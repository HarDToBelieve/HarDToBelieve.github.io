33/*!
34@interface
35@discussion A JSContext is a JavaScript execution environment. All
36 JavaScript execution takes place within a context, and all JavaScript values
37 are tied to a context.
38*/
39JSC_CLASS_AVAILABLE(macos(10.9), ios(7.0))
40@interface JSContext : NSObject
41
42/*!
43@methodgroup Creating New JSContexts
44*/
45/*!
46@method
47@abstract Create a JSContext.
48@result The new context.
49*/
50- (instancetype)init;
51
52/*!
53@method
54@abstract Create a JSContext in the specified virtual machine.
55@param virtualMachine The JSVirtualMachine in which the context will be created.
56@result The new context.
57*/
58- (instancetype)initWithVirtualMachine:(JSVirtualMachine *)virtualMachine;
59
60/*!
61@methodgroup Evaluating Scripts
62*/
63/*!
64@method
65@abstract Evaluate a string of JavaScript code.
66@param script A string containing the JavaScript code to evaluate.
67@result The last value generated by the script.
68*/
69- (JSValue *)evaluateScript:(NSString *)script;
70
71/*!
72@method
73@abstract Evaluate a string of JavaScript code, with a URL for the script's source file.
74@param script A string containing the JavaScript code to evaluate.
75@param sourceURL A URL for the script's source file. Used by debuggers and when reporting exceptions. This parameter is informative only: it does not change the behavior of the script.
76@result The last value generated by the script.
77*/
78- (JSValue *)evaluateScript:(NSString *)script withSourceURL:(NSURL *)sourceURL JSC_API_AVAILABLE(macos(10.10), ios(8.0));
79
80/*!
81@methodgroup Callback Accessors
82*/
83/*!
84@method
85@abstract Get the JSContext that is currently executing.
86@discussion This method may be called from within an Objective-C block or method invoked
87 as a callback from JavaScript to retrieve the callback's context. Outside of
88 a callback from JavaScript this method will return nil.
89@result The currently executing JSContext or nil if there isn't one.
90*/
91+ (JSContext *)currentContext;
92
93/*!
94@method
95@abstract Get the JavaScript function that is currently executing.
96@discussion This method may be called from within an Objective-C block or method invoked
97 as a callback from JavaScript to retrieve the callback's context. Outside of
98 a callback from JavaScript this method will return nil.
99@result The currently executing JavaScript function or nil if there isn't one.
100*/
101+ (JSValue *)currentCallee JSC_API_AVAILABLE(macos(10.10), ios(8.0));
102
103/*!
104@method
105@abstract Get the <code>this</code> value of the currently executing method.
106@discussion This method may be called from within an Objective-C block or method invoked
107 as a callback from JavaScript to retrieve the callback's this value. Outside
108 of a callback from JavaScript this method will return nil.
109@result The current <code>this</code> value or nil if there isn't one.
110*/
111+ (JSValue *)currentThis;
112
113/*!
114@method
115@abstract Get the arguments to the current callback.
116@discussion This method may be called from within an Objective-C block or method invoked
117 as a callback from JavaScript to retrieve the callback's arguments, objects
118 in the returned array are instances of JSValue. Outside of a callback from
119 JavaScript this method will return nil.
120@result An NSArray of the arguments nil if there is no current callback.
121*/
122+ (NSArray *)currentArguments;
123
124/*!
125@functiongroup Global Properties
126*/
127
128/*!
129@property
130@abstract Get the global object of the context.
131@discussion This method retrieves the global object of the JavaScript execution context.
132 Instances of JSContext originating from WebKit will return a reference to the
133 WindowProxy object.
134@result The global object.
135*/
136@property (readonly, strong) JSValue *globalObject;
137
138/*!
139@property
140@discussion The <code>exception</code> property may be used to throw an exception to JavaScript.
141
142 Before a callback is made from JavaScript to an Objective-C block or method,
143 the prior value of the exception property will be preserved and the property
144 will be set to nil. After the callback has completed the new value of the
145 exception property will be read, and prior value restored. If the new value
146 of exception is not nil, the callback will result in that value being thrown.
147
148 This property may also be used to check for uncaught exceptions arising from
149 API function calls (since the default behaviour of <code>exceptionHandler</code> is to
150 assign an uncaught exception to this property).
151*/
152@property (strong) JSValue *exception;
153
154/*!
155@property
156@discussion If a call to an API function results in an uncaught JavaScript exception, the
157 <code>exceptionHandler</code> block will be invoked. The default implementation for the
158 exception handler will store the exception to the exception property on
159 context. As a consequence the default behaviour is for uncaught exceptions
160 occurring within a callback from JavaScript to be rethrown upon return.
161 Setting this value to nil will cause all exceptions occurring
162 within a callback from JavaScript to be silently caught.
163*/
164@property (copy) void(^exceptionHandler)(JSContext *context, JSValue *exception);
165
166/*!
167@property
168@discussion All instances of JSContext are associated with a JSVirtualMachine.
169*/
170@property (readonly, strong) JSVirtualMachine *virtualMachine;
171
172/*!
173@property
174@discussion Name of the JSContext. Exposed when remote debugging the context.
175*/
176@property (copy) NSString *name JSC_API_AVAILABLE(macos(10.10), ios(8.0));
177@end

179/*!
180@category
181@discussion Instances of JSContext implement the following methods in order to enable
182 support for subscript access by key and index, for example:
183
184@textblock
185 JSContext *context;
186 JSValue *v = context[@"X"]; // Get value for "X" from the global object.
187 context[@"Y"] = v; // Assign 'v' to "Y" on the global object.
188@/textblock
189
190 An object key passed as a subscript will be converted to a JavaScript value,
191 and then the value converted to a string used to resolve a property of the
192 global object.
193*/
194@interface JSContext (SubscriptSupport)
195
196/*!
197@method
198@abstract Get a particular property on the global object.
199@result The JSValue for the global object's property.
200*/
201- (JSValue *)objectForKeyedSubscript:(id)key;
202
203/*!
204@method
205@abstract Set a particular property on the global object.
206*/
207- (void)setObject:(id)object forKeyedSubscript:(NSObject <NSCopying> *)key;
208
209@end
210
211/*!
212@category
213@discussion These functions are for bridging between the C API and the Objective-C API.
214*/
215@interface JSContext (JSContextRefSupport)
216
217/*!
218@method
219@abstract Create a JSContext, wrapping its C API counterpart.
220@result The JSContext equivalent of the provided JSGlobalContextRef.
221*/
222+ (JSContext *)contextWithJSGlobalContextRef:(JSGlobalContextRef)jsGlobalContextRef;
223
224/*!
225@property
226@abstract Get the C API counterpart wrapped by a JSContext.
227@result The C API equivalent of this JSContext.
228*/
229@property (readonly) JSGlobalContextRef JSGlobalContextRef;
230
231@end
