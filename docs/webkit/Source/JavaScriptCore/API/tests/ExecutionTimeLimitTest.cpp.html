45static JSGlobalContextRef context = nullptr;
46
47static JSValueRef currentCPUTimeAsJSFunctionCallback(JSContextRef ctx, JSObjectRef functionObject, JSObjectRef thisObject, size_t argumentCount, const JSValueRef arguments[], JSValueRef* exception)
48{
49 UNUSED_PARAM(functionObject);
50 UNUSED_PARAM(thisObject);
51 UNUSED_PARAM(argumentCount);
52 UNUSED_PARAM(arguments);
53 UNUSED_PARAM(exception);
54
55 ASSERT(JSContextGetGlobalContext(ctx) == context);
56 return JSValueMakeNumber(ctx, CPUTime::forCurrentThread().seconds());
57}
58
59bool shouldTerminateCallbackWasCalled = false;
60static bool shouldTerminateCallback(JSContextRef, void*)
61{
62 shouldTerminateCallbackWasCalled = true;
63 return true;
64}
65
66bool cancelTerminateCallbackWasCalled = false;
67static bool cancelTerminateCallback(JSContextRef, void*)
68{
69 cancelTerminateCallbackWasCalled = true;
70 return false;
71}
72
73int extendTerminateCallbackCalled = 0;
74static bool extendTerminateCallback(JSContextRef ctx, void*)
75{
76 extendTerminateCallbackCalled++;
77 if (extendTerminateCallbackCalled == 1) {
78 JSContextGroupRef contextGroup = JSContextGetGroup(ctx);
79 JSContextGroupSetExecutionTimeLimit(contextGroup, .200f, extendTerminateCallback, nullptr);
80 return false;
81 }
82 return true;
83}

89 dispatchTerminateCallbackCalled = true;
90 return true;
91}

94struct TierOptions {
95 const char* tier;
96 Seconds timeLimitAdjustment;
97 const char* optionsStr;

100static void testResetAfterTimeout(bool& failed)
101{
102 JSValueRef v = nullptr;
103 JSValueRef exception = nullptr;
104 const char* reentryScript = "100";
105 JSStringRef script = JSStringCreateWithUTF8CString(reentryScript);
106 v = JSEvaluateScript(context, script, nullptr, nullptr, 1, &exception);
107 JSStringRelease(script);
108 if (exception) {
109 printf("FAIL: Watchdog timeout was not reset.\n");
110 failed = true;
111 } else if (!JSValueIsNumber(context, v) || JSValueToNumber(context, v, nullptr) != 100) {
112 printf("FAIL: Script result is not as expected.\n");
113 failed = true;
114 }
115}

117int testExecutionTimeLimit()
118{
119 static const TierOptions tierOptionsList[] = {
120 { "LLINT", 0_ms, "--useConcurrentJIT=false --useLLInt=true --useJIT=false" },
121 { "Baseline", 0_ms, "--useConcurrentJIT=false --useLLInt=true --useJIT=true --useDFGJIT=false" },
122 { "DFG", 200_ms, "--useConcurrentJIT=false --useLLInt=true --useJIT=true --useDFGJIT=true --useFTLJIT=false" },
123#if ENABLE(FTL_JIT)
124 { "FTL", 500_ms, "--useConcurrentJIT=false --useLLInt=true --useJIT=true --useDFGJIT=true --useFTLJIT=true" },
125#endif
126 };
127
128 bool failed = false;
129
130 JSC::initialize();
131
132 for (auto tierOptions : tierOptionsList) {
133 StringBuilder savedOptionsBuilder;
134 Options::dumpAllOptionsInALine(savedOptionsBuilder);
135
136 Options::setOptions(tierOptions.optionsStr);
137
138 Seconds tierAdjustment = tierOptions.timeLimitAdjustment;
139 Seconds timeLimit;
140
141 context = JSGlobalContextCreateInGroup(nullptr, nullptr);
142
143 JSContextGroupRef contextGroup = JSContextGetGroup(context);
144 JSObjectRef globalObject = JSContextGetGlobalObject(context);
145 ASSERT(JSValueIsObject(context, globalObject));
146
147 JSValueRef exception = nullptr;
148
149 JSStringRef currentCPUTimeStr = JSStringCreateWithUTF8CString("currentCPUTime");
150 JSObjectRef currentCPUTimeFunction = JSObjectMakeFunctionWithCallback(context, currentCPUTimeStr, currentCPUTimeAsJSFunctionCallback);
151 JSObjectSetProperty(context, globalObject, currentCPUTimeStr, currentCPUTimeFunction, kJSPropertyAttributeNone, nullptr);
152 JSStringRelease(currentCPUTimeStr);
153
154 /* Test script on another thread: */
155 timeLimit = 100_ms + tierAdjustment;
156 JSContextGroupSetExecutionTimeLimit(contextGroup, timeLimit.seconds(), shouldTerminateCallback, nullptr);
157 {
158#if OS(LINUX) && (CPU(MIPS) || CPU(ARM_THUMB2))
159 Seconds timeAfterWatchdogShouldHaveFired = 500_ms + tierAdjustment;
160#else
161 Seconds timeAfterWatchdogShouldHaveFired = 300_ms + tierAdjustment;
162#endif
163
164 JSStringRef script = JSStringCreateWithUTF8CString("function foo() { while (true) { } } foo();");
165 exception = nullptr;
166 JSValueRef* exn = &exception;
167 shouldTerminateCallbackWasCalled = false;
168 auto thread = Thread::create("Rogue thread", [=] {
169 JSEvaluateScript(context, script, nullptr, nullptr, 1, exn);
170 });
171
172 sleep(timeAfterWatchdogShouldHaveFired);
173
174 if (shouldTerminateCallbackWasCalled)
175 printf("PASS: %s script timed out as expected.\n", tierOptions.tier);
176 else {
177 printf("FAIL: %s script timeout callback was not called.\n", tierOptions.tier);
178 exit(1);
179 }
180
181 if (!exception) {
182 printf("FAIL: %s TerminationException was not thrown.\n", tierOptions.tier);
183 exit(1);
184 }
185
186 thread->waitForCompletion();
187 testResetAfterTimeout(failed);
188
189 JSStringRelease(script);
190 }
191
192 /* Test script timeout: */
193 timeLimit = 100_ms + tierAdjustment;
194 JSContextGroupSetExecutionTimeLimit(contextGroup, timeLimit.seconds(), shouldTerminateCallback, nullptr);
195 {
196 Seconds timeAfterWatchdogShouldHaveFired = 300_ms + tierAdjustment;
197
198 CString scriptText = makeString(
199 "function foo() {"
200 "var startTime = currentCPUTime();"
201 "while (true) {"
202 "for (var i = 0; i < 1000; i++);"
203 "if (currentCPUTime() - startTime > ", timeAfterWatchdogShouldHaveFired.seconds(), ") break;"
204 "}"
205 "}"
206 "foo();"
207 ).utf8();
208
209 JSStringRef script = JSStringCreateWithUTF8CString(scriptText.data());
210 exception = nullptr;
211 shouldTerminateCallbackWasCalled = false;
212 auto startTime = CPUTime::forCurrentThread();
213 JSEvaluateScript(context, script, nullptr, nullptr, 1, &exception);
214 auto endTime = CPUTime::forCurrentThread();
215 JSStringRelease(script);
216
217 if (((endTime - startTime) < timeAfterWatchdogShouldHaveFired) && shouldTerminateCallbackWasCalled)
218 printf("PASS: %s script timed out as expected.\n", tierOptions.tier);
219 else {
220 if ((endTime - startTime) >= timeAfterWatchdogShouldHaveFired)
221 printf("FAIL: %s script did not time out as expected.\n", tierOptions.tier);
222 if (!shouldTerminateCallbackWasCalled)
223 printf("FAIL: %s script timeout callback was not called.\n", tierOptions.tier);
224 failed = true;
225 }
226
227 if (!exception) {
228 printf("FAIL: %s TerminationException was not thrown.\n", tierOptions.tier);
229 failed = true;
230 }
231
232 testResetAfterTimeout(failed);
233 }

235 /* Test script timeout with tail calls: */
236 timeLimit = 100_ms + tierAdjustment;
237 JSContextGroupSetExecutionTimeLimit(contextGroup, timeLimit.seconds(), shouldTerminateCallback, nullptr);
238 {
239 Seconds timeAfterWatchdogShouldHaveFired = 300_ms + tierAdjustment;
240
241 CString scriptText = makeString(
242 "var startTime = currentCPUTime();"
243 "function recurse(i) {"
244 "'use strict';"
245 "if (i % 1000 === 0) {"
246 "if (currentCPUTime() - startTime >", timeAfterWatchdogShouldHaveFired.seconds(), ") { return; }"
247 "}"
248 "return recurse(i + 1); }"
249 "recurse(0);"
250 ).utf8();
251
252 JSStringRef script = JSStringCreateWithUTF8CString(scriptText.data());
253 exception = nullptr;
254 shouldTerminateCallbackWasCalled = false;
255 auto startTime = CPUTime::forCurrentThread();
256 JSEvaluateScript(context, script, nullptr, nullptr, 1, &exception);
257 auto endTime = CPUTime::forCurrentThread();
258 JSStringRelease(script);
259
260 if (((endTime - startTime) < timeAfterWatchdogShouldHaveFired) && shouldTerminateCallbackWasCalled)
261 printf("PASS: %s script with infinite tail calls timed out as expected .\n", tierOptions.tier);
262 else {
263 if ((endTime - startTime) >= timeAfterWatchdogShouldHaveFired)
264 printf("FAIL: %s script with infinite tail calls did not time out as expected.\n", tierOptions.tier);
265 if (!shouldTerminateCallbackWasCalled)
266 printf("FAIL: %s script with infinite tail calls' timeout callback was not called.\n", tierOptions.tier);
267 failed = true;
268 }
269
270 if (!exception) {
271 printf("FAIL: %s TerminationException was not thrown.\n", tierOptions.tier);
272 failed = true;
273 }
274
275 testResetAfterTimeout(failed);
276 }

278 /* Test the script timeout's TerminationException should NOT be catchable: */
279 timeLimit = 100_ms + tierAdjustment;
280 JSContextGroupSetExecutionTimeLimit(contextGroup, timeLimit.seconds(), shouldTerminateCallback, nullptr);
281 {
282 Seconds timeAfterWatchdogShouldHaveFired = 300_ms + tierAdjustment;
283
284 CString scriptText = makeString(
285 "function foo() {"
286 "var startTime = currentCPUTime();"
287 "try {"
288 "while (true) {"
289 "for (var i = 0; i < 1000; i++);"
290 "if (currentCPUTime() - startTime > ", timeAfterWatchdogShouldHaveFired.seconds(), ") break;"
291 "}"
292 "} catch(e) { }"
293 "}"
294 "foo();"
295 ).utf8();
296
297 JSStringRef script = JSStringCreateWithUTF8CString(scriptText.data());
298 exception = nullptr;
299 shouldTerminateCallbackWasCalled = false;
300
301 auto startTime = CPUTime::forCurrentThread();
302 JSEvaluateScript(context, script, nullptr, nullptr, 1, &exception);
303 auto endTime = CPUTime::forCurrentThread();
304
305 JSStringRelease(script);
306
307 if (((endTime - startTime) >= timeAfterWatchdogShouldHaveFired) || !shouldTerminateCallbackWasCalled) {
308 if (!((endTime - startTime) < timeAfterWatchdogShouldHaveFired))
309 printf("FAIL: %s script did not time out as expected.\n", tierOptions.tier);
310 if (!shouldTerminateCallbackWasCalled)
311 printf("FAIL: %s script timeout callback was not called.\n", tierOptions.tier);
312 failed = true;
313 }
314
315 if (exception)
316 printf("PASS: %s TerminationException was not catchable as expected.\n", tierOptions.tier);
317 else {
318 printf("FAIL: %s TerminationException was caught.\n", tierOptions.tier);
319 failed = true;
320 }
321
322 testResetAfterTimeout(failed);
323 }

376 CString scriptText = makeString(
377 "function foo() {"
378 "var startTime = currentCPUTime();"
379 "while (true) {"
380 "for (var i = 0; i < 1000; i++);"
381 "if (currentCPUTime() - startTime > ", timeAfterWatchdogShouldHaveFired.seconds(), ") break;"
382 "}"
383 "}"
384 "foo();"
385 ).utf8();
386
387 JSStringRef script = JSStringCreateWithUTF8CString(scriptText.data());
388 exception = nullptr;
389 cancelTerminateCallbackWasCalled = false;
390
391 auto startTime = CPUTime::forCurrentThread();
392 JSEvaluateScript(context, script, nullptr, nullptr, 1, &exception);
393 auto endTime = CPUTime::forCurrentThread();
394
395 JSStringRelease(script);

398 printf("PASS: %s script timeout was cancelled as expected.\n", tierOptions.tier);

400 if (((endTime - startTime) < timeAfterWatchdogShouldHaveFired) || exception)

410 }

413 /* Test script timeout extension: */
414 timeLimit = 100_ms + tierAdjustment;
415 JSContextGroupSetExecutionTimeLimit(contextGroup, timeLimit.seconds(), extendTerminateCallback, nullptr);
416 {
417 Seconds timeBeforeExtendedDeadline = 250_ms + tierAdjustment;
418 Seconds timeAfterExtendedDeadline = 600_ms + tierAdjustment;
419 Seconds maxBusyLoopTime = 750_ms + tierAdjustment;
420
421 CString scriptText = makeString(
422 "function foo() {"
423 "var startTime = currentCPUTime();"
424 "while (true) {"
425 "for (var i = 0; i < 1000; i++);"
426 "if (currentCPUTime() - startTime > ", maxBusyLoopTime.seconds(), ") break;"
427 "}"
428 "}"
429 "foo();"
430 ).utf8();
431
432 JSStringRef script = JSStringCreateWithUTF8CString(scriptText.data());
433 exception = nullptr;
434 extendTerminateCallbackCalled = 0;
435
436 auto startTime = CPUTime::forCurrentThread();
437 JSEvaluateScript(context, script, nullptr, nullptr, 1, &exception);
438 auto endTime = CPUTime::forCurrentThread();
439 auto deltaTime = endTime - startTime;
440
441 JSStringRelease(script);
442
443 if ((deltaTime >= timeBeforeExtendedDeadline) && (deltaTime < timeAfterExtendedDeadline) && (extendTerminateCallbackCalled == 2) && exception)
444 printf("PASS: %s script timeout was extended as expected.\n", tierOptions.tier);
445 else {
446 if (deltaTime < timeBeforeExtendedDeadline)
447 printf("FAIL: %s script timeout was not extended as expected.\n", tierOptions.tier);
448 else if (deltaTime >= timeAfterExtendedDeadline)
449 printf("FAIL: %s script did not timeout.\n", tierOptions.tier);
450
451 if (extendTerminateCallbackCalled < 1)
452 printf("FAIL: %s script timeout callback was not called.\n", tierOptions.tier);
453 if (extendTerminateCallbackCalled < 2)
454 printf("FAIL: %s script timeout callback was not called after timeout extension.\n", tierOptions.tier);
455
456 if (!exception)
457 printf("FAIL: %s TerminationException was not thrown during timeout extension test.\n", tierOptions.tier);
458
459 failed = true;
460 }
461 }

463#if HAVE(MACH_EXCEPTIONS)
464 /* Test script timeout from dispatch queue: */
465 timeLimit = 100_ms + tierAdjustment;
466 JSContextGroupSetExecutionTimeLimit(contextGroup, timeLimit.seconds(), dispatchTermitateCallback, nullptr);
467 {
468 Seconds timeAfterWatchdogShouldHaveFired = 300_ms + tierAdjustment;
469
470 CString scriptText = makeString(
471 "function foo() {"
472 "var startTime = currentCPUTime();"
473 "while (true) {"
474 "for (var i = 0; i < 1000; i++);"
475 "if (currentCPUTime() - startTime > ", timeAfterWatchdogShouldHaveFired.seconds(), ") break;"
476 "}"
477 "}"
478 "foo();"
479 ).utf8();
480
481 JSStringRef script = JSStringCreateWithUTF8CString(scriptText.data());
482 exception = nullptr;
483 dispatchTerminateCallbackCalled = false;
484
485 // We have to do this since blocks can only capture things as const.
486 JSGlobalContextRef& contextRef = context;
487 JSStringRef& scriptRef = script;
488 JSValueRef& exceptionRef = exception;
489
490 Lock syncLock;
491 Lock& syncLockRef = syncLock;
492 Condition synchronize;
493 Condition& synchronizeRef = synchronize;
494 bool didSynchronize = false;
495 bool& didSynchronizeRef = didSynchronize;
496
497 Seconds startTime;
498 Seconds endTime;
499
500 Seconds& startTimeRef = startTime;
501 Seconds& endTimeRef = endTime;
502
503 dispatch_group_t group = dispatch_group_create();
504 dispatch_group_async(group, dispatch_get_global_queue(0, 0), ^{
505 startTimeRef = CPUTime::forCurrentThread();
506 JSEvaluateScript(contextRef, scriptRef, nullptr, nullptr, 1, &exceptionRef);
507 endTimeRef = CPUTime::forCurrentThread();
508 Locker locker { syncLockRef };
509 didSynchronizeRef = true;
510 synchronizeRef.notifyAll();
511 });
512
513 Locker locker { syncLock };
514 synchronize.wait(syncLock, [&] { return didSynchronize; });
515
516 if (((endTime - startTime) < timeAfterWatchdogShouldHaveFired) && dispatchTerminateCallbackCalled)
517 printf("PASS: %s script on dispatch queue timed out as expected.\n", tierOptions.tier);
518 else {
519 if ((endTime - startTime) >= timeAfterWatchdogShouldHaveFired)
520 printf("FAIL: %s script on dispatch queue did not time out as expected.\n", tierOptions.tier);
521 if (!shouldTerminateCallbackWasCalled)
522 printf("FAIL: %s script on dispatch queue timeout callback was not called.\n", tierOptions.tier);
523 failed = true;
524 }
525
526 JSStringRelease(script);
527 }
528#endif
529
530 JSGlobalContextRelease(context);
531
532 Options::setOptions(savedOptionsBuilder.toString().ascii().data());
533 }
534
535 return failed;
536}
