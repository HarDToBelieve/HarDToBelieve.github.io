26#include "config.h"
27#include "GlobalContextWithFinalizerTest.h"
28
29#include "JavaScript.h"
30#include <stdio.h>
31
32static bool failed = true;
33
34static void finalize(JSObjectRef)
35{
36 failed = false;
37}
38
39int testGlobalContextWithFinalizer()
40{
41 JSClassDefinition def = kJSClassDefinitionEmpty;
42 def.className = "testClass";
43 def.finalize = finalize;
44 JSClassRef classRef = JSClassCreate(&def);
45
46 JSGlobalContextRef ref = JSGlobalContextCreateInGroup(nullptr, classRef);
47 JSGlobalContextRelease(ref);
48 JSClassRelease(classRef);
49
50 if (failed)
51 printf("FAIL: JSGlobalContextRef did not call its JSClassRef finalizer.\n");
52 else
53 printf("PASS: JSGlobalContextRef called its JSClassRef finalizer as expected.\n");
54
55 return failed;
56}
