26#include "config.h"
27#include "JSONParseTest.h"
28
29#include "JSCInlines.h"
30#include "JSGlobalObject.h"
31#include "JSONObject.h"
32#include "VM.h"
33#include <wtf/RefPtr.h>
34
35using namespace JSC;
36
37int testJSONParse()
38{
39 bool failed = false;
40
41 RefPtr<VM> vm = VM::create();
42
43 JSLockHolder locker(vm.get());
44 JSGlobalObject* globalObject = JSGlobalObject::create(*vm, JSGlobalObject::createStructure(*vm, jsNull()));
45
46 JSValue v0 = JSONParse(globalObject, "");
47 JSValue v1 = JSONParse(globalObject, "#$%^");
48 JSValue v2 = JSONParse(globalObject, String());
49 UChar emptyUCharArray[1] = { '\0' };
50 JSValue v3 = JSONParse(globalObject, String(emptyUCharArray, 0));
51 JSValue v4;
52 JSValue v5 = JSONParse(globalObject, "123");
53
54 failed = failed || (v0 != v1);
55 failed = failed || (v1 != v2);
56 failed = failed || (v2 != v3);
57 failed = failed || (v3 != v4);
58 failed = failed || (v4 == v5);
59
60 vm = nullptr;
61
62 if (failed)
63 printf("FAIL: JSONParse String test.\n");
64 else
65 printf("PASS: JSONParse String test.\n");
66
67 return failed;
68}
