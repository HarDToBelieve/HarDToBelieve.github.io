26#include "config.h"
27#include "JSObjectGetProxyTargetTest.h"
28
29#include "APICast.h"
30#include "JSCInlines.h"
31#include "JSObjectRefPrivate.h"
32#include "JSProxy.h"
33#include "JavaScript.h"
34#include "ProxyObject.h"
35
36using namespace JSC;
37
38int testJSObjectGetProxyTarget()
39{
40 bool overallResult = true;
41
42 printf("JSObjectGetProxyTargetTest:\n");
43
44 auto test = [&] (const char* description, bool currentResult) {
45 printf(" %s: %s\n", description, currentResult ? "PASS" : "FAIL");
46 overallResult &= currentResult;
47 };
48
49 JSContextGroupRef group = JSContextGroupCreate();
50 JSGlobalContextRef context = JSGlobalContextCreateInGroup(group, nullptr);
51
52 VM& vm = *toJS(group);
53 JSObjectRef globalObjectProxy = JSContextGetGlobalObject(context);
54
55 JSGlobalObject* globalObjectObject;
56 JSObjectRef globalObjectRef;
57 JSProxy* jsProxyObject;
58
59 {
60 JSLockHolder locker(vm);
61 JSProxy* globalObjectProxyObject = jsCast<JSProxy*>(toJS(globalObjectProxy));
62 globalObjectObject = jsCast<JSGlobalObject*>(globalObjectProxyObject->target());
63 Structure* proxyStructure = JSProxy::createStructure(vm, globalObjectObject, globalObjectObject->objectPrototype());
64 globalObjectRef = toRef(jsCast<JSObject*>(globalObjectObject));
65 jsProxyObject = JSProxy::create(vm, proxyStructure);
66 }
67
68 JSObjectRef array = JSObjectMakeArray(context, 0, nullptr, nullptr);
69
70 ProxyObject* proxyObjectObject;
71
72 {
73 JSLockHolder locker(vm);
74 Structure* emptyObjectStructure = JSFinalObject::createStructure(vm, globalObjectObject, globalObjectObject->objectPrototype(), 0);
75 JSObject* handler = JSFinalObject::create(vm, emptyObjectStructure);
76 proxyObjectObject = ProxyObject::create(globalObjectObject, toJS(array), handler);
77 }
78
79 JSObjectRef jsProxy = toRef(jsProxyObject);
80 JSObjectRef proxyObject = toRef(proxyObjectObject);
81
82 test("proxy target of null is null", !JSObjectGetProxyTarget(nullptr));
83 test("proxy target of non-proxy is null", !JSObjectGetProxyTarget(array));
84 test("proxy target of uninitialized JSProxy is null", !JSObjectGetProxyTarget(jsProxy));
85
86 {
87 JSLockHolder locker(vm);
88 jsProxyObject->setTarget(vm, globalObjectObject);
89 }
90
91 test("proxy target of initialized JSProxy works", JSObjectGetProxyTarget(jsProxy) == globalObjectRef);
92
93 test("proxy target of ProxyObject works", JSObjectGetProxyTarget(proxyObject) == array);
94
95 test("proxy target of GlobalObject is the globalObject", JSObjectGetProxyTarget(globalObjectProxy) == globalObjectRef);
96
97 JSGlobalContextRelease(context);
98 JSContextGroupRelease(group);
99
100 printf("JSObjectGetProxyTargetTest: %s\n", overallResult ? "PASS" : "FAIL");
101 return !overallResult;
102}
