36extern "C" void checkResult(NSString *description, bool passed);

38@protocol TestClassJSExport <JSExport>
39- (instancetype)init;
40@end
41
42@interface TestClass : NSObject <TestClassJSExport>
43@end
44
45@implementation TestClass
46@end

49@interface JSWrapperMapTests : NSObject
50+ (void)testStructureIdentity;
51@end
52
53
54@implementation JSWrapperMapTests
55+ (void)testStructureIdentity
56{
57 JSContext* context = [[JSContext alloc] init];
58 JSGlobalContextRef contextRef = JSGlobalContextRetain(context.JSGlobalContextRef);
59 JSC::JSGlobalObject* globalObject = toJS(contextRef);
60 JSC::VM& vm = globalObject->vm();
61
62 context[@"TestClass"] = [TestClass class];
63 JSValue* aWrapper = [context evaluateScript:@"new TestClass()"];
64 JSValue* bWrapper = [context evaluateScript:@"new TestClass()"];
65 JSC::JSValue aValue = toJS(globalObject, aWrapper.JSValueRef);
66 JSC::JSValue bValue = toJS(globalObject, bWrapper.JSValueRef);
67 JSC::Structure* aStructure = aValue.structureOrNull(vm);
68 JSC::Structure* bStructure = bValue.structureOrNull(vm);
69 checkResult(@"structure should not be null", !!aStructure);
70 checkResult(@"both wrappers should share the same structure", aStructure == bStructure);
71}
72@end
73
74void runJSWrapperMapTests()
75{
76 @autoreleasepool {
77 [JSWrapperMapTests testStructureIdentity];
78 }
79}
