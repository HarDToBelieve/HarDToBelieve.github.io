36static JSGlobalContextRef context = nullptr;
37static int nativeRecursionCount = 0;
38
39static bool PingPongStackOverflowObject_hasInstance(JSContextRef context, JSObjectRef constructor, JSValueRef possibleValue, JSValueRef* exception)
40{
41 UNUSED_PARAM(context);
42 UNUSED_PARAM(constructor);
43
44 JSStringRef hasInstanceName = JSStringCreateWithUTF8CString("hasInstance");
45 JSValueRef hasInstance = JSObjectGetProperty(context, constructor, hasInstanceName, exception);
46 JSStringRelease(hasInstanceName);
47 if (!hasInstance)
48 return false;
49
50 int countAtEntry = nativeRecursionCount++;
51
52 JSValueRef result = nullptr;
53 if (nativeRecursionCount < 100) {
54 JSObjectRef function = JSValueToObject(context, hasInstance, exception);
55 result = JSObjectCallAsFunction(context, function, constructor, 1, &possibleValue, exception);
56 } else {
57 StringBuilder builder;
58 builder.append("dummy.valueOf([0]");
59 for (int i = 1; i < 35000; i++)
60 builder.append(", [", i, ']');
61 builder.append(");");
62 JSStringRef script = JSStringCreateWithUTF8CString(builder.toString().utf8().data());
63 result = JSEvaluateScript(context, script, nullptr, nullptr, 1, exception);
64 JSStringRelease(script);
65 }
66
67 --nativeRecursionCount;
68 if (nativeRecursionCount != countAtEntry)
69 printf(" ERROR: PingPongStackOverflow test saw a recursion count mismatch\n");
70
71 return result && JSValueToBoolean(context, result);
72}

97static JSClassRef PingPongStackOverflowObject_class(JSContextRef context)
98{
99 UNUSED_PARAM(context);
100
101 static JSClassRef jsClass;
102 if (!jsClass)
103 jsClass = JSClassCreate(&PingPongStackOverflowObject_definition);
104
105 return jsClass;
106}

108// This tests tests a stack overflow on VM reentry into a JS function from a native function
109// after ping-pong'ing back and forth between JS and native functions multiple times.
110// This test should not hang or crash.
111int testPingPongStackOverflow()
112{
113 bool failed = false;
114
115 JSC::initialize();
116
117 auto origSoftReservedZoneSize = Options::softReservedZoneSize();
118 auto origReservedZoneSize = Options::reservedZoneSize();
119 auto origUseLLInt = Options::useLLInt();
120 auto origMaxPerThreadStackUsage = Options::maxPerThreadStackUsage();
121
122 Options::softReservedZoneSize() = 128 * KB;
123 Options::reservedZoneSize() = 64 * KB;
124#if ENABLE(JIT)
125 // Normally, we want to disable the LLINT to force the use of JITted code which is necessary for
126 // reproducing the regression in https://bugs.webkit.org/show_bug.cgi?id=148749. However, we only
127 // want to do this if the LLINT isn't the only available execution engine.
128 Options::useLLInt() = false;
129#endif
130
131 const char* scriptString =
132 "var count = 0;" \
133 "PingPongStackOverflowObject.hasInstance = function f() {" \
134 " return (undefined instanceof PingPongStackOverflowObject);" \
135 "};" \
136 "PingPongStackOverflowObject.__proto__ = undefined;" \
137 "undefined instanceof PingPongStackOverflowObject;";
138
139 JSValueRef exception = nullptr;
140 JSStringRef script = JSStringCreateWithUTF8CString(scriptString);
141
142 nativeRecursionCount = 0;
143 context = JSGlobalContextCreateInGroup(nullptr, nullptr);
144
145 JSObjectRef globalObject = JSContextGetGlobalObject(context);
146 ASSERT(JSValueIsObject(context, globalObject));
147
148 JSObjectRef PingPongStackOverflowObject = JSObjectMake(context, PingPongStackOverflowObject_class(context), nullptr);
149 JSStringRef PingPongStackOverflowObjectString = JSStringCreateWithUTF8CString("PingPongStackOverflowObject");
150 JSObjectSetProperty(context, globalObject, PingPongStackOverflowObjectString, PingPongStackOverflowObject, kJSPropertyAttributeNone, nullptr);
151 JSStringRelease(PingPongStackOverflowObjectString);
152
153 unsigned stackSize = 32 * KB;
154 Options::maxPerThreadStackUsage() = stackSize + Options::softReservedZoneSize();
155
156 exception = nullptr;
157 JSEvaluateScript(context, script, nullptr, nullptr, 1, &exception);
158
159 JSGlobalContextRelease(context);
160 context = nullptr;
161 JSStringRelease(script);
162
163 if (!exception) {
164 printf("FAIL: PingPongStackOverflowError not thrown in PingPongStackOverflow test\n");
165 failed = true;
166 } else if (nativeRecursionCount) {
167 printf("FAIL: Unbalanced native recursion count: %d in PingPongStackOverflow test\n", nativeRecursionCount);
168 failed = true;
169 } else {
170 printf("PASS: PingPongStackOverflow test.\n");
171 }
172
173 Options::softReservedZoneSize() = origSoftReservedZoneSize;
174 Options::reservedZoneSize() = origReservedZoneSize;
175 Options::useLLInt() = origUseLLInt;
176 Options::maxPerThreadStackUsage() = origMaxPerThreadStackUsage;
177
178 return failed;
179}
