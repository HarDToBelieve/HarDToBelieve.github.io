86inline int unexpectedException(const char* name)
87{
88 fprintf(stderr, "%s FAILED: unexpected exception\n", name);
89 return 1;
90}

92static int assertEqualsAsNumber(JSGlobalContextRef context, JSValueRef value, double expectedValue)
93{
94 double number = JSValueToNumber(context, value, nullptr);
95 if (number != expectedValue && !(isnan(number) && isnan(expectedValue))) {
96 fprintf(stderr, "assertEqualsAsNumber FAILED: %p, %lf\n", value, expectedValue);
97 return 1;
98 }
99 return 0;
100}

102static int testAccess(JSGlobalContextRef context, JSObjectRef typedArray, JSTypedArrayType type, unsigned elementLength, void* expectedPtr = nullptr, JSObjectRef expectedBuffer = nullptr, unsigned expectedOffset = 0)
103{
104 JSValueRef exception = nullptr;
105 // Test typedArray basic functions.
106 JSTypedArrayType actualType = JSValueGetTypedArrayType(context, typedArray, &exception);
107 if (type != actualType || exception) {
108 fprintf(stderr, "TypedArray type FAILED: %p, got: %s, expected: %s\n", typedArray, typeToString[actualType], typeToString[type]);
109 return 1;
110 }
111
112 unsigned length = JSObjectGetTypedArrayLength(context, typedArray, &exception);
113 if (elementLength != length || exception) {
114 fprintf(stderr, "TypedArray length FAILED: %p (%s), got: %d, expected: %d\n", typedArray, typeToString[type], length, elementLength);
115 return 1;
116 }
117
118 unsigned byteLength = JSObjectGetTypedArrayByteLength(context, typedArray, &exception);
119 unsigned expectedLength = byteSizes[type] * elementLength;
120 if (byteLength != expectedLength || exception) {
121 fprintf(stderr, "TypedArray byteLength FAILED: %p (%s), got: %d, expected: %d\n", typedArray, typeToString[type], byteLength, expectedLength);
122 return 1;
123 }
124
125 unsigned offset = JSObjectGetTypedArrayByteOffset(context, typedArray, &exception);
126 if (expectedOffset != offset || exception) {
127 fprintf(stderr, "TypedArray byteOffset FAILED: %p (%s), got: %d, expected: %d\n", typedArray, typeToString[type], offset, expectedOffset);
128 return 1;
129 }
130
131 void* ptr = JSObjectGetTypedArrayBytesPtr(context, typedArray, &exception);
132 if (exception)
133 return unexpectedException("TypedArray get bytes ptr");
134
135 JSObjectRef buffer = JSObjectGetTypedArrayBuffer(context, typedArray, &exception);
136 if (exception)
137 return unexpectedException("TypedArray get buffer");
138
139 void* bufferPtr = JSObjectGetArrayBufferBytesPtr(context, buffer, &exception);
140 if (exception)
141 return unexpectedException("ArrayBuffer get bytes ptr");
142
143 if (bufferPtr != ptr) {
144 fprintf(stderr, "FAIL: TypedArray bytes ptr and ArrayBuffer byte ptr were not the same: %p (%s) TypedArray: %p, ArrayBuffer: %p\n", typedArray, typeToString[type], ptr, bufferPtr);
145 return 1;
146 }
147
148 if (expectedPtr && ptr != expectedPtr) {
149 fprintf(stderr, "FAIL: TypedArray bytes ptr and the ptr used to construct the array were not the same: %p (%s) TypedArray: %p, bytes ptr: %p\n", typedArray, typeToString[type], ptr, expectedPtr);
150 return 1;
151 }
152
153 if (expectedBuffer && expectedBuffer != buffer) {
154 fprintf(stderr, "FAIL: TypedArray buffer and the ArrayBuffer buffer used to construct the array were not the same: %p (%s) TypedArray buffer: %p, data: %p\n", typedArray, typeToString[type], buffer, expectedBuffer);
155 return 1;
156 }
157
158 return 0;
159}

161static int testConstructors(JSGlobalContextRef context, JSTypedArrayType type, unsigned length)
162{
163 int failed = 0;
164 JSValueRef exception = nullptr;
165 JSObjectRef typedArray;
166
167 // Test create with length.
168 typedArray = JSObjectMakeTypedArray(context, type, length, &exception);
169 failed = failed || exception || testAccess(context, typedArray, type, length);
170
171 void* ptr = calloc(length, byteSizes[type]); // This is to be freed by data
172 JSObjectRef data = JSObjectMakeArrayBufferWithBytesNoCopy(context, ptr, length * byteSizes[type], freePtr, nullptr, &exception);
173 failed = failed || exception;
174
175 // Test create with existing ptr.
176 typedArray = JSObjectMakeTypedArrayWithBytesNoCopy(context, type, ptr, length * byteSizes[type], id, nullptr, &exception);
177 failed = failed || exception || testAccess(context, typedArray, type, length, ptr);
178
179 // Test create with existing ArrayBuffer.
180 typedArray = JSObjectMakeTypedArrayWithArrayBuffer(context, type, data, &exception);
181 failed = failed || exception || testAccess(context, typedArray, type, length, ptr, data);
182
183 // Test create with existing ArrayBuffer and offset.
184 typedArray = JSObjectMakeTypedArrayWithArrayBufferAndOffset(context, type, data, 0, length, &exception);
185 failed = failed || exception || testAccess(context, typedArray, type, length, ptr, data);
186
187 typedArray = JSObjectMakeTypedArrayWithArrayBufferAndOffset(context, type, data, byteSizes[type], length-1, &exception);
188 if (!length)
189 failed = failed || !exception;
190 else
191 failed = failed || testAccess(context, typedArray, type, length-1, ptr, data, byteSizes[type]) || exception;
192
193 exception = nullptr;
194 typedArray = JSObjectMakeTypedArrayWithArrayBufferAndOffset(context, type, data, byteSizes[type], 3, &exception);
195 if (length < 2)
196 failed = failed || !exception;
197 else
198 failed = failed || testAccess(context, typedArray, type, 3, ptr, data, byteSizes[type]) || exception;
199
200 if (byteSizes[type] > 1) {
201 exception = nullptr;
202 typedArray = JSObjectMakeTypedArrayWithArrayBufferAndOffset(context, type, data, 1, length-1, &exception);
203 failed = failed || !exception;
204 }
205
206 typedArray = JSObjectMakeTypedArrayWithArrayBufferAndOffset(context, type, data, byteSizes[type], length, &exception);
207 failed = failed || !exception;
208
209 exception = nullptr;
210 typedArray = JSObjectMakeTypedArrayWithArrayBufferAndOffset(context, type, data, byteSizes[type], 0, &exception);
211 if (!length)
212 failed = failed || !exception;
213 else
214 failed = failed || testAccess(context, typedArray, type, 0, ptr, data, byteSizes[type]) || exception;
215
216 return failed;
217}

219template <typename Functor>
220static int forEachTypedArrayType(const Functor& functor)
221{
222 int failed = 0;
223 for (unsigned i = 0; i < kJSTypedArrayTypeArrayBuffer; i++) {
224 if (i != kJSTypedArrayTypeNone && i != kJSTypedArrayTypeArrayBuffer)
225 failed = failed || functor(static_cast<JSTypedArrayType>(i));
226 }
227 return failed;
228}

230int testTypedArrayCAPI()
231{
232 int failed = 0;
233 JSGlobalContextRef context = JSGlobalContextCreate(nullptr);
234
235 failed = failed || forEachTypedArrayType([&](JSTypedArrayType type) {
236 int failed = 0;
237 for (unsigned i = 0; i < numLengths; i++)
238 failed = failed || testConstructors(context, type, lengths[i]);
239 return failed;
240 });
241
242 // Test making a typedArray from scratch length.
243 volatile JSObjectRef typedArray = JSObjectMakeTypedArray(context, kJSTypedArrayTypeUint32Array, 10, nullptr);
244 JSObjectRef data = JSObjectGetTypedArrayBuffer(context, typedArray, nullptr);
245 unsigned* buffer = static_cast<unsigned*>(JSObjectGetArrayBufferBytesPtr(context, data, nullptr));
246
247 ASSERT(JSObjectGetTypedArrayLength(context, typedArray, nullptr) == 10);
248
249 // Test buffer is connected to typedArray.
250 buffer[1] = 1;
251 JSValueRef v = JSObjectGetPropertyAtIndex(context, typedArray, 1, nullptr);
252 failed = failed || assertEqualsAsNumber(context, v, 1);
253
254 // Test passing a buffer from a new array to an old array
255 typedArray = JSObjectMakeTypedArrayWithBytesNoCopy(context, kJSTypedArrayTypeUint32Array, buffer, 40, id, nullptr, nullptr);
256 buffer = static_cast<unsigned*>(JSObjectGetTypedArrayBytesPtr(context, typedArray, nullptr));
257 ASSERT(buffer[1] == 1);
258 buffer[1] = 20;
259 ASSERT(((unsigned*)JSObjectGetArrayBufferBytesPtr(context, data, nullptr))[1] == 20);
260
261 // Test constructing with data and the data returned are the same even with an offset.
262 typedArray = JSObjectMakeTypedArrayWithArrayBufferAndOffset(context, kJSTypedArrayTypeUint32Array, data, 4, 9, nullptr);
263 failed = failed || assertEqualsAsNumber(context, JSObjectGetPropertyAtIndex(context, typedArray, 0, nullptr), 20);
264 ASSERT(data == JSObjectGetTypedArrayBuffer(context, typedArray, nullptr));
265
266 // Test attempting to allocate an array too big for memory.
267 forEachTypedArrayType([&](JSTypedArrayType type) {
268 JSValueRef exception = nullptr;
269 JSObjectMakeTypedArray(context, type, UINT_MAX, &exception);
270 return !exception;
271 });
272
273 JSGlobalContextRelease(context);
274
275 if (!failed)
276 printf("PASS: Typed Array C API Tests.\n");
277 else
278 printf("FAIL: Some Typed Array C API Tests failed.\n");
279
280 return failed;
281}
