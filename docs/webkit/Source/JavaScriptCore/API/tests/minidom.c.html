40int main(int argc, char* argv[])
41{
42 const char *scriptPath = "minidom.js";
43 if (argc > 1) {
44 scriptPath = argv[1];
45 }
46
47 JSGlobalContextRef context = JSGlobalContextCreateInGroup(NULL, NULL);
48 JSObjectRef globalObject = JSContextGetGlobalObject(context);
49
50 JSStringRef printIString = JSStringCreateWithUTF8CString("print");
51 JSObjectSetProperty(context, globalObject, printIString, JSObjectMakeFunctionWithCallback(context, printIString, print), kJSPropertyAttributeNone, NULL);
52 JSStringRelease(printIString);
53
54 JSStringRef node = JSStringCreateWithUTF8CString("Node");
55 JSObjectSetProperty(context, globalObject, node, JSObjectMakeConstructor(context, JSNode_class(context), JSNode_construct), kJSPropertyAttributeNone, NULL);
56 JSStringRelease(node);
57
58 char* scriptUTF8 = createStringWithContentsOfFile(scriptPath);
59 JSStringRef script = JSStringCreateWithUTF8CString(scriptUTF8);
60 JSValueRef exception;
61 JSValueRef result = JSEvaluateScript(context, script, NULL, NULL, 1, &exception);
62 if (result)
63 printf("PASS: Test script executed successfully.\n");
64 else {
65 printf("FAIL: Test script threw exception:\n");
66 JSStringRef exceptionIString = JSValueToStringCopy(context, exception, NULL);
67 size_t exceptionUTF8Size = JSStringGetMaximumUTF8CStringSize(exceptionIString);
68 char* exceptionUTF8 = (char*)malloc(exceptionUTF8Size);
69 JSStringGetUTF8CString(exceptionIString, exceptionUTF8, exceptionUTF8Size);
70 printf("%s\n", exceptionUTF8);
71 free(exceptionUTF8);
72 JSStringRelease(exceptionIString);
73 }
74 JSStringRelease(script);
75 free(scriptUTF8);
76
77 globalObject = 0;
78 JSGlobalContextRelease(context);
79 printf("PASS: Program exited normally.\n");
80 return 0;
81}

83static JSValueRef print(JSContextRef context, JSObjectRef object, JSObjectRef thisObject, size_t argumentCount, const JSValueRef arguments[], JSValueRef* exception)
84{
85 UNUSED_PARAM(object);
86 UNUSED_PARAM(thisObject);
87
88 if (argumentCount > 0) {
89 JSStringRef string = JSValueToStringCopy(context, arguments[0], exception);
90 size_t numChars = JSStringGetMaximumUTF8CStringSize(string);
91 char* stringUTF8 = (char*)malloc(numChars);
92 JSStringGetUTF8CString(string, stringUTF8, numChars);
93 printf("%s\n", stringUTF8);
94 JSStringRelease(string);
95 free(stringUTF8);
96 }
97
98 return JSValueMakeUndefined(context);
99}

101static char* createStringWithContentsOfFile(const char* fileName)
102{
103 char* buffer;
104
105 size_t buffer_size = 0;
106 size_t buffer_capacity = 1024;
107 buffer = (char*)malloc(buffer_capacity);
108
109 FILE* f = fopen(fileName, "r");
110 if (!f) {
111 fprintf(stderr, "Could not open file: %s\n", fileName);
112 free(buffer);
113 return 0;
114 }
115
116 while (!feof(f) && !ferror(f)) {
117 buffer_size += fread(buffer + buffer_size, 1, buffer_capacity - buffer_size, f);
118 if (buffer_size == buffer_capacity) { /* guarantees space for trailing '\0' */
119 buffer_capacity *= 2;
120 buffer = (char*)realloc(buffer, buffer_capacity);
121 ASSERT(buffer);
122 }
123
124 ASSERT(buffer_size < buffer_capacity);
125 }
126 fclose(f);
127 buffer[buffer_size] = '\0';
128
129 return buffer;
130}
