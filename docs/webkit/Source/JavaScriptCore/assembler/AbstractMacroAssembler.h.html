176 // ImplicitAddress:
177 //
178 // This class is used for explicit 'load' and 'store' operations
179 // (as opposed to situations in which a memory operand is provided
180 // to a generic operation, such as an integer arithmetic instruction).
181 //
182 // In the case of a load (or store) operation we want to permit
183 // addresses to be implicitly constructed, e.g. the two calls:
184 //
185 // load32(Address(addrReg), destReg);
186 // load32(addrReg, destReg);
187 //
188 // Are equivalent, and the explicit wrapping of the Address in the former
189 // is unnecessary.
190 struct ImplicitAddress {
191 ImplicitAddress(RegisterID base)
192 : base(base)
193 , offset(0)
194 {
195 ASSERT(base != RegisterID::InvalidGPRReg);
196 }
197
198 ImplicitAddress(Address address)
199 : base(address.base)
200 , offset(address.offset)
201 {
202 ASSERT(base != RegisterID::InvalidGPRReg);
203 }
204
205 RegisterID base;
206 int32_t offset;
207 };
208
209 // BaseIndex:
210 //
211 // Describes a complex addressing mode.
212 struct BaseIndex {
213 BaseIndex(RegisterID base, RegisterID index, Scale scale, int32_t offset = 0)
214 : base(base)
215 , index(index)
216 , scale(scale)
217 , offset(offset)
218 {
219 }
220
221 RegisterID base;
222 RegisterID index;
223 Scale scale;
224 int32_t offset;
225
226 BaseIndex withOffset(int32_t additionalOffset)
227 {
228 return BaseIndex(base, index, scale, offset + additionalOffset);
229 }
230
231 BaseIndex withSwappedRegister(RegisterID left, RegisterID right)
232 {
233 return BaseIndex(AbstractMacroAssembler::withSwappedRegister(base, left, right), AbstractMacroAssembler::withSwappedRegister(index, left, right), scale, offset);
234 }
235 };

635 void link(AbstractMacroAssemblerType* masm) const
636 {
637 masm->invalidateAllTempRegisters();
638
639#if ENABLE(DFG_REGISTER_ALLOCATION_VALIDATION)
640 masm->checkRegisterAllocationAgainstBranchRange(m_label.offset(), masm->debugOffset());
641#endif
642
643#if CPU(ARM_THUMB2)
644 masm->m_assembler.linkJump(m_label, masm->m_assembler.label(), m_type, m_condition);
645#elif CPU(ARM64)
646 if ((m_type == ARM64Assembler::JumpCompareAndBranch) || (m_type == ARM64Assembler::JumpCompareAndBranchFixedSize))
647 masm->m_assembler.linkJump(m_label, masm->m_assembler.label(), m_type, m_condition, m_is64Bit, m_compareRegister);
648 else if ((m_type == ARM64Assembler::JumpTestBit) || (m_type == ARM64Assembler::JumpTestBitFixedSize))
649 masm->m_assembler.linkJump(m_label, masm->m_assembler.label(), m_type, m_condition, m_bitNumber, m_compareRegister);
650 else
651 masm->m_assembler.linkJump(m_label, masm->m_assembler.label(), m_type, m_condition);
652#else
653 masm->m_assembler.linkJump(m_label, masm->m_assembler.label());
654#endif
655 }
656
657 void linkTo(Label label, AbstractMacroAssemblerType* masm) const
658 {
659#if ENABLE(DFG_REGISTER_ALLOCATION_VALIDATION)
660 masm->checkRegisterAllocationAgainstBranchRange(label.m_label.offset(), m_label.offset());
661#endif
662
663#if CPU(ARM_THUMB2)
664 masm->m_assembler.linkJump(m_label, label.m_label, m_type, m_condition);
665#elif CPU(ARM64)
666 if ((m_type == ARM64Assembler::JumpCompareAndBranch) || (m_type == ARM64Assembler::JumpCompareAndBranchFixedSize))
667 masm->m_assembler.linkJump(m_label, label.m_label, m_type, m_condition, m_is64Bit, m_compareRegister);
668 else if ((m_type == ARM64Assembler::JumpTestBit) || (m_type == ARM64Assembler::JumpTestBitFixedSize))
669 masm->m_assembler.linkJump(m_label, label.m_label, m_type, m_condition, m_bitNumber, m_compareRegister);
670 else
671 masm->m_assembler.linkJump(m_label, label.m_label, m_type, m_condition);
672#else
673 masm->m_assembler.linkJump(m_label, label.m_label);
674#endif
675 }
676
677 bool isSet() const { return m_label.isSet(); }
678
679 private:
680 AssemblerLabel m_label;
681#if CPU(ARM_THUMB2)
682 ARMv7Assembler::JumpType m_type { ARMv7Assembler::JumpNoCondition };
683 ARMv7Assembler::Condition m_condition { ARMv7Assembler::ConditionInvalid };
684#elif CPU(ARM64)
685 ARM64Assembler::JumpType m_type { ARM64Assembler::JumpNoCondition };
686 ARM64Assembler::Condition m_condition { ARM64Assembler::ConditionInvalid };
687 bool m_is64Bit { false };
688 unsigned m_bitNumber { 0 };
689 ARM64Assembler::RegisterID m_compareRegister { ARM64Registers::InvalidGPRReg };
690#endif
691 };

708 // JumpList:
709 //
710 // A JumpList is a set of Jump objects.
711 // All jumps in the set will be linked to the same destination.
712 class JumpList {
713 public:
714 typedef Vector<Jump, 2> JumpVector;
715
716 JumpList() { }
717
718 JumpList(Jump jump)
719 {
720 if (jump.isSet())
721 append(jump);
722 }
723
724 void link(AbstractMacroAssemblerType* masm) const
725 {
726 size_t size = m_jumps.size();
727 for (size_t i = 0; i < size; ++i)
728 m_jumps[i].link(masm);
729 }
730
731 void linkTo(Label label, AbstractMacroAssemblerType* masm) const
732 {
733 size_t size = m_jumps.size();
734 for (size_t i = 0; i < size; ++i)
735 m_jumps[i].linkTo(label, masm);
736 }
737
738 void append(Jump jump)
739 {
740 if (jump.isSet())
741 m_jumps.append(jump);
742 }
743
744 void append(const JumpList& other)
745 {
746 m_jumps.append(other.m_jumps.begin(), other.m_jumps.size());
747 }
748
749 bool empty() const
750 {
751 return !m_jumps.size();
752 }
753
754 void clear()
755 {
756 m_jumps.clear();
757 }
758
759 const JumpVector& jumps() const { return m_jumps; }
760
761 private:
762 JumpVector m_jumps;
763 };

988 void emitNops(size_t memoryToFillWithNopsInBytes)
989 {
990#if CPU(ARM64)
991 RELEASE_ASSERT(memoryToFillWithNopsInBytes % 4 == 0);
992 for (unsigned i = 0; i < memoryToFillWithNopsInBytes / 4; ++i)
993 m_assembler.nop();
994#else
995 AssemblerBuffer& buffer = m_assembler.buffer();
996 size_t startCodeSize = buffer.codeSize();
997 size_t targetCodeSize = startCodeSize + memoryToFillWithNopsInBytes;
998 buffer.ensureSpace(memoryToFillWithNopsInBytes);
999 AssemblerType::template fillNops<memcpy>(static_cast<char*>(buffer.data()) + startCodeSize, memoryToFillWithNopsInBytes);
1000 buffer.setCodeSize(targetCodeSize);
1001#endif
1002 }
