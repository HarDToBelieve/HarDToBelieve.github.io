26#pragma once
27
28#include "Options.h"
29#include <wtf/HashMap.h>
30#include <wtf/NeverDestroyed.h>
31#include <wtf/PtrTag.h>
32
33namespace JSC {
34
35class JITOperationList {
36public:
37 static JITOperationList& instance();
38 static void initialize();
39
40 void* map(void* pointer) const
41 {
42#if ENABLE(JIT_OPERATION_VALIDATION)
43 return m_validatedOperations.get(removeCodePtrTag(pointer));
44#else
45 return pointer;
46#endif
47 }
48
49 static void populatePointersInJavaScriptCore();
50 static void populatePointersInJavaScriptCoreForLLInt();
51
52 JS_EXPORT_PRIVATE static void populatePointersInEmbedder(const uintptr_t* beginOperations, const uintptr_t* endOperations);
53
54 template<typename T> static void assertIsJITOperation(T function)
55 {
56 UNUSED_PARAM(function);
57#if ENABLE(JIT_OPERATION_VALIDATION)
58 ASSERT(!Options::useJIT() || JITOperationList::instance().map(bitwise_cast<void*>(function)));
59#endif
60 }
61
62private:
63 HashMap<void*, void*> m_validatedOperations;
64};
65
66JS_EXPORT_PRIVATE extern LazyNeverDestroyed<JITOperationList> jitOperationList;
67
68inline JITOperationList& JITOperationList::instance()
69{
70 return jitOperationList.get();
71}
72
73} // namespace JSC
