169 static DoubleCondition invert(DoubleCondition cond)
170 {
171 switch (cond) {
172 case DoubleEqualAndOrdered:
173 return DoubleNotEqualOrUnordered;
174 case DoubleNotEqualAndOrdered:
175 return DoubleEqualOrUnordered;
176 case DoubleGreaterThanAndOrdered:
177 return DoubleLessThanOrEqualOrUnordered;
178 case DoubleGreaterThanOrEqualAndOrdered:
179 return DoubleLessThanOrUnordered;
180 case DoubleLessThanAndOrdered:
181 return DoubleGreaterThanOrEqualOrUnordered;
182 case DoubleLessThanOrEqualAndOrdered:
183 return DoubleGreaterThanOrUnordered;
184 case DoubleEqualOrUnordered:
185 return DoubleNotEqualAndOrdered;
186 case DoubleNotEqualOrUnordered:
187 return DoubleEqualAndOrdered;
188 case DoubleGreaterThanOrUnordered:
189 return DoubleLessThanOrEqualAndOrdered;
190 case DoubleGreaterThanOrEqualOrUnordered:
191 return DoubleLessThanAndOrdered;
192 case DoubleLessThanOrUnordered:
193 return DoubleGreaterThanOrEqualAndOrdered;
194 case DoubleLessThanOrEqualOrUnordered:
195 return DoubleGreaterThanAndOrdered;
196 }
197 RELEASE_ASSERT_NOT_REACHED();
198 return DoubleEqualAndOrdered; // make compiler happy
199 }

217 case Zero:
218 return NonZero;
219 case NonZero:
220 return Zero;
221 case Signed:
222 return PositiveOrZero;
223 case PositiveOrZero:
224 return Signed;

226 RELEASE_ASSERT_NOT_REACHED();
227 return Zero; // Make compiler happy for release builds.

231 static RelationalCondition flip(RelationalCondition cond)
232 {
233 switch (cond) {
234 case Equal:
235 case NotEqual:
236 return cond;
237 case Above:
238 return Below;
239 case AboveOrEqual:
240 return BelowOrEqual;
241 case Below:
242 return Above;
243 case BelowOrEqual:
244 return AboveOrEqual;
245 case GreaterThan:
246 return LessThan;
247 case GreaterThanOrEqual:
248 return LessThanOrEqual;
249 case LessThan:
250 return GreaterThan;
251 case LessThanOrEqual:
252 return GreaterThanOrEqual;
253 }
254
255 RELEASE_ASSERT_NOT_REACHED();
256 return Equal;
257 }

469 // Commute a relational condition, returns a new condition that will produce
470 // the same results given the same inputs but with their positions exchanged.
471 static RelationalCondition commute(RelationalCondition condition)
472 {
473 switch (condition) {
474 case Above:
475 return Below;
476 case AboveOrEqual:
477 return BelowOrEqual;
478 case Below:
479 return Above;
480 case BelowOrEqual:
481 return AboveOrEqual;
482 case GreaterThan:
483 return LessThan;
484 case GreaterThanOrEqual:
485 return LessThanOrEqual;
486 case LessThan:
487 return GreaterThan;
488 case LessThanOrEqual:
489 return GreaterThanOrEqual;
490 default:
491 break;
492 }
493
494 ASSERT(condition == Equal || condition == NotEqual);
495 return condition;
496 }

1255 bool shouldBlindDouble(double value)
1256 {
1257 // Don't trust NaN or +/-Infinity
1258 if (!std::isfinite(value))
1259 return shouldConsiderBlinding();
1260
1261 // Try to force normalisation, and check that there's no change
1262 // in the bit pattern
1263 if (bitwise_cast<uint64_t>(value * 1.0) != bitwise_cast<uint64_t>(value))
1264 return shouldConsiderBlinding();
1265
1266 value = fabs(value);
1267 // Only allow a limited set of fractional components
1268 double scaledValue = value * 8;
1269 if (scaledValue / 8 != value)
1270 return shouldConsiderBlinding();
1271 double frac = scaledValue - floor(scaledValue);
1272 if (frac != 0.0)
1273 return shouldConsiderBlinding();
1274
1275 return value > 0xff;
1276 }

1928 // This function emits code to preserve the CPUState (e.g. registers),
1929 // call a user supplied probe function, and restore the CPUState before
1930 // continuing with other JIT generated code.
1931 //
1932 // The user supplied probe function will be called with a single pointer to
1933 // a Probe::State struct (defined below) which contains, among other things,
1934 // the preserved CPUState. This allows the user probe function to inspect
1935 // the CPUState at that point in the JIT generated code.
1936 //
1937 // If the user probe function alters the register values in the Probe::State,
1938 // the altered values will be loaded into the CPU registers when the probe
1939 // returns.
1940 //
1941 // The Probe::State is stack allocated and is only valid for the duration
1942 // of the call to the user probe function.
1943 //
1944 // The probe function may choose to move the stack pointer (in any direction).
1945 // To do this, the probe function needs to set the new sp value in the CPUState.
1946 //
1947 // The probe function may also choose to fill stack space with some values.
1948 // To do this, the probe function must first:
1949 // 1. Set the new sp value in the Probe::State's CPUState.
1950 // 2. Set the Probe::State's initializeStackFunction to a Probe::Function callback
1951 // which will do the work of filling in the stack values after the probe
1952 // trampoline has adjusted the machine stack pointer.
1953 // 3. Set the Probe::State's initializeStackArgs to any value that the client wants
1954 // to pass to the initializeStackFunction callback.
1955 // 4. Return from the probe function.
1956 //
1957 // Upon returning from the probe function, the probe trampoline will adjust the
1958 // the stack pointer based on the sp value in CPUState. If initializeStackFunction
1959 // is not set, the probe trampoline will restore registers and return to its caller.
1960 //
1961 // If initializeStackFunction is set, the trampoline will move the Probe::State
1962 // beyond the range of the stack pointer i.e. it will place the new Probe::State at
1963 // an address lower than where CPUState.sp() points. This ensures that the
1964 // Probe::State will not be trashed by the initializeStackFunction when it writes to
1965 // the stack. Then, the trampoline will call back to the initializeStackFunction
1966 // Probe::Function to let it fill in the stack values as desired. The
1967 // initializeStackFunction Probe::Function will be passed the moved Probe::State at
1968 // the new location.
1969 //
1970 // initializeStackFunction may now write to the stack at addresses greater or
1971 // equal to CPUState.sp(), but not below that. initializeStackFunction is also
1972 // not allowed to change CPUState.sp(). If the initializeStackFunction does not
1973 // abide by these rules, then behavior is undefined, and bad things may happen.
1974 //
1975 // Note: this version of probe() should be implemented by the target specific
1976 // MacroAssembler.
