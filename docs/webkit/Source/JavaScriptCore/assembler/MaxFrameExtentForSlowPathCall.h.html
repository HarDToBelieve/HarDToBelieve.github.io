27
28#include "Register.h"
29#include "StackAlignment.h"
30#include <wtf/Assertions.h>
31
32namespace JSC {
33
34// The maxFrameExtentForSlowPathCall is the max amount of stack space (in bytes)
35// that can be used for outgoing args when calling a slow path C function
36// from JS code.
37
38#if !ENABLE(ASSEMBLER)
39static constexpr size_t maxFrameExtentForSlowPathCall = 0;
40
41#elif CPU(X86_64) && OS(WINDOWS)
42// 4 args in registers, but stack space needs to be allocated for all args.
43static constexpr size_t maxFrameExtentForSlowPathCall = 64;
44
45#elif CPU(X86_64)
46// All args in registers.
47static constexpr size_t maxFrameExtentForSlowPathCall = 0;
48
49#elif CPU(X86)
50// 7 args on stack (28 bytes).
51static constexpr size_t maxFrameExtentForSlowPathCall = 40;
52
53#elif CPU(ARM64) || CPU(ARM64E)
54// All args in registers.
55static constexpr size_t maxFrameExtentForSlowPathCall = 0;
56
57#elif CPU(ARM)
58// First four args in registers, remaining 4 args on stack.
59static constexpr size_t maxFrameExtentForSlowPathCall = 24;
60
61#elif CPU(MIPS)
62// Though args are in registers, there need to be space on the stack for all args.
63static constexpr size_t maxFrameExtentForSlowPathCall = 40;
64
65#else
66#error "Unsupported CPU: need value for maxFrameExtentForSlowPathCall"
67
68#endif
69
70static_assert(!(maxFrameExtentForSlowPathCall % sizeof(Register)), "Extent must be in multiples of registers");
71
72#if ENABLE(ASSEMBLER)
73// Make sure that cfr - maxFrameExtentForSlowPathCall bytes will make the stack pointer aligned
74static_assert((maxFrameExtentForSlowPathCall % 16) == 16 - sizeof(CallerFrameAndPC), "Extent must align stack from callframe pointer");
75#endif
76
77static constexpr size_t maxFrameExtentForSlowPathCallInRegisters = maxFrameExtentForSlowPathCall / sizeof(Register);
78
79} // namespace JSC
