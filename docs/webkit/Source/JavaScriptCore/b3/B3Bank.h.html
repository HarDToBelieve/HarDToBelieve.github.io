33namespace JSC { namespace B3 {
34
35enum Bank : int8_t {
36 GP,
37 FP
38};
39
40static constexpr unsigned numBanks = 2;
41
42template<typename Func>
43void forEachBank(const Func& func)
44{
45 func(GP);
46 func(FP);
47}
48
49inline Bank bankForType(Type type)
50{
51 switch (type.kind()) {
52 case Void:
53 case Tuple:
54 ASSERT_NOT_REACHED();
55 return GP;
56 case Int32:
57 case Int64:
58 return GP;
59 case Float:
60 case Double:
61 return FP;
62 }
63 ASSERT_NOT_REACHED();
64 return GP;
65}
66
67inline Bank bankForReg(Reg reg)
68{
69 return reg.isGPR() ? GP : FP;
70}
