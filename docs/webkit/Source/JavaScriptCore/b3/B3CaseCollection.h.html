26#pragma once
27
28#if ENABLE(B3_JIT)
29
30#include "B3SwitchCase.h"
31
32namespace JSC { namespace B3 {
33
34class BasicBlock;
35class SwitchValue;
36
37// NOTE: You'll always want to include B3CaseCollectionInlines.h when you use this.
38
39class CaseCollection {
40public:
41 CaseCollection()
42 {
43 }
44
45 CaseCollection(const SwitchValue* terminal, const BasicBlock* owner)
46 : m_switch(terminal)
47 , m_owner(owner)
48 {
49 }
50
51 const FrequentedBlock& fallThrough() const;
52
53 unsigned size() const;
54 SwitchCase at(unsigned index) const;
55
56 SwitchCase operator[](unsigned index) const
57 {
58 return at(index);
59 }
60
61 class iterator {
62 public:
63 iterator()
64 : m_collection(nullptr)
65 , m_index(0)
66 {
67 }
68
69 iterator(const CaseCollection& collection, unsigned index)
70 : m_collection(&collection)
71 , m_index(index)
72 {
73 }
74
75 SwitchCase operator*()
76 {
77 return m_collection->at(m_index);
78 }
79
80 iterator& operator++()
81 {
82 m_index++;
83 return *this;
84 }
85
86 bool operator==(const iterator& other) const
87 {
88 ASSERT(m_collection == other.m_collection);
89 return m_index == other.m_index;
90 }
91
92 bool operator!=(const iterator& other) const
93 {
94 return !(*this == other);
95 }
96
97 private:
98 const CaseCollection* m_collection;
99 unsigned m_index;
100 };
101
102 typedef iterator const_iterator;
103
104 iterator begin() const { return iterator(*this, 0); }
105 iterator end() const { return iterator(*this, size()); }
106
107 void dump(PrintStream&) const;
108
109private:
110 const SwitchValue* m_switch { nullptr };
111 const BasicBlock* m_owner { nullptr };
112};
113
114} } // namespace JSC::B3
115
116#endif // ENABLE(B3_JIT)
