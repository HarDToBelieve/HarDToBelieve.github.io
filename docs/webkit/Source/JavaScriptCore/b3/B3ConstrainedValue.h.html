26#pragma once
27
28#if ENABLE(B3_JIT)
29
30#include "B3ValueRep.h"
31
32namespace JSC { namespace B3 {
33
34class Value;
35
36class ConstrainedValue {
37public:
38 ConstrainedValue()
39 {
40 }
41
42 ConstrainedValue(Value* value)
43 : m_value(value)
44 , m_rep(ValueRep::WarmAny)
45 {
46 }
47
48 ConstrainedValue(Value* value, const ValueRep& rep)
49 : m_value(value)
50 , m_rep(rep)
51 {
52 }
53
54 explicit operator bool() const { return m_value || m_rep; }
55
56 Value* value() const { return m_value; }
57 const ValueRep& rep() const { return m_rep; }
58
59 void dump(PrintStream& out) const;
60
61private:
62 Value* m_value;
63 ValueRep m_rep;
64};
65
66} } // namespace JSC::B3
67
68#endif // ENABLE(B3_JIT)
