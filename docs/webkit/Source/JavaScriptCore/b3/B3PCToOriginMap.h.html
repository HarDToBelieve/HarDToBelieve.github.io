26#pragma once

28#if ENABLE(B3_JIT)
29
30#include "B3Origin.h"
31#include "MacroAssembler.h"
32#include <wtf/Vector.h>
33
34namespace JSC { namespace B3 {
35
36class PCToOriginMap {
37 WTF_MAKE_NONCOPYABLE(PCToOriginMap);
38public:
39 PCToOriginMap()
40 { }
41
42 PCToOriginMap(PCToOriginMap&& other)
43 : m_ranges(WTFMove(other.m_ranges))
44 { }
45
46 struct OriginRange {
47 MacroAssembler::Label label;
48 Origin origin;
49 };
50
51 void appendItem(MacroAssembler::Label label, Origin origin)
52 {
53 if (m_ranges.size()) {
54 if (m_ranges.last().label == label)
55 return;
56 }
57
58 m_ranges.append(OriginRange{label, origin});
59 }
60
61 const Vector<OriginRange>& ranges() const { return m_ranges; }
62
63private:
64 Vector<OriginRange, 0> m_ranges;
65};
66
67} } // namespace JSC::B3

69#endif // ENABLE(B3_JIT)
