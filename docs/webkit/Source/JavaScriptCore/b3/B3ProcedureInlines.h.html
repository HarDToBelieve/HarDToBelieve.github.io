37template<typename ValueType, typename... Arguments>
38ValueType* Procedure::add(Arguments... arguments)
39{
40 return static_cast<ValueType*>(addValueImpl(Value::allocate<ValueType>(arguments...)));
41}
42
43inline Type Procedure::extractFromTuple(Type tuple, unsigned index) const
44{
45 ASSERT(tuple.tupleIndex() < m_tuples.size());
46 ASSERT(index < m_tuples[tuple.tupleIndex()].size());
47 return m_tuples[tuple.tupleIndex()][index];
48}
