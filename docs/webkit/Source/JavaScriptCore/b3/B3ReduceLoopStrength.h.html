40#if CPU(X86_64)
41/*
42 * The semantics of B3 require that loads and stores are not reduced in granularity.
43 * If we would use system memcpy, then it is possible that we would get a byte copy loop,
44 * and this could cause GC tearing.
45 */
46ALWAYS_INLINE void fastForwardCopy32(uint32_t* dst, const uint32_t* src, size_t size)
47{
48 uint64_t i;
49 int64_t counter;
50 uint64_t tmp, tmp2;
51
52 asm volatile(
53 "test %q[size], %q[size]\t\n"
54 "jz 900f\t\n" // exit
55 "xorq %q[i], %q[i]\t\n"
56 // if size < 8
57 "cmp $8, %q[size]\t\n"
58 "jl 100f\t\n" // backporch
59 // if (dst + size*4 <= src || src + size*4 <= dst)
60 "leaq (%q[src], %q[size], 4), %q[tmp]\t\n"
61 "cmp %q[tmp], %q[dst]\t\n"
62 "jae 500f\t\n" // simd no overlap
63 "leaq (%q[dst], %q[size], 4), %q[tmp]\t\n"
64 "cmp %q[tmp], %q[src]\t\n"
65 "jae 500f\t\n" // simd no overlap
66 "jmp 100f\t\n" // Backporch
67
68 // Backporch (assumes we can write at least one int)
69 "100:\t\n"
70 // counter = (size - i)%4
71 "mov %q[size], %q[counter]\t\n"
72 "subq %q[i], %q[counter]\t\n"
73
74 // If we are a multiple of 4, unroll the loop
75 // We already know we are nonzero
76 "and $3, %q[counter]\t\n"
77 "jz 200f\t\n" // unrolled loop
78
79 "negq %q[counter]\t\n"
80 ".balign 32\t\n"
81 "101:\t\n"
82 "mov (%q[src], %q[i], 4), %k[tmp]\t\n"
83 "mov %k[tmp], (%q[dst], %q[i], 4)\t\n"
84 "incq %q[i]\t\n"
85 "incq %q[counter]\t\n"
86 "jnz 101b\t\n"
87 // Do we still have anything left?
88 "cmpq %q[size], %q[i]\t\n"
89 "je 900f\t\n" // exit
90 // Then go to the unrolled loop
91 "jmp 200f\t\n"
92
93 // Unrolled loop (assumes multiple of 4, can do one iteration)
94 "200:\t\n"
95 "shr $2, %q[counter]\t\n"
96 "negq %q[counter]\t\n"
97 ".balign 32\t\n"
98 "201:\t\n"
99 "mov (%q[src], %q[i], 4), %k[tmp]\t\n"
100 "mov %k[tmp], (%q[dst], %q[i], 4)\t\n"
101 "mov 4(%q[src], %q[i], 4), %k[tmp2]\t\n"
102 "mov %k[tmp2], 4(%q[dst], %q[i], 4)\t\n"
103 "mov 8(%q[src], %q[i], 4), %k[tmp2]\t\n"
104 "mov %k[tmp2], 8(%q[dst], %q[i], 4)\t\n"
105 "mov 12(%q[src], %q[i], 4), %k[tmp]\t\n"
106 "mov %k[tmp], 12(%q[dst], %q[i], 4)\t\n"
107 "addq $4, %q[i]\t\n"
108 "cmpq %q[size], %q[i]\t\n"
109 "jb 201b\t\n"
110 "jmp 900f\t\n" // exit
111
112 // simd no overlap
113 // counter = -(size/8);
114 "500:\t\n"
115 "mov %q[size], %q[counter]\t\n"
116 "shrq $3, %q[counter]\t\n"
117 "negq %q[counter]\t\n"
118 ".balign 32\t\n"
119 "501:\t\n"
120 "movups (%q[src], %q[i], 4), %%xmm0\t\n"
121 "movups 16(%q[src], %q[i], 4), %%xmm1\t\n"
122 "movups %%xmm0, (%q[dst], %q[i], 4)\t\n"
123 "movups %%xmm1, 16(%q[dst], %q[i], 4)\t\n"
124 "addq $8, %q[i]\t\n"
125 "incq %q[counter]\t\n"
126 "jnz 501b\t\n"
127 // if (i == size)
128 "cmpq %q[size], %q[i]\t\n"
129 "je 900f\t\n" // end
130 "jmp 100b\t\n" // backporch
131 "900:\t\n"
132 : [i] "=&c" (i), [counter] "=&r" (counter), [tmp] "=&a" (tmp), [tmp2] "=&r" (tmp2)
133 : [dst] "D" (dst), [src] "S" (src), [size] "r" (size)
134 : "xmm0", "xmm1");
135}
