26#pragma once
27
28#if ENABLE(B3_JIT)
29
30#include "B3Value.h"
31#include <wtf/IndexMap.h>
32
33namespace JSC { namespace B3 {
34
35class Procedure;
36
37class UseCounts {
38public:
39 JS_EXPORT_PRIVATE UseCounts(Procedure&);
40 JS_EXPORT_PRIVATE ~UseCounts();
41
42 unsigned numUses(Value* value) const { return m_counts[value].numUses; }
43 unsigned numUsingInstructions(Value* value) const { return m_counts[value].numUsingInstructions; }
44
45private:
46 struct Counts {
47 unsigned numUses { 0 };
48 unsigned numUsingInstructions { 0 };
49 };
50
51 IndexMap<Value*, Counts> m_counts;
52};
53
54} } // namespace JSC::B3
55
56#endif // ENABLE(B3_JIT)
