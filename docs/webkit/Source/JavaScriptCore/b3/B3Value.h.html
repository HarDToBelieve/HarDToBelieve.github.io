143 // value with Identity. Then do a pass of performSubstitution() on all of the values that use

148 // It's often necessary to kill a value. It's tempting to replace the value with Nop or to
149 // just remove it. But unless you are sure that the value is Void, you will probably still
150 // have other values that use this one. Sure, you may kill those later, or you might not. This
151 // method lets you kill a value safely. It will replace Void values with Nop and non-Void
152 // values with Identities on bottom constants. For this reason, this takes a callback that is
153 // responsible for creating bottoms. There's a utility for this, see B3BottomProvider.h. You
154 // can also access that utility using replaceWithBottom(InsertionSet&, size_t).
155 //
156 // You're guaranteed that bottom is zero.

328 // B3 purposefully only represents signed 32-bit offsets because that's what x86 can encode, and
329 // ARM64 cannot encode anything bigger. The IsLegalOffset type trait is then used on B3 Value
330 // methods to prevent implicit conversions by C++ from invalid offset types: these cause compilation
331 // to fail, instead of causing implementation-defined behavior (which often turns to exploit).
332 // OffsetType isn't sufficient to determine offset validity! Each Value opcode further has an
333 // isLegalOffset runtime method used to determine value legality at runtime. This is exposed to users
334 // of B3 to force them to reason about the target's offset.
335 typedef int32_t OffsetType;
336 template<typename Int>
337 struct IsLegalOffset {
338 static constexpr bool value = std::is_integral<Int>::value
339 && std::is_signed<Int>::value
340 && sizeof(Int) <= sizeof(OffsetType);
341 };

381 {
382 switch (kind.opcode()) {
383 case FramePointer:
384 case Nop:
385 case Phi:
386 case Jump:
387 case Oops:
388 case EntrySwitch:
389 case ArgumentReg:
390 case Const32:
391 case Const64:
392 case ConstFloat:
393 case ConstDouble:
394 case BottomTuple:
395 case Fence:
396 case SlotBase:
397 case Get:
398 return 0;
399 case Return:
400 case Identity:
401 case Opaque:
402 case Neg:
403 case Clz:
404 case Abs:
405 case Ceil:
406 case Floor:
407 case Sqrt:
408 case SExt8:
409 case SExt16:
410 case Trunc:
411 case SExt32:
412 case ZExt32:
413 case FloatToDouble:
414 case IToD:
415 case DoubleToFloat:
416 case IToF:
417 case BitwiseCast:
418 case Branch:
419 case Depend:
420 case Load8Z:
421 case Load8S:
422 case Load16Z:
423 case Load16S:
424 case Load:
425 case Switch:
426 case Upsilon:
427 case Extract:
428 case Set:
429 case WasmAddress:
430 case WasmBoundsCheck:
431 return sizeof(Value*);
432 case Add:
433 case Sub:
434 case Mul:
435 case Div:
436 case UDiv:
437 case Mod:
438 case UMod:
439 case BitAnd:
440 case BitOr:
441 case BitXor:
442 case Shl:
443 case SShr:
444 case ZShr:
445 case RotR:
446 case RotL:
447 case Equal:
448 case NotEqual:
449 case LessThan:
450 case GreaterThan:
451 case LessEqual:
452 case GreaterEqual:
453 case Above:
454 case Below:
455 case AboveEqual:
456 case BelowEqual:
457 case EqualOrUnordered:
458 case AtomicXchgAdd:
459 case AtomicXchgAnd:
460 case AtomicXchgOr:
461 case AtomicXchgSub:
462 case AtomicXchgXor:
463 case AtomicXchg:
464 case Store8:
465 case Store16:

479#ifdef NDEBUG
480 default:
481 break;
482#endif
483 }
484 RELEASE_ASSERT_NOT_REACHED();
485 return 0;

489 static char* allocateSpace(Opcode opcode, size_t size)
490 {
491 size_t adjacencyListSpace = Value::adjacencyListSpace(opcode);
492 // We must allocate enough space that replaceWithIdentity can work without buffer overflow.
493 size_t allocIdentitySize = sizeof(Value) + sizeof(Value*);
494 size_t allocSize = std::max(size + adjacencyListSpace, allocIdentitySize);
495 return static_cast<char*>(WTF::fastMalloc(allocSize));
496 }

527 template<typename... Arguments>
528 void buildAdjacencyList(NumChildren numChildren, Arguments... arguments)
529 {
530 if (numChildren == VarArgs) {
531 new (childrenAlloc()) Vector<Value*, 3> { arguments... };
532 return;
533 }
534 ASSERT(numChildren == sizeof...(arguments));
535 new (childrenAlloc()) Value*[sizeof...(arguments)] { arguments... };
536 }
537 void buildAdjacencyList(size_t offset, const Value& valueToClone)
538 {
539 switch (valueToClone.m_numChildren) {
540 case VarArgs:
541 new (bitwise_cast<char*>(this) + offset) Vector<Value*, 3> (valueToClone.childrenVector());
542 break;
543 case Three:
544 bitwise_cast<Value**>(bitwise_cast<char*>(this) + offset)[2] = valueToClone.childrenArray()[2];
545 FALLTHROUGH;
546 case Two:
547 bitwise_cast<Value**>(bitwise_cast<char*>(this) + offset)[1] = valueToClone.childrenArray()[1];
548 FALLTHROUGH;
549 case One:
550 bitwise_cast<Value**>(bitwise_cast<char*>(this) + offset)[0] = valueToClone.childrenArray()[0];
551 break;
552 case Zero:
553 break;
554 }
555 }

557 // Checks that this kind is valid for use with B3::Value.
558 ALWAYS_INLINE static NumChildren numChildrenForKind(Kind kind, unsigned numArgs)
559 {
560 switch (kind.opcode()) {
561 case FramePointer:
562 case Nop:
563 case Phi:
564 case Jump:
565 case Oops:
566 case EntrySwitch:
567 if (UNLIKELY(numArgs))
568 badKind(kind, numArgs);
569 return Zero;
570 case Return:
571 if (UNLIKELY(numArgs > 1))
572 badKind(kind, numArgs);
573 return numArgs ? One : Zero;
574 case Identity:
575 case Opaque:
576 case Neg:
577 case Clz:
578 case Abs:
579 case Ceil:
580 case Floor:
581 case Sqrt:
582 case SExt8:
583 case SExt16:
584 case Trunc:
585 case SExt32:
586 case ZExt32:
587 case FloatToDouble:
588 case IToD:
589 case DoubleToFloat:
590 case IToF:
591 case BitwiseCast:
592 case Branch:
593 case Depend:
594 if (UNLIKELY(numArgs != 1))
595 badKind(kind, numArgs);
596 return One;
597 case Add:
598 case Sub:
599 case Mul:
600 case Div:
601 case UDiv:
602 case Mod:
603 case UMod:
604 case BitAnd:
605 case BitOr:
606 case BitXor:
607 case Shl:
608 case SShr:
609 case ZShr:
610 case RotR:
611 case RotL:
612 case Equal:
613 case NotEqual:
614 case LessThan:
615 case GreaterThan:
616 case LessEqual:
617 case GreaterEqual:
618 case Above:
619 case Below:
620 case AboveEqual:
621 case BelowEqual:
622 case EqualOrUnordered:
623 if (UNLIKELY(numArgs != 2))
624 badKind(kind, numArgs);
625 return Two;
626 case Select:
627 if (UNLIKELY(numArgs != 3))
628 badKind(kind, numArgs);
629 return Three;
630 default:
631 badKind(kind, numArgs);
632 break;
633 }
634 return VarArgs;
635 }

755// The following macros are designed for subclasses of B3::Value to use.
756// They are never required for correctness, but can improve the performance of child/lastChild/numChildren/children methods,
757// for users that already know the specific subclass of Value they are manipulating.
758// The first set is to be used when you know something about the number of children of all values of a class, including its subclasses:
759// - B3_SPECIALIZE_VALUE_FOR_NO_CHILDREN: always 0 children (e.g. Const32Value)
760// - B3_SPECIALIZE_VALUE_FOR_FIXED_CHILDREN(n): always n children, with n in {1, 2, 3} (e.g. UpsilonValue, with n = 1)
761// - B3_SPECIALIZE_VALUE_FOR_NON_VARARGS_CHILDREN: different numbers of children, but never a variable number at runtime (e.g. MemoryValue, that can have between 1 and 3 children)
762// - B3_SPECIALIZE_VALUE_FOR_VARARGS_CHILDREN: always a varargs (e.g. CCallValue)
763// The second set is only to be used by classes that we know are not further subclassed by anyone adding fields,
764// as they hardcode the offset of the children array/vector (which is equal to the size of the object).
765// - B3_SPECIALIZE_VALUE_FOR_FINAL_SIZE_FIXED_CHILDREN
766// - B3_SPECIALIZE_VALUE_FOR_FINAL_SIZE_VARARGS_CHILDREN
767#define B3_SPECIALIZE_VALUE_FOR_NO_CHILDREN \
768 unsigned numChildren() const { return 0; } \
769 WTF::IteratorRange<Value**> children() { return {nullptr, nullptr}; } \
770 WTF::IteratorRange<Value* const*> children() const { return { nullptr, nullptr}; }
771
772#define B3_SPECIALIZE_VALUE_FOR_FIXED_CHILDREN(n) \
773public: \
774 unsigned numChildren() const { return n; } \
775 Value*& child(unsigned index) \
776 { \
777 ASSERT(index <= n); \
778 return childrenArray()[index]; \
779 } \
780 Value* child(unsigned index) const \
781 { \
782 ASSERT(index <= n); \
783 return childrenArray()[index]; \
784 } \
785 Value*& lastChild() \
786 { \
787 return childrenArray()[n - 1]; \
788 } \
789 Value* lastChild() const \
790 { \
791 return childrenArray()[n - 1]; \
792 } \
793 WTF::IteratorRange<Value**> children() \
794 { \
795 Value** buffer = childrenArray(); \
796 return {buffer, buffer + n }; \
797 } \
798 WTF::IteratorRange<Value* const*> children() const \
799 { \
800 Value* const* buffer = childrenArray(); \
801 return {buffer, buffer + n }; \
802 } \
803
804#define B3_SPECIALIZE_VALUE_FOR_NON_VARARGS_CHILDREN \
805public: \
806 unsigned numChildren() const { return m_numChildren; } \
807 Value*& child(unsigned index) { return childrenArray()[index]; } \
808 Value* child(unsigned index) const { return childrenArray()[index]; } \
809 Value*& lastChild() { return childrenArray()[numChildren() - 1]; } \
810 Value* lastChild() const { return childrenArray()[numChildren() - 1]; } \
811 WTF::IteratorRange<Value**> children() \
812 { \
813 Value** buffer = childrenArray(); \
814 return {buffer, buffer + numChildren() }; \
815 } \
816 WTF::IteratorRange<Value* const*> children() const \
817 { \
818 Value* const* buffer = childrenArray(); \
819 return {buffer, buffer + numChildren() }; \
820 } \
821
822#define B3_SPECIALIZE_VALUE_FOR_VARARGS_CHILDREN \
823public: \
824 unsigned numChildren() const { return childrenVector().size(); } \
825 Value*& child(unsigned index) { return childrenVector()[index]; } \
826 Value* child(unsigned index) const { return childrenVector()[index]; } \
827 Value*& lastChild() { return childrenVector().last(); } \
828 Value* lastChild() const { return childrenVector().last(); } \
829 WTF::IteratorRange<Value**> children() \
830 { \
831 Vector<Value*, 3>& vec = childrenVector(); \
832 return WTF::makeIteratorRange(&*vec.begin(), &*vec.end()); \
833 } \
834 WTF::IteratorRange<Value* const*> children() const \
835 { \
836 const Vector<Value*, 3>& vec = childrenVector(); \
837 return WTF::makeIteratorRange(&*vec.begin(), &*vec.end()); \
838 } \

840// Only use this for classes with no subclass that add new fields (as it uses sizeof(*this))
841// Also there is no point in applying this to classes with no children, as they don't have a children array to access.
842#define B3_SPECIALIZE_VALUE_FOR_FINAL_SIZE_FIXED_CHILDREN \
843private: \
844 Value** childrenArray() \
845 { \
846 return bitwise_cast<Value**>(bitwise_cast<char*>(this) + sizeof(*this)); \
847 } \
848 Value* const* childrenArray() const \
849 { \
850 return bitwise_cast<Value* const*>(bitwise_cast<char const*>(this) + sizeof(*this)); \
851 }
852
853// Only use this for classes with no subclass that add new fields (as it uses sizeof(*this))
854#define B3_SPECIALIZE_VALUE_FOR_FINAL_SIZE_VARARGS_CHILDREN \
855private: \
856 Vector<Value*, 3>& childrenVector() \
857 { \
858 return *bitwise_cast<Vector<Value*, 3>*>(bitwise_cast<char*>(this) + sizeof(*this)); \
859 } \
860 const Vector<Value*, 3>& childrenVector() const \
861 { \
862 return *bitwise_cast<Vector<Value*, 3> const*>(bitwise_cast<char const*>(this) + sizeof(*this)); \
863 } \
