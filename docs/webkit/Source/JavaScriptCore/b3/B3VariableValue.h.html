30#include "B3Value.h"

32namespace JSC { namespace B3 {
33
34class Variable;
35
36class JS_EXPORT_PRIVATE VariableValue final : public Value {
37public:
38 static bool accepts(Kind kind) { return kind == Get || kind == Set; }
39
40 ~VariableValue() final;
41
42 Variable* variable() const { return m_variable; }
43
44 B3_SPECIALIZE_VALUE_FOR_NON_VARARGS_CHILDREN
45 B3_SPECIALIZE_VALUE_FOR_FINAL_SIZE_FIXED_CHILDREN
46
47private:
48 void dumpMeta(CommaPrinter&, PrintStream&) const final;
49
50 friend class Procedure;
51 friend class Value;
52
53 // Use this for Set.
54 VariableValue(Kind, Origin, Variable*, Value*);
55
56 // Use this for Get.
57 VariableValue(Kind, Origin, Variable*);
58
59 Variable* m_variable;
60};
61
62} } // namespace JSC::B3

64#endif // ENABLE(B3_JIT)
