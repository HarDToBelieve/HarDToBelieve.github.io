54 // These enum members are intentionally terse because we have to mention them a lot.
55 enum Kind : int8_t {
56 Invalid,
57
58 // This is either an unassigned temporary or a register. All unassigned temporaries
59 // eventually become registers.
60 Tmp,
61
62 // This is an immediate that the instruction will materialize. Imm is the immediate that can be
63 // inlined into most instructions, while BigImm indicates a constant materialization and is
64 // usually only usable with Move. Specials may also admit it, for example for stackmaps used for
65 // OSR exit and tail calls.
66 // BitImm is an immediate for Bitwise operation (And, Xor, etc).
67 Imm,
68 BigImm,
69 BitImm,
70 BitImm64,
71
72 // These are the addresses. Instructions may load from (Use), store to (Def), or evaluate
73 // (UseAddr) addresses.
74 SimpleAddr,
75 Addr,
76 ExtendedOffsetAddr,
77 Stack,
78 CallArg,
79 Index,
80
81 // Immediate operands that customize the behavior of an operation. You can think of them as
82 // secondary opcodes. They are always "Use"'d.
83 RelCond,
84 ResCond,
85 DoubleCond,
86 StatusCond,
87 Special,
88 WidthArg
89 };

107 enum Role : int8_t {
108 // Use means that the Inst will read from this value before doing anything else.
109 //
110 // For Tmp: The Inst will read this Tmp.
111 // For Arg::addr and friends: The Inst will load from this address.
112 // For Arg::imm and friends: The Inst will materialize and use this immediate.
113 // For RelCond/ResCond/Special: This is the only valid role for these kinds.
114 //
115 // Note that Use of an address does not mean escape. It only means that the instruction will
116 // load from the address before doing anything else. This is a bit tricky; for example
117 // Specials could theoretically squirrel away the address and effectively escape it. However,
118 // this is not legal. On the other hand, any address other than Stack is presumed to be
119 // always escaping, and Stack is presumed to be always escaping if it's Locked.
120 Use,
121
122 // Exactly like Use, except that it also implies that the use is cold: that is, replacing the
123 // use with something on the stack is free.
124 ColdUse,
125
126 // LateUse means that the Inst will read from this value after doing its Def's. Note that LateUse
127 // on an Addr or Index still means Use on the internal temporaries. Note that specifying the
128 // same Tmp once as Def and once as LateUse has undefined behavior: the use may happen before
129 // the def, or it may happen after it.
130 LateUse,
131
132 // Combination of LateUse and ColdUse.
133 LateColdUse,
134
135 // Def means that the Inst will write to this value after doing everything else.
136 //
137 // For Tmp: The Inst will write to this Tmp.
138 // For Arg::addr and friends: The Inst will store to this address.
139 // This isn't valid for any other kinds.
140 //
141 // Like Use of address, Def of address does not mean escape.
142 Def,
143
144 // This is a special variant of Def that implies that the upper bits of the target register are
145 // zero-filled. Specifically, if the Width of a ZDef is less than the largest possible width of
146 // the argument (for example, we're on a 64-bit machine and we have a Width32 ZDef of a GPR) then
147 // this has different implications for the upper bits (i.e. the top 32 bits in our example)
148 // depending on the kind of the argument:
149 //
150 // For register: the upper bits are zero-filled.
151 // For anonymous stack slot: the upper bits are zero-filled.
152 // For address: the upper bits are not touched (i.e. we do a 32-bit store in our example).
153 // For tmp: either the upper bits are not touched or they are zero-filled, and we won't know
154 // which until we lower the tmp to either a StackSlot or a Reg.
155 //
156 // The behavior of ZDef is consistent with what happens when you perform 32-bit operations on a
157 // 64-bit GPR. It's not consistent with what happens with 8-bit or 16-bit Defs on x86 GPRs, or
158 // what happens with float Defs in ARM NEON or X86 SSE. Hence why we have both Def and ZDef.
159 ZDef,
160
161 // This is a combined Use and Def. It means that both things happen.
162 UseDef,
163
164 // This is a combined Use and ZDef. It means that both things happen.
165 UseZDef,
166
167 // This is like Def, but implies that the assignment occurs before the start of the Inst's
168 // execution rather than after. Note that specifying the same Tmp once as EarlyDef and once
169 // as Use has undefined behavior: the use may happen before the def, or it may happen after
170 // it.
171 EarlyDef,
172
173 EarlyZDef,
174
175 // Some instructions need a scratch register. We model this by saying that the temporary is
176 // defined early and used late. This role implies that.
177 Scratch,
178
179 // This is a special kind of use that is only valid for addresses. It means that the
180 // instruction will evaluate the address expression and consume the effective address, but it
181 // will neither load nor store. This is an escaping use, because now the address may be
182 // passed along to who-knows-where. Note that this isn't really a Use of the Arg, but it does
183 // imply that we're Use'ing any registers that the Arg contains.
184 UseAddr
185 };

192 // Returns true if the Role implies that the Inst will Use the Arg. It's deliberately false for
193 // UseAddr, since isAnyUse() for an Arg::addr means that we are loading from the address.
194 static bool isAnyUse(Role role)
195 {
196 switch (role) {
197 case Use:
198 case ColdUse:
199 case UseDef:
200 case UseZDef:
201 case LateUse:
202 case LateColdUse:
203 case Scratch:
204 return true;
205 case Def:
206 case ZDef:
207 case UseAddr:
208 case EarlyDef:
209 case EarlyZDef:
210 return false;
211 }
212 ASSERT_NOT_REACHED();
213 }

587 // If you don't pass a Width, this optimistically assumes that you're using the right width.
588 static bool isValidScale(unsigned scale, std::optional<Width> width = std::nullopt)
589 {
590 switch (scale) {
591 case 1:
592 if (isX86() || isARM64())
593 return true;
594 return false;
595 case 2:
596 case 4:
597 case 8:
598 if (isX86())
599 return true;
600 if (isARM64()) {
601 if (!width)
602 return true;
603 return scale == 1 || scale == bytes(*width);
604 }
605 return false;
606 default:
607 return false;
608 }
609 }

945 {

1292 // This is smart enough to know that an address arg in a Def or UseDef rule will use its
1293 // tmps and never def them. For example, this:
1294 //
1295 // mov %rax, (%rcx)
1296 //
1297 // This defs (%rcx) but uses %rcx.

1300 {

1302 case Tmp:

1307 case Addr:

1313 functor(m_index, Use, GP, argRole == UseAddr ? argWidth : pointerWidth());
