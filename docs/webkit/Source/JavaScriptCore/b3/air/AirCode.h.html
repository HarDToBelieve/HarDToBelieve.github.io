144 template<typename Func>
145 void forEachTmp(const Func& func)
146 {
147 for (unsigned bankIndex = 0; bankIndex < numBanks; ++bankIndex) {
148 Bank bank = static_cast<Bank>(bankIndex);
149 unsigned numTmps = this->numTmps(bank);
150 for (unsigned i = 0; i < numTmps; ++i)
151 func(Tmp::tmpForIndex(bank, i));
152 }
153 }

254 class iterator {
255 public:
256 iterator()
257 : m_code(nullptr)
258 , m_index(0)
259 {
260 }
261
262 iterator(const Code& code, unsigned index)
263 : m_code(&code)
264 , m_index(m_code->findFirstBlockIndex(index))
265 {
266 }
267
268 BasicBlock* operator*()
269 {
270 return m_code->at(m_index);
271 }
272
273 iterator& operator++()
274 {
275 m_index = m_code->findFirstBlockIndex(m_index + 1);
276 return *this;
277 }
278
279 bool operator==(const iterator& other) const
280 {
281 return m_index == other.m_index;
282 }
283
284 bool operator!=(const iterator& other) const
285 {
286 return !(*this == other);
287 }
288
289 private:
290 const Code* m_code;
291 unsigned m_index;
292 };
