28#if ENABLE(B3_JIT)
29
30#include "AirArg.h"
31
32namespace JSC { namespace B3 { namespace Air {
33
34static constexpr uint8_t formRoleShift = 0;
35static constexpr uint8_t formRoleMask = 15;
36static constexpr uint8_t formBankShift = 4;
37static constexpr uint8_t formBankMask = 1;
38static constexpr uint8_t formWidthShift = 5;
39static constexpr uint8_t formWidthMask = 3;
40static constexpr uint8_t formInvalidShift = 7;
41
42#define ENCODE_INST_FORM(role, bank, width) (static_cast<uint8_t>(role) << formRoleShift | static_cast<uint8_t>(bank) << formBankShift | static_cast<uint8_t>(width) << formWidthShift)
43
44#define INVALID_INST_FORM (1 << formInvalidShift)
45
46JS_EXPORT_PRIVATE extern const uint8_t g_formTable[];
47
48inline Arg::Role decodeFormRole(uint8_t value)
49{
50 return static_cast<Arg::Role>((value >> formRoleShift) & formRoleMask);
51}
52
53inline Bank decodeFormBank(uint8_t value)
54{
55 return static_cast<Bank>((value >> formBankShift) & formBankMask);
56}
57
58inline Width decodeFormWidth(uint8_t value)
59{
60 return static_cast<Width>((value >> formWidthShift) & formWidthMask);
61}
62
63} } } // namespace JSC::B3::Air

65#endif // ENABLE(B3_JIT)
