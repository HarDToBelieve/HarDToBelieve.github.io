41namespace Air {
42
43class StackSlot {
44 WTF_MAKE_NONCOPYABLE(StackSlot);
45 WTF_MAKE_FAST_ALLOCATED;
46public:
47 unsigned byteSize() const { return m_byteSize; }
48 StackSlotKind kind() const { return m_kind; }
49 bool isLocked() const { return m_kind == StackSlotKind::Locked; }
50 bool isSpill() const { return m_kind == StackSlotKind::Spill; }
51 unsigned index() const { return m_index; }
52
53 void ensureSize(unsigned requestedSize)
54 {
55 ASSERT(!m_offsetFromFP);
56 RELEASE_ASSERT(requestedSize <= std::numeric_limits<uint16_t>::max());
57 m_byteSize = std::max(m_byteSize, static_cast<uint16_t>(requestedSize));
58 }
59
60 unsigned alignment() const
61 {
62 if (byteSize() <= 1)
63 return 1;
64 if (byteSize() <= 2)
65 return 2;
66 if (byteSize() <= 4)
67 return 4;
68 return 8;
69 }
70
71 // Zero means that it's not yet assigned.
72 intptr_t offsetFromFP() const { return m_offsetFromFP; }
73
74 // This should usually just be called from phases that do stack allocation. But you can
75 // totally force a stack slot to land at some offset.
76 void setOffsetFromFP(intptr_t);
77
78 // This computes a hash for comparing this to JSAir's StackSlot.
79 unsigned jsHash() const;
80
81 void dump(PrintStream&) const;
82 void deepDump(PrintStream&) const;
83
84private:
85 friend class Code;
86 friend class SparseCollection<StackSlot>;
87
88 StackSlot(unsigned byteSize, StackSlotKind, intptr_t offsetFromFP = 0);
89
90 uint16_t m_byteSize { 0 };
91 StackSlotKind m_kind { StackSlotKind::Locked };
92 unsigned m_index { UINT_MAX };
93 intptr_t m_offsetFromFP { 0 };
94};
95
96class DeepStackSlotDump {
97public:
98 DeepStackSlotDump(const StackSlot* slot)
99 : m_slot(slot)
100 {
101 }
102
103 void dump(PrintStream& out) const
104 {
105 if (m_slot)
106 m_slot->deepDump(out);
107 else
108 out.print("<null>");
109 }
110
111private:
112 const StackSlot* m_slot;
113};
114
115inline DeepStackSlotDump deepDump(const StackSlot* slot)
116{
117 return DeepStackSlotDump(slot);
118}
119
120} } } // namespace JSC::B3::Air
