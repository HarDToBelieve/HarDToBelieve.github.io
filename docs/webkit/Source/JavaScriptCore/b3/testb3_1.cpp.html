33bool shouldRun(const char* filter, const char* testName)
34{
35 // FIXME: These tests fail <https://bugs.webkit.org/show_bug.cgi?id=199330>.
36 if (!filter && isARM64()) {
37 for (auto& failingTest : {
38 "testNegFloatWithUselessDoubleConversion",
39 "testPinRegisters",
40 }) {
41 if (WTF::findIgnoringASCIICaseWithoutLength(testName, failingTest) != WTF::notFound) {
42 dataLogLn("*** Warning: Skipping known-bad test: ", testName);
43 return false;
44 }
45 }
46 }
47 return !filter || WTF::findIgnoringASCIICaseWithoutLength(testName, filter) != WTF::notFound;
48}

50template<typename T>
51void testRotR(T valueInt, int32_t shift)
52{
53 Procedure proc;
54 BasicBlock* root = proc.addBlock();
55
56 Value* value = root->appendNew<ArgumentRegValue>(proc, Origin(), GPRInfo::argumentGPR0);
57 if (sizeof(T) == 4)
58 value = root->appendNew<Value>(proc, Trunc, Origin(), value);
59
60 Value* ammount = root->appendNew<Value>(proc, Trunc, Origin(),
61 root->appendNew<ArgumentRegValue>(proc, Origin(), GPRInfo::argumentGPR1));
62 root->appendNewControlValue(proc, Return, Origin(),
63 root->appendNew<Value>(proc, RotR, Origin(), value, ammount));
64
65 CHECK_EQ(compileAndRun<T>(proc, valueInt, shift), rotateRight(valueInt, shift));
66}
