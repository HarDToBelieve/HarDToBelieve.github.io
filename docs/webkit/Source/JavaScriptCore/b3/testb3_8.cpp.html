549template<typename T>
550void testAtomicXchg(B3::Opcode opcode)
551{
552 constexpr Type type = NativeTraits<T>::type;
553 constexpr Width width = NativeTraits<T>::width;
554
555 auto doTheMath = [&] (T& memory, T operand) -> T {
556 T oldValue = memory;
557 switch (opcode) {
558 case AtomicXchgAdd:
559 memory += operand;
560 break;
561 case AtomicXchgAnd:
562 memory &= operand;
563 break;
564 case AtomicXchgOr:
565 memory |= operand;
566 break;
567 case AtomicXchgSub:
568 memory -= operand;
569 break;
570 case AtomicXchgXor:
571 memory ^= operand;
572 break;
573 case AtomicXchg:
574 memory = operand;
575 break;
576 default:
577 RELEASE_ASSERT_NOT_REACHED();
578 }
579 return oldValue;
580 };
581
582 auto oldValue = [&] (T memory, T operand) -> T {
583 return doTheMath(memory, operand);
584 };
585
586 auto newValue = [&] (T memory, T operand) -> T {
587 doTheMath(memory, operand);
588 return memory;
589 };
590
591 auto checkMyDisassembly = [&] (Compilation& compilation, bool fenced) {
592 if (isX86()) {
593 // AtomicXchg can be lowered to "xchg" without "lock", and this is OK since "lock" signal is asserted for "xchg" by default.
594 if (AtomicXchg != opcode)
595 checkUsesInstruction(compilation, "lock");
596 } else {
597 if (fenced) {
598 checkUsesInstruction(compilation, "ldax");
599 checkUsesInstruction(compilation, "stlx");
600 } else {
601 checkUsesInstruction(compilation, "ldx");
602 checkUsesInstruction(compilation, "stx");
603 }
604 }
605 };
606
607 {
608 Procedure proc;
609 BasicBlock* root = proc.addBlock();
610 root->appendNew<Value>(
611 proc, Return, Origin(),
612 root->appendNew<AtomicValue>(
613 proc, opcode, Origin(), width,
614 root->appendIntConstant(proc, Origin(), type, 1),
615 root->appendNew<ArgumentRegValue>(proc, Origin(), GPRInfo::argumentGPR0)));
616
617 auto code = compileProc(proc);
618 T value[2];
619 value[0] = 5;
620 value[1] = 100;
621 CHECK_EQ(invoke<T>(*code, value), oldValue(5, 1));
622 CHECK_EQ(value[0], newValue(5, 1));
623 CHECK_EQ(value[1], 100);
624 checkMyDisassembly(*code, true);
625 }
626
627 {
628 Procedure proc;
629 BasicBlock* root = proc.addBlock();
630 root->appendNew<Value>(
631 proc, Return, Origin(),
632 root->appendNew<AtomicValue>(
633 proc, opcode, Origin(), width,
634 root->appendIntConstant(proc, Origin(), type, 42),
635 root->appendNew<ArgumentRegValue>(proc, Origin(), GPRInfo::argumentGPR0)));
636
637 auto code = compileProc(proc);
638 T value[2];
639 value[0] = 5;
640 value[1] = 100;
641 CHECK_EQ(invoke<T>(*code, value), oldValue(5, 42));
642 CHECK_EQ(value[0], newValue(5, 42));
643 CHECK_EQ(value[1], 100);
644 checkMyDisassembly(*code, true);
645 }
646
647 {
648 Procedure proc;
649 BasicBlock* root = proc.addBlock();
650 root->appendNew<AtomicValue>(
651 proc, opcode, Origin(), width,
652 root->appendIntConstant(proc, Origin(), type, 42),
653 root->appendNew<ArgumentRegValue>(proc, Origin(), GPRInfo::argumentGPR0));
654 root->appendNew<Value>(proc, Return, Origin());
655
656 auto code = compileProc(proc);
657 T value[2];
658 value[0] = 5;
659 value[1] = 100;
660 invoke<T>(*code, value);

787 }

873void testFastForwardCopy32()
874{
875#if CPU(X86_64)
876 for (const bool aligned : { true, false }) {
877 for (const bool overlap : { false, true }) {
878 for (size_t arrsize : { 1, 4, 5, 6, 8, 10, 12, 16, 20, 40, 100, 1000}) {
879 size_t overlapAmount = 5;
880
881 UniqueArray<uint32_t> array1, array2;
882 uint32_t* arr1, *arr2;
883
884 if (overlap) {
885 array1 = makeUniqueArray<uint32_t>(arrsize * 2);
886 arr1 = &array1[0];
887 arr2 = arr1 + (arrsize - overlapAmount);
888 } else {
889 array1 = makeUniqueArray<uint32_t>(arrsize);
890 array2 = makeUniqueArray<uint32_t>(arrsize);
891 arr1 = &array1[0];
892 arr2 = &array2[0];
893 }
894
895 if (!aligned && arrsize < 3)
896 continue;
897 if (overlap && arrsize <= overlapAmount + 3)
898 continue;
899
900 if (!aligned) {
901 ++arr1;
902 ++arr2;
903 arrsize -= 1;
904 overlapAmount -= 1;
905 }
906
907 for (size_t i = 0; i < arrsize; ++i)
908 arr1[i] = i;
909
910 fastForwardCopy32(arr2, arr1, arrsize);
911
912 if (overlap) {
913 for (size_t i = 0; i < arrsize - overlapAmount; ++i)
914 CHECK(arr2[i] == i);
915 for (size_t i = arrsize - overlapAmount; i < arrsize; ++i)
916 CHECK(arr2[i] == i - (arrsize - overlapAmount));
917 } else {
918 for (size_t i = 0; i < arrsize; ++i)
919 CHECK(arr2[i] == i);
920 }
921
922 if (!aligned) {
923 --arr1;
924 --arr2;
925 }
926 }
927 }
928 }
929#endif
930}

932void testByteCopyLoop()
933{
934 Procedure proc;
935 BasicBlock* root = proc.addBlock();
936 BasicBlock* head = proc.addBlock();
937 BasicBlock* update = proc.addBlock();
938 BasicBlock* continuation = proc.addBlock();
939
940 auto* arraySrc = root->appendNew<ArgumentRegValue>(proc, Origin(), GPRInfo::argumentGPR0);
941 auto* arrayDst = root->appendNew<ArgumentRegValue>(proc, Origin(), GPRInfo::argumentGPR1);
942 auto* arraySize = root->appendNew<Value>(proc, Trunc, Origin(), root->appendNew<ArgumentRegValue>(proc, Origin(), GPRInfo::argumentGPR2));
943 auto* one = root->appendNew<Const32Value>(proc, Origin(), 1);
944 auto* two = root->appendNew<Const32Value>(proc, Origin(), 2);
945 UpsilonValue* startingIndex = root->appendNew<UpsilonValue>(proc, Origin(), root->appendNew<Const32Value>(proc, Origin(), 0));
946 root->appendNew<Value>(proc, Jump, Origin());
947 root->setSuccessors(FrequentedBlock(head));
948
949 auto* index = head->appendNew<Value>(proc, Phi, Int32, Origin());
950 startingIndex->setPhi(index);
951 auto* loadIndex = head->appendNew<Value>(proc, Add, Origin(), arraySrc,
952 head->appendNew<Value>(proc, ZExt32, Origin(), head->appendNew<Value>(proc, Shl, Origin(), index, two)));
953 auto* storeIndex = head->appendNew<Value>(proc, Add, Origin(), arrayDst,
954 head->appendNew<Value>(proc, ZExt32, Origin(), head->appendNew<Value>(proc, Shl, Origin(), index, two)));
955 head->appendNew<MemoryValue>(proc, Store, Origin(), head->appendNew<MemoryValue>(proc, Load, Int32, Origin(), loadIndex), storeIndex);
956 auto* newIndex = head->appendNew<Value>(proc, Add, Origin(), index, one);
957 auto* cmpValue = head->appendNew<Value>(proc, GreaterThan, Origin(), newIndex, arraySize);
958 head->appendNew<Value>(proc, Branch, Origin(), cmpValue);
959 head->setSuccessors(FrequentedBlock(continuation), FrequentedBlock(update));
960
961 UpsilonValue* updateIndex = update->appendNew<UpsilonValue>(proc, Origin(), newIndex);
962 updateIndex->setPhi(index);
963 update->appendNew<Value>(proc, Jump, Origin());
964 update->setSuccessors(FrequentedBlock(head));
965
966 continuation->appendNewControlValue(proc, Return, Origin());
967
968 int* arr1 = new int[3];
969 int* arr2 = new int[3];
970
971 arr1[0] = 0;
972 arr1[1] = 0;
973 arr1[2] = 0;
974 arr2[0] = 1;
975 arr2[1] = 2;
976 arr2[2] = 3;
977
978 compileAndRun<void>(proc, arr2, arr1, 3);
979
980 CHECK_EQ(arr1[0], 1);
981 CHECK_EQ(arr1[1], 2);
982 CHECK_EQ(arr1[2], 3);
983
984 delete[] arr1;
985 delete [] arr2;
986}
