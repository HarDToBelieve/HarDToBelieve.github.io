245inline PrivateSymbolImpl* BuiltinNames::lookUpPrivateName(const Identifier& ident) const
246{
247 return lookUpPrivateName(ident.impl());
248}

250inline SymbolImpl* BuiltinNames::lookUpWellKnownSymbol(const Identifier& ident) const
251{

254
255inline void BuiltinNames::checkPublicToPrivateMapConsistency(UniquedStringImpl* privateName)
256{
257#if ASSERT_ENABLED
258 for (const auto& key : m_privateNameSet)
259 ASSERT(String(privateName) != key);
260 ASSERT(privateName->isSymbol());
261 ASSERT(static_cast<SymbolImpl*>(privateName)->isPrivate());
262#else
263 UNUSED_PARAM(privateName);
264#endif
265}
266
267inline void BuiltinNames::appendExternalName(const Identifier& publicName, const Identifier& privateName)
268{
269 ASSERT_UNUSED(publicName, String(publicName.impl()) == String(privateName.impl()));
270 checkPublicToPrivateMapConsistency(privateName.impl());
271 m_privateNameSet.add(privateName.impl());
272}
