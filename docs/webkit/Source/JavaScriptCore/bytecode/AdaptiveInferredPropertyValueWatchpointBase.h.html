26#pragma once
27
28#include "ObjectPropertyCondition.h"
29#include "Watchpoint.h"
30#include <wtf/FastMalloc.h>
31#include <wtf/Noncopyable.h>
32
33namespace JSC {
34
35// FIXME: This isn't actually a Watchpoint. We should probably have a name which better reflects that:
36// https://bugs.webkit.org/show_bug.cgi?id=202381
37class AdaptiveInferredPropertyValueWatchpointBase {
38 WTF_MAKE_NONCOPYABLE(AdaptiveInferredPropertyValueWatchpointBase);
39 WTF_MAKE_FAST_ALLOCATED;
40
41public:
42 AdaptiveInferredPropertyValueWatchpointBase(const ObjectPropertyCondition&);
43 AdaptiveInferredPropertyValueWatchpointBase() = default;
44
45 const ObjectPropertyCondition& key() const { return m_key; }
46
47 void initialize(const ObjectPropertyCondition&);
48 void install(VM&);
49
50 virtual ~AdaptiveInferredPropertyValueWatchpointBase() = default;
51
52 class StructureWatchpoint final : public Watchpoint {
53 public:
54 StructureWatchpoint()
55 : Watchpoint(Watchpoint::Type::AdaptiveInferredPropertyValueStructure)
56 { }
57
58 void fireInternal(VM&, const FireDetail&);
59 };
60 // Own destructor may not be called. Keep members trivially destructible.
61 static_assert(sizeof(StructureWatchpoint) == sizeof(Watchpoint), "");
62
63 class PropertyWatchpoint final : public Watchpoint {
64 public:
65 PropertyWatchpoint()
66 : Watchpoint(Watchpoint::Type::AdaptiveInferredPropertyValueProperty)
67 { }
68
69 void fireInternal(VM&, const FireDetail&);
70 };
71 // Own destructor may not be called. Keep members trivially destructible.
72 static_assert(sizeof(PropertyWatchpoint) == sizeof(Watchpoint), "");
73
74protected:
75 virtual bool isValid() const;
76 virtual void handleFire(VM&, const FireDetail&) = 0;
77
78private:
79 void fire(VM&, const FireDetail&);
80
81 ObjectPropertyCondition m_key;
82 StructureWatchpoint m_structureWatchpoint;
83 PropertyWatchpoint m_propertyWatchpoint;
84};
85
86} // namespace JSC
