36// This is a bitfield where each bit represents an type of array access that we have seen.
37// There are 19 indexing types that use the lower bits.
38// There are 9 typed array types taking the bits 16 to 25.
39typedef unsigned ArrayModes;
40
41// The possible IndexingTypes are limited within (0 - 16, 21, 23, 25).
42// This is because CoW types only appear for JSArrays.
43static_assert(CopyOnWriteArrayWithInt32 == 21, "");
44static_assert(CopyOnWriteArrayWithDouble == 23, "");
45static_assert(CopyOnWriteArrayWithContiguous == 25, "");
46const ArrayModes CopyOnWriteArrayWithInt32ArrayMode = 1 << CopyOnWriteArrayWithInt32;
47const ArrayModes CopyOnWriteArrayWithDoubleArrayMode = 1 << CopyOnWriteArrayWithDouble;
48const ArrayModes CopyOnWriteArrayWithContiguousArrayMode = 1 << CopyOnWriteArrayWithContiguous;
49
50const ArrayModes Int8ArrayMode = 1U << 16;
51const ArrayModes Int16ArrayMode = 1U << 17;
52const ArrayModes Int32ArrayMode = 1U << 18;
53const ArrayModes Uint8ArrayMode = 1U << 19;
54const ArrayModes Uint8ClampedArrayMode = 1U << 20; // 21 - 25 are used for CoW arrays.
55const ArrayModes Uint16ArrayMode = 1U << 26;
56const ArrayModes Uint32ArrayMode = 1U << 27;
57const ArrayModes Float32ArrayMode = 1U << 28;
58const ArrayModes Float64ArrayMode = 1U << 29;
59const ArrayModes BigInt64ArrayMode = 1U << 30;
60const ArrayModes BigUint64ArrayMode = 1U << 31;
61
62JS_EXPORT_PRIVATE extern const ArrayModes typedArrayModes[NumberOfTypedArrayTypesExcludingDataView];
63
64constexpr ArrayModes asArrayModesIgnoringTypedArrays(IndexingType indexingMode)
65{
66 return static_cast<unsigned>(1) << static_cast<unsigned>(indexingMode);
67}
68
69#define ALL_TYPED_ARRAY_MODES \
70 (Int8ArrayMode \
71 | Int16ArrayMode \
72 | Int32ArrayMode \
73 | Uint8ArrayMode \
74 | Uint8ClampedArrayMode \
75 | Uint16ArrayMode \
76 | Uint32ArrayMode \
77 | Float32ArrayMode \
78 | Float64ArrayMode \
79 | BigInt64ArrayMode \
80 | BigUint64ArrayMode \
81 )
82
83#define ALL_NON_ARRAY_ARRAY_MODES \
84 (asArrayModesIgnoringTypedArrays(NonArray) \
85 | asArrayModesIgnoringTypedArrays(NonArrayWithInt32) \
86 | asArrayModesIgnoringTypedArrays(NonArrayWithDouble) \
87 | asArrayModesIgnoringTypedArrays(NonArrayWithContiguous) \
88 | asArrayModesIgnoringTypedArrays(NonArrayWithArrayStorage) \
89 | asArrayModesIgnoringTypedArrays(NonArrayWithSlowPutArrayStorage) \
90 | ALL_TYPED_ARRAY_MODES)
91
92#define ALL_COPY_ON_WRITE_ARRAY_MODES \
93 (CopyOnWriteArrayWithInt32ArrayMode \
94 | CopyOnWriteArrayWithDoubleArrayMode \
95 | CopyOnWriteArrayWithContiguousArrayMode)
96
97#define ALL_WRITABLE_ARRAY_ARRAY_MODES \
98 (asArrayModesIgnoringTypedArrays(ArrayClass) \
99 | asArrayModesIgnoringTypedArrays(ArrayWithUndecided) \
100 | asArrayModesIgnoringTypedArrays(ArrayWithInt32) \
101 | asArrayModesIgnoringTypedArrays(ArrayWithDouble) \
102 | asArrayModesIgnoringTypedArrays(ArrayWithContiguous) \
103 | asArrayModesIgnoringTypedArrays(ArrayWithArrayStorage) \
104 | asArrayModesIgnoringTypedArrays(ArrayWithSlowPutArrayStorage))
105
106#define ALL_ARRAY_ARRAY_MODES \
107 (ALL_WRITABLE_ARRAY_ARRAY_MODES \
108 | ALL_COPY_ON_WRITE_ARRAY_MODES)
109
110#define ALL_ARRAY_MODES (ALL_NON_ARRAY_ARRAY_MODES | ALL_ARRAY_ARRAY_MODES)
111
112inline ArrayModes arrayModesFromStructure(Structure* structure)
113{
114 JSType type = structure->typeInfo().type();
115 if (isTypedArrayType(type))
116 return typedArrayModes[type - FirstTypedArrayType];
117 return asArrayModesIgnoringTypedArrays(structure->indexingMode());
118}

120void dumpArrayModes(PrintStream&, ArrayModes);
121MAKE_PRINT_ADAPTOR(ArrayModesDump, ArrayModes, dumpArrayModes);

123inline bool mergeArrayModes(ArrayModes& left, ArrayModes right)
124{
125 ArrayModes newModes = left | right;
126 if (newModes == left)
127 return false;
128 left = newModes;
129 return true;
130}
131
132inline bool arrayModesAreClearOrTop(ArrayModes modes)
133{
134 return !modes || modes == ALL_ARRAY_MODES;
135}
136
137// Checks if proven is a subset of expected.
138inline bool arrayModesAlreadyChecked(ArrayModes proven, ArrayModes expected)
139{
140 return (expected | proven) == expected;
141}
142
143inline bool arrayModesIncludeIgnoringTypedArrays(ArrayModes arrayModes, IndexingType shape)
144{
145 ArrayModes modes = asArrayModesIgnoringTypedArrays(NonArray | shape) | asArrayModesIgnoringTypedArrays(ArrayClass | shape);
146 if (hasInt32(shape) || hasDouble(shape) || hasContiguous(shape))
147 modes |= asArrayModesIgnoringTypedArrays(ArrayClass | shape | CopyOnWrite);
148 return !!(arrayModes & modes);
149}

151inline bool shouldUseSlowPutArrayStorage(ArrayModes arrayModes)
152{
153 return arrayModesIncludeIgnoringTypedArrays(arrayModes, SlowPutArrayStorageShape);
154}
155
156inline bool shouldUseFastArrayStorage(ArrayModes arrayModes)
157{
158 return arrayModesIncludeIgnoringTypedArrays(arrayModes, ArrayStorageShape);
159}
160
161inline bool shouldUseContiguous(ArrayModes arrayModes)
162{
163 return arrayModesIncludeIgnoringTypedArrays(arrayModes, ContiguousShape);
164}
165
166inline bool shouldUseDouble(ArrayModes arrayModes)
167{
168 return arrayModesIncludeIgnoringTypedArrays(arrayModes, DoubleShape);
169}
170
171inline bool shouldUseInt32(ArrayModes arrayModes)
172{
173 return arrayModesIncludeIgnoringTypedArrays(arrayModes, Int32Shape);
174}

176inline bool hasSeenArray(ArrayModes arrayModes)
177{
178 return arrayModes & ALL_ARRAY_ARRAY_MODES;
179}
180
181inline bool hasSeenNonArray(ArrayModes arrayModes)
182{
183 return arrayModes & ALL_NON_ARRAY_ARRAY_MODES;
184}
185
186inline bool hasSeenWritableArray(ArrayModes arrayModes)
187{
188 return arrayModes & ALL_WRITABLE_ARRAY_ARRAY_MODES;
189}
190
191inline bool hasSeenCopyOnWriteArray(ArrayModes arrayModes)
192{
193 return arrayModes & ALL_COPY_ON_WRITE_ARRAY_MODES;
194}

196class ArrayProfile {
197 friend class CodeBlock;
198
199public:
200 explicit ArrayProfile()
201 : m_mayInterceptIndexedAccesses(false)
202 , m_usesOriginalArrayStructures(true)
203 , m_didPerformFirstRunPruning(false)
204 {
205 }
206
207 StructureID* addressOfLastSeenStructureID() { return &m_lastSeenStructureID; }
208 ArrayModes* addressOfArrayModes() { return &m_observedArrayModes; }
209 bool* addressOfMayStoreToHole() { return &m_mayStoreToHole; }
210
211 void setOutOfBounds() { m_outOfBounds = true; }
212 bool* addressOfOutOfBounds() { return &m_outOfBounds; }
213
214 void observeStructureID(StructureID structureID) { m_lastSeenStructureID = structureID; }
215 void observeStructure(Structure* structure) { m_lastSeenStructureID = structure->id(); }
216
217 void computeUpdatedPrediction(const ConcurrentJSLocker&, CodeBlock*);
218 void computeUpdatedPrediction(const ConcurrentJSLocker&, CodeBlock*, Structure* lastSeenStructure);
219
220 void observeArrayMode(ArrayModes mode) { m_observedArrayModes |= mode; }
221 void observeIndexedRead(VM&, JSCell*, unsigned index);
222
223 ArrayModes observedArrayModes(const ConcurrentJSLocker&) const { return m_observedArrayModes; }
224 bool mayInterceptIndexedAccesses(const ConcurrentJSLocker&) const { return m_mayInterceptIndexedAccesses; }
225
226 bool mayStoreToHole(const ConcurrentJSLocker&) const { return m_mayStoreToHole; }
227 bool outOfBounds(const ConcurrentJSLocker&) const { return m_outOfBounds; }
228
229 bool usesOriginalArrayStructures(const ConcurrentJSLocker&) const { return m_usesOriginalArrayStructures; }
230
231 CString briefDescription(const ConcurrentJSLocker&, CodeBlock*);
232 CString briefDescriptionWithoutUpdating(const ConcurrentJSLocker&);
233
234private:
235 friend class LLIntOffsetsExtractor;
236
237 static Structure* polymorphicStructure() { return static_cast<Structure*>(reinterpret_cast<void*>(1)); }
238
239 StructureID m_lastSeenStructureID { 0 };
240 bool m_mayStoreToHole { false }; // This flag may become overloaded to indicate other special cases that were encountered during array access, as it depends on indexing type. Since we currently have basically just one indexing type (two variants of ArrayStorage), this flag for now just means exactly what its name implies.
241 bool m_outOfBounds { false };
242 bool m_mayInterceptIndexedAccesses : 1;
243 bool m_usesOriginalArrayStructures : 1;
244 bool m_didPerformFirstRunPruning : 1;
245 ArrayModes m_observedArrayModes { 0 };
246};
