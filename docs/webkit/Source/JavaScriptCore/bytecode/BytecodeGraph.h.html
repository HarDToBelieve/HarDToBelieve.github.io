39class BytecodeGraph {
40 WTF_MAKE_FAST_ALLOCATED;
41 WTF_MAKE_NONCOPYABLE(BytecodeGraph);
42public:
43 using BasicBlocksVector = BytecodeBasicBlock::BasicBlockVector;
44
45 typedef WTF::IndexedContainerIterator<BytecodeGraph> iterator;
46
47 template <typename CodeBlockType>
48 inline BytecodeGraph(CodeBlockType*, const InstructionStream&);
49
50 WTF::IteratorRange<BasicBlocksVector::reverse_iterator> basicBlocksInReverseOrder()
51 {
52 return WTF::makeIteratorRange(m_basicBlocks.rbegin(), m_basicBlocks.rend());
53 }
54
55 static bool blockContainsBytecodeOffset(const BytecodeBasicBlock& block, InstructionStream::Offset bytecodeOffset)
56 {
57 unsigned leaderOffset = block.leaderOffset();
58 return bytecodeOffset >= leaderOffset && bytecodeOffset < leaderOffset + block.totalLength();
59 }
60
61 BytecodeBasicBlock* findBasicBlockForBytecodeOffset(InstructionStream::Offset bytecodeOffset)
62 {
63 /*
64 for (unsigned i = 0; i < m_basicBlocks.size(); i++) {
65 if (blockContainsBytecodeOffset(m_basicBlocks[i], bytecodeOffset))
66 return &m_basicBlocks[i];
67 }
68 return 0;
69 */
70
71 BytecodeBasicBlock* basicBlock = approximateBinarySearch<BytecodeBasicBlock, unsigned>(m_basicBlocks, m_basicBlocks.size(), bytecodeOffset, [] (BytecodeBasicBlock* basicBlock) { return basicBlock->leaderOffset(); });
72 // We found the block we were looking for.
73 if (blockContainsBytecodeOffset(*basicBlock, bytecodeOffset))
74 return basicBlock;
75
76 // Basic block is to the left of the returned block.
77 if (bytecodeOffset < basicBlock->leaderOffset()) {
78 ASSERT(basicBlock - 1 >= m_basicBlocks.data());
79 ASSERT(blockContainsBytecodeOffset(basicBlock[-1], bytecodeOffset));
80 return &basicBlock[-1];
81 }
82
83 // Basic block is to the right of the returned block.
84 ASSERT(&basicBlock[1] <= &m_basicBlocks.last());
85 ASSERT(blockContainsBytecodeOffset(basicBlock[1], bytecodeOffset));
86 return &basicBlock[1];
87 }
88
89 BytecodeBasicBlock* findBasicBlockWithLeaderOffset(InstructionStream::Offset leaderOffset)
90 {
91 return tryBinarySearch<BytecodeBasicBlock, unsigned>(m_basicBlocks, m_basicBlocks.size(), leaderOffset, [] (BytecodeBasicBlock* basicBlock) { return basicBlock->leaderOffset(); });
92 }
93
94 unsigned size() const { return m_basicBlocks.size(); }
95 BytecodeBasicBlock& at(unsigned index) const { return const_cast<BytecodeGraph*>(this)->m_basicBlocks[index]; }
96 BytecodeBasicBlock& operator[](unsigned index) const { return at(index); }
97
98 iterator begin() { return iterator(*this, 0); }
99 iterator end() { return iterator(*this, size()); }
100 BytecodeBasicBlock& first() { return at(0); }
101 BytecodeBasicBlock& last() { return at(size() - 1); }
102
103
104 template <typename CodeBlockType>
105 void dump(CodeBlockType* codeBlock, const InstructionStream& instructions, std::optional<Vector<Operands<SpeculatedType>>> speculationAtHead, PrintStream& printer = WTF::dataFile())
106 {
107 CodeBlockBytecodeDumper<CodeBlockType>::dumpGraph(codeBlock, instructions, *this, speculationAtHead, printer);
108 }
109
110private:
111 BasicBlocksVector m_basicBlocks;
112};
113
114
115template<typename CodeBlockType>
116BytecodeGraph::BytecodeGraph(CodeBlockType* codeBlock, const InstructionStream& instructions)
117 : m_basicBlocks(BytecodeBasicBlock::compute(codeBlock, instructions))
118{
119 ASSERT(m_basicBlocks.size());
120}
