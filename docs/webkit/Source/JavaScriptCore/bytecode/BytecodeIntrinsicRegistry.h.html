163class BytecodeIntrinsicRegistry {
164 WTF_MAKE_FAST_ALLOCATED;
165 WTF_MAKE_NONCOPYABLE(BytecodeIntrinsicRegistry);
166public:
167 explicit BytecodeIntrinsicRegistry(VM&);
168
169 typedef RegisterID* (BytecodeIntrinsicNode::* EmitterType)(BytecodeGenerator&, RegisterID*);
170
171 enum class Type : uint8_t {
172 Emitter = 0,
173 LinkTimeConstant = 1,
174 };
175
176 class Entry {
177 public:
178 Entry()
179 : m_type(Type::Emitter)
180 {
181 m_emitter = nullptr;
182 }
183
184 Entry(EmitterType emitter)
185 : m_type(Type::Emitter)
186 {
187 m_emitter = emitter;
188 }
189
190 Entry(LinkTimeConstant linkTimeConstant)
191 : m_type(Type::LinkTimeConstant)
192 {
193 m_linkTimeConstant = linkTimeConstant;
194 }
195
196 Type type() const { return m_type; }
197 LinkTimeConstant linkTimeConstant() const { return m_linkTimeConstant; }
198 EmitterType emitter() const { return m_emitter; }
199
200 private:
201 union {
202 EmitterType m_emitter;
203 LinkTimeConstant m_linkTimeConstant;
204 };
205 Type m_type;
206 };
207
208 std::optional<Entry> lookup(const Identifier&) const;
209
210#define JSC_DECLARE_BYTECODE_INTRINSIC_CONSTANT_GENERATORS(name) JSValue name##Value(BytecodeGenerator&);
211 JSC_COMMON_BYTECODE_INTRINSIC_CONSTANTS_EACH_NAME(JSC_DECLARE_BYTECODE_INTRINSIC_CONSTANT_GENERATORS)
212#undef JSC_DECLARE_BYTECODE_INTRINSIC_CONSTANT_GENERATORS
213
214private:
215 VM& m_vm;
216 MemoryCompactLookupOnlyRobinHoodHashMap<RefPtr<UniquedStringImpl>, Entry, IdentifierRepHash> m_bytecodeIntrinsicMap;
217
218#define JSC_DECLARE_BYTECODE_INTRINSIC_CONSTANT_GENERATORS(name) Strong<Unknown> m_##name;

220#undef JSC_DECLARE_BYTECODE_INTRINSIC_CONSTANT_GENERATORS
221};
