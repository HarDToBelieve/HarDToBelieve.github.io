33namespace JSC {
34
35class CheckPrivateBrandStatus;
36
37class CheckPrivateBrandVariant {
38 WTF_MAKE_FAST_ALLOCATED;
39public:
40 CheckPrivateBrandVariant(CacheableIdentifier, const StructureSet& = StructureSet());
41
42 ~CheckPrivateBrandVariant();
43
44 const StructureSet& structureSet() const { return m_structureSet; }
45 StructureSet& structureSet() { return m_structureSet; }
46
47 bool attemptToMerge(const CheckPrivateBrandVariant& other);
48
49 DECLARE_VISIT_AGGREGATE;
50 template<typename Visitor> void markIfCheap(Visitor&);
51 bool finalize(VM&);
52
53 void dump(PrintStream&) const;
54 void dumpInContext(PrintStream&, DumpContext*) const;
55
56 CacheableIdentifier identifier() const { return m_identifier; }
57
58 bool overlaps(const CheckPrivateBrandVariant& other)
59 {
60 if (!!m_identifier != !!other.m_identifier)
61 return true;
62 if (m_identifier) {
63 if (m_identifier != other.m_identifier)
64 return false;
65 }
66 return structureSet().overlaps(other.structureSet());
67 }
68
69private:
70 friend class CheckPrivateBrandStatus;
71
72 StructureSet m_structureSet;
73 CacheableIdentifier m_identifier;
74};
