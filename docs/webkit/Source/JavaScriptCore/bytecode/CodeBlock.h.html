177 template <typename Functor> void forEachRelatedCodeBlock(Functor&& functor)
178 {
179 Functor f(std::forward<Functor>(functor));
180 Vector<CodeBlock*, 4> codeBlocks;
181 codeBlocks.append(this);
182
183 while (!codeBlocks.isEmpty()) {
184 CodeBlock* currentCodeBlock = codeBlocks.takeLast();
185 f(currentCodeBlock);
186
187 if (CodeBlock* alternative = currentCodeBlock->alternative())
188 codeBlocks.append(alternative);
189 if (CodeBlock* osrEntryBlock = currentCodeBlock->specialOSREntryBlockOrNull())
190 codeBlocks.append(osrEntryBlock);
191 }
192 }

376 inline unsigned bytecodeOffset(const Instruction* returnAddress)
377 {
378 const auto* instructionsBegin = instructions().at(0).ptr();
379 const auto* instructionsEnd = reinterpret_cast<const Instruction*>(reinterpret_cast<uintptr_t>(instructionsBegin) + instructions().size());
380 RELEASE_ASSERT(returnAddress >= instructionsBegin && returnAddress < instructionsEnd);
381 return returnAddress - instructionsBegin;
382 }

631 // and appending to others. At this time it is not safe to shrink certain vectors

662 // Functions for controlling when tiered compilation kicks in. This
663 // controls both when the optimizing compiler is invoked and when OSR
664 // entry happens. Two triggers exist: the loop trigger and the return
665 // trigger. In either case, when an addition to m_jitExecuteCounter
666 // causes it to become non-negative, the optimizing compiler is
667 // invoked. This includes a fast check to see if this CodeBlock has
668 // already been optimized (i.e. replacement() returns a CodeBlock
669 // that was optimized with a higher tier JIT than this one). In the
670 // case of the loop trigger, if the optimized compilation succeeds

672 // redirect program flow into the optimized code.

1057template <typename ExecutableType>
1058Exception* ScriptExecutable::prepareForExecution(VM& vm, JSFunction* function, JSScope* scope, CodeSpecializationKind kind, CodeBlock*& resultCodeBlock)
1059{
1060 if (hasJITCodeFor(kind)) {
1061 if constexpr (std::is_same<ExecutableType, EvalExecutable>::value) {
1062 resultCodeBlock = jsCast<CodeBlock*>(jsCast<ExecutableType*>(this)->codeBlock());
1063 return nullptr;
1064 }
1065 if constexpr (std::is_same<ExecutableType, ProgramExecutable>::value) {
1066 resultCodeBlock = jsCast<CodeBlock*>(jsCast<ExecutableType*>(this)->codeBlock());
1067 return nullptr;
1068 }
1069 if constexpr (std::is_same<ExecutableType, ModuleProgramExecutable>::value) {
1070 resultCodeBlock = jsCast<CodeBlock*>(jsCast<ExecutableType*>(this)->codeBlock());
1071 return nullptr;
1072 }
1073 if constexpr (std::is_same<ExecutableType, FunctionExecutable>::value) {
1074 resultCodeBlock = jsCast<CodeBlock*>(jsCast<ExecutableType*>(this)->codeBlockFor(kind));
1075 return nullptr;
1076 }
1077 RELEASE_ASSERT_NOT_REACHED();
1078 return nullptr;
1079 }
1080 return prepareForExecutionImpl(vm, function, scope, kind, resultCodeBlock);
1081}
