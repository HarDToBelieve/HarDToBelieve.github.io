37namespace JSC {
38
39class AccessCase;
40class CodeBlock;
41class StructureStubInfo;
42
43class DeleteByStatus final {
44 WTF_MAKE_FAST_ALLOCATED;
45public:
46 enum State : uint8_t {
47 // It's uncached so we have no information.
48 NoInformation,
49 // It's cached for a simple access.
50 Simple,
51 // It will likely take the slow path.
52 LikelyTakesSlowPath,
53 // It has been seen to take the slow path.
54 ObservedTakesSlowPath,
55 };
56
57 DeleteByStatus()
58 : m_state(NoInformation)
59 {
60 }
61
62 explicit DeleteByStatus(State state)
63 : m_state(state)
64 {
65 ASSERT(state != Simple);
66 }
67
68 static DeleteByStatus computeFor(CodeBlock* baselineBlock, ICStatusMap& baselineMap, ICStatusContextStack& dfgContextStack, CodeOrigin);
69
70 State state() const { return m_state; }
71
72 bool isSet() const { return m_state != NoInformation; }
73 bool operator!() const { return !isSet(); }
74 bool observedSlowPath() const { return m_state == ObservedTakesSlowPath; }
75 bool isSimple() const { return m_state == Simple; }
76 const Vector<DeleteByVariant, 1>& variants() { return m_variants; }
77 CacheableIdentifier singleIdentifier() const;
78
79 DeleteByStatus slowVersion() const;
80
81 // Attempts to reduce the set of variants to fit the given structure set. This may be approximate.
82 void filter(const StructureSet&);
83
84 DECLARE_VISIT_AGGREGATE;
85 template<typename Visitor> void markIfCheap(Visitor&);
86 bool finalize(VM&);
87
88 bool appendVariant(const DeleteByVariant&);
89 void shrinkToFit();
90
91 void dump(PrintStream&) const;
92
93private:
94 explicit DeleteByStatus(StubInfoSummary, StructureStubInfo&);
95 void merge(const DeleteByStatus&);
96
97 static DeleteByStatus computeForBaseline(CodeBlock*, ICStatusMap&, BytecodeIndex, ExitFlag);
98#if ENABLE(JIT)
99 static DeleteByStatus computeForStubInfoWithoutExitSiteFeedback(
100 const ConcurrentJSLocker&, CodeBlock* profiledBlock, StructureStubInfo*);
101#endif
102
103 Vector<DeleteByVariant, 1> m_variants;
104 State m_state;
105};
106
107} // namespace JSC
