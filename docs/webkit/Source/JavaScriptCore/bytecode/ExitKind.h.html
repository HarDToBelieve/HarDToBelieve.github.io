28namespace JSC {
29
30enum ExitKind : uint8_t {
31 ExitKindUnset,
32 BadType, // We exited because a type prediction was wrong.
33 BadConstantValue, // We exited because we made an incorrect assumption about what value we would see. Usually used for function checks.
34 BadIdent, // We exited because we made an incorrect assumption about what identifier we would see. Usually used for cached Id check in get_by_val.
35 BadExecutable, // We exited because we made an incorrect assumption about what executable we would see.
36 BadCache, // We exited because an inline cache was wrong.
37 BadConstantCache, // We exited because a cache on a weak constant (usually a prototype) was wrong.
38 BadIndexingType, // We exited because an indexing type was wrong.
39 BadTypeInfoFlags, // We exited because we made an incorrect assumption about what TypeInfo flags we would see.
40 Overflow, // We exited because of overflow.
41 NegativeZero, // We exited because we encountered negative zero.
42 NegativeIndex, // We exited because we encountered a negative index in a place we didn't want to see it.
43 Int52Overflow, // We exited because of an Int52 overflow.
44 StoreToHole, // We had a store to a hole.
45 LoadFromHole, // We had a load from a hole.
46 OutOfBounds, // We had an out-of-bounds access to an array.
47 InadequateCoverage, // We exited because we ended up in code that didn't have profiling coverage.
48 ArgumentsEscaped, // We exited because arguments escaped but we didn't expect them to.
49 ExoticObjectMode, // We exited because some exotic object that we were accessing was in an exotic mode (like Arguments with slow arguments).
50 VarargsOverflow, // We exited because a varargs call passed more arguments than we expected.
51 TDZFailure, // We exited because we were in the TDZ and accessed the variable.
52 HoistingFailed, // Something that was hoisted exited. So, assume that hoisting is a bad idea.
53 Uncountable, // We exited for none of the above reasons, and we should not count it. Most uses of this should be viewed as a FIXME.
54 UncountableInvalidation, // We exited because the code block was invalidated; this means that we've already counted the reasons why the code block was invalidated.
55 WatchdogTimerFired, // We exited because we need to service the watchdog timer.
56 DebuggerEvent, // We exited because we need to service the debugger.
57 ExceptionCheck, // We exited because a direct exception check showed that we threw an exception from a C call.
58 GenericUnwind, // We exited because we arrived at this OSR exit from genericUnwind.
59 BigInt32Overflow, // We exited because of an BigInt32 overflow.
60};
61
62const char* exitKindToString(ExitKind);
63bool exitKindMayJettison(ExitKind);
64
65} // namespace JSC
