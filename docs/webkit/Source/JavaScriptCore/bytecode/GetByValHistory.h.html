26#pragma once
27
28#include "TinyBloomFilter.h"
29#include <wtf/text/UniquedStringImpl.h>
30
31namespace JSC {
32
33struct GetByValHistory {
34 void observeNonUID()
35 {
36 uintptr_t count = Options::getByValICMaxNumberOfIdentifiers() + 1;
37 update(count, filter());
38 }
39
40 void observe(const UniquedStringImpl* impl)
41 {
42 if (!impl) {
43 observeNonUID();
44 return;
45 }
46
47 uintptr_t count = this->count();
48 uintptr_t filter = this->filter();
49
50 TinyBloomFilter bloomFilter(filter);
51 uintptr_t implBits = bitwise_cast<uintptr_t>(impl);
52 ASSERT(((static_cast<uint64_t>(implBits) << 8) >> 8) == static_cast<uint64_t>(implBits));
53 if (bloomFilter.ruleOut(implBits)) {
54 bloomFilter.add(implBits);
55 ++count;
56 update(count, bloomFilter.bits());
57 }
58 }
59
60 uintptr_t count() const { return static_cast<uintptr_t>(m_payload >> 56); }
61
62private:
63 uintptr_t filter() const { return static_cast<uintptr_t>((m_payload << 8) >> 8); }
64
65 void update(uint64_t count, uint64_t filter)
66 {
67 ASSERT(((filter << 8) >> 8) == filter);
68 m_payload = (count << 56) | filter;
69 }
70
71 uint64_t m_payload { 0 };
72};
73
74} // namespace JSC
