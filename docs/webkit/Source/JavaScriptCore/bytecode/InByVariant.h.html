39class InByStatus;
40struct DumpContext;
41
42class InByVariant {
43 WTF_MAKE_FAST_ALLOCATED;
44public:
45 InByVariant(CacheableIdentifier, const StructureSet& = StructureSet(), PropertyOffset = invalidOffset, const ObjectPropertyConditionSet& = ObjectPropertyConditionSet());
46
47 bool isSet() const { return !!m_structureSet.size(); }
48 explicit operator bool() const { return isSet(); }
49 const StructureSet& structureSet() const { return m_structureSet; }
50 StructureSet& structureSet() { return m_structureSet; }
51
52 // A non-empty condition set means that this is a prototype in-hit/in-miss.
53 const ObjectPropertyConditionSet& conditionSet() const { return m_conditionSet; }
54
55 PropertyOffset offset() const { return m_offset; }
56
57 bool isHit() const { return offset() != invalidOffset; }
58
59 bool attemptToMerge(const InByVariant& other);
60
61 template<typename Visitor> void markIfCheap(Visitor&);
62 bool finalize(VM&);
63
64 void dump(PrintStream&) const;
65 void dumpInContext(PrintStream&, DumpContext*) const;
66
67 CacheableIdentifier identifier() const { return m_identifier; }
68
69 bool overlaps(const InByVariant& other)
70 {
71 if (!!m_identifier != !!other.m_identifier)
72 return true;
73 if (m_identifier) {
74 if (m_identifier != other.m_identifier)
75 return false;
76 }
77 return structureSet().overlaps(other.structureSet());
78 }
79
80private:
81 friend class InByStatus;
82
83 StructureSet m_structureSet;
84 ObjectPropertyConditionSet m_conditionSet;
85 PropertyOffset m_offset;
86 CacheableIdentifier m_identifier;
87};
