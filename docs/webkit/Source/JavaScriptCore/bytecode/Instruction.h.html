33struct JSOpcodeTraits {
34 using OpcodeID = ::JSC::OpcodeID;
35 static constexpr OpcodeID numberOfBytecodesWithCheckpoints = static_cast<OpcodeID>(NUMBER_OF_BYTECODE_WITH_CHECKPOINTS);
36 static constexpr OpcodeID numberOfBytecodesWithMetadata = static_cast<OpcodeID>(NUMBER_OF_BYTECODE_WITH_METADATA);
37 static_assert(numberOfBytecodesWithCheckpoints <= numberOfBytecodesWithMetadata);
38 static constexpr OpcodeID wide16 = op_wide16;
39 static constexpr OpcodeID wide32 = op_wide32;
40 static constexpr const unsigned* opcodeLengths = ::JSC::opcodeLengths;
41 static constexpr const char* const* opcodeNames = ::JSC::opcodeNames;
42 static constexpr auto checkpointCountTable = bytecodeCheckpointCountTable;
43};
44
45struct WasmOpcodeTraits {
46 using OpcodeID = WasmOpcodeID;
47 static constexpr OpcodeID numberOfBytecodesWithCheckpoints = static_cast<OpcodeID>(NUMBER_OF_WASM_WITH_CHECKPOINTS);
48 static constexpr OpcodeID numberOfBytecodesWithMetadata = static_cast<OpcodeID>(NUMBER_OF_WASM_WITH_METADATA);
49 static_assert(numberOfBytecodesWithCheckpoints <= numberOfBytecodesWithMetadata);
50 static constexpr OpcodeID wide16 = wasm_wide16;
51 static constexpr OpcodeID wide32 = wasm_wide32;
52 static constexpr const unsigned* opcodeLengths = wasmOpcodeLengths;
53 static constexpr const char* const* opcodeNames = wasmOpcodeNames;
54 static constexpr auto checkpointCountTable = wasmCheckpointCountTable;
55};

57
58template<typename Opcode>
59struct BaseInstruction {
60
61 struct Metadata { };
62
63protected:
64 BaseInstruction()
65 { }
66
67private:
68 template<OpcodeSize Width>
69 class Impl {
70 public:
71 template<typename Traits = JSOpcodeTraits>
72 typename Traits::OpcodeID opcodeID() const { return static_cast<typename Traits::OpcodeID>(m_opcode); }
73
74 private:
75 typename TypeBySize<OpcodeSize::Narrow>::unsignedType m_opcode;
76 };
77
78public:
79 template<typename Traits = JSOpcodeTraits>
80 typename Traits::OpcodeID opcodeID() const
81 {
82 if (isWide32<Traits>())
83 return wide32<Traits>()->template opcodeID<Traits>();
84 if (isWide16<Traits>())
85 return wide16<Traits>()->template opcodeID<Traits>();
86 return narrow()->template opcodeID<Traits>();
87 }
88
89 template<typename Traits = JSOpcodeTraits>
90 const char* name() const
91 {
92 return Traits::opcodeNames[opcodeID()];
93 }
94
95 template<typename Traits = JSOpcodeTraits>
96 bool isWide16() const
97 {
98 return narrow()->template opcodeID<Traits>() == Traits::wide16;
99 }
100
101 template<typename Traits = JSOpcodeTraits>
102 bool isWide32() const
103 {
104 return narrow()->template opcodeID<Traits>() == Traits::wide32;
105 }
106
107 template<typename Traits = JSOpcodeTraits>
108 OpcodeSize width() const
109 {
110 if (isWide32<Traits>())
111 return OpcodeSize::Wide32;
112 return isWide16<Traits>() ? OpcodeSize::Wide16 : OpcodeSize::Narrow;
113 }
114
115 template<typename Traits = JSOpcodeTraits>
116 bool hasMetadata() const
117 {
118 return opcodeID<Traits>() < Traits::numberOfBytecodesWithMetadata;
119 }
120
121 template<typename Traits = JSOpcodeTraits>
122 bool hasCheckpoints() const
123 {
124 return opcodeID<Traits>() < Traits::numberOfBytecodesWithCheckpoints;
125 }
126
127 template<typename Traits = JSOpcodeTraits>
128 unsigned numberOfCheckpoints() const
129 {
130 if (!hasCheckpoints<Traits>())
131 return 1;
132 return Traits::checkpointCountTable[opcodeID<Traits>()];
133 }
134
135 template<typename Traits = JSOpcodeTraits>
136 int sizeShiftAmount() const
137 {
138 if (isWide32<Traits>())
139 return 2;
140 if (isWide16<Traits>())
141 return 1;
142 return 0;
143 }
144
145 template<typename Traits = JSOpcodeTraits>
146 size_t size() const
147 {
148 auto sizeShiftAmount = this->sizeShiftAmount<Traits>();
149 size_t prefixSize = sizeShiftAmount ? 1 : 0;
150 size_t operandSize = static_cast<size_t>(1) << sizeShiftAmount;
151 size_t sizeOfBytecode = 1;
152 return sizeOfBytecode + (Traits::opcodeLengths[opcodeID<Traits>()] - 1) * operandSize + prefixSize;
153 }
154
155 template<class T, typename Traits = JSOpcodeTraits>
156 bool is() const
157 {
158 return opcodeID<Traits>() == T::opcodeID;
159 }
160
161 template<class T, typename Traits = JSOpcodeTraits>
162 T as() const
163 {
164 ASSERT((is<T, Traits>()));
165 return T::decode(reinterpret_cast<const uint8_t*>(this));
166 }
167
168 template<class T, OpcodeSize width, typename Traits = JSOpcodeTraits>
169 T asKnownWidth() const
170 {
171 ASSERT((is<T, Traits>()));
172 return T(reinterpret_cast<const typename TypeBySize<width>::unsignedType*>(this + (width == OpcodeSize::Narrow ? 1 : 2)));
173 }
174
175 template<class T, typename Traits = JSOpcodeTraits>
176 T* cast()
177 {
178 ASSERT((is<T, Traits>()));
179 return bitwise_cast<T*>(this);
180 }
181
182 template<class T, typename Traits = JSOpcodeTraits>
183 const T* cast() const
184 {
185 ASSERT((is<T, Traits>()));
186 return reinterpret_cast<const T*>(this);
187 }
188
189 const Impl<OpcodeSize::Narrow>* narrow() const
190 {
191 return reinterpret_cast<const Impl<OpcodeSize::Narrow>*>(this);
192 }
193
194 template<typename Traits = JSOpcodeTraits>
195 const Impl<OpcodeSize::Wide16>* wide16() const
196 {
197
198 ASSERT(isWide16<Traits>());
199 return reinterpret_cast<const Impl<OpcodeSize::Wide16>*>(this + 1);
200 }
201
202 template<typename Traits = JSOpcodeTraits>
203 const Impl<OpcodeSize::Wide32>* wide32() const
204 {
205
206 ASSERT(isWide32<Traits>());
207 return reinterpret_cast<const Impl<OpcodeSize::Wide32>*>(this + 1);
208 }
209};

212};
