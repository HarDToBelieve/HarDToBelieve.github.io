42#if ENABLE(JIT)
43 struct StringJumpTable {
44 FixedVector<CodeLocationLabel<JSSwitchPtrTag>> m_ctiOffsets;
45
46 void ensureCTITable(const UnlinkedStringJumpTable& unlinkedTable)
47 {
48 if (!isEmpty())
49 return;
50 m_ctiOffsets = FixedVector<CodeLocationLabel<JSSwitchPtrTag>>(unlinkedTable.m_offsetTable.size() + 1);
51 }
52
53 inline CodeLocationLabel<JSSwitchPtrTag> ctiForValue(const UnlinkedStringJumpTable& unlinkedTable, StringImpl* value) const
54 {
55 auto loc = unlinkedTable.m_offsetTable.find(value);
56 if (loc == unlinkedTable.m_offsetTable.end())
57 return m_ctiOffsets[unlinkedTable.m_offsetTable.size()];
58 return m_ctiOffsets[loc->value.m_indexInTable];
59 }
60
61 CodeLocationLabel<JSSwitchPtrTag> ctiDefault() const { return m_ctiOffsets.last(); }
62
63 bool isEmpty() const { return m_ctiOffsets.isEmpty(); }
64 };
65
66 struct SimpleJumpTable {
67 FixedVector<CodeLocationLabel<JSSwitchPtrTag>> m_ctiOffsets;
68 CodeLocationLabel<JSSwitchPtrTag> m_ctiDefault;
69
70 void ensureCTITable(const UnlinkedSimpleJumpTable& unlinkedTable)
71 {
72 if (!isEmpty())
73 return;
74 m_ctiOffsets = FixedVector<CodeLocationLabel<JSSwitchPtrTag>>(unlinkedTable.m_branchOffsets.size());
75 }
76
77 inline CodeLocationLabel<JSSwitchPtrTag> ctiForValue(int32_t min, int32_t value) const
78 {
79 if (value >= min && static_cast<uint32_t>(value - min) < m_ctiOffsets.size())
80 return m_ctiOffsets[value - min];
81 return m_ctiDefault;
82 }
83
84 bool isEmpty() const { return m_ctiOffsets.isEmpty(); }
85 };
86#endif
