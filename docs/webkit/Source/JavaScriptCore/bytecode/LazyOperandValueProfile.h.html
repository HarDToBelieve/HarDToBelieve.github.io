40class LazyOperandValueProfileKey {
41public:
42 LazyOperandValueProfileKey()
43 : m_operand(VirtualRegister()) // not a valid operand index in our current scheme
44 {
45 }
46
47 LazyOperandValueProfileKey(WTF::HashTableDeletedValueType)
48 : m_bytecodeIndex(WTF::HashTableDeletedValue)
49 , m_operand(VirtualRegister()) // not a valid operand index in our current scheme
50 {
51 }
52
53 LazyOperandValueProfileKey(BytecodeIndex bytecodeIndex, Operand operand)
54 : m_bytecodeIndex(bytecodeIndex)
55 , m_operand(operand)
56 {
57 ASSERT(m_operand.isValid());
58 }
59
60 bool operator!() const
61 {
62 return !m_operand.isValid();
63 }
64
65 bool operator==(const LazyOperandValueProfileKey& other) const
66 {
67 return m_bytecodeIndex == other.m_bytecodeIndex
68 && m_operand == other.m_operand;
69 }
70
71 unsigned hash() const
72 {
73 return m_bytecodeIndex.hash() + m_operand.value() + static_cast<unsigned>(m_operand.kind());
74 }
75
76 BytecodeIndex bytecodeIndex() const
77 {
78 ASSERT(!!*this);
79 return m_bytecodeIndex;
80 }
81
82 Operand operand() const
83 {
84 ASSERT(!!*this);
85 return m_operand;
86 }
87
88 bool isHashTableDeletedValue() const
89 {
90 return !m_operand.isValid() && m_bytecodeIndex.isHashTableDeletedValue();
91 }
92private:
93 BytecodeIndex m_bytecodeIndex;
94 Operand m_operand;
95};
