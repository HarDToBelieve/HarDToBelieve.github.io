36// An object property condition set is used to represent the set of additional conditions
37// that need to be met for some heap access to be valid. The set can have the following
38// interesting states:
39//
40// Empty: There are no special conditions that need to be met.
41// Invalid: The heap access is never valid.
42// Non-empty: The heap access is valid if all the ObjectPropertyConditions in the set are valid.
43
44class ObjectPropertyConditionSet {
45public:
46 ObjectPropertyConditionSet() { }
47
48 static ObjectPropertyConditionSet invalid()
49 {
50 ObjectPropertyConditionSet result;
51 result.m_data = adoptRef(new Data());
52 return result;
53 }
54
55 static ObjectPropertyConditionSet create(Vector<ObjectPropertyCondition>&& vector)
56 {
57 if (vector.isEmpty())
58 return ObjectPropertyConditionSet();
59
60 ObjectPropertyConditionSet result;
61 result.m_data = adoptRef(new Data());
62 result.m_data->m_vector = FixedVector<ObjectPropertyCondition>(WTFMove(vector));
63 return result;
64 }
65
66 bool isValid() const
67 {
68 return !m_data || !m_data->m_vector.isEmpty();
69 }
70
71 bool isValidAndWatchable() const;
72
73 size_t size() const { return m_data ? m_data->m_vector.size() : 0; }
74 bool isEmpty() const
75 {
76 return !m_data;
77 }
78
79 typedef const ObjectPropertyCondition* iterator;
80
81 iterator begin() const
82 {
83 if (!m_data)
84 return nullptr;
85 return m_data->m_vector.begin();
86 }
87 iterator end() const
88 {
89 if (!m_data)
90 return nullptr;
91 return m_data->m_vector.end();
92 }
93
94 unsigned hash() const
95 {
96 Hasher hasher;
97 for (auto& condition : *this)
98 add(hasher, condition.hash());
99 return hasher.hash();
100 }
101
102 friend bool operator==(const ObjectPropertyConditionSet& lhs, const ObjectPropertyConditionSet& rhs)
103 {
104 if (lhs.size() != rhs.size())
105 return false;
106 auto liter = lhs.begin();
107 auto riter = rhs.begin();
108 for (; liter != lhs.end(); ++liter, ++riter) {
109 if (!(*liter == *riter))
110 return false;
111 }
112 return true;
113 }
114
115 friend bool operator!=(const ObjectPropertyConditionSet& lhs, const ObjectPropertyConditionSet& rhs)
116 {
117 return !(lhs == rhs);
118 }

127 // If this is a condition set for a prototype hit, then this is guaranteed to return the
128 // condition on the prototype itself. This allows you to get the object, offset, and
129 // attributes for the prototype. This will RELEASE_ASSERT that there is exactly one Presence
130 // in the set, and it will return that presence.

155 // Helpers for using this in a union.
156 void* releaseRawPointer()
157 {
158 return static_cast<void*>(m_data.leakRef());
159 }
160 static ObjectPropertyConditionSet adoptRawPointer(void* rawPointer)
161 {
162 ObjectPropertyConditionSet result;
163 result.m_data = adoptRef(static_cast<Data*>(rawPointer));
164 return result;
165 }
166 static ObjectPropertyConditionSet fromRawPointer(void* rawPointer)
167 {
168 ObjectPropertyConditionSet result;
169 result.m_data = static_cast<Data*>(rawPointer);
170 return result;
171 }

175 // Internally, this represents Invalid using a pointer to a Data that has an empty vector.
176
177 // FIXME: This could be made more compact by having it internally use a vector that just has

181 class Data : public ThreadSafeRefCounted<Data> {
182 WTF_MAKE_NONCOPYABLE(Data);
183 WTF_MAKE_FAST_ALLOCATED;
184
185 public:
186 Data() { }
187
188 FixedVector<ObjectPropertyCondition> m_vector;
189 };
190
191private:
192 RefPtr<Data> m_data;
193};
