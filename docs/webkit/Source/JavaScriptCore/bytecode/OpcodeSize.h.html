26#pragma once
27
28namespace JSC {
29
30enum OpcodeSize {
31 Narrow = 1,
32 Wide16 = 2,
33 Wide32 = 4,
34};
35
36template<OpcodeSize>
37struct TypeBySize;
38
39template<>
40struct TypeBySize<OpcodeSize::Narrow> {
41 using signedType = int8_t;
42 using unsignedType = uint8_t;
43};
44
45template<>
46struct TypeBySize<OpcodeSize::Wide16> {
47 using signedType = int16_t;
48 using unsignedType = uint16_t;
49};
50
51template<>
52struct TypeBySize<OpcodeSize::Wide32> {
53 using signedType = int32_t;
54 using unsignedType = uint32_t;
55};
56
57template<OpcodeSize>
58struct PaddingBySize;
59
60template<>
61struct PaddingBySize<OpcodeSize::Narrow> {
62 static constexpr uint8_t value = 0;
63};
64
65template<>
66struct PaddingBySize<OpcodeSize::Wide16> {
67 static constexpr uint8_t value = 1;
68};
69
70template<>
71struct PaddingBySize<OpcodeSize::Wide32> {
72 static constexpr uint8_t value = 1;
73};
74
75} // namespace JSC
