28#include "UnlinkedCodeBlock.h"

30namespace JSC {
31
32class CachedFunctionCodeBlock;
33
34class UnlinkedFunctionCodeBlock final : public UnlinkedCodeBlock {
35public:
36 typedef UnlinkedCodeBlock Base;
37 static constexpr unsigned StructureFlags = Base::StructureFlags | StructureIsImmortal;
38
39 template<typename CellType, SubspaceAccess mode>
40 static IsoSubspace* subspaceFor(VM& vm)
41 {
42 return vm.unlinkedFunctionCodeBlockSpace<mode>();
43 }
44
45 static UnlinkedFunctionCodeBlock* create(VM& vm, CodeType codeType, const ExecutableInfo& info, OptionSet<CodeGenerationMode> codeGenerationMode)
46 {
47 UnlinkedFunctionCodeBlock* instance = new (NotNull, allocateCell<UnlinkedFunctionCodeBlock>(vm.heap)) UnlinkedFunctionCodeBlock(vm, vm.unlinkedFunctionCodeBlockStructure.get(), codeType, info, codeGenerationMode);
48 instance->finishCreation(vm);
49 return instance;
50 }
51
52 static void destroy(JSCell*);
53
54private:
55 friend CachedFunctionCodeBlock;
56
57 UnlinkedFunctionCodeBlock(VM& vm, Structure* structure, CodeType codeType, const ExecutableInfo& info, OptionSet<CodeGenerationMode> codeGenerationMode)
58 : Base(vm, structure, codeType, info, codeGenerationMode)
59 {
60 }
61
62 UnlinkedFunctionCodeBlock(Decoder&, const CachedFunctionCodeBlock&);
63
64public:
65 static Structure* createStructure(VM& vm, JSGlobalObject* globalObject, JSValue proto)
66 {
67 return Structure::create(vm, globalObject, proto, TypeInfo(UnlinkedFunctionCodeBlockType, StructureFlags), info());
68 }
69
70 DECLARE_INFO;
71};
