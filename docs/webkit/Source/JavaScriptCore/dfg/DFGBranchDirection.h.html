30namespace JSC { namespace DFG {
31
32enum BranchDirection : uint8_t {
33 // This is not a branch and so there is no branch direction, or
34 // the branch direction has yet to be set.
35 InvalidBranchDirection,
36
37 // The branch takes the true case.
38 TakeTrue,
39
40 // The branch takes the false case.
41 TakeFalse,
42
43 // For all we know, the branch could go either direction, so we
44 // have to assume the worst.
45 TakeBoth
46};
47
48static inline const char* branchDirectionToString(BranchDirection branchDirection)
49{
50 switch (branchDirection) {
51 case InvalidBranchDirection:
52 return "InvalidBranchDirection";
53 case TakeTrue:
54 return "TakeTrue";
55 case TakeFalse:
56 return "TakeFalse";
57 case TakeBoth:
58 return "TakeBoth";
59 }
60
61 RELEASE_ASSERT_NOT_REACHED();
62 return "InvalidBranchDirection";
63}
64
65static inline bool isKnownDirection(BranchDirection branchDirection)
66{
67 switch (branchDirection) {
68 case TakeTrue:
69 case TakeFalse:
70 return true;
71 default:
72 return false;
73 }
74}
75
76static inline bool branchCondition(BranchDirection branchDirection)
77{
78 if (branchDirection == TakeTrue)
79 return true;
80 ASSERT(branchDirection == TakeFalse);
81 return false;
82}
83
84} } // namespace JSC::DFG

86namespace WTF {
87
88inline void printInternal(PrintStream& out, JSC::DFG::BranchDirection direction)
89{
90 out.print(JSC::DFG::branchDirectionToString(direction));
91}
