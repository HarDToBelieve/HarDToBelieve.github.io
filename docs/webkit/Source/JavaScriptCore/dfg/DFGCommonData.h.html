54// CommonData holds the set of data that both DFG and FTL code blocks need to know
55// about themselves.
56
57struct WeakReferenceTransition {
58 WeakReferenceTransition() { }
59
60 WeakReferenceTransition(VM& vm, JSCell* owner, JSCell* codeOrigin, JSCell* from, JSCell* to)
61 : m_from(vm, owner, from)
62 , m_to(vm, owner, to)
63 {
64 if (!!codeOrigin)
65 m_codeOrigin.set(vm, owner, codeOrigin);
66 }
67
68 WriteBarrier<JSCell> m_codeOrigin;
69 WriteBarrier<JSCell> m_from;
70 WriteBarrier<JSCell> m_to;
71};

73class CommonData {
74 WTF_MAKE_NONCOPYABLE(CommonData);
75public:
76 CommonData()
77 : codeOrigins(CodeOriginPool::create())
78 { }
79 ~CommonData();
80
81 void shrinkToFit();
82
83 bool invalidate(); // Returns true if we did invalidate, or false if the code block was already invalidated.
84 bool hasInstalledVMTrapsBreakpoints() const { return isStillValid && hasVMTrapsBreakpointsInstalled; }
85 void installVMTrapBreakpoints(CodeBlock* owner);
86 bool isVMTrapBreakpoint(void* address);
87
88 CatchEntrypointData* catchOSREntryDataForBytecodeIndex(BytecodeIndex bytecodeIndex)
89 {
90 return tryBinarySearch<CatchEntrypointData, BytecodeIndex>(
91 m_catchEntrypoints, m_catchEntrypoints.size(), bytecodeIndex,
92 [] (const CatchEntrypointData* item) { return item->bytecodeIndex; });
93 }
94
95 void finalizeCatchEntrypoints(Vector<CatchEntrypointData>&&);
96
97 unsigned requiredRegisterCountForExecutionAndExit() const
98 {
99 return std::max(frameRegisterCount, requiredRegisterCountForExit);
100 }
101
102 void validateReferences(const TrackedReferences&);
103
104 static ptrdiff_t frameRegisterCountOffset() { return OBJECT_OFFSETOF(CommonData, frameRegisterCount); }

106 void clearWatchpoints();

108 RefPtr<InlineCallFrameSet> inlineCallFrames;
109 Ref<CodeOriginPool> codeOrigins;

111 FixedVector<Identifier> m_dfgIdentifiers;

114 FixedVector<StructureID> m_weakStructureReferences;
115 FixedVector<CatchEntrypointData> m_catchEntrypoints;
116 FixedVector<CodeBlockJettisoningWatchpoint> m_watchpoints;
117 FixedVector<AdaptiveStructureWatchpoint> m_adaptiveStructureWatchpoints;
