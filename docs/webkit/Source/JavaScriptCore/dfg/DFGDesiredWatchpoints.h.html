45enum class WatchpointRegistrationMode : uint8_t { Collect, Add };
46class WatchpointCollector final {
47 WTF_MAKE_NONCOPYABLE(WatchpointCollector);
48public:
49 WatchpointCollector() = default;
50
51 void materialize()
52 {
53 m_watchpoints = FixedVector<CodeBlockJettisoningWatchpoint>(m_watchpointCount);
54 m_adaptiveStructureWatchpoints = FixedVector<AdaptiveStructureWatchpoint>(m_adaptiveStructureWatchpointCount);
55 m_adaptiveInferredPropertyValueWatchpoints = FixedVector<AdaptiveInferredPropertyValueWatchpoint>(m_adaptiveInferredPropertyValueWatchpointCount);
56 m_mode = WatchpointRegistrationMode::Add;
57 }
58
59 template<typename Func>
60 void addWatchpoint(const Func& function)
61 {
62 if (m_mode == WatchpointRegistrationMode::Add)
63 function(m_watchpoints[m_watchpointIndex++]);
64 else
65 ++m_watchpointCount;
66 }
67
68 template<typename Func>
69 void addAdaptiveStructureWatchpoint(const Func& function)
70 {
71 if (m_mode == WatchpointRegistrationMode::Add)
72 function(m_adaptiveStructureWatchpoints[m_adaptiveStructureWatchpointsIndex++]);
73 else
74 ++m_adaptiveStructureWatchpointCount;
75 }
76
77 template<typename Func>
78 void addAdaptiveInferredPropertyValueWatchpoint(const Func& function)
79 {
80 if (m_mode == WatchpointRegistrationMode::Add)
81 function(m_adaptiveInferredPropertyValueWatchpoints[m_adaptiveInferredPropertyValueWatchpointsIndex++]);
82 else
83 ++m_adaptiveInferredPropertyValueWatchpointCount;
84 }
85
86 void finalize(CodeBlock*, CommonData&);
87
88 WatchpointRegistrationMode mode() const { return m_mode; }
89
90private:
91 unsigned m_watchpointCount { 0 };
92 unsigned m_adaptiveStructureWatchpointCount { 0 };
93 unsigned m_adaptiveInferredPropertyValueWatchpointCount { 0 };
94
95 unsigned m_watchpointIndex { 0 };
96 unsigned m_adaptiveStructureWatchpointsIndex { 0 };
97 unsigned m_adaptiveInferredPropertyValueWatchpointsIndex { 0 };
98
99 FixedVector<CodeBlockJettisoningWatchpoint> m_watchpoints;
100 FixedVector<AdaptiveStructureWatchpoint> m_adaptiveStructureWatchpoints;
101 FixedVector<AdaptiveInferredPropertyValueWatchpoint> m_adaptiveInferredPropertyValueWatchpoints;
102 WatchpointRegistrationMode m_mode { WatchpointRegistrationMode::Collect };
103};

105template<typename T>
106struct SetPointerAdaptor {
107 static void add(CodeBlock* codeBlock, T set, WatchpointCollector& collector)
108 {
109 collector.addWatchpoint([&](CodeBlockJettisoningWatchpoint& watchpoint) {
110 {
111 ConcurrentJSLocker locker(codeBlock->m_lock);
112 watchpoint.initialize(codeBlock);
113 }
114 set->add(&watchpoint);
115 });
116 }
117 static bool hasBeenInvalidated(T set)
118 {
119 return set->hasBeenInvalidated();
120 }
121 static void dumpInContext(PrintStream& out, T set, DumpContext*)
122 {
123 out.print(RawPointer(set));
124 }
125};

176template<typename WatchpointSetType, typename Adaptor = SetPointerAdaptor<WatchpointSetType>>
177class GenericDesiredWatchpoints {
178#if ASSERT_ENABLED
179 typedef HashMap<WatchpointSetType, bool> StateMap;
180#endif
181public:
182 GenericDesiredWatchpoints()
183 : m_reallyAdded(false)
184 {
185 }
186
187 void addLazily(const WatchpointSetType& set)
188 {
189 m_sets.add(set);
190 }
191
192 void reallyAdd(CodeBlock* codeBlock, WatchpointCollector& collector)
193 {
194 if (collector.mode() == WatchpointRegistrationMode::Add)
195 RELEASE_ASSERT(!m_reallyAdded);
196
197 for (auto& set : m_sets)
198 Adaptor::add(codeBlock, set, collector);
199
200 if (collector.mode() == WatchpointRegistrationMode::Add)
201 m_reallyAdded = true;
202 }
203
204 bool areStillValid() const
205 {
206 for (auto& set : m_sets) {
207 if (Adaptor::hasBeenInvalidated(set))
208 return false;
209 }
210
211 return true;
212 }
213
214 bool isWatched(const WatchpointSetType& set) const
215 {
216 return m_sets.contains(set);
217 }
218
219 void dumpInContext(PrintStream& out, DumpContext* context) const
220 {
221 CommaPrinter comma;
222 for (const WatchpointSetType& entry : m_sets) {
223 out.print(comma);
224 Adaptor::dumpInContext(out, entry, context);
225 }
226 }
227
228private:
229 HashSet<WatchpointSetType> m_sets;
230 bool m_reallyAdded;
231};

280 }

283private:

289 GenericDesiredWatchpoints<ObjectPropertyCondition, AdaptiveStructureWatchpointAdaptor> m_adaptiveStructureSets;
290 DesiredGlobalProperties m_globalProperties;
291};
