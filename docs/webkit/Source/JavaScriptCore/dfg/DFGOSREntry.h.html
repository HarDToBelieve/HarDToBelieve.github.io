56struct OSREntryData {
57 BytecodeIndex m_bytecodeIndex;
58 CodeLocationLabel<OSREntryPtrTag> m_machineCode;
59 FixedOperands<AbstractValue> m_expectedValues;
60 // Use bitvectors here because they tend to only require one word.
61 BitVector m_localsForcedDouble;
62 BitVector m_localsForcedAnyInt;
63 FixedVector<OSREntryReshuffling> m_reshufflings;
64 BitVector m_machineStackUsed;
65
66 void dumpInContext(PrintStream&, DumpContext*) const;
67 void dump(PrintStream&) const;
68};
69
70inline BytecodeIndex getOSREntryDataBytecodeIndex(OSREntryData* osrEntryData)
71{
72 return osrEntryData->m_bytecodeIndex;
73}

75struct CatchEntrypointData {
76 // We use this when doing OSR entry at catch. We prove the arguments
77 // are of the expected type before entering at a catch block.
78 MacroAssemblerCodePtr<ExceptionHandlerPtrTag> machineCode;
79 FixedVector<FlushFormat> argumentFormats;
80 BytecodeIndex bytecodeIndex;
81};
