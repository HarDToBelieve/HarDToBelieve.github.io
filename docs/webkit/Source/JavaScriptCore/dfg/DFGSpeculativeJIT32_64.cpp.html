57bool SpeculativeJIT::fillJSValue(Edge edge, GPRReg& tagGPR, GPRReg& payloadGPR, FPRReg& fpr)
58{
59 // FIXME: For double we could fill with a FPR.
60 UNUSED_PARAM(fpr);
61
62 VirtualRegister virtualRegister = edge->virtualRegister();
63 GenerationInfo& info = generationInfoFromVirtualRegister(virtualRegister);
64
65 switch (info.registerFormat()) {
66 case DataFormatNone: {
67
68 if (edge->hasConstant()) {
69 tagGPR = allocate();
70 payloadGPR = allocate();
71 JSValue value = edge->asJSValue();
72 m_jit.move(Imm32(value.tag()), tagGPR);
73 m_jit.move(Imm32(value.payload()), payloadGPR);
74 m_gprs.retain(tagGPR, virtualRegister, SpillOrderConstant);
75 m_gprs.retain(payloadGPR, virtualRegister, SpillOrderConstant);
76 info.fillJSValue(*m_stream, tagGPR, payloadGPR, DataFormatJS);
77 } else {
78 DataFormat spillFormat = info.spillFormat();
79 ASSERT(spillFormat != DataFormatNone && spillFormat != DataFormatStorage);
80 tagGPR = allocate();
81 payloadGPR = allocate();
82 switch (spillFormat) {
83 case DataFormatInt32:
84 m_jit.move(TrustedImm32(JSValue::Int32Tag), tagGPR);
85 spillFormat = DataFormatJSInt32; // This will be used as the new register format.
86 break;
87 case DataFormatCell:
88 m_jit.move(TrustedImm32(JSValue::CellTag), tagGPR);
89 spillFormat = DataFormatJSCell; // This will be used as the new register format.
90 break;
91 case DataFormatBoolean:
92 m_jit.move(TrustedImm32(JSValue::BooleanTag), tagGPR);
93 spillFormat = DataFormatJSBoolean; // This will be used as the new register format.
94 break;
95 default:
96 m_jit.load32(JITCompiler::tagFor(virtualRegister), tagGPR);
97 break;
98 }
99 m_jit.load32(JITCompiler::payloadFor(virtualRegister), payloadGPR);
100 m_gprs.retain(tagGPR, virtualRegister, SpillOrderSpilled);
101 m_gprs.retain(payloadGPR, virtualRegister, SpillOrderSpilled);
102 info.fillJSValue(*m_stream, tagGPR, payloadGPR, spillFormat == DataFormatJSDouble ? DataFormatJS : spillFormat);
103 }
104
105 return true;
106 }
107
108 case DataFormatInt32:
109 case DataFormatCell:
110 case DataFormatBoolean: {
111 GPRReg gpr = info.gpr();
112 // If the register has already been locked we need to take a copy.
113 if (m_gprs.isLocked(gpr)) {
114 payloadGPR = allocate();
115 m_jit.move(gpr, payloadGPR);
116 } else {
117 payloadGPR = gpr;
118 m_gprs.lock(gpr);
119 }
120 tagGPR = allocate();
121 int32_t tag = JSValue::EmptyValueTag;
122 DataFormat fillFormat = DataFormatJS;
123 switch (info.registerFormat()) {
124 case DataFormatInt32:
125 tag = JSValue::Int32Tag;
126 fillFormat = DataFormatJSInt32;
127 break;
128 case DataFormatCell:
129 tag = JSValue::CellTag;
130 fillFormat = DataFormatJSCell;
131 break;
132 case DataFormatBoolean:
133 tag = JSValue::BooleanTag;
134 fillFormat = DataFormatJSBoolean;
135 break;
136 default:
137 RELEASE_ASSERT_NOT_REACHED();
138 break;
139 }
140 m_jit.move(TrustedImm32(tag), tagGPR);
141 m_gprs.release(gpr);
142 m_gprs.retain(tagGPR, virtualRegister, SpillOrderJS);
143 m_gprs.retain(payloadGPR, virtualRegister, SpillOrderJS);
144 info.fillJSValue(*m_stream, tagGPR, payloadGPR, fillFormat);
145 return true;
146 }
147
148 case DataFormatJSDouble:
149 case DataFormatJS:
150 case DataFormatJSInt32:
151 case DataFormatJSCell:
152 case DataFormatJSBoolean: {
153 tagGPR = info.tagGPR();
154 payloadGPR = info.payloadGPR();
155 m_gprs.lock(tagGPR);
156 m_gprs.lock(payloadGPR);
157 return true;
158 }
159
160 case DataFormatStorage:
161 case DataFormatDouble:
162 // this type currently never occurs
163 RELEASE_ASSERT_NOT_REACHED();
164
165 default:
166 RELEASE_ASSERT_NOT_REACHED();
167 return true;
168 }
169}

176void SpeculativeJIT::cachedGetById(
177 CodeOrigin codeOrigin, GPRReg baseTagGPROrNone, GPRReg basePayloadGPR, GPRReg resultTagGPR, GPRReg resultPayloadGPR, GPRReg stubInfoGPR,
178 CacheableIdentifier identifier, JITCompiler::Jump slowPathTarget, SpillRegistersMode spillMode, AccessType type)
179{
180 UNUSED_PARAM(stubInfoGPR);
181 // This is a hacky fix for when the register allocator decides to alias the base payload with the result tag. This only happens
182 // in the case of GetByIdFlush/GetByIdDirectFlush, which has a relatively expensive register allocation story already so we probably don't need to
183 // trip over one move instruction.
184 if (basePayloadGPR == resultTagGPR) {
185 RELEASE_ASSERT(basePayloadGPR != resultPayloadGPR);
186
187 if (baseTagGPROrNone == resultPayloadGPR) {
188 m_jit.swap(basePayloadGPR, baseTagGPROrNone);
189 baseTagGPROrNone = resultTagGPR;
190 } else
191 m_jit.move(basePayloadGPR, resultPayloadGPR);
192 basePayloadGPR = resultPayloadGPR;
193 }
194
195 RegisterSet usedRegisters = this->usedRegisters();
196 if (spillMode == DontSpill) {
197 // We've already flushed registers to the stack, we don't need to spill these.
198 usedRegisters.set(JSValueRegs(baseTagGPROrNone, basePayloadGPR), false);
199 usedRegisters.set(JSValueRegs(resultTagGPR, resultPayloadGPR), false);
200 }
201
202 CallSiteIndex callSite = m_jit.recordCallSiteAndGenerateExceptionHandlingOSRExitIfNeeded(codeOrigin, m_stream->size());
203 JITGetByIdGenerator gen(
204 m_jit.codeBlock(), JITType::DFGJIT, codeOrigin, callSite, usedRegisters, identifier,
205 JSValueRegs(baseTagGPROrNone, basePayloadGPR), JSValueRegs(resultTagGPR, resultPayloadGPR), InvalidGPRReg, type);
206
207 gen.generateFastPath(m_jit);
208
209 JITCompiler::JumpList slowCases;
210 if (slowPathTarget.isSet())
211 slowCases.append(slowPathTarget);
212 slowCases.append(gen.slowPathJump());
213
214 std::unique_ptr<SlowPathGenerator> slowPath;
215 if (baseTagGPROrNone == InvalidGPRReg) {
216 slowPath = slowPathCall(
217 slowCases, this, appropriateOptimizingGetByIdFunction(type),
218 JSValueRegs(resultTagGPR, resultPayloadGPR), TrustedImmPtr::weakPointer(m_graph, m_graph.globalObjectFor(codeOrigin)), gen.stubInfo(),
219 CCallHelpers::CellValue(basePayloadGPR),
220 identifier.rawBits());
221 } else {
222 slowPath = slowPathCall(
223 slowCases, this, appropriateOptimizingGetByIdFunction(type),
224 JSValueRegs(resultTagGPR, resultPayloadGPR), TrustedImmPtr::weakPointer(m_graph, m_graph.globalObjectFor(codeOrigin)), gen.stubInfo(), JSValueRegs(baseTagGPROrNone, basePayloadGPR), identifier.rawBits());
225 }
226
227 m_jit.addGetById(gen, slowPath.get());
228 addSlowPathGenerator(WTFMove(slowPath));
229}

271void SpeculativeJIT::nonSpeculativeNonPeepholeCompareNullOrUndefined(Edge operand)
272{
273 JSValueOperand arg(this, operand, ManualOperandSpeculation);
274 GPRReg argTagGPR = arg.tagGPR();
275 GPRReg argPayloadGPR = arg.payloadGPR();
276
277 GPRTemporary resultPayload(this, Reuse, arg, PayloadWord);
278 GPRReg resultPayloadGPR = resultPayload.gpr();
279
280 JITCompiler::Jump notCell;
281 JITCompiler::Jump notMasqueradesAsUndefined;
282 if (masqueradesAsUndefinedWatchpointIsStillValid()) {
283 if (!isKnownCell(operand.node()))
284 notCell = m_jit.branchIfNotCell(arg.jsValueRegs());
285
286 m_jit.move(TrustedImm32(0), resultPayloadGPR);
287 notMasqueradesAsUndefined = m_jit.jump();
288 } else {
289 GPRTemporary localGlobalObject(this);
290 GPRTemporary remoteGlobalObject(this);
291
292 if (!isKnownCell(operand.node()))
293 notCell = m_jit.branchIfNotCell(arg.jsValueRegs());
294
295 JITCompiler::Jump isMasqueradesAsUndefined = m_jit.branchTest8(
296 JITCompiler::NonZero,
297 JITCompiler::Address(argPayloadGPR, JSCell::typeInfoFlagsOffset()),
298 JITCompiler::TrustedImm32(MasqueradesAsUndefined));
299
300 m_jit.move(TrustedImm32(0), resultPayloadGPR);
301 notMasqueradesAsUndefined = m_jit.jump();
302
303 isMasqueradesAsUndefined.link(&m_jit);
304 GPRReg localGlobalObjectGPR = localGlobalObject.gpr();
305 GPRReg remoteGlobalObjectGPR = remoteGlobalObject.gpr();
306 m_jit.move(TrustedImmPtr::weakPointer(m_jit.graph(), m_jit.graph().globalObjectFor(m_currentNode->origin.semantic)), localGlobalObjectGPR);
307 m_jit.loadPtr(JITCompiler::Address(argPayloadGPR, JSCell::structureIDOffset()), resultPayloadGPR);
308 m_jit.loadPtr(JITCompiler::Address(resultPayloadGPR, Structure::globalObjectOffset()), remoteGlobalObjectGPR);
309 m_jit.compare32(JITCompiler::Equal, localGlobalObjectGPR, remoteGlobalObjectGPR, resultPayloadGPR);
310 }
311
312 if (!isKnownCell(operand.node())) {
313 JITCompiler::Jump done = m_jit.jump();
314
315 notCell.link(&m_jit);
316 // null or undefined?
317 COMPILE_ASSERT((JSValue::UndefinedTag | 1) == JSValue::NullTag, UndefinedTag_OR_1_EQUALS_NullTag);
318 m_jit.or32(TrustedImm32(1), argTagGPR, resultPayloadGPR);
319 m_jit.compare32(JITCompiler::Equal, resultPayloadGPR, TrustedImm32(JSValue::NullTag), resultPayloadGPR);
320
321 done.link(&m_jit);
322 }
323
324 notMasqueradesAsUndefined.link(&m_jit);
325
326 booleanResult(resultPayloadGPR, m_currentNode);
327}

329void SpeculativeJIT::nonSpeculativePeepholeBranchNullOrUndefined(Edge operand, Node* branchNode)
330{
331 BasicBlock* taken = branchNode->branchData()->taken.block;
332 BasicBlock* notTaken = branchNode->branchData()->notTaken.block;
333
334 bool invert = false;
335 if (taken == nextBlock()) {
336 invert = !invert;
337 BasicBlock* tmp = taken;
338 taken = notTaken;
339 notTaken = tmp;
340 }
341
342 JSValueOperand arg(this, operand, ManualOperandSpeculation);
343 GPRReg argTagGPR = arg.tagGPR();
344 GPRReg argPayloadGPR = arg.payloadGPR();
345
346 GPRTemporary result(this, Reuse, arg, TagWord);
347 GPRReg resultGPR = result.gpr();
348
349 JITCompiler::Jump notCell;
350
351 if (masqueradesAsUndefinedWatchpointIsStillValid()) {
352 if (!isKnownCell(operand.node()))
353 notCell = m_jit.branchIfNotCell(arg.jsValueRegs());
354
355 jump(invert ? taken : notTaken, ForceJump);
356 } else {
357 GPRTemporary localGlobalObject(this);
358 GPRTemporary remoteGlobalObject(this);
359
360 if (!isKnownCell(operand.node()))
361 notCell = m_jit.branchIfNotCell(arg.jsValueRegs());
362
363 branchTest8(JITCompiler::Zero,
364 JITCompiler::Address(argPayloadGPR, JSCell::typeInfoFlagsOffset()),
365 JITCompiler::TrustedImm32(MasqueradesAsUndefined),
366 invert ? taken : notTaken);
367
368 GPRReg localGlobalObjectGPR = localGlobalObject.gpr();
369 GPRReg remoteGlobalObjectGPR = remoteGlobalObject.gpr();
370 m_jit.move(TrustedImmPtr::weakPointer(m_jit.graph(), m_jit.graph().globalObjectFor(m_currentNode->origin.semantic)), localGlobalObjectGPR);
371 m_jit.loadPtr(JITCompiler::Address(argPayloadGPR, JSCell::structureIDOffset()), resultGPR);
372 m_jit.loadPtr(JITCompiler::Address(resultGPR, Structure::globalObjectOffset()), remoteGlobalObjectGPR);
373 branchPtr(JITCompiler::Equal, localGlobalObjectGPR, remoteGlobalObjectGPR, invert ? notTaken : taken);
374 }
375
376 if (!isKnownCell(operand.node())) {
377 jump(notTaken, ForceJump);
378
379 notCell.link(&m_jit);
380 // null or undefined?
381 COMPILE_ASSERT((JSValue::UndefinedTag | 1) == JSValue::NullTag, UndefinedTag_OR_1_EQUALS_NullTag);
382 m_jit.or32(TrustedImm32(1), argTagGPR, resultGPR);
383 branch32(invert ? JITCompiler::NotEqual : JITCompiler::Equal, resultGPR, JITCompiler::TrustedImm32(JSValue::NullTag), taken);
384 }
385
386 jump(notTaken);
387}

389void SpeculativeJIT::nonSpeculativePeepholeStrictEq(Node* node, Node* branchNode, bool invert)
390{
391 BasicBlock* taken = branchNode->branchData()->taken.block;
392 BasicBlock* notTaken = branchNode->branchData()->notTaken.block;
393
394 // The branch instruction will branch to the taken block.
395 // If taken is next, switch taken with notTaken & invert the branch condition so we can fall through.
396 if (taken == nextBlock()) {
397 invert = !invert;
398 BasicBlock* tmp = taken;
399 taken = notTaken;
400 notTaken = tmp;
401 }
402
403 JSValueOperand arg1(this, node->child1(), ManualOperandSpeculation);
404 JSValueOperand arg2(this, node->child2(), ManualOperandSpeculation);
405 speculate(node, node->child1());
406 speculate(node, node->child2());
407 GPRReg arg1PayloadGPR = arg1.payloadGPR();
408 GPRReg arg2PayloadGPR = arg2.payloadGPR();
409 JSValueRegs arg1Regs = arg1.jsValueRegs();
410 JSValueRegs arg2Regs = arg2.jsValueRegs();
411
412 GPRTemporary resultPayload(this, Reuse, arg1, PayloadWord);
413 GPRReg resultPayloadGPR = resultPayload.gpr();
414
415 arg1.use();
416 arg2.use();
417
418 if (isKnownCell(node->child1().node()) && isKnownCell(node->child2().node())) {
419 // see if we get lucky: if the arguments are cells and they reference the same
420 // cell, then they must be strictly equal.
421 branchPtr(JITCompiler::Equal, arg1PayloadGPR, arg2PayloadGPR, invert ? notTaken : taken);
422
423 silentSpillAllRegisters(resultPayloadGPR);
424 callOperation(operationCompareStrictEqCell, resultPayloadGPR, TrustedImmPtr::weakPointer(m_graph, m_graph.globalObjectFor(node->origin.semantic)), arg1PayloadGPR, arg2PayloadGPR);
425 m_jit.exceptionCheck();
426 silentFillAllRegisters();
427
428 branchTest32(invert ? JITCompiler::Zero : JITCompiler::NonZero, resultPayloadGPR, taken);
429 } else {
430 // FIXME: Add fast paths for twoCells, number etc.
431
432 silentSpillAllRegisters(resultPayloadGPR);
433 callOperation(operationCompareStrictEq, resultPayloadGPR, TrustedImmPtr::weakPointer(m_graph, m_graph.globalObjectFor(node->origin.semantic)), arg1Regs, arg2Regs);
434 m_jit.exceptionCheck();
435 silentFillAllRegisters();
436
437 branchTest32(invert ? JITCompiler::Zero : JITCompiler::NonZero, resultPayloadGPR, taken);
438 }
439
440 jump(notTaken);
441}

443void SpeculativeJIT::genericJSValueNonPeepholeStrictEq(Node* node, bool invert)
444{
445 JSValueOperand arg1(this, node->child1(), ManualOperandSpeculation);
446 JSValueOperand arg2(this, node->child2(), ManualOperandSpeculation);
447 speculate(node, node->child1());
448 speculate(node, node->child2());
449 GPRReg arg1PayloadGPR = arg1.payloadGPR();
450 GPRReg arg2PayloadGPR = arg2.payloadGPR();
451 JSValueRegs arg1Regs = arg1.jsValueRegs();
452 JSValueRegs arg2Regs = arg2.jsValueRegs();
453
454 GPRTemporary resultPayload(this, Reuse, arg1, PayloadWord);
455 GPRReg resultPayloadGPR = resultPayload.gpr();
456
457 arg1.use();
458 arg2.use();
459
460 if (isKnownCell(node->child1().node()) && isKnownCell(node->child2().node())) {
461 // see if we get lucky: if the arguments are cells and they reference the same
462 // cell, then they must be strictly equal.
463 // FIXME: this should flush registers instead of silent spill/fill.
464 JITCompiler::Jump notEqualCase = m_jit.branchPtr(JITCompiler::NotEqual, arg1PayloadGPR, arg2PayloadGPR);
465
466 m_jit.move(JITCompiler::TrustedImm32(!invert), resultPayloadGPR);
467 JITCompiler::Jump done = m_jit.jump();
468
469 notEqualCase.link(&m_jit);
470
471 silentSpillAllRegisters(resultPayloadGPR);
472 callOperation(operationCompareStrictEqCell, resultPayloadGPR, TrustedImmPtr::weakPointer(m_graph, m_graph.globalObjectFor(node->origin.semantic)), arg1PayloadGPR, arg2PayloadGPR);
473 m_jit.exceptionCheck();
474 silentFillAllRegisters();
475
476 m_jit.andPtr(JITCompiler::TrustedImm32(1), resultPayloadGPR);
477
478 done.link(&m_jit);
479 } else {
480 // FIXME: Add fast paths.
481
482 silentSpillAllRegisters(resultPayloadGPR);
483 callOperation(operationCompareStrictEq, resultPayloadGPR, TrustedImmPtr::weakPointer(m_graph, m_graph.globalObjectFor(node->origin.semantic)), arg1Regs, arg2Regs);
484 silentFillAllRegisters();
485 m_jit.exceptionCheck();
486
487 m_jit.andPtr(JITCompiler::TrustedImm32(1), resultPayloadGPR);
488 }
489
490 booleanResult(resultPayloadGPR, node, UseChildrenCalledExplicitly);
491}

506void SpeculativeJIT::emitCall(Node* node)
507{
508 CallLinkInfo::CallType callType;
509 bool isVarargs = false;
510 bool isForwardVarargs = false;
511 bool isTail = false;
512 bool isDirect = false;
513 bool isEmulatedTail = false;
514 switch (node->op()) {
515 case Call:
516 case CallEval:
517 callType = CallLinkInfo::Call;
518 break;
519 case TailCall:
520 callType = CallLinkInfo::TailCall;
521 isTail = true;
522 break;
523 case TailCallInlinedCaller:
524 callType = CallLinkInfo::Call;
525 isEmulatedTail = true;
526 break;
527 case Construct:
528 callType = CallLinkInfo::Construct;
529 break;
530 case CallVarargs:
531 callType = CallLinkInfo::CallVarargs;
532 isVarargs = true;
533 break;
534 case TailCallVarargs:
535 callType = CallLinkInfo::TailCallVarargs;
536 isVarargs = true;
537 isTail = true;
538 break;
539 case TailCallVarargsInlinedCaller:
540 callType = CallLinkInfo::CallVarargs;
541 isVarargs = true;
542 isEmulatedTail = true;
543 break;
544 case ConstructVarargs:
545 callType = CallLinkInfo::ConstructVarargs;
546 isVarargs = true;
547 break;
548 case CallForwardVarargs:
549 callType = CallLinkInfo::CallVarargs;
550 isForwardVarargs = true;
551 break;
552 case TailCallForwardVarargs:
553 callType = CallLinkInfo::TailCallVarargs;
554 isTail = true;
555 isForwardVarargs = true;
556 break;
557 case TailCallForwardVarargsInlinedCaller:
558 callType = CallLinkInfo::CallVarargs;
559 isEmulatedTail = true;
560 isForwardVarargs = true;
561 break;
562 case ConstructForwardVarargs:
563 callType = CallLinkInfo::ConstructVarargs;
564 isForwardVarargs = true;
565 break;
566 case DirectCall:
567 callType = CallLinkInfo::DirectCall;
568 isDirect = true;
569 break;
570 case DirectConstruct:
571 callType = CallLinkInfo::DirectConstruct;
572 isDirect = true;
573 break;
574 case DirectTailCall:
575 callType = CallLinkInfo::DirectTailCall;
576 isTail = true;
577 isDirect = true;
578 break;
579 case DirectTailCallInlinedCaller:
580 callType = CallLinkInfo::DirectCall;
581 isEmulatedTail = true;
582 isDirect = true;
583 break;
584 default:
585 DFG_CRASH(m_jit.graph(), node, "bad node type");
586 break;
587 }
588
589 Edge calleeEdge = m_jit.graph().child(node, 0);
590 GPRReg calleeTagGPR = InvalidGPRReg;
591 GPRReg calleePayloadGPR = InvalidGPRReg;
592 CallFrameShuffleData shuffleData;
593
594 JSGlobalObject* globalObject = m_graph.globalObjectFor(node->origin.semantic);
595 ExecutableBase* executable = nullptr;
596 FunctionExecutable* functionExecutable = nullptr;
597 if (isDirect) {
598 executable = node->castOperand<ExecutableBase*>();
599 functionExecutable = jsDynamicCast<FunctionExecutable*>(vm(), executable);
600 }
601
602 unsigned numPassedArgs = 0;
603 unsigned numAllocatedArgs = 0;
604
605 // Gotta load the arguments somehow. Varargs is trickier.
606 if (isVarargs || isForwardVarargs) {
607 RELEASE_ASSERT(!isDirect);
608 CallVarargsData* data = node->callVarargsData();
609
610 int numUsedStackSlots = m_jit.graph().m_nextMachineLocal;
611
612 if (isForwardVarargs) {
613 flushRegisters();
614 if (node->child3())
615 use(node->child3());
616
617 GPRReg scratchGPR1;
618 GPRReg scratchGPR2;
619 GPRReg scratchGPR3;
620
621 scratchGPR1 = JITCompiler::selectScratchGPR();
622 scratchGPR2 = JITCompiler::selectScratchGPR(scratchGPR1);
623 scratchGPR3 = JITCompiler::selectScratchGPR(scratchGPR1, scratchGPR2);
624
625 m_jit.move(TrustedImm32(numUsedStackSlots), scratchGPR2);
626 JITCompiler::JumpList slowCase;
627 InlineCallFrame* inlineCallFrame;
628 if (node->child3())
629 inlineCallFrame = node->child3()->origin.semantic.inlineCallFrame();
630 else
631 inlineCallFrame = node->origin.semantic.inlineCallFrame();
632 // emitSetupVarargsFrameFastCase modifies the stack pointer if it succeeds.
633 emitSetupVarargsFrameFastCase(vm(), m_jit, scratchGPR2, scratchGPR1, scratchGPR2, scratchGPR3, inlineCallFrame, data->firstVarArgOffset, slowCase);
634 JITCompiler::Jump done = m_jit.jump();
635 slowCase.link(&m_jit);
636 callOperation(operationThrowStackOverflowForVarargs, TrustedImmPtr::weakPointer(m_graph, globalObject));
637 m_jit.exceptionCheck();
638 m_jit.abortWithReason(DFGVarargsThrowingPathDidNotThrow);
639 done.link(&m_jit);
640 } else {
641 GPRReg argumentsPayloadGPR;
642 GPRReg argumentsTagGPR;
643 GPRReg scratchGPR1;
644 GPRReg scratchGPR2;
645 GPRReg scratchGPR3;
646
647 auto loadArgumentsGPR = [&] (GPRReg reservedGPR) {
648 if (reservedGPR != InvalidGPRReg)
649 lock(reservedGPR);
650 JSValueOperand arguments(this, node->child3());
651 argumentsTagGPR = arguments.tagGPR();
652 argumentsPayloadGPR = arguments.payloadGPR();
653 if (reservedGPR != InvalidGPRReg)
654 unlock(reservedGPR);
655 flushRegisters();
656
657 scratchGPR1 = JITCompiler::selectScratchGPR(argumentsPayloadGPR, argumentsTagGPR, reservedGPR);
658 scratchGPR2 = JITCompiler::selectScratchGPR(argumentsPayloadGPR, argumentsTagGPR, scratchGPR1, reservedGPR);
659 scratchGPR3 = JITCompiler::selectScratchGPR(argumentsPayloadGPR, argumentsTagGPR, scratchGPR1, scratchGPR2, reservedGPR);
660 };
661
662 loadArgumentsGPR(InvalidGPRReg);
663
664 DFG_ASSERT(m_jit.graph(), node, isFlushed());
665
666 // Right now, arguments is in argumentsTagGPR/argumentsPayloadGPR and the register file is
667 // flushed.
668 callOperation(operationSizeFrameForVarargs, GPRInfo::returnValueGPR, TrustedImmPtr::weakPointer(m_graph, m_graph.globalObjectFor(node->origin.semantic)), JSValueRegs(argumentsTagGPR, argumentsPayloadGPR), numUsedStackSlots, data->firstVarArgOffset);
669 m_jit.exceptionCheck();
670
671 // Now we have the argument count of the callee frame, but we've lost the arguments operand.
672 // Reconstruct the arguments operand while preserving the callee frame.
673 loadArgumentsGPR(GPRInfo::returnValueGPR);
674 m_jit.move(TrustedImm32(numUsedStackSlots), scratchGPR1);
675 emitSetVarargsFrame(m_jit, GPRInfo::returnValueGPR, false, scratchGPR1, scratchGPR1);
676 m_jit.addPtr(TrustedImm32(-(sizeof(CallerFrameAndPC) + WTF::roundUpToMultipleOf(stackAlignmentBytes(), 6 * sizeof(void*)))), scratchGPR1, JITCompiler::stackPointerRegister);
677
678 callOperation(operationSetupVarargsFrame, GPRInfo::returnValueGPR, TrustedImmPtr::weakPointer(m_graph, m_graph.globalObjectFor(node->origin.semantic)), scratchGPR1, JSValueRegs(argumentsTagGPR, argumentsPayloadGPR), data->firstVarArgOffset, GPRInfo::returnValueGPR);
679 m_jit.exceptionCheck();
680 m_jit.addPtr(TrustedImm32(sizeof(CallerFrameAndPC)), GPRInfo::returnValueGPR, JITCompiler::stackPointerRegister);
681 }
682
683 DFG_ASSERT(m_jit.graph(), node, isFlushed());
684
685 // We don't need the arguments array anymore.
686 if (isVarargs)
687 use(node->child3());
688
689 // Now set up the "this" argument.
690 JSValueOperand thisArgument(this, node->child2());
691 GPRReg thisArgumentTagGPR = thisArgument.tagGPR();
692 GPRReg thisArgumentPayloadGPR = thisArgument.payloadGPR();
693 thisArgument.use();
694
695 m_jit.store32(thisArgumentTagGPR, JITCompiler::calleeArgumentTagSlot(0));
696 m_jit.store32(thisArgumentPayloadGPR, JITCompiler::calleeArgumentPayloadSlot(0));
697 } else {
698 // The call instruction's first child is either the function (normal call) or the
699 // receiver (method call). subsequent children are the arguments.
700 numPassedArgs = node->numChildren() - 1;
701 numAllocatedArgs = numPassedArgs;
702
703 if (functionExecutable) {
704 // Allocate more args if this would let us avoid arity checks. This is throttled by
705 // CallLinkInfo's limit. It's probably good to throttle it - if the callee wants a
706 // ginormous amount of argument space then it's better for them to do it so that when we
707 // make calls to other things, we don't waste space.
708 unsigned desiredNumAllocatedArgs = static_cast<unsigned>(functionExecutable->parameterCount()) + 1;
709 if (desiredNumAllocatedArgs <= Options::maximumDirectCallStackSize()) {
710 numAllocatedArgs = std::max(numAllocatedArgs, desiredNumAllocatedArgs);
711
712 // Whoever converts to DirectCall should do this adjustment. It's too late for us to
713 // do this adjustment now since we will have already emitted code that relied on the
714 // value of m_parameterSlots.
715 DFG_ASSERT(
716 m_jit.graph(), node,
717 Graph::parameterSlotsForArgCount(numAllocatedArgs)
718 <= m_jit.graph().m_parameterSlots);
719 }
720 }
721
722 if (isTail) {
723 JSValueOperand callee(this, calleeEdge);
724 calleeTagGPR = callee.tagGPR();
725 calleePayloadGPR = callee.payloadGPR();
726 if (!isDirect)
727 use(calleeEdge);
728
729 shuffleData.numLocals = m_jit.graph().frameRegisterCount();
730 shuffleData.callee = ValueRecovery::inPair(calleeTagGPR, calleePayloadGPR);
731 shuffleData.args.resize(numAllocatedArgs);
732 shuffleData.numPassedArgs = numPassedArgs;
733
734 for (unsigned i = 0; i < numPassedArgs; ++i) {
735 Edge argEdge = m_jit.graph().varArgChild(node, i + 1);
736 GenerationInfo& info = generationInfo(argEdge.node());
737 if (!isDirect)
738 use(argEdge);
739 shuffleData.args[i] = info.recovery(argEdge->virtualRegister());
740 }
741
742 for (unsigned i = numPassedArgs; i < numAllocatedArgs; ++i)
743 shuffleData.args[i] = ValueRecovery::constant(jsUndefined());
744 } else {
745 m_jit.store32(MacroAssembler::TrustedImm32(numPassedArgs), m_jit.calleeFramePayloadSlot(CallFrameSlot::argumentCountIncludingThis));
746
747 for (unsigned i = 0; i < numPassedArgs; i++) {
748 Edge argEdge = m_jit.graph().m_varArgChildren[node->firstChild() + 1 + i];
749 JSValueOperand arg(this, argEdge);
750 GPRReg argTagGPR = arg.tagGPR();
751 GPRReg argPayloadGPR = arg.payloadGPR();
752 use(argEdge);
753
754 m_jit.store32(argTagGPR, m_jit.calleeArgumentTagSlot(i));
755 m_jit.store32(argPayloadGPR, m_jit.calleeArgumentPayloadSlot(i));
756 }
757
758 for (unsigned i = numPassedArgs; i < numAllocatedArgs; ++i)
759 m_jit.storeTrustedValue(jsUndefined(), JITCompiler::calleeArgumentSlot(i));
760 }
761 }
762
763 if (!isTail || isVarargs || isForwardVarargs) {
764 JSValueOperand callee(this, calleeEdge);
765 calleeTagGPR = callee.tagGPR();
766 calleePayloadGPR = callee.payloadGPR();
767 use(calleeEdge);
768 m_jit.store32(calleePayloadGPR, m_jit.calleeFramePayloadSlot(CallFrameSlot::callee));
769 m_jit.store32(calleeTagGPR, m_jit.calleeFrameTagSlot(CallFrameSlot::callee));
770
771 if (!isTail)
772 flushRegisters();
773 }
774
775 JITCompiler::DataLabelPtr targetToCheck;
776 JITCompiler::JumpList slowPath;
777
778 CodeOrigin staticOrigin = node->origin.semantic;
779 InlineCallFrame* staticInlineCallFrame = staticOrigin.inlineCallFrame();
780 ASSERT(!isTail || !staticInlineCallFrame || !staticInlineCallFrame->getCallerSkippingTailCalls());
781 ASSERT(!isEmulatedTail || (staticInlineCallFrame && staticInlineCallFrame->getCallerSkippingTailCalls()));
782 CodeOrigin dynamicOrigin =
783 isEmulatedTail ? *staticInlineCallFrame->getCallerSkippingTailCalls() : staticOrigin;
784 CallSiteIndex callSite = m_jit.recordCallSiteAndGenerateExceptionHandlingOSRExitIfNeeded(dynamicOrigin, m_stream->size());
785
786 CallLinkInfo* info = m_jit.codeBlock()->addCallLinkInfo(node->origin.semantic);
787 info->setUpCall(callType, calleePayloadGPR);
788
789 auto setResultAndResetStack = [&] () {
790 JSValueRegsFlushedCallResult result(this);
791 JSValueRegs resultRegs = result.regs();
792
793 m_jit.setupResults(resultRegs);
794
795 jsValueResult(resultRegs, node, DataFormatJS, UseChildrenCalledExplicitly);
796 // After the calls are done, we need to reestablish our stack
797 // pointer. We rely on this for varargs calls, calls with arity
798 // mismatch (the callframe is slided) and tail calls.
799 m_jit.addPtr(TrustedImm32(m_jit.graph().stackPointerOffset() * sizeof(Register)), GPRInfo::callFrameRegister, JITCompiler::stackPointerRegister);
800 };
801
802 if (node->op() == CallEval) {
803 // We want to call operationCallEval but we don't want to overwrite the parameter area in
804 // which we have created a prototypical eval call frame. This means that we have to
805 // subtract stack to make room for the call. Lucky for us, at this point we have the whole
806 // register file to ourselves.
807
808 m_jit.emitStoreCallSiteIndex(callSite);
809 m_jit.addPtr(TrustedImm32(-static_cast<ptrdiff_t>(sizeof(CallerFrameAndPC))), JITCompiler::stackPointerRegister, GPRInfo::regT0);
810 m_jit.storePtr(GPRInfo::callFrameRegister, JITCompiler::Address(GPRInfo::regT0, CallFrame::callerFrameOffset()));
811
812 // Now we need to make room for:
813 // - The caller frame and PC of a call to operationCallEval.
814 // - Potentially two arguments on the stack.
815 unsigned requiredBytes = sizeof(CallerFrameAndPC) + sizeof(CallFrame*) * 2;
816 requiredBytes = WTF::roundUpToMultipleOf(stackAlignmentBytes(), requiredBytes);
817 m_jit.subPtr(TrustedImm32(requiredBytes), JITCompiler::stackPointerRegister);
818 m_jit.move(TrustedImm32(node->ecmaMode().value()), GPRInfo::regT1);
819 m_jit.setupArguments<decltype(operationCallEval)>(TrustedImmPtr::weakPointer(m_graph, globalObject), GPRInfo::regT0, GPRInfo::regT1);
820 prepareForExternalCall();
821 m_jit.appendCall(operationCallEval);
822 m_jit.exceptionCheck();
823 JITCompiler::Jump done = m_jit.branchIfNotEmpty(GPRInfo::returnValueGPR2);
824
825 // This is the part where we meant to make a normal call. Oops.
826 m_jit.addPtr(TrustedImm32(requiredBytes), JITCompiler::stackPointerRegister);
827 m_jit.load32(JITCompiler::calleeFrameSlot(CallFrameSlot::callee).withOffset(PayloadOffset), GPRInfo::regT0);
828 m_jit.load32(JITCompiler::calleeFrameSlot(CallFrameSlot::callee).withOffset(TagOffset), GPRInfo::regT1);
829 m_jit.emitVirtualCall(vm(), globalObject, info);
830
831 done.link(&m_jit);
832 setResultAndResetStack();
833 return;
834 }
835
836 if (isDirect) {
837 info->setExecutableDuringCompilation(executable);
838 info->setMaxArgumentCountIncludingThis(numAllocatedArgs);
839
840 if (isTail) {
841 RELEASE_ASSERT(node->op() == DirectTailCall);
842
843 JITCompiler::Label mainPath = m_jit.label();
844
845 m_jit.emitStoreCallSiteIndex(callSite);
846
847 info->emitDirectTailCallFastPath(m_jit, [&] {
848 info->setFrameShuffleData(shuffleData);
849 CallFrameShuffler(m_jit, shuffleData).prepareForTailCall();
850 });
851
852 JITCompiler::Label slowPath = m_jit.label();
853 silentSpillAllRegisters(InvalidGPRReg);
854 callOperation(operationLinkDirectCall, info, calleePayloadGPR);
855 silentFillAllRegisters();
856 m_jit.exceptionCheck();
857 m_jit.jump().linkTo(mainPath, &m_jit);
858
859 useChildren(node);
860
861 m_jit.addJSDirectCall(slowPath, info);
862 return;
863 }
864
865 JITCompiler::Label mainPath = m_jit.label();
866
867 m_jit.emitStoreCallSiteIndex(callSite);
868
869 info->emitDirectFastPath(m_jit);
870 JITCompiler::Jump done = m_jit.jump();
871
872 JITCompiler::Label slowPath = m_jit.label();
873 callOperation(operationLinkDirectCall, info, calleePayloadGPR);
874 m_jit.exceptionCheck();
875 m_jit.jump().linkTo(mainPath, &m_jit);
876
877 done.link(&m_jit);
878
879 setResultAndResetStack();
880
881 m_jit.addJSDirectCall(slowPath, info);
882 return;
883 }
884
885 m_jit.emitStoreCallSiteIndex(callSite);
886
887 slowPath.append(m_jit.branchIfNotCell(JSValueRegs(calleeTagGPR, calleePayloadGPR)));
888 slowPath.append(m_jit.branchPtrWithPatch(MacroAssembler::NotEqual, calleePayloadGPR, targetToCheck));
889
890 CCallHelpers::JumpList slowCases;
891 if (isTail) {
892 slowCases = info->emitTailCallFastPath(m_jit, calleePayloadGPR, InvalidGPRReg, CallLinkInfo::UseDataIC::No, [&] {
893 if (node->op() == TailCall) {
894 info->setFrameShuffleData(shuffleData);
895 CallFrameShuffler(m_jit, shuffleData).prepareForTailCall();
896 } else {
897 m_jit.emitRestoreCalleeSaves();
898 m_jit.prepareForTailCallSlow();
899 }
900 });
901 } else
902 slowCases = info->emitFastPath(m_jit, calleePayloadGPR, InvalidGPRReg, CallLinkInfo::UseDataIC::No);
903
904 JITCompiler::Jump done = m_jit.jump();
905
906 slowPath.link(&m_jit);
907 slowCases.link(&m_jit);
908 auto slowPathStart = m_jit.label();
909
910 if (node->op() == TailCall) {
911 CallFrameShuffler callFrameShuffler(m_jit, shuffleData);
912 callFrameShuffler.setCalleeJSValueRegs(JSValueRegs(
913 GPRInfo::regT1, GPRInfo::regT0));
914 callFrameShuffler.prepareForSlowPath();
915 } else {
916 // Callee payload needs to be in regT0, tag in regT1
917 if (calleeTagGPR == GPRInfo::regT0) {
918 if (calleePayloadGPR == GPRInfo::regT1)
919 m_jit.swap(GPRInfo::regT1, GPRInfo::regT0);
920 else {
921 m_jit.move(calleeTagGPR, GPRInfo::regT1);
922 m_jit.move(calleePayloadGPR, GPRInfo::regT0);
923 }
924 } else {
925 m_jit.move(calleePayloadGPR, GPRInfo::regT0);
926 m_jit.move(calleeTagGPR, GPRInfo::regT1);
927 }
928
929 if (isTail)
930 m_jit.emitRestoreCalleeSaves();
931 }
932
933 m_jit.move(TrustedImmPtr::weakPointer(m_graph, globalObject), GPRInfo::regT3);
934 info->emitSlowPath(vm(), m_jit);
935
936 done.link(&m_jit);
937 auto doneLocation = m_jit.label();
938
939 if (isTail)
940 m_jit.abortWithReason(JITDidReturnFromTailCall);
941 else
942 setResultAndResetStack();
943
944 m_jit.addJSCall(slowPathStart, doneLocation, info);
945}

947template<bool strict>
948GPRReg SpeculativeJIT::fillSpeculateInt32Internal(Edge edge, DataFormat& returnFormat)
949{
950 AbstractValue& value = m_state.forNode(edge);
951 SpeculatedType type = value.m_type;
952 ASSERT(edge.useKind() != KnownInt32Use || !(value.m_type & ~SpecInt32Only));
953
954 m_interpreter.filter(value, SpecInt32Only);
955 if (value.isClear()) {
956 terminateSpeculativeExecution(Uncountable, JSValueRegs(), 0);
957 returnFormat = DataFormatInt32;
958 return allocate();
959 }
960
961 VirtualRegister virtualRegister = edge->virtualRegister();
962 GenerationInfo& info = generationInfoFromVirtualRegister(virtualRegister);
963
964 switch (info.registerFormat()) {
965 case DataFormatNone: {
966 if (edge->hasConstant()) {
967 ASSERT(edge->isInt32Constant());
968 GPRReg gpr = allocate();
969 m_jit.move(MacroAssembler::Imm32(edge->asInt32()), gpr);
970 m_gprs.retain(gpr, virtualRegister, SpillOrderConstant);
971 info.fillInt32(*m_stream, gpr);
972 returnFormat = DataFormatInt32;
973 return gpr;
974 }
975
976 DataFormat spillFormat = info.spillFormat();
977
978 ASSERT_UNUSED(spillFormat, (spillFormat & DataFormatJS) || spillFormat == DataFormatInt32);
979
980 // If we know this was spilled as an integer we can fill without checking.
981 if (type & ~SpecInt32Only)
982 speculationCheck(BadType, JSValueSource(JITCompiler::addressFor(virtualRegister)), edge, m_jit.branch32(MacroAssembler::NotEqual, JITCompiler::tagFor(virtualRegister), TrustedImm32(JSValue::Int32Tag)));
983
984 GPRReg gpr = allocate();
985 m_jit.load32(JITCompiler::payloadFor(virtualRegister), gpr);
986 m_gprs.retain(gpr, virtualRegister, SpillOrderSpilled);
987 info.fillInt32(*m_stream, gpr);
988 returnFormat = DataFormatInt32;
989 return gpr;
990 }
991
992 case DataFormatJSInt32:
993 case DataFormatJS: {
994 // Check the value is an integer.
995 GPRReg tagGPR = info.tagGPR();
996 GPRReg payloadGPR = info.payloadGPR();
997 m_gprs.lock(tagGPR);
998 m_gprs.lock(payloadGPR);
999 if (type & ~SpecInt32Only)
1000 speculationCheck(BadType, JSValueRegs(tagGPR, payloadGPR), edge, m_jit.branchIfNotInt32(tagGPR));
1001 m_gprs.unlock(tagGPR);
1002 m_gprs.release(tagGPR);
1003 m_gprs.release(payloadGPR);
1004 m_gprs.retain(payloadGPR, virtualRegister, SpillOrderInteger);
1005 info.fillInt32(*m_stream, payloadGPR);
1006 // If !strict we're done, return.
1007 returnFormat = DataFormatInt32;
1008 return payloadGPR;
1009 }
1010
1011 case DataFormatInt32: {
1012 GPRReg gpr = info.gpr();
1013 m_gprs.lock(gpr);
1014 returnFormat = DataFormatInt32;
1015 return gpr;
1016 }
1017
1018 case DataFormatCell:
1019 case DataFormatBoolean:
1020 case DataFormatJSDouble:
1021 case DataFormatJSCell:
1022 case DataFormatJSBoolean:
1023 case DataFormatDouble:
1024 case DataFormatStorage:
1025 default:
1026 RELEASE_ASSERT_NOT_REACHED();
1027 return InvalidGPRReg;
1028 }
1029}

1076GPRReg SpeculativeJIT::fillSpeculateCell(Edge edge)
1077{
1078 AbstractValue& value = m_state.forNode(edge);
1079 SpeculatedType type = value.m_type;
1080 ASSERT((edge.useKind() != KnownCellUse && edge.useKind() != KnownStringUse) || !(value.m_type & ~SpecCell));
1081
1082 m_interpreter.filter(value, SpecCell);
1083 if (value.isClear()) {
1084 terminateSpeculativeExecution(Uncountable, JSValueRegs(), 0);
1085 return allocate();
1086 }
1087
1088 VirtualRegister virtualRegister = edge->virtualRegister();
1089 GenerationInfo& info = generationInfoFromVirtualRegister(virtualRegister);
1090
1091 switch (info.registerFormat()) {
1092 case DataFormatNone: {
1093 if (edge->hasConstant()) {
1094 GPRReg gpr = allocate();
1095 m_gprs.retain(gpr, virtualRegister, SpillOrderConstant);
1096 m_jit.move(TrustedImmPtr(edge->constant()), gpr);
1097 info.fillCell(*m_stream, gpr);
1098 return gpr;
1099 }
1100
1101 ASSERT((info.spillFormat() & DataFormatJS) || info.spillFormat() == DataFormatCell);
1102 if (type & ~SpecCell) {
1103 speculationCheck(
1104 BadType,
1105 JSValueSource(JITCompiler::addressFor(virtualRegister)),
1106 edge,
1107 m_jit.branch32(
1108 MacroAssembler::NotEqual,
1109 JITCompiler::tagFor(virtualRegister),
1110 TrustedImm32(JSValue::CellTag)));
1111 }
1112 GPRReg gpr = allocate();
1113 m_jit.load32(JITCompiler::payloadFor(virtualRegister), gpr);
1114 m_gprs.retain(gpr, virtualRegister, SpillOrderSpilled);
1115 info.fillCell(*m_stream, gpr);
1116 return gpr;
1117 }
1118
1119 case DataFormatCell: {
1120 GPRReg gpr = info.gpr();
1121 m_gprs.lock(gpr);
1122 return gpr;
1123 }
1124
1125 case DataFormatJSCell:
1126 case DataFormatJS: {
1127 GPRReg tagGPR = info.tagGPR();
1128 GPRReg payloadGPR = info.payloadGPR();
1129 m_gprs.lock(tagGPR);
1130 m_gprs.lock(payloadGPR);
1131 if (type & ~SpecCell) {
1132 speculationCheck(
1133 BadType, JSValueRegs(tagGPR, payloadGPR), edge,
1134 m_jit.branchIfNotCell(info.jsValueRegs()));
1135 }
1136 m_gprs.unlock(tagGPR);
1137 m_gprs.release(tagGPR);
1138 m_gprs.release(payloadGPR);
1139 m_gprs.retain(payloadGPR, virtualRegister, SpillOrderCell);
1140 info.fillCell(*m_stream, payloadGPR);
1141 return payloadGPR;
1142 }
1143
1144 case DataFormatJSInt32:
1145 case DataFormatInt32:
1146 case DataFormatJSDouble:
1147 case DataFormatJSBoolean:
1148 case DataFormatBoolean:
1149 case DataFormatDouble:
1150 case DataFormatStorage:
1151 RELEASE_ASSERT_NOT_REACHED();
1152
1153 default:
1154 RELEASE_ASSERT_NOT_REACHED();
1155 return InvalidGPRReg;
1156 }
1157}

1159GPRReg SpeculativeJIT::fillSpeculateBoolean(Edge edge)
1160{
1161 AbstractValue& value = m_state.forNode(edge);
1162 SpeculatedType type = value.m_type;
1163 ASSERT(edge.useKind() != KnownBooleanUse || !(value.m_type & ~SpecBoolean));
1164
1165 m_interpreter.filter(value, SpecBoolean);
1166 if (value.isClear()) {
1167 terminateSpeculativeExecution(Uncountable, JSValueRegs(), 0);
1168 return allocate();
1169 }
1170
1171 VirtualRegister virtualRegister = edge->virtualRegister();
1172 GenerationInfo& info = generationInfoFromVirtualRegister(virtualRegister);
1173
1174 switch (info.registerFormat()) {
1175 case DataFormatNone: {
1176 if (edge->hasConstant()) {
1177 JSValue jsValue = edge->asJSValue();
1178 GPRReg gpr = allocate();
1179 m_gprs.retain(gpr, virtualRegister, SpillOrderConstant);
1180 m_jit.move(MacroAssembler::TrustedImm32(jsValue.asBoolean()), gpr);
1181 info.fillBoolean(*m_stream, gpr);
1182 return gpr;
1183 }
1184
1185 ASSERT((info.spillFormat() & DataFormatJS) || info.spillFormat() == DataFormatBoolean);
1186
1187 if (type & ~SpecBoolean)
1188 speculationCheck(BadType, JSValueSource(JITCompiler::addressFor(virtualRegister)), edge, m_jit.branch32(MacroAssembler::NotEqual, JITCompiler::tagFor(virtualRegister), TrustedImm32(JSValue::BooleanTag)));
1189
1190 GPRReg gpr = allocate();
1191 m_jit.load32(JITCompiler::payloadFor(virtualRegister), gpr);
1192 m_gprs.retain(gpr, virtualRegister, SpillOrderSpilled);
1193 info.fillBoolean(*m_stream, gpr);
1194 return gpr;
1195 }
1196
1197 case DataFormatBoolean: {
1198 GPRReg gpr = info.gpr();
1199 m_gprs.lock(gpr);
1200 return gpr;
1201 }
1202
1203 case DataFormatJSBoolean:
1204 case DataFormatJS: {
1205 GPRReg tagGPR = info.tagGPR();
1206 GPRReg payloadGPR = info.payloadGPR();
1207 m_gprs.lock(tagGPR);
1208 m_gprs.lock(payloadGPR);
1209 if (type & ~SpecBoolean)
1210 speculationCheck(BadType, JSValueRegs(tagGPR, payloadGPR), edge, m_jit.branchIfNotBoolean(tagGPR, InvalidGPRReg));
1211
1212 m_gprs.unlock(tagGPR);
1213 m_gprs.release(tagGPR);
1214 m_gprs.release(payloadGPR);
1215 m_gprs.retain(payloadGPR, virtualRegister, SpillOrderBoolean);
1216 info.fillBoolean(*m_stream, payloadGPR);
1217 return payloadGPR;
1218 }
1219
1220 case DataFormatJSInt32:
1221 case DataFormatInt32:
1222 case DataFormatJSDouble:
1223 case DataFormatJSCell:
1224 case DataFormatCell:
1225 case DataFormatDouble:
1226 case DataFormatStorage:
1227 RELEASE_ASSERT_NOT_REACHED();
1228
1229 default:
1230 RELEASE_ASSERT_NOT_REACHED();
1231 return InvalidGPRReg;
1232 }

1286void SpeculativeJIT::compileObjectToObjectOrOtherEquality(Edge leftChild, Edge rightChild)
1287{

1311 }

1315 // or an object, b is usually an object. Balance the branches to make that case fast.
1316 MacroAssembler::Jump rightNotCell = m_jit.branchIfNotCell(op2.jsValueRegs());

1318 // We know that within this branch, rightChild must be a cell.

1327 MacroAssembler::NonZero,
1328 MacroAssembler::Address(op2PayloadGPR, JSCell::typeInfoFlagsOffset()),
1329 MacroAssembler::TrustedImm32(MasqueradesAsUndefined)));
1330 }

1332 // At this point we know that we can perform a straight-forward equality comparison on pointer
1333 // values because both left and right are pointers to objects that have no special equality
1334 // protocols.
1335 MacroAssembler::Jump falseCase = m_jit.branchPtr(MacroAssembler::NotEqual, op1GPR, op2PayloadGPR);
1336 MacroAssembler::Jump trueCase = m_jit.jump();
1337
1338 rightNotCell.link(&m_jit);
1339
1340 // We know that within this branch, rightChild must not be a cell. Check if that is enough to
1341 // prove that it is either null or undefined.
1342 if (needsTypeCheck(rightChild, SpecCell | SpecOther)) {
1343 m_jit.or32(TrustedImm32(1), op2TagGPR, resultGPR);
1344
1345 typeCheck(
1346 JSValueRegs(op2TagGPR, op2PayloadGPR), rightChild, SpecCell | SpecOther,
1347 m_jit.branch32(
1348 MacroAssembler::NotEqual, resultGPR,
1349 MacroAssembler::TrustedImm32(JSValue::NullTag)));
1350 }
1351
1352 falseCase.link(&m_jit);
1353 m_jit.move(TrustedImm32(0), resultGPR);
1354 MacroAssembler::Jump done = m_jit.jump();
1355 trueCase.link(&m_jit);
1356 m_jit.move(TrustedImm32(1), resultGPR);
1357 done.link(&m_jit);
1358
1359 booleanResult(resultGPR, m_currentNode);

1652void SpeculativeJIT::emitBranch(Node* node)
1653{
1654 BasicBlock* taken = node->branchData()->taken.block;
1655 BasicBlock* notTaken = node->branchData()->notTaken.block;
1656
1657 switch (node->child1().useKind()) {
1658 case BooleanUse:
1659 case KnownBooleanUse: {
1660 SpeculateBooleanOperand value(this, node->child1());
1661 MacroAssembler::ResultCondition condition = MacroAssembler::NonZero;
1662
1663 if (taken == nextBlock()) {
1664 condition = MacroAssembler::Zero;
1665 BasicBlock* tmp = taken;
1666 taken = notTaken;
1667 notTaken = tmp;
1668 }
1669
1670 branchTest32(condition, value.gpr(), TrustedImm32(1), taken);
1671 jump(notTaken);
1672
1673 noResult(node);
1674 return;
1675 }
1676
1677 case ObjectOrOtherUse: {
1678 emitObjectOrOtherBranch(node->child1(), taken, notTaken);
1679 return;
1680 }
1681
1682 case StringUse: {
1683 emitStringBranch(node->child1(), taken, notTaken);
1684 return;
1685 }
1686
1687 case StringOrOtherUse: {
1688 emitStringOrOtherBranch(node->child1(), taken, notTaken);
1689 return;
1690 }
1691
1692 case DoubleRepUse:
1693 case Int32Use: {
1694 if (node->child1().useKind() == Int32Use) {
1695 bool invert = false;
1696
1697 if (taken == nextBlock()) {
1698 invert = true;
1699 BasicBlock* tmp = taken;
1700 taken = notTaken;
1701 notTaken = tmp;
1702 }
1703
1704 SpeculateInt32Operand value(this, node->child1());
1705 branchTest32(invert ? MacroAssembler::Zero : MacroAssembler::NonZero, value.gpr(), taken);
1706 } else {
1707 SpeculateDoubleOperand value(this, node->child1());
1708 FPRTemporary scratch(this);
1709 branchDoubleNonZero(value.fpr(), scratch.fpr(), taken);
1710 }
1711
1712 jump(notTaken);
1713
1714 noResult(node);
1715 return;
1716 }
1717
1718 case UntypedUse: {
1719 JSValueOperand value(this, node->child1());
1720 FPRTemporary valueFPR(this);
1721 FPRTemporary tempFPR(this);
1722 GPRTemporary result(this);
1723 GPRTemporary temp(this);
