36enum UseKind {
37 // The DFG has 3 representations of values used:
38
39 // 1. The JSValue representation for a JSValue that must be stored in a GP
40 // register (or a GP register pair), and follows rules for boxing and unboxing
41 // that allow the JSValue to be stored as either fully boxed JSValues, or
42 // unboxed Int32, Booleans, Cells, etc. in 32-bit as appropriate.
43 UntypedUse, // UntypedUse must come first (value 0).
44 Int32Use,
45 KnownInt32Use,
46 AnyIntUse,
47 NumberUse,
48 RealNumberUse,
49 BooleanUse,
50 KnownBooleanUse,
51 CellUse,
52 KnownCellUse,
53 CellOrOtherUse,
54 ObjectUse,
55 ArrayUse,
56 FunctionUse,
57 FinalObjectUse,
58 PromiseObjectUse,
59 RegExpObjectUse,
60 ProxyObjectUse,
61 DerivedArrayUse,
62 ObjectOrOtherUse,
63 StringIdentUse,
64 StringUse,
65 StringOrOtherUse,
66 KnownStringUse,
67 KnownPrimitiveUse, // This bizarre type arises for op_strcat, which has a bytecode guarantee that it will only see primitives (i.e. not objects).
68 SymbolUse,
69 AnyBigIntUse,
70 HeapBigIntUse,
71 BigInt32Use,
72 DateObjectUse,
73 MapObjectUse,
74 SetObjectUse,
75 WeakMapObjectUse,
76 WeakSetObjectUse,
77 DataViewObjectUse,
78 StringObjectUse,
79 StringOrStringObjectUse,
80 NotStringVarUse,
81 NotSymbolUse,
82 NotCellUse,
83 NotCellNorBigIntUse,
84 NotDoubleUse,
85 NeitherDoubleNorHeapBigIntNorStringUse,
86 KnownOtherUse,
87 OtherUse,
88 MiscUse,
89
90 // 2. The Double representation for an unboxed double value that must be stored
91 // in an FP register.
92 DoubleRepUse,
93 DoubleRepRealUse,
94 DoubleRepAnyIntUse,
95
96 // 3. The Int52 representation for an unboxed integer value that must be stored
97 // in a GP register.
98 Int52RepUse,
99
100 LastUseKind // Must always be the last entry in the enum, as it is used to denote the number of enum elements.
101};
102
103inline SpeculatedType typeFilterFor(UseKind useKind)
104{
105 switch (useKind) {
106 case UntypedUse:
107 return SpecBytecodeTop;
108 case Int32Use:
109 case KnownInt32Use:
110 return SpecInt32Only;
111 case Int52RepUse:
112 return SpecInt52Any;
113 case AnyIntUse:
114 return SpecInt32Only | SpecAnyIntAsDouble;
115 case NumberUse:
116 return SpecBytecodeNumber;
117 case RealNumberUse:
118 return SpecBytecodeRealNumber;
119 case DoubleRepUse:
120 return SpecFullDouble;
121 case DoubleRepRealUse:
122 return SpecDoubleReal;
123 case DoubleRepAnyIntUse:
124 return SpecAnyIntAsDouble;
125 case BooleanUse:
126 case KnownBooleanUse:
127 return SpecBoolean;
128 case CellUse:
129 case KnownCellUse:
130 return SpecCellCheck;
131 case CellOrOtherUse:
132 return SpecCellCheck | SpecOther;
133 case ObjectUse:
134 return SpecObject;
135 case ArrayUse:
136 return SpecArray;
137 case FunctionUse:
138 return SpecFunction;
139 case FinalObjectUse:
140 return SpecFinalObject;
141 case RegExpObjectUse:
142 return SpecRegExpObject;
143 case ProxyObjectUse:
144 return SpecProxyObject;
145 case DerivedArrayUse:
146 return SpecDerivedArray;
147 case ObjectOrOtherUse:
148 return SpecObject | SpecOther;
149 case StringIdentUse:
150 return SpecStringIdent;
151 case StringUse:
152 case KnownStringUse:
153 return SpecString;
154 case StringOrOtherUse:
155 return SpecString | SpecOther;
156 case KnownPrimitiveUse:
157 return SpecHeapTop & ~SpecObject;
158 case SymbolUse:
159 return SpecSymbol;
160 case AnyBigIntUse:
161 return SpecBigInt;
162 case HeapBigIntUse:
163 return SpecHeapBigInt;
164 case BigInt32Use:
165 return SpecBigInt32;
166 case PromiseObjectUse:
167 return SpecPromiseObject;
168 case DateObjectUse:
169 return SpecDateObject;
170 case MapObjectUse:
171 return SpecMapObject;
172 case SetObjectUse:
173 return SpecSetObject;
174 case WeakMapObjectUse:
175 return SpecWeakMapObject;
176 case WeakSetObjectUse:
177 return SpecWeakSetObject;
178 case DataViewObjectUse:
179 return SpecDataViewObject;
180 case StringObjectUse:
181 return SpecStringObject;
182 case StringOrStringObjectUse:
183 return SpecString | SpecStringObject;
184 case NotStringVarUse:
185 return ~SpecStringVar;
186 case NotSymbolUse:
187 return ~SpecSymbol;
188 case NotCellUse:
189 return ~SpecCellCheck;
190 case NotCellNorBigIntUse:
191 return ~SpecCellCheck & ~SpecBigInt;
192 case NotDoubleUse:
193 return ~SpecFullDouble;
194 case NeitherDoubleNorHeapBigIntNorStringUse:
195 return ~(SpecFullDouble | SpecHeapBigInt | SpecString);
196 case KnownOtherUse:
197 case OtherUse:
198 return SpecOther;
199 case MiscUse:
200 return SpecMisc;
201 default:
202 RELEASE_ASSERT_NOT_REACHED();
203 return SpecFullTop;
204 }
205}

207inline bool shouldNotHaveTypeCheck(UseKind kind)
208{
209 switch (kind) {
210 case UntypedUse:
211 case KnownInt32Use:
212 case KnownCellUse:
213 case KnownStringUse:
214 case KnownPrimitiveUse:
215 case KnownBooleanUse:
216 case KnownOtherUse:
217 case Int52RepUse:
218 case DoubleRepUse:
219 return true;
220 default:
221 return false;
222 }
223}
224
225inline bool mayHaveTypeCheck(UseKind kind)
226{
227 return !shouldNotHaveTypeCheck(kind);
228}
229
230inline bool isDouble(UseKind kind)
231{
232 switch (kind) {
233 case DoubleRepUse:
234 case DoubleRepRealUse:
235 case DoubleRepAnyIntUse:
236 return true;
237 default:
238 return false;
239 }
240}

242// Returns true if the use kind only admits cells, and is therefore appropriate for
243// SpeculateCellOperand in the DFG or lowCell() in the FTL.
244inline bool isCell(UseKind kind)
245{
246 switch (kind) {
247 case CellUse:
248 case KnownCellUse:
249 case ObjectUse:
250 case ArrayUse:
251 case FunctionUse:
252 case FinalObjectUse:
253 case RegExpObjectUse:
254 case PromiseObjectUse:
255 case ProxyObjectUse:
256 case DerivedArrayUse:
257 case StringIdentUse:
258 case StringUse:
259 case KnownStringUse:
260 case SymbolUse:
261 case HeapBigIntUse:
262 case StringObjectUse:
263 case StringOrStringObjectUse:
264 case DateObjectUse:
265 case MapObjectUse:
266 case SetObjectUse:
267 case WeakMapObjectUse:
268 case WeakSetObjectUse:
269 case DataViewObjectUse:
270 return true;
271 default:
272 return false;
273 }
274}

276// Returns true if we've already guaranteed the type
277inline bool alreadyChecked(UseKind kind, SpeculatedType type)
278{
279 return !(type & ~typeFilterFor(kind));
280}
281
282inline UseKind useKindForResult(NodeFlags result)
283{
284 ASSERT(!(result & ~NodeResultMask));
285 switch (result) {
286 case NodeResultInt52:
287 return Int52RepUse;
288 case NodeResultDouble:
289 return DoubleRepUse;
290 default:
291 return UntypedUse;
292 }
293}
294
295inline bool checkMayCrashIfInputIsEmpty(UseKind kind)
296{
297#if USE(JSVALUE64)
298 switch (kind) {
299 case UntypedUse:
300 case Int32Use:
301 case KnownInt32Use:
302 case AnyIntUse:
303 case NumberUse:
304 case BooleanUse:
305 case KnownBooleanUse:
306 case CellUse:
307 case KnownCellUse:
308 case CellOrOtherUse:
309 case KnownOtherUse:
310 case OtherUse:
311 case MiscUse:
312 case NotCellUse:
313 case NotCellNorBigIntUse:
314 case NotDoubleUse:
315 case NeitherDoubleNorHeapBigIntNorStringUse:
316 return false;
317 default:
318 return true;
319 }
320#else
321 UNUSED_PARAM(kind);
322 return true;
323#endif
324}
