26#pragma once
27
28#include "DOMJITHeapRange.h"
29
30namespace JSC { namespace DOMJIT {
31
32struct Effect {
33
34 constexpr static Effect forWrite(HeapRange writeRange)
35 {
36 return { HeapRange::none(), writeRange };
37 }
38
39 constexpr static Effect forRead(HeapRange readRange)
40 {
41 return { readRange, HeapRange::none() };
42 }
43
44 constexpr static Effect forReadWrite(HeapRange readRange, HeapRange writeRange)
45 {
46 return { readRange, writeRange };
47 }
48
49 constexpr static Effect forPure()
50 {
51 return { HeapRange::none(), HeapRange::none(), HeapRange::none() };
52 }
53
54 constexpr static Effect forDef(HeapRange def)
55 {
56 return { def, HeapRange::none(), def };
57 }
58
59 constexpr static Effect forDef(HeapRange def, HeapRange readRange, HeapRange writeRange)
60 {
61 return { readRange, writeRange, def };
62 }
63
64 constexpr bool mustGenerate() const
65 {
66 return !!writes;
67 }
68
69 HeapRange reads { HeapRange::top() };
70 HeapRange writes { HeapRange::top() };
71 HeapRange def { HeapRange::top() };
72};
73
74} }
