26#pragma once
27
28#if ENABLE(FTL_JIT)
29
30#include "DataFormat.h"
31#include <wtf/PrintStream.h>
32
33namespace JSC { namespace FTL {
34
35struct ExitArgumentRepresentation {
36 DataFormat format;
37 unsigned argument;
38};
39
40class ExitArgument {
41public:
42 ExitArgument()
43 {
44 m_representation.format = DataFormatNone;
45 }
46
47 ExitArgument(DataFormat format, unsigned argument)
48 {
49 m_representation.format = format;
50 m_representation.argument = argument;
51 }
52
53 explicit ExitArgument(ExitArgumentRepresentation representation)
54 {
55 m_representation = representation;
56 }
57
58 bool operator!() const { return m_representation.format == DataFormatNone; }
59
60 DataFormat format() const
61 {
62 ASSERT(*this);
63 return m_representation.format;
64 }
65
66 unsigned argument() const
67 {
68 ASSERT(*this);
69 return m_representation.argument;
70 }
71
72 ExitArgument withFormat(DataFormat format)
73 {
74 return ExitArgument(format, argument());
75 }
76
77 ExitArgumentRepresentation representation() const { return m_representation; }
78
79 void dump(PrintStream&) const;
80
81private:
82 ExitArgumentRepresentation m_representation;
83};
84
85} } // namespace JSC::FTL
86
87#endif // ENABLE(FTL_JIT)
