30#include "DFGPromotedHeapLocation.h"
31#include "FTLExitValue.h"

33namespace JSC {
34
35class TrackedReferences;
36
37namespace FTL {
38
39class ExitPropertyValue {
40public:
41 ExitPropertyValue()
42 {
43 }
44
45 ExitPropertyValue(DFG::PromotedLocationDescriptor location, const ExitValue& value)
46 : m_location(location)
47 , m_value(value)
48 {
49 ASSERT(!!location == !!value);
50 }
51
52 bool operator!() const { return !m_location; }
53
54 DFG::PromotedLocationDescriptor location() const { return m_location; }
55 const ExitValue& value() const { return m_value; }
56
57 ExitPropertyValue withLocalsOffset(int offset) const;
58
59 void dump(PrintStream& out) const;
60
61 void validateReferences(const TrackedReferences&) const;
62
63private:
64 DFG::PromotedLocationDescriptor m_location;
65 ExitValue m_value;
66};
67
68} } // namespace JSC::FTL
