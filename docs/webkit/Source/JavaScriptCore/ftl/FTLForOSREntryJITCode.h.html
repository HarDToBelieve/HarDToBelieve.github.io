33namespace JSC { namespace FTL {
34
35// OSR entry into the FTL has a number of quirks:
36//
37// - OSR entry only happens through special OSR entry compilations. They have their
38// own CodeBlock and their own JITCode.
39//
40// - We only OSR enter in loop headers that have a null inline call frame.
41//
42// - Each OSR entry compilation allows entry through only one bytecode index.
43
44class ForOSREntryJITCode final : public FTL::JITCode {
45public:
46 ForOSREntryJITCode();
47 ~ForOSREntryJITCode() final;
48
49 void initializeEntryBuffer(VM&, unsigned numCalleeLocals);
50 ScratchBuffer* entryBuffer() const { return m_entryBuffer; }
51
52 void setBytecodeIndex(BytecodeIndex value) { m_bytecodeIndex = value; }
53 BytecodeIndex bytecodeIndex() const { return m_bytecodeIndex; }
54
55 void countEntryFailure() { m_entryFailureCount++; }
56 unsigned entryFailureCount() const { return m_entryFailureCount; }
57
58 ForOSREntryJITCode* ftlForOSREntry() final;
59
60 void setArgumentFlushFormats(FixedVector<DFG::FlushFormat>&& argumentFlushFormats)
61 {
62 m_argumentFlushFormats = WTFMove(argumentFlushFormats);
63 }
64 const FixedVector<DFG::FlushFormat>& argumentFlushFormats() { return m_argumentFlushFormats; }
65
66private:
67 FixedVector<DFG::FlushFormat> m_argumentFlushFormats;
68 ScratchBuffer* m_entryBuffer; // Only for OSR entry code blocks.
69 BytecodeIndex m_bytecodeIndex;
70 unsigned m_entryFailureCount;
71};
72
73} } // namespace JSC::FTL
74
75#endif // ENABLE(FTL_JIT)
