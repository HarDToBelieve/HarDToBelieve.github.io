35// This is used for creating some sanity in slow-path calls out of the FTL's inline
36// caches. The idea is that we don't want all of the register save/restore stuff to
37// be generated at each IC site. Instead, the IC slow path call site will just save
38// the registers needed for the arguments. It will arrange for there to be enough
39// space on top of stack to save the remaining registers and the return PC. Then it
40// will call a shared thunk that will save the remaining registers. That thunk needs
41// to know the stack offset at which things get saved along with the call target.
42
43// Note that the offset is *not including* the return PC that would be pushed on X86.
44
45class SlowPathCallKey {
46public:
47 SlowPathCallKey()
48 : m_offset(0)
49 {
50 }
51
52 SlowPathCallKey(
53 const RegisterSet& set, FunctionPtr<CFunctionPtrTag> callTarget, const RegisterSet& argumentRegisters,
54 ptrdiff_t offset, int32_t indirectOffset)
55 : m_usedRegisters(set)
56 , m_callTarget(callTarget.retagged<OperationPtrTag>())
57 , m_argumentRegisters(argumentRegisters)
58 , m_offset(offset)
59 , m_indirectOffset(indirectOffset)
60 {
61 }
62
63 const RegisterSet& usedRegisters() const { return m_usedRegisters; }
64 FunctionPtr<OperationPtrTag> callTarget() const { return m_callTarget; }
65 const RegisterSet& argumentRegisters() const { return m_argumentRegisters; }
66 ptrdiff_t offset() const { return m_offset; }
67 int32_t indirectOffset() const { return m_indirectOffset; }
68
69 SlowPathCallKey withCallTarget(FunctionPtr<CFunctionPtrTag> callTarget)
70 {
71 return SlowPathCallKey(usedRegisters(), callTarget, argumentRegisters(), offset(), indirectOffset());
72 }
73
74 void dump(PrintStream&) const;
75
76 enum EmptyValueTag { EmptyValue };
77 enum DeletedValueTag { DeletedValue };
78
79 SlowPathCallKey(EmptyValueTag)
80 : m_usedRegisters(HashableRegisterSet::EmptyValue)
81 {
82 }
83
84 SlowPathCallKey(DeletedValueTag)
85 : m_usedRegisters(HashableRegisterSet::DeletedValue)
86 {
87 }
88
89 bool isEmptyValue() const { return m_usedRegisters.isEmptyValue(); }
90 bool isDeletedValue() const { return m_usedRegisters.isDeletedValue(); }
91
92 bool operator==(const SlowPathCallKey& other) const
93 {
94 return m_usedRegisters == other.m_usedRegisters
95 && m_callTarget == other.m_callTarget
96 && m_offset == other.m_offset
97 && m_indirectOffset == other.m_indirectOffset;
98 }
99 unsigned hash() const
100 {
101 return m_usedRegisters.hash() + PtrHash<void*>::hash(m_callTarget.executableAddress()) + m_offset + m_indirectOffset;
102 }
103
104private:
105 HashableRegisterSet m_usedRegisters;
106 FunctionPtr<OperationPtrTag> m_callTarget;
107 RegisterSet m_argumentRegisters;
108 ptrdiff_t m_offset { 0 };
109 int32_t m_indirectOffset { 0 };
110};
111
112struct SlowPathCallKeyHash {
113 static unsigned hash(const SlowPathCallKey& key) { return key.hash(); }
114 static bool equal(const SlowPathCallKey& a, const SlowPathCallKey& b) { return a == b; }
115 static constexpr bool safeToCompareToEmptyOrDeleted = false;
116};
