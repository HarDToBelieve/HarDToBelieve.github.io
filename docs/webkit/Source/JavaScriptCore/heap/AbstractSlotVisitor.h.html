53 WTF_MAKE_NONCOPYABLE(AbstractSlotVisitor);
54 WTF_MAKE_FAST_ALLOCATED;
55public:
56 enum OpaqueRootTag { OpaqueRoot };

58 class ReferrerToken {
59 public:
60 ReferrerToken() = default;
61 ReferrerToken(std::nullptr_t) { }
62 ReferrerToken(HeapCell*);
63 ReferrerToken(OpaqueRootTag, void* opaqueRoot);
64 ReferrerToken(RootMarkReason);
65
66 explicit operator bool() const { return m_bits; }
67 bool operator!() const { return !m_bits; }
68
69 HeapCell* asCell() const;
70 void* asOpaqueRoot() const;
71 RootMarkReason asRootMarkReason() const;
72
73 private:
74 enum {
75 HeapCellToken = 0,
76 OpaqueRootToken = 0b01,
77 RootMarkReasonToken = 0b10,
78 };
79
80 static constexpr uintptr_t tokenTypeShift = 2;
81 static constexpr uintptr_t tokenTypeMask = 0x3;
82
83 bool isHeapCell() const { return (m_bits & tokenTypeMask) == HeapCellToken; }
84 bool isOpaqueRoot() const { return (m_bits & tokenTypeMask) == OpaqueRootToken; }
85 bool isRootMarkReason() const { return (m_bits & tokenTypeMask) == RootMarkReasonToken; }
86 uintptr_t m_bits { 0 };
87 };
88

90 public:

92 ReferrerContext(AbstractSlotVisitor&, OpaqueRootTag);
93 ~ReferrerContext();

95 ReferrerToken referrer() const { return m_referrer; }
96 void setReferrer(ReferrerToken referrer)
97 {
98 ASSERT(!m_referrer);
99 m_referrer = referrer;
100 }
101
102 bool isOpaqueRootContext() const { return m_isOpaqueRootContext; }
103
104 private:
105 AbstractSlotVisitor& m_visitor;
106 ReferrerToken m_referrer;
107 ReferrerContext* m_previous;
108 bool m_isOpaqueRootContext { false };
109 };

144 template<typename T, typename Traits> void append(const WriteBarrierBase<T, Traits>&);
145 template<typename T, typename Traits> void appendHidden(const WriteBarrierBase<T, Traits>&);
146 template<typename Iterator> void append(Iterator begin , Iterator end);
147 ALWAYS_INLINE void appendValues(const WriteBarrierBase<Unknown, RawValueTraits<Unknown>>*, size_t count);
148 ALWAYS_INLINE void appendValuesHidden(const WriteBarrierBase<Unknown, RawValueTraits<Unknown>>*, size_t count);

150 // These don't require you to prove that you have a WriteBarrier<>. That makes sense
151 // for:
152 //
153 // - roots.
154 // - sophisticated data structures that barrier through other means (like DFG::Plan and
155 // friends).
156 //
157 // If you are not a root and you don't know what kind of barrier you have, then you
158 // shouldn't call these methods.

237template<typename Visitor>
238class SetRootMarkReasonScope {
239public:
240 SetRootMarkReasonScope(Visitor& visitor, RootMarkReason reason)
241 : m_visitor(visitor)
242 , m_previousReason(visitor.rootMarkReason())
243 , m_context(visitor, reason)
244 {
245 m_visitor.setRootMarkReason(reason);
246 }
247
248 ~SetRootMarkReasonScope()
249 {
250 m_visitor.setRootMarkReason(m_previousReason);
251 }
252
253private:
254 Visitor& m_visitor;
255 RootMarkReason m_previousReason;
256 typename Visitor::ReferrerContext m_context;
257};
