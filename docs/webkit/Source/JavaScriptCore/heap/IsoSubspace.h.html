33namespace JSC {
34
35class IsoAlignedMemoryAllocator;
36class IsoCellSet;
37
38class IsoSubspace : public Subspace {
39public:
40 JS_EXPORT_PRIVATE IsoSubspace(CString name, Heap&, HeapCellType*, size_t size, uint8_t numberOfLowerTierCells);
41 JS_EXPORT_PRIVATE ~IsoSubspace() override;
42
43 size_t cellSize() { return m_directory.cellSize(); }
44
45 Allocator allocatorFor(size_t, AllocatorForMode) override;
46 Allocator allocatorForNonVirtual(size_t, AllocatorForMode);
47
48 void* allocate(VM&, size_t, GCDeferralContext*, AllocationFailureMode) override;
49 void* allocateNonVirtual(VM&, size_t, GCDeferralContext*, AllocationFailureMode);
50
51 void sweepLowerTierCell(PreciseAllocation*);
52 void clearIsoCellSetBit(PreciseAllocation*);
53
54 void* tryAllocateFromLowerTier();
55 void destroyLowerTierFreeList();
56
57 void sweep();
58
59 template<typename Func> void forEachLowerTierFreeListedPreciseAllocation(const Func&);
60
61private:
62 friend class IsoCellSet;
63
64 void didResizeBits(unsigned newSize) override;
65 void didRemoveBlock(unsigned blockIndex) override;
66 void didBeginSweepingToFreeList(MarkedBlock::Handle*) override;
67
68 BlockDirectory m_directory;
69 LocalAllocator m_localAllocator;
70 std::unique_ptr<IsoAlignedMemoryAllocator> m_isoAlignedMemoryAllocator;
71 SentinelLinkedList<PreciseAllocation, PackedRawSentinelNode<PreciseAllocation>> m_lowerTierFreeList;
72 SentinelLinkedList<IsoCellSet, PackedRawSentinelNode<IsoCellSet>> m_cellSets;
73};
74
75ALWAYS_INLINE Allocator IsoSubspace::allocatorForNonVirtual(size_t size, AllocatorForMode)
76{
77 RELEASE_ASSERT(WTF::roundUpToMultipleOf<MarkedBlock::atomSize>(size) == cellSize());
78 return Allocator(&m_localAllocator);
79}
80
81#define ISO_SUBSPACE_INIT(heap, heapCellType, type) ("Isolated " #type " Space", (heap), (heapCellType), sizeof(type), type::numberOfLowerTierCells)
82
83template<typename T>
84struct isAllocatedFromIsoSubspace {
85 static constexpr bool value = std::is_same<std::result_of_t<decltype(T::template subspaceFor<T, SubspaceAccess::OnMainThread>)&(VM&)>, IsoSubspace*>::value;
86};
