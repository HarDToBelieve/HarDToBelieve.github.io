33namespace JSC {

35class AbstractSlotVisitor;
36class Heap;
37class SlotVisitor;

39DECLARE_ALLOCATOR_WITH_HEAP_IDENTIFIER(WeakBlock);
40
41class WeakBlock : public DoublyLinkedListNode<WeakBlock> {
42public:
43 friend class WTF::DoublyLinkedListNode<WeakBlock>;
44 static constexpr size_t blockSize = 256; // 1/16 of MarkedBlock size
45
46 struct FreeCell {
47 FreeCell* next;
48 };
49
50 struct SweepResult {
51 bool isNull() const;
52
53 bool blockIsFree { true };
54 bool blockIsLogicallyEmpty { true };
55 FreeCell* freeList { nullptr };
56 };
57
58 static WeakBlock* create(Heap&, CellContainer);
59 static void destroy(Heap&, WeakBlock*);
60
61 static WeakImpl* asWeakImpl(FreeCell*);
62
63 bool isEmpty();
64 bool isLogicallyEmptyButNotFree() const;
65
66 void sweep();
67 SweepResult takeSweepResult();
68
69 JS_EXPORT_PRIVATE void visit(AbstractSlotVisitor&);
70 JS_EXPORT_PRIVATE void visit(SlotVisitor&);
71
72 void reap();
73
74 void lastChanceToFinalize();
75 void disconnectContainer() { m_container = CellContainer(); }
76
77private:
78 template<typename Visitor> void visitImpl(Visitor&);
79
80 static FreeCell* asFreeCell(WeakImpl*);
81
82 template<typename ContainerType, typename Visitor>
83 void specializedVisit(ContainerType&, Visitor&);
84
85 explicit WeakBlock(CellContainer);
86 void finalize(WeakImpl*);
87 WeakImpl* weakImpls();
88 size_t weakImplCount();
89 void addToFreeList(FreeCell**, WeakImpl*);
90
91 CellContainer m_container;
92 WeakBlock* m_prev;
93 WeakBlock* m_next;
94 SweepResult m_sweepResult;
95};
96
97inline bool WeakBlock::SweepResult::isNull() const
98{
99 return blockIsFree && !freeList; // This state is impossible, so we can use it to mean null.
100}
101
102inline WeakImpl* WeakBlock::asWeakImpl(FreeCell* freeCell)
103{
104 return reinterpret_cast_ptr<WeakImpl*>(freeCell);
105}
106
107inline WeakBlock::SweepResult WeakBlock::takeSweepResult()
108{
109 SweepResult tmp;
110 std::swap(tmp, m_sweepResult);
111 ASSERT(m_sweepResult.isNull());
112 return tmp;
113}
114
115inline WeakBlock::FreeCell* WeakBlock::asFreeCell(WeakImpl* weakImpl)
116{
117 return reinterpret_cast_ptr<FreeCell*>(weakImpl);
118}
119
120inline WeakImpl* WeakBlock::weakImpls()
121{
122 return reinterpret_cast_ptr<WeakImpl*>(this) + ((sizeof(WeakBlock) + sizeof(WeakImpl) - 1) / sizeof(WeakImpl));
123}
124
125inline size_t WeakBlock::weakImplCount()
126{
127 return (blockSize / sizeof(WeakImpl)) - ((sizeof(WeakBlock) + sizeof(WeakImpl) - 1) / sizeof(WeakImpl));
128}
129
130inline void WeakBlock::addToFreeList(FreeCell** freeList, WeakImpl* weakImpl)
131{
132 ASSERT(weakImpl->state() == WeakImpl::Deallocated);
133 FreeCell* freeCell = asFreeCell(weakImpl);
134 ASSERT(!*freeList || ((char*)*freeList > (char*)this && (char*)*freeList < (char*)this + blockSize));
135 ASSERT((char*)freeCell > (char*)this && (char*)freeCell < (char*)this + blockSize);
136 freeCell->next = *freeList;
137 *freeList = freeCell;
138}
139
140inline bool WeakBlock::isEmpty()
141{
142 return !m_sweepResult.isNull() && m_sweepResult.blockIsFree;
143}
144
145inline bool WeakBlock::isLogicallyEmptyButNotFree() const
146{
147 return !m_sweepResult.isNull() && !m_sweepResult.blockIsFree && m_sweepResult.blockIsLogicallyEmpty;
148}
