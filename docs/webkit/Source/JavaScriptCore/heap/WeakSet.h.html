26#pragma once
27
28#include "CellContainer.h"
29#include "WeakBlock.h"
30#include <wtf/SentinelLinkedList.h>
31
32namespace JSC {
33
34class Heap;
35class WeakImpl;
36
37class WeakSet : public BasicRawSentinelNode<WeakSet> {
38 friend class LLIntOffsetsExtractor;
39
40public:
41 static WeakImpl* allocate(JSValue, WeakHandleOwner* = nullptr, void* context = nullptr);
42 static void deallocate(WeakImpl*);
43
44 WeakSet(VM&);
45 ~WeakSet();
46 void lastChanceToFinalize();
47
48 Heap* heap() const;
49 VM& vm() const;
50
51 bool isEmpty() const;
52 bool isTriviallyDestructible() const;
53
54 template<typename Visitor> void visit(Visitor&);
55
56 void reap();
57 void sweep();
58 void shrink();
59 void resetAllocator();
60
61 static ptrdiff_t offsetOfVM() { return OBJECT_OFFSETOF(WeakSet, m_vm); }
62
63private:
64 JS_EXPORT_PRIVATE WeakBlock::FreeCell* findAllocator(CellContainer);
65 WeakBlock::FreeCell* tryFindAllocator();
66 WeakBlock::FreeCell* addAllocator(CellContainer);
67 void removeAllocator(WeakBlock*);
68
69 WeakBlock::FreeCell* m_allocator { nullptr };
70 WeakBlock* m_nextAllocator { nullptr };
71 DoublyLinkedList<WeakBlock> m_blocks;
72 // m_vm must be a pointer (instead of a reference) because the JSCLLIntOffsetsExtractor
73 // cannot handle it being a reference.
74 VM* m_vm;
75};
76
77inline WeakSet::WeakSet(VM& vm)
78 : m_vm(&vm)
79{
80}
81
82inline VM& WeakSet::vm() const
83{
84 return *m_vm;
85}
86
87inline bool WeakSet::isEmpty() const
88{
89 for (WeakBlock* block = m_blocks.head(); block; block = block->next()) {
90 if (!block->isEmpty())
91 return false;
92 }
93
94 return true;
95}
96
97inline bool WeakSet::isTriviallyDestructible() const
98{
99 if (!m_blocks.isEmpty())
100 return false;
101 if (isOnList())
102 return false;
103 return true;
104}
105
106inline void WeakSet::deallocate(WeakImpl* weakImpl)
107{
108 weakImpl->setState(WeakImpl::Deallocated);
109}
110
111inline void WeakSet::lastChanceToFinalize()
112{
113 for (WeakBlock* block = m_blocks.head(); block; block = block->next())
114 block->lastChanceToFinalize();
115}
116
117template<typename Visitor>
118inline void WeakSet::visit(Visitor& visitor)
119{
120 for (WeakBlock* block = m_blocks.head(); block; block = block->next())
121 block->visit(visitor);
122}
123
124inline void WeakSet::reap()
125{
126 for (WeakBlock* block = m_blocks.head(); block; block = block->next())
127 block->reap();
128}
129
130inline void WeakSet::resetAllocator()
131{
132 m_allocator = nullptr;
133 m_nextAllocator = m_blocks.head();
134}
135
136} // namespace JSC
