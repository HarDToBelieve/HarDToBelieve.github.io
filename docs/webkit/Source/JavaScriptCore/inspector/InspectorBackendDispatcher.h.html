58 CallbackBase(Ref<BackendDispatcher>&&, long requestId);

60 bool isActive() const;

66 private:

72 bool isActive() const;

74 bool hasProtocolErrors() const { return m_protocolErrors.size() > 0; }

78 InvalidRequest,

80 InvalidParams,
81 InternalError,
82 ServerError

85 void registerDispatcherForDomain(const String& domain, SupplementalBackendDispatcher*);

88 // Note that 'unused' is a workaround so the compiler can pick the right sendResponse based on arity.

103 String getString(JSON::Object*, const String& name, bool required);
104 RefPtr<JSON::Value> getValue(JSON::Object*, const String& name, bool required);
105 RefPtr<JSON::Object> getObject(JSON::Object*, const String& name, bool required);

107
108private:
109 BackendDispatcher(Ref<FrontendRouter>&&);

114 Ref<FrontendRouter> m_frontendRouter;
115 HashMap<String, SupplementalBackendDispatcher*> m_dispatchers;
116
117 // Protocol errors reported for the top-level request being processed.
118 // If processing a request triggers async responses, then any related errors will
119 // be attributed to the top-level request, but generate separate error messages.
120 Vector<std::tuple<CommonErrorCode, String>> m_protocolErrors;
121
122 // For synchronously handled requests, avoid plumbing requestId through every
123 // call that could potentially fail with a protocol error.
124 std::optional<long> m_currentRequestId { std::nullopt };
125};
