26#pragma once
27
28#include "JSCJSValue.h"
29#include <wtf/StdLibExtras.h>
30
31namespace JSC {
32
33namespace Wasm {
34class Callee;
35}
36
37class JSCell;
38
39class CalleeBits {
40public:
41 CalleeBits() = default;
42 CalleeBits(void* ptr) : m_ptr(ptr) { }
43
44 CalleeBits& operator=(JSCell* cell)
45 {
46 m_ptr = cell;
47 ASSERT(isCell());
48 return *this;
49 }
50
51#if ENABLE(WEBASSEMBLY)
52 static void* boxWasm(Wasm::Callee* callee)
53 {
54 CalleeBits result(reinterpret_cast<void*>(reinterpret_cast<uintptr_t>(callee) | JSValue::WasmTag));
55 ASSERT(result.isWasm());
56 return result.rawPtr();
57 }
58#endif
59
60 bool isWasm() const
61 {
62#if ENABLE(WEBASSEMBLY)
63 return (reinterpret_cast<uintptr_t>(m_ptr) & JSValue::WasmMask) == JSValue::WasmTag;
64#else
65 return false;
66#endif
67 }
68 bool isCell() const { return !isWasm(); }
69
70 JSCell* asCell() const
71 {
72 ASSERT(!isWasm());
73 return static_cast<JSCell*>(m_ptr);
74 }
75
76#if ENABLE(WEBASSEMBLY)
77 Wasm::Callee* asWasmCallee() const
78 {
79 ASSERT(isWasm());
80 return reinterpret_cast<Wasm::Callee*>(reinterpret_cast<uintptr_t>(m_ptr) & ~JSValue::WasmTag);
81 }
82#endif
83
84 void* rawPtr() const { return m_ptr; }
85
86private:
87 void* m_ptr { nullptr };
88};
89
90} // namespace JSC
