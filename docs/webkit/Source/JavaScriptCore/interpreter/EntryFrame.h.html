28#include "StackAlignment.h"
29#include "VMEntryRecord.h"

31namespace JSC {
32
33struct EntryFrame {
34#if !ENABLE(C_LOOP) && NUMBER_OF_CALLEE_SAVES_REGISTERS > 0
35 static ptrdiff_t vmEntryRecordOffset()
36 {
37 EntryFrame* fakeEntryFrame = reinterpret_cast<EntryFrame*>(0x1000);
38 VMEntryRecord* record = vmEntryRecord(fakeEntryFrame);
39 return static_cast<ptrdiff_t>(
40 reinterpret_cast<char*>(record) - reinterpret_cast<char*>(fakeEntryFrame));
41 }
42
43 static ptrdiff_t calleeSaveRegistersBufferOffset()
44 {
45 return vmEntryRecordOffset() + OBJECT_OFFSETOF(VMEntryRecord, calleeSaveRegistersBuffer);
46 }
47#endif
48};
49
50} // namespace JSC
