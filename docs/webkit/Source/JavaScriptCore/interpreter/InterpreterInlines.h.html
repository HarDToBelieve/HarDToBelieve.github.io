48// This function is only available as a debugging tool for development work.
49// It is not currently used except in a RELEASE_ASSERT to ensure that it is
50// working properly.
51inline OpcodeID Interpreter::getOpcodeID(Opcode opcode)
52{
53#if ENABLE(COMPUTED_GOTO_OPCODES)
54 ASSERT(isOpcode(opcode));
55#if ENABLE(LLINT_EMBEDDED_OPCODE_ID)
56 // The OpcodeID is embedded in the int32_t word preceding the location of
57 // the LLInt code for the opcode (see the EMBED_OPCODE_ID_IF_NEEDED macro
58 // in LowLevelInterpreter.cpp).
59 const void* opcodeAddress = removeCodePtrTag(bitwise_cast<const void*>(opcode));
60 const int32_t* opcodeIDAddress = bitwise_cast<int32_t*>(opcodeAddress) - 1;
61 OpcodeID opcodeID = static_cast<OpcodeID>(WTF::unalignedLoad<int32_t>(opcodeIDAddress));
62 ASSERT(opcodeID < NUMBER_OF_BYTECODE_IDS);
63 return opcodeID;
64#else
65 return opcodeIDTable().get(opcode);
66#endif // ENABLE(LLINT_EMBEDDED_OPCODE_ID)
67
68#else // not ENABLE(COMPUTED_GOTO_OPCODES)
69 return opcode;
70#endif
71}

73ALWAYS_INLINE JSValue Interpreter::execute(CallFrameClosure& closure)
74{
75 VM& vm = *closure.vm;
76 auto throwScope = DECLARE_THROW_SCOPE(vm);
77
78 ASSERT(!vm.isCollectorBusyOnCurrentThread());
79 ASSERT(vm.currentThreadIsHoldingAPILock());
80
81 StackStats::CheckPoint stackCheckPoint;
82
83 if (UNLIKELY(vm.traps().needHandling(VMTraps::NonDebuggerAsyncEvents))) {
84 ASSERT(vm.topCallFrame == closure.oldCallFrame);
85 if (vm.hasExceptionsAfterHandlingTraps())
86 return throwScope.exception();
87 }
88
89 // Reload CodeBlock since GC can replace CodeBlock owned by Executable.
90 CodeBlock* codeBlock;
91 Exception* error = closure.functionExecutable->prepareForExecution<FunctionExecutable>(vm, closure.function, closure.scope, CodeForCall, codeBlock);
92 EXCEPTION_ASSERT(throwScope.exception() == error);
93 if (UNLIKELY(error))
94 return checkedReturn(error);
95 codeBlock->m_shouldAlwaysBeInlined = false;
96 {
97 DisallowGC disallowGC; // Ensure no GC happens. GC can replace CodeBlock in Executable.
98 closure.protoCallFrame->setCodeBlock(codeBlock);
99 }
100 // Execute the code:
101 throwScope.release();
102 JSValue result = closure.functionExecutable->generatedJITCodeForCall()->execute(&vm, closure.protoCallFrame);
103
104 return checkedReturn(result);
105}
