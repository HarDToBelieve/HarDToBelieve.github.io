454 // dest = base + index << shift.
455 void shiftAndAdd(RegisterID base, RegisterID index, uint8_t shift, RegisterID dest, std::optional<RegisterID> optionalScratch = { })
456 {
457 ASSERT(shift < 32);
458 if (shift <= 3) {
459 x86Lea64(BaseIndex(base, index, static_cast<Scale>(shift)), dest);
460 return;
461 }
462
463 RegisterID scratch = dest;
464 bool needToPreserveIndexRegister = false;
465 if (base == dest) {
466 scratch = optionalScratch ? optionalScratch.value() : scratchRegister();
467 if (base == scratch) {
468 scratch = index;
469 needToPreserveIndexRegister = true;
470 } else if (index == scratch)
471 needToPreserveIndexRegister = true;
472 if (needToPreserveIndexRegister)
473 push(index);
474 }
475
476 move(index, scratch);
477 lshift64(TrustedImm32(shift), scratch);
478 m_assembler.leaq_mr(0, base, scratch, 0, dest);
479
480 if (needToPreserveIndexRegister)
481 pop(index);
482 }
483

601 JumpList branchIfNotEqual(JSValueRegs regs, JSValue value)
602 {
603#if USE(JSVALUE64)
604 return branch64(NotEqual, regs.gpr(), TrustedImm64(JSValue::encode(value)));
605#else
606 JumpList result;
607 result.append(branch32(NotEqual, regs.tagGPR(), TrustedImm32(value.tag())));
608 if (value.isEmpty() || value.isUndefinedOrNull())
609 return result; // These don't have anything interesting in the payload.
610 result.append(branch32(NotEqual, regs.payloadGPR(), TrustedImm32(value.payload())));
611 return result;
612#endif
613 }
614
615 Jump branchIfEqual(JSValueRegs regs, JSValue value)
616 {
617#if USE(JSVALUE64)
618 return branch64(Equal, regs.gpr(), TrustedImm64(JSValue::encode(value)));
619#else
620 Jump notEqual;
621 // These don't have anything interesting in the payload.
622 if (!value.isEmpty() && !value.isUndefinedOrNull())
623 notEqual = branch32(NotEqual, regs.payloadGPR(), TrustedImm32(value.payload()));
624 Jump result = branch32(Equal, regs.tagGPR(), TrustedImm32(value.tag()));
625 if (notEqual.isSet())
626 notEqual.link(this);
627 return result;
628#endif
629 }

1624 // Emits the branch structure for typeof. The code emitted by this doesn't fall through. The
1625 // functor is called at those points where we have pinpointed a type. One way to use this is to
1626 // have the functor emit the code to put the type string into an appropriate register and then
1627 // jump out. A secondary functor is used for the call trap and masquerades-as-undefined slow
1628 // case. It is passed the unlinked jump to the slow case.
1629 template<typename Functor, typename SlowPathFunctor>
1630 void emitTypeOf(
1631 JSValueRegs regs, GPRReg tempGPR, const Functor& functor,
1632 const SlowPathFunctor& slowPathFunctor)
1633 {
1634 // Implements the following branching structure:
1635 //
1636 // if (is cell) {
1637 // if (is object) {
1638 // if (is function) {
1639 // return function;
1640 // } else if (doesn't have call trap and doesn't masquerade as undefined) {
1641 // return object
1642 // } else {
1643 // return slowPath();
1644 // }
1645 // } else if (is string) {
1646 // return string
1647 // } else if (is heapbigint) {
1648 // return bigint
1649 // } else {
1650 // return symbol
1651 // }
1652 // } else if (is number) {
1653 // return number
1654 // } else if (is null) {
1655 // return object
1656 // } else if (is boolean) {
1657 // return boolean
1658 // } else if (is bigint32) {
1659 // return bigint
1660 // } else {
1661 // return undefined
1662 // }
1663 //
1664 // FIXME: typeof Symbol should be more frequently seen than BigInt.
1665 // We should change the order of type detection based on this frequency.
1666 // https://bugs.webkit.org/show_bug.cgi?id=192650
1667
1668 Jump notCell = branchIfNotCell(regs);
1669
1670 GPRReg cellGPR = regs.payloadGPR();
1671 Jump notObject = branchIfNotObject(cellGPR);
1672
1673 Jump notFunction = branchIfNotFunction(cellGPR);
1674 functor(TypeofType::Function, false);
1675
1676 notFunction.link(this);
1677 slowPathFunctor(
1678 branchTest8(
1679 NonZero,
1680 Address(cellGPR, JSCell::typeInfoFlagsOffset()),
1681 TrustedImm32(MasqueradesAsUndefined | OverridesGetCallData)));
1682 functor(TypeofType::Object, false);
1683
1684 notObject.link(this);
1685
1686 Jump notString = branchIfNotString(cellGPR);
1687 functor(TypeofType::String, false);
1688
1689 notString.link(this);
1690
1691 Jump notHeapBigInt = branchIfNotHeapBigInt(cellGPR);
1692 functor(TypeofType::BigInt, false);
1693
1694 notHeapBigInt.link(this);
1695 functor(TypeofType::Symbol, false);
1696
1697 notCell.link(this);
1698
1699 Jump notNumber = branchIfNotNumber(regs, tempGPR);
1700 functor(TypeofType::Number, false);
1701 notNumber.link(this);
1702
1703 JumpList notNull = branchIfNotEqual(regs, jsNull());
1704 functor(TypeofType::Object, false);
1705 notNull.link(this);
1706
1707 Jump notBoolean = branchIfNotBoolean(regs, tempGPR);
1708 functor(TypeofType::Boolean, false);
1709 notBoolean.link(this);
1710
1711#if USE(BIGINT32)
1712 Jump notBigInt32 = branchIfNotBigInt32(regs, tempGPR);
1713 functor(TypeofType::BigInt, false);
1714 notBigInt32.link(this);
1715#endif
1716
1717 functor(TypeofType::Undefined, true);
1718 }
