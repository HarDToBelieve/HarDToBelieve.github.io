47 // Any register that has been locked or acquired must be released
48 // before calling prepareForTailCall() or prepareForSlowPath().
49 void lockGPR(GPRReg gpr)
50 {
51 ASSERT(!m_lockedRegisters.get(gpr));
52 m_lockedRegisters.set(gpr);
53 if (verbose)
54 dataLog(" * Locking ", gpr, "\n");
55 }
56
57 GPRReg acquireGPR()
58 {
59 ensureGPR();
60 GPRReg gpr { getFreeGPR() };
61 ASSERT(!m_registers[gpr]);
62 lockGPR(gpr);
63 return gpr;
64 }
65
66 void releaseGPR(GPRReg gpr)
67 {
68 if (verbose) {
69 if (m_lockedRegisters.get(gpr))
70 dataLog(" * Releasing ", gpr, "\n");
71 else
72 dataLog(" * ", gpr, " was not locked\n");
73 }
74 m_lockedRegisters.clear(gpr);
75 }
76
77 void restoreGPR(GPRReg gpr)
78 {
79 if (!m_newRegisters[gpr])
80 return;
81
82 ensureGPR();
83#if USE(JSVALUE32_64)
84 GPRReg tempGPR { getFreeGPR() };
85 lockGPR(tempGPR);
86 ensureGPR();
87 releaseGPR(tempGPR);
88#endif
89 emitDisplace(*m_newRegisters[gpr]);
90 }

92 // You can only take a snapshot if the recovery has not started
93 // yet. The only operations that are valid before taking a
94 // snapshot are lockGPR(), acquireGPR() and releaseGPR().
95 //
96 // Locking status is *NOT* preserved by the snapshot: it only
97 // contains information about where the
98 // arguments/callee/callee-save registers are by taking into
99 // account any spilling that acquireGPR() could have done.
100 CallFrameShuffleData snapshot() const
101 {
102 ASSERT(isUndecided());
103
104 CallFrameShuffleData data;
105 data.numLocals = numLocals();
106 data.numPassedArgs = m_numPassedArgs;
107 data.callee = getNew(VirtualRegister { CallFrameSlot::callee })->recovery();
108 data.args.resize(argCount());
109 for (size_t i = 0; i < argCount(); ++i)
110 data.args[i] = getNew(virtualRegisterForArgumentIncludingThis(i))->recovery();
111 for (Reg reg = Reg::first(); reg <= Reg::last(); reg = reg.next()) {
112 CachedRecovery* cachedRecovery { m_newRegisters[reg] };
113 if (!cachedRecovery)
114 continue;
115
116#if USE(JSVALUE64)
117 data.registers[reg] = cachedRecovery->recovery();
118#else
119 RELEASE_ASSERT_NOT_REACHED();
120#endif
121 }
122 return data;
123 }

125 // Ask the shuffler to put the callee into some registers once the
126 // shuffling is done. You should call this before any of the
127 // prepare() methods, and must not take a snapshot afterwards, as
128 // this would crash 32bits platforms.
129 void setCalleeJSValueRegs(JSValueRegs jsValueRegs)
130 {
131 ASSERT(isUndecided());
132 ASSERT(!getNew(jsValueRegs));
133 CachedRecovery* cachedRecovery { getNew(CallFrameSlot::callee) };
134 ASSERT(cachedRecovery);
135 addNew(jsValueRegs, cachedRecovery->recovery());
136 }

138 // Ask the suhffler to assume the callee has already be checked to
139 // be a cell. This is a no-op on 64bit platforms, but allows to
140 // free up a GPR on 32bit platforms.
141 // You obviously must have ensured that this is the case before
142 // running any of the prepare methods.
143 void assumeCalleeIsCell()
144 {
145#if USE(JSVALUE32_64)
146 CachedRecovery& calleeCachedRecovery = *getNew(CallFrameSlot::callee);
147 switch (calleeCachedRecovery.recovery().technique()) {
148 case InPair:
149 updateRecovery(
150 calleeCachedRecovery,
151 ValueRecovery::inGPR(
152 calleeCachedRecovery.recovery().payloadGPR(),
153 DataFormatCell));
154 break;
155 case DisplacedInJSStack:
156 updateRecovery(
157 calleeCachedRecovery,
158 ValueRecovery::displacedInJSStack(
159 calleeCachedRecovery.recovery().virtualRegister(),
160 DataFormatCell));
161 break;
162 case InFPR:
163 case UnboxedCellInGPR:
164 case CellDisplacedInJSStack:
165 break;
166 case Constant:
167 ASSERT(calleeCachedRecovery.recovery().constant().isCell());
168 break;
169 default:
170 RELEASE_ASSERT_NOT_REACHED();
171 break;
172 }
173#endif
174 }

198 // "box" is arguably a bad name here. The meaning is that after
199 // calling emitBox(), your ensure that subsequently calling
200 // emitStore() will be able to store the value without additional
201 // transformation. In particular, this is a no-op for constants,

207 bool canBox(CachedRecovery& cachedRecovery)
208 {
209 if (cachedRecovery.boxingRequiresGPR() && getFreeGPR() == InvalidGPRReg)
210 return false;
211
212 if (cachedRecovery.boxingRequiresFPR() && getFreeFPR() == InvalidFPRReg)
213 return false;
214
215 return true;
216 }
217
218 void ensureBox(CachedRecovery& cachedRecovery)
219 {
220 if (canBox(cachedRecovery))
221 return;
222
223 if (cachedRecovery.boxingRequiresGPR())
224 ensureGPR();
225
226 if (cachedRecovery.boxingRequiresFPR())
227 ensureFPR();
228 }

234 void ensureLoad(CachedRecovery& cachedRecovery)
235 {
236 if (canLoad(cachedRecovery))

246 } else

250 bool canLoadAndBox(CachedRecovery& cachedRecovery)
251 {
252 // We don't have interfering loads & boxes
253 ASSERT(!cachedRecovery.loadsIntoFPR() || !cachedRecovery.boxingRequiresFPR());

398 // This stores the set of locked registers, i.e. registers for
399 // which we have an implicit requirement that they are not changed.
400 //
401 // This will usually contains the link register on architectures
402 // that have one, any scratch register used by the macro assembler
403 // (e.g. r11 on X86_64), as well as any register that we use for
404 // addressing (see m_oldFrameBase and m_newFrameBase).
405 //
406 // We also use this to lock registers temporarily, for instance to
407 // ensure that we have at least 2 available registers for loading
408 // a pair on 32bits.

422 // This stores, for each register, information about the recovery
423 // for the value that should eventually go into that register. The
424 // only registers that have a target recovery will be callee-save
425 // registers, as well as possibly one JSValueRegs for holding the
426 // callee.
427 //
428 // Once the correct value has been put into the registers, and
429 // contrary to what we do with m_newFrame, we keep the entry in
430 // m_newRegisters to simplify spilling.
431 RegisterMap<CachedRecovery*> m_newRegisters;

433 template<typename CheckFunctor>
434 Reg getFreeRegister(const CheckFunctor& check) const
435 {
436 Reg nonTemp { };
437 for (Reg reg = Reg::first(); reg <= Reg::last(); reg = reg.next()) {
438 if (m_lockedRegisters.get(reg))
439 continue;
440
441 if (!check(reg))
442 continue;
443
444 if (!m_registers[reg]) {
445 if (!m_newRegisters[reg])
446 return reg;
447 if (!nonTemp)
448 nonTemp = reg;
449 }
450 }
451
452#if USE(JSVALUE64)
453 if (!nonTemp && m_numberTagRegister != InvalidGPRReg && check(Reg { m_numberTagRegister })) {
454 ASSERT(m_lockedRegisters.get(m_numberTagRegister));
455 m_lockedRegisters.clear(m_numberTagRegister);
456 nonTemp = Reg { m_numberTagRegister };
457 m_numberTagRegister = InvalidGPRReg;
458 }
459#endif
460 return nonTemp;
461 }

492 // This frees up a register satisfying the check functor (this
493 // functor could theoretically have any kind of logic, but it must
494 // ensure that it will only return true for registers - spill
495 // assumes and asserts that it is passed a cachedRecovery stored in a
496 // register).
497 template<typename CheckFunctor>
498 void ensureRegister(const CheckFunctor& check)
499 {
500 // If we can spill a callee-save, that's best, because it will
501 // free up a register that would otherwise been taken for the
502 // longest amount of time.
503 //
504 // We could try to bias towards those that are not in their
505 // target registers yet, but the gain is probably super
506 // small. Unless you have a huge number of argument (at least
507 // around twice the number of available registers on your
508 // architecture), no spilling is going to take place anyways.
509 for (Reg reg = Reg::first(); reg <= Reg::last(); reg = reg.next()) {
510 if (m_lockedRegisters.get(reg))
511 continue;
512
513 CachedRecovery* cachedRecovery { m_newRegisters[reg] };
514 if (!cachedRecovery)
515 continue;
516
517 if (check(*cachedRecovery)) {
518 if (verbose)
519 dataLog(" ", cachedRecovery->recovery(), " looks like a good spill candidate\n");
520 spill(*cachedRecovery);
521 return;
522 }
523 }
524
525 // We use the cachedRecovery associated with the first new slot we
526 // can, because that is the one for which a write will be
527 // possible the latest, i.e. that is the one that we would
528 // have had to retain in registers for the longest.
529 for (VirtualRegister reg = firstNew(); reg <= lastNew(); reg += 1) {
530 CachedRecovery* cachedRecovery { getNew(reg) };
531 if (!cachedRecovery)
532 continue;
533
534 if (check(*cachedRecovery)) {
535 spill(*cachedRecovery);
536 return;
537 }
538 }
539
540 RELEASE_ASSERT_NOT_REACHED();
541 }

543 void ensureRegister()
544 {
545 if (hasFreeRegister())
546 return;
547
548 if (verbose)
549 dataLog(" Finding a register to spill\n");
550 ensureRegister(
551 [this] (const CachedRecovery& cachedRecovery) {
552 if (cachedRecovery.recovery().isInGPR())
553 return !m_lockedRegisters.get(cachedRecovery.recovery().gpr());
554 if (cachedRecovery.recovery().isInFPR())
555 return !m_lockedRegisters.get(cachedRecovery.recovery().fpr());
556#if USE(JSVALUE32_64)
557 if (cachedRecovery.recovery().technique() == InPair) {
558 return !m_lockedRegisters.get(cachedRecovery.recovery().tagGPR())
559 && !m_lockedRegisters.get(cachedRecovery.recovery().payloadGPR());
560 }
561#endif
562 return false;
563 });
564 }

566 void ensureTempGPR()
567 {
568 if (getFreeTempGPR() != InvalidGPRReg)
569 return;
570
571 if (verbose)
572 dataLog(" Finding a temp GPR to spill\n");
573 ensureRegister(
574 [this] (const CachedRecovery& cachedRecovery) {
575 if (cachedRecovery.recovery().isInGPR()) {
576 return !m_lockedRegisters.get(cachedRecovery.recovery().gpr())
577 && !m_newRegisters[cachedRecovery.recovery().gpr()];
578 }
579#if USE(JSVALUE32_64)

581 return !m_lockedRegisters.get(cachedRecovery.recovery().tagGPR())

583 && !m_newRegisters[cachedRecovery.recovery().tagGPR()]
584 && !m_newRegisters[cachedRecovery.recovery().payloadGPR()];
585 }
586#endif
587 return false;
588 });

641 void addNew(JSValueRegs jsValueRegs, ValueRecovery recovery)
642 {
643 ASSERT(jsValueRegs && !getNew(jsValueRegs));
644 CachedRecovery* cachedRecovery = addCachedRecovery(recovery);
645#if USE(JSVALUE64)
646 if (cachedRecovery->wantedJSValueRegs())
647 m_newRegisters[cachedRecovery->wantedJSValueRegs().gpr()] = nullptr;
648 m_newRegisters[jsValueRegs.gpr()] = cachedRecovery;
649#else
650 if (JSValueRegs oldRegs { cachedRecovery->wantedJSValueRegs() }) {
651 if (oldRegs.payloadGPR())
652 m_newRegisters[oldRegs.payloadGPR()] = nullptr;
653 if (oldRegs.tagGPR())
654 m_newRegisters[oldRegs.tagGPR()] = nullptr;
655 }
656 if (jsValueRegs.payloadGPR() != InvalidGPRReg)
657 m_newRegisters[jsValueRegs.payloadGPR()] = cachedRecovery;
658 if (jsValueRegs.tagGPR() != InvalidGPRReg)
659 m_newRegisters[jsValueRegs.tagGPR()] = cachedRecovery;
660#endif
661 ASSERT(!cachedRecovery->wantedJSValueRegs());
662 cachedRecovery->setWantedJSValueRegs(jsValueRegs);
663 }

674 // m_oldFrameBase is the register relative to which we access
675 // slots in the old call frame, with an additional offset of
676 // m_oldFrameOffset.
677 //
678 // - For an actual tail call, m_oldFrameBase is the stack
679 // pointer, and m_oldFrameOffset is the number of locals of the
680 // tail caller's frame. We use such stack pointer-based
681 // addressing because it allows us to load the tail caller's
682 // caller's frame pointer in the frame pointer register
683 // immediately instead of awkwardly keeping it around on the
684 // stack.
685 //
686 // - For a slow path call, m_oldFrameBase is just the frame
687 // pointer, and m_oldFrameOffset is 0.
688 GPRReg m_oldFrameBase { MacroAssembler::framePointerRegister };
689 int m_oldFrameOffset { 0 };
690
691 MacroAssembler::Address addressForOld(VirtualRegister reg) const
692 {
693 return MacroAssembler::Address(m_oldFrameBase,
694 (m_oldFrameOffset + reg.offset()) * sizeof(Register));
695 }
696
697 // m_newFrameBase is the register relative to which we access
698 // slots in the new call frame, and we always make it point to
699 // wherever the stack pointer will be right before making the
700 // actual call/jump. The actual base of the new frame is at offset
701 // m_newFrameOffset relative to m_newFrameBase.
702 //
703 // - For an actual tail call, m_newFrameBase is computed
704 // dynamically, and m_newFrameOffset varies between 0 and -2
705 // depending on the architecture's calling convention (see
706 // prepareForTailCall).
707 //
708 // - For a slow path call, m_newFrameBase is the actual stack
709 // pointer, and m_newFrameOffset is - CallerFrameAndPCSize,
710 // following the convention for a regular call.
711 GPRReg m_newFrameBase { InvalidGPRReg };
712 int m_newFrameOffset { 0};
713
714 bool isUndecided() const
715 {
716 return m_newFrameBase == InvalidGPRReg;
717 }
718
719 bool isSlowPath() const
720 {
721 return m_newFrameBase == MacroAssembler::stackPointerRegister;
722 }
723
724 MacroAssembler::Address addressForNew(VirtualRegister reg) const
725 {
726 return MacroAssembler::Address(m_newFrameBase,
727 (m_newFrameOffset + reg.offset()) * sizeof(Register));
728 }

730 // We use a concept of "danger zone". The danger zone consists of
731 // all the writes in the new frame that could overlap with reads
732 // in the old frame.
733 //
734 // Because we could have a higher actual number of arguments than
735 // parameters, when preparing a tail call, we need to assume that
736 // writing to a slot on the new frame could overlap not only with
737 // the corresponding slot in the old frame, but also with any slot
738 // above it. Thus, the danger zone consists of all writes between
739 // the first write and what I call the "danger frontier": the
740 // highest slot in the old frame we still care about. Thus, the
741 // danger zone contains all the slots between the first slot of
742 // the new frame and the danger frontier. Because the danger
743 // frontier is related to the new frame, it is stored as a virtual
744 // register *in the new frame*.
745 VirtualRegister m_dangerFrontier;
746
747 VirtualRegister dangerFrontier() const
748 {
749 ASSERT(!isUndecided());
750
751 return m_dangerFrontier;
752 }
753
754 bool isDangerNew(VirtualRegister reg) const
755 {
756 ASSERT(!isUndecided() && isValidNew(reg));
757 return reg <= dangerFrontier();
758 }
759
760 void updateDangerFrontier()
761 {
762 ASSERT(!isUndecided());
763
764 m_dangerFrontier = firstNew() - 1;
765 for (VirtualRegister reg = lastNew(); reg >= firstNew(); reg -= 1) {
766 if (!getNew(reg) || !isValidOld(newAsOld(reg)) || !getOld(newAsOld(reg)))
767 continue;
768
769 m_dangerFrontier = reg;
770 if (verbose)
771 dataLog(" Danger frontier now at NEW ", m_dangerFrontier, "\n");
772 break;
773 }
774 if (verbose)
775 dataLog(" All clear! Danger zone is empty.\n");
776 }
777
778 // A safe write is a write that never writes into the danger zone.
779 bool hasOnlySafeWrites(CachedRecovery& cachedRecovery) const
780 {
781 for (VirtualRegister target : cachedRecovery.targets()) {
782 if (isDangerNew(target))
783 return false;
784 }
785 return true;
786 }
787
788 // You must ensure that there is no dangerous writes before
789 // calling this function.
790 bool tryWrites(CachedRecovery&);
791
792 // This function tries to ensure that there is no longer any
793 // possible safe write, i.e. all remaining writes are either to
794 // the danger zone or callee save restorations.
795 //
796 // It returns false if it was unable to perform some safe writes
797 // due to high register pressure.
798 bool performSafeWrites();
799
800 unsigned m_numPassedArgs { UINT_MAX };
801};
