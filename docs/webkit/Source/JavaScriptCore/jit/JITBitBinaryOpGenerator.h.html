26#pragma once
27
28#if ENABLE(JIT)
29
30#include "CCallHelpers.h"
31#include "SnippetOperand.h"
32
33namespace JSC {
34
35class JITBitBinaryOpGenerator {
36public:
37 JITBitBinaryOpGenerator(const SnippetOperand& leftOperand, const SnippetOperand& rightOperand,
38 JSValueRegs result, JSValueRegs left, JSValueRegs right, GPRReg scratchGPR)
39 : m_leftOperand(leftOperand)
40 , m_rightOperand(rightOperand)
41 , m_result(result)
42 , m_left(left)
43 , m_right(right)
44 , m_scratchGPR(scratchGPR)
45 {
46 ASSERT(!m_leftOperand.isConstInt32() || !m_rightOperand.isConstInt32());
47 }
48
49 bool didEmitFastPath() const { return m_didEmitFastPath; }
50 CCallHelpers::JumpList& endJumpList() { return m_endJumpList; }
51 CCallHelpers::JumpList& slowPathJumpList() { return m_slowPathJumpList; }
52
53protected:
54 SnippetOperand m_leftOperand;
55 SnippetOperand m_rightOperand;
56 JSValueRegs m_result;
57 JSValueRegs m_left;
58 JSValueRegs m_right;
59 GPRReg m_scratchGPR;
60 bool m_didEmitFastPath { false };
61
62 CCallHelpers::JumpList m_endJumpList;
63 CCallHelpers::JumpList m_slowPathJumpList;
64};
65
66} // namespace JSC
67
68#endif // ENABLE(JIT)
