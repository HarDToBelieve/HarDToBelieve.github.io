44struct MathICGenerationState {
45 WTF_MAKE_STRUCT_FAST_ALLOCATED;
46 MacroAssembler::Label fastPathStart;
47 MacroAssembler::Label fastPathEnd;
48 MacroAssembler::Label slowPathStart;
49 MacroAssembler::Call slowPathCall;
50 MacroAssembler::JumpList slowPathJumps;
51 bool shouldSlowPathRepatch;

56template <typename GeneratorType, typename ArithProfileType>
57class JITMathIC {
58 WTF_MAKE_FAST_ALLOCATED;
59public:
60 JITMathIC(ArithProfileType* arithProfile)
61 : m_arithProfile(arithProfile)
62 {
63 }
64
65 CodeLocationLabel<JSInternalPtrTag> doneLocation() { return m_inlineEnd; }
66 CodeLocationCall<JSInternalPtrTag> slowPathCallLocation() { return m_slowPathCallLocation; }
67 CodeLocationLabel<JSInternalPtrTag> slowPathStartLocation() { return m_slowPathStartLocation; }
68
69 bool generateInline(CCallHelpers& jit, MathICGenerationState& state, bool shouldEmitProfiling = true)
70 {
71 state.fastPathStart = jit.label();
72 size_t startSize = jit.m_assembler.buffer().codeSize();
73
74 if (m_arithProfile && m_arithProfile->isObservedTypeEmpty()) {
75 // It looks like the MathIC has yet to execute. We don't want to emit code in this
76 // case for a couple reasons. First, the operation may never execute, so if we don't emit
77 // code, it's a win. Second, if the operation does execute, we can emit better code
78 // once we have an idea about the types.
79 state.slowPathJumps.append(jit.patchableJump());
80 size_t inlineSize = jit.m_assembler.buffer().codeSize() - startSize;
81 ASSERT_UNUSED(inlineSize, static_cast<ptrdiff_t>(inlineSize) <= MacroAssembler::patchableJumpSize());
82 state.shouldSlowPathRepatch = true;
83 state.fastPathEnd = jit.label();
84 ASSERT(!m_generateFastPathOnRepatch); // We should have gathered some observed type info about the types before trying to regenerate again.
85 m_generateFastPathOnRepatch = true;
86 return true;
87 }
88
89 JITMathICInlineResult result = m_generator.generateInline(jit, state, m_arithProfile);
90
91 switch (result) {
92 case JITMathICInlineResult::GeneratedFastPath: {
93 size_t inlineSize = jit.m_assembler.buffer().codeSize() - startSize;
94 if (static_cast<ptrdiff_t>(inlineSize) < MacroAssembler::patchableJumpSize()) {
95 size_t nopsToEmitInBytes = MacroAssembler::patchableJumpSize() - inlineSize;
96 jit.emitNops(nopsToEmitInBytes);
97 }
98 state.shouldSlowPathRepatch = true;
99 state.fastPathEnd = jit.label();
100 return true;
101 }
102 case JITMathICInlineResult::GenerateFullSnippet: {
103 MacroAssembler::JumpList endJumpList;
104 bool result = m_generator.generateFastPath(jit, endJumpList, state.slowPathJumps, m_arithProfile, shouldEmitProfiling);
105 if (result) {
106 state.fastPathEnd = jit.label();
107 state.shouldSlowPathRepatch = false;
108 endJumpList.link(&jit);
109 return true;
110 }
111 return false;
112 }
113 case JITMathICInlineResult::DontGenerate: {
114 return false;
115 }
116 default:
117 ASSERT_NOT_REACHED();
118 }
119
120 return false;
121 }
122
123 void generateOutOfLine(CodeBlock* codeBlock, FunctionPtr<CFunctionPtrTag> callReplacement)
124 {
125 auto linkJumpToOutOfLineSnippet = [&] () {
126 CCallHelpers jit(codeBlock);
127 auto jump = jit.jump();
128 // We don't need a nop sled here because nobody should be jumping into the middle of an IC.
129 bool needsBranchCompaction = false;
130 RELEASE_ASSERT(jit.m_assembler.buffer().codeSize() <= static_cast<size_t>(MacroAssembler::differenceBetweenCodePtr(m_inlineStart, m_inlineEnd)));
131 LinkBuffer linkBuffer(jit, m_inlineStart, jit.m_assembler.buffer().codeSize(), LinkBuffer::Profile::InlineCache, JITCompilationMustSucceed, needsBranchCompaction);
132 RELEASE_ASSERT(linkBuffer.isValid());
133 linkBuffer.link(jump, CodeLocationLabel<JITStubRoutinePtrTag>(m_code.code()));
134 FINALIZE_CODE(linkBuffer, NoPtrTag, "JITMathIC: linking constant jump to out of line stub");
135 };
136
137 auto replaceCall = [&] () {
138#if COMPILER(MSVC) && !COMPILER(CLANG)
139 ftlThunkAwareRepatchCall(codeBlock, slowPathCallLocation().retagged<JSInternalPtrTag>(), callReplacement);
140#else
141 ftlThunkAwareRepatchCall(codeBlock, slowPathCallLocation().template retagged<JSInternalPtrTag>(), callReplacement);
142#endif
143 };
144
145 bool shouldEmitProfiling = !JITCode::isOptimizingJIT(codeBlock->jitType());
146
147 if (m_generateFastPathOnRepatch) {
148
149 CCallHelpers jit(codeBlock);
150 MathICGenerationState generationState;
151 bool generatedInline = generateInline(jit, generationState, shouldEmitProfiling);
152
153 // We no longer want to try to regenerate the fast path.
154 m_generateFastPathOnRepatch = false;
155
156 if (generatedInline) {
157 auto jumpToDone = jit.jump();
158
159 LinkBuffer linkBuffer(jit, codeBlock, LinkBuffer::Profile::InlineCache, JITCompilationCanFail);
160 if (!linkBuffer.didFailToAllocate()) {
161 linkBuffer.link(generationState.slowPathJumps, slowPathStartLocation());
162 linkBuffer.link(jumpToDone, doneLocation());
163
164 m_code = FINALIZE_CODE_FOR(
165 codeBlock, linkBuffer, JITStubRoutinePtrTag, "JITMathIC: generating out of line fast IC snippet");
166
167 if (!generationState.shouldSlowPathRepatch) {
168 // We won't need to regenerate, so we can wire the slow path call
169 // to a non repatching variant.
170 replaceCall();
171 }
172
173 linkJumpToOutOfLineSnippet();
174
175 return;
176 }
177 }
178
179 // We weren't able to generate an out of line fast path.
180 // We just generate the snippet in its full generality.
181 }
182
183 // We rewire to the alternate regardless of whether or not we can allocate the out of line path
184 // because if we fail allocating the out of line path, we don't want to waste time trying to
185 // allocate it in the future.
186 replaceCall();
187
188 {
189 CCallHelpers jit(codeBlock);
190
191 MacroAssembler::JumpList endJumpList;
192 MacroAssembler::JumpList slowPathJumpList;
193
194 bool emittedFastPath = m_generator.generateFastPath(jit, endJumpList, slowPathJumpList, m_arithProfile, shouldEmitProfiling);
195 if (!emittedFastPath)
196 return;
197 endJumpList.append(jit.jump());
198
199 LinkBuffer linkBuffer(jit, codeBlock, LinkBuffer::Profile::InlineCache, JITCompilationCanFail);
200 if (linkBuffer.didFailToAllocate())
201 return;
202
203 linkBuffer.link(endJumpList, doneLocation());
204 linkBuffer.link(slowPathJumpList, slowPathStartLocation());
205
206 m_code = FINALIZE_CODE_FOR(
207 codeBlock, linkBuffer, JITStubRoutinePtrTag, "JITMathIC: generating out of line IC snippet");
208 }
209
210 linkJumpToOutOfLineSnippet();
211 }
212
213 void finalizeInlineCode(const MathICGenerationState& state, LinkBuffer& linkBuffer)
214 {
215 CodeLocationLabel<JSInternalPtrTag> start = linkBuffer.locationOf<JSInternalPtrTag>(state.fastPathStart);
216 m_inlineStart = start;
217
218 m_inlineEnd = linkBuffer.locationOf<JSInternalPtrTag>(state.fastPathEnd);
219 ASSERT(m_inlineEnd.untaggedExecutableAddress() > m_inlineStart.untaggedExecutableAddress());
220
221 m_slowPathCallLocation = linkBuffer.locationOf<JSInternalPtrTag>(state.slowPathCall);
222 m_slowPathStartLocation = linkBuffer.locationOf<JSInternalPtrTag>(state.slowPathStart);
223 }
224
225 ArithProfileType* arithProfile() const { return m_arithProfile; }
226
227#if ENABLE(MATH_IC_STATS)
228 size_t m_generatedCodeSize { 0 };
229 size_t codeSize() const
230 {
231 size_t result = m_generatedCodeSize;
232 if (m_code)
233 result += m_code.size();
234 return result;
235 }
236#endif
237
238 ArithProfileType* m_arithProfile;
239 MacroAssemblerCodeRef<JITStubRoutinePtrTag> m_code;
240 CodeLocationLabel<JSInternalPtrTag> m_inlineStart;
241 CodeLocationLabel<JSInternalPtrTag> m_inlineEnd;
242 CodeLocationLabel<JSInternalPtrTag> m_slowPathCallLocation;
243 CodeLocationLabel<JSInternalPtrTag> m_slowPathStartLocation;
244 bool m_generateFastPathOnRepatch { false };
245 GeneratorType m_generator;
246};
