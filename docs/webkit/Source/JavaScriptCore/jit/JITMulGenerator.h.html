27
28#if ENABLE(JIT)
29
30#include "CCallHelpers.h"
31#include "JITMathICInlineResult.h"
32#include "SnippetOperand.h"
33
34namespace JSC {
35
36struct MathICGenerationState;
37
38class JITMulGenerator {
39public:
40 JITMulGenerator()
41 { }
42
43 JITMulGenerator(SnippetOperand leftOperand, SnippetOperand rightOperand,
44 JSValueRegs result, JSValueRegs left, JSValueRegs right,
45 FPRReg leftFPR, FPRReg rightFPR, GPRReg scratchGPR, FPRReg scratchFPR)
46 : m_leftOperand(leftOperand)
47 , m_rightOperand(rightOperand)
48 , m_result(result)
49 , m_left(left)
50 , m_right(right)
51 , m_leftFPR(leftFPR)
52 , m_rightFPR(rightFPR)
53 , m_scratchGPR(scratchGPR)
54 , m_scratchFPR(scratchFPR)
55 {
56 ASSERT(!m_leftOperand.isPositiveConstInt32() || !m_rightOperand.isPositiveConstInt32());
57 }
58
59 JITMathICInlineResult generateInline(CCallHelpers&, MathICGenerationState&, const BinaryArithProfile*);
60 bool generateFastPath(CCallHelpers&, CCallHelpers::JumpList& endJumpList, CCallHelpers::JumpList& slowJumpList, const BinaryArithProfile*, bool shouldEmitProfiling);
61
62 static bool isLeftOperandValidConstant(SnippetOperand leftOperand) { return leftOperand.isPositiveConstInt32(); }
63 static bool isRightOperandValidConstant(SnippetOperand rightOperand) { return rightOperand.isPositiveConstInt32(); }
64
65private:
66 SnippetOperand m_leftOperand;
67 SnippetOperand m_rightOperand;
68 JSValueRegs m_result;
69 JSValueRegs m_left;
70 JSValueRegs m_right;
71 FPRReg m_leftFPR;
72 FPRReg m_rightFPR;
73 GPRReg m_scratchGPR;
74 FPRReg m_scratchFPR;
75};
76
77} // namespace JSC
78
79#endif // ENABLE(JIT)
