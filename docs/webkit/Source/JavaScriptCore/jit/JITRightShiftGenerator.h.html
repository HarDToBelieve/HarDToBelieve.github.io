34class JITRightShiftGenerator : public JITBitBinaryOpGenerator {

36 enum ShiftType {
37 SignedShift,
38 UnsignedShift
39 };
40
41 JITRightShiftGenerator(const SnippetOperand& leftOperand, const SnippetOperand& rightOperand,
42 JSValueRegs result, JSValueRegs left, JSValueRegs right,
43 FPRReg leftFPR, GPRReg scratchGPR, FPRReg scratchFPR, ShiftType type = SignedShift)
44 : JITBitBinaryOpGenerator(leftOperand, rightOperand, result, left, right, scratchGPR)
45 , m_shiftType(type)
46 , m_leftFPR(leftFPR)
47 , m_scratchFPR(scratchFPR)
48 { }
49
50 void generateFastPath(CCallHelpers&);
51
52private:
53 ShiftType m_shiftType;
54 FPRReg m_leftFPR;
55 FPRReg m_scratchFPR;
56};
