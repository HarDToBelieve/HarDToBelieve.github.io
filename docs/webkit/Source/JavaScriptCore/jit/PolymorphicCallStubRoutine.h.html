37namespace JSC {
38
39class CallLinkInfo;
40
41class PolymorphicCallNode : public PackedRawSentinelNode<PolymorphicCallNode> {
42 WTF_MAKE_NONCOPYABLE(PolymorphicCallNode);
43public:
44 PolymorphicCallNode(CallLinkInfo* info)
45 : m_callLinkInfo(info)
46 {
47 }
48
49 ~PolymorphicCallNode();
50
51 void unlink(VM&);
52
53 bool hasCallLinkInfo(CallLinkInfo* info) { return m_callLinkInfo.get() == info; }
54 void clearCallLinkInfo();
55
56private:
57 PackedPtr<CallLinkInfo> m_callLinkInfo;
58};
59
60class PolymorphicCallCase {
61public:
62 PolymorphicCallCase()
63 : m_codeBlock(nullptr)
64 {
65 }
66
67 PolymorphicCallCase(CallVariant variant, CodeBlock* codeBlock)
68 : m_variant(variant)
69 , m_codeBlock(codeBlock)
70 {
71 }
72
73 CallVariant variant() const { return m_variant; }
74 CodeBlock* codeBlock() const { return m_codeBlock; }
75
76 void dump(PrintStream&) const;
77
78private:
79 CallVariant m_variant;
80 CodeBlock* m_codeBlock;
81};
82
83class PolymorphicCallStubRoutine final : public GCAwareJITStubRoutine {
84public:
85 PolymorphicCallStubRoutine(
86 const MacroAssemblerCodeRef<JITStubRoutinePtrTag>&, VM&, const JSCell* owner,
87 CallFrame* callerFrame, CallLinkInfo&, const Vector<PolymorphicCallCase>&,
88 UniqueArray<uint32_t>&& fastCounts);
89
90 ~PolymorphicCallStubRoutine() final;
91
92 CallVariantList variants() const;
93 bool hasEdges() const;
94 CallEdgeList edges() const;
95
96 void clearCallNodesFor(CallLinkInfo*);
97
98 template<typename Functor>
99 void forEachDependentCell(const Functor& functor)
100 {
101 for (auto& variant : m_variants)
102 functor(variant.get());
103 }
104
105 bool visitWeak(VM&) final;
106
107private:
108 template<typename Visitor> void markRequiredObjectsInternalImpl(Visitor&);
109 void markRequiredObjectsInternal(AbstractSlotVisitor&) final;
110 void markRequiredObjectsInternal(SlotVisitor&) final;
111
112 FixedVector<WriteBarrier<JSCell>> m_variants;
113 UniqueArray<uint32_t> m_fastCounts;
114 Bag<PolymorphicCallNode> m_callNodes;
115};
