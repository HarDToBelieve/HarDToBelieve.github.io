26#pragma once
27
28#if ENABLE(ASSEMBLER)
29
30#include "RegisterAtOffset.h"
31#include "RegisterSet.h"
32#include <wtf/FixedVector.h>
33
34namespace JSC {
35
36DECLARE_ALLOCATOR_WITH_HEAP_IDENTIFIER(RegisterAtOffsetList);
37class RegisterAtOffsetList {
38 WTF_MAKE_STRUCT_FAST_ALLOCATED_WITH_HEAP_IDENTIFIER(RegisterAtOffsetList);
39public:
40 enum OffsetBaseType { FramePointerBased, ZeroBased };
41
42 RegisterAtOffsetList();
43 explicit RegisterAtOffsetList(RegisterSet, OffsetBaseType = FramePointerBased);
44
45 void dump(PrintStream&) const;
46
47 size_t size() const
48 {
49 return m_registers.size();
50 }
51
52 const RegisterAtOffset& at(size_t index) const
53 {
54 return m_registers.at(index);
55 }
56
57 RegisterAtOffset& at(size_t index)
58 {
59 return m_registers.at(index);
60 }
61
62 RegisterAtOffset* find(Reg) const;
63 unsigned indexOf(Reg) const; // Returns UINT_MAX if not found.
64
65 FixedVector<RegisterAtOffset>::const_iterator begin() const { return m_registers.begin(); }
66 FixedVector<RegisterAtOffset>::const_iterator end() const { return m_registers.end(); }
67
68 static const RegisterAtOffsetList& llintBaselineCalleeSaveRegisters(); // Registers and Offsets saved and used by the LLInt.
69
70private:
71 FixedVector<RegisterAtOffset> m_registers;
72};
73
74} // namespace JSC
75
76#endif // ENABLE(ASSEMBLER)
