30#include "FPRInfo.h"

33#include "Reg.h"

35namespace JSC {
36
37template<typename T>
38class RegisterMap {
39public:
40 T& operator[](Reg reg)
41 {
42 return m_map[reg.index()];
43 }
44
45 T& operator[](GPRReg gpr)
46 {
47 return m_map[MacroAssembler::registerIndex(gpr)];
48 }
49
50 T& operator[](FPRReg fpr)
51 {
52 return m_map[MacroAssembler::registerIndex(fpr)];
53 }
54
55 const T& operator[](Reg reg) const
56 {
57 return m_map[reg.index()];
58 }
59
60 const T& operator[](GPRReg gpr) const
61 {
62 return m_map[MacroAssembler::registerIndex(gpr)];
63 }
64
65 const T& operator[](FPRReg fpr) const
66 {
67 return m_map[MacroAssembler::registerIndex(fpr)];
68 }
69
70private:
71 std::array<T, MacroAssembler::totalNumberOfRegisters()> m_map { { } };
72};
73
74template<typename T>
75class GPRMap {
76public:
77 T& operator[](GPRReg gpr)
78 {
79 return m_map[MacroAssembler::registerIndex(gpr)];
80 }
81
82 const T& operator[](GPRReg gpr) const
83 {
84 return m_map[MacroAssembler::registerIndex(gpr)];
85 }
86
87private:
88 std::array<T, MacroAssembler::numberOfRegisters()> m_map { { } };
89};
90
91template<typename T>
92class FPRMap {
93public:
94 T& operator[](FPRReg fpr)
95 {
96 return m_map[MacroAssembler::fpRegisterIndex(fpr)];
97 }
98
99 const T& operator[](FPRReg fpr) const
100 {
101 return m_map[MacroAssembler::fpRegisterIndex(fpr)];
102 }
103
104private:
105 std::array<T, MacroAssembler::numberOfFPRegisters()> m_map { { } };
106};
107
108} // namespace JSC
