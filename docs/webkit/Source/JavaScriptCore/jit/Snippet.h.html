31#include "CCallHelpers.h"

33namespace JSC {
34
35class SnippetParams;
36
37typedef CCallHelpers::JumpList SnippetCompilerFunction(CCallHelpers&, SnippetParams&);
38typedef SharedTask<SnippetCompilerFunction> SnippetCompiler;
39
40// Snippet is the way to inject an opaque code generator into DFG and FTL.
41// While B3::Patchpoint is self-contained about its compilation information,
42// Snippet depends on which DFG Node invokes. For example, CheckDOM will
43// link returned failureCases to BadType OSRExit, but this information is offered
44// from CheckDOM DFG Node, not from this snippet. This snippet mainly focuses
45// on injecting a snippet generator that can tell register usage and can be used
46// in both DFG and FTL.
47class Snippet : public ThreadSafeRefCounted<Snippet> {
48public:
49 static Ref<Snippet> create()
50 {
51 return adoptRef(*new Snippet());
52 }
53
54 template<typename Functor>
55 void setGenerator(const Functor& functor)
56 {
57 m_generator = createSharedTask<SnippetCompilerFunction>(functor);
58 }
59
60 RefPtr<SnippetCompiler> generator() const { return m_generator; }
61
62 uint8_t numGPScratchRegisters { 0 };
63 uint8_t numFPScratchRegisters { 0 };
64
65protected:
66 Snippet() = default;
67
68private:
69 RefPtr<SnippetCompiler> m_generator;
70};
71
72}

74#endif
