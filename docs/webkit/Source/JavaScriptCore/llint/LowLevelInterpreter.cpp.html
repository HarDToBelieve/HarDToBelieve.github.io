47// LLInt C Loop opcodes
48// ====================
49// In the implementation of the C loop, the LLint trampoline glue functions
50// (e.g. llint_program_prologue, llint_eval_prologue, etc) are addressed as
51// if they are bytecode handlers. That means the names of the trampoline
52// functions will be added to the OpcodeID list via the
53// FOR_EACH_LLINT_OPCODE_EXTENSION() macro that FOR_EACH_OPCODE_ID()
54// includes.
55//
56// In addition, some JIT trampoline functions which are needed by LLInt
57// (e.g. ctiOpThrowNotCaught) are also added as
58// bytecodes, and the CLoop will provide bytecode handlers for them.
59//
60// In the CLoop, we can only dispatch indirectly to these bytecodes
61// (including the LLInt and JIT extensions). All other dispatches
62// (i.e. goto's) must be to a known label (i.e. local / global labels).
63
64
65// How are the opcodes named?
66// ==========================
67// Here is a table to show examples of how each of the manifestation of the
68// opcodes are named:
69//
70// Type: Opcode Trampoline Glue
71// ====== ===============
72// [In the llint .asm files]
73// llint labels: llint_op_enter llint_program_prologue
74//
75// OpcodeID: op_enter llint_program
76// [in Opcode.h] [in LLIntOpcode.h]
77//
78// When using a switch statement dispatch in the CLoop, each "opcode" is
79// a case statement:
80// Opcode: case op_enter: case llint_program_prologue:
81//
82// When using a computed goto dispatch in the CLoop, each opcode is a label:
83// Opcode: op_enter: llint_program_prologue:
84
85

133// CLoopRegister is the storage for an emulated CPU register.
134// It defines the policy of how ints smaller than intptr_t are packed into the
135// pseudo register, as well as hides endianness differences.
136
137class CLoopRegister {
138public:
139 ALWAYS_INLINE intptr_t i() const { return m_value; };
140 ALWAYS_INLINE uintptr_t u() const { return m_value; }
141 ALWAYS_INLINE int32_t i32() const { return m_value; }
142 ALWAYS_INLINE uint32_t u32() const { return m_value; }
143 ALWAYS_INLINE int8_t i8() const { return m_value; }
144 ALWAYS_INLINE uint8_t u8() const { return m_value; }
145
146 ALWAYS_INLINE intptr_t* ip() const { return bitwise_cast<intptr_t*>(m_value); }
147 ALWAYS_INLINE int8_t* i8p() const { return bitwise_cast<int8_t*>(m_value); }
148 ALWAYS_INLINE void* vp() const { return bitwise_cast<void*>(m_value); }
149 ALWAYS_INLINE const void* cvp() const { return bitwise_cast<const void*>(m_value); }
150 ALWAYS_INLINE CallFrame* callFrame() const { return bitwise_cast<CallFrame*>(m_value); }
151 ALWAYS_INLINE const void* instruction() const { return bitwise_cast<const void*>(m_value); }
152 ALWAYS_INLINE VM* vm() const { return bitwise_cast<VM*>(m_value); }
153 ALWAYS_INLINE JSCell* cell() const { return bitwise_cast<JSCell*>(m_value); }
154 ALWAYS_INLINE ProtoCallFrame* protoCallFrame() const { return bitwise_cast<ProtoCallFrame*>(m_value); }
155 ALWAYS_INLINE NativeFunction nativeFunc() const { return bitwise_cast<NativeFunction>(m_value); }
156#if USE(JSVALUE64)
157 ALWAYS_INLINE int64_t i64() const { return m_value; }
158 ALWAYS_INLINE uint64_t u64() const { return m_value; }
159 ALWAYS_INLINE EncodedJSValue encodedJSValue() const { return bitwise_cast<EncodedJSValue>(m_value); }
160#endif
161 ALWAYS_INLINE Opcode opcode() const { return bitwise_cast<Opcode>(m_value); }
162
163 operator CallFrame*() { return bitwise_cast<CallFrame*>(m_value); }
164 operator const Instruction*() { return bitwise_cast<const Instruction*>(m_value); }
165 operator JSCell*() { return bitwise_cast<JSCell*>(m_value); }
166 operator ProtoCallFrame*() { return bitwise_cast<ProtoCallFrame*>(m_value); }
167 operator Register*() { return bitwise_cast<Register*>(m_value); }
168 operator VM*() { return bitwise_cast<VM*>(m_value); }
169
170 template<typename T, typename = std::enable_if_t<sizeof(T) == sizeof(uintptr_t)>>
171 ALWAYS_INLINE void operator=(T value) { m_value = bitwise_cast<uintptr_t>(value); }
172#if USE(JSVALUE64)
173 ALWAYS_INLINE void operator=(int32_t value) { m_value = static_cast<intptr_t>(value); }
174 ALWAYS_INLINE void operator=(uint32_t value) { m_value = static_cast<uintptr_t>(value); }
175#endif
176 ALWAYS_INLINE void operator=(int16_t value) { m_value = static_cast<intptr_t>(value); }
177 ALWAYS_INLINE void operator=(uint16_t value) { m_value = static_cast<uintptr_t>(value); }
178 ALWAYS_INLINE void operator=(int8_t value) { m_value = static_cast<intptr_t>(value); }
179 ALWAYS_INLINE void operator=(uint8_t value) { m_value = static_cast<uintptr_t>(value); }
180 ALWAYS_INLINE void operator=(bool value) { m_value = static_cast<uintptr_t>(value); }
181
182#if USE(JSVALUE64)
183 ALWAYS_INLINE double bitsAsDouble() const { return bitwise_cast<double>(m_value); }
184 ALWAYS_INLINE int64_t bitsAsInt64() const { return bitwise_cast<int64_t>(m_value); }
185#endif
186
187private:
188 uintptr_t m_value { static_cast<uintptr_t>(0xbadbeef0baddbeef) };
189};

194 explicit operator T() const { return bitwise_cast<T>(m_value); }
195
196 ALWAYS_INLINE double d() const { return m_value; }
197 ALWAYS_INLINE int64_t bitsAsInt64() const { return bitwise_cast<int64_t>(m_value); }

203
204private:
205 double m_value;
206};

244JSValue CLoop::execute(OpcodeID entryOpcodeID, void* executableAddress, VM* vm, ProtoCallFrame* protoCallFrame, bool isInitializationPass)
245{
246#define CAST bitwise_cast
247
248 // One-time initialization of our address tables. We have to put this code
249 // here because our labels are only in scope inside this function. The
250 // caller (or one of its ancestors) is responsible for ensuring that this
251 // is only called once during the initialization of the VM before threads
252 // are at play.
253 if (UNLIKELY(isInitializationPass)) {
254 Opcode* opcodeMap = LLInt::opcodeMap();
255 Opcode* opcodeMapWide16 = LLInt::opcodeMapWide16();
256 Opcode* opcodeMapWide32 = LLInt::opcodeMapWide32();
257
258#if ENABLE(COMPUTED_GOTO_OPCODES)
259 #define OPCODE_ENTRY(__opcode, length) \
260 opcodeMap[__opcode] = bitwise_cast<void*>(&&__opcode); \
261 opcodeMapWide16[__opcode] = bitwise_cast<void*>(&&__opcode##_wide16); \
262 opcodeMapWide32[__opcode] = bitwise_cast<void*>(&&__opcode##_wide32);
263
264 #define LLINT_OPCODE_ENTRY(__opcode, length) \
265 opcodeMap[__opcode] = bitwise_cast<void*>(&&__opcode);
266#else
267 // FIXME: this mapping is unnecessarily expensive in the absence of COMPUTED_GOTO
268 // narrow opcodes don't need any mapping and wide opcodes just need to add numOpcodeIDs
269 #define OPCODE_ENTRY(__opcode, length) \
270 opcodeMap[__opcode] = __opcode; \
271 opcodeMapWide16[__opcode] = static_cast<OpcodeID>(__opcode##_wide16); \
272 opcodeMapWide32[__opcode] = static_cast<OpcodeID>(__opcode##_wide32);
273
274 #define LLINT_OPCODE_ENTRY(__opcode, length) \
275 opcodeMap[__opcode] = __opcode;
276#endif
277 FOR_EACH_BYTECODE_ID(OPCODE_ENTRY)
278 FOR_EACH_CLOOP_BYTECODE_HELPER_ID(LLINT_OPCODE_ENTRY)
279 FOR_EACH_LLINT_NATIVE_HELPER(LLINT_OPCODE_ENTRY)
280 #undef OPCODE_ENTRY
281 #undef LLINT_OPCODE_ENTRY
282
283 // Note: we can only set the exceptionInstructions after we have
284 // initialized the opcodeMap above. This is because getCodePtr()
285 // can depend on the opcodeMap.
286 uint8_t* exceptionInstructions = reinterpret_cast<uint8_t*>(LLInt::exceptionInstructions());
287 for (unsigned i = 0; i < maxOpcodeLength + 1; ++i)
288 exceptionInstructions[i] = llint_throw_from_slow_path_trampoline;
289
290 return JSValue();
291 }
292
293 // Define the pseudo registers used by the LLINT C Loop backend:
294 static_assert(sizeof(CLoopRegister) == sizeof(intptr_t));
295
296 // The CLoop llint backend is initially based on the ARMv7 backend, and
297 // then further enhanced with a few instructions from the x86 backend to
298 // support building for X64 targets. Hence, the shape of the generated
299 // code and the usage convention of registers will look a lot like the
300 // ARMv7 backend's.
301 //
302 // For example, on a 32-bit build:
303 // 1. Outgoing args will be set up as follows:
304 // arg1 in t0 (r0 on ARM)
305 // arg2 in t1 (r1 on ARM)
306 // 2. 32 bit return values will be in t0 (r0 on ARM).
307 // 3. 64 bit return values (e.g. doubles) will be in t0,t1 (r0,r1 on ARM).
308 //
309 // But instead of naming these simulator registers based on their ARM
310 // counterparts, we'll name them based on their original llint asm names.
311 // This will make it easier to correlate the generated code with the
312 // original llint asm code.
313 //
314 // On a 64-bit build, it more like x64 in that the registers are 64 bit.
315 // Hence:
316 // 1. Outgoing args are still the same: arg1 in t0, arg2 in t1, etc.
317 // 2. 32 bit result values will be in the low 32-bit of t0.
318 // 3. 64 bit result values will be in t0.
319
320 CLoopRegister t0, t1, t2, t3, t5, sp, cfr, lr, pc;
321#if USE(JSVALUE64)
322 CLoopRegister numberTag, notCellMask;
323#endif
324 CLoopRegister pcBase;
325 CLoopRegister metadataTable;
326 CLoopDoubleRegister d0, d1;
327
328 struct StackPointerScope {
329 StackPointerScope(CLoopStack& stack)
330 : m_stack(stack)
331 , m_originalStackPointer(stack.currentStackPointer())
332 { }
333
334 ~StackPointerScope()
335 {
336 m_stack.setCurrentStackPointer(m_originalStackPointer);
337 }
338
339 private:
340 CLoopStack& m_stack;
341 void* m_originalStackPointer;
342 };
343
344 CLoopStack& cloopStack = vm->interpreter->cloopStack();
345 StackPointerScope stackPointerScope(cloopStack);
346
347 lr = getOpcode(llint_return_to_host);
348 sp = cloopStack.currentStackPointer();
349 cfr = vm->topCallFrame;
350#ifndef NDEBUG
351 void* startSP = sp.vp();
352 CallFrame* startCFR = cfr.callFrame();
353#endif
354
355 // Initialize the incoming args for doVMEntryToJavaScript:
356 t0 = executableAddress;
357 t1 = vm;
358 t2 = protoCallFrame;
359
360#if USE(JSVALUE64)
361 // For the ASM llint, JITStubs takes care of this initialization. We do
362 // it explicitly here for the C loop:
363 numberTag = JSValue::NumberTag;
364 notCellMask = JSValue::NotCellMask;
365#endif // USE(JSVALUE64)
366
367 // Interpreter variables for value passing between opcodes and/or helpers:
368 NativeFunction nativeFunc = nullptr;
369 JSValue functionReturnValue;
370 Opcode opcode = getOpcode(entryOpcodeID);
371
372#define PUSH(cloopReg) \
373 do { \
374 sp = sp.ip() - 1; \
375 *sp.ip() = cloopReg.i(); \
376 } while (false)
377
378#define POP(cloopReg) \
379 do { \
380 cloopReg = *sp.ip(); \
381 sp = sp.ip() + 1; \
382 } while (false)
383
384#if ENABLE(OPCODE_STATS)
385#define RECORD_OPCODE_STATS(__opcode) OpcodeStats::recordInstruction(__opcode)
386#else
387#define RECORD_OPCODE_STATS(__opcode)
388#endif
389
390#if ENABLE(COMPUTED_GOTO_OPCODES)
391
392 //========================================================================
393 // Loop dispatch mechanism using computed goto statements:
394
395 #define DISPATCH_OPCODE() goto *opcode
396
397 #define DEFINE_OPCODE(__opcode) \
398 __opcode: \
399 RECORD_OPCODE_STATS(__opcode);
400
401 // Dispatch to the current PC's bytecode:
402 DISPATCH_OPCODE();
403
404#else // !ENABLE(COMPUTED_GOTO_OPCODES)
405 //========================================================================
406 // Loop dispatch mechanism using a C switch statement:
407
408 #define DISPATCH_OPCODE() goto dispatchOpcode
409
410 #define DEFINE_OPCODE(__opcode) \
411 case __opcode: \
412 __opcode: \
413 RECORD_OPCODE_STATS(__opcode);
414
415 // Dispatch to the current PC's bytecode:
416 dispatchOpcode:
417 switch (static_cast<unsigned>(opcode))
418
419#endif // !ENABLE(COMPUTED_GOTO_OPCODES)
420
421 //========================================================================
422 // Bytecode handlers:
423 {
424 // This is the file generated by offlineasm, which contains all of the
425 // bytecode handlers for the interpreter, as compiled from
426 // LowLevelInterpreter.asm and its peers.
427
428 IGNORE_CLANG_WARNINGS_BEGIN("unreachable-code")
429 #include "LLIntAssembly.h"
430 IGNORE_CLANG_WARNINGS_END
431
432 OFFLINE_ASM_GLUE_LABEL(llint_return_to_host)
433 {
434 ASSERT(startSP == sp.vp());
435 ASSERT(startCFR == cfr.callFrame());
436#if USE(JSVALUE32_64)
437 return JSValue(t1.i(), t0.i()); // returning JSValue(tag, payload);
438#else
439 return JSValue::decode(t0.encodedJSValue());
440#endif
441 }
442
443#if !ENABLE(COMPUTED_GOTO_OPCODES)
444 default:
445 ASSERT(false);
446#endif
447
448 } // END bytecode handler cases.
449
450#if ENABLE(COMPUTED_GOTO_OPCODES)
451 // Keep the compiler happy so that it doesn't complain about unused
452 // labels for the LLInt trampoline glue. The labels are automatically
453 // emitted by label macros above, and some of them are referenced by
454 // the llint generated code. Since we can't tell ahead of time which
455 // will be referenced and which will be not, we'll just passify the
456 // compiler on all such labels:
457 #define LLINT_OPCODE_ENTRY(__opcode, length) \
458 UNUSED_LABEL(__opcode);
459 FOR_EACH_OPCODE_ID(LLINT_OPCODE_ENTRY);
460 #undef LLINT_OPCODE_ENTRY
461#endif
462
463 #undef DEFINE_OPCODE
464 #undef CHECK_FOR_TIMEOUT
465 #undef CAST
466
467 return JSValue(); // to suppress a compiler warning.
468} // Interpreter::llintCLoopExecute()
