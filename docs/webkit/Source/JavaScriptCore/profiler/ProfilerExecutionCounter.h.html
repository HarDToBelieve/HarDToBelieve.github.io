26#pragma once
27
28#include <wtf/FastMalloc.h>
29#include <wtf/Noncopyable.h>
30
31namespace JSC { namespace Profiler {
32
33class ExecutionCounter {
34 WTF_MAKE_FAST_ALLOCATED; WTF_MAKE_NONCOPYABLE(ExecutionCounter);
35public:
36 ExecutionCounter() : m_counter(0) { }
37
38 uint64_t* address() { return &m_counter; }
39
40 uint64_t count() const { return m_counter; }
41
42private:
43 uint64_t m_counter;
44};
45
46} } // namespace JSC::Profiler
