26#pragma once
27
28#include "BytecodeIndex.h"
29#include "CodeBlockHash.h"
30#include "JSCJSValue.h"
31#include <wtf/PrintStream.h>
32
33namespace JSC {
34
35class CodeBlock;
36
37namespace Profiler {
38
39class Bytecodes;
40class Database;
41
42class Origin {
43public:
44 Origin() = default;
45 Origin(WTF::HashTableDeletedValueType)
46 : m_bytecodeIndex(WTF::HashTableDeletedValue)
47 { }
48
49 Origin(Bytecodes* bytecodes, BytecodeIndex bytecodeIndex)
50 : m_bytecodes(bytecodes)
51 , m_bytecodeIndex(bytecodeIndex)
52 {
53 ASSERT(m_bytecodeIndex.offset() < std::numeric_limits<unsigned>::max() - 1);
54 }
55
56 Origin(Database&, CodeBlock*, BytecodeIndex);
57
58 bool operator!() const { return !m_bytecodeIndex; }
59
60 Bytecodes* bytecodes() const { return m_bytecodes; }
61 BytecodeIndex bytecodeIndex() const { return m_bytecodeIndex; }
62
63 bool operator==(const Origin&) const;
64 bool operator!=(const Origin& other) const { return !(*this == other); }
65 unsigned hash() const;
66
67 bool isHashTableDeletedValue() const;
68
69 void dump(PrintStream&) const;
70 JSValue toJS(JSGlobalObject*) const;
71
72private:
73 Bytecodes* m_bytecodes;
74 BytecodeIndex m_bytecodeIndex;
75};
76
77inline bool Origin::operator==(const Origin& other) const
78{
79 return m_bytecodes == other.m_bytecodes
80 && m_bytecodeIndex == other.m_bytecodeIndex;
81}
82
83inline unsigned Origin::hash() const
84{
85 return WTF::PtrHash<Bytecodes*>::hash(m_bytecodes) + m_bytecodeIndex.hash();
86}
87
88inline bool Origin::isHashTableDeletedValue() const
89{
90 return m_bytecodeIndex.isHashTableDeletedValue();
91}
92
93struct OriginHash {
94 static unsigned hash(const Origin& key) { return key.hash(); }
95 static bool equal(const Origin& a, const Origin& b) { return a == b; }
96 static constexpr bool safeToCompareToEmptyOrDeleted = true;
97};
98
99} } // namespace JSC::Profiler
