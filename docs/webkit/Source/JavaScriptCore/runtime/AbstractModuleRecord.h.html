170 // The loader resolves the given module name to the module key. The module key is the unique value to represent this module.
171 Identifier m_moduleKey;
172
173 // Currently, we don't keep the occurrence order of the import / export entries.
174 // So, we does not guarantee the order of the errors.
175 // e.g. The import declaration that occurr later than the another import declaration may
176 // throw the error even if the former import declaration also has the invalid content.
177 //
178 // import ... // (1) this has some invalid content.
179 // import ... // (2) this also has some invalid content.
180 //
181 // In the above case, (2) may throw the error earlier than (1)
182 //
183 // But, in all the cases, we will throw the syntax error. So except for the content of the syntax error,
184 // there are no difference.
185
186 // Map localName -> ImportEntry.
187 ImportEntries m_importEntries;
188
189 // Map exportName -> ExportEntry.
190 ExportEntries m_exportEntries;
191
192 // Save the occurrence order since resolveExport requires it.
193 OrderedIdentifierSet m_starExportEntries;
194
195 // Save the occurrence order since the module loader loads and runs the modules in this order.
196 // http://www.ecma-international.org/ecma-262/6.0/#sec-moduleevaluation
197 OrderedIdentifierSet m_requestedModules;
198
199 WriteBarrier<JSMap> m_dependenciesMap;
200
201 WriteBarrier<JSModuleNamespaceObject> m_moduleNamespaceObject;
202
203 WriteBarrier<JSModuleEnvironment> m_moduleEnvironment;
204
205 // We assume that all the AbstractModuleRecord are retained by JSModuleLoader's registry.
206 // So here, we don't visit each object for GC. The resolution cache map caches the once
207 // looked up correctly resolved resolution, since (1) we rarely looked up the non-resolved one,
208 // and (2) if we cache all the attempts the size of the map becomes infinitely large.
209 typedef HashMap<RefPtr<UniquedStringImpl>, Resolution, IdentifierRepHash, HashTraits<RefPtr<UniquedStringImpl>>> Resolutions;
210 Resolutions m_resolutionCache;
211};
