26#pragma once
27
28#include <wtf/PrintStream.h>
29#include <wtf/text/ASCIILiteral.h>
30
31namespace JSC {
32
33enum class ArrayBufferSharingMode : uint8_t {
34 Default = 0,
35 Shared
36};
37
38inline ASCIILiteral arrayBufferSharingModeName(ArrayBufferSharingMode sharingMode)
39{
40 switch (sharingMode) {
41 case ArrayBufferSharingMode::Default:
42 return "ArrayBuffer"_s;
43 case ArrayBufferSharingMode::Shared:
44 return "SharedArrayBuffer"_s;
45 }
46 RELEASE_ASSERT_NOT_REACHED();
47 return ASCIILiteral::null();
48}
49
50} // namespace JSC
51
52namespace WTF {
53
54inline void printInternal(PrintStream& out, JSC::ArrayBufferSharingMode mode)
55{
56 out.print(JSC::arrayBufferSharingModeName(mode));
57}
58
59} // namespace WTF
