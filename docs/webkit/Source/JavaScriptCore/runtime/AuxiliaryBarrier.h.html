28namespace JSC {
29
30class JSCell;
31class VM;
32
33// An Auxiliary barrier is a barrier that does not try to reason about the value being stored into
34// it, other than interpreting a falsy value as not needing a barrier. It's OK to use this for either
35// JSCells or any other kind of data, so long as it responds to operator!().
36template<typename T>
37class AuxiliaryBarrier {
38public:
39 typedef T Type;
40
41 AuxiliaryBarrier(): m_value() { }
42
43 template<typename U>
44 AuxiliaryBarrier(VM&, JSCell*, U&&);
45
46 void clear() { m_value = T(); }
47
48 template<typename U>
49 void set(VM&, JSCell*, U&&);
50
51 const T& get() const { return m_value; }
52
53 T* slot() { return &m_value; }
54
55 explicit operator bool() const { return !!m_value; }
56
57 template<typename U>
58 void setWithoutBarrier(U&& value) { m_value = std::forward<U>(value); }
59
60 T operator->() const { return get(); }
61
62private:
63 T m_value;
64};
65
66} // namespace JSC
