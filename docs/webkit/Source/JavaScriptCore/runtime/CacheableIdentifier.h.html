37using WTF::PrintStream;
38using WTF::UniquedStringImpl;

40namespace JSC {
41
42class CodeBlock;
43class Identifier;
44class JSCell;
45
46class CacheableIdentifier {
47public:
48 CacheableIdentifier() = default;
49
50 static inline CacheableIdentifier createFromCell(JSCell* identifier);
51 static inline CacheableIdentifier createFromIdentifierOwnedByCodeBlock(CodeBlock*, const Identifier&);
52 static inline CacheableIdentifier createFromIdentifierOwnedByCodeBlock(CodeBlock*, UniquedStringImpl*);
53 static inline CacheableIdentifier createFromImmortalIdentifier(UniquedStringImpl*);
54 static constexpr CacheableIdentifier createFromRawBits(uintptr_t rawBits) { return CacheableIdentifier(rawBits); }
55
56 CacheableIdentifier(const CacheableIdentifier&) = default;
57 CacheableIdentifier(CacheableIdentifier&&) = default;
58
59 CacheableIdentifier(std::nullptr_t)
60 : m_bits(0)
61 { }
62
63 bool isUid() const { return m_bits & s_uidTag; }
64 bool isCell() const { return !isUid(); }
65 inline bool isSymbolCell() const;
66 inline bool isStringCell() const;
67
68 bool isSymbol() const { return m_bits && uid()->isSymbol(); }
69
70 inline JSCell* cell() const;
71 UniquedStringImpl* uid() const;
72
73 explicit operator bool() const { return m_bits; }
74
75 unsigned hash() const { return uid()->symbolAwareHash(); }
76
77 CacheableIdentifier& operator=(const CacheableIdentifier&) = default;
78 CacheableIdentifier& operator=(CacheableIdentifier&&) = default;
79
80 bool operator==(const CacheableIdentifier& other) const;
81 bool operator!=(const CacheableIdentifier& other) const;
82 bool operator==(const Identifier& other) const;
83
84 static inline bool isCacheableIdentifierCell(JSCell*);
85 static inline bool isCacheableIdentifierCell(JSValue);
86
87 uintptr_t rawBits() const { return m_bits; }
88
89 template<typename Visitor> inline void visitAggregate(Visitor&) const;
90
91 JS_EXPORT_PRIVATE void dump(PrintStream&) const;
92
93private:
94 explicit inline CacheableIdentifier(UniquedStringImpl*);
95 explicit inline CacheableIdentifier(JSCell* identifier);
96 explicit constexpr CacheableIdentifier(uintptr_t rawBits)
97 : m_bits(rawBits)
98 { }
99
100 inline void setCellBits(JSCell*);
101 inline void setUidBits(UniquedStringImpl*);
102
103 // CacheableIdentifier can either hold a cell pointer or a uid. To discern which
104 // it is holding, we tag the low bit if we have a uid. We choose to tag the uid
105 // instead of the cell because this keeps the bits of the cell pointer form
106 // unpolluted, and therefore, it can be scanned by our conservative GC to keep the
107 // cell alive when the CacheableIdentifier is on the stack.
108 static constexpr uintptr_t s_uidTag = 1;
109 uintptr_t m_bits { 0 };
110};
111
112} // namespace JSC
