64inline CacheableIdentifier::CacheableIdentifier(JSCell* identifier)

70inline JSCell* CacheableIdentifier::cell() const
71{
72 ASSERT(isCell());
73 return bitwise_cast<JSCell*>(m_bits);
74}
75
76inline UniquedStringImpl* CacheableIdentifier::uid() const
77{
78 if (!m_bits)
79 return nullptr;
80 if (isUid())
81 return bitwise_cast<UniquedStringImpl*>(m_bits & ~s_uidTag);
82 if (isSymbolCell())
83 return &jsCast<Symbol*>(cell())->uid();
84 ASSERT(isStringCell());
85 JSString* string = jsCast<JSString*>(cell());
86 return bitwise_cast<UniquedStringImpl*>(string->getValueImpl());
87}
88
89inline bool CacheableIdentifier::isCacheableIdentifierCell(JSCell* cell)
90{
91 if (cell->isSymbol())
92 return true;
93 if (!cell->isString())
94 return false;
95 JSString* string = jsCast<JSString*>(cell);
96 if (const StringImpl* impl = string->tryGetValueImpl())
97 return impl->isAtom();
98 return false;
99}
100
101inline bool CacheableIdentifier::isCacheableIdentifierCell(JSValue value)
102{
103 if (!value.isCell())
104 return false;
105 return isCacheableIdentifierCell(value.asCell());
106}
107
108inline bool CacheableIdentifier::isSymbolCell() const
109{
110 return isCell() && cell()->isSymbol();
111}
112
113inline bool CacheableIdentifier::isStringCell() const
114{
115 return isCell() && cell()->isString();
116}
117
118inline void CacheableIdentifier::setCellBits(JSCell* cell)
119{
120 RELEASE_ASSERT(isCacheableIdentifierCell(cell));
121 m_bits = bitwise_cast<uintptr_t>(cell);
122}
123
124inline void CacheableIdentifier::setUidBits(UniquedStringImpl* uid)
125{
126 m_bits = bitwise_cast<uintptr_t>(uid) | s_uidTag;
127}
128
129template<typename Visitor>
130inline void CacheableIdentifier::visitAggregate(Visitor& visitor) const
131{
132 if (m_bits && isCell())
133 visitor.appendUnbarriered(cell());
134}

136
137inline bool CacheableIdentifier::operator==(const CacheableIdentifier& other) const
138{
139 return uid() == other.uid();
140}
141
142inline bool CacheableIdentifier::operator!=(const CacheableIdentifier& other) const
143{
144 return uid() != other.uid();
145}
146
147inline bool CacheableIdentifier::operator==(const Identifier& other) const
148{
149 return uid() == other.impl();
150}
