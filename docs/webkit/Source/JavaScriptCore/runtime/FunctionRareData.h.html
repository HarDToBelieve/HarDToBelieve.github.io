146
147 // Ideally, there would only be one allocation profile for subclassing but due to Reflect.construct we
148 // have two. There are some pros and cons in comparison to our current system to using the same profile
149 // for both JS constructors and subclasses of builtin constructors:
150 //
151 // 1) + Uses less memory.
152 // 2) + Conceptually simplier as there is only one profile.
153 // 3) - We would need a check in all JSFunction object creations (both with classes and without) that the
154 // new.target's profiled structure has a JSFinalObject ClassInfo. This is needed, for example, if we have
155 // `Reflect.construct(Array, args, myConstructor)` since myConstructor will be the new.target of Array
156 // the Array constructor will set the allocation profile of myConstructor to hold an Array structure
157 //
158 // We don't really care about 1) since this memory is rare and small in total. 2) is unfortunate but is
159 // probably outweighed by the cost of 3).
160 ObjectAllocationProfileWithPrototype m_objectAllocationProfile;
161 InlineWatchpointSet m_allocationProfileWatchpointSet;
162 InternalFunctionAllocationProfile m_internalFunctionAllocationProfile;
163 WriteBarrier<Structure> m_boundFunctionStructure;
164 WriteBarrier<ExecutableBase> m_executable;
165 std::unique_ptr<AllocationProfileClearingWatchpoint> m_allocationProfileClearingWatchpoint;
166 bool m_hasReifiedLength : 1;
167 bool m_hasReifiedName : 1;
168 bool m_hasModifiedLengthForNonHostFunction : 1;
169 bool m_hasModifiedNameForNonHostFunction : 1;
170};

172class FunctionRareData::AllocationProfileClearingWatchpoint final : public Watchpoint {
173public:
174 AllocationProfileClearingWatchpoint(FunctionRareData* rareData)
175 : Watchpoint(Watchpoint::Type::FunctionRareDataAllocationProfileClearing)
176 , m_rareData(rareData)
177 { }

179 void fireInternal(VM&, const FireDetail&);
180
181private:
182 // Own destructor may not be called. Keep members trivially destructible.
183 JSC_WATCHPOINT_FIELD(PackedCellPtr<FunctionRareData>, m_rareData);
184};

186inline Watchpoint* FunctionRareData::createAllocationProfileClearingWatchpoint()
187{
188 RELEASE_ASSERT(!hasAllocationProfileClearingWatchpoint());
189 m_allocationProfileClearingWatchpoint = makeUnique<AllocationProfileClearingWatchpoint>(this);
190 return m_allocationProfileClearingWatchpoint.get();
191}
