78enum class Gate : uint8_t {
79#define JSC_DEFINE_GATE_ENUM(gateName, tag) gateName,
80#define JSC_DEFINE_OPCODE_GATE_ENUM(gateName, tag) gateName, gateName##_wide16, gateName##_wide32,
81 JSC_UTILITY_GATES(JSC_DEFINE_GATE_ENUM)
82 JSC_JS_GATE_OPCODES(JSC_DEFINE_OPCODE_GATE_ENUM)
83 JSC_WASM_GATE_OPCODES(JSC_DEFINE_OPCODE_GATE_ENUM)
84#undef JSC_DEFINE_GATE_ENUM
85};
