28#include "CagedBarrierPtr.h"
29#include "JSObject.h"

31namespace JSC {
32
33// This is a mixin for the two kinds of Arguments-class objects that arise when you say
34// "arguments" inside a function. This class doesn't show up in the JSCell inheritance hierarchy.
35template<typename Type>
36class GenericArguments : public JSNonFinalObject {
37public:
38 typedef JSNonFinalObject Base;
39 static constexpr unsigned StructureFlags = Base::StructureFlags | OverridesGetOwnPropertySlot | OverridesGetOwnPropertyNames | OverridesPut | InterceptsGetOwnPropertySlotByIndexEvenWhenLengthIsNotZero;
40
41protected:
42 GenericArguments(VM& vm, Structure* structure)
43 : Base(vm, structure)
44 {
45 }
46
47 DECLARE_VISIT_CHILDREN;
48 static bool getOwnPropertySlot(JSObject*, JSGlobalObject*, PropertyName, PropertySlot&);
49 static bool getOwnPropertySlotByIndex(JSObject*, JSGlobalObject*, unsigned propertyName, PropertySlot&);
50 static void getOwnPropertyNames(JSObject*, JSGlobalObject*, PropertyNameArray&, DontEnumPropertiesMode);
51 static bool put(JSCell*, JSGlobalObject*, PropertyName, JSValue, PutPropertySlot&);
52 static bool putByIndex(JSCell*, JSGlobalObject*, unsigned propertyName, JSValue, bool shouldThrow);
53 static bool deleteProperty(JSCell*, JSGlobalObject*, PropertyName, DeletePropertySlot&);
54 static bool deletePropertyByIndex(JSCell*, JSGlobalObject*, unsigned propertyName);
55 static bool defineOwnProperty(JSObject*, JSGlobalObject*, PropertyName, const PropertyDescriptor&, bool shouldThrow);
56
57 void initModifiedArgumentsDescriptor(JSGlobalObject*, unsigned length);
58 void initModifiedArgumentsDescriptorIfNecessary(JSGlobalObject*, unsigned length);
59 void setModifiedArgumentDescriptor(JSGlobalObject*, unsigned index, unsigned length);
60 bool isModifiedArgumentDescriptor(unsigned index, unsigned length);
61
62 void copyToArguments(JSGlobalObject*, JSValue* firstElementDest, unsigned offset, unsigned length);
63
64 using ModifiedArgumentsPtr = CagedBarrierPtr<Gigacage::Primitive, bool>;
65 ModifiedArgumentsPtr m_modifiedArgumentsDescriptor;
66};
67
68} // namespace JSC
