33template<typename JSCellType>
34class InferredValue {
35 WTF_MAKE_NONCOPYABLE(InferredValue);
36 WTF_MAKE_NONMOVABLE(InferredValue);
37public:
38 // For the purpose of deciding whether or not to watch this variable, you only need
39 // to inspect inferredValue(). If this returns something other than the empty
40 // value, then it means that at all future safepoints, this watchpoint set will be
41 // in one of these states:
42 //
43 // IsWatched: in this case, the variable's value must still be the
44 // inferredValue.
45 //
46 // IsInvalidated: in this case the variable's value may be anything but you'll
47 // either notice that it's invalidated and not install the watchpoint, or
48 // you will have been notified that the watchpoint was fired.
49 JSCellType* inferredValue()
50 {
51 uintptr_t data = m_data;
52 if (isFat(data))
53 return fat(data)->inferredValue();
54 return bitwise_cast<JSCellType*>(data & ValueMask);
55 }
56
57 explicit InferredValue()
58 : m_data(encodeState(ClearWatchpoint))
59 {
60 ASSERT(inferredValue() == nullptr);
61 }
62
63 ~InferredValue()
64 {
65 if (isThin())
66 return;
67 freeFat();
68 }
69
70 // Fast way of getting the state, which only works from the main thread.
71 WatchpointState stateOnJSThread() const
72 {
73 uintptr_t data = m_data;
74 if (isFat(data))
75 return fat(data)->stateOnJSThread();
76 return decodeState(data);
77 }
78
79 // It is safe to call this from another thread. It may return a prior state,
80 // but that should be fine since you should only perform actions based on the
81 // state if you also add a watchpoint.
82 WatchpointState state() const
83 {
84 uintptr_t data = m_data;
85 if (isFat(data))
86 return fat(data)->state();
87 return decodeState(data);
88 }
89
90 // It is safe to call this from another thread. It may return false
91 // even if the set actually had been invalidated, but that ought to happen
92 // only in the case of races, and should be rare.
93 bool hasBeenInvalidated() const
94 {
95 return state() == IsInvalidated;
96 }
97
98 // Like hasBeenInvalidated(), may be called from another thread.
99 bool isStillValid() const
100 {
101 return !hasBeenInvalidated();
102 }
103
104 void add(Watchpoint*);
105
106 void invalidate(VM& vm, const FireDetail& detail)
107 {
108 if (isFat())
109 fat()->invalidate(vm, detail);
110 else
111 m_data = encodeState(IsInvalidated);
112 }
113
114 bool isBeingWatched() const
115 {
116 if (isFat())
117 return fat()->isBeingWatched();
118 return false;
119 }
120
121 void notifyWrite(VM& vm, JSCell* owner, JSCellType* value, const FireDetail& detail)
122 {
123 if (LIKELY(stateOnJSThread() == IsInvalidated))
124 return;
125 notifyWriteSlow(vm, owner, value, detail);
126 }
127
128 void notifyWrite(VM& vm, JSCell* owner, JSCellType* value, const char* reason)
129 {
130 if (LIKELY(stateOnJSThread() == IsInvalidated))
131 return;
132 notifyWriteSlow(vm, owner, value, reason);
133 }
134
135 void finalizeUnconditionally(VM&);
136
137private:
138 class InferredValueWatchpointSet final : public WatchpointSet {
139 public:
140 InferredValueWatchpointSet(WatchpointState state, JSCellType* value)
141 : WatchpointSet(state)
142 , m_value(value)
143 {
144 }
145
146 JSCellType* inferredValue() const { return m_value; }
147
148 void invalidate(VM& vm, const FireDetail& detail)
149 {
150 m_value = nullptr;
151 WatchpointSet::invalidate(vm, detail);
152 }
153
154 void notifyWriteSlow(VM&, JSCell* owner, JSCellType*, const FireDetail&);
155
156 private:
157 JSCellType* m_value;
158 };
159
160 static constexpr uintptr_t IsThinFlag = 1;
161 static constexpr uintptr_t StateMask = 6;
162 static constexpr uintptr_t StateShift = 1;
163 static constexpr uintptr_t ValueMask = ~static_cast<uintptr_t>(IsThinFlag | StateMask);
164
165 static bool isThin(uintptr_t data) { return data & IsThinFlag; }
166 static bool isFat(uintptr_t data) { return !isThin(data); }
167
168 static WatchpointState decodeState(uintptr_t data)
169 {
170 ASSERT(isThin(data));
171 return static_cast<WatchpointState>((data & StateMask) >> StateShift);
172 }
173
174 static uintptr_t encodeState(WatchpointState state)
175 {
176 return (static_cast<uintptr_t>(state) << StateShift) | IsThinFlag;
177 }
178
179 bool isThin() const { return isThin(m_data); }
180 bool isFat() const { return isFat(m_data); };
181
182 static InferredValueWatchpointSet* fat(uintptr_t data)
183 {
184 return bitwise_cast<InferredValueWatchpointSet*>(data);
185 }
186
187 InferredValueWatchpointSet* fat()
188 {
189 ASSERT(isFat());
190 return fat(m_data);
191 }
192
193 const InferredValueWatchpointSet* fat() const
194 {
195 ASSERT(isFat());
196 return fat(m_data);
197 }
198
199 InferredValueWatchpointSet* inflate()
200 {
201 if (LIKELY(isFat()))
202 return fat();
203 return inflateSlow();
204 }
205
206 InferredValueWatchpointSet* inflateSlow();
207 void freeFat();
208
209 void notifyWriteSlow(VM&, JSCell* owner, JSCellType*, const FireDetail&);
210 void notifyWriteSlow(VM&, JSCell* owner, JSCellType*, const char* reason);
211
212 uintptr_t m_data;
213};
214
215template<typename JSCellType>
216void InferredValue<JSCellType>::InferredValueWatchpointSet::notifyWriteSlow(VM& vm, JSCell* owner, JSCellType* value, const FireDetail& detail)
217{
218 switch (state()) {
219 case ClearWatchpoint:
220 m_value = value;
221 vm.heap.writeBarrier(owner, value);
222 startWatching();
223 return;
224
225 case IsWatched:
226 ASSERT(!!m_value);
227 if (m_value == value)
228 return;
229 invalidate(vm, detail);
230 return;
231
232 case IsInvalidated:
233 ASSERT_NOT_REACHED();
234 return;
235 }
236
237 ASSERT_NOT_REACHED();
238}
239
240template<typename JSCellType>
241void InferredValue<JSCellType>::notifyWriteSlow(VM& vm, JSCell* owner, JSCellType* value, const FireDetail& detail)
242{
243 uintptr_t data = m_data;
244 if (isFat(data)) {
245 fat(data)->notifyWriteSlow(vm, owner, value, detail);
246 return;
247 }
248
249 switch (state()) {
250 case ClearWatchpoint:
251 ASSERT(decodeState(m_data) != IsInvalidated);
252 m_data = (bitwise_cast<uintptr_t>(value) & ValueMask) | encodeState(IsWatched);
253 vm.heap.writeBarrier(owner, value);
254 return;
255
256 case IsWatched:
257 ASSERT(!!inferredValue());
258 if (inferredValue() == value)
259 return;
260 invalidate(vm, detail);
261 return;
262
263 case IsInvalidated:
264 ASSERT_NOT_REACHED();
265 return;
266 }
267
268 ASSERT_NOT_REACHED();
269}

271template<typename JSCellType>
272void InferredValue<JSCellType>::notifyWriteSlow(VM& vm, JSCell* owner, JSCellType* value, const char* reason)
273{
274 notifyWriteSlow(vm, owner, value, StringFireDetail(reason));
275}

277template<typename JSCellType>
278void InferredValue<JSCellType>::add(Watchpoint* watchpoint)
279{
280 inflate()->add(watchpoint);
281}
282
283template<typename JSCellType>
284auto InferredValue<JSCellType>::inflateSlow() -> InferredValueWatchpointSet*
285{
286 ASSERT(isThin());
287 ASSERT(!isCompilationThread());
288 uintptr_t data = m_data;
289 InferredValueWatchpointSet* fat = adoptRef(new InferredValueWatchpointSet(decodeState(m_data), bitwise_cast<JSCellType*>(data & ValueMask))).leakRef();
290 WTF::storeStoreFence();
291 m_data = bitwise_cast<uintptr_t>(fat);
292 return fat;
293}
294
295template<typename JSCellType>
296void InferredValue<JSCellType>::freeFat()
297{
298 ASSERT(isFat());
299 fat()->deref();
300}
