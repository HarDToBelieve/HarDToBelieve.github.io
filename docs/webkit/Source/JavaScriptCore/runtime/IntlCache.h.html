26#pragma once
27
28#include <unicode/udatpg.h>
29#include <wtf/Noncopyable.h>
30#include <wtf/text/CString.h>
31#include <wtf/unicode/icu/ICUHelpers.h>
32
33namespace JSC {
34
35class IntlCache {
36 WTF_MAKE_NONCOPYABLE(IntlCache);
37 WTF_MAKE_FAST_ALLOCATED;
38public:
39 IntlCache() = default;
40
41 Vector<UChar, 32> getBestDateTimePattern(const CString& locale, const UChar* skeleton, unsigned skeletonSize, UErrorCode&);
42
43private:
44 UDateTimePatternGenerator* getSharedPatternGenerator(const CString& locale, UErrorCode& status)
45 {
46 if (m_cachedDateTimePatternGenerator) {
47 if (locale == m_cachedDateTimePatternGeneratorLocale)
48 return m_cachedDateTimePatternGenerator.get();
49 }
50 return cacheSharedPatternGenerator(locale, status);
51 }
52
53 UDateTimePatternGenerator* cacheSharedPatternGenerator(const CString& locale, UErrorCode&);
54
55 std::unique_ptr<UDateTimePatternGenerator, ICUDeleter<udatpg_close>> m_cachedDateTimePatternGenerator;
56 CString m_cachedDateTimePatternGeneratorLocale;
57};
58
59} // namespace JSC
