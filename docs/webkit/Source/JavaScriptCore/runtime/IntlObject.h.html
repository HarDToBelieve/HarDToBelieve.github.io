68 template<typename CellType, SubspaceAccess>
69 static IsoSubspace* subspaceFor(VM& vm)
70 {
71 STATIC_ASSERT_ISO_SUBSPACE_SHARABLE(IntlObject, Base);
72 return &vm.plainObjectSpace;
73 }

75 static IntlObject* create(VM&, JSGlobalObject*, Structure*);
76 static Structure* createStructure(VM&, JSGlobalObject*, JSValue);

78 DECLARE_INFO;
