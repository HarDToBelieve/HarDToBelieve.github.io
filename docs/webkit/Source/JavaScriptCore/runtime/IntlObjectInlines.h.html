38template<typename StringType>
39static constexpr uint32_t computeTwoCharacters16Code(const StringType& string)
40{
41 return static_cast<uint16_t>(string.characterAt(0)) | (static_cast<uint32_t>(static_cast<uint16_t>(string.characterAt(1))) << 16);
42}

44template<typename Predicate> String bestAvailableLocale(const String& locale, Predicate predicate)
45{
46 // BestAvailableLocale (availableLocales, locale)
47 // https://tc39.github.io/ecma402/#sec-bestavailablelocale
48
49 String candidate = locale;
50 while (!candidate.isEmpty()) {
51 if (predicate(candidate))
52 return candidate;
53
54 size_t pos = candidate.reverseFind('-');
55 if (pos == notFound)
56 return String();
57
58 if (pos >= 2 && candidate[pos - 2] == '-')
59 pos -= 2;
60
61 candidate = candidate.substring(0, pos);
62 }
63
64 return String();
65}

67template<typename Constructor, typename Factory>
68JSValue constructIntlInstanceWithWorkaroundForLegacyIntlConstructor(JSGlobalObject* globalObject, JSValue thisValue, Constructor* callee, Factory factory)
69{
70 // FIXME: Workaround to provide compatibility with ECMA-402 1.0 call/apply patterns.
71 // https://bugs.webkit.org/show_bug.cgi?id=153679
72 VM& vm = globalObject->vm();
73 auto scope = DECLARE_THROW_SCOPE(vm);
74
75 auto* instance = factory(vm);
76 RETURN_IF_EXCEPTION(scope, JSValue());
77
78 if (thisValue.isObject()) {
79 JSObject* thisObject = asObject(thisValue);
80 ASSERT(!callee->template inherits<JSBoundFunction>(vm));
81 JSValue prototype = callee->getDirect(vm, vm.propertyNames->prototype); // Passed constructors always have `prototype` which cannot be deleted.
82 ASSERT(prototype);
83 bool hasInstance = JSObject::defaultHasInstance(globalObject, thisObject, prototype);
84 RETURN_IF_EXCEPTION(scope, JSValue());
85 if (hasInstance) {
86 PropertyDescriptor descriptor(instance, PropertyAttribute::ReadOnly | PropertyAttribute::DontEnum | PropertyAttribute::DontDelete);
87 scope.release();
88 thisObject->methodTable(vm)->defineOwnProperty(thisObject, globalObject, vm.propertyNames->builtinNames().intlLegacyConstructedSymbol(), descriptor, true);
89 return thisObject;
90 }
91 }
92 return instance;
93}

95template<typename InstanceType>
96InstanceType* unwrapForLegacyIntlConstructor(JSGlobalObject* globalObject, JSValue thisValue, JSObject* constructor)
97{
98 VM& vm = globalObject->vm();
99 auto scope = DECLARE_THROW_SCOPE(vm);
100
101 JSObject* thisObject = jsDynamicCast<JSObject*>(vm, thisValue);
102 if (UNLIKELY(!thisObject))
103 return nullptr;
104
105 auto* instance = jsDynamicCast<InstanceType*>(vm, thisObject);
106 if (LIKELY(instance))
107 return instance;
108
109 ASSERT(!constructor->template inherits<JSBoundFunction>(vm));
110 JSValue prototype = constructor->getDirect(vm, vm.propertyNames->prototype); // Passed constructors always have `prototype` which cannot be deleted.
111 ASSERT(prototype);
112 bool hasInstance = JSObject::defaultHasInstance(globalObject, thisObject, prototype);
113 RETURN_IF_EXCEPTION(scope, nullptr);
114 if (!hasInstance)
115 return nullptr;
116
117 JSValue value = thisObject->get(globalObject, vm.propertyNames->builtinNames().intlLegacyConstructedSymbol());
118 RETURN_IF_EXCEPTION(scope, nullptr);
119 return jsDynamicCast<InstanceType*>(vm, value);
120}

122template<typename ResultType>
123ResultType intlOption(JSGlobalObject* globalObject, JSObject* options, PropertyName property, std::initializer_list<std::pair<ASCIILiteral, ResultType>> values, ASCIILiteral notFoundMessage, ResultType fallback)
124{
125 // GetOption (options, property, type="string", values, fallback)
126 // https://tc39.github.io/ecma402/#sec-getoption
127
128 ASSERT(values.size() > 0);
129
130 VM& vm = globalObject->vm();
131 auto scope = DECLARE_THROW_SCOPE(vm);
132
133 if (!options)
134 return fallback;
135
136 JSValue value = options->get(globalObject, property);
137 RETURN_IF_EXCEPTION(scope, { });
138
139 if (!value.isUndefined()) {
140 String stringValue = value.toWTFString(globalObject);
141 RETURN_IF_EXCEPTION(scope, { });
142
143 for (const auto& entry : values) {
144 if (entry.first == stringValue)
145 return entry.second;
146 }
147 throwException(globalObject, scope, createRangeError(globalObject, notFoundMessage));
148 return { };
149 }
150
151 return fallback;
152}

155ALWAYS_INLINE bool canUseASCIIUCADUCETComparison(UChar character)
156{
157 return isASCII(character) && ducetLevel1Weights[character];
158}

160template<typename CharacterType1, typename CharacterType2>
161UCollationResult compareASCIIWithUCADUCETLevel3(const CharacterType1* characters1, const CharacterType2* characters2, unsigned length)
162{
163 for (unsigned position = 0; position < length; ++position) {
164 auto lhs = characters1[position];
165 auto rhs = characters2[position];
166 uint8_t leftWeight = ducetLevel3Weights[lhs];
167 uint8_t rightWeight = ducetLevel3Weights[rhs];
168 if (leftWeight == rightWeight)
169 continue;
170 return leftWeight > rightWeight ? UCOL_GREATER : UCOL_LESS;
171 }
172 return UCOL_EQUAL;
173}
174
175template<typename CharacterType1, typename CharacterType2>
176inline UCollationResult compareASCIIWithUCADUCET(const CharacterType1* characters1, unsigned length1, const CharacterType2* characters2, unsigned length2)
177{
178 bool notSameCharacters = false;
179 unsigned commonLength = std::min(length1, length2);
180 for (unsigned position = 0; position < commonLength; ++position) {
181 auto lhs = characters1[position];
182 auto rhs = characters2[position];
183 ASSERT(canUseASCIIUCADUCETComparison(lhs));
184 ASSERT(canUseASCIIUCADUCETComparison(rhs));
185 if (lhs == rhs)
186 continue;
187 notSameCharacters = true;
188 uint8_t leftWeight = ducetLevel1Weights[lhs];
189 uint8_t rightWeight = ducetLevel1Weights[rhs];
190 if (leftWeight == rightWeight)
191 continue;
192 return leftWeight > rightWeight ? UCOL_GREATER : UCOL_LESS;
193 }
194
195 if (length1 == length2) {
196 if (notSameCharacters)
197 return compareASCIIWithUCADUCETLevel3(characters1, characters2, length1);
198 return UCOL_EQUAL;
199 }
200 return length1 > length2 ? UCOL_GREATER : UCOL_LESS;
201}

203// https://tc39.es/ecma402/#sec-getoptionsobject
204inline JSObject* intlGetOptionsObject(JSGlobalObject* globalObject, JSValue options)
205{
206 VM& vm = globalObject->vm();
207 auto scope = DECLARE_THROW_SCOPE(vm);
208 if (options.isUndefined())
209 return nullptr;
210 if (LIKELY(options.isObject()))
211 return asObject(options);
212 throwTypeError(globalObject, scope, "options argument is not an object or undefined"_s);
213 return nullptr;
214}

216// https://tc39.es/ecma402/#sec-coerceoptionstoobject
217inline JSObject* intlCoerceOptionsToObject(JSGlobalObject* globalObject, JSValue optionsValue)
218{
219 VM& vm = globalObject->vm();
220 auto scope = DECLARE_THROW_SCOPE(vm);
221 if (optionsValue.isUndefined())
222 return nullptr;
223 JSObject* options = optionsValue.toObject(globalObject);
224 RETURN_IF_EXCEPTION(scope, nullptr);
225 return options;
226}
