27#pragma once
28
29#include "JSFunction.h"
30
31namespace JSC {
32
33class JSAsyncFunction final : public JSFunction {
34 friend class JIT;
35 friend class VM;
36public:
37 typedef JSFunction Base;
38
39 static constexpr unsigned StructureFlags = Base::StructureFlags;
40
41 DECLARE_EXPORT_INFO;
42
43 static JSAsyncFunction* create(VM&, FunctionExecutable*, JSScope*);
44 static JSAsyncFunction* create(VM&, FunctionExecutable*, JSScope*, Structure*);
45 static JSAsyncFunction* createWithInvalidatedReallocationWatchpoint(VM&, FunctionExecutable*, JSScope*);
46
47 static size_t allocationSize(Checked<size_t> inlineCapacity)
48 {
49 ASSERT_UNUSED(inlineCapacity, !inlineCapacity);
50 return sizeof(JSAsyncFunction);
51 }
52
53 static Structure* createStructure(VM& vm, JSGlobalObject* globalObject, JSValue prototype)
54 {
55 ASSERT(globalObject);
56 return Structure::create(vm, globalObject, prototype, TypeInfo(JSFunctionType, StructureFlags), info());
57 }
58
59private:
60 JSAsyncFunction(VM&, FunctionExecutable*, JSScope*, Structure*);
61
62 static JSAsyncFunction* createImpl(VM&, FunctionExecutable*, JSScope*, Structure*);
63};
64static_assert(sizeof(JSAsyncFunction) == sizeof(JSFunction), "Some subclasses of JSFunction should be the same size to share IsoSubspace");
65
66} // namespace JSC
