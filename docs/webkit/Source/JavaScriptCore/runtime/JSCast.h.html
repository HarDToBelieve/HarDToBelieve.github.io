32template<typename To, typename From>
33inline To jsCast(From* from)
34{
35 static_assert(std::is_base_of<JSCell, typename std::remove_pointer<To>::type>::value && std::is_base_of<JSCell, typename std::remove_pointer<From>::type>::value, "JS casting expects that the types you are casting to/from are subclasses of JSCell");
36#if (ASSERT_ENABLED || ENABLE(SECURITY_ASSERTIONS)) && CPU(X86_64)
37 if (from && !from->JSCell::inherits(from->JSCell::vm(), std::remove_pointer<To>::type::info()))
38 reportZappedCellAndCrash(*from->JSCell::heap(), from);
39#else
40 ASSERT_WITH_SECURITY_IMPLICATION(!from || from->JSCell::inherits(from->JSCell::vm(), std::remove_pointer<To>::type::info()));
41#endif
42 return static_cast<To>(from);
43}
44
45template<typename To>
46inline To jsCast(JSValue from)
47{
48 static_assert(std::is_base_of<JSCell, typename std::remove_pointer<To>::type>::value, "JS casting expects that the types you are casting to is a subclass of JSCell");
49#if (ASSERT_ENABLED || ENABLE(SECURITY_ASSERTIONS)) && CPU(X86_64)
50 ASSERT_WITH_SECURITY_IMPLICATION(from.isCell());
51 JSCell* cell = from.asCell();
52 if (!cell->JSCell::inherits(cell->vm(), std::remove_pointer<To>::type::info()))
53 reportZappedCellAndCrash(*cell->JSCell::heap(), cell);
54#else
55 ASSERT_WITH_SECURITY_IMPLICATION(from.isCell() && from.asCell()->JSCell::inherits(from.asCell()->vm(), std::remove_pointer<To>::type::info()));
56#endif
57 return static_cast<To>(from.asCell());
58}

60// The first and last JSType are inclusive
61struct JSTypeRange {
62 bool contains(JSType type) const { return first <= type && type <= last; }
63
64 JSType first;
65 JSType last;
66};

111template<bool isFinal>
112struct FinalTypeDispatcher {
113 template<typename Target, typename From>
114 static inline bool inheritsGeneric(VM& vm, From* from)
115 {
116 static_assert(!std::is_same<JSObject*, Target*>::value, "This ensures our overloads work");
117 static_assert(std::is_base_of<JSCell, Target>::value && std::is_base_of<JSCell, typename std::remove_pointer<From>::type>::value, "JS casting expects that the types you are casting to/from are subclasses of JSCell");
118 // Do not use inherits<Target>(vm) since inherits<T> depends on this function.
119 return from->JSCell::inherits(vm, Target::info());
120 }
121};
122
123template<>
124struct FinalTypeDispatcher</* isFinal */ true> {
125 template<typename Target, typename From>
126 static inline bool inheritsGeneric(VM& vm, From* from)
127 {
128 static_assert(!std::is_same<JSObject*, Target*>::value, "This ensures our overloads work");
129 static_assert(std::is_base_of<JSCell, Target>::value && std::is_base_of<JSCell, typename std::remove_pointer<From>::type>::value, "JS casting expects that the types you are casting to/from are subclasses of JSCell");
130 static_assert(std::is_final<Target>::value, "Target is a final type");
131 bool canCast = from->JSCell::classInfo(vm) == Target::info();
132 // Do not use inherits<Target>(vm) since inherits<T> depends on this function.
133 ASSERT_UNUSED(vm, canCast == from->JSCell::inherits(vm, Target::info()));
134 return canCast;
135 }
136};

138template<typename Target, typename From>
139inline bool inheritsJSTypeImpl(VM& vm, From* from, JSTypeRange range)
140{
141 static_assert(std::is_base_of<JSCell, Target>::value && std::is_base_of<JSCell, typename std::remove_pointer<From>::type>::value, "JS casting expects that the types you are casting to/from are subclasses of JSCell");
142 bool canCast = range.contains(from->type());
143 // Do not use inherits<Target>(vm) since inherits<T> depends on this function.
144 ASSERT_UNUSED(vm, canCast == from->JSCell::inherits(vm, Target::info()));
145 return canCast;
146}

148// C++ has bad syntax so we need to use this struct because C++ doesn't have a
149// way to say that we are overloading just the first type in a template list...
150template<typename Target>
151struct InheritsTraits {
152 static constexpr std::optional<JSTypeRange> typeRange { std::nullopt };
153 template<typename From>
154 static inline bool inherits(VM& vm, From* from) { return FinalTypeDispatcher<std::is_final<Target>::value>::template inheritsGeneric<Target>(vm, from); }
155};
156
157#define DEFINE_TRAITS_FOR_JS_TYPE_OVERLOAD(className, firstJSType, lastJSType) \
158 template<> \
159 struct InheritsTraits<className> { \
160 static constexpr std::optional<JSTypeRange> typeRange { { static_cast<JSType>(firstJSType), static_cast<JSType>(lastJSType) } }; \
161 template<typename From> \
162 static inline bool inherits(VM& vm, From* from) { return inheritsJSTypeImpl<className, From>(vm, from, *typeRange); } \
163 }; \
164
165FOR_EACH_JS_DYNAMIC_CAST_JS_TYPE_OVERLOAD(DEFINE_TRAITS_FOR_JS_TYPE_OVERLOAD)
166
167#undef DEFINE_TRAITS_FOR_JS_TYPE_OVERLOAD

170template<typename Target, typename From>
171bool inherits(VM& vm, From* from)
172{
173 using Dispatcher = InheritsTraits<Target>;
174 return Dispatcher::template inherits(vm, from);
175}

179template<typename To, typename From>
180To jsDynamicCast(VM& vm, From* from)
181{
182 using Dispatcher = JSCastingHelpers::InheritsTraits<typename std::remove_cv<typename std::remove_pointer<To>::type>::type>;
183 if (LIKELY(Dispatcher::template inherits(vm, from)))
184 return static_cast<To>(from);
185 return nullptr;
186}
187
188template<typename To>
189To jsDynamicCast(VM& vm, JSValue from)
190{
191 if (UNLIKELY(!from.isCell()))
192 return nullptr;
193 return jsDynamicCast<To>(vm, from.asCell());
194}
195
196template<typename To, typename From>
197To jsSecureCast(VM& vm, From from)
198{
199 auto* result = jsDynamicCast<To>(vm, from);
200 RELEASE_ASSERT_WITH_SECURITY_IMPLICATION(result);
201 return result;
202}
