120// Get the combined UTC + DST offset for the time passed in.
121//
122// NOTE: The implementation relies on the fact that no time zones have
123// more than one daylight savings offset change per month.
124// If this function is called with NaN it returns random value.
125LocalTimeOffset DateCache::calculateLocalTimeOffset(double millisecondsFromEpoch, WTF::TimeType inputTimeType)
126{
127 int32_t rawOffset = 0;
128 int32_t dstOffset = 0;
129 UErrorCode status = U_ZERO_ERROR;
130
131 // This function can fail input date is invalid: NaN etc.
132 // We can return any values in this case since later we fail when computing non timezone offset part anyway.
133 constexpr LocalTimeOffset failed { false, 0 };
134
135#if HAVE(ICU_C_TIMEZONE_API)
136 auto& timeZoneCache = *this->timeZoneCache();
137 ucal_setMillis(timeZoneCache.m_calendar.get(), millisecondsFromEpoch, &status);
138 if (U_FAILURE(status))
139 return failed;
140
141 if (inputTimeType != WTF::LocalTime) {
142 rawOffset = ucal_get(timeZoneCache.m_calendar.get(), UCAL_ZONE_OFFSET, &status);
143 if (U_FAILURE(status))
144 return failed;
145 dstOffset = ucal_get(timeZoneCache.m_calendar.get(), UCAL_DST_OFFSET, &status);
146 if (U_FAILURE(status))
147 return failed;
148 } else {
149 ucal_getTimeZoneOffsetFromLocal(timeZoneCache.m_calendar.get(), UCAL_TZ_LOCAL_FORMER, UCAL_TZ_LOCAL_FORMER, &rawOffset, &dstOffset, &status);
150 if (U_FAILURE(status))
151 return failed;
152 }
153#else
154 auto& timeZoneCache = *bitwise_cast<icu::TimeZone*>(this->timeZoneCache());
155 if (inputTimeType != WTF::LocalTime) {
156 constexpr bool isLocalTime = false;
157 timeZoneCache.getOffset(millisecondsFromEpoch, isLocalTime, rawOffset, dstOffset, status);
158 if (U_FAILURE(status))
159 return failed;
160 } else {
161 // icu::TimeZone is a timezone instance which inherits icu::BasicTimeZone.
162 // https://unicode-org.atlassian.net/browse/ICU-13705 will move getOffsetFromLocal to icu::TimeZone.
163 static_cast<const icu::BasicTimeZone&>(timeZoneCache).getOffsetFromLocal(millisecondsFromEpoch, icu::BasicTimeZone::kFormer, icu::BasicTimeZone::kFormer, rawOffset, dstOffset, status);
164 if (U_FAILURE(status))
165 return failed;
166 }
167#endif
168
169 return { !!dstOffset, rawOffset + dstOffset };
170}

172LocalTimeOffset DateCache::localTimeOffset(double millisecondsFromEpoch, WTF::TimeType inputTimeType)
173{
174 LocalTimeOffsetCache& cache = inputTimeType == WTF::LocalTime ? m_localTimeOffsetCache : m_utcTimeOffsetCache;
175
176 double start = cache.start;
177 double end = cache.end;
178
179 auto resetCache = [&]() {
180 // Compute the DST offset for the time and shrink the cache interval
181 // to only contain the time. This allows fast repeated DST offset
182 // computations for the same time.
183 LocalTimeOffset offset = calculateLocalTimeOffset(millisecondsFromEpoch, inputTimeType);
184 cache.offset = offset;
185 cache.start = millisecondsFromEpoch;
186 cache.end = millisecondsFromEpoch;
187 cache.incrementStart = WTF::msPerMonth;
188 cache.incrementEnd = WTF::msPerMonth;
189 return offset;
190 };
191
192 // If the time fits in the cached interval, return the cached offset.
193 if (start <= millisecondsFromEpoch && millisecondsFromEpoch <= end)
194 return cache.offset;
195
196 if (start <= millisecondsFromEpoch) {
197 // Compute a possible new interval end.
198 double newEnd = end + cache.incrementEnd;
199 if (!(millisecondsFromEpoch <= newEnd))
200 return resetCache();
201
202 LocalTimeOffset endOffset = calculateLocalTimeOffset(newEnd, inputTimeType);
203 if (cache.offset == endOffset) {
204 // If the offset at the end of the new interval still matches
205 // the offset in the cache, we grow the cached time interval
206 // and return the offset.
207 cache.end = newEnd;
208 cache.incrementStart = WTF::msPerMonth;
209 cache.incrementEnd = WTF::msPerMonth;
210 return endOffset;
211 }
212 LocalTimeOffset offset = calculateLocalTimeOffset(millisecondsFromEpoch, inputTimeType);
213 if (offset == endOffset) {
214 // The offset at the given time is equal to the offset at the
215 // new end of the interval, so that means that we've just skipped
216 // the point in time where the DST offset change occurred. Update
217 // the interval to reflect this and reset the increment.
218 cache.start = millisecondsFromEpoch;
219 cache.end = newEnd;
220 cache.incrementStart = WTF::msPerMonth;
221 cache.incrementEnd = WTF::msPerMonth;
222 } else {
223 // The interval contains a DST offset change and the given time is
224 // before it. Adjust the increment to avoid a linear search for
225 // the offset change point and change the end of the interval.
226 cache.incrementEnd /= 3;
227 cache.end = millisecondsFromEpoch;
228 }
229 // Update the offset in the cache and return it.
230 cache.offset = offset;
231 return offset;
232 }
233
234 // Compute a possible new interval start.
235 double newStart = start - cache.incrementStart;
236 if (!(newStart <= millisecondsFromEpoch))
237 return resetCache();
238
239 LocalTimeOffset startOffset = calculateLocalTimeOffset(newStart, inputTimeType);
240 if (cache.offset == startOffset) {
241 // If the offset at the start of the new interval still matches
242 // the offset in the cache, we grow the cached time interval
243 // and return the offset.
244 cache.start = newStart;
245 cache.incrementStart = WTF::msPerMonth;
246 cache.incrementEnd = WTF::msPerMonth;
247 return startOffset;
248 }
249 LocalTimeOffset offset = calculateLocalTimeOffset(millisecondsFromEpoch, inputTimeType);
250 if (offset == startOffset) {
251 // The offset at the given time is equal to the offset at the
252 // new start of the interval, so that means that we've just skipped
253 // the point in time where the DST offset change occurred. Update
254 // the interval to reflect this and reset the increment.
255 cache.start = newStart;
256 cache.end = millisecondsFromEpoch;
257 cache.incrementStart = WTF::msPerMonth;
258 cache.incrementEnd = WTF::msPerMonth;
259 } else {
260 // The interval contains a DST offset change and the given time is
261 // before it. Adjust the increment to avoid a linear search for
262 // the offset change point and change the end of the interval.
263 cache.incrementStart /= 3;
264 cache.start = millisecondsFromEpoch;
265 }
266 // Update the offset in the cache and return it.
267 cache.offset = offset;
268 return offset;
269}

271static inline double timeToMS(double hour, double min, double sec, double ms)
272{
273 return (((hour * WTF::minutesPerHour + min) * WTF::secondsPerMinute + sec) * WTF::msPerSecond + ms);
274}

276double DateCache::gregorianDateTimeToMS(const GregorianDateTime& t, double milliseconds, WTF::TimeType inputTimeType)
277{
278 double day = dateToDaysFrom1970(t.year(), t.month(), t.monthDay());
279 double ms = timeToMS(t.hour(), t.minute(), t.second(), milliseconds);
280 double localTimeResult = (day * WTF::msPerDay) + ms;
281
282 double localToUTCTimeOffset = inputTimeType == WTF::LocalTime ? localTimeOffset(localTimeResult, inputTimeType).offset : 0;
283
284 return localTimeResult - localToUTCTimeOffset;
285}

287// input is UTC
288void DateCache::msToGregorianDateTime(double millisecondsFromEpoch, WTF::TimeType outputTimeType, GregorianDateTime& tm)
289{
290 LocalTimeOffset localTime;
291 if (outputTimeType == WTF::LocalTime) {
292 localTime = localTimeOffset(millisecondsFromEpoch);
293 millisecondsFromEpoch += localTime.offset;
294 }
295 tm = GregorianDateTime(millisecondsFromEpoch, localTime);
296}

298double DateCache::parseDate(JSGlobalObject* globalObject, VM& vm, const String& date)
299{
300 auto scope = DECLARE_THROW_SCOPE(vm);
301
302 if (date == m_cachedDateString)
303 return m_cachedDateStringValue;
304 auto expectedString = date.tryGetUtf8();
305 if (!expectedString) {
306 if (expectedString.error() == UTF8ConversionError::OutOfMemory)
307 throwOutOfMemoryError(globalObject, scope);
308 // https://tc39.github.io/ecma262/#sec-date-objects section 20.3.3.2 states that:
309 // "Unrecognizable Strings or dates containing illegal element values in the
310 // format String shall cause Date.parse to return NaN."
311 return std::numeric_limits<double>::quiet_NaN();
312 }
313
314 auto parseDateImpl = [this] (const char* dateString) {
315 bool isLocalTime;
316 double value = WTF::parseES5DateFromNullTerminatedCharacters(dateString, isLocalTime);
317 if (std::isnan(value))
318 value = WTF::parseDateFromNullTerminatedCharacters(dateString, isLocalTime);
319
320 if (isLocalTime)
321 value -= localTimeOffset(value, WTF::LocalTime).offset;
322
323 return value;
324 };
325
326 auto dateUtf8 = expectedString.value();
327 double value = parseDateImpl(dateUtf8.data());
328 m_cachedDateString = date;
329 m_cachedDateStringValue = value;
330 return value;
331}

333// https://tc39.es/ecma402/#sec-defaulttimezone
334String DateCache::defaultTimeZone()
335{
336#if HAVE(ICU_C_TIMEZONE_API)
337 auto& timeZone = *timeZoneCache();
338 Vector<UChar, 32> buffer;
339 auto status = callBufferProducingFunction(ucal_getTimeZoneID, timeZone.m_calendar.get(), buffer);
340 if (U_FAILURE(status))
341 return "UTC"_s;
342
343 Vector<UChar, 32> canonicalBuffer;
344 status = callBufferProducingFunction(ucal_getCanonicalTimeZoneID, buffer.data(), buffer.size(), canonicalBuffer, nullptr);
345 if (U_FAILURE(status))
346 return "UTC"_s;
347
348 String canonical = String(canonicalBuffer);
349 if (isUTCEquivalent(canonical))
350 return "UTC"_s;
351
352 return canonical;
353#else
354 icu::UnicodeString timeZoneID;
355 icu::UnicodeString canonicalTimeZoneID;
356 auto& timeZone = *bitwise_cast<icu::TimeZone*>(timeZoneCache());
357 timeZone.getID(timeZoneID);
358
359 UErrorCode status = U_ZERO_ERROR;
360 UBool isSystem = false;
361 icu::TimeZone::getCanonicalID(timeZoneID, canonicalTimeZoneID, isSystem, status);
362 if (U_FAILURE(status))
363 return "UTC"_s;
364
365 String canonical = String(canonicalTimeZoneID.getBuffer(), canonicalTimeZoneID.length());
366 if (isUTCEquivalent(canonical))
367 return "UTC"_s;
368
369 return canonical;
370#endif
371}

382void DateCache::timeZoneCacheSlow()
383{
384 ASSERT(!m_timeZoneCache);
385#if HAVE(ICU_C_TIMEZONE_API)
386 auto* cache = new OpaqueICUTimeZone;
387 Vector<UChar, 32> timeZoneID;
388 auto status = callBufferProducingFunction(ucal_getHostTimeZone, timeZoneID);
389 ASSERT_UNUSED(status, U_SUCCESS(status));
390 cache->m_calendar = std::unique_ptr<UCalendar, ICUDeleter<ucal_close>>(ucal_open(timeZoneID.data(), timeZoneID.size(), "", UCAL_DEFAULT, &status));
391 ASSERT_UNUSED(status, U_SUCCESS(status));
392 ucal_setGregorianChange(cache->m_calendar.get(), minECMAScriptTime, &status); // Ignore "unsupported" error.
393 m_timeZoneCache = std::unique_ptr<OpaqueICUTimeZone, OpaqueICUTimeZoneDeleter>(cache);
394#else
395 // Do not use icu::TimeZone::createDefault. ICU internally has a cache for timezone and createDefault returns this cached value.
396 m_timeZoneCache = std::unique_ptr<OpaqueICUTimeZone, OpaqueICUTimeZoneDeleter>(bitwise_cast<OpaqueICUTimeZone*>(icu::TimeZone::detectHostTimeZone()));
397#endif
398}

400void DateCache::reset()
401{
402 m_timeZoneCache.reset();
403 m_utcTimeOffsetCache = LocalTimeOffsetCache();
404 m_localTimeOffsetCache = LocalTimeOffsetCache();
405 m_cachedDateString = String();
406 m_cachedDateStringValue = std::numeric_limits<double>::quiet_NaN();
407 m_dateInstanceCache.reset();
408}
