32namespace JSC {
33
34class JSMapIterator final : public JSInternalFieldObjectImpl<3> {
35public:
36 using HashMapBucketType = HashMapBucket<HashMapBucketDataKeyValue>;
37 using Base = JSInternalFieldObjectImpl<3>;
38
39 DECLARE_EXPORT_INFO;
40
41 enum class Field : uint8_t {
42 MapBucket = 0,
43 IteratedObject,
44 Kind,
45 };
46 static_assert(numberOfInternalFields == 3);
47
48 static std::array<JSValue, numberOfInternalFields> initialValues()
49 {
50 return { {
51 jsNull(),
52 jsNull(),
53 jsNumber(0),
54 } };
55 }
56
57 const WriteBarrier<Unknown>& internalField(Field field) const { return Base::internalField(static_cast<uint32_t>(field)); }
58 WriteBarrier<Unknown>& internalField(Field field) { return Base::internalField(static_cast<uint32_t>(field)); }
59
60 template<typename CellType, SubspaceAccess mode>
61 static IsoSubspace* subspaceFor(VM& vm)
62 {
63 return vm.mapIteratorSpace<mode>();
64 }
65
66 static Structure* createStructure(VM& vm, JSGlobalObject* globalObject, JSValue prototype)
67 {
68 return Structure::create(vm, globalObject, prototype, TypeInfo(JSMapIteratorType, StructureFlags), info());
69 }
70
71 static JSMapIterator* create(VM& vm, Structure* structure, JSMap* iteratedObject, IterationKind kind)
72 {
73 JSMapIterator* instance = new (NotNull, allocateCell<JSMapIterator>(vm.heap)) JSMapIterator(vm, structure);
74 instance->finishCreation(vm, iteratedObject, kind);
75 return instance;
76 }
77
78 static JSMapIterator* createWithInitialValues(VM&, Structure*);
79
80 ALWAYS_INLINE HashMapBucketType* advanceIter(VM& vm)
81 {
82 HashMapBucketType* prev = iterator();
83 HashMapBucketType* sentinel = jsCast<HashMapBucketType*>(vm.sentinelMapBucket());
84 if (prev == sentinel)
85 return nullptr;
86 HashMapBucketType* bucket = prev->next();
87 while (bucket && bucket->deleted())
88 bucket = bucket->next();
89 if (!bucket) {
90 setIterator(vm, sentinel);
91 return nullptr;
92 }
93 setIterator(vm, bucket); // We keep iterator on the last value since the first thing we do in this function is call next().
94 return bucket;
95 }
96
97 bool next(JSGlobalObject* globalObject, JSValue& value)
98 {
99 HashMapBucketType* bucket = advanceIter(getVM(globalObject));
100 if (!bucket)
101 return false;
102
103 switch (kind()) {
104 case IterationKind::Values:
105 value = bucket->value();
106 break;
107 case IterationKind::Keys:
108 value = bucket->key();
109 break;
110 case IterationKind::Entries:
111 value = createPair(globalObject, bucket->key(), bucket->value());
112 break;
113 }
114 return true;
115 }
116
117 bool nextKeyValue(JSGlobalObject* globalObject, JSValue& key, JSValue& value)
118 {
119 HashMapBucketType* bucket = advanceIter(getVM(globalObject));
120 if (!bucket)
121 return false;
122
123 key = bucket->key();
124 value = bucket->value();
125 return true;
126 }
127
128 IterationKind kind() const { return static_cast<IterationKind>(internalField(Field::Kind).get().asUInt32AsAnyInt()); }
129 JSObject* iteratedObject() const { return jsCast<JSObject*>(internalField(Field::IteratedObject).get()); }
130 HashMapBucketType* iterator() const { return jsCast<HashMapBucketType*>(internalField(Field::MapBucket).get()); }
131
132private:
133 JSMapIterator(VM& vm, Structure* structure)
134 : Base(vm, structure)
135 { }
136
137 void setIterator(VM& vm, HashMapBucketType* bucket)
138 {
139 internalField(Field::MapBucket).set(vm, this, bucket);
140 }
141
142 JS_EXPORT_PRIVATE void finishCreation(VM&, JSMap*, IterationKind);
143 void finishCreation(VM&);
144 JSValue createPair(JSGlobalObject*, JSValue, JSValue);
145 DECLARE_VISIT_CHILDREN;
146};
147STATIC_ASSERT_IS_TRIVIALLY_DESTRUCTIBLE(JSMapIterator);
148
149} // namespace JSC
