233 // This is similar to the putDirect* methods:
234 // - the prototype chain is not consulted
235 // - accessors are not called.
236 // - it will ignore extensibility and read-only properties if PutDirectIndexLikePutDirect is passed as the mode (the default).
237 // This method creates a property with attributes writable, enumerable and configurable all set to true if attributes is zero,
238 // otherwise, it creates a property with the provided attributes. Semantically, this is performing defineOwnProperty.
239 bool putDirectIndex(JSGlobalObject* globalObject, unsigned propertyName, JSValue value, unsigned attributes, PutDirectIndexMode mode)
240 {
241 ASSERT(!value.isCustomGetterSetter());
242 auto canSetIndexQuicklyForPutDirect = [&] () -> bool {
243 switch (indexingMode()) {
244 case ALL_BLANK_INDEXING_TYPES:
245 case ALL_UNDECIDED_INDEXING_TYPES:
246 return false;
247 case ALL_WRITABLE_INT32_INDEXING_TYPES:
248 case ALL_WRITABLE_DOUBLE_INDEXING_TYPES:
249 case ALL_WRITABLE_CONTIGUOUS_INDEXING_TYPES:
250 case ALL_ARRAY_STORAGE_INDEXING_TYPES:
251 return propertyName < m_butterfly->vectorLength();
252 default:
253 if (isCopyOnWrite(indexingMode()))
254 return false;
255 RELEASE_ASSERT_NOT_REACHED();
256 return false;
257 }
258 };
259
260 if (!attributes && canSetIndexQuicklyForPutDirect()) {
261 setIndexQuickly(getVM(globalObject), propertyName, value);
262 return true;
263 }
264 return putDirectIndexSlowOrBeyondVectorLength(globalObject, propertyName, value, attributes, mode);
265 }
266 // This is semantically equivalent to performing defineOwnProperty(propertyName, {configurable:true, writable:true, enumerable:true, value:value}).
267 bool putDirectIndex(JSGlobalObject* globalObject, unsigned propertyName, JSValue value)
268 {
269 return putDirectIndex(globalObject, propertyName, value, 0, PutDirectIndexLikePutDirect);
270 }

276 return putDirect(getVM(globalObject), Identifier::from(getVM(globalObject), propertyName), value, attributes);
277 }

295 bool canGetIndexQuickly(unsigned i) const
296 {
297 const Butterfly* butterfly = this->butterfly();
298 switch (indexingType()) {
299 case ALL_BLANK_INDEXING_TYPES:
300 return canGetIndexQuicklyForTypedArray(i);
301 case ALL_UNDECIDED_INDEXING_TYPES:
302 return false;
303 case ALL_INT32_INDEXING_TYPES:
304 case ALL_CONTIGUOUS_INDEXING_TYPES:
305 return i < butterfly->vectorLength() && butterfly->contiguous().at(this, i);
306 case ALL_DOUBLE_INDEXING_TYPES: {
307 if (i >= butterfly->vectorLength())
308 return false;
309 double value = butterfly->contiguousDouble().at(this, i);
310 if (value != value)
311 return false;
312 return true;
313 }
314 case ALL_ARRAY_STORAGE_INDEXING_TYPES:
315 return i < butterfly->arrayStorage()->vectorLength() && butterfly->arrayStorage()->m_vector[i];
316 default:
317 RELEASE_ASSERT_NOT_REACHED();
318 return false;
319 }
320 }

350 JSValue tryGetIndexQuickly(unsigned i) const
351 {
352 const Butterfly* butterfly = this->butterfly();
353 switch (indexingType()) {
354 case ALL_BLANK_INDEXING_TYPES:
355 if (canGetIndexQuicklyForTypedArray(i))
356 return getIndexQuicklyForTypedArray(i);
357 break;
358 case ALL_UNDECIDED_INDEXING_TYPES:
359 break;
360 case ALL_INT32_INDEXING_TYPES:
361 if (i < butterfly->publicLength()) {
362 JSValue result = butterfly->contiguous().at(this, i).get();
363 ASSERT(result.isInt32() || !result);
364 return result;
365 }
366 break;
367 case ALL_CONTIGUOUS_INDEXING_TYPES:
368 if (i < butterfly->publicLength())
369 return butterfly->contiguous().at(this, i).get();
370 break;
371 case ALL_DOUBLE_INDEXING_TYPES: {
372 if (i >= butterfly->publicLength())
373 break;
374 double result = butterfly->contiguousDouble().at(this, i);
375 if (result != result)
376 break;
377 return JSValue(JSValue::EncodeAsDouble, result);
378 }
379 case ALL_ARRAY_STORAGE_INDEXING_TYPES:
380 if (i < butterfly->arrayStorage()->vectorLength())
381 return butterfly->arrayStorage()->m_vector[i].get();
382 break;
383 default:
384 RELEASE_ASSERT_NOT_REACHED();
385 break;
386 }
387 return JSValue();
388 }

419 {
420 Butterfly* butterfly = this->butterfly();

433 case ArrayWithSlowPutArrayStorage:

435 && !!butterfly->arrayStorage()->m_vector[i];

438 return false;
439 RELEASE_ASSERT_NOT_REACHED();
440 return false;
441 }

444 void setIndexQuickly(VM& vm, unsigned i, JSValue v)
445 {
446 Butterfly* butterfly = m_butterfly.get();
447 ASSERT(!isCopyOnWrite(indexingMode()));
448 switch (indexingType()) {
449 case ALL_INT32_INDEXING_TYPES: {
450 ASSERT(i < butterfly->vectorLength());
451 if (!v.isInt32()) {
452 convertInt32ToDoubleOrContiguousWhilePerformingSetIndex(vm, i, v);
453 return;
454 }
455 FALLTHROUGH;
456 }
457 case ALL_CONTIGUOUS_INDEXING_TYPES: {
458 ASSERT(i < butterfly->vectorLength());
459 butterfly->contiguous().at(this, i).setWithoutWriteBarrier(v);
460 if (i >= butterfly->publicLength())
461 butterfly->setPublicLength(i + 1);
462 vm.heap.writeBarrier(this, v);
463 break;
464 }
465 case ALL_DOUBLE_INDEXING_TYPES: {
466 ASSERT(i < butterfly->vectorLength());
467 if (!v.isNumber()) {
468 convertDoubleToContiguousWhilePerformingSetIndex(vm, i, v);
469 return;
470 }
471 double value = v.asNumber();
472 if (value != value) {
473 convertDoubleToContiguousWhilePerformingSetIndex(vm, i, v);
474 return;
475 }
476 butterfly->contiguousDouble().at(this, i) = value;
477 if (i >= butterfly->publicLength())
478 butterfly->setPublicLength(i + 1);
479 break;
480 }
481 case ALL_ARRAY_STORAGE_INDEXING_TYPES: {
482 ArrayStorage* storage = butterfly->arrayStorage();
483 WriteBarrier<Unknown>& x = storage->m_vector[i];
484 JSValue old = x.get();
485 x.set(vm, this, v);
486 if (!old) {
487 ++storage->m_numValuesInVector;
488 if (i >= storage->length())
489 storage->setLength(i + 1);
490 }
491 break;
492 }
493 case ALL_BLANK_INDEXING_TYPES:
494 setIndexQuicklyForTypedArray(i, v);
495 break;
496 default:
497 RELEASE_ASSERT_NOT_REACHED();
498 }
499 }

506 // NOTE: Clients of this method may call it more than once for any index, and this is supposed
507 // to work.
508 ALWAYS_INLINE void initializeIndex(ObjectInitializationScope& scope, unsigned i, JSValue v, IndexingType indexingType)
509 {
510 VM& vm = scope.vm();
511 Butterfly* butterfly = m_butterfly.get();
512 switch (indexingType) {
513 case ALL_UNDECIDED_INDEXING_TYPES: {
514 setIndexQuicklyToUndecided(vm, i, v);
515 break;
516 }
517 case ALL_INT32_INDEXING_TYPES: {
518 ASSERT(i < butterfly->publicLength());
519 ASSERT(i < butterfly->vectorLength());
520 if (!v.isInt32()) {
521 convertInt32ToDoubleOrContiguousWhilePerformingSetIndex(vm, i, v);
522 break;
523 }
524 FALLTHROUGH;
525 }
526 case ALL_CONTIGUOUS_INDEXING_TYPES: {
527 ASSERT(i < butterfly->publicLength());
528 ASSERT(i < butterfly->vectorLength());
529 butterfly->contiguous().at(this, i).set(vm, this, v);
530 break;
531 }
532 case ALL_DOUBLE_INDEXING_TYPES: {
533 ASSERT(i < butterfly->publicLength());
534 ASSERT(i < butterfly->vectorLength());
535 if (!v.isNumber()) {
536 convertDoubleToContiguousWhilePerformingSetIndex(vm, i, v);
537 return;
538 }
539 double value = v.asNumber();
540 if (value != value) {
541 convertDoubleToContiguousWhilePerformingSetIndex(vm, i, v);
542 return;
543 }
544 butterfly->contiguousDouble().at(this, i) = value;
545 break;
546 }
547 case ALL_ARRAY_STORAGE_INDEXING_TYPES: {
548 ArrayStorage* storage = butterfly->arrayStorage();
549 ASSERT(i < storage->length());
550 ASSERT(i < storage->m_numValuesInVector);
551 storage->m_vector[i].set(vm, this, v);
552 break;
553 }
554 default:
555 RELEASE_ASSERT_NOT_REACHED();
556 }

596 ArrayStorage* storage = butterfly->arrayStorage();
597 ASSERT(i < storage->length());
598 ASSERT(i < storage->m_numValuesInVector);

601 }

643 // putDirect is effectively an unchecked vesion of 'defineOwnProperty':
644 // - the prototype chain is not consulted
645 // - accessors are not called.

1461// It may seem crazy to inline a function this large but it makes a big difference
1462// since this is function very hot in variable lookup
1463template<bool checkNullStructure>
1464ALWAYS_INLINE bool JSObject::getPropertySlot(JSGlobalObject* globalObject, PropertyName propertyName, PropertySlot& slot)
1465{
1466 VM& vm = getVM(globalObject);
1467 auto& structureIDTable = vm.heap.structureIDTable();
1468 JSObject* object = this;
1469 while (true) {
1470 if (UNLIKELY(TypeInfo::overridesGetOwnPropertySlot(object->inlineTypeFlags()))) {
1471 // If propertyName is an index then we may have missed it (as this loop is using
1472 // getOwnNonIndexPropertySlot), so we cannot safely call the overridden getOwnPropertySlot
1473 // (lest we return a property from a prototype that is shadowed). Check now for an index,
1474 // if so we need to start afresh from this object.
1475 if (std::optional<uint32_t> index = parseIndex(propertyName))
1476 return getPropertySlot(globalObject, index.value(), slot);
1477 // Safe to continue searching from current position; call getNonIndexPropertySlot to avoid
1478 // parsing the int again.
1479 return object->getNonIndexPropertySlot(globalObject, propertyName, slot);
1480 }
1481 ASSERT(object->type() != ProxyObjectType);
1482 Structure* structure = structureIDTable.get(object->structureID());
1483#if USE(JSVALUE64)
1484 if (checkNullStructure && UNLIKELY(!structure))
1485 CRASH_WITH_INFO(object->type(), object->structureID(), structureIDTable.size());
1486#endif
1487 if (object->getOwnNonIndexPropertySlot(vm, structure, propertyName, slot))
1488 return true;
1489 // FIXME: This doesn't look like it's following the specification:
1490 // https://bugs.webkit.org/show_bug.cgi?id=172572
1491 JSValue prototype = structure->storedPrototype(object);
1492 if (!prototype.isObject())
1493 break;
1494 object = asObject(prototype);
1495 }
1496
1497 if (std::optional<uint32_t> index = parseIndex(propertyName))
1498 return getPropertySlot(globalObject, index.value(), slot);
1499 return false;
1500}
