47// https://tc39.es/ecma262/#sec-createlistfromarraylike
48template <typename Functor> // A functor should have a type like: (JSValue) -> bool
49void forEachInArrayLike(JSGlobalObject* globalObject, JSObject* arrayLikeObject, Functor functor)
50{
51 VM& vm = getVM(globalObject);
52 auto scope = DECLARE_THROW_SCOPE(vm);
53 uint64_t length = static_cast<uint64_t>(toLength(globalObject, arrayLikeObject));
54 RETURN_IF_EXCEPTION(scope, void());
55 for (uint64_t index = 0; index < length; index++) {
56 JSValue value = arrayLikeObject->getIndex(globalObject, index);
57 RETURN_IF_EXCEPTION(scope, void());
58 if (!functor(value))
59 return;
60 }
61}

63ALWAYS_INLINE bool JSObject::canPerformFastPutInlineExcludingProto(VM& vm)
64{
65 // Check if there are any setters or getters in the prototype chain
66 JSValue prototype;
67 JSObject* obj = this;
68 while (true) {
69 Structure* structure = obj->structure(vm);
70 if (structure->hasReadOnlyOrGetterSetterPropertiesExcludingProto() || structure->typeInfo().overridesGetPrototype())
71 return false;
72 if (obj != this && structure->typeInfo().overridesPut())
73 return false;
74
75 prototype = obj->getPrototypeDirect(vm);
76 if (prototype.isNull())
77 return true;
78
79 obj = asObject(prototype);
80 }
81
82 ASSERT_NOT_REACHED();
83}

85ALWAYS_INLINE bool JSObject::canPerformFastPutInline(VM& vm, PropertyName propertyName)

100ALWAYS_INLINE typename std::result_of<CallbackWhenNoException(bool, PropertySlot&)>::type JSObject::getPropertySlot(JSGlobalObject* globalObject, PropertyName propertyName, PropertySlot& slot, CallbackWhenNoException callback) const
101{
102 VM& vm = getVM(globalObject);
103 auto scope = DECLARE_THROW_SCOPE(vm);
104 bool found = const_cast<JSObject*>(this)->getPropertySlot(globalObject, propertyName, slot);
105 RETURN_IF_EXCEPTION(scope, { });
106 RELEASE_AND_RETURN(scope, callback(found, slot));
107}

109ALWAYS_INLINE bool JSObject::getPropertySlot(JSGlobalObject* globalObject, unsigned propertyName, PropertySlot& slot)
110{
111 VM& vm = getVM(globalObject);
112 auto scope = DECLARE_THROW_SCOPE(vm);
113 auto& structureIDTable = vm.heap.structureIDTable();
114 JSObject* object = this;
115 while (true) {
116 Structure* structure = structureIDTable.get(object->structureID());
117 bool hasSlot = structure->classInfo()->methodTable.getOwnPropertySlotByIndex(object, globalObject, propertyName, slot);
118 RETURN_IF_EXCEPTION(scope, false);
119 if (hasSlot)
120 return true;
121 if (UNLIKELY(slot.isVMInquiry() && slot.isTaintedByOpaqueObject()))
122 return false;
123 if (object->type() == ProxyObjectType && slot.internalMethodType() == PropertySlot::InternalMethodType::HasProperty)
124 return false;
125 if (isTypedArrayType(object->type()) && propertyName >= jsCast<JSArrayBufferView*>(object)->length())
126 return false;
127 JSValue prototype;
128 if (LIKELY(!structure->typeInfo().overridesGetPrototype() || slot.internalMethodType() == PropertySlot::InternalMethodType::VMInquiry))
129 prototype = object->getPrototypeDirect(vm);
130 else {
131 prototype = object->getPrototype(vm, globalObject);
132 RETURN_IF_EXCEPTION(scope, false);
133 }
134 if (!prototype.isObject())
135 return false;
136 object = asObject(prototype);
137 }
138}

147ALWAYS_INLINE bool JSObject::getNonIndexPropertySlot(JSGlobalObject* globalObject, PropertyName propertyName, PropertySlot& slot)
148{
149 // This method only supports non-index PropertyNames.
150 ASSERT(!parseIndex(propertyName));
151
152 VM& vm = getVM(globalObject);
153 auto scope = DECLARE_THROW_SCOPE(vm);
154 auto& structureIDTable = vm.heap.structureIDTable();
155 JSObject* object = this;
156 while (true) {
157 Structure* structure = structureIDTable.get(object->structureID());
158 if (LIKELY(!TypeInfo::overridesGetOwnPropertySlot(object->inlineTypeFlags()))) {
159 if (object->getOwnNonIndexPropertySlot(vm, structure, propertyName, slot))
160 return true;
161 } else {
162 bool hasSlot = structure->classInfo()->methodTable.getOwnPropertySlot(object, globalObject, propertyName, slot);
163 RETURN_IF_EXCEPTION(scope, false);
164 if (hasSlot)
165 return true;
166 if (UNLIKELY(slot.isVMInquiry() && slot.isTaintedByOpaqueObject()))
167 return false;
168 if (object->type() == ProxyObjectType && slot.internalMethodType() == PropertySlot::InternalMethodType::HasProperty)
169 return false;
170 if (isTypedArrayType(object->type()) && isCanonicalNumericIndexString(propertyName))
171 return false;
172 }
173 JSValue prototype;
174 if (LIKELY(!structure->typeInfo().overridesGetPrototype() || slot.internalMethodType() == PropertySlot::InternalMethodType::VMInquiry))
175 prototype = object->getPrototypeDirect(vm);
176 else {
177 prototype = object->getPrototype(vm, globalObject);
178 RETURN_IF_EXCEPTION(scope, false);
179 }
180 if (!prototype.isObject())
181 return false;
182 object = asObject(prototype);
183 }
184}

194inline JSValue JSObject::getIfPropertyExists(JSGlobalObject* globalObject, PropertyName propertyName)
195{
196 VM& vm = globalObject->vm();
197 auto scope = DECLARE_THROW_SCOPE(vm);
198
199 PropertySlot slot(this, PropertySlot::InternalMethodType::HasProperty);
200 bool hasProperty = getPropertySlot(globalObject, propertyName, slot);
201 RETURN_IF_EXCEPTION(scope, { });
202 if (!hasProperty)
203 return { };
204
205 scope.release();
206 if (UNLIKELY(slot.isTaintedByOpaqueObject()))
207 return get(globalObject, propertyName);
208
209 return slot.getValue(globalObject, propertyName);
210}

217inline void JSObject::putDirectWithoutTransition(VM& vm, PropertyName propertyName, JSValue value, unsigned attributes)
218{
219 ASSERT(!value.isGetterSetter() && !(attributes & PropertyAttribute::Accessor));
220 ASSERT(!value.isCustomGetterSetter());
221 StructureID structureID = this->structureID();
222 Structure* structure = vm.heap.structureIDTable().get(structureID);
223 PropertyOffset offset = prepareToPutDirectWithoutTransition(vm, propertyName, attributes, structureID, structure);
224 putDirect(vm, offset, value);
225 if (attributes & PropertyAttribute::ReadOnly)
226 structure->setContainsReadOnlyProperties();
227}
228
229ALWAYS_INLINE PropertyOffset JSObject::prepareToPutDirectWithoutTransition(VM& vm, PropertyName propertyName, unsigned attributes, StructureID structureID, Structure* structure)
230{
231 unsigned oldOutOfLineCapacity = structure->outOfLineCapacity();
232 PropertyOffset result;
233 structure->addPropertyWithoutTransition(
234 vm, propertyName, attributes,
235 [&] (const GCSafeConcurrentJSLocker&, PropertyOffset offset, PropertyOffset newMaxOffset) {
236 unsigned newOutOfLineCapacity = Structure::outOfLineCapacity(newMaxOffset);
237 if (newOutOfLineCapacity != oldOutOfLineCapacity) {
238 Butterfly* butterfly = allocateMoreOutOfLineStorage(vm, oldOutOfLineCapacity, newOutOfLineCapacity);
239 nukeStructureAndSetButterfly(vm, structureID, butterfly);
240 structure->setMaxOffset(vm, newMaxOffset);
241 WTF::storeStoreFence();
242 setStructureIDDirectly(structureID);
243 } else
244 structure->setMaxOffset(vm, newMaxOffset);
245
246 // This assertion verifies that the concurrent GC won't read garbage if the concurrentGC
247 // is running at the same time we put without transitioning.
248 ASSERT(!getDirect(offset) || !JSValue::encode(getDirect(offset)));
249 result = offset;
250 });
251 return result;
252}

254// https://tc39.es/ecma262/#sec-ordinaryset
255ALWAYS_INLINE bool JSObject::putInlineForJSObject(JSCell* cell, JSGlobalObject* globalObject, PropertyName propertyName, JSValue value, PutPropertySlot& slot)
256{
257 VM& vm = getVM(globalObject);
258
259 JSObject* thisObject = jsCast<JSObject*>(cell);
260 ASSERT(value);
261 ASSERT(!Heap::heap(value) || Heap::heap(value) == Heap::heap(thisObject));
262
263 // Try indexed put first. This is required for correctness, since loads on property names that appear like
264 // valid indices will never look in the named property storage.
265 if (std::optional<uint32_t> index = parseIndex(propertyName)) {
266 if (UNLIKELY(isThisValueAltered(slot, thisObject)))
267 return ordinarySetSlow(globalObject, thisObject, propertyName, value, slot.thisValue(), slot.isStrictMode());
268 return thisObject->methodTable(vm)->putByIndex(thisObject, globalObject, index.value(), value, slot.isStrictMode());
269 }
270
271 if (!thisObject->canPerformFastPutInline(vm, propertyName))
272 return thisObject->putInlineSlow(globalObject, propertyName, value, slot);
273 if (UNLIKELY(isThisValueAltered(slot, thisObject)))
274 return definePropertyOnReceiver(globalObject, propertyName, value, slot);
275 if (UNLIKELY(thisObject->hasNonReifiedStaticProperties(vm)))
276 return thisObject->putInlineFastReplacingStaticPropertyIfNeeded(globalObject, propertyName, value, slot);
277 return thisObject->putInlineFast(globalObject, propertyName, value, slot);
278}

280ALWAYS_INLINE bool JSObject::putInlineFast(JSGlobalObject* globalObject, PropertyName propertyName, JSValue value, PutPropertySlot& slot)
281{
282 VM& vm = getVM(globalObject);
283 auto scope = DECLARE_THROW_SCOPE(vm);
284
285 // FIXME: For a failure due to non-extensible structure, the error message is misleading.
286 if (!putDirectInternal<PutModePut>(vm, propertyName, value, 0, slot))
287 return typeError(globalObject, scope, slot.isStrictMode(), ReadonlyPropertyWriteError);
288 return true;
289}

314template<JSObject::PutMode mode>
315ALWAYS_INLINE bool JSObject::putDirectInternal(VM& vm, PropertyName propertyName, JSValue value, unsigned attributes, PutPropertySlot& slot)
316{
317 ASSERT(value);
318 ASSERT(value.isGetterSetter() == !!(attributes & PropertyAttribute::Accessor));
319 ASSERT(value.isCustomGetterSetter() == !!(attributes & PropertyAttribute::CustomAccessorOrValue));
320 ASSERT(!Heap::heap(value) || Heap::heap(value) == Heap::heap(this));
321 ASSERT(!parseIndex(propertyName));
322
323 StructureID structureID = this->structureID();
324 Structure* structure = vm.heap.structureIDTable().get(structureID);
325 if (structure->isDictionary()) {
326 ASSERT(!isCopyOnWrite(indexingMode()));
327
328 unsigned currentAttributes;
329 PropertyOffset offset = structure->get(vm, propertyName, currentAttributes);
330 if (offset != invalidOffset) {
331 if ((mode == PutModePut) && currentAttributes & PropertyAttribute::ReadOnlyOrAccessorOrCustomAccessor)
332 return false;
333
334 putDirect(vm, offset, value);
335 structure->didReplaceProperty(offset);
336
337 // FIXME: Check attributes against PropertyAttribute::CustomAccessorOrValue. Changing GetterSetter should work w/o transition.
338 // https://bugs.webkit.org/show_bug.cgi?id=214342
339 if (mode == PutModeDefineOwnProperty && (attributes != currentAttributes || (attributes & PropertyAttribute::AccessorOrCustomAccessorOrValue)))
340 setStructure(vm, Structure::attributeChangeTransition(vm, structure, propertyName, attributes));
341 else
342 slot.setExistingProperty(this, offset);
343
344 return true;
345 }
346
347 if ((mode == PutModePut) && !isStructureExtensible(vm))
348 return false;
349
350 offset = prepareToPutDirectWithoutTransition(vm, propertyName, attributes, structureID, structure);
351 validateOffset(offset);
352 putDirect(vm, offset, value);
353 slot.setNewProperty(this, offset);
354 if (attributes & PropertyAttribute::ReadOnly)
355 this->structure(vm)->setContainsReadOnlyProperties();
356 return true;
357 }
358
359 PropertyOffset offset;
360 size_t currentCapacity = this->structure(vm)->outOfLineCapacity();
361 Structure* newStructure = Structure::addPropertyTransitionToExistingStructure(
362 structure, propertyName, attributes, offset);
363 if (newStructure) {
364 Butterfly* newButterfly = butterfly();
365 if (currentCapacity != newStructure->outOfLineCapacity()) {
366 ASSERT(newStructure != this->structure(vm));
367 newButterfly = allocateMoreOutOfLineStorage(vm, currentCapacity, newStructure->outOfLineCapacity());
368 nukeStructureAndSetButterfly(vm, structureID, newButterfly);
369 }
370
371 validateOffset(offset);
372 ASSERT(newStructure->isValidOffset(offset));
373
374 // This assertion verifies that the concurrent GC won't read garbage if the concurrentGC
375 // is running at the same time we put without transitioning.
376 ASSERT(!getDirect(offset) || !JSValue::encode(getDirect(offset)));
377 putDirect(vm, offset, value);
378 setStructure(vm, newStructure);
379 slot.setNewProperty(this, offset);
380 return true;
381 }
382
383 // We want the structure transition watchpoint to fire after this object has switched structure.
384 // This allows adaptive watchpoints to observe if the new structure is the one we want.
385 DeferredStructureTransitionWatchpointFire deferredWatchpointFire(vm, structure);
386
387 unsigned currentAttributes;
388 offset = structure->get(vm, propertyName, currentAttributes);
389 if (offset != invalidOffset) {
390 if ((mode == PutModePut) && currentAttributes & PropertyAttribute::ReadOnlyOrAccessorOrCustomAccessor)
391 return false;
392
393 structure->didReplaceProperty(offset);
394 putDirect(vm, offset, value);
395
396 // FIXME: Check attributes against PropertyAttribute::CustomAccessorOrValue. Changing GetterSetter should work w/o transition.
397 // https://bugs.webkit.org/show_bug.cgi?id=214342
398 if (mode == PutModeDefineOwnProperty && (attributes != currentAttributes || (attributes & PropertyAttribute::AccessorOrCustomAccessorOrValue)))
399 setStructure(vm, Structure::attributeChangeTransition(vm, structure, propertyName, attributes, &deferredWatchpointFire));
400 else
401 slot.setExistingProperty(this, offset);
402
403 return true;
404 }
405
406 if ((mode == PutModePut) && !isStructureExtensible(vm))
407 return false;
408
409 newStructure = Structure::addNewPropertyTransition(
410 vm, structure, propertyName, attributes, offset, slot.context(), &deferredWatchpointFire);
411
412 validateOffset(offset);
413 ASSERT(newStructure->isValidOffset(offset));
414 size_t oldCapacity = structure->outOfLineCapacity();
415 size_t newCapacity = newStructure->outOfLineCapacity();
416 ASSERT(oldCapacity <= newCapacity);
417 if (oldCapacity != newCapacity) {
418 Butterfly* newButterfly = allocateMoreOutOfLineStorage(vm, oldCapacity, newCapacity);
419 nukeStructureAndSetButterfly(vm, structureID, newButterfly);
420 }
421
422 // This assertion verifies that the concurrent GC won't read garbage if the concurrentGC
423 // is running at the same time we put without transitioning.
424 ASSERT(!getDirect(offset) || !JSValue::encode(getDirect(offset)));
425 putDirect(vm, offset, value);
426 setStructure(vm, newStructure);
427 slot.setNewProperty(this, offset);
428 if (attributes & PropertyAttribute::ReadOnly)
429 newStructure->setContainsReadOnlyProperties();
430 return true;
431}

505{
506#if ASSERT_ENABLED
507 ASSERT(value);
508 ASSERT(!Heap::heap(value) || Heap::heap(value) == Heap::heap(this));

510 PropertyOffset offset = structure(vm)->get(vm, propertyName, attributes);
511 if (isValidOffset(offset))
512 ASSERT(!(attributes & (PropertyAttribute::Accessor | PropertyAttribute::CustomAccessor | PropertyAttribute::ReadOnly)));
513 else if (TypeInfo::hasStaticPropertyTable(inlineTypeFlags())) {
514 if (auto entry = findPropertyHashEntry(vm, propertyName))
515 ASSERT(!(entry->value->attributes() & (PropertyAttribute::Accessor | PropertyAttribute::CustomAccessor | PropertyAttribute::ReadOnly)));
516 }
517#else // not ASSERT_ENABLED
518 UNUSED_PARAM(vm);
519 UNUSED_PARAM(propertyName);
520 UNUSED_PARAM(value);
521#endif // not ASSERT_ENABLED
522}

600ALWAYS_INLINE bool JSObject::getPrivateFieldSlot(JSObject* object, JSGlobalObject* globalObject, PropertyName propertyName, PropertySlot& slot)
601{
602 ASSERT(propertyName.isPrivateName());
603 VM& vm = getVM(globalObject);
604 Structure* structure = object->structure(vm);
605
606 unsigned attributes;
607 PropertyOffset offset = structure->get(vm, propertyName, attributes);
608 if (offset == invalidOffset)
609 return false;
610
611 JSValue value = object->getDirect(offset);
612#if ASSERT_ENABLED
613 ASSERT(value);
614 if (value.isCell()) {
615 JSCell* cell = value.asCell();
616 JSType type = cell->type();
617 UNUSED_PARAM(cell);
618 ASSERT_UNUSED(type, type != GetterSetterType && type != CustomGetterSetterType);
619 // FIXME: For now, private fields do not support getter/setter fields. Later on, we will need to fill in accessor metadata here,
620 // as in JSObject::getOwnNonIndexPropertySlot()
621 // https://bugs.webkit.org/show_bug.cgi?id=194435
622 }
623#endif
624
625 slot.setValue(object, attributes, value, offset);
626 return true;
627}

680ALWAYS_INLINE void JSObject::getNonReifiedStaticPropertyNames(VM& vm, PropertyNameArray& propertyNames, DontEnumPropertiesMode mode)
681{
682 if (staticPropertiesReified(vm))
683 return;
684
685 JSGlobalObject* globalObject = this->globalObject(vm);
686
687 // Add properties from the static hashtables of properties
688 for (const ClassInfo* info = classInfo(vm); info; info = info->parentClass) {
689 const HashTable* table = info->staticPropHashTable;
690 if (!table)
691 continue;
692
693 for (auto iter = table->begin(); iter != table->end(); ++iter) {
694 if (mode == DontEnumPropertiesMode::Exclude && (iter->attributes() & PropertyAttribute::DontEnum))
695 continue;
696 if ((iter->attributes() & PropertyAttribute::PropertyCallback) && !iter->isLazyPropertyEnabled(globalObject))
697 continue;
698 propertyNames.add(Identifier::fromString(vm, iter.key()));
699 }
700 }
701}
