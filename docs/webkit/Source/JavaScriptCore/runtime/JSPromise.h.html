26#pragma once
27
28#include "JSInternalFieldObjectImpl.h"
29
30namespace JSC {
31
32class JSPromiseConstructor;
33class JSPromise : public JSInternalFieldObjectImpl<2> {
34public:
35 using Base = JSInternalFieldObjectImpl<2>;
36
37 template<typename CellType, SubspaceAccess mode>
38 static IsoSubspace* subspaceFor(VM& vm)
39 {
40 return &vm.promiseSpace;
41 }
42
43 JS_EXPORT_PRIVATE static JSPromise* create(VM&, Structure*);
44 static JSPromise* createWithInitialValues(VM&, Structure*);
45 static Structure* createStructure(VM&, JSGlobalObject*, JSValue);
46
47 DECLARE_EXPORT_INFO;
48
49 enum class Status : unsigned {
50 Pending = 0, // Making this as 0, so that, we can change the status from Pending to others without masking.
51 Fulfilled = 1,
52 Rejected = 2,
53 };
54 static constexpr uint32_t isHandledFlag = 4;
55 static constexpr uint32_t isFirstResolvingFunctionCalledFlag = 8;
56 static constexpr uint32_t stateMask = 0b11;
57
58 enum class Field : unsigned {
59 Flags = 0,
60 ReactionsOrResult = 1,
61 };
62 static_assert(numberOfInternalFields == 2);
63
64 static std::array<JSValue, numberOfInternalFields> initialValues()
65 {
66 return { {
67 jsNumber(static_cast<unsigned>(Status::Pending)),
68 jsUndefined(),
69 } };
70 }
71
72 const WriteBarrier<Unknown>& internalField(Field field) const { return Base::internalField(static_cast<uint32_t>(field)); }
73 WriteBarrier<Unknown>& internalField(Field field) { return Base::internalField(static_cast<uint32_t>(field)); }
74
75 JS_EXPORT_PRIVATE Status status(VM&) const;
76 JS_EXPORT_PRIVATE JSValue result(VM&) const;
77 JS_EXPORT_PRIVATE bool isHandled(VM&) const;
78
79 JS_EXPORT_PRIVATE static JSPromise* resolvedPromise(JSGlobalObject*, JSValue);
80 JS_EXPORT_PRIVATE static JSPromise* rejectedPromise(JSGlobalObject*, JSValue);
81
82 JS_EXPORT_PRIVATE void resolve(JSGlobalObject*, JSValue);
83 JS_EXPORT_PRIVATE void reject(JSGlobalObject*, JSValue);
84 JS_EXPORT_PRIVATE void rejectAsHandled(JSGlobalObject*, JSValue);
85 JS_EXPORT_PRIVATE void reject(JSGlobalObject*, Exception*);
86 JS_EXPORT_PRIVATE void rejectAsHandled(JSGlobalObject*, Exception*);
87
88 JS_EXPORT_PRIVATE JSPromise* rejectWithCaughtException(JSGlobalObject*, ThrowScope&);
89
90 struct DeferredData {
91 WTF_FORBID_HEAP_ALLOCATION;
92 public:
93 JSPromise* promise { nullptr };
94 JSFunction* resolve { nullptr };
95 JSFunction* reject { nullptr };
96 };
97 static DeferredData createDeferredData(JSGlobalObject*, JSPromiseConstructor*);
98
99 DECLARE_VISIT_CHILDREN;
100
101protected:
102 JSPromise(VM&, Structure*);
103 void finishCreation(VM&);
104
105 uint32_t flags() const;
106};
107
108} // namespace JSC
