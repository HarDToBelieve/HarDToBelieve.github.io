104void getEnumerablePropertyNames(JSGlobalObject*, JSObject*, PropertyNameArray&, uint32_t& indexedLength, uint32_t& structurePropertyCount);
105
106inline JSPropertyNameEnumerator* propertyNameEnumerator(JSGlobalObject* globalObject, JSObject* base)
107{
108 VM& vm = getVM(globalObject);
109 auto scope = DECLARE_THROW_SCOPE(vm);
110
111 uint32_t indexedLength = base->getEnumerableLength(globalObject);
112
113 JSPropertyNameEnumerator* enumerator = nullptr;
114
115 Structure* structure = base->structure(vm);
116 if (!indexedLength
117 && (enumerator = structure->cachedPropertyNameEnumerator())
118 && enumerator->cachedPrototypeChain() == structure->prototypeChain(globalObject, base))
119 return enumerator;
120
121 uint32_t numberStructureProperties = 0;
122 PropertyNameArray propertyNames(vm, PropertyNameMode::Strings, PrivateSymbolMode::Exclude);
123 getEnumerablePropertyNames(globalObject, base, propertyNames, indexedLength, numberStructureProperties);
124 RETURN_IF_EXCEPTION(scope, nullptr);
125
126 ASSERT(propertyNames.size() < UINT32_MAX);
127
128 bool sawPolyProto;
129 bool successfullyNormalizedChain = normalizePrototypeChain(globalObject, base, sawPolyProto) != InvalidPrototypeChain;
130
131 Structure* structureAfterGettingPropertyNames = base->structure(vm);
132 if (!structureAfterGettingPropertyNames->canAccessPropertiesQuicklyForEnumeration()) {
133 indexedLength = 0;
134 numberStructureProperties = 0;
135 }
136
137 enumerator = JSPropertyNameEnumerator::create(vm, structureAfterGettingPropertyNames, indexedLength, numberStructureProperties, WTFMove(propertyNames));
138 if (!indexedLength && successfullyNormalizedChain && structureAfterGettingPropertyNames == structure) {
139 enumerator->setCachedPrototypeChain(vm, structure->prototypeChain(globalObject, base));
140 if (structure->canCachePropertyNameEnumerator(vm))
141 structure->setCachedPropertyNameEnumerator(vm, enumerator);
142 }
143 return enumerator;
144}
