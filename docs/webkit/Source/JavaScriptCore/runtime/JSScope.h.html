103class ScopeChainIterator {
104public:
105 ScopeChainIterator(JSScope* node)
106 : m_node(node)
107 {
108 }
109
110 JSObject* get() const { return JSScope::objectAtScope(m_node); }
111 JSObject* operator->() const { return JSScope::objectAtScope(m_node); }
112 JSScope* scope() const { return m_node; }
113
114 ScopeChainIterator& operator++() { m_node = m_node->next(); return *this; }
115
116 // postfix ++ intentionally omitted
117
118 bool operator==(const ScopeChainIterator& other) const { return m_node == other.m_node; }
119 bool operator!=(const ScopeChainIterator& other) const { return m_node != other.m_node; }
120
121private:
122 JSScope* m_node;
123};
124
125inline ScopeChainIterator JSScope::begin()
126{
127 return ScopeChainIterator(this);
128}
129
130inline ScopeChainIterator JSScope::end()
131{
132 return ScopeChainIterator(nullptr);
133}
134
135inline JSScope* JSScope::next()
136{
137 return m_next.get();
138}
139
140inline size_t JSScope::offsetOfNext()
141{
142 return OBJECT_OFFSETOF(JSScope, m_next);
143}
