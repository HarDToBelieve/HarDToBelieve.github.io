32namespace JSC {
33
34class JSSetIterator final : public JSInternalFieldObjectImpl<3> {
35public:
36 using HashMapBucketType = HashMapBucket<HashMapBucketDataKey>;
37 using Base = JSInternalFieldObjectImpl<3>;
38
39 DECLARE_EXPORT_INFO;
40
41 enum class Field : uint8_t {
42 SetBucket = 0,
43 IteratedObject,
44 Kind,
45 };
46 static_assert(numberOfInternalFields == 3);
47
48 static std::array<JSValue, numberOfInternalFields> initialValues()
49 {
50 return { {
51 jsNull(),
52 jsNull(),
53 jsNumber(0),
54 } };
55 }
56
57 const WriteBarrier<Unknown>& internalField(Field field) const { return Base::internalField(static_cast<uint32_t>(field)); }
58 WriteBarrier<Unknown>& internalField(Field field) { return Base::internalField(static_cast<uint32_t>(field)); }
59
60 template<typename CellType, SubspaceAccess mode>
61 static IsoSubspace* subspaceFor(VM& vm)
62 {
63 return vm.setIteratorSpace<mode>();
64 }
65
66 static Structure* createStructure(VM& vm, JSGlobalObject* globalObject, JSValue prototype)
67 {
68 return Structure::create(vm, globalObject, prototype, TypeInfo(JSSetIteratorType, StructureFlags), info());
69 }
70
71 static JSSetIterator* create(VM& vm, Structure* structure, JSSet* iteratedObject, IterationKind kind)
72 {
73 JSSetIterator* instance = new (NotNull, allocateCell<JSSetIterator>(vm.heap)) JSSetIterator(vm, structure);
74 instance->finishCreation(vm, iteratedObject, kind);
75 return instance;
76 }
77
78 static JSSetIterator* createWithInitialValues(VM&, Structure*);
79
80 ALWAYS_INLINE HashMapBucketType* advanceIter(VM& vm)
81 {
82 HashMapBucketType* prev = iterator();
83 HashMapBucketType* sentinel = jsCast<HashMapBucketType*>(vm.sentinelSetBucket());
84 if (prev == sentinel)
85 return nullptr;
86 HashMapBucketType* bucket = prev->next();
87 while (bucket && bucket->deleted())
88 bucket = bucket->next();
89 if (!bucket) {
90 setIterator(vm, sentinel);
91 return nullptr;
92 }
93 setIterator(vm, bucket); // We keep iterator on the last value since the first thing we do in this function is call next().
94 return bucket;
95 }
96
97 bool next(JSGlobalObject* globalObject, JSValue& value)
98 {
99 HashMapBucketType* bucket = advanceIter(getVM(globalObject));
100 if (!bucket)
101 return false;
102
103 switch (kind()) {
104 case IterationKind::Values:
105 case IterationKind::Keys:
106 value = bucket->key();
107 break;
108 case IterationKind::Entries:
109 value = createPair(globalObject, bucket->key(), bucket->key());
110 break;
111 }
112 return true;
113 }
114
115 IterationKind kind() const { return static_cast<IterationKind>(internalField(Field::Kind).get().asUInt32AsAnyInt()); }
116 JSObject* iteratedObject() const { return jsCast<JSObject*>(internalField(Field::IteratedObject).get()); }
117 HashMapBucketType* iterator() const { return jsCast<HashMapBucketType*>(internalField(Field::SetBucket).get()); }
118
119private:
120 JSSetIterator(VM& vm, Structure* structure)
121 : Base(vm, structure)
122 {
123 }
124
125 void setIterator(VM& vm, HashMapBucketType* bucket)
126 {
127 internalField(Field::SetBucket).set(vm, this, bucket);
128 }
129
130 JS_EXPORT_PRIVATE void finishCreation(VM&, JSSet*, IterationKind);
131 void finishCreation(VM&);
132 JS_EXPORT_PRIVATE JSValue createPair(JSGlobalObject*, JSValue, JSValue);
133 DECLARE_VISIT_CHILDREN;
134};
135STATIC_ASSERT_IS_TRIVIALLY_DESTRUCTIBLE(JSSetIterator);
136
137} // namespace JSC
