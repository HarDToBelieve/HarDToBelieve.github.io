55// Hash table generated by the create_hash_table script.
56struct HashTableValue {
57 const char* m_key; // property name
58 unsigned m_attributes; // JSObject attributes
59 Intrinsic m_intrinsic;
60 union ValueStorage {
61 constexpr ValueStorage(intptr_t value1, intptr_t value2)
62 : value1(value1)
63 , value2(value2)
64 { }
65 constexpr ValueStorage(long long constant)
66 : constant(constant)
67 { }
68
69 struct {
70 intptr_t value1;
71 intptr_t value2;
72 };
73 long long constant;
74 } m_values;
75
76 unsigned attributes() const { return m_attributes; }
77
78 Intrinsic intrinsic() const { ASSERT(m_attributes & PropertyAttribute::Function); return m_intrinsic; }
79 BuiltinGenerator builtinGenerator() const { ASSERT(m_attributes & PropertyAttribute::Builtin); return reinterpret_cast<BuiltinGenerator>(m_values.value1); }
80 NativeFunction function() const { ASSERT(m_attributes & PropertyAttribute::Function); return NativeFunction(m_values.value1); }
81 unsigned char functionLength() const
82 {
83 ASSERT(m_attributes & PropertyAttribute::Function);
84 if (m_attributes & PropertyAttribute::DOMJITFunction)
85 return signature()->argumentCount;
86 return static_cast<unsigned char>(m_values.value2);
87 }
88
89 GetFunction propertyGetter() const { ASSERT(!(m_attributes & PropertyAttribute::BuiltinOrFunctionOrAccessorOrLazyPropertyOrConstant)); return reinterpret_cast<GetFunction>(m_values.value1); }
90 PutFunction propertyPutter() const { ASSERT(!(m_attributes & PropertyAttribute::BuiltinOrFunctionOrAccessorOrLazyPropertyOrConstant)); return reinterpret_cast<PutFunction>(m_values.value2); }
91
92 const DOMJIT::GetterSetter* domJIT() const { ASSERT(m_attributes & PropertyAttribute::DOMJITAttribute); return reinterpret_cast<const DOMJIT::GetterSetter*>(m_values.value1); }
93 const DOMJIT::Signature* signature() const { ASSERT(m_attributes & PropertyAttribute::DOMJITFunction); return reinterpret_cast<const DOMJIT::Signature*>(m_values.value2); }
94
95 NativeFunction accessorGetter() const { ASSERT(m_attributes & PropertyAttribute::Accessor); return NativeFunction(m_values.value1); }
96 NativeFunction accessorSetter() const { ASSERT(m_attributes & PropertyAttribute::Accessor); return NativeFunction(m_values.value2); }
97 BuiltinGenerator builtinAccessorGetterGenerator() const;
98 BuiltinGenerator builtinAccessorSetterGenerator() const;
99
100 long long constantInteger() const { ASSERT(m_attributes & PropertyAttribute::ConstantInteger); return m_values.constant; }
101
102 intptr_t lexerValue() const { ASSERT(!m_attributes); return m_values.value1; }
103
104 ptrdiff_t lazyCellPropertyOffset() const { ASSERT(m_attributes & PropertyAttribute::CellProperty); return m_values.value1; }
105 ptrdiff_t lazyClassStructureOffset() const { ASSERT(m_attributes & PropertyAttribute::ClassStructure); return m_values.value1; }
106
107 LazyPropertyCallback lazyPropertyCallback() const { ASSERT(m_attributes & PropertyAttribute::PropertyCallback); return reinterpret_cast<LazyPropertyCallback>(m_values.value1); }
108 bool isLazyPropertyEnabled(JSGlobalObject* globalObject) const
109 {
110 ASSERT(m_attributes & PropertyAttribute::PropertyCallback);
111 auto isEnabledCallback = bitwise_cast<IsLazyPropertyEnabledCallback>(m_values.value2);
112 return !isEnabledCallback || isEnabledCallback(globalObject);
113 }
114
115 HashTableValue makeReadOnlyCopy() const
116 {
117 ASSERT(m_attributes & PropertyAttribute::AccessorOrCustomAccessorOrValue);
118 HashTableValue copy = *this;
119 copy.m_attributes |= PropertyAttribute::ReadOnly;
120 copy.m_values.value2 = 0;
121 return copy;
122 }
123};

125struct HashTable {
126 int numberOfValues;
127 int indexMask;
128 bool hasSetterOrReadonlyProperties;
129 const ClassInfo* classForThis; // Used by DOMAttribute. Attribute accessors perform type check against this classInfo.
130
131 const HashTableValue* values; // Fixed values generated by script.
132 const CompactHashIndex* index;
133
134 // Find an entry in the table, and return the entry.
135 ALWAYS_INLINE const HashTableValue* entry(PropertyName propertyName) const
136 {
137 if (propertyName.isSymbol())
138 return nullptr;
139
140 auto uid = propertyName.uid();
141 if (!uid)
142 return nullptr;
143
144 int indexEntry = IdentifierRepHash::hash(uid) & indexMask;
145 int valueIndex = index[indexEntry].value;
146 if (valueIndex == -1)
147 return nullptr;
148
149 while (true) {
150 if (WTF::equal(uid, values[valueIndex].m_key))
151 return &values[valueIndex];
152
153 indexEntry = index[indexEntry].next;
154 if (indexEntry == -1)
155 return nullptr;
156 valueIndex = index[indexEntry].value;
157 ASSERT(valueIndex != -1);
158 };
159 }

161 class ConstIterator {
162 public:
163 ConstIterator(const HashTable* table, int position)
164 : m_table(table)
165 , m_position(position)
166 {
167 skipInvalidKeys();
168 }
169
170 const HashTableValue* value() const
171 {
172 return &m_table->values[m_position];
173 }
174
175 const HashTableValue& operator*() const { return *value(); }
176
177 const char* key() const
178 {
179 return m_table->values[m_position].m_key;
180 }
181
182 const HashTableValue* operator->() const
183 {
184 return value();
185 }
186
187 bool operator!=(const ConstIterator& other) const
188 {
189 ASSERT(m_table == other.m_table);
190 return m_position != other.m_position;
191 }
192
193 ConstIterator& operator++()
194 {
195 ASSERT(m_position < m_table->numberOfValues);
196 ++m_position;
197 skipInvalidKeys();
198 return *this;
199 }
200
201 private:
202 void skipInvalidKeys()
203 {
204 ASSERT(m_position <= m_table->numberOfValues);
205 while (m_position < m_table->numberOfValues && !m_table->values[m_position].m_key)
206 ++m_position;
207 ASSERT(m_position <= m_table->numberOfValues);
208 }
209
210 const HashTable* m_table;
211 int m_position;
212 };
213
214 ConstIterator begin() const
215 {
216 return ConstIterator(this, 0);
217 }
218 ConstIterator end() const
219 {
220 return ConstIterator(this, numberOfValues);
221 }
222};

241// FIXME: Inline this function into its call site so staticPropertiesReified() check can be hoisted.
242inline bool getStaticPropertySlotFromTable(VM& vm, const ClassInfo* classInfo, const HashTable& table, JSObject* thisObject, PropertyName propertyName, PropertySlot& slot)
243{
244 if (thisObject->staticPropertiesReified(vm))
245 return false;
246
247 auto* entry = table.entry(propertyName);
248 if (!entry)
249 return false;
250
251 if ((entry->attributes() & PropertyAttribute::PropertyCallback) && !entry->isLazyPropertyEnabled(thisObject->globalObject(vm))) {
252 slot.disableCaching();
253 slot.setIsTaintedByOpaqueObject();
254 return false;
255 }
256
257 if (entry->attributes() & PropertyAttribute::BuiltinOrFunctionOrAccessorOrLazyPropertyOrConstant) {
258 setUpStaticPropertySlot(vm, classInfo, entry, thisObject, propertyName, slot);
259 return true;
260 }
261
262 if (entry->attributes() & PropertyAttribute::DOMJITAttribute) {
263 const DOMJIT::GetterSetter* domJIT = entry->domJIT();
264 slot.setCacheableCustom(thisObject, attributesForStructure(entry->attributes()), domJIT->getter(), entry->propertyPutter(), DOMAttributeAnnotation { classInfo, domJIT });
265 return true;
266 }
267
268 if (entry->attributes() & PropertyAttribute::DOMAttribute) {
269 slot.setCacheableCustom(thisObject, attributesForStructure(entry->attributes()), entry->propertyGetter(), entry->propertyPutter(), DOMAttributeAnnotation { classInfo, nullptr });
270 return true;
271 }
272
273 slot.setCacheableCustom(thisObject, attributesForStructure(entry->attributes()), entry->propertyGetter(), entry->propertyPutter());
274 return true;
275}

277inline bool replaceStaticPropertySlot(VM& vm, JSObject* thisObject, PropertyName propertyName, JSValue value)
278{
279 if (!thisObject->putDirect(vm, propertyName, value))
280 return false;
281
282 if (!thisObject->staticPropertiesReified(vm))
283 thisObject->JSObject::setStructure(vm, Structure::attributeChangeTransition(vm, thisObject->structure(vm), propertyName, 0));
284
285 return true;
286}

288inline void reifyStaticProperty(VM& vm, const ClassInfo* classInfo, const PropertyName& propertyName, const HashTableValue& value, JSObject& thisObj)
289{
290 if (value.attributes() & PropertyAttribute::Builtin) {
291 if (value.attributes() & PropertyAttribute::Accessor)
292 reifyStaticAccessor(vm, value, thisObj, propertyName);
293 else
294 thisObj.putDirectBuiltinFunction(vm, thisObj.globalObject(vm), propertyName, value.builtinGenerator()(vm), attributesForStructure(value.attributes()));
295 return;
296 }
297
298 if (value.attributes() & PropertyAttribute::Function) {
299 if (value.attributes() & PropertyAttribute::DOMJITFunction) {
300 thisObj.putDirectNativeFunction(
301 vm, thisObj.globalObject(vm), propertyName, value.functionLength(),
302 value.function(), value.intrinsic(), value.signature(), attributesForStructure(value.attributes()));
303 return;
304 }
305 thisObj.putDirectNativeFunction(
306 vm, thisObj.globalObject(vm), propertyName, value.functionLength(),
307 value.function(), value.intrinsic(), attributesForStructure(value.attributes()));
308 return;
309 }
310
311 if (value.attributes() & PropertyAttribute::ConstantInteger) {
312 thisObj.putDirect(vm, propertyName, jsNumber(value.constantInteger()), attributesForStructure(value.attributes()));
313 return;
314 }
315
316 if (value.attributes() & PropertyAttribute::Accessor) {
317 reifyStaticAccessor(vm, value, thisObj, propertyName);
318 return;
319 }
320
321 if (value.attributes() & PropertyAttribute::CellProperty) {
322 LazyCellProperty* property = bitwise_cast<LazyCellProperty*>(
323 bitwise_cast<char*>(&thisObj) + value.lazyCellPropertyOffset());
324 JSCell* result = property->get(&thisObj);
325 thisObj.putDirect(vm, propertyName, result, attributesForStructure(value.attributes()));
326 return;
327 }
328
329 if (value.attributes() & PropertyAttribute::ClassStructure) {
330 LazyClassStructure* lazyStructure = bitwise_cast<LazyClassStructure*>(
331 bitwise_cast<char*>(&thisObj) + value.lazyClassStructureOffset());
332 JSObject* constructor = lazyStructure->constructor(jsCast<JSGlobalObject*>(&thisObj));
333 thisObj.putDirect(vm, propertyName, constructor, attributesForStructure(value.attributes()));
334 return;
335 }
336
337 if (value.attributes() & PropertyAttribute::PropertyCallback) {
338 ASSERT(value.isLazyPropertyEnabled(thisObj.globalObject(vm)));
339
340 JSValue result = value.lazyPropertyCallback()(vm, &thisObj);
341 if (result.isGetterSetter())
342 thisObj.putDirectNonIndexAccessor(vm, propertyName, jsCast<GetterSetter*>(result), attributesForStructure(value.attributes()) | PropertyAttribute::Accessor);
343 else if (result.isCustomGetterSetter())
344 thisObj.putDirectCustomAccessor(vm, propertyName, result, attributesForStructure(value.attributes()));
345 else
346 thisObj.putDirect(vm, propertyName, result, attributesForStructure(value.attributes()));
347 return;
348 }
349
350 if (value.attributes() & PropertyAttribute::DOMJITAttribute) {
351 ASSERT_WITH_MESSAGE(classInfo, "DOMJITAttribute should have class info for type checking.");
352 const DOMJIT::GetterSetter* domJIT = value.domJIT();
353 auto* customGetterSetter = DOMAttributeGetterSetter::create(vm, domJIT->getter(), value.propertyPutter(), DOMAttributeAnnotation { classInfo, domJIT });
354 thisObj.putDirectCustomAccessor(vm, propertyName, customGetterSetter, attributesForStructure(value.attributes()));
355 return;
356 }
357
358 if (value.attributes() & PropertyAttribute::DOMAttribute) {
359 ASSERT_WITH_MESSAGE(classInfo, "DOMAttribute should have class info for type checking.");
360 auto* customGetterSetter = DOMAttributeGetterSetter::create(vm, value.propertyGetter(), value.propertyPutter(), DOMAttributeAnnotation { classInfo, nullptr });
361 thisObj.putDirectCustomAccessor(vm, propertyName, customGetterSetter, attributesForStructure(value.attributes()));
362 return;
363 }
364
365 CustomGetterSetter* customGetterSetter = CustomGetterSetter::create(vm, value.propertyGetter(), value.propertyPutter());
366 thisObj.putDirectCustomAccessor(vm, propertyName, customGetterSetter, attributesForStructure(value.attributes()));
367}

369template<unsigned numberOfValues>
370inline void reifyStaticProperties(VM& vm, const ClassInfo* classInfo, const HashTableValue (&values)[numberOfValues], JSObject& thisObj)
371{
372 JSGlobalObject* globalObject = thisObj.globalObject(vm);
373 BatchedTransitionOptimizer transitionOptimizer(vm, &thisObj);
374 for (auto& value : values) {
375 if (!value.m_key)
376 continue;
377 if ((value.attributes() & PropertyAttribute::PropertyCallback) && !value.isLazyPropertyEnabled(globalObject))
378 continue;
379 auto key = Identifier::fromString(vm, reinterpret_cast<const LChar*>(value.m_key), strlen(value.m_key));
380 reifyStaticProperty(vm, classInfo, key, value, thisObj);
381 }
382}
