26#pragma once
27
28#include "ExecutableBase.h"
29
30namespace JSC {
31
32class NativeExecutable final : public ExecutableBase {
33 friend class JIT;
34 friend class LLIntOffsetsExtractor;
35public:
36 typedef ExecutableBase Base;
37 static constexpr unsigned StructureFlags = Base::StructureFlags | StructureIsImmortal;
38
39 static NativeExecutable* create(VM&, Ref<JITCode>&& callThunk, TaggedNativeFunction, Ref<JITCode>&& constructThunk, TaggedNativeFunction constructor, const String& name);
40
41 static void destroy(JSCell*);
42
43 template<typename CellType, SubspaceAccess>
44 static IsoSubspace* subspaceFor(VM& vm)
45 {
46 return &vm.nativeExecutableSpace;
47 }
48
49 CodeBlockHash hashFor(CodeSpecializationKind) const;
50
51 TaggedNativeFunction function() const { return m_function; }
52 TaggedNativeFunction constructor() const { return m_constructor; }
53
54 TaggedNativeFunction nativeFunctionFor(CodeSpecializationKind kind)
55 {
56 if (kind == CodeForCall)
57 return function();
58 ASSERT(kind == CodeForConstruct);
59 return constructor();
60 }
61
62 static ptrdiff_t offsetOfNativeFunctionFor(CodeSpecializationKind kind)
63 {
64 if (kind == CodeForCall)
65 return OBJECT_OFFSETOF(NativeExecutable, m_function);
66 ASSERT(kind == CodeForConstruct);
67 return OBJECT_OFFSETOF(NativeExecutable, m_constructor);
68 }
69
70 DECLARE_VISIT_CHILDREN;
71 static Structure* createStructure(VM&, JSGlobalObject*, JSValue proto);
72
73 DECLARE_INFO;
74
75 const String& name() const { return m_name; }
76
77 const DOMJIT::Signature* signatureFor(CodeSpecializationKind) const;
78 Intrinsic intrinsic() const;
79
80 JSString* toString(JSGlobalObject* globalObject)
81 {
82 if (!m_asString)
83 return toStringSlow(globalObject);
84 return m_asString.get();
85 }
86
87 JSString* asStringConcurrently() const { return m_asString.get(); }
88 static inline ptrdiff_t offsetOfAsString() { return OBJECT_OFFSETOF(NativeExecutable, m_asString); }
89
90private:
91 NativeExecutable(VM&, TaggedNativeFunction, TaggedNativeFunction constructor);
92 void finishCreation(VM&, Ref<JITCode>&& callThunk, Ref<JITCode>&& constructThunk, const String& name);
93
94 JSString* toStringSlow(JSGlobalObject*);
95
96 TaggedNativeFunction m_function;
97 TaggedNativeFunction m_constructor;
98
99 String m_name;
100 WriteBarrier<JSString> m_asString;
101};
102
103} // namespace JSC
