21#pragma once
22
23#include "JSWrapperObject.h"
24
25namespace JSC {
26
27class NumberObject : public JSWrapperObject {
28protected:
29 NumberObject(VM&, Structure*);
30 void finishCreation(VM&);
31
32public:
33 using Base = JSWrapperObject;
34
35 template<typename, SubspaceAccess mode>
36 static IsoSubspace* subspaceFor(VM& vm)
37 {
38 return &vm.numberObjectSpace;
39 }
40
41 static NumberObject* create(VM& vm, Structure* structure)
42 {
43 NumberObject* number = new (NotNull, allocateCell<NumberObject>(vm.heap)) NumberObject(vm, structure);
44 number->finishCreation(vm);
45 return number;
46 }
47
48 DECLARE_EXPORT_INFO;
49
50 static Structure* createStructure(VM& vm, JSGlobalObject* globalObject, JSValue prototype)
51 {
52 return Structure::create(vm, globalObject, prototype, TypeInfo(NumberObjectType, StructureFlags), info());
53 }
54};
55static_assert(sizeof(NumberObject) == sizeof(JSWrapperObject));
56
57JS_EXPORT_PRIVATE NumberObject* constructNumber(JSGlobalObject*, JSValue);
58
59} // namespace JSC
