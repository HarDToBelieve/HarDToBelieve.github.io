38// ECMA 262-3 8.6.1
39// Property attributes
40enum class PropertyAttribute : unsigned {
41 None = 0,
42 ReadOnly = 1 << 1, // property can be only read, not written
43 DontEnum = 1 << 2, // property doesn't appear in (for .. in ..)
44 DontDelete = 1 << 3, // property can't be deleted
45 Accessor = 1 << 4, // property is a getter/setter
46 CustomAccessor = 1 << 5,
47 CustomValue = 1 << 6,
48 CustomAccessorOrValue = CustomAccessor | CustomValue,
49 AccessorOrCustomAccessorOrValue = Accessor | CustomAccessor | CustomValue,
50 ReadOnlyOrAccessorOrCustomAccessor = ReadOnly | Accessor | CustomAccessor,
51
52 // Things that are used by static hashtables are not in the attributes byte in PropertyMapEntry.
53 Function = 1 << 8, // property is a function - only used by static hashtables
54 Builtin = 1 << 9, // property is a builtin function - only used by static hashtables
55 ConstantInteger = 1 << 10, // property is a constant integer - only used by static hashtables
56 CellProperty = 1 << 11, // property is a lazy property - only used by static hashtables
57 ClassStructure = 1 << 12, // property is a lazy class structure - only used by static hashtables
58 PropertyCallback = 1 << 13, // property that is a lazy property callback - only used by static hashtables
59 DOMAttribute = 1 << 14, // property is a simple DOM attribute - only used by static hashtables
60 DOMJITAttribute = 1 << 15, // property is a DOM JIT attribute - only used by static hashtables
61 DOMJITFunction = 1 << 16, // property is a DOM JIT function - only used by static hashtables
62 BuiltinOrFunction = Builtin | Function, // helper only used by static hashtables
63 BuiltinOrFunctionOrLazyProperty = Builtin | Function | CellProperty | ClassStructure | PropertyCallback, // helper only used by static hashtables
64 BuiltinOrFunctionOrAccessorOrLazyProperty = Builtin | Function | Accessor | CellProperty | ClassStructure | PropertyCallback, // helper only used by static hashtables
65 BuiltinOrFunctionOrAccessorOrLazyPropertyOrConstant = Builtin | Function | Accessor | CellProperty | ClassStructure | PropertyCallback | ConstantInteger // helper only used by static hashtables
66};
67
68static constexpr unsigned operator| (PropertyAttribute a, PropertyAttribute b) { return static_cast<unsigned>(a) | static_cast<unsigned>(b); }
69static constexpr unsigned operator| (unsigned a, PropertyAttribute b) { return a | static_cast<unsigned>(b); }
70static constexpr unsigned operator| (PropertyAttribute a, unsigned b) { return static_cast<unsigned>(a) | b; }
71static constexpr unsigned operator&(unsigned a, PropertyAttribute b) { return a & static_cast<unsigned>(b); }
72static constexpr bool operator<(PropertyAttribute a, PropertyAttribute b) { return static_cast<unsigned>(a) < static_cast<unsigned>(b); }
73static constexpr unsigned operator~(PropertyAttribute a) { return ~static_cast<unsigned>(a); }
74static constexpr bool operator<(PropertyAttribute a, unsigned b) { return static_cast<unsigned>(a) < b; }
75static inline unsigned& operator|=(unsigned& a, PropertyAttribute b) { return a |= static_cast<unsigned>(b); }

77enum CacheabilityType : uint8_t {
78 CachingDisallowed,
79 CachingAllowed
80};

82inline unsigned attributesForStructure(unsigned attributes)
83{
84 // The attributes that are used just for the static hashtable are at bit 8 and higher.
85 return static_cast<uint8_t>(attributes);
86}

94class PropertySlot {
95
96 // We rely on PropertySlot being stack allocated when used. This is needed
97 // because we rely on some of its fields being a GC root. For example, it
98 // may be the only thing that points to the GetterSetter property it has.
99 WTF_FORBID_HEAP_ALLOCATION;
100
101 enum PropertyType : uint8_t {
102 TypeUnset,
103 TypeValue,
104 TypeGetter,
105 TypeCustom,
106 };
107
108public:
109 enum class InternalMethodType : uint8_t {
110 Get, // [[Get]] internal method in the spec.
111 HasProperty, // [[HasProperty]] internal method in the spec.
112 GetOwnProperty, // [[GetOwnProperty]] internal method in the spec.
113 VMInquiry, // Our VM is just poking around. When this is the InternalMethodType, getOwnPropertySlot is not allowed to do user observable actions.
114 };
115
116 enum class AdditionalDataType : uint8_t {
117 None,
118 DOMAttribute, // Annotated with DOMAttribute information.
119 ModuleNamespace, // ModuleNamespaceObject's environment access.
120 };
121
122 explicit PropertySlot(const JSValue thisValue, InternalMethodType internalMethodType, VM* vmForInquiry = nullptr)
123 : m_thisValue(thisValue)
124 , m_internalMethodType(internalMethodType)
125 {
126 if (isVMInquiry())
127 disallowVMEntry.emplace(*vmForInquiry);
128 }

137 bool isCacheable() const { return isUnset() || m_cacheability == CachingAllowed; }

143 bool isCacheableGetter() const { return isCacheable() && isAccessor(); }
144 bool isCacheableCustom() const { return isCacheable() && isCustom(); }
145 void setIsTaintedByOpaqueObject() { m_isTaintedByOpaqueObject = true; }
146 bool isTaintedByOpaqueObject() const { return m_isTaintedByOpaqueObject; }

148 InternalMethodType internalMethodType() const { return m_internalMethodType; }
149 bool isVMInquiry() const { return m_internalMethodType == InternalMethodType::VMInquiry; }

263 void setCustom(JSObject* slotBase, unsigned attributes, GetValueFunc getValue, PutValueFunc putValue = nullptr)
264 {
265 ASSERT(attributes == attributesForStructure(attributes));
266
267 ASSERT(getValue);
268 assertIsCFunctionPtr(getValue);
269 m_data.custom.getValue = getValue;
270 assertIsNullOrCFunctionPtr(putValue);
271 m_data.custom.putValue = putValue;
272 m_attributes = attributes;
273
274 ASSERT(slotBase);
275 m_slotBase = slotBase;
276 m_propertyType = TypeCustom;
277 ASSERT(m_cacheability == CachingDisallowed);
278 }

372 union {
373 EncodedJSValue value;

402ALWAYS_INLINE JSValue PropertySlot::getValue(JSGlobalObject* globalObject, PropertyName propertyName) const
403{
404 if (m_propertyType == TypeValue)
405 return JSValue::decode(m_data.value);
406 if (m_propertyType == TypeGetter)
407 return functionGetter(globalObject);
408 return customGetter(globalObject, propertyName);
409}
410
411ALWAYS_INLINE JSValue PropertySlot::getValue(JSGlobalObject* globalObject, uint64_t propertyName) const
412{
413 VM& vm = getVM(globalObject);
414 if (m_propertyType == TypeValue)
415 return JSValue::decode(m_data.value);
416 if (m_propertyType == TypeGetter)
417 return functionGetter(globalObject);
418 return customGetter(globalObject, Identifier::from(vm, propertyName));
419}
