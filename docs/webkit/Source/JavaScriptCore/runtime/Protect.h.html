21#pragma once
22
23#include "Heap.h"
24#include "JSCJSValue.h"
25
26namespace JSC {
27
28inline void gcProtect(JSCell* val)
29{
30 Heap::heap(val)->protect(val);
31}
32
33inline void gcUnprotect(JSCell* val)
34{
35 Heap::heap(val)->unprotect(val);
36}
37
38inline void gcProtectNullTolerant(JSCell* val)
39{
40 if (val)
41 gcProtect(val);
42}
43
44inline void gcUnprotectNullTolerant(JSCell* val)
45{
46 if (val)
47 gcUnprotect(val);
48}
49
50inline void gcProtect(JSValue value)
51{
52 if (value && value.isCell())
53 gcProtect(value.asCell());
54}
55
56inline void gcUnprotect(JSValue value)
57{
58 if (value && value.isCell())
59 gcUnprotect(value.asCell());
60}
61
62} // namespace JSC
