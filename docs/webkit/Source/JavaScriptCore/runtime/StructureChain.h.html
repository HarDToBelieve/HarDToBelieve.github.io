34namespace JSC {
35
36class LLIntOffsetsExtractor;
37class Structure;
38
39class StructureChain final : public JSCell {
40 friend class JIT;
41
42public:
43 using Base = JSCell;
44 static constexpr unsigned StructureFlags = Base::StructureFlags | StructureIsImmortal;
45
46 template<typename CellType, SubspaceAccess>
47 static IsoSubspace* subspaceFor(VM& vm)
48 {
49 return &vm.structureChainSpace;
50 }
51
52 static StructureChain* create(VM&, JSObject*);
53 StructureID* head() { return m_vector.get(); }
54 DECLARE_VISIT_CHILDREN;
55
56 static Structure* createStructure(VM& vm, JSGlobalObject* globalObject, JSValue prototype)
57 {
58 return Structure::create(vm, globalObject, prototype, TypeInfo(CellType, StructureFlags), info());
59 }
60
61 DECLARE_INFO;
62
63private:
64 friend class LLIntOffsetsExtractor;
65
66 void finishCreation(VM&, JSObject* head);
67
68 StructureChain(VM&, Structure*, StructureID*);
69 AuxiliaryBarrier<StructureID*> m_vector;
70};

72} // namespace JSC
