75inline bool Structure::mayInterceptIndexedAccesses() const
76{
77 if (indexingModeIncludingHistory() & MayHaveIndexedAccessors)
78 return true;
79
80 // Consider a scenario where object O (of global G1)'s prototype is set to A
81 // (of global G2), and G2 is already having a bad time. If an object B with
82 // indexed accessors is then set as the prototype of A:
83 // O -> A -> B
84 // Then, O should be converted to SlowPutArrayStorage (because it now has an
85 // object with indexed accessors in its prototype chain). But it won't be
86 // converted because this conversion is done by JSGlobalObject::haveAbadTime(),
87 // but G2 is already having a bad time. We solve this by conservatively
88 // treating A as potentially having indexed accessors if its global is already
89 // having a bad time. Hence, when A is set as O's prototype, O will be
90 // converted to SlowPutArrayStorage.
91
92 JSGlobalObject* globalObject = this->globalObject();
93 if (!globalObject)
94 return false;
95 return globalObject->isHavingABadTime();
96}

148ALWAYS_INLINE PropertyOffset Structure::get(VM& vm, PropertyName propertyName, unsigned& attributes)
149{
150 ASSERT(!isCompilationThread());
151 ASSERT(structure(vm)->classInfo() == info());
152
153 if (m_seenProperties.ruleOut(bitwise_cast<uintptr_t>(propertyName.uid())))
154 return invalidOffset;
155
156 PropertyTable* propertyTable = ensurePropertyTableIfNotEmpty(vm);
157 if (!propertyTable)
158 return invalidOffset;
159
160 PropertyMapEntry* entry = propertyTable->get(propertyName.uid());
161 if (!entry)
162 return invalidOffset;
163
164 attributes = entry->attributes;
165 return entry->offset;
166}

168template<typename Functor>
169void Structure::forEachPropertyConcurrently(const Functor& functor)
170{
171 Vector<Structure*, 8> structures;
172 Structure* tableStructure;
173 PropertyTable* table;
174
175 bool didFindStructure = findStructuresAndMapForMaterialization(structures, tableStructure, table);
176
177 HashSet<UniquedStringImpl*> seenProperties;
178
179 for (auto* structure : structures) {
180 if (!structure->m_transitionPropertyName || seenProperties.contains(structure->m_transitionPropertyName.get()))
181 continue;
182
183 seenProperties.add(structure->m_transitionPropertyName.get());
184
185 switch (structure->transitionKind()) {
186 case TransitionKind::PropertyAddition:
187 case TransitionKind::PropertyAttributeChange:
188 break;
189 case TransitionKind::PropertyDeletion:
190 case TransitionKind::SetBrand:
191 continue;
192 default:
193 ASSERT_NOT_REACHED();
194 break;
195 }
196
197 if (!functor(PropertyMapEntry(structure->m_transitionPropertyName.get(), structure->transitionOffset(), structure->transitionPropertyAttributes()))) {
198 if (didFindStructure) {
199 assertIsHeld(tableStructure->m_lock); // Sadly Clang needs some help here.
200 tableStructure->m_lock.unlock();
201 }
202 return;
203 }
204 }
205
206 if (didFindStructure) {
207 assertIsHeld(tableStructure->m_lock); // Sadly Clang needs some help here.
208 for (auto& entry : *table) {
209 if (seenProperties.contains(entry.key))
210 continue;
211
212 if (!functor(entry)) {
213 tableStructure->m_lock.unlock();
214 return;
215 }
216 }
217 tableStructure->m_lock.unlock();
218 }
219}

324inline StructureChain* Structure::prototypeChain(VM& vm, JSGlobalObject* globalObject, JSObject* base) const
325{
326 ASSERT(base->structure(vm) == this);

328 if (!isValid(globalObject, m_cachedPrototypeChain.get(), base)) {
329 JSValue prototype = prototypeForLookup(globalObject, base);

340inline bool Structure::isValid(JSGlobalObject* globalObject, StructureChain* cachedPrototypeChain, JSObject* base) const
341{
342 if (!cachedPrototypeChain)
343 return false;
344
345 VM& vm = globalObject->vm();
346 JSValue prototype = prototypeForLookup(globalObject, base);
347 StructureID* cachedStructure = cachedPrototypeChain->head();
348 while (*cachedStructure && !prototype.isNull()) {
349 if (asObject(prototype)->structureID() != *cachedStructure)
350 return false;
351 ++cachedStructure;
352 prototype = asObject(prototype)->getPrototypeDirect(vm);
353 }
354 return prototype.isNull() && !*cachedStructure;
355}

376template<typename DetailsFunc>
377ALWAYS_INLINE bool Structure::checkOffsetConsistency(PropertyTable* propertyTable, const DetailsFunc& detailsFunc) const
378{
379 // We cannot reliably assert things about the property table in the concurrent
380 // compilation thread. It is possible for the table to be stolen and then have
381 // things added to it, which leads to the offsets being all messed up. We could
382 // get around this by grabbing a lock here, but I think that would be overkill.
383 if (isCompilationThread())
384 return true;
385
386 unsigned totalSize = propertyTable->propertyStorageSize();
387 unsigned inlineOverflowAccordingToTotalSize = totalSize < m_inlineCapacity ? 0 : totalSize - m_inlineCapacity;
388
389 auto fail = [&] (const char* description) {
390 dataLog("Detected offset inconsistency: ", description, "!\n");
391 dataLog("this = ", RawPointer(this), "\n");
392 dataLog("transitionOffset = ", transitionOffset(), "\n");
393 dataLog("maxOffset = ", maxOffset(), "\n");
394 dataLog("m_inlineCapacity = ", m_inlineCapacity, "\n");
395 dataLog("propertyTable = ", RawPointer(propertyTable), "\n");
396 dataLog("numberOfSlotsForMaxOffset = ", numberOfSlotsForMaxOffset(maxOffset(), m_inlineCapacity), "\n");
397 dataLog("totalSize = ", totalSize, "\n");
398 dataLog("inlineOverflowAccordingToTotalSize = ", inlineOverflowAccordingToTotalSize, "\n");
399 dataLog("numberOfOutOfLineSlotsForMaxOffset = ", numberOfOutOfLineSlotsForMaxOffset(maxOffset()), "\n");
400 detailsFunc();
401 UNREACHABLE_FOR_PLATFORM();
402 };
403
404 if (numberOfSlotsForMaxOffset(maxOffset(), m_inlineCapacity) != totalSize)
405 fail("numberOfSlotsForMaxOffset doesn't match totalSize");
406 if (inlineOverflowAccordingToTotalSize != numberOfOutOfLineSlotsForMaxOffset(maxOffset()))
407 fail("inlineOverflowAccordingToTotalSize doesn't match numberOfOutOfLineSlotsForMaxOffset");
408
409 return true;
410}

412ALWAYS_INLINE bool Structure::checkOffsetConsistency() const
413{
414 PropertyTable* propertyTable = propertyTableOrNull();
415
416 if (!propertyTable) {
417 ASSERT(!isPinnedPropertyTable());
418 return true;
419 }
420
421 // We cannot reliably assert things about the property table in the concurrent
422 // compilation thread. It is possible for the table to be stolen and then have
423 // things added to it, which leads to the offsets being all messed up. We could
424 // get around this by grabbing a lock here, but I think that would be overkill.
425 if (isCompilationThread())
426 return true;
427
428 return checkOffsetConsistency(propertyTable, [] () { });
429}

450template<Structure::ShouldPin shouldPin, typename Func>
451inline PropertyOffset Structure::add(VM& vm, PropertyName propertyName, unsigned attributes, const Func& func)
452{
453 PropertyTable* table = ensurePropertyTable(vm);
454
455 GCSafeConcurrentJSLocker locker(m_lock, vm.heap);
456
457 switch (shouldPin) {
458 case ShouldPin::Yes:
459 pin(locker, vm, table);
460 break;
461 case ShouldPin::No:
462 setPropertyTable(vm, table);
463 break;
464 }
465
466 ASSERT(!JSC::isValidOffset(get(vm, propertyName)));
467
468 checkConsistency();
469 if (attributes & PropertyAttribute::DontEnum || propertyName.isSymbol())
470 setIsQuickPropertyAccessAllowedForEnumeration(false);
471 if (propertyName == vm.propertyNames->underscoreProto)
472 setHasUnderscoreProtoPropertyExcludingOriginalProto(true);
473
474 auto rep = propertyName.uid();
475
476 PropertyOffset newOffset = table->nextOffset(m_inlineCapacity);
477
478 m_propertyHash = m_propertyHash ^ rep->existingSymbolAwareHash();
479 m_seenProperties.add(bitwise_cast<uintptr_t>(rep));
480
481 auto result = table->add(vm, PropertyMapEntry(rep, newOffset, attributes));
482 ASSERT_UNUSED(result, result.second);
483 ASSERT_UNUSED(result, result.first.first->offset == newOffset);
484 auto newMaxOffset = std::max(newOffset, maxOffset());
485
486 func(locker, newOffset, newMaxOffset);
487
488 ASSERT(maxOffset() == newMaxOffset);
489
490 checkConsistency();
491 return newOffset;
492}

494template<Structure::ShouldPin shouldPin, typename Func>
495inline PropertyOffset Structure::remove(VM& vm, PropertyName propertyName, const Func& func)
496{
497 PropertyTable* table = ensurePropertyTable(vm);
498 GCSafeConcurrentJSLocker locker(m_lock, vm.heap);
499
500 switch (shouldPin) {
501 case ShouldPin::Yes:
502 pin(locker, vm, table);
503 break;
504 case ShouldPin::No:
505 setPropertyTable(vm, table);
506 break;
507 }
508
509 ASSERT(JSC::isValidOffset(get(vm, propertyName)));
510
511 checkConsistency();
512
513 auto rep = propertyName.uid();
514
515 PropertyTable::find_iterator position = table->find(rep);
516 if (!position.first)
517 return invalidOffset;
518
519 setIsQuickPropertyAccessAllowedForEnumeration(false);
520
521 PropertyOffset offset = position.first->offset;
522
523 table->remove(vm, position);
524 table->addDeletedOffset(offset);
525
526 PropertyOffset newMaxOffset = maxOffset();
527
528 func(locker, offset, newMaxOffset);
529
530 ASSERT(maxOffset() == newMaxOffset);
531 ASSERT(!JSC::isValidOffset(get(vm, propertyName)));
532
533 checkConsistency();
534 return offset;
535}

537template<Structure::ShouldPin shouldPin, typename Func>
538inline PropertyOffset Structure::attributeChange(VM& vm, PropertyName propertyName, unsigned attributes, const Func& func)
539{
540 PropertyTable* table = ensurePropertyTable(vm);
541
542 GCSafeConcurrentJSLocker locker(m_lock, vm.heap);
543
544 switch (shouldPin) {
545 case ShouldPin::Yes:
546 pin(locker, vm, table);
547 break;
548 case ShouldPin::No:
549 setPropertyTable(vm, table);
550 break;
551 }
552
553 ASSERT(JSC::isValidOffset(get(vm, propertyName)));
554
555 checkConsistency();
556 PropertyMapEntry* entry = table->get(propertyName.uid());
557 if (!entry)
558 return invalidOffset;
559
560 PropertyOffset offset = entry->offset;
561
562 if (attributes & PropertyAttribute::DontEnum)
563 setIsQuickPropertyAccessAllowedForEnumeration(false);
564 if (attributes & PropertyAttribute::ReadOnly)
565 setContainsReadOnlyProperties();
566
567 entry->attributes = attributes;
568
569 PropertyOffset newMaxOffset = maxOffset();
570
571 func(locker, offset, newMaxOffset);
572
573 ASSERT(maxOffset() == newMaxOffset);
574 ASSERT(JSC::isValidOffset(get(vm, propertyName)));
575
576 checkConsistency();
577 return offset;
578}

626ALWAYS_INLINE bool Structure::shouldConvertToPolyProto(const Structure* a, const Structure* b)
627{
628 if (!a || !b)
629 return false;
630
631 if (a == b)
632 return false;
633
634 if (a->propertyHash() != b->propertyHash())
635 return false;
636
637 // We only care about objects created via a constructor's to_this. These
638 // all have Structures with rare data and a sharedPolyProtoWatchpoint.
639 if (!a->hasRareData() || !b->hasRareData())
640 return false;
641
642 // We only care about Structure's generated from functions that share
643 // the same executable.
644 const Box<InlineWatchpointSet>& aInlineWatchpointSet = a->rareData()->sharedPolyProtoWatchpoint();
645 const Box<InlineWatchpointSet>& bInlineWatchpointSet = b->rareData()->sharedPolyProtoWatchpoint();
646 if (aInlineWatchpointSet.get() != bInlineWatchpointSet.get() || !aInlineWatchpointSet)
647 return false;
648 ASSERT(aInlineWatchpointSet && bInlineWatchpointSet && aInlineWatchpointSet.get() == bInlineWatchpointSet.get());
649
650 if (a->hasPolyProto() || b->hasPolyProto())
651 return false;
652
653 if (a->storedPrototype() == b->storedPrototype())
654 return false;
655
656 VM& vm = a->vm();
657 JSObject* aObj = a->storedPrototypeObject();
658 JSObject* bObj = b->storedPrototypeObject();
659 while (aObj && bObj) {
660 a = aObj->structure(vm);
661 b = bObj->structure(vm);
662
663 if (a->propertyHash() != b->propertyHash())
664 return false;
665
666 aObj = a->storedPrototypeObject(aObj);
667 bObj = b->storedPrototypeObject(bObj);
668 }
669
670 return !aObj && !bObj;
671}

673inline Structure* Structure::nonPropertyTransition(VM& vm, Structure* structure, TransitionKind transitionKind)
674{
675 IndexingType indexingModeIncludingHistory = newIndexingType(structure->indexingModeIncludingHistory(), transitionKind);
676
677 if (changesIndexingType(transitionKind)) {
678 if (JSGlobalObject* globalObject = structure->m_globalObject.get()) {
679 if (globalObject->isOriginalArrayStructure(structure)) {
680 Structure* result = globalObject->originalArrayStructureForIndexingType(indexingModeIncludingHistory);
681 if (result->indexingModeIncludingHistory() == indexingModeIncludingHistory) {
682 structure->didTransitionFromThisStructure();
683 return result;
684 }
685 }
686 }
687 }
688
689 return nonPropertyTransitionSlow(vm, structure, transitionKind);
690}
