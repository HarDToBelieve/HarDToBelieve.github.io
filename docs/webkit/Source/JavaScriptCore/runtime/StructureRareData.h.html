34namespace JSC {
35
36class JSPropertyNameEnumerator;
37class Structure;
38class CachedSpecialPropertyAdaptiveStructureWatchpoint;
39class CachedSpecialPropertyAdaptiveInferredPropertyValueWatchpoint;
40struct SpecialPropertyCache;
41enum class CachedPropertyNamesKind : uint8_t {
42 Keys = 0,
43 GetOwnPropertyNames,
44};
45static constexpr unsigned numberOfCachedPropertyNames = 2;
46
47enum class CachedSpecialPropertyKey : uint8_t {
48 ToStringTag = 0,
49 ToString,
50 ValueOf,
51 ToPrimitive,
52};
53static constexpr unsigned numberOfCachedSpecialPropertyKeys = 4;
54
55class StructureRareData final : public JSCell {
56public:
57 typedef JSCell Base;
58 static constexpr unsigned StructureFlags = Base::StructureFlags | StructureIsImmortal;
59
60 template<typename CellType, SubspaceAccess>
61 static IsoSubspace* subspaceFor(VM& vm)
62 {
63 return &vm.structureRareDataSpace;
64 }
65
66 static StructureRareData* create(VM&, Structure*);
67
68 static constexpr bool needsDestruction = true;
69 static void destroy(JSCell*);
70
71 DECLARE_VISIT_CHILDREN;
72
73 static Structure* createStructure(VM&, JSGlobalObject*, JSValue prototype);
74
75 Structure* previousID() const
76 {
77 return m_previous.get();
78 }
79 void setPreviousID(VM&, Structure*);
80 void clearPreviousID();
81
82 JSValue cachedSpecialProperty(CachedSpecialPropertyKey) const;
83 void cacheSpecialProperty(JSGlobalObject*, VM&, Structure* baseStructure, JSValue, CachedSpecialPropertyKey, const PropertySlot&);
84
85 JSPropertyNameEnumerator* cachedPropertyNameEnumerator() const;
86 void setCachedPropertyNameEnumerator(VM&, JSPropertyNameEnumerator*);
87
88 JSImmutableButterfly* cachedPropertyNames(CachedPropertyNamesKind) const;
89 JSImmutableButterfly* cachedPropertyNamesIgnoringSentinel(CachedPropertyNamesKind) const;
90 JSImmutableButterfly* cachedPropertyNamesConcurrently(CachedPropertyNamesKind) const;
91 void setCachedPropertyNames(VM&, CachedPropertyNamesKind, JSImmutableButterfly*);
92
93 Box<InlineWatchpointSet> copySharedPolyProtoWatchpoint() const { return m_polyProtoWatchpoint; }
94 const Box<InlineWatchpointSet>& sharedPolyProtoWatchpoint() const { return m_polyProtoWatchpoint; }
95 void setSharedPolyProtoWatchpoint(Box<InlineWatchpointSet>&& sharedPolyProtoWatchpoint) { m_polyProtoWatchpoint = WTFMove(sharedPolyProtoWatchpoint); }
96 bool hasSharedPolyProtoWatchpoint() const { return static_cast<bool>(m_polyProtoWatchpoint); }
97
98 static JSImmutableButterfly* cachedPropertyNamesSentinel() { return bitwise_cast<JSImmutableButterfly*>(static_cast<uintptr_t>(1)); }
99
100 static ptrdiff_t offsetOfCachedPropertyNames(CachedPropertyNamesKind kind)
101 {
102 return OBJECT_OFFSETOF(StructureRareData, m_cachedPropertyNames) + sizeof(WriteBarrier<JSImmutableButterfly>) * static_cast<unsigned>(kind);
103 }
104
105 DECLARE_EXPORT_INFO;
106
107 void finalizeUnconditionally(VM&);
108
109private:
110 friend class Structure;
111 friend class CachedSpecialPropertyAdaptiveStructureWatchpoint;
112 friend class CachedSpecialPropertyAdaptiveInferredPropertyValueWatchpoint;
113
114 StructureRareData(VM&, Structure*);
115
116 void clearCachedSpecialProperty(CachedSpecialPropertyKey);
117 void cacheSpecialPropertySlow(JSGlobalObject*, VM&, Structure* baseStructure, JSValue, CachedSpecialPropertyKey, const PropertySlot&);
118
119 SpecialPropertyCache& ensureSpecialPropertyCache();
120 SpecialPropertyCache& ensureSpecialPropertyCacheSlow();
121 bool canCacheSpecialProperty(CachedSpecialPropertyKey);
122 void giveUpOnSpecialPropertyCache(CachedSpecialPropertyKey);
123
124 WriteBarrier<Structure> m_previous;
125 // FIXME: We should have some story for clearing these property names caches in GC.
126 // https://bugs.webkit.org/show_bug.cgi?id=192659
127 WriteBarrier<JSPropertyNameEnumerator> m_cachedPropertyNameEnumerator;
128 WriteBarrier<JSImmutableButterfly> m_cachedPropertyNames[numberOfCachedPropertyNames] { };
129
130 typedef HashMap<PropertyOffset, RefPtr<WatchpointSet>, WTF::IntHash<PropertyOffset>, WTF::UnsignedWithZeroKeyHashTraits<PropertyOffset>> PropertyWatchpointMap;
131#ifdef NDEBUG
132 static_assert(sizeof(PropertyWatchpointMap) == sizeof(void*), "StructureRareData should remain small");
133#endif
134
135 PropertyWatchpointMap m_replacementWatchpointSets;
136 std::unique_ptr<SpecialPropertyCache> m_specialPropertyCache;
137 Box<InlineWatchpointSet> m_polyProtoWatchpoint;
138
139 PropertyOffset m_maxOffset;
140 PropertyOffset m_transitionOffset;
141};

143} // namespace JSC
