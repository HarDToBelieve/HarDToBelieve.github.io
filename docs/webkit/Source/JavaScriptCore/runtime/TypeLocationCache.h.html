26#pragma once
27
28#include "TypeLocation.h"
29#include <wtf/FastMalloc.h>
30#include <wtf/HashMethod.h>
31#include <wtf/StdUnorderedMap.h>
32
33namespace JSC {
34
35class VM;
36
37class TypeLocationCache {
38public:
39 struct LocationKey {
40 LocationKey() {}
41 bool operator==(const LocationKey& other) const
42 {
43 return m_globalVariableID == other.m_globalVariableID
44 && m_sourceID == other.m_sourceID
45 && m_start == other.m_start
46 && m_end == other.m_end;
47 }
48
49 unsigned hash() const
50 {
51 return m_globalVariableID + m_sourceID + m_start + m_end;
52 }
53
54 GlobalVariableID m_globalVariableID;
55 intptr_t m_sourceID;
56 unsigned m_start;
57 unsigned m_end;
58 };
59
60 std::pair<TypeLocation*, bool> getTypeLocation(GlobalVariableID, intptr_t, unsigned start, unsigned end, RefPtr<TypeSet>&&, VM*);
61private:
62 using LocationMap = StdUnorderedMap<LocationKey, TypeLocation*, HashMethod<LocationKey>>;
63 LocationMap m_locationMap;
64};
65
66} // namespace JSC
