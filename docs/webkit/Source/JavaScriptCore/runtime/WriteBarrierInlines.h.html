31namespace JSC {
32
33template <typename T, typename Traits>
34inline void WriteBarrierBase<T, Traits>::set(VM& vm, const JSCell* owner, T* value)
35{
36 ASSERT(value);
37 ASSERT(!Options::useConcurrentJIT() || !isCompilationThread());
38 validateCell(value);
39 setEarlyValue(vm, owner, value);
40}
41
42template <typename T, typename Traits>
43inline void WriteBarrierBase<T, Traits>::setMayBeNull(VM& vm, const JSCell* owner, T* value)
44{
45 if (value)
46 validateCell(value);
47 setEarlyValue(vm, owner, value);
48}
49
50template <typename T, typename Traits>
51inline void WriteBarrierBase<T, Traits>::setEarlyValue(VM& vm, const JSCell* owner, T* value)
52{
53 Traits::exchange(this->m_cell, value);
54 vm.heap.writeBarrier(owner, static_cast<JSCell*>(value));
55}
56
57inline void WriteBarrierBase<Unknown, RawValueTraits<Unknown>>::set(VM& vm, const JSCell* owner, JSValue value)
58{
59 ASSERT(!Options::useConcurrentJIT() || !isCompilationThread());
60 m_value = JSValue::encode(value);
61 vm.heap.writeBarrier(owner, value);
62}

64} // namespace JSC
