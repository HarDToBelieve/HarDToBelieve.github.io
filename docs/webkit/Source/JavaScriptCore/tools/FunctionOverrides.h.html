33namespace JSC {
34
35class ScriptExecutable;
36
37struct FunctionOverrideInfo {
38 SourceCode sourceCode;
39 unsigned firstLine;
40 unsigned lineCount;
41 unsigned startColumn;
42 unsigned endColumn;
43 unsigned parametersStartOffset;
44 unsigned typeProfilingStartOffset;
45 unsigned typeProfilingEndOffset;
46};
47
48class FunctionOverrides {
49public:
50 using OverrideInfo = FunctionOverrideInfo;
51
52 static FunctionOverrides& overrides();
53 FunctionOverrides(const char* functionOverridesFileName);
54
55 static bool initializeOverrideFor(const SourceCode& origCode, OverrideInfo& result);
56
57 JS_EXPORT_PRIVATE static void reinstallOverrides();
58
59private:
60 void parseOverridesInFile(const char* fileName) WTF_REQUIRES_LOCK(m_lock);
61 void clear() WTF_REQUIRES_LOCK(m_lock) { m_entries.clear(); }
62
63 HashMap<String, String> m_entries WTF_GUARDED_BY_LOCK(m_lock);
64 Lock m_lock;
65};

67} // namespace JSC
