40enum class AuditLevel {
41 None,
42 Minimal,
43 Full,
44 Random,
45};
46
47#ifdef NDEBUG
48static constexpr AuditLevel DefaultAuditLevel = AuditLevel::None;
49#else
50static constexpr AuditLevel DefaultAuditLevel = AuditLevel::Random;
51#endif

53class Random {
54public:
55 Random(VM&);

57 ALWAYS_INLINE bool shouldAudit(VM&);
58
59private:
60 JS_EXPORT_PRIVATE bool reloadAndCheckShouldAuditSlow(VM&);
61
62 uint64_t m_triggerBits;
63 Lock m_lock;
64
65 // The top bit is reserved as a termination bit. Hence, the number of
66 // trigger bits is always 1 less than will fit in m_triggerBits.
67 static constexpr int numberOfTriggerBits = (sizeof(m_triggerBits) * CHAR_BIT) - 1;
68};

70ALWAYS_INLINE void auditCellRandomly(VM&, JSCell*);
71ALWAYS_INLINE void auditCellMinimally(VM&, JSCell*);
72JS_EXPORT_PRIVATE void auditCellMinimallySlow(VM&, JSCell*);
73JS_EXPORT_PRIVATE void auditCellFully(VM&, JSCell*);
74
75template<AuditLevel = AuditLevel::Random, typename T>
76ALWAYS_INLINE void auditCell(VM&, T) { }
77
78template<AuditLevel auditLevel = DefaultAuditLevel>
79ALWAYS_INLINE void auditCell(VM& vm, JSCell* cell)
80{
81 switch (auditLevel) {
82 case AuditLevel::None:
83 return;
84 case AuditLevel::Minimal:
85 return auditCellMinimally(vm, cell);
86 case AuditLevel::Full:
87 return auditCellFully(vm, cell);
88 case AuditLevel::Random:
89 return auditCellRandomly(vm, cell);
90 }
91}
92
93template<AuditLevel auditLevel = DefaultAuditLevel>
94ALWAYS_INLINE void auditCell(VM& vm, JSValue value)
95{
96 if (auditLevel == AuditLevel::None)
97 return;
98
99 if (value.isCell())
100 auditCell<auditLevel>(vm, value.asCell());
101}
