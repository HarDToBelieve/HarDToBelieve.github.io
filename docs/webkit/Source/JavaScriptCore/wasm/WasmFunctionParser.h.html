43template<typename EnclosingStack, typename NewStack>
44void splitStack(BlockSignature signature, EnclosingStack& enclosingStack, NewStack& newStack)
45{
46 newStack.reserveInitialCapacity(signature->argumentCount());
47 ASSERT(enclosingStack.size() >= signature->argumentCount());
48 unsigned offset = enclosingStack.size() - signature->argumentCount();
49 for (unsigned i = 0; i < signature->argumentCount(); ++i)
50 newStack.uncheckedAppend(enclosingStack.at(i + offset));
51 enclosingStack.shrink(offset);
52}

226template<typename Context>
227auto FunctionParser<Context>::parse() -> Result
228{
229 uint32_t localGroupsCount;
230
231 WASM_PARSER_FAIL_IF(!m_context.addArguments(m_signature), "can't add ", m_signature.argumentCount(), " arguments to Function");
232 WASM_PARSER_FAIL_IF(!parseVarUInt32(localGroupsCount), "can't get local groups count");
233
234 WASM_PARSER_FAIL_IF(!m_locals.tryReserveCapacity(m_signature.argumentCount()), "can't allocate enough memory for function's ", m_signature.argumentCount(), " arguments");
235 for (uint32_t i = 0; i < m_signature.argumentCount(); ++i)
236 m_locals.uncheckedAppend(m_signature.argument(i));
237
238 uint64_t totalNumberOfLocals = m_signature.argumentCount();
239 for (uint32_t i = 0; i < localGroupsCount; ++i) {
240 uint32_t numberOfLocals;
241 Type typeOfLocal;
242
243 WASM_PARSER_FAIL_IF(!parseVarUInt32(numberOfLocals), "can't get Function's number of locals in group ", i);
244 totalNumberOfLocals += numberOfLocals;
245 WASM_PARSER_FAIL_IF(totalNumberOfLocals > maxFunctionLocals, "Function's number of locals is too big ", totalNumberOfLocals, " maximum ", maxFunctionLocals);
246 WASM_PARSER_FAIL_IF(!parseValueType(m_info, typeOfLocal), "can't get Function local's type in group ", i);
247 WASM_PARSER_FAIL_IF(!isDefaultableType(typeOfLocal), "Function locals must have a defaultable type");
248
249 WASM_PARSER_FAIL_IF(!m_locals.tryReserveCapacity(totalNumberOfLocals), "can't allocate enough memory for function's ", totalNumberOfLocals, " locals");
250 for (uint32_t i = 0; i < numberOfLocals; ++i)
251 m_locals.uncheckedAppend(typeOfLocal);
252
253 WASM_TRY_ADD_TO_CONTEXT(addLocal(typeOfLocal, numberOfLocals));
254 }
255
256 m_context.didFinishParsingLocals();
257
258 WASM_FAIL_IF_HELPER_FAILS(parseBody());
259
260 return { };
261}

263template<typename Context>
264auto FunctionParser<Context>::parseBody() -> PartialResult
265{
266 m_controlStack.append({ { }, { }, m_context.addTopLevel(&m_signature) });
267 uint8_t op = 0;
268 while (m_controlStack.size()) {
269 m_currentOpcodeStartingOffset = m_offset;
270 WASM_PARSER_FAIL_IF(!parseUInt8(op), "can't decode opcode");
271 WASM_PARSER_FAIL_IF(!isValidOpType(op), "invalid opcode ", op);
272
273 m_currentOpcode = static_cast<OpType>(op);
274
275 if (verbose) {
276 dataLogLn("processing op (", m_unreachableBlocks, "): ", RawPointer(reinterpret_cast<void*>(op)), ", ", makeString(static_cast<OpType>(op)), " at offset: ", RawPointer(reinterpret_cast<void*>(m_offset)));
277 m_context.dump(m_controlStack, &m_expressionStack);
278 }
279
280 if (m_unreachableBlocks)
281 WASM_FAIL_IF_HELPER_FAILS(parseUnreachableExpression());
282 else {
283 WASM_FAIL_IF_HELPER_FAILS(parseExpression());
284 }
285 }
286 WASM_FAIL_IF_HELPER_FAILS(m_context.endTopLevel(&m_signature, m_expressionStack));
287
288 ASSERT(op == OpType::End);
289 return { };
290}

292template<typename Context>
293template<OpType op>
294auto FunctionParser<Context>::binaryCase(Type returnType, Type lhsType, Type rhsType) -> PartialResult
295{
296 TypedExpression right;
297 TypedExpression left;
298
299 WASM_TRY_POP_EXPRESSION_STACK_INTO(right, "binary right");
300 WASM_TRY_POP_EXPRESSION_STACK_INTO(left, "binary left");
301
302 WASM_VALIDATOR_FAIL_IF(left.type() != lhsType, op, " left value type mismatch");
303 WASM_VALIDATOR_FAIL_IF(right.type() != rhsType, op, " right value type mismatch");
304
305 ExpressionType result;
306 WASM_TRY_ADD_TO_CONTEXT(template addOp<op>(left, right, result));
307 m_expressionStack.constructAndAppend(returnType, result);
308 return { };
309}

328{
329 WASM_VALIDATOR_FAIL_IF(!m_info.memory, "load instruction without memory");
330
331 uint32_t alignment;
332 uint32_t offset;
333 TypedExpression pointer;
334 WASM_PARSER_FAIL_IF(!parseVarUInt32(alignment), "can't get load alignment");
335 WASM_PARSER_FAIL_IF(alignment > memoryLog2Alignment(m_currentOpcode), "byte alignment ", 1ull << alignment, " exceeds load's natural alignment ", 1ull << memoryLog2Alignment(m_currentOpcode));
336 WASM_PARSER_FAIL_IF(!parseVarUInt32(offset), "can't get load offset");
337 WASM_TRY_POP_EXPRESSION_STACK_INTO(pointer, "load pointer");
338
339 WASM_VALIDATOR_FAIL_IF(!pointer.type().isI32(), m_currentOpcode, " pointer type mismatch");
340
341 ExpressionType result;
342 WASM_TRY_ADD_TO_CONTEXT(load(static_cast<LoadOpType>(m_currentOpcode), pointer, result, offset));
343 m_expressionStack.constructAndAppend(memoryType, result);
344 return { };

692template<typename Context>
693auto FunctionParser<Context>::checkBranchTarget(const ControlType& target) -> PartialResult
694{
695 if (!target.branchTargetArity())
696 return { };
697
698 WASM_VALIDATOR_FAIL_IF(m_expressionStack.size() < target.branchTargetArity(), ControlType::isTopLevel(target) ? "branch out of function" : "branch to block", " on expression stack of size ", m_expressionStack.size(), ", but block, ", target.signature()->toString() , " expects ", target.branchTargetArity(), " values");
699
700
701 unsigned offset = m_expressionStack.size() - target.branchTargetArity();
702 for (unsigned i = 0; i < target.branchTargetArity(); ++i)
703 WASM_VALIDATOR_FAIL_IF(!isSubtype(m_expressionStack[offset + i].type(), target.branchTargetType(i)), "branch's stack type is not a block's type branch target type. Stack value has type ", m_expressionStack[offset + i].type().kind, " but branch target expects a value of ", target.branchTargetType(i).kind, " at index ", i);
704
705 return { };
706}

712 for (unsigned i = 0; i < controlData.signature()->returnCount(); ++i)
713 WASM_VALIDATOR_FAIL_IF(!isSubtype(m_expressionStack[i].type(), controlData.signature()->returnType(i)), "control flow returns with unexpected type. ", m_expressionStack[i].type().kind, " is not a ", controlData.signature()->returnType(i).kind);
714
715 return { };

718template<typename Context>
719auto FunctionParser<Context>::parseExpression() -> PartialResult
720{
721 switch (m_currentOpcode) {
722#define CREATE_CASE(name, id, b3op, inc, lhsType, rhsType, returnType) case OpType::name: return binaryCase<OpType::name>(Types::returnType, Types::lhsType, Types::rhsType);
723 FOR_EACH_WASM_BINARY_OP(CREATE_CASE)
724#undef CREATE_CASE
725
726#define CREATE_CASE(name, id, b3op, inc, operandType, returnType) case OpType::name: return unaryCase<OpType::name>(Types::returnType, Types::operandType);
727 FOR_EACH_WASM_UNARY_OP(CREATE_CASE)
728#undef CREATE_CASE
729
730 case Select: {
731 TypedExpression condition;
732 TypedExpression zero;
733 TypedExpression nonZero;
734
735 WASM_TRY_POP_EXPRESSION_STACK_INTO(condition, "select condition");
736 WASM_TRY_POP_EXPRESSION_STACK_INTO(zero, "select zero");
737 WASM_TRY_POP_EXPRESSION_STACK_INTO(nonZero, "select non-zero");
738
739 if (Options::useWebAssemblyReferences())
740 WASM_PARSER_FAIL_IF(isRefType(nonZero.type()) || isRefType(nonZero.type()), "can't use ref-types with unannotated select");
741
742 WASM_VALIDATOR_FAIL_IF(!condition.type().isI32(), "select condition must be i32, got ", condition.type().kind);
743 WASM_VALIDATOR_FAIL_IF(nonZero.type() != zero.type(), "select result types must match, got ", nonZero.type().kind, " and ", zero.type().kind);
744
745 ExpressionType result;
746 WASM_TRY_ADD_TO_CONTEXT(addSelect(condition, nonZero, zero, result));
747
748 m_expressionStack.constructAndAppend(zero.type(), result);
749 return { };
750 }
751
752 case AnnotatedSelect: {
753 WASM_PARSER_FAIL_IF(!Options::useWebAssemblyReferences(), "references are not enabled");
754
755 AnnotatedSelectImmediates immediates;
756 WASM_FAIL_IF_HELPER_FAILS(parseAnnotatedSelectImmediates(immediates));
757
758 TypedExpression condition;
759 TypedExpression zero;
760 TypedExpression nonZero;
761
762 WASM_TRY_POP_EXPRESSION_STACK_INTO(condition, "select condition");
763 WASM_TRY_POP_EXPRESSION_STACK_INTO(zero, "select zero");
764 WASM_TRY_POP_EXPRESSION_STACK_INTO(nonZero, "select non-zero");
765
766 WASM_VALIDATOR_FAIL_IF(!condition.type().isI32(), "select condition must be i32, got ", condition.type().kind);
767 WASM_VALIDATOR_FAIL_IF(nonZero.type() != immediates.targetType, "select result types must match, got ", nonZero.type().kind, " and ", immediates.targetType.kind);
768 WASM_VALIDATOR_FAIL_IF(zero.type() != immediates.targetType, "select result types must match, got ", zero.type().kind, " and ", immediates.targetType.kind);
769
770 ExpressionType result;
771 WASM_TRY_ADD_TO_CONTEXT(addSelect(condition, nonZero, zero, result));
772
773 m_expressionStack.constructAndAppend(immediates.targetType, result);
774 return { };
775 }
776
777#define CREATE_CASE(name, id, b3op, inc, memoryType) case OpType::name: return load(Types::memoryType);
778FOR_EACH_WASM_MEMORY_LOAD_OP(CREATE_CASE)
779#undef CREATE_CASE
780
781#define CREATE_CASE(name, id, b3op, inc, memoryType) case OpType::name: return store(Types::memoryType);

783#undef CREATE_CASE

785 case F32Const: {

789 return { };
790 }

792 case I32Const: {

1405
1406 WASM_TRY_POP_EXPRESSION_STACK_INTO(condition, "br_table condition");
1407 WASM_VALIDATOR_FAIL_IF(!condition.type().isI32(), "br_table with non-i32 condition ", condition.type().kind);
1408
1409 for (unsigned i = 0; i < targets.size(); ++i) {
1410 ControlType* target = targets[i];
1411 WASM_VALIDATOR_FAIL_IF(defaultTarget.branchTargetArity() != target->branchTargetArity(), "br_table target type size mismatch. Default has size: ", defaultTarget.branchTargetArity(), "but target: ", i, " has size: ", target->branchTargetArity());
1412 for (unsigned type = 0; type < defaultTarget.branchTargetArity(); ++type)
1413 WASM_VALIDATOR_FAIL_IF(defaultTarget.branchTargetType(type) != target->branchTargetType(type), "br_table target type mismatch at offset ", type, " expected: ", defaultTarget.branchTargetType(type).kind, " but saw: ", target->branchTargetType(type).kind, " when targeting block: ", target->signature()->toString());
1414 }
1415
1416 WASM_FAIL_IF_HELPER_FAILS(checkBranchTarget(defaultTarget));
1417 WASM_TRY_ADD_TO_CONTEXT(addSwitch(condition, targets, defaultTarget, m_expressionStack));
1418
1419 m_unreachableBlocks = 1;
1420 return { };
1421 }
1422
1423 case Return: {
1424 WASM_FAIL_IF_HELPER_FAILS(checkBranchTarget(m_controlStack[0].controlData));
1425 WASM_TRY_ADD_TO_CONTEXT(addReturn(m_controlStack[0].controlData, m_expressionStack));
1426 m_unreachableBlocks = 1;
1427 return { };
1428 }
1429
1430 case End: {
1431 ControlEntry data = m_controlStack.takeLast();
1432 if (ControlType::isIf(data.controlData)) {
1433 WASM_FAIL_IF_HELPER_FAILS(unify(data.controlData));
1434 WASM_TRY_ADD_TO_CONTEXT(addElse(data.controlData, m_expressionStack));
1435 m_expressionStack = WTFMove(data.elseBlockStack);
1436 }
1437 // FIXME: This is a little weird in that it will modify the expressionStack for the result of the block.
1438 // That's a little too effectful for me but I don't have a better API right now.
1439 // see: https://bugs.webkit.org/show_bug.cgi?id=164353
1440 WASM_FAIL_IF_HELPER_FAILS(unify(data.controlData));
1441 WASM_TRY_ADD_TO_CONTEXT(endBlock(data, m_expressionStack));
1442 m_expressionStack.swap(data.enclosedExpressionStack);
1443 return { };
1444 }
1445
1446 case Unreachable: {
1447 WASM_TRY_ADD_TO_CONTEXT(addUnreachable());
1448 m_unreachableBlocks = 1;
1449 return { };
1450 }
1451
1452 case Drop: {
1453 WASM_PARSER_FAIL_IF(!m_expressionStack.size(), "can't drop on empty stack");
1454 m_expressionStack.takeLast();
1455 m_context.didPopValueFromStack();
1456 return { };
1457 }
1458
1459 case Nop: {
1460 return { };
1461 }
1462
1463 case GrowMemory: {
1464 WASM_PARSER_FAIL_IF(!m_info.memory, "grow_memory is only valid if a memory is defined or imported");
1465
1466 uint8_t reserved;
1467 WASM_PARSER_FAIL_IF(!parseUInt8(reserved), "can't parse reserved byte for grow_memory");
1468 WASM_PARSER_FAIL_IF(reserved != 0, "reserved byte for grow_memory must be zero");
1469
1470 TypedExpression delta;
1471 WASM_TRY_POP_EXPRESSION_STACK_INTO(delta, "expect an i32 argument to grow_memory on the stack");
1472 WASM_VALIDATOR_FAIL_IF(!delta.type().isI32(), "grow_memory with non-i32 delta argument has type: ", delta.type().kind);
1473
1474 ExpressionType result;
1475 WASM_TRY_ADD_TO_CONTEXT(addGrowMemory(delta, result));
1476 m_expressionStack.constructAndAppend(Types::I32, result);
1477
1478 return { };
1479 }
1480
1481 case CurrentMemory: {
1482 WASM_PARSER_FAIL_IF(!m_info.memory, "current_memory is only valid if a memory is defined or imported");
1483
1484 uint8_t reserved;
1485 WASM_PARSER_FAIL_IF(!parseUInt8(reserved), "can't parse reserved byte for current_memory");
1486 WASM_PARSER_FAIL_IF(reserved != 0, "reserved byte for current_memory must be zero");
1487
1488 ExpressionType result;
1489 WASM_TRY_ADD_TO_CONTEXT(addCurrentMemory(result));
1490 m_expressionStack.constructAndAppend(Types::I32, result);
1491
1492 return { };
1493 }
1494 }
1495
1496 ASSERT_NOT_REACHED();
1497 return { };

1500// FIXME: We should try to use the same decoder function for both unreachable and reachable code. https://bugs.webkit.org/show_bug.cgi?id=165965
1501template<typename Context>
1502auto FunctionParser<Context>::parseUnreachableExpression() -> PartialResult
1503{
1504 ASSERT(m_unreachableBlocks);
1505#define CREATE_CASE(name, ...) case OpType::name:
1506 switch (m_currentOpcode) {
1507 case Else: {
1508 if (m_unreachableBlocks > 1)
1509 return { };
