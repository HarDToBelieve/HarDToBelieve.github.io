37namespace JSC { namespace Wasm {
38
39class Instance;
40
41class Global final : public ThreadSafeRefCounted<Global> {
42 WTF_MAKE_NONCOPYABLE(Global);
43 WTF_MAKE_FAST_ALLOCATED(Global);
44public:
45 union Value {
46 uint64_t m_primitive;
47 WriteBarrierBase<Unknown> m_externref;
48 Value* m_pointer;
49 };
50
51 static Ref<Global> create(Wasm::Type type, Wasm::GlobalInformation::Mutability mutability, uint64_t initialValue = 0)
52 {
53 return adoptRef(*new Global(type, mutability, initialValue));
54 }
55
56 Wasm::Type type() const { return m_type; }
57 Wasm::GlobalInformation::Mutability mutability() const { return m_mutability; }
58 JSValue get(JSGlobalObject*) const;
59 uint64_t getPrimitive() const { return m_value.m_primitive; }
60 void set(JSGlobalObject*, JSValue);
61 DECLARE_VISIT_AGGREGATE;
62
63 template<typename T> T* owner() const { return reinterpret_cast<T*>(m_owner); }
64 void setOwner(JSObject* owner)
65 {
66 ASSERT(!m_owner);
67 ASSERT(owner);
68 m_owner = owner;
69 }
70
71 static ptrdiff_t offsetOfValue() { return OBJECT_OFFSETOF(Global, m_value); }
72 static ptrdiff_t offsetOfOwner() { return OBJECT_OFFSETOF(Global, m_owner); }
73
74 static Global& fromBinding(Value& value)
75 {
76 return *bitwise_cast<Global*>(bitwise_cast<uint8_t*>(&value) - offsetOfValue());
77 }
78
79 Value* valuePointer() { return &m_value; }
80
81private:
82 Global(Wasm::Type type, Wasm::GlobalInformation::Mutability mutability, uint64_t initialValue)
83 : m_type(type)
84 , m_mutability(mutability)
85 , m_value()
86 {
87 m_value.m_primitive = initialValue;
88 }
89
90 Wasm::Type m_type;
91 Wasm::GlobalInformation::Mutability m_mutability;
92 JSObject* m_owner { nullptr };
93 Value m_value;
94};
95
96} } // namespace JSC::Wasm
97
98#endif // ENABLE(WEBASSEMBLY)
