41struct PinnedSizeRegisterInfo {
42 GPRReg boundsCheckingSizeRegister;
43 unsigned sizeOffset;
44};
45
46class PinnedRegisterInfo {
47public:
48 PinnedRegisterInfo(GPRReg, GPRReg, GPRReg);
49
50 static const PinnedRegisterInfo& get();
51
52 RegisterSet toSave(MemoryMode mode) const
53 {
54 RegisterSet result;
55 result.set(baseMemoryPointer);
56 if (wasmContextInstancePointer != InvalidGPRReg)
57 result.set(wasmContextInstancePointer);
58 if (mode != MemoryMode::Signaling)
59 result.set(boundsCheckingSizeRegister);
60 return result;
61 }
62
63 GPRReg boundsCheckingSizeRegister;
64 GPRReg baseMemoryPointer;
65 GPRReg wasmContextInstancePointer;
66};
67
68class MemoryInformation {
69public:
70 MemoryInformation()
71 {
72 ASSERT(!*this);
73 }
74
75 MemoryInformation(PageCount initial, PageCount maximum, bool isShared, bool isImport);
76
77 PageCount initial() const { return m_initial; }
78 PageCount maximum() const { return m_maximum; }
79 bool isShared() const { return m_isShared; }
80 bool isImport() const { return m_isImport; }
81
82 explicit operator bool() const { return !!m_initial; }
83
84private:
85 PageCount m_initial { };
86 PageCount m_maximum { };
87 bool m_isShared { false };
88 bool m_isImport { false };
89};
