39namespace JSC { namespace Wasm {
40
41class LLIntPlan;
42struct Context;
43struct ModuleInformation;
44
45class Module : public ThreadSafeRefCounted<Module> {
46public:
47 using ValidationResult = Expected<RefPtr<Module>, String>;
48 typedef void CallbackType(ValidationResult&&);
49 using AsyncValidationCallback = RefPtr<SharedTask<CallbackType>>;
50
51 static ValidationResult validateSync(Context*, Vector<uint8_t>&& source);
52 static void validateAsync(Context*, Vector<uint8_t>&& source, Module::AsyncValidationCallback&&);
53
54 static Ref<Module> create(LLIntPlan& plan)
55 {
56 return adoptRef(*new Module(plan));
57 }
58
59 Wasm::SignatureIndex signatureIndexFromFunctionIndexSpace(unsigned functionIndexSpace) const;
60 const Wasm::ModuleInformation& moduleInformation() const { return m_moduleInformation.get(); }
61
62 Ref<CodeBlock> compileSync(Context*, MemoryMode);
63 void compileAsync(Context*, MemoryMode, CodeBlock::AsyncCompilationCallback&&);
64
65 JS_EXPORT_PRIVATE ~Module();
66
67 CodeBlock* codeBlockFor(MemoryMode mode) { return m_codeBlocks[static_cast<uint8_t>(mode)].get(); }
68private:
69 Ref<CodeBlock> getOrCreateCodeBlock(Context*, MemoryMode);
70
71 Module(LLIntPlan&);
72 Ref<ModuleInformation> m_moduleInformation;
73 RefPtr<CodeBlock> m_codeBlocks[Wasm::NumberOfMemoryModes];
74 RefPtr<LLIntCallees> m_llintCallees;
75 MacroAssemblerCodeRef<JITCompilationPtrTag> m_llintEntryThunks;
76 Lock m_lock;
77};
78
79} } // namespace JSC::Wasm
