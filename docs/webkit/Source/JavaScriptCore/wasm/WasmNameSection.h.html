26#pragma once
27
28#include "WasmName.h"
29#include <wtf/Noncopyable.h>
30#include <wtf/text/CString.h>
31#include <wtf/ThreadSafeRefCounted.h>
32#include <wtf/Vector.h>
33#include <utility>
34
35namespace JSC { namespace Wasm {
36
37struct NameSection : public ThreadSafeRefCounted<NameSection> {
38 WTF_MAKE_NONCOPYABLE(NameSection);
39public:
40 NameSection()
41 {
42 setHash(std::nullopt);
43 }
44
45 static Ref<NameSection> create()
46 {
47 return adoptRef(*new NameSection);
48 }
49
50 void setHash(const std::optional<CString> &hash)
51 {
52 moduleHash = Name(hash ? hash->length() : 3);
53 if (hash) {
54 for (size_t i = 0; i < hash->length(); ++i)
55 moduleHash[i] = static_cast<uint8_t>(*(hash->data() + i));
56 } else {
57 moduleHash[0] = '<';
58 moduleHash[1] = '?';
59 moduleHash[2] = '>';
60 }
61 }
62
63 std::pair<const Name*, RefPtr<NameSection>> get(size_t functionIndexSpace)
64 {
65 return std::make_pair(functionIndexSpace < functionNames.size() ? &functionNames[functionIndexSpace] : nullptr, makeRefPtr(this));
66 }
67 Name moduleName;
68 Name moduleHash;
69 Vector<Name> functionNames;
70};
71
72} } // namespace JSC::Wasm
