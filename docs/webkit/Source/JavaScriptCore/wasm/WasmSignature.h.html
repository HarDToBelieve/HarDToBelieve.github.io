46using SignatureArgCount = uint32_t;
47
48class Signature : public ThreadSafeRefCounted<Signature> {
49 WTF_MAKE_FAST_ALLOCATED;
50
51 Signature() = delete;
52 Signature(const Signature&) = delete;
53 Signature(SignatureArgCount retCount, SignatureArgCount argCount)
54 : m_retCount(retCount)
55 , m_argCount(argCount)
56 {
57 }
58
59 Type* storage(SignatureArgCount i)
60 {
61 return i + reinterpret_cast<Type*>(reinterpret_cast<char*>(this) + sizeof(Signature));
62 }
63 Type* storage(SignatureArgCount i) const { return const_cast<Signature*>(this)->storage(i); }
64 static size_t allocatedSize(Checked<SignatureArgCount> retCount, Checked<SignatureArgCount> argCount)
65 {
66 return sizeof(Signature) + (retCount + argCount) * sizeof(Type);
67 }
68
69public:
70 SignatureArgCount returnCount() const { return m_retCount; }
71 SignatureArgCount argumentCount() const { return m_argCount; }
72
73 Type returnType(SignatureArgCount i) const { ASSERT(i < returnCount()); return const_cast<Signature*>(this)->getReturnType(i); }
74 bool returnsVoid() const { return !returnCount(); }
75 Type argument(SignatureArgCount i) const { return const_cast<Signature*>(this)->getArgument(i); }
76 SignatureIndex index() const { return bitwise_cast<SignatureIndex>(this); }
77
78 WTF::String toString() const;
79 void dump(WTF::PrintStream& out) const;
80 bool operator==(const Signature& rhs) const { return this == &rhs; }
81 unsigned hash() const;
82
83 // Signatures are uniqued and, for call_indirect, validated at runtime. Tables can create invalid SignatureIndex values which cause call_indirect to fail. We use 0 as the invalidIndex so that the codegen can easily test for it and trap, and we add a token invalid entry in SignatureInformation.
84 static const constexpr SignatureIndex invalidIndex = 0;
85
86private:
87 friend class SignatureInformation;
88 friend struct ParameterTypes;
89
90 static RefPtr<Signature> tryCreate(SignatureArgCount returnCount, SignatureArgCount argumentCount);
91 Type& getReturnType(SignatureArgCount i) { ASSERT(i < returnCount()); return *storage(i); }
92 Type& getArgument(SignatureArgCount i) { ASSERT(i < argumentCount()); return *storage(returnCount() + i); }
93
94 SignatureArgCount m_retCount;
95 SignatureArgCount m_argCount;
96 // Return Type and arguments are stored here.
97};
98
99struct SignatureHash {
100 RefPtr<Signature> key { nullptr };
101 SignatureHash() = default;
102 explicit SignatureHash(Ref<Signature>&& key)
103 : key(WTFMove(key))
104 { }
105 explicit SignatureHash(WTF::HashTableDeletedValueType)
106 : key(WTF::HashTableDeletedValue)
107 { }
108 bool operator==(const SignatureHash& rhs) const { return equal(*this, rhs); }
109 static bool equal(const SignatureHash& lhs, const SignatureHash& rhs) { return lhs.key == rhs.key; }
110 static unsigned hash(const SignatureHash& signature) { return signature.key ? signature.key->hash() : 0; }
111 static constexpr bool safeToCompareToEmptyOrDeleted = false;
112 bool isHashTableDeletedValue() const { return key.isHashTableDeletedValue(); }
113};

133// Signature information is held globally and shared by the entire process to allow all signatures to be unique. This is required when wasm calls another wasm instance, and must work when modules are shared between multiple VMs.
134class SignatureInformation {
135 WTF_MAKE_NONCOPYABLE(SignatureInformation);
136 WTF_MAKE_FAST_ALLOCATED;
137
138 SignatureInformation();
139
140public:
141 static SignatureInformation& singleton();
142
143 static RefPtr<Signature> signatureFor(const Vector<Type, 1>& returnTypes, const Vector<Type>& argumentTypes);
144 ALWAYS_INLINE const Signature* thunkFor(Type type) const { return thunkSignatures[linearizeType(type.kind)]; }
145
146 static const Signature& get(SignatureIndex);
147 static SignatureIndex get(const Signature&);
148 static void tryCleanup();
149
150private:
151 HashSet<Wasm::SignatureHash> m_signatureSet;
152 const Signature* thunkSignatures[numTypes];
153 Lock m_lock;
154
155 JS_EXPORT_PRIVATE static SignatureInformation* theOne;
156 JS_EXPORT_PRIVATE static std::once_flag signatureInformationFlag;
157};
