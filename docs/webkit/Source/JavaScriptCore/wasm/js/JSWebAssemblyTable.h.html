38namespace JSC {
39
40class JSWebAssemblyTable final : public JSNonFinalObject {
41public:
42 using Base = JSNonFinalObject;
43 static constexpr bool needsDestruction = true;
44 static void destroy(JSCell*);
45
46 template<typename CellType, SubspaceAccess mode>
47 static IsoSubspace* subspaceFor(VM& vm)
48 {
49 return vm.webAssemblyTableSpace<mode>();
50 }
51
52 static JSWebAssemblyTable* tryCreate(JSGlobalObject*, VM&, Structure*, Ref<Wasm::Table>&&);
53 static Structure* createStructure(VM&, JSGlobalObject*, JSValue);
54
55 DECLARE_INFO;
56
57 static bool isValidLength(uint32_t length) { return Wasm::Table::isValidLength(length); }
58 std::optional<uint32_t> maximum() const { return m_table->maximum(); }
59 uint32_t length() const { return m_table->length(); }
60 uint32_t allocatedLength() const { return m_table->allocatedLength(length()); }
61 bool grow(uint32_t delta, JSValue defaultValue) WARN_UNUSED_RETURN;
62 JSValue get(uint32_t);
63 void set(uint32_t, WebAssemblyFunctionBase*);
64 void set(uint32_t, JSValue);
65 void clear(uint32_t);
66 JSObject* type(JSGlobalObject*);
67
68 Wasm::Table* table() { return m_table.ptr(); }
69
70private:
71 JSWebAssemblyTable(VM&, Structure*, Ref<Wasm::Table>&&);
72 void finishCreation(VM&);
73 DECLARE_VISIT_CHILDREN;
74
75 Ref<Wasm::Table> m_table;
76};
77
78} // namespace JSC

80#endif // ENABLE(WEBASSEMBLY)
