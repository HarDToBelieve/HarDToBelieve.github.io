27
28#if ENABLE(WEBASSEMBLY)
29
30#include "ArityCheckMode.h"
31#include "JSToWasmICCallee.h"
32#include "MacroAssemblerCodeRef.h"
33#include "WasmCallee.h"
34#include "WebAssemblyFunctionBase.h"
35#include <wtf/Noncopyable.h>
36
37namespace JSC {
38
39class JSGlobalObject;
40struct ProtoCallFrame;
41class WebAssemblyInstance;
42
43class WebAssemblyFunction final : public WebAssemblyFunctionBase {
44public:
45 using Base = WebAssemblyFunctionBase;
46
47 static constexpr unsigned StructureFlags = Base::StructureFlags;
48
49 static constexpr bool needsDestruction = true;
50 static void destroy(JSCell*);
51
52 template<typename CellType, SubspaceAccess mode>
53 static IsoSubspace* subspaceFor(VM& vm)
54 {
55 return vm.webAssemblyFunctionSpace<mode>();
56 }
57
58 DECLARE_EXPORT_INFO;
59
60 JS_EXPORT_PRIVATE static WebAssemblyFunction* create(VM&, JSGlobalObject*, Structure*, unsigned, const String&, JSWebAssemblyInstance*, Wasm::Callee& jsEntrypoint, WasmToWasmImportableFunction::LoadLocation, Wasm::SignatureIndex);
61 static Structure* createStructure(VM&, JSGlobalObject*, JSValue);
62
63 MacroAssemblerCodePtr<WasmEntryPtrTag> jsEntrypoint(ArityCheckMode arity)
64 {
65 if (arity == ArityCheckNotRequired)
66 return m_jsEntrypoint;
67 ASSERT(arity == MustCheckArity);
68 return m_jsEntrypoint;
69 }
70
71 MacroAssemblerCodePtr<JSEntryPtrTag> jsCallEntrypoint()
72 {
73 if (m_jsCallEntrypoint)
74 return m_jsCallEntrypoint.code();
75 return jsCallEntrypointSlow();
76 }
77
78 RegisterAtOffsetList usedCalleeSaveRegisters() const;
79 Wasm::Instance* previousInstance(CallFrame*);
80
81private:
82 DECLARE_VISIT_CHILDREN;
83 WebAssemblyFunction(VM&, NativeExecutable*, JSGlobalObject*, Structure*, Wasm::Callee& jsEntrypoint, WasmToWasmImportableFunction::LoadLocation entrypointLoadLocation, Wasm::SignatureIndex);
84
85 MacroAssemblerCodePtr<JSEntryPtrTag> jsCallEntrypointSlow();
86 ptrdiff_t previousInstanceOffset() const;
87 bool usesTagRegisters() const;
88
89 RegisterSet calleeSaves() const;
90
91 // It's safe to just hold the raw jsEntrypoint because we have a reference
92 // to our Instance, which points to the Module that exported us, which
93 // ensures that the actual Signature/code doesn't get deallocated.
94 MacroAssemblerCodePtr<WasmEntryPtrTag> m_jsEntrypoint;
95 WriteBarrier<JSToWasmICCallee> m_jsToWasmICCallee;
96 // Used for JS calling into Wasm.
97 MacroAssemblerCodeRef<JSEntryPtrTag> m_jsCallEntrypoint;
98};
99
100} // namespace JSC
101
102#endif // ENABLE(WEBASSEMBLY)
