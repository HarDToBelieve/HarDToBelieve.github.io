30// The behavior of many of the functions in the <ctype.h> header is dependent
31// on the current locale. But in the WebKit project, all uses of those functions
32// are in code processing something that's not locale-specific. These equivalents
33// for some of the <ctype.h> functions are named more explicitly, not dependent
34// on the C library locale, and we should also optimize them as needed.
35
36// All functions return false or leave the character unchanged if passed a character
37// that is outside the range 0-7F. So they can be used on Unicode strings or
38// characters if the intent is to do processing only if the character is ASCII.
39

68// The toASCIILowerUnchecked function can be used for comparing any input character
69// to a lowercase English character. The isASCIIAlphaCaselessEqual function should
70// be used for regular comparison of ASCII alpha characters, but switch statements
71// in the CSS tokenizer, for example, instead make direct use toASCIILowerUnchecked.
72template<typename CharacterType> constexpr CharacterType toASCIILowerUnchecked(CharacterType);

74extern WTF_EXPORT_PRIVATE const unsigned char asciiCaseFoldTable[256];

76template<typename CharacterType> constexpr bool isASCII(CharacterType character)
77{
78 return !(character & ~0x7F);
79}
80
81template<typename CharacterType> constexpr bool isASCIILower(CharacterType character)
82{
83 return character >= 'a' && character <= 'z';
84}
85
86template<typename CharacterType> constexpr CharacterType toASCIILowerUnchecked(CharacterType character)
87{
88 // This function can be used for comparing any input character
89 // to a lowercase English character. The isASCIIAlphaCaselessEqual
90 // below should be used for regular comparison of ASCII alpha
91 // characters, but switch statements in CSS tokenizer instead make
92 // direct use of this function.
93 return character | 0x20;
94}
95
96template<typename CharacterType> constexpr bool isASCIIAlpha(CharacterType character)
97{
98 return isASCIILower(toASCIILowerUnchecked(character));
99}
100
101template<typename CharacterType> constexpr bool isASCIIDigit(CharacterType character)
102{
103 return character >= '0' && character <= '9';
104}
105
106template<typename CharacterType> constexpr bool isASCIIAlphanumeric(CharacterType character)
107{
108 return isASCIIDigit(character) || isASCIIAlpha(character);
109}
110
111template<typename CharacterType> constexpr bool isASCIIHexDigit(CharacterType character)
112{
113 return isASCIIDigit(character) || (toASCIILowerUnchecked(character) >= 'a' && toASCIILowerUnchecked(character) <= 'f');
114}
115
116template<typename CharacterType> constexpr bool isASCIIBinaryDigit(CharacterType character)
117{
118 return character == '0' || character == '1';
119}
120
121template<typename CharacterType> constexpr bool isASCIIOctalDigit(CharacterType character)
122{
123 return character >= '0' && character <= '7';
124}
125
126template<typename CharacterType> constexpr bool isASCIIPrintable(CharacterType character)
127{
128 return character >= ' ' && character <= '~';
129}

131/*
132 Statistics from a run of Apple's page load test for callers of isASCIISpace:
133
134 character count
135 --------- -----
136 non-spaces 689383
137 20 space 294720
138 0A \n 89059
139 09 \t 28320
140 0D \r 0
141 0C \f 0
142 0B \v 0
143
144 Because of those, we first check to quickly return false for non-control characters,
145 then check for space itself to quickly return true for that case, then do the rest.
146*/
147template<typename CharacterType> constexpr bool isASCIISpace(CharacterType character)
148{
149 return character <= ' ' && (character == ' ' || (character <= 0xD && character >= 0x9));
150}

152template<typename CharacterType> constexpr bool isASCIIUpper(CharacterType character)
153{
154 return character >= 'A' && character <= 'Z';
155}
156
157template<typename CharacterType> inline CharacterType toASCIILower(CharacterType character)
158{
159 return character | (isASCIIUpper(character) << 5);
160}
161
162template<> inline char toASCIILower(char character)
163{
164 return static_cast<char>(asciiCaseFoldTable[static_cast<uint8_t>(character)]);
165}
166
167template<> inline LChar toASCIILower(LChar character)
168{
169 return asciiCaseFoldTable[character];
170}
171
172template<typename CharacterType> inline CharacterType toASCIIUpper(CharacterType character)
173{
174 return character & ~(isASCIILower(character) << 5);
175}
176
177template<typename CharacterType> inline uint8_t toASCIIHexValue(CharacterType character)
178{
179 ASSERT(isASCIIHexDigit(character));
180 return character < 'A' ? character - '0' : (character - 'A' + 10) & 0xF;
181}
182
183template<typename CharacterType> inline uint8_t toASCIIHexValue(CharacterType firstCharacter, CharacterType secondCharacter)
184{
185 return toASCIIHexValue(firstCharacter) << 4 | toASCIIHexValue(secondCharacter);
186}

188constexpr char lowerNibbleToASCIIHexDigit(uint8_t value)
189{
190 uint8_t nibble = value & 0xF;
191 return nibble + (nibble < 10 ? '0' : 'A' - 10);
192}
193
194constexpr char upperNibbleToASCIIHexDigit(uint8_t value)
195{
196 uint8_t nibble = value >> 4;
197 return nibble + (nibble < 10 ? '0' : 'A' - 10);
198}
199
200constexpr char lowerNibbleToLowercaseASCIIHexDigit(uint8_t value)
201{
202 uint8_t nibble = value & 0xF;
203 return nibble + (nibble < 10 ? '0' : 'a' - 10);
204}
205
206constexpr char upperNibbleToLowercaseASCIIHexDigit(uint8_t value)
207{
208 uint8_t nibble = value >> 4;
209 return nibble + (nibble < 10 ? '0' : 'a' - 10);
210}

212template<typename CharacterType> constexpr bool isASCIIAlphaCaselessEqual(CharacterType inputCharacter, char expectedASCIILowercaseLetter)
213{
214 // Name of this argument says this must be a lowercase letter, but it can actually be:
215 // - a lowercase letter
216 // - a numeric digit
217 // - a space
218 // - punctuation in the range 0x21-0x3F, including "-", "/", and "+"
219 // It cannot be:
220 // - an uppercase letter
221 // - a non-ASCII character
222 // - other punctuation, such as underscore and backslash
223 // - a control character such as "\n"
224 // FIXME: Would be nice to make both the function name and expectedASCIILowercaseLetter argument name clearer.
225 ASSERT(toASCIILowerUnchecked(expectedASCIILowercaseLetter) == expectedASCIILowercaseLetter);
226 return LIKELY(toASCIILowerUnchecked(inputCharacter) == expectedASCIILowercaseLetter);
227}

229template<typename CharacterType> constexpr bool isASCIIDigitOrPunctuation(CharacterType character)
230{
231 return (character >= '!' && character <= '@') || (character >= '[' && character <= '`') || (character >= '{' && character <= '~');
232}
