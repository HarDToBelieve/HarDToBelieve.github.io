31namespace WTF {

33CrossThreadTaskHandler::CrossThreadTaskHandler(const char* threadName, AutodrainedPoolForRunLoop useAutodrainedPool)
34 : m_useAutodrainedPool(useAutodrainedPool)
35{
36 ASSERT(isMainThread());
37 Locker<Lock> locker(m_taskThreadCreationLock);
38 Thread::create(threadName, [this] {
39 taskRunLoop();
40
41 if (m_completionCallback)
42 m_completionCallback();
43 })->detach();
44}
45
46CrossThreadTaskHandler::~CrossThreadTaskHandler()
47{
48 ASSERT(isMainThread());
49}
50
51void CrossThreadTaskHandler::postTask(CrossThreadTask&& task)
52{
53 m_taskQueue.append(WTFMove(task));
54}
55
56void CrossThreadTaskHandler::postTaskReply(CrossThreadTask&& task)
57{
58 m_taskReplyQueue.append(WTFMove(task));
59
60 Locker locker { m_mainThreadReplyLock };
61 if (m_mainThreadReplyScheduled)
62 return;
63
64 m_mainThreadReplyScheduled = true;
65 callOnMainThread([this] {
66 handleTaskRepliesOnMainThread();
67 });
68}
69
70void CrossThreadTaskHandler::taskRunLoop()
71{
72 ASSERT(!isMainThread());
73 {
74 Locker<Lock> locker(m_taskThreadCreationLock);
75 }
76
77 while (auto task = m_taskQueue.waitForMessage()) {
78 std::unique_ptr<AutodrainedPool> autodrainedPool = (m_useAutodrainedPool == AutodrainedPoolForRunLoop::Use) ? makeUnique<AutodrainedPool>() : nullptr;
79
80 task.performTask();
81 }
82}
83
84void CrossThreadTaskHandler::handleTaskRepliesOnMainThread()
85{
86 {
87 Locker locker { m_mainThreadReplyLock };
88 m_mainThreadReplyScheduled = false;
89 }
90
91 while (auto task = m_taskReplyQueue.tryGetMessage())
92 task->performTask();
93}
94
95void CrossThreadTaskHandler::setCompletionCallback(Function<void ()>&& completionCallback)
96{
97 m_completionCallback = WTFMove(completionCallback);
98}
99
100void CrossThreadTaskHandler::kill()
101{
102 m_taskQueue.kill();
103 m_taskReplyQueue.kill();
104}
