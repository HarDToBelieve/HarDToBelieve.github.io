40static Lock cachedPlatformPreferredLanguagesLock;
41static Vector<String>& cachedFullPlatformPreferredLanguages() WTF_REQUIRES_LOCK(cachedPlatformPreferredLanguagesLock)
42{
43 static NeverDestroyed<Vector<String>> languages;
44 return languages;
45}
46static Vector<String>& cachedMinimizedPlatformPreferredLanguages() WTF_REQUIRES_LOCK(cachedPlatformPreferredLanguagesLock)
47{
48 static NeverDestroyed<Vector<String>> languages;
49 return languages;
50}
51
52static Lock preferredLanguagesOverrideLock;
53static Vector<String>& preferredLanguagesOverride() WTF_REQUIRES_LOCK(preferredLanguagesOverrideLock)
54{
55 static NeverDestroyed<Vector<String>> override;
56 return override;
57}
58
59typedef HashMap<void*, LanguageChangeObserverFunction> ObserverMap;
60static ObserverMap& observerMap()
61{
62 static LazyNeverDestroyed<ObserverMap> map;
63 static std::once_flag onceKey;
64 std::call_once(onceKey, [&] {
65 map.construct();
66 });
67 return map.get();
68}
69
70void addLanguageChangeObserver(void* context, LanguageChangeObserverFunction customObserver)
71{
72 observerMap().set(context, customObserver);
73}
74
75void removeLanguageChangeObserver(void* context)
76{
77 ASSERT(observerMap().contains(context));
78 observerMap().remove(context);
79}
80
81void languageDidChange()
82{
83 {
84 Locker locker { cachedPlatformPreferredLanguagesLock };
85 cachedFullPlatformPreferredLanguages().clear();
86 cachedMinimizedPlatformPreferredLanguages().clear();
87 }
88
89 for (auto& observer : copyToVector(observerMap())) {
90 if (observerMap().contains(observer.key))
91 observer.value(observer.key);
92 }
93}

95String defaultLanguage(ShouldMinimizeLanguages shouldMinimizeLanguages)
96{
97 auto languages = userPreferredLanguages(shouldMinimizeLanguages);
98 if (languages.size())
99 return languages[0];
100
101 return emptyString();
102}
103
104Vector<String> userPreferredLanguagesOverride()
105{
106 Locker locker { preferredLanguagesOverrideLock };
107 return preferredLanguagesOverride();
108}
109
110void overrideUserPreferredLanguages(const Vector<String>& override)
111{
112 {
113 Locker locker { preferredLanguagesOverrideLock };
114 preferredLanguagesOverride() = override;
115 }
116 languageDidChange();
117}

119static Vector<String> isolatedCopy(const Vector<String>& strings)
120{
121 Vector<String> copy;
122 copy.reserveInitialCapacity(strings.size());
123 for (auto& language : strings)
124 copy.uncheckedAppend(language.isolatedCopy());
125 return copy;
126}
127
128Vector<String> userPreferredLanguages(ShouldMinimizeLanguages shouldMinimizeLanguages)
129{
130 {
131 Locker locker { preferredLanguagesOverrideLock };
132 Vector<String>& override = preferredLanguagesOverride();
133 if (!override.isEmpty())
134 return isolatedCopy(override);
135 }
136
137 Locker locker { cachedPlatformPreferredLanguagesLock };
138 auto& languages = shouldMinimizeLanguages == ShouldMinimizeLanguages::Yes ? cachedMinimizedPlatformPreferredLanguages() : cachedFullPlatformPreferredLanguages();
139 if (languages.isEmpty())
140 languages = platformUserPreferredLanguages(shouldMinimizeLanguages);
141 return isolatedCopy(languages);
142}

146static String canonicalLanguageIdentifier(const String& languageCode)
147{
148 String lowercaseLanguageCode = languageCode.convertToASCIILowercase();
149
150 if (lowercaseLanguageCode.length() >= 3 && lowercaseLanguageCode[2] == '_')
151 lowercaseLanguageCode.replace(2, 1, "-");
152
153 return lowercaseLanguageCode;
154}

156size_t indexOfBestMatchingLanguageInList(const String& language, const Vector<String>& languageList, bool& exactMatch)
157{
158 String lowercaseLanguage = language.convertToASCIILowercase();
159 String languageWithoutLocaleMatch;
160 String languageMatchButNotLocale;
161 size_t languageWithoutLocaleMatchIndex = 0;
162 size_t languageMatchButNotLocaleMatchIndex = 0;
163 bool canMatchLanguageOnly = (lowercaseLanguage.length() == 2 || (lowercaseLanguage.length() >= 3 && lowercaseLanguage[2] == '-'));
164
165 for (size_t i = 0; i < languageList.size(); ++i) {
166 String canonicalizedLanguageFromList = canonicalLanguageIdentifier(languageList[i]);
167
168 if (lowercaseLanguage == canonicalizedLanguageFromList) {
169 exactMatch = true;
170 return i;
171 }
172
173 if (canMatchLanguageOnly && canonicalizedLanguageFromList.length() >= 2) {
174 if (lowercaseLanguage[0] == canonicalizedLanguageFromList[0] && lowercaseLanguage[1] == canonicalizedLanguageFromList[1]) {
175 if (!languageWithoutLocaleMatch.length() && canonicalizedLanguageFromList.length() == 2) {
176 languageWithoutLocaleMatch = languageList[i];
177 languageWithoutLocaleMatchIndex = i;
178 }
179 if (!languageMatchButNotLocale.length() && canonicalizedLanguageFromList.length() >= 3) {
180 languageMatchButNotLocale = languageList[i];
181 languageMatchButNotLocaleMatchIndex = i;
182 }
183 }
184 }
185 }
186
187 exactMatch = false;
188
189 // If we have both a language-only match and a languge-but-not-locale match, return the
190 // languge-only match as is considered a "better" match. For example, if the list
191 // provided has both "en-GB" and "en" and the user prefers "en-US" we will return "en".
192 if (languageWithoutLocaleMatch.length())
193 return languageWithoutLocaleMatchIndex;
194
195 if (languageMatchButNotLocale.length())
196 return languageMatchButNotLocaleMatchIndex;
197
198 return languageList.size();
199}

204{

207 return adoptCF(CFLocaleCopyDisplayNameForPropertyValue(adoptCF(CFLocaleCopyCurrent()).get(), kCFLocaleIdentifier, localeName.createCFString().get())).get();
