40static constexpr bool profileLockContention = false;
41
42void Lock::lockSlow()
43{
44 if (profileLockContention)
45 STACK_SHOT_PROFILE(4, 2, 5);
46 DefaultLockAlgorithm::lockSlow(m_byte);
47}
48
49void Lock::unlockSlow()
50{
51 // Heap allocations are forbidden on the certain threads (e.g. audio rendering thread) for performance reasons so we need to
52 // explicitly allow the following allocation(s). In some rare cases, the unlockSlow() algorith may cause allocations.
53 DisableMallocRestrictionsForCurrentThreadScope disableMallocRestrictions;
54
55 DefaultLockAlgorithm::unlockSlow(m_byte, DefaultLockAlgorithm::Unfair);
56}

63
64 DefaultLockAlgorithm::unlockSlow(m_byte, DefaultLockAlgorithm::Fair);
65}

72bool Lock::tryLockWithTimeout(Seconds timeout)
73{
74 // This function may be called from a signal handler (e.g. via visit()). Hence,
75 // it should only use APIs that are safe to call from signal handlers. This is
76 // why we use unistd.h's sleep() instead of its alternatives.
77
78 // We'll be doing sleep(1) between tries below. Hence, sleepPerRetry is 1.
79 unsigned maxRetries = (timeout < Seconds::infinity()) ? timeout.value() : std::numeric_limits<unsigned>::max();
80 unsigned tryCount = 0;
81 while (!tryLock() && tryCount++ <= maxRetries) {
82#if OS(WINDOWS)
83 Sleep(1000);
84#else
85 ::sleep(1);
86#endif
87 }
88 return isHeld();
89}
