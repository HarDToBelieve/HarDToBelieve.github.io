37String Logger::LogSiteIdentifier::toString() const
38{
39 if (className)
40 return makeString(className, "::", methodName, '(', hex(objectPtr), ") ");
41 return makeString(methodName, '(', hex(objectPtr), ") ");
42}
43
44String LogArgument<const void*>::toString(const void* argument)
45{
46 return makeString('(', hex(reinterpret_cast<uintptr_t>(argument)), ')');
47}
48
49Vector<std::reference_wrapper<Logger::Observer>>& Logger::observers()
50{
51 static LazyNeverDestroyed<Vector<std::reference_wrapper<Observer>>> observers;
52 static std::once_flag onceKey;
53 std::call_once(onceKey, [&] {
54 observers.construct();
55 });
56 return observers;
57}
