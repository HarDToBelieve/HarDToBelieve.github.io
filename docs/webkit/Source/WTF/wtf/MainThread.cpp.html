61bool isMainRunLoop()
62{
63 return RunLoop::isMain();
64}
65
66void callOnMainRunLoop(Function<void()>&& function)
67{
68 RunLoop::main().dispatch(WTFMove(function));
69}
70
71void ensureOnMainRunLoop(Function<void()>&& function)
72{
73 if (RunLoop::isMain())
74 function();
75 else
76 RunLoop::main().dispatch(WTFMove(function));
77}
78
79void callOnMainThread(Function<void()>&& function)
80{
81#if USE(WEB_THREAD)
82 if (auto* webRunLoop = RunLoop::webIfExists()) {
83 webRunLoop->dispatch(WTFMove(function));
84 return;
85 }
86#endif
87
88 RunLoop::main().dispatch(WTFMove(function));
89}
90
91void ensureOnMainThread(Function<void()>&& function)
92{
93 if (isMainThread())
94 function();
95 else
96 callOnMainThread(WTFMove(function));
97}

99bool isMainThreadOrGCThread()
100{
101 if (Thread::mayBeGCThread())
102 return true;
103
104 return isMainThread();
105}

107enum class MainStyle : bool {
108 Thread,
109 RunLoop
110};
111
112template <MainStyle mainStyle>
113static void callOnMainAndWait(Function<void()>&& function)
114{
115
116 if (mainStyle == MainStyle::Thread ? isMainThread() : isMainRunLoop()) {
117 function();
118 return;
119 }
120
121 BinarySemaphore semaphore;
122 auto functionImpl = [&semaphore, function = WTFMove(function)] {
123 function();
124 semaphore.signal();
125 };
126
127 switch (mainStyle) {
128 case MainStyle::Thread:
129 callOnMainThread(WTFMove(functionImpl));
130 break;
131 case MainStyle::RunLoop:
132 callOnMainRunLoop(WTFMove(functionImpl));
133 };
134 semaphore.wait();
135}
136
137void callOnMainRunLoopAndWait(Function<void()>&& function)
138{
139 callOnMainAndWait<MainStyle::RunLoop>(WTFMove(function));
140}
141
142void callOnMainThreadAndWait(Function<void()>&& function)
143{
144 callOnMainAndWait<MainStyle::Thread>(WTFMove(function));
145}
