41MemoryPressureHandler& MemoryPressureHandler::singleton()
42{
43 static LazyNeverDestroyed<MemoryPressureHandler> memoryPressureHandler;
44 static std::once_flag onceKey;
45 std::call_once(onceKey, [&] {
46 WTFLogChannel* channels[] = { &LogMemoryPressure };
47 WTFInitializeLogChannelStatesFromString(channels, 1, "");

49 memoryPressureHandler.construct();

66void MemoryPressureHandler::setShouldUsePeriodicMemoryMonitor(bool use)
67{
68 if (!isFastMallocEnabled()) {
69 // If we're running with FastMalloc disabled, some kind of testing or debugging is probably happening.
70 // Let's be nice and not enable the memory kill mechanism.
71 return;
72 }
73
74 if (use) {
75 m_measurementTimer = makeUnique<RunLoop::Timer<MemoryPressureHandler>>(RunLoop::main(), this, &MemoryPressureHandler::measurementTimerFired);
76 m_measurementTimer->startRepeating(30_s);
77 } else
78 m_measurementTimer = nullptr;
79}

94static size_t thresholdForMemoryKillOfInactiveProcess(unsigned tabCount)
95{
96#if CPU(X86_64) || CPU(ARM64)
97 size_t baseThreshold = 3 * GB + tabCount * GB;
98#else
99 size_t baseThreshold = tabCount > 1 ? 3 * GB : 2 * GB;
100#endif
101 return std::min(baseThreshold, static_cast<size_t>(ramSize() * 0.9));
102}

111std::optional<size_t> MemoryPressureHandler::thresholdForMemoryKill()
112{
113 switch (m_processState) {
114 case WebsamProcessState::Inactive:
115 return thresholdForMemoryKillOfInactiveProcess(m_pageCount);
116 case WebsamProcessState::Active:
117 break;
118 }
119 return std::nullopt;
120}
121
122static size_t thresholdForPolicy(MemoryUsagePolicy policy)
123{
124 const size_t baseThresholdForPolicy = std::min(3 * GB, ramSize());
125
126#if PLATFORM(IOS_FAMILY)
127 const double conservativeThresholdFraction = 0.5;
128 const double strictThresholdFraction = 0.65;
129#else
130 const double conservativeThresholdFraction = 0.33;
131 const double strictThresholdFraction = 0.5;
132#endif
133
134 switch (policy) {
135 case MemoryUsagePolicy::Unrestricted:
136 return 0;
137 case MemoryUsagePolicy::Conservative:
138 return baseThresholdForPolicy * conservativeThresholdFraction;
139 case MemoryUsagePolicy::Strict:
140 return baseThresholdForPolicy * strictThresholdFraction;
141 default:
142 ASSERT_NOT_REACHED();
143 return 0;
144 }
145}
146
147static MemoryUsagePolicy policyForFootprint(size_t footprint)
148{
149 if (footprint >= thresholdForPolicy(MemoryUsagePolicy::Strict))
150 return MemoryUsagePolicy::Strict;
151 if (footprint >= thresholdForPolicy(MemoryUsagePolicy::Conservative))
152 return MemoryUsagePolicy::Conservative;
153 return MemoryUsagePolicy::Unrestricted;
154}
155
156MemoryUsagePolicy MemoryPressureHandler::currentMemoryUsagePolicy()
157{
158 return policyForFootprint(memoryFootprint());
159}

161void MemoryPressureHandler::shrinkOrDie(size_t killThreshold)
162{
163 RELEASE_LOG(MemoryPressure, "Process is above the memory kill threshold. Trying to shrink down.");
164 releaseMemory(Critical::Yes, Synchronous::Yes);
165
166 size_t footprint = memoryFootprint();
167 RELEASE_LOG(MemoryPressure, "New memory footprint: %zu MB", footprint / MB);
168
169 if (footprint < killThreshold) {
170 RELEASE_LOG(MemoryPressure, "Shrank below memory kill threshold. Process gets to live.");
171 setMemoryUsagePolicyBasedOnFootprint(footprint);
172 return;
173 }
174
175 WTFLogAlways("Unable to shrink memory footprint of process (%zu MB) below the kill thresold (%zu MB). Killed\n", footprint / MB, killThreshold / MB);
176 RELEASE_ASSERT(m_memoryKillCallback);
177 m_memoryKillCallback();
178}

180void MemoryPressureHandler::setMemoryUsagePolicyBasedOnFootprint(size_t footprint)
181{
182 auto newPolicy = policyForFootprint(footprint);
183 if (newPolicy == m_memoryUsagePolicy)
184 return;
185
186 RELEASE_LOG(MemoryPressure, "Memory usage policy changed: %s -> %s", toString(m_memoryUsagePolicy), toString(newPolicy));
187 m_memoryUsagePolicy = newPolicy;
188 memoryPressureStatusChanged();
189}
190
191void MemoryPressureHandler::measurementTimerFired()
192{
193 size_t footprint = memoryFootprint();
194#if PLATFORM(COCOA)
195 RELEASE_LOG(MemoryPressure, "Current memory footprint: %zu MB", footprint / MB);
196#endif
197 auto killThreshold = thresholdForMemoryKill();
198 if (killThreshold && footprint >= *killThreshold) {
199 shrinkOrDie(*killThreshold);
200 return;
201 }
202
203 setMemoryUsagePolicyBasedOnFootprint(footprint);
204
205 switch (m_memoryUsagePolicy) {
206 case MemoryUsagePolicy::Unrestricted:
207 break;
208 case MemoryUsagePolicy::Conservative:
209 releaseMemory(Critical::No, Synchronous::No);
210 break;
211 case MemoryUsagePolicy::Strict:
212 releaseMemory(Critical::Yes, Synchronous::No);
213 break;
214 }
215
216 if (processState() == WebsamProcessState::Active && footprint > thresholdForMemoryKillOfInactiveProcess(m_pageCount))
217 doesExceedInactiveLimitWhileActive();
218 else
219 doesNotExceedInactiveLimitWhileActive();
220}
221
222void MemoryPressureHandler::doesExceedInactiveLimitWhileActive()
223{
224 if (m_hasInvokedDidExceedInactiveLimitWhileActiveCallback)
225 return;
226 if (m_didExceedInactiveLimitWhileActiveCallback)
227 m_didExceedInactiveLimitWhileActiveCallback();
228 m_hasInvokedDidExceedInactiveLimitWhileActiveCallback = true;
229}
230
231void MemoryPressureHandler::doesNotExceedInactiveLimitWhileActive()
232{
233 m_hasInvokedDidExceedInactiveLimitWhileActiveCallback = false;
234}

243void MemoryPressureHandler::beginSimulatedMemoryPressure()
244{
245 if (m_isSimulatingMemoryPressure)
246 return;
247 m_isSimulatingMemoryPressure = true;
248 memoryPressureStatusChanged();
249 respondToMemoryPressure(Critical::Yes, Synchronous::Yes);
250}
251
252void MemoryPressureHandler::endSimulatedMemoryPressure()
253{
254 if (!m_isSimulatingMemoryPressure)
255 return;
256 m_isSimulatingMemoryPressure = false;
257 memoryPressureStatusChanged();
258}

260void MemoryPressureHandler::releaseMemory(Critical critical, Synchronous synchronous)
261{
262 if (!m_lowMemoryHandler)
263 return;
264
265 ReliefLogger log("Total");
266 m_lowMemoryHandler(critical, synchronous);
267 platformReleaseMemory(critical);
268}

278void MemoryPressureHandler::memoryPressureStatusChanged()

284void MemoryPressureHandler::ReliefLogger::logMemoryUsageChange()
285{
286#if !RELEASE_LOG_DISABLED
287#define MEMORYPRESSURE_LOG(...) RELEASE_LOG(MemoryPressure, __VA_ARGS__)
288#else
289#define MEMORYPRESSURE_LOG(...) WTFLogAlways(__VA_ARGS__)
290#endif
291
292 auto currentMemory = platformMemoryUsage();
293 if (!currentMemory || !m_initialMemory) {
294 MEMORYPRESSURE_LOG("Memory pressure relief: %" PUBLIC_LOG_STRING ": (Unable to get dirty memory information for process)", m_logString);
295 return;
296 }
297
298 long residentDiff = currentMemory->resident - m_initialMemory->resident;
299 long physicalDiff = currentMemory->physical - m_initialMemory->physical;
300
301 MEMORYPRESSURE_LOG("Memory pressure relief: %" PUBLIC_LOG_STRING ": res = %zu/%zu/%ld, res+swap = %zu/%zu/%ld",
302 m_logString,
303 m_initialMemory->resident, currentMemory->resident, residentDiff,
304 m_initialMemory->physical, currentMemory->physical, physicalDiff);
305}
