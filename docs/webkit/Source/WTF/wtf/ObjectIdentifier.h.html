28#include <atomic>
29#include <mutex>
30#include <wtf/HashTraits.h>
31#include <wtf/NeverDestroyed.h>
32#include <wtf/text/TextStream.h>
33#include <wtf/text/WTFString.h>
34
35namespace WTF {
36
37class ObjectIdentifierBase {
38protected:
39 WTF_EXPORT_PRIVATE static uint64_t generateIdentifierInternal();
40 WTF_EXPORT_PRIVATE static uint64_t generateThreadSafeIdentifierInternal();
41};
42
43template<typename T> class ObjectIdentifier : private ObjectIdentifierBase {
44public:
45 static ObjectIdentifier generate()
46 {
47 RELEASE_ASSERT(!m_generationProtected);
48 return ObjectIdentifier { generateIdentifierInternal() };
49 }
50
51 static ObjectIdentifier generateThreadSafe()
52 {
53 RELEASE_ASSERT(!m_generationProtected);
54 return ObjectIdentifier { generateThreadSafeIdentifierInternal() };
55 }
56
57 static void enableGenerationProtection()
58 {
59 m_generationProtected = true;
60 }
61
62 ObjectIdentifier() = default;
63
64 ObjectIdentifier(HashTableDeletedValueType) : m_identifier(hashTableDeletedValue()) { }
65 bool isHashTableDeletedValue() const { return m_identifier == hashTableDeletedValue(); }
66 bool isValid() const { return isValidIdentifier(m_identifier); }
67
68 template<typename Encoder> void encode(Encoder& encoder) const
69 {
70 ASSERT(isValidIdentifier(m_identifier));
71 encoder << m_identifier;
72 }
73
74 template<typename Decoder> static std::optional<ObjectIdentifier> decode(Decoder& decoder)
75 {
76 std::optional<uint64_t> identifier;
77 decoder >> identifier;
78 if (!identifier || !isValidIdentifier(*identifier))
79 return std::nullopt;
80 return ObjectIdentifier { *identifier };
81 }
82
83 bool operator==(const ObjectIdentifier& other) const
84 {
85 return m_identifier == other.m_identifier;
86 }
87
88 bool operator!=(const ObjectIdentifier& other) const
89 {
90 return m_identifier != other.m_identifier;
91 }
92
93 uint64_t toUInt64() const { return m_identifier; }
94 explicit operator bool() const { return m_identifier; }
95
96 String loggingString() const
97 {
98 return String::number(m_identifier);
99 }
100
101 struct MarkableTraits {
102 static bool isEmptyValue(ObjectIdentifier identifier)
103 {
104 return !identifier.m_identifier;
105 }
106
107 static constexpr ObjectIdentifier emptyValue()
108 {
109 return ObjectIdentifier();
110 }
111 };
112
113private:
114 template<typename U> friend ObjectIdentifier<U> makeObjectIdentifier(uint64_t);
115 friend struct HashTraits<ObjectIdentifier>;
116 template<typename U> friend struct ObjectIdentifierHash;
117
118 static uint64_t hashTableDeletedValue() { return std::numeric_limits<uint64_t>::max(); }
119 static bool isValidIdentifier(uint64_t identifier) { return identifier && identifier != hashTableDeletedValue(); }
120
121 explicit constexpr ObjectIdentifier(uint64_t identifier)
122 : m_identifier(identifier)
123 {
124 }
125
126 uint64_t m_identifier { 0 };
127 inline static bool m_generationProtected { false };
128};
129
130template<typename T> inline ObjectIdentifier<T> makeObjectIdentifier(uint64_t identifier)
131{
132 return ObjectIdentifier<T> { identifier };
133}
134
135template<typename T> inline void add(Hasher& hasher, ObjectIdentifier<T> identifier)
136{
137 add(hasher, identifier.toUInt64());
138}
139
140template<typename T> struct ObjectIdentifierHash {
141 static unsigned hash(const ObjectIdentifier<T>& identifier) { return intHash(identifier.m_identifier); }
142 static bool equal(const ObjectIdentifier<T>& a, const ObjectIdentifier<T>& b) { return a == b; }
143 static constexpr bool safeToCompareToEmptyOrDeleted = true;
144};
145
146template<typename T> struct HashTraits<ObjectIdentifier<T>> : SimpleClassHashTraits<ObjectIdentifier<T>> { };
147
148template<typename T> struct DefaultHash<ObjectIdentifier<T>> : ObjectIdentifierHash<T> { };
149
150template<typename T>
151TextStream& operator<<(TextStream& ts, const ObjectIdentifier<T>& identifier)
152{
153 ts << identifier.toUInt64();
154 return ts;
155}
