33ParallelHelperClient::ParallelHelperClient(RefPtr<ParallelHelperPool>&& pool)
34 : m_pool(WTFMove(pool))
35{
36 Locker locker { *m_pool->m_lock };
37 RELEASE_ASSERT(!m_pool->m_isDying);
38 m_pool->m_clients.append(this);
39}
40
41ParallelHelperClient::~ParallelHelperClient()
42{
43 Locker locker { *m_pool->m_lock };
44 finishWithLock();
45
46 for (size_t i = 0; i < m_pool->m_clients.size(); ++i) {
47 if (m_pool->m_clients[i] == this) {
48 m_pool->m_clients[i] = m_pool->m_clients.last();
49 m_pool->m_clients.removeLast();
50 break;
51 }
52 }
53}
54
55void ParallelHelperClient::setTask(RefPtr<SharedTask<void ()>>&& task)
56{
57 Locker locker { *m_pool->m_lock };
58 RELEASE_ASSERT(!m_task);
59 m_task = WTFMove(task);
60 m_pool->didMakeWorkAvailable(locker);
61}
62
63void ParallelHelperClient::finish()
64{
65 Locker locker { *m_pool->m_lock };
66 finishWithLock();
67}
68
69void ParallelHelperClient::doSomeHelping()
70{
71 RefPtr<SharedTask<void ()>> task;
72 {
73 Locker locker { *m_pool->m_lock };
74 task = claimTask();
75 if (!task)
76 return;
77 }
78
79 runTask(task);
80}
81
82void ParallelHelperClient::runTaskInParallel(RefPtr<SharedTask<void ()>>&& task)
83{
84 setTask(WTFMove(task));
85 doSomeHelping();
86 finish();
87}
88
89void ParallelHelperClient::finishWithLock()
90{
91 m_task = nullptr;
92 while (m_numActive)
93 m_pool->m_workCompleteCondition.wait(*m_pool->m_lock);
94}
95
96RefPtr<SharedTask<void ()>> ParallelHelperClient::claimTask()
97{
98 if (!m_task)
99 return nullptr;
100
101 m_numActive++;
102 return m_task;
103}
104
105void ParallelHelperClient::runTask(const RefPtr<SharedTask<void ()>>& task)
106{
107 RELEASE_ASSERT(m_numActive);
108 RELEASE_ASSERT(task);
109
110 task->run();
111
112 {
113 Locker locker { *m_pool->m_lock };
114 RELEASE_ASSERT(m_numActive);
115 // No new task could have been installed, since we were still active.
116 RELEASE_ASSERT(!m_task || m_task == task);
117 m_task = nullptr;
118 m_numActive--;
119 if (!m_numActive)
120 m_pool->m_workCompleteCondition.notifyAll();
121 }
122}

124ParallelHelperPool::ParallelHelperPool(CString&& threadName)
125 : m_lock(Box<Lock>::create())
126 , m_workAvailableCondition(AutomaticThreadCondition::create())
127 , m_threadName(WTFMove(threadName))
128{
129}
130
131ParallelHelperPool::~ParallelHelperPool()
132{
133 RELEASE_ASSERT(m_clients.isEmpty());
134
135 {
136 Locker locker { *m_lock };
137 m_isDying = true;
138 m_workAvailableCondition->notifyAll(locker);
139 }
140
141 for (RefPtr<AutomaticThread>& thread : m_threads)
142 thread->join();
143}

145void ParallelHelperPool::ensureThreads(unsigned numThreads)
146{
147 Locker locker { *m_lock };
148 if (numThreads < m_numThreads)
149 return;
150 m_numThreads = numThreads;
151 if (getClientWithTask())
152 didMakeWorkAvailable(locker);
153}
154
155void ParallelHelperPool::doSomeHelping()
156{
157 ParallelHelperClient* client;
158 RefPtr<SharedTask<void ()>> task;
159 {
160 Locker locker { *m_lock };
161 client = getClientWithTask();
162 if (!client)
163 return;
164 assertIsHeld(*client->m_pool->m_lock);
165 task = client->claimTask();
166 }
167
168 client->runTask(task);
169}

171class ParallelHelperPool::Thread final : public AutomaticThread {
172public:
173 Thread(const AbstractLocker& locker, ParallelHelperPool& pool)
174 : AutomaticThread(locker, pool.m_lock, pool.m_workAvailableCondition.copyRef())
175 , m_pool(pool)
176 {
177 }
178
179 const char* name() const final
180 {
181 return m_pool.m_threadName.data();
182 }
183
184private:
185 PollResult poll(const AbstractLocker&) final
186 {
187 if (m_pool.m_isDying)
188 return PollResult::Stop;
189 assertIsHeld(*m_pool.m_lock);
190 m_client = m_pool.getClientWithTask();
191 if (m_client) {
192 assertIsHeld(*m_client->m_pool->m_lock);
193 m_task = m_client->claimTask();
194 return PollResult::Work;
195 }
196 return PollResult::Wait;
197 }
198
199 WorkResult work() final
200 {
201 m_client->runTask(m_task);
202 m_client = nullptr;
203 m_task = nullptr;
204 return WorkResult::Continue;
205 }
206
207 ParallelHelperPool& m_pool;
208 ParallelHelperClient* m_client { nullptr };
209 RefPtr<SharedTask<void ()>> m_task;
210};
211
212void ParallelHelperPool::didMakeWorkAvailable(const AbstractLocker& locker)
213{
214 while (m_numThreads > m_threads.size())
215 m_threads.append(adoptRef(new Thread(locker, *this)));
216 m_workAvailableCondition->notifyAll(locker);
217}
218
219bool ParallelHelperPool::hasClientWithTask()
220{
221 return !!getClientWithTask();
222}
223
224ParallelHelperClient* ParallelHelperPool::getClientWithTask()
225{
226 // We load-balance by being random.
227 unsigned startIndex = m_random.getUint32(m_clients.size());
228 for (unsigned index = startIndex; index < m_clients.size(); ++index) {
229 ParallelHelperClient* client = m_clients[index];
230 if (client->m_task)
231 return client;
232 }
233 for (unsigned index = 0; index < startIndex; ++index) {
234 ParallelHelperClient* client = m_clients[index];
235 if (client->m_task)
236 return client;
237 }
238
239 return nullptr;
240}
