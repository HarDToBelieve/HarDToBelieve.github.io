37void PrintStream::printf(const char* format, ...)
38{
39 va_list argList;
40 va_start(argList, format);
41 vprintf(format, argList);
42 va_end(argList);
43}

45void PrintStream::printfVariableFormat(const char* format, ...)
46{
47 ALLOW_NONLITERAL_FORMAT_BEGIN
48 IGNORE_GCC_WARNINGS_BEGIN("suggest-attribute=format")
49 va_list argList;
50 va_start(argList, format);
51 vprintf(format, argList);
52 va_end(argList);
53 IGNORE_GCC_WARNINGS_END
54 ALLOW_NONLITERAL_FORMAT_END
55}

75static void printExpectedCStringHelper(PrintStream& out, const char* type, Expected<CString, UTF8ConversionError> expectedCString)
76{
77 if (UNLIKELY(!expectedCString)) {
78 if (expectedCString.error() == UTF8ConversionError::OutOfMemory) {
79 printInternal(out, "(Out of memory while converting ");
80 printInternal(out, type);
81 printInternal(out, " to utf8)");
82 } else {
83 printInternal(out, "(failed to convert ");
84 printInternal(out, type);
85 printInternal(out, " to utf8)");
86 }
87 return;
88 }
89 printInternal(out, expectedCString.value());
90}

92void printInternal(PrintStream& out, const StringView& string)
93{
94 printExpectedCStringHelper(out, "StringView", string.tryGetUtf8());
95}

97void printInternal(PrintStream& out, const CString& string)
98{
99 printInternal(out, string.data());
100}
101
102void printInternal(PrintStream& out, const String& string)
103{
104 printExpectedCStringHelper(out, "String", string.tryGetUtf8());
105}
106
107void printInternal(PrintStream& out, const StringImpl* string)
108{
109 if (!string) {
110 printInternal(out, "(null StringImpl*)");
111 return;
112 }
113 printExpectedCStringHelper(out, "StringImpl*", string->tryGetUtf8());
114}
