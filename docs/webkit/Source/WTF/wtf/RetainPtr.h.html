63// Unlike most most of our smart pointers, RetainPtr can take either the pointer type or the pointed-to type,
64// so both RetainPtr<NSDictionary> and RetainPtr<CFDictionaryRef> will work.
65
66template<typename T> class RetainPtr;
67
68template<typename T> RetainPtr<T> adoptCF(T CF_RELEASES_ARGUMENT) WARN_UNUSED_RETURN;
69#ifdef __OBJC__
70template<typename T> RetainPtr<typename RetainPtr<T>::HelperPtrType> adoptNS(T NS_RELEASES_ARGUMENT) WARN_UNUSED_RETURN;
71#endif
72
73template<typename T> class RetainPtr {
74public:
75 typedef typename std::remove_pointer<T>::type ValueType;
76 typedef ValueType* PtrType;
77 typedef CFTypeRef StorageType;
78
79#ifdef __OBJC__
80 typedef typename std::conditional<std::is_convertible<T, id>::value && !std::is_same<T, id>::value, typename std::remove_pointer<T>::type, T>::type HelperPtrType;
81#else
82 typedef T HelperPtrType;
83#endif
84
85 RetainPtr() : m_ptr(nullptr) { }
86 RetainPtr(PtrType ptr) : m_ptr(toStorageType(ptr)) { if (m_ptr) CFRetain(m_ptr); }
87
88 RetainPtr(const RetainPtr& o) : m_ptr(o.m_ptr) { if (StorageType ptr = m_ptr) CFRetain(ptr); }
89
90 RetainPtr(RetainPtr&& o) : m_ptr(toStorageType(o.leakRef())) { }
91 template<typename U> RetainPtr(RetainPtr<U>&& o) : m_ptr(toStorageType(o.leakRef())) { }
92
93 // Hash table deleted values, which are only constructed and never copied or destroyed.
94 RetainPtr(HashTableDeletedValueType) : m_ptr(hashTableDeletedValue()) { }
95 bool isHashTableDeletedValue() const { return m_ptr == hashTableDeletedValue(); }
96
97 ~RetainPtr();
98
99 template<typename U> RetainPtr(const RetainPtr<U>&);
100
101 void clear();
102 PtrType leakRef() WARN_UNUSED_RETURN;
103 PtrType autorelease();
104#ifdef __OBJC__
105 id bridgingAutorelease();
106#endif
107
108 PtrType get() const { return fromStorageType(m_ptr); }
109 PtrType operator->() const { return fromStorageType(m_ptr); }
110 explicit operator PtrType() const { return fromStorageType(m_ptr); }
111 explicit operator bool() const { return m_ptr; }
112
113 bool operator!() const { return !m_ptr; }
114
115 // This conversion operator allows implicit conversion to bool but not to other integer types.
116 typedef StorageType RetainPtr::*UnspecifiedBoolType;
117 operator UnspecifiedBoolType() const { return m_ptr ? &RetainPtr::m_ptr : nullptr; }
118
119 RetainPtr& operator=(const RetainPtr&);
120 template<typename U> RetainPtr& operator=(const RetainPtr<U>&);
121 RetainPtr& operator=(PtrType);
122 template<typename U> RetainPtr& operator=(U*);
123
124 RetainPtr& operator=(RetainPtr&&);
125 template<typename U> RetainPtr& operator=(RetainPtr<U>&&);
126
127 void swap(RetainPtr&);
128
129 template<typename U> friend RetainPtr<U> adoptCF(U CF_RELEASES_ARGUMENT) WARN_UNUSED_RETURN;
130#ifdef __OBJC__
131 template<typename U> friend RetainPtr<typename RetainPtr<U>::HelperPtrType> adoptNS(U NS_RELEASES_ARGUMENT) WARN_UNUSED_RETURN;
132#endif
133
134private:
135 enum AdoptTag { Adopt };
136 RetainPtr(PtrType ptr, AdoptTag) : m_ptr(toStorageType(ptr)) { }
137
138 static PtrType hashTableDeletedValue() { return reinterpret_cast<PtrType>(-1); }
139
140#ifdef __OBJC__
141 template<typename U>
142 typename std::enable_if<std::is_convertible<U, id>::value, PtrType>::type
143 fromStorageTypeHelper(StorageType ptr) const
144 {
145 return (__bridge PtrType)const_cast<CF_BRIDGED_TYPE(id) void*>(ptr);
146 }
147
148 template<typename U>
149 typename std::enable_if<!std::is_convertible<U, id>::value, PtrType>::type
150 fromStorageTypeHelper(StorageType ptr) const
151 {
152 return (PtrType)const_cast<CF_BRIDGED_TYPE(id) void*>(ptr);
153 }
154
155 PtrType fromStorageType(StorageType ptr) const { return fromStorageTypeHelper<PtrType>(ptr); }
156 StorageType toStorageType(id ptr) const { return (__bridge StorageType)ptr; }
157 StorageType toStorageType(CFTypeRef ptr) const { return (StorageType)ptr; }
158#else
159 PtrType fromStorageType(StorageType ptr) const
160 {
161 return (PtrType)const_cast<CF_BRIDGED_TYPE(id) void*>(ptr);
162 }
163 StorageType toStorageType(PtrType ptr) const { return (StorageType)ptr; }
164#endif
165
166#ifdef __OBJC__
167 template<typename U> std::enable_if_t<std::is_convertible<U, id>::value, PtrType> autoreleaseHelper();
168 template<typename U> std::enable_if_t<!std::is_convertible<U, id>::value, PtrType> autoreleaseHelper();
169#endif
170
171 StorageType m_ptr;
172};

174// Helper function for creating a RetainPtr using template argument deduction.
175template<typename T> RetainPtr<typename RetainPtr<T>::HelperPtrType> retainPtr(T) WARN_UNUSED_RETURN;

177template<typename T> inline RetainPtr<T>::~RetainPtr()
178{
179 if (StorageType ptr = std::exchange(m_ptr, nullptr))
180 CFRelease(ptr);
181}
182
183template<typename T> template<typename U> inline RetainPtr<T>::RetainPtr(const RetainPtr<U>& o)
184 : m_ptr(toStorageType(o.get()))
185{
186 if (StorageType ptr = m_ptr)
187 CFRetain(ptr);
188}
189
190template<typename T> inline void RetainPtr<T>::clear()
191{
192 if (StorageType ptr = std::exchange(m_ptr, nullptr))
193 CFRelease(ptr);
194}
195
196template<typename T> inline auto RetainPtr<T>::leakRef() -> PtrType
197{
198 return fromStorageType(std::exchange(m_ptr, nullptr));
199}
200
201#ifndef __OBJC__
202
203template<typename T> inline auto RetainPtr<T>::autorelease() -> PtrType
204{
205 if (m_ptr)
206 CFAutorelease(m_ptr);
207 return leakRef();
208}
209
210#else
211
212template<typename T> template<typename U> inline auto RetainPtr<T>::autoreleaseHelper() -> std::enable_if_t<std::is_convertible<U, id>::value, PtrType>
213{
214 return CFBridgingRelease(std::exchange(m_ptr, nullptr));
215}
216
217template<typename T> template<typename U> inline auto RetainPtr<T>::autoreleaseHelper() -> std::enable_if_t<!std::is_convertible<U, id>::value, PtrType>
218{
219 if (m_ptr)
220 CFAutorelease(m_ptr);
221 return leakRef();
222}
223
224template<typename T> inline auto RetainPtr<T>::autorelease() -> PtrType
225{
226 return autoreleaseHelper<PtrType>();
227}

229// FIXME: It would be nice if we could base the return type on the type that is toll-free bridged with T rather than using id.
230template<typename T> inline id RetainPtr<T>::bridgingAutorelease()
231{
232 static_assert((!std::is_convertible<PtrType, id>::value), "Don't use bridgingAutorelease for Objective-C pointer types.");
233 return CFBridgingRelease(leakRef());
234}

320template<typename T> inline RetainPtr<T> adoptCF(T CF_RELEASES_ARGUMENT ptr)
321{
322#ifdef __OBJC__
323 static_assert((!std::is_convertible<T, id>::value), "Don't use adoptCF with Objective-C pointer types, use adoptNS.");
324#endif
325 return RetainPtr<T>(ptr, RetainPtr<T>::Adopt);
326}

328#ifdef __OBJC__
329template<typename T> inline RetainPtr<typename RetainPtr<T>::HelperPtrType> adoptNS(T NS_RELEASES_ARGUMENT ptr)
330{
331#if __has_feature(objc_arc)
332 return ptr;
333#elif defined(OBJC_NO_GC)
334 return RetainPtr<typename RetainPtr<T>::HelperPtrType>(ptr, RetainPtr<typename RetainPtr<T>::HelperPtrType>::Adopt);
335#else
336 RetainPtr<typename RetainPtr<T>::HelperPtrType> result = ptr;
337 [ptr release];
338 return result;
339#endif
340}
341#endif

343template<typename T> inline RetainPtr<typename RetainPtr<T>::HelperPtrType> retainPtr(T ptr)

345 return ptr;

357template <typename P>
358struct RetainPtrObjectHashTraits : SimpleClassHashTraits<RetainPtr<P>> {
359 static const RetainPtr<P>& emptyValue()
360 {
361 static RetainPtr<P>& null = *(new RetainPtr<P>);
362 return null;
363 }
364};

366template <typename P>
367struct RetainPtrObjectHash {
368 static unsigned hash(const RetainPtr<P>& o)
369 {
370 ASSERT_WITH_MESSAGE(o.get(), "attempt to use null RetainPtr in HashTable");
371 return static_cast<unsigned>(CFHash(o.get()));
372 }
373 static bool equal(const RetainPtr<P>& a, const RetainPtr<P>& b)
374 {
375 return CFEqual(a.get(), b.get());
376 }
377 static constexpr bool safeToCompareToEmptyOrDeleted = false;
378};

380inline bool safeCFEqual(CFTypeRef a, CFTypeRef b)
381{
382 return (!a && !b) || (a && b && CFEqual(a, b));
383}
384
385inline CFHashCode safeCFHash(CFTypeRef a)
386{
387 return a ? CFHash(a) : 0;
388}
