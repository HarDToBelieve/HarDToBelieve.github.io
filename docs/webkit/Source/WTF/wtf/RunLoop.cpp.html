39// Helper class for ThreadSpecificData.
40class RunLoop::Holder {
41 WTF_MAKE_FAST_ALLOCATED;
42public:
43 Holder()
44 : m_runLoop(adoptRef(*new RunLoop))
45 {
46 }
47
48 ~Holder()
49 {
50 m_runLoop->threadWillExit();
51 }
52
53 RunLoop& runLoop() { return m_runLoop; }
54
55private:
56 Ref<RunLoop> m_runLoop;
57};
58
59void RunLoop::initializeMain()
60{
61 RELEASE_ASSERT(!s_mainRunLoop);
62 s_mainRunLoop = &RunLoop::current();
63}
64
65auto RunLoop::runLoopHolder() -> ThreadSpecific<Holder>&
66{
67 static LazyNeverDestroyed<ThreadSpecific<Holder>> runLoopHolder;
68 static std::once_flag onceKey;
69 std::call_once(onceKey, [&] {
70 runLoopHolder.construct();
71 });
72 return runLoopHolder;
73}
74
75RunLoop& RunLoop::current()
76{
77 return runLoopHolder()->runLoop();
78}

80RunLoop& RunLoop::main()

105bool RunLoop::isMain()
106{
107 ASSERT(s_mainRunLoop);
108 // Avoid constructing the RunLoop for the current thread if it has not been created yet.
109 return runLoopHolder().isSet() && s_mainRunLoop == &RunLoop::current();
110}
111
112void RunLoop::performWork()
113{
114 bool didSuspendFunctions = false;
115
116 {
117 Locker locker { m_nextIterationLock };
118
119 // If the RunLoop re-enters or re-schedules, we're expected to execute all functions in order.
120 while (!m_currentIteration.isEmpty())
121 m_nextIteration.prepend(m_currentIteration.takeLast());
122
123 m_currentIteration = std::exchange(m_nextIteration, { });
124 }
125
126 while (!m_currentIteration.isEmpty()) {
127 if (m_isFunctionDispatchSuspended) {
128 didSuspendFunctions = true;
129 break;
130 }
131
132 auto function = m_currentIteration.takeFirst();
133 function();
134 }
135
136 // Suspend only for a single cycle.
137 m_isFunctionDispatchSuspended = false;
138 m_hasSuspendedFunctions = didSuspendFunctions;
139
140 if (m_hasSuspendedFunctions)
141 wakeUp();
142}
143
144void RunLoop::dispatch(Function<void()>&& function)
145{
146 RELEASE_ASSERT(function);
147 bool needsWakeup = false;
148
149 {
150 Locker locker { m_nextIterationLock };
151 needsWakeup = m_nextIteration.isEmpty();
152 m_nextIteration.append(WTFMove(function));
153 }
154
155 if (needsWakeup)
156 wakeUp();
157}
158
159void RunLoop::dispatchAfter(Seconds delay, Function<void()>&& function)
160{
161 RELEASE_ASSERT(function);
162 auto timer = new DispatchTimer(*this);
163 timer->setFunction([timer, function = WTFMove(function)] {
164 function();
165 delete timer;
166 });
167 timer->startOneShot(delay);
168}
169
170void RunLoop::suspendFunctionDispatchForCurrentCycle()
171{
172 // Don't suspend if there are already suspended functions to avoid unexecuted function pile-up.
173 if (m_isFunctionDispatchSuspended || m_hasSuspendedFunctions)
174 return;
175
176 m_isFunctionDispatchSuspended = true;
177 // Wake up (even if there is nothing to do) to disable suspension.
178 wakeUp();
179}
180
181void RunLoop::threadWillExit()
182{
183 m_currentIteration.clear();
184 {
185 Locker locker { m_nextIterationLock };
186 m_nextIteration.clear();
187 }
188}
