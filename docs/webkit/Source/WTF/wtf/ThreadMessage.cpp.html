32namespace WTF {
33
34MessageStatus sendMessageScoped(Thread& thread, const ThreadMessage& message)
35{
36 static Lock messageLock;
37 Locker locker { messageLock };
38
39 auto result = thread.suspend();
40 if (!result)
41 return MessageStatus::ThreadExited;
42
43 PlatformRegisters registers;
44 thread.getRegisters(registers);
45
46 message(registers);
47
48 thread.resume();
49 return MessageStatus::MessageRan;
50}
