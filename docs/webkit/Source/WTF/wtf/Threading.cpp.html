46static std::optional<size_t> stackSize(ThreadType threadType)
47{
48 // Return the stack size for the created thread based on its type.
49 // If the stack size is not specified, then use the system default. Platforms can tune the values here.
50 // Enable STACK_STATS in StackStats.h to create a build that will track the information for tuning.
51#if PLATFORM(PLAYSTATION)
52 if (threadType == ThreadType::JavaScript)
53 return 512 * KB;
54#elif OS(DARWIN) && ASAN_ENABLED
55 if (threadType == ThreadType::Compiler)
56 return 1 * MB; // ASan needs more stack space (especially on Debug builds).
57#else
58 UNUSED_PARAM(threadType);
59#endif
60
61#if defined(DEFAULT_THREAD_STACK_SIZE_IN_KB) && DEFAULT_THREAD_STACK_SIZE_IN_KB > 0
62 return DEFAULT_THREAD_STACK_SIZE_IN_KB * 1024;
63#elif OS(LINUX) && !defined(__BIONIC__) && !defined(__GLIBC__)
64 // on libcs other than glibc and bionic (e.g. musl) we are either unsure how big
65 // the default thread stack is, or we know it's too small - pick a robust default
66 return 1 * MB;
67#else
68 // Use the platform's default stack size
69 return std::nullopt;
70#endif
71}

111const char* Thread::normalizeThreadName(const char* threadName)
112{
113#if HAVE(PTHREAD_SETNAME_NP)
114 return threadName;
115#else
116 // This name can be com.apple.WebKit.ProcessLauncher or com.apple.CoreIPC.ReceiveQueue.
117 // We are using those names for the thread name, but both are longer than the limit of
118 // the platform thread name length, 32 for Windows and 16 for Linux.
119 StringView result(threadName);
120 size_t size = result.reverseFind('.');
121 if (size != notFound)
122 result = result.substring(size + 1);
123
124#if OS(WINDOWS)
125 constexpr const size_t kVisualStudioThreadNameLimit = 32 - 1;
126 if (result.length() > kVisualStudioThreadNameLimit)
127 result = result.right(kVisualStudioThreadNameLimit);
128#elif OS(LINUX)
129 constexpr const size_t kLinuxThreadNameLimit = 16 - 1;
130 if (result.length() > kLinuxThreadNameLimit)
131 result = result.right(kLinuxThreadNameLimit);
132#endif
133 ASSERT(result.characters8()[result.length()] == '\0');
134 return reinterpret_cast<const char*>(result.characters8());
135#endif
136}

138void Thread::initializeInThread()
139{
140 if (m_stack.isEmpty())
141 m_stack = StackBounds::currentThreadStackBounds();
142 m_savedLastStackTop = stack().origin();
143
144 m_currentAtomStringTable = &m_defaultAtomStringTable;
145#if USE(WEB_THREAD)
146 // On iOS, one AtomStringTable is shared between the main UI thread and the WebThread.
147 if (isWebThread() || isUIThread()) {
148 static LazyNeverDestroyed<AtomStringTable> sharedStringTable;
149 static std::once_flag onceKey;
150 std::call_once(onceKey, [&] {
151 sharedStringTable.construct();
152 });
153 m_currentAtomStringTable = &sharedStringTable.get();
154 }
155#endif
156
157#if OS(LINUX)
158 m_id = currentID();
159#endif
160}

162void Thread::entryPoint(NewThreadContext* newThreadContext)
163{
164 Function<void()> function;
165 {
166 // Ref is already incremented by Thread::create.
167 Ref<NewThreadContext> context = adoptRef(*newThreadContext);
168 // Block until our creating thread has completed any extra setup work, including establishing ThreadIdentifier.
169 MutexLocker locker(context->mutex);
170 ASSERT(context->stage == NewThreadContext::Stage::EstablishedHandle);
171
172 Thread::initializeCurrentThreadInternal(context->name);
173 function = WTFMove(context->entryPoint);
174 context->thread->initializeInThread();
175
176 Thread::initializeTLS(WTFMove(context->thread));
177
178#if !HAVE(STACK_BOUNDS_FOR_NEW_THREAD)
179 // Ack completion of initialization to the creating thread.
180 context->stage = NewThreadContext::Stage::Initialized;
181 context->condition.signal();
182#endif
183 }
184
185 ASSERT(!Thread::current().stack().isEmpty());
186 function();
187}

189Ref<Thread> Thread::create(const char* name, Function<void()>&& entryPoint, ThreadType threadType, QOS qos)
190{
191 WTF::initialize();
192 Ref<Thread> thread = adoptRef(*new Thread());
193 Ref<NewThreadContext> context = adoptRef(*new NewThreadContext { name, WTFMove(entryPoint), thread.copyRef() });
194 // Increment the context ref on behalf of the created thread. We do not just use a unique_ptr and leak it to the created thread because both the creator and created thread has a need to keep the context alive:
195 // 1. the created thread needs to keep it alive because Thread::create() can exit before the created thread has a chance to use the context.
196 // 2. the creator thread (if HAVE(STACK_BOUNDS_FOR_NEW_THREAD) is false) needs to keep it alive because the created thread may exit before the creator has a chance to wake up from waiting for the completion of the created thread's initialization. This waiting uses a condition variable in the context.
197 // Hence, a joint ownership model is needed if HAVE(STACK_BOUNDS_FOR_NEW_THREAD) is false. To simplify the code, we just go with joint ownership by both the creator and created threads,
198 // and make the context ThreadSafeRefCounted.
199 context->ref();
200 {
201 MutexLocker locker(context->mutex);
202 bool success = thread->establishHandle(context.ptr(), stackSize(threadType), qos);
203 RELEASE_ASSERT(success);
204 context->stage = NewThreadContext::Stage::EstablishedHandle;
205
206#if HAVE(STACK_BOUNDS_FOR_NEW_THREAD)
207 thread->m_stack = StackBounds::newThreadStackBounds(thread->m_handle);
208#else
209 // In platforms which do not support StackBounds::newThreadStackBounds(), we do not have a way to get stack
210 // bounds outside the target thread itself. Thus, we need to initialize thread information in the target thread
211 // and wait for completion of initialization in the caller side.
212 while (context->stage != NewThreadContext::Stage::Initialized)
213 context->condition.wait(context->mutex);
214#endif
215 }
216
217 // We must register threads here since threads registered in allThreads are expected to have complete thread data which can be initialized in launched thread side.
218 // However, it is also possible that the launched thread has finished its execution before it is registered in allThreads here! In this case, the thread has already
219 // called Thread::didExit to unregister itself from allThreads. Registering such a thread will register a stale thread pointer to allThreads, which will not be removed
220 // even after Thread is destroyed. Register a thread only when it has not unregistered itself from allThreads yet.
221 {
222 Locker locker { allThreadsLock() };
223 if (!thread->m_didUnregisterFromAllThreads)
224 allThreads().add(thread.ptr());
225 }
226
227 ASSERT(!thread->stack().isEmpty());
228 return thread;
229}

231static bool shouldRemoveThreadFromThreadGroup()
232{
233#if OS(WINDOWS)
234 // On Windows the thread specific destructor is also called when the
235 // main thread is exiting. This may lead to the main thread waiting
236 // forever for the thread group lock when exiting, if the sampling
237 // profiler thread was terminated by the system while holding the
238 // thread group lock.
239 if (WTF::isMainThread())
240 return false;
241#endif
242 return true;
243}
244
245void Thread::didExit()
246{
247 {
248 Locker locker { allThreadsLock() };
249 allThreads().remove(this);
250 m_didUnregisterFromAllThreads = true;
251 }
252
253 if (shouldRemoveThreadFromThreadGroup()) {
254 {
255 Vector<std::shared_ptr<ThreadGroup>> threadGroups;
256 {
257 Locker locker { m_mutex };
258 for (auto& threadGroupPointerPair : m_threadGroupMap) {
259 // If ThreadGroup is just being destroyed,
260 // we do not need to perform unregistering.
261 if (auto retained = threadGroupPointerPair.value.lock())
262 threadGroups.append(WTFMove(retained));
263 }
264 m_isShuttingDown = true;
265 }
266 for (auto& threadGroup : threadGroups) {
267 Locker threadGroupLocker { threadGroup->getLock() };
268 Locker locker { m_mutex };
269 threadGroup->m_threads.remove(*this);
270 }
271 }
272
273 // We would like to say "thread is exited" after unregistering threads from thread groups.
274 // So we need to separate m_isShuttingDown from m_didExit.
275 Locker locker { m_mutex };
276 m_didExit = true;
277 }

280ThreadGroupAddResult Thread::addToThreadGroup(const AbstractLocker& threadGroupLocker, ThreadGroup& threadGroup)
281{
282 UNUSED_PARAM(threadGroupLocker);
283 Locker locker { m_mutex };
284 if (m_isShuttingDown)
285 return ThreadGroupAddResult::NotAdded;
286 if (threadGroup.m_threads.add(*this).isNewEntry) {
287 m_threadGroupMap.add(&threadGroup, threadGroup.weakFromThis());
288 return ThreadGroupAddResult::NewlyAdded;
289 }
290 return ThreadGroupAddResult::AlreadyAdded;
291}
292
293void Thread::removeFromThreadGroup(const AbstractLocker& threadGroupLocker, ThreadGroup& threadGroup)
294{
295 UNUSED_PARAM(threadGroupLocker);
296 Locker locker { m_mutex };
297 if (m_isShuttingDown)
298 return;
299 m_threadGroupMap.remove(&threadGroup);
300}

308bool Thread::exchangeIsCompilationThread(bool newValue)
309{

374void initialize()
375{
376 static std::once_flag onceKey;
377 std::call_once(onceKey, [] {
378 setPermissionsOfConfigPage();
379 Config::AssertNotFrozenScope assertScope;
380 initializeRandomNumberGenerator();
381#if !HAVE(FAST_TLS) && !OS(WINDOWS)
382 Thread::initializeTLSKey();
383#endif
