36TimeWithDynamicClockType TimeWithDynamicClockType::now(ClockType type)
37{
38 switch (type) {
39 case ClockType::Wall:
40 return WallTime::now();
41 case ClockType::Monotonic:
42 return MonotonicTime::now();
43 }
44 RELEASE_ASSERT_NOT_REACHED();
45 return TimeWithDynamicClockType();
46}
47
48TimeWithDynamicClockType TimeWithDynamicClockType::nowWithSameClock() const
49{
50 return now(clockType());
51}
52
53WallTime TimeWithDynamicClockType::wallTime() const
54{
55 RELEASE_ASSERT(m_type == ClockType::Wall);
56 return WallTime::fromRawSeconds(m_value);
57}
58
59MonotonicTime TimeWithDynamicClockType::monotonicTime() const
60{
61 RELEASE_ASSERT(m_type == ClockType::Monotonic);
62 return MonotonicTime::fromRawSeconds(m_value);
63}
64
65WallTime TimeWithDynamicClockType::approximateWallTime() const
66{
67 switch (m_type) {
68 case ClockType::Wall:
69 return wallTime();
70 case ClockType::Monotonic:
71 return monotonicTime().approximateWallTime();
72 }
73 RELEASE_ASSERT_NOT_REACHED();
74 return WallTime();
75}
76
77MonotonicTime TimeWithDynamicClockType::approximateMonotonicTime() const
78{
79 switch (m_type) {
80 case ClockType::Wall:
81 return wallTime().approximateMonotonicTime();
82 case ClockType::Monotonic:
83 return monotonicTime();
84 }
85 RELEASE_ASSERT_NOT_REACHED();
86 return MonotonicTime();
87}
88
89Seconds TimeWithDynamicClockType::operator-(const TimeWithDynamicClockType& other) const
90{
91 RELEASE_ASSERT(m_type == other.m_type);
92 return Seconds(m_value - other.m_value);
93}
94
95bool TimeWithDynamicClockType::operator<(const TimeWithDynamicClockType& other) const
96{
97 RELEASE_ASSERT(m_type == other.m_type);
98 return m_value < other.m_value;
99}
100
101bool TimeWithDynamicClockType::operator>(const TimeWithDynamicClockType& other) const
102{
103 RELEASE_ASSERT(m_type == other.m_type);
104 return m_value > other.m_value;
105}
106
107bool TimeWithDynamicClockType::operator<=(const TimeWithDynamicClockType& other) const
108{
109 RELEASE_ASSERT(m_type == other.m_type);
110 return m_value <= other.m_value;
111}
112
113bool TimeWithDynamicClockType::operator>=(const TimeWithDynamicClockType& other) const
114{
115 RELEASE_ASSERT(m_type == other.m_type);
116 return m_value >= other.m_value;
117}
118
119void TimeWithDynamicClockType::dump(PrintStream& out) const
120{
121 out.print(m_type, "(", m_value, " sec)");
122}
123
124void sleep(const TimeWithDynamicClockType& time)
125{
126 Lock fakeLock;
127 Condition fakeCondition;
128 Locker fakeLocker { fakeLock };
129 fakeCondition.waitUntil(fakeLock, time);
130}
131
132bool hasElapsed(const TimeWithDynamicClockType& time)
133{
134 // Avoid doing now().
135 if (!(time > time.withSameClockAndRawSeconds(0)))
136 return true;
137 if (std::isinf(time.secondsSinceEpoch().value()))
138 return false;
139
140 return time <= time.nowWithSameClock();
141}
