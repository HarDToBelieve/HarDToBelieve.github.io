29#include <wtf/HashMap.h>
30#include <wtf/Lock.h>

32namespace WTF {
33
34namespace {
35
36class State {
37 WTF_MAKE_NONCOPYABLE(State);
38 WTF_MAKE_FAST_ALLOCATED;
39public:
40
41 struct CallData {
42 Seconds totalDuration;
43 unsigned callCount { 0 };
44 Seconds maxDuration;
45
46 Seconds meanDuration() const { return totalDuration / callCount; }
47 };
48
49 State() = default;
50
51 const CallData& addToTotal(const char* name, Seconds duration)
52 {
53 Locker locker { lock };
54 auto& result = totals.add(name, CallData()).iterator->value;
55 ++result.callCount;
56 result.maxDuration = std::max(result.maxDuration, duration);
57 result.totalDuration += duration;
58 return result;
59 }
60
61private:
62 HashMap<const char*, CallData> totals WTF_GUARDED_BY_LOCK(lock);
63 Lock lock;
64};
65
66State& state()
67{
68 static Atomic<State*> s_state;
69 return ensurePointer(s_state, [] { return new State(); });
70}
71
72} // anonymous namespace
73
74void TimingScope::scopeDidEnd()
75{
76 const auto& data = state().addToTotal(m_name, MonotonicTime::now() - m_startTime);
77 if (!(data.callCount % m_logIterationInterval))
78 WTFLogAlways("%s: %u calls, mean duration: %.6fms, total duration: %.6fms, max duration %.6fms", m_name, data.callCount, data.meanDuration().milliseconds(), data.totalDuration.milliseconds(), data.maxDuration.milliseconds());
79}
