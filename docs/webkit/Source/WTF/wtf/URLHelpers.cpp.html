44// This needs to be higher than the UScriptCode for any of the scripts on the IDN allowed list.
45// At one point we used USCRIPT_CODE_LIMIT from ICU, but there are two reasons not to use it.
46// 1) ICU considers it deprecated, so by setting U_HIDE_DEPRECATED we can't see it.
47// 2) No good reason to limit ourselves to scripts that existed in the ICU headers when
48// WebKit was compiled.
49// This is only really important for platforms that load an external IDN allowed script list.
50// Not important for the compiled-in one.
51constexpr auto scriptCodeLimit = static_cast<UScriptCode>(256);
52
53static uint32_t allowedIDNScriptBits[(scriptCodeLimit + 31) / 32];

95template <UScriptCode ScriptType>
96bool isOfScriptType(UChar32 codePoint)
97{
98 UErrorCode error = U_ZERO_ERROR;
99 UScriptCode script = uscript_getScript(codePoint, &error);
100 if (error != U_ZERO_ERROR) {
101 LOG_ERROR("got ICU error while trying to look at scripts: %d", error);
102 return false;
103 }
104 return script == ScriptType;
105}

107template<typename CharacterType> inline bool isASCIIDigitOrValidHostCharacter(CharacterType charCode)
108{
109 if (!isASCIIDigitOrPunctuation(charCode))
110 return false;
111
112 // Things the URL Parser rejects:
113 switch (charCode) {
114 case '#':
115 case '%':
116 case '/':
117 case ':':
118 case '?':
119 case '@':
120 case '[':
121 case '\\':
122 case ']':
123 return false;
124 default:
125 return true;
126 }
127}

129template <UScriptCode ScriptType>
130bool isLookalikeSequence(const std::optional<UChar32>& previousCodePoint, UChar32 codePoint)
131{
132 if (!previousCodePoint || *previousCodePoint == '/')
133 return false;
134
135 auto isLookalikePair = [] (UChar first, UChar second) {
136 return isLookalikeCharacterOfScriptType<ScriptType>(first) && !(isOfScriptType<ScriptType>(second) || isASCIIDigitOrValidHostCharacter(second));
137 };
138 return isLookalikePair(codePoint, *previousCodePoint)
139 || isLookalikePair(*previousCodePoint, codePoint);
140}

142static bool isLookalikeCharacter(const std::optional<UChar32>& previousCodePoint, UChar32 codePoint)
143{
144 // This function treats the following as unsafe, lookalike characters:
145 // any non-printable character, any character considered as whitespace,
146 // any ignorable character, and emoji characters related to locks.
147
148 // We also considered the characters in Mozilla's list of characters <http://kb.mozillazine.org/Network.IDN.blacklist_chars>.
149
150 // Some of the characters here will never appear once ICU has encoded.
151 // For example, ICU transforms most spaces into an ASCII space and most
152 // slashes into an ASCII solidus. But one of the two callers uses this
153 // on characters that have not been processed by ICU, so they are needed here.
154
155 if (!u_isprint(codePoint) || u_isUWhiteSpace(codePoint) || u_hasBinaryProperty(codePoint, UCHAR_DEFAULT_IGNORABLE_CODE_POINT))
156 return true;
157
158 switch (codePoint) {
159 case 0x00BC: /* VULGAR FRACTION ONE QUARTER */
160 case 0x00BD: /* VULGAR FRACTION ONE HALF */
161 case 0x00BE: /* VULGAR FRACTION THREE QUARTERS */
162 case 0x00ED: /* LATIN SMALL LETTER I WITH ACUTE */
163 /* 0x0131 LATIN SMALL LETTER DOTLESS I is intentionally not considered a lookalike character because it is visually distinguishable from i and it has legitimate use in the Turkish language. */
164 case 0x01C0: /* LATIN LETTER DENTAL CLICK */
165 case 0x01C3: /* LATIN LETTER RETROFLEX CLICK */

184 case 0x066A: /* ARABIC PERCENT SIGN */
185 case 0x06D4: /* ARABIC FULL STOP */

187 case 0x0701: /* SYRIAC SUPRALINEAR FULL STOP */

294static void addScriptToIDNAllowedScriptList(int32_t script)
295{
296 if (script >= 0 && script < scriptCodeLimit) {
297 size_t index = script / 32;
298 uint32_t mask = 1 << (script % 32);
299 allowedIDNScriptBits[index] |= mask;
300 }
301}
302
303static void addScriptToIDNAllowedScriptList(UScriptCode script)
304{
305 addScriptToIDNAllowedScriptList(static_cast<int32_t>(script));
306}

310 addScriptToIDNAllowedScriptList(u_getPropertyValueEnum(UCHAR_SCRIPT, scriptName));

341static bool allCharactersInAllowedIDNScriptList(const UChar* buffer, int32_t length)
342{
343 loadIDNAllowedScriptList();
344 int32_t i = 0;
345 std::optional<UChar32> previousCodePoint;
346 while (i < length) {
347 UChar32 c;
348 U16_NEXT(buffer, i, length, c);
349 UErrorCode error = U_ZERO_ERROR;
350 UScriptCode script = uscript_getScript(c, &error);
351 if (error != U_ZERO_ERROR) {
352 LOG_ERROR("got ICU error while trying to look at scripts: %d", error);
353 return false;
354 }
355 if (script < 0) {
356 LOG_ERROR("got negative number for script code from ICU: %d", script);
357 return false;
358 }
359 if (script >= scriptCodeLimit)
360 return false;
361
362 size_t index = script / 32;
363 uint32_t mask = 1 << (script % 32);
364 if (!(allowedIDNScriptBits[index] & mask))
365 return false;
366
367 if (isLookalikeCharacter(previousCodePoint, c))
368 return false;
369 previousCodePoint = c;
370 }
371 return true;
372}

374template<typename Func>
375static inline bool isSecondLevelDomainNameAllowedByTLDRules(const UChar* buffer, int32_t length, Func characterIsAllowed)
376{
377 ASSERT(length > 0);
378
379 for (int32_t i = length - 1; i >= 0; --i) {
380 UChar ch = buffer[i];
381
382 if (characterIsAllowed(ch))
383 continue;
384
385 // Only check the second level domain. Lower level registrars may have different rules.
386 if (ch == '.')
387 break;
388
389 return false;
390 }
391 return true;
392}

394#define CHECK_RULES_IF_SUFFIX_MATCHES(suffix, function) \
395 { \
396 static const int32_t suffixLength = sizeof(suffix) / sizeof(suffix[0]); \
397 if (length > suffixLength && !memcmp(buffer + length - suffixLength, suffix, sizeof(suffix))) \
398 return isSecondLevelDomainNameAllowedByTLDRules(buffer, length - suffixLength, function); \
399 }

401static bool isRussianDomainNameCharacter(UChar ch)
402{
403 // Only modern Russian letters, digits and dashes are allowed.
404 return (ch >= 0x0430 && ch <= 0x044f) || ch == 0x0451 || isASCIIDigit(ch) || ch == '-';
405}

572// Return value of null means no mapping is necessary.
573std::optional<String> mapHostName(const String& hostName, URLDecodeFunction decodeFunction)
574{
575 // destinationBuffer needs to be big enough to hold an IDN-encoded name.
576 // For host names bigger than this, we won't do IDN encoding, which is almost certainly OK.
577 if (hostName.length() > URLParser::hostnameBufferLength)
578 return String();
579
580 if (!hostName.length())
581 return String();
582
583 String string;
584 if (decodeFunction && string.contains('%'))
585 string = (*decodeFunction)(hostName);
586 else
587 string = hostName;
588
589 unsigned length = string.length();
590
591 auto sourceBuffer = string.charactersWithNullTermination();
592
593 UChar destinationBuffer[URLParser::hostnameBufferLength];
594 UErrorCode uerror = U_ZERO_ERROR;
595 UIDNAInfo processingDetails = UIDNA_INFO_INITIALIZER;
596 int32_t numCharactersConverted = (decodeFunction ? uidna_nameToASCII : uidna_nameToUnicode)(&URLParser::internationalDomainNameTranscoder(), sourceBuffer.data(), length, destinationBuffer, URLParser::hostnameBufferLength, &processingDetails, &uerror);
597 int allowedErrors = decodeFunction ? 0 : URLParser::allowedNameToASCIIErrors;
598 if (length && (U_FAILURE(uerror) || processingDetails.errors & ~allowedErrors))
599 return std::nullopt;
600
601 if (numCharactersConverted == static_cast<int32_t>(length) && !memcmp(sourceBuffer.data(), destinationBuffer, length * sizeof(UChar)))
602 return String();
603
604 if (!decodeFunction && !allCharactersInAllowedIDNScriptList(destinationBuffer, numCharactersConverted) && !allCharactersAllowedByTLDRules(destinationBuffer, numCharactersConverted))
605 return String();
606
607 return String(destinationBuffer, numCharactersConverted);
608}

612static void collectRangesThatNeedMapping(const String& string, unsigned location, unsigned length, MappingRangesVector& array, URLDecodeFunction decodeFunction)
613{
614 // Generally, we want to optimize for the case where there is one host name that does not need mapping.
615 // Therefore, we use null to indicate no mapping here and an empty array to indicate error.
616
617 String substring = string.substringSharingImpl(location, length);
618 std::optional<String> host = mapHostName(substring, decodeFunction);
619
620 if (host && !*host)
621 return;
622
623 if (!array)
624 array = Vector<std::tuple<unsigned, unsigned, String>>();
625
626 if (host)
627 array->constructAndAppend(location, length, *host);
628}

630static void applyHostNameFunctionToMailToURLString(const String& string, URLDecodeFunction decodeFunction, MappingRangesVector& array)
631{
632 // In a mailto: URL, host names come after a '@' character and end with a '>' or ',' or '?' character.
633 // Skip quoted strings so that characters in them don't confuse us.
634 // When we find a '?' character, we are past the part of the URL that contains host names.
635
636 unsigned stringLength = string.length();
637 unsigned current = 0;
638
639 while (1) {
640 // Find start of host name or of quoted string.
641 auto hostNameOrStringStart = string.find([](UChar ch) {
642 return ch == '"' || ch == '@' || ch == '?';
643 }, current);
644 if (hostNameOrStringStart == notFound)
645 return;
646
647 UChar c = string[hostNameOrStringStart];
648 current = hostNameOrStringStart + 1;
649
650 if (c == '?')
651 return;
652
653 if (c == '@') {
654 // Find end of host name.
655 unsigned hostNameStart = current;
656 auto hostNameEnd = string.find([](UChar ch) {
657 return ch == '>' || ch == ',' || ch == '?';
658 }, current);
659
660 bool done;
661 if (hostNameEnd == notFound) {
662 hostNameEnd = stringLength;
663 done = true;
664 } else {
665 current = hostNameEnd;
666 done = false;
667 }
668
669 // Process host name range.
670 collectRangesThatNeedMapping(string, hostNameStart, hostNameEnd - hostNameStart, array, decodeFunction);
671
672 if (done)
673 return;
674 } else {
675 // Skip quoted string.
676 ASSERT(c == '"');
677 while (1) {
678 auto escapedCharacterOrStringEnd = string.find([](UChar ch) {
679 return ch == '"' || ch == '\\';
680 }, current);
681 if (escapedCharacterOrStringEnd == notFound)
682 return;
683
684 c = string[escapedCharacterOrStringEnd];
685 current = escapedCharacterOrStringEnd + 1;
686
687 // If we are the end of the string, then break from the string loop back to the host name loop.
688 if (c == '"')
689 break;
690
691 // Skip escaped character.
692 ASSERT(c == '\\');
693 if (current == stringLength)
694 return;
695
696 ++current;
697 }
698 }
699 }
700}

702static void applyHostNameFunctionToURLString(const String& string, URLDecodeFunction decodeFunction, MappingRangesVector& array)
703{
704 // Find hostnames. Too bad we can't use any real URL-parsing code to do this,
705 // but we have to do it before doing all the %-escaping, and this is the only
706 // code we have that parses mailto URLs anyway.
707
708 // Maybe we should implement this using a character buffer instead?
709
710 if (protocolIs(string, "mailto")) {
711 applyHostNameFunctionToMailToURLString(string, decodeFunction, array);
712 return;
713 }
714
715 // Find the host name in a hierarchical URL.
716 // It comes after a "://" sequence, with scheme characters preceding.
717 // If ends with the end of the string or a ":", "/", or a "?".
718 // If there is a "@" character, the host part is just the part after the "@".
719 static const char* separator = "://";
720 auto separatorIndex = string.find(separator);
721 if (separatorIndex == notFound)
722 return;
723
724 unsigned authorityStart = separatorIndex + strlen(separator);
725
726 // Check that all characters before the :// are valid scheme characters.
727 if (StringView { string }.left(separatorIndex).contains([](UChar character) {
728 return !(isASCIIAlphanumeric(character) || character == '+' || character == '-' || character == '.');
729 }))
730 return;
731
732 // Find terminating character.
733 auto hostNameTerminator = string.find([](UChar character) {
734 return character == ':' || character == '/' || character == '?' || character == '#';
735 }, authorityStart);
736 unsigned hostNameEnd = hostNameTerminator == notFound ? string.length() : hostNameTerminator;
737
738 // Find "@" for the start of the host name.
739 auto userInfoTerminator = StringView { string }.left(hostNameEnd).find('@', authorityStart);
740 unsigned hostNameStart = userInfoTerminator == notFound ? authorityStart : userInfoTerminator + 1;
741
742 collectRangesThatNeedMapping(string, hostNameStart, hostNameEnd - hostNameStart, array, decodeFunction);
743}
744
745String mapHostNames(const String& string, URLDecodeFunction decodeFunction)
746{
747 // Generally, we want to optimize for the case where there is one host name that does not need mapping.
748
749 if (decodeFunction && string.isAllASCII())
750 return string;
751
752 // Make a list of ranges that actually need mapping.
753 MappingRangesVector hostNameRanges;
754 applyHostNameFunctionToURLString(string, decodeFunction, hostNameRanges);
755 if (!hostNameRanges)
756 return string;
757
758 if (hostNameRanges->isEmpty())
759 return { };
760
761 // Do the mapping.
762 String result = string;
763 while (!hostNameRanges->isEmpty()) {
764 auto [location, length, mappedHostName] = hostNameRanges->takeLast();
765 result = result.replace(location, length, mappedHostName);
766 }
767 return result;
768}

770static String escapeUnsafeCharacters(const String& sourceBuffer)
771{
772 unsigned length = sourceBuffer.length();
773
774 std::optional<UChar32> previousCodePoint;
775
776 unsigned i;
777 for (i = 0; i < length; ) {
778 UChar32 c = sourceBuffer.characterStartingAt(i);
779 if (isLookalikeCharacter(previousCodePoint, sourceBuffer.characterStartingAt(i)))
780 break;
781 previousCodePoint = c;
782 i += U16_LENGTH(c);
783 }
784
785 if (i == length)
786 return sourceBuffer;
787
788 Vector<UChar, urlBytesBufferLength> outBuffer;
789
790 outBuffer.grow(i);
791 if (sourceBuffer.is8Bit())
792 StringImpl::copyCharacters(outBuffer.data(), sourceBuffer.characters8(), i);
793 else
794 StringImpl::copyCharacters(outBuffer.data(), sourceBuffer.characters16(), i);
795
796 for (; i < length; ) {
797 UChar32 c = sourceBuffer.characterStartingAt(i);
798 unsigned characterLength = U16_LENGTH(c);
799 if (isLookalikeCharacter(previousCodePoint, c)) {
800 uint8_t utf8Buffer[4];
801 size_t offset = 0;
802 UBool failure = false;
803 U8_APPEND(utf8Buffer, offset, 4, c, failure);
804 ASSERT_UNUSED(failure, !failure);
805
806 for (size_t j = 0; j < offset; ++j) {
807 outBuffer.append('%');
808 outBuffer.append(upperNibbleToASCIIHexDigit(utf8Buffer[j]));
809 outBuffer.append(lowerNibbleToASCIIHexDigit(utf8Buffer[j]));
810 }
811 } else {
812 for (unsigned j = 0; j < characterLength; ++j)
813 outBuffer.append(sourceBuffer[i + j]);
814 }
815 previousCodePoint = c;
816 i += characterLength;
817 }
818
819 return String::adopt(WTFMove(outBuffer));
820}

822String userVisibleURL(const CString& url)
823{
824 auto* before = url.dataAsUInt8Ptr();
825 int length = url.length();
826
827 if (!length)
828 return { };
829
830 bool mayNeedHostNameDecoding = false;
831
832 CheckedInt32 bufferLength = length;
833 bufferLength = bufferLength * 3 + 1; // The buffer should be large enough to %-escape every character.
834 if (bufferLength.hasOverflowed())
835 return { };
836 Vector<char, urlBytesBufferLength> after(bufferLength);
837
838 char* q = after.data();
839 {
840 const unsigned char* p = before;
841 for (int i = 0; i < length; i++) {
842 unsigned char c = p[i];
843 // unescape escape sequences that indicate bytes greater than 0x7f
844 if (c == '%' && i + 2 < length && isASCIIHexDigit(p[i + 1]) && isASCIIHexDigit(p[i + 2])) {
845 auto u = toASCIIHexValue(p[i + 1], p[i + 2]);
846 if (u > 0x7f) {
847 // unescape
848 *q++ = u;
849 } else {
850 // do not unescape
851 *q++ = p[i];
852 *q++ = p[i + 1];
853 *q++ = p[i + 2];
854 }
855 i += 2;
856 } else {
857 *q++ = c;
858
859 // Check for "xn--" in an efficient, non-case-sensitive, way.
860 if (c == '-' && i >= 3 && !mayNeedHostNameDecoding && (q[-4] | 0x20) == 'x' && (q[-3] | 0x20) == 'n' && q[-2] == '-')
861 mayNeedHostNameDecoding = true;
862 }
863 }
864 *q = '\0';
865 }
866
867 // Check string to see if it can be converted to display using UTF-8
868 String result = String::fromUTF8(after.data());
869 if (!result) {
870 // Could not convert to UTF-8.
871 // Convert characters greater than 0x7f to escape sequences.
872 // Shift current string to the end of the buffer
873 // then we will copy back bytes to the start of the buffer
874 // as we convert.
875 int afterlength = q - after.data();
876 char* p = after.data() + bufferLength.value() - afterlength - 1;
877 memmove(p, after.data(), afterlength + 1); // copies trailing '\0'
878 char* q = after.data();
879 while (*p) {
880 unsigned char c = *p;
881 if (c > 0x7f) {
882 *q++ = '%';
883 *q++ = upperNibbleToASCIIHexDigit(c);
884 *q++ = lowerNibbleToASCIIHexDigit(c);
885 } else
886 *q++ = *p;
887 p++;
888 }
889 *q = '\0';
890 // Note: after.data() points to a null-terminated, pure ASCII string.
891 result = String::fromUTF8(after.data());
892 ASSERT(!!result);
893 }
894
895 // Note: result is UTF–16 string, created from either a valid UTF-8 string,
896 // or a pure ASCII string (where all bytes with the high bit set are
897 // percent-encoded).
898
899 if (mayNeedHostNameDecoding) {
900 // FIXME: Is it good to ignore the failure of mapHostNames and keep result intact?
901 auto mappedResult = mapHostNames(result, nullptr);
902 if (!!mappedResult)
903 result = mappedResult;
904 }
905
906 return escapeUnsafeCharacters(normalizedNFC(result));
907}
