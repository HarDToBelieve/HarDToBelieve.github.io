115template <bool canMoveWithMemcpy, typename T>
116struct VectorMover;
117
118template<typename T>
119struct VectorMover<false, T>
120{
121 static void move(T* src, T* srcEnd, T* dst)
122 {
123 while (src != srcEnd) {
124 new (NotNull, dst) T(WTFMove(*src));
125 src->~T();
126 ++dst;
127 ++src;
128 }
129 }
130 static void moveOverlapping(T* src, T* srcEnd, T* dst)
131 {
132 if (src > dst)
133 move(src, srcEnd, dst);
134 else {
135 T* dstEnd = dst + (srcEnd - src);
136 while (src != srcEnd) {
137 --srcEnd;
138 --dstEnd;
139 new (NotNull, dstEnd) T(WTFMove(*srcEnd));
140 srcEnd->~T();
141 }
142 }
143 }
144};

147struct VectorMover<true, T>

151 memcpy(static_cast<void*>(dst), static_cast<void*>(const_cast<T*>(src)), reinterpret_cast<const char*>(srcEnd) - reinterpret_cast<const char*>(src));
152 }
153 static void moveOverlapping(const T* src, const T* srcEnd, T* dst)
154 {
155 memmove(static_cast<void*>(dst), static_cast<void*>(const_cast<T*>(src)), reinterpret_cast<const char*>(srcEnd) - reinterpret_cast<const char*>(src));

283template<typename T, typename Malloc>
284class VectorBufferBase {
285 WTF_MAKE_NONCOPYABLE(VectorBufferBase);
286public:
287 template<FailureAction action>
288 bool allocateBuffer(size_t newCapacity)
289 {
290 static_assert(action == FailureAction::Crash || action == FailureAction::Report);
291 ASSERT(newCapacity);
292 if (newCapacity > std::numeric_limits<unsigned>::max() / sizeof(T)) {
293 if constexpr (action == FailureAction::Crash)
294 CRASH();
295 else
296 return false;
297 }
298
299 size_t sizeToAllocate = newCapacity * sizeof(T);
300 T* newBuffer = nullptr;
301 if constexpr (action == FailureAction::Crash)
302 newBuffer = static_cast<T*>(Malloc::malloc(sizeToAllocate));
303 else {
304 newBuffer = static_cast<T*>(Malloc::tryMalloc(sizeToAllocate));
305 if (UNLIKELY(!newBuffer))
306 return false;
307 }
308 m_capacity = sizeToAllocate / sizeof(T);
309 m_buffer = newBuffer;
310 return true;
311 }
312
313 ALWAYS_INLINE void allocateBuffer(size_t newCapacity) { allocateBuffer<FailureAction::Crash>(newCapacity); }
314 ALWAYS_INLINE bool tryAllocateBuffer(size_t newCapacity) { return allocateBuffer<FailureAction::Report>(newCapacity); }
315
316 bool shouldReallocateBuffer(size_t newCapacity) const
317 {
318 return VectorTraits<T>::canMoveWithMemcpy && m_capacity && newCapacity;
319 }
320
321 void reallocateBuffer(size_t newCapacity)
322 {
323 ASSERT(shouldReallocateBuffer(newCapacity));
324 if (newCapacity > std::numeric_limits<size_t>::max() / sizeof(T))
325 CRASH();
326 size_t sizeToAllocate = newCapacity * sizeof(T);
327 m_capacity = sizeToAllocate / sizeof(T);
328 m_buffer = static_cast<T*>(Malloc::realloc(m_buffer, sizeToAllocate));
329 }
330
331 void deallocateBuffer(T* bufferToDeallocate)
332 {
333 if (!bufferToDeallocate)
334 return;
335
336 if (m_buffer == bufferToDeallocate) {
337 m_buffer = nullptr;
338 m_capacity = 0;
339 }
340
341 Malloc::free(bufferToDeallocate);
342 }
343
344 T* buffer() { return m_buffer; }
345 const T* buffer() const { return m_buffer; }
346 static ptrdiff_t bufferMemoryOffset() { return OBJECT_OFFSETOF(VectorBufferBase, m_buffer); }
347 size_t capacity() const { return m_capacity; }
348
349 MallocPtr<T, Malloc> releaseBuffer()
350 {
351 T* buffer = m_buffer;
352 m_buffer = nullptr;
353 m_capacity = 0;
354 return adoptMallocPtr<T, Malloc>(buffer);
355 }
356
357protected:
358 VectorBufferBase()
359 : m_buffer(nullptr)
360 , m_capacity(0)
361 , m_size(0)
362 {
363 }
364
365 VectorBufferBase(T* buffer, size_t capacity, size_t size)
366 : m_buffer(buffer)
367 , m_capacity(capacity)
368 , m_size(size)
369 {
370 }
371
372 ~VectorBufferBase()
373 {
374 // FIXME: It would be nice to find a way to ASSERT that m_buffer hasn't leaked here.
375 }
376
377 T* m_buffer;
378 unsigned m_capacity;
379 unsigned m_size; // Only used by the Vector subclass, but placed here to avoid padding the struct.
380};

443template<typename T, size_t inlineCapacity, typename Malloc>
444class VectorBuffer : private VectorBufferBase<T, Malloc> {
445 WTF_MAKE_NONCOPYABLE(VectorBuffer);
446private:
447 typedef VectorBufferBase<T, Malloc> Base;
448public:
449 VectorBuffer()
450 : Base(inlineBuffer(), inlineCapacity, 0)
451 {
452 }
453
454 VectorBuffer(size_t capacity, size_t size = 0)
455 : Base(inlineBuffer(), inlineCapacity, size)
456 {
457 if (capacity > inlineCapacity)
458 Base::allocateBuffer(capacity);
459 }
460
461 ~VectorBuffer()
462 {
463 deallocateBuffer(buffer());
464 }
465
466 template<FailureAction action>
467 bool allocateBuffer(size_t newCapacity)
468 {
469 // FIXME: This should ASSERT(!m_buffer) to catch misuse/leaks.
470 if (newCapacity > inlineCapacity)
471 return Base::template allocateBuffer<action>(newCapacity);
472 m_buffer = inlineBuffer();
473 m_capacity = inlineCapacity;
474 return true;
475 }
476
477 ALWAYS_INLINE void allocateBuffer(size_t newCapacity) { allocateBuffer<FailureAction::Crash>(newCapacity); }
478 ALWAYS_INLINE bool tryAllocateBuffer(size_t newCapacity) { return allocateBuffer<FailureAction::Report>(newCapacity); }
479
480 void deallocateBuffer(T* bufferToDeallocate)
481 {
482 if (bufferToDeallocate == inlineBuffer())
483 return;
484 Base::deallocateBuffer(bufferToDeallocate);
485 }
486
487 bool shouldReallocateBuffer(size_t newCapacity) const
488 {
489 // We cannot reallocate the inline buffer.
490 return Base::shouldReallocateBuffer(newCapacity) && std::min(static_cast<size_t>(m_capacity), newCapacity) > inlineCapacity;
491 }
492
493 void reallocateBuffer(size_t newCapacity)
494 {
495 ASSERT(shouldReallocateBuffer(newCapacity));
496 Base::reallocateBuffer(newCapacity);
497 }
498
499 void swap(VectorBuffer& other, size_t mySize, size_t otherSize)
500 {
501 if (buffer() == inlineBuffer() && other.buffer() == other.inlineBuffer()) {
502 swapInlineBuffer(other, mySize, otherSize);
503 std::swap(m_capacity, other.m_capacity);
504 } else if (buffer() == inlineBuffer()) {
505 m_buffer = other.m_buffer;
506 other.m_buffer = other.inlineBuffer();
507 swapInlineBuffer(other, mySize, 0);
508 std::swap(m_capacity, other.m_capacity);
509 } else if (other.buffer() == other.inlineBuffer()) {
510 other.m_buffer = m_buffer;
511 m_buffer = inlineBuffer();
512 swapInlineBuffer(other, 0, otherSize);
513 std::swap(m_capacity, other.m_capacity);
514 } else {
515 std::swap(m_buffer, other.m_buffer);
516 std::swap(m_capacity, other.m_capacity);
517 }
518 }
519
520 void restoreInlineBufferIfNeeded()
521 {
522 if (m_buffer)
523 return;
524 m_buffer = inlineBuffer();
525 m_capacity = inlineCapacity;
526 }
527
528#if ASAN_ENABLED
529 void* endOfBuffer()
530 {
531 ASSERT(buffer());
532
533 IGNORE_GCC_WARNINGS_BEGIN("invalid-offsetof")
534 static_assert((offsetof(VectorBuffer, m_inlineBuffer) + sizeof(m_inlineBuffer)) % 8 == 0, "Inline buffer end needs to be on 8 byte boundary for ASan annotations to work.");
535 IGNORE_GCC_WARNINGS_END
536
537 if (buffer() == inlineBuffer())
538 return reinterpret_cast<char*>(m_inlineBuffer) + sizeof(m_inlineBuffer);
539
540 return buffer() + capacity();
541 }
542#endif
543
544 using Base::buffer;
545 using Base::capacity;
546 using Base::bufferMemoryOffset;
547
548 MallocPtr<T, Malloc> releaseBuffer()
549 {
550 if (buffer() == inlineBuffer())
551 return { };
552 return Base::releaseBuffer();
553 }
554
555protected:
556 using Base::m_size;
557
558private:
559 using Base::m_buffer;
560 using Base::m_capacity;
561
562 void swapInlineBuffer(VectorBuffer& other, size_t mySize, size_t otherSize)
563 {
564 // FIXME: We could make swap part of VectorTypeOperations
565 // https://bugs.webkit.org/show_bug.cgi?id=128863
566 swapInlineBuffers(inlineBuffer(), other.inlineBuffer(), mySize, otherSize);
567 }
568
569 static void swapInlineBuffers(T* left, T* right, size_t leftSize, size_t rightSize)
570 {
571 if (left == right)
572 return;
573
574 ASSERT(leftSize <= inlineCapacity);
575 ASSERT(rightSize <= inlineCapacity);
576
577 size_t swapBound = std::min(leftSize, rightSize);
578 for (unsigned i = 0; i < swapBound; ++i)
579 std::swap(left[i], right[i]);
580 VectorTypeOperations<T>::move(left + swapBound, left + leftSize, right + swapBound);
581 VectorTypeOperations<T>::move(right + swapBound, right + rightSize, left + swapBound);
582 }
583
584 T* inlineBuffer() { return reinterpret_cast_ptr<T*>(m_inlineBuffer); }
585 const T* inlineBuffer() const { return reinterpret_cast_ptr<const T*>(m_inlineBuffer); }
586
587#if ASAN_ENABLED
588 // ASan needs the buffer to begin and end on 8-byte boundaries for annotations to work.
589 // FIXME: Add a redzone before the buffer to catch off by one accesses. We don't need a guard after, because the buffer is the last member variable.
590 static constexpr size_t asanInlineBufferAlignment = std::alignment_of<T>::value >= 8 ? std::alignment_of<T>::value : 8;
591 static constexpr size_t asanAdjustedInlineCapacity = ((sizeof(T) * inlineCapacity + 7) & ~7) / sizeof(T);
592 typename std::aligned_storage<sizeof(T), asanInlineBufferAlignment>::type m_inlineBuffer[asanAdjustedInlineCapacity];
593#else
594 typename std::aligned_storage<sizeof(T), std::alignment_of<T>::value>::type m_inlineBuffer[inlineCapacity];
595#endif
596};

1508 iterator holeBegin = end();
1509 iterator holeEnd = end();
1510 unsigned matchCount = 0;
1511 for (auto it = begin() + startIndex, itEnd = end(); it < itEnd; ++it) {
1512 if (matches(*it)) {

1514 holeBegin = it;
1515 else if (holeEnd != it) {

1520 it->~T();

1522 }

1525 TypeOperations::moveOverlapping(holeEnd, end(), holeBegin);
1526 asanBufferSizeWillChangeTo(m_size - matchCount);
1527 m_size -= matchCount;
