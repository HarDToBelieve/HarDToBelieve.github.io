62#if ENABLE(UNIFIED_AND_FREEZABLE_CONFIG_RECORD)
63void setPermissionsOfConfigPage()
64{
65#if OS(DARWIN)
66 static std::once_flag onceFlag;
67 std::call_once(onceFlag, [] {
68 mach_vm_address_t addr = bitwise_cast<uintptr_t>(static_cast<void*>(WebConfig::g_config));
69 auto flags = VM_FLAGS_FIXED | VM_FLAGS_OVERWRITE;
70#if HAVE(VM_FLAGS_PERMANENT)
71 flags |= VM_FLAGS_PERMANENT;
72#endif
73
74 auto attemptVMMapping = [&] {
75 return mach_vm_map(mach_task_self(), &addr, ConfigSizeToProtect, pageSize() - 1, flags, MEMORY_OBJECT_NULL, 0, false, VM_PROT_READ | VM_PROT_WRITE, VM_PROT_READ | VM_PROT_WRITE, VM_INHERIT_DEFAULT);
76 };
77
78 auto result = attemptVMMapping();
79
80#if HAVE(VM_FLAGS_PERMANENT)
81 if (result != KERN_SUCCESS) {
82 flags &= ~VM_FLAGS_PERMANENT;
83 result = attemptVMMapping();
84 }
85#endif
86
87 RELEASE_ASSERT(result == KERN_SUCCESS);
88 });
89#endif // OS(DARWIN)
90}
91#endif // ENABLE(UNIFIED_AND_FREEZABLE_CONFIG_RECORD)

93void Config::permanentlyFreeze()
94{
95 static Lock configLock;
96 Locker locker { configLock };
97
98 RELEASE_ASSERT(roundUpToMultipleOf(pageSize(), ConfigSizeToProtect) == ConfigSizeToProtect);
99
100 if (!g_wtfConfig.isPermanentlyFrozen) {
101 g_wtfConfig.isPermanentlyFrozen = true;
102#if GIGACAGE_ENABLED
103 g_gigacageConfig.isPermanentlyFrozen = true;
104#endif
105 }
106
107 int result = 0;
108
109#if ENABLE(UNIFIED_AND_FREEZABLE_CONFIG_RECORD)
110#if OS(DARWIN)
111 enum {
112 AllowPermissionChangesAfterThis = false,
113 DisallowPermissionChangesAfterThis = true
114 };
115
116 // There's no going back now!
117 result = vm_protect(mach_task_self(), reinterpret_cast<vm_address_t>(&WebConfig::g_config), ConfigSizeToProtect, DisallowPermissionChangesAfterThis, VM_PROT_READ);
118#elif OS(LINUX)
119 result = mprotect(&WebConfig::g_config, ConfigSizeToProtect, PROT_READ);
120#elif OS(WINDOWS)
121 // FIXME: Implement equivalent, maybe with VirtualProtect.
122 // Also need to fix WebKitTestRunner.
123
124 // Note: the Windows port also currently does not support a unified Config
125 // record, which is needed for the current form of the freezing feature to
126 // work. See comments in PlatformEnable.h for UNIFIED_AND_FREEZABLE_CONFIG_RECORD.
127#endif
128#endif // ENABLE(UNIFIED_AND_FREEZABLE_CONFIG_RECORD)
129
130 RELEASE_ASSERT(!result);
131 RELEASE_ASSERT(g_wtfConfig.isPermanentlyFrozen);
132}
