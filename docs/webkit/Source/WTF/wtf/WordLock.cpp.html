37// This data structure serves three purposes:
38//
39// 1) A parking mechanism for threads that go to sleep. That involves just a system mutex and
40// condition variable.
41//
42// 2) A queue node for when a thread is on some WordLock's queue.
43//
44// 3) The queue head. This is kind of funky. When a thread is the head of a queue, it also serves as
45// the basic queue bookkeeping data structure. When a thread is dequeued, the next thread in the
46// queue takes on the queue head duties.
47struct ThreadData {
48 // The parking mechanism.
49 bool shouldPark { false };
50 std::mutex parkingLock;
51 std::condition_variable parkingCondition;
52
53 // The queue node.
54 ThreadData* nextInQueue { nullptr };
55
56 // The queue itself.
57 ThreadData* queueTail { nullptr };
58};
59
60} // anonymous namespace

62NEVER_INLINE void WordLock::lockSlow()
63{
64 unsigned spinCount = 0;
65
66 // This magic number turns out to be optimal based on past JikesRVM experiments.
67 const unsigned spinLimit = 40;
68
69 for (;;) {
70 uintptr_t currentWordValue = m_word.load();
71
72 if (!(currentWordValue & isLockedBit)) {
73 // It's not possible for someone to hold the queue lock while the lock itself is no longer
74 // held, since we will only attempt to acquire the queue lock when the lock is held and
75 // the queue lock prevents unlock.
76 ASSERT(!(currentWordValue & isQueueLockedBit));
77 if (m_word.compareExchangeWeak(currentWordValue, currentWordValue | isLockedBit)) {
78 // Success! We acquired the lock.
79 return;
80 }
81 }
82
83 // If there is no queue and we haven't spun too much, we can just try to spin around again.
84 if (!(currentWordValue & ~queueHeadMask) && spinCount < spinLimit) {
85 spinCount++;
86 Thread::yield();
87 continue;
88 }
89
90 // Need to put ourselves on the queue. Create the queue if one does not exist. This requries
91 // owning the queue for a little bit. The lock that controls the queue is itself a spinlock.
92
93 ThreadData me;
94
95 // Reload the current word value, since some time may have passed.
96 currentWordValue = m_word.load();
97
98 // We proceed only if the queue lock is not held, the WordLock is held, and we succeed in
99 // acquiring the queue lock.
100 if ((currentWordValue & isQueueLockedBit)
101 || !(currentWordValue & isLockedBit)
102 || !m_word.compareExchangeWeak(currentWordValue, currentWordValue | isQueueLockedBit)) {
103 Thread::yield();
104 continue;
105 }
106
107 me.shouldPark = true;
108
109 // We own the queue. Nobody can enqueue or dequeue until we're done. Also, it's not possible
110 // to release the WordLock while we hold the queue lock.
111 ThreadData* queueHead = bitwise_cast<ThreadData*>(currentWordValue & ~queueHeadMask);
112 if (queueHead) {
113 // Put this thread at the end of the queue.
114 queueHead->queueTail->nextInQueue = &me;
115 queueHead->queueTail = &me;
116
117 // Release the queue lock.
118 currentWordValue = m_word.load();
119 ASSERT(currentWordValue & ~queueHeadMask);
120 ASSERT(currentWordValue & isQueueLockedBit);
121 ASSERT(currentWordValue & isLockedBit);
122 m_word.store(currentWordValue & ~isQueueLockedBit);
123 } else {
124 // Make this thread be the queue-head.
125 queueHead = &me;
126 me.queueTail = &me;
127
128 // Release the queue lock and install ourselves as the head. No need for a CAS loop, since
129 // we own the queue lock.
130 currentWordValue = m_word.load();
131 ASSERT(~(currentWordValue & ~queueHeadMask));
132 ASSERT(currentWordValue & isQueueLockedBit);
133 ASSERT(currentWordValue & isLockedBit);
134 uintptr_t newWordValue = currentWordValue;
135 newWordValue |= bitwise_cast<uintptr_t>(queueHead);
136 newWordValue &= ~isQueueLockedBit;
137 m_word.store(newWordValue);
138 }
139
140 // At this point everyone who acquires the queue lock will see me on the queue, and anyone who
141 // acquires me's lock will see that me wants to park. Note that shouldPark may have been
142 // cleared as soon as the queue lock was released above, but it will happen while the
143 // releasing thread holds me's parkingLock.
144
145 {
146 std::unique_lock<std::mutex> locker(me.parkingLock);
147 while (me.shouldPark)
148 me.parkingCondition.wait(locker);
149 }
150
151 ASSERT(!me.shouldPark);
152 ASSERT(!me.nextInQueue);
153 ASSERT(!me.queueTail);
154
155 // Now we can loop around and try to acquire the lock again.
156 }
157}

159NEVER_INLINE void WordLock::unlockSlow()
160{
161 // The fast path can fail either because of spurious weak CAS failure, or because someone put a
162 // thread on the queue, or the queue lock is held. If the queue lock is held, it can only be
163 // because someone *will* enqueue a thread onto the queue.
164
165 // Acquire the queue lock, or release the lock. This loop handles both lock release in case the
166 // fast path's weak CAS spuriously failed and it handles queue lock acquisition if there is
167 // actually something interesting on the queue.
168 for (;;) {
169 uintptr_t currentWordValue = m_word.load();
170
171 ASSERT(currentWordValue & isLockedBit);
172
173 if (currentWordValue == isLockedBit) {
174 if (m_word.compareExchangeWeak(isLockedBit, 0)) {
175 // The fast path's weak CAS had spuriously failed, and now we succeeded. The lock is
176 // unlocked and we're done!
177 return;
178 }
179 // Loop around and try again.
180 Thread::yield();
181 continue;
182 }
183
184 if (currentWordValue & isQueueLockedBit) {
185 Thread::yield();
186 continue;
187 }
188
189 // If it wasn't just a spurious weak CAS failure and if the queue lock is not held, then there
190 // must be an entry on the queue.
191 ASSERT(currentWordValue & ~queueHeadMask);
192
193 if (m_word.compareExchangeWeak(currentWordValue, currentWordValue | isQueueLockedBit))
194 break;
195 }
196
197 uintptr_t currentWordValue = m_word.load();
198
199 // After we acquire the queue lock, the WordLock must still be held and the queue must be
200 // non-empty. The queue must be non-empty since only the lockSlow() method could have held the
201 // queue lock and if it did then it only releases it after putting something on the queue.
202 ASSERT(currentWordValue & isLockedBit);
203 ASSERT(currentWordValue & isQueueLockedBit);
204 ThreadData* queueHead = bitwise_cast<ThreadData*>(currentWordValue & ~queueHeadMask);
205 ASSERT(queueHead);
206
207 ThreadData* newQueueHead = queueHead->nextInQueue;
208 // Either this was the only thread on the queue, in which case we delete the queue, or there
209 // are still more threads on the queue, in which case we create a new queue head.
210 if (newQueueHead)
211 newQueueHead->queueTail = queueHead->queueTail;
212
213 // Change the queue head, possibly removing it if newQueueHead is null. No need for a CAS loop,
214 // since we hold the queue lock and the lock itself so nothing about the lock can change right
215 // now.
216 currentWordValue = m_word.load();
217 ASSERT(currentWordValue & isLockedBit);
218 ASSERT(currentWordValue & isQueueLockedBit);
219 ASSERT((currentWordValue & ~queueHeadMask) == bitwise_cast<uintptr_t>(queueHead));
220 uintptr_t newWordValue = currentWordValue;
221 newWordValue &= ~isLockedBit; // Release the WordLock.
222 newWordValue &= ~isQueueLockedBit; // Release the queue lock.
223 newWordValue &= queueHeadMask; // Clear out the old queue head.
224 newWordValue |= bitwise_cast<uintptr_t>(newQueueHead); // Install new queue head.
225 m_word.store(newWordValue);
226
227 // Now the lock is available for acquisition. But we just have to wake up the old queue head.
228 // After that, we're done!
229
230 queueHead->nextInQueue = nullptr;
231 queueHead->queueTail = nullptr;
232
233 // We do this carefully because this may run either before or during the parkingLock critical
234 // section in lockSlow().
235 {
236 // Be sure to hold the lock across our call to notify_one() because a spurious wakeup could
237 // cause the thread at the head of the queue to exit and delete queueHead.
238 std::scoped_lock<std::mutex> locker(queueHead->parkingLock);
239 queueHead->shouldPark = false;
240
241 // Doesn't matter if we notify_all() or notify_one() here since the only thread that could be
242 // waiting is queueHead.
243 queueHead->parkingCondition.notify_one();
244 }
245
246 // The old queue head can now contend for the lock again. We're done!
247}
