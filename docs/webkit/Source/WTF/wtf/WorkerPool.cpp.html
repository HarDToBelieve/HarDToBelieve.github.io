26#include "config.h"
27#include <wtf/WorkerPool.h>
28
29namespace WTF {
30
31class WorkerPool::Worker final : public AutomaticThread {
32public:
33 friend class WorkerPool;
34
35 Worker(const AbstractLocker& locker, WorkerPool& pool, Box<Lock> lock, Ref<AutomaticThreadCondition>&& condition, Seconds timeout)
36 : AutomaticThread(locker, lock, WTFMove(condition), timeout)
37 , m_pool(pool)
38 {
39 }
40
41 PollResult poll(const AbstractLocker&) final
42 {
43 if (m_pool.m_tasks.isEmpty())
44 return PollResult::Wait;
45 m_task = m_pool.m_tasks.takeFirst();
46 if (!m_task)
47 return PollResult::Stop;
48 return PollResult::Work;
49 }
50
51 WorkResult work() final
52 {
53 m_task();
54 m_task = nullptr;
55 return WorkResult::Continue;
56 }
57
58 void threadDidStart() final
59 {
60 Locker locker { *m_pool.m_lock };
61 m_pool.m_numberOfActiveWorkers++;
62 }
63
64 void threadIsStopping(const AbstractLocker&) final
65 {
66 m_pool.m_numberOfActiveWorkers--;
67 }
68
69 bool shouldSleep(const AbstractLocker& locker) final
70 {
71 return m_pool.shouldSleep(locker);
72 }
73
74 const char* name() const final
75 {
76 return m_pool.name();
77 }
78
79private:
80 WorkerPool& m_pool;
81 Function<void()> m_task;
82};
83
84WorkerPool::WorkerPool(ASCIILiteral name, unsigned numberOfWorkers, Seconds timeout)
85 : m_lock(Box<Lock>::create())
86 , m_condition(AutomaticThreadCondition::create())
87 , m_timeout(timeout)
88 , m_name(name)
89{
90 Locker locker { *m_lock };
91 for (unsigned i = 0; i < numberOfWorkers; ++i)
92 m_workers.append(adoptRef(*new Worker(locker, *this, m_lock, m_condition.copyRef(), timeout)));
93}
94
95WorkerPool::~WorkerPool()
96{
97 {
98 Locker locker { *m_lock };
99 for (unsigned i = m_workers.size(); i--;)
100 m_tasks.append(nullptr); // Use null task to indicate that we want the thread to terminate.
101 m_condition->notifyAll(locker);
102 }
103 for (auto& worker : m_workers)
104 worker->join();
105 ASSERT(!m_numberOfActiveWorkers);
106}
107
108bool WorkerPool::shouldSleep(const AbstractLocker&)
109{
110 if (m_timeout > 0_s && std::isinf(m_timeout))
111 return false;
112
113 MonotonicTime currentTime = MonotonicTime::now();
114 if (std::isnan(m_lastTimeoutTime) || (currentTime >= (m_lastTimeoutTime + m_timeout))) {
115 m_lastTimeoutTime = currentTime;
116 return true;
117 }
118 return false;
119}
120
121void WorkerPool::postTask(Function<void()>&& task)
122{
123 Locker locker { *m_lock };
124 m_tasks.append(WTFMove(task));
125 m_condition->notifyOne(locker);
126}
