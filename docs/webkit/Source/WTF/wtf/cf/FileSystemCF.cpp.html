39CString FileSystem::fileSystemRepresentation(const String& path)
40{
41 RetainPtr<CFStringRef> cfString = path.createCFString();
42
43 if (!cfString)
44 return CString();
45
46 CFIndex size = CFStringGetMaximumSizeOfFileSystemRepresentation(cfString.get());
47
48 Vector<char> buffer(size);
49
50 if (!CFStringGetFileSystemRepresentation(cfString.get(), buffer.data(), buffer.size())) {
51 LOG_ERROR("Failed to get filesystem representation to create CString from cfString");
52 return CString();
53 }
54
55 return CString(buffer.data(), strlen(buffer.data()));
56}
57
58String FileSystem::stringFromFileSystemRepresentation(const char* fileSystemRepresentation)
59{
60 return adoptCF(CFStringCreateWithFileSystemRepresentation(kCFAllocatorDefault, fileSystemRepresentation)).get();
61}
62
63RetainPtr<CFURLRef> FileSystem::pathAsURL(const String& path)
64{
65 CFURLPathStyle pathStyle;
66#if PLATFORM(WIN)
67 pathStyle = kCFURLWindowsPathStyle;
68#else
69 pathStyle = kCFURLPOSIXPathStyle;
70#endif
71 return adoptCF(CFURLCreateWithFileSystemPath(nullptr, path.createCFString().get(), pathStyle, FALSE));
72}
