32namespace WTF {
33
34bool SchedulePair::operator==(const SchedulePair& other) const
35{
36 if (runLoop() != other.runLoop())
37 return false;
38 CFStringRef thisMode = mode();
39 CFStringRef otherMode = other.mode();
40 if (!thisMode || !otherMode)
41 return thisMode == otherMode;
42 return CFEqual(thisMode, otherMode);
43}
