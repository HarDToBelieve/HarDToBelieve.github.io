36URL::URL(CFURLRef url)
37{
38 if (!url) {
39 invalidate();
40 return;
41 }
42
43 // FIXME: Why is it OK to ignore base URL here?
44 CString urlBytes;
45 getURLBytes(url, urlBytes);
46 URLParser parser(urlBytes.data());
47 *this = parser.result();
48}
49
50#if !USE(FOUNDATION)
51RetainPtr<CFURLRef> URL::createCFURL() const
52{
53 RetainPtr<CFURLRef> cfURL;
54 if (LIKELY(m_string.is8Bit() && m_string.isAllASCII()))
55 cfURL = adoptCF(CFURLCreateAbsoluteURLWithBytes(nullptr, m_string.characters8(), m_string.length(), kCFStringEncodingUTF8, nullptr, true));
56 else {
57 CString utf8 = m_string.utf8();
58 cfURL = adoptCF(CFURLCreateAbsoluteURLWithBytes(nullptr, utf8.dataAsUInt8Ptr(), utf8.length(), kCFStringEncodingUTF8, nullptr, true));
59 }
60
61 if (protocolIsInHTTPFamily() && !isCFURLSameOrigin(cfURL.get(), *this))
62 return nullptr;
63
64 return cfURL;
65}
66#endif
67
68String URL::fileSystemPath() const
69{
70 RetainPtr<CFURLRef> cfURL = createCFURL();
71 if (!cfURL)
72 return String();
73
74#if PLATFORM(WIN)
75 CFURLPathStyle pathStyle = kCFURLWindowsPathStyle;
76#else
77 CFURLPathStyle pathStyle = kCFURLPOSIXPathStyle;
78#endif
79 return adoptCF(CFURLCopyFileSystemPath(cfURL.get(), pathStyle)).get();
80}
