35namespace WTF {
36
37static bool hasEntitlement(SecTaskRef task, const char* entitlement)
38{
39 if (!task)
40 return false;
41 auto string = adoptCF(CFStringCreateWithCStringNoCopy(kCFAllocatorDefault, entitlement, kCFStringEncodingASCII, kCFAllocatorNull));
42 auto value = adoptCF(SecTaskCopyValueForEntitlement(task, string.get(), nullptr));
43 return value && CFGetTypeID(value.get()) == CFBooleanGetTypeID() && CFBooleanGetValue(static_cast<CFBooleanRef>(value.get()));
44}
45
46bool hasEntitlement(audit_token_t token, const char* entitlement)
47{
48 return hasEntitlement(adoptCF(SecTaskCreateWithAuditToken(kCFAllocatorDefault, token)).get(), entitlement);
49}
50
51bool hasEntitlement(xpc_connection_t connection, const char *entitlement)
52{
53 auto value = adoptOSObject(xpc_connection_copy_entitlement_value(connection, entitlement));
54 return value && xpc_get_type(value.get()) == XPC_TYPE_BOOL && xpc_bool_get_value(value.get());
55}
56
57bool processHasEntitlement(const char* entitlement)
58{
59 return hasEntitlement(adoptCF(SecTaskCreateFromSelf(kCFAllocatorDefault)).get(), entitlement);
60}
61
62bool hasEntitlementValue(audit_token_t token, const char* entitlement, const char* value)
63{
64 auto secTaskForToken = adoptCF(SecTaskCreateWithAuditToken(kCFAllocatorDefault, token));
65 if (!secTaskForToken)
66 return { };
67
68 auto string = adoptCF(CFStringCreateWithCStringNoCopy(kCFAllocatorDefault, entitlement, kCFStringEncodingASCII, kCFAllocatorNull));
69 String entitlementValue = dynamic_cf_cast<CFStringRef>(adoptCF(SecTaskCopyValueForEntitlement(secTaskForToken.get(), string.get(), nullptr)).get());
70 return entitlementValue == value;
71}
