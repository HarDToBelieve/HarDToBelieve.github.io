66String createTemporaryZipArchive(const String& path)
67{
68 String temporaryFile;
69
70 RetainPtr<NSFileCoordinator> coordinator = adoptNS([[NSFileCoordinator alloc] initWithFilePresenter:nil]);
71 [coordinator coordinateReadingItemAtURL:[NSURL fileURLWithPath:path] options:NSFileCoordinatorReadingWithoutChanges error:nullptr byAccessor:[&](NSURL *newURL) mutable {
72 CString archivePath([NSTemporaryDirectory() stringByAppendingPathComponent:@"WebKitGeneratedFileXXXXXX"].fileSystemRepresentation);
73 if (mkstemp(archivePath.mutableData()) == -1)
74 return;
75
76 NSDictionary *options = @{
77 (__bridge id)kBOMCopierOptionCreatePKZipKey : @YES,
78 (__bridge id)kBOMCopierOptionSequesterResourcesKey : @YES,
79 (__bridge id)kBOMCopierOptionKeepParentKey : @YES,
80 (__bridge id)kBOMCopierOptionCopyResourcesKey : @YES,
81 };
82
83 BOMCopier copier = BOMCopierNew();
84 if (!BOMCopierCopyWithOptions(copier, newURL.path.fileSystemRepresentation, archivePath.data(), (__bridge CFDictionaryRef)options))
85 temporaryFile = String::fromUTF8(archivePath);
86 BOMCopierFree(copier);
87 }];
88
89 return temporaryFile;
90}

92String openTemporaryFile(const String& prefix, PlatformFileHandle& platformFileHandle, const String& suffix)
93{
94 platformFileHandle = invalidPlatformFileHandle;
95
96 Vector<char> temporaryFilePath(PATH_MAX);
97 if (!confstr(_CS_DARWIN_USER_TEMP_DIR, temporaryFilePath.data(), temporaryFilePath.size()))
98 return String();
99
100 // Shrink the vector.
101 temporaryFilePath.shrink(strlen(temporaryFilePath.data()));
102
103 // FIXME: Change to a runtime assertion that the path ends with a slash once <rdar://problem/23579077> is
104 // fixed in all iOS Simulator versions that we use.
105 if (temporaryFilePath.last() != '/')
106 temporaryFilePath.append('/');
107
108 // Append the file name.
109 CString prefixUTF8 = prefix.utf8();
110 temporaryFilePath.append(prefixUTF8.data(), prefixUTF8.length());
111 temporaryFilePath.append("XXXXXX", 6);
112
113 // Append the file name suffix.
114 CString suffixUTF8 = suffix.utf8();
115 temporaryFilePath.append(suffixUTF8.data(), suffixUTF8.length());
116 temporaryFilePath.append('\0');
117
118 platformFileHandle = mkstemps(temporaryFilePath.data(), suffixUTF8.length());
119 if (platformFileHandle == invalidPlatformFileHandle)
120 return String();
121
122 return String::fromUTF8(temporaryFilePath.data());
123}

125NSString *createTemporaryDirectory(NSString *directoryPrefix)
126{
127 NSString *tempDirectory = NSTemporaryDirectory();
128 if (!tempDirectory || ![tempDirectory length])
129 return nil;
130
131 if (!directoryPrefix || ![directoryPrefix length])
132 return nil;
133
134 NSString *tempDirectoryComponent = [directoryPrefix stringByAppendingString:@"-XXXXXXXX"];
135 const char* tempDirectoryCString = [[tempDirectory stringByAppendingPathComponent:tempDirectoryComponent] fileSystemRepresentation];
136 if (!tempDirectoryCString)
137 return nil;
138
139 const size_t length = strlen(tempDirectoryCString);
140 ASSERT(length <= MAXPATHLEN);
141 if (length > MAXPATHLEN)
142 return nil;
143
144 const size_t lengthPlusNullTerminator = length + 1;
145 Vector<char, MAXPATHLEN + 1> path(lengthPlusNullTerminator);
146 memcpy(path.data(), tempDirectoryCString, lengthPlusNullTerminator);
147
148 if (!mkdtemp(path.data()))
149 return nil;
150
151 return [[NSFileManager defaultManager] stringWithFileSystemRepresentation:path.data() length:length];
152}

166bool setAllowsMaterializingDatalessFiles(bool allow, PolicyScope scope)
167{
168#ifdef IOPOL_TYPE_VFS_MATERIALIZE_DATALESS_FILES
169 if (setiopolicy_np(IOPOL_TYPE_VFS_MATERIALIZE_DATALESS_FILES, toIOPolicyScope(scope), allow ? IOPOL_MATERIALIZE_DATALESS_FILES_ON : IOPOL_MATERIALIZE_DATALESS_FILES_OFF) == -1) {
170 LOG_ERROR("FileSystem::setAllowsMaterializingDatalessFiles(%d): setiopolicy_np call failed, errno: %d", allow, errno);
171 return false;
172 }
173 return true;
174#else
175 UNUSED_PARAM(allow);
176 UNUSED_PARAM(scope);
177 return false;
178#endif
179}
180
181std::optional<bool> allowsMaterializingDatalessFiles(PolicyScope scope)
182{
183#ifdef IOPOL_TYPE_VFS_MATERIALIZE_DATALESS_FILES
184 int ret = getiopolicy_np(IOPOL_TYPE_VFS_MATERIALIZE_DATALESS_FILES, toIOPolicyScope(scope));
185 if (ret == IOPOL_MATERIALIZE_DATALESS_FILES_ON)
186 return true;
187 if (ret == IOPOL_MATERIALIZE_DATALESS_FILES_OFF)
188 return false;
189 LOG_ERROR("FileSystem::allowsMaterializingDatalessFiles(): getiopolicy_np call failed, errno: %d", errno);
190 return std::nullopt;
191#else
192 UNUSED_PARAM(scope);
193 return std::nullopt;
194#endif
195}

197#if PLATFORM(IOS_FAMILY)
198bool isSafeToUseMemoryMapForPath(const String& path)
199{
200 NSError *error = nil;
201 NSDictionary<NSFileAttributeKey, id> *attributes = [[NSFileManager defaultManager] attributesOfItemAtPath:path error:&error];
202 if (error) {
203 LOG_ERROR("Unable to get path protection class");
204 return false;
205 }
206 if ([[attributes objectForKey:NSFileProtectionKey] isEqualToString:NSFileProtectionComplete]) {
207 LOG_ERROR("Path protection class is NSFileProtectionComplete, so it is not safe to use memory map");
208 return false;
209 }
210 return true;
211}
212
213void makeSafeToUseMemoryMapForPath(const String& path)
214{
215 if (isSafeToUseMemoryMapForPath(path))
216 return;
217
218 NSError *error = nil;
219 BOOL success = [[NSFileManager defaultManager] setAttributes:@{ NSFileProtectionKey: NSFileProtectionCompleteUnlessOpen } ofItemAtPath:path error:&error];
220 ASSERT(!error);
221 ASSERT_UNUSED(success, success);
222}
