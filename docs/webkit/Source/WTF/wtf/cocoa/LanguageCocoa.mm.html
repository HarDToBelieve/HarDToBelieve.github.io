38size_t indexOfBestMatchingLanguageInList(const String& language, const Vector<String>& languageList, bool& exactMatch)
39{
40 auto matchedLanguages = retainPtr([NSLocale matchedLanguagesFromAvailableLanguages:createNSArray(languageList).get() forPreferredLanguages:@[ static_cast<NSString *>(language) ]]);
41 if (![matchedLanguages count]) {
42 exactMatch = false;
43 return languageList.size();
44 }
45
46 String firstMatchedLanguage = [matchedLanguages firstObject];
47
48 exactMatch = language == firstMatchedLanguage;
49
50 auto index = languageList.find(firstMatchedLanguage);
51 ASSERT(index < languageList.size());
52 return index;
53}
54
55bool canMinimizeLanguages()
56{
57 static const bool result = []() -> bool {
58#if PLATFORM(MAC)
59 if (applicationSDKVersion() < DYLD_MACOSX_VERSION_10_15_4)
60 return false;
61#endif
62#if PLATFORM(IOS)
63 if (applicationSDKVersion() < DYLD_IOS_VERSION_13_4)
64 return false;
65#endif
66 return [NSLocale respondsToSelector:@selector(minimizedLanguagesFromLanguages:)];
67 }();
68 return result;
69}
70
71RetainPtr<CFArrayRef> minimizedLanguagesFromLanguages(CFArrayRef languages)
72{
73 if (!canMinimizeLanguages())
74 return languages;
75
76ALLOW_NEW_API_WITHOUT_GUARDS_BEGIN
77 return (__bridge CFArrayRef)[NSLocale minimizedLanguagesFromLanguages:(__bridge NSArray<NSString *> *)languages];
78ALLOW_NEW_API_WITHOUT_GUARDS_END
79}
