43static void retainSendRight(mach_port_t port)
44{
45 if (port == MACH_PORT_NULL)
46 return;
47
48 auto kr = KERN_SUCCESS;
49 if (port != MACH_PORT_DEAD)
50 kr = mach_port_mod_refs(mach_task_self(), port, MACH_PORT_RIGHT_SEND, 1);
51
52 if (kr == KERN_INVALID_RIGHT || port == MACH_PORT_DEAD)
53 kr = mach_port_mod_refs(mach_task_self(), port, MACH_PORT_RIGHT_DEAD_NAME, 1);
54
55 if (kr != KERN_SUCCESS) {
56 LOG_ERROR("mach_port_mod_refs error for port %d: %s (%x)", port, mach_error_string(kr), kr);
57 if (kr == KERN_INVALID_RIGHT)
58 CRASH();
59 }
60}
61
62static void releaseSendRight(mach_port_t port)
63{
64 if (port == MACH_PORT_NULL)
65 return;
66
67 deallocateSendRightSafely(port);
68}
69
70void deallocateSendRightSafely(mach_port_t port)
71{
72 auto kr = mach_port_deallocate(mach_task_self(), port);
73 if (kr == KERN_SUCCESS)
74 return;
75
76 RELEASE_LOG_ERROR(Process, "mach_port_deallocate error for port %d: %{private}s (%#x)", port, mach_error_string(kr), kr);
77 if (kr == KERN_INVALID_RIGHT || kr == KERN_INVALID_NAME)
78 CRASH();
79}

81MachSendRight MachSendRight::adopt(mach_port_t port)
82{
83 return MachSendRight(port);
84}
85
86MachSendRight MachSendRight::create(mach_port_t port)
87{
88 retainSendRight(port);
89
90 return adopt(port);
91}

93MachSendRight::MachSendRight(mach_port_t port)
94 : m_port(port)
95{
96}
97
98MachSendRight::MachSendRight(MachSendRight&& other)
99 : m_port(other.leakSendRight())
100{
101}
102
103MachSendRight::MachSendRight(const MachSendRight& other)
104 : m_port(other.m_port)
105{
106 retainSendRight(m_port);
107}
108
109MachSendRight::~MachSendRight()
110{
111 releaseSendRight(m_port);
112}
113
114MachSendRight& MachSendRight::operator=(MachSendRight&& other)
115{
116 if (this != &other) {
117 releaseSendRight(m_port);
118 m_port = other.leakSendRight();
119 }
120
121 return *this;
122}
123
124MachSendRight& MachSendRight::operator=(const MachSendRight& other)
125{
126 if (this != &other) {
127 m_port = other.sendRight();
128 retainSendRight(m_port);
129 }
130
131 return *this;
132}
133
134MachSendRight MachSendRight::copySendRight() const
135{
136 return create(m_port);
137}
138
139mach_port_t MachSendRight::leakSendRight()
140{
141 return std::exchange(m_port, MACH_PORT_NULL);
142}
