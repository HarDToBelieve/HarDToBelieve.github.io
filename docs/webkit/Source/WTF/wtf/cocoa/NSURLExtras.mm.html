46static BOOL readIDNAllowedScriptListFile(NSString *filename)
47{
48 if (!filename)
49 return NO;
50
51 FILE *file = fopen([filename fileSystemRepresentation], "r");
52 if (!file)
53 return NO;
54
55 // Read a word at a time.
56 // Allow comments, starting with # character to the end of the line.
57 while (1) {
58 // Skip a comment if present.
59 if (fscanf(file, " #%*[^\n\r]%*[\n\r]") == EOF)
60 break;
61
62 // Read a script name if present.
63 char word[33];
64 int result = fscanf(file, " %32[^# \t\n\r]%*[^# \t\n\r] ", word);
65 if (result == EOF)
66 break;
67
68 if (result == 1) {
69 // Got a word, map to script code and put it into the array.
70 addScriptToIDNAllowedScriptList(word);
71 }
72 }
73 fclose(file);
74 return YES;
75}

80{
81 static dispatch_once_t flag;
82 dispatch_once(&flag, ^{
83 for (NSString *directory in NSSearchPathForDirectoriesInDomains(NSLibraryDirectory, NSAllDomainsMask, YES)) {

87 initializeDefaultIDNAllowedScriptList();
88 });
89}

93static String decodePercentEscapes(const String& string)
94{
95 auto substring = adoptCF(CFURLCreateStringByReplacingPercentEscapes(nullptr, string.createCFString().get(), CFSTR("")));
96 if (!substring)
97 return string;
98 return substring.get();
99}

124NSURL *URLByTruncatingOneCharacterBeforeComponent(NSURL *URL, CFURLComponentType component)
125{
126 if (!URL)
127 return nil;
128
129 CFRange fragRg = CFURLGetByteRangeForComponent((__bridge CFURLRef)URL, component, nullptr);
130 if (fragRg.location == kCFNotFound)
131 return URL;
132
133 Vector<UInt8, urlBytesBufferLength> urlBytes(urlBytesBufferLength);
134 CFIndex numBytes = CFURLGetBytes((__bridge CFURLRef)URL, urlBytes.data(), urlBytes.size());
135 if (numBytes == -1) {
136 numBytes = CFURLGetBytes((__bridge CFURLRef)URL, nullptr, 0);
137 urlBytes.grow(numBytes);
138 CFURLGetBytes((__bridge CFURLRef)URL, urlBytes.data(), numBytes);
139 }
140
141 auto result = adoptCF(CFURLCreateWithBytes(nullptr, urlBytes.data(), fragRg.location - 1, kCFStringEncodingUTF8, nullptr));
142 if (!result)
143 result = adoptCF(CFURLCreateWithBytes(nullptr, urlBytes.data(), fragRg.location - 1, kCFStringEncodingISOLatin1, nullptr));
144
145 return result ? result.bridgingAutorelease() : URL;
146}

153NSURL *URLWithData(NSData *data, NSURL *baseURL)
154{
155 if (!data)
156 return nil;
157
158 size_t length = [data length];
159 if (length > 0) {
160 // work around <rdar://4470771>: CFURLCreateAbsoluteURLWithBytes(.., TRUE) doesn't remove non-path components.
161 baseURL = URLByRemovingResourceSpecifier(baseURL);
162
163 const UInt8 *bytes = static_cast<const UInt8*>([data bytes]);
164
165 // CFURLCreateAbsoluteURLWithBytes would complain to console if we passed a path to it.
166 if (bytes[0] == '/' && !baseURL)
167 return nil;
168
169 // NOTE: We use UTF-8 here since this encoding is used when computing strings when returning URL components
170 // (e.g calls to NSURL -path). However, this function is not tolerant of illegal UTF-8 sequences, which
171 // could either be a malformed string or bytes in a different encoding, like shift-jis, so we fall back
172 // onto using ISO Latin 1 in those cases.
173 auto result = adoptCF(CFURLCreateAbsoluteURLWithBytes(nullptr, bytes, length, kCFStringEncodingUTF8, (__bridge CFURLRef)baseURL, YES));
174 if (!result)
175 result = adoptCF(CFURLCreateAbsoluteURLWithBytes(nullptr, bytes, length, kCFStringEncodingISOLatin1, (__bridge CFURLRef)baseURL, YES));
176 return result.bridgingAutorelease();
177 }
178 return [NSURL URLWithString:@""];
179}
180static NSData *dataWithUserTypedString(NSString *string)
181{
182 NSData *userTypedData = [string dataUsingEncoding:NSUTF8StringEncoding];
183 ASSERT(userTypedData);
184
185 const UInt8* inBytes = static_cast<const UInt8 *>([userTypedData bytes]);
186 int inLength = [userTypedData length];
187 if (!inLength)
188 return nil;
189
190 CheckedInt32 mallocLength = inLength;
191 mallocLength *= 3; // large enough to %-escape every character
192 if (mallocLength.hasOverflowed())
193 return nil;
194
195 char* outBytes = static_cast<char *>(malloc(mallocLength));
196 char* p = outBytes;
197 int outLength = 0;
198 for (int i = 0; i < inLength; i++) {
199 UInt8 c = inBytes[i];
200 if (c <= 0x20 || c >= 0x7f) {
201 *p++ = '%';
202 *p++ = upperNibbleToASCIIHexDigit(c);
203 *p++ = lowerNibbleToASCIIHexDigit(c);
204 outLength += 3;
205 } else {
206 *p++ = c;
207 outLength++;
208 }
209 }
210
211 return [NSData dataWithBytesNoCopy:outBytes length:outLength]; // adopts outBytes
212}

214NSURL *URLWithUserTypedString(NSString *string, NSURL *nsURL)
215{
216 if (!string)
217 return nil;
218
219 auto mappedString = mapHostNames(stringByTrimmingWhitespace(string).get(), decodePercentEscapes);
220 if (!mappedString)
221 return nil;
222
223 // Let's check whether the URL is bogus.
224 URL url { URL { nsURL }, mappedString };
225 if (!url.createCFURL())
226 return nil;
227
228 // FIXME: https://bugs.webkit.org/show_bug.cgi?id=186057
229 // We should be able to use url.createCFURL instead of using directly CFURL parsing routines.
230 NSData *data = dataWithUserTypedString(mappedString);
231 if (!data)
232 return [NSURL URLWithString:@""];
233
234 return URLWithData(data, nsURL);
235}

237NSURL *URLWithUserTypedStringDeprecated(NSString *string, NSURL *URL)
238{
239 if (!string)
240 return nil;
241
242 NSURL *result = URLWithUserTypedString(string, URL);
243 if (!result) {
244 NSData *resultData = dataWithUserTypedString(string);
245 if (!resultData)
246 return [NSURL URLWithString:@""];
247 result = URLWithData(resultData, URL);
248 }
249
250 return result;
251}

253static BOOL hasQuestionMarkOnlyQueryString(NSURL *URL)
254{
255 CFRange rangeWithSeparators;
256 CFURLGetByteRangeForComponent((__bridge CFURLRef)URL, kCFURLComponentQuery, &rangeWithSeparators);
257 if (rangeWithSeparators.location != kCFNotFound && rangeWithSeparators.length == 1)
258 return YES;
259
260 return NO;
261}

263NSData *dataForURLComponentType(NSURL *URL, CFURLComponentType componentType)
264{
265 Vector<UInt8, urlBytesBufferLength> allBytesBuffer(urlBytesBufferLength);
266 CFIndex bytesFilled = CFURLGetBytes((__bridge CFURLRef)URL, allBytesBuffer.data(), allBytesBuffer.size());
267 if (bytesFilled == -1) {
268 CFIndex bytesToAllocate = CFURLGetBytes((__bridge CFURLRef)URL, nullptr, 0);
269 allBytesBuffer.grow(bytesToAllocate);
270 bytesFilled = CFURLGetBytes((__bridge CFURLRef)URL, allBytesBuffer.data(), bytesToAllocate);
271 }
272
273 const CFURLComponentType completeURL = (CFURLComponentType)-1;
274 CFRange range;
275 if (componentType != completeURL) {
276 range = CFURLGetByteRangeForComponent((__bridge CFURLRef)URL, componentType, nullptr);
277 if (range.location == kCFNotFound)
278 return nil;
279 } else {
280 range.location = 0;
281 range.length = bytesFilled;
282 }
283
284 NSData *componentData = [NSData dataWithBytes:allBytesBuffer.data() + range.location length:range.length];
285
286 const unsigned char *bytes = static_cast<const unsigned char *>([componentData bytes]);
287 NSMutableData *resultData = [NSMutableData data];
288 // NOTE: add leading '?' to query strings non-zero length query strings.
289 // NOTE: retain question-mark only query strings.
290 if (componentType == kCFURLComponentQuery) {
291 if (range.length > 0 || hasQuestionMarkOnlyQueryString(URL))
292 [resultData appendBytes:"?" length:1];
293 }
294 for (int i = 0; i < range.length; i++) {
295 unsigned char c = bytes[i];
296 if (c <= 0x20 || c >= 0x7f) {
297 char escaped[3];
298 escaped[0] = '%';
299 escaped[1] = upperNibbleToASCIIHexDigit(c);
300 escaped[2] = lowerNibbleToASCIIHexDigit(c);
301 [resultData appendBytes:escaped length:3];
302 } else {
303 char b[1];
304 b[0] = c;
305 [resultData appendBytes:b length:1];
306 }
307 }
308
309 return resultData;
310}

312static NSURL *URLByRemovingComponentAndSubsequentCharacter(NSURL *URL, CFURLComponentType component)
313{
314 CFRange range = CFURLGetByteRangeForComponent((__bridge CFURLRef)URL, component, 0);
315 if (range.location == kCFNotFound)
316 return URL;
317
318 // Remove one subsequent character.
319 range.length++;
320
321 Vector<UInt8, urlBytesBufferLength> buffer(urlBytesBufferLength);
322 CFIndex numBytes = CFURLGetBytes((__bridge CFURLRef)URL, buffer.data(), buffer.size());
323 if (numBytes == -1) {
324 numBytes = CFURLGetBytes((__bridge CFURLRef)URL, nullptr, 0);
325 buffer.grow(numBytes);
326 CFURLGetBytes((__bridge CFURLRef)URL, buffer.data(), numBytes);
327 }
328 UInt8* urlBytes = buffer.data();
329
330 if (numBytes < range.location)
331 return URL;
332 if (numBytes < range.location + range.length)
333 range.length = numBytes - range.location;
334
335 memmove(urlBytes + range.location, urlBytes + range.location + range.length, numBytes - range.location + range.length);
336
337 auto result = adoptCF(CFURLCreateWithBytes(nullptr, urlBytes, numBytes - range.length, kCFStringEncodingUTF8, nullptr));
338 if (!result)
339 result = adoptCF(CFURLCreateWithBytes(nullptr, urlBytes, numBytes - range.length, kCFStringEncodingISOLatin1, nullptr));
340
341 return result ? result.bridgingAutorelease() : URL;
342}

349NSData *originalURLData(NSURL *URL)
350{
351 UInt8 *buffer = (UInt8 *)malloc(urlBytesBufferLength);
352 CFIndex bytesFilled = CFURLGetBytes((__bridge CFURLRef)URL, buffer, urlBytesBufferLength);
353 if (bytesFilled == -1) {
354 CFIndex bytesToAllocate = CFURLGetBytes((__bridge CFURLRef)URL, nullptr, 0);
355 buffer = (UInt8 *)realloc(buffer, bytesToAllocate);
356 bytesFilled = CFURLGetBytes((__bridge CFURLRef)URL, buffer, bytesToAllocate);
357 ASSERT(bytesFilled == bytesToAllocate);
358 }
359
360 // buffer is adopted by the NSData
361 NSData *data = [NSData dataWithBytesNoCopy:buffer length:bytesFilled freeWhenDone:YES];
362
363 NSURL *baseURL = (__bridge NSURL *)CFURLGetBaseURL((__bridge CFURLRef)URL);
364 if (baseURL)
365 return originalURLData(URLWithData(data, baseURL));
366 return data;
367}

376BOOL isUserVisibleURL(NSString *string)
377{
378 BOOL valid = YES;
379 // get buffer
380
381 char static_buffer[1024];
382 const char *p;
383 BOOL success = CFStringGetCString((__bridge CFStringRef)string, static_buffer, 1023, kCFStringEncodingUTF8);
384 p = success ? static_buffer : [string UTF8String];
385
386 int length = strlen(p);
387
388 // check for characters <= 0x20 or >=0x7f, %-escape sequences of %7f, and xn--, these
389 // are the things that will lead _web_userVisibleString to actually change things.
390 for (int i = 0; i < length; i++) {
391 unsigned char c = p[i];
392 // escape control characters, space, and delete
393 if (c <= 0x20 || c == 0x7f) {
394 valid = NO;
395 break;
396 } else if (c == '%' && (i + 1 < length && isASCIIHexDigit(p[i + 1])) && i + 2 < length && isASCIIHexDigit(p[i + 2])) {
397 auto u = toASCIIHexValue(p[i + 1], p[i + 2]);
398 if (u > 0x7f) {
399 valid = NO;
400 break;
401 }
402 i += 2;
403 } else {
404 // Check for "xn--" in an efficient, non-case-sensitive, way.
405 if (c == '-' && i >= 3 && (p[i - 3] | 0x20) == 'x' && (p[i - 2] | 0x20) == 'n' && p[i - 1] == '-') {
406 valid = NO;
407 break;
408 }
409 }
410 }
411
412 return valid;
413}

415NSRange rangeOfURLScheme(NSString *string)
416{
417 NSRange colon = [string rangeOfString:@":"];
418 if (colon.location != NSNotFound && colon.location > 0) {
419 NSRange scheme = {0, colon.location};
420 /*
421 This stuff is very expensive. 10-15 msec on a 2x1.2GHz. If not cached it swamps
422 everything else when adding items to the autocomplete DB. Makes me wonder if we
423 even need to enforce the character set here.
424 */
425 NSString *acceptableCharacters = @"ABCDEFGHIJKLMNOPQRSTUVWXYZabcdefghijklmnopqrstuvwxyz0123456789+.-";
426 static LazyNeverDestroyed<RetainPtr<NSCharacterSet>> inverseSchemeCharacterSet;
427 static std::once_flag onceKey;
428 std::call_once(onceKey, [&] {
429 inverseSchemeCharacterSet.construct([[NSCharacterSet characterSetWithCharactersInString:acceptableCharacters] invertedSet]);
430 });
431 NSRange illegals = [string rangeOfCharacterFromSet:inverseSchemeCharacterSet.get().get() options:0 range:scheme];
432 if (illegals.location == NSNotFound)
433 return scheme;
434 }
435 return NSMakeRange(NSNotFound, 0);
436}

438BOOL looksLikeAbsoluteURL(NSString *string)
439{
440 // Trim whitespace because _web_URLWithString allows whitespace.
441 return rangeOfURLScheme(stringByTrimmingWhitespace(string).get()).location != NSNotFound;
