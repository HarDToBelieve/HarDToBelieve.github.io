36struct DispatchWorkItem {
37 WTF_MAKE_STRUCT_FAST_ALLOCATED;
38 Ref<WorkQueue> m_workQueue;
39 Function<void()> m_function;
40 void operator()() { m_function(); }
41};
42
43}
44
45template<typename T> static void dispatchWorkItem(void* dispatchContext)
46{
47 T* item = reinterpret_cast<T*>(dispatchContext);
48 (*item)();
49 delete item;
50}
51
52void WorkQueue::dispatch(Function<void()>&& function)
53{
54 dispatch_async_f(m_dispatchQueue.get(), new DispatchWorkItem { makeRef(*this), WTFMove(function) }, dispatchWorkItem<DispatchWorkItem>);
55}
56
57void WorkQueue::dispatchAfter(Seconds duration, Function<void()>&& function)
58{
59 dispatch_after_f(dispatch_time(DISPATCH_TIME_NOW, duration.nanosecondsAs<int64_t>()), m_dispatchQueue.get(), new DispatchWorkItem { makeRef(*this), WTFMove(function) }, dispatchWorkItem<DispatchWorkItem>);
60}
61
62void WorkQueue::dispatchSync(Function<void()>&& function)
63{
64 dispatch_sync_f(m_dispatchQueue.get(), new Function<void()> { WTFMove(function) }, dispatchWorkItem<Function<void()>>);
65}
66
67Ref<WorkQueue> WorkQueue::constructMainWorkQueue()
68{
69 return adoptRef(*new WorkQueue(dispatch_get_main_queue()));
70}
71
72WorkQueue::WorkQueue(OSObjectPtr<dispatch_queue_t>&& dispatchQueue)
73 : m_dispatchQueue(WTFMove(dispatchQueue))
74{
75}
76
77void WorkQueue::platformInitialize(const char* name, Type type, QOS qos)
78{
79 dispatch_queue_attr_t attr = type == Type::Concurrent ? DISPATCH_QUEUE_CONCURRENT : DISPATCH_QUEUE_SERIAL;
80 attr = dispatch_queue_attr_make_with_qos_class(attr, Thread::dispatchQOSClass(qos), 0);
81 m_dispatchQueue = adoptOSObject(dispatch_queue_create(name, attr));
82 dispatch_set_context(m_dispatchQueue.get(), this);
83}

87}

89void WorkQueue::concurrentApply(size_t iterations, WTF::Function<void(size_t index)>&& function)
90{
91 dispatch_apply(iterations, dispatch_get_global_queue(DISPATCH_QUEUE_PRIORITY_DEFAULT, 0), makeBlockPtr([function = WTFMove(function)](size_t index) {
92 function(index);
93 }).get());
94}
