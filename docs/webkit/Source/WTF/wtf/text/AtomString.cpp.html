33template<AtomString::CaseConvertType type>
34ALWAYS_INLINE AtomString AtomString::convertASCIICase() const
35{
36 StringImpl* impl = this->impl();
37 if (UNLIKELY(!impl))
38 return nullAtom();
39
40 // Convert short strings without allocating a new StringImpl, since
41 // there's a good chance these strings are already in the atom
42 // string table and so no memory allocation will be required.
43 unsigned length;
44 const unsigned localBufferSize = 100;
45 if (impl->is8Bit() && (length = impl->length()) <= localBufferSize) {
46 const LChar* characters = impl->characters8();
47 unsigned failingIndex;
48 for (unsigned i = 0; i < length; ++i) {
49 if (type == CaseConvertType::Lower ? UNLIKELY(isASCIIUpper(characters[i])) : LIKELY(isASCIILower(characters[i]))) {
50 failingIndex = i;
51 goto SlowPath;
52 }
53 }
54 return *this;
55SlowPath:
56 LChar localBuffer[localBufferSize];
57 for (unsigned i = 0; i < failingIndex; ++i)
58 localBuffer[i] = characters[i];
59 for (unsigned i = failingIndex; i < length; ++i)
60 localBuffer[i] = type == CaseConvertType::Lower ? toASCIILower(characters[i]) : toASCIIUpper(characters[i]);
61 return AtomString(localBuffer, length);
62 }
63
64 Ref<StringImpl> convertedString = type == CaseConvertType::Lower ? impl->convertToASCIILowercase() : impl->convertToASCIIUppercase();
65 if (LIKELY(convertedString.ptr() == impl))
66 return *this;
67
68 AtomString result;
69 result.m_string = AtomStringImpl::add(convertedString.ptr());
70 return result;
71}
72
73AtomString AtomString::convertToASCIILowercase() const
74{
75 return convertASCIICase<CaseConvertType::Lower>();
76}
77
78AtomString AtomString::convertToASCIIUppercase() const
79{
80 return convertASCIICase<CaseConvertType::Upper>();
81}

115AtomString AtomString::fromUTF8Internal(const char* start, const char* end)
116{
117 ASSERT(start);
118
119 // Caller needs to handle empty string.
120 ASSERT(!end || end > start);
121 ASSERT(end || start[0]);
122
123 return AtomStringImpl::addUTF8(start, end ? end : start + std::strlen(start));
124}

141void AtomString::init()
142{
143 static std::once_flag initializeKey;
144 std::call_once(initializeKey, [] {
145 // Initialization is not thread safe, so this function must be called from the main thread first.
146 ASSERT(isUIThread());
147
148 nullAtomData.construct();
149 emptyAtomData.construct("");
150
151 // When starting WebThread via StartWebThread function, we have special period between the prologue of StartWebThread and spawning WebThread actually.
152 // In this period, `isMainThread()` returns false even if this is called on the main thread since WebThread is now enabled and we are not taking a WebThread lock.
153 // This causes assertion hits in MainThreadLazyNeverDestroyed initialization only in WebThread platforms.
154 // We bypass this by using constructWithoutAccessCheck, which intentionally skips `isMainThread()` check for construction.
155 // In non WebThread environment, we do not lose the assertion coverage since we already have ASSERT(isUIThread()). And ASSERT(isUIThread()) ensures that this
156 // is called in system main thread in WebThread platforms.
157 starAtomData.constructWithoutAccessCheck("*", AtomString::ConstructFromLiteral);
158 xmlAtomData.constructWithoutAccessCheck("xml", AtomString::ConstructFromLiteral);
159 xmlnsAtomData.constructWithoutAccessCheck("xmlns", AtomString::ConstructFromLiteral);
160 });
161}
