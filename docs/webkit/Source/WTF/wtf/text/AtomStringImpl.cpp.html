67static ALWAYS_INLINE StringTableImpl& stringTable()
68{
69 return Thread::current().atomStringTable()->table();
70}

72template<typename T, typename HashTranslator>
73static inline Ref<AtomStringImpl> addToStringTable(AtomStringTableLocker&, StringTableImpl& atomStringTable, const T& value)
74{
75 auto addResult = atomStringTable.add<HashTranslator>(value);
76
77 // If the string is newly-translated, then we need to adopt it.
78 // The boolean in the pair tells us if that is so.
79 if (addResult.isNewEntry)
80 return adoptRef(static_cast<AtomStringImpl&>(*addResult.iterator->get()));
81 return *static_cast<AtomStringImpl*>(addResult.iterator->get());
82}
83
84template<typename T, typename HashTranslator>
85static inline Ref<AtomStringImpl> addToStringTable(const T& value)
86{
87 AtomStringTableLocker locker;
88 return addToStringTable<T, HashTranslator>(locker, stringTable(), value);
89}

93 {

142struct HashAndUTF8Characters {
143 unsigned hash;
144 const char* characters;
145 unsigned length;
146 unsigned utf16Length;
147};
148
149struct HashAndUTF8CharactersTranslator {
150 static unsigned hash(const HashAndUTF8Characters& buffer)
151 {
152 return buffer.hash;
153 }
154
155 static bool equal(PackedPtr<StringImpl> const& passedString, const HashAndUTF8Characters& buffer)
156 {
157 auto* string = passedString.get();
158 if (buffer.utf16Length != string->length())
159 return false;
160
161 // If buffer contains only ASCII characters UTF-8 and UTF16 length are the same.
162 if (buffer.utf16Length != buffer.length) {
163 if (string->is8Bit())
164 return equalLatin1WithUTF8(string->characters8(), buffer.characters, buffer.characters + buffer.length);
165
166 return equalUTF16WithUTF8(string->characters16(), buffer.characters, buffer.characters + buffer.length);
167 }
168
169 if (string->is8Bit()) {
170 const LChar* stringCharacters = string->characters8();
171
172 for (unsigned i = 0; i < buffer.length; ++i) {
173 ASSERT(isASCII(buffer.characters[i]));
174 if (stringCharacters[i] != buffer.characters[i])
175 return false;
176 }
177
178 return true;
179 }
180
181 const UChar* stringCharacters = string->characters16();
182
183 for (unsigned i = 0; i < buffer.length; ++i) {
184 ASSERT(isASCII(buffer.characters[i]));
185 if (stringCharacters[i] != buffer.characters[i])
186 return false;
187 }
188
189 return true;
190 }
191
192 static void translate(PackedPtr<StringImpl>& location, const HashAndUTF8Characters& buffer, unsigned hash)
193 {
194 UChar* target;
195 auto newString = StringImpl::createUninitialized(buffer.utf16Length, target);
196
197 bool isAllASCII;
198 const char* source = buffer.characters;
199 if (!convertUTF8ToUTF16(source, source + buffer.length, &target, target + buffer.utf16Length, &isAllASCII))
200 ASSERT_NOT_REACHED();
201
202 if (isAllASCII)
203 newString = StringImpl::create(buffer.characters, buffer.length);
204
205 auto* pointer = &newString.leakRef();
206 pointer->setHash(hash);
207 pointer->setIsAtom(true);
208 location = pointer;
209 }
210};

224RefPtr<AtomStringImpl> AtomStringImpl::add(const UChar* characters)
225{
226 if (!characters)
227 return nullptr;
228
229 unsigned length = 0;
230 while (characters[length] != UChar(0))
231 ++length;
232
233 if (!length)
234 return static_cast<AtomStringImpl*>(StringImpl::empty());
235
236 UCharBuffer buffer { characters, length };
237 return addToStringTable<UCharBuffer, UCharBufferTranslator>(buffer);
238}

280RefPtr<AtomStringImpl> AtomStringImpl::add(StringImpl* baseString, unsigned start, unsigned length)
281{
282 if (!baseString)
283 return nullptr;
284
285 if (!length || start >= baseString->length())
286 return static_cast<AtomStringImpl*>(StringImpl::empty());
287
288 unsigned maxLength = baseString->length() - start;
289 if (length >= maxLength) {
290 if (!start)
291 return add(baseString);
292 length = maxLength;
293 }
294
295 SubstringLocation buffer = { baseString, start, length };
296 if (baseString->is8Bit())
297 return addToStringTable<SubstringLocation, SubstringTranslator8>(buffer);
298 return addToStringTable<SubstringLocation, SubstringTranslator16>(buffer);
299}

365static Ref<AtomStringImpl> addSymbol(AtomStringTableLocker& locker, StringTableImpl& atomStringTable, StringImpl& base)
366{
367 ASSERT(base.length());
368 ASSERT(base.isSymbol());
369
370 SubstringLocation buffer = { &base, 0, base.length() };
371 if (base.is8Bit())
372 return addToStringTable<SubstringLocation, SubstringTranslator8>(locker, atomStringTable, buffer);
373 return addToStringTable<SubstringLocation, SubstringTranslator16>(locker, atomStringTable, buffer);
374}

382static Ref<AtomStringImpl> addStatic(AtomStringTableLocker& locker, StringTableImpl& atomStringTable, const StringImpl& base)
383{
384 ASSERT(base.length());
385 ASSERT(base.isStatic());
386
387 if (base.is8Bit()) {
388 LCharBuffer buffer { base.characters8(), base.length(), base.hash() };
389 return addToStringTable<LCharBuffer, BufferFromStaticDataTranslator<LChar>>(locker, atomStringTable, buffer);
390 }
391 UCharBuffer buffer { base.characters16(), base.length(), base.hash() };
392 return addToStringTable<UCharBuffer, BufferFromStaticDataTranslator<UChar>>(locker, atomStringTable, buffer);
393}

408Ref<AtomStringImpl> AtomStringImpl::addSlowCase(StringImpl& string)
409{
410 // This check is necessary for null symbols.
411 // Their length is zero, but they are not AtomStringImpl.
412 if (!string.length())
413 return *static_cast<AtomStringImpl*>(StringImpl::empty());
414
415 if (string.isStatic())
416 return addStatic(string);
417
418 if (string.isSymbol())
419 return addSymbol(string);
420
421 ASSERT_WITH_MESSAGE(!string.isAtom(), "AtomStringImpl should not hit the slow case if the string is already an atom.");
422
423 AtomStringTableLocker locker;
424 auto addResult = stringTable().add(&string);
425
426 if (addResult.isNewEntry) {
427 ASSERT(addResult.iterator->get() == &string);
428 string.setIsAtom(true);
429 }
430
431 return *static_cast<AtomStringImpl*>(addResult.iterator->get());
432}
433
434Ref<AtomStringImpl> AtomStringImpl::addSlowCase(AtomStringTable& stringTable, StringImpl& string)
435{
436 // This check is necessary for null symbols.
437 // Their length is zero, but they are not AtomStringImpl.
438 if (!string.length())
439 return *static_cast<AtomStringImpl*>(StringImpl::empty());
440
441 if (string.isStatic()) {
442 AtomStringTableLocker locker;
443 return addStatic(locker, stringTable.table(), string);
444 }
445
446 if (string.isSymbol()) {
447 AtomStringTableLocker locker;
448 return addSymbol(locker, stringTable.table(), string);
449 }
450
451 ASSERT_WITH_MESSAGE(!string.isAtom(), "AtomStringImpl should not hit the slow case if the string is already an atom.");
452
453 AtomStringTableLocker locker;
454 auto addResult = stringTable.table().add(&string);
455
456 if (addResult.isNewEntry) {
457 ASSERT(addResult.iterator->get() == &string);
458 string.setIsAtom(true);
459 }
460
461 return *static_cast<AtomStringImpl*>(addResult.iterator->get());
462}

464void AtomStringImpl::remove(AtomStringImpl* string)
465{
466 ASSERT(string->isAtom());
467 AtomStringTableLocker locker;
468 auto& atomStringTable = stringTable();
469 auto iterator = atomStringTable.find(string);
470 ASSERT_WITH_MESSAGE(iterator != atomStringTable.end(), "The string being removed is an atom in the string table of an other thread!");
471 ASSERT(string == iterator->get());
472 atomStringTable.remove(iterator);
473}

475RefPtr<AtomStringImpl> AtomStringImpl::lookUpSlowCase(StringImpl& string)
476{
477 ASSERT_WITH_MESSAGE(!string.isAtom(), "AtomStringImpl objects should return from the fast case.");
478
479 if (!string.length())
480 return static_cast<AtomStringImpl*>(StringImpl::empty());
481
482 AtomStringTableLocker locker;
483 auto& atomStringTable = stringTable();
484 auto iterator = atomStringTable.find(&string);
485 if (iterator != atomStringTable.end())
486 return static_cast<AtomStringImpl*>(iterator->get());
487 return nullptr;
488}

490RefPtr<AtomStringImpl> AtomStringImpl::addUTF8(const char* charactersStart, const char* charactersEnd)
491{
492 HashAndUTF8Characters buffer;
493 buffer.characters = charactersStart;
494 buffer.hash = calculateStringHashAndLengthFromUTF8MaskingTop8Bits(charactersStart, charactersEnd, buffer.length, buffer.utf16Length);
495
496 if (!buffer.hash)
497 return nullptr;
498
499 return addToStringTable<HashAndUTF8Characters, HashAndUTF8CharactersTranslator>(buffer);
500}
