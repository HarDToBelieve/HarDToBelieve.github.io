96template<typename CharacterType> static void base64EncodeInternal(Span<const uint8_t> inputDataBuffer, Span<CharacterType> destinationDataBuffer, Base64EncodePolicy policy, Base64EncodeMap map)
97{
98 ASSERT(destinationDataBuffer.size() > 0);
99 ASSERT(calculateBase64EncodedSize(inputDataBuffer.size(), policy) == destinationDataBuffer.size());
100
101 auto encodeMap = (map == Base64EncodeMap::URL) ? base64URLEncMap : base64EncMap;
102
103 unsigned sidx = 0;
104 unsigned didx = 0;
105 unsigned count = 0;
106
107 bool insertLFs = (policy == Base64EncodePolicy::InsertLFs && destinationDataBuffer.size() > maximumBase64LineLengthWhenInsertingLFs);
108
109 if (inputDataBuffer.size() > 1) {
110 while (sidx < inputDataBuffer.size() - 2) {
111 if (insertLFs) {
112 if (count && !(count % maximumBase64LineLengthWhenInsertingLFs))
113 destinationDataBuffer[didx++] = '\n';
114 count += 4;
115 }
116
117 destinationDataBuffer[didx++] = encodeMap[ (inputDataBuffer[sidx ] >> 2) & 077];
118 destinationDataBuffer[didx++] = encodeMap[((inputDataBuffer[sidx + 1] >> 4) & 017) | ((inputDataBuffer[sidx ] << 4) & 077)];
119 destinationDataBuffer[didx++] = encodeMap[((inputDataBuffer[sidx + 2] >> 6) & 003) | ((inputDataBuffer[sidx + 1] << 2) & 077)];
120 destinationDataBuffer[didx++] = encodeMap[ inputDataBuffer[sidx + 2] & 077];
121 sidx += 3;
122 }
123 }
124
125 if (sidx < inputDataBuffer.size()) {
126 if (insertLFs && (count > 0) && !(count % maximumBase64LineLengthWhenInsertingLFs))
127 destinationDataBuffer[didx++] = '\n';
128
129 destinationDataBuffer[didx++] = encodeMap[(inputDataBuffer[sidx] >> 2) & 077];
130 if (sidx < inputDataBuffer.size() - 1) {
131 destinationDataBuffer[didx++] = encodeMap[((inputDataBuffer[sidx + 1] >> 4) & 017) | ((inputDataBuffer[sidx] << 4) & 077)];
132 destinationDataBuffer[didx++] = encodeMap[ (inputDataBuffer[sidx + 1] << 2) & 077];
133 } else
134 destinationDataBuffer[didx++] = encodeMap[ (inputDataBuffer[sidx ] << 4) & 077];
135 }
136
137 ASSERT(policy != Base64EncodePolicy::URL || didx == destinationDataBuffer.size());
138
139 while (didx < destinationDataBuffer.size())
140 destinationDataBuffer[didx++] = '=';
141}

143template<typename CharacterType> static void base64EncodeInternal(Span<const std::byte> input, Span<CharacterType> destinationDataBuffer, Base64EncodePolicy policy, Base64EncodeMap map)
144{
145 base64EncodeInternal(Span { reinterpret_cast<const uint8_t*>(input.data()), input.size() }, destinationDataBuffer, policy, map);

148static Vector<uint8_t> base64EncodeInternal(Span<const std::byte> input, Base64EncodePolicy policy, Base64EncodeMap map)
149{
150 auto destinationLength = calculateBase64EncodedSize(input.size(), policy);
151 if (!destinationLength)
152 return { };
153
154 Vector<uint8_t> destinationVector(destinationLength);
155 base64EncodeInternal(input, Span { destinationVector }, policy, map);
156 return destinationVector;
157}

183template<typename T> static std::optional<Vector<uint8_t>> base64DecodeInternal(Span<const T> inputDataBuffer, OptionSet<Base64DecodeOptions> options, Base64DecodeMap map)
184{
185 if (!inputDataBuffer.size())
186 return Vector<uint8_t> { };
187
188 auto decodeMap = (map == Base64DecodeMap::URL) ? base64URLDecMap : base64DecMap;
189
190 Vector<uint8_t> destination(inputDataBuffer.size());
191
192 unsigned equalsSignCount = 0;
193 unsigned destinationLength = 0;
194 for (unsigned idx = 0; idx < inputDataBuffer.size(); ++idx) {
195 unsigned ch = inputDataBuffer[idx];
196 if (ch == '=') {
197 ++equalsSignCount;
198 // There should never be more than 2 padding characters.
199 if (options.contains(Base64DecodeOptions::ValidatePadding) && equalsSignCount > 2) {
200 return std::nullopt;
201 }
202 } else {
203 char decodedCharacter = ch < decodeMapSize ? decodeMap[ch] : nonAlphabet;
204 if (decodedCharacter != nonAlphabet) {
205 if (equalsSignCount)
206 return std::nullopt;
207 destination[destinationLength++] = decodedCharacter;
208 } else if (!options.contains(Base64DecodeOptions::IgnoreSpacesAndNewLines) || (!isLatin1(ch) || !isASCIISpace(ch) || (options.contains(Base64DecodeOptions::DiscardVerticalTab) && ch == '\v'))) {
209 return std::nullopt;
210 }
211 }
212 }
213
214 // Make sure we shrink back the Vector before returning. destinationLength may be shorter than expected
215 // in case of error or in case of ignored spaces.
216 if (destinationLength < destination.size())
217 destination.shrink(destinationLength);
218
219 if (!destinationLength) {
220 if (equalsSignCount)
221 return std::nullopt;
222 return Vector<uint8_t> { };
223 }
224
225 // The should be no padding if length is a multiple of 4.
226 // We use (destinationLength + equalsSignCount) instead of length because we don't want to account for ignored characters (i.e. spaces).
227 if (options.contains(Base64DecodeOptions::ValidatePadding) && equalsSignCount && (destinationLength + equalsSignCount) % 4)
228 return std::nullopt;
229
230 // Valid data is (n * 4 + [0,2,3]) characters long.
231 if ((destinationLength % 4) == 1)
232 return std::nullopt;
233
234 // 4-byte to 3-byte conversion
235 destinationLength -= (destinationLength + 3) / 4;
236 if (!destinationLength)
237 return std::nullopt;
238
239 unsigned sidx = 0;
240 unsigned didx = 0;
241 if (destinationLength > 1) {
242 while (didx < destinationLength - 2) {
243 destination[didx ] = (((destination[sidx ] << 2) & 255) | ((destination[sidx + 1] >> 4) & 003));
244 destination[didx + 1] = (((destination[sidx + 1] << 4) & 255) | ((destination[sidx + 2] >> 2) & 017));
245 destination[didx + 2] = (((destination[sidx + 2] << 6) & 255) | (destination[sidx + 3] & 077));
246 sidx += 4;
247 didx += 3;
248 }
249 }
250
251 if (didx < destinationLength)
252 destination[didx] = (((destination[sidx ] << 2) & 255) | ((destination[sidx + 1] >> 4) & 003));
253
254 if (++didx < destinationLength)
255 destination[didx] = (((destination[sidx + 1] << 4) & 255) | ((destination[sidx + 2] >> 2) & 017));
256
257 if (destinationLength < destination.size())
258 destination.shrink(destinationLength);
259 return destination;
260}

262std::optional<Vector<uint8_t>> base64Decode(Span<const std::byte> input, OptionSet<Base64DecodeOptions> options, Base64DecodeMap map)
263{
264 if (input.size() > std::numeric_limits<unsigned>::max())
265 return std::nullopt;
266 return base64DecodeInternal(Span { reinterpret_cast<const uint8_t*>(input.data()), input.size() }, options, map);
267}

269std::optional<Vector<uint8_t>> base64Decode(StringView input, OptionSet<Base64DecodeOptions> options, Base64DecodeMap map)
270{
271 unsigned length = input.length();
272 if (!length || input.is8Bit())
273 return base64DecodeInternal(Span { input.characters8(), length }, options, map);
274 return base64DecodeInternal(Span { input.characters16(), length }, options, map);
275}
