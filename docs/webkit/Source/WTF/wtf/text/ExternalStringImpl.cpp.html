31WTF_EXPORT_PRIVATE Ref<ExternalStringImpl> ExternalStringImpl::create(const LChar* characters, unsigned length, ExternalStringImplFreeFunction&& free)
32{
33 return adoptRef(*new ExternalStringImpl(characters, length, WTFMove(free)));
34}
35
36WTF_EXPORT_PRIVATE Ref<ExternalStringImpl> ExternalStringImpl::create(const UChar* characters, unsigned length, ExternalStringImplFreeFunction&& free)
37{
38 return adoptRef(*new ExternalStringImpl(characters, length, WTFMove(free)));
39}
40
41ExternalStringImpl::ExternalStringImpl(const LChar* characters, unsigned length, ExternalStringImplFreeFunction&& free)
42 : StringImpl(characters, length, ConstructWithoutCopying)
43 , m_free(WTFMove(free))
44{
45 ASSERT(m_free);
46 m_hashAndFlags = (m_hashAndFlags & ~s_hashMaskBufferOwnership) | BufferExternal;
47}
48
49ExternalStringImpl::ExternalStringImpl(const UChar* characters, unsigned length, ExternalStringImplFreeFunction&& free)
50 : StringImpl(characters, length, ConstructWithoutCopying)
51 , m_free(WTFMove(free))
52{
53 ASSERT(m_free);
54 m_hashAndFlags = (m_hashAndFlags & ~s_hashMaskBufferOwnership) | BufferExternal;
55}
