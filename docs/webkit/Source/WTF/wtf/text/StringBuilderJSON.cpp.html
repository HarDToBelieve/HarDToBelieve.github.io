56template<typename OutputCharacterType, typename InputCharacterType>
57ALWAYS_INLINE static void appendQuotedJSONStringInternal(OutputCharacterType*& output, const InputCharacterType* input, unsigned length)
58{
59 for (auto* end = input + length; input != end; ++input) {
60 auto character = *input;
61 if (LIKELY(character <= 0xFF)) {
62 auto escaped = escapedFormsForJSON[character];
63 if (LIKELY(!escaped)) {
64 *output++ = character;
65 continue;
66 }
67
68 *output++ = '\\';
69 *output++ = escaped;
70 if (UNLIKELY(escaped == 'u')) {
71 *output++ = '0';
72 *output++ = '0';
73 *output++ = upperNibbleToLowercaseASCIIHexDigit(character);
74 *output++ = lowerNibbleToLowercaseASCIIHexDigit(character);
75 }
76 continue;
77 }
78
79 if (LIKELY(!U16_IS_SURROGATE(character))) {
80 *output++ = character;
81 continue;
82 }
83
84 auto next = input + 1;
85 bool isValidSurrogatePair = U16_IS_SURROGATE_LEAD(character) && next != end && U16_IS_TRAIL(*next);
86 if (isValidSurrogatePair) {
87 *output++ = character;
88 *output++ = *next;
89 ++input;
90 continue;
91 }
92
93 uint8_t upper = static_cast<uint32_t>(character) >> 8;
94 uint8_t lower = static_cast<uint8_t>(character);
95 *output++ = '\\';
96 *output++ = 'u';
97 *output++ = upperNibbleToLowercaseASCIIHexDigit(upper);
98 *output++ = lowerNibbleToLowercaseASCIIHexDigit(upper);
99 *output++ = upperNibbleToLowercaseASCIIHexDigit(lower);
100 *output++ = lowerNibbleToLowercaseASCIIHexDigit(lower);
101 continue;
102 }
103}

105void StringBuilder::appendQuotedJSONString(const String& string)
106{
107 if (hasOverflowed())
108 return;
109
110 // Make sure we have enough buffer space to append this string for worst case without reallocating.
111 // The 2 is for the '"' quotes on each end.
112 // The 6 is the worst case for a single code unit that could be encoded as \uNNNN.
113 CheckedUint32 stringLength = string.length();
114 auto maximumCapacityRequired = m_length + 2 + stringLength * 6;
115 if (maximumCapacityRequired.hasOverflowed()) {
116 didOverflow();
117 return;
118 }
119
120 // We need to check maximum length before calling roundUpPowerOfTwo because that function returns 0 for values in the range [2^31, 2^32-2].
121 // FIXME: Instead, roundUpToPowerOfTwo should be fixed to do something more useful in those cases, perhaps using checked or saturated arithmetic.
122 // https://bugs.webkit.org/show_bug.cgi?id=176086
123 auto allocationSize = maximumCapacityRequired.value();
124 if (allocationSize > String::MaxLength) {
125 didOverflow();
126 return;
127 }
128 allocationSize = roundUpToPowerOfTwo(allocationSize);
129 if (allocationSize > String::MaxLength) {
130 didOverflow();
131 return;
132 }
133
134 // FIXME: Consider switching to extendBufferForAppending/shrink instead to share more code with the rest of StringBuilder.
135 if (is8Bit() && !string.is8Bit())
136 allocateBuffer<UChar>(characters<LChar>(), allocationSize);
137 else
138 reserveCapacity(allocationSize);
139 if (UNLIKELY(hasOverflowed()))
140 return;
141
142 if (m_buffer->is8Bit()) {
143 auto characters = const_cast<LChar*>(m_buffer->characters<LChar>());
144 auto output = characters + m_length;
145 *output++ = '"';
146 appendQuotedJSONStringInternal(output, string.characters8(), string.length());
147 *output++ = '"';
148 m_length = output - characters;
149 } else {
150 auto characters = const_cast<UChar*>(m_buffer->characters<UChar>());
151 auto output = characters + m_length;
152 *output++ = '"';
153 if (string.is8Bit())
154 appendQuotedJSONStringInternal(output, string.characters8(), string.length());
155 else
156 appendQuotedJSONStringInternal(output, string.characters16(), string.length());
157 *output++ = '"';
158 m_length = output - characters;
159 }
160 ASSERT(m_buffer->length() >= m_length);
161}
