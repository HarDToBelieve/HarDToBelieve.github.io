46static Variant<TextBreakIteratorICU, TextBreakIteratorPlatform> mapModeToBackingIterator(StringView string, TextBreakIterator::Mode mode, const AtomString& locale)
47{
48 switch (mode) {
49 case TextBreakIterator::Mode::Line:
50 return TextBreakIteratorICU(string, TextBreakIteratorICU::Mode::Line, locale.string().utf8().data());
51 case TextBreakIterator::Mode::Caret:
52 return TextBreakIteratorICU(string, TextBreakIteratorICU::Mode::Character, locale.string().utf8().data());
53 case TextBreakIterator::Mode::Delete:
54 return TextBreakIteratorICU(string, TextBreakIteratorICU::Mode::Character, locale.string().utf8().data());
55 default:
56 ASSERT_NOT_REACHED();
57 return TextBreakIteratorICU(string, TextBreakIteratorICU::Mode::Character, locale.string().utf8().data());
58 }
59}

72static UBreakIterator* initializeIterator(UBreakIteratorType type, const char* locale = currentTextBreakLocaleID())
73{
74 UErrorCode openStatus = U_ZERO_ERROR;
75 UBreakIterator* iterator = ubrk_open(type, locale, nullptr, 0, &openStatus);
76 ASSERT_WITH_MESSAGE(U_SUCCESS(openStatus), "ICU could not open a break iterator: %s (%d)", u_errorName(openStatus), openStatus);
77 return iterator;
78}

80// Iterator text setting
81
82static UBreakIterator* setTextForIterator(UBreakIterator& iterator, StringView string)
83{
84 if (string.is8Bit()) {
85 UTextWithBuffer textLocal;
86 textLocal.text = UTEXT_INITIALIZER;
87 textLocal.text.extraSize = sizeof(textLocal.buffer);
88 textLocal.text.pExtra = textLocal.buffer;
89
90 UErrorCode openStatus = U_ZERO_ERROR;
91 UText* text = openLatin1UTextProvider(&textLocal, string.characters8(), string.length(), &openStatus);
92 if (U_FAILURE(openStatus)) {
93 LOG_ERROR("uTextOpenLatin1 failed with status %d", openStatus);
94 return nullptr;
95 }
96
97 UErrorCode setTextStatus = U_ZERO_ERROR;
98 ubrk_setUText(&iterator, text, &setTextStatus);
99 if (U_FAILURE(setTextStatus)) {
100 LOG_ERROR("ubrk_setUText failed with status %d", setTextStatus);
101 return nullptr;
102 }
103
104 utext_close(text);
105 } else {
106 UErrorCode setTextStatus = U_ZERO_ERROR;
107 ubrk_setText(&iterator, string.characters16(), string.length(), &setTextStatus);
108 if (U_FAILURE(setTextStatus))
109 return nullptr;
110 }
111
112 return &iterator;
113}

115static UBreakIterator* setContextAwareTextForIterator(UBreakIterator& iterator, StringView string, const UChar* priorContext, unsigned priorContextLength)
116{
117 if (string.is8Bit()) {
118 UTextWithBuffer textLocal;
119 textLocal.text = UTEXT_INITIALIZER;
120 textLocal.text.extraSize = sizeof(textLocal.buffer);
121 textLocal.text.pExtra = textLocal.buffer;
122
123 UErrorCode openStatus = U_ZERO_ERROR;
124 UText* text = openLatin1ContextAwareUTextProvider(&textLocal, string.characters8(), string.length(), priorContext, priorContextLength, &openStatus);
125 if (U_FAILURE(openStatus)) {
126 LOG_ERROR("openLatin1ContextAwareUTextProvider failed with status %d", openStatus);
127 return nullptr;
128 }
129
130 UErrorCode setTextStatus = U_ZERO_ERROR;
131 ubrk_setUText(&iterator, text, &setTextStatus);
132 if (U_FAILURE(setTextStatus)) {
133 LOG_ERROR("ubrk_setUText failed with status %d", setTextStatus);
134 return nullptr;
135 }
136
137 utext_close(text);
138 } else {
139 UText textLocal = UTEXT_INITIALIZER;
140
141 UErrorCode openStatus = U_ZERO_ERROR;
142 UText* text = openUTF16ContextAwareUTextProvider(&textLocal, string.characters16(), string.length(), priorContext, priorContextLength, &openStatus);
143 if (U_FAILURE(openStatus)) {
144 LOG_ERROR("openUTF16ContextAwareUTextProvider failed with status %d", openStatus);
145 return nullptr;
146 }
147
148 UErrorCode setTextStatus = U_ZERO_ERROR;
149 ubrk_setUText(&iterator, text, &setTextStatus);
150 if (U_FAILURE(setTextStatus)) {
151 LOG_ERROR("ubrk_setUText failed with status %d", setTextStatus);
152 return nullptr;
153 }
154
155 utext_close(text);
156 }
157
158 return &iterator;
159}

182UBreakIterator* acquireLineBreakIterator(StringView string, const AtomString& locale, const UChar* priorContext, unsigned priorContextLength, LineBreakIteratorMode mode)
183{
184 UBreakIterator* iterator = LineBreakIteratorPool::sharedPool().take(locale, mode);

198UBreakIterator* openLineBreakIterator(const AtomString& locale)
199{
200 bool localeIsEmpty = locale.isEmpty();
201 UErrorCode openStatus = U_ZERO_ERROR;
202 UBreakIterator* ubrkIter = ubrk_open(UBRK_LINE, localeIsEmpty ? currentTextBreakLocaleID() : locale.string().utf8().data(), nullptr, 0, &openStatus);
203 // locale comes from a web page and it can be invalid, leading ICU
204 // to fail, in which case we fall back to the default locale.
205 if (!localeIsEmpty && U_FAILURE(openStatus)) {
206 openStatus = U_ZERO_ERROR;
207 ubrkIter = ubrk_open(UBRK_LINE, currentTextBreakLocaleID(), nullptr, 0, &openStatus);
208 }
209
210 if (U_FAILURE(openStatus)) {
211 LOG_ERROR("ubrk_open failed with status %d", openStatus);
212 return nullptr;
213 }
214
215 return ubrkIter;
216}

218void closeLineBreakIterator(UBreakIterator*& iterator)
219{
220 UBreakIterator* ubrkIter = iterator;
221 ASSERT(ubrkIter);
222 ubrk_close(ubrkIter);
223 iterator = nullptr;
224}

226static std::atomic<UBreakIterator*> nonSharedCharacterBreakIterator = ATOMIC_VAR_INIT(nullptr);
227
228static inline UBreakIterator* getNonSharedCharacterBreakIterator()
229{
230 if (auto *res = nonSharedCharacterBreakIterator.exchange(nullptr, std::memory_order_acquire))
231 return res;
232 return initializeIterator(UBRK_CHARACTER);
233}
234
235static inline void cacheNonSharedCharacterBreakIterator(UBreakIterator* cacheMe)
236{
237 if (auto *old = nonSharedCharacterBreakIterator.exchange(cacheMe, std::memory_order_release))
238 ubrk_close(old);
239}

241NonSharedCharacterBreakIterator::NonSharedCharacterBreakIterator(StringView string)
242{
243 if ((m_iterator = getNonSharedCharacterBreakIterator()))
244 m_iterator = setTextForIterator(*m_iterator, string);
245}
246
247NonSharedCharacterBreakIterator::~NonSharedCharacterBreakIterator()
248{
249 if (m_iterator)
250 cacheNonSharedCharacterBreakIterator(m_iterator);
251}
252
253NonSharedCharacterBreakIterator::NonSharedCharacterBreakIterator(NonSharedCharacterBreakIterator&& other)
254 : m_iterator(nullptr)
255{
256 std::swap(m_iterator, other.m_iterator);
257}

267unsigned numGraphemeClusters(StringView string)
268{
269 unsigned stringLength = string.length();
270
271 if (!stringLength)
272 return 0;
273
274 // The only Latin-1 Extended Grapheme Cluster is CRLF.
275 if (string.is8Bit()) {
276 auto* characters = string.characters8();
277 unsigned numCRLF = 0;
278 for (unsigned i = 1; i < stringLength; ++i)
279 numCRLF += characters[i - 1] == '\r' && characters[i] == '\n';
280 return stringLength - numCRLF;
281 }
282
283 NonSharedCharacterBreakIterator iterator { string };
284 if (!iterator) {
285 ASSERT_NOT_REACHED();
286 return stringLength;
287 }
288
289 unsigned numGraphemeClusters = 0;
290 while (ubrk_next(iterator) != UBRK_DONE)
291 ++numGraphemeClusters;
292 return numGraphemeClusters;
293}
294
295unsigned numCodeUnitsInGraphemeClusters(StringView string, unsigned numGraphemeClusters)
296{
297 unsigned stringLength = string.length();
298
299 if (stringLength <= numGraphemeClusters)
300 return stringLength;
301
302 // The only Latin-1 Extended Grapheme Cluster is CRLF.
303 if (string.is8Bit()) {
304 auto* characters = string.characters8();
305 unsigned i, j;
306 for (i = 0, j = 0; i < numGraphemeClusters && j + 1 < stringLength; ++i, ++j)
307 j += characters[j] == '\r' && characters[j + 1] == '\n';
308 return j + (i < numGraphemeClusters);
309 }
310
311 NonSharedCharacterBreakIterator iterator { string };
312 if (!iterator) {
313 ASSERT_NOT_REACHED();
314 return stringLength;
315 }
316
317 for (unsigned i = 0; i < numGraphemeClusters; ++i) {
318 if (ubrk_next(iterator) == UBRK_DONE)
319 return stringLength;
320 }
321 return ubrk_current(iterator);
322}
