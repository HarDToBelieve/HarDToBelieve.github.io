33static constexpr size_t printBufferSize = 100; // large enough for any integer or floating point value in string format, including trailing null character
34
35static inline bool hasFractions(double val)
36{
37 static constexpr double s_epsilon = 0.0001;
38 int ival = static_cast<int>(val);
39 double dval = static_cast<double>(ival);
40 return fabs(val - dval) > s_epsilon;
41}

90TextStream& TextStream::operator<<(float f)
91{
92 if (m_formattingFlags & Formatting::NumberRespectingIntegers)
93 return *this << FormatNumberRespectingIntegers(f);
94
95 m_text.append(FormattedNumber::fixedWidth(f, 2));
96 return *this;
97}

139TextStream& TextStream::operator<<(const FormatNumberRespectingIntegers& numberToFormat)
140{
141 if (hasFractions(numberToFormat.value)) {
142 m_text.append(FormattedNumber::fixedWidth(numberToFormat.value, 2));
143 return *this;
144 }
145
146 m_text.append(static_cast<int>(numberToFormat.value));
147 return *this;
148}

150String TextStream::release()
151{
152 String result = m_text.toString();
153 m_text.clear();
154 return result;
155}

157void TextStream::startGroup()
158{
159 TextStream& ts = *this;
160
161 if (m_multiLineMode) {
162 ts << "\n";
163 ts.writeIndent();
164 ts << "(";
165 ts.increaseIndent();
166 } else
167 ts << " (";
168}
169
170void TextStream::endGroup()
171{
172 TextStream& ts = *this;
173 ts << ")";
174 if (m_multiLineMode)
175 ts.decreaseIndent();
176}
177
178void TextStream::nextLine()
179{
180 TextStream& ts = *this;
181 if (m_multiLineMode) {
182 ts << "\n";
183 ts.writeIndent();
184 } else
185 ts << " ";
186}
187
188void TextStream::writeIndent()
189{
190 if (m_multiLineMode)
191 WTF::writeIndent(*this, m_indent);
192}
193
194void writeIndent(TextStream& ts, int indent)
195{
196 for (int i = 0; i < indent; ++i)
197 ts << " ";
198}
