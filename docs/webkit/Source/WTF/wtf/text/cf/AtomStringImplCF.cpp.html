34namespace WTF {
35
36RefPtr<AtomStringImpl> AtomStringImpl::add(CFStringRef string)
37{
38 if (!string)
39 return nullptr;
40
41 CFIndex length = CFStringGetLength(string);
42
43 if (const LChar* ptr = reinterpret_cast<const LChar*>(CFStringGetCStringPtr(string, kCFStringEncodingISOLatin1)))
44 return add(ptr, length);
45
46 if (const UniChar* ptr = CFStringGetCharactersPtr(string))
47 return add(reinterpret_cast<const UChar*>(ptr), length);
48
49 Vector<UniChar, 1024> ucharBuffer(length);
50 CFStringGetCharacters(string, CFRangeMake(0, length), ucharBuffer.data());
51 return add(reinterpret_cast<const UChar*>(ucharBuffer.data()), length);
52}
