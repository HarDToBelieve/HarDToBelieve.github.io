29namespace WTF {
30
31void BinarySemaphore::signal()
32{
33 Locker locker { m_lock };
34 m_isSet = true;
35 m_condition.notifyOne();
36}
37
38bool BinarySemaphore::waitUntil(const TimeWithDynamicClockType& absoluteTime)
39{
40 Locker locker { m_lock };
41 bool satisfied = m_condition.waitUntil(m_lock, absoluteTime, [&] {
42 assertIsHeld(m_lock);
43 return m_isSet;
44 });
45 if (satisfied)
46 m_isSet = false;
47 return satisfied;
48}
