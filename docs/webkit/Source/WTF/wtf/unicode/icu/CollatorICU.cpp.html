62static inline char* copyShortASCIIString(CFStringRef string)
63{
64 // OK to have a fixed size buffer and to only handle ASCII since we only use this for locale names.
65 char buffer[256];
66 if (!string || !CFStringGetCString(string, buffer, sizeof(buffer), kCFStringEncodingASCII))
67 return strdup("");
68 return strdup(buffer);
69}

71static char* copyDefaultLocale()
72{
73#if !PLATFORM(IOS_FAMILY)
74 return copyShortASCIIString(static_cast<CFStringRef>(CFLocaleGetValue(adoptCF(CFLocaleCopyCurrent()).get(), kCFLocaleCollatorIdentifier)));
75#else
76 // FIXME: Documentation claims the code above would work on iOS 4.0 and later. After test that works, we should remove this and use that instead.
77 return copyShortASCIIString(adoptCF(static_cast<CFStringRef>(CFPreferencesCopyValue(CFSTR("AppleCollationOrder"), kCFPreferencesAnyApplication, kCFPreferencesCurrentUser, kCFPreferencesAnyHost))).get());
78#endif
79}

81static inline const char* resolveDefaultLocale(const char* locale)
82{
83 if (locale)
84 return locale;
85 // Since iOS and OS X don't set UNIX locale to match the user's selected locale, the ICU default locale is not the right one.
86 // So, instead of passing null to ICU, we pass the name of the user's selected locale.
87 static char* defaultLocale;
88 static std::once_flag initializeDefaultLocaleOnce;
89 std::call_once(initializeDefaultLocaleOnce, []{
90 defaultLocale = copyDefaultLocale();
91 });
92 return defaultLocale;
93}

97static inline bool localesMatch(const char* a, const char* b)
98{
99 // Two null locales are equal, other locales are compared with strcmp.
100 return a == b || (a && b && !strcmp(a, b));
101}
102
103Collator::Collator(const char* locale, bool shouldSortLowercaseFirst)
104{
105 UErrorCode status = U_ZERO_ERROR;
106
107 {
108 Locker locker { cachedCollatorLock };
109 if (cachedCollator && localesMatch(cachedCollatorLocale, locale) && cachedCollatorShouldSortLowercaseFirst == shouldSortLowercaseFirst) {
110 m_collator = cachedCollator;
111 m_locale = cachedCollatorLocale;
112 m_shouldSortLowercaseFirst = shouldSortLowercaseFirst;
113 cachedCollator = nullptr;
114 cachedCollatorLocale = nullptr;
115 return;
116 }
117 }
118
119 m_collator = ucol_open(resolveDefaultLocale(locale), &status);
120 if (U_FAILURE(status)) {
121 status = U_ZERO_ERROR;
122 m_collator = ucol_open("", &status); // Fall back to Unicode Collation Algorithm.
123 }
124 ASSERT(U_SUCCESS(status));
125
126 ucol_setAttribute(m_collator, UCOL_CASE_FIRST, shouldSortLowercaseFirst ? UCOL_LOWER_FIRST : UCOL_UPPER_FIRST, &status);
127 ASSERT(U_SUCCESS(status));
128
129 ucol_setAttribute(m_collator, UCOL_NORMALIZATION_MODE, UCOL_ON, &status);
130 ASSERT(U_SUCCESS(status));
131
132 m_locale = locale ? fastStrDup(locale) : nullptr;
133 m_shouldSortLowercaseFirst = shouldSortLowercaseFirst;
134}
135
136Collator::~Collator()
137{
138 Locker locker { cachedCollatorLock };
139 if (cachedCollator) {
140 ucol_close(cachedCollator);
141 fastFree(cachedCollatorLocale);
142 }
143 cachedCollator = m_collator;
144 cachedCollatorLocale = m_locale;
145 cachedCollatorShouldSortLowercaseFirst = m_shouldSortLowercaseFirst;
146}

148static int32_t getIndexLatin1(UCharIterator* iterator, UCharIteratorOrigin origin)
149{
150 switch (origin) {
151 case UITER_START:
152 return iterator->start;
153 case UITER_CURRENT:
154 return iterator->index;
155 case UITER_LIMIT:
156 return iterator->limit;
157 case UITER_ZERO:
158 return 0;
159 case UITER_LENGTH:
160 return iterator->length;
161 }
162 ASSERT_NOT_REACHED();
163 return U_SENTINEL;
164}
165
166static int32_t moveLatin1(UCharIterator* iterator, int32_t delta, UCharIteratorOrigin origin)
167{
168 return iterator->index = getIndexLatin1(iterator, origin) + delta;
169}
170
171static UBool hasNextLatin1(UCharIterator* iterator)
172{
173 return iterator->index < iterator->limit;
174}
175
176static UBool hasPreviousLatin1(UCharIterator* iterator)
177{
178 return iterator->index > iterator->start;
179}
180
181static UChar32 currentLatin1(UCharIterator* iterator)
182{
183 ASSERT(iterator->index >= iterator->start);
184 if (iterator->index >= iterator->limit)
185 return U_SENTINEL;
186 return static_cast<const LChar*>(iterator->context)[iterator->index];
187}
188
189static UChar32 nextLatin1(UCharIterator* iterator)
190{
191 ASSERT(iterator->index >= iterator->start);
192 if (iterator->index >= iterator->limit)
193 return U_SENTINEL;
194 return static_cast<const LChar*>(iterator->context)[iterator->index++];
195}
196
197static UChar32 previousLatin1(UCharIterator* iterator)
198{
199 if (iterator->index <= iterator->start)
200 return U_SENTINEL;
201 return static_cast<const LChar*>(iterator->context)[--iterator->index];
202}
203
204static uint32_t getStateLatin1(const UCharIterator* iterator)
205{
206 return iterator->index;
207}
208
209static void setStateLatin1(UCharIterator* iterator, uint32_t state, UErrorCode*)
210{
211 iterator->index = state;
212}

214static UCharIterator createLatin1Iterator(const LChar* characters, int length)
215{
216 UCharIterator iterator;
217 iterator.context = characters;
218 iterator.length = length;
219 iterator.start = 0;
220 iterator.index = 0;
221 iterator.limit = length;
222 iterator.reservedField = 0;
223 iterator.getIndex = getIndexLatin1;
224 iterator.move = moveLatin1;
225 iterator.hasNext = hasNextLatin1;
226 iterator.hasPrevious = hasPreviousLatin1;
227 iterator.current = currentLatin1;
228 iterator.next = nextLatin1;
229 iterator.previous = previousLatin1;
230 iterator.reservedFn = nullptr;
231 iterator.getState = getStateLatin1;
232 iterator.setState = setStateLatin1;
233 return iterator;
234}
235
236UCharIterator createIterator(StringView string)
237{
238 if (string.is8Bit())
239 return createLatin1Iterator(string.characters8(), string.length());
240 UCharIterator iterator;
241 uiter_setString(&iterator, string.characters16(), string.length());
242 return iterator;
243}
244
245int Collator::collate(StringView a, StringView b) const
246{
247 UCharIterator iteratorA = createIterator(a);
248 UCharIterator iteratorB = createIterator(b);
249 UErrorCode status = U_ZERO_ERROR;
250 int result = ucol_strcollIter(m_collator, &iteratorA, &iteratorB, &status);
251 ASSERT(U_SUCCESS(status));
252 return result;
253}
254
255static UCharIterator createIteratorUTF8(const char* string)
256{
257 UCharIterator iterator;
258 uiter_setUTF8(&iterator, string, strlen(string));
259 return iterator;
260}
261
262int Collator::collateUTF8(const char* a, const char* b) const
263{
264 UCharIterator iteratorA = createIteratorUTF8(a);
265 UCharIterator iteratorB = createIteratorUTF8(b);
266 UErrorCode status = U_ZERO_ERROR;
267 int result = ucol_strcollIter(m_collator, &iteratorA, &iteratorB, &status);
268 ASSERT(U_SUCCESS(status));
269 return result;
270}
