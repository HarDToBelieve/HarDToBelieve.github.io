45struct ClientState {
46 WTF_MAKE_STRUCT_FAST_ALLOCATED;
47
48 explicit ClientState(WebMediaSessionManagerClient& client, PlaybackTargetClientContextIdentifier contextId)
49 : client(client)
50 , contextId(contextId)
51 {
52 }
53
54 bool operator == (ClientState const& other) const
55 {
56 return contextId == other.contextId && &client == &other.client;
57 }
58
59 WebMediaSessionManagerClient& client;
60 PlaybackTargetClientContextIdentifier contextId;
61 WebCore::MediaProducer::MediaStateFlags flags;
62 bool requestedPicker { false };
63 bool previouslyRequestedPicker { false };
64 bool configurationRequired { true };
65 bool playedToEnd { false };
66};

68static bool flagsAreSet(MediaProducer::MediaStateFlags value, MediaProducer::MediaStateFlags flags)
69{
70 return value.containsAny(flags);
71}

73String mediaProducerStateString(MediaProducer::MediaStateFlags flags)
74{
75 StringBuilder string;
76 string.append(" { ");
77 if (flags & MediaProducer::MediaState::IsPlayingAudio)
78 string.append("IsPlayingAudio+");
79 if (flags & MediaProducer::MediaState::IsPlayingVideo)
80 string.append("IsPlayingVideo+");
81 if (flags & MediaProducer::MediaState::IsPlayingToExternalDevice)
82 string.append("IsPlayingToExternalDevice+");
83 if (flags & MediaProducer::MediaState::HasPlaybackTargetAvailabilityListener)
84 string.append("HasTargetAvailabilityListener+");
85 if (flags & MediaProducer::MediaState::RequiresPlaybackTargetMonitoring)
86 string.append("RequiresTargetMonitoring+");
87 if (flags & MediaProducer::MediaState::ExternalDeviceAutoPlayCandidate)
88 string.append("ExternalDeviceAutoPlayCandidate+");
89 if (flags & MediaProducer::MediaState::DidPlayToEnd)
90 string.append("DidPlayToEnd+");
91 if (flags & MediaProducer::MediaState::HasAudioOrVideo)
92 string.append("HasAudioOrVideo+");
93 if (string.isEmpty())
94 string.append("IsNotPlaying");
95 else
96 string.shrink(string.length() - 1);
97 string.append(" }");
98 return string.toString();
99}

111 template<typename... Arguments>
112 inline void logAlways(const char* methodName, ClientState* state, const Arguments&... arguments) const
113 {
114 if (!state->client.alwaysOnLoggingAllowed())
115 return;
116
117 m_logger->logAlways(LogMedia, makeString("WebMediaSessionManager::", methodName, ' '), state->contextId.toUInt64(), state->flags, arguments...);
118 }

120 template<typename... Arguments>
121 inline void logAlways(const char* methodName, const Arguments&... arguments) const
122 {
123 if (!m_manager.alwaysOnLoggingAllowed())
124 return;
125
126 m_logger->logAlways(LogMedia, makeString("WebMediaSessionManager::", methodName, ' '), arguments...);
127 }

201PlaybackTargetClientContextIdentifier WebMediaSessionManager::addPlaybackTargetPickerClient(WebMediaSessionManagerClient& client, PlaybackTargetClientContextIdentifier contextId)
202{
203 size_t index = find(&client, contextId);
204 ASSERT(index == notFound);
205 if (index != notFound)
206 return { };
207
208 ALWAYS_LOG_MEDIASESSIONMANAGER(__func__, contextId.toUInt64());
209 m_clientState.append(makeUnique<ClientState>(client, contextId));

211 if (m_externalOutputDeviceAvailable || m_playbackTarget)

244void WebMediaSessionManager::showPlaybackTargetPicker(WebMediaSessionManagerClient& client, PlaybackTargetClientContextIdentifier contextId, const IntRect& rect, bool, bool useDarkAppearance)
245{
246 size_t index = find(&client, contextId);
247 ASSERT(index != notFound);
248 if (index == notFound)
249 return;
250
251 auto& clientRequestingPicker = m_clientState[index];
252 for (auto& state : m_clientState) {
253 state->requestedPicker = state == clientRequestingPicker;
254 state->previouslyRequestedPicker = state == clientRequestingPicker;
255 }
256
257 ALWAYS_LOG_MEDIASESSIONMANAGER(__func__, m_clientState[index].get());
258
259 bool hasActiveRoute = flagsAreSet(m_clientState[index]->flags, MediaProducer::MediaState::IsPlayingToExternalDevice);
260 targetPicker().showPlaybackTargetPicker(client.platformView(), FloatRect(rect), hasActiveRoute, useDarkAppearance, client.useiTunesAVOutputContext());
261}

263void WebMediaSessionManager::clientStateDidChange(WebMediaSessionManagerClient& client, PlaybackTargetClientContextIdentifier contextId, MediaProducer::MediaStateFlags newFlags)
264{
265 size_t index = find(&client, contextId);
266 ASSERT(index != notFound);
267 if (index == notFound)
268 return;
269
270 auto& changedClientState = m_clientState[index];
271 MediaProducer::MediaStateFlags oldFlags = changedClientState->flags;
272 if (newFlags == oldFlags)
273 return;
274
275 ALWAYS_LOG_MEDIASESSIONMANAGER(__func__, m_clientState[index].get(), "new flags = ", newFlags);
276
277 changedClientState->flags = newFlags;
278
279 constexpr MediaProducer::MediaStateFlags updateConfigurationFlags { MediaProducer::MediaState::RequiresPlaybackTargetMonitoring, MediaProducer::MediaState::HasPlaybackTargetAvailabilityListener, MediaProducer::MediaState::HasAudioOrVideo };
280 if ((oldFlags & updateConfigurationFlags) != (newFlags & updateConfigurationFlags))
281 scheduleDelayedTask(TargetMonitoringConfigurationTask);
282
283 constexpr MediaProducer::MediaStateFlags playingToTargetFlags { MediaProducer::MediaState::IsPlayingToExternalDevice, MediaProducer::MediaState::IsPlayingVideo };
284 if ((oldFlags & playingToTargetFlags) != (newFlags & playingToTargetFlags)) {
285 if (flagsAreSet(oldFlags, MediaProducer::MediaState::IsPlayingVideo) && !flagsAreSet(newFlags, MediaProducer::MediaState::IsPlayingVideo) && flagsAreSet(newFlags, MediaProducer::MediaState::DidPlayToEnd))
286 changedClientState->playedToEnd = true;
287 scheduleDelayedTask(WatchdogTimerConfigurationTask);
288 }
289
290 if (!m_playbackTarget || !m_playbackTarget->hasActiveRoute() || !flagsAreSet(newFlags, MediaProducer::MediaState::ExternalDeviceAutoPlayCandidate))
291 return;
292
293 // Do not interrupt another element already playing to a device.
294 for (auto& state : m_clientState) {
295 if (state == changedClientState)
296 continue;
297
298 if (flagsAreSet(state->flags, MediaProducer::MediaState::IsPlayingToExternalDevice) && flagsAreSet(state->flags, MediaProducer::MediaState::IsPlayingVideo)) {
299 ALWAYS_LOG_MEDIASESSIONMANAGER(__func__, state.get(), " returning early");
300 return;
301 }
302 }
303
304 // Do not begin playing to the device unless playback has just started.
305 if (!flagsAreSet(newFlags, MediaProducer::MediaState::IsPlayingVideo) || flagsAreSet(oldFlags, MediaProducer::MediaState::IsPlayingVideo)) {
306 ALWAYS_LOG_MEDIASESSIONMANAGER(__func__, "returning early, playback didn't just start");
307 return;
308 }
309
310 for (auto& state : m_clientState) {
311 if (state == changedClientState)
312 continue;
313 ALWAYS_LOG_MEDIASESSIONMANAGER(__func__, state.get(), " calling setShouldPlayToPlaybackTarget(false)");
314 state->client.setShouldPlayToPlaybackTarget(state->contextId, false);
315 }
316
317 ALWAYS_LOG_MEDIASESSIONMANAGER(__func__, changedClientState.get(), " calling setShouldPlayToPlaybackTarget(true)");
318 changedClientState->client.setShouldPlayToPlaybackTarget(changedClientState->contextId, true);
319
320 if (index && m_clientState.size() > 1)
321 std::swap(m_clientState.at(index), m_clientState.at(0));
322}

347void WebMediaSessionManager::configureNewClients()
348{
349 for (auto& state : m_clientState) {
350 if (!state->configurationRequired)
351 continue;
352
353 state->configurationRequired = false;
354 if (m_externalOutputDeviceAvailable)
355 state->client.externalOutputDeviceAvailableDidChange(state->contextId, true);
356
357 if (m_playbackTarget)
358 state->client.setPlaybackTarget(state->contextId, *m_playbackTarget.copyRef());
359 }
360}

362void WebMediaSessionManager::configurePlaybackTargetClients()
363{
364 if (m_clientState.isEmpty())
365 return;
366
367 size_t indexOfClientThatRequestedPicker = notFound;
368 size_t indexOfLastClientToRequestPicker = notFound;
369 size_t indexOfClientWillPlayToTarget = notFound;
370 bool haveActiveRoute = m_playbackTarget && m_playbackTarget->hasActiveRoute();
371
372 for (size_t i = 0; i < m_clientState.size(); ++i) {
373 auto& state = m_clientState[i];
374
375 ALWAYS_LOG_MEDIASESSIONMANAGER(__func__, state.get(), ", requestedPicker = ", state->requestedPicker);
376
377 if ((m_targetChanged || m_playbackTargetPickerDismissed) && state->requestedPicker)
378 indexOfClientThatRequestedPicker = i;
379
380 if (indexOfClientWillPlayToTarget == notFound && flagsAreSet(state->flags, MediaProducer::MediaState::IsPlayingToExternalDevice))
381 indexOfClientWillPlayToTarget = i;
382
383 if (indexOfClientWillPlayToTarget == notFound && haveActiveRoute && state->previouslyRequestedPicker)
384 indexOfLastClientToRequestPicker = i;
385 }
386
387 if (indexOfClientThatRequestedPicker != notFound)
388 indexOfClientWillPlayToTarget = indexOfClientThatRequestedPicker;
389 if (indexOfClientWillPlayToTarget == notFound && indexOfLastClientToRequestPicker != notFound)
390 indexOfClientWillPlayToTarget = indexOfLastClientToRequestPicker;
391 if (indexOfClientWillPlayToTarget == notFound && haveActiveRoute && flagsAreSet(m_clientState[0]->flags, MediaProducer::MediaState::ExternalDeviceAutoPlayCandidate) && !flagsAreSet(m_clientState[0]->flags, MediaProducer::MediaState::IsPlayingVideo))
392 indexOfClientWillPlayToTarget = 0;
393
394 for (size_t i = 0; i < m_clientState.size(); ++i) {
395 auto& state = m_clientState[i];
396
397 if (m_playbackTarget)
398 state->client.setPlaybackTarget(state->contextId, *m_playbackTarget.copyRef());
399
400 if (i != indexOfClientWillPlayToTarget || !haveActiveRoute) {
401 ALWAYS_LOG_MEDIASESSIONMANAGER(__func__, state.get(), " calling setShouldPlayToPlaybackTarget(false)");
402 state->client.setShouldPlayToPlaybackTarget(state->contextId, false);
403 }
404
405 if (state->requestedPicker && m_playbackTargetPickerDismissed) {
406 ALWAYS_LOG_MEDIASESSIONMANAGER(__func__, state.get(), " calling playbackTargetPickerWasDismissed");
407 state->client.playbackTargetPickerWasDismissed(state->contextId);
408 }
409
410 state->configurationRequired = false;
411 if (m_targetChanged || m_playbackTargetPickerDismissed)
412 state->requestedPicker = false;
413 }
414
415 if (haveActiveRoute && indexOfClientWillPlayToTarget != notFound) {
416 auto& state = m_clientState[indexOfClientWillPlayToTarget];
417 if (!flagsAreSet(state->flags, MediaProducer::MediaState::IsPlayingToExternalDevice)) {
418 ALWAYS_LOG_MEDIASESSIONMANAGER(__func__, state.get(), " calling setShouldPlayToPlaybackTarget(true)");
419 state->client.setShouldPlayToPlaybackTarget(state->contextId, true);
420 }
421 }
422
423 m_targetChanged = false;
424 configureWatchdogTimer();
425}

427void WebMediaSessionManager::configurePlaybackTargetMonitoring()
428{
429 bool monitoringRequired = false;
430 bool hasAvailabilityListener = false;
431 bool haveClientWithMedia = false;
432 for (auto& state : m_clientState) {
433 ALWAYS_LOG_MEDIASESSIONMANAGER(__func__, state.get());
434 if (state->flags & MediaProducer::MediaState::RequiresPlaybackTargetMonitoring) {
435 monitoringRequired = true;
436 break;
437 }
438 if (state->flags & MediaProducer::MediaState::HasPlaybackTargetAvailabilityListener)
439 hasAvailabilityListener = true;
440 if (state->flags & MediaProducer::MediaState::HasAudioOrVideo)
441 haveClientWithMedia = true;
442 }
443
444 if (monitoringRequired || (hasAvailabilityListener && haveClientWithMedia)) {
445 ALWAYS_LOG_MEDIASESSIONMANAGER(__func__, "starting monitoring");
446 targetPicker().startingMonitoringPlaybackTargets();
447 } else {
448 ALWAYS_LOG_MEDIASESSIONMANAGER(__func__, "stopping monitoring");
449 targetPicker().stopMonitoringPlaybackTargets();
450 }
451}

453void WebMediaSessionManager::scheduleDelayedTask(ConfigurationTasks tasks)
454{
455 m_taskFlags |= tasks;
456 m_taskTimer.startOneShot(taskDelayInterval);
457}

459void WebMediaSessionManager::taskTimerFired()
460{
461 if (m_taskFlags & InitialConfigurationTask)
462 configureNewClients();
463 if (m_taskFlags & TargetClientsConfigurationTask)
464 configurePlaybackTargetClients();
465 if (m_taskFlags & TargetMonitoringConfigurationTask)
466 configurePlaybackTargetMonitoring();
467 if (m_taskFlags & WatchdogTimerConfigurationTask)
468 configureWatchdogTimer();
469
470 m_taskFlags = NoTask;
471}

473size_t WebMediaSessionManager::find(WebMediaSessionManagerClient* client, PlaybackTargetClientContextIdentifier contextId)
474{
475 for (size_t i = 0; i < m_clientState.size(); ++i) {
476 if (m_clientState[i]->contextId == contextId && &m_clientState[i]->client == client)
477 return i;
478 }
479
480 return notFound;
481}

483void WebMediaSessionManager::configureWatchdogTimer()
484{
485 static const Seconds watchdogTimerIntervalAfterPausing { 1_h };
486 static const Seconds watchdogTimerIntervalAfterPlayingToEnd { 8_min };
487
488 if (!m_playbackTarget || !m_playbackTarget->hasActiveRoute()) {
489 if (m_watchdogTimer.isActive()) {
490 ALWAYS_LOG_MEDIASESSIONMANAGER(__func__, "stopping timer");
491 m_currentWatchdogInterval = { };
492 m_watchdogTimer.stop();
493 }
494
495 return;
496 }
497
498 bool stopTimer = false;
499 bool didPlayToEnd = false;
500 for (auto& state : m_clientState) {
501
502 ALWAYS_LOG_MEDIASESSIONMANAGER(__func__, state.get(), " playedToEnd = ", state->playedToEnd);
503
504 if (flagsAreSet(state->flags, MediaProducer::MediaState::IsPlayingToExternalDevice) && flagsAreSet(state->flags, MediaProducer::MediaState::IsPlayingVideo))
505 stopTimer = true;
506 if (state->playedToEnd)
507 didPlayToEnd = true;
508 state->playedToEnd = false;
509 }
510
511 if (stopTimer) {
512 ALWAYS_LOG_MEDIASESSIONMANAGER(__func__, "stopping timer");
513 m_currentWatchdogInterval = { };
514 m_watchdogTimer.stop();
515 } else {
516 Seconds interval = didPlayToEnd ? watchdogTimerIntervalAfterPlayingToEnd : watchdogTimerIntervalAfterPausing;
517 if (interval != m_currentWatchdogInterval || !m_watchdogTimer.isActive()) {
518 m_watchdogTimer.startOneShot(interval);
519 }
520 ALWAYS_LOG_MEDIASESSIONMANAGER(__func__, "timer scheduled for ", interval.value(), " seconds");
521 m_currentWatchdogInterval = interval;
522 }
523}

525void WebMediaSessionManager::watchdogTimerFired()
526{
527 if (!m_playbackTarget)
528 return;
529
530 ALWAYS_LOG_MEDIASESSIONMANAGER(__func__);
531 targetPicker().invalidatePlaybackTargets();
532}
