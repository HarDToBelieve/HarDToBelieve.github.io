26#pragma once
27
28#if ENABLE(APPLE_PAY)
29
30#include "ApplePayErrorCode.h"
31#include "ApplePayErrorContactField.h"
32#include <wtf/Ref.h>
33#include <wtf/RefCounted.h>
34#include <wtf/RefPtr.h>
35#include <wtf/text/WTFString.h>
36
37namespace WebCore {
38
39class ApplePayError final : public RefCounted<ApplePayError> {
40public:
41 static Ref<ApplePayError> create(ApplePayErrorCode code, std::optional<ApplePayErrorContactField> contactField, const String& message)
42 {
43 return adoptRef(*new ApplePayError(code, contactField, message));
44 }
45
46 virtual ~ApplePayError() = default;
47
48 ApplePayErrorCode code() const { return m_code; }
49 void setCode(ApplePayErrorCode code) { m_code = code; }
50
51 std::optional<ApplePayErrorContactField> contactField() const { return m_contactField; }
52 void setContactField(std::optional<ApplePayErrorContactField> contactField) { m_contactField = contactField; }
53
54 String message() const { return m_message; }
55 void setMessage(String&& message) { m_message = WTFMove(message); }
56
57 template<class Encoder> void encode(Encoder&) const;
58 template<class Decoder> static RefPtr<ApplePayError> decode(Decoder&);
59
60private:
61 ApplePayError(ApplePayErrorCode code, std::optional<ApplePayErrorContactField> contactField, const String& message)
62 : m_code(code)
63 , m_contactField(contactField)
64 , m_message(message)
65 {
66 }
67
68 ApplePayErrorCode m_code;
69 std::optional<ApplePayErrorContactField> m_contactField;
70 String m_message;
71};
72
73template<class Encoder>
74void ApplePayError::encode(Encoder& encoder) const
75{
76 encoder << m_code;
77 encoder << m_contactField;
78 encoder << m_message;
79}
80
81template<class Decoder>
82RefPtr<ApplePayError> ApplePayError::decode(Decoder& decoder)
83{
84#define DECODE(name, type) \
85 std::optional<type> name; \
86 decoder >> name; \
87 if (!name) \
88 return nullptr; \
89
90 DECODE(code, ApplePayErrorCode)
91 DECODE(contactField, std::optional<ApplePayErrorContactField>)
92 DECODE(message, String)
93
94#undef DECODE
95
96 return ApplePayError::create(WTFMove(*code), WTFMove(*contactField), WTFMove(*message));
97}

99} // namespace WebCore

101#endif
