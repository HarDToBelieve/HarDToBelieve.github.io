28#if ENABLE(APPLE_PAY)

30#include "ApplePayDetailsUpdateBase.h"
31#include <wtf/text/WTFString.h>
32
33namespace WebCore {
34
35struct ApplePayPaymentMethodUpdate final : public ApplePayDetailsUpdateBase {
36#if ENABLE(APPLE_PAY_INSTALLMENTS)
37 String installmentGroupIdentifier;
38#endif // ENABLE(APPLE_PAY_INSTALLMENTS)
39
40 template<class Encoder> void encode(Encoder&) const;
41 template<class Decoder> static std::optional<ApplePayPaymentMethodUpdate> decode(Decoder&);
42};
43
44template<class Encoder>
45void ApplePayPaymentMethodUpdate::encode(Encoder& encoder) const
46{
47 ApplePayDetailsUpdateBase::encode(encoder);
48#if ENABLE(APPLE_PAY_INSTALLMENTS)
49 encoder << installmentGroupIdentifier;
50#endif // ENABLE(APPLE_PAY_INSTALLMENTS)
51}
52
53template<class Decoder>
54std::optional<ApplePayPaymentMethodUpdate> ApplePayPaymentMethodUpdate::decode(Decoder& decoder)
55{
56 ApplePayPaymentMethodUpdate result;
57
58 if (!result.decodeBase(decoder))
59 return std::nullopt;
60
61#define DECODE(name, type) \
62 std::optional<type> name; \
63 decoder >> name; \
64 if (!name) \
65 return std::nullopt; \
66 result.name = WTFMove(*name); \
67
68#if ENABLE(APPLE_PAY_INSTALLMENTS)
69 DECODE(installmentGroupIdentifier, String)
70#endif // ENABLE(APPLE_PAY_INSTALLMENTS)
71
72#undef DECODE
73
74 return result;
75}

77} // namespace WebCore
