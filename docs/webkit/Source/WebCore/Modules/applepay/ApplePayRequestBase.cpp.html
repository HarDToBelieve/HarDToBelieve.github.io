40static bool requiresSupportedNetworks(unsigned version, const ApplePayRequestBase& request)
41{
42#if ENABLE(APPLE_PAY_INSTALLMENTS)
43 return version < 8 || !request.installmentConfiguration;
44#else
45 UNUSED_PARAM(version);
46 UNUSED_PARAM(request);
47 return true;
48#endif
49}
50
51static ExceptionOr<Vector<String>> convertAndValidate(Document& document, unsigned version, const Vector<String>& supportedNetworks, const PaymentCoordinator& paymentCoordinator)
52{
53 Vector<String> result;
54 result.reserveInitialCapacity(supportedNetworks.size());
55 for (auto& supportedNetwork : supportedNetworks) {
56 auto validatedNetwork = paymentCoordinator.validatedPaymentNetwork(document, version, supportedNetwork);
57 if (!validatedNetwork)
58 return Exception { TypeError, makeString("\"", supportedNetwork, "\" is not a valid payment network.") };
59 result.uncheckedAppend(*validatedNetwork);
60 }
61
62 return WTFMove(result);
63}

65ExceptionOr<ApplePaySessionPaymentRequest> convertAndValidate(Document& document, unsigned version, ApplePayRequestBase& request, const PaymentCoordinator& paymentCoordinator)
66{
67 if (!version || !paymentCoordinator.supportsVersion(document, version))
68 return Exception { InvalidAccessError, makeString('"', version, "\" is not a supported version.") };
69
70 ApplePaySessionPaymentRequest result;
71 result.setVersion(version);
72 result.setCountryCode(request.countryCode);
73
74 auto merchantCapabilities = convertAndValidate(request.merchantCapabilities);
75 if (merchantCapabilities.hasException())
76 return merchantCapabilities.releaseException();
77 result.setMerchantCapabilities(merchantCapabilities.releaseReturnValue());
78
79 if (requiresSupportedNetworks(version, request) && request.supportedNetworks.isEmpty())
80 return Exception { TypeError, "At least one supported network must be provided." };
81
82 auto supportedNetworks = convertAndValidate(document, version, request.supportedNetworks, paymentCoordinator);
83 if (supportedNetworks.hasException())
84 return supportedNetworks.releaseException();
85 result.setSupportedNetworks(supportedNetworks.releaseReturnValue());
86
87 if (request.requiredBillingContactFields) {
88 auto requiredBillingContactFields = convertAndValidate(version, *request.requiredBillingContactFields);
89 if (requiredBillingContactFields.hasException())
90 return requiredBillingContactFields.releaseException();
91 result.setRequiredBillingContactFields(requiredBillingContactFields.releaseReturnValue());
92 }
93
94 if (request.billingContact)
95 result.setBillingContact(PaymentContact::fromApplePayPaymentContact(version, *request.billingContact));
96
97 if (request.requiredShippingContactFields) {
98 auto requiredShippingContactFields = convertAndValidate(version, *request.requiredShippingContactFields);
99 if (requiredShippingContactFields.hasException())
100 return requiredShippingContactFields.releaseException();
101 result.setRequiredShippingContactFields(requiredShippingContactFields.releaseReturnValue());
102 }
103
104 if (request.shippingContact)
105 result.setShippingContact(PaymentContact::fromApplePayPaymentContact(version, *request.shippingContact));
106
107 result.setApplicationData(request.applicationData);
108
109 if (version >= 3)
110 result.setSupportedCountries(WTFMove(request.supportedCountries));
111
112#if ENABLE(APPLE_PAY_INSTALLMENTS)
113 if (request.installmentConfiguration) {
114 auto installmentConfiguration = PaymentInstallmentConfiguration::create(*request.installmentConfiguration);
115 if (installmentConfiguration.hasException())
116 return installmentConfiguration.releaseException();
117 result.setInstallmentConfiguration(installmentConfiguration.releaseReturnValue());
118 }
119#endif
120
121#if ENABLE(APPLE_PAY_COUPON_CODE)
122 result.setSupportsCouponCode(request.supportsCouponCode);
123 result.setCouponCode(request.couponCode);
124#endif
125
126#if ENABLE(APPLE_PAY_SHIPPING_CONTACT_EDITING_MODE)
127 result.setShippingContactEditingMode(request.shippingContactEditingMode);
128#endif
129
130#if defined(ApplePayRequestBaseAdditions_convertAndValidate_request)
131 ApplePayRequestBaseAdditions_convertAndValidate_request
132#endif
133
134 return WTFMove(result);
135}
