38namespace WebCore {
39
40class ApplePaySetupFeature;
41class DeferredPromise;
42class Document;
43
44class ApplePaySetup : public ActiveDOMObject, public RefCounted<ApplePaySetup> {
45public:
46 static Ref<ApplePaySetup> create(ScriptExecutionContext& context, ApplePaySetupConfiguration&& configuration)
47 {
48 return adoptRef(*new ApplePaySetup(context, WTFMove(configuration)));
49 }
50
51 using SetupFeaturesPromise = DOMPromiseDeferred<IDLSequence<IDLInterface<ApplePaySetupFeature>>>;
52 void getSetupFeatures(Document&, SetupFeaturesPromise&&);
53
54 using BeginPromise = DOMPromiseDeferred<IDLBoolean>;
55 void begin(Document&, Vector<RefPtr<ApplePaySetupFeature>>&&, BeginPromise&&);
56
57private:
58 ApplePaySetup(ScriptExecutionContext&, ApplePaySetupConfiguration&&);
59
60 // ActiveDOMObject
61 const char* activeDOMObjectName() const final { return "ApplePaySetup"; }
62 void stop() final;
63 void suspend(ReasonForSuspension) final;
64
65 ApplePaySetupConfiguration m_configuration;
66 std::optional<SetupFeaturesPromise> m_setupFeaturesPromise;
67 std::optional<BeginPromise> m_beginPromise;
68 RefPtr<PendingActivity<ApplePaySetup>> m_pendingActivity;
69};
