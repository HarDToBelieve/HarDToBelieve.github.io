46ExceptionOr<void> PaymentRequestValidator::validate(const ApplePaySessionPaymentRequest& paymentRequest)
47{
48 auto validatedCountryCode = validateCountryCode(paymentRequest.countryCode());
49 if (validatedCountryCode.hasException())
50 return validatedCountryCode.releaseException();
51
52 auto validatedCurrencyCode = validateCurrencyCode(paymentRequest.currencyCode());
53 if (validatedCurrencyCode.hasException())
54 return validatedCurrencyCode.releaseException();
55
56 auto validatedSupportedNetworks = validateSupportedNetworks(paymentRequest.supportedNetworks());
57 if (validatedSupportedNetworks.hasException())
58 return validatedSupportedNetworks.releaseException();
59
60 auto validatedMerchantCapabilities = validateMerchantCapabilities(paymentRequest.merchantCapabilities());
61 if (validatedMerchantCapabilities.hasException())
62 return validatedMerchantCapabilities.releaseException();
63
64 auto validatedTotal = validateTotal(paymentRequest.total());
65 if (validatedTotal.hasException())
66 return validatedTotal.releaseException();
67
68 auto validatedShippingMethods = validateShippingMethods(paymentRequest.shippingMethods());
69 if (validatedShippingMethods.hasException())
70 return validatedShippingMethods.releaseException();
71
72 for (auto& countryCode : paymentRequest.supportedCountries()) {
73 auto validatedCountryCode = validateCountryCode(countryCode);
74 if (validatedCountryCode.hasException())
75 return validatedCountryCode.releaseException();
76 }
77
78 return { };
79}

81ExceptionOr<void> PaymentRequestValidator::validateTotal(const ApplePayLineItem& total)
82{
83 if (!total.label)
84 return Exception { TypeError, "Missing total label." };
85
86 if (!total.amount)
87 return Exception { TypeError, "Missing total amount." };
88
89 double amount = [NSDecimalNumber decimalNumberWithString:total.amount locale:@{ NSLocaleDecimalSeparator : @"." }].doubleValue;
90
91 if (amount < 0)
92 return Exception { TypeError, "Total amount must not be negative." };
93
94 if (amount > 100000000)
95 return Exception { TypeError, "Total amount is too big." };
96
97 return { };
98}

100static ExceptionOr<void> validateCountryCode(const String& countryCode)
101{
102 if (!countryCode)
103 return Exception { TypeError, "Missing country code." };
104
105 for (auto *countryCodePtr = uloc_getISOCountries(); *countryCodePtr; ++countryCodePtr) {
106 if (countryCode == *countryCodePtr)
107 return { };
108 }
109
110 return Exception { TypeError, makeString("\"" + countryCode, "\" is not a valid country code.") };
111}

113static ExceptionOr<void> validateCurrencyCode(const String& currencyCode)
114{
115 if (!currencyCode)
116 return Exception { TypeError, "Missing currency code." };
117
118 UErrorCode errorCode = U_ZERO_ERROR;
119 auto currencyCodes = std::unique_ptr<UEnumeration, ICUDeleter<uenum_close>>(ucurr_openISOCurrencies(UCURR_ALL, &errorCode));
120
121 int32_t length;
122 while (auto *currencyCodePtr = uenum_next(currencyCodes.get(), &length, &errorCode)) {
123 if (currencyCodePtr == currencyCode)
124 return { };
125 }
126
127 return Exception { TypeError, makeString("\"" + currencyCode, "\" is not a valid currency code.") };
128}

130static ExceptionOr<void> validateMerchantCapabilities(const ApplePaySessionPaymentRequest::MerchantCapabilities& merchantCapabilities)
131{
132 if (!merchantCapabilities.supports3DS && !merchantCapabilities.supportsEMV && !merchantCapabilities.supportsCredit && !merchantCapabilities.supportsDebit)
133 return Exception { TypeError, "Missing merchant capabilities." };
134
135 return { };
136}

146static ExceptionOr<void> validateShippingMethod(const ApplePayShippingMethod& shippingMethod)
147{
148 NSDecimalNumber *amount = [NSDecimalNumber decimalNumberWithString:shippingMethod.amount locale:@{ NSLocaleDecimalSeparator : @"." }];
149 if (amount.integerValue < 0)
150 return Exception { TypeError, "Shipping method amount must be greater than or equal to zero." };
151
152 return { };
153}
154
155static ExceptionOr<void> validateShippingMethods(const Vector<ApplePayShippingMethod>& shippingMethods)
156{
157 for (const auto& shippingMethod : shippingMethods) {
158 auto validatedShippingMethod = validateShippingMethod(shippingMethod);
159 if (validatedShippingMethod.hasException())
160 return validatedShippingMethod.releaseException();
161 }
162
163 return { };
164}
