35unsigned PaymentAPIVersion::current()
36{
37 static unsigned current = [] {
38#if ENABLE(APPLE_PAY_COUPON_CODE) || ENABLE(APPLE_PAY_SHIPPING_CONTACT_EDITING_MODE) || ENABLE(APPLE_PAY_RECURRING_LINE_ITEM) || ENABLE(APPLE_PAY_DEFERRED_LINE_ITEM) || ENABLE(APPLE_PAY_SHIPPING_METHOD_DATE_COMPONENTS_RANGE)
39 return 12;
40#elif ENABLE(APPLE_PAY_SESSION_V11)
41 return 11;
42#elif HAVE(PASSKIT_NEW_BUTTON_TYPES)
43 return 10;
44#elif HAVE(PASSKIT_INSTALLMENTS)
45 if (PAL::getPKPaymentInstallmentConfigurationClass()) {
46 if (PAL::getPKPaymentInstallmentItemClass())
47 return 9;
48 return 8;
49 }
50#endif
51 return 7;
52 }();
53 return current;
54}
