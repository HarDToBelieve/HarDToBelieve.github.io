38static void finishConverting(PKPayment *payment, ApplePayPayment& result)
39{
40#if HAVE(PASSKIT_INSTALLMENTS)
41 if (NSString *installmentAuthorizationToken = payment.installmentAuthorizationToken)
42 result.installmentAuthorizationToken = installmentAuthorizationToken;
43#else
44 UNUSED_PARAM(payment);
45 UNUSED_PARAM(result);
46#endif
47}
48
49static ApplePayPayment::Token convert(PKPaymentToken *paymentToken)
50{
51 ASSERT(paymentToken);
52
53 ApplePayPayment::Token result;
54
55 result.paymentMethod = PaymentMethod(paymentToken.paymentMethod).toApplePayPaymentMethod();
56
57 if (NSString *transactionIdentifier = paymentToken.transactionIdentifier)
58 result.transactionIdentifier = transactionIdentifier;
59 if (NSData *paymentData = paymentToken.paymentData)
60 result.paymentData = String::fromUTF8((const char*)paymentData.bytes, paymentData.length);
61
62 return result;
63}
64
65static ApplePayPayment convert(unsigned version, PKPayment *payment)
66{
67 ASSERT(payment);
68
69 ApplePayPayment result;
70
71 result.token = convert(payment.token);
72
73 if (payment.billingContact)
74 result.billingContact = PaymentContact(payment.billingContact).toApplePayPaymentContact(version);
75 if (payment.shippingContact)
76 result.shippingContact = PaymentContact(payment.shippingContact).toApplePayPaymentContact(version);
77
78 finishConverting(payment, result);
79
80 return result;
81}
