43static RetainPtr<PKContact> convert(unsigned version, const ApplePayPaymentContact& contact)
44{
45 auto result = adoptNS([PAL::allocPKContactInstance() init]);
46
47 NSString *familyName = nil;
48 NSString *phoneticFamilyName = nil;
49 if (!contact.familyName.isEmpty()) {
50 familyName = contact.familyName;
51 if (version >= 3 && !contact.phoneticFamilyName.isEmpty())
52 phoneticFamilyName = contact.phoneticFamilyName;
53 }
54
55 NSString *givenName = nil;
56 NSString *phoneticGivenName = nil;
57 if (!contact.givenName.isEmpty()) {
58 givenName = contact.givenName;
59 if (version >= 3 && !contact.phoneticGivenName.isEmpty())
60 phoneticGivenName = contact.phoneticGivenName;
61 }
62
63 if (familyName || givenName) {
64 auto name = adoptNS([[NSPersonNameComponents alloc] init]);
65 [name setFamilyName:familyName];
66 [name setGivenName:givenName];
67 if (phoneticFamilyName || phoneticGivenName) {
68 auto phoneticName = adoptNS([[NSPersonNameComponents alloc] init]);
69 [phoneticName setFamilyName:phoneticFamilyName];
70 [phoneticName setGivenName:phoneticGivenName];
71 [name setPhoneticRepresentation:phoneticName.get()];
72 }
73 [result setName:name.get()];
74 }
75
76 if (!contact.emailAddress.isEmpty())
77 [result setEmailAddress:contact.emailAddress];
78
79 if (!contact.phoneNumber.isEmpty())
80 [result setPhoneNumber:adoptNS([allocCNPhoneNumberInstance() initWithStringValue:contact.phoneNumber]).get()];
81
82 if (contact.addressLines && !contact.addressLines->isEmpty()) {
83 auto address = adoptNS([allocCNMutablePostalAddressInstance() init]);
84
85 StringBuilder builder;
86 for (unsigned i = 0; i < contact.addressLines->size(); ++i) {
87 builder.append(contact.addressLines->at(i));
88 if (i != contact.addressLines->size() - 1)
89 builder.append('\n');
90 }
91
92 // FIXME: StringBuilder should hava a toNSString() function to avoid the extra String allocation.
93 [address setStreet:builder.toString()];
94
95 if (!contact.subLocality.isEmpty())
96 [address setSubLocality:contact.subLocality];
97 if (!contact.locality.isEmpty())
98 [address setCity:contact.locality];
99 if (!contact.postalCode.isEmpty())
100 [address setPostalCode:contact.postalCode];
101 if (!contact.subAdministrativeArea.isEmpty())
102 [address setSubAdministrativeArea:contact.subAdministrativeArea];
103 if (!contact.administrativeArea.isEmpty())
104 [address setState:contact.administrativeArea];
105 if (!contact.country.isEmpty())
106 [address setCountry:contact.country];
107 if (!contact.countryCode.isEmpty())
108 [address setISOCountryCode:contact.countryCode];
109
110 [result setPostalAddress:address.get()];
111 }
112
113 return result;
114}

116static ApplePayPaymentContact convert(unsigned version, PKContact *contact)
117{
118 ASSERT(contact);
119
120 ApplePayPaymentContact result;
121
122 result.phoneNumber = contact.phoneNumber.stringValue;
123 result.emailAddress = contact.emailAddress;
124
125 NSPersonNameComponents *name = contact.name;
126 result.givenName = name.givenName;
127 result.familyName = name.familyName;
128 if (name)
129 result.localizedName = [NSPersonNameComponentsFormatter localizedStringFromPersonNameComponents:name style:NSPersonNameComponentsFormatterStyleDefault options:0];
130
131 if (version >= 3) {
132 NSPersonNameComponents *phoneticName = name.phoneticRepresentation;
133 result.phoneticGivenName = phoneticName.givenName;
134 result.phoneticFamilyName = phoneticName.familyName;
135 if (phoneticName)
136 result.localizedPhoneticName = [NSPersonNameComponentsFormatter localizedStringFromPersonNameComponents:name style:NSPersonNameComponentsFormatterStyleDefault options:NSPersonNameComponentsFormatterPhonetic];
137 }
138
139 CNPostalAddress *postalAddress = contact.postalAddress;
140 if (postalAddress.street.length) {
141 Vector<String> addressLines = String(postalAddress.street).split('\n');
142 result.addressLines = WTFMove(addressLines);
143 }
144 result.subLocality = postalAddress.subLocality;
145 result.locality = postalAddress.city;
146 result.postalCode = postalAddress.postalCode;
147 result.subAdministrativeArea = postalAddress.subAdministrativeArea;
148 result.administrativeArea = postalAddress.state;
149 result.country = postalAddress.country;
150 result.countryCode = postalAddress.ISOCountryCode;
151
152 return result;
153}
