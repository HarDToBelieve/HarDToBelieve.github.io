56ApplicationManifest ApplicationManifestParser::parseManifest(const String& text, const URL& manifestURL, const URL& documentURL)
57{
58 m_manifestURL = manifestURL;
59
60 auto jsonValue = JSON::Value::parseJSON(text);
61 if (!jsonValue) {
62 logDeveloperWarning("The manifest is not valid JSON data."_s);
63 jsonValue = JSON::Object::create();
64 }
65
66 auto manifest = jsonValue->asObject();
67 if (!manifest) {
68 logDeveloperWarning("The manifest is not a JSON value of type \"object\"."_s);
69 manifest = JSON::Object::create();
70 }
71
72 ApplicationManifest parsedManifest;
73
74 parsedManifest.startURL = parseStartURL(*manifest, documentURL);
75 parsedManifest.display = parseDisplay(*manifest);
76 parsedManifest.name = parseName(*manifest);
77 parsedManifest.description = parseDescription(*manifest);
78 parsedManifest.shortName = parseShortName(*manifest);
79 parsedManifest.scope = parseScope(*manifest, documentURL, parsedManifest.startURL);
80 parsedManifest.themeColor = parseColor(*manifest, "theme_color"_s);
81
82 if (m_document)
83 m_document->processApplicationManifest(parsedManifest);
84
85 return parsedManifest;
86}

104URL ApplicationManifestParser::parseStartURL(const JSON::Object& manifest, const URL& documentURL)
105{
106 auto value = manifest.getValue("start_url"_s);
107 if (!value)
108 return documentURL;
109
110 auto stringValue = value->asString();
111 if (!stringValue) {
112 logManifestPropertyNotAString("start_url"_s);
113 return documentURL;
114 }
115
116 if (stringValue.isEmpty())
117 return documentURL;
118
119 URL startURL(m_manifestURL, stringValue);
120 if (!startURL.isValid()) {
121 logManifestPropertyInvalidURL("start_url"_s);
122 return documentURL;
123 }
124
125 if (!protocolHostAndPortAreEqual(startURL, documentURL)) {
126 auto startURLOrigin = SecurityOrigin::create(startURL);
127 auto documentOrigin = SecurityOrigin::create(documentURL);
128 logDeveloperWarning(makeString("The start_url's origin of \""_s, startURLOrigin->toString(), "\" is different from the document's origin of \""_s, documentOrigin->toString(), "\"."_s));
129 return documentURL;
130 }
131
132 return startURL;
133}

135ApplicationManifest::Display ApplicationManifestParser::parseDisplay(const JSON::Object& manifest)
136{
137 auto value = manifest.getValue("display"_s);
138 if (!value)
139 return ApplicationManifest::Display::Browser;
140
141 auto stringValue = value->asString();
142 if (!stringValue) {
143 logManifestPropertyNotAString("display"_s);
144 return ApplicationManifest::Display::Browser;
145 }
146
147 stringValue = stringValue.stripWhiteSpace().convertToASCIILowercase();
148
149 if (stringValue == "fullscreen")
150 return ApplicationManifest::Display::Fullscreen;
151 if (stringValue == "standalone")
152 return ApplicationManifest::Display::Standalone;
153 if (stringValue == "minimal-ui")
154 return ApplicationManifest::Display::MinimalUI;
155 if (stringValue == "browser")
156 return ApplicationManifest::Display::Browser;
157
158 logDeveloperWarning(makeString("\""_s, stringValue, "\" is not a valid display mode."_s));
159 return ApplicationManifest::Display::Browser;
160}

177static bool isInScope(const URL& scopeURL, const URL& targetURL)
178{
179 // 1. If scopeURL is undefined (i.e., it is unbounded because of an error or it was not declared in the manifest), return true.
180 if (scopeURL.isNull() || scopeURL.isEmpty())
181 return true;
182
183 // 2. Let target be a new URL using targetURL as input. If target is failure, return false.
184 if (!targetURL.isValid())
185 return false;
186
187 // 3. Let scopePath be the elements of scopes's path, separated by U+002F (/).
188 auto scopePath = scopeURL.path();
189
190 // 4. Let targetPath be the elements of target's path, separated by U+002F (/).
191 auto targetPath = targetURL.path();
192
193 // 5. If target is same origin as scope and targetPath starts with scopePath, return true.
194 if (protocolHostAndPortAreEqual(scopeURL, targetURL) && targetPath.startsWith(scopePath))
195 return true;
196
197 // 6. Otherwise, return false.
198 return false;
199}

201URL ApplicationManifestParser::parseScope(const JSON::Object& manifest, const URL& documentURL, const URL& startURL)
202{
203 URL defaultScope { startURL, "./" };
204
205 auto value = manifest.getValue("scope");
206 if (!value)
207 return defaultScope;
208
209 auto stringValue = value->asString();
210 if (!stringValue) {
211 logManifestPropertyNotAString("scope"_s);
212 return defaultScope;
213 }
214
215 if (stringValue.isEmpty())
216 return defaultScope;
217
218 URL scopeURL(m_manifestURL, stringValue);
219 if (!scopeURL.isValid()) {
220 logManifestPropertyInvalidURL("scope"_s);
221 return defaultScope;
222 }
223
224 if (!protocolHostAndPortAreEqual(scopeURL, documentURL)) {
225 auto scopeURLOrigin = SecurityOrigin::create(scopeURL);
226 auto documentOrigin = SecurityOrigin::create(documentURL);
227 logDeveloperWarning(makeString("The scope's origin of \""_s, scopeURLOrigin->toString(), "\" is different from the document's origin of \""_s, documentOrigin->toString(), "\"."_s));
228 return defaultScope;
229 }
230
231 if (!isInScope(scopeURL, startURL)) {
232 logDeveloperWarning("The start URL is not within scope of the provided scope URL."_s);
233 return defaultScope;
234 }
235
236 return scopeURL;
237}

244String ApplicationManifestParser::parseGenericString(const JSON::Object& manifest, const String& propertyName)
245{
246 auto value = manifest.getValue(propertyName);
247 if (!value)
248 return { };
249
250 auto stringValue = value->asString();
251 if (!stringValue) {
252 logManifestPropertyNotAString(propertyName);
253 return { };
254 }
255
256 return stringValue.stripWhiteSpace();
257}
