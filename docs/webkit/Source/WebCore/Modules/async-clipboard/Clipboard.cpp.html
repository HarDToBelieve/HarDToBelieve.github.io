51static bool shouldProceedWithClipboardWrite(const Frame& frame)
52{
53 auto& settings = frame.settings();
54 if (settings.javaScriptCanAccessClipboard() || frame.editor().isCopyingFromMenuOrKeyBinding())
55 return true;
56
57 switch (settings.clipboardAccessPolicy()) {
58 case ClipboardAccessPolicy::Allow:
59 return true;
60 case ClipboardAccessPolicy::RequiresUserGesture:
61 return UserGestureIndicator::processingUserGesture();
62 case ClipboardAccessPolicy::Deny:
63 return false;
64 }
65
66 ASSERT_NOT_REACHED();
67 return false;
68}

101void Clipboard::readText(Ref<DeferredPromise>&& promise)
102{
103 auto frame = makeRefPtr(this->frame());
104 if (!frame) {
105 promise->reject(NotAllowedError);
106 return;
107 }
108
109 auto pasteboard = Pasteboard::createForCopyAndPaste(PagePasteboardContext::create(frame->pageID()));
110 auto changeCountAtStart = pasteboard->changeCount();
111 if (!frame->requestDOMPasteAccess()) {
112 promise->reject(NotAllowedError);
113 return;
114 }
115
116 auto allInfo = pasteboard->allPasteboardItemInfo();
117 if (!allInfo) {
118 promise->reject(NotAllowedError);
119 return;
120 }
121
122 String text;
123 for (size_t index = 0; index < allInfo->size(); ++index) {
124 if (allInfo->at(index).webSafeTypesByFidelity.contains("text/plain"_s)) {
125 PasteboardPlainText plainTextReader;
126 pasteboard->read(plainTextReader, PlainTextURLReadingPolicy::IgnoreURL, index);
127 text = WTFMove(plainTextReader.text);
128 break;
129 }
130 }
131
132 if (changeCountAtStart == pasteboard->changeCount())
133 promise->resolve<IDLDOMString>(WTFMove(text));
134 else
135 promise->reject(NotAllowedError);
136}

138void Clipboard::writeText(const String& data, Ref<DeferredPromise>&& promise)
139{
140 auto frame = makeRefPtr(this->frame());
141 auto document = makeRefPtr(frame ? frame->document() : nullptr);
142 if (!document || !shouldProceedWithClipboardWrite(*frame)) {
143 promise->reject(NotAllowedError);
144 return;
145 }
146
147 PasteboardCustomData customData;
148 customData.writeString("text/plain"_s, data);
149 customData.setOrigin(document->originIdentifierForPasteboard());
150 Pasteboard::createForCopyAndPaste(PagePasteboardContext::create(frame->pageID()))->writeCustomData({ WTFMove(customData) });
151 promise->resolve();
152}

154void Clipboard::read(Ref<DeferredPromise>&& promise)
155{
156 auto rejectPromiseAndClearActiveSession = [&] {
157 m_activeSession = std::nullopt;
158 promise->reject(NotAllowedError);
159 };
160
161 auto frame = makeRefPtr(this->frame());
162 if (!frame) {
163 rejectPromiseAndClearActiveSession();
164 return;
165 }
166
167 auto pasteboard = Pasteboard::createForCopyAndPaste(PagePasteboardContext::create(frame->pageID()));
168 auto changeCountAtStart = pasteboard->changeCount();
169
170 if (!frame->requestDOMPasteAccess()) {
171 rejectPromiseAndClearActiveSession();
172 return;
173 }
174
175 if (!m_activeSession || m_activeSession->changeCount != changeCountAtStart) {
176 auto allInfo = pasteboard->allPasteboardItemInfo();
177 if (!allInfo) {
178 rejectPromiseAndClearActiveSession();
179 return;
180 }
181
182 Vector<Ref<ClipboardItem>> clipboardItems;
183 clipboardItems.reserveInitialCapacity(allInfo->size());
184 for (auto& itemInfo : *allInfo)
185 clipboardItems.uncheckedAppend(ClipboardItem::create(*this, itemInfo));
186 m_activeSession = {{ WTFMove(pasteboard), WTFMove(clipboardItems), changeCountAtStart }};
187 }
188
189 promise->resolve<IDLSequence<IDLInterface<ClipboardItem>>>(m_activeSession->items);
190}

192void Clipboard::getType(ClipboardItem& item, const String& type, Ref<DeferredPromise>&& promise)
193{
194 if (!m_activeSession) {
195 promise->reject(NotAllowedError);
196 return;
197 }
198
199 auto frame = makeRefPtr(this->frame());
200 if (!frame) {
201 m_activeSession = std::nullopt;
202 promise->reject(NotAllowedError);
203 return;
204 }
205
206 auto itemIndex = m_activeSession->items.findMatching([&] (auto& activeItem) {
207 return activeItem.ptr() == &item;
208 });
209
210 if (itemIndex == notFound) {
211 promise->reject(NotAllowedError);
212 return;
213 }
214
215 if (!item.types().contains(type)) {
216 promise->reject(NotAllowedError);
217 return;
218 }
219
220 if (type == "image/png"_s) {
221 ClipboardImageReader imageReader { frame->document(), type };
222 activePasteboard().read(imageReader, itemIndex);
223 auto imageBlob = imageReader.takeResult();
224 if (updateSessionValidity() == SessionIsValid::Yes && imageBlob)
225 promise->resolve<IDLInterface<Blob>>(imageBlob.releaseNonNull());
226 else
227 promise->reject(NotAllowedError);
228 return;
229 }
230
231 String resultAsString;
232
233 if (type == "text/uri-list"_s) {
234 String title;
235 resultAsString = activePasteboard().readURL(itemIndex, title).string();
236 }
237
238 if (type == "text/plain"_s) {
239 PasteboardPlainText plainTextReader;
240 activePasteboard().read(plainTextReader, PlainTextURLReadingPolicy::IgnoreURL, itemIndex);
241 resultAsString = WTFMove(plainTextReader.text);
242 }
243
244 if (type == "text/html"_s) {
245 WebContentMarkupReader markupReader { *frame };
246 activePasteboard().read(markupReader, WebContentReadingPolicy::OnlyRichTextTypes, itemIndex);
247 resultAsString = WTFMove(markupReader.markup);
248 }
249
250 // FIXME: Support reading custom data.
251 if (updateSessionValidity() == SessionIsValid::No || resultAsString.isNull()) {
252 promise->reject(NotAllowedError);
253 return;
254 }
255
256 promise->resolve<IDLInterface<Blob>>(ClipboardItem::blobFromString(frame->document(), resultAsString, type));
257}

259Clipboard::SessionIsValid Clipboard::updateSessionValidity()
260{
261 if (!m_activeSession)
262 return SessionIsValid::No;
263
264 if (m_activeSession->changeCount != activePasteboard().changeCount()) {
265 m_activeSession = std::nullopt;
266 return SessionIsValid::No;
267 }
268
269 return SessionIsValid::Yes;
270}

272void Clipboard::write(const Vector<RefPtr<ClipboardItem>>& items, Ref<DeferredPromise>&& promise)
273{
274 auto frame = makeRefPtr(this->frame());
275 if (!frame || !shouldProceedWithClipboardWrite(*frame)) {
276 promise->reject(NotAllowedError);
277 return;
278 }
279
280 if (auto existingWriter = std::exchange(m_activeItemWriter, ItemWriter::create(*this, WTFMove(promise))))
281 existingWriter->invalidate();
282
283 m_activeItemWriter->write(items);
284}

313void Clipboard::ItemWriter::write(const Vector<RefPtr<ClipboardItem>>& items)
314{
315 ASSERT(m_promise);
316 ASSERT(m_clipboard);
317#if PLATFORM(COCOA)
318 m_changeCountAtStart = m_pasteboard->changeCount();
319#endif
320 m_dataToWrite.fill(std::nullopt, items.size());
321 m_pendingItemCount = items.size();
322 for (size_t index = 0; index < items.size(); ++index) {
323 items[index]->collectDataForWriting(*m_clipboard, [this, protectedThis = makeRef(*this), index] (auto data) {
324 protectedThis->setData(WTFMove(data), index);
325 if (!--m_pendingItemCount)
326 didSetAllData();
327 });
328 }
329 if (items.isEmpty())
330 didSetAllData();
331}

333void Clipboard::ItemWriter::invalidate()
334{
335 if (m_promise)
336 reject();
337}

339void Clipboard::ItemWriter::setData(std::optional<PasteboardCustomData>&& data, size_t index)
340{
341 if (index >= m_dataToWrite.size()) {
342 ASSERT_NOT_REACHED();
343 return;
344 }
345
346 m_dataToWrite[index] = WTFMove(data);
347}
348
349void Clipboard::ItemWriter::didSetAllData()
350{
351 if (!m_promise)
352 return;
353
354#if PLATFORM(COCOA)
355 auto newChangeCount = m_pasteboard->changeCount();
356 if (m_changeCountAtStart != newChangeCount) {
357 // FIXME: Instead of checking the changeCount here, send it over to the client (e.g. the UI process
358 // in WebKit2) and perform it there.
359 reject();
360 return;
361 }
362#endif // PLATFORM(COCOA)
363 auto dataToWrite = std::exchange(m_dataToWrite, { });
364 Vector<PasteboardCustomData> customData;
365 customData.reserveInitialCapacity(dataToWrite.size());
366 for (auto data : dataToWrite) {
367 if (!data) {
368 reject();
369 return;
370 }
371 customData.append(*data);
372 }
373
374 m_pasteboard->writeCustomData(WTFMove(customData));
375 m_promise->resolve();
376 m_promise = nullptr;
377
378 if (auto clipboard = std::exchange(m_clipboard, nullptr))
379 clipboard->didResolveOrReject(*this);
380}
381
382void Clipboard::ItemWriter::reject()
383{
384 if (auto promise = std::exchange(m_promise, nullptr))
385 promise->reject(NotAllowedError);
386
387 if (auto clipboard = std::exchange(m_clipboard, nullptr))
388 clipboard->didResolveOrReject(*this);
389}
