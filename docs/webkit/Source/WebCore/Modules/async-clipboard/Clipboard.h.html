35class ClipboardItem;
36class DeferredPromise;
37class Frame;
38class Navigator;
39class Pasteboard;
40class PasteboardCustomData;

42class Clipboard final : public RefCounted<Clipboard>, public EventTargetWithInlineData {
43 WTF_MAKE_ISO_ALLOCATED(Clipboard);
44public:
45 static Ref<Clipboard> create(Navigator&);
46 ~Clipboard();
47
48 EventTargetInterface eventTargetInterface() const final;
49 ScriptExecutionContext* scriptExecutionContext() const final;
50
51 Frame* frame() const;
52 Navigator* navigator();
53
54 using RefCounted::ref;
55 using RefCounted::deref;
56
57 void readText(Ref<DeferredPromise>&&);
58 void writeText(const String& data, Ref<DeferredPromise>&&);
59
60 void read(Ref<DeferredPromise>&&);
61 void write(const Vector<RefPtr<ClipboardItem>>& data, Ref<DeferredPromise>&&);
62
63 void getType(ClipboardItem&, const String& type, Ref<DeferredPromise>&&);
64
65private:
66 Clipboard(Navigator&);
67
68 struct Session {
69 std::unique_ptr<Pasteboard> pasteboard;
70 Vector<Ref<ClipboardItem>> items;
71 int64_t changeCount;
72 };
73
74 void refEventTarget() final { ref(); }
75 void derefEventTarget() final { deref(); }
76
77 Pasteboard& activePasteboard();
78
79 enum class SessionIsValid { No, Yes };
80 SessionIsValid updateSessionValidity();
81
82 class ItemWriter : public RefCounted<ItemWriter> {
83 public:
84 static Ref<ItemWriter> create(Clipboard& clipboard, Ref<DeferredPromise>&& promise)
85 {
86 return adoptRef(*new ItemWriter(clipboard, WTFMove(promise)));
87 }
88
89 ~ItemWriter();
90
91 void write(const Vector<RefPtr<ClipboardItem>>&);
92 void invalidate();
93
94 private:
95 ItemWriter(Clipboard&, Ref<DeferredPromise>&&);
96
97 void setData(std::optional<PasteboardCustomData>&&, size_t index);
98 void didSetAllData();
99 void reject();
100
101 WeakPtr<Clipboard> m_clipboard;
102 Vector<std::optional<PasteboardCustomData>> m_dataToWrite;
103 RefPtr<DeferredPromise> m_promise;
104 unsigned m_pendingItemCount;
105 std::unique_ptr<Pasteboard> m_pasteboard;
106#if PLATFORM(COCOA)
107 int64_t m_changeCountAtStart { 0 };
108#endif
109 };
110
111 void didResolveOrReject(ItemWriter&);
112
113 std::optional<Session> m_activeSession;
114 WeakPtr<Navigator> m_navigator;
115 Vector<std::optional<PasteboardCustomData>> m_dataToWrite;
116 RefPtr<ItemWriter> m_activeItemWriter;
117};
