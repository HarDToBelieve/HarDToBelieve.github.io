28#include "Blob.h"
29#include "Pasteboard.h"

31namespace WebCore {
32
33class Document;
34class SharedBuffer;
35
36struct ClipboardImageReader : PasteboardFileReader {
37 ClipboardImageReader(Document* document, const String& mimeType)
38 : PasteboardFileReader()
39 , m_document(document)
40 , m_mimeType(mimeType)
41 {
42 }
43
44 RefPtr<Blob> takeResult() { return std::exchange(m_result, nullptr); }
45
46private:
47 void readFilename(const String&) final { ASSERT_NOT_REACHED(); }
48
49 bool shouldReadBuffer(const String&) const final;
50 void readBuffer(const String& filename, const String& type, Ref<SharedBuffer>&&) final;
51
52 RefPtr<Document> m_document;
53 String m_mimeType;
54 RefPtr<Blob> m_result;
55};

57} // namespace WebCore
