27
28#include <wtf/KeyValuePair.h>
29#include <wtf/Ref.h>
30#include <wtf/RefCounted.h>
31#include <wtf/Vector.h>
32#include <wtf/WeakPtr.h>
33#include <wtf/text/WTFString.h>

35namespace WebCore {
36
37class Blob;
38class Clipboard;
39class ClipboardItemDataSource;
40class DeferredPromise;
41class DOMPromise;
42class Navigator;
43class PasteboardCustomData;
44class ScriptExecutionContext;
45struct PasteboardItemInfo;
46
47class ClipboardItem : public RefCounted<ClipboardItem> {
48public:
49 ~ClipboardItem();
50
51 enum class PresentationStyle : uint8_t { Unspecified, Inline, Attachment };
52
53 struct Options {
54 PresentationStyle presentationStyle { PresentationStyle::Unspecified };
55 };
56
57 static Ref<ClipboardItem> create(Vector<KeyValuePair<String, RefPtr<DOMPromise>>>&&, const Options&);
58 static Ref<ClipboardItem> create(Clipboard&, const PasteboardItemInfo&);
59 static Ref<Blob> blobFromString(ScriptExecutionContext*, const String& stringData, const String& type);
60
61 Vector<String> types() const;
62 void getType(const String&, Ref<DeferredPromise>&&);
63
64 void collectDataForWriting(Clipboard& destination, CompletionHandler<void(std::optional<PasteboardCustomData>)>&&);
65
66 PresentationStyle presentationStyle() const { return m_presentationStyle; };
67 Navigator* navigator();
68 Clipboard* clipboard();
69
70private:
71 ClipboardItem(Vector<KeyValuePair<String, RefPtr<DOMPromise>>>&&, const Options&);
72 ClipboardItem(Clipboard&, const PasteboardItemInfo&);
73
74 WeakPtr<Clipboard> m_clipboard;
75 WeakPtr<Navigator> m_navigator;
76 std::unique_ptr<ClipboardItemDataSource> m_dataSource;
77 PresentationStyle m_presentationStyle { PresentationStyle::Unspecified };
78};
79
80} // namespace WebCore
