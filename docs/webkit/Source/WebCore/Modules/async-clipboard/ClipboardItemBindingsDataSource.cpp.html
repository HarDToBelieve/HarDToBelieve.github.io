49static RefPtr<Document> documentFromClipboard(const Clipboard* clipboard)
50{
51 if (!clipboard)
52 return nullptr;
53
54 auto* frame = clipboard->frame();
55 if (!frame)
56 return nullptr;
57
58 return frame->document();
59}

61static FileReaderLoader::ReadType readTypeForMIMEType(const String& type)
62{
63 if (type == "text/uri-list"_s || type == "text/plain"_s || type == "text/html"_s)
64 return FileReaderLoader::ReadAsText;
65 return FileReaderLoader::ReadAsArrayBuffer;
66}

83void ClipboardItemBindingsDataSource::getType(const String& type, Ref<DeferredPromise>&& promise)
84{
85 auto matchIndex = m_itemPromises.findMatching([&] (auto& item) {
86 return type == item.key;
87 });
88
89 if (matchIndex == notFound) {
90 promise->reject(NotFoundError);
91 return;
92 }
93
94 auto itemPromise = m_itemPromises[matchIndex].value;
95 itemPromise->whenSettled([itemPromise, promise = makeRefPtr(promise.get()), type] () mutable {
96 if (itemPromise->status() != DOMPromise::Status::Fulfilled) {
97 promise->reject(AbortError);
98 return;
99 }
100
101 auto result = itemPromise->result();
102 if (!result) {
103 promise->reject(TypeError);
104 return;
105 }
106
107 String string;
108 result.getString(itemPromise->globalObject(), string);
109 if (!string.isNull()) {
110 promise->resolve<IDLInterface<Blob>>(ClipboardItem::blobFromString(promise->scriptExecutionContext(), string, type));
111 return;
112 }
113
114 if (!result.isObject()) {
115 promise->reject(TypeError);
116 return;
117 }
118
119 if (auto blob = JSBlob::toWrapped(result.getObject()->vm(), result.getObject()))
120 promise->resolve<IDLInterface<Blob>>(*blob);
121 else
122 promise->reject(TypeError);
123 });
124}

126void ClipboardItemBindingsDataSource::collectDataForWriting(Clipboard& destination, CompletionHandler<void(std::optional<PasteboardCustomData>)>&& completion)
127{
128 m_itemTypeLoaders.clear();
129 ASSERT(!m_completionHandler);
130 m_completionHandler = WTFMove(completion);
131 m_writingDestination = makeWeakPtr(destination);
132 m_numberOfPendingClipboardTypes = m_itemPromises.size();
133 m_itemTypeLoaders = m_itemPromises.map([&] (auto& typeAndItem) {
134 auto type = typeAndItem.key;
135 auto itemTypeLoader = ClipboardItemTypeLoader::create(type, [this, protectedItem = makeRef(m_item)] {
136 ASSERT(m_numberOfPendingClipboardTypes);
137 if (!--m_numberOfPendingClipboardTypes)
138 invokeCompletionHandler();
139 });
140
141 auto promise = typeAndItem.value;
142 promise->whenSettled([this, protectedItem = makeRefPtr(m_item), destination = m_writingDestination, promise, type, weakItemTypeLoader = makeWeakPtr(itemTypeLoader.ptr())] () mutable {
143 if (!weakItemTypeLoader)
144 return;
145
146 auto itemTypeLoader = makeRef(*weakItemTypeLoader);
147 ASSERT_UNUSED(this, notFound != m_itemTypeLoaders.findMatching([&] (auto& loader) { return loader.ptr() == itemTypeLoader.ptr(); }));
148
149 auto result = promise->result();
150 if (!result) {
151 itemTypeLoader->didFailToResolve();
152 return;
153 }
154
155 auto clipboard = makeRefPtr(destination.get());
156 if (!clipboard) {
157 itemTypeLoader->didFailToResolve();
158 return;
159 }
160
161 if (!clipboard->scriptExecutionContext()) {
162 itemTypeLoader->didFailToResolve();
163 return;
164 }
165
166 String text;
167 result.getString(promise->globalObject(), text);
168 if (!text.isNull()) {
169 itemTypeLoader->didResolveToString(text);
170 return;
171 }
172
173 if (!result.isObject()) {
174 itemTypeLoader->didFailToResolve();
175 return;
176 }
177
178 if (auto blob = makeRefPtr(JSBlob::toWrapped(result.getObject()->vm(), result.getObject())))
179 itemTypeLoader->didResolveToBlob(*clipboard->scriptExecutionContext(), blob.releaseNonNull());
180 else
181 itemTypeLoader->didFailToResolve();
182 });
183
184 return itemTypeLoader;
185 });
186
187 if (!m_numberOfPendingClipboardTypes)
188 invokeCompletionHandler();
189}

191void ClipboardItemBindingsDataSource::invokeCompletionHandler()
192{
193 if (!m_completionHandler) {
194 ASSERT_NOT_REACHED();
195 return;
196 }
197
198 auto completionHandler = std::exchange(m_completionHandler, { });
199 auto itemTypeLoaders = std::exchange(m_itemTypeLoaders, { });
200 auto clipboard = makeRefPtr(m_writingDestination.get());
201 m_writingDestination = nullptr;
202
203 auto document = documentFromClipboard(clipboard.get());
204 if (!document) {
205 completionHandler(std::nullopt);
206 return;
207 }
208
209 PasteboardCustomData customData;
210 for (auto& itemTypeLoader : itemTypeLoaders) {
211 auto type = itemTypeLoader->type();
212 auto& data = itemTypeLoader->data();
213 if (WTF::holds_alternative<String>(data) && !!WTF::get<String>(data))
214 customData.writeString(type, WTF::get<String>(data));
215 else if (WTF::holds_alternative<Ref<SharedBuffer>>(data))
216 customData.writeData(type, WTF::get<Ref<SharedBuffer>>(data).copyRef());
217 else {
218 completionHandler(std::nullopt);
219 return;
220 }
221 }
222
223 customData.setOrigin(document->originIdentifierForPasteboard());
224 completionHandler(WTFMove(customData));
225}

241void ClipboardItemBindingsDataSource::ClipboardItemTypeLoader::didFinishLoading()
242{
243 ASSERT(m_blobLoader);
244 auto stringResult = readTypeForMIMEType(m_type) == FileReaderLoader::ReadAsText ? m_blobLoader->stringResult() : nullString();
245 if (!stringResult.isNull())
246 m_data = { stringResult };
247 else if (auto arrayBuffer = m_blobLoader->arrayBufferResult())
248 m_data = { SharedBuffer::create(static_cast<const char*>(arrayBuffer->data()), arrayBuffer->byteLength()) };
249 m_blobLoader = nullptr;
250 invokeCompletionHandler();
251}
252
253void ClipboardItemBindingsDataSource::ClipboardItemTypeLoader::didFail(ExceptionCode)
254{
255 ASSERT(m_blobLoader);
256 m_blobLoader = nullptr;
257 invokeCompletionHandler();
258}

260void ClipboardItemBindingsDataSource::ClipboardItemTypeLoader::sanitizeDataIfNeeded()
261{
262 if (m_type == "text/html"_s) {
263 String markupToSanitize;
264 if (WTF::holds_alternative<Ref<SharedBuffer>>(m_data)) {
265 auto& buffer = WTF::get<Ref<SharedBuffer>>(m_data);
266 markupToSanitize = String::fromUTF8(buffer->data(), buffer->size());
267 } else if (WTF::holds_alternative<String>(m_data))
268 markupToSanitize = WTF::get<String>(m_data);
269
270 if (markupToSanitize.isEmpty())
271 return;
272
273 m_data = { sanitizeMarkup(markupToSanitize) };
274 }
275
276 if (m_type == "image/png"_s) {
277 RefPtr<SharedBuffer> bufferToSanitize;
278 if (WTF::holds_alternative<Ref<SharedBuffer>>(m_data))
279 bufferToSanitize = WTF::get<Ref<SharedBuffer>>(m_data).ptr();
280 else if (WTF::holds_alternative<String>(m_data))
281 bufferToSanitize = utf8Buffer(WTF::get<String>(m_data));
282
283 if (!bufferToSanitize || bufferToSanitize->isEmpty())
284 return;
285
286 auto bitmapImage = BitmapImage::create();
287 bitmapImage->setData(WTFMove(bufferToSanitize), true);
288 auto imageBuffer = ImageBuffer::create(bitmapImage->size(), RenderingMode::Unaccelerated, 1, DestinationColorSpace::SRGB(), PixelFormat::BGRA8);
289 if (!imageBuffer) {
290 m_data = { nullString() };
291 return;
292 }
293
294 imageBuffer->context().drawImage(bitmapImage.get(), FloatPoint::zero());
295 m_data = { SharedBuffer::create(imageBuffer->toData("image/png"_s)) };
296 }
297}

299void ClipboardItemBindingsDataSource::ClipboardItemTypeLoader::invokeCompletionHandler()
300{
301 if (auto completion = WTFMove(m_completionHandler)) {
302 sanitizeDataIfNeeded();
303 completion();
304 }
305}

307void ClipboardItemBindingsDataSource::ClipboardItemTypeLoader::didResolveToBlob(ScriptExecutionContext& context, Ref<Blob>&& blob)
308{
309 ASSERT(!m_blobLoader);
310 m_blobLoader = makeUnique<FileReaderLoader>(readTypeForMIMEType(m_type), this);
311 m_blobLoader->start(&context, WTFMove(blob));
312}
313
314void ClipboardItemBindingsDataSource::ClipboardItemTypeLoader::didFailToResolve()
315{
316 ASSERT(!m_blobLoader);
317 invokeCompletionHandler();
318}
319
320void ClipboardItemBindingsDataSource::ClipboardItemTypeLoader::didResolveToString(const String& text)
321{
322 ASSERT(!m_blobLoader);
323 m_data = { text };
324 invokeCompletionHandler();
325}
