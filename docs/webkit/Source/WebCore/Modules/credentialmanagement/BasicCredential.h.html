34namespace WebCore {
35
36class BasicCredential : public RefCounted<BasicCredential> {
37public:
38 enum class Type {
39 PublicKey,
40 };
41
42 enum class Discovery {
43 CredentialStore,
44 Remote,
45 };
46
47 BasicCredential(const String&, Type, Discovery);
48 virtual ~BasicCredential();
49
50 virtual Type credentialType() const = 0;
51
52 const String& id() const { return m_id; }
53 String type() const;
54 Discovery discovery() const { return m_discovery; }
55
56private:
57 String m_id;
58 Type m_type;
59 Discovery m_discovery;
60};
61
62} // namespace WebCore
