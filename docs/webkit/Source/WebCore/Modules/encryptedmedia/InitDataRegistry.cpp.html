48 const uint32_t kCencMaxBoxSize = 64 * KB;
49 // ContentEncKeyID has this EBML code [47][E2] in WebM,
50 // as per spec the size of the ContentEncKeyID is encoded on 16 bits.
51 // https://matroska.org/technical/specs/index.html#ContentEncKeyID/
52 const uint32_t kWebMMaxContentEncKeyIDSize = 64 * KB; // 2^16
53 const uint32_t kKeyIdsMinKeyIdSizeInBytes = 1;
54 const uint32_t kKeyIdsMaxKeyIdSizeInBytes = 512;
55}
56
57static std::optional<Vector<Ref<SharedBuffer>>> extractKeyIDsKeyids(const SharedBuffer& buffer)
58{
59 // 1. Format
60 // https://w3c.github.io/encrypted-media/format-registry/initdata/keyids.html#format
61 if (buffer.size() > std::numeric_limits<unsigned>::max())
62 return std::nullopt;
63 String json { buffer.data(), static_cast<unsigned>(buffer.size()) };
64
65 auto value = JSON::Value::parseJSON(json);
66 if (!value)
67 return std::nullopt;
68
69 auto object = value->asObject();
70 if (!object)
71 return std::nullopt;
72
73 auto kidsArray = object->getArray("kids"_s);
74 if (!kidsArray)
75 return std::nullopt;
76
77 Vector<Ref<SharedBuffer>> keyIDs;
78 for (auto& value : *kidsArray) {
79 auto keyID = value->asString();
80 if (!keyID)
81 continue;
82
83 auto keyIDData = base64URLDecode(keyID);
84 if (!keyIDData)
85 continue;
86
87 if (keyIDData->size() < kKeyIdsMinKeyIdSizeInBytes || keyIDData->size() > kKeyIdsMaxKeyIdSizeInBytes)
88 return std::nullopt;
89
90 keyIDs.append(SharedBuffer::create(WTFMove(*keyIDData)));
91 }
92
93 return keyIDs;
94}

96static RefPtr<SharedBuffer> sanitizeKeyids(const SharedBuffer& buffer)
97{
98 // 1. Format
99 // https://w3c.github.io/encrypted-media/format-registry/initdata/keyids.html#format
100 auto keyIDBuffer = extractKeyIDsKeyids(buffer);
101 if (!keyIDBuffer)
102 return nullptr;
103
104 auto object = JSON::Object::create();
105 auto kidsArray = JSON::Array::create();
106 for (auto& buffer : keyIDBuffer.value())
107 kidsArray->pushString(base64URLEncodeToString(buffer->data(), buffer->size()));
108 object->setArray("kids", WTFMove(kidsArray));
109
110 CString jsonData = object->toJSONString().utf8();
111 return SharedBuffer::create(jsonData.data(), jsonData.length());
112}

114std::optional<Vector<std::unique_ptr<ISOProtectionSystemSpecificHeaderBox>>> InitDataRegistry::extractPsshBoxesFromCenc(const SharedBuffer& buffer)
115{
116 // 4. Common SystemID and PSSH Box Format
117 // https://w3c.github.io/encrypted-media/format-registry/initdata/cenc.html#common-system
118 if (buffer.size() >= kCencMaxBoxSize)
119 return std::nullopt;
120
121 unsigned offset = 0;
122 Vector<std::unique_ptr<ISOProtectionSystemSpecificHeaderBox>> psshBoxes;
123
124 auto view = JSC::DataView::create(buffer.tryCreateArrayBuffer(), offset, buffer.size());
125 while (auto optionalBoxType = ISOBox::peekBox(view, offset)) {
126 auto& boxTypeName = optionalBoxType.value().first;
127 auto& boxSize = optionalBoxType.value().second;
128
129 if (boxTypeName != ISOProtectionSystemSpecificHeaderBox::boxTypeName() || boxSize > buffer.size())
130 return std::nullopt;
131
132 auto systemID = ISOProtectionSystemSpecificHeaderBox::peekSystemID(view, offset);
133#if HAVE(FAIRPLAYSTREAMING_CENC_INITDATA)
134 if (systemID == ISOFairPlayStreamingPsshBox::fairPlaySystemID()) {
135 auto fpsPssh = makeUnique<ISOFairPlayStreamingPsshBox>();
136 if (!fpsPssh->read(view, offset))
137 return std::nullopt;
138 psshBoxes.append(WTFMove(fpsPssh));
139 continue;
140 }
141#else
142 UNUSED_PARAM(systemID);
143#endif
144 auto psshBox = makeUnique<ISOProtectionSystemSpecificHeaderBox>();
145 if (!psshBox->read(view, offset))
146 return std::nullopt;
147
148 psshBoxes.append(WTFMove(psshBox));
149 }
150
151 return psshBoxes;
152}

154std::optional<Vector<Ref<SharedBuffer>>> InitDataRegistry::extractKeyIDsCenc(const SharedBuffer& buffer)
155{
156 Vector<Ref<SharedBuffer>> keyIDs;
157
158 auto psshBoxes = extractPsshBoxesFromCenc(buffer);
159 if (!psshBoxes)
160 return std::nullopt;
161
162 for (auto& psshBox : psshBoxes.value()) {
163 ASSERT(psshBox);
164 if (!psshBox)
165 return std::nullopt;
166
167#if HAVE(FAIRPLAYSTREAMING_CENC_INITDATA)
168 if (is<ISOFairPlayStreamingPsshBox>(*psshBox)) {
169 ISOFairPlayStreamingPsshBox& fpsPssh = downcast<ISOFairPlayStreamingPsshBox>(*psshBox);
170
171 FourCC scheme = fpsPssh.initDataBox().info().scheme();
172 if (CDMPrivateFairPlayStreaming::validFairPlayStreamingSchemes().contains(scheme)) {
173 for (const auto& request : fpsPssh.initDataBox().requests()) {
174 auto& keyID = request.requestInfo().keyID();
175 keyIDs.append(SharedBuffer::create(keyID.data(), keyID.size()));
176 }
177 }
178 }
179#endif
180
181 for (auto& value : psshBox->keyIDs())
182 keyIDs.append(SharedBuffer::create(WTFMove(value)));
183 }
184
185 return keyIDs;
186}

188RefPtr<SharedBuffer> InitDataRegistry::sanitizeCenc(const SharedBuffer& buffer)
189{
190 // 4. Common SystemID and PSSH Box Format
191 // https://w3c.github.io/encrypted-media/format-registry/initdata/cenc.html#common-system
192 if (!extractKeyIDsCenc(buffer))
193 return nullptr;
194
195 return buffer.copy();
196}

198static RefPtr<SharedBuffer> sanitizeWebM(const SharedBuffer& buffer)
199{
200 // Check if the buffer is a valid WebM initData.
201 // The WebM initData is the ContentEncKeyID, so should be less than kWebMMaxContentEncKeyIDSize.
202 if (buffer.isEmpty() || buffer.size() > kWebMMaxContentEncKeyIDSize)
203 return nullptr;
204
205 return buffer.copy();
206}
207
208static std::optional<Vector<Ref<SharedBuffer>>> extractKeyIDsWebM(const SharedBuffer& buffer)
209{
210 Vector<Ref<SharedBuffer>> keyIDs;
211 RefPtr<SharedBuffer> sanitizedBuffer = sanitizeWebM(buffer);
212 if (!sanitizedBuffer)
213 return std::nullopt;
214
215 // 1. Format
216 // https://w3c.github.io/encrypted-media/format-registry/initdata/webm.html#format
217 keyIDs.append(sanitizedBuffer.releaseNonNull());
218 return keyIDs;
219}

227InitDataRegistry::InitDataRegistry()
228{
229 registerInitDataType("keyids", { sanitizeKeyids, extractKeyIDsKeyids });
230 registerInitDataType("cenc", { sanitizeCenc, extractKeyIDsCenc });
231 registerInitDataType("webm", { sanitizeWebM, extractKeyIDsWebM });
232}

236RefPtr<SharedBuffer> InitDataRegistry::sanitizeInitData(const AtomString& initDataType, const SharedBuffer& buffer)
237{
238 auto iter = m_types.find(initDataType);
239 if (iter == m_types.end() || !iter->value.sanitizeInitData)
240 return nullptr;
241 return iter->value.sanitizeInitData(buffer);
242}
243
244std::optional<Vector<Ref<SharedBuffer>>> InitDataRegistry::extractKeyIDs(const AtomString& initDataType, const SharedBuffer& buffer)
245{
246 auto iter = m_types.find(initDataType);
247 if (iter == m_types.end() || !iter->value.sanitizeInitData)
248 return std::nullopt;
249 return iter->value.extractKeyIDs(buffer);
250}

252void InitDataRegistry::registerInitDataType(const AtomString& initDataType, InitDataTypeCallbacks&& callbacks)
253{
254 ASSERT(!m_types.contains(initDataType));
255 m_types.set(initDataType, WTFMove(callbacks));
256}
