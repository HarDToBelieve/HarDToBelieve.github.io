134void MediaKeySession::generateRequest(const AtomString& initDataType, const BufferSource& initData, Ref<DeferredPromise>&& promise)
135{
136 // https://w3c.github.io/encrypted-media/#dom-mediakeysession-generaterequest
137 // W3C Editor's Draft 09 November 2016
138
139 // When this method is invoked, the user agent must run the following steps:
140 // 1. If this object is closed, return a promise rejected with an InvalidStateError.
141 // 2. If this object's uninitialized value is false, return a promise rejected with an InvalidStateError.
142 auto identifier = LOGIDENTIFIER;
143 INFO_LOG(identifier, "initDataType(", initDataType, "), initData.length(", initData.length(), ")");
144
145 if (m_closed || !m_uninitialized) {
146 INFO_LOG(identifier, "Rejected: closed(", m_closed, ") or !uninitialized(", !m_uninitialized, ")");
147 promise->reject(InvalidStateError);
148 return;
149 }
150
151 // 3. Let this object's uninitialized value be false.
152 m_uninitialized = false;
153
154 // 4. If initDataType is the empty string, return a promise rejected with a newly created TypeError.
155 // 5. If initData is an empty array, return a promise rejected with a newly created TypeError.
156 if (initDataType.isEmpty() || !initData.length()) {
157 INFO_LOG(identifier, "Rejected: initDataType empty(", initDataType.isEmpty(), ") or initData empty(", !initData.length(), ")");
158 promise->reject(TypeError);
159 return;
160 }
161
162 // 6. If the Key System implementation represented by this object's cdm implementation value does not support
163 // initDataType as an Initialization Data Type, return a promise rejected with a NotSupportedError. String
164 // comparison is case-sensitive.
165 if (!m_implementation->supportsInitDataType(initDataType)) {
166 INFO_LOG(identifier, "Rejected: initDataType(", initDataType, ") unsupported");
167 promise->reject(NotSupportedError);
168 return;
169 }
170
171 // 7. Let init data be a copy of the contents of the initData parameter.
172 // 8. Let session type be this object's session type.
173 // 9. Let promise be a new promise.
174 // 10. Run the following steps in parallel:
175 queueTaskKeepingObjectAlive(*this, TaskSource::Networking, [this, weakThis = makeWeakPtr(*this), initData = SharedBuffer::create(initData.data(), initData.length()), initDataType, promise = WTFMove(promise), identifier = WTFMove(identifier)] () mutable {
176 // 10.1. If the init data is not valid for initDataType, reject promise with a newly created TypeError.
177 // 10.2. Let sanitized init data be a validated and sanitized version of init data.
178 RefPtr<SharedBuffer> sanitizedInitData = m_implementation->sanitizeInitData(initDataType, initData);
179
180 // 10.3. If the preceding step failed, reject promise with a newly created TypeError.
181 if (!sanitizedInitData) {
182 INFO_LOG(identifier, "::task() Rejected: cannot sanitize init data");
183 promise->reject(TypeError);
184 return;
185 }
186
187 // 10.4. If sanitized init data is empty, reject promise with a NotSupportedError.
188 if (sanitizedInitData->isEmpty()) {
189 INFO_LOG(identifier, "::task() Rejected: empty sanitized init data");
190 promise->reject(NotSupportedError);
191 return;
192 }
193
194 // 10.5. Let session id be the empty string.
195 // 10.6. Let message be null.
196 // 10.7. Let message type be null.
197 // 10.8. Let cdm be the CDM instance represented by this object's cdm instance value.
198 // 10.9. Use the cdm to execute the following steps:
199 // 10.9.1. If the sanitized init data is not supported by the cdm, reject promise with a NotSupportedError.
200 if (!m_implementation->supportsInitData(initDataType, *sanitizedInitData)) {
201 INFO_LOG(identifier, "::task() Rejected: unsupported initDataType (", initDataType, ") or sanitized initData");
202 promise->reject(NotSupportedError);
203 return;
204 }
205
206 // 10.9.2 Follow the steps for the value of session type from the following list:
207 // "temporary"
208 // Let requested license type be a temporary non-persistable license.
209 // "persistent-license"
210 // Let requested license type be a persistable license.
211 // "persistent-usage-record"
212 // 1. Initialize this object's record of key usage as follows.
213 // Set the list of key IDs known to the session to an empty list.
214 // Set the first decrypt time to null.
215 // Set the latest decrypt time to null.
216 // 2. Let requested license type be a non-persistable license that will
217 // persist a record of key usage.
218
219 if (m_sessionType == MediaKeySessionType::PersistentUsageRecord) {
220 m_recordOfKeyUsage.clear();
221 m_firstDecryptTime = 0;
222 m_latestDecryptTime = 0;
223 }
224
225 m_instanceSession->requestLicense(m_sessionType, initDataType, sanitizedInitData.releaseNonNull(), [this, weakThis = makeWeakPtr(*weakThis), promise = WTFMove(promise), identifier = WTFMove(identifier)] (Ref<SharedBuffer>&& message, const String& sessionId, bool needsIndividualization, CDMInstanceSession::SuccessValue succeeded) mutable {
226 if (!weakThis)
227 return;
228
229 // 10.9.3. Let session id be a unique Session ID string.
230
231 MediaKeyMessageType messageType;
232 if (!needsIndividualization) {
233 // 10.9.4. If a license request for the requested license type can be generated based on the sanitized init data:
234 // 10.9.4.1. Let message be a license request for the requested license type generated based on the sanitized init data interpreted per initDataType.
235 // 10.9.4.2. Let message type be "license-request".
236 messageType = MediaKeyMessageType::LicenseRequest;
237 } else {
238 // 10.9.5. Otherwise:
239 // 10.9.5.1. Let message be the request that needs to be processed before a license request request for the requested license
240 // type can be generated based on the sanitized init data.
241 // 10.9.5.2. Let message type reflect the type of message, either "license-request" or "individualization-request".
242 messageType = MediaKeyMessageType::IndividualizationRequest;
243 }
244
245 // 10.10. Queue a task to run the following steps:
246 queueTaskKeepingObjectAlive(*this, TaskSource::Networking, [this, promise = WTFMove(promise), message = WTFMove(message), messageType, sessionId, succeeded, identifier = WTFMove(identifier)] () mutable {
247 // 10.10.1. If any of the preceding steps failed, reject promise with a new DOMException whose name is the appropriate error name.
248 if (succeeded == CDMInstanceSession::SuccessValue::Failed) {
249 INFO_LOG(identifier, "::task() Rejected: failed to request license");
250 promise->reject(NotSupportedError);
251 return;
252 }
253 // 10.10.2. Set the sessionId attribute to session id.
254 m_sessionId = sessionId;
255
256 // 10.9.3. Let this object's callable value be true.
257 m_callable = true;
258
259 // 10.9.3. Run the Queue a "message" Event algorithm on the session, providing message type and message.
260 enqueueMessage(messageType, message);
261
262 // 10.9.3. Resolve promise.
263 INFO_LOG(identifier, "::task() Resolved");
264 promise->resolve();
265 });
266 });
267 });
268
269 // 11. Return promise.
270}

272void MediaKeySession::load(const String& sessionId, Ref<DeferredPromise>&& promise)
273{
274 // https://w3c.github.io/encrypted-media/#dom-mediakeysession-load
275 // W3C Editor's Draft 09 November 2016
276
277 auto identifier = LOGIDENTIFIER;
278 INFO_LOG(identifier, "sessionId(", sessionId, ")");
279
280 // 1. If this object is closed, return a promise rejected with an InvalidStateError.
281 // 2. If this object's uninitialized value is false, return a promise rejected with an InvalidStateError.
282 if (m_closed || !m_uninitialized) {
283 INFO_LOG(identifier, "Rejected: closed(", m_closed, ") or !uninitialized(", !m_uninitialized, ")");
284 promise->reject(InvalidStateError);
285 return;
286 }
287
288 // 3. Let this object's uninitialized value be false.
289 m_uninitialized = false;
290
291 // 4. If sessionId is the empty string, return a promise rejected with a newly created TypeError.
292 // 5. If the result of running the Is persistent session type? algorithm on this object's session type is false, return a promise rejected with a newly created TypeError.
293 if (sessionId.isEmpty() || m_sessionType == MediaKeySessionType::Temporary) {
294 INFO_LOG(identifier, "Rejected: sessionID empty(", sessionId.isEmpty(), ") or sessionType == Temporary (", m_sessionType == MediaKeySessionType::Temporary, ")");
295 promise->reject(TypeError);
296 return;
297 }
298
299 // 6. Let origin be the origin of this object's Document.
300 // This is retrieved in the following task.
301
302 // 7. Let promise be a new promise.
303 // 8. Run the following steps in parallel:
304 queueTaskKeepingObjectAlive(*this, TaskSource::Networking, [this, weakThis = makeWeakPtr(*this), sessionId, promise = WTFMove(promise), identifier = WTFMove(identifier)] () mutable {
305 // 8.1. Let sanitized session ID be a validated and/or sanitized version of sessionId.
306 // 8.2. If the preceding step failed, or if sanitized session ID is empty, reject promise with a newly created TypeError.
307 std::optional<String> sanitizedSessionId = m_implementation->sanitizeSessionId(sessionId);
308 if (!sanitizedSessionId || sanitizedSessionId->isEmpty()) {
309 INFO_LOG(identifier, "Rejected: sanitizedSSessionID empty");
310 promise->reject(TypeError);
311 return;
312 }
313
314 // 8.3. If there is a MediaKeySession object that is not closed in this object's Document whose sessionId attribute is sanitized session ID, reject promise with a QuotaExceededError.
315 // FIXME: This needs a global MediaKeySession tracker.
316
317 String origin;
318 if (auto* document = downcast<Document>(scriptExecutionContext()))
319 origin = document->securityOrigin().toString();
320
321 // 8.4. Let expiration time be NaN.
322 // 8.5. Let message be null.
323 // 8.6. Let message type be null.
324 // 8.7. Let cdm be the CDM instance represented by this object's cdm instance value.
325 // 8.8. Use the cdm to execute the following steps:
326 m_instanceSession->loadSession(m_sessionType, *sanitizedSessionId, origin, [this, weakThis = makeWeakPtr(*weakThis), promise = WTFMove(promise), sanitizedSessionId = *sanitizedSessionId, identifier = WTFMove(identifier)] (std::optional<CDMInstanceSession::KeyStatusVector>&& knownKeys, std::optional<double>&& expiration, std::optional<CDMInstanceSession::Message>&& message, CDMInstanceSession::SuccessValue succeeded, CDMInstanceSession::SessionLoadFailure failure) mutable {
327 // 8.8.1. If there is no data stored for the sanitized session ID in the origin, resolve promise with false and abort these steps.
328 // 8.8.2. If the stored session's session type is not the same as the current MediaKeySession session type, reject promise with a newly created TypeError.
329 // 8.8.3. Let session data be the data stored for the sanitized session ID in the origin. This must not include data from other origin(s) or that is not associated with an origin.
330 // 8.8.4. If there is a MediaKeySession object that is not closed in any Document and that represents the session data, reject promise with a QuotaExceededError.
331 // 8.8.5. Load the session data.
332 // 8.8.6. If the session data indicates an expiration time for the session, let expiration time be the expiration time in milliseconds since 01 January 1970 UTC.
333 // 8.8.7. If the CDM needs to send a message:
334 // 8.8.7.1. Let message be a message generated by the CDM based on the session data.
335 // 8.8.7.2. Let message type be the appropriate MediaKeyMessageType for the message.
336 // NOTE: Steps 8.8.1. through 8.8.7. should be implemented in CDMInstance.
337
338 if (succeeded == CDMInstanceSession::SuccessValue::Failed) {
339 switch (failure) {
340 case CDMInstanceSession::SessionLoadFailure::NoSessionData:
341 INFO_LOG(identifier, "::task() Resolved: NoSessionData");
342 promise->resolve<IDLBoolean>(false);
343 return;
344 case CDMInstanceSession::SessionLoadFailure::MismatchedSessionType:
345 INFO_LOG(identifier, "::task() Rejected: MismatchedSessionType");
346 promise->reject(TypeError);
347 return;
348 case CDMInstanceSession::SessionLoadFailure::QuotaExceeded:
349 INFO_LOG(identifier, "::task() Rejected: QuotaExceeded");
350 promise->reject(QuotaExceededError);
351 return;
352 case CDMInstanceSession::SessionLoadFailure::None:
353 case CDMInstanceSession::SessionLoadFailure::Other:
354 // In any other case, the session load failure will cause a rejection in the following task.
355 break;
356 }
357 }
358
359 // 8.9. Queue a task to run the following steps:
360 queueTaskKeepingObjectAlive(*this, TaskSource::Networking, [this, knownKeys = WTFMove(knownKeys), expiration = WTFMove(expiration), message = WTFMove(message), sanitizedSessionId, succeeded, promise = WTFMove(promise), identifier = WTFMove(identifier)] () mutable {
361 // 8.9.1. If any of the preceding steps failed, reject promise with a the appropriate error name.
362 if (succeeded == CDMInstanceSession::SuccessValue::Failed) {
363 INFO_LOG(identifier, "::task() Rejected: Other failure");
364 promise->reject(NotSupportedError);
365 return;
366 }
367
368 // 8.9.2. Set the sessionId attribute to sanitized session ID.
369 // 8.9.3. Let this object's callable value be true.
370 m_sessionId = sanitizedSessionId;
371 m_callable = true;
372
373 // 8.9.4. If the loaded session contains information about any keys (there are known keys), run the Update Key Statuses algorithm on the session, providing each key's key ID along with the appropriate MediaKeyStatus.
374 if (knownKeys)
375 updateKeyStatuses(WTFMove(*knownKeys));
376
377 // 8.9.5. Run the Update Expiration algorithm on the session, providing expiration time.
378 // This must be run, and NaN is the default value if the CDM instance doesn't provide one.
379 updateExpiration(expiration.value_or(std::numeric_limits<double>::quiet_NaN()));
380
381 // 8.9.6. If message is not null, run the Queue a "message" Event algorithm on the session, providing message type and message.
382 if (message)
383 enqueueMessage(message->first, WTFMove(message->second));
384
385 // 8.9.7. Resolve promise with true.
386 INFO_LOG(identifier, "::task() Resolved");
387 promise->resolve<IDLBoolean>(true);
388 });
389 });
390 });
391
392 // 9. Return promise.
393}

395void MediaKeySession::update(const BufferSource& response, Ref<DeferredPromise>&& promise)
396{
397 // https://w3c.github.io/encrypted-media/#dom-mediakeysession-update
398 // W3C Editor's Draft 09 November 2016
399
400 // When this method is invoked, the user agent must run the following steps:
401 // 1. If this object is closed, return a promise rejected with an InvalidStateError.
402 // 2. If this object's callable value is false, return a promise rejected with an InvalidStateError.
403 auto identifier = LOGIDENTIFIER;
404 INFO_LOG(identifier, "response.length(", response.length(), ")");
405
406 if (m_closed || !m_callable) {
407 INFO_LOG(identifier, "Rejected: closed(", m_closed, ") or !callable(", !m_callable, ")");
408 promise->reject(InvalidStateError);
409 return;
410 }
411
412 // 3. If response is an empty array, return a promise rejected with a newly created TypeError.
413 if (!response.length()) {
414 INFO_LOG(identifier, "Rejected: empty response");
415 promise->reject(TypeError);
416 return;
417 }
418
419 // 4. Let response copy be a copy of the contents of the response parameter.
420 // 5. Let promise be a new promise.
421 // 6. Run the following steps in parallel:
422 queueTaskKeepingObjectAlive(*this, TaskSource::Networking, [this, weakThis = makeWeakPtr(*this), response = SharedBuffer::create(response.data(), response.length()), promise = WTFMove(promise), identifier = WTFMove(identifier)] () mutable {
423 // 6.1. Let sanitized response be a validated and/or sanitized version of response copy.
424 RefPtr<SharedBuffer> sanitizedResponse = m_implementation->sanitizeResponse(response);
425
426 // 6.2. If the preceding step failed, or if sanitized response is empty, reject promise with a newly created TypeError.
427 if (!sanitizedResponse || sanitizedResponse->isEmpty()) {
428 INFO_LOG(identifier, "::task - Rejected: empty sanitized response");
429 promise->reject(TypeError);
430 return;
431 }
432
433 // 6.3. Let message be null.
434 // 6.4. Let message type be null.
435 // 6.5. Let session closed be false.
436 // 6.6. Let cdm be the CDM instance represented by this object's cdm instance value.
437 // 6.7. Use the cdm to execute the following steps:
438 m_instanceSession->updateLicense(m_sessionId, m_sessionType, sanitizedResponse.releaseNonNull(), [this, weakThis = makeWeakPtr(*weakThis), promise = WTFMove(promise), identifier = WTFMove(identifier)] (bool sessionWasClosed, std::optional<CDMInstanceSession::KeyStatusVector>&& changedKeys, std::optional<double>&& changedExpiration, std::optional<CDMInstanceSession::Message>&& message, CDMInstanceSession::SuccessValue succeeded) mutable {
439 if (!weakThis)
440 return;
441
442 // 6.7.1. If the format of sanitized response is invalid in any way, reject promise with a newly created TypeError.
443 // 6.7.2. Process sanitized response, following the stipulation for the first matching condition from the following list:
444 // If sanitized response contains a license or key(s)
445 // Process sanitized response, following the stipulation for the first matching condition from the following list:
446 // If sessionType is "temporary" and sanitized response does not specify that session data, including any license, key(s), or similar session data it contains, should be stored
447 // Process sanitized response, not storing any session data.
448 // If sessionType is "persistent-license" and sanitized response contains a persistable license
449 // Process sanitized response, storing the license/key(s) and related session data contained in sanitized response. Such data must be stored such that only the origin of this object's Document can access it.
450 // If sessionType is "persistent-usage-record" and sanitized response contains a non-persistable license
451 // Run the following steps:
452 // 6.7.2.3.1. Process sanitized response, not storing any session data.
453 // 6.7.2.3.2. If processing sanitized response results in the addition of keys to the set of known keys, add the key IDs of these keys to this object's record of key usage.
454 // Otherwise
455 // Reject promise with a newly created TypeError.
456 // If sanitized response contains a record of license destruction acknowledgement and sessionType is "persistent-license"
457 // Run the following steps:
458 // 6.7.2.1. Close the key session and clear all stored session data associated with this object, including the sessionId and record of license destruction.
459 // 6.7.2.2. Set session closed to true.
460 // Otherwise
461 // Process sanitized response, not storing any session data.
462 // NOTE: Steps 6.7.1. and 6.7.2. should be implemented in CDMInstance.
463
464 if (succeeded == CDMInstanceSession::SuccessValue::Failed) {
465 INFO_LOG(identifier, "::task() Rejected");
466 promise->reject(TypeError);
467 return;
468 }
469
470 // 6.7.3. If a message needs to be sent to the server, execute the following steps:
471 // 6.7.3.1. Let message be that message.
472 // 6.7.3.2. Let message type be the appropriate MediaKeyMessageType for the message.
473 // 6.8. Queue a task to run the following steps:
474 queueTaskKeepingObjectAlive(*this, TaskSource::Networking, [this, sessionWasClosed, changedKeys = WTFMove(changedKeys), changedExpiration = WTFMove(changedExpiration), message = WTFMove(message), promise = WTFMove(promise), identifier = WTFMove(identifier)] () mutable {
475 // 6.8.1.
476 if (sessionWasClosed) {
477 // If session closed is true:
478 // Run the Session Closed algorithm on this object.
479 sessionClosed();
480 } else {
481 // Otherwise:
482 // Run the following steps:
483 // 6.8.1.1. If the set of keys known to the CDM for this object changed or the status of any key(s) changed, run the Update Key Statuses
484 // algorithm on the session, providing each known key's key ID along with the appropriate MediaKeyStatus. Should additional
485 // processing be necessary to determine with certainty the status of a key, use "status-pending". Once the additional processing
486 // for one or more keys has completed, run the Update Key Statuses algorithm again with the actual status(es).
487 if (changedKeys)
488 updateKeyStatuses(WTFMove(*changedKeys));
489
490 // 6.8.1.2. If the expiration time for the session changed, run the Update Expiration algorithm on the session, providing the new expiration time.
491 if (changedExpiration)
492 updateExpiration(*changedExpiration);
493
494 // 6.8.1.3. If any of the preceding steps failed, reject promise with a new DOMException whose name is the appropriate error name.
495 // FIXME: At this point the implementations of preceding steps can't fail.
496
497 // 6.8.1.4. If message is not null, run the Queue a "message" Event algorithm on the session, providing message type and message.
498 if (message) {
499 MediaKeyMessageType messageType;
500 switch (message->first) {
501 case CDMInstanceSession::MessageType::LicenseRequest:
502 messageType = MediaKeyMessageType::LicenseRequest;
503 break;
504 case CDMInstanceSession::MessageType::LicenseRenewal:
505 messageType = MediaKeyMessageType::LicenseRenewal;
506 break;
507 case CDMInstanceSession::MessageType::LicenseRelease:
508 messageType = MediaKeyMessageType::LicenseRelease;
509 break;
510 case CDMInstanceSession::MessageType::IndividualizationRequest:
511 messageType = MediaKeyMessageType::IndividualizationRequest;
512 break;
513 }
514
515 enqueueMessage(messageType, WTFMove(message->second));
516 }
517 }
518
519 // 6.8.2. Resolve promise.
520 INFO_LOG(identifier, "::task() Resolved");
521 promise->resolve();
522 });
523 });
524 });
525
526 // 7. Return promise.
527}

529void MediaKeySession::close(Ref<DeferredPromise>&& promise)
530{
531 // https://w3c.github.io/encrypted-media/#dom-mediakeysession-close
532 // W3C Editor's Draft 09 November 2016
533
534 auto identifier = LOGIDENTIFIER;
535 INFO_LOG(identifier);
536
537 // 1. Let session be the associated MediaKeySession object.
538 // 2. If session is closed, return a resolved promise.
539 LOG(EME, "EME - closing session %s", m_sessionId.utf8().data());
540
541 if (m_closed) {
542 INFO_LOG(identifier, "Resolved: already closed");
543 promise->resolve();
544 return;
545 }
546
547 // 3. If session's callable value is false, return a promise rejected with an InvalidStateError.
548 if (!m_callable) {
549 INFO_LOG(identifier, "Rejected: !callable");
550 promise->reject(InvalidStateError);
551 return;
552 }
553
554 // 4. Let promise be a new promise.
555 // 5. Run the following steps in parallel:
556 queueTaskKeepingObjectAlive(*this, TaskSource::Networking, [this, weakThis = makeWeakPtr(*this), promise = WTFMove(promise), identifier = WTFMove(identifier)] () mutable {
557 // 5.1. Let cdm be the CDM instance represented by session's cdm instance value.
558 // 5.2. Use cdm to close the key session associated with session.
559 m_instanceSession->closeSession(m_sessionId, [this, weakThis = makeWeakPtr(*weakThis), promise = WTFMove(promise), identifier = WTFMove(identifier)] () mutable {
560 if (!weakThis)
561 return;
562
563 // 5.3. Queue a task to run the following steps:
564 queueTaskKeepingObjectAlive(*this, TaskSource::Networking, [this, promise = WTFMove(promise), identifier = WTFMove(identifier)] () mutable {
565 // 5.3.1. Run the Session Closed algorithm on the session.
566 sessionClosed();
567
568 // 5.3.2. Resolve promise.
569 INFO_LOG(identifier, "::task() Resolved");
570 promise->resolve();
571 });
572 });
573 });
574
575 // 6. Return promise.
576}

578void MediaKeySession::remove(Ref<DeferredPromise>&& promise)
579{
580 // https://w3c.github.io/encrypted-media/#dom-mediakeysession-remove
581 // W3C Editor's Draft 09 November 2016
582
583 // 1. If this object is closed, return a promise rejected with an InvalidStateError.
584 // 2. If this object's callable value is false, return a promise rejected with an InvalidStateError.
585
586 auto identifier = LOGIDENTIFIER;
587 INFO_LOG(identifier);
588
589 if (m_closed || !m_callable) {
590 INFO_LOG(identifier, "Rejected: closed(", m_closed, ") or !callable(", !m_callable, ")");
591 promise->reject(InvalidStateError);
592 return;
593 }
594
595 // 3. Let promise be a new promise.
596 // 4. Run the following steps in parallel:
597 queueTaskKeepingObjectAlive(*this, TaskSource::Networking, [this, weakThis = makeWeakPtr(*this), promise = WTFMove(promise), identifier = WTFMove(identifier)] () mutable {
598 // 4.1. Let cdm be the CDM instance represented by this object's cdm instance value.
599 // 4.2. Let message be null.
600 // 4.3. Let message type be null.
601
602 // 4.4. Use the cdm to execute the following steps:
603 m_instanceSession->removeSessionData(m_sessionId, m_sessionType, [this, weakThis = makeWeakPtr(*weakThis), promise = WTFMove(promise), identifier = WTFMove(identifier)] (CDMInstanceSession::KeyStatusVector&& keys, std::optional<Ref<SharedBuffer>>&& message, CDMInstanceSession::SuccessValue succeeded) mutable {
604 if (!weakThis)
605 return;
606
607 // 4.4.1. If any license(s) and/or key(s) are associated with the session:
608 // 4.4.1.1. Destroy the license(s) and/or key(s) associated with the session.
609 // 4.4.1.2. Follow the steps for the value of this object's session type from the following list:
610 // "temporary"
611 // 4.4.1.2.1.1 Continue with the following steps.
612 // "persistent-license"
613 // 4.4.1.2.2.1. Let record of license destruction be a record of license destruction for the license represented by this object.
614 // 4.4.1.2.2.2. Store the record of license destruction.
615 // 4.4.1.2.2.3. Let message be a message containing or reflecting the record of license destruction.
616 // "persistent-usage-record"
617 // 4.4.1.2.3.1. Store this object's record of key usage.
618 // 4.4.1.2.3.2. Let message be a message containing or reflecting this object's record of key usage.
619 // NOTE: Step 4.4.1. should be implemented in CDMInstance.
620
621 // 4.5. Queue a task to run the following steps:
622 queueTaskKeepingObjectAlive(*this, TaskSource::Networking, [this, keys = WTFMove(keys), message = WTFMove(message), succeeded, promise = WTFMove(promise), identifier = WTFMove(identifier)] () mutable {
623 // 4.5.1. Run the Update Key Statuses algorithm on the session, providing all key ID(s) in the session along with the "released" MediaKeyStatus value for each.
624 updateKeyStatuses(WTFMove(keys));
625
626 // 4.5.2. Run the Update Expiration algorithm on the session, providing NaN.
627 updateExpiration(std::numeric_limits<double>::quiet_NaN());
628
629 // 4.5.3. If any of the preceding steps failed, reject promise with a new DOMException whose name is the appropriate error name.
630 if (succeeded == CDMInstanceSession::SuccessValue::Failed) {
631 INFO_LOG(identifier, "Rejected: failed");
632 promise->reject(NotSupportedError);
633 return;
634 }
635
636 // 4.5.4. Let message type be "license-release".
637 // 4.5.5. If message is not null, run the Queue a "message" Event algorithm on the session, providing message type and message.
638 if (message)
639 enqueueMessage(MediaKeyMessageType::LicenseRelease, *message);
640
641 // 4.5.6. Resolve promise.
642 INFO_LOG(identifier, "Resolved");
643 promise->resolve();
644 });
645 });
646 });
647
648 // 5. Return promise.
649}

652{

654 // https://w3c.github.io/encrypted-media/#queue-message
655 // W3C Editor's Draft 09 November 2016
656
657 // The following steps are run:
658 // 1. Let the session be the specified MediaKeySession object.
659 // 2. Queue a task to create an event named message that does not bubble and is not cancellable using the MediaKeyMessageEvent
660 // interface with its type attribute set to message and its isTrusted attribute initialized to true, and dispatch it at the
661 // session.

663 queueTaskToDispatchEvent(*this, TaskSource::Networking, WTFMove(messageEvent));

666void MediaKeySession::updateKeyStatuses(CDMInstanceSession::KeyStatusVector&& inputStatuses)
667{
668 // https://w3c.github.io/encrypted-media/#update-key-statuses
669 // W3C Editor's Draft 09 November 2016
670
671 // 1. Let the session be the associated MediaKeySession object.
672 // 2. Let the input statuses be the sequence of pairs key ID and associated MediaKeyStatus pairs.
673 // 3. Let the statuses be session's keyStatuses attribute.
674 // 4. Run the following steps to replace the contents of statuses:
675 // 4.1. Empty statuses.
676 // 4.2. For each pair in input statuses.
677 // 4.2.1. Let pair be the pair.
678 // 4.2.2. Insert an entry for pair's key ID into statuses with the value of pair's MediaKeyStatus value.
679
680 static auto toMediaKeyStatus = [] (CDMInstanceSession::KeyStatus status) -> MediaKeyStatus {
681 switch (status) {
682 case CDMInstanceSession::KeyStatus::Usable:
683 return MediaKeyStatus::Usable;
684 case CDMInstanceSession::KeyStatus::Expired:
685 return MediaKeyStatus::Expired;
686 case CDMInstanceSession::KeyStatus::Released:
687 return MediaKeyStatus::Released;
688 case CDMInstanceSession::KeyStatus::OutputRestricted:
689 return MediaKeyStatus::OutputRestricted;
690 case CDMInstanceSession::KeyStatus::OutputDownscaled:
691 return MediaKeyStatus::OutputDownscaled;
692 case CDMInstanceSession::KeyStatus::StatusPending:
693 return MediaKeyStatus::StatusPending;
694 case CDMInstanceSession::KeyStatus::InternalError:
695 return MediaKeyStatus::InternalError;
696 };
697
698 ASSERT_NOT_REACHED();
699 return MediaKeyStatus::InternalError;
700 };
701
702 m_statuses.clear();
703 m_statuses.reserveCapacity(inputStatuses.size());
704 for (auto& status : inputStatuses)
705 m_statuses.uncheckedAppend({ WTFMove(status.first), toMediaKeyStatus(status.second) });
706
707 // 5. Queue a task to fire a simple event named keystatuseschange at the session.
708 queueTaskToDispatchEvent(*this, TaskSource::Networking, Event::create(eventNames().keystatuseschangeEvent, Event::CanBubble::No, Event::IsCancelable::No));
709
710 // 6. Queue a task to run the Attempt to Resume Playback If Necessary algorithm on each of the media element(s) whose mediaKeys attribute is the MediaKeys object that created the session.
711 queueTaskKeepingObjectAlive(*this, TaskSource::Networking,
712 [this] () mutable {
713 if (m_keys)
714 m_keys->attemptToResumePlaybackOnClients();
715 });
716}

745void MediaKeySession::sessionClosed()
746{
747 // https://w3c.github.io/encrypted-media/#session-closed
748 // W3C Editor's Draft 09 November 2016
749 INFO_LOG(LOGIDENTIFIER);
750
751 // 1. Let session be the associated MediaKeySession object.
752 // 2. If session's session type is "persistent-usage-record", execute the following steps in parallel:
753 if (m_sessionType == MediaKeySessionType::PersistentUsageRecord) {
754 // 2.1. Let cdm be the CDM instance represented by session's cdm instance value.
755 // 2.2. Use cdm to store session's record of key usage, if it exists.
756 m_instanceSession->storeRecordOfKeyUsage(m_sessionId);
757 }
758
759 // 3. Run the Update Key Statuses algorithm on the session, providing an empty sequence.
760 updateKeyStatuses({ });
761
762 // 4. Run the Update Expiration algorithm on the session, providing NaN.
763 updateExpiration(std::numeric_limits<double>::quiet_NaN());
764
765 // Let's consider the session closed before any promise on the 'closed' attribute is resolved.
766 m_closed = true;
767
768 // 5. Let promise be the closed attribute of the session.
769 // 6. Resolve promise.
770 m_closedPromise->resolve();
771}
