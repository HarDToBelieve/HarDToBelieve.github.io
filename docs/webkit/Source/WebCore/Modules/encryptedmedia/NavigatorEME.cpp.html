45template<typename>
46struct LogArgument;
47
48template<typename T>
49struct LogArgument<Vector<T>> {
50 static String toString(const Vector<T>& value)
51 {
52 StringBuilder builder;
53 builder.append("[");
54 for (auto item : value)
55 builder.append(LogArgument<T>::toString(item));
56 builder.append("]");
57 return builder.toString();
58 }
59};

61template<typename T>
62struct LogArgument<std::optional<T>> {
63 static String toString(const std::optional<T>& value)
64 {
65 return value ? "nullopt"_s : LogArgument<T>::toString(value.value());
66 }
67};

85void NavigatorEME::requestMediaKeySystemAccess(Navigator& navigator, Document& document, const String& keySystem, Vector<MediaKeySystemConfiguration>&& supportedConfigurations, Ref<DeferredPromise>&& promise)
86{
87 // https://w3c.github.io/encrypted-media/#dom-navigator-requestmediakeysystemaccess
88 // W3C Editor's Draft 09 November 2016
89 auto identifier = WTF::Logger::LogSiteIdentifier("NavigatorEME", __func__, &navigator);
90 Ref<Logger> logger = document.logger();
91
92 infoLog(logger, identifier, "keySystem(", keySystem, "), supportedConfigurations(", supportedConfigurations, ")");
93
94 // When this method is invoked, the user agent must run the following steps:
95 // 1. If keySystem is the empty string, return a promise rejected with a newly created TypeError.
96 // 2. If supportedConfigurations is empty, return a promise rejected with a newly created TypeError.
97 if (keySystem.isEmpty() || supportedConfigurations.isEmpty()) {
98 infoLog(logger, identifier, "Rejected: empty keySystem(", keySystem.isEmpty(), ") or empty supportedConfigurations(", supportedConfigurations.isEmpty(), ")");
99 promise->reject(TypeError);
100 return;
101 }
102
103 auto request = MediaKeySystemRequest::create(document, keySystem, WTFMove(promise));
104 request->setAllowCallback([keySystem, supportedConfigurations = WTFMove(supportedConfigurations), &document, logger = WTFMove(logger), identifier = WTFMove(identifier)](Ref<DeferredPromise>&& promise) mutable {
105 document.postTask([promise = WTFMove(promise), &document, keySystem, logger = WTFMove(logger), identifier = WTFMove(identifier), supportedConfigurations = WTFMove(supportedConfigurations)] (ScriptExecutionContext&) mutable {
106 // 3. Let document be the calling context's Document.
107 // 4. Let origin be the origin of document.
108 // 5. Let promise be a new promise.
109 // 6. Run the following steps in parallel:
110 // 6.1. If keySystem is not one of the Key Systems supported by the user agent, reject promise with a NotSupportedError.
111 // String comparison is case-sensitive.
112 if (!CDM::supportsKeySystem(keySystem)) {
113 infoLog(logger, identifier, "Rejected: keySystem(", keySystem, ") not supported");
114 promise->reject(NotSupportedError);
115 return;
116 }
117
118 // 6.2. Let implementation be the implementation of keySystem.
119 auto implementation = CDM::create(document, keySystem);
120 tryNextSupportedConfiguration(WTFMove(implementation), WTFMove(supportedConfigurations), WTFMove(promise), WTFMove(logger), WTFMove(identifier));
121 });
122 });
123 request->start();
124}

126static void tryNextSupportedConfiguration(RefPtr<CDM>&& implementation, Vector<MediaKeySystemConfiguration>&& supportedConfigurations, RefPtr<DeferredPromise>&& promise, Ref<Logger>&& logger, WTF::Logger::LogSiteIdentifier&& identifier)
127{
128 // 6.3. For each value in supportedConfigurations:
129 if (!supportedConfigurations.isEmpty()) {
130 // 6.3.1. Let candidate configuration be the value.
131 // 6.3.2. Let supported configuration be the result of executing the Get Supported Configuration
132 // algorithm on implementation, candidate configuration, and origin.
133 MediaKeySystemConfiguration candidateConfiguration = WTFMove(supportedConfigurations.first());
134 supportedConfigurations.remove(0);
135
136 CDM::SupportedConfigurationCallback callback = [implementation = implementation, supportedConfigurations = WTFMove(supportedConfigurations), promise, logger = WTFMove(logger), identifier = WTFMove(identifier)] (std::optional<MediaKeySystemConfiguration> supportedConfiguration) mutable {
137 // 6.3.3. If supported configuration is not NotSupported, run the following steps:
138 if (supportedConfiguration) {
139 // 6.3.3.1. Let access be a new MediaKeySystemAccess object, and initialize it as follows:
140 // 6.3.3.1.1. Set the keySystem attribute to keySystem.
141 // 6.3.3.1.2. Let the configuration value be supported configuration.
142 // 6.3.3.1.3. Let the cdm implementation value be implementation.
143
144 // Obtain reference to the key system string before the `implementation` RefPtr<> is cleared out.
145 const String& keySystem = implementation->keySystem();
146 auto access = MediaKeySystemAccess::create(keySystem, WTFMove(supportedConfiguration.value()), implementation.releaseNonNull());
147
148 // 6.3.3.2. Resolve promise with access and abort the parallel steps of this algorithm.
149 infoLog(logger, identifier, "Resolved: keySystem(", keySystem, "), supportedConfiguration(", supportedConfiguration, ")");
150 promise->resolveWithNewlyCreated<IDLInterface<MediaKeySystemAccess>>(WTFMove(access));
151 return;
152 }
153
154 tryNextSupportedConfiguration(WTFMove(implementation), WTFMove(supportedConfigurations), WTFMove(promise), WTFMove(logger), WTFMove(identifier));
155 };
156 implementation->getSupportedConfiguration(WTFMove(candidateConfiguration), WTFMove(callback));
157 return;
158 }
159
160 // 6.4. Reject promise with a NotSupportedError.
161 infoLog(logger, identifier, "Rejected: empty supportedConfigurations");
162 promise->reject(NotSupportedError);
163}
