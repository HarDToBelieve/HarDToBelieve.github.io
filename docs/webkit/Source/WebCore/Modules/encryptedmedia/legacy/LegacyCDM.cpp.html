44namespace WebCore {

46struct LegacyCDMFactory {
47 CreateCDM constructor;
48 CDMSupportsKeySystem supportsKeySystem;
49 CDMSupportsKeySystemAndMimeType supportsKeySystemAndMimeType;
50};
51
52static void platformRegisterFactories(Vector<LegacyCDMFactory>& factories)
53{
54 factories.append({ [](LegacyCDM* cdm) { return makeUnique<LegacyCDMPrivateClearKey>(cdm); }, LegacyCDMPrivateClearKey::supportsKeySystem, LegacyCDMPrivateClearKey::supportsKeySystemAndMimeType });
55 // FIXME: initialize specific UA CDMs. http://webkit.org/b/109318, http://webkit.org/b/109320
56 factories.append({ [](LegacyCDM* cdm) { return makeUnique<CDMPrivateMediaPlayer>(cdm); }, CDMPrivateMediaPlayer::supportsKeySystem, CDMPrivateMediaPlayer::supportsKeySystemAndMimeType });
57#if (HAVE(AVCONTENTKEYSESSION) || HAVE(AVSTREAMSESSION)) && ENABLE(MEDIA_SOURCE)
58 factories.append({ [](LegacyCDM* cdm) { return makeUnique<CDMPrivateMediaSourceAVFObjC>(cdm); }, CDMPrivateMediaSourceAVFObjC::supportsKeySystem, CDMPrivateMediaSourceAVFObjC::supportsKeySystemAndMimeType });
59#endif
60}
61
62static Vector<LegacyCDMFactory>& installedCDMFactories()
63{
64 static auto cdms = makeNeverDestroyed<Vector<LegacyCDMFactory>>({ });
65 static std::once_flag registerDefaults;
66 std::call_once(registerDefaults, [&] {
67 platformRegisterFactories(cdms);
68 });
69 return cdms;
70}
71
72void LegacyCDM::resetFactories()
73{
74 clearFactories();
75 platformRegisterFactories(installedCDMFactories());
76}
77
78void LegacyCDM::clearFactories()
79{
80 installedCDMFactories().clear();
81}
82
83void LegacyCDM::registerCDMFactory(CreateCDM&& constructor, CDMSupportsKeySystem&& supportsKeySystem, CDMSupportsKeySystemAndMimeType&& supportsKeySystemAndMimeType)
84{
85 installedCDMFactories().append({ WTFMove(constructor), WTFMove(supportsKeySystem), WTFMove(supportsKeySystemAndMimeType) });
86}
87
88static LegacyCDMFactory* CDMFactoryForKeySystem(const String& keySystem)
89{
90 for (auto& factory : installedCDMFactories()) {
91 if (factory.supportsKeySystem(keySystem))
92 return &factory;
93 }
94 return nullptr;
95}
96
97bool LegacyCDM::supportsKeySystem(const String& keySystem)
98{
99 return CDMFactoryForKeySystem(keySystem);
100}
101
102bool LegacyCDM::keySystemSupportsMimeType(const String& keySystem, const String& mimeType)
103{
104 if (LegacyCDMFactory* factory = CDMFactoryForKeySystem(keySystem))
105 return factory->supportsKeySystemAndMimeType(keySystem, mimeType);
106 return false;
107}
108
109std::unique_ptr<LegacyCDM> LegacyCDM::create(const String& keySystem)
110{
111 if (!supportsKeySystem(keySystem))
112 return nullptr;
113
114 return makeUnique<LegacyCDM>(keySystem);
115}
116
117LegacyCDM::LegacyCDM(const String& keySystem)
118 : m_keySystem(keySystem)
119 , m_client(nullptr)
120{
121 m_private = CDMFactoryForKeySystem(keySystem)->constructor(this);
122}
123
124LegacyCDM::~LegacyCDM() = default;
125
126bool LegacyCDM::supportsMIMEType(const String& mimeType) const
127{
128 return m_private->supportsMIMEType(mimeType);
129}
130
131std::unique_ptr<LegacyCDMSession> LegacyCDM::createSession(LegacyCDMSessionClient& client)
132{
133 auto session = m_private->createSession(&client);
134 if (mediaPlayer())
135 mediaPlayer()->setCDMSession(session.get());
136 return session;
137}
138
139RefPtr<MediaPlayer> LegacyCDM::mediaPlayer() const
140{
141 if (!m_client)
142 return nullptr;
143 return m_client->cdmMediaPlayer(this);
144}
