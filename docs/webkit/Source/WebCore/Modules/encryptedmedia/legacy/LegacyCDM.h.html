36namespace WebCore {
37
38class LegacyCDM;
39class CDMPrivateInterface;
40class MediaPlayer;
41
42using CreateCDM = WTF::Function<std::unique_ptr<CDMPrivateInterface>(LegacyCDM*)>;
43using CDMSupportsKeySystem = WTF::Function<bool(const String&)>;
44using CDMSupportsKeySystemAndMimeType = WTF::Function<bool(const String&, const String&)>;
45
46class LegacyCDMClient {
47public:
48 virtual ~LegacyCDMClient() = default;
49
50 virtual RefPtr<MediaPlayer> cdmMediaPlayer(const LegacyCDM*) const = 0;
51};
52
53class WEBCORE_EXPORT LegacyCDM final {
54 WTF_MAKE_FAST_ALLOCATED;
55public:
56 explicit LegacyCDM(const String& keySystem);
57
58 enum CDMErrorCode { NoError, UnknownError, ClientError, ServiceError, OutputError, HardwareChangeError, DomainError };
59 static bool supportsKeySystem(const String&);
60 static bool keySystemSupportsMimeType(const String& keySystem, const String& mimeType);
61 static std::unique_ptr<LegacyCDM> create(const String& keySystem);
62 static void registerCDMFactory(CreateCDM&&, CDMSupportsKeySystem&&, CDMSupportsKeySystemAndMimeType&&);
63 ~LegacyCDM();
64
65 static void resetFactories();
66 static void clearFactories();
67
68 bool supportsMIMEType(const String&) const;
69 std::unique_ptr<LegacyCDMSession> createSession(LegacyCDMSessionClient&);
70
71 const String& keySystem() const { return m_keySystem; }
72
73 LegacyCDMClient* client() const { return m_client; }
74 void setClient(LegacyCDMClient* client) { m_client = client; }
75
76 RefPtr<MediaPlayer> mediaPlayer() const;
77 CDMPrivateInterface* cdmPrivate() const { return m_private.get(); }
78
79private:
80 String m_keySystem;
81 std::unique_ptr<CDMPrivateInterface> m_private;
82 LegacyCDMClient* m_client;
83};
