33namespace WebCore {
34
35class DOMFileSystem;
36class ErrorCallback;
37class FileSystemEntryCallback;
38
39class FileSystemEntry : public ScriptWrappable, public ActiveDOMObject, public RefCounted<FileSystemEntry> {
40 WTF_MAKE_ISO_ALLOCATED(FileSystemEntry);
41public:
42 virtual ~FileSystemEntry();
43
44 virtual bool isFile() const { return false; }
45 virtual bool isDirectory() const { return false; }
46
47 const String& name() const { return m_name; }
48 const String& virtualPath() const { return m_virtualPath; }
49 DOMFileSystem& filesystem() const;
50
51 void getParent(ScriptExecutionContext&, RefPtr<FileSystemEntryCallback>&&, RefPtr<ErrorCallback>&&);
52
53protected:
54 FileSystemEntry(ScriptExecutionContext&, DOMFileSystem&, const String& virtualPath);
55 Document* document() const;
56
57private:
58 const char* activeDOMObjectName() const final;
59
60 Ref<DOMFileSystem> m_filesystem;
61 String m_name;
62 String m_virtualPath;
63};
