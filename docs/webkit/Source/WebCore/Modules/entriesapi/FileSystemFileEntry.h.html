26#pragma once
27
28#include "FileSystemEntry.h"
29
30namespace WebCore {
31
32class ErrorCallback;
33class FileCallback;
34class ScriptExecutionContext;
35
36class FileSystemFileEntry final : public FileSystemEntry {
37public:
38 static Ref<FileSystemFileEntry> create(ScriptExecutionContext& context, DOMFileSystem& filesystem, const String& virtualPath)
39 {
40 return adoptRef(*new FileSystemFileEntry(context, filesystem, virtualPath));
41 }
42
43 void file(ScriptExecutionContext&, Ref<FileCallback>&&, RefPtr<ErrorCallback>&& = nullptr);
44
45private:
46 bool isFile() const final { return true; }
47
48 FileSystemFileEntry(ScriptExecutionContext&, DOMFileSystem&, const String& virtualPath);
49};
50static_assert(sizeof(FileSystemFileEntry) == sizeof(FileSystemEntry), "");
51
52} // namespace WebCore
53
54SPECIALIZE_TYPE_TRAITS_BEGIN(WebCore::FileSystemFileEntry)
55 static bool isType(const WebCore::FileSystemEntry& entry) { return entry.isFile(); }
56SPECIALIZE_TYPE_TRAITS_END()
