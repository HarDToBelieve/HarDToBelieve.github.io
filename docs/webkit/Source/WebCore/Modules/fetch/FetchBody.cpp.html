45ExceptionOr<FetchBody> FetchBody::extract(Init&& value, String& contentType)
46{
47 return WTF::switchOn(value, [&](RefPtr<Blob>& value) mutable -> ExceptionOr<FetchBody> {
48 Ref<const Blob> blob = value.releaseNonNull();
49 if (!blob->type().isEmpty())
50 contentType = blob->type();
51 return FetchBody(WTFMove(blob));
52 }, [&](RefPtr<DOMFormData>& value) mutable -> ExceptionOr<FetchBody> {
53 Ref<DOMFormData> domFormData = value.releaseNonNull();
54 auto formData = FormData::createMultiPart(domFormData.get());
55 contentType = makeString("multipart/form-data; boundary=", formData->boundary().data());
56 return FetchBody(WTFMove(formData));
57 }, [&](RefPtr<URLSearchParams>& value) mutable -> ExceptionOr<FetchBody> {
58 Ref<const URLSearchParams> params = value.releaseNonNull();
59 contentType = HTTPHeaderValues::formURLEncodedContentType();
60 return FetchBody(WTFMove(params));
61 }, [&](RefPtr<ArrayBuffer>& value) mutable -> ExceptionOr<FetchBody> {
62 Ref<const ArrayBuffer> buffer = value.releaseNonNull();
63 return FetchBody(WTFMove(buffer));
64 }, [&](RefPtr<ArrayBufferView>& value) mutable -> ExceptionOr<FetchBody> {
65 Ref<const ArrayBufferView> buffer = value.releaseNonNull();
66 return FetchBody(WTFMove(buffer));
67 }, [&](RefPtr<ReadableStream>& stream) mutable -> ExceptionOr<FetchBody> {
68 if (stream->isDisturbed())
69 return Exception { TypeError, "Input body is disturbed."_s };
70 if (stream->isLocked())
71 return Exception { TypeError, "Input body is locked."_s };
72
73 return FetchBody(stream.releaseNonNull());
74 }, [&](String& value) -> ExceptionOr<FetchBody> {
75 contentType = HTTPHeaderValues::textPlainContentType();
76 return FetchBody(WTFMove(value));
77 });
78}

80std::optional<FetchBody> FetchBody::fromFormData(ScriptExecutionContext& context, FormData& formData)
81{
82 ASSERT(!formData.isEmpty());
83
84 if (auto buffer = formData.asSharedBuffer()) {
85 FetchBody body;
86 body.m_consumer.setData(buffer.releaseNonNull());
87 return body;
88 }
89
90 auto url = formData.asBlobURL();
91 if (!url.isNull()) {
92 // FIXME: Properly set mime type and size of the blob.
93 Ref<const Blob> blob = Blob::deserialize(&context, url, { }, { }, { });
94 return FetchBody { WTFMove(blob) };
95 }
96
97 // FIXME: Support form data bodies.
98 return std::nullopt;
99}

148void FetchBody::consume(FetchBodyOwner& owner, Ref<DeferredPromise>&& promise)
149{
150 if (isArrayBuffer()) {
151 consumeArrayBuffer(owner, WTFMove(promise));
152 return;
153 }
154 if (isArrayBufferView()) {
155 consumeArrayBufferView(owner, WTFMove(promise));
156 return;
157 }
158 if (isText()) {
159 consumeText(owner, WTFMove(promise), textBody());
160 return;
161 }
162 if (isURLSearchParams()) {
163 consumeText(owner, WTFMove(promise), urlSearchParamsBody().toString());
164 return;
165 }
166 if (isBlob()) {
167 consumeBlob(owner, WTFMove(promise));
168 return;
169 }
170 if (isFormData()) {
171 consumeFormData(owner, WTFMove(promise));
172 return;
173 }
174
175 m_consumer.resolve(WTFMove(promise), owner.contentType(), m_readableStream.get());
176}

178void FetchBody::consumeAsStream(FetchBodyOwner& owner, FetchBodySource& source)
179{
180 bool closeStream = false;
181 if (isArrayBuffer()) {
182 closeStream = source.enqueue(ArrayBuffer::tryCreate(arrayBufferBody().data(), arrayBufferBody().byteLength()));
183 m_data = nullptr;
184 } else if (isArrayBufferView()) {
185 closeStream = source.enqueue(ArrayBuffer::tryCreate(arrayBufferViewBody().baseAddress(), arrayBufferViewBody().byteLength()));
186 m_data = nullptr;
187 } else if (isText()) {
188 auto data = UTF8Encoding().encode(textBody(), UnencodableHandling::Entities);
189 closeStream = source.enqueue(ArrayBuffer::tryCreate(data.data(), data.size()));
190 m_data = nullptr;
191 } else if (isURLSearchParams()) {
192 auto data = UTF8Encoding().encode(urlSearchParamsBody().toString(), UnencodableHandling::Entities);
193 closeStream = source.enqueue(ArrayBuffer::tryCreate(data.data(), data.size()));
194 m_data = nullptr;
195 } else if (isBlob()) {
196 owner.loadBlob(blobBody(), nullptr);
197 m_data = nullptr;
198 } else if (isFormData())
199 source.error(Exception { NotSupportedError, "Not implemented"_s });
200 else if (m_consumer.hasData())
201 closeStream = source.enqueue(m_consumer.takeAsArrayBuffer());
202 else
203 closeStream = true;
204
205 if (closeStream)
206 source.close();
207}

235void FetchBody::consumeFormData(FetchBodyOwner& owner, Ref<DeferredPromise>&& promise)
236{
237 if (auto sharedBuffer = formDataBody().asSharedBuffer()) {
238 m_consumer.resolveWithData(WTFMove(promise), owner.contentType(), sharedBuffer->data(), sharedBuffer->size());
239 m_data = nullptr;
240 } else {
241 // FIXME: If the form data contains blobs, load them like we do other blobs.
242 // That will fix the last WPT test in response-consume.html.
243 promise->reject(NotSupportedError);
244 }

257RefPtr<FormData> FetchBody::bodyAsFormData() const
258{
259 if (isText())
260 return FormData::create(UTF8Encoding().encode(textBody(), UnencodableHandling::Entities));
261 if (isURLSearchParams())
262 return FormData::create(UTF8Encoding().encode(urlSearchParamsBody().toString(), UnencodableHandling::Entities));
263 if (isBlob()) {
264 auto body = FormData::create();
265 body->appendBlob(blobBody().url());
266 return body;
267 }
268 if (isArrayBuffer())
269 return FormData::create(arrayBufferBody().data(), arrayBufferBody().byteLength());
270 if (isArrayBufferView())
271 return FormData::create(arrayBufferViewBody().baseAddress(), arrayBufferViewBody().byteLength());
272 if (isFormData()) {
273 auto body = makeRef(const_cast<FormData&>(formDataBody()));
274 return body;
275 }
276 if (auto* data = m_consumer.data())
277 return FormData::create(data->data(), data->size());
278
279 ASSERT_NOT_REACHED();
280 return nullptr;
281}

283FetchBody::TakenData FetchBody::take()
284{
285 if (m_consumer.hasData()) {
286 auto buffer = m_consumer.takeData();
287 if (!buffer)
288 return nullptr;
289 return buffer.releaseNonNull();
290 }
291
292 if (isBlob()) {
293 auto body = FormData::create();
294 body->appendBlob(blobBody().url());
295 return TakenData { WTFMove(body) };
296 }
297
298 if (isFormData())
299 return formDataBody();
300
301 if (isText())
302 return SharedBuffer::create(UTF8Encoding().encode(textBody(), UnencodableHandling::Entities));
303 if (isURLSearchParams())
304 return SharedBuffer::create(UTF8Encoding().encode(urlSearchParamsBody().toString(), UnencodableHandling::Entities));
305
306 if (isArrayBuffer())
307 return SharedBuffer::create(static_cast<const char*>(arrayBufferBody().data()), arrayBufferBody().byteLength());
308 if (isArrayBufferView())
309 return SharedBuffer::create(static_cast<const uint8_t*>(arrayBufferViewBody().baseAddress()), arrayBufferViewBody().byteLength());
310
311 return nullptr;
312}

314FetchBody FetchBody::clone()
315{
316 FetchBody clone(m_consumer);
317
318 if (isArrayBuffer())
319 clone.m_data = arrayBufferBody();
320 else if (isArrayBufferView())
321 clone.m_data = arrayBufferViewBody();
322 else if (isBlob())
323 clone.m_data = blobBody();
324 else if (isFormData())
325 clone.m_data = const_cast<FormData&>(formDataBody());
326 else if (isText())
327 clone.m_data = textBody();
328 else if (isURLSearchParams())
329 clone.m_data = urlSearchParamsBody();
330
331 if (m_readableStream) {
332 auto clones = m_readableStream->tee();
333 if (clones) {
334 m_readableStream = WTFMove(clones->first);
335 clone.m_readableStream = WTFMove(clones->second);
336 }
337 }
338 return clone;
339}
