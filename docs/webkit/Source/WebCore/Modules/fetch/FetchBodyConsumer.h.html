30
31#include "FetchBodySource.h"
32#include "JSDOMPromiseDeferred.h"
33#include "ReadableStreamSink.h"
34#include "SharedBuffer.h"

37namespace WebCore {
38
39class Blob;
40class FetchBodySource;
41class ReadableStream;
42
43class FetchBodyConsumer {
44public:
45 enum class Type { None, ArrayBuffer, Blob, JSON, Text, FormData };
46
47 explicit FetchBodyConsumer(Type type) : m_type(type) { }
48
49 void append(const uint8_t* data, unsigned);
50
51 bool hasData() const { return !!m_buffer; }
52 const SharedBuffer* data() const { return m_buffer.get(); }
53 void setData(Ref<SharedBuffer>&& data) { m_buffer = WTFMove(data); }
54
55 RefPtr<SharedBuffer> takeData();
56 RefPtr<JSC::ArrayBuffer> takeAsArrayBuffer();
57 String takeAsText();
58
59 void setContentType(const String& contentType) { m_contentType = contentType; }
60 void setType(Type type) { m_type = type; }
61
62 void clean();
63
64 void extract(ReadableStream&, ReadableStreamToSharedBufferSink::Callback&&);
65 void resolve(Ref<DeferredPromise>&&, const String& contentType, ReadableStream*);
66 void resolveWithData(Ref<DeferredPromise>&&, const String& contentType, const unsigned char*, unsigned);
67
68 void loadingFailed(const Exception&);
69 void loadingSucceeded(const String& contentType);
70
71 void setConsumePromise(Ref<DeferredPromise>&&);
72 void setSource(Ref<FetchBodySource>&&);
73
74 void setAsLoading() { m_isLoading = true; }
75
76private:
77 Ref<Blob> takeAsBlob(ScriptExecutionContext*);
78 void resetConsumePromise();
79
80 Type m_type;
81 String m_contentType;
82 RefPtr<SharedBuffer> m_buffer;
83 RefPtr<DeferredPromise> m_consumePromise;
84 RefPtr<ReadableStreamToSharedBufferSink> m_sink;
85 RefPtr<FetchBodySource> m_source;
86 bool m_isLoading { false };
87 RefPtr<UserGestureToken> m_userGestureToken;
88};
89
90} // namespace WebCore
