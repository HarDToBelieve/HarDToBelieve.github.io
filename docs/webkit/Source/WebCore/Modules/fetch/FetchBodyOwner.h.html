30#pragma once
31
32#include "ActiveDOMObject.h"
33#include "ExceptionOr.h"
34#include "FetchBody.h"
35#include "FetchBodySource.h"
36#include "FetchHeaders.h"
37#include "FetchLoader.h"
38#include "FetchLoaderClient.h"
39#include "ResourceError.h"
40
41namespace WebCore {
42
43class FetchBodyOwner : public RefCounted<FetchBodyOwner>, public ActiveDOMObject {
44public:
45 FetchBodyOwner(ScriptExecutionContext&, std::optional<FetchBody>&&, Ref<FetchHeaders>&&);
46 ~FetchBodyOwner();
47
48 bool bodyUsed() const { return isDisturbed(); }
49 void arrayBuffer(Ref<DeferredPromise>&&);
50 void blob(Ref<DeferredPromise>&&);
51 void formData(Ref<DeferredPromise>&&);
52 void json(Ref<DeferredPromise>&&);
53 void text(Ref<DeferredPromise>&&);
54
55 bool isDisturbed() const;
56 bool isDisturbedOrLocked() const;
57
58 void loadBlob(const Blob&, FetchBodyConsumer*);
59
60 bool isActive() const { return !!m_blobLoader; }
61
62 ExceptionOr<RefPtr<ReadableStream>> readableStream(JSC::JSGlobalObject&);
63 bool hasReadableStreamBody() const { return m_body && m_body->hasReadableStream(); }
64
65 virtual void consumeBodyAsStream();
66 virtual void feedStream() { }
67 virtual void cancel() { }
68
69 bool hasLoadingError() const;
70 ResourceError loadingError() const;
71 std::optional<Exception> loadingException() const;
72
73 const String& contentType() const { return m_contentType; }
74
75protected:
76 const FetchBody& body() const { return *m_body; }
77 FetchBody& body() { return *m_body; }
78 bool isBodyNull() const { return !m_body; }
79 bool isBodyNullOrOpaque() const { return !m_body || m_isBodyOpaque; }
80 void cloneBody(FetchBodyOwner&);
81
82 ExceptionOr<void> extractBody(FetchBody::Init&&);
83 void updateContentType();
84 void consumeOnceLoadingFinished(FetchBodyConsumer::Type, Ref<DeferredPromise>&&);
85
86 void setBody(FetchBody&& body) { m_body = WTFMove(body); }
87 ExceptionOr<void> createReadableStream(JSC::JSGlobalObject&);
88
89 // ActiveDOMObject API
90 void stop() override;
91
92 void setDisturbed() { m_isDisturbed = true; }
93
94 void setBodyAsOpaque() { m_isBodyOpaque = true; }
95 bool isBodyOpaque() const { return m_isBodyOpaque; }
96
97 void setLoadingError(Exception&&);
98 void setLoadingError(ResourceError&&);
99
100private:
101 // Blob loading routines
102 void blobChunk(const uint8_t*, size_t);
103 void blobLoadingSucceeded();
104 void blobLoadingFailed();
105 void finishBlobLoading();
106
107 // ActiveDOMObject API
108 bool virtualHasPendingActivity() const final;
109
110 struct BlobLoader final : FetchLoaderClient {
111 BlobLoader(FetchBodyOwner&);
112
113 // FetchLoaderClient API
114 void didReceiveResponse(const ResourceResponse&) final;
115 void didReceiveData(const uint8_t* data, size_t size) final { owner.blobChunk(data, size); }
116 void didFail(const ResourceError&) final;
117 void didSucceed() final { owner.blobLoadingSucceeded(); }
118
119 FetchBodyOwner& owner;
120 std::unique_ptr<FetchLoader> loader;
121 };
122
123protected:
124 std::optional<FetchBody> m_body;
125 String m_contentType;
126 bool m_isDisturbed { false };
127 RefPtr<FetchBodySource> m_readableStreamSource;
128 Ref<FetchHeaders> m_headers;
129
130private:
131 std::optional<BlobLoader> m_blobLoader;
132 bool m_isBodyOpaque { false };
133
134 Variant<std::nullptr_t, Exception, ResourceError> m_loadingError;
135};

137} // namespace WebCore
