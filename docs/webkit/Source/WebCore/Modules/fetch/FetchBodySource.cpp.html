32namespace WebCore {
33
34FetchBodySource::FetchBodySource(FetchBodyOwner& bodyOwner)
35 : m_bodyOwner(&bodyOwner)
36{
37}
38
39void FetchBodySource::setActive()
40{
41 ASSERT(m_bodyOwner);
42 ASSERT(!m_pendingActivity);
43 if (m_bodyOwner)
44 m_pendingActivity = m_bodyOwner->makePendingActivity(*m_bodyOwner);
45}
46
47void FetchBodySource::setInactive()
48{
49 ASSERT(m_bodyOwner);
50 ASSERT(m_pendingActivity);
51 m_pendingActivity = nullptr;
52}
53
54void FetchBodySource::doStart()
55{
56 ASSERT(m_bodyOwner);
57 if (m_bodyOwner)
58 m_bodyOwner->consumeBodyAsStream();
59}
60
61void FetchBodySource::doPull()
62{
63 ASSERT(m_bodyOwner);
64 if (m_bodyOwner)
65 m_bodyOwner->feedStream();
66}
67
68void FetchBodySource::doCancel()
69{
70 m_isCancelling = true;
71 ASSERT(m_bodyOwner || m_isClosed);
72 if (!m_bodyOwner)
73 return;
74
75 m_bodyOwner->cancel();
76 m_bodyOwner = nullptr;
77}
78
79void FetchBodySource::close()
80{
81#if ASSERT_ENABLED
82 m_isClosed = true;
83#endif
84
85 controller().close();
86 clean();
87 m_bodyOwner = nullptr;
88}
89
90void FetchBodySource::error(const Exception& value)
91{
92 controller().error(value);
93 clean();
94 m_bodyOwner = nullptr;
95}
