43static std::optional<Exception> setMethod(ResourceRequest& request, const String& initMethod)
44{
45 if (!isValidHTTPToken(initMethod))
46 return Exception { TypeError, "Method is not a valid HTTP token."_s };
47 if (isForbiddenMethod(initMethod))
48 return Exception { TypeError, "Method is forbidden."_s };
49 request.setHTTPMethod(normalizeHTTPMethod(initMethod));
50 return std::nullopt;
51}

53static ExceptionOr<String> computeReferrer(ScriptExecutionContext& context, const String& referrer)
54{
55 if (referrer.isEmpty())
56 return "no-referrer"_str;
57
58 // FIXME: Tighten the URL parsing algorithm according https://url.spec.whatwg.org/#concept-url-parser.
59 URL referrerURL = context.completeURL(referrer);
60 if (!referrerURL.isValid())
61 return Exception { TypeError, "Referrer is not a valid URL."_s };
62
63 if (referrerURL.protocolIsAbout() && referrerURL.path() == "client")
64 return "client"_str;
65
66 if (!(context.securityOrigin() && context.securityOrigin()->canRequest(referrerURL)))
67 return "client"_str;
68
69 return String { referrerURL.string() };
70}

72static std::optional<Exception> buildOptions(FetchOptions& options, ResourceRequest& request, String& referrer, ScriptExecutionContext& context, const FetchRequest::Init& init)
73{
74 if (!init.window.isUndefinedOrNull() && !init.window.isEmpty())
75 return Exception { TypeError, "Window can only be null."_s };
76
77 if (init.hasMembers()) {
78 if (options.mode == FetchOptions::Mode::Navigate)
79 options.mode = FetchOptions::Mode::SameOrigin;
80 referrer = "client"_s;
81 options.referrerPolicy = { };
82 }
83
84 if (!init.referrer.isNull()) {
85 auto result = computeReferrer(context, init.referrer);
86 if (result.hasException())
87 return result.releaseException();
88 referrer = result.releaseReturnValue();
89 }
90
91 if (init.referrerPolicy)
92 options.referrerPolicy = init.referrerPolicy.value();
93
94 if (init.mode) {
95 options.mode = init.mode.value();
96 if (options.mode == FetchOptions::Mode::Navigate)
97 return Exception { TypeError, "Request constructor does not accept navigate fetch mode."_s };
98 }
99
100 if (init.credentials)
101 options.credentials = init.credentials.value();
102
103 if (init.cache)
104 options.cache = init.cache.value();
105 if (options.cache == FetchOptions::Cache::OnlyIfCached && options.mode != FetchOptions::Mode::SameOrigin)
106 return Exception { TypeError, "only-if-cached cache option requires fetch mode to be same-origin."_s };
107
108 if (init.redirect)
109 options.redirect = init.redirect.value();
110
111 if (!init.integrity.isNull())
112 options.integrity = init.integrity;
113
114 if (init.keepalive && init.keepalive.value())
115 options.keepAlive = true;
116
117 if (!init.method.isNull()) {
118 if (auto exception = setMethod(request, init.method))
119 return exception;
120 }
121
122 return std::nullopt;
123}

125static bool methodCanHaveBody(const ResourceRequest& request)
126{
127 return request.httpMethod() != "GET" && request.httpMethod() != "HEAD";
128}

130ExceptionOr<void> FetchRequest::initializeOptions(const Init& init)
131{
132 ASSERT(scriptExecutionContext());
133
134 auto exception = buildOptions(m_options, m_request, m_referrer, *scriptExecutionContext(), init);
135 if (exception)
136 return WTFMove(exception.value());
137
138 if (m_options.mode == FetchOptions::Mode::NoCors) {
139 const String& method = m_request.httpMethod();
140 if (method != "GET" && method != "POST" && method != "HEAD")
141 return Exception { TypeError, "Method must be GET, POST or HEAD in no-cors mode."_s };
142 m_headers->setGuard(FetchHeaders::Guard::RequestNoCors);
143 }
144
145 return { };
146}

148static inline std::optional<Exception> processInvalidSignal(ScriptExecutionContext& context)
149{
150 ASCIILiteral message { "FetchRequestInit.signal should be undefined, null or an AbortSignal object. This will throw in a future release."_s };
151 context.addConsoleMessage(MessageSource::JS, MessageLevel::Warning, message);
152
153 if (is<Document>(context) && downcast<Document>(context).quirks().shouldIgnoreInvalidSignal())
154 return { };
155
156 RELEASE_LOG_ERROR(ResourceLoading, "FetchRequestInit.signal should be undefined, null or an AbortSignal object.");
157 return Exception { TypeError, message };
158}

160ExceptionOr<void> FetchRequest::initializeWith(const String& url, Init&& init)
161{
162 ASSERT(scriptExecutionContext());
163
164 // FIXME: Tighten the URL parsing algorithm according to https://url.spec.whatwg.org/#concept-url-parser.
165 URL requestURL = scriptExecutionContext()->completeURL(url, ScriptExecutionContext::ForceUTF8::Yes);
166 if (!requestURL.isValid() || requestURL.hasCredentials())
167 return Exception { TypeError, "URL is not valid or contains user credentials."_s };
168
169 m_options.mode = Mode::Cors;
170 m_options.credentials = Credentials::SameOrigin;
171 m_referrer = "client"_s;
172 m_request.setURL(requestURL);
173 m_request.setRequester(ResourceRequest::Requester::Fetch);
174 m_request.setInitiatorIdentifier(scriptExecutionContext()->resourceRequestIdentifier());
175
176 auto optionsResult = initializeOptions(init);
177 if (optionsResult.hasException())
178 return optionsResult.releaseException();
179
180 if (init.signal) {
181 if (auto* signal = JSAbortSignal::toWrapped(scriptExecutionContext()->vm(), init.signal))
182 m_signal->signalFollow(*signal);
183 else if (!init.signal.isUndefinedOrNull()) {
184 if (auto exception = processInvalidSignal(*scriptExecutionContext()))
185 return WTFMove(*exception);
186 }
187 }
188
189 if (init.headers) {
190 auto fillResult = m_headers->fill(*init.headers);
191 if (fillResult.hasException())
192 return fillResult.releaseException();
193 }
194
195 if (init.body) {
196 auto setBodyResult = setBody(WTFMove(*init.body));
197 if (setBodyResult.hasException())
198 return setBodyResult.releaseException();
199 }
200
201 updateContentType();
202 return { };
203}

205ExceptionOr<void> FetchRequest::initializeWith(FetchRequest& input, Init&& init)
206{
207 m_request = input.m_request;
208 m_options = input.m_options;
209 m_referrer = input.m_referrer;
210
211 auto optionsResult = initializeOptions(init);
212 if (optionsResult.hasException())
213 return optionsResult.releaseException();
214
215 if (init.signal && !init.signal.isUndefined()) {
216 if (auto* signal = JSAbortSignal::toWrapped(scriptExecutionContext()->vm(), init.signal))
217 m_signal->signalFollow(*signal);
218 else if (!init.signal.isNull()) {
219 if (auto exception = processInvalidSignal(*scriptExecutionContext()))
220 return WTFMove(*exception);
221 }
222
223 } else
224 m_signal->signalFollow(input.m_signal.get());
225
226 if (init.hasMembers()) {
227 auto fillResult = init.headers ? m_headers->fill(*init.headers) : m_headers->fill(input.headers());
228 if (fillResult.hasException())
229 return fillResult;
230 } else
231 m_headers->setInternalHeaders(HTTPHeaderMap { input.headers().internalHeaders() });
232
233 auto setBodyResult = init.body ? setBody(WTFMove(*init.body)) : setBody(input);
234 if (setBodyResult.hasException())
235 return setBodyResult;
236
237 updateContentType();
238 return { };
239}

241ExceptionOr<void> FetchRequest::setBody(FetchBody::Init&& body)
242{
243 if (!methodCanHaveBody(m_request))
244 return Exception { TypeError, makeString("Request has method '", m_request.httpMethod(), "' and cannot have a body") };
245
246 ASSERT(scriptExecutionContext());
247 auto result = extractBody(WTFMove(body));
248 if (result.hasException())
249 return result;
250
251 if (m_options.keepAlive && hasReadableStreamBody())
252 return Exception { TypeError, "Request cannot have a ReadableStream body and keepalive set to true"_s };
253 return { };
254}

256ExceptionOr<void> FetchRequest::setBody(FetchRequest& request)
257{
258 if (request.isDisturbedOrLocked())
259 return Exception { TypeError, "Request input is disturbed or locked."_s };
260
261 if (!request.isBodyNull()) {
262 if (!methodCanHaveBody(m_request))
263 return Exception { TypeError, makeString("Request has method '", m_request.httpMethod(), "' and cannot have a body") };
264 // FIXME: If body has a readable stream, we should pipe it to this new body stream.
265 m_body = WTFMove(*request.m_body);
266 request.setDisturbed();
267 }
268
269 if (m_options.keepAlive && hasReadableStreamBody())
270 return Exception { TypeError, "Request cannot have a ReadableStream body and keepalive set to true"_s };
271 return { };
272}

274ExceptionOr<Ref<FetchRequest>> FetchRequest::create(ScriptExecutionContext& context, Info&& input, Init&& init)
275{
276 auto request = adoptRef(*new FetchRequest(context, std::nullopt, FetchHeaders::create(FetchHeaders::Guard::Request), { }, { }, { }));
277
278 if (WTF::holds_alternative<String>(input)) {
279 auto result = request->initializeWith(WTF::get<String>(input), WTFMove(init));
280 if (result.hasException())
281 return result.releaseException();
282 } else {
283 auto result = request->initializeWith(*WTF::get<RefPtr<FetchRequest>>(input), WTFMove(init));
284 if (result.hasException())
285 return result.releaseException();
286 }
287
288 return request;
289}

291String FetchRequest::referrer() const
292{
293 if (m_referrer == "no-referrer")
294 return String();
295 if (m_referrer == "client")
296 return "about:client"_s;
297 return m_referrer;
298}

300const String& FetchRequest::urlString() const
301{
302 if (m_requestURL.isNull())
303 m_requestURL = m_request.url().string();
304 return m_requestURL;

307ResourceRequest FetchRequest::resourceRequest() const
308{
309 ASSERT(scriptExecutionContext());
310
311 ResourceRequest request = m_request;
312 request.setHTTPHeaderFields(m_headers->internalHeaders());
313
314 if (!isBodyNull())
315 request.setHTTPBody(body().bodyAsFormData());
316
317 return request;
318}

320ExceptionOr<Ref<FetchRequest>> FetchRequest::clone(ScriptExecutionContext& context)
321{
322 if (isDisturbedOrLocked())
323 return Exception { TypeError, "Body is disturbed or locked"_s };
324
325 auto clone = adoptRef(*new FetchRequest(context, std::nullopt, FetchHeaders::create(m_headers.get()), ResourceRequest { m_request }, FetchOptions { m_options}, String { m_referrer }));
326 clone->cloneBody(*this);
327 clone->m_signal->signalFollow(m_signal);
328 return clone;
329}
