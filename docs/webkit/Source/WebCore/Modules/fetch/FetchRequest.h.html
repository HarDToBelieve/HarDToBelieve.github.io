29#pragma once
30
31#include "AbortSignal.h"
32#include "ExceptionOr.h"
33#include "FetchBodyOwner.h"
34#include "FetchOptions.h"
35#include "FetchRequestInit.h"
36#include "ResourceRequest.h"
37
38namespace WebCore {
39
40class Blob;
41class ScriptExecutionContext;
42class URLSearchParams;
43
44class FetchRequest final : public FetchBodyOwner {
45public:
46 using Init = FetchRequestInit;
47 using Info = Variant<RefPtr<FetchRequest>, String>;
48
49 using Cache = FetchOptions::Cache;
50 using Credentials = FetchOptions::Credentials;
51 using Destination = FetchOptions::Destination;
52 using Mode = FetchOptions::Mode;
53 using Redirect = FetchOptions::Redirect;
54
55 static ExceptionOr<Ref<FetchRequest>> create(ScriptExecutionContext&, Info&&, Init&&);
56 static Ref<FetchRequest> create(ScriptExecutionContext& context, std::optional<FetchBody>&& body, Ref<FetchHeaders>&& headers, ResourceRequest&& request, FetchOptions&& options, String&& referrer) { return adoptRef(*new FetchRequest(context, WTFMove(body), WTFMove(headers), WTFMove(request), WTFMove(options), WTFMove(referrer))); }
57
58 const String& method() const { return m_request.httpMethod(); }
59 const String& urlString() const;
60 FetchHeaders& headers() { return m_headers.get(); }
61 const FetchHeaders& headers() const { return m_headers.get(); }
62
63 Destination destination() const { return m_options.destination; }
64 String referrer() const;
65 ReferrerPolicy referrerPolicy() const { return m_options.referrerPolicy; }
66 Mode mode() const { return m_options.mode; }
67 Credentials credentials() const { return m_options.credentials; }
68 Cache cache() const { return m_options.cache; }
69 Redirect redirect() const { return m_options.redirect; }
70 bool keepalive() const { return m_options.keepAlive; };
71 AbortSignal& signal() { return m_signal.get(); }
72
73 const String& integrity() const { return m_options.integrity; }
74
75 ExceptionOr<Ref<FetchRequest>> clone(ScriptExecutionContext&);
76
77 const FetchOptions& fetchOptions() const { return m_options; }
78 const ResourceRequest& internalRequest() const { return m_request; }
79 const String& internalRequestReferrer() const { return m_referrer; }
80 const URL& url() const { return m_request.url(); }
81
82 ResourceRequest resourceRequest() const;
83
84private:
85 FetchRequest(ScriptExecutionContext&, std::optional<FetchBody>&&, Ref<FetchHeaders>&&, ResourceRequest&&, FetchOptions&&, String&& referrer);
86
87 ExceptionOr<void> initializeOptions(const Init&);
88 ExceptionOr<void> initializeWith(FetchRequest&, Init&&);
89 ExceptionOr<void> initializeWith(const String&, Init&&);
90 ExceptionOr<void> setBody(FetchBody::Init&&);
91 ExceptionOr<void> setBody(FetchRequest&);
92
93 const char* activeDOMObjectName() const final;
94
95 ResourceRequest m_request;
96 FetchOptions m_options;
97 String m_referrer;
98 mutable String m_requestURL;
99 Ref<AbortSignal> m_signal;
100};
101
102inline FetchRequest::FetchRequest(ScriptExecutionContext& context, std::optional<FetchBody>&& body, Ref<FetchHeaders>&& headers, ResourceRequest&& request, FetchOptions&& options, String&& referrer)
103 : FetchBodyOwner(context, WTFMove(body), WTFMove(headers))
104 , m_request(WTFMove(request))
105 , m_options(WTFMove(options))
106 , m_referrer(WTFMove(referrer))
107 , m_signal(AbortSignal::create(context))
108{
109 updateContentType();
110}
111
112} // namespace WebCore
