28#include "AbortSignal.h"

30#include "FetchHeaders.h"
31#include "FetchOptions.h"
32#include <JavaScriptCore/JSCJSValue.h>

35namespace WebCore {
36
37struct FetchRequestInit {
38 String method;
39 std::optional<FetchHeaders::Init> headers;
40 std::optional<FetchBody::Init> body;
41 String referrer;
42 std::optional<ReferrerPolicy> referrerPolicy;
43 std::optional<FetchOptions::Mode> mode;
44 std::optional<FetchOptions::Credentials> credentials;
45 std::optional<FetchOptions::Cache> cache;
46 std::optional<FetchOptions::Redirect> redirect;
47 String integrity;
48 std::optional<bool> keepalive;
49 JSC::JSValue signal;
50 JSC::JSValue window;
51
52 bool hasMembers() const { return !method.isEmpty() || headers || body || !referrer.isEmpty() || referrerPolicy || mode || credentials || cache || redirect || !integrity.isEmpty() || keepalive || !window.isUndefined() || !signal.isUndefined(); }
53};
