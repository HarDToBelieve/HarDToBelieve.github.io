45// https://fetch.spec.whatwg.org/#null-body-status
46static inline bool isNullBodyStatus(int status)
47{
48 return status == 101 || status == 204 || status == 205 || status == 304;
49}

51Ref<FetchResponse> FetchResponse::create(ScriptExecutionContext& context, std::optional<FetchBody>&& body, FetchHeaders::Guard guard, ResourceResponse&& response)
52{
53 bool isSynthetic = response.type() == ResourceResponse::Type::Default || response.type() == ResourceResponse::Type::Error;
54 bool isOpaque = response.tainting() == ResourceResponse::Tainting::Opaque;
55 auto headers = isOpaque ? FetchHeaders::create(guard) : FetchHeaders::create(guard, HTTPHeaderMap { response.httpHeaderFields() });
56
57 auto fetchResponse = adoptRef(*new FetchResponse(context, WTFMove(body), WTFMove(headers), WTFMove(response)));
58 fetchResponse->updateContentType();
59 if (!isSynthetic)
60 fetchResponse->m_filteredResponse = ResourceResponseBase::filter(fetchResponse->m_internalResponse, ResourceResponse::PerformExposeAllHeadersCheck::Yes);
61 if (isOpaque)
62 fetchResponse->setBodyAsOpaque();
63 return fetchResponse;
64}

66ExceptionOr<Ref<FetchResponse>> FetchResponse::create(ScriptExecutionContext& context, std::optional<FetchBody::Init>&& body, Init&& init)
67{
68 // 1. If init’s status member is not in the range 200 to 599, inclusive, then throw a RangeError.
69 if (init.status < 200 || init.status > 599)
70 return Exception { RangeError, "Status must be between 200 and 599"_s };
71
72 // 2. If init’s statusText member does not match the reason-phrase token production, then throw a TypeError.
73 if (!isValidReasonPhrase(init.statusText))
74 return Exception { TypeError, "Status text must be a valid reason-phrase."_s };
75
76 // 3. Let r be a new Response object associated with a new response.
77 // NOTE: Creation of the Response object is delayed until all potential exceptional cases are handled.
78
79 // 4. Set r’s headers to a new Headers object, whose header list is r’s response’s header list, and guard is "response".
80 auto headers = FetchHeaders::create(FetchHeaders::Guard::Response);
81
82 // 5. Set r’s response’s status to init’s status member.
83 auto status = init.status;
84
85 // 6. Set r’s response’s status message to init’s statusText member.
86 auto statusText = init.statusText;
87
88 // 7. If init’s headers member is present, then fill r’s headers with init’s headers member.
89 if (init.headers) {
90 auto result = headers->fill(*init.headers);
91 if (result.hasException())
92 return result.releaseException();
93 }
94
95 std::optional<FetchBody> extractedBody;
96
97 // 8. If body is non-null, run these substeps:
98 if (body) {
99 // 8.1 If init’s status member is a null body status, then throw a TypeError.
100 // (NOTE: 101 is included in null body status due to its use elsewhere. It does not affect this step.)
101 if (isNullBodyStatus(init.status))
102 return Exception { TypeError, "Response cannot have a body with the given status."_s };
103
104 // 8.2 Let Content-Type be null.
105 String contentType;
106
107 // 8.3 Set r’s response’s body and Content-Type to the result of extracting body.
108 auto result = FetchBody::extract(WTFMove(*body), contentType);
109 if (result.hasException())
110 return result.releaseException();
111 extractedBody = result.releaseReturnValue();
112
113 // 8.4 If Content-Type is non-null and r’s response’s header list does not contain `Content-Type`, then append
114 // `Content-Type`/Content-Type to r’s response’s header list.
115 if (!contentType.isNull() && !headers->fastHas(HTTPHeaderName::ContentType))
116 headers->fastSet(HTTPHeaderName::ContentType, contentType);
117 }
118
119 // 9. Set r’s MIME type to the result of extracting a MIME type from r’s response’s header list.
120 auto contentType = headers->fastGet(HTTPHeaderName::ContentType);
121
122 // 10. Set r’s response’s HTTPS state to current settings object’s HTTPS state.
123 // FIXME: Implement.
124
125 // 11. Resolve r’s trailer promise with a new Headers object whose guard is "immutable".
126 // FIXME: Implement.
127
128 // 12. Return r.
129 auto r = adoptRef(*new FetchResponse(context, WTFMove(extractedBody), WTFMove(headers), { }));
130
131 r->m_contentType = contentType;
132 auto mimeType = extractMIMETypeFromMediaType(contentType);
133 r->m_internalResponse.setMimeType(mimeType.isEmpty() ? defaultMIMEType() : mimeType);
134 r->m_internalResponse.setTextEncodingName(extractCharsetFromMediaType(contentType));
135
136 r->m_internalResponse.setHTTPStatusCode(status);
137 r->m_internalResponse.setHTTPStatusText(statusText);
138
139 return r;
140}

149ExceptionOr<Ref<FetchResponse>> FetchResponse::redirect(ScriptExecutionContext& context, const String& url, int status)
150{
151 // FIXME: Tighten the URL parsing algorithm according https://url.spec.whatwg.org/#concept-url-parser.
152 URL requestURL = context.completeURL(url);
153 if (!requestURL.isValid())
154 return Exception { TypeError, makeString("Redirection URL '", requestURL.string(), "' is invalid") };
155 if (requestURL.hasCredentials())
156 return Exception { TypeError, "Redirection URL contains credentials"_s };
157 if (!ResourceResponse::isRedirectionStatusCode(status))
158 return Exception { RangeError, makeString("Status code ", status, "is not a redirection status code") };
159 auto redirectResponse = adoptRef(*new FetchResponse(context, { }, FetchHeaders::create(FetchHeaders::Guard::Immutable), { }));
160 redirectResponse->m_internalResponse.setHTTPStatusCode(status);
161 redirectResponse->m_internalResponse.setHTTPHeaderField(HTTPHeaderName::Location, requestURL.string());
162 redirectResponse->m_headers->fastSet(HTTPHeaderName::Location, requestURL.string());
163 return redirectResponse;
164}

172ExceptionOr<Ref<FetchResponse>> FetchResponse::clone(ScriptExecutionContext& context)
173{
174 if (isDisturbedOrLocked())
175 return Exception { TypeError, "Body is disturbed or locked"_s };
176
177 ASSERT(scriptExecutionContext());
178
179 // If loading, let's create a stream so that data is teed on both clones.
180 if (isLoading() && !m_readableStreamSource) {
181 auto voidOrException = createReadableStream(*context.globalObject());
182 if (UNLIKELY(voidOrException.hasException()))
183 return voidOrException.releaseException();
184 }
185
186 // Synthetic responses do not store headers in m_internalResponse.
187 if (m_internalResponse.type() == ResourceResponse::Type::Default)
188 m_internalResponse.setHTTPHeaderFields(HTTPHeaderMap { headers().internalHeaders() });
189
190 auto clone = FetchResponse::create(context, std::nullopt, headers().guard(), ResourceResponse { m_internalResponse });
191 clone->cloneBody(*this);
192 clone->m_opaqueLoadIdentifier = m_opaqueLoadIdentifier;
193 clone->m_bodySizeWithPadding = m_bodySizeWithPadding;
194 return clone;
195}

197void FetchResponse::addAbortSteps(Ref<AbortSignal>&& signal)
198{
199 m_abortSignal = WTFMove(signal);
200 m_abortSignal->addAlgorithm([this, weakThis = makeWeakPtr(this)] {
201 // FIXME: Cancel request body if it is a stream.
202 if (!weakThis)
203 return;
204
205 m_abortSignal = nullptr;
206
207 setLoadingError(Exception { AbortError, "Fetch is aborted"_s });
208
209 if (m_bodyLoader) {
210 if (auto callback = m_bodyLoader->takeNotificationCallback())
211 callback(Exception { AbortError, "Fetch is aborted"_s });
212
213 if (auto callback = m_bodyLoader->takeConsumeDataCallback())
214 callback(Exception { AbortError, "Fetch is aborted"_s });
215 }
216
217 if (m_readableStreamSource) {
218 if (!m_readableStreamSource->isCancelling())
219 m_readableStreamSource->error(*loadingException());
220 m_readableStreamSource = nullptr;
221 }
222 if (m_body)
223 m_body->loadingFailed(*loadingException());
224
225 if (auto bodyLoader = WTFMove(m_bodyLoader))
226 bodyLoader->stop();
227 });
228}

230void FetchResponse::fetch(ScriptExecutionContext& context, FetchRequest& request, NotificationCallback&& responseCallback)
231{
232 if (request.signal().aborted()) {
233 responseCallback(Exception { AbortError, "Request signal is aborted"_s });
234 // FIXME: Cancel request body if it is a stream.
235 return;
236 }
237
238 if (request.hasReadableStreamBody()) {
239 responseCallback(Exception { NotSupportedError, "ReadableStream uploading is not supported"_s });
240 return;
241 }
242
243 InspectorInstrumentation::willFetch(context, request.url().string());
244
245 auto response = adoptRef(*new FetchResponse(context, FetchBody { }, FetchHeaders::create(FetchHeaders::Guard::Immutable), { }));
246
247 response->body().consumer().setAsLoading();
248
249 response->addAbortSteps(request.signal());
250
251 response->m_bodyLoader = makeUnique<BodyLoader>(response.get(), WTFMove(responseCallback));
252 if (!response->m_bodyLoader->start(context, request))
253 response->m_bodyLoader = nullptr;
254}

273void FetchResponse::BodyLoader::didSucceed()
274{
275 ASSERT(m_response.hasPendingActivity());
276 m_response.m_body->loadingSucceeded(m_response.contentType());
277
278 if (m_response.m_readableStreamSource) {
279 if (m_response.body().consumer().hasData())
280 m_response.m_readableStreamSource->enqueue(m_response.body().consumer().takeAsArrayBuffer());
281
282 m_response.closeStream();
283 }
284
285 if (auto consumeDataCallback = WTFMove(m_consumeDataCallback))
286 consumeDataCallback(nullptr);
287
288 if (m_loader->isStarted()) {
289 Ref<FetchResponse> protector(m_response);
290 m_response.m_bodyLoader = nullptr;
291 }
292}

294void FetchResponse::BodyLoader::didFail(const ResourceError& error)
295{
296 ASSERT(m_response.hasPendingActivity());
297
298 m_response.setLoadingError(ResourceError { error });
299
300 if (auto responseCallback = WTFMove(m_responseCallback))
301 responseCallback(Exception { TypeError, error.localizedDescription() });
302
303 if (auto consumeDataCallback = WTFMove(m_consumeDataCallback))
304 consumeDataCallback(Exception { TypeError, error.localizedDescription() });
305
306 if (m_response.m_readableStreamSource) {
307 if (!m_response.m_readableStreamSource->isCancelling())
308 m_response.m_readableStreamSource->error(*m_response.loadingException());
309 m_response.m_readableStreamSource = nullptr;
310 }
311
312 if (m_response.m_body)
313 m_response.m_body->loadingFailed(*m_response.loadingException());
314
315 // Check whether didFail is called as part of FetchLoader::start.
316 if (m_loader && m_loader->isStarted()) {
317 Ref<FetchResponse> protector(m_response);
318 m_response.m_bodyLoader = nullptr;
319 }
320}

333static uint64_t nextOpaqueLoadIdentifier { 0 };
334void FetchResponse::BodyLoader::didReceiveResponse(const ResourceResponse& resourceResponse)
335{
336 auto performCheck = m_credentials == FetchOptions::Credentials::Include ? ResourceResponse::PerformExposeAllHeadersCheck::No : ResourceResponse::PerformExposeAllHeadersCheck::Yes;
337 m_response.m_filteredResponse = ResourceResponseBase::filter(resourceResponse, performCheck);
338 m_response.m_internalResponse = resourceResponse;
339 m_response.m_internalResponse.setType(m_response.m_filteredResponse->type());
340 if (resourceResponse.tainting() == ResourceResponse::Tainting::Opaque) {
341 m_response.m_opaqueLoadIdentifier = ++nextOpaqueLoadIdentifier;
342 m_response.setBodyAsOpaque();
343 }
344
345 m_response.m_headers->filterAndFill(m_response.m_filteredResponse->httpHeaderFields(), FetchHeaders::Guard::Response);
346 m_response.updateContentType();
347
348 if (auto responseCallback = WTFMove(m_responseCallback))
349 responseCallback(m_response);
350}

352void FetchResponse::BodyLoader::didReceiveData(const uint8_t* data, size_t size)
353{
354 ASSERT(m_response.m_readableStreamSource || m_consumeDataCallback);
355
356 if (m_consumeDataCallback) {
357 Span chunk { data, size };
358 m_consumeDataCallback(&chunk);
359 return;
360 }
361
362 auto& source = *m_response.m_readableStreamSource;
363
364 if (!source.isPulling()) {
365 m_response.body().consumer().append(data, size);
366 return;
367 }
368
369 if (m_response.body().consumer().hasData() && !source.enqueue(m_response.body().consumer().takeAsArrayBuffer())) {
370 stop();
371 return;
372 }
373 if (!source.enqueue(ArrayBuffer::tryCreate(data, size))) {
374 stop();
375 return;
376 }
377 source.resolvePullPromise();
378}

407FetchResponse::ResponseData FetchResponse::consumeBody()
408{
409 ASSERT(!isBodyReceivedByChunk());
410
411 if (isBodyNull())
412 return nullptr;
413
414 ASSERT(!m_isDisturbed);
415 m_isDisturbed = true;
416
417 return body().take();
418}

420void FetchResponse::consumeBodyReceivedByChunk(ConsumeDataByChunkCallback&& callback)
421{
422 ASSERT(isBodyReceivedByChunk());
423 ASSERT(!isDisturbed());
424 m_isDisturbed = true;
425
426 if (hasReadableStreamBody()) {
427 m_body->consumer().extract(*m_body->readableStream(), WTFMove(callback));
428 return;
429 }
430
431 ASSERT(isLoading());
432 m_bodyLoader->consumeDataByChunk(WTFMove(callback));
433}

459void FetchResponse::consumeBodyAsStream()
460{
461 ASSERT(m_readableStreamSource);
462 if (!isLoading()) {
463 FetchBodyOwner::consumeBodyAsStream();
464 return;
465 }
466
467 ASSERT(m_bodyLoader);
468
469 auto data = m_bodyLoader->startStreaming();
470 if (data) {
471 if (!m_readableStreamSource->enqueue(data->tryCreateArrayBuffer())) {
472 stop();
473 return;
474 }
475 m_readableStreamSource->resolvePullPromise();
476 }
477}

486void FetchResponse::feedStream()
487{
488 ASSERT(m_readableStreamSource);
489 bool shouldCloseStream = !m_bodyLoader;
490
491 if (body().consumer().hasData()) {
492 if (!m_readableStreamSource->enqueue(body().consumer().takeAsArrayBuffer())) {
493 stop();
494 return;
495 }
496 if (!shouldCloseStream) {
497 m_readableStreamSource->resolvePullPromise();
498 return;
499 }
500 } else if (!shouldCloseStream)
501 return;
502
503 closeStream();
504}

544bool FetchResponse::isCORSSameOrigin() const
545{
546 // https://html.spec.whatwg.org/#cors-same-origin
547 // A response whose type is "basic", "cors", or "default" is CORS-same-origin.
548 switch (type()) {
549 case ResourceResponse::Type::Basic:
550 case ResourceResponse::Type::Cors:
551 case ResourceResponse::Type::Default:
552 return true;
553 default:
554 return false;
555 }
556}
