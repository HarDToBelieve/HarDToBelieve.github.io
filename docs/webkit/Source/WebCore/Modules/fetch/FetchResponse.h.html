50class FetchResponse final : public FetchBodyOwner, public CanMakeWeakPtr<FetchResponse> {
51public:
52 using Type = ResourceResponse::Type;
53
54 struct Init {
55 unsigned short status { 200 };
56 String statusText;
57 std::optional<FetchHeaders::Init> headers;
58 };
59
60 WEBCORE_EXPORT static Ref<FetchResponse> create(ScriptExecutionContext&, std::optional<FetchBody>&&, FetchHeaders::Guard, ResourceResponse&&);
61
62 static ExceptionOr<Ref<FetchResponse>> create(ScriptExecutionContext&, std::optional<FetchBody::Init>&&, Init&&);
63 static Ref<FetchResponse> error(ScriptExecutionContext&);
64 static ExceptionOr<Ref<FetchResponse>> redirect(ScriptExecutionContext&, const String& url, int status);
65
66 using NotificationCallback = WTF::Function<void(ExceptionOr<FetchResponse&>&&)>;
67 static void fetch(ScriptExecutionContext&, FetchRequest&, NotificationCallback&&);
68
69 void startConsumingStream(unsigned);
70 void consumeChunk(Ref<JSC::Uint8Array>&&);
71 void finishConsumingStream(Ref<DeferredPromise>&&);
72
73 Type type() const { return filteredResponse().type(); }
74 const String& url() const;
75 bool redirected() const { return filteredResponse().isRedirected(); }
76 int status() const { return filteredResponse().httpStatusCode(); }
77 bool ok() const { return filteredResponse().isSuccessful(); }
78 const String& statusText() const { return filteredResponse().httpStatusText(); }
79
80 const FetchHeaders& headers() const { return m_headers; }
81 FetchHeaders& headers() { return m_headers; }
82 ExceptionOr<Ref<FetchResponse>> clone(ScriptExecutionContext&);
83
84 void consumeBodyAsStream() final;
85 void feedStream() final;
86 void cancel() final;
87
88 using ResponseData = Variant<std::nullptr_t, Ref<FormData>, Ref<SharedBuffer>>;
89 ResponseData consumeBody();
90 void setBodyData(ResponseData&&, uint64_t bodySizeWithPadding);
91
92 bool isLoading() const { return !!m_bodyLoader; }
93 bool isBodyReceivedByChunk() const { return isLoading() || hasReadableStreamBody(); }
94 bool isBlobBody() const { return !isBodyNull() && body().isBlob(); }
95 bool isBlobFormData() const { return !isBodyNull() && body().isFormData(); }
96
97 using ConsumeDataByChunkCallback = Function<void(ExceptionOr<Span<const uint8_t>*>&&)>;
98 void consumeBodyReceivedByChunk(ConsumeDataByChunkCallback&&);
99
100 WEBCORE_EXPORT ResourceResponse resourceResponse() const;
101 ResourceResponse::Tainting tainting() const { return m_internalResponse.tainting(); }
102
103 uint64_t bodySizeWithPadding() const { return m_bodySizeWithPadding; }
104 void setBodySizeWithPadding(uint64_t size) { m_bodySizeWithPadding = size; }
105 uint64_t opaqueLoadIdentifier() const { return m_opaqueLoadIdentifier; }
106
107 void initializeOpaqueLoadIdentifierForTesting() { m_opaqueLoadIdentifier = 1; }
108
109 const HTTPHeaderMap& internalResponseHeaders() const { return m_internalResponse.httpHeaderFields(); }
110
111 bool isCORSSameOrigin() const;
112 bool hasWasmMIMEType() const;
113
114private:
115 FetchResponse(ScriptExecutionContext&, std::optional<FetchBody>&&, Ref<FetchHeaders>&&, ResourceResponse&&);
116
117 void stop() final;
118 const char* activeDOMObjectName() const final;
119
120 const ResourceResponse& filteredResponse() const;
121
122 void closeStream();
123
124 void addAbortSteps(Ref<AbortSignal>&&);
125
126 class BodyLoader final : public FetchLoaderClient {
127 WTF_MAKE_FAST_ALLOCATED;
128 public:
129 BodyLoader(FetchResponse&, NotificationCallback&&);
130 ~BodyLoader();
131
132 bool start(ScriptExecutionContext&, const FetchRequest&);
133 void stop();
134
135 void consumeDataByChunk(ConsumeDataByChunkCallback&&);
136
137 RefPtr<SharedBuffer> startStreaming();
138 NotificationCallback takeNotificationCallback() { return WTFMove(m_responseCallback); }
139 ConsumeDataByChunkCallback takeConsumeDataCallback() { return WTFMove(m_consumeDataCallback); }
140
141 private:
142 // FetchLoaderClient API
143 void didSucceed() final;
144 void didFail(const ResourceError&) final;
145 void didReceiveResponse(const ResourceResponse&) final;
146 void didReceiveData(const uint8_t*, size_t) final;
147
148 FetchResponse& m_response;
149 NotificationCallback m_responseCallback;
150 ConsumeDataByChunkCallback m_consumeDataCallback;
151 std::unique_ptr<FetchLoader> m_loader;
152 Ref<PendingActivity<FetchResponse>> m_pendingActivity;
153 FetchOptions::Credentials m_credentials;
154 };
155
156 mutable std::optional<ResourceResponse> m_filteredResponse;
157 ResourceResponse m_internalResponse;
158 std::unique_ptr<BodyLoader> m_bodyLoader;
159 mutable String m_responseURL;
160 // Opaque responses will padd their body size when used with Cache API.
161 uint64_t m_bodySizeWithPadding { 0 };
162 uint64_t m_opaqueLoadIdentifier { 0 };
163 RefPtr<AbortSignal> m_abortSignal;
164};
