60void GamepadManager::platformGamepadConnected(PlatformGamepad& platformGamepad, EventMakesGamepadsVisible eventVisibility)
61{
62 if (eventVisibility == EventMakesGamepadsVisible::No)
63 return;
64
65 // Notify blind Navigators and Windows about all gamepads except for this one.
66 for (auto* gamepad : GamepadProvider::singleton().platformGamepads()) {
67 if (!gamepad || gamepad == &platformGamepad)
68 continue;
69
70 makeGamepadVisible(*gamepad, m_gamepadBlindNavigators, m_gamepadBlindDOMWindows);
71 }
72
73 m_gamepadBlindNavigators.clear();
74 m_gamepadBlindDOMWindows.clear();
75
76 // Notify everyone of this new gamepad.
77 makeGamepadVisible(platformGamepad, m_navigators, m_domWindows);
78}

80void GamepadManager::platformGamepadDisconnected(PlatformGamepad& platformGamepad)
81{
82 Vector<WeakPtr<DOMWindow>> weakWindows;
83 for (auto* domWindow : m_domWindows)
84 weakWindows.append(makeWeakPtr(*domWindow));
85
86 HashSet<NavigatorGamepad*> notifiedNavigators;
87
88 // Handle the disconnect for all DOMWindows with event listeners and their Navigators.
89 for (auto& window : weakWindows) {
90 // Event dispatch might have made this window go away.
91 if (!window)
92 continue;
93
94 // This DOMWindow's Navigator might not be accessible. e.g. The DOMWindow might be in the back/forward cache.
95 // If this happens the DOMWindow will not get this gamepaddisconnected event.
96 NavigatorGamepad* navigator = navigatorGamepadFromDOMWindow(window.get());
97 if (!navigator)
98 continue;
99
100 // If this Navigator hasn't seen gamepads yet then its Window should not get the disconnect event.
101 if (m_gamepadBlindNavigators.contains(navigator))
102 continue;
103
104 Ref<Gamepad> gamepad(navigator->gamepadFromPlatformGamepad(platformGamepad));
105
106 navigator->gamepadDisconnected(platformGamepad);
107 notifiedNavigators.add(navigator);
108
109 window->dispatchEvent(GamepadEvent::create(eventNames().gamepaddisconnectedEvent, gamepad.get()), window->document());
110 }
111
112 // Notify all the Navigators that haven't already been notified.
113 for (auto* navigator : m_navigators) {
114 if (!notifiedNavigators.contains(navigator))
115 navigator->gamepadDisconnected(platformGamepad);
116 }
117}

119void GamepadManager::platformGamepadInputActivity(EventMakesGamepadsVisible eventVisibility)
120{
121 if (eventVisibility == EventMakesGamepadsVisible::No)
122 return;
123
124 if (m_gamepadBlindNavigators.isEmpty() && m_gamepadBlindDOMWindows.isEmpty())
125 return;
126
127 for (auto* gamepad : GamepadProvider::singleton().platformGamepads()) {
128 if (gamepad)
129 makeGamepadVisible(*gamepad, m_gamepadBlindNavigators, m_gamepadBlindDOMWindows);
130 }
131
132 m_gamepadBlindNavigators.clear();
133 m_gamepadBlindDOMWindows.clear();
134}

136void GamepadManager::makeGamepadVisible(PlatformGamepad& platformGamepad, HashSet<NavigatorGamepad*>& navigatorSet, HashSet<DOMWindow*>& domWindowSet)
137{
138 LOG(Gamepad, "GamepadManager::makeGamepadVisible - New gamepad '%s' is visible", platformGamepad.id().utf8().data());
139
140 if (navigatorSet.isEmpty() && domWindowSet.isEmpty())
141 return;
142
143 for (auto* navigator : navigatorSet)
144 navigator->gamepadConnected(platformGamepad);
145
146 Vector<WeakPtr<DOMWindow>> weakWindows;
147 for (auto* domWindow : m_domWindows)
148 weakWindows.append(makeWeakPtr(*domWindow));
149
150 for (auto& window : weakWindows) {
151 // Event dispatch might have made this window go away.
152 if (!window)
153 continue;
154
155 // This DOMWindow's Navigator might not be accessible. e.g. The DOMWindow might be in the back/forward cache.
156 // If this happens the DOMWindow will not get this gamepadconnected event.
157 // The new gamepad will still be visibile to it once it is restored from the back/forward cache.
158 NavigatorGamepad* navigator = navigatorGamepadFromDOMWindow(window.get());
159 if (!navigator)
160 continue;
161
162 Ref<Gamepad> gamepad(navigator->gamepadFromPlatformGamepad(platformGamepad));
163
164 LOG(Gamepad, "GamepadManager::makeGamepadVisible - Dispatching gamepadconnected event for gamepad '%s'", platformGamepad.id().utf8().data());
165 window->dispatchEvent(GamepadEvent::create(eventNames().gamepadconnectedEvent, gamepad.get()), window->document());
166 }

169void GamepadManager::registerNavigator(NavigatorGamepad* navigator)
170{
171 LOG(Gamepad, "GamepadManager registering NavigatorGamepad %p", navigator);
172
173 ASSERT(!m_navigators.contains(navigator));
174 m_navigators.add(navigator);
175 m_gamepadBlindNavigators.add(navigator);
176
177 maybeStartMonitoringGamepads();
178}
179
180void GamepadManager::unregisterNavigator(NavigatorGamepad* navigator)
181{
182 LOG(Gamepad, "GamepadManager unregistering NavigatorGamepad %p", navigator);
183
184 ASSERT(m_navigators.contains(navigator));
185 m_navigators.remove(navigator);
186 m_gamepadBlindNavigators.remove(navigator);
187
188 maybeStopMonitoringGamepads();
189}
190
191void GamepadManager::registerDOMWindow(DOMWindow* window)
192{
193 LOG(Gamepad, "GamepadManager registering DOMWindow %p", window);
194
195 ASSERT(!m_domWindows.contains(window));
196 m_domWindows.add(window);
197
198 // Anytime we register a DOMWindow, we should also double check that its NavigatorGamepad is registered.
199 NavigatorGamepad* navigator = navigatorGamepadFromDOMWindow(window);
200 ASSERT(navigator);
201
202 if (m_navigators.add(navigator).isNewEntry)
203 m_gamepadBlindNavigators.add(navigator);
204
205 // If this DOMWindow's NavigatorGamepad was already registered but was still blind,
206 // then this DOMWindow should be blind.
207 if (m_gamepadBlindNavigators.contains(navigator))
208 m_gamepadBlindDOMWindows.add(window);
209
210 maybeStartMonitoringGamepads();
211}
212
213void GamepadManager::unregisterDOMWindow(DOMWindow* window)
214{
215 LOG(Gamepad, "GamepadManager unregistering DOMWindow %p", window);
216
217 ASSERT(m_domWindows.contains(window));
218 m_domWindows.remove(window);
219 m_gamepadBlindDOMWindows.remove(window);
220
221 maybeStopMonitoringGamepads();
222}
223
224void GamepadManager::maybeStartMonitoringGamepads()
225{
226 if (m_isMonitoringGamepads)
227 return;
228
229 if (!m_navigators.isEmpty() || !m_domWindows.isEmpty()) {
230 LOG(Gamepad, "GamepadManager has %i NavigatorGamepads and %i DOMWindows registered, is starting gamepad monitoring", m_navigators.size(), m_domWindows.size());
231 m_isMonitoringGamepads = true;
232 GamepadProvider::singleton().startMonitoringGamepads(*this);
233 }
234}
235
236void GamepadManager::maybeStopMonitoringGamepads()
237{
238 if (!m_isMonitoringGamepads)
239 return;
240
241 if (m_navigators.isEmpty() && m_domWindows.isEmpty()) {
242 LOG(Gamepad, "GamepadManager has no NavigatorGamepads or DOMWindows registered, is stopping gamepad monitoring");
243 m_isMonitoringGamepads = false;
244 GamepadProvider::singleton().stopMonitoringGamepads(*this);
245 }
