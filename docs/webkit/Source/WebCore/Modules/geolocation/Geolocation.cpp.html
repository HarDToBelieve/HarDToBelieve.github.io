68static Ref<GeolocationPositionError> createGeolocationPositionError(GeolocationError& error)
69{
70 auto code = GeolocationPositionError::POSITION_UNAVAILABLE;
71 switch (error.code()) {
72 case GeolocationError::PermissionDenied:
73 code = GeolocationPositionError::PERMISSION_DENIED;
74 break;
75 case GeolocationError::PositionUnavailable:
76 code = GeolocationPositionError::POSITION_UNAVAILABLE;
77 break;
78 }
79
80 return GeolocationPositionError::create(code, error.message());
81}

164{
165 if (reason == ReasonForSuspension::BackForwardCache) {

167 m_resetOnResume = true;
168 }

171 if (hasListeners())

173
174 m_isSuspended = true;
175 m_resumeTimer.stop();

190void Geolocation::resumeTimerFired()
191{
192 m_isSuspended = false;
193
194 if (m_resetOnResume) {
195 resetAllGeolocationPermission();
196 m_resetOnResume = false;
197 }
198
199 // Resume GeoNotifier timeout timers.
200 if (hasListeners()) {
201 for (auto& notifier : m_oneShots)
202 notifier->startTimerIfNeeded();
203 GeoNotifierVector watcherCopy;
204 m_watchers.getNotifiersVector(watcherCopy);
205 for (auto& watcher : watcherCopy)
206 watcher->startTimerIfNeeded();
207 }
208
209 if ((isAllowed() || isDenied()) && !m_pendingForPermissionNotifiers.isEmpty()) {
210 // The pending permission was granted while the object was suspended.
211 setIsAllowed(isAllowed(), authorizationToken());
212 ASSERT(!m_hasChangedPosition);
213 ASSERT(!m_errorWaitingForResume);
214 return;
215 }
216
217 if (isDenied() && hasListeners()) {
218 // The permission was revoked while the object was suspended.
219 setIsAllowed(false, { });
220 return;
221 }
222
223 if (m_hasChangedPosition) {
224 positionChanged();
225 m_hasChangedPosition = false;
226 }
227
228 if (m_errorWaitingForResume) {
229 handleError(*m_errorWaitingForResume);
230 m_errorWaitingForResume = nullptr;
231 }
232}

234void Geolocation::resetAllGeolocationPermission()
235{
236 if (m_isSuspended) {
237 m_resetOnResume = true;
238 return;
239 }
240
241 if (m_allowGeolocation == InProgress) {
242 Page* page = this->page();
243 if (page)
244 GeolocationController::from(page)->cancelPermissionRequest(*this);
245
246 // This return is not technically correct as GeolocationController::cancelPermissionRequest() should have cleared the active request.
247 // Neither iOS nor OS X supports cancelPermissionRequest() (https://bugs.webkit.org/show_bug.cgi?id=89524), so we workaround that and let ongoing requests complete. :(
248 return;
249 }
250
251 // 1) Reset our own state.
252 stopUpdating();
253 resetIsAllowed();
254 m_hasChangedPosition = false;
255 m_errorWaitingForResume = nullptr;
256
257 // 2) Request new permission for the active notifiers.
258 stopTimers();
259
260 // Go over the one shot and re-request permission.
261 for (auto& notifier : m_oneShots)
262 startRequest(notifier.get());
263 // Go over the watchers and re-request permission.
264 GeoNotifierVector watcherCopy;
265 m_watchers.getNotifiersVector(watcherCopy);
266 for (auto& watcher : watcherCopy)
267 startRequest(watcher.get());
268}

270void Geolocation::stop()
271{
272 Page* page = this->page();
273 if (page && m_allowGeolocation == InProgress)
274 GeolocationController::from(page)->cancelPermissionRequest(*this);
275 // The frame may be moving to a new page and we want to get the permissions from the new page's client.
276 resetIsAllowed();
277 cancelAllRequests();
278 stopUpdating();
279 m_hasChangedPosition = false;
280 m_errorWaitingForResume = nullptr;
281 m_pendingForPermissionNotifiers.clear();
282}

311int Geolocation::watchPosition(Ref<PositionCallback>&& successCallback, RefPtr<PositionErrorCallback>&& errorCallback, PositionOptions&& options)
312{
313 if (!frame())
314 return 0;
315
316 auto notifier = GeoNotifier::create(*this, WTFMove(successCallback), WTFMove(errorCallback), WTFMove(options));
317 startRequest(notifier.ptr());
318
319 int watchID;
320 // Keep asking for the next id until we're given one that we don't already have.
321 do {
322 watchID = m_scriptExecutionContext->circularSequentialID();
323 } while (!m_watchers.add(watchID, notifier.copyRef()));
324 return watchID;
325}

327static void logError(const String& target, const bool isSecure, const bool isMixedContent, Document* document)
328{
329 StringBuilder message;
330 message.append("[blocked] Access to geolocation was blocked over");
331
332 if (!isSecure)
333 message.append(" insecure connection to ");
334 else if (isMixedContent)
335 message.append(" secure connection with mixed content to ");
336 else
337 return;
338
339 message.append(target);
340 message.append(".\n");
341 document->addConsoleMessage(MessageSource::Security, MessageLevel::Error, message.toString());
342}

353bool Geolocation::shouldBlockGeolocationRequests()
354{
355 if (!isFeaturePolicyAllowedByDocumentAndAllOwners(FeaturePolicy::Type::Geolocation, *document(), LogFeaturePolicyFailure::Yes))
356 return true;
357
358 bool isSecure = SecurityOrigin::isSecure(document()->url()) || document()->isSecureContext();
359 bool hasMixedContent = !document()->foundMixedContent().isEmpty();
360 bool isLocalOrigin = securityOrigin()->isLocal();
361 if (securityOrigin()->canRequestGeolocation()) {
362 if (isLocalOrigin || (isSecure && !hasMixedContent) || isRequestFromIBooks())
363 return false;
364 }
365
366 logError(securityOrigin()->toString(), isSecure, hasMixedContent, document());
367 return true;
368}

370void Geolocation::startRequest(GeoNotifier* notifier)
371{
372 if (shouldBlockGeolocationRequests()) {
373 notifier->setFatalError(GeolocationPositionError::create(GeolocationPositionError::PERMISSION_DENIED, originCannotRequestGeolocationErrorMessage));
374 return;
375 }
376 document()->setGeolocationAccessed();
377
378 // Check whether permissions have already been denied. Note that if this is the case,
379 // the permission state can not change again in the lifetime of this page.
380 if (isDenied())
381 notifier->setFatalError(GeolocationPositionError::create(GeolocationPositionError::PERMISSION_DENIED, permissionDeniedErrorMessage));
382 else if (haveSuitableCachedPosition(notifier->options()))
383 notifier->setUseCachedPosition();
384 else if (notifier->hasZeroTimeout())
385 notifier->startTimerIfNeeded();
386 else if (!isAllowed()) {
387 // if we don't yet have permission, request for permission before calling startUpdating()
388 m_pendingForPermissionNotifiers.add(notifier);
389 requestPermission();
390 } else if (startUpdating(notifier))
391 notifier->startTimerIfNeeded();
392 else
393 notifier->setFatalError(GeolocationPositionError::create(GeolocationPositionError::POSITION_UNAVAILABLE, failedToStartServiceErrorMessage));
394}

396void Geolocation::fatalErrorOccurred(GeoNotifier* notifier)
397{
398 // This request has failed fatally. Remove it from our lists.
399 m_oneShots.remove(notifier);
400 m_watchers.remove(notifier);
401
402 if (!hasListeners())
403 stopUpdating();

406void Geolocation::requestUsesCachedPosition(GeoNotifier* notifier)
407{
408 // This is called asynchronously, so the permissions could have been denied
409 // since we last checked in startRequest.
410 if (isDenied()) {
411 notifier->setFatalError(GeolocationPositionError::create(GeolocationPositionError::PERMISSION_DENIED, permissionDeniedErrorMessage));
412 return;
413 }
414
415 m_requestsAwaitingCachedPosition.add(notifier);
416
417 // If permissions are allowed, make the callback
418 if (isAllowed()) {
419 makeCachedPositionCallbacks();
420 return;
421 }
422
423 // Request permissions, which may be synchronous or asynchronous.
424 requestPermission();
425}

427void Geolocation::makeCachedPositionCallbacks()
428{
429 // All modifications to m_requestsAwaitingCachedPosition are done
430 // asynchronously, so we don't need to worry about it being modified from
431 // the callbacks.
432 for (auto& notifier : m_requestsAwaitingCachedPosition) {
433 // FIXME: This seems wrong, since makeCachedPositionCallbacks() is called in a branch where
434 // lastPosition() is known to be null in Geolocation::setIsAllowed().
435 notifier->runSuccessCallback(lastPosition());
436
437 // If this is a one-shot request, stop it. Otherwise, if the watch still
438 // exists, start the service to get updates.
439 if (!m_oneShots.remove(notifier.get()) && m_watchers.contains(notifier.get())) {
440 if (notifier->hasZeroTimeout() || startUpdating(notifier.get()))
441 notifier->startTimerIfNeeded();
442 else
443 notifier->setFatalError(GeolocationPositionError::create(GeolocationPositionError::POSITION_UNAVAILABLE, failedToStartServiceErrorMessage));
444 }
445 }
446
447 m_requestsAwaitingCachedPosition.clear();
448
449 if (!hasListeners())
450 stopUpdating();
451}

462bool Geolocation::haveSuitableCachedPosition(const PositionOptions& options)
463{
464 auto* cachedPosition = lastPosition();
465 if (!cachedPosition)
466 return false;
467 if (!options.maximumAge)
468 return false;
469 DOMTimeStamp currentTimeMillis = convertSecondsToDOMTimeStamp(WallTime::now().secondsSinceEpoch());
470 return cachedPosition->timestamp() > currentTimeMillis - options.maximumAge;
471}

474{

486void Geolocation::setIsAllowed(bool allowed, const String& authorizationToken)
487{
488 // Protect the Geolocation object from garbage collection during a callback.
489 Ref<Geolocation> protectedThis(*this);
490
491 // This may be due to either a new position from the service, or a cached
492 // position.
493 m_allowGeolocation = allowed ? Yes : No;
494 m_authorizationToken = authorizationToken;
495
496 if (m_isSuspended)
497 return;
498
499 // Permission request was made during the startRequest process
500 if (!m_pendingForPermissionNotifiers.isEmpty()) {
501 handlePendingPermissionNotifiers();
502 m_pendingForPermissionNotifiers.clear();
503 return;
504 }
505
506 if (!isAllowed()) {
507 auto error = GeolocationPositionError::create(GeolocationPositionError::PERMISSION_DENIED, permissionDeniedErrorMessage);
508 error->setIsFatal(true);
509 handleError(error);
510 m_requestsAwaitingCachedPosition.clear();
511 m_hasChangedPosition = false;
512 m_errorWaitingForResume = nullptr;
513 return;
514 }
515
516 // If the service has a last position, use it to call back for all requests.
517 // If any of the requests are waiting for permission for a cached position,
518 // the position from the service will be at least as fresh.
519 if (RefPtr<GeolocationPosition> position = lastPosition())
520 makeSuccessCallbacks(*position);
521 else
522 makeCachedPositionCallbacks();
523}

577void Geolocation::extractNotifiersWithCachedPosition(GeoNotifierVector& notifiers, GeoNotifierVector* cached)
578{
579 GeoNotifierVector nonCached;
580 for (auto& notifier : notifiers) {
581 if (notifier->useCachedPosition()) {
582 if (cached)
583 cached->append(notifier.get());
584 } else
585 nonCached.append(notifier.get());
586 }
587 notifiers.swap(nonCached);
588}

596void Geolocation::handleError(GeolocationPositionError& error)
597{
598 auto oneShotsCopy = copyToVector(m_oneShots);
599
600 GeoNotifierVector watchersCopy;
601 m_watchers.getNotifiersVector(watchersCopy);
602
603 // Clear the lists before we make the callbacks, to avoid clearing notifiers
604 // added by calls to Geolocation methods from the callbacks, and to prevent
605 // further callbacks to these notifiers.
606 GeoNotifierVector oneShotsWithCachedPosition;
607 m_oneShots.clear();
608 if (error.isFatal())
609 m_watchers.clear();
610 else {
611 // Don't send non-fatal errors to notifiers due to receive a cached position.
612 extractNotifiersWithCachedPosition(oneShotsCopy, &oneShotsWithCachedPosition);
613 extractNotifiersWithCachedPosition(watchersCopy, 0);
614 }
615
616 sendError(oneShotsCopy, error);
617 sendError(watchersCopy, error);
618
619 // hasListeners() doesn't distinguish between notifiers due to receive a
620 // cached position and those requiring a fresh position. Perform the check
621 // before restoring the notifiers below.
622 if (!hasListeners())
623 stopUpdating();
624
625 // Maintain a reference to the cached notifiers until their timer fires.
626 copyToSet(oneShotsWithCachedPosition, m_oneShots);
627}

662void Geolocation::makeSuccessCallbacks(GeolocationPosition& position)
663{
664 ASSERT(lastPosition());
665 ASSERT(isAllowed());
666
667 auto oneShotsCopy = copyToVector(m_oneShots);
668
669 GeoNotifierVector watchersCopy;
670 m_watchers.getNotifiersVector(watchersCopy);
671
672 // Clear the lists before we make the callbacks, to avoid clearing notifiers
673 // added by calls to Geolocation methods from the callbacks, and to prevent
674 // further callbacks to these notifiers.
675 m_oneShots.clear();
676
677 sendPosition(oneShotsCopy, position);
678 sendPosition(watchersCopy, position);
679
680 if (!hasListeners())
681 stopUpdating();
682}

684void Geolocation::positionChanged()
685{
686 ASSERT(isAllowed());
687
688 // Stop all currently running timers.
689 stopTimers();
690
691 if (m_isSuspended) {
692 m_hasChangedPosition = true;
693 return;
694 }
695
696 RefPtr<GeolocationPosition> position = lastPosition();
697 ASSERT(position);
698
699 makeSuccessCallbacks(*position);
700}

732void Geolocation::handlePendingPermissionNotifiers()
733{
734 // While we iterate through the list, we need not worry about list being modified as the permission
735 // is already set to Yes/No and no new listeners will be added to the pending list
736 for (auto& notifier : m_pendingForPermissionNotifiers) {
737 if (isAllowed()) {
738 // start all pending notification requests as permission granted.
739 // The notifier is always ref'ed by m_oneShots or m_watchers.
740 if (startUpdating(notifier.get()))
741 notifier->startTimerIfNeeded();
742 else
743 notifier->setFatalError(GeolocationPositionError::create(GeolocationPositionError::POSITION_UNAVAILABLE, failedToStartServiceErrorMessage));
744 } else
745 notifier->setFatalError(GeolocationPositionError::create(GeolocationPositionError::PERMISSION_DENIED, permissionDeniedErrorMessage));
746 }
