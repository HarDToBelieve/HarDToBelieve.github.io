26#pragma once
27
28#include "GeolocationPositionData.h"
29#include <wtf/Ref.h>
30#include <wtf/RefCounted.h>
31
32namespace WebCore {
33
34class GeolocationCoordinates : public RefCounted<GeolocationCoordinates> {
35public:
36 static Ref<GeolocationCoordinates> create(GeolocationPositionData&& position)
37 {
38 return adoptRef(*new GeolocationCoordinates(WTFMove(position)));
39 }
40
41 Ref<GeolocationCoordinates> isolatedCopy() const
42 {
43 return GeolocationCoordinates::create(GeolocationPositionData(m_position));
44 }
45
46 double latitude() const { return m_position.latitude; }
47 double longitude() const { return m_position.longitude; }
48 std::optional<double> altitude() const { return m_position.altitude; }
49 double accuracy() const { return m_position.accuracy; }
50 std::optional<double> altitudeAccuracy() const { return m_position.altitudeAccuracy; }
51 std::optional<double> heading() const { return m_position.heading; }
52 std::optional<double> speed() const { return m_position.speed; }
53 std::optional<double> floorLevel() const { return m_position.floorLevel; }
54
55private:
56 explicit GeolocationCoordinates(GeolocationPositionData&&);
57
58 GeolocationPositionData m_position;
59};
60
61} // namespace WebCore
