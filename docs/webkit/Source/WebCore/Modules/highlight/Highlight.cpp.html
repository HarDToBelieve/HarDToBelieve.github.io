46Highlight::Highlight(Ref<StaticRange>&& range)
47{
48 auto myRange = WTFMove(range);
49 addToSetLike(myRange.get());
50}
51
52Ref<Highlight> Highlight::create(StaticRange& range)
53{
54 return adoptRef(*new Highlight(range));
55}
56
57void Highlight::initializeSetLike(DOMSetAdapter& set)
58{
59 for (auto& rangeData : m_rangesData)
60 set.add<IDLInterface<StaticRange>>(rangeData->range);
61}
62
63static void repaintRange(const SimpleRange& range)
64{
65 // FIXME: Unclear precisely why we need to handle out of order cases here, but not unordered cases.
66 auto sortedRange = range;
67 if (is_gt(treeOrder<ComposedTree>(range.start, range.end)))
68 std::swap(sortedRange.start, sortedRange.end);
69 for (auto& node : intersectingNodes(sortedRange)) {
70 if (auto renderer = node.renderer()) {
71#if ENABLE(LAYOUT_FORMATTING_CONTEXT)
72 if (auto lineLayout = LayoutIntegration::LineLayout::containing(*renderer))
73 lineLayout->flow().ensureLineBoxes();
74#endif
75 renderer->repaint();
76 }
77 }
78}
79
80bool Highlight::removeFromSetLike(const StaticRange& range)
81{
82 return m_rangesData.removeFirstMatching([&range](const Ref<HighlightRangeData>& current) {
83 repaintRange(range);
84 return current.get().range.get() == range;
85 });
86}
87
88void Highlight::clearFromSetLike()
89{
90 for (auto& data : m_rangesData)
91 repaintRange(data->range);
92 m_rangesData.clear();
93}
94
95bool Highlight::addToSetLike(StaticRange& range)
96{
97 if (notFound != m_rangesData.findMatching([&range](const Ref<HighlightRangeData>& current) { return current.get().range.get() == range; }))
98 return false;
99 repaintRange(range);
100 m_rangesData.append(HighlightRangeData::create(range));
101 return true;
102}
103
104void Highlight::repaint()
105{
106 for (auto& data : m_rangesData)
107 repaintRange(data->range);
108}
