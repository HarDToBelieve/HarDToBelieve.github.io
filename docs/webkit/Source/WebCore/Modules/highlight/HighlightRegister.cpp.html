33namespace WebCore {
34
35void HighlightRegister::initializeMapLike(DOMMapAdapter& map)
36{
37 for (auto& keyValue : m_map)
38 map.set<IDLDOMString, IDLInterface<Highlight>>(keyValue.key, keyValue.value);
39}
40
41void HighlightRegister::setFromMapLike(String&& key, Ref<Highlight>&& value)
42{
43 m_map.set(WTFMove(key), WTFMove(value));
44}
45
46void HighlightRegister::clear()
47{
48 m_map.clear();
49}
50
51bool HighlightRegister::remove(const String& key)
52{
53 return m_map.remove(key);
54}
55#if ENABLE(APP_HIGHLIGHTS)
56ASCIILiteral HighlightRegister::appHighlightKey()
57{
58 return "appHighlightKey"_s;
59}
60
61void HighlightRegister::addAppHighlight(Ref<StaticRange>&& value)
62{
63 if (m_map.contains(appHighlightKey()))
64 m_map.get(appHighlightKey())->addToSetLike(value);
65 else
66 setFromMapLike(appHighlightKey(), Highlight::create(WTFMove(value)));
67}
68
69void HighlightRegister::setHighlightVisibility(HighlightVisibility highlightVisibility)
70{
71 if (m_highlightVisibility == highlightVisibility)
72 return;
73
74 m_highlightVisibility = highlightVisibility;
75
76 for (auto& highlight : m_map)
77 highlight.value->repaint();
78}

80#endif
