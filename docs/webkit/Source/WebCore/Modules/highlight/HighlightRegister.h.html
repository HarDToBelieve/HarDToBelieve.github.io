27
28#include "Highlight.h"
29#include "HighlightVisibility.h"
30#include <wtf/HashMap.h>
31#include <wtf/RefCounted.h>
32
33namespace WebCore {
34
35class DOMMapAdapter;
36class DOMString;
37class Highlight;
38class StaticRange;
39
40class HighlightRegister : public RefCounted<HighlightRegister> {
41public:
42 static Ref<HighlightRegister> create() { return adoptRef(*new HighlightRegister); }
43
44 void initializeMapLike(DOMMapAdapter&);
45 void setFromMapLike(String&&, Ref<Highlight>&&);
46 void clear();
47 bool remove(const String&);
48
49 HighlightVisibility highlightsVisibility() const { return m_highlightVisibility; }
50 WEBCORE_EXPORT void setHighlightVisibility(HighlightVisibility);
51
52#if ENABLE(APP_HIGHLIGHTS)
53 WEBCORE_EXPORT void addAppHighlight(Ref<StaticRange>&&);
54 static ASCIILiteral appHighlightKey();
55#endif
56 const HashMap<String, Ref<Highlight>>& map() const { return m_map; }
57
58private:
59 HighlightRegister() = default;
60 HashMap<String, Ref<Highlight>> m_map;
61
62 HighlightVisibility m_highlightVisibility { HighlightVisibility::Hidden };
63};
64
65}
