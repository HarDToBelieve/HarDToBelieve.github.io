39namespace WebCore {
40
41class IDBGetResult;
42class IDBIndex;
43class IDBObjectStore;
44class IDBTransaction;
45
46class IDBCursor : public ScriptWrappable, public RefCounted<IDBCursor> {
47 WTF_MAKE_ISO_ALLOCATED(IDBCursor);
48public:
49 static Ref<IDBCursor> create(IDBObjectStore&, const IDBCursorInfo&);
50 static Ref<IDBCursor> create(IDBIndex&, const IDBCursorInfo&);
51
52 virtual ~IDBCursor();
53
54 using Source = Variant<RefPtr<IDBObjectStore>, RefPtr<IDBIndex>>;
55
56 const Source& source() const;
57 IDBCursorDirection direction() const;
58
59 IDBKey* key() { return m_key.get(); };
60 IDBKey* primaryKey() { return m_primaryKey.get(); };
61 IDBValue value() { return m_value; };
62 const std::optional<IDBKeyPath>& primaryKeyPath() { return m_keyPath; };
63 JSValueInWrappedObject& keyWrapper() { return m_keyWrapper; }
64 JSValueInWrappedObject& primaryKeyWrapper() { return m_primaryKeyWrapper; }
65 JSValueInWrappedObject& valueWrapper() { return m_valueWrapper; }
66
67 ExceptionOr<Ref<IDBRequest>> update(JSC::JSGlobalObject&, JSC::JSValue);
68 ExceptionOr<void> advance(unsigned);
69 ExceptionOr<void> continueFunction(JSC::JSGlobalObject&, JSC::JSValue key);
70 ExceptionOr<void> continuePrimaryKey(JSC::JSGlobalObject&, JSC::JSValue key, JSC::JSValue primaryKey);
71 ExceptionOr<Ref<IDBRequest>> deleteFunction(JSC::JSGlobalObject&);
72
73 ExceptionOr<void> continueFunction(const IDBKeyData&);
74
75 const IDBCursorInfo& info() const { return m_info; }
76
77 void setRequest(IDBRequest& request) { m_request = makeWeakPtr(&request); }
78 void clearRequest() { m_request.clear(); }
79 void clearWrappers();
80 IDBRequest* request() { return m_request.get(); }
81
82 bool setGetResult(IDBRequest&, const IDBGetResult&, uint64_t operationID);
83
84 virtual bool isKeyCursorWithValue() const { return false; }
85
86 std::optional<IDBGetResult> iterateWithPrefetchedRecords(unsigned count, uint64_t lastWriteOperationID);
87 void clearPrefetchedRecords();
88
89protected:
90 IDBCursor(IDBObjectStore&, const IDBCursorInfo&);
91 IDBCursor(IDBIndex&, const IDBCursorInfo&);
92
93private:
94 bool sourcesDeleted() const;
95 IDBObjectStore& effectiveObjectStore() const;
96 IDBTransaction& transaction() const;
97
98 void uncheckedIterateCursor(const IDBKeyData&, unsigned count);
99 void uncheckedIterateCursor(const IDBKeyData&, const IDBKeyData&);
100
101 IDBCursorInfo m_info;
102 Source m_source;
103 WeakPtr<IDBRequest> m_request;
104
105 bool m_gotValue { false };
106
107 RefPtr<IDBKey> m_key;
108 RefPtr<IDBKey> m_primaryKey;
109 IDBKeyData m_keyData;
110 IDBKeyData m_primaryKeyData;
111 IDBValue m_value;
112 std::optional<IDBKeyPath> m_keyPath;
113
114 JSValueInWrappedObject m_keyWrapper;
115 JSValueInWrappedObject m_primaryKeyWrapper;
116 JSValueInWrappedObject m_valueWrapper;
117
118 Deque<IDBCursorRecord> m_prefetchedRecords;
119 uint64_t m_prefetchOperationID { 0 };
120};
121
122
123inline const IDBCursor::Source& IDBCursor::source() const
124{
125 return m_source;
126}
127
128inline IDBCursorDirection IDBCursor::direction() const
129{
130 return m_info.cursorDirection();
131}
