55 : IDBActiveDOMObject(&context)

60{

76bool IDBDatabase::virtualHasPendingActivity() const
77{
78 ASSERT(canCurrentThreadAccessThreadLocalData(originThread()) || Thread::mayBeGCThread());
79
80 if (m_closedInServer)
81 return false;
82
83 if (!m_activeTransactions.isEmpty() || !m_committingTransactions.isEmpty() || !m_abortingTransactions.isEmpty())
84 return true;
85
86 return hasEventListeners(m_eventNames.abortEvent) || hasEventListeners(m_eventNames.errorEvent) || hasEventListeners(m_eventNames.versionchangeEvent);
87}

135ExceptionOr<Ref<IDBObjectStore>> IDBDatabase::createObjectStore(const String& name, ObjectStoreParameters&& parameters)
136{
137 LOG(IndexedDB, "IDBDatabase::createObjectStore - (%s %s)", m_info.name().utf8().data(), name.utf8().data());
138
139 ASSERT(canCurrentThreadAccessThreadLocalData(originThread()));
140 ASSERT(!m_versionChangeTransaction || m_versionChangeTransaction->isVersionChange());
141
142 if (!m_versionChangeTransaction)
143 return Exception { InvalidStateError, "Failed to execute 'createObjectStore' on 'IDBDatabase': The database is not running a version change transaction."_s };
144
145 if (!m_versionChangeTransaction->isActive())
146 return Exception { TransactionInactiveError };
147
148 auto& keyPath = parameters.keyPath;
149 if (keyPath && !isIDBKeyPathValid(keyPath.value()))
150 return Exception { SyntaxError, "Failed to execute 'createObjectStore' on 'IDBDatabase': The keyPath option is not a valid key path."_s };
151
152 if (m_info.hasObjectStore(name))
153 return Exception { ConstraintError, "Failed to execute 'createObjectStore' on 'IDBDatabase': An object store with the specified name already exists."_s };
154
155 if (keyPath && parameters.autoIncrement && ((WTF::holds_alternative<String>(keyPath.value()) && WTF::get<String>(keyPath.value()).isEmpty()) || WTF::holds_alternative<Vector<String>>(keyPath.value())))
156 return Exception { InvalidAccessError, "Failed to execute 'createObjectStore' on 'IDBDatabase': The autoIncrement option was set but the keyPath option was empty or an array."_s };
157
158 // Install the new ObjectStore into the connection's metadata.
159 auto info = m_info.createNewObjectStore(name, WTFMove(keyPath), parameters.autoIncrement);
160
161 // Create the actual IDBObjectStore from the transaction, which also schedules the operation server side.
162 return m_versionChangeTransaction->createObjectStore(info);
163}

165ExceptionOr<Ref<IDBTransaction>> IDBDatabase::transaction(StringOrVectorOfStrings&& storeNames, IDBTransactionMode mode)
166{
167 LOG(IndexedDB, "IDBDatabase::transaction");
168
169 ASSERT(canCurrentThreadAccessThreadLocalData(originThread()));
170
171 if (m_versionChangeTransaction && !m_versionChangeTransaction->isFinishedOrFinishing())
172 return Exception { InvalidStateError, "Failed to execute 'transaction' on 'IDBDatabase': A version change transaction is running."_s };
173
174 if (m_closePending)
175 return Exception { InvalidStateError, "Failed to execute 'transaction' on 'IDBDatabase': The database connection is closing."_s };
176
177 Vector<String> objectStores;
178 if (WTF::holds_alternative<Vector<String>>(storeNames))
179 objectStores = WTFMove(WTF::get<Vector<String>>(storeNames));
180 else
181 objectStores.append(WTFMove(WTF::get<String>(storeNames)));
182
183 // It is valid for javascript to pass in a list of object store names with the same name listed twice,
184 // so we need to put them all in a set to get a unique list.
185 HashSet<String> objectStoreSet;
186 for (auto& objectStore : objectStores)
187 objectStoreSet.add(objectStore);
188
189 objectStores = copyToVector(objectStoreSet);
190
191 for (auto& objectStoreName : objectStores) {
192 if (m_info.hasObjectStore(objectStoreName))
193 continue;
194 return Exception { NotFoundError, "Failed to execute 'transaction' on 'IDBDatabase': One of the specified object stores was not found."_s };
195 }
196
197 if (objectStores.isEmpty())
198 return Exception { InvalidAccessError, "Failed to execute 'transaction' on 'IDBDatabase': The storeNames parameter was empty."_s };
199
200 if (mode != IDBTransactionMode::Readonly && mode != IDBTransactionMode::Readwrite)
201 return Exception { TypeError };
202
203 auto info = IDBTransactionInfo::clientTransaction(m_connectionProxy.get(), objectStores, mode);
204
205 LOG(IndexedDBOperations, "IDB creating transaction: %s", info.loggingString().utf8().data());
206 auto transaction = IDBTransaction::create(*this, info);
207
208 LOG(IndexedDB, "IDBDatabase::transaction - Added active transaction %s", info.identifier().loggingString().utf8().data());
209
210 m_activeTransactions.set(info.identifier(), transaction.ptr());
211
212 return transaction;
213}

215ExceptionOr<void> IDBDatabase::deleteObjectStore(const String& objectStoreName)
216{
217 LOG(IndexedDB, "IDBDatabase::deleteObjectStore");
218
219 ASSERT(canCurrentThreadAccessThreadLocalData(originThread()));
220
221 if (!m_versionChangeTransaction)
222 return Exception { InvalidStateError, "Failed to execute 'deleteObjectStore' on 'IDBDatabase': The database is not running a version change transaction."_s };
223
224 if (!m_versionChangeTransaction->isActive())
225 return Exception { TransactionInactiveError };
226
227 if (!m_info.hasObjectStore(objectStoreName))
228 return Exception { NotFoundError, "Failed to execute 'deleteObjectStore' on 'IDBDatabase': The specified object store was not found."_s };
229
230 m_info.deleteObjectStore(objectStoreName);
231 m_versionChangeTransaction->deleteObjectStore(objectStoreName);
232
233 return { };
234}

257void IDBDatabase::connectionToServerLost(const IDBError& error)
258{
259 LOG(IndexedDB, "IDBDatabase::connectionToServerLost - %" PRIu64, m_databaseConnectionIdentifier);
260
261 ASSERT(canCurrentThreadAccessThreadLocalData(originThread()));
262
263 m_closePending = true;
264 m_closedInServer = true;
265
266 auto activeTransactions = copyToVector(m_activeTransactions.values());
267 for (auto& transaction : activeTransactions)
268 transaction->connectionClosedFromServer(error);
269
270 auto committingTransactions = copyToVector(m_committingTransactions.values());
271 for (auto& transaction : committingTransactions)
272 transaction->connectionClosedFromServer(error);
273
274 auto errorEvent = Event::create(m_eventNames.errorEvent, Event::CanBubble::Yes, Event::IsCancelable::No);
275 errorEvent->setTarget(this);
276
277 if (scriptExecutionContext())
278 queueTaskToDispatchEvent(*this, TaskSource::DatabaseAccess, WTFMove(errorEvent));
279
280 auto closeEvent = Event::create(m_eventNames.closeEvent, Event::CanBubble::Yes, Event::IsCancelable::No);
281 closeEvent->setTarget(this);
282
283 if (scriptExecutionContext())
284 queueTaskToDispatchEvent(*this, TaskSource::DatabaseAccess, WTFMove(closeEvent));
285}

287void IDBDatabase::maybeCloseInServer()
288{
289 LOG(IndexedDB, "IDBDatabase::maybeCloseInServer - %" PRIu64, m_databaseConnectionIdentifier);
290
291 ASSERT(canCurrentThreadAccessThreadLocalData(originThread()));
292
293 if (m_closedInServer)
294 return;
295
296 // 3.3.9 Database closing steps
297 // Wait for all transactions created using this connection to complete.
298 // Once they are complete, this connection is closed.
299 if (!m_activeTransactions.isEmpty() || !m_committingTransactions.isEmpty())
300 return;
301
302 m_closedInServer = true;
303 m_connectionProxy->databaseConnectionClosed(*this);
304}

312void IDBDatabase::stop()
313{
314 LOG(IndexedDB, "IDBDatabase::stop - %" PRIu64, m_databaseConnectionIdentifier);
315
316 ASSERT(canCurrentThreadAccessThreadLocalData(originThread()));
317
318 removeAllEventListeners();
319
320 Vector<IDBResourceIdentifier> transactionIdentifiers;
321 transactionIdentifiers.reserveInitialCapacity(m_activeTransactions.size());
322
323 for (auto& id : m_activeTransactions.keys())
324 transactionIdentifiers.uncheckedAppend(id);
325
326 for (auto& id : transactionIdentifiers) {
327 IDBTransaction* transaction = m_activeTransactions.get(id);
328 if (transaction)
329 transaction->stop();
330 }
331
332 close();
333}

335Ref<IDBTransaction> IDBDatabase::startVersionChangeTransaction(const IDBTransactionInfo& info, IDBOpenDBRequest& request)
336{
337 LOG(IndexedDB, "IDBDatabase::startVersionChangeTransaction %s", info.identifier().loggingString().utf8().data());
338
339 ASSERT(canCurrentThreadAccessThreadLocalData(originThread()));
340 ASSERT(!m_versionChangeTransaction);
341 ASSERT(info.mode() == IDBTransactionMode::Versionchange);
342 ASSERT(!m_closePending);
343 ASSERT(scriptExecutionContext());
344
345 Ref<IDBTransaction> transaction = IDBTransaction::create(*this, info, request);
346 m_versionChangeTransaction = &transaction.get();
347
348 m_activeTransactions.set(transaction->info().identifier(), &transaction.get());
349
350 return transaction;
351}

353void IDBDatabase::didStartTransaction(IDBTransaction& transaction)
354{
355 LOG(IndexedDB, "IDBDatabase::didStartTransaction %s", transaction.info().identifier().loggingString().utf8().data());
356 ASSERT(!m_versionChangeTransaction);
357 ASSERT(canCurrentThreadAccessThreadLocalData(originThread()));
358
359 // It is possible for the client to have aborted a transaction before the server replies back that it has started.
360 if (m_abortingTransactions.contains(transaction.info().identifier()))
361 return;
362
363 m_activeTransactions.set(transaction.info().identifier(), &transaction);
364}

377void IDBDatabase::didCommitTransaction(IDBTransaction& transaction)

389void IDBDatabase::willAbortTransaction(IDBTransaction& transaction)
390{
391 LOG(IndexedDB, "IDBDatabase::willAbortTransaction %s", transaction.info().identifier().loggingString().utf8().data());
392
393 ASSERT(canCurrentThreadAccessThreadLocalData(originThread()));
394
395 auto refTransaction = m_activeTransactions.take(transaction.info().identifier());
396 if (!refTransaction)
397 refTransaction = m_committingTransactions.take(transaction.info().identifier());
398
399 ASSERT(refTransaction);
400 m_abortingTransactions.set(transaction.info().identifier(), WTFMove(refTransaction));
401
402 if (transaction.isVersionChange()) {
403 ASSERT(transaction.originalDatabaseInfo());
404 m_info = *transaction.originalDatabaseInfo();
405 m_closePending = true;
406 }
407}

409void IDBDatabase::didAbortTransaction(IDBTransaction& transaction)
410{
411 LOG(IndexedDB, "IDBDatabase::didAbortTransaction %s", transaction.info().identifier().loggingString().utf8().data());
412
413 ASSERT(canCurrentThreadAccessThreadLocalData(originThread()));
414
415 if (transaction.isVersionChange()) {
416 ASSERT(transaction.originalDatabaseInfo());
417 ASSERT(m_info.version() == transaction.originalDatabaseInfo()->version());
418 m_closePending = true;
419 maybeCloseInServer();
420 }
421
422 didCommitOrAbortTransaction(transaction);
423}

425void IDBDatabase::didCommitOrAbortTransaction(IDBTransaction& transaction)
426{
427 LOG(IndexedDB, "IDBDatabase::didCommitOrAbortTransaction %s", transaction.info().identifier().loggingString().utf8().data());
428
429 ASSERT(canCurrentThreadAccessThreadLocalData(originThread()));
430
431 if (m_versionChangeTransaction == &transaction)
432 m_versionChangeTransaction = nullptr;
433
434#ifndef NDBEBUG
435 unsigned count = 0;
436 if (m_activeTransactions.contains(transaction.info().identifier()))
437 ++count;
438 if (m_committingTransactions.contains(transaction.info().identifier()))
439 ++count;
440 if (m_abortingTransactions.contains(transaction.info().identifier()))
441 ++count;
442
443 ASSERT(count == 1);
444#endif
445
446 m_activeTransactions.remove(transaction.info().identifier());
447 m_committingTransactions.remove(transaction.info().identifier());
448 m_abortingTransactions.remove(transaction.info().identifier());
449
450 if (m_closePending)
451 maybeCloseInServer();
452}

454void IDBDatabase::fireVersionChangeEvent(const IDBResourceIdentifier& requestIdentifier, uint64_t requestedVersion)
455{
456 uint64_t currentVersion = m_info.version();
457 LOG(IndexedDB, "IDBDatabase::fireVersionChangeEvent - current version %" PRIu64 ", requested version %" PRIu64 ", connection %" PRIu64 " (%p)", currentVersion, requestedVersion, m_databaseConnectionIdentifier, this);
458
459 ASSERT(canCurrentThreadAccessThreadLocalData(originThread()));
460
461 if (!scriptExecutionContext() || m_closePending) {
462 connectionProxy().didFireVersionChangeEvent(m_databaseConnectionIdentifier, requestIdentifier);
463 return;
464 }
465
466 Ref<Event> event = IDBVersionChangeEvent::create(requestIdentifier, currentVersion, requestedVersion, m_eventNames.versionchangeEvent);
467 queueTaskToDispatchEvent(*this, TaskSource::DatabaseAccess, WTFMove(event));
468}

470void IDBDatabase::dispatchEvent(Event& event)
471{
472 LOG(IndexedDB, "IDBDatabase::dispatchEvent (%" PRIu64 ") (%p)", m_databaseConnectionIdentifier, this);
473 ASSERT(canCurrentThreadAccessThreadLocalData(originThread()));
474
475 auto protectedThis = makeRef(*this);
476
477 EventTargetWithInlineData::dispatchEvent(event);
478
479 if (event.isVersionChangeEvent() && event.type() == m_eventNames.versionchangeEvent)
480 m_connectionProxy->didFireVersionChangeEvent(m_databaseConnectionIdentifier, downcast<IDBVersionChangeEvent>(event).requestIdentifier());
481}
