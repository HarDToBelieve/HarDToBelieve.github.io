36
37class DOMStringList;
38class IDBObjectStore;
39class IDBOpenDBRequest;
40class IDBResultData;
41class IDBTransaction;

43
44struct EventNames;
45
46class IDBDatabase final : public ThreadSafeRefCounted<IDBDatabase>, public EventTargetWithInlineData, public IDBActiveDOMObject {
47 WTF_MAKE_ISO_ALLOCATED(IDBDatabase);
48public:
49 static Ref<IDBDatabase> create(ScriptExecutionContext&, IDBClient::IDBConnectionProxy&, const IDBResultData&);
50
51 virtual ~IDBDatabase();
52
53 // IDBDatabase IDL
54 const String name() const;
55 uint64_t version() const;
56 Ref<DOMStringList> objectStoreNames() const;
57
58 struct ObjectStoreParameters {
59 std::optional<IDBKeyPath> keyPath;
60 bool autoIncrement;
61 };
62
63 ExceptionOr<Ref<IDBObjectStore>> createObjectStore(const String& name, ObjectStoreParameters&&);
64
65 using StringOrVectorOfStrings = WTF::Variant<String, Vector<String>>;
66 ExceptionOr<Ref<IDBTransaction>> transaction(StringOrVectorOfStrings&& storeNames, IDBTransactionMode);
67 ExceptionOr<void> deleteObjectStore(const String& name);
68 void close();
69
70 void renameObjectStore(IDBObjectStore&, const String& newName);
71 void renameIndex(IDBIndex&, const String& newName);
72
73 // EventTarget
74 EventTargetInterface eventTargetInterface() const final { return IDBDatabaseEventTargetInterfaceType; }
75 ScriptExecutionContext* scriptExecutionContext() const final { return ActiveDOMObject::scriptExecutionContext(); }
76 void refEventTarget() final { ThreadSafeRefCounted<IDBDatabase>::ref(); }
77 void derefEventTarget() final { ThreadSafeRefCounted<IDBDatabase>::deref(); }
78
79 using ThreadSafeRefCounted<IDBDatabase>::ref;
80 using ThreadSafeRefCounted<IDBDatabase>::deref;
81
82 IDBDatabaseInfo& info() { return m_info; }
83 uint64_t databaseConnectionIdentifier() const { return m_databaseConnectionIdentifier; }
84
85 Ref<IDBTransaction> startVersionChangeTransaction(const IDBTransactionInfo&, IDBOpenDBRequest&);
86 void didStartTransaction(IDBTransaction&);
87
88 void willCommitTransaction(IDBTransaction&);
89 void didCommitTransaction(IDBTransaction&);
90 void willAbortTransaction(IDBTransaction&);
91 void didAbortTransaction(IDBTransaction&);
92
93 void fireVersionChangeEvent(const IDBResourceIdentifier& requestIdentifier, uint64_t requestedVersion);
94 void didCloseFromServer(const IDBError&);
95 void connectionToServerLost(const IDBError&);
96
97 IDBClient::IDBConnectionProxy& connectionProxy() { return m_connectionProxy.get(); }
98
99 void didCreateIndexInfo(const IDBIndexInfo&);
100 void didDeleteIndexInfo(const IDBIndexInfo&);
101
102 bool isClosingOrClosed() const { return m_closePending || m_closedInServer; }
103
104 void dispatchEvent(Event&) final;
105
106 void setIsContextSuspended(bool isContextSuspended) { m_isContextSuspended = isContextSuspended; }
107 bool isContextSuspended() const { return m_isContextSuspended; }
108
109private:
110 IDBDatabase(ScriptExecutionContext&, IDBClient::IDBConnectionProxy&, const IDBResultData&);
111
112 void didCommitOrAbortTransaction(IDBTransaction&);
113
114 // ActiveDOMObject.
115 bool virtualHasPendingActivity() const final;
116 const char* activeDOMObjectName() const final;
117 void stop() final;
118
119 void maybeCloseInServer();
120
121 Ref<IDBClient::IDBConnectionProxy> m_connectionProxy;
122 IDBDatabaseInfo m_info;
123 uint64_t m_databaseConnectionIdentifier { 0 };
124
125 bool m_closePending { false };
126 bool m_closedInServer { false };
127
128 RefPtr<IDBTransaction> m_versionChangeTransaction;
129 HashMap<IDBResourceIdentifier, RefPtr<IDBTransaction>> m_activeTransactions;
130 HashMap<IDBResourceIdentifier, RefPtr<IDBTransaction>> m_committingTransactions;
131 HashMap<IDBResourceIdentifier, RefPtr<IDBTransaction>> m_abortingTransactions;
132
133 const EventNames& m_eventNames; // Need to cache this so we can use it from GC threads.
134
135 bool m_isContextSuspended { false };
136};
