36IDBDatabaseIdentifier::IDBDatabaseIdentifier(const String& databaseName, SecurityOriginData&& openingOrigin, SecurityOriginData&& mainFrameOrigin, bool isTransient)
37 : m_databaseName(databaseName)
38 , m_origin { WTFMove(openingOrigin), WTFMove(mainFrameOrigin) }
39 , m_isTransient(isTransient)
40{
41 // The empty string is a valid database name, but a null string is not.
42 ASSERT(!databaseName.isNull());
43}
44
45IDBDatabaseIdentifier IDBDatabaseIdentifier::isolatedCopy() const
46{
47 IDBDatabaseIdentifier identifier;
48
49 identifier.m_databaseName = m_databaseName.isolatedCopy();
50 identifier.m_origin = m_origin.isolatedCopy();
51 identifier.m_isTransient = m_isTransient;
52
53 return identifier;
54}
55
56String IDBDatabaseIdentifier::databaseDirectoryRelativeToRoot(const String& rootDirectory, const String& versionString) const
57{
58 return databaseDirectoryRelativeToRoot(m_origin.topOrigin, m_origin.clientOrigin, rootDirectory, versionString);
59}
60
61String IDBDatabaseIdentifier::databaseDirectoryRelativeToRoot(const SecurityOriginData& topLevelOrigin, const SecurityOriginData& openingOrigin, const String& rootDirectory, const String& versionString)
62{
63 String versionDirectory = FileSystem::pathByAppendingComponent(rootDirectory, versionString);
64 String mainFrameDirectory = FileSystem::pathByAppendingComponent(versionDirectory, topLevelOrigin.databaseIdentifier());
65
66 // If the opening origin and main frame origins are the same, there is no partitioning.
67 if (openingOrigin == topLevelOrigin)
68 return mainFrameDirectory;
69
70 return FileSystem::pathByAppendingComponent(mainFrameDirectory, openingOrigin.databaseIdentifier());
71}
