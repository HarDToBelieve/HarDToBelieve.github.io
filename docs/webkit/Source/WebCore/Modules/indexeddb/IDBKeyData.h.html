120 unsigned hash() const
121 {
122 Vector<unsigned> hashCodes;
123 hashCodes.append(static_cast<unsigned>(m_type));
124 hashCodes.append(m_isNull ? 1 : 0);
125 hashCodes.append(m_isDeletedValue ? 1 : 0);
126 switch (m_type) {
127 case IndexedDB::KeyType::Invalid:
128 case IndexedDB::KeyType::Max:
129 case IndexedDB::KeyType::Min:
130 break;
131 case IndexedDB::KeyType::Number:
132 case IndexedDB::KeyType::Date:
133 hashCodes.append(StringHasher::hashMemory<sizeof(double)>(&WTF::get<double>(m_value)));
134 break;
135 case IndexedDB::KeyType::String:
136 hashCodes.append(StringHash::hash(WTF::get<String>(m_value)));
137 break;
138 case IndexedDB::KeyType::Binary: {
139 auto* data = WTF::get<ThreadSafeDataBuffer>(m_value).data();
140 if (!data)
141 hashCodes.append(0);
142 else
143 hashCodes.append(StringHasher::hashMemory(data->data(), data->size()));
144 break;
145 }
146 case IndexedDB::KeyType::Array:
147 for (auto& key : WTF::get<Vector<IDBKeyData>>(m_value))
148 hashCodes.append(key.hash());
149 break;
150 }
151
152 return StringHasher::hashMemory(hashCodes.data(), hashCodes.size() * sizeof(unsigned));
153 }

223template<class Encoder>
224void IDBKeyData::encode(Encoder& encoder) const
225{
226 encoder << m_isNull;
227 if (m_isNull)
228 return;
229
230 encoder << m_type;
231
232 switch (m_type) {
233 case IndexedDB::KeyType::Invalid:
234 case IndexedDB::KeyType::Max:
235 case IndexedDB::KeyType::Min:
236 break;
237 case IndexedDB::KeyType::Array:
238 encoder << WTF::get<Vector<IDBKeyData>>(m_value);
239 break;
240 case IndexedDB::KeyType::Binary:
241 encoder << WTF::get<ThreadSafeDataBuffer>(m_value);
242 break;
243 case IndexedDB::KeyType::String:
244 encoder << WTF::get<String>(m_value);
245 break;
246 case IndexedDB::KeyType::Date:
247 case IndexedDB::KeyType::Number:
248 encoder << WTF::get<double>(m_value);
249 break;
250 }
251}

253template<class Decoder>
254std::optional<IDBKeyData> IDBKeyData::decode(Decoder& decoder)
255{
256 IDBKeyData keyData;
257 if (!decoder.decode(keyData.m_isNull))
258 return std::nullopt;
259
260 if (keyData.m_isNull)
261 return keyData;
262
263 if (!decoder.decode(keyData.m_type))
264 return std::nullopt;
265
266 switch (keyData.m_type) {
267 case IndexedDB::KeyType::Invalid:
268 case IndexedDB::KeyType::Max:
269 case IndexedDB::KeyType::Min:
270 break;
271 case IndexedDB::KeyType::Array:
272 keyData.m_value = Vector<IDBKeyData>();
273 if (!decoder.decode(WTF::get<Vector<IDBKeyData>>(keyData.m_value)))
274 return std::nullopt;
275 break;
276 case IndexedDB::KeyType::Binary:
277 keyData.m_value = ThreadSafeDataBuffer();
278 if (!decoder.decode(WTF::get<ThreadSafeDataBuffer>(keyData.m_value)))
279 return std::nullopt;
280 break;
281 case IndexedDB::KeyType::String:
282 keyData.m_value = String();
283 if (!decoder.decode(WTF::get<String>(keyData.m_value)))
284 return std::nullopt;
285 break;
286 case IndexedDB::KeyType::Date:
287 case IndexedDB::KeyType::Number:
288 keyData.m_value = 0.0;
289 if (!decoder.decode(WTF::get<double>(keyData.m_value)))
290 return std::nullopt;
291 break;
292 }
293
294 return keyData;
295}
