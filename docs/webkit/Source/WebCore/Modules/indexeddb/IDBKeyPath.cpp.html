36class IDBKeyPathLexer {
37public:
38 enum TokenType {
39 TokenIdentifier,
40 TokenDot,
41 TokenEnd,
42 TokenError
43 };
44
45 explicit IDBKeyPathLexer(const String& s)
46 : m_string(s)
47 , m_remainingText(s)
48 , m_currentTokenType(TokenError)
49 {
50 }
51
52 TokenType currentTokenType() const { return m_currentTokenType; }
53
54 TokenType nextTokenType()
55 {
56 m_currentTokenType = lex(m_currentElement);
57 return m_currentTokenType;
58 }
59
60 const String& currentElement() { return m_currentElement; }
61
62private:
63 TokenType lex(String&);
64 TokenType lexIdentifier(String&);
65
66 String m_currentElement;
67 const String m_string;
68 StringView m_remainingText;
69 TokenType m_currentTokenType;
70};
71
72IDBKeyPathLexer::TokenType IDBKeyPathLexer::lex(String& element)
73{
74 if (m_remainingText.isEmpty())
75 return TokenEnd;
76
77 if (m_remainingText[0] == '.') {
78 m_remainingText = m_remainingText.substring(1);
79 return TokenDot;
80 }
81
82 return lexIdentifier(element);
83}

85namespace {

87// The following correspond to grammar in ECMA-262.
88const uint32_t unicodeLetter = U_GC_L_MASK | U_GC_NL_MASK;
89const uint32_t unicodeCombiningMark = U_GC_MN_MASK | U_GC_MC_MASK;
90const uint32_t unicodeDigit = U_GC_ND_MASK;
91const uint32_t unicodeConnectorPunctuation = U_GC_PC_MASK;
92const UChar ZWNJ = 0x200C;
93const UChar ZWJ = 0x200D;
94
95static inline bool isIdentifierStartCharacter(UChar c)
96{
97 return (U_GET_GC_MASK(c) & unicodeLetter) || (c == '$') || (c == '_');
98}
99
100static inline bool isIdentifierCharacter(UChar c)
101{
102 return (U_GET_GC_MASK(c) & (unicodeLetter | unicodeCombiningMark | unicodeDigit | unicodeConnectorPunctuation)) || (c == '$') || (c == '_') || (c == ZWNJ) || (c == ZWJ);
103}
104
105} // namespace

107IDBKeyPathLexer::TokenType IDBKeyPathLexer::lexIdentifier(String& element)
108{
109 StringView start = m_remainingText;
110 if (!m_remainingText.isEmpty() && isIdentifierStartCharacter(m_remainingText[0]))
111 m_remainingText = m_remainingText.substring(1);
112 else
113 return TokenError;
114
115 while (!m_remainingText.isEmpty() && isIdentifierCharacter(m_remainingText[0]))
116 m_remainingText = m_remainingText.substring(1);
117
118 element = start.substring(0, start.length() - m_remainingText.length()).toString();
119 return TokenIdentifier;
120}

122static bool IDBIsValidKeyPath(const String& keyPath)
123{
124 IDBKeyPathParseError error;
125 Vector<String> keyPathElements;
126 IDBParseKeyPath(keyPath, keyPathElements, error);
127 return error == IDBKeyPathParseError::None;
128}
129
130void IDBParseKeyPath(const String& keyPath, Vector<String>& elements, IDBKeyPathParseError& error)
131{
132 // IDBKeyPath ::= EMPTY_STRING | identifier ('.' identifier)*
133 // The basic state machine is:
134 // Start => {Identifier, End}
135 // Identifier => {Dot, End}
136 // Dot => {Identifier}
137 // It bails out as soon as it finds an error, but doesn't discard the bits it managed to parse.
138 enum ParserState { Identifier, Dot, End };
139
140 IDBKeyPathLexer lexer(keyPath);
141 IDBKeyPathLexer::TokenType tokenType = lexer.nextTokenType();
142 ParserState state;
143 if (tokenType == IDBKeyPathLexer::TokenIdentifier)
144 state = Identifier;
145 else if (tokenType == IDBKeyPathLexer::TokenEnd)
146 state = End;
147 else {
148 error = IDBKeyPathParseError::Start;
149 return;
150 }
151
152 while (1) {
153 switch (state) {
154 case Identifier : {
155 IDBKeyPathLexer::TokenType tokenType = lexer.currentTokenType();
156 ASSERT(tokenType == IDBKeyPathLexer::TokenIdentifier);
157
158 String element = lexer.currentElement();
159 elements.append(element);
160
161 tokenType = lexer.nextTokenType();
162 if (tokenType == IDBKeyPathLexer::TokenDot)
163 state = Dot;
164 else if (tokenType == IDBKeyPathLexer::TokenEnd)
165 state = End;
166 else {
167 error = IDBKeyPathParseError::Identifier;
168 return;
169 }
170 break;
171 }
172 case Dot: {
173 IDBKeyPathLexer::TokenType tokenType = lexer.currentTokenType();
174 ASSERT(tokenType == IDBKeyPathLexer::TokenDot);
175
176 tokenType = lexer.nextTokenType();
177 if (tokenType == IDBKeyPathLexer::TokenIdentifier)
178 state = Identifier;
179 else {
180 error = IDBKeyPathParseError::Dot;
181 return;
182 }
183 break;
184 }
185 case End: {
186 error = IDBKeyPathParseError::None;
187 return;
188 }
189 }
190 }
191}

193bool isIDBKeyPathValid(const IDBKeyPath& keyPath)
194{
195 auto visitor = WTF::makeVisitor([](const String& string) {
196 return IDBIsValidKeyPath(string);
197 }, [](const Vector<String>& vector) {
198 if (vector.isEmpty())
199 return false;
200 for (auto& key : vector) {
201 if (!IDBIsValidKeyPath(key))
202 return false;
203 }
204 return true;
205 });
206 return WTF::visit(visitor, keyPath);
207}

209IDBKeyPath isolatedCopy(const IDBKeyPath& keyPath)
210{
211 auto visitor = WTF::makeVisitor([](const String& string) -> IDBKeyPath {
212 return string.isolatedCopy();
213 }, [](const Vector<String>& vector) -> IDBKeyPath {
214 Vector<String> vectorCopy;
215 vectorCopy.reserveInitialCapacity(vector.size());
216 for (auto& string : vector)
217 vectorCopy.uncheckedAppend(string.isolatedCopy());
218 return vectorCopy;
219 });
220
221 return WTF::visit(visitor, keyPath);
222}
