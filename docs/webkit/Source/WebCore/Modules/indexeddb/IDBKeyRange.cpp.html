42Ref<IDBKeyRange> IDBKeyRange::create(RefPtr<IDBKey>&& lower, RefPtr<IDBKey>&& upper, bool isLowerOpen, bool isUpperOpen)
43{
44 return adoptRef(*new IDBKeyRange(WTFMove(lower), WTFMove(upper), isLowerOpen, isUpperOpen));

47Ref<IDBKeyRange> IDBKeyRange::create(RefPtr<IDBKey>&& key)
48{
49 auto upper = key;
50 return create(WTFMove(key), WTFMove(upper), false, false);
51}

63ExceptionOr<Ref<IDBKeyRange>> IDBKeyRange::only(RefPtr<IDBKey>&& key)
64{
65 if (!key || !key->isValid())
66 return Exception { DataError };
67
68 return create(WTFMove(key));
69}

71ExceptionOr<Ref<IDBKeyRange>> IDBKeyRange::only(JSGlobalObject& state, JSValue keyValue)
72{
73 VM& vm = state.vm();
74 auto scope = DECLARE_THROW_SCOPE(vm);
75 auto key = scriptValueToIDBKey(state, keyValue);
76 EXCEPTION_ASSERT_UNUSED(scope, !scope.exception() || !key->isValid());
77 return only(WTFMove(key));
78}

80ExceptionOr<Ref<IDBKeyRange>> IDBKeyRange::lowerBound(JSGlobalObject& state, JSValue boundValue, bool open)
81{
82 VM& vm = state.vm();
83 auto scope = DECLARE_THROW_SCOPE(vm);
84
85 auto bound = scriptValueToIDBKey(state, boundValue);
86 EXCEPTION_ASSERT_UNUSED(scope, !scope.exception() || !bound->isValid());
87 if (!bound->isValid())
88 return Exception { DataError };
89
90 return create(WTFMove(bound), nullptr, open, true);
91}

93ExceptionOr<Ref<IDBKeyRange>> IDBKeyRange::upperBound(JSGlobalObject& state, JSValue boundValue, bool open)
94{
95 VM& vm = state.vm();
96 auto scope = DECLARE_THROW_SCOPE(vm);
97
98 auto bound = scriptValueToIDBKey(state, boundValue);
99 EXCEPTION_ASSERT_UNUSED(scope, !scope.exception() || !bound->isValid());
100 if (!bound->isValid())
101 return Exception { DataError };
102
103 return create(nullptr, WTFMove(bound), true, open);
104}

106ExceptionOr<Ref<IDBKeyRange>> IDBKeyRange::bound(JSGlobalObject& state, JSValue lowerValue, JSValue upperValue, bool lowerOpen, bool upperOpen)
107{
108 VM& vm = state.vm();
109 auto scope = DECLARE_THROW_SCOPE(vm);
110
111 auto lower = scriptValueToIDBKey(state, lowerValue);
112 EXCEPTION_ASSERT_UNUSED(scope, !scope.exception() || !lower->isValid());
113 if (!lower->isValid())
114 return Exception { DataError };
115 auto upper = scriptValueToIDBKey(state, upperValue);
116 EXCEPTION_ASSERT_UNUSED(scope, !scope.exception() || !upper->isValid());
117 if (!upper->isValid())
118 return Exception { DataError };
119 if (upper->isLessThan(lower.get()))
120 return Exception { DataError };
121 if (upper->isEqual(lower.get()) && (lowerOpen || upperOpen))
122 return Exception { DataError };
123
124 return create(WTFMove(lower), WTFMove(upper), lowerOpen, upperOpen);
125}

127bool IDBKeyRange::isOnlyKey() const
128{
129 return m_lower && m_upper && !m_isLowerOpen && !m_isUpperOpen && m_lower->isEqual(*m_upper);
130}
131
132ExceptionOr<bool> IDBKeyRange::includes(JSC::JSGlobalObject& state, JSC::JSValue keyValue)
133{
134 VM& vm = state.vm();
135 auto scope = DECLARE_THROW_SCOPE(vm);
136
137 auto key = scriptValueToIDBKey(state, keyValue);
138 EXCEPTION_ASSERT_UNUSED(scope, !scope.exception() || !key->isValid());
139 if (!key->isValid())
140 return Exception { DataError, "Failed to execute 'includes' on 'IDBKeyRange': The passed-in value is not a valid IndexedDB key." };
141
142 if (m_lower) {
143 int compare = m_lower->compare(key.get());
144
145 if (compare > 0)
146 return false;
147 if (m_isLowerOpen && !compare)
148 return false;
149 }
150
151 if (m_upper) {
152 int compare = m_upper->compare(key.get());
153
154 if (compare < 0)
155 return false;
156 if (m_isUpperOpen && !compare)
157 return false;
158 }
159
160 return true;
161}
