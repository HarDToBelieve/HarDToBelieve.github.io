30#include <wtf/RefCounted.h>
31#include <wtf/RefPtr.h>

33namespace JSC {
34class CallFrame;
35class JSGlobalObject;
36class JSValue;
37}
38
39namespace WebCore {
40
41class IDBKey;
42class ScriptExecutionContext;
43
44class IDBKeyRange final : public ScriptWrappable, public RefCounted<IDBKeyRange> {
45 WTF_MAKE_ISO_ALLOCATED(IDBKeyRange);
46public:
47 static Ref<IDBKeyRange> create(RefPtr<IDBKey>&& lower, RefPtr<IDBKey>&& upper, bool isLowerOpen, bool isUpperOpen);
48 static Ref<IDBKeyRange> create(RefPtr<IDBKey>&&);
49 ~IDBKeyRange();
50
51 IDBKey* lower() const { return m_lower.get(); }
52 IDBKey* upper() const { return m_upper.get(); }
53 bool lowerOpen() const { return m_isLowerOpen; }
54 bool upperOpen() const { return m_isUpperOpen; }
55
56 static ExceptionOr<Ref<IDBKeyRange>> only(RefPtr<IDBKey>&& value);
57 static ExceptionOr<Ref<IDBKeyRange>> only(JSC::JSGlobalObject&, JSC::JSValue key);
58
59 static ExceptionOr<Ref<IDBKeyRange>> lowerBound(JSC::JSGlobalObject&, JSC::JSValue bound, bool open);
60 static ExceptionOr<Ref<IDBKeyRange>> upperBound(JSC::JSGlobalObject&, JSC::JSValue bound, bool open);
61
62 static ExceptionOr<Ref<IDBKeyRange>> bound(JSC::JSGlobalObject&, JSC::JSValue lower, JSC::JSValue upper, bool lowerOpen, bool upperOpen);
63
64 ExceptionOr<bool> includes(JSC::JSGlobalObject&, JSC::JSValue key);
65
66 WEBCORE_EXPORT bool isOnlyKey() const;
67
68private:
69 IDBKeyRange(RefPtr<IDBKey>&& lower, RefPtr<IDBKey>&& upper, bool isLowerOpen, bool isUpperOpen);
70
71 RefPtr<IDBKey> m_lower;
72 RefPtr<IDBKey> m_upper;
73 bool m_isLowerOpen;
74 bool m_isUpperOpen;
75};
