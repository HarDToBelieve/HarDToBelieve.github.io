90ExceptionOr<void> IDBObjectStore::setName(const String& name)
91{
92 ASSERT(canCurrentThreadAccessThreadLocalData(m_transaction.database().originThread()));
93
94 if (m_deleted)
95 return Exception { InvalidStateError, "Failed set property 'name' on 'IDBObjectStore': The object store has been deleted."_s };
96
97 if (!m_transaction.isVersionChange())
98 return Exception { InvalidStateError, "Failed set property 'name' on 'IDBObjectStore': The object store's transaction is not a version change transaction."_s };
99
100 if (!m_transaction.isActive())
101 return Exception { TransactionInactiveError, "Failed set property 'name' on 'IDBObjectStore': The object store's transaction is not active."_s };
102
103 if (m_info.name() == name)
104 return { };
105
106 if (m_transaction.database().info().hasObjectStore(name))
107 return Exception { ConstraintError, makeString("Failed set property 'name' on 'IDBObjectStore': The database already has an object store named '", name, "'.") };
108
109 m_transaction.database().renameObjectStore(*this, name);
110 m_info.rename(name);
111
112 return { };
113}

317ExceptionOr<Ref<IDBRequest>> IDBObjectStore::putOrAdd(JSGlobalObject& state, JSValue value, RefPtr<IDBKey> key, IndexedDB::ObjectStoreOverwriteMode overwriteMode, InlineKeyCheck inlineKeyCheck)
318{
319 VM& vm = state.vm();
320 auto scope = DECLARE_CATCH_SCOPE(vm);
321
322 LOG(IndexedDB, "IDBObjectStore::putOrAdd");
323 ASSERT(canCurrentThreadAccessThreadLocalData(m_transaction.database().originThread()));
324
325 auto context = scriptExecutionContextFromExecState(&state);
326 if (!context)
327 return Exception { UnknownError, "Unable to store record in object store because it does not have a valid script execution context"_s };
328
329 if (m_deleted)
330 return Exception { InvalidStateError, "Failed to store record in an IDBObjectStore: The object store has been deleted."_s };
331
332 if (!m_transaction.isActive())
333 return Exception { TransactionInactiveError, "Failed to store record in an IDBObjectStore: The transaction is inactive or finished."_s };
334
335 if (m_transaction.isReadOnly())
336 return Exception { ReadonlyError, "Failed to store record in an IDBObjectStore: The transaction is read-only."_s };
337
338 auto serializedValue = SerializedScriptValue::create(state, value);
339 if (UNLIKELY(scope.exception()))
340 return Exception { DataCloneError, "Failed to store record in an IDBObjectStore: An object could not be cloned."_s };
341
342 bool privateBrowsingEnabled = false;
343 if (is<Document>(*context)) {
344 if (auto* page = downcast<Document>(*context).page())
345 privateBrowsingEnabled = page->sessionID().isEphemeral();
346 }
347
348 if (serializedValue->hasBlobURLs() && privateBrowsingEnabled) {
349 // https://bugs.webkit.org/show_bug.cgi?id=156347 - Support Blobs in private browsing.
350 return Exception { DataCloneError, "Failed to store record in an IDBObjectStore: BlobURLs are not yet supported."_s };
351 }
352
353 if (key && !key->isValid())
354 return Exception { DataError, "Failed to store record in an IDBObjectStore: The parameter is not a valid key."_s };
355
356 bool usesInlineKeys = !!m_info.keyPath();
357 bool usesKeyGenerator = autoIncrement();
358 if (usesInlineKeys && inlineKeyCheck == InlineKeyCheck::Perform) {
359 if (key)
360 return Exception { DataError, "Failed to store record in an IDBObjectStore: The object store uses in-line keys and the key parameter was provided."_s };
361
362 RefPtr<IDBKey> keyPathKey = maybeCreateIDBKeyFromScriptValueAndKeyPath(state, value, m_info.keyPath().value());
363 if (keyPathKey && !keyPathKey->isValid())
364 return Exception { DataError, "Failed to store record in an IDBObjectStore: Evaluating the object store's key path yielded a value that is not a valid key."_s };
365
366 if (!keyPathKey) {
367 if (!usesKeyGenerator)
368 return Exception { DataError, "Failed to store record in an IDBObjectStore: Evaluating the object store's key path did not yield a value."_s };
369 if (!canInjectIDBKeyIntoScriptValue(state, value, m_info.keyPath().value()))
370 return Exception { DataError };
371 }
372
373 if (keyPathKey) {
374 ASSERT(!key);
375 key = keyPathKey;
376 }
377 } else if (!usesKeyGenerator && !key)
378 return Exception { DataError, "Failed to store record in an IDBObjectStore: The object store uses out-of-line keys and has no key generator and the key parameter was not provided."_s };
379
380 return m_transaction.requestPutOrAdd(state, *this, WTFMove(key), *serializedValue, overwriteMode);
381}

390ExceptionOr<Ref<IDBRequest>> IDBObjectStore::doDelete(JSGlobalObject& execState, WTF::Function<ExceptionOr<RefPtr<IDBKeyRange>>()>&& function)
391{
392 LOG(IndexedDB, "IDBObjectStore::deleteFunction");
393 ASSERT(canCurrentThreadAccessThreadLocalData(m_transaction.database().originThread()));
394
395 // The IDB spec for several IDBObjectStore methods states that transaction related exceptions should fire before
396 // the exception for an object store being deleted.
397 // However, a handful of W3C IDB tests expect the deleted exception even though the transaction inactive exception also applies.
398 // Additionally, Chrome and Edge agree with the test, as does Legacy IDB in WebKit.
399 // Until this is sorted out, we'll agree with the test and the majority share browsers.
400 if (m_deleted)
401 return Exception { InvalidStateError, "Failed to execute 'delete' on 'IDBObjectStore': The object store has been deleted."_s };
402
403 if (!m_transaction.isActive())
404 return Exception { TransactionInactiveError, "Failed to execute 'delete' on 'IDBObjectStore': The transaction is inactive or finished."_s };
405
406 if (m_transaction.isReadOnly())
407 return Exception { ReadonlyError, "Failed to execute 'delete' on 'IDBObjectStore': The transaction is read-only."_s };
408
409 auto keyRange = function();
410 if (keyRange.hasException())
411 return keyRange.releaseException();
412
413 IDBKeyRangeData keyRangeData = keyRange.returnValue().get();
414 if (!keyRangeData.isValid())
415 return Exception { DataError, "Failed to execute 'delete' on 'IDBObjectStore': The parameter is not a valid key range."_s };
416
417 return m_transaction.requestDeleteRecord(execState, *this, keyRangeData);
418}

452ExceptionOr<Ref<IDBIndex>> IDBObjectStore::createIndex(JSGlobalObject&, const String& name, IDBKeyPath&& keyPath, const IndexParameters& parameters)
453{
454 LOG(IndexedDB, "IDBObjectStore::createIndex %s (keyPath: %s, unique: %i, multiEntry: %i)", name.utf8().data(), loggingString(keyPath).utf8().data(), parameters.unique, parameters.multiEntry);
455 ASSERT(canCurrentThreadAccessThreadLocalData(m_transaction.database().originThread()));
456
457 if (!m_transaction.isVersionChange())
458 return Exception { InvalidStateError, "Failed to execute 'createIndex' on 'IDBObjectStore': The database is not running a version change transaction."_s };
459
460 if (m_deleted)
461 return Exception { InvalidStateError, "Failed to execute 'createIndex' on 'IDBObjectStore': The object store has been deleted."_s };
462
463 if (!m_transaction.isActive())
464 return Exception { TransactionInactiveError, "Failed to execute 'createIndex' on 'IDBObjectStore': The transaction is inactive."_s};
465
466 if (m_info.hasIndex(name))
467 return Exception { ConstraintError, "Failed to execute 'createIndex' on 'IDBObjectStore': An index with the specified name already exists."_s };
468
469 if (!isIDBKeyPathValid(keyPath))
470 return Exception { SyntaxError, "Failed to execute 'createIndex' on 'IDBObjectStore': The keyPath argument contains an invalid key path."_s };
471
472 if (name.isNull())
473 return Exception { TypeError };
474
475 if (parameters.multiEntry && WTF::holds_alternative<Vector<String>>(keyPath))
476 return Exception { InvalidAccessError, "Failed to execute 'createIndex' on 'IDBObjectStore': The keyPath argument was an array and the multiEntry option is true."_s };
477
478 // Install the new Index into the ObjectStore's info.
479 IDBIndexInfo info = m_info.createNewIndex(m_transaction.database().info().generateNextIndexID(), name, WTFMove(keyPath), parameters.unique, parameters.multiEntry);
480 m_transaction.database().didCreateIndexInfo(info);
481
482 // Create the actual IDBObjectStore from the transaction, which also schedules the operation server side.
483 auto index = m_transaction.createIndex(*this, info);
484
485 Ref<IDBIndex> referencedIndex { *index };
486
487 Locker locker { m_referencedIndexLock };
488 m_referencedIndexes.set(name, WTFMove(index));
489
490 return referencedIndex;
491}

493ExceptionOr<Ref<IDBIndex>> IDBObjectStore::index(const String& indexName)
494{
495 LOG(IndexedDB, "IDBObjectStore::index");
496 ASSERT(canCurrentThreadAccessThreadLocalData(m_transaction.database().originThread()));
497
498 if (!scriptExecutionContext())
499 return Exception { InvalidStateError }; // FIXME: Is this code tested? Is iteven reachable?
500
501 if (m_deleted)
502 return Exception { InvalidStateError, "Failed to execute 'index' on 'IDBObjectStore': The object store has been deleted."_s };
503
504 if (m_transaction.isFinishedOrFinishing())
505 return Exception { InvalidStateError, "Failed to execute 'index' on 'IDBObjectStore': The transaction is finished."_s };
506
507 Locker locker { m_referencedIndexLock };
508 auto iterator = m_referencedIndexes.find(indexName);
509 if (iterator != m_referencedIndexes.end())
510 return Ref<IDBIndex> { *iterator->value };
511
512 auto* info = m_info.infoForExistingIndex(indexName);
513 if (!info)
514 return Exception { NotFoundError, "Failed to execute 'index' on 'IDBObjectStore': The specified index was not found."_s };
515
516 auto index = makeUnique<IDBIndex>(*scriptExecutionContext(), *info, *this);
517
518 Ref<IDBIndex> referencedIndex { *index };
519
520 m_referencedIndexes.set(indexName, WTFMove(index));
521
522 return referencedIndex;
523}

525ExceptionOr<void> IDBObjectStore::deleteIndex(const String& name)
526{
527 LOG(IndexedDB, "IDBObjectStore::deleteIndex %s", name.utf8().data());
528 ASSERT(canCurrentThreadAccessThreadLocalData(m_transaction.database().originThread()));
529
530 if (m_deleted)
531 return Exception { InvalidStateError, "Failed to execute 'deleteIndex' on 'IDBObjectStore': The object store has been deleted."_s };
532
533 if (!m_transaction.isVersionChange())
534 return Exception { InvalidStateError, "Failed to execute 'deleteIndex' on 'IDBObjectStore': The database is not running a version change transaction."_s };
535
536 if (!m_transaction.isActive())
537 return Exception { TransactionInactiveError, "Failed to execute 'deleteIndex' on 'IDBObjectStore': The transaction is inactive or finished."_s };
538
539 if (!m_info.hasIndex(name))
540 return Exception { NotFoundError, "Failed to execute 'deleteIndex' on 'IDBObjectStore': The specified index was not found."_s };
541
542 auto* info = m_info.infoForExistingIndex(name);
543 ASSERT(info);
544 m_transaction.database().didDeleteIndexInfo(*info);
545
546 m_info.deleteIndex(name);
547
548 {
549 Locker locker { m_referencedIndexLock };
550 if (auto index = m_referencedIndexes.take(name)) {
551 index->markAsDeleted();
552 auto identifier = index->info().identifier();
553 m_deletedIndexes.add(identifier, WTFMove(index));
554 }
555 }
556
557 m_transaction.deleteIndex(m_info.identifier(), name);
558
559 return { };
560}

578ExceptionOr<Ref<IDBRequest>> IDBObjectStore::doCount(JSGlobalObject& execState, const IDBKeyRangeData& range)
579{
580 ASSERT(canCurrentThreadAccessThreadLocalData(m_transaction.database().originThread()));
581
582 // The IDB spec for several IDBObjectStore methods states that transaction related exceptions should fire before
583 // the exception for an object store being deleted.
584 // However, a handful of W3C IDB tests expect the deleted exception even though the transaction inactive exception also applies.
585 // Additionally, Chrome and Edge agree with the test, as does Legacy IDB in WebKit.
586 // Until this is sorted out, we'll agree with the test and the majority share browsers.
587 if (m_deleted)
588 return Exception { InvalidStateError, "Failed to execute 'count' on 'IDBObjectStore': The object store has been deleted."_s };
589
590 if (!m_transaction.isActive())
591 return Exception { TransactionInactiveError, "Failed to execute 'count' on 'IDBObjectStore': The transaction is inactive or finished."_s };
592
593 if (!range.isValid())
594 return Exception { DataError, "Failed to execute 'count' on 'IDBObjectStore': The parameter is not a valid key."_s };
595
596 return m_transaction.requestCount(execState, *this, range);
597}

679void IDBObjectStore::rollbackForVersionChangeAbort()
680{
681 ASSERT(canCurrentThreadAccessThreadLocalData(m_transaction.database().originThread()));
682
683 String currentName = m_info.name();
684 m_info = m_originalInfo;
685
686 auto& databaseInfo = transaction().database().info();
687 auto* objectStoreInfo = databaseInfo.infoForExistingObjectStore(m_info.identifier());
688 if (!objectStoreInfo) {
689 m_info.rename(currentName);
690 m_deleted = true;
691 } else {
692 m_deleted = false;
693
694 HashSet<uint64_t> indexesToRemove;
695 for (auto indexIdentifier : objectStoreInfo->indexMap().keys()) {
696 if (!objectStoreInfo->hasIndex(indexIdentifier))
697 indexesToRemove.add(indexIdentifier);
698 }
699
700 for (auto indexIdentifier : indexesToRemove)
701 m_info.deleteIndex(indexIdentifier);
702 }
703
704 Locker locker { m_referencedIndexLock };
705
706 Vector<uint64_t> identifiersToRemove;
707 Vector<std::unique_ptr<IDBIndex>> indexesToDelete;
708 for (auto& iterator : m_deletedIndexes) {
709 if (m_info.hasIndex(iterator.key)) {
710 auto name = iterator.value->info().name();
711 auto result = m_referencedIndexes.add(name, nullptr);
712 if (!result.isNewEntry)
713 indexesToDelete.append(std::exchange(result.iterator->value, nullptr));
714 result.iterator->value = std::exchange(iterator.value, nullptr);
715 identifiersToRemove.append(iterator.key);
716 }
717 }
718
719 for (auto identifier : identifiersToRemove)
720 m_deletedIndexes.remove(identifier);
721
722 for (auto& index : m_referencedIndexes.values())
723 index->rollbackInfoForVersionChangeAbort();
724
725 for (auto& index : indexesToDelete) {
726 index->rollbackInfoForVersionChangeAbort();
727 auto indexIdentifier = index->info().identifier();
728 m_deletedIndexes.set(indexIdentifier, std::exchange(index, nullptr));
729 }
730}

745void IDBObjectStore::renameReferencedIndex(IDBIndex& index, const String& newName)
746{
747 Locker locker { m_referencedIndexLock };
748 LOG(IndexedDB, "IDBObjectStore::renameReferencedIndex");
749
750 auto* indexInfo = m_info.infoForExistingIndex(index.info().identifier());
751 ASSERT(indexInfo);
752 indexInfo->rename(newName);
753
754 ASSERT(m_referencedIndexes.contains(index.info().name()));
755 ASSERT(!m_referencedIndexes.contains(newName));
756 ASSERT(m_referencedIndexes.get(index.info().name()) == &index);
757
758 m_referencedIndexes.set(newName, m_referencedIndexes.take(index.info().name()));
759}
