69void IDBOpenDBRequest::onError(const IDBResultData& data)
70{
71 ASSERT(canCurrentThreadAccessThreadLocalData(originThread()));
72
73 m_domError = data.error().toDOMException();
74 enqueueEvent(IDBRequestCompletionEvent::create(eventNames().errorEvent, Event::CanBubble::Yes, Event::IsCancelable::Yes, *this));
75}

77void IDBOpenDBRequest::versionChangeTransactionDidFinish()
78{
79 ASSERT(canCurrentThreadAccessThreadLocalData(originThread()));
80
81 // 3.3.7 "versionchange" transaction steps
82 // When the transaction is finished, after firing complete/abort on the transaction, immediately set request's transaction property to null.
83 setShouldExposeTransactionToDOM(false);

86void IDBOpenDBRequest::fireSuccessAfterVersionChangeCommit()
87{
88 LOG(IndexedDB, "IDBOpenDBRequest::fireSuccessAfterVersionChangeCommit() - %s", resourceIdentifier().loggingString().utf8().data());
89
90 ASSERT(canCurrentThreadAccessThreadLocalData(originThread()));
91 ASSERT(hasPendingActivity());
92 m_transaction->addRequest(*this);
93
94 auto event = IDBRequestCompletionEvent::create(eventNames().successEvent, Event::CanBubble::No, Event::IsCancelable::No, *this);
95 m_openDatabaseSuccessEvent = &event.get();
96
97 enqueueEvent(WTFMove(event));
98}

100void IDBOpenDBRequest::fireErrorAfterVersionChangeCompletion()
101{
102 LOG(IndexedDB, "IDBOpenDBRequest::fireErrorAfterVersionChangeCompletion() - %s", resourceIdentifier().loggingString().utf8().data());
103
104 ASSERT(canCurrentThreadAccessThreadLocalData(originThread()));
105 ASSERT(hasPendingActivity());
106
107 IDBError idbError(AbortError);
108 m_domError = DOMException::create(AbortError);
109 setResultToUndefined();
110
111 m_transaction->addRequest(*this);
112 enqueueEvent(IDBRequestCompletionEvent::create(eventNames().errorEvent, Event::CanBubble::Yes, Event::IsCancelable::Yes, *this));
113}

120void IDBOpenDBRequest::dispatchEvent(Event& event)
121{
122 ASSERT(canCurrentThreadAccessThreadLocalData(originThread()));
123
124 auto protectedThis = makeRef(*this);
125
126 IDBRequest::dispatchEvent(event);
127
128 if (m_transaction && m_transaction->isVersionChange() && (event.type() == eventNames().errorEvent || event.type() == eventNames().successEvent))
129 m_transaction->database().connectionProxy().didFinishHandlingVersionChangeTransaction(m_transaction->database().databaseConnectionIdentifier(), *m_transaction);
130}

144void IDBOpenDBRequest::onUpgradeNeeded(const IDBResultData& resultData)
145{
146 ASSERT(canCurrentThreadAccessThreadLocalData(originThread()));
147
148 Ref<IDBDatabase> database = IDBDatabase::create(*scriptExecutionContext(), connectionProxy(), resultData);
149 Ref<IDBTransaction> transaction = database->startVersionChangeTransaction(resultData.transactionInfo(), *this);
150
151 ASSERT(transaction->info().mode() == IDBTransactionMode::Versionchange);
152 ASSERT(transaction->originalDatabaseInfo());
153
154 uint64_t oldVersion = transaction->originalDatabaseInfo()->version();
155 uint64_t newVersion = transaction->info().newVersion();
156
157 LOG(IndexedDB, "IDBOpenDBRequest::onUpgradeNeeded() - current version is %" PRIu64 ", new is %" PRIu64, oldVersion, newVersion);
158
159 setResult(WTFMove(database));
160 setReadyState(ReadyState::Done);
161 m_transaction = WTFMove(transaction);
162 m_transaction->addRequest(*this);
163
164 enqueueEvent(IDBVersionChangeEvent::create(oldVersion, newVersion, eventNames().upgradeneededEvent));
165}

167void IDBOpenDBRequest::onDeleteDatabaseSuccess(const IDBResultData& resultData)
168{
169 ASSERT(canCurrentThreadAccessThreadLocalData(originThread()));
170
171 uint64_t oldVersion = resultData.databaseInfo().version();
172
173 LOG(IndexedDB, "IDBOpenDBRequest::onDeleteDatabaseSuccess() - current version is %" PRIu64, oldVersion);
174
175 setReadyState(ReadyState::Done);
176 setResultToUndefined();
177
178 enqueueEvent(IDBVersionChangeEvent::create(oldVersion, 0, eventNames().successEvent));
179}

181void IDBOpenDBRequest::requestCompleted(const IDBResultData& data)
182{
183 LOG(IndexedDB, "IDBOpenDBRequest::requestCompleted");
184
185 ASSERT(canCurrentThreadAccessThreadLocalData(originThread()));
186
187 m_isBlocked = false;
188
189 // If an Open request was completed after the page has navigated, leaving this request
190 // with a stopped script execution context, we need to message back to the server so it
191 // doesn't hang waiting on a database connection or transaction that will never exist.
192 if (isContextStopped()) {
193 switch (data.type()) {
194 case IDBResultType::OpenDatabaseSuccess:
195 connectionProxy().abortOpenAndUpgradeNeeded(data.databaseConnectionIdentifier(), IDBResourceIdentifier::emptyValue());
196 break;
197 case IDBResultType::OpenDatabaseUpgradeNeeded:
198 connectionProxy().abortOpenAndUpgradeNeeded(data.databaseConnectionIdentifier(), data.transactionInfo().identifier());
199 break;
200 default:
201 break;
202 }
203
204 return;
205 }
206
207 switch (data.type()) {
208 case IDBResultType::Error:
209 onError(data);
210 break;
211 case IDBResultType::OpenDatabaseSuccess:
212 onSuccess(data);
213 break;
214 case IDBResultType::OpenDatabaseUpgradeNeeded:
215 onUpgradeNeeded(data);
216 break;
217 case IDBResultType::DeleteDatabaseSuccess:
218 onDeleteDatabaseSuccess(data);
219 break;
220 default:
221 RELEASE_ASSERT_NOT_REACHED();
222 }
223}

225void IDBOpenDBRequest::requestBlocked(uint64_t oldVersion, uint64_t newVersion)
226{
227 ASSERT(canCurrentThreadAccessThreadLocalData(originThread()));
228 ASSERT(!m_isBlocked);
229
230 m_isBlocked = true;
231
232 LOG(IndexedDB, "IDBOpenDBRequest::requestBlocked");
233 enqueueEvent(IDBVersionChangeEvent::create(oldVersion, newVersion, eventNames().blockedEvent));
234}

236void IDBOpenDBRequest::setIsContextSuspended(bool isContextSuspended)
237{
238 m_isContextSuspended = isContextSuspended;
239
240 // If this request is blocked, it means this request is being processed on the server.
241 // The client needs to actively stop the request so it doesn't blocks the processing of subsequent requests.
242 if (m_isBlocked) {
243 IDBRequestData requestData(connectionProxy(), *this);
244 connectionProxy().openDBRequestCancelled(requestData);
245 auto result = IDBResultData::error(requestData.requestIdentifier(), IDBError { UnknownError, "Blocked open request on cached page is aborted to unblock other requests"_s });
246 requestCompleted(result);
247 }
