290void IDBRequest::dispatchEvent(Event& event)
291{
292 LOG(IndexedDB, "IDBRequest::dispatchEvent - %s (%p)", event.type().string().utf8().data(), this);
293
294 ASSERT(canCurrentThreadAccessThreadLocalData(originThread()));
295 ASSERT(m_hasPendingActivity);
296 ASSERT(!isContextStopped());
297
298 auto protectedThis = makeRef(*this);
299 m_dispatchingEvent = true;
300
301 if (event.type() != eventNames().blockedEvent)
302 m_readyState = ReadyState::Done;
303
304 Vector<EventTarget*> targets { this };
305
306 if (&event == m_openDatabaseSuccessEvent)
307 m_openDatabaseSuccessEvent = nullptr;
308 else if (m_transaction && !m_transaction->didDispatchAbortOrCommit())
309 targets = { this, m_transaction.get(), &m_transaction->database() };
310
311 m_hasPendingActivity = false;
312
313 {
314 TransactionActivator activator(m_transaction.get());
315 EventDispatcher::dispatchEvent(targets, event);
316 }
317
318 // Dispatching the event might have set the pending activity flag back to true, suggesting the request will be reused.
319 // We might also re-use the request if this event was the upgradeneeded event for an IDBOpenDBRequest.
320 if (!m_hasPendingActivity)
321 m_hasPendingActivity = isOpenDBRequest() && (event.type() == eventNames().upgradeneededEvent || event.type() == eventNames().blockedEvent);
322
323 m_dispatchingEvent = false;
324 if (!m_transaction)
325 return;
326
327 if (m_hasUncaughtException)
328 m_transaction->abortDueToFailedRequest(DOMException::create(AbortError, "IDBTransaction will abort due to uncaught exception in an event handler"_s));
329 else if (!event.defaultPrevented() && event.type() == eventNames().errorEvent && !m_transaction->isFinishedOrFinishing()) {
330 ASSERT(m_domError);
331 m_transaction->abortDueToFailedRequest(*m_domError);
332 }
333
334 m_transaction->finishedDispatchEventForRequest(*this);
335
336 // The request should only remain in the transaction's request list if it represents a pending cursor operation, or this is an open request that was blocked.
337 if (!m_pendingCursor && event.type() != eventNames().blockedEvent)
338 m_transaction->removeRequest(*this);
339}

341void IDBRequest::uncaughtExceptionInEventHandler()
342{
343 LOG(IndexedDB, "IDBRequest::uncaughtExceptionInEventHandler");
344
345 ASSERT(canCurrentThreadAccessThreadLocalData(originThread()));
346
347 if (m_dispatchingEvent) {
348 ASSERT(!m_hasUncaughtException);
349 m_hasUncaughtException = true;
350 return;
351 }
352 if (m_transaction && m_idbError.code() != AbortError)
353 m_transaction->abortDueToFailedRequest(DOMException::create(AbortError, "IDBTransaction will abort due to uncaught exception in an event handler"_s));
354}

452void IDBRequest::willIterateCursor(IDBCursor& cursor)
453{
454 ASSERT(canCurrentThreadAccessThreadLocalData(originThread()));
455 ASSERT(isDone());
456 ASSERT(scriptExecutionContext());
457 ASSERT(m_transaction);
458 ASSERT(!m_pendingCursor);
459 ASSERT(&cursor == resultCursor());
460
461 m_pendingCursor = &cursor;
462 m_hasPendingActivity = true;
463 m_result = NullResultType::Empty;
464
465 auto* context = scriptExecutionContext();
466 if (!context)
467 return;
468
469 VM& vm = context->vm();
470 JSLockHolder lock(vm);
471
472 if (m_resultWrapper)
473 m_cursorWrapper = m_resultWrapper;
474 m_resultWrapper = { };
475 m_readyState = ReadyState::Pending;
476 m_domError = nullptr;
477 m_idbError = IDBError { };
478}

480void IDBRequest::didOpenOrIterateCursor(const IDBResultData& resultData)
481{
482 ASSERT(canCurrentThreadAccessThreadLocalData(originThread()));
483 ASSERT(m_pendingCursor);
484
485 auto* context = scriptExecutionContext();
486 if (!context)
487 return;
488
489 VM& vm = context->vm();
490 JSLockHolder lock(vm);
491
492 m_result = NullResultType::Empty;
493 m_resultWrapper = { };
494
495 if (resultData.type() == IDBResultType::IterateCursorSuccess || resultData.type() == IDBResultType::OpenCursorSuccess) {
496 if (m_pendingCursor->setGetResult(*this, resultData.getResult(), m_currentTransactionOperationID) && m_cursorWrapper)
497 m_resultWrapper = m_cursorWrapper;
498 if (resultData.getResult().isDefined())
499 m_result = m_pendingCursor;
500 }
501
502 m_pendingCursor = nullptr;
503
504 completeRequestAndDispatchEvent(resultData);
505}

507void IDBRequest::completeRequestAndDispatchEvent(const IDBResultData& resultData)
508{
509 ASSERT(canCurrentThreadAccessThreadLocalData(originThread()));
510
511 m_readyState = ReadyState::Done;
512
513 m_idbError = resultData.error();
514 if (!m_idbError.isNull())
515 onError();
516 else
517 onSuccess();
518}

520void IDBRequest::onError()
521{
522 LOG(IndexedDB, "IDBRequest::onError");
523
524 ASSERT(canCurrentThreadAccessThreadLocalData(originThread()));
525 ASSERT(!m_idbError.isNull());
526
527 m_domError = m_idbError.toDOMException();
528 enqueueEvent(Event::create(eventNames().errorEvent, Event::CanBubble::Yes, Event::IsCancelable::Yes));
529}
