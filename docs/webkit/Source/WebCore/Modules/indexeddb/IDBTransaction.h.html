271class TransactionActivator {
272 WTF_MAKE_NONCOPYABLE(TransactionActivator);
273public:
274 TransactionActivator(IDBTransaction* transaction)
275 : m_transaction(transaction)
276 {
277 if (m_transaction)
278 m_transaction->activate();
279 }
280
281 ~TransactionActivator()
282 {
283 if (m_transaction)
284 m_transaction->deactivate();
285 }
286

288 IDBTransaction* m_transaction;
