31namespace WebCore {
32
33WTF_MAKE_ISO_ALLOCATED_IMPL(IDBVersionChangeEvent);
34
35IDBVersionChangeEvent::IDBVersionChangeEvent(const IDBResourceIdentifier& requestIdentifier, uint64_t oldVersion, uint64_t newVersion, const AtomString& name)
36 : Event(name, CanBubble::No, IsCancelable::No)
37 , m_requestIdentifier(requestIdentifier)
38 , m_oldVersion(oldVersion)
39{
40 if (newVersion)
41 m_newVersion = newVersion;
42 else
43 m_newVersion = std::nullopt;
44}
45
46IDBVersionChangeEvent::IDBVersionChangeEvent(const AtomString& name, const Init& init, IsTrusted isTrusted)
47 : Event(name, init, isTrusted)
48 , m_oldVersion(init.oldVersion)
49 , m_newVersion(init.newVersion)
50{
51}
52
53EventInterface IDBVersionChangeEvent::eventInterface() const
54{
55 return IDBVersionChangeEventInterfaceType;
56}
