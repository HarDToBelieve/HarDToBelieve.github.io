26#pragma once
27
28#include "Event.h"
29#include "IDBResourceIdentifier.h"
30
31namespace WebCore {
32
33class IDBVersionChangeEvent final : public Event {
34 WTF_MAKE_ISO_ALLOCATED(IDBVersionChangeEvent);
35public:
36 static Ref<IDBVersionChangeEvent> create(uint64_t oldVersion, uint64_t newVersion, const AtomString& eventType)
37 {
38 return adoptRef(*new IDBVersionChangeEvent(IDBResourceIdentifier::emptyValue(), oldVersion, newVersion, eventType));
39 }
40
41 static Ref<IDBVersionChangeEvent> create(const IDBResourceIdentifier& requestIdentifier, uint64_t oldVersion, uint64_t newVersion, const AtomString& eventType)
42 {
43 return adoptRef(*new IDBVersionChangeEvent(requestIdentifier, oldVersion, newVersion, eventType));
44 }
45
46 struct Init : EventInit {
47 uint64_t oldVersion { 0 };
48 std::optional<uint64_t> newVersion;
49 };
50
51 static Ref<IDBVersionChangeEvent> create(const AtomString& type, const Init& initializer, IsTrusted isTrusted = IsTrusted::No)
52 {
53 return adoptRef(*new IDBVersionChangeEvent(type, initializer, isTrusted));
54 }
55
56 const IDBResourceIdentifier& requestIdentifier() const { return m_requestIdentifier; }
57
58 bool isVersionChangeEvent() const final { return true; }
59
60 uint64_t oldVersion() const { return m_oldVersion; }
61 std::optional<uint64_t> newVersion() const { return m_newVersion; }
62
63private:
64 IDBVersionChangeEvent(const IDBResourceIdentifier& requestIdentifier, uint64_t oldVersion, uint64_t newVersion, const AtomString& eventType);
65 IDBVersionChangeEvent(const AtomString&, const Init&, IsTrusted);
66
67 EventInterface eventInterface() const;
68
69 IDBResourceIdentifier m_requestIdentifier;
70 uint64_t m_oldVersion;
71 std::optional<uint64_t> m_newVersion;
72};
73
74} // namespace WebCore
75
76SPECIALIZE_TYPE_TRAITS_BEGIN(WebCore::IDBVersionChangeEvent)
77 static bool isType(const WebCore::Event& event) { return event.isVersionChangeEvent(); }
78SPECIALIZE_TYPE_TRAITS_END()
