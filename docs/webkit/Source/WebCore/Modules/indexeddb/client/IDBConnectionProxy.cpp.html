66Ref<IDBOpenDBRequest> IDBConnectionProxy::openDatabase(ScriptExecutionContext& context, const IDBDatabaseIdentifier& databaseIdentifier, uint64_t version)
67{
68 RefPtr<IDBOpenDBRequest> request;
69 {
70 Locker locker { m_openDBRequestMapLock };
71
72 request = IDBOpenDBRequest::createOpenRequest(context, *this, databaseIdentifier, version);
73 ASSERT(!m_openDBRequestMap.contains(request->resourceIdentifier()));
74 m_openDBRequestMap.set(request->resourceIdentifier(), request.get());
75 }
76
77 callConnectionOnMainThread(&IDBConnectionToServer::openDatabase, IDBRequestData(*this, *request));
78
79 return request.releaseNonNull();

108void IDBConnectionProxy::completeOpenDBRequest(const IDBResultData& resultData)
109{
110 ASSERT(isMainThread());
111
112 RefPtr<IDBOpenDBRequest> request;
113 {
114 Locker locker { m_openDBRequestMapLock };
115 request = m_openDBRequestMap.take(resultData.requestIdentifier());
116 }
117
118 if (!request)
119 return;
120
121 if (request->isContextSuspended()) {
122 switch (resultData.type()) {
123 case IDBResultType::OpenDatabaseUpgradeNeeded: {
124 abortOpenAndUpgradeNeeded(resultData.databaseConnectionIdentifier(), resultData.transactionInfo().identifier());
125 auto result = IDBResultData::error(resultData.requestIdentifier(), IDBError { UnknownError, "Version change transaction on cached page is aborted to unblock other connections"_s });
126 request->performCallbackOnOriginThread(*request, &IDBOpenDBRequest::requestCompleted, result);
127 return;
128 }
129 default:
130 break;
131 }
132 }
133
134 request->performCallbackOnOriginThread(*request, &IDBOpenDBRequest::requestCompleted, resultData);
135}

277void IDBConnectionProxy::fireVersionChangeEvent(uint64_t databaseConnectionIdentifier, const IDBResourceIdentifier& requestIdentifier, uint64_t requestedVersion)
278{
279 RefPtr<IDBDatabase> database;
280 {
281 Locker locker { m_databaseConnectionMapLock };
282 database = m_databaseConnectionMap.get(databaseConnectionIdentifier);
283 }
284
285 if (!database)
286 return;
287
288 if (database->isContextSuspended()) {
289 didFireVersionChangeEvent(databaseConnectionIdentifier, requestIdentifier, IndexedDB::ConnectionClosedOnBehalfOfServer::Yes);
290 database->performCallbackOnOriginThread(*database, &IDBDatabase::connectionToServerLost, IDBError { UnknownError, "Connection on cached page closed to unblock other connections"_s});
291 return;
292 }
293
294 database->performCallbackOnOriginThread(*database, &IDBDatabase::fireVersionChangeEvent, requestIdentifier, requestedVersion);
295}

435void IDBConnectionProxy::connectionToServerLost(const IDBError& error)
436{
437 Vector<uint64_t> databaseConnectionIdentifiers;
438 {
439 Locker locker { m_databaseConnectionMapLock };
440 databaseConnectionIdentifiers = copyToVector(m_databaseConnectionMap.keys());
441 }
442
443 for (auto connectionIdentifier : databaseConnectionIdentifiers) {
444 RefPtr<IDBDatabase> database;
445 {
446 Locker locker { m_databaseConnectionMapLock };
447 database = m_databaseConnectionMap.get(connectionIdentifier);
448 }
449
450 if (!database)
451 continue;
452
453 database->performCallbackOnOriginThread(*database, &IDBDatabase::connectionToServerLost, error);
454 }
455
456 Vector<IDBResourceIdentifier> openDBRequestIdentifiers;
457 {
458 Locker locker { m_openDBRequestMapLock };
459 openDBRequestIdentifiers = copyToVector(m_openDBRequestMap.keys());
460 }
461
462 for (auto& requestIdentifier : openDBRequestIdentifiers) {
463 RefPtr<IDBOpenDBRequest> request;
464 {
465 Locker locker { m_openDBRequestMapLock };
466 request = m_openDBRequestMap.get(requestIdentifier);
467 }
468
469 if (!request)
470 continue;
471
472 auto result = IDBResultData::error(requestIdentifier, error);
473 request->performCallbackOnOriginThread(*request, &IDBOpenDBRequest::requestCompleted, result);
474 }
475
476 Vector<IDBResourceIdentifier> infoCallbackIdentifiers;
477 {
478 Locker locker { m_databaseInfoMapLock };
479 infoCallbackIdentifiers = copyToVector(m_databaseInfoCallbacks.keys());
480 }
481
482 for (auto& requestIdentifier : infoCallbackIdentifiers)
483 didGetAllDatabaseNamesAndVersions(requestIdentifier, { });
484}

486void IDBConnectionProxy::scheduleMainThreadTasks()
487{
488 Locker locker { m_mainThreadTaskLock };
489 if (m_mainThreadProtector)
490 return;
491
492 m_mainThreadProtector = &m_connectionToServer;
493 callOnMainThread([this] {
494 handleMainThreadTasks();
495 });
496}
497
498void IDBConnectionProxy::handleMainThreadTasks()
499{
500 RefPtr<IDBConnectionToServer> protector;
501 {
502 Locker locker { m_mainThreadTaskLock };
503 ASSERT(m_mainThreadProtector);
504 protector = WTFMove(m_mainThreadProtector);
505 }
506
507 while (auto task = m_mainThreadQueue.tryGetMessage())
508 task->performTask();
509}

511void IDBConnectionProxy::getAllDatabaseNamesAndVersions(ScriptExecutionContext& context, Function<void(std::optional<Vector<IDBDatabaseNameAndVersion>>&&)>&& callback)
512{
513 ClientOrigin origin { context.securityOrigin()->data(), context.topOrigin().data() };
514
515 IDBDatabaseNameAndVersionRequest* request;
516 {
517 Locker locker { m_databaseInfoMapLock };
518 auto newRequest = IDBDatabaseNameAndVersionRequest::create(context, *this, WTFMove(callback));
519 ASSERT(!m_databaseInfoCallbacks.contains(newRequest->resourceIdentifier()));
520 request = newRequest.ptr();
521 m_databaseInfoCallbacks.add(newRequest->resourceIdentifier(), WTFMove(newRequest));
522 }
523
524 callConnectionOnMainThread(&IDBConnectionToServer::getAllDatabaseNamesAndVersions, request->resourceIdentifier(), origin);
525}

573template<typename KeyType, typename ValueType>
574void removeItemsMatchingCurrentThread(HashMap<KeyType, ValueType>& map)
575{
576 // FIXME: Revisit when introducing WebThread aware thread comparison.
577 // https://bugs.webkit.org/show_bug.cgi?id=204345
578 auto& currentThread = Thread::current();
579
580 Vector<KeyType> keys;
581 keys.reserveInitialCapacity(map.size());
582 for (auto& iterator : map) {
583 if (&iterator.value->originThread() == &currentThread)
584 keys.uncheckedAppend(iterator.key);
585 }
586
587 for (auto& key : keys)
588 map.remove(key);
589}

591template<typename KeyType, typename ValueType>
592void setMatchingItemsContextSuspended(ScriptExecutionContext& currentContext, HashMap<KeyType, ValueType>& map, bool isContextSuspended)
593{
594 // FIXME: Revisit when introducing WebThread aware thread comparison.
595 // https://bugs.webkit.org/show_bug.cgi?id=204345
596 auto& currentThread = Thread::current();
597 for (auto& iterator : map) {
598 if (&iterator.value->originThread() != &currentThread)
599 continue;
600
601 auto* context = iterator.value->scriptExecutionContext();
602 if (!context)
603 continue;
604
605 if (context == &currentContext)
606 iterator.value->setIsContextSuspended(isContextSuspended);
607 }
608}

610void IDBConnectionProxy::forgetActivityForCurrentThread()
611{
612 ASSERT(!isMainThread());
613
614 {
615 Locker locker { m_databaseConnectionMapLock };
616 removeItemsMatchingCurrentThread(m_databaseConnectionMap);
617 }
618 {
619 Locker locker { m_openDBRequestMapLock };
620 removeItemsMatchingCurrentThread(m_openDBRequestMap);
621 }
622 {
623 Locker locker { m_transactionMapLock };
624 removeItemsMatchingCurrentThread(m_pendingTransactions);
625 removeItemsMatchingCurrentThread(m_committingTransactions);
626 removeItemsMatchingCurrentThread(m_abortingTransactions);
627 }
628 {
629 Locker locker { m_transactionOperationLock };
630 removeItemsMatchingCurrentThread(m_activeOperations);
631 }
632 {
633 Locker locker { m_databaseInfoMapLock };
634 removeItemsMatchingCurrentThread(m_databaseInfoCallbacks);
635 }
636}
