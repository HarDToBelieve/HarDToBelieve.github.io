70void IDBConnectionToServer::callResultFunctionWithErrorLater(ResultFunction function, const IDBResourceIdentifier& requestIdentifier)
71{
72 callOnMainThread([this, protectedThis = makeRef(*this), function, requestIdentifier]() {
73 (this->*function)(IDBResultData::error(requestIdentifier, IDBError::serverConnectionLostError()));
74 });
75}

77void IDBConnectionToServer::deleteDatabase(const IDBRequestData& request)
78{
79 LOG(IndexedDB, "IDBConnectionToServer::deleteDatabase - %s", request.databaseIdentifier().loggingString().utf8().data());

499void IDBConnectionToServer::getAllDatabaseNamesAndVersions(const IDBResourceIdentifier& requestIdentifier, const ClientOrigin& origin)
500{
501 LOG(IndexedDB, "IDBConnectionToServer::getAllDatabaseNamesAndVersions");
502 ASSERT(isMainThread());
503
504 if (m_serverConnectionIsValid) {
505 m_delegate->getAllDatabaseNamesAndVersions(requestIdentifier, origin);
506 return;
507 }
508
509 callOnMainThread([this, protectedThis = makeRef(*this), requestIdentifier] {
510 didGetAllDatabaseNamesAndVersions(requestIdentifier, { });
511 });
512}
