39TransactionOperation::TransactionOperation(IDBTransaction& transaction, IDBRequest& request)
40 : TransactionOperation(transaction)
41{
42 m_objectStoreIdentifier = request.sourceObjectStoreIdentifier();
43 m_indexIdentifier = request.sourceIndexIdentifier();
44 if (m_indexIdentifier)
45 m_indexRecordType = request.requestedIndexRecordType();
46 if (auto* cursor = request.pendingCursor())
47 m_cursorIdentifier = makeUnique<IDBResourceIdentifier>(cursor->info().identifier());
48
49 request.setTransactionOperationID(m_operationID);
50 m_idbRequest = &request;
51}
