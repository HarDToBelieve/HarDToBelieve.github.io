47class TransactionOperation : public ThreadSafeRefCounted<TransactionOperation> {
48 WTF_MAKE_ISO_ALLOCATED(TransactionOperation);
49 friend IDBRequestData::IDBRequestData(TransactionOperation&);
50public:
51 virtual ~TransactionOperation()
52 {
53 ASSERT(canCurrentThreadAccessThreadLocalData(originThread()));
54 }
55
56 void perform()
57 {
58 ASSERT(canCurrentThreadAccessThreadLocalData(originThread()));
59 ASSERT(m_performFunction);
60 m_performFunction();
61 m_performFunction = { };
62 }
63
64 void transitionToCompleteOnThisThread(const IDBResultData& data)
65 {
66 ASSERT(canCurrentThreadAccessThreadLocalData(originThread()));
67 m_transaction->operationCompletedOnServer(data, *this);
68 }
69
70 void transitionToComplete(const IDBResultData& data, RefPtr<TransactionOperation>&& lastRef)
71 {
72 ASSERT(isMainThread());
73
74 if (canCurrentThreadAccessThreadLocalData(originThread()))
75 transitionToCompleteOnThisThread(data);
76 else {
77 m_transaction->performCallbackOnOriginThread(*this, &TransactionOperation::transitionToCompleteOnThisThread, data);
78 m_transaction->callFunctionOnOriginThread([lastRef = WTFMove(lastRef)]() {
79 });
80 }
81 }
82
83 void doComplete(const IDBResultData& data)
84 {
85 ASSERT(canCurrentThreadAccessThreadLocalData(originThread()));
86
87 if (m_performFunction)
88 m_performFunction = { };
89
90 // Due to race conditions between the server sending an "operation complete" message and the client
91 // forcefully aborting an operation, it's unavoidable that this method might be called twice.
92 // It's okay to handle that gracefully with an early return.
93 if (m_didComplete)
94 return;
95 m_didComplete = true;
96
97 if (m_completeFunction) {
98 m_completeFunction(data);
99 // m_completeFunction should not hold ref to this TransactionOperation after its execution.
100 m_completeFunction = { };
101 }
102 m_transaction->operationCompletedOnClient(*this);
103 }
104
105 const IDBResourceIdentifier& identifier() const { return m_identifier; }
106
107 Thread& originThread() const { return m_originThread.get(); }
108
109 IDBRequest* idbRequest() { return m_idbRequest.get(); }
110
111 bool nextRequestCanGoToServer() const { return m_nextRequestCanGoToServer && m_idbRequest; }
112 void setNextRequestCanGoToServer(bool nextRequestCanGoToServer) { m_nextRequestCanGoToServer = nextRequestCanGoToServer; }
113
114 uint64_t operationID() const { return m_operationID; }
115
116protected:
117 TransactionOperation(IDBTransaction& transaction)
118 : m_transaction(transaction)
119 , m_identifier(transaction.connectionProxy())
120 , m_operationID(transaction.generateOperationID())
121 {
122 }
123
124 TransactionOperation(IDBTransaction&, IDBRequest&);
125
126 Ref<IDBTransaction> m_transaction;
127 IDBResourceIdentifier m_identifier;
128 uint64_t m_objectStoreIdentifier { 0 };
129 uint64_t m_indexIdentifier { 0 };
130 std::unique_ptr<IDBResourceIdentifier> m_cursorIdentifier;
131 IndexedDB::IndexRecordType m_indexRecordType { IndexedDB::IndexRecordType::Key };
132 Function<void()> m_performFunction;
133 Function<void(const IDBResultData&)> m_completeFunction;
134
135private:
136 IDBResourceIdentifier transactionIdentifier() const { return m_transaction->info().identifier(); }
137 uint64_t objectStoreIdentifier() const { return m_objectStoreIdentifier; }
138 uint64_t indexIdentifier() const { return m_indexIdentifier; }
139 IDBResourceIdentifier* cursorIdentifier() const { return m_cursorIdentifier.get(); }
140 IDBTransaction& transaction() { return m_transaction.get(); }
141 IndexedDB::IndexRecordType indexRecordType() const { return m_indexRecordType; }
142
143 Ref<Thread> m_originThread { Thread::current() };
144 RefPtr<IDBRequest> m_idbRequest;
145 bool m_nextRequestCanGoToServer { true };
146 bool m_didComplete { false };
147
148 uint64_t m_operationID { 0 };
149};

151class TransactionOperationImpl final : public TransactionOperation {
152 WTF_MAKE_ISO_ALLOCATED(TransactionOperationImpl);
153public:
154 template<typename... Args> static Ref<TransactionOperationImpl> create(Args&&... args) { return adoptRef(*new TransactionOperationImpl(std::forward<Args>(args)...)); }
155private:
156 TransactionOperationImpl(IDBTransaction& transaction, Function<void(const IDBResultData&)> completeMethod, Function<void(TransactionOperation&)> performMethod)
157 : TransactionOperation(transaction)
158 {
159 ASSERT(performMethod);
160 m_performFunction = [protectedThis = makeRef(*this), performMethod = WTFMove(performMethod)] {
161 performMethod(protectedThis.get());
162 };
163
164 if (completeMethod) {
165 m_completeFunction = [protectedThis = makeRef(*this), completeMethod = WTFMove(completeMethod)] (const IDBResultData& resultData) {
166 completeMethod(resultData);
167 };
168 }
169 }
170
171 TransactionOperationImpl(IDBTransaction& transaction, IDBRequest& request, Function<void(const IDBResultData&)> completeMethod, Function<void(TransactionOperation&)> performMethod)
172 : TransactionOperation(transaction, request)
173 {
174 ASSERT(performMethod);
175 m_performFunction = [protectedThis = makeRef(*this), performMethod = WTFMove(performMethod)] {
176 performMethod(protectedThis.get());
177 };
178
179 if (completeMethod) {
180 m_completeFunction = [protectedThis = makeRef(*this), completeMethod = WTFMove(completeMethod)] (const IDBResultData& resultData) {
181 completeMethod(resultData);
182 };
183 }
184 }
185};
