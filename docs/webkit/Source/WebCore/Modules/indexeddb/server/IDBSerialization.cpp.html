40enum class KeyPathType { Null, String, Array };
41
42RefPtr<SharedBuffer> serializeIDBKeyPath(const std::optional<IDBKeyPath>& keyPath)
43{
44 auto encoder = KeyedEncoder::encoder();
45
46 if (keyPath) {
47 auto visitor = WTF::makeVisitor([&](const String& string) {
48 encoder->encodeEnum("type", KeyPathType::String);
49 encoder->encodeString("string", string);
50 }, [&](const Vector<String>& vector) {
51 encoder->encodeEnum("type", KeyPathType::Array);
52 encoder->encodeObjects("array", vector.begin(), vector.end(), [](WebCore::KeyedEncoder& encoder, const String& string) {
53 encoder.encodeString("string", string);
54 });
55 });
56 WTF::visit(visitor, keyPath.value());
57 } else
58 encoder->encodeEnum("type", KeyPathType::Null);
59
60 return encoder->finishEncoding();
61}

63bool deserializeIDBKeyPath(const uint8_t* data, size_t size, std::optional<IDBKeyPath>& result)
64{
65 if (!data || !size)
66 return false;
67
68 auto decoder = KeyedDecoder::decoder(data, size);
69
70 KeyPathType type;
71 bool succeeded = decoder->decodeEnum("type", type, [](KeyPathType value) {
72 return value == KeyPathType::Null || value == KeyPathType::String || value == KeyPathType::Array;
73 });
74 if (!succeeded)
75 return false;
76
77 switch (type) {
78 case KeyPathType::Null:
79 break;
80 case KeyPathType::String: {
81 String string;
82 if (!decoder->decodeString("string", string))
83 return false;
84 result = IDBKeyPath(WTFMove(string));
85 break;
86 }
87 case KeyPathType::Array: {
88 Vector<String> vector;
89 succeeded = decoder->decodeObjects("array", vector, [](KeyedDecoder& decoder, String& result) {
90 return decoder.decodeString("string", result);
91 });
92 if (!succeeded)
93 return false;
94 result = IDBKeyPath(WTFMove(vector));
95 break;
96 }
97 }
98 return true;
99}

101static bool isLegacySerializedIDBKeyData(const uint8_t* data, size_t size)
102{
103#if USE(CF)
104 UNUSED_PARAM(size);
105
106 // This is the magic character that begins serialized PropertyLists, and tells us whether
107 // the key we're looking at is an old-style key.
108 static const uint8_t legacySerializedKeyVersion = 'b';
109 if (data[0] == legacySerializedKeyVersion)
110 return true;
111#elif USE(GLIB)
112 // KeyedEncoderGLib uses a GVariant dictionary, so check if the given data is a valid GVariant dictionary.
113 GRefPtr<GBytes> bytes = adoptGRef(g_bytes_new(data, size));
114 GRefPtr<GVariant> variant = g_variant_new_from_bytes(G_VARIANT_TYPE("a{sv}"), bytes.get(), FALSE);
115 return g_variant_is_normal_form(variant.get());
116#else
117 UNUSED_PARAM(data);
118 UNUSED_PARAM(size);
119#endif
120 return false;
121}

124/*
125The IDBKeyData serialization format is as follows:
126[1 byte version header][Key Buffer]
127
128The Key Buffer serialization format is as follows:
129[1 byte key type][Type specific data]
130
131Type specific serialization formats are as follows for each of the types:
132Min:
133[0 bytes]
134
135Number:
136[8 bytes representing a double encoded in little endian]
137
138Date:
139[8 bytes representing a double encoded in little endian]
140
141String:
142[4 bytes representing string "length" in little endian]["length" number of 2-byte pairs representing ECMAScript 16-bit code units]
143
144Binary:
145[8 bytes representing the "size" of the binary blob]["size" bytes]
146
147Array:
148[8 bytes representing the "length" of the key array]["length" individual Key Buffer entries]
149
150Max:
151[0 bytes]
152*/
153
154static const uint8_t SIDBKeyVersion = 0x00;
155enum class SIDBKeyType : uint8_t {
156 Min = 0x00,
157 Number = 0x20,
158 Date = 0x40,
159 String = 0x60,
160 Binary = 0x80,
161 Array = 0xA0,
162 Max = 0xFF,
163};
164
165static SIDBKeyType serializedTypeForKeyType(IndexedDB::KeyType type)
166{
167 switch (type) {
168 case IndexedDB::KeyType::Min:
169 return SIDBKeyType::Min;
170 case IndexedDB::KeyType::Number:
171 return SIDBKeyType::Number;
172 case IndexedDB::KeyType::Date:
173 return SIDBKeyType::Date;
174 case IndexedDB::KeyType::String:
175 return SIDBKeyType::String;
176 case IndexedDB::KeyType::Binary:
177 return SIDBKeyType::Binary;
178 case IndexedDB::KeyType::Array:
179 return SIDBKeyType::Array;
180 case IndexedDB::KeyType::Max:
181 return SIDBKeyType::Max;
182 case IndexedDB::KeyType::Invalid:
183 RELEASE_ASSERT_NOT_REACHED();
184 };
185
186 RELEASE_ASSERT_NOT_REACHED();
187}

189#if CPU(BIG_ENDIAN) || CPU(MIDDLE_ENDIAN) || CPU(NEEDS_ALIGNED_ACCESS)
190template <typename T> static void writeLittleEndian(Vector<uint8_t>& buffer, T value)
191{
192 for (unsigned i = 0; i < sizeof(T); i++) {
193 buffer.append(value & 0xFF);
194 value >>= 8;
195 }
196}
197
198template <typename T> static bool readLittleEndian(const uint8_t*& ptr, const uint8_t* end, T& value)
199{
200 if (ptr > end - sizeof(value))
201 return false;
202
203 value = 0;
204 for (size_t i = 0; i < sizeof(T); i++)
205 value += ((T)*ptr++) << (i * 8);
206 return true;
207}
208#else
209template <typename T> static void writeLittleEndian(Vector<uint8_t>& buffer, T value)
210{
211 buffer.append(reinterpret_cast<uint8_t*>(&value), sizeof(value));
212}
213
214template <typename T> static bool readLittleEndian(const uint8_t*& ptr, const uint8_t* end, T& value)
215{
216 if (ptr > end - sizeof(value))
217 return false;
218
219 value = *reinterpret_cast<const T*>(ptr);
220 ptr += sizeof(T);
221
222 return true;
223}
224#endif
225
226static void writeDouble(Vector<uint8_t>& data, double d)
227{
228 writeLittleEndian(data, *reinterpret_cast<uint64_t*>(&d));
229}
230
231static bool readDouble(const uint8_t*& data, const uint8_t* end, double& d)
232{
233 return readLittleEndian(data, end, *reinterpret_cast<uint64_t*>(&d));
234}
235
236static void encodeKey(Vector<uint8_t>& data, const IDBKeyData& key)
237{
238 SIDBKeyType type = serializedTypeForKeyType(key.type());
239 data.append(static_cast<uint8_t>(type));
240
241 switch (type) {
242 case SIDBKeyType::Number:
243 writeDouble(data, key.number());
244 break;
245 case SIDBKeyType::Date:
246 writeDouble(data, key.date());
247 break;
248 case SIDBKeyType::String: {
249 auto string = key.string();
250 uint32_t length = string.length();
251 writeLittleEndian(data, length);
252
253 for (size_t i = 0; i < length; ++i)
254 writeLittleEndian(data, string[i]);
255
256 break;
257 }
258 case SIDBKeyType::Binary: {
259 auto& buffer = key.binary();
260 uint64_t size = buffer.size();
261 writeLittleEndian(data, size);
262
263 auto* bufferData = buffer.data();
264 ASSERT(bufferData || !size);
265 if (bufferData)
266 data.append(bufferData->data(), bufferData->size());
267
268 break;
269 }
270 case SIDBKeyType::Array: {
271 auto& array = key.array();
272 uint64_t size = array.size();
273 writeLittleEndian(data, size);
274 for (auto& key : array)
275 encodeKey(data, key);
276
277 break;
278 }
279 case SIDBKeyType::Min:
280 case SIDBKeyType::Max:
281 break;
282 }
283}

285RefPtr<SharedBuffer> serializeIDBKeyData(const IDBKeyData& key)
286{
287 Vector<uint8_t> data;
288 data.append(SIDBKeyVersion);
289
290 encodeKey(data, key);
291 return SharedBuffer::create(WTFMove(data));
292}

294static WARN_UNUSED_RETURN bool decodeKey(const uint8_t*& data, const uint8_t* end, IDBKeyData& result)
295{
296 if (!data || data >= end)
297 return false;
298
299 SIDBKeyType type = static_cast<SIDBKeyType>(data++[0]);
300 switch (type) {
301 case SIDBKeyType::Min:
302 result = IDBKeyData::minimum();
303 return true;
304 case SIDBKeyType::Max:
305 result = IDBKeyData::maximum();
306 return true;
307 case SIDBKeyType::Number: {
308 double d;
309 if (!readDouble(data, end, d))
310 return false;
311
312 result.setNumberValue(d);
313 return true;
314 }
315 case SIDBKeyType::Date: {
316 double d;
317 if (!readDouble(data, end, d))
318 return false;
319
320 result.setDateValue(d);
321 return true;
322 }
323 case SIDBKeyType::String: {
324 uint32_t length;
325 if (!readLittleEndian(data, end, length))
326 return false;
327
328 if (static_cast<uint64_t>(end - data) < length * 2)
329 return false;
330
331 Vector<UChar> buffer;
332 buffer.reserveInitialCapacity(length);
333 for (size_t i = 0; i < length; i++) {
334 uint16_t ch;
335 if (!readLittleEndian(data, end, ch))
336 return false;
337 buffer.uncheckedAppend(ch);
338 }
339
340 result.setStringValue(String::adopt(WTFMove(buffer)));
341
342 return true;
343 }
344 case SIDBKeyType::Binary: {
345 uint64_t size64;
346 if (!readLittleEndian(data, end, size64))
347 return false;
348
349 if (static_cast<uint64_t>(end - data) < size64)
350 return false;
351
352 if (size64 > std::numeric_limits<size_t>::max())
353 return false;
354
355 size_t size = static_cast<size_t>(size64);
356 Vector<uint8_t> dataVector;
357
358 dataVector.append(data, size);
359 data += size;
360
361 result.setBinaryValue(ThreadSafeDataBuffer::create(WTFMove(dataVector)));
362 return true;
363 }
364 case SIDBKeyType::Array: {
365 uint64_t size64;
366 if (!readLittleEndian(data, end, size64))
367 return false;
368
369 if (size64 > std::numeric_limits<size_t>::max())
370 return false;
371
372 size_t size = static_cast<size_t>(size64);
373 Vector<IDBKeyData> array;
374 array.reserveInitialCapacity(size);
375
376 for (size_t i = 0; i < size; ++i) {
377 IDBKeyData keyData;
378 if (!decodeKey(data, end, keyData))
379 return false;
380
381 ASSERT(keyData.isValid());
382 array.uncheckedAppend(WTFMove(keyData));
383 }
384
385 result.setArrayValue(array);
386
387 return true;
388 }
389 default:
390 LOG_ERROR("decodeKey encountered unexpected type: %i", (int)type);
391 return false;
392 }
393}
394
395bool deserializeIDBKeyData(const uint8_t* data, size_t size, IDBKeyData& result)
396{
397 if (!data || !size)
398 return false;
399
400 if (isLegacySerializedIDBKeyData(data, size)) {
401 auto decoder = KeyedDecoder::decoder(data, size);
402 return IDBKeyData::decode(*decoder, result);
403 }
404
405 // Verify this is a SerializedIDBKey version we understand.
406 const uint8_t* current = data;
407 const uint8_t* end = data + size;
408 if (current++[0] != SIDBKeyVersion)
409 return false;
410
411 if (decodeKey(current, end, result)) {
412 // Even if we successfully decoded a key, the deserialize is only successful
413 // if we actually consumed all input data.
414 return current == end;
415 }
416
417 return false;
418}
