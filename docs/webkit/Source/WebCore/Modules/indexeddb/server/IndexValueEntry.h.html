26#pragma once
27
28#include "IDBKeyData.h"
29
30namespace WebCore {
31
32class ThreadSafeDataBuffer;
33
34enum class CursorDuplicity;
35
36namespace IDBServer {
37
38class IndexValueEntry {
39 WTF_MAKE_FAST_ALLOCATED;
40public:
41 explicit IndexValueEntry(bool unique);
42 ~IndexValueEntry();
43
44 void addKey(const IDBKeyData&);
45
46 // Returns true if a key was actually removed.
47 bool removeKey(const IDBKeyData&);
48
49 const IDBKeyData* getLowest() const;
50
51 uint64_t getCount() const;
52
53 class Iterator {
54 public:
55 Iterator()
56 {
57 }
58
59 Iterator(IndexValueEntry&);
60 Iterator(IndexValueEntry&, IDBKeyDataSet::iterator);
61 Iterator(IndexValueEntry&, IDBKeyDataSet::reverse_iterator);
62
63 bool isValid() const;
64 void invalidate();
65
66 const IDBKeyData& key() const;
67 const ThreadSafeDataBuffer& value() const;
68
69 Iterator& operator++();
70
71 private:
72 IndexValueEntry* m_entry { nullptr };
73 bool m_forward { true };
74 IDBKeyDataSet::iterator m_forwardIterator;
75 IDBKeyDataSet::reverse_iterator m_reverseIterator;
76 };
77
78 Iterator begin();
79 Iterator reverseBegin(CursorDuplicity);
80
81 // Finds the key, or the next higher record after the key.
82 Iterator find(const IDBKeyData&);
83 // Finds the key, or the next lowest record before the key.
84 Iterator reverseFind(const IDBKeyData&, CursorDuplicity);
85
86 bool unique() const { return m_unique; }
87
88private:
89 union {
90 IDBKeyDataSet* m_orderedKeys;
91 IDBKeyData* m_key;
92 };
93
94 bool m_unique;
95};
