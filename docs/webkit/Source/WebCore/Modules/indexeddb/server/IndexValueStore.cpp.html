52Vector<IDBKeyData> IndexValueStore::allValuesForKey(const IDBKeyData& key, uint32_t limit) const
53{
54 const auto& entry = m_records.get(key);
55 if (!entry)
56 return { };
57
58 Vector<IDBKeyData> results;
59 for (auto iterator = entry->begin(); results.size() < limit && iterator.isValid(); ++iterator)
60 results.append(iterator.key());
61
62 return results;
63}

85IDBError IndexValueStore::addRecord(const IDBKeyData& indexKey, const IDBKeyData& valueKey)
86{
87 auto result = m_records.add(indexKey, nullptr);
88
89 if (!result.isNewEntry && m_unique)
90 return IDBError(ConstraintError);
91
92 if (result.isNewEntry)
93 result.iterator->value = makeUnique<IndexValueEntry>(m_unique);
94
95 result.iterator->value->addKey(valueKey);
96 m_orderedKeys.insert(indexKey);
97
98 return IDBError { };
99}

101void IndexValueStore::removeRecord(const IDBKeyData& indexKey, const IDBKeyData& valueKey)
102{
103 auto iterator = m_records.find(indexKey);
104 if (!iterator->value)
105 return;
106
107 if (iterator->value->removeKey(valueKey))
108 m_records.remove(iterator);
109}
110
111void IndexValueStore::removeEntriesWithValueKey(MemoryIndex& index, const IDBKeyData& valueKey)
112{
113 Vector<IDBKeyData> entryKeysToRemove;
114 entryKeysToRemove.reserveInitialCapacity(m_records.size());
115
116 for (auto& entry : m_records) {
117 if (entry.value->removeKey(valueKey))
118 index.notifyCursorsOfValueChange(entry.key, valueKey);
119 if (!entry.value->getCount())
120 entryKeysToRemove.uncheckedAppend(entry.key);
121 }
122
123 for (auto& entry : entryKeysToRemove) {
124 m_orderedKeys.erase(entry);
125 m_records.remove(entry);
126 }
127}

129IDBKeyData IndexValueStore::lowestKeyWithRecordInRange(const IDBKeyRangeData& range) const
130{
131 LOG(IndexedDB, "IndexValueStore::lowestKeyWithRecordInRange - %s", range.loggingString().utf8().data());
132
133 if (range.isExactlyOneKey())
134 return m_records.contains(range.lowerKey) ? range.lowerKey : IDBKeyData();
135
136 auto iterator = lowestIteratorInRange(range);
137 if (iterator == m_orderedKeys.end())
138 return { };
139
140 return *iterator;
141}

143IDBKeyDataSet::iterator IndexValueStore::lowestIteratorInRange(const IDBKeyRangeData& range) const
144{
145 auto lowestInRange = m_orderedKeys.lower_bound(range.lowerKey);
146
147 if (lowestInRange == m_orderedKeys.end())
148 return lowestInRange;
149
150 if (range.lowerOpen && *lowestInRange == range.lowerKey) {
151 ++lowestInRange;
152
153 if (lowestInRange == m_orderedKeys.end())
154 return lowestInRange;
155 }
156
157 if (!range.upperKey.isNull()) {
158 if (lowestInRange->compare(range.upperKey) > 0)
159 return m_orderedKeys.end();
160 if (range.upperOpen && *lowestInRange == range.upperKey)
161 return m_orderedKeys.end();
162 }
163
164 return lowestInRange;
165}

167IDBKeyDataSet::reverse_iterator IndexValueStore::highestReverseIteratorInRange(const IDBKeyRangeData& range) const
168{
169 auto highestInRange = IDBKeyDataSet::reverse_iterator(m_orderedKeys.upper_bound(range.upperKey));
170
171 if (highestInRange == m_orderedKeys.rend())
172 return highestInRange;
173
174 if (range.upperOpen && *highestInRange == range.upperKey) {
175 ++highestInRange;
176
177 if (highestInRange == m_orderedKeys.rend())
178 return highestInRange;
179 }
180
181 if (!range.lowerKey.isNull()) {
182 if (highestInRange->compare(range.lowerKey) < 0)
183 return m_orderedKeys.rend();
184 if (range.lowerOpen && *highestInRange == range.lowerKey)
185 return m_orderedKeys.rend();
186 }
187
188 return highestInRange;
189}

191IndexValueStore::Iterator IndexValueStore::find(const IDBKeyData& key, bool open)
192{
193 IDBKeyRangeData range;
194 if (!key.isNull())
195 range.lowerKey = key;
196 else
197 range.lowerKey = IDBKeyData::minimum();
198 range.lowerOpen = open;
199
200 auto iterator = lowestIteratorInRange(range);
201 if (iterator == m_orderedKeys.end())
202 return { };
203
204 auto record = m_records.get(*iterator);
205 ASSERT(record);
206
207 auto primaryIterator = record->begin();
208 ASSERT(primaryIterator.isValid());
209 return { *this, iterator, primaryIterator };
210}

212IndexValueStore::Iterator IndexValueStore::find(const IDBKeyData& key, const IDBKeyData& primaryKey)
213{
214 ASSERT(!key.isNull());
215 ASSERT(!primaryKey.isNull());
216
217 IDBKeyRangeData range;
218 range.lowerKey = key;
219 range.lowerOpen = false;
220
221 auto iterator = lowestIteratorInRange(range);
222 if (iterator == m_orderedKeys.end())
223 return { };
224
225 auto record = m_records.get(*iterator);
226 ASSERT(record);
227
228 // If the main record iterator is not equal to the key we were looking for,
229 // we know the primary key record should be the first.
230 if (*iterator != key) {
231 auto primaryIterator = record->begin();
232 ASSERT(primaryIterator.isValid());
233
234 return { *this, iterator, primaryIterator };
235 }
236
237 auto primaryIterator = record->find(primaryKey);
238 if (primaryIterator.isValid())
239 return { *this, iterator, primaryIterator };
240
241 // If we didn't find a primary key iterator in this entry,
242 // we need to move on to start of the next record.
243 iterator++;
244 if (iterator == m_orderedKeys.end())
245 return { };
246
247 record = m_records.get(*iterator);
248 ASSERT(record);
249
250 primaryIterator = record->begin();
251 ASSERT(primaryIterator.isValid());
252
253 return { *this, iterator, primaryIterator };
254}

256IndexValueStore::Iterator IndexValueStore::reverseFind(const IDBKeyData& key, CursorDuplicity duplicity, bool open)
257{
258 IDBKeyRangeData range;
259 if (!key.isNull())
260 range.upperKey = key;
261 else
262 range.upperKey = IDBKeyData::maximum();
263 range.upperOpen = open;
264
265 auto iterator = highestReverseIteratorInRange(range);
266 if (iterator == m_orderedKeys.rend())
267 return { };
268
269 auto record = m_records.get(*iterator);
270 ASSERT(record);
271
272 auto primaryIterator = record->reverseBegin(duplicity);
273 ASSERT(primaryIterator.isValid());
274 return { *this, duplicity, iterator, primaryIterator };
275}

277IndexValueStore::Iterator IndexValueStore::reverseFind(const IDBKeyData& key, const IDBKeyData& primaryKey, CursorDuplicity duplicity)
278{
279 ASSERT(!key.isNull());
280 ASSERT(!primaryKey.isNull());
281
282 IDBKeyRangeData range;
283 range.upperKey = key;
284 range.upperOpen = false;
285
286 auto iterator = highestReverseIteratorInRange(range);
287 if (iterator == m_orderedKeys.rend())
288 return { };
289
290 auto record = m_records.get(*iterator);
291 ASSERT(record);
292
293 auto primaryIterator = record->reverseFind(primaryKey, duplicity);
294 if (primaryIterator.isValid())
295 return { *this, duplicity, iterator, primaryIterator };
296
297 // If we didn't find a primary key iterator in this entry,
298 // we need to move on to start of the next record.
299 iterator++;
300 if (iterator == m_orderedKeys.rend())
301 return { };
302
303 record = m_records.get(*iterator);
304 ASSERT(record);
305
306 primaryIterator = record->reverseBegin(duplicity);
307 ASSERT(primaryIterator.isValid());
308
309 return { *this, duplicity, iterator, primaryIterator };
310}

329IndexValueStore::Iterator& IndexValueStore::Iterator::nextIndexEntry()
330{
331 if (!m_store)
332 return *this;
333
334 if (m_forward) {
335 ++m_forwardIterator;
336 if (m_forwardIterator == m_store->m_orderedKeys.end()) {
337 invalidate();
338 return *this;
339 }
340
341 auto* entry = m_store->m_records.get(*m_forwardIterator);
342 ASSERT(entry);
343
344 m_primaryKeyIterator = entry->begin();
345 ASSERT(m_primaryKeyIterator.isValid());
346 } else {
347 ++m_reverseIterator;
348 if (m_reverseIterator == m_store->m_orderedKeys.rend()) {
349 invalidate();
350 return *this;
351 }
352
353 auto* entry = m_store->m_records.get(*m_reverseIterator);
354 ASSERT(entry);
355
356 m_primaryKeyIterator = entry->reverseBegin(m_duplicity);
357 ASSERT(m_primaryKeyIterator.isValid());
358 }
359
360 return *this;
361}
362
363IndexValueStore::Iterator& IndexValueStore::Iterator::operator++()
364{
365 if (!isValid())
366 return *this;
367
368 ++m_primaryKeyIterator;
369 if (m_primaryKeyIterator.isValid())
370 return *this;
371
372 // Ran out of primary key records, so move the main index iterator.
373 return nextIndexEntry();
374}
375
376void IndexValueStore::Iterator::invalidate()
377{
378 m_store = nullptr;
379 m_primaryKeyIterator.invalidate();
380}
381
382bool IndexValueStore::Iterator::isValid()
383{
384 return m_store && m_primaryKeyIterator.isValid();
385}
386
387const IDBKeyData& IndexValueStore::Iterator::key()
388{
389 ASSERT(isValid());
390 return m_forward ? *m_forwardIterator : *m_reverseIterator;
391}
392
393const IDBKeyData& IndexValueStore::Iterator::primaryKey()
394{
395 ASSERT(isValid());
396 return m_primaryKeyIterator.key();
397}
