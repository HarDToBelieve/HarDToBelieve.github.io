33namespace WebCore {
34
35class IDBError;
36
37struct IDBKeyRangeData;
38
39namespace IDBServer {
40
41class MemoryIndex;
42
43typedef HashMap<IDBKeyData, std::unique_ptr<IndexValueEntry>, IDBKeyDataHash, IDBKeyDataHashTraits> IndexKeyValueMap;
44
45class IndexValueStore {
46 WTF_MAKE_FAST_ALLOCATED;
47public:
48 explicit IndexValueStore(bool unique);
49
50 const IDBKeyData* lowestValueForKey(const IDBKeyData&) const;
51 Vector<IDBKeyData> allValuesForKey(const IDBKeyData&, uint32_t limit) const;
52 uint64_t countForKey(const IDBKeyData&) const;
53 IDBKeyData lowestKeyWithRecordInRange(const IDBKeyRangeData&) const;
54 bool contains(const IDBKeyData&) const;
55
56 IDBError addRecord(const IDBKeyData& indexKey, const IDBKeyData& valueKey);
57 void removeRecord(const IDBKeyData& indexKey, const IDBKeyData& valueKey);
58
59 void removeEntriesWithValueKey(MemoryIndex&, const IDBKeyData& valueKey);
60
61 class Iterator {
62 friend class IndexValueStore;
63 public:
64 Iterator()
65 {
66 }
67
68 Iterator(IndexValueStore&, IDBKeyDataSet::iterator, IndexValueEntry::Iterator);
69 Iterator(IndexValueStore&, CursorDuplicity, IDBKeyDataSet::reverse_iterator, IndexValueEntry::Iterator);
70
71 void invalidate();
72 bool isValid();
73
74 const IDBKeyData& key();
75 const IDBKeyData& primaryKey();
76 const ThreadSafeDataBuffer& value();
77
78 Iterator& operator++();
79 Iterator& nextIndexEntry();
80
81 private:
82 IndexValueStore* m_store { nullptr };
83 bool m_forward { true };
84 CursorDuplicity m_duplicity { CursorDuplicity::Duplicates };
85 IDBKeyDataSet::iterator m_forwardIterator;
86 IDBKeyDataSet::reverse_iterator m_reverseIterator;
87
88 IndexValueEntry::Iterator m_primaryKeyIterator;
89 };
90
91 // Returns an iterator pointing to the first primaryKey record in the requested key, or the next key if it doesn't exist.
92 Iterator find(const IDBKeyData&, bool open = false);
93 Iterator reverseFind(const IDBKeyData&, CursorDuplicity, bool open = false);
94
95 // Returns an iterator pointing to the key/primaryKey record, or the next one after it if it doesn't exist.
96 Iterator find(const IDBKeyData&, const IDBKeyData& primaryKey);
97 Iterator reverseFind(const IDBKeyData&, const IDBKeyData& primaryKey, CursorDuplicity);
98
99#if !LOG_DISABLED
100 String loggingString() const;
101#endif
102
103private:
104 IDBKeyDataSet::iterator lowestIteratorInRange(const IDBKeyRangeData&) const;
105 IDBKeyDataSet::reverse_iterator highestReverseIteratorInRange(const IDBKeyRangeData&) const;
106
107 IndexKeyValueMap m_records;
108 IDBKeyDataSet m_orderedKeys;
109
110 bool m_unique;
111};
