40std::unique_ptr<MemoryBackingStoreTransaction> MemoryBackingStoreTransaction::create(MemoryIDBBackingStore& backingStore, const IDBTransactionInfo& info)
41{
42 return makeUnique<MemoryBackingStoreTransaction>(backingStore, info);

45MemoryBackingStoreTransaction::MemoryBackingStoreTransaction(MemoryIDBBackingStore& backingStore, const IDBTransactionInfo& info)
46 : m_backingStore(backingStore)
47 , m_info(info)
48{
49 if (m_info.mode() == IDBTransactionMode::Versionchange) {
50 IDBDatabaseInfo info;
51 auto error = m_backingStore.getOrEstablishDatabaseInfo(info);
52 if (error.isNull())
53 m_originalDatabaseInfo = makeUnique<IDBDatabaseInfo>(info);
54 }
55}

92void MemoryBackingStoreTransaction::indexDeleted(Ref<MemoryIndex>&& index)
93{
94 m_indexes.remove(&index.get());
95
96 // If this MemoryIndex belongs to an object store that will not get restored if this transaction aborts,
97 // then we can forget about it altogether.
98 auto& objectStore = index->objectStore();
99 if (auto deletedObjectStore = m_deletedObjectStores.get(objectStore.info().name())) {
100 if (deletedObjectStore != &objectStore)
101 return;
102 }
103
104 auto addResult = m_deletedIndexes.add(index->info().name(), nullptr);
105 if (addResult.isNewEntry)
106 addResult.iterator->value = WTFMove(index);
107}

109void MemoryBackingStoreTransaction::addExistingObjectStore(MemoryObjectStore& objectStore)
110{
111 LOG(IndexedDB, "MemoryBackingStoreTransaction::addExistingObjectStore");
112
113 ASSERT(isWriting());
114
115 ASSERT(!m_objectStores.contains(&objectStore));
116 m_objectStores.add(&objectStore);
117
118 objectStore.writeTransactionStarted(*this);
119
120 m_originalKeyGenerators.add(&objectStore, objectStore.currentKeyGeneratorValue());
121}

123void MemoryBackingStoreTransaction::objectStoreDeleted(Ref<MemoryObjectStore>&& objectStore)
124{
125 ASSERT(m_objectStores.contains(&objectStore.get()));
126 m_objectStores.remove(&objectStore.get());
127
128 objectStore->deleteAllIndexes(*this);
129
130 auto addResult = m_deletedObjectStores.add(objectStore->info().name(), nullptr);
131 if (addResult.isNewEntry)
132 addResult.iterator->value = WTFMove(objectStore);
133}

135void MemoryBackingStoreTransaction::objectStoreCleared(MemoryObjectStore& objectStore, std::unique_ptr<KeyValueMap>&& keyValueMap, std::unique_ptr<IDBKeyDataSet>&& orderedKeys)
136{
137 ASSERT(m_objectStores.contains(&objectStore));
138
139 auto addResult = m_clearedKeyValueMaps.add(&objectStore, nullptr);
140
141 // If this object store has already been cleared during this transaction, we shouldn't remember this clearing.
142 if (!addResult.isNewEntry)
143 return;
144
145 addResult.iterator->value = WTFMove(keyValueMap);
146
147 ASSERT(!m_clearedOrderedKeys.contains(&objectStore));
148 m_clearedOrderedKeys.add(&objectStore, WTFMove(orderedKeys));
149}

151void MemoryBackingStoreTransaction::objectStoreRenamed(MemoryObjectStore& objectStore, const String& oldName)
152{
153 ASSERT(m_objectStores.contains(&objectStore));
154 ASSERT(m_info.mode() == IDBTransactionMode::Versionchange);
155
156 // We only care about the first rename in a given transaction, because if the transaction is aborted we want
157 // to go back to the first 'oldName'
158 m_originalObjectStoreNames.add(&objectStore, oldName);
159}

166 // We only care about the first rename in a given transaction, because if the transaction is aborted we want

168 m_originalIndexNames.add(&index, oldName);

171void MemoryBackingStoreTransaction::indexCleared(MemoryIndex& index, std::unique_ptr<IndexValueStore>&& valueStore)

182void MemoryBackingStoreTransaction::recordValueChanged(MemoryObjectStore& objectStore, const IDBKeyData& key, ThreadSafeDataBuffer* value)
183{
184 ASSERT(m_objectStores.contains(&objectStore));
185
186 if (m_isAborting)
187 return;
188
189 // If this object store had been cleared during the transaction, no point in recording this
190 // individual key/value change as its entire key/value map will be restored upon abort.
191 if (m_clearedKeyValueMaps.contains(&objectStore))
192 return;
193
194 auto originalAddResult = m_originalValues.add(&objectStore, nullptr);
195 if (originalAddResult.isNewEntry)
196 originalAddResult.iterator->value = makeUnique<KeyValueMap>();
197
198 auto* map = originalAddResult.iterator->value.get();
199
200 auto addResult = map->add(key, ThreadSafeDataBuffer());
201 if (!addResult.isNewEntry)
202 return;
203
204 if (value)
205 addResult.iterator->value = *value;
206}

208void MemoryBackingStoreTransaction::abort()
209{
210 LOG(IndexedDB, "MemoryBackingStoreTransaction::abort()");
211
212 SetForScope<bool> change(m_isAborting, true);
213
214 for (const auto& iterator : m_originalIndexNames)
215 iterator.key->rename(iterator.value);
216 m_originalIndexNames.clear();
217
218 for (const auto& iterator : m_originalObjectStoreNames)
219 iterator.key->rename(iterator.value);
220 m_originalObjectStoreNames.clear();
221
222 for (const auto& objectStore : m_versionChangeAddedObjectStores)
223 m_backingStore.removeObjectStoreForVersionChangeAbort(*objectStore);
224 m_versionChangeAddedObjectStores.clear();
225
226 for (auto& objectStore : m_deletedObjectStores.values()) {
227 m_backingStore.restoreObjectStoreForVersionChangeAbort(*objectStore);
228 ASSERT(!m_objectStores.contains(objectStore.get()));
229 m_objectStores.add(objectStore);
230 }
231 m_deletedObjectStores.clear();
232
233 if (m_originalDatabaseInfo) {
234 ASSERT(m_info.mode() == IDBTransactionMode::Versionchange);
235 m_backingStore.setDatabaseInfo(*m_originalDatabaseInfo);
236 }
237
238 // Restore cleared index value stores before we re-insert values into object stores
239 // because inserting those values will regenerate the appropriate index values.
240 for (auto& iterator : m_clearedIndexValueStores)
241 iterator.key->replaceIndexValueStore(WTFMove(iterator.value));
242 m_clearedIndexValueStores.clear();
243
244 for (auto& objectStore : m_objectStores) {
245 ASSERT(m_originalKeyGenerators.contains(objectStore.get()));
246 objectStore->setKeyGeneratorValue(m_originalKeyGenerators.get(objectStore.get()));
247
248 auto clearedKeyValueMap = m_clearedKeyValueMaps.take(objectStore.get());
249 if (clearedKeyValueMap) {
250 ASSERT(m_clearedOrderedKeys.contains(objectStore.get()));
251 objectStore->replaceKeyValueStore(WTFMove(clearedKeyValueMap), m_clearedOrderedKeys.take(objectStore.get()));
252 }
253
254 auto keyValueMap = m_originalValues.take(objectStore.get());
255 if (!keyValueMap)
256 continue;
257
258 for (const auto& entry : *keyValueMap) {
259 objectStore->deleteRecord(entry.key);
260 objectStore->addRecord(*this, entry.key, { entry.value });
261 }
262 }
263
264 for (auto& index : m_deletedIndexes.values())
265 index->objectStore().maybeRestoreDeletedIndex(*index);
266 m_deletedIndexes.clear();
267
268 finish();
269}
270
271void MemoryBackingStoreTransaction::commit()
272{
273 LOG(IndexedDB, "MemoryBackingStoreTransaction::commit()");
274
275 finish();
276}
277
278void MemoryBackingStoreTransaction::finish()
279{
280 m_inProgress = false;
281
282 if (!isWriting())
283 return;
284
285 for (auto& objectStore : m_objectStores)
286 objectStore->writeTransactionFinished(*this);
287 for (auto& objectStore : m_deletedObjectStores.values())
288 objectStore->writeTransactionFinished(*this);
289}
