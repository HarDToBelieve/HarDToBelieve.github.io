27
28#include "IDBDatabaseInfo.h"
29#include "IDBKeyData.h"
30#include "IDBTransactionInfo.h"

36namespace WebCore {
37namespace IDBServer {
38
39class MemoryIDBBackingStore;
40class MemoryIndex;
41class MemoryObjectStore;
42
43typedef HashMap<IDBKeyData, ThreadSafeDataBuffer, IDBKeyDataHash, IDBKeyDataHashTraits> KeyValueMap;
44
45class MemoryBackingStoreTransaction {
46 WTF_MAKE_FAST_ALLOCATED;
47public:
48 static std::unique_ptr<MemoryBackingStoreTransaction> create(MemoryIDBBackingStore&, const IDBTransactionInfo&);
49
50 MemoryBackingStoreTransaction(MemoryIDBBackingStore&, const IDBTransactionInfo&);
51 ~MemoryBackingStoreTransaction();
52
53 bool isVersionChange() const { return m_info.mode() == IDBTransactionMode::Versionchange; }
54 bool isWriting() const { return m_info.mode() != IDBTransactionMode::Readonly; }
55 bool isAborting() const { return m_isAborting; }
56
57 const IDBDatabaseInfo& originalDatabaseInfo() const;
58
59 void addNewObjectStore(MemoryObjectStore&);
60 void addExistingObjectStore(MemoryObjectStore&);
61
62 void recordValueChanged(MemoryObjectStore&, const IDBKeyData&, ThreadSafeDataBuffer*);
63 void objectStoreDeleted(Ref<MemoryObjectStore>&&);
64 void objectStoreCleared(MemoryObjectStore&, std::unique_ptr<KeyValueMap>&&, std::unique_ptr<IDBKeyDataSet>&&);
65 void objectStoreRenamed(MemoryObjectStore&, const String& oldName);
66 void indexRenamed(MemoryIndex&, const String& oldName);
67 void indexCleared(MemoryIndex&, std::unique_ptr<IndexValueStore>&&);
68
69 void addNewIndex(MemoryIndex&);
70 void addExistingIndex(MemoryIndex&);
71 void indexDeleted(Ref<MemoryIndex>&&);
72
73 void abort();
74 void commit();
75
76private:
77 void finish();
78
79 MemoryIDBBackingStore& m_backingStore;
80 IDBTransactionInfo m_info;
81
82 std::unique_ptr<IDBDatabaseInfo> m_originalDatabaseInfo;
83
84 bool m_inProgress { true };
85 bool m_isAborting { false };
86
87 HashSet<RefPtr<MemoryObjectStore>> m_objectStores;
88 HashSet<RefPtr<MemoryObjectStore>> m_versionChangeAddedObjectStores;
89 HashSet<RefPtr<MemoryIndex>> m_indexes;
90 HashSet<RefPtr<MemoryIndex>> m_versionChangeAddedIndexes;
91
92 HashMap<MemoryObjectStore*, uint64_t> m_originalKeyGenerators;
93 HashMap<String, RefPtr<MemoryObjectStore>> m_deletedObjectStores;
94 HashMap<String, RefPtr<MemoryIndex>> m_deletedIndexes;
95 HashMap<MemoryObjectStore*, std::unique_ptr<KeyValueMap>> m_originalValues;
96 HashMap<MemoryObjectStore*, std::unique_ptr<KeyValueMap>> m_clearedKeyValueMaps;
97 HashMap<MemoryObjectStore*, std::unique_ptr<IDBKeyDataSet>> m_clearedOrderedKeys;
98 HashMap<MemoryObjectStore*, String> m_originalObjectStoreNames;
99 HashMap<MemoryIndex*, String> m_originalIndexNames;
100 HashMap<MemoryIndex*, std::unique_ptr<IndexValueStore>> m_clearedIndexValueStores;
101};
