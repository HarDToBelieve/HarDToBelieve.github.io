26#include "config.h"
27#include "MemoryCursor.h"
28
29#include "IDBResourceIdentifier.h"
30#include <wtf/HashMap.h>
31#include <wtf/NeverDestroyed.h>
32
33namespace WebCore {
34namespace IDBServer {
35
36static Lock cursorMapLock;
37static HashMap<IDBResourceIdentifier, MemoryCursor*>& cursorMap() WTF_REQUIRES_LOCK(cursorMapLock)
38{
39 static NeverDestroyed<HashMap<IDBResourceIdentifier, MemoryCursor*>> map;
40 return map;
41}
42
43MemoryCursor::MemoryCursor(const IDBCursorInfo& info)
44 : m_info(info)
45{
46 ASSERT(!isMainThread());
47
48 Locker locker { cursorMapLock };
49 ASSERT(!cursorMap().contains(m_info.identifier()));
50 cursorMap().set(m_info.identifier(), this);
51}
52
53MemoryCursor::~MemoryCursor()
54{
55 ASSERT(!isMainThread());
56
57 Locker locker { cursorMapLock };
58 ASSERT(cursorMap().contains(m_info.identifier()));
59 cursorMap().remove(m_info.identifier());
60}
61
62MemoryCursor* MemoryCursor::cursorForIdentifier(const IDBResourceIdentifier& identifier)
63{
64 ASSERT(!isMainThread());
65
66 Locker locker { cursorMapLock };
67 return cursorMap().get(identifier);
68}
