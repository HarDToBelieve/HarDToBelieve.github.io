77IDBError MemoryIDBBackingStore::beginTransaction(const IDBTransactionInfo& info)
78{
79 LOG(IndexedDB, "MemoryIDBBackingStore::beginTransaction");
80
81 if (m_transactions.contains(info.identifier()))
82 return IDBError { InvalidStateError, "Backing store asked to create transaction it already has a record of" };
83
84 auto transaction = MemoryBackingStoreTransaction::create(*this, info);
85
86 // VersionChange transactions are scoped to "every object store".
87 if (transaction->isVersionChange()) {
88 for (auto& objectStore : m_objectStoresByIdentifier.values())
89 transaction->addExistingObjectStore(*objectStore);
90 } else if (transaction->isWriting()) {
91 for (auto& iterator : m_objectStoresByName) {
92 if (info.objectStores().contains(iterator.key))
93 transaction->addExistingObjectStore(*iterator.value);
94 }
95 }
96
97 m_transactions.set(info.identifier(), WTFMove(transaction));
98
99 return IDBError { };
100}

128IDBError MemoryIDBBackingStore::createObjectStore(const IDBResourceIdentifier& transactionIdentifier, const IDBObjectStoreInfo& info)
129{
130 LOG(IndexedDB, "MemoryIDBBackingStore::createObjectStore - adding OS %s with ID %" PRIu64, info.name().utf8().data(), info.identifier());
131
132 ASSERT(m_databaseInfo);
133 if (m_databaseInfo->hasObjectStore(info.name()))
134 return IDBError { ConstraintError };
135
136 ASSERT(!m_objectStoresByIdentifier.contains(info.identifier()));
137 auto objectStore = MemoryObjectStore::create(m_sessionID, info);
138
139 m_databaseInfo->addExistingObjectStore(info);
140
141 auto rawTransaction = m_transactions.get(transactionIdentifier);
142 ASSERT(rawTransaction);
143 ASSERT(rawTransaction->isVersionChange());
144
145 rawTransaction->addNewObjectStore(objectStore.get());
146 registerObjectStore(WTFMove(objectStore));
147
148 return IDBError { };
149}

151IDBError MemoryIDBBackingStore::deleteObjectStore(const IDBResourceIdentifier& transactionIdentifier, uint64_t objectStoreIdentifier)
152{
153 LOG(IndexedDB, "MemoryIDBBackingStore::deleteObjectStore");
154
155 ASSERT(m_databaseInfo);
156 if (!m_databaseInfo->infoForExistingObjectStore(objectStoreIdentifier))
157 return IDBError { ConstraintError };
158
159 auto transaction = m_transactions.get(transactionIdentifier);
160 ASSERT(transaction);
161 ASSERT(transaction->isVersionChange());
162
163 auto objectStore = takeObjectStoreByIdentifier(objectStoreIdentifier);
164 ASSERT(objectStore);
165 if (!objectStore)
166 return IDBError { ConstraintError };
167
168 m_databaseInfo->deleteObjectStore(objectStore->info().name());
169 transaction->objectStoreDeleted(*objectStore);
170
171 return IDBError { };
172}

174IDBError MemoryIDBBackingStore::renameObjectStore(const IDBResourceIdentifier& transactionIdentifier, uint64_t objectStoreIdentifier, const String& newName)
175{
176 LOG(IndexedDB, "MemoryIDBBackingStore::renameObjectStore");
177
178 ASSERT(m_databaseInfo);
179 if (!m_databaseInfo->infoForExistingObjectStore(objectStoreIdentifier))
180 return IDBError { ConstraintError };
181
182 auto transaction = m_transactions.get(transactionIdentifier);
183 ASSERT(transaction);
184 ASSERT(transaction->isVersionChange());
185
186 auto objectStore = m_objectStoresByIdentifier.get(objectStoreIdentifier);
187 ASSERT(objectStore);
188 if (!objectStore)
189 return IDBError { ConstraintError };
190
191 String oldName = objectStore->info().name();
192 objectStore->rename(newName);
193 transaction->objectStoreRenamed(*objectStore, oldName);
194
195 m_objectStoresByName.remove(oldName);
196 m_objectStoresByName.set(newName, objectStore);
197
198 m_databaseInfo->renameObjectStore(objectStoreIdentifier, newName);
199
200 return IDBError { };
201}

224IDBError MemoryIDBBackingStore::createIndex(const IDBResourceIdentifier& transactionIdentifier, const IDBIndexInfo& info)
225{
226 LOG(IndexedDB, "MemoryIDBBackingStore::createIndex");
227
228 ASSERT(m_databaseInfo);
229 auto* objectStoreInfo = m_databaseInfo->infoForExistingObjectStore(info.objectStoreIdentifier());
230 if (!objectStoreInfo)
231 return IDBError { ConstraintError };
232
233 auto rawTransaction = m_transactions.get(transactionIdentifier);
234 ASSERT(rawTransaction);
235 ASSERT(rawTransaction->isVersionChange());
236
237 auto* objectStore = m_objectStoresByIdentifier.get(info.objectStoreIdentifier());
238 if (!objectStore)
239 return IDBError { ConstraintError };
240
241 auto error = objectStore->createIndex(*rawTransaction, info);
242 if (error.isNull()) {
243 objectStoreInfo->addExistingIndex(info);
244 m_databaseInfo->setMaxIndexID(info.identifier());
245 }
246
247 return error;
248}

278IDBError MemoryIDBBackingStore::renameIndex(const IDBResourceIdentifier& transactionIdentifier, uint64_t objectStoreIdentifier, uint64_t indexIdentifier, const String& newName)
279{
280 LOG(IndexedDB, "MemoryIDBBackingStore::renameIndex");
281
282 ASSERT(m_databaseInfo);
283 auto* objectStoreInfo = m_databaseInfo->infoForExistingObjectStore(objectStoreIdentifier);
284 if (!objectStoreInfo)
285 return IDBError { ConstraintError };
286
287 auto* indexInfo = objectStoreInfo->infoForExistingIndex(indexIdentifier);
288 if (!indexInfo)
289 return IDBError { ConstraintError };
290
291 auto transaction = m_transactions.get(transactionIdentifier);
292 ASSERT(transaction);
293 ASSERT(transaction->isVersionChange());
294
295 auto objectStore = m_objectStoresByIdentifier.get(objectStoreIdentifier);
296 ASSERT(objectStore);
297 if (!objectStore)
298 return IDBError { ConstraintError };
299
300 auto* index = objectStore->indexForIdentifier(indexIdentifier);
301 ASSERT(index);
302 if (!index)
303 return IDBError { ConstraintError };
304
305 String oldName = index->info().name();
306 objectStore->renameIndex(*index, newName);
307 transaction->indexRenamed(*index, oldName);
308
309 indexInfo->rename(newName);
310
311 return IDBError { };
312}

378IDBError MemoryIDBBackingStore::getRecord(const IDBResourceIdentifier& transactionIdentifier, uint64_t objectStoreIdentifier, const IDBKeyRangeData& range, IDBGetRecordDataType type, IDBGetResult& outValue)
379{
380 LOG(IndexedDB, "MemoryIDBBackingStore::getRecord");
381
382 ASSERT(objectStoreIdentifier);
383
384 if (!m_transactions.contains(transactionIdentifier))
385 return IDBError { UnknownError, "No backing store transaction found to get record"_s };
386
387 MemoryObjectStore* objectStore = m_objectStoresByIdentifier.get(objectStoreIdentifier);
388 if (!objectStore)
389 return IDBError { UnknownError, "No backing store object store found"_s };
390
391 switch (type) {
392 case IDBGetRecordDataType::KeyAndValue: {
393 auto key = objectStore->lowestKeyWithRecordInRange(range);
394 outValue = { key, key.isNull() ? ThreadSafeDataBuffer() : objectStore->valueForKey(key), objectStore->info().keyPath() };
395 break;
396 }
397 case IDBGetRecordDataType::KeyOnly:
398 outValue = objectStore->lowestKeyWithRecordInRange(range);
399 break;
400 }
401
402 return IDBError { };
403}

405IDBError MemoryIDBBackingStore::getAllRecords(const IDBResourceIdentifier& transactionIdentifier, const IDBGetAllRecordsData& getAllRecordsData, IDBGetAllResult& result)
406{
407 LOG(IndexedDB, "MemoryIDBBackingStore::getAllRecords");
408
409 ASSERT(getAllRecordsData.objectStoreIdentifier);
410
411 if (!m_transactions.contains(transactionIdentifier))
412 return IDBError { UnknownError, "No backing store transaction found to get all records"_s };
413
414 auto* objectStore = m_objectStoresByIdentifier.get(getAllRecordsData.objectStoreIdentifier);
415 if (!objectStore)
416 return IDBError { UnknownError, "No backing store object store found"_s };
417
418 if (getAllRecordsData.indexIdentifier) {
419 auto* index = objectStore->indexForIdentifier(getAllRecordsData.indexIdentifier);
420 if (!index)
421 return IDBError { UnknownError, "No backing store index found"_s };
422
423 index->getAllRecords(getAllRecordsData.keyRangeData, getAllRecordsData.count, getAllRecordsData.getAllType, result);
424 } else
425 objectStore->getAllRecords(getAllRecordsData.keyRangeData, getAllRecordsData.count, getAllRecordsData.getAllType, result);
426
427 return IDBError { };
428}

465IDBError MemoryIDBBackingStore::generateKeyNumber(const IDBResourceIdentifier& transactionIdentifier, uint64_t objectStoreIdentifier, uint64_t& keyNumber)
466{
467 LOG(IndexedDB, "MemoryIDBBackingStore::generateKeyNumber");
468 ASSERT(objectStoreIdentifier);
469 ASSERT_UNUSED(transactionIdentifier, m_transactions.contains(transactionIdentifier));
470 ASSERT_UNUSED(transactionIdentifier, m_transactions.get(transactionIdentifier)->isWriting());
471
472 MemoryObjectStore* objectStore = m_objectStoresByIdentifier.get(objectStoreIdentifier);
473 RELEASE_ASSERT(objectStore);
474
475 keyNumber = objectStore->currentKeyGeneratorValue();
476 if (keyNumber > maxGeneratedKeyValue)
477 return IDBError { ConstraintError, "Cannot generate new key value over 2^53 for object store operation" };
478
479 objectStore->setKeyGeneratorValue(keyNumber + 1);
480
481 return IDBError { };
482}

499IDBError MemoryIDBBackingStore::maybeUpdateKeyGeneratorNumber(const IDBResourceIdentifier& transactionIdentifier, uint64_t objectStoreIdentifier, double newKeyNumber)
500{
501 LOG(IndexedDB, "MemoryIDBBackingStore::maybeUpdateKeyGeneratorNumber");
502 ASSERT(objectStoreIdentifier);
503 ASSERT_UNUSED(transactionIdentifier, m_transactions.contains(transactionIdentifier));
504 ASSERT_UNUSED(transactionIdentifier, m_transactions.get(transactionIdentifier)->isWriting());
505
506 MemoryObjectStore* objectStore = m_objectStoresByIdentifier.get(objectStoreIdentifier);
507 RELEASE_ASSERT(objectStore);
508
509 if (newKeyNumber < objectStore->currentKeyGeneratorValue())
510 return IDBError { };
511
512 if (newKeyNumber >= (double)maxGeneratedKeyValue) {
513 objectStore->setKeyGeneratorValue(maxGeneratedKeyValue + 1);
514 return IDBError { };
515 }
516
517 uint64_t newKeyInteger(newKeyNumber);
518 if (newKeyInteger <= uint64_t(newKeyNumber))
519 ++newKeyInteger;
520
521 ASSERT(newKeyInteger > uint64_t(newKeyNumber));
522
523 objectStore->setKeyGeneratorValue(newKeyInteger);
524
525 return IDBError { };
526}

528IDBError MemoryIDBBackingStore::openCursor(const IDBResourceIdentifier& transactionIdentifier, const IDBCursorInfo& info, IDBGetResult& outData)
529{
530 LOG(IndexedDB, "MemoryIDBBackingStore::openCursor");
531
532 ASSERT(!MemoryCursor::cursorForIdentifier(info.identifier()));
533
534 if (!m_transactions.contains(transactionIdentifier))
535 return IDBError { UnknownError, "No backing store transaction found in which to open a cursor"_s };
536
537 switch (info.cursorSource()) {
538 case IndexedDB::CursorSource::ObjectStore: {
539 auto* objectStore = m_objectStoresByIdentifier.get(info.sourceIdentifier());
540 if (!objectStore)
541 return IDBError { UnknownError, "No backing store object store found"_s };
542
543 MemoryCursor* cursor = objectStore->maybeOpenCursor(info);
544 if (!cursor)
545 return IDBError { UnknownError, "Could not create object store cursor in backing store"_s };
546
547 cursor->currentData(outData);
548 break;
549 }
550 case IndexedDB::CursorSource::Index:
551 auto* objectStore = m_objectStoresByIdentifier.get(info.objectStoreIdentifier());
552 if (!objectStore)
553 return IDBError { UnknownError, "No backing store object store found"_s };
554
555 auto* index = objectStore->indexForIdentifier(info.sourceIdentifier());
556 if (!index)
557 return IDBError { UnknownError, "No backing store index found"_s };
558
559 MemoryCursor* cursor = index->maybeOpenCursor(info);
560 if (!cursor)
561 return IDBError { UnknownError, "Could not create index cursor in backing store"_s };
562
563 cursor->currentData(outData);
564 break;
565 }
566
567 return IDBError { };
568}

586void MemoryIDBBackingStore::registerObjectStore(Ref<MemoryObjectStore>&& objectStore)
587{
588 ASSERT(!m_objectStoresByIdentifier.contains(objectStore->info().identifier()));
589 ASSERT(!m_objectStoresByName.contains(objectStore->info().name()));
590
