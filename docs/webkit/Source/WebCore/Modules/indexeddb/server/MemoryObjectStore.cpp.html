84IDBError MemoryObjectStore::createIndex(MemoryBackingStoreTransaction& transaction, const IDBIndexInfo& info)
85{
86 LOG(IndexedDB, "MemoryObjectStore::createIndex");
87
88 if (!m_writeTransaction || !m_writeTransaction->isVersionChange() || m_writeTransaction != &transaction)
89 return IDBError(ConstraintError);
90
91 ASSERT(!m_indexesByIdentifier.contains(info.identifier()));
92 auto index = MemoryIndex::create(info, *this);
93
94 // If there was an error populating the new index, then the current records in the object store violate its contraints
95 auto error = populateIndexWithExistingRecords(index.get());
96 if (!error.isNull())
97 return error;
98
99 m_info.addExistingIndex(info);
100 transaction.addNewIndex(index.get());
101 registerIndex(WTFMove(index));
102
103 return IDBError { };
104}

106void MemoryObjectStore::maybeRestoreDeletedIndex(Ref<MemoryIndex>&& index)
107{
108 LOG(IndexedDB, "MemoryObjectStore::maybeRestoreDeletedIndex");
109
110 if (m_info.hasIndex(index->info().name()))
111 return;
112
113 m_info.addExistingIndex(index->info());
114
115 ASSERT(!m_indexesByIdentifier.contains(index->info().identifier()));
116 index->clearIndexValueStore();
117 auto error = populateIndexWithExistingRecords(index.get());
118
119 // Since this index was installed in the object store before this transaction started,
120 // assuming things were in a valid state then, we should definitely be able to successfully
121 // repopulate the index with the object store's pre-transaction records.
122 ASSERT_UNUSED(error, error.isNull());
123
124 registerIndex(WTFMove(index));
125}

139IDBError MemoryObjectStore::deleteIndex(MemoryBackingStoreTransaction& transaction, uint64_t indexIdentifier)
140{
141 LOG(IndexedDB, "MemoryObjectStore::deleteIndex");
142
143 if (!m_writeTransaction || !m_writeTransaction->isVersionChange() || m_writeTransaction != &transaction)
144 return IDBError(ConstraintError);
145
146 auto index = takeIndexByIdentifier(indexIdentifier);
147 ASSERT(index);
148 if (!index)
149 return IDBError(ConstraintError);
150
151 m_info.deleteIndex(indexIdentifier);
152 transaction.indexDeleted(*index);
153
154 return IDBError { };
155}

157void MemoryObjectStore::deleteAllIndexes(MemoryBackingStoreTransaction& transaction)
158{
159 Vector<uint64_t> indexIdentifiers;
160 indexIdentifiers.reserveInitialCapacity(m_indexesByName.size());
161
162 for (auto& index : m_indexesByName.values())
163 indexIdentifiers.uncheckedAppend(index->info().identifier());
164
165 for (auto identifier : indexIdentifiers)
166 deleteIndex(transaction, identifier);
167}

199void MemoryObjectStore::deleteRecord(const IDBKeyData& key)
200{
201 LOG(IndexedDB, "MemoryObjectStore::deleteRecord");
202
203 ASSERT(m_writeTransaction);
204
205 if (!m_keyValueStore) {
206 m_writeTransaction->recordValueChanged(*this, key, nullptr);
207 return;
208 }
209
210 ASSERT(m_orderedKeys);
211
212 auto iterator = m_keyValueStore->find(key);
213 if (iterator == m_keyValueStore->end()) {
214 m_writeTransaction->recordValueChanged(*this, key, nullptr);
215 return;
216 }
217
218 m_writeTransaction->recordValueChanged(*this, key, &iterator->value);
219 m_keyValueStore->remove(iterator);
220 m_orderedKeys->erase(key);
221
222 updateIndexesForDeleteRecord(key);
223 updateCursorsForDeleteRecord(key);
224}

226void MemoryObjectStore::deleteRange(const IDBKeyRangeData& inputRange)
227{
228 LOG(IndexedDB, "MemoryObjectStore::deleteRange");
229
230 ASSERT(m_writeTransaction);
231
232 if (inputRange.isExactlyOneKey()) {
233 deleteRecord(inputRange.lowerKey);
234 return;
235 }
236
237 IDBKeyRangeData range = inputRange;
238 while (true) {
239 auto key = lowestKeyWithRecordInRange(range);
240 if (key.isNull())
241 break;
242
243 deleteRecord(key);
244
245 range.lowerKey = key;
246 range.lowerOpen = true;
247 }
248}

250IDBError MemoryObjectStore::addRecord(MemoryBackingStoreTransaction& transaction, const IDBKeyData& keyData, const IDBValue& value)
251{
252 IndexIDToIndexKeyMap indexKeys;
253 callOnIDBSerializationThreadAndWait([info = m_info.isolatedCopy(), keyData = keyData.isolatedCopy(), value = value.isolatedCopy(), &indexKeys](auto& globalObject) {
254 indexKeys = generateIndexKeyMapForValue(globalObject, info, keyData, value);
255 });
256 return addRecord(transaction, keyData, indexKeys, value);
257}

259IDBError MemoryObjectStore::addRecord(MemoryBackingStoreTransaction& transaction, const IDBKeyData& keyData, const IndexIDToIndexKeyMap& indexKeys, const IDBValue& value)
260{
261 LOG(IndexedDB, "MemoryObjectStore::addRecord");
262
263 ASSERT(m_writeTransaction);
264 ASSERT_UNUSED(transaction, m_writeTransaction == &transaction);
265 ASSERT(!m_keyValueStore || !m_keyValueStore->contains(keyData));
266 ASSERT(!m_orderedKeys || m_orderedKeys->find(keyData) == m_orderedKeys->end());
267
268 if (!m_keyValueStore) {
269 ASSERT(!m_orderedKeys);
270 m_keyValueStore = makeUnique<KeyValueMap>();
271 m_orderedKeys = makeUniqueWithoutFastMallocCheck<IDBKeyDataSet>();
272 }
273
274 auto mapResult = m_keyValueStore->set(keyData, value.data());
275 ASSERT(mapResult.isNewEntry);
276 auto listResult = m_orderedKeys->insert(keyData);
277 ASSERT(listResult.second);
278
279 // If there was an error indexing this addition, then revert it.
280 auto error = updateIndexesForPutRecord(keyData, indexKeys);
281 if (!error.isNull()) {
282 m_keyValueStore->remove(mapResult.iterator);
283 m_orderedKeys->erase(listResult.first);
284 } else
285 updateCursorsForPutRecord(listResult.first);
286
287 return error;
288}

308IDBError MemoryObjectStore::updateIndexesForPutRecord(const IDBKeyData& key, const IndexIDToIndexKeyMap& indexKeys)
309{
310 IDBError error;
311 Vector<std::pair<MemoryIndex*, IndexKey>> changedIndexRecords;
312
313 for (const auto& entry : indexKeys) {
314 auto* index = m_indexesByIdentifier.get(entry.key);
315 ASSERT(index);
316 if (!index) {
317 error = IDBError { InvalidStateError, "Missing index metadata" };
318 break;
319 }
320
321 error = index->putIndexKey(key, entry.value);
322 if (!error.isNull())
323 break;
324
325 changedIndexRecords.append(std::make_pair(index, entry.value));
326 }
327
328 // If any of the index puts failed, revert all of the ones that went through.
329 if (!error.isNull()) {
330 for (auto& record : changedIndexRecords)
331 record.first->removeRecord(key, record.second);
332 }
333
334 return error;
335}

337IDBError MemoryObjectStore::populateIndexWithExistingRecords(MemoryIndex& index)
338{
339 if (!m_keyValueStore)
340 return IDBError { };
341
342 for (const auto& iterator : *m_keyValueStore) {
343 std::optional<IndexKey> resultIndexKey;
344 callOnIDBSerializationThreadAndWait([key = iterator.key.isolatedCopy(), value = iterator.value, info = m_info.isolatedCopy(), indexInfo = index.info().isolatedCopy(), &resultIndexKey](auto& globalObject) {
345 auto jsValue = deserializeIDBValueToJSValue(globalObject, value);
346 if (jsValue.isUndefinedOrNull())
347 return;
348
349 IndexKey indexKey;
350 generateIndexKeyForValue(globalObject, indexInfo, jsValue, indexKey, info.keyPath(), key);
351 resultIndexKey = indexKey.isolatedCopy();
352 });
353
354 if (!resultIndexKey)
355 return IDBError { };
356
357 if (resultIndexKey->isNull())
358 continue;
359
360 IDBError error = index.putIndexKey(iterator.key, *resultIndexKey);
361 if (!error.isNull())
362 return error;
363 }
364
365 return IDBError { };
366}

368uint64_t MemoryObjectStore::countForKeyRange(uint64_t indexIdentifier, const IDBKeyRangeData& inRange) const
369{
370 LOG(IndexedDB, "MemoryObjectStore::countForKeyRange");
371
372 if (indexIdentifier) {
373 auto* index = m_indexesByIdentifier.get(indexIdentifier);
374 ASSERT(index);
375 return index->countForKeyRange(inRange);
376 }
377
378 if (!m_keyValueStore)
379 return 0;
380
381 uint64_t count = 0;
382 IDBKeyRangeData range = inRange;
383 while (true) {
384 auto key = lowestKeyWithRecordInRange(range);
385 if (key.isNull())
386 break;
387
388 ++count;
389 range.lowerKey = key;
390 range.lowerOpen = true;
391 }
392
393 return count;
394}

416void MemoryObjectStore::getAllRecords(const IDBKeyRangeData& keyRangeData, std::optional<uint32_t> count, IndexedDB::GetAllType type, IDBGetAllResult& result) const
417{
418 result = { type, m_info.keyPath() };
419
420 uint32_t targetCount;
421 if (count && count.value())
422 targetCount = count.value();
423 else
424 targetCount = std::numeric_limits<uint32_t>::max();
425
426 IDBKeyRangeData range = keyRangeData;
427 uint32_t currentCount = 0;
428 while (currentCount < targetCount) {
429 IDBKeyData key = lowestKeyWithRecordInRange(range);
430 if (key.isNull())
431 return;
432
433 range.lowerKey = key;
434 range.lowerOpen = true;
435 if (type == IndexedDB::GetAllType::Values)
436 result.addValue(valueForKey(key));
437 result.addKey(WTFMove(key));
438
439 ++currentCount;
440 }
441}

452IDBKeyData MemoryObjectStore::lowestKeyWithRecordInRange(const IDBKeyRangeData& keyRangeData) const
453{
454 if (!m_keyValueStore)
455 return { };
456
457 if (keyRangeData.isExactlyOneKey() && m_keyValueStore->contains(keyRangeData.lowerKey))
458 return keyRangeData.lowerKey;
459
460 ASSERT(m_orderedKeys);
461
462 auto lowestInRange = m_orderedKeys->lower_bound(keyRangeData.lowerKey);
463
464 if (lowestInRange == m_orderedKeys->end())
465 return { };
466
467 if (keyRangeData.lowerOpen && *lowestInRange == keyRangeData.lowerKey)
468 ++lowestInRange;
469
470 if (lowestInRange == m_orderedKeys->end())
471 return { };
472
473 if (!keyRangeData.upperKey.isNull()) {
474 if (lowestInRange->compare(keyRangeData.upperKey) > 0)
475 return { };
476 if (keyRangeData.upperOpen && *lowestInRange == keyRangeData.upperKey)
477 return { };
478 }
479
480 return *lowestInRange;
481}

483void MemoryObjectStore::registerIndex(Ref<MemoryIndex>&& index)
484{
485 ASSERT(!m_indexesByIdentifier.contains(index->info().identifier()));
486 ASSERT(!m_indexesByName.contains(index->info().name()));
487
488 auto identifier = index->info().identifier();
489 m_indexesByName.set(index->info().name(), &index.get());
490 m_indexesByIdentifier.set(identifier, WTFMove(index));
491}
492
493void MemoryObjectStore::unregisterIndex(MemoryIndex& index)
494{
495 ASSERT(m_indexesByIdentifier.contains(index.info().identifier()));
496 ASSERT(m_indexesByName.contains(index.info().name()));
497
498 m_indexesByName.remove(index.info().name());
499 m_indexesByIdentifier.remove(index.info().identifier());
500}

502MemoryObjectStoreCursor* MemoryObjectStore::maybeOpenCursor(const IDBCursorInfo& info)
503{
504 auto result = m_cursors.add(info.identifier(), nullptr);
505 if (!result.isNewEntry)
506 return nullptr;

508 result.iterator->value = makeUnique<MemoryObjectStoreCursor>(*this, info);
509 return result.iterator->value.get();

512void MemoryObjectStore::renameIndex(MemoryIndex& index, const String& newName)
513{
514 ASSERT(m_indexesByName.get(index.info().name()) == &index);
515 ASSERT(!m_indexesByName.contains(newName));
516 ASSERT(m_info.infoForExistingIndex(index.info().name()));
517
518 m_info.infoForExistingIndex(index.info().name())->rename(newName);
519 m_indexesByName.set(newName, m_indexesByName.take(index.info().name()));
520 index.rename(newName);
521}
