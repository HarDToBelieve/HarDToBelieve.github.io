38Ref<UniqueIDBDatabaseConnection> UniqueIDBDatabaseConnection::create(UniqueIDBDatabase& database, ServerOpenDBRequest& request)
39{
40 return adoptRef(*new UniqueIDBDatabaseConnection(database, request));
41}

43UniqueIDBDatabaseConnection::UniqueIDBDatabaseConnection(UniqueIDBDatabase& database, ServerOpenDBRequest& request)
44 : m_database(makeWeakPtr(database))
45 , m_server(database.server())
46 , m_connectionToClient(request.connection())
47 , m_openRequestIdentifier(request.requestData().requestIdentifier())
48{
49 m_server.registerDatabaseConnection(*this);
50 m_connectionToClient->registerDatabaseConnection(*this);
51}
52
53UniqueIDBDatabaseConnection::~UniqueIDBDatabaseConnection()
54{
55 ASSERT(m_transactionMap.isEmpty());
56
57 m_server.unregisterDatabaseConnection(*this);
58 m_connectionToClient->unregisterDatabaseConnection(*this);
59}

61bool UniqueIDBDatabaseConnection::hasNonFinishedTransactions() const
62{
63 return !m_transactionMap.isEmpty();
64}
65
66void UniqueIDBDatabaseConnection::abortTransactionWithoutCallback(UniqueIDBDatabaseTransaction& transaction)
67{
68 ASSERT(m_transactionMap.contains(transaction.info().identifier()));
69 ASSERT(m_database);
70
71 const auto& transactionIdentifier = transaction.info().identifier();
72 m_database->abortTransaction(transaction, [this, transactionIdentifier](const IDBError&) {
73 ASSERT(m_transactionMap.contains(transactionIdentifier));
74 m_transactionMap.remove(transactionIdentifier);
75 });
76}

115UniqueIDBDatabaseTransaction& UniqueIDBDatabaseConnection::createVersionChangeTransaction(uint64_t newVersion)
116{
117 LOG(IndexedDB, "UniqueIDBDatabaseConnection::createVersionChangeTransaction - %s - %" PRIu64, m_openRequestIdentifier.loggingString().utf8().data(), identifier());
118 ASSERT(!m_closePending);
119
120 IDBTransactionInfo info = IDBTransactionInfo::versionChange(m_connectionToClient, m_database->info(), newVersion);
121
122 Ref<UniqueIDBDatabaseTransaction> transaction = UniqueIDBDatabaseTransaction::create(*this, info);
123 m_transactionMap.set(transaction->info().identifier(), &transaction.get());
124
125 return transaction.get();
126}
127
128void UniqueIDBDatabaseConnection::establishTransaction(const IDBTransactionInfo& info)
129{
130 LOG(IndexedDB, "UniqueIDBDatabaseConnection::establishTransaction - %s - %" PRIu64, m_openRequestIdentifier.loggingString().utf8().data(), identifier());
131
132 ASSERT(info.mode() != IDBTransactionMode::Versionchange);
133
134 // No transactions should ever come from the client after the client has already told us
135 // the connection is closing.
136 ASSERT(!m_closePending);
137
138 Ref<UniqueIDBDatabaseTransaction> transaction = UniqueIDBDatabaseTransaction::create(*this, info);
139 m_transactionMap.set(transaction->info().identifier(), &transaction.get());
140
141 ASSERT(m_database);
142 m_database->enqueueTransaction(WTFMove(transaction));
143}

145void UniqueIDBDatabaseConnection::didAbortTransaction(UniqueIDBDatabaseTransaction& transaction, const IDBError& error)
146{
147 LOG(IndexedDB, "UniqueIDBDatabaseConnection::didAbortTransaction - %s - %" PRIu64, m_openRequestIdentifier.loggingString().utf8().data(), identifier());
148
149 auto transactionIdentifier = transaction.info().identifier();
150 auto takenTransaction = m_transactionMap.take(transactionIdentifier);
151 ASSERT(takenTransaction);
152
153 m_connectionToClient->didAbortTransaction(transactionIdentifier, error);
154}
155
156void UniqueIDBDatabaseConnection::didCommitTransaction(UniqueIDBDatabaseTransaction& transaction, const IDBError& error)
157{
158 LOG(IndexedDB, "UniqueIDBDatabaseConnection::didCommitTransaction - %s - %" PRIu64, m_openRequestIdentifier.loggingString().utf8().data(), identifier());
159
160 auto transactionIdentifier = transaction.info().identifier();
161
162 ASSERT(m_transactionMap.contains(transactionIdentifier) || !error.isNull());
163 m_transactionMap.remove(transactionIdentifier);
164
165 m_connectionToClient->didCommitTransaction(transactionIdentifier, error);
166}
