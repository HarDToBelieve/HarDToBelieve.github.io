37class IDBError;

40namespace IDBServer {
41
42class IDBConnectionToClient;
43class ServerOpenDBRequest;
44class UniqueIDBDatabase;
45class UniqueIDBDatabaseTransaction;

47class UniqueIDBDatabaseConnection : public RefCounted<UniqueIDBDatabaseConnection>, public Identified<UniqueIDBDatabaseConnection> {
48public:
49 static Ref<UniqueIDBDatabaseConnection> create(UniqueIDBDatabase&, ServerOpenDBRequest&);
50
51 ~UniqueIDBDatabaseConnection();
52
53 const IDBResourceIdentifier& openRequestIdentifier() { return m_openRequestIdentifier; }
54 UniqueIDBDatabase* database() { return m_database.get(); }
55 IDBServer* server() { return &m_server; }
56 IDBConnectionToClient& connectionToClient() { return m_connectionToClient; }
57
58 void connectionPendingCloseFromClient();
59 void connectionClosedFromClient();
60
61 bool closePending() const { return m_closePending; }
62
63 bool hasNonFinishedTransactions() const;
64
65 void fireVersionChangeEvent(const IDBResourceIdentifier& requestIdentifier, uint64_t requestedVersion);
66 UniqueIDBDatabaseTransaction& createVersionChangeTransaction(uint64_t newVersion);
67
68 void establishTransaction(const IDBTransactionInfo&);
69 void didAbortTransaction(UniqueIDBDatabaseTransaction&, const IDBError&);
70 void didCommitTransaction(UniqueIDBDatabaseTransaction&, const IDBError&);
71 void didCreateObjectStore(const IDBResultData&);
72 void didDeleteObjectStore(const IDBResultData&);
73 void didRenameObjectStore(const IDBResultData&);
74 void didClearObjectStore(const IDBResultData&);
75 void didCreateIndex(const IDBResultData&);
76 void didDeleteIndex(const IDBResultData&);
77 void didRenameIndex(const IDBResultData&);
78 void didFireVersionChangeEvent(const IDBResourceIdentifier& requestIdentifier, IndexedDB::ConnectionClosedOnBehalfOfServer);
79 void didFinishHandlingVersionChange(const IDBResourceIdentifier& transactionIdentifier);
80
81 void abortTransactionWithoutCallback(UniqueIDBDatabaseTransaction&);
82
83 bool connectionIsClosing() const;
84
85 void deleteTransaction(UniqueIDBDatabaseTransaction&);
86
87private:
88 UniqueIDBDatabaseConnection(UniqueIDBDatabase&, ServerOpenDBRequest&);
89
90 WeakPtr<UniqueIDBDatabase> m_database;
91 IDBServer& m_server;
92 Ref<IDBConnectionToClient> m_connectionToClient;
93 IDBResourceIdentifier m_openRequestIdentifier;
94
95 bool m_closePending { false };
96
97 HashMap<IDBResourceIdentifier, RefPtr<UniqueIDBDatabaseTransaction>> m_transactionMap;
98};
