26#pragma once
27
28#include "IDBKeyRangeData.h"
29#include "IDBResourceIdentifier.h"
30
31namespace WebCore {
32
33class IDBTransaction;
34
35namespace IndexedDB {
36enum class CursorDirection : uint8_t;
37enum class CursorSource : bool;
38enum class CursorType : bool;
39}
40
41struct IDBKeyRangeData;
42
43enum class CursorDuplicity {
44 Duplicates,
45 NoDuplicates,
46};
47
48class IDBCursorInfo {
49public:
50 static IDBCursorInfo objectStoreCursor(IDBTransaction&, uint64_t objectStoreIdentifier, const IDBKeyRangeData&, IndexedDB::CursorDirection, IndexedDB::CursorType);
51 static IDBCursorInfo indexCursor(IDBTransaction&, uint64_t objectStoreIdentifier, uint64_t indexIdentifier, const IDBKeyRangeData&, IndexedDB::CursorDirection, IndexedDB::CursorType);
52
53 IDBResourceIdentifier identifier() const { return m_cursorIdentifier; }
54 uint64_t sourceIdentifier() const { return m_sourceIdentifier; }
55 uint64_t objectStoreIdentifier() const { return m_objectStoreIdentifier; }
56
57 IndexedDB::CursorSource cursorSource() const { return m_source; }
58 IndexedDB::CursorDirection cursorDirection() const { return m_direction; }
59 IndexedDB::CursorType cursorType() const { return m_type; }
60 const IDBKeyRangeData& range() const { return m_range; }
61
62 bool isDirectionForward() const;
63 CursorDuplicity duplicity() const;
64
65 WEBCORE_EXPORT IDBCursorInfo isolatedCopy() const;
66
67 WEBCORE_EXPORT IDBCursorInfo();
68 template<class Encoder> void encode(Encoder&) const;
69 template<class Decoder> static WARN_UNUSED_RETURN bool decode(Decoder&, IDBCursorInfo&);
70
71#if !LOG_DISABLED
72 String loggingString() const;
73#endif
74
75private:
76 IDBCursorInfo(IDBTransaction&, uint64_t objectStoreIdentifier, const IDBKeyRangeData&, IndexedDB::CursorDirection, IndexedDB::CursorType);
77 IDBCursorInfo(IDBTransaction&, uint64_t objectStoreIdentifier, uint64_t indexIdentifier, const IDBKeyRangeData&, IndexedDB::CursorDirection, IndexedDB::CursorType);
78
79 IDBCursorInfo(const IDBResourceIdentifier&, const IDBResourceIdentifier&, uint64_t, uint64_t, const IDBKeyRangeData&, IndexedDB::CursorSource, IndexedDB::CursorDirection, IndexedDB::CursorType);
80
81 IDBResourceIdentifier m_cursorIdentifier;
82 IDBResourceIdentifier m_transactionIdentifier;
83 uint64_t m_objectStoreIdentifier { 0 };
84 uint64_t m_sourceIdentifier { 0 };
85
86 IDBKeyRangeData m_range;
87
88 IndexedDB::CursorSource m_source;
89 IndexedDB::CursorDirection m_direction;
90 IndexedDB::CursorType m_type;
91};
92
93template<class Encoder>
94void IDBCursorInfo::encode(Encoder& encoder) const
95{
96 encoder << m_cursorIdentifier << m_transactionIdentifier << m_objectStoreIdentifier << m_sourceIdentifier << m_range;
97
98 encoder << m_source;
99 encoder << m_direction;
100 encoder << m_type;
101}
102
103template<class Decoder>
104bool IDBCursorInfo::decode(Decoder& decoder, IDBCursorInfo& info)
105{
106 if (!decoder.decode(info.m_cursorIdentifier))
107 return false;
108
109 if (!decoder.decode(info.m_transactionIdentifier))
110 return false;
111
112 if (!decoder.decode(info.m_objectStoreIdentifier))
113 return false;
114
115 if (!decoder.decode(info.m_sourceIdentifier))
116 return false;
117
118 if (!decoder.decode(info.m_range))
119 return false;
120
121 if (!decoder.decode(info.m_source))
122 return false;
123
124 if (!decoder.decode(info.m_direction))
125 return false;
126
127 if (!decoder.decode(info.m_type))
128 return false;
129
130 return true;
131}
