26#pragma once
27
28#include "IDBKeyData.h"
29#include "IDBValue.h"
30
31namespace WebCore {
32
33struct IDBCursorRecord {
34 IDBKeyData key;
35 IDBKeyData primaryKey;
36 IDBValue value;
37
38 template<class Encoder> void encode(Encoder&) const;
39 template<class Decoder> static WARN_UNUSED_RETURN bool decode(Decoder&, IDBCursorRecord&);
40
41 IDBCursorRecord isolatedCopy() const;
42 size_t size() const { return key.size() + primaryKey.size() + value.size(); }
43};
44
45template<class Encoder>
46void IDBCursorRecord::encode(Encoder& encoder) const
47{
48 encoder << key << primaryKey << value;
49}
50
51template<class Decoder>
52bool IDBCursorRecord::decode(Decoder& decoder, IDBCursorRecord& record)
53{
54 if (!decoder.decode(record.key))
55 return false;
56
57 if (!decoder.decode(record.primaryKey))
58 return false;
59
60 if (!decoder.decode(record.value))
61 return false;
62
63 return true;
64}
65
66inline IDBCursorRecord IDBCursorRecord::isolatedCopy() const
67{
68 return { key.isolatedCopy(), primaryKey.isolatedCopy(), value.isolatedCopy() };
69}
70
71} // namespace WebCore
