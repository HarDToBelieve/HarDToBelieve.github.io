39IDBRequestData::IDBRequestData(const IDBClient::IDBConnectionProxy& connectionProxy, const IDBOpenDBRequest& request)
40 : m_serverConnectionIdentifier(connectionProxy.serverConnectionIdentifier())
41 , m_requestIdentifier(makeUnique<IDBResourceIdentifier>(connectionProxy, request))
42 , m_databaseIdentifier(request.databaseIdentifier())
43 , m_requestedVersion(request.version())
44 , m_requestType(request.requestType())
45{
46}
47
48IDBRequestData::IDBRequestData(IDBClient::TransactionOperation& operation)
49 : m_serverConnectionIdentifier(operation.transaction().database().connectionProxy().serverConnectionIdentifier())
50 , m_requestIdentifier(makeUnique<IDBResourceIdentifier>(operation.identifier()))
51 , m_transactionIdentifier(makeUnique<IDBResourceIdentifier>(operation.transactionIdentifier()))
52 , m_objectStoreIdentifier(operation.objectStoreIdentifier())
53 , m_indexIdentifier(operation.indexIdentifier())
54{
55 if (m_indexIdentifier)
56 m_indexRecordType = operation.indexRecordType();
57
58 if (operation.cursorIdentifier())
59 m_cursorIdentifier = makeUnique<IDBResourceIdentifier>(*operation.cursorIdentifier());
60}

62IDBRequestData::IDBRequestData(const IDBRequestData& other)
63 : m_serverConnectionIdentifier(other.m_serverConnectionIdentifier)
64 , m_objectStoreIdentifier(other.m_objectStoreIdentifier)
65 , m_indexIdentifier(other.m_indexIdentifier)
66 , m_indexRecordType(other.m_indexRecordType)
67 , m_databaseIdentifier(other.m_databaseIdentifier)
68 , m_requestedVersion(other.m_requestedVersion)
69 , m_requestType(other.m_requestType)
70{
71 if (other.m_requestIdentifier)
72 m_requestIdentifier = makeUnique<IDBResourceIdentifier>(*other.m_requestIdentifier);
73 if (other.m_transactionIdentifier)
74 m_transactionIdentifier = makeUnique<IDBResourceIdentifier>(*other.m_transactionIdentifier);
75 if (other.m_cursorIdentifier)
76 m_cursorIdentifier = makeUnique<IDBResourceIdentifier>(*other.m_cursorIdentifier);
77}

89
90void IDBRequestData::isolatedCopy(const IDBRequestData& source, IDBRequestData& destination)
91{
92 destination.m_serverConnectionIdentifier = source.m_serverConnectionIdentifier;
93 destination.m_objectStoreIdentifier = source.m_objectStoreIdentifier;
94 destination.m_indexIdentifier = source.m_indexIdentifier;
95 destination.m_indexRecordType = source.m_indexRecordType;
96 destination.m_requestedVersion = source.m_requestedVersion;
97 destination.m_requestType = source.m_requestType;
98
99 if (source.m_databaseIdentifier)
100 destination.m_databaseIdentifier = source.m_databaseIdentifier->isolatedCopy();
101
102 if (source.m_requestIdentifier)
103 destination.m_requestIdentifier = makeUnique<IDBResourceIdentifier>(*source.m_requestIdentifier);
104 if (source.m_transactionIdentifier)
105 destination.m_transactionIdentifier = makeUnique<IDBResourceIdentifier>(*source.m_transactionIdentifier);
106 if (source.m_cursorIdentifier)
107 destination.m_cursorIdentifier = makeUnique<IDBResourceIdentifier>(*source.m_cursorIdentifier);
108}
