31IndexKey::IndexKey()
32{
33}
34
35IndexKey::IndexKey(Vector<IDBKeyData>&& keys)
36{
37 m_keys.swap(keys);
38}
39
40IndexKey IndexKey::isolatedCopy() const
41{
42 Vector<IDBKeyData> keys;
43 keys.reserveInitialCapacity(m_keys.size());
44 for (auto& key : m_keys)
45 keys.uncheckedAppend(key.isolatedCopy());
46
47 return { WTFMove(keys) };
48}
49
50IDBKeyData IndexKey::asOneKey() const
51{
52 if (m_keys.isEmpty())
53 return { };
54
55 if (m_keys.size() == 1)
56 return m_keys[0];
57
58 IDBKeyData result;
59 result.setArrayValue(m_keys);
60 return result;
61}
62
63Vector<IDBKeyData> IndexKey::multiEntry() const
64{
65 Vector<IDBKeyData> multiEntry;
66 for (auto& key : m_keys) {
67 if (!key.isValid())
68 continue;
69
70 bool skip = false;
71 for (auto& otherKey : multiEntry) {
72 if (key == otherKey) {
73 skip = true;
74 break;
75 }
76 }
77
78 if (!skip)
79 multiEntry.append(key);
80 }
81
82 return multiEntry;
83}
