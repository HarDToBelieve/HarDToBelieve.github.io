50{
51#if PLATFORM(COCOA)

55 UNUSED_PARAM(document);
56 UNUSED_PARAM(value);
57 return false;
58#endif

62ExceptionOr<Ref<MediaRecorder>> MediaRecorder::create(Document& document, Ref<MediaStream>&& stream, Options&& options)
63{
64 auto* page = document.page();
65 if (!page)
66 return Exception { InvalidStateError };
67
68 if (!isTypeSupported(document, options.mimeType))
69 return Exception { NotSupportedError, "mimeType is not supported" };
70
71 auto recorder = adoptRef(*new MediaRecorder(document, WTFMove(stream), WTFMove(options)));
72 recorder->suspendIfNeeded();
73 return recorder;
74}

81ExceptionOr<std::unique_ptr<MediaRecorderPrivate>> MediaRecorder::createMediaRecorderPrivate(Document& document, MediaStreamPrivate& stream, const Options& options)
82{
83 auto* page = document.page();
84 if (!page)
85 return Exception { InvalidStateError };
86
87 if (m_customCreator)
88 return m_customCreator(stream, options);
89
90#if PLATFORM(COCOA)
91 auto result = page->mediaRecorderProvider().createMediaRecorderPrivate(stream, options);
92#else
93 std::unique_ptr<MediaRecorderPrivate> result;
94#endif
95 if (!result)
96 return Exception { NotSupportedError, "The MediaRecorder is unsupported on this platform"_s };
97 return result;
98}

100MediaRecorder::MediaRecorder(Document& document, Ref<MediaStream>&& stream, Options&& options)
101 : ActiveDOMObject(document)
102 , m_options(WTFMove(options))
103 , m_stream(WTFMove(stream))
104 , m_timeSliceTimer([this] { requestData(); })
105{
106 MediaRecorderPrivate::updateOptions(m_options);
107
108 m_tracks = WTF::map(m_stream->getTracks(), [] (auto&& track) -> Ref<MediaStreamTrackPrivate> {
109 return track->privateTrack();
110 });
111 m_stream->privateStream().addObserver(*this);
112}
113
114MediaRecorder::~MediaRecorder()
115{
116 m_stream->privateStream().removeObserver(*this);
117 stopRecordingInternal();
118}

131void MediaRecorder::suspend(ReasonForSuspension reason)
132{
133 if (reason != ReasonForSuspension::BackForwardCache)
134 return;
135
136 if (!m_isActive || state() == RecordingState::Inactive)
137 return;
138
139 stopRecordingInternal();
140
141 queueTaskToDispatchEvent(*this, TaskSource::Networking, MediaRecorderErrorEvent::create(eventNames().errorEvent, Exception { UnknownError, "MediaStream recording was interrupted"_s }));
142}

149ExceptionOr<void> MediaRecorder::startRecording(std::optional<unsigned> timeSlice)
150{
151 if (!m_isActive)
152 return Exception { InvalidStateError, "The MediaRecorder is not active"_s };
153
154 if (state() != RecordingState::Inactive)
155 return Exception { InvalidStateError, "The MediaRecorder's state must be inactive in order to start recording"_s };
156
157 ASSERT(!m_private);
158 auto result = createMediaRecorderPrivate(*document(), m_stream->privateStream(), m_options);
159
160 if (result.hasException())
161 return result.releaseException();
162
163 m_private = result.releaseReturnValue();
164 m_private->startRecording([this, pendingActivity = makePendingActivity(*this)](auto&& mimeTypeOrException, unsigned audioBitsPerSecond, unsigned videoBitsPerSecond) mutable {
165 if (!m_isActive)
166 return;
167
168 if (mimeTypeOrException.hasException()) {
169 stopRecordingInternal();
170 queueTaskKeepingObjectAlive(*this, TaskSource::Networking, [this, exception = mimeTypeOrException.releaseException()]() mutable {
171 if (!m_isActive)
172 return;
173 dispatchError(WTFMove(exception));
174 });
175 return;
176 }
177
178 queueTaskKeepingObjectAlive(*this, TaskSource::Networking, [this, mimeType = mimeTypeOrException.releaseReturnValue(), audioBitsPerSecond, videoBitsPerSecond]() mutable {
179 if (!m_isActive)
180 return;
181 m_options.mimeType = WTFMove(mimeType);
182 m_options.audioBitsPerSecond = audioBitsPerSecond;
183 m_options.videoBitsPerSecond = videoBitsPerSecond;
184
185 dispatchEvent(Event::create(eventNames().startEvent, Event::CanBubble::No, Event::IsCancelable::No));
186 });
187 });
188
189 for (auto& track : m_tracks)
190 track->addObserver(*this);
191
192 m_state = RecordingState::Recording;
193 m_timeSlice = timeSlice;
194 if (m_timeSlice)
195 m_timeSliceTimer.startOneShot(Seconds::fromMilliseconds(*m_timeSlice));
196 return { };
197}

199static inline Ref<BlobEvent> createDataAvailableEvent(ScriptExecutionContext* context, RefPtr<SharedBuffer>&& buffer, const String& mimeType, double timeCode)
200{
201 auto blob = buffer ? Blob::create(context, buffer->takeData(), mimeType) : Blob::create(context);
202 return BlobEvent::create(eventNames().dataavailableEvent, BlobEvent::Init { { false, false, false }, WTFMove(blob), timeCode }, BlobEvent::IsTrusted::Yes);
203}

205void MediaRecorder::stopRecording()
206{
207 if (state() == RecordingState::Inactive)
208 return;
209
210 stopRecordingInternal();
211 fetchData([this](auto&& buffer, auto& mimeType, auto timeCode) {
212 if (!m_isActive)
213 return;
214
215 dispatchEvent(createDataAvailableEvent(scriptExecutionContext(), WTFMove(buffer), mimeType, timeCode));
216
217 if (!m_isActive)
218 return;
219 dispatchEvent(Event::create(eventNames().stopEvent, Event::CanBubble::No, Event::IsCancelable::No));
220 }, TakePrivateRecorder::Yes);
221 return;
222}
223
224ExceptionOr<void> MediaRecorder::requestData()
225{
226 if (state() == RecordingState::Inactive)
227 return Exception { InvalidStateError, "The MediaRecorder's state cannot be inactive"_s };
228
229 if (m_timeSliceTimer.isActive())
230 m_timeSliceTimer.stop();
231
232 fetchData([this](auto&& buffer, auto& mimeType, auto timeCode) {
233 if (!m_isActive)
234 return;
235
236 dispatchEvent(createDataAvailableEvent(scriptExecutionContext(), WTFMove(buffer), mimeType, timeCode));
237
238 if (m_isActive && m_timeSlice)
239 m_timeSliceTimer.startOneShot(Seconds::fromMilliseconds(*m_timeSlice));
240 }, TakePrivateRecorder::No);
241 return { };
242}

244ExceptionOr<void> MediaRecorder::pauseRecording()
245{
246 if (state() == RecordingState::Inactive)
247 return Exception { InvalidStateError, "The MediaRecorder's state cannot be inactive"_s };
248
249 if (state() == RecordingState::Paused)
250 return { };
251
252 m_state = RecordingState::Paused;
253 m_private->pause([this, pendingActivity = makePendingActivity(*this)]() {
254 if (!m_isActive)
255 return;
256 queueTaskKeepingObjectAlive(*this, TaskSource::Networking, [this]() mutable {
257 if (!m_isActive)
258 return;
259 dispatchEvent(Event::create(eventNames().pauseEvent, Event::CanBubble::No, Event::IsCancelable::No));
260 });
261 });
262 return { };
263}

281 });
282 });

286void MediaRecorder::fetchData(FetchDataCallback&& callback, TakePrivateRecorder takeRecorder)
287{
288 auto& privateRecorder = *m_private;
289
290 std::unique_ptr<MediaRecorderPrivate> takenPrivateRecorder;
291 if (takeRecorder == TakePrivateRecorder::Yes)
292 takenPrivateRecorder = WTFMove(m_private);
293
294 auto fetchDataCallback = [this, privateRecorder = WTFMove(takenPrivateRecorder), callback = WTFMove(callback)](auto&& buffer, auto& mimeType, auto timeCode) mutable {
295 queueTaskKeepingObjectAlive(*this, TaskSource::Networking, [buffer = WTFMove(buffer), mimeType, timeCode, callback = WTFMove(callback)]() mutable {
296 callback(WTFMove(buffer), mimeType, timeCode);
297 });
298 };
299
300 if (m_isFetchingData) {
301 m_pendingFetchDataTasks.append(WTFMove(fetchDataCallback));
302 return;
303 }
304
305 m_isFetchingData = true;
306 privateRecorder.fetchData([this, pendingActivity = makePendingActivity(*this), callback = WTFMove(fetchDataCallback)](auto&& buffer, auto& mimeType, auto timeCode) mutable {
307 m_isFetchingData = false;
308 callback(WTFMove(buffer), mimeType, timeCode);
309 for (auto& task : std::exchange(m_pendingFetchDataTasks, { }))
310 task({ }, mimeType, timeCode);
311 });
312}

314void MediaRecorder::stopRecordingInternal(CompletionHandler<void()>&& completionHandler)
315{
316 if (state() == RecordingState::Inactive) {
317 completionHandler();
318 return;
319 }
320
321 for (auto& track : m_tracks)
322 track->removeObserver(*this);
323
324 m_state = RecordingState::Inactive;
325 m_private->stop(WTFMove(completionHandler));
326}

328void MediaRecorder::handleTrackChange()
329{
330 queueTaskKeepingObjectAlive(*this, TaskSource::Networking, [this] {
331 stopRecordingInternal([this, pendingActivity = makePendingActivity(*this)] {
332 queueTaskKeepingObjectAlive(*this, TaskSource::Networking, [this] {
333 if (!m_isActive)
334 return;
335 dispatchError(Exception { InvalidModificationError, "Track cannot be added to or removed from the MediaStream while recording"_s });
336
337 if (!m_isActive)
338 return;
339 dispatchEvent(createDataAvailableEvent(scriptExecutionContext(), { }, { }, 0));
340
341 if (!m_isActive)
342 return;
343 dispatchEvent(Event::create(eventNames().stopEvent, Event::CanBubble::No, Event::IsCancelable::No));
344 });
345 });
346 });
347}

349void MediaRecorder::dispatchError(Exception&& exception)

356void MediaRecorder::trackEnded(MediaStreamTrackPrivate&)
357{
358 auto position = m_tracks.findMatching([](auto& track) {
359 return !track->ended();
360 });
361 if (position != notFound)
362 return;
363
364 queueTaskKeepingObjectAlive(*this, TaskSource::Networking, [this] {
365 stopRecordingInternal([this, pendingActivity = makePendingActivity(*this)] {
366 queueTaskKeepingObjectAlive(*this, TaskSource::Networking, [this] {
367 if (!m_isActive)
368 return;
369 dispatchEvent(createDataAvailableEvent(scriptExecutionContext(), { }, { }, 0));
370
371 if (!m_isActive)
372 return;
373 dispatchEvent(Event::create(eventNames().stopEvent, Event::CanBubble::No, Event::IsCancelable::No));
374 });
375 });
376 });
377}
