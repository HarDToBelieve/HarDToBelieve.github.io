26#pragma once
27
28#include "MediaImage.h"
29
30#if ENABLE(MEDIA_SESSION)
31
32namespace WebCore {
33
34struct MediaMetadataInit {
35 String title;
36 String artist;
37 String album;
38#if ENABLE(MEDIA_SESSION_PLAYLIST)
39 String trackIdentifier;
40#endif
41 Vector<MediaImage> artwork;
42
43 template<class Encoder> void encode(Encoder&) const;
44 template<class Decoder> static std::optional<MediaMetadataInit> decode(Decoder&);
45};
46
47template<class Encoder> inline void MediaMetadataInit::encode(Encoder& encoder) const
48{
49 encoder << title << artist << album
50#if ENABLE(MEDIA_SESSION_PLAYLIST)
51 << trackIdentifier
52#endif
53 << artwork;
54}
55
56template<class Decoder> inline std::optional<MediaMetadataInit> MediaMetadataInit::decode(Decoder& decoder)
57{
58 std::optional<String> title;
59 decoder >> title;
60 if (!title)
61 return { };
62
63 std::optional<String> artist;
64 decoder >> artist;
65 if (!artist)
66 return { };
67
68 std::optional<String> album;
69 decoder >> album;
70 if (!album)
71 return { };
72
73#if ENABLE(MEDIA_SESSION_PLAYLIST)
74 std::optional<String> trackIdentifier;
75 decoder >> trackIdentifier;
76 if (!trackIdentifier)
77 return { };
78#endif
79
80 std::optional<Vector<MediaImage>> artwork;
81 decoder >> artwork;
82 if (!artwork)
83 return { };
84
85 return { {
86 WTFMove(*title),
87 WTFMove(*artist),
88 WTFMove(*album),
89#if ENABLE(MEDIA_SESSION_PLAYLIST)
90 WTFMove(*trackIdentifier),
91#endif
92 WTFMove(*artwork)
93 } };
94}
95
96}
97
98#endif // ENABLE(MEDIA_SESSION)
