57static PlatformMediaSession::RemoteControlCommandType platformCommandForMediaSessionAction(MediaSessionAction action)
58{
59 static const auto commandMap = makeNeverDestroyed([] {
60 using ActionToCommandMap = HashMap<MediaSessionAction, PlatformMediaSession::RemoteControlCommandType, WTF::IntHash<MediaSessionAction>, WTF::StrongEnumHashTraits<MediaSessionAction>>;
61
62 return ActionToCommandMap {
63 { MediaSessionAction::Play, PlatformMediaSession::PlayCommand },
64 { MediaSessionAction::Pause, PlatformMediaSession::PauseCommand },
65 { MediaSessionAction::Seekforward, PlatformMediaSession::SkipForwardCommand },
66 { MediaSessionAction::Seekbackward, PlatformMediaSession::SkipBackwardCommand },
67 { MediaSessionAction::Previoustrack, PlatformMediaSession::NextTrackCommand },
68 { MediaSessionAction::Nexttrack, PlatformMediaSession::PreviousTrackCommand },
69 { MediaSessionAction::Stop, PlatformMediaSession::StopCommand },
70 { MediaSessionAction::Seekto, PlatformMediaSession::SeekToPlaybackPositionCommand },
71 { MediaSessionAction::Skipad, PlatformMediaSession::NextTrackCommand },
72 };
73 }());
74
75 auto it = commandMap.get().find(action);
76 if (it != commandMap.get().end())
77 return it->value;
78
79 return PlatformMediaSession::NoCommand;
80}

82static std::optional<std::pair<PlatformMediaSession::RemoteControlCommandType, PlatformMediaSession::RemoteCommandArgument>> platformCommandForMediaSessionAction(const MediaSessionActionDetails& actionDetails)
83{
84 PlatformMediaSession::RemoteControlCommandType command = PlatformMediaSession::NoCommand;
85 PlatformMediaSession::RemoteCommandArgument argument;
86
87 switch (actionDetails.action) {
88 case MediaSessionAction::Play:
89 command = PlatformMediaSession::PlayCommand;
90 break;
91 case MediaSessionAction::Pause:
92 command = PlatformMediaSession::PauseCommand;
93 break;
94 case MediaSessionAction::Seekbackward:
95 command = PlatformMediaSession::SkipBackwardCommand;
96 argument.time = actionDetails.seekOffset;
97 break;
98 case MediaSessionAction::Seekforward:
99 command = PlatformMediaSession::SkipForwardCommand;
100 argument.time = actionDetails.seekOffset;
101 break;
102 case MediaSessionAction::Previoustrack:
103 command = PlatformMediaSession::PreviousTrackCommand;
104 break;
105 case MediaSessionAction::Nexttrack:
106 command = PlatformMediaSession::NextTrackCommand;
107 break;
108 case MediaSessionAction::Skipad:
109 // Not supported at present.
110 break;
111 case MediaSessionAction::Stop:
112 command = PlatformMediaSession::StopCommand;
113 break;
114 case MediaSessionAction::Seekto:
115 command = PlatformMediaSession::SeekToPlaybackPositionCommand;
116 argument.time = actionDetails.seekTime;
117 argument.fastSeek = actionDetails.fastSeek;
118 break;
119 case MediaSessionAction::Settrack:
120 // Not supported at present.
121 break;
122 }
123 if (command == PlatformMediaSession::NoCommand)
124 return { };
125
126 return std::make_pair(command, argument);
127}

136MediaSession::MediaSession(Navigator& navigator)
137 : ActiveDOMObject(navigator.scriptExecutionContext())
138 , m_navigator(makeWeakPtr(navigator))
139#if ENABLE(MEDIA_SESSION_COORDINATOR)
140 , m_coordinator(MediaSessionCoordinator::create(navigator.scriptExecutionContext()))
141#endif
142{
143 m_logger = makeRefPtr(Document::sharedLogger());
144 m_logIdentifier = nextLogIdentifier();
145
146#if ENABLE(MEDIA_SESSION_COORDINATOR)
147 auto* frame = navigator.frame();
148 auto* page = frame ? frame->page() : nullptr;
149 if (page && page->mediaSessionCoordinator())
150 m_coordinator->setMediaSessionCoordinatorPrivate(*page->mediaSessionCoordinator());
151 m_coordinator->setMediaSession(this);
152#endif
153
154 ALWAYS_LOG(LOGIDENTIFIER);
155}

201ExceptionOr<void> MediaSession::setPlaylist(ScriptExecutionContext& context, Vector<RefPtr<MediaMetadata>>&& playlist)
202{
203 ALWAYS_LOG(LOGIDENTIFIER);
204
205 Vector<Ref<MediaMetadata>> resolvedPlaylist;
206 resolvedPlaylist.reserveInitialCapacity(playlist.size());
207
208 for (auto& entry : playlist) {
209 auto resolvedEntry = MediaMetadata::create(context, { entry->metadata() });
210 if (resolvedEntry.hasException())
211 return resolvedEntry.releaseException();
212
213 resolvedPlaylist.uncheckedAppend(resolvedEntry.releaseReturnValue());
214 }
215
216 m_playlist = WTFMove(resolvedPlaylist);
217
218 return { };
219}

222void MediaSession::setPlaybackState(MediaSessionPlaybackState state)
223{
224 if (m_playbackState == state)
225 return;
226
227 ALWAYS_LOG(LOGIDENTIFIER, state);
228
229 auto currentPosition = this->currentPosition();
230 if (m_positionState && currentPosition) {
231 m_positionState->position = *currentPosition;
232 m_timeAtLastPositionUpdate = MonotonicTime::now();
233 }
234 m_playbackState = state;
235 notifyPlaybackStateObservers();
236}

238void MediaSession::setActionHandler(MediaSessionAction action, RefPtr<MediaSessionActionHandler>&& handler)
239{
240 if (handler) {
241 ALWAYS_LOG(LOGIDENTIFIER, "adding ", action);
242 m_actionHandlers.set(action, handler);
243 auto platformCommand = platformCommandForMediaSessionAction(action);
244 if (platformCommand != PlatformMediaSession::NoCommand)
245 PlatformMediaSessionManager::sharedManager().addSupportedCommand(platformCommand);
246 } else {
247 if (m_actionHandlers.contains(action)) {
248 ALWAYS_LOG(LOGIDENTIFIER, "removing ", action);
249 m_actionHandlers.remove(action);
250 }
251 PlatformMediaSessionManager::sharedManager().removeSupportedCommand(platformCommandForMediaSessionAction(action));
252 }
253
254 notifyActionHandlerObservers();
255}

257void MediaSession::callActionHandler(const MediaSessionActionDetails& actionDetails, DOMPromiseDeferred<void>&& promise)
258{
259 ALWAYS_LOG(LOGIDENTIFIER);
260
261 if (!callActionHandler(actionDetails, TriggerGestureIndicator::No)) {
262 promise.reject(InvalidStateError);
263 return;
264 }
265
266 promise.resolve();
267}

269bool MediaSession::callActionHandler(const MediaSessionActionDetails& actionDetails, TriggerGestureIndicator triggerGestureIndicator)
270{
271 if (auto handler = m_actionHandlers.get(actionDetails.action)) {
272 std::optional<UserGestureIndicator> maybeGestureIndicator;
273 if (triggerGestureIndicator == TriggerGestureIndicator::Yes)
274 maybeGestureIndicator.emplace(ProcessingUserGesture, document());
275 handler->handleEvent(actionDetails);
276 return true;
277 }
278 auto element = activeMediaElement();
279 if (!element)
280 return false;
281
282 auto platformCommand = platformCommandForMediaSessionAction(actionDetails);
283 if (!platformCommand)
284 return false;
285 element->didReceiveRemoteControlCommand(platformCommand->first, platformCommand->second);
286 return true;
287}

289ExceptionOr<void> MediaSession::setPositionState(std::optional<MediaPositionState>&& state)
290{
291 if (state)
292 ALWAYS_LOG(LOGIDENTIFIER, state.value());
293 else
294 ALWAYS_LOG(LOGIDENTIFIER, "{ }");
295
296 if (!state) {
297 m_positionState = std::nullopt;
298 notifyPositionStateObservers();
299 return { };
300 }
301
302 if (!(state->duration >= 0
303 && state->position >= 0
304 && state->position <= state->duration
305 && std::isfinite(state->playbackRate)
306 && state->playbackRate))
307 return Exception { TypeError };
308
309 m_positionState = WTFMove(state);
310 m_lastReportedPosition = m_positionState->position;
311 m_timeAtLastPositionUpdate = MonotonicTime::now();
312 notifyPositionStateObservers();
313
314 return { };
315}
316
317std::optional<double> MediaSession::currentPosition() const
318{
319 if (!m_positionState || !m_lastReportedPosition)
320 return std::nullopt;
321
322 auto actualPlaybackRate = m_playbackState == MediaSessionPlaybackState::Playing ? m_positionState->playbackRate : 0;
323
324 auto elapsedTime = (MonotonicTime::now() - m_timeAtLastPositionUpdate) * actualPlaybackRate;
325
326 return std::max(0., std::min(*m_lastReportedPosition + elapsedTime.value(), m_positionState->duration));
327}
