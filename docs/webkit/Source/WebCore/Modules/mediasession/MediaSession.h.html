53class MediaSession : public RefCounted<MediaSession>, public ActiveDOMObject, public CanMakeWeakPtr<MediaSession> {
54 WTF_MAKE_FAST_ALLOCATED;
55public:
56 static Ref<MediaSession> create(Navigator&);
57 ~MediaSession();
58
59 MediaMetadata* metadata() const { return m_metadata.get(); };
60 void setMetadata(RefPtr<MediaMetadata>&&);
61 void metadataUpdated();
62
63 MediaSessionPlaybackState playbackState() const { return m_playbackState; };
64 void setPlaybackState(MediaSessionPlaybackState);
65
66 void setActionHandler(MediaSessionAction, RefPtr<MediaSessionActionHandler>&&);
67
68 void callActionHandler(const MediaSessionActionDetails&, DOMPromiseDeferred<void>&&);
69
70 ExceptionOr<void> setPositionState(std::optional<MediaPositionState>&&);
71 std::optional<MediaPositionState> positionState() const { return m_positionState; }
72
73 WEBCORE_EXPORT std::optional<double> currentPosition() const;
74
75 Document* document() const;
76
77#if ENABLE(MEDIA_SESSION_COORDINATOR)
78 MediaSessionReadyState readyState() const { return m_readyState; };
79 void setReadyState(MediaSessionReadyState);
80
81 MediaSessionCoordinator& coordinator() const { return m_coordinator.get(); }
82#endif
83
84#if ENABLE(MEDIA_SESSION_PLAYLIST)
85 const Vector<Ref<MediaMetadata>>& playlist() const { return m_playlist; }
86 ExceptionOr<void> setPlaylist(ScriptExecutionContext&, Vector<RefPtr<MediaMetadata>>&&);
87#endif
88
89 bool hasActiveActionHandlers() const { return !m_actionHandlers.isEmpty(); }
90 enum class TriggerGestureIndicator {
91 No,
92 Yes,
93 };
94 WEBCORE_EXPORT bool callActionHandler(const MediaSessionActionDetails&, TriggerGestureIndicator = TriggerGestureIndicator::Yes);
95
96 const Logger& logger() const { return *m_logger.get(); }
97
98 class Observer : public CanMakeWeakPtr<Observer> {
99 public:
100 virtual ~Observer() = default;
101
102 virtual void metadataChanged(const RefPtr<MediaMetadata>&) { }
103 virtual void positionStateChanged(const std::optional<MediaPositionState>&) { }
104 virtual void playbackStateChanged(MediaSessionPlaybackState) { }
105 virtual void actionHandlersChanged() { }
106
107#if ENABLE(MEDIA_SESSION_COORDINATOR)
108 virtual void readyStateChanged(MediaSessionReadyState) { }
109#endif
110 };
111 void addObserver(Observer&);
112 void removeObserver(Observer&);
113
114private:
115 explicit MediaSession(Navigator&);
116
117 const void* logIdentifier() const { return m_logIdentifier; }
118
119 void forEachObserver(const Function<void(Observer&)>&);
120 void notifyMetadataObservers();
121 void notifyPositionStateObservers();
122 void notifyPlaybackStateObservers();
123 void notifyActionHandlerObservers();
124 void notifyReadyStateObservers();
125
126 // ActiveDOMObject
127 const char* activeDOMObjectName() const final { return "MediaSession"; }
128 void suspend(ReasonForSuspension) final;
129 void stop() final;
130
131 RefPtr<HTMLMediaElement> activeMediaElement() const;
132
133 WeakPtr<Navigator> m_navigator;
134 RefPtr<MediaMetadata> m_metadata;
135 MediaSessionPlaybackState m_playbackState { MediaSessionPlaybackState::None };
136 std::optional<MediaPositionState> m_positionState;
137 std::optional<double> m_lastReportedPosition;
138 MonotonicTime m_timeAtLastPositionUpdate;
139 HashMap<MediaSessionAction, RefPtr<MediaSessionActionHandler>, WTF::IntHash<MediaSessionAction>, WTF::StrongEnumHashTraits<MediaSessionAction>> m_actionHandlers;
140 RefPtr<const Logger> m_logger;
141 const void* m_logIdentifier;
142
143 WeakHashSet<Observer> m_observers;
144
145#if ENABLE(MEDIA_SESSION_COORDINATOR)
146 MediaSessionReadyState m_readyState { MediaSessionReadyState::Havenothing };
147 const Ref<MediaSessionCoordinator> m_coordinator;
148#endif
149
150#if ENABLE(MEDIA_SESSION_PLAYLIST)
151 Vector<Ref<MediaMetadata>> m_playlist;
152#endif
153};
