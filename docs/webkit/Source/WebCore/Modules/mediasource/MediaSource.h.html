46class ContentType;

54 , public ActiveDOMObject

61 WTF_MAKE_ISO_ALLOCATED(MediaSource);
62public:

65
66 static Ref<MediaSource> create(ScriptExecutionContext&);
67 virtual ~MediaSource();
68
69 void addedToRegistry();
70 void removedFromRegistry();
71 void openIfInEndedState();
72 bool isOpen() const;
73 bool isClosed() const;
74 bool isEnded() const;
75 void sourceBufferDidChangeActiveState(SourceBuffer&, bool);
76 void sourceBufferDidChangeBufferedDirty(SourceBuffer&, bool);
77
78 enum class EndOfStreamError { Network, Decode };
79 void streamEndedWithError(std::optional<EndOfStreamError>);
80
81 MediaTime duration() const final;
82 std::unique_ptr<PlatformTimeRanges> buffered() const final;
83
84 bool attachToElement(HTMLMediaElement&);
85 void detachFromElement(HTMLMediaElement&);
86#if USE(GSTREAMER)
87 void monitorSourceBuffers() final;
88#else
89 void monitorSourceBuffers();
90#endif
91 bool isSeeking() const { return m_pendingSeekTime.isValid(); }
92 Ref<TimeRanges> seekable();
93 ExceptionOr<void> setLiveSeekableRange(double start, double end);
94 ExceptionOr<void> clearLiveSeekableRange();
95
96 ExceptionOr<void> setDuration(double);
97 ExceptionOr<void> setDurationInternal(const MediaTime&);
98 MediaTime currentTime() const;
99
100 enum class ReadyState { Closed, Open, Ended };
101 ReadyState readyState() const { return m_readyState; }
102 ExceptionOr<void> endOfStream(std::optional<EndOfStreamError>);
103
104 HTMLMediaElement* mediaElement() const { return m_mediaElement.get(); }
105
106 SourceBufferList* sourceBuffers() { return m_sourceBuffers.get(); }
107 SourceBufferList* activeSourceBuffers() { return m_activeSourceBuffers.get(); }
108 ExceptionOr<Ref<SourceBuffer>> addSourceBuffer(const String& type);
109 ExceptionOr<void> removeSourceBuffer(SourceBuffer&);
110 static bool isTypeSupported(ScriptExecutionContext&, const String& type);
111
112 ScriptExecutionContext* scriptExecutionContext() const final;
113
114 using RefCounted::ref;
115 using RefCounted::deref;
116
117 static const MediaTime& currentTimeFudgeFactor();
118 static bool contentTypeShouldGenerateTimestamps(const ContentType&);
119
120#if !RELEASE_LOG_DISABLED
121 const Logger& logger() const final { return m_logger.get(); }
122 const void* logIdentifier() const final { return m_logIdentifier; }
123 const char* logClassName() const final { return "MediaSource"; }
124 WTFLogChannel& logChannel() const final;
125 void setLogIdentifier(const void*) final;
126#endif

128 void failedToCreateRenderer(RendererType) final;

133 // ActiveDOMObject.

136 bool virtualHasPendingActivity() const final;

138
139 void setPrivateAndOpen(Ref<MediaSourcePrivate>&&) final;
140 void seekToTime(const MediaTime&) final;
141
142 void refEventTarget() final { ref(); }
143 void derefEventTarget() final { deref(); }
144 EventTargetInterface eventTargetInterface() const final;
145
146 URLRegistry& registry() const final;

148 void setReadyState(ReadyState);
149 void onReadyStateChange(ReadyState oldState, ReadyState newState);

151 Vector<PlatformTimeRanges> activeRanges() const;

154 void scheduleEvent(const AtomString& eventName);

156 bool hasBufferedTime(const MediaTime&);
157 bool hasCurrentTime();
158 bool hasFutureTime();
159
160 void regenerateActiveSourceBuffers();
161 void updateBufferedIfNeeded();
162
163 void completeSeek();
164

173 MediaTime m_duration;
174 MediaTime m_pendingSeekTime;
175 ReadyState m_readyState { ReadyState::Closed };
176#if !RELEASE_LOG_DISABLED
177 Ref<const Logger> m_logger;
178 const void* m_logIdentifier { nullptr };
179#endif
180 uint64_t m_associatedRegistryCount { 0 };
