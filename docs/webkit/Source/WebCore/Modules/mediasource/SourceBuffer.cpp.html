73SourceBuffer::SourceBuffer(Ref<SourceBufferPrivate>&& sourceBufferPrivate, MediaSource* source)
74 : ActiveDOMObject(source->scriptExecutionContext())
75 , m_private(WTFMove(sourceBufferPrivate))
76 , m_source(source)
77 , m_appendBufferTimer(*this, &SourceBuffer::appendBufferTimerFired)
78 , m_appendWindowStart(MediaTime::zeroTime())
79 , m_appendWindowEnd(MediaTime::positiveInfiniteTime())
80 , m_appendState(WaitingForSegment)
81 , m_timeOfBufferingMonitor(MonotonicTime::now())
82 , m_pendingRemoveStart(MediaTime::invalidTime())
83 , m_pendingRemoveEnd(MediaTime::invalidTime())
84 , m_removeTimer(*this, &SourceBuffer::removeTimerFired)
85#if !RELEASE_LOG_DISABLED
86 , m_logger(m_private->sourceBufferLogger())
87 , m_logIdentifier(m_private->sourceBufferLogIdentifier())
88#endif
89{
90 ASSERT(m_source);
91 ALWAYS_LOG(LOGIDENTIFIER);
92
93 m_private->setClient(this);
94 m_private->setIsAttached(true);
95}

124ExceptionOr<void> SourceBuffer::setTimestampOffset(double offset)
125{
126 // Section 3.1 timestampOffset attribute setter steps.
127 // https://dvcs.w3.org/hg/html-media/raw-file/default/media-source/media-source.html#attributes-1
128 // 1. Let new timestamp offset equal the new value being assigned to this attribute.
129 // 2. If this object has been removed from the sourceBuffers attribute of the parent media source, then throw an
130 // InvalidStateError exception and abort these steps.
131 // 3. If the updating attribute equals true, then throw an InvalidStateError exception and abort these steps.
132 if (isRemoved() || m_updating)
133 return Exception { InvalidStateError };
134
135 // 4. If the readyState attribute of the parent media source is in the "ended" state then run the following steps:
136 // 4.1 Set the readyState attribute of the parent media source to "open"
137 // 4.2 Queue a task to fire a simple event named sourceopen at the parent media source.
138 m_source->openIfInEndedState();
139
140 // 5. If the append state equals PARSING_MEDIA_SEGMENT, then throw an InvalidStateError and abort these steps.
141 if (m_appendState == ParsingMediaSegment)
142 return Exception { InvalidStateError };
143
144 MediaTime newTimestampOffset = MediaTime::createWithDouble(offset);
145
146 // 6. If the mode attribute equals "sequence", then set the group start timestamp to new timestamp offset.
147 if (m_mode == AppendMode::Sequence)
148 m_private->setGroupStartTimestamp(newTimestampOffset);
149
150 // 7. Update the attribute to the new value.
151 m_private->setTimestampOffset(newTimestampOffset);
152
153 m_private->resetTimestampOffsetInTrackBuffers();
154
155 return { };
156}

163ExceptionOr<void> SourceBuffer::setAppendWindowStart(double newValue)
164{
165 // Section 3.1 appendWindowStart attribute setter steps.
166 // W3C Editor's Draft 16 September 2016
167 // https://rawgit.com/w3c/media-source/45627646344eea0170dd1cbc5a3d508ca751abb8/media-source-respec.html#dom-sourcebuffer-appendwindowstart
168 // 1. If this object has been removed from the sourceBuffers attribute of the parent media source,
169 // then throw an InvalidStateError exception and abort these steps.

171 if (isRemoved() || m_updating)

173
174 // 3. If the new value is less than 0 or greater than or equal to appendWindowEnd then
175 // throw an TypeError exception and abort these steps.
176 if (newValue < 0 || newValue >= m_appendWindowEnd.toDouble())
177 return Exception { TypeError };

181 m_private->setAppendWindowStart(m_appendWindowStart);

220void SourceBuffer::resetParserState()
221{
222 // Section 3.5.2 Reset Parser State algorithm steps.
223 // http://www.w3.org/TR/2014/CR-media-source-20140717/#sourcebuffer-reset-parser-state
224 // 1. If the append state equals PARSING_MEDIA_SEGMENT and the input buffer contains some complete coded frames,
225 // then run the coded frame processing algorithm until all of these complete coded frames have been processed.
226 // FIXME: If any implementation will work in pulling mode (instead of async push to SourceBufferPrivate, and forget)
227 // this should be handled somehow either here, or in m_private->abort();
228
229 // 2. Unset the last decode timestamp on all track buffers.
230 // 3. Unset the last frame duration on all track buffers.
231 // 4. Unset the highest presentation timestamp on all track buffers.
232 // 5. Set the need random access point flag on all track buffers to true.
233 m_private->resetTrackBuffers();
234
235 // 6. Remove all bytes from the input buffer.
236 // Note: this is handled by abortIfUpdating()
237 // 7. Set append state to WAITING_FOR_SEGMENT.
238 m_appendState = WaitingForSegment;
239
240 m_private->resetParserState();
241}

243ExceptionOr<void> SourceBuffer::abort()
244{
245 // Section 3.2 abort() method steps.
246 // https://rawgit.com/w3c/media-source/45627646344eea0170dd1cbc5a3d508ca751abb8/media-source-respec.html#dom-sourcebuffer-abort
247 // 1. If this object has been removed from the sourceBuffers attribute of the parent media source
248 // then throw an InvalidStateError exception and abort these steps.
249 // 2. If the readyState attribute of the parent media source is not in the "open" state
250 // then throw an InvalidStateError exception and abort these steps.
251 if (isRemoved() || !m_source->isOpen())
252 return Exception { InvalidStateError };
253
254 // 3. If the range removal algorithm is running, then throw an InvalidStateError exception and abort these steps.
255 if (m_removeTimer.isActive())
256 return Exception { InvalidStateError };
257
258 // 4. If the sourceBuffer.updating attribute equals true, then run the following steps: ...
259 abortIfUpdating();
260
261 // 5. Run the reset parser state algorithm.
262 resetParserState();
263
264 // 6. Set appendWindowStart to the presentation start time.
265 m_appendWindowStart = MediaTime::zeroTime();
266 m_private->setAppendWindowStart(m_appendWindowStart);
267
268 // 7. Set appendWindowEnd to positive Infinity.
269 m_appendWindowEnd = MediaTime::positiveInfiniteTime();
270 m_private->setAppendWindowEnd(m_appendWindowEnd);
271
272 return { };
273}

280ExceptionOr<void> SourceBuffer::remove(const MediaTime& start, const MediaTime& end)
281{
282 DEBUG_LOG(LOGIDENTIFIER, "start = ", start, ", end = ", end);
283
284 // https://rawgit.com/w3c/media-source/45627646344eea0170dd1cbc5a3d508ca751abb8/media-source-respec.html#dom-sourcebuffer-remove
285 // Section 3.2 remove() method steps.
286 // 1. If this object has been removed from the sourceBuffers attribute of the parent media source then throw
287 // an InvalidStateError exception and abort these steps.
288 // 2. If the updating attribute equals true, then throw an InvalidStateError exception and abort these steps.
289 if (isRemoved() || m_updating)
290 return Exception { InvalidStateError };
291
292 // 3. If duration equals NaN, then throw a TypeError exception and abort these steps.
293 // 4. If start is negative or greater than duration, then throw a TypeError exception and abort these steps.
294 // 5. If end is less than or equal to start or end equals NaN, then throw a TypeError exception and abort these steps.
295 if (m_source->duration().isInvalid()
296 || end.isInvalid()
297 || start.isInvalid()
298 || start < MediaTime::zeroTime()
299 || start > m_source->duration()
300 || end <= start) {
301 return Exception { TypeError };
302 }
303
304 // 6. If the readyState attribute of the parent media source is in the "ended" state then run the following steps:
305 // 6.1. Set the readyState attribute of the parent media source to "open"
306 // 6.2. Queue a task to fire a simple event named sourceopen at the parent media source .
307 m_source->openIfInEndedState();
308
309 // 7. Run the range removal algorithm with start and end as the start and end of the removal range.
310 rangeRemoval(start, end);
311
312 return { };
313}

315void SourceBuffer::rangeRemoval(const MediaTime& start, const MediaTime& end)
316{
317 // 3.5.7 Range Removal
318 // https://rawgit.com/w3c/media-source/7bbe4aa33c61ec025bc7acbd80354110f6a000f9/media-source.html#sourcebuffer-range-removal
319 // 1. Let start equal the starting presentation timestamp for the removal range.
320 // 2. Let end equal the end presentation timestamp for the removal range.
321 // 3. Set the updating attribute to true.
322 m_updating = true;
323
324 // 4. Queue a task to fire a simple event named updatestart at this SourceBuffer object.
325 scheduleEvent(eventNames().updatestartEvent);
326
327 // 5. Return control to the caller and run the rest of the steps asynchronously.
328 m_pendingRemoveStart = start;
329 m_pendingRemoveEnd = end;
330 m_removeTimer.startOneShot(0_s);
331}

333ExceptionOr<void> SourceBuffer::changeType(const String& type)
334{
335 // changeType() proposed API. See issue #155: <https://github.com/w3c/media-source/issues/155>
336 // https://rawgit.com/wicg/media-source/codec-switching/index.html#dom-sourcebuffer-changetype
337
338 // 1. If type is an empty string then throw a TypeError exception and abort these steps.
339 if (type.isEmpty())
340 return Exception { TypeError };
341
342 // 2. If this object has been removed from the sourceBuffers attribute of the parent media source,
343 // then throw an InvalidStateError exception and abort these steps.
344 // 3. If the updating attribute equals true, then throw an InvalidStateError exception and abort these steps.
345 if (isRemoved() || m_updating)
346 return Exception { InvalidStateError };
347
348 // 4. If type contains a MIME type that is not supported or contains a MIME type that is not supported with
349 // the types specified (currently or previously) of SourceBuffer objects in the sourceBuffers attribute of
350 // the parent media source, then throw a NotSupportedError exception and abort these steps.
351 ContentType contentType(type);
352 if (!m_private->canSwitchToType(contentType))
353 return Exception { NotSupportedError };
354
355 // 5. If the readyState attribute of the parent media source is in the "ended" state then run the following
356 // steps:
357 // 5.1. Set the readyState attribute of the parent media source to "open"
358 // 5.2. Queue a task to fire a simple event named sourceopen at the parent media source.
359 m_source->openIfInEndedState();
360
361 // 6. Run the reset parser state algorithm.
362 resetParserState();
363
364 // 7. Update the generate timestamps flag on this SourceBuffer object to the value in the "Generate Timestamps
365 // Flag" column of the byte stream format registry [MSE-REGISTRY] entry that is associated with type.
366 setShouldGenerateTimestamps(MediaSource::contentTypeShouldGenerateTimestamps(contentType));
367
368 // If the generate timestamps flag equals true:
369 // Set the mode attribute on this SourceBuffer object to "sequence", including running the associated steps
370 // for that attribute being set.
371 if (m_shouldGenerateTimestamps)
372 setMode(AppendMode::Sequence);
373
374 // Otherwise:
375 // Keep the previous value of the mode attribute on this SourceBuffer object, without running any associated
376 // steps for that attribute being set.
377 // NOTE: No-op.
378
379 // 9. Set pending initialization segment for changeType flag to true.
380 m_pendingInitializationSegmentForChangeType = true;
381 m_private->startChangingType();
382
383 return { };
384}

386void SourceBuffer::abortIfUpdating()
387{
388 // Section 3.2 abort() method step 4 substeps.
389 // https://rawgit.com/w3c/media-source/45627646344eea0170dd1cbc5a3d508ca751abb8/media-source-respec.html#dom-sourcebuffer-abort
390
391 if (!m_updating)
392 return;
393
394 // 4.1. Abort the buffer append algorithm if it is running.
395 m_appendBufferTimer.stop();
396 m_pendingAppendData.clear();
397 m_private->abort();
398
399 // 4.2. Set the updating attribute to false.
400 m_updating = false;
401
402 // 4.3. Queue a task to fire a simple event named abort at this SourceBuffer object.
403 scheduleEvent(eventNames().abortEvent);
404
405 // 4.4. Queue a task to fire a simple event named updateend at this SourceBuffer object.
406 scheduleEvent(eventNames().updateendEvent);
407}

465ExceptionOr<void> SourceBuffer::appendBufferInternal(const unsigned char* data, unsigned size)
466{
467 // Section 3.2 appendBuffer()
468 // https://dvcs.w3.org/hg/html-media/raw-file/default/media-source/media-source.html#widl-SourceBuffer-appendBuffer-void-ArrayBufferView-data
469
470 // Step 1 is enforced by the caller.
471 // 2. Run the prepare append algorithm.
472 // Section 3.5.4 Prepare AppendAlgorithm
473
474 // 1. If the SourceBuffer has been removed from the sourceBuffers attribute of the parent media source
475 // then throw an InvalidStateError exception and abort these steps.
476 // 2. If the updating attribute equals true, then throw an InvalidStateError exception and abort these steps.
477 if (isRemoved() || m_updating)
478 return Exception { InvalidStateError };
479
480 // 3. If the readyState attribute of the parent media source is in the "ended" state then run the following steps:
481 // 3.1. Set the readyState attribute of the parent media source to "open"
482 // 3.2. Queue a task to fire a simple event named sourceopen at the parent media source .
483 m_source->openIfInEndedState();
484
485 // 4. Run the coded frame eviction algorithm.
486 m_private->evictCodedFrames(size, maximumBufferSize(), m_source->currentTime(), m_source->duration(), m_source->isEnded());
487
488 // 5. If the buffer full flag equals true, then throw a QuotaExceededError exception and abort these step.
489 if (m_private->isBufferFullFor(size, maximumBufferSize())) {
490 ERROR_LOG(LOGIDENTIFIER, "buffer full, failing with QuotaExceededError error");
491 return Exception { QuotaExceededError };
492 }
493
494 // NOTE: Return to 3.2 appendBuffer()
495 // 3. Add data to the end of the input buffer.
496 m_pendingAppendData.append(data, size);
497
498 // 4. Set the updating attribute to true.
499 m_updating = true;
500
501 // 5. Queue a task to fire a simple event named updatestart at this SourceBuffer object.
502 scheduleEvent(eventNames().updatestartEvent);
503
504 // 6. Asynchronously run the buffer append algorithm.
505 m_appendBufferTimer.startOneShot(0_s);
506
507 return { };
508}
509
510void SourceBuffer::appendBufferTimerFired()
511{
512 if (isRemoved())
513 return;
514
515 ASSERT(m_updating);
516
517 // Section 3.5.5 Buffer Append Algorithm
518 // https://dvcs.w3.org/hg/html-media/raw-file/default/media-source/media-source.html#sourcebuffer-buffer-append
519
520 // 1. Run the segment parser loop algorithm.
521
522 // Section 3.5.1 Segment Parser Loop
523 // https://dvcs.w3.org/hg/html-media/raw-file/tip/media-source/media-source.html#sourcebuffer-segment-parser-loop
524 // When the segment parser loop algorithm is invoked, run the following steps:
525
526 // 1. Loop Top: If the input buffer is empty, then jump to the need more data step below.
527 if (!m_pendingAppendData.size()) {
528 sourceBufferPrivateAppendComplete(AppendResult::AppendSucceeded);
529 return;
530 }
531
532 // Manually clear out the m_pendingAppendData Vector, in case the platform implementation
533 // rejects appending the buffer for whatever reason.
534 // FIXME: The implementation should guarantee the move from this Vector, and we should
535 // assert here to confirm that. See https://bugs.webkit.org/show_bug.cgi?id=178003.
536 m_private->append(WTFMove(m_pendingAppendData));
537 m_pendingAppendData.clear();
538}
539
540void SourceBuffer::sourceBufferPrivateAppendComplete(AppendResult result)
541{
542 if (isRemoved())
543 return;
544
545 // Section 3.5.5 Buffer Append Algorithm, ctd.
546 // https://dvcs.w3.org/hg/html-media/raw-file/default/media-source/media-source.html#sourcebuffer-buffer-append
547
548 // 2. If the input buffer contains bytes that violate the SourceBuffer byte stream format specification,
549 // then run the append error algorithm with the decode error parameter set to true and abort this algorithm.
550 if (result == AppendResult::ParsingFailed) {
551 ERROR_LOG(LOGIDENTIFIER, "ParsingFailed");
552 appendError(true);
553 return;
554 }
555
556 // NOTE: Steps 3 - 6 enforced by sourceBufferPrivateDidReceiveInitializationSegment() and
557 // sourceBufferPrivateDidReceiveSample below.
558
559 // 7. Need more data: Return control to the calling algorithm.
560
561 // NOTE: return to Section 3.5.5
562 // 2.If the segment parser loop algorithm in the previous step was aborted, then abort this algorithm.
563 if (result != AppendResult::AppendSucceeded)
564 return;
565
566 // 3. Set the updating attribute to false.
567 m_updating = false;
568
569 // 4. Queue a task to fire a simple event named update at this SourceBuffer object.
570 scheduleEvent(eventNames().updateEvent);
571
572 // 5. Queue a task to fire a simple event named updateend at this SourceBuffer object.
573 scheduleEvent(eventNames().updateendEvent);
574
575 m_source->monitorSourceBuffers();
576 m_private->reenqueueMediaIfNeeded(m_source->currentTime());
577
578 DEBUG_LOG(LOGIDENTIFIER);
579}

593void SourceBuffer::removeTimerFired()
594{
595 if (isRemoved())
596 return;
597
598 ASSERT(m_updating);
599 ASSERT(m_pendingRemoveStart.isValid());
600 ASSERT(m_pendingRemoveStart < m_pendingRemoveEnd);
601
602 // Section 3.5.7 Range Removal
603 // http://w3c.github.io/media-source/#sourcebuffer-range-removal
604
605 // 6. Run the coded frame removal algorithm with start and end as the start and end of the removal range.
606
607 m_private->removeCodedFrames(m_pendingRemoveStart, m_pendingRemoveEnd, m_source->currentTime(), m_source->isEnded(), [this, protectedThis = makeRef(*this)] {
608 // 7. Set the updating attribute to false.
609 m_updating = false;
610 m_pendingRemoveStart = MediaTime::invalidTime();
611 m_pendingRemoveEnd = MediaTime::invalidTime();
612
613 // 8. Queue a task to fire a simple event named update at this SourceBuffer object.
614 scheduleEvent(eventNames().updateEvent);
615
616 // 9. Queue a task to fire a simple event named updateend at this SourceBuffer object.
617 scheduleEvent(eventNames().updateendEvent);
618 });
619}

665void SourceBuffer::sourceBufferPrivateDidReceiveInitializationSegment(InitializationSegment&& segment, CompletionHandler<void()>&& completionHandler)
666{
667 if (isRemoved()) {
668 completionHandler();
669 return;
670 }
671
672 ALWAYS_LOG(LOGIDENTIFIER);
673
674 // 3.5.8 Initialization Segment Received (ctd)
675 // https://rawgit.com/w3c/media-source/c3ad59c7a370d04430969ba73d18dc9bcde57a33/index.html#sourcebuffer-init-segment-received [Editor's Draft 09 January 2015]
676
677 // 1. Update the duration attribute if it currently equals NaN:
678 if (m_source->duration().isInvalid()) {
679 // If the initialization segment contains a duration:
680 // Run the duration change algorithm with new duration set to the duration in the initialization segment.
681 // Otherwise:
682 // Run the duration change algorithm with new duration set to positive Infinity.
683 if (segment.duration.isValid() && !segment.duration.isIndefinite())
684 m_source->setDurationInternal(segment.duration);
685 else
686 m_source->setDurationInternal(MediaTime::positiveInfiniteTime());
687 }
688
689 // 2. If the initialization segment has no audio, video, or text tracks, then run the append error algorithm
690 // with the decode error parameter set to true and abort these steps.
691 if (segment.audioTracks.isEmpty() && segment.videoTracks.isEmpty() && segment.textTracks.isEmpty()) {
692 appendError(true);
693 completionHandler();
694 return;
695 }
696
697 // 3. If the first initialization segment flag is true, then run the following steps:
698 if (m_receivedFirstInitializationSegment) {
699 // 3.1. Verify the following properties. If any of the checks fail then run the append error algorithm
700 // with the decode error parameter set to true and abort these steps.
701 if (!validateInitializationSegment(segment)) {
702 appendError(true);
703 completionHandler();
704 return;
705 }
706
707 Vector<std::pair<AtomString, AtomString>> trackIdPairs;
708
709 // 3.2 Add the appropriate track descriptions from this initialization segment to each of the track buffers.
710 ASSERT(segment.audioTracks.size() == audioTracks().length());
711 for (auto& audioTrackInfo : segment.audioTracks) {
712 if (audioTracks().length() == 1) {
713 auto* track = audioTracks().item(0);
714 auto oldId = track->id();
715 auto newId = audioTrackInfo.track->id();
716 track->setPrivate(*audioTrackInfo.track);
717 if (newId != oldId)
718 trackIdPairs.append(std::make_pair(oldId, newId));
719 break;
720 }
721
722 auto audioTrack = audioTracks().getTrackById(audioTrackInfo.track->id());
723 ASSERT(audioTrack);
724 audioTrack->setPrivate(*audioTrackInfo.track);
725 }
726
727 ASSERT(segment.videoTracks.size() == videoTracks().length());
728 for (auto& videoTrackInfo : segment.videoTracks) {
729 if (videoTracks().length() == 1) {
730 auto* track = videoTracks().item(0);
731 auto oldId = track->id();
732 auto newId = videoTrackInfo.track->id();
733 track->setPrivate(*videoTrackInfo.track);
734 if (newId != oldId)
735 trackIdPairs.append(std::make_pair(oldId, newId));
736 break;
737 }
738
739 auto videoTrack = videoTracks().getTrackById(videoTrackInfo.track->id());
740 ASSERT(videoTrack);
741 videoTrack->setPrivate(*videoTrackInfo.track);
742 }
743
744 ASSERT(segment.textTracks.size() == textTracks().length());
745 for (auto& textTrackInfo : segment.textTracks) {
746 if (textTracks().length() == 1) {
747 auto* track = downcast<InbandTextTrack>(textTracks().item(0));
748 auto oldId = track->id();
749 auto newId = textTrackInfo.track->id();
750 track->setPrivate(*textTrackInfo.track);
751 if (newId != oldId)
752 trackIdPairs.append(std::make_pair(oldId, newId));
753 break;
754 }
755
756 auto textTrack = textTracks().getTrackById(textTrackInfo.track->id());
757 ASSERT(textTrack);
758 downcast<InbandTextTrack>(*textTrack).setPrivate(*textTrackInfo.track);
759 }
760
761 if (!trackIdPairs.isEmpty())
762 m_private->updateTrackIds(WTFMove(trackIdPairs));
763
764 // 3.3 Set the need random access point flag on all track buffers to true.
765 m_private->setAllTrackBuffersNeedRandomAccess();
766 }
767
768 // 4. Let active track flag equal false.
769 bool activeTrackFlag = false;
770
771 // 5. If the first initialization segment flag is false, then run the following steps:
772 if (!m_receivedFirstInitializationSegment) {
773 // 5.1 If the initialization segment contains tracks with codecs the user agent does not support,
774 // then run the append error algorithm with the decode error parameter set to true and abort these steps.
775 // NOTE: This check is the responsibility of the SourceBufferPrivate.
776
777 // 5.2 For each audio track in the initialization segment, run following steps:
778 for (auto& audioTrackInfo : segment.audioTracks) {
779 // FIXME: Implement steps 5.2.1-5.2.8.1 as per Editor's Draft 09 January 2015, and reorder this
780 // 5.2.1 Let new audio track be a new AudioTrack object.
781 // 5.2.2 Generate a unique ID and assign it to the id property on new video track.
782 auto newAudioTrack = AudioTrack::create(*this, *audioTrackInfo.track);
783 newAudioTrack->setSourceBuffer(this);
784
785 // 5.2.3 If audioTracks.length equals 0, then run the following steps:
786 if (!audioTracks().length()) {
787 // 5.2.3.1 Set the enabled property on new audio track to true.
788 newAudioTrack->setEnabled(true);
789
790 // 5.2.3.2 Set active track flag to true.
791 activeTrackFlag = true;
792 }
793
794 // 5.2.4 Add new audio track to the audioTracks attribute on this SourceBuffer object.
795 // 5.2.5 Queue a task to fire a trusted event named addtrack, that does not bubble and is
796 // not cancelable, and that uses the TrackEvent interface, at the AudioTrackList object
797 // referenced by the audioTracks attribute on this SourceBuffer object.
798 audioTracks().append(newAudioTrack.copyRef());
799
800 // 5.2.6 Add new audio track to the audioTracks attribute on the HTMLMediaElement.
801 // 5.2.7 Queue a task to fire a trusted event named addtrack, that does not bubble and is
802 // not cancelable, and that uses the TrackEvent interface, at the AudioTrackList object
803 // referenced by the audioTracks attribute on the HTMLMediaElement.
804 m_source->mediaElement()->ensureAudioTracks().append(newAudioTrack.copyRef());
805
806 m_audioCodecs.append(audioTrackInfo.description->codec());
807
808 // 5.2.8 Create a new track buffer to store coded frames for this track.
809 m_private->addTrackBuffer(newAudioTrack->id(), WTFMove(audioTrackInfo.description));
810 }
811
812 // 5.3 For each video track in the initialization segment, run following steps:
813 for (auto& videoTrackInfo : segment.videoTracks) {
814 // FIXME: Implement steps 5.3.1-5.3.8.1 as per Editor's Draft 09 January 2015, and reorder this
815 // 5.3.1 Let new video track be a new VideoTrack object.
816 // 5.3.2 Generate a unique ID and assign it to the id property on new video track.
817 auto newVideoTrack = VideoTrack::create(*this, *videoTrackInfo.track);
818 newVideoTrack->setSourceBuffer(this);
819
820 // 5.3.3 If videoTracks.length equals 0, then run the following steps:
821 if (!videoTracks().length()) {
822 // 5.3.3.1 Set the selected property on new video track to true.
823 newVideoTrack->setSelected(true);
824
825 // 5.3.3.2 Set active track flag to true.
826 activeTrackFlag = true;
827 }
828
829 // 5.3.4 Add new video track to the videoTracks attribute on this SourceBuffer object.
830 // 5.3.5 Queue a task to fire a trusted event named addtrack, that does not bubble and is
831 // not cancelable, and that uses the TrackEvent interface, at the VideoTrackList object
832 // referenced by the videoTracks attribute on this SourceBuffer object.
833 videoTracks().append(newVideoTrack.copyRef());
834
835 // 5.3.6 Add new video track to the videoTracks attribute on the HTMLMediaElement.
836 // 5.3.7 Queue a task to fire a trusted event named addtrack, that does not bubble and is
837 // not cancelable, and that uses the TrackEvent interface, at the VideoTrackList object
838 // referenced by the videoTracks attribute on the HTMLMediaElement.
839 m_source->mediaElement()->ensureVideoTracks().append(newVideoTrack.copyRef());
840
841 m_videoCodecs.append(videoTrackInfo.description->codec());
842
843 // 5.3.8 Create a new track buffer to store coded frames for this track.
844 m_private->addTrackBuffer(newVideoTrack->id(), WTFMove(videoTrackInfo.description));
845 }
846
847 // 5.4 For each text track in the initialization segment, run following steps:
848 for (auto& textTrackInfo : segment.textTracks) {
849 auto& textTrackPrivate = *textTrackInfo.track;
850
851 // FIXME: Implement steps 5.4.1-5.4.8.1 as per Editor's Draft 09 January 2015, and reorder this
852 // 5.4.1 Let new text track be a new TextTrack object with its properties populated with the
853 // appropriate information from the initialization segment.
854 auto newTextTrack = InbandTextTrack::create(document(), *this, textTrackPrivate);
855
856 // 5.4.2 If the mode property on new text track equals "showing" or "hidden", then set active
857 // track flag to true.
858 if (textTrackPrivate.mode() != InbandTextTrackPrivate::Mode::Disabled)
859 activeTrackFlag = true;
860
861 // 5.4.3 Add new text track to the textTracks attribute on this SourceBuffer object.
862 // 5.4.4 Queue a task to fire a trusted event named addtrack, that does not bubble and is
863 // not cancelable, and that uses the TrackEvent interface, at textTracks attribute on this
864 // SourceBuffer object.
865 textTracks().append(newTextTrack.get());
866
867 // 5.4.5 Add new text track to the textTracks attribute on the HTMLMediaElement.
868 // 5.4.6 Queue a task to fire a trusted event named addtrack, that does not bubble and is
869 // not cancelable, and that uses the TrackEvent interface, at the TextTrackList object
870 // referenced by the textTracks attribute on the HTMLMediaElement.
871 m_source->mediaElement()->ensureTextTracks().append(newTextTrack.copyRef());
872
873 m_textCodecs.append(textTrackInfo.description->codec());
874
875 // 5.4.7 Create a new track buffer to store coded frames for this track.
876 m_private->addTrackBuffer(newTextTrack->id(), WTFMove(textTrackInfo.description));
877 }
878
879 // 5.5 If active track flag equals true, then run the following steps:
880 if (activeTrackFlag) {
881 // 5.5.1 Add this SourceBuffer to activeSourceBuffers.
882 // 5.5.2 Queue a task to fire a simple event named addsourcebuffer at activeSourceBuffers
883 setActive(true);
884 }
885
886 // 5.6 Set first initialization segment flag to true.
887 m_receivedFirstInitializationSegment = true;
888 }
889
890 // (Note: Issue #155 adds this step after step 5:)
891 // 6. Set pending initialization segment for changeType flag to false.
892 m_pendingInitializationSegmentForChangeType = false;
893 completionHandler();
894
895 // 6. If the HTMLMediaElement.readyState attribute is HAVE_NOTHING, then run the following steps:
896 if (m_private->readyState() == MediaPlayer::ReadyState::HaveNothing) {
897 // 6.1 If one or more objects in sourceBuffers have first initialization segment flag set to false, then abort these steps.
898 for (auto& sourceBuffer : *m_source->sourceBuffers()) {
899 if (!sourceBuffer->m_receivedFirstInitializationSegment)
900 return;
901 }
902
903 // 6.2 Set the HTMLMediaElement.readyState attribute to HAVE_METADATA.
904 // 6.3 Queue a task to fire a simple event named loadedmetadata at the media element.
905 m_private->setReadyState(MediaPlayer::ReadyState::HaveMetadata);
906 }
907
908 // 7. If the active track flag equals true and the HTMLMediaElement.readyState
909 // attribute is greater than HAVE_CURRENT_DATA, then set the HTMLMediaElement.readyState
910 // attribute to HAVE_METADATA.
911 if (activeTrackFlag && m_private->readyState() > MediaPlayer::ReadyState::HaveCurrentData)
912 m_private->setReadyState(MediaPlayer::ReadyState::HaveMetadata);
913}

915bool SourceBuffer::validateInitializationSegment(const InitializationSegment& segment)
916{
917 // FIXME: ordering of all 3.5.X (X>=7) functions needs to be updated to post-[24 July 2014 Editor's Draft] version
918 // 3.5.8 Initialization Segment Received (ctd)
919 // https://rawgit.com/w3c/media-source/c3ad59c7a370d04430969ba73d18dc9bcde57a33/index.html#sourcebuffer-init-segment-received [Editor's Draft 09 January 2015]
920
921 // Note: those are checks from step 3.1
922 // * The number of audio, video, and text tracks match what was in the first initialization segment.
923 if (segment.audioTracks.size() != audioTracks().length()
924 || segment.videoTracks.size() != videoTracks().length()
925 || segment.textTracks.size() != textTracks().length())
926 return false;
927
928 // * The codecs for each track, match what was specified in the first initialization segment.
929 // (Note: Issue #155 strikes out this check. For broad compatibility when this experimental feature
930 // is not enabled, only perform this check if the "pending initialization segment for changeType flag"
931 // is not set.)
932 for (auto& audioTrackInfo : segment.audioTracks) {
933 if (m_audioCodecs.contains(audioTrackInfo.description->codec()))
934 continue;
935
936 if (!m_pendingInitializationSegmentForChangeType)
937 return false;
938
939 m_audioCodecs.append(audioTrackInfo.description->codec());
940 }
941
942 for (auto& videoTrackInfo : segment.videoTracks) {
943 if (m_videoCodecs.contains(videoTrackInfo.description->codec()))
944 continue;
945
946 if (!m_pendingInitializationSegmentForChangeType)
947 return false;
948
949 m_videoCodecs.append(videoTrackInfo.description->codec());
950 }
951
952 for (auto& textTrackInfo : segment.textTracks) {
953 if (m_textCodecs.contains(textTrackInfo.description->codec()))
954 continue;
955
956 if (!m_pendingInitializationSegmentForChangeType)
957 return false;
958
959 m_textCodecs.append(textTrackInfo.description->codec());
960 }
961
962 return true;
963}

1143void SourceBuffer::monitorBufferingRate()
1144{
1145 MonotonicTime now = MonotonicTime::now();
1146 Seconds interval = now - m_timeOfBufferingMonitor;
1147 double rateSinceLastMonitor = m_bufferedSinceLastMonitor / interval.seconds();
1148
1149 m_timeOfBufferingMonitor = now;
1150 m_bufferedSinceLastMonitor = 0;
1151
1152 m_averageBufferRate += (interval.seconds() * ExponentialMovingAverageCoefficient) * (rateSinceLastMonitor - m_averageBufferRate);
1153
1154 DEBUG_LOG(LOGIDENTIFIER, m_averageBufferRate);
1155}
1156
1157bool SourceBuffer::canPlayThroughRange(PlatformTimeRanges& ranges)
1158{
1159 if (isRemoved())
1160 return false;
1161
1162 monitorBufferingRate();
1163
1164 // Assuming no fluctuations in the buffering rate, loading 1 second per second or greater
1165 // means indefinite playback. This could be improved by taking jitter into account.
1166 if (m_averageBufferRate > 1)
1167 return true;
1168
1169 // Add up all the time yet to be buffered.
1170 MediaTime currentTime = m_source->currentTime();
1171 MediaTime duration = m_source->duration();
1172
1173 PlatformTimeRanges unbufferedRanges = ranges;
1174 unbufferedRanges.invert();
1175 unbufferedRanges.intersectWith(PlatformTimeRanges(currentTime, std::max(currentTime, duration)));
1176 MediaTime unbufferedTime = unbufferedRanges.totalDuration();
1177 if (!unbufferedTime.isValid())
1178 return true;
1179
1180 MediaTime timeRemaining = duration - currentTime;
1181 return unbufferedTime.toDouble() / m_averageBufferRate < timeRemaining.toDouble();
1182}

1229
1230ExceptionOr<void> SourceBuffer::setMode(AppendMode newMode)
1231{
1232 // 3.1 Attributes - mode
1233 // http://www.w3.org/TR/media-source/#widl-SourceBuffer-mode
1234
1235 // On setting, run the following steps:
1236
1237 // 1. Let new mode equal the new value being assigned to this attribute.
1238 // 2. If generate timestamps flag equals true and new mode equals "segments", then throw an InvalidAccessError exception and abort these steps.
1239 if (m_shouldGenerateTimestamps && newMode == AppendMode::Segments)
1240 return Exception { InvalidAccessError };
1241
1242 // 3. If this object has been removed from the sourceBuffers attribute of the parent media source, then throw an InvalidStateError exception and abort these steps.
1243 // 4. If the updating attribute equals true, then throw an InvalidStateError exception and abort these steps.
1244 if (isRemoved() || m_updating)
1245 return Exception { InvalidStateError };
1246
1247 // 5. If the readyState attribute of the parent media source is in the "ended" state then run the following steps:
1248 if (m_source->isEnded()) {
1249 // 5.1. Set the readyState attribute of the parent media source to "open"
1250 // 5.2. Queue a task to fire a simple event named sourceopen at the parent media source.
1251 m_source->openIfInEndedState();
1252 }
1253
1254 // 6. If the append state equals PARSING_MEDIA_SEGMENT, then throw an InvalidStateError and abort these steps.
1255 if (m_appendState == ParsingMediaSegment)
1256 return Exception { InvalidStateError };
1257
1258 // 7. If the new mode equals "sequence", then set the group start timestamp to the group end timestamp.
1259 if (newMode == AppendMode::Sequence)
1260 m_private->setGroupStartTimestampToEndTimestamp();
1261
1262 // 8. Update the attribute to new mode.
1263 m_mode = newMode;
1264
1265 if (newMode == AppendMode::Segments)
1266 m_private->setMode(SourceBufferAppendMode::Segments);
1267 else
1268 m_private->setMode(SourceBufferAppendMode::Sequence);
1269
1270 return { };
1271}
