41namespace WebCore {
42
43WTF_MAKE_ISO_ALLOCATED_IMPL(SourceBufferList);
44
45SourceBufferList::SourceBufferList(ScriptExecutionContext* context)
46 : ActiveDOMObject(context)
47{
48 suspendIfNeeded();
49}
50
51SourceBufferList::~SourceBufferList()
52{
53 ASSERT(m_list.isEmpty());
54}
55
56void SourceBufferList::add(Ref<SourceBuffer>&& buffer)
57{
58 m_list.append(WTFMove(buffer));
59 scheduleEvent(eventNames().addsourcebufferEvent);
60}
61
62void SourceBufferList::remove(SourceBuffer& buffer)
63{
64 size_t index = m_list.find(&buffer);
65 if (index == notFound)
66 return;
67 m_list.remove(index);
68 scheduleEvent(eventNames().removesourcebufferEvent);
69}
70
71void SourceBufferList::clear()
72{
73 m_list.clear();
74 scheduleEvent(eventNames().removesourcebufferEvent);
75}
76
77void SourceBufferList::swap(Vector<RefPtr<SourceBuffer>>& other)
78{
79 int changeInSize = other.size() - m_list.size();
80 int addedEntries = 0;
81 for (auto& sourceBuffer : other) {
82 if (!m_list.contains(sourceBuffer))
83 ++addedEntries;
84 }
85 int removedEntries = addedEntries - changeInSize;
86
87 m_list.swap(other);
88
89 if (addedEntries)
90 scheduleEvent(eventNames().addsourcebufferEvent);
91 if (removedEntries)
92 scheduleEvent(eventNames().removesourcebufferEvent);
93}
94
95void SourceBufferList::scheduleEvent(const AtomString& eventName)
96{
97 queueTaskToDispatchEvent(*this, TaskSource::MediaElement, Event::create(eventName, Event::CanBubble::No, Event::IsCancelable::No));
98}
99
100const char* SourceBufferList::activeDOMObjectName() const
101{
102 return "SourceBufferList";
103}
