39Ref<CanvasCaptureMediaStreamTrack> CanvasCaptureMediaStreamTrack::create(Document& document, Ref<HTMLCanvasElement>&& canvas, std::optional<double>&& frameRequestRate)
40{
41 auto source = CanvasCaptureMediaStreamTrack::Source::create(canvas.get(), WTFMove(frameRequestRate));

59Ref<CanvasCaptureMediaStreamTrack::Source> CanvasCaptureMediaStreamTrack::Source::create(HTMLCanvasElement& canvas, std::optional<double>&& frameRequestRate)
60{
61 auto source = adoptRef(*new Source(canvas, WTFMove(frameRequestRate)));
62 source->start();
63
64 callOnMainThread([source] {
65 if (!source->m_canvas)
66 return;
67 source->captureCanvas();
68 });
69 return source;
70}

87void CanvasCaptureMediaStreamTrack::Source::startProducingData()
88{
89 if (!m_canvas)
90 return;
91 m_canvas->addObserver(*this);
92
93 if (!m_frameRequestRate)
94 return;
95
96 if (m_frameRequestRate.value())
97 m_requestFrameTimer.startRepeating(1_s / m_frameRequestRate.value());
98}
99
100void CanvasCaptureMediaStreamTrack::Source::stopProducingData()
101{
102 m_requestFrameTimer.stop();
103
104 if (!m_canvas)
105 return;
106 m_canvas->removeObserver(*this);
107}

109void CanvasCaptureMediaStreamTrack::Source::requestFrameTimerFired()
110{
111 requestFrame();
112}

114void CanvasCaptureMediaStreamTrack::Source::canvasDestroyed(CanvasBase& canvas)

122const RealtimeMediaSourceSettings& CanvasCaptureMediaStreamTrack::Source::settings()
123{
124 if (m_currentSettings)
125 return m_currentSettings.value();
126
127 RealtimeMediaSourceSupportedConstraints constraints;
128 constraints.setSupportsWidth(true);
129 constraints.setSupportsHeight(true);
130
131 RealtimeMediaSourceSettings settings;
132 settings.setWidth(m_canvas->width());
133 settings.setHeight(m_canvas->height());
134 settings.setSupportedConstraints(constraints);
135
136 m_currentSettings = WTFMove(settings);
137 return m_currentSettings.value();
138}

143 m_currentSettings = std::nullopt;

152void CanvasCaptureMediaStreamTrack::Source::canvasChanged(CanvasBase& canvas, const std::optional<FloatRect>&)
153{
154 ASSERT_UNUSED(canvas, m_canvas == &canvas);
155
156#if ENABLE(WEBGL)
157 // FIXME: We need to preserve drawing buffer as we are currently grabbing frames asynchronously.
158 // We should instead add an anchor point for both 2d and 3d contexts where canvas will actually paint.
159 // And call canvas observers from that point.
160 if (is<WebGLRenderingContextBase>(canvas.renderingContext())) {
161 auto& context = downcast<WebGLRenderingContextBase>(*canvas.renderingContext());
162 if (!context.isPreservingDrawingBuffer()) {
163 canvas.scriptExecutionContext()->addConsoleMessage(MessageSource::JS, MessageLevel::Warning, "Turning drawing buffer preservation for the WebGL canvas being captured"_s);
164 context.enablePreserveDrawingBuffer();
165 }
166 }
167#endif
168
169 // FIXME: We should try to generate the frame at the time the screen is being updated.
170 if (m_captureCanvasTimer.isActive())
171 return;
172 m_captureCanvasTimer.startOneShot(0_s);
173}
174
175void CanvasCaptureMediaStreamTrack::Source::captureCanvas()
176{
177 ASSERT(m_canvas);
178
179 if (!isProducingData())
180 return;
181
182 if (m_frameRequestRate) {
183 if (!m_shouldEmitFrame)
184 return;
185 m_shouldEmitFrame = false;
186 }
187
188 if (!m_canvas->originClean())
189 return;
190
191 auto sample = m_canvas->toMediaSample();
192 if (!sample)
193 return;
194
195 videoSampleAvailable(*sample);
196}

198RefPtr<MediaStreamTrack> CanvasCaptureMediaStreamTrack::clone()
199{
200 if (!scriptExecutionContext())
201 return nullptr;
202
203 auto track = adoptRef(*new CanvasCaptureMediaStreamTrack(downcast<Document>(*scriptExecutionContext()), m_canvas.copyRef(), m_private->clone()));
204 track->suspendIfNeeded();
205 return track;
206}
