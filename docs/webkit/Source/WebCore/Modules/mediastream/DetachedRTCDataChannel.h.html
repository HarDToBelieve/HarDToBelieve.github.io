28#if ENABLE(WEB_RTC)
29
30#include "RTCDataChannelHandler.h"
31#include "RTCDataChannelIdentifier.h"
32#include "RTCDataChannelState.h"
33#include <wtf/text/WTFString.h>
34
35namespace WebCore {
36
37struct DetachedRTCDataChannel {
38 WTF_MAKE_NONCOPYABLE(DetachedRTCDataChannel);
39 WTF_MAKE_FAST_ALLOCATED;
40public:
41 DetachedRTCDataChannel(RTCDataChannelIdentifier identifier, const String& label, const RTCDataChannelInit& options, RTCDataChannelState state)
42 : identifier(identifier)
43 , label(label.isolatedCopy())
44 , options(options.isolatedCopy())
45 , state(state)
46 {
47 }
48
49 size_t memoryCost() const { return label.sizeInBytes(); }
50
51 RTCDataChannelIdentifier identifier;
52 String label;
53 RTCDataChannelInit options;
54 RTCDataChannelState state { RTCDataChannelState::Closed };
55
56 template<class Encoder> void encode(Encoder&) const;
57 template<class Decoder> static std::unique_ptr<DetachedRTCDataChannel> decode(Decoder&);
58};
59
60template<class Encoder> void DetachedRTCDataChannel::encode(Encoder& encoder) const
61{
62 encoder << identifier << label << options << state;
63}
64
65template<class Decoder> std::unique_ptr<DetachedRTCDataChannel> DetachedRTCDataChannel::decode(Decoder& decoder)
66{
67 std::optional<RTCDataChannelIdentifier> identifier;
68 decoder >> identifier;
69 if (!identifier)
70 return { };
71
72 String label;
73 if (!decoder.decode(label))
74 return { };
75
76 std::optional<RTCDataChannelInit> options;
77 decoder >> options;
78 if (!options)
79 return { };
80
81 std::optional<RTCDataChannelState> state;
82 decoder >> state;
83 if (!state)
84 return { };
85
86 return makeUnique<DetachedRTCDataChannel>(*identifier, WTFMove(label), *options, *state);
87}

89} // namespace WebCore

91#endif // ENABLE(WEB_RTC)
