49class Document;
50class MediaDeviceInfo;

54struct MediaTrackSupportedConstraints;

56template<typename IDLType> class DOMPromiseDeferred;
57
58class MediaDevices final : public RefCounted<MediaDevices>, public ActiveDOMObject, public EventTargetWithInlineData {
59 WTF_MAKE_ISO_ALLOCATED(MediaDevices);
60public:
61 static Ref<MediaDevices> create(Document&);
62
63 ~MediaDevices();
64
65 Document* document() const;
66
67 using Promise = DOMPromiseDeferred<IDLInterface<MediaStream>>;
68 using EnumerateDevicesPromise = DOMPromiseDeferred<IDLSequence<IDLInterface<MediaDeviceInfo>>>;
69
70 enum class DisplayCaptureSurfaceType {
71 Monitor,
72 Window,
73 Application,
74 Browser,
75 };
76
77 struct StreamConstraints {
78 Variant<bool, MediaTrackConstraints> video;
79 Variant<bool, MediaTrackConstraints> audio;
80 };
81 void getUserMedia(const StreamConstraints&, Promise&&);
82
83 struct DisplayMediaStreamConstraints {
84 Variant<bool, MediaTrackConstraints> video;
85 Variant<bool, MediaTrackConstraints> audio;
86 };
87 void getDisplayMedia(const DisplayMediaStreamConstraints&, Promise&&);
88
89 void enumerateDevices(EnumerateDevicesPromise&&);
90 MediaTrackSupportedConstraints getSupportedConstraints();
91
92 String deviceIdToPersistentId(const String& deviceId) const { return m_audioOutputDeviceIdToPersistentId.get(deviceId); }
93
94 using RefCounted<MediaDevices>::ref;
95 using RefCounted<MediaDevices>::deref;
96
97private:
98 explicit MediaDevices(Document&);
99
100 void scheduledEventTimerFired();
101 bool addEventListener(const AtomString& eventType, Ref<EventListener>&&, const AddEventListenerOptions&) override;
102
103 void exposeDevices(const Vector<CaptureDevice>&, const String&, EnumerateDevicesPromise&&);
104 void listenForDeviceChanges();
105
106 friend class JSMediaDevicesOwner;
107
108 // ActiveDOMObject
109 const char* activeDOMObjectName() const final;
110 void stop() final;
111 bool virtualHasPendingActivity() const final;
112
113 // EventTargetWithInlineData.
114 EventTargetInterface eventTargetInterface() const final { return MediaDevicesEventTargetInterfaceType; }
115 ScriptExecutionContext* scriptExecutionContext() const final { return ActiveDOMObject::scriptExecutionContext(); }
116 void refEventTarget() final { ref(); }
117 void derefEventTarget() final { deref(); }
118
119 enum class GestureAllowedRequest {
120 Microphone = 1 << 0,
121 Camera = 1 << 1,
122 Display = 1 << 2,
123 };
124 bool computeUserGesturePriviledge(GestureAllowedRequest);
125
126 Timer m_scheduledEventTimer;
127 UserMediaClient::DeviceChangeObserverToken m_deviceChangeToken;
128 const EventNames& m_eventNames; // Need to cache this so we can use it from GC threads.
129 bool m_listeningForDeviceChanges { false };
130
131 String m_groupIdHashSalt;
132
133 OptionSet<GestureAllowedRequest> m_requestTypesForCurrentGesture;
134 WeakPtr<UserGestureToken> m_currentGestureToken;
135
136 HashMap<String, String> m_audioOutputDeviceIdToPersistentId;
137 String m_audioOutputDeviceId;
138};
