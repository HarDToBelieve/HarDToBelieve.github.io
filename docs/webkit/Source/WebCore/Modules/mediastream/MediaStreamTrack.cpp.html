70Ref<MediaStreamTrack> MediaStreamTrack::create(ScriptExecutionContext& context, Ref<MediaStreamTrackPrivate>&& privateTrack)
71{
72 auto track = adoptRef(*new MediaStreamTrack(context, WTFMove(privateTrack)));
73 track->suspendIfNeeded();

77

81MediaStreamTrack::MediaStreamTrack(ScriptExecutionContext& context, Ref<MediaStreamTrackPrivate>&& privateTrack)
82 : ActiveDOMObject(&context)
83 , m_private(WTFMove(privateTrack))
84 , m_muted(m_private->muted())
85 , m_isCaptureTrack(m_private->isCaptureTrack())
86{
87 ALWAYS_LOG(LOGIDENTIFIER);
88
89 m_private->addObserver(*this);
90
91 if (!isCaptureTrack())
92 return;
93
94 allCaptureTracks().add(this);
95
96 if (m_private->type() == RealtimeMediaSource::Type::Audio)
97 PlatformMediaSessionManager::sharedManager().addAudioCaptureSource(*this);
98}

109 if (m_private->type() == RealtimeMediaSource::Type::Audio)

154 default:

159void MediaStreamTrack::setContentHint(const String& hintValue)
160{
161 MediaStreamTrackPrivate::HintValue value;
162 if (m_private->type() == RealtimeMediaSource::Type::Audio) {
163 if (hintValue == "")
164 value = MediaStreamTrackPrivate::HintValue::Empty;
165 else if (hintValue == "speech")
166 value = MediaStreamTrackPrivate::HintValue::Speech;
167 else if (hintValue == "music")
168 value = MediaStreamTrackPrivate::HintValue::Music;
169 else
170 return;
171 } else {
172 if (hintValue == "")
173 value = MediaStreamTrackPrivate::HintValue::Empty;
174 else if (hintValue == "detail")
175 value = MediaStreamTrackPrivate::HintValue::Detail;
176 else if (hintValue == "motion")
177 value = MediaStreamTrackPrivate::HintValue::Motion;
178 else if (hintValue == "text")
179 value = MediaStreamTrackPrivate::HintValue::Text;
180 else
181 return;
182 }
183 m_private->setContentHint(value);
184}

226void MediaStreamTrack::stopTrack(StopMode mode)
227{
228 // NOTE: this method is called when the "stop" method is called from JS, using the "ImplementedAs" IDL attribute.
229 // This is done because ActiveDOMObject requires a "stop" method.
230
231 if (ended())
232 return;
233
234 ALWAYS_LOG(LOGIDENTIFIER, static_cast<int>(mode));
235
236 // An 'ended' event is not posted if m_ended is true when trackEnded is called, so set it now if we are
237 // not supposed to post the event.
238 if (mode == StopMode::Silently)
239 m_ended = true;
240
241 m_private->endTrack();
242 m_ended = true;
243
244 configureTrackRendering();
245}

247MediaStreamTrack::TrackSettings MediaStreamTrack::getSettings() const
248{
249 auto& settings = m_private->settings();
250 TrackSettings result;
251 if (settings.supportsWidth())
252 result.width = settings.width();
253 if (settings.supportsHeight())
254 result.height = settings.height();
255 if (settings.supportsAspectRatio() && settings.aspectRatio()) // FIXME: Why the check for zero here?
256 result.aspectRatio = settings.aspectRatio();
257 if (settings.supportsFrameRate())
258 result.frameRate = settings.frameRate();
259 if (settings.supportsFacingMode())
260 result.facingMode = RealtimeMediaSourceSettings::facingMode(settings.facingMode());
261 if (settings.supportsVolume())
262 result.volume = settings.volume();
263 if (settings.supportsSampleRate())
264 result.sampleRate = settings.sampleRate();
265 if (settings.supportsSampleSize())
266 result.sampleSize = settings.sampleSize();
267 if (settings.supportsEchoCancellation())
268 result.echoCancellation = settings.echoCancellation();
269 if (settings.supportsDeviceId())
270 result.deviceId = settings.deviceId();
271 if (settings.supportsGroupId())
272 result.groupId = settings.groupId();
273
274 // FIXME: shouldn't this include displaySurface and logicalSurface?
275
276 return result;
277}

279static DoubleRange capabilityDoubleRange(const CapabilityValueOrRange& value)
280{
281 DoubleRange range;
282 switch (value.type()) {
283 case CapabilityValueOrRange::Double:
284 range.min = value.value().asDouble;
285 range.max = range.min;
286 break;
287 case CapabilityValueOrRange::DoubleRange:
288 range.min = value.rangeMin().asDouble;
289 range.max = value.rangeMax().asDouble;
290 break;
291 case CapabilityValueOrRange::Undefined:
292 case CapabilityValueOrRange::ULong:
293 case CapabilityValueOrRange::ULongRange:
294 ASSERT_NOT_REACHED();
295 }
296 return range;
297}

338MediaStreamTrack::TrackCapabilities MediaStreamTrack::getCapabilities() const
339{
340 auto capabilities = m_private->capabilities();
341 TrackCapabilities result;
342 if (capabilities.supportsWidth())
343 result.width = capabilityIntRange(capabilities.width());
344 if (capabilities.supportsHeight())
345 result.height = capabilityIntRange(capabilities.height());
346 if (capabilities.supportsAspectRatio())
347 result.aspectRatio = capabilityDoubleRange(capabilities.aspectRatio());
348 if (capabilities.supportsFrameRate())
349 result.frameRate = capabilityDoubleRange(capabilities.frameRate());
350 if (capabilities.supportsFacingMode())
351 result.facingMode = capabilityStringVector(capabilities.facingMode());
352 if (capabilities.supportsVolume())
353 result.volume = capabilityDoubleRange(capabilities.volume());
354 if (capabilities.supportsSampleRate())
355 result.sampleRate = capabilityIntRange(capabilities.sampleRate());
356 if (capabilities.supportsSampleSize())
357 result.sampleSize = capabilityIntRange(capabilities.sampleSize());
358 if (capabilities.supportsEchoCancellation())
359 result.echoCancellation = capabilityBooleanVector(capabilities.echoCancellation());
360 if (capabilities.supportsDeviceId())
361 result.deviceId = capabilities.deviceId();
362 if (capabilities.supportsGroupId())
363 result.groupId = capabilities.groupId();
364 return result;
365}

377void MediaStreamTrack::applyConstraints(const std::optional<MediaTrackConstraints>& constraints, DOMPromiseDeferred<void>&& promise)
378{
379 m_promise = WTF::makeUnique<DOMPromiseDeferred<void>>(WTFMove(promise));
380
381 auto completionHandler = [this, protectedThis = makeRef(*this), constraints](auto&& error) mutable {
382 if (!m_promise)
383 return;
384 if (error) {
385 m_promise->rejectType<IDLInterface<OverconstrainedError>>(OverconstrainedError::create(WTFMove(error->badConstraint), WTFMove(error->message)));
386 return;
387 }
388 m_promise->resolve();
389 m_constraints = constraints.value_or(MediaTrackConstraints { });
390 };
391 m_private->applyConstraints(createMediaConstraints(constraints), WTFMove(completionHandler));
392}

404MediaProducer::MediaStateFlags MediaStreamTrack::mediaState() const
405{
406 if (m_ended || !isCaptureTrack())
407 return MediaProducer::IsNotPlaying;
408
409 auto* document = this->document();
410 if (!document || !document->page())
411 return MediaProducer::IsNotPlaying;
412
413 return sourceCaptureState(source());
414}
415
416MediaProducer::MediaStateFlags sourceCaptureState(RealtimeMediaSource& source)
417{
418 switch (source.deviceType()) {
419 case CaptureDevice::DeviceType::Microphone:
420 if (source.muted())
421 return MediaProducer::MediaState::HasMutedAudioCaptureDevice;
422 if (source.interrupted())
423 return MediaProducer::MediaState::HasInterruptedAudioCaptureDevice;
424 if (source.isProducingData())
425 return MediaProducer::MediaState::HasActiveAudioCaptureDevice;
426 break;
427 case CaptureDevice::DeviceType::Camera:
428 if (source.muted())
429 return MediaProducer::MediaState::HasMutedVideoCaptureDevice;
430 if (source.interrupted())
431 return MediaProducer::MediaState::HasInterruptedVideoCaptureDevice;
432 if (source.isProducingData())
433 return MediaProducer::MediaState::HasActiveVideoCaptureDevice;
434 break;
435 case CaptureDevice::DeviceType::Screen:
436 case CaptureDevice::DeviceType::Window:
437 if (source.muted())
438 return MediaProducer::MediaState::HasMutedDisplayCaptureDevice;
439 if (source.interrupted())
440 return MediaProducer::MediaState::HasInterruptedDisplayCaptureDevice;
441 if (source.isProducingData())
442 return MediaProducer::MediaState::HasActiveDisplayCaptureDevice;
443 break;
444 case CaptureDevice::DeviceType::Speaker:
445 case CaptureDevice::DeviceType::Unknown:
446 ASSERT_NOT_REACHED();
447 }
448
449 return MediaProducer::IsNotPlaying;
450}

452MediaProducer::MediaStateFlags MediaStreamTrack::captureState(Document& document)
453{
454 MediaProducer::MediaStateFlags state;
455 for (auto* captureTrack : allCaptureTracks()) {
456 if (captureTrack->document() != &document || captureTrack->ended())
457 continue;
458 state.add(sourceCaptureState(captureTrack->source()));
459 }
460 return state;
461}

477void MediaStreamTrack::updateCaptureAccordingToMutedState(Document& document)
478{
479#if PLATFORM(IOS_FAMILY)
480 auto* page = document.page();
481 if (!page)
482 return;
483
484 auto* activeAudioSource = RealtimeMediaSourceCenter::singleton().audioCaptureFactory().activeSource();
485 if (activeAudioSource && isSourceCapturingForTrackInDocument(*activeAudioSource, document)) {
486 bool pageMuted = page->mutedState().contains(MediaProducer::MutedState::AudioCaptureIsMuted);
487 activeAudioSource->setMuted(pageMuted || (document.hidden() && document.settings().interruptAudioOnPageVisibilityChangeEnabled()));
488 }
489
490 auto* activeVideoSource = RealtimeMediaSourceCenter::singleton().videoCaptureFactory().activeSource();
491 if (activeVideoSource && isSourceCapturingForTrackInDocument(*activeVideoSource, document)) {
492 bool pageMuted = page->mutedState().contains(MediaProducer::MutedState::VideoCaptureIsMuted);
493 activeVideoSource->setMuted(pageMuted || document.hidden());
494 }
495#else
496 for (auto* captureTrack : allCaptureTracks()) {
497 if (captureTrack->document() == &document && !captureTrack->ended())
498 captureTrack->updateToPageMutedState();
499 }
500#endif
501}

503void MediaStreamTrack::updateToPageMutedState()
504{
505 ASSERT(isCaptureTrack());
506 auto* page = document()->page();
507 if (!page)
508 return;
509
510 switch (source().deviceType()) {
511 case CaptureDevice::DeviceType::Microphone:
512 m_private->setMuted(page->mutedState().contains(MediaProducer::MutedState::AudioCaptureIsMuted));
513 break;
514 case CaptureDevice::DeviceType::Camera:
515 m_private->setMuted(page->mutedState().contains(MediaProducer::MutedState::VideoCaptureIsMuted));
516 break;
517 case CaptureDevice::DeviceType::Screen:
518 case CaptureDevice::DeviceType::Window:
519 m_private->setMuted(page->mutedState().contains(MediaProducer::MutedState::ScreenCaptureIsMuted));
520 break;
521 case CaptureDevice::DeviceType::Speaker:
522 case CaptureDevice::DeviceType::Unknown:
523 ASSERT_NOT_REACHED();
524 break;
525 }
526}

529{
530 switch (kind) {

533 case MediaProducer::MediaCaptureKind::AudioVideo:

541void MediaStreamTrack::endCapture(Document& document, MediaProducer::MediaCaptureKind kind)
542{
543 bool didEndCapture = false;
544 for (auto* captureTrack : allCaptureTracks()) {
545 if (captureTrack->document() != &document || !trackMatchesKind(captureTrack->privateTrack().type(), kind))
546 continue;
547 captureTrack->stopTrack(MediaStreamTrack::StopMode::PostEvent);
548 didEndCapture = true;
549 }
550 if (didEndCapture)
551 document.updateIsPlayingMedia();
552}

560void MediaStreamTrack::trackEnded(MediaStreamTrackPrivate&)
561{
562 if (m_isCaptureTrack && m_private->type() == RealtimeMediaSource::Type::Audio)
563 PlatformMediaSessionManager::sharedManager().removeAudioCaptureSource(*this);
564
565 // http://w3c.github.io/mediacapture-main/#life-cycle
566 // When a MediaStreamTrack track ends for any reason other than the stop() method being invoked, the User Agent must queue a task that runs the following steps:
567 // 1. If the track's readyState attribute has the value ended already, then abort these steps.
568 if (m_ended)
569 return;
570
571 ALWAYS_LOG(LOGIDENTIFIER);
572
573 if (m_isCaptureTrack && m_private->source().captureDidFail())
574 scriptExecutionContext()->addConsoleMessage(MessageSource::JS, MessageLevel::Error, "A MediaStreamTrack ended due to a capture failure"_s);
575
576 // 2. Set track's readyState attribute to ended.
577 m_ended = true;
578
579 if (scriptExecutionContext()->activeDOMObjectsAreSuspended() || scriptExecutionContext()->activeDOMObjectsAreStopped())
580 return;
581
582 ALWAYS_LOG(LOGIDENTIFIER, "firing 'ended' event");
583
584 // 3. Notify track's source that track is ended so that the source may be stopped, unless other MediaStreamTrack objects depend on it.
585 // 4. Fire a simple event named ended at the object.
586 dispatchEvent(Event::create(eventNames().endedEvent, Event::CanBubble::No, Event::IsCancelable::No));
587
588 for (auto& observer : m_observers)
589 observer->trackDidEnd();
590
591 configureTrackRendering();
592}

594void MediaStreamTrack::trackMutedChanged(MediaStreamTrackPrivate&)
595{
596 if (scriptExecutionContext()->activeDOMObjectsAreStopped() || m_ended)
597 return;
598
599 queueTaskKeepingObjectAlive(*this, TaskSource::Networking, [this, muted = m_private->muted()] {
600 if (!scriptExecutionContext() || scriptExecutionContext()->activeDOMObjectsAreStopped())
601 return;
602 m_muted = muted;
603 dispatchEvent(Event::create(muted ? eventNames().muteEvent : eventNames().unmuteEvent, Event::CanBubble::No, Event::IsCancelable::No));
604 });
605 configureTrackRendering();
606}

632void MediaStreamTrack::suspend(ReasonForSuspension reason)
633{
634 if (reason != ReasonForSuspension::BackForwardCache)
635 return;
636
637 // We only end capture tracks, other tracks (capture canvas, remote tracks) can still continue working.
638 if (m_ended || !isCaptureTrack())
639 return;
640
641 stopTrack();
642
643 queueTaskToDispatchEvent(*this, TaskSource::Networking, Event::create(eventNames().endedEvent, Event::CanBubble::No, Event::IsCancelable::No));
644}
