33#include <wtf/RefCounted.h>
34#include <wtf/text/WTFString.h>
35
36namespace WebCore {
37
38class OverconstrainedError : public RefCounted<OverconstrainedError> {
39public:
40 static Ref<OverconstrainedError> create(const String& constraint, const String& message)
41 {
42 return adoptRef(*new OverconstrainedError(constraint, message));
43 }
44
45 String constraint() const { return m_constraint; }
46 String message() const { return m_message; }
47 String name() const { return "OverconstrainedError"_s; }
48
49protected:
50 explicit OverconstrainedError(const String& constraint, const String& message)
51 : m_constraint(constraint)
52 , m_message(message)
53 {
54 }
55
56private:
57 String m_constraint;
58 String m_message;
59};

61} // namespace WebCore
