29#pragma once
30
31#if ENABLE(MEDIA_STREAM)
32
33#include "Event.h"
34#include "OverconstrainedError.h"
35#include <wtf/text/AtomString.h>
36
37namespace WebCore {
38
39class OverconstrainedErrorEvent final : public Event {
40 WTF_MAKE_ISO_ALLOCATED(OverconstrainedErrorEvent);
41public:
42 virtual ~OverconstrainedErrorEvent() = default;
43
44 static Ref<OverconstrainedErrorEvent> create(const AtomString& type, CanBubble canBubble, IsCancelable cancelable, OverconstrainedError* error)
45 {
46 return adoptRef(*new OverconstrainedErrorEvent(type, canBubble, cancelable, error));
47 }
48
49 struct Init : EventInit {
50 RefPtr<OverconstrainedError> error;
51 };
52
53 static Ref<OverconstrainedErrorEvent> create(const AtomString& type, const Init& initializer, IsTrusted isTrusted = IsTrusted::No)
54 {
55 return adoptRef(*new OverconstrainedErrorEvent(type, initializer, isTrusted));
56 }
57
58 OverconstrainedError* error() const { return m_error.get(); }
59 EventInterface eventInterface() const override { return OverconstrainedErrorEventInterfaceType; }
60
61private:
62 explicit OverconstrainedErrorEvent(const AtomString& type, CanBubble canBubble, IsCancelable cancelable, OverconstrainedError* error)
63 : Event(type, canBubble, cancelable)
64 , m_error(error)
65 {
66 }
67 OverconstrainedErrorEvent(const AtomString& type, const Init& initializer, IsTrusted isTrusted)
68 : Event(type, initializer, isTrusted)
69 , m_error(initializer.error)
70 {
71 }
72
73 RefPtr<OverconstrainedError> m_error;
74};
75
76} // namespace WebCore
77
78#endif // ENABLE(MEDIA_STREAM)
