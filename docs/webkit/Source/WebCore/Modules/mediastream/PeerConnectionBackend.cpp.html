97void PeerConnectionBackend::createOfferSucceeded(String&& sdp)

103 m_peerConnection.doTask([this, promise = WTFMove(m_offerAnswerPromise), sdp = filterSDP(WTFMove(sdp))]() mutable {

162static inline bool isLocalDescriptionTypeValidForState(RTCSdpType type, RTCSignalingState state)
163{
164 switch (state) {
165 case RTCSignalingState::Stable:
166 return type == RTCSdpType::Offer;
167 case RTCSignalingState::HaveLocalOffer:
168 return type == RTCSdpType::Offer;
169 case RTCSignalingState::HaveRemoteOffer:
170 return type == RTCSdpType::Answer || type == RTCSdpType::Pranswer;
171 case RTCSignalingState::HaveLocalPranswer:
172 return type == RTCSdpType::Answer || type == RTCSdpType::Pranswer;
173 default:
174 return false;
175 };
176
177 ASSERT_NOT_REACHED();
178 return false;
179}

254void PeerConnectionBackend::setRemoteDescriptionSucceeded()
255{
256 ASSERT(isMainThread());
257 ALWAYS_LOG(LOGIDENTIFIER, "Set remote description succeeded");
258 ASSERT(m_setDescriptionPromise);
259
260 auto promise = WTFMove(m_setDescriptionPromise);
261 auto events = WTFMove(m_pendingTrackEvents);
262 for (auto& event : events) {
263 auto& track = event.track.get();
264
265 m_peerConnection.dispatchEventWhenFeasible(RTCTrackEvent::create(eventNames().trackEvent, Event::CanBubble::No, Event::IsCancelable::No, WTFMove(event.receiver), WTFMove(event.track), WTFMove(event.streams), WTFMove(event.transceiver)));
266 ALWAYS_LOG(LOGIDENTIFIER, "Dispatched if feasible track of type ", track.source().type());
267
268 if (m_peerConnection.isClosed())
269 return;
270
271 // FIXME: As per spec, we should set muted to 'false' when starting to receive the content from network.
272 track.source().setMuted(false);
273 }
274
275 m_peerConnection.doTask([this, promise = WTFMove(promise)]() mutable {
276 if (m_peerConnection.isClosed())
277 return;
278
279 promise->resolve();
280 });
281}

306static String extractIPAddress(StringView sdp)
307{
308 unsigned counter = 0;
309 for (auto item : StringView { sdp }.split(' ')) {
310 if (++counter == 5)
311 return item.toString();
312 }
313 return { };
314}

316static inline bool shouldIgnoreIceCandidate(const RTCIceCandidate& iceCandidate)
317{
318 auto address = extractIPAddress(iceCandidate.candidate());
319 if (!address.endsWithIgnoringASCIICase(".local"_s))
320 return false;
321
322 if (!WTF::isVersion4UUID(StringView { address }.substring(0, address.length() - 6))) {
323 RELEASE_LOG_ERROR(WebRTC, "mDNS candidate is not a Version 4 UUID");
324 return true;
325 }
326 return false;
327}

329void PeerConnectionBackend::addIceCandidate(RTCIceCandidate* iceCandidate, DOMPromiseDeferred<void>&& promise)
330{
331 ASSERT(!m_peerConnection.isClosed());
332
333 if (!iceCandidate) {
334 endOfIceCandidates(WTFMove(promise));
335 return;
336 }
337
338 // FIXME: As per https://w3c.github.io/webrtc-pc/#dom-rtcpeerconnection-addicecandidate(), this check should be done before enqueuing the task.
339 if (iceCandidate->sdpMid().isNull() && !iceCandidate->sdpMLineIndex()) {
340 promise.reject(Exception { TypeError, "Trying to add a candidate that is missing both sdpMid and sdpMLineIndex"_s });
341 return;
342 }
343
344 if (shouldIgnoreIceCandidate(*iceCandidate)) {
345 promise.resolve();
346 return;
347 }
348
349 m_addIceCandidatePromise = WTF::makeUnique<DOMPromiseDeferred<void>>(WTFMove(promise));
350 doAddIceCandidate(*iceCandidate);
351}

401static String filterICECandidate(String&& sdp)
402{
403 ASSERT(!sdp.contains(" host "));
404
405 if (!sdp.contains(" raddr "))
406 return WTFMove(sdp);
407
408 bool skipNextItem = false;
409 bool isFirst = true;
410 StringBuilder filteredSDP;
411 sdp.split(' ', [&](StringView item) {
412 if (skipNextItem) {
413 skipNextItem = false;
414 return;
415 }
416 if (item == "raddr") {
417 filteredSDP.append(" raddr 0.0.0.0");
418 skipNextItem = true;
419 return;
420 }
421 if (item == "rport") {
422 filteredSDP.append(" rport 0");
423 skipNextItem = true;
424 return;
425 }
426 if (isFirst)
427 isFirst = false;
428 else
429 filteredSDP.append(' ');
430 filteredSDP.append(item);
431 });
432 return filteredSDP.toString();
433}

435String PeerConnectionBackend::filterSDP(String&& sdp) const
436{
437 if (!m_shouldFilterICECandidates)
438 return WTFMove(sdp);
439
440 StringBuilder filteredSDP;
441 sdp.split('\n', [this, &filteredSDP](StringView line) {
442 if (line.startsWith("c=IN IP4")) {
443 filteredSDP.append("c=IN IP4 0.0.0.0\r\n");
444 return;
445 }
446 if (line.startsWith("c=IN IP6")) {
447 filteredSDP.append("c=IN IP6 ::\r\n");
448 return;
449 }
450 if (!line.startsWith("a=candidate")) {
451 filteredSDP.append(line);
452 filteredSDP.append('\n');
453 return;
454 }
455 if (line.find(" host ", 11) == notFound) {
456 filteredSDP.append(filterICECandidate(line.toString()));
457 filteredSDP.append('\n');
458 return;
459 }
460
461 auto ipAddress = extractIPAddress(line);
462 auto mdnsName = m_ipAddressToMDNSNameMap.get(ipAddress);
463 if (mdnsName.isEmpty())
464 return;
465 auto sdp = line.toString();
466 sdp.replace(ipAddress, mdnsName);
467 filteredSDP.append(sdp);
468 filteredSDP.append('\n');
469 });
470 return filteredSDP.toString();
471}

473void PeerConnectionBackend::newICECandidate(String&& sdp, String&& mid, unsigned short sdpMLineIndex, String&& serverURL)
474{
475 m_peerConnection.doTask([logSiteIdentifier = LOGIDENTIFIER, this, sdp = WTFMove(sdp), mid = WTFMove(mid), sdpMLineIndex, serverURL = WTFMove(serverURL)]() mutable {
476 UNUSED_PARAM(logSiteIdentifier);
477 ALWAYS_LOG(logSiteIdentifier, "Gathered ice candidate:", sdp);
478 m_finishedGatheringCandidates = false;
479
480 if (!m_shouldFilterICECandidates) {
481 fireICECandidateEvent(RTCIceCandidate::create(WTFMove(sdp), WTFMove(mid), sdpMLineIndex), WTFMove(serverURL));
482 return;
483 }
484 if (sdp.find(" host ", 0) != notFound) {
485 // FIXME: We might need to clear all pending candidates when setting again local description.
486 m_pendingICECandidates.append(PendingICECandidate { String { sdp }, WTFMove(mid), sdpMLineIndex, WTFMove(serverURL) });
487 if (RuntimeEnabledFeatures::sharedFeatures().webRTCMDNSICECandidatesEnabled()) {
488 auto ipAddress = extractIPAddress(sdp);
489 // We restrict to IPv4 candidates for now.
490 if (ipAddress.contains('.'))
491 registerMDNSName(ipAddress);
492 }
493 return;
494 }
495 fireICECandidateEvent(RTCIceCandidate::create(filterICECandidate(WTFMove(sdp)), WTFMove(mid), sdpMLineIndex), WTFMove(serverURL));
496 });
497}

499void PeerConnectionBackend::doneGatheringCandidates()
500{
501 ASSERT(isMainThread());
502 ALWAYS_LOG(LOGIDENTIFIER, "Finished ice candidate gathering");
503 m_finishedGatheringCandidates = true;
504
505 if (m_waitingForMDNSRegistration)
506 return;
507
508 m_peerConnection.dispatchEventWhenFeasible(RTCPeerConnectionIceEvent::create(Event::CanBubble::No, Event::IsCancelable::No, nullptr, { }));
509 m_peerConnection.updateIceGatheringState(RTCIceGatheringState::Complete);
510 m_pendingICECandidates.clear();
511}

518void PeerConnectionBackend::registerMDNSName(const String& ipAddress)
519{
520 ++m_waitingForMDNSRegistration;
521 auto& document = downcast<Document>(*m_peerConnection.scriptExecutionContext());
522 auto& provider = document.page()->libWebRTCProvider();
523 provider.registerMDNSName(document.identifier(), ipAddress, [peerConnection = makeRef(m_peerConnection), this, ipAddress] (LibWebRTCProvider::MDNSNameOrError&& result) {
524 if (peerConnection->isStopped())
525 return;
526
527 --m_waitingForMDNSRegistration;
528 if (!result.has_value()) {
529 m_peerConnection.scriptExecutionContext()->addConsoleMessage(MessageSource::JS, MessageLevel::Warning, makeString("MDNS registration of a host candidate failed with error ", (unsigned)result.error()));
530 return;
531 }
532
533 this->finishedRegisteringMDNSName(ipAddress, result.value());
534 });
535}
536
537void PeerConnectionBackend::finishedRegisteringMDNSName(const String& ipAddress, const String& name)
538{
539 m_ipAddressToMDNSNameMap.add(ipAddress, name);
540 Vector<PendingICECandidate*> candidates;
541 for (auto& candidate : m_pendingICECandidates) {
542 if (candidate.sdp.find(ipAddress) != notFound) {
543 auto sdp = candidate.sdp;
544 sdp.replace(ipAddress, name);
545 fireICECandidateEvent(RTCIceCandidate::create(WTFMove(sdp), WTFMove(candidate.mid), candidate.sdpMLineIndex), WTFMove(candidate.serverURL));
546 candidates.append(&candidate);
547 }
548 }
549 m_pendingICECandidates.removeAllMatching([&] (const auto& candidate) {
550 return candidates.contains(&candidate);
551 });
552
553 if (!m_waitingForMDNSRegistration && m_finishedGatheringCandidates)
554 doneGatheringCandidates();
555}
