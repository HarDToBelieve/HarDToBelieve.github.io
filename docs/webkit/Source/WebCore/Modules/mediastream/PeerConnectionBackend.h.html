148 struct CertificateInformation {
149 enum class Type { RSASSAPKCS1v15, ECDSAP256 };
150 struct RSA {
151 unsigned modulusLength;
152 int publicExponent;
153 };
154
155 static CertificateInformation RSASSA_PKCS1_v1_5()
156 {
157 return CertificateInformation { Type::RSASSAPKCS1v15 };
158 }
159
160 static CertificateInformation ECDSA_P256()
161 {
162 return CertificateInformation { Type::ECDSAP256 };
163 }
164
165 explicit CertificateInformation(Type type)
166 : type(type)
167 {
168 }
169
170 Type type;
171 std::optional<double> expires;
172
173 std::optional<RSA> rsaParameters;
174 };
