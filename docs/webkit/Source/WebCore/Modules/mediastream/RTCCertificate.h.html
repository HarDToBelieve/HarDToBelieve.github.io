25#pragma once
26
27#if ENABLE(WEB_RTC)
28
29#include "SecurityOrigin.h"
30#include <wtf/RefCounted.h>
31#include <wtf/text/WTFString.h>
32
33namespace WebCore {
34
35class RTCCertificate : public RefCounted<RTCCertificate> {
36public:
37 struct DtlsFingerprint {
38 String algorithm;
39 String value;
40 };
41
42 static Ref<RTCCertificate> create(Ref<SecurityOrigin>&&, double expires, Vector<DtlsFingerprint>&&, String&& pemCertificate, String&& pemPrivateKey);
43
44 double expires() const { return m_expires; }
45 const Vector<DtlsFingerprint>& getFingerprints() const { return m_fingerprints; }
46
47 const String& pemCertificate() const { return m_pemCertificate; }
48 const String& pemPrivateKey() const { return m_pemPrivateKey; }
49 const SecurityOrigin& origin() const { return m_origin.get(); }
50
51private:
52 RTCCertificate(Ref<SecurityOrigin>&&, double expires, Vector<DtlsFingerprint>&&, String&& pemCertificate, String&& pemPrivateKey);
53
54 Ref<SecurityOrigin> m_origin;
55 double m_expires { 0 };
56 Vector<DtlsFingerprint> m_fingerprints;
57 String m_pemCertificate;
58 String m_pemPrivateKey;
59};
60
61} // namespace WebCore
62
63#endif // ENABLE(WEB_RTC)
