32#pragma once
33
34#if ENABLE(WEB_RTC)
35
36#include "RTCIceGatheringState.h"
37#include "RTCIceTransportState.h"
38#include "ScriptWrappable.h"
39#include <wtf/RefCounted.h>
40#include <wtf/RefPtr.h>
41
42namespace WebCore {
43
44class RTCIceTransport final : public RefCounted<RTCIceTransport>, public ScriptWrappable {
45 WTF_MAKE_ISO_ALLOCATED(RTCIceTransport);
46public:
47 static Ref<RTCIceTransport> create()
48 {
49 return adoptRef(*new RTCIceTransport());
50 }
51 virtual ~RTCIceTransport() = default;
52
53 RTCIceTransportState state() const { return m_transportState; }
54 void setState(RTCIceTransportState state) { m_transportState = state; }
55
56 RTCIceGatheringState gatheringState() const { return m_gatheringState; }
57 void setGatheringState(RTCIceGatheringState state) { m_gatheringState = state; }
58
59private:
60 RTCIceTransport() { };
61
62 RTCIceTransportState m_transportState { RTCIceTransportState::New };
63 RTCIceGatheringState m_gatheringState { RTCIceGatheringState::New };
64};
65
66} // namespace WebCore
67
68#endif // ENABLE(WEB_RTC)
