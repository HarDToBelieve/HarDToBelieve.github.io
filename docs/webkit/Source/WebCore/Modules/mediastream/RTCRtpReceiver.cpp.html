52RTCRtpReceiver::RTCRtpReceiver(PeerConnectionBackend& connection, Ref<MediaStreamTrack>&& track, std::unique_ptr<RTCRtpReceiverBackend>&& backend)
53 : m_track(WTFMove(track))
54 , m_backend(WTFMove(backend))
55 , m_connection(makeWeakPtr(&connection))
56#if !RELEASE_LOG_DISABLED
57 , m_logger(connection.logger())
58 , m_logIdentifier(connection.logIdentifier())
59#endif
60{
61}
62
63RTCRtpReceiver::~RTCRtpReceiver()
64{
65 if (m_transform)
66 m_transform->detachFromReceiver(*this);
67}
68
69void RTCRtpReceiver::stop()
70{
71 if (!m_backend)
72 return;
73
74 if (m_transform)
75 m_transform->detachFromReceiver(*this);
76
77 m_backend = nullptr;
78 m_track->stopTrack(MediaStreamTrack::StopMode::PostEvent);
79}

81void RTCRtpReceiver::getStats(Ref<DeferredPromise>&& promise)
82{
83 if (!m_connection) {
84 promise->reject(InvalidStateError);
85 return;
86 }
87 m_connection->getStats(*this, WTFMove(promise));
88}

94
95ExceptionOr<void> RTCRtpReceiver::setTransform(std::unique_ptr<RTCRtpTransform>&& transform)
96{
97 ALWAYS_LOG_IF(m_connection, LOGIDENTIFIER_RECEIVER);
98
99 if (transform && m_transform && *transform == *m_transform)
100 return { };
101 if (!transform) {
102 if (m_transform) {
103 m_transform->detachFromReceiver(*this);
104 m_transform = { };
105 }
106 return { };
107 }
108
109 if (transform->isAttached())
110 return Exception { InvalidStateError, "transform is already in use"_s };
111
112 transform->attachToReceiver(*this, m_transform.get());
113 m_transform = WTFMove(transform);
114
115 return { };
116}
117
118std::optional<RTCRtpTransform::Internal> RTCRtpReceiver::transform()
119{
120 if (!m_transform)
121 return { };
122 return m_transform->internalTransform();
123}
