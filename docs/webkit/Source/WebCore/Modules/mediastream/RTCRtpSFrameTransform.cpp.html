61void RTCRtpSFrameTransform::setEncryptionKey(CryptoKey& key, std::optional<uint64_t> keyId, DOMPromiseDeferred<void>&& promise)
62{
63 auto algorithm = key.algorithm();
64 if (!WTF::holds_alternative<CryptoKeyAlgorithm>(algorithm)) {
65 promise.reject(Exception { TypeError, "Invalid key"_s });
66 return;
67 }
68
69 if (WTF::get<CryptoKeyAlgorithm>(algorithm).name != "HKDF") {
70 promise.reject(Exception { TypeError, "Only HKDF is supported"_s });
71 return;
72 }
73
74 auto& rawKey = downcast<CryptoKeyRaw>(key);
75 promise.settle(m_transformer->setEncryptionKey(rawKey.key(), keyId));
76}

93bool RTCRtpSFrameTransform::isAttached() const
94{
95 return m_isAttached || (m_readable && m_readable->isLocked()) || (m_writable && m_writable->isLocked());
96}

98void RTCRtpSFrameTransform::initializeTransformer(RTCRtpTransformBackend& backend, Side side)
99{
100 ASSERT(!isAttached());
101
102 m_isAttached = true;
103 if (m_readable)
104 m_readable->lock();
105 if (m_writable)
106 m_writable->lock();
107
108 m_transformer->setIsEncrypting(side == Side::Sender);
109 m_transformer->setMediaType(backend.mediaType());
110
111 backend.setTransformableFrameCallback([transformer = m_transformer, backend = makeRef(backend)](auto&& frame) {
112 auto chunk = frame->data();
113 auto result = transformer->transform(chunk.data, chunk.size);
114
115 if (result.hasException()) {
116 RELEASE_LOG_ERROR(WebRTC, "RTCRtpSFrameTransform failed transforming a frame");
117 return;
118 }
119
120 frame->setData({ result.returnValue().data(), result.returnValue().size() });
121
122 backend->processTransformedFrame(frame.get());
123 });
124}

141template<typename Frame>
142void transformFrame(Frame& frame, JSDOMGlobalObject& globalObject, RTCRtpSFrameTransformer& transformer, SimpleReadableStreamSource& source)
143{
144 auto chunk = frame.rtcFrame().data();
145 auto result = transformer.transform(chunk.data, chunk.size);
146 RELEASE_LOG_ERROR_IF(result.hasException(), WebRTC, "RTCRtpSFrameTransform failed transforming a frame");
147
148 RTCRtpTransformableFrame::Data transformedChunk;
149 // In case of error, we just pass along the frame with empty data.
150 if (!result.hasException())
151 transformedChunk = { result.returnValue().data(), result.returnValue().size() };
152
153 frame.rtcFrame().setData(transformedChunk);
154 source.enqueue(toJS(&globalObject, &globalObject, frame));
155}
156
157static void transformFrame(const uint8_t* data, size_t size, JSDOMGlobalObject& globalObject, RTCRtpSFrameTransformer& transformer, SimpleReadableStreamSource& source)
158{
159 auto result = transformer.transform(data, size);
160 RELEASE_LOG_ERROR_IF(result.hasException(), WebRTC, "RTCRtpSFrameTransform failed transforming a frame");
161
162 auto buffer = result.hasException() ? SharedBuffer::create() : SharedBuffer::create(result.releaseReturnValue());
163 source.enqueue(toJS(&globalObject, &globalObject, buffer->tryCreateArrayBuffer().get()));
164}

166void RTCRtpSFrameTransform::createStreams(JSC::JSGlobalObject& globalObject)
167{
168 m_readableStreamSource = SimpleReadableStreamSource::create();
169 auto readable = ReadableStream::create(globalObject, m_readableStreamSource.copyRef());
170 if (readable.hasException())
171 return;
172
173 auto writable = WritableStream::create(globalObject, SimpleWritableStreamSink::create([transformer = m_transformer, readableStreamSource = m_readableStreamSource](auto& context, auto value) -> ExceptionOr<void> {
174 if (!context.globalObject())
175 return Exception { InvalidStateError };
176 auto& globalObject = *JSC::jsCast<JSDOMGlobalObject*>(context.globalObject());
177 auto scope = DECLARE_THROW_SCOPE(globalObject.vm());
178
179 auto frame = convert<IDLUnion<IDLArrayBuffer, IDLArrayBufferView, IDLInterface<RTCEncodedAudioFrame>, IDLInterface<RTCEncodedVideoFrame>>>(globalObject, value);
180 if (scope.exception())
181 return Exception { ExistingExceptionError };
182
183 // We do not want to throw any exception in the transform to make sure we do not error the transform.
184 WTF::switchOn(frame, [&](RefPtr<RTCEncodedAudioFrame>& value) {
185 transformFrame(*value, globalObject, transformer.get(), *readableStreamSource);
186 }, [&](RefPtr<RTCEncodedVideoFrame>& value) {
187 transformFrame(*value, globalObject, transformer.get(), *readableStreamSource);
188 }, [&](RefPtr<ArrayBuffer>& value) {
189 transformFrame(static_cast<const uint8_t*>(value->data()), value->byteLength(), globalObject, transformer.get(), *readableStreamSource);
190 }, [&](RefPtr<ArrayBufferView>& value) {
191 transformFrame(static_cast<const uint8_t*>(value->data()), value->byteLength(), globalObject, transformer.get(), *readableStreamSource);
192 });
193 return { };
194 }));
195 if (writable.hasException())
196 return;
197
198 m_readable = readable.releaseReturnValue();
199 m_writable = writable.releaseReturnValue();
200 if (m_isAttached) {
201 m_readable->lock();
202 m_writable->lock();
203 }
204}

206ExceptionOr<RefPtr<ReadableStream>> RTCRtpSFrameTransform::readable()
207{
208 auto* context = scriptExecutionContext();
209 if (!context)
210 return Exception { InvalidStateError };
211
212 auto* globalObject = context->globalObject();
213 if (!globalObject)
214 return Exception { InvalidStateError };
215
216 if (!m_readable)
217 createStreams(*globalObject);
218
219 return m_readable.copyRef();
220}
221
222ExceptionOr<RefPtr<WritableStream>> RTCRtpSFrameTransform::writable()
223{
224 auto* context = scriptExecutionContext();
225 if (!context)
226 return Exception { InvalidStateError };
227
228 auto* globalObject = context->globalObject();
229 if (!globalObject)
230 return Exception { InvalidStateError };
231
232 if (!m_writable)
233 createStreams(*globalObject);
234
235 return m_writable.copyRef();
236}
