40class CryptoKey;
41class RTCRtpTransformBackend;
42class ReadableStream;
43class SimpleReadableStreamSource;
44class WritableStream;

46class RTCRtpSFrameTransform : public RefCounted<RTCRtpSFrameTransform>, private ContextDestructionObserver {
47public:
48 enum class Role { Encrypt, Decrypt };
49 using CompatibilityMode = RTCRtpSFrameTransformer::CompatibilityMode;
50 struct Options {
51 Role role { Role::Encrypt };
52 uint64_t authenticationSize { 10 };
53 CompatibilityMode compatibilityMode { CompatibilityMode::None };
54 };
55
56 static Ref<RTCRtpSFrameTransform> create(ScriptExecutionContext& context, Options options) { return adoptRef(*new RTCRtpSFrameTransform(context, options)); }
57 ~RTCRtpSFrameTransform();
58
59 void setEncryptionKey(CryptoKey&, std::optional<uint64_t>, DOMPromiseDeferred<void>&&);
60
61 bool isAttached() const;
62 void initializeBackendForReceiver(RTCRtpTransformBackend&);
63 void initializeBackendForSender(RTCRtpTransformBackend&);
64 void willClearBackend(RTCRtpTransformBackend&);
65
66 WEBCORE_EXPORT void setCounterForTesting(uint64_t);
67 WEBCORE_EXPORT uint64_t counterForTesting() const;
68 WEBCORE_EXPORT uint64_t keyIdForTesting() const;
69
70 ExceptionOr<RefPtr<ReadableStream>> readable();
71 ExceptionOr<RefPtr<WritableStream>> writable();
72
73private:
74 RTCRtpSFrameTransform(ScriptExecutionContext&, Options);
75
76 enum class Side { Sender, Receiver };
77 void initializeTransformer(RTCRtpTransformBackend&, Side);
78 void createStreams(JSC::JSGlobalObject&);
79
80 bool m_isAttached { false };
81 Ref<RTCRtpSFrameTransformer> m_transformer;
82 RefPtr<ReadableStream> m_readable;
83 RefPtr<WritableStream> m_writable;
84 RefPtr<SimpleReadableStreamSource> m_readableStreamSource;
85};
