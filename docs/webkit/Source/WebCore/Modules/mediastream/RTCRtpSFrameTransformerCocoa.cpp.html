36ExceptionOr<Vector<uint8_t>> RTCRtpSFrameTransformer::computeSaltKey(const Vector<uint8_t>& rawKey)
37{
38 uint8_t usage[] = "SFrame10";
39 uint8_t info[] = "salt";
40 return deriveHDKFSHA256Bits(rawKey.data(), 16, usage, sizeof(usage) - 1, info, sizeof(info) - 1, 96);
41}

43static ExceptionOr<Vector<uint8_t>> createBaseSFrameKey(const Vector<uint8_t>& rawKey)
44{
45 uint8_t usage[] = "SFrame10";
46 uint8_t info[] = "key";
47 return deriveHDKFSHA256Bits(rawKey.data(), 16, usage, sizeof(usage) - 1, info, sizeof(info) - 1, 128);

50ExceptionOr<Vector<uint8_t>> RTCRtpSFrameTransformer::computeAuthenticationKey(const Vector<uint8_t>& rawKey)
51{
52 auto key = createBaseSFrameKey(rawKey);
53 if (key.hasException())
54 return key;
55
56 uint8_t usage[] = "SFrame10 AES CM AEAD";
57 uint8_t info[] = "auth";
58 return deriveHDKFSHA256Bits(key.returnValue().data(), 16, usage, sizeof(usage) - 1, info, sizeof(info) - 1, 256);
59}

82static inline Vector<uint8_t, 8> encodeBigEndian(uint64_t value)
83{
84 Vector<uint8_t, 8> result(8);
85 for (int i = 7; i >= 0; --i) {
86 result.data()[i] = value & 0xff;
87 value = value >> 8;
88 }
89 return result;
90}

92Vector<uint8_t> RTCRtpSFrameTransformer::computeEncryptedDataSignature(const Vector<uint8_t>& nonce, const uint8_t* header, size_t headerSize, const uint8_t* data, size_t dataSize, const Vector<uint8_t>& key)
93{
94 auto headerLength = encodeBigEndian(headerSize);
95 auto dataLength = encodeBigEndian(dataSize);
96
97 Vector<uint8_t> result(CC_SHA256_DIGEST_LENGTH);
98 CCHmacContext context;
99 CCHmacInit(&context, kCCHmacAlgSHA256, key.data(), key.size());
100 CCHmacUpdate(&context, headerLength.data(), headerLength.size());
101 CCHmacUpdate(&context, dataLength.data(), dataLength.size());
102 CCHmacUpdate(&context, nonce.data(), 12);
103 CCHmacUpdate(&context, header, headerSize);
104 CCHmacUpdate(&context, data, dataSize);
105 CCHmacFinal(&context, result.data());
106
107 return result;
108}
