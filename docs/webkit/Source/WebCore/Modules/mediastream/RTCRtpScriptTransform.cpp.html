45ExceptionOr<Ref<RTCRtpScriptTransform>> RTCRtpScriptTransform::create(JSC::JSGlobalObject& state, Worker& worker, JSC::JSValue options, Vector<JSC::Strong<JSC::JSObject>>&& transfer)
46{
47 if (!worker.scriptExecutionContext())
48 return Exception { InvalidStateError, "Worker frame is detached"_s };
49
50 auto* context = JSC::jsCast<JSDOMGlobalObject*>(&state)->scriptExecutionContext();
51 if (!context)
52 return Exception { InvalidStateError, "Invalid context"_s };
53
54 Vector<RefPtr<MessagePort>> transferredPorts;
55 auto serializedOptions = SerializedScriptValue::create(state, options, WTFMove(transfer), transferredPorts);
56 if (serializedOptions.hasException())
57 return serializedOptions.releaseException();
58
59 auto channels = MessagePort::disentanglePorts(WTFMove(transferredPorts));
60 if (channels.hasException())
61 return channels.releaseException();
62
63 auto transform = adoptRef(*new RTCRtpScriptTransform(*context, makeRef(worker)));
64 transform->suspendIfNeeded();
65
66 worker.createRTCRtpScriptTransformer(transform, { serializedOptions.releaseReturnValue(), channels.releaseReturnValue() });
67
68 return transform;
69}

82void RTCRtpScriptTransform::setTransformer(RTCRtpScriptTransformer& transformer)
83{
84 ASSERT(!isMainThread());
85 {
86 Locker locker { m_transformerLock };
87 ASSERT(!m_isTransformerInitialized);
88 m_isTransformerInitialized = true;
89 m_transformer = makeWeakPtr(transformer);
90 }
91 transformer.startPendingActivity();
92 callOnMainThread([this, protectedThis = makeRef(*this)]() mutable {
93 if (m_backend)
94 setupTransformer(m_backend.releaseNonNull());
95 });
96}

113void RTCRtpScriptTransform::initializeTransformer(RTCRtpTransformBackend& backend)
114{
115 m_isAttached = true;
116 if (!setupTransformer(makeRef(backend)))
117 m_backend = makeRef(backend);
118}
119
120bool RTCRtpScriptTransform::setupTransformer(Ref<RTCRtpTransformBackend>&& backend)
121{
122 Locker locker { m_transformerLock };
123 if (!m_isTransformerInitialized)
124 return false;
125
126 m_worker->postTaskToWorkerGlobalScope([transformer = m_transformer, backend = WTFMove(backend)](auto&) mutable {
127 if (transformer)
128 transformer->start(WTFMove(backend));
129 });
130 return true;
131}
132
133void RTCRtpScriptTransform::clear(RTCRtpScriptTransformer::ClearCallback clearCallback)
134{
135 m_isAttached = false;
136
137 Locker locker { m_transformerLock };
138 m_isTransformerInitialized = false;
139 m_worker->postTaskToWorkerGlobalScope([transformer = WTFMove(m_transformer), clearCallback](auto&) mutable {
140 if (transformer)
141 transformer->clear(clearCallback);
142 });
143}
