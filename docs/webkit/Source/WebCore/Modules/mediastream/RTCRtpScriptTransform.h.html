39namespace WebCore {
40
41class RTCRtpScriptTransformer;
42class RTCRtpTransformBackend;
43class Worker;
44
45class RTCRtpScriptTransform final
46 : public ThreadSafeRefCounted<RTCRtpScriptTransform, WTF::DestructionThread::Main>
47 , public ActiveDOMObject
48 , public EventTargetWithInlineData {
49 WTF_MAKE_ISO_ALLOCATED(RTCRtpScriptTransform);
50public:
51 static ExceptionOr<Ref<RTCRtpScriptTransform>> create(JSC::JSGlobalObject&, Worker&, JSC::JSValue, Vector<JSC::Strong<JSC::JSObject>>&&);
52 ~RTCRtpScriptTransform();
53
54 void setTransformer(RTCRtpScriptTransformer&);
55
56 bool isAttached() const { return m_isAttached; }
57 void initializeBackendForReceiver(RTCRtpTransformBackend&);
58 void initializeBackendForSender(RTCRtpTransformBackend&);
59 void willClearBackend(RTCRtpTransformBackend&);
60 void backendTransferedToNewTransform() { clear(RTCRtpScriptTransformer::ClearCallback::No); }
61
62 using ThreadSafeRefCounted::ref;
63 using ThreadSafeRefCounted::deref;
64
65private:
66 RTCRtpScriptTransform(ScriptExecutionContext&, Ref<Worker>&&);
67
68 void initializeTransformer(RTCRtpTransformBackend&);
69 bool setupTransformer(Ref<RTCRtpTransformBackend>&&);
70 void clear(RTCRtpScriptTransformer::ClearCallback);
71
72 // ActiveDOMObject
73 const char* activeDOMObjectName() const final { return "RTCRtpScriptTransform"; }
74
75 // EventTarget
76 EventTargetInterface eventTargetInterface() const final { return RTCRtpScriptTransformEventTargetInterfaceType; }
77 ScriptExecutionContext* scriptExecutionContext() const final { return ActiveDOMObject::scriptExecutionContext(); }
78 void refEventTarget() final { ref(); }
79 void derefEventTarget() final { deref(); }
80
81 Ref<Worker> m_worker;
82
83 bool m_isAttached { false };
84 RefPtr<RTCRtpTransformBackend> m_backend;
85
86 Lock m_transformerLock;
87 bool m_isTransformerInitialized WTF_GUARDED_BY_LOCK(m_transformerLock) { false };
88 WeakPtr<RTCRtpScriptTransformer> m_transformer WTF_GUARDED_BY_LOCK(m_transformerLock);
89};
90
91} // namespace WebCore

93#endif // ENABLE(WEB_RTC)
