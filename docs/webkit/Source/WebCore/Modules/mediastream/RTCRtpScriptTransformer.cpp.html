44ExceptionOr<Ref<RTCRtpScriptTransformer>> RTCRtpScriptTransformer::create(ScriptExecutionContext& context, MessageWithMessagePorts&& options)
45{
46 if (!context.globalObject())
47 return Exception { InvalidStateError };
48
49 auto& globalObject = *JSC::jsCast<JSDOMGlobalObject*>(context.globalObject());
50 JSC::JSLockHolder lock(globalObject.vm());
51 auto readableSource = SimpleReadableStreamSource::create();
52 auto readable = ReadableStream::create(globalObject, readableSource.copyRef());
53 if (readable.hasException())
54 return readable.releaseException();
55 if (!options.message)
56 return Exception { InvalidStateError };
57
58 auto ports = MessagePort::entanglePorts(context, WTFMove(options.transferredPorts));
59 auto transformer = adoptRef(*new RTCRtpScriptTransformer(context, options.message.releaseNonNull(), WTFMove(ports), readable.releaseReturnValue(), WTFMove(readableSource)));
60 transformer->suspendIfNeeded();
61 return transformer;
62}

82ExceptionOr<Ref<WritableStream>> RTCRtpScriptTransformer::writable()
83{
84 if (!m_writable) {
85 auto* context = downcast<WorkerGlobalScope>(scriptExecutionContext());
86 if (!context || !context->globalObject())
87 return Exception { InvalidStateError };
88
89 auto& globalObject = *JSC::jsCast<JSDOMGlobalObject*>(context->globalObject());
90 auto writableOrException = WritableStream::create(globalObject, SimpleWritableStreamSink::create([transformer = makeRef(*this)](auto& context, auto value) -> ExceptionOr<void> {
91 if (!transformer->m_backend)
92 return Exception { InvalidStateError };
93
94 auto& globalObject = *context.globalObject();
95
96 auto scope = DECLARE_THROW_SCOPE(globalObject.vm());
97 auto frame = convert<IDLUnion<IDLInterface<RTCEncodedAudioFrame>, IDLInterface<RTCEncodedVideoFrame>>>(globalObject, value);
98
99 if (scope.exception())
100 return Exception { ExistingExceptionError };
101
102 auto rtcFrame = WTF::switchOn(frame, [&](RefPtr<RTCEncodedAudioFrame>& value) {
103 return makeRef(value->rtcFrame());
104 }, [&](RefPtr<RTCEncodedVideoFrame>& value) {
105 return makeRef(value->rtcFrame());
106 });
107
108 // If no data, skip the frame since there is nothing to packetize or decode.
109 if (rtcFrame->data().data)
110 transformer->m_backend->processTransformedFrame(rtcFrame.get());
111 return { };
112 }));
113 if (writableOrException.hasException())
114 return writableOrException;
115 m_writable = writableOrException.releaseReturnValue();
116 }
117 return makeRef(*m_writable);
118}

120void RTCRtpScriptTransformer::start(Ref<RTCRtpTransformBackend>&& backend)
121{
122 m_backend = WTFMove(backend);
123
124 auto& context = downcast<WorkerGlobalScope>(*scriptExecutionContext());
125 m_backend->setTransformableFrameCallback([readableSource = makeWeakPtr(m_readableSource.get()), isAudio = m_backend->mediaType() == RTCRtpTransformBackend::MediaType::Audio, thread = makeRef(context.thread())](auto&& frame) mutable {
126 thread->runLoop().postTaskForMode([readableSource, isAudio, frame = WTFMove(frame)](auto& context) mutable {
127 if (!readableSource)
128 return;
129
130 auto& globalObject = *JSC::jsCast<JSDOMGlobalObject*>(context.globalObject());
131 auto& vm = globalObject.vm();
132 JSC::JSLockHolder lock(vm);
133
134 auto value = isAudio ? toJS(&globalObject, &globalObject, RTCEncodedAudioFrame::create(WTFMove(frame))) : toJS(&globalObject, &globalObject, RTCEncodedVideoFrame::create(WTFMove(frame)));
135 readableSource->enqueue(value);
136 }, WorkerRunLoop::defaultMode());
137 });
138}

140void RTCRtpScriptTransformer::clear(ClearCallback clearCallback)
141{
142 if (m_backend && clearCallback == ClearCallback::Yes)
143 m_backend->clearTransformableFrameCallback();
144 m_backend = nullptr;
145 stopPendingActivity();
146}

148ExceptionOr<void> RTCRtpScriptTransformer::requestKeyFrame()
149{
150 if (!m_backend)
151 return Exception { InvalidStateError, "Not attached to a receiver or sender"_s };
152
153 if (m_backend->mediaType() != RTCRtpTransformBackend::MediaType::Video)
154 return Exception { InvalidStateError, "Cannot request key frame on audio sender"_s };
155
156 m_backend->requestKeyFrame();
157 return { };
158}
