110void RTCRtpSender::replaceTrack(RefPtr<MediaStreamTrack>&& withTrack, Ref<DeferredPromise>&& promise)
111{
112 if (withTrack && m_trackKind != withTrack->kind()) {
113 promise->reject(TypeError);
114 return;
115 }
116
117 if (!m_connection) {
118 promise->reject(InvalidStateError);
119 return;
120 }
121
122 m_connection->chainOperation(WTFMove(promise), [this, weakThis = makeWeakPtr(this), withTrack = WTFMove(withTrack)](auto&& promise) mutable {
123 if (!weakThis)
124 return;
125 if (isStopped()) {
126 promise->reject(InvalidStateError);
127 return;
128 }
129
130 if (!m_backend->replaceTrack(*this, withTrack.get())) {
131 promise->reject(InvalidModificationError);
132 return;
133 }
134
135 auto* context = m_connection->scriptExecutionContext();
136 if (!context)
137 return;
138
139 context->postTask([this, protectedThis = makeRef(*this), withTrack = WTFMove(withTrack), promise = WTFMove(promise)](auto&) mutable {
140 if (!m_connection || m_connection->isClosed())
141 return;
142
143 m_track = WTFMove(withTrack);
144 promise->resolve();
145 });
146 });
147}

207std::optional<RTCRtpTransceiverDirection> RTCRtpSender::currentTransceiverDirection() const
208{
209 if (!m_connection)
210 return { };
211
212 RTCRtpTransceiver* senderTransceiver = nullptr;
213 for (auto& transceiver : m_connection->currentTransceivers()) {
214 if (&transceiver->sender() == this) {
215 senderTransceiver = transceiver.get();
216 break;
217 }
218 }
219
220 if (!senderTransceiver)
221 return { };
222
223 return senderTransceiver->currentDirection();
224}

226ExceptionOr<void> RTCRtpSender::setTransform(std::unique_ptr<RTCRtpTransform>&& transform)
227{
228 ALWAYS_LOG_IF(m_connection, LOGIDENTIFIER_SENDER);
229
230 if (transform && m_transform && *transform == *m_transform)
231 return { };
232 if (!transform) {
233 if (m_transform) {
234 m_transform->detachFromSender(*this);
235 m_transform = { };
236 }
237 return { };
238 }
239
240 if (transform->isAttached())
241 return Exception { InvalidStateError, "transform is already in use"_s };
242
243 transform->attachToSender(*this, m_transform.get());
244 m_transform = WTFMove(transform);
245
246 return { };
247}
