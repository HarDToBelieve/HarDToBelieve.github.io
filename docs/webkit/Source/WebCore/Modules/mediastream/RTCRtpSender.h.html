32
33#if ENABLE(WEB_RTC)
34
35#include "JSDOMPromiseDeferred.h"
36#include "MediaStreamTrack.h"
37#include "RTCRtpSenderBackend.h"
38#include "RTCRtpTransceiverDirection.h"
39#include "RTCRtpTransform.h"
40#include "ScriptWrappable.h"
41#include <wtf/WeakPtr.h>
42
43namespace WebCore {
44
45class MediaStream;
46class RTCDTMFSender;
47class RTCPeerConnection;
48struct RTCRtpCapabilities;
49
50class RTCRtpSender final : public RefCounted<RTCRtpSender>
51 , public ScriptWrappable
52 , public CanMakeWeakPtr<RTCRtpSender>
53#if !RELEASE_LOG_DISABLED
54 , private LoggerHelper
55#endif
56 {
57 WTF_MAKE_ISO_ALLOCATED(RTCRtpSender);
58public:
59 static Ref<RTCRtpSender> create(RTCPeerConnection&, Ref<MediaStreamTrack>&&, std::unique_ptr<RTCRtpSenderBackend>&&);
60 static Ref<RTCRtpSender> create(RTCPeerConnection&, String&& trackKind, std::unique_ptr<RTCRtpSenderBackend>&&);
61 ~RTCRtpSender();
62
63 static std::optional<RTCRtpCapabilities> getCapabilities(ScriptExecutionContext&, const String& kind);
64
65 MediaStreamTrack* track() { return m_track.get(); }
66
67 const String& trackId() const { return m_trackId; }
68 const String& trackKind() const { return m_trackKind; }
69
70 ExceptionOr<void> setMediaStreamIds(const Vector<String>&);
71 ExceptionOr<void> setStreams(const Vector<std::reference_wrapper<MediaStream>>&);
72
73 bool isStopped() const { return !m_backend; }
74 void stop();
75 void setTrack(Ref<MediaStreamTrack>&&);
76 void setTrackToNull();
77
78 void replaceTrack(RefPtr<MediaStreamTrack>&&, Ref<DeferredPromise>&&);
79
80 RTCRtpSendParameters getParameters();
81 void setParameters(const RTCRtpSendParameters&, DOMPromiseDeferred<void>&&);
82
83 RTCRtpSenderBackend* backend() { return m_backend.get(); }
84
85 void getStats(Ref<DeferredPromise>&&);
86
87 bool isCreatedBy(const RTCPeerConnection&) const;
88
89 RTCDTMFSender* dtmf();
90 std::optional<RTCRtpTransceiverDirection> currentTransceiverDirection() const;
91
92 std::optional<RTCRtpTransform::Internal> transform();
93 ExceptionOr<void> setTransform(std::unique_ptr<RTCRtpTransform>&&);
94
95private:
96 RTCRtpSender(RTCPeerConnection&, String&& trackKind, std::unique_ptr<RTCRtpSenderBackend>&&);
97
98#if !RELEASE_LOG_DISABLED
99 const Logger& logger() const final { return m_logger.get(); }
100 const void* logIdentifier() const final { return m_logIdentifier; }
101 const char* logClassName() const final { return "RTCRtpSender"; }
102 WTFLogChannel& logChannel() const final;
103#endif
104
105 RefPtr<MediaStreamTrack> m_track;
106 String m_trackId;
107 String m_trackKind;
108 std::unique_ptr<RTCRtpSenderBackend> m_backend;
109 WeakPtr<RTCPeerConnection> m_connection;
110 RefPtr<RTCDTMFSender> m_dtmfSender;
111 std::unique_ptr<RTCRtpTransform> m_transform;
112#if !RELEASE_LOG_DISABLED
113 Ref<const Logger> m_logger;
114 const void* m_logIdentifier { nullptr };
115#endif
116};
117
118} // namespace WebCore

120#endif // ENABLE(WEB_RTC)
