45RTCRtpTransceiver::RTCRtpTransceiver(Ref<RTCRtpSender>&& sender, Ref<RTCRtpReceiver>&& receiver, std::unique_ptr<RTCRtpTransceiverBackend>&& backend)
46 : m_direction(RTCRtpTransceiverDirection::Sendrecv)
47 , m_sender(WTFMove(sender))
48 , m_receiver(WTFMove(receiver))
49 , m_iceTransport(RTCIceTransport::create())
50 , m_backend(WTFMove(backend))
51{
52}

54String RTCRtpTransceiver::mid() const
55{
56 return m_backend ? m_backend->mid() : String { };
57}

59bool RTCRtpTransceiver::hasSendingDirection() const
60{
61 return m_direction == RTCRtpTransceiverDirection::Sendrecv || m_direction == RTCRtpTransceiverDirection::Sendonly;
62}
63
64RTCRtpTransceiverDirection RTCRtpTransceiver::direction() const
65{
66 if (!m_backend)
67 return m_direction;
68 return m_backend->direction();
69}
70
71std::optional<RTCRtpTransceiverDirection> RTCRtpTransceiver::currentDirection() const
72{
73 if (!m_backend)
74 return std::nullopt;
75 return m_backend->currentDirection();
76}

78void RTCRtpTransceiver::setDirection(RTCRtpTransceiverDirection direction)
79{
80 m_direction = direction;
81 if (m_backend)
82 m_backend->setDirection(direction);
83}
84
85
86void RTCRtpTransceiver::enableSendingDirection()
87{
88 if (m_direction == RTCRtpTransceiverDirection::Recvonly)
89 m_direction = RTCRtpTransceiverDirection::Sendrecv;
90 else if (m_direction == RTCRtpTransceiverDirection::Inactive)
91 m_direction = RTCRtpTransceiverDirection::Sendonly;
92}
93
94void RTCRtpTransceiver::disableSendingDirection()
95{
96 if (m_direction == RTCRtpTransceiverDirection::Sendrecv)
97 m_direction = RTCRtpTransceiverDirection::Recvonly;
98 else if (m_direction == RTCRtpTransceiverDirection::Sendonly)
99 m_direction = RTCRtpTransceiverDirection::Inactive;
100}
101
102void RTCRtpTransceiver::setConnection(RTCPeerConnection& connection)
103{
104 ASSERT(!m_connection);
105 m_connection = makeWeakPtr(connection);
106}
107
108ExceptionOr<void> RTCRtpTransceiver::stop()
109{
110 if (!m_connection || m_connection->isClosed())
111 return Exception { InvalidStateError, "RTCPeerConnection is closed"_s };
112
113 if (m_stopped)
114 return { };
115
116 m_stopped = true;
117 m_receiver->stop();
118 m_sender->stop();
119 if (m_backend)
120 m_backend->stop();
121
122 m_connection->scheduleNegotiationNeededEvent();
123 return { };
124}
125
126ExceptionOr<void> RTCRtpTransceiver::setCodecPreferences(const Vector<RTCRtpCodecCapability>& codecs)
127{
128 if (!m_backend)
129 return { };
130 return m_backend->setCodecPreferences(codecs);
131}
132
133bool RTCRtpTransceiver::stopped() const
134{
135 if (m_backend)
136 return m_backend->stopped();
137 return m_stopped;
138}
139
140void RtpTransceiverSet::append(Ref<RTCRtpTransceiver>&& transceiver)
141{
142 m_transceivers.append(WTFMove(transceiver));
143}
144
145Vector<std::reference_wrapper<RTCRtpSender>> RtpTransceiverSet::senders() const
146{
147 Vector<std::reference_wrapper<RTCRtpSender>> senders;
148 for (auto& transceiver : m_transceivers) {
149 if (transceiver->stopped())
150 continue;
151 senders.append(transceiver->sender());
152 }
153 return senders;
154}
155
156Vector<std::reference_wrapper<RTCRtpReceiver>> RtpTransceiverSet::receivers() const
157{
158 Vector<std::reference_wrapper<RTCRtpReceiver>> receivers;
159 for (auto& transceiver : m_transceivers) {
160 if (transceiver->stopped())
161 continue;
162 receivers.append(transceiver->receiver());
163 }
164 return receivers;
165}
