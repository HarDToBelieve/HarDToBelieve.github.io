36std::unique_ptr<RTCRtpTransform> RTCRtpTransform::from(std::optional<Internal>&& internal)
37{
38 if (!internal)
39 return nullptr;
40 return makeUnique<RTCRtpTransform>(WTFMove(*internal));
41}
42
43RTCRtpTransform::RTCRtpTransform(Internal&& transform)
44 : m_transform(WTFMove(transform))
45{
46}
47
48RTCRtpTransform::~RTCRtpTransform()
49{
50 clearBackend();
51}
52
53bool RTCRtpTransform::isAttached() const
54{
55 return switchOn(m_transform, [&](const RefPtr<RTCRtpSFrameTransform>& sframeTransform) {
56 return sframeTransform->isAttached();
57 }, [&](const RefPtr<RTCRtpScriptTransform>& scriptTransform) {
58 return scriptTransform->isAttached();
59 });
60}
61
62void RTCRtpTransform::attachToReceiver(RTCRtpReceiver& receiver, RTCRtpTransform* previousTransform)
63{
64 ASSERT(!isAttached());
65
66 if (previousTransform)
67 m_backend = previousTransform->takeBackend();
68 else if (auto* backend = receiver.backend())
69 m_backend = backend->rtcRtpTransformBackend();
70 else
71 return;
72
73 switchOn(m_transform, [&](RefPtr<RTCRtpSFrameTransform>& sframeTransform) {
74 sframeTransform->initializeBackendForReceiver(*m_backend);
75 }, [&](RefPtr<RTCRtpScriptTransform>& scriptTransform) {
76 scriptTransform->initializeBackendForReceiver(*m_backend);
77 });
78}
79
80void RTCRtpTransform::attachToSender(RTCRtpSender& sender, RTCRtpTransform* previousTransform)
81{
82 ASSERT(!isAttached());
83
84 if (previousTransform)
85 m_backend = previousTransform->takeBackend();
86 else if (auto* backend = sender.backend())
87 m_backend = backend->rtcRtpTransformBackend();
88 else
89 return;
90
91 switchOn(m_transform, [&](RefPtr<RTCRtpSFrameTransform>& sframeTransform) {
92 sframeTransform->initializeBackendForSender(*m_backend);
93 }, [&](RefPtr<RTCRtpScriptTransform>& scriptTransform) {
94 scriptTransform->initializeBackendForSender(*m_backend);
95 if (previousTransform)
96 previousTransform->backendTransferedToNewTransform();
97 });
98}
99
100void RTCRtpTransform::backendTransferedToNewTransform()
101{
102 switchOn(m_transform, [&](RefPtr<RTCRtpSFrameTransform>&) {
103 }, [&](RefPtr<RTCRtpScriptTransform>& scriptTransform) {
104 scriptTransform->backendTransferedToNewTransform();
105 });
106}
107
108void RTCRtpTransform::clearBackend()
109{
110 if (!m_backend)
111 return;
112
113 switchOn(m_transform, [&](RefPtr<RTCRtpSFrameTransform>& sframeTransform) {
114 sframeTransform->willClearBackend(*m_backend);
115 }, [&](RefPtr<RTCRtpScriptTransform>& scriptTransform) {
116 scriptTransform->willClearBackend(*m_backend);
117 });
118
119 m_backend = nullptr;
120}
121
122void RTCRtpTransform::detachFromReceiver(RTCRtpReceiver&)
123{
124 clearBackend();
125}
126
127void RTCRtpTransform::detachFromSender(RTCRtpSender&)
128{
129 clearBackend();
130}
131
132bool operator==(const RTCRtpTransform& a, const RTCRtpTransform& b)
133{
134 return switchOn(a.m_transform, [&](const RefPtr<RTCRtpSFrameTransform>& sframeTransformA) {
135 return switchOn(b.m_transform, [&](const RefPtr<RTCRtpSFrameTransform>& sframeTransformB) {
136 return sframeTransformA.get() == sframeTransformB.get();
137 }, [&](const RefPtr<RTCRtpScriptTransform>&) {
138 return false;
139 });
140 }, [&](const RefPtr<RTCRtpScriptTransform>& scriptTransformA) {
141 return switchOn(b.m_transform, [&](const RefPtr<RTCRtpSFrameTransform>&) {
142 return false;
143 }, [&](const RefPtr<RTCRtpScriptTransform>& scriptTransformB) {
144 return scriptTransformA.get() == scriptTransformB.get();
145 });
146 });
147}
