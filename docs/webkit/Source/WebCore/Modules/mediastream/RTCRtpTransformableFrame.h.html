25#pragma once
26
27#if ENABLE(WEB_RTC)
28
29#include <wtf/RefCounted.h>
30#include <wtf/Vector.h>
31
32namespace WebCore {
33
34struct RTCEncodedAudioFrameMetadata {
35 uint32_t synchronizationSource;
36 Vector<uint32_t> contributingSources;
37};
38
39struct RTCEncodedVideoFrameMetadata {
40 std::optional<int64_t> frameId;
41 Vector<int64_t> dependencies;
42 uint16_t width;
43 uint16_t height;
44 std::optional<int32_t> spatialIndex;
45 std::optional<int32_t> temporalIndex;
46 uint32_t synchronizationSource;
47};
48
49class RTCRtpTransformableFrame : public RefCounted<RTCRtpTransformableFrame> {
50public:
51 virtual ~RTCRtpTransformableFrame() = default;
52
53 struct Data {
54 const uint8_t* data { nullptr };
55 size_t size { 0 };
56 };
57 virtual Data data() const = 0;
58 virtual void setData(Data) = 0;
59
60 virtual uint64_t timestamp() const = 0;
61 virtual RTCEncodedAudioFrameMetadata audioMetadata() const = 0;
62 virtual RTCEncodedVideoFrameMetadata videoMetadata() const = 0;
63
64 virtual bool isKeyFrame() const = 0;
65};
66
67} // namespace WebCore
68
69#endif // ENABLE(WEB_RTC)
