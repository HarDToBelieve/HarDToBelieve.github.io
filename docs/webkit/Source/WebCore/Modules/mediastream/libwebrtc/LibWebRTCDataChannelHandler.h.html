47class Document;
48class RTCDataChannelEvent;
49class RTCDataChannelHandlerClient;
50struct RTCDataChannelInit;
51class ScriptExecutionContext;
52
53class LibWebRTCDataChannelHandler final : public RTCDataChannelHandler, private webrtc::DataChannelObserver {
54 WTF_MAKE_FAST_ALLOCATED;
55public:
56 explicit LibWebRTCDataChannelHandler(rtc::scoped_refptr<webrtc::DataChannelInterface>&&);
57 ~LibWebRTCDataChannelHandler();
58
59 static webrtc::DataChannelInit fromRTCDataChannelInit(const RTCDataChannelInit&);
60 static Ref<RTCDataChannelEvent> channelEvent(Document&, rtc::scoped_refptr<webrtc::DataChannelInterface>&&);
61
62private:
63 // RTCDataChannelHandler API
64 void setClient(RTCDataChannelHandlerClient&, ScriptExecutionContextIdentifier) final;
65 bool sendStringData(const CString&) final;
66 bool sendRawData(const uint8_t*, size_t) final;
67 void close() final;
68
69 // webrtc::DataChannelObserver API
70 void OnStateChange();
71 void OnMessage(const webrtc::DataBuffer&);
72 void OnBufferedAmountChange(uint64_t);
73
74 void checkState();
75
76 using Message = Variant<RTCDataChannelState, String, Ref<SharedBuffer>>;
77 using PendingMessages = Vector<Message>;
78 void storeMessage(PendingMessages&, const webrtc::DataBuffer&);
79 void processMessage(const webrtc::DataBuffer&);
80 void processStoredMessage(Message&);
81
82 void postTask(Function<void()>&&);
83
84 rtc::scoped_refptr<webrtc::DataChannelInterface> m_channel;
85 Lock m_clientLock;
86 RTCDataChannelHandlerClient* m_client WTF_GUARDED_BY_LOCK(m_clientLock) { nullptr };
87 ScriptExecutionContextIdentifier m_contextIdentifier;
88 PendingMessages m_bufferedMessages;
89};
