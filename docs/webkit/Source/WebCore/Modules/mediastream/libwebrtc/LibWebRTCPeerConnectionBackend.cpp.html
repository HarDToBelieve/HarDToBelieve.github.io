53static std::unique_ptr<PeerConnectionBackend> createLibWebRTCPeerConnectionBackend(RTCPeerConnection& peerConnection)
54{
55 if (!LibWebRTCProvider::webRTCAvailable()) {
56 RELEASE_LOG_ERROR(WebRTC, "LibWebRTC is not available to create a backend");
57 return nullptr;
58 }
59
60 auto* page = downcast<Document>(*peerConnection.scriptExecutionContext()).page();
61 if (!page)
62 return nullptr;
63
64 page->libWebRTCProvider().setEnableWebRTCEncryption(page->settings().webRTCEncryptionEnabled());
65
66 return makeUnique<LibWebRTCPeerConnectionBackend>(peerConnection, page->libWebRTCProvider());
67}

112static inline webrtc::PeerConnectionInterface::BundlePolicy bundlePolicyfromConfiguration(const MediaEndpointConfiguration& configuration)
113{
114 switch (configuration.bundlePolicy) {
115 case RTCBundlePolicy::MaxCompat:
116 return webrtc::PeerConnectionInterface::kBundlePolicyMaxCompat;
117 case RTCBundlePolicy::MaxBundle:
118 return webrtc::PeerConnectionInterface::kBundlePolicyMaxBundle;
119 case RTCBundlePolicy::Balanced:
120 return webrtc::PeerConnectionInterface::kBundlePolicyBalanced;
121 }
122
123 ASSERT_NOT_REACHED();
124 return webrtc::PeerConnectionInterface::kBundlePolicyMaxCompat;
125}

140static inline webrtc::PeerConnectionInterface::IceTransportsType iceTransportPolicyfromConfiguration(const MediaEndpointConfiguration& configuration)
141{
142 switch (configuration.iceTransportPolicy) {
143 case RTCIceTransportPolicy::Relay:
144 return webrtc::PeerConnectionInterface::kRelay;
145 case RTCIceTransportPolicy::All:
146 return webrtc::PeerConnectionInterface::kAll;
147 }
148
149 ASSERT_NOT_REACHED();
150 return webrtc::PeerConnectionInterface::kNone;
151}

153static webrtc::PeerConnectionInterface::RTCConfiguration configurationFromMediaEndpointConfiguration(MediaEndpointConfiguration&& configuration)
154{
155 webrtc::PeerConnectionInterface::RTCConfiguration rtcConfiguration;
156
157 rtcConfiguration.type = iceTransportPolicyfromConfiguration(configuration);
158 rtcConfiguration.bundle_policy = bundlePolicyfromConfiguration(configuration);
159 rtcConfiguration.rtcp_mux_policy = rtcpMuxPolicyfromConfiguration(configuration);
160
161 for (auto& server : configuration.iceServers) {
162 webrtc::PeerConnectionInterface::IceServer iceServer;
163 iceServer.username = server.username.utf8().data();
164 iceServer.password = server.credential.utf8().data();
165 for (auto& url : server.urls)
166 iceServer.urls.push_back({ url.string().utf8().data() });
167 rtcConfiguration.servers.push_back(WTFMove(iceServer));
168 }
169
170 rtcConfiguration.set_cpu_adaptation(false);
171 // FIXME: Activate ice candidate pool size once it no longer bothers test bots.
172 // rtcConfiguration.ice_candidate_pool_size = configuration.iceCandidatePoolSize;
173
174 for (auto& pem : configuration.certificates) {
175 rtcConfiguration.certificates.push_back(rtc::RTCCertificate::FromPEM(rtc::RTCCertificatePEM {
176 pem.privateKey.utf8().data(), pem.certificate.utf8().data()
177 }));
178 }
179
180 return rtcConfiguration;
181}

230void LibWebRTCPeerConnectionBackend::doSetLocalDescription(const RTCSessionDescription* description)
231{
232 m_endpoint->doSetLocalDescription(description);
233 if (!m_isLocalDescriptionSet) {
234 if (m_isRemoteDescriptionSet) {
235 for (auto& candidate : m_pendingCandidates)
236 m_endpoint->addIceCandidate(*candidate);
237 m_pendingCandidates.clear();
238 }
239 m_isLocalDescriptionSet = true;
240 }
241}
242
243void LibWebRTCPeerConnectionBackend::doSetRemoteDescription(const RTCSessionDescription& description)
244{
245 m_endpoint->doSetRemoteDescription(description);
246 if (!m_isRemoteDescriptionSet) {
247 if (m_isLocalDescriptionSet) {
248 for (auto& candidate : m_pendingCandidates)
249 m_endpoint->addIceCandidate(*candidate);
250 }
251 m_isRemoteDescriptionSet = true;
252 }
253}

280void LibWebRTCPeerConnectionBackend::doAddIceCandidate(RTCIceCandidate& candidate)
281{
282 webrtc::SdpParseError error;
283 int sdpMLineIndex = candidate.sdpMLineIndex() ? candidate.sdpMLineIndex().value() : 0;
284 std::unique_ptr<webrtc::IceCandidateInterface> rtcCandidate(webrtc::CreateIceCandidate(candidate.sdpMid().utf8().data(), sdpMLineIndex, candidate.candidate().utf8().data(), &error));
285
286 if (!rtcCandidate) {
287 addIceCandidateFailed(Exception { OperationError, String::fromUTF8(error.description.data(), error.description.length()) });
288 return;
289 }
290
291 // libwebrtc does not like that ice candidates are set before the description.
292 if (!m_isLocalDescriptionSet || !m_isRemoteDescriptionSet)
293 m_pendingCandidates.append(WTFMove(rtcCandidate));
294 else if (!m_endpoint->addIceCandidate(*rtcCandidate.get())) {
295 ASSERT_NOT_REACHED();
296 addIceCandidateFailed(Exception { OperationError, "Failed to apply the received candidate"_s });
297 return;
298 }
299 addIceCandidateSucceeded();
300}

302Ref<RTCRtpReceiver> LibWebRTCPeerConnectionBackend::createReceiver(std::unique_ptr<LibWebRTCRtpReceiverBackend>&& backend)
303{
304 auto& document = downcast<Document>(*m_peerConnection.scriptExecutionContext());
305
306 auto source = backend->createSource();
307 // Remote source is initially muted and will be unmuted when receiving the first packet.
308 source->setMuted(true);
309 auto trackID = source->persistentID();
310 auto remoteTrackPrivate = MediaStreamTrackPrivate::create(document.logger(), WTFMove(source), WTFMove(trackID));
311 auto remoteTrack = MediaStreamTrack::create(document, WTFMove(remoteTrackPrivate));
312
313 return RTCRtpReceiver::create(*this, WTFMove(remoteTrack), WTFMove(backend));
314}

360static inline RefPtr<RTCRtpSender> findExistingSender(const Vector<RefPtr<RTCRtpTransceiver>>& transceivers, LibWebRTCRtpSenderBackend& senderBackend)
361{
362 ASSERT(senderBackend.rtcSender());
363 for (auto& transceiver : transceivers) {
364 auto& sender = transceiver->sender();
365 if (!sender.isStopped() && senderBackend.rtcSender() == backendFromRTPSender(sender).rtcSender())
366 return makeRef(sender);
367 }
368 return nullptr;
369}

371ExceptionOr<Ref<RTCRtpSender>> LibWebRTCPeerConnectionBackend::addTrack(MediaStreamTrack& track, Vector<String>&& mediaStreamIds)
372{
373 auto senderBackend = makeUnique<LibWebRTCRtpSenderBackend>(*this, nullptr);
374 if (!m_endpoint->addTrack(*senderBackend, track, mediaStreamIds))
375 return Exception { TypeError, "Unable to add track"_s };
376
377 if (auto sender = findExistingSender(m_peerConnection.currentTransceivers(), *senderBackend)) {
378 backendFromRTPSender(*sender).takeSource(*senderBackend);
379 sender->setTrack(makeRef(track));
380 sender->setMediaStreamIds(mediaStreamIds);
381 return sender.releaseNonNull();
382 }
383
384 auto transceiverBackend = m_endpoint->transceiverBackendFromSender(*senderBackend);
385
386 auto sender = RTCRtpSender::create(m_peerConnection, makeRef(track), WTFMove(senderBackend));
387 sender->setMediaStreamIds(mediaStreamIds);
388 auto receiver = createReceiver(transceiverBackend->createReceiverBackend());
389 auto transceiver = RTCRtpTransceiver::create(sender.copyRef(), WTFMove(receiver), WTFMove(transceiverBackend));
390 m_peerConnection.addInternalTransceiver(WTFMove(transceiver));
391 return sender;
392}

394template<typename T>
395ExceptionOr<Ref<RTCRtpTransceiver>> LibWebRTCPeerConnectionBackend::addTransceiverFromTrackOrKind(T&& trackOrKind, const RTCRtpTransceiverInit& init)
396{
397 auto result = m_endpoint->addTransceiver(trackOrKind, init);
398 if (result.hasException())
399 return result.releaseException();
400
401 auto backends = result.releaseReturnValue();
402 auto sender = RTCRtpSender::create(m_peerConnection, WTFMove(trackOrKind), WTFMove(backends.senderBackend));
403 auto receiver = createReceiver(WTFMove(backends.receiverBackend));
404 auto transceiver = RTCRtpTransceiver::create(WTFMove(sender), WTFMove(receiver), WTFMove(backends.transceiverBackend));
405 m_peerConnection.addInternalTransceiver(transceiver.copyRef());
406 return transceiver;
407}

438RTCRtpTransceiver& LibWebRTCPeerConnectionBackend::newRemoteTransceiver(std::unique_ptr<LibWebRTCRtpTransceiverBackend>&& transceiverBackend, RealtimeMediaSource::Type type)
439{
440 auto sender = RTCRtpSender::create(m_peerConnection, type == RealtimeMediaSource::Type::Audio ? "audio"_s : "video"_s, transceiverBackend->createSenderBackend(*this, nullptr));
441 auto receiver = createReceiver(transceiverBackend->createReceiverBackend());
442 auto transceiver = RTCRtpTransceiver::create(WTFMove(sender), WTFMove(receiver), WTFMove(transceiverBackend));
443 m_peerConnection.addInternalTransceiver(transceiver.copyRef());
444 return transceiver.get();
445}

457void LibWebRTCPeerConnectionBackend::applyRotationForOutgoingVideoSources()
458{

460 if (!transceiver->sender().isStopped()) {
461 if (auto* videoSource = backendFromRTPSender(transceiver->sender()).videoSource())
462 videoSource->applyRotation();
