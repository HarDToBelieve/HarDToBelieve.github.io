58static inline void fillRTCRtpContributingSource(RTCRtpContributingSource& source, const webrtc::RtpSource& rtcSource)
59{
60 source.timestamp = rtcSource.timestamp_ms();
61 source.rtpTimestamp = rtcSource.rtp_timestamp();
62 source.source = rtcSource.source_id();
63 if (rtcSource.audio_level())
64 source.audioLevel = (*rtcSource.audio_level() == 127) ? 0 : pow(10, -*rtcSource.audio_level() / 20);
65}
66
67static inline RTCRtpContributingSource toRTCRtpContributingSource(const webrtc::RtpSource& rtcSource)
68{
69 RTCRtpContributingSource source;
70 fillRTCRtpContributingSource(source, rtcSource);
71 return source;
72}
73
74static inline RTCRtpSynchronizationSource toRTCRtpSynchronizationSource(const webrtc::RtpSource& rtcSource)
75{
76 RTCRtpSynchronizationSource source;
77 fillRTCRtpContributingSource(source, rtcSource);
78 return source;
79}

81Vector<RTCRtpContributingSource> LibWebRTCRtpReceiverBackend::getContributingSources() const
82{
83 Vector<RTCRtpContributingSource> sources;
84 for (auto& rtcSource : m_rtcReceiver->GetSources()) {
85 if (rtcSource.source_type() == webrtc::RtpSourceType::CSRC)
86 sources.append(toRTCRtpContributingSource(rtcSource));
87 }
88 return sources;
89}
90
91Vector<RTCRtpSynchronizationSource> LibWebRTCRtpReceiverBackend::getSynchronizationSources() const
92{
93 Vector<RTCRtpSynchronizationSource> sources;
94 for (auto& rtcSource : m_rtcReceiver->GetSources()) {
95 if (rtcSource.source_type() == webrtc::RtpSourceType::SSRC)
96 sources.append(toRTCRtpSynchronizationSource(rtcSource));
97 }
98 return sources;
99}

101Ref<RealtimeMediaSource> LibWebRTCRtpReceiverBackend::createSource()
102{
103 auto rtcTrack = m_rtcReceiver->track();
104 switch (m_rtcReceiver->media_type()) {
105 case cricket::MEDIA_TYPE_DATA:
106 case cricket::MEDIA_TYPE_UNSUPPORTED:
107 break;
108 case cricket::MEDIA_TYPE_AUDIO: {
109 rtc::scoped_refptr<webrtc::AudioTrackInterface> audioTrack = static_cast<webrtc::AudioTrackInterface*>(rtcTrack.get());
110 return RealtimeIncomingAudioSource::create(WTFMove(audioTrack), fromStdString(rtcTrack->id()));
111 }
112 case cricket::MEDIA_TYPE_VIDEO: {
113 rtc::scoped_refptr<webrtc::VideoTrackInterface> videoTrack = static_cast<webrtc::VideoTrackInterface*>(rtcTrack.get());
114 return RealtimeIncomingVideoSource::create(WTFMove(videoTrack), fromStdString(rtcTrack->id()));
115 }
116 }
117 RELEASE_ASSERT_NOT_REACHED();
118}

120Ref<RTCRtpTransformBackend> LibWebRTCRtpReceiverBackend::rtcRtpTransformBackend()

122 if (!m_transformBackend)
