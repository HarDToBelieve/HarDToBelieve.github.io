61static bool operator==(const LibWebRTCRtpSenderBackend::Source& a, const LibWebRTCRtpSenderBackend::Source& b)
62{
63 return switchOn(a, [&b](const Ref<RealtimeOutgoingAudioSource>& source) {
64 return WTF::holds_alternative<Ref<RealtimeOutgoingAudioSource>>(b) && source.ptr() == WTF::get<Ref<RealtimeOutgoingAudioSource>>(b).ptr();
65 }, [&b](const Ref<RealtimeOutgoingVideoSource>& source) {
66 return WTF::holds_alternative<Ref<RealtimeOutgoingVideoSource>>(b) && source.ptr() == WTF::get<Ref<RealtimeOutgoingVideoSource>>(b).ptr();
67 }, [&b](std::nullptr_t) {
68 return WTF::holds_alternative<std::nullptr_t>(b);
69 });
70}

72void LibWebRTCRtpSenderBackend::startSource()
73{
74 // We asynchronously start the sources to guarantee media goes through the transform if a transform is set when creating the track.
75 callOnMainThread([weakThis = makeWeakPtr(this), source = m_source]() mutable {
76 if (!weakThis || weakThis->m_source != source)
77 return;
78 switchOn(source, [](Ref<RealtimeOutgoingAudioSource>& source) {
79 source->start();
80 }, [](Ref<RealtimeOutgoingVideoSource>& source) {
81 source->start();
82 }, [](std::nullptr_t&) {
83 });
84 });
85}
86
87void LibWebRTCRtpSenderBackend::stopSource()
88{
89 switchOn(m_source, [](Ref<RealtimeOutgoingAudioSource>& source) {
90 source->stop();
91 }, [](Ref<RealtimeOutgoingVideoSource>& source) {
92 source->stop();
93 }, [](std::nullptr_t&) {
94 });
95 m_source = nullptr;
96}

98bool LibWebRTCRtpSenderBackend::replaceTrack(RTCRtpSender& sender, MediaStreamTrack* track)
99{
100 if (!track) {
101 stopSource();
102 return true;
103 }
104
105 if (sender.track()) {
106 switchOn(m_source, [&](Ref<RealtimeOutgoingAudioSource>& source) {
107 ASSERT(track->source().type() == RealtimeMediaSource::Type::Audio);
108 source->stop();
109 source->setSource(track->privateTrack());
110 source->start();
111 }, [&](Ref<RealtimeOutgoingVideoSource>& source) {
112 ASSERT(track->source().type() == RealtimeMediaSource::Type::Video);
113 source->stop();
114 source->setSource(track->privateTrack());
115 source->start();
116 }, [](std::nullptr_t&) {
117 ASSERT_NOT_REACHED();
118 });
119 }
120
121 m_peerConnectionBackend->setSenderSourceFromTrack(*this, *track);
122 return true;
123}

125RTCRtpSendParameters LibWebRTCRtpSenderBackend::getParameters() const
126{
127 if (!m_rtcSender)
128 return { };
129
130 m_currentParameters = m_rtcSender->GetParameters();
131 return toRTCRtpSendParameters(*m_currentParameters);
132}
133
134void LibWebRTCRtpSenderBackend::setParameters(const RTCRtpSendParameters& parameters, DOMPromiseDeferred<void>&& promise)
135{
136 if (!m_rtcSender) {
137 promise.reject(NotSupportedError);
138 return;
139 }
140
141 if (!m_currentParameters) {
142 promise.reject(Exception { InvalidStateError, "getParameters must be called before setParameters"_s });
143 return;
144 }
145
146 auto rtcParameters = WTFMove(*m_currentParameters);
147 updateRTCRtpSendParameters(parameters, rtcParameters);
148 m_currentParameters = std::nullopt;
149
150 auto error = m_rtcSender->SetParameters(rtcParameters);
151 if (!error.ok()) {
152 promise.reject(Exception { InvalidStateError, error.message() });
153 return;
154 }
155 promise.resolve();
156}

170void LibWebRTCRtpSenderBackend::setMediaStreamIds(const Vector<String>& streamIds)
171{
172 std::vector<std::string> ids;
173 for (auto& id : streamIds)
174 ids.push_back(id.utf8().data());
175 m_rtcSender->SetStreams(ids);

178RealtimeOutgoingVideoSource* LibWebRTCRtpSenderBackend::videoSource()
179{
180 return switchOn(m_source,
181 [](Ref<RealtimeOutgoingVideoSource>& source) { return source.ptr(); },
182 [](const auto&) -> RealtimeOutgoingVideoSource* { return nullptr; }
183 );
184}
185
186bool LibWebRTCRtpSenderBackend::hasSource() const
187{
188 return switchOn(m_source,
189 [](const std::nullptr_t&) { return false; },
190 [](const auto&) { return true; }
191 );
192}
193
194void LibWebRTCRtpSenderBackend::setSource(Source&& source)
195{
196 stopSource();
197 m_source = WTFMove(source);
198 startSource();
199}
200
201void LibWebRTCRtpSenderBackend::takeSource(LibWebRTCRtpSenderBackend& backend)
202{
203 ASSERT(backend.hasSource());
204 stopSource();
205 m_source = WTFMove(backend.m_source);
206 backend.m_source = nullptr;
207}
