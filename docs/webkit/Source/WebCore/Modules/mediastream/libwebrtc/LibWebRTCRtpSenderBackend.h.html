49public:
50 using Source = Variant<std::nullptr_t, Ref<RealtimeOutgoingAudioSource>, Ref<RealtimeOutgoingVideoSource>>;

52 LibWebRTCRtpSenderBackend(LibWebRTCPeerConnectionBackend&, rtc::scoped_refptr<webrtc::RtpSenderInterface>&&);

57
58 RealtimeOutgoingVideoSource* videoSource();

60 void setSource(Source&&);
61 void takeSource(LibWebRTCRtpSenderBackend&);
62
63private:
64 bool replaceTrack(RTCRtpSender&, MediaStreamTrack*) final;
65 RTCRtpSendParameters getParameters() const final;

68 Ref<RTCRtpTransformBackend> rtcRtpTransformBackend() final;
69 void setMediaStreamIds(const Vector<String>&) final;
70
71 void startSource();
72 void stopSource();
73 bool hasSource() const;
74
75 WeakPtr<LibWebRTCPeerConnectionBackend> m_peerConnectionBackend;
76 rtc::scoped_refptr<webrtc::RtpSenderInterface> m_rtcSender;
77 Source m_source;
