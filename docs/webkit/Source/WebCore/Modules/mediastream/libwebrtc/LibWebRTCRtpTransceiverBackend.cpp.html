84static inline ExceptionOr<webrtc::RtpCodecCapability> toRtpCodecCapability(const RTCRtpCodecCapability& codec)
85{
86 webrtc::RtpCodecCapability rtcCodec;
87 if (codec.mimeType.startsWith("video/"))
88 rtcCodec.kind = cricket::MEDIA_TYPE_VIDEO;
89 else if (codec.mimeType.startsWith("audio/"))
90 rtcCodec.kind = cricket::MEDIA_TYPE_AUDIO;
91 else
92 return Exception { InvalidModificationError, "RTCRtpCodecCapability bad mimeType" };
93
94 rtcCodec.name = codec.mimeType.substring(6).utf8().data();
95 rtcCodec.clock_rate = codec.clockRate;
96 if (codec.channels)
97 rtcCodec.num_channels = *codec.channels;
98
99 for (auto parameter : StringView(codec.sdpFmtpLine).split(';')) {
100 auto position = parameter.find('=');
101 if (position == notFound)
102 return Exception { InvalidModificationError, "RTCRtpCodecCapability sdpFmtLine badly formated" };
103 rtcCodec.parameters.emplace(parameter.substring(0, position).utf8().data(), parameter.substring(position + 1).utf8().data());
104 }
105
106 return rtcCodec;
107}
108
109ExceptionOr<void> LibWebRTCRtpTransceiverBackend::setCodecPreferences(const Vector<RTCRtpCodecCapability>& codecs)
110{
111 std::vector<webrtc::RtpCodecCapability> rtcCodecs;
112 for (auto& codec : codecs) {
113 auto result = toRtpCodecCapability(codec);
114 if (result.hasException())
115 return result.releaseException();
116 rtcCodecs.push_back(result.releaseReturnValue());
117 }
118 auto result = m_rtcTransceiver->SetCodecPreferences(rtcCodecs);
119 if (!result.ok())
120 return toException(result);
121 return { };
122}
