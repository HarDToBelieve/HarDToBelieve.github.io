80static inline double toWebRTCBitRatePriority(RTCPriorityType priority)
81{
82 switch (priority) {
83 case RTCPriorityType::VeryLow:
84 return 0.5;
85 case RTCPriorityType::Low:
86 return 1;
87 case RTCPriorityType::Medium:
88 return 2;
89 case RTCPriorityType::High:
90 return 4;
91 }
92
93 RELEASE_ASSERT_NOT_REACHED();
94}

96static inline RTCPriorityType fromWebRTCBitRatePriority(double priority)
97{
98 if (priority < 0.7)
99 return RTCPriorityType::VeryLow;
100 if (priority < 1.5)
101 return RTCPriorityType::Low;
102 if (priority < 2.5)
103 return RTCPriorityType::Medium;
104 return RTCPriorityType::High;
105}

107static inline RTCRtpEncodingParameters toRTCEncodingParameters(const webrtc::RtpEncodingParameters& rtcParameters)
108{
109 RTCRtpEncodingParameters parameters;
110
111 if (rtcParameters.ssrc)
112 parameters.ssrc = *rtcParameters.ssrc;
113
114 parameters.active = rtcParameters.active;
115 if (rtcParameters.max_bitrate_bps)
116 parameters.maxBitrate = *rtcParameters.max_bitrate_bps;
117 if (rtcParameters.max_framerate)
118 parameters.maxFramerate = *rtcParameters.max_framerate;
119 parameters.rid = fromStdString(rtcParameters.rid);
120 if (rtcParameters.scale_resolution_down_by)
121 parameters.scaleResolutionDownBy = *rtcParameters.scale_resolution_down_by;
122
123 parameters.priority = fromWebRTCBitRatePriority(rtcParameters.bitrate_priority);
124 parameters.networkPriority = toRTCPriorityType(rtcParameters.network_priority);
125
126 return parameters;
127}

171static inline RTCRtpCodecParameters toRTCCodecParameters(const webrtc::RtpCodecParameters& rtcParameters)
172{
173 RTCRtpCodecParameters parameters;
174
175 parameters.payloadType = rtcParameters.payload_type;
176 parameters.mimeType = fromStdString(rtcParameters.mime_type());
177 if (rtcParameters.clock_rate)
178 parameters.clockRate = *rtcParameters.clock_rate;
179 if (rtcParameters.num_channels)
180 parameters.channels = *rtcParameters.num_channels;
181
182 StringBuilder sdpFmtpLineBuilder;
183 sdpFmtpLineBuilder.append("a=fmtp:", parameters.payloadType, ' ');
184
185 bool isFirst = true;
186 for (auto& keyValue : rtcParameters.parameters) {
187 if (!isFirst)
188 sdpFmtpLineBuilder.append(';');
189 else
190 isFirst = false;
191 sdpFmtpLineBuilder.append(keyValue.first.c_str(), '=', keyValue.second.c_str());
192 }
193 parameters.sdpFmtpLine = sdpFmtpLineBuilder.toString();
194
195 return parameters;
196}

198RTCRtpParameters toRTCRtpParameters(const webrtc::RtpParameters& rtcParameters)
199{
200 RTCRtpParameters parameters;
201
202 for (auto& extension : rtcParameters.header_extensions)
203 parameters.headerExtensions.append(toRTCHeaderExtensionParameters(extension));
204 for (auto& codec : rtcParameters.codecs)
205 parameters.codecs.append(toRTCCodecParameters(codec));
206
207 parameters.rtcp.reducedSize = rtcParameters.rtcp.reduced_size;
208 if (rtcParameters.rtcp.cname.length())
209 parameters.rtcp.cname = fromStdString(rtcParameters.rtcp.cname);
210
211 return parameters;
212}

214RTCRtpSendParameters toRTCRtpSendParameters(const webrtc::RtpParameters& rtcParameters)
215{
216 RTCRtpSendParameters parameters { toRTCRtpParameters(rtcParameters) };
217
218 parameters.transactionId = fromStdString(rtcParameters.transaction_id);
219 for (auto& rtcEncoding : rtcParameters.encodings)
220 parameters.encodings.append(toRTCEncodingParameters(rtcEncoding));
221
222 if (rtcParameters.degradation_preference) {
223 switch (*rtcParameters.degradation_preference) {
224 // FIXME: Support DegradationPreference::DISABLED.
225 case webrtc::DegradationPreference::DISABLED:
226 case webrtc::DegradationPreference::MAINTAIN_FRAMERATE:
227 parameters.degradationPreference = RTCDegradationPreference::MaintainFramerate;
228 break;
229 case webrtc::DegradationPreference::MAINTAIN_RESOLUTION:
230 parameters.degradationPreference = RTCDegradationPreference::MaintainResolution;
231 break;
232 case webrtc::DegradationPreference::BALANCED:
233 parameters.degradationPreference = RTCDegradationPreference::Balanced;
234 break;
235 };
236 }
237 return parameters;
238}

240void updateRTCRtpSendParameters(const RTCRtpSendParameters& parameters, webrtc::RtpParameters& rtcParameters)
241{
242 rtcParameters.transaction_id = parameters.transactionId.utf8().data();
243
244 if (parameters.encodings.size() != rtcParameters.encodings.size()) {
245 // If encodings size is different, setting parameters will fail. Let's make it so.
246 rtcParameters.encodings.clear();
247 return;
248 }
249
250 // We copy all current encodings parameters and only update parameters that can actually be usefully updated.
251 for (size_t i = 0; i < parameters.encodings.size(); ++i) {
252 rtcParameters.encodings[i].active = parameters.encodings[i].active;
253 if (parameters.encodings[i].maxBitrate)
254 rtcParameters.encodings[i].max_bitrate_bps = parameters.encodings[i].maxBitrate;
255 if (parameters.encodings[i].maxFramerate)
256 rtcParameters.encodings[i].max_framerate = parameters.encodings[i].maxFramerate;
257 if (parameters.encodings[i].scaleResolutionDownBy)
258 rtcParameters.encodings[i].scale_resolution_down_by = parameters.encodings[i].scaleResolutionDownBy;
259 rtcParameters.encodings[i].bitrate_priority = toWebRTCBitRatePriority(parameters.encodings[i].priority);
260 if (parameters.encodings[i].networkPriority)
261 rtcParameters.encodings[i].network_priority = fromRTCPriorityType(*parameters.encodings[i].networkPriority);
262 }
263
264 rtcParameters.header_extensions.clear();
265 for (auto& extension : parameters.headerExtensions)
266 rtcParameters.header_extensions.push_back(fromRTCHeaderExtensionParameters(extension));
267 // Codecs parameters are readonly
268
269 switch (parameters.degradationPreference) {
270 case RTCDegradationPreference::MaintainFramerate:
271 rtcParameters.degradation_preference = webrtc::DegradationPreference::MAINTAIN_FRAMERATE;
272 break;
273 case RTCDegradationPreference::MaintainResolution:
274 rtcParameters.degradation_preference = webrtc::DegradationPreference::MAINTAIN_RESOLUTION;
275 break;
276 case RTCDegradationPreference::Balanced:
277 rtcParameters.degradation_preference = webrtc::DegradationPreference::BALANCED;
278 break;
279 }
280
281 if (parameters.rtcp.reducedSize)
282 rtcParameters.rtcp.reduced_size = *parameters.rtcp.reducedSize;
283 if (!parameters.rtcp.cname.isNull())
284 rtcParameters.rtcp.cname = parameters.rtcp.cname.utf8().data();
285}

320webrtc::RtpTransceiverInit fromRtpTransceiverInit(const RTCRtpTransceiverInit& init, cricket::MediaType type)
321{
322 webrtc::RtpTransceiverInit rtcInit;
323 rtcInit.direction = fromRTCRtpTransceiverDirection(init.direction);
324 for (auto& stream : init.streams)
325 rtcInit.stream_ids.push_back(stream->id().utf8().data());
326
327 if (type == cricket::MediaType::MEDIA_TYPE_AUDIO) {
328 if (!init.sendEncodings.isEmpty())
329 rtcInit.send_encodings.push_back(fromRTCEncodingParameters(init.sendEncodings[0]));
330 } else {
331 for (auto& encoding : init.sendEncodings)
332 rtcInit.send_encodings.push_back(fromRTCEncodingParameters(encoding));
333 }
334 return rtcInit;
335}

337ExceptionCode toExceptionCode(webrtc::RTCErrorType type)
338{
339 switch (type) {
340 case webrtc::RTCErrorType::INVALID_PARAMETER:
341 return InvalidAccessError;
342 case webrtc::RTCErrorType::INVALID_RANGE:
343 return RangeError;
344 case webrtc::RTCErrorType::SYNTAX_ERROR:
345 return SyntaxError;
346 case webrtc::RTCErrorType::INVALID_STATE:
347 return InvalidStateError;
348 case webrtc::RTCErrorType::INVALID_MODIFICATION:
349 return InvalidModificationError;
350 case webrtc::RTCErrorType::NETWORK_ERROR:
351 return NetworkError;
352 default:
353 return OperationError;
354 }
355}

369static inline std::optional<RTCIceProtocol> toRTCIceProtocol(const std::string& protocol)
370{
371 if (protocol == "")
372 return { };
373 if (protocol == "udp")
374 return RTCIceProtocol::Udp;
375 ASSERT(protocol == "tcp" || protocol == "ssltcp");
376 return RTCIceProtocol::Tcp;

391static inline std::optional<RTCIceCandidateType> toRTCIceCandidateType(const std::string& type)

405std::optional<RTCIceCandidate::Fields> parseIceCandidateSDP(const String& sdp)
406{
407 cricket::Candidate candidate;
408 if (!webrtc::ParseCandidate(sdp.utf8().data(), &candidate, nullptr, true))
409 return { };
410
411 RTCIceCandidate::Fields fields;
412 fields.foundation = fromStdString(candidate.foundation());
413 fields.component = toRTCIceComponent(candidate.component());
414 fields.priority = candidate.priority();
415 fields.protocol = toRTCIceProtocol(candidate.protocol());
416 if (!candidate.address().IsNil()) {
417 fields.address = fromStdString(candidate.address().HostAsURIString());
418 fields.port = candidate.address().port();
419 }
420 fields.type = toRTCIceCandidateType(candidate.type());
421 fields.tcpType = toRTCIceTcpCandidateType(candidate.tcptype());
422 if (!candidate.related_address().IsNil()) {
423 fields.relatedAddress = fromStdString(candidate.related_address().HostAsURIString());
424 fields.relatedPort = candidate.related_address().port();
425 }
426
427 fields.usernameFragment = fromStdString(candidate.username());
428 return fields;
429}
