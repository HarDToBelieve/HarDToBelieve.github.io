39Ref<MerchantValidationEvent> MerchantValidationEvent::create(const AtomString& type, const String& methodName, URL&& validationURL)
40{
41 return adoptRef(*new MerchantValidationEvent(type, methodName, WTFMove(validationURL)));
42}
43
44ExceptionOr<Ref<MerchantValidationEvent>> MerchantValidationEvent::create(Document& document, const AtomString& type, Init&& eventInit)
45{
46 URL validationURL { document.url(), eventInit.validationURL };
47 if (!validationURL.isValid())
48 return Exception { TypeError };
49
50 auto methodName = WTFMove(eventInit.methodName);
51 if (!methodName.isEmpty()) {
52 auto validatedMethodName = convertAndValidatePaymentMethodIdentifier(methodName);
53 if (!validatedMethodName)
54 return Exception { RangeError, makeString('"', methodName, "\" is an invalid payment method identifier.") };
55 }
56
57 return adoptRef(*new MerchantValidationEvent(type, WTFMove(methodName), WTFMove(validationURL), WTFMove(eventInit)));
58}

60MerchantValidationEvent::MerchantValidationEvent(const AtomString& type, const String& methodName, URL&& validationURL)
61 : Event { type, Event::CanBubble::No, Event::IsCancelable::No }
62 , m_methodName { methodName }
63 , m_validationURL { WTFMove(validationURL) }
64{
65 ASSERT(isTrusted());
66 ASSERT(m_validationURL.isValid());
67}

69MerchantValidationEvent::MerchantValidationEvent(const AtomString& type, String&& methodName, URL&& validationURL, Init&& eventInit)
70 : Event { type, WTFMove(eventInit), IsTrusted::No }
71 , m_methodName { WTFMove(methodName) }
72 , m_validationURL { WTFMove(validationURL) }
73{
74 ASSERT(!isTrusted());
75 ASSERT(m_validationURL.isValid());
76}

78EventInterface MerchantValidationEvent::eventInterface() const

82
83ExceptionOr<void> MerchantValidationEvent::complete(Ref<DOMPromise>&& merchantSessionPromise)
84{
85 if (!isTrusted())
86 return Exception { InvalidStateError };
87
88 if (m_isCompleted)
89 return Exception { InvalidStateError };
90
91 auto exception = downcast<PaymentRequest>(target())->completeMerchantValidation(*this, WTFMove(merchantSessionPromise));
92 if (exception.hasException())
93 return exception.releaseException();
94
95 m_isCompleted = true;
96 return { };
97}
