28#if ENABLE(PAYMENT_REQUEST)
29
30#include <wtf/RefCounted.h>
31#include <wtf/Vector.h>
32#include <wtf/text/WTFString.h>
33
34namespace WebCore {
35
36class PaymentAddress final : public RefCounted<PaymentAddress> {
37public:
38 template <typename... Args> static Ref<PaymentAddress> create(Args&&... args)
39 {
40 return adoptRef(*new PaymentAddress(std::forward<Args>(args)...));
41 }
42
43 const String& country() const { return m_country; }
44 const Vector<String>& addressLine() const { return m_addressLine; }
45 const String& region() const { return m_region; }
46 const String& city() const { return m_city; }
47 const String& dependentLocality() const { return m_dependentLocality; }
48 const String& postalCode() const { return m_postalCode; }
49 const String& sortingCode() const { return m_sortingCode; }
50 const String& organization() const { return m_organization; }
51 const String& recipient() const { return m_recipient; }
52 const String& phone() const { return m_phone; }
53
54private:
55 PaymentAddress(const String& country, const Vector<String>& addressLine, const String& region, const String& city, const String& dependentLocality, const String& postalCode, const String& sortingCode, const String& organization, const String& recipient, const String& phone);
56
57 String m_country;
58 Vector<String> m_addressLine;
59 String m_region;
60 String m_city;
61 String m_dependentLocality;
62 String m_postalCode;
63 String m_sortingCode;
64 String m_organization;
65 String m_recipient;
66 String m_phone;
67};
68
69} // namespace WebCore
70
71#endif // ENABLE(PAYMENT_REQUEST)
