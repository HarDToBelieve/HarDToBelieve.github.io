63// Implements the IsWellFormedCurrencyCode abstract operation from ECMA 402
64// https://tc39.github.io/ecma402/#sec-iswellformedcurrencycode
65static bool isWellFormedCurrencyCode(const String& currency)
66{
67 if (currency.length() == 3)
68 return currency.isAllSpecialCharacters<isASCIIAlpha>();
69 return false;
70}

72template<typename T>
73static ExceptionOr<String> checkAndCanonicalizeData(ScriptExecutionContext& context, T& value)
74{
75 String serializedData;
76 if (value.data) {
77 auto* globalObject = context.globalObject();
78 auto scope = DECLARE_THROW_SCOPE(globalObject->vm());
79 serializedData = JSONStringify(globalObject, value.data.get(), 0);
80 if (scope.exception())
81 return Exception { ExistingExceptionError };
82 value.data.clear();
83 }
84 return { WTFMove(serializedData) };
85}

87// Implements the "check and canonicalize amount" validity checker
88// https://www.w3.org/TR/payment-request/#dfn-check-and-canonicalize-amount
89static ExceptionOr<void> checkAndCanonicalizeAmount(PaymentCurrencyAmount& amount)
90{
91 if (!isWellFormedCurrencyCode(amount.currency))
92 return Exception { RangeError, makeString("\"", amount.currency, "\" is not a valid currency code.") };
93
94 if (!isValidDecimalMonetaryValue(amount.value))
95 return Exception { TypeError, makeString("\"", amount.value, "\" is not a valid decimal monetary value.") };
96
97 amount.currency = amount.currency.convertToASCIIUppercase();
98 return { };
99}

101enum class NegativeAmountAllowed { Yes, No };
102static ExceptionOr<void> checkAndCanonicalizePaymentItem(PaymentItem& item, NegativeAmountAllowed negativeAmountAllowed)
103{
104 auto exception = checkAndCanonicalizeAmount(item.amount);
105 if (exception.hasException())
106 return exception.releaseException();
107
108 if (negativeAmountAllowed == NegativeAmountAllowed::No && item.amount.value[0] == '-')
109 return Exception { TypeError, "Total currency values cannot be negative."_s };
110
111 return { };
112}

121// Implements "validate a standardized payment method identifier"
122// https://www.w3.org/TR/payment-method-id/#validity-0
123static bool isValidStandardizedPaymentMethodIdentifier(StringView identifier)
124{
125 enum class State {
126 Start,
127 Hyphen,
128 LowerAlpha,
129 Digit,
130 };
131
132 auto state = State::Start;
133 for (auto character : identifier.codeUnits()) {
134 switch (state) {
135 case State::Start:
136 case State::Hyphen:
137 if (isASCIILower(character)) {
138 state = State::LowerAlpha;
139 break;
140 }
141
142 return false;
143
144 case State::LowerAlpha:
145 case State::Digit:
146 if (isASCIILower(character)) {
147 state = State::LowerAlpha;
148 break;
149 }
150
151 if (isASCIIDigit(character)) {
152 state = State::Digit;
153 break;
154 }
155
156 if (character == '-') {
157 state = State::Hyphen;
158 break;
159 }
160
161 return false;
162 }
163 }
164
165 return state == State::LowerAlpha || state == State::Digit;
166}

168// Implements "validate a URL-based payment method identifier"
169// https://www.w3.org/TR/payment-method-id/#validation
170static bool isValidURLBasedPaymentMethodIdentifier(const URL& url)
171{
172 return url.protocolIs("https") && !url.hasCredentials();
173}

175// Implements "validate a payment method identifier"
176// https://www.w3.org/TR/payment-method-id/#validity
177std::optional<PaymentRequest::MethodIdentifier> convertAndValidatePaymentMethodIdentifier(const String& identifier)
178{
179 URL url { URL(), identifier };
180 if (!url.isValid()) {
181 if (isValidStandardizedPaymentMethodIdentifier(identifier))
182 return { identifier };
183 return std::nullopt;
184 }
185
186 if (isValidURLBasedPaymentMethodIdentifier(url))
187 return { WTFMove(url) };
188
189 return std::nullopt;
190}

192enum class IsUpdate {
193 No,
194 Yes,

197static ExceptionOr<std::tuple<String, Vector<String>>> checkAndCanonicalizeDetails(ScriptExecutionContext& context, PaymentDetailsBase& details, bool requestShipping, IsUpdate isUpdate)
198{
199 if (details.displayItems) {
200 for (auto& item : *details.displayItems) {
201 auto paymentItemResult = checkAndCanonicalizePaymentItem(item, NegativeAmountAllowed::Yes);
202 if (paymentItemResult.hasException())
203 return paymentItemResult.releaseException();
204 }
205 }
206
207 String selectedShippingOption;
208 if (requestShipping) {
209 if (details.shippingOptions) {
210 HashSet<String> seenShippingOptionIDs;
211 bool didLog = false;
212 for (auto& shippingOption : *details.shippingOptions) {
213 auto exception = checkAndCanonicalizeAmount(shippingOption.amount);
214 if (exception.hasException())
215 return exception.releaseException();
216
217 auto addResult = seenShippingOptionIDs.add(shippingOption.id);
218 if (!addResult.isNewEntry)
219 return Exception { TypeError, "Shipping option IDs must be unique." };
220
221 // FIXME: <rdar://problem/73464404>
222 if (!selectedShippingOption)
223 selectedShippingOption = shippingOption.id;
224 else if (!didLog && shippingOption.selected) {
225 context.addConsoleMessage(JSC::MessageSource::PaymentRequest, JSC::MessageLevel::Warning, "WebKit currently uses the first shipping option even if other shipping options are marked as selected."_s);
226 didLog = true;
227 }
228 }
229 } else if (isUpdate == IsUpdate::No)
230 details.shippingOptions = { { } };
231 } else if (isUpdate == IsUpdate::No)
232 details.shippingOptions = std::nullopt;
233
234 Vector<String> serializedModifierData;
235 if (details.modifiers) {
236 serializedModifierData.reserveInitialCapacity(details.modifiers->size());
237 for (auto& modifier : *details.modifiers) {
238 if (isUpdate == IsUpdate::Yes) {
239 auto paymentMethodIdentifier = convertAndValidatePaymentMethodIdentifier(modifier.supportedMethods);
240 if (!paymentMethodIdentifier)
241 return Exception { RangeError, makeString('"', modifier.supportedMethods, "\" is an invalid payment method identifier.") };
242 }
243
244 if (modifier.total) {
245 auto totalResult = checkAndCanonicalizeTotal(*modifier.total);
246 if (totalResult.hasException())
247 return totalResult.releaseException();
248 }
249
250 for (auto& item : modifier.additionalDisplayItems) {
251 auto paymentItemResult = checkAndCanonicalizePaymentItem(item, NegativeAmountAllowed::Yes);
252 if (paymentItemResult.hasException())
253 return paymentItemResult.releaseException();
254 }
255
256 String serializedData;
257 if (modifier.data) {
258 auto dataResult = checkAndCanonicalizeData(context, modifier);
259 if (dataResult.hasException())
260 return dataResult.releaseException();
261
262 serializedData = dataResult.releaseReturnValue();
263 }
264 serializedModifierData.uncheckedAppend(WTFMove(serializedData));
265 }
266 } else if (isUpdate == IsUpdate::No)
267 details.modifiers = { { } };
268
269 return std::make_tuple(WTFMove(selectedShippingOption), WTFMove(serializedModifierData));
270}

289// Implements the PaymentRequest Constructor
290// https://www.w3.org/TR/payment-request/#constructor
291ExceptionOr<Ref<PaymentRequest>> PaymentRequest::create(Document& document, Vector<PaymentMethodData>&& methodData, PaymentDetailsInit&& details, PaymentOptions&& options)
292{
293 auto canCreateSession = PaymentHandler::canCreateSession(document);
294 if (canCreateSession.hasException())
295 return canCreateSession.releaseException();
296
297 if (details.id.isNull())
298 details.id = createCanonicalUUIDString();
299
300 if (methodData.isEmpty())
301 return Exception { TypeError, "At least one payment method is required."_s };
302
303 Vector<Method> serializedMethodData;
304 serializedMethodData.reserveInitialCapacity(methodData.size());
305 HashSet<String> seenMethodIDs;
306 for (auto& paymentMethod : methodData) {
307 auto identifier = convertAndValidatePaymentMethodIdentifier(paymentMethod.supportedMethods);
308 if (!identifier)
309 return Exception { RangeError, makeString('"', paymentMethod.supportedMethods, "\" is an invalid payment method identifier.") };
310
311 if (!seenMethodIDs.add(stringify(*identifier)))
312 return Exception { RangeError, "Payment method IDs must be unique."_s };
313
314 String serializedData;
315 if (paymentMethod.data) {
316 auto scope = DECLARE_THROW_SCOPE(document.globalObject()->vm());
317 serializedData = JSONStringify(document.globalObject(), paymentMethod.data.get(), 0);
318 if (scope.exception())
319 return Exception { ExistingExceptionError };
320
321 auto parsedDataOrException = parse(document, serializedData);
322 if (parsedDataOrException.hasException())
323 return parsedDataOrException.releaseException();
324
325 auto exception = PaymentHandler::validateData(document, parsedDataOrException.releaseReturnValue(), *identifier);
326 if (exception.hasException())
327 return exception.releaseException();
328 }
329 serializedMethodData.uncheckedAppend({ WTFMove(*identifier), WTFMove(serializedData) });
330 }
331
332 auto totalResult = checkAndCanonicalizeTotal(details.total);
333 if (totalResult.hasException())
334 return totalResult.releaseException();
335
336 auto detailsResult = checkAndCanonicalizeDetails(document, details, options.requestShipping, IsUpdate::No);
337 if (detailsResult.hasException())
338 return detailsResult.releaseException();
339
340 auto shippingOptionAndModifierData = detailsResult.releaseReturnValue();
341 return adoptRef(*new PaymentRequest(document, WTFMove(options), WTFMove(details), WTFMove(std::get<1>(shippingOptionAndModifierData)), WTFMove(serializedMethodData), WTFMove(std::get<0>(shippingOptionAndModifierData))));
342}

366// https://www.w3.org/TR/payment-request/#show()-method
367void PaymentRequest::show(Document& document, RefPtr<DOMPromise>&& detailsPromise, ShowPromise&& promise)
368{
369 if (!document.frame()) {
370 promise.reject(Exception { AbortError });
371 return;
372 }
373
374 if (!UserGestureIndicator::processingUserGesture()) {
375 promise.reject(Exception { SecurityError, "show() must be triggered by user activation." });
376 return;
377 }
378
379 if (m_state != State::Created) {
380 promise.reject(Exception { InvalidStateError });
381 return;
382 }
383
384 if (PaymentHandler::hasActiveSession(document)) {
385 promise.reject(Exception { AbortError });
386 m_state = State::Closed;
387 return;
388 }
389
390 m_state = State::Interactive;
391 ASSERT(!m_showPromise);
392 m_showPromise = WTF::makeUnique<ShowPromise>(WTFMove(promise));
393
394 RefPtr<PaymentHandler> selectedPaymentHandler;
395 for (auto& paymentMethod : m_serializedMethodData) {
396 auto data = parse(document, paymentMethod.serializedData);
397 if (data.hasException()) {
398 settleShowPromise(data.releaseException());
399 return;
400 }
401
402 auto handler = PaymentHandler::create(document, *this, paymentMethod.identifier);
403 if (!handler)
404 continue;
405
406 auto result = handler->convertData(data.releaseReturnValue());
407 if (result.hasException()) {
408 settleShowPromise(result.releaseException());
409 return;
410 }
411
412 if (!selectedPaymentHandler)
413 selectedPaymentHandler = WTFMove(handler);
414 }
415
416 if (!selectedPaymentHandler) {
417 settleShowPromise(Exception { NotSupportedError });
418 return;
419 }
420
421 auto exception = selectedPaymentHandler->show(document);
422 if (exception.hasException()) {
423 settleShowPromise(exception.releaseException());
424 return;
425 }
426
427 ASSERT(!m_activePaymentHandler);
428 m_activePaymentHandler = PaymentHandlerWithPendingActivity { selectedPaymentHandler.releaseNonNull(), makePendingActivity(*this) };
429
430 if (!detailsPromise)
431 return;
432
433 exception = updateWith(UpdateReason::ShowDetailsResolved, detailsPromise.releaseNonNull());
434 ASSERT(!exception.hasException());
435}

438{
439 ASSERT(m_state == State::Interactive);
440 closeActivePaymentHandler();

443 m_response->abortWithException(WTFMove(exception));
444 else
445 settleShowPromise(WTFMove(exception));

500// https://www.w3.org/TR/payment-request/#canmakepayment()-method
501void PaymentRequest::canMakePayment(Document& document, CanMakePaymentPromise&& promise)
502{
503 if (m_state != State::Created) {
504 promise.reject(Exception { InvalidStateError });
505 return;
506 }
507
508 for (auto& paymentMethod : m_serializedMethodData) {
509 auto handler = PaymentHandler::create(document, *this, paymentMethod.identifier);
510 if (!handler)
511 continue;
512
513 handler->canMakePayment(document, [promise = WTFMove(promise)](bool canMakePayment) mutable {
514 promise.resolve(canMakePayment);
515 });
516 return;
517 }
518
519 promise.resolve(false);

561ExceptionOr<void> PaymentRequest::updateWith(UpdateReason reason, Ref<DOMPromise>&& promise)
562{
563 if (m_state != State::Interactive)
564 return Exception { InvalidStateError };
565
566 if (m_isUpdating)
567 return Exception { InvalidStateError };
568
569 m_isUpdating = true;
570
571 ASSERT(!m_detailsPromise);
572 m_detailsPromise = WTFMove(promise);
573 m_detailsPromise->whenSettled([this, protectedThis = makeRefPtr(this), reason]() {
574 settleDetailsPromise(reason);
575 });
576
577 return { };
578}

580ExceptionOr<void> PaymentRequest::completeMerchantValidation(Event& event, Ref<DOMPromise>&& merchantSessionPromise)
581{
582 if (m_state != State::Interactive)
583 return Exception { InvalidStateError };
584
585 event.stopPropagation();
586 event.stopImmediatePropagation();
587
588 m_merchantSessionPromise = WTFMove(merchantSessionPromise);
589 m_merchantSessionPromise->whenSettled([this, protectedThis = makeRefPtr(this)]() {
590 if (m_state != State::Interactive)
591 return;
592
593 if (m_merchantSessionPromise->status() == DOMPromise::Status::Rejected) {
594 abortWithException(Exception { AbortError });
595 return;
596 }
597
598 auto exception = activePaymentHandler()->merchantValidationCompleted(m_merchantSessionPromise->result());
599 if (exception.hasException()) {
600 abortWithException(exception.releaseException());
601 return;
602 }
603 });
604
605 return { };

618void PaymentRequest::settleDetailsPromise(UpdateReason reason)
619{
620 auto scopeExit = makeScopeExit([&] {
621 m_isUpdating = false;
622 m_isCancelPending = false;
623 m_detailsPromise = nullptr;
624 });
625
626 if (m_state != State::Interactive)
627 return;
628
629 if (m_isCancelPending || m_detailsPromise->status() == DOMPromise::Status::Rejected) {
630 abortWithException(Exception { AbortError });
631 return;
632 }
633
634 auto& context = *m_detailsPromise->scriptExecutionContext();
635 auto throwScope = DECLARE_THROW_SCOPE(context.vm());
636 auto detailsUpdate = convertDictionary<PaymentDetailsUpdate>(*context.globalObject(), m_detailsPromise->result());
637 if (throwScope.exception()) {
638 abortWithException(Exception { ExistingExceptionError });
639 return;
640 }
641
642 if (detailsUpdate.total) {
643 auto totalResult = checkAndCanonicalizeTotal(*detailsUpdate.total);
644 if (totalResult.hasException()) {
645 abortWithException(totalResult.releaseException());
646 return;
647 }
648 }
649
650 auto detailsResult = checkAndCanonicalizeDetails(context, detailsUpdate, m_options.requestShipping, IsUpdate::Yes);
651 if (detailsResult.hasException()) {
652 abortWithException(detailsResult.releaseException());
653 return;
654 }
655
656 auto shippingOptionAndModifierData = detailsResult.releaseReturnValue();
657
658 if (detailsUpdate.total)
659 m_details.total = WTFMove(*detailsUpdate.total);
660 if (detailsUpdate.displayItems)
661 m_details.displayItems = WTFMove(*detailsUpdate.displayItems);
662 if (detailsUpdate.shippingOptions && m_options.requestShipping) {
663 m_details.shippingOptions = WTFMove(detailsUpdate.shippingOptions);
664 m_shippingOption = WTFMove(std::get<0>(shippingOptionAndModifierData));
665 }
666 if (detailsUpdate.modifiers) {
667 m_details.modifiers = WTFMove(*detailsUpdate.modifiers);
668 m_serializedModifierData = WTFMove(std::get<1>(shippingOptionAndModifierData));
669 }
670
671 auto result = activePaymentHandler()->detailsUpdated(reason, WTFMove(detailsUpdate.error), WTFMove(detailsUpdate.shippingAddressErrors), WTFMove(detailsUpdate.payerErrors), detailsUpdate.paymentMethodErrors.get());
672 if (result.hasException()) {
673 abortWithException(result.releaseException());
674 return;
675 }
676}

678void PaymentRequest::whenDetailsSettled(std::function<void()>&& callback)
679{
680 auto whenSettledFunction = [this, callback = WTFMove(callback)] {
681 ASSERT(m_state == State::Interactive);
682 ASSERT(!m_isUpdating);
683 ASSERT(!m_isCancelPending);
684 ASSERT_UNUSED(this, this);
685 callback();
686 };
687
688 if (!m_detailsPromise) {
689 whenSettledFunction();
690 return;
691 }
692
693 m_detailsPromise->whenSettled([this, protectedThis = makeRefPtr(this), whenSettledFunction = WTFMove(whenSettledFunction)] {
694 if (m_state == State::Interactive)
695 whenSettledFunction();
696 });
697}

699void PaymentRequest::accept(const String& methodName, PaymentResponse::DetailsFunction&& detailsFunction, Ref<PaymentAddress>&& shippingAddress, const String& payerName, const String& payerEmail, const String& payerPhone)
700{
701 ASSERT(!m_isUpdating);
702 ASSERT(m_state == State::Interactive);
703
704 bool isRetry = m_response;
705 if (!isRetry) {
706 m_response = PaymentResponse::create(scriptExecutionContext(), *this);
707 m_response->setRequestId(m_details.id);
708 }
709
710 m_response->setMethodName(methodName);
711 m_response->setDetailsFunction(WTFMove(detailsFunction));
712 m_response->setShippingAddress(m_options.requestShipping ? shippingAddress.ptr() : nullptr);
713 m_response->setShippingOption(m_options.requestShipping ? m_shippingOption : String { });
714 m_response->setPayerName(m_options.requestPayerName ? payerName : String { });
715 m_response->setPayerEmail(m_options.requestPayerEmail ? payerEmail : String { });
716 m_response->setPayerPhone(m_options.requestPayerPhone ? payerPhone : String { });
717
718 if (!isRetry)
719 settleShowPromise(*m_response);
720 else {
721 ASSERT(m_response->hasRetryPromise());
722 m_response->settleRetryPromise();
723 }
724
725 m_state = State::Closed;
726}

728ExceptionOr<void> PaymentRequest::complete(std::optional<PaymentComplete>&& result)
729{
730 ASSERT(m_state == State::Closed);
731 if (!m_activePaymentHandler)
732 return Exception { AbortError };
733
734 activePaymentHandler()->complete(WTFMove(result));
735 m_activePaymentHandler = std::nullopt;
736 return { };
737}

739ExceptionOr<void> PaymentRequest::retry(PaymentValidationErrors&& errors)
740{
741 ASSERT(m_state == State::Closed);
742 if (!m_activePaymentHandler)
743 return Exception { AbortError };
744
745 m_state = State::Interactive;
746 return activePaymentHandler()->retry(WTFMove(errors));
747}
