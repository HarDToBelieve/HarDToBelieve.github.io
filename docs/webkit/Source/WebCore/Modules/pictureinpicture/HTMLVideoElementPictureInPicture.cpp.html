65HTMLVideoElementPictureInPicture* HTMLVideoElementPictureInPicture::from(HTMLVideoElement& videoElement)
66{
67 HTMLVideoElementPictureInPicture* supplement = static_cast<HTMLVideoElementPictureInPicture*>(Supplement<HTMLVideoElement>::from(&videoElement, supplementName()));
68 if (!supplement) {
69 auto newSupplement = makeUnique<HTMLVideoElementPictureInPicture>(videoElement);
70 supplement = newSupplement.get();
71 provideTo(&videoElement, supplementName(), WTFMove(newSupplement));
72 }
73 return supplement;
74}

81void HTMLVideoElementPictureInPicture::requestPictureInPicture(HTMLVideoElement& videoElement, Ref<DeferredPromise>&& promise)
82{
83 if (!supportsPictureInPicture()) {
84 promise->reject(NotSupportedError, "The Picture-in-Picture mode is not supported.");
85 return;
86 }
87
88 if (videoElement.readyState() == HTMLMediaElementEnums::HAVE_NOTHING) {
89 promise->reject(InvalidStateError, "The video element is not ready to enter the Picture-in-Picture mode.");
90 return;
91 }
92
93 if (!videoElement.videoTracks() || !videoElement.videoTracks()->length()) {
94 promise->reject(InvalidStateError, "The video element does not have a video track or it has not detected a video track yet.");
95 return;
96 }
97
98 bool userActivationRequired = !videoElement.document().pictureInPictureElement();
99 if (userActivationRequired && !UserGestureIndicator::processingUserGesture()) {
100 promise->reject(NotAllowedError, "The request is not triggered by a user activation.");
101 return;
102 }
103
104 auto videoElementPictureInPicture = HTMLVideoElementPictureInPicture::from(videoElement);
105 if (videoElement.document().pictureInPictureElement() == &videoElement) {
106 promise->resolve<IDLInterface<PictureInPictureWindow>>(*(videoElementPictureInPicture->m_pictureInPictureWindow));
107 return;
108 }
109
110 if (videoElementPictureInPicture->m_enterPictureInPicturePromise || videoElementPictureInPicture->m_exitPictureInPicturePromise) {
111 promise->reject(NotAllowedError, "The video element is processing a Picture-in-Picture request.");
112 return;
113 }
114
115 if (videoElement.webkitSupportsPresentationMode(HTMLVideoElement::VideoPresentationMode::PictureInPicture)) {
116 videoElementPictureInPicture->m_enterPictureInPicturePromise = WTFMove(promise);
117 videoElement.webkitSetPresentationMode(HTMLVideoElement::VideoPresentationMode::PictureInPicture);
118 } else
119 promise->reject(NotSupportedError, "The video element does not support the Picture-in-Picture mode.");
120}

142void HTMLVideoElementPictureInPicture::exitPictureInPicture(Ref<DeferredPromise>&& promise)
143{
144 INFO_LOG(LOGIDENTIFIER);
145 if (m_enterPictureInPicturePromise || m_exitPictureInPicturePromise) {
146 promise->reject(NotAllowedError);
147 return;
148 }
149
150 m_exitPictureInPicturePromise = WTFMove(promise);
151 m_videoElement.webkitSetPresentationMode(HTMLVideoElement::VideoPresentationMode::Inline);
152}

154void HTMLVideoElementPictureInPicture::didEnterPictureInPicture(const IntSize& windowSize)
155{
156 INFO_LOG(LOGIDENTIFIER);
157 m_videoElement.invalidateStyle();
158 m_videoElement.document().setPictureInPictureElement(&m_videoElement);
159 m_pictureInPictureWindow->setSize(windowSize);
160
161 EnterPictureInPictureEvent::Init initializer;
162 initializer.bubbles = true;
163 initializer.pictureInPictureWindow = m_pictureInPictureWindow;
164 m_videoElement.scheduleEvent(EnterPictureInPictureEvent::create(eventNames().enterpictureinpictureEvent, WTFMove(initializer)));
165
166 if (m_enterPictureInPicturePromise) {
167 m_enterPictureInPicturePromise->resolve<IDLInterface<PictureInPictureWindow>>(*m_pictureInPictureWindow);
168 m_enterPictureInPicturePromise = nullptr;
169 }
170}
171
172void HTMLVideoElementPictureInPicture::didExitPictureInPicture()
173{
174 INFO_LOG(LOGIDENTIFIER);
175 m_videoElement.invalidateStyle();
176 m_pictureInPictureWindow->close();
177 m_videoElement.document().setPictureInPictureElement(nullptr);
178 m_videoElement.scheduleEvent(Event::create(eventNames().leavepictureinpictureEvent, Event::CanBubble::Yes, Event::IsCancelable::No));
179
180 if (m_exitPictureInPicturePromise) {
181 m_exitPictureInPicturePromise->resolve();
182 m_exitPictureInPicturePromise = nullptr;
183 }
184}

186void HTMLVideoElementPictureInPicture::pictureInPictureWindowResized(const IntSize& windowSize)
187{
188 if (m_pictureInPictureWindow->width() == windowSize.width() && m_pictureInPictureWindow->height() == windowSize.height())
189 return;
190
191 m_pictureInPictureWindow->setSize(windowSize);
192 auto resizeEvent = Event::create(eventNames().resizeEvent, Event::CanBubble::Yes, Event::IsCancelable::No);
193 resizeEvent->setTarget(m_pictureInPictureWindow);
194 m_videoElement.scheduleEvent(WTFMove(resizeEvent));
195}
