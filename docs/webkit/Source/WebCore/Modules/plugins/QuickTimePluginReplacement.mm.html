135bool QuickTimePluginReplacement::ensureReplacementScriptInjected()
136{
137 if (!m_parentElement->document().frame())
138 return false;
139
140 DOMWrapperWorld& world = isolatedWorld();
141 ScriptController& scriptController = m_parentElement->document().frame()->script();
142 JSDOMGlobalObject* globalObject = JSC::jsCast<JSDOMGlobalObject*>(scriptController.globalObject(world));
143 JSC::VM& vm = globalObject->vm();
144 JSC::JSLockHolder lock(vm);
145 auto scope = DECLARE_CATCH_SCOPE(vm);
146 JSC::JSGlobalObject* lexicalGlobalObject = globalObject;
147
148 JSC::JSValue replacementFunction = globalObject->get(lexicalGlobalObject, JSC::Identifier::fromString(vm, "createPluginReplacement"));
149 if (replacementFunction.isCallable(vm))
150 return true;
151
152 scriptController.evaluateInWorldIgnoringException(ScriptSourceCode(quickTimePluginReplacementScript()), world);
153 if (UNLIKELY(scope.exception())) {
154 LOG(Plugins, "%p - Exception when evaluating QuickTime plugin replacement script", this);
155 scope.clearException();
156 return false;
157 }
158
159 return true;
160}

162auto QuickTimePluginReplacement::installReplacement(ShadowRoot& root) -> InstallResult
163{
164 if (!ensureReplacementScriptInjected())
165 return { false };
166
167 if (!m_parentElement->document().frame())
168 return { false };
169
170 DOMWrapperWorld& world = isolatedWorld();
171 ScriptController& scriptController = m_parentElement->document().frame()->script();
172 JSDOMGlobalObject* globalObject = JSC::jsCast<JSDOMGlobalObject*>(scriptController.globalObject(world));
173 JSC::VM& vm = globalObject->vm();
174 JSC::JSLockHolder lock(vm);
175 auto scope = DECLARE_CATCH_SCOPE(vm);
176 JSC::JSGlobalObject* lexicalGlobalObject = globalObject;
177
178 auto clearExceptionAndReturnFalse = [&] () -> InstallResult {
179 scope.clearException();
180 return { false };
181 };
182
183 // Lookup the "createPluginReplacement" function.
184 JSC::JSValue replacementFunction = globalObject->get(lexicalGlobalObject, JSC::Identifier::fromString(vm, "createPluginReplacement"));
185 if (replacementFunction.isUndefinedOrNull())
186 return { false };
187 JSC::JSObject* replacementObject = replacementFunction.toObject(lexicalGlobalObject);
188 RETURN_IF_EXCEPTION(scope, clearExceptionAndReturnFalse());
189
190 auto callData = getCallData(vm, replacementObject);
191 if (callData.type == JSC::CallData::Type::None)
192 return { false };
193
194 JSC::MarkedArgumentBuffer argList;
195 argList.append(toJS(lexicalGlobalObject, globalObject, &root));
196 argList.append(toJS(lexicalGlobalObject, globalObject, m_parentElement.get()));
197 argList.append(toJS(lexicalGlobalObject, globalObject, this));
198 argList.append(toJS<IDLSequence<IDLNullable<IDLDOMString>>>(*lexicalGlobalObject, *globalObject, m_names));
199 argList.append(toJS<IDLSequence<IDLNullable<IDLDOMString>>>(*lexicalGlobalObject, *globalObject, m_values));
200 ASSERT(!argList.hasOverflowed());
201 JSC::JSValue replacement = call(lexicalGlobalObject, replacementObject, callData, globalObject, argList);
202 RETURN_IF_EXCEPTION(scope, clearExceptionAndReturnFalse());
203
204 // Get the <video> created to replace the plug-in.
205 JSC::JSValue value = replacement.get(lexicalGlobalObject, JSC::Identifier::fromString(vm, "video"));
206 if (!scope.exception() && !value.isUndefinedOrNull())
207 m_mediaElement = JSHTMLVideoElement::toWrapped(vm, value);
208
209 if (!m_mediaElement) {
210 LOG(Plugins, "%p - Failed to find <video> element created by QuickTime plugin replacement script.", this);
211 scope.clearException();
212 return { false };
213 }
214
215 // Get the scripting interface.
216 value = replacement.get(lexicalGlobalObject, JSC::Identifier::fromString(vm, "scriptObject"));
217 if (!value.isObject()) {
218 LOG(Plugins, "%p - Failed to find script object created by QuickTime plugin replacement.", this);
219 scope.clearException();
220 return { false };
221 }
222
223 return { true, value };
224}

243static JSValue *jsValueWithDataInContext(NSData *data, const String& mimeType, JSContext *context)
244{
245 // FIXME: Add makeCFString/makeNSString to avoid unnecessary String allocation.
246 String data64;
247 if (!mimeType.isEmpty())
248 data64 = makeString("data:", mimeType, ";base64,", base64Encoded([data bytes], [data length]));
249 else
250 data64 = makeString("data:text/plain;base64,", base64Encoded([data bytes], [data length]));
251
252 return [JSValue valueWithObject:(id)data64.createCFString().get() inContext:context];
253}

255static JSValue *jsValueWithArrayInContext(NSArray *array, JSContext *context)
256{
257 JSValueRef exception = 0;
258 JSValue *result = [JSValue valueWithNewArrayInContext:context];
259 JSObjectRef resultObject = JSValueToObject([context JSGlobalContextRef], [result JSValueRef], &exception);
260 if (exception)
261 return [JSValue valueWithUndefinedInContext:context];
262
263 NSUInteger count = [array count];
264 for (NSUInteger i = 0; i < count; ++i) {
265 JSValue *value = jsValueWithValueInContext([array objectAtIndex:i], context);
266 if (!value)
267 continue;
268
269 JSObjectSetPropertyAtIndex([context JSGlobalContextRef], resultObject, (unsigned)i, [value JSValueRef], &exception);
270 if (exception)
271 continue;
272 }
273
274 return result;
275}

277static JSValue *jsValueWithDictionaryInContext(NSDictionary *dictionary, JSContext *context)
278{
279 JSValueRef exception = 0;
280 JSValue *result = [JSValue valueWithNewObjectInContext:context];
281 JSObjectRef resultObject = JSValueToObject([context JSGlobalContextRef], [result JSValueRef], &exception);
282 if (exception)
283 return [JSValue valueWithUndefinedInContext:context];
284
285 for (id key in [dictionary keyEnumerator]) {
286 if (![key isKindOfClass:[NSString class]])
287 continue;
288
289 JSValue *value = jsValueWithValueInContext([dictionary objectForKey:key], context);
290 if (!value)
291 continue;
292
293 auto name = OpaqueJSString::tryCreate(key);
294 JSObjectSetProperty([context JSGlobalContextRef], resultObject, name.get(), [value JSValueRef], 0, &exception);
295 if (exception)
296 continue;
297 }
298
299 return result;
300}

302static JSValue *jsValueWithValueInContext(id value, JSContext *context)
303{
304 if ([value isKindOfClass:[NSString class]] || [value isKindOfClass:[NSNumber class]])
305 return [JSValue valueWithObject:value inContext:context];
306 else if ([value isKindOfClass:[NSLocale class]])
307 return [JSValue valueWithObject:[value localeIdentifier] inContext:context];
308 else if ([value isKindOfClass:[NSDictionary class]])
309 return jsValueWithDictionaryInContext(value, context);
310 else if ([value isKindOfClass:[NSArray class]])
311 return jsValueWithArrayInContext(value, context);
312 else if ([value isKindOfClass:[NSData class]])
313 return jsValueWithDataInContext(value, emptyString(), context);
314 else if ([value isKindOfClass:PAL::getAVMetadataItemClass()])
315 return jsValueWithAVMetadataItemInContext(value, context);
316
317 return nil;
318}

320static JSValue *jsValueWithAVMetadataItemInContext(AVMetadataItem *item, JSContext *context)
321{
322 NSMutableDictionary* dictionary = [NSMutableDictionary dictionaryWithDictionary:[item extraAttributes]];
323
324 if (item.keySpace)
325 [dictionary setObject:item.keySpace forKey:@"keyspace"];
326
327 if (item.key)
328 [dictionary setObject:item.key forKey:@"key"];
329
330 if (item.locale)
331 [dictionary setObject:item.locale forKey:@"locale"];
332
333 if (CMTIME_IS_VALID(item.time)) {
334 if (auto timeDictionary = adoptCF(PAL::CMTimeCopyAsDictionary(item.time, kCFAllocatorDefault)))
335 [dictionary setObject:(__bridge NSDictionary *)timeDictionary.get() forKey:@"timestamp"];
336 }
337
338 if (item.value) {
339 id value = item.value;
340 NSString *mimeType = [[item extraAttributes] objectForKey:@"MIMEtype"];
341 if ([value isKindOfClass:[NSData class]] && mimeType) {
342 auto data64 = makeString("data:", String(mimeType), ";base64,", base64Encoded([value bytes], [value length]));
343 [dictionary setObject:(__bridge NSString *)data64.createCFString().get() forKey:@"value"];
344 } else
345 [dictionary setObject:value forKey:@"value"];
346 }
347
348 return jsValueWithDictionaryInContext(dictionary, context);
349}

353JSC::JSValue JSQuickTimePluginReplacement::timedMetaData(JSC::JSGlobalObject& state) const
354{
355#if PLATFORM(IOS_FAMILY)
356 HTMLVideoElement* parent = wrapped().parentElement();
357 if (!parent || !parent->player())
358 return JSC::jsNull();
359
360 Frame* frame = parent->document().frame();
361 if (!frame)
362 return JSC::jsNull();
363
364 NSArray *metaData = parent->player()->timedMetadata();
365 if (!metaData)
366 return JSC::jsNull();
367
368 JSContext *jsContext = frame->script().javaScriptContext();
369 JSValue *metaDataValue = jsValueWithValueInContext(metaData, jsContext);
370
371 return toJS(&state, [metaDataValue JSValueRef]);
372#else
373 UNUSED_PARAM(state);
374 return JSC::jsNull();
375#endif
376}
