36namespace WebCore {
37
38class DeferredPromise;
39class HTMLMediaElement;
40class MediaPlaybackTarget;
41class RemotePlaybackAvailabilityCallback;
42
43class RemotePlayback final : public RefCounted<RemotePlayback>, public ActiveDOMObject, public EventTargetWithInlineData {
44 WTF_MAKE_ISO_ALLOCATED(RemotePlayback);
45public:
46 static Ref<RemotePlayback> create(HTMLMediaElement&);
47 ~RemotePlayback();
48
49 void watchAvailability(Ref<RemotePlaybackAvailabilityCallback>&&, Ref<DeferredPromise>&&);
50 void cancelWatchAvailability(std::optional<int32_t> id, Ref<DeferredPromise>&&);
51 void prompt(Ref<DeferredPromise>&&);
52
53 bool hasAvailabilityCallbacks() const;
54 void availabilityChanged(bool);
55 void playbackTargetPickerWasDismissed();
56 void shouldPlayToRemoteTargetChanged(bool);
57 void isPlayingToRemoteTargetChanged(bool);
58
59 enum class State {
60 Connecting,
61 Connected,
62 Disconnected,
63 };
64 State state() const { return m_state; }
65
66 void invalidate();
67
68 using RefCounted::ref;
69 using RefCounted::deref;
70
71 void* opaqueRootConcurrently() const;
72 Node* ownerNode() const;
73
74private:
75 explicit RemotePlayback(HTMLMediaElement&);
76
77 void refEventTarget() final { ref(); }
78 void derefEventTarget() final { deref(); }
79
80 void setState(State);
81 void establishConnection();
82 void disconnect();
83
84 // ActiveDOMObject.
85 const char* activeDOMObjectName() const final;
86
87 // EventTargetWithInlineData.
88 EventTargetInterface eventTargetInterface() const final { return RemotePlaybackEventTargetInterfaceType; }
89 ScriptExecutionContext* scriptExecutionContext() const final { return ActiveDOMObject::scriptExecutionContext(); }
90
91 WeakPtr<HTMLMediaElement> m_mediaElement;
92 uint32_t m_nextId { 0 };
93
94 using CallbackMap = HashMap<int32_t, Ref<RemotePlaybackAvailabilityCallback>>;
95 CallbackMap m_callbackMap;
96
97 using PromiseVector = Vector<Ref<DeferredPromise>>;
98 PromiseVector m_availabilityPromises;
99 PromiseVector m_cancelAvailabilityPromises;
100 PromiseVector m_promptPromises;
101 State m_state { State::Disconnected };
102 bool m_available { false };
103};
104
105}
