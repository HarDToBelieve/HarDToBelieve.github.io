51SpeechRecognition::SpeechRecognition(Document& document)
52 : ActiveDOMObject(document)
53{
54 if (auto* page = document.page()) {
55 m_connection = makeRefPtr(page->speechRecognitionConnection());
56 m_connection->registerClient(*this);
57 }
58}

65ExceptionOr<void> SpeechRecognition::startRecognition()
66{
67 if (m_state != State::Inactive)
68 return Exception { InvalidStateError, "Recognition is being started or already started"_s };
69
70 if (!m_connection)
71 return Exception { UnknownError, "Recognition does not have a valid connection"_s };
72
73 auto& document = downcast<Document>(*scriptExecutionContext());
74 auto* frame = document.frame();
75 if (!frame)
76 return Exception { UnknownError, "Recognition is not in a valid frame"_s };
77
78 auto optionalFrameIdentifier = document.frameID();
79 auto frameIdentifier = optionalFrameIdentifier ? *optionalFrameIdentifier : FrameIdentifier { };
80 m_connection->start(identifier(), m_lang, m_continuous, m_interimResults, m_maxAlternatives, ClientOrigin { document.topOrigin().data(), document.securityOrigin().data() }, frameIdentifier);
81 m_state = State::Starting;
82 return { };
83}

85void SpeechRecognition::stopRecognition()
86{
87 if (m_state == State::Inactive || m_state == State::Stopping || m_state == State::Aborting)
88 return;
89
90 m_connection->stop(identifier());
91 m_state = State::Stopping;
92}
93
94void SpeechRecognition::abortRecognition()
95{
96 if (m_state == State::Inactive || m_state == State::Aborting)
97 return;
98
99 m_connection->abort(identifier());
100 m_state = State::Aborting;
101}

108void SpeechRecognition::stop()
109{
110 abortRecognition();
111
112 if (!m_connection)
113 return;
114 m_connection->unregisterClient(*this);
115
116 auto& document = downcast<Document>(*scriptExecutionContext());
117 document.setActiveSpeechRecognition(nullptr);
118}

169void SpeechRecognition::didReceiveResult(Vector<SpeechRecognitionResultData>&& resultDatas)
170{
171 Vector<Ref<SpeechRecognitionResult>> allResults;
172 allResults.reserveCapacity(m_finalResults.size() + resultDatas.size());
173 allResults.appendVector(m_finalResults);
174
175 auto firstChangedIndex = allResults.size();
176 for (auto resultData : resultDatas) {
177 auto alternatives = WTF::map(resultData.alternatives, [](auto& alternativeData) {
178 return SpeechRecognitionAlternative::create(WTFMove(alternativeData.transcript), alternativeData.confidence);
179 });
180
181 auto newResult = SpeechRecognitionResult::create(WTFMove(alternatives), resultData.isFinal);
182 if (resultData.isFinal)
183 m_finalResults.append(newResult);
184
185 allResults.append(WTFMove(newResult));
186 }
187
188 auto resultList = SpeechRecognitionResultList::create(WTFMove(allResults));
189 queueTaskToDispatchEvent(*this, TaskSource::Speech, SpeechRecognitionEvent::create(eventNames().resultEvent, firstChangedIndex, WTFMove(resultList)));
190}
191
192void SpeechRecognition::didError(const SpeechRecognitionError& error)
193{
194 m_finalResults.clear();
195 m_state = State::Inactive;
196
197 queueTaskToDispatchEvent(*this, TaskSource::Speech, SpeechRecognitionErrorEvent::create(eventNames().errorEvent, error.type, error.message));
198}
199
200void SpeechRecognition::didEnd()
201{
202 m_finalResults.clear();
203 m_state = State::Inactive;
204
205 queueTaskToDispatchEvent(*this, TaskSource::Speech, Event::create(eventNames().endEvent, Event::CanBubble::No, Event::IsCancelable::No));
206}
