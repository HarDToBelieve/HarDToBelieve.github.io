46std::optional<CaptureDevice> SpeechRecognitionCaptureSource::findCaptureDevice()
47{
48 std::optional<CaptureDevice> captureDevice;
49 auto devices = RealtimeMediaSourceCenter::singleton().audioCaptureFactory().audioCaptureDeviceManager().captureDevices();
50 for (auto device : devices) {
51 if (!device.enabled())
52 continue;
53
54 if (!captureDevice)
55 captureDevice = device;
56
57 if (device.isDefault()) {
58 captureDevice = device;
59 break;
60 }
61 }
62 return captureDevice;
63}
