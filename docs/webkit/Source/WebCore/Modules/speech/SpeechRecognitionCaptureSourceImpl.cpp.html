54SpeechRecognitionCaptureSourceImpl::SpeechRecognitionCaptureSourceImpl(SpeechRecognitionConnectionClientIdentifier identifier, DataCallback&& dataCallback, StateUpdateCallback&& stateUpdateCallback, Ref<RealtimeMediaSource>&& source)
55 : m_clientIdentifier(identifier)
56 , m_dataCallback(WTFMove(dataCallback))
57 , m_stateUpdateCallback(WTFMove(stateUpdateCallback))
58 , m_source(WTFMove(source))
59{
60 m_source->addAudioSampleObserver(*this);
61 m_source->addObserver(*this);
62 m_source->start();
63
64#if !RELEASE_LOG_DISABLED
65 if (!nullLogger().get()) {
66 nullLogger() = Logger::create(this);
67 nullLogger()->setEnabled(this, false);
68 }
69
70 m_source->setLogger(*nullLogger(), nextLogIdentifier());
71#endif
72
73 auto weakThis = makeWeakPtr(this);
74}
75
76SpeechRecognitionCaptureSourceImpl::~SpeechRecognitionCaptureSourceImpl()
77{
78 m_source->removeAudioSampleObserver(*this);
79 m_source->removeObserver(*this);
80 m_source->stop();
81}

83#if PLATFORM(COCOA)
84bool SpeechRecognitionCaptureSourceImpl::updateDataSource(const CAAudioStreamDescription& audioDescription)
85{
86 if (!m_dataSourceLock.tryLock())
87 return false;
88
89 Locker locker { AdoptLock, m_dataSourceLock };
90
91 auto dataSource = AudioSampleDataSource::create(audioDescription.sampleRate() * 1, m_source.get());
92 if (dataSource->setInputFormat(audioDescription)) {
93 callOnMainThread([this, weakThis = makeWeakPtr(this)] {
94 if (weakThis)
95 m_stateUpdateCallback(SpeechRecognitionUpdate::createError(m_clientIdentifier, SpeechRecognitionError { SpeechRecognitionErrorType::AudioCapture, "Unable to set input format" }));
96 });
97 return false;
98 }
99
100 if (dataSource->setOutputFormat(audioDescription)) {
101 callOnMainThread([this, weakThis = makeWeakPtr(this)] {
102 if (weakThis)
103 m_stateUpdateCallback(SpeechRecognitionUpdate::createError(m_clientIdentifier, SpeechRecognitionError { SpeechRecognitionErrorType::AudioCapture, "Unable to set output format" }));
104 });
105 return false;
106 }
107
108 m_dataSource = WTFMove(dataSource);
109 return true;
110}

112void SpeechRecognitionCaptureSourceImpl::pullSamplesAndCallDataCallback(const MediaTime& time, const CAAudioStreamDescription& audioDescription, size_t sampleCount)
113{
114 ASSERT(isMainThread());
115
116 auto data = WebAudioBufferList { audioDescription, static_cast<uint32_t>(sampleCount) };
117 {
118 Locker locker { m_dataSourceLock };
119 m_dataSource->pullSamples(*data.list(), sampleCount, time.timeValue(), 0, AudioSampleDataSource::Copy);
120 }
121
122 m_dataCallback(time, data, audioDescription, sampleCount);
123}
124#endif

126// FIXME: It is unclear why it is safe to use m_dataSource without locking in this function.
127void SpeechRecognitionCaptureSourceImpl::audioSamplesAvailable(const MediaTime& time, const PlatformAudioData& data, const AudioStreamDescription& description, size_t sampleCount) WTF_IGNORES_THREAD_SAFETY_ANALYSIS
128{
129#if PLATFORM(COCOA)
130 DisableMallocRestrictionsForCurrentThreadScope scope;
131
132 ASSERT(description.platformDescription().type == PlatformDescription::CAAudioStreamBasicType);
133 auto audioDescription = toCAAudioStreamDescription(description);
134 if (!m_dataSource || !m_dataSource->inputDescription() || *m_dataSource->inputDescription() != description) {
135 if (!updateDataSource(audioDescription))
136 return;
137 }
138
139 m_dataSource->pushSamples(time, data, sampleCount);
140
141 callOnMainThread([weakThis = makeWeakPtr(this), time, audioDescription, sampleCount] {
142 if (weakThis)
143 weakThis->pullSamplesAndCallDataCallback(time, audioDescription, sampleCount);
144 });
145#else
146 m_dataCallback(time, data, description, sampleCount);
147#endif
148}

150void SpeechRecognitionCaptureSourceImpl::sourceStarted()
151{
152 ASSERT(isMainThread());
153 m_stateUpdateCallback(SpeechRecognitionUpdate::create(m_clientIdentifier, SpeechRecognitionUpdateType::AudioStart));
154}
155
156void SpeechRecognitionCaptureSourceImpl::sourceStopped()
157{
158 ASSERT(isMainThread());
159 ASSERT(m_source->captureDidFail());
160 m_stateUpdateCallback(SpeechRecognitionUpdate::createError(m_clientIdentifier, SpeechRecognitionError { SpeechRecognitionErrorType::AudioCapture, "Source is stopped" }));
161}
162
163void SpeechRecognitionCaptureSourceImpl::sourceMutedChanged()
164{
165 ASSERT(isMainThread());
166 m_stateUpdateCallback(SpeechRecognitionUpdate::createError(m_clientIdentifier, SpeechRecognitionError { SpeechRecognitionErrorType::AudioCapture, "Source is muted" }));
167}
168
169void SpeechRecognitionCaptureSourceImpl::mute()
170{
171 m_source->setMuted(true);
172}
