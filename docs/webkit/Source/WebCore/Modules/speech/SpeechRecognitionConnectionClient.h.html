26#pragma once
27
28#include "SpeechRecognitionConnectionClientIdentifier.h"
29#include <wtf/WeakPtr.h>
30
31namespace WebCore {
32
33struct SpeechRecognitionError;
34struct SpeechRecognitionResultData;
35
36class SpeechRecognitionConnectionClient : public CanMakeWeakPtr<SpeechRecognitionConnectionClient> {
37public:
38 SpeechRecognitionConnectionClient()
39 : m_identifier(SpeechRecognitionConnectionClientIdentifier::generate())
40 {
41 }
42
43 virtual ~SpeechRecognitionConnectionClient() = default;
44
45 virtual void didStart() = 0;
46 virtual void didStartCapturingAudio() = 0;
47 virtual void didStartCapturingSound() = 0;
48 virtual void didStartCapturingSpeech() = 0;
49 virtual void didStopCapturingSpeech() = 0;
50 virtual void didStopCapturingSound() = 0;
51 virtual void didStopCapturingAudio() = 0;
52 virtual void didFindNoMatch() = 0;
53 virtual void didReceiveResult(Vector<SpeechRecognitionResultData>&& resultDatas) = 0;
54 virtual void didError(const SpeechRecognitionError&) = 0;
55 virtual void didEnd() = 0;
56
57 SpeechRecognitionConnectionClientIdentifier identifier() const { return m_identifier; };
58
59private:
60 SpeechRecognitionConnectionClientIdentifier m_identifier;
61};
62
63} // namespace WebCore
