27#include "SpeechRecognitionResultList.h"
28
29#include "SpeechRecognitionResult.h"
30#include <wtf/IsoMallocInlines.h>
31
32namespace WebCore {
33
34WTF_MAKE_ISO_ALLOCATED_IMPL(SpeechRecognitionResultList);
35
36Ref<SpeechRecognitionResultList> SpeechRecognitionResultList::create()
37{
38 return adoptRef(*new SpeechRecognitionResultList());
39}
40
41Ref<SpeechRecognitionResultList> SpeechRecognitionResultList::create(Vector<Ref<SpeechRecognitionResult>>&& list)
42{
43 return adoptRef(*new SpeechRecognitionResultList(WTFMove(list)));
44}
45
46SpeechRecognitionResult* SpeechRecognitionResultList::item(uint64_t index) const
47{
48 return (index < m_list.size()) ? m_list[index].ptr() : nullptr;
49}
50
51void SpeechRecognitionResultList::add(SpeechRecognitionResult& result)
52{
53 m_list.append(makeRef(result));
54}
55
56SpeechRecognitionResultList::SpeechRecognitionResultList(Vector<Ref<SpeechRecognitionResult>>&& list)
57 : m_list(WTFMove(list))
58{
59}
