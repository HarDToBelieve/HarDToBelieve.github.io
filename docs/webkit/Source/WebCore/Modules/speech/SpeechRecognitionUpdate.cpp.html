31String convertEnumerationToString(SpeechRecognitionUpdateType enumerationValue)
32{
33 static const NeverDestroyed<String> values[] = {
34 MAKE_STATIC_STRING_IMPL("UpdateTypeStart"),
35 MAKE_STATIC_STRING_IMPL("UpdateTypeAudioStart"),
36 MAKE_STATIC_STRING_IMPL("UpdateTypeSoundStart"),
37 MAKE_STATIC_STRING_IMPL("UpdateTypeSpeechStart"),
38 MAKE_STATIC_STRING_IMPL("UpdateTypeSpeechEnd"),
39 MAKE_STATIC_STRING_IMPL("UpdateTypeSoundEnd"),
40 MAKE_STATIC_STRING_IMPL("UpdateTypeAudioEnd"),
41 MAKE_STATIC_STRING_IMPL("UpdateTypeResult"),
42 MAKE_STATIC_STRING_IMPL("UpdateTypeNoMatch"),
43 MAKE_STATIC_STRING_IMPL("UpdateTypeError"),
44 MAKE_STATIC_STRING_IMPL("UpdateTypeEnd"),
45 };
46 static_assert(static_cast<size_t>(SpeechRecognitionUpdateType::Start) == 0, "SpeechRecognitionUpdateType::Start is not 1 as expected");
47 static_assert(static_cast<size_t>(SpeechRecognitionUpdateType::AudioStart) == 1, "SpeechRecognitionUpdateType::AudioStart is not 2 as expected");
48 static_assert(static_cast<size_t>(SpeechRecognitionUpdateType::SoundStart) == 2, "SpeechRecognitionUpdateType::SoundStart is not 3 as expected");
49 static_assert(static_cast<size_t>(SpeechRecognitionUpdateType::SpeechStart) == 3, "SpeechRecognitionUpdateType::SpeechStart is not 4 as expected");
50 static_assert(static_cast<size_t>(SpeechRecognitionUpdateType::SpeechEnd) == 4, "SpeechRecognitionUpdateType::SpeechEnd is not 5 as expected");
51 static_assert(static_cast<size_t>(SpeechRecognitionUpdateType::SoundEnd) == 5, "SpeechRecognitionUpdateType::SoundEnd is not 6 as expected");
52 static_assert(static_cast<size_t>(SpeechRecognitionUpdateType::AudioEnd) == 6, "SpeechRecognitionUpdateType::AudioEnd is not 7 as expected");
53 static_assert(static_cast<size_t>(SpeechRecognitionUpdateType::Result) == 7, "SpeechRecognitionUpdateType::Result is not 8 as expected");
54 static_assert(static_cast<size_t>(SpeechRecognitionUpdateType::NoMatch) == 8, "SpeechRecognitionUpdateType::NoMatch is not 9 as expected");
55 static_assert(static_cast<size_t>(SpeechRecognitionUpdateType::Error) == 9, "SpeechRecognitionUpdateType::Error is not 10 as expected");
56 static_assert(static_cast<size_t>(SpeechRecognitionUpdateType::End) == 10, "SpeechRecognitionUpdateType::End is not 11 as expected");
57 ASSERT(static_cast<size_t>(enumerationValue) < WTF_ARRAY_LENGTH(values));
58 return values[static_cast<size_t>(enumerationValue)];
59}

61SpeechRecognitionUpdate SpeechRecognitionUpdate::create(SpeechRecognitionConnectionClientIdentifier clientIdentifier, SpeechRecognitionUpdateType type)
62{
63 return SpeechRecognitionUpdate { clientIdentifier, type, WTF::Monostate() };
64}
65
66SpeechRecognitionUpdate SpeechRecognitionUpdate::createError(SpeechRecognitionConnectionClientIdentifier clientIdentifier, const SpeechRecognitionError& error)
67{
68 return SpeechRecognitionUpdate { clientIdentifier, SpeechRecognitionUpdateType::Error, error };
69}
70
71SpeechRecognitionUpdate SpeechRecognitionUpdate::createResult(SpeechRecognitionConnectionClientIdentifier clientIdentifier, const Vector<SpeechRecognitionResultData>& results)
72{
73 return SpeechRecognitionUpdate { clientIdentifier, SpeechRecognitionUpdateType::Result, results };
74}

76SpeechRecognitionUpdate::SpeechRecognitionUpdate(SpeechRecognitionConnectionClientIdentifier clientIdentifier, SpeechRecognitionUpdateType type, Content content)
77 : m_clientIdentifier(clientIdentifier)
78 , m_type(type)
79 , m_content(content)
80{
81}

83SpeechRecognitionError SpeechRecognitionUpdate::error() const
84{
85 return WTF::switchOn(m_content,
86 [] (const SpeechRecognitionError& error) { return error; },
87 [] (const auto&) { return SpeechRecognitionError(); }
88 );
89}
90
91Vector<SpeechRecognitionResultData> SpeechRecognitionUpdate::result() const
92{
93 return WTF::switchOn(m_content,
94 [] (const Vector<SpeechRecognitionResultData>& data) { return data; },
95 [] (const auto&) { return Vector<SpeechRecognitionResultData> { }; }
96 );
97}
