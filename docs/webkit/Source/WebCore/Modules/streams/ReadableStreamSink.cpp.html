35namespace WebCore {
36
37ReadableStreamToSharedBufferSink::ReadableStreamToSharedBufferSink(Callback&& callback)
38 : m_callback { WTFMove(callback) }
39{
40}
41
42void ReadableStreamToSharedBufferSink::pipeFrom(ReadableStream& stream)
43{
44 stream.pipeTo(*this);
45}
46
47void ReadableStreamToSharedBufferSink::enqueue(const BufferSource& buffer)
48{
49 if (!buffer.length())
50 return;
51
52 if (m_callback) {
53 Span chunk { buffer.data(), buffer.length() };
54 m_callback(&chunk);
55 }
56}
57
58void ReadableStreamToSharedBufferSink::close()
59{
60 if (m_callback)
61 m_callback(nullptr);
62}
63
64void ReadableStreamToSharedBufferSink::error(String&& message)
65{
66 if (auto callback = WTFMove(m_callback))
67 callback(Exception { TypeError, WTFMove(message) });
68}
