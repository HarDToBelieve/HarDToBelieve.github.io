27#pragma once
28
29#include "ExceptionOr.h"
30#include <wtf/Function.h>
31#include <wtf/RefCounted.h>
32#include <wtf/Span.h>
33
34namespace WebCore {
35
36class BufferSource;
37class ReadableStream;
38class SharedBuffer;
39
40class ReadableStreamSink : public RefCounted<ReadableStreamSink> {
41public:
42 virtual ~ReadableStreamSink() = default;
43
44 virtual void enqueue(const BufferSource&) = 0;
45 virtual void close() = 0;
46 virtual void error(String&&) = 0;
47};
48
49class ReadableStreamToSharedBufferSink final : public ReadableStreamSink {
50public:
51 using Callback = Function<void(ExceptionOr<Span<const uint8_t>*>&&)>;
52 static Ref<ReadableStreamToSharedBufferSink> create(Callback&& callback) { return adoptRef(*new ReadableStreamToSharedBufferSink(WTFMove(callback))); }
53 void pipeFrom(ReadableStream&);
54 void clearCallback() { m_callback = { }; }
55
56private:
57 explicit ReadableStreamToSharedBufferSink(Callback&&);
58
59 void enqueue(const BufferSource&) final;
60 void close() final;
61 void error(String&&) final;
62
63 Callback m_callback;
64};
65
66} // namespace WebCore
