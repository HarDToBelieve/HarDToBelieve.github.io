30namespace WebCore {
31
32ReadableStreamSource::~ReadableStreamSource() = default;
33
34void ReadableStreamSource::start(ReadableStreamDefaultController&& controller, DOMPromiseDeferred<void>&& promise)
35{
36 ASSERT(!m_promise);
37 m_promise = WTF::makeUnique<DOMPromiseDeferred<void>>(WTFMove(promise));
38 m_controller = WTFMove(controller);
39
40 setActive();
41 doStart();
42}
43
44void ReadableStreamSource::pull(DOMPromiseDeferred<void>&& promise)
45{
46 ASSERT(!m_promise);
47 ASSERT(m_controller);
48
49 m_promise = WTF::makeUnique<DOMPromiseDeferred<void>>(WTFMove(promise));
50
51 setActive();
52 doPull();
53}
54
55void ReadableStreamSource::startFinished()
56{
57 ASSERT(m_promise);
58 m_promise->resolve();
59 m_promise = nullptr;
60 setInactive();
61}
62
63void ReadableStreamSource::pullFinished()
64{
65 ASSERT(m_promise);
66 m_promise->resolve();
67 m_promise = nullptr;
68 setInactive();
69}
70
71void ReadableStreamSource::cancel(JSC::JSValue)
72{
73 clean();
74 doCancel();
75}
76
77void ReadableStreamSource::clean()
78{
79 if (m_promise) {
80 m_promise = nullptr;
81 setInactive();
82 }
83}
84
85void SimpleReadableStreamSource::doCancel()
86{
87 m_isCancelled = true;
88}
89
90void SimpleReadableStreamSource::close()
91{
92 if (!m_isCancelled)
93 controller().close();
94}
95
96void SimpleReadableStreamSource::enqueue(JSC::JSValue value)
97{
98 if (!m_isCancelled)
99 controller().enqueue(value);
100}
