35namespace WebCore {
36
37class ReadableStreamSource : public RefCounted<ReadableStreamSource> {
38public:
39 virtual ~ReadableStreamSource();
40
41 void start(ReadableStreamDefaultController&&, DOMPromiseDeferred<void>&&);
42 void pull(DOMPromiseDeferred<void>&&);
43 void cancel(JSC::JSValue);
44
45 bool isPulling() const { return !!m_promise; }
46
47protected:
48 ReadableStreamDefaultController& controller() { return m_controller.value(); }
49 const ReadableStreamDefaultController& controller() const { return m_controller.value(); }
50
51 void startFinished();
52 void pullFinished();
53 void cancelFinished();
54 void clean();
55
56 virtual void setActive() = 0;
57 virtual void setInactive() = 0;
58
59 virtual void doStart() = 0;
60 virtual void doPull() = 0;
61 virtual void doCancel() = 0;
62
63private:
64 std::unique_ptr<DOMPromiseDeferred<void>> m_promise;
65 std::optional<ReadableStreamDefaultController> m_controller;
66};
67
68class SimpleReadableStreamSource
69 : public ReadableStreamSource
70 , public CanMakeWeakPtr<SimpleReadableStreamSource> {
71public:
72 static Ref<SimpleReadableStreamSource> create() { return adoptRef(*new SimpleReadableStreamSource); }
73
74 void close();
75 void enqueue(JSC::JSValue);
76
77private:
78 SimpleReadableStreamSource() = default;
79
80 // ReadableStreamSource
81 void setActive() final { }
82 void setInactive() final { }
83 void doStart() final { }
84 void doPull() final { }
85 void doCancel() final;
86
87 bool m_isCancelled { false };
88};
