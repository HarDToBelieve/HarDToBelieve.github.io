39ExceptionOr<Ref<AnalyserNode>> AnalyserNode::create(BaseAudioContext& context, const AnalyserOptions& options)
40{
41 auto analyser = adoptRef(*new AnalyserNode(context));
42
43 auto result = analyser->handleAudioNodeOptions(options, { 2, ChannelCountMode::Max, ChannelInterpretation::Speakers });
44 if (result.hasException())
45 return result.releaseException();
46
47 result = analyser->setMinMaxDecibels(options.minDecibels, options.maxDecibels);
48 if (result.hasException())
49 return result.releaseException();
50
51 result = analyser->setFftSize(options.fftSize);
52 if (result.hasException())
53 return result.releaseException();
54
55 result = analyser->setSmoothingTimeConstant(options.smoothingTimeConstant);
56 if (result.hasException())
57 return result.releaseException();
58
59 return analyser;
60}

75void AnalyserNode::process(size_t framesToProcess)
76{
77 AudioBus* outputBus = output(0)->bus();
78
79 if (!isInitialized()) {
80 outputBus->zero();
81 return;
82 }
83
84 AudioBus* inputBus = input(0)->bus();
85
86 // Give the analyser the audio which is passing through this AudioNode. This must always
87 // be done so that the state of the Analyser reflects the current input.
88 m_analyser.writeInput(inputBus, framesToProcess);
89
90 if (!input(0)->isConnected()) {
91 outputBus->zero();
92 return;
93 }
94
95 // For in-place processing, our override of pullInputs() will just pass the audio data through unchanged if the channel count matches from input to output
96 // (resulting in inputBus == outputBus). Otherwise, do an up-mix to stereo.
97 if (inputBus != outputBus)
98 outputBus->copyFrom(*inputBus);
99}

108ExceptionOr<void> AnalyserNode::setMinMaxDecibels(double minDecibels, double maxDecibels)
109{
110 if (maxDecibels <= minDecibels)
111 return Exception { IndexSizeError, "minDecibels must be less than maxDecibels."_s };
112
113 m_analyser.setMinDecibels(minDecibels);
114 m_analyser.setMaxDecibels(maxDecibels);
115 return { };
116}

118ExceptionOr<void> AnalyserNode::setMinDecibels(double k)
119{
120 if (k >= maxDecibels())
121 return Exception { IndexSizeError, "minDecibels must be less than maxDecibels."_s };
122
123 m_analyser.setMinDecibels(k);

127ExceptionOr<void> AnalyserNode::setMaxDecibels(double k)
128{

136ExceptionOr<void> AnalyserNode::setSmoothingTimeConstant(double k)

151void AnalyserNode::updatePullStatus()
152{
153 ASSERT(context().isGraphOwner());
154
155 if (output(0)->isConnected()) {
156 // When an AudioBasicInspectorNode is connected to a downstream node, it
157 // will get pulled by the downstream node, thus remove it from the context's
158 // automatic pull list.
159 if (m_needAutomaticPull) {
160 context().removeAutomaticPullNode(*this);
161 m_needAutomaticPull = false;
162 }
163 } else {
164 unsigned numberOfInputConnections = input(0)->numberOfRenderingConnections();
165 // When an AnalyserNode is not connected to any downstream node
166 // while still connected from upstream node(s), add it to the context's
167 // automatic pull list.
168 //
169 // But don't remove the AnalyserNode if there are no inputs
170 // connected to the node. The node needs to be pulled so that the
171 // internal state is updated with the correct input signal (of
172 // zeroes).
173 if (numberOfInputConnections && !m_needAutomaticPull) {
174 context().addAutomaticPullNode(*this);
175 m_needAutomaticPull = true;
176 }
177 }
178}

180bool AnalyserNode::propagatesSilence() const
181{
182 // An AnalyserNode does actually propogate silence, but to get the
183 // time and FFT data updated correctly, process() needs to be
184 // called even if all the inputs are silent.
185 return false;
186}

188double AnalyserNode::tailTime() const
189{
190 return RealtimeAnalyser::MaxFFTSize / static_cast<double>(context().sampleRate());
191}
