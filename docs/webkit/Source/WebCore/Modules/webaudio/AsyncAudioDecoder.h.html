27#include "ExceptionOr.h"

34namespace JSC {
35class ArrayBuffer;
36}
37
38namespace WebCore {
39
40class AudioBuffer;
41class AudioBufferCallback;
42
43// AsyncAudioDecoder asynchronously decodes audio file data from an ArrayBuffer in a worker thread.
44// Upon successful decoding, a completion callback will be invoked with the decoded PCM data in an AudioBuffer.
45
46class AsyncAudioDecoder final {
47 WTF_MAKE_FAST_ALLOCATED;
48 WTF_MAKE_NONCOPYABLE(AsyncAudioDecoder);
49public:
50 AsyncAudioDecoder();
51 ~AsyncAudioDecoder();
52
53 // Must be called on the main thread.
54 void decodeAsync(Ref<JSC::ArrayBuffer>&& audioData, float sampleRate, Function<void(ExceptionOr<Ref<AudioBuffer>>&&)>&&);
55
56private:
57 class DecodingTask {
58 WTF_MAKE_NONCOPYABLE(DecodingTask);
59 WTF_MAKE_FAST_ALLOCATED;
60 public:
61 DecodingTask(Ref<JSC::ArrayBuffer>&& audioData, float sampleRate, Function<void(ExceptionOr<Ref<AudioBuffer>>&&)>&& callback);
62 void decode();
63
64 private:
65 JSC::ArrayBuffer& audioData() { return m_audioData; }
66 float sampleRate() const { return m_sampleRate; }
67 Function<void(ExceptionOr<Ref<AudioBuffer>>&&)>& callback() { return m_callback; }
68 AudioBuffer* audioBuffer() { return m_audioBuffer.get(); }
69
70 void notifyComplete();
71
72 Ref<JSC::ArrayBuffer> m_audioData;
73 float m_sampleRate;
74 Function<void(ExceptionOr<Ref<AudioBuffer>>&&)> m_callback;
75 RefPtr<AudioBuffer> m_audioBuffer;
76 };
77
78 void runLoop();
79
80 RefPtr<Thread> m_thread;
81 Lock m_threadCreationMutex;
82 MessageQueue<DecodingTask> m_queue;
83};
84
85} // namespace WebCore
