39AudioBasicInspectorNode::AudioBasicInspectorNode(BaseAudioContext& context, NodeType type)
40 : AudioNode(context, type)
41{
42 addInput();
43}
44
45// We override pullInputs() as an optimization allowing this node to take advantage of in-place processing,
46// where the input is simply passed through unprocessed to the output.
47// Note: this only applies if the input and output channel counts match.
48void AudioBasicInspectorNode::pullInputs(size_t framesToProcess)
49{
50 // Render input stream - try to render directly into output bus for pass-through processing where process() doesn't need to do anything...
51 auto* output = this->output(0);
52 input(0)->pull(output ? output->bus() : nullptr, framesToProcess);
53}
54
55void AudioBasicInspectorNode::checkNumberOfChannelsForInput(AudioNodeInput* input)
56{
57 ASSERT(context().isAudioThread() && context().isGraphOwner());
58
59 ASSERT(input == this->input(0));
60 if (input != this->input(0))
61 return;
62
63 if (auto* output = this->output(0)) {
64 unsigned numberOfChannels = input->numberOfChannels();
65
66 if (numberOfChannels != output->numberOfChannels()) {
67 // This will propagate the channel count to any nodes connected further downstream in the graph.
68 output->setNumberOfChannels(numberOfChannels);
69 }
70 }
71
72 AudioNode::checkNumberOfChannelsForInput(input);
73
74 updatePullStatus();
75}
76
77void AudioBasicInspectorNode::updatePullStatus()
78{
79 ASSERT(context().isGraphOwner());
80
81 auto output = this->output(0);
82 if (output && output->isConnected()) {
83 // When an AudioBasicInspectorNode is connected to a downstream node, it will get pulled by the
84 // downstream node, thus remove it from the context's automatic pull list.
85 if (m_needAutomaticPull) {
86 context().removeAutomaticPullNode(*this);
87 m_needAutomaticPull = false;
88 }
89 } else {
90 unsigned numberOfInputConnections = input(0)->numberOfRenderingConnections();
91 if (numberOfInputConnections && !m_needAutomaticPull) {
92 // When an AudioBasicInspectorNode is not connected to any downstream node while still connected from
93 // upstream node(s), add it to the context's automatic pull list.
94 context().addAutomaticPullNode(*this);
95 m_needAutomaticPull = true;
96 } else if (!numberOfInputConnections && m_needAutomaticPull) {
97 // The AudioBasicInspectorNode is connected to nothing and is not an AnalyserNode, remove it from the
98 // context's automatic pull list. AnalyserNode's need to be pulled even with no inputs so that the
99 // internal state gets updated to hold the right time and FFT data.
100 context().removeAutomaticPullNode(*this);
101 m_needAutomaticPull = false;
102 }
103 }
