51void AudioBasicProcessorNode::initialize()
52{
53 if (isInitialized())
54 return;
55
56 ASSERT(processor());
57 processor()->initialize();
58
59 AudioNode::initialize();
60}

62void AudioBasicProcessorNode::uninitialize()
63{
64 if (!isInitialized())
65 return;
66
67 ASSERT(processor());
68 processor()->uninitialize();
69
70 AudioNode::uninitialize();
71}

73void AudioBasicProcessorNode::process(size_t framesToProcess)
74{
75 AudioBus* destinationBus = output(0)->bus();
76
77 if (!isInitialized() || !processor() || processor()->numberOfChannels() != numberOfChannels())
78 destinationBus->zero();
79 else {
80 AudioBus* sourceBus = input(0)->bus();
81
82 // FIXME: if we take "tail time" into account, then we can avoid calling processor()->process() once the tail dies down.
83 if (!input(0)->isConnected())
84 sourceBus->zero();
85
86 processor()->process(sourceBus, destinationBus, framesToProcess);
87 }
88}

90void AudioBasicProcessorNode::processOnlyAudioParams(size_t framesToProcess)
91{
92 if (!isInitialized() || !processor())
93 return;
94
95 processor()->processOnlyAudioParams(framesToProcess);
96}

98// Nice optimization in the very common case allowing for "in-place" processing
99void AudioBasicProcessorNode::pullInputs(size_t framesToProcess)
100{
101 // Render input stream - suggest to the input to render directly into output bus for in-place processing in process() if possible.
102 input(0)->pull(output(0)->bus(), framesToProcess);

105// As soon as we know the channel count of our input, we can lazily initialize.
106// Sometimes this may be called more than once with different channel counts, in which case we must safely
107// uninitialize and then re-initialize with the new channel count.
108void AudioBasicProcessorNode::checkNumberOfChannelsForInput(AudioNodeInput* input)
109{
110 ASSERT(context().isAudioThread() && context().isGraphOwner());
111
112 ASSERT(input == this->input(0));
113 if (input != this->input(0))
114 return;
115
116 ASSERT(processor());
117 if (!processor())
118 return;
119
120 unsigned numberOfChannels = input->numberOfChannels();
121
122 if (isInitialized() && numberOfChannels != output(0)->numberOfChannels()) {
123 // We're already initialized but the channel count has changed.
124 uninitialize();
125 }
126
127 if (!isInitialized()) {
128 // This will propagate the channel count to any nodes connected further down the chain...
129 output(0)->setNumberOfChannels(numberOfChannels);
130
131 // Re-initialize the processor with the new channel count.
132 processor()->setNumberOfChannels(numberOfChannels);
133 initialize();
134 }
135
136 AudioNode::checkNumberOfChannelsForInput(input);
137}
