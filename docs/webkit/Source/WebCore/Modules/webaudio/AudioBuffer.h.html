30#pragma once
31
32#include "AudioBufferOptions.h"
33#include "ExceptionOr.h"
34#include "JSValueInWrappedObject.h"
35#include <JavaScriptCore/Float32Array.h>
36#include <wtf/Lock.h>
37#include <wtf/Vector.h>
38
39namespace WebCore {
40
41class AudioBus;
42
43class AudioBuffer : public RefCounted<AudioBuffer> {
44public:
45 enum class LegacyPreventDetaching : bool { No, Yes };
46 static RefPtr<AudioBuffer> create(unsigned numberOfChannels, size_t numberOfFrames, float sampleRate, LegacyPreventDetaching = LegacyPreventDetaching::No);
47 static ExceptionOr<Ref<AudioBuffer>> create(const AudioBufferOptions&);
48 // Returns nullptr if data is not a valid audio file.
49 static RefPtr<AudioBuffer> createFromAudioFileData(const void* data, size_t dataSize, bool mixToMono, float sampleRate);
50
51 // Format
52 size_t originalLength() const { return m_originalLength; }
53 double originalDuration() const { return originalLength() / static_cast<double>(sampleRate()); }
54 float sampleRate() const { return m_sampleRate; }
55
56 // The following function may start returning 0 if any of the underlying channel buffers gets detached.
57 size_t length() const { return hasDetachedChannelBuffer() ? 0 : m_originalLength; }
58 double duration() const { return length() / static_cast<double>(sampleRate()); }
59
60 // Channel data access
61 unsigned numberOfChannels() const { return m_channels.size(); }
62 ExceptionOr<JSC::JSValue> getChannelData(JSDOMGlobalObject&, unsigned channelIndex);
63 ExceptionOr<void> copyFromChannel(Ref<Float32Array>&&, unsigned channelNumber, unsigned bufferOffset);
64 ExceptionOr<void> copyToChannel(Ref<Float32Array>&&, unsigned channelNumber, unsigned startInChannel);
65
66 // Native channel data access.
67 RefPtr<Float32Array> channelData(unsigned channelIndex);
68 float* rawChannelData(unsigned channelIndex);
69 void zero();
70
71 // Because an AudioBuffer has a JavaScript wrapper, which will be garbage collected, it may take a while for this object to be deleted.
72 // releaseMemory() can be called when the AudioContext goes away, so we can release the memory earlier than when the garbage collection happens.
73 // Careful! Only call this when the page unloads, after the AudioContext is no longer processing.
74 void releaseMemory();
75
76 size_t memoryCost() const;
77
78 template<typename Visitor> void visitChannelWrappers(Visitor&);
79
80 bool copyTo(AudioBuffer&) const;
81
82 enum class ShouldCopyChannelData : bool { No, Yes };
83 Ref<AudioBuffer> clone(ShouldCopyChannelData = ShouldCopyChannelData::Yes) const;
84
85 bool topologyMatches(const AudioBuffer&) const;
86
87private:
88 AudioBuffer(unsigned numberOfChannels, size_t length, float sampleRate, LegacyPreventDetaching = LegacyPreventDetaching::No);
89 explicit AudioBuffer(AudioBus&);
90
91 void invalidate();
92
93 bool hasDetachedChannelBuffer() const;
94
95 float m_sampleRate;
96 mutable Lock m_channelsLock;
97 size_t m_originalLength;
98 Vector<RefPtr<Float32Array>> m_channels;
99 Vector<JSValueInWrappedObject> m_channelWrappers;
100 bool m_isDetachable { true };
101};
102
103} // namespace WebCore
