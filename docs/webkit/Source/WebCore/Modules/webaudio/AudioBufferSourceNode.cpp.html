45constexpr double DefaultGrainDuration = 0.020; // 20ms
46
47// Arbitrary upper limit on playback rate.
48// Higher than expected rates can be useful when playing back oversampled buffers
49// to minimize linear interpolation aliasing.
50const double MaxRate = 1024;
51
52static float computeSampleUsingLinearInterpolation(const float* source, unsigned readIndex, unsigned readIndex2, float interpolationFactor)
53{
54 if (readIndex == readIndex2 && readIndex >= 1) {
55 // We're at the end of the buffer, so just linearly extrapolate from the last two samples.
56 float sample1 = source[readIndex - 1];
57 float sample2 = source[readIndex];
58 return sample2 + (sample2 - sample1) * interpolationFactor;
59 }
60 float sample1 = source[readIndex];
61 float sample2 = source[readIndex2];
62 return sample1 + interpolationFactor * (sample2 - sample1);
63}

65ExceptionOr<Ref<AudioBufferSourceNode>> AudioBufferSourceNode::create(BaseAudioContext& context, AudioBufferSourceOptions&& options)
66{
67 auto node = adoptRef(*new AudioBufferSourceNode(context));

69 node->setBufferForBindings(WTFMove(options.buffer));
70 node->detune().setValue(options.detune);
71 node->setLoopForBindings(options.loop);

96void AudioBufferSourceNode::process(size_t framesToProcess)
97{
98 auto& outputBus = *output(0)->bus();
99
100 if (!isInitialized()) {
101 outputBus.zero();
102 return;
103 }
104
105 // The audio thread can't block on this lock, so we use tryLock() instead.
106 if (!m_processLock.tryLock()) {
107 // Too bad - tryLock() failed. We must be in the middle of changing buffers and were already outputting silence anyway.
108 outputBus.zero();
109 return;
110 }
111 Locker locker { AdoptLock, m_processLock };
112
113 if (!m_buffer) {
114 outputBus.zero();
115 return;
116 }
117
118 // After calling setBuffer() with a buffer having a different number of channels, there can in rare cases be a slight delay
119 // before the output bus is updated to the new number of channels because of use of tryLocks() in the context's updating system.
120 // In this case, if the buffer has just been changed and we're not quite ready yet, then just output silence.
121 if (numberOfChannels() != m_buffer->numberOfChannels()) {
122 outputBus.zero();
123 return;
124 }
125
126 size_t quantumFrameOffset = 0;
127 size_t bufferFramesToProcess = 0;
128 double startFrameOffset = 0;
129 updateSchedulingInfo(framesToProcess, outputBus, quantumFrameOffset, bufferFramesToProcess, startFrameOffset);
130
131 if (!bufferFramesToProcess) {
132 outputBus.zero();
133 return;
134 }
135
136 for (unsigned i = 0; i < outputBus.numberOfChannels(); ++i)
137 m_destinationChannels[i] = outputBus.channel(i)->mutableData();
138
139 // Render by reading directly from the buffer.
140 if (!renderFromBuffer(&outputBus, quantumFrameOffset, bufferFramesToProcess, startFrameOffset)) {
141 outputBus.zero();
142 return;
143 }
144
145 outputBus.clearSilentFlag();
146}

148// Returns true if we're finished.
149bool AudioBufferSourceNode::renderSilenceAndFinishIfNotLooping(AudioBus*, unsigned index, size_t framesToProcess)
150{
151 if (!m_isLooping) {
152 // If we're not looping, then stop playing when we get to the end.
153
154 if (framesToProcess > 0) {
155 // We're not looping and we've reached the end of the sample data, but we still need to provide more output,
156 // so generate silence for the remaining.
157 for (unsigned i = 0; i < numberOfChannels(); ++i)
158 memset(m_destinationChannels[i] + index, 0, sizeof(float) * framesToProcess);
159 }
160
161 if (!hasFinished())
162 finish();
163 return true;
164 }
165 return false;
166}

168bool AudioBufferSourceNode::renderFromBuffer(AudioBus* bus, unsigned destinationFrameOffset, size_t numberOfFrames, double startFrameOffset)
169{
170 ASSERT(context().isAudioThread());
171
172 // Basic sanity checking
173 ASSERT(bus);
174 ASSERT(m_buffer);
175 if (!bus || !m_buffer)
176 return false;
177
178 unsigned numberOfChannels = this->numberOfChannels();
179 unsigned busNumberOfChannels = bus->numberOfChannels();
180
181 bool channelCountGood = numberOfChannels && numberOfChannels == busNumberOfChannels;
182 ASSERT(channelCountGood);
183 if (!channelCountGood)
184 return false;
185
186 // Sanity check destinationFrameOffset, numberOfFrames.
187 size_t destinationLength = bus->length();
188
189 bool isLengthGood = destinationLength <= 4096 && numberOfFrames <= 4096;
190 ASSERT(isLengthGood);
191 if (!isLengthGood)
192 return false;
193
194 bool isOffsetGood = destinationFrameOffset <= destinationLength && destinationFrameOffset + numberOfFrames <= destinationLength;
195 ASSERT(isOffsetGood);
196 if (!isOffsetGood)
197 return false;
198
199 // Potentially zero out initial frames leading up to the offset.
200 if (destinationFrameOffset) {
201 for (unsigned i = 0; i < numberOfChannels; ++i)
202 memset(m_destinationChannels[i], 0, sizeof(float) * destinationFrameOffset);
203 }
204
205 // Offset the pointers to the correct offset frame.
206 unsigned writeIndex = destinationFrameOffset;
207
208 size_t bufferLength = m_buffer->length();
209 double bufferSampleRate = m_buffer->sampleRate();
210 double pitchRate = totalPitchRate();
211 bool reverse = pitchRate < 0;
212
213 // Avoid converting from time to sample-frames twice by computing
214 // the grain end time first before computing the sample frame.
215 unsigned maxFrame;
216 if (m_isGrain)
217 maxFrame = AudioUtilities::timeToSampleFrame(m_grainOffset + m_grainDuration, bufferSampleRate);
218 else
219 maxFrame = bufferLength;
220
221 // Do some sanity checking.
222 if (maxFrame > bufferLength)
223 maxFrame = bufferLength;
224
225 // If the .loop attribute is true, then values of m_loopStart == 0 && m_loopEnd == 0 implies
226 // that we should use the entire buffer as the loop, otherwise use the loop values in m_loopStart and m_loopEnd.
227 double virtualMaxFrame = maxFrame;
228 double virtualMinFrame = 0;
229 double virtualDeltaFrames = maxFrame;
230
231 if (m_isLooping && (m_loopStart || m_loopEnd) && m_loopStart >= 0 && m_loopEnd > 0 && m_loopStart < m_loopEnd) {
232 // Convert from seconds to sample-frames.
233 double loopMinFrame = m_loopStart * m_buffer->sampleRate();
234 double loopMaxFrame = m_loopEnd * m_buffer->sampleRate();
235
236 virtualMaxFrame = std::min(loopMaxFrame, virtualMaxFrame);
237 virtualMinFrame = std::max(loopMinFrame, virtualMinFrame);
238 virtualDeltaFrames = virtualMaxFrame - virtualMinFrame;
239 }
240
241 // If we're looping and the offset (virtualReadIndex) is past the end of the loop, wrap back to the
242 // beginning of the loop. For other cases, nothing needs to be done.
243 if (m_isLooping && m_virtualReadIndex >= virtualMaxFrame) {
244 m_virtualReadIndex = (m_loopStart < 0) ? 0 : (m_loopStart * m_buffer->sampleRate());
245 m_virtualReadIndex = std::min(m_virtualReadIndex, static_cast<double>(bufferLength - 1));
246 }
247
248 // Sanity check that our playback rate isn't larger than the loop size.
249 if (std::abs(pitchRate) > virtualDeltaFrames)
250 return false;
251
252 // Get local copy.
253 double virtualReadIndex = m_virtualReadIndex;
254
255 // Adjust the read index by the startFrameOffset (compensated by the pitch rate) because
256 // we always start output on a frame boundary with interpolation if necessary.
257 if (startFrameOffset < 0 && pitchRate)
258 virtualReadIndex += std::abs(startFrameOffset * pitchRate);
259
260 bool needsInterpolation = virtualReadIndex != floor(virtualReadIndex)
261 || virtualDeltaFrames != floor(virtualDeltaFrames)
262 || virtualMaxFrame != floor(virtualMaxFrame)
263 || virtualMinFrame != floor(virtualMinFrame);
264
265 // Render loop - reading from the source buffer to the destination using linear interpolation.
266 int framesToProcess = numberOfFrames;
267
268 const float** sourceChannels = m_sourceChannels.get();
269 float** destinationChannels = m_destinationChannels.get();
270
271 // Optimize for the very common case of playing back with pitchRate == 1.
272 // We can avoid the linear interpolation.
273 if (pitchRate == 1 && !needsInterpolation) {
274 unsigned readIndex = static_cast<unsigned>(virtualReadIndex);
275 unsigned deltaFrames = static_cast<unsigned>(virtualDeltaFrames);
276 maxFrame = static_cast<unsigned>(virtualMaxFrame);
277 while (framesToProcess > 0) {
278 int framesToEnd = maxFrame - readIndex;
279 int framesThisTime = std::min(framesToProcess, framesToEnd);
280 framesThisTime = std::max(0, framesThisTime);
281
282 for (unsigned i = 0; i < numberOfChannels; ++i)
283 memcpy(destinationChannels[i] + writeIndex, sourceChannels[i] + readIndex, sizeof(float) * framesThisTime);
284
285 writeIndex += framesThisTime;
286 readIndex += framesThisTime;
287 framesToProcess -= framesThisTime;
288
289 // Wrap-around.
290 if (readIndex >= maxFrame) {
291 readIndex -= deltaFrames;
292 if (renderSilenceAndFinishIfNotLooping(bus, writeIndex, framesToProcess))
293 break;
294 }
295 }
296 virtualReadIndex = readIndex;
297 } else if (pitchRate == -1 && !needsInterpolation) {
298 int readIndex = static_cast<int>(virtualReadIndex);
299 int deltaFrames = static_cast<int>(virtualDeltaFrames);
300 int minFrame = static_cast<int>(virtualMinFrame) - 1;
301 while (framesToProcess > 0) {
302 int framesToEnd = readIndex - minFrame;
303 int framesThisTime = std::min<int>(framesToProcess, framesToEnd);
304 framesThisTime = std::max<int>(0, framesThisTime);
305
306 while (framesThisTime--) {
307 for (unsigned i = 0; i < numberOfChannels; ++i) {
308 float* destination = destinationChannels[i];
309 const float* source = sourceChannels[i];
310
311 destination[writeIndex] = source[readIndex];
312 }
313
314 ++writeIndex;
315 --readIndex;
316 --framesToProcess;
317 }
318
319 // Wrap-around.
320 if (readIndex <= minFrame) {
321 readIndex += deltaFrames;
322 if (renderSilenceAndFinishIfNotLooping(bus, writeIndex, framesToProcess))
323 break;
324 }
325 }
326 virtualReadIndex = readIndex;
327 } else if (!pitchRate) {
328 unsigned readIndex = static_cast<unsigned>(virtualReadIndex);
329
330 for (unsigned i = 0; i < numberOfChannels; ++i)
331 std::fill_n(destinationChannels[i] + writeIndex, framesToProcess, sourceChannels[i][readIndex]);
332 } else if (reverse) {
333 unsigned maxFrame = static_cast<unsigned>(virtualMaxFrame);
334 unsigned minFrame = static_cast<unsigned>(floorf(virtualMinFrame));
335
336 while (framesToProcess--) {
337 unsigned readIndex = static_cast<unsigned>(floorf(virtualReadIndex));
338 float interpolationFactor = virtualReadIndex - readIndex;
339
340 unsigned readIndex2 = readIndex + 1;
341 if (readIndex2 >= maxFrame)
342 readIndex2 = m_isLooping ? minFrame : readIndex;
343
344 // Linear interpolation.
345 for (unsigned i = 0; i < numberOfChannels; ++i) {
346 float* destination = destinationChannels[i];
347 const float* source = sourceChannels[i];
348
349 destination[writeIndex] = computeSampleUsingLinearInterpolation(source, readIndex, readIndex2, interpolationFactor);
350 }
351
352 writeIndex++;
353
354 virtualReadIndex += pitchRate;
355
356 // Wrap-around, retaining sub-sample position since virtualReadIndex is floating-point.
357 if (virtualReadIndex < virtualMinFrame) {
358 virtualReadIndex += virtualDeltaFrames;
359 if (renderSilenceAndFinishIfNotLooping(bus, writeIndex, framesToProcess))
360 break;
361 }
362 }
363 } else {
364 while (framesToProcess--) {
365 unsigned readIndex = static_cast<unsigned>(virtualReadIndex);
366 float interpolationFactor = virtualReadIndex - readIndex;
367
368 // For linear interpolation we need the next sample-frame too.
369 unsigned readIndex2 = readIndex + 1;
370 if (readIndex2 >= bufferLength) {
371 if (m_isLooping) {
372 // Make sure to wrap around at the end of the buffer.
373 readIndex2 = static_cast<unsigned>(virtualReadIndex + 1 - virtualDeltaFrames);
374 } else
375 readIndex2 = readIndex;
376 }
377
378 // Final sanity check on buffer access.
379 // FIXME: as an optimization, try to get rid of this inner-loop check and put assertions and guards before the loop.
380 if (readIndex >= bufferLength || readIndex2 >= bufferLength)
381 break;
382
383 // Linear interpolation.
384 for (unsigned i = 0; i < numberOfChannels; ++i) {
385 float* destination = destinationChannels[i];
386 const float* source = sourceChannels[i];
387
388 destination[writeIndex] = computeSampleUsingLinearInterpolation(source, readIndex, readIndex2, interpolationFactor);
389 }
390 writeIndex++;
391
392 virtualReadIndex += pitchRate;
393
394 // Wrap-around, retaining sub-sample position since virtualReadIndex is floating-point.
395 if (virtualReadIndex >= virtualMaxFrame) {
396 virtualReadIndex -= virtualDeltaFrames;
397 if (renderSilenceAndFinishIfNotLooping(bus, writeIndex, framesToProcess))
398 break;
399 }
400 }
401 }
402
403 bus->clearSilentFlag();
404
405 m_virtualReadIndex = virtualReadIndex;
406
407 return true;
408}

410ExceptionOr<void> AudioBufferSourceNode::setBufferForBindings(RefPtr<AudioBuffer>&& buffer)
411{
412 ASSERT(isMainThread());
413 DEBUG_LOG(LOGIDENTIFIER);
414
415 // The context must be locked since changing the buffer can re-configure the number of channels that are output.
416 Locker contextLocker { context().graphLock() };
417
418 // This synchronizes with process().
419 Locker locker { m_processLock };
420
421 if (buffer && m_wasBufferSet)
422 return Exception { InvalidStateError, "The buffer was already set"_s };
423
424 if (buffer) {
425 m_wasBufferSet = true;
426
427 // Do any necesssary re-configuration to the buffer's number of channels.
428 unsigned numberOfChannels = buffer->numberOfChannels();
429 ASSERT(numberOfChannels <= AudioContext::maxNumberOfChannels);
430
431 output(0)->setNumberOfChannels(numberOfChannels);
432
433 m_sourceChannels = makeUniqueArray<const float*>(numberOfChannels);
434 m_destinationChannels = makeUniqueArray<float*>(numberOfChannels);
435
436 for (unsigned i = 0; i < numberOfChannels; ++i)
437 m_sourceChannels[i] = buffer->channelData(i)->data();
438 }
439
440 m_virtualReadIndex = 0;
441 m_buffer = WTFMove(buffer);
442
443 // In case the buffer gets set after playback has started, we need to clamp the grain parameters now.
444 if (m_isGrain)
445 adjustGrainParameters();
446
447 return { };
448}

481ExceptionOr<void> AudioBufferSourceNode::startPlaying(double when, double grainOffset, std::optional<double> grainDuration)
482{
483 ASSERT(isMainThread());
484 ALWAYS_LOG(LOGIDENTIFIER, "when = ", when, ", offset = ", grainOffset, ", duration = ", grainDuration.value_or(0));
485
486 if (m_playbackState != UNSCHEDULED_STATE)
487 return Exception { InvalidStateError, "Cannot call start more than once."_s };
488
489 if (!std::isfinite(when) || (when < 0))
490 return Exception { RangeError, "when value should be positive"_s };
491
492 if (!std::isfinite(grainOffset) || (grainOffset < 0))
493 return Exception { RangeError, "offset value should be positive"_s };
494
495 if (grainDuration && (!std::isfinite(*grainDuration) || (*grainDuration < 0)))
496 return Exception { RangeError, "duration value should be positive"_s };
497
498 context().sourceNodeWillBeginPlayback(*this);
499
500 // This synchronizes with process().
501 Locker locker { m_processLock };
502
503 m_isGrain = true;
504 m_grainOffset = grainOffset;
505 m_grainDuration = grainDuration.value_or(0);
506 m_wasGrainDurationGiven = !!grainDuration;
507 m_startTime = when;
508
509 adjustGrainParameters();
510
511 m_playbackState = SCHEDULED_STATE;
512
513 return { };
514}

516void AudioBufferSourceNode::adjustGrainParameters()
517{
518 ASSERT(m_processLock.isHeld());
519
520 if (!m_buffer)
521 return;
522
523 // Do sanity checking of grain parameters versus buffer size.
524 double bufferDuration = m_buffer->duration();
525
526 m_grainOffset = std::min(bufferDuration, m_grainOffset);
527
528 if (!m_wasGrainDurationGiven)
529 m_grainDuration = bufferDuration - m_grainOffset;
530
531 if (m_wasGrainDurationGiven && m_isLooping) {
532 // We're looping a grain with a grain duration specified. Schedule the loop
533 // to stop after grainDuration seconds after starting, possibly running the
534 // loop multiple times if grainDuration is larger than the buffer duration.
535 // The net effect is as if the user called stop(when + grainDuration).
536 m_grainDuration = clampTo(m_grainDuration, 0.0, std::numeric_limits<double>::infinity());
537 m_endTime = m_startTime + m_grainDuration;
538 } else
539 m_grainDuration = clampTo(m_grainDuration, 0.0, bufferDuration - m_grainOffset);
540
541 // We call timeToSampleFrame here since at playbackRate == 1 we don't want to go through linear interpolation
542 // at a sub-sample position since it will degrade the quality.
543 // When aligned to the sample-frame the playback will be identical to the PCM data stored in the buffer.
544 // Since playbackRate == 1 is very common, it's worth considering quality.
545 if (playbackRate().value() < 0)
546 m_virtualReadIndex = AudioUtilities::timeToSampleFrame(m_grainOffset + m_grainDuration, m_buffer->sampleRate()) - 1;
547 else
548 m_virtualReadIndex = AudioUtilities::timeToSampleFrame(m_grainOffset, m_buffer->sampleRate());
549}

551double AudioBufferSourceNode::totalPitchRate()
552{
553 // Incorporate buffer's sample-rate versus AudioContext's sample-rate.
554 // Normally it's not an issue because buffers are loaded at the AudioContext's sample-rate, but we can handle it in any case.
555 double sampleRateFactor = 1.0;
556 if (m_buffer)
557 sampleRateFactor = m_buffer->sampleRate() / static_cast<double>(sampleRate());
558
559 double basePitchRate = playbackRate().finalValue();
560 double detune = pow(2, m_detune->finalValue() / 1200);
561
562 double totalRate = sampleRateFactor * basePitchRate * detune;
563
564 totalRate = std::clamp(totalRate, -MaxRate, MaxRate);
565
566 bool isTotalRateValid = !std::isnan(totalRate) && !std::isinf(totalRate);
567 ASSERT(isTotalRateValid);
568 if (!isTotalRateValid)
569 totalRate = 1.0;
570
571 return totalRate;
572}

574bool AudioBufferSourceNode::propagatesSilence() const
575{
576 ASSERT(context().isAudioThread());
577 if (!isPlayingOrScheduled() || hasFinished())
578 return true;
579
580 if (!m_processLock.tryLock()) {
581 // We weren't able to get the lock so we assume we have a buffer.
582 return false;
583 }
584 Locker locker { AdoptLock, m_processLock };
585 return !m_buffer;
586}
