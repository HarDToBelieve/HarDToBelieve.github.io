38class DOMWindow;
39class HTMLMediaElement;

44struct AudioContextOptions;

47class AudioContext final
48 : public BaseAudioContext
49 , public MediaProducer
50 , public MediaCanStartListener
51 , private PlatformMediaSessionClient
52 , private VisibilityChangeClient {
53 WTF_MAKE_ISO_ALLOCATED(AudioContext);
54public:
55 // Create an AudioContext for rendering to the audio hardware.
56 static ExceptionOr<Ref<AudioContext>> create(Document&, AudioContextOptions&&);
57 ~AudioContext();
58
59 WEBCORE_EXPORT static void setDefaultSampleRateForTesting(std::optional<float>);
60
61 void close(DOMPromiseDeferred<void>&&);
62
63 DefaultAudioDestinationNode& destination() final { return m_destinationNode.get(); }
64 const DefaultAudioDestinationNode& destination() const final { return m_destinationNode.get(); }
65
66 double baseLatency();
67
68 AudioTimestamp getOutputTimestamp(DOMWindow&);
69
70#if ENABLE(VIDEO)
71 ExceptionOr<Ref<MediaElementAudioSourceNode>> createMediaElementSource(HTMLMediaElement&);
72#endif
73#if ENABLE(MEDIA_STREAM)
74 ExceptionOr<Ref<MediaStreamAudioSourceNode>> createMediaStreamSource(MediaStream&);
75 ExceptionOr<Ref<MediaStreamAudioDestinationNode>> createMediaStreamDestination();
76#endif
77
78 void suspendRendering(DOMPromiseDeferred<void>&&);
79 void resumeRendering(DOMPromiseDeferred<void>&&);
80
81 void sourceNodeWillBeginPlayback(AudioNode&) final;
82 void lazyInitialize() final;
83
84 void startRendering();
85
86 void isPlayingAudioDidChange();
87
88 // Restrictions to change default behaviors.
89 enum BehaviorRestrictionFlags {
90 NoRestrictions = 0,
91 RequireUserGestureForAudioStartRestriction = 1 << 0,
92 RequirePageConsentForAudioStartRestriction = 1 << 1,
93 };
94 typedef unsigned BehaviorRestrictions;
95 BehaviorRestrictions behaviorRestrictions() const { return m_restrictions; }
96 void addBehaviorRestriction(BehaviorRestrictions restriction) { m_restrictions |= restriction; }
97 void removeBehaviorRestriction(BehaviorRestrictions restriction) { m_restrictions &= ~restriction; }
98
99private:
100 AudioContext(Document&, const AudioContextOptions&);
101
102 bool willBeginPlayback();
103
104#if !RELEASE_LOG_DISABLED
105 const Logger& logger() const final;
106#endif
107
108 void constructCommon();
109
110 bool userGestureRequiredForAudioStart() const { return m_restrictions & RequireUserGestureForAudioStartRestriction; }
111 bool pageConsentRequiredForAudioStart() const { return m_restrictions & RequirePageConsentForAudioStartRestriction; }
112
113 bool willPausePlayback();
114
115 void uninitialize() final;
116 bool isOfflineContext() const final { return false; }
117
118 // MediaProducer
119 MediaProducer::MediaStateFlags mediaState() const final;
120 void pageMutedStateDidChange() final;
121
122 // PlatformMediaSessionClient
123 PlatformMediaSession::MediaType mediaType() const final { return isSuspended() || isStopped() ? PlatformMediaSession::MediaType::None : PlatformMediaSession::MediaType::WebAudio; }
124 PlatformMediaSession::MediaType presentationType() const final { return PlatformMediaSession::MediaType::WebAudio; }
125 void mayResumePlayback(bool shouldResume) final;
126 void suspendPlayback() final;
127 bool canReceiveRemoteControlCommands() const final { return false; }
128 void didReceiveRemoteControlCommand(PlatformMediaSession::RemoteControlCommandType, const PlatformMediaSession::RemoteCommandArgument&) final { }
129 bool supportsSeeking() const final { return false; }
130 bool shouldOverrideBackgroundPlaybackRestriction(PlatformMediaSession::InterruptionType) const final { return false; }
131 bool canProduceAudio() const final { return true; }
132 bool isSuspended() const final;
133 MediaSessionGroupIdentifier mediaSessionGroupIdentifier() const final;
134
135 // MediaCanStartListener.
136 void mediaCanStart(Document&) final;
137
138 // VisibilityChangeClient
139 void visibilityStateChanged() final;
140
141 // ActiveDOMObject
142 const char* activeDOMObjectName() const final;
143 void suspend(ReasonForSuspension) final;
144 void resume() final;
145 bool virtualHasPendingActivity() const final;
146
147 UniqueRef<DefaultAudioDestinationNode> m_destinationNode;
148 std::unique_ptr<PlatformMediaSession> m_mediaSession;
149
150 BehaviorRestrictions m_restrictions { NoRestrictions };
151
152 // [[suspended by user]] flag in the specification:
153 // https://www.w3.org/TR/webaudio/#dom-audiocontext-suspended-by-user-slot
154 bool m_wasSuspendedByScript { false };
155};

157} // WebCore
