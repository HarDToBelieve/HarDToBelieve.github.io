26#pragma once
27
28#include "AudioNode.h"
29#include <wtf/CompletionHandler.h>
30
31namespace WebCore {
32
33class AudioBus;
34struct AudioIOPosition;
35
36class AudioDestinationNode : public AudioNode {
37 WTF_MAKE_ISO_ALLOCATED(AudioDestinationNode);
38public:
39 ~AudioDestinationNode();
40
41 // AudioNode
42 void process(size_t) final { } // we're pulled by hardware so this is never called
43
44 float sampleRate() const final { return m_sampleRate; }
45
46 size_t currentSampleFrame() const { return m_currentSampleFrame; }
47 double currentTime() const { return currentSampleFrame() / static_cast<double>(sampleRate()); }
48
49 virtual unsigned maxChannelCount() const = 0;
50
51 // Enable local/live input for the specified device.
52 virtual void enableInput(const String& inputDeviceId) = 0;
53
54 virtual void startRendering(CompletionHandler<void(std::optional<Exception>&&)>&&) = 0;
55 virtual void restartRendering() { }
56
57 // AudioDestinationNodes are owned by the BaseAudioContext so we forward the refcounting to its BaseAudioContext.
58 void ref() final;
59 void deref() final;
60
61protected:
62 AudioDestinationNode(BaseAudioContext&, float sampleRate);
63
64 double tailTime() const final { return 0; }
65 double latencyTime() const final { return 0; }
66
67 void renderQuantum(AudioBus* destinationBus, size_t numberOfFrames, const AudioIOPosition& outputPosition);
68
69private:
70 // Counts the number of sample-frames processed by the destination.
71 std::atomic<size_t> m_currentSampleFrame { 0 };
72 float m_sampleRate;
73};
74
75} // namespace WebCore
