92void AudioListener::updateValuesIfNeeded(size_t framesToProcess)
93{
94 if (!positionX().context())
95 return;
96
97 double currentTime = positionX().context()->currentTime();
98 if (m_lastUpdateTime != currentTime) {
99 // Time has changed. Update all of the automation values now.
100 m_lastUpdateTime = currentTime;
101
102 ASSERT(framesToProcess <= m_positionXValues.size());
103 ASSERT(framesToProcess <= m_positionYValues.size());
104 ASSERT(framesToProcess <= m_positionZValues.size());
105 ASSERT(framesToProcess <= m_forwardXValues.size());
106 ASSERT(framesToProcess <= m_forwardYValues.size());
107 ASSERT(framesToProcess <= m_forwardZValues.size());
108 ASSERT(framesToProcess <= m_upXValues.size());
109 ASSERT(framesToProcess <= m_upYValues.size());
110 ASSERT(framesToProcess <= m_upZValues.size());
111
112 positionX().calculateSampleAccurateValues(m_positionXValues.data(), framesToProcess);
113 positionY().calculateSampleAccurateValues(m_positionYValues.data(), framesToProcess);
114 positionZ().calculateSampleAccurateValues(m_positionZValues.data(), framesToProcess);
115
116 forwardX().calculateSampleAccurateValues(m_forwardXValues.data(), framesToProcess);
117 forwardY().calculateSampleAccurateValues(m_forwardYValues.data(), framesToProcess);
118 forwardZ().calculateSampleAccurateValues(m_forwardZValues.data(), framesToProcess);
119
120 upX().calculateSampleAccurateValues(m_upXValues.data(), framesToProcess);
121 upY().calculateSampleAccurateValues(m_upYValues.data(), framesToProcess);
122 upZ().calculateSampleAccurateValues(m_upZValues.data(), framesToProcess);
123 }
124}

180ExceptionOr<void> AudioListener::setPosition(float x, float y, float z)
181{
182 ASSERT(isMainThread());
183 if (!m_positionX->context())
184 return { };
185
186 double now = m_positionX->context()->currentTime();
187
188 auto result = m_positionX->setValueAtTime(x, now);
189 if (result.hasException())
190 return result.releaseException();
191 result = m_positionY->setValueAtTime(y, now);
192 if (result.hasException())
193 return result.releaseException();
194 result = m_positionZ->setValueAtTime(z, now);
195 if (result.hasException())
196 return result.releaseException();
197
198 return { };
199}

206ExceptionOr<void> AudioListener::setOrientation(float x, float y, float z, float upX, float upY, float upZ)

233 return { };
234}
