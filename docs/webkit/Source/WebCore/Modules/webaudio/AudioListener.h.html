30#pragma once
31
32#include "AudioArray.h"
33#include "ExceptionOr.h"
34#include "FloatPoint3D.h"
35#include <wtf/Ref.h>
36#include <wtf/RefCounted.h>
37
38namespace WebCore {
39
40class AudioParam;
41class BaseAudioContext;
42
43// AudioListener maintains the state of the listener in the audio scene as defined in the OpenAL specification.
44
45class AudioListener : public RefCounted<AudioListener> {
46public:
47 static Ref<AudioListener> create(BaseAudioContext& context)
48 {
49 return adoptRef(*new AudioListener(context));
50 }
51 ~AudioListener();
52
53 AudioParam& positionX() { return m_positionX.get(); }
54 AudioParam& positionY() { return m_positionY.get(); }
55 AudioParam& positionZ() { return m_positionZ.get(); }
56 AudioParam& forwardX() { return m_forwardX.get(); }
57 AudioParam& forwardY() { return m_forwardY.get(); }
58 AudioParam& forwardZ() { return m_forwardZ.get(); }
59 AudioParam& upX() { return m_upX.get(); }
60 AudioParam& upY() { return m_upY.get(); }
61 AudioParam& upZ() { return m_upZ.get(); }
62
63 // Position
64 ExceptionOr<void> setPosition(float x, float y, float z);
65 FloatPoint3D position() const;
66
67 // Orientation
68 ExceptionOr<void> setOrientation(float x, float y, float z, float upX, float upY, float upZ);
69 FloatPoint3D orientation() const;
70
71 FloatPoint3D upVector() const;
72
73 bool hasSampleAccurateValues() const;
74 bool shouldUseARate() const;
75
76 const float* positionXValues(size_t framesToProcess);
77 const float* positionYValues(size_t framesToProcess);
78 const float* positionZValues(size_t framesToProcess);
79
80 const float* forwardXValues(size_t framesToProcess);
81 const float* forwardYValues(size_t framesToProcess);
82 const float* forwardZValues(size_t framesToProcess);
83
84 const float* upXValues(size_t framesToProcess);
85 const float* upYValues(size_t framesToProcess);
86 const float* upZValues(size_t framesToProcess);
87
88 void updateValuesIfNeeded(size_t framesToProcess);
89
90protected:
91 explicit AudioListener(BaseAudioContext&);
92
93private:
94
95 Ref<AudioParam> m_positionX;
96 Ref<AudioParam> m_positionY;
97 Ref<AudioParam> m_positionZ;
98 Ref<AudioParam> m_forwardX;
99 Ref<AudioParam> m_forwardY;
100 Ref<AudioParam> m_forwardZ;
101 Ref<AudioParam> m_upX;
102 Ref<AudioParam> m_upY;
103 Ref<AudioParam> m_upZ;
104
105 // Last time that the automations were updated.
106 double m_lastUpdateTime { -1 };
107
108 AudioFloatArray m_positionXValues;
109 AudioFloatArray m_positionYValues;
110 AudioFloatArray m_positionZValues;
111
112 AudioFloatArray m_forwardXValues;
113 AudioFloatArray m_forwardYValues;
114 AudioFloatArray m_forwardZValues;
115
116 AudioFloatArray m_upXValues;
117 AudioFloatArray m_upYValues;
118 AudioFloatArray m_upZValues;
119};
120
121} // namespace WebCore
