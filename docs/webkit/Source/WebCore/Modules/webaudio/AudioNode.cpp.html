101auto AudioNode::toWeakOrStrongContext(BaseAudioContext& context, NodeType nodeType) -> WeakOrStrongContext
102{
103 // Destination nodes are owned by the BaseAudioContext so we use WeakPtr to avoid a retain cycle.
104 if (nodeType == AudioNode::NodeTypeDestination)
105 return makeWeakPtr(context, EnableWeakPtrThreadingAssertions::No); // WebAudio code uses locking when accessing the context.
106 return makeRef(context);
107}

111 , m_context(toWeakOrStrongContext(context, type))
112#if !RELEASE_LOG_DISABLED

114 , m_logIdentifier(context.nextAudioNodeLogIdentifier())
115#endif

130}

181ExceptionOr<void> AudioNode::connect(AudioNode& destination, unsigned outputIndex, unsigned inputIndex)
182{
183 ASSERT(isMainThread());
184 Locker locker { context().graphLock() };
185
186 ALWAYS_LOG(LOGIDENTIFIER, destination.nodeType(), ", output = ", outputIndex, ", input = ", inputIndex);
187
188 // Sanity check input and output indices.
189 if (outputIndex >= numberOfOutputs())
190 return Exception { IndexSizeError, "Output index exceeds number of outputs"_s };
191
192 if (inputIndex >= destination.numberOfInputs())
193 return Exception { IndexSizeError, "Input index exceeds number of inputs"_s };
194
195 if (&context() != &destination.context())
196 return Exception { SyntaxError, "Source and destination nodes belong to different audio contexts"_s };
197
198 auto* input = destination.input(inputIndex);
199 auto* output = this->output(outputIndex);
200
201 if (!output->numberOfChannels())
202 return Exception { InvalidAccessError, "Node has zero output channels"_s };
203
204 input->connect(output);
205
206 updatePullStatus();
207
208 return { };
209}

262ExceptionOr<void> AudioNode::disconnect(AudioNode& destinationNode)
263{
264 ASSERT(isMainThread());
265 Locker locker { context().graphLock() };
266
267 bool didDisconnection = false;
268 for (unsigned outputIndex = 0; outputIndex < numberOfOutputs(); ++outputIndex) {
269 auto* output = this->output(outputIndex);
270 for (unsigned inputIndex = 0; inputIndex < destinationNode.numberOfInputs(); ++inputIndex) {
271 auto* input = destinationNode.input(inputIndex);
272 if (output->isConnectedTo(*input)) {
273 input->disconnect(output);
274 didDisconnection = true;
275 }
276 }
277 }
278
279 if (!didDisconnection)
280 return Exception { InvalidAccessError, "The given destination is not connected"_s };
281
282 updatePullStatus();
283 return { };
284}

377ExceptionOr<void> AudioNode::setChannelCount(unsigned channelCount)
378{
379 ASSERT(isMainThread());
380 Locker locker { context().graphLock() };
381
382 ALWAYS_LOG(LOGIDENTIFIER, channelCount);
383
384 if (!channelCount)
385 return Exception { NotSupportedError, "Channel count cannot be 0"_s };
386
387 if (channelCount > AudioContext::maxNumberOfChannels)
388 return Exception { IndexSizeError, "Channel count exceeds maximum limit"_s };
389
390 if (m_channelCount == channelCount)
391 return { };
392
393 m_channelCount = channelCount;
394 if (m_channelCountMode != ChannelCountMode::Max)
395 updateChannelsForInputs();
396 return { };
397}

450void AudioNode::processIfNecessary(size_t framesToProcess)
451{
452 ASSERT(context().isAudioThread());
453
454 if (!isInitialized())
455 return;
456
457 // Ensure that we only process once per rendering quantum.
458 // This handles the "fanout" problem where an output is connected to multiple inputs.
459 // The first time we're called during this time slice we process, but after that we don't want to re-process,
460 // instead our output(s) will already have the results cached in their bus;
461 double currentTime = context().currentTime();
462 if (m_lastProcessingTime != currentTime) {
463 m_lastProcessingTime = currentTime; // important to first update this time because of feedback loops in the rendering graph
464
465 pullInputs(framesToProcess);
466
467 bool silentInputs = inputsAreSilent();
468 if (!silentInputs)
469 m_lastNonSilentTime = (context().currentSampleFrame() + framesToProcess) / static_cast<double>(context().sampleRate());
470
471 if (silentInputs && propagatesSilence()) {
472 silenceOutputs();
473 // AudioParams still need to be processed so that the value can be updated
474 // if there are automations or so that the upstream nodes get pulled if
475 // any are connected to the AudioParam.
476 processOnlyAudioParams(framesToProcess);
477 } else
478 process(framesToProcess);
479 }
480}

519void AudioNode::enableOutputsIfNecessary()
520{
521 Locker locker { context().graphLock() };
522 if (isTailProcessing())
523 context().removeTailProcessingNode(*this);
524
525 if (m_isDisabled && m_connectionRefCount > 0) {
526 ASSERT(isMainThread());
527
528 m_isDisabled = false;
529 for (auto& output : m_outputs)
530 output->enable();
531 }
532}
533
534void AudioNode::disableOutputsIfNecessary()
535{
536 // Disable outputs if appropriate. We do this if the number of connections is 0 or 1. The case
537 // of 0 is from decrementConnectionCountWithLock() where there are no connections left. The case of 1 is from
538 // AudioNodeInput::disable() where we want to disable outputs when there's only one connection
539 // left because we're ready to go away, but can't quite yet.
540 if (m_connectionRefCount <= 1 && !m_isDisabled) {
541 // Still may have JavaScript references, but no more "active" connection references, so put all of our outputs in a "dormant" disabled state.
542 // Garbage collection may take a very long time after this time, so the "dormant" disabled nodes should not bog down the rendering...
543
544 // As far as JavaScript is concerned, our outputs must still appear to be connected.
545 // But internally our outputs should be disabled from the inputs they're connected to.
546 // disable() can recursively deref connections (and call disable()) down a whole chain of connected nodes.
547
548 // If a node requires tail processing, we defer the disabling of the outputs so that the tail for the node can be output.
549 // Otherwise, we can disable the outputs right away.
550 if (requiresTailProcessing())
551 context().addTailProcessingNode(*this);
552 else
553 disableOutputs();
554 }
555}

564void AudioNode::incrementConnectionCount()
565{
566 ++m_connectionRefCount;
567
568 // See the disabling code in decrementConnectionCountWithLock() below. This handles the case where a node
569 // is being re-connected after being used at least once and disconnected.
570 // In this case, we need to re-enable.
571 enableOutputsIfNecessary();
572
573#if DEBUG_AUDIONODE_REFERENCES
574 fprintf(stderr, "%p: %d: AudioNode::incrementConnectionCount() %d %d\n", this, nodeType(), m_normalRefCount, m_connectionRefCount);
575#endif
576}
577
578void AudioNode::decrementConnectionCount()
579{
580 // The actually work for deref happens completely within the audio context's graph lock.
581 // In the case of the audio thread, we must use a tryLock to avoid glitches.
582 if (auto locker = context().isAudioThread() ? Locker<RecursiveLock>::tryLock(context().graphLock()) : Locker { context().graphLock() }) {
583 // This is where the real deref work happens.
584 decrementConnectionCountWithLock();
585 } else {
586 // We were unable to get the lock, so put this in a list to finish up later.
587 ASSERT(context().isAudioThread());
588 context().addDeferredDecrementConnectionCount(this);
589 }
590
591 // Once AudioContext::uninitialize() is called there's no more chances for deleteMarkedNodes() to get called, so we call here.
592 // We can't call in AudioContext::~AudioContext() since it will never be called as long as any AudioNode is alive
593 // because AudioNodes keep a reference to the context.
594 if (context().isAudioThreadFinished())
595 context().deleteMarkedNodes();
596}
597
598void AudioNode::decrementConnectionCountWithLock()
599{
600 ASSERT(context().isGraphOwner());
601
602 ASSERT(m_connectionRefCount > 0);
603 --m_connectionRefCount;
604
605#if DEBUG_AUDIONODE_REFERENCES
606 fprintf(stderr, "%p: %d: AudioNode::decrementConnectionCountWithLock() %d %d\n", this, nodeType(), m_normalRefCount, m_connectionRefCount);
607#endif
608
609 if (!m_connectionRefCount && m_normalRefCount)
610 disableOutputsIfNecessary();
611
612 markNodeForDeletionIfNecessary();
613}
614
615void AudioNode::markNodeForDeletionIfNecessary()
616{
617 ASSERT(context().isGraphOwner());
618
619 if (m_connectionRefCount || m_normalRefCount || m_isMarkedForDeletion)
620 return;
621
622 // AudioDestinationNodes are owned by their BaseAudioContext so there is no need to mark them for deletion.
623 if (nodeType() == NodeTypeDestination)
624 return;
625
626 // All references are gone - we need to go away.
627 for (auto& output : m_outputs)
628 output->disconnectAll(); // This will deref() nodes we're connected to.
629
630 // Mark for deletion at end of each render quantum or when context shuts down.
631 context().markForDeletion(*this);
632 m_isMarkedForDeletion = true;
633}

644void AudioNode::deref()
645{
646 ASSERT(!context().isAudioThread());
647
648 {
649 Locker locker { context().graphLock() };
650 // This is where the real deref work happens.
651 derefWithLock();
652 }
653
654 // Once AudioContext::uninitialize() is called there's no more chances for deleteMarkedNodes() to get called, so we call here.
655 // We can't call in AudioContext::~AudioContext() since it will never be called as long as any AudioNode is alive
656 // because AudioNodes keep a reference to the context.
657 if (context().isAudioThreadFinished())
658 context().deleteMarkedNodes();
659}
660
661void AudioNode::derefWithLock()
662{
663 ASSERT(context().isGraphOwner());
664
665 ASSERT(m_normalRefCount > 0);
666 --m_normalRefCount;
667
668#if DEBUG_AUDIONODE_REFERENCES
669 fprintf(stderr, "%p: %d: AudioNode::deref() %d %d\n", this, nodeType(), m_normalRefCount, m_connectionRefCount);
670#endif
671
672 markNodeForDeletionIfNecessary();
673}

675ExceptionOr<void> AudioNode::handleAudioNodeOptions(const AudioNodeOptions& options, const DefaultAudioNodeOptions& defaults)
676{
677 auto result = setChannelCount(options.channelCount.value_or(defaults.channelCount));
678 if (result.hasException())
679 return result.releaseException();
680
681 result = setChannelCountMode(options.channelCountMode.value_or(defaults.channelCountMode));
682 if (result.hasException())
683 return result.releaseException();
684
685 result = setChannelInterpretation(options.channelInterpretation.value_or(defaults.channelInterpretation));
686 if (result.hasException())
687 return result.releaseException();
688
689 return { };
690}
