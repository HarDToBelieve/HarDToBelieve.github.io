46// An AudioNode is the basic building block for handling audio within an AudioContext.
47// It may be an audio source, an intermediate processing module, or an audio destination.
48// Each AudioNode can have inputs and/or outputs. An AudioSourceNode has no inputs and a single output.
49// An AudioDestinationNode has one input and no outputs and represents the final destination to the audio hardware.
50// Most processing nodes such as filters will have one input and one output, although multiple inputs and outputs are possible.
51
52class AudioNode
53 : public EventTargetWithInlineData
54#if !RELEASE_LOG_DISABLED
55 , private LoggerHelper
56#endif
57{
58 WTF_MAKE_NONCOPYABLE(AudioNode);
59 WTF_MAKE_ISO_ALLOCATED(AudioNode);
60public:
61 enum NodeType {
62 NodeTypeDestination,
63 NodeTypeOscillator,
64 NodeTypeAudioBufferSource,
65 NodeTypeMediaElementAudioSource,
66 NodeTypeMediaStreamAudioDestination,
67 NodeTypeMediaStreamAudioSource,
68 NodeTypeJavaScript,
69 NodeTypeBiquadFilter,
70 NodeTypePanner,
71 NodeTypeConvolver,
72 NodeTypeDelay,
73 NodeTypeGain,
74 NodeTypeChannelSplitter,
75 NodeTypeChannelMerger,
76 NodeTypeAnalyser,
77 NodeTypeDynamicsCompressor,
78 NodeTypeWaveShaper,
79 NodeTypeConstant,
80 NodeTypeStereoPanner,
81 NodeTypeIIRFilter,
82 NodeTypeWorklet,
83 NodeTypeLast = NodeTypeWorklet
84 };
85
86 AudioNode(BaseAudioContext&, NodeType);
87 virtual ~AudioNode();
88
89 BaseAudioContext& context();
90 const BaseAudioContext& context() const;
91
92 NodeType nodeType() const { return m_nodeType; }
93
94 // Can be called from main thread or context's audio thread.
95 virtual void ref();
96 virtual void deref();
97 void incrementConnectionCount();
98 void decrementConnectionCount();
99
100 // Can be called from main thread or context's audio thread. It must be called while the context's graph lock is held.
101 void decrementConnectionCountWithLock();
102
103 // The AudioNodeInput(s) (if any) will already have their input data available when process() is called.
104 // Subclasses will take this input data and put the results in the AudioBus(s) of its AudioNodeOutput(s) (if any).
105 // Called from context's audio thread.
106 virtual void process(size_t framesToProcess) = 0;
107
108 // Like process(), but only causes the automations to process; the
109 // normal processing of the node is bypassed. By default, we assume
110 // no AudioParams need to be updated.
111 virtual void processOnlyAudioParams(size_t) { }
112
113 // No significant resources should be allocated until initialize() is called.
114 // Processing may not occur until a node is initialized.
115 virtual void initialize();
116 virtual void uninitialize();
117
118 bool isInitialized() const { return m_isInitialized; }
119
120 unsigned numberOfInputs() const { return m_inputs.size(); }
121 unsigned numberOfOutputs() const { return m_outputs.size(); }
122
123 AudioNodeInput* input(unsigned);
124 AudioNodeOutput* output(unsigned);
125
126 // Called from main thread by corresponding JavaScript methods.
127 ExceptionOr<void> connect(AudioNode&, unsigned outputIndex, unsigned inputIndex);
128 ExceptionOr<void> connect(AudioParam&, unsigned outputIndex);
129
130 void disconnect();
131 ExceptionOr<void> disconnect(unsigned output);
132 ExceptionOr<void> disconnect(AudioNode& destinationNode);
133 ExceptionOr<void> disconnect(AudioNode& destinationNode, unsigned output);
134 ExceptionOr<void> disconnect(AudioNode& destinationNode, unsigned output, unsigned input);
135 ExceptionOr<void> disconnect(AudioParam& destinationParam);
136 ExceptionOr<void> disconnect(AudioParam& destinationParam, unsigned output);
137
138 virtual float sampleRate() const;
139
140 // processIfNecessary() is called by our output(s) when the rendering graph needs this AudioNode to process.
141 // This method ensures that the AudioNode will only process once per rendering time quantum even if it's called repeatedly.
142 // This handles the case of "fanout" where an output is connected to multiple AudioNode inputs.
143 // Called from context's audio thread.
144 void processIfNecessary(size_t framesToProcess);
145
146 // Called when a new connection has been made to one of our inputs or the connection number of channels has changed.
147 // This potentially gives us enough information to perform a lazy initialization or, if necessary, a re-initialization.
148 // Called from main thread.
149 virtual void checkNumberOfChannelsForInput(AudioNodeInput*);
150
151#if DEBUG_AUDIONODE_REFERENCES
152 static void printNodeCounts();
153#endif
154
155 bool isMarkedForDeletion() const { return m_isMarkedForDeletion; }
156
157 // tailTime() is the length of time (not counting latency time) where non-zero output may occur after continuous silent input.
158 virtual double tailTime() const = 0;
159 // latencyTime() is the length of time it takes for non-zero output to appear after non-zero input is provided. This only applies to
160 // processing delay which is an artifact of the processing algorithm chosen and is *not* part of the intrinsic desired effect. For
161 // example, a "delay" effect is expected to delay the signal, and thus would not be considered latency.
162 virtual double latencyTime() const = 0;
163
164 // True if the node has a tail time or latency time that requires
165 // special tail processing to behave properly. Ideally, this can be
166 // checked using tailTime and latencyTime, but these aren't
167 // available on the main thread, and the tail processing check can
168 // happen on the main thread.
169 virtual bool requiresTailProcessing() const = 0;
170
171 // propagatesSilence() should return true if the node will generate silent output when given silent input. By default, AudioNode
172 // will take tailTime() and latencyTime() into account when determining whether the node will propagate silence.
173 virtual bool propagatesSilence() const;
174 bool inputsAreSilent();
175 void silenceOutputs();
176
177 void enableOutputsIfNecessary();
178 void disableOutputsIfNecessary();
179 void disableOutputs();
180
181 unsigned channelCount() const { return m_channelCount; }
182 virtual ExceptionOr<void> setChannelCount(unsigned);
183
184 ChannelCountMode channelCountMode() const { return m_channelCountMode; }
185 virtual ExceptionOr<void> setChannelCountMode(ChannelCountMode);
186
187 ChannelInterpretation channelInterpretation() const { return m_channelInterpretation; }
188 virtual ExceptionOr<void> setChannelInterpretation(ChannelInterpretation);
189
190 bool isFinishedSourceNode() const { return m_isFinishedSourceNode; }
191 void setIsFinishedSourceNode() { m_isFinishedSourceNode = true; }
192
193 // Flag indicating the node is in the context's m_tailProcessingNodes or m_finishTailProcessingNodes.
194 // We rely on this flag to avoid unnecessary linear searches in those vectors.
195 bool isTailProcessing() const { return m_isTailProcessing; }
196 void setIsTailProcessing(bool isTailProcessing) { m_isTailProcessing = isTailProcessing; }
197
198protected:
199 // Inputs and outputs must be created before the AudioNode is initialized.
200 void addInput();
201 void addOutput(unsigned numberOfChannels);
202
203 void markNodeForDeletionIfNecessary();
204 void derefWithLock();
205
206 struct DefaultAudioNodeOptions {
207 unsigned channelCount;
208 ChannelCountMode channelCountMode;
209 ChannelInterpretation channelInterpretation;
210 };
211
212 ExceptionOr<void> handleAudioNodeOptions(const AudioNodeOptions&, const DefaultAudioNodeOptions&);
213
214 // Called by processIfNecessary() to cause all parts of the rendering graph connected to us to process.
215 // Each rendering quantum, the audio data for each of the AudioNode's inputs will be available after this method is called.
216 // Called from context's audio thread.
217 virtual void pullInputs(size_t framesToProcess);
218
219 // Force all inputs to take any channel interpretation changes into account.
220 void updateChannelsForInputs();
221
222#if !RELEASE_LOG_DISABLED
223 const Logger& logger() const final { return m_logger.get(); }
224 const void* logIdentifier() const final { return m_logIdentifier; }
225 const char* logClassName() const final { return "AudioNode"; }
226 WTFLogChannel& logChannel() const final;
227#endif
228
229 void initializeDefaultNodeOptions(unsigned count, ChannelCountMode, ChannelInterpretation);
230
231 virtual void updatePullStatus() { }
232
233private:
234 using WeakOrStrongContext = Variant<Ref<BaseAudioContext>, WeakPtr<BaseAudioContext>>;
235 static WeakOrStrongContext toWeakOrStrongContext(BaseAudioContext&, NodeType);
236
237 // EventTarget
238 EventTargetInterface eventTargetInterface() const override;
239 ScriptExecutionContext* scriptExecutionContext() const final;

241 volatile bool m_isInitialized { false };
242 NodeType m_nodeType;
243
244 WeakOrStrongContext m_context;
245
246 Vector<std::unique_ptr<AudioNodeInput>> m_inputs;
247 Vector<std::unique_ptr<AudioNodeOutput>> m_outputs;
248
249 double m_lastProcessingTime { -1 };
250 double m_lastNonSilentTime { -1 };
251
252 // Ref-counting
253 // start out with normal refCount == 1 (like WTF::RefCounted class).
254 std::atomic<int> m_normalRefCount { 1 };
255 std::atomic<int> m_connectionRefCount { 0 };
256
257 bool m_isMarkedForDeletion { false };
258 bool m_isDisabled { false };
259 bool m_isFinishedSourceNode { false };
260 bool m_isTailProcessing { false };
261
262#if DEBUG_AUDIONODE_REFERENCES
263 static bool s_isNodeCountInitialized;
264 static int s_nodeCount[NodeTypeLast + 1];
265#endif
266
267 void refEventTarget() override { ref(); }
268 void derefEventTarget() override { deref(); }
269
270#if !RELEASE_LOG_DISABLED
271 mutable Ref<const Logger> m_logger;
272 const void* m_logIdentifier;
273#endif
274
275 unsigned m_channelCount { 2 };
276 ChannelCountMode m_channelCountMode { ChannelCountMode::Max };
277 ChannelInterpretation m_channelInterpretation { ChannelInterpretation::Speakers };
278};

280template<typename T> struct AudioNodeConnectionRefDerefTraits {
281 static ALWAYS_INLINE void refIfNotNull(T* ptr)
282 {
283 if (LIKELY(ptr != nullptr))
284 ptr->incrementConnectionCount();
285 }
286
287 static ALWAYS_INLINE void derefIfNotNull(T* ptr)
288 {
289 if (LIKELY(ptr != nullptr))
290 ptr->decrementConnectionCount();
291 }
292};
293
294template<typename T>
295using AudioConnectionRefPtr = RefPtr<T, RawPtrTraits<T>, AudioNodeConnectionRefDerefTraits<T>>;
