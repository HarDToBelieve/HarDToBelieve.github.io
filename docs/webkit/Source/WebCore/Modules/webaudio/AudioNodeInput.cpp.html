40AudioNodeInput::AudioNodeInput(AudioNode* node)
41 : AudioSummingJunction(node->context())
42 , m_node(node)
43{
44 // Set to mono by default.
45 m_internalSummingBus = AudioBus::create(1, AudioUtilities::renderQuantumSize);
46}

48void AudioNodeInput::connect(AudioNodeOutput* output)
49{
50 ASSERT(context());
51 ASSERT(context()->isGraphOwner());
52
53 ASSERT(output && node());
54 if (!output || !node())
55 return;
56
57 auto addPotentiallyDisabledOutput = [this](AudioNodeOutput& output) {
58 if (output.isEnabled())
59 return addOutput(output);
60 return m_disabledOutputs.add(&output).isNewEntry;
61 };
62
63 if (addPotentiallyDisabledOutput(*output))
64 output->addInput(this);
65}
66
67void AudioNodeInput::disconnect(AudioNodeOutput* output)
68{
69 ASSERT(context());
70 ASSERT(context()->isGraphOwner());
71
72 ASSERT(output && node());
73 if (!output || !node())
74 return;
75
76 // First try to disconnect from "active" connections.
77 if (removeOutput(*output)) {
78 output->removeInput(this); // Note: it's important to return immediately after this since the node may be deleted.
79 return;
80 }
81
82 // Otherwise, try to disconnect from disabled connections.
83 if (m_disabledOutputs.remove(output)) {
84 output->removeInput(this); // Note: it's important to return immediately after this since the node may be deleted.
85 return;
86 }
87
88 ASSERT_NOT_REACHED();
89}

91void AudioNodeInput::disable(AudioNodeOutput* output)
92{
93 ASSERT(context());
94 ASSERT(context()->isGraphOwner());
95
96 ASSERT(output && node());
97 if (!output || !node())
98 return;
99
100 {
101 // Heap allocations are forbidden on the audio thread for performance reasons so we need to
102 // explicitly allow the following allocation(s).
103 DisableMallocRestrictionsForCurrentThreadScope disableMallocRestrictions;
104 m_disabledOutputs.add(output);
105 bool wasRemoved = removeOutput(*output);
106 ASSERT_UNUSED(wasRemoved, wasRemoved);
107 }
108
109 // Propagate disabled state to outputs.
110 node()->disableOutputsIfNecessary();
111}
112
113void AudioNodeInput::enable(AudioNodeOutput* output)
114{
115 ASSERT(context());
116 ASSERT(context()->isGraphOwner());
117
118 ASSERT(output && node());
119 if (!output || !node())
120 return;
121
122 ASSERT(m_disabledOutputs.contains(output));
123
124 // Move output from disabled list to active list.
125 {
126 // Heap allocations are forbidden on the audio thread for performance reasons so we need to
127 // explicitly allow the following allocation(s).
128 DisableMallocRestrictionsForCurrentThreadScope disableMallocRestrictions;
129 addOutput(*output);
130 }
131 m_disabledOutputs.remove(output);
132
133 // Propagate enabled state to outputs.
134 node()->enableOutputsIfNecessary();
135}

142void AudioNodeInput::updateInternalBus()
143{
144 ASSERT(context());
145 ASSERT(context()->isAudioThread() && context()->isGraphOwner());
146
147 unsigned numberOfInputChannels = numberOfChannels();
148
149 if (numberOfInputChannels == m_internalSummingBus->numberOfChannels())
150 return;
151
152 // Heap allocations are forbidden on the audio thread for performance reasons so we need to
153 // explicitly allow the following allocation(s).
154 DisableMallocRestrictionsForCurrentThreadScope disableMallocRestrictions;
155 m_internalSummingBus = AudioBus::create(numberOfInputChannels, AudioUtilities::renderQuantumSize);
156}

158unsigned AudioNodeInput::numberOfChannels() const
159{
160 auto mode = node()->channelCountMode();
161 if (mode == ChannelCountMode::Explicit)
162 return node()->channelCount();
163
164 // Find the number of channels of the connection with the largest number of channels.
165 unsigned maxChannels = std::max(maximumNumberOfChannels(), 1u); // One channel is the minimum allowed.
166
167 if (mode == ChannelCountMode::ClampedMax)
168 maxChannels = std::min(maxChannels, static_cast<unsigned>(node()->channelCount()));
169
170 return maxChannels;
171}

173AudioBus* AudioNodeInput::bus()
174{
175 ASSERT(context());
176 ASSERT(context()->isAudioThread());
177
178 // Handle single connection specially to allow for in-place processing.
179 if (numberOfRenderingConnections() == 1 && node()->channelCountMode() == ChannelCountMode::Max)
180 return renderingOutput(0)->bus();
181
182 // Multiple connections case or complex ChannelCountMode (or no connections).
183 return internalSummingBus();
184}

186AudioBus* AudioNodeInput::internalSummingBus()
187{
188 ASSERT(context());
189 ASSERT(context()->isAudioThread());
190
191 return m_internalSummingBus.get();
192}

194void AudioNodeInput::sumAllConnections(AudioBus* summingBus, size_t framesToProcess)
195{
196 ASSERT(context());
197 ASSERT(context()->isAudioThread());
198
199 // We shouldn't be calling this method if there's only one connection, since it's less efficient.
200 ASSERT(numberOfRenderingConnections() > 1 || node()->channelCountMode() != ChannelCountMode::Max);
201
202 ASSERT(summingBus);
203 if (!summingBus)
204 return;
205
206 summingBus->zero();
207
208 auto interpretation = node()->channelInterpretation();
209
210 for (auto& output : m_renderingOutputs) {
211 ASSERT(output);
212
213 // Render audio from this output.
214 AudioBus* connectionBus = output->pull(0, framesToProcess);
215
216 // Sum, with unity-gain.
217 summingBus->sumFrom(*connectionBus, interpretation);
218 }
219}
220
221AudioBus* AudioNodeInput::pull(AudioBus* inPlaceBus, size_t framesToProcess)
222{
223 ASSERT(context());
224 ASSERT(context()->isAudioThread());
225
226 // Handle single connection case.
227 if (numberOfRenderingConnections() == 1 && node()->channelCountMode() == ChannelCountMode::Max) {
228 // The output will optimize processing using inPlaceBus if it's able.
229 AudioNodeOutput* output = this->renderingOutput(0);
230 return output->pull(inPlaceBus, framesToProcess);
231 }
232
233 AudioBus* internalSummingBus = this->internalSummingBus();
234
235 if (!numberOfRenderingConnections()) {
236 // At least, generate silence if we're not connected to anything.
237 // FIXME: if we wanted to get fancy, we could propagate a 'silent hint' here to optimize the downstream graph processing.
238 internalSummingBus->zero();
239 return internalSummingBus;
240 }
241
242 // Handle multiple connections case.
243 sumAllConnections(internalSummingBus, framesToProcess);
244
245 return internalSummingBus;
246}
