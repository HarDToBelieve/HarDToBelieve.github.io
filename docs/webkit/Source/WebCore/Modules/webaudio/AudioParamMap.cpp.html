40AudioParamMap::AudioParamMap() = default;
41

44void AudioParamMap::initializeMapLike(DOMMapAdapter& map)
45{
46 for (auto& keyValue : m_map)
47 map.set<IDLDOMString, IDLInterface<AudioParam>>(keyValue.key, keyValue.value);
48}
49
50void AudioParamMap::add(const String& name, Ref<AudioParam>&& param)
51{
52 m_map.add(name, WTFMove(param));
53}
