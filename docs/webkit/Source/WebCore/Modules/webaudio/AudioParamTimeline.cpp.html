40static void fillWithValue(float* values, float value, unsigned endFrame, unsigned& writeIndex)
41{
42 if (writeIndex < endFrame) {
43 std::fill_n(values + writeIndex, endFrame - writeIndex, value);
44 writeIndex = endFrame;
45 }
46}

48// Test that for a SetTarget event, the current value is close enough to the target value that
49// we can consider the event to have converged to the target.
50static bool hasSetTargetConverged(float value, float target, Seconds currentTime, Seconds startTime, double timeConstant)
51{
52 // For a SetTarget event, we want the event to terminate eventually so that we can stop using
53 // the timeline to compute the values.
54 constexpr float timeConstantsToConverge = 10;
55 constexpr float setTargetThreshold = 4.539992976248485e-05;
56
57 // Converged if enough time constants (|timeConstantsToConverge|) have passed since the start
58 // of the event.
59 if (currentTime.value() > startTime.value() + timeConstantsToConverge * timeConstant)
60 return true;
61
62 // If |target| is 0, converged if |value| is less than |setTargetThreshold|.
63 if (!target && std::abs(value) < setTargetThreshold)
64 return true;
65
66 // If |target| is not zero, converged if relative difference between |value|
67 // and |target| is small. That is |target - value| / |target| < |setTargetThreshold|.
68 if (target && std::abs(target - value) < setTargetThreshold * std::abs(value))
69 return true;
70
71 return false;
72}

111ExceptionOr<void> AudioParamTimeline::setValueCurveAtTime(Vector<float>&& curve, Seconds time, Seconds duration)
112{
113 Locker locker { m_eventsLock };
114
115 float curveEndValue = curve.last();
116 auto result = insertEvent(ParamEvent::createSetValueCurveEvent(WTFMove(curve), time, duration));
117 if (result.hasException())
118 return result.releaseException();
119
120 // The specification says an implicit call to setValueAtTime() is made at time T0+TD with value V[N-1]
121 // so that following automations will start from the end of the setValueCurveAtTime() event.
122 return insertEvent(ParamEvent::createSetValueEvent(curveEndValue, time + duration));
123}

135ExceptionOr<void> AudioParamTimeline::insertEvent(ParamEvent&& event)
136{
137 // Sanity check the event. Be super careful we're not getting infected with NaN or Inf.
138 bool isValid = event.type() < ParamEvent::LastType
139 && isValidNumber(event.value())
140 && isValidNumber(event.time())
141 && isValidNumber(event.timeConstant())
142 && isValidNumber(event.duration())
143 && event.duration() >= 0_s;
144
145 if (!isValid)
146 return { };
147
148 ASSERT(m_eventsLock.isLocked());
149
150 unsigned i = 0;
151 auto insertTime = event.time();
152
153 for (auto& paramEvent : m_events) {
154 if (event.type() == ParamEvent::SetValueCurve) {
155 if (paramEvent.type() != ParamEvent::CancelValues) {
156 // If this event is a SetValueCurve, make sure it doesn't overlap any existing event.
157 // It's ok if the SetValueCurve starts at the same time as the end of some other duration.
158 auto endTime = event.time() + event.duration();
159 if (paramEvent.type() == ParamEvent::SetValueCurve) {
160 auto paramEventEndTime = paramEvent.time() + paramEvent.duration();
161 if ((paramEvent.time() >= event.time() && paramEvent.time() < endTime)
162 || (paramEventEndTime > event.time() && paramEventEndTime < endTime)
163 || (event.time() >= paramEvent.time() && event.time() < paramEventEndTime)
164 || (endTime >= paramEvent.time() && endTime < paramEventEndTime)) {
165 return Exception { NotSupportedError, "Events are overlapping"_s };
166 }
167 } else if (paramEvent.time() > event.time() && paramEvent.time() < endTime)
168 return Exception { NotSupportedError, "Events are overlapping"_s };
169 }
170 } else if (paramEvent.type() == ParamEvent::SetValueCurve) {
171 // Otherwise, make sure this event doesn't overlap any existing SetValueCurve event.
172 auto parentEventEndTime = paramEvent.time() + paramEvent.duration();
173 if (event.time() >= paramEvent.time() && event.time() < parentEventEndTime)
174 return Exception { NotSupportedError, "Events are overlapping" };
175 }
176
177 if (paramEvent.time() > insertTime)
178 break;
179
180 ++i;
181 }
182
183 m_events.insert(i, WTFMove(event));
184 return { };
185}

187void AudioParamTimeline::cancelScheduledValues(Seconds cancelTime)
188{
189 Locker locker { m_eventsLock };
190
191 // Remove all events whose start time is greater than or equal to the cancel time.
192 // Also handle the special case where the cancel time lies in the middle of a
193 // SetValueCurve event.
194 //
195 // This critically depends on the fact that no event can be scheduled in the middle
196 // of the curve or at the same start time. Then removing the SetValueCurve doesn't
197 // remove any events that shouldn't have been.
198 auto isAfter = [](const ParamEvent& event, Seconds cancelTime) {
199 auto eventTime = event.time();
200 if (eventTime >= cancelTime)
201 return true;
202 return event.type() == ParamEvent::SetValueCurve
203 && eventTime <= cancelTime
204 && (eventTime + event.duration() > cancelTime);
205 };
206
207 // Remove all events starting at cancelTime.
208 for (unsigned i = 0; i < m_events.size(); ++i) {
209 if (isAfter(m_events[i], cancelTime)) {
210 m_events.remove(i, m_events.size() - i);
211 break;
212 }
213 }
214}

216ExceptionOr<void> AudioParamTimeline::cancelAndHoldAtTime(Seconds cancelTime)
217{
218 Locker locker { m_eventsLock };
219
220 // Find the first event at or just past cancelTime.
221 size_t i = m_events.findMatching([&](auto& event) {
222 return event.time() > cancelTime;
223 });
224 i = (i == notFound) ? m_events.size() : i;
225
226 // The event that is being cancelled. This is the event just past cancelTime, if any.
227 size_t cancelledEventIndex = i;
228
229 // If the event just before cancelTime is a SetTarget or SetValueCurve event, we need
230 // to handle that event specially instead of the event after.
231 if (i > 0 && ((m_events[i - 1].type() == ParamEvent::SetTarget) || (m_events[i - 1].type() == ParamEvent::SetValueCurve)))
232 cancelledEventIndex = i - 1;
233 else if (i >= m_events.size()) {
234 // If there were no events occurring after |cancelTime| (and the
235 // previous event is not SetTarget or SetValueCurve, we're done.
236 return { };
237 }
238
239 // cancelledEvent is the event that is being cancelled.
240 auto& cancelledEvent = m_events[cancelledEventIndex];
241 auto eventType = cancelledEvent.type();
242
243 // New event to be inserted, if any, and a SetValueEvent if needed.
244 std::optional<ParamEvent> newEvent;
245 std::optional<ParamEvent> newSetValueEvent;
246
247 switch (eventType) {
248 case ParamEvent::LinearRampToValue:
249 case ParamEvent::ExponentialRampToValue: {
250 // For these events we need to remember the parameters of the event
251 // for a CancelValues event so that we can properly cancel the event
252 // and hold the value.
253 auto savedEvent = ParamEvent::SavedEvent { eventType, cancelledEvent.value(), cancelledEvent.time() };
254 newEvent = ParamEvent::createCancelValuesEvent(cancelTime, WTFMove(savedEvent));
255 break;
256 }
257 case ParamEvent::SetTarget: {
258 if (cancelledEvent.time() < cancelTime) {
259 // Don't want to remove the SetTarget event if it started before the
260 // cancel time, so bump the index. But we do want to insert a
261 // cancelEvent so that we stop this automation and hold the value when
262 // we get there.
263 ++cancelledEventIndex;
264
265 newEvent = ParamEvent::createCancelValuesEvent(cancelTime, std::nullopt);
266 }
267 break;
268 }
269 case ParamEvent::SetValueCurve: {
270 // If the setValueCurve event started strictly before the cancel time,
271 // there might be something to do....
272 if (cancelledEvent.time() < cancelTime) {
273 if (cancelTime > cancelledEvent.time() + cancelledEvent.duration()) {
274 // If the cancellation time is past the end of the curve there's
275 // nothing to do except remove the following events.
276 ++cancelledEventIndex;
277 } else {
278 // Cancellation time is in the middle of the curve. Therefore,
279 // create a new SetValueCurve event with the appropriate new
280 // parameters to cancel this event properly. Since it's illegal
281 // to insert any event within a SetValueCurve event, we can
282 // compute the new end value now instead of doing when running
283 // the timeline.
284 auto newDuration = cancelTime - cancelledEvent.time();
285 float endValue = valueCurveAtTime(cancelTime, cancelledEvent.time(), cancelledEvent.duration(), cancelledEvent.curve().data(), cancelledEvent.curve().size());
286
287 // Replace the existing SetValueCurve with this new one that is identical except for the duration.
288 newEvent = ParamEvent { eventType, cancelledEvent.value(), cancelledEvent.time(), cancelledEvent.timeConstant(), newDuration, Vector<float> { cancelledEvent.curve() }, cancelledEvent.curvePointsPerSecond(), endValue, std::nullopt };
289 newSetValueEvent = ParamEvent::createSetValueEvent(endValue, cancelledEvent.time() + newDuration);
290 }
291 }
292 break;
293 }
294 case ParamEvent::SetValue:
295 case ParamEvent::CancelValues:
296 // Nothing needs to be done for a SetValue or CancelValues event.
297 break;
298 case ParamEvent::LastType:
299 ASSERT_NOT_REACHED();
300 break;
301 }
302
303 // Now remove all the following events from the timeline.
304 if (cancelledEventIndex < m_events.size())
305 removeCancelledEvents(cancelledEventIndex);
306
307 // Insert the new event, if any.
308 if (newEvent) {
309 auto result = insertEvent(WTFMove(*newEvent));
310 if (result.hasException())
311 return result.releaseException();
312 if (newSetValueEvent) {
313 insertEvent(WTFMove(*newSetValueEvent));
314 if (result.hasException())
315 return result.releaseException();
316 }
317 }
318
319 return { };
320}

322void AudioParamTimeline::removeCancelledEvents(size_t firstEventToRemove)
323{
324 m_events.remove(firstEventToRemove, m_events.size() - firstEventToRemove);
325}

327std::optional<float> AudioParamTimeline::valueForContextTime(BaseAudioContext& context, float defaultValue, float minValue, float maxValue)
328{
329 {
330 if (!m_eventsLock.tryLock())
331 return std::nullopt;
332 Locker locker { AdoptLock, m_eventsLock };
333 if (!m_events.size() || Seconds { context.currentTime() } < m_events[0].time())
334 return std::nullopt;
335 }
336
337 // Ask for just a single value.
338 float value;
339 double sampleRate = context.sampleRate();
340 size_t startFrame = context.currentSampleFrame();
341 size_t endFrame = startFrame + 1;
342 double controlRate = sampleRate / AudioUtilities::renderQuantumSize; // one parameter change per render quantum
343 value = valuesForFrameRange(startFrame, endFrame, defaultValue, minValue, maxValue, &value, 1, sampleRate, controlRate);
344 return value;
345}

347float AudioParamTimeline::valuesForFrameRange(size_t startFrame, size_t endFrame, float defaultValue, float minValue, float maxValue, float* values, unsigned numberOfValues, double sampleRate, double controlRate)
348{
349 // We can't contend the lock in the realtime audio thread.
350 if (!m_eventsLock.tryLock()) {
351 std::fill_n(values, numberOfValues, defaultValue);
352 return defaultValue;
353 }
354 Locker locker { AdoptLock, m_eventsLock };
355
356 float value = valuesForFrameRangeImpl(startFrame, endFrame, defaultValue, values, numberOfValues, sampleRate, controlRate);
357
358 // Clamp values based on range allowed by AudioParam's min and max values.
359 VectorMath::clamp(values, minValue, maxValue, values, numberOfValues);
360
361 return value;
362}

364float AudioParamTimeline::valuesForFrameRangeImpl(size_t startFrame, size_t endFrame, float defaultValue, float* values, unsigned numberOfValues, double sampleRate, double controlRate)
365{
366 ASSERT(values);
367 if (!values)
368 return defaultValue;
369
370 double samplingPeriod = 1. / sampleRate;
371
372 // Return default value if there are no events matching the desired time range.
373 if (!m_events.size() || endFrame * samplingPeriod <= m_events[0].time().value()) {
374 std::fill_n(values, numberOfValues, defaultValue);
375 return defaultValue;
376 }
377
378 // Maintain a running time and index for writing the values buffer.
379 size_t currentFrame = startFrame;
380 unsigned writeIndex = 0;
381
382 // If first event is after startTime then fill initial part of values buffer with defaultValue
383 // until we reach the first event time.
384 auto firstEventTime = m_events[0].time();
385 if (firstEventTime.value() > startFrame * samplingPeriod) {
386 size_t fillToEndFrame = endFrame;
387 double firstEventFrame = ceil(firstEventTime.value() * sampleRate);
388 if (endFrame > firstEventFrame)
389 fillToEndFrame = firstEventFrame;
390 ASSERT(fillToEndFrame >= startFrame);
391
392 unsigned fillToFrame = static_cast<unsigned>(fillToEndFrame - startFrame);
393 fillToFrame = std::min(fillToFrame, numberOfValues);
394 fillWithValue(values, defaultValue, fillToFrame, writeIndex);
395
396 currentFrame += fillToFrame;
397 }
398
399 float value = defaultValue;
400
401 // Go through each event and render the value buffer where the times overlap,
402 // stopping when we've rendered all the requested values.
403 // FIXME: could try to optimize by avoiding having to iterate starting from the very first event
404 // and keeping track of a "current" event index.
405 int n = m_events.size();
406 for (int i = 0; i < n && writeIndex < numberOfValues; ++i) {
407 auto* event = &m_events[i];
408 auto* nextEvent = i < n - 1 ? &m_events[i + 1] : nullptr;
409
410 // Wait until we get a more recent event.
411 if (!isEventCurrent(*event, nextEvent, currentFrame, sampleRate))
412 continue;
413
414 auto nextEventType = nextEvent ? static_cast<ParamEvent::Type>(nextEvent->type()) : ParamEvent::LastType /* unknown */;
415
416 processSetTargetFollowedByRamp(i, event, nextEventType, currentFrame, sampleRate, controlRate, value);
417
418 float value1 = event->value();
419 auto time1 = event->time();
420 float value2 = nextEvent ? nextEvent->value() : value1;
421 auto time2 = nextEvent ? nextEvent->time() : Seconds { endFrame * samplingPeriod + 1 };
422
423 ASSERT(time2 >= time1);
424
425 handleCancelValues(*event, nextEvent, value2, time2, nextEventType);
426
427 size_t fillToEndFrame = endFrame;
428 if (endFrame > time2.value() * sampleRate)
429 fillToEndFrame = static_cast<size_t>(ceil(time2.value() * sampleRate));
430
431 ASSERT(fillToEndFrame >= startFrame);
432 unsigned fillToFrame = static_cast<unsigned>(fillToEndFrame - startFrame);
433 fillToFrame = std::min(fillToFrame, numberOfValues);
434
435 const AutomationState currentState = {
436 numberOfValues,
437 startFrame,
438 endFrame,
439 sampleRate,
440 controlRate,
441 samplingPeriod,
442 fillToFrame,
443 fillToEndFrame,
444 value1,
445 time1,
446 value2,
447 time2,
448 event,
449 i,
450 };
451
452 // First handle linear and exponential ramps which require looking ahead to the next event.
453 if (nextEventType == ParamEvent::LinearRampToValue)
454 processLinearRamp(currentState, values, currentFrame, value, writeIndex);
455 else if (nextEventType == ParamEvent::ExponentialRampToValue)
456 processExponentialRamp(currentState, values, currentFrame, value, writeIndex);
457 else {
458 // Handle event types not requiring looking ahead to the next event.
459 switch (event->type()) {
460 case ParamEvent::SetValue:
461 case ParamEvent::LinearRampToValue:
462 case ParamEvent::ExponentialRampToValue:
463 currentFrame = fillToEndFrame;
464
465 // Simply stay at a constant value.
466 value = event->value();
467 fillWithValue(values, value, fillToFrame, writeIndex);
468 break;
469 case ParamEvent::CancelValues:
470 processCancelValues(currentState, values, currentFrame, value, writeIndex);
471 break;
472 case ParamEvent::SetTarget:
473 processSetTarget(currentState, values, currentFrame, value, writeIndex);
474 break;
475 case ParamEvent::SetValueCurve:
476 processSetValueCurve(currentState, values, currentFrame, value, writeIndex);
477 break;
478 case ParamEvent::LastType:
479 ASSERT_NOT_REACHED();
480 break;
481 }
482 }
483 }
484
485 // If there's any time left after processing the last event then just propagate the last value
486 // to the end of the values buffer.
487 fillWithValue(values, value, numberOfValues, writeIndex);
488
489 return value;
490}

492void AudioParamTimeline::processLinearRamp(const AutomationState& currentState, float* values, size_t& currentFrame, float& value, unsigned& writeIndex)
493{
494 auto deltaTime = currentState.time2 - currentState.time1;
495 float valueDelta = currentState.value2 - currentState.value1;
496
497 // Since deltaTime is a double, 1/deltaTime can easily overflow a float. Thus, if deltaTime
498 // is close enough to zero (less than float min), treat it as zero.
499 float k = deltaTime.value() <= std::numeric_limits<float>::min() ? 0 : 1 / deltaTime.value();
500
501 unsigned fillToFrameTrunc = writeIndex + ((currentState.fillToFrame - writeIndex) / 4) * 4;
502 if (fillToFrameTrunc > writeIndex) {
503 // Minimize in-loop operations. Calculate starting value and increment.
504 // Next step: value += inc.
505 // value = value1 + (currentFrame/sampleRate - time1) * k * (value2 - value1);
506 // inc = 4 / sampleRate * k * (value2 - value1);
507 // Resolve recursion by expanding constants to achieve a 4-step loop unrolling.
508 // value = value1 + ((currentFrame/sampleRate - time1) + i * sampleFrameTimeIncr) * k * (value2 - value1), i in 0..3
509 values[writeIndex] = 0;
510 values[writeIndex + 1] = 1;
511 values[writeIndex + 2] = 2;
512 values[writeIndex + 3] = 3;
513 VectorMath::multiplyByScalar(values + writeIndex, currentState.samplingPeriod, values + writeIndex, 4);
514 VectorMath::addScalar(values + writeIndex, currentFrame * currentState.samplingPeriod - currentState.time1.value(), values + writeIndex, 4);
515 VectorMath::multiplyByScalar(values + writeIndex, k * valueDelta, values + writeIndex, 4);
516 VectorMath::addScalar(values + writeIndex, currentState.value1, values + writeIndex, 4);
517
518 float inc = 4 * currentState.samplingPeriod * k * valueDelta;
519
520 // Truncate loop steps to multiple of 4.
521 unsigned fillToFrameTrunc = writeIndex + ((currentState.fillToFrame - writeIndex) / 4) * 4;
522 // Compute final frame.
523 currentFrame += fillToFrameTrunc - writeIndex;
524
525 // Process 4 loop steps.
526 writeIndex += 4;
527 for (; writeIndex < fillToFrameTrunc; writeIndex += 4)
528 VectorMath::addScalar(values + writeIndex - 4, inc, values + writeIndex, 4);
529 }
530 // Update |value| with the last value computed so that the .value attribute of the AudioParam gets
531 // the correct linear ramp value, in case the following loop doesn't execute.
532 if (writeIndex >= 1)
533 value = values[writeIndex - 1];
534
535 // Serially process remaining values.
536 for (; writeIndex < currentState.fillToFrame; ++writeIndex) {
537 float x = (currentFrame * currentState.samplingPeriod - currentState.time1.value()) * k;
538 value = currentState.value1 + valueDelta * x;
539 values[writeIndex] = value;
540 ++currentFrame;
541 }
542}

544void AudioParamTimeline::processExponentialRamp(const AutomationState& currentState, float* values, size_t& currentFrame, float& value, unsigned& writeIndex)
545{
546 if (!currentState.value1 || currentState.value1 * currentState.value2 < 0) {
547 // Per the specification:
548 // If value1 and value2 have opposite signs or if value1 is zero, then v(t) = value1 for T0 <= t < T1.
549 value = currentState.value1;
550 fillWithValue(values, value, currentState.fillToFrame, writeIndex);
551 return;
552 }
553
554 auto deltaTime = currentState.time2 - currentState.time1;
555 float numSampleFrames = deltaTime.value() * currentState.sampleRate;
556 // The value goes exponentially from value1 to value2 in a duration of deltaTime seconds (corresponding to numSampleFrames).
557 // Compute the per-sample multiplier.
558 float multiplier = powf(currentState.value2 / currentState.value1, 1 / numSampleFrames);
559
560 // Set the starting value of the exponential ramp.
561 value = currentState.value1 * pow(currentState.value2 / static_cast<double>(currentState.value1), (currentFrame * currentState.samplingPeriod - currentState.time1.value()) / deltaTime.value());
562
563 for (; writeIndex < currentState.fillToFrame; ++writeIndex) {
564 values[writeIndex] = value;
565 value *= multiplier;
566 ++currentFrame;
567 }
568
569 // |value| got updated one extra time in the above loop. Restore it to the last computed value.
570 if (writeIndex >= 1)
571 value /= multiplier;
572}

574void AudioParamTimeline::processCancelValues(const AutomationState& currentState, float* values, size_t& currentFrame, float& value, unsigned& writeIndex)
575{
576 // If the previous event was a SetTarget or ExponentialRamp
577 // event, the current value is one sample behind. Update
578 // the sample value by one sample, but only at the start of
579 // this CancelValues event.
580 if (currentState.event->hasDefaultCancelledValue())
581 value = currentState.event->value();
582 else {
583 double cancelFrame = currentState.time1.value() * currentState.sampleRate;
584 if (currentState.eventIndex >= 1 && cancelFrame <= currentFrame && currentFrame < cancelFrame + 1) {
585 auto lastEventType = m_events[currentState.eventIndex - 1].type();
586 if (lastEventType == ParamEvent::SetTarget) {
587 float target = m_events[currentState.eventIndex - 1].value();
588 float timeConstant = m_events[currentState.eventIndex - 1].timeConstant();
589 float discreteTimeConstant = static_cast<float>(AudioUtilities::discreteTimeConstantForSampleRate(timeConstant, currentState.controlRate));
590 value += (target - value) * discreteTimeConstant;
591 }
592 }
593 }
594
595 fillWithValue(values, value, currentState.fillToFrame, writeIndex);
596
597 currentFrame = currentState.fillToEndFrame;
598}

600void AudioParamTimeline::processSetTarget(const AutomationState& currentState, float* values, size_t& currentFrame, float& value, unsigned& writeIndex)
601{
602 // Exponential approach to target value with given time constant.
603 float target = currentState.event->value();
604 float timeConstant = currentState.event->timeConstant();
605 float discreteTimeConstant = static_cast<float>(AudioUtilities::discreteTimeConstantForSampleRate(timeConstant, currentState.controlRate));
606
607 // Set the starting value correctly. This is only needed when the
608 // current time is "equal" to the start time of this event. This is
609 // to get the sampling correct if the start time of this automation
610 // isn't on a frame boundary. Otherwise, we can just continue from
611 // where we left off from the previous rendering quantum.
612 double rampStartFrame = currentState.time1.value() * currentState.sampleRate;
613 // Condition is c - 1 < r <= c where c = currentFrame and r =
614 // rampStartFrame. Compute it this way because currentFrame is
615 // unsigned and could be 0.
616 if (rampStartFrame <= currentFrame && currentFrame < rampStartFrame + 1)
617 value = target + (value - target) * exp(-(currentFrame * currentState.samplingPeriod - currentState.time1.value()) / timeConstant);
618 else {
619 // Otherwise, need to compute a new value because |value| is the
620 // last computed value of SetTarget. Time has progressed by one
621 // frame, so we need to update the value for the new frame.
622 value += (target - value) * discreteTimeConstant;
623 }
624
625 // If the value is close enough to the target, just fill in the data
626 // with the target value.
627 if (hasSetTargetConverged(value, target, Seconds { currentFrame * currentState.samplingPeriod }, currentState.time1, timeConstant)) {
628 currentFrame += currentState.fillToFrame - writeIndex;
629 fillWithValue(values, target, currentState.fillToFrame, writeIndex);
630 value = target;
631 return;
632 }
633
634 if (currentState.fillToFrame > writeIndex) {
635 // Resolve recursion by expanding constants to achieve a 4-step loop unrolling.
636 //
637 // v1 = v0 + (t - v0) * c
638 // v2 = v1 + (t - v1) * c
639 // v2 = v0 + (t - v0) * c + (t - (v0 + (t - v0) * c)) * c
640 // v2 = v0 + (t - v0) * c + (t - v0) * c - (t - v0) * c * c
641 // v2 = v0 + (t - v0) * c * (2 - c)
642 // Thus c0 = c, c1 = c*(2-c). The same logic applies to c2 and c3.
643 const float c0 = discreteTimeConstant;
644 const float c1 = c0 * (2 - c0);
645 const float c2 = c0 * ((c0 - 3) * c0 + 3);
646 const float c3 = c0 * (c0 * ((4 - c0) * c0 - 6) + 4);
647 float delta;
648
649 // Process 4 loop steps.
650 unsigned fillToFrameTrunc = writeIndex + ((currentState.fillToFrame - writeIndex) / 4) * 4;
651 const float cVector[4] = { 0, c0, c1, c2 };
652
653 for (; writeIndex < fillToFrameTrunc; writeIndex += 4) {
654 delta = target - value;
655
656 VectorMath::multiplyByScalar(&cVector[0], delta, &values[writeIndex], 4);
657 VectorMath::addScalar(&values[writeIndex], value, &values[writeIndex], 4);
658
659 value += delta * c3;
660 }
661 }
662
663 // Serially process remaining values.
664 for (; writeIndex < currentState.fillToFrame; ++writeIndex) {
665 values[writeIndex] = value;
666 value += (target - value) * discreteTimeConstant;
667 }
668
669 // The previous loops may have updated |value| one extra time.
670 // Reset it to the last computed value.
671 if (writeIndex >= 1)
672 value = values[writeIndex - 1];
673
674 currentFrame = currentState.fillToEndFrame;
675}

677void AudioParamTimeline::processSetValueCurve(const AutomationState& currentState, float* values, size_t& currentFrame, float& value, unsigned& writeIndex)
678{
679 auto* curveData = currentState.event->curve().data();
680 unsigned numberOfCurvePoints = currentState.event->curve().size();
681 float curveEndValue = currentState.event->curveEndValue();
682 size_t fillToEndFrame = currentState.fillToEndFrame;
683 unsigned fillToFrame = currentState.fillToFrame;
684
685 // Curve events have duration, so don't just use next event time.
686 auto duration = currentState.event->duration();
687 double curvePointsPerFrame = currentState.event->curvePointsPerSecond() * currentState.samplingPeriod;
688
689 if (!curveData || !numberOfCurvePoints || duration <= 0_s || currentState.sampleRate <= 0) {
690 // Error condition - simply propagate previous value.
691 currentFrame = fillToEndFrame;
692 fillWithValue(values, value, fillToFrame, writeIndex);
693 return;
694 }
695
696 // Save old values and recalculate information based on the curve's duration
697 // instead of the next event time.
698 unsigned nextEventFillToFrame = fillToFrame;
699
700 double curveEndFrame = ceil(currentState.sampleRate * (currentState.time1 + duration).value());
701 if (currentState.endFrame > curveEndFrame)
702 fillToEndFrame = static_cast<size_t>(curveEndFrame);
703 else
704 fillToEndFrame = currentState.endFrame;
705
706 fillToFrame = (fillToEndFrame < currentState.startFrame) ? 0 : static_cast<unsigned>(fillToEndFrame - currentState.startFrame);
707 fillToFrame = std::min(fillToFrame, currentState.numberOfValues);
708
709 // Index into the curve data using a floating-point value.
710 // We're scaling the number of curve points by the duration (see curvePointsPerFrame).
711 double curveVirtualIndex = 0;
712 if (currentState.time1.value() < currentFrame * currentState.samplingPeriod) {
713 // Index somewhere in the middle of the curve data.
714 // Don't use timeToSampleFrame() since we want the exact floating-point frame.
715 double frameOffset = currentFrame - currentState.time1.value() * currentState.sampleRate;
716 curveVirtualIndex = curvePointsPerFrame * frameOffset;
717 }
718
719 // Set the default value in case fillToFrame is 0.
720 value = curveEndValue;
721
722 // Render the stretched curve data using nearest neighbor sampling.
723 // Oversampled curve data can be provided if smoothness is desired.
724 int k = 0;
725 for (; writeIndex < fillToFrame; ++writeIndex, ++k) {
726 // Compute current index this way to minimize round-off that would
727 // have occurred by incrementing the index by curvePointsPerFrame.
728 double currentVirtualIndex = curveVirtualIndex + k * curvePointsPerFrame;
729 unsigned curveIndex0;
730
731 // Clamp index to the last element of the array.
732 if (currentVirtualIndex < numberOfCurvePoints)
733 curveIndex0 = static_cast<unsigned>(currentVirtualIndex);
734 else
735 curveIndex0 = numberOfCurvePoints - 1;
736
737 unsigned curveIndex1 = std::min(curveIndex0 + 1, numberOfCurvePoints - 1);
738
739 // Linearly interpolate between the two nearest curve points.
740 // |delta| is clamped to 1 because currentVirtualIndex can exceed
741 // curveIndex0 by more than one. This can happen when we reached
742 // the end of the curve but still need values to fill out the
743 // current rendering quantum.
744 ASSERT(curveIndex0 < numberOfCurvePoints);
745 ASSERT(curveIndex1 < numberOfCurvePoints);
746 float c0 = curveData[curveIndex0];
747 float c1 = curveData[curveIndex1];
748 double delta = std::min(currentVirtualIndex - curveIndex0, 1.0);
749
750 value = c0 + (c1 - c0) * delta;
751
752 values[writeIndex] = value;
753 }
754
755 // If there's any time left after the duration of this event and the start
756 // of the next, then just propagate the last value.
757 if (writeIndex < nextEventFillToFrame) {
758 value = curveEndValue;
759 fillWithValue(values, value, nextEventFillToFrame, writeIndex);
760 }
761
762 // Re-adjust current time
763 currentFrame += nextEventFillToFrame;
764}

766void AudioParamTimeline::processSetTargetFollowedByRamp(int eventIndex, ParamEvent*& event, ParamEvent::Type nextEventType, size_t currentFrame, double sampleRate, double controlRate, float& value)
767{
768 // If the current event is SetTarget and the next event is a LinearRampToValue or ExponentialRampToValue,
769 // special handling is needed. In this case, the linear and exponential ramp should start at wherever
770 // the SetTarget processing has reached.
771 if (event->type() != ParamEvent::SetTarget)
772 return;
773
774 if (nextEventType != ParamEvent::LinearRampToValue && nextEventType != ParamEvent::ExponentialRampToValue)
775 return;
776
777 // Replace the SetTarget with a SetValue to set the starting time and value for the ramp using the
778 // current frame. We need to update |value| appropriately depending on whether the ramp has started
779 // or not.
780 //
781 // If SetTarget starts somewhere between currentFrame - 1 and currentFrame, we directly compute the
782 // value it would have at currentFrame. If not, we update the value from the value from currentFrame - 1.
783 //
784 // Can't use the condition currentFrame - 1 <= t0 * sampleRate <= currentFrame because currentFrame
785 // is unsigned and could be 0. Instead, compute the condition this way, where f = currentFrame and
786 // Fs = sampleRate:
787 //
788 // f - 1 <= t0 * Fs <= f
789 // 2 * f - 2 <= 2 * Fs * t0 <= 2 * f
790 // -2 <= 2 * Fs * t0 - 2 * f <= 0
791 // -1 <= 2 * Fs * t0 - 2 * f + 1 <= 1
792 // abs(2 * Fs * t0 - 2 * f + 1) <= 1
793
794 if (fabs(2 * sampleRate * event->time().value() - 2 * currentFrame + 1) <= 1) {
795 // SetTarget is starting somewhere between currentFrame - 1 and currentFrame. Compute the value
796 // the SetTarget would have at the currentFrame.
797 value = event->value() + (value - event->value()) * exp(-(currentFrame / sampleRate - event->time().value()) / event->timeConstant());
798 } else {
799 // SetTarget has already started. Update |value| one frame because it's the value from the previous frame.
800 float discreteTimeConstant = static_cast<float>(AudioUtilities::discreteTimeConstantForSampleRate(event->timeConstant(), controlRate));
801 value += (event->value() - value) * discreteTimeConstant;
802 }
803 // Insert a SetValueEvent to mark the starting value and time.
804 // Clear the clamp check because this doesn't need it.
805 m_events[eventIndex] = ParamEvent::createSetValueEvent(value, Seconds { currentFrame / sampleRate });
806
807 // Update our pointer to the current event because we just changed it.
808 event = &m_events[eventIndex];
809}

812float AudioParamTimeline::linearRampAtTime(Seconds t, float value1, Seconds time1, float value2, Seconds time2)
813{
814 return value1 + (value2 - value1) * (t - time1).value() / (time2 - time1).value();
815}

817float AudioParamTimeline::exponentialRampAtTime(Seconds t, float value1, Seconds time1, float value2, Seconds time2)
818{
819 return value1 * pow(value2 / value1, (t - time1).value() / (time2 - time1).value());
820}

822float AudioParamTimeline::valueCurveAtTime(Seconds t, Seconds time1, Seconds duration, const float* curveData, size_t curveLength)
823{
824 double curveIndex = (curveLength - 1) / duration.value() * (t - time1).value();
825 size_t k = std::min(static_cast<size_t>(curveIndex), curveLength - 1);
826 size_t k1 = std::min(k + 1, curveLength - 1);
827 float c0 = curveData[k];
828 float c1 = curveData[k1];
829 float delta = std::min(curveIndex - k, 1.0);
830
831 return c0 + (c1 - c0) * delta;
832}

834void AudioParamTimeline::handleCancelValues(ParamEvent& event, ParamEvent* nextEvent, float& value2, Seconds& time2, ParamEvent::Type& nextEventType)
835{
836 if (!nextEvent || nextEvent->type() != ParamEvent::CancelValues || !nextEvent->savedEvent())
837 return;
838
839 float value1 = event.value();
840 auto time1 = event.time();
841
842 switch (event.type()) {
843 case ParamEvent::CancelValues:
844 case ParamEvent::LinearRampToValue:
845 case ParamEvent::ExponentialRampToValue:
846 case ParamEvent::SetValue: {
847 // These three events potentially establish a starting value for
848 // the following event, so we need to examine the cancelled
849 // event to see what to do.
850 auto* savedEvent = nextEvent->savedEvent();
851
852 // Update the end time and type to pretend that we're running
853 // this saved event type.
854 time2 = nextEvent->time();
855 nextEventType = savedEvent->type;
856
857 if (nextEvent->hasDefaultCancelledValue()) {
858 // We've already established a value for the cancelled
859 // event, so just return it.
860 value2 = nextEvent->value();
861 } else {
862 // If the next event would have been a LinearRamp or
863 // ExponentialRamp, we need to compute a new end value for
864 // the event so that the curve works continues as if it were
865 // not cancelled.
866 switch (savedEvent->type) {
867 case ParamEvent::LinearRampToValue:
868 value2 = linearRampAtTime(nextEvent->time(), value1, time1, savedEvent->value, savedEvent->time);
869 break;
870 case ParamEvent::ExponentialRampToValue:
871 value2 = exponentialRampAtTime(nextEvent->time(), value1, time1, savedEvent->value, savedEvent->time);
872 break;
873 case ParamEvent::SetValueCurve:
874 case ParamEvent::SetValue:
875 case ParamEvent::SetTarget:
876 case ParamEvent::CancelValues:
877 // These cannot be possible types for the saved event because they can't be created.
878 // createCancelValuesEvent doesn't allow them (SetValue, SetTarget, CancelValues) or
879 // cancelScheduledValues() doesn't create such an event (SetValueCurve).
880 ASSERT_NOT_REACHED();
881 break;
882 case ParamEvent::LastType:
883 ASSERT_NOT_REACHED();
884 break;
885 }
886
887 // Cache the new value so we don't keep computing it over and over.
888 nextEvent->setCancelledValue(value2);
889 }
890 } break;
891 case ParamEvent::SetValueCurve:
892 // Everything needed for this was handled when cancelling was
893 // done.
894 break;
895 case ParamEvent::SetTarget:
896 // Nothing special needs to be done for SetTarget
897 // followed by CancelValues.
898 break;
899 case ParamEvent::LastType:
900 ASSERT_NOT_REACHED();
901 break;
902 }
903}

951bool AudioParamTimeline::isEventCurrent(const ParamEvent& event, const ParamEvent* nextEvent, size_t currentFrame, double sampleRate) const
952{
953 // WARNING: due to round-off it might happen that nextEvent->time() is
954 // just larger than currentFrame/sampleRate. This means that we will end
955 // up running the |event| again. The code below had better be prepared
956 // for this case! What should happen is the fillToFrame should be 0 so
957 // that while the event is actually run again, nothing actually gets
958 // computed, and we move on to the next event.
959 //
960 // An example of this case is setValueCurveAtTime. The time at which
961 // setValueCurveAtTime ends (and the setValueAtTime begins) might be
962 // just past currentTime/sampleRate. Then setValueCurveAtTime will be
963 // processed again before advancing to setValueAtTime. The number of
964 // frames to be processed should be zero in this case.
965 if (nextEvent && nextEvent->time().value() < currentFrame / sampleRate) {
966 // But if the current event is a SetValue event and the event time is
967 // between currentFrame - 1 and curentFrame (in time). we don't want to
968 // skip it. If we do skip it, the SetValue event is completely skipped
969 // and not applied, which is wrong. Other events don't have this problem.
970 // (Because currentFrame is unsigned, we do the time check in this funny,
971 // but equivalent way.)
972 double eventFrame = event.time().value() * sampleRate;
973
974 // Condition is currentFrame - 1 < eventFrame <= currentFrame, but
975 // currentFrame is unsigned and could be 0, so use
976 // currentFrame < eventFrame + 1 instead.
977 if (!((event.type() == ParamEvent::SetValue && (eventFrame <= currentFrame) && (currentFrame < eventFrame + 1)))) {
978 // This is not the special SetValue event case, and nextEvent is
979 // in the past. We can skip processing of this event since it's
980 // in past.
981 return false;
982 }
983 }
984 return true;
985}

987bool AudioParamTimeline::hasValues(size_t startFrame, double sampleRate) const
988{
989 if (!m_eventsLock.tryLock())
990 return true;
991 Locker locker { AdoptLock, m_eventsLock };
992
993 // Return false if there are no events in the time range.
994 auto endFrame = startFrame + AudioUtilities::renderQuantumSize;

996}
