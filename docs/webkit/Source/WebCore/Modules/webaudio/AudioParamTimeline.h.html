38class AudioParamTimeline {
39 WTF_MAKE_NONCOPYABLE(AudioParamTimeline);
40 WTF_MAKE_FAST_ALLOCATED;
41public:
42 AudioParamTimeline() = default;
43
44 ExceptionOr<void> setValueAtTime(float value, Seconds time);
45 ExceptionOr<void> linearRampToValueAtTime(float targetValue, Seconds endTime, float currentValue, Seconds currentTime);
46 ExceptionOr<void> exponentialRampToValueAtTime(float targetValue, Seconds endTime, float currentValue, Seconds currentTime);
47 ExceptionOr<void> setTargetAtTime(float target, Seconds time, float timeConstant);
48 ExceptionOr<void> setValueCurveAtTime(Vector<float>&& curve, Seconds time, Seconds duration);
49 void cancelScheduledValues(Seconds cancelTime);
50 ExceptionOr<void> cancelAndHoldAtTime(Seconds cancelTime);
51
52 // hasValue is set to true if a valid timeline value is returned.
53 // otherwise defaultValue is returned.
54 std::optional<float> valueForContextTime(BaseAudioContext&, float defaultValue, float minValue, float maxValue);
55
56 // Given the time range, calculates parameter values into the values buffer
57 // and returns the last parameter value calculated for "values" or the defaultValue if none were calculated.
58 // controlRate is the rate (number per second) at which parameter values will be calculated.
59 // It should equal sampleRate for sample-accurate parameter changes, and otherwise will usually match
60 // the render quantum size such that the parameter value changes once per render quantum.
61 float valuesForFrameRange(size_t startFrame, size_t endFrame, float defaultValue, float minValue, float maxValue, float* values, unsigned numberOfValues, double sampleRate, double controlRate);
62
63 bool hasValues(size_t startFrame, double sampleRate) const;
64
65private:
66 class ParamEvent {
67 WTF_MAKE_FAST_ALLOCATED;
68 public:
69 enum Type {
70 SetValue,
71 LinearRampToValue,
72 ExponentialRampToValue,
73 SetTarget,
74 SetValueCurve,
75 CancelValues,
76 LastType
77 };
78
79 struct SavedEvent {
80 Type type;
81 float value;
82 Seconds time;
83 };
84
85 static ParamEvent createSetValueEvent(float value, Seconds time);
86 static ParamEvent createLinearRampEvent(float value, Seconds time);
87 static ParamEvent createExponentialRampEvent(float value, Seconds time);
88 static ParamEvent createSetTargetEvent(float target, Seconds time, float timeConstant);
89 static ParamEvent createSetValueCurveEvent(Vector<float>&& curve, Seconds time, Seconds duration);
90 static ParamEvent createCancelValuesEvent(Seconds cancelTime, std::optional<SavedEvent>&&);
91
92 ParamEvent(Type type, float value, Seconds time, float timeConstant, Seconds duration, Vector<float>&& curve, double curvePointsPerSecond, float curveEndValue, std::optional<SavedEvent>&& savedEvent)
93 : m_type(type)
94 , m_value(value)
95 , m_time(time)
96 , m_timeConstant(timeConstant)
97 , m_duration(duration)
98 , m_curve(WTFMove(curve))
99 , m_curvePointsPerSecond(curvePointsPerSecond)
100 , m_curveEndValue(curveEndValue)
101 , m_savedEvent(WTFMove(savedEvent))
102 {
103 }
104
105 Type type() const { return m_type; }
106 float value() const { return m_value; }
107 Seconds time() const { return m_time; }
108 float timeConstant() const { return m_timeConstant; }
109 Seconds duration() const { return m_duration; }
110 const Vector<float>& curve() const { return m_curve; }
111 SavedEvent* savedEvent() { return m_savedEvent ? &m_savedEvent.value() : nullptr; }
112
113 void setCancelledValue(float cancelledValue)
114 {
115 ASSERT(m_type == Type::CancelValues);
116 m_value = cancelledValue;
117 m_hasDefaultCancelledValue = true;
118 }
119 bool hasDefaultCancelledValue() const
120 {
121 ASSERT(m_type == Type::CancelValues);
122 return m_hasDefaultCancelledValue;
123 }
124
125 double curvePointsPerSecond() const { return m_curvePointsPerSecond; }
126 float curveEndValue() const { return m_curveEndValue; }
127
128 private:
129 Type m_type;
130 float m_value { 0 };
131 Seconds m_time;
132
133 // Only used for SetTarget events.
134 float m_timeConstant { 0 };
135
136 // The duration of the curve.
137 Seconds m_duration;
138
139 // The array of curve points.
140 Vector<float> m_curve;
141
142 // The number of curve points per second. it is used to compute
143 // the curve index step when running the automation.
144 double m_curvePointsPerSecond { 0 };
145
146 // The default value to use at the end of the curve. Normally
147 // it's the last entry in m_curve, but cancelling a SetValueCurve
148 // will set this to a new value.
149 float m_curveEndValue { 0 };
150
151 // True if a default value has been assigned to the CancelValues event.
152 bool m_hasDefaultCancelledValue { false };
153
154 // For CancelValues. If CancelValues is in the middle of an event, this
155 // holds the event that is being cancelled, so that processing can
156 // continue as if the event still existed up until we reach the actual
157 // scheduled cancel time.
158 std::optional<SavedEvent> m_savedEvent;
159 };
160
161 // State of the timeline for the current event.
162 struct AutomationState {
163 // Parameters for the current automation request. Number of
164 // values to be computed for the automation request
165 const unsigned numberOfValues;
166 // Start and end frames for this automation request
167 const size_t startFrame;
168 const size_t endFrame;
169
170 // Sample rate and control rate for this request
171 const double sampleRate;
172 const double controlRate;
173 const double samplingPeriod;
174
175 // Parameters needed for processing the current event.
176 const unsigned fillToFrame;
177 const size_t fillToEndFrame;
178
179 // Value and time for the current event
180 const float value1;
181 const Seconds time1;
182
183 // Value and time for the next event, if any.
184 const float value2;
185 const Seconds time2;
186
187 // The current event, and it's index in the event vector.
188 const ParamEvent* event;
189 const int eventIndex;
190 };
