37AudioSummingJunction::AudioSummingJunction(BaseAudioContext& context)
38 : m_context(makeWeakPtr(context, EnableWeakPtrThreadingAssertions::No)) // WebAudio code uses locking when accessing the context.
39{
40}
41
42AudioSummingJunction::~AudioSummingJunction()
43{
44 if (m_renderingStateNeedUpdating && context())
45 context()->removeMarkedSummingJunction(this);
46}
47
48void AudioSummingJunction::markRenderingStateAsDirty()
49{
50 ASSERT(context());
51 ASSERT(context()->isGraphOwner());
52 if (!m_renderingStateNeedUpdating && canUpdateState()) {
53 context()->markSummingJunctionDirty(this);
54 m_renderingStateNeedUpdating = true;
55 }
56}
57
58bool AudioSummingJunction::addOutput(AudioNodeOutput& output)
59{
60 ASSERT(context());
61 ASSERT(context()->isGraphOwner());
62 if (!m_outputs.add(&output).isNewEntry)
63 return false;
64
65 if (m_pendingRenderingOutputs.isEmpty())
66 m_pendingRenderingOutputs = copyToVector(m_outputs);
67 else
68 m_pendingRenderingOutputs.append(&output);
69
70 markRenderingStateAsDirty();
71 return true;
72}
73
74bool AudioSummingJunction::removeOutput(AudioNodeOutput& output)
75{
76 ASSERT(context());
77 ASSERT(context()->isGraphOwner());
78 if (!m_outputs.remove(&output))
79 return false;
80
81 if (m_pendingRenderingOutputs.isEmpty()) {
82 // Heap allocations are forbidden on the audio thread for performance reasons so we need to
83 // explicitly allow the following allocation(s).
84 DisableMallocRestrictionsForCurrentThreadScope disableMallocRestrictions;
85 m_pendingRenderingOutputs = copyToVector(m_outputs);
86 } else
87 m_pendingRenderingOutputs.removeFirst(&output);
88
89 markRenderingStateAsDirty();
90 return true;
91}
92
93void AudioSummingJunction::updateRenderingState()
94{
95 ASSERT(context());
96 ASSERT(context()->isAudioThread() && context()->isGraphOwner());
97
98 if (m_renderingStateNeedUpdating && canUpdateState()) {
99 // Copy from m_outputs to m_renderingOutputs.
100 m_renderingOutputs = std::exchange(m_pendingRenderingOutputs, { });
101 for (auto& output : m_renderingOutputs)
102 output->updateRenderingState();
103
104 didUpdate();
105
106 m_renderingStateNeedUpdating = false;
107 }
108}
109
110unsigned AudioSummingJunction::maximumNumberOfChannels() const
111{
112 unsigned maxChannels = 0;
113 for (auto* output : m_outputs) {
114 // Use output()->numberOfChannels() instead of output->bus()->numberOfChannels(),
115 // because the calling of AudioNodeOutput::bus() is not safe here.
116 maxChannels = std::max(maxChannels, output->numberOfChannels());
117 }
118 return maxChannels;
119}
