46Ref<AudioWorklet> AudioWorklet::create(BaseAudioContext& audioContext)
47{
48 auto worklet = adoptRef(*new AudioWorklet(audioContext));
49 worklet->suspendIfNeeded();
50 return worklet;
51}
52
53AudioWorklet::AudioWorklet(BaseAudioContext& audioContext)
54 : Worklet(*audioContext.document())
55 , m_audioContext(makeWeakPtr(audioContext))
56{
57}
58
59Vector<Ref<WorkletGlobalScopeProxy>> AudioWorklet::createGlobalScopes()
60{
61 // WebAudio uses a single global scope.
62 return { AudioWorkletMessagingProxy::create(*this) };
63}
64
65AudioWorkletMessagingProxy* AudioWorklet::proxy() const
66{
67 auto& proxies = this->proxies();
68 if (proxies.isEmpty())
69 return nullptr;
70 return downcast<AudioWorkletMessagingProxy>(proxies.first().ptr());
71}
72
73BaseAudioContext* AudioWorklet::audioContext() const
74{
75 return m_audioContext.get();
76}
77
78void AudioWorklet::createProcessor(const String& name, TransferredMessagePort port, Ref<SerializedScriptValue>&& options, AudioWorkletNode& node)
79{
80 auto* proxy = this->proxy();
81 ASSERT(proxy);
82 if (!proxy)
83 return;
84
85 proxy->postTaskForModeToWorkletGlobalScope([name = name.isolatedCopy(), port, options = WTFMove(options), node = makeRef(node)](ScriptExecutionContext& context) mutable {
86 node->setProcessor(downcast<AudioWorkletGlobalScope>(context).createProcessor(name, port, WTFMove(options)));
87 callOnMainThread([node = WTFMove(node)] { });
88 }, WorkerRunLoop::defaultMode());
89}
