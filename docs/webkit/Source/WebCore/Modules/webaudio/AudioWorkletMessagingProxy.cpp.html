48static WorkletParameters generateWorkletParameters(AudioWorklet& worklet)
49{
50 auto* document = worklet.document();
51 auto jsRuntimeFlags = document->settings().javaScriptRuntimeFlags();
52
53 return {
54 document->url(),
55 jsRuntimeFlags,
56 worklet.audioContext() ? worklet.audioContext()->sampleRate() : 0.0f,
57 worklet.identifier(),
58 document->settingsValues(),
59 worklet.audioContext() ? !worklet.audioContext()->isOfflineContext() : false
60 };
61}
62
63AudioWorkletMessagingProxy::AudioWorkletMessagingProxy(AudioWorklet& worklet)
64 : m_worklet(makeWeakPtr(worklet))
65 , m_document(*worklet.document())
66 , m_workletThread(AudioWorkletThread::create(*this, generateWorkletParameters(worklet)))
67{
68 ASSERT(isMainThread());
69
70 m_workletThread->start();
71}
72
73AudioWorkletMessagingProxy::~AudioWorkletMessagingProxy()
74{
75 m_workletThread->stop();
76}
77
78bool AudioWorkletMessagingProxy::postTaskForModeToWorkletGlobalScope(ScriptExecutionContext::Task&& task, const String& mode)
79{
80 m_workletThread->runLoop().postTaskForMode(WTFMove(task), mode);
81 return true;
82}
83
84RefPtr<CacheStorageConnection> AudioWorkletMessagingProxy::createCacheStorageConnection()
85{
86 ASSERT_NOT_REACHED();
87 return nullptr;
88}
89
90RefPtr<RTCDataChannelRemoteHandlerConnection> AudioWorkletMessagingProxy::createRTCDataChannelRemoteHandlerConnection()
91{
92 ASSERT(isMainThread());
93 if (!m_document->page())
94 return nullptr;
95 return m_document->page()->libWebRTCProvider().createRTCDataChannelRemoteHandlerConnection();
96}

98void AudioWorkletMessagingProxy::postTaskToLoader(ScriptExecutionContext::Task&& task)
99{
100 m_document->postTask(WTFMove(task));
101}
102
103bool AudioWorkletMessagingProxy::postTaskForModeToWorkerOrWorkletGlobalScope(ScriptExecutionContext::Task&& task, const String& mode)
104{
105 return postTaskForModeToWorkletGlobalScope(WTFMove(task), mode);
106}
107
108void AudioWorkletMessagingProxy::postTaskToAudioWorklet(Function<void(AudioWorklet&)>&& task)
109{
110 m_document->postTask([this, protectedThis = makeRef(*this), task = WTFMove(task)](ScriptExecutionContext&) {
111 if (m_worklet)
112 task(*m_worklet);
113 });
114}
