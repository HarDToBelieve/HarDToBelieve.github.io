30
31#if ENABLE(WEB_AUDIO)
32#include "WorkerLoaderProxy.h"
33#include "WorkletGlobalScopeProxy.h"
34
35namespace WebCore {
36
37class AudioWorklet;
38class AudioWorkletThread;
39class Document;
40
41class AudioWorkletMessagingProxy : public WorkletGlobalScopeProxy, public WorkerLoaderProxy {
42public:
43 static Ref<AudioWorkletMessagingProxy> create(AudioWorklet& worklet)
44 {
45 return adoptRef(*new AudioWorkletMessagingProxy(worklet));
46 }
47
48 ~AudioWorkletMessagingProxy();
49
50 // This method is used in the main thread to post task back to the worklet thread.
51 bool postTaskForModeToWorkletGlobalScope(ScriptExecutionContext::Task&&, const String& mode) final;
52
53 AudioWorkletThread& workletThread() { return m_workletThread.get(); }
54
55 void postTaskToAudioWorklet(Function<void(AudioWorklet&)>&&);
56
57private:
58 explicit AudioWorkletMessagingProxy(AudioWorklet&);
59
60 // WorkerLoaderProxy.
61 RefPtr<CacheStorageConnection> createCacheStorageConnection() final;
62 RefPtr<RTCDataChannelRemoteHandlerConnection> createRTCDataChannelRemoteHandlerConnection() final;
63 void postTaskToLoader(ScriptExecutionContext::Task&&) final;
64 bool postTaskForModeToWorkerOrWorkletGlobalScope(ScriptExecutionContext::Task&&, const String& mode) final;
65
66 bool isAudioWorkletMessagingProxy() const final { return true; }
67
68 WeakPtr<AudioWorklet> m_worklet;
69 Ref<Document> m_document;
70 Ref<AudioWorkletThread> m_workletThread;
71};
72
73} // namespace WebCore
74
75SPECIALIZE_TYPE_TRAITS_BEGIN(WebCore::AudioWorkletMessagingProxy)
76static bool isType(const WebCore::WorkletGlobalScopeProxy& proxy) { return proxy.isAudioWorkletMessagingProxy(); }
77SPECIALIZE_TYPE_TRAITS_END()
78
79#endif // ENABLE(WEB_AUDIO)
