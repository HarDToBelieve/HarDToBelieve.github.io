61ExceptionOr<Ref<AudioWorkletNode>> AudioWorkletNode::create(JSC::JSGlobalObject& globalObject, BaseAudioContext& context, String&& name, AudioWorkletNodeOptions&& options)
62{
63 if (!options.numberOfInputs && !options.numberOfOutputs)
64 return Exception { NotSupportedError, "Number of inputs and outputs cannot both be 0"_s };
65
66 if (options.outputChannelCount) {
67 if (options.numberOfOutputs != options.outputChannelCount->size())
68 return Exception { IndexSizeError, "Length of specified outputChannelCount does not match the given number of outputs"_s };
69
70 for (auto& channelCount : *options.outputChannelCount) {
71 if (channelCount < 1 || channelCount > AudioContext::maxNumberOfChannels)
72 return Exception { NotSupportedError, "Provided number of channels for output is outside supported range"_s };
73 }
74 }
75
76 auto it = context.parameterDescriptorMap().find(name);
77 if (it == context.parameterDescriptorMap().end())
78 return Exception { InvalidStateError, "No ScriptProcessor was registered with this name"_s };
79 auto& parameterDescriptors = it->value;
80
81 if (!context.scriptExecutionContext())
82 return Exception { InvalidStateError, "Audio context's frame is detached"_s };
83
84 auto messageChannel = MessageChannel::create(*context.scriptExecutionContext());
85 auto nodeMessagePort = messageChannel->port1();
86 auto processorMessagePort = messageChannel->port2();
87
88 RefPtr<SerializedScriptValue> serializedOptions;
89 {
90 auto lock = JSC::JSLockHolder { &globalObject };
91 auto* jsOptions = convertDictionaryToJS(globalObject, *JSC::jsCast<JSDOMGlobalObject*>(&globalObject), options);
92 serializedOptions = SerializedScriptValue::create(globalObject, jsOptions, SerializationErrorMode::NonThrowing);
93 if (!serializedOptions)
94 serializedOptions = SerializedScriptValue::nullValue();
95 }
96
97 auto parameterData = WTFMove(options.parameterData);
98 auto node = adoptRef(*new AudioWorkletNode(context, name, WTFMove(options), *nodeMessagePort));
99 node->suspendIfNeeded();
100
101 auto result = node->handleAudioNodeOptions(options, { 2, ChannelCountMode::Max, ChannelInterpretation::Speakers });
102 if (result.hasException())
103 return result.releaseException();
104
105 node->initializeAudioParameters(parameterDescriptors, parameterData);
106
107 // Will cause the context to ref the node until playback has finished.
108 // Note that a node with zero outputs cannot be a source node.
109 if (node->numberOfOutputs() > 0)
110 context.sourceNodeWillBeginPlayback(node);
111
112 context.audioWorklet().createProcessor(name, processorMessagePort->disentangle(), serializedOptions.releaseNonNull(), node);
113
114 {
115 // The node should be manually added to the automatic pull node list, even without a connect() call.
116 Locker contextLocker { context.graphLock() };
117 node->updatePullStatus();
118 }
119
120 return node;
121}

123AudioWorkletNode::AudioWorkletNode(BaseAudioContext& context, const String& name, AudioWorkletNodeOptions&& options, Ref<MessagePort>&& port)
124 : AudioNode(context, NodeTypeWorklet)
125 , ActiveDOMObject(context.scriptExecutionContext())
126 , m_name(name)
127 , m_parameters(AudioParamMap::create())
128 , m_port(WTFMove(port))
129 , m_wasOutputChannelCountGiven(!!options.outputChannelCount)
130{
131 ASSERT(isMainThread());
132 for (unsigned i = 0; i < options.numberOfInputs; ++i)
133 addInput();
134 for (unsigned i = 0; i < options.numberOfOutputs; ++i)
135 addOutput(options.outputChannelCount ? options.outputChannelCount->at(i): 1);
136
137 m_inputs.resize(options.numberOfInputs);
138 m_outputs.resize(options.numberOfOutputs);
139
140 initialize();
141}

143AudioWorkletNode::~AudioWorkletNode()
144{
145 ASSERT(isMainThread());
146 {
147 Locker locker { m_processLock };
148 if (m_processor) {
149 if (auto* workletProxy = context().audioWorklet().proxy())
150 workletProxy->postTaskForModeToWorkletGlobalScope([m_processor = WTFMove(m_processor)](ScriptExecutionContext&) { }, WorkerRunLoop::defaultMode());
151 }
152 }
153 uninitialize();
154}

156void AudioWorkletNode::initializeAudioParameters(const Vector<AudioParamDescriptor>& descriptors, const std::optional<Vector<WTF::KeyValuePair<String, double>>>& paramValues)
157{
158 ASSERT(isMainThread());
159 ASSERT(m_parameters->map().isEmpty());
160
161 Locker locker { m_processLock };
162
163 for (auto& descriptor : descriptors) {
164 auto parameter = AudioParam::create(context(), descriptor.name, descriptor.defaultValue, descriptor.minValue, descriptor.maxValue, descriptor.automationRate);
165 m_parameters->add(descriptor.name, WTFMove(parameter));
166 }
167
168 if (paramValues) {
169 for (auto& paramValue : *paramValues) {
170 if (auto* audioParam = m_parameters->map().get(paramValue.key))
171 audioParam->setValue(paramValue.value);
172 }
173 }
174
175 for (auto& parameterName : m_parameters->map().keys())
176 m_paramValuesMap.add(parameterName, makeUnique<AudioFloatArray>(AudioUtilities::renderQuantumSize));
177}

179void AudioWorkletNode::setProcessor(RefPtr<AudioWorkletProcessor>&& processor)
180{
181 ASSERT(!isMainThread());
182 if (processor) {
183 Locker locker { m_processLock };
184 m_processor = WTFMove(processor);
185 m_workletThread = &Thread::current();
186 } else
187 fireProcessorErrorOnMainThread(ProcessorError::ConstructorError);
188}

190void AudioWorkletNode::process(size_t framesToProcess)
191{
192 ASSERT(!isMainThread());
193
194 auto zeroOutput = [&] {
195 for (unsigned i = 0; i < numberOfOutputs(); ++i)
196 output(i)->bus()->zero();
197 };
198
199 if (!m_processLock.tryLock()) {
200 zeroOutput();
201 return;
202 }
203 Locker locker { AdoptLock, m_processLock };
204 if (!m_processor || &Thread::current() != m_workletThread) {
205 // We're not ready yet or we are getting destroyed. In this case, we output silence.
206 zeroOutput();
207 return;
208 }
209
210 // If the input is not connected, pass nullptr to the processor.
211 for (unsigned i = 0; i < numberOfInputs(); ++i)
212 m_inputs[i] = input(i)->isConnected() ? input(i)->bus() : nullptr;
213 for (unsigned i = 0; i < numberOfOutputs(); ++i)
214 m_outputs[i] = *output(i)->bus();
215
216 for (auto& audioParam : m_parameters->map().values()) {
217 auto* paramValues = m_paramValuesMap.get(audioParam->name());
218 ASSERT(paramValues);
219 RELEASE_ASSERT(paramValues->size() >= framesToProcess);
220 if (audioParam->hasSampleAccurateValues() && audioParam->automationRate() == AutomationRate::ARate)
221 audioParam->calculateSampleAccurateValues(paramValues->data(), framesToProcess);
222 else
223 std::fill_n(paramValues->data(), framesToProcess, audioParam->finalValue());
224 }
225
226 bool threwException = false;
227 if (!m_processor->process(m_inputs, m_outputs, m_paramValuesMap, threwException))
228 didFinishProcessingOnRenderingThread(threwException);
229}

231void AudioWorkletNode::didFinishProcessingOnRenderingThread(bool threwException)
232{
233 if (threwException)
234 fireProcessorErrorOnMainThread(ProcessorError::ProcessError);
235
236 m_processor = nullptr;
237 m_tailTime = 0;
238
239 if (numberOfOutputs() > 0)
240 context().sourceNodeDidFinishPlayback(*this);
241}

243void AudioWorkletNode::updatePullStatus()
244{
245 ASSERT(context().isGraphOwner());
246
247 bool hasConnectedOutput = false;
248 for (unsigned i = 0; i < numberOfOutputs(); ++i) {
249 if (output(i)->isConnected()) {
250 hasConnectedOutput = true;
251 break;
252 }
253 }
254
255 // If no output is connected, add the node to the automatic pull list.
256 // Otherwise, remove it out of the list.
257 if (!hasConnectedOutput)
258 context().addAutomaticPullNode(*this);
259 else
260 context().removeAutomaticPullNode(*this);
261}

263void AudioWorkletNode::checkNumberOfChannelsForInput(AudioNodeInput* input)
264{
265 ASSERT(context().isAudioThread() && context().isGraphOwner());
266
267 // Dynamic channel count only works when the node has 1 input, 1 output and |outputChannelCount|
268 // is not given. Otherwise the channel count(s) should not be dynamically changed.
269 if (numberOfInputs() == 1 && numberOfOutputs() == 1 && !m_wasOutputChannelCountGiven) {
270 ASSERT(input == this->input(0));
271 unsigned numberOfInputChannels = input->numberOfChannels();
272 if (numberOfInputChannels != output(0)->numberOfChannels()) {
273 // This will propagate the channel count to any nodes connected further downstream in the graph.
274 output(0)->setNumberOfChannels(numberOfInputChannels);
275 }
276 }
277
278 // Update the input's internal bus if needed.
279 AudioNode::checkNumberOfChannelsForInput(input);
280 updatePullStatus();
281}

283void AudioWorkletNode::fireProcessorErrorOnMainThread(ProcessorError error)
284{
285 ASSERT(!isMainThread());
286
287 // Heap allocations are forbidden on the audio thread for performance reasons so we need to
288 // explicitly allow the following allocation(s).
289 DisableMallocRestrictionsForCurrentThreadScope disableMallocRestrictions;
290 callOnMainThread([this, protectedThis = makeRef(*this), error]() mutable {
291 String errorMessage;
292 switch (error) {
293 case ProcessorError::ConstructorError:
294 errorMessage = "An error was thrown from AudioWorkletProcessor constructor"_s;
295 break;
296 case ProcessorError::ProcessError:
297 errorMessage = "An error was thrown from AudioWorkletProcessor::process() method"_s;
298 break;
299 }
300 queueTaskToDispatchEvent(*this, TaskSource::MediaElement, ErrorEvent::create(eventNames().processorerrorEvent, errorMessage, { }, 0, 0, { }));
301 });
302}
