38namespace WebCore {
39
40AudioWorkletThread::AudioWorkletThread(AudioWorkletMessagingProxy& messagingProxy, const WorkletParameters& parameters)
41 : WorkerOrWorkletThread(parameters.identifier.isolatedCopy())
42 , m_messagingProxy(messagingProxy)
43 , m_parameters(parameters.isolatedCopy())
44{
45}
46
47AudioWorkletThread::~AudioWorkletThread() = default;
48
49RefPtr<WorkerOrWorkletGlobalScope> AudioWorkletThread::createGlobalScope()
50{
51 return AudioWorkletGlobalScope::tryCreate(*this, m_parameters);
52}
53
54WorkerLoaderProxy& AudioWorkletThread::workerLoaderProxy()
55{
56 return m_messagingProxy;
57}
58
59WorkerDebuggerProxy* AudioWorkletThread::workerDebuggerProxy() const
60{
61 // FIXME: Add debugging support for AudioWorklets.
62 return nullptr;
63}
64
65Ref<Thread> AudioWorkletThread::createThread()
66{
67 return Thread::create("WebCore: AudioWorklet", [this] {
68 workerOrWorkletThread();
69 }, ThreadType::Audio, m_parameters.isAudioContextRealTime ? Thread::QOS::UserInteractive : Thread::QOS::Default);
70}
71
72AudioWorkletGlobalScope* AudioWorkletThread::globalScope() const
73{
74 return downcast<AudioWorkletGlobalScope>(WorkerOrWorkletThread::globalScope());
75}
