155void BaseAudioContext::lazyInitialize()
156{
157 if (isStopped() || isClosed())
158 return;
159
160 if (m_isInitialized)
161 return;
162
163 // Don't allow the context to initialize a second time after it's already been explicitly uninitialized.
164 ASSERT(!m_isAudioThreadFinished);
165 if (m_isAudioThreadFinished)
166 return;
167
168 destination().initialize();
169
170 m_isInitialized = true;
171}

173void BaseAudioContext::clear()
174{
175 auto protectedThis = makeRef(*this);
176
177 // Audio thread is dead. Nobody will schedule node deletion action. Let's do it ourselves.
178 do {
179 deleteMarkedNodes();
180 m_nodesToDelete = std::exchange(m_nodesMarkedForDeletion, { });
181 } while (!m_nodesToDelete.isEmpty());
182}

184void BaseAudioContext::uninitialize()
185{
186 ALWAYS_LOG(LOGIDENTIFIER);
187
188 ASSERT(isMainThread());
189
190 if (!m_isInitialized)
191 return;
192
193 // This stops the audio thread and all audio rendering.
194 destination().uninitialize();
195
196 // Don't allow the context to initialize a second time after it's already been explicitly uninitialized.
197 m_isAudioThreadFinished = true;
198
199 finishTailProcessing();
200
201 {
202 Locker locker { graphLock() };
203 // This should have been called from handlePostRenderTasks() at the end of rendering.
204 // However, in case of lock contention, the tryLock() call could have failed in handlePostRenderTasks(),
205 // leaving nodes in m_referencedSourceNodes. Now that the audio thread is gone, make sure we deref those nodes
206 // before the BaseAudioContext gets destroyed.
207 derefFinishedSourceNodes();
208 }
209
210 // Get rid of the sources which may still be playing.
211 derefUnfinishedSourceNodes();
212
213 m_isInitialized = false;
214}

216void BaseAudioContext::addReaction(State state, DOMPromiseDeferred<void>&& promise)
217{
218 size_t stateIndex = static_cast<size_t>(state);
219 if (stateIndex >= m_stateReactions.size())
220 m_stateReactions.grow(stateIndex + 1);
221
222 m_stateReactions[stateIndex].append(WTFMove(promise));
223}
224
225void BaseAudioContext::setState(State state)
226{
227 if (m_state != state) {
228 m_state = state;
229 queueTaskToDispatchEvent(*this, TaskSource::MediaElement, Event::create(eventNames().statechangeEvent, Event::CanBubble::Yes, Event::IsCancelable::No));
230 }
231
232 size_t stateIndex = static_cast<size_t>(state);
233 if (stateIndex >= m_stateReactions.size())
234 return;
235
236 Vector<DOMPromiseDeferred<void>> reactions;
237 m_stateReactions[stateIndex].swap(reactions);
238
239 for (auto& promise : reactions)
240 promise.resolve();
241}

243void BaseAudioContext::stop()
244{
245 ASSERT(isMainThread());
246 ALWAYS_LOG(LOGIDENTIFIER);
247
248 // Usually ScriptExecutionContext calls stop twice.
249 if (m_isStopScheduled)
250 return;
251
252 auto protectedThis = makeRef(*this);
253
254 m_isStopScheduled = true;
255
256 ASSERT(document());
257 document()->updateIsPlayingMedia();
258
259 uninitialize();
260 clear();
261}

284void BaseAudioContext::decodeAudioData(Ref<ArrayBuffer>&& audioData, RefPtr<AudioBufferCallback>&& successCallback, RefPtr<AudioBufferCallback>&& errorCallback, std::optional<Ref<DeferredPromise>>&& promise)
285{
286 if (promise && (!document() || !document()->isFullyActive())) {
287 promise.value()->reject(Exception { InvalidStateError, "Document is not fully active"_s });
288 return;
289 }
290
291 if (!m_audioDecoder)
292 m_audioDecoder = makeUnique<AsyncAudioDecoder>();
293
294 m_audioDecoder->decodeAsync(WTFMove(audioData), sampleRate(), [this, activity = makePendingActivity(*this), successCallback = WTFMove(successCallback), errorCallback = WTFMove(errorCallback), promise = WTFMove(promise)](ExceptionOr<Ref<AudioBuffer>>&& result) mutable {
295 queueTaskKeepingObjectAlive(*this, TaskSource::InternalAsyncTask, [successCallback = WTFMove(successCallback), errorCallback = WTFMove(errorCallback), promise = WTFMove(promise), result = WTFMove(result)]() mutable {
296 if (result.hasException()) {
297 if (promise)
298 promise.value()->reject(result.releaseException());
299 if (errorCallback)
300 errorCallback->handleEvent(nullptr);
301 return;
302 }
303 auto audioBuffer = result.releaseReturnValue();
304 if (promise)
305 promise.value()->resolve<IDLInterface<AudioBuffer>>(audioBuffer.get());
306 if (successCallback)
307 successCallback->handleEvent(audioBuffer.ptr());
308 });
309 });
310}

320ExceptionOr<Ref<ScriptProcessorNode>> BaseAudioContext::createScriptProcessor(size_t bufferSize, size_t numberOfInputChannels, size_t numberOfOutputChannels)
321{
322 ALWAYS_LOG(LOGIDENTIFIER);
323
324 ASSERT(isMainThread());
325
326 // W3C Editor's Draft 06 June 2017
327 // https://webaudio.github.io/web-audio-api/#widl-BaseAudioContext-createScriptProcessor-ScriptProcessorNode-unsigned-long-bufferSize-unsigned-long-numberOfInputChannels-unsigned-long-numberOfOutputChannels
328
329 // The bufferSize parameter determines the buffer size in units of sample-frames. If it's not passed in,
330 // or if the value is 0, then the implementation will choose the best buffer size for the given environment,
331 // which will be constant power of 2 throughout the lifetime of the node. ... If the value of this parameter
332 // is not one of the allowed power-of-2 values listed above, an IndexSizeError must be thrown.
333 switch (bufferSize) {
334 case 0:
335#if USE(AUDIO_SESSION)
336 // Pick a value between 256 (2^8) and 16384 (2^14), based on the buffer size of the current AudioSession:
337 bufferSize = 1 << std::max<size_t>(8, std::min<size_t>(14, std::log2(AudioSession::sharedSession().bufferSize())));
338#else
339 bufferSize = 2048;
340#endif
341 break;
342 case 256:
343 case 512:
344 case 1024:
345 case 2048:
346 case 4096:
347 case 8192:
348 case 16384:
349 break;
350 default:
351 return Exception { IndexSizeError, "Unsupported buffer size for ScriptProcessorNode"_s };
352 }
353
354 // An IndexSizeError exception must be thrown if bufferSize or numberOfInputChannels or numberOfOutputChannels
355 // are outside the valid range. It is invalid for both numberOfInputChannels and numberOfOutputChannels to be zero.
356 // In this case an IndexSizeError must be thrown.
357
358 if (!numberOfInputChannels && !numberOfOutputChannels)
359 return Exception { NotSupportedError, "numberOfInputChannels and numberOfOutputChannels cannot both be 0"_s };
360
361 // This parameter [numberOfInputChannels] determines the number of channels for this node's input. Values of
362 // up to 32 must be supported. A NotSupportedError must be thrown if the number of channels is not supported.
363
364 if (numberOfInputChannels > maxNumberOfChannels)
365 return Exception { NotSupportedError, "numberOfInputChannels exceeds maximum number of channels"_s };
366
367 // This parameter [numberOfOutputChannels] determines the number of channels for this node's output. Values of
368 // up to 32 must be supported. A NotSupportedError must be thrown if the number of channels is not supported.
369
370 if (numberOfOutputChannels > maxNumberOfChannels)
371 return Exception { NotSupportedError, "numberOfOutputChannels exceeds maximum number of channels"_s };
372
373 return ScriptProcessorNode::create(*this, bufferSize, numberOfInputChannels, numberOfOutputChannels);
374}

513{

539void BaseAudioContext::handlePreRenderTasks(const AudioIOPosition& outputPosition)
540{
541 ASSERT(isAudioThread());
542
543 // At the beginning of every render quantum, try to update the internal rendering graph state (from main thread changes).
544 // It's OK if the tryLock() fails, we'll just take slightly longer to pick up the changes.
545 if (auto locker = Locker<RecursiveLock>::tryLock(graphLock())) {
546 // Fixup the state of any dirty AudioSummingJunctions and AudioNodeOutputs.
547 handleDirtyAudioSummingJunctions();
548 handleDirtyAudioNodeOutputs();
549
550 updateAutomaticPullNodes();
551 m_outputPosition = outputPosition;
552 }
553}
554
555AudioIOPosition BaseAudioContext::outputPosition()
556{
557 ASSERT(isMainThread());
558 Locker locker { graphLock() };
559 return m_outputPosition;
560}
561
562void BaseAudioContext::handlePostRenderTasks()
563{
564 ASSERT(isAudioThread());
565
566 // Must use a tryLock() here too. Don't worry, the lock will very rarely be contended and this method is called frequently.
567 // The worst that can happen is that there will be some nodes which will take slightly longer than usual to be deleted or removed
568 // from the render graph (in which case they'll render silence).
569 auto locker = Locker<RecursiveLock>::tryLock(graphLock());
570 if (!locker)
571 return;
572
573 // Take care of finishing any derefs where the tryLock() failed previously.
574 handleDeferredDecrementConnectionCounts();
575
576 // Dynamically clean up nodes which are no longer needed.
577 derefFinishedSourceNodes();
578
579 // Don't delete in the real-time thread. Let the main thread do it.
580 // Ref-counted objects held by certain AudioNodes may not be thread-safe.
581 scheduleNodeDeletion();
582
583 // Fixup the state of any dirty AudioSummingJunctions and AudioNodeOutputs.
584 handleDirtyAudioSummingJunctions();
585 handleDirtyAudioNodeOutputs();
586
587 updateAutomaticPullNodes();
588 updateTailProcessingNodes();
589}

591void BaseAudioContext::handleDeferredDecrementConnectionCounts()
592{
593 ASSERT(isAudioThread() && isGraphOwner());
594 for (auto& node : m_deferredBreakConnectionList)
595 node->decrementConnectionCountWithLock();
596
597 m_deferredBreakConnectionList.clear();
598}

600void BaseAudioContext::addTailProcessingNode(AudioNode& node)
601{
602 ASSERT(isGraphOwner());
603 if (node.isTailProcessing()) {
604 ASSERT(m_tailProcessingNodes.contains(node) || m_finishedTailProcessingNodes.contains(node));
605 return;
606 }
607
608 // Ideally we'd find a way to avoid this vector append since we try to avoid potential heap allocations
609 // on the audio thread for performance reasons.
610 DisableMallocRestrictionsForCurrentThreadScope disableMallocRestrictions;
611 ASSERT(!m_tailProcessingNodes.contains(node));
612 m_tailProcessingNodes.append(node);
613}
614
615void BaseAudioContext::removeTailProcessingNode(AudioNode& node)
616{
617 ASSERT(isGraphOwner());
618 ASSERT(node.isTailProcessing());
619
620 if (m_tailProcessingNodes.removeFirst(node))
621 return;
622
623 // Remove the node from finished tail processing nodes so we don't end up disabling its outputs later on the main thread.
624 ASSERT(m_finishedTailProcessingNodes.contains(node));
625 m_finishedTailProcessingNodes.removeFirst(node);
626}
627
628void BaseAudioContext::updateTailProcessingNodes()
629{
630 ASSERT(isAudioThread());
631 ASSERT(isGraphOwner());
632 // Go backwards as the current node may be removed from m_tailProcessingNodes as we iterate.
633 // We are on the audio thread so we want to avoid allocations as much as possible.
634 for (auto i = m_tailProcessingNodes.size(); i > 0; --i) {
635 auto& node = m_tailProcessingNodes[i - 1];
636 if (!node->propagatesSilence())
637 continue; // Node is not done processing its tail.
638
639 // Ideally we'd find a way to avoid this vector append since we try to avoid potential heap allocations
640 // on the audio thread for performance reasons.
641 DisableMallocRestrictionsForCurrentThreadScope disableMallocRestrictions;
642
643 // Disabling of outputs should happen on the main thread we add the node to m_finishedTailProcessingNodes
644 // for disableOutputsForFinishedTailProcessingNodes() to process later on the main thread.
645 ASSERT(!m_finishedTailProcessingNodes.contains(node));
646 m_finishedTailProcessingNodes.append(WTFMove(node));
647 m_tailProcessingNodes.remove(i - 1);
648 }
649
650 if (m_finishedTailProcessingNodes.isEmpty() || m_disableOutputsForTailProcessingScheduled)
651 return;
652
653 m_disableOutputsForTailProcessingScheduled = true;
654
655 // We try to avoid heap allocations on the audio thread but there is no way to do a main thread dispatch
656 // without one.
657 DisableMallocRestrictionsForCurrentThreadScope disableMallocRestrictions;
658 callOnMainThread([this, protectedThis = makeRef(*this)]() mutable {
659 Locker locker { graphLock() };
660 disableOutputsForFinishedTailProcessingNodes();
661 m_disableOutputsForTailProcessingScheduled = false;
662 });
663}
664
665void BaseAudioContext::disableOutputsForFinishedTailProcessingNodes()
666{
667 ASSERT(isMainThread());
668 ASSERT(isGraphOwner());
669 for (auto& finishedTailProcessingNode : std::exchange(m_finishedTailProcessingNodes, { }))
670 finishedTailProcessingNode->disableOutputs();
671}
672
673void BaseAudioContext::finishTailProcessing()
674{
675 ASSERT(isMainThread());
676 Locker locker { graphLock() };
677
678 // disableOutputs() can cause new nodes to start tail processing so we need to loop until both vectors are empty.
679 while (!m_tailProcessingNodes.isEmpty() || !m_finishedTailProcessingNodes.isEmpty()) {
680 for (auto& tailProcessingNode : std::exchange(m_tailProcessingNodes, { }))
681 tailProcessingNode->disableOutputs();
682 disableOutputsForFinishedTailProcessingNodes();
683 }
684}

686void BaseAudioContext::markForDeletion(AudioNode& node)
687{
688 ASSERT(isGraphOwner());
689 ASSERT_WITH_MESSAGE(node.nodeType() != AudioNode::NodeTypeDestination, "Destination node is owned by the BaseAudioContext");
690
691 if (isAudioThreadFinished())
692 m_nodesToDelete.append(&node);
693 else {
694 // Heap allocations are forbidden on the audio thread for performance reasons so we need to
695 // explicitly allow the following allocation(s).
696 DisableMallocRestrictionsForCurrentThreadScope disableMallocRestrictions;
697 m_nodesMarkedForDeletion.append(&node);
698 }
699
700 // This is probably the best time for us to remove the node from automatic pull list,
701 // since all connections are gone and we hold the graph lock. Then when handlePostRenderTasks()
702 // gets a chance to schedule the deletion work, updateAutomaticPullNodes() also gets a chance to
703 // modify m_renderingAutomaticPullNodes.
704 removeAutomaticPullNode(node);
705}
706
707void BaseAudioContext::scheduleNodeDeletion()
708{
709 bool isGood = m_isInitialized && isGraphOwner();
710 ASSERT(isGood);
711 if (!isGood)
712 return;
713
714 // Make sure to call deleteMarkedNodes() on main thread.
715 if (!m_nodesMarkedForDeletion.isEmpty() && !m_isDeletionScheduled) {
716 ASSERT(m_nodesToDelete.isEmpty());
717 m_nodesToDelete = std::exchange(m_nodesMarkedForDeletion, { });
718
719 m_isDeletionScheduled = true;
720
721 // Heap allocations are forbidden on the audio thread for performance reasons so we need to
722 // explicitly allow the following allocation(s).
723 DisableMallocRestrictionsForCurrentThreadScope disableMallocRestrictions;
724 callOnMainThread([protectedThis = makeRef(*this)]() mutable {
725 protectedThis->deleteMarkedNodes();
726 });
727 }

730void BaseAudioContext::deleteMarkedNodes()
731{
732 ASSERT(isMainThread());
733
734 // Protect this object from being deleted before we release the lock.
735 auto protectedThis = makeRef(*this);
736
737 Locker locker { graphLock() };
738
739 while (m_nodesToDelete.size()) {
740 AudioNode* node = m_nodesToDelete.takeLast();
741
742 // Before deleting the node, clear out any AudioNodeInputs from m_dirtySummingJunctions.
743 unsigned numberOfInputs = node->numberOfInputs();
744 for (unsigned i = 0; i < numberOfInputs; ++i)
745 m_dirtySummingJunctions.remove(node->input(i));
746
747 // Before deleting the node, clear out any AudioNodeOutputs from m_dirtyAudioNodeOutputs.
748 unsigned numberOfOutputs = node->numberOfOutputs();
749 for (unsigned i = 0; i < numberOfOutputs; ++i)
750 m_dirtyAudioNodeOutputs.remove(node->output(i));
751
752 ASSERT_WITH_MESSAGE(node->nodeType() != AudioNode::NodeTypeDestination, "Destination node is owned by the BaseAudioContext");
753
754 // Finally, delete it.
755 delete node;
756 }
757 m_isDeletionScheduled = false;
758}

826void BaseAudioContext::updateAutomaticPullNodes()
827{
828 ASSERT(isGraphOwner());
829
830 if (!m_automaticPullNodesNeedUpdating)
831 return;
832
833 // Heap allocations are forbidden on the audio thread for performance reasons so we need to
834 // explicitly allow the following allocation(s).
835 DisableMallocRestrictionsForCurrentThreadScope disableMallocRestrictions;
