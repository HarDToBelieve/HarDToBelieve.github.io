80// AudioContext is the cornerstone of the web audio API and all AudioNodes are created from it.
81// For thread safety between the audio thread and the main thread, it has a rendering graph locking mechanism.

292 class TailProcessingNode {
293 public:
294 TailProcessingNode(AudioNode& node)
295 : m_node(&node)
296 {
297 ASSERT(!node.isTailProcessing());
298 node.setIsTailProcessing(true);
299 }
300 TailProcessingNode(TailProcessingNode&& other)
301 : m_node(std::exchange(other.m_node, nullptr))
302 { }
303 ~TailProcessingNode()
304 {
305 if (m_node)
306 m_node->setIsTailProcessing(false);
307 }
308 TailProcessingNode& operator=(const TailProcessingNode&) = delete;
309 TailProcessingNode& operator=(TailProcessingNode&&) = delete;
310 AudioNode* operator->() const { return m_node.get(); }
311 bool operator==(const TailProcessingNode& other) const { return m_node == other.m_node; }
312 bool operator==(const AudioNode& node) const { return m_node == &node; }
313 private:
314 RefPtr<AudioNode> m_node;
315 };
