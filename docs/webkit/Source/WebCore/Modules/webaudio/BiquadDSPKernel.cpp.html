41static bool hasConstantValue(float* values, size_t framesToProcess)
42{
43 if (framesToProcess <= 1)
44 return true;
45
46 float constant = values[0];
47 for (size_t i = 1; i < framesToProcess; ++i) {
48 if (values[i] != constant)
49 return false;
50 }
51 return true;
52}

54void BiquadDSPKernel::updateCoefficientsIfNecessary(size_t framesToProcess)
55{
56 if (biquadProcessor()->filterCoefficientsDirty()) {
57 if (biquadProcessor()->hasSampleAccurateValues() && biquadProcessor()->shouldUseARate()) {
58 // Use float arrays instead of AudioFloatArray to avoid heap allocations on the audio thread.
59 float cutoffFrequency[AudioUtilities::renderQuantumSize];
60 float q[AudioUtilities::renderQuantumSize];
61 float gain[AudioUtilities::renderQuantumSize];
62 float detune[AudioUtilities::renderQuantumSize]; // in Cents
63
64 RELEASE_ASSERT(framesToProcess <= AudioUtilities::renderQuantumSize);
65
66 biquadProcessor()->parameter1().calculateSampleAccurateValues(cutoffFrequency, framesToProcess);
67 biquadProcessor()->parameter2().calculateSampleAccurateValues(q, framesToProcess);
68 biquadProcessor()->parameter3().calculateSampleAccurateValues(gain, framesToProcess);
69 biquadProcessor()->parameter4().calculateSampleAccurateValues(detune, framesToProcess);
70
71 // If all the values are actually constant for this render (or the
72 // automation rate is "k-rate" for all of the AudioParams), we don't need
73 // to compute filter coefficients for each frame since they would be the
74 // same as the first.
75 bool isConstant = hasConstantValue(cutoffFrequency, framesToProcess)
76 && hasConstantValue(q, framesToProcess)
77 && hasConstantValue(gain, framesToProcess)
78 && hasConstantValue(detune, framesToProcess);
79
80 updateCoefficients(isConstant ? 1 : framesToProcess, cutoffFrequency, q, gain, detune);
81 } else {
82 float cutoffFrequency = biquadProcessor()->parameter1().finalValue();
83 float q = biquadProcessor()->parameter2().finalValue();
84 float gain = biquadProcessor()->parameter3().finalValue();
85 float detune = biquadProcessor()->parameter4().finalValue();
86 updateCoefficients(1, &cutoffFrequency, &q, &gain, &detune);
87 }
88 }
89}

91void BiquadDSPKernel::updateCoefficients(size_t numberOfFrames, const float* cutoffFrequency, const float* q, const float* gain, const float* detune)
92{
93 // Convert from Hertz to normalized frequency 0 -> 1.
94 double nyquist = this->nyquist();
95
96 m_biquad.setHasSampleAccurateValues(numberOfFrames > 1);
97
98 for (size_t k = 0; k < numberOfFrames; ++k) {
99 double normalizedFrequency = cutoffFrequency[k] / nyquist;
100
101 // Offset frequency by detune.
102 if (detune[k]) {
103 // Detune multiplies the frequency by 2^(detune[k] / 1200).
104 normalizedFrequency *= std::exp2(detune[k] / 1200);
105 }
106
107 // Configure the biquad with the new filter parameters for the appropriate
108 // type of filter.
109 switch (biquadProcessor()->type()) {
110 case BiquadFilterType::Lowpass:
111 m_biquad.setLowpassParams(k, normalizedFrequency, q[k]);
112 break;
113
114 case BiquadFilterType::Highpass:
115 m_biquad.setHighpassParams(k, normalizedFrequency, q[k]);
116 break;
117
118 case BiquadFilterType::Bandpass:
119 m_biquad.setBandpassParams(k, normalizedFrequency, q[k]);
120 break;
121
122 case BiquadFilterType::Lowshelf:
123 m_biquad.setLowShelfParams(k, normalizedFrequency, gain[k]);
124 break;
125
126 case BiquadFilterType::Highshelf:
127 m_biquad.setHighShelfParams(k, normalizedFrequency, gain[k]);
128 break;
129
130 case BiquadFilterType::Peaking:
131 m_biquad.setPeakingParams(k, normalizedFrequency, q[k], gain[k]);
132 break;
133
134 case BiquadFilterType::Notch:
135 m_biquad.setNotchParams(k, normalizedFrequency, q[k]);
136 break;
137
138 case BiquadFilterType::Allpass:
139 m_biquad.setAllpassParams(k, normalizedFrequency, q[k]);
140 break;
141 }
142 }
143
144 ASSERT(numberOfFrames);
145 updateTailTime(numberOfFrames - 1);
146}

148void BiquadDSPKernel::process(const float* source, float* destination, size_t framesToProcess)
149{
150 ASSERT(source && destination && biquadProcessor());
151
152 // Recompute filter coefficients if any of the parameters have changed.
153 // FIXME: as an optimization, implement a way that a Biquad object can simply copy its internal filter coefficients from another Biquad object.
154 // Then re-factor this code to only run for the first BiquadDSPKernel of each BiquadProcessor.
155
156 updateCoefficientsIfNecessary(framesToProcess);
157
158 m_biquad.process(source, destination, framesToProcess);
159}

161void BiquadDSPKernel::getFrequencyResponse(unsigned nFrequencies, const float* frequencyHz, float* magResponse, float* phaseResponse)
162{
163 bool isGood = nFrequencies > 0 && frequencyHz && magResponse && phaseResponse;
164 ASSERT(isGood);
165 if (!isGood)
166 return;
167
168 Vector<float> frequency(nFrequencies);
169
170 double nyquist = this->nyquist();
171
172 // Convert from frequency in Hz to normalized frequency (0 -> 1),
173 // with 1 equal to the Nyquist frequency.
174 for (unsigned k = 0; k < nFrequencies; ++k)
175 frequency[k] = frequencyHz[k] / nyquist;
176
177 m_biquad.getFrequencyResponse(nFrequencies, frequency.data(), magResponse, phaseResponse);
178}

190void BiquadDSPKernel::updateTailTime(size_t coefIndex)
191{
192 // A reasonable upper limit for the tail time. While it's easy to
193 // create biquad filters whose tail time can be much larger than
194 // this, limit the maximum to this value so that we don't keep such
195 // nodes alive "forever".
196 // TODO: What is a reasonable upper limit?
197 constexpr double maxTailTime = 30;
198
199 double sampleRate = this->sampleRate();
200 double tail = m_biquad.tailFrame(coefIndex, maxTailTime * sampleRate) / sampleRate;
201
202 m_tailTime = std::clamp(tail, 0.0, maxTailTime);
203}

205bool BiquadDSPKernel::requiresTailProcessing() const
206{
207 // Always return true even if the tail time and latency might both
208 // be zero. This is for simplicity and because TailTime() is 0
209 // basically only when the filter response H(z) = 0 or H(z) = 1. And
210 // it's ok to return true. It just means the node lives a little
211 // longer than strictly necessary.
212 return true;
213}
