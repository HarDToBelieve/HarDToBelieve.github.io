25#pragma once
26
27#include "AudioBasicProcessorNode.h"
28#include "BiquadFilterOptions.h"
29#include "BiquadProcessor.h"
30
31namespace WebCore {
32
33class AudioParam;
34
35class BiquadFilterNode final : public AudioBasicProcessorNode {
36 WTF_MAKE_ISO_ALLOCATED(BiquadFilterNode);
37public:
38 static ExceptionOr<Ref<BiquadFilterNode>> create(BaseAudioContext& context, const BiquadFilterOptions& = { });
39
40 BiquadFilterType type() const;
41 void setType(BiquadFilterType);
42
43 AudioParam& frequency() { return biquadProcessor()->parameter1(); }
44 AudioParam& q() { return biquadProcessor()->parameter2(); }
45 AudioParam& gain() { return biquadProcessor()->parameter3(); }
46 AudioParam& detune() { return biquadProcessor()->parameter4(); }
47
48 // Get the magnitude and phase response of the filter at the given
49 // set of frequencies (in Hz). The phase response is in radians.
50 ExceptionOr<void> getFrequencyResponse(const Ref<Float32Array>& frequencyHz, const Ref<Float32Array>& magResponse, const Ref<Float32Array>& phaseResponse);
51
52private:
53 explicit BiquadFilterNode(BaseAudioContext&);
54
55 BiquadProcessor* biquadProcessor() { return static_cast<BiquadProcessor*>(processor()); }
56};
57
58} // namespace WebCore
