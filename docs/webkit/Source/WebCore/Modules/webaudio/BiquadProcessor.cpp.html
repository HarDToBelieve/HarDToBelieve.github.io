36BiquadProcessor::BiquadProcessor(BaseAudioContext& context, float sampleRate, size_t numberOfChannels, bool autoInitialize)
37 : AudioDSPKernelProcessor(sampleRate, numberOfChannels)
38 , m_parameter1(AudioParam::create(context, "frequency", 350.0, 0.0, 0.5 * sampleRate, AutomationRate::ARate))
39 , m_parameter2(AudioParam::create(context, "Q", 1, -FLT_MAX, FLT_MAX, AutomationRate::ARate))
40 , m_parameter3(AudioParam::create(context, "gain", 0.0, -FLT_MAX, 40 * std::log10(std::numeric_limits<float>::max()), AutomationRate::ARate))
41 , m_parameter4(AudioParam::create(context, "detune", 0.0, -153600, 153600, AutomationRate::ARate))
42{
43 if (autoInitialize)
44 initialize();
45}
46
47BiquadProcessor::~BiquadProcessor()
48{
49 if (isInitialized())
50 uninitialize();
51}

58void BiquadProcessor::checkForDirtyCoefficients()
59{
60 // Deal with smoothing / de-zippering. Start out assuming filter parameters are not changing.
61
62 // The BiquadDSPKernel objects rely on this value to see if they need to re-compute their internal filter coefficients.
63 m_filterCoefficientsDirty = false;
64 m_hasSampleAccurateValues = false;
65
66 if (m_parameter1->hasSampleAccurateValues() || m_parameter2->hasSampleAccurateValues() || m_parameter3->hasSampleAccurateValues() || m_parameter4->hasSampleAccurateValues()) {
67 m_filterCoefficientsDirty = true;
68 m_hasSampleAccurateValues = true;
69
70 m_shouldUseARate = m_parameter1->automationRate() == AutomationRate::ARate
71 || m_parameter2->automationRate() == AutomationRate::ARate
72 || m_parameter3->automationRate() == AutomationRate::ARate
73 || m_parameter4->automationRate() == AutomationRate::ARate;
74 } else {
75 if (m_hasJustReset) {
76 // Snap to exact values first time after reset, then smooth for subsequent changes.
77 m_parameter1->resetSmoothedValue();
78 m_parameter2->resetSmoothedValue();
79 m_parameter3->resetSmoothedValue();
80 m_parameter4->resetSmoothedValue();
81 m_filterCoefficientsDirty = true;
82 m_hasJustReset = false;
83 } else {
84 // Smooth all of the filter parameters. If they haven't yet converged to their target value then mark coefficients as dirty.
85 bool isStable1 = m_parameter1->smooth();
86 bool isStable2 = m_parameter2->smooth();
87 bool isStable3 = m_parameter3->smooth();
88 bool isStable4 = m_parameter4->smooth();
89 if (!(isStable1 && isStable2 && isStable3 && isStable4))
90 m_filterCoefficientsDirty = true;
91 }
92 }
93}

95void BiquadProcessor::process(const AudioBus* source, AudioBus* destination, size_t framesToProcess)
96{
97 if (!isInitialized()) {
98 destination->zero();
99 return;
100 }
101
102 checkForDirtyCoefficients();
103
104 // For each channel of our input, process using the corresponding BiquadDSPKernel into the output channel.
105 for (unsigned i = 0; i < m_kernels.size(); ++i)
106 m_kernels[i]->process(source->channel(i)->data(), destination->channel(i)->mutableData(), framesToProcess);
107}

109void BiquadProcessor::processOnlyAudioParams(size_t framesToProcess)
110{
111 float values[AudioUtilities::renderQuantumSize];
112 ASSERT(framesToProcess <= AudioUtilities::renderQuantumSize);
113
114 m_parameter1->calculateSampleAccurateValues(values, framesToProcess);
115 m_parameter2->calculateSampleAccurateValues(values, framesToProcess);
116 m_parameter3->calculateSampleAccurateValues(values, framesToProcess);
117 m_parameter4->calculateSampleAccurateValues(values, framesToProcess);
118}

120void BiquadProcessor::setType(BiquadFilterType type)
121{
122 if (type != m_type) {
123 m_type = type;
124 reset(); // The filter state must be reset only if the type has changed.
125 }
126}

128void BiquadProcessor::getFrequencyResponse(unsigned nFrequencies, const float* frequencyHz, float* magResponse, float* phaseResponse)
129{
130 // Compute the frequency response on a separate temporary kernel
131 // to avoid interfering with the processing running in the audio
132 // thread on the main kernels.
133
134 auto responseKernel = makeUnique<BiquadDSPKernel>(this);
135
136 // Get a copy of the current biquad filter coefficients so we can update
137 // |responseKernel| with these values.
138 float cutoffFrequency = parameter1().value();
139 float q = parameter2().value();
140 float gain = parameter3().value();
141 float detune = parameter4().value();
142
143 responseKernel->updateCoefficients(1, &cutoffFrequency, &q, &gain, &detune);
144 responseKernel->getFrequencyResponse(nFrequencies, frequencyHz, magResponse, phaseResponse);
145}
