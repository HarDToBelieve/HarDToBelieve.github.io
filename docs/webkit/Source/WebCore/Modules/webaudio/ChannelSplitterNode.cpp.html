40ExceptionOr<Ref<ChannelSplitterNode>> ChannelSplitterNode::create(BaseAudioContext& context, const ChannelSplitterOptions& options)
41{
42 if (options.numberOfOutputs > AudioContext::maxNumberOfChannels || !options.numberOfOutputs)
43 return Exception { IndexSizeError, "Number of outputs is not in the allowed range"_s };
44
45 auto splitter = adoptRef(*new ChannelSplitterNode(context, options.numberOfOutputs));
46
47 auto result = splitter->handleAudioNodeOptions(options, { options.numberOfOutputs, ChannelCountMode::Explicit, ChannelInterpretation::Discrete });
48 if (result.hasException())
49 return result.releaseException();
50
51 return splitter;
52}

54ChannelSplitterNode::ChannelSplitterNode(BaseAudioContext& context, unsigned numberOfOutputs)
55 : AudioNode(context, NodeTypeChannelSplitter)
56{
57 addInput();
58
59 // Create a fixed number of outputs (able to handle the maximum number of channels fed to an input).
60 for (unsigned i = 0; i < numberOfOutputs; ++i)
61 addOutput(1);
62
63 initialize();
64}
65
66void ChannelSplitterNode::process(size_t framesToProcess)
67{
68 AudioBus* source = input(0)->bus();
69 ASSERT(source);
70 ASSERT_UNUSED(framesToProcess, framesToProcess == source->length());
71
72 unsigned numberOfSourceChannels = source->numberOfChannels();
73
74 for (unsigned i = 0; i < numberOfOutputs(); ++i) {
75 AudioBus* destination = output(i)->bus();
76 ASSERT(destination);
77
78 if (i < numberOfSourceChannels) {
79 // Split the channel out if it exists in the source.
80 // It would be nice to avoid the copy and simply pass along pointers, but this becomes extremely difficult with fanout and fanin.
81 destination->channel(0)->copyFrom(source->channel(i));
82 } else if (output(i)->renderingFanOutCount() > 0) {
83 // Only bother zeroing out the destination if it's connected to anything
84 destination->zero();
85 }
86 }
87}

89ExceptionOr<void> ChannelSplitterNode::setChannelCount(unsigned channelCount)
90{
91 if (channelCount != numberOfOutputs())
92 return Exception { IndexSizeError, "Channel count must be set to number of outputs."_s };
93
94 return AudioNode::setChannelCount(channelCount);
95}
96
97ExceptionOr<void> ChannelSplitterNode::setChannelCountMode(ChannelCountMode mode)
98{
99 if (mode != ChannelCountMode::Explicit)
100 return Exception { InvalidStateError, "Channel count mode cannot be changed from explicit."_s };
101
102 return AudioNode::setChannelCountMode(mode);
103}
104
105ExceptionOr<void> ChannelSplitterNode::setChannelInterpretation(ChannelInterpretation interpretation)
106{
107 if (interpretation != ChannelInterpretation::Discrete)
108 return Exception { InvalidStateError, "Channel interpretation cannot be changed from discrete."_s };
109
110 return AudioNode::setChannelInterpretation(interpretation);
111}
