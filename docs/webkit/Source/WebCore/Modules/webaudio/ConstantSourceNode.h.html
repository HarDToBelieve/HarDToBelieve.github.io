28
29#include "AudioScheduledSourceNode.h"
30#include "ConstantSourceOptions.h"
31#include <wtf/Lock.h>
32
33namespace WebCore {
34
35class ConstantSourceNode final : public AudioScheduledSourceNode {
36 WTF_MAKE_ISO_ALLOCATED(ConstantSourceNode);
37public:
38 static ExceptionOr<Ref<ConstantSourceNode>> create(BaseAudioContext&, const ConstantSourceOptions& = { });
39
40 virtual ~ConstantSourceNode();
41
42 AudioParam& offset() { return m_offset.get(); }
43
44private:
45 void process(size_t framesToProcess) final;
46
47 ConstantSourceNode(BaseAudioContext&, float);
48
49 double tailTime() const final { return 0; }
50 double latencyTime() const final { return 0; }
51
52 // If we are no longer playing, propogate silence ahead to downstream nodes.
53 bool propagatesSilence() const final;
54
55 const char* activeDOMObjectName() const override { return "ConstantSourceNode"; }
56
57 Ref<AudioParam> m_offset;
58
59 // Stores sample-accurate values calculated.
60 AudioFloatArray m_sampleAccurateValues;
61};

63} // namespace WebCore
