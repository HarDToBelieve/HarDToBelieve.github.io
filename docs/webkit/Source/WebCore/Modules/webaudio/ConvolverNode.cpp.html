39// Note about empirical tuning:
40// The maximum FFT size affects reverb performance and accuracy.
41// If the reverb is single-threaded and processes entirely in the real-time audio thread,
42// it's important not to make this too high. In this case 8192 is a good value.
43// But, the Reverb object is multi-threaded, so we want this as high as possible without losing too much accuracy.
44// Very large FFTs will have worse phase errors. Given these constraints 32768 is a good compromise.
45constexpr size_t MaxFFTSize = 32768;

51static unsigned computeNumberOfOutputChannels(unsigned inputChannels, unsigned responseChannels)
52{
53 // The number of output channels for a Convolver must be one or two. And can only be one if
54 // there's a mono source and a mono response buffer.
55 return (inputChannels == 1 && responseChannels == 1) ? 1u : 2u;
56}

58ExceptionOr<Ref<ConvolverNode>> ConvolverNode::create(BaseAudioContext& context, ConvolverOptions&& options)
59{
60 auto node = adoptRef(*new ConvolverNode(context));
61
62 auto result = node->handleAudioNodeOptions(options, { 2, ChannelCountMode::ClampedMax, ChannelInterpretation::Speakers });
63 if (result.hasException())
64 return result.releaseException();
65
66 node->setNormalizeForBindings(!options.disableNormalization);
67
68 result = node->setBufferForBindings(WTFMove(options.buffer));
69 if (result.hasException())
70 return result.releaseException();
71
72 return node;
73}

89void ConvolverNode::process(size_t framesToProcess)
90{
91 AudioBus* outputBus = output(0)->bus();
92 ASSERT(outputBus);
93
94 // Synchronize with possible dynamic changes to the impulse response.
95 if (!m_processLock.tryLock()) {
96 // Too bad - tryLock() failed. We must be in the middle of setting a new impulse response.
97 outputBus->zero();
98 return;
99 }
100 Locker locker { AdoptLock, m_processLock };
101
102 if (!isInitialized() || !m_reverb.get())
103 outputBus->zero();
104 else {
105 // Process using the convolution engine.
106 // Note that we can handle the case where nothing is connected to the input, in which case we'll just feed silence into the convolver.
107 // FIXME: If we wanted to get fancy we could try to factor in the 'tail time' and stop processing once the tail dies down if
108 // we keep getting fed silence.
109 m_reverb->process(input(0)->bus(), outputBus, framesToProcess);
110 }
111}

113ExceptionOr<void> ConvolverNode::setBufferForBindings(RefPtr<AudioBuffer>&& buffer)
114{
115 ASSERT(isMainThread());
116
117 if (!buffer)
118 return { };
119
120 if (buffer->sampleRate() != context().sampleRate())
121 return Exception { NotSupportedError, "Buffer sample rate does not match the context's sample rate"_s };
122
123 unsigned numberOfChannels = buffer->numberOfChannels();
124 size_t bufferLength = buffer->length();
125
126 // The current implementation supports only 1-, 2-, or 4-channel impulse responses, with the
127 // 4-channel response being interpreted as true-stereo (see Reverb class).
128 bool isChannelCountGood = (numberOfChannels == 1 || numberOfChannels == 2 || numberOfChannels == 4);
129
130 if (!isChannelCountGood)
131 return Exception { NotSupportedError, "Buffer should have 1, 2 or 4 channels"_s };
132
133 // Wrap the AudioBuffer by an AudioBus. It's an efficient pointer set and not a memcpy().
134 // This memory is simply used in the Reverb constructor and no reference to it is kept for later use in that class.
135 auto bufferBus = AudioBus::create(numberOfChannels, bufferLength, false);
136 for (unsigned i = 0; i < numberOfChannels; ++i)
137 bufferBus->setChannelMemory(i, buffer->channelData(i)->data(), bufferLength);
138
139 bufferBus->setSampleRate(buffer->sampleRate());
140
141 // Create the reverb with the given impulse response.
142 bool useBackgroundThreads = !context().isOfflineContext();
143 auto reverb = makeUnique<Reverb>(bufferBus.get(), AudioUtilities::renderQuantumSize, MaxFFTSize, useBackgroundThreads, m_normalize);
144
145 {
146 // The context must be locked since changing the buffer can re-configure the number of channels that are output.
147 Locker contextLocker { context().graphLock() };
148
149 // Synchronize with process().
150 Locker locker { m_processLock };
151
152 m_reverb = WTFMove(reverb);
153 m_buffer = WTFMove(buffer);
154 if (m_buffer) {
155 // This will propagate the channel count to any nodes connected further downstream in the graph.
156 output(0)->setNumberOfChannels(computeNumberOfOutputChannels(input(0)->numberOfChannels(), m_buffer->numberOfChannels()));
157 }
158 }
159
160 return { };
161}

175double ConvolverNode::tailTime() const
176{
177 ASSERT(context().isAudioThread());
178 if (!m_processLock.tryLock())
179 return std::numeric_limits<double>::infinity();
180 Locker locker { AdoptLock, m_processLock };
181 return m_reverb ? m_reverb->impulseResponseLength() / static_cast<double>(sampleRate()) : 0;
182}
183
184double ConvolverNode::latencyTime() const
185{
186 ASSERT(context().isAudioThread());
187 if (!m_processLock.tryLock())
188 return std::numeric_limits<double>::infinity();
189 Locker locker { AdoptLock, m_processLock };
190 return m_reverb ? m_reverb->latencyFrames() / static_cast<double>(sampleRate()) : 0;
191}

213void ConvolverNode::checkNumberOfChannelsForInput(AudioNodeInput* input)
214{
215 ASSERT(context().isAudioThread() && context().isGraphOwner());
216 std::optional<unsigned> numberOfBufferChannels;
217 if (m_processLock.tryLock()) {
218 Locker locker { AdoptLock, m_processLock };
219 if (m_buffer)
220 numberOfBufferChannels = m_buffer->numberOfChannels();
221 }
222
223 if (numberOfBufferChannels) {
224 unsigned numberOfOutputChannels = computeNumberOfOutputChannels(input->numberOfChannels(), *numberOfBufferChannels);
225
226 if (isInitialized() && numberOfOutputChannels != output(0)->numberOfChannels()) {
227 // We're already initialized but the channel count has changed.
228 uninitialize();
229 }
230
231 if (!isInitialized()) {
232 // This will propagate the channel count to any nodes connected further
233 // downstream in the graph.
234 output(0)->setNumberOfChannels(numberOfOutputChannels);
235 initialize();
236 }
237 }
238
239 // Update the input's internal bus if needed.
240 AudioNode::checkNumberOfChannelsForInput(input);
241}
