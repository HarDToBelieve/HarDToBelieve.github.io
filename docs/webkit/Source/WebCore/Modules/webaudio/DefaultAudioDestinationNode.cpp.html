50DefaultAudioDestinationNode::DefaultAudioDestinationNode(AudioContext& context, std::optional<float> sampleRate)
51 : AudioDestinationNode(context, sampleRate.value_or(AudioDestination::hardwareSampleRate()))
52{
53 ASSERT(BaseAudioContext::isSupportedSampleRate(AudioDestination::hardwareSampleRate()));
54 initializeDefaultNodeOptions(2, ChannelCountMode::Explicit, ChannelInterpretation::Speakers);
55}

72void DefaultAudioDestinationNode::initialize()
73{
74 ASSERT(isMainThread());
75 if (isInitialized())
76 return;
77 ALWAYS_LOG(LOGIDENTIFIER);
78
79 createDestination();
80 AudioNode::initialize();
81}
82
83void DefaultAudioDestinationNode::uninitialize()
84{
85 ASSERT(isMainThread());
86 if (!isInitialized())
87 return;
88
89 ALWAYS_LOG(LOGIDENTIFIER);
90 clearDestination();
91 m_numberOfInputChannels = 0;
92
93 AudioNode::uninitialize();
94}
95
96void DefaultAudioDestinationNode::clearDestination()
97{
98 ASSERT(m_destination);
99 if (m_wasDestinationStarted) {
100 m_destination->stop();
101 m_wasDestinationStarted = false;
102 }
103 m_destination->clearCallback();
104 m_destination = nullptr;
105}
106
107void DefaultAudioDestinationNode::createDestination()
108{
109 ALWAYS_LOG(LOGIDENTIFIER, "contextSampleRate = ", sampleRate(), ", hardwareSampleRate = ", AudioDestination::hardwareSampleRate());
110 ASSERT(!m_destination);
111 m_destination = platformStrategies()->mediaStrategy().createAudioDestination(*this, m_inputDeviceId, m_numberOfInputChannels, channelCount(), sampleRate());
112}
113
114void DefaultAudioDestinationNode::recreateDestination()
115{
116 bool wasDestinationStarted = m_wasDestinationStarted;
117 clearDestination();
118 createDestination();
119 if (wasDestinationStarted) {
120 m_wasDestinationStarted = true;
121 m_destination->start(dispatchToRenderThreadFunction());
122 }
123}
124
125void DefaultAudioDestinationNode::enableInput(const String& inputDeviceId)
126{
127 ALWAYS_LOG(LOGIDENTIFIER);
128
129 ASSERT(isMainThread());
130 if (m_numberOfInputChannels != EnabledInputChannels) {
131 m_numberOfInputChannels = EnabledInputChannels;
132 m_inputDeviceId = inputDeviceId;
133
134 if (isInitialized())
135 recreateDestination();
136 }
137}

139Function<void(Function<void()>&&)> DefaultAudioDestinationNode::dispatchToRenderThreadFunction()
140{
141 if (auto* workletProxy = context().audioWorklet().proxy()) {
142 return [workletProxy = makeRef(*workletProxy)](Function<void()>&& function) {
143 workletProxy->postTaskForModeToWorkletGlobalScope([function = WTFMove(function)](ScriptExecutionContext&) mutable {
144 function();
145 }, WorkerRunLoop::defaultMode());
146 };
147 }
148 return nullptr;
149}

151void DefaultAudioDestinationNode::startRendering(CompletionHandler<void(std::optional<Exception>&&)>&& completionHandler)
152{
153 ASSERT(isInitialized());
154 if (!isInitialized())
155 return completionHandler(Exception { InvalidStateError, "AudioDestinationNode is not initialized"_s });
156
157 auto innerCompletionHandler = [completionHandler = WTFMove(completionHandler)](bool success) mutable {
158 completionHandler(success ? std::nullopt : std::make_optional(Exception { InvalidStateError, "Failed to start the audio device"_s }));
159 };
160
161 m_wasDestinationStarted = true;
162 m_destination->start(dispatchToRenderThreadFunction(), WTFMove(innerCompletionHandler));
163}
164
165void DefaultAudioDestinationNode::resume(CompletionHandler<void(std::optional<Exception>&&)>&& completionHandler)
166{
167 ASSERT(isInitialized());
168 if (!isInitialized()) {
169 context().postTask([completionHandler = WTFMove(completionHandler)]() mutable {
170 completionHandler(Exception { InvalidStateError, "AudioDestinationNode is not initialized"_s });
171 });
172 return;
173 }
174 m_wasDestinationStarted = true;
175 m_destination->start(dispatchToRenderThreadFunction(), [completionHandler = WTFMove(completionHandler)](bool success) mutable {
176 completionHandler(success ? std::nullopt : std::make_optional(Exception { InvalidStateError, "Failed to start the audio device"_s }));
177 });
178}
179
180void DefaultAudioDestinationNode::suspend(CompletionHandler<void(std::optional<Exception>&&)>&& completionHandler)
181{
182 ASSERT(isInitialized());
183 if (!isInitialized()) {
184 context().postTask([completionHandler = WTFMove(completionHandler)]() mutable {
185 completionHandler(Exception { InvalidStateError, "AudioDestinationNode is not initialized"_s });
186 });
187 return;
188 }
189
190 m_wasDestinationStarted = false;
191 m_destination->stop([completionHandler = WTFMove(completionHandler)](bool success) mutable {
192 completionHandler(success ? std::nullopt : std::make_optional(Exception { InvalidStateError, "Failed to stop the audio device"_s }));
193 });
194}
195
196void DefaultAudioDestinationNode::restartRendering()
197{
198 if (!m_wasDestinationStarted)
199 return;
200
201 m_destination->stop();
202 m_destination->start(dispatchToRenderThreadFunction());
203}
204
205void DefaultAudioDestinationNode::close(CompletionHandler<void()>&& completionHandler)
206{
207 ASSERT(isInitialized());
208 uninitialize();
209 context().postTask(WTFMove(completionHandler));
210}

213{
214 return AudioDestination::maxChannelCount();

217ExceptionOr<void> DefaultAudioDestinationNode::setChannelCount(unsigned channelCount)
218{
219 // The channelCount for the input to this node controls the actual number of channels we
220 // send to the audio hardware. It can only be set depending on the maximum number of
221 // channels supported by the hardware.
222
223 ASSERT(isMainThread());
224 ALWAYS_LOG(LOGIDENTIFIER, channelCount);
225
226 if (channelCount > maxChannelCount())
227 return Exception { IndexSizeError, "Channel count exceeds maximum limit"_s };
228
229 auto oldChannelCount = this->channelCount();
230 auto result = AudioNode::setChannelCount(channelCount);
231 if (result.hasException())
232 return result;
233
234 if (this->channelCount() != oldChannelCount && isInitialized())
235 recreateDestination();
236
237 return { };
238}

245void DefaultAudioDestinationNode::render(AudioBus*, AudioBus* destinationBus, size_t numberOfFrames, const AudioIOPosition& outputPosition)
246{
247 renderQuantum(destinationBus, numberOfFrames, outputPosition);
248
249 setIsSilent(destinationBus->isSilent());
250
251 // The reason we are handling mute after the call to setIsSilent() is because the muted state does
252 // not affect the audio destination node's effective playing state.
253 if (m_muted)
254 destinationBus->zero();
255}
256
257void DefaultAudioDestinationNode::setIsSilent(bool isSilent)
258{
259 if (m_isSilent == isSilent)
260 return;
261
262 m_isSilent = isSilent;
263 updateIsEffectivelyPlayingAudio();
264}
265
266void DefaultAudioDestinationNode::isPlayingDidChange()
267{
268 updateIsEffectivelyPlayingAudio();
269}
270
271void DefaultAudioDestinationNode::updateIsEffectivelyPlayingAudio()
272{
273 bool isEffectivelyPlayingAudio = m_destination && m_destination->isPlaying() && !m_isSilent;
274 if (m_isEffectivelyPlayingAudio == isEffectivelyPlayingAudio)
275 return;
276
277 m_isEffectivelyPlayingAudio = isEffectivelyPlayingAudio;
278 context().isPlayingAudioDidChange();
279}
