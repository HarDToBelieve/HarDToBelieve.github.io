37static size_t bufferLengthForDelay(double maxDelayTime, double sampleRate)
38{
39 // Compute the length of the buffer needed to handle a max delay of |maxDelayTime|. Add an additional render quantum frame size so we can
40 // vectorize the delay processing. The extra space is needed so that writes to the buffer won't overlap reads from the buffer.
41 return AudioUtilities::renderQuantumSize + AudioUtilities::timeToSampleFrame(maxDelayTime, sampleRate, AudioUtilities::SampleFrameRounding::Up);
42}
43
44// Returns (a - b) if a is greater than b, 0 otherwise.
45template<typename T> static inline size_t positiveSubtract(T a, T b)
46{
47 return a <= b ? 0 : static_cast<size_t>(a - b);
48}
49
50static void copyToCircularBuffer(float* buffer, size_t writeIndex, size_t bufferLength, const float* source, size_t framesToProcess)
51{
52 // The algorithm below depends on this being true because we don't expect to have to fill the entire buffer more than once.
53 RELEASE_ASSERT(bufferLength >= framesToProcess);
54
55 // Copy |framesToProcess| values from |source| to the circular buffer that starts at |buffer| of length |bufferLength|. The
56 // copy starts at index |writeIndex| into the buffer.
57 auto* writePointer = &buffer[writeIndex];
58 size_t remainder = positiveSubtract(bufferLength, writeIndex);
59
60 // Copy the frames over, carefully handling the case where we need to wrap around to the beginning of the buffer.
61 memcpy(writePointer, source, sizeof(*writePointer) * std::min(framesToProcess, remainder));
62 memcpy(buffer, source + remainder, sizeof(*buffer) * positiveSubtract(framesToProcess, remainder));
63}

65DelayDSPKernel::DelayDSPKernel(DelayProcessor* processor)
66 : AudioDSPKernel(processor)
67 , m_delayTimes(AudioUtilities::renderQuantumSize)
68 , m_tempBuffer(AudioUtilities::renderQuantumSize)
69{
70 ASSERT(processor && processor->sampleRate() > 0);
71 if (!(processor && processor->sampleRate() > 0))
72 return;
73
74 m_maxDelayTime = processor->maxDelayTime();
75 ASSERT(m_maxDelayTime >= 0);
76 if (m_maxDelayTime < 0)
77 return;
78
79 m_buffer.resize(bufferLengthForDelay(m_maxDelayTime, processor->sampleRate()));
80}
81
82DelayDSPKernel::DelayDSPKernel(double maxDelayTime, float sampleRate)
83 : AudioDSPKernel(sampleRate)
84 , m_maxDelayTime(maxDelayTime)
85 , m_tempBuffer(AudioUtilities::renderQuantumSize)
86{
87 ASSERT(maxDelayTime > 0.0);
88 if (maxDelayTime <= 0.0)
89 return;
90
91 size_t bufferLength = bufferLengthForDelay(maxDelayTime, sampleRate);
92 ASSERT(bufferLength);
93 if (!bufferLength)
94 return;
95
96 m_buffer.resize(bufferLength);
97}

99void DelayDSPKernel::process(const float* source, float* destination, size_t framesToProcess)
100{
101 ASSERT(m_buffer.size());
102 ASSERT(source && destination);
103 if (UNLIKELY(m_buffer.isEmpty() || !source || !destination))
104 return;
105
106 bool sampleAccurate = delayProcessor() && delayProcessor()->delayTime().hasSampleAccurateValues();
107 bool shouldUseARate = delayProcessor() && delayProcessor()->delayTime().automationRate() == AutomationRate::ARate;
108 if (sampleAccurate && shouldUseARate)
109 processARate(source, destination, framesToProcess);
110 else
111 processKRate(source, destination, framesToProcess);
112}
113
114void DelayDSPKernel::processARate(const float* source, float* destination, size_t framesToProcess)
115{
116 size_t bufferLength = m_buffer.size();
117 auto* buffer = m_buffer.data();
118
119 delayProcessor()->delayTime().calculateSampleAccurateValues(m_delayTimes.data(), framesToProcess);
120
121 copyToCircularBuffer(buffer, m_writeIndex, bufferLength, source, framesToProcess);
122
123 for (unsigned i = 0; i < framesToProcess; ++i) {
124 double delayTime = std::clamp<double>(m_delayTimes[i], 0.0, maxDelayTime());
125 double desiredDelayFrames = delayTime * sampleRate();
126
127 double readPosition = m_writeIndex + bufferLength - desiredDelayFrames;
128 if (readPosition >= bufferLength)
129 readPosition -= bufferLength;
130
131 // Linearly interpolate in-between delay times.
132 size_t readIndex1 = static_cast<size_t>(readPosition);
133 size_t readIndex2 = (readIndex1 + 1) % bufferLength;
134 float interpolationFactor = readPosition - readIndex1;
135
136 m_writeIndex = (m_writeIndex + 1) % bufferLength;
137
138 float sample1 = buffer[readIndex1];
139 float sample2 = buffer[readIndex2];
140 destination[i] = sample1 + interpolationFactor * (sample2 - sample1);
141 }
142}

144// Optimized version of processARate() when the delayTime is constant.
145void DelayDSPKernel::processKRate(const float* source, float* destination, size_t framesToProcess)
146{
147 size_t bufferLength = m_buffer.size();
148 auto* buffer = m_buffer.data();
149
150 double delayTime = delayProcessor() ? delayProcessor()->delayTime().finalValue() : m_desiredDelayFrames / sampleRate();
151 // Make sure the delay time is in a valid range.
152 delayTime = std::clamp(delayTime, 0.0, maxDelayTime());
153 double desiredDelayFrames = delayTime * sampleRate();
154
155 double readPosition = m_writeIndex + bufferLength - desiredDelayFrames;
156 if (readPosition >= bufferLength)
157 readPosition -= bufferLength;
158
159 // Linearly interpolate in-between delay times. |readIndex1| and |readIndex2| are the indices of the frames to be used
160 // for interpolation.
161 size_t readIndex1 = static_cast<size_t>(readPosition);
162 float interpolationFactor = readPosition - readIndex1;
163 auto* bufferEnd = &buffer[bufferLength];
164 ASSERT(static_cast<unsigned>(bufferLength) >= framesToProcess);
165
166 // sample1 and sample2 hold the current and next samples in the buffer. These are used for interoplating the delay value.
167 // To reduce memory usage and an extra memcpy, sample1 can be the same as destination.
168 // VectorMath::interpolate() below has an optimization in the case where the input buffer is the same as the output one.
169 auto* sample1 = destination;
170
171 // Copy data from the source into the buffer, starting at the write index. The buffer is circular, so carefully handle
172 // the wrapping of the write pointer.
173 copyToCircularBuffer(buffer, m_writeIndex, bufferLength, source, framesToProcess);
174 m_writeIndex = (m_writeIndex + framesToProcess) % bufferLength;
175
176 // Now copy out the samples from the buffer, starting at the read pointer, carefully handling wrapping of the read pointer.
177 auto* readPointer = &buffer[readIndex1];
178
179 size_t remainder = positiveSubtract(bufferEnd, readPointer);
180 memcpy(sample1, readPointer, sizeof(*sample1) * std::min(framesToProcess, remainder));
181 memcpy(sample1 + remainder, buffer, sizeof(*sample1) * positiveSubtract(framesToProcess, remainder));
182
183 // If interpolationFactor is 0, we don't need to do any interpolation and sample1 contains the desired values.
184 if (!interpolationFactor)
185 return;
186
187 ASSERT(framesToProcess <= m_tempBuffer.size());
188
189 size_t readIndex2 = (readIndex1 + 1) % bufferLength;
190 auto* sample2 = m_tempBuffer.data();
191
192 readPointer = &buffer[readIndex2];
193 remainder = positiveSubtract(bufferEnd, readPointer);
194 memcpy(sample2, readPointer, sizeof(*sample2) * std::min(framesToProcess, remainder));
195 memcpy(sample2 + remainder, buffer, sizeof(*sample2) * positiveSubtract(framesToProcess, remainder));
196
197 // Interpolate samples.
198 // destination[k] = sample1[k] + interpolationFactor * (sample2[k] - sample1[k]);
199 VectorMath::interpolate(sample1, sample2, interpolationFactor, destination, framesToProcess);
200}

202void DelayDSPKernel::processOnlyAudioParams(size_t framesToProcess)
203{
204 if (!delayProcessor())
205 return;
206
207 float values[AudioUtilities::renderQuantumSize];
208 ASSERT(framesToProcess <= AudioUtilities::renderQuantumSize);
209
210 delayProcessor()->delayTime().calculateSampleAccurateValues(values, framesToProcess);
211}

228bool DelayDSPKernel::requiresTailProcessing() const
229{
230 // Always return true even if the tail time and latency might both
231 // be zero. This is for simplicity; most interesting delay nodes
232 // have non-zero delay times anyway. And it's ok to return true. It
233 // just means the node lives a little longer than strictly
234 // necessary.
235 return true;
236}
