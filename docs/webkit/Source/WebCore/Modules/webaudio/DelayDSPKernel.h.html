26
27#include "AudioArray.h"
28#include "AudioDSPKernel.h"
29#include "DelayProcessor.h"

31namespace WebCore {
32
33class DelayProcessor;
34
35class DelayDSPKernel final : public AudioDSPKernel {
36public:
37 explicit DelayDSPKernel(DelayProcessor*);
38 DelayDSPKernel(double maxDelayTime, float sampleRate);
39
40 void process(const float* source, float* destination, size_t framesToProcess) override;
41 void processOnlyAudioParams(size_t framesToProcess) final;
42 void reset() override;
43
44 double maxDelayTime() const { return m_maxDelayTime; }
45
46 void setDelayFrames(double numberOfFrames) { m_desiredDelayFrames = numberOfFrames; }
47
48 double tailTime() const override;
49 double latencyTime() const override;
50 bool requiresTailProcessing() const final;
51
52private:
53 void processARate(const float* source, float* destination, size_t framesToProcess);
54 void processKRate(const float* source, float* destination, size_t framesToProcess);
55
56 AudioFloatArray m_buffer;
57 double m_maxDelayTime;
58 size_t m_writeIndex { 0 };
59 double m_desiredDelayFrames;
60
61 AudioFloatArray m_delayTimes;
62 // Temporary buffer used to hold the second sample for interpolation if needed.
63 AudioFloatArray m_tempBuffer;
64
65 DelayProcessor* delayProcessor() { return static_cast<DelayProcessor*>(processor()); }
66};

68} // namespace WebCore
