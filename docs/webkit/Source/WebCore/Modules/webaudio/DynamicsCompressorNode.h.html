26#pragma once
27
28#include "AudioNode.h"
29#include "AudioParam.h"
30#include "DynamicsCompressorOptions.h"
31#include <memory>
32
33namespace WebCore {
34
35class DynamicsCompressor;
36
37class DynamicsCompressorNode final : public AudioNode {
38 WTF_MAKE_ISO_ALLOCATED(DynamicsCompressorNode);
39public:
40 static ExceptionOr<Ref<DynamicsCompressorNode>> create(BaseAudioContext&, const DynamicsCompressorOptions& = { });
41
42 ~DynamicsCompressorNode();
43
44 // AudioNode
45 void process(size_t framesToProcess) final;
46 void processOnlyAudioParams(size_t framesToProcess) final;
47 void initialize() final;
48 void uninitialize() final;
49
50 // Static compression curve parameters.
51 AudioParam& threshold() { return m_threshold.get(); }
52 AudioParam& knee() { return m_knee.get(); }
53 AudioParam& ratio() { return m_ratio.get(); }
54 AudioParam& attack() { return m_attack.get(); }
55 AudioParam& release() { return m_release.get(); }
56
57 // Amount by which the compressor is currently compressing the signal in decibels.
58 float reduction() const { return m_reduction; }
59
60 ExceptionOr<void> setChannelCount(unsigned) final;
61 ExceptionOr<void> setChannelCountMode(ChannelCountMode) final;
62
63protected:
64 explicit DynamicsCompressorNode(BaseAudioContext&, const DynamicsCompressorOptions& = { });
65 virtual void setReduction(float reduction) { m_reduction = reduction; }
66
67private:
68 double tailTime() const final;
69 double latencyTime() const final;
70 bool requiresTailProcessing() const final;
71
72 std::unique_ptr<DynamicsCompressor> m_dynamicsCompressor;
73 Ref<AudioParam> m_threshold;
74 Ref<AudioParam> m_knee;
75 Ref<AudioParam> m_ratio;
76 Ref<AudioParam> m_attack;
77 Ref<AudioParam> m_release;
78 float m_reduction { 0 };
79};
80
81} // namespace WebCore
