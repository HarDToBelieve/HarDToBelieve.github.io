41ExceptionOr<Ref<GainNode>> GainNode::create(BaseAudioContext& context, const GainOptions& options)
42{
43 auto gainNode = adoptRef(*new GainNode(context));
44
45 auto result = gainNode->handleAudioNodeOptions(options, { 2, ChannelCountMode::Max, ChannelInterpretation::Speakers });
46 if (result.hasException())
47 return result.releaseException();
48
49 gainNode->gain().setValue(options.gain);
50
51 return gainNode;
52}

54GainNode::GainNode(BaseAudioContext& context)
55 : AudioNode(context, NodeTypeGain)
56 , m_sampleAccurateGainValues(AudioUtilities::renderQuantumSize) // FIXME: can probably share temp buffer in context
57 , m_gain(AudioParam::create(context, "gain"_s, 1.0, -FLT_MAX, FLT_MAX, AutomationRate::ARate))
58{
59 addInput();
60 addOutput(1);
61
62 initialize();
63}

65void GainNode::process(size_t framesToProcess)
66{
67 // FIXME: for some cases there is a nice optimization to avoid processing here, and let the gain change
68 // happen in the summing junction input of the AudioNode we're connected to.
69 // Then we can avoid all of the following:
70
71 AudioBus* outputBus = output(0)->bus();
72 ASSERT(outputBus);
73
74 if (!isInitialized() || !input(0)->isConnected())
75 outputBus->zero();
76 else {
77 AudioBus* inputBus = input(0)->bus();
78
79 if (gain().hasSampleAccurateValues() && gain().automationRate() == AutomationRate::ARate) {
80 // Apply sample-accurate gain scaling for precise envelopes, grain windows, etc.
81 ASSERT(framesToProcess <= m_sampleAccurateGainValues.size());
82 if (framesToProcess <= m_sampleAccurateGainValues.size()) {
83 float* gainValues = m_sampleAccurateGainValues.data();
84 gain().calculateSampleAccurateValues(gainValues, framesToProcess);
85 outputBus->copyWithSampleAccurateGainValuesFrom(*inputBus, gainValues, framesToProcess);
86 }
87 } else {
88 // Apply the gain with de-zippering into the output bus.
89 float gain = this->gain().hasSampleAccurateValues() ? this->gain().finalValue() : this->gain().value();
90 if (!gain) {
91 // If the gain is 0 just zero the bus.
92 outputBus->zero();
93 } else
94 outputBus->copyWithGainFrom(*inputBus, gain);
95 }
96 }
97}

99void GainNode::processOnlyAudioParams(size_t framesToProcess)
100{
101 float values[AudioUtilities::renderQuantumSize];
102 ASSERT(framesToProcess <= AudioUtilities::renderQuantumSize);
103
104 m_gain->calculateSampleAccurateValues(values, framesToProcess);
105}

107// FIXME: this can go away when we do mixing with gain directly in summing junction of AudioNodeInput
108//
109// As soon as we know the channel count of our input, we can lazily initialize.
110// Sometimes this may be called more than once with different channel counts, in which case we must safely
111// uninitialize and then re-initialize with the new channel count.
112void GainNode::checkNumberOfChannelsForInput(AudioNodeInput* input)
113{
114 ASSERT(context().isAudioThread() && context().isGraphOwner());
115
116 ASSERT(input && input == this->input(0));
117 if (input != this->input(0))
118 return;
119
120 unsigned numberOfChannels = input->numberOfChannels();
121
122 if (isInitialized() && numberOfChannels != output(0)->numberOfChannels()) {
123 // We're already initialized but the channel count has changed.
124 uninitialize();
125 }
126
127 if (!isInitialized()) {
128 // This will propagate the channel count to any nodes connected further downstream in the graph.
129 output(0)->setNumberOfChannels(numberOfChannels);
130 initialize();
131 }
132
133 AudioNode::checkNumberOfChannelsForInput(input);
134}
