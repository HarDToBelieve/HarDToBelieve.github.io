35IIRProcessor::IIRProcessor(float sampleRate, unsigned numberOfChannels, const Vector<double>& feedforward, const Vector<double>& feedback, bool isFilterStable)
36 : AudioDSPKernelProcessor(sampleRate, numberOfChannels)
37 , m_feedforward(feedforward)
38 , m_feedback(feedback)
39 , m_isFilterStable(isFilterStable)
40{
41 unsigned feedbackLength = feedback.size();
42 unsigned feedforwardLength = feedforward.size();
43 ASSERT(feedbackLength > 0);
44 ASSERT(feedforwardLength > 0);
45
46 // Need to scale the feedback and feedforward coefficients appropriately.
47 // (It's up to the caller to ensure feedbackCoef[0] is not 0.)
48 ASSERT(!!feedback[0]);
49
50 if (feedback[0] != 1) {
51 // The provided filter is:
52 //
53 // a[0]*y(n) + a[1]*y(n-1) + ... = b[0]*x(n) + b[1]*x(n-1) + ...
54 //
55 // We want the leading coefficient of y(n) to be 1:
56 //
57 // y(n) + a[1]/a[0]*y(n-1) + ... = b[0]/a[0]*x(n) + b[1]/a[0]*x(n-1) + ...
58 //
59 // Thus, the feedback and feedforward coefficients need to be scaled by 1/a[0].
60 float scale = feedback[0];
61 for (unsigned k = 1; k < feedbackLength; ++k)
62 m_feedback[k] /= scale;
63
64 for (unsigned k = 0; k < feedforwardLength; ++k)
65 m_feedforward[k] /= scale;
66
67 // The IIRFilter checks to make sure this coefficient is 1, so make it so.
68 m_feedback[0] = 1;
69 }
70
71 m_responseKernel = makeUnique<IIRDSPKernel>(*this);
72}
73
74IIRProcessor::~IIRProcessor()
75{
76 if (isInitialized())
77 uninitialize();
78}
79
80std::unique_ptr<AudioDSPKernel> IIRProcessor::createKernel()
81{
82 return makeUnique<IIRDSPKernel>(*this);
83}
84
85void IIRProcessor::process(const AudioBus* source, AudioBus* destination, size_t framesToProcess)
86{
87 if (!isInitialized()) {
88 destination->zero();
89 return;
90 }
91
92 // For each channel of our input, process using the corresponding IIRDSPKernel
93 // into the output channel.
94 for (size_t i = 0; i < m_kernels.size(); ++i)
95 m_kernels[i]->process(source->channel(i)->data(), destination->channel(i)->mutableData(), framesToProcess);
96}

98void IIRProcessor::getFrequencyResponse(unsigned length, const float* frequencyHz, float* magResponse, float* phaseResponse)
99{
100 m_responseKernel->getFrequencyResponse(length, frequencyHz, magResponse, phaseResponse);
101}
