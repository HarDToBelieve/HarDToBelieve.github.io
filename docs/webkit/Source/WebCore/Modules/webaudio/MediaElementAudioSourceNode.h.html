27
28#if ENABLE(WEB_AUDIO) && ENABLE(VIDEO)
29
30#include "AudioNode.h"
31#include "AudioSourceProviderClient.h"
32#include "HTMLMediaElement.h"
33#include "MultiChannelResampler.h"
34#include <memory>
35#include <wtf/Lock.h>
36
37namespace WebCore {
38
39class AudioContext;
40struct MediaElementAudioSourceOptions;
41
42class MediaElementAudioSourceNode final : public AudioNode, public AudioSourceProviderClient {
43 WTF_MAKE_ISO_ALLOCATED(MediaElementAudioSourceNode);
44public:
45 static ExceptionOr<Ref<MediaElementAudioSourceNode>> create(BaseAudioContext&, MediaElementAudioSourceOptions&&);
46
47 virtual ~MediaElementAudioSourceNode();
48
49 HTMLMediaElement& mediaElement() { return m_mediaElement; }
50
51 // AudioNode
52 void process(size_t framesToProcess) override;
53
54 // AudioSourceProviderClient
55 void setFormat(size_t numberOfChannels, float sampleRate) override;
56
57 Lock& processLock() WTF_RETURNS_LOCK(m_processLock) { return m_processLock; }
58
59private:
60 MediaElementAudioSourceNode(BaseAudioContext&, Ref<HTMLMediaElement>&&);
61 void provideInput(AudioBus*, size_t framesToProcess);
62
63 double tailTime() const override { return 0; }
64 double latencyTime() const override { return 0; }
65 bool requiresTailProcessing() const final { return false; }
66
67 // As an audio source, we will never propagate silence.
68 bool propagatesSilence() const override { return false; }
69
70 bool wouldTaintOrigin();
71
72 Ref<HTMLMediaElement> m_mediaElement;
73 Lock m_processLock;
74
75 unsigned m_sourceNumberOfChannels WTF_GUARDED_BY_LOCK(m_processLock) { 0 };
76 double m_sourceSampleRate WTF_GUARDED_BY_LOCK(m_processLock) { 0 };
77 bool m_muted WTF_GUARDED_BY_LOCK(m_processLock) { false };
78
79 std::unique_ptr<MultiChannelResampler> m_multiChannelResampler WTF_GUARDED_BY_LOCK(m_processLock);
80};
81
82} // namespace WebCore

84#endif // ENABLE(WEB_AUDIO) && ENABLE(VIDEO)
