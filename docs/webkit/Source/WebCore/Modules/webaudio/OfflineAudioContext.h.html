35namespace WebCore {
36
37struct OfflineAudioContextOptions;
38
39class OfflineAudioContext final : public BaseAudioContext {
40 WTF_MAKE_ISO_ALLOCATED(OfflineAudioContext);
41public:
42 static ExceptionOr<Ref<OfflineAudioContext>> create(ScriptExecutionContext&, const OfflineAudioContextOptions&);
43 static ExceptionOr<Ref<OfflineAudioContext>> create(ScriptExecutionContext&, unsigned numberOfChannels, unsigned length, float sampleRate);
44 void startRendering(Ref<DeferredPromise>&&);
45 void suspendRendering(double suspendTime, Ref<DeferredPromise>&&);
46 void resumeRendering(Ref<DeferredPromise>&&);
47 void finishedRendering(bool didRendering);
48 void didSuspendRendering(size_t frame);
49
50 unsigned length() const { return m_length; }
51 bool shouldSuspend();
52
53 OfflineAudioDestinationNode& destination() final { return m_destinationNode.get(); }
54 const OfflineAudioDestinationNode& destination() const final { return m_destinationNode.get(); }
55
56private:
57 OfflineAudioContext(Document&, const OfflineAudioContextOptions&);
58
59 AudioBuffer* renderTarget() const { return destination().renderTarget(); }
60
61 // ActiveDOMObject
62 const char* activeDOMObjectName() const final;
63 bool virtualHasPendingActivity() const final;
64
65 void settleRenderingPromise(ExceptionOr<Ref<AudioBuffer>>&&);
66 void uninitialize() final;
67 bool isOfflineContext() const final { return true; }
68
69 UniqueRef<OfflineAudioDestinationNode> m_destinationNode;
70 RefPtr<DeferredPromise> m_pendingRenderingPromise;
71 HashMap<unsigned /* frame */, RefPtr<DeferredPromise>, WTF::IntHash<unsigned>, WTF::UnsignedWithZeroKeyHashTraits<unsigned>> m_suspendRequests;
72 unsigned m_length;
73 bool m_didStartRendering { false };
74};
75
76} // namespace WebCore
