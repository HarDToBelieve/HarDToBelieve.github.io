42// Breakpoints where we deicde to do linear interoplation, 3-point interpolation or 5-point interpolation. See doInterpolation().
43constexpr float interpolate2Point = 0.3;
44constexpr float interpolate3Point = 0.16;
45
46// Convert the detune value (in cents) to a frequency scale multiplier: 2^(d/1200).
47static inline float detuneToFrequencyMultiplier(float detuneValue)
48{
49 return std::exp2(detuneValue / 1200);
50}
51
52// Clamp the frequency value to lie within Nyquist frequency. For NaN, arbitrarily clamp to +Nyquist.
53static void clampFrequency(float* frequency, size_t framesToProcess, float nyquist)
54{
55 for (size_t k = 0; k < framesToProcess; ++k) {
56 float f = frequency[k];
57 frequency[k] = std::isnan(f) ? nyquist : clampTo(f, -nyquist, nyquist);
58 }
59}

61ExceptionOr<Ref<OscillatorNode>> OscillatorNode::create(BaseAudioContext& context, const OscillatorOptions& options)
62{
63 if (options.type == OscillatorType::Custom && !options.periodicWave)
64 return Exception { InvalidStateError, "Must provide periodicWave when using custom type."_s };
65
66 auto oscillator = adoptRef(*new OscillatorNode(context, options));
67
68 auto result = oscillator->handleAudioNodeOptions(options, { 2, ChannelCountMode::Max, ChannelInterpretation::Speakers });
69 if (result.hasException())
70 return result.releaseException();
71
72 if (options.periodicWave)
73 oscillator->setPeriodicWave(*options.periodicWave);
74 else {
75 result = oscillator->setTypeForBindings(options.type);
76 if (result.hasException())
77 return result.releaseException();
78 }
79
80 return oscillator;
81}

100ExceptionOr<void> OscillatorNode::setTypeForBindings(OscillatorType type)
101{
102 ALWAYS_LOG(LOGIDENTIFIER, type);
103 ASSERT(isMainThread());
104
105 if (type == OscillatorType::Custom) {
106 if (m_type != OscillatorType::Custom)
107 return Exception { InvalidStateError, "OscillatorNode.type cannot be changed to 'custom'"_s };
108 return { };
109 }
110
111 setPeriodicWave(context().periodicWave(type));
112 m_type = type;
113
114 return { };

117bool OscillatorNode::calculateSampleAccuratePhaseIncrements(size_t framesToProcess)
118{
119 bool isGood = framesToProcess <= m_phaseIncrements.size() && framesToProcess <= m_detuneValues.size();
120 ASSERT(isGood);
121 if (!isGood)
122 return false;
123
124 if (m_firstRender) {
125 m_firstRender = false;
126 m_frequency->resetSmoothedValue();
127 m_detune->resetSmoothedValue();
128 }
129
130 bool hasSampleAccurateValues = false;
131 bool hasFrequencyChanges = false;
132 float* phaseIncrements = m_phaseIncrements.data();
133
134 float finalScale = m_periodicWave->rateScale();
135
136 if (m_frequency->hasSampleAccurateValues() && m_frequency->automationRate() == AutomationRate::ARate) {
137 hasSampleAccurateValues = true;
138 hasFrequencyChanges = true;
139
140 // Get the sample-accurate frequency values and convert to phase increments.
141 // They will be converted to phase increments below.
142 m_frequency->calculateSampleAccurateValues(phaseIncrements, framesToProcess);
143 } else {
144 float frequency = m_frequency->finalValue();
145 finalScale *= frequency;
146 }
147
148 if (m_detune->hasSampleAccurateValues() && m_detune->automationRate() == AutomationRate::ARate) {
149 hasSampleAccurateValues = true;
150
151 // Get the sample-accurate detune values.
152 float* detuneValues = hasFrequencyChanges ? m_detuneValues.data() : phaseIncrements;
153 m_detune->calculateSampleAccurateValues(detuneValues, framesToProcess);
154
155 // Convert from cents to rate scalar.
156 VectorMath::multiplyByScalar(detuneValues, 1.0 / 1200, detuneValues, framesToProcess);
157 for (unsigned i = 0; i < framesToProcess; ++i)
158 detuneValues[i] = std::exp2(detuneValues[i]);
159
160 if (hasFrequencyChanges) {
161 // Multiply frequencies by detune scalings.
162 VectorMath::multiply(detuneValues, phaseIncrements, phaseIncrements, framesToProcess);
163 }
164 } else {
165 float detune = m_detune->finalValue();
166 float detuneScale = detuneToFrequencyMultiplier(detune);
167 finalScale *= detuneScale;
168 }
169
170 if (hasSampleAccurateValues) {
171 clampFrequency(phaseIncrements, framesToProcess, context().sampleRate() / 2);
172 // Convert from frequency to wave increment.
173 VectorMath::multiplyByScalar(phaseIncrements, finalScale, phaseIncrements, framesToProcess);
174 }
175
176 return hasSampleAccurateValues;
177}

179static float doInterpolation(double virtualReadIndex, float incr, unsigned readIndexMask, float tableInterpolationFactor, const float* lowerWaveData, const float* higherWaveData)
180{
181 ASSERT(incr >= 0);
182 ASSERT(std::isfinite(virtualReadIndex));
183
184 double sampleLower = 0;
185 double sampleHigher = 0;
186
187 unsigned readIndex0 = static_cast<unsigned>(virtualReadIndex);
188
189 // Consider a typical sample rate of 44100 Hz and max periodic wave
190 // size of 4096. The relationship between |incr| and the frequency
191 // of the oscillator is |incr| = freq * 4096/44100. Or freq =
192 // |incr|*44100/4096 = 10.8*|incr|.
193 //
194 // For the |incr| thresholds below, this means that we use linear
195 // interpolation for all freq >= 3.2 Hz, 3-point Lagrange
196 // for freq >= 1.7 Hz and 5-point Lagrange for every thing else.
197 //
198 // We use Lagrange interpolation because it's relatively simple to
199 // implement and fairly inexpensive, and the interpolator always
200 // passes through known points.
201 if (incr >= interpolate2Point) {
202 // Increment is fairly large, so we're doing no more than about 3
203 // points between each wave table entry. Assume linear
204 // interpolation between points is good enough.
205 unsigned readIndex2 = readIndex0 + 1;
206
207 // Contain within valid range.
208 readIndex0 = readIndex0 & readIndexMask;
209 readIndex2 = readIndex2 & readIndexMask;
210
211 float sample1Lower = lowerWaveData[readIndex0];
212 float sample2Lower = lowerWaveData[readIndex2];
213 float sample1Higher = higherWaveData[readIndex0];
214 float sample2Higher = higherWaveData[readIndex2];
215
216 // Linearly interpolate within each table (lower and higher).
217 double interpolationFactor = static_cast<float>(virtualReadIndex) - readIndex0;
218 sampleHigher = (1 - interpolationFactor) * sample1Higher + interpolationFactor * sample2Higher;
219 sampleLower = (1 - interpolationFactor) * sample1Lower + interpolationFactor * sample2Lower;
220
221 } else if (incr >= interpolate3Point) {
222 // We're doing about 6 interpolation values between each wave
223 // table sample. Just use a 3-point Lagrange interpolator to get a
224 // better estimate than just linear.
225 //
226 // See 3-point formula in http://dlmf.nist.gov/3.3#ii
227 unsigned readIndex[3];
228
229 for (int k = -1; k <= 1; ++k)
230 readIndex[k + 1] = (readIndex0 + k) & readIndexMask;
231
232 double a[3];
233 double t = virtualReadIndex - readIndex0;
234
235 a[0] = 0.5 * t * (t - 1);
236 a[1] = 1 - t * t;
237 a[2] = 0.5 * t * (t + 1);
238
239 for (int k = 0; k < 3; ++k) {
240 sampleLower += a[k] * lowerWaveData[readIndex[k]];
241 sampleHigher += a[k] * higherWaveData[readIndex[k]];
242 }
243 } else {
244 // For everything else (more than 6 points per entry), we'll do a
245 // 5-point Lagrange interpolator. This is a trade-off between
246 // quality and speed.
247 //
248 // See 5-point formula in http://dlmf.nist.gov/3.3#ii
249 unsigned readIndex[5];
250 for (int k = -2; k <= 2; ++k)
251 readIndex[k + 2] = (readIndex0 + k) & readIndexMask;
252
253 double a[5];
254 double t = virtualReadIndex - readIndex0;
255 double t2 = t * t;
256
257 a[0] = t * (t2 - 1) * (t - 2) / 24;
258 a[1] = -t * (t - 1) * (t2 - 4) / 6;
259 a[2] = (t2 - 1) * (t2 - 4) / 4;
260 a[3] = -t * (t + 1) * (t2 - 4) / 6;
261 a[4] = t * (t2 - 1) * (t + 2) / 24;
262
263 for (int k = 0; k < 5; ++k) {
264 sampleLower += a[k] * lowerWaveData[readIndex[k]];
265 sampleHigher += a[k] * higherWaveData[readIndex[k]];
266 }
267 }
268
269 // Then interpolate between the two tables.
270 float sample = (1 - tableInterpolationFactor) * sampleHigher + tableInterpolationFactor * sampleLower;
271 return sample;
272}

274double OscillatorNode::processARate(int n, float* destP, double virtualReadIndex, float* phaseIncrements)
275{
276 float rateScale = m_periodicWave->rateScale();
277 float invRateScale = 1 / rateScale;
278 unsigned periodicWaveSize = m_periodicWave->periodicWaveSize();
279 double invPeriodicWaveSize = 1.0 / periodicWaveSize;
280 unsigned readIndexMask = periodicWaveSize - 1;
281
282 float* higherWaveData = nullptr;
283 float* lowerWaveData = nullptr;
284 float tableInterpolationFactor = 0;
285
286 for (int k = 0; k < n; ++k) {
287 float incr = *phaseIncrements++;
288
289 float frequency = invRateScale * incr;
290 m_periodicWave->waveDataForFundamentalFrequency(frequency, lowerWaveData, higherWaveData, tableInterpolationFactor);
291
292 float sample = doInterpolation(virtualReadIndex, fabs(incr), readIndexMask, tableInterpolationFactor, lowerWaveData, higherWaveData);
293
294 *destP++ = sample;
295
296 // Increment virtual read index and wrap virtualReadIndex into the range
297 // 0 -> periodicWaveSize.
298 virtualReadIndex += incr;
299 virtualReadIndex -= floor(virtualReadIndex * invPeriodicWaveSize) * periodicWaveSize;
300 }
301
302 return virtualReadIndex;
303}

305double OscillatorNode::processKRate(int n, float* destP, double virtualReadIndex)
306{
307 unsigned periodicWaveSize = m_periodicWave->periodicWaveSize();
308 double invPeriodicWaveSize = 1.0 / periodicWaveSize;
309 unsigned readIndexMask = periodicWaveSize - 1;
310
311 float frequency = 0;
312 float* higherWaveData = nullptr;
313 float* lowerWaveData = nullptr;
314 float tableInterpolationFactor = 0;
315
316 frequency = m_frequency->finalValue();
317 float detune = m_detune->finalValue();
318 float detuneScale = detuneToFrequencyMultiplier(detune);
319 frequency *= detuneScale;
320 clampFrequency(&frequency, 1, context().sampleRate() / 2);
321 m_periodicWave->waveDataForFundamentalFrequency(frequency, lowerWaveData, higherWaveData, tableInterpolationFactor);
322
323 float rateScale = m_periodicWave->rateScale();
324 float incr = frequency * rateScale;
325
326 for (int k = 0; k < n; ++k) {
327 float sample = doInterpolation(virtualReadIndex, fabs(incr), readIndexMask, tableInterpolationFactor, lowerWaveData, higherWaveData);
328
329 *destP++ = sample;
330
331 // Increment virtual read index and wrap virtualReadIndex into the range
332 // 0 -> periodicWaveSize.
333 virtualReadIndex += incr;
334 virtualReadIndex -= floor(virtualReadIndex * invPeriodicWaveSize) * periodicWaveSize;
335 }
336
337 return virtualReadIndex;
338}

340void OscillatorNode::process(size_t framesToProcess)
341{
342 auto& outputBus = *output(0)->bus();
343
344 if (!isInitialized() || !outputBus.numberOfChannels()) {
345 outputBus.zero();
346 return;
347 }
348
349 ASSERT(framesToProcess <= m_phaseIncrements.size());
350 if (framesToProcess > m_phaseIncrements.size())
351 return;
352
353 // The audio thread can't block on this lock, so we use tryLock() instead.
354 if (!m_processLock.tryLock()) {
355 // Too bad - tryLock() failed. We must be in the middle of changing wave-tables.
356 outputBus.zero();
357 return;
358 }
359 Locker locker { AdoptLock, m_processLock };
360
361 // We must access m_periodicWave only inside the lock.
362 if (!m_periodicWave.get()) {
363 outputBus.zero();
364 return;
365 }
366
367 size_t quantumFrameOffset = 0;
368 size_t nonSilentFramesToProcess = 0;
369 double startFrameOffset = 0;
370 updateSchedulingInfo(framesToProcess, outputBus, quantumFrameOffset, nonSilentFramesToProcess, startFrameOffset);
371
372 if (!nonSilentFramesToProcess) {
373 outputBus.zero();
374 return;
375 }
376
377 float* destP = outputBus.channel(0)->mutableData();
378
379 ASSERT(quantumFrameOffset <= framesToProcess);
380
381 // We keep virtualReadIndex double-precision since we're accumulating values.
382 double virtualReadIndex = m_virtualReadIndex;
383
384 float rateScale = m_periodicWave->rateScale();
385 bool hasSampleAccurateValues = calculateSampleAccuratePhaseIncrements(framesToProcess);
386
387 float frequency = 0;
388 float* higherWaveData = nullptr;
389 float* lowerWaveData = nullptr;
390 float tableInterpolationFactor = 0;
391
392 if (!hasSampleAccurateValues) {
393 frequency = m_frequency->finalValue();
394 float detune = m_detune->finalValue();
395 float detuneScale = detuneToFrequencyMultiplier(detune);
396 frequency *= detuneScale;
397 clampFrequency(&frequency, 1, context().sampleRate() / 2);
398 m_periodicWave->waveDataForFundamentalFrequency(frequency, lowerWaveData, higherWaveData, tableInterpolationFactor);
399 }
400
401 float* phaseIncrements = m_phaseIncrements.data();
402
403 // Start rendering at the correct offset.
404 destP += quantumFrameOffset;
405 int n = nonSilentFramesToProcess;
406
407 // If startFrameOffset is not 0, that means the oscillator doesn't actually
408 // start at quantumFrameOffset, but just past that time. Adjust destP and n
409 // to reflect that, and adjust virtualReadIndex to start the value at
410 // startFrameOffset.
411 if (startFrameOffset > 0) {
412 ++destP;
413 --n;
414 virtualReadIndex += (1 - startFrameOffset) * frequency * rateScale;
415 ASSERT(virtualReadIndex < m_periodicWave->periodicWaveSize());
416 } else if (startFrameOffset < 0)
417 virtualReadIndex = -startFrameOffset * frequency * rateScale;
418
419 if (hasSampleAccurateValues)
420 virtualReadIndex = processARate(n, destP, virtualReadIndex, phaseIncrements);
421 else
422 virtualReadIndex = processKRate(n, destP, virtualReadIndex);
423
424 m_virtualReadIndex = virtualReadIndex;
425
426 outputBus.clearSilentFlag();
427}

429void OscillatorNode::setPeriodicWave(PeriodicWave& periodicWave)
430{
431 ALWAYS_LOG(LOGIDENTIFIER, "sample rate = ", periodicWave.sampleRate(), ", wave size = ", periodicWave.periodicWaveSize(), ", rate scale = ", periodicWave.rateScale());
432 ASSERT(isMainThread());
433
434 // This synchronizes with process().
435 Locker locker { m_processLock };
436 m_periodicWave = &periodicWave;
437 m_type = OscillatorType::Custom;
438}

440bool OscillatorNode::propagatesSilence() const
441{
442 ASSERT(context().isAudioThread());
443 if (!isPlayingOrScheduled() || hasFinished())
444 return true;
445 if (!m_processLock.tryLock())
446 return false; // Assume we have a periodic wave if we are unable to grab the lock.
447 Locker locker { AdoptLock, m_processLock };
448 return !m_periodicWave.get();
449}
