30#include <wtf/Lock.h>

32namespace WebCore {
33
34class PeriodicWave;
35
36// OscillatorNode is an audio generator of periodic waveforms.
37
38class OscillatorNode final : public AudioScheduledSourceNode {
39 WTF_MAKE_ISO_ALLOCATED(OscillatorNode);
40public:
41 static ExceptionOr<Ref<OscillatorNode>> create(BaseAudioContext&, const OscillatorOptions& = { });
42
43 virtual ~OscillatorNode();
44
45 const char* activeDOMObjectName() const final { return "OscillatorNode"; }
46
47 OscillatorType typeForBindings() const { ASSERT(isMainThread()); return m_type; }
48 ExceptionOr<void> setTypeForBindings(OscillatorType);
49
50 AudioParam* frequency() { return m_frequency.get(); }
51 AudioParam* detune() { return m_detune.get(); }
52
53 void setPeriodicWave(PeriodicWave&);
54
55protected:
56 explicit OscillatorNode(BaseAudioContext&, const OscillatorOptions& = { });
57
58private:
59 void process(size_t framesToProcess) final;
60
61 double tailTime() const final { return 0; }
62 double latencyTime() const final { return 0; }
63
64 // Returns true if there are sample-accurate timeline parameter changes.
65 bool calculateSampleAccuratePhaseIncrements(size_t framesToProcess) WTF_REQUIRES_LOCK(m_processLock);
66
67 double processARate(int, float* destP, double virtualReadIndex, float* phaseIncrements) WTF_REQUIRES_LOCK(m_processLock);
68 double processKRate(int, float* destP, double virtualReadIndex) WTF_REQUIRES_LOCK(m_processLock);
69
70 bool propagatesSilence() const final;
71
72 // One of the waveform types defined in the enum.
73 OscillatorType m_type; // Only used on the main thread.
74
75 // Frequency value in Hertz.
76 RefPtr<AudioParam> m_frequency;
77
78 // Detune value (deviating from the frequency) in Cents.
79 RefPtr<AudioParam> m_detune;
80
81 bool m_firstRender { true };
82
83 // m_virtualReadIndex is a sample-frame index into our buffer representing the current playback position.
84 // Since it's floating-point, it has sub-sample accuracy.
85 double m_virtualReadIndex { 0 };
86
87 // This synchronizes process().
88 mutable Lock m_processLock;
89
90 // Stores sample-accurate values calculated according to frequency and detune.
91 AudioFloatArray m_phaseIncrements;
92 AudioFloatArray m_detuneValues;
93
94 RefPtr<PeriodicWave> m_periodicWave WTF_GUARDED_BY_LOCK(m_processLock);
95};
96
