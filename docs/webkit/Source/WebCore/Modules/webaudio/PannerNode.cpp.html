53ExceptionOr<Ref<PannerNode>> PannerNode::create(BaseAudioContext& context, const PannerOptions& options)
54{
55 auto panner = adoptRef(*new PannerNode(context, options));
56
57 auto result = panner->handleAudioNodeOptions(options, { 2, ChannelCountMode::ClampedMax, ChannelInterpretation::Speakers });
58 if (result.hasException())
59 return result.releaseException();
60
61 result = panner->setMaxDistanceForBindings(options.maxDistance);
62 if (result.hasException())
63 return result.releaseException();
64
65 result = panner->setRefDistanceForBindings(options.refDistance);
66 if (result.hasException())
67 return result.releaseException();
68
69 result = panner->setRolloffFactorForBindings(options.rolloffFactor);
70 if (result.hasException())
71 return result.releaseException();
72
73 result = panner->setConeOuterGainForBindings(options.coneOuterGain);
74 if (result.hasException())
75 return result.releaseException();
76
77 return panner;
78}

80PannerNode::PannerNode(BaseAudioContext& context, const PannerOptions& options)
81 : AudioNode(context, NodeTypePanner)
82 // Load the HRTF database asynchronously so we don't block the Javascript thread while creating the HRTF database.
83 , m_hrtfDatabaseLoader(HRTFDatabaseLoader::createAndLoadAsynchronouslyIfNecessary(context.sampleRate()))
84 , m_panningModel(options.panningModel)
85 , m_panner(Panner::create(m_panningModel, sampleRate(), m_hrtfDatabaseLoader.ptr()))
86 , m_positionX(AudioParam::create(context, "positionX"_s, options.positionX, -FLT_MAX, FLT_MAX, AutomationRate::ARate))
87 , m_positionY(AudioParam::create(context, "positionY"_s, options.positionY, -FLT_MAX, FLT_MAX, AutomationRate::ARate))
88 , m_positionZ(AudioParam::create(context, "positionZ"_s, options.positionZ, -FLT_MAX, FLT_MAX, AutomationRate::ARate))
89 , m_orientationX(AudioParam::create(context, "orientationX"_s, options.orientationX, -FLT_MAX, FLT_MAX, AutomationRate::ARate))
90 , m_orientationY(AudioParam::create(context, "orientationY"_s, options.orientationY, -FLT_MAX, FLT_MAX, AutomationRate::ARate))
91 , m_orientationZ(AudioParam::create(context, "orientationZ"_s, options.orientationZ, -FLT_MAX, FLT_MAX, AutomationRate::ARate))
92{
93 setDistanceModelForBindings(options.distanceModel);
94 setConeInnerAngleForBindings(options.coneInnerAngle);
95 setConeOuterAngleForBindings(options.coneOuterAngle);
96
97 addInput();
98 addOutput(2);
99
100 initialize();
101}

108void PannerNode::process(size_t framesToProcess)
109{
110 AudioBus* destination = output(0)->bus();
111
112 if (!isInitialized() || !input(0)->isConnected()) {
113 destination->zero();
114 return;
115 }
116
117 AudioBus* source = input(0)->bus();
118 if (!source) {
119 destination->zero();
120 return;
121 }
122
123 // The audio thread can't block on this lock, so we use tryLock() instead.
124 if (!m_processLock.tryLock()) {
125 // Too bad - tryLock() failed. We must be in the middle of changing the panner.
126 destination->zero();
127 return;
128 }
129 Locker locker { AdoptLock, m_processLock };
130
131 if (!m_panner) {
132 destination->zero();
133 return;
134 }
135
136 // HRTFDatabase should be loaded before proceeding for offline audio context when m_panningModel is "HRTF".
137 if (m_panningModel == PanningModelType::HRTF && !m_hrtfDatabaseLoader->isLoaded()) {
138 if (context().isOfflineContext())
139 m_hrtfDatabaseLoader->waitForLoaderThreadCompletion();
140 else {
141 destination->zero();
142 return;
143 }
144 }
145
146 if ((hasSampleAccurateValues() || listener().hasSampleAccurateValues()) && (shouldUseARate() || listener().shouldUseARate())) {
147 processSampleAccurateValues(destination, source, framesToProcess);
148 return;
149 }
150
151 // Apply the panning effect.
152 double azimuth;
153 double elevation;
154 azimuthElevation(&azimuth, &elevation);
155 m_panner->pan(azimuth, elevation, source, destination, framesToProcess);
156
157 // Get the distance and cone gain.
158 double totalGain = distanceConeGain();
159
160 // Apply gain in-place.
161 destination->copyWithGainFrom(*destination, totalGain);
162}

164void PannerNode::processOnlyAudioParams(size_t framesToProcess)
165{
166 ASSERT(context().isAudioThread());
167 if (!m_processLock.tryLock())
168 return;
169
170 Locker locker { AdoptLock, m_processLock };
171 float values[AudioUtilities::renderQuantumSize];
172 ASSERT(framesToProcess <= AudioUtilities::renderQuantumSize);
173
174 m_positionX->calculateSampleAccurateValues(values, framesToProcess);
175 m_positionY->calculateSampleAccurateValues(values, framesToProcess);
176 m_positionZ->calculateSampleAccurateValues(values, framesToProcess);
177
178 m_orientationX->calculateSampleAccurateValues(values, framesToProcess);
179 m_orientationY->calculateSampleAccurateValues(values, framesToProcess);
180 m_orientationZ->calculateSampleAccurateValues(values, framesToProcess);
181
182 listener().updateValuesIfNeeded(framesToProcess);
183}

185void PannerNode::processSampleAccurateValues(AudioBus* destination, const AudioBus* source, size_t framesToProcess)
186{
187 // Get the sample accurate values from all of the AudioParams, including the
188 // values from the AudioListener.
189 float pannerX[AudioUtilities::renderQuantumSize];
190 float pannerY[AudioUtilities::renderQuantumSize];
191 float pannerZ[AudioUtilities::renderQuantumSize];
192
193 float orientationX[AudioUtilities::renderQuantumSize];
194 float orientationY[AudioUtilities::renderQuantumSize];
195 float orientationZ[AudioUtilities::renderQuantumSize];
196
197 m_positionX->calculateSampleAccurateValues(pannerX, framesToProcess);
198 m_positionY->calculateSampleAccurateValues(pannerY, framesToProcess);
199 m_positionZ->calculateSampleAccurateValues(pannerZ, framesToProcess);
200 m_orientationX->calculateSampleAccurateValues(orientationX, framesToProcess);
201 m_orientationY->calculateSampleAccurateValues(orientationY, framesToProcess);
202 m_orientationZ->calculateSampleAccurateValues(orientationZ, framesToProcess);
203
204 // Get the automation values from the listener.
205 const float* listenerX = listener().positionXValues(AudioUtilities::renderQuantumSize);
206 const float* listenerY = listener().positionYValues(AudioUtilities::renderQuantumSize);
207 const float* listenerZ = listener().positionZValues(AudioUtilities::renderQuantumSize);
208
209 const float* forwardX = listener().forwardXValues(AudioUtilities::renderQuantumSize);
210 const float* forwardY = listener().forwardYValues(AudioUtilities::renderQuantumSize);
211 const float* forwardZ = listener().forwardZValues(AudioUtilities::renderQuantumSize);
212
213 const float* upX = listener().upXValues(AudioUtilities::renderQuantumSize);
214 const float* upY = listener().upYValues(AudioUtilities::renderQuantumSize);
215 const float* upZ = listener().upZValues(AudioUtilities::renderQuantumSize);
216
217 // Compute the azimuth, elevation, and total gains for each position.
218 double azimuth[AudioUtilities::renderQuantumSize];
219 double elevation[AudioUtilities::renderQuantumSize];
220 float totalGain[AudioUtilities::renderQuantumSize];
221
222 for (size_t k = 0; k < framesToProcess; ++k) {
223 FloatPoint3D pannerPosition(pannerX[k], pannerY[k], pannerZ[k]);
224 FloatPoint3D orientation(orientationX[k], orientationY[k], orientationZ[k]);
225 FloatPoint3D listenerPosition(listenerX[k], listenerY[k], listenerZ[k]);
226 FloatPoint3D listenerFront(forwardX[k], forwardY[k], forwardZ[k]);
227 FloatPoint3D listenerUp(upX[k], upY[k], upZ[k]);
228
229 calculateAzimuthElevation(&azimuth[k], &elevation[k], pannerPosition, listenerPosition, listenerFront, listenerUp);
230
231 // Get distance and cone gain
232 totalGain[k] = calculateDistanceConeGain(pannerPosition, orientation, listenerPosition);
233 }
234
235 m_panner->panWithSampleAccurateValues(azimuth, elevation, source, destination, framesToProcess);
236 destination->copyWithSampleAccurateGainValuesFrom(*destination, totalGain, framesToProcess);
237}

442void PannerNode::calculateAzimuthElevation(double* outAzimuth, double* outElevation, const FloatPoint3D& position, const FloatPoint3D& listenerPosition, const FloatPoint3D& listenerFront, const FloatPoint3D& listenerUp)
443{
444 // FIXME: we should cache azimuth and elevation (if possible), so we only re-calculate if a change has been made.
445
446 // Calculate the source-listener vector
447 FloatPoint3D sourceListener = position - listenerPosition;
448
449 if (sourceListener.isZero()) {
450 // degenerate case if source and listener are at the same point
451 *outAzimuth = 0.0;
452 *outElevation = 0.0;
453 return;
454 }
455
456 sourceListener.normalize();
457
458 // Align axes
459 FloatPoint3D listenerRight = listenerFront.cross(listenerUp);
460 listenerRight.normalize();
461
462 FloatPoint3D listenerFrontNorm = listenerFront;
463 listenerFrontNorm.normalize();
464
465 FloatPoint3D up = listenerRight.cross(listenerFrontNorm);
466
467 float upProjection = sourceListener.dot(up);
468
469 FloatPoint3D projectedSource = sourceListener - upProjection * up;
470 projectedSource.normalize();
471
472 double azimuth = rad2deg(std::acos(std::clamp(projectedSource.dot(listenerRight), -1.0f, 1.0f)));
473 fixNANs(azimuth); // avoid illegal values
474
475 // Source in front or behind the listener
476 double frontBack = projectedSource.dot(listenerFrontNorm);
477 if (frontBack < 0.0)
478 azimuth = 360.0 - azimuth;
479
480 // Make azimuth relative to "front" and not "right" listener vector
481 if ((azimuth >= 0.0) && (azimuth <= 270.0))
482 azimuth = 90.0 - azimuth;
483 else
484 azimuth = 450.0 - azimuth;
485
486 // Elevation
487 double elevation = 90.0 - 180.0 * acos(sourceListener.dot(up)) / piDouble;
488 fixNANs(elevation); // avoid illegal values
489
490 if (elevation > 90.0)
491 elevation = 180.0 - elevation;
492 else if (elevation < -90.0)
493 elevation = -180.0 - elevation;
494
495 if (outAzimuth)
496 *outAzimuth = azimuth;
497 if (outElevation)
498 *outElevation = elevation;
499}

508bool PannerNode::requiresTailProcessing() const
509{
510 if (!m_processLock.tryLock())
511 return true;
512 Locker locker { AdoptLock, m_processLock };
513 // If there's no internal panner method set up yet, assume we require tail
514 // processing in case the HRTF panner is set later, which does require tail
515 // processing.
516 return !m_panner || m_panner->requiresTailProcessing();
517}

519float PannerNode::calculateDistanceConeGain(const FloatPoint3D& sourcePosition, const FloatPoint3D& orientation, const FloatPoint3D& listenerPosition)
520{
521 double listenerDistance = sourcePosition.distanceTo(listenerPosition);
522 double distanceGain = m_distanceEffect.gain(listenerDistance);
523
524 // FIXME: could optimize by caching coneGain
525 double coneGain = m_coneEffect.gain(sourcePosition, orientation, listenerPosition);
526
527 return float(distanceGain * coneGain);
528}

530float PannerNode::distanceConeGain()
531{
532 ASSERT(context().isAudioThread());
533
534 return calculateDistanceConeGain(position(), orientation(), listener().position());
535}
