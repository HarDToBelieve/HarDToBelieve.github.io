40// The number of bands per octave. Each octave will have this many entries in the wave tables.
41constexpr unsigned NumberOfOctaveBands = 3;
42
43// The max length of a periodic wave. This must be a power of two greater than
44// or equal to 2048 and must be supported by the FFT routines.
45constexpr unsigned MaxPeriodicWaveSize = 16384;
46
47constexpr float CentsPerRange = 1200 / NumberOfOctaveBands;

51Ref<PeriodicWave> PeriodicWave::create(float sampleRate, Float32Array& real, Float32Array& imaginary)
52{
53 ASSERT(real.length() == imaginary.length());
54
55 auto waveTable = adoptRef(*new PeriodicWave(sampleRate));
56 waveTable->createBandLimitedTables(real.data(), imaginary.data(), real.length());
57 return waveTable;
58}

60ExceptionOr<Ref<PeriodicWave>> PeriodicWave::create(BaseAudioContext& context, PeriodicWaveOptions&& options)
61{
62 Vector<float> real;
63 Vector<float> imag;
64
65 if (options.real && options.imag) {
66 if (options.real->size() != options.imag->size())
67 return Exception { IndexSizeError, "real and imag have different lengths"_s };
68 if (options.real->size() < 2)
69 return Exception { IndexSizeError, "real's length cannot be less than 2"_s };
70 if (options.imag->size() < 2)
71 return Exception { IndexSizeError, "imag's length cannot be less than 2"_s };
72 real = WTFMove(*options.real);
73 imag = WTFMove(*options.imag);
74 } else if (options.real) {
75 if (options.real->size() < 2)
76 return Exception { IndexSizeError, "real's length cannot be less than 2"_s };
77 real = WTFMove(*options.real);
78 imag.fill(0, real.size());
79 } else if (options.imag) {
80 if (options.imag->size() < 2)
81 return Exception { IndexSizeError, "imag's length cannot be less than 2"_s };
82 imag = WTFMove(*options.imag);
83 real.fill(0, imag.size());
84 } else {
85 real.fill(0, 2);
86 imag.fill(0, 2);
87 imag[1] = 1;
88 }
89
90 real[0] = 0;
91 imag[0] = 0;
92
93 auto waveTable = adoptRef(*new PeriodicWave(context.sampleRate()));
94 waveTable->createBandLimitedTables(real.data(), imag.data(), real.size(), options.disableNormalization ? ShouldDisableNormalization::Yes : ShouldDisableNormalization::No);
95 return waveTable;
96}

126PeriodicWave::PeriodicWave(float sampleRate)
127 : m_sampleRate(sampleRate)
128 , m_numberOfRanges(0.5 + NumberOfOctaveBands * log2f(periodicWaveSize()))
129{
130 float nyquist = 0.5 * m_sampleRate;
131 m_lowestFundamentalFrequency = nyquist / maxNumberOfPartials();
132 m_rateScale = periodicWaveSize() / m_sampleRate;
133}

135void PeriodicWave::waveDataForFundamentalFrequency(float fundamentalFrequency, float* &lowerWaveData, float* &higherWaveData, float& tableInterpolationFactor)
136{
137 // Negative frequencies are allowed, in which case we alias to the positive frequency.
138 fundamentalFrequency = fabsf(fundamentalFrequency);
139
140 // Calculate the pitch range.
141 float ratio = fundamentalFrequency > 0 ? fundamentalFrequency / m_lowestFundamentalFrequency : 0.5;
142 float centsAboveLowestFrequency = log2f(ratio) * 1200;
143
144 // Add one to round-up to the next range just in time to truncate partials before aliasing occurs.
145 float pitchRange = 1 + centsAboveLowestFrequency / CentsPerRange;
146
147 pitchRange = std::max(pitchRange, 0.0f);
148 pitchRange = std::min(pitchRange, static_cast<float>(m_numberOfRanges - 1));
149
150 // The words "lower" and "higher" refer to the table data having the lower and higher numbers of partials.
151 // It's a little confusing since the range index gets larger the more partials we cull out.
152 // So the lower table data will have a larger range index.
153 unsigned rangeIndex1 = static_cast<unsigned>(pitchRange);
154 unsigned rangeIndex2 = rangeIndex1 < m_numberOfRanges - 1 ? rangeIndex1 + 1 : rangeIndex1;
155
156 lowerWaveData = m_bandLimitedTables[rangeIndex2]->data();
157 higherWaveData = m_bandLimitedTables[rangeIndex1]->data();
158
159 // Ranges from 0 -> 1 to interpolate between lower -> higher.
160 tableInterpolationFactor = pitchRange - rangeIndex1;
161}

163unsigned PeriodicWave::maxNumberOfPartials() const
164{
165 return periodicWaveSize() / 2;
166}
167
168unsigned PeriodicWave::numberOfPartialsForRange(unsigned rangeIndex) const
169{
170 // Number of cents below nyquist where we cull partials.
171 float centsToCull = rangeIndex * CentsPerRange;
172
173 // A value from 0 -> 1 representing what fraction of the partials to keep.
174 float cullingScale = pow(2, -centsToCull / 1200);
175
176 // The very top range will have all the partials culled.
177 unsigned numberOfPartials = cullingScale * maxNumberOfPartials();
178
179 return numberOfPartials;
180}

182// Convert into time-domain wave tables.
183// One table is created for each range for non-aliasing playback at different playback rates.
184// Thus, higher ranges have more high-frequency partials culled out.
185void PeriodicWave::createBandLimitedTables(const float* realData, const float* imagData, unsigned numberOfComponents, ShouldDisableNormalization disableNormalization)
186{
187 float normalizationScale = 0.5;
188
189 unsigned fftSize = periodicWaveSize();
190 unsigned halfSize = fftSize / 2;
191
192 numberOfComponents = std::min(numberOfComponents, halfSize);
193
194 m_bandLimitedTables.reserveCapacity(m_numberOfRanges);
195
196 for (unsigned rangeIndex = 0; rangeIndex < m_numberOfRanges; ++rangeIndex) {
197 // This FFTFrame is used to cull partials (represented by frequency bins).
198 FFTFrame frame(fftSize);
199 auto& realP = frame.realData();
200 auto& imagP = frame.imagData();
201
202 RELEASE_ASSERT(realP.size() >= numberOfComponents);
203 RELEASE_ASSERT(imagP.size() >= numberOfComponents);
204
205 // Copy from loaded frequency data and scale.
206 VectorMath::multiplyByScalar(realData, fftSize, realP.data(), numberOfComponents);
207 VectorMath::multiplyByScalar(imagData, -static_cast<float>(fftSize), imagP.data(), numberOfComponents);
208
209 // Find the starting bin where we should start culling.
210 // We need to clear out the highest frequencies to band-limit the waveform.
211 unsigned numberOfPartials = numberOfPartialsForRange(rangeIndex);
212
213 // If fewer components were provided than 1/2 FFT size, then clear the
214 // remaining bins. We also need to cull the aliasing partials for this
215 // pitch range.
216 unsigned clampedNumberOfComponents = std::min(numberOfComponents, numberOfPartials + 1);
217 if (clampedNumberOfComponents < halfSize) {
218 size_t numBytes = (halfSize - clampedNumberOfComponents) * sizeof(float);
219 memset(&realP[clampedNumberOfComponents], 0, numBytes);
220 memset(&imagP[clampedNumberOfComponents], 0, numBytes);
221 }
222
223 // Clear packed-nyquist and any DC-offset.
224 realP[0] = 0;
225 imagP[0] = 0;
226
227 // Create the band-limited table.
228 unsigned waveSize = periodicWaveSize();
229 m_bandLimitedTables.append(makeUnique<AudioFloatArray>(waveSize));
230
231 // Apply an inverse FFT to generate the time-domain table data.
232 float* data = m_bandLimitedTables[rangeIndex]->data();
233 frame.doInverseFFT(data);
234
235 // For the first range (which has the highest power), calculate its peak value then compute normalization scale.
236 if (disableNormalization == ShouldDisableNormalization::No) {
237 if (!rangeIndex) {
238 float maxValue = VectorMath::maximumMagnitude(data, fftSize);
239
240 if (maxValue)
241 normalizationScale = 1.0f / maxValue;
242 }
243 }
244
245 // Apply normalization scale.
246 VectorMath::multiplyByScalar(data, normalizationScale, data, fftSize);
247 }

250void PeriodicWave::generateBasicWaveform(Type shape)
251{
252 unsigned fftSize = periodicWaveSize();
253 unsigned halfSize = fftSize / 2;
254
255 AudioFloatArray real(halfSize);
256 AudioFloatArray imag(halfSize);
257 float* realP = real.data();
258 float* imagP = imag.data();
259
260 // Clear DC and Nyquist.
261 realP[0] = 0;
262 imagP[0] = 0;
263
264 for (unsigned n = 1; n < halfSize; ++n) {
265 float piFactor = 2 / (n * piFloat);
266
267 // All waveforms are odd functions with a positive slope at time 0. Hence
268 // the coefficients for cos() are always 0.
269
270 // Fourier coefficients according to standard definition:
271 // b = 1/pi*integrate(f(x)*sin(n*x), x, -pi, pi)
272 // = 2/pi*integrate(f(x)*sin(n*x), x, 0, pi)
273 // since f(x) is an odd function.
274
275 float b; // Coefficient for sin().
276
277 // Calculate Fourier coefficients depending on the shape.
278 // Note that the overall scaling (magnitude) of the waveforms is normalized in createBandLimitedTables().
279 switch (shape) {
280 case Type::Sine:
281 // Standard sine wave function.
282 b = (n == 1) ? 1 : 0;
283 break;
284 case Type::Square:
285 // Square-shaped waveform with the first half its maximum value and the
286 // second half its minimum value.
287 //
288 // See http://mathworld.wolfram.com/FourierSeriesSquareWave.html
289 //
290 // b[n] = 2/n/pi*(1-(-1)^n)
291 // = 4/n/pi for n odd and 0 otherwise.
292 // = 2*(2/(n*pi)) for n odd
293 b = (n & 1) ? 2 * piFactor : 0;
294 break;
295 case Type::Sawtooth:
296 // Sawtooth-shaped waveform with the first half ramping from zero to
297 // maximum and the second half from minimum to zero.
298 //
299 // b[n] = -2*(-1)^n/pi/n
300 // = (2/(n*pi))*(-1)^(n+1)
301 b = piFactor * ((n & 1) ? 1 : -1);
302 break;
303 case Type::Triangle:
304 // Triangle-shaped waveform going from 0 at time 0 to 1 at time pi/2 and
305 // back to 0 at time pi.
306 //
307 // See http://mathworld.wolfram.com/FourierSeriesTriangleWave.html
308 //
309 // b[n] = 8*sin(pi*k/2)/(pi*k)^2
310 // = 8/pi^2/n^2*(-1)^((n-1)/2) for n odd and 0 otherwise
311 // = 2*(2/(n*pi))^2 * (-1)^((n-1)/2)
312 if (n & 1)
313 b = 2 * (piFactor * piFactor) * ((((n - 1) >> 1) & 1) ? -1 : 1);
314 else
315 b = 0;
316 break;
317 }
318
319 realP[n] = 0;
320 imagP[n] = b;
321 }
322
323 createBandLimitedTables(realP, imagP, halfSize);
324}

326unsigned PeriodicWave::periodicWaveSize() const
327{
328 // Choose an appropriate wave size for the given sample rate. This allows us
329 // to use shorter FFTs when possible to limit the complexity. The breakpoints
330 // here are somewhat arbitrary, but we want sample rates around 44.1 kHz or so
331 // to have a size of 4096 to preserve backward compatibility.
332 if (m_sampleRate <= 24000)
333 return 2048;
334
335 if (m_sampleRate <= 88200)
336 return 4096;
337
338 return MaxPeriodicWaveSize;
339}
