54bool RealtimeAnalyser::setFftSize(size_t size)
55{
56 ASSERT(isMainThread());
57
58 // Only allow powers of two.
59 unsigned log2size = static_cast<unsigned>(log2(size));
60 bool isPOT(1UL << log2size == size);
61
62 if (!isPOT || size > MaxFFTSize || size < MinFFTSize)
63 return false;
64
65 if (m_fftSize != size) {
66 m_analysisFrame = makeUnique<FFTFrame>(size);
67 // m_magnitudeBuffer has size = fftSize / 2 because it contains floats reduced from complex values in m_analysisFrame.
68 m_magnitudeBuffer.resize(size / 2);
69 m_fftSize = size;
70 }
71
72 return true;
73}

75void RealtimeAnalyser::writeInput(AudioBus* bus, size_t framesToProcess)
76{
77 bool isBusGood = bus && bus->numberOfChannels() > 0 && bus->channel(0)->length() >= framesToProcess;
78 ASSERT(isBusGood);
79 if (!isBusGood)
80 return;
81
82 // FIXME : allow to work with non-FFTSize divisible chunking
83 bool isDestinationGood = m_writeIndex < m_inputBuffer.size() && m_writeIndex + framesToProcess <= m_inputBuffer.size();
84 ASSERT(isDestinationGood);
85 if (!isDestinationGood)
86 return;
87
88 // Perform real-time analysis
89 float* dest = m_inputBuffer.data() + m_writeIndex;
90
91 // Clear the bus and downmix the input according to the down mixing rules.
92 // Then save the result in the m_inputBuffer at the appropriate place.
93 m_downmixBus->zero();
94 m_downmixBus->sumFrom(*bus);
95 memcpy(dest, m_downmixBus->channel(0)->data(), sizeof(float) * framesToProcess);
96
97 m_writeIndex += framesToProcess;
98 if (m_writeIndex >= InputBufferSize)
99 m_writeIndex = 0;
100
101 // A new render quantum has been processed so we should do the FFT analysis again.
102 m_shouldDoFFTAnalysis = true;
103}

107void applyWindow(float* p, size_t n)
108{
109 ASSERT(isMainThread());
110
111 // Blackman window
112 double alpha = 0.16;
113 double a0 = 0.5 * (1 - alpha);
114 double a1 = 0.5;
115 double a2 = 0.5 * alpha;
116
117 for (unsigned i = 0; i < n; ++i) {
118 double x = static_cast<double>(i) / static_cast<double>(n);
119 double window = a0 - a1 * cos(2 * piDouble * x) + a2 * cos(4 * piDouble * x);
120 p[i] *= float(window);
121 }
122}

126void RealtimeAnalyser::doFFTAnalysisIfNecessary()
127{
128 ASSERT(isMainThread());
129
130 if (!m_shouldDoFFTAnalysis)
131 return;
132
133 m_shouldDoFFTAnalysis = false;
134
135 // Unroll the input buffer into a temporary buffer, where we'll apply an analysis window followed by an FFT.
136 size_t fftSize = this->fftSize();
137
138 AudioFloatArray temporaryBuffer(fftSize);
139 float* inputBuffer = m_inputBuffer.data();
140 float* tempP = temporaryBuffer.data();
141
142 // Take the previous fftSize values from the input buffer and copy into the temporary buffer.
143 unsigned writeIndex = m_writeIndex;
144 if (writeIndex < fftSize) {
145 memcpy(tempP, inputBuffer + writeIndex - fftSize + InputBufferSize, sizeof(*tempP) * (fftSize - writeIndex));
146 memcpy(tempP + fftSize - writeIndex, inputBuffer, sizeof(*tempP) * writeIndex);
147 } else
148 memcpy(tempP, inputBuffer + writeIndex - fftSize, sizeof(*tempP) * fftSize);
149
150
151 // Window the input samples.
152 applyWindow(tempP, fftSize);
153
154 // Do the analysis.
155 m_analysisFrame->doFFT(tempP);
156
157 auto& realP = m_analysisFrame->realData();
158 auto& imagP = m_analysisFrame->imagData();
159
160 // Blow away the packed nyquist component.
161 imagP[0] = 0;
162
163 // Normalize so than an input sine wave at 0dBfs registers as 0dBfs (undo FFT scaling factor).
164 const double magnitudeScale = 1.0 / fftSize;
165
166 // A value of 0 does no averaging with the previous result. Larger values produce slower, but smoother changes.
167 double k = m_smoothingTimeConstant;
168 k = std::max(0.0, k);
169 k = std::min(1.0, k);
170
171 // Convert the analysis data from complex to magnitude and average with the previous result.
172 float* destination = magnitudeBuffer().data();
173 size_t n = magnitudeBuffer().size();
174 for (size_t i = 0; i < n; ++i) {
175 std::complex<double> c(realP[i], imagP[i]);
176 double scalarMagnitude = abs(c) * magnitudeScale;
177 destination[i] = static_cast<float>(k * destination[i] + (1 - k) * scalarMagnitude);
178 }
179}

181void RealtimeAnalyser::getFloatFrequencyData(Float32Array& destinationArray)
182{
183 ASSERT(isMainThread());
184
185 doFFTAnalysisIfNecessary();
186
187 // Convert from linear magnitude to floating-point decibels.
188 size_t length = std::min<size_t>(magnitudeBuffer().size(), destinationArray.length());
189 VectorMath::linearToDecibels(magnitudeBuffer().data(), destinationArray.data(), length);
190}

192void RealtimeAnalyser::getByteFrequencyData(Uint8Array& destinationArray)
193{
194 ASSERT(isMainThread());
195
196 doFFTAnalysisIfNecessary();
197
198 // Convert from linear magnitude to unsigned-byte decibels.
199 unsigned sourceLength = magnitudeBuffer().size();
200 size_t length = std::min(sourceLength, destinationArray.length());
201 if (length > 0) {
202 const double rangeScaleFactor = m_maxDecibels == m_minDecibels ? 1 : 1 / (m_maxDecibels - m_minDecibels);
203 const double minDecibels = m_minDecibels;
204
205 const float* source = magnitudeBuffer().data();
206 unsigned char* destination = destinationArray.data();
207
208 for (size_t i = 0; i < length; ++i) {
209 float linearValue = source[i];
210 double dbMag = !linearValue ? minDecibels : AudioUtilities::linearToDecibels(linearValue);
211
212 // The range m_minDecibels to m_maxDecibels will be scaled to byte values from 0 to UCHAR_MAX.
213 double scaledValue = UCHAR_MAX * (dbMag - minDecibels) * rangeScaleFactor;
214
215 // Clip to valid range.
216 if (scaledValue < 0)
217 scaledValue = 0;
218 if (scaledValue > UCHAR_MAX)
219 scaledValue = UCHAR_MAX;
220
221 destination[i] = static_cast<unsigned char>(scaledValue);
222 }
223 }

226void RealtimeAnalyser::getFloatTimeDomainData(Float32Array& destinationArray)
227{
228 ASSERT(isMainThread());
229
230 unsigned fftSize = this->fftSize();
231 size_t length = std::min(fftSize, destinationArray.length());
232 if (length > 0) {
233 bool isInputBufferGood = m_inputBuffer.size() == InputBufferSize && m_inputBuffer.size() > fftSize;
234 ASSERT(isInputBufferGood);
235 if (!isInputBufferGood)
236 return;
237
238 float* inputBuffer = m_inputBuffer.data();
239 float* destination = destinationArray.data();
240
241 unsigned writeIndex = m_writeIndex;
242
243 for (size_t i = 0; i < length; ++i) {
244 // Buffer access is protected due to modulo operation.
245 destination[i] = inputBuffer[(i + writeIndex - fftSize + InputBufferSize) % InputBufferSize];
246 }
247 }

250void RealtimeAnalyser::getByteTimeDomainData(Uint8Array& destinationArray)
251{
252 ASSERT(isMainThread());
253
254 unsigned fftSize = this->fftSize();
255 size_t length = std::min(fftSize, destinationArray.length());
256 if (length > 0) {
257 bool isInputBufferGood = m_inputBuffer.size() == InputBufferSize && m_inputBuffer.size() > fftSize;
258 ASSERT(isInputBufferGood);
259 if (!isInputBufferGood)
260 return;
261
262 float* inputBuffer = m_inputBuffer.data();
263 unsigned char* destination = destinationArray.data();
264
265 unsigned writeIndex = m_writeIndex;
266
267 for (size_t i = 0; i < length; ++i) {
268 // Buffer access is protected due to modulo operation.
269 float value = inputBuffer[(i + writeIndex - fftSize + InputBufferSize) % InputBufferSize];
270
271 // Scale from nominal -1 -> +1 to unsigned byte.
272 double scaledValue = 128 * (value + 1);
273
274 // Clip to valid range.
275 if (scaledValue < 0)
276 scaledValue = 0;
277 if (scaledValue > UCHAR_MAX)
278 scaledValue = UCHAR_MAX;
279
280 destination[i] = static_cast<unsigned char>(scaledValue);
281 }
282 }
