53ScriptProcessorNode::ScriptProcessorNode(BaseAudioContext& context, size_t bufferSize, unsigned numberOfInputChannels, unsigned numberOfOutputChannels)
54 : AudioNode(context, NodeTypeJavaScript)
55 , ActiveDOMObject(context.scriptExecutionContext())
56 , m_bufferSize(bufferSize)
57 , m_numberOfInputChannels(numberOfInputChannels)
58 , m_numberOfOutputChannels(numberOfOutputChannels)
59 , m_internalInputBus(AudioBus::create(numberOfInputChannels, AudioUtilities::renderQuantumSize, false))
60{
61 // Regardless of the allowed buffer sizes, we still need to process at the granularity of the AudioNode.
62 if (m_bufferSize < AudioUtilities::renderQuantumSize)
63 m_bufferSize = AudioUtilities::renderQuantumSize;
64
65 ASSERT(numberOfInputChannels <= AudioContext::maxNumberOfChannels);
66
67 initializeDefaultNodeOptions(numberOfInputChannels, ChannelCountMode::Explicit, ChannelInterpretation::Speakers);
68 addInput();
69 addOutput(numberOfOutputChannels);
70
71 initialize();
72 suspendIfNeeded();
73}

75ScriptProcessorNode::~ScriptProcessorNode()

81void ScriptProcessorNode::initialize()
82{
83 if (isInitialized())
84 return;
85
86 float sampleRate = context().sampleRate();
87
88 // Create double buffers on both the input and output sides.
89 // These AudioBuffers will be directly accessed in the main thread by JavaScript.
90 for (unsigned i = 0; i < bufferCount; ++i) {
91 // We prevent detaching the AudioBuffers here since we pass those to JS and reuse them.
92 m_inputBuffers[i] = m_numberOfInputChannels ? AudioBuffer::create(m_numberOfInputChannels, bufferSize(), sampleRate, AudioBuffer::LegacyPreventDetaching::Yes) : 0;
93 m_outputBuffers[i] = m_numberOfOutputChannels ? AudioBuffer::create(m_numberOfOutputChannels, bufferSize(), sampleRate, AudioBuffer::LegacyPreventDetaching::Yes) : 0;
94 }
95
96 AudioNode::initialize();
97}

99RefPtr<AudioBuffer> ScriptProcessorNode::createInputBufferForJS(AudioBuffer* inputBuffer) const
100{
101 if (!inputBuffer)
102 return nullptr;
103
104 // As an optimization, we reuse the same buffer as last time when possible.
105 if (!m_cachedInputBufferForJS || !inputBuffer->copyTo(*m_cachedInputBufferForJS))
106 m_cachedInputBufferForJS = inputBuffer->clone();
107
108 return m_cachedInputBufferForJS;
109}
110
111RefPtr<AudioBuffer> ScriptProcessorNode::createOutputBufferForJS(AudioBuffer& outputBuffer) const
112{
113 // As an optimization, we reuse the same buffer as last time when possible.
114 if (!m_cachedOutputBufferForJS || !m_cachedOutputBufferForJS->topologyMatches(outputBuffer))
115 m_cachedOutputBufferForJS = outputBuffer.clone(AudioBuffer::ShouldCopyChannelData::No);
116 else
117 m_cachedOutputBufferForJS->zero();
118
119 return m_cachedOutputBufferForJS;
120}

122void ScriptProcessorNode::uninitialize()
123{
124 if (!isInitialized())
125 return;
126
127 for (unsigned i = 0; i < bufferCount; ++i) {
128 Locker locker { m_bufferLocks[i] };
129 m_inputBuffers[i] = nullptr;
130 m_outputBuffers[i] = nullptr;
131 }
132
133 AudioNode::uninitialize();
134}

136void ScriptProcessorNode::process(size_t framesToProcess)
137{
138 // Discussion about inputs and outputs:
139 // As in other AudioNodes, ScriptProcessorNode uses an AudioBus for its input and output (see inputBus and outputBus below).
140 // Additionally, there is a double-buffering for input and output (see inputBuffer and outputBuffer below).
141 // This node is the producer for inputBuffer and the consumer for outputBuffer.
142 // The JavaScript code is the consumer of inputBuffer and the producer for outputBuffer. The JavaScript gets its own copy
143 // of the buffers for safety reasons.
144
145 // Get input and output busses.
146 AudioBus* inputBus = this->input(0)->bus();
147 AudioBus* outputBus = this->output(0)->bus();
148
149 // Get input and output buffers. We double-buffer both the input and output sides.
150 unsigned bufferIndex = this->bufferIndex();
151 ASSERT(bufferIndex < bufferCount);
152
153 if (!m_bufferLocks[bufferIndex].tryLock()) {
154 // We're late in handling the previous request. The main thread must be
155 // very busy. The best we can do is clear out the buffer ourself here.
156 outputBus->zero();
157 return;
158 }
159 Locker locker { AdoptLock, m_bufferLocks[bufferIndex] };
160
161 AudioBuffer* inputBuffer = m_inputBuffers[bufferIndex].get();
162 AudioBuffer* outputBuffer = m_outputBuffers[bufferIndex].get();
163
164 // Check the consistency of input and output buffers.
165 unsigned numberOfInputChannels = m_internalInputBus->numberOfChannels();
166 bool buffersAreGood = outputBuffer && bufferSize() == outputBuffer->length() && m_bufferReadWriteIndex + framesToProcess <= bufferSize();
167
168 // If the number of input channels is zero, it's ok to have inputBuffer = 0.
169 if (m_internalInputBus->numberOfChannels())
170 buffersAreGood = buffersAreGood && inputBuffer && bufferSize() == inputBuffer->length();
171
172 ASSERT(buffersAreGood);
173 if (!buffersAreGood)
174 return;
175
176 // We assume that bufferSize() is evenly divisible by framesToProcess - should always be true, but we should still check.
177 bool isFramesToProcessGood = framesToProcess && bufferSize() >= framesToProcess && !(bufferSize() % framesToProcess);
178 ASSERT(isFramesToProcessGood);
179 if (!isFramesToProcessGood)
180 return;
181
182 unsigned numberOfOutputChannels = outputBus->numberOfChannels();
183
184 bool channelsAreGood = (numberOfInputChannels == m_numberOfInputChannels) && (numberOfOutputChannels == m_numberOfOutputChannels);
185 ASSERT(channelsAreGood);
186 if (!channelsAreGood)
187 return;
188
189 for (unsigned i = 0; i < numberOfInputChannels; i++)
190 m_internalInputBus->setChannelMemory(i, inputBuffer->rawChannelData(i) + m_bufferReadWriteIndex, framesToProcess);
191
192 if (numberOfInputChannels)
193 m_internalInputBus->copyFrom(*inputBus);
194
195 // Copy from the output buffer to the output.
196 for (unsigned i = 0; i < numberOfOutputChannels; ++i)
197 memcpy(outputBus->channel(i)->mutableData(), outputBuffer->rawChannelData(i) + m_bufferReadWriteIndex, sizeof(float) * framesToProcess);
198
199 // Update the buffering index.
200 m_bufferReadWriteIndex = (m_bufferReadWriteIndex + framesToProcess) % bufferSize();
201
202 // m_bufferReadWriteIndex will wrap back around to 0 when the current input and output buffers are full.
203 // When this happens, fire an event and swap buffers.
204 if (!m_bufferReadWriteIndex) {
205 // Heap allocations are forbidden on the audio thread for performance reasons so we need to
206 // explicitly allow the following allocation(s).
207 DisableMallocRestrictionsForCurrentThreadScope disableMallocRestrictions;
208
209 // Reference ourself so we don't accidentally get deleted before fireProcessEvent() gets called.
210 // We only wait for script code execution when the context is an offline one for performance reasons.
211 if (context().isOfflineContext()) {
212 callOnMainThreadAndWait([this, bufferIndex, protector = makeRef(*this)] {
213 fireProcessEvent(bufferIndex);
214 });
215 } else {
216 callOnMainThread([this, bufferIndex, protector = makeRef(*this)] {
217 Locker locker { m_bufferLocks[bufferIndex] };
218 fireProcessEvent(bufferIndex);
219 });
220 }
221
222 swapBuffers();
223 }
224}

226void ScriptProcessorNode::fireProcessEvent(unsigned bufferIndex)
227{
228 ASSERT(isMainThread());
229
230 AudioBuffer* inputBuffer = m_inputBuffers[bufferIndex].get();
231 AudioBuffer* outputBuffer = m_outputBuffers[bufferIndex].get();
232 ASSERT(outputBuffer);
233 if (!outputBuffer)
234 return;
235
236 // Avoid firing the event if the document has already gone away.
237 if (context().isStopped())
238 return;
239
240 // Calculate playbackTime with the buffersize which needs to be processed each time when onaudioprocess is called.
241 // The outputBuffer being passed to JS will be played after exhausting previous outputBuffer by double-buffering.
242 double playbackTime = (context().currentSampleFrame() + m_bufferSize) / static_cast<double>(context().sampleRate());
243
244 auto inputBufferForJS = createInputBufferForJS(inputBuffer);
245 auto outputBufferForJS = createOutputBufferForJS(*outputBuffer);
246
247 // Call the JavaScript event handler which will do the audio processing.
248 dispatchEvent(AudioProcessingEvent::create(inputBufferForJS.get(), outputBufferForJS.get(), playbackTime));
249
250 if (!outputBufferForJS->copyTo(*outputBuffer))
251 outputBuffer->zero();
252}
