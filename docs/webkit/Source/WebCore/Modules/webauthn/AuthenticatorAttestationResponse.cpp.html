35Ref<AuthenticatorAttestationResponse> AuthenticatorAttestationResponse::create(Ref<ArrayBuffer>&& rawId, Ref<ArrayBuffer>&& attestationObject, AuthenticatorAttachment attachment)
36{
37 return adoptRef(*new AuthenticatorAttestationResponse(WTFMove(rawId), WTFMove(attestationObject), attachment));
38}
39
40Ref<AuthenticatorAttestationResponse> AuthenticatorAttestationResponse::create(const Vector<uint8_t>& rawId, const Vector<uint8_t>& attestationObject, AuthenticatorAttachment attachment)
41{
42 return create(ArrayBuffer::create(rawId.data(), rawId.size()), ArrayBuffer::create(attestationObject.data(), attestationObject.size()), attachment);
43}
44
45AuthenticatorAttestationResponse::AuthenticatorAttestationResponse(Ref<ArrayBuffer>&& rawId, Ref<ArrayBuffer>&& attestationObject, AuthenticatorAttachment attachment)
46 : AuthenticatorResponse(WTFMove(rawId), attachment)
47 , m_attestationObject(WTFMove(attestationObject))
48{
49}
50
51AuthenticatorResponseData AuthenticatorAttestationResponse::data() const
52{
53 auto data = AuthenticatorResponse::data();
54 data.isAuthenticatorAttestationResponse = true;
55 data.attestationObject = m_attestationObject.copyRef();
56 return data;
57}
