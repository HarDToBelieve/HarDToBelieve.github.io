30#include <JavaScriptCore/ArrayBuffer.h>
31#include <wtf/Forward.h>

33namespace WebCore {
34
35class AuthenticatorResponse;
36
37struct AuthenticatorResponseData {
38 bool isAuthenticatorAttestationResponse;
39
40 // AuthenticatorResponse
41 RefPtr<ArrayBuffer> rawId;
42
43 // Extensions
44 std::optional<bool> appid;
45
46 // AuthenticatorAttestationResponse
47 RefPtr<ArrayBuffer> attestationObject;
48
49 // AuthenticatorAssertionResponse
50 RefPtr<ArrayBuffer> authenticatorData;
51 RefPtr<ArrayBuffer> signature;
52 RefPtr<ArrayBuffer> userHandle;
53
54 template<class Encoder> void encode(Encoder&) const;
55 template<class Decoder> static std::optional<AuthenticatorResponseData> decode(Decoder&);
56};
57
58template<class Encoder>
59static void encodeArrayBuffer(Encoder& encoder, const ArrayBuffer& buffer)
60{
61 encoder << static_cast<uint64_t>(buffer.byteLength());
62 encoder.encodeFixedLengthData(static_cast<const uint8_t*>(buffer.data()), buffer.byteLength(), 1);
63}
64
65template<class Decoder>
66RefPtr<ArrayBuffer> decodeArrayBuffer(Decoder& decoder)
67{
68 std::optional<uint64_t> length;
69 decoder >> length;
70 if (!length)
71 return nullptr;
72
73 if (!decoder.template bufferIsLargeEnoughToContain<uint8_t>(length.value()))
74 return nullptr;
75 auto buffer = ArrayBuffer::tryCreate(length.value(), sizeof(uint8_t));
76 if (!buffer)
77 return nullptr;
78 if (!decoder.decodeFixedLengthData(static_cast<uint8_t*>(buffer->data()), length.value(), 1))
79 return nullptr;
80 return buffer;
81}
82
83template<class Encoder>
84void AuthenticatorResponseData::encode(Encoder& encoder) const
85{
86 if (!rawId) {
87 encoder << true;
88 return;
89 }
90 encoder << false;
91 encodeArrayBuffer(encoder, *rawId);
92
93 encoder << isAuthenticatorAttestationResponse;
94
95 if (isAuthenticatorAttestationResponse && attestationObject) {
96 encodeArrayBuffer(encoder, *attestationObject);
97 return;
98 }
99
100 if (!authenticatorData || !signature)
101 return;
102 encodeArrayBuffer(encoder, *authenticatorData);
103 encodeArrayBuffer(encoder, *signature);
104
105 // Encode AppID before user handle to avoid the userHandle flag.
106 encoder << appid;
107
108 if (!userHandle) {
109 encoder << false;
110 return;
111 }
112 encoder << true;
113 encodeArrayBuffer(encoder, *userHandle);
114}
115
116template<class Decoder>
117std::optional<AuthenticatorResponseData> AuthenticatorResponseData::decode(Decoder& decoder)
118{
119 AuthenticatorResponseData result;
120
121 std::optional<bool> isEmpty;
122 decoder >> isEmpty;
123 if (!isEmpty)
124 return std::nullopt;
125 if (isEmpty.value())
126 return result;
127
128 result.rawId = decodeArrayBuffer(decoder);
129 if (!result.rawId)
130 return std::nullopt;
131
132 std::optional<bool> isAuthenticatorAttestationResponse;
133 decoder >> isAuthenticatorAttestationResponse;
134 if (!isAuthenticatorAttestationResponse)
135 return std::nullopt;
136 result.isAuthenticatorAttestationResponse = isAuthenticatorAttestationResponse.value();
137
138 if (result.isAuthenticatorAttestationResponse) {
139 result.attestationObject = decodeArrayBuffer(decoder);
140 if (!result.attestationObject)
141 return std::nullopt;
142 return result;
143 }
144
145 result.authenticatorData = decodeArrayBuffer(decoder);
146 if (!result.authenticatorData)
147 return std::nullopt;
148
149 result.signature = decodeArrayBuffer(decoder);
150 if (!result.signature)
151 return std::nullopt;
152
153 std::optional<std::optional<bool>> appid;
154 decoder >> appid;
155 if (!appid)
156 return std::nullopt;
157 result.appid = WTFMove(*appid);
158
159 std::optional<bool> hasUserHandle;
160 decoder >> hasUserHandle;
161 if (!hasUserHandle)
162 return std::nullopt;
163 if (!*hasUserHandle)
164 return result;
165
166 result.userHandle = decodeArrayBuffer(decoder);
167 if (!result.userHandle)
168 return std::nullopt;
169
170 return result;
171}
