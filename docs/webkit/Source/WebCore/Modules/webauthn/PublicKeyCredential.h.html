34namespace WebCore {
35
36class AuthenticatorResponse;
37class Document;
38
39struct AuthenticationExtensionsClientOutputs;
40
41template<typename IDLType> class DOMPromiseDeferred;
42
43class PublicKeyCredential final : public BasicCredential {
44public:
45 static Ref<PublicKeyCredential> create(Ref<AuthenticatorResponse>&&);
46
47 ArrayBuffer* rawId() const;
48 AuthenticatorResponse* response() const { return m_response.ptr(); }
49 AuthenticationExtensionsClientOutputs getClientExtensionResults() const;
50
51 static void isUserVerifyingPlatformAuthenticatorAvailable(Document&, DOMPromiseDeferred<IDLBoolean>&&);
52
53private:
54 PublicKeyCredential(Ref<AuthenticatorResponse>&&);
55
56 Type credentialType() const final { return Type::PublicKey; }
57
58 Ref<AuthenticatorResponse> m_response;
59};

61} // namespace WebCore
