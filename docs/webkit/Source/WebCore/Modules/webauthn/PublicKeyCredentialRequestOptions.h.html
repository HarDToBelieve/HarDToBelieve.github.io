27
28#if ENABLE(WEB_AUTHN)
29#include "AuthenticationExtensionsClientInputs.h"
30#include "BufferSource.h"
31#include "PublicKeyCredentialDescriptor.h"
32#include "UserVerificationRequirement.h"
33#include <wtf/Forward.h>
34#endif // ENABLE(WEB_AUTHN)
35
36namespace WebCore {
37
38struct PublicKeyCredentialRequestOptions {
39#if ENABLE(WEB_AUTHN)
40 BufferSource challenge;
41 std::optional<unsigned> timeout;
42 mutable String rpId;
43 Vector<PublicKeyCredentialDescriptor> allowCredentials;
44 UserVerificationRequirement userVerification { UserVerificationRequirement::Preferred };
45 std::optional<AuthenticatorAttachment> authenticatorAttachment;
46 mutable std::optional<AuthenticationExtensionsClientInputs> extensions;
47
48 template<class Encoder> void encode(Encoder&) const;
49 template<class Decoder> static std::optional<PublicKeyCredentialRequestOptions> decode(Decoder&);
50#endif // ENABLE(WEB_AUTHN)
51};
52
53#if ENABLE(WEB_AUTHN)
54// Not every member is encoded.
55template<class Encoder>
56void PublicKeyCredentialRequestOptions::encode(Encoder& encoder) const
57{
58 encoder << timeout << rpId << allowCredentials << userVerification << extensions;
59}
60
61template<class Decoder>
62std::optional<PublicKeyCredentialRequestOptions> PublicKeyCredentialRequestOptions::decode(Decoder& decoder)
63{
64 PublicKeyCredentialRequestOptions result;
65
66 std::optional<std::optional<unsigned>> timeout;
67 decoder >> timeout;
68 if (!timeout)
69 return std::nullopt;
70 result.timeout = WTFMove(*timeout);
71
72 if (!decoder.decode(result.rpId))
73 return std::nullopt;
74 if (!decoder.decode(result.allowCredentials))
75 return std::nullopt;
76
77 std::optional<UserVerificationRequirement> userVerification;
78 decoder >> userVerification;
79 if (!userVerification)
80 return std::nullopt;
81 result.userVerification = WTFMove(*userVerification);
82
83 std::optional<std::optional<AuthenticationExtensionsClientInputs>> extensions;
84 decoder >> extensions;
85 if (!extensions)
86 return std::nullopt;
87 result.extensions = WTFMove(*extensions);
88
89 return result;
90}
91#endif // ENABLE(WEB_AUTHN)

93} // namespace WebCore
