43Vector<uint8_t> produceRpIdHash(const String& rpId)
44{
45 auto crypto = PAL::CryptoDigest::create(PAL::CryptoDigest::Algorithm::SHA_256);
46 auto rpIdUtf8 = rpId.utf8();
47 crypto->addBytes(rpIdUtf8.data(), rpIdUtf8.length());
48 return crypto->computeHash();
49}

51Vector<uint8_t> encodeES256PublicKeyAsCBOR(Vector<uint8_t>&& x, Vector<uint8_t>&& y)
52{
53 cbor::CBORValue::MapValue publicKeyMap;
54 publicKeyMap[cbor::CBORValue(COSE::kty)] = cbor::CBORValue(COSE::EC2);
55 publicKeyMap[cbor::CBORValue(COSE::alg)] = cbor::CBORValue(COSE::ES256);
56 publicKeyMap[cbor::CBORValue(COSE::crv)] = cbor::CBORValue(COSE::P_256);
57 publicKeyMap[cbor::CBORValue(COSE::x)] = cbor::CBORValue(WTFMove(x));
58 publicKeyMap[cbor::CBORValue(COSE::y)] = cbor::CBORValue(WTFMove(y));
59
60 auto cosePublicKey = cbor::CBORWriter::write(cbor::CBORValue(WTFMove(publicKeyMap)));
61 ASSERT(cosePublicKey);
62 return *cosePublicKey;
63}

65Vector<uint8_t> buildAttestedCredentialData(const Vector<uint8_t>& aaguid, const Vector<uint8_t>& credentialId, const Vector<uint8_t>& coseKey)
66{
67 Vector<uint8_t> attestedCredentialData;
68 attestedCredentialData.reserveInitialCapacity(aaguidLength + credentialIdLengthLength + credentialId.size() + coseKey.size());
69
70 // aaguid
71 ASSERT(aaguid.size() == aaguidLength);
72 attestedCredentialData.appendVector(aaguid);
73
74 // credentialIdLength
75 ASSERT(credentialId.size() <= std::numeric_limits<uint16_t>::max());
76 attestedCredentialData.append(credentialId.size() >> 8 & 0xff);
77 attestedCredentialData.append(credentialId.size() & 0xff);
78
79 // credentialId
80 attestedCredentialData.appendVector(credentialId);
81
82 // credentialPublicKey
83 attestedCredentialData.appendVector(coseKey);
84
85 return attestedCredentialData;
86}

88Vector<uint8_t> buildAuthData(const String& rpId, const uint8_t flags, const uint32_t counter, const Vector<uint8_t>& optionalAttestedCredentialData)
89{
90 Vector<uint8_t> authData;
91 authData.reserveInitialCapacity(rpIdHashLength + flagsLength + signCounterLength + optionalAttestedCredentialData.size());
92
93 // RP ID hash
94 authData.appendVector(produceRpIdHash(rpId));
95
96 // FLAGS
97 authData.append(flags);
98
99 // COUNTER
100 authData.append(counter >> 24 & 0xff);
101 authData.append(counter >> 16 & 0xff);
102 authData.append(counter >> 8 & 0xff);
103 authData.append(counter & 0xff);
104
105 // ATTESTED CRED. DATA
106 authData.appendVector(optionalAttestedCredentialData);
107
108 return authData;
109}

111Vector<uint8_t> buildAttestationObject(Vector<uint8_t>&& authData, String&& format, cbor::CBORValue::MapValue&& statementMap, const AttestationConveyancePreference& attestation)
112{
113 cbor::CBORValue::MapValue attestationObjectMap;
114 // The following implements Step 20 with regard to AttestationConveyancePreference
115 // of https://www.w3.org/TR/webauthn/#createCredential as of 4 March 2019.
116 // None attestation is always returned if it is requested to keep consistency, and therefore skip the
117 // step to return self attestation.
118 if (attestation == AttestationConveyancePreference::None) {
119 const size_t aaguidOffset = rpIdHashLength + flagsLength + signCounterLength;
120 if (authData.size() >= aaguidOffset + aaguidLength)
121 memset(authData.data() + aaguidOffset, 0, aaguidLength);
122 format = noneAttestationValue;
123 statementMap.clear();
124 }
125 attestationObjectMap[cbor::CBORValue("authData")] = cbor::CBORValue(WTFMove(authData));
126 attestationObjectMap[cbor::CBORValue("fmt")] = cbor::CBORValue(WTFMove(format));
127 attestationObjectMap[cbor::CBORValue("attStmt")] = cbor::CBORValue(WTFMove(statementMap));
128
129 auto attestationObject = cbor::CBORWriter::write(cbor::CBORValue(WTFMove(attestationObjectMap)));
130 ASSERT(attestationObject);
131 return *attestationObject;
132}
