47static ProtocolVersion convertStringToProtocolVersion(const String& version)
48{
49 if (version == kCtap2Version)
50 return ProtocolVersion::kCtap;
51 if (version == kU2fVersion)
52 return ProtocolVersion::kU2f;
53
54 return ProtocolVersion::kUnknown;
55}

57std::optional<cbor::CBORValue> decodeResponseMap(const Vector<uint8_t>& inBuffer)
58{
59 if (inBuffer.size() <= kResponseCodeLength || getResponseCode(inBuffer) != CtapDeviceResponseCode::kSuccess)
60 return std::nullopt;
61
62 Vector<uint8_t> buffer { inBuffer.data() + 1, inBuffer.size() - 1 };
63 std::optional<CBOR> decodedResponse = cbor::CBORReader::read(buffer);
64 if (!decodedResponse || !decodedResponse->isMap())
65 return std::nullopt;
66 return decodedResponse;
67}
68
69CtapDeviceResponseCode getResponseCode(const Vector<uint8_t>& buffer)
70{
71 if (buffer.isEmpty())
72 return CtapDeviceResponseCode::kCtap2ErrInvalidCBOR;
73
74 auto code = static_cast<CtapDeviceResponseCode>(buffer[0]);
75 return isCtapDeviceResponseCode(code) ? code : CtapDeviceResponseCode::kCtap2ErrInvalidCBOR;
76}

78static Vector<uint8_t> getCredentialId(const Vector<uint8_t>& authenticatorData)
79{
80 const size_t credentialIdLengthOffset = rpIdHashLength + flagsLength + signCounterLength + aaguidLength;
81
82 if (authenticatorData.size() < credentialIdLengthOffset + credentialIdLengthLength)
83 return { };
84 size_t credentialIdLength = (static_cast<size_t>(authenticatorData[credentialIdLengthOffset]) << 8) | static_cast<size_t>(authenticatorData[credentialIdLengthOffset + 1]);
85
86 if (authenticatorData.size() < credentialIdLengthOffset + credentialIdLengthLength + credentialIdLength)
87 return { };
88 Vector<uint8_t> credentialId;
89 credentialId.reserveInitialCapacity(credentialIdLength);
90 auto beginIt = authenticatorData.begin() + credentialIdLengthOffset + credentialIdLengthLength;
91 credentialId.appendRange(beginIt, beginIt + credentialIdLength);
92 return credentialId;
93}

96// Decodes byte array response from authenticator to CBOR value object and
97// checks for correct encoding format.
98RefPtr<AuthenticatorAttestationResponse> readCTAPMakeCredentialResponse(const Vector<uint8_t>& inBuffer, WebCore::AuthenticatorAttachment attachment, const AttestationConveyancePreference& attestation)
99{
100 auto decodedMap = decodeResponseMap(inBuffer);
101 if (!decodedMap)
102 return nullptr;
103 const auto& responseMap = decodedMap->getMap();
104
105 auto it = responseMap.find(CBOR(1));
106 if (it == responseMap.end() || !it->second.isString())
107 return nullptr;
108 auto format = it->second.clone();
109
110 it = responseMap.find(CBOR(2));
111 if (it == responseMap.end() || !it->second.isByteString())
112 return nullptr;
113 auto authenticatorData = it->second.clone();
114
115 auto credentialId = getCredentialId(authenticatorData.getByteString());
116 if (credentialId.isEmpty())
117 return nullptr;
118
119 it = responseMap.find(CBOR(3));
120 if (it == responseMap.end() || !it->second.isMap())
121 return nullptr;
122 auto attStmt = it->second.clone();
123
124 std::optional<Vector<uint8_t>> attestationObject;
125 if (attestation == AttestationConveyancePreference::None) {
126 // The reason why we can't directly pass authenticatorData/format/attStmt to buildAttestationObject
127 // is that they are CBORValue instead of the raw type.
128 // Also, format and attStmt are omitted as they are not useful in none attestation.
129 attestationObject = buildAttestationObject(Vector<uint8_t>(authenticatorData.getByteString()), "", { }, attestation);
130 } else {
131 CBOR::MapValue attestationObjectMap;
132 attestationObjectMap[CBOR("authData")] = WTFMove(authenticatorData);
133 attestationObjectMap[CBOR("fmt")] = WTFMove(format);
134 attestationObjectMap[CBOR("attStmt")] = WTFMove(attStmt);
135 attestationObject = cbor::CBORWriter::write(CBOR(WTFMove(attestationObjectMap)));
136 }
137
138 return AuthenticatorAttestationResponse::create(credentialId, *attestationObject, attachment);
139}

141RefPtr<AuthenticatorAssertionResponse> readCTAPGetAssertionResponse(const Vector<uint8_t>& inBuffer, WebCore::AuthenticatorAttachment attachment)
142{
143 auto decodedMap = decodeResponseMap(inBuffer);
144 if (!decodedMap)
145 return nullptr;
146 const auto& responseMap = decodedMap->getMap();
147
148 auto it = responseMap.find(CBOR(1));
149 if (it == responseMap.end() || !it->second.isMap())
150 return nullptr;
151 auto& credential = it->second.getMap();
152 auto itr = credential.find(CBOR(kCredentialIdKey));
153 if (itr == credential.end() || !itr->second.isByteString())
154 return nullptr;
155 auto& credentialId = itr->second.getByteString();
156
157 it = responseMap.find(CBOR(2));
158 if (it == responseMap.end() || !it->second.isByteString())
159 return nullptr;
160 auto& authData = it->second.getByteString();
161
162 it = responseMap.find(CBOR(3));
163 if (it == responseMap.end() || !it->second.isByteString())
164 return nullptr;
165 auto& signature = it->second.getByteString();
166
167 RefPtr<AuthenticatorAssertionResponse> response;
168 it = responseMap.find(CBOR(4));
169 if (it != responseMap.end() && it->second.isMap()) {
170 auto& user = it->second.getMap();
171 auto itr = user.find(CBOR(kEntityIdMapKey));
172 if (itr == user.end() || !itr->second.isByteString())
173 return nullptr;
174 auto& userHandle = itr->second.getByteString();
175 response = AuthenticatorAssertionResponse::create(credentialId, authData, signature, userHandle, attachment);
176
177 itr = user.find(CBOR(kEntityNameMapKey));
178 if (itr != user.end()) {
179 if (!itr->second.isString())
180 return nullptr;
181 response->setName(itr->second.getString());
182 }
183
184 itr = user.find(CBOR(kDisplayNameMapKey));
185 if (itr != user.end()) {
186 if (!itr->second.isString())
187 return nullptr;
188 response->setDisplayName(itr->second.getString());
189 }
190 } else {
191 response = AuthenticatorAssertionResponse::create(credentialId, authData, signature, { }, attachment);
192 }
193
194 it = responseMap.find(CBOR(5));
195 if (it != responseMap.end() && it->second.isUnsigned())
196 response->setNumberOfCredentials(it->second.getUnsigned());
197
198 return response;
199}

201std::optional<AuthenticatorGetInfoResponse> readCTAPGetInfoResponse(const Vector<uint8_t>& inBuffer)
202{
203 auto decodedMap = decodeResponseMap(inBuffer);
204 if (!decodedMap)
205 return std::nullopt;
206 const auto& responseMap = decodedMap->getMap();
207
208 auto it = responseMap.find(CBOR(1));
209 if (it == responseMap.end() || !it->second.isArray())
210 return std::nullopt;
211 StdSet<ProtocolVersion> protocolVersions;
212 for (const auto& version : it->second.getArray()) {
213 if (!version.isString())
214 return std::nullopt;
215
216 auto protocol = convertStringToProtocolVersion(version.getString());
217 if (protocol == ProtocolVersion::kUnknown) {
218 LOG_ERROR("Unexpected protocol version received.");
219 continue;
220 }
221
222 if (!protocolVersions.insert(protocol).second)
223 return std::nullopt;
224 }
225 if (protocolVersions.empty())
226 return std::nullopt;
227
228 it = responseMap.find(CBOR(3));
229 if (it == responseMap.end() || !it->second.isByteString() || it->second.getByteString().size() != aaguidLength)
230 return std::nullopt;
231
232 AuthenticatorGetInfoResponse response(WTFMove(protocolVersions), Vector<uint8_t>(it->second.getByteString()));
233
234 it = responseMap.find(CBOR(2));
235 if (it != responseMap.end()) {
236 if (!it->second.isArray())
237 return std::nullopt;
238
239 Vector<String> extensions;
240 for (const auto& extension : it->second.getArray()) {
241 if (!extension.isString())
242 return std::nullopt;
243
244 extensions.append(extension.getString());
245 }
246 response.setExtensions(WTFMove(extensions));
247 }
248
249 AuthenticatorSupportedOptions options;
250 it = responseMap.find(CBOR(4));
251 if (it != responseMap.end()) {
252 if (!it->second.isMap())
253 return std::nullopt;
254 const auto& optionMap = it->second.getMap();
255 auto optionMapIt = optionMap.find(CBOR(kPlatformDeviceMapKey));
256 if (optionMapIt != optionMap.end()) {
257 if (!optionMapIt->second.isBool())
258 return std::nullopt;
259
260 options.setIsPlatformDevice(optionMapIt->second.getBool());
261 }
262
263 optionMapIt = optionMap.find(CBOR(kResidentKeyMapKey));
264 if (optionMapIt != optionMap.end()) {
265 if (!optionMapIt->second.isBool())
266 return std::nullopt;
267
268 options.setSupportsResidentKey(optionMapIt->second.getBool());
269 }
270
271 optionMapIt = optionMap.find(CBOR(kUserPresenceMapKey));
272 if (optionMapIt != optionMap.end()) {
273 if (!optionMapIt->second.isBool())
274 return std::nullopt;
275
276 options.setUserPresenceRequired(optionMapIt->second.getBool());
277 }
278
279 optionMapIt = optionMap.find(CBOR(kUserVerificationMapKey));
280 if (optionMapIt != optionMap.end()) {
281 if (!optionMapIt->second.isBool())
282 return std::nullopt;
283
284 if (optionMapIt->second.getBool())
285 options.setUserVerificationAvailability(AuthenticatorSupportedOptions::UserVerificationAvailability::kSupportedAndConfigured);
286 else
287 options.setUserVerificationAvailability(AuthenticatorSupportedOptions::UserVerificationAvailability::kSupportedButNotConfigured);
288 }
289
290 optionMapIt = optionMap.find(CBOR(kClientPinMapKey));
291 if (optionMapIt != optionMap.end()) {
292 if (!optionMapIt->second.isBool())
293 return std::nullopt;
294
295 if (optionMapIt->second.getBool())
296 options.setClientPinAvailability(AuthenticatorSupportedOptions::ClientPinAvailability::kSupportedAndPinSet);
297 else
298 options.setClientPinAvailability(AuthenticatorSupportedOptions::ClientPinAvailability::kSupportedButPinNotSet);
299 }
300 response.setOptions(WTFMove(options));
301 }
302
303 it = responseMap.find(CBOR(5));
304 if (it != responseMap.end()) {
305 if (!it->second.isUnsigned())
306 return std::nullopt;
307
308 response.setMaxMsgSize(it->second.getUnsigned());
309 }
310
311 it = responseMap.find(CBOR(6));
312 if (it != responseMap.end()) {
313 if (!it->second.isArray())
314 return std::nullopt;
315
316 Vector<uint8_t> supportedPinProtocols;
317 for (const auto& protocol : it->second.getArray()) {
318 if (!protocol.isUnsigned())
319 return std::nullopt;
320
321 supportedPinProtocols.append(protocol.getUnsigned());
322 }
323 response.setPinProtocols(WTFMove(supportedPinProtocols));
324 }
325
326 return WTFMove(response);
327}
