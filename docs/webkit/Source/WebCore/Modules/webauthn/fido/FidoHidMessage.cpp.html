40// static
41std::optional<FidoHidMessage> FidoHidMessage::create(uint32_t channelId, FidoHidDeviceCommand type, const Vector<uint8_t>& data)
42{
43 if (data.size() > kHidMaxMessageSize)
44 return std::nullopt;
45
46 switch (type) {
47 case FidoHidDeviceCommand::kPing:
48 break;
49 case FidoHidDeviceCommand::kMsg:
50 case FidoHidDeviceCommand::kCbor: {
51 if (data.isEmpty())
52 return std::nullopt;
53 break;
54 }
55
56 case FidoHidDeviceCommand::kCancel:
57 case FidoHidDeviceCommand::kWink: {
58 if (!data.isEmpty())
59 return std::nullopt;
60 break;
61 }
62 case FidoHidDeviceCommand::kLock: {
63 if (data.size() != 1 || data[0] > kHidMaxLockSeconds)
64 return std::nullopt;
65 break;
66 }
67 case FidoHidDeviceCommand::kInit: {
68 if (data.size() != 8)
69 return std::nullopt;
70 break;
71 }
72 case FidoHidDeviceCommand::kKeepAlive:
73 case FidoHidDeviceCommand::kError:
74 if (data.size() != 1)
75 return std::nullopt;
76 }
77
78 return FidoHidMessage(channelId, type, data);
79}

81// static
82std::optional<FidoHidMessage> FidoHidMessage::createFromSerializedData(const Vector<uint8_t>& serializedData)
83{
84 size_t remainingSize = 0;
85 if (serializedData.size() > kHidPacketSize || serializedData.size() < kHidInitPacketHeaderSize)
86 return std::nullopt;
87
88 auto initPacket = FidoHidInitPacket::createFromSerializedData(serializedData, &remainingSize);
89
90 if (!initPacket)
91 return std::nullopt;
92
93 return FidoHidMessage(WTFMove(initPacket), remainingSize);
94}

96bool FidoHidMessage::messageComplete() const
97{
98 return !m_remainingSize;
99}
100
101Vector<uint8_t> FidoHidMessage::getMessagePayload() const
102{
103 Vector<uint8_t> data;
104 size_t dataSize = 0;
105 for (const auto& packet : m_packets)
106 dataSize += packet->getPacketPayload().size();
107 data.reserveInitialCapacity(dataSize);
108
109 for (const auto& packet : m_packets) {
110 const auto& packet_data = packet->getPacketPayload();
111 data.appendVector(packet_data);
112 }
113
114 return data;
115}
116
117Vector<uint8_t> FidoHidMessage::popNextPacket()
118{
119 if (m_packets.isEmpty())
120 return { };
121
122 Vector<uint8_t> data = m_packets.first()->getSerializedData();
123 m_packets.removeFirst();
124 return data;
125}
126
127bool FidoHidMessage::addContinuationPacket(const Vector<uint8_t>& buf)
128{
129 size_t remainingSize = m_remainingSize;
130 auto contPacket = FidoHidContinuationPacket::createFromSerializedData(buf, &remainingSize);
131
132 // Reject packets with a different channel id.
133 if (!contPacket || m_channelId != contPacket->channelId())
134 return false;
135
136 m_remainingSize = remainingSize;
137 m_packets.append(WTFMove(contPacket));
138 return true;
139}
140
141size_t FidoHidMessage::numPackets() const
142{
143 return m_packets.size();
144}
145
146FidoHidMessage::FidoHidMessage(uint32_t channelId, FidoHidDeviceCommand type, const Vector<uint8_t>& data)
147 : m_channelId(channelId)
148 , m_cmd(type)
149{
150 uint8_t sequence = 0;
151
152 size_t pos = data.size() > kHidInitPacketDataSize ? kHidInitPacketDataSize : data.size();
153 m_packets.append(makeUnique<FidoHidInitPacket>(channelId, type, getInitPacketData(data), data.size()));
154 for (; pos < data.size(); pos += kHidContinuationPacketDataSize)
155 m_packets.append(makeUnique<FidoHidContinuationPacket>(channelId, sequence++, getContinuationPacketData(data, pos)));
156}
157
158FidoHidMessage::FidoHidMessage(std::unique_ptr<FidoHidInitPacket> initPacket, size_t remainingSize)
159 : m_remainingSize(remainingSize)
160{
161 m_channelId = initPacket->channelId();
162 m_cmd = initPacket->command();
163 m_packets.append(WTFMove(initPacket));
164}
