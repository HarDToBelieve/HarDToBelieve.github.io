55// hasAtLeastFourCodepoints returns true if |pin| contains
56// four or more code points. This reflects the "4 Unicode characters"
57// requirement in CTAP2.
58static bool hasAtLeastFourCodepoints(const String& pin)
59{
60 return pin.length() >= 4;
61}
62
63// makePinAuth returns `LEFT(HMAC-SHA-256(secret, data), 16)`.
64static Vector<uint8_t> makePinAuth(const CryptoKeyHMAC& key, const Vector<uint8_t>& data)
65{
66 auto result = CryptoAlgorithmHMAC::platformSign(key, data);
67 ASSERT(!result.hasException());
68 auto pinAuth = result.releaseReturnValue();
69 pinAuth.shrink(16);
70 return pinAuth;
71}

73Vector<uint8_t> encodeRawPublicKey(const Vector<uint8_t>& x, const Vector<uint8_t>& y)
74{
75 Vector<uint8_t> rawKey;
76 rawKey.reserveCapacity(1 + x.size() + y.size());
77 rawKey.append(0x04);
78 rawKey.appendVector(x);
79 rawKey.appendVector(y);
80 return rawKey;
81}

83std::optional<CString> validateAndConvertToUTF8(const String& pin)
84{
85 if (!hasAtLeastFourCodepoints(pin))
86 return std::nullopt;
87 auto result = pin.utf8();
88 if (result.length() < kMinBytes || result.length() > kMaxBytes)
89 return std::nullopt;
90 return result;
91}

93// encodePINCommand returns a CTAP2 PIN command for the operation |subcommand|.
94// Additional elements of the top-level CBOR map can be added with the optional
95// |addAdditional| callback.
96static Vector<uint8_t> encodePinCommand(Subcommand subcommand, Function<void(CBORValue::MapValue*)> addAdditional = nullptr)
97{
98 CBORValue::MapValue map;
99 map.emplace(static_cast<int64_t>(RequestKey::kProtocol), kProtocolVersion);
100 map.emplace(static_cast<int64_t>(RequestKey::kSubcommand), static_cast<int64_t>(subcommand));
101
102 if (addAdditional)
103 addAdditional(&map);
104
105 auto serializedParam = CBORWriter::write(CBORValue(WTFMove(map)));
106 ASSERT(serializedParam);
107
108 Vector<uint8_t> cborRequest({ static_cast<uint8_t>(CtapRequestCommand::kAuthenticatorClientPin) });
109 cborRequest.appendVector(*serializedParam);
110 return cborRequest;
111}

115std::optional<RetriesResponse> RetriesResponse::parse(const Vector<uint8_t>& inBuffer)
116{
117 auto decodedMap = decodeResponseMap(inBuffer);
118 if (!decodedMap)
119 return std::nullopt;
120 const auto& responseMap = decodedMap->getMap();
121
122 auto it = responseMap.find(CBORValue(static_cast<int64_t>(ResponseKey::kRetries)));
123 if (it == responseMap.end() || !it->second.isUnsigned())
124 return std::nullopt;
125
126 RetriesResponse ret;
127 ret.retries = static_cast<uint64_t>(it->second.getUnsigned());
128 return ret;
129}

136std::optional<KeyAgreementResponse> KeyAgreementResponse::parse(const Vector<uint8_t>& inBuffer)
137{
138 auto decodedMap = decodeResponseMap(inBuffer);
139 if (!decodedMap)
140 return std::nullopt;
141 const auto& responseMap = decodedMap->getMap();
142
143 // The ephemeral key is encoded as a COSE structure.
144 auto it = responseMap.find(CBORValue(static_cast<int64_t>(ResponseKey::kKeyAgreement)));
145 if (it == responseMap.end() || !it->second.isMap())
146 return std::nullopt;
147 const auto& coseKey = it->second.getMap();
148
149 return parseFromCOSE(coseKey);
150}

152std::optional<KeyAgreementResponse> KeyAgreementResponse::parseFromCOSE(const CBORValue::MapValue& coseKey)
153{
154 // The COSE key must be a P-256 point. See
155 // https://tools.ietf.org/html/rfc8152#section-7.1
156 for (const auto& pair : Vector<std::pair<int64_t, int64_t>>({
157 { static_cast<int64_t>(COSE::kty), static_cast<int64_t>(COSE::EC2) },
158 { static_cast<int64_t>(COSE::alg), static_cast<int64_t>(COSE::ECDH256) },
159 { static_cast<int64_t>(COSE::crv), static_cast<int64_t>(COSE::P_256) },
160 })) {
161 auto it = coseKey.find(CBORValue(pair.first));
162 if (it == coseKey.end() || !it->second.isInteger() || it->second.getInteger() != pair.second)
163 return std::nullopt;
164 }
165
166 // See https://tools.ietf.org/html/rfc8152#section-13.1.1
167 const auto& xIt = coseKey.find(CBORValue(static_cast<int64_t>(COSE::x)));
168 const auto& yIt = coseKey.find(CBORValue(static_cast<int64_t>(COSE::y)));
169 if (xIt == coseKey.end() || yIt == coseKey.end() || !xIt->second.isByteString() || !yIt->second.isByteString())
170 return std::nullopt;
171
172 const auto& x = xIt->second.getByteString();
173 const auto& y = yIt->second.getByteString();
174 auto peerKey = CryptoKeyEC::importRaw(CryptoAlgorithmIdentifier::ECDH, "P-256", encodeRawPublicKey(x, y), true, CryptoKeyUsageDeriveBits);
175 if (!peerKey)
176 return std::nullopt;
177
178 return KeyAgreementResponse(peerKey.releaseNonNull());
179}

181cbor::CBORValue::MapValue encodeCOSEPublicKey(const Vector<uint8_t>& rawPublicKey)
182{
183 ASSERT(rawPublicKey.size() == 65);
184 Vector<uint8_t> x { rawPublicKey.data() + 1, ES256FieldElementLength };
185 Vector<uint8_t> y { rawPublicKey.data() + 1 + ES256FieldElementLength, ES256FieldElementLength };
186
187 cbor::CBORValue::MapValue publicKeyMap;
188 publicKeyMap[cbor::CBORValue(COSE::kty)] = cbor::CBORValue(COSE::EC2);
189 publicKeyMap[cbor::CBORValue(COSE::alg)] = cbor::CBORValue(COSE::ECDH256);
190 publicKeyMap[cbor::CBORValue(COSE::crv)] = cbor::CBORValue(COSE::P_256);
191 publicKeyMap[cbor::CBORValue(COSE::x)] = cbor::CBORValue(WTFMove(x));
192 publicKeyMap[cbor::CBORValue(COSE::y)] = cbor::CBORValue(WTFMove(y));
193
194 return publicKeyMap;
195}

202std::optional<TokenResponse> TokenResponse::parse(const WebCore::CryptoKeyAES& sharedKey, const Vector<uint8_t>& inBuffer)
203{
204 auto decodedMap = decodeResponseMap(inBuffer);
205 if (!decodedMap)
206 return std::nullopt;
207 const auto& responseMap = decodedMap->getMap();
208
209 auto it = responseMap.find(CBORValue(static_cast<int64_t>(ResponseKey::kPinToken)));
210 if (it == responseMap.end() || !it->second.isByteString())
211 return std::nullopt;
212 const auto& encryptedToken = it->second.getByteString();
213
214 auto tokenResult = CryptoAlgorithmAES_CBC::platformDecrypt({ }, sharedKey, encryptedToken, CryptoAlgorithmAES_CBC::Padding::No);
215 if (tokenResult.hasException())
216 return std::nullopt;
217 auto token = tokenResult.releaseReturnValue();
218
219 auto tokenKey = CryptoKeyHMAC::importRaw(token.size() * 8, CryptoAlgorithmIdentifier::SHA_256, WTFMove(token), true, CryptoKeyUsageSign);
220 ASSERT(tokenKey);
221
222 return TokenResponse(tokenKey.releaseNonNull());
223}

245std::optional<TokenRequest> TokenRequest::tryCreate(const CString& pin, const CryptoKeyEC& peerKey)
246{
247 // The following implements Section 5.5.4 Getting sharedSecret from Authenticator.
248 // https://fidoalliance.org/specs/fido-v2.0-ps-20190130/fido-client-to-authenticator-protocol-v2.0-ps-20190130.html#gettingSharedSecret
249 // 1. Generate a P256 key pair.
250 auto keyPairResult = CryptoKeyEC::generatePair(CryptoAlgorithmIdentifier::ECDH, "P-256", true, CryptoKeyUsageDeriveBits);
251 ASSERT(!keyPairResult.hasException());
252 auto keyPair = keyPairResult.releaseReturnValue();
253
254 // 2. Use ECDH and SHA-256 to compute the shared AES-CBC key.
255 auto sharedKeyResult = CryptoAlgorithmECDH::platformDeriveBits(downcast<CryptoKeyEC>(*keyPair.privateKey), peerKey);
256 if (!sharedKeyResult)
257 return std::nullopt;
258
259 auto crypto = PAL::CryptoDigest::create(PAL::CryptoDigest::Algorithm::SHA_256);
260 crypto->addBytes(sharedKeyResult->data(), sharedKeyResult->size());
261 auto sharedKeyHash = crypto->computeHash();
262
263 auto sharedKey = CryptoKeyAES::importRaw(CryptoAlgorithmIdentifier::AES_CBC, WTFMove(sharedKeyHash), true, CryptoKeyUsageEncrypt | CryptoKeyUsageDecrypt);
264 ASSERT(sharedKey);
265
266 // The following encodes the public key of the above key pair into COSE format.
267 auto rawPublicKeyResult = downcast<CryptoKeyEC>(*keyPair.publicKey).exportRaw();
268 ASSERT(!rawPublicKeyResult.hasException());
269 auto coseKey = encodeCOSEPublicKey(rawPublicKeyResult.returnValue());
270
271 // The following calculates a SHA-256 digest of the PIN, and shrink to the left 16 bytes.
272 crypto = PAL::CryptoDigest::create(PAL::CryptoDigest::Algorithm::SHA_256);
273 crypto->addBytes(pin.data(), pin.length());
274 auto pinHash = crypto->computeHash();
275 pinHash.shrink(16);
276
277 return TokenRequest(sharedKey.releaseNonNull(), WTFMove(coseKey), WTFMove(pinHash));
278}

292Vector<uint8_t> encodeAsCBOR(const TokenRequest& request)
293{
294 auto result = CryptoAlgorithmAES_CBC::platformEncrypt({ }, request.sharedKey(), request.m_pinHash, CryptoAlgorithmAES_CBC::Padding::No);
295 ASSERT(!result.hasException());
296
297 return encodePinCommand(Subcommand::kGetPinToken, [coseKey = WTFMove(request.m_coseKey), encryptedPin = result.releaseReturnValue()] (CBORValue::MapValue* map) mutable {
298 map->emplace(static_cast<int64_t>(RequestKey::kKeyAgreement), WTFMove(coseKey));
299 map->emplace(static_cast<int64_t>(RequestKey::kPinHashEnc), WTFMove(encryptedPin));
300 });
301}
