48static Vector<uint8_t> constructU2fRegisterCommand(const Vector<uint8_t>& applicationParameter, const Vector<uint8_t>& challengeParameter)
49{
50 Vector<uint8_t> data;
51 data.reserveInitialCapacity(kU2fChallengeParamLength + kU2fApplicationParamLength);
52 data.appendVector(challengeParameter);
53 data.appendVector(applicationParameter);
54
55 apdu::ApduCommand command;
56 command.setIns(static_cast<uint8_t>(U2fApduInstruction::kRegister));
57 // This is needed for test of user presence even though the spec doesn't specify it.
58 command.setP1(kP1EnforceUserPresenceAndSign);
59 command.setData(WTFMove(data));
60 command.setResponseLength(apdu::ApduCommand::kApduMaxResponseLength);
61 return command.getEncodedCommand();
62}
63
64static std::optional<Vector<uint8_t>> constructU2fSignCommand(const Vector<uint8_t>& applicationParameter, const Vector<uint8_t>& challengeParameter, const Vector<uint8_t>& keyHandle, bool checkOnly)
65{
66 if (keyHandle.size() > kMaxKeyHandleLength)
67 return std::nullopt;
68
69 Vector<uint8_t> data;
70 data.reserveInitialCapacity(kU2fChallengeParamLength + kU2fApplicationParamLength + 1 + keyHandle.size());
71 data.appendVector(challengeParameter);
72 data.appendVector(applicationParameter);
73 data.append(static_cast<uint8_t>(keyHandle.size()));
74 data.appendVector(keyHandle);
75
76 apdu::ApduCommand command;
77 command.setIns(static_cast<uint8_t>(U2fApduInstruction::kSign));
78 command.setP1(checkOnly ? kP1CheckOnly : kP1EnforceUserPresenceAndSign);
79 command.setData(WTFMove(data));
80 command.setResponseLength(apdu::ApduCommand::kApduMaxResponseLength);
81 return command.getEncodedCommand();
82}

86bool isConvertibleToU2fRegisterCommand(const PublicKeyCredentialCreationOptions& request)
87{
88 if (request.authenticatorSelection && (request.authenticatorSelection->userVerification == UserVerificationRequirement::Required || request.authenticatorSelection->requireResidentKey))
89 return false;
90 if (request.pubKeyCredParams.findMatching([](auto& item) { return item.alg == COSE::ES256; }) == notFound)
91 return false;
92 return true;
93}
94
95bool isConvertibleToU2fSignCommand(const PublicKeyCredentialRequestOptions& request)
96{
97 return (request.userVerification != UserVerificationRequirement::Required) && !request.allowCredentials.isEmpty();
98}
99
100std::optional<Vector<uint8_t>> convertToU2fRegisterCommand(const Vector<uint8_t>& clientDataHash, const PublicKeyCredentialCreationOptions& request)
101{
102 if (!isConvertibleToU2fRegisterCommand(request))
103 return std::nullopt;
104
105 auto appId = processGoogleLegacyAppIdSupportExtension(request.extensions);
106 return constructU2fRegisterCommand(produceRpIdHash(!appId ? request.rp.id : appId), clientDataHash);
107}
108
109std::optional<Vector<uint8_t>> convertToU2fCheckOnlySignCommand(const Vector<uint8_t>& clientDataHash, const PublicKeyCredentialCreationOptions& request, const PublicKeyCredentialDescriptor& keyHandle)
110{
111 if (keyHandle.type != PublicKeyCredentialType::PublicKey)
112 return std::nullopt;
113
114 return constructU2fSignCommand(produceRpIdHash(request.rp.id), clientDataHash, keyHandle.idVector, true /* checkOnly */);
115}
116
117std::optional<Vector<uint8_t>> convertToU2fSignCommand(const Vector<uint8_t>& clientDataHash, const PublicKeyCredentialRequestOptions& request, const Vector<uint8_t>& keyHandle, bool isAppId)
118{
119 if (!isConvertibleToU2fSignCommand(request))
120 return std::nullopt;
121
122 if (!isAppId)
123 return constructU2fSignCommand(produceRpIdHash(request.rpId), clientDataHash, keyHandle, false);
124 ASSERT(request.extensions && !request.extensions->appid.isNull());
125 return constructU2fSignCommand(produceRpIdHash(request.extensions->appid), clientDataHash, keyHandle, false);
126}

128Vector<uint8_t> constructBogusU2fRegistrationCommand()
129{
130 return constructU2fRegisterCommand(convertBytesToVector(kBogusAppParam, sizeof(kBogusAppParam)), convertBytesToVector(kBogusChallenge, sizeof(kBogusChallenge)));
131}

133String processGoogleLegacyAppIdSupportExtension(const std::optional<AuthenticationExtensionsClientInputs>& extensions)
134{
135 if (!extensions) {
136 // AuthenticatorCoordinator::create should always set it.
137 ASSERT_NOT_REACHED();
138 return String();
139 }
140
141 if (!extensions->googleLegacyAppidSupport)
142 return String();
143 return "https://www.gstatic.com/securitykey/origins.json"_s;
144}
