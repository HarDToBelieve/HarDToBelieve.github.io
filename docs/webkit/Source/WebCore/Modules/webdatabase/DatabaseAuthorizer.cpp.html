61void DatabaseAuthorizer::addAllowedFunctions()
62{
63 // SQLite functions used to help implement some operations
64 // ALTER TABLE helpers
65 m_allowedFunctions.add("sqlite_rename_table");
66 m_allowedFunctions.add("sqlite_rename_trigger");
67 // GLOB helpers
68 m_allowedFunctions.add("glob");
69
70 // SQLite core functions
71 m_allowedFunctions.add("abs");
72 m_allowedFunctions.add("changes");
73 m_allowedFunctions.add("coalesce");
74 m_allowedFunctions.add("glob");
75 m_allowedFunctions.add("ifnull");
76 m_allowedFunctions.add("hex");
77 m_allowedFunctions.add("last_insert_rowid");
78 m_allowedFunctions.add("length");
79 m_allowedFunctions.add("like");
80 m_allowedFunctions.add("lower");
81 m_allowedFunctions.add("ltrim");
82 m_allowedFunctions.add("max");
83 m_allowedFunctions.add("min");
84 m_allowedFunctions.add("nullif");
85 m_allowedFunctions.add("quote");
86 m_allowedFunctions.add("replace");
87 m_allowedFunctions.add("round");
88 m_allowedFunctions.add("rtrim");
89 m_allowedFunctions.add("soundex");
90 m_allowedFunctions.add("sqlite_source_id");
91 m_allowedFunctions.add("sqlite_version");
92 m_allowedFunctions.add("substr");
93 m_allowedFunctions.add("total_changes");
94 m_allowedFunctions.add("trim");
95 m_allowedFunctions.add("typeof");
96 m_allowedFunctions.add("upper");
97 m_allowedFunctions.add("zeroblob");
98
99 // SQLite date and time functions
100 m_allowedFunctions.add("date");
101 m_allowedFunctions.add("time");
102 m_allowedFunctions.add("datetime");
103 m_allowedFunctions.add("julianday");
104 m_allowedFunctions.add("strftime");
105
106 // SQLite aggregate functions
107 // max() and min() are already in the list
108 m_allowedFunctions.add("avg");
109 m_allowedFunctions.add("count");
110 m_allowedFunctions.add("group_concat");
111 m_allowedFunctions.add("sum");
112 m_allowedFunctions.add("total");
113
114 // SQLite FTS functions
115 m_allowedFunctions.add("match");
116 m_allowedFunctions.add("snippet");
117 m_allowedFunctions.add("offsets");
118 m_allowedFunctions.add("optimize");
119
120 // SQLite ICU functions
121 // like(), lower() and upper() are already in the list
122 m_allowedFunctions.add("regexp");
123}

125int DatabaseAuthorizer::createTable(const String& tableName)
126{
127 if (!allowWrite())
128 return SQLAuthDeny;
129
130 m_lastActionChangedDatabase = true;
131 return denyBasedOnTableName(tableName);
132}

134int DatabaseAuthorizer::createTempTable(const String& tableName)
135{
136 // SQLITE_CREATE_TEMP_TABLE results in a UPDATE operation, which is not
137 // allowed in read-only transactions or private browsing, so we might as
138 // well disallow SQLITE_CREATE_TEMP_TABLE in these cases
139 if (!allowWrite())
140 return SQLAuthDeny;
141
142 return denyBasedOnTableName(tableName);
143}

388int DatabaseAuthorizer::denyBasedOnTableName(const String& tableName) const
389{
390 if (!m_securityEnabled)
391 return SQLAuthAllow;
392
393 // Sadly, normal creates and drops end up affecting sqlite_master in an authorizer callback, so
394 // it will be tough to enforce all of the following policies.
395 // if (equalIgnoringASCIICase(tableName, "sqlite_master")
396 // || equalIgnoringASCIICase(tableName, "sqlite_temp_master")
397 // || equalIgnoringASCIICase(tableName, "sqlite_sequence")
398 // || equalIgnoringASCIICase(tableName, Database::databaseInfoTableName()))
399 // return SQLAuthDeny;
400
401 if (equalIgnoringASCIICase(tableName, m_databaseInfoTableName))
402 return SQLAuthDeny;
403
404 return SQLAuthAllow;
405}
406
407int DatabaseAuthorizer::updateDeletesBasedOnTableName(const String& tableName)
408{
409 int allow = denyBasedOnTableName(tableName);
410 if (allow)
411 m_hadDeletes = true;
412 return allow;
413}
