38DatabaseTaskSynchronizer::DatabaseTaskSynchronizer()
39{
40}
41
42void DatabaseTaskSynchronizer::waitForTaskCompletion()
43{
44 Locker locker { m_synchronousLock };
45 while (!m_taskCompleted)
46 m_synchronousCondition.wait(m_synchronousLock);
47}
48
49void DatabaseTaskSynchronizer::taskCompleted()
50{
51 Locker locker { m_synchronousLock };
52 m_taskCompleted = true;
53 m_synchronousCondition.notifyOne();
54}

56DatabaseTask::DatabaseTask(Database& database, DatabaseTaskSynchronizer* synchronizer)
57 : m_database(database)
58 , m_synchronizer(synchronizer)
59{
60}
61
62DatabaseTask::~DatabaseTask()
63{
64 ASSERT(m_complete || !m_synchronizer);
65}
66
67void DatabaseTask::performTask()
68{
69 // Database tasks are meant to be used only once, so make sure this one hasn't been performed before.
70 ASSERT(!m_complete);
71
72 LOG(StorageAPI, "Performing %s %p\n", debugTaskName(), this);
73
74 m_database.resetAuthorizer();
75
76 doPerformTask();
77
78 if (m_synchronizer)
79 m_synchronizer->taskCompleted();
80
81#if ASSERT_ENABLED
82 m_complete = true;
83#endif
84}

132// *** DatabaseTransactionTask ***
133// Starts a transaction that will report its results via a callback.
134
135DatabaseTransactionTask::DatabaseTransactionTask(RefPtr<SQLTransaction>&& transaction)
136 : DatabaseTask(transaction->database(), 0)
137 , m_transaction(WTFMove(transaction))
138 , m_didPerformTask(false)
139{
140}
141
142DatabaseTransactionTask::~DatabaseTransactionTask()
143{
144 // If the task is being destructed without the transaction ever being run,
145 // then we must either have an error or an interruption. Give the
146 // transaction a chance to clean up since it may not have been able to
147 // run to its clean up state.
148
149 // Transaction phase 2 cleanup. See comment on "What happens if a
150 // transaction is interrupted?" at the top of SQLTransactionBackend.cpp.
151
152 if (!m_didPerformTask)
153 m_transaction->notifyDatabaseThreadIsShuttingDown();
154}
155
156void DatabaseTransactionTask::doPerformTask()
157{
158 m_transaction->performNextStep();
159 m_didPerformTask = true;
160}
