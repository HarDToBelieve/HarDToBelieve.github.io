29#pragma once
30
31#include <memory>
32#include <wtf/HashSet.h>
33#include <wtf/Lock.h>
34#include <wtf/MessageQueue.h>
35#include <wtf/RefPtr.h>
36#include <wtf/Threading.h>
37
38namespace WebCore {
39
40class Database;
41class DatabaseTask;
42class DatabaseTaskSynchronizer;
43class Document;
44class SQLTransactionCoordinator;
45
46class DatabaseThread : public ThreadSafeRefCounted<DatabaseThread> {
47public:
48 static Ref<DatabaseThread> create() { return adoptRef(*new DatabaseThread); }
49 ~DatabaseThread();
50
51 void start();
52 void requestTermination(DatabaseTaskSynchronizer* cleanupSync);
53 bool terminationRequested(DatabaseTaskSynchronizer* = nullptr) const;
54
55 void scheduleTask(std::unique_ptr<DatabaseTask>&&);
56 void scheduleImmediateTask(std::unique_ptr<DatabaseTask>&&); // This just adds the task to the front of the queue - the caller needs to be extremely careful not to create deadlocks when waiting for completion.
57 void unscheduleDatabaseTasks(Database&);
58 bool hasPendingDatabaseActivity() const;
59
60 void recordDatabaseOpen(Database&);
61 void recordDatabaseClosed(Database&);
62 Thread* getThread() { return m_thread.get(); }
63
64 SQLTransactionCoordinator* transactionCoordinator() { return m_transactionCoordinator.get(); }
65
66private:
67 DatabaseThread();
68
69 void databaseThread();
70
71 Lock m_threadCreationMutex;
72 RefPtr<Thread> m_thread;
73 RefPtr<DatabaseThread> m_selfRef;
74
75 MessageQueue<DatabaseTask> m_queue;
76
77 // This set keeps track of the open databases that have been used on this thread.
78 using DatabaseSet = HashSet<RefPtr<Database>>;
79 mutable Lock m_openDatabaseSetLock;
80 DatabaseSet m_openDatabaseSet WTF_GUARDED_BY_LOCK(m_openDatabaseSetLock);
81
82 std::unique_ptr<SQLTransactionCoordinator> m_transactionCoordinator;
83 DatabaseTaskSynchronizer* m_cleanupSync { nullptr };
84};
85
86} // namespace WebCore
