39ExceptionOr<int64_t> SQLResultSet::insertId() const
40{
41 // 4.11.4 - Return the id of the last row inserted as a result of the query
42 // If the query didn't result in any rows being added, raise an InvalidAccessError exception
43 if (!m_insertId)
44 return Exception { InvalidAccessError };
45 return m_insertId.value();
46}
