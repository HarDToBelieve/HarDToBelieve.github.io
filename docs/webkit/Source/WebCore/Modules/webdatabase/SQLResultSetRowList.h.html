29#pragma once
30
31#include "ExceptionOr.h"
32#include "SQLValue.h"
33
34namespace WebCore {
35
36class SQLResultSetRowList : public RefCounted<SQLResultSetRowList> {
37public:
38 static Ref<SQLResultSetRowList> create() { return adoptRef(*new SQLResultSetRowList); }
39
40 const Vector<String>& columnNames() const { return m_columns; }
41 const Vector<SQLValue>& values() const { return m_result; }
42
43 void addColumn(const String& name) { m_columns.append(name); }
44 void addResult(const SQLValue& result) { m_result.append(result); }
45
46 unsigned length() const;
47 ExceptionOr<Vector<WTF::KeyValuePair<String, SQLValue>>> item(unsigned index) const;
48
49private:
50 SQLResultSetRowList() { }
51
52 Vector<String> m_columns;
53 Vector<SQLValue> m_result;
54};
55
56} // namespace WebCore
