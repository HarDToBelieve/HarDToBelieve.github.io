44// The Life-Cycle of a SQLStatement i.e. Who's keeping the SQLStatement alive?
45// ==========================================================================
46// The RefPtr chain goes something like this:
47//
48// At birth (in SQLTransactionBackend::executeSQL()):
49// =================================================
50// SQLTransactionBackend // Deque<RefPtr<SQLStatement>> m_statementQueue points to ...
51// --> SQLStatement // std::unique_ptr<SQLStatement> m_frontend points to ...
52// --> SQLStatement
53//
54// After grabbing the statement for execution (in SQLTransactionBackend::getNextStatement()):
55// =========================================================================================
56// SQLTransactionBackend // RefPtr<SQLStatement> m_currentStatementBackend points to ...
57// --> SQLStatement // std::unique_ptr<SQLStatement> m_frontend points to ...
58// --> SQLStatement
59//
60// Then we execute the statement in SQLTransactionBackend::runCurrentStatementAndGetNextState().
61// And we callback to the script in SQLTransaction::deliverStatementCallback() if
62// necessary.
63// - Inside SQLTransaction::deliverStatementCallback(), we operate on a raw SQLStatement*.
64// This pointer is valid because it is owned by SQLTransactionBackend's
65// SQLTransactionBackend::m_currentStatementBackend.
66//
67// After we're done executing the statement (in SQLTransactionBackend::getNextStatement()):
68// =======================================================================================
69// When we're done executing, we'll grab the next statement. But before we
70// do that, getNextStatement() nullify SQLTransactionBackend::m_currentStatementBackend.
71// This will trigger the deletion of the SQLStatement and SQLStatement.
72//
73// Note: unlike with SQLTransaction, there is no JS representation of SQLStatement.
74// Hence, there is no GC dependency at play here.
75

78SQLStatement::SQLStatement(Database& database, const String& statement, Vector<SQLValue>&& arguments, RefPtr<SQLStatementCallback>&& callback, RefPtr<SQLStatementErrorCallback>&& errorCallback, int permissions)
79 : m_statement(statement.isolatedCopy())
80 , m_arguments(WTFMove(arguments))
81 , m_statementCallbackWrapper(WTFMove(callback), &database.document())
82 , m_statementErrorCallbackWrapper(WTFMove(errorCallback), &database.document())

85}

99bool SQLStatement::execute(Database& db)
100{
101 ASSERT(!m_resultSet);
102
103 // If we're re-running this statement after a quota violation, we need to clear that error now
104 clearFailureDueToQuota();
105
106 // This transaction might have been marked bad while it was being set up on the main thread,
107 // so if there is still an error, return false.
108 if (m_error)
109 return false;
110
111 db.setAuthorizerPermissions(m_permissions);
112
113 SQLiteDatabase& database = db.sqliteDatabase();
114
115 auto statement = database.prepareStatementSlow(m_statement);
116 if (!statement) {
117 LOG(StorageAPI, "Unable to verify correctness of statement %s - error %i (%s)", m_statement.ascii().data(), statement.error(), database.lastErrorMsg());
118 if (statement.error() == SQLITE_INTERRUPT)
119 m_error = SQLError::create(SQLError::DATABASE_ERR, "could not prepare statement", statement.error(), "interrupted");
120 else
121 m_error = SQLError::create(SQLError::SYNTAX_ERR, "could not prepare statement", statement.error(), database.lastErrorMsg());
122 return false;
123 }
124
125 // FIXME: If the statement uses the ?### syntax supported by sqlite, the bind parameter count is very likely off from the number of question marks.
126 // If this is the case, they might be trying to do something fishy or malicious
127 if (statement->bindParameterCount() != m_arguments.size()) {
128 LOG(StorageAPI, "Bind parameter count doesn't match number of question marks");
129 m_error = SQLError::create(SQLError::SYNTAX_ERR, "number of '?'s in statement string does not match argument count");
130 return false;
131 }
132
133 for (unsigned i = 0; i < m_arguments.size(); ++i) {
134 int result = statement->bindValue(i + 1, m_arguments[i]);
135 if (result == SQLITE_FULL) {
136 setFailureDueToQuota();
137 return false;
138 }
139
140 if (result != SQLITE_OK) {
141 LOG(StorageAPI, "Failed to bind value index %i to statement for query '%s'", i + 1, m_statement.ascii().data());
142 m_error = SQLError::create(SQLError::DATABASE_ERR, "could not bind value", result, database.lastErrorMsg());
143 return false;
144 }
145 }
146
147 auto resultSet = SQLResultSet::create();
148
149 // Step so we can fetch the column names.
150 int result = statement->step();
151 switch (result) {
152 case SQLITE_ROW: {
153 int columnCount = statement->columnCount();
154 auto& rows = resultSet->rows();
155
156 for (int i = 0; i < columnCount; i++)
157 rows.addColumn(statement->columnName(i));
158
159 do {
160 for (int i = 0; i < columnCount; i++)
161 rows.addResult(statement->columnValue(i));
162
163 result = statement->step();
164 } while (result == SQLITE_ROW);
165
166 if (result != SQLITE_DONE) {
167 m_error = SQLError::create(SQLError::DATABASE_ERR, "could not iterate results", result, database.lastErrorMsg());
168 return false;
169 }
170 break;
171 }
172 case SQLITE_DONE: {
173 // Didn't find anything, or was an insert
174 if (db.lastActionWasInsert())
175 resultSet->setInsertId(database.lastInsertRowID());
176 break;
177 }
178 case SQLITE_FULL:
179 // Return the Quota error - the delegate will be asked for more space and this statement might be re-run
180 setFailureDueToQuota();
181 return false;
182 case SQLITE_CONSTRAINT:
183 m_error = SQLError::create(SQLError::CONSTRAINT_ERR, "could not execute statement due to a constaint failure", result, database.lastErrorMsg());
184 return false;
185 default:
186 m_error = SQLError::create(SQLError::DATABASE_ERR, "could not execute statement", result, database.lastErrorMsg());
187 return false;
188 }
189
190 // rowsAffected should be 0 for read only statements (e.g. SELECT statement). However, SQLiteDatabase::lastChanges() returns
191 // the number of changes made by the most recent INSERT, UPDATE or DELETE statement.
192 if (!statement->isReadOnly())
193 resultSet->setRowsAffected(database.lastChanges());
194
195 m_resultSet = WTFMove(resultSet);
196 return true;
197}

199bool SQLStatement::performCallback(SQLTransaction& transaction)
200{
201 // Call the appropriate statement callback and track if it resulted in an error,
202 // because then we need to jump to the transaction error callback.
203
204 if (m_error) {
205 if (auto errorCallback = m_statementErrorCallbackWrapper.unwrap()) {
206 auto result = errorCallback->handleEvent(transaction, *m_error);
207
208 // The spec says:
209 // "If the error callback returns false, then move on to the next statement..."
210 // "Otherwise, the error callback did not return false, or there was no error callback"
211 // Therefore an exception and returning true are the same thing - so, return true on an exception
212
213 switch (result.type()) {
214 case CallbackResultType::Success:
215 return result.releaseReturnValue();
216 case CallbackResultType::ExceptionThrown:
217 case CallbackResultType::UnableToExecute:
218 return true;
219 }
220 }
221 return false;
222 }
223
224 if (auto callback = m_statementCallbackWrapper.unwrap()) {
225 ASSERT(m_resultSet);
226
227 auto result = callback->handleEvent(transaction, *m_resultSet);
228 return result.type() == CallbackResultType::ExceptionThrown;
229 }
230
231 return false;
232}

234void SQLStatement::setDatabaseDeletedError()
235{
236 ASSERT(!m_error && !m_resultSet);
237 m_error = SQLError::create(SQLError::UNKNOWN_ERR, "unable to execute statement, because the user deleted the database");
238}
239
240void SQLStatement::setVersionMismatchedError()
241{
242 ASSERT(!m_error && !m_resultSet);
243 m_error = SQLError::create(SQLError::VERSION_ERR, "current version of the database and `oldVersion` argument do not match");
244}
245
246void SQLStatement::setFailureDueToQuota()
247{
248 ASSERT(!m_error && !m_resultSet);
249 m_error = SQLError::create(SQLError::QUOTA_ERR, "there was not enough remaining storage space, or the storage quota was reached and the user declined to allow more space");
250}
251
252void SQLStatement::clearFailureDueToQuota()
253{
254 if (lastExecutionFailedDueToQuota())
255 m_error = nullptr;
256}
257
258bool SQLStatement::lastExecutionFailedDueToQuota() const
259{
260 return m_error && m_error->code() == SQLError::QUOTA_ERR;
261}
