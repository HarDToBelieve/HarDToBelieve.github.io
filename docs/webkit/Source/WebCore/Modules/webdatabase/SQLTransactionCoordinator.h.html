33
34#include <wtf/Deque.h>
35#include <wtf/HashMap.h>
36#include <wtf/HashSet.h>
37#include <wtf/RefPtr.h>
38#include <wtf/text/StringHash.h>
39
40namespace WebCore {
41
42class SQLTransaction;
43
44class SQLTransactionCoordinator {
45 WTF_MAKE_NONCOPYABLE(SQLTransactionCoordinator); WTF_MAKE_FAST_ALLOCATED;
46public:
47 SQLTransactionCoordinator();
48 void acquireLock(SQLTransaction&);
49 void releaseLock(SQLTransaction&);
50 void shutdown();
51private:
52 typedef Deque<RefPtr<SQLTransaction>> TransactionsQueue;
53 struct CoordinationInfo {
54 TransactionsQueue pendingTransactions;
55 HashSet<RefPtr<SQLTransaction>> activeReadTransactions;
56 RefPtr<SQLTransaction> activeWriteTransaction;
57 };
58 // Maps database names to information about pending transactions
59 typedef HashMap<String, CoordinationInfo> CoordinationInfoMap;
60 CoordinationInfoMap m_coordinationInfoMap;
61 bool m_isShuttingDown;
62
63 void processPendingTransactions(CoordinationInfo&);
64};

66} // namespace WebCore
