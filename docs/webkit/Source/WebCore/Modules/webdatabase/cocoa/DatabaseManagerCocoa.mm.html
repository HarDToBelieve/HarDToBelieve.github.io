32#import <wtf/FileSystem.h>

34namespace WebCore {
35
36void DatabaseManager::platformInitialize(const String& databasePath)
37{
38 if (databasePath.isEmpty())
39 return;
40
41 // The path needs to exist so that we can resolve it via realPath().
42 if (!FileSystem::fileExists(databasePath))
43 FileSystem::makeAllDirectories(databasePath);
44
45 // Disable vnode-guard related simulated crashes for WebSQL (rdar://problem/40674034). We do not overwrite the value here as it may have already been set by WKTR / DRT.
46 int overwrite = 0;
47 setenv("SQLITE_EXEMPT_PATH_FROM_VNODE_GUARDS", FileSystem::realPath(databasePath).utf8().data(), overwrite);
48}
