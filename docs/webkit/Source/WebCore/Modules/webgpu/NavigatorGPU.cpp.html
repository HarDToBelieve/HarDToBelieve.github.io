36NavigatorGPU* NavigatorGPU::from(Navigator* navigator)
37{
38 NavigatorGPU* supplement = static_cast<NavigatorGPU*>(Supplement<Navigator>::from(navigator, supplementName()));
39 if (!supplement) {
40 auto newSupplement = makeUnique<NavigatorGPU>();
41 supplement = newSupplement.get();
42 provideTo(navigator, supplementName(), WTFMove(newSupplement));
43 }
44 return supplement;
45}
46
47const char* NavigatorGPU::supplementName()
48{
49 return "NavigatorGPU";
50}
51
52const WebGPU* NavigatorGPU::gpu(Navigator& navigator)
53{
54 auto scriptExecutionContext = navigator.scriptExecutionContext();
55 if (!scriptExecutionContext)
56 return nullptr;
57
58 ASSERT(scriptExecutionContext->settingsValues().webGPUEnabled);
59 return NavigatorGPU::from(&navigator)->gpu();
60}
61
62const WebGPU* NavigatorGPU::gpu() const
63{
64 if (!m_gpu)
65 m_gpu = WebGPU::create();
66 return m_gpu.get();
67}
