30#include "WHLSLBaseSemantic.h"
31#include "WHLSLCodeLocation.h"
32#include <wtf/FastMalloc.h>

34namespace WebCore {
35
36namespace WHLSL {
37
38namespace AST {
39
40class BuiltInSemantic : public BaseSemantic {
41 WTF_MAKE_FAST_ALLOCATED;
42public:
43 enum class Variable : uint8_t {
44 SVInstanceID,
45 SVVertexID,
46 PSize,
47 SVPosition,
48 SVIsFrontFace,
49 SVSampleIndex,
50 SVInnerCoverage,
51 SVTarget,
52 SVDepth,
53 SVCoverage,
54 SVDispatchThreadID,
55 SVGroupID,
56 SVGroupIndex,
57 SVGroupThreadID
58 };
59
60 BuiltInSemantic(CodeLocation location, Variable variable, std::optional<unsigned>&& targetIndex = std::nullopt)
61 : BaseSemantic(location)
62 , m_variable(variable)
63 , m_targetIndex(WTFMove(targetIndex))
64 {
65 }
66
67 virtual ~BuiltInSemantic() = default;
68
69 BuiltInSemantic(const BuiltInSemantic&) = delete;
70 BuiltInSemantic(BuiltInSemantic&&) = default;
71
72 Variable variable() const { return m_variable; }
73 std::optional<unsigned>& targetIndex() { return m_targetIndex; }
74
75 bool operator==(const BuiltInSemantic& other) const
76 {
77 return m_variable == other.m_variable && m_targetIndex == other.m_targetIndex;
78 }
79
80 bool operator!=(const BuiltInSemantic& other) const
81 {
82 return !(*this == other);
83 }
84
85 StringView toString() const
86 {
87 switch (m_variable) {
88 case Variable::SVInstanceID:
89 return "SVInstanceID";
90 case Variable::SVVertexID:
91 return "SVVertexID";
92 case Variable::PSize:
93 return "PSize";
94 case Variable::SVPosition:
95 return "SVPosition";
96 case Variable::SVIsFrontFace:
97 return "SVIsFrontFace";
98 case Variable::SVSampleIndex:
99 return "SVSampleIndex";
100 case Variable::SVInnerCoverage:
101 return "SVInnerCoverage";
102 case Variable::SVTarget:
103 return "SVTarget";
104 case Variable::SVDepth:
105 return "SVDepth";
106 case Variable::SVCoverage:
107 return "SVCoverage";
108 case Variable::SVDispatchThreadID:
109 return "SVDispatchThreadID";
110 case Variable::SVGroupID:
111 return "SVGroupID";
112 case Variable::SVGroupIndex:
113 return "SVGroupIndex";
114 case Variable::SVGroupThreadID:
115 return "SVGroupThreadID";
116 }
117 }
118
119 bool isAcceptableType(const UnnamedType&, const Intrinsics&) const override;
120 bool isAcceptableForShaderItemDirection(ShaderItemDirection, const std::optional<EntryPointType>&) const override;
121
122private:
123 Variable m_variable;
124 std::optional<unsigned> m_targetIndex;
125};
126
127} // namespace AST

129}

131}
