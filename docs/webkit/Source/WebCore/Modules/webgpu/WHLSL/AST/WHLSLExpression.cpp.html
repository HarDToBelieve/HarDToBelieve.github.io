165bool Expression::mayBeEffectful() const
166{
167 auto& expression = const_cast<Expression&>(*this);
168
169 switch (expression.kind()) {
170 case Expression::Kind::BooleanLiteral:
171 case Expression::Kind::FloatLiteral:
172 case Expression::Kind::IntegerLiteral:
173 case Expression::Kind::UnsignedIntegerLiteral:
174 case Expression::Kind::EnumerationMemberLiteral:
175 case Expression::Kind::GlobalVariableReference:
176 case Expression::Kind::VariableReference:
177 return false;
178
179 case Expression::Kind::Dereference:
180 return downcast<DereferenceExpression>(expression).pointer().mayBeEffectful();
181
182 case Expression::Kind::Logical:
183 return downcast<LogicalExpression>(expression).left().mayBeEffectful() || downcast<LogicalExpression>(expression).right().mayBeEffectful();
184
185 case Expression::Kind::LogicalNot:
186 return downcast<LogicalNotExpression>(expression).operand().mayBeEffectful();
187
188 case Expression::Kind::MakeArrayReference:
189 return downcast<MakeArrayReferenceExpression>(expression).leftValue().mayBeEffectful();
190
191 case Expression::Kind::MakePointer:
192 return downcast<MakePointerExpression>(expression).leftValue().mayBeEffectful();
193
194 case Expression::Kind::Dot:
195 return downcast<DotExpression>(expression).base().mayBeEffectful();
196
197 case Expression::Kind::Index:
198 return downcast<IndexExpression>(expression).base().mayBeEffectful() || downcast<IndexExpression>(expression).indexExpression().mayBeEffectful();
199
200 default:
201 break;
202 }
203
204 return true;
205}
