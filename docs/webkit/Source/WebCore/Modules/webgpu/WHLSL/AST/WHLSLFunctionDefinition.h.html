40class FunctionDefinition : public FunctionDeclaration {
41 WTF_MAKE_FAST_ALLOCATED;
42public:
43 FunctionDefinition(FunctionDeclaration&& functionDeclaration, Block&& block)
44 : FunctionDeclaration(WTFMove(functionDeclaration))
45 , m_block(WTFMove(block))
46 {
47 }
48
49 virtual ~FunctionDefinition() = default;
50
51 FunctionDefinition(const FunctionDefinition&) = delete;
52 FunctionDefinition(FunctionDefinition&&) = default;
53
54 bool isFunctionDefinition() const override { return true; }
55
56 Block& block() { return m_block; }
57
58private:
59 Block m_block;
60};

62} // namespace AST
