39void Statement::destroy(Statement& statement)
40{
41 switch (statement.kind()) {
42 case Kind::Block:
43 delete &downcast<Block>(statement);
44 break;
45 case Kind::Break:
46 delete &downcast<Break>(statement);
47 break;
48 case Kind::Continue:
49 delete &downcast<Continue>(statement);
50 break;
51 case Kind::DoWhileLoop:
52 delete &downcast<DoWhileLoop>(statement);
53 break;
54 case Kind::EffectfulExpression:
55 delete &downcast<EffectfulExpressionStatement>(statement);
56 break;
57 case Kind::Fallthrough:
58 delete &downcast<Fallthrough>(statement);
59 break;
60 case Kind::ForLoop:
61 delete &downcast<ForLoop>(statement);
62 break;
63 case Kind::If:
64 delete &downcast<IfStatement>(statement);
65 break;
66 case Kind::Return:
67 delete &downcast<Return>(statement);
68 break;
69 case Kind::StatementList:
70 delete &downcast<StatementList>(statement);
71 break;
72 case Kind::SwitchCase:
73 delete &downcast<SwitchCase>(statement);
74 break;
75 case Kind::Switch:
76 delete &downcast<SwitchStatement>(statement);
77 break;
78 case Kind::VariableDeclarations:
79 delete &downcast<VariableDeclarationsStatement>(statement);
80 break;
81 case Kind::WhileLoop:
82 delete &downcast<WhileLoop>(statement);
83 break;
84 }
85}
86
87void Statement::destruct(Statement& statement)
88{
89 switch (statement.kind()) {
90 case Kind::Block:
91 downcast<Block>(statement).~Block();
92 break;
93 case Kind::Break:
94 downcast<Break>(statement).~Break();
95 break;
96 case Kind::Continue:
97 downcast<Continue>(statement).~Continue();
98 break;
99 case Kind::DoWhileLoop:
100 downcast<DoWhileLoop>(statement).~DoWhileLoop();
101 break;
102 case Kind::EffectfulExpression:
103 downcast<EffectfulExpressionStatement>(statement).~EffectfulExpressionStatement();
104 break;
105 case Kind::Fallthrough:
106 downcast<Fallthrough>(statement).~Fallthrough();
107 break;
108 case Kind::ForLoop:
109 downcast<ForLoop>(statement).~ForLoop();
110 break;
111 case Kind::If:
112 downcast<IfStatement>(statement).~IfStatement();
113 break;
114 case Kind::Return:
115 downcast<Return>(statement).~Return();
116 break;
117 case Kind::StatementList:
118 downcast<StatementList>(statement).~StatementList();
119 break;
120 case Kind::SwitchCase:
121 downcast<SwitchCase>(statement).~SwitchCase();
122 break;
123 case Kind::Switch:
124 downcast<SwitchStatement>(statement).~SwitchStatement();
125 break;
126 case Kind::VariableDeclarations:
127 downcast<VariableDeclarationsStatement>(statement).~VariableDeclarationsStatement();
128 break;
129 case Kind::WhileLoop:
130 downcast<WhileLoop>(statement).~WhileLoop();
131 break;
132 }
