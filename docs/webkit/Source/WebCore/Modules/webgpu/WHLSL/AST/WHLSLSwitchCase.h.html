30#include "WHLSLBlock.h"

34#include <wtf/FastMalloc.h>

36namespace WebCore {
37
38namespace WHLSL {
39
40namespace AST {
41
42class SwitchCase final : public Statement {
43 WTF_MAKE_FAST_ALLOCATED;
44public:
45 SwitchCase(CodeLocation location, std::optional<ConstantExpression>&& value, Block&& block)
46 : Statement(location, Kind::SwitchCase)
47 , m_value(WTFMove(value))
48 , m_block(WTFMove(block))
49 {
50 }
51
52 ~SwitchCase() = default;
53
54 SwitchCase(const SwitchCase&) = delete;
55 SwitchCase(SwitchCase&&) = default;
56
57 std::optional<ConstantExpression>& value() { return m_value; }
58 Block& block() { return m_block; }
59
60private:
61 std::optional<ConstantExpression> m_value;
62 Block m_block;
63};
64
65} // namespace AST
66
67}

69}

71DEFINE_DEFAULT_DELETE(SwitchCase)
