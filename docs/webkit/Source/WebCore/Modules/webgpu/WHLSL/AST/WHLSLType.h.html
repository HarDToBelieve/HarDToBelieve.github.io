34namespace WebCore {
35
36namespace WHLSL {
37
38namespace AST {
39
40class Type {
41 WTF_MAKE_FAST_ALLOCATED;
42
43protected:
44 ~Type() = default;
45
46public:
47 enum class Kind : uint8_t {
48 // UnnamedTypes
49 TypeReference,
50 Pointer,
51 ArrayReference,
52 Array,
53 // NamedTypes
54 TypeDefinition,
55 StructureDefinition,
56 EnumerationDefinition,
57 NativeTypeDeclaration,
58 // ResolvableTypes
59 FloatLiteral,
60 IntegerLiteral,
61 UnsignedIntegerLiteral,
62 };
63
64 Type(Kind kind)
65 : m_kind(kind)
66 { }
67 static void destroy(Type&);
68 static void destruct(Type&);
69
70 explicit Type(const Type&) = delete;
71 Type(Type&&) = default;
72
73 Type& operator=(const Type&) = delete;
74 Type& operator=(Type&&) = default;
75
76 Kind kind() const { return m_kind; }
77
78 bool isUnnamedType() const { return kind() >= Kind::TypeReference && kind() <= Kind::Array; }
79 bool isNamedType() const { return kind() >= Kind::TypeDefinition && kind() <= Kind::NativeTypeDeclaration; }
80 bool isResolvableType() const { return kind() >= Kind::FloatLiteral && kind() <= Kind::UnsignedIntegerLiteral; }
81
82 bool isTypeReference() const { return kind() == Kind::TypeReference; }
83 bool isPointerType() const { return kind() == Kind::Pointer; }
84 bool isArrayReferenceType() const { return kind() == Kind::ArrayReference; }
85 bool isArrayType() const { return kind() == Kind::Array; }
86 bool isReferenceType() const { return isPointerType() || isArrayReferenceType(); }
87
88 bool isTypeDefinition() const { return kind() == Kind::TypeDefinition; }
89 bool isStructureDefinition() const { return kind() == Kind::StructureDefinition; }
90 bool isEnumerationDefinition() const { return kind() == Kind::EnumerationDefinition; }
91 bool isNativeTypeDeclaration() const { return kind() == Kind::NativeTypeDeclaration; }
92
93 bool isFloatLiteralType() const { return kind() == Kind::FloatLiteral; }
94 bool isIntegerLiteralType() const { return kind() == Kind::IntegerLiteral; }
95 bool isUnsignedIntegerLiteralType() const { return kind() == Kind::UnsignedIntegerLiteral; }
96
97 Type& unifyNode();
98 const Type& unifyNode() const
99 {
100 return const_cast<Type*>(this)->unifyNode();
101 }
102
103private:
104 Kind m_kind;
105};
106
107} // namespace AST
108
109}

111}
