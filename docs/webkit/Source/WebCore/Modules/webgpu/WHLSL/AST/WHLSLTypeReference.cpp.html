40Ref<TypeReference> TypeReference::wrap(CodeLocation location, NamedType& resolvedType)
41{
42 TypeArguments typeArguments;
43 if (is<NativeTypeDeclaration>(resolvedType))
44 typeArguments = AST::clone(downcast<NativeTypeDeclaration>(resolvedType).typeArguments());
45 auto result = TypeReference::create(location, String(resolvedType.name()), WTFMove(typeArguments));
46 result->setResolvedType(resolvedType);
47 return result;
48}
