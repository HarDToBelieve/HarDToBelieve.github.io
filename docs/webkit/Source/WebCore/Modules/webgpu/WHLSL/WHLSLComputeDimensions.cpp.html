40std::optional<ComputeDimensions> computeDimensions(Program& program, AST::FunctionDefinition& entryPoint)
41{
42 std::optional<ComputeDimensions> computeDimensions;
43
44 for (auto& functionDefinition : program.functionDefinitions()) {
45 bool foundNumThreadsFunctionAttribute = false;
46 for (auto& functionAttribute : functionDefinition->attributeBlock()) {
47 auto success = WTF::visit(WTF::makeVisitor([&](AST::NumThreadsFunctionAttribute& numThreadsFunctionAttribute) {
48 if (foundNumThreadsFunctionAttribute)
49 return false;
50 foundNumThreadsFunctionAttribute = true;
51 if (&functionDefinition == &entryPoint) {
52 ASSERT(!computeDimensions);
53 computeDimensions = {{ numThreadsFunctionAttribute.width(), numThreadsFunctionAttribute.height(), numThreadsFunctionAttribute.depth() }};
54 }
55 return true;
56 }), functionAttribute);
57 if (!success) {
58 // Cannot declare multiple numthread attributes on a single function.
59 return std::nullopt;
60 }
61 }
62 }
63
64 return computeDimensions;
65}
