36namespace WHLSL {

38class HighZombieFinder : public Visitor {
39public:
40 void visit(AST::ReadModifyWriteExpression& readModifyWrite) override
41 {
42 checkErrorAndVisit(readModifyWrite.newValueExpression());
43 checkErrorAndVisit(readModifyWrite.resultExpression());
44
45 RELEASE_ASSERT(!readModifyWrite.leftValue().mayBeEffectful());
46 }
47
48 void visit(AST::DotExpression& dotExpression) override
49 {
50 RELEASE_ASSERT(!dotExpression.base().mayBeEffectful());
51 }
52
53 void visit(AST::IndexExpression& indexExpression) override
54 {
55 RELEASE_ASSERT(!indexExpression.base().mayBeEffectful());
56 RELEASE_ASSERT(!indexExpression.indexExpression().mayBeEffectful());
57 }
58};
59
60void findHighZombies(Program& program)
61{
62 UNUSED_PARAM(program);
63
64#if ASSERT_ENABLED
65 HighZombieFinder finder;
66 finder.Visitor::visit(program);
67#endif
68}
