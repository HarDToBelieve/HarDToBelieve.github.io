47static bool matches(const AST::Type& unifyThis, const AST::Type& unifyOther)
48{
49 if (&unifyThis == &unifyOther)
50 return true;
51
52 if (is<AST::NamedType>(unifyThis) && is<AST::NamedType>(unifyOther)) {
53#if ASSERT_ENABLED
54 auto& namedThis = downcast<AST::NamedType>(unifyThis);
55 auto& namedOther = downcast<AST::NamedType>(unifyOther);
56 ASSERT(!is<AST::TypeDefinition>(namedThis) && !is<AST::TypeDefinition>(namedOther));
57#endif
58 return false;
59 }
60 if (is<AST::UnnamedType>(unifyThis) && is<AST::UnnamedType>(unifyOther)) {
61 auto& unnamedThis = downcast<AST::UnnamedType>(unifyThis);
62 auto& unnamedOther = downcast<AST::UnnamedType>(unifyOther);
63 ASSERT(!is<AST::TypeReference>(unnamedThis) && !is<AST::TypeReference>(unnamedOther));
64 return unnamedThis == unnamedOther;
65 }
66 return false;
67}

84static RefPtr<AST::UnnamedType> matchAndCommit(AST::Type& unifyNode, AST::ResolvableType& resolvableType)
85{
86 ASSERT(!resolvableType.maybeResolvedType());
87 if (!resolvableType.canResolve(unifyNode))
88 return nullptr;
89 if (is<AST::NamedType>(unifyNode)) {
90 auto& namedUnifyNode = downcast<AST::NamedType>(unifyNode);
91 auto result = AST::TypeReference::wrap(namedUnifyNode.codeLocation(), namedUnifyNode);
92 resolvableType.resolve(result.copyRef());
93 return { WTFMove(result) };
94 }
95
96 Ref<AST::UnnamedType> result = downcast<AST::UnnamedType>(unifyNode);
97 resolvableType.resolve(result.copyRef());
98 return result;
99}

111RefPtr<AST::UnnamedType> matchAndCommit(AST::ResolvableType& resolvableType1, AST::ResolvableType& resolvableType2)
112{
113 ASSERT(!resolvableType1.maybeResolvedType());
114 ASSERT(!resolvableType2.maybeResolvedType());
115 if (is<AST::FloatLiteralType>(resolvableType1) && is<AST::FloatLiteralType>(resolvableType2)) {
116 resolvableType1.resolve(downcast<AST::FloatLiteralType>(resolvableType1).preferredType());
117 resolvableType2.resolve(downcast<AST::FloatLiteralType>(resolvableType2).preferredType());
118 return &downcast<AST::FloatLiteralType>(resolvableType1).preferredType();
119 }
120 if (is<AST::IntegerLiteralType>(resolvableType1) && is<AST::IntegerLiteralType>(resolvableType2)) {
121 resolvableType1.resolve(downcast<AST::IntegerLiteralType>(resolvableType1).preferredType());
122 resolvableType2.resolve(downcast<AST::IntegerLiteralType>(resolvableType2).preferredType());
123 return &downcast<AST::IntegerLiteralType>(resolvableType1).preferredType();
124 }
125 if (is<AST::UnsignedIntegerLiteralType>(resolvableType1) && is<AST::UnsignedIntegerLiteralType>(resolvableType2)) {
126 resolvableType1.resolve(downcast<AST::UnsignedIntegerLiteralType>(resolvableType1).preferredType());
127 resolvableType2.resolve(downcast<AST::UnsignedIntegerLiteralType>(resolvableType2).preferredType());
128 return &downcast<AST::UnsignedIntegerLiteralType>(resolvableType1).preferredType();
129 }
130 return nullptr;
131}

133RefPtr<AST::UnnamedType> commit(AST::ResolvableType& resolvableType)
134{
135 ASSERT(!resolvableType.maybeResolvedType());
136 if (is<AST::FloatLiteralType>(resolvableType)) {
137 auto& floatLiteralType = downcast<AST::FloatLiteralType>(resolvableType);
138 resolvableType.resolve(floatLiteralType.preferredType());
139 return &floatLiteralType.preferredType();
140 }
141 if (is<AST::IntegerLiteralType>(resolvableType)) {
142 auto& integerLiteralType = downcast<AST::IntegerLiteralType>(resolvableType);
143 resolvableType.resolve(integerLiteralType.preferredType());
144 return &integerLiteralType.preferredType();
145 }
146 if (is<AST::UnsignedIntegerLiteralType>(resolvableType)) {
147 auto& unsignedIntegerLiteralType = downcast<AST::UnsignedIntegerLiteralType>(resolvableType);
148 resolvableType.resolve(unsignedIntegerLiteralType.preferredType());
149 return &unsignedIntegerLiteralType.preferredType();
150 }
151 return nullptr;
152}

154bool inferTypesForTypeArguments(AST::NamedType& possibleType, AST::TypeArguments& typeArguments)
155{
156 if (is<AST::TypeDefinition>(possibleType)
157 || is<AST::StructureDefinition>(possibleType)
158 || is<AST::EnumerationDefinition>(possibleType)) {
159 return typeArguments.isEmpty();
160 }
161
162 auto& nativeTypeDeclaration = downcast<AST::NativeTypeDeclaration>(possibleType);
163 if (nativeTypeDeclaration.typeArguments().size() != typeArguments.size())
164 return false;
165 for (size_t i = 0; i < nativeTypeDeclaration.typeArguments().size(); ++i) {
166 AST::ConstantExpression* typeArgumentExpression = nullptr;
167 AST::TypeReference* typeArgumentTypeReference = nullptr;
168 AST::ConstantExpression* nativeTypeArgumentExpression = nullptr;
169 AST::TypeReference* nativeTypeArgumentTypeReference = nullptr;
170
171 auto assign = [&](AST::TypeArgument& typeArgument, AST::ConstantExpression*& expression, AST::TypeReference*& typeReference) {
172 WTF::visit(WTF::makeVisitor([&](AST::ConstantExpression& constantExpression) {
173 expression = &constantExpression;
174 }, [&](Ref<AST::TypeReference>& theTypeReference) {
175 typeReference = theTypeReference.ptr();
176 }), typeArgument);
177 };
178
179 assign(typeArguments[i], typeArgumentExpression, typeArgumentTypeReference);
180 assign(nativeTypeDeclaration.typeArguments()[i], nativeTypeArgumentExpression, nativeTypeArgumentTypeReference);
181
182 if (typeArgumentExpression && nativeTypeArgumentExpression) {
183 if (!typeArgumentExpression->matches(*nativeTypeArgumentExpression))
184 return false;
185 } else if (typeArgumentTypeReference && nativeTypeArgumentTypeReference) {
186 if (!matches(*typeArgumentTypeReference, *nativeTypeArgumentTypeReference))
187 return false;
188 }
189 }
190
191 return true;
192}

194template <typename TypeKind>
195ALWAYS_INLINE bool inferTypesForCallImpl(AST::FunctionDeclaration& possibleFunction, Vector<std::reference_wrapper<ResolvingType>>& argumentTypes, const TypeKind* castReturnType)
196{
197 if (possibleFunction.parameters().size() != argumentTypes.size())
198 return false;
199 for (size_t i = 0; i < possibleFunction.parameters().size(); ++i) {
200 auto success = argumentTypes[i].get().visit(WTF::makeVisitor([&](Ref<AST::UnnamedType>& unnamedType) -> bool {
201 return matches(*possibleFunction.parameters()[i]->type(), unnamedType);
202 }, [&](RefPtr<ResolvableTypeReference>& resolvableTypeReference) -> bool {
203 return resolvableTypeReference->resolvableType().canResolve(possibleFunction.parameters()[i]->type()->unifyNode());
204 }));
205 if (!success)
206 return false;
207 }
208 if (castReturnType && !matches(possibleFunction.type(), *castReturnType))
209 return false;
210 return true;
211}

213bool inferTypesForCall(AST::FunctionDeclaration& possibleFunction, Vector<std::reference_wrapper<ResolvingType>>& argumentTypes, const AST::NamedType* castReturnType)
214{
215 return inferTypesForCallImpl(possibleFunction, argumentTypes, castReturnType);
216}

218bool inferTypesForCall(AST::FunctionDeclaration& possibleFunction, Vector<std::reference_wrapper<ResolvingType>>& argumentTypes, const AST::UnnamedType* castReturnType)
219{
220 return inferTypesForCallImpl(possibleFunction, argumentTypes, castReturnType);
