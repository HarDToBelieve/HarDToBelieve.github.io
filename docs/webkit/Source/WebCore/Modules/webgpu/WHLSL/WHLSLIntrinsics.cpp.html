49void Intrinsics::add(AST::NativeFunctionDeclaration& nativeFunctionDeclaration)

63bool Intrinsics::addPrimitive(AST::NativeTypeDeclaration& nativeTypeDeclaration)
64{
65 if (nativeTypeDeclaration.typeArguments().size())
66 return false;
67
68 if (nativeTypeDeclaration.name() == "void")
69 m_voidType = &nativeTypeDeclaration;
70 else if (nativeTypeDeclaration.name() == "bool")
71 m_boolType = &nativeTypeDeclaration;
72 else if (nativeTypeDeclaration.name() == "uint") {
73 nativeTypeDeclaration.setIsInt();
74 nativeTypeDeclaration.setIsNumber();
75 nativeTypeDeclaration.setCanRepresentInteger([](int x) {
76 return x >= 0;
77 });
78 nativeTypeDeclaration.setCanRepresentUnsignedInteger([](unsigned) {
79 return true;
80 });
81 nativeTypeDeclaration.setCanRepresentFloat([](float x) {
82 return static_cast<float>(static_cast<uint32_t>(x)) == x;
83 });
84 nativeTypeDeclaration.setSuccessor([](int64_t x) -> int64_t {
85 return static_cast<uint32_t>(x + 1);
86 });
87 nativeTypeDeclaration.setFormatValueFromInteger([](int x) -> int64_t {
88 return static_cast<uint32_t>(x);
89 });
90 nativeTypeDeclaration.setFormatValueFromUnsignedInteger([](unsigned x) -> int64_t {
91 return static_cast<uint32_t>(x);
92 });
93 nativeTypeDeclaration.setIterateAllValues([](const std::function<bool(int64_t)>& callback) {
94 for (int64_t i = 0; i < 0x100000000; ++i) {
95 if (callback(i))
96 break;
97 }
98 });
99 m_uintType = &nativeTypeDeclaration;
100 } else if (nativeTypeDeclaration.name() == "int") {
101 nativeTypeDeclaration.setIsInt();
102 nativeTypeDeclaration.setIsNumber();
103 nativeTypeDeclaration.setIsSigned();
104 nativeTypeDeclaration.setCanRepresentInteger([](int) {
105 return true;
106 });
107 nativeTypeDeclaration.setCanRepresentUnsignedInteger([](unsigned x) {
108 return x <= 2147483647;
109 });
110 nativeTypeDeclaration.setCanRepresentFloat([](float x) {
111 return static_cast<float>(static_cast<int32_t>(x)) == x;
112 });
113 nativeTypeDeclaration.setSuccessor([](int64_t x) -> int64_t {
114 return static_cast<int32_t>(x + 1);
115 });
116 nativeTypeDeclaration.setFormatValueFromInteger([](int x) -> int64_t {
117 return static_cast<int32_t>(x);
118 });
119 nativeTypeDeclaration.setFormatValueFromUnsignedInteger([](unsigned x) -> int64_t {
120 return static_cast<int32_t>(x);
121 });
122 nativeTypeDeclaration.setIterateAllValues([](const std::function<bool(int64_t)>& callback) {
123 for (int64_t i = -2147483647; i < 2147483648; ++i) {
124 if (callback(i))
125 break;
126 }
127 });
128 m_intType = &nativeTypeDeclaration;
129 } else if (nativeTypeDeclaration.name() == "float") {
130 nativeTypeDeclaration.setIsNumber();
131 nativeTypeDeclaration.setIsFloating();
132 nativeTypeDeclaration.setCanRepresentInteger([](int) {
133 return true;
134 });
135 nativeTypeDeclaration.setCanRepresentUnsignedInteger([](unsigned) {
136 return true;
137 });
138 nativeTypeDeclaration.setCanRepresentFloat([](float) {
139 return true;
140 });
141 m_floatType = &nativeTypeDeclaration;
142 } else if (nativeTypeDeclaration.name() == "atomic_int") {
143 nativeTypeDeclaration.setIsAtomic();
144 m_atomicIntType = &nativeTypeDeclaration;
145 } else if (nativeTypeDeclaration.name() == "atomic_uint") {
146 nativeTypeDeclaration.setIsAtomic();
147 m_atomicUintType = &nativeTypeDeclaration;
148 } else if (nativeTypeDeclaration.name() == "sampler") {
149 m_samplerType = &nativeTypeDeclaration;
150 nativeTypeDeclaration.setIsOpaqueType();
151 } else
152 ASSERT_NOT_REACHED();
153 return true;
154}

156bool Intrinsics::addVector(AST::NativeTypeDeclaration& nativeTypeDeclaration)
157{
158 if (nativeTypeDeclaration.name() != "vector")
159 return false;
160
161 ASSERT(nativeTypeDeclaration.typeArguments().size() == 2);
162 ASSERT(WTF::holds_alternative<Ref<AST::TypeReference>>(nativeTypeDeclaration.typeArguments()[0]));
163 ASSERT(WTF::holds_alternative<AST::ConstantExpression>(nativeTypeDeclaration.typeArguments()[1]));
164 auto& innerType = static_cast<AST::TypeReference&>(WTF::get<Ref<AST::TypeReference>>(nativeTypeDeclaration.typeArguments()[0]));
165 auto& lengthExpression = WTF::get<AST::ConstantExpression>(nativeTypeDeclaration.typeArguments()[1]);
166 ASSERT(!innerType.typeArguments().size());
167 AST::NativeTypeDeclaration** array;
168 if (innerType.name() == "bool")
169 array = m_vectorBool;
170 else if (innerType.name() == "uint")
171 array = m_vectorUint;
172 else if (innerType.name() == "int")
173 array = m_vectorInt;
174 else {
175 ASSERT(innerType.name() == "float");
176 array = m_vectorFloat;
177 }
178 int length = lengthExpression.integerLiteral().value();
179 ASSERT(length >= 2 && length <= 4);
180 nativeTypeDeclaration.setIsVector();
181 array[length - 2] = &nativeTypeDeclaration;
182 return true;
183}

195 ASSERT(WTF::holds_alternative<AST::ConstantExpression>(nativeTypeDeclaration.typeArguments()[2]));
196 return true;
197}

199bool Intrinsics::addFullTexture(AST::NativeTypeDeclaration& nativeTypeDeclaration, AST::TypeReference& innerType)
200{
201 auto textureTypeIndex = std::find(m_textureTypeNames, m_textureTypeNames + WTF_ARRAY_LENGTH(m_textureTypeNames), nativeTypeDeclaration.name()) - m_textureTypeNames;
202 if (textureTypeIndex == WTF_ARRAY_LENGTH(m_textureTypeNames))
203 return false;
204
205 unsigned innerTypeIndex = WTF_ARRAY_LENGTH(m_textureInnerTypeNames);
206 unsigned vectorLength = 0;
207 for (unsigned i = 0; i < WTF_ARRAY_LENGTH(m_textureInnerTypeNames); ++i) {
208 if (innerType.name().startsWith(m_textureInnerTypeNames[i])) {
209 innerTypeIndex = i;
210 if (innerType.name() == m_textureInnerTypeNames[i])
211 vectorLength = 1;
212 else {
213 ASSERT_WITH_SECURITY_IMPLICATION(innerType.name().length() == strlen(m_textureInnerTypeNames[i]) + 1);
214 ASSERT_WITH_SECURITY_IMPLICATION(innerType.name()[innerType.name().length() - 1] == '2'
215 || innerType.name()[innerType.name().length() - 1] == '3'
216 || innerType.name()[innerType.name().length() - 1] == '4');
217 vectorLength = innerType.name()[innerType.name().length() - 1] - '0';
218 }
219 }
220 }
221 RELEASE_ASSERT(vectorLength && vectorLength < 5);
222 ASSERT_WITH_SECURITY_IMPLICATION(innerTypeIndex != WTF_ARRAY_LENGTH(m_textureInnerTypeNames));
223 nativeTypeDeclaration.setIsTexture();
224 nativeTypeDeclaration.setIsOpaqueType();
225 if (nativeTypeDeclaration.name() == "Texture1DArray" || nativeTypeDeclaration.name() == "RWTexture1DArray" || nativeTypeDeclaration.name() == "Texture2DArray" || nativeTypeDeclaration.name() == "RWTexture2DArray")
226 nativeTypeDeclaration.setIsTextureArray();
227 if (nativeTypeDeclaration.name() == "RWTexture1D" || nativeTypeDeclaration.name() == "RWTexture2D" || nativeTypeDeclaration.name() == "RWTexture3D" || nativeTypeDeclaration.name() == "RWTexture1DArray" || nativeTypeDeclaration.name() == "RWTexture2DArray")
228 nativeTypeDeclaration.setIsWritableTexture();
229 if (nativeTypeDeclaration.name() == "Texture1D" || nativeTypeDeclaration.name() == "RWTexture1D" || nativeTypeDeclaration.name() == "Texture1DArray" || nativeTypeDeclaration.name() == "RWTexture1DArray")
230 nativeTypeDeclaration.setTextureDimension(1);
231 if (nativeTypeDeclaration.name() == "Texture2D" || nativeTypeDeclaration.name() == "RWTexture2D" || nativeTypeDeclaration.name() == "TextureCube" || nativeTypeDeclaration.name() == "Texture2DArray" || nativeTypeDeclaration.name() == "RWTexture2DArray")
232 nativeTypeDeclaration.setTextureDimension(2);
233 if (nativeTypeDeclaration.name() == "Texture3D" || nativeTypeDeclaration.name() == "RWTexture3D")
234 nativeTypeDeclaration.setTextureDimension(3);
235 if (nativeTypeDeclaration.name() == "TextureCube")
236 nativeTypeDeclaration.setIsCubeTexture();
237 m_fullTextures[textureTypeIndex][innerTypeIndex][vectorLength - 1] = &nativeTypeDeclaration;
238 return true;
239}

241void Intrinsics::addDepthTexture(AST::NativeTypeDeclaration& nativeTypeDeclaration, AST::TypeReference& innerType)
242{
243 AST::NativeTypeDeclaration** texture = nullptr;
244 if (nativeTypeDeclaration.name() == "TextureDepth2D")
245 texture = m_textureDepth2D;
246 else if (nativeTypeDeclaration.name() == "TextureDepth2DArray")
247 texture = m_textureDepth2DArray;
248 else {
249 ASSERT(nativeTypeDeclaration.name() == "TextureDepthCube");
250 texture = m_textureDepthCube;
251 }
252 auto innerTypeIndex = std::find(m_depthTextureInnerTypes, m_depthTextureInnerTypes + WTF_ARRAY_LENGTH(m_depthTextureInnerTypes), innerType.name()) - m_depthTextureInnerTypes;
253 ASSERT(innerTypeIndex != WTF_ARRAY_LENGTH(m_depthTextureInnerTypes));
254 nativeTypeDeclaration.setIsTexture();
255 nativeTypeDeclaration.setIsOpaqueType();
256 if (texture == m_textureDepth2DArray)
257 nativeTypeDeclaration.setIsTextureArray();
258 if (texture == m_textureDepthCube)
259 nativeTypeDeclaration.setIsCubeTexture();
260 nativeTypeDeclaration.setTextureDimension(2);
261 nativeTypeDeclaration.setIsDepthTexture();
262 texture[innerTypeIndex] = &nativeTypeDeclaration;
263}

265void Intrinsics::addTexture(AST::NativeTypeDeclaration& nativeTypeDeclaration)
266{
267 ASSERT(nativeTypeDeclaration.typeArguments().size() == 1);
268 ASSERT(WTF::holds_alternative<Ref<AST::TypeReference>>(nativeTypeDeclaration.typeArguments()[0]));
269 auto& innerType = static_cast<AST::TypeReference&>(WTF::get<Ref<AST::TypeReference>>(nativeTypeDeclaration.typeArguments()[0]));
270 ASSERT(!innerType.typeArguments().size());
271 if (addFullTexture(nativeTypeDeclaration, innerType)) {
272 m_textureSet.add(&nativeTypeDeclaration);
273 return;
274 }
275 addDepthTexture(nativeTypeDeclaration, innerType);
276 m_textureSet.add(&nativeTypeDeclaration);
277}
278
279void Intrinsics::add(AST::NativeTypeDeclaration& nativeTypeDeclaration)
280{
281 if (addPrimitive(nativeTypeDeclaration))
282 return;
283 if (addVector(nativeTypeDeclaration))
284 return;
285 if (addMatrix(nativeTypeDeclaration))
286 return;
287 addTexture(nativeTypeDeclaration);
288}
