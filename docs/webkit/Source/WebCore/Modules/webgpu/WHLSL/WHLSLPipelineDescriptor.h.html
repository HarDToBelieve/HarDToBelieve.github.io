38enum class VertexFormat : uint8_t {
39 FloatR32G32B32A32,
40 FloatR32G32B32,
41 FloatR32G32,
42 FloatR32
43};

45struct VertexAttribute {
46 VertexFormat vertexFormat;
47 unsigned shaderLocation;
48 unsigned metalLocation;
49};

51using VertexAttributes = Vector<VertexAttribute>;

91struct AttachmentDescriptor {
92 TextureFormat textureFormat;
93 unsigned name;
94};
95
96struct AttachmentsStateDescriptor {
97 Vector<AttachmentDescriptor> attachmentDescriptors;
98 std::optional<AttachmentDescriptor> depthStencilAttachmentDescriptor;
99};
100
101enum class ShaderStage : uint8_t {
102 None = 0,
103 Vertex = 1 << 0,
104 Fragment = 1 << 1,
105 Compute = 1 << 2
106};
107
108struct UniformBufferBinding {
109 unsigned lengthName;
110};
111
112struct SamplerBinding {
113};
114
115struct TextureBinding {
116};
117
118struct StorageBufferBinding {
119 unsigned lengthName;
120};
121
122// FIXME: https://bugs.webkit.org/show_bug.cgi?id=198168 Add the dynamic types
123
124struct Binding {
125 using BindingDetails = Variant<UniformBufferBinding, SamplerBinding, TextureBinding, StorageBufferBinding>;
126 OptionSet<ShaderStage> visibility;
127 BindingDetails binding;
128 unsigned internalName;
129 unsigned externalName;
130};
131
132struct BindGroup {
133 Vector<Binding> bindings;
134 unsigned name;
135};
136
137using Layout = Vector<BindGroup>;
138
139struct RenderPipelineDescriptor {
140 VertexAttributes vertexAttributes;
141 AttachmentsStateDescriptor attachmentsStateDescriptor;
142 Layout layout;
143 String vertexEntryPointName;
144 String fragmentEntryPointName;
145};
146
147struct ComputePipelineDescriptor {
148 Layout layout;
149 String entryPointName;
150};
151
152using PipelineDescriptor = Variant<RenderPipelineDescriptor, ComputePipelineDescriptor>;
