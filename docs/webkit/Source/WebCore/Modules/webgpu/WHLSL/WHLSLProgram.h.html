41namespace WebCore {
42
43namespace WHLSL {
44
45class Program {
46public:
47 Program() = default;
48 Program(Program&&) = default;
49
50 Expected<void, Error> append(AST::TypeDefinition&& typeDefinition)
51 {
52 m_typeDefinitions.append(makeUniqueRef<AST::TypeDefinition>(WTFMove(typeDefinition)));
53 return m_nameContext.add(m_typeDefinitions.last());
54 }
55
56 Expected<void, Error> append(AST::StructureDefinition&& structureDefinition)
57 {
58 m_structureDefinitions.append(makeUniqueRef<AST::StructureDefinition>(WTFMove(structureDefinition)));
59 return m_nameContext.add(m_structureDefinitions.last());
60 }
61
62 Expected<void, Error> append(AST::EnumerationDefinition&& enumerationDefinition)
63 {
64 m_enumerationDefinitions.append(makeUniqueRef<AST::EnumerationDefinition>(WTFMove(enumerationDefinition)));
65 return m_nameContext.add(m_enumerationDefinitions.last());
66 }
67
68 Expected<void, Error> append(AST::FunctionDefinition&& functionDefinition)
69 {
70 m_functionDefinitions.append(makeUniqueRef<AST::FunctionDefinition>(WTFMove(functionDefinition)));
71 return m_nameContext.add(m_functionDefinitions.last());
72 }
73
74 Expected<void, Error> append(AST::NativeFunctionDeclaration&& nativeFunctionDeclaration)
75 {
76 m_nativeFunctionDeclarations.append(makeUniqueRef<AST::NativeFunctionDeclaration>(WTFMove(nativeFunctionDeclaration)));
77 m_intrinsics.add(m_nativeFunctionDeclarations.last());
78 return m_nameContext.add(m_nativeFunctionDeclarations.last());
79 }
80
81 Expected<void, Error> append(AST::NativeTypeDeclaration&& nativeTypeDeclaration)
82 {
83 m_nativeTypeDeclarations.append(makeUniqueRef<AST::NativeTypeDeclaration>(WTFMove(nativeTypeDeclaration)));
84 m_intrinsics.add(m_nativeTypeDeclarations.last());
85 return m_nameContext.add(m_nativeTypeDeclarations.last());
86 }
87
88 bool isValidVectorProperty(String);
89
90 NameContext& nameContext() { return m_nameContext; }
91 Intrinsics& intrinsics() { return m_intrinsics; }
92 const Intrinsics& intrinsics() const { return m_intrinsics; }
93 Vector<UniqueRef<AST::TypeDefinition>>& typeDefinitions() { return m_typeDefinitions; }
94 Vector<UniqueRef<AST::StructureDefinition>>& structureDefinitions() { return m_structureDefinitions; }
95 Vector<UniqueRef<AST::EnumerationDefinition>>& enumerationDefinitions() { return m_enumerationDefinitions; }
96 const Vector<UniqueRef<AST::FunctionDefinition>>& functionDefinitions() const { return m_functionDefinitions; }
97 Vector<UniqueRef<AST::FunctionDefinition>>& functionDefinitions() { return m_functionDefinitions; }
98 const Vector<UniqueRef<AST::NativeFunctionDeclaration>>& nativeFunctionDeclarations() const { return m_nativeFunctionDeclarations; }
99 Vector<UniqueRef<AST::NativeFunctionDeclaration>>& nativeFunctionDeclarations() { return m_nativeFunctionDeclarations; }
100 Vector<UniqueRef<AST::NativeTypeDeclaration>>& nativeTypeDeclarations() { return m_nativeTypeDeclarations; }
101
102private:
103 NameContext m_nameContext;
104 Intrinsics m_intrinsics;
105 Vector<UniqueRef<AST::TypeDefinition>> m_typeDefinitions;
106 Vector<UniqueRef<AST::StructureDefinition>> m_structureDefinitions;
107 Vector<UniqueRef<AST::EnumerationDefinition>> m_enumerationDefinitions;
108 Vector<UniqueRef<AST::FunctionDefinition>> m_functionDefinitions;
109 Vector<UniqueRef<AST::NativeFunctionDeclaration>> m_nativeFunctionDeclarations;
110 Vector<UniqueRef<AST::NativeTypeDeclaration>> m_nativeTypeDeclarations;
111};
