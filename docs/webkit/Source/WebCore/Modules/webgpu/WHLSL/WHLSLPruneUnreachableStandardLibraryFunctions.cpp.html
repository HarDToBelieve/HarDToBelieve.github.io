35namespace WebCore {

37namespace WHLSL {
38
39class ReachableStdlibFunctions final : public Visitor {
40public:
41 void visit(AST::FunctionDefinition& function) override
42 {
43 auto addResult = m_reachableFunctions.add(&function);
44 if (addResult.isNewEntry)
45 Visitor::visit(function);
46 }
47
48 void visit(AST::CallExpression& callExpression) override
49 {
50 Visitor::visit(callExpression);
51 if (is<AST::FunctionDefinition>(callExpression.function()))
52 visit(downcast<AST::FunctionDefinition>(callExpression.function()));
53 }
54
55 HashSet<AST::FunctionDefinition*> takeReachableFunctions() { return WTFMove(m_reachableFunctions); }
56
57private:
58 HashSet<AST::FunctionDefinition*> m_reachableFunctions;
59};
60
61void pruneUnreachableStandardLibraryFunctions(Program& program)
62{
63 ReachableStdlibFunctions reachableStdlibFunctions;
64 Vector<UniqueRef<AST::FunctionDefinition>> functionDefinitions = WTFMove(program.functionDefinitions());
65 for (auto& function : functionDefinitions) {
66 if (function->parsingMode() != ParsingMode::StandardLibrary)
67 reachableStdlibFunctions.visit(function.get());
68 else
69 RELEASE_ASSERT(!function->entryPointType());
70 }
71
72 auto reachableFunctions = reachableStdlibFunctions.takeReachableFunctions();
73 Vector<UniqueRef<AST::FunctionDefinition>> newFunctionDefinitions;
74 for (UniqueRef<AST::FunctionDefinition>& entry : functionDefinitions) {
75 if (reachableFunctions.contains(&entry.get()))
76 newFunctionDefinitions.append(WTFMove(entry));
77 }
78
79 program.functionDefinitions() = WTFMove(newFunctionDefinitions);
80}
