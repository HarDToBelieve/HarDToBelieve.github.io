38Ref<WebGPUAdapter> WebGPUAdapter::create(std::optional<GPURequestAdapterOptions>&& options)
39{
40 return adoptRef(*new WebGPUAdapter(WTFMove(options)));
41}
42
43WebGPUAdapter::WebGPUAdapter(std::optional<GPURequestAdapterOptions>&& options)
44 : m_options(WTFMove(options))
45{
46}
47
48void WebGPUAdapter::requestDevice(Document& document, DeviceRequestPromise&& promise) const
49{
50 document.postTask([protectedThis = makeRef(*this), promise = WTFMove(promise)] (ScriptExecutionContext& context) mutable {
51 if (auto device = WebGPUDevice::tryCreate(context, protectedThis.get())) {
52 InspectorInstrumentation::didCreateWebGPUDevice(*device);
53
54 promise.resolve(device.releaseNonNull());
55 } else
56 promise.reject();
57 });
58}
