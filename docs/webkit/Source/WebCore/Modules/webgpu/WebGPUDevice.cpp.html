98RefPtr<WebGPUDevice> WebGPUDevice::tryCreate(ScriptExecutionContext& context, Ref<const WebGPUAdapter>&& adapter)
99{
100 if (auto device = GPUDevice::tryCreate(adapter->options())) {
101 auto result = adoptRef(new WebGPUDevice(context, WTFMove(adapter), makeRef(*device)));
102 device->setErrorScopes(result->m_errorScopes.copyRef());
103 return result;
104 }
105 return nullptr;
106}

119WebGPUDevice::WebGPUDevice(ScriptExecutionContext& context, Ref<const WebGPUAdapter>&& adapter, Ref<GPUDevice>&& device)
120 : m_scriptExecutionContext(context)
121 , m_adapter(WTFMove(adapter))
122 , m_device(WTFMove(device))
123 , m_errorScopes(GPUErrorScopes::create([this, weakThis = makeWeakPtr(this)] (GPUError&& error) {
124 if (weakThis)
125 dispatchUncapturedError(WTFMove(error));
126 }))
127{
128 ASSERT(m_scriptExecutionContext.isDocument());
129
130 {
131 Locker locker { instancesLock() };
132 instances().add(this);
133 }
134}
135
136WebGPUDevice::~WebGPUDevice()
137{
138 InspectorInstrumentation::willDestroyWebGPUDevice(*this);
139
140 {
141 Locker locker { WebGPUPipeline::instancesLock() };
142 for (auto& entry : WebGPUPipeline::instances()) {
143 if (entry.value == this) {
144 // Don't remove any WebGPUPipeline from the instances list, as they may still exist.
145 // Only remove the association with a WebGPU device.
146 entry.value = nullptr;
147 }
148 }
149 }
150
151 {
152 Locker locker { instancesLock() };
153 ASSERT(instances().contains(this));
154 instances().remove(this);
155 }
156}

166Vector<JSC::JSValue> WebGPUDevice::createBufferMapped(JSC::JSGlobalObject& lexicalGlobalObject, const GPUBufferDescriptor& descriptor) const
167{
168 m_errorScopes->setErrorPrefix("GPUDevice.createBufferMapped(): ");
169
170 JSC::JSValue wrappedArrayBuffer = JSC::jsNull();
171
172 auto buffer = m_device->tryCreateBuffer(descriptor, GPUBufferMappedOption::IsMapped, m_errorScopes);
173 if (buffer) {
174 auto arrayBuffer = buffer->mapOnCreation();
175 wrappedArrayBuffer = toJS(&lexicalGlobalObject, JSC::jsCast<JSDOMGlobalObject*>(&lexicalGlobalObject), arrayBuffer);
176 }
177
178 auto webBuffer = WebGPUBuffer::create(WTFMove(buffer), m_errorScopes);
179 auto wrappedWebBuffer = toJS(&lexicalGlobalObject, JSC::jsCast<JSDOMGlobalObject*>(&lexicalGlobalObject), webBuffer);
180
181 return { wrappedWebBuffer, wrappedArrayBuffer };
182}

231Ref<WebGPURenderPipeline> WebGPUDevice::createRenderPipeline(const WebGPURenderPipelineDescriptor& descriptor)
232{
233 m_errorScopes->setErrorPrefix("GPUDevice.createRenderPipeline(): ");
234
235 auto gpuDescriptor = descriptor.tryCreateGPURenderPipelineDescriptor(m_errorScopes);
236 if (!gpuDescriptor)
237 return WebGPURenderPipeline::create(*this, nullptr, m_errorScopes, { }, { });
238
239 auto gpuPipeline = m_device->tryCreateRenderPipeline(*gpuDescriptor, m_errorScopes);
240
241 WebGPUPipeline::ShaderData vertexShader = { descriptor.vertexStage.module, descriptor.vertexStage.entryPoint };
242
243 WebGPUPipeline::ShaderData fragmentShader;
244 if (descriptor.fragmentStage)
245 fragmentShader = { descriptor.fragmentStage.value().module, descriptor.fragmentStage.value().entryPoint };
246
247 auto webGPUPipeline = WebGPURenderPipeline::create(*this, WTFMove(gpuPipeline), m_errorScopes, WTFMove(vertexShader), WTFMove(fragmentShader));
248 if (webGPUPipeline->isValid())
249 InspectorInstrumentation::didCreateWebGPUPipeline(*this, webGPUPipeline.get());
250 return webGPUPipeline;
251}

285void WebGPUDevice::popErrorScope(ErrorPromise&& promise)
286{
287 String failMessage;
288 std::optional<GPUError> error = m_errorScopes->popErrorScope(failMessage);
289 if (failMessage.isEmpty())
290 promise.resolve(error);
291 else
292 promise.reject(Exception { OperationError, "GPUDevice::popErrorScope(): " + failMessage });
293}

295// Errors reported via the validation error event should also appear in the console as warnings.
296static void printValidationErrorToConsole(GPUError& error, ScriptExecutionContext& context)
297{
298 if (!WTF::holds_alternative<RefPtr<GPUValidationError>>(error))
299 return;
300
301 auto validationError = WTF::get<RefPtr<GPUValidationError>>(error);
302 if (!validationError)
303 return;
304
305 auto message = validationError->message();
306 auto consoleMessage = makeUnique<Inspector::ConsoleMessage>(MessageSource::Rendering, MessageType::Log, MessageLevel::Warning, message);
307
308 downcast<Document>(context).addConsoleMessage(WTFMove(consoleMessage));
309}
310
311void WebGPUDevice::dispatchUncapturedError(GPUError&& error)
312{
313 printValidationErrorToConsole(error, m_scriptExecutionContext);
314
315 callOnMainThread([error = WTFMove(error), this, protectedThis = makeRef(*this)] () mutable {
316 dispatchEvent(GPUUncapturedErrorEvent::create(eventNames().uncapturederrorEvent, WTFMove(error)));
317 });
318}
