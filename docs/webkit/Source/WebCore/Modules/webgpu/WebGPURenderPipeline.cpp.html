48WebGPURenderPipeline::WebGPURenderPipeline(WebGPUDevice& device, RefPtr<GPURenderPipeline>&& pipeline, GPUErrorScopes& errorScopes, WebGPUPipeline::ShaderData&& vertexShader, WebGPUPipeline::ShaderData&& fragmentShader)

58bool WebGPURenderPipeline::cloneShaderModules(const WebGPUDevice& device)
59{
60 if (m_vertexShader.module) {
61 bool sharesVertexFragmentShaderModule = m_fragmentShader.module == m_vertexShader.module;
62
63 const auto& vertexSource = m_vertexShader.module->source();
64 m_vertexShader.module = WebGPUShaderModule::create(GPUShaderModule::tryCreate(device.device(), { vertexSource }), vertexSource);
65
66 if (m_fragmentShader.module) {
67 if (sharesVertexFragmentShaderModule)
68 m_fragmentShader.module = m_vertexShader.module;
69 else {
70 const auto& fragmentSource = m_fragmentShader.module->source();
71 m_fragmentShader.module = WebGPUShaderModule::create(GPUShaderModule::tryCreate(device.device(), { fragmentSource }), fragmentSource);
72 }
73 }
74
75 return true;
76 }
77 return false;
78}
79
80bool WebGPURenderPipeline::recompile(const WebGPUDevice& device)
81{
82 if (m_renderPipeline && m_vertexShader.module) {
83 if (auto* gpuVertexShaderModule = m_vertexShader.module->module()) {
84 GPUProgrammableStageDescriptor vertexStage(makeRef(*gpuVertexShaderModule), { m_vertexShader.entryPoint });
85 std::optional<GPUProgrammableStageDescriptor> fragmentStage;
86 if (m_fragmentShader.module) {
87 if (auto* gpuFragmentShaderModule = m_fragmentShader.module->module())
88 fragmentStage = GPUProgrammableStageDescriptor(makeRef(*gpuFragmentShaderModule), { m_fragmentShader.entryPoint });
89 }
90 return m_renderPipeline->recompile(device.device(), WTFMove(vertexStage), WTFMove(fragmentStage));
91 }
92 }
93 return false;
94}
