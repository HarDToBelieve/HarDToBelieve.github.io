35std::optional<GPURenderPipelineDescriptor> WebGPURenderPipelineDescriptor::tryCreateGPURenderPipelineDescriptor(GPUErrorScopes& errorScopes) const
36{
37 auto pipelineLayout = layout ? makeRefPtr(layout->pipelineLayout()) : nullptr;
38
39 auto vertex = vertexStage.tryCreateGPUProgrammableStageDescriptor();
40
41 std::optional<GPUProgrammableStageDescriptor> fragment;
42 if (fragmentStage)
43 fragment = fragmentStage->tryCreateGPUProgrammableStageDescriptor();
44
45 if (!vertex || (fragmentStage && !fragment)) {
46 errorScopes.generatePrefixedError("Invalid GPUProgrammableStageDescriptor!");
47 return std::nullopt;
48 }
49
50 // FIXME: Web GPU validation, e.g. fail if colorStates is larger than (max number of supported color states).
51
52 return GPURenderPipelineDescriptor { WTFMove(pipelineLayout), WTFMove(*vertex), WTFMove(fragment), *this };
53}
