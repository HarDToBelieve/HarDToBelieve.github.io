31namespace WebCore {

33Ref<WebGPUTextureView> WebGPUTextureView::create(RefPtr<GPUTexture>&& view)
34{
35 return adoptRef(*new WebGPUTextureView(WTFMove(view)));
36}
37
38WebGPUTextureView::WebGPUTextureView(RefPtr<GPUTexture>&& view)
39 : m_texture(WTFMove(view))
40{
41}
42
43void WebGPUTextureView::destroy()
44{
45 if (!m_texture)
46 return;
47
48 m_texture->destroy();
49 m_texture = nullptr;
50}
