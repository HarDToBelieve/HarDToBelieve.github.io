53Ref<ThreadableWebSocketChannel> ThreadableWebSocketChannel::create(Document& document, WebSocketChannelClient& client, SocketProvider& provider)
54{
55#if USE(SOUP)
56 auto channel = provider.createWebSocketChannel(document, client);
57 ASSERT(channel);
58 return channel.releaseNonNull();
59#else
60
61#if HAVE(NSURLSESSION_WEBSOCKET)
62 if (RuntimeEnabledFeatures::sharedFeatures().isNSURLSessionWebSocketEnabled()) {
63 if (auto channel = provider.createWebSocketChannel(document, client))
64 return channel.releaseNonNull();
65 }
66#endif
67
68 return WebSocketChannel::create(document, client, provider);
69#endif
70}

72Ref<ThreadableWebSocketChannel> ThreadableWebSocketChannel::create(ScriptExecutionContext& context, WebSocketChannelClient& client, SocketProvider& provider)
73{
74 if (is<WorkerGlobalScope>(context)) {
75 WorkerGlobalScope& workerGlobalScope = downcast<WorkerGlobalScope>(context);
76 WorkerRunLoop& runLoop = workerGlobalScope.thread().runLoop();
77 return WorkerThreadableWebSocketChannel::create(workerGlobalScope, client, makeString("webSocketChannelMode", runLoop.createUniqueId()), provider);
78 }
79
80 return create(downcast<Document>(context), client, provider);
81}

88std::optional<ThreadableWebSocketChannel::ValidatedURL> ThreadableWebSocketChannel::validateURL(Document& document, const URL& requestedURL)
89{
90 ValidatedURL validatedURL { requestedURL, true };
91 if (auto* page = document.page()) {
92 if (!page->allowsLoadFromURL(requestedURL))
93 return { };
94#if ENABLE(CONTENT_EXTENSIONS)
95 if (auto* documentLoader = document.loader()) {
96 auto results = page->userContentProvider().processContentRuleListsForLoad(*page, validatedURL.url, ContentExtensions::ResourceType::WebSocket, *documentLoader);
97 if (results.summary.blockedLoad)
98 return { };
99 if (results.summary.madeHTTPS) {
100 ASSERT(validatedURL.url.protocolIs("ws"));
101 validatedURL.url.setProtocol("wss");
102 }
103 validatedURL.areCookiesAllowed = !results.summary.blockedCookies;
104 }
105#else
106 UNUSED_PARAM(document);
107#endif
108 }
109 return validatedURL;
110}

112std::optional<ResourceRequest> ThreadableWebSocketChannel::webSocketConnectRequest(Document& document, const URL& url)
113{
114 auto validatedURL = validateURL(document, url);
115 if (!validatedURL)
116 return { };
117
118 ResourceRequest request { validatedURL->url };
119 request.setHTTPUserAgent(document.userAgent(validatedURL->url));
120 request.setDomainForCachePartition(document.domainForCachePartition());
121 request.setAllowCookies(validatedURL->areCookiesAllowed);
122 request.setFirstPartyForCookies(document.firstPartyForCookies());
123 request.setHTTPHeaderField(HTTPHeaderName::Origin, document.securityOrigin().toString());
124 FrameLoader::addSameSiteInfoToRequestIfNeeded(request, &document);
125
126 // Add no-cache headers to avoid compatibility issue.
127 // There are some proxies that rewrite "Connection: upgrade"
128 // to "Connection: close" in the response if a request doesn't contain
129 // these headers.
130 request.addHTTPHeaderField(HTTPHeaderName::Pragma, HTTPHeaderValues::noCache());
131 request.addHTTPHeaderField(HTTPHeaderName::CacheControl, HTTPHeaderValues::noCache());
132
133 return request;
134}
