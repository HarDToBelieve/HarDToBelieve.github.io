42namespace WebCore {
43
44class ScriptExecutionContext;
45class WebSocketChannelClient;
46
47class ThreadableWebSocketChannelClientWrapper : public ThreadSafeRefCounted<ThreadableWebSocketChannelClientWrapper> {
48public:
49 static Ref<ThreadableWebSocketChannelClientWrapper> create(ScriptExecutionContext&, WebSocketChannelClient&);
50
51 void clearSyncMethodDone();
52 void setSyncMethodDone();
53 bool syncMethodDone() const;
54
55 WorkerThreadableWebSocketChannel::Peer* peer() const;
56 void didCreateWebSocketChannel(WorkerThreadableWebSocketChannel::Peer*);
57 void clearPeer();
58
59 bool failedWebSocketChannelCreation() const;
60 void setFailedWebSocketChannelCreation();
61
62 // Subprotocol and extensions will be available when didConnect() callback is invoked.
63 String subprotocol() const;
64 void setSubprotocol(const String&);
65 String extensions() const;
66 void setExtensions(const String&);
67
68 ThreadableWebSocketChannel::SendResult sendRequestResult() const;
69 void setSendRequestResult(ThreadableWebSocketChannel::SendResult);
70
71 unsigned bufferedAmount() const;
72 void setBufferedAmount(unsigned);
73
74 void clearClient();
75
76 void didConnect();
77 void didReceiveMessage(const String& message);
78 void didReceiveBinaryData(Vector<uint8_t>&&);
79 void didUpdateBufferedAmount(unsigned bufferedAmount);
80 void didStartClosingHandshake();
81 void didClose(unsigned unhandledBufferedAmount, WebSocketChannelClient::ClosingHandshakeCompletionStatus, unsigned short code, const String& reason);
82 void didReceiveMessageError();
83 void didUpgradeURL();
84
85 void suspend();
86 void resume();
87
88private:
89 ThreadableWebSocketChannelClientWrapper(ScriptExecutionContext&, WebSocketChannelClient&);
90
91 void processPendingTasks();
92
93 ScriptExecutionContext& m_context;
94 WebSocketChannelClient* m_client;
95 WorkerThreadableWebSocketChannel::Peer* m_peer;
96 bool m_failedWebSocketChannelCreation;
97 bool m_syncMethodDone;
98 // ThreadSafeRefCounted must not have String member variables.
99 Vector<UChar> m_subprotocol;
100 Vector<UChar> m_extensions;
101 ThreadableWebSocketChannel::SendResult m_sendRequestResult;
102 unsigned m_bufferedAmount;
103 bool m_suspended;
104 Vector<std::unique_ptr<ScriptExecutionContext::Task>> m_pendingTasks;
105};
