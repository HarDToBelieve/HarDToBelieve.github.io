31#pragma once
32
33#include <wtf/Forward.h>
34#include <wtf/WeakPtr.h>
35
36namespace WebCore {
37
38class WebSocketChannelClient : public CanMakeWeakPtr<WebSocketChannelClient> {
39public:
40 virtual ~WebSocketChannelClient() = default;
41 virtual void didConnect() = 0;
42 virtual void didReceiveMessage(const String&) = 0;
43 virtual void didReceiveBinaryData(Vector<uint8_t>&&) = 0;
44 virtual void didReceiveMessageError() = 0;
45 virtual void didUpdateBufferedAmount(unsigned bufferedAmount) = 0;
46 virtual void didStartClosingHandshake() = 0;
47 enum ClosingHandshakeCompletionStatus {
48 ClosingHandshakeIncomplete,
49 ClosingHandshakeComplete
50 };
51 virtual void didClose(unsigned unhandledBufferedAmount, ClosingHandshakeCompletionStatus, unsigned short code, const String& reason) = 0;
52 virtual void didUpgradeURL() = 0;
53
54protected:
55 WebSocketChannelClient() = default;
56};
57
58} // namespace WebCore
