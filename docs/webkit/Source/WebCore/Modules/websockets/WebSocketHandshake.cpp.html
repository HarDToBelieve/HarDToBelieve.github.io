64static String resourceName(const URL& url)
65{
66 auto path = url.path();
67 auto result = makeString(
68 path,
69 path.isEmpty() ? "/" : "",
70 url.queryWithLeadingQuestionMark()
71 );
72 ASSERT(!result.isEmpty());
73 ASSERT(!result.contains(' '));
74 return result;
75}

77static String hostName(const URL& url, bool secure)
78{
79 ASSERT(url.protocolIs("wss") == secure);
80 String host = url.host().convertToASCIILowercase();
81 if (url.port() && ((!secure && url.port().value() != 80) || (secure && url.port().value() != 443)))
82 return makeString(host, ':', url.port().value());
83 return host;
84}

86static const size_t maxInputSampleSize = 128;
87static String trimInputSample(const uint8_t* p, size_t len)
88{
89 String s = String(p, std::min<size_t>(len, maxInputSampleSize));
90 if (len > maxInputSampleSize)
91 s.append(horizontalEllipsis);
92 return s;
93}

95static String generateSecWebSocketKey()
96{
97 static const size_t nonceSize = 16;
98 unsigned char key[nonceSize];
99 cryptographicallyRandomValues(key, nonceSize);
100 return base64EncodeToString(key, nonceSize);
101}

103String WebSocketHandshake::getExpectedWebSocketAccept(const String& secWebSocketKey)
104{
105 constexpr uint8_t webSocketKeyGUID[] = "258EAFA5-E914-47DA-95CA-C5AB0DC85B11";
106 SHA1 sha1;
107 CString keyData = secWebSocketKey.ascii();
108 sha1.addBytes(keyData.dataAsUInt8Ptr(), keyData.length());
109 sha1.addBytes(webSocketKeyGUID, std::size(webSocketKeyGUID) - 1);
110 SHA1::Digest hash;
111 sha1.computeHash(hash);
112 return base64EncodeToString(hash.data(), SHA1::hashSize);
113}

166CString WebSocketHandshake::clientHandshakeMessage() const
167{
168 // Keep the following consistent with clientHandshakeRequest just below.
169
170 // Cookies are not retrieved in the WebContent process. Instead, a proxy object is
171 // added in the handshake, and is exchanged for actual cookies in the Network process.
172
173 // Add no-cache headers to avoid a compatibility issue. There are some proxies that
174 // rewrite "Connection: upgrade" to "Connection: close" in the response if a request
175 // doesn't contain these headers.
176
177 auto extensions = m_extensionDispatcher.createHeaderValue();
178 return makeString("GET ", resourceName(m_url), " HTTP/1.1\r\n"
179 "Upgrade: websocket\r\n"
180 "Connection: Upgrade\r\n"
181 "Host: ", hostName(m_url, m_secure), "\r\n"
182 "Origin: ", m_clientOrigin, "\r\n",
183 m_clientProtocol.isEmpty() ? "" : "Sec-WebSocket-Protocol: ", m_clientProtocol, m_clientProtocol.isEmpty() ? "" : "\r\n",
184 "Pragma: no-cache\r\n"
185 "Cache-Control: no-cache\r\n"
186 "Sec-WebSocket-Key: ", m_secWebSocketKey, "\r\n"
187 "Sec-WebSocket-Version: 13\r\n",
188 extensions.isEmpty() ? "" : "Sec-WebSocket-Extensions: ", extensions, extensions.isEmpty() ? "" : "\r\n",
189 "User-Agent: ", m_userAgent, "\r\n"
190 "\r\n").utf8();
191}

193ResourceRequest WebSocketHandshake::clientHandshakeRequest(const Function<String(const URL&)>& cookieRequestHeaderFieldValue) const
194{
195 // Keep the following consistent with clientHandshakeMessage just above.
196
197 auto cookie = m_allowCookies ? cookieRequestHeaderFieldValue(httpURLForAuthenticationAndCookies()) : emptyString();
198 auto extensions = m_extensionDispatcher.createHeaderValue();
199 ResourceRequest request { m_url };
200 request.setHTTPMethod("GET"_s);
201 request.setHTTPHeaderField(HTTPHeaderName::Connection, "Upgrade"_s);
202 request.setHTTPHeaderField(HTTPHeaderName::Host, hostName(m_url, m_secure));
203 request.setHTTPHeaderField(HTTPHeaderName::Origin, m_clientOrigin);
204 if (!m_clientProtocol.isEmpty())
205 request.setHTTPHeaderField(HTTPHeaderName::SecWebSocketProtocol, m_clientProtocol);
206 if (!cookie.isEmpty())
207 request.setHTTPHeaderField(HTTPHeaderName::Cookie, cookie);
208 request.setHTTPHeaderField(HTTPHeaderName::Pragma, HTTPHeaderValues::noCache());
209 request.setHTTPHeaderField(HTTPHeaderName::CacheControl, HTTPHeaderValues::noCache());
210 request.setHTTPHeaderField(HTTPHeaderName::SecWebSocketKey, m_secWebSocketKey);
211 request.setHTTPHeaderField(HTTPHeaderName::SecWebSocketVersion, "13"_s);
212 if (!extensions.isEmpty())
213 request.setHTTPHeaderField(HTTPHeaderName::SecWebSocketExtensions, extensions);
214 request.setHTTPUserAgent(m_userAgent);
215 return request;
216}

224int WebSocketHandshake::readServerHandshake(const uint8_t* header, size_t len)
225{
226 m_mode = Incomplete;
227 int statusCode;
228 String statusText;
229 int lineLength = readStatusLine(header, len, statusCode, statusText);
230 if (lineLength == -1)
231 return -1;
232 if (statusCode == -1) {
233 m_mode = Failed; // m_failureReason is set inside readStatusLine().
234 return len;
235 }
236 LOG(Network, "WebSocketHandshake %p readServerHandshake() Status code is %d", this, statusCode);
237
238 m_serverHandshakeResponse = ResourceResponse();
239 m_serverHandshakeResponse.setHTTPStatusCode(statusCode);
240 m_serverHandshakeResponse.setHTTPStatusText(statusText);
241
242 if (statusCode != 101) {
243 m_mode = Failed;
244 m_failureReason = makeString("Unexpected response code: ", statusCode);
245 return len;
246 }
247 m_mode = Normal;
248 if (!memmem(header, len, "\r\n\r\n", 4)) {
249 // Just hasn't been received fully yet.
250 m_mode = Incomplete;
251 return -1;
252 }
253 auto p = readHTTPHeaders(header + lineLength, header + len);
254 if (!p) {
255 LOG(Network, "WebSocketHandshake %p readServerHandshake() readHTTPHeaders() failed", this);
256 m_mode = Failed; // m_failureReason is set inside readHTTPHeaders().
257 return len;
258 }
259 if (!checkResponseHeaders()) {
260 LOG(Network, "WebSocketHandshake %p readServerHandshake() checkResponseHeaders() failed", this);
261 m_mode = Failed;
262 return p - header;
263 }
264
265 m_mode = Connected;
266 return p - header;
267}

327// https://tools.ietf.org/html/rfc6455#section-4.1
328// "The HTTP version MUST be at least 1.1."
329static inline bool headerHasValidHTTPVersion(StringView httpStatusLine)
330{
331 constexpr char preamble[] = "HTTP/";
332 if (!httpStatusLine.startsWith(preamble))
333 return false;
334
335 // Check that there is a version number which should be at least three characters after "HTTP/"
336 unsigned preambleLength = strlen(preamble);
337 if (httpStatusLine.length() < preambleLength + 3)
338 return false;
339
340 auto dotPosition = httpStatusLine.find('.', preambleLength);
341 if (dotPosition == notFound)
342 return false;
343
344 auto majorVersion = parseInteger<int>(httpStatusLine.substring(preambleLength, dotPosition - preambleLength));
345 if (!majorVersion)
346 return false;
347
348 unsigned minorVersionLength;
349 unsigned charactersLeftAfterDotPosition = httpStatusLine.length() - dotPosition;
350 for (minorVersionLength = 1; minorVersionLength < charactersLeftAfterDotPosition; minorVersionLength++) {
351 if (!isASCIIDigit(httpStatusLine[dotPosition + minorVersionLength]))
352 break;
353 }
354 auto minorVersion = parseInteger<int>(httpStatusLine.substring(dotPosition + 1, minorVersionLength));
355 if (!minorVersion)
356 return false;
357
358 return (*majorVersion >= 1 && *minorVersion >= 1) || *majorVersion >= 2;
359}

361// Returns the header length (including "\r\n"), or -1 if we have not received enough data yet.
362// If the line is malformed or the status code is not a 3-digit number,
363// statusCode and statusText will be set to -1 and a null string, respectively.
364int WebSocketHandshake::readStatusLine(const uint8_t* header, size_t headerLength, int& statusCode, String& statusText)
365{
366 // Arbitrary size limit to prevent the server from sending an unbounded
367 // amount of data with no newlines and forcing us to buffer it all.
368 static const int maximumLength = 1024;
369
370 statusCode = -1;
371 statusText = String();
372
373 const uint8_t* space1 = nullptr;
374 const uint8_t* space2 = nullptr;
375 const uint8_t* p;
376 size_t consumedLength;
377
378 for (p = header, consumedLength = 0; consumedLength < headerLength; p++, consumedLength++) {
379 if (*p == ' ') {
380 if (!space1)
381 space1 = p;
382 else if (!space2)
383 space2 = p;
384 } else if (*p == '\0') {
385 // The caller isn't prepared to deal with null bytes in status
386 // line. WebSockets specification doesn't prohibit this, but HTTP
387 // does, so we'll just treat this as an error.
388 m_failureReason = "Status line contains embedded null"_s;
389 return p + 1 - header;
390 } else if (!isASCII(*p)) {
391 m_failureReason = "Status line contains non-ASCII character"_s;
392 return p + 1 - header;
393 } else if (*p == '\n')
394 break;
395 }
396 if (consumedLength == headerLength)
397 return -1; // We have not received '\n' yet.
398
399 auto end = p + 1;
400 int lineLength = end - header;
401 if (lineLength > maximumLength) {
402 m_failureReason = "Status line is too long"_s;
403 return maximumLength;
404 }
405
406 // The line must end with "\r\n".
407 if (lineLength < 2 || *(end - 2) != '\r') {
408 m_failureReason = "Status line does not end with CRLF"_s;
409 return lineLength;
410 }
411
412 if (!space1 || !space2) {
413 m_failureReason = makeString("No response code found: ", trimInputSample(header, lineLength - 2));
414 return lineLength;
415 }
416
417 StringView httpStatusLine(header, space1 - header);
418 if (!headerHasValidHTTPVersion(httpStatusLine)) {
419 m_failureReason = makeString("Invalid HTTP version string: ", httpStatusLine);
420 return lineLength;
421 }
422
423 StringView statusCodeString(space1 + 1, space2 - space1 - 1);
424 if (statusCodeString.length() != 3) // Status code must consist of three digits.
425 return lineLength;
426 for (int i = 0; i < 3; ++i) {
427 if (!isASCIIDigit(statusCodeString[i])) {
428 m_failureReason = makeString("Invalid status code: ", statusCodeString);
429 return lineLength;
430 }
431 }
432
433 statusCode = parseInteger<int>(statusCodeString).value();
434 statusText = String(space2 + 1, end - space2 - 3); // Exclude "\r\n".
435 return lineLength;
436}

438const uint8_t* WebSocketHandshake::readHTTPHeaders(const uint8_t* start, const uint8_t* end)
439{
440 StringView name;
441 String value;
442 bool sawSecWebSocketExtensionsHeaderField = false;
443 bool sawSecWebSocketAcceptHeaderField = false;
444 bool sawSecWebSocketProtocolHeaderField = false;
445 auto p = start;
446 for (; p < end; p++) {
447 size_t consumedLength = parseHTTPHeader(p, end - p, m_failureReason, name, value);
448 if (!consumedLength)
449 return nullptr;
450 p += consumedLength;
451
452 // Stop once we consumed an empty line.
453 if (name.isEmpty())
454 break;
455
456 HTTPHeaderName headerName;
457 if (!findHTTPHeaderName(name, headerName)) {
458 // Evidence in the wild shows that services make use of custom headers in the handshake
459 m_serverHandshakeResponse.addHTTPHeaderField(name.toString(), value);
460 continue;
461 }
462
463 // https://tools.ietf.org/html/rfc7230#section-3.2.4
464 // "Newly defined header fields SHOULD limit their field values to US-ASCII octets."
465 if ((headerName == HTTPHeaderName::SecWebSocketExtensions
466 || headerName == HTTPHeaderName::SecWebSocketAccept
467 || headerName == HTTPHeaderName::SecWebSocketProtocol)
468 && !value.isAllASCII()) {
469 m_failureReason = makeString(name, " header value should only contain ASCII characters");
470 return nullptr;
471 }
472
473 if (headerName == HTTPHeaderName::SecWebSocketExtensions) {
474 if (sawSecWebSocketExtensionsHeaderField) {
475 m_failureReason = "The Sec-WebSocket-Extensions header must not appear more than once in an HTTP response"_s;
476 return nullptr;
477 }
478 if (!m_extensionDispatcher.processHeaderValue(value)) {
479 m_failureReason = m_extensionDispatcher.failureReason();
480 return nullptr;
481 }
482 sawSecWebSocketExtensionsHeaderField = true;
483 } else {
484 if (headerName == HTTPHeaderName::SecWebSocketAccept) {
485 if (sawSecWebSocketAcceptHeaderField) {
486 m_failureReason = "The Sec-WebSocket-Accept header must not appear more than once in an HTTP response"_s;
487 return nullptr;
488 }
489 sawSecWebSocketAcceptHeaderField = true;
490 } else if (headerName == HTTPHeaderName::SecWebSocketProtocol) {
491 if (sawSecWebSocketProtocolHeaderField) {
492 m_failureReason = "The Sec-WebSocket-Protocol header must not appear more than once in an HTTP response"_s;
493 return nullptr;
494 }
495 sawSecWebSocketProtocolHeaderField = true;
496 }
497
498 m_serverHandshakeResponse.addHTTPHeaderField(headerName, value);
499 }
500 }
501 return p;
502}

504bool WebSocketHandshake::checkResponseHeaders()
505{
506 const String& serverWebSocketProtocol = this->serverWebSocketProtocol();
507 const String& serverUpgrade = this->serverUpgrade();
508 const String& serverConnection = this->serverConnection();
509 const String& serverWebSocketAccept = this->serverWebSocketAccept();
510
511 if (serverUpgrade.isNull()) {
512 m_failureReason = "Error during WebSocket handshake: 'Upgrade' header is missing"_s;
513 return false;
514 }
515 if (serverConnection.isNull()) {
516 m_failureReason = "Error during WebSocket handshake: 'Connection' header is missing"_s;
517 return false;
518 }
519 if (serverWebSocketAccept.isNull()) {
520 m_failureReason = "Error during WebSocket handshake: 'Sec-WebSocket-Accept' header is missing"_s;
521 return false;
522 }
523
524 if (!equalLettersIgnoringASCIICase(serverUpgrade, "websocket")) {
525 m_failureReason = "Error during WebSocket handshake: 'Upgrade' header value is not 'WebSocket'"_s;
526 return false;
527 }
528 if (!equalLettersIgnoringASCIICase(serverConnection, "upgrade")) {
529 m_failureReason = "Error during WebSocket handshake: 'Connection' header value is not 'Upgrade'"_s;
530 return false;
531 }
532
533 if (serverWebSocketAccept != m_expectedAccept) {
534 m_failureReason = "Error during WebSocket handshake: Sec-WebSocket-Accept mismatch"_s;
535 return false;
536 }
537 if (!serverWebSocketProtocol.isNull()) {
538 if (m_clientProtocol.isEmpty()) {
539 m_failureReason = "Error during WebSocket handshake: Sec-WebSocket-Protocol mismatch"_s;
540 return false;
541 }
542 Vector<String> result = m_clientProtocol.split(WebSocket::subprotocolSeparator());
543 if (!result.contains(serverWebSocketProtocol)) {
544 m_failureReason = "Error during WebSocket handshake: Sec-WebSocket-Protocol mismatch"_s;
545 return false;
546 }
547 }
548 return true;
549}
