34namespace WebCore {
35
36WebXRSystem& NavigatorWebXR::xr(Navigator& navigatorObject)
37{
38 auto& navigator = NavigatorWebXR::from(navigatorObject);
39 if (!navigator.m_xr)
40 navigator.m_xr = WebXRSystem::create(navigatorObject);
41 return *navigator.m_xr;
42}
43
44NavigatorWebXR& NavigatorWebXR::from(Navigator& navigator)
45{
46 auto* supplement = static_cast<NavigatorWebXR*>(Supplement<Navigator>::from(&navigator, "NavigatorWebXR"));
47 if (!supplement) {
48 auto newSupplement = makeUnique<NavigatorWebXR>();
49 supplement = newSupplement.get();
50 provideTo(&navigator, "NavigatorWebXR", WTFMove(newSupplement));
51 }
52 return *supplement;
53}
