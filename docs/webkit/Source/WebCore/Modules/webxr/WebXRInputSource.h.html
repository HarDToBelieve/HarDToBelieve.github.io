26#pragma once
27
28#if ENABLE(WEBXR)
29
30#include "PlatformXR.h"
31#include "WebXRGamepad.h"
32#include "WebXRInputSpace.h"
33#include "XRHandedness.h"
34#include "XRTargetRayMode.h"
35#include <wtf/IsoMalloc.h>
36#include <wtf/Ref.h>
37#include <wtf/RefCounted.h>
38#include <wtf/RefPtr.h>
39#include <wtf/Vector.h>
40#include <wtf/text/WTFString.h>
41
42namespace WebCore {
43
44#if ENABLE(GAMEPAD)
45class Gamepad;
46#endif
47class XRInputSourceEvent;
48class WebXRInputSpace;
49
50class WebXRInputSource : public RefCounted<WebXRInputSource> {
51 WTF_MAKE_ISO_ALLOCATED(WebXRInputSource);
52public:
53 using InputSource = PlatformXR::Device::FrameData::InputSource;
54 using InputSourceButton = PlatformXR::Device::FrameData::InputSourceButton;
55
56 static Ref<WebXRInputSource> create(Document&, WebXRSession&, double timestamp, const InputSource&);
57 ~WebXRInputSource();
58
59 PlatformXR::InputSourceHandle handle() const { return m_source.handle; }
60 XRHandedness handedness() const { return m_source.handeness; }
61 XRTargetRayMode targetRayMode() const { return m_source.targetRayMode; };
62 const WebXRSpace& targetRaySpace() const {return m_targetRaySpace.get(); };
63 WebXRSpace* gripSpace() const { return m_gripSpace.get(); }
64 const Vector<String>& profiles() const { return m_source.profiles; };
65 double connectTime() const { return m_connectTime; }
66#if ENABLE(GAMEPAD)
67 const Gamepad* gamepad() const { return m_gamepad.ptr(); }
68#endif
69
70 void update(double timestamp, const InputSource&);
71 bool requiresInputSourceChange(const InputSource&);
72 void disconnect();
73
74 void pollEvents(Vector<Ref<XRInputSourceEvent>>&);
75
76 // For GC reachablitiy.
77 WebXRSession* session();
78
79private:
80 WebXRInputSource(Document&, WebXRSession&, double timestamp, const InputSource&);
81
82 WeakPtr<WebXRSession> m_session;
83 InputSource m_source;
84 Ref<WebXRInputSpace> m_targetRaySpace;
85 RefPtr<WebXRInputSpace> m_gripSpace;
86 double m_connectTime { 0 };
87 bool m_connected { true };
88#if ENABLE(GAMEPAD)
89 Ref<Gamepad> m_gamepad;
90#endif
91
92 bool m_selectStarted { false };
93 bool m_squeezeStarted { false };
94};
95
96} // namespace WebCore
97
98#endif // ENABLE(WEBXR)
