35#include <wtf/UniqueRef.h>

37namespace WebCore {
38
39class Document;
40class WebXRInputSource;
41class XRInputSourceEvent;
42class WebXRSession;
43
44class WebXRInputSourceArray final : public ScriptWrappable {
45 WTF_MAKE_ISO_ALLOCATED(WebXRInputSourceArray);
46 friend UniqueRef<WebXRInputSourceArray> WTF::makeUniqueRefWithoutFastMallocCheck<WebXRInputSourceArray, WebCore::WebXRSession&>(WebCore::WebXRSession&);
47public:
48 using InputSourceList = Vector<PlatformXR::Device::FrameData::InputSource>;
49 static UniqueRef<WebXRInputSourceArray> create(WebXRSession&);
50 ~WebXRInputSourceArray();
51
52 void ref();
53 void deref();
54
55 unsigned length() const;
56 WebXRInputSource* item(unsigned) const;
57
58 void clear();
59 void update(double timestamp, const InputSourceList&);
60
61 // For GC reachablitiy.
62 WebXRSession* session() const { return &m_session; }
63
64private:
65 WebXRInputSourceArray(WebXRSession&);
66
67 void handleRemovedInputSources(const InputSourceList&, Vector<RefPtr<WebXRInputSource>>&, Vector<Ref<XRInputSourceEvent>>&);
68 void handleAddedOrUpdatedInputSources(double timestamp, const InputSourceList&, Vector<RefPtr<WebXRInputSource>>&, Vector<RefPtr<WebXRInputSource>>&, Vector<Ref<XRInputSourceEvent>>&);
69
70 WebXRSession& m_session;
71 Vector<Ref<WebXRInputSource>> m_inputSources;
72};

74} // namespace WebCore

76#endif // ENABLE(WEBXR)
