36namespace WebCore {
37
38WTF_MAKE_ISO_ALLOCATED_IMPL(WebXRInputSpace);
39
40// WebXRInputSpace
41
42Ref<WebXRInputSpace> WebXRInputSpace::create(Document& document, WebXRSession& session, const PlatformXR::Device::FrameData::InputSourcePose& pose)
43{
44 return adoptRef(*new WebXRInputSpace(document, session, pose));
45}
46
47WebXRInputSpace::WebXRInputSpace(Document& document, WebXRSession& session, const PlatformXR::Device::FrameData::InputSourcePose& pose)
48 : WebXRSpace(document, WebXRRigidTransform::create())
49 , m_session(session)
50 , m_pose(pose)
51{
52}
53
54WebXRInputSpace::~WebXRInputSpace() = default;
55
56TransformationMatrix WebXRInputSpace::nativeOrigin() const
57{
58 return WebXRFrame::matrixFromPose(m_pose.pose);
59}
