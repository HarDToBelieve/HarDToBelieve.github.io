27
28#if ENABLE(WEBXR)
29
30#include "WebXRSpace.h"
31#include "XRHandedness.h"
32#include "XRTargetRayMode.h"
33#include <wtf/IsoMalloc.h>
34#include <wtf/RefCounted.h>
35
36namespace WebCore {
37
38class WebXRInputSpace : public RefCounted<WebXRInputSpace>, public WebXRSpace {
39 WTF_MAKE_ISO_ALLOCATED(WebXRInputSpace);
40public:
41 static Ref<WebXRInputSpace> create(Document&, WebXRSession&, const PlatformXR::Device::FrameData::InputSourcePose&);
42 virtual ~WebXRInputSpace();
43
44 using RefCounted<WebXRInputSpace>::ref;
45 using RefCounted<WebXRInputSpace>::deref;
46
47 bool isPositionEmulated() const final { return m_pose.isPositionEmulated; }
48 void setPose(const PlatformXR::Device::FrameData::InputSourcePose& pose) { m_pose = pose; }
49
50private:
51 WebXRInputSpace(Document&, WebXRSession&, const PlatformXR::Device::FrameData::InputSourcePose&);
52 WebXRSession& session() const final { return m_session.get(); }
53 TransformationMatrix nativeOrigin() const final;
54
55 void refEventTarget() final { ref(); }
56 void derefEventTarget() final { deref(); }
57
58 Ref<WebXRSession> m_session;
59 PlatformXR::Device::FrameData::InputSourcePose m_pose;
60};
61
62} // namespace WebCore
63
64#endif // ENABLE(WEBXR)
