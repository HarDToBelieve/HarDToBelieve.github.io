62std::unique_ptr<WebXROpaqueFramebuffer> WebXROpaqueFramebuffer::create(PlatformXR::LayerHandle handle, WebGLRenderingContextBase& context, Attributes&& attributes, uint32_t width, uint32_t height)
63{
64 auto framebuffer = WebGLFramebuffer::createOpaque(context);
65 auto opaque = std::unique_ptr<WebXROpaqueFramebuffer>(new WebXROpaqueFramebuffer(handle, WTFMove(framebuffer), context, WTFMove(attributes), width, height));
66 if (!opaque->setupFramebuffer())
67 return nullptr;
68 return opaque;
69}

71WebXROpaqueFramebuffer::WebXROpaqueFramebuffer(PlatformXR::LayerHandle handle, Ref<WebGLFramebuffer>&& framebuffer, WebGLRenderingContextBase& context, Attributes&& attributes, uint32_t width, uint32_t height)
72 : m_handle(handle)
73 , m_framebuffer(WTFMove(framebuffer))
74 , m_context(context)
75 , m_attributes(WTFMove(attributes))
76 , m_width(width)
77 , m_height(height)
78{
79}

81WebXROpaqueFramebuffer::~WebXROpaqueFramebuffer()
82{
83 if (auto gl = m_context.graphicsContextGL()) {
84#if USE(IOSURFACE_FOR_XR_LAYER_DATA)
85 if (m_opaqueTexture)
86 gl->deleteTexture(m_opaqueTexture);
87#endif
88 if (m_stencilBuffer)
89 gl->deleteRenderbuffer(m_stencilBuffer);
90 if (m_depthStencilBuffer)
91 gl->deleteRenderbuffer(m_depthStencilBuffer);
92 if (m_multisampleColorBuffer)
93 gl->deleteRenderbuffer(m_multisampleColorBuffer);
94 if (m_resolvedFBO)
95 gl->deleteFramebuffer(m_resolvedFBO);
96 m_context.deleteFramebuffer(m_framebuffer.ptr());
97 }
98}

100void WebXROpaqueFramebuffer::startFrame(const PlatformXR::Device::FrameData::LayerData& data)
101{
102 if (!m_context.graphicsContextGL())
103 return;
104 auto& gl = *m_context.graphicsContextGL();
105
106 m_framebuffer->setOpaqueActive(true);
107
108 GCGLint boundFBO { 0 };
109
110 gl.getIntegerv(GL::FRAMEBUFFER_BINDING, makeGCGLSpan(&boundFBO, 1));
111 auto scopedFBOs = makeScopeExit([&gl, boundFBO]() {
112 gl.bindFramebuffer(GL::FRAMEBUFFER, boundFBO);
113 });
114
115 gl.bindFramebuffer(GraphicsContextGL::FRAMEBUFFER, m_framebuffer->object());
116 // https://immersive-web.github.io/webxr/#opaque-framebuffer
117 // The buffers attached to an opaque framebuffer MUST be cleared to the values in the provided table when first created,
118 // or prior to the processing of each XR animation frame.
119 // FIXME: Actually do the clearing (not using invalidateFramebuffer). This will have to be done after we've attached
120 // the textures/renderbuffers.
121
122#if USE(IOSURFACE_FOR_XR_LAYER_DATA)
123 ASSERT(data.surface);
124
125 auto gCGL = static_cast<GraphicsContextGLOpenGL*>(m_context.graphicsContextGL());
126 GCGLenum textureTarget = GraphicsContextGLOpenGL::drawingBufferTextureTarget();
127
128
129 if (!m_opaqueTexture)
130 m_opaqueTexture = gCGL->createTexture();
131
132 gCGL->bindTexture(textureTarget, m_opaqueTexture);
133 gCGL->texParameteri(textureTarget, GL::TEXTURE_MAG_FILTER, GL::LINEAR);
134 gCGL->texParameteri(textureTarget, GL::TEXTURE_MIN_FILTER, GL::LINEAR);
135 gCGL->texParameteri(textureTarget, GL::TEXTURE_WRAP_S, GL::CLAMP_TO_EDGE);
136 gCGL->texParameteri(textureTarget, GL::TEXTURE_WRAP_T, GL::CLAMP_TO_EDGE);
137
138 // Tell the GraphicsContextGL to use the IOSurface as the backing store for m_opaqueTexture.
139 if (data.isShared) {
140#if !PLATFORM(IOS_FAMILY_SIMULATOR)
141 m_ioSurfaceTextureHandle = gCGL->attachIOSurfaceToSharedTexture(textureTarget, data.surface.get());
142 m_ioSurfaceTextureHandleIsShared = true;
143#else
144 ASSERT_NOT_REACHED();
145#endif
146 } else {
147 auto size = data.surface->size();
148 if (!size.width() || !size.height())
149 return;
150 m_ioSurfaceTextureHandle = gCGL->createPbufferAndAttachIOSurface(textureTarget, GraphicsContextGLOpenGL::PbufferAttachmentUsage::Write, GL::BGRA, size.width(), size.height(), GL::UNSIGNED_BYTE, data.surface->surface(), 0);
151 m_ioSurfaceTextureHandleIsShared = false;
152 }
153
154 if (!m_ioSurfaceTextureHandle) {
155 gCGL->deleteTexture(m_opaqueTexture);
156 m_opaqueTexture = 0;
157 return;
158 }
159
160 // Now set up the framebuffer to use the texture that points to the IOSurface. The depth and
161 // stencil buffers were attached by startFrame.
162 gl.framebufferTexture2D(GL::FRAMEBUFFER, GL::COLOR_ATTACHMENT0, GL::TEXTURE_2D, m_opaqueTexture, 0);
163
164 // FIXME: This is assuming multisampling is turned off and we're rendering directly into the framebuffer.
165
166 // At this point the framebuffer should be "complete".
167 ASSERT(gl.checkFramebufferStatus(GL::FRAMEBUFFER) == GL::FRAMEBUFFER_COMPLETE);
168#else
169 m_opaqueTexture = data.opaqueTexture;
170
171#if USE(OPENGL_ES)
172 auto& extensions = reinterpret_cast<ExtensionsGLOpenGLES&>(gl.getExtensions());
173 if (m_attributes.antialias && extensions.isImagination()) {
174 extensions.framebufferTexture2DMultisampleIMG(GL::FRAMEBUFFER, GL::COLOR_ATTACHMENT0, GL::TEXTURE_2D, m_opaqueTexture, 0, m_sampleCount);
175 return;
176 }
177#endif
178
179 if (!m_multisampleColorBuffer)
180 gl.framebufferTexture2D(GL::FRAMEBUFFER, GL::COLOR_ATTACHMENT0, GL::TEXTURE_2D, m_opaqueTexture, 0);
181#endif
182}

184void WebXROpaqueFramebuffer::endFrame()
185{
186 m_framebuffer->setOpaqueActive(false);
187
188 if (!m_context.graphicsContextGL())
189 return;
190
191 auto& gl = *m_context.graphicsContextGL();
192
193#if USE(IOSURFACE_FOR_XR_LAYER_DATA)
194 // FIXME: We have to call finish rather than flush because we only want to disconnect
195 // the IOSurface and signal the DeviceProxy when we know the content has been rendered.
196 // It might be possible to set this up so the completion of the rendering triggers
197 // the endFrame call.
198 gl.finish();
199
200 if (m_ioSurfaceTextureHandle) {
201 auto gCGL = static_cast<GraphicsContextGLOpenGL*>(&gl);
202 if (m_ioSurfaceTextureHandleIsShared) {
203#if !PLATFORM(IOS_FAMILY_SIMULATOR)
204 gCGL->detachIOSurfaceFromSharedTexture(m_ioSurfaceTextureHandle);
205#else
206 ASSERT_NOT_REACHED();
207#endif
208 } else
209 gCGL->destroyPbufferAndDetachIOSurface(m_ioSurfaceTextureHandle);
210 m_ioSurfaceTextureHandle = nullptr;
211 m_ioSurfaceTextureHandleIsShared = false;
212 }
213#else
214
215 if (m_multisampleColorBuffer) {
216#if !USE(ANGLE)
217 // FIXME: These may be needed when using ANGLE, but it didn't compile in the initial implementation.
218 TemporaryOpenGLSetting scopedScissor(GL::SCISSOR_TEST, 0);
219 TemporaryOpenGLSetting scopedDither(GL::DITHER, 0);
220 TemporaryOpenGLSetting scopedDepth(GL::DEPTH_TEST, 0);
221 TemporaryOpenGLSetting scopedStencil(GL::STENCIL_TEST, 0);
222#endif
223
224 GCGLint boundReadFBO { 0 };
225 GCGLint boundDrawFBO { 0 };
226 gl.getIntegerv(GL::READ_FRAMEBUFFER_BINDING, makeGCGLSpan(&boundReadFBO, 1));
227 gl.getIntegerv(GL::DRAW_FRAMEBUFFER_BINDING, makeGCGLSpan(&boundDrawFBO, 1));
228
229 auto scopedFBOs = makeScopeExit([&gl, boundReadFBO, boundDrawFBO]() {
230 gl.bindFramebuffer(GL::READ_FRAMEBUFFER, boundReadFBO);
231 gl.bindFramebuffer(GL::DRAW_FRAMEBUFFER, boundDrawFBO);
232 });
233
234 gl.bindFramebuffer(GL::DRAW_FRAMEBUFFER, m_resolvedFBO);
235 gl.framebufferTexture2D(GL::FRAMEBUFFER, GL::COLOR_ATTACHMENT0, GL::TEXTURE_2D, m_opaqueTexture, 0);
236
237 // Resolve multisample framebuffer
238 gl.bindFramebuffer(GL::READ_FRAMEBUFFER, m_framebuffer->object());
239 gl.blitFramebuffer(0, 0, m_width, m_height, 0, 0, m_width, m_height, GL::COLOR_BUFFER_BIT, GL::LINEAR);
240 }
241
242 gl.flush();
243#endif
244}

246bool WebXROpaqueFramebuffer::setupFramebuffer()
247{
248 if (!m_context.graphicsContextGL())
249 return false;
250 auto& gl = *m_context.graphicsContextGL();
251
252 // Bind current FBOs when exiting the function
253 GCGLint boundFBO { 0 };
254 GCGLint boundRenderbuffer { 0 };
255 gl.getIntegerv(GL::FRAMEBUFFER_BINDING, makeGCGLSpan(&boundFBO, 1));
256 gl.getIntegerv(GL::RENDERBUFFER_BINDING, makeGCGLSpan(&boundRenderbuffer, 1));
257 auto scopedFBO = makeScopeExit([&gl, boundFBO, boundRenderbuffer]() {
258 gl.bindFramebuffer(GL::FRAMEBUFFER, boundFBO);
259 gl.bindRenderbuffer(GL::RENDERBUFFER, boundRenderbuffer);
260 });
261
262 // Set up color, depth and stencil formats
263 bool hasDepthOrStencil = m_attributes.stencil || m_attributes.depth;
264 auto colorFormat = m_attributes.alpha ? GL::RGBA8 : GL::RGB8;
265#if USE(OPENGL_ES)
266 auto& extensions = reinterpret_cast<ExtensionsGLOpenGLES&>(gl.getExtensions());
267 bool platformSupportsPackedDepthStencil = hasDepthOrStencil && extensions.supports("GL_OES_packed_depth_stencil");
268 auto depthFormat = platformSupportsPackedDepthStencil ? GL::DEPTH24_STENCIL8 : GL::DEPTH_COMPONENT16;
269 auto stencilFormat = GL::STENCIL_INDEX8;
270#elif USE(ANGLE)
271 bool platformSupportsPackedDepthStencil = true;
272 auto depthFormat = platformSupportsPackedDepthStencil ? GL::DEPTH24_STENCIL8 : GL::DEPTH_COMPONENT;
273 auto stencilFormat = GL::STENCIL_INDEX8;
274#else
275 auto& extensions = reinterpret_cast<ExtensionsGLOpenGLCommon&>(gl.getExtensions());
276 bool platformSupportsPackedDepthStencil = hasDepthOrStencil && extensions.supports("GL_EXT_packed_depth_stencil");
277 auto depthFormat = platformSupportsPackedDepthStencil ? GL::DEPTH24_STENCIL8 : GL::DEPTH_COMPONENT;
278 auto stencilFormat = GL::STENCIL_COMPONENT;
279#endif
280
281 // Set up recommended samples for WebXR.
282 // FIXME: check if we can get recommended values from each device platform.
283 if (m_attributes.antialias) {
284 GCGLint maxSampleCount;
285#if USE(ANGLE)
286 gl.getIntegerv(GL::MAX_SAMPLES, makeGCGLSpan(&maxSampleCount, 1));
287#else
288 gl.getIntegerv(ExtensionsGL::MAX_SAMPLES, makeGCGLSpan(&maxSampleCount, 1));
289#endif
290 // Cap the maxiumum multisample count at 4. Any more than this is likely overkill and will impact performance.
291 m_sampleCount = std::min(4, maxSampleCount);
292 }
293
294#if USE(OPENGL_ES)
295 // Use multisampled_render_to_texture extension if available.
296 if (m_attributes.antialias && extensions.isImagination()) {
297 // framebufferTexture2DMultisampleIMG is set up in startFrame call.
298 if (!hasDepthOrStencil)
299 return true;
300
301 gl.bindFramebuffer(GL::FRAMEBUFFER, m_framebuffer->object());
302 m_depthStencilBuffer = gl.createRenderbuffer();
303 if (platformSupportsPackedDepthStencil) {
304 gl.bindRenderbuffer(GL::RENDERBUFFER, m_depthStencilBuffer);
305 extensions.renderbufferStorageMultisampleANGLE(GL::RENDERBUFFER, m_sampleCount, depthFormat, m_width, m_height);
306 if (m_attributes.stencil)
307 gl.framebufferRenderbuffer(GL::FRAMEBUFFER, GL::STENCIL_ATTACHMENT, GL::RENDERBUFFER, m_depthStencilBuffer);
308 if (m_attributes.depth)
309 gl.framebufferRenderbuffer(GL::FRAMEBUFFER, GL::DEPTH_ATTACHMENT, GL::RENDERBUFFER, m_depthStencilBuffer);
310 } else {
311 if (m_attributes.stencil) {
312 m_stencilBuffer = gl.createRenderbuffer();
313 gl.bindRenderbuffer(GL::RENDERBUFFER, m_stencilBuffer);
314 extensions.renderbufferStorageMultisampleANGLE(GL::RENDERBUFFER, m_sampleCount, stencilFormat, m_width, m_height);
315 gl.framebufferRenderbuffer(GL::FRAMEBUFFER, GL::STENCIL_ATTACHMENT, GL::RENDERBUFFER, m_stencilBuffer);
316 }
317 if (m_attributes.depth) {
318 gl.bindRenderbuffer(GL::RENDERBUFFER, m_depthStencilBuffer);
319 extensions.renderbufferStorageMultisampleANGLE(GL::RENDERBUFFER, m_sampleCount, depthFormat, m_width, m_height);
320 gl.framebufferRenderbuffer(GL::FRAMEBUFFER, GL::DEPTH_ATTACHMENT, GL::RENDERBUFFER, m_depthStencilBuffer);
321 }
322 }
323 return true;
324 }
325#endif // USE(OPENGL_ES)
326
327 if (m_attributes.antialias && m_context.isWebGL2()) {
328 m_resolvedFBO = gl.createFramebuffer();
329 m_multisampleColorBuffer = gl.createRenderbuffer();
330 gl.bindFramebuffer(GL::FRAMEBUFFER, m_framebuffer->object());
331 gl.bindRenderbuffer(GL::RENDERBUFFER, m_multisampleColorBuffer);
332 gl.renderbufferStorageMultisample(GL::RENDERBUFFER, m_sampleCount, colorFormat, m_width, m_height);
333 gl.framebufferRenderbuffer(GL::FRAMEBUFFER, GL::COLOR_ATTACHMENT0, GL::RENDERBUFFER, m_multisampleColorBuffer);
334 if (hasDepthOrStencil) {
335 m_depthStencilBuffer = gl.createRenderbuffer();
336 if (platformSupportsPackedDepthStencil) {
337 gl.bindRenderbuffer(GL::RENDERBUFFER, m_depthStencilBuffer);
338 gl.renderbufferStorageMultisample(GL::RENDERBUFFER, m_sampleCount, depthFormat, m_width, m_height);
339 if (m_attributes.stencil)
340 gl.framebufferRenderbuffer(GL::FRAMEBUFFER, GL::STENCIL_ATTACHMENT, GL::RENDERBUFFER, m_depthStencilBuffer);
341 if (m_attributes.depth)
342 gl.framebufferRenderbuffer(GL::FRAMEBUFFER, GL::DEPTH_ATTACHMENT, GL::RENDERBUFFER, m_depthStencilBuffer);
343 } else {
344 if (m_attributes.stencil) {
345 m_stencilBuffer = gl.createRenderbuffer();
346 gl.bindRenderbuffer(GL::RENDERBUFFER, m_stencilBuffer);
347 gl.renderbufferStorageMultisample(GL::RENDERBUFFER, m_sampleCount, stencilFormat, m_width, m_height);
348 gl.framebufferRenderbuffer(GL::FRAMEBUFFER, GL::STENCIL_ATTACHMENT, GL::RENDERBUFFER, m_stencilBuffer);
349 }
350 if (m_attributes.depth) {
351 gl.bindRenderbuffer(GL::RENDERBUFFER, m_depthStencilBuffer);
352 gl.renderbufferStorageMultisample(GL::RENDERBUFFER, m_sampleCount, depthFormat, m_width, m_height);
353 gl.framebufferRenderbuffer(GL::FRAMEBUFFER, GL::DEPTH_ATTACHMENT, GL::RENDERBUFFER, m_depthStencilBuffer);
354 }
355 }
356 }
357 return gl.checkFramebufferStatus(GL::FRAMEBUFFER) == GL::FRAMEBUFFER_COMPLETE;
358 }
359
360 if (hasDepthOrStencil) {
361 gl.bindFramebuffer(GL::FRAMEBUFFER, m_framebuffer->object());
362 m_depthStencilBuffer = gl.createRenderbuffer();
363 if (platformSupportsPackedDepthStencil) {
364 gl.bindRenderbuffer(GL::RENDERBUFFER, m_depthStencilBuffer);
365 gl.renderbufferStorage(GL::RENDERBUFFER, depthFormat, m_width, m_height);
366 if (m_context.isWebGL2()) {
367 if (m_attributes.stencil)
368 gl.framebufferRenderbuffer(GL::FRAMEBUFFER, GL::STENCIL_ATTACHMENT, GL::RENDERBUFFER, m_depthStencilBuffer);
369 if (m_attributes.depth)
370 gl.framebufferRenderbuffer(GL::FRAMEBUFFER, GL::DEPTH_ATTACHMENT, GL::RENDERBUFFER, m_depthStencilBuffer);
371 } else {
372 if (m_attributes.stencil || m_attributes.depth)
373 gl.framebufferRenderbuffer(GL::FRAMEBUFFER, GL::DEPTH_STENCIL_ATTACHMENT, GL::RENDERBUFFER, m_depthStencilBuffer);
374 }
375 } else {
376 if (m_attributes.stencil) {
377 m_stencilBuffer = gl.createRenderbuffer();
378 gl.bindRenderbuffer(GL::RENDERBUFFER, m_stencilBuffer);
379 gl.renderbufferStorage(GL::RENDERBUFFER, stencilFormat, m_width, m_height);
380 gl.framebufferRenderbuffer(GL::FRAMEBUFFER, GL::STENCIL_ATTACHMENT, GL::RENDERBUFFER, m_stencilBuffer);
381 }
382 if (m_attributes.depth) {
383 gl.bindRenderbuffer(GL::RENDERBUFFER, m_depthStencilBuffer);
384 gl.renderbufferStorage(GL::RENDERBUFFER, depthFormat, m_width, m_height);
385 gl.framebufferRenderbuffer(GL::FRAMEBUFFER, GL::DEPTH_ATTACHMENT, GL::RENDERBUFFER, m_depthStencilBuffer);
386 }
387 }
388 }
389
390 return true;
391}
