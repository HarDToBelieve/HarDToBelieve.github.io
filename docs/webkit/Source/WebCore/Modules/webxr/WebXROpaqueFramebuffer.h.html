28
29#if ENABLE(WEBXR)
30
31#include "GraphicsTypesGL.h"
32#include "PlatformXR.h"
33#include "WebXRLayer.h"
34#include <wtf/Ref.h>
35
36namespace WebCore {
37
38class IntSize;
39class WebGLFramebuffer;
40class WebGLRenderingContextBase;
41struct XRWebGLLayerInit;
42
43class WebXROpaqueFramebuffer {
44public:
45 struct Attributes {
46 bool alpha { true };
47 bool antialias { true };
48 bool depth { true };
49 bool stencil { false };
50 };
51
52 static std::unique_ptr<WebXROpaqueFramebuffer> create(PlatformXR::LayerHandle, WebGLRenderingContextBase&, Attributes&&, uint32_t width, uint32_t height);
53 ~WebXROpaqueFramebuffer();
54
55 PlatformXR::LayerHandle handle() const { return m_handle; }
56 const WebGLFramebuffer& framebuffer() const { return m_framebuffer.get(); }
57 uint32_t width() const { return m_width; }
58 uint32_t height() const { return m_height; }
59
60 void startFrame(const PlatformXR::Device::FrameData::LayerData&);
61 void endFrame();
62
63private:
64 WebXROpaqueFramebuffer(PlatformXR::LayerHandle, Ref<WebGLFramebuffer>&&, WebGLRenderingContextBase&, Attributes&&, uint32_t width, uint32_t height);
65
66 bool setupFramebuffer();
67
68 PlatformXR::LayerHandle m_handle;
69 Ref<WebGLFramebuffer> m_framebuffer;
70 WebGLRenderingContextBase& m_context;
71 Attributes m_attributes;
72 uint32_t m_width { 0 };
73 uint32_t m_height { 0 };
74 PlatformGLObject m_depthStencilBuffer { 0 };
75 PlatformGLObject m_stencilBuffer { 0 };
76 PlatformGLObject m_multisampleColorBuffer { 0 };
77 PlatformGLObject m_resolvedFBO { 0 };
78 GCGLint m_sampleCount { 0 };
79 PlatformGLObject m_opaqueTexture { 0 };
80#if USE(IOSURFACE_FOR_XR_LAYER_DATA)
81 void* m_ioSurfaceTextureHandle { nullptr };
82 bool m_ioSurfaceTextureHandleIsShared { false };
83#endif
84};
85
86} // namespace WebCore

88#endif // ENABLE(WEBXR)
