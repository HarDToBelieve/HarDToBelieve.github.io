26#pragma once
27
28#if ENABLE(WEBXR)
29
30#include "HTMLCanvasElement.h"
31#include "WebXRWebGLLayer.h"
32#include "XRSessionMode.h"
33
34namespace WebCore {
35
36struct XRRenderStateInit;
37
38class WebXRRenderState : public RefCounted<WebXRRenderState> {
39 WTF_MAKE_ISO_ALLOCATED(WebXRRenderState);
40public:
41 static Ref<WebXRRenderState> create(XRSessionMode);
42 ~WebXRRenderState();
43
44 Ref<WebXRRenderState> clone() const;
45
46 double depthNear() const { return m_depth.near; }
47 void setDepthNear(double near) { m_depth.near = near; }
48
49 double depthFar() const { return m_depth.far; }
50 void setDepthFar(double far) { m_depth.far = far; };
51
52 std::optional<double> inlineVerticalFieldOfView() const { return m_inlineVerticalFieldOfView; }
53 void setInlineVerticalFieldOfView(double fieldOfView) { m_inlineVerticalFieldOfView = fieldOfView; }
54
55 RefPtr<WebXRWebGLLayer> baseLayer() const { return m_baseLayer; }
56 void setBaseLayer(WebXRWebGLLayer* baseLayer) { m_baseLayer = baseLayer; }
57
58 HTMLCanvasElement* outputCanvas() const { return m_outputCanvas.get(); }
59 void setOutputCanvas(HTMLCanvasElement* canvas) { m_outputCanvas = makeWeakPtr(canvas); }
60
61 bool isCompositionEnabled() const { return m_compositionEnabled; }
62 void setCompositionEnabled(bool compositionEnabled) { m_compositionEnabled = compositionEnabled; }
63
64private:
65 explicit WebXRRenderState(std::optional<double> fieldOfView);
66 explicit WebXRRenderState(const WebXRRenderState&);
67
68 // https://immersive-web.github.io/webxr/#initialize-the-render-state
69 struct {
70 double near { 0.1 }; // in meters
71 double far { 1000 }; // in meters
72 } m_depth;
73 std::optional<double> m_inlineVerticalFieldOfView; // in radians
74 RefPtr<WebXRWebGLLayer> m_baseLayer;
75 WeakPtr<HTMLCanvasElement> m_outputCanvas;
76 bool m_compositionEnabled { true };
77};
78
79} // namespace WebCore
80
81#endif // ENABLE(WEBXR)
