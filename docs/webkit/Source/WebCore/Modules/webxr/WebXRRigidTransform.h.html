40struct DOMPointInit;

43class WebXRRigidTransform : public RefCounted<WebXRRigidTransform>, public CanMakeWeakPtr<WebXRRigidTransform> {
44 WTF_MAKE_ISO_ALLOCATED_EXPORT(WebXRRigidTransform, WEBCORE_EXPORT);
45public:
46 static Ref<WebXRRigidTransform> create();
47 static Ref<WebXRRigidTransform> create(const TransformationMatrix&);
48 WEBCORE_EXPORT static ExceptionOr<Ref<WebXRRigidTransform>> create(const DOMPointInit&, const DOMPointInit&);
49 WEBCORE_EXPORT ~WebXRRigidTransform();
50
51 const DOMPointReadOnly& position() const;
52 const DOMPointReadOnly& orientation() const;
53 const Float32Array& matrix();
54 const WebXRRigidTransform& inverse();
55 const TransformationMatrix& rawTransform() const;
56
57 JSValueInWrappedObject& cachedMatrix() { return m_cachedMatrix; }
58
59private:
60 WebXRRigidTransform(const DOMPointInit&, const DOMPointInit&);
61 WebXRRigidTransform(const TransformationMatrix&);
62
63 Ref<DOMPointReadOnly> m_position;
64 Ref<DOMPointReadOnly> m_orientation;
65 TransformationMatrix m_rawTransform;
66 RefPtr<Float32Array> m_matrix;
67 RefPtr<WebXRRigidTransform> m_inverse;
68 WeakPtr<WebXRRigidTransform> m_parentInverse;
69 JSValueInWrappedObject m_cachedMatrix;
70};
