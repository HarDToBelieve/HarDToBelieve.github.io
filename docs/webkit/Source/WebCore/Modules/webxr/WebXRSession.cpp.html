53WebXRSession::WebXRSession(Document& document, WebXRSystem& system, XRSessionMode mode, PlatformXR::Device& device, FeatureList&& requestedFeatures)
54 : ActiveDOMObject(&document)
55 , m_inputSources(WebXRInputSourceArray::create(*this))
56 , m_xrSystem(system)
57 , m_mode(mode)
58 , m_device(makeWeakPtr(device))
59 , m_requestedFeatures(WTFMove(requestedFeatures))
60 , m_activeRenderState(WebXRRenderState::create(mode))
61 , m_viewerReferenceSpace(makeUnique<WebXRViewerSpace>(document, *this))
62 , m_timeOrigin(MonotonicTime::now())
63 , m_views(device.views(mode))
64{
65 m_device->setTrackingAndRenderingClient(makeWeakPtr(*this));
66 m_device->initializeTrackingAndRendering(mode);
67
68 // https://immersive-web.github.io/webxr/#ref-for-dom-xrreferencespacetype-viewer%E2%91%A2
69 // Every session MUST support viewer XRReferenceSpaces.
70 m_device->initializeReferenceSpace(XRReferenceSpaceType::Viewer);
71
72 suspendIfNeeded();
73}

111// https://immersive-web.github.io/webxr/#dom-xrsession-updaterenderstate
112ExceptionOr<void> WebXRSession::updateRenderState(const XRRenderStateInit& newState)
113{
114 // 1. Let session be this.
115 // 2. If session's ended value is true, throw an InvalidStateError and abort these steps.
116 if (m_ended)
117 return Exception { InvalidStateError };
118
119 // 3. If newState's baseLayer was created with an XRSession other than session,
120 // throw an InvalidStateError and abort these steps.
121 if (newState.baseLayer && &newState.baseLayer->session() != this)
122 return Exception { InvalidStateError };
123
124 // 4. If newState's inlineVerticalFieldOfView is set and session is an immersive session,
125 // throw an InvalidStateError and abort these steps.
126 if (newState.inlineVerticalFieldOfView && isImmersive(m_mode))
127 return Exception { InvalidStateError };
128
129 // 5. If none of newState's depthNear, depthFar, inlineVerticalFieldOfView, baseLayer,
130 // layers are set, abort these steps.
131 if (!newState.depthNear && !newState.depthFar && !newState.inlineVerticalFieldOfView && !newState.baseLayer && !newState.layers)
132 return { };
133
134 // 6. Run update the pending layers state with session and newState.
135 // https://immersive-web.github.io/webxr/#update-the-pending-layers-state
136 if (newState.layers)
137 return Exception { NotSupportedError };
138
139 // 7. Let activeState be session's active render state.
140 // 8. If session's pending render state is null, set it to a copy of activeState.
141 if (!m_pendingRenderState)
142 m_pendingRenderState = m_activeRenderState->clone();
143
144 // 9. If newState's depthNear value is set, set session's pending render state's depthNear to newState's depthNear.
145 if (newState.depthNear)
146 m_pendingRenderState->setDepthNear(newState.depthNear.value());
147
148 // 10. If newState's depthFar value is set, set session's pending render state's depthFar to newState's depthFar.
149 if (newState.depthFar)
150 m_pendingRenderState->setDepthFar(newState.depthFar.value());
151
152 // 11. If newState's inlineVerticalFieldOfView is set, set session's pending render state's inlineVerticalFieldOfView
153 // to newState's inlineVerticalFieldOfView.
154 if (newState.inlineVerticalFieldOfView)
155 m_pendingRenderState->setInlineVerticalFieldOfView(newState.inlineVerticalFieldOfView.value());
156
157 // 12. If newState's baseLayer is set, set session's pending render state's baseLayer to newState's baseLayer.
158 if (newState.baseLayer)
159 m_pendingRenderState->setBaseLayer(newState.baseLayer.get());
160
161 return { };
162}

164// https://immersive-web.github.io/webxr/#reference-space-is-supported
165bool WebXRSession::referenceSpaceIsSupported(XRReferenceSpaceType type) const
166{
167 // 1. If type is not contained in session’s XR device's list of enabled features for mode return false.
168 if (!m_requestedFeatures.contains(type))
169 return false;
170
171 // 2. If type is viewer, return true.
172 if (type == XRReferenceSpaceType::Viewer)
173 return true;
174
175 bool isImmersiveSession = isImmersive(m_mode);
176 if (type == XRReferenceSpaceType::Local || type == XRReferenceSpaceType::LocalFloor) {
177 // 3. If type is local or local-floor, and session is an immersive session, return true.
178 if (isImmersiveSession)
179 return true;
180
181 // 4. If type is local or local-floor, and the XR device supports reporting orientation data, return true.
182 if (m_device->supportsOrientationTracking())
183 return true;
184 }
185
186 // 5. If type is bounded-floor and session is an immersive session, return the result of whether bounded
187 // reference spaces are supported by the XR device.
188 // https://immersive-web.github.io/webxr/#bounded-reference-spaces-are-supported
189 // TODO: add API to PlatformXR::Device
190 if (type == XRReferenceSpaceType::BoundedFloor && isImmersiveSession)
191 return true;
192
193 // 6. If type is unbounded, session is an immersive session, and the XR device supports stable tracking
194 // near the user over an unlimited distance, return true.
195 // TODO: add API to PlatformXR::Device to check stable tracking over unlimited distance.
196 if (type == XRReferenceSpaceType::Unbounded && isImmersiveSession)
197 return true;
198
199 // 7. Return false.
200 return false;
201}

203// https://immersive-web.github.io/webxr/#dom-xrsession-requestreferencespace
204void WebXRSession::requestReferenceSpace(XRReferenceSpaceType type, RequestReferenceSpacePromise&& promise)
205{
206 if (!scriptExecutionContext()) {
207 promise.reject(Exception { InvalidStateError });
208 return;
209 }
210
211 // 1. Let promise be a new Promise.
212 // 2. Run the following steps in parallel:
213 scriptExecutionContext()->postTask([this, weakThis = makeWeakPtr(*this), promise = WTFMove(promise), type](auto&) mutable {
214 if (!weakThis)
215 return;
216 // 2.1. If the result of running reference space is supported for type and session is false, queue a task to reject promise
217 // with a NotSupportedError and abort these steps.
218 if (!referenceSpaceIsSupported(type)) {
219 queueTaskKeepingObjectAlive(*this, TaskSource::WebXR, [promise = WTFMove(promise)]() mutable {
220 promise.reject(Exception { NotSupportedError });
221 });
222 return;
223 }
224 // 2.2. Set up any platform resources required to track reference spaces of type type.
225 m_device->initializeReferenceSpace(type);
226
227 // 2.3. Queue a task to run the following steps:
228 queueTaskKeepingObjectAlive(*this, TaskSource::WebXR, [this, type, promise = WTFMove(promise)]() mutable {
229 if (!scriptExecutionContext()) {
230 promise.reject(Exception { InvalidStateError });
231 return;
232 }
233 auto& document = downcast<Document>(*scriptExecutionContext());
234 // 2.4. Create a reference space, referenceSpace, with type and session.
235 // https://immersive-web.github.io/webxr/#create-a-reference-space
236 RefPtr<WebXRReferenceSpace> referenceSpace;
237 if (type == XRReferenceSpaceType::BoundedFloor)
238 referenceSpace = WebXRBoundedReferenceSpace::create(document, makeRef(*this), type);
239 else
240 referenceSpace = WebXRReferenceSpace::create(document, makeRef(*this), type);
241
242 // 2.5. Resolve promise with referenceSpace.
243 promise.resolve(referenceSpace.releaseNonNull());
244 });
245 });
246}

248// https://immersive-web.github.io/webxr/#dom-xrsession-requestanimationframe
249unsigned WebXRSession::requestAnimationFrame(Ref<XRFrameRequestCallback>&& callback)
250{
251 // Ignore any new frame requests once the session is ended.
252 if (m_ended)
253 return 0;
254
255 // 1. Let session be the target XRSession object.
256 // 2. Increment session's animation frame callback identifier by one.
257 unsigned newId = m_nextCallbackId++;
258
259 // 3. Append callback to session's list of animation frame callbacks, associated with session's
260 // animation frame callback identifier's current value.
261 callback->setCallbackId(newId);
262 m_callbacks.append(WTFMove(callback));
263
264 // Script can add multiple requestAnimationFrame callbacks but we should only request a device frame once.
265 // When requestAnimationFrame is called during processing RAF callbacks the next requestFrame is scheduled
266 // at the end of WebXRSession::onFrame() to prevent requesting a new frame before the current one has ended.
267 if (m_callbacks.size() == 1)
268 requestFrame();
269
270 // 4. Return session's animation frame callback identifier's current value.
271 return newId;
272}

274// https://immersive-web.github.io/webxr/#dom-xrsession-cancelanimationframe
275void WebXRSession::cancelAnimationFrame(unsigned callbackId)
276{
277 // 1. Let session be the target XRSession object.
278 // 2. Find the entry in session's list of animation frame callbacks or session's list of
279 // currently running animation frame callbacks that is associated with the value handle.
280 // 3. If there is such an entry, set its cancelled boolean to true and remove it from
281 // session's list of animation frame callbacks.
282 size_t position = m_callbacks.findMatching([callbackId] (auto& item) {
283 return item->callbackId() == callbackId;
284 });
285
286 if (position != notFound) {
287 m_callbacks[position]->setFiredOrCancelled();
288 return;
289 }
290}

323// https://immersive-web.github.io/webxr/#shut-down-the-session
324void WebXRSession::shutdown(InitiatedBySystem initiatedBySystem)
325{
326 auto protectedThis = makeRef(*this);
327
328 if (m_ended) {
329 // This method was called earlier with initiatedBySystem=No when the
330 // session termination was requested manually via XRSession.end(). When
331 // the system has completed the shutdown, this method is now called again
332 // with initiatedBySystem=Yes to do the final cleanup.
333 if (initiatedBySystem == InitiatedBySystem::Yes)
334 didCompleteShutdown();
335 return;
336 }
337
338 // 1. Let session be the target XRSession object.
339 // 2. Set session's ended value to true.
340 m_ended = true;
341
342 // 3. If the active immersive session is equal to session, set the active immersive session to null.
343 // 4. Remove session from the list of inline sessions.
344 m_xrSystem.sessionEnded(*this);
345
346 m_inputSources->clear();
347
348 if (initiatedBySystem == InitiatedBySystem::Yes) {
349 // If we get here, the session termination was triggered by the system rather than
350 // via XRSession.end(). Since the system has completed the session shutdown, we can
351 // immediately do the final cleanup.
352 didCompleteShutdown();
353 return;
354 }
355
356 // TODO: complete the implementation
357 // 5. Reject any outstanding promises returned by session with an InvalidStateError, except for any promises returned by end().
358 // 6. If no other features of the user agent are actively using them, perform the necessary platform-specific steps to shut down the device's tracking and rendering capabilities. This MUST include:
359 // 6.1. Releasing exclusive access to the XR device if session is an immersive session.
360 // 6.2. Deallocating any graphics resources acquired by session for presentation to the XR device.
361 // 6.3. Putting the XR device in a state such that a different source may be able to initiate a session with the same device if session is an immersive session.
362 if (m_device)
363 m_device->shutDownTrackingAndRendering();
364
365 // If device will not report shutdown completion via the TrackingAndRenderingClient,
366 // complete the shutdown cleanup here.
367 if (!m_device || !m_device->supportsSessionShutdownNotification())
368 didCompleteShutdown();
369}
370
371void WebXRSession::didCompleteShutdown()
372{
373 if (m_device)
374 m_device->setTrackingAndRenderingClient(nullptr);
375
376 // Resolve end promise from XRSession::end()
377 if (m_endPromise) {
378 m_endPromise->resolve();
379 m_endPromise = std::nullopt;
380 }
381
382 // From https://immersive-web.github.io/webxr/#shut-down-the-session
383 // 7. Queue a task that fires an XRSessionEvent named end on session.
384 auto event = XRSessionEvent::create(eventNames().endEvent, { makeRefPtr(*this) });
385 queueTaskToDispatchEvent(*this, TaskSource::WebXR, WTFMove(event));
386}

388// https://immersive-web.github.io/webxr/#dom-xrsession-end
389ExceptionOr<void> WebXRSession::end(EndPromise&& promise)
390{
391 // The shutdown() call below might remove the sole reference to session
392 // that could exist (the XRSystem owns the sessions) so let's protect this.
393 auto protectedThis = makeRef(*this);
394
395 if (m_ended)
396 return Exception { InvalidStateError, "Cannot end a session more than once"_s };
397
398 ASSERT(!m_endPromise);
399 m_endPromise = WTFMove(promise);
400
401 // 1. Let promise be a new Promise.
402 // 2. Shut down the target XRSession object.
403 shutdown(InitiatedBySystem::No);
404
405 // 3. Queue a task to perform the following steps:
406 // 3.1 Wait until any platform-specific steps related to shutting down the session have completed.
407 // 3.2 Resolve promise.
408 // 4. Return promise.
409 return { };
410}

421void WebXRSession::sessionDidInitializeInputSources(Vector<PlatformXR::Device::FrameData::InputSource>&& inputSources)
422{
423 // https://immersive-web.github.io/webxr/#dom-xrsystem-requestsession
424 // 5.4.11 Queue a task to perform the following steps: NOTE: These steps ensure that initial inputsourceschange
425 // events occur after the initial session is resolved.
426 queueTaskKeepingObjectAlive(*this, TaskSource::WebXR, [this, inputSources = WTFMove(inputSources)]() mutable {
427 // 1. Set session's promise resolved flag to true.
428 m_inputInitialized = true;
429 // 2. Let sources be any existing input sources attached to session.
430 // 3. If sources is non-empty, perform the following steps:
431 if (!inputSources.isEmpty()) {
432 auto timestamp = (MonotonicTime::now() - m_timeOrigin).milliseconds();
433 // 3.1. Set session's list of active XR input sources to sources.
434 // 3.2. Fire an XRInputSourcesChangeEvent named inputsourceschange on session with added set to sources.
435 // Note: 3.1 and 3.2 steps are handled inside the update() call.
436 m_inputSources->update(timestamp, inputSources);
437 }
438 });
439}

448void WebXRSession::applyPendingRenderState()
449{
450 // https: //immersive-web.github.io/webxr/#apply-the-pending-render-state
451 // 1. Let activeState be session’s active render state.
452 // 2. Let newState be session’s pending render state.
453 // 3. Set session’s pending render state to null.
454 auto newState = m_pendingRenderState;
455 ASSERT(newState);
456
457 // 4. Let oldBaseLayer be activeState’s baseLayer.
458 // 5. Let oldLayers be activeState’s layers.
459 // FIXME: those are only needed for step 6.2.
460
461 // 6.1 Set activeState to newState.
462 m_activeRenderState = newState;
463
464 // 6.2 If oldBaseLayer is not equal to activeState’s baseLayer, oldLayers is not equal to activeState’s layers, or the dimensions of any of the layers have changed, update the viewports for session.
465 // FIXME: implement this.
466
467 // 6.3 If activeState’s inlineVerticalFieldOfView is less than session’s minimum inline field of view set activeState’s inlineVerticalFieldOfView to session’s minimum inline field of view.
468 if (m_activeRenderState->inlineVerticalFieldOfView() < m_minimumInlineFOV)
469 m_activeRenderState->setInlineVerticalFieldOfView(m_minimumInlineFOV);
470
471 // 6.4 If activeState’s inlineVerticalFieldOfView is greater than session’s maximum inline field of view set activeState’s inlineVerticalFieldOfView to session’s maximum inline field of view.
472 if (m_activeRenderState->inlineVerticalFieldOfView() > m_maximumInlineFOV)
473 m_activeRenderState->setInlineVerticalFieldOfView(m_maximumInlineFOV);
474
475 // 6.5 If activeState’s depthNear is less than session’s minimum near clip plane set activeState’s depthNear to session’s minimum near clip plane.
476 if (m_activeRenderState->depthNear() < m_minimumNearClipPlane)
477 m_activeRenderState->setDepthNear(m_minimumNearClipPlane);
478
479 // 6.6 If activeState’s depthFar is greater than session’s maximum far clip plane set activeState’s depthFar to session’s maximum far clip plane.
480 if (m_activeRenderState->depthFar() > m_maximumFarClipPlane)
481 m_activeRenderState->setDepthFar(m_maximumFarClipPlane);
482
483 // 6.7 Let baseLayer be activeState’s baseLayer.
484 auto baseLayer = m_activeRenderState->baseLayer();
485
486 // 6.8 Set activeState’s composition enabled and output canvas as follows:
487 if (m_mode == XRSessionMode::Inline && is<WebXRWebGLLayer>(baseLayer) && !baseLayer->isCompositionEnabled()) {
488 m_activeRenderState->setCompositionEnabled(false);
489 m_activeRenderState->setOutputCanvas(baseLayer->canvas());
490 } else {
491 m_activeRenderState->setCompositionEnabled(true);
492 m_activeRenderState->setOutputCanvas(nullptr);
493 }
494}

512
513void WebXRSession::onFrame(PlatformXR::Device::FrameData&& frameData)
514{
515 ASSERT(isMainThread());
516
517 if (m_ended)
518 return;
519
520 // Queue a task to perform the following steps.
521 queueTaskKeepingObjectAlive(*this, TaskSource::WebXR, [this, frameData = WTFMove(frameData)]() mutable {
522 if (m_ended)
523 return;
524
525 m_frameData = WTFMove(frameData);
526 // 1.Let now be the current high resolution time.
527 auto now = (MonotonicTime::now() - m_timeOrigin).milliseconds();
528
529 auto frame = WebXRFrame::create(*this, WebXRFrame::IsAnimationFrame::Yes);
530 // 2.Let frame be session’s animation frame.
531 // 3.Set frame’s time to frameTime.
532 frame->setTime(static_cast<DOMHighResTimeStamp>(m_frameData.predictedDisplayTime));
533
534 // 4. For each view in list of views, set view’s viewport modifiable flag to true.
535 // 5. If the active flag of any view in the list of views has changed since the last XR animation frame, update the viewports.
536 // FIXME: implement.
537
538 // FIXME: I moved step 7 before 6 because of https://github.com/immersive-web/webxr/issues/1164
539 // 7.If session’s pending render state is not null, apply the pending render state.
540 if (m_pendingRenderState)
541 applyPendingRenderState();
542
543 // 6. If the frame should be rendered for session:
544 if (frameShouldBeRendered() && m_frameData.shouldRender) {
545 // Prepare all layers for render
546 if (m_mode == XRSessionMode::ImmersiveVr && m_activeRenderState->baseLayer())
547 m_activeRenderState->baseLayer()->startFrame(m_frameData);
548
549 // 6.1.Set session’s list of currently running animation frame callbacks to be session’s list of animation frame callbacks.
550 // 6.2.Set session’s list of animation frame callbacks to the empty list.
551 auto callbacks = m_callbacks;
552
553 // 6.3.Set frame’s active boolean to true.
554 frame->setActive(true);
555
556 // 6.4.Apply frame updates for frame.
557 if (m_inputInitialized)
558 m_inputSources->update(now, m_frameData.inputSources);
559
560 // 6.5.For each entry in session’s list of currently running animation frame callbacks, in order:
561 for (auto& callback : callbacks) {
562 // 6.6.If the entry’s cancelled boolean is true, continue to the next entry.
563 if (callback->isFiredOrCancelled())
564 continue;
565 callback->setFiredOrCancelled();
566 // 6.7.Invoke the Web IDL callback function for entry, passing now and frame as the arguments
567 callback->handleEvent(now, frame.get());
568
569 // 6.8.If an exception is thrown, report the exception.
570 }
571 // 6.9.Set session’s list of currently running animation frame callbacks to the empty list.
572 m_callbacks.removeAllMatching([](auto& callback) {
573 return callback->isFiredOrCancelled();
574 });
575
576 // 6.10.Set frame’s active boolean to false.
577 // If the session is ended, m_animationFrame->setActive false is set in shutdown().
578 frame->setActive(false);
579
580
581 // Submit current frame layers to the device.
582 Vector<PlatformXR::Device::Layer> frameLayers;
583 if (m_mode == XRSessionMode::ImmersiveVr && m_activeRenderState->baseLayer())
584 frameLayers.append(m_activeRenderState->baseLayer()->endFrame());
585
586 m_device->submitFrame(WTFMove(frameLayers));
587 }
588
589 if (!m_callbacks.isEmpty())
590 requestFrame();
591
592 });
593}

595// https://immersive-web.github.io/webxr/#poses-may-be-reported
596bool WebXRSession::posesCanBeReported(const Document& document) const
597{
598 // 1. If session’s relevant global object is not the current global object, return false.
599 auto* sessionDocument = downcast<Document>(scriptExecutionContext());
600 if (!sessionDocument || sessionDocument->domWindow() != document.domWindow())
601 return false;
602
603 // 2. If session's visibilityState in not "visible", return false.
604 if (m_visibilityState != XRVisibilityState::Visible)
605 return false;
606
607 // 5. Determine if the pose data can be returned as follows:
608 // The procedure in the specs tries to ensure that we apply measures to
609 // prevent fingerprintint in pose data and return false in case we don't.
610 // We're going to apply them so let's just return true.
