55class WebXRSession final : public RefCounted<WebXRSession>, public EventTargetWithInlineData, public ActiveDOMObject, public PlatformXR::TrackingAndRenderingClient {
56 WTF_MAKE_ISO_ALLOCATED(WebXRSession);
57public:
58 using RequestReferenceSpacePromise = DOMPromiseDeferred<IDLInterface<WebXRReferenceSpace>>;
59 using EndPromise = DOMPromiseDeferred<void>;
60 using FeatureList = PlatformXR::Device::FeatureList;
61
62 static Ref<WebXRSession> create(Document&, WebXRSystem&, XRSessionMode, PlatformXR::Device&, FeatureList&&);
63 virtual ~WebXRSession();
64
65 using RefCounted<WebXRSession>::ref;
66 using RefCounted<WebXRSession>::deref;
67
68 using TrackingAndRenderingClient::weakPtrFactory;
69 using WeakValueType = TrackingAndRenderingClient::WeakValueType;
70
71 XREnvironmentBlendMode environmentBlendMode() const;
72 XRInteractionMode interactionMode() const;
73 XRVisibilityState visibilityState() const;
74 const WebXRRenderState& renderState() const;
75 const WebXRInputSourceArray& inputSources() const;
76 PlatformXR::Device* device() const { return m_device.get(); }
77
78 ExceptionOr<void> updateRenderState(const XRRenderStateInit&);
79 void requestReferenceSpace(XRReferenceSpaceType, RequestReferenceSpacePromise&&);
80
81 unsigned requestAnimationFrame(Ref<XRFrameRequestCallback>&&);
82 void cancelAnimationFrame(unsigned callbackId);
83
84 IntSize nativeWebGLFramebufferResolution() const;
85 IntSize recommendedWebGLFramebufferResolution() const;
86 bool supportsViewportScaling() const;
87 bool isPositionEmulated() const;
88
89 // EventTarget.
90 ScriptExecutionContext* scriptExecutionContext() const final { return ActiveDOMObject::scriptExecutionContext(); }
91
92 ExceptionOr<void> end(EndPromise&&);
93 bool ended() const { return m_ended; }
94
95 XRSessionMode mode() const { return m_mode; }
96
97 const Vector<PlatformXR::Device::ViewData>& views() const { return m_views; }
98 const PlatformXR::Device::FrameData& frameData() const { return m_frameData; }
99 const WebXRViewerSpace& viewerReferenceSpace() const { return *m_viewerReferenceSpace; }
100 bool posesCanBeReported(const Document&) const;
101
102private:
103 WebXRSession(Document&, WebXRSystem&, XRSessionMode, PlatformXR::Device&, FeatureList&&);
104
105 // EventTarget
106 EventTargetInterface eventTargetInterface() const override { return WebXRSessionEventTargetInterfaceType; }
107 void refEventTarget() override { ref(); }
108 void derefEventTarget() override { deref(); }
109
110 // ActiveDOMObject
111 const char* activeDOMObjectName() const override;
112 void stop() override;
113
114 // PlatformXR::TrackingAndRenderingClient
115 void sessionDidInitializeInputSources(Vector<PlatformXR::Device::FrameData::InputSource>&&) final;
116 void sessionDidEnd() final;
117
118 enum class InitiatedBySystem : bool { No, Yes };
119 void shutdown(InitiatedBySystem);
120 void didCompleteShutdown();
121
122 bool referenceSpaceIsSupported(XRReferenceSpaceType) const;
123
124 bool frameShouldBeRendered() const;
125 void requestFrame();
126 void onFrame(PlatformXR::Device::FrameData&&);
127 void applyPendingRenderState();
128
129 XREnvironmentBlendMode m_environmentBlendMode;
130 XRInteractionMode m_interactionMode;
131 XRVisibilityState m_visibilityState;
132 UniqueRef<WebXRInputSourceArray> m_inputSources;
133 bool m_ended { false };
134 std::optional<EndPromise> m_endPromise;
135
136 WebXRSystem& m_xrSystem;
137 XRSessionMode m_mode;
138 WeakPtr<PlatformXR::Device> m_device;
139 FeatureList m_requestedFeatures;
140 RefPtr<WebXRRenderState> m_activeRenderState;
141 RefPtr<WebXRRenderState> m_pendingRenderState;
142 std::unique_ptr<WebXRViewerSpace> m_viewerReferenceSpace;
143 MonotonicTime m_timeOrigin;
144
145 unsigned m_nextCallbackId { 1 };
146 Vector<Ref<XRFrameRequestCallback>> m_callbacks;
147
148 Vector<PlatformXR::Device::ViewData> m_views;
149 PlatformXR::Device::FrameData m_frameData;
150
151 double m_minimumInlineFOV { 0.0 };
152 double m_maximumInlineFOV { piFloat };
153
154 // In meters.
155 double m_minimumNearClipPlane { 0.1 };
156 double m_maximumFarClipPlane { 1000.0 };
157
158 // https://immersive-web.github.io/webxr/#xrsession-promise-resolved
159 bool m_inputInitialized { false };
160};
