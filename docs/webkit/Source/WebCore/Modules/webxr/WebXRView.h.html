27
28#if ENABLE(WEBXR)
29
30#include "JSValueInWrappedObject.h"
31#include "WebXRRigidTransform.h"
32#include "XREye.h"
33#include <JavaScriptCore/Float32Array.h>
34#include <wtf/IsoMalloc.h>
35#include <wtf/Ref.h>
36#include <wtf/RefCounted.h>
37#include <wtf/Vector.h>
38
39namespace WebCore {
40
41class WebXRFrame;
42class WebXRRigidTransform;
43class WebXRSession;
44
45class WebXRView : public RefCounted<WebXRView> {
46 WTF_MAKE_ISO_ALLOCATED_EXPORT(WebXRView, WEBCORE_EXPORT);
47public:
48 WEBCORE_EXPORT static Ref<WebXRView> create(Ref<WebXRFrame>&&, XREye, Ref<WebXRRigidTransform>&&, Ref<Float32Array>&&);
49 WEBCORE_EXPORT ~WebXRView();
50
51 const WebXRFrame& frame() const { return m_frame.get(); }
52 XREye eye() const { return m_eye; }
53 const Float32Array& projectionMatrix() const { return m_projection.get(); }
54 const WebXRRigidTransform& transform() const { return m_transform.get(); }
55
56 std::optional<double> recommendedViewportScale() const;
57 void requestViewportScale(std::optional<double>);
58
59 double requestedViewportScale() const { return m_requestedViewportScale; }
60 bool isViewportModifiable() const { return m_viewportModifiable; }
61 void setViewportModifiable(bool modifiable) { m_viewportModifiable = modifiable; }
62
63 JSValueInWrappedObject& cachedProjectionMatrix() { return m_cachedProjectionMatrix; }
64
65private:
66 WebXRView(Ref<WebXRFrame>&&, XREye, Ref<WebXRRigidTransform>&&, Ref<Float32Array>&&);
67
68 Ref<WebXRFrame> m_frame;
69 XREye m_eye;
70 Ref<WebXRRigidTransform> m_transform;
71 Ref<Float32Array> m_projection;
72 bool m_viewportModifiable { false };
73 double m_requestedViewportScale { 1.0 };
74 JSValueInWrappedObject m_cachedProjectionMatrix;
75};
76
77} // namespace WebCore
78
79#endif // ENABLE(WEBXR)
