26#pragma once
27
28#if ENABLE(WEBXR)
29
30#include "Event.h"
31#include <wtf/RefPtr.h>
32
33namespace WebCore {
34
35class WebXRFrame;
36class WebXRInputSource;
37
38class XRInputSourceEvent final : public Event {
39 WTF_MAKE_ISO_ALLOCATED(XRInputSourceEvent);
40public:
41 struct Init : EventInit {
42 RefPtr<WebXRFrame> frame;
43 RefPtr<WebXRInputSource> inputSource;
44 };
45
46 static Ref<XRInputSourceEvent> create(const AtomString&, const Init&, IsTrusted = IsTrusted::No);
47 virtual ~XRInputSourceEvent();
48
49 const WebXRFrame& frame() const;
50 const WebXRInputSource& inputSource() const;
51 void setFrameActive(bool);
52
53 EventInterface eventInterface() const final { return XRInputSourceEventInterfaceType; }
54
55private:
56 XRInputSourceEvent(const AtomString&, const Init&, IsTrusted);
57
58 RefPtr<WebXRFrame> m_frame;
59 RefPtr<WebXRInputSource> m_inputSource;
60 std::optional<int> m_buttonIndex;
61};
62
63} // namespace WebCore
64
65#endif // ENABLE(WEBXR)
