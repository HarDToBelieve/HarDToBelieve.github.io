36namespace PAL {
37
38void registerNotifyCallback(const String& notifyID, WTF::Function<void()>&& callback)
39{
40#if PLATFORM(COCOA)
41 int token;
42 notify_register_dispatch(notifyID.utf8().data(), &token, dispatch_get_main_queue(), makeBlockPtr([callback = WTFMove(callback)](int) {
43 callback();
44 }).get());
45#else
46 UNUSED_PARAM(notifyID);
47 UNUSED_PARAM(callback);
48#endif
49}

51} // namespace WebCore
