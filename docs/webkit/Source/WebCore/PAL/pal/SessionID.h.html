26#pragma once
27
28#include <optional>
29#include <wtf/HashFunctions.h>
30#include <wtf/HashTraits.h>
31
32namespace PAL {
33
34class SessionID {
35public:
36 SessionID() = delete;
37
38 enum SessionConstants : uint64_t {
39 EphemeralSessionMask = 0x8000000000000000,
40 DefaultSessionID = 1,
41 LegacyPrivateSessionID = DefaultSessionID | EphemeralSessionMask,
42 HashTableEmptyValueID = 0,
43 HashTableDeletedValueID = std::numeric_limits<uint64_t>::max(),
44 };
45
46 static SessionID defaultSessionID() { return SessionID(DefaultSessionID); }
47 static SessionID legacyPrivateSessionID() { return SessionID(LegacyPrivateSessionID); }
48
49 explicit SessionID(WTF::HashTableDeletedValueType)
50 : m_identifier(HashTableDeletedValueID)
51 {
52 }
53
54 explicit SessionID(WTF::HashTableEmptyValueType)
55 : m_identifier(HashTableEmptyValueID)
56 {
57 }
58
59 explicit SessionID(uint64_t identifier)
60 : m_identifier(identifier)
61 {
62 }
63
64 PAL_EXPORT static SessionID generateEphemeralSessionID();
65 PAL_EXPORT static SessionID generatePersistentSessionID();
66 PAL_EXPORT static void enableGenerationProtection();
67
68 bool isValid() const { return isValidSessionIDValue(m_identifier); }
69 bool isEphemeral() const { return m_identifier & EphemeralSessionMask && m_identifier != HashTableDeletedValueID; }
70 bool isHashTableDeletedValue() const { return m_identifier == HashTableDeletedValueID; }
71
72 uint64_t toUInt64() const { return m_identifier; }
73 bool operator==(SessionID sessionID) const { return m_identifier == sessionID.m_identifier; }
74 bool operator!=(SessionID sessionID) const { return m_identifier != sessionID.m_identifier; }
75 bool isAlwaysOnLoggingAllowed() const { return !isEphemeral(); }
76
77 template<class Encoder> void encode(Encoder&) const;
78 template<class Decoder> static std::optional<SessionID> decode(Decoder&);
79
80 SessionID isolatedCopy() const { return *this; }
81
82 explicit operator bool() const { return m_identifier; }
83
84private:
85 static bool isValidSessionIDValue(uint64_t sessionID) { return sessionID != HashTableEmptyValueID && sessionID != HashTableDeletedValueID; }
86
87 uint64_t m_identifier;
88};
89
90template<class Encoder>
91void SessionID::encode(Encoder& encoder) const
92{
93 ASSERT(isValid());
94 encoder << m_identifier;
95}
96
97template<class Decoder>
98std::optional<SessionID> SessionID::decode(Decoder& decoder)
99{
100 std::optional<uint64_t> sessionID;
101 decoder >> sessionID;
102 if (!sessionID || !isValidSessionIDValue(*sessionID))
103 return std::nullopt;
104 return SessionID { *sessionID };
105}
106
107} // namespace PAL
108
109namespace WTF {
110
111struct SessionIDHash {
112 static unsigned hash(const PAL::SessionID& p) { return intHash(p.toUInt64()); }
113 static bool equal(const PAL::SessionID& a, const PAL::SessionID& b) { return a == b; }
114 static const bool safeToCompareToEmptyOrDeleted = true;
115};
116
117template<> struct HashTraits<PAL::SessionID> : GenericHashTraits<PAL::SessionID> {
118 static PAL::SessionID emptyValue() { return PAL::SessionID(HashTableEmptyValue); }
119 static void constructDeletedValue(PAL::SessionID& slot) { new (NotNull, &slot) PAL::SessionID(HashTableDeletedValue); }
120 static bool isDeletedValue(const PAL::SessionID& slot) { return slot.isHashTableDeletedValue(); }
121};
122
123template<> struct DefaultHash<PAL::SessionID> : SessionIDHash { };
