46std::optional<OutputContext>& OutputContext::sharedAudioPresentationOutputContext()
47{
48 static NeverDestroyed<std::optional<OutputContext>> sharedAudioPresentationOutputContext = [] () -> std::optional<OutputContext> {
49 if (![PAL::getAVOutputContextClass() respondsToSelector:@selector(sharedAudioPresentationOutputContext)])
50 return std::nullopt;
51
52#if PLATFORM(MAC) || PLATFORM(MACCATALYST)
53 AVOutputContext* context = [getAVOutputContextClass() sharedSystemAudioContext];
54#else
55 auto context = [getAVOutputContextClass() sharedAudioPresentationOutputContext];
56#endif
57 if (!context)
58 return std::nullopt;
59
60 return OutputContext(retainPtr(context));
61 }();
62 return sharedAudioPresentationOutputContext;
63}

65bool OutputContext::supportsMultipleOutputDevices()
66{
67 return [m_context respondsToSelector:@selector(supportsMultipleOutputDevices)]
68 && [m_context respondsToSelector:@selector(outputDevices)]
69 && [m_context supportsMultipleOutputDevices];
70}
71
72String OutputContext::deviceName()
73{
74 if (!supportsMultipleOutputDevices())
75 return [m_context deviceName];
76
77 StringBuilder builder;
78 auto devices = outputDevices();
79 auto iterator = devices.begin();
80
81 while (iterator != devices.end()) {
82 builder.append(iterator->name());
83
84 if (++iterator != devices.end())
85 builder.append(" + ");
86 }
87
88 return builder.toString();
89}
90
91Vector<OutputDevice> OutputContext::outputDevices() const
92{
93 if (![m_context respondsToSelector:@selector(outputDevices)]) {
94 if (auto *outputDevice = [m_context outputDevice])
95 return { retainPtr(outputDevice) };
96 return { };
97 }
98
99 auto *avOutputDevices = [m_context outputDevices];
100 Vector<OutputDevice> outputDevices;
101 outputDevices.reserveInitialCapacity(avOutputDevices.count);
102 for (AVOutputDevice *device in avOutputDevices)
103 outputDevices.uncheckedAppend({ retainPtr(device) });
104 return outputDevices;
105}
