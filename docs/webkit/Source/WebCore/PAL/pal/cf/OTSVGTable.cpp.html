25#include "config.h"
26#include "OTSVGTable.h"
27
28#include "CoreTextSoftLink.h"

30namespace PAL {
31
32OTSVGTable::OTSVGTable() = default;
33
34OTSVGTable::OTSVGTable(CFDataRef svgTable, unsigned unitsPerEm, CGFloat fontSize)
35 : table(isOTSVGFrameworkAvailable() ? softLinkOTSVGOTSVGTableCreateFromData(svgTable, unitsPerEm, fontSize) : nullptr)
36{
37}
38
39OTSVGTable::OTSVGTable(OTSVGTable&& other)
40{
41 std::swap(table, other.table);
42}
43
44OTSVGTable::~OTSVGTable()
45{
46 if (isOTSVGFrameworkAvailable() && table)
47 softLinkOTSVGOTSVGTableRelease(table);
48}
49
50OTSVGTable& OTSVGTable::operator=(OTSVGTable&& other)
51{
52 if (isOTSVGFrameworkAvailable() && table) {
53 softLinkOTSVGOTSVGTableRelease(table);
54 table = nullptr;
55 }
56 std::swap(table, other.table);
57 return *this;
58}
