47PAL_EXPORT void* AVFoundationLibrary(bool isOptional = false);
48void* AVFoundationLibrary(bool isOptional)
49{
50 static void* frameworkLibrary;
51 static dispatch_once_t once;
52 dispatch_once(&once, ^{
53 frameworkLibrary = dlopen("/System/Library/Frameworks/AVFoundation.framework/AVFoundation", RTLD_NOW);
54 if (!isOptional)
55 RELEASE_ASSERT_WITH_MESSAGE(frameworkLibrary, "%s", dlerror());
56
57 class_addMethod(objc_getClass("AVPlayerItem"), @selector(automaticallyNotifiesObserversOfSuppressesVideoLayers), (IMP)justReturnsNO, "B@:");
58 });
59 return frameworkLibrary;
60}
