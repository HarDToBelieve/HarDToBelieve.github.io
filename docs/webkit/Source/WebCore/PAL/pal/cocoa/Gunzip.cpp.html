26#include "config.h"
27#include "Gunzip.h"
28
29#include <compression.h>
30
31namespace PAL {
32
33Vector<LChar> gunzip(const unsigned char* data, size_t length)
34{
35 Vector<LChar> result;
36
37 // Parse the gzip header.
38 auto checks = [&]() {
39 return length >= 10 && data[0] == 0x1f && data[1] == 0x8b && data[2] == 0x8 && data[3] == 0x0;
40 };
41 ASSERT(checks());
42 if (!checks())
43 return { };
44
45 constexpr auto ignoredByteCount = 10;
46
47 compression_stream stream;
48 auto status = compression_stream_init(&stream, COMPRESSION_STREAM_DECODE, COMPRESSION_ZLIB);
49 ASSERT(status == COMPRESSION_STATUS_OK);
50 if (status != COMPRESSION_STATUS_OK)
51 return { };
52 stream.dst_ptr = result.data();
53 stream.dst_size = result.size();
54 stream.src_ptr = data + ignoredByteCount;
55 stream.src_size = length - ignoredByteCount;
56 size_t offset = 0;
57
58 do {
59 uint8_t* originalDestinationPointer = stream.dst_ptr;
60 status = compression_stream_process(&stream, COMPRESSION_STREAM_FINALIZE);
61 uint8_t* newDestinationPointer = stream.dst_ptr;
62 offset += newDestinationPointer - originalDestinationPointer;
63 switch (status) {
64 case COMPRESSION_STATUS_OK: {
65 auto newSize = offset * 1.5 + 1;
66 // FIXME: We can get better performance if we, instead of resizing the buffer, we allocate distinct chunks and have a postprocessing step which concatenates the chunks together.
67 if (newSize > result.size())
68 result.grow(newSize);
69 stream.dst_ptr = result.data() + offset;
70 stream.dst_size = result.size() - offset;
71 break;
72 }
73 case COMPRESSION_STATUS_END:
74 status = compression_stream_destroy(&stream);
75 ASSERT(status == COMPRESSION_STATUS_OK);
76 if (status == COMPRESSION_STATUS_OK) {
77 result.shrink(stream.dst_ptr - result.data());
78 return result;
79 }
80 return { };
81 case COMPRESSION_STATUS_ERROR:
82 default:
83 ASSERT_NOT_REACHED();
84 compression_stream_destroy(&stream);
85 return { };
86 }
87 } while (true);
88}
