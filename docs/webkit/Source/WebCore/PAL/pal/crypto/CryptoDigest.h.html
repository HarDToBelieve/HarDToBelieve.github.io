31namespace PAL {

33struct CryptoDigestContext;

35class CryptoDigest {
36 WTF_MAKE_NONCOPYABLE(CryptoDigest);
37public:
38 enum class Algorithm {
39 SHA_1,
40 SHA_224,
41 SHA_256,
42 SHA_384,
43 SHA_512,
44 };
45 PAL_EXPORT static std::unique_ptr<CryptoDigest> create(Algorithm);
46 PAL_EXPORT ~CryptoDigest();
47
48 PAL_EXPORT void addBytes(const void* input, size_t length);
49 PAL_EXPORT Vector<uint8_t> computeHash();
50
51private:
52 CryptoDigest();
53
54 std::unique_ptr<CryptoDigestContext> m_context;
55};

57} // namespace PAL
