32#include <wtf/RetainPtr.h>

34namespace PAL {
35
36std::unique_ptr<SleepDisabler> SleepDisabler::create(const char* reason, Type type)
37{
38 return std::unique_ptr<SleepDisabler>(new SleepDisablerCocoa(reason, type));
39}
40
41SleepDisablerCocoa::SleepDisablerCocoa(const char* reason, Type type)
42 : SleepDisabler(reason, type)
43{
44 RetainPtr<CFStringRef> reasonCF = adoptCF(CFStringCreateWithCString(kCFAllocatorDefault, reason, kCFStringEncodingUTF8));
45
46 CFStringRef assertionType;
47 switch (type) {
48 case Type::Display:
49 assertionType = kIOPMAssertionTypePreventUserIdleDisplaySleep;
50 break;
51 case Type::System:
52 assertionType = kIOPMAssertionTypePreventUserIdleSystemSleep;
53 break;
54 default:
55 ASSERT_NOT_REACHED();
56 assertionType = nullptr;
57 break;
58 }
59 IOPMAssertionCreateWithDescription(assertionType, reasonCF.get(), nullptr, nullptr, nullptr, 0, nullptr, &m_sleepAssertion);
60}
61
62SleepDisablerCocoa::~SleepDisablerCocoa()
63{
64 IOPMAssertionRelease(m_sleepAssertion);
65}

67} // namespace PAL
