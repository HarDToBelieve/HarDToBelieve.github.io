32#import <wtf/MainThread.h>

34namespace PAL {
35
36std::unique_ptr<SystemSleepListener> SystemSleepListener::create(Client& client)
37{
38 return std::unique_ptr<SystemSleepListener>(new SystemSleepListenerMac(client));
39}
40
41SystemSleepListenerMac::SystemSleepListenerMac(Client& client)
42 : SystemSleepListener(client)
43 , m_sleepObserver(nil)
44 , m_wakeObserver(nil)
45{
46 NSNotificationCenter *center = [[NSWorkspace sharedWorkspace] notificationCenter];
47 NSOperationQueue *queue = [NSOperationQueue mainQueue];
48
49 auto weakThis = makeWeakPtr(*this);
50
51 m_sleepObserver = [center addObserverForName:NSWorkspaceWillSleepNotification object:nil queue:queue usingBlock:^(NSNotification *) {
52 callOnMainThread([weakThis] {
53 if (weakThis)
54 weakThis->m_client.systemWillSleep();
55 });
56 }];
57
58 m_wakeObserver = [center addObserverForName:NSWorkspaceDidWakeNotification object:nil queue:queue usingBlock:^(NSNotification *) {
59 callOnMainThread([weakThis] {
60 if (weakThis)
61 weakThis->m_client.systemDidWake();
62 });
63 }];
64}
65
66SystemSleepListenerMac::~SystemSleepListenerMac()
67{
68 NSNotificationCenter* center = [[NSWorkspace sharedWorkspace] notificationCenter];
69 [center removeObserver:m_sleepObserver];
70 [center removeObserver:m_wakeObserver];
71}
