48AccessibilityRole AXImage::roleValue() const
49{
50 auto ariaRole = ariaRoleAttribute();
51 if (ariaRole != AccessibilityRole::Unknown)
52 return ariaRole;
53
54 return AccessibilityRole::Image;
55}
56
57std::optional<AXCoreObject::AccessibilityChildrenVector> AXImage::imageOverlayElements()
58{
59 AXTRACE("AXImage::imageOverlayElements");
60
61 auto& children = this->children();
62 if (children.size())
63 return children;
64
65#if ENABLE(IMAGE_ANALYSIS)
66 auto* page = this->page();
67 if (!page)
68 return std::nullopt;
69
70 page->chrome().client().requestTextRecognition(*element(), [] (RefPtr<Element>&& imageOverlayHost) {
71 if (!imageOverlayHost)
72 return;
73
74 if (auto* axObjectCache = imageOverlayHost->document().existingAXObjectCache())
75 axObjectCache->postNotification(imageOverlayHost.get(), AXObjectCache::AXImageOverlayChanged);
76 });
77#endif
78
79 return std::nullopt;
80}
