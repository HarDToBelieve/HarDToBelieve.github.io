53AccessibilityTable* AccessibilityARIAGridCell::parentTable() const
54{
55 // ARIA gridcells may have multiple levels of unignored ancestors that are not the parent table,
56 // including rows and interactive rowgroups. In addition, poorly-formed grids may contain elements
57 // which pass the tests for inclusion.
58 for (auto* parent = parentObjectUnignored(); parent; parent = parent->parentObjectUnignored()) {
59 if (is<AccessibilityTable>(*parent) && downcast<AccessibilityTable>(*parent).isExposable())
60 return downcast<AccessibilityTable>(parent);
61 }
62
63 return nullptr;
64}

66std::pair<unsigned, unsigned> AccessibilityARIAGridCell::rowIndexRange() const
67{
68 std::pair<unsigned, unsigned> rowRange { 0, 1 };
69 AXCoreObject* parent = parentObjectUnignored();
70 if (!parent)
71 return rowRange;
72
73 if (is<AccessibilityTableRow>(*parent)) {
74 // We already got a table row, use its API.
75 rowRange.first = downcast<AccessibilityTableRow>(*parent).rowIndex();
76 } else if (is<AccessibilityTable>(*parent) && downcast<AccessibilityTable>(*parent).isExposable()) {
77 // We reached the parent table, so we need to inspect its
78 // children to determine the row index for the cell in it.
79 unsigned columnCount = downcast<AccessibilityTable>(*parent).columnCount();
80 if (!columnCount)
81 return rowRange;
82
83 const auto& siblings = parent->children();
84 unsigned childrenSize = siblings.size();
85 for (unsigned k = 0; k < childrenSize; ++k) {
86 if (siblings[k].get() == this) {
87 rowRange.first = k / columnCount;
88 break;
89 }
90 }
91 }
92
93 // ARIA 1.1, aria-rowspan attribute is intended for cells and gridcells which are not contained in a native table.
94 // So we should check for that attribute here.
95 rowRange.second = axRowSpanWithRowIndex(rowRange.first);
96
97 return rowRange;
98}

100unsigned AccessibilityARIAGridCell::axRowSpanWithRowIndex(unsigned rowIndex) const
101{
102 int rowSpan = AccessibilityTableCell::axRowSpan();
103 if (rowSpan == -1) {
104 auto range = AccessibilityTableCell::rowIndexRange();
105 return std::max(range.second, static_cast<unsigned>(1));
106 }
107
108 AXCoreObject* parent = parentObjectUnignored();
109 if (!parent)
110 return 1;
111
112 // Setting the value to 0 indicates that the cell or gridcell is to span all the remaining rows in the row group.
113 if (!rowSpan) {
114 // rowSpan defaults to 1.
115 rowSpan = 1;
116 if (AccessibilityObject* parentRowGroup = this->parentRowGroup()) {
117 // If the row group is the parent table, we use total row count to calculate the span.
118 if (is<AccessibilityTable>(*parentRowGroup))
119 rowSpan = downcast<AccessibilityTable>(*parentRowGroup).rowCount() - rowIndex;
120 // Otherwise, we have to get the index for the current row within the parent row group.
121 else if (is<AccessibilityTableRow>(*parent)) {
122 const auto& siblings = parentRowGroup->children();
123 unsigned rowCount = siblings.size();
124 for (unsigned k = 0; k < rowCount; ++k) {
125 if (siblings[k].get() == parent) {
126 rowSpan = rowCount - k;
127 break;
128 }
129 }
130 }
131 }
132 }
133
134 return rowSpan;
135}

137std::pair<unsigned, unsigned> AccessibilityARIAGridCell::columnIndexRange() const
138{
139 std::pair<unsigned, unsigned> columnRange { 0, 1 };
140 AXCoreObject* parent = parentObjectUnignored();
141 if (!parent)
142 return columnRange;
143
144 if (!is<AccessibilityTableRow>(*parent)
145 && !(is<AccessibilityTable>(*parent) && downcast<AccessibilityTable>(*parent).isExposable()))
146 return columnRange;
147
148 const AccessibilityChildrenVector& siblings = parent->children();
149 unsigned childrenSize = siblings.size();
150 unsigned indexWithSpan = 0;
151 for (unsigned k = 0; k < childrenSize; ++k) {
152 auto* child = siblings[k].get();
153 if (child == this) {
154 columnRange.first = indexWithSpan;
155 break;
156 }
157 indexWithSpan += is<AccessibilityTableCell>(*child) ? std::max(downcast<AccessibilityTableCell>(*child).axColumnSpan(), 1) : 1;
158 }
159
160 // ARIA 1.1, aria-colspan attribute is intended for cells and gridcells which are not contained in a native table.
161 // So we should check for that attribute here.
162 int columnSpan = AccessibilityTableCell::axColumnSpan();
163 if (columnSpan == -1) {
164 auto range = AccessibilityTableCell::columnIndexRange();
165 columnSpan = range.second;
166 }
167
168 columnRange.second = std::max(columnSpan, 1);
169
170 return columnRange;
171}

173AccessibilityObject* AccessibilityARIAGridCell::parentRowGroup() const
174{
175 for (AccessibilityObject* parent = parentObject(); parent; parent = parent->parentObject()) {
176 if (parent->hasTagName(theadTag) || parent->hasTagName(tbodyTag) || parent->hasTagName(tfootTag) || parent->roleValue() == AccessibilityRole::RowGroup)
177 return parent;
178 }
179
180 // If there's no row group found, we use the parent table as the row group.
181 return parentTable();
182}

184String AccessibilityARIAGridCell::readOnlyValue() const
185{
186 if (hasAttribute(aria_readonlyAttr))
187 return getAttribute(aria_readonlyAttr).string().convertToASCIILowercase();
188
189 // ARIA 1.1 requires user agents to propagate the grid's aria-readonly value to all
190 // gridcell elements if the property is not present on the gridcell element itelf.
191 if (AccessibilityObject* parent = parentTable())
192 return parent->readOnlyValue();
193
194 return String();
195}
