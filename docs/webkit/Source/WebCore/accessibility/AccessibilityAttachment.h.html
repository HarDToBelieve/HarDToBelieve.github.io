29#if ENABLE(ATTACHMENT_ELEMENT)
30
31#include "AccessibilityRenderObject.h"
32
33namespace WebCore {
34
35class HTMLAttachmentElement;
36class RenderAttachment;
37
38class AccessibilityAttachment final : public AccessibilityRenderObject {
39public:
40 static Ref<AccessibilityAttachment> create(RenderAttachment*);
41 HTMLAttachmentElement* attachmentElement() const;
42 bool hasProgress(float* progress = nullptr) const;
43
44private:
45 AccessibilityRole roleValue() const override { return AccessibilityRole::Button; }
46 bool isAttachmentElement() const override { return true; }
47
48 String roleDescription() const override;
49 float valueForRange() const override;
50 bool computeAccessibilityIsIgnored() const override;
51 void accessibilityText(Vector<AccessibilityText>&) const override;
52 explicit AccessibilityAttachment(RenderAttachment*);
53};
54
55} // namespace WebCore

57SPECIALIZE_TYPE_TRAITS_ACCESSIBILITY(AccessibilityAttachment, isAttachmentElement())

59#endif // ENABLE(ATTACHMENT_ELEMENT)
