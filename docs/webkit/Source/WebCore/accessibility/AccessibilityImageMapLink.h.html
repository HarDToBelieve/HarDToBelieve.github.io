31#include "AccessibilityMockObject.h"
32#include "HTMLAreaElement.h"
33#include "HTMLMapElement.h"
34
35namespace WebCore {
36
37class AccessibilityImageMapLink final : public AccessibilityMockObject {
38public:
39 static Ref<AccessibilityImageMapLink> create();
40 virtual ~AccessibilityImageMapLink();
41
42 void setHTMLAreaElement(HTMLAreaElement* element) { m_areaElement = element; }
43 HTMLAreaElement* areaElement() const { return m_areaElement.get(); }
44
45 void setHTMLMapElement(HTMLMapElement* element) { m_mapElement = element; }
46 HTMLMapElement* mapElement() const { return m_mapElement.get(); }
47
48 Node* node() const override { return m_areaElement.get(); }
49
50 AccessibilityRole roleValue() const override;
51 bool isEnabled() const override { return true; }
52
53 Element* anchorElement() const override;
54 Element* actionElement() const override;
55 URL url() const override;
56 bool isLink() const override { return true; }
57 bool isLinked() const override { return true; }
58 String title() const override;
59 String accessibilityDescription() const override;
60 AccessibilityObject* parentObject() const override;
61
62 String stringValueForMSAA() const override;
63 String nameForMSAA() const override;
64
65 LayoutRect elementRect() const override;
66
67private:
68 AccessibilityImageMapLink();
69
70 void detachFromParent() override;
71 Path elementPath() const override;
72 RenderElement* imageMapLinkRenderer() const;
73 void accessibilityText(Vector<AccessibilityText>&) const override;
74 bool isImageMapLink() const override { return true; }
75 bool supportsPath() const override { return true; }
76
77 RefPtr<HTMLAreaElement> m_areaElement;
78 RefPtr<HTMLMapElement> m_mapElement;
79};
80
81} // namespace WebCore
