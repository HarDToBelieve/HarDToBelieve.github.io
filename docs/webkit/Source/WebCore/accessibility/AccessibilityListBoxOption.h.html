31#include "AccessibilityObject.h"

33namespace WebCore {
34
35class HTMLElement;
36class HTMLSelectElement;
37
38class AccessibilityListBoxOption final : public AccessibilityObject {
39public:
40 static Ref<AccessibilityListBoxOption> create(HTMLElement&);
41 virtual ~AccessibilityListBoxOption();
42
43 bool isSelected() const final;
44 void setSelected(bool) final;
45
46private:
47 explicit AccessibilityListBoxOption(HTMLElement&);
48
49 AccessibilityRole roleValue() const final { return AccessibilityRole::ListBoxOption; }
50 bool isEnabled() const final;
51 bool isSelectedOptionActive() const final;
52 String stringValue() const final;
53 Element* actionElement() const final;
54 Node* node() const final;
55 bool canSetSelectedAttribute() const final;
56
57 LayoutRect elementRect() const final;
58 AccessibilityObject* parentObject() const final;
59
60 bool isListBoxOption() const final { return true; }
61 bool canHaveChildren() const final { return false; }
62 HTMLSelectElement* listBoxOptionParentNode() const;
63 int listBoxOptionIndex() const;
64 IntRect listBoxOptionRect() const;
65 AccessibilityObject* listBoxOptionAccessibilityObject(HTMLElement*) const;
66 bool computeAccessibilityIsIgnored() const final;
67
68 WeakPtr<HTMLElement> m_optionElement;
69};
70
71} // namespace WebCore
