26#pragma once
27
28#include "AccessibilityObject.h"
29
30namespace WebCore {
31
32class HTMLOptionElement;
33
34class AccessibilityMenuListOption final : public AccessibilityObject {
35public:
36 static Ref<AccessibilityMenuListOption> create(HTMLOptionElement&);
37
38private:
39 explicit AccessibilityMenuListOption(HTMLOptionElement&);
40
41 bool isMenuListOption() const final { return true; }
42
43 AccessibilityRole roleValue() const final { return AccessibilityRole::MenuListOption; }
44 bool canHaveChildren() const final { return false; }
45
46 Element* actionElement() const final;
47 Node* node() const final;
48 bool isEnabled() const final;
49 bool isVisible() const final;
50 bool isOffScreen() const final;
51 bool isSelected() const final;
52 String nameForMSAA() const final;
53 void setSelected(bool) final;
54 bool canSetSelectedAttribute() const final;
55 LayoutRect elementRect() const final;
56 String stringValue() const final;
57 bool computeAccessibilityIsIgnored() const final;
58
59 WeakPtr<HTMLOptionElement> m_element;
60};
61
62} // namespace WebCore
