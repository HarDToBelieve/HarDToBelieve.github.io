71AccessibilityMenuListOption* AccessibilityMenuListPopup::menuListOptionAccessibilityObject(HTMLElement* element) const
72{
73 if (!element || !element->inRenderedDocument())
74 return nullptr;
75
76 return downcast<AccessibilityMenuListOption>(document()->axObjectCache()->getOrCreate(element));
77}

79bool AccessibilityMenuListPopup::press()
80{
81 if (!m_parent)

84 m_parent->press();

88void AccessibilityMenuListPopup::addChildren()
89{
90 if (!m_parent)
91 return;
92
93 Node* selectNode = m_parent->node();
94 if (!selectNode)
95 return;
96
97 m_haveChildren = true;
98
99 for (const auto& listItem : downcast<HTMLSelectElement>(*selectNode).listItems()) {
100 // FIXME: Why does AccessibilityListBox::addChildren check accessibilityIsIgnored but this does not?
101 if (auto option = menuListOptionAccessibilityObject(listItem))
102 m_children.append(option);
103 }
104}
105
106void AccessibilityMenuListPopup::childrenChanged()
107{
108 AXObjectCache* cache = axObjectCache();
109 for (size_t i = m_children.size(); i > 0 ; --i) {
110 AXCoreObject* child = m_children[i - 1].get();
111 if (child->actionElement() && !child->actionElement()->inRenderedDocument()) {
112 child->detachFromParent();
113 cache->remove(child->objectID());
114 }
115 }
116
117 m_children.clear();
118 m_haveChildren = false;
119 addChildren();
120}
121
122void AccessibilityMenuListPopup::didUpdateActiveOption(int optionIndex)
123{
124 ASSERT_ARG(optionIndex, optionIndex >= 0);
125 ASSERT_ARG(optionIndex, optionIndex < static_cast<int>(m_children.size()));
126
127 RefPtr<AXCoreObject> child = m_children[optionIndex].get();
128
129 if (auto* cache = axObjectCache()) {
130 cache->postNotification(child.get(), document(), AXObjectCache::AXFocusedUIElementChanged);
131 cache->postNotification(child.get(), document(), AXObjectCache::AXMenuListItemSelected);
132 }
