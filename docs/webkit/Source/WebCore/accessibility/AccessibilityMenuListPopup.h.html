27
28#include "AccessibilityMockObject.h"
29
30namespace WebCore {
31
32class AccessibilityMenuList;
33class AccessibilityMenuListOption;
34class HTMLElement;
35
36class AccessibilityMenuListPopup final : public AccessibilityMockObject {
37public:
38 static Ref<AccessibilityMenuListPopup> create() { return adoptRef(*new AccessibilityMenuListPopup); }
39
40 bool isEnabled() const override;
41 bool isOffScreen() const override;
42
43 void didUpdateActiveOption(int optionIndex);
44
45private:
46 AccessibilityMenuListPopup();
47
48 bool isMenuListPopup() const override { return true; }
49
50 LayoutRect elementRect() const override { return LayoutRect(); }
51 AccessibilityRole roleValue() const override { return AccessibilityRole::MenuListPopup; }
52
53 bool isVisible() const override;
54 bool press() override;
55 void addChildren() override;
56 void childrenChanged() override;
57 bool computeAccessibilityIsIgnored() const override;
58
59 AccessibilityMenuListOption* menuListOptionAccessibilityObject(HTMLElement*) const;
60};
61
62} // namespace WebCore
