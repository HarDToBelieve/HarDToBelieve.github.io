58AccessibilityObject* AccessibilitySVGElement::targetForUseElement() const
59{
60 if (!is<SVGUseElement>(element()))
61 return nullptr;
62
63 SVGUseElement& use = downcast<SVGUseElement>(*element());
64 String href = use.href();
65 if (href.isEmpty())
66 href = getAttribute(HTMLNames::hrefAttr);
67
68 auto target = SVGURIReference::targetElementFromIRIString(href, use.treeScope());
69 if (!target.element)
70 return nullptr;
71 return axObjectCache()->getOrCreate(target.element.get());
72}

74template <typename ChildrenType>
75Element* AccessibilitySVGElement::childElementWithMatchingLanguage(ChildrenType& children) const
76{
77 String languageCode = language();
78 if (languageCode.isEmpty())
79 languageCode = defaultLanguage();
80
81 // The best match for a group of child SVG2 'title' or 'desc' elements may be the one
82 // which lacks a 'lang' attribute value. However, indexOfBestMatchingLanguageInList()
83 // currently bases its decision on non-empty strings. Furthermore, we cannot count on
84 // that child element having a given position. So we'll look for such an element while
85 // building the language list and save it as our fallback.
86
87 Element* fallback = nullptr;
88 Vector<String> childLanguageCodes;
89 Vector<Element*> elements;
90 for (auto& child : children) {
91 auto& lang = child.attributeWithoutSynchronization(SVGNames::langAttr);
92 childLanguageCodes.append(lang);
93 elements.append(&child);
94
95 // The current draft of the SVG2 spec states if there are multiple equally-valid
96 // matches, the first match should be used.
97 if (lang.isEmpty() && !fallback)
98 fallback = &child;
99 }
100
101 bool exactMatch;
102 size_t index = indexOfBestMatchingLanguageInList(languageCode, childLanguageCodes, exactMatch);
103 if (index < childLanguageCodes.size())
104 return elements[index];
105
106 return fallback;
107}

109void AccessibilitySVGElement::accessibilityText(Vector<AccessibilityText>& textOrder) const
110{
111 String description = accessibilityDescription();
112 if (!description.isEmpty())
113 textOrder.append(AccessibilityText(description, AccessibilityTextSource::Alternative));
114
115 String helptext = helpText();
116 if (!helptext.isEmpty())
117 textOrder.append(AccessibilityText(helptext, AccessibilityTextSource::Help));
118}

120String AccessibilitySVGElement::accessibilityDescription() const
121{
122 // According to the SVG Accessibility API Mappings spec, the order of priority is:
123 // 1. aria-labelledby
124 // 2. aria-label
125 // 3. a direct child title element (selected according to language)
126 // 4. xlink:title attribute
127 // 5. for a use element, the accessible name calculated for the re-used content
128 // 6. for text container elements, the text content
129
130 String ariaDescription = ariaAccessibilityDescription();
131 if (!ariaDescription.isEmpty())
132 return ariaDescription;
133
134 auto titleElements = childrenOfType<SVGTitleElement>(*element());
135 if (auto titleChild = childElementWithMatchingLanguage(titleElements))
136 return titleChild->textContent();
137
138 if (is<SVGAElement>(element())) {
139 auto& xlinkTitle = element()->attributeWithoutSynchronization(XLinkNames::titleAttr);
140 if (!xlinkTitle.isEmpty())
141 return xlinkTitle;
142 }
143
144 if (is<SVGUseElement>(element())) {
145 if (AccessibilityObject* target = targetForUseElement())
146 return target->accessibilityDescription();
147 }
148
149 // FIXME: This is here to not break the svg-image.html test. But 'alt' is not
150 // listed as a supported attribute of the 'image' element in the SVG spec:
151 // https://www.w3.org/TR/SVG/struct.html#ImageElement
152 if (m_renderer->isSVGImage()) {
153 const AtomString& alt = getAttribute(HTMLNames::altAttr);
154 if (!alt.isNull())
155 return alt;
156 }
157
158 return String();
159}

161String AccessibilitySVGElement::helpText() const
162{
163 // According to the SVG Accessibility API Mappings spec, the order of priority is:
164 // 1. aria-describedby
165 // 2. a direct child desc element
166 // 3. for a use element, the accessible description calculated for the re-used content
167 // 4. for text container elements, the text content, if not used for the name
168 // 5. a direct child title element that provides a tooltip, if not used for the name
169
170 String describedBy = ariaDescribedByAttribute();
171 if (!describedBy.isEmpty())
172 return describedBy;
173
174 auto descriptionElements = childrenOfType<SVGDescElement>(*element());
175 if (auto descriptionChild = childElementWithMatchingLanguage(descriptionElements))
176 return descriptionChild->textContent();
177
178 if (is<SVGUseElement>(element())) {
179 AccessibilityObject* target = targetForUseElement();
180 if (target)
181 return target->helpText();
182 }
183
184 auto titleElements = childrenOfType<SVGTitleElement>(*element());
185 if (auto titleChild = childElementWithMatchingLanguage(titleElements)) {
186 if (titleChild->textContent() != accessibilityDescription())
187 return titleChild->textContent();
188 }
189
190 return String();
191}

193bool AccessibilitySVGElement::computeAccessibilityIsIgnored() const
194{
195 // According to the SVG Accessibility API Mappings spec, items should be excluded if:
196 // * They would be excluded according to the Core Accessibility API Mappings.
197 // * They are neither perceivable nor interactive.
198 // * Their first mappable role is presentational, unless they have a global ARIA
199 // attribute (covered by Core AAM) or at least one 'title' or 'desc' child element.
200 // * They have an ancestor with Children Presentational: True (covered by Core AAM)
201
202 AccessibilityObjectInclusion decision = defaultObjectInclusion();
203 if (decision == AccessibilityObjectInclusion::IgnoreObject)
204 return true;
205
206 if (m_renderer->isSVGHiddenContainer())
207 return true;
208
209 // The SVG AAM states objects with at least one 'title' or 'desc' element MUST be included.
210 // At this time, the presence of a matching 'lang' attribute is not mentioned in the spec.
211 for (const auto& child : childrenOfType<SVGElement>(*element())) {
212 if ((is<SVGTitleElement>(child) || is<SVGDescElement>(child)))
213 return false;
214 }
215
216 if (roleValue() == AccessibilityRole::Presentational || inheritsPresentationalRole())
217 return true;
218
219 if (ariaRoleAttribute() != AccessibilityRole::Unknown)
220 return false;
221
222 // The SVG AAM states text elements should also be included, if they have content.
223 if (m_renderer->isSVGText() || m_renderer->isSVGTextPath()) {
224 for (auto& child : childrenOfType<RenderText>(downcast<RenderElement>(*m_renderer))) {
225 if (!child.isAllCollapsibleWhitespace())
226 return false;
227 }
228 }
229
230 // SVG shapes should not be included unless there's a concrete reason for inclusion.
231 // https://rawgit.com/w3c/aria/master/svg-aam/svg-aam.html#exclude_elements
232 if (m_renderer->isSVGShape()) {
233 if (canSetFocusAttribute() || element()->hasEventListeners())
234 return false;
235 if (auto* svgParent = Accessibility::findAncestor<AccessibilityObject>(*this, true, [] (const AccessibilityObject& object) {
236 return object.hasAttributesRequiredForInclusion() || object.isAccessibilitySVGRoot();
237 }))
238 return !svgParent->hasAttributesRequiredForInclusion();
239 return true;
240 }
241
242 return AccessibilityRenderObject::computeAccessibilityIsIgnored();
243}

245bool AccessibilitySVGElement::inheritsPresentationalRole() const
246{
247 if (canSetFocusAttribute())
248 return false;
249
250 AccessibilityRole role = roleValue();
251 if (role != AccessibilityRole::SVGTextPath && role != AccessibilityRole::SVGTSpan)
252 return false;
253
254 for (AccessibilityObject* parent = parentObject(); parent; parent = parent->parentObject()) {
255 if (is<AccessibilityRenderObject>(*parent) && parent->element()->hasTagName(SVGNames::textTag))
256 return parent->roleValue() == AccessibilityRole::Presentational;
257 }
258
259 return false;
260}

262AccessibilityRole AccessibilitySVGElement::determineAriaRoleAttribute() const
263{
264 AccessibilityRole role = AccessibilityRenderObject::determineAriaRoleAttribute();
265 if (role != AccessibilityRole::Presentational)
266 return role;
267
268 // The presence of a 'title' or 'desc' child element trumps PresentationalRole.
269 // https://lists.w3.org/Archives/Public/public-svg-a11y/2016Apr/0016.html
270 // At this time, the presence of a matching 'lang' attribute is not mentioned.
271 for (const auto& child : childrenOfType<SVGElement>(*element())) {
272 if ((is<SVGTitleElement>(child) || is<SVGDescElement>(child)))
273 return AccessibilityRole::Unknown;
274 }
275
276 return role;
277}
278
279AccessibilityRole AccessibilitySVGElement::determineAccessibilityRole()
280{
281 if ((m_ariaRole = determineAriaRoleAttribute()) != AccessibilityRole::Unknown)
282 return m_ariaRole;
283
284 Element* svgElement = element();
285
286 if (m_renderer->isSVGShape() || m_renderer->isSVGPath() || m_renderer->isSVGImage() || is<SVGUseElement>(svgElement))
287 return AccessibilityRole::Image;
288 if (m_renderer->isSVGForeignObject() || is<SVGGElement>(svgElement))
289 return AccessibilityRole::Group;
290 if (m_renderer->isSVGText())
291 return AccessibilityRole::SVGText;
292 if (m_renderer->isSVGTextPath())
293 return AccessibilityRole::SVGTextPath;
294 if (m_renderer->isSVGTSpan())
295 return AccessibilityRole::SVGTSpan;
296 if (is<SVGAElement>(svgElement))
297 return AccessibilityRole::WebCoreLink;
298
299 return AccessibilityRenderObject::determineAccessibilityRole();
300}
