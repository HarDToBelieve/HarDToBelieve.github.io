58bool AccessibilityTableCell::computeAccessibilityIsIgnored() const
59{
60 AccessibilityObjectInclusion decision = defaultObjectInclusion();
61 if (decision == AccessibilityObjectInclusion::IncludeObject)
62 return false;
63 if (decision == AccessibilityObjectInclusion::IgnoreObject)
64 return true;
65
66 // Ignore anonymous table cells as long as they're not in a table (ie. when display:table is used).
67 RenderObject* renderTable = is<RenderTableCell>(renderer()) ? downcast<RenderTableCell>(*m_renderer).table() : nullptr;
68 bool inTable = renderTable && renderTable->node() && (renderTable->node()->hasTagName(tableTag) || nodeHasRole(renderTable->node(), "grid"));
69 if (!node() && !inTable)
70 return true;
71
72 if (!isTableCell())
73 return AccessibilityRenderObject::computeAccessibilityIsIgnored();
74
75 return false;
76}

78AccessibilityTable* AccessibilityTableCell::parentTable() const
79{
80 if (!is<RenderTableCell>(renderer()))
81 return nullptr;
82
83 // If the document no longer exists, we might not have an axObjectCache.
84 if (!axObjectCache())
85 return nullptr;
86
87 // Do not use getOrCreate. parentTable() can be called while the render tree is being modified
88 // by javascript, and creating a table element may try to access the render tree while in a bad state.
89 // By using only get() implies that the AXTable must be created before AXTableCells. This should
90 // always be the case when AT clients access a table.
91 // https://bugs.webkit.org/show_bug.cgi?id=42652
92 AccessibilityObject* parentTable = axObjectCache()->get(downcast<RenderTableCell>(*m_renderer).table());
93 if (!is<AccessibilityTable>(parentTable))
94 return nullptr;
95
96 // The RenderTableCell's table() object might be anonymous sometimes. We should handle it gracefully
97 // by finding the right table.
98 if (!parentTable->node()) {
99 for (AccessibilityObject* parent = parentObject(); parent; parent = parent->parentObject()) {
100 // If this is a non-anonymous table object, but not an accessibility table, we should stop because
101 // we don't want to choose another ancestor table as this cell's table.
102 if (is<AccessibilityTable>(*parent)) {
103 auto& parentTable = downcast<AccessibilityTable>(*parent);
104 if (parentTable.isExposable())
105 return &parentTable;
106 if (parentTable.node())
107 break;
108 }
109 }
110 return nullptr;
111 }
112
113 return downcast<AccessibilityTable>(parentTable);
114}

116bool AccessibilityTableCell::isTableCell() const
117{
118 // If the parent table is an accessibility table, then we are a table cell.
119 // This used to check if the unignoredParent was a row, but that exploded performance if
120 // this was in nested tables. This check should be just as good.
121 AccessibilityObject* parentTable = this->parentTable();
122 return is<AccessibilityTable>(parentTable) && downcast<AccessibilityTable>(*parentTable).isExposable();
123}

125AccessibilityRole AccessibilityTableCell::determineAccessibilityRole()
126{
127 // AccessibilityRenderObject::determineAccessibleRole provides any ARIA-supplied
128 // role, falling back on the role to be used if we determine here that the element
129 // should not be exposed as a cell. Thus if we already know it's a cell, return that.
130 AccessibilityRole defaultRole = AccessibilityRenderObject::determineAccessibilityRole();
131 if (defaultRole == AccessibilityRole::ColumnHeader || defaultRole == AccessibilityRole::RowHeader || defaultRole == AccessibilityRole::Cell || defaultRole == AccessibilityRole::GridCell)
132 return defaultRole;
133
134 if (!isTableCell())
135 return defaultRole;
136 if (isColumnHeaderCell())
137 return AccessibilityRole::ColumnHeader;
138 if (isRowHeaderCell())
139 return AccessibilityRole::RowHeader;
140
141 return AccessibilityRole::Cell;
142}

149bool AccessibilityTableCell::isColumnHeaderCell() const
150{
151 const AtomString& scope = getAttribute(scopeAttr);
152 if (scope == "col" || scope == "colgroup")
153 return true;
154 if (scope == "row" || scope == "rowgroup")
155 return false;
156 if (!isTableHeaderCell())
157 return false;
158
159 // We are in a situation after checking the scope attribute.
160 // It is an attempt to resolve the type of th element without support in the specification.
161 // Checking tableTag and tbodyTag allows to check the case of direct row placement in the table and lets stop the loop at the table level.
162 for (Node* parentNode = node(); parentNode; parentNode = parentNode->parentNode()) {
163 if (parentNode->hasTagName(theadTag))
164 return true;
165 if (parentNode->hasTagName(tfootTag))
166 return false;
167 if (parentNode->hasTagName(tableTag) || parentNode->hasTagName(tbodyTag)) {
168 auto rowRange = rowIndexRange();
169 if (!rowRange.first)
170 return true;
171 return false;
172 }
173 }
174 return false;
175}

177bool AccessibilityTableCell::isRowHeaderCell() const
178{
179 const AtomString& scope = getAttribute(scopeAttr);
180 if (scope == "row" || scope == "rowgroup")
181 return true;
182 if (scope == "col" || scope == "colgroup")
183 return false;
184 if (!isTableHeaderCell())
185 return false;
186
187 // We are in a situation after checking the scope attribute.
188 // It is an attempt to resolve the type of th element without support in the specification.
189 // Checking tableTag allows to check the case of direct row placement in the table and lets stop the loop at the table level.
190 for (Node* parentNode = node(); parentNode; parentNode = parentNode->parentNode()) {
191 if (parentNode->hasTagName(tfootTag) || parentNode->hasTagName(tbodyTag) || parentNode->hasTagName(tableTag)) {
192 auto colRange = columnIndexRange();
193 if (!colRange.first)
194 return true;
195 return false;
196 }
197 if (parentNode->hasTagName(theadTag))
198 return false;
199 }
200 return false;
201}

203bool AccessibilityTableCell::isTableCellInSameRowGroup(AXCoreObject* otherTableCell)
204{
205 Node* parentNode = node();
206 for ( ; parentNode; parentNode = parentNode->parentNode()) {
207 if (parentNode->hasTagName(theadTag) || parentNode->hasTagName(tbodyTag) || parentNode->hasTagName(tfootTag))
208 break;
209 }
210
211 Node* otherParentNode = otherTableCell->node();
212 for ( ; otherParentNode; otherParentNode = otherParentNode->parentNode()) {
213 if (otherParentNode->hasTagName(theadTag) || otherParentNode->hasTagName(tbodyTag) || otherParentNode->hasTagName(tfootTag))
214 break;
215 }
216
217 return otherParentNode == parentNode;
218}

220bool AccessibilityTableCell::isTableCellInSameColGroup(AXCoreObject* tableCell)
221{
222 auto colRange = columnIndexRange();
223 auto otherColRange = tableCell->columnIndexRange();
224
225 if (colRange.first <= (otherColRange.first + otherColRange.second))
226 return true;
227 return false;
228}

240AXCoreObject::AccessibilityChildrenVector AccessibilityTableCell::columnHeaders()
241{
242 AccessibilityChildrenVector headers;
243 AccessibilityTable* parent = parentTable();
244 if (!parent)
245 return headers;
246
247 // Choose columnHeaders as the place where the "headers" attribute is reported.
248 ariaElementsFromAttribute(headers, headersAttr);
249 // If the headers attribute returned valid values, then do not further search for column headers.
250 if (!headers.isEmpty())
251 return headers;
252
253 auto rowRange = rowIndexRange();
254 auto colRange = columnIndexRange();
255
256 for (unsigned row = 0; row < rowRange.first; row++) {
257 auto* tableCell = parent->cellForColumnAndRow(colRange.first, row);
258 if (!tableCell || tableCell == this || headers.contains(tableCell))
259 continue;
260
261 const AtomString& scope = tableCell->getAttribute(scopeAttr);
262 if (scope == "colgroup" && isTableCellInSameColGroup(tableCell))
263 headers.append(tableCell);
264 else if (tableCell->isColumnHeaderCell())
265 headers.append(tableCell);
266 }
267
268 return headers;
269}
270
271AXCoreObject::AccessibilityChildrenVector AccessibilityTableCell::rowHeaders()
272{
273 AccessibilityChildrenVector headers;
274 AccessibilityTable* parent = parentTable();
275 if (!parent)
276 return headers;
277
278 auto rowRange = rowIndexRange();
279 auto colRange = columnIndexRange();
280
281 for (unsigned column = 0; column < colRange.first; column++) {
282 auto* tableCell = parent->cellForColumnAndRow(column, rowRange.first);
283 if (!tableCell || tableCell == this || headers.contains(tableCell))
284 continue;
285
286 const AtomString& scope = tableCell->getAttribute(scopeAttr);
287 if (scope == "rowgroup" && isTableCellInSameRowGroup(tableCell))
288 headers.append(tableCell);
289 else if (tableCell->isRowHeaderCell())
290 headers.append(tableCell);
291 }
292
293 return headers;
294}

296AccessibilityTableRow* AccessibilityTableCell::ariaOwnedByParent() const
297{
298 AccessibilityChildrenVector ariaOwnedBy;
299 ariaOwnsReferencingElements(ariaOwnedBy);
300 if (ariaOwnedBy.size() == 1 && ariaOwnedBy[0]->isTableRow())
301 return downcast<AccessibilityTableRow>(ariaOwnedBy[0].get());
302
303 return nullptr;
304}

324std::pair<unsigned, unsigned> AccessibilityTableCell::rowIndexRange() const
325{
326 std::pair<unsigned, unsigned> rowRange { 0, 1 };
327 if (!is<RenderTableCell>(renderer()))
328 return rowRange;
329
330 RenderTableCell& renderCell = downcast<RenderTableCell>(*m_renderer);
331
332 // ARIA 1.1's aria-rowspan attribute is intended for cells and gridcells which are not contained
333 // in a native table. But if we have a valid author-provided value and do not have an explicit
334 // native host language value for the rowspan, expose the ARIA value.
335 rowRange.second = axRowSpan();
336 if (static_cast<int>(rowRange.second) == -1)
337 rowRange.second = renderCell.rowSpan();
338
339 if (AccessibilityTableRow* parentRow = this->parentRow())
340 rowRange.first = parentRow->rowIndex();
341
342 return rowRange;
343}
344
345std::pair<unsigned, unsigned> AccessibilityTableCell::columnIndexRange() const
346{
347 std::pair<unsigned, unsigned> columnRange { 0, 1 };
348 if (!is<RenderTableCell>(renderer()))
349 return columnRange;
350
351 const RenderTableCell& cell = downcast<RenderTableCell>(*m_renderer);
352 columnRange.first = cell.table()->colToEffCol(cell.col());
353
354 // ARIA 1.1's aria-colspan attribute is intended for cells and gridcells which are not contained
355 // in a native table. But if we have a valid author-provided value and do not have an explicit
356 // native host language value for the colspan, expose the ARIA value.
357 columnRange.second = axColumnSpan();
358 if (static_cast<int>(columnRange.second) != -1)
359 return columnRange;
360
361 columnRange.second = cell.table()->colToEffCol(cell.col() + cell.colSpan()) - columnRange.first;
362
363 return columnRange;
364}

366AccessibilityObject* AccessibilityTableCell::titleUIElement() const
367{
368 // Try to find if the first cell in this row is a <th>. If it is,
369 // then it can act as the title ui element. (This is only in the
370 // case when the table is not appearing as an AXTable.)
371 if (isTableCell() || !is<RenderTableCell>(renderer()))
372 return nullptr;
373
374 // Table cells that are th cannot have title ui elements, since by definition
375 // they are title ui elements
376 Node* node = m_renderer->node();
377 if (node && node->hasTagName(thTag))
378 return nullptr;
379
380 RenderTableCell& renderCell = downcast<RenderTableCell>(*m_renderer);
381
382 // If this cell is in the first column, there is no need to continue.
383 int col = renderCell.col();
384 if (!col)
385 return nullptr;
386
387 int row = renderCell.rowIndex();
388
389 RenderTableSection* section = renderCell.section();
390 if (!section)
391 return nullptr;
392
393 RenderTableCell* headerCell = section->primaryCellAt(row, 0);
394 if (!headerCell || headerCell == &renderCell)
395 return nullptr;
396
397 if (!headerCell->element() || !headerCell->element()->hasTagName(thTag))
398 return nullptr;
399
400 return axObjectCache()->getOrCreate(headerCell);
401}

403int AccessibilityTableCell::axColumnIndex() const
404{
405 if (int value = getIntegralAttribute(aria_colindexAttr); value >= 1)
406 return value;
407
408 // "ARIA 1.1: If the set of columns which is present in the DOM is contiguous, and if there are no cells which span more than one row
409 // or column in that set, then authors may place aria-colindex on each row, setting the value to the index of the first column of the set."
410 // Here, we let its parent row to set its index beforehand, so we don't have to go through the siblings to calculate the index.
411 AccessibilityTableRow* parentRow = this->parentRow();
412 if (parentRow && m_axColIndexFromRow != -1)
413 return m_axColIndexFromRow;
414
415 return -1;
416}
417
418int AccessibilityTableCell::axRowIndex() const
419{
420 // ARIA 1.1: Authors should place aria-rowindex on each row. Authors may also place
421 // aria-rowindex on all of the children or owned elements of each row.
422 if (int value = getIntegralAttribute(aria_rowindexAttr); value >= 1)
423 return value;
424
425 if (AccessibilityTableRow* parentRow = this->parentRow())
426 return parentRow->axRowIndex();
427
428 return -1;
429}

431int AccessibilityTableCell::axColumnSpan() const
432{
433 // According to the ARIA spec, "If aria-colpan is used on an element for which the host language
434 // provides an equivalent attribute, user agents must ignore the value of aria-colspan."
435 if (hasAttribute(colspanAttr))
436 return -1;
437
438 // ARIA 1.1: Authors must set the value of aria-colspan to an integer greater than or equal to 1.
439 if (int value = getIntegralAttribute(aria_colspanAttr); value >= 1)
440 return value;
441
442 return -1;
443}
444
445int AccessibilityTableCell::axRowSpan() const
446{
447 // According to the ARIA spec, "If aria-rowspan is used on an element for which the host language
448 // provides an equivalent attribute, user agents must ignore the value of aria-rowspan."
449 if (hasAttribute(rowspanAttr))
450 return -1;
451
452 // ARIA 1.1: Authors must set the value of aria-rowspan to an integer greater than or equal to 0.
453 // Setting the value to 0 indicates that the cell or gridcell is to span all the remaining rows in the row group.
454 if (getAttribute(aria_rowspanAttr) == "0")
455 return 0;
456 if (int value = getIntegralAttribute(aria_rowspanAttr); value >= 1)
457 return value;
458
459 return -1;
460}
