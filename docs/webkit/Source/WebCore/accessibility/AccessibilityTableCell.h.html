29#pragma once
30
31#include "AccessibilityRenderObject.h"
32
33namespace WebCore {
34
35class AccessibilityTable;
36class AccessibilityTableRow;
37
38class AccessibilityTableCell : public AccessibilityRenderObject {
39public:
40 static Ref<AccessibilityTableCell> create(RenderObject*);
41 virtual ~AccessibilityTableCell();
42
43 bool isTableCell() const final;
44 bool isTableHeaderCell() const;
45 bool isColumnHeaderCell() const override;
46 bool isRowHeaderCell() const override;
47
48 // Returns the start location and row span of the cell.
49 std::pair<unsigned, unsigned> rowIndexRange() const override;
50 // Returns the start location and column span of the cell.
51 std::pair<unsigned, unsigned> columnIndexRange() const override;
52
53 AccessibilityChildrenVector columnHeaders() override;
54 AccessibilityChildrenVector rowHeaders() override;
55
56 int axColumnIndex() const override;
57 int axRowIndex() const override;
58 int axColumnSpan() const;
59 int axRowSpan() const;
60 void setAXColIndexFromRow(int index) { m_axColIndexFromRow = index; }
61
62protected:
63 explicit AccessibilityTableCell(RenderObject*);
64
65 AccessibilityTableRow* parentRow() const;
66 virtual AccessibilityTable* parentTable() const;
67 AccessibilityRole determineAccessibilityRole() final;
68 AXCoreObject* parentObjectUnignored() const override;
69
70 int m_rowIndex;
71 int m_axColIndexFromRow;
72
73private:
74 // If a table cell is not exposed as a table cell, a TH element can serve as its title UI element.
75 AccessibilityObject* titleUIElement() const final;
76 bool exposesTitleUIElement() const final { return true; }
77 bool computeAccessibilityIsIgnored() const final;
78 String expandedTextValue() const final;
79 bool supportsExpandedTextValue() const final;
80 AccessibilityTableRow* ariaOwnedByParent() const;
81
82 bool isTableCellInSameRowGroup(AXCoreObject*);
83 bool isTableCellInSameColGroup(AXCoreObject*);
84};
85
86} // namespace WebCore
87
88SPECIALIZE_TYPE_TRAITS_ACCESSIBILITY(AccessibilityTableCell, isTableCell())
