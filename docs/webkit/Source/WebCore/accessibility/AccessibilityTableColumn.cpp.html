60LayoutRect AccessibilityTableColumn::elementRect() const
61{
62 // This used to be cached during the call to addChildren(), but calling elementRect()
63 // can invalidate elements, so its better to ask for this on demand.
64 LayoutRect columnRect;
65 AccessibilityChildrenVector childrenCopy = m_children;
66 for (const auto& cell : childrenCopy)
67 columnRect.unite(cell->elementRect());
68
69 return columnRect;
70}

72AXCoreObject* AccessibilityTableColumn::columnHeader()
73{
74 if (!m_parent || !is<AccessibilityTable>(*m_parent)
75 || !m_parent->isExposable())
76 return nullptr;
77
78 RenderObject* renderer = m_parent->renderer();
79 if (!renderer)
80 return nullptr;
81
82 auto& parentTable = downcast<AccessibilityTable>(*m_parent);
83
84 if (parentTable.isAriaTable()) {
85 for (const auto& cell : children()) {
86 if (cell->ariaRoleAttribute() == AccessibilityRole::ColumnHeader)
87 return cell.get();
88 }
89
90 return nullptr;
91 }
92
93 if (!is<RenderTable>(*renderer))
94 return nullptr;
95
96 RenderTable& table = downcast<RenderTable>(*renderer);
97
98 // try the <thead> section first. this doesn't require th tags
99 if (auto* headerObject = headerObjectForSection(table.header(), false))
100 return headerObject;
101
102 RenderTableSection* bodySection = table.firstBody();
103 while (bodySection && bodySection->isAnonymous())
104 bodySection = table.sectionBelow(bodySection, SkipEmptySections);
105
106 // now try for <th> tags in the first body. If the first body is
107 return headerObjectForSection(bodySection, true);
108}

110AccessibilityObject* AccessibilityTableColumn::headerObjectForSection(RenderTableSection* section, bool thTagRequired)
111{
112 if (!section)
113 return nullptr;
114
115 unsigned numCols = section->numColumns();
116 if (m_columnIndex >= numCols)
117 return nullptr;
118
119 if (!section->numRows())
120 return nullptr;
121
122 RenderTableCell* cell = nullptr;
123 // also account for cells that have a span
124 for (int testCol = m_columnIndex; testCol >= 0; --testCol) {
125
126 // Run down the rows in case initial rows are invalid (like when a <caption> is used).
127 unsigned rowCount = section->numRows();
128 for (unsigned testRow = 0; testRow < rowCount; testRow++) {
129 RenderTableCell* testCell = section->primaryCellAt(testRow, testCol);
130 // No cell at this index, keep checking more rows and columns.
131 if (!testCell)
132 continue;
133
134 // If we've reached a cell that doesn't even overlap our column it can't be the header.
135 if ((testCell->col() + (testCell->colSpan()-1)) < m_columnIndex)
136 break;
137
138 Node* testCellNode = testCell->element();
139 // If the RenderTableCell doesn't have an element because its anonymous,
140 // try to see if we can find the original cell element to check if it has a <th> tag.
141 if (!testCellNode && testCell->isAnonymous()) {
142 if (Element* parentElement = testCell->parent() ? testCell->parent()->element() : nullptr) {
143 if (parentElement->hasTagName(trTag) && testCol < static_cast<int>(parentElement->countChildNodes()))
144 testCellNode = parentElement->traverseToChildAt(testCol);
145 }
146 }
147
148 if (!testCellNode)
149 continue;
150
151 // If th is required, but we found an element that doesn't have a th tag, we can stop looking.
152 if (thTagRequired && !testCellNode->hasTagName(thTag))
153 break;
154
155 cell = testCell;
156 break;
157 }
158 }
159
160 if (!cell)
161 return nullptr;
162
163 auto* cellObject = axObjectCache()->getOrCreate(cell);
164 if (!cellObject || cellObject->accessibilityIsIgnored())
165 return nullptr;
166
167 return cellObject;
168}

170bool AccessibilityTableColumn::computeAccessibilityIsIgnored() const
171{
172 if (!m_parent)
173 return true;
174
175#if PLATFORM(IOS_FAMILY) || USE(ATK)
176 return true;
177#endif
178
179 return m_parent->accessibilityIsIgnored();
180}

182void AccessibilityTableColumn::addChildren()
183{
184 ASSERT(!m_haveChildren);
185
186 m_haveChildren = true;
187 if (!is<AccessibilityTable>(m_parent))
188 return;
189
190 auto& parentTable = downcast<AccessibilityTable>(*m_parent);
191 if (!parentTable.isExposable())
192 return;
193
194 int numRows = parentTable.rowCount();
195
196 for (int i = 0; i < numRows; ++i) {
197 auto* cell = parentTable.cellForColumnAndRow(m_columnIndex, i);
198 if (!cell)
199 continue;
200
201 // make sure the last one isn't the same as this one (rowspan cells)
202 if (m_children.size() > 0 && m_children.last() == cell)
203 continue;
204
205 m_children.append(cell);
206 }
207}
