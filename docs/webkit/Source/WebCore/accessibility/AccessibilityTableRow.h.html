29#pragma once
30
31#include "AccessibilityRenderObject.h"
32
33namespace WebCore {
34
35class AccessibilityTable;
36
37class AccessibilityTableRow : public AccessibilityRenderObject {
38public:
39 static Ref<AccessibilityTableRow> create(RenderObject*);
40 virtual ~AccessibilityTableRow();
41
42 // retrieves the "row" header (a th tag in the rightmost column)
43 virtual AXCoreObject* headerObject();
44 virtual AccessibilityTable* parentTable() const;
45
46 void setRowIndex(unsigned rowIndex) { m_rowIndex = rowIndex; }
47 unsigned rowIndex() const override { return m_rowIndex; }
48
49 // allows the table to add other children that may not originate
50 // in the row, but their col/row spans overlap into it
51 void appendChild(AccessibilityObject*);
52
53 void addChildren() override;
54
55 int axColumnIndex() const override;
56 int axRowIndex() const override;
57
58protected:
59 explicit AccessibilityTableRow(RenderObject*);
60
61 AccessibilityRole determineAccessibilityRole() final;
62
63private:
64 bool isTableRow() const final;
65 AccessibilityObject* observableObject() const final;
66 bool computeAccessibilityIsIgnored() const final;
67
68 unsigned m_rowIndex;
69};
70
71} // namespace WebCore
72
73SPECIALIZE_TYPE_TRAITS_ACCESSIBILITY(AccessibilityTableRow, isTableRow())
