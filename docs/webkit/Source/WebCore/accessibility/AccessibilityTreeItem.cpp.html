54}
55
56AccessibilityRole AccessibilityTreeItem::determineAccessibilityRole()
57{
58
59 // Walk the parent chain looking for a parent that is a tree. A treeitem is
60 // only considered valid if it is in a tree.
61 AccessibilityObject* parent = nullptr;
62 for (parent = parentObject(); parent && !parent->isTree(); parent = parent->parentObject()) { }
63 m_isTreeItemValid = parent;
64
65 return AccessibilityRenderObject::determineAccessibilityRole();
66}
