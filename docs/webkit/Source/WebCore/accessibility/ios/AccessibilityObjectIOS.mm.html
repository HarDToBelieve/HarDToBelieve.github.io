50FloatRect AccessibilityObject::convertRectToPlatformSpace(const FloatRect& rect, AccessibilityConversionSpace space) const
51{
52 auto* frameView = documentFrameView();
53 WAKView *documentView = frameView ? frameView->documentView() : nullptr;
54 if (documentView) {
55 CGPoint point = CGPointMake(rect.x(), rect.y());
56 CGSize size = CGSizeMake(rect.size().width(), rect.size().height());
57 CGRect cgRect = CGRectMake(point.x, point.y, size.width, size.height);
58
59 cgRect = [documentView convertRect:cgRect toView:nil];
60
61 // we need the web document view to give us our final screen coordinates
62 // because that can take account of the scroller
63 id webDocument = [wrapper() _accessibilityWebDocumentView];
64 if (webDocument)
65 cgRect = [webDocument convertRect:cgRect toView:nil];
66 return cgRect;
67 }
68
69 return convertFrameToSpace(rect, space);
70}

82// In iPhone only code for now. It's debateable whether this is desired on all platforms.
83int AccessibilityObject::accessibilityPasswordFieldLength()
84{
85 if (!isPasswordField())
86 return 0;
87 RenderObject* renderObject = downcast<AccessibilityRenderObject>(*this).renderer();
88
89 if (!renderObject || !is<HTMLInputElement>(renderObject->node()))
90 return false;
91
92 HTMLInputElement& inputElement = downcast<HTMLInputElement>(*renderObject->node());
93 return inputElement.value().length();
94}

106bool AccessibilityObject::hasTouchEventListener() const
107{
108 // Check whether this->node or any of its ancestors has any of the touch-related event listeners.
109 auto touchEventNames = eventNames().touchRelatedEventNames();
110 // If the node is in a shadowRoot, going up the node parent tree will stop and
111 // not check the entire chain of ancestors. Thus, use the parentInComposedTree instead.
112 for (auto* node = this->node(); node; node = node->parentInComposedTree()) {
113 for (auto eventName : touchEventNames) {
114 if (node->hasEventListeners(eventName))
115 return true;
116 }
117 }
118 return false;
119}

121bool AccessibilityObject::isInputTypePopupButton() const
122{

124 return roleValue() == AccessibilityRole::PopUpButton;
125 return false;
