91Ref<AXIsolatedTree> AXIsolatedTree::create(AXObjectCache* axObjectCache)
92{
93 AXTRACE("AXIsolatedTree::create");
94 ASSERT(isMainThread());
95 ASSERT(axObjectCache && axObjectCache->pageID());
96
97 auto tree = adoptRef(*new AXIsolatedTree(axObjectCache));
98
99 // Generate the nodes of the tree and set its root and focused objects.
100 // For this, we need the root and focused objects of the AXObject tree.
101 auto* axRoot = axObjectCache->getOrCreate(axObjectCache->document().view());
102 if (axRoot)
103 tree->generateSubtree(*axRoot, nullptr, true);
104 auto* axFocus = axObjectCache->focusedObjectForPage(axObjectCache->document().page());
105 if (axFocus)
106 tree->setFocusedNodeID(axFocus->objectID());
107
108 // Now that the tree is ready to take client requests, add it to the tree
109 // maps so that it can be found.
110 auto pageID = axObjectCache->pageID();
111 Locker locker { s_cacheLock };
112 ASSERT(!treePageCache().contains(*pageID));
113 treePageCache().set(*pageID, tree.copyRef());
114 treeIDCache().set(tree->treeID(), tree.copyRef());
115
116 return tree;
117}

141RefPtr<AXIsolatedObject> AXIsolatedTree::nodeForID(AXID axID) const
142{
143 // In isolated tree mode 2, only access m_readerThreadNodeMap on the AX thread.
144 if (m_usedOnAXThread && isMainThread())
145 return nullptr;
146

182void AXIsolatedTree::generateSubtree(AXCoreObject& axObject, AXCoreObject* axParent, bool attachWrapper)
183{
184 AXTRACE("AXIsolatedTree::generateSubtree");
185 ASSERT(isMainThread());
186
187 if (axObject.objectID() == InvalidAXID)
188 return;
189
190 auto object = createSubtree(axObject, axParent ? axParent->objectID() : InvalidAXID, attachWrapper);
191 Locker locker { m_changeLogLock };
192 if (!axParent)
193 setRootNode(object.ptr());
194 else if (axParent->objectID() != InvalidAXID) // Need to check for the objectID of axParent again because it may have been detached while traversing the tree.
195 updateChildrenIDs(axParent->objectID(), axParent->childrenIDs());
196}

198Ref<AXIsolatedObject> AXIsolatedTree::createSubtree(AXCoreObject& axObject, AXID parentID, bool attachWrapper)
199{
200 AXTRACE("AXIsolatedTree::createSubtree");
201 ASSERT(isMainThread());
202
203 auto object = AXIsolatedObject::create(axObject, this, parentID);
204 if (object->objectID() == InvalidAXID) {
205 // Either the axObject has an invalid ID or something else went terribly wrong. Don't bother doing anything else.
206 ASSERT_NOT_REACHED();
207 return object;
208 }
209
210 NodeChange nodeChange { object, nullptr };
211 if (attachWrapper)
212 object->attachPlatformWrapper(axObject.wrapper());
213 else {
214 // Set the wrapper in the NodeChange so that it is set on the AX thread.
215 nodeChange.wrapper = axObject.wrapper();
216 }
217
218 auto axChildren = axObject.children();
219 Vector<AXID> childrenIDs;
220 childrenIDs.reserveCapacity(axChildren.size());
221 for (const auto& axChild : axChildren) {
222 auto child = createSubtree(*axChild, axObject.objectID(), attachWrapper);
223 childrenIDs.uncheckedAppend(child->objectID());
224 }
225
226 {
227 Locker locker { m_changeLogLock };
228 m_pendingAppends.append(WTFMove(nodeChange));
229 updateChildrenIDs(object->objectID(), WTFMove(childrenIDs));
230 }
231
232 return object;
233}

235void AXIsolatedTree::updateNode(AXCoreObject& axObject)
236{
237 AXTRACE("AXIsolatedTree::updateNode");
238 AXLOG(&axObject);
239 ASSERT(isMainThread());
240
241 AXID axID = axObject.objectID();
242 auto* axParent = axObject.parentObject();
243 AXID parentID = axParent ? axParent->objectID() : InvalidAXID;
244
245 auto newObject = AXIsolatedObject::create(axObject, this, parentID);
246 newObject->m_childrenIDs = axObject.childrenIDs();
247
248 Locker locker { m_changeLogLock };
249 // Remove the old object and set the new one to be updated on the AX thread.
250 m_pendingNodeRemovals.append(axID);
251 m_pendingAppends.append({ newObject, axObject.wrapper() });
252}

254void AXIsolatedTree::updateNodeProperty(const AXCoreObject& axObject, AXPropertyName property)
255{
256 AXTRACE("AXIsolatedTree::updateNodeProperty");
257 ASSERT(isMainThread());
258
259 AXPropertyMap propertyMap;
260 switch (property) {
261 case AXPropertyName::IsChecked:
262 propertyMap.set(AXPropertyName::IsChecked, axObject.isChecked());
263 break;
264 case AXPropertyName::SortDirection:
265 propertyMap.set(AXPropertyName::SortDirection, static_cast<int>(axObject.sortDirection()));
266 break;
267 case AXPropertyName::IdentifierAttribute:
268 propertyMap.set(AXPropertyName::IdentifierAttribute, axObject.identifierAttribute().isolatedCopy());
269 break;
270 default:
271 return;
272 }
273
274 Locker locker { m_changeLogLock };
275 m_pendingPropertyChanges.append({ axObject.objectID(), propertyMap });
276}

278void AXIsolatedTree::updateSubtree(AXCoreObject& axObject)
279{
280 AXTRACE("AXIsolatedTree::updateSubtree");
281 AXLOG(&axObject);
282 ASSERT(isMainThread());
283
284 removeSubtree(axObject.objectID());
285 generateSubtree(axObject, axObject.parentObject(), false);

288void AXIsolatedTree::updateChildren(AXCoreObject& axObject)
289{
290 AXTRACE("AXIsolatedTree::updateChildren");
291 AXLOG("For AXObject:");
292 AXLOG(&axObject);
293 ASSERT(isMainThread());
294
295 if (m_nodeMap.isEmpty()) {
296 ASSERT_NOT_REACHED();
297 return;
298 }
299
300 if (!axObject.document() || !axObject.document()->hasLivingRenderTree())
301 return;
302
303 // updateChildren may be called as the result of a children changed
304 // notification for an axObject that has no associated isolated object.
305 // An example of this is when an empty element such as a <canvas> or <div>
306 // is added a new child. So find the closest ancestor of axObject that has
307 // an associated isolated object and update its children.
308 auto iterator = m_nodeMap.end();
309 auto* axAncestor = Accessibility::findAncestor(axObject, true, [&iterator, this] (const AXCoreObject& ancestor) {
310 auto it = m_nodeMap.find(ancestor.objectID());
311 if (it != m_nodeMap.end()) {
312 iterator = it;
313 return true;
314 }
315 return false;
316 });
317 if (!axAncestor || axAncestor->objectID() == InvalidAXID || iterator == m_nodeMap.end()) {
318 // This update triggered before the isolated tree has been repopulated.
319 // Return here since there is nothing to update.
320 return;
321 }
322
323 // iterator is pointing to the m_nodeMap entry corresponding to axAncestor->objectID().
324 ASSERT(iterator->key == axAncestor->objectID());
325 auto removals = iterator->value;
326
327 const auto& axChildren = axAncestor->children();
328 auto axChildrenIDs = axAncestor->childrenIDs();
329
330 bool updatedChild = false; // Set to true if at least one child's subtree is updated.
331 for (size_t i = 0; i < axChildren.size() && i < axChildrenIDs.size(); ++i) {
332 size_t index = removals.find(axChildrenIDs[i]);
333 if (index != notFound)
334 removals.remove(index);
335 else {
336 ASSERT(axChildren[i]->objectID() == axChildrenIDs[i]);
337 // This is a new child, add it to the tree.
338 AXLOG("Adding a new child for:");
339 AXLOG(axChildren[i]);
340 generateSubtree(*axChildren[i], axAncestor, true);
341 updatedChild = true;
342 }
343 }
344
345 // What is left in removals are the IDs that are no longer children of
346 // axObject. Thus, remove them from the tree.
347 for (const AXID& childID : removals)
348 removeSubtree(childID);
349
350 if (updatedChild || removals.size()) {
351 // Make the children IDs of the isolated object to be the same as the AXObject's.
352 Locker locker { m_changeLogLock };
353 updateChildrenIDs(axAncestor->objectID(), WTFMove(axChildrenIDs));
354 } else {
355 // Nothing was updated. As a last resort, update the subtree.
356 updateSubtree(*axAncestor);
357 }
358}

415void AXIsolatedTree::removeSubtree(AXID axID)
416{
417 AXTRACE("AXIsolatedTree::removeSubtree");
418 AXLOG(makeString("Removing subtree for axID ", axID));
419 ASSERT(isMainThread());
420
421 Vector<AXID> removals = { axID };
422 while (removals.size()) {
423 AXID axID = removals.takeLast();
424 if (axID == InvalidAXID)
425 continue;
426
427 auto it = m_nodeMap.find(axID);
428 if (it != m_nodeMap.end()) {
429 removals.appendVector(it->value);
430 m_nodeMap.remove(axID);
431 }
432 }
433
434 Locker locker { m_changeLogLock };
435 m_pendingSubtreeRemovals.append(axID);
436}

438void AXIsolatedTree::applyPendingChanges()
439{
440 AXTRACE("AXIsolatedTree::applyPendingChanges");
441
442 // In isolated tree mode 2, only apply pending changes on the AX thread.
443 if (m_usedOnAXThread && isMainThread())
444 return;
445
446 Locker locker { m_changeLogLock };
447
448 if (m_pendingFocusedNodeID != m_focusedNodeID) {
449 AXLOG(makeString("focusedNodeID ", m_focusedNodeID, " pendingFocusedNodeID ", m_pendingFocusedNodeID));
450
451 if (m_focusedNodeID != InvalidAXID) {
452 // Set the old focused object's IsFocused property to false.
453 AXPropertyMap propertyMap;
454 propertyMap.set(AXPropertyName::IsFocused, false);
455 m_pendingPropertyChanges.append({ m_focusedNodeID, propertyMap });
456 }
457 m_focusedNodeID = m_pendingFocusedNodeID;
458 }
459
460 while (m_pendingNodeRemovals.size()) {
461 auto axID = m_pendingNodeRemovals.takeLast();
462 AXLOG(makeString("removing axID ", axID));
463 if (auto object = nodeForID(axID)) {
464 object->detach(AccessibilityDetachmentType::ElementDestroyed);
465 m_readerThreadNodeMap.remove(axID);
466 }
467 }
468
469 while (m_pendingSubtreeRemovals.size()) {
470 auto axID = m_pendingSubtreeRemovals.takeLast();
471 AXLOG(makeString("removing subtree axID ", axID));
472 if (auto object = nodeForID(axID)) {
473 object->detach(AccessibilityDetachmentType::ElementDestroyed);
474 m_pendingSubtreeRemovals.appendVector(object->m_childrenIDs);
475 m_readerThreadNodeMap.remove(axID);
476 }
477 }
478
479 for (const auto& item : m_pendingAppends) {
480 AXID axID = item.isolatedObject->objectID();
481 AXLOG(makeString("appending axID ", axID));
482 if (axID == InvalidAXID)
483 continue;
484
485 auto& wrapper = item.wrapper ? item.wrapper : item.isolatedObject->wrapper();
486 if (!wrapper)
487 continue;
488
489 if (auto object = m_readerThreadNodeMap.get(axID)) {
490 if (object != &item.isolatedObject.get()
491 && object->wrapper() == wrapper.get()) {
492 // The new IsolatedObject is a replacement for an existing object
493 // as the result of an update. Thus detach the wrapper from the
494 // existing object and attach it to the new one.
495 object->detachWrapper(AccessibilityDetachmentType::ElementChanged);
496 item.isolatedObject->attachPlatformWrapper(wrapper.get());
497 }
498 m_readerThreadNodeMap.remove(axID);
499 }
500
501 if (!item.isolatedObject->wrapper()) {
502 // The new object hasn't been attached a wrapper yet, so attach it.
503 item.isolatedObject->attachPlatformWrapper(wrapper.get());
504 }
505
506 auto addResult = m_readerThreadNodeMap.add(axID, item.isolatedObject.get());
507 // The newly added object must have a wrapper.
508 ASSERT_UNUSED(addResult, addResult.iterator->value->wrapper());
509 // The reference count of the just added IsolatedObject must be 2
510 // because it is referenced by m_readerThreadNodeMap and m_pendingAppends.
511 // When m_pendingAppends is cleared, the object will be held only by m_readerThreadNodeMap. The exception is the root node whose reference count is 3.
512 ASSERT_UNUSED(addResult, addResult.iterator->value->refCount() == 2
513 || (addResult.iterator->value.ptr() == m_rootNode.get() && m_rootNode->refCount() == 3));
514 }
515 m_pendingAppends.clear();
516
517 for (auto& update : m_pendingChildrenUpdates) {
518 AXLOG(makeString("updating children for axID ", update.first));
519 if (auto object = nodeForID(update.first))
520 object->m_childrenIDs = WTFMove(update.second);
521 }
522 m_pendingChildrenUpdates.clear();
523
524 for (auto& change : m_pendingPropertyChanges) {
525 if (auto object = nodeForID(change.axID)) {
526 for (auto& property : change.properties)
527 object->setProperty(property.key, WTFMove(property.value));
528 }
529 }
530 m_pendingPropertyChanges.clear();
531}
