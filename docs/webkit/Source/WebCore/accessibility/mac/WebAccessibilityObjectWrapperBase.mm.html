243static NSArray *convertMathPairsToNSArray(const AccessibilityObject::AccessibilityMathMultiscriptPairs& pairs, NSString *subscriptKey, NSString *superscriptKey)
244{
245 return createNSArray(pairs, [&] (auto& pair) {
246 WebAccessibilityObjectWrapper *wrappers[2];
247 NSString *keys[2];
248 NSUInteger count = 0;
249 if (pair.first && pair.first->wrapper() && !pair.first->accessibilityIsIgnored()) {
250 wrappers[0] = pair.first->wrapper();
251 keys[0] = subscriptKey;
252 count = 1;
253 }
254 if (pair.second && pair.second->wrapper() && !pair.second->accessibilityIsIgnored()) {
255 wrappers[count] = pair.second->wrapper();
256 keys[count] = superscriptKey;
257 count += 1;
258 }
259 return adoptNS([[NSDictionary alloc] initWithObjects:wrappers forKeys:keys count:count]);
260 }).autorelease();
261}

263NSArray *convertToNSArray(const WebCore::AXCoreObject::AccessibilityChildrenVector& children)
264{
265 return createNSArray(children, [] (const auto& child) -> id {
266 auto wrapper = child->wrapper();
267
268 // We want to return the attachment view instead of the object representing the attachment,
269 // otherwise, we get palindrome errors in the AX hierarchy.
270 if (child->isAttachment() && wrapper.attachmentView)
271 return wrapper.attachmentView;
272
273 return wrapper;
274 }).autorelease();
275}

377struct PathConversionInfo {
378 WebAccessibilityObjectWrapperBase *wrapper;
379 CGMutablePathRef path;
380};
381
382static void convertPathToScreenSpaceFunction(PathConversionInfo& conversion, const PathElement& element)
383{
384 WebAccessibilityObjectWrapperBase *wrapper = conversion.wrapper;
385 CGMutablePathRef newPath = conversion.path;
386 FloatRect rect;
387 switch (element.type) {
388 case PathElement::Type::MoveToPoint:
389 {
390 rect = FloatRect(element.points[0], FloatSize());
391 CGPoint newPoint = [wrapper convertRectToSpace:rect space:AccessibilityConversionSpace::Screen].origin;
392 CGPathMoveToPoint(newPath, nil, newPoint.x, newPoint.y);
393 break;
394 }
395 case PathElement::Type::AddLineToPoint:
396 {
397 rect = FloatRect(element.points[0], FloatSize());
398 CGPoint newPoint = [wrapper convertRectToSpace:rect space:AccessibilityConversionSpace::Screen].origin;
399 CGPathAddLineToPoint(newPath, nil, newPoint.x, newPoint.y);
400 break;
401 }
402 case PathElement::Type::AddQuadCurveToPoint:
403 {
404 rect = FloatRect(element.points[0], FloatSize());
405 CGPoint newPoint1 = [wrapper convertRectToSpace:rect space:AccessibilityConversionSpace::Screen].origin;
406
407 rect = FloatRect(element.points[1], FloatSize());
408 CGPoint newPoint2 = [wrapper convertRectToSpace:rect space:AccessibilityConversionSpace::Screen].origin;
409 CGPathAddQuadCurveToPoint(newPath, nil, newPoint1.x, newPoint1.y, newPoint2.x, newPoint2.y);
410 break;
411 }
412 case PathElement::Type::AddCurveToPoint:
413 {
414 rect = FloatRect(element.points[0], FloatSize());
415 CGPoint newPoint1 = [wrapper convertRectToSpace:rect space:AccessibilityConversionSpace::Screen].origin;
416
417 rect = FloatRect(element.points[1], FloatSize());
418 CGPoint newPoint2 = [wrapper convertRectToSpace:rect space:AccessibilityConversionSpace::Screen].origin;
419
420 rect = FloatRect(element.points[2], FloatSize());
421 CGPoint newPoint3 = [wrapper convertRectToSpace:rect space:AccessibilityConversionSpace::Screen].origin;
422 CGPathAddCurveToPoint(newPath, nil, newPoint1.x, newPoint1.y, newPoint2.x, newPoint2.y, newPoint3.x, newPoint3.y);
423 break;
424 }
425 case PathElement::Type::CloseSubpath:
426 {
427 CGPathCloseSubpath(newPath);
428 break;
429 }
430 }
431}
432
433- (CGPathRef)convertPathToScreenSpace:(const Path&)path
434{
435 auto convertedPath = adoptCF(CGPathCreateMutable());
436 PathConversionInfo conversion = { self, convertedPath.get() };
437 path.apply([&conversion](const PathElement& pathElement) {
438 convertPathToScreenSpaceFunction(conversion, pathElement);
439 });
440 return convertedPath.autorelease();
441}

491{
492 NSMutableDictionary<NSString *, id> *results = [NSMutableDictionary dictionary];
493 auto editingStyles = self.axBackingObject->resolvedEditingStyles();

496 id result = WTF::switchOn(value,
497 [] (String& typedValue) -> id { return (NSString *)typedValue; },

500 [] (auto&) { return nil; }
501 );
502 results[(NSString *)key] = result;
503 }
504 return results;

523static bool isValueTypeSupported(id value)
524{
525 return [value isKindOfClass:[NSString class]] || [value isKindOfClass:[NSNumber class]] || [value isKindOfClass:[WebAccessibilityObjectWrapperBase class]];
526}

528static NSArray *arrayRemovingNonSupportedTypes(NSArray *array)
529{
530 ASSERT([array isKindOfClass:[NSArray class]]);
531 auto mutableArray = adoptNS([array mutableCopy]);
532 for (NSUInteger i = 0; i < [mutableArray count];) {
533 id value = [mutableArray objectAtIndex:i];
534 if ([value isKindOfClass:[NSDictionary class]])
535 [mutableArray replaceObjectAtIndex:i withObject:dictionaryRemovingNonSupportedTypes(value)];
536 else if ([value isKindOfClass:[NSArray class]])
537 [mutableArray replaceObjectAtIndex:i withObject:arrayRemovingNonSupportedTypes(value)];
538 else if (!isValueTypeSupported(value)) {
539 [mutableArray removeObjectAtIndex:i];
540 continue;
541 }
542 i++;
543 }
544 return mutableArray.autorelease();
545}
546
547static NSDictionary *dictionaryRemovingNonSupportedTypes(NSDictionary *dictionary)
548{
549 if (!dictionary)
550 return nil;
551 ASSERT([dictionary isKindOfClass:[NSDictionary class]]);
552 auto mutableDictionary = adoptNS([dictionary mutableCopy]);
553 for (NSString *key in dictionary) {
554 id value = [dictionary objectForKey:key];
555 if ([value isKindOfClass:[NSDictionary class]])
556 [mutableDictionary setObject:dictionaryRemovingNonSupportedTypes(value) forKey:key];
557 else if ([value isKindOfClass:[NSArray class]])
558 [mutableDictionary setObject:arrayRemovingNonSupportedTypes(value) forKey:key];
559 else if (!isValueTypeSupported(value))
560 [mutableDictionary removeObjectForKey:key];
561 }
562 return mutableDictionary.autorelease();
563}

565- (void)accessibilityPostedNotification:(NSString *)notificationName userInfo:(NSDictionary *)userInfo
566{
567 if (accessibilityShouldRepostNotifications) {
568 ASSERT(notificationName);
569 userInfo = dictionaryRemovingNonSupportedTypes(userInfo);
570 NSDictionary *info = [NSDictionary dictionaryWithObjectsAndKeys:notificationName, @"notificationName", userInfo, @"userInfo", nil];
571 [[NSNotificationCenter defaultCenter] postNotificationName:@"AXDRTNotification" object:self userInfo:info];
572 }
573}

651static AccessibilitySearchKey accessibilitySearchKeyForString(const String& value)
652{
653 if (value.isEmpty())
654 return AccessibilitySearchKey::AnyType;

668AccessibilitySearchCriteria accessibilitySearchCriteriaForSearchPredicateParameterizedAttribute(const NSDictionary *parameterizedAttribute)
669{
670 NSString *directionParameter = [parameterizedAttribute objectForKey:@"AXDirection"];
671 NSNumber *immediateDescendantsOnlyParameter = [parameterizedAttribute objectForKey:NSAccessibilityImmediateDescendantsOnly];
672 NSNumber *resultsLimitParameter = [parameterizedAttribute objectForKey:@"AXResultsLimit"];
673 NSString *searchTextParameter = [parameterizedAttribute objectForKey:@"AXSearchText"];
674 WebAccessibilityObjectWrapperBase *startElementParameter = [parameterizedAttribute objectForKey:@"AXStartElement"];
675 NSNumber *visibleOnlyParameter = [parameterizedAttribute objectForKey:@"AXVisibleOnly"];
676 id searchKeyParameter = [parameterizedAttribute objectForKey:@"AXSearchKey"];
677
678 AccessibilitySearchDirection direction = AccessibilitySearchDirection::Next;
679 if ([directionParameter isKindOfClass:[NSString class]])
680 direction = [directionParameter isEqualToString:@"AXDirectionNext"] ? AccessibilitySearchDirection::Next : AccessibilitySearchDirection::Previous;
681
682 bool immediateDescendantsOnly = false;
683 if ([immediateDescendantsOnlyParameter isKindOfClass:[NSNumber class]])
684 immediateDescendantsOnly = [immediateDescendantsOnlyParameter boolValue];
685
686 unsigned resultsLimit = 0;
687 if ([resultsLimitParameter isKindOfClass:[NSNumber class]])
688 resultsLimit = [resultsLimitParameter unsignedIntValue];
689
690 String searchText;
691 if ([searchTextParameter isKindOfClass:[NSString class]])
692 searchText = searchTextParameter;
693
694 AXCoreObject* startElement = nullptr;
695 if ([startElementParameter isKindOfClass:[WebAccessibilityObjectWrapperBase class]])
696 startElement = [startElementParameter axBackingObject];
697
698 bool visibleOnly = false;
