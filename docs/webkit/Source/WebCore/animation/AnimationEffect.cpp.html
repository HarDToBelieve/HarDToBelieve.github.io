70BasicEffectTiming AnimationEffect::getBasicTiming(std::optional<Seconds> startTime) const
71{
72 // The Web Animations spec introduces a number of animation effect time-related definitions that refer
73 // to each other a fair bit, so rather than implementing them as individual methods, it's more efficient
74 // to return them all as a single BasicEffectTiming.
75
76 auto localTime = [this, startTime]() -> std::optional<Seconds> {
77 // 4.5.4. Local time
78 // https://drafts.csswg.org/web-animations-1/#local-time-section
79
80 // The local time of an animation effect at a given moment is based on the first matching condition from the following:
81 // If the animation effect is associated with an animation, the local time is the current time of the animation.
82 // Otherwise, the local time is unresolved.
83 if (m_animation)
84 return m_animation->currentTime(startTime);
85 return std::nullopt;
86 }();
87
88 auto phase = [this, localTime]() -> AnimationEffectPhase {
89 // 3.5.5. Animation effect phases and states
90 // https://drafts.csswg.org/web-animations-1/#animation-effect-phases-and-states
91
92 bool animationIsBackwards = m_animation && m_animation->playbackRate() < 0;
93 auto beforeActiveBoundaryTime = std::max(std::min(m_delay, m_endTime), 0_s);
94 auto activeAfterBoundaryTime = std::max(std::min(m_delay + m_activeDuration, m_endTime), 0_s);
95
96 // (This should be the last statement, but it's more efficient to cache the local time and return right away if it's not resolved.)
97 // Furthermore, it is often convenient to refer to the case when an animation effect is in none of the above phases
98 // as being in the idle phase.
99 if (!localTime)
100 return AnimationEffectPhase::Idle;
101
102 // An animation effect is in the before phase if the animation effect’s local time is not unresolved and
103 // either of the following conditions are met:
104 // 1. the local time is less than the before-active boundary time, or
105 // 2. the animation direction is ‘backwards’ and the local time is equal to the before-active boundary time.
106 if ((*localTime + timeEpsilon) < beforeActiveBoundaryTime || (animationIsBackwards && std::abs(localTime->microseconds() - beforeActiveBoundaryTime.microseconds()) < timeEpsilon.microseconds()))
107 return AnimationEffectPhase::Before;
108
109 // An animation effect is in the after phase if the animation effect’s local time is not unresolved and
110 // either of the following conditions are met:
111 // 1. the local time is greater than the active-after boundary time, or
112 // 2. the animation direction is ‘forwards’ and the local time is equal to the active-after boundary time.
113 if ((*localTime - timeEpsilon) > activeAfterBoundaryTime || (!animationIsBackwards && std::abs(localTime->microseconds() - activeAfterBoundaryTime.microseconds()) < timeEpsilon.microseconds()))
114 return AnimationEffectPhase::After;
115
116 // An animation effect is in the active phase if the animation effect’s local time is not unresolved and it is not
117 // in either the before phase nor the after phase.
118 // (No need to check, we've already established that local time was resolved).
119 return AnimationEffectPhase::Active;
120 }();
121
122 auto activeTime = [this, localTime, phase]() -> std::optional<Seconds> {
123 // 3.8.3.1. Calculating the active time
124 // https://drafts.csswg.org/web-animations-1/#calculating-the-active-time
125
126 // The active time is based on the local time and start delay. However, it is only defined
127 // when the animation effect should produce an output and hence depends on its fill mode
128 // and phase as follows,
129
130 // If the animation effect is in the before phase, the result depends on the first matching
131 // condition from the following,
132 if (phase == AnimationEffectPhase::Before) {
133 // If the fill mode is backwards or both, return the result of evaluating
134 // max(local time - start delay, 0).
135 if (m_fill == FillMode::Backwards || m_fill == FillMode::Both)
136 return std::max(*localTime - m_delay, 0_s);
137 // Otherwise, return an unresolved time value.
138 return std::nullopt;
139 }
140
141 // If the animation effect is in the active phase, return the result of evaluating local time - start delay.
142 if (phase == AnimationEffectPhase::Active)
143 return *localTime - m_delay;
144
145 // If the animation effect is in the after phase, the result depends on the first matching
146 // condition from the following,
147 if (phase == AnimationEffectPhase::After) {
148 // If the fill mode is forwards or both, return the result of evaluating
149 // max(min(local time - start delay, active duration), 0).
150 if (m_fill == FillMode::Forwards || m_fill == FillMode::Both)
151 return std::max(std::min(*localTime - m_delay, m_activeDuration), 0_s);
152 // Otherwise, return an unresolved time value.
153 return std::nullopt;
154 }
155
156 // Otherwise (the local time is unresolved), return an unresolved time value.
157 return std::nullopt;
158 }();
159
160 return { localTime, activeTime, m_endTime, m_activeDuration, phase };
161}

170ComputedEffectTiming AnimationEffect::getComputedTiming(std::optional<Seconds> startTime) const
171{
172 // The Web Animations spec introduces a number of animation effect time-related definitions that refer
173 // to each other a fair bit, so rather than implementing them as individual methods, it's more efficient
174 // to return them all as a single ComputedEffectTiming.
175
176 auto basicEffectTiming = getBasicTiming(startTime);
177 auto activeTime = basicEffectTiming.activeTime;
178 auto phase = basicEffectTiming.phase;
179
180 auto overallProgress = [this, phase, activeTime]() -> std::optional<double> {
181 // 3.8.3.2. Calculating the overall progress
182 // https://drafts.csswg.org/web-animations-1/#calculating-the-overall-progress
183
184 // The overall progress describes the number of iterations that have completed (including partial iterations) and is defined as follows:
185
186 // 1. If the active time is unresolved, return unresolved.
187 if (!activeTime)
188 return std::nullopt;
189
190 // 2. Calculate an initial value for overall progress based on the first matching condition from below,
191 double overallProgress;
192
193 if (!m_iterationDuration) {
194 // If the iteration duration is zero, if the animation effect is in the before phase, let overall progress be zero,
195 // otherwise, let it be equal to the iteration count.
196 overallProgress = phase == AnimationEffectPhase::Before ? 0 : m_iterations;
197 } else {
198 // Otherwise, let overall progress be the result of calculating active time / iteration duration.
199 overallProgress = secondsToWebAnimationsAPITime(*activeTime) / secondsToWebAnimationsAPITime(m_iterationDuration);
200 }
201
202 // 3. Return the result of calculating overall progress + iteration start.
203 overallProgress += m_iterationStart;
204 return std::abs(overallProgress);
205 }();
206
207 auto simpleIterationProgress = [this, overallProgress, phase, activeTime]() -> std::optional<double> {
208 // 3.8.3.3. Calculating the simple iteration progress
209 // https://drafts.csswg.org/web-animations-1/#calculating-the-simple-iteration-progress
210
211 // The simple iteration progress is a fraction of the progress through the current iteration that
212 // ignores transformations to the time introduced by the playback direction or timing functions
213 // applied to the effect, and is calculated as follows:
214
215 // 1. If the overall progress is unresolved, return unresolved.
216 if (!overallProgress)
217 return std::nullopt;
218
219 // 2. If overall progress is infinity, let the simple iteration progress be iteration start % 1.0,
220 // otherwise, let the simple iteration progress be overall progress % 1.0.
221 double simpleIterationProgress = std::isinf(*overallProgress) ? fmod(m_iterationStart, 1) : fmod(*overallProgress, 1);
222
223 // 3. If all of the following conditions are true,
224 //
225 // the simple iteration progress calculated above is zero, and
226 // the animation effect is in the active phase or the after phase, and
227 // the active time is equal to the active duration, and
228 // the iteration count is not equal to zero.
229 // let the simple iteration progress be 1.0.
230 if (!simpleIterationProgress && (phase == AnimationEffectPhase::Active || phase == AnimationEffectPhase::After) && std::abs(activeTime->microseconds() - m_activeDuration.microseconds()) < timeEpsilon.microseconds() && m_iterations)
231 return 1;
232
233 return simpleIterationProgress;
234 }();
235
236 auto currentIteration = [this, activeTime, phase, simpleIterationProgress, overallProgress]() -> std::optional<double> {
237 // 3.8.4. Calculating the current iteration
238 // https://drafts.csswg.org/web-animations-1/#calculating-the-current-iteration
239
240 // The current iteration can be calculated using the following steps:
241
242 // 1. If the active time is unresolved, return unresolved.
243 if (!activeTime)
244 return std::nullopt;
245
246 // 2. If the animation effect is in the after phase and the iteration count is infinity, return infinity.
247 if (phase == AnimationEffectPhase::After && std::isinf(m_iterations))
248 return std::numeric_limits<double>::infinity();
249
250 // 3. If the simple iteration progress is 1.0, return floor(overall progress) - 1.
251 if (*simpleIterationProgress == 1)
252 return floor(*overallProgress) - 1;
253
254 // 4. Otherwise, return floor(overall progress).
255 return floor(*overallProgress);
256 }();
257
258 auto currentDirection = [this, currentIteration]() -> AnimationEffect::ComputedDirection {
259 // 3.9.1. Calculating the directed progress
260 // https://drafts.csswg.org/web-animations-1/#calculating-the-directed-progress
261
262 // If playback direction is normal, let the current direction be forwards.
263 if (m_direction == PlaybackDirection::Normal)
264 return AnimationEffect::ComputedDirection::Forwards;
265
266 // If playback direction is reverse, let the current direction be reverse.
267 if (m_direction == PlaybackDirection::Reverse)
268 return AnimationEffect::ComputedDirection::Reverse;
269
270 if (!currentIteration)
271 return AnimationEffect::ComputedDirection::Forwards;
272
273 // Otherwise, let d be the current iteration.
274 auto d = *currentIteration;
275 // If playback direction is alternate-reverse increment d by 1.
276 if (m_direction == PlaybackDirection::AlternateReverse)
277 d++;
278 // If d % 2 == 0, let the current direction be forwards, otherwise let the current direction be reverse.
279 // If d is infinity, let the current direction be forwards.
280 if (std::isinf(d) || !fmod(d, 2))
281 return AnimationEffect::ComputedDirection::Forwards;
282 return AnimationEffect::ComputedDirection::Reverse;
283 }();
284
285 auto directedProgress = [simpleIterationProgress, currentDirection]() -> std::optional<double> {
286 // 3.9.1. Calculating the directed progress
287 // https://drafts.csswg.org/web-animations-1/#calculating-the-directed-progress
288
289 // The directed progress is calculated from the simple iteration progress using the following steps:
290
291 // 1. If the simple iteration progress is unresolved, return unresolved.
292 if (!simpleIterationProgress)
293 return std::nullopt;
294
295 // 2. Calculate the current direction (we implement this as a separate method).
296
297 // 3. If the current direction is forwards then return the simple iteration progress.
298 if (currentDirection == AnimationEffect::ComputedDirection::Forwards)
299 return *simpleIterationProgress;
300
301 // Otherwise, return 1.0 - simple iteration progress.
302 return 1 - *simpleIterationProgress;
303 }();
304
305 auto transformedProgress = [this, directedProgress, currentDirection, phase]() -> std::optional<double> {
306 // 3.10.1. Calculating the transformed progress
307 // https://drafts.csswg.org/web-animations-1/#calculating-the-transformed-progress
308
309 // The transformed progress is calculated from the directed progress using the following steps:
310 //
311 // 1. If the directed progress is unresolved, return unresolved.
312 if (!directedProgress)
313 return std::nullopt;
314
315 if (auto iterationDuration = m_iterationDuration.seconds()) {
316 bool before = false;
317 // 2. Calculate the value of the before flag as follows:
318 if (is<StepsTimingFunction>(m_timingFunction)) {
319 // 1. Determine the current direction using the procedure defined in §3.9.1 Calculating the directed progress.
320 // 2. If the current direction is forwards, let going forwards be true, otherwise it is false.
321 bool goingForwards = currentDirection == AnimationEffect::ComputedDirection::Forwards;
322 // 3. The before flag is set if the animation effect is in the before phase and going forwards is true;
323 // or if the animation effect is in the after phase and going forwards is false.
324 before = (phase == AnimationEffectPhase::Before && goingForwards) || (phase == AnimationEffectPhase::After && !goingForwards);
325 }
326
327 // 3. Return the result of evaluating the animation effect’s timing function passing directed progress as the
328 // input progress value and before flag as the before flag.
329 return m_timingFunction->transformTime(*directedProgress, iterationDuration, before);
330 }
331
332 return *directedProgress;
333 }();
334
335 ComputedEffectTiming computedTiming;
336 computedTiming.delay = secondsToWebAnimationsAPITime(m_delay);
337 computedTiming.endDelay = secondsToWebAnimationsAPITime(m_endDelay);
338 computedTiming.fill = m_fill == FillMode::Auto ? FillMode::None : m_fill;
339 computedTiming.iterationStart = m_iterationStart;
340 computedTiming.iterations = m_iterations;
341 computedTiming.duration = secondsToWebAnimationsAPITime(m_iterationDuration);
342 computedTiming.direction = m_direction;
343 computedTiming.easing = m_timingFunction->cssText();
344 computedTiming.endTime = secondsToWebAnimationsAPITime(m_endTime);
345 computedTiming.activeDuration = secondsToWebAnimationsAPITime(m_activeDuration);
346 if (basicEffectTiming.localTime)
347 computedTiming.localTime = secondsToWebAnimationsAPITime(*basicEffectTiming.localTime);
348 computedTiming.simpleIterationProgress = simpleIterationProgress;
349 computedTiming.progress = transformedProgress;
350 computedTiming.currentIteration = currentIteration;
351 computedTiming.phase = phase;
352 return computedTiming;
353}

363ExceptionOr<void> AnimationEffect::updateTiming(std::optional<OptionalEffectTiming> timing)
364{
365 // 6.5.4. Updating the timing of an AnimationEffect
366 // https://drafts.csswg.org/web-animations/#updating-animationeffect-timing
367
368 // To update the timing properties of an animation effect, effect, from an EffectTiming or OptionalEffectTiming object, input, perform the following steps:
369 if (!timing)
370 return { };
371
372 // 1. If the iterationStart member of input is present and less than zero, throw a TypeError and abort this procedure.
373 if (timing->iterationStart) {
374 if (timing->iterationStart.value() < 0)
375 return Exception { TypeError };
376 }
377
378 // 2. If the iterations member of input is present, and less than zero or is the value NaN, throw a TypeError and abort this procedure.
379 if (timing->iterations) {
380 if (timing->iterations.value() < 0 || std::isnan(timing->iterations.value()))
381 return Exception { TypeError };
382 }
383
384 // 3. If the duration member of input is present, and less than zero or is the value NaN, throw a TypeError and abort this procedure.
385 // FIXME: should it not throw an exception on a string other than "auto"?
386 if (timing->duration) {
387 if (WTF::holds_alternative<double>(timing->duration.value())) {
388 auto durationAsDouble = WTF::get<double>(timing->duration.value());
389 if (durationAsDouble < 0 || std::isnan(durationAsDouble))
390 return Exception { TypeError };
391 } else {
392 if (WTF::get<String>(timing->duration.value()) != "auto")
393 return Exception { TypeError };
394 }
395 }
396
397 // 4. If the easing member of input is present but cannot be parsed using the <timing-function> production [CSS-EASING-1], throw a TypeError and abort this procedure.
398 if (!timing->easing.isNull()) {
399 auto timingFunctionResult = TimingFunction::createFromCSSText(timing->easing);
400 if (timingFunctionResult.hasException())
401 return timingFunctionResult.releaseException();
402 m_timingFunction = timingFunctionResult.returnValue();
403 }
404
405 // 5. Assign each member present in input to the corresponding timing property of effect as follows:
406 //
407 // delay → start delay
408 // endDelay → end delay
409 // fill → fill mode
410 // iterationStart → iteration start
411 // iterations → iteration count
412 // duration → iteration duration
413 // direction → playback direction
414 // easing → timing function
415
416 if (timing->delay)
417 m_delay = Seconds::fromMilliseconds(timing->delay.value());
418
419 if (timing->endDelay)
420 m_endDelay = Seconds::fromMilliseconds(timing->endDelay.value());
421
422 if (timing->fill)
423 m_fill = timing->fill.value();
424
425 if (timing->iterationStart)
426 m_iterationStart = timing->iterationStart.value();
427
428 if (timing->iterations)
429 m_iterations = timing->iterations.value();
430
431 if (timing->duration)
432 m_iterationDuration = WTF::holds_alternative<double>(timing->duration.value()) ? Seconds::fromMilliseconds(WTF::get<double>(timing->duration.value())) : 0_s;
433
434 if (timing->direction)
435 m_direction = timing->direction.value();
436
437 updateStaticTimingProperties();
438
439 if (m_animation)
440 m_animation->effectTimingDidChange();
441
442 return { };
443}

445void AnimationEffect::updateStaticTimingProperties()
446{
447 // 3.8.2. Calculating the active duration
448 // https://drafts.csswg.org/web-animations-1/#calculating-the-active-duration
449
450 // The active duration is calculated as follows:
451 // active duration = iteration duration × iteration count
452 // If either the iteration duration or iteration count are zero, the active duration is zero.
453 if (!m_iterationDuration || !m_iterations)
454 m_activeDuration = 0_s;
455 else
456 m_activeDuration = m_iterationDuration * m_iterations;
457
458 // 3.5.3 The active interval
459 // https://drafts.csswg.org/web-animations-1/#end-time
460
461 // The end time of an animation effect is the result of evaluating max(start delay + active duration + end delay, 0).
462 m_endTime = m_delay + m_activeDuration + m_endDelay;
463 if (m_endTime < 0_s)
464 m_endTime = 0_s;
465}

544std::optional<double> AnimationEffect::progressUntilNextStep(double iterationProgress) const
545{
546 if (!is<StepsTimingFunction>(m_timingFunction))
547 return std::nullopt;
548
549 auto numberOfSteps = downcast<StepsTimingFunction>(*m_timingFunction).numberOfSteps();
550 auto nextStepProgress = ceil(iterationProgress * numberOfSteps) / numberOfSteps;
551 return nextStepProgress - iterationProgress;
552}
