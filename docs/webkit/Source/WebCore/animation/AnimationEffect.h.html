48namespace WebCore {
49
50class AnimationEffect : public RefCounted<AnimationEffect>, public CanMakeWeakPtr<AnimationEffect> {
51public:
52 virtual ~AnimationEffect();
53
54 virtual bool isKeyframeEffect() const { return false; }
55
56 EffectTiming getBindingsTiming() const;
57 EffectTiming getTiming() const;
58 BasicEffectTiming getBasicTiming(std::optional<Seconds> = std::nullopt) const;
59 ComputedEffectTiming getBindingsComputedTiming() const;
60 ComputedEffectTiming getComputedTiming(std::optional<Seconds> = std::nullopt) const;
61 ExceptionOr<void> bindingsUpdateTiming(std::optional<OptionalEffectTiming>);
62 ExceptionOr<void> updateTiming(std::optional<OptionalEffectTiming>);
63
64 virtual void apply(RenderStyle& targetStyle, const RenderStyle* parentElementStyle, std::optional<Seconds> = std::nullopt) = 0;
65 virtual void invalidate() = 0;
66 virtual void animationDidTick() = 0;
67 virtual void animationDidPlay() = 0;
68 virtual void animationDidChangeTimingProperties() = 0;
69 virtual void animationWasCanceled() = 0;
70 virtual void animationSuspensionStateDidChange(bool) = 0;
71 virtual void animationTimelineDidChange(AnimationTimeline*) = 0;
72
73 WebAnimation* animation() const { return m_animation.get(); }
74 virtual void setAnimation(WebAnimation* animation) { m_animation = makeWeakPtr(animation); }
75
76 Seconds delay() const { return m_delay; }
77 void setDelay(const Seconds&);
78
79 Seconds endDelay() const { return m_endDelay; }
80 void setEndDelay(const Seconds&);
81
82 FillMode fill() const { return m_fill; }
83 void setFill(FillMode);
84
85 double iterationStart() const { return m_iterationStart; }
86 ExceptionOr<void> setIterationStart(double);
87
88 double iterations() const { return m_iterations; }
89 ExceptionOr<void> setIterations(double);
90
91 Seconds iterationDuration() const { return m_iterationDuration; }
92 void setIterationDuration(const Seconds&);
93
94 PlaybackDirection direction() const { return m_direction; }
95 void setDirection(PlaybackDirection);
96
97 TimingFunction* timingFunction() const { return m_timingFunction.get(); }
98 void setTimingFunction(const RefPtr<TimingFunction>&);
99
100 Seconds activeDuration() const { return m_activeDuration; }
101 Seconds endTime() const { return m_endTime; }
102
103 void updateStaticTimingProperties();
104
105 virtual Seconds timeToNextTick() const { return Seconds::infinity(); }
106
107protected:
108 explicit AnimationEffect();
109
110 virtual std::optional<double> progressUntilNextStep(double) const;
111
112private:
113 enum class ComputedDirection : uint8_t { Forwards, Reverse };
114
115 FillMode m_fill { FillMode::Auto };
116 PlaybackDirection m_direction { PlaybackDirection::Normal };
117
118 WeakPtr<WebAnimation> m_animation;
119 RefPtr<TimingFunction> m_timingFunction;
120
121 double m_iterationStart { 0 };
122 double m_iterations { 1 };
123
124 Seconds m_delay { 0_s };
125 Seconds m_endDelay { 0_s };
126 Seconds m_iterationDuration { 0_s };
127 Seconds m_activeDuration { 0_s };
128 Seconds m_endTime { 0_s };
129};
130
131} // namespace WebCore
