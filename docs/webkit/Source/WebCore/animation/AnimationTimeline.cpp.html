38AnimationTimeline::AnimationTimeline() = default;
39AnimationTimeline::~AnimationTimeline() = default;
40
41void AnimationTimeline::forgetAnimation(WebAnimation* animation)
42{
43 m_allAnimations.removeFirst(animation);
44}
45
46void AnimationTimeline::animationTimingDidChange(WebAnimation& animation)
47{
48 updateGlobalPosition(animation);
49
50 if (m_animations.add(&animation)) {
51 m_allAnimations.append(makeWeakPtr(&animation));
52 auto* timeline = animation.timeline();
53 if (timeline && timeline != this)
54 timeline->removeAnimation(animation);
55 }
56}
57
58void AnimationTimeline::updateGlobalPosition(WebAnimation& animation)
59{
60 static uint64_t s_globalPosition = 0;
61 if (!animation.globalPosition() && animation.canHaveGlobalPosition())
62 animation.setGlobalPosition(++s_globalPosition);
63}
64
65void AnimationTimeline::removeAnimation(WebAnimation& animation)
66{
67 ASSERT(!animation.timeline() || animation.timeline() == this);
68 m_animations.remove(&animation);
69 if (is<KeyframeEffect>(animation.effect())) {
70 if (auto styleable = downcast<KeyframeEffect>(animation.effect())->targetStyleable()) {
71 styleable->animationWasRemoved(animation);
72 styleable->ensureKeyframeEffectStack().removeEffect(*downcast<KeyframeEffect>(animation.effect()));
73 }
74 }
75}
76
77std::optional<double> AnimationTimeline::bindingsCurrentTime()
78{
79 auto time = currentTime();
80 if (!time)
81 return std::nullopt;
82 return secondsToWebAnimationsAPITime(*time);
83}
