37namespace WebCore {
38
39class AnimationTimeline : public RefCounted<AnimationTimeline>, public CanMakeWeakPtr<AnimationTimeline> {
40public:
41 virtual ~AnimationTimeline();
42
43 virtual bool isDocumentTimeline() const { return false; }
44
45 const AnimationCollection& relevantAnimations() const { return m_animations; }
46
47 void forgetAnimation(WebAnimation*);
48 virtual void animationTimingDidChange(WebAnimation&);
49 virtual void removeAnimation(WebAnimation&);
50
51 std::optional<double> bindingsCurrentTime();
52 virtual std::optional<Seconds> currentTime() { return m_currentTime; }
53
54protected:
55 explicit AnimationTimeline();
56
57 Vector<WeakPtr<WebAnimation>> m_allAnimations;
58 AnimationCollection m_animations;
59
60private:
61 void updateGlobalPosition(WebAnimation&);
62
63 Markable<Seconds, Seconds::MarkableTraits> m_currentTime;
64};
65
66} // namespace WebCore
