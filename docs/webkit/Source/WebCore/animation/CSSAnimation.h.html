33namespace WebCore {
34
35class Animation;
36class RenderStyle;
37
38class CSSAnimation final : public DeclarativeAnimation {
39 WTF_MAKE_ISO_ALLOCATED(CSSAnimation);
40public:
41 static Ref<CSSAnimation> create(const Styleable&, const Animation&, const RenderStyle* oldStyle, const RenderStyle& newStyle, const RenderStyle* parentElementStyle);
42 ~CSSAnimation() = default;
43
44 bool isCSSAnimation() const override { return true; }
45 const String& animationName() const { return m_animationName; }
46
47 void effectTimingWasUpdatedUsingBindings(OptionalEffectTiming);
48 void effectKeyframesWereSetUsingBindings();
49
50private:
51 CSSAnimation(const Styleable&, const Animation&);
52
53 void syncPropertiesWithBackingAnimation() final;
54 Ref<AnimationEventBase> createEvent(const AtomString& eventType, double elapsedTime, const String& pseudoId, std::optional<Seconds> timelineTime) final;
55
56 ExceptionOr<void> bindingsPlay() final;
57 ExceptionOr<void> bindingsPause() final;
58 void setBindingsEffect(RefPtr<AnimationEffect>&&) final;
59 void setBindingsStartTime(std::optional<double>) final;
60 ExceptionOr<void> bindingsReverse() final;
61
62 enum class Property : uint8_t {
63 Name = 1 << 0,
64 Duration = 1 << 1,
65 TimingFunction = 1 << 2,
66 IterationCount = 1 << 3,
67 Direction = 1 << 4,
68 PlayState = 1 << 5,
69 Delay = 1 << 6,
70 FillMode = 1 << 7
71 };
72
73 String m_animationName;
74 OptionSet<Property> m_overriddenProperties;
75};

77} // namespace WebCore
