36namespace WebCore {
37
38class Animation;
39class RenderStyle;
40
41class CSSTransition final : public DeclarativeAnimation {
42 WTF_MAKE_ISO_ALLOCATED(CSSTransition);
43public:
44 static Ref<CSSTransition> create(const Styleable&, CSSPropertyID, MonotonicTime generationTime, const Animation&, const RenderStyle* oldStyle, const RenderStyle& newStyle, Seconds delay, Seconds duration, const RenderStyle& reversingAdjustedStartStyle, double);
45 ~CSSTransition() = default;
46
47 bool isCSSTransition() const override { return true; }
48 String transitionProperty() const { return getPropertyNameString(m_property); }
49 CSSPropertyID property() const { return m_property; }
50 MonotonicTime generationTime() const { return m_generationTime; }
51 std::optional<Seconds> timelineTimeAtCreation() const { return m_timelineTimeAtCreation; }
52 const RenderStyle& targetStyle() const { return *m_targetStyle; }
53 const RenderStyle& currentStyle() const { return *m_currentStyle; }
54 const RenderStyle& reversingAdjustedStartStyle() const { return *m_reversingAdjustedStartStyle; }
55 double reversingShorteningFactor() const { return m_reversingShorteningFactor; }
56
57private:
58 CSSTransition(const Styleable&, CSSPropertyID, MonotonicTime generationTime, const Animation&, const RenderStyle& oldStyle, const RenderStyle& targetStyle, const RenderStyle& reversingAdjustedStartStyle, double);
59 void setTimingProperties(Seconds delay, Seconds duration);
60 Ref<AnimationEventBase> createEvent(const AtomString& eventType, double elapsedTime, const String& pseudoId, std::optional<Seconds> timelineTime) final;
61 void resolve(RenderStyle& targetStyle, const RenderStyle* parentElementStyle, std::optional<Seconds>) final;
62
63 CSSPropertyID m_property;
64 MonotonicTime m_generationTime;
65 Markable<Seconds, Seconds::MarkableTraits> m_timelineTimeAtCreation;
66 std::unique_ptr<RenderStyle> m_targetStyle;
67 std::unique_ptr<RenderStyle> m_currentStyle;
68 std::unique_ptr<RenderStyle> m_reversingAdjustedStartStyle;
69 double m_reversingShorteningFactor;
70
71};
72
73} // namespace WebCore
