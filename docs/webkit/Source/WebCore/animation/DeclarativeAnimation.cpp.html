65void DeclarativeAnimation::tick()
66{
67 LOG_WITH_STREAM(Animations, stream << "DeclarativeAnimation::tick for element " << m_owningElement);
68
69 bool wasRelevant = isRelevant();
70
71 WebAnimation::tick();
72 invalidateDOMEvents();
73
74 // If a declarative animation transitions from a non-idle state to an idle state, it means it was
75 // canceled using the Web Animations API and it should be disassociated from its owner element.
76 // From this point on, this animation is like any other animation and should not appear in the
77 // maps containing running CSS Transitions and CSS Animations for a given element.
78 if (wasRelevant && playState() == WebAnimation::PlayState::Idle)
79 disassociateFromOwningElement();
80}

82bool DeclarativeAnimation::canHaveGlobalPosition()
83{
84 // https://drafts.csswg.org/css-animations-2/#animation-composite-order
85 // https://drafts.csswg.org/css-transitions-2/#animation-composite-order
86 // CSS Animations and CSS Transitions generated using the markup defined in this specification are not added
87 // to the global animation list when they are created. Instead, these animations are appended to the global
88 // animation list at the first moment when they transition out of the idle play state after being disassociated
89 // from their owning element.
90 return !m_owningElement && playState() != WebAnimation::PlayState::Idle;
91}

93void DeclarativeAnimation::disassociateFromOwningElement()
94{
95 if (!m_owningElement)
96 return;
97
98 owningElement()->removeDeclarativeAnimationFromListsForOwningElement(*this);
99 m_owningElement = nullptr;
100}

108void DeclarativeAnimation::initialize(const RenderStyle* oldStyle, const RenderStyle& newStyle, const RenderStyle* parentElementStyle)
109{
110 // We need to suspend invalidation of the animation's keyframe effect during its creation
111 // as it would otherwise trigger invalidation of the document's style and this would be
112 // incorrect since it would happen during style invalidation.
113 suspendEffectInvalidation();
114
115 ASSERT(m_owningElement);
116
117 setEffect(KeyframeEffect::create(*m_owningElement, m_owningPseudoId));
118 setTimeline(&m_owningElement->document().timeline());
119 downcast<KeyframeEffect>(effect())->computeDeclarativeAnimationBlendingKeyframes(oldStyle, newStyle, parentElementStyle);
120 syncPropertiesWithBackingAnimation();
121 if (backingAnimation().playState() == AnimationPlayState::Playing)
122 play();
123 else
124 pause();
125
126 unsuspendEffectInvalidation();
127}

217void DeclarativeAnimation::cancel()
218{
219 auto cancelationTime = 0_s;
220 if (auto* animationEffect = effect()) {
221 if (auto activeTime = animationEffect->getBasicTiming().activeTime)
222 cancelationTime = *activeTime;
223 }
224
225 WebAnimation::cancel();
226
227 invalidateDOMEvents(cancelationTime);
228}

230void DeclarativeAnimation::cancelFromStyle()

236AnimationEffectPhase DeclarativeAnimation::phaseWithoutEffect() const
237{
238 // This shouldn't be called if we actually have an effect.
239 ASSERT(!effect());
240
241 auto animationCurrentTime = currentTime();
242 if (!animationCurrentTime)
243 return AnimationEffectPhase::Idle;
244
245 // Since we don't have an effect, the duration will be zero so the phase is 'before' if the current time is less than zero.
246 return *animationCurrentTime < 0_s ? AnimationEffectPhase::Before : AnimationEffectPhase::After;
247}

249void DeclarativeAnimation::invalidateDOMEvents(Seconds elapsedTime)
250{
251 if (!m_owningElement)
252 return;
253
254 auto isPending = pending();
255 if (isPending && m_wasPending)
256 return;
257
258 double iteration = 0;
259 AnimationEffectPhase currentPhase;
260 Seconds intervalStart;
261 Seconds intervalEnd;
262
263 auto* animationEffect = effect();
264 if (animationEffect) {
265 auto timing = animationEffect->getComputedTiming();
266 if (auto computedIteration = timing.currentIteration)
267 iteration = *computedIteration;
268 currentPhase = timing.phase;
269 intervalStart = std::max(0_s, Seconds::fromMilliseconds(std::min(-timing.delay, timing.activeDuration)));
270 intervalEnd = std::max(0_s, Seconds::fromMilliseconds(std::min(timing.endTime - timing.delay, timing.activeDuration)));
271 } else {
272 iteration = 0;
273 currentPhase = phaseWithoutEffect();
274 intervalStart = 0_s;
275 intervalEnd = 0_s;
276 }
277
278 bool wasActive = m_previousPhase == AnimationEffectPhase::Active;
279 bool wasAfter = m_previousPhase == AnimationEffectPhase::After;
280 bool wasBefore = m_previousPhase == AnimationEffectPhase::Before;
281 bool wasIdle = m_previousPhase == AnimationEffectPhase::Idle;
282
283 bool isActive = currentPhase == AnimationEffectPhase::Active;
284 bool isAfter = currentPhase == AnimationEffectPhase::After;
285 bool isBefore = currentPhase == AnimationEffectPhase::Before;
286 bool isIdle = currentPhase == AnimationEffectPhase::Idle;
287
288 if (is<CSSAnimation>(this)) {
289 // https://drafts.csswg.org/css-animations-2/#events
290 if ((wasIdle || wasBefore) && isActive)
291 enqueueDOMEvent(eventNames().animationstartEvent, intervalStart);
292 else if ((wasIdle || wasBefore) && isAfter) {
293 enqueueDOMEvent(eventNames().animationstartEvent, intervalStart);
294 enqueueDOMEvent(eventNames().animationendEvent, intervalEnd);
295 } else if (wasActive && isBefore)
296 enqueueDOMEvent(eventNames().animationendEvent, intervalStart);
297 else if (wasActive && isActive && m_previousIteration != iteration) {
298 auto iterationBoundary = iteration;
299 if (m_previousIteration > iteration)
300 iterationBoundary++;
301 auto elapsedTime = animationEffect ? animationEffect->iterationDuration() * (iterationBoundary - animationEffect->iterationStart()) : 0_s;
302 enqueueDOMEvent(eventNames().animationiterationEvent, elapsedTime);
303 } else if (wasActive && isAfter)
304 enqueueDOMEvent(eventNames().animationendEvent, intervalEnd);
305 else if (wasAfter && isActive)
306 enqueueDOMEvent(eventNames().animationstartEvent, intervalEnd);
307 else if (wasAfter && isBefore) {
308 enqueueDOMEvent(eventNames().animationstartEvent, intervalEnd);
309 enqueueDOMEvent(eventNames().animationendEvent, intervalStart);
310 } else if ((!wasIdle && !wasAfter) && isIdle)
311 enqueueDOMEvent(eventNames().animationcancelEvent, elapsedTime);
312 } else if (is<CSSTransition>(this)) {
313 // https://drafts.csswg.org/css-transitions-2/#transition-events
314 if (wasIdle && (isPending || isBefore))
315 enqueueDOMEvent(eventNames().transitionrunEvent, intervalStart);
316 else if (wasIdle && isActive) {
317 enqueueDOMEvent(eventNames().transitionrunEvent, intervalStart);
318 enqueueDOMEvent(eventNames().transitionstartEvent, intervalStart);
319 } else if (wasIdle && isAfter) {
320 enqueueDOMEvent(eventNames().transitionrunEvent, intervalStart);
321 enqueueDOMEvent(eventNames().transitionstartEvent, intervalStart);
322 enqueueDOMEvent(eventNames().transitionendEvent, intervalEnd);
323 } else if ((m_wasPending || wasBefore) && isActive)
324 enqueueDOMEvent(eventNames().transitionstartEvent, intervalStart);
325 else if ((m_wasPending || wasBefore) && isAfter) {
326 enqueueDOMEvent(eventNames().transitionstartEvent, intervalStart);
327 enqueueDOMEvent(eventNames().transitionendEvent, intervalEnd);
328 } else if (wasActive && isAfter)
329 enqueueDOMEvent(eventNames().transitionendEvent, intervalEnd);
330 else if (wasActive && isBefore)
331 enqueueDOMEvent(eventNames().transitionendEvent, intervalStart);
332 else if (wasAfter && isActive)
333 enqueueDOMEvent(eventNames().transitionstartEvent, intervalEnd);
334 else if (wasAfter && isBefore) {
335 enqueueDOMEvent(eventNames().transitionstartEvent, intervalEnd);
336 enqueueDOMEvent(eventNames().transitionendEvent, intervalStart);
337 } else if ((!wasIdle && !wasAfter) && isIdle)
338 enqueueDOMEvent(eventNames().transitioncancelEvent, elapsedTime);
339 }
340
341 m_wasPending = isPending;
342 m_previousPhase = currentPhase;
343 m_previousIteration = iteration;
344}

346void DeclarativeAnimation::enqueueDOMEvent(const AtomString& eventType, Seconds elapsedTime)
347{
348 if (!m_owningElement)
349 return;
350
351 auto time = secondsToWebAnimationsAPITime(elapsedTime) / 1000;
352 auto pseudoId = pseudoIdAsString(m_owningPseudoId);
353 auto timelineTime = timeline() ? timeline()->currentTime() : std::nullopt;
354 auto event = createEvent(eventType, time, pseudoId, timelineTime);
355 event->setTarget(m_owningElement.get());
356 enqueueAnimationEvent(WTFMove(event));
357}
