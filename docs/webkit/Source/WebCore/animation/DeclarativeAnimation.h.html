26#pragma once
27
28#include "AnimationEffect.h"
29#include "AnimationEffectPhase.h"
30#include "Styleable.h"
31#include "WebAnimation.h"
32#include <wtf/Ref.h>
33#include <wtf/WeakPtr.h>
34
35namespace WebCore {
36
37class Animation;
38class AnimationEventBase;
39class Element;
40class RenderStyle;
41
42class DeclarativeAnimation : public WebAnimation {
43 WTF_MAKE_ISO_ALLOCATED(DeclarativeAnimation);
44public:
45 ~DeclarativeAnimation();
46
47 bool isDeclarativeAnimation() const final { return true; }
48
49 const std::optional<const Styleable> owningElement() const;
50 const Animation& backingAnimation() const { return m_backingAnimation; }
51 void setBackingAnimation(const Animation&);
52 void cancelFromStyle();
53
54 std::optional<double> bindingsStartTime() const final;
55 void setBindingsStartTime(std::optional<double>) override;
56 std::optional<double> bindingsCurrentTime() const final;
57 ExceptionOr<void> setBindingsCurrentTime(std::optional<double>) final;
58 WebAnimation::PlayState bindingsPlayState() const final;
59 WebAnimation::ReplaceState bindingsReplaceState() const final;
60 bool bindingsPending() const final;
61 WebAnimation::ReadyPromise& bindingsReady() final;
62 WebAnimation::FinishedPromise& bindingsFinished() final;
63 ExceptionOr<void> bindingsPlay() override;
64 ExceptionOr<void> bindingsPause() override;
65
66 void setTimeline(RefPtr<AnimationTimeline>&&) final;
67 void cancel() final;
68
69 void tick() override;
70
71 bool canHaveGlobalPosition() final;
72
73 void flushPendingStyleChanges() const;
74
75protected:
76 DeclarativeAnimation(const Styleable&, const Animation&);
77
78 virtual void initialize(const RenderStyle* oldStyle, const RenderStyle& newStyle, const RenderStyle* parentElementStyle);
79 virtual void syncPropertiesWithBackingAnimation();
80 // elapsedTime is the animation's current time at the time the event is added and is exposed through the DOM API, timelineTime is the animations'
81 // timeline current time and is not exposed through the DOM API but used by the DocumentTimeline for sorting events before dispatch.
82 virtual Ref<AnimationEventBase> createEvent(const AtomString& eventType, double elapsedTime, const String& pseudoId, std::optional<Seconds> timelineTime) = 0;
83 void invalidateDOMEvents(Seconds elapsedTime = 0_s);
84
85private:
86 void disassociateFromOwningElement();
87 AnimationEffectPhase phaseWithoutEffect() const;
88 void enqueueDOMEvent(const AtomString&, Seconds);
89
90 bool m_wasPending { false };
91 AnimationEffectPhase m_previousPhase { AnimationEffectPhase::Idle };
92
93 WeakPtr<Element> m_owningElement;
94 PseudoId m_owningPseudoId;
95 Ref<Animation> m_backingAnimation;
96 double m_previousIteration;
97};
98
99} // namespace WebCore
100
101SPECIALIZE_TYPE_TRAITS_WEB_ANIMATION(DeclarativeAnimation, isDeclarativeAnimation())
