38class ElementAnimationRareData {
39 WTF_MAKE_NONCOPYABLE(ElementAnimationRareData);
40 WTF_MAKE_FAST_ALLOCATED;
41public:
42 explicit ElementAnimationRareData(PseudoId);
43 ~ElementAnimationRareData();
44
45 PseudoId pseudoId() const { return m_pseudoId; }
46
47 KeyframeEffectStack* keyframeEffectStack() { return m_keyframeEffectStack.get(); }
48 KeyframeEffectStack& ensureKeyframeEffectStack();
49
50 AnimationCollection& animations() { return m_animations; }
51 CSSAnimationCollection& animationsCreatedByMarkup() { return m_animationsCreatedByMarkup; }
52 void setAnimationsCreatedByMarkup(CSSAnimationCollection&&);
53 PropertyToTransitionMap& completedTransitionsByProperty() { return m_completedTransitionsByProperty; }
54 PropertyToTransitionMap& runningTransitionsByProperty() { return m_runningTransitionsByProperty; }
55 const RenderStyle* lastStyleChangeEventStyle() const { return m_lastStyleChangeEventStyle.get(); }
56 void setLastStyleChangeEventStyle(std::unique_ptr<const RenderStyle>&& style) { m_lastStyleChangeEventStyle = WTFMove(style); }
57
58private:
59
60 std::unique_ptr<KeyframeEffectStack> m_keyframeEffectStack;
61 std::unique_ptr<const RenderStyle> m_lastStyleChangeEventStyle;
62 AnimationCollection m_animations;
63 CSSAnimationCollection m_animationsCreatedByMarkup;
64 PropertyToTransitionMap m_completedTransitionsByProperty;
65 PropertyToTransitionMap m_runningTransitionsByProperty;
66 PseudoId m_pseudoId;
67};
