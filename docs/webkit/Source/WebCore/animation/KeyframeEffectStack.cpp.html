45bool KeyframeEffectStack::addEffect(KeyframeEffect& effect)
46{
47 // To qualify for membership in an effect stack, an effect must have a target, an animation, a timeline and be relevant.
48 // This method will be called in WebAnimation and KeyframeEffect as those properties change.
49 if (!effect.targetStyleable() || !effect.animation() || !effect.animation()->timeline() || !effect.animation()->isRelevant())
50 return false;
51
52 m_effects.append(makeWeakPtr(&effect));
53 m_isSorted = false;
54 return true;
55}
56
57void KeyframeEffectStack::removeEffect(KeyframeEffect& effect)
58{
59 m_effects.removeFirst(&effect);
60}

95void KeyframeEffectStack::ensureEffectsAreSorted()
96{
97 if (m_isSorted || m_effects.size() < 2)
98 return;
99
100 std::stable_sort(m_effects.begin(), m_effects.end(), [&](auto& lhs, auto& rhs) {
101 RELEASE_ASSERT(lhs.get());
102 RELEASE_ASSERT(rhs.get());
103
104 auto* lhsAnimation = lhs->animation();
105 auto* rhsAnimation = rhs->animation();
106
107 RELEASE_ASSERT(lhsAnimation);
108 RELEASE_ASSERT(rhsAnimation);
109
110 return compareAnimationsByCompositeOrder(*lhsAnimation, *rhsAnimation);
111 });
112
113 m_isSorted = true;
114}

123OptionSet<AnimationImpact> KeyframeEffectStack::applyKeyframeEffects(RenderStyle& targetStyle, const RenderStyle& previousLastStyleChangeEventStyle, const RenderStyle* parentElementStyle)
124{
125 OptionSet<AnimationImpact> impact;
126
127 auto transformRelatedPropertyChanged = [&]() -> bool {
128 return !arePointingToEqualData(targetStyle.translate(), previousLastStyleChangeEventStyle.translate())
129 || !arePointingToEqualData(targetStyle.scale(), previousLastStyleChangeEventStyle.scale())
130 || !arePointingToEqualData(targetStyle.rotate(), previousLastStyleChangeEventStyle.rotate())
131 || targetStyle.transform() != previousLastStyleChangeEventStyle.transform();
132 }();
133
134 for (const auto& effect : sortedEffects()) {
135 ASSERT(effect->animation());
136 effect->animation()->resolve(targetStyle, parentElementStyle);
137
138 if (effect->isRunningAccelerated() || effect->isAboutToRunAccelerated())
139 impact.add(AnimationImpact::RequiresRecomposite);
140
141 if (effect->triggersStackingContext())
142 impact.add(AnimationImpact::ForcesStackingContext);
143
144 if (transformRelatedPropertyChanged && effect->isRunningAcceleratedTransformRelatedAnimation())
145 effect->transformRelatedPropertyDidChange();
146 }
147
148 return impact;
149}
