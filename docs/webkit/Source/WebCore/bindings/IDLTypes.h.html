60template<typename T>
61struct IDLType {
62 using ImplementationType = T;
63 using StorageType = T;
64 using SequenceStorageType = T;
65
66 using ParameterType = T;
67 using NullableParameterType = std::optional<ImplementationType>;
68
69 using InnerParameterType = T;
70 using NullableInnerParameterType = std::optional<ImplementationType>;
71
72 using NullableType = std::optional<ImplementationType>;
73 static NullableType nullValue() { return std::nullopt; }
74 static bool isNullValue(const NullableType& value) { return !value; }
75 static ImplementationType extractValueFromNullable(const NullableType& value) { return value.value(); }
76};

193template<typename T> struct IDLNullable : IDLType<typename T::NullableType> {
194 using InnerType = T;
195
196 using ParameterType = typename T::NullableParameterType;
197 using NullableParameterType = typename T::NullableParameterType;
198
199 using InnerParameterType = typename T::NullableInnerParameterType;
200 using NullableInnerParameterType = typename T::NullableInnerParameterType;
201
202 using NullableType = typename T::NullableType;
203 static inline auto nullValue() -> decltype(T::nullValue()) { return T::nullValue(); }
204 template<typename U> static inline bool isNullValue(U&& value) { return T::isNullValue(std::forward<U>(value)); }
205 template<typename U> static inline auto extractValueFromNullable(U&& value) -> decltype(T::extractValueFromNullable(std::forward<U>(value))) { return T::extractValueFromNullable(std::forward<U>(value)); }
206};
