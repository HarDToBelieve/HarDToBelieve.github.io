44CachedResourceHandle<CachedScript> CachedScriptFetcher::requestModuleScript(Document& document, const URL& sourceURL, String&& integrity) const
45{
46 return requestScriptWithCache(document, sourceURL, String { }, WTFMove(integrity), { });

49CachedResourceHandle<CachedScript> CachedScriptFetcher::requestScriptWithCache(Document& document, const URL& sourceURL, const String& crossOriginMode, String&& integrity, std::optional<ResourceLoadPriority> resourceLoadPriority) const
50{
51 if (!document.settings().isScriptEnabled())
52 return nullptr;
53
54 ASSERT(document.contentSecurityPolicy());
55 bool hasKnownNonce = document.contentSecurityPolicy()->allowScriptWithNonce(m_nonce, m_isInUserAgentShadowTree);
56 ResourceLoaderOptions options = CachedResourceLoader::defaultCachedResourceOptions();
57 options.contentSecurityPolicyImposition = hasKnownNonce ? ContentSecurityPolicyImposition::SkipPolicyCheck : ContentSecurityPolicyImposition::DoPolicyCheck;
58 options.sameOriginDataURLFlag = SameOriginDataURLFlag::Set;
59 options.integrity = WTFMove(integrity);
60 options.referrerPolicy = m_referrerPolicy;
61
62 auto request = createPotentialAccessControlRequest(sourceURL, WTFMove(options), document, crossOriginMode);
63 request.upgradeInsecureRequestIfNeeded(document);
64 request.setCharset(m_charset);
65 request.setPriority(WTFMove(resourceLoadPriority));
66 if (!m_initiatorName.isNull())
67 request.setInitiator(m_initiatorName);
68
69 return document.cachedResourceLoader().requestScript(WTFMove(request)).value_or(nullptr);
70}
