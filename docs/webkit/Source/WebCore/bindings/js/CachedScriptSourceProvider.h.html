26#pragma once
27
28#include "CachedResourceClient.h"
29#include "CachedResourceHandle.h"
30#include "CachedScript.h"
31#include "CachedScriptFetcher.h"
32#include <JavaScriptCore/SourceProvider.h>
33
34namespace WebCore {
35
36class CachedScriptSourceProvider : public JSC::SourceProvider, public CachedResourceClient {
37 WTF_MAKE_FAST_ALLOCATED;
38public:
39 static Ref<CachedScriptSourceProvider> create(CachedScript* cachedScript, JSC::SourceProviderSourceType sourceType, Ref<CachedScriptFetcher>&& scriptFetcher) { return adoptRef(*new CachedScriptSourceProvider(cachedScript, sourceType, WTFMove(scriptFetcher))); }
40
41 virtual ~CachedScriptSourceProvider()
42 {
43 m_cachedScript->removeClient(*this);
44 }
45
46 unsigned hash() const override { return m_cachedScript->scriptHash(); }
47 StringView source() const override { return m_cachedScript->script(); }
48
49private:
50 CachedScriptSourceProvider(CachedScript* cachedScript, JSC::SourceProviderSourceType sourceType, Ref<CachedScriptFetcher>&& scriptFetcher)
51 : SourceProvider(JSC::SourceOrigin { cachedScript->response().url(), WTFMove(scriptFetcher) }, String(cachedScript->response().url().string()), TextPosition(), sourceType)
52 , m_cachedScript(cachedScript)
53 {
54 m_cachedScript->addClient(*this);
55 }
56
57 CachedResourceHandle<CachedScript> m_cachedScript;
58};
59
60} // namespace WebCore
