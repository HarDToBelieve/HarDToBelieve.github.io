40DOMGCOutputConstraint::DOMGCOutputConstraint(VM& vm, JSVMClientData& clientData)
41 : MarkingConstraint("Domo", "DOM Output", ConstraintVolatility::SeldomGreyed, ConstraintConcurrency::Concurrent, ConstraintParallelism::Parallel)
42 , m_vm(vm)
43 , m_clientData(clientData)
44 , m_lastExecutionVersion(vm.heap.mutatorExecutionVersion())
45{
46}

48DOMGCOutputConstraint::~DOMGCOutputConstraint()
49{
50}

52template<typename Visitor>
53void DOMGCOutputConstraint::executeImplImpl(Visitor& visitor)
54{
55 Heap& heap = m_vm.heap;
56
57 if (heap.mutatorExecutionVersion() == m_lastExecutionVersion)
58 return;
59
60 m_lastExecutionVersion = heap.mutatorExecutionVersion();
61
62 m_clientData.forEachOutputConstraintSpace(
63 [&] (Subspace& subspace) {
64 auto func = [] (Visitor& visitor, HeapCell* heapCell, HeapCell::Kind) {
65 SetRootMarkReasonScope rootScope(visitor, RootMarkReason::DOMGCOutput);
66 JSCell* cell = static_cast<JSCell*>(heapCell);
67 cell->methodTable(visitor.vm())->visitOutputConstraints(cell, visitor);
68 };
69
70 RefPtr<SharedTask<void(Visitor&)>> task = subspace.template forEachMarkedCellInParallel<Visitor>(func);
71 visitor.addParallelConstraintTask(task);
72 });
73}
74
75void DOMGCOutputConstraint::executeImpl(AbstractSlotVisitor& visitor) { executeImplImpl(visitor); }
76void DOMGCOutputConstraint::executeImpl(SlotVisitor& visitor) { executeImplImpl(visitor); }
