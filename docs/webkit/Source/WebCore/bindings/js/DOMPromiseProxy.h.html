81// Instead of storing the value of the resolution directly, DOMPromiseProxyWithResolveCallback
82// allows the owner to specify callback to be called when the resolved value is needed. This is
83// needed to avoid reference cycles when the resolved value is the owner, such as is the case with
84// FontFace and FontFaceSet.
85template<typename IDLType>
86class DOMPromiseProxyWithResolveCallback {
87 WTF_MAKE_FAST_ALLOCATED;
88public:
89 using ResolveCallback = WTF::Function<typename IDLType::ParameterType ()>;
90
91 template <typename Class, typename BaseClass>
92 DOMPromiseProxyWithResolveCallback(Class&, typename IDLType::ParameterType (BaseClass::*)());
93 DOMPromiseProxyWithResolveCallback(ResolveCallback&&);
94 ~DOMPromiseProxyWithResolveCallback() = default;
95
96 JSC::JSValue promise(JSC::JSGlobalObject&, JSDOMGlobalObject&);
97
98 void clear();
99
100 bool isFulfilled() const;
101
102 void resolve(typename IDLType::ParameterType);
103 void resolveWithNewlyCreated(typename IDLType::ParameterType);
104 void reject(Exception, RejectAsHandled = RejectAsHandled::No);
105
106private:
107 ResolveCallback m_resolveCallback;
108 std::optional<ExceptionOr<void>> m_valueOrException;
109 Vector<Ref<DeferredPromise>, 1> m_deferredPromises;
110};

113// MARK: - DOMPromiseProxy<IDLType> generic implementation
114
115template<typename IDLType>
116inline JSC::JSValue DOMPromiseProxy<IDLType>::promise(JSC::JSGlobalObject& lexicalGlobalObject, JSDOMGlobalObject& globalObject)
117{
118 UNUSED_PARAM(lexicalGlobalObject);
119 for (auto& deferredPromise : m_deferredPromises) {
120 if (deferredPromise->globalObject() == &globalObject)
121 return deferredPromise->promise();
122 }
123
124 // DeferredPromise can fail construction during worker abrupt termination.
125 auto deferredPromise = DeferredPromise::create(globalObject, DeferredPromise::Mode::RetainPromiseOnResolve);
126 if (!deferredPromise)
127 return JSC::jsUndefined();
128
129 if (m_valueOrException) {
130 if (m_valueOrException->hasException())
131 deferredPromise->reject(m_valueOrException->exception());
132 else
133 deferredPromise->template resolve<IDLType>(m_valueOrException->returnValue());
134 }
135
136 auto result = deferredPromise->promise();
137 m_deferredPromises.append(deferredPromise.releaseNonNull());
138 return result;
139}

154template<typename IDLType>
155inline void DOMPromiseProxy<IDLType>::resolve(typename IDLType::ParameterType value)
156{
157 ASSERT(!m_valueOrException);
158
159 m_valueOrException = ExceptionOr<Value> { std::forward<typename IDLType::ParameterType>(value) };
160 for (auto& deferredPromise : m_deferredPromises)
161 deferredPromise->template resolve<IDLType>(m_valueOrException->returnValue());
162}
