31namespace WebCore {
32using namespace JSC;
33
34DOMWrapperWorld::DOMWrapperWorld(JSC::VM& vm, Type type, const String& name)
35 : m_vm(vm)
36 , m_name(name)
37 , m_type(type)
38{
39 VM::ClientData* clientData = m_vm.clientData;
40 ASSERT(clientData);
41 static_cast<JSVMClientData*>(clientData)->rememberWorld(*this);
42}
43
44DOMWrapperWorld::~DOMWrapperWorld()
45{
46 VM::ClientData* clientData = m_vm.clientData;
47 ASSERT(clientData);
48 static_cast<JSVMClientData*>(clientData)->forgetWorld(*this);
49
50 // These items are created lazily.
51 while (!m_jsWindowProxies.isEmpty())
52 (*m_jsWindowProxies.begin())->destroyJSWindowProxy(*this);
53}
54
55void DOMWrapperWorld::clearWrappers()
56{
57 m_wrappers.clear();
58
59 // These items are created lazily.
60 while (!m_jsWindowProxies.isEmpty())
61 (*m_jsWindowProxies.begin())->destroyJSWindowProxy(*this);
62}
63
64DOMWrapperWorld& normalWorld(JSC::VM& vm)
65{
66 VM::ClientData* clientData = vm.clientData;
67 ASSERT(clientData);
68 return static_cast<JSVMClientData*>(clientData)->normalWorld();
69}
70
71DOMWrapperWorld& mainThreadNormalWorld()
72{
73 ASSERT(isMainThread());
74 static DOMWrapperWorld& cachedNormalWorld = normalWorld(commonVM());
75 return cachedNormalWorld;
76}
