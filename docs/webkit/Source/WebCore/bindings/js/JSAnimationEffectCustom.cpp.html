37JSValue toJSNewlyCreated(JSGlobalObject*, JSDOMGlobalObject* globalObject, Ref<AnimationEffect>&& value)
38{
39 if (value->isKeyframeEffect())
40 return createWrapper<KeyframeEffect>(globalObject, WTFMove(value));
41 return createWrapper<AnimationEffect>(globalObject, WTFMove(value));
42}
43
44JSValue toJS(JSGlobalObject* lexicalGlobalObject, JSDOMGlobalObject* globalObject, AnimationEffect& value)
45{
46 return wrap(lexicalGlobalObject, globalObject, value);
47}
