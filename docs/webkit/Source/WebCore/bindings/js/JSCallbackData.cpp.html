41// https://heycam.github.io/webidl/#call-a-user-objects-operation
42JSValue JSCallbackData::invokeCallback(JSDOMGlobalObject& globalObject, JSObject* callback, JSValue thisValue, MarkedArgumentBuffer& args, CallbackType method, PropertyName functionName, NakedPtr<JSC::Exception>& returnedException)
43{
44 ASSERT(callback);
45
46 JSGlobalObject* lexicalGlobalObject = &globalObject;
47 VM& vm = lexicalGlobalObject->vm();
48 auto scope = DECLARE_CATCH_SCOPE(vm);
49
50 JSValue function;
51 CallData callData;
52
53 if (method != CallbackType::Object) {
54 function = callback;
55 callData = getCallData(vm, callback);
56 }
57 if (callData.type == CallData::Type::None) {
58 if (method == CallbackType::Function) {
59 returnedException = JSC::Exception::create(vm, createTypeError(lexicalGlobalObject));
60 return JSValue();
61 }
62
63 ASSERT(!functionName.isNull());
64 function = callback->get(lexicalGlobalObject, functionName);
65 if (UNLIKELY(scope.exception())) {
66 returnedException = scope.exception();
67 scope.clearException();
68 return JSValue();
69 }
70
71 callData = getCallData(vm, function);
72 if (callData.type == CallData::Type::None) {
73 returnedException = JSC::Exception::create(vm, createTypeError(lexicalGlobalObject, makeString("'", String(functionName.uid()), "' property of callback interface should be callable")));
74 return JSValue();
75 }
76
77 thisValue = callback;
78 }
79
80 ASSERT(!function.isEmpty());
81 ASSERT(callData.type != CallData::Type::None);
82
83 ScriptExecutionContext* context = globalObject.scriptExecutionContext();
84 // We will fail to get the context if the frame has been detached.
85 if (!context)
86 return JSValue();
87
88 JSExecState::instrumentFunction(context, callData);
89
90 returnedException = nullptr;
91 JSValue result = JSExecState::profiledCall(lexicalGlobalObject, JSC::ProfilingReason::Other, function, callData, thisValue, args, returnedException);
92
93 InspectorInstrumentation::didCallFunction(context);
94
95 return result;
96}
