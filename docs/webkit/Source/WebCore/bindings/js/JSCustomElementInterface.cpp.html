60Ref<Element> JSCustomElementInterface::constructElementWithFallback(Document& document, const AtomString& localName)
61{
62 if (auto element = tryToConstructCustomElement(document, localName))
63 return element.releaseNonNull();
64
65 auto element = HTMLUnknownElement::create(QualifiedName(nullAtom(), localName, HTMLNames::xhtmlNamespaceURI), document);
66 element->setIsCustomElementUpgradeCandidate();
67 element->setIsFailedCustomElement();
68
69 return element;
70}

72Ref<Element> JSCustomElementInterface::constructElementWithFallback(Document& document, const QualifiedName& name)
73{
74 if (auto element = tryToConstructCustomElement(document, name.localName())) {
75 if (!name.prefix().isNull())
76 element->setPrefix(name.prefix());
77 return element.releaseNonNull();
78 }
79
80 auto element = HTMLUnknownElement::create(name, document);
81 element->setIsCustomElementUpgradeCandidate();
82 element->setIsFailedCustomElement();
83
84 return element;
85}

87RefPtr<Element> JSCustomElementInterface::tryToConstructCustomElement(Document& document, const AtomString& localName)
88{
89 if (!canInvokeCallback())
90 return nullptr;
91
92 Ref<JSCustomElementInterface> protectedThis(*this);
93
94 VM& vm = m_isolatedWorld->vm();
95 JSLockHolder lock(vm);
96 auto scope = DECLARE_CATCH_SCOPE(vm);
97
98 if (!m_constructor)
99 return nullptr;
100
101 ASSERT(&document == scriptExecutionContext());
102 auto* lexicalGlobalObject = document.globalObject();
103 ASSERT(lexicalGlobalObject);
104 if (!lexicalGlobalObject)
105 return nullptr;
106 auto element = constructCustomElementSynchronously(document, vm, *lexicalGlobalObject, m_constructor.get(), localName);
107 EXCEPTION_ASSERT(!!scope.exception() == !element);
108 if (!element) {
109 auto* exception = scope.exception();
110 scope.clearException();
111 reportException(lexicalGlobalObject, exception);
112 return nullptr;
113 }
114
115 return element;
116}

118// https://dom.spec.whatwg.org/#concept-create-element
119// 6. 1. If the synchronous custom elements flag is set
120static RefPtr<Element> constructCustomElementSynchronously(Document& document, VM& vm, JSGlobalObject& lexicalGlobalObject, JSObject* constructor, const AtomString& localName)
121{
122 auto scope = DECLARE_THROW_SCOPE(vm);
123 auto constructData = getConstructData(vm, constructor);
124 if (constructData.type == CallData::Type::None) {
125 ASSERT_NOT_REACHED();
126 return nullptr;
127 }
128
129 JSExecState::instrumentFunction(&document, constructData);
130 MarkedArgumentBuffer args;
131 ASSERT(!args.hasOverflowed());
132 JSValue newElement = construct(&lexicalGlobalObject, constructor, constructData, args);
133 InspectorInstrumentation::didCallFunction(&document);
134 RETURN_IF_EXCEPTION(scope, nullptr);
135
136 ASSERT(!newElement.isEmpty());
137 HTMLElement* wrappedElement = JSHTMLElement::toWrapped(vm, newElement);
138 if (!wrappedElement) {
139 throwTypeError(&lexicalGlobalObject, scope, "The result of constructing a custom element must be a HTMLElement"_s);
140 return nullptr;
141 }
142
143 if (wrappedElement->hasAttributes()) {
144 throwNotSupportedError(lexicalGlobalObject, scope, "A newly constructed custom element must not have attributes"_s);
145 return nullptr;
146 }
147 if (wrappedElement->hasChildNodes()) {
148 throwNotSupportedError(lexicalGlobalObject, scope, "A newly constructed custom element must not have child nodes"_s);
149 return nullptr;
150 }
151 if (wrappedElement->parentNode()) {
152 throwNotSupportedError(lexicalGlobalObject, scope, "A newly constructed custom element must not have a parent node"_s);
153 return nullptr;
154 }
155 if (&wrappedElement->document() != &document) {
156 throwNotSupportedError(lexicalGlobalObject, scope, "A newly constructed custom element belongs to a wrong document"_s);
157 return nullptr;
158 }
159 ASSERT(wrappedElement->namespaceURI() == HTMLNames::xhtmlNamespaceURI);
160 if (wrappedElement->localName() != localName) {
161 throwNotSupportedError(lexicalGlobalObject, scope, "A newly constructed custom element has incorrect local name"_s);
162 return nullptr;
163 }
164
165 return wrappedElement;
166}

168// https://html.spec.whatwg.org/multipage/custom-elements.html#concept-upgrade-an-element
169void JSCustomElementInterface::upgradeElement(Element& element)
170{
171 ASSERT(element.tagQName() == name());
172
173 if (element.isDefinedCustomElement() || element.isFailedCustomElement())
174 return; // If element's custom element state is not "undefined" or "uncustomized", then return.
175
176 ASSERT(element.isCustomElementUpgradeCandidate());
177 if (!canInvokeCallback())
178 return;
179
180 Ref<JSCustomElementInterface> protectedThis(*this);
181 VM& vm = m_isolatedWorld->vm();
182 JSLockHolder lock(vm);
183 auto scope = DECLARE_THROW_SCOPE(vm);
184
185 if (!m_constructor)
186 return;
187
188 auto* context = scriptExecutionContext();
189 if (!context)
190 return;
191 auto* globalObject = toJSDOMWindow(downcast<Document>(*context).frame(), m_isolatedWorld);
192 if (!globalObject)
193 return;
194 JSGlobalObject* lexicalGlobalObject = globalObject;
195
196 auto constructData = getConstructData(vm, m_constructor.get());
197 if (constructData.type == CallData::Type::None) {
198 ASSERT_NOT_REACHED();
199 return;
200 }
201
202 CustomElementReactionQueue::enqueuePostUpgradeReactions(element);
203
204 // Unlike spec, set element's custom element state to "failed" after enqueueing post-upgrade reactions
205 // to avoid hitting debug assertions in enqueuePostUpgradeReactions.
206 element.setIsFailedCustomElementWithoutClearingReactionQueue();
207
208 m_constructionStack.append(&element);
209
210 MarkedArgumentBuffer args;
211 ASSERT(!args.hasOverflowed());
212 JSExecState::instrumentFunction(context, constructData);
213 JSValue returnedElement = construct(lexicalGlobalObject, m_constructor.get(), constructData, args);
214 InspectorInstrumentation::didCallFunction(context);
215
216 m_constructionStack.removeLast();
217
218 if (UNLIKELY(scope.exception())) {
219 element.clearReactionQueueFromFailedCustomElement();
220 reportException(lexicalGlobalObject, scope.exception());
221 return;
222 }
223
224 Element* wrappedElement = JSElement::toWrapped(vm, returnedElement);
225 if (!wrappedElement || wrappedElement != &element) {
226 element.clearReactionQueueFromFailedCustomElement();
227 reportException(lexicalGlobalObject, createDOMException(lexicalGlobalObject, TypeError, "Custom element constructor returned a wrong element"));
228 return;
229 }
230 element.setIsDefinedCustomElement(*this);
231}

233void JSCustomElementInterface::invokeCallback(Element& element, JSObject* callback, const WTF::Function<void(JSGlobalObject*, JSDOMGlobalObject*, MarkedArgumentBuffer&)>& addArguments)
234{
235 if (!canInvokeCallback())
236 return;
237
238 auto* context = scriptExecutionContext();
239 if (!context)
240 return;
241
242 Ref<JSCustomElementInterface> protectedThis(*this);
243 VM& vm = m_isolatedWorld->vm();
244 JSLockHolder lock(vm);
245
246 auto* globalObject = toJSDOMWindow(downcast<Document>(*context).frame(), m_isolatedWorld);
247 if (!globalObject)
248 return;
249 JSGlobalObject* lexicalGlobalObject = globalObject;
250
251 JSObject* jsElement = asObject(toJS(lexicalGlobalObject, globalObject, element));
252
253 auto callData = getCallData(vm, callback);
254 ASSERT(callData.type != CallData::Type::None);
255
256 MarkedArgumentBuffer args;
257 addArguments(lexicalGlobalObject, globalObject, args);
258 RELEASE_ASSERT(!args.hasOverflowed());
259
260 JSExecState::instrumentFunction(context, callData);
261
262 NakedPtr<JSC::Exception> exception;
263 JSExecState::call(lexicalGlobalObject, callback, callData, jsElement, args, exception);
264
265 InspectorInstrumentation::didCallFunction(context);
266
267 if (exception)
268 reportException(lexicalGlobalObject, exception);
269}

296void JSCustomElementInterface::invokeAdoptedCallback(Element& element, Document& oldDocument, Document& newDocument)
297{
298 invokeCallback(element, m_adoptedCallback.get(), [&](JSGlobalObject* lexicalGlobalObject, JSDOMGlobalObject* globalObject, MarkedArgumentBuffer& args) {

304void JSCustomElementInterface::setAttributeChangedCallback(JSC::JSObject* callback, const Vector<String>& observedAttributes)
305{
306 m_attributeChangedCallback = callback;
307 m_observedAttributes.clear();
308 for (auto& name : observedAttributes)
309 m_observedAttributes.add(name);
310}

312void JSCustomElementInterface::invokeAttributeChangedCallback(Element& element, const QualifiedName& attributeName, const AtomString& oldValue, const AtomString& newValue)
313{
314 invokeCallback(element, m_attributeChangedCallback.get(), [&](JSGlobalObject* lexicalGlobalObject, JSDOMGlobalObject*, MarkedArgumentBuffer& args) {
315 args.append(toJS<IDLDOMString>(*lexicalGlobalObject, attributeName.localName()));
316 args.append(toJS<IDLNullable<IDLDOMString>>(*lexicalGlobalObject, oldValue));
317 args.append(toJS<IDLNullable<IDLDOMString>>(*lexicalGlobalObject, newValue));
318 args.append(toJS<IDLNullable<IDLDOMString>>(*lexicalGlobalObject, attributeName.namespaceURI()));
319 });
320}
