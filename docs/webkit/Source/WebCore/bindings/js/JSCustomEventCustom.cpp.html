37JSC::JSValue JSCustomEvent::detail(JSC::JSGlobalObject& lexicalGlobalObject) const
38{
39 return cachedPropertyValue(lexicalGlobalObject, *this, wrapped().cachedDetail(), [this] {
40 JSC::JSValue detail = wrapped().detail();
41 return detail ? detail : JSC::jsNull();
42 });
43}
44
45template<typename Visitor>
46void JSCustomEvent::visitAdditionalChildren(Visitor& visitor)
47{
48 wrapped().detail().visit(visitor);
49 wrapped().cachedDetail().visit(visitor);
50}
