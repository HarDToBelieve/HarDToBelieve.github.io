38// An implementation of the 'named property visibility algorithm'
39// https://heycam.github.io/webidl/#dfn-named-property-visibility
40template<LegacyOverrideBuiltIns overrideBuiltins, class JSClass>
41static bool isVisibleNamedProperty(JSC::JSGlobalObject& lexicalGlobalObject, JSClass& thisObject, JSC::PropertyName propertyName)
42{
43 // FIXME: It seems unfortunate that have to do two lookups for the property name,
44 // one for isSupportedPropertyName and one by the user of this algorithm to access
45 // that property. It would be nice if we could smuggle the result, or an iterator
46 // out so the duplicate lookup could be avoided.
47
48 // NOTE: While it is not specified, a Symbol can never be a 'supported property
49 // name' so we check that first.
50 if (propertyName.isSymbol())
51 return false;
52
53 auto& impl = thisObject.wrapped();
54
55 // 1. If P is not a supported property name of O, then return false.
56 if (!impl.isSupportedPropertyName(propertyNameToString(propertyName)))
57 return false;
58
59 // 2. If O has an own property named P, then return false.
60 JSC::PropertySlot slot { &thisObject, JSC::PropertySlot::InternalMethodType::VMInquiry, &lexicalGlobalObject.vm() };
61 if (JSC::JSObject::getOwnPropertySlot(&thisObject, &lexicalGlobalObject, propertyName, slot))
62 return false;
63
64 // 3. If O implements an interface that has the [LegacyOverrideBuiltIns] extended attribute, then return true.
65 if (overrideBuiltins == LegacyOverrideBuiltIns::Yes)
66 return true;
67
68 // 4. Initialize prototype to be the value of the internal [[Prototype]] property of O.
69 // 5. While prototype is not null:
70 // 1. If prototype is not a named properties object, and prototype has an own property named P, then return false.
71 // FIXME: Implement checking for 'named properties object'.
72 // 2. Set prototype to be the value of the internal [[Prototype]] property of prototype.
73 auto prototype = thisObject.getPrototypeDirect(JSC::getVM(&lexicalGlobalObject));
74 if (prototype.isObject() && JSC::asObject(prototype)->getPropertySlot(&lexicalGlobalObject, propertyName, slot))
75 return false;
76
77 // 6. Return true.
78 return true;
79}

81// NOTE: Named getters are little odd. To avoid doing duplicate lookups (once when checking if
82// the property name is a 'supported property name' and once to get the value) we signal
83// that a property is supported by whether or not it is 'null' (where what null means is
84// dependant on the IDL type). This is based on the assumption that no named getter will
85// ever actually want to return null as an actual return value, which seems like an ok
86// assumption to make (should it turn out this doesn't hold in the future, we have lots
87// of options; do two lookups, add an extra layer of Optional, etc.).
88template<typename IDLType, typename JSClass, typename InnerItemAccessor>
89static decltype(auto) visibleNamedPropertyItemAccessorFunctor(InnerItemAccessor&& innerItemAccessor)
90{
91 if constexpr (IsExceptionOr<std::invoke_result_t<InnerItemAccessor, JSClass&, JSC::PropertyName>>) {
92 using ReturnType = ExceptionOr<typename IDLType::ImplementationType>;
93
94 return [innerItemAccessor = std::forward<InnerItemAccessor>(innerItemAccessor)] (JSClass& thisObject, JSC::PropertyName propertyName) -> std::optional<ReturnType> {
95 auto result = innerItemAccessor(thisObject, propertyName);
96 if (result.hasException())
97 return ReturnType { result.releaseException() };
98 if (!IDLType::isNullValue(result.returnValue()))
99 return ReturnType { IDLType::extractValueFromNullable(result.releaseReturnValue()) };
100 return std::nullopt;
101 };
102 } else {
103 using ReturnType = typename IDLType::ImplementationType;
104
105 return [innerItemAccessor = std::forward<InnerItemAccessor>(innerItemAccessor)] (JSClass& thisObject, JSC::PropertyName propertyName) -> std::optional<ReturnType> {
106 auto result = innerItemAccessor(thisObject, propertyName);
107 if (!IDLType::isNullValue(result))
108 return ReturnType { IDLType::extractValueFromNullable(result) };
109 return std::nullopt;
110 };
111 }
112}

114// An implementation of the 'named property visibility algorithm' augmented to replace the
115// 'supported property name' check with direct access to the implementation value returned
116// for the property name, via passed in functor. This allows us to avoid two looking up the
117// the property name twice; once for 'named property visibility algorithm' check, and then
118// again when the value is needed.
119template<LegacyOverrideBuiltIns overrideBuiltins, class JSClass, class ItemAccessor>
120static auto accessVisibleNamedProperty(JSC::JSGlobalObject& lexicalGlobalObject, JSClass& thisObject, JSC::PropertyName propertyName, ItemAccessor&& itemAccessor) -> decltype(itemAccessor(thisObject, propertyName))
121{
122 // NOTE: While it is not specified, a Symbol can never be a 'supported property
123 // name' so we check that first.
124 if (propertyName.isSymbol())
125 return std::nullopt;
126
127 // 1. If P is not a supported property name of O, then return false.
128 auto result = itemAccessor(thisObject, propertyName);
129 if (!result)
130 return std::nullopt;
131
132 // 2. If O has an own property named P, then return false.
133 JSC::PropertySlot slot { &thisObject, JSC::PropertySlot::InternalMethodType::VMInquiry, &lexicalGlobalObject.vm() };
134 if (JSC::JSObject::getOwnPropertySlot(&thisObject, &lexicalGlobalObject, propertyName, slot))
135 return std::nullopt;
136
137 // 3. If O implements an interface that has the [LegacyOverrideBuiltIns] extended attribute, then return true.
138 if (overrideBuiltins == LegacyOverrideBuiltIns::Yes && !worldForDOMObject(thisObject).shouldDisableLegacyOverrideBuiltInsBehavior())
139 return result;
140
141 // 4. Initialize prototype to be the value of the internal [[Prototype]] property of O.
142 // 5. While prototype is not null:
143 // 1. If prototype is not a named properties object, and prototype has an own property named P, then return false.
144 // FIXME: Implement checking for 'named properties object'.
145 // 2. Set prototype to be the value of the internal [[Prototype]] property of prototype.
146 auto prototype = thisObject.getPrototypeDirect(JSC::getVM(&lexicalGlobalObject));
147 if (prototype.isObject() && JSC::asObject(prototype)->getPropertySlot(&lexicalGlobalObject, propertyName, slot))
148 return std::nullopt;
149
150 // 6. Return true.
151 return result;
152}

154// This implements steps 2.2 through 2.5 of https://heycam.github.io/webidl/#legacy-platform-object-delete.
155template<typename Functor> bool performLegacyPlatformObjectDeleteOperation(JSC::JSGlobalObject& lexicalGlobalObject, Functor&& functor)
156{
157 using ReturnType = std::invoke_result_t<Functor>;
158
159 if constexpr (IsExceptionOr<ReturnType>) {
160 auto result = functor();
161 if (result.hasException()) {
162 auto throwScope = DECLARE_THROW_SCOPE(JSC::getVM(&lexicalGlobalObject));
163 propagateException(lexicalGlobalObject, throwScope, result.releaseException());
164 return true;
165 }
166
167 if constexpr (std::is_same_v<ReturnType, ExceptionOr<bool>>)
168 return result.releaseReturnValue();
169 return true;
170 }
171
172 if constexpr (std::is_same_v<ReturnType, bool>)
173 return functor();
174
175 functor();
176 return true;
177}
