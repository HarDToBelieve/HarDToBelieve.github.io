26#include "JSDOMCastThisValue.h"
27#include "JSDOMExceptionHandling.h"

29namespace WebCore {
30
31template<typename JSClass>
32class IDLAttribute {
33public:
34 using Setter = bool(JSC::JSGlobalObject&, JSClass&, JSC::JSValue);
35 using SetterPassingPropertyName = bool(JSC::JSGlobalObject&, JSClass&, JSC::JSValue, JSC::PropertyName);
36 using StaticSetter = bool(JSC::JSGlobalObject&, JSC::JSValue);
37 using Getter = JSC::JSValue(JSC::JSGlobalObject&, JSClass&);
38 using GetterPassingPropertyName = JSC::JSValue(JSC::JSGlobalObject&, JSClass&, JSC::PropertyName);
39 using StaticGetter = JSC::JSValue(JSC::JSGlobalObject&);
40
41 template<Setter setter, CastedThisErrorBehavior shouldThrow = CastedThisErrorBehavior::Throw>
42 static bool set(JSC::JSGlobalObject& lexicalGlobalObject, JSC::EncodedJSValue thisValue, JSC::EncodedJSValue encodedValue, JSC::PropertyName attributeName)
43 {
44 auto throwScope = DECLARE_THROW_SCOPE(JSC::getVM(&lexicalGlobalObject));
45
46 auto* thisObject = castThisValue<JSClass>(lexicalGlobalObject, JSC::JSValue::decode(thisValue));
47 if (UNLIKELY(!thisObject)) {
48 if constexpr (shouldThrow == CastedThisErrorBehavior::Throw)
49 return JSC::throwVMDOMAttributeSetterTypeError(&lexicalGlobalObject, throwScope, JSClass::info(), attributeName);
50 else
51 return false;
52 }
53
54 RELEASE_AND_RETURN(throwScope, (setter(lexicalGlobalObject, *thisObject, JSC::JSValue::decode(encodedValue))));
55 }
56
57 // FIXME: FIXME: This can be merged with `set` if we replace the explicit setter function template parameter with a generic lambda.
58 template<SetterPassingPropertyName setter, CastedThisErrorBehavior shouldThrow = CastedThisErrorBehavior::Throw>
59 static bool setPassingPropertyName(JSC::JSGlobalObject& lexicalGlobalObject, JSC::EncodedJSValue thisValue, JSC::EncodedJSValue encodedValue, JSC::PropertyName attributeName)
60 {
61 auto throwScope = DECLARE_THROW_SCOPE(JSC::getVM(&lexicalGlobalObject));
62
63 auto* thisObject = castThisValue<JSClass>(lexicalGlobalObject, JSC::JSValue::decode(thisValue));
64 if (UNLIKELY(!thisObject)) {
65 if constexpr (shouldThrow == CastedThisErrorBehavior::Throw)
66 return JSC::throwVMDOMAttributeSetterTypeError(&lexicalGlobalObject, throwScope, JSClass::info(), attributeName);
67 else
68 return false;
69 }
70
71 RELEASE_AND_RETURN(throwScope, (setter(lexicalGlobalObject, *thisObject, JSC::JSValue::decode(encodedValue), attributeName)));
72 }
73
74 template<StaticSetter setter, CastedThisErrorBehavior shouldThrow = CastedThisErrorBehavior::Throw>
75 static bool setStatic(JSC::JSGlobalObject& lexicalGlobalObject, JSC::EncodedJSValue, JSC::EncodedJSValue encodedValue, JSC::PropertyName)
76 {
77 return setter(lexicalGlobalObject, JSC::JSValue::decode(encodedValue));
78 }
79
80 template<Getter getter, CastedThisErrorBehavior shouldThrow = CastedThisErrorBehavior::Throw>
81 static JSC::EncodedJSValue get(JSC::JSGlobalObject& lexicalGlobalObject, JSC::EncodedJSValue thisValue, JSC::PropertyName attributeName)
82 {
83 auto throwScope = DECLARE_THROW_SCOPE(JSC::getVM(&lexicalGlobalObject));
84
85 if constexpr (shouldThrow == CastedThisErrorBehavior::Assert) {
86 ASSERT(castThisValue<JSClass>(lexicalGlobalObject, JSC::JSValue::decode(thisValue)));
87 auto* thisObject = JSC::jsCast<JSClass*>(JSC::JSValue::decode(thisValue));
88 RELEASE_AND_RETURN(throwScope, (JSC::JSValue::encode(getter(lexicalGlobalObject, *thisObject))));
89 } else {
90 auto* thisObject = castThisValue<JSClass>(lexicalGlobalObject, JSC::JSValue::decode(thisValue));
91 if (UNLIKELY(!thisObject)) {
92 if constexpr (shouldThrow == CastedThisErrorBehavior::Throw)
93 return JSC::throwVMDOMAttributeGetterTypeError(&lexicalGlobalObject, throwScope, JSClass::info(), attributeName);
94 else if constexpr (shouldThrow == CastedThisErrorBehavior::RejectPromise)
95 RELEASE_AND_RETURN(throwScope, rejectPromiseWithGetterTypeError(lexicalGlobalObject, JSClass::info(), attributeName));
96 else
97 return JSC::JSValue::encode(JSC::jsUndefined());
98 }
99
100 RELEASE_AND_RETURN(throwScope, (JSC::JSValue::encode(getter(lexicalGlobalObject, *thisObject))));
101 }
102 }
103
104 // FIXME: This can be merged with `get` if we replace the explicit setter function template parameter with a generic lambda.
105 template<GetterPassingPropertyName getter, CastedThisErrorBehavior shouldThrow = CastedThisErrorBehavior::Throw>
106 static JSC::EncodedJSValue getPassingPropertyName(JSC::JSGlobalObject& lexicalGlobalObject, JSC::EncodedJSValue thisValue, JSC::PropertyName attributeName)
107 {
108 auto throwScope = DECLARE_THROW_SCOPE(JSC::getVM(&lexicalGlobalObject));
109
110 if constexpr (shouldThrow == CastedThisErrorBehavior::Assert) {
111 ASSERT(castThisValue<JSClass>(lexicalGlobalObject, JSC::JSValue::decode(thisValue)));
112 auto* thisObject = JSC::jsCast<JSClass*>(JSC::JSValue::decode(thisValue));
113 RELEASE_AND_RETURN(throwScope, (JSC::JSValue::encode(getter(lexicalGlobalObject, *thisObject, attributeName))));
114 } else {
115 auto* thisObject = castThisValue<JSClass>(lexicalGlobalObject, JSC::JSValue::decode(thisValue));
116 if (UNLIKELY(!thisObject)) {
117 if constexpr (shouldThrow == CastedThisErrorBehavior::Throw)
118 return JSC::throwVMDOMAttributeGetterTypeError(&lexicalGlobalObject, throwScope, JSClass::info(), attributeName);
119 else if constexpr (shouldThrow == CastedThisErrorBehavior::RejectPromise)
120 RELEASE_AND_RETURN(throwScope, rejectPromiseWithGetterTypeError(lexicalGlobalObject, JSClass::info(), attributeName));
121 else
122 return JSC::JSValue::encode(JSC::jsUndefined());
123 }
124
125 RELEASE_AND_RETURN(throwScope, (JSC::JSValue::encode(getter(lexicalGlobalObject, *thisObject, attributeName))));
126 }
127 }
128
129 template<StaticGetter getter, CastedThisErrorBehavior shouldThrow = CastedThisErrorBehavior::Throw>
130 static JSC::EncodedJSValue getStatic(JSC::JSGlobalObject& lexicalGlobalObject, JSC::EncodedJSValue, JSC::PropertyName)
131 {
132 return JSC::JSValue::encode(getter(lexicalGlobalObject));
133 }
134};
