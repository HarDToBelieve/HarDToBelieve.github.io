28template<typename JSClass> class JSDOMBuiltinConstructor final : public JSDOMBuiltinConstructorBase {
29public:
30 using Base = JSDOMBuiltinConstructorBase;
31 static const unsigned StructureFlags;
32
33 static JSDOMBuiltinConstructor* create(JSC::VM&, JSC::Structure*, JSDOMGlobalObject&);
34 static JSC::Structure* createStructure(JSC::VM&, JSC::JSGlobalObject&, JSC::JSValue prototype);
35
36 DECLARE_INFO;
37
38 // Usually defined for each specialization class.
39 static JSC::JSValue prototypeForStructure(JSC::VM&, const JSDOMGlobalObject&);
40
41 static JSC::EncodedJSValue JSC_HOST_CALL_ATTRIBUTES construct(JSC::JSGlobalObject*, JSC::CallFrame*);
42
43private:
44 JSDOMBuiltinConstructor(JSC::Structure* structure, JSDOMGlobalObject& globalObject)
45 : Base(structure, globalObject)
46 {
47 }
48
49 void finishCreation(JSC::VM&, JSDOMGlobalObject&);
50 static JSC::CallData getConstructData(JSC::JSCell*);
51
52 JSC::EncodedJSValue callConstructor(JSC::JSGlobalObject&, JSC::CallFrame&, JSC::JSObject&);
53 JSC::EncodedJSValue callConstructor(JSC::JSGlobalObject&, JSC::CallFrame&, JSC::JSObject*);
54
55 // Usually defined for each specialization class.
56 void initializeProperties(JSC::VM&, JSDOMGlobalObject&) { }
57 // Must be defined for each specialization class.
58 JSC::FunctionExecutable* initializeExecutable(JSC::VM&);
59};

61template<typename JSClass> inline JSDOMBuiltinConstructor<JSClass>* JSDOMBuiltinConstructor<JSClass>::create(JSC::VM& vm, JSC::Structure* structure, JSDOMGlobalObject& globalObject)
62{
63 JSDOMBuiltinConstructor* constructor = new (NotNull, JSC::allocateCell<JSDOMBuiltinConstructor>(vm.heap)) JSDOMBuiltinConstructor(structure, globalObject);
64 constructor->finishCreation(vm, globalObject);
65 return constructor;
66}
67
68template<typename JSClass> inline JSC::Structure* JSDOMBuiltinConstructor<JSClass>::createStructure(JSC::VM& vm, JSC::JSGlobalObject& globalObject, JSC::JSValue prototype)
69{
70 return JSC::Structure::create(vm, &globalObject, prototype, JSC::TypeInfo(JSC::ObjectType, StructureFlags), info());
71}
72
73template<typename JSClass> inline void JSDOMBuiltinConstructor<JSClass>::finishCreation(JSC::VM& vm, JSDOMGlobalObject& globalObject)
74{
75 Base::finishCreation(vm);
76 ASSERT(inherits(vm, info()));
77 setInitializeFunction(vm, *JSC::JSFunction::create(vm, initializeExecutable(vm), &globalObject));
78 initializeProperties(vm, globalObject);
79}
80
81template<typename JSClass> inline JSC::EncodedJSValue JSDOMBuiltinConstructor<JSClass>::callConstructor(JSC::JSGlobalObject& lexicalGlobalObject, JSC::CallFrame& callFrame, JSC::JSObject& object)
82{
83 Base::callFunctionWithCurrentArguments(lexicalGlobalObject, callFrame, object, *initializeFunction());
84 return JSC::JSValue::encode(&object);
85}
86
87template<typename JSClass> inline JSC::EncodedJSValue JSDOMBuiltinConstructor<JSClass>::callConstructor(JSC::JSGlobalObject& lexicalGlobalObject, JSC::CallFrame& callFrame, JSC::JSObject* object)
88{
89 JSC::VM& vm = JSC::getVM(&lexicalGlobalObject);
90 auto scope = DECLARE_THROW_SCOPE(vm);
91 if (!object)
92 return throwConstructorScriptExecutionContextUnavailableError(lexicalGlobalObject, scope, info()->className);
93 return callConstructor(lexicalGlobalObject, callFrame, *object);
94}

96template<typename JSClass>
97typename std::enable_if<JSDOMObjectInspector<JSClass>::isSimpleWrapper, JSC::JSObject&>::type createJSObject(JSDOMBuiltinConstructor<JSClass>& constructor)
98{
99 return *createWrapper<typename JSClass::DOMWrapped>(constructor.globalObject(), JSClass::DOMWrapped::create());
100}
101
102template<typename JSClass>
103typename std::enable_if<JSDOMObjectInspector<JSClass>::isBuiltin, JSC::JSObject&>::type createJSObject(JSDOMBuiltinConstructor<JSClass>& constructor)
104{
105 auto& globalObject = *constructor.globalObject();
106 return *JSClass::create(getDOMStructure<JSClass>(globalObject.vm(), globalObject), &globalObject);
107}
108
109template<typename JSClass>
110typename std::enable_if<JSDOMObjectInspector<JSClass>::isComplexWrapper, JSC::JSObject*>::type createJSObject(JSDOMBuiltinConstructor<JSClass>& constructor)
111{
112 auto* context = constructor.scriptExecutionContext();
113 return context ? createWrapper<typename JSClass::DOMWrapped>(constructor.globalObject(), JSClass::DOMWrapped::create(*context)) : nullptr;
114}
115
116template<typename JSClass> inline JSC::EncodedJSValue JSC_HOST_CALL_ATTRIBUTES JSDOMBuiltinConstructor<JSClass>::construct(JSC::JSGlobalObject* lexicalGlobalObject, JSC::CallFrame* callFrame)
117{
118 ASSERT(callFrame);
119 auto* castedThis = JSC::jsCast<JSDOMBuiltinConstructor*>(callFrame->jsCallee());
120 return castedThis->callConstructor(*lexicalGlobalObject, *callFrame, createJSObject(*castedThis));
121}
122
123template<typename JSClass> inline JSC::CallData JSDOMBuiltinConstructor<JSClass>::getConstructData(JSC::JSCell*)
124{
125 JSC::CallData constructData;
126 constructData.type = JSC::CallData::Type::Native;
127 constructData.native.function = construct;
128 return constructData;
129}
