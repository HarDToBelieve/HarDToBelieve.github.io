28#include "JSDOMWrapper.h"
29#include "WebCoreJSClientData.h"

31namespace WebCore {
32
33template<typename JSClass> class JSDOMConstructorNotCallable : public JSDOMObject {
34public:
35 using Base = JSDOMObject;
36 static const unsigned StructureFlags;
37
38 static constexpr bool needsDestruction = false;
39 static JSDOMConstructorNotCallable* create(JSC::VM&, JSC::Structure*, JSDOMGlobalObject&);
40 static JSC::Structure* createStructure(JSC::VM&, JSC::JSGlobalObject&, JSC::JSValue prototype);
41
42 DECLARE_INFO;
43
44 static JSC::JSValue prototypeForStructure(JSC::VM&, const JSDOMGlobalObject&);
45
46 template<typename CellType, JSC::SubspaceAccess>
47 static JSC::IsoSubspace* subspaceFor(JSC::VM& vm)
48 {
49 static_assert(sizeof(CellType) == sizeof(JSDOMConstructorNotCallable));
50 STATIC_ASSERT_ISO_SUBSPACE_SHARABLE(CellType, JSDOMConstructorNotCallable);
51 static_assert(CellType::destroy == JSC::JSCell::destroy, "JSDOMConstructorNotCallable<JSClass> is not destructible actually");
52 return &static_cast<JSVMClientData*>(vm.clientData)->domConstructorSpace();
53 }
54
55private:
56 JSDOMConstructorNotCallable(JSC::Structure* structure, JSDOMGlobalObject& globalObject)
57 : Base(structure, globalObject)
58 {
59 }
60
61 void finishCreation(JSC::VM&, JSDOMGlobalObject&);
62 void initializeProperties(JSC::VM&, JSDOMGlobalObject&) { }
63};
64
65template<typename JSClass> inline JSDOMConstructorNotCallable<JSClass>* JSDOMConstructorNotCallable<JSClass>::create(JSC::VM& vm, JSC::Structure* structure, JSDOMGlobalObject& globalObject)
66{
67 JSDOMConstructorNotCallable* constructor = new (NotNull, JSC::allocateCell<JSDOMConstructorNotCallable>(vm.heap)) JSDOMConstructorNotCallable(structure, globalObject);
68 constructor->finishCreation(vm, globalObject);
69 return constructor;
70}
71
72template<typename JSClass> inline JSC::Structure* JSDOMConstructorNotCallable<JSClass>::createStructure(JSC::VM& vm, JSC::JSGlobalObject& globalObject, JSC::JSValue prototype)
73{
74 return JSC::Structure::create(vm, &globalObject, prototype, JSC::TypeInfo(JSC::ObjectType, StructureFlags), info());
75}
76
77template<typename JSClass> inline void JSDOMConstructorNotCallable<JSClass>::finishCreation(JSC::VM& vm, JSDOMGlobalObject& globalObject)
78{
79 Base::finishCreation(vm);
80 ASSERT(inherits(vm, info()));
81 initializeProperties(vm, globalObject);
82}
