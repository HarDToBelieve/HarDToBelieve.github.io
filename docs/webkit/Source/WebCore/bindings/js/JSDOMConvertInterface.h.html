34template<typename ImplementationClass> struct JSDOMWrapperConverterTraits;
35
36template<typename T, typename Enable = void>
37struct JSToWrappedOverloader {
38 using ReturnType = typename JSDOMWrapperConverterTraits<T>::ToWrappedReturnType;
39 using WrapperType = typename JSDOMWrapperConverterTraits<T>::WrapperClass;
40
41 static ReturnType toWrapped(JSC::JSGlobalObject& lexicalGlobalObject, JSC::JSValue value)
42 {
43 return WrapperType::toWrapped(JSC::getVM(&lexicalGlobalObject), value);
44 }
45};
46
47template<typename T>
48struct JSToWrappedOverloader<T, typename std::enable_if<JSDOMWrapperConverterTraits<T>::needsState>::type> {
49 using ReturnType = typename JSDOMWrapperConverterTraits<T>::ToWrappedReturnType;
50 using WrapperType = typename JSDOMWrapperConverterTraits<T>::WrapperClass;
51
52 static ReturnType toWrapped(JSC::JSGlobalObject& lexicalGlobalObject, JSC::JSValue value)
53 {
54 return WrapperType::toWrapped(lexicalGlobalObject, value);
55 }
56};
57
58template<typename T> struct Converter<IDLInterface<T>> : DefaultConverter<IDLInterface<T>> {
59 using ReturnType = typename JSDOMWrapperConverterTraits<T>::ToWrappedReturnType;
60 using WrapperType = typename JSDOMWrapperConverterTraits<T>::WrapperClass;
61
62 template<typename ExceptionThrower = DefaultExceptionThrower>
63 static ReturnType convert(JSC::JSGlobalObject& lexicalGlobalObject, JSC::JSValue value, ExceptionThrower&& exceptionThrower = ExceptionThrower())
64 {
65 auto& vm = JSC::getVM(&lexicalGlobalObject);
66 auto scope = DECLARE_THROW_SCOPE(vm);
67 ReturnType object = JSToWrappedOverloader<T>::toWrapped(lexicalGlobalObject, value);
68 if (UNLIKELY(!object))
69 exceptionThrower(lexicalGlobalObject, scope);
70 return object;
71 }
72};
73
74template<typename T> struct JSConverter<IDLInterface<T>> {
75 static constexpr bool needsState = true;
76 static constexpr bool needsGlobalObject = true;
77
78 template <typename U>
79 static JSC::JSValue convert(JSC::JSGlobalObject& lexicalGlobalObject, JSDOMGlobalObject& globalObject, const U& value)
80 {
81 return toJS(&lexicalGlobalObject, &globalObject, Detail::getPtrOrRef(value));
82 }
83
84 template<typename U>
85 static JSC::JSValue convertNewlyCreated(JSC::JSGlobalObject& lexicalGlobalObject, JSDOMGlobalObject& globalObject, U&& value)
86 {
87 return toJSNewlyCreated(&lexicalGlobalObject, &globalObject, std::forward<U>(value));
88 }
89};
90
91template<typename T> struct VariadicConverter<IDLInterface<T>> {
92 using Item = std::reference_wrapper<T>;
93
94 static std::optional<Item> convert(JSC::JSGlobalObject& lexicalGlobalObject, JSC::JSValue value)
95 {
96 auto* result = Converter<IDLInterface<T>>::convert(lexicalGlobalObject, value);
97 if (!result)
98 return std::nullopt;
99 return std::optional<Item> { *result };
100 }
101};
