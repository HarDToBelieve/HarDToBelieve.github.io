280template<> struct Converter<IDLFloat> : DefaultConverter<IDLFloat> {
281
282 static inline float convert(JSC::JSGlobalObject& lexicalGlobalObject, JSC::ThrowScope& scope, double number)
283 {
284 if (UNLIKELY(!std::isfinite(number)))
285 throwNonFiniteTypeError(lexicalGlobalObject, scope);
286 return static_cast<float>(number);
287 }
288
289 static float convert(JSC::JSGlobalObject& lexicalGlobalObject, JSC::JSValue value)
290 {
291 JSC::VM& vm = JSC::getVM(&lexicalGlobalObject);
292 auto scope = DECLARE_THROW_SCOPE(vm);
293 double number = value.toNumber(&lexicalGlobalObject);
294 RETURN_IF_EXCEPTION(scope, 0.0);
295 if (UNLIKELY(number < std::numeric_limits<float>::lowest() || number > std::numeric_limits<float>::max()))
296 throwTypeError(&lexicalGlobalObject, scope, "The provided value is outside the range of a float");
297 if (UNLIKELY(!std::isfinite(number)))
298 throwNonFiniteTypeError(lexicalGlobalObject, scope);
299 return static_cast<float>(number);
300 }
301};

315template<> struct Converter<IDLUnrestrictedFloat> : DefaultConverter<IDLUnrestrictedFloat> {
316 static inline float convert(JSC::JSGlobalObject&, JSC::ThrowScope&, double number)
317 {
318 return static_cast<float>(number);
319 }
320
321 static float convert(JSC::JSGlobalObject& lexicalGlobalObject, JSC::JSValue value)
322 {
323 JSC::VM& vm = JSC::getVM(&lexicalGlobalObject);
324 auto scope = DECLARE_THROW_SCOPE(vm);
325 double number = value.toNumber(&lexicalGlobalObject);
326 RETURN_IF_EXCEPTION(scope, 0.0);
327
328 if (UNLIKELY(number < std::numeric_limits<float>::lowest()))
329 return -std::numeric_limits<float>::infinity();
330 if (UNLIKELY(number > std::numeric_limits<float>::max()))
331 return std::numeric_limits<float>::infinity();
332 return static_cast<float>(number);
333 }
334};
