41template<typename IDLType>
42struct GenericSequenceConverter {
43 using ReturnType = Vector<typename IDLType::SequenceStorageType>;
44
45 static ReturnType convert(JSC::JSGlobalObject& lexicalGlobalObject, JSC::JSObject* object)
46 {
47 return convert(lexicalGlobalObject, object, ReturnType());
48 }
49
50 static ReturnType convert(JSC::JSGlobalObject& lexicalGlobalObject, JSC::JSObject* object, ReturnType&& result)
51 {
52 forEachInIterable(&lexicalGlobalObject, object, [&result](JSC::VM& vm, JSC::JSGlobalObject* lexicalGlobalObject, JSC::JSValue nextValue) {
53 auto scope = DECLARE_THROW_SCOPE(vm);
54
55 auto convertedValue = Converter<IDLType>::convert(*lexicalGlobalObject, nextValue);
56 if (UNLIKELY(scope.exception()))
57 return;
58 result.append(WTFMove(convertedValue));
59 });
60 return WTFMove(result);
61 }
62
63 static ReturnType convert(JSC::JSGlobalObject& lexicalGlobalObject, JSC::JSObject* object, JSC::JSValue method)
64 {
65 return convert(lexicalGlobalObject, object, method, ReturnType());
66 }
67
68 static ReturnType convert(JSC::JSGlobalObject& lexicalGlobalObject, JSC::JSObject* object, JSC::JSValue method, ReturnType&& result)
69 {
70 forEachInIterable(lexicalGlobalObject, object, method, [&result](JSC::VM& vm, JSC::JSGlobalObject& lexicalGlobalObject, JSC::JSValue nextValue) {
71 auto scope = DECLARE_THROW_SCOPE(vm);
72
73 auto convertedValue = Converter<IDLType>::convert(lexicalGlobalObject, nextValue);
74 if (UNLIKELY(scope.exception()))
75 return;
76 result.append(WTFMove(convertedValue));
77 });
78 return WTFMove(result);
79 }
80};

82// Specialization for numeric types
83// FIXME: This is only implemented for the IDLFloatingPointTypes and IDLLong. To add
84// support for more numeric types, add an overload of Converter<IDLType>::convert that
85// takes a JSGlobalObject, ThrowScope and double as its arguments.
86template<typename IDLType>
87struct NumericSequenceConverter {
88 using GenericConverter = GenericSequenceConverter<IDLType>;
89 using ReturnType = typename GenericConverter::ReturnType;
90
91 static ReturnType convertArray(JSC::JSGlobalObject& lexicalGlobalObject, JSC::ThrowScope& scope, JSC::JSArray* array, unsigned length, JSC::IndexingType indexingType, ReturnType&& result)
92 {
93 if (indexingType == JSC::Int32Shape) {
94 for (unsigned i = 0; i < length; i++) {
95 auto indexValue = array->butterfly()->contiguousInt32().at(array, i).get();
96 ASSERT(!indexValue || indexValue.isInt32());
97 if (!indexValue)
98 result.uncheckedAppend(0);
99 else
100 result.uncheckedAppend(indexValue.asInt32());
101 }
102 return WTFMove(result);
103 }
104
105 ASSERT(indexingType == JSC::DoubleShape);
106 for (unsigned i = 0; i < length; i++) {
107 double doubleValue = array->butterfly()->contiguousDouble().at(array, i);
108 if (std::isnan(doubleValue))
109 result.uncheckedAppend(0);
110 else {
111 auto convertedValue = Converter<IDLType>::convert(lexicalGlobalObject, scope, doubleValue);
112 RETURN_IF_EXCEPTION(scope, { });
113
114 result.uncheckedAppend(convertedValue);
115 }
116 }
117 return WTFMove(result);
118 }
119
120 static ReturnType convert(JSC::JSGlobalObject& lexicalGlobalObject, JSC::JSValue value)
121 {
122 auto& vm = JSC::getVM(&lexicalGlobalObject);
123 auto scope = DECLARE_THROW_SCOPE(vm);
124
125 if (!value.isObject()) {
126 throwSequenceTypeError(lexicalGlobalObject, scope);
127 return { };
128 }
129
130 JSC::JSObject* object = JSC::asObject(value);
131 if (!JSC::isJSArray(object))
132 RELEASE_AND_RETURN(scope, GenericConverter::convert(lexicalGlobalObject, object));
133
134 JSC::JSArray* array = JSC::asArray(object);
135 if (!array->isIteratorProtocolFastAndNonObservable())
136 RELEASE_AND_RETURN(scope, GenericConverter::convert(lexicalGlobalObject, object));
137
138 unsigned length = array->length();
139 ReturnType result;
140 // If we're not an int32/double array, it's possible that converting a
141 // JSValue to a number could cause the iterator protocol to change, hence,
142 // we may need more capacity, or less. In such cases, we use the length
143 // as a proxy for the capacity we will most likely need (it's unlikely that
144 // a program is written with a valueOf that will augment the iterator protocol).
145 // If we are an int32/double array, then length is precisely the capacity we need.
146 if (!result.tryReserveCapacity(length)) {
147 // FIXME: Is the right exception to throw?
148 throwTypeError(&lexicalGlobalObject, scope);
149 return { };
150 }
151
152 JSC::IndexingType indexingType = array->indexingType() & JSC::IndexingShapeMask;
153 if (indexingType != JSC::Int32Shape && indexingType != JSC::DoubleShape)
154 RELEASE_AND_RETURN(scope, GenericConverter::convert(lexicalGlobalObject, object, WTFMove(result)));
155
156 return convertArray(lexicalGlobalObject, scope, array, length, indexingType, WTFMove(result));
157 }
158
159 static ReturnType convert(JSC::JSGlobalObject& lexicalGlobalObject, JSC::JSObject* object, JSC::JSValue method)
160 {
161 auto& vm = JSC::getVM(&lexicalGlobalObject);
162 auto scope = DECLARE_THROW_SCOPE(vm);
163
164 if (!JSC::isJSArray(object))
165 RELEASE_AND_RETURN(scope, GenericConverter::convert(lexicalGlobalObject, object, method));
166
167 JSC::JSArray* array = JSC::asArray(object);
168 if (!array->isIteratorProtocolFastAndNonObservable())
169 RELEASE_AND_RETURN(scope, GenericConverter::convert(lexicalGlobalObject, object, method));
170
171 unsigned length = array->length();
172 ReturnType result;
173 // If we're not an int32/double array, it's possible that converting a
174 // JSValue to a number could cause the iterator protocol to change, hence,
175 // we may need more capacity, or less. In such cases, we use the length
176 // as a proxy for the capacity we will most likely need (it's unlikely that
177 // a program is written with a valueOf that will augment the iterator protocol).
178 // If we are an int32/double array, then length is precisely the capacity we need.
179 if (!result.tryReserveCapacity(length)) {
180 // FIXME: Is the right exception to throw?
181 throwTypeError(&lexicalGlobalObject, scope);
182 return { };
183 }
184
185 JSC::IndexingType indexingType = array->indexingType() & JSC::IndexingShapeMask;
186 if (indexingType != JSC::Int32Shape && indexingType != JSC::DoubleShape)
187 RELEASE_AND_RETURN(scope, GenericConverter::convert(lexicalGlobalObject, object, method, WTFMove(result)));
188
189 return convertArray(lexicalGlobalObject, scope, array, length, indexingType, WTFMove(result));
190 }

193template<typename IDLType>
194struct SequenceConverter {
195 using GenericConverter = GenericSequenceConverter<IDLType>;
196 using ReturnType = typename GenericConverter::ReturnType;
197
198 static ReturnType convertArray(JSC::JSGlobalObject& lexicalGlobalObject, JSC::JSArray* array)
199 {
200 auto& vm = lexicalGlobalObject.vm();
201 auto scope = DECLARE_THROW_SCOPE(vm);
202 unsigned length = array->length();
203
204 ReturnType result;
205 if (!result.tryReserveCapacity(length)) {
206 // FIXME: Is the right exception to throw?
207 throwTypeError(&lexicalGlobalObject, scope);
208 return { };
209 }
210
211 JSC::IndexingType indexingType = array->indexingType() & JSC::IndexingShapeMask;
212
213 if (indexingType == JSC::ContiguousShape) {
214 for (unsigned i = 0; i < length; i++) {
215 auto indexValue = array->butterfly()->contiguous().at(array, i).get();
216 if (!indexValue)
217 indexValue = JSC::jsUndefined();
218
219 auto convertedValue = Converter<IDLType>::convert(lexicalGlobalObject, indexValue);
220 RETURN_IF_EXCEPTION(scope, { });
221
222 result.uncheckedAppend(convertedValue);
223 }
224 return result;
225 }
226
227 for (unsigned i = 0; i < length; i++) {
228 auto indexValue = array->getDirectIndex(&lexicalGlobalObject, i);
229 RETURN_IF_EXCEPTION(scope, { });
230
231 if (!indexValue)
232 indexValue = JSC::jsUndefined();
233
234 auto convertedValue = Converter<IDLType>::convert(lexicalGlobalObject, indexValue);
235 RETURN_IF_EXCEPTION(scope, { });
236
237 result.uncheckedAppend(convertedValue);
238 }
239 return result;
240 }
241
242 static ReturnType convert(JSC::JSGlobalObject& lexicalGlobalObject, JSC::JSValue value)
243 {
244 auto& vm = JSC::getVM(&lexicalGlobalObject);
245 auto scope = DECLARE_THROW_SCOPE(vm);
246
247 if (!value.isObject()) {
248 throwSequenceTypeError(lexicalGlobalObject, scope);
249 return { };
250 }
251
252 JSC::JSObject* object = JSC::asObject(value);
253 if (Converter<IDLType>::conversionHasSideEffects)
254 RELEASE_AND_RETURN(scope, (GenericConverter::convert(lexicalGlobalObject, object)));
255
256 if (!JSC::isJSArray(object))
257 RELEASE_AND_RETURN(scope, (GenericConverter::convert(lexicalGlobalObject, object)));
258
259 JSC::JSArray* array = JSC::asArray(object);
260 if (!array->isIteratorProtocolFastAndNonObservable())
261 RELEASE_AND_RETURN(scope, (GenericConverter::convert(lexicalGlobalObject, object)));
262
263 RELEASE_AND_RETURN(scope, (convertArray(lexicalGlobalObject, array)));
264 }
265
266 static ReturnType convert(JSC::JSGlobalObject& lexicalGlobalObject, JSC::JSObject* object, JSC::JSValue method)
267 {
268 if (Converter<IDLType>::conversionHasSideEffects)
269 return GenericConverter::convert(lexicalGlobalObject, object, method);
270
271 if (!JSC::isJSArray(object))
272 return GenericConverter::convert(lexicalGlobalObject, object, method);
273
274 JSC::JSArray* array = JSC::asArray(object);
275 if (!array->isIteratorProtocolFastAndNonObservable())
276 return GenericConverter::convert(lexicalGlobalObject, object, method);
277
278 return convertArray(lexicalGlobalObject, array);
279 }

373template<typename T> struct JSConverter<IDLSequence<T>> {
374 static constexpr bool needsState = true;
375 static constexpr bool needsGlobalObject = true;
376
377 template<typename U, size_t inlineCapacity>
378 static JSC::JSValue convert(JSC::JSGlobalObject& lexicalGlobalObject, JSDOMGlobalObject& globalObject, const Vector<U, inlineCapacity>& vector)
379 {
380 JSC::VM& vm = JSC::getVM(&lexicalGlobalObject);
381 auto scope = DECLARE_THROW_SCOPE(vm);
382 JSC::MarkedArgumentBuffer list;
383 for (auto& element : vector) {
384 auto jsValue = toJS<T>(lexicalGlobalObject, globalObject, element);
385 RETURN_IF_EXCEPTION(scope, { });
386 list.append(jsValue);
387 }
388 if (UNLIKELY(list.hasOverflowed())) {
389 throwOutOfMemoryError(&lexicalGlobalObject, scope);
390 return { };
391 }
392 RELEASE_AND_RETURN(scope, JSC::constructArray(&globalObject, static_cast<JSC::ArrayAllocationProfile*>(nullptr), list));
393 }
394};

411 static constexpr bool needsState = true;
412 static constexpr bool needsGlobalObject = true;
413
414 template<typename U, size_t inlineCapacity>
415 static JSC::JSValue convert(JSC::JSGlobalObject& lexicalGlobalObject, JSDOMGlobalObject& globalObject, const Vector<U, inlineCapacity>& vector)
416 {
417 JSC::VM& vm = JSC::getVM(&lexicalGlobalObject);
418 auto scope = DECLARE_THROW_SCOPE(vm);
419 JSC::MarkedArgumentBuffer list;
420 for (auto& element : vector) {
421 auto jsValue = toJS<T>(lexicalGlobalObject, globalObject, element);
422 RETURN_IF_EXCEPTION(scope, { });
423 list.append(jsValue);
424 }
425 if (UNLIKELY(list.hasOverflowed())) {
426 throwOutOfMemoryError(&lexicalGlobalObject, scope);
427 return { };
428 }
429 auto* array = JSC::constructArray(&globalObject, static_cast<JSC::ArrayAllocationProfile*>(nullptr), list);
430 RETURN_IF_EXCEPTION(scope, { });
431 RELEASE_AND_RETURN(scope, JSC::objectConstructorFreeze(&lexicalGlobalObject, array));
432 }
