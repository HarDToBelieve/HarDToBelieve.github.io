79// FIXME: This should use the IDLUnion JSConverter.
80JSValue convertToJSValue(JSGlobalObject& lexicalGlobalObject, JSDOMGlobalObject& globalObject, const WebGLAny& any)
81{
82 return WTF::switchOn(any,
83 [] (std::nullptr_t) {
84 return jsNull();
85 },
86 [] (bool value) {
87 return jsBoolean(value);
88 },
89 [] (int value) {
90 return jsNumber(value);
91 },
92 [] (unsigned value) {
93 return jsNumber(value);
94 },
95 [] (long long value) {
96 return jsNumber(value);
97 },
98 [] (float value) {
99 return jsNumber(value);
100 },
101 [&] (const String& value) {
102 return jsStringWithCache(lexicalGlobalObject.vm(), value);
103 },
104 [&] (const Vector<bool>& values) {
105 MarkedArgumentBuffer list;
106 for (auto& value : values)
107 list.append(jsBoolean(value));
108 RELEASE_ASSERT(!list.hasOverflowed());
109 return constructArray(&globalObject, static_cast<JSC::ArrayAllocationProfile*>(nullptr), list);
110 },
111 [&] (const Vector<int>& values) {
112 MarkedArgumentBuffer list;
113 for (auto& value : values)
114 list.append(jsNumber(value));
115 RELEASE_ASSERT(!list.hasOverflowed());
116 return constructArray(&globalObject, static_cast<JSC::ArrayAllocationProfile*>(nullptr), list);
117 },
118 [&] (const Vector<unsigned>& values) {
119 MarkedArgumentBuffer list;
120 for (auto& value : values)
121 list.append(jsNumber(value));
122 RELEASE_ASSERT(!list.hasOverflowed());
123 return constructArray(&globalObject, static_cast<JSC::ArrayAllocationProfile*>(nullptr), list);
124 },
125 [&] (const RefPtr<Float32Array>& array) {
126 return toJS(&lexicalGlobalObject, &globalObject, array.get());
127 },
128 [&] (const RefPtr<Int32Array>& array) {
129 return toJS(&lexicalGlobalObject, &globalObject, array.get());
130 },
131 [&] (const RefPtr<Uint8Array>& array) {
132 return toJS(&lexicalGlobalObject, &globalObject, array.get());
133 },
134 [&] (const RefPtr<Uint32Array>& array) {
135 return toJS(&lexicalGlobalObject, &globalObject, array.get());
136 },
137 [&] (const RefPtr<WebGLBuffer>& buffer) {
138 return toJS(&lexicalGlobalObject, &globalObject, buffer.get());
139 },
140 [&] (const RefPtr<WebGLFramebuffer>& buffer) {
141 return toJS(&lexicalGlobalObject, &globalObject, buffer.get());
142 },
143 [&] (const RefPtr<WebGLProgram>& program) {
144 return toJS(&lexicalGlobalObject, &globalObject, program.get());
145 },
146 [&] (const RefPtr<WebGLRenderbuffer>& buffer) {
147 return toJS(&lexicalGlobalObject, &globalObject, buffer.get());
148 },
149 [&] (const RefPtr<WebGLTexture>& texture) {
150 return toJS(&lexicalGlobalObject, &globalObject, texture.get());
151 },
152 [&] (const RefPtr<WebGLVertexArrayObjectOES>& array) {
153 return toJS(&lexicalGlobalObject, &globalObject, array.get());
154 }
155#if ENABLE(WEBGL2)
156 ,
157 [&] (const RefPtr<WebGLSampler>& sampler) {
158 return toJS(&lexicalGlobalObject, &globalObject, sampler.get());
159 },
160 [&] (const RefPtr<WebGLTransformFeedback>& transformFeedback) {
161 return toJS(&lexicalGlobalObject, &globalObject, transformFeedback.get());
162 },
163 [&] (const RefPtr<WebGLVertexArrayObject>& array) {
164 return toJS(&lexicalGlobalObject, &globalObject, array.get());
165 }
166#endif
167 );
168}

170JSValue convertToJSValue(JSGlobalObject& lexicalGlobalObject, JSDOMGlobalObject& globalObject, WebGLExtension& extension)
171{
172 switch (extension.getName()) {
173 case WebGLExtension::WebGLLoseContextName:
174 return toJS(&lexicalGlobalObject, &globalObject, static_cast<WebGLLoseContext&>(extension));
175 case WebGLExtension::EXTShaderTextureLODName:
176 return toJS(&lexicalGlobalObject, &globalObject, static_cast<EXTShaderTextureLOD&>(extension));
177 case WebGLExtension::EXTTextureCompressionRGTCName:
178 return toJS(&lexicalGlobalObject, &globalObject, static_cast<EXTTextureCompressionRGTC&>(extension));
179 case WebGLExtension::EXTTextureFilterAnisotropicName:
180 return toJS(&lexicalGlobalObject, &globalObject, static_cast<EXTTextureFilterAnisotropic&>(extension));
181 case WebGLExtension::EXTsRGBName:
182 return toJS(&lexicalGlobalObject, &globalObject, static_cast<EXTsRGB&>(extension));
183 case WebGLExtension::EXTFragDepthName:
184 return toJS(&lexicalGlobalObject, &globalObject, static_cast<EXTFragDepth&>(extension));
185 case WebGLExtension::EXTBlendMinMaxName:
186 return toJS(&lexicalGlobalObject, &globalObject, static_cast<EXTBlendMinMax&>(extension));
187 case WebGLExtension::KHRParallelShaderCompileName:
188 return toJS(&lexicalGlobalObject, &globalObject, static_cast<KHRParallelShaderCompile&>(extension));
189 case WebGLExtension::OESStandardDerivativesName:
190 return toJS(&lexicalGlobalObject, &globalObject, static_cast<OESStandardDerivatives&>(extension));
191 case WebGLExtension::OESTextureFloatName:
192 return toJS(&lexicalGlobalObject, &globalObject, static_cast<OESTextureFloat&>(extension));
193 case WebGLExtension::OESTextureFloatLinearName:
194 return toJS(&lexicalGlobalObject, &globalObject, static_cast<OESTextureFloatLinear&>(extension));
195 case WebGLExtension::OESTextureHalfFloatName:
196 return toJS(&lexicalGlobalObject, &globalObject, static_cast<OESTextureHalfFloat&>(extension));
197 case WebGLExtension::OESTextureHalfFloatLinearName:
198 return toJS(&lexicalGlobalObject, &globalObject, static_cast<OESTextureHalfFloatLinear&>(extension));
199 case WebGLExtension::OESVertexArrayObjectName:
200 return toJS(&lexicalGlobalObject, &globalObject, static_cast<OESVertexArrayObject&>(extension));
201 case WebGLExtension::OESElementIndexUintName:
202 return toJS(&lexicalGlobalObject, &globalObject, static_cast<OESElementIndexUint&>(extension));
203 case WebGLExtension::OESFBORenderMipmapName:
204 return toJS(&lexicalGlobalObject, &globalObject, static_cast<OESFBORenderMipmap&>(extension));
205 case WebGLExtension::WebGLDebugRendererInfoName:
206 return toJS(&lexicalGlobalObject, &globalObject, static_cast<WebGLDebugRendererInfo&>(extension));
207 case WebGLExtension::WebGLDebugShadersName:
208 return toJS(&lexicalGlobalObject, &globalObject, static_cast<WebGLDebugShaders&>(extension));
209 case WebGLExtension::WebGLCompressedTextureATCName:
210 return toJS(&lexicalGlobalObject, &globalObject, static_cast<WebGLCompressedTextureATC&>(extension));
211 case WebGLExtension::WebGLCompressedTextureETCName:
212 return toJS(&lexicalGlobalObject, &globalObject, static_cast<WebGLCompressedTextureETC&>(extension));
213 case WebGLExtension::WebGLCompressedTextureETC1Name:
214 return toJS(&lexicalGlobalObject, &globalObject, static_cast<WebGLCompressedTextureETC1&>(extension));
215 case WebGLExtension::WebGLCompressedTexturePVRTCName:
216 return toJS(&lexicalGlobalObject, &globalObject, static_cast<WebGLCompressedTexturePVRTC&>(extension));
217 case WebGLExtension::WebGLCompressedTextureS3TCName:
218 return toJS(&lexicalGlobalObject, &globalObject, static_cast<WebGLCompressedTextureS3TC&>(extension));
219 case WebGLExtension::WebGLCompressedTextureS3TCsRGBName:
220 return toJS(&lexicalGlobalObject, &globalObject, static_cast<WebGLCompressedTextureS3TCsRGB&>(extension));
221 case WebGLExtension::WebGLCompressedTextureASTCName:
222 return toJS(&lexicalGlobalObject, &globalObject, static_cast<WebGLCompressedTextureASTC&>(extension));
223 case WebGLExtension::WebGLDepthTextureName:
224 return toJS(&lexicalGlobalObject, &globalObject, static_cast<WebGLDepthTexture&>(extension));
225 case WebGLExtension::WebGLDrawBuffersName:
226 return toJS(&lexicalGlobalObject, &globalObject, static_cast<WebGLDrawBuffers&>(extension));
227 case WebGLExtension::ANGLEInstancedArraysName:
228 return toJS(&lexicalGlobalObject, &globalObject, static_cast<ANGLEInstancedArrays&>(extension));
229 case WebGLExtension::EXTColorBufferHalfFloatName:
230 return toJS(&lexicalGlobalObject, &globalObject, static_cast<EXTColorBufferHalfFloat&>(extension));
231 case WebGLExtension::EXTFloatBlendName:
232 return toJS(&lexicalGlobalObject, &globalObject, static_cast<EXTFloatBlend&>(extension));
233 case WebGLExtension::WebGLColorBufferFloatName:
234 return toJS(&lexicalGlobalObject, &globalObject, static_cast<WebGLColorBufferFloat&>(extension));
235 case WebGLExtension::EXTColorBufferFloatName:
236 return toJS(&lexicalGlobalObject, &globalObject, static_cast<EXTColorBufferFloat&>(extension));
237 case WebGLExtension::WebGLMultiDrawName:
238 return toJS(&lexicalGlobalObject, &globalObject, static_cast<WebGLMultiDraw&>(extension));
239 }
240 ASSERT_NOT_REACHED();
241 return jsNull();
242}
