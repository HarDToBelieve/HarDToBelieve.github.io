33namespace WebCore {
34
35JSC::JSValue convertToJSValue(JSC::JSGlobalObject&, JSDOMGlobalObject&, const WebGLAny&);
36JSC::JSValue convertToJSValue(JSC::JSGlobalObject&, JSDOMGlobalObject&, WebGLExtension&);
37
38inline JSC::JSValue convertToJSValue(JSC::JSGlobalObject& lexicalGlobalObject, JSDOMGlobalObject& globalObject, WebGLExtension* extension)
39{
40 if (!extension)
41 return JSC::jsNull();
42 return convertToJSValue(lexicalGlobalObject, globalObject, *extension);
43}
44
45template<> struct JSConverter<IDLWebGLAny> {
46 static constexpr bool needsState = true;
47 static constexpr bool needsGlobalObject = true;
48
49 static JSC::JSValue convert(JSC::JSGlobalObject& lexicalGlobalObject, JSDOMGlobalObject& globalObject, const WebGLAny& value)
50 {
51 return convertToJSValue(lexicalGlobalObject, globalObject, value);
52 }
53};
54
55template<> struct JSConverter<IDLWebGLExtension> {
56 static constexpr bool needsState = true;
57 static constexpr bool needsGlobalObject = true;
58
59 template <typename T>
60 static JSC::JSValue convert(JSC::JSGlobalObject& lexicalGlobalObject, JSDOMGlobalObject& globalObject, const T& value)
61 {
62 return convertToJSValue(lexicalGlobalObject, globalObject, Detail::getPtrOrRef(value));
63 }
64};
