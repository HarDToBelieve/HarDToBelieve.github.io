75// Convert a DOM implementation exception into a JavaScript exception in the execution lexicalGlobalObject.
76WEBCORE_EXPORT void propagateExceptionSlowPath(JSC::JSGlobalObject&, JSC::ThrowScope&, Exception&&);
77
78ALWAYS_INLINE void propagateException(JSC::JSGlobalObject& lexicalGlobalObject, JSC::ThrowScope& throwScope, Exception&& exception)
79{
80 if (throwScope.exception())
81 return;
82 propagateExceptionSlowPath(lexicalGlobalObject, throwScope, WTFMove(exception));
83}
84
85inline void propagateException(JSC::JSGlobalObject& lexicalGlobalObject, JSC::ThrowScope& throwScope, ExceptionOr<void>&& value)
86{
87 if (UNLIKELY(value.hasException()))
88 propagateException(lexicalGlobalObject, throwScope, value.releaseException());
89}
90
91template<typename Functor> void invokeFunctorPropagatingExceptionIfNecessary(JSC::JSGlobalObject& lexicalGlobalObject, JSC::ThrowScope& throwScope, Functor&& functor)
92{
93 using ReturnType = std::invoke_result_t<Functor>;
94
95 if constexpr (IsExceptionOr<ReturnType>) {
96 auto result = functor();
97 if (UNLIKELY(result.hasException()))
98 propagateException(lexicalGlobalObject, throwScope, result.releaseException());
99 } else
100 functor();
101}
