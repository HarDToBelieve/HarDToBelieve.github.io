97JSC_DEFINE_HOST_FUNCTION(makeThisTypeErrorForBuiltins, (JSGlobalObject* globalObject, CallFrame* callFrame))
98{
99 ASSERT(callFrame);
100 ASSERT(callFrame->argumentCount() == 2);
101 VM& vm = globalObject->vm();
102 DeferTermination deferScope(vm);
103 auto scope = DECLARE_CATCH_SCOPE(vm);
104
105 auto interfaceName = callFrame->uncheckedArgument(0).getString(globalObject);
106 scope.assertNoException();
107 auto functionName = callFrame->uncheckedArgument(1).getString(globalObject);
108 scope.assertNoException();
109 return JSValue::encode(createTypeError(globalObject, makeThisTypeErrorMessage(interfaceName.utf8().data(), functionName.utf8().data())));
110}
111
112JSC_DEFINE_HOST_FUNCTION(makeGetterTypeErrorForBuiltins, (JSGlobalObject* globalObject, CallFrame* callFrame))
113{
114 ASSERT(callFrame);
115 ASSERT(callFrame->argumentCount() == 2);
116 VM& vm = globalObject->vm();
117 DeferTermination deferScope(vm);
118 auto scope = DECLARE_CATCH_SCOPE(vm);
119
120 auto interfaceName = callFrame->uncheckedArgument(0).getString(globalObject);
121 scope.assertNoException();
122 auto attributeName = callFrame->uncheckedArgument(1).getString(globalObject);
123 scope.assertNoException();
124
125 auto error = static_cast<ErrorInstance*>(createTypeError(globalObject, JSC::makeDOMAttributeGetterTypeErrorMessage(interfaceName.utf8().data(), attributeName)));
126 error->setNativeGetterTypeError();
127 return JSValue::encode(error);
128}

130JSC_DEFINE_HOST_FUNCTION(makeDOMExceptionForBuiltins, (JSGlobalObject* globalObject, CallFrame* callFrame))
131{
132 ASSERT(callFrame);
133 ASSERT(callFrame->argumentCount() == 2);
134
135 auto& vm = globalObject->vm();
136 DeferTermination deferScope(vm);
137 auto scope = DECLARE_CATCH_SCOPE(vm);
138
139 auto codeValue = callFrame->uncheckedArgument(0).getString(globalObject);
140 scope.assertNoException();
141
142 auto message = callFrame->uncheckedArgument(1).getString(globalObject);
143 scope.assertNoException();
144
145 ExceptionCode code { TypeError };
146 if (codeValue == "AbortError")
147 code = AbortError;
148 auto value = createDOMException(globalObject, code, message);
149
150 EXCEPTION_ASSERT(!scope.exception() || vm.hasPendingTerminationException());
151
152 return JSValue::encode(value);
153}

165JSC_DEFINE_HOST_FUNCTION(whenSignalAborted, (JSGlobalObject* globalObject, CallFrame* callFrame))
166{
167 ASSERT(callFrame);
168 ASSERT(callFrame->argumentCount() == 2);
169
170 auto& vm = globalObject->vm();
171 auto* abortSignal = jsDynamicCast<JSAbortSignal*>(vm, callFrame->uncheckedArgument(0));
172 if (UNLIKELY(!abortSignal))
173 return JSValue::encode(JSValue(JSC::JSValue::JSFalse));
174
175 auto* jsDOMGlobalObject = JSC::jsCast<JSDOMGlobalObject*>(globalObject);
176 Ref<AbortAlgorithm> abortAlgorithm = JSAbortAlgorithm::create(callFrame->uncheckedArgument(1).getObject(), jsDOMGlobalObject);
177
178 bool result = AbortSignal::whenSignalAborted(abortSignal->wrapped(), WTFMove(abortAlgorithm));
179 return JSValue::encode(result ? JSValue(JSC::JSValue::JSTrue) : JSValue(JSC::JSValue::JSFalse));
180}

253template<typename Visitor>
254void JSDOMGlobalObject::visitChildrenImpl(JSCell* cell, Visitor& visitor)
255{
256 JSDOMGlobalObject* thisObject = jsCast<JSDOMGlobalObject*>(cell);
257 ASSERT_GC_OBJECT_INHERITS(thisObject, info());
258 Base::visitChildren(thisObject, visitor);
259
260 {
261 // The GC thread has to grab the GC lock even though it is not mutating the containers.
262 Locker locker { thisObject->m_gcLock };
263
264 for (auto& structure : thisObject->m_structures.values())
265 visitor.append(structure);
266
267 for (auto& guarded : thisObject->m_guardedObjects)
268 guarded->visitAggregate(visitor);
269 }
270
271 for (auto& constructor : thisObject->constructors().array())
272 visitor.append(constructor);
273
274 thisObject->m_builtinInternalFunctions.visit(visitor);
275}

279void JSDOMGlobalObject::promiseRejectionTracker(JSGlobalObject* jsGlobalObject, JSPromise* promise, JSPromiseRejectionOperation operation)
280{
281 // https://html.spec.whatwg.org/multipage/webappapis.html#the-hostpromiserejectiontracker-implementation
282
283 auto& globalObject = *JSC::jsCast<JSDOMGlobalObject*>(jsGlobalObject);
284 auto* context = globalObject.scriptExecutionContext();
285 if (!context)
286 return;
287
288 // FIXME: If script has muted errors (cross origin), terminate these steps.
289 // <https://webkit.org/b/171415> Implement the `muted-errors` property of Scripts to avoid onerror/onunhandledrejection for cross-origin scripts
290
291 switch (operation) {
292 case JSPromiseRejectionOperation::Reject:
293 context->ensureRejectedPromiseTracker().promiseRejected(globalObject, *promise);
294 break;
295 case JSPromiseRejectionOperation::Handle:
296 context->ensureRejectedPromiseTracker().promiseHandled(globalObject, *promise);
297 break;
298 }
299}

306void JSDOMGlobalObject::clearDOMGuardedObjects() const
307{

309 // Calling JSDOMGuardedObject::clear() will however modify the guarded objects container but

316JSFunction* JSDOMGlobalObject::createCrossOriginFunction(JSGlobalObject* lexicalGlobalObject, PropertyName propertyName, NativeFunction nativeFunction, unsigned length)
317{
318 auto& vm = lexicalGlobalObject->vm();
319 CrossOriginMapKey key = std::make_pair(lexicalGlobalObject, nativeFunction.rawPointer());
320
321 // WeakGCMap::ensureValue's functor must not invoke GC since GC can modify WeakGCMap in the middle of HashMap::ensure.
322 // We use DeferGC here (1) not to invoke GC when executing WeakGCMap::ensureValue and (2) to avoid looking up HashMap twice.
323 DeferGC deferGC(vm.heap);
324 return m_crossOriginFunctionMap.ensureValue(key, [&] {
325 return JSFunction::create(vm, lexicalGlobalObject, length, propertyName.publicName(), nativeFunction);
326 });
327}

329GetterSetter* JSDOMGlobalObject::createCrossOriginGetterSetter(JSGlobalObject* lexicalGlobalObject, PropertyName propertyName, GetValueFunc getter, PutValueFunc setter)
330{
331 ASSERT(getter || setter);
332 auto& vm = lexicalGlobalObject->vm();
333 CrossOriginMapKey key = std::make_pair(lexicalGlobalObject, getter ? reinterpret_cast<void*>(getter) : reinterpret_cast<void*>(setter));
334
335 // WeakGCMap::ensureValue's functor must not invoke GC since GC can modify WeakGCMap in the middle of HashMap::ensure.
336 // We use DeferGC here (1) not to invoke GC when executing WeakGCMap::ensureValue and (2) to avoid looking up HashMap twice.
337 DeferGC deferGC(vm.heap);
338 return m_crossOriginGetterSetterMap.ensureValue(key, [&] {
339 return GetterSetter::create(vm, lexicalGlobalObject,
340 getter ? JSCustomGetterFunction::create(vm, lexicalGlobalObject, propertyName, getter) : nullptr,
341 setter ? JSCustomSetterFunction::create(vm, lexicalGlobalObject, propertyName, setter) : nullptr);
342 });
343}

345#if ENABLE(WEBASSEMBLY)
346// https://webassembly.github.io/spec/web-api/index.html#compile-a-potential-webassembly-response
347static JSC::JSPromise* handleResponseOnStreamingAction(JSC::JSGlobalObject* globalObject, JSC::JSValue source, JSC::Wasm::CompilerMode compilerMode, JSC::JSObject* importObject)
348{
349 VM& vm = globalObject->vm();
350 JSLockHolder lock(vm);
351
352 auto deferred = DeferredPromise::create(*jsCast<JSDOMGlobalObject*>(globalObject), DeferredPromise::Mode::RetainPromiseOnResolve);
353
354 auto inputResponse = JSFetchResponse::toWrapped(vm, source);
355 if (!inputResponse) {
356 deferred->reject(TypeError, "first argument must be an Response or Promise for Response"_s);
357 return jsCast<JSC::JSPromise*>(deferred->promise());
358 }
359
360 if (auto exception = inputResponse->loadingException()) {
361 deferred->reject(*exception);
362 return jsCast<JSC::JSPromise*>(deferred->promise());
363 }
364
365 // 4. If response is not CORS-same-origin, reject returnValue with a TypeError and abort these substeps.
366 // If response is opaque, content-type becomes "".
367 if (!inputResponse->isCORSSameOrigin()) {
368 deferred->reject(TypeError, "Response is not CORS-same-origin"_s);
369 return jsCast<JSC::JSPromise*>(deferred->promise());
370 }
371
372 // 3. If mimeType is not `application/wasm`, reject returnValue with a TypeError and abort these substeps.
373 if (!inputResponse->hasWasmMIMEType()) {
374 deferred->reject(TypeError, "Unexpected response MIME type. Expected 'application/wasm'"_s);
375 return jsCast<JSC::JSPromise*>(deferred->promise());
376 }
377
378 // 5. If response’s status is not an ok status, reject returnValue with a TypeError and abort these substeps.
379 if (!inputResponse->ok()) {
380 deferred->reject(TypeError, "Response has not returned OK status"_s);
381 return jsCast<JSC::JSPromise*>(deferred->promise());
382 }
383
384 // https://fetch.spec.whatwg.org/#concept-body-consume-body
385 if (inputResponse->isDisturbedOrLocked()) {
386 deferred->reject(TypeError, "Response is disturbed or locked"_s);
387 return jsCast<JSC::JSPromise*>(deferred->promise());
388 }
389
390 // FIXME: for efficiency, we should load blobs directly instead of going through the readableStream path.
391 if (inputResponse->isBlobBody()) {
392 auto streamOrException = inputResponse->readableStream(*globalObject);
393 if (UNLIKELY(streamOrException.hasException())) {
394 deferred->reject(streamOrException.releaseException());
395 return jsCast<JSC::JSPromise*>(deferred->promise());
396 }
397 }
398
399 auto compiler = JSC::Wasm::StreamingCompiler::create(vm, compilerMode, globalObject, jsCast<JSC::JSPromise*>(deferred->promise()), importObject);
400
401 if (inputResponse->isBodyReceivedByChunk()) {
402 inputResponse->consumeBodyReceivedByChunk([globalObject, compiler = WTFMove(compiler)](auto&& result) mutable {
403 VM& vm = globalObject->vm();
404 JSLockHolder lock(vm);
405
406 if (result.hasException()) {
407 auto exception = result.exception();
408 if (exception.code() == ExistingExceptionError) {
409 auto scope = DECLARE_CATCH_SCOPE(vm);
410
411 EXCEPTION_ASSERT(scope.exception());
412
413 auto error = scope.exception()->value();
414 scope.clearException();
415
416 compiler->fail(globalObject, error);
417 return;
418 }
419
420 auto scope = DECLARE_THROW_SCOPE(vm);
421 auto error = createDOMException(*globalObject, WTFMove(exception));
422 if (UNLIKELY(scope.exception())) {
423 ASSERT(vm.hasPendingTerminationException());
424 compiler->cancel();
425 return;
426 }
427
428 compiler->fail(globalObject, error);
429 return;
430 }
431
432 if (auto* chunk = result.returnValue())
433 compiler->addBytes(chunk->data(), chunk->size());
434 else
435 compiler->finalize(globalObject);
436 });
437 return jsCast<JSC::JSPromise*>(deferred->promise());
438 }
439
440 auto body = inputResponse->consumeBody();
441 WTF::switchOn(body, [&](Ref<FormData>& formData) {
442 if (auto buffer = formData->asSharedBuffer()) {
443 compiler->addBytes(buffer->data(), buffer->size());
444 compiler->finalize(globalObject);
445 return;
446 }
447 // FIXME: Support FormData loading.
448 // https://bugs.webkit.org/show_bug.cgi?id=221248
449 compiler->fail(globalObject, createDOMException(*globalObject, Exception { NotSupportedError, "Not implemented"_s }));
450 }, [&](Ref<SharedBuffer>& buffer) {
451 compiler->addBytes(buffer->data(), buffer->size());
452 compiler->finalize(globalObject);
453 }, [&](std::nullptr_t&) {
454 compiler->finalize(globalObject);
455 });
456
457 return jsCast<JSC::JSPromise*>(deferred->promise());
458}

473static ScriptModuleLoader* scriptModuleLoader(JSDOMGlobalObject* globalObject)
474{
475 VM& vm = globalObject->vm();
476 if (globalObject->inherits<JSDOMWindowBase>(vm)) {
477 if (auto document = jsCast<const JSDOMWindowBase*>(globalObject)->wrapped().document())
478 return &document->moduleLoader();
479 return nullptr;
480 }
481 if (globalObject->inherits<JSRemoteDOMWindowBase>(vm))
482 return nullptr;
483 if (globalObject->inherits<JSWorkerGlobalScopeBase>(vm))
484 return &jsCast<const JSWorkerGlobalScopeBase*>(globalObject)->wrapped().moduleLoader();
485 if (globalObject->inherits<JSWorkletGlobalScopeBase>(vm))
486 return &jsCast<const JSWorkletGlobalScopeBase*>(globalObject)->wrapped().moduleLoader();
487 if (globalObject->inherits<JSIDBSerializationGlobalObject>(vm))
488 return nullptr;
489
490 dataLog("Unexpected global object: ", JSValue(globalObject), "\n");
491 RELEASE_ASSERT_NOT_REACHED();
492 return nullptr;
493}

545JSDOMGlobalObject& callerGlobalObject(JSGlobalObject& lexicalGlobalObject, CallFrame& callFrame)
546{
547 class GetCallerGlobalObjectFunctor {
548 public:
549 GetCallerGlobalObjectFunctor() = default;
550
551 StackVisitor::Status operator()(StackVisitor& visitor) const
552 {
553 if (!m_hasSkippedFirstFrame) {
554 m_hasSkippedFirstFrame = true;
555 return StackVisitor::Continue;
556 }
557
558 if (auto* codeBlock = visitor->codeBlock())
559 m_globalObject = codeBlock->globalObject();
560 else {
561 ASSERT(visitor->callee().rawPtr());
562 // FIXME: Callee is not an object if the caller is Web Assembly.
563 // Figure out what to do here. We can probably get the global object
564 // from the top-most Wasm Instance. https://bugs.webkit.org/show_bug.cgi?id=165721
565 if (visitor->callee().isCell() && visitor->callee().asCell()->isObject())
566 m_globalObject = jsCast<JSObject*>(visitor->callee().asCell())->globalObject();
567 }
568 return StackVisitor::Done;
569 }
570
571 JSGlobalObject* globalObject() const { return m_globalObject; }
572
573 private:
574 mutable bool m_hasSkippedFirstFrame { false };
575 mutable JSGlobalObject* m_globalObject { nullptr };
576 };
577
578 GetCallerGlobalObjectFunctor iter;
579 callFrame.iterate(lexicalGlobalObject.vm(), iter);
580 if (iter.globalObject())
581 return *jsCast<JSDOMGlobalObject*>(iter.globalObject());
582
583 // If we cannot find JSGlobalObject in caller frames, we just return the current lexicalGlobalObject.
584 return *jsCast<JSDOMGlobalObject*>(&lexicalGlobalObject);
585}
586
587JSDOMGlobalObject* toJSDOMGlobalObject(ScriptExecutionContext& context, DOMWrapperWorld& world)
588{
589 if (is<Document>(context))
590 return toJSDOMWindow(downcast<Document>(context).frame(), world);
591
592 if (is<WorkerOrWorkletGlobalScope>(context))
593 return downcast<WorkerOrWorkletGlobalScope>(context).script()->globalScopeWrapper();
594
595 ASSERT_NOT_REACHED();
596 return nullptr;
597}
