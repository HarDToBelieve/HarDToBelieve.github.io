33
34void addValueIterableMethods(JSC::JSGlobalObject& globalObject, JSC::JSObject& prototype)
35{
36 JSC::ArrayPrototype* arrayPrototype = globalObject.arrayPrototype();
37 ASSERT(arrayPrototype);
38
39 JSC::JSGlobalObject* lexicalGlobalObject = &globalObject;
40 ASSERT(lexicalGlobalObject);
41 JSC::VM& vm = lexicalGlobalObject->vm();
42
43 auto copyProperty = [&] (const JSC::Identifier& arrayIdentifier, const JSC::Identifier& otherIdentifier, unsigned attributes = 0) {
44 JSC::JSValue value = arrayPrototype->getDirect(vm, arrayIdentifier);
45 ASSERT(value);
46 prototype.putDirect(vm, otherIdentifier, value, attributes);
47 };
48
49 copyProperty(vm.propertyNames->builtinNames().entriesPrivateName(), vm.propertyNames->builtinNames().entriesPublicName());
50 copyProperty(vm.propertyNames->builtinNames().forEachPrivateName(), vm.propertyNames->builtinNames().forEachPublicName());
51 copyProperty(vm.propertyNames->builtinNames().keysPrivateName(), vm.propertyNames->builtinNames().keysPublicName());
52 copyProperty(vm.propertyNames->builtinNames().valuesPrivateName(), vm.propertyNames->builtinNames().valuesPublicName());
53}
