20#pragma once
21
22#include "JSDOMConstructorWithDocument.h"
23
24namespace WebCore {
25
26// FIMXE: Why can't LegacyFactoryFunctions be used with workers?
27template<typename JSClass> class JSDOMLegacyFactoryFunction final : public JSDOMConstructorWithDocument {
28public:
29 using Base = JSDOMConstructorWithDocument;
30
31 static JSDOMLegacyFactoryFunction* create(JSC::VM&, JSC::Structure*, JSDOMGlobalObject&);
32 static JSC::Structure* createStructure(JSC::VM&, JSC::JSGlobalObject&, JSC::JSValue prototype);
33
34 DECLARE_INFO;
35
36 // Must be defined for each specialization class.
37 static JSC::JSValue prototypeForStructure(JSC::VM&, const JSDOMGlobalObject&);
38
39 // Must be defined for each specialization class.
40 static JSC::EncodedJSValue JSC_HOST_CALL_ATTRIBUTES construct(JSC::JSGlobalObject*, JSC::CallFrame*);
41
42private:
43 JSDOMLegacyFactoryFunction(JSC::Structure* structure, JSDOMGlobalObject& globalObject)
44 : Base(structure, globalObject)
45 {
46 }
47
48 void finishCreation(JSC::VM&, JSDOMGlobalObject&);
49 static JSC::CallData getConstructData(JSC::JSCell*);
50
51 // Usually defined for each specialization class.
52 void initializeProperties(JSC::VM&, JSDOMGlobalObject&) { }
53};
54
55template<typename JSClass> inline JSDOMLegacyFactoryFunction<JSClass>* JSDOMLegacyFactoryFunction<JSClass>::create(JSC::VM& vm, JSC::Structure* structure, JSDOMGlobalObject& globalObject)
56{
57 JSDOMLegacyFactoryFunction* constructor = new (NotNull, JSC::allocateCell<JSDOMLegacyFactoryFunction>(vm.heap)) JSDOMLegacyFactoryFunction(structure, globalObject);
58 constructor->finishCreation(vm, globalObject);
59 return constructor;
60}
61
62template<typename JSClass> inline JSC::Structure* JSDOMLegacyFactoryFunction<JSClass>::createStructure(JSC::VM& vm, JSC::JSGlobalObject& globalObject, JSC::JSValue prototype)
63{
64 return JSC::Structure::create(vm, &globalObject, prototype, JSC::TypeInfo(JSC::ObjectType, StructureFlags), info());
65}
66
67template<typename JSClass> inline void JSDOMLegacyFactoryFunction<JSClass>::finishCreation(JSC::VM& vm, JSDOMGlobalObject& globalObject)
68{
69 Base::finishCreation(globalObject);
70 ASSERT(inherits(vm, info()));
71 initializeProperties(vm, globalObject);
72}
73
74template<typename JSClass> inline JSC::CallData JSDOMLegacyFactoryFunction<JSClass>::getConstructData(JSC::JSCell*)
75{
76 JSC::CallData constructData;
77 constructData.type = JSC::CallData::Type::Native;
78 constructData.native.function = construct;
79 return constructData;
80}
