36std::pair<bool, std::reference_wrapper<JSC::JSObject>> getBackingMap(JSC::JSGlobalObject& lexicalGlobalObject, JSC::JSObject& mapLike)
37{
38 auto& vm = lexicalGlobalObject.vm();
39 auto backingMap = mapLike.get(&lexicalGlobalObject, static_cast<JSVMClientData*>(vm.clientData)->builtinNames().backingMapPrivateName());
40 if (!backingMap.isUndefined())
41 return { false, *JSC::asObject(backingMap) };
42
43 JSC::DeferTermination deferScope(vm);
44 auto scope = DECLARE_CATCH_SCOPE(vm);
45
46 backingMap = JSC::JSMap::create(&lexicalGlobalObject, vm, lexicalGlobalObject.mapStructure());
47 scope.releaseAssertNoException();
48
49 mapLike.putDirect(vm, static_cast<JSVMClientData*>(vm.clientData)->builtinNames().backingMapPrivateName(), backingMap, static_cast<unsigned>(JSC::PropertyAttribute::DontEnum));
50 return { true, *JSC::asObject(backingMap) };
51}

53void clearBackingMap(JSC::JSGlobalObject& lexicalGlobalObject, JSC::JSObject& backingMap)
54{
55 auto& vm = JSC::getVM(&lexicalGlobalObject);
56 auto function = backingMap.get(&lexicalGlobalObject, vm.propertyNames->clear);
57
58 auto callData = JSC::getCallData(vm, function);
59 if (callData.type == JSC::CallData::Type::None)
60 return;
61
62 JSC::MarkedArgumentBuffer arguments;
63 JSC::call(&lexicalGlobalObject, function, callData, &backingMap, arguments);
64}

66void setToBackingMap(JSC::JSGlobalObject& lexicalGlobalObject, JSC::JSObject& backingMap, JSC::JSValue key, JSC::JSValue value)
67{
68 auto& vm = JSC::getVM(&lexicalGlobalObject);
69 auto function = backingMap.get(&lexicalGlobalObject, vm.propertyNames->set);
70
71 auto callData = JSC::getCallData(vm, function);
72 if (callData.type == JSC::CallData::Type::None)
73 return;
74
75 JSC::MarkedArgumentBuffer arguments;
76 arguments.append(key);
77 arguments.append(value);
78 JSC::call(&lexicalGlobalObject, function, callData, &backingMap, arguments);
79}

86JSC::JSValue forwardFunctionCallToBackingMap(JSC::JSGlobalObject& lexicalGlobalObject, JSC::CallFrame& callFrame, JSC::JSObject& backingMap, const JSC::Identifier& functionName)
87{
88 auto function = backingMap.get(&lexicalGlobalObject, functionName);
89
90 auto callData = JSC::getCallData(lexicalGlobalObject.vm(), function);
91 if (callData.type == JSC::CallData::Type::None)
92 return JSC::jsUndefined();
93
94 JSC::MarkedArgumentBuffer arguments;
95 for (size_t cptr = 0; cptr < callFrame.argumentCount(); ++cptr)
96 arguments.append(callFrame.uncheckedArgument(cptr));
97 ASSERT(!arguments.hasOverflowed());
98 return JSC::call(&lexicalGlobalObject, function, callData, &backingMap, arguments);
99}

101JSC::JSValue forwardForEachCallToBackingMap(JSDOMGlobalObject& globalObject, JSC::CallFrame& callFrame, JSC::JSObject& mapLike)
102{
103 auto result = getBackingMap(globalObject, mapLike);
104 ASSERT(!result.first);
105
106 auto* function = globalObject.builtinInternalFunctions().jsDOMBindingInternals().m_forEachWrapperFunction.get();
107 ASSERT(function);
108
109 auto callData = JSC::getCallData(globalObject.vm(), function);
110 ASSERT(callData.type != JSC::CallData::Type::None);
111
112 JSC::MarkedArgumentBuffer arguments;
113 arguments.append(&result.second.get());
114 for (size_t cptr = 0; cptr < callFrame.argumentCount(); ++cptr)
115 arguments.append(callFrame.uncheckedArgument(cptr));
116 ASSERT(!arguments.hasOverflowed());
117 return JSC::call(&globalObject, function, callData, &mapLike, arguments);
118}
