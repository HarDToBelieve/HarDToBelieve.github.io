27#include "JSDOMPromiseDeferred.h"

29namespace WebCore {
30
31template<typename JSClass>
32class IDLOperationReturningPromise {
33public:
34 using ClassParameter = JSClass*;
35 using Operation = JSC::EncodedJSValue(JSC::JSGlobalObject*, JSC::CallFrame*, ClassParameter, Ref<DeferredPromise>&&);
36 using StaticOperation = JSC::EncodedJSValue(JSC::JSGlobalObject*, JSC::CallFrame*, Ref<DeferredPromise>&&);
37
38 template<Operation operation, CastedThisErrorBehavior shouldThrow = CastedThisErrorBehavior::RejectPromise>
39 static JSC::EncodedJSValue call(JSC::JSGlobalObject& lexicalGlobalObject, JSC::CallFrame& callFrame, const char* operationName)
40 {
41 return JSC::JSValue::encode(callPromiseFunction(lexicalGlobalObject, callFrame, [&operationName] (JSC::JSGlobalObject& lexicalGlobalObject, JSC::CallFrame& callFrame, Ref<DeferredPromise>&& promise) {
42 auto* thisObject = IDLOperation<JSClass>::cast(lexicalGlobalObject, callFrame);
43 if constexpr (shouldThrow != CastedThisErrorBehavior::Assert) {
44 if (UNLIKELY(!thisObject))
45 return rejectPromiseWithThisTypeError(promise.get(), JSClass::info()->className, operationName);
46 } else
47 ASSERT(thisObject);
48
49 ASSERT_GC_OBJECT_INHERITS(thisObject, JSClass::info());
50
51 // FIXME: We should refactor the binding generated code to use references for lexicalGlobalObject and thisObject.
52 return operation(&lexicalGlobalObject, &callFrame, thisObject, WTFMove(promise));
53 }));
54 }
55
56 // This function is a special case for custom operations want to handle the creation of the promise themselves.
57 // It is triggered via the extended attribute [ReturnsOwnPromise].
58 template<typename IDLOperation<JSClass>::Operation operation, CastedThisErrorBehavior shouldThrow = CastedThisErrorBehavior::RejectPromise>
59 static JSC::EncodedJSValue callReturningOwnPromise(JSC::JSGlobalObject& lexicalGlobalObject, JSC::CallFrame& callFrame, const char* operationName)
60 {
61 auto* thisObject = IDLOperation<JSClass>::cast(lexicalGlobalObject, callFrame);
62 if constexpr (shouldThrow != CastedThisErrorBehavior::Assert) {
63 if (UNLIKELY(!thisObject))
64 return rejectPromiseWithThisTypeError(lexicalGlobalObject, JSClass::info()->className, operationName);
65 } else
66 ASSERT(thisObject);
67
68 ASSERT_GC_OBJECT_INHERITS(thisObject, JSClass::info());
69
70 // FIXME: We should refactor the binding generated code to use references for lexicalGlobalObject and thisObject.
71 return operation(&lexicalGlobalObject, &callFrame, thisObject);
72 }
73
74 template<StaticOperation operation, CastedThisErrorBehavior shouldThrow = CastedThisErrorBehavior::RejectPromise>
75 static JSC::EncodedJSValue callStatic(JSC::JSGlobalObject& lexicalGlobalObject, JSC::CallFrame& callFrame, const char*)
76 {
77 return JSC::JSValue::encode(callPromiseFunction(lexicalGlobalObject, callFrame, [] (JSC::JSGlobalObject& lexicalGlobalObject, JSC::CallFrame& callFrame, Ref<DeferredPromise>&& promise) {
78 // FIXME: We should refactor the binding generated code to use references for lexicalGlobalObject.
79 return operation(&lexicalGlobalObject, &callFrame, WTFMove(promise));
80 }));
81 }
82
83 // This function is a special case for custom operations want to handle the creation of the promise themselves.
84 // It is triggered via the extended attribute [ReturnsOwnPromise].
85 template<typename IDLOperation<JSClass>::StaticOperation operation, CastedThisErrorBehavior shouldThrow = CastedThisErrorBehavior::RejectPromise>
86 static JSC::EncodedJSValue callStaticReturningOwnPromise(JSC::JSGlobalObject& lexicalGlobalObject, JSC::CallFrame& callFrame, const char*)
87 {
88 // FIXME: We should refactor the binding generated code to use references for lexicalGlobalObject.
89 return operation(&lexicalGlobalObject, &callFrame);
90 }
91};
