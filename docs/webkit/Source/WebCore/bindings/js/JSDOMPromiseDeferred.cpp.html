51void DeferredPromise::callFunction(JSGlobalObject& lexicalGlobalObject, ResolveMode mode, JSValue resolution)
52{
53 if (shouldIgnoreRequestToFulfill())
54 return;
55
56 if (activeDOMObjectsAreSuspended()) {
57 JSC::Strong<JSC::Unknown, ShouldStrongDestructorGrabLock::Yes> strongResolution(lexicalGlobalObject.vm(), resolution);
58 scriptExecutionContext()->eventLoop().queueTask(TaskSource::Networking, [this, protectedThis = makeRef(*this), mode, strongResolution = WTFMove(strongResolution)]() mutable {
59 if (shouldIgnoreRequestToFulfill())
60 return;
61
62 JSC::JSGlobalObject* lexicalGlobalObject = globalObject();
63 JSC::JSLockHolder locker(lexicalGlobalObject);
64 callFunction(*globalObject(), mode, strongResolution.get());
65 });
66 return;
67 }
68
69 // FIXME: We could have error since any JS call can throw stack-overflow errors.
70 // https://bugs.webkit.org/show_bug.cgi?id=203402
71 switch (mode) {
72 case ResolveMode::Resolve:
73 deferred()->resolve(&lexicalGlobalObject, resolution);
74 break;
75 case ResolveMode::Reject:
76 deferred()->reject(&lexicalGlobalObject, resolution);
77 break;
78 case ResolveMode::RejectAsHandled:
79 deferred()->rejectAsHandled(&lexicalGlobalObject, resolution);
80 break;
81 }
82
83 if (m_mode == Mode::ClearPromiseOnResolve)
84 clear();
85}

87void DeferredPromise::whenSettled(Function<void()>&& callback)
88{
89 if (shouldIgnoreRequestToFulfill())
90 return;
91
92 if (activeDOMObjectsAreSuspended()) {
93 scriptExecutionContext()->eventLoop().queueTask(TaskSource::Networking, [this, protectedThis = makeRef(*this), callback = WTFMove(callback)]() mutable {
94 whenSettled(WTFMove(callback));
95 });
96 return;
97 }
98
99 DOMPromise::whenPromiseIsSettled(globalObject(), deferred(), WTFMove(callback));
100}

126void DeferredPromise::reject(Exception exception, RejectAsHandled rejectAsHandled)
127{
128 if (shouldIgnoreRequestToFulfill())
129 return;
130
131 ASSERT(deferred());
132 ASSERT(m_globalObject);
133 auto& lexicalGlobalObject = *m_globalObject;
134 JSC::VM& vm = lexicalGlobalObject.vm();
135 JSC::JSLockHolder locker(vm);
136 auto scope = DECLARE_CATCH_SCOPE(vm);
137
138 if (exception.code() == ExistingExceptionError) {
139 EXCEPTION_ASSERT(scope.exception());
140 auto error = scope.exception()->value();
141 bool isTerminating = handleTerminationExceptionIfNeeded(scope, lexicalGlobalObject);
142 scope.clearException();
143
144 if (!isTerminating)
145 reject<IDLAny>(error, rejectAsHandled);
146 return;
147 }
148
149 auto error = createDOMException(lexicalGlobalObject, WTFMove(exception));
150 if (UNLIKELY(scope.exception())) {
151 handleUncaughtException(scope, lexicalGlobalObject);
152 return;
153 }
154
155 reject(lexicalGlobalObject, error, rejectAsHandled);
156 if (UNLIKELY(scope.exception()))
157 handleUncaughtException(scope, lexicalGlobalObject);
158}

160void DeferredPromise::reject(ExceptionCode ec, const String& message, RejectAsHandled rejectAsHandled)
161{
162 if (shouldIgnoreRequestToFulfill())
163 return;
164
165 ASSERT(deferred());
166 ASSERT(m_globalObject);
167 auto& lexicalGlobalObject = *m_globalObject;
168 JSC::VM& vm = lexicalGlobalObject.vm();
169 JSC::JSLockHolder locker(vm);
170 auto scope = DECLARE_CATCH_SCOPE(vm);
171
172 if (ec == ExistingExceptionError) {
173 EXCEPTION_ASSERT(scope.exception());
174 auto error = scope.exception()->value();
175 bool isTerminating = handleTerminationExceptionIfNeeded(scope, lexicalGlobalObject);
176 scope.clearException();
177
178 if (!isTerminating)
179 reject<IDLAny>(error, rejectAsHandled);
180 return;
181 }
182
183 auto error = createDOMException(&lexicalGlobalObject, ec, message);
184 if (UNLIKELY(scope.exception())) {
185 handleUncaughtException(scope, lexicalGlobalObject);
186 return;
187 }
188
189 reject(lexicalGlobalObject, error, rejectAsHandled);
190 if (UNLIKELY(scope.exception()))
191 handleUncaughtException(scope, lexicalGlobalObject);
192}

194void DeferredPromise::reject(const JSC::PrivateName& privateName, RejectAsHandled rejectAsHandled)
195{
196 if (shouldIgnoreRequestToFulfill())
197 return;
198
199 ASSERT(deferred());
200 ASSERT(m_globalObject);
201 JSC::JSGlobalObject* lexicalGlobalObject = m_globalObject.get();
202 JSC::JSLockHolder locker(lexicalGlobalObject);
203 reject(*lexicalGlobalObject, JSC::Symbol::create(lexicalGlobalObject->vm(), privateName.uid()), rejectAsHandled);
204}

206void rejectPromiseWithExceptionIfAny(JSC::JSGlobalObject& lexicalGlobalObject, JSDOMGlobalObject& globalObject, JSPromise& promise, JSC::CatchScope& catchScope)
207{
208 UNUSED_PARAM(lexicalGlobalObject);
209 if (LIKELY(!catchScope.exception()))
210 return;
211
212 JSValue error = catchScope.exception()->value();
213 catchScope.clearException();
214
215 DeferredPromise::create(globalObject, promise)->reject<IDLAny>(error);
216}

218JSC::EncodedJSValue createRejectedPromiseWithTypeError(JSC::JSGlobalObject& lexicalGlobalObject, const String& errorMessage, RejectedPromiseWithTypeErrorCause cause)
219{
220 auto& globalObject = lexicalGlobalObject;
221 auto& vm = lexicalGlobalObject.vm();
222 auto scope = DECLARE_THROW_SCOPE(vm);
223
224 auto promiseConstructor = globalObject.promiseConstructor();
225 auto rejectFunction = promiseConstructor->get(&lexicalGlobalObject, vm.propertyNames->builtinNames().rejectPrivateName());
226 RETURN_IF_EXCEPTION(scope, { });
227 auto* rejectionValue = static_cast<ErrorInstance*>(createTypeError(&lexicalGlobalObject, errorMessage));
228 if (cause == RejectedPromiseWithTypeErrorCause::NativeGetter)
229 rejectionValue->setNativeGetterTypeError();
230
231 auto callData = getCallData(vm, rejectFunction);
232 ASSERT(callData.type != CallData::Type::None);
233
234 MarkedArgumentBuffer arguments;
235 arguments.append(rejectionValue);
236 ASSERT(!arguments.hasOverflowed());
237
238 RELEASE_AND_RETURN(scope, JSValue::encode(call(&lexicalGlobalObject, rejectFunction, callData, promiseConstructor, arguments)));
239}

241static inline JSC::JSValue parseAsJSON(JSC::JSGlobalObject* lexicalGlobalObject, const String& data)
242{
243 JSC::JSLockHolder lock(lexicalGlobalObject);
244 return JSC::JSONParse(lexicalGlobalObject, data);
245}

247void fulfillPromiseWithJSON(Ref<DeferredPromise>&& promise, const String& data)
248{
249 JSC::JSValue value = parseAsJSON(promise->globalObject(), data);
250 if (!value)
251 promise->reject(SyntaxError);
252 else
253 promise->resolve<IDLAny>(value);
254}

256void fulfillPromiseWithArrayBuffer(Ref<DeferredPromise>&& promise, ArrayBuffer* arrayBuffer)
257{
258 if (!arrayBuffer) {
259 promise->reject<IDLAny>(createOutOfMemoryError(promise->globalObject()));
260 return;
261 }
262 promise->resolve<IDLInterface<ArrayBuffer>>(*arrayBuffer);
263}

265void fulfillPromiseWithArrayBuffer(Ref<DeferredPromise>&& promise, const void* data, size_t length)

267 fulfillPromiseWithArrayBuffer(WTFMove(promise), ArrayBuffer::tryCreate(data, length).get());

270bool DeferredPromise::handleTerminationExceptionIfNeeded(CatchScope& scope, JSDOMGlobalObject& lexicalGlobalObject)
271{
272 auto* exception = scope.exception();
273 VM& vm = scope.vm();
274
275 auto& scriptExecutionContext = *lexicalGlobalObject.scriptExecutionContext();
276 if (is<WorkerGlobalScope>(scriptExecutionContext)) {
277 auto& scriptController = *downcast<WorkerGlobalScope>(scriptExecutionContext).script();
278 bool terminatorCausedException = vm.isTerminationException(exception);
279 if (terminatorCausedException || scriptController.isTerminatingExecution()) {
280 scriptController.forbidExecution();
281 return true;
282 }
283 }
284 return false;
285}
286
287void DeferredPromise::handleUncaughtException(CatchScope& scope, JSDOMGlobalObject& lexicalGlobalObject)
288{
289 auto* exception = scope.exception();
290 handleTerminationExceptionIfNeeded(scope, lexicalGlobalObject);
291 reportException(&lexicalGlobalObject, exception);
