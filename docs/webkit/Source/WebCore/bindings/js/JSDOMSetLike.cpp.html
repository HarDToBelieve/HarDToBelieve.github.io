42std::pair<bool, std::reference_wrapper<JSC::JSObject>> getBackingSet(JSC::JSGlobalObject& lexicalGlobalObject, JSC::JSObject& setLike)
43{
44 auto& vm = lexicalGlobalObject.vm();
45 auto backingSet = setLike.get(&lexicalGlobalObject, static_cast<JSVMClientData*>(vm.clientData)->builtinNames().backingSetPrivateName());
46 if (backingSet.isUndefined()) {
47 auto& vm = lexicalGlobalObject.vm();
48 JSC::DeferTermination deferScope(vm);
49 auto scope = DECLARE_CATCH_SCOPE(vm);
50
51 backingSet = JSC::JSSet::create(&lexicalGlobalObject, vm, lexicalGlobalObject.setStructure());
52 scope.releaseAssertNoException();
53
54 setLike.putDirect(vm, static_cast<JSVMClientData*>(vm.clientData)->builtinNames().backingSetPrivateName(), backingSet, static_cast<unsigned>(JSC::PropertyAttribute::DontEnum));
55 return { true, *JSC::asObject(backingSet) };
56 }
57 return { false, *JSC::asObject(backingSet) };
58}

60void clearBackingSet(JSC::JSGlobalObject& lexicalGlobalObject, JSC::JSObject& backingSet)
61{
62 auto& vm = JSC::getVM(&lexicalGlobalObject);
63 JSC::JSValue function = backingSet.get(&lexicalGlobalObject, vm.propertyNames->clear);
64 ASSERT(!function.isUndefined());
65
66 auto callData = JSC::getCallData(vm, function);
67 ASSERT(callData.type != JSC::CallData::Type::None);
68 JSC::MarkedArgumentBuffer arguments;
69 JSC::call(&lexicalGlobalObject, function, callData, &backingSet, arguments);
70}

90JSC::JSValue forwardFunctionCallToBackingSet(JSC::JSGlobalObject& lexicalGlobalObject, JSC::CallFrame& callFrame, JSC::JSObject& backingSet, const JSC::Identifier& functionName)
91{
92 JSC::JSValue function = backingSet.get(&lexicalGlobalObject, functionName);
93 ASSERT(!function.isUndefined());
94
95 auto callData = JSC::getCallData(lexicalGlobalObject.vm(), function);
96 ASSERT(callData.type != JSC::CallData::Type::None);
97 JSC::MarkedArgumentBuffer arguments;
98 for (size_t cptr = 0; cptr < callFrame.argumentCount(); ++cptr)
99 arguments.append(callFrame.uncheckedArgument(cptr));
100 ASSERT(!arguments.hasOverflowed());
101 return JSC::call(&lexicalGlobalObject, function, callData, &backingSet, arguments);
102}

104JSC::JSValue forwardForEachCallToBackingSet(JSDOMGlobalObject& globalObject, JSC::CallFrame& callFrame, JSC::JSObject& setLike)
105{
106 auto result = getBackingSet(globalObject, setLike);
107 ASSERT(!result.first);
108
109 auto* function = globalObject.builtinInternalFunctions().jsDOMBindingInternals().m_forEachWrapperFunction.get();
110 ASSERT(function);
111
112 auto callData = JSC::getCallData(globalObject.vm(), function);
113 ASSERT(callData.type != JSC::CallData::Type::None);
114
115 JSC::MarkedArgumentBuffer arguments;
116 arguments.append(&result.second.get());
117 for (size_t cptr = 0; cptr < callFrame.argumentCount(); ++cptr)
118 arguments.append(callFrame.uncheckedArgument(cptr));
119 ASSERT(!arguments.hasOverflowed());
120 return JSC::call(&globalObject, function, callData, &setLike, arguments);
121}
