131void JSDOMWindowBase::updateDocument()
132{
133 // Since "document" property is defined as { configurable: false, writable: false, enumerable: true },
134 // users cannot change its attributes further.
135 // Reaching here, the attributes of "document" property should be never changed.
136 ASSERT(m_wrapped->document());
137 JSGlobalObject* lexicalGlobalObject = this;
138 VM& vm = lexicalGlobalObject->vm();
139 auto scope = DECLARE_CATCH_SCOPE(vm);
140
141 bool shouldThrowReadOnlyError = false;
142 bool ignoreReadOnlyErrors = true;
143 bool putResult = false;
144 symbolTablePutTouchWatchpointSet(this, lexicalGlobalObject, static_cast<JSVMClientData*>(vm.clientData)->builtinNames().documentPublicName(), toJS(lexicalGlobalObject, this, m_wrapped->document()), shouldThrowReadOnlyError, ignoreReadOnlyErrors, putResult);
145 EXCEPTION_ASSERT_UNUSED(scope, !scope.exception());
146}

158bool JSDOMWindowBase::supportsRichSourceInfo(const JSGlobalObject* object)
159{
160 const JSDOMWindowBase* thisObject = static_cast<const JSDOMWindowBase*>(object);
161 Frame* frame = thisObject->wrapped().frame();
162 if (!frame)
163 return false;
164
165 Page* page = frame->page();
166 if (!page)
167 return false;
168
169 bool enabled = page->inspectorController().enabled();
170 ASSERT(enabled || !thisObject->debugger());
171 return enabled;
172}

174static inline bool shouldInterruptScriptToPreventInfiniteRecursionWhenClosingPage(Page* page)
175{
176 // See <rdar://problem/5479443>. We don't think that page can ever be NULL
177 // in this case, but if it is, we've gotten into a lexicalGlobalObject where we may have
178 // hung the UI, with no way to ask the client whether to cancel execution.
179 // For now, our solution is just to cancel execution no matter what,
180 // ensuring that we never hang. We might want to consider other solutions
181 // if we discover problems with this one.
182 ASSERT(page);
183 return !page;
184}

186bool JSDOMWindowBase::shouldInterruptScript(const JSGlobalObject* object)
187{
188 const JSDOMWindowBase* thisObject = static_cast<const JSDOMWindowBase*>(object);
189 ASSERT(thisObject->wrapped().frame());
190 Page* page = thisObject->wrapped().frame()->page();
191 return shouldInterruptScriptToPreventInfiniteRecursionWhenClosingPage(page);
192}
193
194bool JSDOMWindowBase::shouldInterruptScriptBeforeTimeout(const JSGlobalObject* object)
195{
196 const JSDOMWindowBase* thisObject = static_cast<const JSDOMWindowBase*>(object);
197 ASSERT(thisObject->wrapped().frame());
198 Page* page = thisObject->wrapped().frame()->page();
199
200 if (shouldInterruptScriptToPreventInfiniteRecursionWhenClosingPage(page))
201 return true;
202
203#if PLATFORM(IOS_FAMILY)
204 if (page->chrome().client().isStopping())
205 return true;
206#endif
207
208 return JSGlobalObject::shouldInterruptScriptBeforeTimeout(object);
209}

220void JSDOMWindowBase::queueMicrotaskToEventLoop(JSGlobalObject& object, Ref<JSC::Microtask>&& task)
221{
222 JSDOMWindowBase& thisObject = static_cast<JSDOMWindowBase&>(object);
223
224 auto callback = JSMicrotaskCallback::create(thisObject, WTFMove(task));
225 auto& eventLoop = thisObject.scriptExecutionContext()->eventLoop();
226 // Propagating media only user gesture for Fetch API's promise chain.
227 auto userGestureToken = UserGestureIndicator::currentUserGesture();
228 if (userGestureToken && (!userGestureToken->isPropagatedFromFetch() || !RuntimeEnabledFeatures::sharedFeatures().userGesturePromisePropagationEnabled()))
229 userGestureToken = nullptr;
230 eventLoop.queueMicrotask([callback = WTFMove(callback), userGestureToken = WTFMove(userGestureToken)]() mutable {
231 if (!userGestureToken) {
232 callback->call();
233 return;
234 }
235
236 UserGestureIndicator gestureIndicator(userGestureToken, UserGestureToken::GestureScope::MediaOnly, UserGestureToken::IsPropagatedFromFetch::Yes);
237 callback->call();
238 });
239}

311void JSDOMWindowBase::fireFrameClearedWatchpointsForWindow(DOMWindow* window)
312{
313 JSC::VM& vm = commonVM();
314 JSVMClientData* clientData = static_cast<JSVMClientData*>(vm.clientData);
315 Vector<Ref<DOMWrapperWorld>> wrapperWorlds;
316 clientData->getAllWorlds(wrapperWorlds);
317 for (unsigned i = 0; i < wrapperWorlds.size(); ++i) {
318 auto& wrappers = wrapperWorlds[i]->wrappers();
319 auto result = wrappers.find(window);
320 if (result == wrappers.end())
321 continue;
322 JSC::JSObject* wrapper = result->value.get();
323 if (!wrapper)
324 continue;
325 JSDOMWindowBase* jsWindow = JSC::jsCast<JSDOMWindowBase*>(wrapper);
326 jsWindow->m_windowCloseWatchpoints->fireAll(vm, "Frame cleared");
327 }
