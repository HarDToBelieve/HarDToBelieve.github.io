98static ALWAYS_INLINE bool allowsLegacyExpandoIndexedProperties()
99{
100#if PLATFORM(COCOA)
101 // Given that WindowProxy is the default |this| value for sloppy mode functions, it's hard to prove
102 // that older iOS and macOS apps don't accidentally depend on this behavior, so we keep it for them.
103 static bool requiresQuirk = !linkedOnOrAfter(SDKVersion::FirstWithoutExpandoIndexedPropertiesOnWindow);
104 return requiresQuirk;
105#else
106 return false;
107#endif
108}

110template <DOMWindowType windowType>
111bool jsDOMWindowGetOwnPropertySlotRestrictedAccess(JSDOMGlobalObject* thisObject, AbstractDOMWindow& window, JSGlobalObject& lexicalGlobalObject, PropertyName propertyName, PropertySlot& slot, const String& errorMessage)
112{
113 VM& vm = lexicalGlobalObject.vm();
114 auto scope = DECLARE_THROW_SCOPE(vm);
115
116 auto& builtinNames = static_cast<JSVMClientData*>(vm.clientData)->builtinNames();
117
118 // https://html.spec.whatwg.org/#crossorigingetownpropertyhelper-(-o,-p-)
119
120 // These are the functions we allow access to cross-origin (DoNotCheckSecurity in IDL).
121 auto* classInfo = windowType == DOMWindowType::Remote ? JSRemoteDOMWindow::info() : JSDOMWindow::info();
122 if (propertyName == builtinNames.closePublicName()
123 || propertyName == builtinNames.focusPublicName()
124 || propertyName == builtinNames.blurPublicName()
125 || propertyName == builtinNames.postMessagePublicName()) {
126 auto* entry = classInfo->staticPropHashTable->entry(propertyName);
127 auto* jsFunction = thisObject->createCrossOriginFunction(&lexicalGlobalObject, propertyName, entry->function(), entry->functionLength());
128 slot.setValue(thisObject, PropertyAttribute::ReadOnly | PropertyAttribute::DontEnum, jsFunction);
129 return true;
130 }
131
132 // When accessing cross-origin known Window properties, we always use the original property getter,
133 // even if the property was removed / redefined. As of early 2016, this matches Firefox and Chrome's
134 // behavior.
135 if (propertyName == builtinNames.windowPublicName()
136 || propertyName == builtinNames.selfPublicName()
137 || propertyName == builtinNames.locationPublicName()
138 || propertyName == builtinNames.closedPublicName()
139 || propertyName == builtinNames.framesPublicName()
140 || propertyName == vm.propertyNames->length
141 || propertyName == builtinNames.topPublicName()
142 || propertyName == builtinNames.openerPublicName()
143 || propertyName == builtinNames.parentPublicName()) {
144 auto* entry = classInfo->staticPropHashTable->entry(propertyName);
145 auto* setter = propertyName == builtinNames.locationPublicName() ? entry->propertyPutter() : nullptr;
146 auto* getterSetter = thisObject->createCrossOriginGetterSetter(&lexicalGlobalObject, propertyName, entry->propertyGetter(), setter);
147 slot.setGetterSlot(thisObject, PropertyAttribute::Accessor | PropertyAttribute::DontEnum, getterSetter);
148 return true;
149 }
150
151 // Check for child frames by name before built-in properties to match Mozilla. This does
152 // not match IE, but some sites end up naming frames things that conflict with window
153 // properties that are in Moz but not IE. Since we have some of these, we have to do it
154 // the Moz way.
155 // FIXME: Add support to named attributes on RemoteFrames.
156 auto* frame = window.frame();
157 if (frame && is<Frame>(*frame)) {
158 if (auto* scopedChild = downcast<Frame>(*frame).tree().scopedChild(propertyNameToAtomString(propertyName))) {
159 slot.setValue(thisObject, PropertyAttribute::ReadOnly | PropertyAttribute::DontEnum, toJS(&lexicalGlobalObject, scopedChild->document()->domWindow()));
160 return true;
161 }
162 }
163
164 if (handleCommonCrossOriginProperties(thisObject, vm, propertyName, slot))
165 return true;
166
167 throwSecurityError(lexicalGlobalObject, scope, errorMessage);
168 slot.setUndefined();
169 return false;
170}
171template bool jsDOMWindowGetOwnPropertySlotRestrictedAccess<DOMWindowType::Local>(JSDOMGlobalObject*, AbstractDOMWindow&, JSGlobalObject&, PropertyName, PropertySlot&, const String&);
172template bool jsDOMWindowGetOwnPropertySlotRestrictedAccess<DOMWindowType::Remote>(JSDOMGlobalObject*, AbstractDOMWindow&, JSGlobalObject&, PropertyName, PropertySlot&, const String&);

174// https://html.spec.whatwg.org/#crossorigingetownpropertyhelper-(-o,-p-)
175bool handleCommonCrossOriginProperties(JSObject* thisObject, VM& vm, PropertyName propertyName, PropertySlot& slot)
176{
177 auto& propertyNames = vm.propertyNames;
178 if (propertyName == propertyNames->builtinNames().thenPublicName() || propertyName == propertyNames->toStringTagSymbol || propertyName == propertyNames->hasInstanceSymbol || propertyName == propertyNames->isConcatSpreadableSymbol) {
179 slot.setValue(thisObject, JSC::PropertyAttribute::ReadOnly | JSC::PropertyAttribute::DontEnum, jsUndefined());
180 return true;
181 }
182 return false;
183}

185// Property access sequence is:
186// (1) indexed properties,
187// (2) regular own properties,
188// (3) named properties (in fact, these shouldn't be on the window, should be on the NPO).
189bool JSDOMWindow::getOwnPropertySlot(JSObject* object, JSGlobalObject* lexicalGlobalObject, PropertyName propertyName, PropertySlot& slot)
190{
191 // (1) First, indexed properties.
192 // Hand off all indexed access to getOwnPropertySlotByIndex, which supports the indexed getter.
193 if (std::optional<unsigned> index = parseIndex(propertyName))
194 return getOwnPropertySlotByIndex(object, lexicalGlobalObject, index.value(), slot);
195
196 auto* thisObject = jsCast<JSDOMWindow*>(object);
197
198 // Hand off all cross-domain access to jsDOMWindowGetOwnPropertySlotRestrictedAccess.
199 String errorMessage;
200 if (!BindingSecurity::shouldAllowAccessToDOMWindow(*lexicalGlobalObject, thisObject->wrapped(), errorMessage))
201 return jsDOMWindowGetOwnPropertySlotRestrictedAccess<DOMWindowType::Local>(thisObject, thisObject->wrapped(), *lexicalGlobalObject, propertyName, slot, errorMessage);
202
203 if (UNLIKELY(!thisObject->m_windowCloseWatchpoints))
204 thisObject->m_windowCloseWatchpoints = WatchpointSet::create(thisObject->wrapped().frame() ? IsWatched : IsInvalidated);
205 // We use m_windowCloseWatchpoints to clear any inline caches once the frame is cleared.
206 // This is sound because DOMWindow can be associated with at most one frame in its lifetime.
207 if (thisObject->m_windowCloseWatchpoints->isStillValidOnJSThread())
208 slot.setWatchpointSet(*thisObject->m_windowCloseWatchpoints);
209
210 // (2) Regular own properties.
211 PropertySlot slotCopy = slot;
212 if (Base::getOwnPropertySlot(thisObject, lexicalGlobalObject, propertyName, slot)) {
213 auto* frame = thisObject->wrapped().frame();
214
215 // Detect when we're getting the property 'showModalDialog', this is disabled, and has its original value.
216 bool isShowModalDialogAndShouldHide = propertyName == static_cast<JSVMClientData*>(lexicalGlobalObject->vm().clientData)->builtinNames().showModalDialogPublicName()
217 && (!frame || !DOMWindow::canShowModalDialog(*frame))
218 && slot.isValue() && isHostFunction(slot.getValue(lexicalGlobalObject, propertyName), s_info.staticPropHashTable->entry(propertyName)->function());
219 // Unless we're in the showModalDialog special case, we're done.
220 if (!isShowModalDialogAndShouldHide)
221 return true;
222 slot = slotCopy;
223
224 } else if (UNLIKELY(slot.isVMInquiry() && slot.isTaintedByOpaqueObject()))
225 return false;
226
227#if ENABLE(USER_MESSAGE_HANDLERS)
228 if (propertyName == static_cast<JSVMClientData*>(lexicalGlobalObject->vm().clientData)->builtinNames().webkitPublicName() && thisObject->wrapped().shouldHaveWebKitNamespaceForWorld(thisObject->world())) {
229 slot.setCacheableCustom(thisObject, JSC::PropertyAttribute::DontDelete | JSC::PropertyAttribute::ReadOnly, jsDOMWindow_webkit);
230 return true;
231 }
232#endif
233
234 return false;
235}

237bool JSDOMWindow::getOwnPropertySlotByIndex(JSObject* object, JSGlobalObject* lexicalGlobalObject, unsigned index, PropertySlot& slot)
238{
239 auto* thisObject = jsCast<JSDOMWindow*>(object);
240 auto& window = thisObject->wrapped();
241 auto* frame = window.frame();
242
243 // Indexed getters take precendence over regular properties, so caching would be invalid.
244 slot.disableCaching();
245
246 // These are also allowed cross-origin, so come before the access check.
247 if (frame && index < frame->tree().scopedChildCount()) {
248 slot.setValue(thisObject, static_cast<unsigned>(JSC::PropertyAttribute::ReadOnly), toJS(lexicalGlobalObject, frame->tree().scopedChild(index)->document()->domWindow()));
249 return true;
250 }
251
252 if (!BindingSecurity::shouldAllowAccessToDOMWindow(lexicalGlobalObject, window, ThrowSecurityError))
253 return false;
254 if (allowsLegacyExpandoIndexedProperties())
255 return Base::getOwnPropertySlotByIndex(thisObject, lexicalGlobalObject, index, slot);
256 return false;
257}

259bool JSDOMWindow::put(JSCell* cell, JSGlobalObject* lexicalGlobalObject, PropertyName propertyName, JSValue value, PutPropertySlot& slot)
260{
261 VM& vm = lexicalGlobalObject->vm();
262 auto scope = DECLARE_THROW_SCOPE(vm);
263
264 auto* thisObject = jsCast<JSDOMWindow*>(cell);
265
266 if (propertyName != static_cast<JSVMClientData*>(vm.clientData)->builtinNames().locationPublicName()) {
267 // We only allow setting "location" attribute cross-origin.
268 if (!BindingSecurity::shouldAllowAccessToDOMWindow(lexicalGlobalObject, thisObject->wrapped(), ThrowSecurityError))
269 return false;
270 }
271
272 if (parseIndex(propertyName) && !allowsLegacyExpandoIndexedProperties())
273 return typeError(lexicalGlobalObject, scope, slot.isStrictMode(), makeUnsupportedIndexedSetterErrorMessage("Window"));
274 RELEASE_AND_RETURN(scope, Base::put(thisObject, lexicalGlobalObject, propertyName, value, slot));
275}

277bool JSDOMWindow::putByIndex(JSCell* cell, JSGlobalObject* lexicalGlobalObject, unsigned index, JSValue value, bool shouldThrow)
278{
279 VM& vm = lexicalGlobalObject->vm();
280 auto* thisObject = jsCast<JSDOMWindow*>(cell);
281 auto scope = DECLARE_THROW_SCOPE(vm);
282
283 String errorMessage;
284 if (!BindingSecurity::shouldAllowAccessToDOMWindow(*lexicalGlobalObject, thisObject->wrapped(), errorMessage)) {
285 throwSecurityError(*lexicalGlobalObject, scope, errorMessage);
286 return false;
287 }
288
289 if (allowsLegacyExpandoIndexedProperties()) {
290 if (auto* document = thisObject->wrapped().document())
291 document->addConsoleMessage(MessageSource::JS, MessageLevel::Warning, "Adding expando indexed properties to 'window' was a non-standard behavior that is now removed."_s);
292 RELEASE_AND_RETURN(scope, Base::putByIndex(thisObject, lexicalGlobalObject, index, value, shouldThrow));
293 }
294 return typeError(lexicalGlobalObject, scope, shouldThrow, makeUnsupportedIndexedSetterErrorMessage("Window"));
295}

297bool JSDOMWindow::deleteProperty(JSCell* cell, JSGlobalObject* lexicalGlobalObject, PropertyName propertyName, DeletePropertySlot& slot)
298{
299 JSDOMWindow* thisObject = jsCast<JSDOMWindow*>(cell);
300 // Only allow deleting properties by frames in the same origin.
301 if (!BindingSecurity::shouldAllowAccessToDOMWindow(lexicalGlobalObject, thisObject->wrapped(), ThrowSecurityError))
302 return false;
303 if (std::optional<uint32_t> index = parseIndex(propertyName)) {
304 if (!allowsLegacyExpandoIndexedProperties())
305 return index.value() >= thisObject->wrapped().length();
306 }
307 return Base::deleteProperty(thisObject, lexicalGlobalObject, propertyName, slot);
308}

330// https://html.spec.whatwg.org/#crossoriginproperties-(-o-)
331template <CrossOriginObject objectType>
332static void addCrossOriginPropertyNames(VM& vm, PropertyNameArray& propertyNames)
333{
334 auto& builtinNames = static_cast<JSVMClientData*>(vm.clientData)->builtinNames();
335 switch (objectType) {
336 case CrossOriginObject::Location: {
337 static const Identifier* const properties[] = { &builtinNames.hrefPublicName(), &vm.propertyNames->replace };
338 for (auto* property : properties)
339 propertyNames.add(*property);
340 break;
341 }
342 case CrossOriginObject::Window: {
343 static const Identifier* const properties[] = {
344 &builtinNames.blurPublicName(), &builtinNames.closePublicName(), &builtinNames.closedPublicName(),
345 &builtinNames.focusPublicName(), &builtinNames.framesPublicName(), &vm.propertyNames->length,
346 &builtinNames.locationPublicName(), &builtinNames.openerPublicName(), &builtinNames.parentPublicName(),
347 &builtinNames.postMessagePublicName(), &builtinNames.selfPublicName(), &builtinNames.topPublicName(),
348 &builtinNames.windowPublicName()
349 };
350
351 for (auto* property : properties)
352 propertyNames.add(*property);
353 break;
354 }
355 }
356}

358// https://html.spec.whatwg.org/#crossoriginownpropertykeys-(-o-)
359template <CrossOriginObject objectType>
360void addCrossOriginOwnPropertyNames(JSC::JSGlobalObject& lexicalGlobalObject, JSC::PropertyNameArray& propertyNames)
361{
362 auto& vm = lexicalGlobalObject.vm();
363 addCrossOriginPropertyNames<objectType>(vm, propertyNames);
364
365 static const Identifier* const properties[] = {
366 &vm.propertyNames->builtinNames().thenPublicName(), &vm.propertyNames->toStringTagSymbol, &vm.propertyNames->hasInstanceSymbol, &vm.propertyNames->isConcatSpreadableSymbol
367 };
368
369 for (auto* property : properties)
370 propertyNames.add(*property);
371
372}

376static void addScopedChildrenIndexes(JSGlobalObject& lexicalGlobalObject, DOMWindow& window, PropertyNameArray& propertyNames)
377{
378 auto* document = window.document();
379 if (!document)
380 return;
381
382 auto* frame = document->frame();
383 if (!frame)
384 return;
385
386 VM& vm = lexicalGlobalObject.vm();
387 unsigned scopedChildCount = frame->tree().scopedChildCount();
388 for (unsigned i = 0; i < scopedChildCount; ++i)
389 propertyNames.add(Identifier::from(vm, i));
390}

392// https://html.spec.whatwg.org/#windowproxy-ownpropertykeys
393void JSDOMWindow::getOwnPropertyNames(JSObject* object, JSGlobalObject* lexicalGlobalObject, PropertyNameArray& propertyNames, DontEnumPropertiesMode mode)
394{
395 JSDOMWindow* thisObject = jsCast<JSDOMWindow*>(object);
396
397 addScopedChildrenIndexes(*lexicalGlobalObject, thisObject->wrapped(), propertyNames);
398
399 if (!BindingSecurity::shouldAllowAccessToDOMWindow(lexicalGlobalObject, thisObject->wrapped(), DoNotReportSecurityError)) {
400 if (mode == DontEnumPropertiesMode::Include)
401 addCrossOriginOwnPropertyNames<CrossOriginObject::Window>(*lexicalGlobalObject, propertyNames);
402 return;
403 }
404 Base::getOwnPropertyNames(thisObject, lexicalGlobalObject, propertyNames, mode);
405}

407bool JSDOMWindow::defineOwnProperty(JSC::JSObject* object, JSC::JSGlobalObject* lexicalGlobalObject, JSC::PropertyName propertyName, const JSC::PropertyDescriptor& descriptor, bool shouldThrow)
408{
409 VM& vm = lexicalGlobalObject->vm();
410 auto scope = DECLARE_THROW_SCOPE(vm);
411
412 JSDOMWindow* thisObject = jsCast<JSDOMWindow*>(object);
413 // Only allow defining properties in this way by frames in the same origin, as it allows setters to be introduced.
414 if (!BindingSecurity::shouldAllowAccessToDOMWindow(lexicalGlobalObject, thisObject->wrapped(), ThrowSecurityError))
415 return false;
416 if (parseIndex(propertyName) && !allowsLegacyExpandoIndexedProperties())
417 return typeError(lexicalGlobalObject, scope, shouldThrow, makeUnsupportedIndexedSetterErrorMessage("Window"));
418 scope.release();
419
420 auto& builtinNames = static_cast<JSVMClientData*>(vm.clientData)->builtinNames();
421 if (propertyName == builtinNames.documentPublicName() || propertyName == builtinNames.windowPublicName())
422 return JSObject::defineOwnProperty(thisObject, lexicalGlobalObject, propertyName, descriptor, shouldThrow);
423
424 return Base::defineOwnProperty(thisObject, lexicalGlobalObject, propertyName, descriptor, shouldThrow);
425}

470inline void DialogHandler::dialogCreated(DOMWindow& dialog)
471{
472 VM& vm = m_globalObject.vm();
473 m_frame = dialog.frame();
474
475 // FIXME: This looks like a leak between the normal world and an isolated
476 // world if dialogArguments comes from an isolated world.
477 JSDOMWindow* globalObject = toJSDOMWindow(m_frame.get(), normalWorld(vm));
478 if (JSValue dialogArguments = m_callFrame.argument(1))
479 globalObject->putDirect(vm, Identifier::fromString(vm, "dialogArguments"), dialogArguments);
480}
481
482inline JSValue DialogHandler::returnValue() const
483{
484 VM& vm = m_globalObject.vm();
485 JSDOMWindow* globalObject = toJSDOMWindow(m_frame.get(), normalWorld(vm));
486 if (!globalObject)
487 return jsUndefined();
488 Identifier identifier = Identifier::fromString(vm, "returnValue");
489 PropertySlot slot(globalObject, PropertySlot::InternalMethodType::Get);
490 if (!JSGlobalObject::getOwnPropertySlot(globalObject, &m_globalObject, identifier, slot))
491 return jsUndefined();
492 return slot.getValue(&m_globalObject, identifier);
493}

495JSValue JSDOMWindow::showModalDialog(JSGlobalObject& lexicalGlobalObject, CallFrame& callFrame)
496{
497 VM& vm = lexicalGlobalObject.vm();
498 auto scope = DECLARE_THROW_SCOPE(vm);
499
500 if (UNLIKELY(callFrame.argumentCount() < 1))
501 return throwException(&lexicalGlobalObject, scope, createNotEnoughArgumentsError(&lexicalGlobalObject));
502
503 String urlString = convert<IDLNullable<IDLDOMString>>(lexicalGlobalObject, callFrame.argument(0));
504 RETURN_IF_EXCEPTION(scope, JSValue());
505 String dialogFeaturesString = convert<IDLNullable<IDLDOMString>>(lexicalGlobalObject, callFrame.argument(2));
506 RETURN_IF_EXCEPTION(scope, JSValue());
507
508 DialogHandler handler(lexicalGlobalObject, callFrame);
509
510 wrapped().showModalDialog(urlString, dialogFeaturesString, activeDOMWindow(lexicalGlobalObject), firstDOMWindow(lexicalGlobalObject), [&handler](DOMWindow& dialog) {
511 handler.dialogCreated(dialog);
512 });
513
514 return handler.returnValue();
515}

517JSValue JSDOMWindow::queueMicrotask(JSGlobalObject& lexicalGlobalObject, CallFrame& callFrame)
518{
519 VM& vm = lexicalGlobalObject.vm();
520 auto scope = DECLARE_THROW_SCOPE(vm);
521
522 if (UNLIKELY(callFrame.argumentCount() < 1))
523 return throwException(&lexicalGlobalObject, scope, createNotEnoughArgumentsError(&lexicalGlobalObject));
524
525 JSValue functionValue = callFrame.uncheckedArgument(0);
526 if (UNLIKELY(!functionValue.isCallable(vm)))
527 return JSValue::decode(throwArgumentMustBeFunctionError(lexicalGlobalObject, scope, 0, "callback", "Window", "queueMicrotask"));
528
529 scope.release();
530 Base::queueMicrotask(JSC::createJSMicrotask(vm, functionValue));
531 return jsUndefined();
532}

534DOMWindow* JSDOMWindow::toWrapped(VM& vm, JSValue value)
535{
536 if (!value.isObject())
537 return nullptr;
538 JSObject* object = asObject(value);
539 if (object->inherits<JSDOMWindow>(vm))
540 return &jsCast<JSDOMWindow*>(object)->wrapped();
541 if (object->inherits<JSWindowProxy>(vm)) {
542 if (auto* jsDOMWindow = jsDynamicCast<JSDOMWindow*>(vm, jsCast<JSWindowProxy*>(object)->window()))
543 return &jsDOMWindow->wrapped();
544 }
545 return nullptr;
546}

576static inline JSC::EncodedJSValue jsDOMWindowInstanceFunction_openDatabaseBody(JSC::JSGlobalObject* lexicalGlobalObject, CallFrame* callFrame, typename IDLOperation<JSDOMWindow>::ClassParameter castedThis)
577{
578 auto& vm = lexicalGlobalObject->vm();
579 auto throwScope = DECLARE_THROW_SCOPE(vm);
580
