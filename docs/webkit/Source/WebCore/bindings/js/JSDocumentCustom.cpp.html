33static inline JSValue createNewDocumentWrapper(JSGlobalObject& lexicalGlobalObject, JSDOMGlobalObject& globalObject, Ref<Document>&& passedDocument)
34{
35 auto& document = passedDocument.get();
36 JSObject* wrapper;
37 if (document.isHTMLDocument())
38 wrapper = createWrapper<HTMLDocument>(&globalObject, WTFMove(passedDocument));
39 else if (document.isXMLDocument())
40 wrapper = createWrapper<XMLDocument>(&globalObject, WTFMove(passedDocument));
41 else
42 wrapper = createWrapper<Document>(&globalObject, WTFMove(passedDocument));
43
44 reportMemoryForDocumentIfFrameless(lexicalGlobalObject, document);
45
46 return wrapper;
47}
48
49JSObject* cachedDocumentWrapper(JSGlobalObject& lexicalGlobalObject, JSDOMGlobalObject& globalObject, Document& document)
50{
51 if (auto* wrapper = getCachedWrapper(globalObject.world(), document))
52 return wrapper;
53
54 auto* window = document.domWindow();
55 if (!window)
56 return nullptr;
57
58 auto* documentGlobalObject = toJSDOMGlobalObject<JSDOMWindow>(lexicalGlobalObject.vm(), toJS(&lexicalGlobalObject, *window));
59 if (!documentGlobalObject)
60 return nullptr;
61
62 // Creating a wrapper for domWindow might have created a wrapper for document as well.
63 return getCachedWrapper(documentGlobalObject->world(), document);
64}
65
66void reportMemoryForDocumentIfFrameless(JSGlobalObject& lexicalGlobalObject, Document& document)
67{
68 // Make sure the document is kept around by the window object, and works right with the back/forward cache.
69 if (document.frame())
70 return;
71
72 VM& vm = lexicalGlobalObject.vm();
73 size_t memoryCost = 0;
74 for (Node* node = &document; node; node = NodeTraversal::next(*node))
75 memoryCost += node->approximateMemoryCost();
76
77 // FIXME: Adopt reportExtraMemoryVisited, and switch to reportExtraMemoryAllocated.
78 // https://bugs.webkit.org/show_bug.cgi?id=142595
79 vm.heap.deprecatedReportExtraMemory(memoryCost);
80}
81
82JSValue toJSNewlyCreated(JSGlobalObject* lexicalGlobalObject, JSDOMGlobalObject* globalObject, Ref<Document>&& document)
83{
84 return createNewDocumentWrapper(*lexicalGlobalObject, *globalObject, WTFMove(document));
85}
86
87JSValue toJS(JSGlobalObject* lexicalGlobalObject, JSDOMGlobalObject* globalObject, Document& document)
88{
89 if (auto* wrapper = cachedDocumentWrapper(*lexicalGlobalObject, *globalObject, document))
90 return wrapper;
91 return toJSNewlyCreated(lexicalGlobalObject, globalObject, Ref<Document>(document));
92}

94template<typename Visitor>
95void JSDocument::visitAdditionalChildren(Visitor& visitor)
96{
97 visitor.addOpaqueRoot(static_cast<ScriptExecutionContext*>(&wrapped()));
98}

102void JSDocument::analyzeHeap(JSCell* cell, HeapAnalyzer& analyzer)
103{
104 Base::analyzeHeap(cell, analyzer);
105 auto* thisObject = jsCast<JSDocument*>(cell);
106 analyzer.setLabelForCell(cell, thisObject->wrapped().url().string());
107}
