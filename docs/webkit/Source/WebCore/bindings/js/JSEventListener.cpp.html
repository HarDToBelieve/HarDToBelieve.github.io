48JSEventListener::JSEventListener(JSObject* function, JSObject* wrapper, bool isAttribute, DOMWrapperWorld& isolatedWorld)
49 : EventListener(JSEventListenerType)

53{

55 ASSERT(wrapper);

94static void handleBeforeUnloadEventReturnValue(BeforeUnloadEvent& event, const String& returnValue)
95{
96 if (returnValue.isNull())
97 return;
98
99 event.preventDefault();
100 if (event.returnValue().isEmpty())
101 event.setReturnValue(returnValue);
102}

104void JSEventListener::handleEvent(ScriptExecutionContext& scriptExecutionContext, Event& event)
105{
106 if (scriptExecutionContext.isJSExecutionForbidden())
107 return;
108
109 VM& vm = scriptExecutionContext.vm();
110 JSLockHolder lock(vm);
111 auto scope = DECLARE_CATCH_SCOPE(vm);
112
113 // See https://dom.spec.whatwg.org/#dispatching-events spec on calling handleEvent.
114 // "If this throws an exception, report the exception." It should not propagate the
115 // exception.
116
117 JSObject* jsFunction = ensureJSFunction(scriptExecutionContext);
118 if (!jsFunction)
119 return;
120
121 JSDOMGlobalObject* globalObject = toJSDOMGlobalObject(scriptExecutionContext, m_isolatedWorld);
122 if (!globalObject)
123 return;
124
125 if (scriptExecutionContext.isDocument()) {
126 JSDOMWindow* window = jsCast<JSDOMWindow*>(globalObject);
127 if (!window->wrapped().isCurrentlyDisplayedInFrame())
128 return;
129 if (wasCreatedFromMarkup() && !scriptExecutionContext.contentSecurityPolicy()->allowInlineEventHandlers(sourceURL().string(), sourcePosition().m_line))
130 return;
131 // FIXME: Is this check needed for other contexts?
132 ScriptController& script = window->wrapped().frame()->script();
133 if (!script.canExecuteScripts(AboutToExecuteScript) || script.isPaused())
134 return;
135 }
136
137 JSGlobalObject* lexicalGlobalObject = globalObject;
138
139 JSValue handleEventFunction = jsFunction;
140
141 auto callData = getCallData(vm, handleEventFunction);
142
143 // If jsFunction is not actually a function and this is an EventListener, see if it implements callback interface.
144 if (callData.type == CallData::Type::None) {
145 if (m_isAttribute)
146 return;
147
148 handleEventFunction = jsFunction->get(lexicalGlobalObject, Identifier::fromString(vm, "handleEvent"));
149 if (UNLIKELY(scope.exception())) {
150 auto* exception = scope.exception();
151 scope.clearException();
152 event.target()->uncaughtExceptionInEventHandler();
153 reportException(lexicalGlobalObject, exception);
154 return;
155 }
156 callData = getCallData(vm, handleEventFunction);
157 if (callData.type == CallData::Type::None) {
158 event.target()->uncaughtExceptionInEventHandler();
159 reportException(lexicalGlobalObject, createTypeError(lexicalGlobalObject, "'handleEvent' property of event listener should be callable"_s));
160 return;
161 }
162 }
163
164 Ref<JSEventListener> protectedThis(*this);
165
166 MarkedArgumentBuffer args;
167 args.append(toJS(lexicalGlobalObject, globalObject, &event));
168 ASSERT(!args.hasOverflowed());
169
170 RefPtr<Event> savedEvent;
171 auto* jsFunctionWindow = jsDynamicCast<JSDOMWindow*>(vm, jsFunction->globalObject());
172 if (jsFunctionWindow) {
173 savedEvent = jsFunctionWindow->currentEvent();
174
175 // window.event should not be set when the target is inside a shadow tree, as per the DOM specification.
176 if (!event.currentTargetIsInShadowTree())
177 jsFunctionWindow->setCurrentEvent(&event);
178 }
179
180 VMEntryScope entryScope(vm, vm.entryScope ? vm.entryScope->globalObject() : globalObject);
181
182 JSExecState::instrumentFunction(&scriptExecutionContext, callData);
183
184 JSValue thisValue = handleEventFunction == jsFunction ? toJS(lexicalGlobalObject, globalObject, event.currentTarget()) : jsFunction;
185 NakedPtr<JSC::Exception> uncaughtException;
186 JSValue retval = JSExecState::profiledCall(lexicalGlobalObject, JSC::ProfilingReason::Other, handleEventFunction, callData, thisValue, args, uncaughtException);
187
188 InspectorInstrumentation::didCallFunction(&scriptExecutionContext);
189
190 if (jsFunctionWindow)
191 jsFunctionWindow->setCurrentEvent(savedEvent.get());
192
193 auto handleExceptionIfNeeded = [&] (JSC::Exception* exception) -> bool {
194 if (is<WorkerGlobalScope>(scriptExecutionContext)) {
195 auto& scriptController = *downcast<WorkerGlobalScope>(scriptExecutionContext).script();
196 bool terminatorCausedException = (exception && vm.isTerminationException(exception));
197 if (terminatorCausedException || scriptController.isTerminatingExecution())
198 scriptController.forbidExecution();
199 }
200
201 if (exception) {
202 event.target()->uncaughtExceptionInEventHandler();
203 reportException(lexicalGlobalObject, exception);
204 return true;
205 }
206 return false;
207 };
208
209 if (handleExceptionIfNeeded(uncaughtException))
210 return;
211
212 if (!m_isAttribute) {
213 // This is an EventListener and there is therefore no need for any return value handling.
214 return;
215 }
216
217 // Do return value handling for event handlers (https://html.spec.whatwg.org/#the-event-handler-processing-algorithm).
218
219 if (event.type() == eventNames().beforeunloadEvent) {
220 // This is a OnBeforeUnloadEventHandler, and therefore the return value must be coerced into a String.
221 if (is<BeforeUnloadEvent>(event)) {
222 String resultStr = convert<IDLNullable<IDLDOMString>>(*lexicalGlobalObject, retval);
223 if (UNLIKELY(scope.exception())) {
224 if (handleExceptionIfNeeded(scope.exception()))
225 return;
226 }
227 handleBeforeUnloadEventReturnValue(downcast<BeforeUnloadEvent>(event), resultStr);
228 }
229 return;
230 }
231
232 if (retval.isFalse())
233 event.preventDefault();
234}

236bool JSEventListener::operator==(const EventListener& listener) const
237{
238 if (!is<JSEventListener>(listener))
239 return false;
240 auto& other = downcast<JSEventListener>(listener);
241 return m_jsFunction == other.m_jsFunction && m_isAttribute == other.m_isAttribute;
242}
243
244String JSEventListener::functionName() const
245{
246 if (!m_wrapper || !m_jsFunction)
247 return { };
248
249 auto& vm = isolatedWorld().vm();
250 JSC::JSLockHolder lock(vm);
251
252 auto* handlerFunction = JSC::jsDynamicCast<JSC::JSFunction*>(vm, m_jsFunction.get());
253 if (!handlerFunction)
254 return { };
255
256 return handlerFunction->name(vm);
257}

259static inline JSC::JSValue eventHandlerAttribute(EventListener* abstractListener, ScriptExecutionContext& context)
260{
261 if (!is<JSEventListener>(abstractListener))
262 return jsNull();
263
264 auto* function = downcast<JSEventListener>(*abstractListener).ensureJSFunction(context);
265 if (!function)
266 return jsNull();
267
268 return function;
269}

271static inline RefPtr<JSEventListener> createEventListenerForEventHandlerAttribute(JSC::JSGlobalObject& lexicalGlobalObject, JSC::JSValue listener, JSC::JSObject& wrapper)
272{
273 if (!listener.isObject())
274 return nullptr;
275 return JSEventListener::create(asObject(listener), &wrapper, true, currentWorld(lexicalGlobalObject));
276}

278JSC::JSValue eventHandlerAttribute(EventTarget& target, const AtomString& eventType, DOMWrapperWorld& isolatedWorld)
279{
280 auto* context = target.scriptExecutionContext();
281 if (!context)
282 return jsNull();
283 return eventHandlerAttribute(target.attributeEventListener(eventType, isolatedWorld), *context);
284}

286void setEventHandlerAttribute(JSC::JSGlobalObject& lexicalGlobalObject, JSC::JSObject& wrapper, EventTarget& target, const AtomString& eventType, JSC::JSValue value)

319void setDocumentEventHandlerAttribute(JSC::JSGlobalObject& lexicalGlobalObject, JSC::JSObject& wrapper, HTMLElement& element, const AtomString& eventType, JSC::JSValue value)
320{
321 ASSERT(wrapper.globalObject());
322 auto& document = element.document();
323 auto* documentWrapper = JSC::jsCast<JSDocument*>(toJS(&lexicalGlobalObject, JSC::jsCast<JSDOMGlobalObject*>(wrapper.globalObject()), document));

326}
