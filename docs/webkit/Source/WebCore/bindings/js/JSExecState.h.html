42class JSExecState {
43 WTF_MAKE_NONCOPYABLE(JSExecState);
44 WTF_FORBID_HEAP_ALLOCATION;
45 friend class JSMainThreadNullState;
46public:
47 static JSC::JSGlobalObject* currentState()
48 {
49 return threadGlobalData().currentState();

52 static JSC::JSValue call(JSC::JSGlobalObject* lexicalGlobalObject, JSC::JSValue functionObject, const JSC::CallData& callData, JSC::JSValue thisValue, const JSC::ArgList& args, NakedPtr<JSC::Exception>& returnedException)

62 };

94 static JSC::JSInternalPromise& loadModule(JSC::JSGlobalObject& lexicalGlobalObject, const String& moduleName, JSC::JSValue parameters, JSC::JSValue scriptFetcher)
95 {

100 static JSC::JSInternalPromise& loadModule(JSC::JSGlobalObject& lexicalGlobalObject, const JSC::SourceCode& sourceCode, JSC::JSValue scriptFetcher)
101 {
102 JSExecState currentState(&lexicalGlobalObject);
103 return *JSC::loadModule(&lexicalGlobalObject, sourceCode, scriptFetcher);
104 }
105
106 static JSC::JSValue linkAndEvaluateModule(JSC::JSGlobalObject& lexicalGlobalObject, const JSC::Identifier& moduleKey, JSC::JSValue scriptFetcher, NakedPtr<JSC::Exception>& returnedException)
107 {
108 JSC::VM& vm = JSC::getVM(&lexicalGlobalObject);
109 auto scope = DECLARE_CATCH_SCOPE(vm);
110
111 JSExecState currentState(&lexicalGlobalObject);
112 auto returnValue = JSC::linkAndEvaluateModule(&lexicalGlobalObject, moduleKey, scriptFetcher);
113 if (UNLIKELY(scope.exception())) {
114 returnedException = scope.exception();
115 scope.clearException();
116 return JSC::jsUndefined();
117 }
118 return returnValue;
119 }
120
121 static void instrumentFunction(ScriptExecutionContext*, const JSC::CallData&);
122
123private:
124 explicit JSExecState(JSC::JSGlobalObject* lexicalGlobalObject)
125 : m_previousState(currentState())
126 , m_lock(lexicalGlobalObject)
127 {
128 setCurrentState(lexicalGlobalObject);
129 };
130
131 ~JSExecState()
132 {
133 JSC::VM& vm = currentState()->vm();
134 auto scope = DECLARE_CATCH_SCOPE(vm);
135 scope.assertNoException();
136
137 JSC::JSGlobalObject* lexicalGlobalObject = currentState();
138 bool didExitJavaScript = lexicalGlobalObject && !m_previousState;
139
140 setCurrentState(m_previousState);
141
142 if (didExitJavaScript) {
143 didLeaveScriptContext(lexicalGlobalObject);
144 // We need to clear any exceptions from microtask drain.
145 scope.clearException();
146 }
147 }
148
149 static void setCurrentState(JSC::JSGlobalObject* lexicalGlobalObject)
150 {
151 threadGlobalData().setCurrentState(lexicalGlobalObject);
152 }
153
154 JSC::JSGlobalObject* m_previousState;
155 JSC::JSLockHolder m_lock;
156
157 static void didLeaveScriptContext(JSC::JSGlobalObject*);
158};

160// Null lexicalGlobalObject prevents origin security checks.
161// Used by non-JavaScript bindings (ObjC, GObject).
162class JSMainThreadNullState {
163 WTF_MAKE_NONCOPYABLE(JSMainThreadNullState);
164 WTF_FORBID_HEAP_ALLOCATION;
165public:
166 explicit JSMainThreadNullState()
167 : m_previousState(JSExecState::currentState())
168 , m_customElementReactionStack(m_previousState)
169 {
170 ASSERT(isMainThread());
171 JSExecState::setCurrentState(nullptr);
172 }
173
174 ~JSMainThreadNullState()
175 {
176 ASSERT(isMainThread());
177 JSExecState::setCurrentState(m_previousState);
178 }
179
180private:
181 JSC::JSGlobalObject* m_previousState;
182 CustomElementReactionStack m_customElementReactionStack;
183};
