33namespace WebCore {
34
35inline void JSExecState::instrumentFunction(ScriptExecutionContext* context, const JSC::CallData& callData)
36{
37 if (!InspectorInstrumentation::timelineAgentTracking(context))
38 return;
39
40 String resourceName;
41 int lineNumber = 1;
42 int columnNumber = 1;
43 if (callData.type == JSC::CallData::Type::JS) {
44 resourceName = callData.js.functionExecutable->sourceURL();
45 lineNumber = callData.js.functionExecutable->firstLine();
46 columnNumber = callData.js.functionExecutable->startColumn();
47 } else
48 resourceName = "undefined";
49 InspectorInstrumentation::willCallFunction(context, resourceName, lineNumber, columnNumber);
50}
