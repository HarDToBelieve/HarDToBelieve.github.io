40static JSC_DECLARE_HOST_FUNCTION(callJSHTMLAllCollection);
41
42// https://html.spec.whatwg.org/multipage/common-dom-interfaces.html#HTMLAllCollection-call
43JSC_DEFINE_HOST_FUNCTION(callJSHTMLAllCollection, (JSGlobalObject* lexicalGlobalObject, CallFrame* callFrame))
44{
45 VM& vm = lexicalGlobalObject->vm();
46 auto scope = DECLARE_THROW_SCOPE(vm);
47
48 auto* castedThis = jsCast<JSHTMLAllCollection*>(callFrame->jsCallee());
49 ASSERT(castedThis);
50 auto& impl = castedThis->wrapped();
51 if (callFrame->argument(0).isUndefined())
52 return JSValue::encode(jsNull());
53
54 AtomString nameOrIndex = callFrame->uncheckedArgument(0).toString(lexicalGlobalObject)->toAtomString(lexicalGlobalObject);
55 RETURN_IF_EXCEPTION(scope, { });
56 RELEASE_AND_RETURN(scope, JSValue::encode(toJS<IDLNullable<IDLUnion<IDLInterface<HTMLCollection>, IDLInterface<Element>>>>(*lexicalGlobalObject, *castedThis->globalObject(), impl.namedOrIndexedItemOrItems(WTFMove(nameOrIndex)))));
57}
58
59CallData JSHTMLAllCollection::getCallData(JSCell*)
60{
61 CallData callData;
62 callData.type = CallData::Type::Native;
63 callData.native.function = callJSHTMLAllCollection;
64 return callData;
65}
