30namespace WebCore {
31using namespace JSC;

33JSValue toJSNewlyCreated(JSGlobalObject* lexicalGlobalObject, JSDOMGlobalObject* globalObject, Ref<HTMLDocument>&& passedDocument)
34{
35 auto& document = passedDocument.get();
36 auto* wrapper = createWrapper<HTMLDocument>(globalObject, WTFMove(passedDocument));
37 reportMemoryForDocumentIfFrameless(*lexicalGlobalObject, document);
38 return wrapper;
39}
40
41JSValue toJS(JSGlobalObject* lexicalGlobalObject, JSDOMGlobalObject* globalObject, HTMLDocument& document)
42{
43 if (auto* wrapper = cachedDocumentWrapper(*lexicalGlobalObject, *globalObject, document))
44 return wrapper;
45 return toJSNewlyCreated(lexicalGlobalObject, globalObject, Ref<HTMLDocument>(document));
46}
