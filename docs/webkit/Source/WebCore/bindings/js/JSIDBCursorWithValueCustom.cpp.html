36JSC::JSValue JSIDBCursorWithValue::value(JSC::JSGlobalObject& lexicalGlobalObject) const
37{
38 return cachedPropertyValue(lexicalGlobalObject, *this, wrapped().valueWrapper(), [&] {
39 auto result = deserializeIDBValueWithKeyInjection(lexicalGlobalObject, wrapped().value(), wrapped().primaryKey(), wrapped().primaryKeyPath());
40 return result ? result.value() : jsNull();
41 });
42}
