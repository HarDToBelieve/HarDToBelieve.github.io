39JSC::JSValue JSIDBRequest::result(JSC::JSGlobalObject& lexicalGlobalObject) const
40{
41 return cachedPropertyValue(lexicalGlobalObject, *this, wrapped().resultWrapper(), [&] {
42 auto result = wrapped().result();
43 if (UNLIKELY(result.hasException())) {
44 auto throwScope = DECLARE_THROW_SCOPE(lexicalGlobalObject.vm());
45 propagateException(lexicalGlobalObject, throwScope, result.releaseException());
46 return jsNull();
47 }
48
49 IDBRequest::Result resultValue = result.releaseReturnValue();
50 return WTF::switchOn(resultValue, [&lexicalGlobalObject] (RefPtr<IDBCursor>& cursor) {
51 auto throwScope = DECLARE_THROW_SCOPE(lexicalGlobalObject.vm());
52 return toJS<IDLInterface<IDBCursor>>(lexicalGlobalObject, *jsCast<JSDOMGlobalObject*>(&lexicalGlobalObject), throwScope, cursor.get());
53 }, [&lexicalGlobalObject] (RefPtr<IDBDatabase>& database) {
54 auto throwScope = DECLARE_THROW_SCOPE(lexicalGlobalObject.vm());
55 return toJS<IDLInterface<IDBDatabase>>(lexicalGlobalObject, *jsCast<JSDOMGlobalObject*>(&lexicalGlobalObject), throwScope, database.get());
56 }, [&lexicalGlobalObject] (IDBKeyData keyData) {
57 return toJS<IDLIDBKeyData>(lexicalGlobalObject, *jsCast<JSDOMGlobalObject*>(&lexicalGlobalObject), keyData);
58 }, [&lexicalGlobalObject] (Vector<IDBKeyData> keyDatas) {
59 return toJS<IDLSequence<IDLIDBKeyData>>(lexicalGlobalObject, *jsCast<JSDOMGlobalObject*>(&lexicalGlobalObject), keyDatas);
60 }, [&lexicalGlobalObject] (IDBGetResult getResult) {
61 auto result = deserializeIDBValueWithKeyInjection(lexicalGlobalObject, getResult.value(), getResult.keyData(), getResult.keyPath());
62 return result ? result.value() : jsNull();
63 }, [&lexicalGlobalObject] (IDBGetAllResult getAllResult) {
64 auto& keys = getAllResult.keys();
65 auto& values = getAllResult.values();
66 auto& keyPath = getAllResult.keyPath();
67 auto scope = DECLARE_THROW_SCOPE(lexicalGlobalObject.vm());
68 JSC::MarkedArgumentBuffer list;
69 for (unsigned i = 0; i < values.size(); i ++) {
70 auto result = deserializeIDBValueWithKeyInjection(lexicalGlobalObject, values[i], keys[i], keyPath);
71 if (!result)
72 return jsNull();
73 list.append(result.value());
74 if (UNLIKELY(list.hasOverflowed())) {
75 propagateException(lexicalGlobalObject, scope, Exception(UnknownError));
76 return jsNull();
77 }
78 }
79 return JSValue(JSC::constructArray(&lexicalGlobalObject, static_cast<JSC::ArrayAllocationProfile*>(nullptr), list));
80 }, [] (uint64_t number) {
81 return toJS<IDLUnsignedLongLong>(number);
82 }, [] (IDBRequest::NullResultType other) {
83 if (other == IDBRequest::NullResultType::Empty)
84 return JSC::jsNull();
85 return JSC::jsUndefined();
86 });
87 });
88}

90template<typename Visitor>
91void JSIDBRequest::visitAdditionalChildren(Visitor& visitor)
92{
93 auto& request = wrapped();
94 request.resultWrapper().visit(visitor);
95 request.cursorWrapper().visit(visitor);
96}
