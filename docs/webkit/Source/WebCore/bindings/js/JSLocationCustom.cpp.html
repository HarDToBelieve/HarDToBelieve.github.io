38static bool getOwnPropertySlotCommon(JSLocation& thisObject, JSGlobalObject& lexicalGlobalObject, PropertyName propertyName, PropertySlot& slot)
39{
40 VM& vm = lexicalGlobalObject.vm();
41 auto scope = DECLARE_THROW_SCOPE(vm);
42
43 auto* window = thisObject.wrapped().window();
44
45 // When accessing Location cross-domain, functions are always the native built-in ones.
46 // See JSDOMWindow::getOwnPropertySlotDelegate for additional details.
47
48 // Our custom code is only needed to implement the Window cross-domain scheme, so if access is
49 // allowed, return false so the normal lookup will take place.
50 String message;
51 if (BindingSecurity::shouldAllowAccessToDOMWindow(lexicalGlobalObject, window, message))
52 return false;
53
54 // https://html.spec.whatwg.org/#crossorigingetownpropertyhelper-(-o,-p-)
55
56 // We only allow access to Location.replace() cross origin.
57 if (propertyName == vm.propertyNames->replace) {
58 auto* entry = JSLocation::info()->staticPropHashTable->entry(propertyName);
59 auto* jsFunction = thisObject.globalObject()->createCrossOriginFunction(&lexicalGlobalObject, propertyName, entry->function(), entry->functionLength());
60 slot.setValue(&thisObject, PropertyAttribute::ReadOnly | PropertyAttribute::DontEnum, jsFunction);
61 return true;
62 }
63
64 // Getting location.href cross origin needs to throw. However, getOwnPropertyDescriptor() needs to return
65 // a descriptor that has a setter but no getter.
66 if (slot.internalMethodType() == PropertySlot::InternalMethodType::GetOwnProperty && propertyName == static_cast<JSVMClientData*>(vm.clientData)->builtinNames().hrefPublicName()) {
67 auto* entry = JSLocation::info()->staticPropHashTable->entry(propertyName);
68 auto* getterSetter = thisObject.globalObject()->createCrossOriginGetterSetter(&lexicalGlobalObject, propertyName, nullptr, entry->propertyPutter());
69 slot.setGetterSlot(&thisObject, PropertyAttribute::Accessor | PropertyAttribute::DontEnum, getterSetter);
70 return true;
71 }
72
73 if (handleCommonCrossOriginProperties(&thisObject, vm, propertyName, slot))
74 return true;
75
76 throwSecurityError(lexicalGlobalObject, scope, message);
77 slot.setUndefined();
78 return false;
79}

81bool JSLocation::getOwnPropertySlot(JSObject* object, JSGlobalObject* lexicalGlobalObject, PropertyName propertyName, PropertySlot& slot)
82{
83 VM& vm = lexicalGlobalObject->vm();
84 auto scope = DECLARE_THROW_SCOPE(vm);
85 auto* thisObject = jsCast<JSLocation*>(object);
86 ASSERT_GC_OBJECT_INHERITS(thisObject, info());
87
88 bool result = getOwnPropertySlotCommon(*thisObject, *lexicalGlobalObject, propertyName, slot);
89 EXCEPTION_ASSERT(!scope.exception() || !result);
90 RETURN_IF_EXCEPTION(scope, false);
91 if (result)
92 return true;
93 RELEASE_AND_RETURN(scope, JSObject::getOwnPropertySlot(object, lexicalGlobalObject, propertyName, slot));
94}

96bool JSLocation::getOwnPropertySlotByIndex(JSObject* object, JSGlobalObject* lexicalGlobalObject, unsigned index, PropertySlot& slot)
97{
98 VM& vm = lexicalGlobalObject->vm();
99 auto scope = DECLARE_THROW_SCOPE(vm);
100 auto* thisObject = jsCast<JSLocation*>(object);
101 ASSERT_GC_OBJECT_INHERITS(thisObject, info());

103 bool result = getOwnPropertySlotCommon(*thisObject, *lexicalGlobalObject, Identifier::from(vm, index), slot);
104 EXCEPTION_ASSERT(!scope.exception() || !result);
105 RETURN_IF_EXCEPTION(scope, false);
106 if (result)
107 return true;
108 RELEASE_AND_RETURN(scope, JSObject::getOwnPropertySlotByIndex(object, lexicalGlobalObject, index, slot));
109}

111bool JSLocation::put(JSCell* cell, JSGlobalObject* lexicalGlobalObject, PropertyName propertyName, JSValue value, PutPropertySlot& putPropertySlot)
112{
113 VM& vm = lexicalGlobalObject->vm();
114 auto* thisObject = jsCast<JSLocation*>(cell);
115 ASSERT_GC_OBJECT_INHERITS(thisObject, info());
116
117 // Always allow assigning to the whole location.
118 // However, allowing assigning of pieces might inadvertently disclose parts of the original location.
119 // So fall through to the access check for those.
120 if (propertyName != static_cast<JSVMClientData*>(vm.clientData)->builtinNames().hrefPublicName()) {
121 if (!BindingSecurity::shouldAllowAccessToDOMWindow(lexicalGlobalObject, thisObject->wrapped().window(), ThrowSecurityError))
122 return false;
123 }
124
125 return JSObject::put(thisObject, lexicalGlobalObject, propertyName, value, putPropertySlot);
126}

157void JSLocation::getOwnPropertyNames(JSObject* object, JSGlobalObject* lexicalGlobalObject, PropertyNameArray& propertyNames, DontEnumPropertiesMode mode)
158{
159 JSLocation* thisObject = jsCast<JSLocation*>(object);
160 if (!BindingSecurity::shouldAllowAccessToDOMWindow(lexicalGlobalObject, thisObject->wrapped().window(), DoNotReportSecurityError)) {
161 if (mode == DontEnumPropertiesMode::Include)
162 addCrossOriginOwnPropertyNames<CrossOriginObject::Location>(*lexicalGlobalObject, propertyNames);
163 return;
164 }
165 Base::getOwnPropertyNames(thisObject, lexicalGlobalObject, propertyNames, mode);
166}
