33namespace WebCore {
34
35inline void* root(MediaList* mediaList)
36{
37 if (CSSRule* parentRule = mediaList->parentRule())
38 return root(parentRule);
39 if (CSSStyleSheet* parentStyleSheet = mediaList->parentStyleSheet())
40 return root(parentStyleSheet);
41 return mediaList;
42}
