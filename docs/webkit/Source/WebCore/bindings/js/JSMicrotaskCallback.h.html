28#pragma once
29
30#include "JSExecState.h"
31
32namespace WebCore {
33
34class JSMicrotaskCallback : public RefCounted<JSMicrotaskCallback> {
35public:
36 static Ref<JSMicrotaskCallback> create(JSDOMGlobalObject& globalObject, Ref<JSC::Microtask>&& task)
37 {
38 return adoptRef(*new JSMicrotaskCallback(globalObject, WTFMove(task)));
39 }
40
41 void call()
42 {
43 auto protectedThis { makeRef(*this) };
44 JSC::VM& vm = m_globalObject->vm();
45 JSC::JSLockHolder lock(vm);
46 auto scope = DECLARE_CATCH_SCOPE(vm);
47 JSExecState::runTask(m_globalObject.get(), m_task);
48 scope.assertNoExceptionExceptTermination();
49 }
50
51private:
52 JSMicrotaskCallback(JSDOMGlobalObject& globalObject, Ref<JSC::Microtask>&& task)
53 : m_globalObject { globalObject.vm(), &globalObject }
54 , m_task { WTFMove(task) }
55 {
56 }
57
58 JSC::Strong<JSDOMGlobalObject> m_globalObject;
59 Ref<JSC::Microtask> m_task;
60};
61
62} // namespace WebCore
