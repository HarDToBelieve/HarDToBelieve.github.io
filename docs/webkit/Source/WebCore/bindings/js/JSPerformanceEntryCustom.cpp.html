51JSValue toJSNewlyCreated(JSGlobalObject*, JSDOMGlobalObject* globalObject, Ref<PerformanceEntry>&& entry)
52{
53 switch (entry->performanceEntryType()) {
54 case PerformanceEntry::Type::Navigation:
55 return createWrapper<PerformanceNavigationTiming>(globalObject, WTFMove(entry));
56 case PerformanceEntry::Type::Mark:
57 return createWrapper<PerformanceMark>(globalObject, WTFMove(entry));
58 case PerformanceEntry::Type::Measure:
59 return createWrapper<PerformanceMeasure>(globalObject, WTFMove(entry));
60 case PerformanceEntry::Type::Resource:
61 return createWrapper<PerformanceResourceTiming>(globalObject, WTFMove(entry));
62 case PerformanceEntry::Type::Paint:
63 return createWrapper<PerformancePaintTiming>(globalObject, WTFMove(entry));
64 }
65
66 ASSERT_NOT_REACHED();
67 return createWrapper<PerformanceEntry>(globalObject, WTFMove(entry));
68}
69
70JSValue toJS(JSC::JSGlobalObject* lexicalGlobalObject, JSDOMGlobalObject* globalObject, PerformanceEntry& entry)
71{
72 return wrap(lexicalGlobalObject, globalObject, entry);
73}
