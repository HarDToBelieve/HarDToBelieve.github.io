36// JavaScript access to plug-in-exported properties for JSHTMLEmbedElement and JSHTMLObjectElement.
37
38static JSC_DECLARE_HOST_FUNCTION(callPlugin);
39static JSC_DECLARE_CUSTOM_GETTER(pluginElementPropertyGetter);
40
41Instance* pluginInstance(HTMLElement& element)
42{
43 // The plugin element holds an owning reference, so we don't have to.
44 if (!is<HTMLPlugInElement>(element))
45 return nullptr;
46 auto* instance = downcast<HTMLPlugInElement>(element).bindingsInstance();
47 if (!instance || !instance->rootObject())
48 return nullptr;
49 return instance;
50}
51
52static JSObject* pluginScriptObjectFromPluginViewBase(HTMLPlugInElement& pluginElement, JSGlobalObject* globalObject)
53{
54 Widget* pluginWidget = pluginElement.pluginWidget();
55 if (!is<PluginViewBase>(pluginWidget))
56 return nullptr;
57
58 return downcast<PluginViewBase>(*pluginWidget).scriptObject(globalObject);
59}
60
61static JSObject* pluginScriptObjectFromPluginViewBase(JSHTMLElement* jsHTMLElement)
62{
63 HTMLElement& element = jsHTMLElement->wrapped();
64 if (!is<HTMLPlugInElement>(element))
65 return nullptr;
66
67 HTMLPlugInElement& pluginElement = downcast<HTMLPlugInElement>(element);
68 return pluginScriptObjectFromPluginViewBase(pluginElement, jsHTMLElement->globalObject());
69}
70
71JSObject* pluginScriptObject(JSGlobalObject* lexicalGlobalObject, JSHTMLElement* jsHTMLElement)
72{
73 HTMLElement& element = jsHTMLElement->wrapped();
74 if (!is<HTMLPlugInElement>(element))
75 return nullptr;
76
77 auto& pluginElement = downcast<HTMLPlugInElement>(element);
78
79 // Choke point for script/plugin interaction; notify DOMTimer of the event.
80 DOMTimer::scriptDidInteractWithPlugin(pluginElement);
81
82 // First, see if the element has a plug-in replacement with a script.
83 if (auto* scriptObject = pluginElement.scriptObjectForPluginReplacement())
84 return scriptObject;
85
86 // Next, see if we can ask the plug-in view for its script object.
87 if (auto* scriptObject = pluginScriptObjectFromPluginViewBase(pluginElement, jsHTMLElement->globalObject()))
88 return scriptObject;
89
90 // Otherwise, fall back to getting the object from the instance.
91
92 // The plugin element holds an owning reference, so we don't have to.
93 auto* instance = pluginElement.bindingsInstance();
94 if (!instance || !instance->rootObject())
95 return nullptr;
96
97 return instance->createRuntimeObject(lexicalGlobalObject);
98}

100JSC_DEFINE_CUSTOM_GETTER(pluginElementPropertyGetter, (JSGlobalObject* lexicalGlobalObject, EncodedJSValue thisValue, PropertyName propertyName))
101{
102 VM& vm = lexicalGlobalObject->vm();
103 auto scope = DECLARE_THROW_SCOPE(vm);
104
105 JSHTMLElement* thisObject = jsDynamicCast<JSHTMLElement*>(vm, JSValue::decode(thisValue));
106 if (!thisObject)
107 return throwVMTypeError(lexicalGlobalObject, scope);
108 JSObject* scriptObject = pluginScriptObject(lexicalGlobalObject, thisObject);
109 if (!scriptObject)
110 return JSValue::encode(jsUndefined());
111
112 return JSValue::encode(scriptObject->get(lexicalGlobalObject, propertyName));
113}

115bool pluginElementCustomGetOwnPropertySlot(JSHTMLElement* element, JSGlobalObject* lexicalGlobalObject, PropertyName propertyName, PropertySlot& slot)
116{
117 VM& vm = lexicalGlobalObject->vm();
118 slot.setIsTaintedByOpaqueObject();
119
120 if (propertyName.uid() == vm.propertyNames->toPrimitiveSymbol.impl())
121 return false;
122
123 if (!element->globalObject()->world().isNormal()) {
124 JSValue proto = element->getPrototypeDirect(vm);
125 if (proto.isObject() && JSC::jsCast<JSC::JSObject*>(asObject(proto))->hasProperty(lexicalGlobalObject, propertyName))
126 return false;
127 }
128
129 if (slot.isVMInquiry()) {
130 slot.setValue(element, static_cast<unsigned>(JSC::PropertyAttribute::None), jsUndefined());
131 return false; // Can't execute stuff below because they can call back into JS.
132 }
133
134 JSObject* scriptObject = pluginScriptObject(lexicalGlobalObject, element);
135 if (!scriptObject)
136 return false;
137
138 if (!scriptObject->hasProperty(lexicalGlobalObject, propertyName))
139 return false;
140
141 slot.setCustom(element, JSC::PropertyAttribute::DontDelete | JSC::PropertyAttribute::DontEnum, pluginElementPropertyGetter);
142 return true;
143}

145bool pluginElementCustomPut(JSHTMLElement* element, JSGlobalObject* lexicalGlobalObject, PropertyName propertyName, JSValue value, PutPropertySlot& slot, bool& putResult)
146{
147 JSObject* scriptObject = pluginScriptObject(lexicalGlobalObject, element);
148 if (!scriptObject)
149 return false;
150 if (!scriptObject->hasProperty(lexicalGlobalObject, propertyName))
151 return false;
152 putResult = scriptObject->methodTable(lexicalGlobalObject->vm())->put(scriptObject, lexicalGlobalObject, propertyName, value, slot);
153 return true;
154}

156JSC_DEFINE_HOST_FUNCTION(callPlugin, (JSGlobalObject* lexicalGlobalObject, CallFrame* callFrame))
157{
158 JSHTMLElement* element = jsCast<JSHTMLElement*>(callFrame->jsCallee());
159
160 // Get the plug-in script object.
161 JSObject* scriptObject = pluginScriptObject(lexicalGlobalObject, element);
162 ASSERT(scriptObject);
163
164 size_t argumentCount = callFrame->argumentCount();
165 MarkedArgumentBuffer argumentList;
166 for (size_t i = 0; i < argumentCount; i++)
167 argumentList.append(callFrame->argument(i));
168 ASSERT(!argumentList.hasOverflowed());
169
170 auto callData = getCallData(lexicalGlobalObject->vm(), scriptObject);
171 ASSERT(callData.type == CallData::Type::Native);
172
173 // Call the object.
174 JSValue result = call(lexicalGlobalObject, scriptObject, callData, callFrame->thisValue(), argumentList);
175 return JSValue::encode(result);
176}
177
178CallData pluginElementCustomGetCallData(JSHTMLElement* element)
179{
180 CallData callData;
181
182 // First, ask the plug-in view base for its runtime object.
183 if (JSObject* scriptObject = pluginScriptObjectFromPluginViewBase(element)) {
184 VM& vm = scriptObject->vm();
185 auto scriptObjectCallData = getCallData(vm, scriptObject);
186 if (scriptObjectCallData.type != CallData::Type::None) {
187 callData.type = CallData::Type::Native;
188 callData.native.function = callPlugin;
189 }
190 } else {
191 Instance* instance = pluginInstance(element->wrapped());
192 if (instance && instance->supportsInvokeDefaultMethod()) {
193 callData.type = CallData::Type::Native;
194 callData.native.function = callPlugin;
195 }
196 }
197
198 return callData;
199}
