43JSValue JSPopStateEvent::state(JSGlobalObject& lexicalGlobalObject) const
44{
45 if (m_state) {
46 // We cannot use a cached object if we are in a different world than the one it was created in.
47 if (isWorldCompatible(lexicalGlobalObject, m_state.get()))
48 return m_state.get();
49 ASSERT_NOT_REACHED();
50 }
51
52 // Save the lexicalGlobalObject value to the m_state member of a JSPopStateEvent, and return it, for convenience.
53 auto cacheState = [&lexicalGlobalObject, this] (JSC::JSValue eventState) {
54 m_state.set(lexicalGlobalObject.vm(), this, eventState);
55 return eventState;
56 };
57
58 PopStateEvent& event = wrapped();
59
60 if (JSC::JSValue eventState = event.state()) {
61 // We need to make sure a PopStateEvent does not leak objects in its lexicalGlobalObject property across isolated DOM worlds.
62 // Ideally, we would check that the worlds have different privileges but that's not possible yet.
63 if (!isWorldCompatible(lexicalGlobalObject, eventState)) {
64 if (auto serializedValue = event.trySerializeState(lexicalGlobalObject))
65 eventState = serializedValue->deserialize(lexicalGlobalObject, globalObject());
66 else
67 eventState = jsNull();
68 }
69 return cacheState(eventState);
70 }
71
72 History* history = event.history();
73 if (!history || !event.serializedState())
74 return cacheState(jsNull());
75
76 // There's no cached value from a previous invocation, nor a lexicalGlobalObject value was provided by the
77 // event, but there is a history object, so first we need to see if the lexicalGlobalObject object has been
78 // deserialized through the history object already.
79 // The current history lexicalGlobalObject object might've changed in the meantime, so we need to take care
80 // of using the correct one, and always share the same deserialization with history.lexicalGlobalObject.
81
82 bool isSameState = history->isSameAsCurrentState(event.serializedState());
83 JSValue result;
84
85 if (isSameState) {
86 JSHistory* jsHistory = jsCast<JSHistory*>(toJS(&lexicalGlobalObject, globalObject(), *history).asCell());
87 result = jsHistory->state(lexicalGlobalObject);
88 } else
89 result = event.serializedState()->deserialize(lexicalGlobalObject, globalObject());
90
91 return cacheState(result);
92}
