38JSValue JSRTCRtpSFrameTransform::setEncryptionKey(JSGlobalObject& lexicalGlobalObject, CallFrame& callFrame, Ref<DeferredPromise>&& promise)
39{
40 auto& vm = getVM(&lexicalGlobalObject);
41 auto throwScope = DECLARE_THROW_SCOPE(vm);
42
43 if (UNLIKELY(callFrame.argumentCount() < 1)) {
44 throwVMError(&lexicalGlobalObject, throwScope, createNotEnoughArgumentsError(&lexicalGlobalObject));
45 return jsUndefined();
46 }
47
48 EnsureStillAliveScope argument0 = callFrame.uncheckedArgument(0);
49 auto key = convert<IDLInterface<CryptoKey>>(lexicalGlobalObject, argument0.value(), [](auto& lexicalGlobalObject, auto& scope) {
50 throwArgumentTypeError(lexicalGlobalObject, scope, 0, "key", "SFrameTransform", "setEncryptionKey", "CryptoKey");
51 });
52 RETURN_IF_EXCEPTION(throwScope, jsUndefined());
53
54 EnsureStillAliveScope argument1 = callFrame.argument(1);
55 std::optional<uint64_t> keyID;
56 if (!argument1.value().isUndefined()) {
57 if (argument1.value().isBigInt()) {
58 if (argument1.value().asHeapBigInt()->length() > 1) {
59 throwException(&lexicalGlobalObject, throwScope, createDOMException(&lexicalGlobalObject, RangeError, "Not a 64 bits integer"_s));
60 return jsUndefined();
61 }
62 keyID = JSBigInt::toBigUInt64(argument1.value());
63 } else
64 keyID = std::optional<Converter<IDLUnsignedLongLong>::ReturnType>(convert<IDLUnsignedLongLong>(lexicalGlobalObject, argument1.value()));
65 }
66 RETURN_IF_EXCEPTION(throwScope, jsUndefined());
67 throwScope.release();
68
69 wrapped().setEncryptionKey(*key, keyID, WTFMove(promise));
70 return jsUndefined();
71}
