37JSValue JSReadableStreamSource::start(JSGlobalObject& lexicalGlobalObject, CallFrame& callFrame, Ref<DeferredPromise>&& promise)
38{
39 VM& vm = lexicalGlobalObject.vm();
40
41 // FIXME: Why is it ok to ASSERT the argument count here?
42 ASSERT(callFrame.argumentCount());
43 JSReadableStreamDefaultController* controller = jsDynamicCast<JSReadableStreamDefaultController*>(vm, callFrame.uncheckedArgument(0));
44 ASSERT(controller);
45
46 m_controller.set(vm, this, controller);
47
48 wrapped().start(ReadableStreamDefaultController(controller), WTFMove(promise));
49
50 return jsUndefined();
51}
52
53JSValue JSReadableStreamSource::pull(JSGlobalObject&, CallFrame&, Ref<DeferredPromise>&& promise)
54{
55 wrapped().pull(WTFMove(promise));
56 return jsUndefined();
57}

59JSValue JSReadableStreamSource::controller(JSGlobalObject&) const
60{
61 ASSERT_NOT_REACHED();
62 return jsUndefined();
63}
