36bool JSRemoteDOMWindow::getOwnPropertySlot(JSObject* object, JSGlobalObject* lexicalGlobalObject, PropertyName propertyName, PropertySlot& slot)
37{
38 if (std::optional<unsigned> index = parseIndex(propertyName))
39 return getOwnPropertySlotByIndex(object, lexicalGlobalObject, index.value(), slot);
40
41 auto* thisObject = jsCast<JSRemoteDOMWindow*>(object);
42 return jsDOMWindowGetOwnPropertySlotRestrictedAccess<DOMWindowType::Remote>(thisObject, thisObject->wrapped(), *lexicalGlobalObject, propertyName, slot, String());
43}

45bool JSRemoteDOMWindow::getOwnPropertySlotByIndex(JSObject* object, JSGlobalObject* lexicalGlobalObject, unsigned index, PropertySlot& slot)
46{
47 VM& vm = lexicalGlobalObject->vm();
48 auto* thisObject = jsCast<JSRemoteDOMWindow*>(object);
49
50 // Indexed getters take precendence over regular properties, so caching would be invalid.
51 slot.disableCaching();
52
53 // FIXME: Add support for indexed properties.
54
55 return jsDOMWindowGetOwnPropertySlotRestrictedAccess<DOMWindowType::Remote>(thisObject, thisObject->wrapped(), *lexicalGlobalObject, Identifier::from(vm, index), slot, String());
56}

58bool JSRemoteDOMWindow::put(JSCell* cell, JSGlobalObject* lexicalGlobalObject, PropertyName propertyName, JSValue value, PutPropertySlot& slot)
59{
60 VM& vm = lexicalGlobalObject->vm();
61 auto scope = DECLARE_THROW_SCOPE(vm);
62
63 auto* thisObject = jsCast<JSRemoteDOMWindow*>(cell);
64 if (!thisObject->wrapped().frame())
65 return false;
66
67 String errorMessage;
68
69 // We only allow setting "location" attribute cross-origin.
70 if (propertyName == static_cast<JSVMClientData*>(vm.clientData)->builtinNames().locationPublicName()) {
71 auto* setter = s_info.staticPropHashTable->entry(propertyName)->propertyPutter();
72 scope.release();
73 setter(lexicalGlobalObject, JSValue::encode(slot.thisValue()), JSValue::encode(value), propertyName);
74 return true;
75 }
76
77 throwSecurityError(*lexicalGlobalObject, scope, errorMessage);
78 return false;
79}
