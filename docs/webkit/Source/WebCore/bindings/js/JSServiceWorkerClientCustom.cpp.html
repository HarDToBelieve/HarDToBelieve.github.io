38JSValue toJSNewlyCreated(JSGlobalObject*, JSDOMGlobalObject* globalObject, Ref<ServiceWorkerClient>&& client)
39{
40 if (is<ServiceWorkerWindowClient>(client))
41 return createWrapper<ServiceWorkerWindowClient>(globalObject, WTFMove(client));
42 return createWrapper<ServiceWorkerClient>(globalObject, WTFMove(client));
43}

45JSValue toJS(JSGlobalObject* lexicalGlobalObject, JSDOMGlobalObject* globalObject, ServiceWorkerClient& client)
46{
47 return wrap(lexicalGlobalObject, globalObject, client);
48}
