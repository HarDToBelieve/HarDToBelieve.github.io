33template<typename Visitor>
34void JSStyleSheet::visitAdditionalChildren(Visitor& visitor)
35{
36 visitor.addOpaqueRoot(root(&wrapped()));
37}
38
39DEFINE_VISIT_ADDITIONAL_CHILDREN(JSStyleSheet);
40
41JSC::JSValue toJSNewlyCreated(JSC::JSGlobalObject*, JSDOMGlobalObject* globalObject, Ref<StyleSheet>&& styleSheet)
42{
43 if (styleSheet->isCSSStyleSheet())
44 return createWrapper<CSSStyleSheet>(globalObject, WTFMove(styleSheet));
45 return createWrapper<StyleSheet>(globalObject, WTFMove(styleSheet));
46}
47
48JSC::JSValue toJS(JSC::JSGlobalObject* lexicalGlobalObject, JSDOMGlobalObject* globalObject, StyleSheet& stylesheet)
49{
50 return wrap(lexicalGlobalObject, globalObject, stylesheet);
51}
