41bool JSTextTrackCueOwner::isReachableFromOpaqueRoots(JSC::Handle<JSC::Unknown> handle, void*, AbstractSlotVisitor& visitor, const char** reason)
42{
43 JSTextTrackCue* jsTextTrackCue = jsCast<JSTextTrackCue*>(handle.slot()->asCell());
44 TextTrackCue& textTrackCue = jsTextTrackCue->wrapped();
45
46 if (!textTrackCue.isContextStopped() && textTrackCue.hasPendingActivity()) {
47 if (UNLIKELY(reason))
48 *reason = "TextTrackCue with pending activity";
49 return true;
50 }
51
52 // If the cue is not associated with a track, it is not reachable.
53 if (!textTrackCue.track())
54 return false;
55
56 if (UNLIKELY(reason))
57 *reason = "TextTrack is an opaque root";
58
59 return visitor.containsOpaqueRoot(root(textTrackCue.track()));
60}

62JSValue toJSNewlyCreated(JSGlobalObject*, JSDOMGlobalObject* globalObject, Ref<TextTrackCue>&& cue)
63{
64 switch (cue->cueType()) {
65 case TextTrackCue::Data:
66 return createWrapper<DataCue>(globalObject, WTFMove(cue));
67 case TextTrackCue::WebVTT:
68 case TextTrackCue::ConvertedToWebVTT:
69 return createWrapper<VTTCue>(globalObject, WTFMove(cue));
70 case TextTrackCue::Generic:
71 return createWrapper<TextTrackCue>(globalObject, WTFMove(cue));
72 }
73
74 ASSERT_NOT_REACHED();
75 return jsNull();
76}
77
78JSValue toJS(JSGlobalObject* lexicalGlobalObject, JSDOMGlobalObject* globalObject, TextTrackCue& cue)
79{
80 return wrap(lexicalGlobalObject, globalObject, cue);
81}

83template<typename Visitor>
84void JSTextTrackCue::visitAdditionalChildren(Visitor& visitor)
85{
86 if (TextTrack* textTrack = wrapped().track())
87 visitor.addOpaqueRoot(root(textTrack));
88}
