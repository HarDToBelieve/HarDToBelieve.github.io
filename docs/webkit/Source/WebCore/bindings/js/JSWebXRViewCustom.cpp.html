37JSC::JSValue JSWebXRView::projectionMatrix(JSC::JSGlobalObject& lexicalGlobalObject) const
38{
39 auto throwScope = DECLARE_THROW_SCOPE(lexicalGlobalObject.vm());
40 return cachedPropertyValue(lexicalGlobalObject, *this, wrapped().cachedProjectionMatrix(), [&] {
41 JSC::JSValue matrix = toJS<IDLFloat32Array>(lexicalGlobalObject, *globalObject(), throwScope, wrapped().projectionMatrix());
42 return matrix;
43 });
44}
