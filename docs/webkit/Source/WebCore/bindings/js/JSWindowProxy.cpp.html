61void JSWindowProxy::finishCreation(VM& vm, AbstractDOMWindow& window)
62{
63 Base::finishCreation(vm);
64 ASSERT(inherits(vm, info()));
65 setWindow(window);
66}

68JSWindowProxy& JSWindowProxy::create(VM& vm, AbstractDOMWindow& window, DOMWrapperWorld& world)
69{
70 auto& structure = *Structure::create(vm, 0, jsNull(), TypeInfo(PureForwardingProxyType, StructureFlags), info());
71 auto& proxy = *new (NotNull, allocateCell<JSWindowProxy>(vm.heap)) JSWindowProxy(vm, structure, world);
72 proxy.finishCreation(vm, window);
73 return proxy;
74}

81void JSWindowProxy::setWindow(VM& vm, JSDOMGlobalObject& window)
82{
83 ASSERT(window.classInfo(vm) == JSDOMWindow::info() || window.classInfo(vm) == JSRemoteDOMWindow::info());
84 setTarget(vm, &window);
85 structure(vm)->setGlobalObject(vm, &window);
86 GCController::singleton().garbageCollectSoon();
87}
88
89void JSWindowProxy::setWindow(AbstractDOMWindow& domWindow)
90{
91 // Replacing JSDOMWindow via telling JSWindowProxy to use the same DOMWindow it already uses makes no sense,
92 // so we'd better never try to.
93 ASSERT(!window() || &domWindow != &wrapped());
94
95 bool isRemoteDOMWindow = is<RemoteDOMWindow>(domWindow);
96
97 VM& vm = commonVM();
98 auto& prototypeStructure = isRemoteDOMWindow ? *JSRemoteDOMWindowPrototype::createStructure(vm, nullptr, jsNull()) : *JSDOMWindowPrototype::createStructure(vm, nullptr, jsNull());
99
100 // Explicitly protect the prototype so it isn't collected when we allocate the global object.
101 // (Once the global object is fully constructed, it will mark its own prototype.)
102 JSNonFinalObject* prototype = isRemoteDOMWindow ? static_cast<JSNonFinalObject*>(JSRemoteDOMWindowPrototype::create(vm, nullptr, &prototypeStructure)) : static_cast<JSNonFinalObject*>(JSDOMWindowPrototype::create(vm, nullptr, &prototypeStructure));
103 JSC::EnsureStillAliveScope protectedPrototype(prototype);
104
105 JSDOMGlobalObject* window = nullptr;
106 if (isRemoteDOMWindow) {
107 auto& windowStructure = *JSRemoteDOMWindow::createStructure(vm, nullptr, prototype);
108 window = JSRemoteDOMWindow::create(vm, &windowStructure, downcast<RemoteDOMWindow>(domWindow), this);
109 } else {
110 auto& localWindow = downcast<DOMWindow>(domWindow);
111 auto& windowStructure = *JSDOMWindow::createStructure(vm, nullptr, prototype);
112 window = JSDOMWindow::create(vm, &windowStructure, localWindow, this);
113 bool linkedWithNewSDK = true;
114#if PLATFORM(COCOA)
115 linkedWithNewSDK = linkedOnOrAfter(SDKVersion::FirstWithDOMWindowReuseRestriction);
116#endif
117 if (!localWindow.document()->haveInitializedSecurityOrigin() && linkedWithNewSDK)
118 localWindow.setAsWrappedWithoutInitializedSecurityOrigin();
119 }
120
121 prototype->structure(vm)->setGlobalObject(vm, window);
122
123 auto& propertiesStructure = *JSDOMWindowProperties::createStructure(vm, window, JSEventTarget::prototype(vm, *window));
124 auto& properties = *JSDOMWindowProperties::create(&propertiesStructure, *window);
125 properties.didBecomePrototype();
126 prototype->structure(vm)->setPrototypeWithoutTransition(vm, &properties);
127
128 setWindow(vm, *window);
129
130 ASSERT(window->globalObject() == window);
131 ASSERT(prototype->globalObject() == window);
132}

134WindowProxy* JSWindowProxy::windowProxy() const
135{
136 auto& window = wrapped();
137 return window.frame() ? &window.frame()->windowProxy() : nullptr;
138}
139
140void JSWindowProxy::attachDebugger(JSC::Debugger* debugger)
141{
142 auto* globalObject = window();
143 JSLockHolder lock(globalObject->vm());
144
145 if (debugger)
146 debugger->attach(globalObject);
147 else if (auto* currentDebugger = globalObject->debugger())
148 currentDebugger->detach(globalObject, JSC::Debugger::TerminatingDebuggingSession);
149}
150
151AbstractDOMWindow& JSWindowProxy::wrapped() const
152{
153 auto* window = this->window();
154 if (auto* jsWindow = jsDynamicCast<JSRemoteDOMWindowBase*>(window->vm(), window))
155 return jsWindow->wrapped();
156 return jsCast<JSDOMWindowBase*>(window)->wrapped();
157}
158
159JSValue toJS(JSGlobalObject* lexicalGlobalObject, WindowProxy& windowProxy)
160{
161 auto* jsWindowProxy = windowProxy.jsWindowProxy(currentWorld(*lexicalGlobalObject));
162 return jsWindowProxy ? JSValue(jsWindowProxy) : jsNull();
163}
164
165JSWindowProxy* toJSWindowProxy(WindowProxy& windowProxy, DOMWrapperWorld& world)
166{
167 return windowProxy.jsWindowProxy(world);
168}
169
170WindowProxy* JSWindowProxy::toWrapped(VM& vm, JSValue value)
171{
172 if (!value.isObject())
173 return nullptr;
174 JSObject* object = asObject(value);
175 if (object->inherits<JSWindowProxy>(vm))
176 return jsCast<JSWindowProxy*>(object)->windowProxy();
177 return nullptr;
178}

180JSC::IsoSubspace* JSWindowProxy::subspaceForImpl(JSC::VM& vm)
181{
182 return &static_cast<JSVMClientData*>(vm.clientData)->windowProxySpace();
