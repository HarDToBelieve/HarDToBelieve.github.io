36template<typename Visitor>
37void JSWorkerGlobalScope::visitAdditionalChildren(Visitor& visitor)
38{
39 if (auto* location = wrapped().optionalLocation())
40 visitor.addOpaqueRoot(location);
41 if (auto* navigator = wrapped().optionalNavigator())
42 visitor.addOpaqueRoot(navigator);
43 ScriptExecutionContext& context = wrapped();
44 visitor.addOpaqueRoot(&context);
45
46 // Normally JSEventTargetCustom.cpp's JSEventTarget::visitAdditionalChildren() would call this. But
47 // even though WorkerGlobalScope is an EventTarget, JSWorkerGlobalScope does not subclass
48 // JSEventTarget, so we need to do this here.
49 wrapped().visitJSEventListeners(visitor);
50}
51
52DEFINE_VISIT_ADDITIONAL_CHILDREN(JSWorkerGlobalScope);

54JSValue JSWorkerGlobalScope::queueMicrotask(JSGlobalObject& lexicalGlobalObject, CallFrame& callFrame)
55{
56 VM& vm = lexicalGlobalObject.vm();
57 auto scope = DECLARE_THROW_SCOPE(vm);
58
59 if (UNLIKELY(callFrame.argumentCount() < 1))
60 return throwException(&lexicalGlobalObject, scope, createNotEnoughArgumentsError(&lexicalGlobalObject));
61
62 JSValue functionValue = callFrame.uncheckedArgument(0);
63 if (UNLIKELY(!functionValue.isCallable(vm)))
64 return JSValue::decode(throwArgumentMustBeFunctionError(lexicalGlobalObject, scope, 0, "callback", "WorkerGlobalScope", "queueMicrotask"));
65
66 scope.release();
67 Base::queueMicrotask(JSC::createJSMicrotask(vm, functionValue));
68 return jsUndefined();
69}
