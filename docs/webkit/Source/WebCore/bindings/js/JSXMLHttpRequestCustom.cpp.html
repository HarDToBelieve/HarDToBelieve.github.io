44template<typename Visitor>
45void JSXMLHttpRequest::visitAdditionalChildren(Visitor& visitor)
46{
47 if (auto* upload = wrapped().optionalUpload())
48 visitor.addOpaqueRoot(upload);
49
50 if (auto* responseDocument = wrapped().optionalResponseXML())
51 visitor.addOpaqueRoot(responseDocument);
52}
53
54DEFINE_VISIT_ADDITIONAL_CHILDREN(JSXMLHttpRequest);

56JSValue JSXMLHttpRequest::response(JSGlobalObject& lexicalGlobalObject) const
57{
58 auto cacheResult = [&] (JSValue value) -> JSValue {
59 m_response.set(lexicalGlobalObject.vm(), this, value);
60 return value;
61 };
62
63
64 if (wrapped().responseCacheIsValid())
65 return m_response.get();
66
67 auto type = wrapped().responseType();
68
69 switch (type) {
70 case XMLHttpRequest::ResponseType::EmptyString:
71 case XMLHttpRequest::ResponseType::Text: {
72 auto scope = DECLARE_THROW_SCOPE(lexicalGlobalObject.vm());
73 return cacheResult(toJS<IDLNullable<IDLUSVString>>(lexicalGlobalObject, scope, wrapped().responseText()));
74 }
75 default:
76 break;
77 }
78
79 if (!wrapped().doneWithoutErrors())
80 return cacheResult(jsNull());
81
82 JSValue value;
83 switch (type) {
84 case XMLHttpRequest::ResponseType::EmptyString:
85 case XMLHttpRequest::ResponseType::Text:
86 ASSERT_NOT_REACHED();
87 return jsUndefined();
88
89 case XMLHttpRequest::ResponseType::Json:
90 value = toJS<IDLJSON>(*globalObject(), wrapped().responseTextIgnoringResponseType());
91 if (!value)
92 value = jsNull();
93 break;
94
95 case XMLHttpRequest::ResponseType::Document: {
96 auto document = wrapped().responseXML();
97 ASSERT(!document.hasException());
98 value = toJS<IDLInterface<Document>>(lexicalGlobalObject, *globalObject(), document.releaseReturnValue());
99 break;
100 }
101
102 case XMLHttpRequest::ResponseType::Blob:
103 value = toJSNewlyCreated<IDLInterface<Blob>>(lexicalGlobalObject, *globalObject(), wrapped().createResponseBlob());
104 break;
105
106 case XMLHttpRequest::ResponseType::Arraybuffer:
107 value = toJS<IDLInterface<ArrayBuffer>>(lexicalGlobalObject, *globalObject(), wrapped().createResponseArrayBuffer());
108 break;
109 }
110
111 wrapped().didCacheResponse();
112 return cacheResult(value);
113}
