35template<typename Visitor>
36void JSXPathResult::visitAdditionalChildren(Visitor& visitor)
37{
38 auto& value = wrapped().value();
39 if (value.isNodeSet()) {
40 // FIXME: This looks like it might race, but I'm not sure.
41 for (auto& node : value.toNodeSet())
42 visitor.addOpaqueRoot(root(node.get()));
43 }
44}
