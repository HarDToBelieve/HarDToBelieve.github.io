36class ReadableStreamSink;

39class ReadableStream final : public DOMGuarded<JSReadableStream> {
40public:
41 static Ref<ReadableStream> create(JSDOMGlobalObject& globalObject, JSReadableStream& readableStream) { return adoptRef(*new ReadableStream(globalObject, readableStream)); }
42
43 static ExceptionOr<Ref<ReadableStream>> create(JSC::JSGlobalObject&, RefPtr<ReadableStreamSource>&&);
44
45 WEBCORE_EXPORT static bool isDisturbed(JSC::JSGlobalObject&, JSC::JSValue);
46
47 std::optional<std::pair<Ref<ReadableStream>, Ref<ReadableStream>>> tee();
48
49 void lock();
50 void pipeTo(ReadableStreamSink&);
51 bool isLocked() const;
52 bool isDisturbed() const;
53
54 JSReadableStream* readableStream() const { return guarded(); }
55
56private:
57 ReadableStream(JSDOMGlobalObject& globalObject, JSReadableStream& readableStream) : DOMGuarded<JSReadableStream>(globalObject, readableStream) { }
58};
59
60struct JSReadableStreamWrapperConverter {
61 static RefPtr<ReadableStream> toWrapped(JSC::JSGlobalObject& lexicalGlobalObject, JSC::JSValue value)
62 {
63 JSC::VM& vm = JSC::getVM(&lexicalGlobalObject);
64 auto* globalObject = JSC::jsDynamicCast<JSDOMGlobalObject*>(vm, &lexicalGlobalObject);
65 if (!globalObject)
66 return nullptr;
67
68 auto* readableStream = JSC::jsDynamicCast<JSReadableStream*>(vm, value);
69 if (!readableStream)
70 return nullptr;
71
72 return ReadableStream::create(*globalObject, *readableStream);
73 }
74};
75
76template<> struct JSDOMWrapperConverterTraits<ReadableStream> {
77 using WrapperClass = JSReadableStreamWrapperConverter;
78 using ToWrappedReturnType = RefPtr<ReadableStream>;
79 static constexpr bool needsState = true;
80};
81
82inline JSC::JSValue toJS(JSC::JSGlobalObject*, JSC::JSGlobalObject*, ReadableStream* stream)
83{
84 return stream ? stream->readableStream() : JSC::jsUndefined();
85}
86
87inline JSC::JSValue toJS(JSC::JSGlobalObject*, JSC::JSGlobalObject*, ReadableStream& stream)
88{
89 return stream.readableStream();
90}
91
92inline JSC::JSValue toJSNewlyCreated(JSC::JSGlobalObject*, JSDOMGlobalObject*, Ref<ReadableStream>&& stream)
93{
94 return stream->readableStream();
95}
