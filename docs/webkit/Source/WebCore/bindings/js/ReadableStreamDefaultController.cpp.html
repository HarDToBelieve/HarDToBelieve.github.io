42static bool invokeReadableStreamDefaultControllerFunction(JSC::JSGlobalObject& lexicalGlobalObject, const JSC::Identifier& identifier, const JSC::MarkedArgumentBuffer& arguments)
43{
44 JSC::VM& vm = lexicalGlobalObject.vm();
45 JSC::JSLockHolder lock(vm);
46
47 auto function = lexicalGlobalObject.get(&lexicalGlobalObject, identifier);
48 ASSERT(function.isCallable(lexicalGlobalObject.vm()));
49
50 auto scope = DECLARE_CATCH_SCOPE(vm);
51 auto callData = JSC::getCallData(vm, function);
52 call(&lexicalGlobalObject, function, callData, JSC::jsUndefined(), arguments);
53 EXCEPTION_ASSERT(!scope.exception() || vm.hasPendingTerminationException());
54 return !scope.exception();
55}

57void ReadableStreamDefaultController::close()
58{
59 JSC::MarkedArgumentBuffer arguments;
60 arguments.append(&jsController());
61
62 auto* clientData = static_cast<JSVMClientData*>(globalObject().vm().clientData);
63 auto& privateName = clientData->builtinFunctions().readableStreamInternalsBuiltins().readableStreamDefaultControllerClosePrivateName();
64
65 invokeReadableStreamDefaultControllerFunction(globalObject(), privateName, arguments);
66}

69void ReadableStreamDefaultController::error(const Exception& exception)
70{
71 JSC::JSGlobalObject& lexicalGlobalObject = this->globalObject();
72 auto& vm = lexicalGlobalObject.vm();
73 JSC::JSLockHolder lock(vm);
74 auto scope = DECLARE_CATCH_SCOPE(vm);
75 auto value = createDOMException(&lexicalGlobalObject, exception.code(), exception.message());
76
77 if (UNLIKELY(scope.exception())) {
78 ASSERT(vm.hasPendingTerminationException());
79 return;
80 }
81
82 JSC::MarkedArgumentBuffer arguments;
83 arguments.append(&jsController());
84 arguments.append(value);
85
86 auto* clientData = static_cast<JSVMClientData*>(vm.clientData);
87 auto& privateName = clientData->builtinFunctions().readableStreamInternalsBuiltins().readableStreamDefaultControllerErrorPrivateName();
88
89 invokeReadableStreamDefaultControllerFunction(globalObject(), privateName, arguments);
90}

92bool ReadableStreamDefaultController::enqueue(JSC::JSValue value)
93{
94 JSC::JSGlobalObject& lexicalGlobalObject = this->globalObject();
95 auto& vm = lexicalGlobalObject.vm();
96 JSC::JSLockHolder lock(vm);
97
98 JSC::MarkedArgumentBuffer arguments;
99 arguments.append(&jsController());
100 arguments.append(value);
101
102 auto* clientData = static_cast<JSVMClientData*>(lexicalGlobalObject.vm().clientData);
103 auto& privateName = clientData->builtinFunctions().readableStreamInternalsBuiltins().readableStreamDefaultControllerEnqueuePrivateName();
104
105 return invokeReadableStreamDefaultControllerFunction(globalObject(), privateName, arguments);
106}
107
108bool ReadableStreamDefaultController::enqueue(RefPtr<JSC::ArrayBuffer>&& buffer)
109{
110 if (!buffer) {
111 error(Exception { OutOfMemoryError });
112 return false;
113 }
114
115 JSC::JSGlobalObject& lexicalGlobalObject = this->globalObject();
116 auto& vm = lexicalGlobalObject.vm();
117 JSC::JSLockHolder lock(vm);
118 auto scope = DECLARE_CATCH_SCOPE(vm);
119 auto length = buffer->byteLength();
120 auto chunk = JSC::Uint8Array::create(WTFMove(buffer), 0, length);
121 auto value = toJS(&lexicalGlobalObject, &lexicalGlobalObject, chunk.get());
122
123 if (UNLIKELY(scope.exception())) {
124 ASSERT(vm.hasPendingTerminationException());
125 return false;
126 }
127
128 return enqueue(value);
129}
