50ScriptCachedFrameData::ScriptCachedFrameData(Frame& frame)
51{
52 JSLockHolder lock(commonVM());
53
54 for (auto windowProxy : frame.windowProxy().jsWindowProxiesAsVector()) {
55 auto* window = jsCast<JSDOMWindow*>(windowProxy->window());
56 m_windows.add(&windowProxy->world(), Strong<JSDOMWindow>(window->vm(), window));
57 window->setConsoleClient(nullptr);
58 }
59
60 frame.windowProxy().attachDebugger(nullptr);
61}
62
63ScriptCachedFrameData::~ScriptCachedFrameData()
64{
65 clear();
66}
67
68void ScriptCachedFrameData::restore(Frame& frame)
69{
70 JSLockHolder lock(commonVM());
71
72 Page* page = frame.page();
73
74 for (auto windowProxy : frame.windowProxy().jsWindowProxiesAsVector()) {
75 auto* world = &windowProxy->world();
76
77 if (auto* window = m_windows.get(world).get())
78 windowProxy->setWindow(window->vm(), *window);
79 else {
80 ASSERT(frame.document()->domWindow());
81 auto& domWindow = *frame.document()->domWindow();
82 if (&windowProxy->wrapped() == &domWindow)
83 continue;
84
85 windowProxy->setWindow(domWindow);
86
87 if (page) {
88 windowProxy->attachDebugger(page->debugger());
89 windowProxy->window()->setProfileGroup(page->group().identifier());
90 }
91 }
92
93 if (page)
94 windowProxy->window()->setConsoleClient(makeWeakPtr(page->console()));
95 }
96}
97
98void ScriptCachedFrameData::clear()
99{
100 if (m_windows.isEmpty())
101 return;
102
103 JSLockHolder lock(commonVM());
104 m_windows.clear();
105 GCController::singleton().garbageCollectSoon();
106}
