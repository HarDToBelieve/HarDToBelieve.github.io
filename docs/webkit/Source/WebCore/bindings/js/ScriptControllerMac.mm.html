63RefPtr<JSC::Bindings::Instance> ScriptController::createScriptInstanceForWidget(Widget* widget)
64{
65 NSView* widgetView = widget->platformWidget();
66 if (!widgetView)
67 return nullptr;
68
69 auto rootObject = createRootObject((__bridge void*)widgetView);
70
71 if ([widgetView respondsToSelector:@selector(createPluginBindingsInstance:)])
72 return [widgetView createPluginBindingsInstance:WTFMove(rootObject)];
73
74 if ([widgetView respondsToSelector:@selector(objectForWebScript)]) {
75 id objectForWebScript = [widgetView objectForWebScript];
76 if (!objectForWebScript)
77 return nullptr;
78 return JSC::Bindings::ObjcInstance::create(objectForWebScript, WTFMove(rootObject));
79 }
80
81 if ([widgetView respondsToSelector:@selector(createPluginScriptableObject)]) {
82#if !ENABLE(NETSCAPE_PLUGIN_API)
83 return nullptr;
84#else
85 NPObject* npObject = [widgetView createPluginScriptableObject];
86 if (!npObject)
87 return nullptr;
88 auto instance = JSC::Bindings::CInstance::create(npObject, WTFMove(rootObject));
89 // -createPluginScriptableObject returns a retained NPObject. The caller is expected to release it.
90 _NPN_ReleaseObject(npObject);
91 return WTFMove(instance);
92#endif
93 }
94
95 return nullptr;
96}

98WebScriptObject *ScriptController::windowScriptObject()
99{
100 if (!canExecuteScripts(NotAboutToExecuteScript))
101 return nil;
102
103 if (!m_windowScriptObject) {
104 JSC::JSLockHolder lock(commonVM());
105 JSC::Bindings::RootObject* root = bindingRootObject();
106 m_windowScriptObject = [WebScriptObject scriptObjectForJSObject:toRef(&jsWindowProxy(pluginWorld())) originRootObject:root rootObject:root];
107 }
108
109 return m_windowScriptObject.get();
110}
111
112JSContext *ScriptController::javaScriptContext()
113{
114#if JSC_OBJC_API_ENABLED
115 if (!canExecuteScripts(NotAboutToExecuteScript))
116 return 0;
117 JSContext *context = [JSContext contextWithJSGlobalContextRef:toGlobalRef(bindingRootObject()->globalObject())];
118 return context;
119#else
120 return 0;
121#endif
122}
123
124void ScriptController::updatePlatformScriptObjects()
125{
126 if (m_windowScriptObject) {
127 JSC::Bindings::RootObject* root = bindingRootObject();
128 [m_windowScriptObject.get() _setOriginRootObject:root andRootObject:root];
129 }
130}
131
132void ScriptController::disconnectPlatformScriptObjects()
133{
134 if (m_windowScriptObject)
135 disconnectWindowWrapper(m_windowScriptObject.get());
136}
