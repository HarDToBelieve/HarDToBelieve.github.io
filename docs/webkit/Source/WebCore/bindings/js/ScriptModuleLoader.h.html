51class ScriptModuleLoader final : private ModuleScriptLoaderClient {
52 WTF_MAKE_NONCOPYABLE(ScriptModuleLoader); WTF_MAKE_FAST_ALLOCATED;
53public:
54 enum class OwnerType : uint8_t { Document, WorkerOrWorklet };
55 explicit ScriptModuleLoader(ScriptExecutionContext&, OwnerType);
56 ~ScriptModuleLoader();

58 ScriptExecutionContext& context() { return m_context; }
59
60 JSC::Identifier resolve(JSC::JSGlobalObject*, JSC::JSModuleLoader*, JSC::JSValue moduleName, JSC::JSValue importerModuleKey, JSC::JSValue scriptFetcher);

62 JSC::JSValue evaluate(JSC::JSGlobalObject*, JSC::JSModuleLoader*, JSC::JSValue moduleKey, JSC::JSValue moduleRecord, JSC::JSValue scriptFetcher, JSC::JSValue awaitedValue, JSC::JSValue resumeMode);
63 JSC::JSInternalPromise* importModule(JSC::JSGlobalObject*, JSC::JSModuleLoader*, JSC::JSString*, JSC::JSValue parameters, const JSC::SourceOrigin&);
64 JSC::JSObject* createImportMetaProperties(JSC::JSGlobalObject*, JSC::JSModuleLoader*, JSC::JSValue, JSC::JSModuleRecord*, JSC::JSValue);
65
66private:
67 void notifyFinished(ModuleScriptLoader&, URL&&, Ref<DeferredPromise>) final;
68
69 URL moduleURL(JSC::JSGlobalObject&, JSC::JSValue);
70 URL responseURLFromRequestURL(JSC::JSGlobalObject&, JSC::JSValue);
71
72 ScriptExecutionContext& m_context;

74 HashSet<Ref<ModuleScriptLoader>> m_loaders;
75 OwnerType m_ownerType;
