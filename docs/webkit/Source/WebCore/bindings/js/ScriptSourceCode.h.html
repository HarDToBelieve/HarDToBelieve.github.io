31#pragma once
32
33#include "CachedResourceHandle.h"
34#include "CachedScript.h"
35#include "CachedScriptFetcher.h"
36#include "CachedScriptSourceProvider.h"
37#include "ScriptBufferSourceProvider.h"
38#include <JavaScriptCore/SourceCode.h>
39#include <JavaScriptCore/SourceProvider.h>
40#include <wtf/RefPtr.h>
41#include <wtf/URL.h>
42#include <wtf/text/TextPosition.h>
43
44namespace WebCore {
45
46class ScriptSourceCode {
47public:
48 ScriptSourceCode(const String& source, URL&& url = URL(), const TextPosition& startPosition = TextPosition(), JSC::SourceProviderSourceType sourceType = JSC::SourceProviderSourceType::Program)
49 : m_provider(JSC::StringSourceProvider::create(source, JSC::SourceOrigin { url }, url.string(), startPosition, sourceType))
50 , m_code(m_provider.copyRef(), startPosition.m_line.oneBasedInt(), startPosition.m_column.oneBasedInt())
51 {
52 }
53
54 ScriptSourceCode(const ScriptBuffer& source, URL&& url = URL(), const TextPosition& startPosition = TextPosition(), JSC::SourceProviderSourceType sourceType = JSC::SourceProviderSourceType::Program)
55 : m_provider(ScriptBufferSourceProvider::create(source, JSC::SourceOrigin { url }, url.string(), startPosition, sourceType))
56 , m_code(m_provider.copyRef(), startPosition.m_line.oneBasedInt(), startPosition.m_column.oneBasedInt())
57 {
58 }
59
60 ScriptSourceCode(CachedScript* cachedScript, JSC::SourceProviderSourceType sourceType, Ref<CachedScriptFetcher>&& scriptFetcher)
61 : m_provider(CachedScriptSourceProvider::create(cachedScript, sourceType, WTFMove(scriptFetcher)))
62 , m_code(m_provider.copyRef())
63 , m_cachedScript(cachedScript)
64 {
65 }
66
67 ScriptSourceCode(const String& source, URL&& url, const TextPosition& startPosition, JSC::SourceProviderSourceType sourceType, Ref<JSC::ScriptFetcher>&& scriptFetcher)
68 : m_provider(JSC::StringSourceProvider::create(source, JSC::SourceOrigin { url, WTFMove(scriptFetcher) }, url.string(), startPosition, sourceType))
69 , m_code(m_provider.copyRef(), startPosition.m_line.oneBasedInt(), startPosition.m_column.oneBasedInt())
70 {
71 }
72
73 ScriptSourceCode(const ScriptBuffer& source, URL&& url, const TextPosition& startPosition, JSC::SourceProviderSourceType sourceType, Ref<JSC::ScriptFetcher>&& scriptFetcher)
74 : m_provider(ScriptBufferSourceProvider::create(source, JSC::SourceOrigin { url, WTFMove(scriptFetcher) }, url.string(), startPosition, sourceType))
75 , m_code(m_provider.copyRef(), startPosition.m_line.oneBasedInt(), startPosition.m_column.oneBasedInt())
76 {
77 }
78
79 bool isEmpty() const { return m_code.length() == 0; }
80
81 const JSC::SourceCode& jsSourceCode() const { return m_code; }
82
83 JSC::SourceProvider& provider() { return m_provider.get(); }
84 StringView source() const { return m_provider->source(); }
85
86 int startLine() const { return m_code.firstLine().oneBasedInt(); }
87 int startColumn() const { return m_code.startColumn().oneBasedInt(); }
88
89 CachedScript* cachedScript() const { return m_cachedScript.get(); }
90
91 const URL& url() const { return m_provider->sourceOrigin().url(); }
92
93private:
94 Ref<JSC::SourceProvider> m_provider;
95 JSC::SourceCode m_code;
96 CachedResourceHandle<CachedScript> m_cachedScript;
97};
98
99} // namespace WebCore
