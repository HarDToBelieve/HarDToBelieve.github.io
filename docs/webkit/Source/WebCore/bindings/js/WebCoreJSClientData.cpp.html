92JSVMClientData::~JSVMClientData()
93{
94 ASSERT(m_worldSet.contains(m_normalWorld.get()));
95 ASSERT(m_worldSet.size() == 1);
96 ASSERT(m_normalWorld->hasOneRef());
97 m_normalWorld = nullptr;
98 ASSERT(m_worldSet.isEmpty());
99}

101void JSVMClientData::getAllWorlds(Vector<Ref<DOMWrapperWorld>>& worlds)
102{
103 ASSERT(worlds.isEmpty());
104
105 worlds.reserveInitialCapacity(m_worldSet.size());
106
107 // It is necessary to order the `DOMWrapperWorld`s because certain callers expect the main world
108 // to be the first item in the list, as they use the main world as an indicator of when the page
109 // is ready to start evaluating JavaScript. For example, Web Inspector waits for the main world
110 // change to clear any injected scripts and debugger/breakpoint state.
111
112 auto& mainNormalWorld = mainThreadNormalWorld();
113
114 // Add main normal world.
115 if (m_worldSet.contains(&mainNormalWorld))
116 worlds.uncheckedAppend(mainNormalWorld);
117
118 // Add other normal worlds.
119 for (auto* world : m_worldSet) {
120 if (world->type() != DOMWrapperWorld::Type::Normal)
121 continue;
122 if (world == &mainNormalWorld)
123 continue;
124 worlds.uncheckedAppend(*world);
125 }
126
127 // Add non-normal worlds.
128 for (auto* world : m_worldSet) {
129 if (world->type() == DOMWrapperWorld::Type::Normal)
130 continue;
131 worlds.uncheckedAppend(*world);
132 }
133}

135void JSVMClientData::initNormalWorld(VM* vm, WorkerThreadType type)
136{
137 JSVMClientData* clientData = new JSVMClientData(*vm);
138 vm->clientData = clientData; // ~VM deletes this pointer.
139
140 vm->heap.addMarkingConstraint(makeUnique<DOMGCOutputConstraint>(*vm, *clientData));
141
142 clientData->m_normalWorld = DOMWrapperWorld::create(*vm, DOMWrapperWorld::Type::Normal);
143 vm->m_typedArrayController = adoptRef(new WebCoreTypedArrayController(type == WorkerThreadType::DedicatedWorker || type == WorkerThreadType::Worklet));
144}
