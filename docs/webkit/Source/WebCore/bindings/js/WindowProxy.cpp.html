43static void collectGarbageAfterWindowProxyDestruction()
44{
45 // Make sure to GC Extra Soon(tm) during memory pressure conditions
46 // to soften high peaks of memory usage during navigation.
47 if (MemoryPressureHandler::singleton().isUnderMemoryPressure()) {
48 // NOTE: We do the collection on next runloop to ensure that there's no pointer
49 // to the window object on the stack.
50 GCController::singleton().garbageCollectOnNextRunLoop();
51 } else
52 GCController::singleton().garbageCollectSoon();
53}

72void WindowProxy::detachFromFrame()
73{
74 ASSERT(m_frame);
75
76 m_frame = nullptr;
77
78 // It's likely that destroying windowProxies will create a lot of garbage.
79 if (!m_jsWindowProxies->isEmpty()) {
80 while (!m_jsWindowProxies->isEmpty()) {
81 auto it = m_jsWindowProxies->begin();
82 it->value->window()->setConsoleClient(nullptr);
83 destroyJSWindowProxy(*it->key);
84 }
85 collectGarbageAfterWindowProxyDestruction();
86 }
87}

89void WindowProxy::destroyJSWindowProxy(DOMWrapperWorld& world)
90{
91 ASSERT(m_jsWindowProxies->contains(&world));
92 m_jsWindowProxies->remove(&world);
93 world.didDestroyWindowProxy(this);
94}
95
96JSWindowProxy& WindowProxy::createJSWindowProxy(DOMWrapperWorld& world)
97{
98 ASSERT(m_frame);
99
100 ASSERT(!m_jsWindowProxies->contains(&world));
101 ASSERT(m_frame->window());
102
103 VM& vm = world.vm();
104
105 Strong<JSWindowProxy> jsWindowProxy(vm, &JSWindowProxy::create(vm, *m_frame->window(), world));
106 m_jsWindowProxies->add(&world, jsWindowProxy);
107 world.didCreateWindowProxy(this);
108 return *jsWindowProxy.get();
109}

123JSWindowProxy& WindowProxy::createJSWindowProxyWithInitializedScript(DOMWrapperWorld& world)
124{
125 ASSERT(m_frame);
126
127 JSLockHolder lock(world.vm());
128 auto& windowProxy = createJSWindowProxy(world);
129 if (is<Frame>(*m_frame))
130 downcast<Frame>(*m_frame).script().initScriptForWindowProxy(windowProxy);
131 return windowProxy;
132}

134void WindowProxy::clearJSWindowProxiesNotMatchingDOMWindow(AbstractDOMWindow* newDOMWindow, bool goingIntoBackForwardCache)
135{
136 if (m_jsWindowProxies->isEmpty())
137 return;
138
139 JSLockHolder lock(commonVM());
140
141 for (auto& windowProxy : jsWindowProxiesAsVector()) {
142 if (&windowProxy->wrapped() == newDOMWindow)
143 continue;
144
145 // Clear the debugger and console from the current window before setting the new window.
146 windowProxy->attachDebugger(nullptr);
147 windowProxy->window()->setConsoleClient(nullptr);
148 if (auto* jsDOMWindow = jsDynamicCast<JSDOMWindowBase*>(windowProxy->vm(), windowProxy->window()))
149 jsDOMWindow->willRemoveFromWindowProxy();
150 }
151
152 // It's likely that resetting our windows created a lot of garbage, unless
153 // it went in a back/forward cache.
154 if (!goingIntoBackForwardCache)
155 collectGarbageAfterWindowProxyDestruction();
156}

158void WindowProxy::setDOMWindow(AbstractDOMWindow* newDOMWindow)
159{
160 ASSERT(newDOMWindow);
161
162 if (m_jsWindowProxies->isEmpty())
163 return;
164
165 ASSERT(m_frame);
166
167 JSLockHolder lock(commonVM());
168
169 for (auto& windowProxy : jsWindowProxiesAsVector()) {
170 if (&windowProxy->wrapped() == newDOMWindow)
171 continue;
172
173 windowProxy->setWindow(*newDOMWindow);
174
175 ScriptController* scriptController = nullptr;
176 Page* page = nullptr;
177 if (is<Frame>(*m_frame)) {
178 auto& frame = downcast<Frame>(*m_frame);
179 scriptController = &frame.script();
180 page = frame.page();
181 }
182
183 // ScriptController's m_cacheableBindingRootObject persists between page navigations
184 // so needs to know about the new JSDOMWindow.
185 if (auto* cacheableBindingRootObject = scriptController ? scriptController->existingCacheableBindingRootObject() : nullptr)
186 cacheableBindingRootObject->updateGlobalObject(windowProxy->window());
187
188 windowProxy->attachDebugger(page ? page->debugger() : nullptr);
189 if (page)
190 windowProxy->window()->setProfileGroup(page->group().identifier());
191 windowProxy->window()->setConsoleClient(page ? makeWeakPtr(page->console()) : nullptr);
192 }
193}
