61bool WorkerModuleScriptLoader::load(ScriptExecutionContext& context, URL&& sourceURL)
62{
63 m_sourceURL = WTFMove(sourceURL);
64
65#if ENABLE(SERVICE_WORKER)
66 if (is<ServiceWorkerGlobalScope>(context)) {
67 if (auto* scriptResource = downcast<ServiceWorkerGlobalScope>(context).scriptResource(m_sourceURL)) {
68 m_script = scriptResource->script;
69 m_responseURL = scriptResource->responseURL;
70 m_responseMIMEType = scriptResource->mimeType;
71 m_retrievedFromServiceWorkerCache = true;
72 notifyClientFinished();
73 return true;
74 }
75 }
76#endif
77
78 ResourceRequest request { m_sourceURL };
79
80 FetchOptions fetchOptions;
81 fetchOptions.mode = FetchOptions::Mode::Cors;
82 fetchOptions.cache = FetchOptions::Cache::Default;
83 fetchOptions.redirect = FetchOptions::Redirect::Follow;
84 fetchOptions.credentials = static_cast<WorkerScriptFetcher&>(scriptFetcher()).credentials();
85 fetchOptions.destination = static_cast<WorkerScriptFetcher&>(scriptFetcher()).destination();
86 fetchOptions.referrerPolicy = static_cast<WorkerScriptFetcher&>(scriptFetcher()).referrerPolicy();
87 auto contentSecurityPolicyEnforcement = context.shouldBypassMainWorldContentSecurityPolicy() ? ContentSecurityPolicyEnforcement::DoNotEnforce : ContentSecurityPolicyEnforcement::EnforceChildSrcDirective;
88
89 // https://html.spec.whatwg.org/multipage/webappapis.html#fetch-a-single-module-script
90 // If destination is "worker" or "sharedworker" and the top-level module fetch flag is set, then set request's mode to "same-origin".
91 if (fetchOptions.destination == FetchOptions::Destination::Worker || fetchOptions.destination == FetchOptions::Destination::Serviceworker) {
92 if (parameters() && parameters()->isTopLevelModule())
93 fetchOptions.mode = FetchOptions::Mode::SameOrigin;
94 }
95
96 m_scriptLoader->loadAsynchronously(context, WTFMove(request), WTFMove(fetchOptions), contentSecurityPolicyEnforcement, ServiceWorkersMode::All, *this, taskMode());
97 return true;
98}

100ReferrerPolicy WorkerModuleScriptLoader::referrerPolicy()
101{
102 if (auto policy = parseReferrerPolicy(m_scriptLoader->referrerPolicy(), ReferrerPolicySource::HTTPHeader))
103 return *policy;
104 return ReferrerPolicy::EmptyString;
105}

107void WorkerModuleScriptLoader::notifyFinished()
108{
109 ASSERT(m_promise);
110
111 if (m_scriptLoader->failed())
112 m_failed = true;
113 else {
114 m_script = m_scriptLoader->script();
115 m_responseURL = m_scriptLoader->responseURL();
116 m_responseMIMEType = m_scriptLoader->responseMIMEType();
117 }
118
119 notifyClientFinished();
120}
121
122void WorkerModuleScriptLoader::notifyClientFinished()
123{
124 auto protectedThis = makeRef(*this);
125
126 if (m_client)
127 m_client->notifyFinished(*this, WTFMove(m_sourceURL), m_promise.releaseNonNull());
128}

130String WorkerModuleScriptLoader::taskMode()
131{
132 return "loadModulesInWorkerOrWorkletMode"_s;
133}
