26#pragma once
27
28#include "FetchOptions.h"
29#include "LoadableScript.h"
30#include "ModuleFetchParameters.h"
31#include <JavaScriptCore/ScriptFetcher.h>
32#include <wtf/text/WTFString.h>
33
34namespace WebCore {
35
36class CachedScript;
37class Document;
38
39class WorkerScriptFetcher final : public JSC::ScriptFetcher {
40public:
41 static Ref<WorkerScriptFetcher> create(FetchOptions::Credentials credentials, FetchOptions::Destination destination, ReferrerPolicy referrerPolicy)
42 {
43 return adoptRef(*new WorkerScriptFetcher(credentials, destination, referrerPolicy));
44 }
45
46 FetchOptions::Credentials credentials() const { return m_credentials; }
47 FetchOptions::Destination destination() const { return m_destination; }
48 ReferrerPolicy referrerPolicy() const { return m_referrerPolicy; }
49
50 void notifyLoadCompleted(UniquedStringImpl& moduleKey)
51 {
52 m_moduleKey = &moduleKey;
53 m_isLoaded = true;
54 }
55
56 void notifyLoadFailed(LoadableScript::Error&& error)
57 {
58 m_error = WTFMove(error);
59 m_isLoaded = true;
60 }
61
62 void notifyLoadWasCanceled()
63 {
64 m_wasCanceled = true;
65 m_isLoaded = true;
66 }
67
68 bool isLoaded() const { return m_isLoaded; }
69 std::optional<LoadableScript::Error> error() const { return m_error; }
70 bool wasCanceled() const { return m_wasCanceled; }
71 UniquedStringImpl* moduleKey() const { return m_moduleKey.get(); }
72 ModuleFetchParameters& parameters() { return m_parameters.get(); }
73
74 void setReferrerPolicy(ReferrerPolicy referrerPolicy)
75 {
76 m_referrerPolicy = referrerPolicy;
77 }
78
79protected:
80 WorkerScriptFetcher(FetchOptions::Credentials credentials, FetchOptions::Destination destination, ReferrerPolicy referrerPolicy)
81 : m_credentials(credentials)
82 , m_destination(destination)
83 , m_referrerPolicy(referrerPolicy)
84 , m_parameters(ModuleFetchParameters::create(emptyString(), /* isTopLevelModule */ true))
85 {
86 }
87
88private:
89 FetchOptions::Credentials m_credentials;
90 FetchOptions::Destination m_destination;
91 ReferrerPolicy m_referrerPolicy { ReferrerPolicy::EmptyString };
92 RefPtr<UniquedStringImpl> m_moduleKey;
93 Ref<ModuleFetchParameters> m_parameters;
94 std::optional<LoadableScript::Error> m_error;
95 bool m_wasCanceled { false };
96 bool m_isLoaded { false };
97};
98
99} // namespace WebCore
