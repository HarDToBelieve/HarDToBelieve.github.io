34namespace WebCore {

36class WritableStreamSink;
37
38class WritableStream final : public DOMGuarded<JSWritableStream> {
39public:
40 static Ref<WritableStream> create(JSDOMGlobalObject& globalObject, JSWritableStream& jsStream) { return adoptRef(*new WritableStream(globalObject, jsStream)); }
41
42 static ExceptionOr<Ref<WritableStream>> create(JSC::JSGlobalObject&, RefPtr<WritableStreamSink>&&);
43 JSWritableStream* writableStream() const { return guarded(); }
44
45 void lock();
46 bool isLocked() const;
47
48private:
49 WritableStream(JSDOMGlobalObject&, JSWritableStream&);
50};
51
52struct JSWritableStreamWrapperConverter {
53 static RefPtr<WritableStream> toWrapped(JSC::JSGlobalObject& lexicalGlobalObject, JSC::JSValue value)
54 {
55 JSC::VM& vm = JSC::getVM(&lexicalGlobalObject);
56 auto* globalObject = JSC::jsDynamicCast<JSDOMGlobalObject*>(vm, &lexicalGlobalObject);
57 if (!globalObject)
58 return nullptr;
59
60 auto* WritableStream = JSC::jsDynamicCast<JSWritableStream*>(vm, value);
61 if (!WritableStream)
62 return nullptr;
63
64 return WritableStream::create(*globalObject, *WritableStream);
65 }
66};
67
68template<> struct JSDOMWrapperConverterTraits<WritableStream> {
69 using WrapperClass = JSWritableStreamWrapperConverter;
70 using ToWrappedReturnType = RefPtr<WritableStream>;
71 static constexpr bool needsState = true;
72};
73
74inline WritableStream::WritableStream(JSDOMGlobalObject& globalObject, JSWritableStream& WritableStream)
75 : DOMGuarded<JSWritableStream>(globalObject, WritableStream)
76{
77}
78
79inline JSC::JSValue toJS(JSC::JSGlobalObject*, JSC::JSGlobalObject*, WritableStream* stream)
80{
81 return stream ? stream->writableStream() : JSC::jsUndefined();
82}
83
84inline JSC::JSValue toJS(JSC::JSGlobalObject*, JSC::JSGlobalObject*, WritableStream& stream)
85{
86 return stream.writableStream();
87}
88
89inline JSC::JSValue toJSNewlyCreated(JSC::JSGlobalObject*, JSDOMGlobalObject*, Ref<WritableStream>&& stream)
90{
91 return stream->writableStream();
92}
