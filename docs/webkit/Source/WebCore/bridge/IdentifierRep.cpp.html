38typedef HashSet<IdentifierRep*> IdentifierSet;
39
40static IdentifierSet& identifierSet()
41{
42 static NeverDestroyed<IdentifierSet> identifierSet;
43 return identifierSet;
44}
45
46typedef HashMap<int, IdentifierRep*> IntIdentifierMap;
47
48static IntIdentifierMap& intIdentifierMap()
49{
50 static NeverDestroyed<IntIdentifierMap> intIdentifierMap;
51 return intIdentifierMap;
52}
53
54IdentifierRep* IdentifierRep::get(int intID)
55{
56 if (intID == 0 || intID == -1) {
57 static IdentifierRep* negativeOneAndZeroIdentifiers[2];
58
59 IdentifierRep* identifier = negativeOneAndZeroIdentifiers[intID + 1];
60 if (!identifier) {
61 identifier = new IdentifierRep(intID);
62
63 negativeOneAndZeroIdentifiers[intID + 1] = identifier;
64 }
65
66 return identifier;
67 }
68
69 IntIdentifierMap::AddResult result = intIdentifierMap().add(intID, nullptr);
70 if (result.isNewEntry) {
71 ASSERT(!result.iterator->value);
72 result.iterator->value = new IdentifierRep(intID);
73
74 identifierSet().add(result.iterator->value);
75 }
76
77 return result.iterator->value;
78}
79
80typedef HashMap<RefPtr<StringImpl>, IdentifierRep*> StringIdentifierMap;
81
82static StringIdentifierMap& stringIdentifierMap()
83{
84 static NeverDestroyed<StringIdentifierMap> stringIdentifierMap;
85 return stringIdentifierMap;
86}
87
88IdentifierRep* IdentifierRep::get(const char* name)
89{
90 ASSERT(name);
91 if (!name)
92 return 0;
93
94 String string = String::fromUTF8WithLatin1Fallback(name, strlen(name));
95 StringIdentifierMap::AddResult result = stringIdentifierMap().add(string.impl(), nullptr);
96 if (result.isNewEntry) {
97 ASSERT(!result.iterator->value);
98 result.iterator->value = new IdentifierRep(name);
99
100 identifierSet().add(result.iterator->value);
101 }
102
103 return result.iterator->value;
104}
105
106bool IdentifierRep::isValid(IdentifierRep* identifier)
107{
108 return identifierSet().contains(identifier);
109}
