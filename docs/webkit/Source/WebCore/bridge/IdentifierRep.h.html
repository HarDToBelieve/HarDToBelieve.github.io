26#pragma once
27
28#include <wtf/Assertions.h>
29#include <wtf/FastMalloc.h>
30#include <string.h>
31
32namespace WebCore {
33
34class IdentifierRep {
35 WTF_MAKE_FAST_ALLOCATED;
36public:
37 WEBCORE_EXPORT static IdentifierRep* get(int);
38 WEBCORE_EXPORT static IdentifierRep* get(const char*);
39
40 WEBCORE_EXPORT static bool isValid(IdentifierRep*);
41
42 bool isString() const { return m_isString; }
43
44 int number() const { return m_isString ? 0 : m_value.m_number; }
45 const char* string() const { return m_isString ? m_value.m_string : 0; }
46
47private:
48 explicit IdentifierRep(int number)
49 : m_isString(false)
50 {
51 m_value.m_number = number;
52 }
53
54 explicit IdentifierRep(const char* name)
55 : m_isString(true)
56 {
57 m_value.m_string = fastStrDup(name);
58 }
59
60 ~IdentifierRep(); // Not implemented
61
62 union {
63 const char* m_string;
64 int m_number;
65 } m_value;
66 bool m_isString;
67};
68
69} // namespace WebCore
