41JSValue CField::valueFromInstance(JSGlobalObject* lexicalGlobalObject, const Instance* inst) const
42{
43 const CInstance* instance = static_cast<const CInstance*>(inst);
44 NPObject* obj = instance->getObject();
45 if (obj->_class->getProperty) {
46 NPVariant property;
47 VOID_TO_NPVARIANT(property);
48
49 bool result;
50 {
51 JSLock::DropAllLocks dropAllLocks(lexicalGlobalObject);
52 result = obj->_class->getProperty(obj, _fieldIdentifier, &property);
53 CInstance::moveGlobalExceptionToExecState(lexicalGlobalObject);
54 }
55 if (result) {
56 JSValue result = convertNPVariantToValue(lexicalGlobalObject, &property, instance->rootObject());
57 _NPN_ReleaseVariantValue(&property);
58 return result;
59 }
60 }
61 return jsUndefined();
62}
63
64bool CField::setValueToInstance(JSGlobalObject* lexicalGlobalObject, const Instance *inst, JSValue aValue) const
65{
66 const CInstance* instance = static_cast<const CInstance*>(inst);
67 NPObject* obj = instance->getObject();
68 if (obj->_class->setProperty) {
69 NPVariant variant;
70 convertValueToNPVariant(lexicalGlobalObject, aValue, &variant);
71
72 bool result = false;
73 {
74 JSLock::DropAllLocks dropAllLocks(lexicalGlobalObject);
75 result = obj->_class->setProperty(obj, _fieldIdentifier, &variant);
76 CInstance::moveGlobalExceptionToExecState(lexicalGlobalObject);
77 }
78
79 _NPN_ReleaseVariantValue(&variant);
80 return result;
81 }
82 return false;
83}
