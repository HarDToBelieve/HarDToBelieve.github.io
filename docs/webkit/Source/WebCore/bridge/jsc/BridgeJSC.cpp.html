71JSObject* Instance::createRuntimeObject(JSGlobalObject* lexicalGlobalObject)
72{
73 ASSERT(m_rootObject);
74 ASSERT(m_rootObject->isValid());
75 if (RuntimeObject* existingObject = m_runtimeObject.get())
76 return existingObject;
77
78 JSLockHolder lock(lexicalGlobalObject);
79 RuntimeObject* newObject = newRuntimeObject(lexicalGlobalObject);
80 m_runtimeObject = JSC::Weak<RuntimeObject>(newObject);
81 m_rootObject->addRuntimeObject(lexicalGlobalObject->vm(), newObject);
82 return newObject;
83}
84
85RuntimeObject* Instance::newRuntimeObject(JSGlobalObject* lexicalGlobalObject)
86{
87 JSLockHolder lock(lexicalGlobalObject);
88
89 // FIXME: deprecatedGetDOMStructure uses the prototype off of the wrong global object.
90 return RuntimeObject::create(lexicalGlobalObject->vm(), WebCore::deprecatedGetDOMStructure<RuntimeObject>(lexicalGlobalObject), this);
91}
92
93void Instance::willInvalidateRuntimeObject()
94{
95 m_runtimeObject.clear();
96}
97
98RootObject* Instance::rootObject() const
99{
100 return m_rootObject && m_rootObject->isValid() ? m_rootObject.get() : 0;
101}
