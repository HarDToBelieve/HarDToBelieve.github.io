26#ifndef ObjCRuntimeObject_h
27#define ObjCRuntimeObject_h
28
29#include "runtime_object.h"
30
31namespace JSC {
32namespace Bindings {
33
34class ObjcInstance;
35
36class ObjCRuntimeObject final : public RuntimeObject {
37public:
38 using Base = RuntimeObject;
39
40 static ObjCRuntimeObject* create(VM& vm, Structure* structure, RefPtr<ObjcInstance>&& inst)
41 {
42 ObjCRuntimeObject* object = new (NotNull, allocateCell<ObjCRuntimeObject>(vm.heap)) ObjCRuntimeObject(vm, structure, WTFMove(inst));
43 object->finishCreation(vm);
44 return object;
45 }
46
47 ObjcInstance* getInternalObjCInstance() const;
48
49 DECLARE_INFO;
50
51 static Structure* createStructure(VM& vm, JSGlobalObject* globalObject, JSValue prototype)
52 {
53 return Structure::create(vm, globalObject, prototype, TypeInfo(ObjectType, StructureFlags), info());
54 }
55
56private:
57 ObjCRuntimeObject(VM&, Structure*, RefPtr<ObjcInstance>&&);
58 void finishCreation(VM&);
59};
