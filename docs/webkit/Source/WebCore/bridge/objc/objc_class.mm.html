44static RetainPtr<CFMutableDictionaryRef>& classesByIsA()
45{
46 static NeverDestroyed<RetainPtr<CFMutableDictionaryRef>> classesByIsA;
47 return classesByIsA;
48}

50static void _createClassesByIsAIfNecessary()
51{
52 if (!classesByIsA())
53 classesByIsA() = adoptCF(CFDictionaryCreateMutable(NULL, 0, NULL, NULL));
54}

56ObjcClass* ObjcClass::classForIsA(ClassStructPtr isa)
57{
58 _createClassesByIsAIfNecessary();
59
60 auto aClass = reinterpret_cast<ObjcClass*>(const_cast<void*>(CFDictionaryGetValue(classesByIsA().get(), (__bridge CFTypeRef)isa)));
61 if (!aClass) {
62 aClass = new ObjcClass(isa);
63 CFDictionaryAddValue(classesByIsA().get(), (__bridge CFTypeRef)isa, aClass);
64 }
65
66 return aClass;
67}

69/*
70 By default, a JavaScript method name is produced by concatenating the
71 components of an ObjectiveC method name, replacing ':' with '_', and
72 escaping '_' and '$' with a leading '$', such that '_' becomes "$_" and
73 '$' becomes "$$". For example:
74
75 ObjectiveC name Default JavaScript name
76 moveTo:: moveTo__
77 moveTo_ moveTo$_
78 moveTo$_ moveTo$$$_
79
80 This function performs the inverse of that operation.
81
82 @result Fills 'buffer' with the ObjectiveC method name that corresponds to 'JSName'.
83*/
84typedef Vector<char, 256> JSNameConversionBuffer;
85static inline void convertJSMethodNameToObjc(const CString& jsName, JSNameConversionBuffer& buffer)
86{
87 buffer.reserveInitialCapacity(jsName.length() + 1);
88
89 const char* source = jsName.data();
90 while (true) {
91 if (*source == '$') {
92 ++source;
93 buffer.uncheckedAppend(*source);
94 } else if (*source == '_')
95 buffer.uncheckedAppend(':');
96 else
97 buffer.uncheckedAppend(*source);
98
99 if (!*source)
100 return;
101
102 ++source;
103 }
104}

106Method* ObjcClass::methodNamed(PropertyName propertyName, Instance*) const
107{
108 String name(propertyName.publicName());
109 if (name.isNull())
110 return nullptr;
111
112 if (Method* method = m_methodCache.get(name.impl()))
113 return method;
114
115 CString jsName = name.ascii();
116 JSNameConversionBuffer buffer;
117 convertJSMethodNameToObjc(jsName, buffer);
118 RetainPtr<CFStringRef> methodName = adoptCF(CFStringCreateWithCString(NULL, buffer.data(), kCFStringEncodingASCII));
119
120 Method* methodPtr = 0;
121 ClassStructPtr thisClass = _isa;
122
123 while (thisClass && !methodPtr) {
124 unsigned numMethodsInClass = 0;
125 MethodStructPtr* objcMethodList = class_copyMethodList(thisClass, &numMethodsInClass);
126 for (unsigned i = 0; i < numMethodsInClass; i++) {
127 MethodStructPtr objcMethod = objcMethodList[i];
128 SEL objcMethodSelector = method_getName(objcMethod);
129 const char* objcMethodSelectorName = sel_getName(objcMethodSelector);
130 NSString* mappedName = nil;
131
132 // See if the class wants to exclude the selector from visibility in JavaScript.
133 if ([thisClass respondsToSelector:@selector(isSelectorExcludedFromWebScript:)])
134 if ([thisClass isSelectorExcludedFromWebScript:objcMethodSelector])
135 continue;
136
137 // See if the class want to provide a different name for the selector in JavaScript.
138 // Note that we do not do any checks to guarantee uniqueness. That's the responsiblity
139 // of the class.
140 if ([thisClass respondsToSelector:@selector(webScriptNameForSelector:)])
141 mappedName = [thisClass webScriptNameForSelector:objcMethodSelector];
142
143 if ((mappedName && [mappedName isEqual:(__bridge NSString*)methodName.get()]) || !strcmp(objcMethodSelectorName, buffer.data())) {
144 auto method = makeUnique<ObjcMethod>(thisClass, objcMethodSelector);
145 methodPtr = method.get();
146 m_methodCache.add(name.impl(), WTFMove(method));
147 break;
148 }
149 }
150 thisClass = class_getSuperclass(thisClass);
151 free(objcMethodList);
152 }
153
154 return methodPtr;
155}

157Field* ObjcClass::fieldNamed(PropertyName propertyName, Instance* instance) const
158{
159 String name(propertyName.publicName());
160 if (name.isNull())
161 return nullptr;
162
163 Field* field = m_fieldCache.get(name.impl());
164 if (field)
165 return field;
166
167 ClassStructPtr thisClass = _isa;
168
169 CString jsName = name.ascii();
170 RetainPtr<CFStringRef> fieldName = adoptCF(CFStringCreateWithCString(NULL, jsName.data(), kCFStringEncodingASCII));
171 id targetObject = (static_cast<ObjcInstance*>(instance))->getObject();
172#if PLATFORM(IOS_FAMILY)
173 IGNORE_WARNINGS_BEGIN("undeclared-selector")
174 id attributes = [targetObject respondsToSelector:@selector(attributeKeys)] ? [targetObject performSelector:@selector(attributeKeys)] : nil;
175 IGNORE_WARNINGS_END
176#else
177 id attributes = [targetObject attributeKeys];
178#endif
179 if (attributes) {
180 // Class overrides attributeKeys, use that array of key names.
181 unsigned count = [attributes count];
182 for (unsigned i = 0; i < count; i++) {
183 NSString* keyName = [attributes objectAtIndex:i];
184 const char* UTF8KeyName = [keyName UTF8String]; // ObjC actually only supports ASCII names.
185
186 // See if the class wants to exclude the selector from visibility in JavaScript.
187 if ([thisClass respondsToSelector:@selector(isKeyExcludedFromWebScript:)])
188 if ([thisClass isKeyExcludedFromWebScript:UTF8KeyName])
189 continue;
190
191 // See if the class want to provide a different name for the selector in JavaScript.
192 // Note that we do not do any checks to guarantee uniqueness. That's the responsiblity
193 // of the class.
194 NSString* mappedName = nil;
195 if ([thisClass respondsToSelector:@selector(webScriptNameForKey:)])
196 mappedName = [thisClass webScriptNameForKey:UTF8KeyName];
197
198 if ((mappedName && [mappedName isEqual:(__bridge NSString *)fieldName.get()]) || [keyName isEqual:(__bridge NSString *)fieldName.get()]) {
199 auto newField = makeUnique<ObjcField>((__bridge CFStringRef)keyName);
200 field = newField.get();
201 m_fieldCache.add(name.impl(), WTFMove(newField));
202 break;
203 }
204 }
205 } else {
206 // Class doesn't override attributeKeys, so fall back on class runtime
207 // introspection.
208
209 while (thisClass) {
210 unsigned numFieldsInClass = 0;
211 IvarStructPtr* ivarsInClass = class_copyIvarList(thisClass, &numFieldsInClass);
212
213 for (unsigned i = 0; i < numFieldsInClass; i++) {
214 IvarStructPtr objcIVar = ivarsInClass[i];
215 const char* objcIvarName = ivar_getName(objcIVar);
216 NSString* mappedName = 0;
217
218 // See if the class wants to exclude the selector from visibility in JavaScript.
219 if ([thisClass respondsToSelector:@selector(isKeyExcludedFromWebScript:)])
220 if ([thisClass isKeyExcludedFromWebScript:objcIvarName])
221 continue;
222
223 // See if the class want to provide a different name for the selector in JavaScript.
224 // Note that we do not do any checks to guarantee uniqueness. That's the responsiblity
225 // of the class.
226 if ([thisClass respondsToSelector:@selector(webScriptNameForKey:)])
227 mappedName = [thisClass webScriptNameForKey:objcIvarName];
228
229 if ((mappedName && [mappedName isEqual:(__bridge NSString *)fieldName.get()]) || !strcmp(objcIvarName, jsName.data())) {
230 auto newField = makeUnique<ObjcField>(objcIVar);
231 field = newField.get();
232 m_fieldCache.add(name.impl(), WTFMove(newField));
233 break;
234 }
235 }
236
237 thisClass = class_getSuperclass(thisClass);
238 free(ivarsInClass);
239 }
240 }
241
242 return field;
243}

245JSValue ObjcClass::fallbackObject(JSGlobalObject* lexicalGlobalObject, Instance* instance, PropertyName propertyName)
246{
247 ObjcInstance* objcInstance = static_cast<ObjcInstance*>(instance);
248 id targetObject = objcInstance->getObject();
249
250 if (![targetObject respondsToSelector:@selector(invokeUndefinedMethodFromWebScript:withArguments:)])
251 return jsUndefined();
252
253 if (!propertyName.publicName())
254 return jsUndefined();
255
256 return ObjcFallbackObjectImp::create(lexicalGlobalObject, lexicalGlobalObject, objcInstance, propertyName.publicName());
257}
