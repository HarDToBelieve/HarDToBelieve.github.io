26#import "objc_class.h"
27#import "objc_utility.h"
28
29namespace JSC {
30
31namespace Bindings {
32
33class ObjcClass;
34
35class ObjcInstance : public Instance {
36public:
37 static Ref<ObjcInstance> create(ObjectStructPtr, RefPtr<RootObject>&&);
38 virtual ~ObjcInstance();
39
40 static void setGlobalException(NSString*, JSGlobalObject* exceptionEnvironment = 0); // A null exceptionEnvironment means the exception should propogate to any execution environment.
41
42 virtual Class* getClass() const;
43
44 virtual JSValue valueOf(JSGlobalObject*) const;
45 virtual JSValue defaultValue(JSGlobalObject*, PreferredPrimitiveType) const;
46
47 virtual JSValue getMethod(JSGlobalObject*, PropertyName);
48 JSValue invokeObjcMethod(JSGlobalObject*, CallFrame*, ObjcMethod* method);
49 virtual JSValue invokeMethod(JSGlobalObject*, CallFrame*, RuntimeMethod* method);
50 virtual bool supportsInvokeDefaultMethod() const;
51 virtual JSValue invokeDefaultMethod(JSGlobalObject*, CallFrame*);
52
53 JSValue getValueOfUndefinedField(JSGlobalObject*, PropertyName) const;
54 virtual bool setValueOfUndefinedField(JSGlobalObject*, PropertyName, JSValue);
55
56 ObjectStructPtr getObject() const { return _instance.get(); }
57
58 JSValue stringValue(JSGlobalObject*) const;
59 JSValue numberValue(JSGlobalObject*) const;
60 JSValue booleanValue() const;
61
62 static bool isInStringValue();
63
64protected:
65 virtual void virtualBegin();
66 virtual void virtualEnd();
67
68private:
69 friend class ObjcField;
70 static void moveGlobalExceptionToExecState(JSGlobalObject*);
71
72 ObjcInstance(ObjectStructPtr, RefPtr<RootObject>&&);
73
74 virtual RuntimeObject* newRuntimeObject(JSGlobalObject*);
75
76 RetainPtr<ObjectStructPtr> _instance;
77 mutable ObjcClass* _class { nullptr };
78 void* m_autoreleasePool { nullptr };
79 int _beginCount { 0 };
80};
81
82} // namespace Bindings
83
84} // namespace JSC
