90void ObjcInstance::moveGlobalExceptionToExecState(JSGlobalObject* lexicalGlobalObject)
91{
92 VM& vm = lexicalGlobalObject->vm();
93 auto scope = DECLARE_THROW_SCOPE(vm);
94
95 if (!globalException()) {
96 ASSERT(!s_exceptionEnvironment);
97 return;
98 }
99
100 if (!s_exceptionEnvironment || s_exceptionEnvironment == vm.deprecatedVMEntryGlobalObject(lexicalGlobalObject)) {
101 JSLockHolder lock(vm);
102 throwError(lexicalGlobalObject, scope, globalException().get());
103 }
104
105 globalException() = nil;
106 s_exceptionEnvironment = nullptr;
107}

115Ref<ObjcInstance> ObjcInstance::create(id instance, RefPtr<RootObject>&& rootObject)
116{
117 auto result = wrapperCache().add((__bridge CFTypeRef)instance, nullptr);
118 if (result.isNewEntry) {
119 auto wrapper = adoptRef(*new ObjcInstance(instance, WTFMove(rootObject)));
120 result.iterator->value = wrapper.ptr();
121 return wrapper;
122 }
123
124 ASSERT(result.iterator->value);
125 return *result.iterator->value;
126}

128ObjcInstance::~ObjcInstance()
129{
130 // Both -finalizeForWebScript and -dealloc/-finalize of _instance may require autorelease pools.
131 @autoreleasepool {
132 ASSERT(_instance);
133 wrapperCache().remove((__bridge CFTypeRef)_instance.get());
134
135 if ([_instance.get() respondsToSelector:@selector(finalizeForWebScript)])
136 [_instance.get() performSelector:@selector(finalizeForWebScript)];
137 _instance = 0;
138 }
139}

141void ObjcInstance::virtualBegin()
142{
143 if (!m_autoreleasePool)
144 m_autoreleasePool = objc_autoreleasePoolPush();
145 _beginCount++;
146}
147
148void ObjcInstance::virtualEnd()
149{
150 _beginCount--;
151 ASSERT(_beginCount >= 0);
152 if (!_beginCount) {
153 ASSERT(m_autoreleasePool);
154 objc_autoreleasePoolPop(m_autoreleasePool);
155 m_autoreleasePool = nullptr;
156 }

175 static ObjCRuntimeMethod* create(JSGlobalObject* lexicalGlobalObject, JSGlobalObject* globalObject, const String& name, Bindings::Method* method)
176 {
177 VM& vm = globalObject->vm();
178 // FIXME: deprecatedGetDOMStructure uses the prototype off of the wrong global object
179 // We need to pass in the right global object for "i".
180 Structure* domStructure = WebCore::deprecatedGetDOMStructure<ObjCRuntimeMethod>(lexicalGlobalObject);
181 ObjCRuntimeMethod* runtimeMethod = new (NotNull, allocateCell<ObjCRuntimeMethod>(vm.heap)) ObjCRuntimeMethod(vm, domStructure, method);
182 runtimeMethod->finishCreation(vm, name);
183 return runtimeMethod;
184 }
185
186 static Structure* createStructure(VM& vm, JSC::JSGlobalObject* globalObject, JSC::JSValue prototype)
187 {
188 return Structure::create(vm, globalObject, prototype, TypeInfo(InternalFunctionType, StructureFlags), &s_info);
189 }
190
191 DECLARE_INFO;
192
193private:
194 using Base = RuntimeMethod;
195
196 ObjCRuntimeMethod(VM& vm, Structure* structure, Bindings::Method* method)
197 : Base(vm, structure, method)
198 {
199 }
200
201 void finishCreation(VM& vm, const String& name)
202 {
203 Base::finishCreation(vm, name);
204 ASSERT(inherits(vm, info()));
205 }
206};

216JSC::JSValue ObjcInstance::invokeMethod(JSGlobalObject* lexicalGlobalObject, CallFrame* callFrame, RuntimeMethod* runtimeMethod)
217{
218 JSC::VM& vm = lexicalGlobalObject->vm();
219 auto scope = DECLARE_THROW_SCOPE(vm);
220
221 if (!asObject(runtimeMethod)->inherits<ObjCRuntimeMethod>(vm))
222 return throwTypeError(lexicalGlobalObject, scope, "Attempt to invoke non-plug-in method on plug-in object."_s);
223
224 ObjcMethod *method = static_cast<ObjcMethod*>(runtimeMethod->method());
225 ASSERT(method);
226
227 return invokeObjcMethod(lexicalGlobalObject, callFrame, method);
228}

230JSC::JSValue ObjcInstance::invokeObjcMethod(JSGlobalObject* lexicalGlobalObject, CallFrame* callFrame, ObjcMethod* method)
231{
232 JSValue result = jsUndefined();
233
234 JSLock::DropAllLocks dropAllLocks(lexicalGlobalObject); // Can't put this inside the @try scope because it unwinds incorrectly.
235
236 setGlobalException(nil);
237
238@try {
239 NSMethodSignature* signature = method->getMethodSignature();
240 NSInvocation* invocation = [NSInvocation invocationWithMethodSignature:signature];
241 [invocation setSelector:method->selector()];
242 [invocation setTarget:_instance.get()];
243
244 if (method->isFallbackMethod()) {
245 if (objcValueTypeForType([signature methodReturnType]) != ObjcObjectType) {
246 NSLog(@"Incorrect signature for invokeUndefinedMethodFromWebScript:withArguments: -- return type must be object.");
247 return result;
248 }
249
250 // Invoke invokeUndefinedMethodFromWebScript:withArguments:, pass JavaScript function
251 // name as first (actually at 2) argument and array of args as second.
252 NSString* jsName = (__bridge NSString *)method->javaScriptName();
253 [invocation setArgument:&jsName atIndex:2];
254
255 NSMutableArray* objcArgs = [NSMutableArray array];
256 int count = callFrame->argumentCount();
257 for (int i = 0; i < count; i++) {
258 ObjcValue value = convertValueToObjcValue(lexicalGlobalObject, callFrame->uncheckedArgument(i), ObjcObjectType);
259 [objcArgs addObject:(__bridge id)value.objectValue];
260 }
261 [invocation setArgument:&objcArgs atIndex:3];
262 } else {
263 unsigned count = [signature numberOfArguments];
264 for (unsigned i = 2; i < count; ++i) {
265 const char* type = [signature getArgumentTypeAtIndex:i];
266 ObjcValueType objcValueType = objcValueTypeForType(type);
267
268 // Must have a valid argument type. This method signature should have
269 // been filtered already to ensure that it has acceptable argument
270 // types.
271 ASSERT(objcValueType != ObjcInvalidType && objcValueType != ObjcVoidType);
272
273 ObjcValue value = convertValueToObjcValue(lexicalGlobalObject, callFrame->argument(i - 2), objcValueType);
274
275 switch (objcValueType) {
276 case ObjcObjectType:
277 [invocation setArgument:&value.objectValue atIndex:i];
278 break;
279 case ObjcCharType:
280 case ObjcUnsignedCharType:
281 [invocation setArgument:&value.charValue atIndex:i];
282 break;
283 case ObjcShortType:
284 case ObjcUnsignedShortType:
285 [invocation setArgument:&value.shortValue atIndex:i];
286 break;
287 case ObjcBoolType:
288 [invocation setArgument:&value.booleanValue atIndex:i];
289 break;
290 case ObjcIntType:
291 case ObjcUnsignedIntType:
292 [invocation setArgument:&value.intValue atIndex:i];
293 break;
294 case ObjcLongType:
295 case ObjcUnsignedLongType:
296 [invocation setArgument:&value.longValue atIndex:i];
297 break;
298 case ObjcLongLongType:
299 case ObjcUnsignedLongLongType:
300 [invocation setArgument:&value.longLongValue atIndex:i];
301 break;
302 case ObjcFloatType:
303 [invocation setArgument:&value.floatValue atIndex:i];
304 break;
305 case ObjcDoubleType:
306 [invocation setArgument:&value.doubleValue atIndex:i];
307 break;
308 default:
309 // Should never get here. Argument types are filtered (and
310 // the assert above should have fired in the impossible case
311 // of an invalid type anyway).
312 fprintf(stderr, "%s: invalid type (%d)\n", __PRETTY_FUNCTION__, (int)objcValueType);
313 ASSERT_NOT_REACHED();
314 }
315 }
316 }
317
318 [invocation invoke];
319
320 // Get the return value type.
321 const char* type = [signature methodReturnType];
322 ObjcValueType objcValueType = objcValueTypeForType(type);
323
324 // Must have a valid return type. This method signature should have
325 // been filtered already to ensure that it have an acceptable return
326 // type.
327 ASSERT(objcValueType != ObjcInvalidType);
328
329 // Get the return value and convert it to a JavaScript value. Length
330 // of return value will never exceed the size of largest scalar
331 // or a pointer.
332 char buffer[1024];
333 ASSERT([signature methodReturnLength] < 1024);
334
335 if (*type != 'v') {
336 [invocation getReturnValue:buffer];
337 result = convertObjcValueToValue(lexicalGlobalObject, buffer, objcValueType, m_rootObject.get());
338 }
339} @catch(NSException* localException) {
340}
341 moveGlobalExceptionToExecState(lexicalGlobalObject);
342
343 // Work around problem in some versions of GCC where result gets marked volatile and
344 // it can't handle copying from a volatile to non-volatile.
345 return const_cast<JSValue&>(result);
346}

348JSC::JSValue ObjcInstance::invokeDefaultMethod(JSGlobalObject* lexicalGlobalObject, CallFrame* callFrame)
349{
350 JSValue result = jsUndefined();
351
352 JSLock::DropAllLocks dropAllLocks(lexicalGlobalObject); // Can't put this inside the @try scope because it unwinds incorrectly.
353 setGlobalException(nil);
354
355@try {
356 if (![_instance.get() respondsToSelector:@selector(invokeDefaultMethodWithArguments:)])
357 return result;
358
359 NSMethodSignature* signature = [_instance.get() methodSignatureForSelector:@selector(invokeDefaultMethodWithArguments:)];
360 NSInvocation* invocation = [NSInvocation invocationWithMethodSignature:signature];
361 [invocation setSelector:@selector(invokeDefaultMethodWithArguments:)];
362 [invocation setTarget:_instance.get()];
363
364 if (objcValueTypeForType([signature methodReturnType]) != ObjcObjectType) {
365 NSLog(@"Incorrect signature for invokeDefaultMethodWithArguments: -- return type must be object.");
366 return result;
367 }
368
369 NSMutableArray* objcArgs = [NSMutableArray array];
370 unsigned count = callFrame->argumentCount();
371 for (unsigned i = 0; i < count; i++) {
372 ObjcValue value = convertValueToObjcValue(lexicalGlobalObject, callFrame->uncheckedArgument(i), ObjcObjectType);
373 [objcArgs addObject:(__bridge id)value.objectValue];
374 }
375 [invocation setArgument:&objcArgs atIndex:2];
376
377 [invocation invoke];
378
379 // Get the return value type, should always be "@" because of
380 // check above.
381 const char* type = [signature methodReturnType];
382 ObjcValueType objcValueType = objcValueTypeForType(type);
383
384 // Get the return value and convert it to a JavaScript value. Length
385 // of return value will never exceed the size of a pointer, so we're
386 // OK with 32 here.
387 char buffer[32];
388 [invocation getReturnValue:buffer];
389 result = convertObjcValueToValue(lexicalGlobalObject, buffer, objcValueType, m_rootObject.get());
390} @catch(NSException* localException) {
391}
392 moveGlobalExceptionToExecState(lexicalGlobalObject);
393
394 // Work around problem in some versions of GCC where result gets marked volatile and
395 // it can't handle copying from a volatile to non-volatile.
396 return const_cast<JSValue&>(result);
397}

399bool ObjcInstance::setValueOfUndefinedField(JSGlobalObject* lexicalGlobalObject, PropertyName propertyName, JSValue aValue)
400{
401 String name(propertyName.publicName());
402 if (name.isNull())
403 return false;
404
405 id targetObject = getObject();
406 if (![targetObject respondsToSelector:@selector(setValue:forUndefinedKey:)])
407 return false;
408
409 JSLock::DropAllLocks dropAllLocks(lexicalGlobalObject); // Can't put this inside the @try scope because it unwinds incorrectly.
410
411 // This check is not really necessary because NSObject implements
412 // setValue:forUndefinedKey:, and unfortunately the default implementation
413 // throws an exception.
414 if ([targetObject respondsToSelector:@selector(setValue:forUndefinedKey:)]){
415 setGlobalException(nil);
416
417 ObjcValue objcValue = convertValueToObjcValue(lexicalGlobalObject, aValue, ObjcObjectType);
418
419 @try {
420 [targetObject setValue:(__bridge id)objcValue.objectValue forUndefinedKey:[NSString stringWithCString:name.ascii().data() encoding:NSASCIIStringEncoding]];
421 } @catch(NSException* localException) {
422 // Do nothing. Class did not override valueForUndefinedKey:.
423 }
424
425 moveGlobalExceptionToExecState(lexicalGlobalObject);
426 }
427
428 return true;
429}

431JSC::JSValue ObjcInstance::getValueOfUndefinedField(JSGlobalObject* lexicalGlobalObject, PropertyName propertyName) const
432{
433 String name(propertyName.publicName());
434 if (name.isNull())
435 return jsUndefined();
436
437 JSValue result = jsUndefined();
438
439 id targetObject = getObject();
440
441 JSLock::DropAllLocks dropAllLocks(lexicalGlobalObject); // Can't put this inside the @try scope because it unwinds incorrectly.
442
443 // This check is not really necessary because NSObject implements
444 // valueForUndefinedKey:, and unfortunately the default implementation
445 // throws an exception.
446 if ([targetObject respondsToSelector:@selector(valueForUndefinedKey:)]){
447 setGlobalException(nil);
448
449 @try {
450 id objcValue = [targetObject valueForUndefinedKey:[NSString stringWithCString:name.ascii().data() encoding:NSASCIIStringEncoding]];
451 result = convertObjcValueToValue(lexicalGlobalObject, &objcValue, ObjcObjectType, m_rootObject.get());
452 } @catch(NSException* localException) {
453 // Do nothing. Class did not override valueForUndefinedKey:.
454 }
455
456 moveGlobalExceptionToExecState(lexicalGlobalObject);
457 }
458
459 // Work around problem in some versions of GCC where result gets marked volatile and
460 // it can't handle copying from a volatile to non-volatile.
461 return const_cast<JSValue&>(result);
462}

464JSC::JSValue ObjcInstance::defaultValue(JSGlobalObject* lexicalGlobalObject, PreferredPrimitiveType hint) const
465{
466 if (hint == PreferString)
467 return stringValue(lexicalGlobalObject);
468 if (hint == PreferNumber)
469 return numberValue(lexicalGlobalObject);
470 if ([_instance.get() isKindOfClass:[NSString class]])
471 return stringValue(lexicalGlobalObject);
472 if ([_instance.get() isKindOfClass:[NSNumber class]])
473 return numberValue(lexicalGlobalObject);
474 return valueOf(lexicalGlobalObject);
475}

477static ThreadSpecific<uint32_t>* s_descriptionDepth;
478
479JSC::JSValue ObjcInstance::stringValue(JSGlobalObject* lexicalGlobalObject) const
480{
481 static std::once_flag initializeDescriptionDepthOnceFlag;
482 std::call_once(initializeDescriptionDepthOnceFlag, [] {
483 s_descriptionDepth = new ThreadSpecific<uint32_t>();
484 **s_descriptionDepth = 0;
485
486 auto descriptionMethod = class_getInstanceMethod([NSObject class], @selector(description));
487 auto webDescriptionMethod = class_getInstanceMethod([NSObject class], @selector(_web_description));
488 method_exchangeImplementations(descriptionMethod, webDescriptionMethod);
489 });
490
491 (**s_descriptionDepth)++;
492 JSC::JSValue result = convertNSStringToString(lexicalGlobalObject, [getObject() description]);
493 (**s_descriptionDepth)--;
494 return result;
495}
496
497bool ObjcInstance::isInStringValue()
498{
499 return s_descriptionDepth && s_descriptionDepth->isSet() && **s_descriptionDepth;
500}
