91class ObjcFallbackObjectImp final : public JSDestructibleObject {
92public:
93 using Base = JSDestructibleObject;
94 static constexpr unsigned StructureFlags = Base::StructureFlags | OverridesGetOwnPropertySlot | OverridesGetCallData | OverridesPut;
95 static constexpr bool needsDestruction = true;
96
97 template<typename CellType, JSC::SubspaceAccess>
98 static IsoSubspace* subspaceFor(JSC::VM& vm)
99 {
100 return subspaceForImpl(vm);
101 }
102
103 static ObjcFallbackObjectImp* create(JSGlobalObject* exec, JSGlobalObject* globalObject, ObjcInstance* instance, const String& propertyName)
104 {
105 VM& vm = globalObject->vm();
106 // FIXME: deprecatedGetDOMStructure uses the prototype off of the wrong global object
107 Structure* domStructure = WebCore::deprecatedGetDOMStructure<ObjcFallbackObjectImp>(exec);
108 ObjcFallbackObjectImp* fallbackObject = new (NotNull, allocateCell<ObjcFallbackObjectImp>(vm.heap)) ObjcFallbackObjectImp(globalObject, domStructure, instance, propertyName);
109 fallbackObject->finishCreation(globalObject);
110 return fallbackObject;
111 }
112
113 DECLARE_INFO;
114
115 const String& propertyName() const { return m_item; }
116
117 static ObjectPrototype* createPrototype(VM&, JSGlobalObject& globalObject)
118 {
119 return globalObject.objectPrototype();
120 }
121
122 static Structure* createStructure(VM& vm, JSGlobalObject* globalObject, JSValue prototype)
123 {
124 return Structure::create(vm, globalObject, prototype, TypeInfo(ObjectType, StructureFlags), info());
125 }
126
127 ObjcInstance* getInternalObjCInstance() const { return _instance.get(); }
128
129private:
130 ObjcFallbackObjectImp(JSGlobalObject*, Structure*, ObjcInstance*, const String& propertyName);
131 void finishCreation(JSGlobalObject*);
132
133 static void destroy(JSCell*);
134 static bool getOwnPropertySlot(JSObject*, JSGlobalObject*, PropertyName, PropertySlot&);
135 static bool put(JSCell*, JSGlobalObject*, PropertyName, JSValue, PutPropertySlot&);
136 static CallData getCallData(JSCell*);
137 static bool deleteProperty(JSCell*, JSGlobalObject*, PropertyName, DeletePropertySlot&);
138
139 bool toBoolean(JSGlobalObject*) const; // FIXME: Currently this is broken because none of the superclasses are marked virtual. We need to solve this in the longer term.
140
141 static IsoSubspace* subspaceForImpl(VM&);
142
143 RefPtr<ObjcInstance> _instance;
144 String m_item;
145};
