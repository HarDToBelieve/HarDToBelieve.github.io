67JSC_DEFINE_CUSTOM_GETTER(arrayLengthGetter, (JSGlobalObject* lexicalGlobalObject, EncodedJSValue thisValue, PropertyName))
68{
69 VM& vm = lexicalGlobalObject->vm();
70 auto scope = DECLARE_THROW_SCOPE(vm);
71
72 RuntimeArray* thisObject = jsDynamicCast<RuntimeArray*>(vm, JSValue::decode(thisValue));
73 if (!thisObject)
74 return throwVMTypeError(lexicalGlobalObject, scope);
75 return JSValue::encode(jsNumber(thisObject->getLength()));
76}

78void RuntimeArray::getOwnPropertyNames(JSObject* object, JSGlobalObject* lexicalGlobalObject, PropertyNameArray& propertyNames, DontEnumPropertiesMode mode)
79{
80 VM& vm = lexicalGlobalObject->vm();
81 RuntimeArray* thisObject = jsCast<RuntimeArray*>(object);
82 unsigned length = thisObject->getLength();
83 for (unsigned i = 0; i < length; ++i)
84 propertyNames.add(Identifier::from(vm, i));
85
86 if (mode == DontEnumPropertiesMode::Include)
87 propertyNames.add(vm.propertyNames->length);
88
89 thisObject->getOwnNonIndexPropertyNames(lexicalGlobalObject, propertyNames, mode);
90}

92bool RuntimeArray::getOwnPropertySlot(JSObject* object, JSGlobalObject* lexicalGlobalObject, PropertyName propertyName, PropertySlot& slot)
93{
94 VM& vm = lexicalGlobalObject->vm();
95 RuntimeArray* thisObject = jsCast<RuntimeArray*>(object);
96 if (propertyName == vm.propertyNames->length) {
97 slot.setCacheableCustom(thisObject, PropertyAttribute::DontDelete | PropertyAttribute::ReadOnly | PropertyAttribute::DontEnum, arrayLengthGetter);
98 return true;
99 }
100
101 std::optional<uint32_t> index = parseIndex(propertyName);
102 if (index && index.value() < thisObject->getLength()) {
103 slot.setValue(thisObject, static_cast<unsigned>(PropertyAttribute::DontDelete),
104 thisObject->getConcreteArray()->valueAt(lexicalGlobalObject, index.value()));
105 return true;
106 }
107
108 return JSObject::getOwnPropertySlot(thisObject, lexicalGlobalObject, propertyName, slot);
109}

111bool RuntimeArray::getOwnPropertySlotByIndex(JSObject* object, JSGlobalObject* lexicalGlobalObject, unsigned index, PropertySlot& slot)
112{
113 RuntimeArray* thisObject = jsCast<RuntimeArray*>(object);
114 if (index < thisObject->getLength()) {
115 slot.setValue(thisObject, static_cast<unsigned>(PropertyAttribute::DontDelete),
116 thisObject->getConcreteArray()->valueAt(lexicalGlobalObject, index));
117 return true;
118 }
119
120 return JSObject::getOwnPropertySlotByIndex(thisObject, lexicalGlobalObject, index, slot);
121}

123bool RuntimeArray::put(JSCell* cell, JSGlobalObject* lexicalGlobalObject, PropertyName propertyName, JSValue value, PutPropertySlot& slot)
124{
125 VM& vm = lexicalGlobalObject->vm();
126 auto scope = DECLARE_THROW_SCOPE(vm);
127
128 RuntimeArray* thisObject = jsCast<RuntimeArray*>(cell);
129 if (propertyName == vm.propertyNames->length) {
130 throwException(lexicalGlobalObject, scope, createRangeError(lexicalGlobalObject, "Range error"));
131 return false;
132 }
133
134 if (std::optional<uint32_t> index = parseIndex(propertyName))
135 return thisObject->getConcreteArray()->setValueAt(lexicalGlobalObject, index.value(), value);
136
137 RELEASE_AND_RETURN(scope, JSObject::put(thisObject, lexicalGlobalObject, propertyName, value, slot));
138}
139
140bool RuntimeArray::putByIndex(JSCell* cell, JSGlobalObject* lexicalGlobalObject, unsigned index, JSValue value, bool)
141{
142 VM& vm = lexicalGlobalObject->vm();
143 auto scope = DECLARE_THROW_SCOPE(vm);
144
145 RuntimeArray* thisObject = jsCast<RuntimeArray*>(cell);
146 if (index >= thisObject->getLength()) {
147 throwException(lexicalGlobalObject, scope, createRangeError(lexicalGlobalObject, "Range error"));
148 return false;
149 }
150
151 return thisObject->getConcreteArray()->setValueAt(lexicalGlobalObject, index, value);
152}
