77JSC_DEFINE_CUSTOM_GETTER(fallbackObjectGetter, (JSGlobalObject* lexicalGlobalObject, EncodedJSValue thisValue, PropertyName propertyName))
78{
79 VM& vm = lexicalGlobalObject->vm();
80 auto scope = DECLARE_THROW_SCOPE(vm);
81
82 RuntimeObject* thisObj = jsCast<RuntimeObject*>(JSValue::decode(thisValue));
83 RefPtr<Instance> instance = thisObj->getInternalInstance();
84
85 if (!instance)
86 return JSValue::encode(throwRuntimeObjectInvalidAccessError(lexicalGlobalObject, scope));
87
88 instance->begin();
89
90 Class *aClass = instance->getClass();
91 JSValue result = aClass->fallbackObject(lexicalGlobalObject, instance.get(), propertyName);
92
93 instance->end();
94
95 return JSValue::encode(result);
96}

98JSC_DEFINE_CUSTOM_GETTER(fieldGetter, (JSGlobalObject* lexicalGlobalObject, EncodedJSValue thisValue, PropertyName propertyName))
99{
100 VM& vm = lexicalGlobalObject->vm();
101 auto scope = DECLARE_THROW_SCOPE(vm);
102
103 RuntimeObject* thisObj = jsCast<RuntimeObject*>(JSValue::decode(thisValue));
104 RefPtr<Instance> instance = thisObj->getInternalInstance();
105
106 if (!instance)
107 return JSValue::encode(throwRuntimeObjectInvalidAccessError(lexicalGlobalObject, scope));
108
109 instance->begin();
110
111 Class *aClass = instance->getClass();
112 Field* aField = aClass->fieldNamed(propertyName, instance.get());
113 JSValue result = aField->valueFromInstance(lexicalGlobalObject, instance.get());
114
115 instance->end();
116
117 return JSValue::encode(result);
118}

140bool RuntimeObject::getOwnPropertySlot(JSObject* object, JSGlobalObject* lexicalGlobalObject, PropertyName propertyName, PropertySlot& slot)
141{
142 VM& vm = lexicalGlobalObject->vm();
143 auto scope = DECLARE_THROW_SCOPE(vm);
144
145 RuntimeObject* thisObject = jsCast<RuntimeObject*>(object);
146 if (!thisObject->m_instance) {
147 throwRuntimeObjectInvalidAccessError(lexicalGlobalObject, scope);
148 return false;
149 }
150
151 if (propertyName.uid() == vm.propertyNames->toPrimitiveSymbol.impl())
152 return JSObject::getOwnPropertySlot(thisObject, lexicalGlobalObject, propertyName, slot);
153
154 RefPtr<Instance> instance = thisObject->m_instance;
155
156 instance->begin();
157
158 Class *aClass = instance->getClass();
159
160 if (aClass) {
161 // See if the instance has a field with the specified name.
162 Field *aField = aClass->fieldNamed(propertyName, instance.get());
163 if (aField) {
164 slot.setCustom(thisObject, static_cast<unsigned>(JSC::PropertyAttribute::DontDelete), fieldGetter);
165 instance->end();
166 return true;
167 } else {
168 // Now check if a method with specified name exists, if so return a function object for
169 // that method.
170 if (aClass->methodNamed(propertyName, instance.get())) {
171 slot.setCustom(thisObject, PropertyAttribute::DontDelete | PropertyAttribute::ReadOnly, methodGetter);
172
173 instance->end();
174 return true;
175 }
176 }
177
178 // Try a fallback object.
179 if (!aClass->fallbackObject(lexicalGlobalObject, instance.get(), propertyName).isUndefined()) {
180 slot.setCustom(thisObject, PropertyAttribute::DontDelete | PropertyAttribute::ReadOnly | PropertyAttribute::DontEnum, fallbackObjectGetter);
181 instance->end();
182 return true;
183 }
184 }
185
186 instance->end();
187
188 return instance->getOwnPropertySlot(thisObject, lexicalGlobalObject, propertyName, slot);
189}

191bool RuntimeObject::put(JSCell* cell, JSGlobalObject* lexicalGlobalObject, PropertyName propertyName, JSValue value, PutPropertySlot& slot)
192{
193 VM& vm = lexicalGlobalObject->vm();
194 auto scope = DECLARE_THROW_SCOPE(vm);
195
196 RuntimeObject* thisObject = jsCast<RuntimeObject*>(cell);
197 if (!thisObject->m_instance) {
198 throwRuntimeObjectInvalidAccessError(lexicalGlobalObject, scope);
199 return false;
200 }
201
202 RefPtr<Instance> instance = thisObject->m_instance;
203 instance->begin();
204
205 // Set the value of the property.
206 bool result = false;
207 Field *aField = instance->getClass()->fieldNamed(propertyName, instance.get());
208 if (aField)
209 result = aField->setValueToInstance(lexicalGlobalObject, instance.get(), value);
210 else if (!instance->setValueOfUndefinedField(lexicalGlobalObject, propertyName, value))
211 result = instance->put(thisObject, lexicalGlobalObject, propertyName, value, slot);
212
213 instance->end();
214 return result;
215}

223JSC_DEFINE_HOST_FUNCTION(convertRuntimeObjectToPrimitive, (JSGlobalObject* lexicalGlobalObject, CallFrame* callFrame))
224{
225 VM& vm = lexicalGlobalObject->vm();
226 auto scope = DECLARE_THROW_SCOPE(vm);
227
228 auto* thisObject = jsDynamicCast<RuntimeObject*>(vm, callFrame->thisValue());
229 if (!thisObject)
230 return throwVMTypeError(lexicalGlobalObject, scope, "RuntimeObject[Symbol.toPrimitive] method called on incompatible |this| value."_s);
231 auto instance = makeRefPtr(thisObject->getInternalInstance());
232 if (!instance)
233 return JSValue::encode(throwRuntimeObjectInvalidAccessError(lexicalGlobalObject, scope));
234 auto hint = toPreferredPrimitiveType(lexicalGlobalObject, callFrame->argument(0));
235 RETURN_IF_EXCEPTION(scope, { });
236
237 instance->begin();
238 JSValue result = instance->defaultValue(lexicalGlobalObject, hint);
239 instance->end();
240 return JSValue::encode(result);
241}

243JSC_DEFINE_HOST_FUNCTION(callRuntimeObject, (JSGlobalObject* globalObject, CallFrame* callFrame))
244{
245 ASSERT_UNUSED(globalObject, callFrame->jsCallee()->inherits<RuntimeObject>(globalObject->vm()));
246 RefPtr<Instance> instance(static_cast<RuntimeObject*>(callFrame->jsCallee())->getInternalInstance());
247 instance->begin();
248 JSValue result = instance->invokeDefaultMethod(globalObject, callFrame);
249 instance->end();
250 return JSValue::encode(result);
251}

253CallData RuntimeObject::getCallData(JSCell* cell)
254{
255 CallData callData;
256
257 RuntimeObject* thisObject = jsCast<RuntimeObject*>(cell);
258 if (thisObject->m_instance && thisObject->m_instance->supportsInvokeDefaultMethod()) {
259 callData.type = CallData::Type::Native;
260 callData.native.function = callRuntimeObject;
261 }
262
263 return callData;
264}
265
266JSC_DEFINE_HOST_FUNCTION(callRuntimeConstructor, (JSGlobalObject* globalObject, CallFrame* callFrame))
267{
268 JSObject* constructor = callFrame->jsCallee();
269 ASSERT_UNUSED(globalObject, constructor->inherits<RuntimeObject>(globalObject->vm()));
270 RefPtr<Instance> instance(static_cast<RuntimeObject*>(callFrame->jsCallee())->getInternalInstance());
271 instance->begin();
272 ArgList args(callFrame);
273 JSValue result = instance->invokeConstruct(globalObject, callFrame, args);
274 instance->end();
275
276 ASSERT(result);
277 return JSValue::encode(result.isObject() ? jsCast<JSObject*>(result.asCell()) : constructor);
278}

293void RuntimeObject::getOwnPropertyNames(JSObject* object, JSGlobalObject* lexicalGlobalObject, PropertyNameArray& propertyNames, DontEnumPropertiesMode)
294{
295 VM& vm = lexicalGlobalObject->vm();
296 auto scope = DECLARE_THROW_SCOPE(vm);
297
298 RuntimeObject* thisObject = jsCast<RuntimeObject*>(object);
299 if (!thisObject->m_instance) {
300 throwRuntimeObjectInvalidAccessError(lexicalGlobalObject, scope);
301 return;
302 }
303
304 RefPtr<Instance> instance = thisObject->m_instance;
305
306 instance->begin();
307 instance->getPropertyNames(lexicalGlobalObject, propertyNames);
308 instance->end();
309}
