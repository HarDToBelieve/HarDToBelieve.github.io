30#include "CSSStyleSheet.h"

32#include <wtf/Ref.h>
33#include <wtf/RefCounted.h>
34#include <wtf/text/WTFString.h>
35
36namespace WebCore {
37
38class Document;
39
40namespace ContentExtensions {
41
42class ContentExtensionStyleSheet : public RefCounted<ContentExtensionStyleSheet> {
43public:
44 static Ref<ContentExtensionStyleSheet> create(Document& document)
45 {
46 return adoptRef(*new ContentExtensionStyleSheet(document));
47 }
48 virtual ~ContentExtensionStyleSheet();
49
50 bool addDisplayNoneSelector(const String& selector, uint32_t selectorID);
51
52 CSSStyleSheet& styleSheet() { return m_styleSheet.get(); }
53
54private:
55 ContentExtensionStyleSheet(Document&);
56
57 Ref<CSSStyleSheet> m_styleSheet;
58 HashSet<uint32_t, DefaultHash<uint32_t>, WTF::UnsignedWithZeroKeyHashTraits<uint32_t>> m_addedSelectorIDs;
59};
60
61} // namespace ContentExtensions
62} // namespace WebCore
63
64#endif // ENABLE(CONTENT_EXTENSIONS)
