60static void makeSecureIfNecessary(ContentRuleListResults& results, const URL& url, const URL& redirectFrom = { })
61{
62 if (redirectFrom.host() == url.host() && redirectFrom.protocolIs("https"))
63 return;
64
65 if (!url.protocolIs("http"))
66 return;
67 if (url.host() == "www.opengl.org"
68 || url.host() == "webkit.org"
69 || url.host() == "download")
70 results.summary.madeHTTPS = true;
71}
72#endif
73
74bool ContentExtensionsBackend::shouldBeMadeSecure(const URL& url)
75{
76 ContentRuleListResults results;
77 makeSecureIfNecessary(results, url);
78 return results.summary.madeHTTPS;
79}

101auto ContentExtensionsBackend::actionsForResourceLoad(const ResourceLoadInfo& resourceLoadInfo) const -> Vector<ActionsFromContentRuleList>
102{
103#if CONTENT_EXTENSIONS_PERFORMANCE_REPORTING
104 MonotonicTime addedTimeStart = MonotonicTime::now();
105#endif
106 if (m_contentExtensions.isEmpty()
107 || !resourceLoadInfo.resourceURL.isValid()
108 || resourceLoadInfo.resourceURL.protocolIsData())
109 return { };
110
111 const String& urlString = resourceLoadInfo.resourceURL.string();
112 ASSERT_WITH_MESSAGE(urlString.isAllASCII(), "A decoded URL should only contain ASCII characters. The matching algorithm assumes the input is ASCII.");
113 const auto urlCString = urlString.utf8();
114
115 Vector<ActionsFromContentRuleList> actionsVector;
116 actionsVector.reserveInitialCapacity(m_contentExtensions.size());
117 const ResourceFlags flags = resourceLoadInfo.getResourceFlags();
118 for (auto& contentExtension : m_contentExtensions.values()) {
119 ActionsFromContentRuleList actionsStruct;
120 actionsStruct.contentRuleListIdentifier = contentExtension->identifier();
121
122 const CompiledContentExtension& compiledExtension = contentExtension->compiledExtension();
123
124 DFABytecodeInterpreter withoutConditionsInterpreter(compiledExtension.filtersWithoutConditionsBytecode(), compiledExtension.filtersWithoutConditionsBytecodeLength());
125 DFABytecodeInterpreter::Actions withoutConditionsActions = withoutConditionsInterpreter.interpret(urlCString, flags);
126
127 URL topURL = resourceLoadInfo.mainDocumentURL;
128 DFABytecodeInterpreter withConditionsInterpreter(compiledExtension.filtersWithConditionsBytecode(), compiledExtension.filtersWithConditionsBytecodeLength());
129 DFABytecodeInterpreter::Actions withConditionsActions = withConditionsInterpreter.interpretWithConditions(urlCString, flags, contentExtension->topURLActions(topURL));
130
131 const SerializedActionByte* actions = compiledExtension.actions();
132 const unsigned actionsLength = compiledExtension.actionsLength();
133
134 const Vector<uint32_t>& universalWithConditions = contentExtension->universalActionsWithConditions(topURL);
135 const Vector<uint32_t>& universalWithoutConditions = contentExtension->universalActionsWithoutConditions();
136 if (!withoutConditionsActions.isEmpty() || !withConditionsActions.isEmpty() || !universalWithConditions.isEmpty() || !universalWithoutConditions.isEmpty()) {
137 Vector<uint32_t> actionLocations;
138 actionLocations.reserveInitialCapacity(withoutConditionsActions.size() + withConditionsActions.size() + universalWithoutConditions.size() + universalWithConditions.size());
139 for (uint64_t actionLocation : withoutConditionsActions)
140 actionLocations.uncheckedAppend(static_cast<uint32_t>(actionLocation));
141 for (uint64_t actionLocation : withConditionsActions)
142 actionLocations.uncheckedAppend(static_cast<uint32_t>(actionLocation));
143 for (uint32_t actionLocation : universalWithoutConditions)
144 actionLocations.uncheckedAppend(actionLocation);
145 for (uint32_t actionLocation : universalWithConditions)
146 actionLocations.uncheckedAppend(actionLocation);
147 std::sort(actionLocations.begin(), actionLocations.end());
148
149 // Add actions in reverse order to properly deal with IgnorePreviousRules.
150 for (unsigned i = actionLocations.size(); i; i--) {
151 Action action = Action::deserialize(actions, actionsLength, actionLocations[i - 1]);
152 if (action.type() == ActionType::IgnorePreviousRules) {
153 actionsStruct.sawIgnorePreviousRules = true;
154 break;
155 }
156 actionsStruct.actions.append(WTFMove(action));
157 }
158 }
159 actionsVector.uncheckedAppend(WTFMove(actionsStruct));
160 }
161#if CONTENT_EXTENSIONS_PERFORMANCE_REPORTING
162 MonotonicTime addedTimeEnd = MonotonicTime::now();
163 dataLogF("Time added: %f microseconds %s \n", (addedTimeEnd - addedTimeStart).microseconds(), resourceLoadInfo.resourceURL.string().utf8().data());
164#endif
165 return actionsVector;
166}

180ContentRuleListResults ContentExtensionsBackend::processContentRuleListsForLoad(Page& page, const URL& url, OptionSet<ResourceType> resourceType, DocumentLoader& initiatingDocumentLoader, const URL& redirectFrom)
181{
182 Document* currentDocument = nullptr;
183 URL mainDocumentURL;
184 bool mainFrameContext = false;
185
186 if (auto* frame = initiatingDocumentLoader.frame()) {
187 mainFrameContext = frame->isMainFrame();
188 currentDocument = frame->document();
189
190 if (initiatingDocumentLoader.isLoadingMainResource()
191 && frame->isMainFrame()
192 && resourceType == ResourceType::Document)
193 mainDocumentURL = url;
194 else if (auto* mainDocument = frame->mainFrame().document())
195 mainDocumentURL = mainDocument->url();
196 }
197
198 ResourceLoadInfo resourceLoadInfo = { url, mainDocumentURL, resourceType, mainFrameContext };
199 auto actions = actionsForResourceLoad(resourceLoadInfo);
200
201 ContentRuleListResults results;
202 if (page.httpsUpgradeEnabled())
203 makeSecureIfNecessary(results, url, redirectFrom);
204 results.results.reserveInitialCapacity(actions.size());
205 for (const auto& actionsFromContentRuleList : actions) {
206 const String& contentRuleListIdentifier = actionsFromContentRuleList.contentRuleListIdentifier;
207 ContentRuleListResults::Result result;
208 for (const auto& action : actionsFromContentRuleList.actions) {
209 switch (action.type()) {
210 case ContentExtensions::ActionType::BlockLoad:
211 results.summary.blockedLoad = true;
212 result.blockedLoad = true;
213 break;
214 case ContentExtensions::ActionType::BlockCookies:
215 results.summary.blockedCookies = true;
216 result.blockedCookies = true;
217 break;
218 case ContentExtensions::ActionType::CSSDisplayNoneSelector:
219 if (resourceType == ResourceType::Document)
220 initiatingDocumentLoader.addPendingContentExtensionDisplayNoneSelector(contentRuleListIdentifier, action.stringArgument(), action.actionID());
221 else if (currentDocument)
222 currentDocument->extensionStyleSheets().addDisplayNoneSelector(contentRuleListIdentifier, action.stringArgument(), action.actionID());
223 break;
224 case ContentExtensions::ActionType::Notify:
225 results.summary.hasNotifications = true;
226 result.notifications.append(action.stringArgument());
227 break;
228 case ContentExtensions::ActionType::MakeHTTPS: {
229 if ((url.protocolIs("http") || url.protocolIs("ws"))
230 && (!url.port() || WTF::isDefaultPortForProtocol(url.port().value(), url.protocol()))) {
231 results.summary.madeHTTPS = true;
232 result.madeHTTPS = true;
233 }
234 break;
235 }
236 case ContentExtensions::ActionType::IgnorePreviousRules:
237 RELEASE_ASSERT_NOT_REACHED();
238 }
239 }
240
241 if (!actionsFromContentRuleList.sawIgnorePreviousRules) {
242 if (auto* styleSheetContents = globalDisplayNoneStyleSheet(contentRuleListIdentifier)) {
243 if (resourceType == ResourceType::Document)
244 initiatingDocumentLoader.addPendingContentExtensionSheet(contentRuleListIdentifier, *styleSheetContents);
245 else if (currentDocument)
246 currentDocument->extensionStyleSheets().maybeAddContentExtensionSheet(contentRuleListIdentifier, *styleSheetContents);
247 }
248 }
249
250 results.results.uncheckedAppend({ contentRuleListIdentifier, WTFMove(result) });
251 }
252
253 if (currentDocument) {
254 if (results.summary.madeHTTPS) {
255 ASSERT(url.protocolIs("http") || url.protocolIs("ws"));
256 String newProtocol = url.protocolIs("http") ? "https"_s : "wss"_s;
257 currentDocument->addConsoleMessage(MessageSource::ContentBlocker, MessageLevel::Info, makeString("Promoted URL from ", url.string(), " to ", newProtocol));
258 }
259 if (results.summary.blockedLoad) {
260 currentDocument->addConsoleMessage(MessageSource::ContentBlocker, MessageLevel::Info, makeString("Content blocker prevented frame displaying ", mainDocumentURL.string(), " from loading a resource from ", url.string()));
261
262 // Quirk for content-blocker interference with Google's anti-flicker optimization (rdar://problem/45968770).
263 // https://developers.google.com/optimize/
264 if (currentDocument->settings().googleAntiFlickerOptimizationQuirkEnabled()
265 && ((equalLettersIgnoringASCIICase(url.host(), "www.google-analytics.com") && equalLettersIgnoringASCIICase(url.path(), "/analytics.js"))
266 || (equalLettersIgnoringASCIICase(url.host(), "www.googletagmanager.com") && equalLettersIgnoringASCIICase(url.path(), "/gtm.js")))) {
267 if (auto* frame = currentDocument->frame())
268 frame->script().evaluateIgnoringException(ScriptSourceCode { "try { window.dataLayer.hide.end(); console.log('Called window.dataLayer.hide.end() in frame ' + document.URL + ' because the content blocker blocked the load of the https://www.google-analytics.com/analytics.js script'); } catch (e) { }"_s });
269 }
270 }
271 }
272
273 return results;
274}

276ContentRuleListResults ContentExtensionsBackend::processContentRuleListsForPingLoad(const URL& url, const URL& mainDocumentURL)
277{
278 ResourceLoadInfo resourceLoadInfo = { url, mainDocumentURL, ResourceType::Ping };
279 auto actions = actionsForResourceLoad(resourceLoadInfo);
280
281 ContentRuleListResults results;
282 makeSecureIfNecessary(results, url);
283 for (const auto& actionsFromContentRuleList : actions) {
284 for (const auto& action : actionsFromContentRuleList.actions) {
285 switch (action.type()) {
286 case ContentExtensions::ActionType::BlockLoad:
287 results.summary.blockedLoad = true;
288 break;
289 case ContentExtensions::ActionType::BlockCookies:
290 results.summary.blockedCookies = true;
291 break;
292 case ContentExtensions::ActionType::MakeHTTPS:
293 if ((url.protocolIs("http") || url.protocolIs("ws")) && (!url.port() || WTF::isDefaultPortForProtocol(url.port().value(), url.protocol())))
294 results.summary.madeHTTPS = true;
295 break;
296 case ContentExtensions::ActionType::CSSDisplayNoneSelector:
297 case ContentExtensions::ActionType::Notify:
298 // We currently have not implemented notifications from the NetworkProcess to the UIProcess.
299 break;
300 case ContentExtensions::ActionType::IgnorePreviousRules:
301 RELEASE_ASSERT_NOT_REACHED();
302 }
303 }
304 }
305
306 return results;
307}

315void applyResultsToRequest(ContentRuleListResults&& results, Page* page, ResourceRequest& request)
316{
317 if (results.summary.blockedCookies)
318 request.setAllowCookies(false);
319
320 if (results.summary.madeHTTPS) {
321 const URL& originalURL = request.url();
322 ASSERT(originalURL.protocolIs("http"));
323 ASSERT(!originalURL.port() || WTF::isDefaultPortForProtocol(originalURL.port().value(), originalURL.protocol()));
324
325 URL newURL = originalURL;
326 newURL.setProtocol("https");
327 if (originalURL.port())
328 newURL.setPort(WTF::defaultPortForProtocol("https").value());
329 request.setURL(newURL);
330 }
331
332 if (page && results.shouldNotifyApplication()) {
333 results.results.removeAllMatching([](const auto& pair) {
334 return !pair.second.shouldNotifyApplication();
335 });
336 page->chrome().client().contentRuleListNotification(request.url(), results);
337 }
338}
