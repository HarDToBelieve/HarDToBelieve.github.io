42 MatchesEverything,
43 NonASCII,
44 UnsupportedCharacterClass,
45 BackReference,
46 ForwardReference,
47 MisplacedStartOfLine,
48 WordBoundary,
49 AtomCharacter,
50 Group,
51 Disjunction,
52 MisplacedEndOfLine,
53 EmptyPattern,
54 YarrError,
55 InvalidQuantifier,
56 };

61
62private:
63 CombinedURLFilters& m_combinedURLFilters;
64};
