95template<typename ResultCallbackType, typename OperationType>
96static void dispatchAlgorithmOperation(WorkQueue& workQueue, ScriptExecutionContext& context, ResultCallbackType&& callback, CryptoAlgorithm::ExceptionCallback&& exceptionCallback, OperationType&& operation)
97{
98 workQueue.dispatch(
99 [operation = WTFMove(operation), callback = WTFMove(callback), exceptionCallback = WTFMove(exceptionCallback), contextIdentifier = context.contextIdentifier()]() mutable {
100 auto result = operation();
101 ScriptExecutionContext::postTaskTo(contextIdentifier, [result = crossThreadCopy(result), callback = WTFMove(callback), exceptionCallback = WTFMove(exceptionCallback)](auto&) mutable {
102 if (result.hasException()) {
103 exceptionCallback(result.releaseException().code());
104 return;
105 }
106 callback(result.releaseReturnValue());
107 });
108 });
109}
