36CryptoAlgorithmRegistry& CryptoAlgorithmRegistry::singleton()
37{
38 static LazyNeverDestroyed<CryptoAlgorithmRegistry> registry;
39 static std::once_flag onceKey;
40 std::call_once(onceKey, [&] {
41 registry.construct();
42 });
43 return registry;
44}

46CryptoAlgorithmRegistry::CryptoAlgorithmRegistry()
47{
48 platformRegisterAlgorithms();
49}

51std::optional<CryptoAlgorithmIdentifier> CryptoAlgorithmRegistry::identifier(const String& name)
52{
53 if (name.isEmpty())
54 return std::nullopt;
55
56 Locker locker { m_lock };
57
58 // FIXME: How is it helpful to call isolatedCopy on the argument to find?
59 auto identifier = m_identifiers.find(name.isolatedCopy());
60 if (identifier == m_identifiers.end())
61 return std::nullopt;
62
63 return identifier->value;
64}
65
66String CryptoAlgorithmRegistry::name(CryptoAlgorithmIdentifier identifier)
67{
68 Locker locker { m_lock };
69
70 auto contructor = m_constructors.find(static_cast<unsigned>(identifier));
71 if (contructor == m_constructors.end())
72 return { };
73
74 return contructor->value.first.isolatedCopy();
75}
76
77RefPtr<CryptoAlgorithm> CryptoAlgorithmRegistry::create(CryptoAlgorithmIdentifier identifier)
78{
79 Locker locker { m_lock };
80
81 auto contructor = m_constructors.find(static_cast<unsigned>(identifier));
82 if (contructor == m_constructors.end())
83 return nullptr;
84
85 return contructor->value.second();
86}
87
88void CryptoAlgorithmRegistry::registerAlgorithm(const String& name, CryptoAlgorithmIdentifier identifier, CryptoAlgorithmConstructor constructor)
89{
90 Locker locker { m_lock };
91
92 ASSERT(!m_identifiers.contains(name));
93 ASSERT(!m_constructors.contains(static_cast<unsigned>(identifier)));
94
95 m_identifiers.add(name, identifier);
96 m_constructors.add(static_cast<unsigned>(identifier), std::make_pair(name, constructor));
97}
