36CryptoKey::CryptoKey(CryptoAlgorithmIdentifier algorithmIdentifier, Type type, bool extractable, CryptoKeyUsageBitmap usages)
37 : m_algorithmIdentifier(algorithmIdentifier)
38 , m_type(type)
39 , m_extractable(extractable)
40 , m_usages(usages)
41{
42}
43
44CryptoKey::~CryptoKey() = default;
45
46auto CryptoKey::usages() const -> Vector<CryptoKeyUsage>
47{
48 // The result is ordered alphabetically.
49 Vector<CryptoKeyUsage> result;
50 if (m_usages & CryptoKeyUsageDecrypt)
51 result.append(CryptoKeyUsage::Decrypt);
52 if (m_usages & CryptoKeyUsageDeriveBits)
53 result.append(CryptoKeyUsage::DeriveBits);
54 if (m_usages & CryptoKeyUsageDeriveKey)
55 result.append(CryptoKeyUsage::DeriveKey);
56 if (m_usages & CryptoKeyUsageEncrypt)
57 result.append(CryptoKeyUsage::Encrypt);
58 if (m_usages & CryptoKeyUsageSign)
59 result.append(CryptoKeyUsage::Sign);
60 if (m_usages & CryptoKeyUsageUnwrapKey)
61 result.append(CryptoKeyUsage::UnwrapKey);
62 if (m_usages & CryptoKeyUsageVerify)
63 result.append(CryptoKeyUsage::Verify);
64 if (m_usages & CryptoKeyUsageWrapKey)
65 result.append(CryptoKeyUsage::WrapKey);
66 return result;
67}

70Vector<uint8_t> CryptoKey::randomData(size_t size)
71{
72 Vector<uint8_t> result(size);
73 cryptographicallyRandomValues(result.data(), result.size());
