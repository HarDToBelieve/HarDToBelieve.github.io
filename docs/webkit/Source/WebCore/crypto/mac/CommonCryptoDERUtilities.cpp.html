33size_t bytesUsedToEncodedLength(uint8_t octet)
34{
35 if (octet < MaxLengthInOneByte)
36 return 1;
37 return octet - MaxLengthInOneByte + 1;
38}
39
40size_t extraBytesNeededForEncodedLength(size_t length)
41{
42 if (!length)
43 return 0;
44 size_t result = 1;
45 while (result < sizeof(length) && length >= (1 << (result * 8)))
46 result += 1;
47 return result;
48}
49
50void addEncodedASN1Length(Vector<uint8_t>& in, size_t length)
51{
52 if (length < MaxLengthInOneByte) {
53 in.append(length);
54 return;
55 }
56
57 size_t extraBytes = extraBytesNeededForEncodedLength(length);
58 in.append(128 + extraBytes); // 128 is used to set the first bit of this byte.
59
60 size_t lastPosition = in.size() + extraBytes - 1;
61 in.grow(in.size() + extraBytes);
62 for (size_t i = 0; i < extraBytes; i++) {
63 in[lastPosition - i] = length & 0xff;
64 length = length >> 8;
65 }
66}
67
68size_t bytesNeededForEncodedLength(size_t length)
69{
70 if (length < MaxLengthInOneByte)
71 return 1;
72 return 1 + extraBytesNeededForEncodedLength(length);
73}
