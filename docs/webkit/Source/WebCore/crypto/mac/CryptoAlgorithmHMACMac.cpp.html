38static std::optional<CCHmacAlgorithm> commonCryptoHMACAlgorithm(CryptoAlgorithmIdentifier hashFunction)
39{
40 switch (hashFunction) {
41 case CryptoAlgorithmIdentifier::SHA_1:
42 return kCCHmacAlgSHA1;
43 case CryptoAlgorithmIdentifier::SHA_224:
44 return kCCHmacAlgSHA224;
45 case CryptoAlgorithmIdentifier::SHA_256:
46 return kCCHmacAlgSHA256;
47 case CryptoAlgorithmIdentifier::SHA_384:
48 return kCCHmacAlgSHA384;
49 case CryptoAlgorithmIdentifier::SHA_512:
50 return kCCHmacAlgSHA512;
51 default:
52 return std::nullopt;
53 }
54}
55
56ExceptionOr<Vector<uint8_t>> CryptoAlgorithmHMAC::platformSign(const CryptoKeyHMAC& key, const Vector<uint8_t>& data)
57{
58 auto algorithm = commonCryptoHMACAlgorithm(key.hashAlgorithmIdentifier());
59 if (!algorithm)
60 return Exception { OperationError };
61
62 return calculateHMACSignature(*algorithm, key.key(), data.data(), data.size());
63}
64
65ExceptionOr<bool> CryptoAlgorithmHMAC::platformVerify(const CryptoKeyHMAC& key, const Vector<uint8_t>& signature, const Vector<uint8_t>& data)
66{
67 auto algorithm = commonCryptoHMACAlgorithm(key.hashAlgorithmIdentifier());
68 if (!algorithm)
69 return Exception { OperationError };
70
71 auto expectedSignature = calculateHMACSignature(*algorithm, key.key(), data.data(), data.size());
72 // Using a constant time comparison to prevent timing attacks.
73 return signature.size() == expectedSignature.size() && !constantTimeMemcmp(expectedSignature.data(), signature.data(), expectedSignature.size());
74}
