38static CCPseudoRandomAlgorithm commonCryptoHMACAlgorithm(CryptoAlgorithmIdentifier hashFunction)
39{
40 switch (hashFunction) {
41 case CryptoAlgorithmIdentifier::SHA_1:
42 return kCCPRFHmacAlgSHA1;
43 case CryptoAlgorithmIdentifier::SHA_224:
44 return kCCPRFHmacAlgSHA224;
45 case CryptoAlgorithmIdentifier::SHA_256:
46 return kCCPRFHmacAlgSHA256;
47 case CryptoAlgorithmIdentifier::SHA_384:
48 return kCCPRFHmacAlgSHA384;
49 case CryptoAlgorithmIdentifier::SHA_512:
50 return kCCPRFHmacAlgSHA512;
51 default:
52 ASSERT_NOT_REACHED();
53 return 0;
54 }
55}
56}
57
58ExceptionOr<Vector<uint8_t>> CryptoAlgorithmPBKDF2::platformDeriveBits(const CryptoAlgorithmPbkdf2Params& parameters, const CryptoKeyRaw& key, size_t length)
59{
60 Vector<uint8_t> result(length / 8);
61 // <rdar://problem/32439955> Currently, CCKeyDerivationPBKDF bails out when an empty password/salt is provided.
62 if (CCKeyDerivationPBKDF(kCCPBKDF2, reinterpret_cast<const char *>(key.key().data()), key.key().size(), parameters.saltVector().data(), parameters.saltVector().size(), CryptoAlgorithmPBKDF2MacInternal::commonCryptoHMACAlgorithm(parameters.hashIdentifier), parameters.iterations, result.data(), length / 8))
63 return Exception { OperationError };
64 return WTFMove(result);
65}
