53void CryptoAlgorithmRegistry::platformRegisterAlgorithms()
54{
55 registerAlgorithm<CryptoAlgorithmAES_CBC>();
56 registerAlgorithm<CryptoAlgorithmAES_CFB>();
57 registerAlgorithm<CryptoAlgorithmAES_CTR>();
58 registerAlgorithm<CryptoAlgorithmAES_GCM>();
59 registerAlgorithm<CryptoAlgorithmAES_KW>();
60 registerAlgorithm<CryptoAlgorithmECDH>();
61 registerAlgorithm<CryptoAlgorithmECDSA>();
62 registerAlgorithm<CryptoAlgorithmHKDF>();
63 registerAlgorithm<CryptoAlgorithmHMAC>();
64 registerAlgorithm<CryptoAlgorithmPBKDF2>();
65 registerAlgorithm<CryptoAlgorithmRSAES_PKCS1_v1_5>();
66 registerAlgorithm<CryptoAlgorithmRSASSA_PKCS1_v1_5>();
67 registerAlgorithm<CryptoAlgorithmRSA_OAEP>();
68#if HAVE(RSA_PSS)
69 registerAlgorithm<CryptoAlgorithmRSA_PSS>();
70#endif
71 registerAlgorithm<CryptoAlgorithmSHA1>();
72 registerAlgorithm<CryptoAlgorithmSHA224>();
73 registerAlgorithm<CryptoAlgorithmSHA256>();
74 registerAlgorithm<CryptoAlgorithmSHA384>();
75 registerAlgorithm<CryptoAlgorithmSHA512>();
76}
