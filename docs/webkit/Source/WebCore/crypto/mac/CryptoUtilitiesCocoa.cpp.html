36ExceptionOr<Vector<uint8_t>> transformAES_CTR(CCOperation operation, const Vector<uint8_t>& counter, size_t counterLength, const Vector<uint8_t>& key, const uint8_t* data, size_t dataSize)
37{
38 // FIXME: We should remove the following hack once <rdar://problem/31361050> is fixed.
39 // counter = nonce + counter
40 // CommonCrypto currently can neither reset the counter nor detect overflow once the counter reaches its max value restricted
41 // by the counterLength. It then increments the nonce which should stay same for the whole operation. To remedy this issue,
42 // we detect the overflow ahead and divide the operation into two parts.
43 size_t numberOfBlocks = dataSize % kCCBlockSizeAES128 ? dataSize / kCCBlockSizeAES128 + 1 : dataSize / kCCBlockSizeAES128;
44
45 // Detect loop
46 if (counterLength < sizeof(size_t) * 8 && numberOfBlocks > (static_cast<size_t>(1) << counterLength))
47 return Exception { OperationError };
48
49 // Calculate capacity before overflow
50 CryptoAlgorithmAES_CTR::CounterBlockHelper counterBlockHelper(counter, counterLength);
51 size_t capacity = counterBlockHelper.countToOverflowSaturating();
52
53 // Divide data into two parts if necessary.
54 size_t headSize = dataSize;
55 if (capacity < numberOfBlocks)
56 headSize = capacity * kCCBlockSizeAES128;
57
58 // first part: compute the first n=capacity blocks of data if capacity is insufficient. Otherwise, return the result.
59 CCCryptorRef cryptor;
60 CCCryptorStatus status = CCCryptorCreateWithMode(operation, kCCModeCTR, kCCAlgorithmAES128, ccNoPadding, counter.data(), key.data(), key.size(), 0, 0, 0, kCCModeOptionCTR_BE, &cryptor);
61 if (status)
62 return Exception { OperationError };
63
64 Vector<uint8_t> head(CCCryptorGetOutputLength(cryptor, headSize, true));
65
66 size_t bytesWritten;
67 status = CCCryptorUpdate(cryptor, data, headSize, head.data(), head.size(), &bytesWritten);
68 if (status)
69 return Exception { OperationError };
70
71 uint8_t* p = head.data() + bytesWritten;
72 status = CCCryptorFinal(cryptor, p, head.end() - p, &bytesWritten);
73 p += bytesWritten;
74 if (status)
75 return Exception { OperationError };
76
77 ASSERT_WITH_SECURITY_IMPLICATION(p <= head.end());
78 head.shrink(p - head.begin());
79
80 CCCryptorRelease(cryptor);
81
82 if (capacity >= numberOfBlocks)
83 return WTFMove(head);
84
85 // second part: compute the remaining data and append them to the head.
86 // reset counter
87 Vector<uint8_t> remainingCounter = counterBlockHelper.counterVectorAfterOverflow();
88 status = CCCryptorCreateWithMode(operation, kCCModeCTR, kCCAlgorithmAES128, ccNoPadding, remainingCounter.data(), key.data(), key.size(), 0, 0, 0, kCCModeOptionCTR_BE, &cryptor);
89 if (status)
90 return Exception { OperationError };
91
92 size_t tailSize = dataSize - headSize;
93 Vector<uint8_t> tail(CCCryptorGetOutputLength(cryptor, tailSize, true));
94
95 status = CCCryptorUpdate(cryptor, data + headSize, tailSize, tail.data(), tail.size(), &bytesWritten);
96 if (status)
97 return Exception { OperationError };
98
99 p = tail.data() + bytesWritten;
100 status = CCCryptorFinal(cryptor, p, tail.end() - p, &bytesWritten);
101 p += bytesWritten;
102 if (status)
103 return Exception { OperationError };
104
105 ASSERT_WITH_SECURITY_IMPLICATION(p <= tail.end());
106 tail.shrink(p - tail.begin());
107
108 CCCryptorRelease(cryptor);
109
110 head.appendVector(tail);
111 return WTFMove(head);
112}

133Vector<uint8_t> calculateHMACSignature(CCHmacAlgorithm algorithm, const Vector<uint8_t>& key, const uint8_t* data, size_t size)
134{
135 size_t digestLength;
136 switch (algorithm) {
137 case kCCHmacAlgSHA1:
138 digestLength = CC_SHA1_DIGEST_LENGTH;
139 break;
140 case kCCHmacAlgSHA224:
141 digestLength = CC_SHA224_DIGEST_LENGTH;
142 break;
143 case kCCHmacAlgSHA256:
144 digestLength = CC_SHA256_DIGEST_LENGTH;
145 break;
146 case kCCHmacAlgSHA384:
147 digestLength = CC_SHA384_DIGEST_LENGTH;
148 break;
149 case kCCHmacAlgSHA512:
150 digestLength = CC_SHA512_DIGEST_LENGTH;
151 break;
152 default:
153 ASSERT_NOT_REACHED();
154 return Vector<uint8_t>();
155 }
156
157 Vector<uint8_t> result(digestLength);
158 CCHmac(algorithm, key.data(), key.size(), data, size, result.data());
159 return result;
160}
161
162Vector<uint8_t> calculateSHA256Signature(const Vector<uint8_t>& key, const uint8_t* data, size_t size)
163{
164 return calculateHMACSignature(kCCHmacAlgSHA256, key, data, size);
165}
