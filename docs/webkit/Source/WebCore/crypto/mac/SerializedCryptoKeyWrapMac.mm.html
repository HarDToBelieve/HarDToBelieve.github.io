65static NSString* masterKeyAccountNameForCurrentApplication()
66{
67#if PLATFORM(IOS_FAMILY)
68 NSString *bundleIdentifier = [[NSBundle mainBundle] bundleIdentifier];
69#else
70 NSString *bundleIdentifier = [[NSRunningApplication currentApplication] bundleIdentifier];
71#endif
72 if (!bundleIdentifier)
73 bundleIdentifier = [NSString stringWithCString:*_NSGetProgname() encoding:NSASCIIStringEncoding];
74 return [NSString stringWithFormat:@"com.apple.WebKit.WebCrypto.master+%@", bundleIdentifier];
75}

77static std::optional<Vector<uint8_t>> createAndStoreMasterKey()
78{
79 RELEASE_ASSERT(hasProcessPrivilege(ProcessPrivilege::CanAccessCredentials));
80
81 Vector<uint8_t> masterKeyData(masterKeySizeInBytes);
82 auto rc = CCRandomGenerateBytes(masterKeyData.data(), masterKeyData.size());
83 RELEASE_ASSERT(rc == kCCSuccess);
84
85#if PLATFORM(IOS_FAMILY)
86 NSBundle *mainBundle = [NSBundle mainBundle];
87 NSString *applicationName = [mainBundle objectForInfoDictionaryKey:@"CFBundleDisplayName"];
88 if (!applicationName)
89 applicationName = [mainBundle objectForInfoDictionaryKey:(NSString *)kCFBundleNameKey];
90 if (!applicationName)
91 applicationName = [mainBundle bundleIdentifier];
92 NSString *localizedItemName = webCryptoMasterKeyKeychainLabel(applicationName);
93#else
94 NSString *localizedItemName = webCryptoMasterKeyKeychainLabel([[NSRunningApplication currentApplication] localizedName]);
95#endif
96
97 OSStatus status;
98
99#if USE(KEYCHAIN_ACCESS_CONTROL_LISTS)
100 SecAccessRef accessRef;
101 status = SecAccessCreate((__bridge CFStringRef)localizedItemName, nullptr, &accessRef);
102 if (status) {
103 WTFLogAlways("Cannot create a security access object for storing WebCrypto master key, error %d", (int)status);
104 return std::nullopt;
105 }
106 RetainPtr<SecAccessRef> access = adoptCF(accessRef);
107
108 RetainPtr<CFArrayRef> acls = adoptCF(SecAccessCopyMatchingACLList(accessRef, kSecACLAuthorizationExportClear));
109 SecACLRef acl = checked_cf_cast<SecACLRef>(CFArrayGetValueAtIndex(acls.get(), 0));
110
111 SecTrustedApplicationRef trustedAppRef;
112ALLOW_DEPRECATED_DECLARATIONS_BEGIN
113 status = SecTrustedApplicationCreateFromPath(0, &trustedAppRef);
114ALLOW_DEPRECATED_DECLARATIONS_END
115 if (status) {
116 WTFLogAlways("Cannot create a trusted application object for storing WebCrypto master key, error %d", (int)status);
117 return std::nullopt;
118 }
119 RetainPtr<SecTrustedApplicationRef> trustedApp = adoptCF(trustedAppRef);
120
121 status = SecACLSetContents(acl, (__bridge CFArrayRef)@[ (__bridge id)trustedApp.get() ], (__bridge CFStringRef)localizedItemName, kSecKeychainPromptRequirePassphase);
122 if (status) {
123 WTFLogAlways("Cannot set ACL for WebCrypto master key, error %d", (int)status);
124 return std::nullopt;
125 }
126#endif
127
128 auto base64EncodedMasterKeyData = base64EncodeToVector(masterKeyData);
129
130 // Cannot use kSecClassKey because of <rdar://problem/16068207>.
131 NSDictionary *attributes = @{
132 (id)kSecClass : (id)kSecClassGenericPassword,
133 (id)kSecAttrSynchronizable : @NO,
134#if USE(KEYCHAIN_ACCESS_CONTROL_LISTS)
135 (id)kSecAttrAccess : (__bridge id)access.get(),
136#endif
137 (id)kSecAttrComment : webCryptoMasterKeyKeychainComment(),
138 (id)kSecAttrLabel : localizedItemName,
139 (id)kSecAttrAccount : masterKeyAccountNameForCurrentApplication(),
140 (id)kSecValueData : [NSData dataWithBytes:base64EncodedMasterKeyData.data() length:base64EncodedMasterKeyData.size()],
141 };
142
143 status = SecItemAdd((CFDictionaryRef)attributes, nullptr);
144 if (status) {
145 WTFLogAlways("Cannot store WebCrypto master key, error %d", (int)status);
146 return std::nullopt;
147 }
148
149 return masterKeyData;
150}

152static std::optional<Vector<uint8_t>> findMasterKey()
153{
154 RELEASE_ASSERT(hasProcessPrivilege(ProcessPrivilege::CanAccessCredentials));
155
156 NSDictionary *query = @{
157 (id)kSecClass : (id)kSecClassGenericPassword,
158 (id)kSecAttrAccount : masterKeyAccountNameForCurrentApplication(),
159 (id)kSecReturnData : @YES,
160 };
161
162 CFDataRef keyDataRef;
163 OSStatus status = SecItemCopyMatching((CFDictionaryRef)query, (CFTypeRef*)&keyDataRef);
164 if (status) {
165 if (status != errSecItemNotFound && status != errSecUserCanceled)
166 WTFLogAlways("Could not find WebCrypto master key in Keychain, error %d", (int)status);
167 return std::nullopt;
168 }
169 RetainPtr<CFDataRef> keyData = adoptCF(keyDataRef);
170 return base64Decode(CFDataGetBytePtr(keyData.get()), CFDataGetLength(keyData.get()));
171}

173std::optional<Vector<uint8_t>> defaultWebCryptoMasterKey()
174{
175 if (auto masterKey = findMasterKey()) {
176 RELEASE_ASSERT(masterKey->size() == masterKeySizeInBytes);
177 return masterKey;
178 }
179
180 if (auto masterKey = createAndStoreMasterKey()) {
181 RELEASE_ASSERT(masterKey->size() == masterKeySizeInBytes);
182 return masterKey;
183 }
184
185 return std::nullopt;
186}

188bool deleteDefaultWebCryptoMasterKey()
189{
190 RELEASE_ASSERT(hasProcessPrivilege(ProcessPrivilege::CanAccessCredentials));
191
192 NSDictionary *query = @{
193 (id)kSecClass : (id)kSecClassGenericPassword,
194 (id)kSecAttrAccount : masterKeyAccountNameForCurrentApplication(),
195 };
196
197 OSStatus status = SecItemDelete((CFDictionaryRef)query);
198 if (status) {
199 if (status != errSecItemNotFound && status != errSecUserCanceled)
200 WTFLogAlways("Could not delete WebCrypto master key in Keychain, error %d", (int)status);
201 return false;
202 }
203 return true;
204}

206bool wrapSerializedCryptoKey(const Vector<uint8_t>& masterKey, const Vector<uint8_t>& key, Vector<uint8_t>& result)
207{
208 Vector<uint8_t> kek(16);
209 auto rc = CCRandomGenerateBytes(kek.data(), kek.size());
210 RELEASE_ASSERT(rc == kCCSuccess);
211
212 Vector<uint8_t> wrappedKEK(CCSymmetricWrappedSize(kCCWRAPAES, kek.size()));
213
214 size_t wrappedKEKSize = wrappedKEK.size();
215 CCCryptorStatus status = CCSymmetricKeyWrap(kCCWRAPAES, CCrfc3394_iv, CCrfc3394_ivLen, masterKey.data(), masterKey.size(), kek.data(), kek.size(), wrappedKEK.data(), &wrappedKEKSize);
216 if (status != kCCSuccess)
217 return false;
218
219 wrappedKEK.shrink(wrappedKEKSize);
220
221 Vector<uint8_t> encryptedKey(key.size());
222 constexpr size_t maxTagLength = 16;
223 size_t tagLength = maxTagLength;
224 uint8_t tag[maxTagLength];
225
226 ALLOW_DEPRECATED_DECLARATIONS_BEGIN
227 status = CCCryptorGCM(kCCEncrypt, kCCAlgorithmAES128, kek.data(), kek.size(),
228 nullptr, 0, // iv
229 nullptr, 0, // auth data
230 key.data(), key.size(),
231 encryptedKey.data(),
232 tag, &tagLength);
233 ALLOW_DEPRECATED_DECLARATIONS_END
234
235 if (status != kCCSuccess)
236 return false;
237 RELEASE_ASSERT(tagLength == 16);
238
239 auto dictionary = @{
240 versionKey: [NSNumber numberWithUnsignedInteger:currentSerializationVersion],
241 wrappedKEKKey: [NSData dataWithBytes:wrappedKEK.data() length:wrappedKEK.size()],
242 encryptedKeyKey: [NSData dataWithBytes:encryptedKey.data() length:encryptedKey.size()],
243 tagKey: [NSData dataWithBytes:tag length:tagLength]
244 };
245
246 NSData* serialization = [NSPropertyListSerialization dataWithPropertyList:dictionary format:NSPropertyListBinaryFormat_v1_0 options:0 error:nullptr];
247 if (!serialization)
248 return false;
249
250 result = vectorFromNSData(serialization);
251 return true;
252}

254bool unwrapSerializedCryptoKey(const Vector<uint8_t>& masterKey, const Vector<uint8_t>& wrappedKey, Vector<uint8_t>& key)
255{
256 NSDictionary* dictionary = [NSPropertyListSerialization propertyListWithData:[NSData dataWithBytesNoCopy:(void*)wrappedKey.data() length:wrappedKey.size() freeWhenDone:NO] options:0 format:nullptr error:nullptr];
257 if (!dictionary)
258 return false;
259
260 id versionObject = [dictionary objectForKey:versionKey];
261 if (![versionObject isKindOfClass:[NSNumber class]])
262 return false;
263 if ([versionObject unsignedIntegerValue] > currentSerializationVersion)
264 return false;
265
266 id wrappedKEKObject = [dictionary objectForKey:wrappedKEKKey];
267 if (![wrappedKEKObject isKindOfClass:[NSData class]])
268 return false;
269 Vector<uint8_t> wrappedKEK = vectorFromNSData(wrappedKEKObject);
270
271 id encryptedKeyObject = [dictionary objectForKey:encryptedKeyKey];
272 if (![encryptedKeyObject isKindOfClass:[NSData class]])
273 return false;
274 Vector<uint8_t> encryptedKey = vectorFromNSData(encryptedKeyObject);
275
276 id tagObject = [dictionary objectForKey:tagKey];
277 if (![tagObject isKindOfClass:[NSData class]])
278 return false;
279 Vector<uint8_t> tag = vectorFromNSData(tagObject);
280 if (tag.size() != 16)
281 return false;
282
283 Vector<uint8_t> kek(CCSymmetricUnwrappedSize(kCCWRAPAES, wrappedKEK.size()));
284 size_t kekSize = kek.size();
285 CCCryptorStatus status = CCSymmetricKeyUnwrap(kCCWRAPAES, CCrfc3394_iv, CCrfc3394_ivLen, masterKey.data(), masterKey.size(), wrappedKEK.data(), wrappedKEK.size(), kek.data(), &kekSize);
286 if (status != kCCSuccess)
287 return false;
288 kek.shrink(kekSize);
289
290 constexpr size_t maxTagLength = 16;
291 size_t tagLength = maxTagLength;
292 uint8_t actualTag[maxTagLength];
293
294 key.resize(encryptedKey.size());
295 ALLOW_DEPRECATED_DECLARATIONS_BEGIN
296 status = CCCryptorGCM(kCCDecrypt, kCCAlgorithmAES128, kek.data(), kek.size(),
297 nullptr, 0, // iv
298 nullptr, 0, // auth data
299 encryptedKey.data(), encryptedKey.size(),
300 key.data(),
301 actualTag, &tagLength);
302 ALLOW_DEPRECATED_DECLARATIONS_END
303
304 if (status != kCCSuccess)
305 return false;
306 RELEASE_ASSERT(tagLength == 16);
307
308 if (constantTimeMemcmp(tag.data(), actualTag, tagLength))
309 return false;
310
311 return true;
312}
