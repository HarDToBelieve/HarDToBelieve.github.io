26#pragma once
27
28#include "BufferSource.h"
29#include "CryptoAlgorithmParameters.h"
30#include <wtf/Vector.h>
31
32#if ENABLE(WEB_CRYPTO)
33
34namespace WebCore {
35
36class CryptoAlgorithmAesCtrParams final : public CryptoAlgorithmParameters {
37public:
38 BufferSource counter;
39 size_t length;
40
41 Class parametersClass() const final { return Class::AesCtrParams; }
42
43 const Vector<uint8_t>& counterVector() const
44 {
45 if (!m_counterVector.isEmpty() || !counter.length())
46 return m_counterVector;
47
48 m_counterVector.append(counter.data(), counter.length());
49 return m_counterVector;
50 }
51
52 CryptoAlgorithmAesCtrParams isolatedCopy() const
53 {
54 CryptoAlgorithmAesCtrParams result;
55 result.identifier = identifier;
56 result.m_counterVector = counterVector();
57 result.length = length;
58
59 return result;
60 }
61
62private:
63 mutable Vector<uint8_t> m_counterVector;
64};
65
66} // namespace WebCore
67
68SPECIALIZE_TYPE_TRAITS_CRYPTO_ALGORITHM_PARAMETERS(AesCtrParams)
69
70#endif // ENABLE(WEB_CRYPTO)
