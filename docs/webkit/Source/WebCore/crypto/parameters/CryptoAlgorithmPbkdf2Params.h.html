26#pragma once
27
28#include "BufferSource.h"
29#include "CryptoAlgorithmParameters.h"
30#include <JavaScriptCore/JSObject.h>
31#include <JavaScriptCore/Strong.h>
32#include <wtf/Vector.h>
33
34#if ENABLE(WEB_CRYPTO)
35
36namespace WebCore {
37
38class CryptoAlgorithmPbkdf2Params final : public CryptoAlgorithmParameters {
39public:
40 BufferSource salt;
41 unsigned long iterations;
42 // FIXME: Consider merging hash and hashIdentifier.
43 Variant<JSC::Strong<JSC::JSObject>, String> hash;
44 CryptoAlgorithmIdentifier hashIdentifier;
45
46 const Vector<uint8_t>& saltVector() const
47 {
48 if (!m_saltVector.isEmpty() || !salt.length())
49 return m_saltVector;
50
51 m_saltVector.append(salt.data(), salt.length());
52 return m_saltVector;
53 }
54
55 Class parametersClass() const final { return Class::Pbkdf2Params; }
56
57 CryptoAlgorithmPbkdf2Params isolatedCopy() const
58 {
59 CryptoAlgorithmPbkdf2Params result;
60 result.identifier = identifier;
61 result.m_saltVector = saltVector();
62 result.iterations = iterations;
63 result.hashIdentifier = hashIdentifier;
64
65 return result;
66 }
67
68private:
69 mutable Vector<uint8_t> m_saltVector;
70};
71
72} // namespace WebCore
73
74SPECIALIZE_TYPE_TRAITS_CRYPTO_ALGORITHM_PARAMETERS(Pbkdf2Params)
75
76#endif // ENABLE(WEB_CRYPTO)
