33Ref<CSSCustomPropertyValue> CSSCustomPropertyValue::createEmpty(const AtomString& name)

35 return adoptRef(*new CSSCustomPropertyValue(name, Monostate { }));

38bool CSSCustomPropertyValue::equals(const CSSCustomPropertyValue& other) const
39{
40 if (m_name != other.m_name || m_value.index() != other.m_value.index())
41 return false;
42 return WTF::switchOn(m_value, [&](const Monostate&) {
43 return true;
44 }, [&](const Ref<CSSVariableReferenceValue>& value) {
45 return value.get() == WTF::get<Ref<CSSVariableReferenceValue>>(other.m_value).get();
46 }, [&](const CSSValueID& value) {
47 return value == WTF::get<CSSValueID>(other.m_value);
48 }, [&](const Ref<CSSVariableData>& value) {
49 return value.get() == WTF::get<Ref<CSSVariableData>>(other.m_value).get();
50 }, [&](const Length& value) {
51 return value == WTF::get<Length>(other.m_value);
52 }, [&](const Ref<StyleImage>& value) {
53 return value.get() == WTF::get<Ref<StyleImage>>(other.m_value).get();
54 });
55}

57String CSSCustomPropertyValue::customCSSText() const
58{
59 if (m_stringValue.isNull()) {
60 WTF::switchOn(m_value, [&](const Monostate&) {
61 m_stringValue = emptyString();
62 }, [&](const Ref<CSSVariableReferenceValue>& value) {
63 m_stringValue = value->cssText();
64 }, [&](const CSSValueID& value) {
65 m_stringValue = getValueName(value);
66 }, [&](const Ref<CSSVariableData>& value) {
67 m_stringValue = value->tokenRange().serialize();
68 }, [&](const Length& value) {
69 m_stringValue = CSSPrimitiveValue::create(value.value(), CSSUnitType::CSS_PX)->cssText();
70 }, [&](const Ref<StyleImage>& value) {
71 m_stringValue = value->cssValue()->cssText();
72 });
73 }
74 return m_stringValue;
75}

77Vector<CSSParserToken> CSSCustomPropertyValue::tokens() const
78{
79 Vector<CSSParserToken> result;
80 WTF::switchOn(m_value, [&](const Monostate&) {
81 // Do nothing.
82 }, [&](const Ref<CSSVariableReferenceValue>&) {
83 ASSERT_NOT_REACHED();
84 }, [&](const CSSValueID&) {
85 // Do nothing.
86 }, [&](const Ref<CSSVariableData>& value) {
87 result.appendVector(value->tokens());
88 }, [&](auto&) {
89 CSSTokenizer tokenizer(customCSSText());
90 auto tokenizerRange = tokenizer.tokenRange();
91 while (!tokenizerRange.atEnd())
92 result.append(tokenizerRange.consume());
93 });
94 return result;
95}
