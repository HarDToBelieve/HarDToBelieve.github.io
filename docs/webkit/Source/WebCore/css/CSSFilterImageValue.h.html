38
39class FilterSubimageObserverProxy;
40class RenderElement;
41
42namespace Style {
43class BuilderState;

45
46class CSSFilterImageValue final : public CSSImageGeneratorValue {
47 friend class FilterSubimageObserverProxy;
48public:
49 static Ref<CSSFilterImageValue> create(Ref<CSSValue>&& imageValue, Ref<CSSValue>&& filterValue)
50 {
51 return adoptRef(*new CSSFilterImageValue(WTFMove(imageValue), WTFMove(filterValue)));
52 }
53
54 ~CSSFilterImageValue();
55
56 String customCSSText() const;
57
58 RefPtr<Image> image(RenderElement&, const FloatSize&);
59 bool isFixedSize() const { return true; }
60 FloatSize fixedSize(const RenderElement&);
61
62 bool isPending() const;
63 bool knownToBeOpaque(const RenderElement&) const;
64
65 void loadSubimages(CachedResourceLoader&, const ResourceLoaderOptions&);
66
67 bool traverseSubresources(const WTF::Function<bool (const CachedResource&)>& handler) const;
68
69 bool equals(const CSSFilterImageValue&) const;
70
71 bool equalInputImages(const CSSFilterImageValue&) const;
72
73 const FilterOperations& filterOperations() const { return m_filterOperations; }
74 void setFilterOperations(const FilterOperations& filterOperations) { m_filterOperations = filterOperations; }
75 CachedImage* cachedImage() const { return m_cachedImage.get(); }
76
77 Ref<CSSFilterImageValue> valueWithStylesResolved(Style::BuilderState&);
78
79private:
80 CSSFilterImageValue(Ref<CSSValue>&& imageValue, Ref<CSSValue>&& filterValue)
81 : CSSImageGeneratorValue(FilterImageClass)
82 , m_imageValue(WTFMove(imageValue))
83 , m_filterValue(WTFMove(filterValue))
84 , m_filterSubimageObserver(this)
85 {
86 }
87
88 class FilterSubimageObserverProxy final : public CachedImageClient {
89 public:
90 FilterSubimageObserverProxy(CSSFilterImageValue* ownerValue)
91 : m_ownerValue(ownerValue)
92 , m_ready(false)
93 {
94 }
95
96 virtual ~FilterSubimageObserverProxy() = default;
97 void imageChanged(CachedImage*, const IntRect* = nullptr) final;
98 void setReady(bool ready) { m_ready = ready; }
99 private:
100 CSSFilterImageValue* m_ownerValue;
101 bool m_ready;
102 };
103
104 void filterImageChanged(const IntRect&);
105 void createFilterOperations(Style::BuilderState&);
106
107 Ref<CSSValue> m_imageValue;
108 Ref<CSSValue> m_filterValue;
109
110 FilterOperations m_filterOperations;
111
112 CachedResourceHandle<CachedImage> m_cachedImage;
113
114 FilterSubimageObserverProxy m_filterSubimageObserver;
115};

117} // namespace WebCore
