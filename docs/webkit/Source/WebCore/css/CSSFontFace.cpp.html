54template<typename T> void iterateClients(HashSet<CSSFontFace::Client*>& clients, T callback)
55{
56 Vector<Ref<CSSFontFace::Client>> clientsCopy;
57 clientsCopy.reserveInitialCapacity(clients.size());
58 for (auto* client : clients)
59 clientsCopy.uncheckedAppend(*client);
60
61 for (auto& client : clientsCopy)
62 callback(client);
63}

65void CSSFontFace::appendSources(CSSFontFace& fontFace, CSSValueList& srcList, ScriptExecutionContext* context, bool isInitiatingElementInUserAgentShadowTree)
66{
67 for (auto& src : srcList) {
68 // An item in the list either specifies a string (local font name) or a URL (remote font to download).
69 CSSFontFaceSrcValue& item = downcast<CSSFontFaceSrcValue>(src.get());
70 std::unique_ptr<CSSFontFaceSource> source;
71 SVGFontFaceElement* fontFaceElement = nullptr;
72 bool foundSVGFont = false;
73
74 foundSVGFont = item.isSVGFontFaceSrc() || item.svgFontFaceElement();
75 fontFaceElement = item.svgFontFaceElement();
76 if (!item.isLocal()) {
77 const auto* settings = context ? &context->settingsValues() : nullptr;
78 bool allowDownloading = foundSVGFont || (settings && settings->downloadableBinaryFontsEnabled);
79 if (allowDownloading && item.isSupportedFormat()) {
80 if (auto fontRequest = item.fontLoadRequest(context, foundSVGFont, isInitiatingElementInUserAgentShadowTree))
81 source = makeUnique<CSSFontFaceSource>(fontFace, item.resource(), *context->cssFontSelector(), makeUniqueRefFromNonNullUniquePtr(WTFMove(fontRequest)));
82 }
83 } else
84 source = (fontFaceElement ? makeUnique<CSSFontFaceSource>(fontFace, item.resource(), *fontFaceElement)
85 : makeUnique<CSSFontFaceSource>(fontFace, item.resource()));
86
87 if (source)
88 fontFace.adoptSource(WTFMove(source));
89 }
90 fontFace.sourcesPopulated();
91}

143static FontSelectionRange calculateWeightRange(CSSValue& value)
144{
145 if (value.isValueList()) {
146 auto& valueList = downcast<CSSValueList>(value);
147 ASSERT(valueList.length() == 2);
148 if (valueList.length() != 2)
149 return { normalWeightValue(), normalWeightValue() };
150 ASSERT(valueList.item(0)->isPrimitiveValue());
151 ASSERT(valueList.item(1)->isPrimitiveValue());
152 auto& value0 = downcast<CSSPrimitiveValue>(*valueList.item(0));
153 auto& value1 = downcast<CSSPrimitiveValue>(*valueList.item(1));
154 auto result0 = Style::BuilderConverter::convertFontWeightFromValue(value0);
155 auto result1 = Style::BuilderConverter::convertFontWeightFromValue(value1);
156 return { result0, result1 };
157 }
158
159 ASSERT(is<CSSPrimitiveValue>(value));
160 auto& primitiveValue = downcast<CSSPrimitiveValue>(value);
161 FontSelectionValue result = Style::BuilderConverter::convertFontWeightFromValue(primitiveValue);
162 return { result, result };
163}

219static FontSelectionRange calculateItalicRange(CSSValue& value)
220{
221 if (value.isFontStyleValue()) {
222 auto result = Style::BuilderConverter::convertFontStyleFromValue(value);
223 return { result.value_or(normalItalicValue()), result.value_or(normalItalicValue()) };
224 }
225
226 ASSERT(value.isFontStyleRangeValue());
227 auto& rangeValue = downcast<CSSFontStyleRangeValue>(value);
228 ASSERT(rangeValue.fontStyleValue->isValueID());
229 auto valueID = rangeValue.fontStyleValue->valueID();
230 if (!rangeValue.obliqueValues) {
231 if (valueID == CSSValueNormal)
232 return { normalItalicValue(), normalItalicValue() };
233 ASSERT(valueID == CSSValueItalic || valueID == CSSValueOblique);
234 return { italicValue(), italicValue() };
235 }
236 ASSERT(valueID == CSSValueOblique);
237 auto length = rangeValue.obliqueValues->length();
238 if (length == 1) {
239 auto& primitiveValue = downcast<CSSPrimitiveValue>(*rangeValue.obliqueValues->item(0));
240 FontSelectionValue result(primitiveValue.value<float>(CSSUnitType::CSS_DEG));
241 return { result, result };
242 }
243 ASSERT(length == 2);
244 auto& primitiveValue1 = downcast<CSSPrimitiveValue>(*rangeValue.obliqueValues->item(0));
245 auto& primitiveValue2 = downcast<CSSPrimitiveValue>(*rangeValue.obliqueValues->item(1));
246 FontSelectionValue result1(primitiveValue1.value<float>(CSSUnitType::CSS_DEG));
247 FontSelectionValue result2(primitiveValue2.value<float>(CSSUnitType::CSS_DEG));
248 return { result1, result2 };
249}

267bool CSSFontFace::setUnicodeRange(CSSValue& unicodeRange)
268{
269 if (!is<CSSValueList>(unicodeRange))
270 return false;
271
272 auto& list = downcast<CSSValueList>(unicodeRange);
273
274 Vector<UnicodeRange> ranges;
275 ranges.reserveInitialCapacity(list.length());
276
277 for (auto& rangeValue : list) {
278 auto& range = downcast<CSSUnicodeRangeValue>(rangeValue.get());
279 ranges.uncheckedAppend({ range.from(), range.to() });
280 }
281
282 if (ranges == m_ranges)
283 return true;
284
285 m_ranges = WTFMove(ranges);
286
287 if (m_cssConnection)
288 m_cssConnection->mutableProperties().setProperty(CSSPropertyUnicodeRange, &unicodeRange);
289
290 iterateClients(m_clients, [&](Client& client) {
291 client.fontPropertyChanged(*this);
292 });
293
294 return true;
295}

297void CSSFontFace::setFeatureSettings(CSSValue& featureSettings)
298{
299 // Can only call this with a primitive value of normal, or a value list containing font feature values.
300 ASSERT(is<CSSPrimitiveValue>(featureSettings) || is<CSSValueList>(featureSettings));
301
302 FontFeatureSettings settings;
303
304 if (is<CSSValueList>(featureSettings)) {
305 auto& list = downcast<CSSValueList>(featureSettings);
306 for (auto& rangeValue : list) {
307 auto& feature = downcast<CSSFontFeatureValue>(rangeValue.get());
308 settings.insert({ feature.tag(), feature.value() });
309 }
310 }
311
312 if (m_featureSettings == settings)
313 return;
314
315 m_featureSettings = WTFMove(settings);
316
317 if (m_cssConnection)
318 m_cssConnection->mutableProperties().setProperty(CSSPropertyFontFeatureSettings, &featureSettings);
319
320 iterateClients(m_clients, [&](Client& client) {
321 client.fontPropertyChanged(*this);
322 });

354void CSSFontFace::fontLoadEventOccurred()
355{
356 // If the font is already in the cache, CSSFontFaceSource may report it's loaded before it is added here as a source.
357 // Let's not pump the state machine until we've got all our sources. font() and load() are smart enough to act correctly
358 // when a source is failed or succeeded before we have asked it to load.
359 if (m_sourcesPopulated)
360 pump(ExternalResourceDownloadPolicy::Forbid);
361
362 iterateClients(m_clients, [&](Client& client) {
363 client.fontLoaded(*this);
364 });
365}

367void CSSFontFace::timeoutFired()
368{
369 Ref<CSSFontFace> protectedThis(*this);
370
371 switch (status()) {
372 case Status::Loading:
373 setStatus(Status::TimedOut);
374 break;
375 case Status::TimedOut:
376 // Cancelling the network request here could lead to a situation where a site's font never gets
377 // shown as the user navigates around to different pages on the site. This would occur if the
378 // download always takes longer than the timeout (even though the user may spend substantial time
379 // on each page). Therefore, we shouldn't cancel the network request here, but should use the
380 // loading infrastructure's timeout policies instead.
381 setStatus(Status::Failure);
382 break;
383 default:
384 ASSERT_NOT_REACHED();
385 break;
386 }
387
388 fontLoadEventOccurred();
389}

427void CSSFontFace::initializeWrapper()
428{
429 switch (m_status) {
430 case Status::Pending:
431 break;
432 case Status::Loading:
433 m_wrapper->fontStateChanged(*this, Status::Pending, Status::Loading);
434 break;
435 case Status::TimedOut:
436 m_wrapper->fontStateChanged(*this, Status::Pending, Status::Loading);
437 m_wrapper->fontStateChanged(*this, Status::Loading, Status::TimedOut);
438 break;
439 case Status::Success:
440 m_wrapper->fontStateChanged(*this, Status::Pending, Status::Loading);
441 m_wrapper->fontStateChanged(*this, Status::Pending, Status::Success);
442 break;
443 case Status::Failure:
444 m_wrapper->fontStateChanged(*this, Status::Pending, Status::Loading);
445 m_wrapper->fontStateChanged(*this, Status::Pending, Status::Failure);
446 break;
447 }
448 m_mayBePurged = false;
449}

478auto CSSFontFace::fontLoadTiming() const -> FontLoadTiming
479{
480 switch (m_fontLoadTimingOverride) {
481 case FontLoadTimingOverride::None:
482 switch (m_loadingBehavior) {
483 case FontLoadingBehavior::Auto:
484 case FontLoadingBehavior::Block:
485 return { 3_s, Seconds::infinity() };
486 case FontLoadingBehavior::Swap:
487 return { 0_s, Seconds::infinity() };
488 case FontLoadingBehavior::Fallback:
489 return { 0.1_s, 3_s };
490 case FontLoadingBehavior::Optional:
491 return { 0.1_s, 0_s };
492 }
493 RELEASE_ASSERT_NOT_REACHED();
494 case FontLoadTimingOverride::Block:
495 return { Seconds::infinity(), 0_s };
496 case FontLoadTimingOverride::Swap:
497 return { 0_s, Seconds::infinity() };
498 case FontLoadTimingOverride::Failure:
499 return { 0_s, 0_s };
500 }
501 RELEASE_ASSERT_NOT_REACHED();
502}

504void CSSFontFace::setStatus(Status newStatus)
505{
506 switch (newStatus) {
507 case Status::Pending:
508 ASSERT_NOT_REACHED();
509 break;
510 case Status::Loading:
511 ASSERT(m_status == Status::Pending);
512 break;
513 case Status::TimedOut:
514 ASSERT(m_status == Status::Loading);
515 break;
516 case Status::Success:
517 ASSERT(m_status == Status::Loading || m_status == Status::TimedOut);
518 break;
519 case Status::Failure:
520 ASSERT(m_status == Status::Loading || m_status == Status::TimedOut);
521 break;
522 }
523
524 iterateClients(m_clients, [&](Client& client) {
525 client.fontStateChanged(*this, m_status, newStatus);
526 });
527
528 m_status = newStatus;
529
530 Seconds blockPeriodTimeout;
531 Seconds swapPeriodTimeout;
532 auto timeouts = fontLoadTiming();
533 blockPeriodTimeout = timeouts.blockPeriod;
534 swapPeriodTimeout = timeouts.swapPeriod;
535
536 // Transfer across 0-delay timers synchronously. Layouts/script may
537 // take arbitrarily long time, and we shouldn't be in a 0-duration
538 // state for an arbitrarily long time. Also it's necessary for
539 // testing so we don't have a race with the font load.
540 switch (newStatus) {
541 case Status::Pending:
542 ASSERT_NOT_REACHED();
543 break;
544 case Status::Loading:
545 if (blockPeriodTimeout == 0_s)
546 setStatus(Status::TimedOut);
547 else if (std::isfinite(blockPeriodTimeout.value()))
548 m_timeoutTimer.startOneShot(blockPeriodTimeout);
549 break;
550 case Status::TimedOut:
551 if (swapPeriodTimeout == 0_s)
552 setStatus(Status::Failure);
553 else if (std::isfinite(swapPeriodTimeout.value()))
554 m_timeoutTimer.startOneShot(swapPeriodTimeout);
555 break;
556 case Status::Success:
557 case Status::Failure:
558 m_timeoutTimer.stop();
559 break;
560 }
561}

577size_t CSSFontFace::pump(ExternalResourceDownloadPolicy policy)
578{
579 if (status() == Status::Failure)
580 return 0;
581
582 size_t i;
583 for (i = 0; i < m_sources.size(); ++i) {
584 auto& source = m_sources[i];
585
586 if (source->status() == CSSFontFaceSource::Status::Pending) {
587 ASSERT(m_status == Status::Pending || m_status == Status::Loading || m_status == Status::TimedOut);
588 // This is a little tricky. After calling CSSFontFace::font(Forbid), a font must never fail later in
589 // this turn of the runloop because the return value of CSSFontFace::font() shouldn't get nulled out
590 // from under an existing FontRanges object. Remote fonts are all downloaded asynchronously, so this
591 // isn't a problem for them because CSSFontFace::font() will always return the interstitial font.
592 // However, local fonts may synchronously fail when you call load() on them. Therefore, we have to call
593 // load() here in order to guarantee that, if the font synchronously fails, it happens now during the
594 // first call to CSSFontFace::font() and the FontRanges object sees a consistent view of the
595 // CSSFontFace. This means we eagerly create some internal font objects when they may not be needed,
596 // but it seems that this behavior is a requirement of the design of FontRanges. FIXME: Perhaps rethink
597 // this design.
598 if (policy == ExternalResourceDownloadPolicy::Allow || !source->requiresExternalResource()) {
599 if (policy == ExternalResourceDownloadPolicy::Allow && m_status == Status::Pending)
600 setStatus(Status::Loading);
601 source->load(document());
602 }
603 }
604
605 switch (source->status()) {
606 case CSSFontFaceSource::Status::Pending:
607 ASSERT(policy == ExternalResourceDownloadPolicy::Forbid);
608 return i;
609 case CSSFontFaceSource::Status::Loading:
610 ASSERT(m_status == Status::Pending || m_status == Status::Loading || m_status == Status::TimedOut || m_status == Status::Failure);
611 if (policy == ExternalResourceDownloadPolicy::Allow && m_status == Status::Pending)
612 setStatus(Status::Loading);
613 return i;
614 case CSSFontFaceSource::Status::Success:
615 ASSERT(m_status == Status::Pending || m_status == Status::Loading || m_status == Status::TimedOut || m_status == Status::Success || m_status == Status::Failure);
616 if (m_status == Status::Pending)
617 setStatus(Status::Loading);
618 if (m_status == Status::Loading || m_status == Status::TimedOut)
619 setStatus(Status::Success);
620 return i;
621 case CSSFontFaceSource::Status::Failure:
622 if (policy == ExternalResourceDownloadPolicy::Allow && m_status == Status::Pending)
623 setStatus(Status::Loading);
624 break;
625 }
626 }
627 if (m_sources.isEmpty() && m_status == Status::Pending)
628 setStatus(Status::Loading);
629 if (m_status == Status::Loading || m_status == Status::TimedOut)
630 setStatus(Status::Failure);
631 return m_sources.size();
632}

639static Font::Visibility visibility(CSSFontFace::Status status, CSSFontFace::FontLoadTiming timing)
640{
641 switch (status) {
642 case CSSFontFace::Status::Pending:

644 case CSSFontFace::Status::Loading:
645 return Font::Visibility::Invisible;

649 default:

654RefPtr<Font> CSSFontFace::font(const FontDescription& fontDescription, bool syntheticBold, bool syntheticItalic, ExternalResourceDownloadPolicy policy)
655{
656 if (computeFailureState())
657 return nullptr;
658
659 Ref<CSSFontFace> protectedThis(*this);
660
661 // Our status is derived from the first non-failed source. However, this source may
662 // return null from font(), which means we need to continue looping through the remainder
663 // of the sources to try to find a font to use. These subsequent tries should not affect
664 // our own state, though.
665 size_t startIndex = pump(policy);
666
667 if (computeFailureState())
668 return nullptr;
669
670 for (size_t i = startIndex; i < m_sources.size(); ++i) {
671 auto& source = m_sources[i];
672 if (source->status() == CSSFontFaceSource::Status::Pending && (policy == ExternalResourceDownloadPolicy::Allow || !source->requiresExternalResource()))
673 source->load(document());
674
675 switch (source->status()) {
676 case CSSFontFaceSource::Status::Pending:
677 case CSSFontFaceSource::Status::Loading: {
678 auto& fontCache = fontCacheFallingBackToSingleton();
679 Font::Visibility visibility = WebCore::visibility(status(), fontLoadTiming());
680 return Font::create(fontCache.lastResortFallbackFont(fontDescription)->platformData(), Font::Origin::Local, &fontCache, Font::Interstitial::Yes, visibility);
681 }
682 case CSSFontFaceSource::Status::Success:
683 if (auto result = source->font(fontDescription, syntheticBold, syntheticItalic, m_featureSettings, m_fontSelectionCapabilities))
684 return result;
685 break;
686 case CSSFontFaceSource::Status::Failure:
687 break;
688 }
689 }
690
691 return nullptr;
692}
