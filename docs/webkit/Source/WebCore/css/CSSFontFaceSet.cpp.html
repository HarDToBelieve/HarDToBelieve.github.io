74void CSSFontFaceSet::incrementActiveCount()
75{
76 ++m_activeCount;
77 if (m_activeCount == 1) {

85void CSSFontFaceSet::decrementActiveCount()
86{
87 --m_activeCount;
88 if (!m_activeCount) {
89 m_status = Status::Loaded;
90 m_fontEventClients.forEach([] (auto& client) {
91 client.completedLoading();
92 });
93 }

96bool CSSFontFaceSet::hasFace(const CSSFontFace& face) const
97{
98 for (auto& myFace : m_faces) {
99 if (myFace.ptr() == &face)
100 return true;
101 }
102
103 return false;

106void CSSFontFaceSet::ensureLocalFontFacesForFamilyRegistered(const String& familyName)
107{
108 ASSERT(m_owningFontSelector);
109 if (m_locallyInstalledFacesLookupTable.contains(familyName))
110 return;
111
112 AllowUserInstalledFonts allowUserInstalledFonts = AllowUserInstalledFonts::Yes;
113 if (m_owningFontSelector->scriptExecutionContext())
114 allowUserInstalledFonts = m_owningFontSelector->scriptExecutionContext()->settingsValues().shouldAllowUserInstalledFonts ? AllowUserInstalledFonts::Yes : AllowUserInstalledFonts::No;
115 Vector<FontSelectionCapabilities> capabilities = m_owningFontSelector->scriptExecutionContext()->fontCache().getFontSelectionCapabilitiesInFamily(familyName, allowUserInstalledFonts);
116 if (capabilities.isEmpty())
117 return;
118
119 Vector<Ref<CSSFontFace>> faces;
120 for (auto item : capabilities) {
121 auto face = CSSFontFace::create(*m_owningFontSelector, nullptr, nullptr, true);
122
123 Ref<CSSValueList> familyList = CSSValueList::createCommaSeparated();
124 familyList->append(m_owningFontSelector->scriptExecutionContext()->cssValuePool().createFontFamilyValue(familyName));
125 face->setFamilies(familyList.get());
126 face->setFontSelectionCapabilities(item);
127 face->adoptSource(makeUnique<CSSFontFaceSource>(face.get(), familyName));
128 ASSERT(!face->computeFailureState());
129 faces.append(WTFMove(face));
130 }
131 m_locallyInstalledFacesLookupTable.add(familyName, WTFMove(faces));
132}

134String CSSFontFaceSet::familyNameFromPrimitive(const CSSPrimitiveValue& value)
135{
136 if (value.isFontFamily())
137 return value.fontFamily().familyName;
138 if (!value.isValueID())
139 return { };
140
141 // We need to use the raw text for all the generic family types, since @font-face is a way of actually
142 // defining what font to use for those types.
143 switch (value.valueID()) {
144 case CSSValueSerif:
145 return serifFamily.get();
146 case CSSValueSansSerif:
147 return sansSerifFamily.get();
148 case CSSValueCursive:
149 return cursiveFamily.get();
150 case CSSValueFantasy:
151 return fantasyFamily.get();
152 case CSSValueMonospace:
153 return monospaceFamily.get();
154 case CSSValueWebkitPictograph:
155 return pictographFamily.get();
156 case CSSValueSystemUi:
157 return systemUiFamily.get();
158 default:
159 return { };
160 }
161}

163void CSSFontFaceSet::addToFacesLookupTable(CSSFontFace& face)
164{
165 if (!face.families())
166 return;
167 auto families = face.families().value();
168
169 for (auto& item : *families) {
170 String familyName = CSSFontFaceSet::familyNameFromPrimitive(downcast<CSSPrimitiveValue>(item.get()));
171 if (familyName.isEmpty())
172 continue;
173
174 auto addResult = m_facesLookupTable.add(familyName, Vector<Ref<CSSFontFace>>());
175 auto& familyFontFaces = addResult.iterator->value;
176 if (addResult.isNewEntry) {
177 // m_locallyInstalledFontFaces grows without bound, eventually encorporating every font installed on the system.
178 // This is by design.
179 if (m_owningFontSelector)
180 ensureLocalFontFacesForFamilyRegistered(familyName);
181 familyFontFaces = { };
182 }
183
184 familyFontFaces.append(face);
185 }
186}

188void CSSFontFaceSet::add(CSSFontFace& face)
189{
190 ASSERT(!hasFace(face));
191
192 m_fontModifiedObservers.forEach([] (auto& observer) {
193 observer();
194 });
195
196 face.addClient(*this);
197 m_cache.clear();
198
199 if (face.cssConnection())
200 m_faces.insert(m_facesPartitionIndex++, face);
201 else
202 m_faces.append(face);
203
204 addToFacesLookupTable(face);
205
206 if (face.status() == CSSFontFace::Status::Loading || face.status() == CSSFontFace::Status::TimedOut)
207 incrementActiveCount();
208
209 if (face.cssConnection()) {
210 ASSERT(!m_constituentCSSConnections.contains(face.cssConnection()));
211 m_constituentCSSConnections.add(face.cssConnection(), &face);
212 }
213}

215void CSSFontFaceSet::removeFromFacesLookupTable(const CSSFontFace& face, const CSSValueList& familiesToSearchFor)
216{
217 for (auto& item : familiesToSearchFor) {
218 String familyName = CSSFontFaceSet::familyNameFromPrimitive(downcast<CSSPrimitiveValue>(item.get()));
219 if (familyName.isEmpty())
220 continue;
221
222 auto iterator = m_facesLookupTable.find(familyName);
223 ASSERT(iterator != m_facesLookupTable.end());
224 bool found = false;
225 for (size_t i = 0; i < iterator->value.size(); ++i) {
226 if (iterator->value[i].ptr() == &face) {
227 found = true;
228 iterator->value.remove(i);
229 break;
230 }
231 }
232 ASSERT_UNUSED(found, found);
233 if (!iterator->value.size())
234 m_facesLookupTable.remove(iterator);
235 }
236}

238void CSSFontFaceSet::remove(const CSSFontFace& face)
239{
240 auto protect = makeRef(face);
241
242 m_cache.clear();
243
244 m_fontModifiedObservers.forEach([] (auto& observer) {
245 observer();
246 });
247
248 if (face.families())
249 removeFromFacesLookupTable(face, *face.families().value());
250
251 if (face.cssConnection()) {
252 ASSERT(m_constituentCSSConnections.get(face.cssConnection()) == &face);
253 m_constituentCSSConnections.remove(face.cssConnection());
254 }
255
256 for (size_t i = 0; i < m_faces.size(); ++i) {
257 if (m_faces[i].ptr() == &face) {
258 if (i < m_facesPartitionIndex)
259 --m_facesPartitionIndex;
260 m_faces[i]->removeClient(*this);
261 m_faces.remove(i);
262 if (face.status() == CSSFontFace::Status::Loading || face.status() == CSSFontFace::Status::TimedOut)
263 decrementActiveCount();
264 return;
265 }
266 }
267 ASSERT_NOT_REACHED();
268}

311static FontSelectionRequest computeFontSelectionRequest(CSSPropertyParserHelpers::FontRaw& font)
312{
313 auto weightSelectionValue = font.weight
314 ? WTF::switchOn(*font.weight, [&] (CSSValueID keyword) {
315 switch (keyword) {
316 case CSSValueNormal:
317 return normalWeightValue();
318 case CSSValueBold:
319 case CSSValueBolder:
320 return boldWeightValue();
321 case CSSValueLighter:
322 return lightWeightValue();
323 default:
324 ASSERT_NOT_REACHED();
325 return normalWeightValue();
326 }
327 }, [&] (double weight) {
328 return FontSelectionValue::clampFloat(weight);
329 }) : normalWeightValue();
330
331 // Because this is a FontRaw, we know we should be able to dereference stretchSelectionValue as
332 // consumeFontStretchKeywordValueRaw only returns results valid to pass to fontStretchValue.
333 auto stretchSelectionValue = fontStretchValue(font.stretch.value_or(CSSValueNormal));
334 ASSERT(stretchSelectionValue);
335
336 auto styleKeyword = font.style ? font.style->style : CSSValueNormal;
337 auto styleSelectionValue = [&] () -> std::optional<FontSelectionValue> {
338 if (styleKeyword == CSSValueNormal)
339 return std::nullopt;
340 if (styleKeyword == CSSValueItalic)
341 return italicValue();
342 ASSERT(font.style && styleKeyword == CSSValueOblique);
343 float degrees = 0;
344 if (font.style->angle)
345 degrees = static_cast<float>(CSSPrimitiveValue::computeDegrees(font.style->angle->type, font.style->angle->value));
346 return FontSelectionValue(degrees);
347 }();
348
349 return { weightSelectionValue, *stretchSelectionValue, styleSelectionValue };
350}

352using CodePointsMap = HashSet<UChar32, DefaultHash<UChar32>, WTF::UnsignedWithZeroKeyHashTraits<UChar32>>;
353static CodePointsMap codePointsFromString(StringView stringView)
354{
355 CodePointsMap result;
356 auto graphemeClusters = stringView.graphemeClusters();
357 for (auto cluster : graphemeClusters) {
358 ASSERT(cluster.length() > 0);
359 UChar32 character = 0;
360 if (cluster.is8Bit())
361 character = cluster[0];
362 else
363 U16_GET(cluster.characters16(), 0, 0, cluster.length(), character);
364 result.add(character);
365 }
366 return result;
367}

369ExceptionOr<Vector<std::reference_wrapper<CSSFontFace>>> CSSFontFaceSet::matchingFacesExcludingPreinstalledFonts(const String& fontShorthand, const String& string)
370{
371 auto font = CSSPropertyParserWorkerSafe::parseFont(fontShorthand, HTMLStandardMode);
372 if (!font)
373 return Exception { SyntaxError };
374
375 HashSet<AtomString> uniqueFamilies;
376 Vector<AtomString> familyOrder;
377 for (auto& familyRaw : font->family) {
378 AtomString familyAtom;
379 WTF::switchOn(familyRaw, [&] (CSSValueID familyKeyword) {
380 if (familyKeyword != CSSValueWebkitBody)
381 familyAtom = familyNamesData->at(CSSPropertyParserHelpers::genericFontFamilyIndex(familyKeyword));
382 else {
383 ASSERT(m_owningFontSelector && m_owningFontSelector->scriptExecutionContext());
384 familyAtom = m_owningFontSelector->scriptExecutionContext()->settingsValues().fontGenericFamilies.standardFontFamily();
385 }
386 }, [&] (const String& familyString) {
387 familyAtom = familyString;
388 });
389
390 if (!familyAtom.isEmpty() && uniqueFamilies.add(familyAtom).isNewEntry)
391 familyOrder.append(familyAtom);
392 }
393
394 HashSet<CSSFontFace*> resultConstituents;
395 auto request = computeFontSelectionRequest(*font);
396 for (auto codePoint : codePointsFromString(string)) {
397 bool found = false;
398 for (auto& family : familyOrder) {
399 auto* faces = fontFace(request, family);
400 if (!faces)
401 continue;
402 for (auto& constituentFace : faces->constituentFaces()) {
403 if (constituentFace->isLocalFallback())
404 continue;
405 if (constituentFace->rangesMatchCodePoint(codePoint)) {
406 resultConstituents.add(constituentFace.ptr());
407 found = true;
408 break;
409 }
410 }
411 if (found)
412 break;
413 }
414 }
415
416 Vector<std::reference_wrapper<CSSFontFace>> result;
417 result.reserveInitialCapacity(resultConstituents.size());
418 for (auto* constituent : resultConstituents)
419 result.uncheckedAppend(*constituent);
420 return result;
421}

423ExceptionOr<bool> CSSFontFaceSet::check(const String& font, const String& text)
424{
425 auto matchingFaces = this->matchingFacesExcludingPreinstalledFonts(font, text);
426 if (matchingFaces.hasException())
427 return matchingFaces.releaseException();
428
429 for (auto& face : matchingFaces.releaseReturnValue()) {
430 if (face.get().status() == CSSFontFace::Status::Pending)
431 return false;
432 }
433 return true;
434}

436CSSSegmentedFontFace* CSSFontFaceSet::fontFace(FontSelectionRequest request, const AtomString& family)
437{
438 auto iterator = m_facesLookupTable.find(family);
439 if (iterator == m_facesLookupTable.end())
440 return nullptr;
441 auto& familyFontFaces = iterator->value;
442
443 auto& segmentedFontFaceCache = m_cache.add(family, FontSelectionHashMap()).iterator->value;
444
445 auto& face = segmentedFontFaceCache.add(request, nullptr).iterator->value;
446 if (face)
447 return face.get();
448
449 face = CSSSegmentedFontFace::create();
450
451 Vector<std::reference_wrapper<CSSFontFace>, 32> candidateFontFaces;
452 for (int i = familyFontFaces.size() - 1; i >= 0; --i) {
453 CSSFontFace& candidate = familyFontFaces[i];
454 if (auto capabilities = candidate.fontSelectionCapabilities()) {
455 if (!isItalic(request.slope) && isItalic(capabilities->slope.minimum))
456 continue;
457 candidateFontFaces.append(candidate);
458 }
459 }
460
461 auto localIterator = m_locallyInstalledFacesLookupTable.find(family);
462 if (localIterator != m_locallyInstalledFacesLookupTable.end()) {
463 for (auto& candidate : localIterator->value) {
464 if (auto capabilities = candidate->fontSelectionCapabilities()) {
465 if (!isItalic(request.slope) && isItalic(capabilities->slope.minimum))
466 continue;
467 candidateFontFaces.append(candidate);
468 }
469 }
470 }
471
472 if (!candidateFontFaces.isEmpty()) {
473 Vector<FontSelectionCapabilities> capabilities;
474 capabilities.reserveInitialCapacity(candidateFontFaces.size());
475 for (auto& face : candidateFontFaces) {
476 auto fontSelectionCapabilities = face.get().fontSelectionCapabilities();
477 capabilities.uncheckedAppend(*fontSelectionCapabilities);
478 }
479 FontSelectionAlgorithm fontSelectionAlgorithm(request, capabilities);
480 std::stable_sort(candidateFontFaces.begin(), candidateFontFaces.end(), [&fontSelectionAlgorithm](const CSSFontFace& first, const CSSFontFace& second) {
481 auto firstCapabilities = first.fontSelectionCapabilities();
482 auto secondCapabilities = second.fontSelectionCapabilities();
483
484 auto stretchDistanceFirst = fontSelectionAlgorithm.stretchDistance(*firstCapabilities).distance;
485 auto stretchDistanceSecond = fontSelectionAlgorithm.stretchDistance(*secondCapabilities).distance;
486 if (stretchDistanceFirst < stretchDistanceSecond)
487 return true;
488 if (stretchDistanceFirst > stretchDistanceSecond)
489 return false;
490
491 auto styleDistanceFirst = fontSelectionAlgorithm.styleDistance(*firstCapabilities).distance;
492 auto styleDistanceSecond = fontSelectionAlgorithm.styleDistance(*secondCapabilities).distance;
493 if (styleDistanceFirst < styleDistanceSecond)
494 return true;
495 if (styleDistanceFirst > styleDistanceSecond)
496 return false;
497
498 auto weightDistanceFirst = fontSelectionAlgorithm.weightDistance(*firstCapabilities).distance;
499 auto weightDistanceSecond = fontSelectionAlgorithm.weightDistance(*secondCapabilities).distance;
500 if (weightDistanceFirst < weightDistanceSecond)
501 return true;
502 return false;
503 });
504 CSSFontFace* previousCandidate = nullptr;
505 for (auto& candidate : candidateFontFaces) {
506 if (&candidate.get() == previousCandidate)
507 continue;
508 previousCandidate = &candidate.get();
509 face->appendFontFace(candidate.get());
510 }
511 }
512
513 return face.get();
514}

516void CSSFontFaceSet::fontStateChanged(CSSFontFace& face, CSSFontFace::Status oldState, CSSFontFace::Status newState)
517{
518 ASSERT(hasFace(face));
519 if (oldState == CSSFontFace::Status::Pending) {
520 ASSERT(newState == CSSFontFace::Status::Loading);
521 incrementActiveCount();
522 }
523 if (newState == CSSFontFace::Status::Success || newState == CSSFontFace::Status::Failure) {
524 ASSERT(oldState == CSSFontFace::Status::Loading || oldState == CSSFontFace::Status::TimedOut);
525 m_fontEventClients.forEach([&] (auto& client) {
526 client.faceFinished(face, newState);
527 });
528 decrementActiveCount();
529 }
530}
