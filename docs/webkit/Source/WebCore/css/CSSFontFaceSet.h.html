102 using FontSelectionKey = std::optional<FontSelectionRequest>;
103 struct FontSelectionKeyHash {
104 static unsigned hash(const FontSelectionKey& key) { return computeHash(key); }
105 static bool equal(const FontSelectionKey& a, const FontSelectionKey& b) { return a == b; }
106 static const bool safeToCompareToEmptyOrDeleted = true;
107 };
108 struct FontSelectionKeyHashTraits : SimpleClassHashTraits<FontSelectionKey> {
109 static const bool emptyValueIsZero = false;
110 static FontSelectionKey emptyValue() { return FontSelectionRequest { }; }
111 static void constructDeletedValue(FontSelectionKey& slot) { slot = std::nullopt; }
112 static bool isDeletedValue(const FontSelectionKey& value) { return !value; }
113 };
114 using FontSelectionHashMap = HashMap<FontSelectionKey, RefPtr<CSSSegmentedFontFace>, FontSelectionKeyHash, FontSelectionKeyHashTraits>;
