49inline void CSSFontFaceSource::setStatus(Status newStatus)
50{
51 switch (newStatus) {
52 case Status::Pending:
53 ASSERT_NOT_REACHED();
54 break;
55 case Status::Loading:
56 ASSERT(status() == Status::Pending);
57 break;
58 case Status::Success:
59 ASSERT(status() == Status::Loading);
60 break;
61 case Status::Failure:
62 ASSERT(status() == Status::Loading);
63 break;
64 }
65
66 m_status = newStatus;
67}

75CSSFontFaceSource::CSSFontFaceSource(CSSFontFace& owner, const String& familyNameOrURI, CSSFontSelector& fontSelector, UniqueRef<FontLoadRequest>&& request)
76 : m_familyNameOrURI(familyNameOrURI)
77 , m_face(owner)
78 , m_fontSelector(makeWeakPtr(fontSelector))
79 , m_fontRequest(request.moveToUniquePtr())
80{
81 // This may synchronously call fontLoaded().
82 m_fontRequest->setClient(this);
83
84 if (status() == Status::Pending && !m_fontRequest->isPending()) {
85 setStatus(Status::Loading);
86 if (!shouldIgnoreFontLoadCompletions()) {
87 if (m_fontRequest->errorOccurred())
88 setStatus(Status::Failure);
89 else
90 setStatus(Status::Success);
91 }
92 }
93}

127void CSSFontFaceSource::fontLoaded(FontLoadRequest& fontRequest)
128{
129 ASSERT_UNUSED(fontRequest, &fontRequest == m_fontRequest.get());
130
131 if (shouldIgnoreFontLoadCompletions())
132 return;
133
134 Ref<CSSFontFace> protectedFace(m_face);
135
136 // If the font is in the cache, this will be synchronously called from FontLoadRequest::addClient().
137 if (m_status == Status::Pending)
138 setStatus(Status::Loading);
139 else if (m_status == Status::Failure) {
140 // This function may be called twice if loading was cancelled.
141 ASSERT(m_fontRequest->errorOccurred());
142 return;
143 }
144
145 if (m_fontRequest->errorOccurred() || !m_fontRequest->ensureCustomFontData(m_familyNameOrURI))
146 setStatus(Status::Failure);
147 else
148 setStatus(Status::Success);
149
150 m_face.fontLoaded(*this);
151}

153void CSSFontFaceSource::load(Document* document)
154{
155 setStatus(Status::Loading);
156
157 if (m_fontRequest) {
158 ASSERT(m_fontSelector);
159 if (auto* context = m_fontSelector->scriptExecutionContext())
160 context->beginLoadingFontSoon(*m_fontRequest);
161 } else {
162 bool success = false;
163 if (m_hasSVGFontFaceElement) {
164 if (m_svgFontFaceElement && is<SVGFontElement>(m_svgFontFaceElement->parentNode())) {
165 ASSERT(!m_inDocumentCustomPlatformData);
166 SVGFontElement& fontElement = downcast<SVGFontElement>(*m_svgFontFaceElement->parentNode());
167 if (auto otfFont = convertSVGToOTFFont(fontElement))
168 m_generatedOTFBuffer = SharedBuffer::create(WTFMove(otfFont.value()));
169 if (m_generatedOTFBuffer) {
170 m_inDocumentCustomPlatformData = createFontCustomPlatformData(*m_generatedOTFBuffer, String());
171 success = static_cast<bool>(m_inDocumentCustomPlatformData);
172 }
173 }
174 } else if (m_immediateSource) {
175 ASSERT(!m_immediateFontCustomPlatformData);
176 bool wrapping;
177 auto buffer = SharedBuffer::create(static_cast<const char*>(m_immediateSource->baseAddress()), m_immediateSource->byteLength());
178 m_immediateFontCustomPlatformData = CachedFont::createCustomFontData(buffer.get(), String(), wrapping);
179 success = static_cast<bool>(m_immediateFontCustomPlatformData);
180 } else {
181 // We are only interested in whether or not fontForFamily() returns null or not. Luckily, none of
182 // the values in the FontDescription other than the family name can cause the function to return
183 // null if it wasn't going to otherwise (and vice-versa).
184 FontCascadeDescription fontDescription;
185 fontDescription.setOneFamily(m_familyNameOrURI);
186 fontDescription.setComputedSize(1);
187 fontDescription.setShouldAllowUserInstalledFonts(m_face.allowUserInstalledFonts());
188 success = FontCache::singleton().fontForFamily(fontDescription, m_familyNameOrURI, nullptr, FontSelectionSpecifiedCapabilities(), true);
189 if (document && RuntimeEnabledFeatures::sharedFeatures().webAPIStatisticsEnabled())
190 ResourceLoadObserver::shared().logFontLoad(*document, m_familyNameOrURI.string(), success);
191 }
192 setStatus(success ? Status::Success : Status::Failure);
193 }
194}

196RefPtr<Font> CSSFontFaceSource::font(const FontDescription& fontDescription, bool syntheticBold, bool syntheticItalic, const FontFeatureSettings& fontFaceFeatures, FontSelectionSpecifiedCapabilities fontFaceCapabilities)
197{
198 ASSERT(status() == Status::Success);
199
200 bool usesInDocumentSVGFont = m_hasSVGFontFaceElement;
201
202 if (!m_fontRequest && !usesInDocumentSVGFont) {
203 if (m_immediateSource) {
204 if (!m_immediateFontCustomPlatformData)
205 return nullptr;
206 return Font::create(CachedFont::platformDataFromCustomData(*m_immediateFontCustomPlatformData, fontDescription, syntheticBold, syntheticItalic, fontFaceFeatures, fontFaceCapabilities), Font::Origin::Remote);
207 }
208
209 // We're local. Just return a Font from the normal cache.
210 // We don't want to check alternate font family names here, so pass true as the checkingAlternateName parameter.
211 return FontCache::singleton().fontForFamily(fontDescription, m_familyNameOrURI, &fontFaceFeatures, fontFaceCapabilities, true);
212 }
213
214 if (m_fontRequest) {
215 auto success = m_fontRequest->ensureCustomFontData(m_familyNameOrURI);
216 ASSERT_UNUSED(success, success);
217
218 ASSERT(status() == Status::Success);
219 auto result = m_fontRequest->createFont(fontDescription, m_familyNameOrURI, syntheticBold, syntheticItalic, fontFaceFeatures, fontFaceCapabilities);
220 ASSERT(result);
221 return result;
222 }
223
224 if (!usesInDocumentSVGFont)
225 return nullptr;
226
227 if (!m_svgFontFaceElement || !is<SVGFontElement>(m_svgFontFaceElement->parentNode()))
228 return nullptr;
229 if (!m_inDocumentCustomPlatformData)
230 return nullptr;
231 return Font::create(m_inDocumentCustomPlatformData->fontPlatformData(fontDescription, syntheticBold, syntheticItalic, fontFaceFeatures, fontFaceCapabilities), Font::Origin::Remote);
232}

234bool CSSFontFaceSource::isSVGFontFaceSource() const
235{
236 return m_hasSVGFontFaceElement || (is<CachedFontLoadRequest>(m_fontRequest.get()) && is<CachedSVGFont>(downcast<CachedFontLoadRequest>(m_fontRequest.get())->cachedFont()));
237}
