42bool CSSFontFaceSrcValue::isSVGFontFaceSrc() const
43{
44 return equalLettersIgnoringASCIICase(m_format, "svg");
45}
46
47bool CSSFontFaceSrcValue::isSVGFontTarget() const
48{
49 return isSVGFontFaceSrc() || svgFontFaceElement();
50}
51
52bool CSSFontFaceSrcValue::isSupportedFormat() const
53{
54 // Normally we would just check the format, but in order to avoid conflicts with the old WinIE style of font-face,
55 // we will also check to see if the URL ends with .eot. If so, we'll assume that we shouldn't load it.
56 if (m_format.isEmpty()) {
57 // Check for .eot.
58 if (!protocolIs(m_resource, "data") && m_resource.endsWithIgnoringASCIICase(".eot"))
59 return false;
60 return true;
61 }
62
63 return FontCustomPlatformData::supportsFormat(m_format) || isSVGFontFaceSrc();
64}
65
66String CSSFontFaceSrcValue::customCSSText() const
67{
68 const char* prefix = isLocal() ? "local(" : "url(";
69 if (m_format.isEmpty())
70 return makeString(prefix, m_resource, ')');
71 return makeString(prefix, m_resource, ')', " format(", m_format, ')');
72}
73
74bool CSSFontFaceSrcValue::traverseSubresources(const WTF::Function<bool (const CachedResource&)>& handler) const
75{
76 if (!m_cachedFont)
77 return false;
78 return handler(*m_cachedFont);
79}
80
81std::unique_ptr<FontLoadRequest> CSSFontFaceSrcValue::fontLoadRequest(ScriptExecutionContext* context, bool isSVG, bool isInitiatingElementInUserAgentShadowTree)
82{
83 if (m_cachedFont)
84 return makeUnique<CachedFontLoadRequest>(*m_cachedFont);
85
86 auto request = context->fontLoadRequest(m_resource, isSVG, isInitiatingElementInUserAgentShadowTree, m_loadedFromOpaqueSource);
87 if (is<CachedFontLoadRequest>(request.get()))
88 m_cachedFont = &downcast<CachedFontLoadRequest>(request.get())->cachedFont();
89
90 return request;
91}
92
93bool CSSFontFaceSrcValue::equals(const CSSFontFaceSrcValue& other) const
94{
95 return m_isLocal == other.m_isLocal && m_format == other.m_format && m_resource == other.m_resource;
96}
