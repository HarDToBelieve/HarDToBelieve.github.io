30namespace WebCore {
31
32// We use a CSSFontFamily instead of a String to store the font family inside CSSPrimitiveValue.
33// This is because require an extra bit to determine if the font family was resolved by the CSS
34// parser from a system font ID (caption | icon | menu | message-box | small-caption |
35// status-bar).
36//
37// This is needed by the StyleBuilder to correctly set the "isSpecifiedFont" flag on the
38// FontDescription. This flag is used to determine if we should do the "use backslash as Yen
39// sign" hack.
40
41struct CSSFontFamily {
42 String familyName;
43 bool fromSystemFontID;
44};
45
46inline bool operator==(const CSSFontFamily& a, const CSSFontFamily& b)
47{
48 return a.familyName == b.familyName && a.fromSystemFontID == b.fromSystemFontID;
49}

51} // namespace WebCore
