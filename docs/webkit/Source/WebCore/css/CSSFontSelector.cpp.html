65CSSFontSelector::CSSFontSelector(ScriptExecutionContext& context)
66 : ActiveDOMObject(&context)
67 , m_context(makeWeakPtr(context))
68 , m_fontCache(makeRef(context.fontCache()))
69 , m_cssFontFaceSet(CSSFontFaceSet::create(this))
70 , m_fontModifiedObserver([this] { fontModified(); })
71 , m_uniqueId(++fontSelectorId)
72 , m_version(0)
73{
74 if (is<Document>(context)) {
75 m_fontFamilyNames.reserveInitialCapacity(familyNames->size());
76 for (auto& familyName : familyNames.get())
77 m_fontFamilyNames.uncheckedConstructAndAppend(familyName);
78 } else {
79 m_fontFamilyNames.reserveInitialCapacity(familyNamesData->size());
80 for (auto& familyName : familyNamesData.get())
81 m_fontFamilyNames.uncheckedAppend(familyName);
82 }
83
84 m_fontCache->addClient(*this);
85 m_cssFontFaceSet->addFontModifiedObserver(m_fontModifiedObserver);
86 LOG(Fonts, "CSSFontSelector %p ctor", this);
87
88 suspendIfNeeded();
89}

124void CSSFontSelector::buildStarted()
125{
126 m_buildIsUnderway = true;
127 m_cssFontFaceSet->purge();
128 ++m_version;
129
130 ASSERT(m_cssConnectionsPossiblyToRemove.isEmpty());
131 ASSERT(m_cssConnectionsEncounteredDuringBuild.isEmpty());
132 ASSERT(m_stagingArea.isEmpty());
133 for (size_t i = 0; i < m_cssFontFaceSet->faceCount(); ++i) {
134 CSSFontFace& face = m_cssFontFaceSet.get()[i];
135 if (face.cssConnection())
136 m_cssConnectionsPossiblyToRemove.add(&face);
137 }
138}

140void CSSFontSelector::buildCompleted()
141{
142 if (!m_buildIsUnderway)
143 return;
144
145 m_buildIsUnderway = false;
146
147 // Some font faces weren't re-added during the build process.
148 for (auto& face : m_cssConnectionsPossiblyToRemove) {
149 auto* connection = face->cssConnection();
150 ASSERT(connection);
151 if (!m_cssConnectionsEncounteredDuringBuild.contains(connection))
152 m_cssFontFaceSet->remove(*face);
153 }
154
155 for (auto& item : m_stagingArea)
156 addFontFaceRule(item.styleRuleFontFace, item.isInitiatingElementInUserAgentShadowTree);
157 m_cssConnectionsEncounteredDuringBuild.clear();
158 m_stagingArea.clear();
159 m_cssConnectionsPossiblyToRemove.clear();
160}

162void CSSFontSelector::addFontFaceRule(StyleRuleFontFace& fontFaceRule, bool isInitiatingElementInUserAgentShadowTree)
163{
164 if (m_buildIsUnderway) {
165 m_cssConnectionsEncounteredDuringBuild.add(&fontFaceRule);
166 m_stagingArea.append({fontFaceRule, isInitiatingElementInUserAgentShadowTree});
167 return;
168 }
169
170 const StyleProperties& style = fontFaceRule.properties();
171 RefPtr<CSSValue> fontFamily = style.getPropertyCSSValue(CSSPropertyFontFamily);
172 RefPtr<CSSValue> fontStyle = style.getPropertyCSSValue(CSSPropertyFontStyle);
173 RefPtr<CSSValue> fontWeight = style.getPropertyCSSValue(CSSPropertyFontWeight);
174 RefPtr<CSSValue> fontStretch = style.getPropertyCSSValue(CSSPropertyFontStretch);
175 RefPtr<CSSValue> src = style.getPropertyCSSValue(CSSPropertySrc);
176 RefPtr<CSSValue> unicodeRange = style.getPropertyCSSValue(CSSPropertyUnicodeRange);
177 RefPtr<CSSValue> featureSettings = style.getPropertyCSSValue(CSSPropertyFontFeatureSettings);
178 RefPtr<CSSValue> loadingBehavior = style.getPropertyCSSValue(CSSPropertyFontDisplay);
179 if (!is<CSSValueList>(fontFamily) || !is<CSSValueList>(src) || (unicodeRange && !is<CSSValueList>(*unicodeRange)))
180 return;
181
182 CSSValueList& familyList = downcast<CSSValueList>(*fontFamily);
183 if (!familyList.length())
184 return;
185
186 CSSValueList* rangeList = downcast<CSSValueList>(unicodeRange.get());
187
188 CSSValueList& srcList = downcast<CSSValueList>(*src);
189 if (!srcList.length())
190 return;
191
192 SetForScope<bool> creatingFont(m_creatingFont, true);
193 auto fontFace = CSSFontFace::create(*this, &fontFaceRule);
194
195 if (!fontFace->setFamilies(*fontFamily))
196 return;
197 if (fontStyle)
198 fontFace->setStyle(*fontStyle);
199 if (fontWeight)
200 fontFace->setWeight(*fontWeight);
201 if (fontStretch)
202 fontFace->setStretch(*fontStretch);
203 if (rangeList && !fontFace->setUnicodeRange(*rangeList))
204 return;
205 if (featureSettings)
206 fontFace->setFeatureSettings(*featureSettings);
207 if (loadingBehavior)
208 fontFace->setLoadingBehavior(*loadingBehavior);
209
210 CSSFontFace::appendSources(fontFace, srcList, m_context.get(), isInitiatingElementInUserAgentShadowTree);
211 if (fontFace->computeFailureState())
212 return;
213
214 if (RefPtr<CSSFontFace> existingFace = m_cssFontFaceSet->lookUpByCSSConnection(fontFaceRule)) {
215 // This adoption is fairly subtle. Script can trigger a purge of m_cssFontFaceSet at any time,
216 // which will cause us to just rely on the memory cache to retain the bytes of the file the next
217 // time we build up the CSSFontFaceSet. However, when the CSS Font Loading API is involved,
218 // the FontFace and FontFaceSet objects need to retain state. We create the new CSSFontFace object
219 // while the old one is still in scope so that the memory cache will be forced to retain the bytes
220 // of the resource. This means that the CachedFont will temporarily have two clients (until the
221 // old CSSFontFace goes out of scope, which should happen at the end of this "if" block). Because
222 // the CSSFontFaceSource objects will inspect their CachedFonts, the new CSSFontFace is smart enough
223 // to enter the correct state() during the next pump(). This approach of making a new CSSFontFace is
224 // simpler than computing and applying a diff of the StyleProperties.
225 m_cssFontFaceSet->remove(*existingFace);
226 if (auto* existingWrapper = existingFace->existingWrapper())
227 existingWrapper->adopt(fontFace.get());
228 }
229
230 m_cssFontFaceSet->add(fontFace.get());
231 ++m_version;
232}

283std::optional<AtomString> CSSFontSelector::resolveGenericFamily(const FontDescription& fontDescription, const AtomString& familyName)
284{
285 auto platformResult = FontDescription::platformResolveGenericFamily(fontDescription.script(), fontDescription.computedLocale(), familyName);
286 if (!platformResult.isNull())
287 return platformResult;
288
289 if (!m_context)
290 return std::nullopt;
291
292 const auto& settings = m_context->settingsValues();
293
294 UScriptCode script = fontDescription.script();
295 auto familyNameIndex = m_fontFamilyNames.find(familyName);
296 if (familyNameIndex != notFound) {
297 if (auto familyString = settings.fontGenericFamilies.fontFamily(static_cast<FamilyNamesIndex>(familyNameIndex), script))
298 return AtomString(*familyString);
299 }
300
301 return std::nullopt;
302}

304FontRanges CSSFontSelector::fontRangesForFamily(const FontDescription& fontDescription, const AtomString& familyName)
305{
306 // If this ASSERT() fires, it usually means you forgot a document.updateStyleIfNeeded() somewhere.
307 ASSERT(!m_buildIsUnderway || m_computingRootStyleFontCount);
308
309 // FIXME: The spec (and Firefox) says user specified generic families (sans-serif etc.) should be resolved before the @font-face lookup too.
310 bool resolveGenericFamilyFirst = familyName == m_fontFamilyNames.at(FamilyNamesIndex::StandardFamily);
311
312 AtomString familyForLookup = familyName;
313 std::optional<FontDescription> overrideFontDescription;
314 const FontDescription* fontDescriptionForLookup = &fontDescription;
315 auto resolveAndAssignGenericFamily = [&]() {
316 if (auto genericFamilyOptional = resolveGenericFamily(fontDescription, familyName))
317 familyForLookup = *genericFamilyOptional;
318 };
319
320 if (resolveGenericFamilyFirst)
321 resolveAndAssignGenericFamily();
322 Document* document = is<Document>(m_context.get()) ? &downcast<Document>(*m_context) : nullptr;
323 auto* face = m_cssFontFaceSet->fontFace(fontDescriptionForLookup->fontSelectionRequest(), familyForLookup);
324 if (face) {
325 if (document && RuntimeEnabledFeatures::sharedFeatures().webAPIStatisticsEnabled())
326 ResourceLoadObserver::shared().logFontLoad(*document, familyForLookup.string(), true);
327 return face->fontRanges(*fontDescriptionForLookup);
328 }
329
330 if (!resolveGenericFamilyFirst)
331 resolveAndAssignGenericFamily();
332 auto font = m_fontCache->fontForFamily(*fontDescriptionForLookup, familyForLookup);
333 if (document && RuntimeEnabledFeatures::sharedFeatures().webAPIStatisticsEnabled())
334 ResourceLoadObserver::shared().logFontLoad(*document, familyForLookup.string(), !!font);
335 return FontRanges { WTFMove(font) };
336}

345size_t CSSFontSelector::fallbackFontCount()
346{
347 if (m_isStopped)
348 return 0;
349
350 return m_context->settingsValues().fontFallbackPrefersPictographs ? 1 : 0;
351}

353RefPtr<Font> CSSFontSelector::fallbackFontAt(const FontDescription& fontDescription, size_t index)
354{
355 ASSERT_UNUSED(index, !index);
356
357 if (m_isStopped)
358 return nullptr;
359
360 if (!m_context->settingsValues().fontFallbackPrefersPictographs)
361 return nullptr;
362 auto& pictographFontFamily = m_context->settingsValues().fontGenericFamilies.pictographFontFamily();
363 auto font = m_fontCache->fontForFamily(fontDescription, pictographFontFamily);
364 if (RuntimeEnabledFeatures::sharedFeatures().webAPIStatisticsEnabled() && is<Document>(m_context.get()))
365 ResourceLoadObserver::shared().logFontLoad(downcast<Document>(*m_context), pictographFontFamily, !!font);
366
367 return font;
368}
