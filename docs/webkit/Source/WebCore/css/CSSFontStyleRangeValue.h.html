26#pragma once
27
28#include "CSSPrimitiveValue.h"
29#include "CSSValue.h"
30#include "CSSValueList.h"
31
32namespace WebCore {
33
34class CSSFontStyleRangeValue final : public CSSValue {
35public:
36 static Ref<CSSFontStyleRangeValue> create(Ref<CSSPrimitiveValue>&& fontStyleValue)
37 {
38 return adoptRef(*new CSSFontStyleRangeValue(WTFMove(fontStyleValue), nullptr));
39 }
40 static Ref<CSSFontStyleRangeValue> create(Ref<CSSPrimitiveValue>&& fontStyleValue, RefPtr<CSSValueList>&& obliqueValues)
41 {
42 return adoptRef(*new CSSFontStyleRangeValue(WTFMove(fontStyleValue), WTFMove(obliqueValues)));
43 }
44
45 String customCSSText() const;
46
47 bool equals(const CSSFontStyleRangeValue&) const;
48
49 Ref<CSSPrimitiveValue> fontStyleValue;
50 RefPtr<CSSValueList> obliqueValues;
51
52private:
53 CSSFontStyleRangeValue(Ref<CSSPrimitiveValue>&& fontStyleValue, RefPtr<CSSValueList>&& obliqueValues)
54 : CSSValue(FontStyleRangeClass)
55 , fontStyleValue(WTFMove(fontStyleValue))
56 , obliqueValues(WTFMove(obliqueValues))
57 {
58 }
59};
60
61}
