33String CSSFontStyleValue::customCSSText() const
34{
35 if (!obliqueValue)
36 return fontStyleValue->cssText();
37
38 StringBuilder builder;
39 builder.append(fontStyleValue->cssText());
40 builder.append(' ');
41 builder.append(obliqueValue->cssText());
42 return builder.toString();
43}
44
45bool CSSFontStyleValue::equals(const CSSFontStyleValue& other) const
46{
47 if (!obliqueValue && !other.obliqueValue)
48 return fontStyleValue.get() == other.fontStyleValue.get();
49 return fontStyleValue.get() == other.fontStyleValue.get()
50 && obliqueValue && other.obliqueValue && *obliqueValue == *other.obliqueValue;
51}
