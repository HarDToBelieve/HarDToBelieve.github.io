26#pragma once
27
28#include "CSSValueKeywords.h"
29#include "CSSValueList.h"
30
31namespace WebCore {
32
33class CSSFunctionValue final : public CSSValueList {
34public:
35 static Ref<CSSFunctionValue> create(CSSValueID name)
36 {
37 return adoptRef(*new CSSFunctionValue(name));
38 }
39
40 String customCSSText() const;
41
42 CSSValueID name() const { return m_name; }
43
44 bool equals(const CSSFunctionValue& other) const { return m_name == other.m_name && CSSValueList::equals(other); }
45
46private:
47 CSSFunctionValue(CSSValueID name)
48 : CSSValueList(FunctionClass, CommaSeparator)
49 , m_name(name)
50 {
51 }
52
53 CSSValueID m_name { CSSValueInvalid };
54};
55
56} // namespace WebCore
57
58SPECIALIZE_TYPE_TRAITS_CSS_VALUE(CSSFunctionValue, isFunctionValue())
