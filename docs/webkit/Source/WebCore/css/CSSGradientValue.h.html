38enum CSSGradientType {
39 CSSDeprecatedLinearGradient,
40 CSSDeprecatedRadialGradient,
41 CSSPrefixedLinearGradient,
42 CSSPrefixedRadialGradient,
43 CSSLinearGradient,
44 CSSRadialGradient,
45 CSSConicGradient
46};
47enum CSSGradientRepeat { NonRepeating, Repeating };
48
49struct CSSGradientColorStop {
50 RefPtr<CSSPrimitiveValue> color;
51 RefPtr<CSSPrimitiveValue> position; // percentage or length
52 Color resolvedColor;
53};
54
55inline bool operator==(const CSSGradientColorStop& a, const CSSGradientColorStop& b)
56{
57 return compareCSSValuePtr(a.color, b.color) && compareCSSValuePtr(a.position, b.position);
58}

60class CSSGradientValue : public CSSImageGeneratorValue {
61public:
62 void setFirstX(RefPtr<CSSPrimitiveValue>&& value) { m_firstX = WTFMove(value); }
63 void setFirstY(RefPtr<CSSPrimitiveValue>&& value) { m_firstY = WTFMove(value); }
64 void setSecondX(RefPtr<CSSPrimitiveValue>&& value) { m_secondX = WTFMove(value); }
65 void setSecondY(RefPtr<CSSPrimitiveValue>&& value) { m_secondY = WTFMove(value); }
66 void addStop(CSSGradientColorStop&& stop) { m_stops.append(WTFMove(stop)); }
67 void doneAddingStops() { m_stops.shrinkToFit(); }
68 bool hasAtLeastTwoStops() const { return m_stops.size() >= 2; }
69 void resolveRGBColors();
70
71 CSSGradientType gradientType() const { return m_gradientType; }
72
73 RefPtr<Image> image(RenderElement&, const FloatSize&);
74 bool knownToBeOpaque(const RenderElement&) const;
75
76 static constexpr bool isFixedSize() { return false; }
77 static FloatSize fixedSize(const RenderElement&) { return FloatSize(); }
78 static constexpr bool isPending() { return false; }
79 static void loadSubimages(CachedResourceLoader&, const ResourceLoaderOptions&) { }
80
81 Ref<CSSGradientValue> valueWithStylesResolved(Style::BuilderState&);
82
83protected:
84 CSSGradientValue(ClassType classType, CSSGradientRepeat repeat, CSSGradientType gradientType)
85 : CSSImageGeneratorValue(classType)
86 , m_gradientType(gradientType)
87 , m_repeating(repeat == Repeating)
88 {
89 }
90
91 CSSGradientValue(const CSSGradientValue& other, ClassType classType)
92 : CSSImageGeneratorValue(classType)
93 , m_firstX(other.m_firstX)
94 , m_firstY(other.m_firstY)
95 , m_secondX(other.m_secondX)
96 , m_secondY(other.m_secondY)
97 , m_stops(other.m_stops)
98 , m_gradientType(other.m_gradientType)
99 , m_repeating(other.m_repeating)
100 , m_hasColorDerivedFromElement(other.m_hasColorDerivedFromElement)
101 {
102 }
103
104 template<typename GradientAdapter> Gradient::ColorStopVector computeStops(GradientAdapter&, const CSSToLengthConversionData&, const RenderStyle&, float maxLengthForRepeat);
105
106 auto firstX() const { return m_firstX.get(); }
107 auto firstY() const { return m_firstY.get(); }
108 auto secondX() const { return m_secondX.get(); }
109 auto secondY() const { return m_secondY.get(); }
110 auto& stops() const { return m_stops; }
111 bool isRepeating() const { return m_repeating; }
112
113 bool equals(const CSSGradientValue&) const;
114
115private:
116 bool hasColorDerivedFromElement() const;
117 bool isCacheable() const;
118
119 RefPtr<CSSPrimitiveValue> m_firstX;
120 RefPtr<CSSPrimitiveValue> m_firstY;
121 RefPtr<CSSPrimitiveValue> m_secondX;
122 RefPtr<CSSPrimitiveValue> m_secondY;
123 Vector<CSSGradientColorStop, 2> m_stops;
124 CSSGradientType m_gradientType;
125 bool m_repeating { false };
126
127 mutable std::optional<bool> m_hasColorDerivedFromElement;
128};
