40CSSGridTemplateAreasValue::CSSGridTemplateAreasValue(const NamedGridAreaMap& gridAreaMap, size_t rowCount, size_t columnCount)
41 : CSSValue(GridTemplateAreasClass)
42 , m_gridAreaMap(gridAreaMap)
43 , m_rowCount(rowCount)
44 , m_columnCount(columnCount)
45{
46 ASSERT(m_rowCount);
47 ASSERT(m_columnCount);
48}
49
50static String stringForPosition(const NamedGridAreaMap& gridAreaMap, size_t row, size_t column)
51{
52 Vector<String> candidates;
53
54 for (const auto& it : gridAreaMap) {
55 const GridArea& area = it.value;
56 if (row >= area.rows.startLine() && row < area.rows.endLine())
57 candidates.append(it.key);
58 }
59
60 for (const auto& it : gridAreaMap) {
61 const GridArea& area = it.value;
62 if (column >= area.columns.startLine() && column < area.columns.endLine() && candidates.contains(it.key))
63 return it.key;
64 }
65
66 return ".";
67}
68
69String CSSGridTemplateAreasValue::customCSSText() const
70{
71 StringBuilder builder;
72 for (size_t row = 0; row < m_rowCount; ++row) {
73 builder.append('\"');
74 for (size_t column = 0; column < m_columnCount; ++column) {
75 builder.append(stringForPosition(m_gridAreaMap, row, column));
76 if (column != m_columnCount - 1)
77 builder.append(' ');
78 }
79 builder.append('\"');
80 if (row != m_rowCount - 1)
81 builder.append(' ');
82 }
83 return builder.toString();
84}

86bool CSSGridTemplateAreasValue::equals(const CSSGridTemplateAreasValue& other) const
87{
88 return m_gridAreaMap == other.m_gridAreaMap && m_rowCount == other.m_rowCount && m_columnCount == other.m_columnCount;
89}
