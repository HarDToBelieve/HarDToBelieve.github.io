42CSSGroupingRule::CSSGroupingRule(StyleRuleGroup& groupRule, CSSStyleSheet* parent)
43 : CSSRule(parent)
44 , m_groupRule(groupRule)
45 , m_childRuleCSSOMWrappers(groupRule.childRules().size())
46{
47}
48
49CSSGroupingRule::~CSSGroupingRule()
50{
51 ASSERT(m_childRuleCSSOMWrappers.size() == m_groupRule->childRules().size());
52 for (auto& wrapper : m_childRuleCSSOMWrappers) {
53 if (wrapper)
54 wrapper->setParentRule(nullptr);
55 }
56}
57
58ExceptionOr<unsigned> CSSGroupingRule::insertRule(const String& ruleString, unsigned index)
59{
60 ASSERT(m_childRuleCSSOMWrappers.size() == m_groupRule->childRules().size());
61
62 if (index > m_groupRule->childRules().size()) {
63 // IndexSizeError: Raised if the specified index is not a valid insertion point.
64 return Exception { IndexSizeError };
65 }
66
67 CSSStyleSheet* styleSheet = parentStyleSheet();
68 RefPtr<StyleRuleBase> newRule = CSSParser::parseRule(parserContext(), styleSheet ? &styleSheet->contents() : nullptr, ruleString);
69 if (!newRule) {
70 // SyntaxError: Raised if the specified rule has a syntax error and is unparsable.
71 return Exception { SyntaxError };
72 }
73
74 if (newRule->isImportRule() || newRule->isNamespaceRule()) {
75 // FIXME: an HierarchyRequestError should also be thrown for a @charset or a nested
76 // @media rule. They are currently not getting parsed, resulting in a SyntaxError
77 // to get raised above.
78
79 // HierarchyRequestError: Raised if the rule cannot be inserted at the specified
80 // index, e.g., if an @import rule is inserted after a standard rule set or other
81 // at-rule.
82 return Exception { HierarchyRequestError };
83 }
84 CSSStyleSheet::RuleMutationScope mutationScope(this);
85
86 m_groupRule->wrapperInsertRule(index, newRule.releaseNonNull());
87
88 m_childRuleCSSOMWrappers.insert(index, RefPtr<CSSRule>());
89 return index;
90}
91
92ExceptionOr<void> CSSGroupingRule::deleteRule(unsigned index)
93{
94 ASSERT(m_childRuleCSSOMWrappers.size() == m_groupRule->childRules().size());
95
96 if (index >= m_groupRule->childRules().size()) {
97 // IndexSizeError: Raised if the specified index does not correspond to a
98 // rule in the media rule list.
99 return Exception { IndexSizeError };
100 }
101
102 CSSStyleSheet::RuleMutationScope mutationScope(this);
103
104 m_groupRule->wrapperRemoveRule(index);
105
106 if (m_childRuleCSSOMWrappers[index])
107 m_childRuleCSSOMWrappers[index]->setParentRule(0);
108 m_childRuleCSSOMWrappers.remove(index);
109
110 return { };
111}

113void CSSGroupingRule::appendCSSTextForItems(StringBuilder& result) const
114{
115 for (unsigned i = 0, size = length(); i < size; ++i)
116 result.append(" ", item(i)->cssText(), '\n');
117}
118
119unsigned CSSGroupingRule::length() const
120{
121 return m_groupRule->childRules().size();
122}
123
124CSSRule* CSSGroupingRule::item(unsigned index) const
125{
126 if (index >= length())
127 return nullptr;
128 ASSERT(m_childRuleCSSOMWrappers.size() == m_groupRule->childRules().size());
129 auto& rule = m_childRuleCSSOMWrappers[index];
130 if (!rule)
131 rule = m_groupRule->childRules()[index]->createCSSOMWrapper(const_cast<CSSGroupingRule*>(this));
132 return rule.get();
133}
134
135CSSRuleList& CSSGroupingRule::cssRules() const
136{
137 if (!m_ruleListCSSOMWrapper)
138 m_ruleListCSSOMWrapper = makeUnique<LiveCSSRuleList<CSSGroupingRule>>(const_cast<CSSGroupingRule&>(*this));
139 return *m_ruleListCSSOMWrapper;
140}

142void CSSGroupingRule::reattach(StyleRuleBase& rule)
143{
144 m_groupRule = static_cast<StyleRuleGroup&>(rule);
145 for (unsigned i = 0; i < m_childRuleCSSOMWrappers.size(); ++i) {
146 if (m_childRuleCSSOMWrappers[i])
147 m_childRuleCSSOMWrappers[i]->reattach(*m_groupRule.get().childRules()[i]);
148 }
149}
