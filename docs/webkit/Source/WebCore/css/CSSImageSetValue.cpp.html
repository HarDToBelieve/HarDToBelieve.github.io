51void CSSImageSetValue::fillImageSet()
52{
53 size_t length = this->length();
54 for (size_t i = 0; i + 1 < length; i += 2) {
55 CSSValue* imageValue = item(i);
56 CSSValue* scaleFactorValue = item(i + 1);
57
58 ASSERT(is<CSSImageValue>(imageValue) || is<CSSImageGeneratorValue>(imageValue));
59 ASSERT(is<CSSPrimitiveValue>(scaleFactorValue));
60
61 float scaleFactor = downcast<CSSPrimitiveValue>(scaleFactorValue)->floatValue(CSSUnitType::CSS_DPPX);
62 m_imagesInSet.append({ imageValue, scaleFactor });
63 }
64
65 // Sort the images so that they are stored in order from lowest resolution to highest.
66 std::stable_sort(m_imagesInSet.begin(), m_imagesInSet.end(), CSSImageSetValue::compareByScaleFactor);
67}
68
69ImageWithScale CSSImageSetValue::bestImageForScaleFactor()
70{
71 if (!m_imagesInSet.size())
72 fillImageSet();
73
74 ImageWithScale image;
75 size_t numberOfImages = m_imagesInSet.size();
76 for (size_t i = 0; i < numberOfImages; ++i) {
77 image = m_imagesInSet.at(i);
78 if (image.scaleFactor >= m_deviceScaleFactor)
79 return image;
80 }
81 return image;
82}

84CachedImage* CSSImageSetValue::cachedImage() const
85{
86 if (is<CSSImageValue>(m_selectedImageValue))
87 return downcast<CSSImageValue>(*m_selectedImageValue).cachedImage();
88 return nullptr;
89}
90
91ImageWithScale CSSImageSetValue::selectBestFitImage(const Document& document)
92{
93 updateDeviceScaleFactor(document);
94
95 if (!m_accessedBestFitImage) {
96 m_accessedBestFitImage = true;
97 m_bestFitImage = bestImageForScaleFactor();
98 }
99
100 return m_bestFitImage;
101}
102
103void CSSImageSetValue::updateDeviceScaleFactor(const Document& document)
104{
105 // FIXME: In the future, we want to take much more than deviceScaleFactor into acount here.
106 // All forms of scale should be included: Page::pageScaleFactor(), Frame::pageZoomFactor(),
107 // and any CSS transforms. https://bugs.webkit.org/show_bug.cgi?id=81698
108 float deviceScaleFactor = document.page() ? document.page()->deviceScaleFactor() : 1;
109 if (deviceScaleFactor == m_deviceScaleFactor)
110 return;
111 m_deviceScaleFactor = deviceScaleFactor;
112 m_accessedBestFitImage = false;
113 m_selectedImageValue = nullptr;
114}

116Ref<CSSImageSetValue> CSSImageSetValue::valueWithStylesResolved(Style::BuilderState& builderState)
117{
118 auto result = create();
119 for (size_t i = 0, length = this->length(); i + 1 < length; i += 2) {
120 result->append(builderState.resolveImageStyles(*itemWithoutBoundsCheck(i)));
121 result->append(*itemWithoutBoundsCheck(i + 1));
122 }
123 return equals(result) ? Ref { *this } : result;
124}
125
126String CSSImageSetValue::customCSSText() const
127{
128 StringBuilder result;
129 result.append("image-set(");
130 size_t length = this->length();
131 for (size_t i = 0; i + 1 < length; i += 2) {
132 if (i > 0)
133 result.append(", ");
134 result.append(item(i)->cssText(), ' ', item(i + 1)->cssText());
135 }
136 result.append(')');
137 return result.toString();
138}
