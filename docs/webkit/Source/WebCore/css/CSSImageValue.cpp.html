38static bool operator==(const ResolvedURL& a, const ResolvedURL& b)
39{
40 return a.specifiedURLString == b.specifiedURLString && a.resolvedURL == b.resolvedURL;
41}

43static ResolvedURL makeResolvedURL(URL&& resolvedURL)
44{
45 auto string = resolvedURL.string();
46 return { WTFMove(string), WTFMove(resolvedURL) };
47}

85URL CSSImageValue::reresolvedURL(const Document& document) const
86{
87 // Re-resolving the URL is important for cases where resolvedURL is still not an absolute URL.
88 // This can happen if there was no absolute base URL when the value was created, like a style from a document without a base URL.
89 return document.completeURL(m_location.resolvedURL.string());
90}
91
92Ref<CSSImageValue> CSSImageValue::valueWithStylesResolved(Style::BuilderState& state)
93{
94 auto location = makeResolvedURL(reresolvedURL(state.document()));
95 if (m_location == location)
96 return *this;
97 auto result = create(WTFMove(location), m_loadedFromOpaqueSource);
98 result->m_cachedImage = m_cachedImage;
99 result->m_initiatorName = m_initiatorName;
100 result->m_unresolvedValue = this;
101 return result;
102}

104CachedImage* CSSImageValue::loadImage(CachedResourceLoader& loader, const ResourceLoaderOptions& options)
105{
106 if (!m_cachedImage) {
107 ResourceLoaderOptions loadOptions = options;
108 loadOptions.loadedFromOpaqueSource = m_loadedFromOpaqueSource;
109 CachedResourceRequest request(ResourceRequest(reresolvedURL(*loader.document())), loadOptions);
110 if (m_initiatorName.isEmpty())
111 request.setInitiator(cachedResourceRequestInitiators().css);
112 else
113 request.setInitiator(m_initiatorName);
114 if (options.mode == FetchOptions::Mode::Cors) {
115 ASSERT(loader.document());
116 request.updateForAccessControl(*loader.document());
117 }
118 m_cachedImage = loader.requestImage(WTFMove(request)).value_or(nullptr);
119 for (auto imageValue = this; (imageValue = imageValue->m_unresolvedValue.get()); )
120 imageValue->m_cachedImage = m_cachedImage;
121 }
122 return m_cachedImage.value().get();
123}

125bool CSSImageValue::traverseSubresources(const WTF::Function<bool (const CachedResource&)>& handler) const
126{
127 return m_cachedImage.value_or(nullptr) && handler(**m_cachedImage);

140Ref<DeprecatedCSSOMValue> CSSImageValue::createDeprecatedCSSOMWrapper(CSSStyleDeclaration& styleDeclaration) const
141{
142 // NOTE: We expose CSSImageValues as URI primitive values in CSSOM to maintain old behavior.
143 return DeprecatedCSSOMPrimitiveValue::create(CSSPrimitiveValue::create(m_location.resolvedURL.string(), CSSUnitType::CSS_URI), styleDeclaration);
144}

146bool CSSImageValue::knownToBeOpaque(const RenderElement& renderer) const
147{
148 return m_cachedImage.value_or(nullptr) && (**m_cachedImage).currentFrameKnownToBeOpaque(&renderer);
