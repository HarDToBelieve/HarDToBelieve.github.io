53StyleRuleKeyframes::StyleRuleKeyframes(const StyleRuleKeyframes& o)
54 : StyleRuleBase(o)
55 , m_name(o.m_name)
56{
57 m_keyframes.reserveInitialCapacity(o.keyframes().size());
58 for (auto& keyframe : o.keyframes())
59 m_keyframes.uncheckedAppend(keyframe.copyRef());
60}

74void StyleRuleKeyframes::parseDeferredRulesIfNeeded() const
75{
76 if (!m_deferredRules)
77 return;
78
79 m_deferredRules->parseDeferredKeyframes(const_cast<StyleRuleKeyframes&>(*this));
80 m_deferredRules = nullptr;
81}

108std::optional<size_t> StyleRuleKeyframes::findKeyframeIndex(const String& key) const
109{
110 parseDeferredRulesIfNeeded();
111 auto keys = CSSParser::parseKeyframeKeyList(key);
112 if (keys.isEmpty())
113 return std::nullopt;
114 for (auto i = m_keyframes.size(); i--; ) {
115 if (m_keyframes[i]->keys() == keys)
116 return i;
117 }
118 return std::nullopt;
119}

133CSSKeyframesRule::~CSSKeyframesRule()
134{
135 ASSERT(m_childRuleCSSOMWrappers.size() == m_keyframesRule->keyframes().size());
136
137 for (unsigned i = 0; i < m_childRuleCSSOMWrappers.size(); ++i) {
138 if (m_childRuleCSSOMWrappers[i])
139 m_childRuleCSSOMWrappers[i]->setParentRule(0);
140 }
141}

150void CSSKeyframesRule::appendRule(const String& ruleText)
151{
152 ASSERT(m_childRuleCSSOMWrappers.size() == m_keyframesRule->keyframes().size());
153
154 CSSParser parser(parserContext());
155 RefPtr<StyleRuleKeyframe> keyframe = parser.parseKeyframeRule(ruleText);
156 if (!keyframe)
157 return;
158
159 CSSStyleSheet::RuleMutationScope mutationScope(this);
160
161 m_keyframesRule->wrapperAppendKeyframe(keyframe.releaseNonNull());
162
163 m_childRuleCSSOMWrappers.grow(length());
164}

166void CSSKeyframesRule::insertRule(const String& ruleText)
167{
168 if (CSSStyleSheet* parent = parentStyleSheet()) {
169 if (Document* ownerDocument = parent->ownerDocument())
170 ownerDocument->addConsoleMessage(MessageSource::JS, MessageLevel::Warning, "CSSKeyframesRule 'insertRule' function is deprecated. Use 'appendRule' instead."_s);
171 }
172 appendRule(ruleText);
173}
174
175void CSSKeyframesRule::deleteRule(const String& s)
176{
177 ASSERT(m_childRuleCSSOMWrappers.size() == m_keyframesRule->keyframes().size());
178
179 auto i = m_keyframesRule->findKeyframeIndex(s);
180 if (!i)
181 return;
182
183 CSSStyleSheet::RuleMutationScope mutationScope(this);
184
185 m_keyframesRule->wrapperRemoveKeyframe(*i);
186
187 if (m_childRuleCSSOMWrappers[*i])
188 m_childRuleCSSOMWrappers[*i]->setParentRule(nullptr);
189 m_childRuleCSSOMWrappers.remove(*i);
190}
191
192CSSKeyframeRule* CSSKeyframesRule::findRule(const String& s)
193{
194 auto i = m_keyframesRule->findKeyframeIndex(s);
195 return i ? item(*i) : nullptr;
196}

198String CSSKeyframesRule::cssText() const
199{
200 StringBuilder result;
201 result.append("@-webkit-keyframes ", name(), " { \n");
202 for (unsigned i = 0, size = length(); i < size; ++i)
203 result.append(" ", m_keyframesRule->keyframes()[i]->cssText(), '\n');
204 result.append('}');
205 return result.toString();
206}
207
208unsigned CSSKeyframesRule::length() const
209{
210 return m_keyframesRule->keyframes().size();
211}
212
213CSSKeyframeRule* CSSKeyframesRule::item(unsigned index) const
214{
215 if (index >= length())
216 return nullptr;
217 ASSERT(m_childRuleCSSOMWrappers.size() == m_keyframesRule->keyframes().size());
218 auto& rule = m_childRuleCSSOMWrappers[index];
219 if (!rule)
220 rule = adoptRef(*new CSSKeyframeRule(m_keyframesRule->keyframes()[index], const_cast<CSSKeyframesRule*>(this)));
221 return rule.get();
222}
