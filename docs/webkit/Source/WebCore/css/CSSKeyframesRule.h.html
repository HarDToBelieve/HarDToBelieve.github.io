36class CSSKeyframeRule;

40class StyleRuleKeyframes final : public StyleRuleBase {
41public:
42 static Ref<StyleRuleKeyframes> create(const AtomString& name);
43 static Ref<StyleRuleKeyframes> create(const AtomString& name, std::unique_ptr<DeferredStyleGroupRuleList>&&);
44 ~StyleRuleKeyframes();
45
46 const Vector<Ref<StyleRuleKeyframe>>& keyframes() const;
47 const Vector<Ref<StyleRuleKeyframe>>* keyframesWithoutDeferredParsing() const
48 {
49 return !m_deferredRules ? &m_keyframes : nullptr;
50 }
51
52 void parserAppendKeyframe(RefPtr<StyleRuleKeyframe>&&);
53 void wrapperAppendKeyframe(Ref<StyleRuleKeyframe>&&);
54 void wrapperRemoveKeyframe(unsigned);
55
56 const AtomString& name() const { return m_name; }
57 void setName(const AtomString& name) { m_name = name; }
58
59 std::optional<size_t> findKeyframeIndex(const String& key) const;
60
61 Ref<StyleRuleKeyframes> copy() const { return adoptRef(*new StyleRuleKeyframes(*this)); }
62
63 void shrinkToFit();
64
65private:
66 explicit StyleRuleKeyframes(const AtomString&);
67 StyleRuleKeyframes(const AtomString&, std::unique_ptr<DeferredStyleGroupRuleList>&&);
68 StyleRuleKeyframes(const StyleRuleKeyframes&);
69
70 void parseDeferredRulesIfNeeded() const;
71
72 mutable Vector<Ref<StyleRuleKeyframe>> m_keyframes;
73 AtomString m_name;
74
75 mutable std::unique_ptr<DeferredStyleGroupRuleList> m_deferredRules;
76};
77
78class CSSKeyframesRule final : public CSSRule {
79public:
80 static Ref<CSSKeyframesRule> create(StyleRuleKeyframes& rule, CSSStyleSheet* sheet) { return adoptRef(*new CSSKeyframesRule(rule, sheet)); }
81
82 virtual ~CSSKeyframesRule();
83
84 CSSRule::Type type() const final { return KEYFRAMES_RULE; }
85 String cssText() const final;
86 void reattach(StyleRuleBase&) final;
87
88 String name() const { return m_keyframesRule->name(); }
89 void setName(const String&);
90
91 CSSRuleList& cssRules();
92
93 void insertRule(const String& rule);
94 void appendRule(const String& rule);
95 void deleteRule(const String& key);
96 CSSKeyframeRule* findRule(const String& key);
97
98 // For IndexedGetter and CSSRuleList.
99 unsigned length() const;
100 CSSKeyframeRule* item(unsigned index) const;
101
102private:
103 CSSKeyframesRule(StyleRuleKeyframes&, CSSStyleSheet* parent);
104
105 Ref<StyleRuleKeyframes> m_keyframesRule;
106 mutable Vector<RefPtr<CSSKeyframeRule>> m_childRuleCSSOMWrappers;
107 mutable std::unique_ptr<CSSRuleList> m_ruleListCSSOMWrapper;
108};
