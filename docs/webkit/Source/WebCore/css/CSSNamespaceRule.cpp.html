42
43AtomString CSSNamespaceRule::namespaceURI() const
44{
45 return m_namespaceRule->uri();
46}
47
48AtomString CSSNamespaceRule::prefix() const
49{
50 return m_namespaceRule->prefix();
51}
52
53String CSSNamespaceRule::cssText() const
54{
55 auto prefix = this->prefix();
56 StringBuilder result;
57 result.append("@namespace ");
58 serializeIdentifier(prefix, result);
59 result.append(prefix.isEmpty() ? "" : " ", "url(", serializeString(namespaceURI()), ");");
60 return result.toString();
61}
