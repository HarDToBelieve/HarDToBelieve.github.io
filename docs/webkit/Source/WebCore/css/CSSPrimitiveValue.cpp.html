192CSSUnitType CSSPrimitiveValue::primitiveType() const
193{
194 // FIXME: Use a switch statement here.
195
196 if (primitiveUnitType() == CSSUnitType::CSS_PROPERTY_ID || primitiveUnitType() == CSSUnitType::CSS_VALUE_ID || primitiveUnitType() == CSSUnitType::CustomIdent)
197 return CSSUnitType::CSS_IDENT;
198
199 // Web-exposed content expects font family values to have CSSUnitType::CSS_STRING primitive type
200 // so we need to map our internal CSSUnitType::CSS_FONT_FAMILY type here.
201 if (primitiveUnitType() == CSSUnitType::CSS_FONT_FAMILY)
202 return CSSUnitType::CSS_STRING;
203
204 if (primitiveUnitType() != CSSUnitType::CSS_CALC)
205 return primitiveUnitType();
206
207 switch (m_value.calc->category()) {
208 case CalculationCategory::Number:
209 return CSSUnitType::CSS_NUMBER;
210 case CalculationCategory::Percent:
211 return CSSUnitType::CSS_PERCENTAGE;
212 case CalculationCategory::PercentNumber:
213 return CSSUnitType::CSS_CALC_PERCENTAGE_WITH_NUMBER;
214 case CalculationCategory::PercentLength:
215 return CSSUnitType::CSS_CALC_PERCENTAGE_WITH_LENGTH;
216 case CalculationCategory::Length:
217 case CalculationCategory::Angle:
218 case CalculationCategory::Time:
219 case CalculationCategory::Frequency:
220 return m_value.calc->primitiveType();
221 case CalculationCategory::Other:
222 return CSSUnitType::CSS_UNKNOWN;
223 }
224 return CSSUnitType::CSS_UNKNOWN;
225}

338void CSSPrimitiveValue::init(const Length& length)
339{
340 switch (length.type()) {
341 case LengthType::Auto:
342 setPrimitiveUnitType(CSSUnitType::CSS_VALUE_ID);
343 m_value.valueID = CSSValueAuto;
344 return;
345 case LengthType::Fixed:
346 setPrimitiveUnitType(CSSUnitType::CSS_PX);
347 m_value.num = length.value();
348 return;
349 case LengthType::Intrinsic:
350 setPrimitiveUnitType(CSSUnitType::CSS_VALUE_ID);
351 m_value.valueID = CSSValueIntrinsic;
352 return;
353 case LengthType::MinIntrinsic:
354 setPrimitiveUnitType(CSSUnitType::CSS_VALUE_ID);
355 m_value.valueID = CSSValueMinIntrinsic;
356 return;
357 case LengthType::MinContent:
358 setPrimitiveUnitType(CSSUnitType::CSS_VALUE_ID);
359 m_value.valueID = CSSValueMinContent;
360 return;
361 case LengthType::MaxContent:
362 setPrimitiveUnitType(CSSUnitType::CSS_VALUE_ID);
363 m_value.valueID = CSSValueMaxContent;
364 return;
365 case LengthType::FillAvailable:
366 setPrimitiveUnitType(CSSUnitType::CSS_VALUE_ID);
367 m_value.valueID = CSSValueWebkitFillAvailable;
368 return;
369 case LengthType::FitContent:
370 setPrimitiveUnitType(CSSUnitType::CSS_VALUE_ID);
371 m_value.valueID = CSSValueFitContent;
372 return;
373 case LengthType::Percent:
374 setPrimitiveUnitType(CSSUnitType::CSS_PERCENTAGE);
375 ASSERT(std::isfinite(length.percent()));
376 m_value.num = length.percent();
377 return;
378 case LengthType::Calculated:
379 case LengthType::Relative:
380 case LengthType::Undefined:
381 ASSERT_NOT_REACHED();
382 return;
383 }
384 ASSERT_NOT_REACHED();
385}

441void CSSPrimitiveValue::cleanup()
442{
443 auto type = primitiveUnitType();
444 switch (type) {
445 case CSSUnitType::CSS_STRING:
446 case CSSUnitType::CustomIdent:
447 case CSSUnitType::CSS_URI:
448 case CSSUnitType::CSS_ATTR:
449 case CSSUnitType::CSS_COUNTER_NAME:
450 if (m_value.string)
451 m_value.string->deref();
452 break;
453 case CSSUnitType::CSS_COUNTER:
454 m_value.counter->deref();
455 break;
456 case CSSUnitType::CSS_RECT:
457 m_value.rect->deref();
458 break;
459 case CSSUnitType::CSS_QUAD:
460 m_value.quad->deref();
461 break;
462 case CSSUnitType::CSS_PAIR:
463 m_value.pair->deref();
464 break;
465 case CSSUnitType::CSS_CALC:
466 if (m_value.calc)
467 m_value.calc->deref();
468 break;
469 case CSSUnitType::CSS_CALC_PERCENTAGE_WITH_NUMBER:
470 case CSSUnitType::CSS_CALC_PERCENTAGE_WITH_LENGTH:
471 ASSERT_NOT_REACHED();
472 break;
473 case CSSUnitType::CSS_SHAPE:
474 m_value.shape->deref();
475 break;
476 case CSSUnitType::CSS_FONT_FAMILY:
477 ASSERT(m_value.fontFamily);
478 delete m_value.fontFamily;
479 m_value.fontFamily = nullptr;
480 break;
481 case CSSUnitType::CSS_RGBCOLOR:
482 ASSERT(m_value.color);
483 delete m_value.color;
484 m_value.color = nullptr;
485 break;
486 case CSSUnitType::CSS_DIMENSION:
487 case CSSUnitType::CSS_NUMBER:
488 case CSSUnitType::CSS_PERCENTAGE:
489 case CSSUnitType::CSS_EMS:
490 case CSSUnitType::CSS_QUIRKY_EMS:
491 case CSSUnitType::CSS_EXS:
492 case CSSUnitType::CSS_REMS:
493 case CSSUnitType::CSS_CHS:
494 case CSSUnitType::CSS_PX:
495 case CSSUnitType::CSS_CM:
496 case CSSUnitType::CSS_MM:
497 case CSSUnitType::CSS_IN:
498 case CSSUnitType::CSS_PT:
499 case CSSUnitType::CSS_PC:
500 case CSSUnitType::CSS_DEG:
501 case CSSUnitType::CSS_RAD:
502 case CSSUnitType::CSS_GRAD:
503 case CSSUnitType::CSS_MS:
504 case CSSUnitType::CSS_S:
505 case CSSUnitType::CSS_HZ:
506 case CSSUnitType::CSS_KHZ:
507 case CSSUnitType::CSS_TURN:
508 case CSSUnitType::CSS_VW:
509 case CSSUnitType::CSS_VH:
510 case CSSUnitType::CSS_VMIN:
511 case CSSUnitType::CSS_VMAX:
512 case CSSUnitType::CSS_DPPX:
513 case CSSUnitType::CSS_DPI:
514 case CSSUnitType::CSS_DPCM:
515 case CSSUnitType::CSS_FR:
516 case CSSUnitType::CSS_Q:
517 case CSSUnitType::CSS_LHS:
518 case CSSUnitType::CSS_RLHS:
519 case CSSUnitType::CSS_IDENT:
520 case CSSUnitType::CSS_UNKNOWN:
521 case CSSUnitType::CSS_UNICODE_RANGE:
522 case CSSUnitType::CSS_PROPERTY_ID:
523 case CSSUnitType::CSS_VALUE_ID:
524 ASSERT(!isStringType(type));
525 break;
526 }
527 setPrimitiveUnitType(CSSUnitType::CSS_UNKNOWN);
528 if (m_hasCachedCSSText) {
529 cssTextCache().remove(this);
530 m_hasCachedCSSText = false;
531 }
532}

579double CSSPrimitiveValue::computeLengthDouble(const CSSToLengthConversionData& conversionData) const
580{
581 if (primitiveUnitType() == CSSUnitType::CSS_CALC) {
582 // The multiplier and factor is applied to each value in the calc expression individually
583 return m_value.calc->computeLengthPx(conversionData);
584 }
585
586 return computeNonCalcLengthDouble(conversionData, primitiveType(), m_value.num);

589static constexpr double mmPerInch = 25.4;
590static constexpr double cmPerInch = 2.54;
591static constexpr double QPerInch = 25.4 * 4.0;
592
593double CSSPrimitiveValue::computeUnzoomedNonCalcLengthDouble(CSSUnitType primitiveType, double value, CSSPropertyID propertyToCompute, const FontMetrics* fontMetrics, const FontCascadeDescription* fontDescription, const FontCascadeDescription* rootFontDescription, const RenderView* renderView)
594{
595 switch (primitiveType) {
596 case CSSUnitType::CSS_EMS:
597 case CSSUnitType::CSS_QUIRKY_EMS:
598 ASSERT(fontDescription);
599 return ((propertyToCompute == CSSPropertyFontSize) ? fontDescription->specifiedSize() : fontDescription->computedSize()) * value;
600 case CSSUnitType::CSS_EXS:
601 ASSERT(fontMetrics);
602 if (fontMetrics->hasXHeight())
603 return fontMetrics->xHeight() * value;
604 ASSERT(fontDescription);
605 return ((propertyToCompute == CSSPropertyFontSize) ? fontDescription->specifiedSize() : fontDescription->computedSize()) / 2.0 * value;
606 case CSSUnitType::CSS_REMS:
607 if (!rootFontDescription)
608 return value;
609 return ((propertyToCompute == CSSPropertyFontSize) ? rootFontDescription->specifiedSize() : rootFontDescription->computedSize()) * value;
610 case CSSUnitType::CSS_CHS:
611 ASSERT(fontMetrics);
612 return fontMetrics->zeroWidth() * value;
613 case CSSUnitType::CSS_PX:
614 return value;
615 case CSSUnitType::CSS_CM:
616 return cssPixelsPerInch / cmPerInch * value;
617 case CSSUnitType::CSS_MM:
618 return cssPixelsPerInch / mmPerInch * value;
619 case CSSUnitType::CSS_Q:
620 return cssPixelsPerInch / QPerInch * value;
621 case CSSUnitType::CSS_LHS:
622 case CSSUnitType::CSS_RLHS:
623 ASSERT_NOT_REACHED();
624 return -1.0;
625 case CSSUnitType::CSS_IN:
626 return cssPixelsPerInch * value;
627 case CSSUnitType::CSS_PT:
628 return cssPixelsPerInch / 72.0 * value;
629 case CSSUnitType::CSS_PC:
630 // 1 pc == 12 pt
631 return cssPixelsPerInch * 12.0 / 72.0 * value;
632 case CSSUnitType::CSS_CALC_PERCENTAGE_WITH_LENGTH:
633 case CSSUnitType::CSS_CALC_PERCENTAGE_WITH_NUMBER:
634 ASSERT_NOT_REACHED();
635 return -1.0;
636 case CSSUnitType::CSS_VH:
637 return renderView ? renderView->viewportSizeForCSSViewportUnits().height() / 100.0 * value : 0;
638 case CSSUnitType::CSS_VW:
639 return renderView ? renderView->viewportSizeForCSSViewportUnits().width() / 100.0 * value : 0;
640 case CSSUnitType::CSS_VMAX:
641 if (renderView) {
642 IntSize viewportSizeForCSSViewportUnits = renderView->viewportSizeForCSSViewportUnits();
643 return std::max(viewportSizeForCSSViewportUnits.width(), viewportSizeForCSSViewportUnits.height()) / 100.0 * value;
644 }
645 return value;
646 case CSSUnitType::CSS_VMIN:
647 if (renderView) {
648 IntSize viewportSizeForCSSViewportUnits = renderView->viewportSizeForCSSViewportUnits();
649 return std::min(viewportSizeForCSSViewportUnits.width(), viewportSizeForCSSViewportUnits.height()) / 100.0 * value;
650 }
651 return value;
652 default:
653 ASSERT_NOT_REACHED();
654 return -1.0;
655 }
656}

658double CSSPrimitiveValue::computeNonCalcLengthDouble(const CSSToLengthConversionData& conversionData, CSSUnitType primitiveType, double value)
659{
660 switch (primitiveType) {
661 case CSSUnitType::CSS_EMS:
662 case CSSUnitType::CSS_QUIRKY_EMS:
663 ASSERT(conversionData.style());
664 value = computeUnzoomedNonCalcLengthDouble(primitiveType, value, conversionData.propertyToCompute(), nullptr, &conversionData.style()->fontDescription());
665 break;
666
667 case CSSUnitType::CSS_EXS:
668 // FIXME: We have a bug right now where the zoom will be applied twice to EX units.
669 // We really need to compute EX using fontMetrics for the original specifiedSize and not use
670 // our actual constructed rendering font.
671 ASSERT(conversionData.style());
672 value = computeUnzoomedNonCalcLengthDouble(primitiveType, value, conversionData.propertyToCompute(), &conversionData.style()->fontMetrics(), &conversionData.style()->fontDescription());
673 break;
674
675 case CSSUnitType::CSS_REMS:
676 value = computeUnzoomedNonCalcLengthDouble(primitiveType, value, conversionData.propertyToCompute(), nullptr, nullptr, conversionData.rootStyle() ? &conversionData.rootStyle()->fontDescription() : nullptr);
677 break;
678
679 case CSSUnitType::CSS_CHS:
680 ASSERT(conversionData.style());
681 value = computeUnzoomedNonCalcLengthDouble(primitiveType, value, conversionData.propertyToCompute(), &conversionData.style()->fontMetrics());
682 break;
683
684 case CSSUnitType::CSS_PX:
685 case CSSUnitType::CSS_CM:
686 case CSSUnitType::CSS_MM:
687 case CSSUnitType::CSS_Q:
688 case CSSUnitType::CSS_IN:
689 case CSSUnitType::CSS_PT:
690 case CSSUnitType::CSS_PC:
691 case CSSUnitType::CSS_CALC_PERCENTAGE_WITH_LENGTH:
692 case CSSUnitType::CSS_CALC_PERCENTAGE_WITH_NUMBER:
693 value = computeUnzoomedNonCalcLengthDouble(primitiveType, value, conversionData.propertyToCompute());
694 break;
695
696 case CSSUnitType::CSS_VH:
697 return value * conversionData.viewportHeightFactor();
698
699 case CSSUnitType::CSS_VW:
700 return value * conversionData.viewportWidthFactor();
701
702 case CSSUnitType::CSS_VMAX:
703 return value * conversionData.viewportMaxFactor();
704
705 case CSSUnitType::CSS_VMIN:
706 return value * conversionData.viewportMinFactor();
707
708 case CSSUnitType::CSS_LHS:
709 ASSERT(conversionData.style());
710 if (conversionData.computingLineHeight() || conversionData.computingFontSize()) {
711 // Try to get the parent's computed line-height, or fall back to the initial line-height of this element's font spacing.
712 value *= conversionData.parentStyle() ? conversionData.parentStyle()->computedLineHeight() : conversionData.style()->fontMetrics().lineSpacing();
713 } else
714 value *= conversionData.style()->computedLineHeight();
715 break;
716
717 case CSSUnitType::CSS_RLHS:
718 if (conversionData.rootStyle()) {
719 if (conversionData.computingLineHeight() || conversionData.computingFontSize())
720 value *= conversionData.rootStyle()->computeLineHeight(conversionData.rootStyle()->specifiedLineHeight());
721 else
722 value *= conversionData.rootStyle()->computedLineHeight();
723 }
724 break;
725
726 default:
727 ASSERT_NOT_REACHED();
728 return -1.0;
729 }
730
731 // We do not apply the zoom factor when we are computing the value of the font-size property. The zooming
732 // for font sizes is much more complicated, since we have to worry about enforcing the minimum font size preference
733 // as well as enforcing the implicit "smart minimum."
734 if (conversionData.computingFontSize() || isFontRelativeLength(primitiveType))
735 return value;
736
737 return value * conversionData.zoom();
738}

759double CSSPrimitiveValue::conversionToCanonicalUnitsScaleFactor(CSSUnitType unitType)
760{
761 double factor = 1.0;
762 // FIXME: the switch can be replaced by an array of scale factors.
763 switch (unitType) {
764 // These are "canonical" units in their respective categories.
765 case CSSUnitType::CSS_PX:
766 case CSSUnitType::CSS_DEG:
767 case CSSUnitType::CSS_MS:
768 case CSSUnitType::CSS_HZ:
769 case CSSUnitType::CSS_DPPX:
770 break;
771 case CSSUnitType::CSS_CM:
772 factor = cssPixelsPerInch / cmPerInch;
773 break;
774 case CSSUnitType::CSS_DPCM:
775 factor = cmPerInch / cssPixelsPerInch; // (2.54 cm/in)
776 break;
777 case CSSUnitType::CSS_MM:
778 factor = cssPixelsPerInch / mmPerInch;
779 break;
780 case CSSUnitType::CSS_Q:
781 factor = cssPixelsPerInch / QPerInch;
782 break;
783 case CSSUnitType::CSS_IN:
784 factor = cssPixelsPerInch;
785 break;
786 case CSSUnitType::CSS_DPI:
787 factor = 1 / cssPixelsPerInch;
788 break;
789 case CSSUnitType::CSS_PT:
790 factor = cssPixelsPerInch / 72.0;
791 break;
792 case CSSUnitType::CSS_PC:
793 factor = cssPixelsPerInch * 12.0 / 72.0; // 1 pc == 12 pt
794 break;
795 case CSSUnitType::CSS_RAD:
796 factor = degreesPerRadianDouble;
797 break;
798 case CSSUnitType::CSS_GRAD:
799 factor = degreesPerGradientDouble;
800 break;
801 case CSSUnitType::CSS_TURN:
802 factor = degreesPerTurnDouble;
803 break;
804 case CSSUnitType::CSS_S:
805 case CSSUnitType::CSS_KHZ:
806 factor = 1000;
807 break;
808 default:
809 break;
810 }
811
812 return factor;
813}

859std::optional<double> CSSPrimitiveValue::doubleValueInternal(CSSUnitType requestedUnitType) const
860{
861 if (!isValidCSSUnitTypeForDoubleConversion(primitiveUnitType()) || !isValidCSSUnitTypeForDoubleConversion(requestedUnitType))
862 return std::nullopt;
863
864 CSSUnitType sourceUnitType = primitiveType();
865 if (requestedUnitType == sourceUnitType || requestedUnitType == CSSUnitType::CSS_DIMENSION)
866 return doubleValue();
867
868 CSSUnitCategory sourceCategory = unitCategory(sourceUnitType);
869 ASSERT(sourceCategory != CSSUnitCategory::Other);
870
871 CSSUnitType targetUnitType = requestedUnitType;
872 CSSUnitCategory targetCategory = unitCategory(targetUnitType);
873 ASSERT(targetCategory != CSSUnitCategory::Other);
874
875 // Cannot convert between unrelated unit categories if one of them is not CSSUnitCategory::Number.
876 if (sourceCategory != targetCategory && sourceCategory != CSSUnitCategory::Number && targetCategory != CSSUnitCategory::Number)
877 return std::nullopt;
878
879 if (targetCategory == CSSUnitCategory::Number) {
880 // We interpret conversion to CSSUnitType::CSS_NUMBER as conversion to a canonical unit in this value's category.
881 targetUnitType = canonicalUnitTypeForCategory(sourceCategory);
882 if (targetUnitType == CSSUnitType::CSS_UNKNOWN)
883 return std::nullopt;
884 }
885
886 if (sourceUnitType == CSSUnitType::CSS_NUMBER) {
887 // We interpret conversion from CSSUnitType::CSS_NUMBER in the same way as CSSParser::validUnit() while using non-strict mode.
888 sourceUnitType = canonicalUnitTypeForCategory(targetCategory);
889 if (sourceUnitType == CSSUnitType::CSS_UNKNOWN)
890 return std::nullopt;
891 }
892
893 double convertedValue = doubleValue();
894
895 // First convert the value from primitiveUnitType() to canonical type.
896 double factor = conversionToCanonicalUnitsScaleFactor(sourceUnitType);
897 convertedValue *= factor;
898
899 // Now convert from canonical type to the target unitType.
900 factor = conversionToCanonicalUnitsScaleFactor(targetUnitType);
901 convertedValue /= factor;
902
903 return convertedValue;
904}
