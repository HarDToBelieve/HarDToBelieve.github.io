37CSSPropertyID StylePropertyMetadata::shorthandID() const
38{
39 if (!m_isSetFromShorthand)
40 return CSSPropertyInvalid;
41
42 auto shorthands = matchingShorthandsForLonghand(static_cast<CSSPropertyID>(m_propertyID));
43 ASSERT(shorthands.size() && m_indexInShorthandsVector >= 0 && m_indexInShorthandsVector < shorthands.size());
44 return shorthands[m_indexInShorthandsVector].id();
45}
46
47void CSSProperty::wrapValueInCommaSeparatedList()
48{
49 auto list = CSSValueList::createCommaSeparated();
50 list.get().append(m_value.releaseNonNull());
51 m_value = WTFMove(list);
52}
