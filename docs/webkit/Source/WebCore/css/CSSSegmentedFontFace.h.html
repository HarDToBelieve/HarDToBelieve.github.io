34namespace WebCore {
35
36class CSSFontSelector;
37class FontDescription;
38class FontRanges;
39
40class CSSSegmentedFontFace final : public RefCounted<CSSSegmentedFontFace>, public CSSFontFace::Client {
41 WTF_MAKE_FAST_ALLOCATED;
42public:
43 static Ref<CSSSegmentedFontFace> create()
44 {
45 return adoptRef(*new CSSSegmentedFontFace());
46 }
47 ~CSSSegmentedFontFace();
48
49 void appendFontFace(Ref<CSSFontFace>&&);
50
51 FontRanges fontRanges(const FontDescription&);
52
53 Vector<Ref<CSSFontFace>, 1>& constituentFaces() { return m_fontFaces; }
54
55 // CSSFontFace::Client needs to be able to be held in a RefPtr.
56 void ref() final { RefCounted<CSSSegmentedFontFace>::ref(); }
57 void deref() final { RefCounted<CSSSegmentedFontFace>::deref(); }
58
59private:
60 CSSSegmentedFontFace();
61 void fontLoaded(CSSFontFace&) final;
62
63 HashMap<FontDescriptionKey, FontRanges, FontDescriptionKeyHash, WTF::SimpleClassHashTraits<FontDescriptionKey>> m_cache;
64 Vector<Ref<CSSFontFace>, 1> m_fontFaces;
65};
