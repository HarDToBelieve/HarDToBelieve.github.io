438static inline bool pseudoClassIsRelativeToSiblings(CSSSelector::PseudoClassType type)
439{
440 return type == CSSSelector::PseudoClassEmpty
441 || type == CSSSelector::PseudoClassFirstChild
442 || type == CSSSelector::PseudoClassFirstOfType
443 || type == CSSSelector::PseudoClassLastChild
444 || type == CSSSelector::PseudoClassLastOfType
445 || type == CSSSelector::PseudoClassOnlyChild
446 || type == CSSSelector::PseudoClassOnlyOfType
447 || type == CSSSelector::PseudoClassNthChild
448 || type == CSSSelector::PseudoClassNthOfType
449 || type == CSSSelector::PseudoClassNthLastChild
450 || type == CSSSelector::PseudoClassNthLastOfType;
451}
452
453static inline bool isTreeStructuralPseudoClass(CSSSelector::PseudoClassType type)
454{
455 return pseudoClassIsRelativeToSiblings(type) || type == CSSSelector::PseudoClassRoot;
456}

458inline bool CSSSelector::isSiblingSelector() const
459{
460 return relation() == DirectAdjacent
461 || relation() == IndirectAdjacent
462 || (match() == CSSSelector::PseudoClass && pseudoClassIsRelativeToSiblings(pseudoClassType()));
463}

465inline bool CSSSelector::isAttributeSelector() const
466{
467 return match() == CSSSelector::Exact
468 || match() == CSSSelector::Set
469 || match() == CSSSelector::List
470 || match() == CSSSelector::Hyphen
471 || match() == CSSSelector::Contain
472 || match() == CSSSelector::Begin
473 || match() == CSSSelector::End;
474}

476inline void CSSSelector::setValue(const AtomString& value, bool matchLowerCase)
477{
478 ASSERT(match() != Tag);
479 AtomString matchingValue = matchLowerCase ? value.convertToASCIILowercase() : value;
480 if (!m_hasRareData && matchingValue != value)
481 createRareData();
482
483 // Need to do ref counting manually for the union.
484 if (!m_hasRareData) {
485 if (m_data.m_value)
486 m_data.m_value->deref();
487 m_data.m_value = value.impl();
488 m_data.m_value->ref();
489 return;
490 }
491
492 m_data.m_rareData->m_matchingValue = WTFMove(matchingValue);
493 m_data.m_rareData->m_serializingValue = value;
494}

513inline CSSSelector::CSSSelector(const CSSSelector& o)
514 : m_relation(o.m_relation)
515 , m_match(o.m_match)
516 , m_pseudoType(o.m_pseudoType)
517 , m_isLastInSelectorList(o.m_isLastInSelectorList)
518 , m_isLastInTagHistory(o.m_isLastInTagHistory)
519 , m_hasRareData(o.m_hasRareData)
520 , m_hasNameWithCase(o.m_hasNameWithCase)
521 , m_isForPage(o.m_isForPage)
522 , m_tagIsForNamespaceRule(o.m_tagIsForNamespaceRule)
523 , m_caseInsensitiveAttributeValueMatching(o.m_caseInsensitiveAttributeValueMatching)
524#if !ASSERT_WITH_SECURITY_IMPLICATION_DISABLED
525 , m_destructorHasBeenCalled(false)
526#endif
527{
528 if (o.m_hasRareData) {
529 m_data.m_rareData = o.m_data.m_rareData;
530 m_data.m_rareData->ref();
531 } else if (o.m_hasNameWithCase) {
532 m_data.m_nameWithCase = o.m_data.m_nameWithCase;
533 m_data.m_nameWithCase->ref();
534 } if (o.match() == Tag) {
535 m_data.m_tagQName = o.m_data.m_tagQName;
536 m_data.m_tagQName->ref();
537 } else if (o.m_data.m_value) {
538 m_data.m_value = o.m_data.m_value;
539 m_data.m_value->ref();
540 }
541}

543inline CSSSelector::~CSSSelector()
544{
545 ASSERT_WITH_SECURITY_IMPLICATION(!m_destructorHasBeenCalled);
546#if !ASSERT_WITH_SECURITY_IMPLICATION_DISABLED
547 m_destructorHasBeenCalled = true;
548#endif
549 if (m_hasRareData) {
550 m_data.m_rareData->deref();
551 m_data.m_rareData = nullptr;
552 m_hasRareData = false;
553 } else if (m_hasNameWithCase) {
554 m_data.m_nameWithCase->deref();
555 m_data.m_nameWithCase = nullptr;
556 m_hasNameWithCase = false;
557 } else if (match() == Tag) {
558 m_data.m_tagQName->deref();
559 m_data.m_tagQName = nullptr;
560 m_match = Unknown;
561 } else if (m_data.m_value) {
562 m_data.m_value->deref();
563 m_data.m_value = nullptr;
564 }
565}
