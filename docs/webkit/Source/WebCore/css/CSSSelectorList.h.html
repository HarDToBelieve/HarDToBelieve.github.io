26#pragma once
27
28#include "CSSSelector.h"
29#include <memory>
30#include <wtf/UniqueArray.h>
31
32namespace WebCore {
33
34class CSSParserSelector;
35
36class CSSSelectorList {
37 WTF_MAKE_FAST_ALLOCATED;
38public:
39 CSSSelectorList() = default;
40 CSSSelectorList(const CSSSelectorList&);
41 CSSSelectorList(CSSSelectorList&&) = default;
42 explicit CSSSelectorList(Vector<std::unique_ptr<CSSParserSelector>>&&);
43 explicit CSSSelectorList(UniqueArray<CSSSelector>&& array)
44 : m_selectorArray(WTFMove(array)) { }
45
46 bool isEmpty() const { return !m_selectorArray; }
47 const CSSSelector* first() const { return m_selectorArray.get(); }
48 static const CSSSelector* next(const CSSSelector*);
49 const CSSSelector* selectorAt(size_t index) const { return &m_selectorArray[index]; }
50
51 size_t indexOfNextSelectorAfter(size_t index) const
52 {
53 const CSSSelector* current = selectorAt(index);
54 current = next(current);
55 if (!current)
56 return notFound;
57 return current - m_selectorArray.get();
58 }
59
60 bool selectorsNeedNamespaceResolution();
61 bool hasInvalidSelector() const;
62
63 String selectorsText() const;
64 void buildSelectorsText(StringBuilder&) const;
65
66 unsigned componentCount() const;
67 unsigned listSize() const;
68
69 CSSSelectorList& operator=(CSSSelectorList&&) = default;
70private:
71 // End of a multipart selector is indicated by m_isLastInTagHistory bit in the last item.
72 // End of the array is indicated by m_isLastInSelectorList bit in the last item.
73 UniqueArray<CSSSelector> m_selectorArray;
74};
75
76inline const CSSSelector* CSSSelectorList::next(const CSSSelector* current)
77{
78 // Skip subparts of compound selectors.
79 while (!current->isLastInTagHistory())
80 current++;
81 return current->isLastInSelectorList() ? 0 : current + 1;
82}
83
84} // namespace WebCore
