28// Used for text-shadow and box-shadow
29CSSShadowValue::CSSShadowValue(RefPtr<CSSPrimitiveValue>&& x, RefPtr<CSSPrimitiveValue>&& y, RefPtr<CSSPrimitiveValue>&& blur, RefPtr<CSSPrimitiveValue>&& spread, RefPtr<CSSPrimitiveValue>&& style, RefPtr<CSSPrimitiveValue>&& color)
30 : CSSValue(ShadowClass)
31 , x(WTFMove(x))
32 , y(WTFMove(y))
33 , blur(WTFMove(blur))
34 , spread(WTFMove(spread))
35 , style(WTFMove(style))
36 , color(WTFMove(color))
37{
38}

40String CSSShadowValue::customCSSText() const
41{
42 StringBuilder text;
43
44 if (color)
45 text.append(color->cssText());
46 if (x) {
47 if (!text.isEmpty())
48 text.append(' ');
49 text.append(x->cssText());
50 }
51 if (y) {
52 if (!text.isEmpty())
53 text.append(' ');
54 text.append(y->cssText());
55 }
56 if (blur) {
57 if (!text.isEmpty())
58 text.append(' ');
59 text.append(blur->cssText());
60 }
61 if (spread) {
62 if (!text.isEmpty())
63 text.append(' ');
64 text.append(spread->cssText());
65 }
66 if (style) {
67 if (!text.isEmpty())
68 text.append(' ');
69 text.append(style->cssText());
70 }
71
72 return text.toString();
73}
74
75bool CSSShadowValue::equals(const CSSShadowValue& other) const
76{
77 return compareCSSValuePtr(color, other.color)
78 && compareCSSValuePtr(x, other.x)
79 && compareCSSValuePtr(y, other.y)
80 && compareCSSValuePtr(blur, other.blur)
81 && compareCSSValuePtr(spread, other.spread)
82 && compareCSSValuePtr(style, other.style);
83}
