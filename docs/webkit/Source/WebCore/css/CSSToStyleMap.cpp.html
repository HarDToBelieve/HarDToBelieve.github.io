181static inline bool convertToLengthSize(const CSSPrimitiveValue& primitiveValue, CSSToLengthConversionData conversionData, LengthSize& size)
182{
183 if (auto* pair = primitiveValue.pairValue()) {
184 size.width = pair->first()->convertToLength<AnyConversion>(conversionData);
185 size.height = pair->second()->convertToLength<AnyConversion>(conversionData);
186 } else
187 size.width = primitiveValue.convertToLength<AnyConversion>(conversionData);
188 return !size.width.isUndefined() && !size.height.isUndefined();
189}

191void CSSToStyleMap::mapFillSize(CSSPropertyID propertyID, FillLayer& layer, const CSSValue& value)
192{
193 if (value.treatAsInitialValue(propertyID)) {
194 layer.setSize(FillLayer::initialFillSize(layer.type()));
195 return;
196 }
197
198 if (!is<CSSPrimitiveValue>(value))
199 return;
200
201 auto& primitiveValue = downcast<CSSPrimitiveValue>(value);
202 FillSize fillSize;
203 switch (primitiveValue.valueID()) {
204 case CSSValueContain:
205 fillSize.type = FillSizeType::Contain;
206 break;
207 case CSSValueCover:
208 fillSize.type = FillSizeType::Cover;
209 break;
210 default:
211 ASSERT(fillSize.type == FillSizeType::Size);
212 if (!convertToLengthSize(primitiveValue, m_builderState.cssToLengthConversionData(), fillSize.size))
213 return;
214 break;
215 }
216 layer.setSize(fillSize);
217}

219void CSSToStyleMap::mapFillXPosition(CSSPropertyID propertyID, FillLayer& layer, const CSSValue& value)
220{
221 if (value.treatAsInitialValue(propertyID)) {
222 layer.setXPosition(FillLayer::initialFillXPosition(layer.type()));
223 return;
224 }
225
226 if (!is<CSSPrimitiveValue>(value))
227 return;
228
229 auto* primitiveValue = &downcast<CSSPrimitiveValue>(value);
230 Pair* pair = primitiveValue->pairValue();
231 Length length;
232 if (pair) {
233 ASSERT_UNUSED(propertyID, propertyID == CSSPropertyBackgroundPositionX || propertyID == CSSPropertyWebkitMaskPositionX);
234 length = Style::BuilderConverter::convertLength(m_builderState, *pair->second());
235 } else
236 length = Style::BuilderConverter::convertPositionComponentX(m_builderState, value);
237
238 layer.setXPosition(length);
239 if (pair)
240 layer.setBackgroundXOrigin(*pair->first());
241}

424void CSSToStyleMap::mapAnimationProperty(Animation& animation, const CSSValue& value)
425{
426 if (value.treatAsInitialValue(CSSPropertyAnimation)) {

428 return;
429 }

432 return;

437 return;

446 return;
447 }
448
449 animation.setProperty({ Animation::TransitionMode::SingleProperty, primitiveValue.propertyID() });

452void CSSToStyleMap::mapAnimationTimingFunction(Animation& animation, const CSSValue& value)
453{
454 if (value.treatAsInitialValue(CSSPropertyAnimationTimingFunction)) {
455 animation.setTimingFunction(Animation::initialTimingFunction());
456 return;
457 }
458
459 if (is<CSSPrimitiveValue>(value)) {
460 switch (downcast<CSSPrimitiveValue>(value).valueID()) {
461 case CSSValueLinear:
462 animation.setTimingFunction(LinearTimingFunction::create());
463 break;
464 case CSSValueEase:
465 animation.setTimingFunction(CubicBezierTimingFunction::create());
466 break;
467 case CSSValueEaseIn:
468 animation.setTimingFunction(CubicBezierTimingFunction::create(CubicBezierTimingFunction::EaseIn));
469 break;
470 case CSSValueEaseOut:
471 animation.setTimingFunction(CubicBezierTimingFunction::create(CubicBezierTimingFunction::EaseOut));
472 break;
473 case CSSValueEaseInOut:
474 animation.setTimingFunction(CubicBezierTimingFunction::create(CubicBezierTimingFunction::EaseInOut));
475 break;
476 case CSSValueStepStart:
477 animation.setTimingFunction(StepsTimingFunction::create(1, StepsTimingFunction::StepPosition::Start));
478 break;
479 case CSSValueStepEnd:
480 animation.setTimingFunction(StepsTimingFunction::create(1, StepsTimingFunction::StepPosition::End));
481 break;
482 default:
483 break;
484 }
485 return;
486 }
487
488 if (is<CSSCubicBezierTimingFunctionValue>(value)) {
489 auto& cubicTimingFunction = downcast<CSSCubicBezierTimingFunctionValue>(value);
490 animation.setTimingFunction(CubicBezierTimingFunction::create(cubicTimingFunction.x1(), cubicTimingFunction.y1(), cubicTimingFunction.x2(), cubicTimingFunction.y2()));
491 } else if (is<CSSStepsTimingFunctionValue>(value)) {
492 auto& stepsTimingFunction = downcast<CSSStepsTimingFunctionValue>(value);
493 animation.setTimingFunction(StepsTimingFunction::create(stepsTimingFunction.numberOfSteps(), stepsTimingFunction.stepPosition()));
494 } else if (is<CSSSpringTimingFunctionValue>(value)) {
495 auto& springTimingFunction = downcast<CSSSpringTimingFunctionValue>(value);
496 animation.setTimingFunction(SpringTimingFunction::create(springTimingFunction.mass(), springTimingFunction.stiffness(), springTimingFunction.damping(), springTimingFunction.initialVelocity()));
497 }
498}

500void CSSToStyleMap::mapNinePieceImage(CSSPropertyID property, CSSValue* value, NinePieceImage& image)
501{
502 // If we're not a value list, then we are "none" and don't need to alter the empty image at all.
503 if (!is<CSSValueList>(value))
504 return;
505
506 // Retrieve the border image value.
507 CSSValueList& borderImage = downcast<CSSValueList>(*value);
508
509 for (auto& current : borderImage) {
510 if (is<CSSImageValue>(current) || is<CSSImageGeneratorValue>(current) || is<CSSImageSetValue>(current))
511 image.setImage(styleImage(current.get()));
512 else if (is<CSSBorderImageSliceValue>(current))
513 mapNinePieceImageSlice(current, image);
514 else if (is<CSSValueList>(current)) {
515 CSSValueList& slashList = downcast<CSSValueList>(current.get());
516 // Map in the image slices.
517 if (is<CSSBorderImageSliceValue>(slashList.item(0)))
518 mapNinePieceImageSlice(*slashList.item(0), image);
519
520 // Map in the border slices.
521 if (slashList.item(1))
522 image.setBorderSlices(mapNinePieceImageQuad(*slashList.item(1)));
523
524 // Map in the outset.
525 if (slashList.item(2))
526 image.setOutset(mapNinePieceImageQuad(*slashList.item(2)));
527 } else if (is<CSSPrimitiveValue>(current)) {
528 // Set the appropriate rules for stretch/round/repeat of the slices.
529 mapNinePieceImageRepeat(current, image);
530 }
531 }
532
533 if (property == CSSPropertyWebkitBorderImage) {
534 // We have to preserve the legacy behavior of -webkit-border-image and make the border slices
535 // also set the border widths. We don't need to worry about percentages, since we don't even support
536 // those on real borders yet.
537 if (image.borderSlices().top().isFixed())
538 style()->setBorderTopWidth(image.borderSlices().top().value());
539 if (image.borderSlices().right().isFixed())
540 style()->setBorderRightWidth(image.borderSlices().right().value());
541 if (image.borderSlices().bottom().isFixed())
542 style()->setBorderBottomWidth(image.borderSlices().bottom().value());
543 if (image.borderSlices().left().isFixed())
544 style()->setBorderLeftWidth(image.borderSlices().left().value());
545 }
546}

548void CSSToStyleMap::mapNinePieceImageSlice(CSSValue& value, NinePieceImage& image)
549{
550 if (!is<CSSBorderImageSliceValue>(value))
551 return;
552
553 // Retrieve the border image value.
554 auto& borderImageSlice = downcast<CSSBorderImageSliceValue>(value);
555
556 // Set up a length box to represent our image slices.
557 LengthBox box;
558 Quad* slices = borderImageSlice.slices();
559 if (slices->top()->isPercentage())
560 box.top() = Length(slices->top()->doubleValue(), LengthType::Percent);
561 else
562 box.top() = Length(slices->top()->intValue(CSSUnitType::CSS_NUMBER), LengthType::Fixed);
563 if (slices->bottom()->isPercentage())
564 box.bottom() = Length(slices->bottom()->doubleValue(), LengthType::Percent);
565 else
566 box.bottom() = Length((int)slices->bottom()->floatValue(CSSUnitType::CSS_NUMBER), LengthType::Fixed);
567 if (slices->left()->isPercentage())
568 box.left() = Length(slices->left()->doubleValue(), LengthType::Percent);
569 else
570 box.left() = Length(slices->left()->intValue(CSSUnitType::CSS_NUMBER), LengthType::Fixed);
571 if (slices->right()->isPercentage())
572 box.right() = Length(slices->right()->doubleValue(), LengthType::Percent);
573 else
574 box.right() = Length(slices->right()->intValue(CSSUnitType::CSS_NUMBER), LengthType::Fixed);
575 image.setImageSlices(box);
576
577 // Set our fill mode.
578 image.setFill(borderImageSlice.m_fill);
579}

581LengthBox CSSToStyleMap::mapNinePieceImageQuad(CSSValue& value)
582{
583 if (!is<CSSPrimitiveValue>(value))
584 return LengthBox();
585
586 // Get our zoom value.
587 CSSToLengthConversionData conversionData = useSVGZoomRules() ? m_builderState.cssToLengthConversionData().copyWithAdjustedZoom(1.0f) : m_builderState.cssToLengthConversionData();
588
589 // Retrieve the primitive value.
590 auto& borderWidths = downcast<CSSPrimitiveValue>(value);
591
592 // Set up a length box to represent our image slices.
593 LengthBox box; // Defaults to 'auto' so we don't have to handle that explicitly below.
594 Quad* slices = borderWidths.quadValue();
595 if (slices->top()->isNumber())
596 box.top() = Length(slices->top()->floatValue(), LengthType::Relative);
597 else if (slices->top()->isPercentage())
598 box.top() = Length(slices->top()->doubleValue(CSSUnitType::CSS_PERCENTAGE), LengthType::Percent);
599 else if (slices->top()->isCalculatedPercentageWithLength())
600 box.top() = Length(slices->top()->cssCalcValue()->createCalculationValue(conversionData));
601 else if (slices->top()->valueID() != CSSValueAuto)
602 box.top() = slices->top()->computeLength<Length>(conversionData);
603
604 if (slices->right()->isNumber())
605 box.right() = Length(slices->right()->floatValue(), LengthType::Relative);
606 else if (slices->right()->isPercentage())
607 box.right() = Length(slices->right()->doubleValue(CSSUnitType::CSS_PERCENTAGE), LengthType::Percent);
608 else if (slices->right()->isCalculatedPercentageWithLength())
609 box.right() = Length(slices->right()->cssCalcValue()->createCalculationValue(conversionData));
610 else if (slices->right()->valueID() != CSSValueAuto)
611 box.right() = slices->right()->computeLength<Length>(conversionData);
612
613 if (slices->bottom()->isNumber())
614 box.bottom() = Length(slices->bottom()->floatValue(), LengthType::Relative);
615 else if (slices->bottom()->isPercentage())
616 box.bottom() = Length(slices->bottom()->doubleValue(CSSUnitType::CSS_PERCENTAGE), LengthType::Percent);
617 else if (slices->bottom()->isCalculatedPercentageWithLength())
618 box.bottom() = Length(slices->bottom()->cssCalcValue()->createCalculationValue(conversionData));
619 else if (slices->bottom()->valueID() != CSSValueAuto)
620 box.bottom() = slices->bottom()->computeLength<Length>(conversionData);
621
622 if (slices->left()->isNumber())
623 box.left() = Length(slices->left()->floatValue(), LengthType::Relative);
624 else if (slices->left()->isPercentage())
625 box.left() = Length(slices->left()->doubleValue(CSSUnitType::CSS_PERCENTAGE), LengthType::Percent);
626 else if (slices->left()->isCalculatedPercentageWithLength())
627 box.left() = Length(slices->left()->cssCalcValue()->createCalculationValue(conversionData));
628 else if (slices->left()->valueID() != CSSValueAuto)
629 box.left() = slices->left()->computeLength<Length>(conversionData);
630
631 return box;
632}

634void CSSToStyleMap::mapNinePieceImageRepeat(CSSValue& value, NinePieceImage& image)
635{
636 if (!is<CSSPrimitiveValue>(value))
637 return;
638
639 CSSPrimitiveValue& primitiveValue = downcast<CSSPrimitiveValue>(value);
640 Pair* pair = primitiveValue.pairValue();
641 if (!pair || !pair->first() || !pair->second())
642 return;
643
644 CSSValueID firstIdentifier = pair->first()->valueID();
645 CSSValueID secondIdentifier = pair->second()->valueID();
646
647 NinePieceImageRule horizontalRule;
648 switch (firstIdentifier) {
649 case CSSValueStretch:
650 horizontalRule = NinePieceImageRule::Stretch;
651 break;
652 case CSSValueRound:
653 horizontalRule = NinePieceImageRule::Round;
654 break;
655 case CSSValueSpace:
656 horizontalRule = NinePieceImageRule::Space;
657 break;
658 default: // CSSValueRepeat
659 horizontalRule = NinePieceImageRule::Repeat;
660 break;
661 }
662 image.setHorizontalRule(horizontalRule);
663
664 NinePieceImageRule verticalRule;
665 switch (secondIdentifier) {
666 case CSSValueStretch:
667 verticalRule = NinePieceImageRule::Stretch;
668 break;
669 case CSSValueRound:
670 verticalRule = NinePieceImageRule::Round;
671 break;
672 case CSSValueSpace:
673 verticalRule = NinePieceImageRule::Space;
674 break;
675 default: // CSSValueRepeat
676 verticalRule = NinePieceImageRule::Repeat;
677 break;
678 }
679 image.setVerticalRule(verticalRule);
680}
