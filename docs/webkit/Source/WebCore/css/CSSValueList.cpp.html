41bool CSSValueList::removeAll(CSSValue* value)
42{
43 // FIXME: Why even take a pointer?
44 if (!value)
45 return false;
46
47 return m_values.removeAllMatching([value](auto& current) {
48 return current->equals(*value);
49 }) > 0;
50}
51
52bool CSSValueList::hasValue(CSSValue* val) const
53{
54 // FIXME: Why even take a pointer?
55 if (!val)
56 return false;
57
58 for (unsigned i = 0, size = m_values.size(); i < size; ++i) {
59 if (m_values[i].get().equals(*val))
60 return true;
61 }
62 return false;
63}

65Ref<CSSValueList> CSSValueList::copy()
66{
67 RefPtr<CSSValueList> newList;
68 switch (m_valueSeparator) {
69 case SpaceSeparator:
70 newList = createSpaceSeparated();
71 break;
72 case CommaSeparator:
73 newList = createCommaSeparated();
74 break;
75 case SlashSeparator:
76 newList = createSlashSeparated();
77 break;
78 default:
79 ASSERT_NOT_REACHED();
80 }
81 for (auto& value : m_values)
82 newList->append(value.get());
83 return newList.releaseNonNull();
84}

86String CSSValueList::customCSSText() const
87{
88 StringBuilder result;
89 auto separator = separatorCSSText();
90 for (auto& value : m_values)
91 result.append(result.isEmpty() ? ""_s : separator, value.get().cssText());
92 return result.toString();
93}
94
95bool CSSValueList::equals(const CSSValueList& other) const
96{
97 if (m_valueSeparator != other.m_valueSeparator)
98 return false;
99
100 if (m_values.size() != other.m_values.size())
101 return false;
102
103 for (unsigned i = 0, size = m_values.size(); i < size; ++i) {
104 if (!m_values[i].get().equals(other.m_values[i]))
105 return false;
106 }
107 return true;
108}
109
110bool CSSValueList::equals(const CSSValue& other) const
111{
112 if (m_values.size() != 1)
113 return false;
114
115 return m_values[0].get().equals(other);
116}

118bool CSSValueList::traverseSubresources(const WTF::Function<bool (const CachedResource&)>& handler) const
119{
120 for (unsigned i = 0; i < m_values.size(); ++i) {
121 if (m_values[i].get().traverseSubresources(handler))
122 return true;
123 }
124 return false;
125}
