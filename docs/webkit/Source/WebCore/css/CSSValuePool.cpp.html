37LazyNeverDestroyed<StaticCSSValuePool> staticCSSValuePool;
38
39StaticCSSValuePool::StaticCSSValuePool()
40{
41 m_inheritedValue.construct(CSSValue::StaticCSSValue);
42 m_implicitInitialValue.construct(CSSValue::StaticCSSValue, true);
43 m_explicitInitialValue.construct(CSSValue::StaticCSSValue, false);
44 m_unsetValue.construct(CSSValue::StaticCSSValue);
45 m_revertValue.construct(CSSValue::StaticCSSValue);
46
47 m_transparentColor.construct(CSSValue::StaticCSSValue, Color::transparentBlack);
48 m_whiteColor.construct(CSSValue::StaticCSSValue, Color::white);
49 m_blackColor.construct(CSSValue::StaticCSSValue, Color::black);
50
51 for (unsigned i = firstCSSValueKeyword; i <= lastCSSValueKeyword; ++i)
52 m_identifierValues[i].construct(CSSValue::StaticCSSValue, static_cast<CSSValueID>(i));
53
54 for (unsigned i = 0; i < (maximumCacheableIntegerValue + 1); ++i) {
55 m_pixelValues[i].construct(CSSValue::StaticCSSValue, i, CSSUnitType::CSS_PX);
56 m_percentValues[i].construct(CSSValue::StaticCSSValue, i, CSSUnitType::CSS_PERCENTAGE);
57 m_numberValues[i].construct(CSSValue::StaticCSSValue, i, CSSUnitType::CSS_NUMBER);
58 }
59}

80Ref<CSSPrimitiveValue> CSSValuePool::createIdentifierValue(CSSValueID ident)
81{
82 RELEASE_ASSERT(ident >= firstCSSValueKeyword && ident <= lastCSSValueKeyword);
83 return staticCSSValuePool->m_identifierValues[ident].get();

91Ref<CSSPrimitiveValue> CSSValuePool::createColorValue(const Color& color)
92{
93 // These are the empty and deleted values of the hash table.
94 if (color == Color::transparentBlack)
95 return staticCSSValuePool->m_transparentColor.get();
96 if (color == Color::white)
97 return staticCSSValuePool->m_whiteColor.get();
98 // Just because it is common.
99 if (color == Color::black)
100 return staticCSSValuePool->m_blackColor.get();
101
102 // Remove one entry at random if the cache grows too large.
103 // FIXME: Use TinyLRUCache instead?
104 const int maximumColorCacheSize = 512;
105 if (m_colorValueCache.size() >= maximumColorCacheSize)
106 m_colorValueCache.remove(m_colorValueCache.random());
107
108 return *m_colorValueCache.ensure(color, [&color] {
109 return CSSPrimitiveValue::create(color);
110 }).iterator->value;
111}
112
113Ref<CSSPrimitiveValue> CSSValuePool::createValue(double value, CSSUnitType type)
114{
115 ASSERT(std::isfinite(value));
116
117 if (value < 0 || value > StaticCSSValuePool::maximumCacheableIntegerValue)
118 return CSSPrimitiveValue::create(value, type);
119
120 int intValue = static_cast<int>(value);
121 if (value != intValue)
122 return CSSPrimitiveValue::create(value, type);
123
124 switch (type) {
125 case CSSUnitType::CSS_PX:
126 return staticCSSValuePool->m_pixelValues[intValue].get();
127 case CSSUnitType::CSS_PERCENTAGE:
128 return staticCSSValuePool->m_percentValues[intValue].get();
129 case CSSUnitType::CSS_NUMBER:
130 return staticCSSValuePool->m_numberValues[intValue].get();
131 default:
132 return CSSPrimitiveValue::create(value, type);
133 }
134}

136Ref<CSSPrimitiveValue> CSSValuePool::createFontFamilyValue(const String& familyName, FromSystemFontID fromSystemFontID)
137{
138 // Remove one entry at random if the cache grows too large.
139 // FIXME: Use TinyLRUCache instead?
140 const int maximumFontFamilyCacheSize = 128;
141 if (m_fontFamilyValueCache.size() >= maximumFontFamilyCacheSize)
142 m_fontFamilyValueCache.remove(m_fontFamilyValueCache.random());
143
144 bool isFromSystemID = fromSystemFontID == FromSystemFontID::Yes;
145 return *m_fontFamilyValueCache.ensure({ familyName, isFromSystemID }, [&familyName, isFromSystemID] {
146 return CSSPrimitiveValue::create(CSSFontFamily { familyName, isFromSystemID });
147 }).iterator->value;
148}

150RefPtr<CSSValueList> CSSValuePool::createFontFaceValue(const AtomString& string)
151{
152 // Remove one entry at random if the cache grows too large.
153 // FIXME: Use TinyLRUCache instead?
154 const int maximumFontFaceCacheSize = 128;
155 if (m_fontFaceValueCache.size() >= maximumFontFaceCacheSize)
156 m_fontFaceValueCache.remove(m_fontFaceValueCache.random());
157
158 return m_fontFaceValueCache.ensure(string, [&string] () -> RefPtr<CSSValueList> {
159 auto result = CSSParser::parseSingleValue(CSSPropertyFontFamily, string);
160 if (!is<CSSValueList>(result))
161 return nullptr;
162 // FIXME: Make downcast work on RefPtr, remove the get() below, and save one reference count churn.
163 return downcast<CSSValueList>(result.get());
164 }).iterator->value;
165}

167void CSSValuePool::drain()
168{
169 m_colorValueCache.clear();
170 m_fontFaceValueCache.clear();
171 m_fontFamilyValueCache.clear();
172}
