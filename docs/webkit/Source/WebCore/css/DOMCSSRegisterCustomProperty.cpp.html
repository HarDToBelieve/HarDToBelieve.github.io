44ExceptionOr<void> DOMCSSRegisterCustomProperty::registerProperty(Document& document, const DOMCSSCustomPropertyDescriptor& descriptor)
45{
46 if (!isCustomPropertyName(descriptor.name))
47 return Exception { SyntaxError, "The name of this property is not a custom property name." };
48
49 RefPtr<CSSCustomPropertyValue> initialValue;
50 if (!descriptor.initialValue.isEmpty()) {
51 CSSTokenizer tokenizer(descriptor.initialValue);
52 auto styleResolver = Style::Resolver::create(document);
53
54 // We need to initialize this so that we can successfully parse computationally dependent values (like em units).
55 // We don't actually need the values to be accurate, since they will be rejected later anyway
56 auto style = styleResolver->defaultStyleForElement(nullptr);
57
58 HashSet<CSSPropertyID> dependencies;
59 CSSPropertyParser::collectParsedCustomPropertyValueDependencies(descriptor.syntax, false, dependencies, tokenizer.tokenRange(), strictCSSParserContext());
60
61 if (!dependencies.isEmpty())
62 return Exception { SyntaxError, "The given initial value must be computationally independent." };
63
64
65 Style::MatchResult matchResult;
66
67 auto parentStyle = RenderStyle::clone(*style);
68 Style::Builder dummyBuilder(*style, { document, parentStyle }, matchResult, { });
69
70 initialValue = CSSPropertyParser::parseTypedCustomPropertyValue(descriptor.name, descriptor.syntax, tokenizer.tokenRange(), dummyBuilder.state(), strictCSSParserContext());
71
72 if (!initialValue || !initialValue->isResolved())
73 return Exception { SyntaxError, "The given initial value does not parse for the given syntax." };
74
75 initialValue->collectDirectComputationalDependencies(dependencies);
76 initialValue->collectDirectRootComputationalDependencies(dependencies);
77
78 if (!dependencies.isEmpty())
79 return Exception { SyntaxError, "The given initial value must be computationally independent." };
80 }
81
82 CSSRegisteredCustomProperty property { descriptor.name, descriptor.syntax, descriptor.inherits, WTFMove(initialValue) };
83 if (!document.registerCSSProperty(WTFMove(property)))
84 return Exception { InvalidModificationError, "This property has already been registered." };
85
86 document.styleScope().didChangeStyleSheetEnvironment();
87
88 return { };
89}

91DOMCSSRegisterCustomProperty* DOMCSSRegisterCustomProperty::from(DOMCSSNamespace& css)
92{
93 auto* supplement = static_cast<DOMCSSRegisterCustomProperty*>(Supplement<DOMCSSNamespace>::from(&css, supplementName()));
94 if (!supplement) {
95 auto newSupplement = makeUnique<DOMCSSRegisterCustomProperty>(css);
96 supplement = newSupplement.get();
97 provideTo(&css, supplementName(), WTFMove(newSupplement));
98 }
99 return supplement;
100}

102const char* DOMCSSRegisterCustomProperty::supplementName()
103{
104 return "DOMCSSRegisterCustomProperty";
105}
