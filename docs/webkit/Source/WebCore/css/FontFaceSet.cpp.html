58FontFaceSet::FontFaceSet(ScriptExecutionContext& context, const Vector<RefPtr<FontFace>>& initialFaces)
59 : ActiveDOMObject(&context)
60 , m_backing(CSSFontFaceSet::create())
61 , m_readyPromise(makeUniqueRef<ReadyPromise>(*this, &FontFaceSet::readyPromiseResolve))
62{
63 m_backing->addFontEventClient(*this);
64 for (auto& face : initialFaces)
65 add(*face);
66}
67
68FontFaceSet::FontFaceSet(ScriptExecutionContext& context, CSSFontFaceSet& backing)
69 : ActiveDOMObject(&context)
70 , m_backing(backing)
71 , m_readyPromise(makeUniqueRef<ReadyPromise>(*this, &FontFaceSet::readyPromiseResolve))
72{
73 if (is<Document>(context)) {
74 auto& document = downcast<Document>(context);
75 if (document.frame())
76 m_isDocumentLoaded = document.loadEventFinished() && !document.processingLoadEvent();
77 }
78
79 if (m_isDocumentLoaded && !backing.hasActiveFontFaces())
80 m_readyPromise->resolve(*this);
81
82 m_backing->addFontEventClient(*this);
83}

139void FontFaceSet::load(const String& font, const String& text, LoadPromise&& promise)
140{
141 auto matchingFacesResult = m_backing->matchingFacesExcludingPreinstalledFonts(font, text);
142 if (matchingFacesResult.hasException()) {
143 promise.reject(matchingFacesResult.releaseException());
144 return;
145 }
146 auto matchingFaces = matchingFacesResult.releaseReturnValue();
147
148 if (matchingFaces.isEmpty()) {
149 promise.resolve({ });
150 return;
151 }
152
153 for (auto& face : matchingFaces)
154 face.get().load();
155
156 for (auto& face : matchingFaces) {
157 if (face.get().status() == CSSFontFace::Status::Failure) {
158 promise.reject(NetworkError);
159 return;
160 }
161 }
162
163 auto pendingPromise = PendingPromise::create(WTFMove(promise));
164 bool waiting = false;
165
166 for (auto& face : matchingFaces) {
167 pendingPromise->faces.append(face.get().wrapper(scriptExecutionContext()));
168 if (face.get().status() == CSSFontFace::Status::Success)
169 continue;
170 waiting = true;
171 ASSERT(face.get().existingWrapper());
172 m_pendingPromises.add(face.get().existingWrapper(), Vector<Ref<PendingPromise>>()).iterator->value.append(pendingPromise.copyRef());
173 }
174
175 if (!waiting)
176 pendingPromise->promise->resolve(pendingPromise->faces);
177}

184auto FontFaceSet::status() const -> LoadStatus
185{
186 switch (m_backing->status()) {
187 case CSSFontFaceSet::Status::Loading:
188 return LoadStatus::Loading;
189 case CSSFontFaceSet::Status::Loaded:
190 return LoadStatus::Loaded;
191 }
192 ASSERT_NOT_REACHED();
193 return LoadStatus::Loaded;
194}
195
196void FontFaceSet::faceFinished(CSSFontFace& face, CSSFontFace::Status newStatus)
197{
198 if (!face.existingWrapper())
199 return;
200
201 auto pendingPromises = m_pendingPromises.take(face.existingWrapper());
202 if (pendingPromises.isEmpty())
203 return;
204
205 for (auto& pendingPromise : pendingPromises) {
206 if (pendingPromise->hasReachedTerminalState)
207 continue;
208 if (newStatus == CSSFontFace::Status::Success) {
209 if (pendingPromise->hasOneRef()) {
210 pendingPromise->promise->resolve(pendingPromise->faces);
211 pendingPromise->hasReachedTerminalState = true;
212 }
213 } else {
214 ASSERT(newStatus == CSSFontFace::Status::Failure);
215 pendingPromise->promise->reject(NetworkError);
216 pendingPromise->hasReachedTerminalState = true;
217 }
218 }

226void FontFaceSet::documentDidFinishLoading()
227{
228 m_isDocumentLoaded = true;
229 if (!m_backing->hasActiveFontFaces() && !m_readyPromise->isFulfilled())
230 m_readyPromise->resolve(*this);

233void FontFaceSet::completedLoading()
234{
235 if (m_isDocumentLoaded && !m_readyPromise->isFulfilled())
236 m_readyPromise->resolve(*this);
