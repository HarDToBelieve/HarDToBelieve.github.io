34int intValueForLength(const Length& length, LayoutUnit maximumValue)
35{
36 return static_cast<int>(valueForLength(length, maximumValue));
37}
38
39LayoutUnit valueForLength(const Length& length, LayoutUnit maximumValue)
40{
41 switch (length.type()) {
42 case LengthType::Fixed:
43 case LengthType::Percent:
44 case LengthType::Calculated:
45 return minimumValueForLength(length, maximumValue);
46 case LengthType::FillAvailable:
47 case LengthType::Auto:
48 return maximumValue;
49 case LengthType::Relative:
50 case LengthType::Intrinsic:
51 case LengthType::MinIntrinsic:
52 case LengthType::MinContent:
53 case LengthType::MaxContent:
54 case LengthType::FitContent:
55 case LengthType::Undefined:
56 ASSERT_NOT_REACHED();
57 return 0;
58 }
59 ASSERT_NOT_REACHED();
60 return 0;
61}
62
63// FIXME: when subpixel layout is supported this copy of floatValueForLength() can be removed. See bug 71143.
64float floatValueForLength(const Length& length, LayoutUnit maximumValue)
65{
66 switch (length.type()) {
67 case LengthType::Fixed:
68 return length.value();
69 case LengthType::Percent:
70 return static_cast<float>(maximumValue * length.percent() / 100.0f);
71 case LengthType::FillAvailable:
72 case LengthType::Auto:
73 return static_cast<float>(maximumValue);
74 case LengthType::Calculated:
75 return length.nonNanCalculatedValue(maximumValue);
76 case LengthType::Relative:
77 case LengthType::Intrinsic:
78 case LengthType::MinIntrinsic:
79 case LengthType::MinContent:
80 case LengthType::MaxContent:
81 case LengthType::FitContent:
82 case LengthType::Undefined:
83 ASSERT_NOT_REACHED();
84 return 0;
85 }
86 ASSERT_NOT_REACHED();
87 return 0;
88}

90float floatValueForLength(const Length& length, float maximumValue)
91{
92 switch (length.type()) {
93 case LengthType::Fixed:
94 return length.value();
95 case LengthType::Percent:
96 return static_cast<float>(maximumValue * length.percent() / 100.0f);
97 case LengthType::FillAvailable:
98 case LengthType::Auto:
99 return static_cast<float>(maximumValue);
100 case LengthType::Calculated:
101 return length.nonNanCalculatedValue(maximumValue);
102 case LengthType::Relative:
103 case LengthType::Intrinsic:
104 case LengthType::MinIntrinsic:
105 case LengthType::MinContent:
106 case LengthType::MaxContent:
107 case LengthType::FitContent:
108 case LengthType::Undefined:
109 ASSERT_NOT_REACHED();
110 return 0;
111 }
112 ASSERT_NOT_REACHED();
113 return 0;
114}

116LayoutSize sizeForLengthSize(const LengthSize& length, const LayoutSize& maximumValue)
117{
118 return { valueForLength(length.width, maximumValue.width()), valueForLength(length.height, maximumValue.height()) };
119}
120
121LayoutPoint pointForLengthPoint(const LengthPoint& lengthPoint, const LayoutSize& maximumValue)
122{
123 return { valueForLength(lengthPoint.x(), maximumValue.width()), valueForLength(lengthPoint.y(), maximumValue.height()) };
124}
125
126FloatSize floatSizeForLengthSize(const LengthSize& lengthSize, const FloatSize& boxSize)
127{
128 return { floatValueForLength(lengthSize.width, boxSize.width()), floatValueForLength(lengthSize.height, boxSize.height()) };
129}
130
131FloatPoint floatPointForLengthPoint(const LengthPoint& lengthPoint, const FloatSize& boxSize)
132{
133 return { floatValueForLength(lengthPoint.x(), boxSize.width()), floatValueForLength(lengthPoint.y(), boxSize.height()) };
134}
