80static bool isAccessibilitySettingsDependent(const AtomString& mediaFeature)
81{
82 return mediaFeature == MediaFeatureNames::invertedColors
83 || mediaFeature == MediaFeatureNames::maxMonochrome
84 || mediaFeature == MediaFeatureNames::minMonochrome
85 || mediaFeature == MediaFeatureNames::monochrome
86 || mediaFeature == MediaFeatureNames::prefersReducedMotion
87 || mediaFeature == MediaFeatureNames::prefersContrast;
88}
89
90static bool isViewportDependent(const AtomString& mediaFeature)
91{
92 return mediaFeature == MediaFeatureNames::width
93 || mediaFeature == MediaFeatureNames::height
94 || mediaFeature == MediaFeatureNames::minWidth
95 || mediaFeature == MediaFeatureNames::minHeight
96 || mediaFeature == MediaFeatureNames::maxWidth
97 || mediaFeature == MediaFeatureNames::maxHeight
98 || mediaFeature == MediaFeatureNames::orientation
99 || mediaFeature == MediaFeatureNames::aspectRatio
100 || mediaFeature == MediaFeatureNames::minAspectRatio
101 || mediaFeature == MediaFeatureNames::maxAspectRatio;
102}

104static bool isAppearanceDependent(const AtomString& mediaFeature)
105{
106 return mediaFeature == MediaFeatureNames::prefersDarkInterface
107#if ENABLE(DARK_MODE_CSS)
108 || mediaFeature == MediaFeatureNames::prefersColorScheme
109#endif
110 ;
111}

131bool MediaQueryEvaluator::mediaTypeMatch(const String& mediaTypeToMatch) const
132{
133 return mediaTypeToMatch.isEmpty()
134 || equalLettersIgnoringASCIICase(mediaTypeToMatch, "all")
135 || equalIgnoringASCIICase(mediaTypeToMatch, m_mediaType);
136}

138bool MediaQueryEvaluator::mediaTypeMatchSpecific(const char* mediaTypeToMatch) const
139{
140 // Like mediaTypeMatch, but without the special cases for "" and "all".
141 ASSERT(mediaTypeToMatch);
142 ASSERT(mediaTypeToMatch[0] != '\0');
143 ASSERT(!equalLettersIgnoringASCIICase(StringView(mediaTypeToMatch), "all"));
144 return equalIgnoringASCIICase(m_mediaType, mediaTypeToMatch);
145}

147static bool applyRestrictor(MediaQuery::Restrictor r, bool value)
148{
149 return r == MediaQuery::Not ? !value : value;

152bool MediaQueryEvaluator::evaluate(const MediaQuerySet& querySet, MediaQueryDynamicResults* dynamicResults, Mode mode) const
153{
154 LOG_WITH_STREAM(MediaQueries, stream << "MediaQueryEvaluator::evaluate on " << (m_document ? m_document->url().string() : emptyString()));
155
156 auto& queries = querySet.queryVector();
157 if (!queries.size()) {
158 LOG_WITH_STREAM(MediaQueries, stream << "MediaQueryEvaluator::evaluate " << querySet << " returning true");
159 return true; // Empty query list evaluates to true.
160 }
161
162 // Iterate over queries, stop if any of them eval to true (OR semantics).
163 bool result = false;
164 for (size_t i = 0; i < queries.size() && !result; ++i) {
165 auto& query = queries[i];
166
167 if (query.ignored() || (!query.expressions().size() && query.mediaType().isEmpty()))
168 continue;
169
170 if (mediaTypeMatch(query.mediaType())) {
171 auto& expressions = query.expressions();
172 // Iterate through expressions, stop if any of them eval to false (AND semantics).
173 bool isDynamic = false;
174 size_t j = 0;
175 for (; j < expressions.size(); ++j) {
176 bool expressionResult = evaluate(expressions[j]);
177 if (dynamicResults) {
178 if (isViewportDependent(expressions[j].mediaFeature())) {
179 isDynamic = true;
180 dynamicResults->viewport.append({ expressions[j], expressionResult });
181 }
182 if (isAppearanceDependent(expressions[j].mediaFeature())) {
183 isDynamic = true;
184 dynamicResults->appearance.append({ expressions[j], expressionResult });
185 }
186 if (isAccessibilitySettingsDependent(expressions[j].mediaFeature())) {
187 isDynamic = true;
188 dynamicResults->accessibilitySettings.append({ expressions[j], expressionResult });
189 }
190 }
191 if (mode == Mode::AlwaysMatchDynamic && isDynamic)
192 continue;
193
194 if (!expressionResult)
195 break;
196 }
197
198 if (mode == Mode::AlwaysMatchDynamic && isDynamic) {
199 result = true;
200 continue;
201 }
202
203 // Assume true if we are at the end of the list, otherwise assume false.
204 result = applyRestrictor(query.restrictor(), expressions.size() == j);
205 } else
206 result = applyRestrictor(query.restrictor(), false);
207 }
208
209 LOG_WITH_STREAM(MediaQueries, stream << "MediaQueryEvaluator::evaluate " << querySet << " returning " << result);
210 return result;
211}

213bool MediaQueryEvaluator::evaluateForChanges(const MediaQueryDynamicResults& dynamicResults) const
214{
215 auto hasChanges = [&](auto& dynamicResultsVector) {
216 for (auto& dynamicResult : dynamicResultsVector) {
217 if (evaluate(dynamicResult.expression) != dynamicResult.result)
218 return true;
219 }
220 return false;
221 };
222
223 return hasChanges(dynamicResults.viewport) || hasChanges(dynamicResults.appearance) || hasChanges(dynamicResults.accessibilitySettings);
224}

226template<typename T, typename U> bool compareValue(T a, U b, MediaFeaturePrefix op)
227{
228 switch (op) {
229 case MinPrefix:
230 return a >= b;
231 case MaxPrefix:
232 return a <= b;
233 case NoPrefix:
234 return a == b;
235 }
236 return false;
237}

252static bool compareAspectRatioValue(CSSValue* value, int width, int height, MediaFeaturePrefix op)
253{
254 if (!is<CSSAspectRatioValue>(value))
255 return false;
256 auto& aspectRatio = downcast<CSSAspectRatioValue>(*value);
257 return compareValue(width * aspectRatio.denominatorValue(), height * aspectRatio.numeratorValue(), op);
258}

260static std::optional<double> doubleValue(CSSValue* value)
261{
262 if (!is<CSSPrimitiveValue>(value) || !downcast<CSSPrimitiveValue>(*value).isNumber())
263 return std::nullopt;
264 return downcast<CSSPrimitiveValue>(*value).doubleValue(CSSUnitType::CSS_NUMBER);
265}

267static bool zeroEvaluate(CSSValue* value, MediaFeaturePrefix op)
268{
269 auto numericValue = doubleValue(value);
270 return numericValue && compareValue(0, numericValue.value(), op);
271}

273static bool oneEvaluate(CSSValue* value, MediaFeaturePrefix op)
274{
275 if (!value)
276 return true;
277 auto numericValue = doubleValue(value);
278 return numericValue && compareValue(1, numericValue.value(), op);
279}

281static bool colorEvaluate(CSSValue* value, const CSSToLengthConversionData&, Frame& frame, MediaFeaturePrefix op)
282{
283 int bitsPerComponent = screenDepthPerComponent(frame.mainFrame().view());
284 auto numericValue = doubleValue(value);
285 if (!numericValue)
286 return bitsPerComponent;
287 return compareValue(bitsPerComponent, numericValue.value(), op);
288}

296static bool colorGamutEvaluate(CSSValue* value, const CSSToLengthConversionData&, Frame& frame, MediaFeaturePrefix)
297{
298 if (!value)
299 return true;
300
301 switch (downcast<CSSPrimitiveValue>(*value).valueID()) {
302 case CSSValueSRGB:
303 return true;
304 case CSSValueP3:
305 // FIXME: For the moment we just assume any "extended color" display is at least as good as P3.
306 return screenSupportsExtendedColor(frame.mainFrame().view());
307 case CSSValueRec2020:
308 // FIXME: At some point we should start detecting displays that support more colors.
309 return false;
310 default:
311 return false; // Any unknown value should not be considered a match.
312 }
313}

315static bool monochromeEvaluate(CSSValue* value, const CSSToLengthConversionData& conversionData, Frame& frame, MediaFeaturePrefix op)
316{
317 bool isMonochrome;
318
319 if (frame.settings().forcedDisplayIsMonochromeAccessibilityValue() == ForcedAccessibilityValue::On)
320 isMonochrome = true;
321 else if (frame.settings().forcedDisplayIsMonochromeAccessibilityValue() == ForcedAccessibilityValue::Off)
322 isMonochrome = false;
323 else
324 isMonochrome = screenIsMonochrome(frame.mainFrame().view());
325
326 if (!isMonochrome)
327 return zeroEvaluate(value, op);
328 return colorEvaluate(value, conversionData, frame, op);
329}

348static bool orientationEvaluate(CSSValue* value, const CSSToLengthConversionData&, Frame& frame, MediaFeaturePrefix)
349{
350 FrameView* view = frame.view();
351 if (!view)
352 return false;
353
354 auto width = view->layoutWidth();
355 auto height = view->layoutHeight();
356
357 if (!is<CSSPrimitiveValue>(value)) {
358 // Expression (orientation) evaluates to true if width and height >= 0.
359 return height >= 0 && width >= 0;
360 }
361
362 auto keyword = downcast<CSSPrimitiveValue>(*value).valueID();
363 bool result;
364 if (width > height) // Square viewport is portrait.
365 result = keyword == CSSValueLandscape;
366 else
367 result = keyword == CSSValuePortrait;
368
369 LOG_WITH_STREAM(MediaQueries, stream << " orientationEvaluate: view size " << width << "x" << height << " is " << value->cssText() << ": " << result);
370 return result;
371}

373static bool aspectRatioEvaluate(CSSValue* value, const CSSToLengthConversionData&, Frame& frame, MediaFeaturePrefix op)
374{
375 // ({,min-,max-}aspect-ratio)
376 // assume if we have a device, its aspect ratio is non-zero
377 if (!value)
378 return true;
379 FrameView* view = frame.view();
380 if (!view)
381 return true;
382 bool result = compareAspectRatioValue(value, view->layoutWidth(), view->layoutHeight(), op);
383 LOG_WITH_STREAM(MediaQueries, stream << " aspectRatioEvaluate: " << op << " " << aspectRatioValueAsString(value) << " actual view size " << view->layoutWidth() << "x" << view->layoutHeight() << " : " << result);
384 return result;
385}

400static bool evaluateResolution(CSSValue* value, Frame& frame, MediaFeaturePrefix op)
401{
402 // FIXME: Possible handle other media types than 'screen' and 'print'.
403 FrameView* view = frame.view();
404 if (!view)
405 return false;
406
407 float deviceScaleFactor = 0;
408
409 // This checks the actual media type applied to the document, and we know
410 // this method only got called if this media type matches the one defined
411 // in the query. Thus, if if the document's media type is "print", the
412 // media type of the query will either be "print" or "all".
413 String mediaType = view->mediaType();
414 if (equalLettersIgnoringASCIICase(mediaType, "screen"))
415 deviceScaleFactor = frame.page() ? frame.page()->deviceScaleFactor() : 1;
416 else if (equalLettersIgnoringASCIICase(mediaType, "print")) {
417 // The resolution of images while printing should not depend on the dpi
418 // of the screen. Until we support proper ways of querying this info
419 // we use 300px which is considered minimum for current printers.
420 deviceScaleFactor = 3.125; // 300dpi / 96dpi;
421 }
422
423 if (!value)
424 return !!deviceScaleFactor;
425
426 if (!is<CSSPrimitiveValue>(value))
427 return false;
428
429 auto& resolution = downcast<CSSPrimitiveValue>(*value);
430 float resolutionValue = resolution.isNumber() ? resolution.floatValue() : resolution.floatValue(CSSUnitType::CSS_DPPX);
431 bool result = compareValue(deviceScaleFactor, resolutionValue, op);
432 LOG_WITH_STREAM(MediaQueries, stream << " evaluateResolution: " << op << " " << resolutionValue << " device scale factor " << deviceScaleFactor << ": " << result);
433 return result;
434}

453static bool dynamicRangeEvaluate(CSSValue* value, const CSSToLengthConversionData&, Frame& frame, MediaFeaturePrefix)
454{
455 if (!value)
456 return false;
457
458 if (!frame.settings().hdrMediaCapabilitiesEnabled())
459 return false;
460
461 bool supportsHighDynamicRange;
462
463 if (frame.settings().forcedSupportsHighDynamicRangeValue() == ForcedAccessibilityValue::On)
464 supportsHighDynamicRange = true;
465 else if (frame.settings().forcedSupportsHighDynamicRangeValue() == ForcedAccessibilityValue::Off)
466 supportsHighDynamicRange = false;
467 else
468 supportsHighDynamicRange = screenSupportsHighDynamicRange(frame.mainFrame().view());
469
470 switch (downcast<CSSPrimitiveValue>(*value).valueID()) {
471 case CSSValueHigh:
472 return supportsHighDynamicRange;
473 case CSSValueStandard:
474 return true;
475 default:
476 return false; // Any unknown value should not be considered a match.
477 }
478}

485static std::optional<double> computeLength(CSSValue* value, bool strict, const CSSToLengthConversionData& conversionData)
486{
487 if (!is<CSSPrimitiveValue>(value))
488 return std::nullopt;
489
490 auto& primitiveValue = downcast<CSSPrimitiveValue>(*value);
491 if (primitiveValue.isNumber()) {
492 double value = primitiveValue.doubleValue();
493 // The only unitless number value allowed in strict mode is zero.
494 if (strict && value)
495 return std::nullopt;
496 return value;
497 }
498
499 if (primitiveValue.isLength())
500 return primitiveValue.computeLength<double>(conversionData);
501
502 return std::nullopt;
503}

505static bool deviceHeightEvaluate(CSSValue* value, const CSSToLengthConversionData& conversionData, Frame& frame, MediaFeaturePrefix op)
506{
507 // ({,min-,max-}device-height)
508 // assume if we have a device, assume non-zero
509 if (!value)
510 return true;
511 auto length = computeLength(value, !frame.document()->inQuirksMode(), conversionData);
512 if (!length)
513 return false;
514
515 auto height = frame.mainFrame().screenSize().height();
516 LOG_WITH_STREAM(MediaQueries, stream << " deviceHeightEvaluate: query " << op << " height " << *length << ", actual height " << height << " result: " << compareValue(height, *length, op));
517 return compareValue(height, *length, op);
518}

535static bool heightEvaluate(CSSValue* value, const CSSToLengthConversionData& conversionData, Frame& frame, MediaFeaturePrefix op)
536{
537 FrameView* view = frame.view();
538 if (!view)
539 return false;
540 int height = view->layoutHeight();
541 if (!value)
542 return height;
543
544 auto length = computeLength(value, !frame.document()->inQuirksMode(), conversionData);
545 if (!length)
546 return false;
547
548 if (auto* renderView = frame.document()->renderView())
549 height = adjustForAbsoluteZoom(height, *renderView);
550
551 LOG_WITH_STREAM(MediaQueries, stream << " heightEvaluate: query " << op << " height " << *length << ", actual height " << height << " result: " << compareValue(height, *length, op));
552 return compareValue(height, *length, op);
553}

900bool MediaQueryEvaluator::evaluate(const MediaQueryExpression& expression) const
901{
902 if (!m_document)
903 return m_fallbackResult;
904
905 auto& document = *m_document;
906 auto* frame = document.frame();
907 if (!frame || !frame->view() || !m_style)
908 return m_fallbackResult;
909
910 if (!expression.isValid())
911 return false;
912
913 static NeverDestroyed<MediaQueryFunctionMap> map = [] {
914 MediaQueryFunctionMap map;
915#define ADD_TO_FUNCTIONMAP(name, str) add(map, MediaFeatureNames::name->impl(), name##Evaluate);
916 CSS_MEDIAQUERY_NAMES_FOR_EACH_MEDIAFEATURE(ADD_TO_FUNCTIONMAP);
917#undef ADD_TO_FUNCTIONMAP
918 return map;
919 }();
920
921 auto function = map.get().get(expression.mediaFeature().impl());
922 if (!function)
923 return false;
924
925 if (!document.documentElement())
926 return false;
927
928 // Pass `nullptr` for `parentStyle` because we are in the context of a media query.
929 return function(expression.value(), { m_style, document.documentElement()->renderStyle(), nullptr, document.renderView(), 1, std::nullopt }, *frame, NoPrefix);
930}
