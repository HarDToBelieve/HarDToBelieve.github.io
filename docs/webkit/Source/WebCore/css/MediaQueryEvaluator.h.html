31#include <wtf/WeakPtr.h>

33namespace WebCore {
34
35class Document;
36class MediaQuerySet;
37class RenderStyle;
38
39struct MediaQueryResult {
40 MediaQueryExpression expression;
41 bool result;
42};
43
44struct MediaQueryDynamicResults {
45 Vector<MediaQueryResult> viewport;
46 Vector<MediaQueryResult> appearance;
47 Vector<MediaQueryResult> accessibilitySettings;
48
49 void append(const MediaQueryDynamicResults& other)
50 {
51 viewport.appendVector(other.viewport);
52 appearance.appendVector(other.appearance);
53 accessibilitySettings.appendVector(other.accessibilitySettings);
54 }
55 bool isEmpty() const { return viewport.isEmpty() && appearance.isEmpty() && accessibilitySettings.isEmpty(); }
56};
57
58// Some of the constructors are used for cases where the device characteristics are not known.
59// These can be used to prune the loading of stylesheets to only those which are not already known to not match.
60
61class MediaQueryEvaluator {
62public:
63 // Creates evaluator which evaluates only simple media queries.
64 // Evaluator returns true for "all", and returns value of \mediaFeatureResult for any media features.
65 explicit MediaQueryEvaluator(bool mediaFeatureResult = false);
66
67 // Creates evaluator which evaluates only simple media queries.
68 // Evaluator returns true for acceptedMediaType and returns value of \mediaFeatureResult for any media features.
69 MediaQueryEvaluator(const String& acceptedMediaType, bool mediaFeatureResult = false);
70
71 // Creates evaluator which evaluates full media queries.
72 WEBCORE_EXPORT MediaQueryEvaluator(const String& acceptedMediaType, const Document&, const RenderStyle*);
73
74 bool mediaTypeMatch(const String& mediaTypeToMatch) const;
75 bool mediaTypeMatchSpecific(const char* mediaTypeToMatch) const;
76
77 // Evaluates media query subexpression, ie "and (media-feature: value)" part.
78 bool evaluate(const MediaQueryExpression&) const;
79 bool evaluateForChanges(const MediaQueryDynamicResults&) const;
80
81 enum class Mode { Normal, AlwaysMatchDynamic };
82 WEBCORE_EXPORT bool evaluate(const MediaQuerySet&, MediaQueryDynamicResults* = nullptr, Mode = Mode::Normal) const;
83
84 static bool mediaAttributeMatches(Document&, const String& attributeValue);
85
86private:
87 String m_mediaType;
88 WeakPtr<const Document> m_document;
89 const RenderStyle* m_style { nullptr };
90 bool m_fallbackResult { false };
91};
