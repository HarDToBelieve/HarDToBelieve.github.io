40static inline bool isValidValueForIdentMediaFeature(const AtomString& feature, const CSSPrimitiveValue& value)
41{
42 auto valueID = value.valueID();
43
44 if (feature == MediaFeatureNames::orientation)
45 return valueID == CSSValuePortrait || valueID == CSSValueLandscape;
46 if (feature == MediaFeatureNames::colorGamut)
47 return valueID == CSSValueSRGB || valueID == CSSValueP3 || valueID == CSSValueRec2020;
48 if (feature == MediaFeatureNames::anyHover || feature == MediaFeatureNames::hover)
49 return valueID == CSSValueHover || valueID == CSSValueNone;
50 if (feature == MediaFeatureNames::anyPointer || feature == MediaFeatureNames::pointer)
51 return valueID == CSSValueFine || valueID == CSSValueCoarse || valueID == CSSValueNone;
52 if (feature == MediaFeatureNames::invertedColors)
53 return valueID == CSSValueInverted || valueID == CSSValueNone;
54#if ENABLE(APPLICATION_MANIFEST)
55 if (feature == MediaFeatureNames::displayMode)
56 return valueID == CSSValueFullscreen || valueID == CSSValueStandalone || valueID == CSSValueMinimalUi || valueID == CSSValueBrowser;
57#endif
58#if ENABLE(DARK_MODE_CSS)
59 if (feature == MediaFeatureNames::prefersColorScheme)
60 return valueID == CSSValueLight || valueID == CSSValueDark;
61#endif
62 if (feature == MediaFeatureNames::prefersContrast)
63 return valueID == CSSValueNoPreference || valueID == CSSValueMore || valueID == CSSValueLess;
64 if (feature == MediaFeatureNames::prefersReducedMotion)
65 return valueID == CSSValueNoPreference || valueID == CSSValueReduce;
66 if (feature == MediaFeatureNames::prefersDarkInterface)
67 return valueID == CSSValuePrefers || valueID == CSSValueNoPreference;
68 if (feature == MediaFeatureNames::dynamicRange)
69 return valueID == CSSValueHigh || valueID == CSSValueStandard;
70
71 return false;
72}

74static inline bool featureWithValidIdent(const AtomString& mediaFeature, const CSSPrimitiveValue& value, const MediaQueryParserContext& context)
75{
76 if (value.primitiveType() != CSSUnitType::CSS_IDENT || !isValidValueForIdentMediaFeature(mediaFeature, value))
77 return false;
78
79 return mediaFeature == MediaFeatureNames::orientation
80 || mediaFeature == MediaFeatureNames::colorGamut
81 || mediaFeature == MediaFeatureNames::anyHover
82 || mediaFeature == MediaFeatureNames::anyPointer
83 || mediaFeature == MediaFeatureNames::hover
84 || mediaFeature == MediaFeatureNames::invertedColors
85 || mediaFeature == MediaFeatureNames::pointer
86#if ENABLE(APPLICATION_MANIFEST)
87 || mediaFeature == MediaFeatureNames::displayMode
88#endif
89#if ENABLE(DARK_MODE_CSS)
90 || mediaFeature == MediaFeatureNames::prefersColorScheme
91#endif
92 || mediaFeature == MediaFeatureNames::prefersContrast
93 || mediaFeature == MediaFeatureNames::prefersReducedMotion
94 || (mediaFeature == MediaFeatureNames::prefersDarkInterface && (context.useSystemAppearance || isUASheetBehavior(context.mode)))
95 || mediaFeature == MediaFeatureNames::dynamicRange;
96}

98static inline bool featureWithValidDensity(const String& mediaFeature, const CSSPrimitiveValue& value)
99{
100 if (!value.isResolution() || value.doubleValue() <= 0)
101 return false;
102
103 return mediaFeature == MediaFeatureNames::resolution
104 || mediaFeature == MediaFeatureNames::minResolution
105 || mediaFeature == MediaFeatureNames::maxResolution;
106}
107
108static inline bool featureWithValidPositiveLength(const String& mediaFeature, const CSSPrimitiveValue& value)
109{
110 if (!(value.isLength() || (value.isNumber() && !value.doubleValue())) || value.doubleValue() < 0)
111 return false;
112
113 return mediaFeature == MediaFeatureNames::height
114 || mediaFeature == MediaFeatureNames::maxHeight
115 || mediaFeature == MediaFeatureNames::minHeight
116 || mediaFeature == MediaFeatureNames::width
117 || mediaFeature == MediaFeatureNames::maxWidth
118 || mediaFeature == MediaFeatureNames::minWidth
119 || mediaFeature == MediaFeatureNames::deviceHeight
120 || mediaFeature == MediaFeatureNames::maxDeviceHeight
121 || mediaFeature == MediaFeatureNames::minDeviceHeight
122 || mediaFeature == MediaFeatureNames::deviceWidth
123 || mediaFeature == MediaFeatureNames::minDeviceWidth
124 || mediaFeature == MediaFeatureNames::maxDeviceWidth;
125}

127static inline bool featureExpectingPositiveInteger(const String& mediaFeature)
128{
129 return mediaFeature == MediaFeatureNames::color
130 || mediaFeature == MediaFeatureNames::maxColor
131 || mediaFeature == MediaFeatureNames::minColor
132 || mediaFeature == MediaFeatureNames::colorIndex
133 || mediaFeature == MediaFeatureNames::maxColorIndex
134 || mediaFeature == MediaFeatureNames::minColorIndex
135 || mediaFeature == MediaFeatureNames::monochrome
136 || mediaFeature == MediaFeatureNames::maxMonochrome
137 || mediaFeature == MediaFeatureNames::minMonochrome;
138}
139
140static inline bool featureWithPositiveInteger(const String& mediaFeature, const CSSPrimitiveValue& value)
141{
142 if (!value.isNumber())
143 return false;
144 return featureExpectingPositiveInteger(mediaFeature);
145}

147static inline bool featureWithPositiveNumber(const String& mediaFeature, const CSSPrimitiveValue& value)
148{
149 if (!value.isNumber())
150 return false;
151
152 return mediaFeature == MediaFeatureNames::transform3d
153 || mediaFeature == MediaFeatureNames::devicePixelRatio
154 || mediaFeature == MediaFeatureNames::maxDevicePixelRatio
155 || mediaFeature == MediaFeatureNames::minDevicePixelRatio
156 || mediaFeature == MediaFeatureNames::transition
157 || mediaFeature == MediaFeatureNames::animation
158 || mediaFeature == MediaFeatureNames::transform2d;
159}
160
161static inline bool featureWithZeroOrOne(const String& mediaFeature, const CSSPrimitiveValue& value)
162{
163 if (!value.isNumber() || !(value.doubleValue() == 1 || !value.doubleValue()))
164 return false;
165
166 return mediaFeature == MediaFeatureNames::grid;
167}

169static inline bool isAspectRatioFeature(const AtomString& mediaFeature)
170{
171 return mediaFeature == MediaFeatureNames::aspectRatio
172 || mediaFeature == MediaFeatureNames::deviceAspectRatio
173 || mediaFeature == MediaFeatureNames::minAspectRatio
174 || mediaFeature == MediaFeatureNames::maxAspectRatio
175 || mediaFeature == MediaFeatureNames::minDeviceAspectRatio
176 || mediaFeature == MediaFeatureNames::maxDeviceAspectRatio;
177}

179static inline bool isFeatureValidWithoutValue(const AtomString& mediaFeature, const MediaQueryParserContext& context)
180{
181 // Media features that are prefixed by min/max cannot be used without a value.
182 return mediaFeature == MediaFeatureNames::anyHover
183 || mediaFeature == MediaFeatureNames::anyPointer
184 || mediaFeature == MediaFeatureNames::monochrome
185 || mediaFeature == MediaFeatureNames::color
186 || mediaFeature == MediaFeatureNames::colorIndex
187 || mediaFeature == MediaFeatureNames::grid
188 || mediaFeature == MediaFeatureNames::height
189 || mediaFeature == MediaFeatureNames::width
190 || mediaFeature == MediaFeatureNames::deviceHeight
191 || mediaFeature == MediaFeatureNames::deviceWidth
192 || mediaFeature == MediaFeatureNames::orientation
193 || mediaFeature == MediaFeatureNames::aspectRatio
194 || mediaFeature == MediaFeatureNames::deviceAspectRatio
195 || mediaFeature == MediaFeatureNames::hover
196 || mediaFeature == MediaFeatureNames::transform2d
197 || mediaFeature == MediaFeatureNames::transform3d
198 || mediaFeature == MediaFeatureNames::transition
199 || mediaFeature == MediaFeatureNames::animation
200 || mediaFeature == MediaFeatureNames::invertedColors
201 || mediaFeature == MediaFeatureNames::pointer
202 || mediaFeature == MediaFeatureNames::prefersContrast
203 || mediaFeature == MediaFeatureNames::prefersReducedMotion
204 || (mediaFeature == MediaFeatureNames::prefersDarkInterface && (context.useSystemAppearance || isUASheetBehavior(context.mode)))
205#if ENABLE(DARK_MODE_CSS)
206 || (mediaFeature == MediaFeatureNames::prefersColorScheme)
207#endif
208 || mediaFeature == MediaFeatureNames::devicePixelRatio
209 || mediaFeature == MediaFeatureNames::resolution
210#if ENABLE(APPLICATION_MANIFEST)
211 || mediaFeature == MediaFeatureNames::displayMode
212#endif
213 || mediaFeature == MediaFeatureNames::videoPlayableInline;
214}

216inline RefPtr<CSSPrimitiveValue> consumeFirstValue(const String& mediaFeature, CSSParserTokenRange& range)
217{
218 if (auto value = CSSPropertyParserHelpers::consumeInteger(range, 0))
219 return value;
220
221 if (!featureExpectingPositiveInteger(mediaFeature) && !isAspectRatioFeature(mediaFeature)) {
222 if (auto value = CSSPropertyParserHelpers::consumeNumber(range, ValueRange::NonNegative))
223 return value;
224 }
225
226 if (auto value = CSSPropertyParserHelpers::consumeLength(range, HTMLStandardMode, ValueRange::NonNegative))
227 return value;
228
229 if (auto value = CSSPropertyParserHelpers::consumeResolution(range))
230 return value;
231
232 if (auto value = CSSPropertyParserHelpers::consumeIdent(range))
233 return value;
234
235 return nullptr;
236}

238MediaQueryExpression::MediaQueryExpression(const String& feature, CSSParserTokenRange& range, MediaQueryParserContext& context)
239 : m_mediaFeature(feature.convertToASCIILowercase())
240 , m_isValid(false)
241{
242 RefPtr<CSSPrimitiveValue> firstValue = consumeFirstValue(m_mediaFeature, range);
243 if (!firstValue) {
244 if (isFeatureValidWithoutValue(m_mediaFeature, context)) {
245 // Valid, creates a MediaQueryExp with an 'invalid' MediaQueryExpValue
246 m_isValid = true;
247 }
248 return;
249 }
250 // Create value for media query expression that must have 1 or more values.
251 if (isAspectRatioFeature(m_mediaFeature)) {
252 if (!firstValue->isNumber() || !firstValue->doubleValue())
253 return;
254 if (!CSSPropertyParserHelpers::consumeSlashIncludingWhitespace(range))
255 return;
256 auto denominatorValue = CSSPropertyParserHelpers::consumePositiveIntegerRaw(range);
257 if (!denominatorValue)
258 return;
259
260 unsigned numerator = clampTo<unsigned>(firstValue->doubleValue());
261 m_value = CSSAspectRatioValue::create(numerator, *denominatorValue);
262 m_isValid = true;
263 return;
264 }
265 if (featureWithPositiveInteger(m_mediaFeature, *firstValue) || featureWithPositiveNumber(m_mediaFeature, *firstValue)
266 || featureWithZeroOrOne(m_mediaFeature, *firstValue) || featureWithValidDensity(m_mediaFeature, *firstValue)
267 || featureWithValidPositiveLength(m_mediaFeature, *firstValue) || featureWithValidIdent(m_mediaFeature, *firstValue, context)) {
268 m_value = firstValue;
269 m_isValid = true;
270 return;
271 }
272}

274String MediaQueryExpression::serialize() const
275{
276 if (m_serializationCache.isNull())
277 m_serializationCache = makeString('(', m_mediaFeature.convertToASCIILowercase(), m_value ? ": " : "", m_value ? m_value->cssText() : "", ')');
278 return m_serializationCache;
279}

284 return ts;
