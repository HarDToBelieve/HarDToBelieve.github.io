29WTF_MAKE_ISO_ALLOCATED_IMPL(MediaQueryList);

31MediaQueryList::MediaQueryList(Document& document, MediaQueryMatcher& matcher, Ref<MediaQuerySet>&& media, bool matches)
32 : ActiveDOMObject(&document)
33 , m_matcher(&matcher)
34 , m_media(WTFMove(media))
35 , m_evaluationRound(matcher.evaluationRound())
36 , m_changeRound(m_evaluationRound - 1) // Any value that is not the same as m_evaluationRound would do.
37 , m_matches(matches)
38{
39 matcher.addMediaQueryList(*this);
40}
41
42Ref<MediaQueryList> MediaQueryList::create(Document& document, MediaQueryMatcher& matcher, Ref<MediaQuerySet>&& media, bool matches)
43{
44 auto list = adoptRef(*new MediaQueryList(document, matcher, WTFMove(media), matches));
45 list->suspendIfNeeded();
46 return list;
47}
48
49MediaQueryList::~MediaQueryList()
50{
51 if (m_matcher)
52 m_matcher->removeMediaQueryList(*this);
53}
54
55void MediaQueryList::detachFromMatcher()
56{
57 m_matcher = nullptr;
58}
59
60String MediaQueryList::media() const
61{
62 return m_media->mediaText();
63}
64
65void MediaQueryList::addListener(RefPtr<EventListener>&& listener)
66{
67 if (!listener)
68 return;
69
70 addEventListener(eventNames().changeEvent, listener.releaseNonNull(), { });
71}
72
73void MediaQueryList::removeListener(RefPtr<EventListener>&& listener)
74{
75 if (!listener)
76 return;
77
78 removeEventListener(eventNames().changeEvent, *listener, { });
79}
80
81void MediaQueryList::evaluate(MediaQueryEvaluator& evaluator, bool& notificationNeeded)
82{
83 if (!m_matcher) {
84 notificationNeeded = false;
85 return;
86 }
87
88 if (m_evaluationRound != m_matcher->evaluationRound())
89 setMatches(evaluator.evaluate(m_media.get()));
90 notificationNeeded = m_changeRound == m_matcher->evaluationRound();
91}
92
93void MediaQueryList::setMatches(bool newValue)
94{
95 ASSERT(m_matcher);
96 m_evaluationRound = m_matcher->evaluationRound();
97
98 if (newValue == m_matches)
99 return;
100
101 m_matches = newValue;
102 m_changeRound = m_evaluationRound;
103}
104
105bool MediaQueryList::matches()
106{
107 if (m_matcher && m_evaluationRound != m_matcher->evaluationRound())
108 setMatches(m_matcher->evaluate(m_media.get()));
109 return m_matches;
110}
111
112void MediaQueryList::eventListenersDidChange()
113{
114 m_hasChangeEventListener = hasEventListeners(eventNames().changeEvent);
115}
116
117const char* MediaQueryList::activeDOMObjectName() const
118{
119 return "MediaQueryList";
120}
121
122bool MediaQueryList::virtualHasPendingActivity() const
123{
124 return m_hasChangeEventListener && m_matcher;
125}
