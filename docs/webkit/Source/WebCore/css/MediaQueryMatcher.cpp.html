41MediaQueryMatcher::MediaQueryMatcher(Document& document)
42 : m_document(makeWeakPtr(document))
43{
44}

46MediaQueryMatcher::~MediaQueryMatcher() = default;

48void MediaQueryMatcher::documentDestroyed()
49{
50 m_document = nullptr;
51 auto mediaQueryLists = std::exchange(m_mediaQueryLists, { });
52 for (auto& mediaQueryList : mediaQueryLists) {
53 if (mediaQueryList)
54 mediaQueryList->detachFromMatcher();
55 }
56}
57
58String MediaQueryMatcher::mediaType() const
59{
60 if (!m_document || !m_document->frame() || !m_document->frame()->view())
61 return String();
62
63 return m_document->frame()->view()->mediaType();
64}
65
66std::unique_ptr<RenderStyle> MediaQueryMatcher::documentElementUserAgentStyle() const
67{
68 if (!m_document || !m_document->frame())
69 return nullptr;
70
71 auto* documentElement = m_document->documentElement();
72 if (!documentElement)
73 return nullptr;
74
75 return m_document->styleScope().resolver().styleForElement(*documentElement, m_document->renderStyle(), nullptr, RuleMatchingBehavior::MatchOnlyUserAgentRules).renderStyle;
76}
77
78bool MediaQueryMatcher::evaluate(const MediaQuerySet& media)
79{
80 auto style = documentElementUserAgentStyle();
81 if (!style)
82 return false;
83 return MediaQueryEvaluator { mediaType(), *m_document, style.get() }.evaluate(media);
84}

86void MediaQueryMatcher::addMediaQueryList(MediaQueryList& list)
87{
88 ASSERT(!m_mediaQueryLists.contains(&list));
89 m_mediaQueryLists.append(makeWeakPtr(&list));
90}
91
92void MediaQueryMatcher::removeMediaQueryList(MediaQueryList& list)
93{
94 m_mediaQueryLists.removeFirst(&list);
95}
96
97RefPtr<MediaQueryList> MediaQueryMatcher::matchMedia(const String& query)
98{
99 if (!m_document)
100 return nullptr;
101
102 auto media = MediaQuerySet::create(query, MediaQueryParserContext(*m_document));
103 reportMediaQueryWarningIfNeeded(m_document.get(), media.ptr());
104 bool matches = evaluate(media.get());
105 return MediaQueryList::create(*m_document, *this, WTFMove(media), matches);
106}
107
108void MediaQueryMatcher::evaluateAll()
109{
110 ASSERT(m_document);
111
112 ++m_evaluationRound;
113
114 auto style = documentElementUserAgentStyle();
115 if (!style)
116 return;
117
118 LOG_WITH_STREAM(MediaQueries, stream << "MediaQueryMatcher::styleResolverChanged " << m_document->url());
119
120 MediaQueryEvaluator evaluator { mediaType(), *m_document, style.get() };
121
122 auto mediaQueryLists = m_mediaQueryLists;
123 for (auto& list : mediaQueryLists) {
124 if (!list)
125 continue;
126 bool notify;
127 list->evaluate(evaluator, notify);
128 if (notify) {
129 if (m_document && m_document->quirks().shouldSilenceMediaQueryListChangeEvents())
130 continue;
131
132 list->dispatchEvent(MediaQueryListEvent::create(eventNames().changeEvent, list->media(), list->matches()));
133 }
134 }
