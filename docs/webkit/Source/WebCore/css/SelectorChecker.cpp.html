81static inline void addStyleRelation(SelectorChecker::CheckingContext& checkingContext, const Element& element, Style::Relation::Type type, unsigned value = 1)
82{
83 ASSERT(value == 1 || type == Style::Relation::NthChildIndex || type == Style::Relation::AffectedByEmpty);
84 if (checkingContext.resolvingMode != SelectorChecker::Mode::ResolvingStyle)
85 return;
86 if (type == Style::Relation::AffectsNextSibling && !checkingContext.styleRelations.isEmpty()) {
87 auto& last = checkingContext.styleRelations.last();
88 if (last.type == Style::Relation::AffectsNextSibling && last.element == element.nextElementSibling()) {
89 ++last.value;
90 last.element = &element;
91 return;
92 }
93 }
94 checkingContext.styleRelations.append({ element, type, value });
95}

173bool SelectorChecker::match(const CSSSelector& selector, const Element& element, CheckingContext& checkingContext) const
174{
175 LocalContext context(selector, element, checkingContext.resolvingMode == SelectorChecker::Mode::QueryingRules ? VisitedMatchType::Disabled : VisitedMatchType::Enabled, checkingContext.pseudoId);
176
177 if (checkingContext.isMatchingHostPseudoClass) {
178 ASSERT(element.shadowRoot());
179 context.mayMatchHostPseudoClass = true;
180 }
181
182 PseudoIdSet pseudoIdSet;
183 MatchResult result = matchRecursively(checkingContext, context, pseudoIdSet);
184 if (result.match != Match::SelectorMatches)
185 return false;
186 if (checkingContext.pseudoId != PseudoId::None && !pseudoIdSet.has(checkingContext.pseudoId))
187 return false;
188
189 if (checkingContext.pseudoId == PseudoId::None && pseudoIdSet) {
190 PseudoIdSet publicPseudoIdSet = pseudoIdSet & PseudoIdSet::fromMask(static_cast<unsigned>(PseudoId::PublicPseudoIdMask));
191 if (checkingContext.resolvingMode == Mode::ResolvingStyle && publicPseudoIdSet)
192 checkingContext.pseudoIDSet = publicPseudoIdSet;
193
194 // When ignoring virtual pseudo elements, the context's pseudo should also be PseudoId::None but that does
195 // not cause a failure.
196 return checkingContext.resolvingMode == Mode::CollectingRulesIgnoringVirtualPseudoElements || result.matchType == MatchType::Element;
197 }
198 return true;
199}

201bool SelectorChecker::matchHostPseudoClass(const CSSSelector& selector, const Element& element, CheckingContext& checkingContext) const
202{
203 ASSERT(element.shadowRoot());
204 ASSERT(selector.match() == CSSSelector::PseudoClass && selector.pseudoClassType() == CSSSelector::PseudoClassHost);
205
206 if (auto* selectorList = selector.selectorList()) {
207 LocalContext context(*selectorList->first(), element, VisitedMatchType::Enabled, PseudoId::None);
208 context.inFunctionalPseudoClass = true;
209 context.pseudoElementEffective = false;
210 PseudoIdSet ignoreDynamicPseudo;
211 if (matchRecursively(checkingContext, context, ignoreDynamicPseudo).match != Match::SelectorMatches)
212 return false;
213 }
214 return true;
215}

217inline static bool hasScrollbarPseudoElement(const PseudoIdSet& dynamicPseudoIdSet)
218{
219 PseudoIdSet scrollbarIdSet = { PseudoId::Scrollbar, PseudoId::ScrollbarThumb, PseudoId::ScrollbarButton, PseudoId::ScrollbarTrack, PseudoId::ScrollbarTrackPiece, PseudoId::ScrollbarCorner };
220 if (dynamicPseudoIdSet & scrollbarIdSet)
221 return true;
222
223 // PseudoId::Resizer does not always have a scrollbar but it is a scrollbar-like pseudo element
224 // because it can have more than one pseudo element.
225 return dynamicPseudoIdSet.has(PseudoId::Resizer);
226}

228static SelectorChecker::LocalContext localContextForParent(const SelectorChecker::LocalContext& context)
229{
230 SelectorChecker::LocalContext updatedContext(context);
231 // Disable :visited matching when we see the first link.
232 if (context.element->isLink())
233 updatedContext.visitedMatchType = VisitedMatchType::Disabled;
234
235 updatedContext.isMatchElement = false;
236 updatedContext.isSubjectOrAdjacentElement = false;
237
238 if (updatedContext.mayMatchHostPseudoClass) {
239 updatedContext.element = nullptr;
240 return updatedContext;
241 }
242
243 // Move to the shadow host if matching :host and the parent is the shadow root.
244 if (context.selector->match() == CSSSelector::PseudoClass && context.selector->pseudoClassType() == CSSSelector::PseudoClassHost && is<ShadowRoot>(context.element->parentNode())) {
245 updatedContext.element = downcast<ShadowRoot>(*context.element->parentNode()).host();
246 updatedContext.mayMatchHostPseudoClass = true;
247 return updatedContext;
248 }
249
250 updatedContext.element = context.element->parentElement();
251 return updatedContext;
252}

254// Recursive check of selectors and combinators
255// It can return 4 different values:
256// * SelectorMatches - the selector matches the element e
257// * SelectorFailsLocally - the selector fails for the element e
258// * SelectorFailsAllSiblings - the selector fails for e and any sibling of e
259// * SelectorFailsCompletely - the selector fails for e and any sibling or ancestor of e
260SelectorChecker::MatchResult SelectorChecker::matchRecursively(CheckingContext& checkingContext, const LocalContext& context, PseudoIdSet& dynamicPseudoIdSet) const
261{
262 MatchType matchType = MatchType::Element;
263
264 // The first selector has to match.
265 if (!checkOne(checkingContext, context, matchType))
266 return MatchResult::fails(Match::SelectorFailsLocally);
267
268 if (context.selector->match() == CSSSelector::PseudoElement) {
269 if (context.selector->isCustomPseudoElement()) {
270 // In functional pseudo class, custom pseudo elements are always disabled.
271 // FIXME: We should accept custom pseudo elements inside :is()/:matches().
272 if (context.inFunctionalPseudoClass)
273 return MatchResult::fails(Match::SelectorFailsCompletely);
274 if (ShadowRoot* root = context.element->containingShadowRoot()) {
275 if (context.element->shadowPseudoId() != context.selector->value())
276 return MatchResult::fails(Match::SelectorFailsLocally);
277
278 if (context.selector->isWebKitCustomPseudoElement() && root->mode() != ShadowRootMode::UserAgent)
279 return MatchResult::fails(Match::SelectorFailsLocally);
280 } else
281 return MatchResult::fails(Match::SelectorFailsLocally);
282 } else {
283 if (!context.pseudoElementEffective)
284 return MatchResult::fails(Match::SelectorFailsCompletely);
285
286 if (checkingContext.resolvingMode == Mode::QueryingRules)
287 return MatchResult::fails(Match::SelectorFailsCompletely);
288
289 PseudoId pseudoId = CSSSelector::pseudoId(context.selector->pseudoElementType());
290 if (pseudoId != PseudoId::None)
291 dynamicPseudoIdSet.add(pseudoId);
292 matchType = MatchType::VirtualPseudoElementOnly;
293 }
294 }
295
296 // The rest of the selectors has to match
297 auto relation = context.selector->relation();
298
299 // Prepare next selector
300 const CSSSelector* leftSelector = context.selector->tagHistory();
301 if (!leftSelector)
302 return MatchResult::matches(matchType);
303
304 LocalContext nextContext(context);
305 nextContext.selector = leftSelector;
306
307 if (relation != CSSSelector::Subselector) {
308 // Bail-out if this selector is irrelevant for the pseudoId
309 if (context.pseudoId != PseudoId::None && !dynamicPseudoIdSet.has(context.pseudoId))
310 return MatchResult::fails(Match::SelectorFailsCompletely);
311
312 // Disable :visited matching when we try to match anything else than an ancestors.
313 if (!context.selector->hasDescendantOrChildRelation())
314 nextContext.visitedMatchType = VisitedMatchType::Disabled;
315
316 nextContext.pseudoId = PseudoId::None;
317
318 bool nextIsPart = leftSelector->match() == CSSSelector::PseudoElement && leftSelector->pseudoElementType() == CSSSelector::PseudoElementPart;
319 bool allowMultiplePseudoElements = relation == CSSSelector::ShadowDescendant && nextIsPart;
320 // Virtual pseudo element is only effective in the rightmost fragment.
321 if (!allowMultiplePseudoElements)
322 nextContext.pseudoElementEffective = false;
323
324 nextContext.isMatchElement = false;
325 }
326
327 switch (relation) {
328 case CSSSelector::DescendantSpace:
329 nextContext = localContextForParent(nextContext);
330 nextContext.firstSelectorOfTheFragment = nextContext.selector;
331 for (; nextContext.element; nextContext = localContextForParent(nextContext)) {
332 PseudoIdSet ignoreDynamicPseudo;
333 MatchResult result = matchRecursively(checkingContext, nextContext, ignoreDynamicPseudo);
334 ASSERT(!nextContext.pseudoElementEffective && !ignoreDynamicPseudo);
335
336 if (result.match == Match::SelectorMatches || result.match == Match::SelectorFailsCompletely)
337 return MatchResult::updateWithMatchType(result, matchType);
338 }
339 return MatchResult::fails(Match::SelectorFailsCompletely);
340
341 case CSSSelector::Child:
342 {
343 nextContext = localContextForParent(nextContext);
344 if (!nextContext.element)
345 return MatchResult::fails(Match::SelectorFailsCompletely);
346 nextContext.firstSelectorOfTheFragment = nextContext.selector;
347 PseudoIdSet ignoreDynamicPseudo;
348 MatchResult result = matchRecursively(checkingContext, nextContext, ignoreDynamicPseudo);
349 ASSERT(!nextContext.pseudoElementEffective && !ignoreDynamicPseudo);
350
351 if (result.match == Match::SelectorMatches || result.match == Match::SelectorFailsCompletely)
352 return MatchResult::updateWithMatchType(result, matchType);
353 return MatchResult::fails(Match::SelectorFailsAllSiblings);
354 }
355
356 case CSSSelector::DirectAdjacent:
357 {
358 auto relation = context.isMatchElement ? Style::Relation::AffectedByPreviousSibling : Style::Relation::DescendantsAffectedByPreviousSibling;
359 addStyleRelation(checkingContext, *context.element, relation);
360
361 Element* previousElement = context.element->previousElementSibling();
362 if (!previousElement)
363 return MatchResult::fails(Match::SelectorFailsAllSiblings);
364
365 addStyleRelation(checkingContext, *previousElement, Style::Relation::AffectsNextSibling);
366
367 nextContext.element = previousElement;
368 nextContext.firstSelectorOfTheFragment = nextContext.selector;
369 PseudoIdSet ignoreDynamicPseudo;
370 MatchResult result = matchRecursively(checkingContext, nextContext, ignoreDynamicPseudo);
371 ASSERT(!nextContext.pseudoElementEffective && !ignoreDynamicPseudo);
372
373 return MatchResult::updateWithMatchType(result, matchType);
374 }
375 case CSSSelector::IndirectAdjacent: {
376 auto relation = context.isMatchElement ? Style::Relation::AffectedByPreviousSibling : Style::Relation::DescendantsAffectedByPreviousSibling;
377 addStyleRelation(checkingContext, *context.element, relation);
378
379 nextContext.element = context.element->previousElementSibling();
380 nextContext.firstSelectorOfTheFragment = nextContext.selector;
381 for (; nextContext.element; nextContext.element = nextContext.element->previousElementSibling()) {
382 addStyleRelation(checkingContext, *nextContext.element, Style::Relation::AffectsNextSibling);
383
384 PseudoIdSet ignoreDynamicPseudo;
385 MatchResult result = matchRecursively(checkingContext, nextContext, ignoreDynamicPseudo);
386 ASSERT(!nextContext.pseudoElementEffective && !ignoreDynamicPseudo);
387
388 if (result.match == Match::SelectorMatches || result.match == Match::SelectorFailsAllSiblings || result.match == Match::SelectorFailsCompletely)
389 return MatchResult::updateWithMatchType(result, matchType);
390 };
391 return MatchResult::fails(Match::SelectorFailsAllSiblings);
392 }
393 case CSSSelector::Subselector:
394 {
395 // a selector is invalid if something follows a pseudo-element
396 // We make an exception for scrollbar pseudo elements and allow a set of pseudo classes (but nothing else)
397 // to follow the pseudo elements.
398 nextContext.hasScrollbarPseudo = hasScrollbarPseudoElement(dynamicPseudoIdSet);
399 nextContext.hasSelectionPseudo = dynamicPseudoIdSet.has(PseudoId::Selection);
400 if ((context.isMatchElement || checkingContext.resolvingMode == Mode::CollectingRules) && dynamicPseudoIdSet
401 && !nextContext.hasSelectionPseudo
402 && !(nextContext.hasScrollbarPseudo && nextContext.selector->match() == CSSSelector::PseudoClass))
403 return MatchResult::fails(Match::SelectorFailsCompletely);
404
405 MatchResult result = matchRecursively(checkingContext, nextContext, dynamicPseudoIdSet);
406
407 return MatchResult::updateWithMatchType(result, matchType);
408 }
409 case CSSSelector::ShadowDescendant:
410 {
411 // When matching foo::part(bar) we skip directly to the tree of element 'foo'.
412 bool isPart = context.selector->match() == CSSSelector::PseudoElement && context.selector->pseudoElementType() == CSSSelector::PseudoElementPart;
413 auto* shadowHost = isPart ? checkingContext.shadowHostInPartRuleScope : context.element->shadowHost();
414 if (!shadowHost)
415 return MatchResult::fails(Match::SelectorFailsCompletely);
416 nextContext.element = shadowHost;
417 nextContext.firstSelectorOfTheFragment = nextContext.selector;
418 nextContext.isSubjectOrAdjacentElement = false;
419 PseudoIdSet ignoreDynamicPseudo;
420 MatchResult result = matchRecursively(checkingContext, nextContext, ignoreDynamicPseudo);
421
422 return MatchResult::updateWithMatchType(result, matchType);
423 }
424 }
425
426
427 ASSERT_NOT_REACHED();
428 return MatchResult::fails(Match::SelectorFailsCompletely);
429}

431static bool attributeValueMatches(const Attribute& attribute, CSSSelector::Match match, const AtomString& selectorValue, bool caseSensitive)
432{
433 const AtomString& value = attribute.value();
434 ASSERT(!value.isNull());
435
436 switch (match) {
437 case CSSSelector::Set:
438 break;
439 case CSSSelector::Exact:
440 if (caseSensitive ? selectorValue != value : !equalIgnoringASCIICase(selectorValue, value))
441 return false;
442 break;
443 case CSSSelector::List:
444 {
445 // Ignore empty selectors or selectors containing spaces.
446 if (selectorValue.isEmpty() || selectorValue.find(isHTMLSpace<UChar>) != notFound)
447 return false;
448
449 unsigned startSearchAt = 0;
450 while (true) {
451 size_t foundPos;
452 if (caseSensitive)
453 foundPos = value.find(selectorValue, startSearchAt);
454 else
455 foundPos = value.findIgnoringASCIICase(selectorValue, startSearchAt);
456 if (foundPos == notFound)
457 return false;
458 if (!foundPos || isHTMLSpace(value[foundPos - 1])) {
459 unsigned endStr = foundPos + selectorValue.length();
460 if (endStr == value.length() || isHTMLSpace(value[endStr]))
461 break; // We found a match.
462 }
463
464 // No match. Keep looking.
465 startSearchAt = foundPos + 1;
466 }
467 break;
468 }
469 case CSSSelector::Contain: {
470 bool valueContainsSelectorValue;
471 if (caseSensitive)
472 valueContainsSelectorValue = value.contains(selectorValue);
473 else
474 valueContainsSelectorValue = value.containsIgnoringASCIICase(selectorValue);
475
476 if (!valueContainsSelectorValue || selectorValue.isEmpty())
477 return false;
478
479 break;
480 }
481 case CSSSelector::Begin:
482 if (selectorValue.isEmpty())
483 return false;
484 if (caseSensitive) {
485 if (!value.startsWith(selectorValue))
486 return false;
487 } else {
488 if (!value.startsWithIgnoringASCIICase(selectorValue))
489 return false;
490 }
491 break;
492 case CSSSelector::End:
493 if (selectorValue.isEmpty())
494 return false;
495 if (caseSensitive) {
496 if (!value.endsWith(selectorValue))
497 return false;
498 } else {
499 if (!value.endsWithIgnoringASCIICase(selectorValue))
500 return false;
501 }
502 break;
503 case CSSSelector::Hyphen:
504 if (value.length() < selectorValue.length())
505 return false;
506 if (caseSensitive) {
507 if (!value.startsWith(selectorValue))
508 return false;
509 } else {
510 if (!value.startsWithIgnoringASCIICase(selectorValue))
511 return false;
512 }
513 // It they start the same, check for exact match or following '-':
514 if (value.length() != selectorValue.length() && value[selectorValue.length()] != '-')
515 return false;
516 break;
517 default:
518 ASSERT_NOT_REACHED();
519 return false;
520 }
521
522 return true;
523}

525static bool anyAttributeMatches(const Element& element, const CSSSelector& selector, const QualifiedName& selectorAttr, bool caseSensitive)
526{
527 ASSERT(element.hasAttributesWithoutUpdate());
528 for (const Attribute& attribute : element.attributesIterator()) {
529 if (!attribute.matches(selectorAttr.prefix(), element.isHTMLElement() ? selector.attributeCanonicalLocalName() : selectorAttr.localName(), selectorAttr.namespaceURI()))
530 continue;
531
532 if (attributeValueMatches(attribute, selector.match(), selector.value(), caseSensitive))
533 return true;
534 }
535
536 return false;
537}

539bool SelectorChecker::attributeSelectorMatches(const Element& element, const QualifiedName& attributeName, const AtomString& attributeValue, const CSSSelector& selector)
540{
541 ASSERT(selector.isAttributeSelector());
542 auto& selectorAttribute = selector.attribute();
543 auto& selectorName = element.isHTMLElement() ? selector.attributeCanonicalLocalName() : selectorAttribute.localName();
544 if (!Attribute::nameMatchesFilter(attributeName, selectorAttribute.prefix(), selectorName, selectorAttribute.namespaceURI()))
545 return false;
546 bool caseSensitive = true;
547 if (selector.attributeValueMatchingIsCaseInsensitive())
548 caseSensitive = false;
549 else if (element.document().isHTMLDocument() && element.isHTMLElement() && !HTMLDocument::isCaseSensitiveAttribute(selector.attribute()))
550 caseSensitive = false;
551 return attributeValueMatches(Attribute(attributeName, attributeValue), selector.match(), selector.value(), caseSensitive);
552}

554static bool canMatchHoverOrActiveInQuirksMode(const SelectorChecker::LocalContext& context)
555{
556 // For quirks mode, follow this: http://quirks.spec.whatwg.org/#the-:active-and-:hover-quirk
557 // In quirks mode, a compound selector 'selector' that matches the following conditions must not match elements that would not also match the ':any-link' selector.
558 //
559 // selector uses the ':active' or ':hover' pseudo-classes.
560 // selector does not use a type selector.
561 // selector does not use an attribute selector.
562 // selector does not use an ID selector.
563 // selector does not use a class selector.
564 // selector does not use a pseudo-class selector other than ':active' and ':hover'.
565 // selector does not use a pseudo-element selector.
566 // selector is not part of an argument to a functional pseudo-class or pseudo-element.
567 if (context.inFunctionalPseudoClass)
568 return true;
569
570 for (const CSSSelector* selector = context.firstSelectorOfTheFragment; selector; selector = selector->tagHistory()) {
571 switch (selector->match()) {
572 case CSSSelector::Tag:
573 if (selector->tagQName() != anyQName())
574 return true;
575 break;
576 case CSSSelector::PseudoClass: {
577 CSSSelector::PseudoClassType pseudoClassType = selector->pseudoClassType();
578 if (pseudoClassType != CSSSelector::PseudoClassHover && pseudoClassType != CSSSelector::PseudoClassActive)
579 return true;
580 break;
581 }
582 case CSSSelector::Id:
583 case CSSSelector::Class:
584 case CSSSelector::Exact:
585 case CSSSelector::Set:
586 case CSSSelector::List:
587 case CSSSelector::Hyphen:
588 case CSSSelector::Contain:
589 case CSSSelector::Begin:
590 case CSSSelector::End:
591 case CSSSelector::PagePseudoClass:
592 case CSSSelector::PseudoElement:
593 return true;
594 case CSSSelector::Unknown:
595 ASSERT_NOT_REACHED();
596 break;
597 }
598
599 auto relation = selector->relation();
600 if (relation == CSSSelector::ShadowDescendant)
601 return true;
602
603 if (relation != CSSSelector::Subselector)
604 return false;
605 }
606 return false;
607}

609static inline bool tagMatches(const Element& element, const CSSSelector& simpleSelector)
610{
611 const QualifiedName& tagQName = simpleSelector.tagQName();
612
613 if (tagQName == anyQName())
614 return true;
615
616 const AtomString& localName = (element.isHTMLElement() && element.document().isHTMLDocument()) ? simpleSelector.tagLowercaseLocalName() : tagQName.localName();
617
618 if (localName != starAtom() && localName != element.localName())
619 return false;
620 const AtomString& namespaceURI = tagQName.namespaceURI();
621 return namespaceURI == starAtom() || namespaceURI == element.namespaceURI();
622}

624bool SelectorChecker::checkOne(CheckingContext& checkingContext, const LocalContext& context, MatchType& matchType) const
625{
626 const Element& element = *context.element;
627 const CSSSelector& selector = *context.selector;
628
629 if (context.mayMatchHostPseudoClass) {
630 // :host doesn't combine with anything except pseudo elements.
631 bool isHostPseudoClass = selector.match() == CSSSelector::PseudoClass && selector.pseudoClassType() == CSSSelector::PseudoClassHost;
632 bool isPseudoElement = selector.match() == CSSSelector::PseudoElement;
633 if (!isHostPseudoClass && !isPseudoElement)
634 return false;
635 }
636
637 if (selector.match() == CSSSelector::Tag)
638 return tagMatches(element, selector);
639
640 if (selector.match() == CSSSelector::Class)
641 return element.hasClass() && element.classNames().contains(selector.value());
642
643 if (selector.match() == CSSSelector::Id) {
644 ASSERT(!selector.value().isNull());
645 return element.idForStyleResolution() == selector.value();
646 }
647
648 if (selector.isAttributeSelector()) {
649 if (!element.hasAttributes())
650 return false;
651
652 const QualifiedName& attr = selector.attribute();
653 bool caseSensitive = true;

657 caseSensitive = false;

660 }
661
662 if (selector.match() == CSSSelector::PseudoClass) {

664 if (selector.pseudoClassType() == CSSSelector::PseudoClassNot) {
665 const CSSSelectorList* selectorList = selector.selectorList();
666
667 for (const CSSSelector* subselector = selectorList->first(); subselector; subselector = CSSSelectorList::next(subselector)) {
668 LocalContext subcontext(context);
669 subcontext.inFunctionalPseudoClass = true;
670 subcontext.pseudoElementEffective = false;
671 subcontext.selector = subselector;
672 subcontext.firstSelectorOfTheFragment = selectorList->first();
673 PseudoIdSet ignoreDynamicPseudo;
674
675 if (matchRecursively(checkingContext, subcontext, ignoreDynamicPseudo).match == Match::SelectorMatches) {
676 ASSERT(!ignoreDynamicPseudo);
677 return false;
678 }
679 }
680 return true;
681 }
682 if (context.hasScrollbarPseudo) {
683 // CSS scrollbars match a specific subset of pseudo classes, and they have specialized rules for each
684 // (since there are no elements involved except with window-inactive).
685 return checkScrollbarPseudoClass(checkingContext, element, selector);

699 break;

703 if (!textNode.data().isEmpty()) {
704 result = false;
705 break;
706 }
707 }
708 }
709 addStyleRelation(checkingContext, *context.element, Style::Relation::AffectedByEmpty, result);
710
711 return result;
712 }
713 case CSSSelector::PseudoClassFirstChild: {
714 // first-child matches the first child that is an element
715 bool isFirstChild = isFirstChildElement(element);
716 auto* parent = element.parentNode();

718 addStyleRelation(checkingContext, downcast<Element>(*parent), Style::Relation::ChildrenAffectedByFirstChildRules);
719 else if (!is<ShadowRoot>(parent))
720 break; // FIXME: Add the support for specifying relations on ShadowRoot.
721 if (!isFirstChild)
722 break;

728 auto* parent = element.parentNode();

742 if (!parentElement.isFinishedParsingChildren())

745 } else if (!is<ShadowRoot>(parent))

747 if (!isLastChild)

751 }

951 // Inside functional pseudo class except for :not, :visited never matches.

1223unsigned SelectorChecker::determineLinkMatchType(const CSSSelector* selector)
1224{
1225 unsigned linkMatchType = MatchAll;
1226
1227 // Statically determine if this selector will match a link in visited, unvisited or any state, or never.
1228 // :visited never matches other elements than the innermost link element.
1229 for (; selector; selector = selector->tagHistory()) {
1230 if (selector->match() == CSSSelector::PseudoClass) {
1231 switch (selector->pseudoClassType()) {
1232 case CSSSelector::PseudoClassLink:
1233 linkMatchType &= ~SelectorChecker::MatchVisited;
1234 break;
1235 case CSSSelector::PseudoClassVisited:
1236 linkMatchType &= ~SelectorChecker::MatchLink;
1237 break;
1238 default:
1239 break;
1240 }
1241 }
1242 auto relation = selector->relation();
1243 if (relation == CSSSelector::Subselector)
1244 continue;
1245 if (!selector->hasDescendantOrChildRelation())
1246 return linkMatchType;
1247 if (linkMatchType != MatchAll)
1248 return linkMatchType;
1249 }
1250 return linkMatchType;
1251}
