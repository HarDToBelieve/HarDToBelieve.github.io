40Color StyleColor::colorFromKeyword(CSSValueID keyword, OptionSet<Options> options)
41{
42 if (const char* valueName = getValueName(keyword)) {
43 if (const NamedColor* namedColor = findColor(valueName, strlen(valueName)))
44 return asSRGBA(PackedColor::ARGB { namedColor->ARGBValue });
45 }
46
47 return RenderTheme::singleton().systemColor(keyword, options);
48}
49
50bool StyleColor::isColorKeyword(CSSValueID id)
51{
52 return (id >= CSSValueAlpha && id <= CSSValueWebkitText) || (id >= CSSValueAliceblue && id <= CSSValueYellowgreen) || isSystemColor(id);
53}
54
55bool StyleColor::isSystemColor(CSSValueID id)
56{
57 return (id >= CSSValueWebkitLink && id <= CSSValueWebkitFocusRingColor) || id == CSSValueMenu || id == CSSValueText;
58}
