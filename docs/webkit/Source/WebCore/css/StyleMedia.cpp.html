47String StyleMedia::type() const
48{
49 auto* frame = this->frame();
50 FrameView* view = frame ? frame->view() : nullptr;
51 if (view)
52 return view->mediaType();
53
54 return String();
55}
56
57bool StyleMedia::matchMedium(const String& query) const
58{
59 auto* frame = this->frame();
60 if (!frame)
61 return false;
62
63 Document* document = frame->document();
64 ASSERT(document);
65 Element* documentElement = document->documentElement();
66 if (!documentElement)
67 return false;
68
69 auto rootStyle = document->styleScope().resolver().styleForElement(*documentElement, document->renderStyle(), nullptr, RuleMatchingBehavior::MatchOnlyUserAgentRules).renderStyle;
70
71 auto media = MediaQuerySet::create(query, MediaQueryParserContext(*document));
72
73 return MediaQueryEvaluator { type(), *document, rootStyle.get() }.evaluate(media.get());
74}
