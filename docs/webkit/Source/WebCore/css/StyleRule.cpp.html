133Ref<CSSRule> StyleRuleBase::createCSSOMWrapper(CSSStyleSheet* parentSheet, CSSRule* parentRule) const
134{
135 RefPtr<CSSRule> rule;
136 StyleRuleBase& self = const_cast<StyleRuleBase&>(*this);
137 switch (type()) {
138 case StyleRuleType::Style:
139 rule = CSSStyleRule::create(downcast<StyleRule>(self), parentSheet);
140 break;
141 case StyleRuleType::Page:
142 rule = CSSPageRule::create(downcast<StyleRulePage>(self), parentSheet);
143 break;
144 case StyleRuleType::FontFace:
145 rule = CSSFontFaceRule::create(downcast<StyleRuleFontFace>(self), parentSheet);
146 break;
147 case StyleRuleType::Media:
148 rule = CSSMediaRule::create(downcast<StyleRuleMedia>(self), parentSheet);
149 break;
150 case StyleRuleType::Supports:
151 rule = CSSSupportsRule::create(downcast<StyleRuleSupports>(self), parentSheet);
152 break;
153 case StyleRuleType::Import:
154 rule = CSSImportRule::create(downcast<StyleRuleImport>(self), parentSheet);
155 break;
156 case StyleRuleType::Keyframes:
157 rule = CSSKeyframesRule::create(downcast<StyleRuleKeyframes>(self), parentSheet);
158 break;
159 case StyleRuleType::Namespace:
160 rule = CSSNamespaceRule::create(downcast<StyleRuleNamespace>(self), parentSheet);
161 break;
162 case StyleRuleType::CounterStyle:
163 rule = CSSCounterStyleRule::create(downcast<StyleRuleCounterStyle>(self), parentSheet);
164 break;
165 case StyleRuleType::Unknown:
166 case StyleRuleType::Charset:
167 case StyleRuleType::Keyframe:
168 ASSERT_NOT_REACHED();
169 break;
170 }
171 ASSERT(rule);
172
173 if (parentRule)
174 rule->setParentRule(parentRule);
175 return rule.releaseNonNull();
176}

223Ref<StyleRule> StyleRule::createForSplitting(const Vector<const CSSSelector*>& selectors, Ref<StyleProperties>&& properties, bool hasDocumentSecurityOrigin)
224{
225 ASSERT_WITH_SECURITY_IMPLICATION(!selectors.isEmpty());
226 auto selectorListArray = makeUniqueArray<CSSSelector>(selectors.size());
227 for (unsigned i = 0; i < selectors.size(); ++i)
228 new (NotNull, &selectorListArray[i]) CSSSelector(*selectors.at(i));
229 selectorListArray[selectors.size() - 1].setLastInSelectorList();
230 return StyleRule::create(WTFMove(properties), hasDocumentSecurityOrigin, CSSSelectorList(WTFMove(selectorListArray)));
231}

233Vector<RefPtr<StyleRule>> StyleRule::splitIntoMultipleRulesWithMaximumSelectorComponentCount(unsigned maxCount) const
234{
235 ASSERT(selectorList().componentCount() > maxCount);
236
237 Vector<RefPtr<StyleRule>> rules;
238 Vector<const CSSSelector*> componentsSinceLastSplit;
239
240 for (const CSSSelector* selector = selectorList().first(); selector; selector = CSSSelectorList::next(selector)) {
241 Vector<const CSSSelector*, 8> componentsInThisSelector;
242 for (const CSSSelector* component = selector; component; component = component->tagHistory())
243 componentsInThisSelector.append(component);
244
245 if (componentsInThisSelector.size() + componentsSinceLastSplit.size() > maxCount && !componentsSinceLastSplit.isEmpty()) {
246 rules.append(createForSplitting(componentsSinceLastSplit, const_cast<StyleProperties&>(properties()), hasDocumentSecurityOrigin()));
247 componentsSinceLastSplit.clear();
248 }
249
250 componentsSinceLastSplit.appendVector(componentsInThisSelector);
251 }
252
253 if (!componentsSinceLastSplit.isEmpty())
254 rules.append(createForSplitting(componentsSinceLastSplit, const_cast<StyleProperties&>(properties()), hasDocumentSecurityOrigin()));
255
256 return rules;
257}
