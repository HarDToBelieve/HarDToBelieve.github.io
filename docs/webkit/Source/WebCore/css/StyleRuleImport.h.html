23
24#include "CachedResourceHandle.h"
25#include "CachedStyleSheetClient.h"
26#include "StyleRule.h"
27#include <wtf/TypeCasts.h>
28
29namespace WebCore {
30
31class CachedCSSStyleSheet;
32class MediaQuerySet;
33class StyleSheetContents;
34
35class StyleRuleImport final : public StyleRuleBase {
36 WTF_MAKE_FAST_ALLOCATED;
37public:
38 static Ref<StyleRuleImport> create(const String& href, Ref<MediaQuerySet>&&);
39
40 ~StyleRuleImport();
41
42 StyleSheetContents* parentStyleSheet() const { return m_parentStyleSheet; }
43 void setParentStyleSheet(StyleSheetContents* sheet) { ASSERT(sheet); m_parentStyleSheet = sheet; }
44 void clearParentStyleSheet() { m_parentStyleSheet = 0; }
45
46 String href() const { return m_strHref; }
47 StyleSheetContents* styleSheet() const { return m_styleSheet.get(); }
48
49 bool isLoading() const;
50 MediaQuerySet* mediaQueries() const { return m_mediaQueries.get(); }
51
52 void requestStyleSheet();
53 const CachedCSSStyleSheet* cachedCSSStyleSheet() const { return m_cachedSheet.get(); }
54
55private:
56 // NOTE: We put the CachedStyleSheetClient in a member instead of inheriting from it
57 // to avoid adding a vptr to StyleRuleImport.
58class ImportedStyleSheetClient final : public CachedStyleSheetClient {
59 public:
60 ImportedStyleSheetClient(StyleRuleImport* ownerRule) : m_ownerRule(ownerRule) { }
61 virtual ~ImportedStyleSheetClient() = default;
62 void setCSSStyleSheet(const String& href, const URL& baseURL, const String& charset, const CachedCSSStyleSheet* sheet) final
63 {
64 m_ownerRule->setCSSStyleSheet(href, baseURL, charset, sheet);
65 }
66 private:
67 StyleRuleImport* m_ownerRule;
68 };
69
70 void setCSSStyleSheet(const String& href, const URL& baseURL, const String& charset, const CachedCSSStyleSheet*);
71 friend class ImportedStyleSheetClient;
72
73 StyleRuleImport(const String& href, Ref<MediaQuerySet>&&);
74
75 StyleSheetContents* m_parentStyleSheet { nullptr };
76
77 ImportedStyleSheetClient m_styleSheetClient;
78 String m_strHref;
79 RefPtr<MediaQuerySet> m_mediaQueries;
80 RefPtr<StyleSheetContents> m_styleSheet;
81 CachedResourceHandle<CachedCSSStyleSheet> m_cachedSheet;
82 bool m_loading { false };
83};
84
85} // namespace WebCore
86
87SPECIALIZE_TYPE_TRAITS_BEGIN(WebCore::StyleRuleImport)
