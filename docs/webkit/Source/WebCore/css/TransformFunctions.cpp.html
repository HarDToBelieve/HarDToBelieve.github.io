49static TransformOperation::OperationType transformOperationType(CSSValueID type)
50{
51 switch (type) {
52 case CSSValueScale:
53 return TransformOperation::SCALE;
54 case CSSValueScaleX:
55 return TransformOperation::SCALE_X;
56 case CSSValueScaleY:
57 return TransformOperation::SCALE_Y;
58 case CSSValueScaleZ:
59 return TransformOperation::SCALE_Z;
60 case CSSValueScale3d:
61 return TransformOperation::SCALE_3D;
62 case CSSValueTranslate:
63 return TransformOperation::TRANSLATE;
64 case CSSValueTranslateX:
65 return TransformOperation::TRANSLATE_X;
66 case CSSValueTranslateY:
67 return TransformOperation::TRANSLATE_Y;
68 case CSSValueTranslateZ:
69 return TransformOperation::TRANSLATE_Z;
70 case CSSValueTranslate3d:
71 return TransformOperation::TRANSLATE_3D;
72 case CSSValueRotate:
73 return TransformOperation::ROTATE;
74 case CSSValueRotateX:
75 return TransformOperation::ROTATE_X;
76 case CSSValueRotateY:
77 return TransformOperation::ROTATE_Y;
78 case CSSValueRotateZ:
79 return TransformOperation::ROTATE_Z;
80 case CSSValueRotate3d:
81 return TransformOperation::ROTATE_3D;
82 case CSSValueSkew:
83 return TransformOperation::SKEW;
84 case CSSValueSkewX:
85 return TransformOperation::SKEW_X;
86 case CSSValueSkewY:
87 return TransformOperation::SKEW_Y;
88 case CSSValueMatrix:
89 return TransformOperation::MATRIX;
90 case CSSValueMatrix3d:
91 return TransformOperation::MATRIX_3D;
92 case CSSValuePerspective:
93 return TransformOperation::PERSPECTIVE;
94 default:
95 break;
96 }
97 return TransformOperation::NONE;
98}

100Length convertToFloatLength(const CSSPrimitiveValue* primitiveValue, const CSSToLengthConversionData& conversionData)
101{
102 return primitiveValue ? primitiveValue->convertToLength<FixedFloatConversion | PercentConversion | CalculatedConversion>(conversionData) : Length(LengthType::Undefined);
103}

105bool transformsForValue(const CSSValue& value, const CSSToLengthConversionData& conversionData, TransformOperations& outOperations)
106{
107 ASSERT(!outOperations.size());
108 if (!is<CSSValueList>(value))
109 return false;
110
111 auto& operations = outOperations.operations();
112 for (auto& currentValue : downcast<CSSValueList>(value)) {
113 if (!is<CSSFunctionValue>(currentValue))
114 continue;
115
116 auto& transformValue = downcast<CSSFunctionValue>(currentValue.get());
117 if (!transformValue.length())
118 continue;
119
120 bool haveNonPrimitiveValue = false;
121 for (unsigned j = 0; j < transformValue.length(); ++j) {
122 if (!is<CSSPrimitiveValue>(*transformValue.itemWithoutBoundsCheck(j))) {
123 haveNonPrimitiveValue = true;
124 break;
125 }
126 }
127 if (haveNonPrimitiveValue)
128 continue;
129
130 auto& firstValue = downcast<CSSPrimitiveValue>(*transformValue.itemWithoutBoundsCheck(0));
131
132 switch (transformValue.name()) {
133 case CSSValueScale:
134 case CSSValueScaleX:
135 case CSSValueScaleY: {
136 double sx = 1.0;
137 double sy = 1.0;
138 if (transformValue.name() == CSSValueScaleY)
139 sy = firstValue.doubleValue();
140 else {
141 sx = firstValue.doubleValue();
142 if (transformValue.name() != CSSValueScaleX) {
143 if (transformValue.length() > 1) {
144 auto& secondValue = downcast<CSSPrimitiveValue>(*transformValue.itemWithoutBoundsCheck(1));
145 sy = secondValue.doubleValue();
146 } else
147 sy = sx;
148 }
149 }
150 operations.append(ScaleTransformOperation::create(sx, sy, 1.0, transformOperationType(transformValue.name())));
151 break;
152 }
153 case CSSValueScaleZ:
154 case CSSValueScale3d: {
155 double sx = 1.0;
156 double sy = 1.0;
157 double sz = 1.0;
158 if (transformValue.name() == CSSValueScaleZ)
159 sz = firstValue.doubleValue();
160 else if (transformValue.name() == CSSValueScaleY)
161 sy = firstValue.doubleValue();
162 else {
163 sx = firstValue.doubleValue();
164 if (transformValue.name() != CSSValueScaleX) {
165 if (transformValue.length() > 2) {
166 auto& thirdValue = downcast<CSSPrimitiveValue>(*transformValue.itemWithoutBoundsCheck(2));
167 sz = thirdValue.doubleValue();
168 }
169 if (transformValue.length() > 1) {
170 auto& secondValue = downcast<CSSPrimitiveValue>(*transformValue.itemWithoutBoundsCheck(1));
171 sy = secondValue.doubleValue();
172 } else
173 sy = sx;
174 }
175 }
176 operations.append(ScaleTransformOperation::create(sx, sy, sz, transformOperationType(transformValue.name())));
177 break;
178 }
179 case CSSValueTranslate:
180 case CSSValueTranslateX:
181 case CSSValueTranslateY: {
182 Length tx = Length(0, LengthType::Fixed);
183 Length ty = Length(0, LengthType::Fixed);
184 if (transformValue.name() == CSSValueTranslateY)
185 ty = convertToFloatLength(&firstValue, conversionData);
186 else {
187 tx = convertToFloatLength(&firstValue, conversionData);
188 if (transformValue.name() != CSSValueTranslateX) {
189 if (transformValue.length() > 1) {
190 auto& secondValue = downcast<CSSPrimitiveValue>(*transformValue.itemWithoutBoundsCheck(1));
191 ty = convertToFloatLength(&secondValue, conversionData);
192 }
193 }
194 }
195
196 if (tx.isUndefined() || ty.isUndefined()) {
197 operations.clear();
198 return false;
199 }
200
201 operations.append(TranslateTransformOperation::create(tx, ty, Length(0, LengthType::Fixed), transformOperationType(transformValue.name())));
202 break;
203 }
204 case CSSValueTranslateZ:
205 case CSSValueTranslate3d: {
206 Length tx = Length(0, LengthType::Fixed);
207 Length ty = Length(0, LengthType::Fixed);
208 Length tz = Length(0, LengthType::Fixed);
209 if (transformValue.name() == CSSValueTranslateZ)
210 tz = convertToFloatLength(&firstValue, conversionData);
211 else if (transformValue.name() == CSSValueTranslateY)
212 ty = convertToFloatLength(&firstValue, conversionData);
213 else {
214 tx = convertToFloatLength(&firstValue, conversionData);
215 if (transformValue.name() != CSSValueTranslateX) {
216 if (transformValue.length() > 2) {
217 auto& thirdValue = downcast<CSSPrimitiveValue>(*transformValue.itemWithoutBoundsCheck(2));
218 tz = convertToFloatLength(&thirdValue, conversionData);
219 }
220 if (transformValue.length() > 1) {
221 auto& secondValue = downcast<CSSPrimitiveValue>(*transformValue.itemWithoutBoundsCheck(1));
222 ty = convertToFloatLength(&secondValue, conversionData);
223 }
224 }
225 }
226
227 if (tx.isUndefined() || ty.isUndefined() || tz.isUndefined()) {
228 operations.clear();
229 return false;
230 }
231
232 operations.append(TranslateTransformOperation::create(tx, ty, tz, transformOperationType(transformValue.name())));
233 break;
234 }
235 case CSSValueRotate: {
236 double angle = firstValue.computeDegrees();
237 operations.append(RotateTransformOperation::create(0, 0, 1, angle, transformOperationType(transformValue.name())));
238 break;
239 }
240 case CSSValueRotateX:
241 case CSSValueRotateY:
242 case CSSValueRotateZ: {
243 double x = 0;
244 double y = 0;
245 double z = 0;
246 double angle = firstValue.computeDegrees();
247
248 if (transformValue.name() == CSSValueRotateX)
249 x = 1;
250 else if (transformValue.name() == CSSValueRotateY)
251 y = 1;
252 else
253 z = 1;
254 operations.append(RotateTransformOperation::create(x, y, z, angle, transformOperationType(transformValue.name())));
255 break;
256 }
257 case CSSValueRotate3d: {
258 if (transformValue.length() < 4)
259 break;
260 auto& secondValue = downcast<CSSPrimitiveValue>(*transformValue.itemWithoutBoundsCheck(1));
261 auto& thirdValue = downcast<CSSPrimitiveValue>(*transformValue.itemWithoutBoundsCheck(2));
262 auto& fourthValue = downcast<CSSPrimitiveValue>(*transformValue.itemWithoutBoundsCheck(3));
263 double x = firstValue.doubleValue();
264 double y = secondValue.doubleValue();
265 double z = thirdValue.doubleValue();
266 double angle = fourthValue.computeDegrees();
267 operations.append(RotateTransformOperation::create(x, y, z, angle, transformOperationType(transformValue.name())));
268 break;
269 }
270 case CSSValueSkew:
271 case CSSValueSkewX:
272 case CSSValueSkewY: {
273 double angleX = 0;
274 double angleY = 0;
275 double angle = firstValue.computeDegrees();
276 if (transformValue.name() == CSSValueSkewY)
277 angleY = angle;
278 else {
279 angleX = angle;
280 if (transformValue.name() == CSSValueSkew) {
281 if (transformValue.length() > 1) {
282 auto& secondValue = downcast<CSSPrimitiveValue>(*transformValue.itemWithoutBoundsCheck(1));
283 angleY = secondValue.computeDegrees();
284 }
285 }
286 }
287 operations.append(SkewTransformOperation::create(angleX, angleY, transformOperationType(transformValue.name())));
288 break;
289 }
290 case CSSValueMatrix: {
291 if (transformValue.length() < 6)
292 break;
293 double a = firstValue.doubleValue();
294 double b = downcast<CSSPrimitiveValue>(*transformValue.itemWithoutBoundsCheck(1)).doubleValue();
295 double c = downcast<CSSPrimitiveValue>(*transformValue.itemWithoutBoundsCheck(2)).doubleValue();
296 double d = downcast<CSSPrimitiveValue>(*transformValue.itemWithoutBoundsCheck(3)).doubleValue();
297 double e = conversionData.zoom() * downcast<CSSPrimitiveValue>(*transformValue.itemWithoutBoundsCheck(4)).doubleValue();
298 double f = conversionData.zoom() * downcast<CSSPrimitiveValue>(*transformValue.itemWithoutBoundsCheck(5)).doubleValue();
299 operations.append(MatrixTransformOperation::create(a, b, c, d, e, f));
300 break;
301 }
302 case CSSValueMatrix3d: {
303 if (transformValue.length() < 16)
304 break;
305 TransformationMatrix matrix(downcast<CSSPrimitiveValue>(*transformValue.itemWithoutBoundsCheck(0)).doubleValue(),
306 downcast<CSSPrimitiveValue>(*transformValue.itemWithoutBoundsCheck(1)).doubleValue(),
307 downcast<CSSPrimitiveValue>(*transformValue.itemWithoutBoundsCheck(2)).doubleValue(),
308 downcast<CSSPrimitiveValue>(*transformValue.itemWithoutBoundsCheck(3)).doubleValue(),
309 downcast<CSSPrimitiveValue>(*transformValue.itemWithoutBoundsCheck(4)).doubleValue(),
310 downcast<CSSPrimitiveValue>(*transformValue.itemWithoutBoundsCheck(5)).doubleValue(),
311 downcast<CSSPrimitiveValue>(*transformValue.itemWithoutBoundsCheck(6)).doubleValue(),
312 downcast<CSSPrimitiveValue>(*transformValue.itemWithoutBoundsCheck(7)).doubleValue(),
313 downcast<CSSPrimitiveValue>(*transformValue.itemWithoutBoundsCheck(8)).doubleValue(),
314 downcast<CSSPrimitiveValue>(*transformValue.itemWithoutBoundsCheck(9)).doubleValue(),
315 downcast<CSSPrimitiveValue>(*transformValue.itemWithoutBoundsCheck(10)).doubleValue(),
316 downcast<CSSPrimitiveValue>(*transformValue.itemWithoutBoundsCheck(11)).doubleValue(),
317 conversionData.zoom() * downcast<CSSPrimitiveValue>(*transformValue.itemWithoutBoundsCheck(12)).doubleValue(),
318 conversionData.zoom() * downcast<CSSPrimitiveValue>(*transformValue.itemWithoutBoundsCheck(13)).doubleValue(),
319 downcast<CSSPrimitiveValue>(*transformValue.itemWithoutBoundsCheck(14)).doubleValue(),
320 downcast<CSSPrimitiveValue>(*transformValue.itemWithoutBoundsCheck(15)).doubleValue());
321 operations.append(Matrix3DTransformOperation::create(matrix));
322 break;
323 }
324 case CSSValuePerspective: {
325 Length p = Length(0, LengthType::Fixed);
326 if (firstValue.isLength())
327 p = convertToFloatLength(&firstValue, conversionData);
328 else {
329 // This is a quirk that should go away when 3d transforms are finalized.
330 double val = firstValue.doubleValue();
331 p = val >= 0 ? Length(clampToPositiveInteger(val), LengthType::Fixed) : Length(LengthType::Undefined);
332 }
333
334 if (p.isUndefined()) {
335 operations.clear();
336 return false;
337 }
338
339 operations.append(PerspectiveTransformOperation::create(p));
340 break;
341 }
342 default:
343 ASSERT_NOT_REACHED();
344 break;
345 }
346 }
347
348 return true;
349}

351RefPtr<TranslateTransformOperation> translateForValue(const CSSValue& value, const CSSToLengthConversionData& conversionData)
352{
353 if (!is<CSSValueList>(value))
354 return nullptr;
355
356 auto& valueList = downcast<CSSValueList>(value);
357 if (!valueList.length())
358 return nullptr;
359
360 auto type = TransformOperation::TRANSLATE;
361 Length tx = Length(0, LengthType::Fixed);
362 Length ty = Length(0, LengthType::Fixed);
363 Length tz = Length(0, LengthType::Fixed);
364 for (unsigned i = 0; i < valueList.length(); ++i) {
365 auto* valueItem = valueList.itemWithoutBoundsCheck(i);
366 if (!is<CSSPrimitiveValue>(valueItem))
367 return nullptr;
368 if (!i)
369 tx = convertToFloatLength(downcast<CSSPrimitiveValue>(valueItem), conversionData);
370 else if (i == 1)
371 ty = convertToFloatLength(downcast<CSSPrimitiveValue>(valueItem), conversionData);
372 else if (i == 2) {
373 type = TransformOperation::TRANSLATE_3D;
374 tz = convertToFloatLength(downcast<CSSPrimitiveValue>(valueItem), conversionData);
375 }
376 }
377
378 return TranslateTransformOperation::create(tx, ty, tz, type);
379}

381RefPtr<ScaleTransformOperation> scaleForValue(const CSSValue& value)
382{
383 if (!is<CSSValueList>(value))
384 return nullptr;
385
386 auto& valueList = downcast<CSSValueList>(value);
387 if (!valueList.length())
388 return nullptr;
389
390 auto type = TransformOperation::SCALE;
391 double sx = 1.0;
392 double sy = 1.0;
393 double sz = 1.0;
394 for (unsigned i = 0; i < valueList.length(); ++i) {
395 auto* valueItem = valueList.itemWithoutBoundsCheck(i);
396 if (!is<CSSPrimitiveValue>(valueItem))
397 return nullptr;
398 if (!i) {
399 sx = downcast<CSSPrimitiveValue>(*valueItem).doubleValue();
400 sy = sx;
401 } else if (i == 1)
402 sy = downcast<CSSPrimitiveValue>(*valueItem).doubleValue();
403 else if (i == 2) {
404 type = TransformOperation::SCALE_3D;
405 sz = downcast<CSSPrimitiveValue>(*valueItem).doubleValue();
406 }
407 }
408
409 return ScaleTransformOperation::create(sx, sy, sz, type);
410}

412RefPtr<RotateTransformOperation> rotateForValue(const CSSValue& value)
413{
414 if (!is<CSSValueList>(value))
415 return nullptr;
416
417 auto& valueList = downcast<CSSValueList>(value);
418 auto numberOfItems = valueList.length();
419
420 // There are three scenarios here since the rotation axis is defined either as:
421 // - no value: implicit 2d rotation
422 // - 1 value: an axis identifier (x/y/z)
423 // - 3 values: three numbers defining an x/y/z vector
424 // The angle is specified as the last value.
425 if (numberOfItems != 1 && numberOfItems != 2 && numberOfItems != 4)
426 return nullptr;
427
428 auto* lastValue = valueList.itemWithoutBoundsCheck(numberOfItems - 1);
429 if (!is<CSSPrimitiveValue>(lastValue))
430 return nullptr;
431 auto angle = downcast<CSSPrimitiveValue>(*lastValue).computeDegrees();
432
433 if (numberOfItems == 1)
434 return RotateTransformOperation::create(angle, TransformOperation::ROTATE);
435
436 double x = 0.0;
437 double y = 0.0;
438 double z = 0.0;
439 auto type = TransformOperation::ROTATE;
440
441 if (numberOfItems == 2) {
442 // An axis identifier was specified.
443 auto* axisIdentifierItem = valueList.itemWithoutBoundsCheck(0);
444 if (!is<CSSPrimitiveValue>(axisIdentifierItem))
445 return nullptr;
446 auto axisIdentifier = downcast<CSSPrimitiveValue>(*axisIdentifierItem).valueID();
447 if (axisIdentifier == CSSValueX) {
448 type = TransformOperation::ROTATE_X;
449 x = 1.0;
450 } else if (axisIdentifier == CSSValueY) {
451 type = TransformOperation::ROTATE_Y;
452 y = 1.0;
453 } else if (axisIdentifier == CSSValueZ) {
454 type = TransformOperation::ROTATE_3D;
455 z = 1.0;
456 } else
457 return nullptr;
458 } else if (numberOfItems == 4) {
459 // The axis was specified using a vector.
460 type = TransformOperation::ROTATE;
461 for (unsigned i = 0; i < 3; ++i) {
462 auto* valueItem = valueList.itemWithoutBoundsCheck(i);
463 if (!is<CSSPrimitiveValue>(valueItem))
464 return nullptr;
465 if (!i)
466 x = downcast<CSSPrimitiveValue>(*valueItem).doubleValue();
467 else if (i == 1)
468 y = downcast<CSSPrimitiveValue>(*valueItem).doubleValue();
469 else if (i == 2) {
470 type = TransformOperation::ROTATE_3D;
471 z = downcast<CSSPrimitiveValue>(*valueItem).doubleValue();
472 }
473 }
474 }
475
476 return RotateTransformOperation::create(x, y, z, angle, type);
477}
