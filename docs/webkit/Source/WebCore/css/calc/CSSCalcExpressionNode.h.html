34namespace WebCore {

36class CSSToLengthConversionData;
37
38enum CSSPropertyID : uint16_t;
39
40enum class CSSUnitType : uint8_t;
41
42class CSSCalcExpressionNode : public RefCounted<CSSCalcExpressionNode> {
43public:
44 enum Type {
45 CssCalcPrimitiveValue = 1,
46 CssCalcOperation,
47 CssCalcNegate,
48 CssCalcInvert,
49 };
50
51 virtual ~CSSCalcExpressionNode() = default;
52 virtual bool isZero() const = 0;
53 virtual std::unique_ptr<CalcExpressionNode> createCalcExpression(const CSSToLengthConversionData&) const = 0;
54 virtual double doubleValue(CSSUnitType) const = 0;
55 virtual double computeLengthPx(const CSSToLengthConversionData&) const = 0;
56 virtual bool equals(const CSSCalcExpressionNode& other) const { return m_category == other.m_category; }
57 virtual Type type() const = 0;
58 virtual CSSUnitType primitiveType() const = 0;
59
60 virtual void collectDirectComputationalDependencies(HashSet<CSSPropertyID>&) const = 0;
61 virtual void collectDirectRootComputationalDependencies(HashSet<CSSPropertyID>&) const = 0;
62
63 CalculationCategory category() const { return m_category; }
64
65 virtual void dump(TextStream&) const = 0;
66
67protected:
68 CSSCalcExpressionNode(CalculationCategory category)
69 : m_category(category)
70 {
71 }
72
73private:
74 CalculationCategory m_category;
75};
