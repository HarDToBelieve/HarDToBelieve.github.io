44static constexpr int maxExpressionDepth = 100;
45
46// <https://drafts.csswg.org/css-values-4/#calc-syntax>:
47// <calc()> = calc( <calc-sum> )
48// <min()> = min( <calc-sum># )
49// <max()> = max( <calc-sum># )
50// <clamp()> = clamp( <calc-sum>#{3} )
51// <sin()> = sin( <calc-sum> )
52// <cos()> = cos( <calc-sum> )
53// <tan()> = tan( <calc-sum> )
54// <asin()> = asin( <calc-sum> )
55// <acos()> = acos( <calc-sum> )
56// <atan()> = atan( <calc-sum> )
57// <atan2()> = atan2( <calc-sum>, <calc-sum> )
58// <pow()> = pow( <calc-sum>, <calc-sum> )
59// <sqrt()> = sqrt( <calc-sum> )
60// <hypot()> = hypot( <calc-sum># )
61// <calc-sum> = <calc-product> [ [ '+' | '-' ] <calc-product> ]*
62// <calc-product> = <calc-value> [ [ '*' | '/' ] <calc-value> ]*
63// <calc-value> = <number> | <dimension> | <percentage> | ( <calc-sum> )
64RefPtr<CSSCalcExpressionNode> CSSCalcExpressionNodeParser::parseCalc(CSSParserTokenRange tokens, CSSValueID function)
65{
66 tokens.consumeWhitespace();
67
68 RefPtr<CSSCalcExpressionNode> result;
69 bool ok = parseCalcFunction(tokens, function, 0, result);
70 if (!ok || !tokens.atEnd())
71 return nullptr;
72
73 if (!result)
74 return nullptr;
75
76 LOG_WITH_STREAM(Calc, stream << "CSSCalcExpressionNodeParser::parseCalc " << prettyPrintNode(*result));
77
78 result = CSSCalcOperationNode::simplify(result.releaseNonNull());
79
80 LOG_WITH_STREAM(Calc, stream << "CSSCalcExpressionNodeParser::parseCalc - after simplification " << prettyPrintNode(*result));
81
82 return result;
83}

85char CSSCalcExpressionNodeParser::operatorValue(const CSSParserToken& token)
86{
87 if (token.type() == DelimiterToken)
88 return token.delimiter();
89 return 0;

92enum ParseState {
93 OK,
94 TooDeep,
95 NoMoreTokens
96};
97
98static ParseState checkDepthAndIndex(int depth, CSSParserTokenRange tokens)
99{
100 if (tokens.atEnd())
101 return NoMoreTokens;
102 if (depth > maxExpressionDepth) {
103 LOG_WITH_STREAM(Calc, stream << "Depth " << depth << " exceeded maxExpressionDepth " << maxExpressionDepth);
104 return TooDeep;
105 }
106 return OK;
107}

109bool CSSCalcExpressionNodeParser::parseCalcFunction(CSSParserTokenRange& tokens, CSSValueID functionID, int depth, RefPtr<CSSCalcExpressionNode>& result)
110{
111 if (checkDepthAndIndex(depth, tokens) != OK)
112 return false;
113
114 // "arguments" refers to things between commas.
115 unsigned minArgumentCount = 1;
116 std::optional<unsigned> maxArgumentCount;
117
118 switch (functionID) {
119 case CSSValueMin:
120 case CSSValueMax:
121 maxArgumentCount = std::nullopt;
122 break;
123 case CSSValueClamp:
124 minArgumentCount = 3;
125 maxArgumentCount = 3;
126 break;
127 case CSSValueCalc:
128 maxArgumentCount = 1;
129 break;
130 // TODO: clamp, sin, cos, tan, asin, acos, atan, atan2, pow, sqrt, hypot.
131 default:
132 break;
133 }
134
135 Vector<Ref<CSSCalcExpressionNode>> nodes;
136
137 bool requireComma = false;
138 unsigned argumentCount = 0;
139 while (!tokens.atEnd()) {
140 tokens.consumeWhitespace();
141 if (requireComma && !CSSPropertyParserHelpers::consumeCommaIncludingWhitespace(tokens))
142 return false;
143
144 RefPtr<CSSCalcExpressionNode> node;
145 if (!parseCalcSum(tokens, depth, node))
146 return false;
147
148 ++argumentCount;
149 if (maxArgumentCount && argumentCount > maxArgumentCount.value())
150 return false;
151
152 nodes.append(node.releaseNonNull());
153 requireComma = true;
154 }
155
156 if (argumentCount < minArgumentCount)
157 return false;
158
159 switch (functionID) {
160 case CSSValueMin:
161 result = CSSCalcOperationNode::createMinOrMaxOrClamp(CalcOperator::Min, WTFMove(nodes), m_destinationCategory);
162 break;
163 case CSSValueMax:
164 result = CSSCalcOperationNode::createMinOrMaxOrClamp(CalcOperator::Max, WTFMove(nodes), m_destinationCategory);
165 break;
166 case CSSValueClamp:
167 result = CSSCalcOperationNode::createMinOrMaxOrClamp(CalcOperator::Clamp, WTFMove(nodes), m_destinationCategory);
168 break;
169 case CSSValueWebkitCalc:
170 case CSSValueCalc:
171 result = CSSCalcOperationNode::createSum(WTFMove(nodes));
172 break;
173 // TODO: clamp, sin, cos, tan, asin, acos, atan, atan2, pow, sqrt, hypot
174 default:
175 break;
176 }
177
178 return !!result;
179}

181bool CSSCalcExpressionNodeParser::parseValue(CSSParserTokenRange& tokens, RefPtr<CSSCalcExpressionNode>& result)
182{
183 auto makeCSSCalcPrimitiveValueNode = [&] (CSSUnitType type, double value) -> bool {
184 if (calcUnitCategory(type) == CalculationCategory::Other)
185 return false;
186
187 result = CSSCalcPrimitiveValueNode::create(CSSPrimitiveValue::create(value, type));
188 return true;
189 };
190
191 auto token = tokens.consumeIncludingWhitespace();
192
193 switch (token.type()) {
194 case IdentToken: {
195 auto value = m_symbolTable.get(token.id());
196 if (!value)
197 return false;
198 return makeCSSCalcPrimitiveValueNode(value->type, value->value);
199 }
200
201 case NumberToken:
202 case PercentageToken:
203 case DimensionToken:
204 return makeCSSCalcPrimitiveValueNode(token.unitType(), token.numericValue());
205
206 default:
207 return false;
208 }
209
210 ASSERT_NOT_REACHED();
211 return false;
212}

214bool CSSCalcExpressionNodeParser::parseCalcValue(CSSParserTokenRange& tokens, int depth, RefPtr<CSSCalcExpressionNode>& result)
215{
216 if (checkDepthAndIndex(depth, tokens) != OK)
217 return false;
218
219 auto findFunctionId = [&](CSSValueID& functionId) {
220 if (tokens.peek().type() == LeftParenthesisToken) {
221 functionId = CSSValueCalc;
222 return true;
223 }
224
225 functionId = tokens.peek().functionId();
226 return CSSCalcValue::isCalcFunction(functionId);
227 };
228
229 CSSValueID functionId;
230 if (findFunctionId(functionId)) {
231 CSSParserTokenRange innerRange = tokens.consumeBlock();
232 tokens.consumeWhitespace();
233 innerRange.consumeWhitespace();
234 return parseCalcFunction(innerRange, functionId, depth + 1, result);
235 }
236
237 return parseValue(tokens, result);
238}

240bool CSSCalcExpressionNodeParser::parseCalcProduct(CSSParserTokenRange& tokens, int depth, RefPtr<CSSCalcExpressionNode>& result)
241{
242 if (checkDepthAndIndex(depth, tokens) != OK)
243 return false;
244
245 RefPtr<CSSCalcExpressionNode> firstValue;
246 if (!parseCalcValue(tokens, depth, firstValue))
247 return false;
248
249 Vector<Ref<CSSCalcExpressionNode>> nodes;
250
251 while (!tokens.atEnd()) {
252 char operatorCharacter = operatorValue(tokens.peek());
253 if (operatorCharacter != static_cast<char>(CalcOperator::Multiply) && operatorCharacter != static_cast<char>(CalcOperator::Divide))
254 break;
255 tokens.consumeIncludingWhitespace();
256
257 RefPtr<CSSCalcExpressionNode> nextValue;
258 if (!parseCalcValue(tokens, depth, nextValue) || !nextValue)
259 return false;
260
261 if (operatorCharacter == static_cast<char>(CalcOperator::Divide))
262 nextValue = CSSCalcInvertNode::create(nextValue.releaseNonNull());
263
264 if (firstValue)
265 nodes.append(firstValue.releaseNonNull());
266
267 nodes.append(nextValue.releaseNonNull());
268 }
269
270 if (nodes.isEmpty()) {
271 result = WTFMove(firstValue);
272 return !!result;
273 }
274
275 result = CSSCalcOperationNode::createProduct(WTFMove(nodes));
276 return !!result;
277}

279bool CSSCalcExpressionNodeParser::parseCalcSum(CSSParserTokenRange& tokens, int depth, RefPtr<CSSCalcExpressionNode>& result)
280{
281 if (checkDepthAndIndex(depth, tokens) != OK)
282 return false;
283
284 RefPtr<CSSCalcExpressionNode> firstValue;
285 if (!parseCalcProduct(tokens, depth, firstValue))
286 return false;
287
288 Vector<Ref<CSSCalcExpressionNode>> nodes;
289
290 while (!tokens.atEnd()) {
291 char operatorCharacter = operatorValue(tokens.peek());
292 if (operatorCharacter != static_cast<char>(CalcOperator::Add) && operatorCharacter != static_cast<char>(CalcOperator::Subtract))
293 break;
294
295 if ((&tokens.peek() - 1)->type() != WhitespaceToken)
296 return false; // calc(1px+ 2px) is invalid
297
298 tokens.consume();
299 if (tokens.peek().type() != WhitespaceToken)
300 return false; // calc(1px +2px) is invalid
301
302 tokens.consumeIncludingWhitespace();
303
304 RefPtr<CSSCalcExpressionNode> nextValue;
305 if (!parseCalcProduct(tokens, depth, nextValue) || !nextValue)
306 return false;
307
308 if (operatorCharacter == static_cast<char>(CalcOperator::Subtract))
309 nextValue = CSSCalcNegateNode::create(nextValue.releaseNonNull());
310
311 if (firstValue)
312 nodes.append(firstValue.releaseNonNull());
313
314 nodes.append(nextValue.releaseNonNull());
315 }
316
317 if (nodes.isEmpty()) {
318 result = WTFMove(firstValue);
319 return !!result;
320 }
321
322 result = CSSCalcOperationNode::createSum(WTFMove(nodes));
323 return !!result;
324}
