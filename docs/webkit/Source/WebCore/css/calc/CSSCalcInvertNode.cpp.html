34std::unique_ptr<CalcExpressionNode> CSSCalcInvertNode::createCalcExpression(const CSSToLengthConversionData& conversionData) const
35{
36 auto childNode = m_child->createCalcExpression(conversionData);
37 return makeUnique<CalcExpressionInversion>(WTFMove(childNode));
38}
39
40double CSSCalcInvertNode::doubleValue(CSSUnitType unitType) const
41{
42 auto childValue = m_child->doubleValue(unitType);
43 if (!childValue)
44 return std::numeric_limits<double>::infinity();
45 return 1.0 / childValue;
46}
47
48double CSSCalcInvertNode::computeLengthPx(const CSSToLengthConversionData& conversionData) const
49{
50 auto childValue = m_child->computeLengthPx(conversionData);
51 if (!childValue)
52 return std::numeric_limits<double>::infinity();
53 return 1.0 / childValue;
54}
55
56void CSSCalcInvertNode::dump(TextStream& ts) const
57{
58 ts << "1/" << m_child.get();
59}
