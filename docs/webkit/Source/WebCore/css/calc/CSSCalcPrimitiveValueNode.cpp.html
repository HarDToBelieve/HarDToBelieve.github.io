67bool CSSCalcPrimitiveValueNode::isNumericValue() const
68{
69 return m_value->isLength() || m_value->isNumber() || m_value->isPercentage() || m_value->isAngle()

73bool CSSCalcPrimitiveValueNode::isNegative() const
74{
75 return isNumericValue() && m_value->doubleValue() < 0.0;
76}

78void CSSCalcPrimitiveValueNode::negate()
79{
80 ASSERT(isNumericValue());
81 m_value = CSSPrimitiveValue::create(0.0 - m_value->doubleValue(), m_value->primitiveType());
82}

84void CSSCalcPrimitiveValueNode::invert()
85{
86 ASSERT(isNumericValue());
87 if (!m_value->doubleValue())
88 m_value = CSSPrimitiveValue::create(std::numeric_limits<double>::infinity(), m_value->primitiveType());
89
90 m_value = CSSPrimitiveValue::create(1.0 / m_value->doubleValue(), m_value->primitiveType());
91}

93void CSSCalcPrimitiveValueNode::add(const CSSCalcPrimitiveValueNode& node, UnitConversion unitConversion)
94{
95 auto valueType = m_value->primitiveType();
96
97 switch (unitConversion) {
98 case UnitConversion::Invalid:
99 ASSERT_NOT_REACHED();
100 break;
101 case UnitConversion::Preserve:
102 ASSERT(node.primitiveType() == valueType);
103 m_value = CSSPrimitiveValue::create(m_value->doubleValue() + node.doubleValue(valueType), valueType);
104 break;
105 case UnitConversion::Canonicalize: {
106 auto valueCategory = unitCategory(valueType);
107 // FIXME: It's awkward that canonicalUnitTypeForCategory() has special handling for CSSUnitCategory::Percent.
108 auto canonicalType = valueCategory == CSSUnitCategory::Percent ? CSSUnitType::CSS_PERCENTAGE : canonicalUnitTypeForCategory(valueCategory);
109 ASSERT(canonicalType != CSSUnitType::CSS_UNKNOWN);
110 double leftValue = m_value->doubleValue(canonicalType);
111 double rightValue = node.doubleValue(canonicalType);
112 m_value = CSSPrimitiveValue::create(leftValue + rightValue, canonicalType);
113 break;
114 }
115 }
116}

125void CSSCalcPrimitiveValueNode::convertToUnitType(CSSUnitType unitType)
126{
127 ASSERT(unitCategory(unitType) == unitCategory(m_value->primitiveType()));

132void CSSCalcPrimitiveValueNode::canonicalizeUnit()
133{
134 auto category = calculationCategoryForCombination(m_value->primitiveType());
135 if (category == CalculationCategory::Other)
136 return;
137
138 auto canonicalType = canonicalUnitTypeForCalculationCategory(category);
139 if (canonicalType == m_value->primitiveType())
140 return;
141
142 double newValue = m_value->doubleValue(canonicalType);
143 m_value = CSSPrimitiveValue::create(newValue, canonicalType);
144}

146std::unique_ptr<CalcExpressionNode> CSSCalcPrimitiveValueNode::createCalcExpression(const CSSToLengthConversionData& conversionData) const
147{
148 switch (category()) {
149 case CalculationCategory::Number:
150 return makeUnique<CalcExpressionNumber>(m_value->floatValue());
151 case CalculationCategory::Length:
152 return makeUnique<CalcExpressionLength>(Length(m_value->computeLength<float>(conversionData), LengthType::Fixed));
153 case CalculationCategory::Percent:
154 case CalculationCategory::PercentLength: {
155 return makeUnique<CalcExpressionLength>(m_value->convertToLength<FixedFloatConversion | PercentConversion>(conversionData));
156 }
157 // Only types that could be part of a Length expression can be converted
158 // to a CalcExpressionNode. CalculationCategory::PercentNumber makes no sense as a Length.
159 case CalculationCategory::PercentNumber:
160 case CalculationCategory::Angle:
161 case CalculationCategory::Time:
162 case CalculationCategory::Frequency:
163 case CalculationCategory::Other:
164 ASSERT_NOT_REACHED();
165 }
166 ASSERT_NOT_REACHED();
167 return nullptr;
168}

170double CSSCalcPrimitiveValueNode::doubleValue(CSSUnitType unitType) const
171{
172 if (hasDoubleValue(unitType)) {
173 // FIXME: This should ASSERT(unitCategory(m_value->primitiveType()) == unitCategory(unitType)), but only when all callers are fixed (e.g. webkit.org/b/204826).
174 if (unitCategory(m_value->primitiveType()) != unitCategory(unitType)) {
175 LOG_WITH_STREAM(Calc, stream << "Calling doubleValue() with unit " << unitType << " on a node of unit type " << m_value->primitiveType() << " which is incompatible");
176 return 0;
177 }
178
179 return m_value->doubleValue(unitType);
180 }
181 ASSERT_NOT_REACHED();
182 return 0;
183}

185double CSSCalcPrimitiveValueNode::computeLengthPx(const CSSToLengthConversionData& conversionData) const
186{
187 switch (category()) {
188 case CalculationCategory::Length:
189 return m_value->computeLength<double>(conversionData);
190 case CalculationCategory::Percent:
191 case CalculationCategory::Number:
192 return m_value->doubleValue();
193 case CalculationCategory::PercentLength:
194 case CalculationCategory::PercentNumber:
195 case CalculationCategory::Angle:
196 case CalculationCategory::Time:
197 case CalculationCategory::Frequency:
198 case CalculationCategory::Other:
199 ASSERT_NOT_REACHED();
200 break;
201 }
202 ASSERT_NOT_REACHED();
203 return 0;
204}
