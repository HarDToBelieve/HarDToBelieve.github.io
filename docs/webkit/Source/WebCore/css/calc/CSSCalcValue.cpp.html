61static inline RefPtr<CSSCalcOperationNode> createBlendHalf(const Length& length, const RenderStyle& style, float progress)
62{
63 return CSSCalcOperationNode::create(CalcOperator::Multiply, createCSS(length, style),
64 CSSCalcPrimitiveValueNode::create(CSSPrimitiveValue::create(progress, CSSUnitType::CSS_NUMBER)));
65}

67static Vector<Ref<CSSCalcExpressionNode>> createCSS(const Vector<std::unique_ptr<CalcExpressionNode>>& nodes, const RenderStyle& style)
68{
69 Vector<Ref<CSSCalcExpressionNode>> values;
70 values.reserveInitialCapacity(nodes.size());
71 for (auto& node : nodes) {
72 auto cssNode = createCSS(*node, style);
73 if (!cssNode)
74 continue;
75 values.uncheckedAppend(cssNode.releaseNonNull());
76 }
77 return values;
78}

80static RefPtr<CSSCalcExpressionNode> createCSS(const CalcExpressionNode& node, const RenderStyle& style)
81{
82 switch (node.type()) {
83 case CalcExpressionNodeType::Number: {
84 float value = downcast<CalcExpressionNumber>(node).value(); // double?
85 return CSSCalcPrimitiveValueNode::create(CSSPrimitiveValue::create(value, CSSUnitType::CSS_NUMBER));
86 }
87 case CalcExpressionNodeType::Length: {
88 auto& length = downcast<CalcExpressionLength>(node).length();
89 if (!length.isPercent() && length.isZero())
90 return nullptr;
91 return createCSS(length, style);
92 }
93
94 case CalcExpressionNodeType::Negation: {
95 auto childNode = createCSS(*downcast<CalcExpressionNegation>(node).child(), style);
96 if (!childNode)
97 return nullptr;
98 return CSSCalcNegateNode::create(childNode.releaseNonNull());
99 }
100 case CalcExpressionNodeType::Inversion: {
101 auto childNode = createCSS(*downcast<CalcExpressionInversion>(node).child(), style);
102 if (!childNode)
103 return nullptr;
104 return CSSCalcInvertNode::create(childNode.releaseNonNull());
105 }
106 case CalcExpressionNodeType::Operation: {
107 auto& operationNode = downcast<CalcExpressionOperation>(node);
108 auto& operationChildren = operationNode.children();
109 CalcOperator op = operationNode.getOperator();
110
111 switch (op) {
112 case CalcOperator::Add: {
113 auto children = createCSS(operationChildren, style);
114 if (children.isEmpty())
115 return nullptr;
116 if (children.size() == 1)
117 return WTFMove(children[0]);
118 return CSSCalcOperationNode::createSum(WTFMove(children));
119 }
120 case CalcOperator::Subtract: {
121 ASSERT(operationChildren.size() == 2);
122
123 Vector<Ref<CSSCalcExpressionNode>> values;
124 values.reserveInitialCapacity(operationChildren.size());
125
126 auto firstChild = createCSS(*operationChildren[0], style);
127 auto secondChild = createCSS(*operationChildren[1], style);
128
129 if (!secondChild)
130 return firstChild;
131
132 auto negateNode = CSSCalcNegateNode::create(secondChild.releaseNonNull());
133 if (!firstChild)
134 return negateNode;
135
136 values.append(firstChild.releaseNonNull());
137 values.append(WTFMove(negateNode));
138
139 return CSSCalcOperationNode::createSum(WTFMove(values));
140 }
141 case CalcOperator::Multiply: {
142 auto children = createCSS(operationChildren, style);
143 if (children.isEmpty())
144 return nullptr;
145 return CSSCalcOperationNode::createProduct(WTFMove(children));
146 }
147 case CalcOperator::Divide: {
148 ASSERT(operationChildren.size() == 2);
149
150 Vector<Ref<CSSCalcExpressionNode>> values;
151 values.reserveInitialCapacity(operationChildren.size());
152
153 auto firstChild = createCSS(*operationChildren[0], style);
154 if (!firstChild)
155 return nullptr;
156
157 auto secondChild = createCSS(*operationChildren[1], style);
158 if (!secondChild)
159 return nullptr;
160 auto invertNode = CSSCalcInvertNode::create(secondChild.releaseNonNull());
161
162 values.append(firstChild.releaseNonNull());
163 values.append(WTFMove(invertNode));
164
165 return CSSCalcOperationNode::createProduct(createCSS(operationChildren, style));
166 }
167 case CalcOperator::Min:
168 case CalcOperator::Max:
169 case CalcOperator::Clamp: {
170 auto children = createCSS(operationChildren, style);
171 if (children.isEmpty())
172 return nullptr;
173 return CSSCalcOperationNode::createMinOrMaxOrClamp(op, WTFMove(children), operationNode.destinationCategory());
174 }
175 }
176 return nullptr;
177 }
178 case CalcExpressionNodeType::BlendLength: {
179 // FIXME: (http://webkit.org/b/122036) Create a CSSCalcExpressionNode equivalent of CalcExpressionBlendLength.
180 auto& blend = downcast<CalcExpressionBlendLength>(node);
181 float progress = blend.progress();
182 return CSSCalcOperationNode::create(CalcOperator::Add, createBlendHalf(blend.from(), style, 1 - progress), createBlendHalf(blend.to(), style, progress));
183 }
184 case CalcExpressionNodeType::Undefined:
185 ASSERT_NOT_REACHED();
186 }
187 return nullptr;
188}

190static RefPtr<CSSCalcExpressionNode> createCSS(const Length& length, const RenderStyle& style)
191{
192 switch (length.type()) {
193 case LengthType::Percent:
194 case LengthType::Fixed:
195 return CSSCalcPrimitiveValueNode::create(CSSPrimitiveValue::create(length, style));
196 case LengthType::Calculated:
197 return createCSS(length.calculationValue().expression(), style);
198 case LengthType::Auto:
199 case LengthType::Intrinsic:
200 case LengthType::MinIntrinsic:
201 case LengthType::MinContent:
202 case LengthType::MaxContent:
203 case LengthType::FillAvailable:
204 case LengthType::FitContent:
205 case LengthType::Relative:
206 case LengthType::Undefined:
207 ASSERT_NOT_REACHED();
208 }
209 return nullptr;
210}

278bool CSSCalcValue::isCalcFunction(CSSValueID functionId)
279{
280 switch (functionId) {
281 case CSSValueCalc:
282 case CSSValueWebkitCalc:
283 case CSSValueMin:
284 case CSSValueMax:
285 case CSSValueClamp:
286 return true;
287 default:
288 return false;
289 }
290 return false;
291}

293void CSSCalcValue::dump(TextStream& ts) const
294{
295 ts << indent << "(" << "CSSCalcValue";
296
297 TextStream multilineStream;
298 multilineStream.setIndent(ts.indent() + 2);
299
300 multilineStream.dumpProperty("should clamp non-negative", m_shouldClampToNonNegative);
301 multilineStream.dumpProperty("expression", m_expression.get());
302
303 ts << multilineStream.release();
304 ts << ")\n";
305}

307RefPtr<CSSCalcValue> CSSCalcValue::create(CSSValueID function, const CSSParserTokenRange& tokens, CalculationCategory destinationCategory, ValueRange range, const CSSCalcSymbolTable& symbolTable)
308{
309 CSSCalcExpressionNodeParser parser(destinationCategory, symbolTable);
310 auto expression = parser.parseCalc(tokens, function);
311 if (!expression)
312 return nullptr;
313 auto result = adoptRef(new CSSCalcValue(expression.releaseNonNull(), range != ValueRange::All));
314 LOG_WITH_STREAM(Calc, stream << "CSSCalcValue::create " << *result);
315 return result;
316}

318RefPtr<CSSCalcValue> CSSCalcValue::create(CSSValueID function, const CSSParserTokenRange& tokens, CalculationCategory destinationCategory, ValueRange range)
319{
320 return create(function, tokens, destinationCategory, range, { });
321}

323RefPtr<CSSCalcValue> CSSCalcValue::create(const CalculationValue& value, const RenderStyle& style)
324{
325 auto expression = createCSS(value.expression(), style);
326 if (!expression)
327 return nullptr;
328
329 auto simplifiedExpression = CSSCalcOperationNode::simplify(expression.releaseNonNull());
330
331 auto result = adoptRef(new CSSCalcValue(WTFMove(simplifiedExpression), value.shouldClampToNonNegative()));
332 LOG_WITH_STREAM(Calc, stream << "CSSCalcValue::create from CalculationValue: " << *result);
333 return result;
334}
