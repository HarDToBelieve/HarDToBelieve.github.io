33namespace WebCore {
34
35CSSDeferredParser::CSSDeferredParser(const CSSParserContext& context, const String& sheetText, StyleSheetContents& styleSheet)
36 : m_context(context)
37 , m_sheetText(sheetText)
38 , m_styleSheet(makeWeakPtr(styleSheet))
39{
40}
41
42StyleSheetContents* CSSDeferredParser::styleSheet() const
43{
44 return m_styleSheet.get();
45}
46
47Ref<ImmutableStyleProperties> CSSDeferredParser::parseDeclaration(const CSSParserTokenRange& range)
48{
49 return CSSParserImpl::parseDeferredDeclaration(range, m_context, m_styleSheet.get());
50}
51
52void CSSDeferredParser::parseRuleList(const CSSParserTokenRange& range, Vector<RefPtr<StyleRuleBase>>& childRules)
53{
54 CSSParserImpl::parseDeferredRuleList(range, *this, childRules);
55}
56
57void CSSDeferredParser::parseKeyframeList(const CSSParserTokenRange& range, StyleRuleKeyframes&keyframesRule)
58{
59 CSSParserImpl::parseDeferredKeyframeList(range, *this, keyframesRule);
60}
