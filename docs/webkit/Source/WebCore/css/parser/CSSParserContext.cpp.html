51#if ENABLE(OVERFLOW_SCROLLING_TOUCH)
52static bool shouldEnableLegacyOverflowScrollingTouch(const Document& document)
53{
54 // The legacy -webkit-overflow-scrolling: touch behavior may have been disabled through the website policy,
55 // in that case we want to disable the legacy behavior regardless of what the setting says.
56 if (auto* loader = document.loader()) {
57 if (loader->legacyOverflowScrollingTouchPolicy() == LegacyOverflowScrollingTouchPolicy::Disable)
58 return false;
59 }
60 return document.settings().legacyOverflowScrollingTouchEnabled();
61}
62#endif

64CSSParserContext::CSSParserContext(const Document& document, const URL& sheetBaseURL, const String& charset)
65 : baseURL { sheetBaseURL.isNull() ? document.baseURL() : sheetBaseURL }
66 , charset { charset }
67 , mode { document.inQuirksMode() ? HTMLQuirksMode : HTMLStandardMode }
68 , isHTMLDocument { document.isHTMLDocument() }
69 , hasDocumentSecurityOrigin { sheetBaseURL.isNull() || document.securityOrigin().canRequest(baseURL) }
70 , useSystemAppearance { document.page() ? document.page()->useSystemAppearance() : false }
71 , aspectRatioEnabled { document.settings().aspectRatioEnabled() }
72 , colorContrastEnabled { document.settings().cssColorContrastEnabled() }
73 , colorFilterEnabled { document.settings().colorFilterEnabled() }
74 , colorMixEnabled { document.settings().cssColorMixEnabled() }
75 , constantPropertiesEnabled { document.settings().constantPropertiesEnabled() }
76 , containmentEnabled { document.settings().cssContainmentEnabled() }
77 , counterStyleAtRulesEnabled { document.settings().cssCounterStyleAtRulesEnabled() }
78 , counterStyleAtRuleImageSymbolsEnabled { document.settings().cssCounterStyleAtRuleImageSymbolsEnabled() }
79 , cssColor4 { document.settings().cssColor4() }
80 , deferredCSSParserEnabled { document.settings().deferredCSSParserEnabled() }
81 , individualTransformPropertiesEnabled { document.settings().cssIndividualTransformPropertiesEnabled() }
82#if ENABLE(OVERFLOW_SCROLLING_TOUCH)
83 , legacyOverflowScrollingTouchEnabled { shouldEnableLegacyOverflowScrollingTouch(document) }
84#endif
85 , overscrollBehaviorEnabled { document.settings().overscrollBehaviorEnabled() }
86 , relativeColorSyntaxEnabled { document.settings().cssRelativeColorSyntaxEnabled() }
87 , scrollBehaviorEnabled { document.settings().CSSOMViewSmoothScrollingEnabled() }
88 , springTimingFunctionEnabled { document.settings().springTimingFunctionEnabled() }
89#if ENABLE(TEXT_AUTOSIZING)
90 , textAutosizingEnabled { document.settings().textAutosizingEnabled() }
91#endif
92#if ENABLE(CSS_TRANSFORM_STYLE_OPTIMIZED_3D)
93 , transformStyleOptimized3DEnabled { document.settings().cssTransformStyleOptimized3DEnabled() }

95 , useLegacyBackgroundSizeShorthandBehavior { document.settings().useLegacyBackgroundSizeShorthandBehavior() }
96 , focusVisibleEnabled { document.settings().focusVisibleEnabled() }

98 , attachmentEnabled { RuntimeEnabledFeatures::sharedFeatures().attachmentElementEnabled() }

104{

110 && a.hasDocumentSecurityOrigin == b.hasDocumentSecurityOrigin
111 && a.isContentOpaque == b.isContentOpaque

145void add(Hasher& hasher, const CSSParserContext& context)
146{
147 unsigned bits = context.isHTMLDocument << 0
148 | context.hasDocumentSecurityOrigin << 1
149 | context.isContentOpaque << 2
150 | context.useSystemAppearance << 3
151 | context.aspectRatioEnabled << 4
152 | context.colorContrastEnabled << 5
153 | context.colorFilterEnabled << 6
154 | context.colorMixEnabled << 7
155 | context.constantPropertiesEnabled << 8
156 | context.containmentEnabled << 9
157 | context.cssColor4 << 10
158 | context.deferredCSSParserEnabled << 11
159 | context.individualTransformPropertiesEnabled << 12
160#if ENABLE(OVERFLOW_SCROLLING_TOUCH)
161 | context.legacyOverflowScrollingTouchEnabled << 13
162#endif
163 | context.overscrollBehaviorEnabled << 14
164 | context.relativeColorSyntaxEnabled << 15
165 | context.scrollBehaviorEnabled << 16
166 | context.springTimingFunctionEnabled << 17
167#if ENABLE(TEXT_AUTOSIZING)
168 | context.textAutosizingEnabled << 18
169#endif
170#if ENABLE(CSS_TRANSFORM_STYLE_OPTIMIZED_3D)
171 | context.transformStyleOptimized3DEnabled << 19
172#endif
173 | context.useLegacyBackgroundSizeShorthandBehavior << 20
174 | context.focusVisibleEnabled << 21
175#if ENABLE(ATTACHMENT_ELEMENT)
176 | context.attachmentEnabled << 22
177#endif
178 | context.mode << 23; // This is multiple bits, so keep it last.
179 add(hasher, context.baseURL, context.charset, bits);
180}

182bool CSSParserContext::isPropertyRuntimeDisabled(CSSPropertyID property) const
183{
184 switch (property) {
185 case CSSPropertyAdditiveSymbols:
186 case CSSPropertyFallback:
187 case CSSPropertyPad:
188 case CSSPropertySymbols:
189 case CSSPropertyNegative:
190 case CSSPropertyPrefix:
191 case CSSPropertyRange:
192 case CSSPropertySuffix:
193 case CSSPropertySystem:
194 return !counterStyleAtRulesEnabled;
195 case CSSPropertyAspectRatio:
196 return !aspectRatioEnabled;
197 case CSSPropertyContain:
198 return !containmentEnabled;
199 case CSSPropertyAppleColorFilter:
200 return !colorFilterEnabled;
201 case CSSPropertyTranslate:
202 case CSSPropertyRotate:
203 case CSSPropertyScale:
204 return !individualTransformPropertiesEnabled;
205 case CSSPropertyOverscrollBehavior:
206 case CSSPropertyOverscrollBehaviorX:
207 case CSSPropertyOverscrollBehaviorY:
208 return !overscrollBehaviorEnabled;
209 case CSSPropertyScrollBehavior:
210 return !scrollBehaviorEnabled;
211#if ENABLE(TEXT_AUTOSIZING) && !PLATFORM(IOS_FAMILY)
212 case CSSPropertyWebkitTextSizeAdjust:
213 return !textAutosizingEnabled;
214#endif
215#if ENABLE(OVERFLOW_SCROLLING_TOUCH)
216 case CSSPropertyWebkitOverflowScrolling:
217 return !legacyOverflowScrollingTouchEnabled;
218#endif
219 default:
220 return false;
221 }
222}

224ResolvedURL CSSParserContext::completeURL(const String& string) const
225{
226 auto result = [&] () -> ResolvedURL {
227 if (string.isNull())
228 return { };
229 if (charset.isEmpty())
230 return { string, { baseURL, string } };
231 auto encodingForURLParsing = TextEncoding { charset }.encodingForFormSubmissionOrURLParsing();
232 return { string, { baseURL, string, encodingForURLParsing == UTF8Encoding() ? nullptr : &encodingForURLParsing } };
233 }();
234
235 if (mode == WebVTTMode && !result.resolvedURL.protocolIsData())
236 return { };
237
238 return result;
239}
