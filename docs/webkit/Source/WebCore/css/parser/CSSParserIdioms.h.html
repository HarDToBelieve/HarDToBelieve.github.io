39// Space characters as defined by the CSS specification.
40// http://www.w3.org/TR/css3-syntax/#whitespace
41inline bool isCSSSpace(UChar c)
42{
43 return c == ' ' || c == '\t' || c == '\n';
44}
45
46// http://dev.w3.org/csswg/css-syntax/#name-start-code-point
47template <typename CharacterType>
48bool isNameStartCodePoint(CharacterType c)
49{
50 return isASCIIAlpha(c) || c == '_' || !isASCII(c);
51}
52
53// http://dev.w3.org/csswg/css-syntax/#name-code-point
54template <typename CharacterType>
55bool isNameCodePoint(CharacterType c)
56{
57 return isNameStartCodePoint(c) || isASCIIDigit(c) || c == '-';
58}
