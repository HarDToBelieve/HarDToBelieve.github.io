71 enum AllowedRulesType {
72 // As per css-syntax, css-cascade and css-namespaces, @charset rules
73 // must come first, followed by @import then @namespace.
74 // AllowImportRules actually means we allow @import and any rules thay
75 // may follow it, i.e. @namespace rules and regular rules.
76 // AllowCharsetRules and AllowNamespaceRules behave similarly.
77 AllowCharsetRules,
78 AllowImportRules,
79 AllowNamespaceRules,
80 RegularRules,
81 KeyframeRules,
82 ApplyRules, // For @apply inside style rules
83 CounterStyleRules,
84 NoRules, // For parsing at-rules inside declaration lists
85 };
