30#pragma once
31
32#include "CSSParserObserver.h"
33
34namespace WebCore {
35
36class CSSParserObserverWrapper {
37public:
38 explicit CSSParserObserverWrapper(CSSParserObserver& observer)
39 : m_observer(observer)
40 { }
41
42 unsigned startOffset(const CSSParserTokenRange&);
43 unsigned previousTokenStartOffset(const CSSParserTokenRange&);
44 unsigned endOffset(const CSSParserTokenRange&); // Includes trailing comments
45
46 void skipCommentsBefore(const CSSParserTokenRange&, bool leaveDirectlyBefore);
47 void yieldCommentsBefore(const CSSParserTokenRange&);
48
49 CSSParserObserver& observer() { return m_observer; }
50 void addComment(unsigned startOffset, unsigned endOffset, unsigned tokensBefore)
51 {
52 CommentPosition position = {startOffset, endOffset, tokensBefore};
53 m_commentOffsets.append(position);
54 }
55 void addToken(unsigned startOffset) { m_tokenOffsets.append(startOffset); }
56 void finalizeConstruction(CSSParserToken* firstParserToken)
57 {
58 m_firstParserToken = firstParserToken;
59 m_commentIterator = m_commentOffsets.begin();
60 }
61
62private:
63 CSSParserObserver& m_observer;
64 Vector<unsigned> m_tokenOffsets;
65 CSSParserToken* m_firstParserToken;
66
67 struct CommentPosition {
68 unsigned startOffset;
69 unsigned endOffset;
70 unsigned tokensBefore;
71 };
72
73 Vector<CommentPosition> m_commentOffsets;
74 Vector<CommentPosition>::iterator m_commentIterator;
75};
76
77} // namespace WebCore
