43template<typename CharacterType>
44CSSUnitType cssPrimitiveValueUnitFromTrie(const CharacterType* data, unsigned length)
45{
46 ASSERT(data);
47 ASSERT(length);
48 switch (length) {
49 case 1:
50 switch (toASCIILower(data[0])) {
51 case 'q':
52 return CSSUnitType::CSS_Q;
53 case 's':
54 return CSSUnitType::CSS_S;
55 }
56 break;
57 case 2:
58 switch (toASCIILower(data[0])) {
59 case 'c':
60 switch (toASCIILower(data[1])) {
61 case 'h':
62 return CSSUnitType::CSS_CHS;
63 case 'm':
64 return CSSUnitType::CSS_CM;
65 }
66 break;
67 case 'e':
68 switch (toASCIILower(data[1])) {
69 case 'm':
70 return CSSUnitType::CSS_EMS;
71 case 'x':
72 return CSSUnitType::CSS_EXS;
73 }
74 break;
75 case 'f':
76 if (toASCIILower(data[1]) == 'r')
77 return CSSUnitType::CSS_FR;
78 break;
79 case 'h':
80 if (toASCIILower(data[1]) == 'z')
81 return CSSUnitType::CSS_HZ;
82 break;
83 case 'i':
84 if (toASCIILower(data[1]) == 'n')
85 return CSSUnitType::CSS_IN;
86 break;
87 case 'l':
88 if (toASCIILower(data[1]) == 'h' && RuntimeEnabledFeatures::sharedFeatures().lineHeightUnitsEnabled())
89 return CSSUnitType::CSS_LHS;
90 break;
91 case 'm':
92 switch (toASCIILower(data[1])) {
93 case 'm':
94 return CSSUnitType::CSS_MM;
95 case 's':
96 return CSSUnitType::CSS_MS;
97 }
98 break;
99 case 'p':
100 switch (toASCIILower(data[1])) {
101 case 'c':
102 return CSSUnitType::CSS_PC;
103 case 't':
104 return CSSUnitType::CSS_PT;
105 case 'x':
106 return CSSUnitType::CSS_PX;
107 }
108 break;
109 case 'v':
110 switch (toASCIILower(data[1])) {
111 case 'h':
112 return CSSUnitType::CSS_VH;
113 case 'w':
114 return CSSUnitType::CSS_VW;
115 }
116 break;
117 }
118 break;
119 case 3:
120 switch (toASCIILower(data[0])) {
121 case 'd':
122 switch (toASCIILower(data[1])) {
123 case 'e':
124 if (toASCIILower(data[2]) == 'g')
125 return CSSUnitType::CSS_DEG;
126 break;
127 case 'p':
128 if (toASCIILower(data[2]) == 'i')
129 return CSSUnitType::CSS_DPI;
130 break;
131 }
132 break;
133 case 'k':
134 if (toASCIILower(data[1]) == 'h' && toASCIILower(data[2]) == 'z')
135 return CSSUnitType::CSS_KHZ;
136 break;
137 case 'r':
138 switch (toASCIILower(data[1])) {
139 case 'a':
140 if (toASCIILower(data[2]) == 'd')
141 return CSSUnitType::CSS_RAD;
142 break;
143 case 'e':
144 if (toASCIILower(data[2]) == 'm')
145 return CSSUnitType::CSS_REMS;
146 break;
147 case 'l':
148 if (toASCIILower(data[2]) == 'h' && RuntimeEnabledFeatures::sharedFeatures().lineHeightUnitsEnabled())
149 return CSSUnitType::CSS_RLHS;
150 break;
151 }
152 break;
153 }
154 break;
155 case 4:
156 switch (toASCIILower(data[0])) {
157 case 'd':
158 switch (toASCIILower(data[1])) {
159 case 'p':
160 switch (toASCIILower(data[2])) {
161 case 'c':
162 if (toASCIILower(data[3]) == 'm')
163 return CSSUnitType::CSS_DPCM;
164 break;
165 case 'p':
166 if (toASCIILower(data[3]) == 'x')
167 return CSSUnitType::CSS_DPPX;
168 break;
169 }
170 break;
171 }
172 break;
173 case 'g':
174 if (toASCIILower(data[1]) == 'r' && toASCIILower(data[2]) == 'a' && toASCIILower(data[3]) == 'd')
175 return CSSUnitType::CSS_GRAD;
176 break;
177 case 't':
178 if (toASCIILower(data[1]) == 'u' && toASCIILower(data[2]) == 'r' && toASCIILower(data[3]) == 'n')
179 return CSSUnitType::CSS_TURN;
180 break;
181 case 'v':
182 switch (toASCIILower(data[1])) {
183 case 'm':
184 switch (toASCIILower(data[2])) {
185 case 'a':
186 if (toASCIILower(data[3]) == 'x')
187 return CSSUnitType::CSS_VMAX;
188 break;
189 case 'i':
190 if (toASCIILower(data[3]) == 'n')
191 return CSSUnitType::CSS_VMIN;
192 break;
193 }
194 break;
195 }
196 break;
197 }
198 break;
199 case 5:
200 switch (toASCIILower(data[0])) {
201 case '_':
202 if (toASCIILower(data[1]) == '_' && toASCIILower(data[2]) == 'q' && toASCIILower(data[3]) == 'e' && toASCIILower(data[4]) == 'm')
203 return CSSUnitType::CSS_QUIRKY_EMS;
204 break;
205 }
206 break;
207 }
208 return CSSUnitType::CSS_UNKNOWN;
209}

211static CSSUnitType stringToUnitType(StringView stringView)
212{
213 if (stringView.is8Bit())
214 return cssPrimitiveValueUnitFromTrie(stringView.characters8(), stringView.length());
215 return cssPrimitiveValueUnitFromTrie(stringView.characters16(), stringView.length());
216}

260static StringView mergeIfAdjacent(StringView a, StringView b)
261{
262 if (a.is8Bit() && b.is8Bit()) {
263 auto characters = a.characters8();
264 if (characters + a.length() == b.characters8())
265 return { characters, a.length() + b.length() };
266 } else if (!a.is8Bit() && !b.is8Bit()) {
267 auto characters = a.characters16();
268 if (characters + a.length() == b.characters16())
269 return { characters, a.length() + b.length() };
270 }
271 return { };
272}

274void CSSParserToken::convertToDimensionWithUnit(StringView unit)
275{
276 ASSERT(m_type == NumberToken);
277 auto originalNumberText = originalText();
278 auto originalNumberTextLength = originalNumberText.length();
279 auto string = unit;
280 if (originalNumberTextLength && originalNumberTextLength < 16) {
281 if (auto merged = mergeIfAdjacent(originalNumberText, unit))
282 string = merged;
283 }
284 m_type = DimensionToken;
285 m_unit = static_cast<unsigned>(stringToUnitType(unit));
286 m_nonUnitPrefixLength = string == unit ? 0 : originalNumberTextLength;
287 initValueFromStringView(string);
288}

290void CSSParserToken::convertToPercentage()
291{
292 ASSERT(m_type == NumberToken);
293 m_type = PercentageToken;
294 m_unit = static_cast<unsigned>(CSSUnitType::CSS_PERCENTAGE);

359bool CSSParserToken::hasStringBacking() const
360{
361 switch (type()) {
362 case AtKeywordToken:
363 case DimensionToken:
364 case FunctionToken:
365 case HashToken:
366 case IdentToken:
367 case NumberToken:
368 case PercentageToken:
369 case StringToken:
370 case UrlToken:
371 return true;
372 case BadStringToken:
373 case BadUrlToken:
374 case CDCToken:
375 case CDOToken:
376 case ColonToken:
377 case ColumnToken:
378 case CommaToken:
379 case CommentToken:
380 case DashMatchToken:
381 case DelimiterToken:
382 case EOFToken:
383 case IncludeMatchToken:
384 case LeftBraceToken:
385 case LeftBracketToken:
386 case LeftParenthesisToken:
387 case PrefixMatchToken:
388 case RightBraceToken:
389 case RightBracketToken:
390 case RightParenthesisToken:
391 case SemicolonToken:
392 case SubstringMatchToken:
393 case SuffixMatchToken:
394 case WhitespaceToken:
395 return false;
396 }
397 ASSERT_NOT_REACHED();
398 return false;
399}

409bool CSSParserToken::operator==(const CSSParserToken& other) const
410{
411 if (m_type != other.m_type)
412 return false;
413 switch (m_type) {
414 case DelimiterToken:
415 return delimiter() == other.delimiter();
416 case HashToken:
417 if (m_hashTokenType != other.m_hashTokenType)
418 return false;
419 FALLTHROUGH;
420 case IdentToken:
421 case FunctionToken:
422 case StringToken:
423 case UrlToken:
424 return value() == other.value();
425 case DimensionToken:
426 if (unitString() != other.unitString())
427 return false;
428 FALLTHROUGH;
429 case NumberToken:
430 case PercentageToken:
431 return m_numericSign == other.m_numericSign && m_numericValue == other.m_numericValue && m_numericValueType == other.m_numericValueType;
432 default:
433 return true;
434 }
435}

437struct NextTokenNeedsCommentBuilder {
438 constexpr NextTokenNeedsCommentBuilder(std::initializer_list<CSSParserTokenType> tokens)
439 {
440 for (auto token : tokens)
441 buffer[token] = true;
442 }
443
444 std::array<bool, numberOfCSSParserTokenTypes> buffer { false };
445};

447void CSSParserToken::serialize(StringBuilder& builder, const CSSParserToken* nextToken) const
448{
449 // This is currently only used for @supports CSSOM. To keep our implementation
450 // simple we handle some of the edge cases incorrectly (see comments below).
451 auto appendCommentIfNeeded = [&] (const NextTokenNeedsCommentBuilder& tokensNeedingComment, auto... delimitersNeedingComment) {
452 if (!nextToken)
453 return;
454
455 CSSParserTokenType nextType = nextToken->type();
456 if (tokensNeedingComment.buffer[nextType]) {
457 builder.append("/**/");
458 return;
459 }
460
461 if (nextType == DelimiterToken && ((delimitersNeedingComment == nextToken->delimiter()) || ... || false)) {
462 builder.append("/**/");
463 return;
464 }
465 };
466
467 switch (type()) {
468 case IdentToken:
469 serializeIdentifier(value().toString(), builder);
470 appendCommentIfNeeded({ IdentToken, FunctionToken, UrlToken, BadUrlToken, NumberToken, PercentageToken, DimensionToken, CDCToken, LeftParenthesisToken }, '-');
471 break;
472 case FunctionToken:
473 serializeIdentifier(value().toString(), builder);
474 builder.append('(');
475 break;
476 case AtKeywordToken:
477 builder.append('@');
478 serializeIdentifier(value().toString(), builder);
479 appendCommentIfNeeded({ IdentToken, FunctionToken, UrlToken, BadUrlToken, NumberToken, PercentageToken, DimensionToken, CDCToken }, '-');
480 break;
481 case HashToken:
482 builder.append('#');
483 serializeIdentifier(value().toString(), builder, (getHashTokenType() == HashTokenUnrestricted));
484 appendCommentIfNeeded({ IdentToken, FunctionToken, UrlToken, BadUrlToken, NumberToken, PercentageToken, DimensionToken, CDCToken }, '-');
485 break;
486 case UrlToken:
487 builder.append("url(");
488 serializeIdentifier(value().toString(), builder);
489 builder.append(')');
490 break;
491 case DelimiterToken:
492 switch (delimiter()) {
493 case '\\':
494 builder.append("\\\n");
495 break;
496
497 case '#':
498 case '-':
499 builder.append(delimiter());
500 appendCommentIfNeeded({ IdentToken, FunctionToken, UrlToken, BadUrlToken, NumberToken, PercentageToken, DimensionToken }, '-');
501 break;
502
503 case '@':
504 builder.append('@');
505 appendCommentIfNeeded({ IdentToken, FunctionToken, UrlToken, BadUrlToken }, '-');
506 break;
507
508 case '.':
509 case '+':
510 builder.append(delimiter());
511 appendCommentIfNeeded({ NumberToken, PercentageToken, DimensionToken });
512 break;
513
514 case '/':
515 builder.append('/');
516 // Weirdly Clang errors if you try to use the fold expression in buildNextTokenNeedsCommentTable() because the true value is unused.
517 // So we just build the table by hand here instead. See: rdar://69710661
518 appendCommentIfNeeded({ }, '*');
519 break;
520
521 default:
522 builder.append(delimiter());
523 break;
524 }
525 break;
526 case NumberToken:
527 // These won't properly preserve the NumericValueType flag.
528 if (m_numericSign == PlusSign)
529 builder.append('+');
530 builder.append(numericValue());
531 appendCommentIfNeeded({ IdentToken, FunctionToken, UrlToken, BadUrlToken, NumberToken, PercentageToken, DimensionToken }, '%');
532 break;
533 case PercentageToken:
534 builder.append(numericValue(), '%');
535 break;
536 case DimensionToken:
537 // This will incorrectly serialize e.g. 4e3e2 as 4000e2.
538 builder.append(numericValue());
539 serializeIdentifier(unitString().toString(), builder);
540 appendCommentIfNeeded({ IdentToken, FunctionToken, UrlToken, BadUrlToken, NumberToken, PercentageToken, DimensionToken, CDCToken }, '-');
541 break;
542 case StringToken:
543 serializeString(value().toString(), builder);
544 break;
545
546 case IncludeMatchToken:
547 builder.append("~=");
548 break;
549 case DashMatchToken:
550 builder.append("|=");
551 break;
552 case PrefixMatchToken:
553 builder.append("^=");
554 break;
555 case SuffixMatchToken:
556 builder.append("$=");
557 break;
558 case SubstringMatchToken:
559 builder.append("*=");
560 break;
561 case ColumnToken:
562 builder.append("||");
563 break;
564 case CDOToken:
565 builder.append("<!--");
566 break;
567 case CDCToken:
568 builder.append("-->");
569 break;
570 case BadStringToken:
571 builder.append("'\n");
572 break;
573 case BadUrlToken:
574 builder.append("url(()");
575 break;
576 case WhitespaceToken:
577 builder.append(' ');
578 break;
579 case ColonToken:
580 builder.append(':');
581 break;
582 case SemicolonToken:
583 builder.append(';');
584 break;
585 case CommaToken:
586 builder.append(',');
587 break;
588 case LeftParenthesisToken:
589 builder.append('(');
590 break;
591 case RightParenthesisToken:
592 builder.append(')');
593 break;
594 case LeftBracketToken:
595 builder.append('[');
596 break;
597 case RightBracketToken:
598 builder.append(']');
599 break;
600 case LeftBraceToken:
601 builder.append('{');
602 break;
603 case RightBraceToken:
604 builder.append('}');
605 break;
606
607 case EOFToken:
608 case CommentToken:
609 ASSERT_NOT_REACHED();
610 break;
611 }
