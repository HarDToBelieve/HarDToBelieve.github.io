190// Template and inline implementations are at the bottom of the file for readability.
191
192template<typename... emptyBaseCase> inline bool identMatches(CSSValueID) { return false; }
193template<CSSValueID head, CSSValueID... tail> inline bool identMatches(CSSValueID id)
194{
195 return id == head || identMatches<tail...>(id);
196}
197
198template<CSSValueID... names> std::optional<CSSValueID> consumeIdentRaw(CSSParserTokenRange& range)
199{
200 if (range.peek().type() != IdentToken || !identMatches<names...>(range.peek().id()))
201 return std::nullopt;
202 return range.consumeIncludingWhitespace().id();
203}
204
205template<CSSValueID... names> RefPtr<CSSPrimitiveValue> consumeIdent(CSSParserTokenRange& range)
206{
207 return consumeIdentWorkerSafe<names...>(range, CSSValuePool::singleton());
208}
209
210template<CSSValueID... names> RefPtr<CSSPrimitiveValue> consumeIdentWorkerSafe(CSSParserTokenRange& range, CSSValuePool& cssValuePool)
211{
212 if (range.peek().type() != IdentToken || !identMatches<names...>(range.peek().id()))
213 return nullptr;
214 return cssValuePool.createIdentifierValue(range.consumeIncludingWhitespace().id());
215}

217inline bool isFontStyleAngleInRange(double angleInDegrees)
218{
219 return angleInDegrees > -90 && angleInDegrees < 90;
220}
