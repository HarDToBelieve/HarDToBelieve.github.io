62std::optional<CSSSelectorList> parseCSSSelector(CSSParserTokenRange range, const CSSParserContext& context, StyleSheetContents* styleSheet)
63{
64 CSSSelectorParser parser(context, styleSheet);
65 range.consumeWhitespace();
66 CSSSelectorList result = parser.consumeComplexSelectorList(range);
67 if (result.isEmpty() || !range.atEnd())
68 return { };
69 return result;
70}

78CSSSelectorList CSSSelectorParser::consumeComplexSelectorList(CSSParserTokenRange& range)
79{
80 Vector<std::unique_ptr<CSSParserSelector>> selectorList;
81 auto selector = consumeComplexSelector(range);
82 if (!selector)
83 return { };
84
85 selectorList.append(WTFMove(selector));
86 while (!range.atEnd() && range.peek().type() == CommaToken) {
87 range.consumeIncludingWhitespace();
88 selector = consumeComplexSelector(range);
89 if (!selector)
90 return { };
91 selectorList.append(WTFMove(selector));
92 }
93
94 if (m_failedParsing)
95 return { };
96
97 return CSSSelectorList { WTFMove(selectorList) };
98}
99
100CSSSelectorList CSSSelectorParser::consumeComplexForgivingSelectorList(CSSParserTokenRange& range)
101{
102 if (m_failedParsing)
103 return { };
104
105 Vector<std::unique_ptr<CSSParserSelector>> selectorList;
106
107 auto consumeForgiving = [&] {
108 auto selector = consumeComplexSelector(range);
109
110 if (m_failedParsing) {
111 selector = { };
112 m_failedParsing = false;
113 }
114 if (!range.atEnd() && range.peek().type() != CommaToken) {
115 while (!range.atEnd() && range.peek().type() != CommaToken)
116 range.consume();
117 return;
118 }
119 if (selector)
120 selectorList.append(WTFMove(selector));
121 };
122
123 consumeForgiving();
124
125 while (!range.atEnd() && range.peek().type() == CommaToken) {
126 range.consumeIncludingWhitespace();
127 consumeForgiving();
128 }
129
130 if (selectorList.isEmpty())
131 return { };
132
133 return CSSSelectorList { WTFMove(selectorList) };
134}

136bool CSSSelectorParser::supportsComplexSelector(CSSParserTokenRange range, const CSSParserContext& context)
137{
138 range.consumeWhitespace();
139 CSSSelectorParser parser(context, nullptr);
140
141 // FIXME: https://bugs.webkit.org/show_bug.cgi?id=215635
142 // Unknown css selector combinator is not addressed correctly in |CSSSelectorParser::consumeComplexSelector|.
143 auto parserSelector = parser.consumeComplexSelector(range);
144 if (parser.m_failedParsing || !range.atEnd() || !parserSelector)
145 return false;
146
147 auto complexSelector = parserSelector->releaseSelector();
148 ASSERT(complexSelector);
149
150 return !containsUnknownWebKitPseudoElements(*complexSelector);
151}

153CSSSelectorList CSSSelectorParser::consumeCompoundSelectorList(CSSParserTokenRange& range)
154{
155 Vector<std::unique_ptr<CSSParserSelector>> selectorList;
156 auto selector = consumeCompoundSelector(range);
157 range.consumeWhitespace();
158 if (!selector)
159 return CSSSelectorList();
160 selectorList.append(WTFMove(selector));
161 while (!range.atEnd() && range.peek().type() == CommaToken) {
162 range.consumeIncludingWhitespace();
163 selector = consumeCompoundSelector(range);
164 range.consumeWhitespace();
165 if (!selector)
166 return CSSSelectorList();
167 selectorList.append(WTFMove(selector));
168 }
169 return CSSSelectorList { WTFMove(selectorList) };
170}

172static bool consumeLangArgumentList(std::unique_ptr<Vector<AtomString>>& argumentList, CSSParserTokenRange& range)
173{
174 const CSSParserToken& ident = range.consumeIncludingWhitespace();
175 if (ident.type() != IdentToken && ident.type() != StringToken)
176 return false;
177 StringView string = ident.value();
178 if (string.startsWith("--"))
179 return false;
180 argumentList->append(string.toAtomString());
181 while (!range.atEnd() && range.peek().type() == CommaToken) {
182 range.consumeIncludingWhitespace();
183 const CSSParserToken& ident = range.consumeIncludingWhitespace();
184 if (ident.type() != IdentToken && ident.type() != StringToken)
185 return false;
186 StringView string = ident.value();
187 if (string.startsWith("--"))
188 return false;
189 argumentList->append(string.toAtomString());
190 }
191 return range.atEnd();
192}

194enum class CompoundSelectorFlag {
195 HasPseudoElementForRightmostCompound = 1 << 0,
196};
197
198static OptionSet<CompoundSelectorFlag> extractCompoundFlags(const CSSParserSelector& simpleSelector, CSSParserMode parserMode)
199{
200 if (simpleSelector.match() != CSSSelector::PseudoElement)
201 return { };
202
203 // FIXME: https://bugs.webkit.org/show_bug.cgi?id=161747
204 // The UASheetMode check is a work-around to allow this selector in mediaControls(New).css:
205 // input[type="range" i]::-webkit-media-slider-container > div {
206 if (parserMode == UASheetMode && simpleSelector.pseudoElementType() == CSSSelector::PseudoElementWebKitCustom)
207 return { };
208
209 return CompoundSelectorFlag::HasPseudoElementForRightmostCompound;
210}

212static bool isDescendantCombinator(CSSSelector::RelationType relation)
213{
214 return relation == CSSSelector::DescendantSpace;

216
217std::unique_ptr<CSSParserSelector> CSSSelectorParser::consumeComplexSelector(CSSParserTokenRange& range)
218{
219 auto selector = consumeCompoundSelector(range);
220 if (!selector)
221 return nullptr;
222
223 OptionSet<CompoundSelectorFlag> previousCompoundFlags;
224
225 for (CSSParserSelector* simple = selector.get(); simple && !previousCompoundFlags; simple = simple->tagHistory())
226 previousCompoundFlags = extractCompoundFlags(*simple, m_context.mode);
227
228 while (auto combinator = consumeCombinator(range)) {
229 auto nextSelector = consumeCompoundSelector(range);
230 if (!nextSelector)
231 return isDescendantCombinator(combinator) ? WTFMove(selector) : nullptr;
232 if (previousCompoundFlags.contains(CompoundSelectorFlag::HasPseudoElementForRightmostCompound))
233 return nullptr;
234 CSSParserSelector* end = nextSelector.get();
235 auto compoundFlags = extractCompoundFlags(*end, m_context.mode);
236 while (end->tagHistory()) {
237 end = end->tagHistory();
238 compoundFlags.add(extractCompoundFlags(*end, m_context.mode));
239 }
240 end->setRelation(combinator);
241 previousCompoundFlags = compoundFlags;
242 end->setTagHistory(WTFMove(selector));
243
244 selector = WTFMove(nextSelector);
245 }
246
247 return selector;
248}

288static bool isPseudoClassValidAfterPseudoElement(CSSSelector::PseudoClassType pseudoClass, CSSSelector::PseudoElementType compoundPseudoElement)
289{
290 switch (compoundPseudoElement) {
291 case CSSSelector::PseudoElementPart:
292 return !isTreeStructuralPseudoClass(pseudoClass);
293 case CSSSelector::PseudoElementResizer:
294 case CSSSelector::PseudoElementScrollbar:
295 case CSSSelector::PseudoElementScrollbarCorner:
296 case CSSSelector::PseudoElementScrollbarButton:
297 case CSSSelector::PseudoElementScrollbarThumb:
298 case CSSSelector::PseudoElementScrollbarTrack:
299 case CSSSelector::PseudoElementScrollbarTrackPiece:
300 return isScrollbarPseudoClass(pseudoClass);
301 case CSSSelector::PseudoElementSelection:
302 return pseudoClass == CSSSelector::PseudoClassWindowInactive;
303 case CSSSelector::PseudoElementWebKitCustom:
304 case CSSSelector::PseudoElementWebKitCustomLegacyPrefixed:
305 return isUserActionPseudoClass(pseudoClass);
306 default:
307 return false;
308 }

311static bool isSimpleSelectorValidAfterPseudoElement(const CSSParserSelector& simpleSelector, CSSSelector::PseudoElementType compoundPseudoElement)
312{
313 if (compoundPseudoElement == CSSSelector::PseudoElementUnknown)
314 return true;
315 if (compoundPseudoElement == CSSSelector::PseudoElementPart) {
316 if (simpleSelector.match() == CSSSelector::PseudoElement && simpleSelector.pseudoElementType() != CSSSelector::PseudoElementPart)
317 return true;
318 }
319 if (simpleSelector.match() != CSSSelector::PseudoClass)
320 return false;
321 CSSSelector::PseudoClassType pseudo = simpleSelector.pseudoClassType();
322 if (pseudo == CSSSelector::PseudoClassNot) {
323 ASSERT(simpleSelector.selectorList());
324 ASSERT(simpleSelector.selectorList()->first());
325 pseudo = simpleSelector.selectorList()->first()->pseudoClassType();
326 }
327 return isPseudoClassValidAfterPseudoElement(pseudo, compoundPseudoElement);
328}

336std::unique_ptr<CSSParserSelector> CSSSelectorParser::consumeCompoundSelector(CSSParserTokenRange& range)
337{
338 std::unique_ptr<CSSParserSelector> compoundSelector;
339
340 AtomString namespacePrefix;
341 AtomString elementName;
342 CSSSelector::PseudoElementType compoundPseudoElement = CSSSelector::PseudoElementUnknown;
343 const bool hasName = consumeName(range, elementName, namespacePrefix);
344 if (!hasName) {
345 compoundSelector = consumeSimpleSelector(range);
346 if (!compoundSelector)
347 return nullptr;
348 if (compoundSelector->match() == CSSSelector::PseudoElement)
349 compoundPseudoElement = compoundSelector->pseudoElementType();
350 }
351
352 while (auto simpleSelector = consumeSimpleSelector(range)) {
353 // FIXME: https://bugs.webkit.org/show_bug.cgi?id=161747
354 // The UASheetMode check is a work-around to allow this selector in mediaControls(New).css:
355 // video::-webkit-media-text-track-region-container.scrolling
356 if (m_context.mode != UASheetMode && !isSimpleSelectorValidAfterPseudoElement(*simpleSelector.get(), compoundPseudoElement)) {
357 m_failedParsing = true;
358 return nullptr;
359 }
360 if (simpleSelector->match() == CSSSelector::PseudoElement)
361 compoundPseudoElement = simpleSelector->pseudoElementType();
362
363 if (compoundSelector)
364 compoundSelector->appendTagHistory(CSSSelector::Subselector, WTFMove(simpleSelector));
365 else
366 compoundSelector = WTFMove(simpleSelector);
367 }
368
369
370 // While inside a nested selector like :is(), the default namespace shall be ignored when [1]:
371 // * The compound selector represents the subject [2], and
372 // * The compound selector does not contain a type/universal selector.
373 //
374 // [1] https://drafts.csswg.org/selectors/#matches
375 // [2] https://drafts.csswg.org/selectors/#selector-subject
376 SetForScope<bool> ignoreDefaultNamespace(m_ignoreDefaultNamespace, m_resistDefaultNamespace && !hasName && atEndIgnoringWhitespace(range));
377 if (!compoundSelector) {
378 AtomString namespaceURI = determineNamespace(namespacePrefix);
379 if (namespaceURI.isNull()) {
380 m_failedParsing = true;
381 return nullptr;
382 }
383 if (namespaceURI == defaultNamespace())
384 namespacePrefix = nullAtom();
385
386 return makeUnique<CSSParserSelector>(QualifiedName(namespacePrefix, elementName, namespaceURI));
387 }
388 prependTypeSelectorIfNeeded(namespacePrefix, elementName, *compoundSelector);
389 return splitCompoundAtImplicitShadowCrossingCombinator(WTFMove(compoundSelector), m_context);
390}

392std::unique_ptr<CSSParserSelector> CSSSelectorParser::consumeSimpleSelector(CSSParserTokenRange& range)
393{
394 const CSSParserToken& token = range.peek();
395 std::unique_ptr<CSSParserSelector> selector;
396 if (token.type() == HashToken)
397 selector = consumeId(range);
398 else if (token.type() == DelimiterToken && token.delimiter() == '.')
399 selector = consumeClass(range);
400 else if (token.type() == LeftBracketToken)
401 selector = consumeAttribute(range);
402 else if (token.type() == ColonToken)
403 selector = consumePseudo(range);
404 else
405 return nullptr;
406 if (!selector)
407 m_failedParsing = true;
408 return selector;
409}

411bool CSSSelectorParser::consumeName(CSSParserTokenRange& range, AtomString& name, AtomString& namespacePrefix)
412{
413 name = nullAtom();
414 namespacePrefix = nullAtom();
415
416 const CSSParserToken& firstToken = range.peek();
417 if (firstToken.type() == IdentToken) {
418 name = firstToken.value().toAtomString();
419 range.consume();
420 } else if (firstToken.type() == DelimiterToken && firstToken.delimiter() == '*') {
421 name = starAtom();
422 range.consume();
423 } else if (firstToken.type() == DelimiterToken && firstToken.delimiter() == '|') {
424 // This is an empty namespace, which'll get assigned this value below
425 name = emptyAtom();
426 } else
427 return false;
428
429 if (range.peek().type() != DelimiterToken || range.peek().delimiter() != '|')
430 return true;
431 range.consume();
432
433 namespacePrefix = name;
434 const CSSParserToken& nameToken = range.consume();
435 if (nameToken.type() == IdentToken) {
436 name = nameToken.value().toAtomString();
437 } else if (nameToken.type() == DelimiterToken && nameToken.delimiter() == '*')
438 name = starAtom();
439 else {
440 name = nullAtom();
441 namespacePrefix = nullAtom();
442 return false;
443 }
444
445 return true;
446}

448std::unique_ptr<CSSParserSelector> CSSSelectorParser::consumeId(CSSParserTokenRange& range)
449{
450 ASSERT(range.peek().type() == HashToken);
451 if (range.peek().getHashTokenType() != HashTokenId)
452 return nullptr;
453
454 auto selector = makeUnique<CSSParserSelector>();
455 selector->setMatch(CSSSelector::Id);
456
457 // FIXME-NEWPARSER: Avoid having to do this, but the old parser does and we need
458 // to be compatible for now.
459 CSSParserToken token = range.consume();
460 selector->setValue(token.value().toAtomString(), m_context.mode == HTMLQuirksMode);
461 return selector;
462}

483std::unique_ptr<CSSParserSelector> CSSSelectorParser::consumeAttribute(CSSParserTokenRange& range)
484{
485 ASSERT(range.peek().type() == LeftBracketToken);
486 CSSParserTokenRange block = range.consumeBlock();
487 block.consumeWhitespace();
488
489 AtomString namespacePrefix;
490 AtomString attributeName;
491 if (!consumeName(block, attributeName, namespacePrefix))
492 return nullptr;
493 block.consumeWhitespace();
494
495 AtomString namespaceURI = determineNamespace(namespacePrefix);
496 if (namespaceURI.isNull())
497 return nullptr;
498
499 QualifiedName qualifiedName = namespacePrefix.isNull()
500 ? QualifiedName(nullAtom(), attributeName, nullAtom())
501 : QualifiedName(namespacePrefix, attributeName, namespaceURI);
502
503 auto selector = makeUnique<CSSParserSelector>();
504
505 if (block.atEnd()) {
506 selector->setAttribute(qualifiedName, m_context.isHTMLDocument, CSSSelector::CaseSensitive);
507 selector->setMatch(CSSSelector::Set);
508 return selector;
509 }
510
511 selector->setMatch(consumeAttributeMatch(block));
512
513 const CSSParserToken& attributeValue = block.consumeIncludingWhitespace();
514 if (attributeValue.type() != IdentToken && attributeValue.type() != StringToken)
515 return nullptr;
516 selector->setValue(attributeValue.value().toAtomString());
517
518 selector->setAttribute(qualifiedName, m_context.isHTMLDocument, consumeAttributeFlags(block));
519
520 if (!block.atEnd())
521 return nullptr;
522 return selector;
523}

562std::unique_ptr<CSSParserSelector> CSSSelectorParser::consumePseudo(CSSParserTokenRange& range)
563{
564 ASSERT(range.peek().type() == ColonToken);
565 range.consume();
566
567 int colons = 1;
568 if (range.peek().type() == ColonToken) {
569 range.consume();
570 colons++;
571 }
572
573 const CSSParserToken& token = range.peek();
574 if (token.type() != IdentToken && token.type() != FunctionToken)
575 return nullptr;
576
577 std::unique_ptr<CSSParserSelector> selector;
578
579 if (colons == 1) {
580 selector = CSSParserSelector::parsePseudoClassSelector(token.value());
581 if (!selector)
582 return nullptr;
583 if (selector->match() == CSSSelector::PseudoClass) {
584 if (m_context.mode != UASheetMode && selector->pseudoClassType() == CSSSelector::PseudoClassDirectFocus)
585 return nullptr;
586 if (m_context.mode != UASheetMode && selector->pseudoClassType() == CSSSelector::PseudoClassModalDialog)
587 return nullptr;
588 if (!m_context.focusVisibleEnabled && selector->pseudoClassType() == CSSSelector::PseudoClassFocusVisible)
589 return nullptr;
590#if ENABLE(ATTACHMENT_ELEMENT)
591 if (!m_context.attachmentEnabled && selector->pseudoClassType() == CSSSelector::PseudoClassHasAttachment)
592 return nullptr;
593#endif
594 }
595 } else {
596 selector = CSSParserSelector::parsePseudoElementSelector(token.value());
597#if ENABLE(VIDEO)
598 // Treat the ident version of cue as PseudoElementWebkitCustom.
599 if (token.type() == IdentToken && selector && selector->match() == CSSSelector::PseudoElement && selector->pseudoElementType() == CSSSelector::PseudoElementCue)
600 selector->setPseudoElementType(CSSSelector::PseudoElementWebKitCustom);
601#endif
602 }
603
604 if (!selector || (selector->match() == CSSSelector::PseudoElement && m_disallowPseudoElements))
605 return nullptr;
606
607 if (token.type() == IdentToken) {
608 range.consume();
609 if ((selector->match() == CSSSelector::PseudoElement && (selector->pseudoElementType() == CSSSelector::PseudoElementUnknown || isOnlyPseudoElementFunction(selector->pseudoElementType())))
610 || (selector->match() == CSSSelector::PseudoClass && (selector->pseudoClassType() == CSSSelector::PseudoClassUnknown || isOnlyPseudoClassFunction(selector->pseudoClassType()))))
611 return nullptr;
612 return selector;
613 }
614
615 CSSParserTokenRange block = range.consumeBlock();
616 block.consumeWhitespace();
617 if (token.type() != FunctionToken)
618 return nullptr;
619
620 if (selector->match() == CSSSelector::PseudoClass) {
621 switch (selector->pseudoClassType()) {
622 case CSSSelector::PseudoClassNot: {
623 SetForScope<bool> resistDefaultNamespace(m_resistDefaultNamespace, true);
624 DisallowPseudoElementsScope scope(*this);
625 auto selectorList = makeUnique<CSSSelectorList>();
626 *selectorList = consumeComplexSelectorList(block);
627 if (!selectorList->first() || !block.atEnd())
628 return nullptr;
629 selector->setSelectorList(WTFMove(selectorList));
630 return selector;
631 }
632 case CSSSelector::PseudoClassNthChild:
633 case CSSSelector::PseudoClassNthLastChild:
634 case CSSSelector::PseudoClassNthOfType:
635 case CSSSelector::PseudoClassNthLastOfType: {
636 std::pair<int, int> ab;
637 if (!consumeANPlusB(block, ab))
638 return nullptr;
639 block.consumeWhitespace();
640 // FIXME: We should be able to do this lazily. See: https://bugs.webkit.org/show_bug.cgi?id=217149
641 selector->setArgument(serializeANPlusB(ab));
642 if (!block.atEnd()) {
643 if (block.peek().type() != IdentToken)
644 return nullptr;
645 const CSSParserToken& ident = block.consume();
646 if (!equalIgnoringASCIICase(ident.value(), "of"))
647 return nullptr;
648 if (block.peek().type() != WhitespaceToken)
649 return nullptr;
650 DisallowPseudoElementsScope scope(*this);
651 block.consumeWhitespace();
652 auto selectorList = makeUnique<CSSSelectorList>();
653 *selectorList = consumeComplexSelectorList(block);
654 if (selectorList->isEmpty() || !block.atEnd())
655 return nullptr;
656 selector->setSelectorList(WTFMove(selectorList));
657 }
658 selector->setNth(ab.first, ab.second);
659 return selector;
660 }
661 case CSSSelector::PseudoClassLang: {
662 // FIXME: CSS Selectors Level 4 allows :lang(*-foo)
663 auto argumentList = makeUnique<Vector<AtomString>>();
664 if (!consumeLangArgumentList(argumentList, block))
665 return nullptr;
666 selector->setArgumentList(WTFMove(argumentList));
667 return selector;
668 }
669 case CSSSelector::PseudoClassIs:
670 case CSSSelector::PseudoClassMatches:
671 case CSSSelector::PseudoClassWhere: {
672 SetForScope<bool> resistDefaultNamespace(m_resistDefaultNamespace, true);
673 DisallowPseudoElementsScope scope(*this);
674 auto selectorList = makeUnique<CSSSelectorList>();
675 *selectorList = consumeComplexForgivingSelectorList(block);
676 if (!block.atEnd())
677 return nullptr;
678 selector->setSelectorList(WTFMove(selectorList));
679 return selector;
680 }
681 case CSSSelector::PseudoClassAny:
682 case CSSSelector::PseudoClassHost: {
683 auto selectorList = makeUnique<CSSSelectorList>();
684 *selectorList = consumeCompoundSelectorList(block);
685 if (selectorList->isEmpty() || !block.atEnd())
686 return nullptr;
687 selector->setSelectorList(WTFMove(selectorList));
688 return selector;
689 }
690#if ENABLE(CSS_SELECTORS_LEVEL4)
691 case CSSSelector::PseudoClassDir:
692 case CSSSelector::PseudoClassRole: {
693 const CSSParserToken& ident = block.consumeIncludingWhitespace();
694 if (ident.type() != IdentToken || !block.atEnd())
695 return nullptr;
696 selector->setArgument(ident.value().toAtomString());
697 return selector;
698 }
699#endif
700 default:
701 break;
702 }
703 }
704
705 if (selector->match() == CSSSelector::PseudoElement) {
706 switch (selector->pseudoElementType()) {
707#if ENABLE(VIDEO)
708 case CSSSelector::PseudoElementCue: {
709 DisallowPseudoElementsScope scope(*this);
710 auto selectorList = makeUnique<CSSSelectorList>();
711 *selectorList = consumeCompoundSelectorList(block);
712 if (selectorList->isEmpty() || !block.atEnd())
713 return nullptr;
714 selector->setSelectorList(WTFMove(selectorList));
715 return selector;
716 }
717#endif
718 case CSSSelector::PseudoElementHighlight: {
719 DisallowPseudoElementsScope scope(*this);
720
721 auto& ident = block.consumeIncludingWhitespace();
722 if (ident.type() != IdentToken || !block.atEnd())
723 return nullptr;
724
725 auto argumentList = makeUnique<Vector<AtomString>>();
726 argumentList->append(ident.value().toAtomString());
727 selector->setArgumentList(WTFMove(argumentList));
728
729 return selector;
730 }
731 case CSSSelector::PseudoElementPart: {
732 auto argumentList = makeUnique<Vector<AtomString>>();
733 do {
734 auto& ident = block.consumeIncludingWhitespace();
735 if (ident.type() != IdentToken)
736 return nullptr;
737 argumentList->append(ident.value().toAtomString());
738 } while (!block.atEnd());
739
740 selector->setArgumentList(WTFMove(argumentList));
741 return selector;
742 }
743 case CSSSelector::PseudoElementSlotted: {
744 DisallowPseudoElementsScope scope(*this);
745
746 auto innerSelector = consumeCompoundSelector(block);
747 block.consumeWhitespace();
748 if (!innerSelector || !block.atEnd())
749 return nullptr;
750 selector->adoptSelectorVector(Vector<std::unique_ptr<CSSParserSelector>>::from(WTFMove(innerSelector)));
751 return selector;
752 }
753 default:
754 break;
755 }
756 }
757
758 return nullptr;
759}

761CSSSelector::RelationType CSSSelectorParser::consumeCombinator(CSSParserTokenRange& range)
762{
763 auto fallbackResult = CSSSelector::Subselector;
764 while (range.peek().type() == WhitespaceToken) {
765 range.consume();
766 fallbackResult = CSSSelector::DescendantSpace;
767 }
768
769 if (range.peek().type() != DelimiterToken)
770 return fallbackResult;
771
772 UChar delimiter = range.peek().delimiter();
773
774 if (delimiter == '+' || delimiter == '~' || delimiter == '>') {
775 range.consumeIncludingWhitespace();
776 if (delimiter == '+')
777 return CSSSelector::DirectAdjacent;
778 if (delimiter == '~')
779 return CSSSelector::IndirectAdjacent;
780 return CSSSelector::Child;
781 }
782
783 return fallbackResult;
784}

786CSSSelector::Match CSSSelectorParser::consumeAttributeMatch(CSSParserTokenRange& range)
787{
788 const CSSParserToken& token = range.consumeIncludingWhitespace();
789 switch (token.type()) {
790 case IncludeMatchToken:
791 return CSSSelector::List;
792 case DashMatchToken:
793 return CSSSelector::Hyphen;
794 case PrefixMatchToken:
795 return CSSSelector::Begin;
796 case SuffixMatchToken:
797 return CSSSelector::End;
798 case SubstringMatchToken:
799 return CSSSelector::Contain;
800 case DelimiterToken:
801 if (token.delimiter() == '=')
802 return CSSSelector::Exact;
803 FALLTHROUGH;
804 default:
805 m_failedParsing = true;
806 return CSSSelector::Exact;
807 }

821// <an+b> token sequences have special serialization rules: https://www.w3.org/TR/css-syntax-3/#serializing-anb
822static AtomString serializeANPlusB(const std::pair<int, int>& ab)
823{
824 if (!ab.first)
825 return AtomString::number(ab.second);
826
827 StringBuilder builder;
828
829 if (ab.first == -1)
830 builder.append('-');
831 else if (ab.first != 1)
832 builder.append(ab.first);
833 builder.append('n');
834
835 if (ab.second) {
836 if (ab.second > 0)
837 builder.append('+');
838 builder.append(ab.second);
839 }
840
841 return builder.toAtomString();
842}
843
844static bool consumeANPlusB(CSSParserTokenRange& range, std::pair<int, int>& result)
845{
846 const CSSParserToken& token = range.consume();
847 if (token.type() == NumberToken && token.numericValueType() == IntegerValueType) {
848 result = std::make_pair(0, static_cast<int>(token.numericValue()));
849 return true;
850 }
851 if (token.type() == IdentToken) {
852 if (equalIgnoringASCIICase(token.value(), "odd")) {
853 result = std::make_pair(2, 1);
854 return true;
855 }
856 if (equalIgnoringASCIICase(token.value(), "even")) {
857 result = std::make_pair(2, 0);
858 return true;
859 }
860 }
861
862 // The 'n' will end up as part of an ident or dimension. For a valid <an+b>,
863 // this will store a string of the form 'n', 'n-', or 'n-123'.
864 String nString;
865
866 if (token.type() == DelimiterToken && token.delimiter() == '+' && range.peek().type() == IdentToken) {
867 result.first = 1;
868 nString = range.consume().value().toString();
869 } else if (token.type() == DimensionToken && token.numericValueType() == IntegerValueType) {
870 result.first = token.numericValue();
871 nString = token.unitString().toString();
872 } else if (token.type() == IdentToken) {
873 if (token.value()[0] == '-') {
874 result.first = -1;
875 nString = token.value().substring(1).toString();
876 } else {
877 result.first = 1;
878 nString = token.value().toString();
879 }
880 }
881
882 range.consumeWhitespace();
883
884 if (nString.isEmpty() || !isASCIIAlphaCaselessEqual(nString[0], 'n'))
885 return false;
886 if (nString.length() > 1 && nString[1] != '-')
887 return false;
888
889 if (nString.length() > 2) {
890 auto parsedNumber = parseInteger<int>(StringView { nString }.substring(1));
891 result.second = parsedNumber.value_or(0);
892 return parsedNumber.has_value();
893 }
894
895 NumericSign sign = nString.length() == 1 ? NoSign : MinusSign;
896 if (sign == NoSign && range.peek().type() == DelimiterToken) {
897 char delimiterSign = range.consumeIncludingWhitespace().delimiter();
898 if (delimiterSign == '+')
899 sign = PlusSign;
900 else if (delimiterSign == '-')
901 sign = MinusSign;
902 else
903 return false;
904 }
905
906 if (sign == NoSign && range.peek().type() != NumberToken) {
907 result.second = 0;
908 return true;
909 }
910
911 const CSSParserToken& b = range.consume();
912 if (b.type() != NumberToken || b.numericValueType() != IntegerValueType)
913 return false;
914 if ((b.numericSign() == NoSign) == (sign == NoSign))
915 return false;
916 result.second = b.numericValue();
917 if (sign == MinusSign)
918 result.second = -result.second;
919 return true;
920}

942void CSSSelectorParser::prependTypeSelectorIfNeeded(const AtomString& namespacePrefix, const AtomString& elementName, CSSParserSelector& compoundSelector)
943{
944 bool isShadowDOM = compoundSelector.needsImplicitShadowCombinatorForMatching();
945
946 if (elementName.isNull() && defaultNamespace() == starAtom() && !isShadowDOM)
947 return;
948
949 AtomString determinedElementName = elementName.isNull() ? starAtom() : elementName;
950 AtomString namespaceURI = determineNamespace(namespacePrefix);
951 if (namespaceURI.isNull()) {
952 m_failedParsing = true;
953 return;
954 }
955 AtomString determinedPrefix = namespacePrefix;
956 if (namespaceURI == defaultNamespace())
957 determinedPrefix = nullAtom();
958 QualifiedName tag(determinedPrefix, determinedElementName, namespaceURI);
959
960 // *:host never matches, so we can't discard the *,
961 // otherwise we can't tell the difference between *:host and just :host.
962 //
963 // Also, selectors where we use a ShadowPseudo combinator between the
964 // element and the pseudo element for matching (custom pseudo elements,
965 // ::cue), we need a universal selector to set the combinator
966 // (relation) on in the cases where there are no simple selectors preceding
967 // the pseudo element.
968 bool explicitForHost = compoundSelector.isHostPseudoSelector() && !elementName.isNull();
969 if (tag != anyQName() || explicitForHost || isShadowDOM)
970 compoundSelector.prependTagSelector(tag, determinedPrefix == nullAtom() && determinedElementName == starAtom() && !explicitForHost);
971}

973std::unique_ptr<CSSParserSelector> CSSSelectorParser::splitCompoundAtImplicitShadowCrossingCombinator(std::unique_ptr<CSSParserSelector> compoundSelector, const CSSParserContext& context)
974{
975 // The tagHistory is a linked list that stores combinator separated compound selectors
976 // from right-to-left. Yet, within a single compound selector, stores the simple selectors
977 // from left-to-right.
978 //
979 // ".a.b > div#id" is stored in a tagHistory as [div, #id, .a, .b], each element in the
980 // list stored with an associated relation (combinator or Subselector).
981 //
982 // ::cue, ::shadow, and custom pseudo elements have an implicit ShadowPseudo combinator
983 // to their left, which really makes for a new compound selector, yet it's consumed by
984 // the selector parser as a single compound selector.
985 //
986 // Example: input#x::-webkit-clear-button -> [ ::-webkit-clear-button, input, #x ]
987 //
988 CSSParserSelector* splitAfter = compoundSelector.get();
989 while (splitAfter->tagHistory() && !splitAfter->tagHistory()->needsImplicitShadowCombinatorForMatching())
990 splitAfter = splitAfter->tagHistory();
991
992 if (!splitAfter || !splitAfter->tagHistory())
993 return compoundSelector;
994
995 // ::part() combines with other pseudo elements.
996 bool isPart = splitAfter->tagHistory()->match() == CSSSelector::PseudoElement && splitAfter->tagHistory()->pseudoElementType() == CSSSelector::PseudoElementPart;
997
998 std::unique_ptr<CSSParserSelector> secondCompound;
999 if (context.mode == UASheetMode || isPart) {
1000 // FIXME: https://bugs.webkit.org/show_bug.cgi?id=161747
1001 // We have to recur, since we have rules in media controls like video::a::b. This should not be allowed, and
1002 // we should remove this recursion once those rules are gone.
1003 secondCompound = splitCompoundAtImplicitShadowCrossingCombinator(splitAfter->releaseTagHistory(), context);
1004 } else
1005 secondCompound = splitAfter->releaseTagHistory();
1006
1007 secondCompound->appendTagHistory(CSSSelector::ShadowDescendant, WTFMove(compoundSelector));
1008 return secondCompound;
1009}
