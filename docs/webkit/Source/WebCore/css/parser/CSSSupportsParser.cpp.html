38CSSSupportsParser::SupportsResult CSSSupportsParser::supportsCondition(CSSParserTokenRange range, CSSParserImpl& parser, SupportsParsingMode mode)
39{
40 // FIXME: The spec allows leading whitespace in @supports but not CSS.supports,
41 // but major browser vendors allow it in CSS.supports also.
42 range.consumeWhitespace();
43 CSSSupportsParser supportsParser(parser);
44
45 auto result = supportsParser.consumeCondition(range);
46 if (mode != ForWindowCSS || result != Invalid)
47 return result;
48
49 // window.CSS.supports requires parsing as-if the condition was wrapped in
50 // parenthesis. The only productions that wouldn't have parsed above are the
51 // declaration condition or the general enclosed productions.
52 return supportsParser.consumeSupportsFeatureOrGeneralEnclosed(range);
53}
54
55enum ClauseType { Unresolved, Conjunction, Disjunction };
56
57CSSSupportsParser::SupportsResult CSSSupportsParser::consumeCondition(CSSParserTokenRange range)
58{
59 if (range.peek().type() == IdentToken || range.peek().type() == FunctionToken) {
60 if (equalIgnoringASCIICase(range.peek().value(), "not"))
61 return consumeNegation(range);
62 }
63
64 bool result = false;
65 ClauseType clauseType = Unresolved;
66
67 auto previousTokenType = IdentToken;
68
69 while (true) {
70 auto nextResult = consumeConditionInParenthesis(range, previousTokenType);
71 if (nextResult == Invalid)
72 return Invalid;
73 bool nextSupported = nextResult;
74 if (clauseType == Unresolved)
75 result = nextSupported;
76 else if (clauseType == Conjunction)
77 result &= nextSupported;
78 else
79 result |= nextSupported;
80
81 if (range.atEnd())
82 break;
83 range.consumeWhitespace();
84 if (range.atEnd())
85 break;
86
87 const CSSParserToken& token = range.peek();
88 if (token.type() != IdentToken && token.type() != FunctionToken)
89 return Invalid;
90
91 previousTokenType = token.type();
92
93 if (clauseType == Unresolved)
94 clauseType = token.value().length() == 3 ? Conjunction : Disjunction;
95 if ((clauseType == Conjunction && !equalIgnoringASCIICase(token.value(), "and"))
96 || (clauseType == Disjunction && !equalIgnoringASCIICase(token.value(), "or")))
97 return Invalid;
98
99 if (token.type() == IdentToken)
100 range.consumeIncludingWhitespace();
101 }
102 return result ? Supported : Unsupported;
103}

105CSSSupportsParser::SupportsResult CSSSupportsParser::consumeNegation(CSSParserTokenRange range)
106{
107 ASSERT(range.peek().type() == IdentToken || range.peek().type() == FunctionToken);
108 auto tokenType = range.peek().type();
109
110 if (range.peek().type() == IdentToken)
111 range.consumeIncludingWhitespace();
112 auto result = consumeConditionInParenthesis(range, tokenType);
113 range.consumeWhitespace();
114 if (!range.atEnd() || result == Invalid)
115 return Invalid;
116
117 return result ? Unsupported : Supported;
118}

120CSSSupportsParser::SupportsResult CSSSupportsParser::consumeSupportsFeatureOrGeneralEnclosed(CSSParserTokenRange& range)
121{
122 if (range.peek().type() == FunctionToken) {
123 if (range.peek().functionId() == CSSValueSelector)
124 return consumeSupportsSelectorFunction(range);
125
126 range.consumeComponentValue();
127 return Unsupported;
128 }
129
130 return range.peek().type() == IdentToken && m_parser.supportsDeclaration(range) ? Supported : Unsupported;
131}
132
133CSSSupportsParser::SupportsResult CSSSupportsParser::consumeSupportsSelectorFunction(CSSParserTokenRange& range)
134{
135 if (range.peek().type() != FunctionToken || range.peek().functionId() != CSSValueSelector)
136 return Invalid;
137
138 auto block = range.consumeBlock();
139 block.consumeWhitespace();
140
141 return CSSSelectorParser::supportsComplexSelector(block, m_parser.context()) ? Supported : Unsupported;
142}
143
144CSSSupportsParser::SupportsResult CSSSupportsParser::consumeConditionInParenthesis(CSSParserTokenRange& range, CSSParserTokenType startTokenType)
145{
146 // <supports-in-parens> = ( <supports-condition> ) | <supports-feature> | <general-enclosed>
147 if (startTokenType == IdentToken && range.peek().type() != LeftParenthesisToken) {
148 if (range.peek().type() == FunctionToken && range.peek().functionId() == CSSValueSelector)
149 return consumeSupportsSelectorFunction(range);
150
151 return Invalid;
152 }
153
154 auto innerRange = range.consumeBlock();
155 innerRange.consumeWhitespace();
156
157 auto result = consumeCondition(innerRange);
158 if (result != Invalid)
159 return result;
160
161 // <supports-feature> = <supports-selector-fn> | <supports-decl>
162 // <general-enclosed>
163 return consumeSupportsFeatureOrGeneralEnclosed(innerRange);
164}
