37CSSTokenizerInputStream::CSSTokenizerInputStream(const String& input)
38 : m_offset(0)
39 , m_stringLength(input.length())
40 , m_string(input.impl())
41{
42}
43
44void CSSTokenizerInputStream::advanceUntilNonWhitespace()
45{
46 // Using HTML space here rather than CSS space since we don't do preprocessing
47 if (m_string->is8Bit()) {
48 const LChar* characters = m_string->characters8();
49 while (m_offset < m_stringLength && isHTMLSpace(characters[m_offset]))
50 ++m_offset;
51 } else {
52 const UChar* characters = m_string->characters16();
53 while (m_offset < m_stringLength && isHTMLSpace(characters[m_offset]))
54 ++m_offset;
55 }
56}
57
58double CSSTokenizerInputStream::getDouble(unsigned start, unsigned end) const
59{
60 ASSERT(start <= end && ((m_offset + end) <= m_stringLength));
61 bool isResultOK = false;
62 double result = 0.0;
63 if (start < end) {
64 if (m_string->is8Bit())
65 result = charactersToDouble(m_string->characters8() + m_offset + start, end - start, &isResultOK);
66 else
67 result = charactersToDouble(m_string->characters16() + m_offset + start, end - start, &isResultOK);
68 }
69 // FIXME: It looks like callers ensure we have a valid number
70 return isResultOK ? result : 0.0;
71}
