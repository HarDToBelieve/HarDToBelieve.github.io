30#pragma once
31
32#include <wtf/text/StringView.h>
33
34namespace WebCore {
35
36constexpr LChar kEndOfFileMarker = 0;
37
38class CSSTokenizerInputStream {
39 WTF_MAKE_NONCOPYABLE(CSSTokenizerInputStream);
40 WTF_MAKE_FAST_ALLOCATED;
41public:
42 explicit CSSTokenizerInputStream(const String& input);
43
44 // Gets the char in the stream replacing NUL characters with a unicode
45 // replacement character. Will return (NUL) kEndOfFileMarker when at the
46 // end of the stream.
47 UChar nextInputChar() const
48 {
49 if (m_offset >= m_stringLength)
50 return kEndOfFileMarker;
51 return (*m_string)[m_offset];
52 }
53
54 // Gets the char at lookaheadOffset from the current stream position. Will
55 // return NUL (kEndOfFileMarker) if the stream position is at the end.
56 UChar peek(unsigned lookaheadOffset) const
57 {
58 if ((m_offset + lookaheadOffset) >= m_stringLength)
59 return kEndOfFileMarker;
60 return (*m_string)[m_offset + lookaheadOffset];
61 }
62
63 void advance(unsigned offset = 1) { m_offset += offset; }
64 void pushBack(UChar cc)
65 {
66 --m_offset;
67 ASSERT_UNUSED(cc, nextInputChar() == cc);
68 }
69
70 double getDouble(unsigned start, unsigned end) const;
71
72 template<bool characterPredicate(UChar)>
73 unsigned skipWhilePredicate(unsigned offset)
74 {
75 if (m_string->is8Bit()) {
76 const LChar* characters8 = m_string->characters8();
77 while ((m_offset + offset) < m_stringLength && characterPredicate(characters8[m_offset + offset]))
78 ++offset;
79 } else {
80 const UChar* characters16 = m_string->characters16();
81 while ((m_offset + offset) < m_stringLength && characterPredicate(characters16[m_offset + offset]))
82 ++offset;
83 }
84 return offset;
85 }
86
87 void advanceUntilNonWhitespace();
88
89 unsigned length() const { return m_stringLength; }
90 unsigned offset() const { return std::min(m_offset, m_stringLength); }
91
92 StringView rangeAt(unsigned start, unsigned length) const
93 {
94 ASSERT(start + length <= m_stringLength);
95 return StringView(m_string.get()).substring(start, length); // FIXME: Should make a constructor on StringView for this.
96 }
97
98private:
99 size_t m_offset;
100 const size_t m_stringLength;
101 RefPtr<StringImpl> m_string;
102};
103
104} // namespace WebCore
