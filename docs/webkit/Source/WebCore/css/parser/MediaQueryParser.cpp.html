40RefPtr<MediaQuerySet> MediaQueryParser::parseMediaQuerySet(const String& queryString, MediaQueryParserContext context)
41{
42 auto tokenizer = CSSTokenizer::tryCreate(queryString);
43 if (UNLIKELY(!tokenizer))
44 return nullptr;
45 return parseMediaQuerySet(tokenizer->tokenRange(), context);
46}

71MediaQueryParser::MediaQueryParser(ParserType parserType, MediaQueryParserContext context)
72 : m_parserType(parserType)
73 , m_mediaQueryData(context)
74 , m_querySet(MediaQuerySet::create())
75
76{
77 if (parserType == MediaQuerySetParser)
78 m_state = &MediaQueryParser::readRestrictor;
79 else // MediaConditionParser
80 m_state = &MediaQueryParser::readMediaNot;
81}

91// State machine member functions start here
92void MediaQueryParser::readRestrictor(CSSParserTokenType type, const CSSParserToken& token, CSSParserTokenRange& range)
93{
94 readMediaType(type, token, range);
95}
96
97void MediaQueryParser::readMediaNot(CSSParserTokenType type, const CSSParserToken& token, CSSParserTokenRange& range)
98{
99 if (type == IdentToken && equalIgnoringASCIICase(token.value(), "not"))
100 setStateAndRestrict(ReadFeatureStart, MediaQuery::Not);
101 else
102 readFeatureStart(type, token, range);
103}
104
105static bool isRestrictorOrLogicalOperator(const CSSParserToken& token)
106{
107 // FIXME: it would be more efficient to use lower-case always for tokenValue.
108 return equalIgnoringASCIICase(token.value(), "not")
109 || equalIgnoringASCIICase(token.value(), "and")
110 || equalIgnoringASCIICase(token.value(), "or")
111 || equalIgnoringASCIICase(token.value(), "only");
112}
113
114void MediaQueryParser::readMediaType(CSSParserTokenType type, const CSSParserToken& token, CSSParserTokenRange& range)
115{
116 if (type == LeftParenthesisToken) {
117 if (m_mediaQueryData.restrictor() != MediaQuery::None)
118 m_state = SkipUntilComma;
119 else
120 m_state = ReadFeature;
121 } else if (type == IdentToken) {
122 if (m_state == ReadRestrictor && equalIgnoringASCIICase(token.value(), "not"))
123 setStateAndRestrict(ReadMediaType, MediaQuery::Not);
124 else if (m_state == ReadRestrictor && equalIgnoringASCIICase(token.value(), "only"))
125 setStateAndRestrict(ReadMediaType, MediaQuery::Only);
126 else if (m_mediaQueryData.restrictor() != MediaQuery::None
127 && isRestrictorOrLogicalOperator(token)) {
128 m_state = SkipUntilComma;
129 } else {
130 m_mediaQueryData.setMediaType(token.value().toString());
131 m_state = ReadAnd;
132 }
133 } else if (type == EOFToken && (!m_querySet->queryVector().size() || m_state != ReadRestrictor))
134 m_state = Done;
135 else {
136 m_state = SkipUntilComma;
137 if (type == CommaToken)
138 skipUntilComma(type, token, range);
139 }
140}

142void MediaQueryParser::commitMediaQuery()
143{
144 // FIXME-NEWPARSER: Convoluted and awful, but we can't change the MediaQuerySet yet because of the
145 // old parser.
146 static const NeverDestroyed<String> defaultMediaType { "all"_s };
147 MediaQuery mediaQuery { m_mediaQueryData.restrictor(), m_mediaQueryData.mediaType().value_or(defaultMediaType), WTFMove(m_mediaQueryData.expressions()) };
148 m_mediaQueryData.clear();
149 m_querySet->addMediaQuery(WTFMove(mediaQuery));
150}

152void MediaQueryParser::readAnd(CSSParserTokenType type, const CSSParserToken& token, CSSParserTokenRange& /*range*/)
153{
154 if (type == IdentToken && equalIgnoringASCIICase(token.value(), "and")) {
155 m_state = ReadFeatureStart;
156 } else if (type == CommaToken && m_parserType != MediaConditionParser) {
157 commitMediaQuery();
158 m_state = ReadRestrictor;
159 } else if (type == EOFToken)
160 m_state = Done;
161 else
162 m_state = SkipUntilComma;
163}
164
165void MediaQueryParser::readFeatureStart(CSSParserTokenType type, const CSSParserToken& /*token*/, CSSParserTokenRange& /*range*/)
166{
167 if (type == LeftParenthesisToken)
168 m_state = ReadFeature;
169 else
170 m_state = SkipUntilComma;
171}
172
173void MediaQueryParser::readFeature(CSSParserTokenType type, const CSSParserToken& token, CSSParserTokenRange& /*range*/)
174{
175 if (type == IdentToken) {
176 m_mediaQueryData.setMediaFeature(token.value().toString());
177 m_state = ReadFeatureColon;
178 } else
179 m_state = SkipUntilComma;
180}
181
182void MediaQueryParser::readFeatureColon(CSSParserTokenType type, const CSSParserToken& token, CSSParserTokenRange& range)
183{
184 if (type == ColonToken) {
185 while (range.peek().type() == WhitespaceToken)
186 range.consume();
187 if (range.peek().type() == RightParenthesisToken || range.peek().type() == EOFToken)
188 m_state = SkipUntilBlockEnd;
189 else
190 m_state = ReadFeatureValue;
191 } else if (type == RightParenthesisToken || type == EOFToken) {
192 m_mediaQueryData.addExpression(range);
193 readFeatureEnd(type, token, range);
194 } else
195 m_state = SkipUntilBlockEnd;
196}
197
198void MediaQueryParser::readFeatureValue(CSSParserTokenType type, const CSSParserToken& token, CSSParserTokenRange& range)
199{
200 if (type == DimensionToken && token.unitType() == CSSUnitType::CSS_UNKNOWN) {
201 range.consume();
202 m_state = SkipUntilComma;
203 } else {
204 m_mediaQueryData.addExpression(range);
205 m_state = ReadFeatureEnd;
206 }
207}
208
209void MediaQueryParser::readFeatureEnd(CSSParserTokenType type, const CSSParserToken& /*token*/, CSSParserTokenRange& /*range*/)
210{
211 if (type == RightParenthesisToken || type == EOFToken) {
212 if (type != EOFToken && m_mediaQueryData.lastExpressionValid())
213 m_state = ReadAnd;
214 else
215 m_state = SkipUntilComma;
216 } else {
217 m_mediaQueryData.removeLastExpression();
218 m_state = SkipUntilBlockEnd;
219 }
220}
221
222void MediaQueryParser::skipUntilComma(CSSParserTokenType type, const CSSParserToken& /*token*/, CSSParserTokenRange& /*range*/)
223{
224 if ((type == CommaToken && !m_blockWatcher.blockLevel()) || type == EOFToken) {
225 m_state = ReadRestrictor;
226 m_mediaQueryData.clear();
227 MediaQuery query = MediaQuery(MediaQuery::Not, "all", Vector<MediaQueryExpression>());
228 m_querySet->addMediaQuery(WTFMove(query));
229 }
230}
231
232void MediaQueryParser::skipUntilBlockEnd(CSSParserTokenType /*type */, const CSSParserToken& token, CSSParserTokenRange& /*range*/)
233{
234 if (token.getBlockType() == CSSParserToken::BlockEnd && !m_blockWatcher.blockLevel())
235 m_state = SkipUntilComma;
236}

238void MediaQueryParser::done(CSSParserTokenType /*type*/, const CSSParserToken& /*token*/, CSSParserTokenRange& /*range*/) { }
239
240void MediaQueryParser::handleBlocks(const CSSParserToken& token)
241{
242 if (token.getBlockType() == CSSParserToken::BlockStart
243 && (token.type() != LeftParenthesisToken || m_blockWatcher.blockLevel()))
244 m_state = SkipUntilBlockEnd;
245}
246
247void MediaQueryParser::processToken(const CSSParserToken& token, CSSParserTokenRange& range)
248{
249 CSSParserTokenType type = token.type();
250
251 if (m_state != ReadFeatureValue || type == WhitespaceToken) {
252 handleBlocks(token);
253 m_blockWatcher.handleToken(token);
254 range.consume();
255 }
256
257 // Call the function that handles current state
258 if (type != WhitespaceToken)
259 ((this)->*(m_state))(type, token, range);
260}
261
262// The state machine loop
263RefPtr<MediaQuerySet> MediaQueryParser::parseInternal(CSSParserTokenRange range)
264{
265 while (!range.atEnd())
266 processToken(range.peek(), range);
267
268 // FIXME: Can we get rid of this special case?
269 if (m_parserType == MediaQuerySetParser)
270 processToken(CSSParserToken(EOFToken), range);
271
272 if (m_state != ReadAnd && m_state != ReadRestrictor && m_state != Done && m_state != ReadMediaNot) {
273 MediaQuery query = MediaQuery(MediaQuery::Not, "all", Vector<MediaQueryExpression>());
274 m_querySet->addMediaQuery(WTFMove(query));
275 } else if (m_mediaQueryData.currentMediaQueryChanged())
276 commitMediaQuery();
277
278 m_querySet->shrinkToFit();
279
280 return m_querySet;
281}

288void MediaQueryParser::MediaQueryData::clear()
289{
290 m_restrictor = MediaQuery::None;
291 m_mediaType = std::nullopt;
292 m_mediaFeature = String();
293 m_expressions.clear();
294}
295
296void MediaQueryParser::MediaQueryData::addExpression(CSSParserTokenRange& range)
297{
298 m_expressions.append(MediaQueryExpression { m_mediaFeature, range, m_context });
299}
300
301bool MediaQueryParser::MediaQueryData::lastExpressionValid()
302{
303 return m_expressions.last().isValid();
304}

306void MediaQueryParser::MediaQueryData::removeLastExpression()
307{
308 m_expressions.removeLast();
