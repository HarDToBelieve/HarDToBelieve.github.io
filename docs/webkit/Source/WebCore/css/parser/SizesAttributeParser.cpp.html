47float SizesAttributeParser::computeLength(double value, CSSUnitType type, const Document& document)
48{
49 auto* renderer = document.renderView();
50 if (!renderer)
51 return 0;
52 auto& style = renderer->style();
53
54 CSSToLengthConversionData conversionData(&style, &style, renderer->parentStyle(), renderer);
55 // Because we evaluate "sizes" at parse time (before style has been resolved), the font metrics used for these specific units
56 // are not available. The font selector's internal consistency isn't guaranteed just yet, so we can just temporarily clear
57 // the pointer to it for the duration of the unit evaluation. This is acceptible because the style always comes from the
58 // RenderView, which has its font information hardcoded in resolveForDocument() to be -webkit-standard, whose operations
59 // don't require a font selector.
60 if (type == CSSUnitType::CSS_EXS || type == CSSUnitType::CSS_CHS) {
61 RefPtr<FontSelector> fontSelector = style.fontCascade().fontSelector();
62 style.fontCascade().update(nullptr);
63 float result = CSSPrimitiveValue::computeNonCalcLengthDouble(conversionData, type, value);
64 style.fontCascade().update(fontSelector.get());
65 return result;
66 }
67 return clampTo<float>(CSSPrimitiveValue::computeNonCalcLengthDouble(conversionData, type, value));
68}

70SizesAttributeParser::SizesAttributeParser(const String& attribute, const Document& document, MediaQueryDynamicResults* mediaQueryDynamicResults)
71 : m_document(document)
72 , m_mediaQueryDynamicResults(mediaQueryDynamicResults)
73{
74 m_isValid = parse(CSSTokenizer(attribute).tokenRange());
75}
76
77float SizesAttributeParser::length()
78{
79 if (m_isValid)
80 return effectiveSize();
81 return effectiveSizeDefaultValue();
82}

84bool SizesAttributeParser::calculateLengthInPixels(CSSParserTokenRange range, float& result)
85{
86 const CSSParserToken& startToken = range.peek();
87 CSSParserTokenType type = startToken.type();
88 if (type == DimensionToken) {
89 if (!CSSPrimitiveValue::isLength(startToken.unitType()))
90 return false;
91 result = computeLength(startToken.numericValue(), startToken.unitType(), m_document);
92 if (result >= 0)
93 return true;
94 } else if (type == FunctionToken) {
95 SizesCalcParser calcParser(range, m_document);
96 if (!calcParser.isValid())
97 return false;
98 result = calcParser.result();
99 return true;
100 } else if (type == NumberToken && !startToken.numericValue()) {
101 result = 0;
102 return true;
103 }
104
105 return false;
106}

108bool SizesAttributeParser::mediaConditionMatches(const MediaQuerySet& mediaCondition)
109{
110 // A Media Condition cannot have a media type other than screen.
111 auto* renderer = m_document.renderView();
112 if (!renderer)
113 return false;
114 auto& style = renderer->style();
115 return MediaQueryEvaluator { "screen", m_document, &style }.evaluate(mediaCondition, m_mediaQueryDynamicResults);
116}

118bool SizesAttributeParser::parse(CSSParserTokenRange range)
119{
120 // Split on a comma token and parse the result tokens as (media-condition, length) pairs
121 while (!range.atEnd()) {
122 const CSSParserToken* mediaConditionStart = &range.peek();
123 // The length is the last component value before the comma which isn't whitespace or a comment
124 const CSSParserToken* lengthTokenStart = &range.peek();
125 const CSSParserToken* lengthTokenEnd = &range.peek();
126 while (!range.atEnd() && range.peek().type() != CommaToken) {
127 lengthTokenStart = &range.peek();
128 range.consumeComponentValue();
129 lengthTokenEnd = &range.peek();
130 range.consumeWhitespace();
131 }
132 range.consume();
133
134 float length;
135 if (!calculateLengthInPixels(range.makeSubRange(lengthTokenStart, lengthTokenEnd), length))
136 continue;
137 RefPtr<MediaQuerySet> mediaCondition = MediaQueryParser::parseMediaCondition(range.makeSubRange(mediaConditionStart, lengthTokenStart), MediaQueryParserContext(m_document));
138 if (!mediaCondition || !mediaConditionMatches(*mediaCondition))
139 continue;
140 m_length = length;
141 m_lengthWasSet = true;
142 return true;
143 }
144 return false;
145}
146
147float SizesAttributeParser::effectiveSize()
148{
149 if (m_lengthWasSet)
150 return m_length;
151 return effectiveSizeDefaultValue();
152}
153
154unsigned SizesAttributeParser::effectiveSizeDefaultValue()
155{
156 auto* renderer = m_document.renderView();
157 if (!renderer)
158 return 0;
159 auto& style = renderer->style();
160 return clampTo<float>(CSSPrimitiveValue::computeNonCalcLengthDouble({ &style, &style, renderer->parentStyle(), renderer }, CSSUnitType::CSS_VW, 100.0));
161}
