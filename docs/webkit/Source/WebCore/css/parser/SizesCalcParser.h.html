36namespace WebCore {

38class Document;
39
40struct SizesCalcValue {
41 double value;
42 bool isLength;
43 UChar operation;
44
45 SizesCalcValue()
46 : value(0)
47 , isLength(false)
48 , operation(0)
49 {
50 }
51
52 SizesCalcValue(double numericValue, bool length)
53 : value(numericValue)
54 , isLength(length)
55 , operation(0)
56 {
57 }
58};
59
60class SizesCalcParser {
61public:
62 SizesCalcParser(CSSParserTokenRange, const Document&);
63
64 float result() const;
65 bool isValid() const { return m_isValid; }
66
67private:
68 bool calcToReversePolishNotation(CSSParserTokenRange);
69 bool calculate();
70 void appendNumber(const CSSParserToken&);
71 bool appendLength(const CSSParserToken&);
72 bool handleOperator(Vector<CSSParserToken>& stack, const CSSParserToken&);
73 void appendOperator(const CSSParserToken&);
74
75 Vector<SizesCalcValue> m_valueList;
76 bool m_isValid;
77 float m_result;
78 const Document& m_document;
79};
