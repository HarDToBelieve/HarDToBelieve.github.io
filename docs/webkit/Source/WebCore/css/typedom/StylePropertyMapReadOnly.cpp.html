46RefPtr<TypedOMCSSStyleValue> StylePropertyMapReadOnly::reifyValue(CSSValue* value, Document& document, Element*)
47{
48 if (!value)
49 return nullptr;
50
51 // FIXME: Properly reify all length values.
52 if (is<CSSPrimitiveValue>(*value) && downcast<CSSPrimitiveValue>(*value).primitiveType() == CSSUnitType::CSS_PX)
53 return TypedOMCSSUnitValue::create(downcast<CSSPrimitiveValue>(*value).doubleValue(), "px");
54
55 if (is<CSSImageValue>(*value))
56 return TypedOMCSSImageValue::create(downcast<CSSImageValue>(*value), document);
57
58 // FIXME: should use raw TypedOMCSSStyleValue
59 return TypedOMCSSUnparsedValue::create(value->cssText());
60}
61
62RefPtr<TypedOMCSSStyleValue> StylePropertyMapReadOnly::customPropertyValueOrDefault(const String& name, Document& document, CSSValue* inputValue, Element* element)
63{
64 if (!inputValue) {
65 auto* registered = document.getCSSRegisteredCustomPropertySet().get(name);
66
67 if (registered && registered->initialValue()) {
68 auto value = registered->initialValueCopy();
69 return StylePropertyMapReadOnly::reifyValue(value.get(), document, element);
70 }
71
72 return TypedOMCSSUnparsedValue::create(emptyString());
73 }
74
75 return StylePropertyMapReadOnly::reifyValue(inputValue, document, element);
76}
