54void LayerController::RootLayerClient::notifyFlushRequired(const GraphicsLayer*)

59void LayerController::RootLayerClient::paintContents(const GraphicsLayer* layer, GraphicsContext& context, const FloatRect& dirtyRect, GraphicsLayerPaintBehavior)
60{
61 ASSERT_UNUSED(layer, layer == m_layerController.contentLayer());
62
63 if (auto* displayTree = m_layerController.view().tree()) {
64 PaintingContext paintingContext { context, deviceScaleFactor() };
65 CSSPainter::paintTree(*displayTree, paintingContext, enclosingIntRect(dirtyRect));
66 }
67}

82void LayerController::prepareForDisplay(Tree& displayTree)
83{
84 auto viewSize = displayTree.rootBox().absoluteBorderBoxRect().size();
85 // FIXME: Do overflow etc.
86 auto contentSize = viewSize;
87
88 LOG_WITH_STREAM(FormattingContextLayout, stream << "LayerController::prepareForDisplay - viewSize " << viewSize << " contentSize " << contentSize);
89
90 ensureRootLayer(viewSize, contentSize);
91
92 // FIXME: For now, repaint the world on every update.
93 m_contentLayer->setNeedsDisplay();
94
95 auto& settings = m_view.frame().settings();
96 bool showDebugBorders = settings.showDebugBorders();
97 bool showRepaintCounter = settings.showRepaintCounter();
98 bool acceleratedDrawingEnabled = settings.acceleratedDrawingEnabled();
99
100 m_rootLayer->setShowDebugBorder(showDebugBorders);
101 m_contentHostLayer->setShowDebugBorder(showDebugBorders);
102
103 m_contentLayer->setShowDebugBorder(showDebugBorders);
104 m_contentLayer->setShowRepaintCounter(showRepaintCounter);
105 m_contentLayer->setAcceleratesDrawing(acceleratedDrawingEnabled);
106}
107
108void LayerController::setNeedsDisplay()
109{
110 m_contentLayer->setNeedsDisplay();
111}
112
113void LayerController::flushLayers()
114{
115 if (!m_rootLayer)
116 return;
117
118 FloatRect visibleRect = visibleRectForLayerFlushing();
119 m_rootLayer->flushCompositingState(visibleRect);
120}
121
122FloatRect LayerController::visibleRectForLayerFlushing() const
123{
124 const auto& frameView = m_view.frameView();
125 return frameView.visibleContentRect();
126}
127
128void LayerController::scheduleRenderingUpdate()
129{
130 auto page = m_view.page();
131 if (!page)
132 return;
133
134 page->scheduleRenderingUpdate(RenderingUpdateStep::LayerFlush);
135}
136
137void LayerController::ensureRootLayer(FloatSize viewSize, FloatSize contentSize)
138{
139 if (m_rootLayer) {
140 updateRootLayerGeometry(viewSize, contentSize);
141 return;
142 }
143
144 setupRootLayerHierarchy();
145 updateRootLayerGeometry(viewSize, contentSize);
146
147 attachRootLayer();
148}
149
150void LayerController::setupRootLayerHierarchy()
151{
152 // FIXME: This layer hierarchy is just enough to get topContentInset working.
153 m_rootLayer = GraphicsLayer::create(graphicsLayerFactory(), m_rootLayerClient);
154 m_rootLayer->setName(MAKE_STATIC_STRING_IMPL("display root"));
155 m_rootLayer->setPosition({ });
156 m_rootLayer->setAnchorPoint({ });
157
158 m_contentHostLayer = GraphicsLayer::create(graphicsLayerFactory(), m_rootLayerClient);
159 m_contentHostLayer->setName(MAKE_STATIC_STRING_IMPL("content host"));
160 m_contentHostLayer->setAnchorPoint({ });
161
162 m_contentLayer = GraphicsLayer::create(graphicsLayerFactory(), m_rootLayerClient);
163 m_contentLayer->setName(MAKE_STATIC_STRING_IMPL("content"));
164 m_contentLayer->setPosition({ });
165 m_contentLayer->setAnchorPoint({ });
166 m_contentLayer->setDrawsContent(true);
167
168 m_rootLayer->addChild(*m_contentHostLayer);
169 m_contentHostLayer->addChild(*m_contentLayer);
170}
171
172void LayerController::updateRootLayerGeometry(FloatSize viewSize, FloatSize contentSize)
173{
174 m_rootLayer->setSize(viewSize);
175
176 float topContentInset = m_view.frameView().topContentInset();
177 m_contentHostLayer->setPosition({ 0, topContentInset });
178 m_contentHostLayer->setSize(viewSize);
179
180 m_contentLayer->setSize(contentSize);
181}
182
183void LayerController::attachRootLayer()
184{
185 auto page = m_view.page();
186 if (!page)
187 return;
188
189 page->chrome().client().attachRootGraphicsLayer(m_view.frame(), rootGraphicsLayer());
190}
191
192void LayerController::detachRootLayer()
193{
194 auto page = m_view.page();
195 if (!page)
196 return;
197
198 page->chrome().client().attachRootGraphicsLayer(m_view.frame(), nullptr);
199}
200
201void LayerController::setIsInWindow(bool isInWindow)
202{
203 if (isInWindow)
204 attachRootLayer();
205 else
206 detachRootLayer();
207}
208
209GraphicsLayerFactory* LayerController::graphicsLayerFactory() const
210{
211 auto page = m_view.page();
212 if (!page)
213 return nullptr;
214
215 return page->chrome().client().graphicsLayerFactory();
216}
