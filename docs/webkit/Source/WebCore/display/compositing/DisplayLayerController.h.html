35class GraphicsLayer;
36class GraphicsLayerFactory;

38namespace Display {
39
40class Tree;
41class View;
42
43// A controller object that makes layerization decisions for a display tree, for a single document.
44class LayerController {
45 WTF_MAKE_FAST_ALLOCATED(LayerController);
46public:
47 explicit LayerController(View&);
48 ~LayerController();
49
50 void prepareForDisplay(Tree&);
51 void flushLayers();
52
53 void setIsInWindow(bool);
54
55 // FIXME: Temporary.
56 void setNeedsDisplay();
57
58 const View& view() const { return m_view; }
59
60private:
61 void attachRootLayer();
62 void detachRootLayer();
63
64 void ensureRootLayer(FloatSize viewSize, FloatSize contentSize);
65 void updateRootLayerGeometry(FloatSize viewSize, FloatSize contentSize);
66
67 void setupRootLayerHierarchy();
68 void scheduleRenderingUpdate();
69
70 GraphicsLayer* rootGraphicsLayer() const { return m_rootLayer.get(); }
71 GraphicsLayer* contentLayer() const { return m_contentLayer.get(); }
72
73 GraphicsLayerFactory* graphicsLayerFactory() const;
74
75 FloatRect visibleRectForLayerFlushing() const;
76
77 class RootLayerClient : public GraphicsLayerClient {
78 public:
79 RootLayerClient(LayerController&);
80
81 private:
82 // GraphicsLayerClient implementation
83 void notifyFlushRequired(const GraphicsLayer*) final;
84 void paintContents(const GraphicsLayer*, GraphicsContext&, const FloatRect&, GraphicsLayerPaintBehavior) final;
85 float deviceScaleFactor() const final;
86
87 LayerController& m_layerController;
88 };
89
90 View& m_view;
91 RootLayerClient m_rootLayerClient;
92
93 RefPtr<GraphicsLayer> m_rootLayer;
94 RefPtr<GraphicsLayer> m_contentHostLayer;
95 RefPtr<GraphicsLayer> m_contentLayer;
96};

98} // namespace Display
99} // namespace WebCore
