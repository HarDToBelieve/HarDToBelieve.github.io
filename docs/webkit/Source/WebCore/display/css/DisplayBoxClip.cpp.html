36DEFINE_ALLOCATOR_WITH_HEAP_IDENTIFIER(BoxClip);
37
38BoxClip::BoxClip() = default;
39BoxClip::~BoxClip() = default;
40
41BoxClip::BoxClip(const BoxClip& other)
42 : m_clipRect(other.clipRect())
43 , m_clipStack(other.clipStack())
44 , m_affectedByBorderRadius(other.affectedByBorderRadius())
45{
46}
47
48Ref<BoxClip> BoxClip::copy() const
49{
50 return adoptRef(*new BoxClip(*this));
51}
52
53void BoxClip::pushClip(const AbsoluteFloatRect& rect)
54{
55 if (m_clipRect) {
56 m_clipRect->intersect(rect);
57 return;
58 }
59
60 m_clipRect = rect;
61}
62
63void BoxClip::pushRoundedClip(const FloatRoundedRect& roundedRect)
64{
65 ASSERT(roundedRect.isRounded());
66 pushClip(roundedRect.rect());
67
68 m_affectedByBorderRadius = true;
69 m_clipStack.append(roundedRect);
70}
