48// Pixel-snapped border edge.
49class BorderEdge {
50public:
51 BorderEdge() = default;
52 BorderEdge(float width, float innerWidth, float outerWidth, Color, BorderStyle, bool isTransparent, bool isPresent);
53
54 float width() const { return m_width; }
55 BorderStyle style() const { return m_style; }
56 const Color& color() const { return m_color; }
57 bool isTransparent() const { return m_isTransparent; }
58 bool isPresent() const { return m_isPresent; }
59
60 bool hasVisibleColorAndStyle() const { return m_style > BorderStyle::Hidden && !m_isTransparent; }
61 bool shouldRender() const { return widthForPainting() && hasVisibleColorAndStyle(); }
62 bool presentButInvisible() const { return widthForPainting() && !hasVisibleColorAndStyle(); }
63 float widthForPainting() const { return m_isPresent ? m_width : 0; }
64
65 float innerWidth() const { return m_innerWidth; }
66 float outerWidth() const { return m_outerWidth; }
67
68 bool obscuresBackground() const;
69 bool obscuresBackgroundEdge(float scale) const;
70
71private:
72 Color m_color;
73 float m_width { 0 };
74 float m_innerWidth { 0 }; // Only used for double borders.
75 float m_outerWidth { 0 }; // Only used for double borders.
76 BorderStyle m_style { BorderStyle::Hidden };
77 bool m_isTransparent { false };
78 bool m_isPresent { false };
79};
80
81RectEdges<BorderEdge> calculateBorderEdges(const RenderStyle&, float pixelSnappingFactor, bool includeLogicalLeftEdge = true, bool includeLogicalRightEdge = true);
82
83std::pair<BoxSide, BoxSide> adjacentSidesForSide(BoxSide);
84
85inline RectEdges<float> borderWidths(const RectEdges<BorderEdge>& edges)
86{
87 return {
88 edges.top().width(),
89 edges.right().width(),
90 edges.bottom().width(),
91 edges.left().width()
92 };
93}

114private:

117};
