48static void applyClipIfNecessary(const Box& box, PaintingContext& paintingContext, GraphicsContextStateSaver& stateSaver)
49{
50 if (is<BoxModelBox>(box) && box.style().hasClippedOverflow()) {
51 stateSaver.save();
52 auto roundedInnerBorder = downcast<BoxModelBox>(box).innerBorderRoundedRect();
53 paintingContext.context.clipRoundedRect(roundedInnerBorder);
54 }
55}
56
57static void applyAncestorClip(const BoxModelBox& box, PaintingContext& paintingContext)
58{
59 auto* boxClip = box.ancestorClip();
60 if (!boxClip || !boxClip->clipRect())
61 return;
62
63 if (!boxClip->affectedByBorderRadius()) {
64 paintingContext.context.clip(*boxClip->clipRect());
65 return;
66 }
67
68 for (auto& roundedRect : boxClip->clipStack())
69 paintingContext.context.clipRoundedRect(roundedRect);
70}

72static void applyEffects(const StackingItem& stackingItem, const Box& box, PaintingContext& paintingContext, TransparencyLayerScope& transparencyScope)
73{
74 if (box.style().opacity() < 1) {
75 paintingContext.context.clip(stackingItem.paintedBoundsIncludingDescendantItems());
76 transparencyScope.beginLayer(box.style().opacity());
77 }
78}

81void CSSPainter::recursivePaintDescendantsForPhase(const ContainerBox& containerBox, PaintingContext& paintingContext, PaintPhase paintPhase)
82{
83 auto stateSaver = GraphicsContextStateSaver { paintingContext.context, false };
84 applyClipIfNecessary(containerBox, paintingContext, stateSaver);
85
86 for (const auto* child = containerBox.firstChild(); child; child = child->nextSibling()) {
87 auto& box = *child;
88 ASSERT(!box.participatesInZOrderSorting());
89
90 switch (paintPhase) {
91 case PaintPhase::BlockBackgrounds:
92 if (!box.style().isFloating() && !box.style().isPositioned() && is<BoxModelBox>(box))
93 BoxPainter::paintBoxDecorations(downcast<BoxModelBox>(box), paintingContext);
94 break;
95 case PaintPhase::Floats:
96 if (box.style().isFloating() && !box.style().isPositioned() && is<BoxModelBox>(box))
97 BoxPainter::paintBoxDecorations(downcast<BoxModelBox>(box), paintingContext);
98 break;
99 case PaintPhase::BlockForegrounds:
100 if (!box.style().isFloating() && !box.style().isPositioned())
101 BoxPainter::paintBoxContent(box, paintingContext);
102 };
103 if (is<ContainerBox>(box))
104 recursivePaintDescendantsForPhase(downcast<ContainerBox>(box), paintingContext, paintPhase);
105 }
106}

108void CSSPainter::recursivePaintDescendants(const ContainerBox& containerBox, PaintingContext& paintingContext)
109{
110 // For all its in-flow, non-positioned, block-level descendants in tree order: If the element is a block, list-item, or other block equivalent:
111 // Box decorations.
112 // Table decorations.
113 recursivePaintDescendantsForPhase(containerBox, paintingContext, PaintPhase::BlockBackgrounds);
114
115 // All non-positioned floating descendants, in tree order. For each one of these, treat the element as if it created a new stacking context,
116 // but any positioned descendants and descendants which actually create a new stacking context should be considered part of the parent
117 // stacking context, not this new one.
118 recursivePaintDescendantsForPhase(containerBox, paintingContext, PaintPhase::Floats);
119
120 // If the element is an inline element that generates a stacking context, then:
121 // FIXME: Handle this case.
122
123 // Otherwise: first for the element, then for all its in-flow, non-positioned, block-level descendants in tree order:
124 // 1. If the element is a block-level replaced element, then: the replaced content, atomically.
125 // 2. Otherwise, for each line box of that element...
126 recursivePaintDescendantsForPhase(containerBox, paintingContext, PaintPhase::BlockForegrounds);
127}

129void CSSPainter::paintAtomicallyPaintedBox(const StackingItem& stackingItem, PaintingContext& paintingContext, const IntRect& dirtyRect, IncludeStackingContextDescendants includeStackingContextDescendants)
130{
131 UNUSED_PARAM(dirtyRect);
132
133 auto& box = stackingItem.box();
134
135 auto needToSaveState = [](const Box& box) {
136 if (!is<BoxModelBox>(box))
137 return false;
138
139 auto& boxModelBox = downcast<BoxModelBox>(box);
140 return boxModelBox.hasAncestorClip() || boxModelBox.style().hasTransform() || boxModelBox.style().opacity() < 1;
141 };
142
143 auto stateSaver = GraphicsContextStateSaver { paintingContext.context, needToSaveState(box) };
144
145 if (is<BoxModelBox>(box))
146 applyAncestorClip(downcast<BoxModelBox>(box), paintingContext);
147
148 if (is<BoxModelBox>(box) && box.style().hasTransform()) {
149 auto transformationMatrix = downcast<BoxModelBox>(box).rareGeometry()->transform();
150 auto absoluteBorderBox = box.absoluteBoxRect();
151
152 // Equivalent to adjusting the CTM so that the origin is in the top left of the border box.
153 transformationMatrix.translateRight(absoluteBorderBox.x(), absoluteBorderBox.y());
154 // Allow descendants rendered using absolute coordinates to paint relative to this box.
155 transformationMatrix.translate(-absoluteBorderBox.x(), -absoluteBorderBox.y());
156
157 auto affineTransform = transformationMatrix.toAffineTransform();
158 paintingContext.context.concatCTM(affineTransform);
159 }
160
161 auto transparencyScope = TransparencyLayerScope { paintingContext.context, 1, false };
162 applyEffects(stackingItem, box, paintingContext, transparencyScope);
163
164#if SHOW_ITEM_EXTENTS
165 {
166 GraphicsContextStateSaver saver(paintingContext.context);
167 paintingContext.context.setStrokeColor(Color::gray);
168 paintingContext.context.strokeRect(stackingItem.paintedBoundsIncludingDescendantItems(), 1);
169 }
170#endif
171
172 BoxPainter::paintBox(box, paintingContext, dirtyRect);
173 if (!is<ContainerBox>(box))
174 return;
175
176 if (includeStackingContextDescendants == IncludeStackingContextDescendants::Yes) {
177 // Stacking contexts formed by positioned descendants with negative z-indices (excluding 0) in z-index order (most negative first) then tree order.
178 for (auto& stackingItem : stackingItem.negativeZOrderList())
179 paintStackingContext(*stackingItem, paintingContext, dirtyRect);
180 }
181
182 auto& containerBox = downcast<ContainerBox>(box);
183 recursivePaintDescendants(containerBox, paintingContext);
184
185 if (includeStackingContextDescendants == IncludeStackingContextDescendants::Yes) {
186 // All positioned descendants with 'z-index: auto' or 'z-index: 0', in tree order. For those with 'z-index: auto', treat the element
187 // as if it created a new stacking context, but any positioned descendants and descendants which actually create a new stacking context
188 // should be considered part of the parent stacking context, not this new one. For those with 'z-index: 0', treat the stacking context
189 // generated atomically.
190 for (auto& stackingItem : stackingItem.positiveZOrderList()) {
191 if (stackingItem->isStackingContext())
192 paintStackingContext(*stackingItem, paintingContext, dirtyRect);
193 else
194 paintAtomicallyPaintedBox(*stackingItem, paintingContext, dirtyRect);
195 }
196 }
197}

199void CSSPainter::paintStackingContext(const StackingItem& stackingItem, PaintingContext& paintingContext, const IntRect& dirtyRect)
200{
201 paintAtomicallyPaintedBox(stackingItem, paintingContext, dirtyRect, IncludeStackingContextDescendants::Yes);
202}

204void CSSPainter::paintTree(const Tree& displayTree, PaintingContext& paintingContext, const IntRect& dirtyRect)
205{
206 paintStackingContext(displayTree.rootStackingItem(), paintingContext, dirtyRect);
