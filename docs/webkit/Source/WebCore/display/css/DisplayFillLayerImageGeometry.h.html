47// Pixel-snapped image geometry for a single fill layer (used for backgrounds and masks).
48class FillLayerImageGeometry {
49public:
50 FillLayerImageGeometry(const FloatRect& destinationRect, const FloatSize& tileSize, const FloatSize& phase, const FloatSize& space, FillAttachment attachment)
51 : m_destRect(destinationRect)
52 , m_destOrigin(m_destRect.location())
53 , m_tileSize(tileSize)
54 , m_phase(phase)
55 , m_space(space)
56 , m_hasNonLocalGeometry(attachment == FillAttachment::FixedBackground)
57 {
58 }
59
60 FloatRect destRect() const { return m_destRect; }
61 FloatSize phase() const { return m_phase; }
62 FloatSize tileSize() const { return m_tileSize; }
63 FloatSize spaceSize() const { return m_space; }
64 bool hasNonLocalGeometry() const { return m_hasNonLocalGeometry; }
65
66 FloatSize relativePhase() const
67 {
68 FloatSize phase = m_phase;
69 phase += m_destRect.location() - m_destOrigin;
70 return phase;
71 }
72
73 void clip(const FloatRect& clipRect) { m_destRect.intersect(clipRect); }
74
75private:
76 FloatRect m_destRect;
77 FloatPoint m_destOrigin;
78 FloatSize m_tileSize;
79 FloatSize m_phase;
80 FloatSize m_space;
81 bool m_hasNonLocalGeometry { false }; // Has background-attachment: fixed. Implies that we can't always cheaply compute destRect.
82};
