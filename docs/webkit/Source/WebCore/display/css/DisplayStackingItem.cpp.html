35namespace WebCore {
36namespace Display {
37
38StackingItem::StackingItem(std::unique_ptr<BoxModelBox>&& box)
39 : m_box(WTFMove(box))
40{
41 ASSERT(m_box);
42}
43
44bool StackingItem::isStackingContext() const
45{
46 return m_box->style().isStackingContext();
47}
48
49void StackingItem::addChildStackingItem(std::unique_ptr<StackingItem>&& item)
50{
51 auto zIndex = item->box().style().zIndex().value_or(0);
52 if (zIndex < 0)
53 m_negativeZOrderList.append(WTFMove(item));
54 else
55 m_positiveZOrderList.append(WTFMove(item));
56}
57
58void StackingItem::sortLists()
59{
60 auto compareZIndex = [](const std::unique_ptr<StackingItem>& a, const std::unique_ptr<StackingItem>& b) {
61 return a->box().style().zIndex().value_or(0) < b->box().style().zIndex().value_or(0);
62 };
63
64 std::stable_sort(m_positiveZOrderList.begin(), m_positiveZOrderList.end(), compareZIndex);
65 std::stable_sort(m_negativeZOrderList.begin(), m_negativeZOrderList.end(), compareZIndex);
66}
