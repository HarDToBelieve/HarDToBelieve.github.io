42static RefPtr<FillLayer> deepCopy(const FillLayer& layer)
43{
44 RefPtr<FillLayer> firstLayer;
45 FillLayer* currCopiedLayer = nullptr;
46
47 for (auto* currLayer = &layer; currLayer; currLayer = currLayer->next()) {
48 RefPtr<FillLayer> layerCopy = currLayer->copy();
49
50 if (!firstLayer) {
51 firstLayer = layerCopy;
52 currCopiedLayer = layerCopy.get();
53 } else {
54 auto nextCopiedLayer = layerCopy.get();
55 currCopiedLayer->setNext(WTFMove(layerCopy));
56 currCopiedLayer = nextCopiedLayer;
57 }
58 }
59 return firstLayer;
60}
61
62static std::unique_ptr<ShadowData> deepCopy(const ShadowData* shadow, const RenderStyle& style)
63{
64 std::unique_ptr<ShadowData> firstShadow;
65 ShadowData* currCopiedShadow = nullptr;
66
67 for (auto* currShadow = shadow; currShadow; currShadow = currShadow->next()) {
68 auto shadowCopy = makeUnique<ShadowData>(*currShadow);
69 shadowCopy->setColor(style.colorByApplyingColorFilter(shadowCopy->color()));
70
71 if (!firstShadow) {
72 currCopiedShadow = shadowCopy.get();
73 firstShadow = WTFMove(shadowCopy);
74 } else {
75 auto nextCopiedShadow = shadowCopy.get();
76 currCopiedShadow->setNext(WTFMove(shadowCopy));
77 currCopiedShadow = nextCopiedShadow;
78 }
79 }
80 return firstShadow;
81}

83Style::Style(const RenderStyle& style)
84 : Style(style, &style)

86}

88Style::Style(const RenderStyle& style, const RenderStyle* styleForBackground)
89 : m_overflowX(style.overflowX())
90 , m_overflowY(style.overflowY())
91 , m_fontCascade(style.fontCascade())
92 , m_whiteSpace(style.whiteSpace())
93 , m_tabSize(style.tabSize())
94 , m_opacity(style.opacity())
95{
96 // FIXME: Is currentColor resolved here?
97 m_color = style.visitedDependentColorWithColorFilter(CSSPropertyColor);
98
99 if (styleForBackground)
100 setupBackground(*styleForBackground);
101
102 m_boxShadow = deepCopy(style.boxShadow(), style);
103
104 if (!style.hasAutoUsedZIndex())
105 m_zIndex = style.usedZIndex();
106
107 setIsPositioned(style.position() != PositionType::Static);
108 setIsFloating(style.floating() != Float::None);
109 setHasTransform(style.hasTransform());
110}
111
112void Style::setupBackground(const RenderStyle& style)
113{
114 m_backgroundColor = style.visitedDependentColorWithColorFilter(CSSPropertyBackgroundColor);
115 m_backgroundLayers = deepCopy(style.backgroundLayers());
116}
117
118bool Style::hasBackground() const
119{
120 return m_backgroundColor.isVisible() || hasBackgroundImage();
121}
122
123bool Style::hasBackgroundImage() const
124{
125 return m_backgroundLayers && m_backgroundLayers->hasImage();
126}
127
128bool Style::backgroundHasOpaqueTopLayer() const
129{
130 auto* fillLayer = backgroundLayers();
131 if (!fillLayer)
132 return false;
133
134 if (fillLayer->clip() != FillBox::Border)
135 return false;
136
137 // FIXME: Check for overflow clip and local attachment.
138 // FIXME: Check for repeated, opaque and renderable image.
139
140 // If there is only one layer and no image, check whether the background color is opaque.
141 if (!fillLayer->next() && !fillLayer->hasImage()) {
142 if (backgroundColor().isOpaque())
143 return true;
144 }
145
146 return false;
147}
