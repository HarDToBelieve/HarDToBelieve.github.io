56// https://dom.spec.whatwg.org/#abortsignal-signal-abort
57void AbortSignal::signalAbort()
58{
59 // 1. If signal's aborted flag is set, then return.
60 if (m_aborted)
61 return;
62
63 // 2. Set signal’s aborted flag.
64 m_aborted = true;
65
66 auto protectedThis = makeRef(*this);
67 auto algorithms = std::exchange(m_algorithms, { });
68 for (auto& algorithm : algorithms)
69 algorithm();
70
71 // 5. Fire an event named abort at signal.
72 dispatchEvent(Event::create(eventNames().abortEvent, Event::CanBubble::No, Event::IsCancelable::No));
73}
74
75// https://dom.spec.whatwg.org/#abortsignal-follow
76void AbortSignal::signalFollow(AbortSignal& signal)
77{
78 if (aborted())
79 return;
80
81 if (signal.aborted()) {
82 signalAbort();
83 return;
84 }
85
86 ASSERT(!m_followingSignal);
87 m_followingSignal = makeWeakPtr(signal);
88 signal.addAlgorithm([weakThis = makeWeakPtr(this)] {
89 if (weakThis)
90 weakThis->signalAbort();
91 });
92}
93
94bool AbortSignal::whenSignalAborted(AbortSignal& signal, Ref<AbortAlgorithm>&& algorithm)
95{
96 if (signal.aborted()) {
97 algorithm->handleEvent();
98 return true;
99 }
100 signal.addAlgorithm([algorithm = WTFMove(algorithm)]() mutable {
101 algorithm->handleEvent();
102 });
103 return false;
104}
