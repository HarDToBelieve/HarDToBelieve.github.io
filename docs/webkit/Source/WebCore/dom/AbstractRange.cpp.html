31namespace WebCore {
32
33SimpleRange makeSimpleRange(const AbstractRange& range)
34{
35 return { { range.startContainer(), range.startOffset() }, { range.endContainer(), range.endOffset() } };
36}
37
38SimpleRange makeSimpleRange(const Ref<AbstractRange>& range)
39{
40 return makeSimpleRange(range.get());
41}
42
43std::optional<SimpleRange> makeSimpleRange(const AbstractRange* range)
44{
45 if (!range)
46 return std::nullopt;
47 return makeSimpleRange(*range);
48}
49
50std::optional<SimpleRange> makeSimpleRange(const RefPtr<AbstractRange>& range)
51{
52 return makeSimpleRange(range.get());
53}
