34#include "ScriptExecutionContext.h"

36namespace WebCore {
37
38ActiveDOMCallback::ActiveDOMCallback(ScriptExecutionContext* context)
39 : ContextDestructionObserver(context)
40{
41}
42
43ActiveDOMCallback::~ActiveDOMCallback() = default;
44
45bool ActiveDOMCallback::canInvokeCallback() const
46{
47 ScriptExecutionContext* context = scriptExecutionContext();
48 return context && !context->activeDOMObjectsAreSuspended() && !context->activeDOMObjectsAreStopped();
49}
50
51bool ActiveDOMCallback::activeDOMObjectsAreSuspended() const
52{
53 auto* context = scriptExecutionContext();
54 return context && context->activeDOMObjectsAreSuspended();
55}
56
57bool ActiveDOMCallback::activeDOMObjectAreStopped() const
58{
59 auto* context = scriptExecutionContext();
60 return !context || context->activeDOMObjectsAreStopped();
61}
