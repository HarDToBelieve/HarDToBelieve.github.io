36static inline ScriptExecutionContext* suitableScriptExecutionContext(ScriptExecutionContext* scriptExecutionContext)
37{
38 // For detached documents, make sure we observe their context document instead.
39 return is<Document>(scriptExecutionContext) ? &downcast<Document>(*scriptExecutionContext).contextDocument() : scriptExecutionContext;
40}

42inline ActiveDOMObject::ActiveDOMObject(ScriptExecutionContext* context, CheckedScriptExecutionContextType)
43 : ContextDestructionObserver(context)
44{
45 ASSERT(!is<Document>(context) || &downcast<Document>(context)->contextDocument() == downcast<Document>(context));
46 if (!context)
47 return;
48
49 ASSERT(context->isContextThread());
50 context->didCreateActiveDOMObject(*this);
51}

68ActiveDOMObject::~ActiveDOMObject()
69{
70 ASSERT(canCurrentThreadAccessThreadLocalData(m_creationThread));
71
72 // ActiveDOMObject may be inherited by a sub-class whose life-cycle
73 // exceeds that of the associated ScriptExecutionContext. In those cases,
74 // m_scriptExecutionContext would/should have been nullified by
75 // ContextDestructionObserver::contextDestroyed() (which we implement /
76 // inherit). Hence, we should ensure that this is not 0 before use it
77 // here.
78 if (!m_scriptExecutionContext)
79 return;
80
81 ASSERT(m_suspendIfNeededWasCalled);
82 ASSERT(m_scriptExecutionContext->isContextThread());
83 m_scriptExecutionContext->willDestroyActiveDOMObject(*this);
84}

86void ActiveDOMObject::suspendIfNeeded()
87{
88#if ASSERT_ENABLED
89 ASSERT(!m_suspendIfNeededWasCalled);
90 m_suspendIfNeededWasCalled = true;
91#endif
92 if (!m_scriptExecutionContext)
93 return;
94
95 m_scriptExecutionContext->suspendActiveDOMObjectIfNeeded(*this);
96}

121bool ActiveDOMObject::isContextStopped() const
122{
123 return !scriptExecutionContext() || scriptExecutionContext()->activeDOMObjectsAreStopped();
124}

126bool ActiveDOMObject::isAllowedToRunScript() const
127{
128 return scriptExecutionContext() && !scriptExecutionContext()->activeDOMObjectsAreStopped() && !scriptExecutionContext()->activeDOMObjectsAreSuspended();
129}

131void ActiveDOMObject::queueTaskInEventLoop(TaskSource source, Function<void ()>&& function)
132{
133 auto* context = scriptExecutionContext();
134 if (!context)
135 return;
136 context->eventLoop().queueTask(source, WTFMove(function));
137}

139class ActiveDOMObjectEventDispatchTask : public EventLoopTask {
140public:
141 ActiveDOMObjectEventDispatchTask(TaskSource source, EventLoopTaskGroup& group, ActiveDOMObject& object, Function<void()>&& dispatchEvent)
142 : EventLoopTask(source, group)
143 , m_object(object)
144 , m_dispatchEvent(WTFMove(dispatchEvent))
145 {
146 ++m_object.m_pendingActivityInstanceCount;
147 }
148
149 ~ActiveDOMObjectEventDispatchTask()
150 {
151 ASSERT(m_object.m_pendingActivityInstanceCount);
152 --m_object.m_pendingActivityInstanceCount;
153 }
154
155 void execute() final
156 {
157 // If this task executes after the script execution context has been stopped, don't
158 // actually dispatch the event.
159 if (m_object.isAllowedToRunScript())
160 m_dispatchEvent();
161 }
162
163private:
164 ActiveDOMObject& m_object;
165 Function<void()> m_dispatchEvent;
166};

168void ActiveDOMObject::queueTaskToDispatchEventInternal(EventTarget& target, TaskSource source, Ref<Event>&& event)
169{
170 ASSERT(!event->target() || &target == event->target());
171 auto* context = scriptExecutionContext();
172 if (!context)
173 return;
174 auto& eventLoopTaskGroup = context->eventLoop();
175 auto task = makeUnique<ActiveDOMObjectEventDispatchTask>(source, eventLoopTaskGroup, *this, [target = makeRef(target), event = WTFMove(event)] {
176 target->dispatchEvent(event);
177 });
178 eventLoopTaskGroup.queueTask(WTFMove(task));
179}

181void ActiveDOMObject::queueCancellableTaskToDispatchEventInternal(EventTarget& target, TaskSource source, TaskCancellationGroup& cancellationGroup, Ref<Event>&& event)
182{
183 ASSERT(!event->target() || &target == event->target());
184 auto* context = scriptExecutionContext();
185 if (!context)
186 return;
187 auto& eventLoopTaskGroup = context->eventLoop();
188 auto task = makeUnique<ActiveDOMObjectEventDispatchTask>(source, eventLoopTaskGroup, *this, CancellableTask(cancellationGroup, [target = makeRef(target), event = WTFMove(event)] {
189 target->dispatchEvent(event);
190 }));
191 eventLoopTaskGroup.queueTask(WTFMove(task));
192}
