45enum class ReasonForSuspension {
46 JavaScriptDebuggerPaused,
47 WillDeferLoading,
48 BackForwardCache,
49 PageWillBeSuspended,
50};
51
52class WEBCORE_EXPORT ActiveDOMObject : public ContextDestructionObserver {
53public:
54 // The suspendIfNeeded must be called exactly once after object construction to update
55 // the suspended state to match that of the ScriptExecutionContext.
56 void suspendIfNeeded();
57 void assertSuspendIfNeededWasCalled() const;
58
59 // This function is used by JS bindings to determine if the JS wrapper should be kept alive or not.
60 bool hasPendingActivity() const { return m_pendingActivityInstanceCount || virtualHasPendingActivity(); }
61
62 // However, the suspend function will sometimes be called even if canSuspendForDocumentSuspension() returns false.
63 // That happens in step-by-step JS debugging for example - in this case it would be incorrect
64 // to stop the object. Exact semantics of suspend is up to the object in cases like that.
65
66 virtual const char* activeDOMObjectName() const = 0;
67
68 // These functions must not have a side effect of creating or destroying
69 // any ActiveDOMObject. That means they must not result in calls to arbitrary JavaScript.
70 virtual void suspend(ReasonForSuspension);
71 virtual void resume();
72
73 // This function must not have a side effect of creating an ActiveDOMObject.
74 // That means it must not result in calls to arbitrary JavaScript.
75 // It can, however, have a side effect of deleting an ActiveDOMObject.
76 virtual void stop();
77
78 template<class T>
79 class PendingActivity : public RefCounted<PendingActivity<T>> {
80 public:
81 explicit PendingActivity(T& thisObject)
82 : m_thisObject(thisObject)
83 {
84 ++(m_thisObject->m_pendingActivityInstanceCount);
85 }
86
87 ~PendingActivity()
88 {
89 ASSERT(m_thisObject->m_pendingActivityInstanceCount > 0);
90 --(m_thisObject->m_pendingActivityInstanceCount);
91 }
92
93 private:
94 Ref<T> m_thisObject;
95 };
96
97 template<class T> Ref<PendingActivity<T>> makePendingActivity(T& thisObject)
98 {
99 ASSERT(&thisObject == this);
100 return adoptRef(*new PendingActivity<T>(thisObject));
101 }
102
103 bool isContextStopped() const;
104 bool isAllowedToRunScript() const;
105
106 template<typename T>
107 static void queueTaskKeepingObjectAlive(T& object, TaskSource source, Function<void ()>&& task)
108 {
109 object.queueTaskInEventLoop(source, [protectedObject = makeRef(object), activity = object.ActiveDOMObject::makePendingActivity(object), task = WTFMove(task)] () {
110 task();
111 });
112 }
113
114 template<typename T>
115 static void queueCancellableTaskKeepingObjectAlive(T& object, TaskSource source, TaskCancellationGroup& cancellationGroup, Function<void()>&& task)
116 {
117 CancellableTask cancellableTask(cancellationGroup, WTFMove(task));
118 object.queueTaskInEventLoop(source, [protectedObject = makeRef(object), activity = object.ActiveDOMObject::makePendingActivity(object), cancellableTask = WTFMove(cancellableTask)]() mutable {
119 cancellableTask();
120 });
121 }
122
123 template<typename EventTargetType>
124 static void queueTaskToDispatchEvent(EventTargetType& target, TaskSource source, Ref<Event>&& event)
125 {
126 target.queueTaskToDispatchEventInternal(target, source, WTFMove(event));
127 }
128
129 template<typename EventTargetType>
130 static void queueCancellableTaskToDispatchEvent(EventTargetType& target, TaskSource source, TaskCancellationGroup& cancellationGroup, Ref<Event>&& event)
131 {
132 target.queueCancellableTaskToDispatchEventInternal(target, source, cancellationGroup, WTFMove(event));
133 }
134
135protected:
136 explicit ActiveDOMObject(ScriptExecutionContext*);
137 explicit ActiveDOMObject(Document*);
138 explicit ActiveDOMObject(Document&);
139 virtual ~ActiveDOMObject();
140
141private:
142 enum CheckedScriptExecutionContextType { CheckedScriptExecutionContext };
143 ActiveDOMObject(ScriptExecutionContext*, CheckedScriptExecutionContextType);
144
145 // This is used by subclasses to indicate that they have pending activity, meaning that they would
146 // like the JS wrapper to stay alive (because they may still fire JS events).
147 virtual bool virtualHasPendingActivity() const { return false; }
148
149 void queueTaskInEventLoop(TaskSource, Function<void ()>&&);
150 void queueTaskToDispatchEventInternal(EventTarget&, TaskSource, Ref<Event>&&);
151 void queueCancellableTaskToDispatchEventInternal(EventTarget&, TaskSource, TaskCancellationGroup&, Ref<Event>&&);
152
153 uint64_t m_pendingActivityInstanceCount { 0 };
154#if ASSERT_ENABLED
155 bool m_suspendIfNeededWasCalled { false };
156 Ref<Thread> m_creationThread { Thread::current() };
157#endif
158
159 friend class ActiveDOMObjectEventDispatchTask;
160};
