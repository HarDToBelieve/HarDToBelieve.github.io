44static bool canUseSetDataOptimization(const CharacterData& node)
45{
46 auto& document = node.document();
47 return !document.hasListenerType(Document::DOMCHARACTERDATAMODIFIED_LISTENER) && !document.hasMutationObserversOfType(MutationObserver::CharacterData)
48 && !document.hasListenerType(Document::DOMSUBTREEMODIFIED_LISTENER);
49}

51void CharacterData::setData(const String& data)
52{
53 const String& nonNullData = !data.isNull() ? data : emptyString();
54 unsigned oldLength = length();
55
56 if (m_data == nonNullData && canUseSetDataOptimization(*this)) {
57 document().textRemoved(*this, 0, oldLength);
58 if (document().frame())
59 document().frame()->selection().textWasReplaced(this, 0, oldLength, oldLength);
60 return;
61 }
62
63 Ref<CharacterData> protectedThis(*this);
64
65 setDataAndUpdate(nonNullData, 0, oldLength, nonNullData.length());
66}

76unsigned CharacterData::parserAppendData(const String& string, unsigned offset, unsigned lengthLimit)
77{
78 unsigned oldLength = m_data.length();
79
80 ASSERT(lengthLimit >= oldLength);
81
82 unsigned characterLength = string.length() - offset;
83 unsigned characterLengthLimit = std::min(characterLength, lengthLimit - oldLength);
84
85 // Check that we are not on an unbreakable boundary.
86 // Some text break iterator implementations work best if the passed buffer is as small as possible,
87 // see <https://bugs.webkit.org/show_bug.cgi?id=29092>.
88 // We need at least two characters look-ahead to account for UTF-16 surrogates.
89 if (characterLengthLimit < characterLength) {
90 NonSharedCharacterBreakIterator it(StringView(string).substring(offset, (characterLengthLimit + 2 > characterLength) ? characterLength : characterLengthLimit + 2));
91 if (!ubrk_isBoundary(it, characterLengthLimit))
92 characterLengthLimit = ubrk_preceding(it, characterLengthLimit);
93 }
94
95 if (!characterLengthLimit)
96 return 0;
97
98 String oldData = m_data;
99 if (string.is8Bit())
100 m_data.append(string.characters8() + offset, characterLengthLimit);
101 else
102 m_data.append(string.characters16() + offset, characterLengthLimit);
103
104 ASSERT(!renderer() || is<Text>(*this));
105 if (is<Text>(*this))
106 downcast<Text>(*this).updateRendererAfterContentChange(oldLength, 0);
107
108 notifyParentAfterChange(ContainerNode::ChildChange::Source::Parser);
109
110 auto mutationRecipients = MutationObserverInterestGroup::createForCharacterDataMutation(*this);
111 if (UNLIKELY(mutationRecipients))
112 mutationRecipients->enqueueMutationRecord(MutationRecord::createCharacterData(*this, oldData));
113
114 return characterLengthLimit;
115}

117void CharacterData::appendData(const String& data)
118{
119 setDataAndUpdate(m_data + data, m_data.length(), 0, data.length(), UpdateLiveRanges::No);
120}

122ExceptionOr<void> CharacterData::insertData(unsigned offset, const String& data)
123{
124 if (offset > length())
125 return Exception { IndexSizeError };
126
127 String newData = m_data;
128 newData.insert(data, offset);
129 setDataAndUpdate(newData, offset, 0, data.length());
130
131 return { };
132}
133
134ExceptionOr<void> CharacterData::deleteData(unsigned offset, unsigned count)
135{
136 if (offset > length())
137 return Exception { IndexSizeError };
138
139 count = std::min(count, length() - offset);
140
141 String newData = m_data;
142 newData.remove(offset, count);
143 setDataAndUpdate(newData, offset, count, 0);
144
145 return { };
146}
147
148ExceptionOr<void> CharacterData::replaceData(unsigned offset, unsigned count, const String& data)
149{
150 if (offset > length())
151 return Exception { IndexSizeError };
152
153 count = std::min(count, length() - offset);
154
155 String newData = m_data;
156 newData.remove(offset, count);
157 newData.insert(data, offset);
158 setDataAndUpdate(newData, offset, count, data.length());
159
160 return { };
161}

174void CharacterData::setDataAndUpdate(const String& newData, unsigned offsetOfReplacedData, unsigned oldLength, unsigned newLength, UpdateLiveRanges shouldUpdateLiveRanges)
175{
176 String oldData = m_data;
177 m_data = newData;
178
179 if (oldLength && shouldUpdateLiveRanges != UpdateLiveRanges::No)
180 document().textRemoved(*this, offsetOfReplacedData, oldLength);
181 if (newLength && shouldUpdateLiveRanges != UpdateLiveRanges::No)
182 document().textInserted(*this, offsetOfReplacedData, newLength);
183
184 ASSERT(!renderer() || is<Text>(*this));
185 if (is<Text>(*this))
186 downcast<Text>(*this).updateRendererAfterContentChange(offsetOfReplacedData, oldLength);
187
188 if (is<ProcessingInstruction>(*this))
189 downcast<ProcessingInstruction>(*this).checkStyleSheet();
190
191 if (document().frame())
192 document().frame()->selection().textWasReplaced(this, offsetOfReplacedData, oldLength, newLength);
193
194 notifyParentAfterChange(ContainerNode::ChildChange::Source::API);
195
196 dispatchModifiedEvent(oldData);
197}
198
199void CharacterData::notifyParentAfterChange(ContainerNode::ChildChange::Source source)
200{
201 document().incDOMTreeVersion();
202
203 if (!parentNode())
204 return;
205
206 ContainerNode::ChildChange change = {
207 ContainerNode::ChildChange::Type::TextChanged,
208 ElementTraversal::previousSibling(*this),
209 ElementTraversal::nextSibling(*this),
210 source
211 };
212
213 parentNode()->childrenChanged(change);
214}

216void CharacterData::dispatchModifiedEvent(const String& oldData)
217{
218 if (auto mutationRecipients = MutationObserverInterestGroup::createForCharacterDataMutation(*this))
219 mutationRecipients->enqueueMutationRecord(MutationRecord::createCharacterData(*this, oldData));
220
221 if (!isInShadowTree()) {
222 if (document().hasListenerType(Document::DOMCHARACTERDATAMODIFIED_LISTENER))
223 dispatchScopedEvent(MutationEvent::create(eventNames().DOMCharacterDataModifiedEvent, Event::CanBubble::Yes, nullptr, oldData, m_data));
224 dispatchSubtreeModifiedEvent();
225 }
226
227 InspectorInstrumentation::characterDataModified(document(), *this);
228}
