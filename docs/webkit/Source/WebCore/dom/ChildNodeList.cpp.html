70void ChildNodeList::collectionTraverseForward(Node*& current, unsigned count, unsigned& traversedCount) const
71{
72 ASSERT(count);
73 for (traversedCount = 0; traversedCount < count; ++traversedCount) {
74 current = current->nextSibling();
75 if (!current)
76 return;
77 }
78}
79
80void ChildNodeList::collectionTraverseBackward(Node*& current, unsigned count) const
81{
82 ASSERT(count);
83 for (; count && current ; --count)
84 current = current->previousSibling();
85}

87void ChildNodeList::invalidateCache()

89 m_indexCache.invalidate();
90}
