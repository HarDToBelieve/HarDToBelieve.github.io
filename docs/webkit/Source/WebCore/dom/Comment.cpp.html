30WTF_MAKE_ISO_ALLOCATED_IMPL(Comment);

32inline Comment::Comment(Document& document, const String& text)
33 : CharacterData(document, text)
34{
35}
36
37Ref<Comment> Comment::create(Document& document, const String& text)
38{
39 return adoptRef(*new Comment(document, text));
40}
41
42String Comment::nodeName() const
43{
44 return "#comment"_s;
45}
46
47Node::NodeType Comment::nodeType() const
48{
49 return COMMENT_NODE;
50}
51
52Ref<Node> Comment::cloneNodeInternal(Document& targetDocument, CloningOperation)
53{
54 return create(targetDocument, data());
55}
56
57bool Comment::childTypeAllowed(NodeType) const
58{
59 return false;
60}
