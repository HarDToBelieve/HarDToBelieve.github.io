32namespace WebCore {
33
34WTF_MAKE_ISO_ALLOCATED_IMPL(CompositionEvent);
35
36CompositionEvent::CompositionEvent() = default;
37
38CompositionEvent::CompositionEvent(const AtomString& type, RefPtr<WindowProxy>&& view, const String& data)
39 : UIEvent(type, CanBubble::Yes, IsCancelable::Yes, IsComposed::Yes, WTFMove(view), 0)
40 , m_data(data)
41{
42}
43
44CompositionEvent::CompositionEvent(const AtomString& type, const Init& initializer)
45 : UIEvent(type, initializer)
46 , m_data(initializer.data)
47{
48}
49
50CompositionEvent::~CompositionEvent() = default;
51
52void CompositionEvent::initCompositionEvent(const AtomString& type, bool canBubble, bool cancelable, RefPtr<WindowProxy>&& view, const String& data)
53{
54 if (isBeingDispatched())
55 return;
56
57 initUIEvent(type, canBubble, cancelable, WTFMove(view), 0);
58
59 m_data = data;
60}
61
62EventInterface CompositionEvent::eventInterface() const
63{
64 return CompositionEventInterfaceType;
65}
66
67bool CompositionEvent::isCompositionEvent() const
68{
69 return true;
70}
