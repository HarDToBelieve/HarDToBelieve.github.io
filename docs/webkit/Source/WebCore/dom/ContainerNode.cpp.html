81ALWAYS_INLINE NodeVector ContainerNode::removeAllChildrenWithScriptAssertion(ChildChange::Source source, DeferChildrenChanged deferChildrenChanged)
82{
83 auto children = collectChildNodes(*this);
84
85 if (UNLIKELY(isDocumentFragmentForInnerOuterHTML())) {
86 ScriptDisallowedScope::InMainThread scriptDisallowedScope;
87 RELEASE_ASSERT(!connectedSubframeCount() && !hasRareData() && !wrapper());
88 ASSERT(!weakPtrFactory().isInitialized());
89 while (RefPtr<Node> child = m_firstChild)
90 removeBetween(nullptr, child->nextSibling(), *child);
91 document().incDOMTreeVersion();
92 return children;
93 }
94
95 if (source == ChildChange::Source::API) {
96 ChildListMutationScope mutation(*this);
97 for (auto& child : children) {
98 mutation.willRemoveChild(child.get());
99 child->notifyMutationObserversNodeWillDetach();
100 dispatchChildRemovalEvents(child);
101 }
102 } else {
103 ASSERT(source == ChildChange::Source::Parser);
104 ScriptDisallowedScope::InMainThread scriptDisallowedScope;
105 if (UNLIKELY(document().hasMutationObserversOfType(MutationObserver::ChildList))) {
106 ChildListMutationScope mutation(*this);
107 for (auto& child : children)
108 mutation.willRemoveChild(child.get());
109 }
110 }
111
112 disconnectSubframesIfNeeded(*this, DescendantsOnly);
113
114 WidgetHierarchyUpdatesSuspensionScope suspendWidgetHierarchyUpdates;
115 ScriptDisallowedScope::InMainThread scriptDisallowedScope;
116
117 if (UNLIKELY(isShadowRoot() || isInShadowTree()))
118 containingShadowRoot()->willRemoveAllChildren(*this);
119
120 document().nodeChildrenWillBeRemoved(*this);
121
122 while (RefPtr<Node> child = m_firstChild) {
123 removeBetween(nullptr, child->nextSibling(), *child);
124 auto subtreeObservability = notifyChildNodeRemoved(*this, *child);
125 if (source == ChildChange::Source::API && subtreeObservability == RemovedSubtreeObservability::MaybeObservableByRefPtr)
126 willCreatePossiblyOrphanedTreeByRemoval(child.get());
127 }
128
129 ASSERT_WITH_SECURITY_IMPLICATION(!document().selection().selection().isOrphan());
130
131 if (deferChildrenChanged == DeferChildrenChanged::No)
132 childrenChanged(ContainerNode::ChildChange { ChildChange::Type::AllChildrenRemoved, nullptr, nullptr, source });
133
134 return children;
135}

137ALWAYS_INLINE bool ContainerNode::removeNodeWithScriptAssertion(Node& childToRemove, ChildChange::Source source)
138{
139 Ref<Node> protectedChildToRemove(childToRemove);
140 ASSERT_WITH_SECURITY_IMPLICATION(childToRemove.parentNode() == this);
141 {
142 ScriptDisallowedScope::InMainThread scriptDisallowedScope;
143 ChildListMutationScope(*this).willRemoveChild(childToRemove);
144 }
145
146 ASSERT_WITH_SECURITY_IMPLICATION(ScriptDisallowedScope::InMainThread::isEventDispatchAllowedInSubtree(childToRemove));
147 if (source == ChildChange::Source::API) {
148 childToRemove.notifyMutationObserversNodeWillDetach();
149 dispatchChildRemovalEvents(protectedChildToRemove);
150 if (childToRemove.parentNode() != this)
151 return false;
152 }
153
154 if (source == ChildChange::Source::Parser) {
155 // FIXME: Merge these two code paths. It's a bug in the parser not to update connectedSubframeCount in time.
156 disconnectSubframesIfNeeded(*this, DescendantsOnly);
157 } else {
158 if (is<ContainerNode>(childToRemove))
159 disconnectSubframesIfNeeded(downcast<ContainerNode>(childToRemove), RootAndDescendants);
160 }
161
162 if (childToRemove.parentNode() != this)
163 return false;
164
165 ChildChange change;
166 RemovedSubtreeObservability subtreeObservability;
167 {
168 WidgetHierarchyUpdatesSuspensionScope suspendWidgetHierarchyUpdates;
169 ScriptDisallowedScope::InMainThread scriptDisallowedScope;
170
171 if (UNLIKELY(isShadowRoot() || isInShadowTree()))
172 containingShadowRoot()->resolveSlotsBeforeNodeInsertionOrRemoval();
173
174 document().nodeWillBeRemoved(childToRemove);
175
176 ASSERT_WITH_SECURITY_IMPLICATION(childToRemove.parentNode() == this);
177 ASSERT(!childToRemove.isDocumentFragment());
178
179 RefPtr<Node> previousSibling = childToRemove.previousSibling();
180 RefPtr<Node> nextSibling = childToRemove.nextSibling();
181 removeBetween(previousSibling.get(), nextSibling.get(), childToRemove);
182 subtreeObservability = notifyChildNodeRemoved(*this, childToRemove);
183
184 change.type = is<Element>(childToRemove) ?
185 ChildChange::Type::ElementRemoved :
186 (is<Text>(childToRemove) ?
187 ChildChange::Type::TextRemoved :
188 ChildChange::Type::NonContentsChildRemoved);
189 change.previousSiblingElement = (!previousSibling || is<Element>(*previousSibling)) ? downcast<Element>(previousSibling.get()) : ElementTraversal::previousSibling(*previousSibling);
190 change.nextSiblingElement = (!nextSibling || is<Element>(*nextSibling)) ? downcast<Element>(nextSibling.get()) : ElementTraversal::nextSibling(*nextSibling);
191 change.source = source;
192 }
193
194 if (source == ChildChange::Source::API && subtreeObservability == RemovedSubtreeObservability::MaybeObservableByRefPtr)
195 willCreatePossiblyOrphanedTreeByRemoval(&childToRemove);
196
197 ASSERT_WITH_SECURITY_IMPLICATION(!document().selection().selection().isOrphan());
198
199 // FIXME: Move childrenChanged into ScriptDisallowedScope block.
200 childrenChanged(change);
201
202 return true;
203}

207template<typename DOMInsertionWork>
208static ALWAYS_INLINE void executeNodeInsertionWithScriptAssertion(ContainerNode& containerNode, Node& child,
209 ContainerNode::ChildChange::Source source, ReplacedAllChildren replacedAllChildren, DOMInsertionWork doNodeInsertion)
210{
211 NodeVector postInsertionNotificationTargets;
212 {
213 WidgetHierarchyUpdatesSuspensionScope suspendWidgetHierarchyUpdates;
214 ScriptDisallowedScope::InMainThread scriptDisallowedScope;
215
216 if (UNLIKELY(containerNode.isShadowRoot() || containerNode.isInShadowTree()))
217 containerNode.containingShadowRoot()->resolveSlotsBeforeNodeInsertionOrRemoval();
218
219 doNodeInsertion();
220 ChildListMutationScope(containerNode).childAdded(child);
221 postInsertionNotificationTargets = notifyChildNodeInserted(containerNode, child);
222 }
223
224 // FIXME: Move childrenChanged into ScriptDisallowedScope block.
225 if (replacedAllChildren == ReplacedAllChildren::Yes)
226 containerNode.childrenChanged(ContainerNode::ChildChange { ContainerNode::ChildChange::Type::AllChildrenReplaced, nullptr, nullptr, source });
227 else {
228 containerNode.childrenChanged(ContainerNode::ChildChange {
229 child.isElementNode() ?
230 ContainerNode::ChildChange::Type::ElementInserted :
231 (child.isTextNode() ?
232 ContainerNode::ChildChange::Type::TextInserted :
233 ContainerNode::ChildChange::Type::NonContentsChildInserted),
234 ElementTraversal::previousSibling(child),
235 ElementTraversal::nextSibling(child),
236 source
237 });
238 }
239
240 ASSERT(ScriptDisallowedScope::InMainThread::isEventDispatchAllowedInSubtree(child));
241 for (auto& target : postInsertionNotificationTargets)
242 target->didFinishInsertingNode();
243
244 if (source == ContainerNode::ChildChange::Source::API)
245 dispatchChildInsertionEvents(child);
246}

248ExceptionOr<void> ContainerNode::removeSelfOrChildNodesForInsertion(Node& child, NodeVector& nodesForInsertion)
249{
250 if (!is<DocumentFragment>(child)) {
251 nodesForInsertion.append(child);
252 auto oldParent = makeRefPtr(child.parentNode());
253 if (!oldParent)
254 return { };
255 return oldParent->removeChild(child);
256 }
257
258 auto& fragment = downcast<DocumentFragment>(child);
259 if (!fragment.hasChildNodes())
260 return { };
261
262 auto removedChildNodes = fragment.removeAllChildrenWithScriptAssertion(ContainerNode::ChildChange::Source::API);
263 nodesForInsertion.swap(removedChildNodes);
264
265 fragment.rebuildSVGExtensionsElementsIfNecessary();
266 fragment.dispatchSubtreeModifiedEvent();
267
268 return { };
269}

285static inline void destroyRenderTreeIfNeeded(Node& child)
286{
287 bool isElement = is<Element>(child);
288 auto hasDisplayContents = isElement && downcast<Element>(child).hasDisplayContents();
289 if (!child.renderer() && !hasDisplayContents)
290 return;
291 if (isElement)
292 RenderTreeUpdater::tearDownRenderers(downcast<Element>(child));
293 else if (is<Text>(child))
294 RenderTreeUpdater::tearDownRenderer(downcast<Text>(child));

297void ContainerNode::takeAllChildrenFrom(ContainerNode* oldParent)
298{
299 ASSERT(oldParent);
300
301 auto children = oldParent->removeAllChildrenWithScriptAssertion(ChildChange::Source::Parser);
302
303 // FIXME: assert that we don't dispatch events here since this container node is still disconnected.
304 for (auto& child : children) {
305 RELEASE_ASSERT(!child->parentNode() && &child->treeScope() == &treeScope());
306 ASSERT(!ensurePreInsertionValidity(child, nullptr).hasException());
307 child->setTreeScopeRecursively(treeScope());
308 parserAppendChild(child);
309 }
310}

319static inline bool isChildTypeAllowed(ContainerNode& newParent, Node& child)
320{
321 if (!child.isDocumentFragment())
322 return newParent.childTypeAllowed(child.nodeType());
323
324 for (Node* node = child.firstChild(); node; node = node->nextSibling()) {
325 if (!newParent.childTypeAllowed(node->nodeType()))
326 return false;
327 }
328 return true;
329}

331static bool containsIncludingHostElements(const Node& possibleAncestor, const Node& node)
332{
333 const Node* currentNode = &node;
334 do {
335 if (currentNode == &possibleAncestor)
336 return true;
337 const ContainerNode* parent = currentNode->parentNode();
338 if (!parent) {
339 if (is<ShadowRoot>(currentNode))
340 parent = downcast<ShadowRoot>(currentNode)->host();
341 else if (is<DocumentFragment>(*currentNode) && downcast<DocumentFragment>(*currentNode).isTemplateContent())
342 parent = static_cast<const TemplateContentDocumentFragment*>(currentNode)->host();
343 }
344 currentNode = parent;
345 } while (currentNode);
346
347 return false;
348}

350enum class ShouldValidateChildParent { No, Yes };
351static inline ExceptionOr<void> checkAcceptChild(ContainerNode& newParent, Node& newChild, const Node* refChild, Document::AcceptChildOperation operation, ShouldValidateChildParent shouldValidateChildParent)
352{
353 if (containsIncludingHostElements(newChild, newParent))
354 return Exception { HierarchyRequestError };
355
356 // Use common case fast path if possible.
357 if ((newChild.isElementNode() || newChild.isTextNode()) && newParent.isElementNode()) {
358 ASSERT(!newParent.isDocumentTypeNode());
359 ASSERT(isChildTypeAllowed(newParent, newChild));
360 if (shouldValidateChildParent == ShouldValidateChildParent::Yes && refChild && refChild->parentNode() != &newParent)
361 return Exception { NotFoundError };
362 return { };
363 }
364
365 // This should never happen, but also protect release builds from tree corruption.
366 ASSERT(!newChild.isPseudoElement());
367 if (newChild.isPseudoElement())
368 return Exception { HierarchyRequestError };
369
370 if (shouldValidateChildParent == ShouldValidateChildParent::Yes && refChild && refChild->parentNode() != &newParent)
371 return Exception { NotFoundError };
372
373 if (is<Document>(newParent)) {
374 if (!downcast<Document>(newParent).canAcceptChild(newChild, refChild, operation))
375 return Exception { HierarchyRequestError };
376 } else if (!isChildTypeAllowed(newParent, newChild))
377 return Exception { HierarchyRequestError };
378
379 return { };
380}

382static inline ExceptionOr<void> checkAcceptChildGuaranteedNodeTypes(ContainerNode& newParent, Node& newChild)
383{
384 ASSERT(!newParent.isDocumentTypeNode());
385 ASSERT(isChildTypeAllowed(newParent, newChild));
386 if (containsIncludingHostElements(newChild, newParent))
387 return Exception { HierarchyRequestError };
388 return { };

403ExceptionOr<void> ContainerNode::insertBefore(Node& newChild, Node* refChild)
404{
405 // Check that this node is not "floating".
406 // If it is, it can be deleted as a side effect of sending mutation events.
407 ASSERT(refCount() || parentOrShadowHostNode());
408
409 // Make sure adding the new child is OK.
410 auto validityCheckResult = ensurePreInsertionValidity(newChild, refChild);
411 if (validityCheckResult.hasException())
412 return validityCheckResult.releaseException();
413
414 if (refChild == &newChild)
415 refChild = newChild.nextSibling();
416
417 // insertBefore(node, null) is equivalent to appendChild(node)
418 if (!refChild)
419 return appendChildWithoutPreInsertionValidityCheck(newChild);
420
421 Ref<ContainerNode> protectedThis(*this);
422 Ref<Node> next(*refChild);
423
424 NodeVector targets;
425 auto removeResult = removeSelfOrChildNodesForInsertion(newChild, targets);
426 if (removeResult.hasException())
427 return removeResult.releaseException();
428 if (targets.isEmpty())
429 return { };
430
431 // We need this extra check because removeSelfOrChildNodesForInsertion() can fire mutation events.
432 for (auto& child : targets) {
433 auto checkAcceptResult = checkAcceptChildGuaranteedNodeTypes(*this, child);
434 if (checkAcceptResult.hasException())
435 return checkAcceptResult.releaseException();
436 }
437
438 InspectorInstrumentation::willInsertDOMNode(document(), *this);
439
440 ChildListMutationScope mutation(*this);
441 for (auto& child : targets) {
442 // Due to arbitrary code running in response to a DOM mutation event it's
443 // possible that "next" is no longer a child of "this".
444 // It's also possible that "child" has been inserted elsewhere.
445 // In either of those cases, we'll just stop.
446 if (next->parentNode() != this)
447 break;
448 if (child->parentNode())
449 break;
450
451 executeNodeInsertionWithScriptAssertion(*this, child.get(), ChildChange::Source::API, ReplacedAllChildren::No, [&] {
452 child->setTreeScopeRecursively(treeScope());
453 insertBeforeCommon(next, child);
454 });
455 }
456
457 dispatchSubtreeModifiedEvent();
458 return { };
459}

461void ContainerNode::insertBeforeCommon(Node& nextChild, Node& newChild)
462{
463 ScriptDisallowedScope::InMainThread scriptDisallowedScope;
464
465 ASSERT(!newChild.parentNode()); // Use insertBefore if you need to handle reparenting (and want DOM mutation events).
466 ASSERT(!newChild.nextSibling());
467 ASSERT(!newChild.previousSibling());
468 ASSERT(!newChild.isShadowRoot());
469
470 Node* prev = nextChild.previousSibling();
471 ASSERT(m_lastChild != prev);
472 nextChild.setPreviousSibling(&newChild);
473 if (prev) {
474 ASSERT(m_firstChild != &nextChild);
475 ASSERT(prev->nextSibling() == &nextChild);
476 prev->setNextSibling(&newChild);
477 } else {
478 ASSERT(m_firstChild == &nextChild);
479 m_firstChild = &newChild;
480 }
481 newChild.setParentNode(this);
482 newChild.setPreviousSibling(prev);
483 newChild.setNextSibling(&nextChild);
484}

486void ContainerNode::appendChildCommon(Node& child)
487{
488 ScriptDisallowedScope::InMainThread scriptDisallowedScope;
489
490 child.setParentNode(this);
491
492 if (m_lastChild) {
493 child.setPreviousSibling(m_lastChild);
494 m_lastChild->setNextSibling(&child);
495 } else
496 m_firstChild = &child;
497
498 m_lastChild = &child;
499}

501void ContainerNode::parserInsertBefore(Node& newChild, Node& nextChild)
502{
503 ASSERT(nextChild.parentNode() == this);
504 ASSERT(!newChild.isDocumentFragment());
505 ASSERT(!hasTagName(HTMLNames::templateTag));
506
507 if (nextChild.previousSibling() == &newChild || &nextChild == &newChild) // nothing to do
508 return;
509
510 executeNodeInsertionWithScriptAssertion(*this, newChild, ChildChange::Source::Parser, ReplacedAllChildren::No, [&] {
511 if (&document() != &newChild.document())
512 document().adoptNode(newChild);
513
514 insertBeforeCommon(nextChild, newChild);
515
516 newChild.updateAncestorConnectedSubframeCountForInsertion();
517 });
518}

520ExceptionOr<void> ContainerNode::replaceChild(Node& newChild, Node& oldChild)
521{
522 // Check that this node is not "floating".
523 // If it is, it can be deleted as a side effect of sending mutation events.
524 ASSERT(refCount() || parentOrShadowHostNode());
525
526 Ref<ContainerNode> protectedThis(*this);
527
528 // Make sure replacing the old child with the new is ok
529 auto validityResult = checkPreReplacementValidity(*this, newChild, oldChild, ShouldValidateChildParent::Yes);
530 if (validityResult.hasException())
531 return validityResult.releaseException();
532
533 RefPtr<Node> refChild = oldChild.nextSibling();
534 if (refChild.get() == &newChild)
535 refChild = refChild->nextSibling();
536
537 NodeVector targets;
538 {
539 ChildListMutationScope mutation(*this);
540 auto collectResult = removeSelfOrChildNodesForInsertion(newChild, targets);
541 if (collectResult.hasException())
542 return collectResult.releaseException();
543 }
544
545 // Do this one more time because removeSelfOrChildNodesForInsertion() fires a MutationEvent.
546 for (auto& child : targets) {
547 validityResult = checkPreReplacementValidity(*this, child, oldChild, ShouldValidateChildParent::No);
548 if (validityResult.hasException())
549 return validityResult.releaseException();
550 }
551
552 // Remove the node we're replacing.
553 Ref<Node> protectOldChild(oldChild);
554
555 ChildListMutationScope mutation(*this);
556
557 // If oldChild == newChild then oldChild no longer has a parent at this point.
558 if (oldChild.parentNode()) {
559 auto removeResult = removeChild(oldChild);
560 if (removeResult.hasException())
561 return removeResult.releaseException();
562
563 // Does this one more time because removeChild() fires a MutationEvent.
564 for (auto& child : targets) {
565 validityResult = checkPreReplacementValidity(*this, child, oldChild, ShouldValidateChildParent::No);
566 if (validityResult.hasException())
567 return validityResult.releaseException();
568 }
569 }
570
571 InspectorInstrumentation::willInsertDOMNode(document(), *this);
572
573 // Add the new child(ren).
574 for (auto& child : targets) {
575 // Due to arbitrary code running in response to a DOM mutation event it's
576 // possible that "refChild" is no longer a child of "this".
577 // It's also possible that "child" has been inserted elsewhere.
578 // In either of those cases, we'll just stop.
579 if (refChild && refChild->parentNode() != this)
580 break;
581 if (child->parentNode())
582 break;
583
584 executeNodeInsertionWithScriptAssertion(*this, child.get(), ChildChange::Source::API, ReplacedAllChildren::No, [&] {
585 child->setTreeScopeRecursively(treeScope());
586 if (refChild)
587 insertBeforeCommon(*refChild, child.get());
588 else
589 appendChildCommon(child);
590 });
591 }
592
593 dispatchSubtreeModifiedEvent();
594 return { };
595}

602ExceptionOr<void> ContainerNode::removeChild(Node& oldChild)
603{
604 // Check that this node is not "floating".
605 // If it is, it can be deleted as a side effect of sending mutation events.
606 ASSERT(refCount() || parentOrShadowHostNode());
607
608 Ref<ContainerNode> protectedThis(*this);
609
610 // NotFoundError: Raised if oldChild is not a child of this node.
611 if (oldChild.parentNode() != this)
612 return Exception { NotFoundError };
613
614 if (!removeNodeWithScriptAssertion(oldChild, ChildChange::Source::API))
615 return Exception { NotFoundError };
616
617 rebuildSVGExtensionsElementsIfNecessary();
618 dispatchSubtreeModifiedEvent();
619
620 return { };
621}

623void ContainerNode::removeBetween(Node* previousChild, Node* nextChild, Node& oldChild)
624{
625 InspectorInstrumentation::didRemoveDOMNode(oldChild.document(), oldChild);
626
627 ScriptDisallowedScope::InMainThread scriptDisallowedScope;
628
629 ASSERT(oldChild.parentNode() == this);
630
631 destroyRenderTreeIfNeeded(oldChild);
632
633 if (UNLIKELY(hasShadowRootContainingSlots()))
634 shadowRoot()->willRemoveAssignedNode(oldChild);
635
636 if (nextChild) {
637 nextChild->setPreviousSibling(previousChild);
638 oldChild.setNextSibling(nullptr);
639 } else {
640 ASSERT(m_lastChild == &oldChild);
641 m_lastChild = previousChild;
642 }
643 if (previousChild) {
644 previousChild->setNextSibling(nextChild);
645 oldChild.setPreviousSibling(nullptr);
646 } else {
647 ASSERT(m_firstChild == &oldChild);
648 m_firstChild = nextChild;
649 }
650
651 ASSERT(m_firstChild != &oldChild);
652 ASSERT(m_lastChild != &oldChild);
653 ASSERT(!oldChild.previousSibling());
654 ASSERT(!oldChild.nextSibling());
655 oldChild.setParentNode(nullptr);
656
657 oldChild.setTreeScopeRecursively(document());
658}

665// https://dom.spec.whatwg.org/#concept-node-replace-all
666void ContainerNode::replaceAll(Node* node)
667{
668 if (!node) {
669 ChildListMutationScope mutation(*this);
670 removeChildren();
671 return;
672 }
673
674 // FIXME: The code below is roughly correct for a new text node with no parent, but needs enhancement to work properly for more complex cases.
675
676 if (!hasChildNodes()) {
677 appendChildWithoutPreInsertionValidityCheck(*node);
678 return;
679 }
680
681 Ref<ContainerNode> protectedThis(*this);
682 ChildListMutationScope mutation(*this);
683 removeAllChildrenWithScriptAssertion(ChildChange::Source::API, DeferChildrenChanged::Yes);
684
685 executeNodeInsertionWithScriptAssertion(*this, *node, ChildChange::Source::API, ReplacedAllChildren::Yes, [&] {
686 InspectorInstrumentation::willInsertDOMNode(document(), *this);
687 node->setTreeScopeRecursively(treeScope());
688 appendChildCommon(*node);
689 });

721ExceptionOr<void> ContainerNode::appendChild(Node& newChild)
722{
723 // Check that this node is not "floating".
724 // If it is, it can be deleted as a side effect of sending mutation events.
725 ASSERT(refCount() || parentOrShadowHostNode());
726
727 // Make sure adding the new child is ok
728 auto validityCheckResult = ensurePreInsertionValidity(newChild, nullptr);
729 if (validityCheckResult.hasException())
730 return validityCheckResult.releaseException();
731
732 return appendChildWithoutPreInsertionValidityCheck(newChild);
733}
734
735ExceptionOr<void> ContainerNode::appendChildWithoutPreInsertionValidityCheck(Node& newChild)
736{
737 Ref<ContainerNode> protectedThis(*this);
738
739 NodeVector targets;
740 auto removeResult = removeSelfOrChildNodesForInsertion(newChild, targets);
741 if (removeResult.hasException())
742 return removeResult.releaseException();
743
744 if (targets.isEmpty())
745 return { };
746
747 // We need this extra check because removeSelfOrChildNodesForInsertion() can fire mutation events.
748 for (auto& child : targets) {
749 auto nodeTypeResult = checkAcceptChildGuaranteedNodeTypes(*this, child);
750 if (nodeTypeResult.hasException())
751 return nodeTypeResult.releaseException();
752 }
753
754 InspectorInstrumentation::willInsertDOMNode(document(), *this);
755
756 // Now actually add the child(ren)
757 ChildListMutationScope mutation(*this);
758 for (auto& child : targets) {
759 // If the child has a parent again, just stop what we're doing, because
760 // that means someone is doing something with DOM mutation -- can't re-parent
761 // a child that already has a parent.
762 if (child->parentNode())
763 break;
764
765 // Append child to the end of the list
766 executeNodeInsertionWithScriptAssertion(*this, child.get(), ChildChange::Source::API, ReplacedAllChildren::No, [&] {
767 child->setTreeScopeRecursively(treeScope());
768 appendChildCommon(child);
769 });
770 }
771
772 dispatchSubtreeModifiedEvent();
773 return { };
774}

776void ContainerNode::parserAppendChild(Node& newChild)
777{
778 ASSERT(!newChild.parentNode()); // Use appendChild if you need to handle reparenting (and want DOM mutation events).
779 ASSERT(!newChild.isDocumentFragment());
780 ASSERT(!hasTagName(HTMLNames::templateTag));
781
782 executeNodeInsertionWithScriptAssertion(*this, newChild, ChildChange::Source::Parser, ReplacedAllChildren::No, [&] {
783 if (&document() != &newChild.document())
784 document().adoptNode(newChild);
785
786 appendChildCommon(newChild);
787 newChild.setTreeScopeRecursively(treeScope());
788 newChild.updateAncestorConnectedSubframeCountForInsertion();
789 });
790}

862static void dispatchChildInsertionEvents(Node& child)
863{
864 if (child.isInShadowTree())
865 return;
866
867 ASSERT_WITH_SECURITY_IMPLICATION(ScriptDisallowedScope::InMainThread::isEventDispatchAllowedInSubtree(child));
868
869 RefPtr<Node> c = &child;
870 Ref<Document> document(child.document());
871
872 if (c->parentNode() && document->hasListenerType(Document::DOMNODEINSERTED_LISTENER))
873 c->dispatchScopedEvent(MutationEvent::create(eventNames().DOMNodeInsertedEvent, Event::CanBubble::Yes, c->parentNode()));
874
875 // dispatch the DOMNodeInsertedIntoDocument event to all descendants
876 if (c->isConnected() && document->hasListenerType(Document::DOMNODEINSERTEDINTODOCUMENT_LISTENER)) {
877 for (; c; c = NodeTraversal::next(*c, &child))
878 c->dispatchScopedEvent(MutationEvent::create(eventNames().DOMNodeInsertedIntoDocumentEvent, Event::CanBubble::No));
879 }
880}

882static void dispatchChildRemovalEvents(Ref<Node>& child)
883{
884 ASSERT_WITH_SECURITY_IMPLICATION(ScriptDisallowedScope::InMainThread::isEventDispatchAllowedInSubtree(child));
885 InspectorInstrumentation::willRemoveDOMNode(child->document(), child.get());
886
887 if (child->isInShadowTree())
888 return;
889
890 Ref<Document> document = child->document();
891
892 // dispatch pre-removal mutation events
893 if (child->parentNode() && document->hasListenerType(Document::DOMNODEREMOVED_LISTENER))
894 child->dispatchScopedEvent(MutationEvent::create(eventNames().DOMNodeRemovedEvent, Event::CanBubble::Yes, child->parentNode()));
895
896 // dispatch the DOMNodeRemovedFromDocument event to all descendants
897 if (child->isConnected() && document->hasListenerType(Document::DOMNODEREMOVEDFROMDOCUMENT_LISTENER)) {
898 for (RefPtr<Node> currentNode = child.copyRef(); currentNode; currentNode = NodeTraversal::next(*currentNode, child.ptr()))
899 currentNode->dispatchScopedEvent(MutationEvent::create(eventNames().DOMNodeRemovedFromDocumentEvent, Event::CanBubble::No));
900 }

1000// https://dom.spec.whatwg.org/#dom-parentnode-replacechildren
1001ExceptionOr<void> ContainerNode::replaceChildren(Vector<NodeOrString>&& vector)
1002{
1003 // step 1
1004 auto result = convertNodesOrStringsIntoNode(WTFMove(vector));
1005 if (result.hasException())
1006 return result.releaseException();
1007 auto node = result.releaseReturnValue();
1008
1009 // step 2
1010 if (node) {
1011 if (auto checkResult = ensurePreInsertionValidity(*node, nullptr); checkResult.hasException())
1012 return checkResult;
1013 }
1014
1015 // step 3
1016 auto protectedThis = makeRef(*this);
1017 ChildListMutationScope mutation(*this);
1018 removeAllChildrenWithScriptAssertion(ChildChange::Source::API, DeferChildrenChanged::No);
1019
1020 if (node) {
1021 if (auto appendResult = appendChildWithoutPreInsertionValidityCheck(*node); appendResult.hasException())
1022 return appendResult;
1023 }
1024
1025 rebuildSVGExtensionsElementsIfNecessary();
1026 dispatchSubtreeModifiedEvent();
1027
1028 return { };
1029}
