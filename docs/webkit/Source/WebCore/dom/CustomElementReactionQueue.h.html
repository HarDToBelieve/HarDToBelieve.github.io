97class CustomElementReactionDisallowedScope {
98public:
99 CustomElementReactionDisallowedScope()
100 {
101#if ASSERT_ENABLED
102 s_customElementReactionDisallowedCount++;
103#endif
104 }
105
106 ~CustomElementReactionDisallowedScope()
107 {
108#if ASSERT_ENABLED
109 ASSERT(s_customElementReactionDisallowedCount);
110 s_customElementReactionDisallowedCount--;
111#endif
112 }
113
114#if ASSERT_ENABLED
115 static bool isReactionAllowed() { return !s_customElementReactionDisallowedCount; }
116#endif
117
118 class AllowedScope {
119#if ASSERT_ENABLED
120 public:
121 AllowedScope()
122 : m_originalCount(s_customElementReactionDisallowedCount)
123 {
124 s_customElementReactionDisallowedCount = 0;
125 }
126
127 ~AllowedScope()
128 {
129 s_customElementReactionDisallowedCount = m_originalCount;
130 }
131
132 private:
133 unsigned m_originalCount;
134#endif // ASSERT_ENABLED
135 };
136
137private:
138#if ASSERT_ENABLED
139 WEBCORE_EXPORT static unsigned s_customElementReactionDisallowedCount;
140
141 friend class AllowedScope;
142#endif
143};

145class CustomElementReactionStack : public CustomElementReactionDisallowedScope::AllowedScope {
146public:
147 ALWAYS_INLINE CustomElementReactionStack(JSC::JSGlobalObject* state)
148 : m_previousProcessingStack(s_currentProcessingStack)
149 , m_state(state)
150 {
151 s_currentProcessingStack = this;
152 }
153
154 ALWAYS_INLINE CustomElementReactionStack(JSC::JSGlobalObject& state)
155 : CustomElementReactionStack(&state)
156 { }
157
158 ALWAYS_INLINE ~CustomElementReactionStack()
159 {
160 if (UNLIKELY(m_queue))
161 processQueue(m_state);
162 s_currentProcessingStack = m_previousProcessingStack;
163 }
164
165private:
166 WEBCORE_EXPORT void processQueue(JSC::JSGlobalObject*);
167
168 CustomElementQueue* m_queue { nullptr }; // Use raw pointer to avoid generating delete in the destructor.
169 CustomElementReactionStack* m_previousProcessingStack;
170 JSC::JSGlobalObject* m_state;
171
172 WEBCORE_EXPORT static CustomElementReactionStack* s_currentProcessingStack;
173
174 friend CustomElementReactionQueue;
175};
