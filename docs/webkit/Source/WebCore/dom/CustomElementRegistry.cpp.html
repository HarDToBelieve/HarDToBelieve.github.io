56// https://dom.spec.whatwg.org/#concept-shadow-including-tree-order
57static void enqueueUpgradeInShadowIncludingTreeOrder(ContainerNode& node, JSCustomElementInterface& elementInterface)
58{
59 for (Element* element = ElementTraversal::firstWithin(node); element; element = ElementTraversal::next(*element)) {
60 if (element->isCustomElementUpgradeCandidate() && element->tagQName() == elementInterface.name())
61 element->enqueueToUpgrade(elementInterface);
62 if (auto* shadowRoot = element->shadowRoot()) {
63 if (shadowRoot->mode() != ShadowRootMode::UserAgent)
64 enqueueUpgradeInShadowIncludingTreeOrder(*shadowRoot, elementInterface);
65 }
66 }
67}

69RefPtr<DeferredPromise> CustomElementRegistry::addElementDefinition(Ref<JSCustomElementInterface>&& elementInterface)
70{
71 AtomString localName = elementInterface->name().localName();
72 ASSERT(!m_nameMap.contains(localName));
73 m_constructorMap.add(elementInterface->constructor(), elementInterface.ptr());
74 m_nameMap.add(localName, elementInterface.copyRef());
75
76 if (auto* document = m_window.document())
77 enqueueUpgradeInShadowIncludingTreeOrder(*document, elementInterface.get());
78
79 return m_promiseMap.take(localName);
80}

116static void upgradeElementsInShadowIncludingDescendants(ContainerNode& root)
117{
118 for (auto& element : descendantsOfType<Element>(root)) {
119 if (element.isCustomElementUpgradeCandidate())
120 CustomElementReactionQueue::tryToUpgradeElement(element);
121 if (auto* shadowRoot = element.shadowRoot())
122 upgradeElementsInShadowIncludingDescendants(*shadowRoot);
123 }
124}
125
126void CustomElementRegistry::upgrade(Node& root)
127{
128 if (!is<ContainerNode>(root))
129 return;
130
131 if (is<Element>(root) && downcast<Element>(root).isCustomElementUpgradeCandidate())
132 CustomElementReactionQueue::tryToUpgradeElement(downcast<Element>(root));
133
134 upgradeElementsInShadowIncludingDescendants(downcast<ContainerNode>(root));
135}
