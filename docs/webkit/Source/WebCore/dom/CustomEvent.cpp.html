33namespace WebCore {
34
35WTF_MAKE_ISO_ALLOCATED_IMPL(CustomEvent);
36
37inline CustomEvent::CustomEvent(IsTrusted isTrusted)
38 : Event(isTrusted)
39{
40}
41
42inline CustomEvent::CustomEvent(const AtomString& type, const Init& initializer, IsTrusted isTrusted)
43 : Event(type, initializer, isTrusted)
44 , m_detail(initializer.detail)
45{
46}
47
48CustomEvent::~CustomEvent() = default;
49
50Ref<CustomEvent> CustomEvent::create(IsTrusted isTrusted)
51{
52 return adoptRef(*new CustomEvent(isTrusted));
53}
54
55Ref<CustomEvent> CustomEvent::create(const AtomString& type, const Init& initializer, IsTrusted isTrusted)
56{
57 return adoptRef(*new CustomEvent(type, initializer, isTrusted));
58}
59
60void CustomEvent::initCustomEvent(const AtomString& type, bool canBubble, bool cancelable, JSC::JSValue detail)
61{
62 if (isBeingDispatched())
63 return;
64
65 initEvent(type, canBubble, cancelable);
66
67 m_detail = detail;
68 m_cachedDetail = { };
69}
70
71EventInterface CustomEvent::eventInterface() const
72{
73 return CustomEventInterfaceType;
74}
