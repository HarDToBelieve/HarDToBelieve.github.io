29#include "DOMRect.h"

31namespace WebCore {
32
33DOMRectList::DOMRectList(const Vector<FloatQuad>& quads)
34{
35 m_items.reserveInitialCapacity(quads.size());
36 for (auto& quad : quads)
37 m_items.uncheckedAppend(DOMRect::create(quad.boundingBox()));
38}
39
40DOMRectList::DOMRectList(const Vector<FloatRect>& rects)
41{
42 m_items.reserveInitialCapacity(rects.size());
43 for (auto& rect : rects)
44 m_items.uncheckedAppend(DOMRect::create(rect));
45}
46
47DOMRectList::~DOMRectList() = default;
48
