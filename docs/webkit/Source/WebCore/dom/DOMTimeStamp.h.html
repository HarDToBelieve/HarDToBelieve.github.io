31#pragma once
32
33#include <wtf/Seconds.h>
34
35namespace WebCore {
36
37typedef unsigned long long DOMTimeStamp;
38
39inline DOMTimeStamp convertSecondsToDOMTimeStamp(double seconds)
40{
41 return static_cast<DOMTimeStamp>(seconds * 1000.0);
42}
43
44inline DOMTimeStamp convertSecondsToDOMTimeStamp(Seconds seconds)
45{
46 return static_cast<DOMTimeStamp>(seconds.milliseconds());
47}
48
49inline double convertDOMTimeStampToSeconds(DOMTimeStamp milliseconds)
50{
51 return milliseconds / 1000.0;
52}
53
54} // namespace WebCore
