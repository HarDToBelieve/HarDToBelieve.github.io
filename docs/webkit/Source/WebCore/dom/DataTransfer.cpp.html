123static String normalizeType(const String& type)
124{
125 if (type.isNull())
126 return type;
127
128 String lowercaseType = stripLeadingAndTrailingHTMLSpaces(type).convertToASCIILowercase();
129 if (lowercaseType == "text" || lowercaseType.startsWith("text/plain;"))
130 return "text/plain";
131 if (lowercaseType == "url" || lowercaseType.startsWith("text/uri-list;"))
132 return "text/uri-list";
133 if (lowercaseType.startsWith("text/html;"))
134 return "text/html";
135
136 return lowercaseType;
137}

139void DataTransfer::clearData(const String& type)
140{
141 if (!canWriteData())
142 return;
143
144 String normalizedType = normalizeType(type);
145 if (normalizedType.isNull())
146 m_pasteboard->clear();
147 else
148 m_pasteboard->clear(normalizedType);
149 if (m_itemList)
150 m_itemList->didClearStringData(normalizedType);
151}

153static String readURLsFromPasteboardAsString(Pasteboard& pasteboard, Function<bool(const String&)>&& shouldIncludeURL)
154{
155 StringBuilder urlList;
156 for (const auto& urlString : pasteboard.readAllStrings("text/uri-list"_s)) {
157 if (!shouldIncludeURL(urlString))
158 continue;
159 if (!urlList.isEmpty())
160 urlList.append(newlineCharacter);
161 urlList.append(urlString);
162 }
163 return urlList.toString();
164}

166String DataTransfer::getDataForItem(Document& document, const String& type) const
167{
168 if (!canReadData())
169 return { };
170
171 auto lowercaseType = stripLeadingAndTrailingHTMLSpaces(type).convertToASCIILowercase();
172 if (shouldSuppressGetAndSetDataToAvoidExposingFilePaths()) {
173 if (lowercaseType == "text/uri-list") {
174 return readURLsFromPasteboardAsString(*m_pasteboard, [] (auto& urlString) {
175 return Pasteboard::canExposeURLToDOMWhenPasteboardContainsFiles(urlString);
176 });
177 }
178
179 if (lowercaseType == "text/html" && RuntimeEnabledFeatures::sharedFeatures().customPasteboardDataEnabled()) {
180 // If the pasteboard contains files and the page requests 'text/html', we only read from rich text types to prevent file
181 // paths from leaking (e.g. from plain text data on the pasteboard) since we sanitize cross-origin markup. However, if
182 // custom pasteboard data is disabled, then we can't ensure that the markup we deliver is sanitized, so we fall back to
183 // current behavior and return an empty string.
184 return readStringFromPasteboard(document, lowercaseType, WebContentReadingPolicy::OnlyRichTextTypes);
185 }
186
187 return { };
188 }
189
190 return readStringFromPasteboard(document, lowercaseType, WebContentReadingPolicy::AnyType);
191}
192
193String DataTransfer::readStringFromPasteboard(Document& document, const String& lowercaseType, WebContentReadingPolicy policy) const
194{
195 if (!RuntimeEnabledFeatures::sharedFeatures().customPasteboardDataEnabled())
196 return m_pasteboard->readString(lowercaseType);
197
198 // StaticPasteboard is only used to stage data written by websites before being committed to the system pasteboard.
199 bool isSameOrigin = is<StaticPasteboard>(*m_pasteboard) || (!m_originIdentifier.isNull() && m_originIdentifier == m_pasteboard->readOrigin());
200 if (isSameOrigin) {
201 String value = m_pasteboard->readStringInCustomData(lowercaseType);
202 if (!value.isNull())
203 return value;
204 }
205 if (!Pasteboard::isSafeTypeForDOMToReadAndWrite(lowercaseType))
206 return { };
207
208 if (!is<StaticPasteboard>(*m_pasteboard) && lowercaseType == "text/html") {
209 if (!document.frame())
210 return { };
211 WebContentMarkupReader reader { *document.frame() };
212 m_pasteboard->read(reader, policy);
213 return reader.markup;
214 }
215
216 if (!is<StaticPasteboard>(*m_pasteboard) && lowercaseType == "text/uri-list") {
217 return readURLsFromPasteboardAsString(*m_pasteboard, [] (auto&) {
218 return true;
219 });
220 }
221
222 return m_pasteboard->readString(lowercaseType);
223}

237void DataTransfer::setData(const String& type, const String& data)
238{
239 if (!canWriteData())
240 return;
241
242 if (shouldSuppressGetAndSetDataToAvoidExposingFilePaths())
243 return;
244
245 auto normalizedType = normalizeType(type);
246 setDataFromItemList(normalizedType, data);
247 if (m_itemList)
248 m_itemList->didSetStringData(normalizedType);
249}
250
251void DataTransfer::setDataFromItemList(const String& type, const String& data)
252{
253 ASSERT(canWriteData());
254 RELEASE_ASSERT(is<StaticPasteboard>(*m_pasteboard));
255
256 if (!RuntimeEnabledFeatures::sharedFeatures().customPasteboardDataEnabled()) {
257 m_pasteboard->writeString(type, data);
258 return;
259 }
260
261 String sanitizedData;
262 if (type == "text/html")
263 sanitizedData = sanitizeMarkup(data);
264 else if (type == "text/uri-list") {
265 auto url = URL({ }, data);
266 if (url.isValid())
267 sanitizedData = url.string();
268 } else if (type == "text/plain")
269 sanitizedData = data; // Nothing to sanitize.
270
271 if (sanitizedData != data)
272 downcast<StaticPasteboard>(*m_pasteboard).writeStringInCustomData(type, data);
273
274 if (Pasteboard::isSafeTypeForDOMToReadAndWrite(type) && !sanitizedData.isNull())
275 m_pasteboard->writeString(type, sanitizedData);
276}

313Vector<String> DataTransfer::types(AddFilesType addFilesType) const
314{
315 if (!canReadTypes())
316 return { };
317
318 if (!RuntimeEnabledFeatures::sharedFeatures().customPasteboardDataEnabled()) {
319 auto types = m_pasteboard->typesForLegacyUnsafeBindings();
320 ASSERT(!types.contains("Files"));
321 if (m_pasteboard->fileContentState() != Pasteboard::FileContentState::NoFileOrImageData && addFilesType == AddFilesType::Yes)
322 types.append("Files");
323 return types;
324 }
325
326 auto safeTypes = m_pasteboard->typesSafeForBindings(m_originIdentifier);
327 bool hasFileBackedItem = m_itemList && m_itemList->hasItems() && notFound != m_itemList->items().findMatching([] (const auto& item) {
328 return item->isFile();
329 });
330
331 auto fileContentState = m_pasteboard->fileContentState();
332 if (hasFileBackedItem || fileContentState != Pasteboard::FileContentState::NoFileOrImageData) {
333 Vector<String> types;
334 if (addFilesType == AddFilesType::Yes)
335 types.append("Files"_s);
336
337 if (fileContentState != Pasteboard::FileContentState::MayContainFilePaths) {
338 types.appendVector(WTFMove(safeTypes));
339 return types;
340 }
341
342 if (safeTypes.contains("text/uri-list"))
343 types.append("text/uri-list"_s);
344 if (safeTypes.contains("text/html") && RuntimeEnabledFeatures::sharedFeatures().customPasteboardDataEnabled())
345 types.append("text/html"_s);
346 return types;
347 }
348
349 ASSERT(!safeTypes.contains("Files"));
350 return safeTypes;
351}

353Vector<Ref<File>> DataTransfer::filesFromPasteboardAndItemList(ScriptExecutionContext* context) const
354{
355 bool addedFilesFromPasteboard = false;
356 Vector<Ref<File>> files;
357 if ((!forDrag() || forFileDrag()) && m_pasteboard->fileContentState() != Pasteboard::FileContentState::NoFileOrImageData) {
358 WebCorePasteboardFileReader reader(context);
359 m_pasteboard->read(reader);
360 files = WTFMove(reader.files);
361 addedFilesFromPasteboard = !files.isEmpty();
362 }
363
364 bool itemListContainsItems = false;
365 if (m_itemList && m_itemList->hasItems()) {
366 for (auto& item : m_itemList->items()) {
367 if (auto file = item->file())
368 files.append(file.releaseNonNull());
369 }
370 itemListContainsItems = true;
371 }
372
373 bool containsItemsAndFiles = itemListContainsItems && addedFilesFromPasteboard;
374 ASSERT_UNUSED(containsItemsAndFiles, !containsItemsAndFiles);
375 return files;
376}

436void DataTransfer::commitToPasteboard(Pasteboard& nativePasteboard)
437{
438 ASSERT(is<StaticPasteboard>(*m_pasteboard) && !is<StaticPasteboard>(nativePasteboard));
439 auto& staticPasteboard = downcast<StaticPasteboard>(*m_pasteboard);
440 if (!staticPasteboard.hasNonDefaultData()) {
441 // We clear the platform pasteboard here to ensure that the pasteboard doesn't contain any data
442 // that may have been written before starting the drag or copying, and after ending the last
443 // drag session or paste. After pushing the static pasteboard's contents to the platform, the
444 // pasteboard should only contain data that was in the static pasteboard.
445 nativePasteboard.clear();
446 return;
447 }
448
449 auto customData = staticPasteboard.takeCustomData();
450 if (RuntimeEnabledFeatures::sharedFeatures().customPasteboardDataEnabled()) {
451 customData.setOrigin(m_originIdentifier);
452 nativePasteboard.writeCustomData({ customData });
453 return;
454 }
455
456 nativePasteboard.clear();
457 customData.forEachPlatformString([&] (auto& type, auto& string) {
458 nativePasteboard.writeString(type, string);
459 });
460
461 customData.forEachCustomString([&] (auto& type, auto& string) {
462 nativePasteboard.writeString(type, string);
463 });
464}

520void DataTransfer::setDragImage(Element& element, int x, int y)
521{
522 if (!forDrag() || !canWriteData())
523 return;
524
525 CachedImage* image = nullptr;
526 if (is<HTMLImageElement>(element) && !element.isConnected())
527 image = downcast<HTMLImageElement>(element).cachedImage();
528
529 m_dragLocation = IntPoint(x, y);
530
531 if (m_dragImageLoader && m_dragImage)
532 m_dragImageLoader->stopLoading(m_dragImage);
533 m_dragImage = image;
534 if (m_dragImage) {
535 if (!m_dragImageLoader)
536 m_dragImageLoader = makeUnique<DragImageLoader>(this);
537 m_dragImageLoader->startLoading(m_dragImage);
538 }
539
540 m_dragImageElement = image ? nullptr : &element;
541
542 updateDragImage();
543}

545void DataTransfer::updateDragImage()
546{
547 // Don't allow setting the image if we haven't started dragging yet; we'll rely on the dragging code

552 IntPoint computedHotSpot;

554 if (!computedImage)
555 return;

557 m_pasteboard->setDragImage(WTFMove(computedImage), computedHotSpot);
558}

611static OptionSet<DragOperation> dragOpFromIEOp(const String& operation)
612{
613 if (operation == "uninitialized")
614 return anyDragOperation();
615 if (operation == "none")
616 return { };
617 if (operation == "copy")
618 return { DragOperation::Copy };
619 if (operation == "link")
620 return { DragOperation::Link };
621 if (operation == "move")
622 return { DragOperation::Generic, DragOperation::Move };
623 if (operation == "copyLink")
624 return { DragOperation::Copy, DragOperation::Link };
625 if (operation == "copyMove")
626 return { DragOperation::Copy, DragOperation::Generic, DragOperation::Move };
627 if (operation == "linkMove")
628 return { DragOperation::Link, DragOperation::Generic, DragOperation::Move };
629 if (operation == "all")
630 return anyDragOperation();
631 return { DragOperation::Private }; // Really a marker for "no conversion".
632}
633
634static const char* IEOpFromDragOp(OptionSet<DragOperation> operationMask)
635{
636 bool isGenericMove = operationMask.containsAny({ DragOperation::Generic, DragOperation::Move });
637
638 if ((isGenericMove && operationMask.containsAll({ DragOperation::Copy, DragOperation::Link })) || operationMask.containsAll({ DragOperation::Copy, DragOperation::Link, DragOperation::Generic, DragOperation::Private, DragOperation::Move, DragOperation::Delete }))
639 return "all";
640 if (isGenericMove && operationMask.contains(DragOperation::Copy))
641 return "copyMove";
642 if (isGenericMove && operationMask.contains(DragOperation::Link))
643 return "linkMove";
644 if (operationMask.containsAll({ DragOperation::Copy, DragOperation::Link }))
645 return "copyLink";
646 if (isGenericMove)
647 return "move";
648 if (operationMask.contains(DragOperation::Copy))
649 return "copy";
650 if (operationMask.contains(DragOperation::Link))
651 return "link";
652 return "none";
653}

722void DataTransfer::moveDragState(Ref<DataTransfer>&& other)
723{
724 RELEASE_ASSERT(is<StaticPasteboard>(other->pasteboard()));
