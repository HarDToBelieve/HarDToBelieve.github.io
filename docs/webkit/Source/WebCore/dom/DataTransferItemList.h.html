32#pragma once
33
34#include "ContextDestructionObserver.h"
35#include "DataTransfer.h"
36#include "ExceptionOr.h"
37#include "ScriptWrappable.h"
38#include <wtf/Forward.h>
39#include <wtf/Ref.h>
40#include <wtf/RefCounted.h>
41#include <wtf/WeakPtr.h>
42
43namespace WebCore {
44
45class DataTransferItem;
46class File;
47
48class DataTransferItemList final : public ScriptWrappable, public ContextDestructionObserver, public CanMakeWeakPtr<DataTransferItemList> {
49 WTF_MAKE_NONCOPYABLE(DataTransferItemList);
50 WTF_MAKE_ISO_ALLOCATED(DataTransferItemList);
51public:
52 DataTransferItemList(Document&, DataTransfer&);
53 ~DataTransferItemList();
54
55 // DataTransfer owns DataTransferItemList, and DataTransfer is kept alive as long as DataTransferItemList is alive.
56 void ref() { m_dataTransfer.ref(); }
57 void deref() { m_dataTransfer.deref(); }
58 DataTransfer& dataTransfer() { return m_dataTransfer; }
59
60 // DOM API
61 unsigned length() const;
62 RefPtr<DataTransferItem> item(unsigned index);
63 ExceptionOr<RefPtr<DataTransferItem>> add(const String& data, const String& type);
64 RefPtr<DataTransferItem> add(Ref<File>&&);
65 ExceptionOr<void> remove(unsigned index);
66 void clear();
67
68 void didClearStringData(const String& type);
69 void didSetStringData(const String& type);
70 bool hasItems() const { return m_items.has_value(); }
71 const Vector<Ref<DataTransferItem>>& items() const
72 {
73 ASSERT(m_items);
74 return *m_items;
75 }
76
77private:
78 Vector<Ref<DataTransferItem>>& ensureItems() const;
79 Document* document() const;
80
81 DataTransfer& m_dataTransfer;
82 mutable std::optional<Vector<Ref<DataTransferItem>>> m_items;
83};
84
85} // namespace WebCore
