35DecodedDataDocumentParser::DecodedDataDocumentParser(Document& document)
36 : DocumentParser(document)
37{
38}
39
40void DecodedDataDocumentParser::appendBytes(DocumentWriter& writer, const uint8_t* data, size_t length)
41{
42 if (!length)
43 return;
44
45 String decoded = writer.decoder().decode(data, length);
46 if (decoded.isEmpty())
47 return;
48
49 writer.reportDataReceived();
50 append(decoded.releaseImpl());
51}
52
53void DecodedDataDocumentParser::flush(DocumentWriter& writer)
54{
55 String remainingData = writer.decoder().flush();
56 if (remainingData.isEmpty())
57 return;
58
59 writer.reportDataReceived();
60 append(remainingData.releaseImpl());
61}
