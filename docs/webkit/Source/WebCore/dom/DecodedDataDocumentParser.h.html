32class DecodedDataDocumentParser : public DocumentParser {

35 // XMLHttpRequest if the responseXML was well formed.

38protected:
39 explicit DecodedDataDocumentParser(Document&);
40
41private:
42 // append is used by DocumentWriter::replaceDocument.
43 void append(RefPtr<StringImpl>&&) override = 0;
44
45 // appendBytes and flush are used by DocumentWriter (the loader).
46 void appendBytes(DocumentWriter&, const uint8_t* bytes, size_t length) override;
47 void flush(DocumentWriter&) override;
48};
