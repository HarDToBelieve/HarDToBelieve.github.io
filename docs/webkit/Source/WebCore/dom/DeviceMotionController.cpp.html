38DeviceMotionController::DeviceMotionController(DeviceMotionClient& client)
39 : DeviceController(client)
40{
41 deviceMotionClient().setController(this);
42}
43
44#if PLATFORM(IOS_FAMILY)
45
46// FIXME: We should look to reconcile the iOS vs. non-iOS differences with this class
47// so that we can either remove these functions or the PLATFORM(IOS_FAMILY)-guard.
48
49void DeviceMotionController::suspendUpdates()
50{
51 m_client.stopUpdating();
52}
53
54void DeviceMotionController::resumeUpdates()
55{
56 if (!m_listeners.isEmpty())
57 m_client.startUpdating();
58}
59
60#endif
61
62void DeviceMotionController::didChangeDeviceMotion(DeviceMotionData* deviceMotionData)
63{
64 dispatchDeviceEvent(DeviceMotionEvent::create(eventNames().devicemotionEvent, deviceMotionData));
65}
66
67DeviceMotionClient& DeviceMotionController::deviceMotionClient()
68{
69 return static_cast<DeviceMotionClient&>(m_client);
70}
71
72bool DeviceMotionController::hasLastData()
73{
74 return deviceMotionClient().lastMotion();
75}
76
77RefPtr<Event> DeviceMotionController::getLastEvent()
78{
79 return DeviceMotionEvent::create(eventNames().devicemotionEvent, deviceMotionClient().lastMotion());
80}
81
82const char* DeviceMotionController::supplementName()
83{
84 return "DeviceMotionController";
85}
86
87DeviceMotionController* DeviceMotionController::from(Page* page)
88{
89 return static_cast<DeviceMotionController*>(Supplement<Page>::from(page, supplementName()));
90}
91
92bool DeviceMotionController::isActiveAt(Page* page)
93{
94 if (DeviceMotionController* self = DeviceMotionController::from(page))
95 return self->isActive();
96 return false;
97}
