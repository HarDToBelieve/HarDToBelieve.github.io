26#pragma once
27
28#include "DeviceOrientationOrMotionPermissionState.h"
29#include "Event.h"
30#include "IDLTypes.h"

32namespace WebCore {
33
34class DeviceMotionData;
35template<typename IDLType> class DOMPromiseDeferred;
36
37class DeviceMotionEvent final : public Event {
38 WTF_MAKE_ISO_ALLOCATED(DeviceMotionEvent);
39public:
40 virtual ~DeviceMotionEvent();
41
42 // FIXME: Merge this with DeviceMotionData::Acceleration
43 struct Acceleration {
44 std::optional<double> x;
45 std::optional<double> y;
46 std::optional<double> z;
47 };
48
49 // FIXME: Merge this with DeviceMotionData::RotationRate
50 struct RotationRate {
51 std::optional<double> alpha;
52 std::optional<double> beta;
53 std::optional<double> gamma;
54 };
55
56 static Ref<DeviceMotionEvent> create(const AtomString& eventType, DeviceMotionData* deviceMotionData)
57 {
58 return adoptRef(*new DeviceMotionEvent(eventType, deviceMotionData));
59 }
60
61 static Ref<DeviceMotionEvent> createForBindings()
62 {
63 return adoptRef(*new DeviceMotionEvent);
64 }
65
66 std::optional<Acceleration> acceleration() const;
67 std::optional<Acceleration> accelerationIncludingGravity() const;
68 std::optional<RotationRate> rotationRate() const;
69 std::optional<double> interval() const;
70
71 void initDeviceMotionEvent(const AtomString& type, bool bubbles, bool cancelable, std::optional<Acceleration>&&, std::optional<Acceleration>&&, std::optional<RotationRate>&&, std::optional<double>);
72
73#if ENABLE(DEVICE_ORIENTATION)
74 using PermissionState = DeviceOrientationOrMotionPermissionState;
75 using PermissionPromise = DOMPromiseDeferred<IDLEnumeration<PermissionState>>;
76 static void requestPermission(Document&, PermissionPromise&&);
77#endif
78
79private:
80 DeviceMotionEvent();
81 DeviceMotionEvent(const AtomString& eventType, DeviceMotionData*);
82
83 EventInterface eventInterface() const override;
84
85 RefPtr<DeviceMotionData> m_deviceMotionData;
86};

88} // namespace WebCore
