26#pragma once
27
28#include <optional>
29#include <wtf/Ref.h>
30#include <wtf/RefCounted.h>
31
32namespace WebCore {
33
34class DeviceOrientationData : public RefCounted<DeviceOrientationData> {
35public:
36 static Ref<DeviceOrientationData> create()
37 {
38 return adoptRef(*new DeviceOrientationData);
39 }
40
41#if PLATFORM(IOS_FAMILY)
42 WEBCORE_EXPORT static Ref<DeviceOrientationData> create(std::optional<double> alpha, std::optional<double> beta, std::optional<double> gamma, std::optional<double> compassHeading, std::optional<double> compassAccuracy);
43#else
44 WEBCORE_EXPORT static Ref<DeviceOrientationData> create(std::optional<double> alpha, std::optional<double> beta, std::optional<double> gamma, std::optional<bool> absolute);
45#endif
46
47 std::optional<double> alpha() const { return m_alpha; }
48 std::optional<double> beta() const { return m_beta; }
49 std::optional<double> gamma() const { return m_gamma; }
50#if PLATFORM(IOS_FAMILY)
51 std::optional<double> compassHeading() const { return m_compassHeading; }
52 std::optional<double> compassAccuracy() const { return m_compassAccuracy; }
53#else
54 std::optional<bool> absolute() const { return m_absolute; }
55#endif
56
57private:
58 DeviceOrientationData() = default;
59#if PLATFORM(IOS_FAMILY)
60 DeviceOrientationData(std::optional<double> alpha, std::optional<double> beta, std::optional<double> gamma, std::optional<double> compassHeading, std::optional<double> compassAccuracy);
61#else
62 DeviceOrientationData(std::optional<double> alpha, std::optional<double> beta, std::optional<double> gamma, std::optional<bool> absolute);
63#endif
64
65 std::optional<double> m_alpha;
66 std::optional<double> m_beta;
67 std::optional<double> m_gamma;
68#if PLATFORM(IOS_FAMILY)
69 std::optional<double> m_compassHeading;
70 std::optional<double> m_compassAccuracy;
71#else
72 std::optional<bool> m_absolute;
73#endif
74};
75
76} // namespace WebCore
