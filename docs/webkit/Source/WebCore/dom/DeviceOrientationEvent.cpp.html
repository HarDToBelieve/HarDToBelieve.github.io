118void DeviceOrientationEvent::requestPermission(Document& document, PermissionPromise&& promise)
119{
120 auto* window = document.domWindow();
121 auto* page = document.page();
122 if (!window || !page)
123 return promise.reject(Exception { InvalidStateError, "No browsing context"_s });
124
125 String errorMessage;
126 if (!window->isAllowedToUseDeviceOrientation(errorMessage)) {
127 document.addConsoleMessage(MessageSource::JS, MessageLevel::Warning, makeString("Call to requestPermission() failed, reason: ", errorMessage, "."));
128 return promise.resolve(PermissionState::Denied);
129 }
130
131 document.deviceOrientationAndMotionAccessController().shouldAllowAccess(document, [promise = WTFMove(promise)](PermissionState permissionState) mutable {
132 if (permissionState == PermissionState::Prompt)
133 return promise.reject(Exception { NotAllowedError, "Requesting device orientation access requires a user gesture to prompt"_s });
134 promise.resolve(permissionState);
135 });
136}
137#endif
