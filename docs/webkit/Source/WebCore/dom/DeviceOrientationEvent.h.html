28#include "DeviceOrientationOrMotionPermissionState.h"
29#include "Event.h"
30#include "IDLTypes.h"

32namespace WebCore {
33
34class DeviceOrientationData;
35template<typename IDLType> class DOMPromiseDeferred;
36
37class DeviceOrientationEvent final : public Event {
38 WTF_MAKE_ISO_ALLOCATED(DeviceOrientationEvent);
39public:
40 static Ref<DeviceOrientationEvent> create(const AtomString& eventType, DeviceOrientationData* orientation)
41 {
42 return adoptRef(*new DeviceOrientationEvent(eventType, orientation));
43 }
44
45 static Ref<DeviceOrientationEvent> createForBindings()
46 {
47 return adoptRef(*new DeviceOrientationEvent);
48 }
49
50 virtual ~DeviceOrientationEvent();
51
52 std::optional<double> alpha() const;
53 std::optional<double> beta() const;
54 std::optional<double> gamma() const;
55
56#if PLATFORM(IOS_FAMILY)
57 std::optional<double> compassHeading() const;
58 std::optional<double> compassAccuracy() const;
59
60 void initDeviceOrientationEvent(const AtomString& type, bool bubbles, bool cancelable, std::optional<double> alpha, std::optional<double> beta, std::optional<double> gamma, std::optional<double> compassHeading, std::optional<double> compassAccuracy);
61#else
62 std::optional<bool> absolute() const;
63
64 void initDeviceOrientationEvent(const AtomString& type, bool bubbles, bool cancelable, std::optional<double> alpha, std::optional<double> beta, std::optional<double> gamma, std::optional<bool> absolute);
65#endif
66
67#if ENABLE(DEVICE_ORIENTATION)
68 using PermissionState = DeviceOrientationOrMotionPermissionState;
69 using PermissionPromise = DOMPromiseDeferred<IDLEnumeration<PermissionState>>;
70 static void requestPermission(Document&, PermissionPromise&&);
71#endif
72
73private:
74 DeviceOrientationEvent();
75 DeviceOrientationEvent(const AtomString& eventType, DeviceOrientationData*);
76
77 EventInterface eventInterface() const override;
78
79 RefPtr<DeviceOrientationData> m_orientation;
80};
