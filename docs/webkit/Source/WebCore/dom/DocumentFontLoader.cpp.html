41DocumentFontLoader::DocumentFontLoader(Document& document)
42 : m_document(document)
43 , m_fontLoadingTimer(*this, &DocumentFontLoader::fontLoadingTimerFired)
44{
45}

47DocumentFontLoader::~DocumentFontLoader()
48{
49 stopLoadingAndClearFonts();

52CachedFont* DocumentFontLoader::cachedFont(URL&& url, bool isSVG, bool isInitiatingElementInUserAgentShadowTree, LoadedFromOpaqueSource loadedFromOpaqueSource)
53{
54 ResourceLoaderOptions options = CachedResourceLoader::defaultCachedResourceOptions();
55 options.contentSecurityPolicyImposition = isInitiatingElementInUserAgentShadowTree ? ContentSecurityPolicyImposition::SkipPolicyCheck : ContentSecurityPolicyImposition::DoPolicyCheck;
56 options.loadedFromOpaqueSource = loadedFromOpaqueSource;
57
58 CachedResourceRequest request(ResourceRequest(WTFMove(url)), options);
59 request.setInitiator(cachedResourceRequestInitiators().css);
60 return m_document.cachedResourceLoader().requestFont(WTFMove(request), isSVG).value_or(nullptr).get();
61}

63void DocumentFontLoader::beginLoadingFontSoon(CachedFont& font)
64{
65 if (m_isStopped)
66 return;
67
68 m_fontsToBeginLoading.append(&font);
69 // Increment the request count now, in order to prevent didFinishLoad from being dispatched
70 // after this font has been requested but before it began loading. Balanced by
71 // decrementRequestCount() in fontLoadingTimerFired() and in stopLoadingAndClearFonts().
72 m_document.cachedResourceLoader().incrementRequestCount(font);
73
74 if (!m_isFontLoadingSuspended && !m_fontLoadingTimer.isActive())
75 m_fontLoadingTimer.startOneShot(0_s);
76}
77
78void DocumentFontLoader::loadPendingFonts()
79{
80 if (m_isFontLoadingSuspended)
81 return;
82
83 Vector<CachedResourceHandle<CachedFont>> fontsToBeginLoading;
84 fontsToBeginLoading.swap(m_fontsToBeginLoading);
85
86 auto& cachedResourceLoader = m_document.cachedResourceLoader();
87 for (auto& fontHandle : fontsToBeginLoading) {
88 fontHandle->beginLoadIfNeeded(cachedResourceLoader);
89 // Balances incrementRequestCount() in beginLoadingFontSoon().
90 cachedResourceLoader.decrementRequestCount(*fontHandle);
91 }
92}
93
94void DocumentFontLoader::fontLoadingTimerFired()
95{
96 loadPendingFonts();
97
98 // FIXME: Use SubresourceLoader instead.
99 // Call FrameLoader::loadDone before FrameLoader::subresourceLoadDone to match the order in SubresourceLoader::notifyDone.
100 m_document.cachedResourceLoader().loadDone(LoadCompletionType::Finish);
101 // Ensure that if the request count reaches zero, the frame loader will know about it.
102 // New font loads may be triggered by layout after the document load is complete but before we have dispatched
103 // didFinishLoading for the frame. Make sure the delegate is always dispatched by checking explicitly.
104 if (m_document.frame())
105 m_document.frame()->loader().checkLoadComplete();
106}

108void DocumentFontLoader::stopLoadingAndClearFonts()
109{
110 if (m_isStopped)
111 return;
112
113 m_fontLoadingTimer.stop();
114 auto& cachedResourceLoader = m_document.cachedResourceLoader();
115 for (auto& fontHandle : m_fontsToBeginLoading) {
116 // Balances incrementRequestCount() in beginLoadingFontSoon().
117 cachedResourceLoader.decrementRequestCount(*fontHandle);
118 }
119 m_fontsToBeginLoading.clear();
120 m_document.fontSelector().clearFonts();
121
122 m_isFontLoadingSuspended = true;
123 m_isStopped = true;
124}
125
126void DocumentFontLoader::suspendFontLoading()
127{
128 if (m_isFontLoadingSuspended)
129 return;
130
131 m_fontLoadingTimer.stop();
132 m_isFontLoadingSuspended = true;
133}
134
135void DocumentFontLoader::resumeFontLoading()
136{
137 if (!m_isFontLoadingSuspended || m_isStopped)
138 return;
139
140 m_isFontLoadingSuspended = false;
141 if (!m_fontsToBeginLoading.isEmpty())
142 m_fontLoadingTimer.startOneShot(0_s);
143}
