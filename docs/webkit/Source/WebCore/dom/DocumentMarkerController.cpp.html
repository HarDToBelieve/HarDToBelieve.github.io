63auto DocumentMarkerController::collectTextRanges(const SimpleRange& range) -> Vector<TextRange>
64{
65 Vector<TextRange> ranges;
66 for (TextIterator iterator(range); !iterator.atEnd(); iterator.advance()) {
67 auto currentRange = iterator.range();
68 ranges.append({ WTFMove(currentRange.start.container), { currentRange.start.offset, currentRange.end.offset } });
69 }
70 return ranges;
71}

96void DocumentMarkerController::filterMarkers(const SimpleRange& range, const Function<bool(const DocumentMarker&)>& filter, OptionSet<DocumentMarker::MarkerType> types, RemovePartiallyOverlappingMarker overlapRule)

154void DocumentMarkerController::updateRectsForInvalidatedMarkersOfType(DocumentMarker::MarkerType type)
155{
156 if (!possiblyHasMarkers(type))
157 return;
158 ASSERT(!m_markers.isEmpty());
159
160 bool updatedLayout = false;
161 for (auto& nodeMarkers : m_markers) {
162 for (auto& marker : *nodeMarkers.value) {
163 if (marker.type() != type || marker.isValid())
164 continue;
165 if (!updatedLayout) {
166 updateMainFrameLayoutIfNeeded(m_document);
167 updatedLayout = true;
168 }
169 updateRenderedRectsForMarker(marker, *nodeMarkers.key);
170 }
171 }

174Vector<FloatRect> DocumentMarkerController::renderedRectsForMarkers(DocumentMarker::MarkerType type)
175{
176 Vector<FloatRect> result;
177
178 if (!possiblyHasMarkers(type))
179 return result;
180 ASSERT(!m_markers.isEmpty());
181
182 RefPtr<Frame> frame = m_document.frame();
183 if (!frame)
184 return result;
185 FrameView* frameView = frame->view();
186 if (!frameView)
187 return result;
188
189 updateRectsForInvalidatedMarkersOfType(type);
190
191 bool isSubframe = !frame->isMainFrame();
192 IntRect subframeClipRect;
193 if (isSubframe)
194 subframeClipRect = frameView->windowToContents(frameView->windowClipRect());
195
196 for (auto& nodeMarkers : m_markers) {
197 auto renderer = nodeMarkers.key->renderer();
198 FloatRect overflowClipRect;
199 if (renderer)
200 overflowClipRect = renderer->absoluteClippedOverflowRectForRepaint();
201 for (auto& marker : *nodeMarkers.value) {
202 if (marker.type() != type)
203 continue;
204
205 auto renderedRects = marker.unclippedAbsoluteRects();
206
207 // Clip document markers by their overflow clip.
208 if (renderer) {
209 for (auto& rect : renderedRects)
210 rect.intersect(overflowClipRect);
211 }
212
213 // Clip subframe document markers by their frame.
214 if (isSubframe) {
215 for (auto& rect : renderedRects)
216 rect.intersect(subframeClipRect);
217 }
218
219 for (const auto& rect : renderedRects) {
220 if (!rect.isEmpty())
221 result.append(rect);
222 }
223 }
224 }
225
226 return result;
227}

229static bool shouldInsertAsSeparateMarker(const DocumentMarker& marker)
230{
231#if ENABLE(PLATFORM_DRIVEN_TEXT_CHECKING)
232 if (marker.type() == DocumentMarker::PlatformTextChecking)
233 return true;
234#endif
235
236#if PLATFORM(IOS_FAMILY)
237 if (marker.type() == DocumentMarker::DictationPhraseWithAlternatives || marker.type() == DocumentMarker::DictationResult)
238 return true;
239#endif
240
241 if (marker.type() == DocumentMarker::DraggedContent)
242 return is<RenderReplaced>(WTF::get<RefPtr<Node>>(marker.data())->renderer());
243
244 return false;
245}

247// Markers are stored in order sorted by their start offset.
248// Markers of the same type do not overlap each other.
249
250void DocumentMarkerController::addMarker(Node& node, DocumentMarker&& newMarker)
251{
252 ASSERT(newMarker.endOffset() >= newMarker.startOffset());
253 if (newMarker.endOffset() == newMarker.startOffset())
254 return;
255
256#if ENABLE(LAYOUT_FORMATTING_CONTEXT)
257 if (auto* renderer = node.renderer()) {
258 // FIXME: Factor the marker painting code out of LegacyInlineTextBox and teach simple line layout to use it.
259 if (auto* lineLayout = LayoutIntegration::LineLayout::containing(*renderer))
260 lineLayout->flow().ensureLineBoxes();
261 }
262#endif
263
264 m_possiblyExistingMarkerTypes.add(newMarker.type());
265
266 auto& list = m_markers.add(&node, nullptr).iterator->value;
267
268 if (!list) {
269 list = makeUnique<Vector<RenderedDocumentMarker>>();
270 list->append(RenderedDocumentMarker(WTFMove(newMarker)));
271 } else if (shouldInsertAsSeparateMarker(newMarker)) {
272 // We don't merge dictation markers.
273 size_t i;
274 size_t numberOfMarkers = list->size();
275 for (i = 0; i < numberOfMarkers; ++i) {
276 DocumentMarker marker = list->at(i);
277 if (marker.startOffset() > newMarker.startOffset())
278 break;
279 }
280 list->insert(i, RenderedDocumentMarker(WTFMove(newMarker)));
281 } else {
282 RenderedDocumentMarker toInsert(WTFMove(newMarker));
283 size_t numMarkers = list->size();
284 size_t i;
285 // Iterate over all markers whose start offset is less than or equal to the new marker's.
286 // If one of them is of the same type as the new marker and touches it or intersects with it
287 // (there is at most one), remove it and adjust the new marker's start offset to encompass it.
288 for (i = 0; i < numMarkers; ++i) {
289 DocumentMarker marker = list->at(i);
290 if (marker.startOffset() > toInsert.startOffset())
291 break;
292 if (marker.type() == toInsert.type() && marker.endOffset() >= toInsert.startOffset()) {
293 toInsert.setStartOffset(marker.startOffset());
294 list->remove(i);
295 numMarkers--;
296 break;
297 }
298 }
299 size_t j = i;
300 // Iterate over all markers whose end offset is less than or equal to the new marker's,
301 // removing markers of the same type as the new marker which touch it or intersect with it,
302 // adjusting the new marker's end offset to cover them if necessary.
303 while (j < numMarkers) {
304 DocumentMarker marker = list->at(j);
305 if (marker.startOffset() > toInsert.endOffset())
306 break;
307 if (marker.type() == toInsert.type()) {
308 list->remove(j);
309 if (toInsert.endOffset() <= marker.endOffset()) {
310 toInsert.setEndOffset(marker.endOffset());
311 break;
312 }
313 numMarkers--;
314 } else
315 j++;
316 }
317 // At this point i points to the node before which we want to insert.
318 list->insert(i, RenderedDocumentMarker(toInsert));
319 }
320
321 if (node.renderer())
322 node.renderer()->repaint();
323
324 invalidateRectsForMarkersInNode(node);
325}

327// Copies markers from source to destination, applying the specified shift delta to the copies. The shift is
328// useful if, e.g., the caller has created the destination from a non-prefix substring of the source.
329void DocumentMarkerController::copyMarkers(Node& source, OffsetRange range, Node& destination)
330{
331 if (range.start >= range.end)
332 return;
333
334 if (!possiblyHasMarkers(DocumentMarker::allMarkers()))
335 return;
336 ASSERT(!m_markers.isEmpty());
337
338 auto list = m_markers.get(&source);
339 if (!list)
340 return;
341
342 bool needRepaint = false;
343 for (auto& marker : *list) {
344 // Stop if we are now past the specified range.
345 if (marker.startOffset() >= range.end)
346 break;
347
348 // Skip marker that is before the specified range.
349 if (marker.endOffset() < range.start)
350 continue;
351
352 // Pin the marker to the specified range and apply the shift delta.
353 auto copiedMarker = marker;
354 if (copiedMarker.startOffset() < range.start)
355 copiedMarker.setStartOffset(range.start);
356 if (copiedMarker.endOffset() >= range.end)
357 copiedMarker.setEndOffset(range.end);
358
359 addMarker(destination, WTFMove(copiedMarker));
360 needRepaint = true;
361 }
362
363 if (needRepaint) {
364 if (auto renderer = destination.renderer())
365 renderer->repaint();
366 }
367}

369void DocumentMarkerController::removeMarkers(Node& node, OffsetRange range, OptionSet<DocumentMarker::MarkerType> types, const Function<bool(const DocumentMarker&)>& filter, RemovePartiallyOverlappingMarker overlapRule)
370{
371 if (range.start >= range.end)
372 return;
373
374 if (!possiblyHasMarkers(types))
375 return;
376 ASSERT(!m_markers.isEmpty());
377
378 auto list = m_markers.get(&node);
379 if (!list)
380 return;
381
382 bool needRepaint = false;
383 for (size_t i = 0; i < list->size(); ) {
384 auto& marker = list->at(i);
385
386 // markers are returned in order, so stop if we are now past the specified range
387 if (marker.startOffset() >= range.end)
388 break;
389
390 // skip marker that is wrong type or before target
391 if (marker.endOffset() <= range.start || !types.contains(marker.type())) {
392 i++;
393 continue;
394 }
395
396 if (filter && !filter(marker)) {
397 i++;
398 continue;
399 }
400
401 // At this point we know that marker and target intersect in some way.
402 needRepaint = true;
403
404 DocumentMarker copiedMarker = marker;
405 list->remove(i);
406 if (overlapRule == RemovePartiallyOverlappingMarker::Yes)
407 continue;
408
409 // Add either of the resulting slices that remain after removing target.
410 if (range.start > copiedMarker.startOffset()) {
411 auto newLeft = copiedMarker;
412 newLeft.setEndOffset(range.start);
413 list->insert(i, RenderedDocumentMarker(WTFMove(newLeft)));
414 i++;
415 }
416 if (copiedMarker.endOffset() > range.end) {
417 copiedMarker.setStartOffset(range.end);
418 list->insert(i, RenderedDocumentMarker(WTFMove(copiedMarker)));
419 i++;
420 }
421 }
422
423 if (list->isEmpty()) {
424 m_markers.remove(&node);
425 if (m_markers.isEmpty())
426 m_possiblyExistingMarkerTypes = { };
427 }
428
429 if (needRepaint) {
430 if (auto renderer = node.renderer())
431 renderer->repaint();
432 }
433}

435DocumentMarker* DocumentMarkerController::markerContainingPoint(const LayoutPoint& point, DocumentMarker::MarkerType type)
436{
437 if (!possiblyHasMarkers(type))
438 return nullptr;
439 ASSERT(!m_markers.isEmpty());
440
441 updateRectsForInvalidatedMarkersOfType(type);
442 for (auto& nodeMarkers : m_markers.values()) {
443 for (auto& marker : *nodeMarkers) {
444 if (marker.type() == type && marker.contains(point))
445 return &marker;
446 }
447 }
448 return nullptr;
449}

469void DocumentMarkerController::forEach(const SimpleRange& range, OptionSet<DocumentMarker::MarkerType> types, Function<bool(RenderedDocumentMarker&)> function)
470{
471 if (!possiblyHasMarkers(types))
472 return;
473 ASSERT(!m_markers.isEmpty());
474
475 for (auto& node : intersectingNodes(range)) {
476 if (auto list = m_markers.get(&node)) {
477 auto offsetRange = characterDataOffsetRange(range, node);
478 for (auto& marker : *list) {
479 // Markers are stored in order, so stop if we are now past the specified range.
480 if (marker.startOffset() >= offsetRange.end)
481 break;
482 if (marker.endOffset() > offsetRange.start && types.contains(marker.type())) {
483 if (function(marker))
484 return;
485 }
486 }
487 }
488 }
489}

524void DocumentMarkerController::removeMarkersFromList(MarkerMap::iterator iterator, OptionSet<DocumentMarker::MarkerType> types)
525{
526 bool needsRepainting = false;
527 bool listCanBeRemoved;
528
529 if (types == DocumentMarker::allMarkers()) {
530 needsRepainting = true;
531 listCanBeRemoved = true;
532 } else {
533 auto list = iterator->value.get();
534
535 for (size_t i = 0; i != list->size(); ) {
536 DocumentMarker marker = list->at(i);
537
538 // skip nodes that are not of the specified type
539 if (!types.contains(marker.type())) {
540 ++i;
541 continue;
542 }
543
544 // pitch the old marker
545 list->remove(i);
546 needsRepainting = true;
547 // i now is the index of the next marker
548 }
549
550 listCanBeRemoved = list->isEmpty();
551 }
552
553 if (needsRepainting) {
554 if (auto renderer = iterator->key->renderer())
555 renderer->repaint();
556 }
557
558 if (listCanBeRemoved) {
559 m_markers.remove(iterator);
560 if (m_markers.isEmpty())
561 m_possiblyExistingMarkerTypes = { };
562 }
563}

582void DocumentMarkerController::shiftMarkers(Node& node, unsigned startOffset, int delta)
583{
584 if (!possiblyHasMarkers(DocumentMarker::allMarkers()))
585 return;
586 ASSERT(!m_markers.isEmpty());
587
588 auto list = m_markers.get(&node);
589 if (!list)
590 return;
591
592 bool didShiftMarker = false;
593 for (size_t i = 0; i != list->size(); ) {
594 auto& marker = list->at(i);
595
596#if PLATFORM(IOS_FAMILY)
597 // FIXME: No obvious reason this should be iOS-specific. Remove the #if at some point.
598 auto targetStartOffset = clampTo<unsigned>(static_cast<int>(marker.startOffset()) + delta);
599 auto targetEndOffset = clampTo<unsigned>(static_cast<int>(marker.endOffset()) + delta);
600 if (targetStartOffset >= node.length() || targetEndOffset <= 0) {
601 list->remove(i);
602 continue;
603 }
604#endif
605
606 if (marker.startOffset() >= startOffset) {
607 ASSERT((int)marker.startOffset() + delta >= 0);
608 marker.shiftOffsets(delta);
609 didShiftMarker = true;
610 }
611#if PLATFORM(IOS_FAMILY)
612 // FIXME: No obvious reason this should be iOS-specific. Remove the #if at some point.
613 else if (marker.endOffset() > startOffset) {
614 if (targetEndOffset <= marker.startOffset()) {
615 list->remove(i);
616 continue;
617 }
618 marker.setEndOffset(std::min(targetEndOffset, node.length()));
619 didShiftMarker = true;
620 }
621#endif
622
623 ++i;
624 }
625
626 if (didShiftMarker) {
627 invalidateRectsForMarkersInNode(node);
628 if (auto renderer = node.renderer())
629 renderer->repaint();
630 }
631}
