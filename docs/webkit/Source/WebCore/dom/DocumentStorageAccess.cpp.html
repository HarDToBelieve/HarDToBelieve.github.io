77std::optional<bool> DocumentStorageAccess::hasStorageAccessQuickCheck()
78{
79 ASSERT(m_document.settings().storageAccessAPIEnabled());
80
81 auto* frame = m_document.frame();
82 if (frame && hasFrameSpecificStorageAccess())
83 return true;
84
85 if (!frame || m_document.securityOrigin().isUnique())
86 return false;
87
88 if (frame->isMainFrame())
89 return true;
90
91 auto& securityOrigin = m_document.securityOrigin();
92 auto& topSecurityOrigin = m_document.topDocument().securityOrigin();
93 if (securityOrigin.equal(&topSecurityOrigin))
94 return true;
95
96 auto* page = frame->page();
97 if (!page)
98 return false;
99
100 return std::nullopt;
101}

103void DocumentStorageAccess::hasStorageAccess(Ref<DeferredPromise>&& promise)
104{
105 ASSERT(m_document.settings().storageAccessAPIEnabled());
106
107 auto quickCheckResult = hasStorageAccessQuickCheck();
108 if (quickCheckResult) {
109 promise->resolve<IDLBoolean>(*quickCheckResult);
110 return;
111 }
112
113 // The existence of a frame and page has been checked in requestStorageAccessQuickCheck().
114 auto* frame = m_document.frame();
115 if (!frame) {
116 ASSERT_NOT_REACHED();
117 promise->resolve<IDLBoolean>(false);
118 return;
119 }
120 auto* page = frame->page();
121 if (!page) {
122 ASSERT_NOT_REACHED();
123 promise->resolve<IDLBoolean>(false);
124 return;
125 }
126
127 page->chrome().client().hasStorageAccess(RegistrableDomain::uncheckedCreateFromHost(m_document.securityOrigin().host()), RegistrableDomain::uncheckedCreateFromHost(m_document.topDocument().securityOrigin().host()), *frame, [weakThis = makeWeakPtr(*this), promise = WTFMove(promise)] (bool hasAccess) {
128 if (!weakThis)
129 return;
130
131 promise->resolve<IDLBoolean>(hasAccess);
132 });
133}

148std::optional<StorageAccessQuickResult> DocumentStorageAccess::requestStorageAccessQuickCheck()
149{
150 ASSERT(m_document.settings().storageAccessAPIEnabled());
151
152 auto* frame = m_document.frame();
153 if (frame && hasFrameSpecificStorageAccess())
154 return StorageAccessQuickResult::Grant;
155
156 if (!frame || m_document.securityOrigin().isUnique() || !isAllowedToRequestStorageAccess())
157 return StorageAccessQuickResult::Reject;
158
159 if (frame->isMainFrame())
160 return StorageAccessQuickResult::Grant;
161
162 auto& topDocument = m_document.topDocument();
163 auto& topSecurityOrigin = topDocument.securityOrigin();
164 auto& securityOrigin = m_document.securityOrigin();
165 if (securityOrigin.equal(&topSecurityOrigin))
166 return StorageAccessQuickResult::Grant;
167
168 // If there is a sandbox, it has to allow the storage access API to be called.
169 if (m_document.sandboxFlags() != SandboxNone && m_document.isSandboxed(SandboxStorageAccessByUserActivation))
170 return StorageAccessQuickResult::Reject;
171
172 if (!UserGestureIndicator::processingUserGesture())
173 return StorageAccessQuickResult::Reject;
174
175 return std::nullopt;
176}

178void DocumentStorageAccess::requestStorageAccess(Ref<DeferredPromise>&& promise)
179{
180 ASSERT(m_document.settings().storageAccessAPIEnabled());
181
182 auto quickCheckResult = requestStorageAccessQuickCheck();
183 if (quickCheckResult) {
184 *quickCheckResult == StorageAccessQuickResult::Grant ? promise->resolve() : promise->reject();
185 return;
186 }
187
188 // The existence of a frame and page has been checked in requestStorageAccessQuickCheck().
189 auto* frame = m_document.frame();
190 if (!frame) {
191 ASSERT_NOT_REACHED();
192 promise->reject();
193 return;
194 }
195 auto* page = frame->page();
196 if (!page) {
197 ASSERT_NOT_REACHED();
198 promise->reject();
199 return;
200 }
201
202 if (!page->settings().storageAccessAPIPerPageScopeEnabled())
203 m_storageAccessScope = StorageAccessScope::PerFrame;
204
205 page->chrome().client().requestStorageAccess(RegistrableDomain::uncheckedCreateFromHost(m_document.securityOrigin().host()), RegistrableDomain::uncheckedCreateFromHost(m_document.topDocument().securityOrigin().host()), *frame, m_storageAccessScope, [this, weakThis = makeWeakPtr(*this), promise = WTFMove(promise)] (RequestStorageAccessResult result) mutable {
206 if (!weakThis)
207 return;
208
209 // Consume the user gesture only if the user explicitly denied access.
210 bool shouldPreserveUserGesture = result.wasGranted == StorageAccessWasGranted::Yes || result.promptWasShown == StorageAccessPromptWasShown::No;
211
212 if (shouldPreserveUserGesture) {
213 m_document.eventLoop().queueMicrotask([this, weakThis] {
214 if (weakThis)
215 enableTemporaryTimeUserGesture();
216 });
217 }
218
219 if (result.wasGranted == StorageAccessWasGranted::Yes)
220 promise->resolve();
221 else {
222 if (result.promptWasShown == StorageAccessPromptWasShown::Yes)
223 setWasExplicitlyDeniedFrameSpecificStorageAccess();
224 promise->reject();
225 }
226
227 if (shouldPreserveUserGesture) {
228 m_document.eventLoop().queueMicrotask([this, weakThis] {
229 if (weakThis)
230 consumeTemporaryTimeUserGesture();
231 });
232 }
233 });
234}

256void DocumentStorageAccess::requestStorageAccessForNonDocumentQuirk(RegistrableDomain&& requestingDomain, CompletionHandler<void(StorageAccessWasGranted)>&& completionHandler)

265void DocumentStorageAccess::requestStorageAccessQuirk(RegistrableDomain&& requestingDomain, CompletionHandler<void(StorageAccessWasGranted)>&& completionHandler)
266{
267 ASSERT(m_document.settings().storageAccessAPIEnabled());
268 RELEASE_ASSERT(m_document.frame() && m_document.frame()->page());
269
270 auto topFrameDomain = RegistrableDomain(m_document.topDocument().url());
271
272 m_document.frame()->page()->chrome().client().requestStorageAccess(WTFMove(requestingDomain), WTFMove(topFrameDomain), *m_document.frame(), m_storageAccessScope, [this, weakThis = makeWeakPtr(*this), completionHandler = WTFMove(completionHandler)] (RequestStorageAccessResult result) mutable {
273 if (!weakThis)
274 return;
275
276 // Consume the user gesture only if the user explicitly denied access.
277 bool shouldPreserveUserGesture = result.wasGranted == StorageAccessWasGranted::Yes || result.promptWasShown == StorageAccessPromptWasShown::No;
278
279 if (shouldPreserveUserGesture) {
280 m_document.eventLoop().queueMicrotask([this, weakThis] {
281 if (weakThis)
282 enableTemporaryTimeUserGesture();
283 });
284 }
285
286 if (result.wasGranted == StorageAccessWasGranted::Yes)
287 completionHandler(StorageAccessWasGranted::Yes);
288 else {
289 if (result.promptWasShown == StorageAccessPromptWasShown::Yes)
290 setWasExplicitlyDeniedFrameSpecificStorageAccess();
291 completionHandler(StorageAccessWasGranted::No);
292 }
293
294 if (shouldPreserveUserGesture) {
295 m_document.eventLoop().queueMicrotask([this, weakThis] {
296 if (weakThis)
297 consumeTemporaryTimeUserGesture();
298 });
299 }
300 });
301}
