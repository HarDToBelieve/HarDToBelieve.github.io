34DocumentType::DocumentType(Document& document, const String& name, const String& publicId, const String& systemId)
35 : Node(document, CreateOther)
36 , m_name(name)
37 , m_publicId(publicId.isNull() ? emptyString() : publicId)
38 , m_systemId(systemId.isNull() ? emptyString() : systemId)
39{
40}
41
42String DocumentType::nodeName() const
43{
44 return name();
45}
46
47Node::NodeType DocumentType::nodeType() const
48{
49 return DOCUMENT_TYPE_NODE;
50}
51
52Ref<Node> DocumentType::cloneNodeInternal(Document& documentTarget, CloningOperation)
53{
54 return create(documentTarget, m_name, m_publicId, m_systemId);
55}
