36Ref<DragEvent> DragEvent::create(const AtomString& eventType, DragEventInit&& init)
37{
38 return adoptRef(*new DragEvent(eventType, WTFMove(init)));
39}
40
41Ref<DragEvent> DragEvent::createForBindings()
42{
43 return adoptRef(*new DragEvent);
44}

47 const IntPoint& screenLocation, const IntPoint& windowLocation, const IntPoint& movementDelta, OptionSet<Modifier> modifiers, short button, unsigned short buttons,
48 EventTarget* relatedTarget, double force, unsigned short syntheticClickType, DataTransfer* dataTransfer, IsSimulated isSimulated, IsTrusted isTrusted)

52}

54DragEvent::DragEvent(const AtomString& eventType, DragEventInit&& init)
55 : MouseEvent(eventType, init)

58}
