26#pragma once
27
28#include "DocumentIdentifier.h"
29#include "ElementIdentifier.h"
30#include "FloatRect.h"
31#include "PageIdentifier.h"
32
33namespace WebCore {
34
35struct ElementContext {
36 FloatRect boundingRect;
37
38 PageIdentifier webPageIdentifier;
39 DocumentIdentifier documentIdentifier;
40 ElementIdentifier elementIdentifier;
41
42 ~ElementContext() = default;
43
44 bool isSameElement(const ElementContext& other) const
45 {
46 return webPageIdentifier == other.webPageIdentifier && documentIdentifier == other.documentIdentifier && elementIdentifier == other.elementIdentifier;
47 }
48
49 template<class Encoder> void encode(Encoder&) const;
50 template<class Decoder> static std::optional<ElementContext> decode(Decoder&);
51};
52
53inline bool operator==(const ElementContext& a, const ElementContext& b)
54{
55 return a.boundingRect == b.boundingRect && a.isSameElement(b);
56}
57
58template<class Encoder>
59void ElementContext::encode(Encoder& encoder) const
60{
61 encoder << boundingRect;
62 encoder << webPageIdentifier;
63 encoder << documentIdentifier;
64 encoder << elementIdentifier;
65}
66
67template<class Decoder>
68std::optional<ElementContext> ElementContext::decode(Decoder& decoder)
69{
70 ElementContext context;
71
72 if (!decoder.decode(context.boundingRect))
73 return std::nullopt;
74
75 auto pageIdentifier = PageIdentifier::decode(decoder);
76 if (!pageIdentifier)
77 return std::nullopt;
78 context.webPageIdentifier = *pageIdentifier;
79
80 auto documentIdentifier = DocumentIdentifier::decode(decoder);
81 if (!documentIdentifier)
82 return std::nullopt;
83 context.documentIdentifier = *documentIdentifier;
84
85 auto elementIdentifier = ElementIdentifier::decode(decoder);
86 if (!elementIdentifier)
87 return std::nullopt;
88 context.elementIdentifier = *elementIdentifier;
89
90 return context;
91}
