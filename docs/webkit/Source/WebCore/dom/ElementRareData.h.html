41inline IntSize defaultMinimumSizeForResizing()
42{
43 return IntSize(LayoutUnit::max(), LayoutUnit::max());
44}

115#if DUMP_NODE_STATISTICS
116 OptionSet<UseType> useTypes() const
117 {
118 auto result = NodeRareData::useTypes();
119 if (m_unusualTabIndex)
120 result.add(UseType::TabIndex);
121 if (m_minimumSizeForResizing != defaultMinimumSizeForResizing())
122 result.add(UseType::MinimumSize);
123 if (!m_savedLayerScrollPosition.isZero())
124 result.add(UseType::ScrollingPosition);
125 if (m_computedStyle)
126 result.add(UseType::ComputedStyle);
127 if (m_dataset)
128 result.add(UseType::Dataset);
129 if (m_classList)
130 result.add(UseType::ClassList);
131 if (m_shadowRoot)
132 result.add(UseType::ShadowRoot);
133 if (m_customElementReactionQueue)
134 result.add(UseType::CustomElementQueue);
135 if (m_attributeMap)
136 result.add(UseType::AttributeMap);
137 if (m_intersectionObserverData)
138 result.add(UseType::InteractionObserver);
139#if ENABLE(RESIZE_OBSERVER)
140 if (m_resizeObserverData)
141 result.add(UseType::ResizeObserver);
142#endif
143 if (!m_animationRareData.isEmpty())
144 result.add(UseType::Animations);
145 if (m_beforePseudoElement || m_afterPseudoElement)
146 result.add(UseType::PseudoElements);
147#if ENABLE(CSS_TYPED_OM)
148 if (m_attributeStyleMap)
149 result.add(UseType::StyleMap);
150#endif
151 if (m_partList)
152 result.add(UseType::PartList);
153 if (!m_partNames.isEmpty())
154 result.add(UseType::PartNames);
155 return result;
156 }
157#endif

192inline ElementRareData::ElementRareData()
193 : NodeRareData(Type::Element)
194 , m_minimumSizeForResizing(defaultMinimumSizeForResizing())
195{
196}
197
198inline ElementRareData::~ElementRareData()
199{
200 ASSERT(!m_shadowRoot);
201 ASSERT(!m_beforePseudoElement);
202 ASSERT(!m_afterPseudoElement);
203}

205inline void ElementRareData::setBeforePseudoElement(RefPtr<PseudoElement>&& pseudoElement)
206{
207 ASSERT(!m_beforePseudoElement || !pseudoElement);
208 m_beforePseudoElement = WTFMove(pseudoElement);
209}
210
211inline void ElementRareData::setAfterPseudoElement(RefPtr<PseudoElement>&& pseudoElement)
212{
213 ASSERT(!m_afterPseudoElement || !pseudoElement);
214 m_afterPseudoElement = WTFMove(pseudoElement);
215}
216
217inline void ElementRareData::resetComputedStyle()
218{
219 m_computedStyle = nullptr;
220}
221
222inline int ElementRareData::unusualTabIndex() const
223{
224 ASSERT(m_unusualTabIndex); // setUnusualTabIndex must have been called before this.
225 return m_unusualTabIndex;
226}
227
228inline void ElementRareData::setUnusualTabIndex(int tabIndex)
229{
230 ASSERT(tabIndex && tabIndex != -1); // Common values of 0 and -1 are stored as TabIndexState in Node.
231 m_unusualTabIndex = tabIndex;
232}
233
234inline ElementAnimationRareData* ElementRareData::animationRareData(PseudoId pseudoId) const
235{
236 for (auto& animationRareData : m_animationRareData) {
237 if (animationRareData->pseudoId() == pseudoId)
238 return animationRareData.get();
239 }
240 return nullptr;
241}
242
243inline ElementAnimationRareData& ElementRareData::ensureAnimationRareData(PseudoId pseudoId)
244{
245 if (auto* animationRareData = this->animationRareData(pseudoId))
246 return *animationRareData;
247
248 m_animationRareData.append(makeUnique<ElementAnimationRareData>(pseudoId));
249 return *m_animationRareData.last().get();
250}
