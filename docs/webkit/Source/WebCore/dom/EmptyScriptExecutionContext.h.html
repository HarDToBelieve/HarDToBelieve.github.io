36namespace WebCore {
37
38class EmptyScriptExecutionContext final : public RefCounted<EmptyScriptExecutionContext>, public ScriptExecutionContext {
39public:
40 static Ref<EmptyScriptExecutionContext> create(JSC::VM& vm)
41 {
42 return adoptRef(*new EmptyScriptExecutionContext(vm));
43 }
44
45 bool isSecureContext() const final { return false; }
46 bool isJSExecutionForbidden() const final { return false; }
47 EventLoopTaskGroup& eventLoop() final
48 {
49 ASSERT_NOT_REACHED();
50 return *m_eventLoopTaskGroup;
51 }
52 const URL& url() const final { return m_url; }
53 URL completeURL(const String&, ForceUTF8 = ForceUTF8::No) const final { return { }; };
54 String userAgent(const URL&) const final { return emptyString(); }
55 ReferrerPolicy referrerPolicy() const final { return ReferrerPolicy::EmptyString; }
56
57 void disableEval(const String&) final { };
58 void disableWebAssembly(const String&) final { };
59
60 IDBClient::IDBConnectionProxy* idbConnectionProxy() final { return nullptr; }
61 SocketProvider* socketProvider() final { return nullptr; }
62
63 void addConsoleMessage(std::unique_ptr<Inspector::ConsoleMessage>&&) final { }
64 void addConsoleMessage(MessageSource, MessageLevel, const String&, unsigned long) final { };
65
66 SecurityOrigin& topOrigin() const final { return m_origin.get(); };
67
68 void postTask(Task&&) final { ASSERT_NOT_REACHED(); }
69 EventTarget* errorEventTarget() final { return nullptr; };
70
71#if ENABLE(WEB_CRYPTO)
72 bool wrapCryptoKey(const Vector<uint8_t>&, Vector<uint8_t>&) final { return false; }
73 bool unwrapCryptoKey(const Vector<uint8_t>&, Vector<uint8_t>&) final { return false; }
74#endif
75
76 JSC::VM& vm() final { return m_vm; }
77
78 using RefCounted::ref;
79 using RefCounted::deref;
80
81private:
82 EmptyScriptExecutionContext(JSC::VM& vm)
83 : m_vm(vm)
84 , m_origin(SecurityOrigin::createUnique())
85 , m_eventLoop(EmptyEventLoop::create(vm))
86 , m_eventLoopTaskGroup(makeUnique<EventLoopTaskGroup>(m_eventLoop))
87 {
88 }
89
90 void addMessage(MessageSource, MessageLevel, const String&, const String&, unsigned, unsigned, RefPtr<Inspector::ScriptCallStack>&&, JSC::JSGlobalObject* = nullptr, unsigned long = 0) final { }
91 void logExceptionToConsole(const String&, const String&, int, int, RefPtr<Inspector::ScriptCallStack>&&) final { };
92 void refScriptExecutionContext() final { ref(); };
93 void derefScriptExecutionContext() final { deref(); };
94
95 const Settings::Values& settingsValues() const final { return m_settingsValues; }
96
97 class EmptyEventLoop final : public EventLoop {
98 public:
99 static Ref<EmptyEventLoop> create(JSC::VM& vm)
100 {
101 return adoptRef(*new EmptyEventLoop(vm));
102 }
103
104 MicrotaskQueue& microtaskQueue() final { return m_queue; };
105
106 private:
107 explicit EmptyEventLoop(JSC::VM& vm)
108 : m_queue(MicrotaskQueue(vm))
109 {
110 }
111
112 void scheduleToRun() final { ASSERT_NOT_REACHED(); }
113 bool isContextThread() const final { return true; }
114
115 MicrotaskQueue m_queue;
116 };
117
118 Ref<JSC::VM> m_vm;
119 Ref<SecurityOrigin> m_origin;
120 URL m_url;
121 Ref<EmptyEventLoop> m_eventLoop;
122 std::unique_ptr<EventLoopTaskGroup> m_eventLoopTaskGroup;
123 Settings::Values m_settingsValues;
124};
