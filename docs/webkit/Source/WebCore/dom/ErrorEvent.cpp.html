78JSValue ErrorEvent::error(JSGlobalObject& globalObject)
79{
80 JSValue error = m_error;
81 if (!error)
82 return jsNull();
83
84 if (!isWorldCompatible(globalObject, error)) {
85 // We need to make sure ErrorEvents do not leak their error property across isolated DOM worlds.
86 // Ideally, we would check that the worlds have different privileges but that's not possible yet.
87 auto serializedError = trySerializeError(globalObject);
88 if (!serializedError)
89 return jsNull();
90 return serializedError->deserialize(globalObject, &globalObject);
91 }
92
93 return error;
94}
95
96RefPtr<SerializedScriptValue> ErrorEvent::trySerializeError(JSGlobalObject& exec)
97{
98 if (!m_serializedError && !m_triedToSerialize) {
99 m_serializedError = SerializedScriptValue::create(exec, m_error, SerializationErrorMode::NonThrowing);
100 m_triedToSerialize = true;
101 }
102 return m_serializedError;
103}
