32#pragma once
33
34#include "Event.h"
35#include "JSValueInWrappedObject.h"
36#include "SerializedScriptValue.h"
37#include <JavaScriptCore/Strong.h>
38#include <wtf/text/WTFString.h>
39
40namespace WebCore {
41
42class ErrorEvent final : public Event {
43 WTF_MAKE_ISO_ALLOCATED(ErrorEvent);
44public:
45 static Ref<ErrorEvent> create(const String& message, const String& fileName, unsigned lineNumber, unsigned columnNumber, JSC::Strong<JSC::Unknown> error)
46 {
47 return adoptRef(*new ErrorEvent(message, fileName, lineNumber, columnNumber, error));
48 }
49
50 static Ref<ErrorEvent> create(const AtomString& type, const String& message, const String& fileName, unsigned lineNumber, unsigned columnNumber, JSC::Strong<JSC::Unknown> error)
51 {
52 return adoptRef(*new ErrorEvent(type, message, fileName, lineNumber, columnNumber, error));
53 }
54
55 struct Init : EventInit {
56 String message;
57 String filename;
58 unsigned lineno { 0 };
59 unsigned colno { 0 };
60 JSC::JSValue error;
61 };
62
63 static Ref<ErrorEvent> create(const AtomString& type, const Init& initializer, IsTrusted isTrusted = IsTrusted::No)
64 {
65 return adoptRef(*new ErrorEvent(type, initializer, isTrusted));
66 }
67
68 virtual ~ErrorEvent();
69
70 const String& message() const { return m_message; }
71 const String& filename() const { return m_fileName; }
72 unsigned lineno() const { return m_lineNumber; }
73 unsigned colno() const { return m_columnNumber; }
74 JSC::JSValue error(JSC::JSGlobalObject&);
75
76 const JSValueInWrappedObject& originalError() const { return m_error; }
77 SerializedScriptValue* serializedError() const { return m_serializedError.get(); }
78
79 EventInterface eventInterface() const override;
80
81 RefPtr<SerializedScriptValue> trySerializeError(JSC::JSGlobalObject&);
82
83private:
84 ErrorEvent(const AtomString& type, const String& message, const String& fileName, unsigned lineNumber, unsigned columnNumber, JSC::Strong<JSC::Unknown> error);
85 ErrorEvent(const String& message, const String& fileName, unsigned lineNumber, unsigned columnNumber, JSC::Strong<JSC::Unknown> error);
86 ErrorEvent(const AtomString&, const Init&, IsTrusted);
87
88 bool isErrorEvent() const override;
89
90 String m_message;
91 String m_fileName;
92 unsigned m_lineNumber;
93 unsigned m_columnNumber;
94 JSValueInWrappedObject m_error;
95 RefPtr<SerializedScriptValue> m_serializedError;
96 bool m_triedToSerialize { false };
97};
98
99} // namespace WebCore

101SPECIALIZE_TYPE_TRAITS_EVENT(ErrorEvent)
