106void Event::initEvent(const AtomString& eventTypeArg, bool canBubbleArg, bool cancelableArg)
107{
108 if (isBeingDispatched())
109 return;
110
111 m_isInitialized = true;
112 m_propagationStopped = false;
113 m_immediatePropagationStopped = false;
114 m_wasCanceled = false;
115 m_isTrusted = false;
116 m_target = nullptr;
117 m_type = eventTypeArg;
118 m_canBubble = canBubbleArg;
119 m_cancelable = cancelableArg;
120
121 m_underlyingEvent = nullptr;
122}

124void Event::setTarget(RefPtr<EventTarget>&& target)
125{
126 if (m_target == target)
127 return;
128
129 m_target = WTFMove(target);
130 if (m_target)
131 receivedTarget();
132}

134void Event::setCurrentTarget(EventTarget* currentTarget, std::optional<bool> isInShadowTree)
135{
136 m_currentTarget = currentTarget;
137 m_currentTargetIsInShadowTree = isInShadowTree ? *isInShadowTree : (is<Node>(currentTarget) && downcast<Node>(*currentTarget).isInShadowTree());
138}

140Vector<EventTarget*> Event::composedPath() const
141{
142 if (!m_eventPath)
143 return Vector<EventTarget*>();
144 return m_eventPath->computePathUnclosedToTarget(*m_currentTarget);
145}

147void Event::setUnderlyingEvent(Event* underlyingEvent)
148{
149 // Prohibit creation of a cycle by doing nothing if a cycle would be created.
150 for (Event* event = underlyingEvent; event; event = event->underlyingEvent()) {
151 if (event == this)
152 return;
153 }
154 m_underlyingEvent = underlyingEvent;
155}

157DOMHighResTimeStamp Event::timeStampForBindings(ScriptExecutionContext& context) const
158{
159 Performance* performance = nullptr;
160 if (is<WorkerGlobalScope>(context))
161 performance = &downcast<WorkerGlobalScope>(context).performance();
162 else if (auto* window = downcast<Document>(context).domWindow())
163 performance = &window->performance();
164
165 if (!performance)
166 return 0;
167
168 return std::max(performance->relativeTimeFromTimeOriginInReducedResolution(m_createTime), 0.);
169}

171void Event::resetBeforeDispatch()

176void Event::resetAfterDispatch()
177{
178 m_eventPath = nullptr;
179 setCurrentTarget(nullptr);
180 m_eventPhase = NONE;
181 m_propagationStopped = false;
182 m_immediatePropagationStopped = false;
183
184 InspectorInstrumentation::eventDidResetAfterDispatch(*this);
185}

187String Event::debugDescription() const
188{
189 return makeString(type(), " phase ", eventPhase(), bubbles() ? " bubbles " : " ", cancelable() ? "cancelable " : " ", "0x"_s, hex(reinterpret_cast<uintptr_t>(this), Lowercase));
190}
191
192TextStream& operator<<(TextStream& ts, const Event& event)
193{
194 ts << event.debugDescription();
195 return ts;
196}
