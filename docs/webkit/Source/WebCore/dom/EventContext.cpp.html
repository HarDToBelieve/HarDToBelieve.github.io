43void EventContext::handleLocalEvents(Event& event, EventInvokePhase phase) const
44{
45 event.setTarget(m_target.get());
46 event.setCurrentTarget(m_currentTarget.get(), m_currentTargetIsInShadowTree);
47
48 if (m_relatedTarget) {
49 ASSERT(m_type == Type::MouseOrFocus);
50 event.setRelatedTarget(m_relatedTarget.get());
51 }
52
53#if ENABLE(TOUCH_EVENTS)
54 if (m_type == Type::Touch) {
55
56#if ASSERT_ENABLED
57 auto checkReachability = [&](const Ref<TouchList>& touchList) {
58 size_t length = touchList->length();
59 for (size_t i = 0; i < length; ++i)
60 ASSERT(!isUnreachableNode(downcast<Node>(touchList->item(i)->target())));
61 };
62 checkReachability(*m_touches);
63 checkReachability(*m_targetTouches);
64 checkReachability(*m_changedTouches);
65#endif
66
67 auto& touchEvent = downcast<TouchEvent>(event);
68 touchEvent.setTouches(m_touches.get());
69 touchEvent.setTargetTouches(m_targetTouches.get());
70 touchEvent.setChangedTouches(m_changedTouches.get());
71 }
72#endif
73
74 if (!m_node || UNLIKELY(m_type == Type::Window)) {
75 m_currentTarget->fireEventListeners(event, phase);
76 return;
77 }
78
79 if (UNLIKELY(m_contextNodeIsFormElement)) {
80 ASSERT(is<HTMLFormElement>(*m_node));
81 if ((event.type() == eventNames().submitEvent || event.type() == eventNames().resetEvent)
82 && event.eventPhase() != Event::CAPTURING_PHASE && event.target() != m_node && is<Node>(event.target())) {
83 event.stopPropagation();
84 return;
85 }
86 }
87
88 if (!m_node->hasEventTargetData())
89 return;
90
91 // FIXME: Should we deliver wheel events to disabled form controls or not?
92 if (event.isTrusted() && is<Element>(m_node) && downcast<Element>(*m_node).isDisabledFormControl() && event.isMouseEvent() && !event.isWheelEvent())
93 return;
94
95 m_node->fireEventListeners(event, phase);
96}

111bool EventContext::isUnreachableNode(EventTarget* target) const
112{
113 // FIXME: Checks also for SVG elements.
114 return is<Node>(target) && !downcast<Node>(*target).isSVGElement() && m_node->isClosedShadowHidden(downcast<Node>(*target));
115}
