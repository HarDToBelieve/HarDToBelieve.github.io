57bool EventListenerMap::containsCapturing(const AtomString& eventType) const
58{
59 auto* listeners = find(eventType);
60 if (!listeners)
61 return false;
62
63 for (auto& eventListener : *listeners) {
64 if (eventListener->useCapture())
65 return true;
66 }
67 return false;
68}

70bool EventListenerMap::containsActive(const AtomString& eventType) const
71{
72 auto* listeners = find(eventType);
73 if (!listeners)
74 return false;
75
76 for (auto& eventListener : *listeners) {
77 if (!eventListener->isPassive())
78 return true;
79 }
80 return false;
81}

83void EventListenerMap::clear()
84{
85 Locker locker { m_lock };
86
87 assertNoActiveIterators();
88
89 for (auto& entry : m_entries) {
90 for (auto& listener : *entry.second)
91 listener->markAsRemoved();
92 }
93
94 m_entries.clear();
95}

97Vector<AtomString> EventListenerMap::eventTypes() const
98{
99 Vector<AtomString> types;
100 types.reserveInitialCapacity(m_entries.size());
101
102 for (auto& entry : m_entries)
103 types.uncheckedAppend(entry.first);

105 return types;

108static inline size_t findListener(const EventListenerVector& listeners, EventListener& listener, bool useCapture)
109{
110 for (size_t i = 0; i < listeners.size(); ++i) {
111 auto& registeredListener = listeners[i];
112 if (registeredListener->callback() == listener && registeredListener->useCapture() == useCapture)
113 return i;
114 }
115 return notFound;
116}

118void EventListenerMap::replace(const AtomString& eventType, EventListener& oldListener, Ref<EventListener>&& newListener, const RegisteredEventListener::Options& options)
119{
120 Locker locker { m_lock };
121
122 assertNoActiveIterators();
123
124 auto* listeners = find(eventType);
125 ASSERT(listeners);
126 size_t index = findListener(*listeners, oldListener, options.capture);
127 ASSERT(index != notFound);
128 auto& registeredListener = listeners->at(index);
129 registeredListener->markAsRemoved();
130 registeredListener = RegisteredEventListener::create(WTFMove(newListener), options);
131}

133bool EventListenerMap::add(const AtomString& eventType, Ref<EventListener>&& listener, const RegisteredEventListener::Options& options)
134{
135 Locker locker { m_lock };
136
137 assertNoActiveIterators();
138
139 if (auto* listeners = find(eventType)) {
140 if (findListener(*listeners, listener, options.capture) != notFound)
141 return false; // Duplicate listener.
142 listeners->append(RegisteredEventListener::create(WTFMove(listener), options));
143 return true;
144 }
145
146 auto listeners = makeUnique<EventListenerVector>();
147 listeners->uncheckedAppend(RegisteredEventListener::create(WTFMove(listener), options));
148 m_entries.append({ eventType, WTFMove(listeners) });
149 return true;
150}

152static bool removeListenerFromVector(EventListenerVector& listeners, EventListener& listener, bool useCapture)
153{
154 size_t indexOfRemovedListener = findListener(listeners, listener, useCapture);
155 if (UNLIKELY(indexOfRemovedListener == notFound))
156 return false;
157
158 listeners[indexOfRemovedListener]->markAsRemoved();
159 listeners.remove(indexOfRemovedListener);
160 return true;
161}

163bool EventListenerMap::remove(const AtomString& eventType, EventListener& listener, bool useCapture)
164{
165 Locker locker { m_lock };
166
167 assertNoActiveIterators();
168
169 for (unsigned i = 0; i < m_entries.size(); ++i) {
170 if (m_entries[i].first == eventType) {
171 bool wasRemoved = removeListenerFromVector(*m_entries[i].second, listener, useCapture);
172 if (m_entries[i].second->isEmpty())
173 m_entries.remove(i);
174 return wasRemoved;
175 }
176 }
177
178 return false;
179}

181EventListenerVector* EventListenerMap::find(const AtomString& eventType) const
182{
183 for (auto& entry : m_entries) {
184 if (entry.first == eventType)
185 return entry.second.get();
186 }
187
188 return nullptr;
189}

191static void removeFirstListenerCreatedFromMarkup(EventListenerVector& listenerVector)
192{
193 bool foundListener = listenerVector.removeFirstMatching([] (const auto& registeredListener) {
194 if (registeredListener->callback().wasCreatedFromMarkup()) {
195 registeredListener->markAsRemoved();
196 return true;
197 }
198 return false;
199 });
200 ASSERT_UNUSED(foundListener, foundListener);
201}
202
203void EventListenerMap::removeFirstEventListenerCreatedFromMarkup(const AtomString& eventType)
204{
205 Locker locker { m_lock };
206
207 assertNoActiveIterators();
208
209 for (unsigned i = 0; i < m_entries.size(); ++i) {
210 if (m_entries[i].first == eventType) {
211 removeFirstListenerCreatedFromMarkup(*m_entries[i].second);
212 if (m_entries[i].second->isEmpty())
213 m_entries.remove(i);
214 return;
215 }
216 }
217}

219static void copyListenersNotCreatedFromMarkupToTarget(const AtomString& eventType, EventListenerVector& listenerVector, EventTarget* target)
220{
221 for (auto& registeredListener : listenerVector) {
222 // Event listeners created from markup have already been transfered to the shadow tree during cloning.
223 if (registeredListener->callback().wasCreatedFromMarkup())
224 continue;
225 target->addEventListener(eventType, registeredListener->callback(), registeredListener->useCapture());
226 }
227}
228
229void EventListenerMap::copyEventListenersNotCreatedFromMarkupToTarget(EventTarget* target)
230{
231 for (auto& entry : m_entries)
232 copyListenersNotCreatedFromMarkupToTarget(entry.first, *entry.second, target);
233}

267EventListener* EventListenerIterator::nextListener()
268{
269 if (!m_map)
270 return nullptr;
271
272 for (; m_entryIndex < m_map->m_entries.size(); ++m_entryIndex) {
273 EventListenerVector& listeners = *m_map->m_entries[m_entryIndex].second;
274 if (m_index < listeners.size())
275 return &listeners[m_index++]->callback();
276 m_index = 0;
277 }
278
279 return nullptr;
280}
