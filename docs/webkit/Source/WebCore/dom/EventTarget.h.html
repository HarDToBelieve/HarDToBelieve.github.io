49struct EventTargetData {
50 WTF_MAKE_NONCOPYABLE(EventTargetData); WTF_MAKE_FAST_ALLOCATED;
51public:
52 EventTargetData() = default;
53 EventListenerMap eventListenerMap;
54 bool isFiringEventListeners { false };

57class EventTarget : public ScriptWrappable, public CanMakeWeakPtr<EventTarget> {
58 WTF_MAKE_ISO_ALLOCATED(EventTarget);
59public:
60 static Ref<EventTarget> create(ScriptExecutionContext&);
61
62 void ref() { refEventTarget(); }
63 void deref() { derefEventTarget(); }
64
65 virtual EventTargetInterface eventTargetInterface() const = 0;
66 virtual ScriptExecutionContext* scriptExecutionContext() const = 0;
67
68 WEBCORE_EXPORT virtual bool isNode() const;
69 WEBCORE_EXPORT virtual bool isPaymentRequest() const;
70
71 using AddEventListenerOptionsOrBoolean = Variant<AddEventListenerOptions, bool>;
72 WEBCORE_EXPORT void addEventListenerForBindings(const AtomString& eventType, RefPtr<EventListener>&&, AddEventListenerOptionsOrBoolean&&);
73 using EventListenerOptionsOrBoolean = Variant<EventListenerOptions, bool>;
74 WEBCORE_EXPORT void removeEventListenerForBindings(const AtomString& eventType, RefPtr<EventListener>&&, EventListenerOptionsOrBoolean&&);
75 WEBCORE_EXPORT ExceptionOr<bool> dispatchEventForBindings(Event&);
76
77 WEBCORE_EXPORT virtual bool addEventListener(const AtomString& eventType, Ref<EventListener>&&, const AddEventListenerOptions&);
78 WEBCORE_EXPORT virtual bool removeEventListener(const AtomString& eventType, EventListener&, const EventListenerOptions& = { });
79
80 WEBCORE_EXPORT virtual void removeAllEventListeners();
81 WEBCORE_EXPORT virtual void dispatchEvent(Event&);
82 WEBCORE_EXPORT virtual void uncaughtExceptionInEventHandler();
83
84 // Used for legacy "onevent" attributes.
85 bool setAttributeEventListener(const AtomString& eventType, RefPtr<EventListener>&&, DOMWrapperWorld&);
86 EventListener* attributeEventListener(const AtomString& eventType, DOMWrapperWorld&);
87
88 bool hasEventListeners() const;
89 bool hasEventListeners(const AtomString& eventType) const;
90 bool hasCapturingEventListeners(const AtomString& eventType);
91 bool hasActiveEventListeners(const AtomString& eventType) const;
92
93 Vector<AtomString> eventTypes();
94 const EventListenerVector& eventListeners(const AtomString& eventType);
95
96 enum class EventInvokePhase { Capturing, Bubbling };
97 void fireEventListeners(Event&, EventInvokePhase);
98 bool isFiringEventListeners() const;
99
100 template<typename Visitor> void visitJSEventListeners(Visitor&);
101 void invalidateJSEventListeners(JSC::JSObject*);
102
103 const EventTargetData* eventTargetData() const;
104
105protected:
106 WEBCORE_EXPORT virtual ~EventTarget();
107
108 virtual EventTargetData* eventTargetData() = 0;
109 virtual EventTargetData* eventTargetDataConcurrently() = 0;
110 virtual EventTargetData& ensureEventTargetData() = 0;
111
112 virtual void eventListenersDidChange() { }
113
114private:
115 virtual void refEventTarget() = 0;
116 virtual void derefEventTarget() = 0;
117
118 void innerInvokeEventListeners(Event&, EventListenerVector, EventInvokePhase);
119 void invalidateEventListenerRegions();
120
121 friend class EventListenerIterator;
122};

124class EventTargetWithInlineData : public EventTarget {
125 WTF_MAKE_ISO_ALLOCATED_EXPORT(EventTargetWithInlineData, WEBCORE_EXPORT);
126protected:
127 EventTargetData* eventTargetData() final { return &m_eventTargetData; }
128 EventTargetData* eventTargetDataConcurrently() final { return &m_eventTargetData; }
129 EventTargetData& ensureEventTargetData() final { return m_eventTargetData; }
130private:
131 EventTargetData m_eventTargetData;
132};
133
134inline const EventTargetData* EventTarget::eventTargetData() const
135{
136 return const_cast<EventTarget*>(this)->eventTargetData();
137}
138
139inline bool EventTarget::isFiringEventListeners() const
140{
141 auto* data = eventTargetData();
142 return data && data->isFiringEventListeners;
143}
144
145inline bool EventTarget::hasEventListeners() const
146{
147 auto* data = eventTargetData();
148 return data && !data->eventListenerMap.isEmpty();
149}
150
151inline bool EventTarget::hasEventListeners(const AtomString& eventType) const
152{
153 auto* data = eventTargetData();
154 return data && data->eventListenerMap.contains(eventType);
155}
156
157inline bool EventTarget::hasCapturingEventListeners(const AtomString& eventType)
158{
159 auto* data = eventTargetData();
160 return data && data->eventListenerMap.containsCapturing(eventType);
161}
