23namespace WebCore {
24
25enum ExceptionCode {
26 // DOMException error names (https://heycam.github.io/webidl/#idl-DOMException-error-names).
27 // Those need to be kept in sync with the array in DOMException.cpp.
28 IndexSizeError, // Deprecated. Use RangeError instead.
29 HierarchyRequestError,
30 WrongDocumentError,
31 InvalidCharacterError,
32 NoModificationAllowedError,
33 NotFoundError,
34 NotSupportedError,
35 InUseAttributeError,
36 InvalidStateError,
37 SyntaxError,
38 InvalidModificationError,
39 NamespaceError,
40 InvalidAccessError, // Deprecated. use NotAllowedError instead.
41 TypeMismatchError, // Deprecated. Use TypeError instead.
42 SecurityError,
43 NetworkError,
44 AbortError,
45 URLMismatchError,
46 QuotaExceededError,
47 TimeoutError,
48 InvalidNodeTypeError,
49 DataCloneError,
50 EncodingError,
51 NotReadableError,
52 UnknownError,
53 ConstraintError,
54 DataError,
55 TransactionInactiveError,
56 ReadonlyError,
57 VersionError,
58 OperationError,
59 NotAllowedError,
60
61 // Simple exceptions (https://heycam.github.io/webidl/#idl-exceptions).
62 RangeError,
63 TypeError,
64 JSSyntaxError, // Different from DOM SYNTAX_ERR.
65
66 // Non-standard error.
67 StackOverflowError,
68 OutOfMemoryError,
69
70 // Used to indicate to the bindings that a JS exception was thrown below and it should be propagated.
71 ExistingExceptionError,
72};
