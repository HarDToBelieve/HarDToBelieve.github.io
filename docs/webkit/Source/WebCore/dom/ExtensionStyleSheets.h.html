67 WEBCORE_EXPORT void addUserStyleSheet(Ref<StyleSheetContents>&&);

76 void injectPageSpecificUserStyleSheet(const UserStyleSheet&);

97#if ENABLE(CONTENT_EXTENSIONS)

100#endif
101};
