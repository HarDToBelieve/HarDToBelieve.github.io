30namespace WebCore {
31
32enum ParserContentPolicy {
33 DisallowScriptingAndPluginContent,
34 DisallowScriptingContent,
35 AllowScriptingContent,
36 AllowScriptingContentAndDoNotMarkAlreadyStarted,
37};
38
39static inline bool scriptingContentIsAllowed(ParserContentPolicy parserContentPolicy)
40{
41 return parserContentPolicy == AllowScriptingContent || parserContentPolicy == AllowScriptingContentAndDoNotMarkAlreadyStarted;
42}
43
44static inline ParserContentPolicy disallowScriptingContent(ParserContentPolicy parserContentPolicy)
45{
46 if (!scriptingContentIsAllowed(parserContentPolicy))
47 return parserContentPolicy;
48 return DisallowScriptingContent;
49}
50
51static inline bool pluginContentIsAllowed(ParserContentPolicy parserContentPolicy)
52{
53 return parserContentPolicy != DisallowScriptingAndPluginContent;
54}
55
56static inline ParserContentPolicy allowPluginContent(ParserContentPolicy parserContentPolicy)
57{
58 if (pluginContentIsAllowed(parserContentPolicy))
59 return parserContentPolicy;
60 return DisallowScriptingContent;
61}
62
63} // namespace WebCore
