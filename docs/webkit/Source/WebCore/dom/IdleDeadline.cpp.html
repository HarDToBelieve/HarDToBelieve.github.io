34namespace WebCore {
35
36DOMHighResTimeStamp IdleDeadline::timeRemaining(Document& document) const
37{
38 auto window = makeRefPtr(document.domWindow());
39 if (!window)
40 return 0;
41 return window->performance().relativeTimeFromTimeOriginInReducedResolution(m_deadline);
42}
43
44bool IdleDeadline::didTimeout(Document& document) const
45{
46 auto window = makeRefPtr(document.domWindow());
47 if (!window)
48 return true;
49
50 // Reduce the resolution before the comparision to prevent resolution leakage.
51 auto deadline = window->performance().relativeTimeFromTimeOriginInReducedResolution(m_deadline);
52 auto now = window->performance().now();
53 return deadline >= now;
54}
