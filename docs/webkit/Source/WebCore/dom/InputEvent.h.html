28#include "StaticRange.h"
29#include "UIEvent.h"

31namespace WebCore {
32
33class DataTransfer;
34class WindowProxy;
35
36class InputEvent final : public UIEvent {
37 WTF_MAKE_ISO_ALLOCATED(InputEvent);
38public:
39 struct Init : UIEventInit {
40 String data;
41 };
42
43 static Ref<InputEvent> create(const AtomString& eventType, const String& inputType, IsCancelable, RefPtr<WindowProxy>&& view,
44 const String& data, RefPtr<DataTransfer>&&, const Vector<RefPtr<StaticRange>>& targetRanges, int detail);
45
46 static Ref<InputEvent> create(const AtomString& type, const Init& initializer)
47 {
48 return adoptRef(*new InputEvent(type, initializer));
49 }
50
51 bool isInputEvent() const override { return true; }
52 EventInterface eventInterface() const final { return InputEventInterfaceType; }
53 const String& inputType() const { return m_inputType; }
54 const String& data() const { return m_data; }
55 RefPtr<DataTransfer> dataTransfer() const;
56 const Vector<RefPtr<StaticRange>>& getTargetRanges() { return m_targetRanges; }
57
58private:
59 InputEvent(const AtomString& eventType, const String& inputType, IsCancelable, RefPtr<WindowProxy>&&, const String& data, RefPtr<DataTransfer>&&, const Vector<RefPtr<StaticRange>>& targetRanges, int detail);
60 InputEvent(const AtomString& eventType, const Init&);
61
62 String m_inputType;
63 String m_data;
64 RefPtr<DataTransfer> m_dataTransfer;
65 Vector<RefPtr<StaticRange>> m_targetRanges;
66};

68} // namespace WebCore
