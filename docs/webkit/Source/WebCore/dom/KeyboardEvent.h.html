24#pragma once
25
26#include "EventModifierInit.h"
27#include "KeypressCommand.h"
28#include "UIEventWithKeyState.h"
29#include <memory>
30#include <wtf/Vector.h>
31
32namespace WebCore {
33
34class Node;
35class PlatformKeyboardEvent;
36
37class KeyboardEvent final : public UIEventWithKeyState {
38 WTF_MAKE_ISO_ALLOCATED(KeyboardEvent);
39public:
40 enum KeyLocationCode {
41 DOM_KEY_LOCATION_STANDARD = 0x00,
42 DOM_KEY_LOCATION_LEFT = 0x01,
43 DOM_KEY_LOCATION_RIGHT = 0x02,
44 DOM_KEY_LOCATION_NUMPAD = 0x03
45 };
46
47 WEBCORE_EXPORT static Ref<KeyboardEvent> create(const PlatformKeyboardEvent&, RefPtr<WindowProxy>&&);
48 static Ref<KeyboardEvent> createForBindings();
49
50 struct Init : public EventModifierInit {
51 String key;
52 String code;
53 unsigned location;
54 bool repeat;
55 bool isComposing;
56
57 // Legacy.
58 String keyIdentifier;
59 std::optional<unsigned> keyLocation;
60 unsigned charCode;
61 unsigned keyCode;
62 unsigned which;
63 };
64
65 static Ref<KeyboardEvent> create(const AtomString& type, const Init&, IsTrusted = IsTrusted::No);
66
67 virtual ~KeyboardEvent();
68
69 WEBCORE_EXPORT void initKeyboardEvent(const AtomString& type, bool canBubble, bool cancelable, RefPtr<WindowProxy>&&,
70 const String& keyIdentifier, unsigned location,
71 bool ctrlKey, bool altKey, bool shiftKey, bool metaKey, bool altGraphKey = false);
72
73 const String& key() const { return m_key; }
74 const String& code() const { return m_code; }
75 const String& keyIdentifier() const { return m_keyIdentifier; }
76 unsigned location() const { return m_location; }
77 bool repeat() const { return m_repeat; }
78
79 const PlatformKeyboardEvent* underlyingPlatformEvent() const { return m_underlyingPlatformEvent.get(); }
80 PlatformKeyboardEvent* underlyingPlatformEvent() { return m_underlyingPlatformEvent.get(); }
81
82 WEBCORE_EXPORT int keyCode() const; // key code for keydown and keyup, character for keypress
83 WEBCORE_EXPORT int charCode() const; // character code for keypress, 0 for keydown and keyup
84
85 EventInterface eventInterface() const final;
86 bool isKeyboardEvent() const final;
87 int which() const final;
88
89 bool isComposing() const { return m_isComposing; }
90
91#if PLATFORM(COCOA)
92 bool handledByInputMethod() const { return m_handledByInputMethod; }
93 const Vector<KeypressCommand>& keypressCommands() const { return m_keypressCommands; }
94 Vector<KeypressCommand>& keypressCommands() { return m_keypressCommands; }
95#endif
96
97private:
98 KeyboardEvent();
99 KeyboardEvent(const PlatformKeyboardEvent&, RefPtr<WindowProxy>&&);
100 KeyboardEvent(const AtomString&, const Init&, IsTrusted = IsTrusted::No);
101
102 std::unique_ptr<PlatformKeyboardEvent> m_underlyingPlatformEvent;
103 String m_key;
104 String m_code;
105 String m_keyIdentifier;
106 unsigned m_location { DOM_KEY_LOCATION_STANDARD };
107 bool m_repeat { false };
108 bool m_isComposing { false };
109 std::optional<unsigned> m_charCode;
110 std::optional<unsigned> m_keyCode;
111 std::optional<unsigned> m_which;
112
113#if PLATFORM(COCOA)
114 // Commands that were sent by AppKit when interpreting the event. Doesn't include input method commands.
115 bool m_handledByInputMethod { false };
116 Vector<KeypressCommand> m_keypressCommands;
117#endif
118};
119
120} // namespace WebCore
121
122SPECIALIZE_TYPE_TRAITS_EVENT(KeyboardEvent)
