56std::optional<LoadableScript::Error> LoadableClassicScript::error() const
57{
58 ASSERT(m_cachedScript);
59 if (m_error)
60 return m_error;
61
62 if (m_cachedScript->errorOccurred())
63 return Error { ErrorType::CachedScript, std::nullopt };
64
65 return std::nullopt;
66}

74void LoadableClassicScript::notifyFinished(CachedResource& resource, const NetworkLoadMetrics&)
75{
76 ASSERT(m_cachedScript);
77 if (resource.resourceError().isAccessControl()) {
78 static NeverDestroyed<String> consoleMessage(MAKE_STATIC_STRING_IMPL("Cross-origin script load denied by Cross-Origin Resource Sharing policy."));
79 m_error = Error {
80 ErrorType::CrossOriginLoad,
81 ConsoleMessage {
82 MessageSource::JS,
83 MessageLevel::Error,
84 consoleMessage
85 }
86 };
87 }
88
89 if (!m_error && !isScriptAllowedByNosniff(m_cachedScript->response())) {
90 m_error = Error {
91 ErrorType::Nosniff,
92 ConsoleMessage {
93 MessageSource::Security,
94 MessageLevel::Error,
95 makeString("Refused to execute ", m_cachedScript->url().stringCenterEllipsizedToLength(), " as script because \"X-Content-Type-Options: nosniff\" was given and its Content-Type is not a script MIME type.")
96 }
97 };
98 }
99
100 if (!m_error && shouldBlockResponseDueToMIMEType(m_cachedScript->response(), m_cachedScript->options().destination)) {
101 m_error = Error {
102 ErrorType::MIMEType,
103 ConsoleMessage {
104 MessageSource::Security,
105 MessageLevel::Error,
106 makeString("Refused to execute ", m_cachedScript->url().stringCenterEllipsizedToLength(), " as script because ", m_cachedScript->response().mimeType(), " is not a script MIME type.")
107 }
108 };
109 }
110
111 if (!m_error && !resource.errorOccurred() && !matchIntegrityMetadata(resource, m_integrity)) {
112 m_error = Error {
113 ErrorType::FailedIntegrityCheck,
114 ConsoleMessage { MessageSource::Security, MessageLevel::Error, makeString("Cannot load script ", integrityMismatchDescription(resource, m_integrity)) }
115 };
116 }
117
118 notifyClientFinished();
119}

121void LoadableClassicScript::execute(ScriptElement& scriptElement)
122{
123 ASSERT(!error());
124 scriptElement.executeClassicScript(ScriptSourceCode(m_cachedScript.get(), JSC::SourceProviderSourceType::Program, *this));
125}

127bool LoadableClassicScript::load(Document& document, const URL& sourceURL)
128{
129 ASSERT(!m_cachedScript);
130
131 auto priority = [&]() -> std::optional<ResourceLoadPriority> {
132 if (m_isAsync)
133 return DefaultResourceLoadPriority::asyncScript;
134 // Use default.
135 return { };
136 };
137
138 m_cachedScript = requestScriptWithCache(document, sourceURL, crossOriginMode(), String { m_integrity }, priority());
139 if (!m_cachedScript)
140 return false;
141 m_cachedScript->addClient(*this);
142 return true;
143}
