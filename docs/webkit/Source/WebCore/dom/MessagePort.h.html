44namespace WebCore {
45
46class Frame;
47
48class MessagePort final : public ActiveDOMObject, public EventTargetWithInlineData {
49 WTF_MAKE_NONCOPYABLE(MessagePort);
50 WTF_MAKE_ISO_ALLOCATED(MessagePort);
51public:
52 static Ref<MessagePort> create(ScriptExecutionContext&, const MessagePortIdentifier& local, const MessagePortIdentifier& remote);
53 virtual ~MessagePort();
54
55 ExceptionOr<void> postMessage(JSC::JSGlobalObject&, JSC::JSValue message, PostMessageOptions&&);
56
57 void start();
58 void close();
59 void entangle();
60
61 // Returns nullptr if the passed-in vector is empty.
62 static ExceptionOr<TransferredMessagePortArray> disentanglePorts(Vector<RefPtr<MessagePort>>&&);
63 static Vector<RefPtr<MessagePort>> entanglePorts(ScriptExecutionContext&, TransferredMessagePortArray&&);
64
65 WEBCORE_EXPORT static bool isExistingMessagePortLocallyReachable(const MessagePortIdentifier&);
66 WEBCORE_EXPORT static void notifyMessageAvailable(const MessagePortIdentifier&);
67
68 WEBCORE_EXPORT void messageAvailable();
69 bool started() const { return m_started; }
70 bool closed() const { return m_closed; }
71
72 void dispatchMessages();
73
74 // Returns null if there is no entangled port, or if the entangled port is run by a different thread.
75 // This is used solely to enable a GC optimization. Some platforms may not be able to determine ownership
76 // of the remote port (since it may live cross-process) - those platforms may always return null.
77 MessagePort* locallyEntangledPort() const;
78
79 const MessagePortIdentifier& identifier() const { return m_identifier; }
80 const MessagePortIdentifier& remoteIdentifier() const { return m_remoteIdentifier; }
81
82 WEBCORE_EXPORT void ref() const;
83 WEBCORE_EXPORT void deref() const;
84
85 WEBCORE_EXPORT bool isLocallyReachable() const;
86
87 // EventTargetWithInlineData.
88 EventTargetInterface eventTargetInterface() const final { return MessagePortEventTargetInterfaceType; }
89 ScriptExecutionContext* scriptExecutionContext() const final { return ActiveDOMObject::scriptExecutionContext(); }
90 void refEventTarget() final { ref(); }
91 void derefEventTarget() final { deref(); }
92
93 void dispatchEvent(Event&) final;
94
95 TransferredMessagePort disentangle();
96 static Ref<MessagePort> entangle(ScriptExecutionContext&, TransferredMessagePort&&);
97
98private:
99 explicit MessagePort(ScriptExecutionContext&, const MessagePortIdentifier& local, const MessagePortIdentifier& remote);
100
101 bool addEventListener(const AtomString& eventType, Ref<EventListener>&&, const AddEventListenerOptions&) final;
102 bool removeEventListener(const AtomString& eventType, EventListener&, const EventListenerOptions&) final;
103
104 // ActiveDOMObject
105 const char* activeDOMObjectName() const final;
106 void contextDestroyed() final;
107 void stop() final { close(); }
108 bool virtualHasPendingActivity() const final;
109
110 void registerLocalActivity();
111
112 // A port starts out its life entangled, and remains entangled until it is closed or is cloned.
113 bool isEntangled() const { return !m_closed && m_entangled; }
114
115 void updateActivity(MessagePortChannelProvider::HasActivity);
116
117 bool m_started { false };
118 bool m_closed { false };
119 bool m_entangled { true };
120
121 // Flags to manage querying the remote port for GC purposes
122 mutable bool m_mightBeEligibleForGC { false };
123 mutable bool m_hasHadLocalActivitySinceLastCheck { false };
124 mutable bool m_isRemoteEligibleForGC { false };
125 mutable bool m_isAskingRemoteAboutGC { false };
126 bool m_hasMessageEventListener { false };
127
128 MessagePortIdentifier m_identifier;
129 MessagePortIdentifier m_remoteIdentifier;
130
131 mutable std::atomic<unsigned> m_refCount { 1 };
132};

134} // namespace WebCore
