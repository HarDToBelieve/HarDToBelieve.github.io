42{

46void MicrotaskQueue::performMicrotaskCheckpoint()
47{
48 if (m_performingMicrotaskCheckpoint)
49 return;
50
51 SetForScope<bool> change(m_performingMicrotaskCheckpoint, true);
52 JSC::JSLockHolder locker(vm());
53
54 Vector<std::unique_ptr<EventLoopTask>> toKeep;
55 while (!m_microtaskQueue.isEmpty()) {
56 Vector<std::unique_ptr<EventLoopTask>> queue = WTFMove(m_microtaskQueue);
57 for (auto& task : queue) {
58 auto* group = task->group();
59 if (!group || group->isStoppedPermanently())
60 continue;
61 if (group->isSuspended())
62 toKeep.append(WTFMove(task));
63 else
64 task->execute();
65 }
66 }
67
68 vm().finalizeSynchronousJSExecution();
69 m_microtaskQueue = WTFMove(toKeep);
70
71 auto checkpointTasks = std::exchange(m_checkpointTasks, { });
72 for (auto& checkpointTask : checkpointTasks) {
73 auto* group = checkpointTask->group();
74 if (!group || group->isStoppedPermanently())
75 continue;
76
77 if (group->isSuspended()) {
78 m_checkpointTasks.append(WTFMove(checkpointTask));
79 continue;
80 }
81
82 checkpointTask->execute();
83 }
84}
85
86void MicrotaskQueue::addCheckpointTask(std::unique_ptr<EventLoopTask>&& task)
87{
88 m_checkpointTasks.append(WTFMove(task));
89}
