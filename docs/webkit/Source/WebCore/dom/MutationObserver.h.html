42namespace JSC {
43class AbstractSlotVisitor;
44}
45
46namespace WebCore {
47
48class Document;
49class HTMLSlotElement;
50class MutationCallback;
51class MutationObserverRegistration;
52class MutationRecord;
53class Node;
54class WindowEventLoop;
55
56using MutationObserverOptions = unsigned char;
57using MutationRecordDeliveryOptions = unsigned char;
58
59class MutationObserver final : public RefCounted<MutationObserver> {
60 WTF_MAKE_ISO_ALLOCATED(MutationObserver);
61public:
62 enum MutationType {
63 ChildList = 1 << 0,
64 Attributes = 1 << 1,
65 CharacterData = 1 << 2,
66
67 AllMutationTypes = ChildList | Attributes | CharacterData
68 };
69
70 enum ObservationFlags {
71 Subtree = 1 << 3,
72 AttributeFilter = 1 << 4
73 };
74
75 enum DeliveryFlags {
76 AttributeOldValue = 1 << 5,
77 CharacterDataOldValue = 1 << 6,
78 };
79
80 static Ref<MutationObserver> create(Ref<MutationCallback>&&);
81
82 ~MutationObserver();
83
84 struct Init {
85 bool childList;
86 std::optional<bool> attributes;
87 std::optional<bool> characterData;
88 bool subtree;
89 std::optional<bool> attributeOldValue;
90 std::optional<bool> characterDataOldValue;
91 std::optional<Vector<String>> attributeFilter;
92 };
93
94 ExceptionOr<void> observe(Node&, const Init&);
95
96 struct TakenRecords {
97 Vector<Ref<MutationRecord>> records;
98 HashSet<GCReachableRef<Node>> pendingTargets;
99 };
100 TakenRecords takeRecords();
101 void disconnect();
102
103 void observationStarted(MutationObserverRegistration&);
104 void observationEnded(MutationObserverRegistration&);
105 void enqueueMutationRecord(Ref<MutationRecord>&&);
106 void setHasTransientRegistration(Document&);
107 bool canDeliver();
108
109 bool isReachableFromOpaqueRoots(JSC::AbstractSlotVisitor&) const;
110
111 MutationCallback& callback() const { return m_callback.get(); }
112
113 static void enqueueSlotChangeEvent(HTMLSlotElement&);
114
115 static void notifyMutationObservers(WindowEventLoop&);
116
117private:
118 explicit MutationObserver(Ref<MutationCallback>&&);
119 void deliver();
120
121 static bool validateOptions(MutationObserverOptions);
122
123 Ref<MutationCallback> m_callback;
124 Vector<Ref<MutationRecord>> m_records;
125 HashSet<GCReachableRef<Node>> m_pendingTargets;
126 WeakHashSet<MutationObserverRegistration> m_registrations;
127 unsigned m_priority;
128};
129
130} // namespace WebCore
