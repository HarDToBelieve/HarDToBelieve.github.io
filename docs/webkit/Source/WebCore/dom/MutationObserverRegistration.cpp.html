41MutationObserverRegistration::MutationObserverRegistration(MutationObserver& observer, Node& node, MutationObserverOptions options, const HashSet<AtomString>& attributeFilter)
42 : m_observer(observer)
43 , m_node(node)
44 , m_options(options)
45 , m_attributeFilter(attributeFilter)
46{
47 m_observer->observationStarted(*this);
48}
49
50MutationObserverRegistration::~MutationObserverRegistration()
51{
52 takeTransientRegistrations();
53 m_observer->observationEnded(*this);
54}
55
56void MutationObserverRegistration::resetObservation(MutationObserverOptions options, const HashSet<AtomString>& attributeFilter)
57{
58 takeTransientRegistrations();
59 m_options = options;
60 m_attributeFilter = attributeFilter;
61}
62
63void MutationObserverRegistration::observedSubtreeNodeWillDetach(Node& node)
64{
65 if (!isSubtree())
66 return;
67
68 node.registerTransientMutationObserver(*this);
69 m_observer->setHasTransientRegistration(node.document());
70
71 if (!m_transientRegistrationNodes) {
72 m_transientRegistrationNodes = makeUnique<HashSet<GCReachableRef<Node>>>();
73
74 ASSERT(!m_nodeKeptAlive);
75 m_nodeKeptAlive = &m_node; // Balanced in takeTransientRegistrations.
76 }
77 m_transientRegistrationNodes->add(node);
78}
79
80std::unique_ptr<HashSet<GCReachableRef<Node>>> MutationObserverRegistration::takeTransientRegistrations()
81{
82 if (!m_transientRegistrationNodes) {
83 ASSERT(!m_nodeKeptAlive);
84 return nullptr;
85 }
86
87 for (auto& node : *m_transientRegistrationNodes)
88 node->unregisterTransientMutationObserver(*this);
89
90 auto returnValue = WTFMove(m_transientRegistrationNodes);
91
92 ASSERT(m_nodeKeptAlive);
93 m_nodeKeptAlive = nullptr; // Balanced in observeSubtreeNodeWillDetach.
94
95 return returnValue;
96}

98bool MutationObserverRegistration::shouldReceiveMutationFrom(Node& node, MutationObserver::MutationType type, const QualifiedName* attributeName) const
99{
100 ASSERT((type == MutationObserver::Attributes && attributeName) || !attributeName);
101 if (!(m_options & type))
102 return false;
103
104 if (&m_node != &node && !isSubtree())
105 return false;
106
107 if (type != MutationObserver::Attributes || !(m_options & MutationObserver::AttributeFilter))
108 return true;
109
110 if (!attributeName->namespaceURI().isNull())
111 return false;
112
113 return m_attributeFilter.contains(attributeName->localName());
114}

116bool MutationObserverRegistration::isReachableFromOpaqueRoots(JSC::AbstractSlotVisitor& visitor) const
117{
118 if (visitor.containsOpaqueRoot(root(m_node)))
119 return true;
120
121 if (!m_transientRegistrationNodes)
122 return false;
123
124 for (auto& node : *m_transientRegistrationNodes) {
125 if (visitor.containsOpaqueRoot(root(node)))
126 return true;
127 }
128
129 return false;
130}
