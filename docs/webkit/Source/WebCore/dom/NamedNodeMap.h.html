32class Attr;
33class Element;
34
35class NamedNodeMap final : public ScriptWrappable {
36 WTF_MAKE_ISO_ALLOCATED(NamedNodeMap);
37public:
38 explicit NamedNodeMap(Element& element)

40 {
41 }
42
43 WEBCORE_EXPORT void ref();
44 WEBCORE_EXPORT void deref();
45
46 WEBCORE_EXPORT unsigned length() const;
47 WEBCORE_EXPORT RefPtr<Attr> item(unsigned index) const;
48 WEBCORE_EXPORT RefPtr<Attr> getNamedItem(const AtomString&) const;
49 WEBCORE_EXPORT RefPtr<Attr> getNamedItemNS(const AtomString& namespaceURI, const AtomString& localName) const;
50 WEBCORE_EXPORT ExceptionOr<RefPtr<Attr>> setNamedItem(Attr&);
51 WEBCORE_EXPORT ExceptionOr<Ref<Attr>> removeNamedItem(const AtomString& name);
52 WEBCORE_EXPORT ExceptionOr<Ref<Attr>> removeNamedItemNS(const AtomString& namespaceURI, const AtomString& localName);
53
54 Vector<String> supportedPropertyNames() const;
55
56 Element& element() { return m_element; }
57
58private:
59 Element& m_element;
60};
