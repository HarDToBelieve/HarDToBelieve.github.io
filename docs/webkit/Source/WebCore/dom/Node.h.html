775ALWAYS_INLINE void Node::ref() const
776{
777 ASSERT(isMainThread());
778 ASSERT(!m_deletionHasBegun);
779 ASSERT(!m_inRemovedLastRefFunction);
780 ASSERT(!m_adoptionIsRequired);
781 m_refCountAndParentBit += s_refCountIncrement;

783
784ALWAYS_INLINE void Node::deref() const
785{
786 ASSERT(isMainThread());
787 ASSERT(refCount());
788 ASSERT(!m_deletionHasBegun);
789 ASSERT(!m_inRemovedLastRefFunction);
790 ASSERT(!m_adoptionIsRequired);
791 auto updatedRefCount = m_refCountAndParentBit - s_refCountIncrement;
792 if (!updatedRefCount) {
793 // Don't update m_refCountAndParentBit to avoid double destruction through use of Ref<T>/RefPtr<T>.
794 // (This is a security mitigation in case of programmer error. It will ASSERT in debug builds.)
795#if ASSERT_ENABLED
796 m_inRemovedLastRefFunction = true;
797#endif
798IGNORE_ERRONEOUS_GCC_NULL_CHECK_WARNINGS_BEGIN
799 const_cast<Node&>(*this).removedLastRef();
800IGNORE_ERRONEOUS_GCC_NULL_CHECK_WARNINGS_END
801 return;
802 }
803 m_refCountAndParentBit = updatedRefCount;
804}
