25#pragma once
26
27#include "ActiveDOMCallback.h"
28#include "CallbackResult.h"
29#include <wtf/RefCounted.h>
30
31namespace WebCore {
32
33class Node;
34
35class NodeFilter : public RefCounted<NodeFilter>, public ActiveDOMCallback {
36public:
37 using ActiveDOMCallback::ActiveDOMCallback;
38
39 virtual CallbackResult<unsigned short> acceptNode(Node&) = 0;
40
41 virtual bool hasCallback() const = 0;
42
43 /*
44 * The following constants are returned by the acceptNode()
45 * method:
46 */
47 enum {
48 FILTER_ACCEPT = 1,
49 FILTER_REJECT = 2,
50 FILTER_SKIP = 3
51 };
52
53 /*
54 * These are the available values for the whatToShow parameter.
55 * They are the same as the set of possible types for Node, and
56 * their values are derived by using a bit position corresponding
57 * to the value of NodeType for the equivalent node type.
58 */
59 enum : unsigned long {
60 SHOW_ALL = 0xFFFFFFFF,
61 SHOW_ELEMENT = 0x00000001,
62 SHOW_ATTRIBUTE = 0x00000002,
63 SHOW_TEXT = 0x00000004,
64 SHOW_CDATA_SECTION = 0x00000008,
65 SHOW_ENTITY_REFERENCE = 0x00000010,
66 SHOW_ENTITY = 0x00000020,
67 SHOW_PROCESSING_INSTRUCTION = 0x00000040,
68 SHOW_COMMENT = 0x00000080,
69 SHOW_DOCUMENT = 0x00000100,
70 SHOW_DOCUMENT_TYPE = 0x00000200,
71 SHOW_DOCUMENT_FRAGMENT = 0x00000400,
72 SHOW_NOTATION = 0x00000800
73 };
74};
75
76} // namespace WebCore
