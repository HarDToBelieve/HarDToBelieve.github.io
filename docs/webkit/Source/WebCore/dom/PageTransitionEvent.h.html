27
28#include "Event.h"

30namespace WebCore {
31
32class PageTransitionEvent final : public Event {
33 WTF_MAKE_ISO_ALLOCATED(PageTransitionEvent);
34public:
35 static Ref<PageTransitionEvent> create(const AtomString& type, bool persisted)
36 {
37 return adoptRef(*new PageTransitionEvent(type, persisted));
38 }
39
40 struct Init : EventInit {
41 bool persisted { false };
42 };
43
44 static Ref<PageTransitionEvent> create(const AtomString& type, const Init& initializer, IsTrusted isTrusted = IsTrusted::No)
45 {
46 return adoptRef(*new PageTransitionEvent(type, initializer, isTrusted));
47 }
48
49 virtual ~PageTransitionEvent();
50
51 EventInterface eventInterface() const override;
52
53 bool persisted() const { return m_persisted; }
54
55private:
56 PageTransitionEvent(const AtomString& type, bool persisted);
57 PageTransitionEvent(const AtomString&, const Init&, IsTrusted);
58
59 bool m_persisted;
60};
