37static AtomString pointerEventType(const AtomString& mouseEventType)
38{
39 auto& names = eventNames();
40 if (mouseEventType == names.mousedownEvent)
41 return names.pointerdownEvent;
42 if (mouseEventType == names.mouseoverEvent)
43 return names.pointeroverEvent;
44 if (mouseEventType == names.mouseenterEvent)
45 return names.pointerenterEvent;
46 if (mouseEventType == names.mousemoveEvent)
47 return names.pointermoveEvent;
48 if (mouseEventType == names.mouseleaveEvent)
49 return names.pointerleaveEvent;
50 if (mouseEventType == names.mouseoutEvent)
51 return names.pointeroutEvent;
52 if (mouseEventType == names.mouseupEvent)
53 return names.pointerupEvent;
54
55 return nullAtom();
56}

58RefPtr<PointerEvent> PointerEvent::create(short button, const MouseEvent& mouseEvent, PointerID pointerId, const String& pointerType)
59{
60 auto type = pointerEventType(mouseEvent.type());
61 if (type.isEmpty())
62 return nullptr;
63
64 return create(type, button, mouseEvent, pointerId, pointerType);
65}
