31namespace WebCore {
32
33// A Position iterator with constant-time
34// increment, decrement, and several predicates on the Position it is at.
35// Conversion to/from Position is O(n) in the offset.
36class PositionIterator {
37public:
38 PositionIterator(const Position& pos)
39 : m_anchorNode(pos.anchorNode())
40 , m_nodeAfterPositionInAnchor(m_anchorNode->traverseToChildAt(pos.deprecatedEditingOffset()))
41 , m_offsetInAnchor(m_nodeAfterPositionInAnchor ? 0 : pos.deprecatedEditingOffset())
42 {
43 }
44 operator Position() const;
45
46 void increment();
47 void decrement();
48
49 Node* node() const { return m_anchorNode; }
50 int offsetInLeafNode() const { return m_offsetInAnchor; }
51
52 bool atStart() const;
53 bool atEnd() const;
54 bool atStartOfNode() const;
55 bool atEndOfNode() const;
56 bool isCandidate() const;
57
58private:
59 Node* m_anchorNode { nullptr };
60 Node* m_nodeAfterPositionInAnchor { nullptr }; // If this is non-null, m_nodeAfterPositionInAnchor->parentNode() == m_anchorNode;
61 int m_offsetInAnchor { 0 };
62};
63
64} // namespace WebCore
