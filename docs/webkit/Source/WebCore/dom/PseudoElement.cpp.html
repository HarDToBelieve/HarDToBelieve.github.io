43WTF_MAKE_ISO_ALLOCATED_IMPL(PseudoElement);
44
45const QualifiedName& pseudoElementTagName()
46{
47 static NeverDestroyed<QualifiedName> name(nullAtom(), "<pseudo>", nullAtom());
48 return name;
49}
50
51PseudoElement::PseudoElement(Element& host, PseudoId pseudoId)
52 : Element(pseudoElementTagName(), host.document(), CreatePseudoElement)
53 , m_hostElement(&host)
54 , m_pseudoId(pseudoId)
55{
56 ASSERT(pseudoId == PseudoId::Before || pseudoId == PseudoId::After);
57 setHasCustomStyleResolveCallbacks();
58}
59
60PseudoElement::~PseudoElement()
61{
62 ASSERT(!m_hostElement);
63}
64
65Ref<PseudoElement> PseudoElement::create(Element& host, PseudoId pseudoId)
66{
67 auto pseudoElement = adoptRef(*new PseudoElement(host, pseudoId));
68
69 InspectorInstrumentation::pseudoElementCreated(host.document().page(), pseudoElement.get());
70
71 return pseudoElement;
72}
73
74void PseudoElement::clearHostElement()
75{
76 InspectorInstrumentation::pseudoElementDestroyed(document().page(), *this);
77
78 Styleable::fromElement(*this).elementWasRemoved();
79
80 m_hostElement = nullptr;
81}
82
83bool PseudoElement::rendererIsNeeded(const RenderStyle& style)
84{
85 if (pseudoElementRendererIsNeeded(&style))
86 return true;
87
88 if (m_hostElement) {
89 if (auto* stack = m_hostElement->keyframeEffectStack(pseudoId()))
90 return stack->requiresPseudoElement();
91 }
92 return false;
93}
