29DEFINE_ALLOCATOR_WITH_HEAP_IDENTIFIER(QualifiedName);
30DEFINE_ALLOCATOR_WITH_HEAP_IDENTIFIER(QualifiedNameQualifiedNameImpl);

32QualifiedName::QualifiedName(const AtomString& p, const AtomString& l, const AtomString& n)
33 : m_impl(threadGlobalData().qualifiedNameCache().getOrCreate(QualifiedNameComponents { p.impl(), l.impl(), n.isEmpty() ? nullptr : n.impl() }))
34{
35}
36
37QualifiedName::QualifiedNameImpl::~QualifiedNameImpl()
38{
39 threadGlobalData().qualifiedNameCache().remove(*this);
40}
41
42// Global init routines
43LazyNeverDestroyed<const QualifiedName> anyName;
44
45void QualifiedName::init()
46{
47 static bool initialized = false;
48 if (initialized)
49 return;
50
51 ASSERT_WITH_MESSAGE(WTF::nullAtomData.isConstructed(), "AtomString::init should have been called");
52 anyName.construct(nullAtom(), starAtom(), starAtom());
53 initialized = true;
54}
55
56const QualifiedName& nullQName()
57{
58 static NeverDestroyed<QualifiedName> nullName(nullAtom(), nullAtom(), nullAtom());
59 return nullName;
60}
61
62const AtomString& QualifiedName::localNameUpper() const
63{
64 if (!m_impl->m_localNameUpper)
65 m_impl->m_localNameUpper = m_impl->m_localName.convertToASCIIUppercase();
66 return m_impl->m_localNameUpper;
67}
68
69unsigned QualifiedName::QualifiedNameImpl::computeHash() const
70{
71 QualifiedNameComponents components = { m_prefix.impl(), m_localName.impl(), m_namespace.impl() };
72 return hashComponents(components);
73}
