40ScopedEventQueue& ScopedEventQueue::singleton()
41{
42 static NeverDestroyed<ScopedEventQueue> scopedEventQueue;
43 return scopedEventQueue;
44}
45
46void ScopedEventQueue::enqueueEvent(Ref<Event>&& event)
47{
48 ASSERT(is<Node>(event->target()));
49 auto& target = downcast<Node>(*event->target());
50 ScopedEvent scopedEvent = { WTFMove(event), target };
51 if (m_scopingLevel)
52 m_queuedEvents.append(WTFMove(scopedEvent));
53 else
54 dispatchEvent(scopedEvent);
55}
56
57void ScopedEventQueue::dispatchEvent(const ScopedEvent& event) const
58{
59 if (event.event->eventInterface() == MutationEventInterfaceType && event.target->isInShadowTree())
60 return;
61 event.target->dispatchEvent(event.event);
62}
63
64void ScopedEventQueue::dispatchAllEvents()
65{
66 auto queuedEvents = std::exchange(m_queuedEvents, { });
67 for (auto& queuedEvent : queuedEvents)
68 dispatchEvent(queuedEvent);
69}
70
71void ScopedEventQueue::incrementScopingLevel()
72{
73 ++m_scopingLevel;
74}
75
76void ScopedEventQueue::decrementScopingLevel()
77{
78 ASSERT(m_scopingLevel);
79 --m_scopingLevel;
80 if (!m_scopingLevel)
81 dispatchAllEvents();
82}
