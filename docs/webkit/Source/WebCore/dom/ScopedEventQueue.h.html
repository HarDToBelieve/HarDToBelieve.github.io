31#pragma once
32
33#include "GCReachableRef.h"
34#include <wtf/Forward.h>
35#include <wtf/Noncopyable.h>
36#include <wtf/Ref.h>
37#include <wtf/Vector.h>
38
39namespace WebCore {
40
41class Event;
42class EventQueueScope;
43
44class ScopedEventQueue {
45 WTF_MAKE_NONCOPYABLE(ScopedEventQueue); WTF_MAKE_FAST_ALLOCATED;
46public:
47 static ScopedEventQueue& singleton();
48 void enqueueEvent(Ref<Event>&&);
49
50private:
51 ScopedEventQueue() = default;
52 ~ScopedEventQueue() = delete;
53
54 struct ScopedEvent {
55 Ref<Event> event;
56 GCReachableRef<Node> target;
57 };
58
59 void dispatchEvent(const ScopedEvent&) const;
60 void dispatchAllEvents();
61 void incrementScopingLevel();
62 void decrementScopingLevel();
63
64 Vector<ScopedEvent> m_queuedEvents;
65 unsigned m_scopingLevel { 0 };
66
67 friend class WTF::NeverDestroyed<WebCore::ScopedEventQueue>;
68 friend class EventQueueScope;
69};
70
71class EventQueueScope {
72 WTF_MAKE_NONCOPYABLE(EventQueueScope);
73public:
74 EventQueueScope() { ScopedEventQueue::singleton().incrementScopingLevel(); }
75 ~EventQueueScope() { ScopedEventQueue::singleton().decrementScopingLevel(); }
76};

78} // namespace WebCore
