36class ScriptDisallowedScope {
37public:
38 // This variant is expensive. Use ScriptDisallowedScope::InMainThread whenever possible.
39 ScriptDisallowedScope()
40 {
41 if (!isMainThread())
42 return;
43 ++s_count;
44 }
45
46 ScriptDisallowedScope(const ScriptDisallowedScope&)
47 : ScriptDisallowedScope()
48 {
49 }
50
51 ~ScriptDisallowedScope()
52 {
53 if (!isMainThread())
54 return;
55 ASSERT(s_count);
56 s_count--;
57 }
58
59 ScriptDisallowedScope& operator=(const ScriptDisallowedScope&)
60 {
61 return *this;
62 }
63
64 static bool isEventAllowedInMainThread()
65 {
66 return !isInWebProcess() || !isMainThread() || !s_count;
67 }
68
69 class InMainThread {
70 public:
71 InMainThread()
72 {
73 ASSERT(isMainThread());
74 ++s_count;
75 }
76
77 ~InMainThread()
78 {
79 ASSERT(isMainThread());
80 ASSERT(s_count);
81 --s_count;
82 }
83
84 // Don't enable this assertion in release since it's O(n).
85 // Release asserts in canExecuteScript should be sufficient for security defense purposes.
86 static bool isEventDispatchAllowedInSubtree(Node& node)
87 {
88#if ASSERT_ENABLED || ENABLE(SECURITY_ASSERTIONS)
89 return !isInWebProcess() || isScriptAllowed() || EventAllowedScope::isAllowedNode(node);
90#else
91 UNUSED_PARAM(node);
92 return true;
93#endif
94 }
95
96 static bool hasDisallowedScope()
97 {
98 ASSERT(isMainThread());
99 return s_count;
100 }
101
102 static bool isScriptAllowed()
103 {
104 ASSERT(isMainThread());
105#if PLATFORM(IOS_FAMILY)
106 return !s_count || webThreadDelegateMessageScopeCount;
107#else
108 return !s_count;
109#endif
110 }
111 };
112
113#if ASSERT_ENABLED
114 class EventAllowedScope {
115 public:
116 explicit EventAllowedScope(ContainerNode& userAgentContentRoot)
117 : m_eventAllowedTreeRoot(userAgentContentRoot)
118 , m_previousScope(s_currentScope)
119 {
120 s_currentScope = this;
121 }
122
123 ~EventAllowedScope()
124 {
125 s_currentScope = m_previousScope;
126 }
127
128 static bool isAllowedNode(Node& node)
129 {
130 return s_currentScope && s_currentScope->isAllowedNodeInternal(node);
131 }
132
133 private:
134 bool isAllowedNodeInternal(Node& node)
135 {
136 return m_eventAllowedTreeRoot->contains(&node) || (m_previousScope && m_previousScope->isAllowedNodeInternal(node));
137 }
138
139 Ref<ContainerNode> m_eventAllowedTreeRoot;
140
141 EventAllowedScope* m_previousScope;
142 static EventAllowedScope* s_currentScope;
143 };
144#else // not ASSERT_ENABLED
145 class EventAllowedScope {
146 public:
147 explicit EventAllowedScope(ContainerNode&) { }
148 static bool isAllowedNode(Node&) { return true; }
149 };
150#endif // not ASSERT_ENABLED

152 // FIXME: Remove this class once the sync layout inside SVGImage::draw is removed,
153 // CachedSVGFont::ensureCustomFontData no longer synchronously creates a document during style resolution,
154 // and refactored the code in RenderFrameBase::performLayoutWithFlattening.
155 class DisableAssertionsInScope {
156 public:
157 DisableAssertionsInScope()
158 {
159 ASSERT(isMainThread());
160 std::swap(s_count, m_originalCount);
161 }
162
163 ~DisableAssertionsInScope()
164 {
165 s_count = m_originalCount;
166 }
167 private:
168 unsigned m_originalCount { 0 };
169 };
170
171private:
172 WEBCORE_EXPORT static unsigned s_count;
173};
