34const ASCIILiteral ScriptElementCachedScriptFetcher::defaultCrossOriginModeForModule { "anonymous"_s };
35
36CachedResourceHandle<CachedScript> ScriptElementCachedScriptFetcher::requestModuleScript(Document& document, const URL& sourceURL, String&& integrity) const
37{
38 // https://html.spec.whatwg.org/multipage/urls-and-fetching.html#cors-settings-attributes
39 // If the fetcher is not module script, credential mode is always "same-origin" ("anonymous").
40 // This code is for dynamic module import (`import` operator).
41
42 return requestScriptWithCache(document, sourceURL, isClassicScript() ? defaultCrossOriginModeForModule : m_crossOriginMode, WTFMove(integrity), { });
43}
