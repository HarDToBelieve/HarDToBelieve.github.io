113ScriptExecutionContextIdentifier ScriptExecutionContext::contextIdentifier() const
114{
115 ASSERT(isContextThread());
116 if (!m_contextIdentifier) {
117 Locker locker { allScriptExecutionContextsMapLock };
118
119 m_contextIdentifier = ScriptExecutionContextIdentifier::generate();
120
121 ASSERT(!allScriptExecutionContextsMap().contains(m_contextIdentifier));
122 allScriptExecutionContextsMap().add(m_contextIdentifier, const_cast<ScriptExecutionContext*>(this));
123 }
124 return m_contextIdentifier;
125}

144void ScriptExecutionContext::checkConsistency() const
145{
146 for (auto* messagePort : m_messagePorts)
147 ASSERT(messagePort->scriptExecutionContext() == this);
148
149 for (auto* destructionObserver : m_destructionObservers)
150 ASSERT(destructionObserver->scriptExecutionContext() == this);
151
152 for (auto* activeDOMObject : m_activeDOMObjects) {
153 ASSERT(activeDOMObject->scriptExecutionContext() == this);
154 activeDOMObject->assertSuspendIfNeededWasCalled();
155 }
156}
157
158#endif // ASSERT_ENABLED

160ScriptExecutionContext::~ScriptExecutionContext()
161{
162 checkConsistency();
163
164#if ASSERT_ENABLED
165 if (m_contextIdentifier) {
166 Locker locker { allScriptExecutionContextsMapLock };
167 ASSERT_WITH_MESSAGE(!allScriptExecutionContextsMap().contains(m_contextIdentifier),
168 "A ScriptExecutionContext subclass instance implementing postTask should have already removed itself from the map");
169 }
170
171 m_inScriptExecutionContextDestructor = true;
172#endif // ASSERT_ENABLED
173
174#if ENABLE(SERVICE_WORKER)
175 setActiveServiceWorker(nullptr);
176#endif
177
178 while (auto* destructionObserver = m_destructionObservers.takeAny())
179 destructionObserver->contextDestroyed();
180
181#if ASSERT_ENABLED
182 m_inScriptExecutionContextDestructor = false;
183#endif
184}

186void ScriptExecutionContext::processMessageWithMessagePortsSoon()
187{
188 if (m_willprocessMessageWithMessagePortsSoon)
189 return;
190
191 m_willprocessMessageWithMessagePortsSoon = true;
192 postTask([] (ScriptExecutionContext& context) {
193 context.dispatchMessagePortEvents();
194 });
195}
196
197void ScriptExecutionContext::dispatchMessagePortEvents()
198{
199 checkConsistency();
200
201 Ref<ScriptExecutionContext> protectedThis(*this);
202 ASSERT(m_willprocessMessageWithMessagePortsSoon);
203 m_willprocessMessageWithMessagePortsSoon = false;
204
205 // Make a frozen copy of the ports so we can iterate while new ones might be added or destroyed.
206 for (auto* messagePort : copyToVector(m_messagePorts)) {
207 // The port may be destroyed, and another one created at the same address,
208 // but this is harmless. The worst that can happen as a result is that
209 // dispatchMessages() will be called needlessly.
210 if (m_messagePorts.contains(messagePort) && messagePort->started())
211 messagePort->dispatchMessages();
212 }
213}

248void ScriptExecutionContext::forEachActiveDOMObject(const Function<ShouldContinue(ActiveDOMObject&)>& apply) const
249{
250 // It is not allowed to run arbitrary script or construct new ActiveDOMObjects while we are iterating over ActiveDOMObjects.
251 // An ASSERT_WITH_SECURITY_IMPLICATION or RELEASE_ASSERT will fire if this happens, but it's important to code
252 // suspend() / resume() / stop() functions so it will not happen!
253 ScriptDisallowedScope scriptDisallowedScope;
254 SetForScope<bool> activeDOMObjectAdditionForbiddenScope(m_activeDOMObjectAdditionForbidden, true);
255
256 // Make a frozen copy of the objects so we can iterate while new ones might be destroyed.
257 auto possibleActiveDOMObjects = copyToVector(m_activeDOMObjects);
258
259 for (auto* activeDOMObject : possibleActiveDOMObjects) {
260 // Check if this object was deleted already. If so, just skip it.
261 // Calling contains on a possibly-already-deleted object is OK because we guarantee
262 // no new object can be added, so even if a new object ends up allocated with the
263 // same address, that will be *after* this function exits.
264 if (!m_activeDOMObjects.contains(activeDOMObject))
265 continue;
266
267 if (apply(*activeDOMObject) == ShouldContinue::No)
268 break;
269 }
270}

281void ScriptExecutionContext::suspendActiveDOMObjects(ReasonForSuspension why)
282{
283 checkConsistency();
284
285 if (m_activeDOMObjectsAreSuspended) {
286 // A page may subsequently suspend DOM objects, say as part of entering the back/forward cache, after the embedding
287 // client requested the page be suspended. We ignore such requests so long as the embedding client requested
288 // the suspension first. See <rdar://problem/13754896> for more details.
289 ASSERT(m_reasonForSuspendingActiveDOMObjects == ReasonForSuspension::PageWillBeSuspended);
290 return;
291 }
292
293 m_activeDOMObjectsAreSuspended = true;
294
295 forEachActiveDOMObject([why](auto& activeDOMObject) {
296 activeDOMObject.suspend(why);
297 return ShouldContinue::Yes;
298 });
299
300 m_reasonForSuspendingActiveDOMObjects = why;
301}
302
303void ScriptExecutionContext::resumeActiveDOMObjects(ReasonForSuspension why)
304{
305 checkConsistency();
306
307 if (m_reasonForSuspendingActiveDOMObjects != why)
308 return;
309
310 forEachActiveDOMObject([](auto& activeDOMObject) {
311 activeDOMObject.resume();
312 return ShouldContinue::Yes;
313 });
314
315 vm().deferredWorkTimer->didResumeScriptExecutionOwner();
316
317 m_activeDOMObjectsAreSuspended = false;
318
319 // In case there were pending messages at the time the script execution context entered the BackForwardCache,
320 // make sure those get dispatched shortly after restoring from the BackForwardCache.
321 processMessageWithMessagePortsSoon();
322}

347void ScriptExecutionContext::didCreateActiveDOMObject(ActiveDOMObject& activeDOMObject)
348{
349 // The m_activeDOMObjectAdditionForbidden check is a RELEASE_ASSERT because of the
350 // consequences of having an ActiveDOMObject that is not correctly reflected in the set.
351 // If we do have one of those, it can possibly be a security vulnerability. So we'd
352 // rather have a crash than continue running with the set possibly compromised.
353 ASSERT(!m_inScriptExecutionContextDestructor);
354 RELEASE_ASSERT(!m_activeDOMObjectAdditionForbidden);
355 m_activeDOMObjects.add(&activeDOMObject);
356}

386void ScriptExecutionContext::reportException(const String& errorMessage, int lineNumber, int columnNumber, const String& sourceURL, JSC::Exception* exception, RefPtr<ScriptCallStack>&& callStack, CachedScript* cachedScript)
387{
388 if (m_inDispatchErrorEvent) {
389 if (!m_pendingExceptions)
390 m_pendingExceptions = makeUnique<Vector<std::unique_ptr<PendingException>>>();
391 m_pendingExceptions->append(makeUnique<PendingException>(errorMessage, lineNumber, columnNumber, sourceURL, WTFMove(callStack)));
392 return;
393 }
394
395 // First report the original exception and only then all the nested ones.
396 if (!dispatchErrorEvent(errorMessage, lineNumber, columnNumber, sourceURL, exception, cachedScript))
397 logExceptionToConsole(errorMessage, sourceURL, lineNumber, columnNumber, callStack.copyRef());
398
399 if (!m_pendingExceptions)
400 return;
401
402 auto pendingExceptions = WTFMove(m_pendingExceptions);
403 for (auto& exception : *pendingExceptions)
404 logExceptionToConsole(exception->m_errorMessage, exception->m_sourceURL, exception->m_lineNumber, exception->m_columnNumber, WTFMove(exception->m_callStack));
405}

407void ScriptExecutionContext::reportUnhandledPromiseRejection(JSC::JSGlobalObject& state, JSC::JSPromise& promise, RefPtr<Inspector::ScriptCallStack>&& callStack)
408{
409 Page* page = nullptr;
410 if (is<Document>(this))
411 page = downcast<Document>(this)->page();
412 // FIXME: allow Workers to mute unhandled promise rejection messages.
413
414 if (page && !page->settings().unhandledPromiseRejectionToConsoleEnabled())
415 return;
416
417 JSC::VM& vm = state.vm();
418 auto scope = DECLARE_CATCH_SCOPE(vm);
419 JSC::JSValue result = promise.result(vm);
420 String resultMessage = retrieveErrorMessage(state, vm, result, scope);
421 String errorMessage;
422
423 auto tryMakeErrorString = [&] (unsigned length) -> String {
424 bool addEllipsis = length != resultMessage.length();
425 return tryMakeString("Unhandled Promise Rejection: ", StringView(resultMessage).left(length), addEllipsis ? "..." : "");
426 };
427
428 if (!!resultMessage && !scope.exception()) {
429 constexpr unsigned maxLength = 200;
430 constexpr unsigned shortLength = 10;
431 errorMessage = tryMakeErrorString(std::min(resultMessage.length(), maxLength));
432 if (!errorMessage && resultMessage.length() > shortLength)
433 errorMessage = tryMakeErrorString(shortLength);
434 }
435
436 if (!errorMessage)
437 errorMessage = "Unhandled Promise Rejection"_s;
438
439 std::unique_ptr<Inspector::ConsoleMessage> message;
440 if (callStack)
441 message = makeUnique<Inspector::ConsoleMessage>(MessageSource::JS, MessageType::Log, MessageLevel::Error, errorMessage, callStack.releaseNonNull());
442 else
443 message = makeUnique<Inspector::ConsoleMessage>(MessageSource::JS, MessageType::Log, MessageLevel::Error, errorMessage);
444 addConsoleMessage(WTFMove(message));
445}

452bool ScriptExecutionContext::dispatchErrorEvent(const String& errorMessage, int lineNumber, int columnNumber, const String& sourceURL, JSC::Exception* exception, CachedScript* cachedScript)
453{
454 auto* target = errorEventTarget();
455 if (!target)
456 return false;
457
458 RefPtr<ErrorEvent> errorEvent;
459 if (canIncludeErrorDetails(cachedScript, sourceURL))
460 errorEvent = ErrorEvent::create(errorMessage, sourceURL, lineNumber, columnNumber, { vm(), exception ? exception->value() : JSC::jsNull() });
461 else
462 errorEvent = ErrorEvent::create("Script error."_s, { }, 0, 0, { });
463
464 ASSERT(!m_inDispatchErrorEvent);
465 m_inDispatchErrorEvent = true;
466 target->dispatchEvent(*errorEvent);
467 m_inDispatchErrorEvent = false;
468 return errorEvent->defaultPrevented();
469}

625bool ScriptExecutionContext::postTaskTo(ScriptExecutionContextIdentifier identifier, Task&& task)
626{
627 Locker locker { allScriptExecutionContextsMapLock };
628 auto* context = allScriptExecutionContextsMap().get(identifier);
629
630 if (!context)

633 context->postTask(WTFMove(task));
