36ScriptRunner::ScriptRunner(Document& document)
37 : m_document(document)
38 , m_timer(*this, &ScriptRunner::timerFired)
39{
40}
41
42ScriptRunner::~ScriptRunner()
43{
44 for (auto& pendingScript : m_scriptsToExecuteSoon) {
45 UNUSED_PARAM(pendingScript);
46 m_document.decrementLoadEventDelayCount();
47 }
48 for (auto& pendingScript : m_scriptsToExecuteInOrder) {
49 if (pendingScript->watchingForLoad())
50 pendingScript->clearClient();
51 m_document.decrementLoadEventDelayCount();
52 }
53 for (auto& pendingScript : m_pendingAsyncScripts) {
54 if (pendingScript->watchingForLoad())
55 pendingScript->clearClient();
56 m_document.decrementLoadEventDelayCount();
57 }
58}

60void ScriptRunner::queueScriptForExecution(ScriptElement& scriptElement, LoadableScript& loadableScript, ExecutionType executionType)
61{
62 ASSERT(scriptElement.element().isConnected());
63
64 m_document.incrementLoadEventDelayCount();
65
66 auto pendingScript = PendingScript::create(scriptElement, loadableScript);
67 switch (executionType) {
68 case ASYNC_EXECUTION:
69 m_pendingAsyncScripts.add(pendingScript.copyRef());
70 break;
71 case IN_ORDER_EXECUTION:
72 m_scriptsToExecuteInOrder.append(pendingScript.copyRef());
73 break;
74 }
75 pendingScript->setClient(*this);
76}

78void ScriptRunner::suspend()
79{
80 m_timer.stop();
81}

83void ScriptRunner::resume()
84{
85 if (hasPendingScripts() && !m_document.hasActiveParserYieldToken())
86 m_timer.startOneShot(0_s);
87}
88
89void ScriptRunner::documentFinishedParsing()
90{
91 if (!m_scriptsToExecuteSoon.isEmpty() && !m_timer.isActive())
92 resume();
93}
94
95void ScriptRunner::notifyFinished(PendingScript& pendingScript)
96{
97 if (pendingScript.element().willExecuteInOrder())
98 ASSERT(!m_scriptsToExecuteInOrder.isEmpty());
99 else
100 m_scriptsToExecuteSoon.append(m_pendingAsyncScripts.take(pendingScript).releaseNonNull());
101 pendingScript.clearClient();
102
103 if (!m_document.hasActiveParserYieldToken())
104 m_timer.startOneShot(0_s);
105}
106
107void ScriptRunner::timerFired()
108{
109 Ref<Document> protect(m_document);
110
111 Vector<RefPtr<PendingScript>> scripts;
112
113 if (m_document.shouldDeferAsynchronousScriptsUntilParsingFinishes()) {
114 // Scripts not added by the parser are executed asynchronously and yet do not have the 'async' attribute set.
115 // We only want to delay scripts that were explicitly marked as 'async' by the developer.
116 m_scriptsToExecuteSoon.removeAllMatching([&](auto& pendingScript) {
117 if (pendingScript->element().hasAsyncAttribute())
118 return false;
119 scripts.append(WTFMove(pendingScript));
120 return true;
121 });
122 } else
123 scripts.swap(m_scriptsToExecuteSoon);
124
125 size_t numInOrderScriptsToExecute = 0;
126 for (; numInOrderScriptsToExecute < m_scriptsToExecuteInOrder.size() && m_scriptsToExecuteInOrder[numInOrderScriptsToExecute]->isLoaded(); ++numInOrderScriptsToExecute)
127 scripts.append(m_scriptsToExecuteInOrder[numInOrderScriptsToExecute].ptr());
128 if (numInOrderScriptsToExecute)
129 m_scriptsToExecuteInOrder.remove(0, numInOrderScriptsToExecute);
130
131 for (auto& currentScript : scripts) {
132 auto script = WTFMove(currentScript);
133 ASSERT(script);
134 // Paper over https://bugs.webkit.org/show_bug.cgi?id=144050
135 if (!script)
136 continue;
137 ASSERT(script->needsLoading());
138 script->element().executePendingScript(*script);
139 m_document.decrementLoadEventDelayCount();
140 }
141}
