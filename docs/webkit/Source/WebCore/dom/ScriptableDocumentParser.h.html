26#pragma once
27
28#include "DecodedDataDocumentParser.h"
29#include "FragmentScriptingPermission.h"
30#include "Timer.h"
31#include <wtf/text/TextPosition.h>
32
33namespace WebCore {
34
35class ScriptableDocumentParser : public DecodedDataDocumentParser {
36public:
37 // Only used by Document::open for deciding if its safe to act on a
38 // JavaScript document.open() call right now, or it should be ignored.
39 virtual bool isExecutingScript() const { return false; }
40
41 virtual TextPosition textPosition() const = 0;
42
43 virtual bool hasScriptsWaitingForStylesheets() const { return false; }
44
45 void executeScriptsWaitingForStylesheetsSoon();
46
47 // Returns true if the parser didn't yield or pause or synchronously execute a script,
48 // so calls to PageConsoleClient should be associated with the parser's text position.
49 virtual bool shouldAssociateConsoleMessagesWithTextPosition() const = 0;
50
51 void setWasCreatedByScript(bool wasCreatedByScript) { m_wasCreatedByScript = wasCreatedByScript; }
52 bool wasCreatedByScript() const { return m_wasCreatedByScript; }
53
54 ParserContentPolicy parserContentPolicy() { return m_parserContentPolicy; }
55
56protected:
57 explicit ScriptableDocumentParser(Document&, ParserContentPolicy = AllowScriptingContent);
58
59 virtual void executeScriptsWaitingForStylesheets() { }
60
61 void detach() override;
62
63private:
64 ScriptableDocumentParser* asScriptableDocumentParser() final { return this; }
65
66 void scriptsWaitingForStylesheetsExecutionTimerFired();
67
68 // http://www.whatwg.org/specs/web-apps/current-work/#script-created-parser
69 bool m_wasCreatedByScript;
70 ParserContentPolicy m_parserContentPolicy;
71 Timer m_scriptsWaitingForStylesheetsExecutionTimer;
72};
73
74} // namespace WebCore
