49bool ScriptedAnimationController::requestAnimationFrameEnabled() const
50{
51 return m_document && m_document->settings().requestAnimationFrameEnabled();
52}

54void ScriptedAnimationController::suspend()
55{
56 ++m_suspendCount;
57}
58
59void ScriptedAnimationController::resume()
60{
61 // It would be nice to put an ASSERT(m_suspendCount > 0) here, but in WK1 resume() can be called
62 // even when suspend hasn't (if a tab was created in the background).
63 if (m_suspendCount > 0)
64 --m_suspendCount;
65
66 if (!m_suspendCount && m_callbackDataList.size())
67 scheduleAnimation();
68}

80Seconds ScriptedAnimationController::preferredScriptedAnimationInterval() const
81{
82 auto* page = this->page();
83 if (!page)
84 return FullSpeedAnimationInterval;
85
86 return preferredFrameInterval(throttlingReasons(), page->displayNominalFramesPerSecond(), page->settings().preferPageRenderingUpdatesNear60FPSEnabled());
87}
88
89OptionSet<ThrottlingReason> ScriptedAnimationController::throttlingReasons() const
90{
91 if (auto* page = this->page())
92 return page->throttlingReasons() | m_throttlingReasons;
93
94 return m_throttlingReasons;
95}

97bool ScriptedAnimationController::isThrottledRelativeToPage() const
98{
99 if (auto* page = this->page())
100 return preferredScriptedAnimationInterval() > page->preferredRenderingUpdateInterval();
101 return false;
102}

104bool ScriptedAnimationController::shouldRescheduleRequestAnimationFrame(ReducedResolutionSeconds timestamp) const
105{
106 LOG_WITH_STREAM(RequestAnimationFrame, stream << "ScriptedAnimationController::shouldRescheduleRequestAnimationFrame - throttled relative to page " << isThrottledRelativeToPage()
107 << ", last delta " << (timestamp - m_lastAnimationFrameTimestamp).milliseconds() << "ms, preferred interval " << preferredScriptedAnimationInterval().milliseconds() << ")");
108
109 return timestamp <= m_lastAnimationFrameTimestamp || (isThrottledRelativeToPage() && (timestamp - m_lastAnimationFrameTimestamp < preferredScriptedAnimationInterval()));
110}

112ScriptedAnimationController::CallbackId ScriptedAnimationController::registerCallback(Ref<RequestAnimationFrameCallback>&& callback)
113{
114 CallbackId callbackId = ++m_nextCallbackId;
115 callback->m_firedOrCancelled = false;
116 callback->m_id = callbackId;
117 m_callbackDataList.append({ WTFMove(callback), UserGestureIndicator::currentUserGesture() });
118
119 if (m_document)
120 InspectorInstrumentation::didRequestAnimationFrame(*m_document, callbackId);
121
122 if (!m_suspendCount)
123 scheduleAnimation();
124 return callbackId;
125}

127void ScriptedAnimationController::cancelCallback(CallbackId callbackId)
128{
129 bool cancelled = m_callbackDataList.removeFirstMatching([callbackId](auto& data) {
130 if (data.callback->m_id != callbackId)
131 return false;
132 data.callback->m_firedOrCancelled = true;
133 return true;
134 });
135
136 if (cancelled && m_document)
137 InspectorInstrumentation::didCancelAnimationFrame(*m_document, callbackId);
138}

140void ScriptedAnimationController::serviceRequestAnimationFrameCallbacks(ReducedResolutionSeconds timestamp)
141{
142 if (!m_callbackDataList.size() || m_suspendCount || !requestAnimationFrameEnabled())
143 return;
144
145 if (shouldRescheduleRequestAnimationFrame(timestamp)) {
146 LOG_WITH_STREAM(RequestAnimationFrame, stream << "ScriptedAnimationController::serviceRequestAnimationFrameCallbacks - rescheduling (page update interval " << (page() ? page()->preferredRenderingUpdateInterval() : 0_s) << ", raf interval " << preferredScriptedAnimationInterval() << ")");
147 scheduleAnimation();
148 return;
149 }
150
151 TraceScope tracingScope(RAFCallbackStart, RAFCallbackEnd);
152
153 auto highResNowMs = std::round(1000 * timestamp.seconds());
154 if (m_document && m_document->quirks().needsMillisecondResolutionForHighResTimeStamp())
155 highResNowMs += 0.1;
156
157 LOG_WITH_STREAM(RequestAnimationFrame, stream << "ScriptedAnimationController::serviceRequestAnimationFrameCallbacks at " << highResNowMs << " (throttling reasons " << throttlingReasons() << ", preferred interval " << preferredScriptedAnimationInterval().milliseconds() << "ms)");
158
159 // First, generate a list of callbacks to consider. Callbacks registered from this point
160 // on are considered only for the "next" frame, not this one.
161 Vector<CallbackData> callbackDataList(m_callbackDataList);
162
163 // Invoking callbacks may detach elements from our document, which clears the document's
164 // reference to us, so take a defensive reference.
165 Ref<ScriptedAnimationController> protectedThis(*this);
166 Ref<Document> protectedDocument(*m_document);
167
168 for (auto& [callback, userGestureTokenToForward] : callbackDataList) {
169 if (callback->m_firedOrCancelled)
170 continue;
171 callback->m_firedOrCancelled = true;
172
173 if (userGestureTokenToForward && userGestureTokenToForward->hasExpired(UserGestureToken::maximumIntervalForUserGestureForwarding))
174 userGestureTokenToForward = nullptr;
175 UserGestureIndicator gestureIndicator(userGestureTokenToForward);
176
177 InspectorInstrumentation::willFireAnimationFrame(protectedDocument, callback->m_id);
178 callback->handleEvent(highResNowMs);
179 InspectorInstrumentation::didFireAnimationFrame(protectedDocument);
180 }
181
182 // Remove any callbacks we fired from the list of pending callbacks.
183 m_callbackDataList.removeAllMatching([](auto& data) {
184 return data.callback->m_firedOrCancelled;
185 });
186
187 m_lastAnimationFrameTimestamp = timestamp;
188
189 if (m_callbackDataList.size())
190 scheduleAnimation();
191}
192
193void ScriptedAnimationController::scheduleAnimation()
194{
195 if (!requestAnimationFrameEnabled())
196 return;
197
198 if (auto* page = this->page())
199 page->scheduleRenderingUpdate(RenderingUpdateStep::AnimationFrameCallbacks);
200}
