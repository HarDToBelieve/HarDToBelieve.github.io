27#pragma once
28
29#include "AnimationFrameRate.h"
30#include "ReducedResolutionSeconds.h"
31#include "Timer.h"
32#include <wtf/OptionSet.h>
33#include <wtf/RefCounted.h>
34#include <wtf/RefPtr.h>
35#include <wtf/Vector.h>
36
37namespace WebCore {
38
39class Document;
40class Page;
41class RequestAnimationFrameCallback;
42class UserGestureToken;
43
44class ScriptedAnimationController : public RefCounted<ScriptedAnimationController>
45{
46public:
47 static Ref<ScriptedAnimationController> create(Document& document)
48 {
49 return adoptRef(*new ScriptedAnimationController(document));
50 }
51 ~ScriptedAnimationController();
52 void clearDocumentPointer() { m_document = nullptr; }
53 bool requestAnimationFrameEnabled() const;
54
55 WEBCORE_EXPORT Seconds interval() const;
56 WEBCORE_EXPORT OptionSet<ThrottlingReason> throttlingReasons() const;
57
58 void suspend();
59 void resume();
60
61 void addThrottlingReason(ThrottlingReason reason) { m_throttlingReasons.add(reason); }
62 void removeThrottlingReason(ThrottlingReason reason) { m_throttlingReasons.remove(reason); }
63
64 using CallbackId = int;
65 CallbackId registerCallback(Ref<RequestAnimationFrameCallback>&&);
66 void cancelCallback(CallbackId);
67 void serviceRequestAnimationFrameCallbacks(ReducedResolutionSeconds);
68
69private:
70 ScriptedAnimationController(Document&);
71
72 Page* page() const;
73 Seconds preferredScriptedAnimationInterval() const;
74 bool isThrottledRelativeToPage() const;
75 bool shouldRescheduleRequestAnimationFrame(ReducedResolutionSeconds) const;
76 void scheduleAnimation();
77
78 struct CallbackData {
79 Ref<RequestAnimationFrameCallback> callback;
80 RefPtr<UserGestureToken> userGestureTokenToForward;
81 };
82 Vector<CallbackData> m_callbackDataList;
83
84 WeakPtr<Document> m_document;
85 CallbackId m_nextCallbackId { 0 };
86 int m_suspendCount { 0 };
87
88 ReducedResolutionSeconds m_lastAnimationFrameTimestamp;
89 OptionSet<ThrottlingReason> m_throttlingReasons;
90};
91
92} // namespace WebCore
