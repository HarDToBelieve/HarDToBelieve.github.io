51enum class IdMatchingType : uint8_t {
52 None,
53 Rightmost,
54 Filter
55};
56
57static bool canBeUsedForIdFastPath(const CSSSelector& selector)
58{
59 return selector.match() == CSSSelector::Id
60 || (selector.match() == CSSSelector::Exact && selector.attribute() == HTMLNames::idAttr && !selector.attributeValueMatchingIsCaseInsensitive());
61}
62
63static IdMatchingType findIdMatchingType(const CSSSelector& firstSelector)
64{
65 bool inRightmost = true;
66 for (const CSSSelector* selector = &firstSelector; selector; selector = selector->tagHistory()) {
67 if (canBeUsedForIdFastPath(*selector)) {
68 if (inRightmost)
69 return IdMatchingType::Rightmost;
70 return IdMatchingType::Filter;
71 }
72 if (selector->relation() != CSSSelector::Subselector)
73 inRightmost = false;
74 }
75 return IdMatchingType::None;
76}

78SelectorDataList::SelectorDataList(const CSSSelectorList& selectorList)
79{
80 unsigned selectorCount = 0;
81 for (const CSSSelector* selector = selectorList.first(); selector; selector = CSSSelectorList::next(selector))
82 selectorCount++;
83
84 m_selectors.reserveInitialCapacity(selectorCount);
85 for (const CSSSelector* selector = selectorList.first(); selector; selector = CSSSelectorList::next(selector))
86 m_selectors.uncheckedAppend({ selector });
87
88 if (selectorCount == 1) {
89 const CSSSelector& selector = *m_selectors.first().selector;
90 if (selector.isLastInTagHistory()) {
91 switch (selector.match()) {
92 case CSSSelector::Tag:
93 m_matchType = TagNameMatch;
94 break;
95 case CSSSelector::Class:
96 m_matchType = ClassNameMatch;
97 break;
98 default:
99 if (canBeUsedForIdFastPath(selector))
100 m_matchType = RightMostWithIdMatch;
101 else
102 m_matchType = CompilableSingle;
103 break;
104 }
105 } else {
106 switch (findIdMatchingType(selector)) {
107 case IdMatchingType::None:
108 m_matchType = CompilableSingle;
109 break;
110 case IdMatchingType::Rightmost:
111 m_matchType = RightMostWithIdMatch;
112 break;
113 case IdMatchingType::Filter:
114 m_matchType = CompilableSingleWithRootFilter;
115 break;
116 }
117 }
118 } else
119 m_matchType = CompilableMultipleSelectorMatch;
120}

122inline bool SelectorDataList::selectorMatches(const SelectorData& selectorData, Element& element, const ContainerNode& rootNode) const
123{
124 SelectorChecker selectorChecker(element.document());
125 SelectorChecker::CheckingContext selectorCheckingContext(SelectorChecker::Mode::QueryingRules);
126 selectorCheckingContext.scope = rootNode.isDocumentNode() ? nullptr : &rootNode;
127 return selectorChecker.match(*selectorData.selector, element, selectorCheckingContext);
128}

130inline Element* SelectorDataList::selectorClosest(const SelectorData& selectorData, Element& element, const ContainerNode& rootNode) const
131{
132 SelectorChecker selectorChecker(element.document());
133 SelectorChecker::CheckingContext selectorCheckingContext(SelectorChecker::Mode::QueryingRules);
134 selectorCheckingContext.scope = rootNode.isDocumentNode() ? nullptr : &rootNode;
135 if (!selectorChecker.match(*selectorData.selector, element, selectorCheckingContext))
136 return nullptr;
137 return &element;
138}

140bool SelectorDataList::matches(Element& targetElement) const
141{
142 for (auto& selctor : m_selectors) {
143 if (selectorMatches(selctor, targetElement, targetElement))
144 return true;
145 }
146 return false;
147}

149Element* SelectorDataList::closest(Element& targetElement) const
150{
151 for (auto& currentElement : lineageOfType<Element>(targetElement)) {
152 for (auto& selector : m_selectors) {
153 if (auto* candidateElement = selectorClosest(selector, currentElement, targetElement))
154 return candidateElement;
155 }
156 }
157 return nullptr;
158}

191static const CSSSelector* selectorForIdLookup(const ContainerNode& rootNode, const CSSSelector& firstSelector)
192{
193 if (!rootNode.isConnected())
194 return nullptr;
195 if (rootNode.document().inQuirksMode())
196 return nullptr;
197
198 for (const CSSSelector* selector = &firstSelector; selector; selector = selector->tagHistory()) {
199 if (canBeUsedForIdFastPath(*selector))
200 return selector;
201 if (selector->relation() != CSSSelector::Subselector)
202 break;
203 }
204
205 return nullptr;
206}

213template <typename SelectorQueryTrait>
214ALWAYS_INLINE void SelectorDataList::executeFastPathForIdSelector(const ContainerNode& rootNode, const SelectorData& selectorData, const CSSSelector* idSelector, typename SelectorQueryTrait::OutputType& output) const
215{
216 ASSERT(m_selectors.size() == 1);
217 ASSERT(idSelector);
218
219 const AtomString& idToMatch = idSelector->value();
220 if (UNLIKELY(rootNode.treeScope().containsMultipleElementsWithId(idToMatch))) {
221 const Vector<Element*>* elements = rootNode.treeScope().getAllElementsById(idToMatch);
222 ASSERT(elements);
223 bool rootNodeIsTreeScopeRoot = isTreeScopeRoot(rootNode);
224 for (auto& element : *elements) {
225 if ((rootNodeIsTreeScopeRoot || element->isDescendantOf(rootNode)) && selectorMatches(selectorData, *element, rootNode)) {
226 SelectorQueryTrait::appendOutputForElement(output, element);
227 if (SelectorQueryTrait::shouldOnlyMatchFirstElement)
228 return;
229 }
230 }
231 return;
232 }
233
234 Element* element = rootNode.treeScope().getElementById(idToMatch);
235 if (!element || !(isTreeScopeRoot(rootNode) || element->isDescendantOf(rootNode)))
236 return;
237 if (selectorMatches(selectorData, *element, rootNode))
238 SelectorQueryTrait::appendOutputForElement(output, element);
239}

241static ContainerNode& filterRootById(ContainerNode& rootNode, const CSSSelector& firstSelector)
242{
243 if (!rootNode.isConnected())
244 return rootNode;
245 if (rootNode.document().inQuirksMode())
246 return rootNode;
247
248 // If there was an Id match in the rightmost Simple Selector, we should be in a RightMostWithIdMatch, not in filter.
249 // Thus we can skip the rightmost match.
250 const CSSSelector* selector = &firstSelector;
251 do {
252 ASSERT(!canBeUsedForIdFastPath(*selector));
253 if (selector->relation() != CSSSelector::Subselector)
254 break;
255 selector = selector->tagHistory();
256 } while (selector);
257
258 bool inAdjacentChain = false;
259 for (; selector; selector = selector->tagHistory()) {
260 if (canBeUsedForIdFastPath(*selector)) {
261 const AtomString& idToMatch = selector->value();
262 if (ContainerNode* searchRoot = rootNode.treeScope().getElementById(idToMatch)) {
263 if (LIKELY(!rootNode.treeScope().containsMultipleElementsWithId(idToMatch))) {
264 if (inAdjacentChain)
265 searchRoot = searchRoot->parentNode();
266 if (searchRoot && (isTreeScopeRoot(rootNode) || searchRoot == &rootNode || searchRoot->isDescendantOf(rootNode)))
267 return *searchRoot;
268 }
269 }
270 }
271 if (selector->relation() == CSSSelector::Subselector)
272 continue;
273 if (selector->relation() == CSSSelector::DirectAdjacent || selector->relation() == CSSSelector::IndirectAdjacent)
274 inAdjacentChain = true;
275 else
276 inAdjacentChain = false;
277 }
278 return rootNode;
279}

289template <typename SelectorQueryTrait>
290static inline void elementsForLocalName(const ContainerNode& rootNode, const AtomString& localName, const AtomString& lowercaseLocalName, typename SelectorQueryTrait::OutputType& output)
291{
292 if (localName == lowercaseLocalName) {
293 for (auto& element : descendantsOfType<Element>(const_cast<ContainerNode&>(rootNode))) {
294 if (element.tagQName().localName() == localName) {
295 SelectorQueryTrait::appendOutputForElement(output, &element);
296 if (SelectorQueryTrait::shouldOnlyMatchFirstElement)
297 return;
298 }
299 }
300 } else {
301 for (auto& element : descendantsOfType<Element>(const_cast<ContainerNode&>(rootNode))) {
302 if (localNameMatches(element, localName, lowercaseLocalName)) {
303 SelectorQueryTrait::appendOutputForElement(output, &element);
304 if (SelectorQueryTrait::shouldOnlyMatchFirstElement)
305 return;
306 }
307 }
308 }
309}

322template <typename SelectorQueryTrait>
323ALWAYS_INLINE void SelectorDataList::executeSingleTagNameSelectorData(const ContainerNode& rootNode, const SelectorData& selectorData, typename SelectorQueryTrait::OutputType& output) const
324{
325 ASSERT(m_selectors.size() == 1);
326 ASSERT(isSingleTagNameSelector(*selectorData.selector));
327
328 const QualifiedName& tagQualifiedName = selectorData.selector->tagQName();
329 const AtomString& selectorLocalName = tagQualifiedName.localName();
330 const AtomString& selectorLowercaseLocalName = selectorData.selector->tagLowercaseLocalName();
331 const AtomString& selectorNamespaceURI = tagQualifiedName.namespaceURI();
332
333 if (selectorNamespaceURI == starAtom()) {
334 if (selectorLocalName != starAtom()) {
335 // Common case: name defined, selectorNamespaceURI is a wildcard.
336 elementsForLocalName<SelectorQueryTrait>(rootNode, selectorLocalName, selectorLowercaseLocalName, output);
337 } else {
338 // Other fairly common case: both are wildcards.
339 anyElement<SelectorQueryTrait>(rootNode, output);
340 }
341 } else {
342 // Fallback: NamespaceURI is set, selectorLocalName may be starAtom().
343 for (auto& element : descendantsOfType<Element>(const_cast<ContainerNode&>(rootNode))) {
344 if (element.namespaceURI() == selectorNamespaceURI && localNameMatches(element, selectorLocalName, selectorLowercaseLocalName)) {
345 SelectorQueryTrait::appendOutputForElement(output, &element);
346 if (SelectorQueryTrait::shouldOnlyMatchFirstElement)
347 return;
348 }
349 }
350 }

353template <typename SelectorQueryTrait>
354ALWAYS_INLINE void SelectorDataList::executeSingleClassNameSelectorData(const ContainerNode& rootNode, const SelectorData& selectorData, typename SelectorQueryTrait::OutputType& output) const
355{
356 ASSERT(m_selectors.size() == 1);
357 ASSERT(isSingleClassNameSelector(*selectorData.selector));
358
359 const AtomString& className = selectorData.selector->value();
360 for (auto& element : descendantsOfType<Element>(const_cast<ContainerNode&>(rootNode))) {
361 if (element.hasClass() && element.classNames().contains(className)) {
362 SelectorQueryTrait::appendOutputForElement(output, &element);
363 if (SelectorQueryTrait::shouldOnlyMatchFirstElement)
364 return;
365 }
366 }

369template <typename SelectorQueryTrait>
370ALWAYS_INLINE void SelectorDataList::executeSingleSelectorData(const ContainerNode& rootNode, const ContainerNode& searchRootNode, const SelectorData& selectorData, typename SelectorQueryTrait::OutputType& output) const
371{
372 ASSERT(m_selectors.size() == 1);
373
374 for (auto& element : descendantsOfType<Element>(const_cast<ContainerNode&>(searchRootNode))) {
375 if (selectorMatches(selectorData, element, rootNode)) {
376 SelectorQueryTrait::appendOutputForElement(output, &element);
377 if (SelectorQueryTrait::shouldOnlyMatchFirstElement)
378 return;
379 }

430template <typename SelectorQueryTrait>
431ALWAYS_INLINE void SelectorDataList::executeCompiledSingleMultiSelectorData(const ContainerNode& rootNode, typename SelectorQueryTrait::OutputType& output) const
432{
433 SelectorChecker::CheckingContext checkingContext(SelectorChecker::Mode::QueryingRules);
434 checkingContext.scope = rootNode.isDocumentNode() ? nullptr : &rootNode;
435 for (auto& element : descendantsOfType<Element>(const_cast<ContainerNode&>(rootNode))) {
436 for (auto& selector : m_selectors) {
437 selector.compiledSelector.wasUsed();
438
439 bool matched = false;
440 if (selector.compiledSelector.status == SelectorCompilationStatus::SimpleSelectorChecker)
441 matched = SelectorCompiler::querySelectorSimpleSelectorChecker(selector.compiledSelector, &element);
442 else {
443 ASSERT(selector.compiledSelector.status == SelectorCompilationStatus::SelectorCheckerWithCheckingContext);
444 matched = SelectorCompiler::querySelectorSelectorCheckerWithCheckingContext(selector.compiledSelector, &element, &checkingContext);
445 }
446 if (matched) {
447 SelectorQueryTrait::appendOutputForElement(output, &element);
448 if (SelectorQueryTrait::shouldOnlyMatchFirstElement)
449 return;
450 break;
451 }
452 }
453 }

456bool SelectorDataList::compileSelector(const SelectorData& selectorData)
457{
458 auto& compiledSelector = selectorData.compiledSelector;
459
460 if (compiledSelector.status == SelectorCompilationStatus::NotCompiled)
461 SelectorCompiler::compileSelector(compiledSelector, selectorData.selector, SelectorCompiler::SelectorContext::QuerySelector);
462
463 return compiledSelector.status != SelectorCompilationStatus::CannotCompile;
464}

468template <typename SelectorQueryTrait>
469ALWAYS_INLINE void SelectorDataList::execute(ContainerNode& rootNode, typename SelectorQueryTrait::OutputType& output) const
470{
471 ContainerNode* searchRootNode = &rootNode;
472 switch (m_matchType) {
473 case RightMostWithIdMatch:
474 {
475 const SelectorData& selectorData = m_selectors.first();
476 if (const CSSSelector* idSelector = selectorForIdLookup(*searchRootNode, *selectorData.selector)) {
477 executeFastPathForIdSelector<SelectorQueryTrait>(*searchRootNode, m_selectors.first(), idSelector, output);
478 break;
479 }
480#if ENABLE(CSS_SELECTOR_JIT)
481 if (compileSelector(selectorData))
482 goto CompiledSingleCase;
483#endif // ENABLE(CSS_SELECTOR_JIT)
484 goto SingleSelectorCase;
485 ASSERT_NOT_REACHED();
486 }
487
488 case CompilableSingleWithRootFilter:
489 case CompilableSingle:
490 {
491#if ENABLE(CSS_SELECTOR_JIT)
492 const SelectorData& selectorData = m_selectors.first();
493 ASSERT(selectorData.compiledSelector.status == SelectorCompilationStatus::NotCompiled);
494 ASSERT(m_matchType == CompilableSingle || m_matchType == CompilableSingleWithRootFilter);
495 if (compileSelector(selectorData)) {
496 if (m_matchType == CompilableSingle) {
497 m_matchType = CompiledSingle;
498 goto CompiledSingleCase;
499 }
500 ASSERT(m_matchType == CompilableSingleWithRootFilter);
501 m_matchType = CompiledSingleWithRootFilter;
502 goto CompiledSingleWithRootFilterCase;
503 }
504#endif // ENABLE(CSS_SELECTOR_JIT)
505 if (m_matchType == CompilableSingle) {
506 m_matchType = SingleSelector;
507 goto SingleSelectorCase;
508 }
509 ASSERT(m_matchType == CompilableSingleWithRootFilter);
510 m_matchType = SingleSelectorWithRootFilter;
511 goto SingleSelectorWithRootFilterCase;
512 ASSERT_NOT_REACHED();
513 }
514
515#if ENABLE(CSS_SELECTOR_JIT)
516 case CompiledSingleWithRootFilter:
517 CompiledSingleWithRootFilterCase:
518 searchRootNode = &filterRootById(*searchRootNode, *m_selectors.first().selector);
519 FALLTHROUGH;
520 case CompiledSingle:
521 {
522 CompiledSingleCase:
523 const SelectorData& selectorData = m_selectors.first();
524 if (selectorData.compiledSelector.status == SelectorCompilationStatus::SimpleSelectorChecker) {
525 executeCompiledSimpleSelectorChecker<SelectorQueryTrait>(*searchRootNode, [&] (const Element* element) {
526 return SelectorCompiler::querySelectorSimpleSelectorChecker(selectorData.compiledSelector, element);
527 }, output, selectorData);
528 } else {
529 ASSERT(selectorData.compiledSelector.status == SelectorCompilationStatus::SelectorCheckerWithCheckingContext);
530 executeCompiledSelectorCheckerWithCheckingContext<SelectorQueryTrait>(rootNode, *searchRootNode, [&] (const Element* element, const SelectorChecker::CheckingContext* context) {
531 return SelectorCompiler::querySelectorSelectorCheckerWithCheckingContext(selectorData.compiledSelector, element, context);
532 }, output, selectorData);
533 }
534 break;
535 }
536#else
537 case CompiledSingleWithRootFilter:
538 case CompiledSingle:
539 ASSERT_NOT_REACHED();
540#if !ASSERT_ENABLED
541 FALLTHROUGH;
542#endif
543#endif // ENABLE(CSS_SELECTOR_JIT)
544
545 case SingleSelectorWithRootFilter:
546 SingleSelectorWithRootFilterCase:
547 searchRootNode = &filterRootById(*searchRootNode, *m_selectors.first().selector);
548 FALLTHROUGH;
549 case SingleSelector:
550 SingleSelectorCase:
551 executeSingleSelectorData<SelectorQueryTrait>(rootNode, *searchRootNode, m_selectors.first(), output);
552 break;
553
554 case TagNameMatch:
555 executeSingleTagNameSelectorData<SelectorQueryTrait>(*searchRootNode, m_selectors.first(), output);
556 break;
557 case ClassNameMatch:
558 executeSingleClassNameSelectorData<SelectorQueryTrait>(*searchRootNode, m_selectors.first(), output);
559 break;
560 case CompilableMultipleSelectorMatch:
561#if ENABLE(CSS_SELECTOR_JIT)
562 {
563 for (auto& selector : m_selectors) {
564 if (!compileSelector(selector)) {
565 m_matchType = MultipleSelectorMatch;
566 goto MultipleSelectorMatch;
567 }
568 }
569 m_matchType = CompiledMultipleSelectorMatch;
570 goto CompiledMultipleSelectorMatch;
571 }
572#else
573 FALLTHROUGH;
574#endif // ENABLE(CSS_SELECTOR_JIT)
575 case CompiledMultipleSelectorMatch:
576#if ENABLE(CSS_SELECTOR_JIT)
577 CompiledMultipleSelectorMatch:
578 executeCompiledSingleMultiSelectorData<SelectorQueryTrait>(*searchRootNode, output);
579 break;
580#else
581 FALLTHROUGH;
582#endif // ENABLE(CSS_SELECTOR_JIT)
583 case MultipleSelectorMatch:
584#if ENABLE(CSS_SELECTOR_JIT)
585 MultipleSelectorMatch:
586#endif
587 executeSingleMultiSelectorData<SelectorQueryTrait>(*searchRootNode, output);
588 break;
589 }

598ExceptionOr<SelectorQuery&> SelectorQueryCache::add(const String& selectors, Document& document)
599{
600 if (auto* entry = m_entries.get(selectors))
601 return *entry;
602
603 CSSParser parser(document);
604 auto selectorList = parser.parseSelector(selectors);
605
606 if (!selectorList || selectorList->hasInvalidSelector())
607 return Exception { SyntaxError };
608
609 if (selectorList->selectorsNeedNamespaceResolution())
610 return Exception { SyntaxError };
611
612 const int maximumSelectorQueryCacheSize = 256;
613 if (m_entries.size() == maximumSelectorQueryCacheSize)
614 m_entries.remove(m_entries.random());
615
616 return *m_entries.add(selectors, makeUnique<SelectorQuery>(WTFMove(*selectorList))).iterator->value;
617}
