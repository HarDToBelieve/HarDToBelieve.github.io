43class SelectorDataList {

45 explicit SelectorDataList(const CSSSelectorList&);

49 Element* queryFirst(ContainerNode& rootNode) const;

51private:
52 struct SelectorData {
53 const CSSSelector* selector;
54#if ENABLE(CSS_SELECTOR_JIT)
55 mutable CompiledSelector compiledSelector { };

57 };

59 bool selectorMatches(const SelectorData&, Element&, const ContainerNode& rootNode) const;

62 template <typename SelectorQueryTrait> void execute(ContainerNode& rootNode, typename SelectorQueryTrait::OutputType&) const;

67 template <typename SelectorQueryTrait> void executeSingleMultiSelectorData(const ContainerNode& rootNode, typename SelectorQueryTrait::OutputType&) const;
68#if ENABLE(CSS_SELECTOR_JIT)

92class SelectorQuery {
93 WTF_MAKE_NONCOPYABLE(SelectorQuery);
94 WTF_MAKE_FAST_ALLOCATED;
95
96public:
97 explicit SelectorQuery(CSSSelectorList&&);
98 bool matches(Element&) const;
99 Element* closest(Element&) const;
100 Ref<NodeList> queryAll(ContainerNode& rootNode) const;
101 Element* queryFirst(ContainerNode& rootNode) const;
102
103private:
104 CSSSelectorList m_selectorList;
105 SelectorDataList m_selectors;
106};

108class SelectorQueryCache {
109 WTF_MAKE_FAST_ALLOCATED;
110public:
111 ExceptionOr<SelectorQuery&> add(const String&, Document&);
112private:
113 HashMap<String, std::unique_ptr<SelectorQuery>> m_entries;
114};

116inline bool SelectorQuery::matches(Element& element) const
117{
118 return m_selectors.matches(element);
119}
120
121inline Element* SelectorQuery::closest(Element& element) const
122{
123 return m_selectors.closest(element);
124}
125
126inline Ref<NodeList> SelectorQuery::queryAll(ContainerNode& rootNode) const
127{
128 return m_selectors.queryAll(rootNode);
129}
130
131inline Element* SelectorQuery::queryFirst(ContainerNode& rootNode) const
132{
133 return m_selectors.queryFirst(rootNode);
134}
