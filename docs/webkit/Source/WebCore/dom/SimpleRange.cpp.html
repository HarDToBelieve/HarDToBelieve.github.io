54std::optional<SimpleRange> makeRangeSelectingNode(Node& node)
55{
56 auto parent = node.parentNode();
57 if (!parent)
58 return std::nullopt;
59 unsigned offset = node.computeNodeIndex();
60 return SimpleRange { { *parent, offset }, { *parent, offset + 1 } };
61}

63SimpleRange makeRangeSelectingNodeContents(Node& node)
64{

66}

68OffsetRange characterDataOffsetRange(const SimpleRange& range, const Node& node)
69{
70 return { &node == range.start.container.ptr() ? range.start.offset : 0,
71 &node == range.end.container.ptr() ? range.end.offset : std::numeric_limits<unsigned>::max() };
72}

74static RefPtr<Node> firstIntersectingNode(const SimpleRange& range)
75{
76 if (range.start.container->isCharacterDataNode())
77 return range.start.container.ptr();
78 if (auto child = range.start.container->traverseToChildAt(range.start.offset))
79 return child;
80 return NodeTraversal::nextSkippingChildren(range.start.container);
81}
82
83static RefPtr<Node> nodePastLastIntersectingNode(const SimpleRange& range)
84{
85 if (range.end.container->isCharacterDataNode())
86 return NodeTraversal::nextSkippingChildren(range.end.container);
87 if (auto child = range.end.container->traverseToChildAt(range.end.offset))
88 return child;
89 return NodeTraversal::nextSkippingChildren(range.end.container);
90}
91
92static RefPtr<Node> firstIntersectingNodeWithDeprecatedZeroOffsetStartQuirk(const SimpleRange& range)
93{
94 if (range.start.container->isCharacterDataNode())
95 return range.start.container.ptr();
96 if (auto child = range.start.container->traverseToChildAt(range.start.offset))
97 return child;
98 if (!range.start.offset)
99 return range.start.container.ptr();
100 return NodeTraversal::nextSkippingChildren(range.start.container);
101}

117void IntersectingNodeIterator::advance()
118{
119 ASSERT(m_node);
120 m_node = NodeTraversal::next(*m_node);
121 enforceEndInvariant();
122}
123
124void IntersectingNodeIterator::advanceSkippingChildren()
125{
126 ASSERT(m_node);
127 m_node = m_node->contains(m_pastLastNode.get()) ? nullptr : NodeTraversal::nextSkippingChildren(*m_node);
128 enforceEndInvariant();
129}
130
131void IntersectingNodeIterator::enforceEndInvariant()
132{
133 if (m_node == m_pastLastNode || !m_node) {
134 m_node = nullptr;
135 m_pastLastNode = nullptr;
136 }
137}

174template<TreeType treeType> PartialOrdering treeOrder(const SimpleRange& range, const BoundaryPoint& point)
175{
176 if (auto order = treeOrder<treeType>(range.start, point); !is_lt(order))
177 return order;
178 if (auto order = treeOrder<treeType>(range.end, point); !is_gt(order))
179 return order;
180 return PartialOrdering::equivalent;
181}

186 return order;

189 return PartialOrdering::equivalent;

244SimpleRange unionRange(const SimpleRange& a, const SimpleRange& b)
245{
246 return { std::min(a.start, b.start, compareByComposedTreeOrder), std::max(a.end, b.end, compareByComposedTreeOrder) };

249std::optional<SimpleRange> intersection(const std::optional<SimpleRange>& a, const std::optional<SimpleRange>& b)
250{
251 // FIXME: Can this be done more efficiently, with fewer calls to treeOrder?
252 if (!a || !b || !intersects<ComposedTree>(*a, *b))
253 return std::nullopt;
254 return { { std::max(a->start, b->start, compareByComposedTreeOrder), std::min(a->end, b->end, compareByComposedTreeOrder) } };
255}

282template<TreeType treeType> bool contains(const Node& outer, const Node& inner)
283{
284 for (auto inclusiveAncestor = &inner; inclusiveAncestor; inclusiveAncestor = parent<treeType>(*inclusiveAncestor)) {
285 if (inclusiveAncestor == &outer)
286 return true;
287 }
288 return false;
289}

297template<TreeType treeType> bool intersects(const SimpleRange& range, const Node& node)
298{
299 // FIXME: Consider a more efficient algorithm that avoids always computing the node index.
300 // FIXME: Does this const_cast point to a design problem?
301 auto nodeRange = makeRangeSelectingNode(const_cast<Node&>(node));
302 if (!nodeRange)
303 return contains<treeType>(node, range.start.container);
304 return is_lt(treeOrder<treeType>(nodeRange->start, range.end)) && is_lt(treeOrder<treeType>(range.start, nodeRange->end));
305}

324bool containsCrossingDocumentBoundaries(const SimpleRange& range, Node& node)
325{
326 auto* ancestor = &node;
327 while (&range.start.document() != &ancestor->document()) {
328 ancestor = ancestor->document().ownerElement();
329 if (!ancestor)
330 return false;
331 }
332 return contains<ComposedTree>(range, *ancestor);
333}
