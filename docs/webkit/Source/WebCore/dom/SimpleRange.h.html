32struct SimpleRange {
33 BoundaryPoint start;
34 BoundaryPoint end;
35
36 Node& startContainer() const { return start.container.get(); }
37 unsigned startOffset() const { return start.offset; }
38 Node& endContainer() const { return end.container.get(); }
39 unsigned endOffset() const { return end.offset; }
40
41 bool collapsed() const { return start == end; }
42
43 WEBCORE_EXPORT SimpleRange(const BoundaryPoint&, const BoundaryPoint&);
44 WEBCORE_EXPORT SimpleRange(BoundaryPoint&&, BoundaryPoint&&);
45};

175inline SimpleRange makeSimpleRangeHelper(BoundaryPoint&& start, BoundaryPoint&& end)
176{
177 return { WTFMove(start), WTFMove(end) };
178}
179
180inline std::optional<SimpleRange> makeSimpleRangeHelper(std::optional<BoundaryPoint>&& start, std::optional<BoundaryPoint>&& end)
181{
182 if (!start || !end)
183 return std::nullopt;
184 return makeSimpleRangeHelper(WTFMove(*start), WTFMove(*end));
185}
186
187inline SimpleRange makeSimpleRangeHelper(BoundaryPoint&& point)
188{
189 auto end = point;
190 return makeSimpleRangeHelper(WTFMove(point), WTFMove(end));
191}
192
193inline std::optional<SimpleRange> makeSimpleRangeHelper(std::optional<BoundaryPoint>&& point)
194{
195 if (!point)
196 return std::nullopt;
197 return makeSimpleRangeHelper(WTFMove(*point));
198}
