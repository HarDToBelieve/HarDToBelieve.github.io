29#include "config.h"
30#include "StaticNodeList.h"
31
32#include <wtf/IsoMallocInlines.h>
33
34namespace WebCore {
35
36WTF_MAKE_ISO_ALLOCATED_IMPL(StaticNodeList);
37WTF_MAKE_ISO_ALLOCATED_IMPL(StaticElementList);
38
39unsigned StaticNodeList::length() const
40{
41 return m_nodes.size();
42}
43
44Node* StaticNodeList::item(unsigned index) const
45{
46 if (index < m_nodes.size())
47 return const_cast<Node*>(m_nodes[index].ptr());
48 return nullptr;
49}
50
51unsigned StaticElementList::length() const
52{
53 return m_elements.size();
54}
55
56Element* StaticElementList::item(unsigned index) const
57{
58 if (index < m_elements.size())
59 return const_cast<Element*>(m_elements[index].ptr());
60 return nullptr;
61}
