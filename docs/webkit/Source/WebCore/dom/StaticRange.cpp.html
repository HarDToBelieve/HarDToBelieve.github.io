54static bool isDocumentTypeOrAttr(Node& node)
55{
56 // Before calling nodeType, do two fast non-virtual checks that cover almost all normal nodes, but are false for DocumentType and Attr.
57 if (is<ContainerNode>(node) || is<Text>(node))
58 return false;
59
60 // Call nodeType explicitly and use a switch so we don't have to call it twice.
61 switch (node.nodeType()) {
62 case Node::ATTRIBUTE_NODE:
63 case Node::DOCUMENT_TYPE_NODE:
64 return true;
65 default:
66 return false;
67 }
68}
69
70ExceptionOr<Ref<StaticRange>> StaticRange::create(Init&& init)
71{
72 ASSERT(init.startContainer);
73 ASSERT(init.endContainer);
74 if (isDocumentTypeOrAttr(*init.startContainer) || isDocumentTypeOrAttr(*init.endContainer))
75 return Exception { InvalidNodeTypeError };
76 return create({ { init.startContainer.releaseNonNull(), init.startOffset }, { init.endContainer.releaseNonNull(), init.endOffset } });
77}

79void StaticRange::visitNodesConcurrently(JSC::AbstractSlotVisitor& visitor) const
80{
81 visitor.addOpaqueRoot(root(start.container.get()));
82 visitor.addOpaqueRoot(root(end.container.get()));
83}
