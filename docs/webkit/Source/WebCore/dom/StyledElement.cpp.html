60void StyledElement::synchronizeStyleAttributeInternalImpl()
61{
62 ASSERT(elementData());
63 ASSERT(elementData()->styleAttributeIsDirty());
64 elementData()->setStyleAttributeIsDirty(false);
65 if (const StyleProperties* inlineStyle = this->inlineStyle())
66 setSynchronizedLazyAttribute(styleAttr, inlineStyle->asText());

82class StyledElementInlineStylePropertyMap final : public StylePropertyMap {
83public:
84 static Ref<StylePropertyMap> create(StyledElement& element)
85 {
86 return adoptRef(*new StyledElementInlineStylePropertyMap(element));
87 }
88
89private:
90 RefPtr<TypedOMCSSStyleValue> get(const String& property) const final
91 {
92 ASSERT(m_element); // Hitting this assertion would imply a GC bug. Element is collected while this property map is alive.
93 if (!m_element)
94 return nullptr;
95 return extractInlineProperty(property, *m_element);
96 }
97
98 explicit StyledElementInlineStylePropertyMap(StyledElement& element)
99 : m_element(&element)
100 {
101 }
102
103 void clearElement() override { m_element = nullptr; }
104
105 static RefPtr<TypedOMCSSStyleValue> extractInlineProperty(const String& name, StyledElement& element)
106 {
107 if (!element.inlineStyle())
108 return nullptr;
109
110 if (isCustomPropertyName(name)) {
111 auto value = element.inlineStyle()->getCustomPropertyCSSValue(name);
112 return StylePropertyMapReadOnly::customPropertyValueOrDefault(name, element.document(), value.get(), &element);
113 }
114
115 CSSPropertyID propertyID = cssPropertyID(name);
116 if (!propertyID)
117 return nullptr;
118
119 auto value = element.inlineStyle()->getPropertyCSSValue(propertyID);
120 return StylePropertyMapReadOnly::reifyValue(value.get(), element.document(), &element);
121 }
122
123 StyledElement* m_element { nullptr };
124};

134MutableStyleProperties& StyledElement::ensureMutableInlineStyle()
135{
136 RefPtr<StyleProperties>& inlineStyle = ensureUniqueElementData().m_inlineStyle;
137 if (!inlineStyle)
138 inlineStyle = MutableStyleProperties::create(strictToCSSParserMode(isHTMLElement() && !document().inQuirksMode()));
139 else if (!is<MutableStyleProperties>(*inlineStyle))
140 inlineStyle = inlineStyle->mutableCopy();
141 return downcast<MutableStyleProperties>(*inlineStyle);
142}

144void StyledElement::attributeChanged(const QualifiedName& name, const AtomString& oldValue, const AtomString& newValue, AttributeModificationReason reason)
145{
146 if (oldValue != newValue) {
147 if (name == styleAttr)
148 styleAttributeChanged(newValue, reason);
149 else if (hasPresentationalHintsForAttribute(name)) {
150 elementData()->setPresentationalHintStyleIsDirty(true);
151 invalidateStyle();
152 }
153 }
154
155 Element::attributeChanged(name, oldValue, newValue, reason);
156}

172void StyledElement::setInlineStyleFromString(const AtomString& newStyleString)
173{
174 auto& inlineStyle = elementData()->m_inlineStyle;
175
176 // Avoid redundant work if we're using shared attribute data with already parsed inline style.
177 if (inlineStyle && !elementData()->isUnique())
178 return;
179
180 // We reconstruct the property set instead of mutating if there is no CSSOM wrapper.
181 // This makes wrapperless property sets immutable and so cacheable.
182 if (inlineStyle && !is<MutableStyleProperties>(*inlineStyle))
183 inlineStyle = nullptr;
184
185 if (!inlineStyle)
186 inlineStyle = CSSParser::parseInlineStyleDeclaration(newStyleString, this);
187 else
188 downcast<MutableStyleProperties>(*inlineStyle).parseDeclaration(newStyleString, document());
189
190 if (usesStyleBasedEditability(*inlineStyle))
191 document().setHasElementUsingStyleBasedEditability();
192}

194void StyledElement::styleAttributeChanged(const AtomString& newStyleString, AttributeModificationReason reason)
195{
196 auto startLineNumber = WTF::OrdinalNumber::beforeFirst();
197 if (document().scriptableDocumentParser() && !document().isInDocumentWrite())
198 startLineNumber = document().scriptableDocumentParser()->textPosition().m_line;
199
200 if (reason == ModifiedByCloning || document().contentSecurityPolicy()->allowInlineStyle(document().url().string(), startLineNumber, { }, isInUserAgentShadowTree()))
201 setInlineStyleFromString(newStyleString);
202
203 elementData()->setStyleAttributeIsDirty(false);
204
205 invalidateStyle();
206 InspectorInstrumentation::didInvalidateStyleAttr(*this);
207}

209void StyledElement::invalidateStyleAttribute()
210{
211 if (auto* inlineStyle = this->inlineStyle()) {
212 if (usesStyleBasedEditability(*inlineStyle))
213 document().setHasElementUsingStyleBasedEditability();
214 }
215
216 elementData()->setStyleAttributeIsDirty(true);
217 invalidateStyle();
218
219 // In the rare case of selectors like "[style] ~ div" we need to synchronize immediately to invalidate.
220 if (styleResolver().ruleSets().hasComplexSelectorsForStyleAttribute()) {
221 if (auto* inlineStyle = this->inlineStyle()) {
222 elementData()->setStyleAttributeIsDirty(false);
223 auto newValue = inlineStyle->asText();
224 Style::AttributeChangeInvalidation styleInvalidation(*this, styleAttr, attributeWithoutSynchronization(styleAttr), newValue);
225 setSynchronizedLazyAttribute(styleAttr, newValue);
226 }
227 }
228}

294void StyledElement::rebuildPresentationalHintStyle()
295{
296 auto style = MutableStyleProperties::create(isSVGElement() ? SVGAttributeMode : HTMLQuirksMode);
297 for (const Attribute& attribute : attributesIterator())
298 collectPresentationalHintsForAttribute(attribute.name(), attribute.value(), style);
299
300 if (is<HTMLImageElement>(*this))
301 collectExtraStyleForPresentationalHints(style);
302
303 // ShareableElementData doesn't store presentation attribute style, so make sure we have a UniqueElementData.
304 UniqueElementData& elementData = ensureUniqueElementData();
305
306 elementData.setPresentationalHintStyleIsDirty(false);
307 if (style->isEmpty())
308 elementData.m_presentationalHintStyle = nullptr;
309 else
310 elementData.m_presentationalHintStyle = WTFMove(style);
311}
