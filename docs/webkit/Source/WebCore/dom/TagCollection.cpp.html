32WTF_MAKE_ISO_ALLOCATED_IMPL(TagCollection);
33WTF_MAKE_ISO_ALLOCATED_IMPL(TagCollectionNS);
34WTF_MAKE_ISO_ALLOCATED_IMPL(HTMLTagCollection);

36TagCollectionNS::TagCollectionNS(ContainerNode& rootNode, const AtomString& namespaceURI, const AtomString& localName)
37 : CachedHTMLCollection(rootNode, ByTag)
38 , m_namespaceURI(namespaceURI)
39 , m_localName(localName)
40{
41 ASSERT(m_namespaceURI.isNull() || !m_namespaceURI.isEmpty());
42}
43
44TagCollectionNS::~TagCollectionNS()
45{
46 ownerNode().nodeLists()->removeCachedTagCollectionNS(*this, m_namespaceURI, m_localName);
47}
48
49TagCollection::TagCollection(ContainerNode& rootNode, const AtomString& qualifiedName)
50 : CachedHTMLCollection(rootNode, ByTag)
51 , m_qualifiedName(qualifiedName)
52{
53 ASSERT(qualifiedName != starAtom());
54}
55
56TagCollection::~TagCollection()
57{
58 ownerNode().nodeLists()->removeCachedCollection(this, m_qualifiedName);
59}
60
61HTMLTagCollection::HTMLTagCollection(ContainerNode& rootNode, const AtomString& qualifiedName)
62 : CachedHTMLCollection(rootNode, ByHTMLTag)
63 , m_qualifiedName(qualifiedName)
64 , m_loweredQualifiedName(qualifiedName.convertToASCIILowercase())
65{
66 ASSERT(qualifiedName != starAtom());
67}
68
69HTMLTagCollection::~HTMLTagCollection()
70{
71 ownerNode().nodeLists()->removeCachedCollection(this, m_qualifiedName);
72}
