28#pragma once
29
30#include "DocumentFragment.h"
31
32namespace WebCore {
33
34class TemplateContentDocumentFragment final : public DocumentFragment {
35 WTF_MAKE_ISO_ALLOCATED(TemplateContentDocumentFragment);
36public:
37 static Ref<TemplateContentDocumentFragment> create(Document& document, const Element* host)
38 {
39 return adoptRef(*new TemplateContentDocumentFragment(document, host));
40 }
41
42 const Element* host() const { return m_host; }
43 void clearHost() { m_host = nullptr; }
44
45private:
46 TemplateContentDocumentFragment(Document& document, const Element* host)
47 : DocumentFragment(document, CreateDocumentFragment)
48 , m_host(host)
49 {
50 }
51
52 bool isTemplateContent() const override { return true; }
53
54 const Element* m_host;
55};

57} // namespace WebCore
