42ExceptionOr<Ref<TextDecoder>> TextDecoder::create(const String& label, Options options)
43{
44 String strippedLabel = stripLeadingAndTrailingHTMLSpaces(label);
45 const UChar nullCharacter = '\0';
46 if (strippedLabel.contains(nullCharacter))
47 return Exception { RangeError };
48 auto decoder = adoptRef(*new TextDecoder(strippedLabel.utf8().data(), options));
49 if (!decoder->m_textEncoding.isValid() || !strcmp(decoder->m_textEncoding.name(), "replacement"))
50 return Exception { RangeError };
51 return decoder;
52}
53
54ExceptionOr<String> TextDecoder::decode(std::optional<BufferSource::VariantType> input, DecodeOptions options)
55{
56 std::optional<BufferSource> inputBuffer;
57 const uint8_t* data = nullptr;
58 size_t length = 0;
59 if (input) {
60 inputBuffer = BufferSource(WTFMove(input.value()));
61 data = inputBuffer->data();
62 length = inputBuffer->length();
63 }
64
65 if (!m_codec) {
66 m_codec = newTextCodec(m_textEncoding);
67 if (!m_options.ignoreBOM)
68 m_codec->stripByteOrderMark();
69 }
70
71 bool sawError = false;
72 String result = m_codec->decode(reinterpret_cast<const char*>(data), length, !options.stream, m_options.fatal, sawError);
73
74 if (!options.stream && !m_options.ignoreBOM)
75 m_codec->stripByteOrderMark();
76
77 if (sawError && m_options.fatal)
78 return Exception { TypeError };
79 return result;
80}
81
82String TextDecoder::encoding() const
83{
84 return String(m_textEncoding.name()).convertToASCIILowercase();
85}
