34RefPtr<Uint8Array> TextEncoderStreamEncoder::encode(const String& input)
35{
36 StringView view(input);
37
38 if (!view.length())
39 return nullptr;
40
41 Vector<uint8_t> bytes(WTF::checkedProduct<size_t>(view.length() + 1, 3));
42 size_t bytesWritten = 0;
43
44 for (size_t cptr = 0; cptr < view.length(); cptr++) {
45 // https://encoding.spec.whatwg.org/#convert-code-unit-to-scalar-value
46 auto token = view[cptr];
47 if (m_pendingHighSurrogate) {
48 auto highSurrogate = *m_pendingHighSurrogate;
49 m_pendingHighSurrogate = std::nullopt;
50 if (token >= 0xDC00 && token <= 0xDFFF) {
51 auto codePoint = 0x10000 + ((highSurrogate - 0xD800) << 10) + (token - 0xDC00);
52 U8_APPEND_UNSAFE(bytes.data(), bytesWritten, codePoint);
53 continue;
54 }
55 U8_APPEND_UNSAFE(bytes.data(), bytesWritten, 0XFFFD);
56 }
57 if (token >= 0xD800 && token <= 0xDBFF) {
58 m_pendingHighSurrogate = token;
59 continue;
60 }
61 if (token >= 0xDC00 && token <= 0xDFFF) {
62 U8_APPEND_UNSAFE(bytes.data(), bytesWritten, 0XFFFD);
63 continue;
64 }
65 U8_APPEND_UNSAFE(bytes.data(), bytesWritten, token);
66 }
67
68 if (!bytesWritten)
69 return nullptr;
70
71 bytes.shrink(bytesWritten);
72 return Uint8Array::tryCreate(bytes.data(), bytesWritten);
73}
74
75RefPtr<Uint8Array> TextEncoderStreamEncoder::flush()
76{
77 if (!m_pendingHighSurrogate)
78 return nullptr;
79
80 constexpr uint8_t byteSequence[] = { 0xEF, 0xBF, 0xBD };
81 return Uint8Array::tryCreate(byteSequence, std::size(byteSequence));
82}
