183Ref<Node> TreeScope::retargetToScope(Node& node) const
184{
185 auto& scope = node.treeScope();
186 if (LIKELY(this == &scope || !node.isInShadowTree()))
187 return node;
188 ASSERT(is<ShadowRoot>(scope.rootNode()));
189
190 Vector<TreeScope*, 8> nodeTreeScopes;
191 for (auto* currentScope = &scope; currentScope; currentScope = currentScope->parentTreeScope())
192 nodeTreeScopes.append(currentScope);
193 ASSERT(nodeTreeScopes.size() >= 2);
194
195 Vector<const TreeScope*, 8> ancestorScopes;
196 for (auto* currentScope = this; currentScope; currentScope = currentScope->parentTreeScope())
197 ancestorScopes.append(currentScope);
198
199 auto i = nodeTreeScopes.size();
200 auto j = ancestorScopes.size();
201 while (i > 0 && j > 0 && nodeTreeScopes[i - 1] == ancestorScopes[j - 1]) {
202 --i;
203 --j;
204 }
205
206 bool nodeIsInOuterTreeScope = !i;
207 if (nodeIsInOuterTreeScope)
208 return node;
209
210 auto& shadowRootInLowestCommonTreeScope = downcast<ShadowRoot>(nodeTreeScopes[i - 1]->rootNode());
211 return *shadowRootInLowestCommonTreeScope.host();
212}

296HTMLLabelElement* TreeScope::labelElementForId(const AtomString& forAttributeValue)
297{
298 if (forAttributeValue.isEmpty())
299 return nullptr;
300
301 if (!m_labelsByForAttribute) {
302 // Populate the map on first access.
303 m_labelsByForAttribute = makeUnique<TreeScopeOrderedMap>();
304
305 for (auto& label : descendantsOfType<HTMLLabelElement>(m_rootNode)) {
306 const AtomString& forValue = label.attributeWithoutSynchronization(forAttr);
307 if (!forValue.isEmpty())
308 addLabel(*forValue.impl(), label);
309 }
310 }
311
312 return m_labelsByForAttribute->getElementByLabelForAttribute(*forAttributeValue.impl(), *this);
313}

315static std::optional<LayoutPoint> absolutePointIfNotClipped(Document& document, const LayoutPoint& clientPoint)
316{
317 if (!document.frame() || !document.view())
318 return std::nullopt;
319
320 const auto& settings = document.frame()->settings();
321 if (settings.visualViewportEnabled() && settings.clientCoordinatesRelativeToLayoutViewport()) {
322 document.updateLayout();
323 if (!document.view() || !document.hasLivingRenderTree())
324 return std::nullopt;
325 auto* view = document.view();
326 FloatPoint layoutViewportPoint = view->clientToLayoutViewportPoint(clientPoint);
327 FloatRect layoutViewportBounds({ }, view->layoutViewportRect().size());
328 if (!layoutViewportBounds.contains(layoutViewportPoint))
329 return std::nullopt;
330 return LayoutPoint(view->layoutViewportToAbsolutePoint(layoutViewportPoint));
331 }
332
333 auto* frame = document.frame();
334 auto* view = document.view();
335 float scaleFactor = frame->pageZoomFactor() * frame->frameScaleFactor();
336
337 LayoutPoint absolutePoint = clientPoint;
338 absolutePoint.scale(scaleFactor);
339 absolutePoint.moveBy(view->contentsScrollPosition());
340
341 LayoutRect visibleRect;
342#if PLATFORM(IOS_FAMILY)
343 visibleRect = view->unobscuredContentRect();
344#else
345 visibleRect = view->visibleContentRect();
346#endif
347 if (visibleRect.contains(absolutePoint))
348 return absolutePoint;
349 return std::nullopt;
350}

352RefPtr<Node> TreeScope::nodeFromPoint(const LayoutPoint& clientPoint, LayoutPoint* localPoint)
353{
354 auto absolutePoint = absolutePointIfNotClipped(documentScope(), clientPoint);
355 if (!absolutePoint)
356 return nullptr;
357
358 HitTestResult result(absolutePoint.value());
359 documentScope().hitTest(HitTestRequest(), result);
360 if (localPoint)
361 *localPoint = result.localPoint();
362 return result.innerNode();
363}

365RefPtr<Element> TreeScope::elementFromPoint(double clientX, double clientY)
366{
367 Document& document = documentScope();
368 if (!document.hasLivingRenderTree())
369 return nullptr;
370
371 auto node = nodeFromPoint(LayoutPoint { clientX, clientY }, nullptr);
372 if (!node)
373 return nullptr;
374
375 node = retargetToScope(*node);
376 while (!is<Element>(*node)) {
377 node = node->parentInComposedTree();
378 if (!node)
379 break;
380 node = retargetToScope(*node);
381 }
382
383 return static_pointer_cast<Element>(node);
384}

386Vector<RefPtr<Element>> TreeScope::elementsFromPoint(double clientX, double clientY)
387{
388 Vector<RefPtr<Element>> elements;
389
390 Document& document = documentScope();
391 if (!document.hasLivingRenderTree())
392 return elements;
393
394 auto absolutePoint = absolutePointIfNotClipped(document, LayoutPoint(clientX, clientY));
395 if (!absolutePoint)
396 return elements;
397
398 constexpr OptionSet<HitTestRequest::Type> hitType { HitTestRequest::Type::ReadOnly, HitTestRequest::Type::Active, HitTestRequest::Type::DisallowUserAgentShadowContent, HitTestRequest::Type::CollectMultipleElements, HitTestRequest::Type::IncludeAllElementsUnderPoint };
399 HitTestResult result { absolutePoint.value() };
400 documentScope().hitTest(hitType, result);
401
402 RefPtr<Node> lastNode;
403 auto& nodeSet = result.listBasedTestResult();
404 elements.reserveInitialCapacity(nodeSet.size());
405 for (auto& listBasedNode : nodeSet) {
406 RefPtr<Node> node = retargetToScope(listBasedNode);
407 while (!is<Element>(node)) {
408 node = node->parentInComposedTree();
409 if (!node)
410 break;
411 node = retargetToScope(*node);
412 }
413
414 if (!node)
415 continue;
416
417 if (is<PseudoElement>(node))
418 node = downcast<PseudoElement>(*node).hostElement();
419
420 // Prune duplicate entries. A pseudo ::before content above its parent
421 // node should only result in one entry.
422 if (node == lastNode)
423 continue;
424
425 elements.append(static_pointer_cast<Element>(node));
426 lastNode = node;
427 }
428
429 if (m_rootNode.isDocumentNode()) {
430 if (Element* rootElement = downcast<Document>(m_rootNode).documentElement()) {
431 if (elements.isEmpty() || elements.last() != rootElement)
432 elements.append(rootElement);
433 }
434 }
435
436 return elements;
437}

444// FIXME: Would be nice to change this to take a StringView, since that's what callers have
445// and there is no particular advantage to already having a String.
446Element* TreeScope::findAnchor(StringView name)
447{
448 if (name.isEmpty())
449 return nullptr;
450 if (Element* element = getElementById(name))
451 return element;
452 for (auto& anchor : descendantsOfType<HTMLAnchorElement>(m_rootNode)) {
453 if (m_rootNode.document().inQuirksMode()) {
454 // Quirks mode, ASCII case-insensitive comparison of names.
455 // FIXME: This behavior is not mentioned in the HTML specification.
456 // We should either remove this or get this into the specification.
457 if (equalIgnoringASCIICase(anchor.name(), name))
458 return &anchor;
459 } else {
460 // Strict mode, names need to match exactly.
461 if (anchor.name() == name)
462 return &anchor;
463 }
464 }
465 return nullptr;
466}

504static void listTreeScopes(Node* node, Vector<TreeScope*, 5>& treeScopes)
505{
506 while (true) {
507 treeScopes.append(&node->treeScope());
508 Element* ancestor = node->shadowHost();
509 if (!ancestor)
510 break;
511 node = ancestor;
512 }
513}
514
515TreeScope* commonTreeScope(Node* nodeA, Node* nodeB)
516{
517 if (!nodeA || !nodeB)
518 return nullptr;
519
520 if (&nodeA->treeScope() == &nodeB->treeScope())
521 return &nodeA->treeScope();
522
523 Vector<TreeScope*, 5> treeScopesA;
524 listTreeScopes(nodeA, treeScopesA);
525
526 Vector<TreeScope*, 5> treeScopesB;
527 listTreeScopes(nodeB, treeScopesB);
528
529 size_t indexA = treeScopesA.size();
530 size_t indexB = treeScopesB.size();
531
532 for (; indexA > 0 && indexB > 0 && treeScopesA[indexA - 1] == treeScopesB[indexB - 1]; --indexA, --indexB) { }
533
534 // If the nodes had no common tree scope, return immediately.
535 if (indexA == treeScopesA.size())
536 return nullptr;
537
538 return treeScopesA[indexA] == treeScopesB[indexB] ? treeScopesA[indexA] : nullptr;
539}
