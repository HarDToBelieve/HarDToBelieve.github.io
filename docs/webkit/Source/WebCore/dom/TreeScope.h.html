145inline bool TreeScope::hasElementWithId(const AtomStringImpl& id) const
146{
147 return m_elementsById && m_elementsById->contains(id);
148}
149
150inline bool TreeScope::containsMultipleElementsWithId(const AtomString& id) const
151{
152 return m_elementsById && id.impl() && m_elementsById->containsMultiple(*id.impl());
153}
154
155inline bool TreeScope::hasElementWithName(const AtomStringImpl& id) const
156{
157 return m_elementsByName && m_elementsByName->contains(id);
158}
159
160inline bool TreeScope::containsMultipleElementsWithName(const AtomString& name) const
161{
162 return m_elementsByName && name.impl() && m_elementsByName->containsMultiple(*name.impl());
163}
