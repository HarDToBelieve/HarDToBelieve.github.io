139template<typename ElementType> ElementDescendantRange<ElementType>::ElementDescendantRange(const ContainerNode& root)

144template<typename ElementType> ElementDescendantIterator<ElementType> ElementDescendantRange<ElementType>::begin() const
145{
146 return ElementDescendantIterator<ElementType>(m_root, Traversal<ElementType>::firstWithin(m_root));
147}
148
149template<typename ElementType> ElementDescendantIterator<ElementType> ElementDescendantRange<ElementType>::beginAt(ElementType& descendant) const
150{
151 ASSERT(descendant.isDescendantOf(m_root));
152 return ElementDescendantIterator<ElementType>(m_root, &descendant);
153}
154
155template<typename ElementType> ElementDescendantIterator<ElementType> ElementDescendantRange<ElementType>::from(Element& descendant) const
156{
157 ASSERT(descendant.isDescendantOf(m_root));
158 if (is<ElementType>(descendant))
159 return ElementDescendantIterator<ElementType>(m_root, downcast<ElementType>(&descendant));
160 ElementType* next = Traversal<ElementType>::next(descendant, &m_root);
161 return ElementDescendantIterator<ElementType>(m_root, next);
162}
163
164template<typename ElementType> ElementType* ElementDescendantRange<ElementType>::first() const
165{
166 return Traversal<ElementType>::firstWithin(m_root);
167}
168
169template<typename ElementType> ElementType* ElementDescendantRange<ElementType>::last() const
170{
171 return Traversal<ElementType>::lastWithin(m_root);
172}

193template<typename ElementType> auto DoubleElementDescendantIterator<ElementType>::operator*() const -> ReferenceProxy

196}
197
198template<typename ElementType> constexpr bool DoubleElementDescendantIterator<ElementType>::operator!=(std::nullptr_t) const
199{
200 ASSERT(!m_pair.first == !m_pair.second);
201 return m_pair.first;
202}

204template<typename ElementType> DoubleElementDescendantIterator<ElementType>& DoubleElementDescendantIterator<ElementType>::operator++()
205{
206 ++m_pair.first;
207 ++m_pair.second;
208 return *this;

213template<typename ElementType, bool filter(const ElementType&)> FilteredElementDescendantIterator<ElementType, filter>::FilteredElementDescendantIterator(const ContainerNode& root, ElementType* element)

218template<typename ElementType, bool filter(const ElementType&)> FilteredElementDescendantIterator<ElementType, filter>& FilteredElementDescendantIterator<ElementType, filter>::operator++()
219{
220 do {
221 ElementIterator<ElementType>::traverseNext();
222 } while (*this && !filter(**this));
223 return *this;
224}

233template<typename ElementType, bool filter(const ElementType&)> auto FilteredElementDescendantRange<ElementType, filter>::begin() const -> Iterator
234{
235 return { m_root, first() };
236}

238template<typename ElementType, bool filter(const ElementType&)> ElementType* FilteredElementDescendantRange<ElementType, filter>::first() const
239{
240 for (auto* element = Traversal<ElementType>::firstWithin(m_root); element; element = Traversal<ElementType>::next(*element, &m_root)) {
241 if (filter(*element))
242 return element;
243 }
244 return nullptr;
245}
