39static RefPtr<UserGestureToken>& currentToken()
40{
41 ASSERT(isMainThread());
42 static NeverDestroyed<RefPtr<UserGestureToken>> token;
43 return token;
44}

46UserGestureToken::UserGestureToken(ProcessingUserGestureState state, UserGestureType gestureType, Document* document)
47 : m_state(state)
48 , m_gestureType(gestureType)
49{
50 if (!document || !processingUserGesture())
51 return;
52
53 // User gesture is valid for the document that received the user gesture, all of its ancestors
54 // as well as all same-origin documents on the page.
55 m_documentsImpactedByUserGesture.add(*document);
56
57 auto* documentFrame = document->frame();
58 if (!documentFrame)
59 return;
60
61 for (auto* ancestorFrame = documentFrame->tree().parent(); ancestorFrame; ancestorFrame = ancestorFrame->tree().parent()) {
62 if (auto* ancestorDocument = ancestorFrame->document())
63 m_documentsImpactedByUserGesture.add(ancestorDocument);
64 }
65
66 auto& documentOrigin = document->securityOrigin();
67 for (auto* frame = &documentFrame->tree().top(); frame; frame = frame->tree().traverseNext()) {
68 auto* frameDocument = frame->document();
69 if (frameDocument && documentOrigin.isSameOriginDomain(frameDocument->securityOrigin()))
70 m_documentsImpactedByUserGesture.add(*frameDocument);
71 }
72}

96UserGestureIndicator::UserGestureIndicator(std::optional<ProcessingUserGestureState> state, Document* document, UserGestureType gestureType, ProcessInteractionStyle processInteractionStyle)
97 : m_previousToken { currentToken() }
98{
99 ASSERT(isMainThread());
100
101 if (state)
102 currentToken() = UserGestureToken::create(state.value(), gestureType, document);
103
104 if (state && document && currentToken()->processingUserGesture()) {
105 document->updateLastHandledUserGestureTimestamp(currentToken()->startTime());
106 if (processInteractionStyle == ProcessInteractionStyle::Immediate)
107 ResourceLoadObserver::shared().logUserInteractionWithReducedTimeResolution(document->topDocument());
108 document->topDocument().setUserDidInteractWithPage(true);
109 if (auto* frame = document->frame()) {
110 if (!frame->hasHadUserInteraction()) {
111 for (; frame; frame = frame->tree().parent())
112 frame->setHasHadUserInteraction();
113 }
114 }
115
116 if (auto* window = document->domWindow())
117 window->notifyActivated(currentToken()->startTime());
118 }
119}

121UserGestureIndicator::UserGestureIndicator(RefPtr<UserGestureToken> token, UserGestureToken::GestureScope scope, UserGestureToken::IsPropagatedFromFetch isPropagatedFromFetch)
122{
123 // Silently ignore UserGestureIndicators on non main threads.
124 if (!isMainThread())
125 return;
126
127 // It is only safe to use currentToken() on the main thread.
128 m_previousToken = currentToken();
129
130 if (token) {
131 token->setScope(scope);
132 token->setIsPropagatedFromFetch(isPropagatedFromFetch);
133 currentToken() = token;
134 }
135}
136
137UserGestureIndicator::~UserGestureIndicator()
138{
139 if (!isMainThread())
140 return;
141
142 if (auto token = currentToken()) {
143 token->resetDOMPasteAccess();
144 token->resetScope();
145 token->resetIsPropagatedFromFetch();
146 }
147
148 currentToken() = m_previousToken;
149}

151RefPtr<UserGestureToken> UserGestureIndicator::currentUserGesture()
152{
153 if (!isMainThread())
154 return nullptr;
155
156 return currentToken();
157}
158
159bool UserGestureIndicator::processingUserGesture(const Document* document)
160{
161 if (!isMainThread())
162 return false;
163
164 if (!currentToken() || !currentToken()->processingUserGesture())
165 return false;
166
167 return !document || currentToken()->isValidForDocument(*document);
168}
169
170bool UserGestureIndicator::processingUserGestureForMedia()
171{
172 if (!isMainThread())
173 return false;
174
175 return currentToken() ? currentToken()->processingUserGestureForMedia() : false;
176}
