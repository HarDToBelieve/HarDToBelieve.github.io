30namespace WebCore {
31
32class WebKitTransitionEvent final : public Event {
33 WTF_MAKE_ISO_ALLOCATED(WebKitTransitionEvent);
34public:
35 static Ref<WebKitTransitionEvent> create(const AtomString& type, const String& propertyName, double elapsedTime, const String& pseudoElement)
36 {
37 return adoptRef(*new WebKitTransitionEvent(type, propertyName, elapsedTime, pseudoElement));
38 }
39
40 struct Init : EventInit {
41 String propertyName;
42 double elapsedTime { 0 };
43 String pseudoElement;
44 };
45
46 static Ref<WebKitTransitionEvent> create(const AtomString& type, const Init& initializer, IsTrusted isTrusted = IsTrusted::No)
47 {
48 return adoptRef(*new WebKitTransitionEvent(type, initializer, isTrusted));
49 }
50
51 virtual ~WebKitTransitionEvent();
52
53 const String& propertyName() const;
54 double elapsedTime() const;
55 const String& pseudoElement() const;
56
57 EventInterface eventInterface() const override;
58
59private:
60 WebKitTransitionEvent(const AtomString& type, const String& propertyName, double elapsedTime, const String& pseudoElement);
61 WebKitTransitionEvent(const AtomString& type, const Init& initializer, IsTrusted);
62
63 String m_propertyName;
64 double m_elapsedTime;
65 String m_pseudoElement;
66};
