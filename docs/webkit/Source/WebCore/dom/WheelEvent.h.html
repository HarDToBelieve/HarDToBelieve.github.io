25#pragma once
26
27#include "MouseEvent.h"
28#include "PlatformWheelEvent.h"
29
30namespace WebCore {
31
32class WheelEvent final : public MouseEvent {
33 WTF_MAKE_ISO_ALLOCATED(WheelEvent);
34public:
35 enum { TickMultiplier = 120 };
36
37 enum {
38 DOM_DELTA_PIXEL = 0,
39 DOM_DELTA_LINE,
40 DOM_DELTA_PAGE
41 };
42
43 static Ref<WheelEvent> create(const PlatformWheelEvent&, RefPtr<WindowProxy>&&, IsCancelable = IsCancelable::Yes);
44 static Ref<WheelEvent> createForBindings();
45
46 struct Init : MouseEventInit {
47 double deltaX { 0 };
48 double deltaY { 0 };
49 double deltaZ { 0 };
50 unsigned deltaMode { DOM_DELTA_PIXEL };
51 int wheelDeltaX { 0 }; // Deprecated.
52 int wheelDeltaY { 0 }; // Deprecated.
53 };
54
55 static Ref<WheelEvent> create(const AtomString& type, const Init&);
56
57 WEBCORE_EXPORT void initWebKitWheelEvent(int rawDeltaX, int rawDeltaY, RefPtr<WindowProxy>&&, int screenX, int screenY, int pageX, int pageY, bool ctrlKey, bool altKey, bool shiftKey, bool metaKey);
58
59 const std::optional<PlatformWheelEvent>& underlyingPlatformEvent() const { return m_underlyingPlatformEvent; }
60
61 double deltaX() const { return m_deltaX; } // Positive when scrolling right.
62 double deltaY() const { return m_deltaY; } // Positive when scrolling down.
63 double deltaZ() const { return m_deltaZ; }
64 int wheelDelta() const { return wheelDeltaY() ? wheelDeltaY() : wheelDeltaX(); } // Deprecated.
65 int wheelDeltaX() const { return m_wheelDelta.x(); } // Deprecated, negative when scrolling right.
66 int wheelDeltaY() const { return m_wheelDelta.y(); } // Deprecated, negative when scrolling down.
67 unsigned deltaMode() const { return m_deltaMode; }
68
69 bool webkitDirectionInvertedFromDevice() const { return m_underlyingPlatformEvent && m_underlyingPlatformEvent.value().directionInvertedFromDevice(); }
70
71#if PLATFORM(MAC)
72 PlatformWheelEventPhase phase() const { return m_underlyingPlatformEvent ? m_underlyingPlatformEvent.value().phase() : PlatformWheelEventPhase::None; }
73 PlatformWheelEventPhase momentumPhase() const { return m_underlyingPlatformEvent ? m_underlyingPlatformEvent.value().momentumPhase() : PlatformWheelEventPhase::None; }
74#endif
75
76private:
77 WheelEvent();
78 WheelEvent(const AtomString&, const Init&);
79 WheelEvent(const PlatformWheelEvent&, RefPtr<WindowProxy>&&, IsCancelable);
80
81 EventInterface eventInterface() const final;
82
83 bool isWheelEvent() const final;
84
85 IntPoint m_wheelDelta;
86 double m_deltaX { 0 };
87 double m_deltaY { 0 };
88 double m_deltaZ { 0 };
89 unsigned m_deltaMode { DOM_DELTA_PIXEL };
90 std::optional<PlatformWheelEvent> m_underlyingPlatformEvent;
91};
92
93} // namespace WebCore
94
95SPECIALIZE_TYPE_TRAITS_EVENT(WheelEvent)
