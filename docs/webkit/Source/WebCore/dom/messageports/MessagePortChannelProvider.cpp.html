35namespace WebCore {
36
37static MessagePortChannelProvider* globalProvider;
38
39MessagePortChannelProvider& MessagePortChannelProvider::singleton()
40{
41 ASSERT(isMainThread());
42 static std::once_flag onceFlag;
43 std::call_once(onceFlag, [] {
44 if (!globalProvider)
45 globalProvider = new MessagePortChannelProviderImpl;
46 });
47
48 return *globalProvider;
49}
50
51void MessagePortChannelProvider::setSharedProvider(MessagePortChannelProvider& provider)
52{
53 RELEASE_ASSERT(isMainThread());
54 RELEASE_ASSERT(!globalProvider);
55 globalProvider = &provider;
56}
57
58MessagePortChannelProvider& MessagePortChannelProvider::fromContext(ScriptExecutionContext& context)
59{
60 if (is<Document>(context))
61 return downcast<Document>(context).messagePortChannelProvider();
62
63 if (is<WorkletGlobalScope>(context))
64 return downcast<WorkletGlobalScope>(context).messagePortChannelProvider();
65
66 return downcast<WorkerGlobalScope>(context).messagePortChannelProvider();
67}
