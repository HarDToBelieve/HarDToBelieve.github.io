35static inline MessagePortChannelRegistry::CheckProcessLocalPortForActivityCallback checkActivityCallback()
36{
37 return [](auto& messagePortIdentifier, auto, auto&& callback) {
38 ASSERT(isMainThread());
39 callback(MessagePort::isExistingMessagePortLocallyReachable(messagePortIdentifier) ? MessagePortChannelProvider::HasActivity::Yes : MessagePortChannelProvider::HasActivity::No);
40 };
41}

43MessagePortChannelProviderImpl::MessagePortChannelProviderImpl()
44 : m_registry(checkActivityCallback())

46}

81void MessagePortChannelProviderImpl::postMessageToRemote(MessageWithMessagePorts&& message, const MessagePortIdentifier& remoteTarget)
82{
83 ensureOnMainThread([registry = &m_registry, message = WTFMove(message), remoteTarget]() mutable {
84 if (registry->didPostMessageToRemote(WTFMove(message), remoteTarget))
85 MessagePort::notifyMessageAvailable(remoteTarget);
86 });
87}
88
89void MessagePortChannelProviderImpl::takeAllMessagesForPort(const MessagePortIdentifier& port, CompletionHandler<void(Vector<MessageWithMessagePorts>&&, Function<void()>&&)>&& outerCallback)
90{
91 // It is the responsibility of outerCallback to get itself to the appropriate thread (e.g. WebWorker thread)
92 auto callback = [outerCallback = WTFMove(outerCallback)](Vector<MessageWithMessagePorts>&& messages, Function<void()>&& messageDeliveryCallback) mutable {
93 ASSERT(isMainThread());
94 outerCallback(WTFMove(messages), WTFMove(messageDeliveryCallback));
95 };
96
97 ensureOnMainThread([registry = &m_registry, port, callback = WTFMove(callback)]() mutable {
98 registry->takeAllMessagesForPort(port, WTFMove(callback));
99 });
100}
101
102void MessagePortChannelProviderImpl::checkRemotePortForActivity(const MessagePortIdentifier& remoteTarget, CompletionHandler<void(HasActivity)>&& outerCallback)
103{
104 auto callback = Function<void(HasActivity)> { [outerCallback = WTFMove(outerCallback)](HasActivity hasActivity) mutable {
105 ASSERT(isMainThread());
106 outerCallback(hasActivity);
107 } };
108
109 ensureOnMainThread([registry = &m_registry, remoteTarget, callback = WTFMove(callback)]() mutable {
110 registry->checkRemotePortForActivity(remoteTarget, WTFMove(callback));
111 });
112}
