26#pragma once
27
28#include "MessagePortIdentifier.h"
29#include "SerializedScriptValue.h"
30#include <wtf/RefPtr.h>
31
32namespace WebCore {
33
34// When a message port is transferred, it is represented by a pair of identifiers.
35// The first identifier is the port being transferred and the second is its remote port.
36using TransferredMessagePort = std::pair<WebCore::MessagePortIdentifier, WebCore::MessagePortIdentifier>;
37using TransferredMessagePortArray = Vector<TransferredMessagePort>;
38
39struct MessageWithMessagePorts {
40 RefPtr<SerializedScriptValue> message;
41 TransferredMessagePortArray transferredPorts;
42
43 template<class Encoder> void encode(Encoder&) const;
44 template<class Decoder> static std::optional<MessageWithMessagePorts> decode(Decoder&);
45};
46
47
48template<class Encoder>
49void MessageWithMessagePorts::encode(Encoder& encoder) const
50{
51 ASSERT(message);
52 encoder << *message << transferredPorts;
53}
54
55template<class Decoder>
56std::optional<MessageWithMessagePorts> MessageWithMessagePorts::decode(Decoder& decoder)
57{
58 MessageWithMessagePorts result;
59
60 result.message = SerializedScriptValue::decode(decoder);
61 if (!result.message)
62 return std::nullopt;
63
64 if (!decoder.decode(result.transferredPorts))
65 return std::nullopt;
66
67 return result;
68}
