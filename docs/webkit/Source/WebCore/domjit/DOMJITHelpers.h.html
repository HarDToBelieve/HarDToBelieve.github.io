63template<typename WrappedType>
64void tryLookUpWrapperCache(CCallHelpers& jit, CCallHelpers::JumpList& failureCases, GPRReg wrapped, GPRReg resultGPR)
65{
66 jit.loadPtr(CCallHelpers::Address(wrapped, ScriptWrappable::offsetOfWrapper<WrappedType>()), resultGPR);
67 failureCases.append(jit.branchTestPtr(CCallHelpers::Zero, resultGPR));
68 failureCases.append(branchIfNotWeakIsLive(jit, resultGPR));
69 jit.loadPtr(CCallHelpers::Address(resultGPR, JSC::WeakImpl::offsetOfJSValue() + JSC::JSValue::offsetOfPayload()), resultGPR);
70}

72template<typename WrappedType, typename ToJSFunction>
73void toWrapper(CCallHelpers& jit, JSC::SnippetParams& params, GPRReg wrapped, GPRReg globalObject, JSValueRegs result, ToJSFunction function, JSC::JSValue globalObjectConstant)
74{
75 ASSERT(wrapped != result.payloadGPR());
76 ASSERT(globalObject != result.payloadGPR());
77 GPRReg payloadGPR = result.payloadGPR();
78 CCallHelpers::JumpList slowCases;
79
80 if (globalObjectConstant) {
81 if (!JSC::jsCast<JSDOMGlobalObject*>(globalObjectConstant)->worldIsNormal()) {
82 slowCases.append(jit.jump());
83 params.addSlowPathCall(slowCases, jit, function, result, globalObject, wrapped);
84 return;
85 }
86 } else
87 slowCases.append(branchIfNotWorldIsNormal(jit, globalObject));
88
89 tryLookUpWrapperCache<WrappedType>(jit, slowCases, wrapped, payloadGPR);
90 jit.boxCell(payloadGPR, result);
91 params.addSlowPathCall(slowCases, jit, function, result, globalObject, wrapped);
92}

94inline CCallHelpers::Jump branchIfDOMWrapper(CCallHelpers& jit, GPRReg target)
95{
96 return jit.branch8(
97 CCallHelpers::AboveOrEqual,
98 CCallHelpers::Address(target, JSC::JSCell::typeInfoTypeOffset()),
99 CCallHelpers::TrustedImm32(JSC::JSType(JSDOMWrapperType)));
100}

103{
104 return jit.branch8(
105 CCallHelpers::Below,
106 CCallHelpers::Address(target, JSC::JSCell::typeInfoTypeOffset()),
107 CCallHelpers::TrustedImm32(JSC::JSType(JSDOMWrapperType)));
108}
