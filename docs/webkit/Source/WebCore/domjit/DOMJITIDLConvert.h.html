26#pragma once
27
28#include "IDLTypes.h"
29
30namespace WebCore { namespace DOMJIT {
31
32template<typename IDLType>
33struct DirectConverter;
34
35template<>
36struct DirectConverter<IDLDOMString> {
37 static String directConvert(JSC::JSGlobalObject& lexicalGlobalObject, JSC::JSString* string)
38 {
39 return string->value(&lexicalGlobalObject);
40 }
41};
42
43template<>
44struct DirectConverter<IDLAtomStringAdaptor<IDLDOMString>> {
45 static String directConvert(JSC::JSGlobalObject& lexicalGlobalObject, JSC::JSString* string)
46 {
47 return string->toAtomString(&lexicalGlobalObject);
48 }
49};
50
51template<>
52struct DirectConverter<IDLRequiresExistingAtomStringAdaptor<IDLDOMString>> {
53 static String directConvert(JSC::JSGlobalObject& lexicalGlobalObject, JSC::JSString* string)
54 {
55 return string->toExistingAtomString(&lexicalGlobalObject);
56 }
57};
58
59} }
