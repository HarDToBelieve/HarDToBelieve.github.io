74static bool markersHaveIdenticalDescription(const Vector<RenderedDocumentMarker*>& markers)
75{
76 if (markers.isEmpty())
77 return true;
78
79 const String& description = markers[0]->description();
80 for (size_t i = 1; i < markers.size(); ++i) {
81 if (description != markers[i]->description())
82 return false;
83 }
84 return true;
85}

98void AlternativeTextController::startAlternativeTextUITimer(AlternativeTextType type)
99{
100 const Seconds correctionPanelTimerInterval { 300_ms };
101 if (!isAutomaticSpellingCorrectionEnabled())
102 return;
103
104 // If type is PanelTypeReversion, then the new range has been set. So we shouldn't clear it.
105 if (type == AlternativeTextTypeCorrection)
106 m_rangeWithAlternative = std::nullopt;
107 m_type = type;
108 m_timer.startOneShot(correctionPanelTimerInterval);
109}

113 m_timer.stop();
114 m_rangeWithAlternative = std::nullopt;

128void AlternativeTextController::applyPendingCorrection(const VisibleSelection& selectionAfterTyping)
129{
130 // Apply pending autocorrection before next round of spell checking.
131 bool doApplyCorrection = true;
132 VisiblePosition startOfSelection = selectionAfterTyping.visibleStart();
133 VisibleSelection currentWord = VisibleSelection(startOfWord(startOfSelection, LeftWordIfOnBoundary), endOfWord(startOfSelection, RightWordIfOnBoundary));
134 if (currentWord.visibleEnd() == startOfSelection) {
135 if (auto wordRange = currentWord.firstRange()) {
136 String wordText = plainText(*wordRange);
137 if (!wordText.isEmpty() && isAmbiguousBoundaryCharacter(wordText[wordText.length() - 1]))
138 doApplyCorrection = false;
139 }
140 }
141 if (doApplyCorrection)
142 handleAlternativeTextUIResult(dismissSoon(ReasonForDismissingAlternativeTextAccepted));
143 else
144 m_rangeWithAlternative = std::nullopt;
145}

199bool AlternativeTextController::applyAutocorrectionBeforeTypingIfAppropriate()
200{
201 if (!m_rangeWithAlternative || !m_isActive)
202 return false;
203
204 if (m_type != AlternativeTextTypeCorrection)
205 return false;
206
207 Position caretPosition = m_document.selection().selection().start();
208
209 if (makeDeprecatedLegacyPosition(m_rangeWithAlternative->end) == caretPosition) {
210 handleAlternativeTextUIResult(dismissSoon(ReasonForDismissingAlternativeTextAccepted));
211 return true;
212 }
213
214 // Pending correction should always be where caret is. But in case this is not always true, we still want to dismiss the panel without accepting the correction.
215 ASSERT(makeDeprecatedLegacyPosition(m_rangeWithAlternative->end) == caretPosition);
216 dismiss(ReasonForDismissingAlternativeTextIgnored);
217 return false;
218}

220void AlternativeTextController::respondToUnappliedSpellCorrection(const VisibleSelection& selectionOfCorrected, const String& corrected, const String& correction)
221{
222 if (auto client = alternativeTextClient())
223 client->recordAutocorrectionResponse(AutocorrectionResponse::Reverted, corrected, correction);
224
225 RefPtr<Frame> protector(m_document.frame());
226 m_document.updateLayout();
227
228 m_document.selection().setSelection(selectionOfCorrected, FrameSelection::defaultSetSelectionOptions() | FrameSelection::SpellCorrectionTriggered);
229 auto range = m_document.selection().selection().firstRange();
230 if (!range)
231 return;
232 removeMarkers(*range, OptionSet<DocumentMarker::MarkerType> { DocumentMarker::Spelling, DocumentMarker::Autocorrected }, RemovePartiallyOverlappingMarker::Yes);
233 addMarker(*range, DocumentMarker::Replacement);
234 addMarker(*range, DocumentMarker::SpellCheckingExemption);
235}

237void AlternativeTextController::timerFired()
238{
239 m_isDismissedByEditing = false;
240 switch (m_type) {
241 case AlternativeTextTypeCorrection: {
242 VisibleSelection selection(m_document.selection().selection());
243 VisiblePosition start(selection.start(), selection.affinity());
244 VisiblePosition p = startOfWord(start, LeftWordIfOnBoundary);
245 VisibleSelection adjacentWords = VisibleSelection(p, start);
246 auto adjacentWordRange = adjacentWords.toNormalizedRange();
247 m_document.editor().markAllMisspellingsAndBadGrammarInRanges({ TextCheckingType::Spelling, TextCheckingType::Replacement, TextCheckingType::ShowCorrectionPanel }, adjacentWordRange, adjacentWordRange, std::nullopt);
248 }
249 break;
250 case AlternativeTextTypeReversion: {
251 if (!m_rangeWithAlternative)
252 break;
253 String replacementString = WTF::get<AutocorrectionReplacement>(m_details);
254 if (replacementString.isEmpty())
255 break;
256 m_isActive = true;
257 m_originalText = plainText(*m_rangeWithAlternative);
258 auto boundingBox = rootViewRectForRange(*m_rangeWithAlternative);
259 if (!boundingBox.isEmpty()) {
260 if (AlternativeTextClient* client = alternativeTextClient())
261 client->showCorrectionAlternative(m_type, boundingBox, m_originalText, replacementString, { });
262 }
263 }
264 break;
265 case AlternativeTextTypeSpellingSuggestions: {
266 if (!m_rangeWithAlternative || plainText(*m_rangeWithAlternative) != m_originalText)
267 break;
268 String paragraphText = plainText(TextCheckingParagraph(*m_rangeWithAlternative).paragraphRange());
269 Vector<String> suggestions;
270 textChecker()->getGuessesForWord(m_originalText, paragraphText, m_document.selection().selection(), suggestions);
271 if (suggestions.isEmpty()) {
272 m_rangeWithAlternative = std::nullopt;
273 break;
274 }
275 String topSuggestion = suggestions.first();
276 suggestions.remove(0);
277 m_isActive = true;
278 auto boundingBox = rootViewRectForRange(*m_rangeWithAlternative);
279 if (!boundingBox.isEmpty()) {
280 if (AlternativeTextClient* client = alternativeTextClient())
281 client->showCorrectionAlternative(m_type, boundingBox, m_originalText, topSuggestion, suggestions);
282 }
283 }
284 break;
285 case AlternativeTextTypeDictationAlternatives:
286 {
287#if USE(DICTATION_ALTERNATIVES)
288 if (!m_rangeWithAlternative)
289 return;
290 auto dictationContext = WTF::get<DictationContext>(m_details);
291 if (!dictationContext)
292 return;
293 auto boundingBox = rootViewRectForRange(*m_rangeWithAlternative);
294 m_isActive = true;
295 if (!boundingBox.isEmpty()) {
296 if (auto client = alternativeTextClient())
297 client->showDictationAlternativeUI(boundingBox, dictationContext);
298 }
299#endif
300 }
301 break;
302 }
303}

305void AlternativeTextController::handleAlternativeTextUIResult(const String& result)
306{
307 if (!m_rangeWithAlternative || &m_document != &m_rangeWithAlternative->start.document())
308 return;
309
310 String currentWord = plainText(*m_rangeWithAlternative);
311 // Check to see if the word we are about to correct has been changed between timer firing and callback being triggered.
312 if (currentWord != m_originalText)
313 return;
314
315 m_isActive = false;
316
317 switch (m_type) {
318 case AlternativeTextTypeCorrection:
319 if (result.length())
320 applyAlternativeTextToRange(*m_rangeWithAlternative, result, m_type, markerTypesForAutocorrection());
321 else if (!m_isDismissedByEditing)
322 addMarker(*m_rangeWithAlternative, DocumentMarker::RejectedCorrection, m_originalText);
323 break;
324 case AlternativeTextTypeReversion:
325 case AlternativeTextTypeSpellingSuggestions:
326 if (result.length())
327 applyAlternativeTextToRange(*m_rangeWithAlternative, result, m_type, markerTypesForReplacement());
328 break;
329 case AlternativeTextTypeDictationAlternatives:
330 if (result.length())
331 applyAlternativeTextToRange(*m_rangeWithAlternative, result, m_type, markerTypesForAppliedDictationAlternative());
332 break;
333 }
334
335 m_rangeWithAlternative = std::nullopt;
336}

372void AlternativeTextController::respondToChangedSelection(const VisibleSelection& oldSelection)
373{
374 VisibleSelection currentSelection(m_document.selection().selection());
375 // When user moves caret to the end of autocorrected word and pauses, we show the panel
376 // containing the original pre-correction word so that user can quickly revert the
377 // undesired autocorrection. Here, we start correction panel timer once we confirm that
378 // the new caret position is at the end of a word.
379 if (!currentSelection.isCaret() || currentSelection == oldSelection || !currentSelection.isContentEditable())
380 return;
381
382 VisiblePosition selectionPosition = currentSelection.start();
383
384 // Creating a Visible position triggers a layout and there is no
385 // guarantee that the selection is still valid.
386 if (selectionPosition.isNull())
387 return;
388
389 VisiblePosition endPositionOfWord = endOfWord(selectionPosition, LeftWordIfOnBoundary);
390 if (selectionPosition != endPositionOfWord)
391 return;
392
393 Position position = endPositionOfWord.deepEquivalent();
394 if (position.anchorType() != Position::PositionIsOffsetInAnchor)
395 return;
396
397 Node* node = position.containerNode();
398 ASSERT(node);
399 for (auto* marker : node->document().markers().markersFor(*node)) {
400 ASSERT(marker);
401 if (respondToMarkerAtEndOfWord(*marker, position))
402 break;
403 }
404}

462void AlternativeTextController::recordSpellcheckerResponseForModifiedCorrection(const SimpleRange& rangeOfCorrection, const String& corrected, const String& correction)
463{
464 DocumentMarkerController& markers = rangeOfCorrection.startContainer().document().markers();
465 Vector<RenderedDocumentMarker*> correctedOnceMarkers = markers.markersInRange(rangeOfCorrection, DocumentMarker::Autocorrected);
466 if (correctedOnceMarkers.isEmpty())
467 return;
468
469 if (AlternativeTextClient* client = alternativeTextClient()) {
470 // Spelling corrected text has been edited. We need to determine whether user has reverted it to original text or
471 // edited it to something else, and notify spellchecker accordingly.
472 if (markersHaveIdenticalDescription(correctedOnceMarkers) && correctedOnceMarkers[0]->description() == corrected)
473 client->recordAutocorrectionResponse(AutocorrectionResponse::Reverted, corrected, correction);
474 else
475 client->recordAutocorrectionResponse(AutocorrectionResponse::Edited, corrected, correction);
476 }
477
478 removeMarkers(rangeOfCorrection, DocumentMarker::Autocorrected, RemovePartiallyOverlappingMarker::Yes);
479}

487void AlternativeTextController::markPrecedingWhitespaceForDeletedAutocorrectionAfterCommand(EditCommand* command)
488{
489 Position endOfSelection = command->endingSelection().end();
490 if (endOfSelection != m_positionForLastDeletedAutocorrection)
491 return;
492
493 Position precedingCharacterPosition = endOfSelection.previous();
494 if (endOfSelection == precedingCharacterPosition)
495 return;
496
497 auto precedingCharacterRange = makeSimpleRange(precedingCharacterPosition, endOfSelection);
498 if (!precedingCharacterRange)
499 return;
500 String string = plainText(*precedingCharacterRange);
501 if (string.isEmpty() || !deprecatedIsEditingWhitespace(string[string.length() - 1]))
502 return;
503
504 // Mark this whitespace to indicate we have deleted an autocorrection following this
505 // whitespace. So if the user types the same original word again at this position, we
506 // won't autocorrect it again.
507 addMarker(*precedingCharacterRange, DocumentMarker::DeletedAutocorrection, m_originalStringForLastDeletedAutocorrection);
508}

510bool AlternativeTextController::processMarkersOnTextToBeReplacedByResult(const TextCheckingResult& result, const SimpleRange& rangeWithAlternative, const String& stringToBeReplaced)
511{
512 auto& markers = m_document.markers();
513 if (markers.hasMarkers(rangeWithAlternative, DocumentMarker::Replacement)) {
514 if (result.type == TextCheckingType::Correction)
515 recordSpellcheckerResponseForModifiedCorrection(rangeWithAlternative, stringToBeReplaced, result.replacement);
516 return false;
517 }
518
519 if (markers.hasMarkers(rangeWithAlternative, DocumentMarker::RejectedCorrection))
520 return false;
521
522 if (markers.hasMarkers(rangeWithAlternative, DocumentMarker::AcceptedCandidate))
523 return false;
524
525 auto precedingCharacterRange = makeSimpleRange(makeDeprecatedLegacyPosition(rangeWithAlternative.start).previous(), rangeWithAlternative.start);
526 if (!precedingCharacterRange)
527 return false;
528
529 for (auto& marker : markers.markersInRange(*precedingCharacterRange, DocumentMarker::DeletedAutocorrection)) {
530 if (marker->description() == stringToBeReplaced)
531 return false;
532 }
533
534 return true;
535}

537bool AlternativeTextController::shouldStartTimerFor(const WebCore::DocumentMarker &marker, int endOffset) const
538{
539 return (((marker.type() == DocumentMarker::Replacement && !marker.description().isNull()) || marker.type() == DocumentMarker::Spelling || marker.type() == DocumentMarker::DictationAlternatives) && static_cast<int>(marker.endOffset()) == endOffset);

542bool AlternativeTextController::respondToMarkerAtEndOfWord(const DocumentMarker& marker, const Position& endOfWordPosition)
543{
544 if (!shouldStartTimerFor(marker, endOfWordPosition.offsetInContainerNode()))
545 return false;
546 Node* node = endOfWordPosition.containerNode();
547 auto wordRange = makeSimpleRange(*node, marker);
548 String currentWord = plainText(wordRange);
549 if (!currentWord.length())
550 return false;
551 m_originalText = currentWord;
552 switch (marker.type()) {
553 case DocumentMarker::Spelling:
554 m_rangeWithAlternative = WTFMove(wordRange);
555 m_details = emptyString();
556 startAlternativeTextUITimer(AlternativeTextTypeSpellingSuggestions);
557 break;
558 case DocumentMarker::Replacement:
559 m_rangeWithAlternative = WTFMove(wordRange);
560 m_details = marker.description();
561 startAlternativeTextUITimer(AlternativeTextTypeReversion);
562 break;
563 case DocumentMarker::DictationAlternatives: {
564 auto& markerData = WTF::get<DocumentMarker::DictationData>(marker.data());
565 if (currentWord != markerData.originalText)
566 return false;
567 m_rangeWithAlternative = WTFMove(wordRange);
568 m_details = markerData.context;
569 startAlternativeTextUITimer(AlternativeTextTypeDictationAlternatives);
570 }
571 break;
572 default:
573 ASSERT_NOT_REACHED();
574 break;
575 }
576 return true;
577}

600void AlternativeTextController::applyAlternativeTextToRange(const SimpleRange& range, const String& alternative, AlternativeTextType alternativeType, OptionSet<DocumentMarker::MarkerType> markerTypesToAdd)
601{
602 // After we replace the word at range rangeWithAlternative, we need to add markers to that range.
603 // So before we carry out the replacement,store the start position relative to the start position
604 // of the containing paragraph.
605
606 // Take note of the location of autocorrection so that we can add marker after the replacement took place.
607 auto paragraphStart = makeBoundaryPoint(startOfParagraph(makeDeprecatedLegacyPosition(range.start)));
608 if (!paragraphStart)
609 return;
610 auto treeScopeRoot = makeRef(range.start.container->treeScope().rootNode());
611 auto treeScopeStart = BoundaryPoint { treeScopeRoot.get(), 0 };
612 auto correctionOffsetInParagraph = characterCount({ *paragraphStart, range.start });
613 auto paragraphOffsetInTreeScope = characterCount({ treeScopeStart, *paragraphStart });
614
615 SpellingCorrectionCommand::create(range, alternative)->apply();
616
617 // Recalculate pragraphRangeContainingCorrection, since SpellingCorrectionCommand modified the DOM, such that the original paragraphRangeContainingCorrection is no longer valid. Radar: 10305315 Bugzilla: 89526
618 auto updatedParagraphStartContainingCorrection = resolveCharacterLocation(makeRangeSelectingNodeContents(treeScopeRoot), paragraphOffsetInTreeScope);
619 auto updatedParagraphEndContainingCorrection = makeBoundaryPoint(m_document.selection().selection().start());
620 if (!updatedParagraphEndContainingCorrection)
621 return;
622 auto replacementRange = resolveCharacterRange({ updatedParagraphStartContainingCorrection, *updatedParagraphEndContainingCorrection }, CharacterRange(correctionOffsetInParagraph, alternative.length()));
623
624 // Check to see if replacement succeeded.
625 if (plainText(replacementRange) != alternative)
626 return;
627
628 for (auto markerType : markerTypesToAdd)
629 addMarker(replacementRange, markerType, markerDescriptionForAppliedAlternativeText(alternativeType, markerType));
630}

673void AlternativeTextController::applyDictationAlternative(const String& alternativeString)
