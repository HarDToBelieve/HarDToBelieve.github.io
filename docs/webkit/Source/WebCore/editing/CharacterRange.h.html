26#pragma once
27
28#if USE(CF)
29#include <CoreFoundation/CoreFoundation.h>
30#endif
31
32#if USE(FOUNDATION)
33typedef struct _NSRange NSRange;
34#endif
35
36namespace WebCore {
37
38struct CharacterRange {
39 uint64_t location { 0 };
40 uint64_t length { 0 };
41
42 CharacterRange() = default;
43 constexpr CharacterRange(uint64_t location, uint64_t length);
44
45#if USE(CF)
46 constexpr CharacterRange(CFRange);
47 constexpr operator CFRange() const;
48#endif
49
50#if USE(FOUNDATION)
51 constexpr CharacterRange(NSRange);
52 constexpr operator NSRange() const;
53#endif
54};
55
56constexpr CharacterRange::CharacterRange(uint64_t location, uint64_t length)
57 : location(location)
58 , length(length)
59{
60}
61
62#if USE(CF)
63
64constexpr CharacterRange::CharacterRange(CFRange range)
65 : CharacterRange(range.location, range.length)
66{
67 ASSERT(range.location != kCFNotFound);
68}
69
70constexpr CharacterRange::operator CFRange() const
71{
72 CFIndex locationCF = location;
73 CFIndex lengthCF = length;
74 return { locationCF, lengthCF };
75}
76
77#endif
78
79#if USE(FOUNDATION) && defined(__OBJC__)
80
81constexpr CharacterRange::CharacterRange(NSRange range)
82 : CharacterRange { range.location, range.length }
83{
84 ASSERT(range.location != NSNotFound);
85}
86
87constexpr CharacterRange::operator NSRange() const
88{
89 NSUInteger locationNS = location;
90 NSUInteger lengthNS = length;
91 return { locationNS, lengthNS };
92}
93
94#endif
95
96} // namespace WebCore
