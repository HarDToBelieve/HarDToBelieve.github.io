36DeleteFromTextNodeCommand::DeleteFromTextNodeCommand(Ref<Text>&& node, unsigned offset, unsigned count, EditAction editingAction)
37 : SimpleEditCommand(node->document(), editingAction)
38 , m_node(WTFMove(node))
39 , m_offset(offset)
40 , m_count(count)
41{
42 ASSERT(m_offset <= m_node->length());
43 ASSERT(m_offset + m_count <= m_node->length());
44}
45
46void DeleteFromTextNodeCommand::doApply()
47{
48 if (!isEditableNode(m_node))
49 return;
50
51 auto result = m_node->substringData(m_offset, m_count);
52 if (result.hasException())
53 return;
54 m_text = result.releaseReturnValue();
55 m_node->deleteData(m_offset, m_count);
56}
57
58void DeleteFromTextNodeCommand::doUnapply()
59{
60 if (!m_node->hasEditableStyle())
61 return;
62
63 m_node->insertData(m_offset, m_text);
64}

67void DeleteFromTextNodeCommand::getNodesInCommand(HashSet<Ref<Node>>& nodes)
68{
69 addNodeAndDescendants(m_node.ptr(), nodes);
70}
