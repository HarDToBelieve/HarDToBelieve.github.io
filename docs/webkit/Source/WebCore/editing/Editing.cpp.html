82ContainerNode* highestEditableRoot(const Position& position, EditableType editableType)
83{
84 ContainerNode* highestEditableRoot = editableRootForPosition(position, editableType);
85 if (!highestEditableRoot)
86 return nullptr;
87
88 for (ContainerNode* node = highestEditableRoot; !is<HTMLBodyElement>(*node); ) {
89 node = node->parentNode();
90 if (!node)
91 break;
92 // FIXME: Can this ever be a Document or DocumentFragment? If not, this should return Element* instead.
93 if (hasEditableStyle(*node, editableType))
94 highestEditableRoot = node;
95 }
96
97 return highestEditableRoot;
98}

122static bool computeEditability(const Node& node, EditableType editableType, Node::ShouldUpdateStyle shouldUpdateStyle)
123{
124 if (node.computeEditability(Node::UserSelectAllIsAlwaysNonEditable, shouldUpdateStyle) != Node::Editability::ReadOnly)
125 return true;
126
127 switch (editableType) {
128 case ContentIsEditable:
129 return false;
130 case HasEditableAXRole:
131 return isEditableToAccessibility(node);
132 }
133 ASSERT_NOT_REACHED();
134 return false;
135}

165Element* editableRootForPosition(const Position& position, EditableType editableType)
166{
167 Node* node = position.containerNode();
168 if (!node)
169 return nullptr;
170
171 switch (editableType) {
172 case HasEditableAXRole:
173 if (auto* cache = node->document().existingAXObjectCache())
174 return const_cast<Element*>(cache->rootAXEditableElement(node));
175 FALLTHROUGH;
176 case ContentIsEditable:
177 return node->rootEditableElement();
178 }
179 return nullptr;
180}

182// Finds the enclosing element until which the tree can be split.
183// When a user hits ENTER, he/she won't expect this element to be split into two.
184// You may pass it as the second argument of splitTreeToNode.
185Element* unsplittableElementForPosition(const Position& position)
186{
187 // Since enclosingNodeOfType won't search beyond the highest root editable node,
188 // this code works even if the closest table cell was outside of the root editable node.
189 if (auto* enclosingCell = downcast<Element>(enclosingNodeOfType(position, &isTableCell)))
190 return enclosingCell;
191 return editableRootForPosition(position);
192}

194Position nextCandidate(const Position& position)
195{
196 for (PositionIterator nextPosition = position; !nextPosition.atEnd(); ) {
197 nextPosition.increment();
198 if (nextPosition.isCandidate())
199 return nextPosition;
200 }
201 return { };
202}
203
204Position nextVisuallyDistinctCandidate(const Position& position)
205{
206 // FIXME: Use PositionIterator instead.
207 Position nextPosition = position;
208 Position downstreamStart = nextPosition.downstream();
209 while (!nextPosition.atEndOfTree()) {
210 nextPosition = nextPosition.next(Character);
211 if (nextPosition.isCandidate() && nextPosition.downstream() != downstreamStart)
212 return nextPosition;
213 if (auto* node = nextPosition.containerNode()) {
214 if (!node->renderer())
215 nextPosition = lastPositionInOrAfterNode(node);
216 }
217 }
218 return { };
219}
220
221Position previousCandidate(const Position& position)
222{
223 PositionIterator previousPosition = position;
224 while (!previousPosition.atStart()) {
225 previousPosition.decrement();
226 if (previousPosition.isCandidate())
227 return previousPosition;
228 }
229 return { };
230}
231
232Position previousVisuallyDistinctCandidate(const Position& position)
233{
234 // FIXME: Use PositionIterator instead.
235 Position previousPosition = position;
236 Position downstreamStart = previousPosition.downstream();
237 while (!previousPosition.atStartOfTree()) {
238 previousPosition = previousPosition.previous(Character);
239 if (previousPosition.isCandidate() && previousPosition.downstream() != downstreamStart)
240 return previousPosition;
241 if (auto* node = previousPosition.containerNode()) {
242 if (!node->renderer())
243 previousPosition = firstPositionInOrBeforeNode(node);
244 }
245 }
246 return { };
247}

249Position firstEditablePositionAfterPositionInRoot(const Position& position, ContainerNode* highestRoot)
250{
251 if (!highestRoot)
252 return { };
253
254 // position falls before highestRoot.
255 if (position < firstPositionInNode(highestRoot) && highestRoot->hasEditableStyle())
256 return firstPositionInNode(highestRoot);
257
258 Position candidate = position;
259
260 if (&position.deprecatedNode()->treeScope() != &highestRoot->treeScope()) {
261 auto* shadowAncestor = highestRoot->treeScope().ancestorNodeInThisScope(position.deprecatedNode());
262 if (!shadowAncestor)
263 return { };
264
265 candidate = positionAfterNode(shadowAncestor);
266 }
267
268 while (candidate.deprecatedNode() && !isEditablePosition(candidate) && candidate.deprecatedNode()->isDescendantOf(*highestRoot))
269 candidate = isAtomicNode(candidate.deprecatedNode()) ? positionInParentAfterNode(candidate.deprecatedNode()) : nextVisuallyDistinctCandidate(candidate);
270
271 if (candidate.deprecatedNode() && candidate.deprecatedNode() != highestRoot && !candidate.deprecatedNode()->isDescendantOf(*highestRoot))
272 return { };
273
274 return candidate;
275}
276
277Position lastEditablePositionBeforePositionInRoot(const Position& position, ContainerNode* highestRoot)
278{
279 if (!highestRoot)
280 return { };
281
282 // When position falls after highestRoot, the result is easy to compute.
283 if (position > lastPositionInNode(highestRoot))
284 return lastPositionInNode(highestRoot);
285
286 Position candidate = position;
287
288 if (&position.deprecatedNode()->treeScope() != &highestRoot->treeScope()) {
289 auto* shadowAncestor = highestRoot->treeScope().ancestorNodeInThisScope(position.deprecatedNode());
290 if (!shadowAncestor)
291 return { };
292
293 candidate = firstPositionInOrBeforeNode(shadowAncestor);
294 }
295
296 while (candidate.deprecatedNode() && !isEditablePosition(candidate) && candidate.deprecatedNode()->isDescendantOf(*highestRoot))
297 candidate = isAtomicNode(candidate.deprecatedNode()) ? positionInParentBeforeNode(candidate.deprecatedNode()) : previousVisuallyDistinctCandidate(candidate);
298
299 if (candidate.deprecatedNode() && candidate.deprecatedNode() != highestRoot && !candidate.deprecatedNode()->isDescendantOf(*highestRoot))
300 return { };
301
302 return candidate;
303}

338// This method is used to create positions in the DOM. It returns the maximum valid offset
339// in a node. It returns 1 for some elements even though they do not have children, which
340// creates technically invalid DOM Positions. Be sure to call parentAnchoredEquivalent
341// on a Position before using it to create a DOM Range, or an exception will be thrown.
342int lastOffsetForEditing(const Node& node)
343{
344 if (node.isCharacterDataNode() || node.hasChildNodes())
345 return node.length();
346
347 // FIXME: Might be more helpful to return 1 for any node where editingIgnoresContent is true, even one that happens to have child nodes, like a select element with option node children.
348 return editingIgnoresContent(node) ? 1 : 0;
349}

351bool isAmbiguousBoundaryCharacter(UChar character)
352{
353 // These are characters that can behave as word boundaries, but can appear within words.
354 // If they are just typed, i.e. if they are immediately followed by a caret, we want to delay text checking until the next character has been typed.
355 // FIXME: this is required until <rdar://problem/6853027> is fixed and text checking can do this for us.
356 return character == '\'' || character == '@' || character == rightSingleQuotationMark || character == hebrewPunctuationGershayim;
357}

359String stringWithRebalancedWhitespace(const String& string, bool startIsStartOfParagraph, bool endIsEndOfParagraph)
360{
361 StringBuilder rebalancedString;
362
363 bool previousCharacterWasSpace = false;
364 unsigned length = string.length();
365 for (unsigned i = 0; i < length; ++i) {
366 auto character = string[i];
367 if (!deprecatedIsEditingWhitespace(character)) {
368 previousCharacterWasSpace = false;
369 continue;
370 }
371 LChar selectedWhitespaceCharacter;
372 if (previousCharacterWasSpace || (!i && startIsStartOfParagraph) || (i == length - 1 && endIsEndOfParagraph)) {
373 selectedWhitespaceCharacter = noBreakSpace;
374 previousCharacterWasSpace = false;
375 } else {
376 selectedWhitespaceCharacter = ' ';
377 previousCharacterWasSpace = true;
378 }
379 if (character == selectedWhitespaceCharacter)
380 continue;
381 rebalancedString.reserveCapacity(length);
382 rebalancedString.appendSubstring(string, rebalancedString.length(), i - rebalancedString.length());
383 rebalancedString.append(selectedWhitespaceCharacter);
384 }
385
386 if (rebalancedString.isEmpty())
387 return string;
388
389 rebalancedString.reserveCapacity(length);
390 rebalancedString.appendSubstring(string, rebalancedString.length(), length - rebalancedString.length());
391 return rebalancedString.toString();
392}

450VisiblePosition closestEditablePositionInElementForAbsolutePoint(const Element& element, const IntPoint& point)
451{
452 if (!element.isConnected() || !element.document().frame())
453 return { };
454
455 Ref<const Element> protectedElement { element };
456 element.document().updateLayoutIgnorePendingStylesheets();
457
458 RenderObject* renderer = element.renderer();
459 // Look at the inner element of a form control, not the control itself, as it is the editable part.
460 if (is<HTMLTextFormControlElement>(element)) {
461 auto& formControlElement = downcast<HTMLTextFormControlElement>(element);
462 if (!formControlElement.isInnerTextElementEditable())
463 return { };
464 if (auto innerTextElement = formControlElement.innerTextElement())
465 renderer = innerTextElement->renderer();
466 }
467 if (!renderer)
468 return { };
469 auto absoluteBoundingBox = renderer->absoluteBoundingBoxRect();
470 auto constrainedAbsolutePoint = point.constrainedBetween(absoluteBoundingBox.minXMinYCorner(), absoluteBoundingBox.maxXMaxYCorner());
471 auto localPoint = renderer->absoluteToLocal(constrainedAbsolutePoint, UseTransforms);
472 auto visiblePosition = renderer->positionForPoint(flooredLayoutPoint(localPoint), nullptr);
473 return isEditablePosition(visiblePosition.deepEquivalent()) ? visiblePosition : VisiblePosition { };
474}

486Element* enclosingElementWithTag(const Position& position, const QualifiedName& tagName)
487{
488 auto* root = highestEditableRoot(position);
489 for (Node* node = position.deprecatedNode(); node; node = node->parentNode()) {
490 if (root && !node->hasEditableStyle())
491 continue;
492 if (!is<Element>(*node))
493 continue;
494 if (downcast<Element>(*node).hasTagName(tagName))
495 return &downcast<Element>(*node);
496 if (node == root)
497 return nullptr;
498 }
499 return nullptr;
500}

502Node* enclosingNodeOfType(const Position& position, bool (*nodeIsOfType)(const Node*), EditingBoundaryCrossingRule rule)
503{
504 // FIXME: support CanSkipCrossEditingBoundary
505 ASSERT(rule == CanCrossEditingBoundary || rule == CannotCrossEditingBoundary);
506 auto* root = rule == CannotCrossEditingBoundary ? highestEditableRoot(position) : nullptr;
507 for (Node* n = position.deprecatedNode(); n; n = n->parentNode()) {
508 // Don't return a non-editable node if the input position was editable, since
509 // the callers from editing will no doubt want to perform editing inside the returned node.
510 if (root && !n->hasEditableStyle())
511 continue;
512 if (nodeIsOfType(n))
513 return n;
514 if (n == root)
515 return nullptr;
516 }
517 return nullptr;
518}

520Node* highestEnclosingNodeOfType(const Position& position, bool (*nodeIsOfType)(const Node*), EditingBoundaryCrossingRule rule, Node* stayWithin)
521{
522 Node* highest = nullptr;
523 auto* root = rule == CannotCrossEditingBoundary ? highestEditableRoot(position) : nullptr;
524 for (Node* n = position.containerNode(); n && n != stayWithin; n = n->parentNode()) {
525 if (root && !n->hasEditableStyle())
526 continue;
527 if (nodeIsOfType(n))
528 highest = n;
529 if (n == root)
530 break;
531 }
532 return highest;
533}

535static bool hasARenderedDescendant(Node* node, Node* excludedNode)
536{
537 for (Node* n = node->firstChild(); n;) {
538 if (n == excludedNode) {
539 n = NodeTraversal::nextSkippingChildren(*n, node);
540 continue;
541 }
542 if (n->renderer())
543 return true;
544 n = NodeTraversal::next(*n, node);
545 }
546 return false;
547}

549Node* highestNodeToRemoveInPruning(Node* node)
550{
551 Node* previousNode = nullptr;
552 auto* rootEditableElement = node ? node->rootEditableElement() : nullptr;
553 for (auto currentNode = makeRefPtr(node); currentNode; currentNode = currentNode->parentNode()) {
554 if (auto* renderer = currentNode->renderer()) {
555 if (!renderer->canHaveChildren() || hasARenderedDescendant(currentNode.get(), previousNode) || rootEditableElement == currentNode.get())
556 return previousNode;
557 }
558 previousNode = currentNode.get();
559 }
560 return nullptr;
561}

578{
579 if (!node)
580 return nullptr;

583

594Node* enclosingListChild(Node *node)
595{
596 if (!node)
597 return nullptr;
598
599 // Check for a list item element, or for a node whose parent is a list element. Such a node
600 // will appear visually as a list item (but without a list marker)
601 auto* root = highestEditableRoot(firstPositionInOrBeforeNode(node));
602
603 // FIXME: This function is inappropriately named since it starts with node instead of node->parentNode()
604 for (Node* n = node; n && n->parentNode(); n = n->parentNode()) {
605 if (is<HTMLLIElement>(*n) || (isListHTMLElement(n->parentNode()) && n != root))
606 return n;
607 if (n == root || isTableCell(n))
608 return nullptr;
609 }
610
611 return nullptr;
612}

637// FIXME: This function should not need to call isStartOfParagraph/isEndOfParagraph.
638Node* enclosingEmptyListItem(const VisiblePosition& position)
639{
640 // Check that position is on a line by itself inside a list item
641 auto* listChildNode = enclosingListChild(position.deepEquivalent().deprecatedNode());
642 if (!listChildNode || !isStartOfParagraph(position) || !isEndOfParagraph(position))
643 return nullptr;

649 return nullptr;

652 return nullptr;
653
654 return listChildNode;

672bool canMergeLists(Element* firstList, Element* secondList)
673{
674 if (!is<HTMLElement>(firstList) || !is<HTMLElement>(secondList))
675 return false;
676
677 auto& first = downcast<HTMLElement>(*firstList);
678 auto& second = downcast<HTMLElement>(*secondList);
679
680 return first.localName() == second.localName() // make sure the list types match (ol vs. ul)
681 && first.hasEditableStyle() && second.hasEditableStyle() // both lists are editable
682 && first.rootEditableElement() == second.rootEditableElement() // don't cross editing boundaries
683 // Make sure there is no visible content between this li and the previous list.
684 && isVisiblyAdjacent(positionInParentAfterNode(&first), positionInParentBeforeNode(&second));
685}

687static Node* previousNodeConsideringAtomicNodes(const Node* node)
688{
689 if (node->previousSibling()) {
690 Node* n = node->previousSibling();

700static Node* nextNodeConsideringAtomicNodes(const Node* node)
701{
702 if (!isAtomicNode(node) && node->firstChild())
703 return node->firstChild();
704 if (node->nextSibling())
705 return node->nextSibling();
706 const Node* n = node;
707 while (n && !n->nextSibling())

749bool isEmptyTableCell(const Node* node)
750{
751 // Returns true IFF the passed in node is one of:
752 // .) a table cell with no children,
753 // .) a table cell with a single BR child, and which has no other child renderers, including :before and :after renderers
754 // .) the BR child of such a table cell
755
756 // Find rendered node
757 while (node && !node->renderer())
758 node = node->parentNode();
759 if (!node)
760 return false;
761
762 // Make sure the rendered node is a table cell or <br>.
763 // If it's a <br>, then the parent node has to be a table cell.
764 auto* renderer = node->renderer();
765 if (renderer->isBR()) {
766 renderer = renderer->parent();
767 if (!renderer)
768 return false;
769 }
770 if (!is<RenderTableCell>(*renderer))
771 return false;
772
773 // Check that the table cell contains no child renderers except for perhaps a single <br>.
774 auto* childRenderer = downcast<RenderTableCell>(*renderer).firstChild();
775 if (!childRenderer)
776 return true;
777 if (!childRenderer->isBR())
778 return false;
779 return !childRenderer->nextSibling();

937// Modifies selections that have an end point at the edge of a table
938// that contains the other endpoint so that they don't confuse
939// code that iterates over selected paragraphs.
940VisibleSelection selectionForParagraphIteration(const VisibleSelection& original)
941{
942 VisibleSelection newSelection(original);
943 VisiblePosition startOfSelection(newSelection.visibleStart());
944 VisiblePosition endOfSelection(newSelection.visibleEnd());
945
946 // If the end of the selection to modify is just after a table, and
947 // if the start of the selection is inside that table, then the last paragraph
948 // that we'll want modify is the last one inside the table, not the table itself
949 // (a table is itself a paragraph).
950 if (auto* table = isFirstPositionAfterTable(endOfSelection)) {
951 if (startOfSelection.deepEquivalent().deprecatedNode()->isDescendantOf(*table))
952 newSelection = VisibleSelection(startOfSelection, endOfSelection.previous(CannotCrossEditingBoundary));
953 }
954
955 // If the start of the selection to modify is just before a table,
956 // and if the end of the selection is inside that table, then the first paragraph
957 // we'll want to modify is the first one inside the table, not the paragraph
958 // containing the table itself.
959 if (auto* table = isLastPositionBeforeTable(startOfSelection)) {
960 if (endOfSelection.deepEquivalent().deprecatedNode()->isDescendantOf(*table))
961 newSelection = VisibleSelection(startOfSelection.next(CannotCrossEditingBoundary), endOfSelection);
962 }
963
964 return newSelection;
965}

967// FIXME: indexForVisiblePosition and visiblePositionForIndex use TextIterators to convert between
968// VisiblePositions and indices. But TextIterator iteration using TextIteratorBehavior::EmitsCharactersBetweenAllVisiblePositions
969// does not exactly match VisiblePosition iteration, so using them to preserve a selection during an editing
970// opertion is unreliable. TextIterator's TextIteratorBehavior::EmitsCharactersBetweenAllVisiblePositions mode needs to be fixed,
971// or these functions need to be changed to iterate using actual VisiblePositions.
972// FIXME: Deploy these functions everywhere that TextIterators are used to convert between VisiblePositions and indices.
973int indexForVisiblePosition(const VisiblePosition& visiblePosition, RefPtr<ContainerNode>& scope)
974{

1102Position adjustedSelectionStartForStyleComputation(const VisibleSelection& selection)
1103{
1104 // This function is used by range style computations to avoid bugs like:
1105 // <rdar://problem/4017641> REGRESSION (Mail): you can only bold/unbold a selection starting from end of line once
1106 // It is important to skip certain irrelevant content at the start of the selection, so we do not wind up
1107 // with a spurious "mixed" style.
1108
1109 auto visiblePosition = selection.visibleStart();
1110 if (visiblePosition.isNull())
1111 return { };
1112
1113 // if the selection is a caret, just return the position, since the style
1114 // behind us is relevant
1115 if (selection.isCaret())
1116 return visiblePosition.deepEquivalent();
1117
1118 // if the selection starts just before a paragraph break, skip over it
1119 if (isEndOfParagraph(visiblePosition))
1120 return visiblePosition.next().deepEquivalent().downstream();
1121
1122 // otherwise, make sure to be at the start of the first selected node,
1123 // instead of possibly at the end of the last node before the selection
1124 return visiblePosition.deepEquivalent().downstream();
