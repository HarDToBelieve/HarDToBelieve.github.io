27
28#include "ApplyBlockElementCommand.h"
29#include "EditAction.h"
30#include "QualifiedName.h"

32namespace WebCore {
33
34class Document;
35class Element;
36class Position;
37class Range;
38class VisiblePosition;
39
40class FormatBlockCommand : public ApplyBlockElementCommand {
41public:
42 static Ref<FormatBlockCommand> create(Document& document, const QualifiedName& tagName)
43 {
44 return adoptRef(*new FormatBlockCommand(document, tagName));
45 }
46
47 bool preservesTypingStyle() const override { return true; }
48
49 static Element* elementForFormatBlockCommand(const std::optional<SimpleRange>&);
50 bool didApply() const { return m_didApply; }
51
52private:
53 FormatBlockCommand(Document&, const QualifiedName& tagName);
54
55 void formatSelection(const VisiblePosition& startOfSelection, const VisiblePosition& endOfSelection) override;
56 void formatRange(const Position& start, const Position& end, const Position& endOfSelection, RefPtr<Element>&) override;
57 EditAction editingAction() const override { return EditAction::FormatBlock; }
58
59 bool m_didApply;
60};

62} // namespace WebCore
