37String convertHTMLTextToInterchangeFormat(const String& in, const Text* node)
38{
39 // Assume all the text comes from node.
40 if (node->renderer() && node->renderer()->style().preserveNewline())
41 return in;
42
43 const char convertedSpaceString[] = "<span class=\"" AppleConvertedSpace "\">\xA0</span>";
44 COMPILE_ASSERT((static_cast<unsigned char>('\xA0') == noBreakSpace), ConvertedSpaceStringSpaceIsNoBreakSpace);
45
46 StringBuilder s;
47
48 unsigned i = 0;
49 unsigned consumed = 0;
50 while (i < in.length()) {
51 consumed = 1;
52 if (deprecatedIsCollapsibleWhitespace(in[i])) {
53 // count number of adjoining spaces
54 unsigned j = i + 1;
55 while (j < in.length() && deprecatedIsCollapsibleWhitespace(in[j]))
56 j++;
57 unsigned count = j - i;
58 consumed = count;
59 while (count) {
60 unsigned add = count % 3;
61 switch (add) {
62 case 0:
63 s.append(convertedSpaceString, ' ', convertedSpaceString);
64 add = 3;
65 break;
66 case 1:
67 if (i == 0 || i + 1 == in.length()) // at start or end of string
68 s.append(convertedSpaceString);
69 else
70 s.append(' ');
71 break;
72 case 2:
73 if (i == 0) {
74 // at start of string
75 s.append(convertedSpaceString, ' ');
76 } else if (i + 2 == in.length()) {
77 // at end of string
78 s.append(convertedSpaceString, convertedSpaceString);
79 } else {
80 s.append(convertedSpaceString, ' ');
81 }
82 break;
83 }
84 count -= add;
85 }
86 } else
87 s.append(in[i]);
88 i += consumed;
89 }
90
91 return s.toString();
92}
