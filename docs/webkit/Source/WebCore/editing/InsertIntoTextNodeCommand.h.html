26#pragma once
27
28#include "EditCommand.h"
29
30namespace WebCore {
31
32class Text;
33
34class InsertIntoTextNodeCommand : public SimpleEditCommand {
35public:
36 static Ref<InsertIntoTextNodeCommand> create(Ref<Text>&& node, unsigned offset, const String& text, EditAction editingAction = EditAction::Insert)
37 {
38 return adoptRef(*new InsertIntoTextNodeCommand(WTFMove(node), offset, text, editingAction));
39 }
40
41 const String& insertedText();
42
43protected:
44 InsertIntoTextNodeCommand(Ref<Text>&& node, unsigned offset, const String& text, EditAction editingAction);
45
46private:
47 void doApply() override;
48 void doUnapply() override;
49 void doReapply() override;
50
51#ifndef NDEBUG
52 void getNodesInCommand(HashSet<Ref<Node>>&) override;
53#endif
54
55 Ref<Text> m_node;
56 unsigned m_offset;
57 String m_text;
58};
59
60inline const String& InsertIntoTextNodeCommand::insertedText()
61{
62 return m_text;
63}

65} // namespace WebCore
