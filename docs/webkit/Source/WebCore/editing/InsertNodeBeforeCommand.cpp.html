37InsertNodeBeforeCommand::InsertNodeBeforeCommand(Ref<Node>&& insertChild, Node& refChild, ShouldAssumeContentIsAlwaysEditable shouldAssumeContentIsAlwaysEditable, EditAction editingAction)
38 : SimpleEditCommand(refChild.document(), editingAction)
39 , m_insertChild(WTFMove(insertChild))
40 , m_refChild(refChild)
41 , m_shouldAssumeContentIsAlwaysEditable(shouldAssumeContentIsAlwaysEditable)
42{
43 ASSERT(!m_insertChild->parentNode());
44 ASSERT(m_refChild->parentNode());
45
46 ASSERT(m_refChild->parentNode()->hasEditableStyle() || !m_refChild->parentNode()->renderer());
47}
48
49void InsertNodeBeforeCommand::doApply()
50{
51 ContainerNode* parent = m_refChild->parentNode();
52 if (!parent || (m_shouldAssumeContentIsAlwaysEditable == DoNotAssumeContentIsAlwaysEditable && !isEditableNode(*parent)))
53 return;
54 ASSERT(isEditableNode(*parent));
55
56 parent->insertBefore(m_insertChild, m_refChild.ptr());
57}
58
59void InsertNodeBeforeCommand::doUnapply()
60{
61 if (!isEditableNode(m_insertChild))
62 return;
63
64 m_insertChild->remove();
65}

67#ifndef NDEBUG
68void InsertNodeBeforeCommand::getNodesInCommand(HashSet<Ref<Node>>& nodes)
69{
70 addNodeAndDescendants(m_insertChild.ptr(), nodes);
71 addNodeAndDescendants(m_refChild.ptr(), nodes);
72}
73#endif
