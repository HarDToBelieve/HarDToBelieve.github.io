31namespace WebCore {

33MergeIdenticalElementsCommand::MergeIdenticalElementsCommand(Ref<Element>&& first, Ref<Element>&& second)
34 : SimpleEditCommand(first->document())
35 , m_element1(WTFMove(first))
36 , m_element2(WTFMove(second))
37{
38 ASSERT(m_element1->nextSibling() == m_element2.ptr());
39}
40
41void MergeIdenticalElementsCommand::doApply()
42{
43 if (m_element1->nextSibling() != m_element2.ptr() || !m_element1->hasEditableStyle() || !m_element2->hasEditableStyle())
44 return;
45
46 m_atChild = m_element2->firstChild();
47
48 Vector<Ref<Node>> children;
49 for (Node* child = m_element1->firstChild(); child; child = child->nextSibling())
50 children.append(*child);
51
52 for (auto& child : children)
53 m_element2->insertBefore(child, m_atChild.get());
54
55 m_element1->remove();
56}
57
58void MergeIdenticalElementsCommand::doUnapply()
59{
60 RefPtr<Node> atChild = WTFMove(m_atChild);
61
62 auto* parent = m_element2->parentNode();
63 if (!parent || !parent->hasEditableStyle())
64 return;
65
66 if (parent->insertBefore(m_element1, m_element2.ptr()).hasException())
67 return;
68
69 Vector<Ref<Node>> children;
70 for (Node* child = m_element2->firstChild(); child && child != atChild; child = child->nextSibling())
71 children.append(*child);
72
73 for (auto& child : children)
74 m_element1->appendChild(child);
75}

77#ifndef NDEBUG
78void MergeIdenticalElementsCommand::getNodesInCommand(HashSet<Ref<Node>>& nodes)
79{
80 addNodeAndDescendants(m_element1.ptr(), nodes);
81 addNodeAndDescendants(m_element2.ptr(), nodes);
82}
