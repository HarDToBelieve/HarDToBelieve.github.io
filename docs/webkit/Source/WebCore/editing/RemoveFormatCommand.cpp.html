48static bool isElementForRemoveFormatCommand(const Element* element)
49{
50 static const auto elements = makeNeverDestroyed(MemoryCompactLookupOnlyRobinHoodHashSet<QualifiedName> {
51 acronymTag,
52 bTag,
53 bdoTag,
54 bigTag,
55 citeTag,
56 codeTag,
57 dfnTag,
58 emTag,
59 fontTag,
60 iTag,
61 insTag,
62 kbdTag,
63 nobrTag,
64 qTag,
65 sTag,
66 sampTag,
67 smallTag,
68 strikeTag,
69 strongTag,
70 subTag,
71 supTag,
72 ttTag,
73 uTag,
74 varTag,
75 });
76 return elements.get().contains(element->tagQName());
77}
78
79void RemoveFormatCommand::doApply()
80{
81 if (endingSelection().isNoneOrOrphaned())
82 return;
83
84 // Get the default style for this editable root, it's the style that we'll give the
85 // content that we're operating on.
86 Node* root = endingSelection().rootEditableElement();
87 auto defaultStyle = EditingStyle::create(root);
88
89 // We want to remove everything but transparent background.
90 // FIXME: We shouldn't access style().
91 defaultStyle->style()->setProperty(CSSPropertyBackgroundColor, CSSValueTransparent);
92
93 applyCommandToComposite(ApplyStyleCommand::create(document(), defaultStyle.ptr(), isElementForRemoveFormatCommand, editingAction()));
94}
