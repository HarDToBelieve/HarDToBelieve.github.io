39ReplaceNodeWithSpanCommand::ReplaceNodeWithSpanCommand(Ref<HTMLElement>&& element)
40 : SimpleEditCommand(element->document())
41 , m_elementToReplace(WTFMove(element))
42{
43}
44
45static void swapInNodePreservingAttributesAndChildren(HTMLElement& newNode, HTMLElement& nodeToReplace)
46{
47 ASSERT(nodeToReplace.isConnected());
48 RefPtr<ContainerNode> parentNode = nodeToReplace.parentNode();
49
50 // FIXME: Fix this to send the proper MutationRecords when MutationObservers are present.
51 newNode.cloneDataFromElement(nodeToReplace);
52 auto children = collectChildNodes(nodeToReplace);
53 for (auto& child : children)
54 newNode.appendChild(child);
55
56 parentNode->insertBefore(newNode, &nodeToReplace);
57 parentNode->removeChild(nodeToReplace);
58}
59
60void ReplaceNodeWithSpanCommand::doApply()
61{
62 if (!m_elementToReplace->isConnected())
63 return;
64 if (!m_spanElement)
65 m_spanElement = HTMLSpanElement::create(m_elementToReplace->document());
66 swapInNodePreservingAttributesAndChildren(*m_spanElement, m_elementToReplace);
67}
68
69void ReplaceNodeWithSpanCommand::doUnapply()
70{
71 if (!m_spanElement->isConnected())
72 return;
73 swapInNodePreservingAttributesAndChildren(m_elementToReplace, *m_spanElement);
74}

76#ifndef NDEBUG
77void ReplaceNodeWithSpanCommand::getNodesInCommand(HashSet<Ref<Node>>& nodes)
78{
79 addNodeAndDescendants(m_elementToReplace.ptr(), nodes);
80 addNodeAndDescendants(m_spanElement.get(), nodes);
81}
