35namespace WebCore {
36
37class HTMLElement;
38
39// More accurately, this is ReplaceElementWithSpanPreservingChildrenAndAttributesCommand
40class ReplaceNodeWithSpanCommand : public SimpleEditCommand {
41public:
42 static Ref<ReplaceNodeWithSpanCommand> create(Ref<HTMLElement>&& element)
43 {
44 return adoptRef(*new ReplaceNodeWithSpanCommand(WTFMove(element)));
45 }
46
47 HTMLElement* spanElement() { return m_spanElement.get(); }
48
49private:
50 explicit ReplaceNodeWithSpanCommand(Ref<HTMLElement>&&);
51
52 void doApply() override;
53 void doUnapply() override;
54
55#ifndef NDEBUG
56 void getNodesInCommand(HashSet<Ref<Node>>&) override;
57#endif
58
59 Ref<HTMLElement> m_elementToReplace;
60 RefPtr<HTMLElement> m_spanElement;
61};
