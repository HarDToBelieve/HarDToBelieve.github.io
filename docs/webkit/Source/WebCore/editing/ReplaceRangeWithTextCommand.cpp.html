43ReplaceRangeWithTextCommand::ReplaceRangeWithTextCommand(const SimpleRange& rangeToBeReplaced, const String& text)
44 : CompositeEditCommand(rangeToBeReplaced.start.document(), EditAction::InsertReplacement)
45 , m_rangeToBeReplaced(rangeToBeReplaced)
46 , m_text(text)
47{
48}
49
50bool ReplaceRangeWithTextCommand::willApplyCommand()
51{
52 m_textFragment = createFragmentFromText(m_rangeToBeReplaced, m_text);
53 return CompositeEditCommand::willApplyCommand();
54}
55
56void ReplaceRangeWithTextCommand::doApply()
57{
58 VisibleSelection selection { m_rangeToBeReplaced };
59
60 if (!document().selection().shouldChangeSelection(selection))
61 return;
62
63 if (!characterCount(m_rangeToBeReplaced))
64 return;
65
66 applyCommandToComposite(SetSelectionCommand::create(selection, FrameSelection::defaultSetSelectionOptions()));
67 applyCommandToComposite(ReplaceSelectionCommand::create(document(), WTFMove(m_textFragment), ReplaceSelectionCommand::MatchStyle, EditAction::Paste));
68}
69
70String ReplaceRangeWithTextCommand::inputEventData() const
71{
72 if (isEditingTextAreaOrTextInput())
73 return m_text;
74
75 return CompositeEditCommand::inputEventData();
76}
77
78RefPtr<DataTransfer> ReplaceRangeWithTextCommand::inputEventDataTransfer() const
79{
80 if (!isEditingTextAreaOrTextInput())
81 return DataTransfer::createForInputEvent(m_text, serializeFragment(*m_textFragment, SerializedNodes::SubtreeIncludingNode));
82
83 return CompositeEditCommand::inputEventDataTransfer();
84}
85
86Vector<RefPtr<StaticRange>> ReplaceRangeWithTextCommand::targetRanges() const
87{
88 return { 1, StaticRange::create(m_rangeToBeReplaced) };
89}
