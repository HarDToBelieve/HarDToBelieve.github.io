40SelectionGeometryGatherer::SelectionGeometryGatherer(RenderView& renderView)
41 : m_renderView(renderView)
42 , m_isTextOnly(true)
43{
44}
45
46void SelectionGeometryGatherer::addQuad(RenderLayerModelObject *repaintContainer, const FloatQuad& quad)
47{
48 if (!quad.boundingBoxIsEmpty())
49 m_quads.append(repaintContainer ? repaintContainer->localToAbsoluteQuad(quad) : quad);
50}
51
52void SelectionGeometryGatherer::addGapRects(RenderLayerModelObject *repaintContainer, const GapRects& rects)
53{
54 if (repaintContainer) {
55 GapRects absoluteGapRects;
56 absoluteGapRects.uniteLeft(LayoutRect(repaintContainer->localToAbsoluteQuad(FloatQuad(rects.left())).boundingBox()));
57 absoluteGapRects.uniteCenter(LayoutRect(repaintContainer->localToAbsoluteQuad(FloatQuad(rects.center())).boundingBox()));
58 absoluteGapRects.uniteRight(LayoutRect(repaintContainer->localToAbsoluteQuad(FloatQuad(rects.right())).boundingBox()));
59 m_gapRects.append(absoluteGapRects);
60 } else
61 m_gapRects.append(rects);
62}
63
64SelectionGeometryGatherer::Notifier::Notifier(SelectionGeometryGatherer& gatherer)
65 : m_gatherer(gatherer)
66{
67}
68
69SelectionGeometryGatherer::Notifier::~Notifier()
70{
71 auto page = m_gatherer.m_renderView.view().frame().page();
72 if (!page)
73 return;
74
75 page->servicesOverlayController().selectionRectsDidChange(m_gatherer.boundingRects(), m_gatherer.m_gapRects, m_gatherer.isTextOnly());
76 page->imageOverlayController().selectionQuadsDidChange(m_gatherer.m_renderView.frame(), m_gatherer.m_quads);
77}
78
79Vector<LayoutRect> SelectionGeometryGatherer::boundingRects() const
80{
81 return m_quads.map([](auto& quad) {
82 return LayoutRect { quad.boundingBox() };
83 });
84}
85
86std::unique_ptr<SelectionGeometryGatherer::Notifier> SelectionGeometryGatherer::clearAndCreateNotifier()
87{
88 m_quads.clear();
89 m_gapRects.clear();
90 m_isTextOnly = true;
91
92 return makeUnique<Notifier>(*this);
93}
