48static void findGrammaticalErrors(TextCheckerClient& client, StringView text, Vector<TextCheckingResult>& results)
49{
50 for (unsigned checkLocation = 0; checkLocation < text.length(); ) {
51 int badGrammarLocation = -1;
52 int badGrammarLength = 0;
53 Vector<GrammarDetail> badGrammarDetails;
54 client.checkGrammarOfString(text.substring(checkLocation), badGrammarDetails, &badGrammarLocation, &badGrammarLength);
55 if (!badGrammarLength)
56 break;
57
58 ASSERT(badGrammarLocation >= 0);
59 ASSERT(static_cast<unsigned>(badGrammarLocation) <= text.length() - checkLocation);
60 ASSERT(badGrammarLength > 0);
61 ASSERT(static_cast<unsigned>(badGrammarLength) <= text.length() - checkLocation - badGrammarLocation);
62
63 TextCheckingResult badGrammar;
64 badGrammar.type = TextCheckingType::Grammar;
65 badGrammar.range = CharacterRange(checkLocation + badGrammarLocation, badGrammarLength);
66 badGrammar.details = WTFMove(badGrammarDetails);
67 results.append(badGrammar);
68
69 checkLocation += badGrammarLocation + badGrammarLength;
70 }
71}

73static void findMisspellings(TextCheckerClient& client, StringView text, Vector<TextCheckingResult>& results)
74{
75 UBreakIterator* iterator = wordBreakIterator(text);
76 if (!iterator)
77 return;
78 for (int wordStart = ubrk_current(iterator); wordStart >= 0; ) {
79 int wordEnd = ubrk_next(iterator);
80 if (wordEnd < 0)
81 break;
82
83 int wordLength = wordEnd - wordStart;
84 int misspellingLocation = -1;
85 int misspellingLength = 0;
86 client.checkSpellingOfString(text.substring(wordStart, wordLength), &misspellingLocation, &misspellingLength);
87
88 if (misspellingLength > 0) {
89 ASSERT(misspellingLocation >= 0);
90 ASSERT(misspellingLocation <= wordLength);
91 ASSERT(misspellingLength > 0);
92 ASSERT(misspellingLocation + misspellingLength <= wordLength);
93
94 TextCheckingResult misspelling;
95 misspelling.type = TextCheckingType::Spelling;
96 misspelling.range = CharacterRange(wordStart + misspellingLocation, misspellingLength);
97 misspelling.replacement = client.getAutoCorrectSuggestionForMisspelledWord(text.substring(misspelling.range.location, misspelling.range.length).toStringWithoutCopying());
98 results.append(misspelling);
99 }
100
101 wordStart = wordEnd;
102 }
103}

107static SimpleRange expandToParagraphBoundary(const SimpleRange& range)
108{
109 auto start = makeBoundaryPoint(startOfParagraph(makeDeprecatedLegacyPosition(range.start)));
110 auto end = makeBoundaryPoint(endOfParagraph(makeDeprecatedLegacyPosition(range.end)));
111 if (!start || !end)
112 return range;
113 return { *start, *end };

231auto TextCheckingHelper::findMisspelledWords(Operation operation) const -> std::pair<MisspelledWord, std::optional<SimpleRange>>
232{
233 std::pair<MisspelledWord, std::optional<SimpleRange>> first;
234
235 uint64_t currentChunkOffset = 0;
236
237 for (WordAwareIterator it(m_range); !it.atEnd(); currentChunkOffset += it.text().length(), it.advance()) {
238 StringView text = it.text();
239
240 if (text == " "_s)
241 continue;
242
243 int misspellingLocation = -1;
244 int misspellingLength = 0;
245 m_client.textChecker()->checkSpellingOfString(text, &misspellingLocation, &misspellingLength);
246
247 int textLength = text.length();
248
249 // 5490627 shows that there was some code path here where the String constructor below crashes.
250 // We don't know exactly what combination of bad input caused this, so we're making this much
251 // more robust against bad input on release builds.
252 ASSERT(misspellingLength >= 0);
253 ASSERT(misspellingLocation >= -1);
254 ASSERT(!misspellingLength || misspellingLocation >= 0);
255 ASSERT(misspellingLocation < textLength);
256 ASSERT(misspellingLength <= textLength);
257 ASSERT(misspellingLocation + misspellingLength <= textLength);
258 if (!(misspellingLocation >= 0 && misspellingLength > 0 && misspellingLocation < textLength && misspellingLength <= textLength && misspellingLocation + misspellingLength <= textLength))
259 continue;
260
261 auto misspellingRange = resolveCharacterRange(m_range, CharacterRange(currentChunkOffset + misspellingLocation, misspellingLength));
262
263 if (operation == Operation::MarkAll)
264 addMarker(misspellingRange, DocumentMarker::Spelling);
265
266 if (first.first.word.isNull()) {
267 first = {
268 {
269 text.substring(misspellingLocation, misspellingLength).toString(),
270 currentChunkOffset + misspellingLocation
271 },
272 WTFMove(misspellingRange)
273 };
274 }
275
276 if (operation == Operation::FindFirst)
277 break;
278 }
279
280 return first;
281}

288auto TextCheckingHelper::findFirstMisspelledWordOrUngrammaticalPhrase(bool checkGrammar) const -> Variant<MisspelledWord, UngrammaticalPhrase>
289{
290 if (!unifiedTextCheckerEnabled())
291 return { };
292
293 if (platformDrivenTextCheckerEnabled())
294 return { };
295
296 Variant<MisspelledWord, UngrammaticalPhrase> firstFoundItem;
297 GrammarDetail grammarDetail;
298
299 String misspelledWord;
300 std::optional<SimpleRange> misspelledWordRange;
301 String badGrammarPhrase;
302
303 // Expand the search range to encompass entire paragraphs, since text checking needs that much context.
304 // Determine the character offset from the start of the paragraph to the start of the original search range,
305 // since we will want to ignore results in this area.
306 auto paragraphRange = *makeSimpleRange(startOfParagraph(makeDeprecatedLegacyPosition(m_range.start)), m_range.end);
307 auto totalRangeLength = characterCount(paragraphRange);
308 paragraphRange.end = *makeBoundaryPoint(endOfParagraph(makeDeprecatedLegacyPosition(m_range.start)));
309
310 auto rangeStartOffset = characterCount({ paragraphRange.start, m_range.start });
311 uint64_t totalLengthProcessed = 0;
312
313 bool firstIteration = true;
314 bool lastIteration = false;
315 while (totalLengthProcessed < totalRangeLength) {
316 // Iterate through the search range by paragraphs, checking each one for spelling and grammar.
317 auto currentLength = characterCount(paragraphRange);
318 uint64_t currentStartOffset = firstIteration ? rangeStartOffset : 0;
319 uint64_t currentEndOffset = currentLength;
320 if (inSameParagraph(makeDeprecatedLegacyPosition(paragraphRange.start), makeDeprecatedLegacyPosition(m_range.end))) {
321 // Determine the character offset from the end of the original search range to the end of the paragraph,
322 // since we will want to ignore results in this area.
323 currentEndOffset = characterCount({ paragraphRange.start, m_range.end });
324 lastIteration = true;
325 }
326 if (currentStartOffset < currentEndOffset) {
327 String paragraphString = plainText(paragraphRange);
328 if (paragraphString.length() > 0) {
329 bool foundGrammar = false;
330 uint64_t spellingLocation = 0;
331 uint64_t grammarPhraseLocation = 0;
332 uint64_t grammarDetailLocation = 0;
333 unsigned grammarDetailIndex = 0;
334
335 Vector<TextCheckingResult> results;
336 OptionSet<TextCheckingType> checkingTypes { TextCheckingType::Spelling };
337 if (checkGrammar)
338 checkingTypes.add(TextCheckingType::Grammar);
339 VisibleSelection currentSelection;
340 if (Frame* frame = paragraphRange.start.document().frame())
341 currentSelection = frame->selection().selection();
342 checkTextOfParagraph(*m_client.textChecker(), paragraphString, checkingTypes, results, currentSelection);
343
344 for (auto& result : results) {
345 if (result.type == TextCheckingType::Spelling && result.range.location >= currentStartOffset && result.range.location + result.range.length <= currentEndOffset) {
346 ASSERT(result.range.length > 0);
347 spellingLocation = result.range.location;
348 misspelledWord = paragraphString.substring(result.range.location, result.range.length);
349 ASSERT(misspelledWord.length());
350 break;
351 }
352 if (checkGrammar && result.type == TextCheckingType::Grammar && result.range.location < currentEndOffset && result.range.location + result.range.length > currentStartOffset) {
353 ASSERT(result.range.length > 0);
354 // We can't stop after the first grammar result, since there might still be a spelling result after
355 // it begins but before the first detail in it, but we can stop if we find a second grammar result.
356 if (foundGrammar)
357 break;
358 for (unsigned j = 0; j < result.details.size(); j++) {
359 const GrammarDetail* detail = &result.details[j];
360 ASSERT(detail->range.length > 0);
361 if (result.range.location + detail->range.location >= currentStartOffset && result.range.location + detail->range.location + detail->range.length <= currentEndOffset && (!foundGrammar || result.range.location + detail->range.location < grammarDetailLocation)) {
362 grammarDetailIndex = j;
363 grammarDetailLocation = result.range.location + detail->range.location;
364 foundGrammar = true;
365 }
366 }
367 if (foundGrammar) {
368 grammarPhraseLocation = result.range.location;
369 grammarDetail = result.details[grammarDetailIndex];
370 badGrammarPhrase = paragraphString.substring(result.range.location, result.range.length);
371 ASSERT(badGrammarPhrase.length());
372 }
373 }
374 }
375
376 if (!misspelledWord.isEmpty() && (!checkGrammar || badGrammarPhrase.isEmpty() || spellingLocation <= grammarDetailLocation)) {
377 uint64_t spellingOffset = spellingLocation - currentStartOffset;
378 if (!firstIteration)
379 spellingOffset += characterCount({ m_range.start, paragraphRange.start });
380 firstFoundItem = MisspelledWord {
381 misspelledWord,
382 spellingOffset
383 };
384 break;
385 }
386 if (checkGrammar && !badGrammarPhrase.isEmpty()) {
387 uint64_t grammarPhraseOffset = grammarPhraseLocation - currentStartOffset;
388 if (!firstIteration)
389 grammarPhraseOffset += characterCount({ m_range.start, paragraphRange.start });
390 firstFoundItem = UngrammaticalPhrase {
391 badGrammarPhrase,
392 grammarPhraseOffset,
393 grammarDetail
394 };
395 break;
396 }
397 }
398 }
399 if (lastIteration || totalLengthProcessed + currentLength >= totalRangeLength)
400 break;
401
402 auto nextStart = startOfNextParagraph(makeDeprecatedLegacyPosition(paragraphRange.end));
403 auto nextParagraphRange = makeSimpleRange(nextStart, endOfParagraph(nextStart));
404 if (!nextParagraphRange)
405 break;
406 paragraphRange = WTFMove(*nextParagraphRange);
407
408 firstIteration = false;
409 totalLengthProcessed += currentLength;
410 }
411 return firstFoundItem;
412}

414int TextCheckingHelper::findUngrammaticalPhrases(Operation operation, const Vector<GrammarDetail>& grammarDetails, uint64_t badGrammarPhraseLocation, uint64_t startOffset, uint64_t endOffset) const
415{
416 // Found some bad grammar. Find the earliest detail range that starts in our search range (if any).
417 // Optionally add a DocumentMarker for each detail in the range.
418 uint64_t earliestDetailLocationSoFar = 0;
419 int earliestDetailIndex = -1;
420 for (unsigned i = 0; i < grammarDetails.size(); i++) {
421 const GrammarDetail* detail = &grammarDetails[i];
422 ASSERT(detail->range.length > 0);
423
424 uint64_t detailStartOffsetInParagraph = badGrammarPhraseLocation + detail->range.location;
425
426 // Skip this detail if it starts before the original search range
427 if (detailStartOffsetInParagraph < startOffset)
428 continue;
429
430 // Skip this detail if it starts after the original search range
431 if (detailStartOffsetInParagraph >= endOffset)
432 continue;
433
434 if (operation == Operation::MarkAll) {
435 auto badGrammarRange = resolveCharacterRange(m_range, { badGrammarPhraseLocation - startOffset + detail->range.location, detail->range.length });
436 addMarker(badGrammarRange, DocumentMarker::Grammar, detail->userDescription);
437 }
438
439 // Remember this detail only if it's earlier than our current candidate (the details aren't in a guaranteed order)
440 if (earliestDetailIndex < 0 || earliestDetailLocationSoFar > detail->range.location) {
441 earliestDetailIndex = i;
442 earliestDetailLocationSoFar = detail->range.location;
443 }
444 }
445 return earliestDetailIndex;
446}

448auto TextCheckingHelper::findUngrammaticalPhrases(Operation operation) const -> UngrammaticalPhrase
449{
450 UngrammaticalPhrase result;
451
452 // Expand the search range to encompass entire paragraphs, since grammar checking needs that much context.
453 // Determine the character offset from the start of the paragraph to the start of the original search range,
454 // since we will want to ignore results in this area.
455 TextCheckingParagraph paragraph(m_range);
456
457 // Start checking from beginning of paragraph, but skip past results that occur before the start of the original search range.
458 for (uint64_t startOffset = 0; startOffset < paragraph.checkingEnd(); ) {
459 Vector<GrammarDetail> grammarDetails;
460 int badGrammarPhraseLocation = -1;
461 int badGrammarPhraseLength = 0;
462 m_client.textChecker()->checkGrammarOfString(paragraph.text().substring(startOffset), grammarDetails, &badGrammarPhraseLocation, &badGrammarPhraseLength);
463
464 if (!badGrammarPhraseLength) {
465 ASSERT(badGrammarPhraseLocation == -1);
466 return { };
467 }
468
469 ASSERT(badGrammarPhraseLocation >= 0);
470 badGrammarPhraseLocation += startOffset;
471
472 // Found some bad grammar. Find the earliest detail range that starts in our search range (if any).
473 int badGrammarIndex = findUngrammaticalPhrases(operation, grammarDetails, badGrammarPhraseLocation, paragraph.checkingStart(), paragraph.checkingEnd());
474
475 if (badGrammarIndex >= 0 && result.phrase.isEmpty()) {
476 result.offset = badGrammarPhraseLocation - paragraph.checkingStart();
477 result.phrase = paragraph.text().substring(badGrammarPhraseLocation, badGrammarPhraseLength).toString();
478 ASSERT(static_cast<unsigned>(badGrammarIndex) < grammarDetails.size());
479 result.detail = grammarDetails[badGrammarIndex];
480
481 // Found one. We're done now, unless we're marking each instance.
482 if (operation == Operation::FindFirst)
483 break;
484 }
485
486 // These results were all between the start of the paragraph and the start of the search range; look beyond this phrase.
487 startOffset = badGrammarPhraseLocation + badGrammarPhraseLength;
488 }
489
490 return result;
491}

495 return findUngrammaticalPhrases(Operation::FindFirst);

498TextCheckingGuesses TextCheckingHelper::guessesForMisspelledWordOrUngrammaticalPhrase(bool checkGrammar) const
499{
500 if (!unifiedTextCheckerEnabled())
501 return { };
502
503 if (platformDrivenTextCheckerEnabled())
504 return { };
505
506 if (m_range.collapsed())
507 return { };
508
509 // Expand the range to encompass entire paragraphs, since text checking needs that much context.
510 TextCheckingParagraph paragraph(m_range);
511 if (paragraph.isEmpty())
512 return { };
513
514 Vector<TextCheckingResult> results;
515 OptionSet<TextCheckingType> checkingTypes { TextCheckingType::Spelling };
516 if (checkGrammar)
517 checkingTypes.add(TextCheckingType::Grammar);
518 VisibleSelection currentSelection;
519 if (auto frame = m_range.start.document().frame())
520 currentSelection = frame->selection().selection();
521 checkTextOfParagraph(*m_client.textChecker(), paragraph.text(), checkingTypes, results, currentSelection);
522
523 for (auto& result : results) {
524 if (result.type == TextCheckingType::Spelling && paragraph.checkingRangeMatches(result.range)) {
525 String misspelledWord = paragraph.checkingSubstring().toString();
526 ASSERT(misspelledWord.length());
527 Vector<String> guesses;
528 m_client.textChecker()->getGuessesForWord(misspelledWord, String(), currentSelection, guesses);
529 m_client.updateSpellingUIWithMisspelledWord(misspelledWord);
530 return { WTFMove(guesses), true, false };
531 }
532 }
533
534 if (!checkGrammar)
535 return { };
536
537 for (auto& result : results) {
538 if (result.type == TextCheckingType::Grammar && paragraph.isCheckingRangeCoveredBy(result.range)) {
539 for (auto& detail : result.details) {
540 ASSERT(detail.range.length > 0);
541 if (paragraph.checkingRangeMatches({ result.range.location + detail.range.location, detail.range.length })) {
542 String badGrammarPhrase = paragraph.text().substring(result.range.location, result.range.length).toString();
543 ASSERT(badGrammarPhrase.length());
544 m_client.updateSpellingUIWithGrammarString(badGrammarPhrase, detail);
545 return { WTFMove(detail.guesses), false, true };
546 }
547 }
548 }
549 }
550
551 return { };
552}

569void checkTextOfParagraph(TextCheckerClient& client, StringView text, OptionSet<TextCheckingType> checkingTypes, Vector<TextCheckingResult>& results, const VisibleSelection& currentSelection)
570{
571#if USE(UNIFIED_TEXT_CHECKING)
572 results = client.checkTextOfParagraph(text, checkingTypes, currentSelection);
573#else
574 UNUSED_PARAM(currentSelection);
575
576 Vector<TextCheckingResult> misspellings;
577 if (checkingTypes.contains(TextCheckingType::Spelling))
578 findMisspellings(client, text, misspellings);
579
580 // Look for grammatical errors that occur before the first misspelling.
581 Vector<TextCheckingResult> grammaticalErrors;
582 if (checkingTypes.contains(TextCheckingType::Grammar)) {
583 unsigned grammarCheckLength = text.length();
584 for (auto& misspelling : misspellings)
585 grammarCheckLength = std::min<unsigned>(grammarCheckLength, misspelling.range.location);
586 findGrammaticalErrors(client, text.substring(0, grammarCheckLength), grammaticalErrors);
587 }
588
589 results = WTFMove(grammaticalErrors);
590
591 if (results.isEmpty())
592 results = WTFMove(misspellings);
593 else
594 results.appendVector(misspellings);
595#endif // USE(UNIFIED_TEXT_CHECKING)
596}
