37// FIXME: Should move this class to its own header.
38class TextCheckingParagraph {
39public:
40 explicit TextCheckingParagraph(const SimpleRange& checkingAndAutomaticReplacementRange);
41 TextCheckingParagraph(const SimpleRange& checkingRange, const SimpleRange& automaticReplacementRange, const std::optional<SimpleRange>& paragraphRange);
42
43 uint64_t rangeLength() const;
44 SimpleRange subrange(CharacterRange) const;
45 ExceptionOr<uint64_t> offsetTo(const Position&) const;
46 void expandRangeToNextEnd();
47
48 StringView text() const;
49
50 bool isEmpty() const;
51
52 uint64_t checkingStart() const;
53 uint64_t checkingEnd() const;
54 uint64_t checkingLength() const;
55 StringView checkingSubstring() const { return text().substring(checkingStart(), checkingLength()); }
56
57 uint64_t automaticReplacementStart() const;
58 uint64_t automaticReplacementLength() const;
59
60 bool checkingRangeMatches(CharacterRange range) const { return range.location == checkingStart() && range.length == checkingLength(); }
61 bool isCheckingRangeCoveredBy(CharacterRange range) const { return range.location <= checkingStart() && range.location + range.length >= checkingStart() + checkingLength(); }
62 bool checkingRangeCovers(CharacterRange range) const { return range.location < checkingEnd() && range.location + range.length > checkingStart(); }
63
64 const SimpleRange& paragraphRange() const;
65
66private:
67 void invalidateParagraphRangeValues();
68 const SimpleRange& offsetAsRange() const;
69
70 SimpleRange m_checkingRange;
71 SimpleRange m_automaticReplacementRange;
72 mutable std::optional<SimpleRange> m_paragraphRange;
73 mutable std::optional<SimpleRange> m_offsetAsRange;
74 mutable String m_text;
75 mutable std::optional<uint64_t> m_checkingStart;
76 mutable std::optional<uint64_t> m_checkingLength;
77 mutable std::optional<uint64_t> m_automaticReplacementStart;
78 mutable std::optional<uint64_t> m_automaticReplacementLength;
79};
