43void TextInsertionBaseCommand::applyTextInsertionCommand(Frame* frame, TextInsertionBaseCommand& command, const VisibleSelection& selectionForInsertion, const VisibleSelection& endingSelection)
44{
45 bool changeSelection = selectionForInsertion != endingSelection;
46 if (changeSelection) {
47 command.setStartingSelection(selectionForInsertion);
48 command.setEndingSelection(selectionForInsertion);
49 }
50 command.apply();
51 if (changeSelection) {
52 command.setEndingSelection(endingSelection);
53 frame->selection().setSelection(endingSelection);
54 }
55}

57String dispatchBeforeTextInsertedEvent(const String& text, const VisibleSelection& selectionForInsertion, bool insertionIsForUpdatingComposition)
58{
59 if (insertionIsForUpdatingComposition)
60 return text;
61
62 String newText = text;
63 if (Node* startNode = selectionForInsertion.start().containerNode()) {
64 if (startNode->rootEditableElement()) {
65 // Send BeforeTextInsertedEvent. The event handler will update text if necessary.
66 Ref<BeforeTextInsertedEvent> event = BeforeTextInsertedEvent::create(text);
67 startNode->rootEditableElement()->dispatchEvent(event);
68 newText = event->text();
69 }
70 }
71 return newText;
72}
73
74bool canAppendNewLineFeedToSelection(const VisibleSelection& selection)
75{
76 Node* node = selection.rootEditableElement();
77 if (!node)
78 return false;
79
80 Ref<BeforeTextInsertedEvent> event = BeforeTextInsertedEvent::create(String("\n"));
81 node->dispatchEvent(event);
82 return event->text().length();
83}
