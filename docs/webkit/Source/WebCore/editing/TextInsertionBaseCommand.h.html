48// LineOperation should define member function "opeartor (size_t lineOffset, size_t lineLength, bool isLastLine)".
49// lienLength doesn't include the newline character. So the value of lineLength could be 0.
50template <class LineOperation>
51void forEachLineInString(const String& string, const LineOperation& operation)
52{
53 unsigned offset = 0;
54 size_t newline;
55 while ((newline = string.find('\n', offset)) != notFound) {
56 operation(offset, newline - offset, false);
57 offset = newline + 1;
58 }
59 if (!offset)
60 operation(0, string.length(), true);
61 else {
62 unsigned length = string.length();
63 if (length != offset)
64 operation(offset, length - offset, true);
65 }
66}
