33class VisiblePosition {
34public:
35 // VisiblePosition default affinity is downstream for callers that do not really care because it is more efficient than upstream.
36 static constexpr auto defaultAffinity = Affinity::Downstream;
37
38 VisiblePosition() = default;
39
40 // This constructor will ignore the passed-in affinity if the position is not at the end of a line.
41 WEBCORE_EXPORT VisiblePosition(const Position&, Affinity = defaultAffinity);
42
43 bool isNull() const { return m_deepPosition.isNull(); }
44 bool isNotNull() const { return m_deepPosition.isNotNull(); }
45 bool isOrphan() const { return m_deepPosition.isOrphan(); }
46
47 Position deepEquivalent() const { return m_deepPosition; }
48 Affinity affinity() const { return m_affinity; }
49
50 void setAffinity(Affinity affinity) { m_affinity = affinity; }
51
52 // FIXME: Change the following functions' parameter from a boolean to StayInEditableContent.
53
54 // next() and previous() increment/decrement by a character cluster.
55 WEBCORE_EXPORT VisiblePosition next(EditingBoundaryCrossingRule = CanCrossEditingBoundary, bool* reachedBoundary = nullptr) const;
56 WEBCORE_EXPORT VisiblePosition previous(EditingBoundaryCrossingRule = CanCrossEditingBoundary, bool* reachedBoundary = nullptr) const;
57
58 VisiblePosition honorEditingBoundaryAtOrBefore(const VisiblePosition&, bool* reachedBoundary = nullptr) const;
59 VisiblePosition honorEditingBoundaryAtOrAfter(const VisiblePosition&, bool* reachedBoundary = nullptr) const;
60
61 WEBCORE_EXPORT VisiblePosition left(bool stayInEditableContent = false, bool* reachedBoundary = nullptr) const;
62 WEBCORE_EXPORT VisiblePosition right(bool stayInEditableContent = false, bool* reachedBoundary = nullptr) const;
63
64 WEBCORE_EXPORT UChar32 characterAfter() const;
65 UChar32 characterBefore() const { return previous().characterAfter(); }
66
67 // FIXME: This does not handle [table, 0] correctly.
68 Element* rootEditableElement() const { return m_deepPosition.isNotNull() ? m_deepPosition.deprecatedNode()->rootEditableElement() : 0; }
69
70 InlineRunAndOffset inlineRunAndOffset() const;
71 InlineRunAndOffset inlineRunAndOffset(TextDirection primaryDirection) const;
72

74 LayoutRect rect;
75 RenderObject* renderer { nullptr };
76 };
77 WEBCORE_EXPORT LocalCaretRect localCaretRect() const;
78
79 // Bounds of (possibly transformed) caret in absolute coords.
80 WEBCORE_EXPORT IntRect absoluteCaretBounds(bool* insideFixed = nullptr) const;
81
82 // Abs x/y position of the caret ignoring transforms.
83 // FIXME: navigation with transforms should be smarter.
84 WEBCORE_EXPORT int lineDirectionPointForBlockDirectionNavigation() const;
85
86 WEBCORE_EXPORT FloatRect absoluteSelectionBoundsForLine() const;
87
88 // This is a tentative enhancement of operator== to account for affinity.
89 // FIXME: Combine this function with operator==.
90 bool equals(const VisiblePosition&) const;
91
92#if ENABLE(TREE_DEBUGGING)
93 void debugPosition(const char* msg = "") const;
94 String debugDescription() const;
95 void showTreeForThis() const;
96#endif

98private:
99 static Position canonicalPosition(const Position&);
100
101 Position leftVisuallyDistinctCandidate() const;
102 Position rightVisuallyDistinctCandidate() const;
103
104 Position m_deepPosition;
105 Affinity m_affinity { defaultAffinity };
106};

148// inlines
149
150inline bool operator==(const VisiblePosition& a, const VisiblePosition& b)
151{
152 // FIXME: Is it correct and helpful for this to be ignoring differences in affinity?
153 return a.deepEquivalent() == b.deepEquivalent();
154}
155
156inline bool operator!=(const VisiblePosition& a, const VisiblePosition& b)
157{
158 return !(a == b);
159}
160
161inline bool operator<(const VisiblePosition& a, const VisiblePosition& b)
162{
163 return is_lt(documentOrder(a, b));
164}
165
166inline bool operator>(const VisiblePosition& a, const VisiblePosition& b)
167{
168 return is_gt(documentOrder(a, b));
169}
170
171inline bool operator<=(const VisiblePosition& a, const VisiblePosition& b)
172{
173 return is_lteq(documentOrder(a, b));
174}
175
176inline bool operator>=(const VisiblePosition& a, const VisiblePosition& b)
177{
178 return is_gteq(documentOrder(a, b));
179}
