136bool VisibleSelection::isOrphan() const
137{
138 if (m_base.isOrphan() || m_extent.isOrphan() || m_start.isOrphan() || m_end.isOrphan())
139 return true;
140 if (m_anchor.isOrphan() && m_anchor.document()->settings().liveRangeSelectionEnabled())
141 return true;
142 if (m_focus.isOrphan() && m_focus.document()->settings().liveRangeSelectionEnabled())
143 return true;
144 return false;

148{

150 if (!baseDocument)
151 return nullptr;
152

154 return nullptr;
155
156 if (baseDocument->settings().liveRangeSelectionEnabled() && (m_anchor.document() != baseDocument.get() || m_focus.document() != baseDocument.get()))
157 return nullptr;

170std::optional<SimpleRange> VisibleSelection::toNormalizedRange() const
171{
172 if (isNoneOrOrphaned())
173 return std::nullopt;
174
175 // Make sure we have an updated layout since this function is called
176 // in the course of running edit commands which modify the DOM.
177 // Failing to call this can result in equivalentXXXPosition calls returning
178 // incorrect results.
179 m_start.anchorNode()->document().updateLayout();
180
181 // Check again, because updating layout can clear the selection.
182 if (isNoneOrOrphaned())
183 return std::nullopt;
184
185 Position s, e;
186 if (isCaret()) {
187 // If the selection is a caret, move the range start upstream. This helps us match
188 // the conventions of text editors tested, which make style determinations based
189 // on the character before the caret, if any.
190 s = m_start.upstream().parentAnchoredEquivalent();
191 e = s;
192 } else {
193 // If the selection is a range, select the minimum range that encompasses the selection.
194 // Again, this is to match the conventions of text editors tested, which make style
195 // determinations based on the first character of the selection.
196 // For instance, this operation helps to make sure that the "X" selected below is the
197 // only thing selected. The range should not be allowed to "leak" out to the end of the
198 // previous text node, or to the beginning of the next text node, each of which has a
199 // different style.
200 //
201 // On a treasure map, <b>X</b> marks the spot.
202 // ^ selected
203 //
204 ASSERT(isRange());
205 s = m_start.downstream().parentAnchoredEquivalent();
206 e = m_end.upstream().parentAnchoredEquivalent();
207 // Make sure the start is before the end.
208 // The end can wind up before the start if collapsed whitespace is the only thing selected.
209 if (s > e)
210 std::swap(s, e);
211 }
212
213 return makeSimpleRange(s, e);
214}

230void VisibleSelection::appendTrailingWhitespace()
231{
232 auto scope = deprecatedEnclosingBlockFlowElement(m_end.deprecatedNode());
233 if (!scope)
234 return;
235
236 CharacterIterator charIt(*makeSimpleRange(m_end, makeBoundaryPointAfterNodeContents(*scope)), TextIteratorBehavior::EmitsCharactersBetweenAllVisiblePositions);
237 for (; !charIt.atEnd() && charIt.text().length(); charIt.advance(1)) {
238 UChar c = charIt.text()[0];
239 if ((!isSpaceOrNewline(c) && c != noBreakSpace) || c == '\n')
240 break;
241 m_end = makeDeprecatedLegacyPosition(charIt.range().end);
242 }

245void VisibleSelection::setBaseAndExtentToDeepEquivalents()
246{
247 // If only one of anchor and focus is null, convert to a caret selection.
248 // FIXME: Seems like a better rule would be to convert to no selection.
249 if (m_anchor.isNull())
250 m_anchor = m_focus;
251 if (m_focus.isNull())
252 m_focus = m_anchor;
253
254 m_anchorIsFirst = m_anchor <= m_focus;
255
256 m_base = VisiblePosition(m_anchor, m_affinity).deepEquivalent();
257 if (m_anchor == m_focus)
258 m_extent = m_base;
259 else
260 m_extent = VisiblePosition(m_focus, m_affinity).deepEquivalent();
261 if (m_base.isNull() != m_extent.isNull()) {
262 if (m_base.isNull())
263 m_base = m_extent;
264 else
265 m_extent = m_base;
266 }

269void VisibleSelection::adjustSelectionRespectingGranularity(TextGranularity granularity)
270{
271 switch (granularity) {
272 case TextGranularity::CharacterGranularity:
273 // Don't do any expansion.
274 break;
275 case TextGranularity::WordGranularity: {
276 // General case: Select the word the caret is positioned inside of, or at the start of (RightWordIfOnBoundary).
277 // Edge case: If the caret is after the last word in a soft-wrapped line or the last word in
278 // the document, select that last word (LeftWordIfOnBoundary).
279 // Edge case: If the caret is after the last word in a paragraph, select from the end of the
280 // last word to the line break (also RightWordIfOnBoundary);
281 VisiblePosition start = VisiblePosition(m_start, m_affinity);
282 VisiblePosition originalEnd(m_end, m_affinity);
283 EWordSide side = RightWordIfOnBoundary;
284 if (isEndOfEditableOrNonEditableContent(start) || (isEndOfLine(start) && !isStartOfLine(start) && !isEndOfParagraph(start)))
285 side = LeftWordIfOnBoundary;
286 m_start = startOfWord(start, side).deepEquivalent();
287 side = RightWordIfOnBoundary;
288 if (isEndOfEditableOrNonEditableContent(originalEnd) || (isEndOfLine(originalEnd) && !isStartOfLine(originalEnd) && !isEndOfParagraph(originalEnd)))
289 side = LeftWordIfOnBoundary;
290
291 VisiblePosition wordEnd(endOfWord(originalEnd, side));
292 VisiblePosition end(wordEnd);
293
294 if (isEndOfParagraph(originalEnd) && !isEmptyTableCell(m_start.deprecatedNode())) {
295 // Select the paragraph break (the space from the end of a paragraph to the start of
296 // the next one) to match TextEdit.
297 end = wordEnd.next();
298
299 if (auto* table = isFirstPositionAfterTable(end)) {
300 // The paragraph break after the last paragraph in the last cell of a block table ends
301 // at the start of the paragraph after the table.
302 if (isBlock(table))
303 end = end.next(CannotCrossEditingBoundary);
304 else
305 end = wordEnd;
306 }
307
308 if (end.isNull())
309 end = wordEnd;
310
311 }
312
313 m_end = end.deepEquivalent();
314
315 // End must not be before start.
316 if (m_start.deprecatedNode() == m_end.deprecatedNode() && m_start.deprecatedEditingOffset() > m_end.deprecatedEditingOffset())
317 std::swap(m_start, m_end);
318 break;
319 }
320 case TextGranularity::SentenceGranularity:
321 m_start = startOfSentence(VisiblePosition(m_start, m_affinity)).deepEquivalent();
322 m_end = endOfSentence(VisiblePosition(m_end, m_affinity)).deepEquivalent();
323 break;
324 case TextGranularity::LineGranularity: {
325 m_start = startOfLine(VisiblePosition(m_start, m_affinity)).deepEquivalent();
326 VisiblePosition end = endOfLine(VisiblePosition(m_end, m_affinity));
327 // If the end of this line is at the end of a paragraph, include the space
328 // after the end of the line in the selection.
329 if (isEndOfParagraph(end)) {
330 VisiblePosition next = end.next();
331 if (next.isNotNull())
332 end = next;
333 }
334 m_end = end.deepEquivalent();
335 break;
336 }
337 case TextGranularity::LineBoundary:
338 m_start = startOfLine(VisiblePosition(m_start, m_affinity)).deepEquivalent();
339 m_end = endOfLine(VisiblePosition(m_end, m_affinity)).deepEquivalent();
340 break;
341 case TextGranularity::ParagraphGranularity: {
342 VisiblePosition position(m_start, m_affinity);
343 if (isStartOfLine(position) && isEndOfEditableOrNonEditableContent(position))
344 position = position.previous();
345 m_start = startOfParagraph(position).deepEquivalent();
346 VisiblePosition visibleParagraphEnd = endOfParagraph(VisiblePosition(m_end, m_affinity));
347
348 // Include the "paragraph break" (the space from the end of this paragraph to the start
349 // of the next one) in the selection.
350 VisiblePosition end(visibleParagraphEnd.next());
351
352 if (Node* table = isFirstPositionAfterTable(end)) {
353 // The paragraph break after the last paragraph in the last cell of a block table ends
354 // at the start of the paragraph after the table, not at the position just after the table.
355 if (isBlock(table))
356 end = end.next(CannotCrossEditingBoundary);
357 // There is no parargraph break after the last paragraph in the last cell of an inline table.
358 else
359 end = visibleParagraphEnd;
360 }
361
362 if (end.isNull())
363 end = visibleParagraphEnd;
364
365 m_end = end.deepEquivalent();
366 break;
367 }
368 case TextGranularity::DocumentBoundary:
369 m_start = startOfDocument(m_start.document()).deepEquivalent();
370 m_end = endOfDocument(m_end.document()).deepEquivalent();
371 break;
372 case TextGranularity::ParagraphBoundary:
373 m_start = startOfParagraph(VisiblePosition(m_start, m_affinity)).deepEquivalent();
374 m_end = endOfParagraph(VisiblePosition(m_end, m_affinity)).deepEquivalent();
375 break;
376 case TextGranularity::SentenceBoundary:
377 m_start = startOfSentence(VisiblePosition(m_start, m_affinity)).deepEquivalent();
378 m_end = endOfSentence(VisiblePosition(m_end, m_affinity)).deepEquivalent();
379 break;
380 case TextGranularity::DocumentGranularity:
381 ASSERT_NOT_REACHED();
382 break;
383 }
384
385 // Make sure we do not have a dangling start or end.
386 if (m_start.isNull())
387 m_start = m_end;
388 if (m_end.isNull())
389 m_end = m_start;
390}

392void VisibleSelection::updateSelectionType()
393{
394 if (m_start.isNull()) {
395 ASSERT(m_end.isNull());
396 m_type = Type::None;
397 m_affinity = Affinity::Downstream;
398 } else if (m_start == m_end || m_start.upstream() == m_end.upstream())
399 m_type = Type::Caret;
400 else {
401 m_type = Type::Range;
402 m_affinity = Affinity::Downstream;
403 }

406void VisibleSelection::validate(TextGranularity granularity)
407{
408 setBaseAndExtentToDeepEquivalents();
409
410 m_start = m_anchorIsFirst ? m_base : m_extent;
411 m_end = m_anchorIsFirst ? m_extent : m_base;
412
413 auto startBeforeAdjustments = m_start;
414 auto endBeforeAdjustments = m_end;
415
416 adjustSelectionRespectingGranularity(granularity);
417 adjustSelectionToAvoidCrossingShadowBoundaries();
418 adjustSelectionToAvoidCrossingEditingBoundaries();
419 updateSelectionType();
420
421 bool shouldUpdateAnchor = false; // Set to false because of <rdar://problem/69542459>. Can be returned to original logic when this problem is fully fixed.
422 bool shouldUpdateFocus = false; // Ditto.
423
424 if (isRange()) {
425 // "Constrain" the selection to be the smallest equivalent range of nodes.
426 // This is a somewhat arbitrary choice, but experience shows that it is
427 // useful to make to make the selection "canonical" (if only for
428 // purposes of comparing selections). This is an ideal point of the code
429 // to do this operation, since all selection changes that result in a RANGE
430 // come through here before anyone uses it.
431 // FIXME: Canonicalizing is good, but haven't we already done it (when we
432 // set these two positions to VisiblePosition deepEquivalent()s above)?
433 m_start = m_start.downstream();
434 m_end = m_end.upstream();
435
436 // Position::downstream() or Position::upstream() might violate editing boundaries
437 // if an anchor node has a Shadow DOM even though they should not. But because this
438 // happens in practice, adjust selection to avoid crossing editing boundaries again.
439 // See https://bugs.webkit.org/show_bug.cgi?id=87463.
440 adjustSelectionToAvoidCrossingEditingBoundaries();
441 }
442
443 if (shouldUpdateAnchor) {
444 m_anchor = m_anchorIsFirst ? m_start : m_end;
445 m_base = m_anchor;
446 }
447 if (shouldUpdateFocus) {
448 m_focus = m_anchorIsFirst ? m_end : m_start;
449 m_extent = m_focus;
450 }
451}
452
453// Because we use VisibleSelection to store values in editing commands for use when
454// undoing the command, we need to be able to create a selection that, while currently
455// invalid, will be valid once the changes are undone. This is a design problem.
456// The best fix is likely to get rid of canonicalization from VisibleSelection entirely,
457// and then remove this function.
458void VisibleSelection::setWithoutValidation(const Position& anchor, const Position& focus)
459{
460 ASSERT(anchor.isNull() == focus.isNull());
461 ASSERT(m_affinity == Affinity::Downstream);
462 m_anchor = anchor;
463 m_focus = focus;
464 m_anchorIsFirst = m_anchor <= m_focus;
465 m_base = anchor;
466 m_extent = focus;
467 m_start = m_anchorIsFirst ? anchor : focus;
468 m_end = m_anchorIsFirst ? focus : anchor;
469 m_type = anchor == focus ? Type::Caret : Type::Range;
470}

472Position VisibleSelection::adjustPositionForEnd(const Position& currentPosition, Node* startContainerNode)
473{
474 TreeScope& treeScope = startContainerNode->treeScope();
475
476 ASSERT(&currentPosition.containerNode()->treeScope() != &treeScope);
477
478 if (Node* ancestor = treeScope.ancestorNodeInThisScope(currentPosition.containerNode())) {
479 if (ancestor->contains(startContainerNode))
480 return positionAfterNode(ancestor);
481 return positionBeforeNode(ancestor);
482 }
483
484 if (Node* lastChild = treeScope.rootNode().lastChild())
485 return positionAfterNode(lastChild);
486
487 return Position();
488}
489
490Position VisibleSelection::adjustPositionForStart(const Position& currentPosition, Node* endContainerNode)
491{
492 TreeScope& treeScope = endContainerNode->treeScope();
493
494 ASSERT(&currentPosition.containerNode()->treeScope() != &treeScope);
495
496 if (Node* ancestor = treeScope.ancestorNodeInThisScope(currentPosition.containerNode())) {
497 if (ancestor->contains(endContainerNode))
498 return positionBeforeNode(ancestor);
499 return positionAfterNode(ancestor);
500 }
501
502 if (Node* firstChild = treeScope.rootNode().firstChild())
503 return positionBeforeNode(firstChild);
504
505 return Position();
506}

508static bool isInUserAgentShadowRootOrHasEditableShadowAncestor(Node& node)
509{
510 auto* shadowRoot = node.containingShadowRoot();
511 if (!shadowRoot)
512 return false;
513
514 if (shadowRoot->mode() == ShadowRootMode::UserAgent)
515 return true;
516
517 for (RefPtr<Node> currentNode = &node; currentNode; currentNode = currentNode->parentOrShadowHostNode()) {
518 if (currentNode->hasEditableStyle())
519 return true;
520 }
521 return false;
522}

524void VisibleSelection::adjustSelectionToAvoidCrossingShadowBoundaries()
525{
526 if (m_start.isNull() || m_end.isNull())
527 return;
528
529 auto startNode = makeRef(*m_start.anchorNode());
530 auto endNode = makeRef(*m_end.anchorNode());
531 if (&startNode->treeScope() == &endNode->treeScope())
532 return;
533
534 if (startNode->document().settings().selectionAcrossShadowBoundariesEnabled()) {
535 if (!isInUserAgentShadowRootOrHasEditableShadowAncestor(startNode)
536 && !isInUserAgentShadowRootOrHasEditableShadowAncestor(endNode))
537 return;
538 }
539
540 // Correct the focus if necessary.
541 if (m_anchorIsFirst) {
542 m_extent = adjustPositionForEnd(m_end, m_start.containerNode());
543 m_end = m_extent;
544 } else {
545 m_extent = adjustPositionForStart(m_start, m_end.containerNode());
546 m_start = m_extent;
547 }
548 m_focus = m_extent;
549}
550
551void VisibleSelection::adjustSelectionToAvoidCrossingEditingBoundaries()
552{
553 if (m_start.isNull() || m_end.isNull())
554 return;
555
556 // Early return in the caret case (the state hasn't actually been set yet, so we can't use isCaret()) to avoid the
557 // expense of computing highestEditableRoot.
558 if (m_base == m_start && m_base == m_end)
559 return;
560
561 auto* baseRoot = highestEditableRoot(m_base);
562 auto* startRoot = highestEditableRoot(m_start);
563 auto* endRoot = highestEditableRoot(m_end);
564
565 auto* baseEditableAncestor = lowestEditableAncestor(m_base.containerNode());
566
567 // The base, start and end are all in the same region. No adjustment necessary.
568 if (baseRoot == startRoot && baseRoot == endRoot)
569 return;
570
571 // The selection is based in editable content.
572 if (baseRoot) {
573 // If the start is outside the base's editable root, cap it at the start of that root.
574 // If the start is in non-editable content that is inside the base's editable root, put it
575 // at the first editable position after start inside the base's editable root.
576 if (startRoot != baseRoot) {
577 VisiblePosition first = firstEditablePositionAfterPositionInRoot(m_start, baseRoot);
578 m_start = first.deepEquivalent();
579 if (m_start.isNull()) {
580 ASSERT_NOT_REACHED();
581 m_start = m_end;
582 }
583 }
584 // If the end is outside the base's editable root, cap it at the end of that root.
585 // If the end is in non-editable content that is inside the base's root, put it
586 // at the last editable position before the end inside the base's root.
587 if (endRoot != baseRoot) {
588 VisiblePosition last = lastEditablePositionBeforePositionInRoot(m_end, baseRoot);
589 m_end = last.deepEquivalent();
590 if (m_end.isNull())
591 m_end = m_start;
592 }
593 // The selection is based in non-editable content.
594 } else {
595 // FIXME: Non-editable pieces inside editable content should be atomic, in the same way that editable
596 // pieces in non-editable content are atomic.
597
598 // The selection ends in editable content or non-editable content inside a different editable ancestor,
599 // move backward until non-editable content inside the same lowest editable ancestor is reached.
600 auto* endEditableAncestor = lowestEditableAncestor(m_end.containerNode());
601 if (endRoot || endEditableAncestor != baseEditableAncestor) {
602
603 Position p = previousVisuallyDistinctCandidate(m_end);
604 Node* shadowAncestor = endRoot ? endRoot->shadowHost() : 0;
605 if (p.isNull() && shadowAncestor)
606 p = positionAfterNode(shadowAncestor);
607 while (p.isNotNull() && !(lowestEditableAncestor(p.containerNode()) == baseEditableAncestor && !isEditablePosition(p))) {
608 Node* root = editableRootForPosition(p);
609 shadowAncestor = root ? root->shadowHost() : 0;
610 p = isAtomicNode(p.containerNode()) ? positionInParentBeforeNode(p.containerNode()) : previousVisuallyDistinctCandidate(p);
611 if (p.isNull() && shadowAncestor)
612 p = positionAfterNode(shadowAncestor);
613 }
614 VisiblePosition previous(p);
615
616 if (previous.isNull()) {
617 *this = { };
618 return;
619 }
620 m_end = previous.deepEquivalent();
621 }
622
623 // The selection starts in editable content or non-editable content inside a different editable ancestor,
624 // move forward until non-editable content inside the same lowest editable ancestor is reached.
625 auto* startEditableAncestor = lowestEditableAncestor(m_start.containerNode());
626 if (startRoot || startEditableAncestor != baseEditableAncestor) {
627 Position p = nextVisuallyDistinctCandidate(m_start);
628 Node* shadowAncestor = startRoot ? startRoot->shadowHost() : 0;
629 if (p.isNull() && shadowAncestor)
630 p = positionBeforeNode(shadowAncestor);
631 while (p.isNotNull() && !(lowestEditableAncestor(p.containerNode()) == baseEditableAncestor && !isEditablePosition(p))) {
632 Node* root = editableRootForPosition(p);
633 shadowAncestor = root ? root->shadowHost() : 0;
634 p = isAtomicNode(p.containerNode()) ? positionInParentAfterNode(p.containerNode()) : nextVisuallyDistinctCandidate(p);
635 if (p.isNull() && shadowAncestor)
636 p = positionBeforeNode(shadowAncestor);
637 }
638 VisiblePosition next(p);
639
640 if (next.isNull()) {
641 *this = { };
642 return;
643 }
644 m_start = next.deepEquivalent();
645 }
646 }
647
648 // Correct the focus if necessary.
649 if (baseEditableAncestor != lowestEditableAncestor(m_extent.containerNode())) {
650 m_extent = m_anchorIsFirst ? m_end : m_start;
651 m_focus = m_extent;
