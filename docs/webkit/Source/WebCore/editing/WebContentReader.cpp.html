34void WebContentReader::addFragment(Ref<DocumentFragment>&& newFragment)
35{
36 if (!fragment)
37 fragment = WTFMove(newFragment);
38 else
39 fragment->appendChild(newFragment.get());
40}
41
42bool FrameWebContentReader::shouldSanitize() const
43{
44 ASSERT(frame.document());
45 return frame.document()->originIdentifierForPasteboard() != contentOrigin;
46}
47
48MSOListQuirks FrameWebContentReader::msoListQuirksForMarkup() const
49{
50 return contentOrigin.isNull() ? MSOListQuirks::CheckIfNeeded : MSOListQuirks::Disabled;
51}

53#if PLATFORM(COCOA) || PLATFORM(GTK)
54bool WebContentReader::readFilePaths(const Vector<String>& paths)
55{
56 if (paths.isEmpty() || !frame.document())
57 return false;
58
59 for (auto& path : paths)
60 readFilePath(path);
61
62 return true;
63}
