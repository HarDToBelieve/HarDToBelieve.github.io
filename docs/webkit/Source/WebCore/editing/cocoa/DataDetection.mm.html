75static std::optional<DetectedItem> detectItem(const VisiblePosition& position, const SimpleRange& contextRange)
76{
77 if (position.isNull())
78 return { };
79 String fullPlainTextString = plainText(contextRange);
80 CFIndex hitLocation = characterCount(*makeSimpleRange(contextRange.start, position));
81
82 auto scanner = adoptCF(DDScannerCreate(DDScannerTypeStandard, 0, nullptr));
83 auto scanQuery = adoptCF(DDScanQueryCreateFromString(kCFAllocatorDefault, fullPlainTextString.createCFString().get(), CFRangeMake(0, fullPlainTextString.length())));
84
85 if (!DDScannerScanQuery(scanner.get(), scanQuery.get()))
86 return { };
87
88 auto results = adoptCF(DDScannerCopyResultsWithOptions(scanner.get(), DDScannerCopyResultsOptionsNoOverlap));
89
90 // Find the DDResultRef that intersects the hitTestResult's VisiblePosition.
91 DDResultRef mainResult = nullptr;
92 std::optional<SimpleRange> mainResultRange;
93 CFIndex resultCount = CFArrayGetCount(results.get());
94 for (CFIndex i = 0; i < resultCount; i++) {
95 auto result = checked_cf_cast<DDResultRef>(CFArrayGetValueAtIndex(results.get(), i));
96 CFRange resultRangeInContext = DDResultGetRange(result);
97 if (hitLocation >= resultRangeInContext.location && (hitLocation - resultRangeInContext.location) < resultRangeInContext.length) {
98 mainResult = result;
99 mainResultRange = resolveCharacterRange(contextRange, resultRangeInContext);
100 break;
101 }
102 }
103
104 if (!mainResult)
105 return { };
106
107 auto view = mainResultRange->start.document().view();
108 if (!view)
109 return { };
110
111 auto actionContext = adoptNS([PAL::allocDDActionContextInstance() init]);
112 [actionContext setAllResults:@[ (__bridge id)mainResult ]];
113 [actionContext setMainResult:mainResult];
114
115 return { {
116 WTFMove(actionContext),
117 view->contentsToWindow(enclosingIntRect(unitedBoundingBoxes(RenderObject::absoluteTextQuads(*mainResultRange)))),
118 *mainResultRange,
119 } };
120}

122std::optional<DetectedItem> DataDetection::detectItemAroundHitTestResult(const HitTestResult& hitTestResult)
123{
124 if (!PAL::isDataDetectorsFrameworkAvailable())
125 return { };
126
127 Node* node = hitTestResult.innerNonSharedNode();
128 if (!node)
129 return { };
130 auto renderer = node->renderer();
131 if (!renderer)
132 return { };
133
134 VisiblePosition position;
135 std::optional<SimpleRange> contextRange;
136
137 if (!is<HTMLTextFormControlElement>(*node)) {
138 position = renderer->positionForPoint(hitTestResult.localPoint(), nullptr);
139 if (position.isNull())
140 position = firstPositionInOrBeforeNode(node);
141
142 contextRange = rangeExpandedAroundPositionByCharacters(position, 250);
143 } else {
144 Frame* frame = node->document().frame();
145 if (!frame)
146 return { };
147
148 IntPoint framePoint = hitTestResult.roundedPointInInnerNodeFrame();
149 if (!frame->rangeForPoint(framePoint))
150 return { };
151
152 position = frame->visiblePositionForPoint(framePoint);
153 if (position.isNull())
154 return { };
155
156 contextRange = enclosingTextUnitOfGranularity(position, TextGranularity::LineGranularity, SelectionDirection::Forward);
157 }
158
159 if (!contextRange)
160 return { };
161
162 return detectItem(position, *contextRange);
163}

192bool DataDetection::canPresentDataDetectorsUIForElement(Element& element)
193{
194 if (!isDataDetectorLink(element))
195 return false;
196
197 if (PAL::softLink_DataDetectorsCore_DDShouldImmediatelyShowActionSheetForURL(downcast<HTMLAnchorElement>(element).href()))
198 return true;
199
200 auto& resultAttribute = element.attributeWithoutSynchronization(x_apple_data_detectors_resultAttr);
201 if (resultAttribute.isEmpty())
202 return false;
203
204 auto* dataDetectionResults = element.document().frame()->dataDetectionResultsIfExists();
205 if (!dataDetectionResults)
206 return false;
207
208 NSArray *results = dataDetectionResults->documentLevelResults();
209 if (!results)
210 return false;
211
212 auto resultIndices = StringView { resultAttribute }.split('/');
213 auto indexIterator = resultIndices.begin();
214 auto result = [results[parseIntegerAllowingTrailingJunk<int>(*indexIterator).value_or(0)] coreResult];
215
216 // Handle the case of a signature block, where we need to follow the path down one or more subresult levels.
217 while (++indexIterator != resultIndices.end()) {
218 results = (__bridge NSArray *)PAL::softLink_DataDetectorsCore_DDResultGetSubResults(result);
219 result = (__bridge DDResultRef)results[parseIntegerAllowingTrailingJunk<int>(*indexIterator).value_or(0)];
220 }
221
222 return PAL::softLink_DataDetectorsCore_DDShouldImmediatelyShowActionSheetForResult(result);
223}

225static BOOL resultIsURL(DDResultRef result)
226{
227 if (!result)
228 return NO;
229
230 static NeverDestroyed<RetainPtr<NSSet>> urlTypes = [NSSet setWithObjects:
231 (NSString *)PAL::get_DataDetectorsCore_DDBinderHttpURLKey(),
232 (NSString *)PAL::get_DataDetectorsCore_DDBinderWebURLKey(),
233 (NSString *)PAL::get_DataDetectorsCore_DDBinderMailURLKey(),
234 (NSString *)PAL::get_DataDetectorsCore_DDBinderGenericURLKey(),
235 (NSString *)PAL::get_DataDetectorsCore_DDBinderEmailKey(), nil];
236 return [urlTypes.get() containsObject:(NSString *)PAL::softLink_DataDetectorsCore_DDResultGetType(result)];
237}

239static NSString *constructURLStringForResult(DDResultRef currentResult, NSString *resultIdentifier, NSDate *referenceDate, NSTimeZone *referenceTimeZone, OptionSet<DataDetectorType> detectionTypes)
240{
241 if (!PAL::softLink_DataDetectorsCore_DDResultHasProperties(currentResult, DDResultPropertyPassiveDisplay))
242 return nil;
243
244 auto phoneTypes = detectionTypes.contains(DataDetectorType::PhoneNumber) ? DDURLifierPhoneNumberDetectionRegular : DDURLifierPhoneNumberDetectionNone;
245 auto category = PAL::softLink_DataDetectorsCore_DDResultGetCategory(currentResult);
246 auto type = PAL::softLink_DataDetectorsCore_DDResultGetType(currentResult);
247
248 if ((detectionTypes.contains(DataDetectorType::Address) && DDResultCategoryAddress == category)
249 || (detectionTypes.contains(DataDetectorType::TrackingNumber) && CFEqual(PAL::get_DataDetectorsCore_DDBinderTrackingNumberKey(), type))
250 || (detectionTypes.contains(DataDetectorType::FlightNumber) && CFEqual(PAL::get_DataDetectorsCore_DDBinderFlightInformationKey(), type))
251 || (detectionTypes.contains(DataDetectorType::LookupSuggestion) && CFEqual(PAL::get_DataDetectorsCore_DDBinderParsecSourceKey(), type))
252 || (detectionTypes.contains(DataDetectorType::PhoneNumber) && DDResultCategoryPhoneNumber == category)
253 || (detectionTypes.contains(DataDetectorType::Link) && resultIsURL(currentResult))) {
254 return PAL::softLink_DataDetectorsCore_DDURLStringForResult(currentResult, resultIdentifier, phoneTypes, referenceDate, referenceTimeZone);
255 }
256 if (detectionTypes.contains(DataDetectorType::CalendarEvent) && DDResultCategoryCalendarEvent == category) {
257 if (!PAL::softLink_DataDetectorsCore_DDResultIsPastDate(currentResult, (CFDateRef)referenceDate, (CFTimeZoneRef)referenceTimeZone))
258 return PAL::softLink_DataDetectorsCore_DDURLStringForResult(currentResult, resultIdentifier, phoneTypes, referenceDate, referenceTimeZone);
259 }
260 return nil;
261}

263static void removeResultLinksFromAnchor(Element& element)
264{
265 // Perform a depth-first search for anchor nodes, which have the data detectors attribute set to true,
266 // take their children and insert them before the anchor, and then remove the anchor.
267
268 // Note that this is not using ElementChildIterator because we potentially prepend children as we iterate over them.
269 for (auto* child = ElementTraversal::firstChild(element); child; child = ElementTraversal::nextSibling(*child))
270 removeResultLinksFromAnchor(*child);
271
272 auto* elementParent = element.parentElement();
273 if (!elementParent)
274 return;
275
276 bool elementIsDDAnchor = is<HTMLAnchorElement>(element) && equalIgnoringASCIICase(element.attributeWithoutSynchronization(x_apple_data_detectorsAttr), "true");
277 if (!elementIsDDAnchor)
278 return;
279
280 // Iterate over the children and move them all onto the same level as this anchor. Remove the anchor afterwards.
281 while (auto* child = element.firstChild())
282 elementParent->insertBefore(*child, &element);
283
284 elementParent->removeChild(element);
285}

287static bool searchForLinkRemovingExistingDDLinks(Node& startNode, Node& endNode)
288{
289 for (Node* node = &startNode; node; node = NodeTraversal::next(*node)) {
290 if (is<HTMLAnchorElement>(*node)) {
291 auto& anchor = downcast<HTMLAnchorElement>(*node);
292 if (!equalIgnoringASCIICase(anchor.attributeWithoutSynchronization(x_apple_data_detectorsAttr), "true"))
293 return true;
294 removeResultLinksFromAnchor(anchor);
295 }
296
297 if (node == &endNode) {
298 // If we found the end node and no link, return false unless an ancestor node is a link.
299 // The only ancestors not tested at this point are in the direct line from self's parent to the top.
300 for (auto& anchor : ancestorsOfType<HTMLAnchorElement>(startNode)) {
301 if (!equalIgnoringASCIICase(anchor.attributeWithoutSynchronization(x_apple_data_detectorsAttr), "true"))
302 return true;
303 removeResultLinksFromAnchor(anchor);
304 }
305 return false;
306 }
307 }
308 return false;
309}

329static String dataDetectorStringForPath(NSIndexPath *path)
330{
331 auto length = path.length;
332 switch (length) {
333 case 0:
334 return { };
335 case 1:
336 return makeString([path indexAtPosition:0]);
337 case 2:
338 return makeString([path indexAtPosition:0], '/', [path indexAtPosition:1]);
339 default:
340 StringBuilder builder;
341 builder.append([path indexAtPosition:0]);
342 for (NSUInteger i = 1; i < length; i++)
343 builder.append('/', [path indexAtPosition:i]);
344 return builder.toString();
345 }
346}

348static void buildQuery(DDScanQueryRef scanQuery, const SimpleRange& contextRange)
349{
350 // Once we're over this number of fragments, stop at the first hard break.
351 const CFIndex maxFragmentWithHardBreak = 1000;
352 // Once we're over this number of fragments, we stop at the line.
353 const CFIndex maxFragmentWithLinebreak = 5000;
354 // Once we're over this number of fragments, we stop at the space.
355 const CFIndex maxFragmentSpace = 10000;
356
357 CFCharacterSetRef whiteSpacesSet = CFCharacterSetGetPredefined(kCFCharacterSetWhitespaceAndNewline);
358 CFCharacterSetRef newLinesSet = CFCharacterSetGetPredefined(kCFCharacterSetNewline);
359
360 CFIndex iteratorCount = 0;
361 CFIndex fragmentCount = 0;
362
363 // Build the scan query adding separators.
364 // For each fragment the iterator increment is stored as metadata.
365 for (TextIterator iterator(contextRange); !iterator.atEnd(); iterator.advance(), iteratorCount++) {
366 StringView currentText = iterator.text();
367 size_t currentTextLength = currentText.length();
368 if (!currentTextLength) {
369 PAL::softLink_DataDetectorsCore_DDScanQueryAddSeparator(scanQuery, DDTextCoalescingTypeHardBreak);
370 if (iteratorCount > maxFragmentWithHardBreak)
371 break;
372 continue;
373 }
374 // Test for white space nodes, we're coalescing them.
375 auto currentTextUpconvertedCharacters = currentText.upconvertedCharacters();
376 auto currentCharPtr = currentTextUpconvertedCharacters.get();
377
378 bool containsOnlyWhiteSpace = true;
379 bool hasTab = false;
380 bool hasNewline = false;
381 int nbspCount = 0;
382 for (NSUInteger i = 0; i < currentTextLength; i++) {
383 if (!CFCharacterSetIsCharacterMember(whiteSpacesSet, *currentCharPtr)) {
384 containsOnlyWhiteSpace = false;
385 break;
386 }
387
388 if (CFCharacterSetIsCharacterMember(newLinesSet, *currentCharPtr))
389 hasNewline = true;
390 else if (*currentCharPtr == '\t')
391 hasTab = true;
392
393 // Multiple consecutive non breakable spaces are most likely simulated tabs.
394 if (*currentCharPtr == 0xa0) {
395 if (++nbspCount > 2)
396 hasTab = true;
397 } else
398 nbspCount = 0;
399
400 currentCharPtr++;
401 }
402 if (containsOnlyWhiteSpace) {
403 if (hasNewline) {
404 PAL::softLink_DataDetectorsCore_DDScanQueryAddLineBreak(scanQuery);
405 if (iteratorCount > maxFragmentWithLinebreak)
406 break;
407 } else {
408 PAL::softLink_DataDetectorsCore_DDScanQueryAddSeparator(scanQuery, hasTab ? DDTextCoalescingTypeTab : DDTextCoalescingTypeSpace);
409 if (iteratorCount > maxFragmentSpace)
410 break;
411 }
412 continue;
413 }
414
415 auto currentTextCFString = adoptCF(CFStringCreateWithCharacters(kCFAllocatorDefault, reinterpret_cast<const UniChar*>(currentTextUpconvertedCharacters.get()), currentTextLength));
416 PAL::softLink_DataDetectorsCore_DDScanQueryAddTextFragment(scanQuery, currentTextCFString.get(), CFRangeMake(0, currentTextLength), (void *)iteratorCount, (DDTextFragmentMode)0, DDTextCoalescingTypeNone);
417 fragmentCount++;
418 }
419}

421static inline CFComparisonResult queryOffsetCompare(DDQueryOffset o1, DDQueryOffset o2)
422{
423 if (o1.queryIndex < o2.queryIndex)
424 return kCFCompareLessThan;
425 if (o1.queryIndex > o2.queryIndex)
426 return kCFCompareGreaterThan;
427 if (o1.offset < o2.offset)
428 return kCFCompareLessThan;
429 if (o1.offset > o2.offset)
430 return kCFCompareGreaterThan;
431 return kCFCompareEqualTo;
432}

434void DataDetection::removeDataDetectedLinksInDocument(Document& document)
435{
436 Vector<Ref<HTMLAnchorElement>> allAnchorElements;
437 for (auto& anchor : descendantsOfType<HTMLAnchorElement>(document))
438 allAnchorElements.append(anchor);
439

441 removeResultLinksFromAnchor(anchor.get());

444NSArray *DataDetection::detectContentInRange(const SimpleRange& contextRange, OptionSet<DataDetectorType> types, NSDictionary *context)
445{
446 auto scanner = adoptCF(PAL::softLink_DataDetectorsCore_DDScannerCreate(DDScannerTypeStandard, 0, nullptr));
447 auto scanQuery = adoptCF(PAL::softLink_DataDetectorsCore_DDScanQueryCreate(NULL));
448 buildQuery(scanQuery.get(), contextRange);
449
450 if (types.contains(DataDetectorType::LookupSuggestion))
451 PAL::softLink_DataDetectorsCore_DDScannerEnableOptionalSource(scanner.get(), DDScannerSourceSpotlight, true);
452
453 // FIXME: we should add a timeout to this call to make sure it doesn't take too much time.
454 if (!PAL::softLink_DataDetectorsCore_DDScannerScanQuery(scanner.get(), scanQuery.get()))
455 return nil;
456
457 auto scannerResults = adoptCF(PAL::softLink_DataDetectorsCore_DDScannerCopyResultsWithOptions(scanner.get(), PAL::get_DataDetectorsCore_DDScannerCopyResultsOptionsForPassiveUse() | DDScannerCopyResultsOptionsCoalesceSignatures));
458 if (!scannerResults)
459 return nil;
460
461 if (!CFArrayGetCount(scannerResults.get()))
462 return nil;
463
464 Vector<RetainPtr<DDResultRef>> allResults;
465 Vector<RetainPtr<NSIndexPath>> indexPaths;
466 NSInteger currentTopLevelIndex = 0;
467
468 // Iterate through the scanner results to find signatures and extract all the subresults while
469 // populating the array of index paths to use in the href of the anchors being created.
470 for (id resultObject in (NSArray *)scannerResults.get()) {
471 DDResultRef result = (DDResultRef)resultObject;
472 NSIndexPath *indexPath = [NSIndexPath indexPathWithIndex:currentTopLevelIndex];
473 if (CFEqual(PAL::softLink_DataDetectorsCore_DDResultGetType(result), PAL::get_DataDetectorsCore_DDBinderSignatureBlockKey())) {
474 NSArray *subresults = (NSArray *)PAL::softLink_DataDetectorsCore_DDResultGetSubResults(result);
475
476 for (NSUInteger subResultIndex = 0 ; subResultIndex < [subresults count] ; subResultIndex++) {
477 indexPaths.append([indexPath indexPathByAddingIndex:subResultIndex]);
478 allResults.append((DDResultRef)[subresults objectAtIndex:subResultIndex]);
479 }
480 } else {
481 allResults.append(result);
482 indexPaths.append(indexPath);
483 }
484 currentTopLevelIndex++;
485 }
486
487 Vector<Vector<SimpleRange>> allResultRanges;
488 TextIterator iterator(contextRange);
489 CFIndex iteratorCount = 0;
490
491 // Iterate through the array of the expanded results to create a vector of Range objects that indicate
492 // where the DOM needs to be modified.
493 // Each result can be contained all in one text node or can span multiple text nodes.
494 for (auto& result : allResults) {
495 DDQueryRange queryRange = PAL::softLink_DataDetectorsCore_DDResultGetQueryRangeForURLification(result.get());
496 CFIndex iteratorTargetAdvanceCount = (CFIndex)PAL::softLink_DataDetectorsCore_DDScanQueryGetFragmentMetaData(scanQuery.get(), queryRange.start.queryIndex);
497 for (; iteratorCount < iteratorTargetAdvanceCount; ++iteratorCount)
498 iterator.advance();
499
500 Vector<SimpleRange> fragmentRanges;
501 CFIndex fragmentIndex = queryRange.start.queryIndex;
502 if (fragmentIndex == queryRange.end.queryIndex) {
503 CharacterRange fragmentRange;
504 fragmentRange.location = queryRange.start.offset;
505 fragmentRange.length = queryRange.end.offset - queryRange.start.offset;
506 fragmentRanges.append(resolveCharacterRange(iterator.range(), fragmentRange));
507 } else {
508 auto range = iterator.range();
509 range.start.offset += queryRange.start.offset;
510 fragmentRanges.append(range);
511 }
512
513 while (fragmentIndex < queryRange.end.queryIndex) {
514 ++fragmentIndex;
515 iteratorTargetAdvanceCount = (CFIndex)PAL::softLink_DataDetectorsCore_DDScanQueryGetFragmentMetaData(scanQuery.get(), fragmentIndex);
516 for (; iteratorCount < iteratorTargetAdvanceCount; ++iteratorCount)
517 iterator.advance();
518
519 auto fragmentRange = iterator.range();
520 if (fragmentIndex == queryRange.end.queryIndex)
521 fragmentRange.end.offset = fragmentRange.start.offset + queryRange.end.offset;
522 auto& previousRange = fragmentRanges.last();
523 if (previousRange.start.container.ptr() == fragmentRange.start.container.ptr())
524 previousRange.end = fragmentRange.end;
525 else
526 fragmentRanges.append(fragmentRange);
527 }
528 allResultRanges.append(WTFMove(fragmentRanges));
529 }
530
531 auto tz = adoptCF(CFTimeZoneCopyDefault());
532 NSDate *referenceDate = [context objectForKey:getkDataDetectorsReferenceDateKey()] ?: [NSDate date];
533 RefPtr<Text> lastTextNodeToUpdate;
534 String lastNodeContent;
535 unsigned contentOffset = 0;
536 DDQueryOffset lastModifiedQueryOffset = { -1, 0 };
537
538 // For each result add the link.
539 // Since there could be multiple results in the same text node, the node is only modified when
540 // we are about to process a different text node.
541 CFIndex resultCount = allResults.size();
542 for (CFIndex resultIndex = 0; resultIndex < resultCount; ++resultIndex) {
543 DDResultRef coreResult = allResults[resultIndex].get();
544 DDQueryRange queryRange = PAL::softLink_DataDetectorsCore_DDResultGetQueryRangeForURLification(coreResult);
545 auto& resultRanges = allResultRanges[resultIndex];
546
547 // Compare the query offsets to make sure we don't go backwards
548 if (queryOffsetCompare(lastModifiedQueryOffset, queryRange.start) >= 0)
549 continue;
550
551 if (resultRanges.isEmpty())
552 continue;
553
554 NSString *identifier = dataDetectorStringForPath(indexPaths[resultIndex].get());
555 NSString *correspondingURL = constructURLStringForResult(coreResult, identifier, referenceDate, (NSTimeZone *)tz.get(), types);
556
557 if (!correspondingURL || searchForLinkRemovingExistingDDLinks(resultRanges.first().start.container, resultRanges.last().end.container))
558 continue;
559
560 lastModifiedQueryOffset = queryRange.end;
561 BOOL shouldUseLightLinks = PAL::softLink_DataDetectorsCore_DDShouldUseLightLinksForResult(coreResult, [indexPaths[resultIndex] length] > 1);
562
563 for (auto& range : resultRanges) {
564 auto* parentNode = range.start.container->parentNode();
565 if (!parentNode)
566 continue;
567
568 if (!is<Text>(range.start.container))
569 continue;
570
571 auto& currentTextNode = downcast<Text>(range.start.container.get());
572 Document& document = currentTextNode.document();
573 String textNodeData;
574
575 if (lastTextNodeToUpdate != &currentTextNode) {
576 if (lastTextNodeToUpdate)
577 lastTextNodeToUpdate->setData(lastNodeContent);
578 contentOffset = 0;
579 if (range.start.offset > 0)
580 textNodeData = currentTextNode.data().substring(0, range.start.offset);
581 } else
582 textNodeData = currentTextNode.data().substring(contentOffset, range.start.offset - contentOffset);
583
584 if (!textNodeData.isEmpty()) {
585 parentNode->insertBefore(Text::create(document, textNodeData), &currentTextNode);
586 contentOffset = range.start.offset;
587 }
588
589 // Create the actual anchor node and insert it before the current node.
590 textNodeData = currentTextNode.data().substring(range.start.offset, range.end.offset - range.start.offset);
591 auto newTextNode = Text::create(document, textNodeData);
592 parentNode->insertBefore(newTextNode.copyRef(), &currentTextNode);
593
594 Ref<HTMLAnchorElement> anchorElement = HTMLAnchorElement::create(document);
595 anchorElement->setHref(correspondingURL);
596 anchorElement->setDir("ltr");
597
598 if (shouldUseLightLinks) {
599 document.updateStyleIfNeeded();
600
601 auto* renderStyle = parentNode->computedStyle();
602 if (renderStyle) {
603 auto textColor = renderStyle->visitedDependentColor(CSSPropertyColor);
604 if (textColor.isValid()) {
605 // FIXME: Consider using LCHA<float> rather than HSLA<float> for better perceptual results and to avoid clamping to sRGB gamut, which is what HSLA does.
606 auto hsla = textColor.toColorTypeLossy<HSLA<float>>();
607
608 // Force the lightness of the underline color to the middle, and multiply the alpha by 38%,
609 // so the color will appear on light and dark backgrounds, since only one color can be specified.
610 hsla.lightness = 50.0f;
611 hsla.alpha *= 0.38f;
612
613 // FIXME: Consider keeping color in LCHA (if that change is made) or converting back to the initial underlying color type to avoid unnecessarily clamping colors outside of sRGB.
614 auto underlineColor = convertColor<SRGBA<uint8_t>>(hsla);
615
616 anchorElement->setInlineStyleProperty(CSSPropertyColor, CSSValueCurrentcolor);
617 anchorElement->setInlineStyleProperty(CSSPropertyTextDecorationColor, serializationForCSS(underlineColor));
618 }
619 }
620 }

622 anchorElement->appendChild(WTFMove(newTextNode));
623
624 // Add a special attribute to mark this URLification as the result of data detectors.
625 anchorElement->setAttributeWithoutSynchronization(x_apple_data_detectorsAttr, AtomString("true", AtomString::ConstructFromLiteral));
626 anchorElement->setAttributeWithoutSynchronization(x_apple_data_detectors_typeAttr, dataDetectorTypeForCategory(PAL::softLink_DataDetectorsCore_DDResultGetCategory(coreResult)));
627 anchorElement->setAttributeWithoutSynchronization(x_apple_data_detectors_resultAttr, identifier);
628
629 parentNode->insertBefore(WTFMove(anchorElement), &currentTextNode);
630
631 contentOffset = range.end.offset;
632
633 lastNodeContent = currentTextNode.data().substring(range.end.offset, currentTextNode.length() - range.end.offset);
634 lastTextNodeToUpdate = &currentTextNode;
635 }
636 }
637
638 if (lastTextNodeToUpdate)
639 lastTextNodeToUpdate->setData(lastNodeContent);
640
641 return [PAL::getDDScannerResultClass() resultsFromCoreResults:scannerResults.get()];
642}
