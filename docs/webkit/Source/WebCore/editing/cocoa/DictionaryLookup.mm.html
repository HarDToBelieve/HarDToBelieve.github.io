111- (void)revealContext:(RVPresentingContext *)context drawRectsForItem:(RVItem *)item
112{
113 UNUSED_PARAM(item);
114
115 for (NSValue *rectVal in context.itemRectsInView) {
116 NSRect rect = rectVal.rectValue;
117
118 // Get current font attributes from the attributed string above, and add paragraph style attribute in order to center text.
119 auto attributes = adoptNS([[NSMutableDictionary alloc] initWithDictionary:[self.attributedString fontAttributesInRange:NSMakeRange(0, [self.attributedString length])]]);
120 auto paragraph = adoptNS([[NSMutableParagraphStyle alloc] init]);
121 [paragraph setAlignment:NSTextAlignmentCenter];
122 [attributes setObject:paragraph.get() forKey:NSParagraphStyleAttributeName];
123
124 auto string = adoptNS([[NSAttributedString alloc] initWithString:[self.attributedString string] attributes:attributes.get()]);
125 [string drawInRect:rect];
126 }
127}

219- (void)drawHighlightContentForItem:(RVItem *)item context:(CGContextRef)context
220{
221 NSArray <NSValue *> *rects = [self highlightRectsForItem:item];
222 if (!rects.count)
223 return;
224
225 CGRect highlightRect = rects.firstObject.CGRectValue;
226 for (NSValue *rect in rects)
227 highlightRect = CGRectUnion(highlightRect, rect.CGRectValue);
228 highlightRect = [_view convertRect:highlightRect fromView:nil];
229
230 WebCore::CGContextStateSaver saveState(context);
231 CGAffineTransform contextTransform = CGContextGetCTM(context);
232 CGFloat backingScale = contextTransform.a;
233 CGFloat macCatalystScaleFactor = [PAL::getUIApplicationClass() sharedApplication]._iOSMacScale;
234 CGAffineTransform transform = CGAffineTransformMakeScale(macCatalystScaleFactor * backingScale, macCatalystScaleFactor * backingScale);
235 CGContextSetCTM(context, transform);
236
237 for (NSValue *v in rects) {
238 CGRect imgSrcRect = [_view convertRect:v.CGRectValue fromView:nil];
239 auto nativeImage = _image->nativeImage();
240 CGRect origin = CGRectMake(imgSrcRect.origin.x - highlightRect.origin.x, imgSrcRect.origin.y - highlightRect.origin.y, highlightRect.size.width, highlightRect.size.height);
241 CGContextDrawImage(context, origin, nativeImage->platformImage().get());
242 }
243}

255static bool canCreateRevealItems()
256{
257 static bool result;
258 static std::once_flag onceFlag;
259 std::call_once(onceFlag, [&] {
260 result = PAL::isRevealFrameworkAvailable() && PAL::isRevealCoreFrameworkAvailable() && PAL::getRVItemClass();
261 });
262 return result;
263}

265std::optional<std::tuple<SimpleRange, NSDictionary *>> DictionaryLookup::rangeForSelection(const VisibleSelection& selection)
266{
267 BEGIN_BLOCK_OBJC_EXCEPTIONS
268
269 if (!canCreateRevealItems())
270 return std::nullopt;
271
272 // Since we already have the range we want, we just need to grab the returned options.
273 auto selectionStart = selection.visibleStart();
274 auto selectionEnd = selection.visibleEnd();
275
276 // As context, we are going to use the surrounding paragraphs of text.
277 auto paragraphStart = startOfParagraph(selectionStart);
278 auto paragraphEnd = endOfParagraph(selectionEnd);
279 if (paragraphStart.isNull() || paragraphEnd.isNull())
280 return std::nullopt;
281
282 auto lengthToSelectionStart = characterCount(*makeSimpleRange(paragraphStart, selectionStart));
283 auto selectionCharacterCount = characterCount(*makeSimpleRange(selectionStart, selectionEnd));
284 NSRange rangeToPass = NSMakeRange(lengthToSelectionStart, selectionCharacterCount);
285
286 auto fullCharacterRange = *makeSimpleRange(paragraphStart, paragraphEnd);
287 String itemString = plainText(fullCharacterRange);
288 NSRange highlightRange = adoptNS([PAL::allocRVItemInstance() initWithText:itemString selectedRange:rangeToPass]).get().highlightRange;
289
290 return { { resolveCharacterRange(fullCharacterRange, highlightRange), nil } };
291
292 END_BLOCK_OBJC_EXCEPTIONS
293
294 return std::nullopt;
295}

297std::optional<std::tuple<SimpleRange, NSDictionary *>> DictionaryLookup::rangeAtHitTestResult(const HitTestResult& hitTestResult)
298{
299 BEGIN_BLOCK_OBJC_EXCEPTIONS
300
301 if (!canCreateRevealItems())
302 return std::nullopt;
303
304 auto* node = hitTestResult.innerNonSharedNode();
305 if (!node || !node->renderer())
306 return std::nullopt;
307
308 auto* frame = node->document().frame();
309 if (!frame)
310 return std::nullopt;
311
312 // Don't do anything if there is no character at the point.
313 auto framePoint = hitTestResult.roundedPointInInnerNodeFrame();
314 if (!frame->rangeForPoint(framePoint))
315 return std::nullopt;
316
317 auto position = frame->visiblePositionForPoint(framePoint);
318 if (position.isNull())
319 position = firstPositionInOrBeforeNode(node);
320
321 auto selection = frame->page()->focusController().focusedOrMainFrame().selection().selection();
322 NSRange selectionRange;
323 NSUInteger hitIndex;
324 std::optional<SimpleRange> fullCharacterRange;
325
326 if (selection.isRange()) {
327 auto selectionStart = selection.visibleStart();
328 auto selectionEnd = selection.visibleEnd();
329
330 // As context, we are going to use the surrounding paragraphs of text.
331 fullCharacterRange = makeSimpleRange(startOfParagraph(selectionStart), endOfParagraph(selectionEnd));
332 if (!fullCharacterRange)
333 return std::nullopt;
334
335 selectionRange = NSMakeRange(characterCount(*makeSimpleRange(fullCharacterRange->start, selectionStart)),
336 characterCount(*makeSimpleRange(selectionStart, selectionEnd)));
337 hitIndex = characterCount(*makeSimpleRange(fullCharacterRange->start, position));
338 } else {
339 VisibleSelection selectionAccountingForLineRules { position };
340 selectionAccountingForLineRules.expandUsingGranularity(TextGranularity::WordGranularity);
341 position = selectionAccountingForLineRules.start();
342
343 // As context, we are going to use 250 characters of text before and after the point.
344 fullCharacterRange = rangeExpandedAroundPositionByCharacters(position, 250);
345 if (!fullCharacterRange)
346 return std::nullopt;
347
348 selectionRange = NSMakeRange(NSNotFound, 0);
349 hitIndex = characterCount(*makeSimpleRange(fullCharacterRange->start, position));
350 }
351
352 NSRange selectedRange = [PAL::getRVSelectionClass() revealRangeAtIndex:hitIndex selectedRanges:@[[NSValue valueWithRange:selectionRange]] shouldUpdateSelection:nil];
353
354 String itemString = plainText(*fullCharacterRange);
355 auto highlightRange = adoptNS([PAL::allocRVItemInstance() initWithText:itemString selectedRange:selectedRange]).get().highlightRange;
356
357 if (highlightRange.location == NSNotFound || !highlightRange.length)
358 return std::nullopt;
359
360 return { { resolveCharacterRange(*fullCharacterRange, highlightRange), nil } };
361
362 END_BLOCK_OBJC_EXCEPTIONS
363
364 return std::nullopt;
365
366}

368static void expandSelectionByCharacters(PDFSelection *selection, NSInteger numberOfCharactersToExpand, NSInteger& charactersAddedBeforeStart, NSInteger& charactersAddedAfterEnd)
369{
370 BEGIN_BLOCK_OBJC_EXCEPTIONS
371
372 size_t originalLength = selection.string.length;
373 [selection extendSelectionAtStart:numberOfCharactersToExpand];
374
375 charactersAddedBeforeStart = selection.string.length - originalLength;
376
377 [selection extendSelectionAtEnd:numberOfCharactersToExpand];
378 charactersAddedAfterEnd = selection.string.length - originalLength - charactersAddedBeforeStart;
379
380 END_BLOCK_OBJC_EXCEPTIONS
381}

383std::tuple<NSString *, NSDictionary *> DictionaryLookup::stringForPDFSelection(PDFSelection *selection)
384{
385 BEGIN_BLOCK_OBJC_EXCEPTIONS
386
387 if (!canCreateRevealItems())
388 return { nullptr, nil };
389
390 // Don't do anything if there is no character at the point.
391 if (!selection || !selection.string.length)
392 return { @"", nil };
393
394 RetainPtr<PDFSelection> selectionForLookup = adoptNS([selection copy]);
395
396 // As context, we are going to use 250 characters of text before and after the point.
397 auto originalLength = [selectionForLookup string].length;
398 NSInteger charactersAddedBeforeStart = 0;
399 NSInteger charactersAddedAfterEnd = 0;
400 expandSelectionByCharacters(selectionForLookup.get(), 250, charactersAddedBeforeStart, charactersAddedAfterEnd);
401
402 auto fullPlainTextString = [selectionForLookup string];
403 auto rangeToPass = NSMakeRange(charactersAddedBeforeStart, 0);
404
405 auto item = adoptNS([PAL::allocRVItemInstance() initWithText:fullPlainTextString selectedRange:rangeToPass]);
406 NSRange extractedRange = item.get().highlightRange;
407
408 if (extractedRange.location == NSNotFound)
409 return { selection.string, nil };
410
411 NSInteger lookupAddedBefore = rangeToPass.location - extractedRange.location;
412 NSInteger lookupAddedAfter = (extractedRange.location + extractedRange.length) - (rangeToPass.location + originalLength);
413
414 [selection extendSelectionAtStart:lookupAddedBefore];
415 [selection extendSelectionAtEnd:lookupAddedAfter];
416
417 ASSERT([selection.string isEqualToString:[fullPlainTextString substringWithRange:extractedRange]]);
418 return { selection.string, nil };
419
420 END_BLOCK_OBJC_EXCEPTIONS
421
422 return { @"", nil };
423}

425static WKRevealController showPopupOrCreateAnimationController(bool createAnimationController, const DictionaryPopupInfo& dictionaryPopupInfo, CocoaView *view, const WTF::Function<void(TextIndicator&)>& textIndicatorInstallationCallback, const WTF::Function<FloatRect(FloatRect)>& rootViewToViewConversionCallback, WTF::Function<void()>&& clearTextIndicator)
426{
427 BEGIN_BLOCK_OBJC_EXCEPTIONS
428
429#if PLATFORM(MAC)
430 if (!canCreateRevealItems() || !PAL::getRVPresenterClass())
431 return nil;
432
433 auto mutableOptions = adoptNS([[NSMutableDictionary alloc] init]);
434 if (NSDictionary *options = dictionaryPopupInfo.options.get())
435 [mutableOptions addEntriesFromDictionary:options];
436
437 auto textIndicator = TextIndicator::create(dictionaryPopupInfo.textIndicator);
438
439 auto presenter = adoptNS([PAL::allocRVPresenterInstance() init]);
440
441 NSRect highlightRect;
442 NSPoint pointerLocation;
443
444 if (textIndicator.get().contentImage()) {
445 textIndicatorInstallationCallback(textIndicator.get());
446
447 FloatRect firstTextRectInViewCoordinates = textIndicator.get().textRectsInBoundingRectCoordinates()[0];
448 FloatRect textBoundingRectInViewCoordinates = textIndicator.get().textBoundingRectInRootViewCoordinates();
449 FloatRect selectionBoundingRectInViewCoordinates = textIndicator.get().selectionRectInRootViewCoordinates();
450
451 if (rootViewToViewConversionCallback) {
452 textBoundingRectInViewCoordinates = rootViewToViewConversionCallback(textBoundingRectInViewCoordinates);
453 selectionBoundingRectInViewCoordinates = rootViewToViewConversionCallback(selectionBoundingRectInViewCoordinates);
454 }
455
456 firstTextRectInViewCoordinates.moveBy(textBoundingRectInViewCoordinates.location());
457 highlightRect = selectionBoundingRectInViewCoordinates;
458 pointerLocation = firstTextRectInViewCoordinates.location();
459 } else {
460 NSPoint textBaselineOrigin = dictionaryPopupInfo.origin;
461 highlightRect = textIndicator->selectionRectInRootViewCoordinates();
462 pointerLocation = [view convertPoint:textBaselineOrigin toView:nil];
463 }
464
465 auto webHighlight = adoptNS([[WebRevealHighlight alloc] initWithHighlightRect: highlightRect useDefaultHighlight:!textIndicator.get().contentImage() attributedString:dictionaryPopupInfo.attributedString.get()]);
466 auto context = adoptNS([PAL::allocRVPresentingContextInstance() initWithPointerLocationInView:pointerLocation inView:view highlightDelegate:webHighlight.get()]);
467 auto item = adoptNS([PAL::allocRVItemInstance() initWithText:dictionaryPopupInfo.attributedString.get().string selectedRange:NSMakeRange(0, dictionaryPopupInfo.attributedString.get().string.length)]);
468
469 [webHighlight setClearTextIndicator:[webHighlight = WTFMove(webHighlight), clearTextIndicator = WTFMove(clearTextIndicator)] {
470 if (clearTextIndicator)
471 clearTextIndicator();
472 }];
473
474 if (createAnimationController)
475 return [presenter animationControllerForItem:item.get() documentContext:nil presentingContext:context.get() options:nil];
476
477 [presenter revealItem:item.get() documentContext:nil presentingContext:context.get() options:@{ @"forceLookup": @YES }];
478 return nil;
479#elif PLATFORM(MACCATALYST)
480 UNUSED_PARAM(textIndicatorInstallationCallback);
481 UNUSED_PARAM(rootViewToViewConversionCallback);
482 UNUSED_PARAM(clearTextIndicator);
483 ASSERT_UNUSED(createAnimationController, !createAnimationController);
484 auto textIndicator = TextIndicator::create(dictionaryPopupInfo.textIndicator);
485 auto webHighlight = adoptNS([[WebRevealHighlight alloc] initWithHighlightRect:[view convertRect:textIndicator->selectionRectInRootViewCoordinates() toView:nil] view:view image:textIndicator->contentImage()]);
486 auto item = adoptNS([PAL::allocRVItemInstance() initWithText:dictionaryPopupInfo.attributedString.get().string selectedRange:NSMakeRange(0, dictionaryPopupInfo.attributedString.get().string.length)]);
487
488 [UINSSharedRevealController() revealItem:item.get() locationInWindow:dictionaryPopupInfo.origin window:view.window highlighter:webHighlight.get()];
489 return nil;
490#else // PLATFORM(IOS_FAMILY)
491 UNUSED_PARAM(createAnimationController);
492 UNUSED_PARAM(dictionaryPopupInfo);
493 UNUSED_PARAM(view);
494 UNUSED_PARAM(textIndicatorInstallationCallback);
495 UNUSED_PARAM(rootViewToViewConversionCallback);
496 UNUSED_PARAM(clearTextIndicator);
497 return nil;
498#endif // PLATFORM(IOS_FAMILY)
499
500 END_BLOCK_OBJC_EXCEPTIONS
501}
