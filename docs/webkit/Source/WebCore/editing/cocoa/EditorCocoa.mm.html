74static RefPtr<SharedBuffer> archivedDataForAttributedString(NSAttributedString *attributedString)
75{
76 if (!attributedString.length)
77 return nullptr;
78
79 return SharedBuffer::create([NSKeyedArchiver archivedDataWithRootObject:attributedString requiringSecureCoding:YES error:nullptr]);
80}

110static RetainPtr<NSAttributedString> selectionAsAttributedString(const Document& document)
111{
112 auto range = document.selection().selection().firstRange();
113 return range ? attributedString(*range).string : adoptNS([[NSAttributedString alloc] init]);
114}

116void Editor::writeSelectionToPasteboard(Pasteboard& pasteboard)
117{
118 auto string = selectionAsAttributedString(m_document);
119
120 PasteboardWebContent content;
121 content.contentOrigin = m_document.originIdentifierForPasteboard();
122 content.canSmartCopyOrDelete = canSmartCopyOrDelete();
123 if (!pasteboard.isStatic()) {
124 content.dataInWebArchiveFormat = selectionInWebArchiveFormat();
125 content.dataInRTFDFormat = [string containsAttachments] ? dataInRTFDFormat(string.get()) : nullptr;
126 content.dataInRTFFormat = dataInRTFFormat(string.get());
127 content.dataInAttributedStringFormat = archivedDataForAttributedString(string.get());
128 client()->getClientPasteboardData(selectedRange(), content.clientTypes, content.clientData);
129 }
130 content.dataInHTMLFormat = selectionInHTMLFormat();
131 content.dataInStringFormat = stringSelectionForPasteboardWithImageAltText();
132
133 pasteboard.write(content);
134}

136void Editor::writeSelection(PasteboardWriterData& pasteboardWriterData)
137{
138 auto string = selectionAsAttributedString(m_document);
139
140 PasteboardWriterData::WebContent webContent;
141 webContent.contentOrigin = m_document.originIdentifierForPasteboard();
142 webContent.canSmartCopyOrDelete = canSmartCopyOrDelete();
143 webContent.dataInWebArchiveFormat = selectionInWebArchiveFormat();
144 webContent.dataInRTFDFormat = [string containsAttachments] ? dataInRTFDFormat(string.get()) : nullptr;
145 webContent.dataInRTFFormat = dataInRTFFormat(string.get());
146 webContent.dataInAttributedStringFormat = archivedDataForAttributedString(string.get());
147 webContent.dataInHTMLFormat = selectionInHTMLFormat();
148 webContent.dataInStringFormat = stringSelectionForPasteboardWithImageAltText();
149 client()->getClientPasteboardData(selectedRange(), webContent.clientTypes, webContent.clientData);
150
151 pasteboardWriterData.setWebContent(WTFMove(webContent));
152}

154RefPtr<SharedBuffer> Editor::selectionInWebArchiveFormat()
155{

182void Editor::replaceSelectionWithAttributedString(NSAttributedString *attributedString, MailBlockquoteHandling mailBlockquoteHandling)
183{
184 if (m_document.selection().isNone())
185 return;
186
187 if (m_document.selection().selection().isContentRichlyEditable()) {
188 if (auto fragment = createFragmentAndAddResources(*m_document.frame(), attributedString)) {
189 if (shouldInsertFragment(*fragment, selectedRange(), EditorInsertAction::Pasted))
190 pasteAsFragment(fragment.releaseNonNull(), false, false, mailBlockquoteHandling);
191 }
192 } else {
193 String text = attributedString.string;
194 if (shouldInsertText(text, selectedRange(), EditorInsertAction::Pasted))
195 pasteAsPlainText(text, false);
196 }
197}

204RefPtr<SharedBuffer> Editor::dataInRTFDFormat(NSAttributedString *string)
205{
206 NSUInteger length = string.length;
207 if (!length)
208 return nullptr;
209
210 BEGIN_BLOCK_OBJC_EXCEPTIONS
211 return SharedBuffer::create([string RTFDFromRange:NSMakeRange(0, length) documentAttributes:@{ }]);
212 END_BLOCK_OBJC_EXCEPTIONS
213
214 return nullptr;
215}
216
217RefPtr<SharedBuffer> Editor::dataInRTFFormat(NSAttributedString *string)
218{
219 NSUInteger length = string.length;
220 if (!length)
221 return nullptr;
222
223 BEGIN_BLOCK_OBJC_EXCEPTIONS
224 return SharedBuffer::create([string RTFFromRange:NSMakeRange(0, length) documentAttributes:@{ }]);
225 END_BLOCK_OBJC_EXCEPTIONS
226
227 return nullptr;
228}

240void Editor::takeFindStringFromSelection()
241{
242 if (!canCopyExcludingStandaloneImages()) {
243 SystemSoundManager::singleton().systemBeep();
244 return;
245 }
246
247 auto stringFromSelection = m_document.frame()->displayStringModifiedByEncoding(selectedTextForDataTransfer());
248#if PLATFORM(MAC)
249 Vector<String> types;
250 types.append(String(legacyStringPasteboardType()));
251 auto context = PagePasteboardContext::create(m_document.pageID());
252 ALLOW_DEPRECATED_DECLARATIONS_BEGIN
253 platformStrategies()->pasteboardStrategy()->setTypes(types, NSFindPboard, context.get());
254 platformStrategies()->pasteboardStrategy()->setStringForType(WTFMove(stringFromSelection), legacyStringPasteboardType(), NSFindPboard, context.get());
255 ALLOW_DEPRECATED_DECLARATIONS_END
256#else
257 if (auto* client = this->client()) {
258 // Since the find pasteboard doesn't exist on iOS, WebKit maintains its own notion of the latest find string,
259 // which SPI clients may respect when presenting find-in-page UI.
260 client->updateStringForFind(stringFromSelection);
261 }
262#endif
