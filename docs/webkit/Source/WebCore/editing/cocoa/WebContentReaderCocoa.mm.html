98static NSDictionary *attributesForAttributedStringConversion()
99{
100 // This function needs to be kept in sync with identically named one in WebKitLegacy, which is used on older OS versions.
101 RetainPtr<NSMutableArray> excludedElements = adoptNS([[NSMutableArray alloc] initWithObjects:
102 // Omit style since we want style to be inline so the fragment can be easily inserted.
103 @"style",
104 // Omit xml so the result is not XHTML.
105 @"xml",
106 // Omit tags that will get stripped when converted to a fragment anyway.
107 @"doctype", @"html", @"head", @"body",
108 // Omit deprecated tags.
109 @"applet", @"basefont", @"center", @"dir", @"font", @"menu", @"s", @"strike", @"u",
110#if !ENABLE(ATTACHMENT_ELEMENT)
111 // Omit object so no file attachments are part of the fragment.
112 @"object",
113#endif
114 nil]);
115
116#if ENABLE(ATTACHMENT_ELEMENT)
117 if (!RuntimeEnabledFeatures::sharedFeatures().attachmentElementEnabled())
118 [excludedElements addObject:@"object"];
119#endif
120
121 NSURL *baseURL = URL::fakeURLWithRelativePart(emptyString());
122
123 // The output base URL needs +1 refcount to work around the fact that NSHTMLReader over-releases it.
124 CFRetain((__bridge CFTypeRef)baseURL);
125
126 return @{
127 NSExcludedElementsDocumentAttribute: excludedElements.get(),
128 @"InterchangeNewline": @YES,
129 @"CoalesceTabSpans": @YES,
130 @"OutputBaseURL": baseURL,
131 @"WebResourceHandler": adoptNS([WebArchiveResourceWebResourceHandler new]).get(),
132 };
133}

135static FragmentAndResources createFragment(Frame& frame, NSAttributedString *string)
136{
137 FragmentAndResources result;
138 Document& document = *frame.document();
139
140#if PLATFORM(MAC)
141 auto* view = frame.view();
142 LocalDefaultSystemAppearance localAppearance(view ? view->useDarkAppearance() : false);
143#endif
144
145 NSArray *subresources = nil;
146 NSString *fragmentString = [string _htmlDocumentFragmentString:NSMakeRange(0, [string length]) documentAttributes:attributesForAttributedStringConversion() subresources:&subresources];
147 auto fragment = DocumentFragment::create(document);
148 auto dummyBodyToForceInBodyInsertionMode = HTMLBodyElement::create(document);
149 fragment->parseHTML(fragmentString, dummyBodyToForceInBodyInsertionMode.ptr(), DisallowScriptingAndPluginContent);
150
151 result.fragment = WTFMove(fragment);
152 for (WebArchiveResourceFromNSAttributedString *resource in subresources)
153 result.resources.append(*resource->resource);
154
155 return result;
156}

170class DeferredLoadingScope {
171public:
172 DeferredLoadingScope(Frame& frame)
173 : m_frame(frame)
174 , m_cachedResourceLoader(frame.document()->cachedResourceLoader())
175 {
176 if (!frame.page()->defersLoading()) {
177 frame.page()->setDefersLoading(true);
178 m_didEnabledDeferredLoading = true;
179 }
180
181 if (m_cachedResourceLoader->imagesEnabled()) {
182 m_cachedResourceLoader->setImagesEnabled(false);
183 m_didDisableImage = true;
184 }
185 }
186
187 ~DeferredLoadingScope()
188 {
189 if (m_didDisableImage)
190 m_cachedResourceLoader->setImagesEnabled(true);
191 if (m_didEnabledDeferredLoading)
192 m_frame->page()->setDefersLoading(false);
193 }
194
195private:
196 Ref<Frame> m_frame;
197 Ref<CachedResourceLoader> m_cachedResourceLoader;
198 bool m_didEnabledDeferredLoading { false };
199 bool m_didDisableImage { false };
200};

219static String mimeTypeFromContentType(const String& contentType)
220{
221ALLOW_DEPRECATED_DECLARATIONS_BEGIN
222 if (contentType == String(kUTTypeVCard)) {
223 // CoreServices erroneously reports that "public.vcard" maps to "text/directory", rather
224 // than either "text/vcard" or "text/x-vcard". Work around this by special casing the
225 // "public.vcard" UTI type. See <rdar://problem/49478229> for more detail.
226 return "text/vcard"_s;
227 }
228ALLOW_DEPRECATED_DECLARATIONS_END
229 return isDeclaredUTI(contentType) ? MIMETypeFromUTI(contentType) : contentType;
230}

247static Ref<DocumentFragment> createFragmentForImageAttachment(Frame& frame, Document& document, Ref<SharedBuffer>&& buffer, const String& contentType, PresentationSize preferredSize)
248{
249#if ENABLE(ATTACHMENT_ELEMENT)
250 auto attachment = HTMLAttachmentElement::create(HTMLNames::attachmentTag, document);
251 // FIXME: This fallback image name needs to be a localized string.
252 String defaultImageAttachmentName { "image"_s };
253
254 auto fragment = document.createDocumentFragment();
255 if (supportsClientSideAttachmentData(frame)) {
256 frame.editor().registerAttachmentIdentifier(attachment->ensureUniqueIdentifier(), contentType, defaultImageAttachmentName, WTFMove(buffer));
257 if (contentTypeIsSuitableForInlineImageRepresentation(contentType)) {
258 auto image = HTMLImageElement::create(document);
259 image->setAttributeWithoutSynchronization(HTMLNames::srcAttr, DOMURL::createObjectURL(document, Blob::create(&document, buffer->takeData(), contentType)));
260 image->setAttachmentElement(WTFMove(attachment));
261 if (preferredSize.width)
262 image->setAttributeWithoutSynchronization(HTMLNames::widthAttr, AtomString::number(*preferredSize.width));
263 if (preferredSize.height)
264 image->setAttributeWithoutSynchronization(HTMLNames::heightAttr, AtomString::number(*preferredSize.height));
265 fragment->appendChild(WTFMove(image));
266 } else {
267 attachment->updateAttributes(buffer->size(), contentType, defaultImageAttachmentName);
268 fragment->appendChild(WTFMove(attachment));
269 }
270 } else {
271 attachment->setFile(File::create(&document, Blob::create(&document, buffer->takeData(), contentType), defaultImageAttachmentName), HTMLAttachmentElement::UpdateDisplayAttributes::Yes);
272 fragment->appendChild(WTFMove(attachment));
273 }
274 return fragment;
275#else
276 UNUSED_PARAM(blob);
277 return document.createDocumentFragment();
278#endif
279}

281static void replaceRichContentWithAttachments(Frame& frame, DocumentFragment& fragment, const Vector<Ref<ArchiveResource>>& subresources)
282{
283#if ENABLE(ATTACHMENT_ELEMENT)
284 struct AttachmentInsertionInfo {
285 String fileName;
286 String contentType;
287 Ref<SharedBuffer> data;
288 Ref<Element> originalElement;
289 };
290
291 ASSERT(RuntimeEnabledFeatures::sharedFeatures().attachmentElementEnabled());
292 if (subresources.isEmpty())
293 return;
294
295 // FIXME: Handle resources in subframe archives.
296 HashMap<AtomString, Ref<ArchiveResource>> urlToResourceMap;
297 for (auto& subresource : subresources) {
298 auto& url = subresource->url();
299 if (shouldReplaceSubresourceURL(url))
300 urlToResourceMap.set(url.string(), subresource.copyRef());
301 }
302
303 Vector<SerializedAttachmentData> serializedAttachmentData;
304 for (auto& attachment : descendantsOfType<HTMLAttachmentElement>(fragment)) {
305 auto resourceURL = HTMLAttachmentElement::archiveResourceURL(attachment.uniqueIdentifier());
306 auto resourceEntry = urlToResourceMap.find(resourceURL.string());
307 if (resourceEntry == urlToResourceMap.end())
308 continue;
309
310 auto& resource = resourceEntry->value;
311 serializedAttachmentData.append({ attachment.uniqueIdentifier(), resource->mimeType(), resource->data() });
312 }
313
314 if (!serializedAttachmentData.isEmpty())
315 frame.editor().registerAttachments(WTFMove(serializedAttachmentData));
316
317 Vector<Ref<Element>> elementsToRemove;
318 Vector<AttachmentInsertionInfo> attachmentInsertionInfo;
319 for (auto& image : descendantsOfType<HTMLImageElement>(fragment)) {
320 auto resourceURLString = image.attributeWithoutSynchronization(HTMLNames::srcAttr);
321 if (resourceURLString.isEmpty())
322 continue;
323
324 auto resource = urlToResourceMap.find(resourceURLString);
325 if (resource == urlToResourceMap.end())
326 continue;
327
328 String name = image.attributeWithoutSynchronization(HTMLNames::altAttr);
329 if (name.isEmpty())
330 name = URL({ }, resourceURLString).lastPathComponent().toString();
331 if (name.isEmpty())
332 name = "media"_s;
333
334 attachmentInsertionInfo.append({ WTFMove(name), resource->value->mimeType(), resource->value->data(), image });
335 }
336
337 for (auto& object : descendantsOfType<HTMLObjectElement>(fragment)) {
338 auto resourceURLString = object.attributeWithoutSynchronization(HTMLNames::dataAttr);
339 if (resourceURLString.isEmpty()) {
340 elementsToRemove.append(object);
341 continue;
342 }
343
344 auto resource = urlToResourceMap.find(resourceURLString);
345 if (resource == urlToResourceMap.end())
346 continue;
347
348 String name = URL({ }, resourceURLString).lastPathComponent().toString();
349 if (name.isEmpty())
350 name = "file"_s;
351
352 attachmentInsertionInfo.append({ WTFMove(name), resource->value->mimeType(), resource->value->data(), object });
353 }
354
355 for (auto& info : attachmentInsertionInfo) {
356 auto originalElement = WTFMove(info.originalElement);
357 auto parent = makeRefPtr(originalElement->parentNode());
358 if (!parent)
359 continue;
360
361 auto attachment = HTMLAttachmentElement::create(HTMLNames::attachmentTag, fragment.document());
362 if (supportsClientSideAttachmentData(frame)) {
363 if (is<HTMLImageElement>(originalElement.get()) && contentTypeIsSuitableForInlineImageRepresentation(info.contentType)) {
364 auto& image = downcast<HTMLImageElement>(originalElement.get());
365 image.setAttributeWithoutSynchronization(HTMLNames::srcAttr, DOMURL::createObjectURL(*frame.document(), Blob::create(frame.document(), info.data->copyData(), info.contentType)));
366 image.setAttachmentElement(attachment.copyRef());
367 } else {
368 attachment->updateAttributes(info.data->size(), info.contentType, info.fileName);
369 parent->replaceChild(attachment, WTFMove(originalElement));
370 }
371 frame.editor().registerAttachmentIdentifier(attachment->ensureUniqueIdentifier(), WTFMove(info.contentType), WTFMove(info.fileName), WTFMove(info.data));
372 } else {
373 attachment->setFile(File::create(frame.document(), Blob::create(frame.document(), info.data->copyData(), WTFMove(info.contentType)), WTFMove(info.fileName)), HTMLAttachmentElement::UpdateDisplayAttributes::Yes);
374 parent->replaceChild(WTFMove(attachment), WTFMove(originalElement));
375 }
376 }
377
378 for (auto& elementToRemove : elementsToRemove)
379 elementToRemove->remove();
380#else
381 UNUSED_PARAM(fragment);
382 UNUSED_PARAM(subresources);
383#endif
384}

386RefPtr<DocumentFragment> createFragmentAndAddResources(Frame& frame, NSAttributedString *string)
387{
388 if (!frame.page() || !frame.document())
389 return nullptr;
390
391 auto& document = *frame.document();
392 if (!document.isHTMLDocument() || !string)
393 return nullptr;
394
395 DeferredLoadingScope scope(frame);
396 auto fragmentAndResources = createFragment(frame, string);
397 if (!fragmentAndResources.fragment)
398 return nullptr;
399
400 if (!RuntimeEnabledFeatures::sharedFeatures().customPasteboardDataEnabled()) {
401 if (DocumentLoader* loader = frame.loader().documentLoader()) {
402 for (auto& resource : fragmentAndResources.resources)
403 loader->addArchiveResource(resource.copyRef());
404 }
405 return WTFMove(fragmentAndResources.fragment);
406 }
407
408 if (shouldReplaceRichContentWithAttachments()) {
409 replaceRichContentWithAttachments(frame, *fragmentAndResources.fragment, fragmentAndResources.resources);
410 return WTFMove(fragmentAndResources.fragment);
411 }
412
413 HashMap<AtomString, AtomString> blobURLMap;
414 for (auto& subresource : fragmentAndResources.resources) {
415 auto blob = Blob::create(&document, subresource->data().copyData(), subresource->mimeType());
416 String blobURL = DOMURL::createObjectURL(document, blob);
417 blobURLMap.set(subresource->url().string(), blobURL);
418 }
419
420 replaceSubresourceURLs(*fragmentAndResources.fragment, WTFMove(blobURLMap));
421 return WTFMove(fragmentAndResources.fragment);
422}

430static std::optional<MarkupAndArchive> extractMarkupAndArchive(SharedBuffer& buffer, const std::function<bool(const String)>& canShowMIMETypeAsHTML)
431{
432 auto archive = LegacyWebArchive::create(URL(), buffer);
433 if (!archive)
434 return std::nullopt;
435
436 RefPtr<ArchiveResource> mainResource = archive->mainResource();
437 if (!mainResource)
438 return std::nullopt;
439
440 auto type = mainResource->mimeType();
441 if (!canShowMIMETypeAsHTML(type))
442 return std::nullopt;
443
444 return MarkupAndArchive { String::fromUTF8(mainResource->data().data(), mainResource->data().size()), mainResource.releaseNonNull(), archive.releaseNonNull() };
445}

447static String sanitizeMarkupWithArchive(Frame& frame, Document& destinationDocument, MarkupAndArchive& markupAndArchive, MSOListQuirks msoListQuirks, const std::function<bool(const String)>& canShowMIMETypeAsHTML)
448{
449 auto page = createPageForSanitizingWebContent();
450 Document* stagingDocument = page->mainFrame().document();
451 ASSERT(stagingDocument);
452 auto fragment = createFragmentFromMarkup(*stagingDocument, markupAndArchive.markup, markupAndArchive.mainResource->url().string(), DisallowScriptingAndPluginContent);
453
454 if (shouldReplaceRichContentWithAttachments()) {
455 replaceRichContentWithAttachments(frame, fragment, markupAndArchive.archive->subresources());
456 return sanitizedMarkupForFragmentInDocument(WTFMove(fragment), *stagingDocument, msoListQuirks, markupAndArchive.markup);
457 }
458
459 HashMap<AtomString, AtomString> blobURLMap;
460 for (const Ref<ArchiveResource>& subresource : markupAndArchive.archive->subresources()) {
461 auto& subresourceURL = subresource->url();
462 if (!shouldReplaceSubresourceURL(subresourceURL))
463 continue;
464 auto blob = Blob::create(&destinationDocument, subresource->data().copyData(), subresource->mimeType());
465 String blobURL = DOMURL::createObjectURL(destinationDocument, blob);
466 blobURLMap.set(subresourceURL.string(), blobURL);
467 }
468
469 auto contentOrigin = SecurityOrigin::create(markupAndArchive.mainResource->url());
470 for (const Ref<Archive>& subframeArchive : markupAndArchive.archive->subframeArchives()) {
471 RefPtr<ArchiveResource> subframeMainResource = subframeArchive->mainResource();
472 if (!subframeMainResource)
473 continue;
474
475 auto type = subframeMainResource->mimeType();
476 if (!canShowMIMETypeAsHTML(type))
477 continue;
478
479 auto subframeURL = subframeMainResource->url();
480 if (!shouldReplaceSubresourceURL(subframeURL))
481 continue;
482
483 MarkupAndArchive subframeContent = { String::fromUTF8(subframeMainResource->data().data(), subframeMainResource->data().size()),
484 subframeMainResource.releaseNonNull(), subframeArchive.copyRef() };
485 auto subframeMarkup = sanitizeMarkupWithArchive(frame, destinationDocument, subframeContent, MSOListQuirks::Disabled, canShowMIMETypeAsHTML);
486
487 CString utf8 = subframeMarkup.utf8();
488 auto blob = Blob::create(&destinationDocument, Vector { utf8.dataAsUInt8Ptr(), utf8.length() }, type);
489
490 String subframeBlobURL = DOMURL::createObjectURL(destinationDocument, blob);
491 blobURLMap.set(subframeURL.string(), subframeBlobURL);
492 }
493
494 replaceSubresourceURLs(fragment.get(), WTFMove(blobURLMap));
495
496 return sanitizedMarkupForFragmentInDocument(WTFMove(fragment), *stagingDocument, msoListQuirks, markupAndArchive.markup);
497}

499bool WebContentReader::readWebArchive(SharedBuffer& buffer)
500{
501 if (frame.settings().preferMIMETypeForImages() || !frame.document())
502 return false;
503
504 DeferredLoadingScope scope(frame);
505 auto result = extractMarkupAndArchive(buffer, [&] (const String& type) {
506 return frame.loader().client().canShowMIMETypeAsHTML(type);
507 });
508 if (!result)
509 return false;
510
511 if (!RuntimeEnabledFeatures::sharedFeatures().customPasteboardDataEnabled()) {
512 fragment = createFragmentFromMarkup(*frame.document(), result->markup, result->mainResource->url().string(), DisallowScriptingAndPluginContent);
513 if (DocumentLoader* loader = frame.loader().documentLoader())
514 loader->addAllArchiveResources(result->archive.get());
515 return true;
516 }
517
518 if (!shouldSanitize()) {
519 fragment = createFragmentFromMarkup(*frame.document(), result->markup, result->mainResource->url().string(), DisallowScriptingAndPluginContent);
520 return true;
521 }
522
523 String sanitizedMarkup = sanitizeMarkupWithArchive(frame, *frame.document(), *result, msoListQuirksForMarkup(), [&] (const String& type) {
524 return frame.loader().client().canShowMIMETypeAsHTML(type);
525 });
526 fragment = createFragmentFromMarkup(*frame.document(), sanitizedMarkup, aboutBlankURL().string(), DisallowScriptingAndPluginContent);
527
528 if (!fragment)
529 return false;
530
531 return true;
532}

557static String stripMicrosoftPrefix(const String& string)
558{
559#if PLATFORM(MAC)
560 // This code was added to make HTML paste from Microsoft Word on Mac work, back in 2004.
561 // It's a simple-minded way to ignore the CF_HTML clipboard format, just skipping over the
562 // description part and parsing the entire context plus fragment.
563 if (string.startsWith("Version:")) {
564 size_t location = string.findIgnoringASCIICase("<html");
565 if (location != notFound)
566 return string.substring(location);
567 }
568#endif
569 return string;
570}

572bool WebContentReader::readHTML(const String& string)
573{
574 if (frame.settings().preferMIMETypeForImages() || !frame.document())
575 return false;
576 Document& document = *frame.document();
577
578 String stringOmittingMicrosoftPrefix = stripMicrosoftPrefix(string);
579 if (stringOmittingMicrosoftPrefix.isEmpty())
580 return false;
581
582 String markup;
583 if (RuntimeEnabledFeatures::sharedFeatures().customPasteboardDataEnabled() && shouldSanitize()) {
584 markup = sanitizeMarkup(stringOmittingMicrosoftPrefix, msoListQuirksForMarkup(), WTF::Function<void (DocumentFragment&)> { [] (DocumentFragment& fragment) {
585 removeSubresourceURLAttributes(fragment, [] (const URL& url) {
586 return shouldReplaceSubresourceURL(url);
587 });
588 } });
589 } else
590 markup = stringOmittingMicrosoftPrefix;
591
592 addFragment(createFragmentFromMarkup(document, markup, emptyString(), DisallowScriptingAndPluginContent));
593 return true;
594}
595
596bool WebContentMarkupReader::readHTML(const String& string)
597{
598 if (!frame.document())
599 return false;
600
601 String rawHTML = stripMicrosoftPrefix(string);
602 if (shouldSanitize()) {
603 markup = sanitizeMarkup(rawHTML, msoListQuirksForMarkup(), WTF::Function<void (DocumentFragment&)> { [] (DocumentFragment& fragment) {
604 removeSubresourceURLAttributes(fragment, [] (const URL& url) {
605 return shouldReplaceSubresourceURL(url);
606 });
607 } });
608 } else
609 markup = rawHTML;
610
611 return !markup.isEmpty();
612}

667bool WebContentReader::readPlainText(const String& text)
668{
669 if (!allowPlainText)
670 return false;
671
672 addFragment(createFragmentFromText(context, [text precomposedStringWithCanonicalMapping]));
673

704static Ref<HTMLElement> attachmentForFilePath(Frame& frame, const String& path, PresentationSize preferredSize, const String& explicitContentType)
705{
706 auto document = makeRef(*frame.document());
707 auto attachment = HTMLAttachmentElement::create(HTMLNames::attachmentTag, document);
708 if (!supportsClientSideAttachmentData(frame)) {
709 attachment->setFile(File::create(document.ptr(), path), HTMLAttachmentElement::UpdateDisplayAttributes::Yes);
710 return attachment;
711 }
712
713 bool isDirectory = FileSystem::fileTypeFollowingSymlinks(path) == FileSystem::FileType::Directory;
714 String contentType = typeForAttachmentElement(explicitContentType);
715 if (contentType.isEmpty()) {
716ALLOW_DEPRECATED_DECLARATIONS_BEGIN
717 if (isDirectory)
718 contentType = kUTTypeDirectory;
719 else {
720 contentType = File::contentTypeForFile(path);
721 if (contentType.isEmpty())
722 contentType = kUTTypeData;
723 }
724ALLOW_DEPRECATED_DECLARATIONS_END
725 }
726
727 std::optional<uint64_t> fileSizeForDisplay;
728 if (!isDirectory)
729 fileSizeForDisplay = FileSystem::fileSize(path).value_or(0);
730
731 frame.editor().registerAttachmentIdentifier(attachment->ensureUniqueIdentifier(), contentType, path);
732
733 if (contentTypeIsSuitableForInlineImageRepresentation(contentType)) {
734 auto image = HTMLImageElement::create(document);
735 image->setAttributeWithoutSynchronization(HTMLNames::srcAttr, DOMURL::createObjectURL(document, File::create(document.ptr(), path)));
736 image->setAttachmentElement(WTFMove(attachment));
737 if (preferredSize.width)
738 image->setAttributeWithoutSynchronization(HTMLNames::widthAttr, AtomString::number(*preferredSize.width));
739 if (preferredSize.height)
740 image->setAttributeWithoutSynchronization(HTMLNames::heightAttr, AtomString::number(*preferredSize.height));
741 return image;
742 }
743
744 attachment->updateAttributes(WTFMove(fileSizeForDisplay), contentType, FileSystem::pathFileName(path));
745 return attachment;
746}

748static Ref<HTMLElement> attachmentForData(Frame& frame, SharedBuffer& buffer, const String& contentType, const String& name, PresentationSize preferredSize)
749{
750 auto document = makeRef(*frame.document());
751 auto attachment = HTMLAttachmentElement::create(HTMLNames::attachmentTag, document);
752 auto attachmentType = typeForAttachmentElement(contentType);
753
754 // FIXME: We should instead ask CoreServices for a preferred name corresponding to the given content type.
755 static const char* defaultAttachmentName = "file";
756
757 String fileName;
758 if (name.isEmpty())
759 fileName = defaultAttachmentName;
760 else
761 fileName = name;
762
763 if (!supportsClientSideAttachmentData(frame)) {
764 attachment->setFile(File::create(document.ptr(), Blob::create(document.ptr(), buffer.copyData(), WTFMove(attachmentType)), fileName));
765 return attachment;
766 }
767
768 frame.editor().registerAttachmentIdentifier(attachment->ensureUniqueIdentifier(), attachmentType, fileName, buffer);
769
770 if (contentTypeIsSuitableForInlineImageRepresentation(attachmentType)) {
771 auto image = HTMLImageElement::create(document);
772 image->setAttributeWithoutSynchronization(HTMLNames::srcAttr, DOMURL::createObjectURL(document, File::create(document.ptr(), Blob::create(document.ptr(), buffer.copyData(), WTFMove(attachmentType)), WTFMove(fileName))));
773 image->setAttachmentElement(WTFMove(attachment));
774 if (preferredSize.width)
775 image->setAttributeWithoutSynchronization(HTMLNames::widthAttr, AtomString::number(*preferredSize.width));
776 if (preferredSize.height)
777 image->setAttributeWithoutSynchronization(HTMLNames::heightAttr, AtomString::number(*preferredSize.height));
778 return image;
779 }
780
781 attachment->updateAttributes({ buffer.size() }, WTFMove(attachmentType), WTFMove(fileName));
782 return attachment;
783}

804bool WebContentReader::readURL(const URL& url, const String& title)
805{
806 if (url.isEmpty())
807 return false;
808
809#if PLATFORM(IOS_FAMILY)
810 // FIXME: This code shouldn't be accessing selection and changing the behavior.
811 if (!frame.editor().client()->hasRichlyEditableSelection()) {
812 if (readPlainText([(NSURL *)url absoluteString]))
813 return true;
814 }
815
816 if ([(NSURL *)url isFileURL])
817 return false;
818#endif // PLATFORM(IOS_FAMILY)
819
820 auto document = makeRef(*frame.document());
821 auto anchor = HTMLAnchorElement::create(document.get());
822 anchor->setAttributeWithoutSynchronization(HTMLNames::hrefAttr, url.string());
823
824 NSString *linkText = title.isEmpty() ? [(NSURL *)url absoluteString] : (NSString *)title;
825 anchor->appendChild(document->createTextNode([linkText precomposedStringWithCanonicalMapping]));
826
827 auto newFragment = document->createDocumentFragment();
828 if (fragment)
829 newFragment->appendChild(HTMLBRElement::create(document.get()));
830 newFragment->appendChild(anchor);
831 addFragment(WTFMove(newFragment));
832 return true;
833}
