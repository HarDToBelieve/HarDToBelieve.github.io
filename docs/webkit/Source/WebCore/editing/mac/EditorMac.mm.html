81void Editor::pasteWithPasteboard(Pasteboard* pasteboard, OptionSet<PasteOption> options)
82{
83 auto range = selectedRange();
84
85 ALLOW_DEPRECATED_DECLARATIONS_BEGIN
86 // FIXME: How can this hard-coded pasteboard name be right, given that the passed-in pasteboard has a name?
87 client()->setInsertionPasteboard(NSGeneralPboard);
88 ALLOW_DEPRECATED_DECLARATIONS_END
89
90 bool chosePlainText;
91 RefPtr<DocumentFragment> fragment = webContentFromPasteboard(*pasteboard, *range, options.contains(PasteOption::AllowPlainText), chosePlainText);
92
93 if (fragment && options.contains(PasteOption::AsQuotation))
94 quoteFragmentForPasting(*fragment);
95
96 if (fragment && shouldInsertFragment(*fragment, range, EditorInsertAction::Pasted))
97 pasteAsFragment(fragment.releaseNonNull(), canSmartReplaceWithPasteboard(*pasteboard), false, options.contains(PasteOption::IgnoreMailBlockquote) ? MailBlockquoteHandling::IgnoreBlockquote : MailBlockquoteHandling::RespectBlockquote );
98
99 client()->setInsertionPasteboard(String());
100}

102void Editor::readSelectionFromPasteboard(const String& pasteboardName)
103{
104 Pasteboard pasteboard(PagePasteboardContext::create(m_document.pageID()), pasteboardName);
105 if (m_document.selection().selection().isContentRichlyEditable())
106 pasteWithPasteboard(&pasteboard, { PasteOption::AllowPlainText });
107 else
108 pasteAsPlainTextWithPasteboard(pasteboard);
109}

111static void maybeCopyNodeAttributesToFragment(const Node& node, DocumentFragment& fragment)
112{
113 // This is only supported for single-Node fragments.
114 Node* firstChild = fragment.firstChild();
115 if (!firstChild || firstChild != fragment.lastChild())
116 return;
117
118 // And only supported for HTML elements.
119 if (!node.isHTMLElement() || !firstChild->isHTMLElement())
120 return;
121
122 // And only if the source Element and destination Element have the same HTML tag name.
123 const HTMLElement& oldElement = downcast<HTMLElement>(node);
124 HTMLElement& newElement = downcast<HTMLElement>(*firstChild);
125 if (oldElement.localName() != newElement.localName())
126 return;
127
128 for (const Attribute& attribute : oldElement.attributesIterator()) {
129 if (newElement.hasAttribute(attribute.name()))
130 continue;
131 newElement.setAttribute(attribute.name(), attribute.value());
132 }
133}

135void Editor::replaceNodeFromPasteboard(Node* node, const String& pasteboardName)
136{
137 ASSERT(node);
138
139 if (node->document() != m_document)
140 return;
141
142 Ref<Document> protector(m_document);
143
144 auto range = makeRangeSelectingNodeContents(*node);
145 m_document.selection().setSelection(VisibleSelection(range), FrameSelection::DoNotSetFocus);
146
147 Pasteboard pasteboard(PagePasteboardContext::create(m_document.pageID()), pasteboardName);
148
149 if (!m_document.selection().selection().isContentRichlyEditable()) {
150 pasteAsPlainTextWithPasteboard(pasteboard);
151 return;
152 }
153
154 ALLOW_DEPRECATED_DECLARATIONS_BEGIN
155 // FIXME: How can this hard-coded pasteboard name be right, given that the passed-in pasteboard has a name?
156 client()->setInsertionPasteboard(NSGeneralPboard);
157 ALLOW_DEPRECATED_DECLARATIONS_END
158
159 bool chosePlainText;
160 if (auto fragment = webContentFromPasteboard(pasteboard, range, true, chosePlainText)) {
161 maybeCopyNodeAttributesToFragment(*node, *fragment);
162 if (shouldInsertFragment(*fragment, range, EditorInsertAction::Pasted))
163 pasteAsFragment(fragment.releaseNonNull(), canSmartReplaceWithPasteboard(pasteboard), false, MailBlockquoteHandling::IgnoreBlockquote);
164 }
165
166 client()->setInsertionPasteboard(String());
167}

177RefPtr<SharedBuffer> Editor::dataSelectionForPasteboard(const String& pasteboardType)
178{
179 // FIXME: The interface to this function is awkward. We'd probably be better off with three separate functions. As of this writing, this is only used in WebKit2 to implement the method -[WKView writeSelectionToPasteboard:types:], which is only used to support OS X services.
180
181 // FIXME: Does this function really need to use adjustedSelectionRange()? Because writeSelectionToPasteboard() just uses selectedRange(). This is the only function in WebKit that uses adjustedSelectionRange.
182 if (!canCopy())
183 return nullptr;
184
185ALLOW_DEPRECATED_DECLARATIONS_BEGIN
186 if (pasteboardType == WebArchivePboardType || pasteboardType == String(kUTTypeWebArchive))
187 return selectionInWebArchiveFormat();
188ALLOW_DEPRECATED_DECLARATIONS_END
189
190 if (pasteboardType == String(legacyRTFDPasteboardType()))
191 return dataInRTFDFormat(attributedString(*adjustedSelectionRange()).string.get());
192
193 if (pasteboardType == String(legacyRTFPasteboardType())) {
194 auto string = attributedString(*adjustedSelectionRange()).string;
195 // FIXME: Why is this stripping needed here, but not in writeSelectionToPasteboard?
196 if ([string containsAttachments])
197 string = attributedStringByStrippingAttachmentCharacters(string.get());
198 return dataInRTFFormat(string.get());
199 }
200
201 return nullptr;
202}

204static void getImage(Element& imageElement, RefPtr<Image>& image, CachedImage*& cachedImage)
205{
206 auto* renderer = imageElement.renderer();
207 if (!is<RenderImage>(renderer))
208 return;
209
210 CachedImage* tentativeCachedImage = downcast<RenderImage>(*renderer).cachedImage();
211 if (!tentativeCachedImage || tentativeCachedImage->errorOccurred())
212 return;
213
214 image = tentativeCachedImage->imageForRenderer(renderer);
215 if (!image)
216 return;
217
218 cachedImage = tentativeCachedImage;
219}

230String Editor::plainTextFromPasteboard(const PasteboardPlainText& text)
231{
232 auto string = text.text;
233
234 // FIXME: It's not clear this is 100% correct since we know -[NSURL URLWithString:] does not handle
235 // all the same cases we handle well in the URL code for creating an NSURL.
236 if (text.isURL)
237 string = WTF::userVisibleString([NSURL URLWithString:string]);
238
239 // FIXME: WTF should offer a non-Mac-specific way to convert string to precomposed form so we can do it for all platforms.
240 return [(NSString *)string precomposedStringWithCanonicalMapping];
241}

243void Editor::writeImageToPasteboard(Pasteboard& pasteboard, Element& imageElement, const URL& url, const String& title)
244{
245 PasteboardImage pasteboardImage;
246
247 CachedImage* cachedImage = nullptr;
248 getImage(imageElement, pasteboardImage.image, cachedImage);
249 if (!pasteboardImage.image)
250 return;
251 ASSERT(cachedImage);
252
253 if (!pasteboard.isStatic())
254 pasteboardImage.dataInWebArchiveFormat = imageInWebArchiveFormat(imageElement);
255
256 if (auto imageRange = makeRangeSelectingNode(imageElement)) {
257 auto serializeComposedTree = m_document.settings().selectionAcrossShadowBoundariesEnabled() ? SerializeComposedTree::Yes : SerializeComposedTree::No;
258 pasteboardImage.dataInHTMLFormat = serializePreservingVisualAppearance(VisibleSelection { *imageRange }, ResolveURLs::YesExcludingLocalFileURLsForPrivacy, serializeComposedTree);
259 }
260
261 pasteboardImage.url.url = url;
262 pasteboardImage.url.title = title;
263 pasteboardImage.url.userVisibleForm = WTF::userVisibleString(pasteboardImage.url.url);
264 pasteboardImage.resourceData = cachedImage->resourceBuffer();
265 pasteboardImage.resourceMIMEType = cachedImage->response().mimeType();
266
267 pasteboard.write(pasteboardImage);
268}
