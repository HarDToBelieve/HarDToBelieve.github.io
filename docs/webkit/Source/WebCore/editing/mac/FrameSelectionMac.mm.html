37void FrameSelection::notifyAccessibilityForSelectionChange(const AXTextStateChangeIntent& intent)
38{
39 if (m_selection.start().isNotNull() && m_selection.end().isNotNull()) {
40 if (AXObjectCache* cache = m_document->existingAXObjectCache())
41 cache->postTextStateChangeNotification(m_selection.start(), intent, m_selection);
42 }
43
44#if !PLATFORM(IOS_FAMILY)
45 // if zoom feature is enabled, insertion point changes should update the zoom
46 if (!m_selection.isCaret())
47 return;
48
49 RenderView* renderView = m_document->renderView();
50 if (!renderView)
51 return;
52 FrameView* frameView = m_document->view();
53 if (!frameView)
54 return;
55
56 IntRect selectionRect = absoluteCaretBounds();
57 IntRect viewRect = snappedIntRect(renderView->viewRect());
58
59 selectionRect = frameView->contentsToScreen(selectionRect);
60 viewRect = frameView->contentsToScreen(viewRect);
61
62 if (!m_document->page())
63 return;
64
65 m_document->page()->chrome().client().changeUniversalAccessZoomFocus(viewRect, selectionRect);
66#endif // !PLATFORM(IOS_FAMILY)
67}
