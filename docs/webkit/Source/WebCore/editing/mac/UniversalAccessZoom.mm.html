32namespace WebCore {
33
34void changeUniversalAccessZoomFocus(const IntRect& viewRect, const IntRect& selectionRect)
35{
36#if PLATFORM(MAC)
37 if (!UAZoomEnabled())
38 return;
39
40 auto cgCaretRect = NSRectToCGRect(toUserSpaceForPrimaryScreen(selectionRect));
41 auto cgViewRect = NSRectToCGRect(toUserSpaceForPrimaryScreen(viewRect));
42
43 UAZoomChangeFocus(&cgViewRect, &cgCaretRect, kUAZoomFocusTypeInsertionPoint);
44#else
45 UNUSED_PARAM(viewRect);
46 UNUSED_PARAM(selectionRect);
47#endif
48}
