119static void completeURLs(DocumentFragment* fragment, const String& baseURL)
120{
121 Vector<AttributeChange> changes;
122
123 URL parsedBaseURL({ }, baseURL);
124
125 for (auto& element : descendantsOfType<Element>(*fragment)) {
126 if (!element.hasAttributes())
127 continue;
128 for (const Attribute& attribute : element.attributesIterator()) {
129 if (element.attributeContainsURL(attribute) && !attribute.value().isEmpty())
130 changes.append(AttributeChange(&element, attribute.name(), element.completeURLsInAttributeValue(parsedBaseURL, attribute)));
131 }
132 }
133
134 for (auto& change : changes)
135 change.apply();
136}

138void replaceSubresourceURLs(Ref<DocumentFragment>&& fragment, HashMap<AtomString, AtomString>&& replacementMap)
139{
140 Vector<AttributeChange> changes;
141 for (auto& element : descendantsOfType<Element>(fragment)) {
142 if (!element.hasAttributes())
143 continue;
144 for (const Attribute& attribute : element.attributesIterator()) {
145 // FIXME: This won't work for srcset.
146 if (element.attributeContainsURL(attribute) && !attribute.value().isEmpty()) {
147 auto replacement = replacementMap.get(attribute.value());
148 if (!replacement.isNull())
149 changes.append({ &element, attribute.name(), replacement });
150 }
151 }
152 }
153 for (auto& change : changes)
154 change.apply();
155}

162void removeSubresourceURLAttributes(Ref<DocumentFragment>&& fragment, WTF::Function<bool(const URL&)> shouldRemoveURL)
163{
164 Vector<ElementAttribute> attributesToRemove;
165 for (auto& element : descendantsOfType<Element>(fragment)) {
166 if (!element.hasAttributes())
167 continue;
168 for (const Attribute& attribute : element.attributesIterator()) {
169 // FIXME: This won't work for srcset.
170 if (element.attributeContainsURL(attribute) && !attribute.value().isEmpty()) {
171 URL url({ }, attribute.value());
172 if (shouldRemoveURL(url))
173 attributesToRemove.append({ element, attribute.name() });
174 }
175 }
176 }
177 for (auto& item : attributesToRemove)
178 item.element->removeAttribute(item.attributeName);
179}

181std::unique_ptr<Page> createPageForSanitizingWebContent()
182{
183 auto pageConfiguration = pageConfigurationWithEmptyClients(PAL::SessionID::defaultSessionID());
184
185 auto page = makeUnique<Page>(WTFMove(pageConfiguration));
186#if ENABLE(VIDEO)
187 page->settings().setMediaEnabled(false);
188#endif
189 page->settings().setScriptEnabled(false);
190 page->settings().setHTMLParserScriptingFlagPolicy(HTMLParserScriptingFlagPolicy::Enabled);
191 page->settings().setPluginsEnabled(false);
192 page->settings().setAcceleratedCompositingEnabled(false);
193
194 Frame& frame = page->mainFrame();
195 frame.setView(FrameView::create(frame, IntSize { 800, 600 }));
196 frame.init();
197
198 FrameLoader& loader = frame.loader();
199 static char markup[] = "<!DOCTYPE html><html><body></body></html>";
200 ASSERT(loader.activeDocumentLoader());
201 auto& writer = loader.activeDocumentLoader()->writer();
202 writer.setMIMEType("text/html");
203 writer.begin();
204 writer.insertDataSynchronously(String(markup));
205 writer.end();
206 RELEASE_ASSERT(page->mainFrame().document()->body());
207
208 return page;
209}

211String sanitizeMarkup(const String& rawHTML, MSOListQuirks msoListQuirks, std::optional<WTF::Function<void(DocumentFragment&)>> fragmentSanitizer)
212{
213 auto page = createPageForSanitizingWebContent();
214 Document* stagingDocument = page->mainFrame().document();
215 ASSERT(stagingDocument);
216
217 auto fragment = createFragmentFromMarkup(*stagingDocument, rawHTML, emptyString(), DisallowScriptingAndPluginContent);
218
219 if (fragmentSanitizer)
220 (*fragmentSanitizer)(fragment);
221
222 return sanitizedMarkupForFragmentInDocument(WTFMove(fragment), *stagingDocument, msoListQuirks, rawHTML);
223}

412String StyledMarkupAccumulator::takeResults()
413{
414 CheckedUint32 length = this->length();
415 for (auto& string : m_reversedPrecedingMarkup)
416 length += string.length();
417 StringBuilder result;
418 result.reserveCapacity(length);
419 for (auto& string : makeReversedRange(m_reversedPrecedingMarkup))
420 result.append(string);
421 result.append(takeMarkup());
422 // Remove '\0' characters because they are not visibly rendered to the user.
423 return result.toString().replaceWithLiteral('\0', "");
424}

426void StyledMarkupAccumulator::appendText(StringBuilder& out, const Text& text)
427{
428 const bool parentIsTextarea = is<HTMLTextAreaElement>(text.parentElement());
429 const bool wrappingSpan = shouldApplyWrappingStyle(text) && !parentIsTextarea;
430 if (wrappingSpan) {
431 auto wrappingStyle = m_wrappingStyle->copy();
432 // FIXME: <rdar://problem/5371536> Style rules that match pasted content can change it's appearance
433 // Make sure spans are inline style in paste side e.g. span { display: block }.
434 wrappingStyle->forceInline();
435 // FIXME: Should this be included in forceInline?
436 wrappingStyle->style()->setProperty(CSSPropertyFloat, CSSValueNone);
437
438 appendStyleNodeOpenTag(out, wrappingStyle->style(), text.document());
439 }
440
441 if (!shouldAnnotate() || parentIsTextarea) {
442 auto content = textContentRespectingRange(text);
443 appendCharactersReplacingEntities(out, content, 0, content.length(), entityMaskForText(text));
444 } else {
445 const bool useRenderedText = !enclosingElementWithTag(firstPositionInNode(const_cast<Text*>(&text)), selectTag);
446 String content = useRenderedText ? renderedTextRespectingRange(text) : textContentRespectingRange(text);
447 StringBuilder buffer;
448 appendCharactersReplacingEntities(buffer, content, 0, content.length(), EntityMaskInPCDATA);
449 out.append(convertHTMLTextToInterchangeFormat(buffer.toString(), &text));
450 }
451
452 if (wrappingSpan)
453 out.append(styleNodeCloseTag());
454}

456String StyledMarkupAccumulator::renderedTextRespectingRange(const Text& text)
457{
458 TextIteratorBehaviors behaviors;
459 Position start = &text == m_start.containerNode() ? m_start : firstPositionInNode(const_cast<Text*>(&text));
460 Position end;
461 if (&text == m_end.containerNode())
462 end = m_end;
463 else {
464 end = lastPositionInNode(const_cast<Text*>(&text));
465 if (!m_end.isNull())
466 behaviors.add(TextIteratorBehavior::BehavesAsIfNodesFollowing);
467 }
468
469 auto range = makeSimpleRange(start, end);
470 return range ? plainText(*range, behaviors) : emptyString();
471}
472
473String StyledMarkupAccumulator::textContentRespectingRange(const Text& text)
474{
475 if (m_start.isNull() && m_end.isNull())
476 return text.data();
477
478 unsigned start = 0;
479 unsigned end = std::numeric_limits<unsigned>::max();
480 if (&text == m_start.containerNode())
481 start = m_start.offsetInContainerNode();
482 if (&text == m_end.containerNode())
483 end = m_end.offsetInContainerNode();
484 ASSERT(start < end);
485 return text.data().substring(start, end - start);
486}

488void StyledMarkupAccumulator::appendCustomAttributes(StringBuilder& out, const Element& element, Namespaces* namespaces)
489{
490#if ENABLE(ATTACHMENT_ELEMENT)
491 if (!RuntimeEnabledFeatures::sharedFeatures().attachmentElementEnabled())
492 return;
493
494 if (is<HTMLAttachmentElement>(element)) {
495 auto& attachment = downcast<HTMLAttachmentElement>(element);
496 appendAttribute(out, element, { webkitattachmentidAttr, attachment.uniqueIdentifier() }, namespaces);
497 if (auto* file = attachment.file()) {
498 // These attributes are only intended for File deserialization, and are removed from the generated attachment
499 // element after we've deserialized and set its backing File, in restoreAttachmentElementsInFragment.
500 appendAttribute(out, element, { webkitattachmentpathAttr, file->path() }, namespaces);
501 appendAttribute(out, element, { webkitattachmentbloburlAttr, file->url().string() }, namespaces);
502 }
503 } else if (is<HTMLImageElement>(element)) {
504 if (auto attachment = downcast<HTMLImageElement>(element).attachmentElement())
505 appendAttribute(out, element, { webkitattachmentidAttr, attachment->uniqueIdentifier() }, namespaces);
506 }
507#else
508 UNUSED_PARAM(out);
509 UNUSED_PARAM(element);
510 UNUSED_PARAM(namespaces);
511#endif
512}

538void StyledMarkupAccumulator::appendStartTag(StringBuilder& out, const Element& element, bool addDisplayInline, RangeFullySelectsNode rangeFullySelectsNode)
539{
540 const bool documentIsHTML = element.document().isHTMLDocument();
541
542 auto replacementType = spanReplacementForElement(element);
543 if (UNLIKELY(replacementType != SpanReplacementType::None))
544 out.append("<span");
545 else
546 appendOpenTag(out, element, nullptr);
547
548 appendCustomAttributes(out, element, nullptr);
549
550 const bool shouldAnnotateOrForceInline = element.isHTMLElement() && (shouldAnnotate() || addDisplayInline);
551 bool shouldOverrideStyleAttr = (shouldAnnotateOrForceInline || shouldApplyWrappingStyle(element) || replacementType != SpanReplacementType::None) && !shouldPreserveMSOListStyleForElement(element);
552 if (element.hasAttributes()) {
553 for (const Attribute& attribute : element.attributesIterator()) {
554 // We'll handle the style attribute separately, below.
555 if (attribute.name() == styleAttr && shouldOverrideStyleAttr)
556 continue;
557 if (element.isEventHandlerAttribute(attribute) || element.isJavaScriptURLAttribute(attribute))
558 continue;
559#if ENABLE(DATA_DETECTION)
560 if (replacementType == SpanReplacementType::DataDetector && DataDetection::isDataDetectorAttribute(attribute.name()))
561 continue;
562#endif
563 appendAttribute(out, element, attribute, 0);
564 }
565 }
566
567 if (shouldOverrideStyleAttr) {
568 RefPtr<EditingStyle> newInlineStyle;
569
570 if (shouldApplyWrappingStyle(element)) {
571 newInlineStyle = m_wrappingStyle->copy();
572 newInlineStyle->removePropertiesInElementDefaultStyle(*const_cast<Element*>(&element));
573 newInlineStyle->removeStyleConflictingWithStyleOfNode(*const_cast<Element*>(&element));
574 } else
575 newInlineStyle = EditingStyle::create();
576
577 if (replacementType == SpanReplacementType::Slot)
578 newInlineStyle->addDisplayContents();
579
580 if (is<StyledElement>(element) && downcast<StyledElement>(element).inlineStyle())
581 newInlineStyle->overrideWithStyle(*downcast<StyledElement>(element).inlineStyle());
582
583#if ENABLE(DATA_DETECTION)
584 if (replacementType == SpanReplacementType::DataDetector && newInlineStyle->style())
585 newInlineStyle->style()->removeProperty(CSSPropertyTextDecorationColor);
586#endif
587
588 if (shouldAnnotateOrForceInline) {
589 if (shouldAnnotate())
590 newInlineStyle->mergeStyleFromRulesForSerialization(downcast<HTMLElement>(*const_cast<Element*>(&element)), m_standardFontFamilySerializationMode);
591
592 if (addDisplayInline)
593 newInlineStyle->forceInline();
594
595 if (m_needsPositionStyleConversion) {
596 m_needRelativeStyleWrapper |= newInlineStyle->convertPositionStyle();
597 m_needClearingDiv |= newInlineStyle->isFloating();
598 }
599
600 // If the node is not fully selected by the range, then we don't want to keep styles that affect its relationship to the nodes around it
601 // only the ones that affect it and the nodes within it.
602 if (rangeFullySelectsNode == DoesNotFullySelectNode && newInlineStyle->style())
603 newInlineStyle->style()->removeProperty(CSSPropertyFloat);
604 }
605
606 if (!newInlineStyle->isEmpty()) {
607 out.append(" style=\"");
608 appendAttributeValue(out, newInlineStyle->style()->asText(), documentIsHTML);
609 out.append('"');
610 }
611 }
612
613 appendCloseTag(out, element);
614}

624Node* StyledMarkupAccumulator::serializeNodes(const Position& start, const Position& end)
625{
626 ASSERT(start <= end);
627 auto startNode = start.firstNode();
628 Node* pastEnd = end.computeNodeAfterPosition();
629 if (!pastEnd && end.containerNode())
630 pastEnd = nextSkippingChildren(*end.containerNode());
631
632 if (!m_highestNodeToBeSerialized) {
633 Node* lastClosed = traverseNodesForSerialization(startNode.get(), pastEnd, NodeTraversalMode::DoNotEmitString);
634 m_highestNodeToBeSerialized = lastClosed;
635 }
636
637 if (m_highestNodeToBeSerialized && m_highestNodeToBeSerialized->parentNode())
638 m_wrappingStyle = EditingStyle::wrappingStyleForSerialization(*m_highestNodeToBeSerialized->parentNode(), shouldAnnotate(), m_standardFontFamilySerializationMode);
639
640 return traverseNodesForSerialization(startNode.get(), pastEnd, NodeTraversalMode::EmitString);
641}
642
643Node* StyledMarkupAccumulator::traverseNodesForSerialization(Node* startNode, Node* pastEnd, NodeTraversalMode traversalMode)
644{
645 const bool shouldEmit = traversalMode == NodeTraversalMode::EmitString;
646
647 m_inMSOList = false;
648
649 unsigned depth = 0;
650 auto enterNode = [&] (Node& node) {
651 if (UNLIKELY(m_shouldPreserveMSOList) && shouldEmit) {
652 if (appendNodeToPreserveMSOList(node))
653 return false;
654 }
655
656 bool isDisplayContents = is<Element>(node) && downcast<Element>(node).hasDisplayContents();
657 if (!node.renderer() && !isDisplayContents && !enclosingElementWithTag(firstPositionInOrBeforeNode(&node), selectTag))
658 return false;
659
660 ++depth;
661 if (shouldEmit)
662 startAppendingNode(node);
663
664 return true;
665 };
666
667 Node* lastClosed = nullptr;
668 auto exitNode = [&] (Node& node) {
669 bool closing = depth;
670 if (depth)
671 --depth;
672 if (shouldEmit) {
673 if (closing)
674 endAppendingNode(node);
675 else
676 wrapWithNode(node);
677 }
678 lastClosed = &node;
679 };
680
681 Node* lastNode = nullptr;
682 Node* next = nullptr;
683 for (auto* n = startNode; n != pastEnd; lastNode = n, n = next) {
684
685 Vector<Node*, 8> exitedAncestors;
686 next = nullptr;
687 if (auto* child = firstChild(*n))
688 next = child;
689 else if (auto* sibling = nextSibling(*n))
690 next = sibling;
691 else {
692 for (auto* ancestor = parentNode(*n); ancestor; ancestor = parentNode(*ancestor)) {
693 exitedAncestors.append(ancestor);
694 if (auto* sibling = nextSibling(*ancestor)) {
695 next = sibling;
696 break;
697 }
698 }
699 }
700 ASSERT(next || !pastEnd || n->containsIncludingShadowDOM(pastEnd));
701
702 if (isBlock(n) && canHaveChildrenForEditing(*n) && next == pastEnd) {
703 // Don't write out empty block containers that aren't fully selected.
704 continue;
705 }
706
707 bool didEnterNode = false;
708 if (!enterNode(*n))
709 next = nextSkippingChildren(*n);
710 else if (!hasChildNodes(*n))
711 exitNode(*n);
712 else
713 didEnterNode = true;
714
715 bool aboutToGoPastEnd = pastEnd && !didEnterNode && (!next || isDescendantOf(*pastEnd, *n));
716 if (aboutToGoPastEnd)
717 next = pastEnd;
718
719 for (auto* ancestor : exitedAncestors) {
720 if (!depth && next == pastEnd)
721 break;
722 exitNode(*ancestor);
723 }
724 }
725
726 ASSERT(lastNode || !depth);
727 if (depth) {
728 for (auto* ancestor = parentNode(pastEnd ? *pastEnd : *lastNode); ancestor && depth; ancestor = parentNode(*ancestor))
729 exitNode(*ancestor);
730 }
731
732 return lastClosed;
733}

735bool StyledMarkupAccumulator::appendNodeToPreserveMSOList(Node& node)
736{
737 if (is<Comment>(node)) {
738 auto& commentNode = downcast<Comment>(node);
739 if (!m_inMSOList && commentNode.data() == "[if !supportLists]")
740 m_inMSOList = true;
741 else if (m_inMSOList && commentNode.data() == "[endif]")
742 m_inMSOList = false;
743 else
744 return false;
745 startAppendingNode(commentNode);
746 return true;
747 }
748 if (is<HTMLStyleElement>(node)) {
749 auto* firstChild = node.firstChild();
750 if (!is<Text>(firstChild))
751 return false;
752
753 auto& textChild = downcast<Text>(*firstChild);
754 auto& styleContent = textChild.data();
755
756 const auto msoStyleDefinitionsStart = styleContent.find("/* Style Definitions */");
757 const auto msoListDefinitionsStart = styleContent.find("/* List Definitions */");
758 const auto lastListItem = styleContent.reverseFind("\n@list");
759 if (msoListDefinitionsStart == notFound || lastListItem == notFound)
760 return false;
761 const auto start = msoStyleDefinitionsStart != notFound && msoStyleDefinitionsStart < msoListDefinitionsStart ? msoStyleDefinitionsStart : msoListDefinitionsStart;
762
763 const auto msoListDefinitionsEnd = styleContent.find(";}\n", lastListItem);
764 if (msoListDefinitionsEnd == notFound || start >= msoListDefinitionsEnd)
765 return false;
766
767 append("<head><style class=\"" WebKitMSOListQuirksStyle "\">\n<!--\n",
768 StringView(textChild.data()).substring(start, msoListDefinitionsEnd - start + 3),
769 "\n-->\n</style></head>");
770
771 return true;
772 }
773 return false;
774}

834static Node* highestAncestorToWrapMarkup(const Position& start, const Position& end, Node& commonAncestor, AnnotateForInterchange annotate)
835{
836 Node* specialCommonAncestor = nullptr;
837 if (annotate == AnnotateForInterchange::Yes) {
838 // Include ancestors that aren't completely inside the range but are required to retain
839 // the structure and appearance of the copied markup.
840 specialCommonAncestor = ancestorToRetainStructureAndAppearance(&commonAncestor);
841
842 if (auto* parentListNode = enclosingNodeOfType(start, isListItem)) {
843 if (!editingIgnoresContent(*parentListNode) && VisibleSelection::selectionFromContentsOfNode(parentListNode) == VisibleSelection(start, end)) {
844 specialCommonAncestor = parentListNode->parentNode();
845 while (specialCommonAncestor && !isListHTMLElement(specialCommonAncestor))
846 specialCommonAncestor = specialCommonAncestor->parentNode();
847 }
848 }
849
850 // Retain the Mail quote level by including all ancestor mail block quotes.
851 if (Node* highestMailBlockquote = highestEnclosingNodeOfType(start, isMailBlockquote, CanCrossEditingBoundary))
852 specialCommonAncestor = highestMailBlockquote;
853 }
854
855 auto* checkAncestor = specialCommonAncestor ? specialCommonAncestor : &commonAncestor;
856 if (checkAncestor->renderer() && checkAncestor->renderer()->containingBlock()) {
857 Node* newSpecialCommonAncestor = highestEnclosingNodeOfType(firstPositionInNode(checkAncestor), &isElementPresentational, CanCrossEditingBoundary, checkAncestor->renderer()->containingBlock()->element());
858 if (newSpecialCommonAncestor)
859 specialCommonAncestor = newSpecialCommonAncestor;
860 }
861
862 // If a single tab is selected, commonAncestor will be a text node inside a tab span.
863 // If two or more tabs are selected, commonAncestor will be the tab span.
864 // In either case, if there is a specialCommonAncestor already, it will necessarily be above
865 // any tab span that needs to be included.
866 if (!specialCommonAncestor && isTabSpanTextNode(&commonAncestor))
867 specialCommonAncestor = commonAncestor.parentNode();
868 if (!specialCommonAncestor && isTabSpanNode(&commonAncestor))
869 specialCommonAncestor = &commonAncestor;
870
871 if (auto* enclosingAnchor = enclosingElementWithTag(firstPositionInNode(specialCommonAncestor ? specialCommonAncestor : &commonAncestor), aTag))
872 specialCommonAncestor = enclosingAnchor;
873
874 return specialCommonAncestor;
875}

877static String serializePreservingVisualAppearanceInternal(const Position& start, const Position& end, Vector<Node*>* nodes, ResolveURLs resolveURLs, SerializeComposedTree serializeComposedTree,
878 AnnotateForInterchange annotate, ConvertBlocksToInlines convertBlocksToInlines, StandardFontFamilySerializationMode standardFontFamilySerializationMode, MSOListMode msoListMode)
879{
880 static NeverDestroyed<const String> interchangeNewlineString(MAKE_STATIC_STRING_IMPL("<br class=\"" AppleInterchangeNewline "\">"));
881
882 if (!(start < end))
883 return emptyString();
884
885 auto commonAncestor = commonInclusiveAncestor(start, end);
886 if (!commonAncestor)
887 return emptyString();
888
889 auto& document = *start.document();
890 document.updateLayoutIgnorePendingStylesheets();
891
892 VisiblePosition visibleStart { start };
893 VisiblePosition visibleEnd { end };
894
895 auto body = makeRefPtr(enclosingElementWithTag(firstPositionInNode(commonAncestor), bodyTag));
896 RefPtr<Element> fullySelectedRoot;
897 // FIXME: Do this for all fully selected blocks, not just the body.
898 if (body && VisiblePosition(firstPositionInNode(body.get())) == visibleStart && VisiblePosition(lastPositionInNode(body.get())) == visibleEnd)
899 fullySelectedRoot = body;
900 bool needsPositionStyleConversion = body && fullySelectedRoot == body && document.settings().shouldConvertPositionStyleOnCopy();
901
902 Node* specialCommonAncestor = highestAncestorToWrapMarkup(start, end, *commonAncestor, annotate);
903
904 StyledMarkupAccumulator accumulator(start, end, nodes, resolveURLs, serializeComposedTree, annotate, standardFontFamilySerializationMode, msoListMode, needsPositionStyleConversion, specialCommonAncestor);
905
906 Position startAdjustedForInterchangeNewline = start;
907 if (annotate == AnnotateForInterchange::Yes && needInterchangeNewlineAfter(visibleStart)) {
908 if (visibleStart == visibleEnd.previous())
909 return interchangeNewlineString;
910
911 accumulator.append(interchangeNewlineString.get());
912 startAdjustedForInterchangeNewline = visibleStart.next().deepEquivalent();
913
914 if (!(startAdjustedForInterchangeNewline < end))
915 return interchangeNewlineString;
916 }
917
918 Node* lastClosed = accumulator.serializeNodes(startAdjustedForInterchangeNewline, end);
919
920 if (specialCommonAncestor && lastClosed) {
921 // Also include all of the ancestors of lastClosed up to this special ancestor.
922 for (ContainerNode* ancestor = accumulator.parentNode(*lastClosed); ancestor; ancestor = accumulator.parentNode(*ancestor)) {
923 if (ancestor == fullySelectedRoot && convertBlocksToInlines == ConvertBlocksToInlines::No) {
924 RefPtr<EditingStyle> fullySelectedRootStyle = styleFromMatchedRulesAndInlineDecl(*fullySelectedRoot);
925
926 // Bring the background attribute over, but not as an attribute because a background attribute on a div
927 // appears to have no effect.
928 if ((!fullySelectedRootStyle || !fullySelectedRootStyle->style() || !fullySelectedRootStyle->style()->getPropertyCSSValue(CSSPropertyBackgroundImage))
929 && fullySelectedRoot->hasAttributeWithoutSynchronization(backgroundAttr))
930 fullySelectedRootStyle->style()->setProperty(CSSPropertyBackgroundImage, "url('" + fullySelectedRoot->getAttribute(backgroundAttr) + "')");
931
932 if (fullySelectedRootStyle->style()) {
933 // Reset the CSS properties to avoid an assertion error in addStyleMarkup().
934 // This assertion is caused at least when we select all text of a <body> element whose
935 // 'text-decoration' property is "inherit", and copy it.
936 if (!propertyMissingOrEqualToNone(fullySelectedRootStyle->style(), CSSPropertyTextDecoration))
937 fullySelectedRootStyle->style()->setProperty(CSSPropertyTextDecoration, CSSValueNone);
938 if (!propertyMissingOrEqualToNone(fullySelectedRootStyle->style(), CSSPropertyWebkitTextDecorationsInEffect))
939 fullySelectedRootStyle->style()->setProperty(CSSPropertyWebkitTextDecorationsInEffect, CSSValueNone);
940 accumulator.wrapWithStyleNode(fullySelectedRootStyle->style(), document, true);
941 }
942 } else {
943 // Since this node and all the other ancestors are not in the selection we want to set RangeFullySelectsNode to DoesNotFullySelectNode
944 // so that styles that affect the exterior of the node are not included.
945 accumulator.wrapWithNode(*ancestor, convertBlocksToInlines == ConvertBlocksToInlines::Yes, StyledMarkupAccumulator::DoesNotFullySelectNode);
946 }
947 if (nodes)
948 nodes->append(ancestor);
949
950 if (ancestor == specialCommonAncestor)
951 break;
952 }
953 }
954
955 if (accumulator.needRelativeStyleWrapper() && needsPositionStyleConversion) {
956 if (accumulator.needClearingDiv())
957 accumulator.append("<div style=\"clear: both;\"></div>");
958 RefPtr<EditingStyle> positionRelativeStyle = styleFromMatchedRulesAndInlineDecl(*body);
959 positionRelativeStyle->style()->setProperty(CSSPropertyPosition, CSSValueRelative);
960 accumulator.wrapWithStyleNode(positionRelativeStyle->style(), document, true);
961 }
962
963 // FIXME: The interchange newline should be placed in the block that it's in, not after all of the content, unconditionally.
964 if (annotate == AnnotateForInterchange::Yes && needInterchangeNewlineAfter(visibleEnd.previous()))
965 accumulator.append(interchangeNewlineString.get());
966
967#if PLATFORM(COCOA)
968 // On Cocoa platforms, this markup is eventually persisted to the pasteboard and read back as UTF-8 data,
969 // so this meta tag is needed for clients that read this data in the future from the pasteboard and load it.
970 accumulator.prependMetaCharsetUTF8TagIfNonASCIICharactersArePresent();
971#endif
972
973 return accumulator.takeResults();
974}

1027{
1028#if ENABLE(ATTACHMENT_ELEMENT)
1029 if (!RuntimeEnabledFeatures::sharedFeatures().attachmentElementEnabled())
1030 return;
1031
1032 // When creating a fragment we must strip the webkit-attachment-path attribute after restoring the File object.
1033 Vector<Ref<HTMLAttachmentElement>> attachments;

1035 attachments.append(attachment);
1036

1038 attachment->setUniqueIdentifier(attachment->attributeWithoutSynchronization(webkitattachmentidAttr));
1039
1040 auto attachmentPath = attachment->attachmentPath();

1042 if (!attachmentPath.isEmpty())
1043 attachment->setFile(File::create(fragment.ownerDocument(), attachmentPath));
1044 else if (!blobURL.isEmpty())

1048 attachment->removeAttribute(webkitattachmentidAttr);
1049 attachment->removeAttribute(webkitattachmentpathAttr);
1050 attachment->removeAttribute(webkitattachmentbloburlAttr);

1054 for (auto& image : descendantsOfType<HTMLImageElement>(fragment))
1055 images.append(image);
1056

1059 if (attachmentIdentifier.isEmpty())
1060 continue;
1061

1067#else
1068 UNUSED_PARAM(fragment);
1069#endif
1070}

1092static void fillContainerFromString(ContainerNode& paragraph, const String& string)
1093{
1094 Document& document = paragraph.document();
1095
1096 if (string.isEmpty()) {
1097 paragraph.appendChild(createBlockPlaceholderElement(document));
1098 return;
1099 }
1100
1101 ASSERT(string.find('\n') == notFound);
1102
1103 Vector<String> tabList = string.splitAllowingEmptyEntries('\t');
1104 String tabText = emptyString();
1105 bool first = true;
1106 size_t numEntries = tabList.size();
1107 for (size_t i = 0; i < numEntries; ++i) {
1108 const String& s = tabList[i];
1109
1110 // append the non-tab textual part
1111 if (!s.isEmpty()) {
1112 if (!tabText.isEmpty()) {
1113 paragraph.appendChild(createTabSpanElement(document, tabText));
1114 tabText = emptyString();
1115 }
1116 Ref<Node> textNode = document.createTextNode(stringWithRebalancedWhitespace(s, first, i + 1 == numEntries));
1117 paragraph.appendChild(textNode);
1118 }
1119
1120 // there is a tab after every entry, except the last entry
1121 // (if the last character is a tab, the list gets an extra empty entry)
1122 if (i + 1 != numEntries)
1123 tabText.append('\t');
1124 else if (!tabText.isEmpty())
1125 paragraph.appendChild(createTabSpanElement(document, tabText));
1126
1127 first = false;
1128 }
1129}

1161Ref<DocumentFragment> createFragmentFromText(const SimpleRange& context, const String& text)
1162{
1163 auto& document = context.start.document();
1164 auto fragment = document.createDocumentFragment();
1165
1166 if (text.isEmpty())
1167 return fragment;
1168
1169 String string = text;
1170 string.replace("\r\n", "\n");
1171 string.replace('\r', '\n');
1172
1173 auto createHTMLBRElement = [&document]() {
1174 auto element = HTMLBRElement::create(document);
1175 element->setAttributeWithoutSynchronization(classAttr, AppleInterchangeNewline);
1176 return element;
1177 };
1178
1179 if (contextPreservesNewline(context)) {
1180 fragment->appendChild(document.createTextNode(string));
1181 if (string.endsWith('\n')) {
1182 fragment->appendChild(createHTMLBRElement());
1183 }
1184 return fragment;
1185 }
1186
1187 // A string with no newlines gets added inline, rather than being put into a paragraph.
1188 if (string.find('\n') == notFound) {
1189 fillContainerFromString(fragment, string);
1190 return fragment;
1191 }
1192
1193 if (string.length() == 1 && string[0] == '\n') {
1194 // This is a single newline char, thus just create one HTMLBRElement.
1195 fragment->appendChild(createHTMLBRElement());
1196 return fragment;
1197 }
1198
1199 // Break string into paragraphs. Extra line breaks turn into empty paragraphs.
1200 auto start = makeDeprecatedLegacyPosition(context.start);
1201 auto block = enclosingBlock(start.firstNode().get());
1202 bool useClonesOfEnclosingBlock = block
1203 && !block->hasTagName(bodyTag)
1204 && !block->hasTagName(htmlTag)
1205 // Avoid using table as paragraphs due to its special treatment in Position::upstream/downstream.
1206 && !isRenderedTable(block)
1207 && block != editableRootForPosition(start);
1208 bool useLineBreak = enclosingTextFormControl(start);
1209
1210 Vector<String> list = string.splitAllowingEmptyEntries('\n');
1211 size_t numLines = list.size();
1212 for (size_t i = 0; i < numLines; ++i) {
1213 const String& s = list[i];
1214
1215 RefPtr<Element> element;
1216 if (s.isEmpty() && i + 1 == numLines) {
1217 // For last line, use the "magic BR" rather than a P.
1218 element = createHTMLBRElement();
1219 } else if (useLineBreak) {
1220 element = HTMLBRElement::create(document);
1221 fillContainerFromString(fragment, s);
1222 } else {
1223 if (useClonesOfEnclosingBlock)
1224 element = block->cloneElementWithoutChildren(document);
1225 else
1226 element = createDefaultParagraphElement(document);
1227 fillContainerFromString(*element, s);
1228 }
1229 fragment->appendChild(*element);
1230 }
1231 return fragment;
1232}

1256 ASSERT(!fragment->hasChildNodes());

1258 fragment->parseHTML(markup, &contextElement, parserContentPolicy);

1260 }

1273RefPtr<DocumentFragment> createFragmentForTransformToFragment(Document& outputDoc, const String& sourceString, const String& sourceMIMEType)
1274{
1275 RefPtr<DocumentFragment> fragment = outputDoc.createDocumentFragment();
1276
1277 if (sourceMIMEType == "text/html") {
1278 // As far as I can tell, there isn't a spec for how transformToFragment is supposed to work.
1279 // Based on the documentation I can find, it looks like we want to start parsing the fragment in the InBody insertion mode.
1280 // Unfortunately, that's an implementation detail of the parser.
1281 // We achieve that effect here by passing in a fake body element as context for the fragment.
1282 auto fakeBody = HTMLBodyElement::create(outputDoc);
1283 fragment->parseHTML(sourceString, fakeBody.ptr());
1284 } else if (sourceMIMEType == "text/plain")
1285 fragment->parserAppendChild(Text::create(outputDoc, sourceString));
1286 else {
1287 bool successfulParse = fragment->parseXML(sourceString, 0);
1288 if (!successfulParse)
1289 return nullptr;
1290 }
1291
1292 // FIXME: Do we need to mess with URLs here?
1293
1294 return fragment;

1326static void removeElementFromFragmentPreservingChildren(DocumentFragment& fragment, HTMLElement& element)
1327{

1338{

1340 if (result.hasException())
1341 return result.releaseException();
1342
1343 auto fragment = result.releaseReturnValue();

1345 // We need to pop <html> and <body> elements and remove <head> to
1346 // accommodate folks passing complete HTML documents to make the
1347 // child of an element.
1348 auto toRemove = collectElementsToRemoveFromFragment(fragment);
1349 for (auto& element : toRemove)
1350 removeElementFromFragmentPreservingChildren(fragment, element);

1376ExceptionOr<void> replaceChildrenWithFragment(ContainerNode& container, Ref<DocumentFragment>&& fragment)
1377{
1378 Ref<ContainerNode> containerNode(container);
1379 ChildListMutationScope mutation(containerNode);
1380
1381 if (!fragment->firstChild()) {
1382 containerNode->removeChildren();
1383 return { };
1384 }
1385
1386 auto* containerChild = containerNode->firstChild();
1387 if (containerChild && !containerChild->nextSibling()) {
1388 if (is<Text>(*containerChild) && hasOneTextChild(fragment) && canUseSetDataOptimization(downcast<Text>(*containerChild), mutation)) {
1389 downcast<Text>(*containerChild).setData(downcast<Text>(*fragment->firstChild()).data());
1390 return { };
1391 }
1392
1393 return containerNode->replaceChild(fragment, *containerChild);
1394 }
1395
1396 containerNode->removeChildren();
1397 auto result = containerNode->appendChild(fragment);
1398 ASSERT(!fragment->hasChildNodes());
1399 ASSERT(!fragment->wrapper());
1400 return result;
1401}
