94Blob::Blob(ScriptExecutionContext& context, Vector<BlobPartVariant>&& blobPartVariants, const BlobPropertyBag& propertyBag)
95 : ActiveDOMObject(&context)
96 , m_internalURL(BlobURL::createInternalURL())
97 , m_type(normalizedContentType(propertyBag.type))
98{
99 BlobBuilder builder(propertyBag.endings);
100 for (auto& blobPartVariant : blobPartVariants) {
101 WTF::switchOn(blobPartVariant,
102 [&] (auto& part) {
103 builder.append(WTFMove(part));
104 }
105 );
106 }
107
108 ThreadableBlobRegistry::registerBlobURL(m_internalURL, builder.finalize(), m_type);
109}

111Blob::Blob(ScriptExecutionContext* context, Vector<uint8_t>&& data, const String& contentType)
112 : ActiveDOMObject(context)
113 , m_type(contentType)
114 , m_size(data.size())
115{
116 Vector<BlobPart> blobParts;
117 blobParts.append(BlobPart(WTFMove(data)));
118 m_internalURL = BlobURL::createInternalURL();
119 ThreadableBlobRegistry::registerBlobURL(m_internalURL, WTFMove(blobParts), contentType);
120}

131Blob::Blob(DeserializationContructor, ScriptExecutionContext* context, const URL& srcURL, const String& type, std::optional<unsigned long long> size, const String& fileBackedPath)
132 : ActiveDOMObject(context)
133 , m_type(normalizedContentType(type))
134 , m_size(size)
135{
136 m_internalURL = BlobURL::createInternalURL();
137 if (fileBackedPath.isEmpty())

141}

152Blob::~Blob()
153{
154 while (!m_blobLoaders.isEmpty())
155 (*m_blobLoaders.begin())->cancel();
156
157 ThreadableBlobRegistry::unregisterBlobURL(m_internalURL);
158}

160unsigned long long Blob::size() const
161{
162 if (!m_size) {
163 // FIXME: JavaScript cannot represent sizes as large as unsigned long long, we need to
164 // come up with an exception to throw if file size is not representable.
165 unsigned long long actualSize = ThreadableBlobRegistry::blobSize(m_internalURL);
166 m_size = isInBounds<long long>(actualSize) ? actualSize : 0;
167 }
168
169 return *m_size;
170}

172bool Blob::isValidContentType(const String& contentType)
173{
174 // FIXME: Do we really want to treat the empty string and null string as valid content types?
175 unsigned length = contentType.length();
176 for (unsigned i = 0; i < length; ++i) {
177 if (contentType[i] < 0x20 || contentType[i] > 0x7e)
178 return false;
179 }
180 return true;
181}

183String Blob::normalizedContentType(const String& contentType)
184{
185 if (!isValidContentType(contentType))
186 return emptyString();
187 return contentType.convertToASCIILowercase();
188}

190void Blob::loadBlob(ScriptExecutionContext& context, FileReaderLoader::ReadType readType, CompletionHandler<void(BlobLoader&)>&& completionHandler)
191{
192 auto blobLoader = makeUnique<BlobLoader>([this, pendingActivity = makePendingActivity(*this), completionHandler = WTFMove(completionHandler)](BlobLoader& blobLoader) mutable {
193 completionHandler(blobLoader);
194 m_blobLoaders.take(&blobLoader);
195 });
196
197 blobLoader->start(*this, &context, readType);
198
199 if (blobLoader->isLoading())
200 m_blobLoaders.add(WTFMove(blobLoader));
201}

203void Blob::text(ScriptExecutionContext& context, Ref<DeferredPromise>&& promise)
204{
205 loadBlob(context, FileReaderLoader::ReadAsText, [promise = WTFMove(promise)](BlobLoader& blobLoader) mutable {
206 if (auto optionalErrorCode = blobLoader.errorCode()) {
207 promise->reject(Exception { *optionalErrorCode });
208 return;
209 }
210 promise->resolve<IDLDOMString>(blobLoader.stringResult());
211 });
212}

214void Blob::arrayBuffer(ScriptExecutionContext& context, Ref<DeferredPromise>&& promise)
215{
216 loadBlob(context, FileReaderLoader::ReadAsArrayBuffer, [promise = WTFMove(promise)](BlobLoader& blobLoader) mutable {
217 if (auto optionalErrorCode = blobLoader.errorCode()) {
218 promise->reject(Exception { *optionalErrorCode });
219 return;
220 }
221 auto arrayBuffer = blobLoader.arrayBufferResult();
222 if (!arrayBuffer) {
223 promise->reject(Exception { InvalidStateError });
224 return;
225 }
226 promise->resolve<IDLArrayBuffer>(*arrayBuffer);
227 });
228}

230ExceptionOr<Ref<ReadableStream>> Blob::stream(ScriptExecutionContext& scriptExecutionContext)
231{
232 class BlobStreamSource : public FileReaderLoaderClient, public ReadableStreamSource {
233 public:
234 BlobStreamSource(ScriptExecutionContext& scriptExecutionContext, Blob& blob)
235 : m_loader(makeUniqueRef<FileReaderLoader>(FileReaderLoader::ReadType::ReadAsArrayBuffer, this))
236 {
237 m_loader->start(&scriptExecutionContext, blob);
238 }
239
240 private:
241 // ReadableStreamSource
242 void setActive() final { }
243 void setInactive() final { }
244 void doStart() final
245 {
246 m_isStarted = true;
247 if (m_exception)
248 controller().error(*m_exception);
249 }
250
251 void doPull() final { }
252 void doCancel() final
253 {
254 m_loader->cancel();
255 }
256
257 // FileReaderLoaderClient
258 void didStartLoading() final { }
259 void didReceiveData() final
260 {
261 auto result = m_loader->arrayBufferResult();
262 if (!result)
263 return;
264
265 if (m_loader->isCompleted() && !m_bytesRead)
266 controller().enqueue(WTFMove(result));
267 else {
268 auto bytesLoaded = m_loader->bytesLoaded();
269 controller().enqueue(result->slice(m_bytesRead, bytesLoaded));
270 m_bytesRead = bytesLoaded;
271 }
272 }
273 void didFinishLoading() final
274 {
275 controller().close();
276 }
277 void didFail(ExceptionCode code) final
278 {
279 Exception exception { code };
280 if (!m_isStarted) {
281 m_exception = WTFMove(exception);
282 return;
283 }
284 controller().error(exception);
285 }
286
287 UniqueRef<FileReaderLoader> m_loader;
288 size_t m_bytesRead { 0 };
289 bool m_isStarted { false };
290 std::optional<Exception> m_exception;
291 };
292
293 auto* globalObject = scriptExecutionContext.globalObject();
294 if (!globalObject)
295 return Exception { InvalidStateError };
296 return ReadableStream::create(*globalObject, adoptRef(*new BlobStreamSource(scriptExecutionContext, *this)));
297}

301{
302 // FIXME: Do we really want to treat the empty string and null string as valid content types?
303 unsigned length = contentType.length();
304 for (size_t i = 0; i < length; ++i) {

313bool Blob::isNormalizedContentType(const CString& contentType)
314{
315 // FIXME: Do we really want to treat the empty string and null string as valid content types?
316 size_t length = contentType.length();
317 const char* characters = contentType.data();
318 for (size_t i = 0; i < length; ++i) {
319 if (characters[i] < 0x20 || characters[i] > 0x7e)
320 return false;
321 if (isASCIIUpper(characters[i]))
322 return false;
323 }
324 return true;
