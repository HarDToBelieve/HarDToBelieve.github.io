26#pragma once
27
28#include "Blob.h"
29#include "Document.h"
30#include "ExceptionCode.h"
31#include "FileReaderLoader.h"
32#include "FileReaderLoaderClient.h"
33#include "SharedBuffer.h"
34#include <JavaScriptCore/ArrayBuffer.h>
35#include <wtf/CompletionHandler.h>
36
37namespace WebCore {
38
39class BlobLoader final : public FileReaderLoaderClient {
40 WTF_MAKE_FAST_ALLOCATED;
41public:
42 explicit BlobLoader(CompletionHandler<void(BlobLoader&)>&&);
43 ~BlobLoader();
44
45 void start(Blob&, ScriptExecutionContext*, FileReaderLoader::ReadType);
46
47 void cancel();
48 bool isLoading() const { return m_loader && m_completionHandler; }
49 String stringResult() const { return m_loader ? m_loader->stringResult() : String(); }
50 RefPtr<JSC::ArrayBuffer> arrayBufferResult() const { return m_loader ? m_loader->arrayBufferResult() : nullptr; }
51 std::optional<ExceptionCode> errorCode() const { return m_loader ? m_loader->errorCode() : std::nullopt; }
52
53private:
54 void didStartLoading() final { }
55 void didReceiveData() final { }
56
57 void didFinishLoading() final;
58 void didFail(ExceptionCode errorCode) final;
59 void complete();
60
61 std::unique_ptr<FileReaderLoader> m_loader;
62 CompletionHandler<void(BlobLoader&)> m_completionHandler;
63};
64
65inline BlobLoader::BlobLoader(CompletionHandler<void(BlobLoader&)>&& completionHandler)
66 : m_completionHandler(WTFMove(completionHandler))
67{
68}
69
70inline BlobLoader::~BlobLoader()
71{
72 if (isLoading())
73 cancel();
74}
75
76inline void BlobLoader::cancel()
77{
78 if (m_loader)
79 m_loader->cancel();
80
81 if (m_completionHandler)
82 m_completionHandler(*this);
83}
84
85inline void BlobLoader::start(Blob& blob, ScriptExecutionContext* context, FileReaderLoader::ReadType readType)
86{
87 ASSERT(!m_loader);
88 m_loader = makeUnique<FileReaderLoader>(readType, this);
89 m_loader->start(context, blob);
90}
91
92inline void BlobLoader::didFinishLoading()
93{
94 m_completionHandler(*this);
95}
96
97inline void BlobLoader::didFail(ExceptionCode)
98{
99 m_completionHandler(*this);
100}
101
102} // namespace WebCore
