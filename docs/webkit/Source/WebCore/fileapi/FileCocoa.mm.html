39bool File::shouldReplaceFile(const String& path)
40{
41 if (path.isEmpty())
42 return false;
43
44 NSError *error;
45 NSURL *pathURL = [NSURL URLByResolvingAliasFileAtURL:[NSURL fileURLWithPath:path isDirectory:NO] options:NSURLBookmarkResolutionWithoutUI error:&error];
46 if (!pathURL) {
47 LOG_ERROR("Failed to resolve alias at path %s with error %@.\n", path.utf8().data(), error);
48 return false;
49 }
50
51 NSString *uti;
52 if (![pathURL getResourceValue:&uti forKey:NSURLTypeIdentifierKey error:&error]) {
53 LOG_ERROR("Failed to get type identifier of resource at URL %@ with error %@.\n", pathURL, error);
54 return false;
55 }
56
57ALLOW_DEPRECATED_DECLARATIONS_BEGIN
58 return UTTypeConformsTo((__bridge CFStringRef)uti, kUTTypePackage);
59ALLOW_DEPRECATED_DECLARATIONS_END
60}

62void File::computeNameAndContentTypeForReplacedFile(const String& path, const String& nameOverride, String& effectiveName, String& effectiveContentType)
63{
64 ASSERT(!FileSystem::pathFileName(path).endsWith('/')); // Expecting to get a path without trailing slash, even for directories.
65 effectiveContentType = "application/zip"_s;
66 effectiveName = makeString((nameOverride.isNull() ? FileSystem::pathFileName(path) : nameOverride), ".zip"_s);
67}
