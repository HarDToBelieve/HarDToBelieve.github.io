34NetworkSendQueue::NetworkSendQueue(ScriptExecutionContext& context, WriteString&& writeString, WriteRawData&& writeRawData, ProcessError&& processError)
35 : ContextDestructionObserver(&context)
36 , m_writeString(WTFMove(writeString))
37 , m_writeRawData(WTFMove(writeRawData))
38 , m_processError(WTFMove(processError))
39{
40}

42NetworkSendQueue::~NetworkSendQueue() = default;
43
44void NetworkSendQueue::enqueue(CString&& utf8)
45{
46 if (m_queue.isEmpty()) {
47 m_writeString(utf8);
48 return;
49 }
50 m_queue.append(WTFMove(utf8));
51}
52
53void NetworkSendQueue::enqueue(const JSC::ArrayBuffer& binaryData, unsigned byteOffset, unsigned byteLength)
54{
55 if (m_queue.isEmpty()) {
56 auto* data = static_cast<const uint8_t*>(binaryData.data());
57 m_writeRawData(Span { data + byteOffset, byteLength });
58 return;
59 }
60 m_queue.append(SharedBuffer::create(static_cast<const uint8_t*>(binaryData.data()) + byteOffset, byteLength));
61}
62
63void NetworkSendQueue::enqueue(WebCore::Blob& blob)
64{
65 auto* context = scriptExecutionContext();
66 if (!context)
67 return;
68
69 auto byteLength = blob.size();
70 if (!byteLength) {
71 enqueue(JSC::ArrayBuffer::create(0U, 1), 0, 0);
72 return;
73 }
74 auto blobLoader = makeUniqueRef<BlobLoader>([this](BlobLoader&) {
75 processMessages();
76 });
77 auto* blobLoaderPtr = &blobLoader.get();
78 m_queue.append(WTFMove(blobLoader));
79 blobLoaderPtr->start(blob, context, FileReaderLoader::ReadAsArrayBuffer);
80}
81
82void NetworkSendQueue::clear()
83{
84 // Do not call m_queue.clear() here since destroying a BlobLoader will cause its completion
85 // handler to get called, which will call processMessages() to iterate over m_queue.
86 std::exchange(m_queue, { });
87}
88
89void NetworkSendQueue::processMessages()
90{
91 while (!m_queue.isEmpty()) {
92 bool shouldStopProcessing = false;
93 switchOn(m_queue.first(), [this](const CString& utf8) {
94 m_writeString(utf8);
95 }, [this](Ref<SharedBuffer>& data) {
96 data->forEachSegment(m_writeRawData);
97 }, [this, &shouldStopProcessing](UniqueRef<BlobLoader>& loader) {
98 auto errorCode = loader->errorCode();
99 if (loader->isLoading() || (errorCode && errorCode.value() == AbortError)) {
100 shouldStopProcessing = true;
101 return;
102 }
103
104 if (const auto& result = loader->arrayBufferResult()) {
105 m_writeRawData(Span { static_cast<const uint8_t*>(result->data()), result->byteLength() });
106 return;
107 }
108 ASSERT(errorCode);
109 shouldStopProcessing = m_processError(errorCode.value()) == Continue::No;
110 });
111 if (shouldStopProcessing)
112 return;
113 m_queue.removeFirst();
114 }
