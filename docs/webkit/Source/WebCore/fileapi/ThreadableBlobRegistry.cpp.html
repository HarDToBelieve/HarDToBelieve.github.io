54typedef HashMap<String, RefPtr<SecurityOrigin>> BlobUrlOriginMap;
55
56static ThreadSpecific<BlobUrlOriginMap>& originMap()
57{
58 static std::once_flag onceFlag;
59 static ThreadSpecific<BlobUrlOriginMap>* map;
60 std::call_once(onceFlag, []{
61 map = new ThreadSpecific<BlobUrlOriginMap>;
62 });
63
64 return *map;
65}

67void ThreadableBlobRegistry::registerFileBlobURL(const URL& url, const String& path, const String& replacementPath, const String& contentType)
68{
69 String effectivePath = !replacementPath.isNull() ? replacementPath : path;
70
71 if (isMainThread()) {
72 blobRegistry().registerFileBlobURL(url, BlobDataFileReference::create(effectivePath), path, contentType);
73 return;
74 }
75
76 callOnMainThread([url = url.isolatedCopy(), effectivePath = effectivePath.isolatedCopy(), path = path.isolatedCopy(), contentType = contentType.isolatedCopy()] {
77 blobRegistry().registerFileBlobURL(url, BlobDataFileReference::create(effectivePath), path, contentType);
78 });
79}

81void ThreadableBlobRegistry::registerBlobURL(const URL& url, Vector<BlobPart>&& blobParts, const String& contentType)
82{
83 if (isMainThread()) {
84 blobRegistry().registerBlobURL(url, WTFMove(blobParts), contentType);
85 return;
86 }
87 for (auto& part : blobParts)
88 part.detachFromCurrentThread();
89 callOnMainThread([url = url.isolatedCopy(), blobParts = WTFMove(blobParts), contentType = contentType.isolatedCopy()]() mutable {
90 blobRegistry().registerBlobURL(url, WTFMove(blobParts), contentType);
91 });
92}

94static inline bool isBlobURLContainsNullOrigin(const URL& url)
95{
96 ASSERT(url.protocolIsBlob());
97 unsigned startIndex = url.pathStart();
98 unsigned endIndex = url.pathAfterLastSlash();
99 return url.string().substring(startIndex, endIndex - startIndex - 1) == "null";
100}

102void ThreadableBlobRegistry::registerBlobURL(SecurityOrigin* origin, const URL& url, const URL& srcURL)
103{
104 // If the blob URL contains null origin, as in the context with unique security origin or file URL, save the mapping between url and origin so that the origin can be retrived when doing security origin check.
105 if (origin && isBlobURLContainsNullOrigin(url))
106 originMap()->add(url.string(), origin);
107
108 if (isMainThread()) {
109 blobRegistry().registerBlobURL(url, srcURL);
110 return;
111 }
112
113 callOnMainThread([url = url.isolatedCopy(), srcURL = srcURL.isolatedCopy()] {
114 blobRegistry().registerBlobURL(url, srcURL);
115 });
116}

141unsigned long long ThreadableBlobRegistry::blobSize(const URL& url)
142{
143 if (isMainThread())
144 return blobRegistry().blobSize(url);
145
146 unsigned long long resultSize;
147 callOnMainThreadAndWait([url = url.isolatedCopy(), &resultSize] {
148 resultSize = blobRegistry().blobSize(url);
149 });
150 return resultSize;
151}

153void ThreadableBlobRegistry::unregisterBlobURL(const URL& url)
154{
155 if (isBlobURLContainsNullOrigin(url))
156 originMap()->remove(url.string());
157
158 if (isMainThread()) {
159 blobRegistry().unregisterBlobURL(url);
160 return;
161 }
162 callOnMainThread([url = url.isolatedCopy()] {
163 blobRegistry().unregisterBlobURL(url);
164 });
165}

167RefPtr<SecurityOrigin> ThreadableBlobRegistry::getCachedOrigin(const URL& url)
168{
169 if (auto cachedOrigin = originMap()->get(url.string()))
170 return cachedOrigin;
171
172 if (!url.protocolIsBlob() || !isBlobURLContainsNullOrigin(url))
173 return nullptr;
174
175 // If we do not have a cached origin for null blob URLs, we use a unique origin.
176 return SecurityOrigin::createUnique();
177}
