76static bool canCacheFrame(Frame& frame, DiagnosticLoggingClient& diagnosticLoggingClient, unsigned indentLevel)
77{
78 PCLOG("+---");
79 FrameLoader& frameLoader = frame.loader();
80
81 // Prevent page caching if a subframe is still in provisional load stage.
82 // We only do this check for subframes because the main frame is reused when navigating to a new page.
83 if (!frame.isMainFrame() && frameLoader.state() == FrameState::Provisional) {
84 PCLOG(" -Frame is in provisional load stage");
85 logBackForwardCacheFailureDiagnosticMessage(diagnosticLoggingClient, DiagnosticLoggingKeys::provisionalLoadKey());
86 return false;
87 }
88
89 if (frame.isMainFrame() && frameLoader.stateMachine().isDisplayingInitialEmptyDocument()) {
90 PCLOG(" -MainFrame is displaying initial empty document");
91 return false;
92 }
93
94 if (!frame.document()) {
95 PCLOG(" -Frame has no document");
96 return false;
97 }
98
99 if (frame.document()->shouldPreventEnteringBackForwardCacheForTesting()) {
100 PCLOG(" -Back/Forward caching is disabled for testing");
101 return false;
102 }
103
104 if (!frame.document()->frame()) {
105 PCLOG(" -Document is detached from frame");
106 ASSERT_NOT_REACHED();
107 return false;
108 }
109
110 DocumentLoader* documentLoader = frameLoader.documentLoader();
111 if (!documentLoader) {
112 PCLOG(" -There is no DocumentLoader object");
113 logBackForwardCacheFailureDiagnosticMessage(diagnosticLoggingClient, DiagnosticLoggingKeys::noDocumentLoaderKey());
114 return false;
115 }
116
117 URL currentURL = documentLoader->url();
118 URL newURL = frameLoader.provisionalDocumentLoader() ? frameLoader.provisionalDocumentLoader()->url() : URL();
119 if (!newURL.isEmpty())
120 PCLOG(" Determining if frame can be cached navigating from (", currentURL.string(), ") to (", newURL.string(), "):");
121 else
122 PCLOG(" Determining if subframe with URL (", currentURL.string(), ") can be cached:");
123
124 bool isCacheable = true;
125
126 if (frame.isMainFrame() && frame.document()->quirks().shouldBypassBackForwardCache()) {
127 PCLOG(" -Disabled by site-specific quirk");
128 logBackForwardCacheFailureDiagnosticMessage(diagnosticLoggingClient, DiagnosticLoggingKeys::siteSpecificQuirkKey());
129 isCacheable = false;
130 }
131
132 // Do not cache error pages (these can be recognized as pages with substitute data or unreachable URLs).
133 if (documentLoader->substituteData().isValid() && !documentLoader->substituteData().failingURL().isEmpty()) {
134 PCLOG(" -Frame is an error page");
135 logBackForwardCacheFailureDiagnosticMessage(diagnosticLoggingClient, DiagnosticLoggingKeys::isErrorPageKey());
136 isCacheable = false;
137 }
138 if (frameLoader.subframeLoader().containsPlugins() && !frame.page()->settings().backForwardCacheSupportsPlugins()) {
139 PCLOG(" -Frame contains plugins");
140 isCacheable = false;
141 }
142 if (frame.isMainFrame() && frame.document() && frame.document()->url().protocolIs("https") && documentLoader->response().cacheControlContainsNoStore()) {
143 PCLOG(" -Frame is HTTPS, and cache control prohibits storing");
144 logBackForwardCacheFailureDiagnosticMessage(diagnosticLoggingClient, DiagnosticLoggingKeys::httpsNoStoreKey());
145 isCacheable = false;
146 }
147 if (frame.isMainFrame() && !frameLoader.history().currentItem()) {
148 PCLOG(" -Main frame has no current history item");
149 logBackForwardCacheFailureDiagnosticMessage(diagnosticLoggingClient, DiagnosticLoggingKeys::noCurrentHistoryItemKey());
150 isCacheable = false;
151 }
152 if (frame.isMainFrame() && frame.view() && !frame.view()->isVisuallyNonEmpty()) {
153 PCLOG(" -Main frame is visually empty");
154 logBackForwardCacheFailureDiagnosticMessage(diagnosticLoggingClient, DiagnosticLoggingKeys::visuallyEmptyKey());
155 isCacheable = false;
156 }
157 if (frameLoader.quickRedirectComing()) {
158 PCLOG(" -Quick redirect is coming");
159 logBackForwardCacheFailureDiagnosticMessage(diagnosticLoggingClient, DiagnosticLoggingKeys::quirkRedirectComingKey());
160 isCacheable = false;
161 }
162 if (documentLoader->isLoading()) {
163 PCLOG(" -DocumentLoader is still loading");
164 logBackForwardCacheFailureDiagnosticMessage(diagnosticLoggingClient, DiagnosticLoggingKeys::isLoadingKey());
165 isCacheable = false;
166 }
167 if (documentLoader->isStopping()) {
168 PCLOG(" -DocumentLoader is in the middle of stopping");
169 logBackForwardCacheFailureDiagnosticMessage(diagnosticLoggingClient, DiagnosticLoggingKeys::documentLoaderStoppingKey());
170 isCacheable = false;
171 }
172
173 // FIXME: We should investigating caching frames that have an associated
174 // application cache. <rdar://problem/5917899> tracks that work.
175 if (!documentLoader->applicationCacheHost().canCacheInBackForwardCache()) {
176 PCLOG(" -The DocumentLoader uses an application cache");
177 logBackForwardCacheFailureDiagnosticMessage(diagnosticLoggingClient, DiagnosticLoggingKeys::applicationCacheKey());
178 isCacheable = false;
179 }
180 if (!frameLoader.client().canCachePage()) {
181 PCLOG(" -The client says this frame cannot be cached");
182 logBackForwardCacheFailureDiagnosticMessage(diagnosticLoggingClient, DiagnosticLoggingKeys::deniedByClientKey());
183 isCacheable = false;
184 }
185
186 for (Frame* child = frame.tree().firstChild(); child; child = child->tree().nextSibling()) {
187 if (!canCacheFrame(*child, diagnosticLoggingClient, indentLevel + 1))
188 isCacheable = false;
189 }
190
191 PCLOG(isCacheable ? " Frame CAN be cached" : " Frame CANNOT be cached");
192 PCLOG("+---");
193
194 return isCacheable;
195}

197static bool canCachePage(Page& page)
198{
199 RELEASE_ASSERT(!page.isRestoringCachedPage());
200
201 unsigned indentLevel = 0;
202 PCLOG("--------\n Determining if page can be cached:");
203
204 DiagnosticLoggingClient& diagnosticLoggingClient = page.diagnosticLoggingClient();
205 bool isCacheable = canCacheFrame(page.mainFrame(), diagnosticLoggingClient, indentLevel + 1);
206
207 if (!page.settings().usesBackForwardCache() || page.isResourceCachingDisabledByWebInspector()) {
208 PCLOG(" -Page settings says b/f cache disabled");
209 logBackForwardCacheFailureDiagnosticMessage(diagnosticLoggingClient, DiagnosticLoggingKeys::isDisabledKey());
210 isCacheable = false;
211 }
212#if ENABLE(DEVICE_ORIENTATION) && !PLATFORM(IOS_FAMILY)
213 if (DeviceMotionController::isActiveAt(&page)) {
214 PCLOG(" -Page is using DeviceMotion");
215 logBackForwardCacheFailureDiagnosticMessage(diagnosticLoggingClient, DiagnosticLoggingKeys::deviceMotionKey());
216 isCacheable = false;
217 }
218 if (DeviceOrientationController::isActiveAt(&page)) {
219 PCLOG(" -Page is using DeviceOrientation");
220 logBackForwardCacheFailureDiagnosticMessage(diagnosticLoggingClient, DiagnosticLoggingKeys::deviceOrientationKey());
221 isCacheable = false;
222 }
223#endif
224
225 FrameLoadType loadType = page.mainFrame().loader().loadType();
226 switch (loadType) {
227 case FrameLoadType::Reload:
228 // No point writing to the cache on a reload, since we will just write over it again when we leave that page.
229 PCLOG(" -Load type is: Reload");
230 logBackForwardCacheFailureDiagnosticMessage(diagnosticLoggingClient, DiagnosticLoggingKeys::reloadKey());
231 isCacheable = false;
232 break;
233 case FrameLoadType::Same: // user loads same URL again (but not reload button)
234 // No point writing to the cache on a same load, since we will just write over it again when we leave that page.
235 PCLOG(" -Load type is: Same");
236 logBackForwardCacheFailureDiagnosticMessage(diagnosticLoggingClient, DiagnosticLoggingKeys::sameLoadKey());
237 isCacheable = false;
238 break;
239 case FrameLoadType::RedirectWithLockedBackForwardList:
240 // Don't write to the cache if in the middle of a redirect, since we will want to store the final page we end up on.
241 PCLOG(" -Load type is: RedirectWithLockedBackForwardList");
242 logBackForwardCacheFailureDiagnosticMessage(diagnosticLoggingClient, DiagnosticLoggingKeys::redirectKey());
243 isCacheable = false;
244 break;
245 case FrameLoadType::Replace:
246 // No point writing to the cache on a replace, since we will just write over it again when we leave that page.
247 PCLOG(" -Load type is: Replace");
248 logBackForwardCacheFailureDiagnosticMessage(diagnosticLoggingClient, DiagnosticLoggingKeys::replaceKey());
249 isCacheable = false;
250 break;
251 case FrameLoadType::ReloadFromOrigin: {
252 // No point writing to the cache on a reload, since we will just write over it again when we leave that page.
253 PCLOG(" -Load type is: ReloadFromOrigin");
254 logBackForwardCacheFailureDiagnosticMessage(diagnosticLoggingClient, DiagnosticLoggingKeys::reloadFromOriginKey());
255 isCacheable = false;
256 break;
257 }
258 case FrameLoadType::ReloadExpiredOnly: {
259 // No point writing to the cache on a reload, since we will just write over it again when we leave that page.
260 PCLOG(" -Load type is: ReloadRevalidatingExpired");
261 logBackForwardCacheFailureDiagnosticMessage(diagnosticLoggingClient, DiagnosticLoggingKeys::reloadRevalidatingExpiredKey());
262 isCacheable = false;
263 break;
264 }
265 case FrameLoadType::Standard:
266 case FrameLoadType::Back:
267 case FrameLoadType::Forward:
268 case FrameLoadType::IndexedBackForward: // a multi-item hop in the backforward list
269 // Cacheable.
270 break;
271 }
272
273 if (isCacheable)
274 PCLOG(" Page CAN be cached\n--------");
275 else
276 PCLOG(" Page CANNOT be cached\n--------");
277
278 diagnosticLoggingClient.logDiagnosticMessageWithResult(DiagnosticLoggingKeys::backForwardCacheKey(), DiagnosticLoggingKeys::canCacheKey(), isCacheable ? DiagnosticLoggingResultPass : DiagnosticLoggingResultFail, ShouldSample::No);
279 return isCacheable;
280}

307bool BackForwardCache::canCache(Page& page) const
308{
309 if (!m_maxSize) {
310 logBackForwardCacheFailureDiagnosticMessage(&page, DiagnosticLoggingKeys::isDisabledKey());
311 return false;
312 }
313
314 if (MemoryPressureHandler::singleton().isUnderMemoryPressure()) {
315 logBackForwardCacheFailureDiagnosticMessage(&page, DiagnosticLoggingKeys::underMemoryPressureKey());
316 return false;
317 }
318
319 return canCachePage(page);
320}

373static String pruningReasonToDiagnosticLoggingKey(PruningReason pruningReason)
374{
375 switch (pruningReason) {
376 case PruningReason::MemoryPressure:
377 return DiagnosticLoggingKeys::prunedDueToMemoryPressureKey();
378 case PruningReason::ProcessSuspended:
379 return DiagnosticLoggingKeys::prunedDueToProcessSuspended();
380 case PruningReason::ReachedMaxSize:
381 return DiagnosticLoggingKeys::prunedDueToMaxSizeReached();
382 case PruningReason::None:
383 break;
384 }
385 ASSERT_NOT_REACHED();
386 return emptyString();
387}

389static void setBackForwardCacheState(Page& page, Document::BackForwardCacheState BackForwardCacheState)
390{
391 for (Frame* frame = &page.mainFrame(); frame; frame = frame->tree().traverseNext()) {
392 if (auto* document = frame->document())
393 document->setBackForwardCacheState(BackForwardCacheState);
394 }
395}
396
397// When entering back/forward cache, tear down the render tree before setting the in-cache flag.
398// This maintains the invariant that render trees are never present in the back/forward cache.
399// Note that destruction happens bottom-up so that the main frame's tree dies last.
400static void destroyRenderTree(Frame& mainFrame)
401{
402 for (Frame* frame = mainFrame.tree().traversePrevious(CanWrap::Yes); frame; frame = frame->tree().traversePrevious(CanWrap::No)) {
403 if (!frame->document())
404 continue;
405 auto& document = *frame->document();
406 if (document.hasLivingRenderTree())
407 document.destroyRenderTree();
408 }
409}

411static void firePageHideEventRecursively(Frame& frame)
412{
413 auto* document = frame.document();
414 if (!document)
415 return;
416
417 // stopLoading() will fire the pagehide event in each subframe and the HTML specification states
418 // that the parent document's ignore-opens-during-unload counter should be incremented while the
419 // pagehide event is being fired in its subframes:
420 // https://html.spec.whatwg.org/multipage/browsers.html#unload-a-document
421 IgnoreOpensDuringUnloadCountIncrementer ignoreOpensDuringUnloadCountIncrementer(document);
422
423 frame.loader().stopLoading(UnloadEventPolicy::UnloadAndPageHide);
424
425 for (RefPtr<Frame> child = frame.tree().firstChild(); child; child = child->tree().nextSibling())
426 firePageHideEventRecursively(*child);
427}

429std::unique_ptr<CachedPage> BackForwardCache::trySuspendPage(Page& page, ForceSuspension forceSuspension)
430{
431 page.mainFrame().loader().stopForBackForwardCache();
432
433 if (forceSuspension == ForceSuspension::No && !canCache(page))
434 return nullptr;
435
436 ASSERT_WITH_MESSAGE(!page.isUtilityPage(), "Utility pages such as SVGImage pages should never go into BackForwardCache");
437
438 setBackForwardCacheState(page, Document::AboutToEnterBackForwardCache);
439
440 // Focus the main frame, defocusing a focused subframe (if we have one). We do this here,
441 // before the page enters the back/forward cache, while we still can dispatch DOM blur/focus events.
442 if (page.focusController().focusedFrame())
443 page.focusController().setFocusedFrame(&page.mainFrame());
444
445 // Fire the pagehide event in all frames.
446 firePageHideEventRecursively(page.mainFrame());
447
448 destroyRenderTree(page.mainFrame());
449
450 // Stop all loads again before checking if we can still cache the page after firing the pagehide
451 // event, since the page may have started ping loads in its pagehide event handler.
452 page.mainFrame().loader().stopForBackForwardCache();
453
454 // Check that the page is still page-cacheable after firing the pagehide event. The JS event handlers
455 // could have altered the page in a way that could prevent caching.
456 if (forceSuspension == ForceSuspension::No && !canCache(page)) {
457 setBackForwardCacheState(page, Document::NotInBackForwardCache);
458 return nullptr;
459 }
460
461 setBackForwardCacheState(page, Document::InBackForwardCache);
462
463 // Make sure we don't fire any JS events in this scope.
464 ScriptDisallowedScope::InMainThread scriptDisallowedScope;
465 return makeUnique<CachedPage>(page);
466}

468bool BackForwardCache::addIfCacheable(HistoryItem& item, Page* page)
469{
470 if (item.isInBackForwardCache())
471 return false;
472
473 if (!page)
474 return false;
475
476 auto cachedPage = trySuspendPage(*page, ForceSuspension::No);
477 if (!cachedPage)
478 return false;
479
480 {
481 // Make sure we don't fire any JS events in this scope.
482 ScriptDisallowedScope::InMainThread scriptDisallowedScope;
483
484 item.setCachedPage(WTFMove(cachedPage));
485 item.m_pruningReason = PruningReason::None;
486 m_items.add(&item);
487 }
488 prune(PruningReason::ReachedMaxSize);
489
490 RELEASE_LOG(BackForwardCache, "BackForwardCache::addIfCacheable item: %s, size: %u / %u", item.identifier().string().utf8().data(), pageCount(), maxSize());
491
492 return true;
493}

495std::unique_ptr<CachedPage> BackForwardCache::suspendPage(Page& page)
496{
497 RELEASE_LOG(BackForwardCache, "BackForwardCache::suspendPage()");
498
499 return trySuspendPage(page, ForceSuspension::Yes);

502std::unique_ptr<CachedPage> BackForwardCache::take(HistoryItem& item, Page* page)
503{
504 if (!item.m_cachedPage) {
505 if (item.m_pruningReason != PruningReason::None)
506 logBackForwardCacheFailureDiagnosticMessage(page, pruningReasonToDiagnosticLoggingKey(item.m_pruningReason));
507 return nullptr;
508 }
509
510 m_items.remove(&item);
511 std::unique_ptr<CachedPage> cachedPage = item.takeCachedPage();
512
513 RELEASE_LOG(BackForwardCache, "BackForwardCache::take item: %s, size: %u / %u", item.identifier().string().utf8().data(), pageCount(), maxSize());
514
515 if (cachedPage->hasExpired() || (page && page->isResourceCachingDisabledByWebInspector())) {
516 LOG(BackForwardCache, "Not restoring page for %s from back/forward cache because cache entry has expired", item.url().string().ascii().data());
517 logBackForwardCacheFailureDiagnosticMessage(page, DiagnosticLoggingKeys::expiredKey());
518 return nullptr;
519 }
520
521 return cachedPage;
522}

524void BackForwardCache::removeAllItemsForPage(Page& page)
525{
526#if ASSERT_ENABLED
527 ASSERT_WITH_MESSAGE(!m_isInRemoveAllItemsForPage, "We should not reenter this method");
528 SetForScope<bool> inRemoveAllItemsForPageScope { m_isInRemoveAllItemsForPage, true };
529#endif
530
531 for (auto it = m_items.begin(); it != m_items.end();) {
532 // Increment iterator first so it stays valid after the removal.
533 auto current = it;
534 ++it;
535 if (&(*current)->m_cachedPage->page() == &page) {
536 RELEASE_LOG(BackForwardCache, "BackForwardCache::removeAllItemsForPage removing item: %s, size: %u / %u", (*current)->identifier().string().utf8().data(), pageCount() - 1, maxSize());
537 (*current)->setCachedPage(nullptr);
538 m_items.remove(current);
539 }
540 }

543CachedPage* BackForwardCache::get(HistoryItem& item, Page* page)
544{
545 CachedPage* cachedPage = item.m_cachedPage.get();
546 if (!cachedPage) {
547 if (item.m_pruningReason != PruningReason::None)
548 logBackForwardCacheFailureDiagnosticMessage(page, pruningReasonToDiagnosticLoggingKey(item.m_pruningReason));
549 return nullptr;
550 }
551
552 if (cachedPage->hasExpired() || (page && page->isResourceCachingDisabledByWebInspector())) {
553 LOG(BackForwardCache, "Not restoring page for %s from back/forward cache because cache entry has expired", item.url().string().ascii().data());
554 logBackForwardCacheFailureDiagnosticMessage(page, DiagnosticLoggingKeys::expiredKey());
555 remove(item);
556 return nullptr;
557 }
558 return cachedPage;
559}

561void BackForwardCache::remove(HistoryItem& item)
562{
563 // Safely ignore attempts to remove items not in the cache.
564 if (!item.m_cachedPage)
565 return;
566
567 m_items.remove(&item);
568 item.setCachedPage(nullptr);
569
570 RELEASE_LOG(BackForwardCache, "BackForwardCache::remove item: %s, size: %u / %u", item.identifier().string().utf8().data(), pageCount(), maxSize());
571
572}
573
574void BackForwardCache::prune(PruningReason pruningReason)
575{
576 while (pageCount() > maxSize()) {
577 auto oldestItem = m_items.takeFirst();
578 oldestItem->setCachedPage(nullptr);
579 oldestItem->m_pruningReason = pruningReason;
580 RELEASE_LOG(BackForwardCache, "BackForwardCache::prune removing item: %s, size: %u / %u", oldestItem->identifier().string().utf8().data(), pageCount(), maxSize());
581 }
