61CachedFrameBase::CachedFrameBase(Frame& frame)
62 : m_document(frame.document())
63 , m_documentLoader(frame.loader().documentLoader())
64 , m_view(frame.view())
65 , m_url(frame.document()->url())
66 , m_isMainFrame(!frame.tree().parent())
67{
68}

70CachedFrameBase::~CachedFrameBase()
71{
72#ifndef NDEBUG
73 cachedFrameCounter.decrement();
74#endif
75 // CachedFrames should always have had destroy() called by their parent CachedPage
76 ASSERT(!m_document);
77}

79void CachedFrameBase::pruneDetachedChildFrames()
80{
81 m_childFrames.removeAllMatching([] (auto& childFrame) {
82 if (childFrame->view()->frame().page())
83 return false;
84 childFrame->destroy();
85 return true;
86 });
87}

89void CachedFrameBase::restore()
90{
91 ASSERT(m_document->view() == m_view);
92
93 if (m_isMainFrame)
94 m_view->setParentVisible(true);
95
96 auto frame = makeRef(m_view->frame());
97 {
98 Style::PostResolutionCallbackDisabler disabler(*m_document);
99 WidgetHierarchyUpdatesSuspensionScope suspendWidgetHierarchyUpdates;
100 NavigationDisabler disableNavigation { nullptr }; // Disable navigation globally.
101
102 m_cachedFrameScriptData->restore(frame.get());
103
104 if (m_document->svgExtensions())
105 m_document->accessSVGExtensions().unpauseAnimations();
106
107 m_document->resume(ReasonForSuspension::BackForwardCache);
108
109 // It is necessary to update any platform script objects after restoring the
110 // cached page.
111 frame->script().updatePlatformScriptObjects();
112
113 frame->loader().client().didRestoreFromBackForwardCache();
114
115 pruneDetachedChildFrames();
116
117 // Reconstruct the FrameTree. And open the child CachedFrames in their respective FrameLoaders.
118 for (auto& childFrame : m_childFrames) {
119 ASSERT(childFrame->view()->frame().page());
120 frame->tree().appendChild(childFrame->view()->frame());
121 childFrame->open();
122 ASSERT_WITH_SECURITY_IMPLICATION(m_document == frame->document());
123 }
124 }
125
126#if PLATFORM(IOS_FAMILY)
127 if (m_isMainFrame) {
128 frame->loader().client().didRestoreFrameHierarchyForCachedFrame();
129
130 if (DOMWindow* domWindow = m_document->domWindow()) {
131 // FIXME: Add SCROLL_LISTENER to the list of event types on Document, and use m_document->hasListenerType(). See <rdar://problem/9615482>.
132 // FIXME: Can use Document::hasListenerType() now.
133 if (domWindow->scrollEventListenerCount() && frame->page())
134 frame->page()->chrome().client().setNeedsScrollNotifications(frame, true);
135 }
136 }
137#endif
138
139 frame->view()->didRestoreFromBackForwardCache();
140}

142CachedFrame::CachedFrame(Frame& frame)
143 : CachedFrameBase(frame)
144{
145#ifndef NDEBUG
146 cachedFrameCounter.increment();
147#endif
148 ASSERT(m_document);
149 ASSERT(m_documentLoader);
150 ASSERT(m_view);
151 ASSERT(m_document->backForwardCacheState() == Document::InBackForwardCache);
152
153 RELEASE_ASSERT(m_document->domWindow());
154 RELEASE_ASSERT(m_document->frame());
155 RELEASE_ASSERT(m_document->domWindow()->frame());
156
157 // FIXME: We have evidence that constructing CachedFrames for descendant frames may detach the document from its frame (rdar://problem/49877867).
158 // This sets the flag to help find the guilty code.
159 m_document->setMayBeDetachedFromFrame(false);
160
161 // Create the CachedFrames for all Frames in the FrameTree.
162 for (Frame* child = frame.tree().firstChild(); child; child = child->tree().nextSibling())
163 m_childFrames.append(makeUniqueRef<CachedFrame>(*child));
164
165 RELEASE_ASSERT(m_document->domWindow());
166 RELEASE_ASSERT(m_document->frame());
167 RELEASE_ASSERT(m_document->domWindow()->frame());
168
169 // Active DOM objects must be suspended before we cache the frame script data.
170 m_document->suspend(ReasonForSuspension::BackForwardCache);
171
172 m_cachedFrameScriptData = makeUnique<ScriptCachedFrameData>(frame);
173
174 m_document->domWindow()->suspendForBackForwardCache();
175
176 // Clear FrameView to reset flags such as 'firstVisuallyNonEmptyLayoutCallbackPending' so that the
177 // 'DidFirstVisuallyNonEmptyLayout' callback gets called against when restoring from the BackForwardCache.
178 m_view->resetLayoutMilestones();
179
180 // The main frame is reused for the navigation and the opener link to its should thus persist.
181 if (!frame.isMainFrame())
182 frame.loader().detachFromAllOpenedFrames();
183
184 frame.loader().client().savePlatformDataToCachedFrame(this);
185
186 // documentWillSuspendForBackForwardCache() can set up a layout timer on the FrameView, so clear timers after that.
187 frame.clearTimers();
188
189 // Deconstruct the FrameTree, to restore it later.
190 // We do this for two reasons:
191 // 1 - We reuse the main frame, so when it navigates to a new page load it needs to start with a blank FrameTree.
192 // 2 - It's much easier to destroy a CachedFrame while it resides in the BackForwardCache if it is disconnected from its parent.
193 for (unsigned i = 0; i < m_childFrames.size(); ++i)
194 frame.tree().removeChild(m_childFrames[i]->view()->frame());
195
196#ifndef NDEBUG
197 if (m_isMainFrame)
198 LOG(BackForwardCache, "Finished creating CachedFrame for main frame url '%s' and DocumentLoader %p\n", m_url.string().utf8().data(), m_documentLoader.get());
199 else
200 LOG(BackForwardCache, "Finished creating CachedFrame for child frame with url '%s' and DocumentLoader %p\n", m_url.string().utf8().data(), m_documentLoader.get());
201#endif
202
203#if PLATFORM(IOS_FAMILY)
204 if (m_isMainFrame) {
205 if (DOMWindow* domWindow = m_document->domWindow()) {
206 if (domWindow->scrollEventListenerCount() && frame.page())
207 frame.page()->chrome().client().setNeedsScrollNotifications(frame, false);
208 }
209 }
210#endif
211
212 m_document->setMayBeDetachedFromFrame(true);
213 m_document->detachFromCachedFrame(*this);
214
215 ASSERT_WITH_SECURITY_IMPLICATION(!m_documentLoader->isLoading());
216}

222
223 m_view->frame().loader().open(*this);

226void CachedFrame::clear()
227{
228 if (!m_document)
229 return;
230
231 // clear() should only be called for Frames representing documents that are no longer in the back/forward cache.
232 // This means the CachedFrame has been:
233 // 1 - Successfully restore()'d by going back/forward.
234 // 2 - destroy()'ed because the BackForwardCache is pruning or the WebView was closed.
235 ASSERT(m_document->backForwardCacheState() == Document::NotInBackForwardCache);
236 ASSERT(m_view);
237 ASSERT(!m_document->frame() || m_document->frame() == &m_view->frame());
238
239 for (int i = m_childFrames.size() - 1; i >= 0; --i)
240 m_childFrames[i]->clear();
241
242 m_document = nullptr;
243 m_view = nullptr;
244 m_url = URL();
245
246 m_cachedFramePlatformData = nullptr;
247 m_cachedFrameScriptData = nullptr;
248}
249
250void CachedFrame::destroy()
251{
252 if (!m_document)
253 return;
254
255 // Only CachedFrames that are still in the BackForwardCache should be destroyed in this manner
256 ASSERT(m_document->backForwardCacheState() == Document::InBackForwardCache);
257 ASSERT(m_view);
258 ASSERT(!m_document->frame());
259
260 m_document->domWindow()->willDestroyCachedFrame();
261
262 if (!m_isMainFrame && m_view->frame().page()) {
263 m_view->frame().loader().detachViewsAndDocumentLoader();
264 m_view->frame().detachFromPage();
265 }
266
267 for (int i = m_childFrames.size() - 1; i >= 0; --i)
268 m_childFrames[i]->destroy();
269
270 if (m_cachedFramePlatformData)
271 m_cachedFramePlatformData->clear();
272
273 Frame::clearTimers(m_view.get(), m_document.get());
274
275 // FIXME: Why do we need to call removeAllEventListeners here? When the document is in back/forward cache, this method won't work
276 // fully anyway, because the document won't be able to access its DOMWindow object (due to being frameless).
277 m_document->removeAllEventListeners();
278
279 m_document->setBackForwardCacheState(Document::NotInBackForwardCache);
280 m_document->willBeRemovedFromFrame();
281
282 clear();
283}

295size_t CachedFrame::descendantFrameCount() const
296{
297 size_t count = m_childFrames.size();
298 for (const auto& childFrame : m_childFrames)
299 count += childFrame->descendantFrameCount();
300 return count;
301}

303UsedLegacyTLS CachedFrame::usedLegacyTLS() const
304{
305 if (auto* document = this->document()) {
306 if (document->usedLegacyTLS())
307 return UsedLegacyTLS::Yes;
308 }
309
310 for (const auto& cachedFrame : m_childFrames) {
311 if (cachedFrame->usedLegacyTLS() == UsedLegacyTLS::Yes)
312 return UsedLegacyTLS::Yes;
313 }
314
315 return UsedLegacyTLS::No;
316}
317
318HasInsecureContent CachedFrame::hasInsecureContent() const
319{
320 if (auto* document = this->document()) {
321 if (!document->isSecureContext() || !document->foundMixedContent().isEmpty())
322 return HasInsecureContent::Yes;
323 }
324
325 for (const auto& cachedFrame : m_childFrames) {
326 if (cachedFrame->hasInsecureContent() == HasInsecureContent::Yes)
327 return HasInsecureContent::Yes;
328 }
329
330 return HasInsecureContent::No;
331}
