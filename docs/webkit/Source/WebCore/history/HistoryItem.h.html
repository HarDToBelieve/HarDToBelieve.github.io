65 static Ref<HistoryItem> create()

72 return adoptRef(*new HistoryItem(urlString, title));
73 }
74
75 static Ref<HistoryItem> create(const String& urlString, const String& title, const String& alternateTitle)
76 {
77 return adoptRef(*new HistoryItem(urlString, title, alternateTitle));
78 }
79
80 static Ref<HistoryItem> create(const String& urlString, const String& title, const String& alternateTitle, BackForwardItemIdentifier identifier)
81 {
82 return adoptRef(*new HistoryItem(urlString, title, alternateTitle, identifier));
83 }

251 // If two HistoryItems have the same item sequence number, then they are
252 // clones of one another. Traversing history from one such HistoryItem to
253 // another is a no-op. HistoryItem clones are created for parent and
254 // sibling frames when only a subframe navigates.
255 int64_t m_itemSequenceNumber { generateSequenceNumber() };
256
257 // If two HistoryItems have the same document sequence number, then they
258 // refer to the same instance of a document. Traversing history from one
259 // such HistoryItem to another preserves the document.
260 int64_t m_documentSequenceNumber { generateSequenceNumber() };

282#endif
