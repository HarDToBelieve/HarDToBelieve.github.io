124
125AutofillFieldName toAutofillFieldName(const AtomString& value)
126{
127 auto map = fieldNameMap();
128 auto it = map.find(value);
129 if (it == map.end())
130 return AutofillFieldName::None;
131 return it->value.fieldName;
132}

134static inline bool isContactToken(const AtomString& token)
135{
136 static MainThreadNeverDestroyed<const AtomString> home("home", AtomString::ConstructFromLiteral);
137 static MainThreadNeverDestroyed<const AtomString> work("work", AtomString::ConstructFromLiteral);
138 static MainThreadNeverDestroyed<const AtomString> mobile("mobile", AtomString::ConstructFromLiteral);
139 static MainThreadNeverDestroyed<const AtomString> fax("fax", AtomString::ConstructFromLiteral);
140 static MainThreadNeverDestroyed<const AtomString> pager("pager", AtomString::ConstructFromLiteral);
141
142 return token == home || token == work || token == mobile || token == fax || token == pager;
143}

145static unsigned maxTokensForAutofillFieldCategory(AutofillCategory category)
146{
147 switch (category) {
148 case AutofillCategory::Automatic:
149 case AutofillCategory::Off:
150 return 1;
151
152 case AutofillCategory::Normal:
153 return 3;
154
155 case AutofillCategory::Contact:
156 return 4;
157 }
158 ASSERT_NOT_REACHED();
159 return 0;
160}

162// https://html.spec.whatwg.org/multipage/forms.html#processing-model-3
163AutofillData AutofillData::createFromHTMLFormControlElement(const HTMLFormControlElement& element)
164{
165 static MainThreadNeverDestroyed<const AtomString> on("on", AtomString::ConstructFromLiteral);
166 static MainThreadNeverDestroyed<const AtomString> off("off", AtomString::ConstructFromLiteral);
167
168 // Label: Default
169 // 26. Let the element's IDL-exposed autofill value be the empty string, and its autofill hint set and autofill scope be empty.
170 // 27. If the element's autocomplete attribute is wearing the autofill anchor mantle, then let the element's autofill field name be the empty string and abort these steps.
171 // 28. Let form be the element's form owner, if any, or null otherwise.
172 // 29. If form is not null and form's autocomplete attribute is in the off state, then let the element's autofill field name be "off". Otherwise, let the element's autofill field name be "on".
173 auto defaultLabel = [&] () -> AutofillData {
174 if (element.autofillMantle() == AutofillMantle::Anchor)
175 return { emptyString(), emptyString() };
176
177 auto form = element.form();
178 if (form && form->autocomplete() == off)
179 return { off, emptyString() };
180 return { on, emptyString() };
181 };
182
183
184 const AtomString& attributeValue = element.attributeWithoutSynchronization(HTMLNames::autocompleteAttr);
185
186 // 1. If the element has no autocomplete attribute, then jump to the step labeled default.
187 if (attributeValue.isNull())
188 return defaultLabel();
189
190 // 2. Let tokens be the result of splitting the attribute's value on spaces.
191 SpaceSplitString tokens(attributeValue, true);
192
193 // 3. If tokens is empty, then jump to the step labeled default.
194 if (tokens.isEmpty())
195 return defaultLabel();
196
197 // 4. Let index be the index of the last token in tokens
198 unsigned index = tokens.size() - 1;
199
200 // 5. If the indexth token in tokens is not an ASCII case-insensitive match for one of the
201 // tokens given in the first column of the following table, or if the number of tokens in
202 // tokens is greater than the maximum number given in the cell in the second column of that
203 // token's row, then jump to the step labeled default. Otherwise, let field be the string given
204 // in the cell of the first column of the matching row, and let category be the value of the
205 // cell in the third column of that same row.
206 auto& map = fieldNameMap();
207
208 auto it = map.find(tokens[index]);
209 if (it == map.end())
210 return defaultLabel();
211
212 auto category = it->value.category;
213
214 if (tokens.size() > maxTokensForAutofillFieldCategory(category))
215 return defaultLabel();
216
217 const auto& field = tokens[index];
218
219 // 6. If category is Off or Automatic but the element's autocomplete attribute is wearing the
220 // autofill anchor mantle, then jump to the step labeled default.
221 auto mantle = element.autofillMantle();
222 if ((category == AutofillCategory::Off || category == AutofillCategory::Automatic) && mantle == AutofillMantle::Anchor)
223 return defaultLabel();
224
225 // 7. If category is Off, let the element's autofill field name be the string "off", let its
226 // autofill hint set be empty, and let its IDL-exposed autofill value be the string "off".
227 // Then, abort these steps.
228 if (category == AutofillCategory::Off)
229 return { off, off.get().string() };
230
231 // 8. If category is Automatic, let the element's autofill field name be the string "on",
232 // let its autofill hint set be empty, and let its IDL-exposed autofill value be the string
233 // "on". Then, abort these steps.
234 if (category == AutofillCategory::Automatic)
235 return { on, on.get().string() };
236
237 // 9. Let scope tokens be an empty list.
238 // 10. Let hint tokens be an empty set.
239
240 // NOTE: We are ignoring these steps as we don't currently make use of scope tokens or hint tokens anywhere.
241
242 // 11. Let IDL value have the same value as field.
243 String idlValue = field;
244
245 // 12, If the indexth token in tokens is the first entry, then skip to the step labeled done.
246 if (index == 0)
247 return { field, idlValue };
248
249 // 13. Decrement index by one
250 index--;
251
252 // 14. If category is Contact and the indexth token in tokens is an ASCII case-insensitive match
253 // for one of the strings in the following list, then run the substeps that follow:
254 const auto& contactToken = tokens[index];
255 if (category == AutofillCategory::Contact && isContactToken(contactToken)) {
256 // 1. Let contact be the matching string from the list above.
257 const auto& contact = contactToken;
258
259 // 2. Insert contact at the start of scope tokens.
260 // 3. Add contact to hint tokens.
261
262 // NOTE: We are ignoring these steps as we don't currently make use of scope tokens or hint tokens anywhere.
263
264 // 4. Let IDL value be the concatenation of contact, a U+0020 SPACE character, and the previous
265 // value of IDL value (which at this point will always be field).
266 idlValue = WTF::makeString(contact, " ", idlValue);
267
268 // 5. If the indexth entry in tokens is the first entry, then skip to the step labeled done.
269 if (index == 0)
270 return { field, idlValue };
271
272 // 6. Decrement index by one.
273 index--;
274 }
275
276 // 15. If the indexth token in tokens is an ASCII case-insensitive match for one of the strings
277 // in the following list, then run the substeps that follow:
278 const auto& modeToken = tokens[index];
279 if (equalIgnoringASCIICase(modeToken, "shipping") || equalIgnoringASCIICase(modeToken, "billing")) {
280 // 1. Let mode be the matching string from the list above.
281 const auto& mode = modeToken;
282
283 // 2. Insert mode at the start of scope tokens.
284 // 3. Add mode to hint tokens.
285
286 // NOTE: We are ignoring these steps as we don't currently make use of scope tokens or hint tokens anywhere.
287
288 // 4. Let IDL value be the concatenation of mode, a U+0020 SPACE character, and the previous
289 // value of IDL value (which at this point will either be field or the concatenation of contact,
290 // a space, and field).
291 idlValue = WTF::makeString(mode, " ", idlValue);
292
293 // 5. If the indexth entry in tokens is the first entry, then skip to the step labeled done.
294 if (index == 0)
295 return { field, idlValue };
296
297 // 6. Decrement index by one.
298 index--;
299 }
300
301 // 16. If the indexth entry in tokens is not the first entry, then jump to the step labeled default.
302 if (index != 0)
303 return defaultLabel();
304
305 // 17. If the first eight characters of the indexth token in tokens are not an ASCII case-insensitive
306 // match for the string "section-", then jump to the step labeled default.
307 const auto& sectionToken = tokens[index];
308 if (!startsWithLettersIgnoringASCIICase(StringView(sectionToken), "section-"))
309 return defaultLabel();
310
311 // 18. Let section be the indexth token in tokens, converted to ASCII lowercase.
312 const auto& section = sectionToken;
313
314 // 19. Insert section at the start of scope tokens.
315
316 // NOTE: We are ignoring this step as we don't currently make use of scope tokens or hint tokens anywhere.
317
318 // 20. Let IDL value be the concatenation of section, a U+0020 SPACE character, and the previous
319 // value of IDL value.
320 idlValue = WTF::makeString(section, " ", idlValue);
321
322 return { field, idlValue };
323}
