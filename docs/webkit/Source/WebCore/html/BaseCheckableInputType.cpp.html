46FormControlState BaseCheckableInputType::saveFormControlState() const
47{
48 ASSERT(element());
49 return { element()->checked() ? "on"_s : "off"_s };
50}
51
52void BaseCheckableInputType::restoreFormControlState(const FormControlState& state)
53{
54 ASSERT(element());
55 element()->setChecked(state[0] == "on");
56}
57
58bool BaseCheckableInputType::appendFormData(DOMFormData& formData, bool) const
59{
60 ASSERT(element());
61 if (!element()->checked())
62 return false;
63 formData.append(element()->name(), element()->value());
64 return true;
65}

67auto BaseCheckableInputType::handleKeydownEvent(KeyboardEvent& event) -> ShouldCallBaseEventHandler
68{
69 const String& key = event.keyIdentifier();
70 if (key == "U+0020") {
71 ASSERT(element());
72 element()->setActive(true, true);
73 // No setDefaultHandled(), because IE dispatches a keypress in this case
74 // and the caller will only dispatch a keypress if we don't call setDefaultHandled().
75 return ShouldCallBaseEventHandler::No;
76 }
77 return ShouldCallBaseEventHandler::Yes;
78}
79
80void BaseCheckableInputType::handleKeypressEvent(KeyboardEvent& event)
81{
82 if (event.charCode() == ' ') {
83 // Prevent scrolling down the page.
84 event.setDefaultHandled();
85 }
86}

117void BaseCheckableInputType::fireInputAndChangeEvents()
118{
119 if (!element()->isConnected())
120 return;
121
122 if (!shouldSendChangeEventAfterCheckedChanged())
123 return;
124
125 auto protectedThis = makeRef(*this);
126 element()->setTextAsOfLastFormControlChangeEvent(String());
127 element()->dispatchInputEvent();
128 if (auto* element = this->element())
129 element->dispatchFormControlChangeEvent();
130}
