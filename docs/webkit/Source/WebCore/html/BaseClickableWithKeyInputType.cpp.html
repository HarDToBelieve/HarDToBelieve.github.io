42auto BaseClickableWithKeyInputType::handleKeydownEvent(HTMLInputElement& element, KeyboardEvent& event) -> ShouldCallBaseEventHandler
43{
44 const String& key = event.keyIdentifier();
45 if (key == "U+0020") {
46 element.setActive(true, true);
47 // No setDefaultHandled(), because IE dispatches a keypress in this case
48 // and the caller will only dispatch a keypress if we don't call setDefaultHandled().
49 return ShouldCallBaseEventHandler::No;
50 }
51 return ShouldCallBaseEventHandler::Yes;
52}
53
54void BaseClickableWithKeyInputType::handleKeypressEvent(HTMLInputElement& element, KeyboardEvent& event)
55{
56 int charCode = event.charCode();
57 if (charCode == '\r') {
58 element.dispatchSimulatedClick(&event);
59 event.setDefaultHandled();
60 return;
61 }
62 if (charCode == ' ') {
63 // Prevent scrolling down the page.
64 event.setDefaultHandled();
65 }
66}
67
68void BaseClickableWithKeyInputType::handleKeyupEvent(InputType& inputType, KeyboardEvent& event)
69{
70 const String& key = event.keyIdentifier();
71 if (key != "U+0020")
72 return;
73 // Simulate mouse click for spacebar for button types.
74 inputType.dispatchSimulatedClickIfActive(event);
75}

77// FIXME: Could share this with BaseCheckableInputType and RangeInputType if we had a common base class.
78bool BaseClickableWithKeyInputType::accessKeyAction(HTMLInputElement& element, bool sendMouseEvents)
79{
80 return element.dispatchSimulatedClick(0, sendMouseEvents ? SendMouseUpDownEvents : SendNoEvents);
81}
82
83auto BaseClickableWithKeyInputType::handleKeydownEvent(KeyboardEvent& event) -> ShouldCallBaseEventHandler
84{
85 ASSERT(element());
86 return handleKeydownEvent(*element(), event);
87}
88
89void BaseClickableWithKeyInputType::handleKeypressEvent(KeyboardEvent& event)
90{
91 ASSERT(element());
92 handleKeypressEvent(*element(), event);
93}
94
95void BaseClickableWithKeyInputType::handleKeyupEvent(KeyboardEvent& event)
96{
97 handleKeyupEvent(*this, event);
98}
99
100bool BaseClickableWithKeyInputType::accessKeyAction(bool sendMouseEvents)
101{
102 InputType::accessKeyAction(sendMouseEvents);
103 ASSERT(element());
104 return accessKeyAction(*element(), sendMouseEvents);
105}
