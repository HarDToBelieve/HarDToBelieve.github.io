68void BaseDateAndTimeInputType::DateTimeFormatValidator::visitField(DateTimeFormat::FieldType fieldType, int)
69{
70 switch (fieldType) {
71 case DateTimeFormat::FieldTypeYear:
72 m_results.add(DateTimeFormatValidationResults::HasYear);
73 break;
74
75 case DateTimeFormat::FieldTypeMonth:
76 case DateTimeFormat::FieldTypeMonthStandAlone:
77 m_results.add(DateTimeFormatValidationResults::HasMonth);
78 break;
79
80 case DateTimeFormat::FieldTypeWeekOfYear:
81 m_results.add(DateTimeFormatValidationResults::HasWeek);
82 break;
83
84 case DateTimeFormat::FieldTypeDayOfMonth:
85 m_results.add(DateTimeFormatValidationResults::HasDay);
86 break;
87
88 case DateTimeFormat::FieldTypePeriod:
89 m_results.add(DateTimeFormatValidationResults::HasMeridiem);
90 break;
91
92 case DateTimeFormat::FieldTypeHour11:
93 case DateTimeFormat::FieldTypeHour12:
94 m_results.add(DateTimeFormatValidationResults::HasHour);
95 break;
96
97 case DateTimeFormat::FieldTypeHour23:
98 case DateTimeFormat::FieldTypeHour24:
99 m_results.add(DateTimeFormatValidationResults::HasHour);
100 m_results.add(DateTimeFormatValidationResults::HasMeridiem);
101 break;
102
103 case DateTimeFormat::FieldTypeMinute:
104 m_results.add(DateTimeFormatValidationResults::HasMinute);
105 break;
106
107 case DateTimeFormat::FieldTypeSecond:
108 m_results.add(DateTimeFormatValidationResults::HasSecond);
109 break;
110
111 default:
112 break;
113 }
114}

165Decimal BaseDateAndTimeInputType::defaultValueForStepUp() const
166{
167 double ms = WallTime::now().secondsSinceEpoch().milliseconds();
168 int offset = calculateLocalTimeOffset(ms).offset / msPerMinute;
169 return Decimal::fromDouble(ms + (offset * msPerMinute));
170}

172Decimal BaseDateAndTimeInputType::parseToNumber(const String& source, const Decimal& defaultValue) const
173{
174 auto date = parseToDateComponents(source);
175 if (!date)
176 return defaultValue;
177 double msec = date->millisecondsSinceEpoch();
178 ASSERT(std::isfinite(msec));
179 return Decimal::fromDouble(msec);
180}

182String BaseDateAndTimeInputType::serialize(const Decimal& value) const
183{
184 if (!value.isFinite())
185 return { };
186 auto date = setMillisecondToDateComponents(value.toDouble());
187 if (!date)
188 return { };
189 return serializeWithComponents(*date);
190}
191
192String BaseDateAndTimeInputType::serializeWithComponents(const DateComponents& date) const
193{
194 ASSERT(element());
195 Decimal step;
196 if (!element()->getAllowedValueStep(&step) || step.remainder(msecPerMinute).isZero())
197 return date.toString();
198 if (step.remainder(msecPerSecond).isZero())
199 return date.toString(SecondFormat::Second);
200 return date.toString(SecondFormat::Millisecond);
201}
202
203String BaseDateAndTimeInputType::serializeWithMilliseconds(double value) const
204{
205 return serialize(Decimal::fromDouble(value));
206}

208String BaseDateAndTimeInputType::localizeValue(const String& proposedValue) const

210 auto date = parseToDateComponents(proposedValue);
211 if (!date)

214 ASSERT(element());

258bool BaseDateAndTimeInputType::shouldHaveSecondField(const DateComponents& date) const
259{
260 if (date.second())
261 return true;
262
263 auto stepRange = createStepRange(AnyStepHandling::Default);
264 return !stepRange.minimum().remainder(msecPerMinute).isZero()
265 || !stepRange.step().remainder(msecPerMinute).isZero();
266}
267
268bool BaseDateAndTimeInputType::shouldHaveMillisecondField(const DateComponents& date) const
269{
270 if (date.millisecond())
271 return true;
272
273 auto stepRange = createStepRange(AnyStepHandling::Default);
274 return !stepRange.minimum().remainder(msecPerSecond).isZero()
275 || !stepRange.step().remainder(msecPerSecond).isZero();
276}

285void BaseDateAndTimeInputType::handleDOMActivateEvent(Event&)
286{
287 ASSERT(element());
288 if (element()->isDisabledOrReadOnly() || !element()->renderer() || !UserGestureIndicator::processingUserGesture())
289 return;
290
291 if (m_dateTimeChooser)
292 return;
293 if (!element()->document().page())
294 return;
295
296 DateTimeChooserParameters parameters;
297 if (!setupDateTimeChooserParameters(parameters))
298 return;
299
300 if (auto chrome = this->chrome()) {
301 m_dateTimeChooser = chrome->createDateTimeChooser(*this);
302 if (m_dateTimeChooser)
303 m_dateTimeChooser->showChooser(parameters);
304 }
305}

307void BaseDateAndTimeInputType::createShadowSubtreeAndUpdateInnerTextElementEditability(ContainerNode::ChildChange::Source source, bool)
308{
309 ASSERT(needsShadowSubtree());
310 ASSERT(element());
311
312 auto& element = *this->element();
313 auto& document = element.document();
314
315 if (document.settings().dateTimeInputsEditableComponentsEnabled()) {
316 m_dateTimeEditElement = DateTimeEditElement::create(document, *this);
317 element.userAgentShadowRoot()->appendChild(source, *m_dateTimeEditElement);
318 } else {
319 static MainThreadNeverDestroyed<const AtomString> valueContainerPseudo("-webkit-date-and-time-value", AtomString::ConstructFromLiteral);
320 auto valueContainer = HTMLDivElement::create(document);
321 valueContainer->setPseudo(valueContainerPseudo);
322 element.userAgentShadowRoot()->appendChild(source, valueContainer);
323 }
324 updateInnerTextValue();
325}

333void BaseDateAndTimeInputType::updateInnerTextValue()
334{
335 ASSERT(element());
336 if (!m_dateTimeEditElement) {
337 auto node = element()->userAgentShadowRoot()->firstChild();
338 if (!is<HTMLElement>(node))
339 return;
340 auto displayValue = visibleValue();
341 if (displayValue.isEmpty()) {
342 // Need to put something to keep text baseline.
343 displayValue = " "_s;
344 }
345 downcast<HTMLElement>(*node).setInnerText(displayValue);
346 return;
347 }
348
349 DateTimeEditElement::LayoutParameters layoutParameters(element()->locale());
350
351 auto date = parseToDateComponents(element()->value());
352 if (date)
353 setupLayoutParameters(layoutParameters, *date);
354 else {
355 if (auto dateForLayout = setMillisecondToDateComponents(createStepRange(AnyStepHandling::Default).minimum().toDouble()))
356 setupLayoutParameters(layoutParameters, *dateForLayout);
357 else
358 setupLayoutParameters(layoutParameters, DateComponents());
359 }
360
361 if (!DateTimeFormatValidator().validateFormat(layoutParameters.dateTimeFormat, *this))
362 layoutParameters.dateTimeFormat = layoutParameters.fallbackDateTimeFormat;
363
364 if (date)
365 m_dateTimeEditElement->setValueAsDate(layoutParameters, *date);
366 else
367 m_dateTimeEditElement->setEmptyValue(layoutParameters);
368}

377void BaseDateAndTimeInputType::attributeChanged(const QualifiedName& name)
378{
379 if (name == maxAttr || name == minAttr) {
380 if (auto* element = this->element())
381 element->invalidateStyleForSubtree();
382 } else if (name == valueAttr) {
383 if (auto* element = this->element()) {
384 if (!element->hasDirtyValue())
385 updateInnerTextValue();
386 }
387 } else if (name == stepAttr && m_dateTimeEditElement)
388 updateInnerTextValue();
389
390 InputType::attributeChanged(name);
391}

431void BaseDateAndTimeInputType::handleFocusEvent(Node* oldFocusedNode, FocusDirection direction)
432{
433 if (!m_dateTimeEditElement) {
434 InputType::handleFocusEvent(oldFocusedNode, direction);
435 return;
436 }
437
438 // If the element contains editable components, the element itself should not
439 // be focused. Instead, one of it's children should receive focus.
440
441 if (direction == FocusDirection::Backward) {
442 // If the element received focus when going backwards, advance the focus one more time
443 // so that this element no longer has focus. In this case, one of the children should
444 // not be focused as the element is losing focus entirely.
445 if (auto* page = element()->document().page())
446 page->focusController().advanceFocus(direction, 0);
447 } else {
448 // If the element received focus in any other direction, transfer focus to the first focusable child.
449 m_dateTimeEditElement->focusByOwner();
450 }
451}

507bool BaseDateAndTimeInputType::setupDateTimeChooserParameters(DateTimeChooserParameters& parameters)
508{
509 ASSERT(element());
510
511 auto& element = *this->element();
512 auto& document = element.document();
513
514 if (!document.view())
515 return false;
516
517 parameters.type = element.type();
518 parameters.minimum = element.minimum();
519 parameters.maximum = element.maximum();
520 parameters.required = element.isRequired();
521
522 if (!document.settings().langAttributeAwareFormControlUIEnabled())
523 parameters.locale = defaultLanguage();
524 else {
525 AtomString computedLocale = element.computeInheritedLanguage();
526 parameters.locale = computedLocale.isEmpty() ? AtomString(defaultLanguage()) : computedLocale;
527 }
528
529 auto stepRange = createStepRange(AnyStepHandling::Reject);
530 if (stepRange.hasStep()) {
531 parameters.step = stepRange.step().toDouble();
532 parameters.stepBase = stepRange.stepBase().toDouble();
533 } else {
534 parameters.step = 1.0;
535 parameters.stepBase = 0;
536 }
537
538 if (RenderElement* renderer = element.renderer())
539 parameters.anchorRectInRootView = document.view()->contentsToRootView(renderer->absoluteBoundingBoxRect());
540 else
541 parameters.anchorRectInRootView = IntRect();
542 parameters.currentValue = element.value();
543
544 auto* computedStyle = element.computedStyle();
545 parameters.isAnchorElementRTL = computedStyle->direction() == TextDirection::RTL;
546 parameters.useDarkAppearance = document.useDarkAppearance(computedStyle);
547
548 auto date = parseToDateComponents(element.value()).value_or(DateComponents());
549 parameters.hasSecondField = shouldHaveSecondField(date);
550 parameters.hasMillisecondField = shouldHaveMillisecondField(date);
551
552#if ENABLE(DATALIST_ELEMENT)
553 if (auto dataList = element.dataList()) {
554 for (auto& option : dataList->suggestions()) {
555 auto label = option.label();
556 auto value = option.value();
557 if (!element.isValidValue(value))
558 continue;
559 parameters.suggestionValues.append(element.sanitizeValue(value));
560 parameters.localizedSuggestionValues.append(element.localizeValue(value));
561 parameters.suggestionLabels.append(value == label ? String() : label);
562 }
563 }
564#endif
565
566 return true;
567}
