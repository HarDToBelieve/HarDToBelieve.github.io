49// A super class of date, datetime, datetime-local, month, time, and week types.
50class BaseDateAndTimeInputType : public InputType, private DateTimeChooserClient, private DateTimeEditElement::EditControlOwner {
51protected:
52 enum class DateTimeFormatValidationResults : uint8_t {
53 HasYear = 1 << 0,
54 HasMonth = 1 << 1,
55 HasWeek = 1 << 2,
56 HasDay = 1 << 3,
57 HasHour = 1 << 4,
58 HasMinute = 1 << 5,
59 HasSecond = 1 << 6,
60 HasMeridiem = 1 << 7,
61 };
62
63 BaseDateAndTimeInputType(Type type, HTMLInputElement& element)
64 : InputType(type, element)
65 {
66 ASSERT(needsShadowSubtree());
67 }
68
69 ~BaseDateAndTimeInputType();
70
71 Decimal parseToNumber(const String&, const Decimal&) const override;
72 String serialize(const Decimal&) const override;
73 String serializeWithComponents(const DateComponents&) const;
74
75 bool shouldHaveSecondField(const DateComponents&) const;
76 bool shouldHaveMillisecondField(const DateComponents&) const;
77 bool typeMismatchFor(const String&) const final;
78 bool typeMismatch() const final;
79 bool valueMissing(const String&) const final;
80
81private:
82 class DateTimeFormatValidator final : public DateTimeFormat::TokenHandler {
83 public:
84 DateTimeFormatValidator() { }
85
86 void visitField(DateTimeFormat::FieldType, int);
87 void visitLiteral(const String&) { }
88
89 bool validateFormat(const String& format, const BaseDateAndTimeInputType&);
90
91 private:
92 OptionSet<DateTimeFormatValidationResults> m_results;
93 };
94
95 virtual std::optional<DateComponents> parseToDateComponents(const StringView&) const = 0;
96 virtual std::optional<DateComponents> setMillisecondToDateComponents(double) const = 0;
97 virtual void setupLayoutParameters(DateTimeEditElement::LayoutParameters&, const DateComponents&) const = 0;
98 virtual bool isValidFormat(OptionSet<DateTimeFormatValidationResults>) const = 0;
99 virtual String serializeWithMilliseconds(double) const;

103 String sanitizeValue(const String&) const final;
104 void setValue(const String&, bool valueChanged, TextFieldEventBehavior) final;
105 double valueAsDate() const override;

107 double valueAsDouble() const final;
108 ExceptionOr<void> setValueAsDecimal(const Decimal&, TextFieldEventBehavior) const final;
109 Decimal defaultValueForStepUp() const override;
110 String localizeValue(const String&) const final;
111 bool supportsReadOnly() const final;
112 bool shouldRespectListAttribute() final;
113 bool isKeyboardFocusable(KeyboardEvent*) const final;
114 bool isMouseFocusable() const final;
115
116 void handleDOMActivateEvent(Event&) override;
117 void createShadowSubtreeAndUpdateInnerTextElementEditability(ContainerNode::ChildChange::Source, bool) final;

119 void updateInnerTextValue() final;

123 void elementDidBlur() final;
124 void detach() final;

126 ShouldCallBaseEventHandler handleKeydownEvent(KeyboardEvent&) override;
127 void handleKeypressEvent(KeyboardEvent&) override;

129 void handleFocusEvent(Node* oldFocusedNode, FocusDirection) final;
130 bool accessKeyAction(bool sendMouseEvents) override;

134 void didChangeValueFromControl() final;
