36bool BaseTextInputType::patternMismatch(const String& value) const
37{
38 ASSERT(element());
39 // FIXME: We should execute RegExp parser first to check validity instead of creating an actual RegularExpression.
40 // https://bugs.webkit.org/show_bug.cgi?id=183361
41 const AtomString& rawPattern = element()->attributeWithoutSynchronization(patternAttr);
42 if (rawPattern.isNull() || value.isEmpty() || !JSC::Yarr::RegularExpression(rawPattern, JSC::Yarr::TextCaseSensitive, JSC::Yarr::MultilineDisabled, JSC::Yarr::UnicodeAwareMode).isValid())
43 return false;
44 String pattern = "^(?:" + rawPattern + ")$";
45 int matchLength = 0;
46 int valueLength = value.length();
47 int matchOffset = JSC::Yarr::RegularExpression(pattern, JSC::Yarr::TextCaseSensitive, JSC::Yarr::MultilineDisabled, JSC::Yarr::UnicodeAwareMode).match(value, 0, &matchLength);
48 return matchOffset || matchLength != valueLength;
49}
