58CanvasBase::~CanvasBase()
59{

65GraphicsContext* CanvasBase::drawingContext() const
66{
67 auto* context = renderingContext();
68 if (context && !context->is2d() && !context->isOffscreen2d())
69 return nullptr;
70
71 return buffer() ? &m_imageBuffer->context() : nullptr;
72}

74GraphicsContext* CanvasBase::existingDrawingContext() const
75{
76 if (!hasCreatedImageBuffer())
77 return nullptr;
78
79 return drawingContext();
80}

101size_t CanvasBase::memoryCost() const
102{
103 // memoryCost() may be invoked concurrently from a GC thread, and we need to be careful
104 // about what data we access here and how. We need to hold a lock to prevent m_imageBuffer
105 // from being changed while we access it.
106 Locker locker { m_imageBufferAssignmentLock };
107 if (!m_imageBuffer)
108 return 0;
109 return m_imageBuffer->memoryCost();
110}

120 return m_imageBuffer->externalMemoryCost();

151void CanvasBase::notifyObserversCanvasDestroyed()
152{
153 ASSERT(!m_didNotifyObserversCanvasDestroyed);
154
155 for (auto& observer : copyToVector(m_observers))
156 observer->canvasDestroyed(*this);
157
158 m_observers.clear();
159
160#if ASSERT_ENABLED
161 m_didNotifyObserversCanvasDestroyed = true;
162#endif
163}

165HashSet<Element*> CanvasBase::cssCanvasClients() const
166{
167 HashSet<Element*> cssCanvasClients;
168 for (auto& observer : m_observers) {
169 if (!is<CSSCanvasValue::CanvasObserverProxy>(observer))
170 continue;
171
172 auto clients = downcast<CSSCanvasValue::CanvasObserverProxy>(observer)->ownerValue().clients();
173 for (auto& entry : clients) {
174 if (RefPtr<Element> element = entry.key->element())
175 cssCanvasClients.add(element.get());
176 }
177 }
178 return cssCanvasClients;
179}

187RefPtr<ImageBuffer> CanvasBase::setImageBuffer(RefPtr<ImageBuffer>&& buffer) const
188{
189 RefPtr<ImageBuffer> returnBuffer;
190 {
191 Locker locker { m_imageBufferAssignmentLock };
192 m_contextStateSaver = nullptr;
193 returnBuffer = std::exchange(m_imageBuffer, WTFMove(buffer));
194 }
195
196 if (m_imageBuffer && m_size != m_imageBuffer->logicalSize())
197 m_size = m_imageBuffer->logicalSize();
198
199 size_t previousMemoryCost = m_imageBufferCost;
200 m_imageBufferCost = memoryCost();
201 s_activePixelMemory += m_imageBufferCost - previousMemoryCost;
202
203 auto* context = renderingContext();
204 if (context && m_imageBuffer && previousMemoryCost != m_imageBufferCost)
205 InspectorInstrumentation::didChangeCanvasMemory(*context);
206
207 if (m_imageBuffer) {
208 m_imageBuffer->context().setShadowsIgnoreTransforms(true);
209 m_imageBuffer->context().setImageInterpolationQuality(defaultInterpolationQuality);
210 m_imageBuffer->context().setStrokeThickness(1);
211 m_contextStateSaver = makeUnique<GraphicsContextStateSaver>(m_imageBuffer->context());
212
213 JSC::JSLockHolder lock(scriptExecutionContext()->vm());
214 scriptExecutionContext()->vm().heap.reportExtraMemoryAllocated(memoryCost());
215 }
216
217 return returnBuffer;
218}

225void CanvasBase::resetGraphicsContextState() const
226{
227 if (m_contextStateSaver) {

230 m_contextStateSaver->save();
