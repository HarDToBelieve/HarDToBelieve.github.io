57// https://html.spec.whatwg.org/multipage/infrastructure.html#valid-simple-colour
58static bool isValidSimpleColor(StringView string)
59{
60 if (string.length() != 7)
61 return false;
62 if (string[0] != '#')
63 return false;
64 for (unsigned i = 1; i < 7; ++i) {
65 if (!isASCIIHexDigit(string[i]))
66 return false;
67 }
68 return true;
69}
70
71// https://html.spec.whatwg.org/multipage/common-microsyntaxes.html#rules-for-parsing-simple-colour-values
72static std::optional<SRGBA<uint8_t>> parseSimpleColorValue(StringView string)
73{
74 if (!isValidSimpleColor(string))
75 return std::nullopt;
76 return { { toASCIIHexValue(string[1], string[2]), toASCIIHexValue(string[3], string[4]), toASCIIHexValue(string[5], string[6]) } };
77}

123String ColorInputType::sanitizeValue(const String& proposedValue) const
124{
125 if (!isValidSimpleColor(proposedValue))
126 return fallbackValue();

128 return proposedValue.convertToASCIILowercase();

137void ColorInputType::createShadowSubtreeAndUpdateInnerTextElementEditability(ContainerNode::ChildChange::Source source, bool)
138{
139 ASSERT(needsShadowSubtree());
140 ASSERT(element());
141 ASSERT(element()->shadowRoot());
142
143 static MainThreadNeverDestroyed<const AtomString> webkitColorSwatchName("-webkit-color-swatch", AtomString::ConstructFromLiteral);
144 static MainThreadNeverDestroyed<const AtomString> webkitColorSwatchWrapperName("-webkit-color-swatch-wrapper", AtomString::ConstructFromLiteral);
145
146 Document& document = element()->document();
147 auto wrapperElement = HTMLDivElement::create(document);
148 wrapperElement->setPseudo(webkitColorSwatchWrapperName);
149 auto colorSwatch = HTMLDivElement::create(document);
150 colorSwatch->setPseudo(webkitColorSwatchName);
151 wrapperElement->appendChild(source, colorSwatch);
152 element()->userAgentShadowRoot()->appendChild(source, wrapperElement);
153
154 updateColorSwatch();
155}

157void ColorInputType::setValue(const String& value, bool valueChanged, TextFieldEventBehavior eventBehavior)
158{
159 InputType::setValue(value, valueChanged, eventBehavior);
160
161 if (!valueChanged)
162 return;
163
164 updateColorSwatch();
165 if (m_chooser)
166 m_chooser->setSelectedColor(valueAsColor());
167}

177void ColorInputType::handleDOMActivateEvent(Event& event)
178{
179 ASSERT(element());
180 if (element()->isDisabledFormControl() || !element()->renderer())
181 return;
182
183 if (!UserGestureIndicator::processingUserGesture())
184 return;
185
186 if (Chrome* chrome = this->chrome()) {
187 if (!m_chooser)
188 m_chooser = chrome->createColorChooser(*this, valueAsColor());
189 else
190 m_chooser->reattachColorChooser(valueAsColor());
191 }
192
193 event.setDefaultHandled();
194}

221void ColorInputType::didChooseColor(const Color& color)
222{
223 ASSERT(element());
224 if (element()->isDisabledFormControl() || color == valueAsColor())
225 return;
226 EventQueueScope scope;
227 element()->setValueFromRenderer(serializationForHTML(color));
228 updateColorSwatch();
229 element()->dispatchFormControlChangeEvent();
230}
231
232void ColorInputType::didEndChooser()
233{
234 m_chooser = nullptr;
235 if (element()->renderer())
236 element()->renderer()->repaint();
237}

239void ColorInputType::endColorChooser()

241 if (m_chooser)
242 m_chooser->endChooser();
243}

245void ColorInputType::updateColorSwatch()
246{
247 RefPtr<HTMLElement> colorSwatch = shadowColorSwatch();
248 if (!colorSwatch)
249 return;
250
251 ASSERT(element());
252 colorSwatch->setInlineStyleProperty(CSSPropertyBackgroundColor, element()->value(), false);
253}
254
255HTMLElement* ColorInputType::shadowColorSwatch() const
256{
257 ASSERT(element());
258 RefPtr<ShadowRoot> shadow = element()->userAgentShadowRoot();
259 if (!shadow)
260 return nullptr;
261
262 auto wrapper = childrenOfType<HTMLDivElement>(*shadow).first();
263 if (!wrapper)
264 return nullptr;
265
266 return childrenOfType<HTMLDivElement>(*wrapper).first();
267}

277Vector<Color> ColorInputType::suggestedColors() const
278{
279 Vector<Color> suggestions;
280#if ENABLE(DATALIST_ELEMENT)
281 ASSERT(element());
282 if (auto dataList = element()->dataList()) {
283 for (auto& option : dataList->suggestions()) {
284 if (auto color = parseSimpleColorValue(option.value()))
285 suggestions.append(*color);
286 }
287 }
288#endif
289 return suggestions;
290}
291
292void ColorInputType::selectColor(StringView string)
293{
294 if (auto color = parseSimpleColorValue(string))
295 didChooseColor(color.value());
296}
