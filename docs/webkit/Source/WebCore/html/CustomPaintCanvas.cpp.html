42CustomPaintCanvas::CustomPaintCanvas(ScriptExecutionContext& context, unsigned width, unsigned height)
43 : CanvasBase(IntSize(width, height))
44 , ContextDestructionObserver(&context)
45{
46}
47
48CustomPaintCanvas::~CustomPaintCanvas()
49{
50 notifyObserversCanvasDestroyed();
51
52 m_context = nullptr; // Ensure this goes away before the ImageBuffer.
53 setImageBuffer(nullptr);
54}
55
56ExceptionOr<RefPtr<PaintRenderingContext2D>> CustomPaintCanvas::getContext()
57{
58 if (m_context)
59 return { RefPtr<PaintRenderingContext2D> { &downcast<PaintRenderingContext2D>(*m_context) } };
60
61 m_context = PaintRenderingContext2D::create(*this);
62 downcast<PaintRenderingContext2D>(*m_context).setUsesDisplayListDrawing(true);
63
64 return { RefPtr<PaintRenderingContext2D> { &downcast<PaintRenderingContext2D>(*m_context) } };
65}

67void CustomPaintCanvas::replayDisplayList(GraphicsContext* ctx) const
68{
69 ASSERT(!m_destinationGraphicsContext);
70 if (!width() || !height())
71 return;
72
73 // FIXME: Using an intermediate buffer is not needed if there are no composite operations.
74 auto clipBounds = ctx->clipBounds();
75
76 auto image = ImageBuffer::createCompatibleBuffer(clipBounds.size(), *ctx);
77 if (!image)
78 return;
79
80 m_destinationGraphicsContext = &image->context();
81 m_destinationGraphicsContext->translate(-clipBounds.location());
82 if (m_context)
83 m_context->paintRenderingResultsToCanvas();
84 m_destinationGraphicsContext = nullptr;
85
86 ctx->drawImageBuffer(*image, clipBounds);
87}
88
89Image* CustomPaintCanvas::copiedImage() const
90{
91 ASSERT(!m_destinationGraphicsContext);
92 if (!width() || !height())
93 return nullptr;
94
95 m_copiedBuffer = ImageBuffer::create(size(), RenderingMode::Unaccelerated, 1, DestinationColorSpace::SRGB(), PixelFormat::BGRA8, nullptr);
96 if (!m_copiedBuffer)
97 return nullptr;
98
99 m_destinationGraphicsContext = &m_copiedBuffer->context();
100 if (m_context)
101 m_context->paintRenderingResultsToCanvas();
102 m_destinationGraphicsContext = nullptr;
103
104 m_copiedImage = m_copiedBuffer->copyImage(DontCopyBackingStore, PreserveResolution::Yes);
105 return m_copiedImage.get();
106}
107
108void CustomPaintCanvas::clearCopiedImage() const
109{
110 m_copiedImage = nullptr;
111}
