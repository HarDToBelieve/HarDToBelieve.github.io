62static RefPtr<TypedOMCSSStyleValue> extractComputedProperty(const String& name, Element& element)
63{
64 ComputedStyleExtractor extractor(&element);
65
66 if (isCustomPropertyName(name)) {
67 auto value = extractor.customPropertyValue(name);
68 return StylePropertyMapReadOnly::customPropertyValueOrDefault(name, element.document(), value.get(), &element);
69 }
70
71 CSSPropertyID propertyID = cssPropertyID(name);
72 if (!propertyID)
73 return nullptr;
74
75 auto value = extractor.propertyValue(propertyID, DoNotUpdateLayout);
76 return StylePropertyMapReadOnly::reifyValue(value.get(), element.document(), &element);
77}

116ImageDrawResult CustomPaintImage::doCustomPaint(GraphicsContext& destContext, const FloatSize& destSize)
117{
118 if (!m_element || !m_element->element() || !m_paintDefinition)
119 return ImageDrawResult::DidNothing;
120
121 JSC::JSValue paintConstructor = m_paintDefinition->paintConstructor;
122
123 if (!paintConstructor)
124 return ImageDrawResult::DidNothing;
125
126 ASSERT(!m_element->needsLayout());
127 ASSERT(!m_element->element()->document().needsStyleRecalc());
128
129 JSCSSPaintCallback& callback = static_cast<JSCSSPaintCallback&>(m_paintDefinition->paintCallback.get());
130 auto* scriptExecutionContext = callback.scriptExecutionContext();
131 if (!scriptExecutionContext)
132 return ImageDrawResult::DidNothing;
133
134 auto canvas = CustomPaintCanvas::create(*scriptExecutionContext, destSize.width(), destSize.height());
135 ExceptionOr<RefPtr<PaintRenderingContext2D>> contextOrException = canvas->getContext();
136
137 if (contextOrException.hasException())
138 return ImageDrawResult::DidNothing;
139 auto context = contextOrException.releaseReturnValue();
140
141 HashMap<String, RefPtr<TypedOMCSSStyleValue>> propertyValues;
142
143 if (auto* element = m_element->element()) {
144 for (auto& name : m_inputProperties)
145 propertyValues.add(name, extractComputedProperty(name, *element));
146 }
147
148 auto size = CSSPaintSize::create(destSize.width(), destSize.height());
149 Ref<StylePropertyMapReadOnly> propertyMap = HashMapStylePropertyMap::create(WTFMove(propertyValues));
150
151 auto& vm = paintConstructor.getObject()->vm();
152 JSC::JSLockHolder lock(vm);
153 auto scope = DECLARE_THROW_SCOPE(vm);
154 auto& globalObject = *paintConstructor.getObject()->globalObject();
155
156 auto& lexicalGlobalObject = globalObject;
157 JSC::ArgList noArgs;
158 JSC::JSValue thisObject = { JSC::construct(&lexicalGlobalObject, paintConstructor, noArgs, "Failed to construct paint class") };
159
160 if (UNLIKELY(scope.exception())) {
161 reportException(&lexicalGlobalObject, scope.exception());
162 return ImageDrawResult::DidNothing;
163 }
164
165 auto result = callback.handleEvent(WTFMove(thisObject), *context, size, propertyMap, m_arguments);
166 if (result.type() != CallbackResultType::Success)
167 return ImageDrawResult::DidNothing;
168
169 canvas->replayDisplayList(&destContext);
170
171 return ImageDrawResult::DidDraw;
172}

174ImageDrawResult CustomPaintImage::draw(GraphicsContext& destContext, const FloatRect& destRect, const FloatRect& srcRect, const ImagePaintingOptions& options)
175{
176 GraphicsContextStateSaver stateSaver(destContext);
177 destContext.setCompositeOperation(options.compositeOperator(), options.blendMode());
178 destContext.clip(destRect);
179 destContext.translate(destRect.location());
180 if (destRect.size() != srcRect.size())
181 destContext.scale(destRect.size() / srcRect.size());
182 destContext.translate(-srcRect.location());
183 return doCustomPaint(destContext, size());
184}

186void CustomPaintImage::drawPattern(GraphicsContext& destContext, const FloatRect& destRect, const FloatRect& srcRect, const AffineTransform& patternTransform,
187 const FloatPoint& phase, const FloatSize& spacing, const ImagePaintingOptions& options)
188{
189 // Allow the generator to provide visually-equivalent tiling parameters for better performance.
190 FloatSize adjustedSize = size();
191 FloatRect adjustedSrcRect = srcRect;
192
193 // Factor in the destination context's scale to generate at the best resolution
194 AffineTransform destContextCTM = destContext.getCTM(GraphicsContext::DefinitelyIncludeDeviceScale);
195 double xScale = fabs(destContextCTM.xScale());
196 double yScale = fabs(destContextCTM.yScale());
197 AffineTransform adjustedPatternCTM = patternTransform;
198 adjustedPatternCTM.scale(1.0 / xScale, 1.0 / yScale);
199 adjustedSrcRect.scale(xScale, yScale);
200
201 auto buffer = ImageBuffer::createCompatibleBuffer(adjustedSize, DestinationColorSpace::SRGB(), destContext);
202 if (!buffer)
203 return;
204 doCustomPaint(buffer->context(), adjustedSize);
205
206 if (destContext.drawLuminanceMask())
207 buffer->convertToLuminanceMask();
208
209 buffer->drawPattern(destContext, destRect, adjustedSrcRect, adjustedPatternCTM, phase, spacing, options);
210 destContext.setDrawLuminanceMask(false);
211}
