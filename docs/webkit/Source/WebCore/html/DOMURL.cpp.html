42inline DOMURL::DOMURL(URL&& completeURL, const URL& baseURL)
43 : m_baseURL(baseURL)
44 , m_url(WTFMove(completeURL))

46}

48ExceptionOr<Ref<DOMURL>> DOMURL::create(const String& url, const URL& base)
49{
50 ASSERT(base.isValid() || base.isNull());
51 URL completeURL { base, url };
52 if (!completeURL.isValid())
53 return Exception { TypeError };
54 return adoptRef(*new DOMURL(WTFMove(completeURL), base));
55}
56
57ExceptionOr<Ref<DOMURL>> DOMURL::create(const String& url, const String& base)
58{
59 URL baseURL { URL { }, base };
60 if (!base.isNull() && !baseURL.isValid())
61 return Exception { TypeError };
62 return create(url, baseURL);
63}
64
65ExceptionOr<Ref<DOMURL>> DOMURL::create(const String& url, const DOMURL& base)
66{
67 return create(url, base.href());
68}
69
70DOMURL::~DOMURL()
71{
72 if (m_searchParams)
73 m_searchParams->associatedURLDestroyed();
74}
75
76ExceptionOr<void> DOMURL::setHref(const String& url)
77{
78 URL completeURL { m_baseURL, url };
79 if (!completeURL.isValid())
80 return Exception { TypeError };
81 m_url = WTFMove(completeURL);
82 if (m_searchParams)
83 m_searchParams->updateFromAssociatedURL();
84 return { };
85}
86
87void DOMURL::setQuery(const String& query)
88{
89 m_url.setQuery(query);
90}
91
92String DOMURL::createObjectURL(ScriptExecutionContext& scriptExecutionContext, Blob& blob)
93{
94 return createPublicURL(scriptExecutionContext, blob);
95}
96
97String DOMURL::createPublicURL(ScriptExecutionContext& scriptExecutionContext, URLRegistrable& registrable)
98{
99 URL publicURL = BlobURL::createPublicURL(scriptExecutionContext.securityOrigin());
100 if (publicURL.isEmpty())
101 return String();
102
103 scriptExecutionContext.publicURLManager().registerURL(publicURL, registrable);
104
105 return publicURL.string();
106}
107
108URLSearchParams& DOMURL::searchParams()
109{
110 if (!m_searchParams)
111 m_searchParams = URLSearchParams::create(search(), this);
112 return *m_searchParams;
113}
114
115void DOMURL::revokeObjectURL(ScriptExecutionContext& scriptExecutionContext, const String& urlString)
116{
117 URL url(URL(), urlString);
118 ResourceRequest request(url);
119 request.setDomainForCachePartition(scriptExecutionContext.domainForCachePartition());
120
121 MemoryCache::removeRequestFromSessionCaches(scriptExecutionContext, request);
122
123 scriptExecutionContext.publicURLManager().revoke(url);
124}
