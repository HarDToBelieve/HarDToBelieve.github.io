29#if ENABLE(DATE_AND_TIME_INPUT_TYPES)

31#include <optional>
32
33namespace WebCore {
34
35struct DateTimeFieldsState {
36 enum class Meridiem : bool { AM, PM };
37
38 std::optional<unsigned> hour23() const
39 {
40 if (!hour || !meridiem)
41 return std::nullopt;
42 return (*hour % 12) + (*meridiem == DateTimeFieldsState::Meridiem::PM ? 12 : 0);
43 }
44
45 std::optional<unsigned> year;
46 std::optional<unsigned> month;
47 std::optional<unsigned> dayOfMonth;
48 std::optional<unsigned> hour;
49 std::optional<unsigned> minute;
50 std::optional<unsigned> second;
51 std::optional<unsigned> millisecond;
52 std::optional<Meridiem> meridiem;
53};
54
55} // namespace WebCore
56
57#endif // ENABLE(DATE_AND_TIME_INPUT_TYPES)
