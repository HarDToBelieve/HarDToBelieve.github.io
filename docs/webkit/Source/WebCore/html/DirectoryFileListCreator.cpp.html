42struct FileInformation {
43 String path;
44 String relativePath;
45 String displayName;
46
47 FileInformation isolatedCopy() const
48 {
49 return FileInformation { path.isolatedCopy(), relativePath.isolatedCopy(), displayName.isolatedCopy() };
50 }
51};
52
53static void appendDirectoryFiles(const String& directory, const String& relativePath, Vector<FileInformation>& files)
54{
55 ASSERT(!isMainThread());
56 for (auto& childName : FileSystem::listDirectory(directory)) {
57 auto childPath = FileSystem::pathByAppendingComponent(directory, childName);
58 if (FileSystem::isHiddenFile(childPath))
59 continue;
60
61 auto fileType = FileSystem::fileType(childPath);
62 if (!fileType)
63 continue;
64
65 String childRelativePath = relativePath + "/" + childName;
66 if (*fileType == FileSystem::FileType::Directory)
67 appendDirectoryFiles(childPath, childRelativePath, files);
68 else if (*fileType == FileSystem::FileType::Regular)
69 files.append(FileInformation { childPath, childRelativePath, { } });
70 }
71}
72
73static Vector<FileInformation> gatherFileInformation(const Vector<FileChooserFileInfo>& paths)
74{
75 ASSERT(!isMainThread());
76 Vector<FileInformation> files;
77 for (auto& info : paths) {
78 if (FileSystem::fileType(info.path) == FileSystem::FileType::Directory)
79 appendDirectoryFiles(info.path, FileSystem::pathFileName(info.path), files);
80 else
81 files.append(FileInformation { info.path, { }, info.displayName });
82 }
83 return files;
84}
85
86static Ref<FileList> toFileList(Document* document, const Vector<FileInformation>& files)
87{
88 ASSERT(isMainThread());
89 Vector<Ref<File>> fileObjects;
90 for (auto& file : files) {
91 if (file.relativePath.isNull())
92 fileObjects.append(File::create(document, file.path, { }, file.displayName));
93 else
94 fileObjects.append(File::createWithRelativePath(document, file.path, file.relativePath));
95 }
96 return FileList::create(WTFMove(fileObjects));
97}

99DirectoryFileListCreator::DirectoryFileListCreator(CompletionHandler&& completionHandler)
100 : m_workQueue(WorkQueue::create("DirectoryFileListCreator Work Queue"))
101 , m_completionHandler(WTFMove(completionHandler))
102{
103}

105void DirectoryFileListCreator::start(Document* document, const Vector<FileChooserFileInfo>& paths)
106{
107 // Resolve directories on a background thread to avoid blocking the main thread.
108 m_workQueue->dispatch([this, protectedThis = makeRef(*this), document = makeRefPtr(document), paths = crossThreadCopy(paths)]() mutable {
109 auto files = gatherFileInformation(paths);
110 callOnMainThread([this, protectedThis = WTFMove(protectedThis), document = WTFMove(document), files = crossThreadCopy(files)]() mutable {
111 if (auto completionHandler = std::exchange(m_completionHandler, nullptr))
112 completionHandler(toFileList(document.get(), files));
113 });
114 });
115}
116
117void DirectoryFileListCreator::cancel()
118{
119 m_completionHandler = nullptr;
120 m_workQueue = nullptr;
121}
