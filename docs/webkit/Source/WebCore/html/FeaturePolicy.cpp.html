40static const char* policyTypeName(FeaturePolicy::Type type)
41{
42 switch (type) {
43 case FeaturePolicy::Type::Camera:
44 return "Camera";
45 case FeaturePolicy::Type::Microphone:
46 return "Microphone";
47 case FeaturePolicy::Type::SpeakerSelection:
48 return "SpeakerSelection";
49 case FeaturePolicy::Type::DisplayCapture:
50 return "DisplayCapture";
51 case FeaturePolicy::Type::Geolocation:
52 return "Geolocation";
53 case FeaturePolicy::Type::SyncXHR:
54 return "SyncXHR";
55 case FeaturePolicy::Type::Fullscreen:
56 return "Fullscreen";
57#if ENABLE(DEVICE_ORIENTATION)
58 case FeaturePolicy::Type::Gyroscope:
59 return "Gyroscope";

71 return "";

74bool isFeaturePolicyAllowedByDocumentAndAllOwners(FeaturePolicy::Type type, const Document& document, LogFeaturePolicyFailure logFailure)
75{
76 auto& topDocument = document.topDocument();
77 auto* ancestorDocument = &document;
78 while (ancestorDocument != &topDocument) {
79 if (!ancestorDocument) {
80 if (logFailure == LogFeaturePolicyFailure::Yes && document.domWindow())
81 document.domWindow()->printErrorMessage(makeString("Feature policy '", policyTypeName(type), "' check failed."));
82 return false;
83 }
84
85 auto* ownerElement = ancestorDocument->ownerElement();
86 if (is<HTMLIFrameElement>(ownerElement)) {
87 const auto& featurePolicy = downcast<HTMLIFrameElement>(ownerElement)->featurePolicy();
88 if (!featurePolicy.allows(type, ancestorDocument->securityOrigin().data())) {
89 if (logFailure == LogFeaturePolicyFailure::Yes && document.domWindow()) {
90 auto& allowValue = downcast<HTMLIFrameElement>(ownerElement)->attributeWithoutSynchronization(HTMLNames::allowAttr);
91 document.domWindow()->printErrorMessage(makeString("Feature policy '", policyTypeName(type), "' check failed for iframe with origin '", document.securityOrigin().toString(), "' and allow attribute '", allowValue, "'."));
92 }
93 return false;
94 }
95 }
96
97 ancestorDocument = ancestorDocument->parentDocument();
98 }
99
100 return true;
101}

103static bool isAllowedByFeaturePolicy(const FeaturePolicy::AllowRule& rule, const SecurityOriginData& origin)
104{
105 switch (rule.type) {
106 case FeaturePolicy::AllowRule::Type::None:
107 return false;
108 case FeaturePolicy::AllowRule::Type::All:
109 return true;
110 case FeaturePolicy::AllowRule::Type::List:
111 return rule.allowedList.contains(origin);
112 }
113 ASSERT_NOT_REACHED();
114 return false;
115}
116
117static inline void processOriginItem(Document& document, FeaturePolicy::AllowRule& rule, StringView item)
118{
119 if (rule.type == FeaturePolicy::AllowRule::Type::None)
120 return;
121
122 item = item.stripLeadingAndTrailingMatchedCharacters(isHTMLSpace<UChar>);
123 // FIXME: Support 'src'.
124 if (item == "'src'")
125 return;
126
127 if (item == "*") {
128 rule.type = FeaturePolicy::AllowRule::Type::All;
129 return;
130 }
131
132 if (item == "'self'") {
133 rule.allowedList.add(document.securityOrigin().data());
134 return;
135 }
136 if (item == "'none'") {
137 rule.type = FeaturePolicy::AllowRule::Type::None;
138 return;
139 }
140 URL url { { }, item.toString() };
141 if (url.isValid())
142 rule.allowedList.add(SecurityOriginData::fromURL(url));
143}
144
145static inline void updateList(Document& document, FeaturePolicy::AllowRule& rule, StringView value)
146{
147 // We keep the empty string value equivalent to '*' for existing websites.
148 if (value.isEmpty()) {
149 rule.type = FeaturePolicy::AllowRule::Type::All;
150 return;
151 }
152
153 while (!value.isEmpty()) {
154 auto position = value.find(isHTMLSpace<UChar>);
155 if (position == notFound) {
156 processOriginItem(document, rule, value);
157 return;
158 }
159
160 processOriginItem(document, rule, value.substring(0, position));
161 value = value.substring(position + 1).stripLeadingAndTrailingMatchedCharacters(isHTMLSpace<UChar>);
162 }
163}

165FeaturePolicy FeaturePolicy::parse(Document& document, const HTMLIFrameElement& iframe, StringView allowAttributeValue)
166{
167 FeaturePolicy policy;
168 bool isCameraInitialized = false;
169 bool isMicrophoneInitialized = false;
170 bool isSpeakerSelectionInitialized = false;
171 bool isDisplayCaptureInitialized = false;
172 bool isGeolocationInitialized = false;
173 bool isSyncXHRInitialized = false;
174 bool isFullscreenInitialized = false;
175#if ENABLE(DEVICE_ORIENTATION)
176 bool isGyroscopeInitialized = false;
177 bool isAccelerometerInitialized = false;
178 bool isMagnetometerInitialized = false;
179#endif
180#if ENABLE(WEBXR)
181 bool isXRSpatialTrackingInitialized = false;
182#endif
183 for (auto allowItem : allowAttributeValue.split(';')) {
184 auto item = allowItem.stripLeadingAndTrailingMatchedCharacters(isHTMLSpace<UChar>);
185 if (item.startsWith("camera")) {
186 isCameraInitialized = true;
187 updateList(document, policy.m_cameraRule, item.substring(7));
188 continue;
189 }
190 if (item.startsWith("microphone")) {
191 isMicrophoneInitialized = true;
192 updateList(document, policy.m_microphoneRule, item.substring(11));
193 continue;
194 }
195 if (item.startsWith("speaker-selection")) {
196 isSpeakerSelectionInitialized = true;
197 updateList(document, policy.m_speakerSelectionRule, item.substring(18));
198 continue;
199 }
200 if (item.startsWith("display-capture")) {
201 isDisplayCaptureInitialized = true;
202 updateList(document, policy.m_displayCaptureRule, item.substring(16));
203 continue;
204 }
205 if (item.startsWith("geolocation")) {
206 isGeolocationInitialized = true;
207 updateList(document, policy.m_geolocationRule, item.substring(12));
208 continue;
209 }
210 if (item.startsWith("sync-xhr")) {
211 isSyncXHRInitialized = true;
212 updateList(document, policy.m_syncXHRRule, item.substring(9));
213 continue;
214 }
215 if (item.startsWith("fullscreen")) {
216 isFullscreenInitialized = true;
217 updateList(document, policy.m_fullscreenRule, item.substring(11));
218 continue;
219 }
220#if ENABLE(DEVICE_ORIENTATION)
221 if (item.startsWith("gyroscope")) {
222 isGyroscopeInitialized = true;
223 updateList(document, policy.m_gyroscopeRule, item.substring(10));
224 continue;
225 }
226 if (item.startsWith("accelerometer")) {
227 isAccelerometerInitialized = true;
228 updateList(document, policy.m_accelerometerRule, item.substring(14));
229 continue;
230 }
231 if (item.startsWith("magnetometer")) {
232 isMagnetometerInitialized = true;
233 updateList(document, policy.m_magnetometerRule, item.substring(13));
234 continue;
235 }
236#endif
237#if ENABLE(WEBXR)
238 if (item.startsWith("xr-spatial-tracking")) {
239 isXRSpatialTrackingInitialized = true;
240 updateList(document, policy.m_xrSpatialTrackingRule, item.substring(19));
241 continue;
242 }
243#endif
244 }
245
246 // By default, camera, microphone, speaker-selection, display-capture, fullscreen and xr-spatial-tracking policy is 'self'.
247 if (!isCameraInitialized)
248 policy.m_cameraRule.allowedList.add(document.securityOrigin().data());
249 if (!isMicrophoneInitialized)
250 policy.m_microphoneRule.allowedList.add(document.securityOrigin().data());
251 if (!isSpeakerSelectionInitialized)
252 policy.m_speakerSelectionRule.allowedList.add(document.securityOrigin().data());
253 if (!isDisplayCaptureInitialized)
254 policy.m_displayCaptureRule.allowedList.add(document.securityOrigin().data());
255 if (!isGeolocationInitialized)
256 policy.m_geolocationRule.allowedList.add(document.securityOrigin().data());
257#if ENABLE(DEVICE_ORIENTATION)
258 if (!isGyroscopeInitialized)
259 policy.m_gyroscopeRule.allowedList.add(document.securityOrigin().data());
260 if (!isAccelerometerInitialized)
261 policy.m_accelerometerRule.allowedList.add(document.securityOrigin().data());
262 if (!isMagnetometerInitialized)
263 policy.m_magnetometerRule.allowedList.add(document.securityOrigin().data());
264#endif
265#if ENABLE(WEBXR)
266 if (!isXRSpatialTrackingInitialized)
267 policy.m_xrSpatialTrackingRule.allowedList.add(document.securityOrigin().data());
268#endif
269
270 // https://w3c.github.io/webappsec-feature-policy/#process-feature-policy-attributes
271 // 9.5 Process Feature Policy Attributes
272 // 3.1 If element’s allowfullscreen attribute is specified, and container policy does
273 // not contain an allowlist for fullscreen,
274 if (!isFullscreenInitialized) {
275 if (iframe.hasAttribute(allowfullscreenAttr) || iframe.hasAttribute(webkitallowfullscreenAttr)) {
276 // 3.1.1 Construct a new declaration for fullscreen, whose allowlist is the special value *.
277 policy.m_fullscreenRule.type = FeaturePolicy::AllowRule::Type::All;
278 } else {
279 // https://fullscreen.spec.whatwg.org/#feature-policy-integration
280 // The default allowlist is 'self'.
281 policy.m_fullscreenRule.allowedList.add(document.securityOrigin().data());
282 }
283 }
284
285 if (!isSyncXHRInitialized)
286 policy.m_syncXHRRule.type = AllowRule::Type::All;
287
288 return policy;
289}

291bool FeaturePolicy::allows(Type type, const SecurityOriginData& origin) const
292{
293 switch (type) {
294 case Type::Camera:
295 return isAllowedByFeaturePolicy(m_cameraRule, origin);
296 case Type::Microphone:
297 return isAllowedByFeaturePolicy(m_microphoneRule, origin);
298 case Type::SpeakerSelection:
299 return isAllowedByFeaturePolicy(m_speakerSelectionRule, origin);
300 case Type::DisplayCapture:
301 return isAllowedByFeaturePolicy(m_displayCaptureRule, origin);
302 case Type::Geolocation:
303 return isAllowedByFeaturePolicy(m_geolocationRule, origin);
304 case Type::SyncXHR:
305 return isAllowedByFeaturePolicy(m_syncXHRRule, origin);
306 case Type::Fullscreen:
307 return isAllowedByFeaturePolicy(m_fullscreenRule, origin);
308#if ENABLE(DEVICE_ORIENTATION)
309 case Type::Gyroscope:
310 return isAllowedByFeaturePolicy(m_gyroscopeRule, origin);
311 case Type::Accelerometer:
312 return isAllowedByFeaturePolicy(m_accelerometerRule, origin);
313 case Type::Magnetometer:
314 return isAllowedByFeaturePolicy(m_magnetometerRule, origin);
315#endif
316#if ENABLE(WEBXR)
317 case Type::XRSpatialTracking:
318 return isAllowedByFeaturePolicy(m_xrSpatialTrackingRule, origin);
319#endif
320 }
321 ASSERT_NOT_REACHED();
322 return false;
323}
