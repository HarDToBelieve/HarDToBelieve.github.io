26#pragma once

28#include "SecurityOriginData.h"
29#include <wtf/HashSet.h>
30#include <wtf/text/StringHash.h>

32namespace WebCore {
33
34class Document;
35class HTMLIFrameElement;
36
37class FeaturePolicy {
38public:
39 static FeaturePolicy parse(Document&, const HTMLIFrameElement&, StringView);
40
41 enum class Type {
42 Camera,
43 Microphone,
44 SpeakerSelection,
45 DisplayCapture,
46 Geolocation,
47 SyncXHR,
48 Fullscreen,
49#if ENABLE(DEVICE_ORIENTATION)
50 Gyroscope,
51 Accelerometer,
52 Magnetometer,
53#endif
54#if ENABLE(WEBXR)
55 XRSpatialTracking,
56#endif
57 };
58 bool allows(Type, const SecurityOriginData&) const;
59
60 struct AllowRule {
61 enum class Type { All, None, List };
62 Type type { Type::List };
63 HashSet<SecurityOriginData> allowedList;
64 };
65
66private:
67 AllowRule m_cameraRule;
68 AllowRule m_microphoneRule;
69 AllowRule m_speakerSelectionRule;
70 AllowRule m_displayCaptureRule;
71 AllowRule m_geolocationRule;
72 AllowRule m_syncXHRRule;
73 AllowRule m_fullscreenRule;
74#if ENABLE(DEVICE_ORIENTATION)
75 AllowRule m_gyroscopeRule;
76 AllowRule m_accelerometerRule;
77 AllowRule m_magnetometerRule;
78#endif
79#if ENABLE(WEBXR)
80 AllowRule m_xrSpatialTrackingRule;
81#endif
82};

84enum class LogFeaturePolicyFailure { No, Yes };
85extern bool isFeaturePolicyAllowedByDocumentAndAllOwners(FeaturePolicy::Type, const Document&, LogFeaturePolicyFailure = LogFeaturePolicyFailure::Yes);
