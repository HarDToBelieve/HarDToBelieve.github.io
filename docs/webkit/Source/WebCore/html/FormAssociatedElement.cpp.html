71void FormAssociatedElement::insertedIntoAncestor(Node::InsertionType insertionType, ContainerNode&)
72{
73 HTMLElement& element = asHTMLElement();
74 if (m_formSetByParser) {
75 // The form could have been removed by a script during parsing.
76 if (m_formSetByParser->isConnected())
77 setForm(m_formSetByParser.get());
78 m_formSetByParser = nullptr;
79 }
80
81 if (m_form && element.rootElement() != m_form->rootElement())
82 setForm(nullptr);
83
84 if (!insertionType.connectedToDocument)
85 return;
86
87 if (element.hasAttributeWithoutSynchronization(formAttr))
88 resetFormAttributeTargetObserver();
89}
90
91void FormAssociatedElement::removedFromAncestor(Node::RemovalType, ContainerNode&)
92{
93 m_formAttributeTargetObserver = nullptr;
94
95 // If the form and element are both in the same tree, preserve the connection to the form.
96 // Otherwise, null out our form and remove ourselves from the form's list of elements.
97 // Do not rely on rootNode() because our IsInTreeScope is outdated.
98 if (m_form && &asHTMLElement().traverseToRootNode() != &m_form->traverseToRootNode())
99 setForm(nullptr);
100}

102HTMLFormElement* FormAssociatedElement::findAssociatedForm(const HTMLElement* element, HTMLFormElement* currentAssociatedForm)
103{
104 const AtomString& formId(element->attributeWithoutSynchronization(formAttr));
105 if (!formId.isNull() && element->isConnected()) {
106 // The HTML5 spec says that the element should be associated with
107 // the first element in the document to have an ID that equal to
108 // the value of form attribute, so we put the result of
109 // treeScope().getElementById() over the given element.
110 RefPtr<Element> newFormCandidate = element->treeScope().getElementById(formId);
111 if (!is<HTMLFormElement>(newFormCandidate))
112 return nullptr;
113 if (&element->traverseToRootNode() == &element->treeScope().rootNode()) {
114 ASSERT(&element->traverseToRootNode() == &newFormCandidate->traverseToRootNode());
115 return downcast<HTMLFormElement>(newFormCandidate.get());
116 }
117 }
118
119 if (!currentAssociatedForm)
120 return HTMLFormElement::findClosestFormAncestor(*element);
121
122 return currentAssociatedForm;
123}

130void FormAssociatedElement::formOwnerRemovedFromTree(const Node& formRoot)
131{
132 ASSERT(m_form);
133 // Can't use RefPtr here beacuse this function might be called inside ~ShadowRoot via addChildNodesToDeletionQueue. See webkit.org/b/189493.
134 Node* rootNode = &asHTMLElement();
135 for (auto* ancestor = asHTMLElement().parentNode(); ancestor; ancestor = ancestor->parentNode()) {
136 if (ancestor == m_form) {
137 // Form is our ancestor so we don't need to reset our owner, we also no longer
138 // need an id observer since we are no longer connected.
139 m_formAttributeTargetObserver = nullptr;
140 return;
141 }
142 rootNode = ancestor;
143 }
144
145 // We are no longer in the same tree as our form owner so clear our owner.
146 if (rootNode != &formRoot)
147 setForm(nullptr);
148}

150void FormAssociatedElement::setForm(HTMLFormElement* newForm)
151{
152 if (m_form == newForm)
153 return;
154 willChangeForm();
155 if (m_form)
156 m_form->removeFormElement(this);
157 m_form = makeWeakPtr(newForm);
158 if (newForm)
159 newForm->registerFormElement(this);
160 didChangeForm();
161}

181void FormAssociatedElement::resetFormOwner()
182{
183 RefPtr<HTMLFormElement> originalForm = m_form.get();
184 setForm(findAssociatedForm(&asHTMLElement(), originalForm.get()));
185 HTMLElement& element = asHTMLElement();
186 auto* newForm = m_form.get();
187 if (newForm && newForm != originalForm && newForm->isConnected())
188 element.document().didAssociateFormControl(element);
189}
190
191void FormAssociatedElement::formAttributeChanged()
192{
193 HTMLElement& element = asHTMLElement();
194 if (!element.hasAttributeWithoutSynchronization(formAttr)) {
195 // The form attribute removed. We need to reset form owner here.
196 RefPtr<HTMLFormElement> originalForm = m_form.get();
197 // FIXME: Why does this not pass originalForm to findClosestFormAncestor?
198 setForm(HTMLFormElement::findClosestFormAncestor(element));
199 auto* newForm = m_form.get();
200 if (newForm && newForm != originalForm && newForm->isConnected())
201 element.document().didAssociateFormControl(element);
202 m_formAttributeTargetObserver = nullptr;
203 } else {
204 resetFormOwner();
205 if (element.isConnected())
206 resetFormAttributeTargetObserver();
207 }
208}

255bool FormAssociatedElement::isValid() const
256{
257 bool someError = typeMismatch() || stepMismatch() || rangeUnderflow() || rangeOverflow()
258 || tooShort() || tooLong() || patternMismatch() || valueMissing() || hasBadInput() || customError();
259 return !someError;
260}

282void FormAssociatedElement::resetFormAttributeTargetObserver()
283{
284 ASSERT_WITH_SECURITY_IMPLICATION(asHTMLElement().isConnected());
285 m_formAttributeTargetObserver = makeUnique<FormAttributeTargetObserver>(asHTMLElement().attributeWithoutSynchronization(formAttr), *this);
286}
