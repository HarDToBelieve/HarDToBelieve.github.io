32template <CollectionTraversalType traversalType>
33class GenericCachedHTMLCollection final : public CachedHTMLCollection<GenericCachedHTMLCollection<traversalType>, traversalType> {
34 WTF_MAKE_ISO_ALLOCATED(GenericCachedHTMLCollection);
35 static_assert(traversalType != CollectionTraversalType::CustomForwardOnly, "CustomForwardOnly should use non GenericCachedHTMLCollection.");
36public:
37 static Ref<GenericCachedHTMLCollection> create(ContainerNode& base, CollectionType collectionType)
38 {
39 return adoptRef(*new GenericCachedHTMLCollection(base, collectionType));
40 }
41
42 bool elementMatches(Element&) const;
43
44private:
45 GenericCachedHTMLCollection(ContainerNode& base, CollectionType collectionType)
46 : CachedHTMLCollection<GenericCachedHTMLCollection<traversalType>, traversalType>(base, collectionType)
47 { }
48};
