49// https://html.spec.whatwg.org/multipage/infrastructure.html#dom-htmlallcollection-item
50std::optional<Variant<RefPtr<HTMLCollection>, RefPtr<Element>>> HTMLAllCollection::namedOrIndexedItemOrItems(const AtomString& nameOrIndex) const
51{
52 if (nameOrIndex.isNull())
53 return std::nullopt;
54
55 if (auto index = JSC::parseIndex(*nameOrIndex.impl()))
56 return Variant<RefPtr<HTMLCollection>, RefPtr<Element>> { RefPtr<Element> { item(index.value()) } };
57
58 return namedItemOrItems(nameOrIndex);
59}
60
61// https://html.spec.whatwg.org/multipage/infrastructure.html#concept-get-all-named
62std::optional<Variant<RefPtr<HTMLCollection>, RefPtr<Element>>> HTMLAllCollection::namedItemOrItems(const AtomString& name) const
63{
64 auto namedItems = this->namedItems(name);
65
66 if (namedItems.isEmpty())
67 return std::nullopt;
68 if (namedItems.size() == 1)
69 return Variant<RefPtr<HTMLCollection>, RefPtr<Element>> { RefPtr<Element> { WTFMove(namedItems[0]) } };
70
71 return Variant<RefPtr<HTMLCollection>, RefPtr<Element>> { RefPtr<HTMLCollection> { downcast<Document>(ownerNode()).allFilteredByName(name) } };
72}
73
74HTMLAllNamedSubCollection::~HTMLAllNamedSubCollection()
75{
76 ownerNode().nodeLists()->removeCachedCollection(this, m_name);
77}
78
79bool HTMLAllNamedSubCollection::elementMatches(Element& element) const
80{
81 const auto& id = element.getIdAttribute();
82 if (id == m_name)
83 return true;
84
85 if (!nameShouldBeVisibleInDocumentAll(element))
86 return false;
87
88 const auto& name = element.getNameAttribute();
89 if (name == m_name)
90 return true;
91
92 return false;
93}
