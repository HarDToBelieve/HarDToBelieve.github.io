114static bool hasNonEmptyBox(RenderBoxModelObject* renderer)
115{
116 if (!renderer)
117 return false;
118
119 // Before calling absoluteRects, check for the common case where borderBoundingBox
120 // is non-empty, since this is a faster check and almost always returns true.
121 // FIXME: Why do we need to call absoluteRects at all?
122 if (!renderer->borderBoundingBox().isEmpty())
123 return true;
124
125 // FIXME: Since all we are checking is whether the rects are empty, could we just
126 // pass in 0,0 for the layout point instead of calling localToAbsolute?
127 Vector<IntRect> rects;
128 renderer->absoluteRects(rects, flooredLayoutPoint(renderer->localToAbsolute()));
129 for (auto& rect : rects) {
130 if (!rect.isEmpty())
131 return true;
132 }
133
134 return false;
135}

137bool HTMLAnchorElement::isKeyboardFocusable(KeyboardEvent* event) const
138{
139 if (!isLink())
140 return HTMLElement::isKeyboardFocusable(event);
141
142 if (!isFocusable())
143 return false;
144
145 if (!document().frame())
146 return false;
147
148 if (!document().frame()->eventHandler().tabsToLinks(event))
149 return false;
150
151 if (!renderer() && ancestorsOfType<HTMLCanvasElement>(*this).first())
152 return true;
153
154 return hasNonEmptyBox(renderBoxModelObject());
155}

157static void appendServerMapMousePosition(StringBuilder& url, Event& event)
158{
159 if (!is<MouseEvent>(event))
160 return;
161 auto& mouseEvent = downcast<MouseEvent>(event);
162
163 if (!is<HTMLImageElement>(mouseEvent.target()))
164 return;
165
166 auto& imageElement = downcast<HTMLImageElement>(*mouseEvent.target());
167 if (!imageElement.isServerMap())
168 return;
169
170 auto* renderer = imageElement.renderer();
171 if (!is<RenderImage>(renderer))
172 return;
173
174 // FIXME: This should probably pass UseTransforms in the MapCoordinatesFlags.
175 auto absolutePosition = downcast<RenderImage>(*renderer).absoluteToLocal(FloatPoint(mouseEvent.pageX(), mouseEvent.pageY()));
176 url.append('?', std::lround(absolutePosition.x()), ',', std::lround(absolutePosition.y()));
177}

179void HTMLAnchorElement::defaultEventHandler(Event& event)
180{
181 if (isLink()) {
182 if (focused() && isEnterKeyKeydownEvent(event) && treatLinkAsLiveForEventType(NonMouseEvent)) {
183 event.setDefaultHandled();
184 dispatchSimulatedClick(&event);
185 return;
186 }
187
188 if (MouseEvent::canTriggerActivationBehavior(event) && treatLinkAsLiveForEventType(eventType(event))) {
189 handleClick(event);
190 return;
191 }
192
193 if (hasEditableStyle()) {
194 // This keeps track of the editable block that the selection was in (if it was in one) just before the link was clicked
195 // for the LiveWhenNotFocused editable link behavior
196 if (event.type() == eventNames().mousedownEvent && is<MouseEvent>(event) && downcast<MouseEvent>(event).button() != RightButton && document().frame()) {
197 setRootEditableElementForSelectionOnMouseDown(document().frame()->selection().selection().rootEditableElement());
198 m_wasShiftKeyDownOnMouseDown = downcast<MouseEvent>(event).shiftKey();
199 } else if (event.type() == eventNames().mouseoverEvent) {
200 // These are cleared on mouseover and not mouseout because their values are needed for drag events,
201 // but drag events happen after mouse out events.
202 clearRootEditableElementForSelectionOnMouseDown();
203 m_wasShiftKeyDownOnMouseDown = false;
204 }
205 }
206 }
207
208 HTMLElement::defaultEventHandler(event);
209}

211void HTMLAnchorElement::setActive(bool down, bool pause, Style::InvalidationScope invalidationScope)
212{
213 if (hasEditableStyle()) {
214 switch (document().settings().editableLinkBehavior()) {
215 case EditableLinkBehavior::Default:
216 case EditableLinkBehavior::AlwaysLive:
217 break;
218
219 // Don't set the link to be active if the current selection is in the same editable block as this link.
220 case EditableLinkBehavior::LiveWhenNotFocused:
221 if (down && document().frame() && document().frame()->selection().selection().rootEditableElement() == rootEditableElement())
222 return;
223 break;
224
225 case EditableLinkBehavior::NeverLive:
226 case EditableLinkBehavior::OnlyLiveWithShiftKey:
227 return;
228
229 }
230 }
231
232 HTMLElement::setActive(down, pause, invalidationScope);
233}

235void HTMLAnchorElement::parseAttribute(const QualifiedName& name, const AtomString& value)
236{
237 if (name == hrefAttr) {
238 bool wasLink = isLink();
239 setIsLink(!value.isNull() && !shouldProhibitLinks(this));
240 if (wasLink != isLink())
241 invalidateStyleForSubtree();
242 if (isLink()) {
243 String parsedURL = stripLeadingAndTrailingHTMLSpaces(value);
244 if (document().isDNSPrefetchEnabled() && document().frame()) {
245 if (protocolIsInHTTPFamily(parsedURL) || parsedURL.startsWith("//"))
246 document().frame()->loader().client().prefetchDNS(document().completeURL(parsedURL).host().toString());
247 }
248 }
249 } else if (name == nameAttr || name == titleAttr) {
250 // Do nothing.
251 } else if (name == relAttr) {
252 // Update HTMLAnchorElement::relList() if more rel attributes values are supported.
253 static MainThreadNeverDestroyed<const AtomString> noReferrer("noreferrer", AtomString::ConstructFromLiteral);
254 static MainThreadNeverDestroyed<const AtomString> noOpener("noopener", AtomString::ConstructFromLiteral);
255 static MainThreadNeverDestroyed<const AtomString> opener("opener", AtomString::ConstructFromLiteral);
256 const bool shouldFoldCase = true;
257 SpaceSplitString relValue(value, shouldFoldCase);
258 if (relValue.contains(noReferrer))
259 m_linkRelations.add(Relation::NoReferrer);
260 if (relValue.contains(noOpener))
261 m_linkRelations.add(Relation::NoOpener);
262 if (relValue.contains(opener))
263 m_linkRelations.add(Relation::Opener);
264 if (m_relList)
265 m_relList->associatedAttributeValueChanged(value);
266 }
267 else
268 HTMLElement::parseAttribute(name, value);
269}

362void HTMLAnchorElement::sendPings(const URL& destinationURL)
363{
364 if (!document().frame())
365 return;
366
367 if (!hasAttributeWithoutSynchronization(pingAttr) || !document().settings().hyperlinkAuditingEnabled())
368 return;
369
370 SpaceSplitString pingURLs(attributeWithoutSynchronization(pingAttr), false);
371 for (unsigned i = 0; i < pingURLs.size(); i++)
372 PingLoader::sendPing(*document().frame(), document().completeURL(pingURLs[i]), destinationURL);
373}

398std::optional<PrivateClickMeasurement> HTMLAnchorElement::parsePrivateClickMeasurement() const
399{
400 using SourceID = PrivateClickMeasurement::SourceID;
401 using SourceSite = PrivateClickMeasurement::SourceSite;
402 using AttributionDestinationSite = PrivateClickMeasurement::AttributionDestinationSite;
403
404 auto* page = document().page();
405 if (!page || page->sessionID().isEphemeral()
406 || !document().settings().privateClickMeasurementEnabled()
407 || !UserGestureIndicator::processingUserGesture())
408 return std::nullopt;
409
410 auto hasAttributionSourceIDAttr = hasAttributeWithoutSynchronization(attributionsourceidAttr);
411 auto hasAttributionDestinationAttr = hasAttributeWithoutSynchronization(attributiondestinationAttr);
412 if (!hasAttributionSourceIDAttr && !hasAttributionDestinationAttr)
413 return std::nullopt;
414
415 auto attributionSourceIDAttr = attributeWithoutSynchronization(attributionsourceidAttr);
416 auto attributionDestinationAttr = attributeWithoutSynchronization(attributiondestinationAttr);
417
418 if (!hasAttributionSourceIDAttr || !hasAttributionDestinationAttr || attributionSourceIDAttr.isEmpty() || attributionDestinationAttr.isEmpty()) {
419 document().addConsoleMessage(MessageSource::Other, MessageLevel::Warning, "Both attributionsourceid and attributiondestination need to be set for Private Click Measurement to work."_s);
420 return std::nullopt;
421 }
422
423 RefPtr<Frame> frame = document().frame();
424 if (!frame || !frame->isMainFrame()) {
425 document().addConsoleMessage(MessageSource::Other, MessageLevel::Warning, "Private Click Measurement is only supported in the main frame."_s);
426 return std::nullopt;
427 }
428
429 auto attributionSourceID = parseHTMLNonNegativeInteger(attributionSourceIDAttr);
430 if (!attributionSourceID) {
431 document().addConsoleMessage(MessageSource::Other, MessageLevel::Warning, "attributionsourceid is not a non-negative integer which is required for Private Click Measurement."_s);
432 return std::nullopt;
433 }
434
435 if (attributionSourceID.value() > PrivateClickMeasurement::SourceID::MaxEntropy) {
436 document().addConsoleMessage(MessageSource::Other, MessageLevel::Warning, makeString("attributionsourceid must have a non-negative value less than or equal to ", PrivateClickMeasurement::SourceID::MaxEntropy, " for Private Click Measurement."));
437 return std::nullopt;
438 }
439
440 URL destinationURL { URL(), attributionDestinationAttr };
441 if (!destinationURL.isValid() || !destinationURL.protocolIsInHTTPFamily()) {
442 document().addConsoleMessage(MessageSource::Other, MessageLevel::Warning, "attributiondestination could not be converted to a valid HTTP-family URL."_s);
443 return std::nullopt;
444 }
445
446 RegistrableDomain documentRegistrableDomain { document().url() };
447 if (documentRegistrableDomain.matches(destinationURL)) {
448 document().addConsoleMessage(MessageSource::Other, MessageLevel::Warning, "attributiondestination can not be the same site as the current website."_s);
449 return std::nullopt;
450 }
451
452 auto privateClickMeasurement = PrivateClickMeasurement { SourceID(attributionSourceID.value()), SourceSite(documentRegistrableDomain), AttributionDestinationSite(destinationURL) };
453
454 auto attributionSourceNonceAttr = attributeWithoutSynchronization(attributionsourcenonceAttr);
455 if (!attributionSourceNonceAttr.isEmpty()) {
456 auto ephemeralNonce = PrivateClickMeasurement::EphemeralSourceNonce { attributionSourceNonceAttr };
457 if (!ephemeralNonce.isValid()) {
458 document().addConsoleMessage(MessageSource::Other, MessageLevel::Warning, "attributionsourcenonce was not valid."_s);
459 return std::nullopt;
460 }
461 privateClickMeasurement.setEphemeralSourceNonce(WTFMove(ephemeralNonce));
462 }
463
464 return privateClickMeasurement;
465}

467void HTMLAnchorElement::handleClick(Event& event)
468{
469 event.setDefaultHandled();
470
471 RefPtr<Frame> frame = document().frame();
472 if (!frame)
473 return;
474
475 if (!hasTagName(aTag) && !isConnected())
476 return;
477
478 StringBuilder url;
479 url.append(stripLeadingAndTrailingHTMLSpaces(attributeWithoutSynchronization(hrefAttr)));
480 appendServerMapMousePosition(url, event);
481 URL completedURL = document().completeURL(url.toString());
482
483#if ENABLE(DATA_DETECTION) && PLATFORM(IOS_FAMILY)
484 if (DataDetection::isDataDetectorLink(*this) && DataDetection::canPresentDataDetectorsUIForElement(*this)) {
485 if (auto* page = document().page()) {
486 if (page->chrome().client().showDataDetectorsUIForElement(*this, event))
487 return;
488 }
489 }
490#endif
491
492 String downloadAttribute;
493#if ENABLE(DOWNLOAD_ATTRIBUTE)
494 if (document().settings().downloadAttributeEnabled()) {
495 // Ignore the download attribute completely if the href URL is cross origin.
496 bool isSameOrigin = completedURL.protocolIsData() || document().securityOrigin().canRequest(completedURL);
497 if (isSameOrigin)
498 downloadAttribute = ResourceResponse::sanitizeSuggestedFilename(attributeWithoutSynchronization(downloadAttr));
499 else if (hasAttributeWithoutSynchronization(downloadAttr))
500 document().addConsoleMessage(MessageSource::Security, MessageLevel::Warning, "The download attribute on anchor was ignored because its href URL has a different security origin.");
501 }
502#endif
503
504 SystemPreviewInfo systemPreviewInfo;
505#if USE(SYSTEM_PREVIEW)
506 systemPreviewInfo.isPreview = isSystemPreviewLink() && document().settings().systemPreviewEnabled();
507
508 if (systemPreviewInfo.isPreview) {
509 systemPreviewInfo.element.elementIdentifier = document().identifierForElement(*this);
510 systemPreviewInfo.element.documentIdentifier = document().identifier();
511 systemPreviewInfo.element.webPageIdentifier = document().frame()->loader().pageID().value_or(PageIdentifier { });
512 if (auto* child = firstElementChild())
513 systemPreviewInfo.previewRect = child->boundsInRootViewSpace();
514 }
515#endif
516
517 auto referrerPolicy = hasRel(Relation::NoReferrer) ? ReferrerPolicy::NoReferrer : this->referrerPolicy();
518
519 auto effectiveTarget = this->effectiveTarget();
520 std::optional<NewFrameOpenerPolicy> newFrameOpenerPolicy;
521 if (hasRel(Relation::Opener))
522 newFrameOpenerPolicy = NewFrameOpenerPolicy::Allow;
523 else if (hasRel(Relation::NoOpener) || (document().settings().blankAnchorTargetImpliesNoOpenerEnabled() && equalIgnoringASCIICase(effectiveTarget, "_blank")))
524 newFrameOpenerPolicy = NewFrameOpenerPolicy::Suppress;
525
526 auto privateClickMeasurement = parsePrivateClickMeasurement();
527 // A matching triggering event needs to happen before an attribution report can be sent.
528 // Thus, URLs should be empty for now.
529 ASSERT(!privateClickMeasurement || (privateClickMeasurement->attributionReportSourceURL().isNull() && privateClickMeasurement->attributionReportAttributeOnURL().isNull()));
530
531 frame->loader().changeLocation(completedURL, effectiveTarget, &event, referrerPolicy, document().shouldOpenExternalURLsPolicyToPropagate(), newFrameOpenerPolicy, downloadAttribute, systemPreviewInfo, WTFMove(privateClickMeasurement));
532
533 sendPings(completedURL);
534}

552bool HTMLAnchorElement::treatLinkAsLiveForEventType(EventType eventType) const
553{
554 if (!hasEditableStyle())
555 return true;
556
557 switch (document().settings().editableLinkBehavior()) {
558 case EditableLinkBehavior::Default:
559 case EditableLinkBehavior::AlwaysLive:
560 return true;
561
562 case EditableLinkBehavior::NeverLive:
563 return false;
564
565 // If the selection prior to clicking on this link resided in the same editable block as this link,
566 // and the shift key isn't pressed, we don't want to follow the link.
567 case EditableLinkBehavior::LiveWhenNotFocused:
568 return eventType == MouseEventWithShiftKey || (eventType == MouseEventWithoutShiftKey && rootEditableElementForSelectionOnMouseDown() != rootEditableElement());
569
570 case EditableLinkBehavior::OnlyLiveWithShiftKey:
571 return eventType == MouseEventWithShiftKey;
572 }
573
574 ASSERT_NOT_REACHED();
575 return false;
576}
