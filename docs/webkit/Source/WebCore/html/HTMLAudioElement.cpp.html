41inline HTMLAudioElement::HTMLAudioElement(const QualifiedName& tagName, Document& document, bool createdByParser)
42 : HTMLMediaElement(tagName, document, createdByParser)
43{
44 ASSERT(hasTagName(audioTag));
45}
46
47Ref<HTMLAudioElement> HTMLAudioElement::create(const QualifiedName& tagName, Document& document, bool createdByParser)
48{
49 auto element = adoptRef(*new HTMLAudioElement(tagName, document, createdByParser));
50 element->suspendIfNeeded();
51 return element;
52}
53
54Ref<HTMLAudioElement> HTMLAudioElement::createForLegacyFactoryFunction(Document& document, const AtomString& src)
55{
56 auto element = create(audioTag, document, false);
57 element->setAttributeWithoutSynchronization(preloadAttr, "auto");
58 element->setAttributeWithoutSynchronization(srcAttr, src);
59 return element;
60}
