54bool HTMLBRElement::hasPresentationalHintsForAttribute(const QualifiedName& name) const
55{
56 if (name == clearAttr)
57 return true;
58 return HTMLElement::hasPresentationalHintsForAttribute(name);
59}
60
61void HTMLBRElement::collectPresentationalHintsForAttribute(const QualifiedName& name, const AtomString& value, MutableStyleProperties& style)
62{
63 if (name == clearAttr) {
64 // If the string is empty, then don't add the clear property.
65 // <br clear> and <br clear=""> are just treated like <br> by Gecko, Mac IE, etc. -dwh
66 if (!value.isEmpty()) {
67 if (equalLettersIgnoringASCIICase(value, "all"))
68 addPropertyToPresentationalHintStyle(style, CSSPropertyClear, CSSValueBoth);
69 else
70 addPropertyToPresentationalHintStyle(style, CSSPropertyClear, value);
71 }
72 } else
73 HTMLElement::collectPresentationalHintsForAttribute(name, value, style);
74}

76RenderPtr<RenderElement> HTMLBRElement::createElementRenderer(RenderStyle&& style, const RenderTreePosition&)
77{
78 if (style.hasContent())
79 return RenderElement::createFor(*this, WTFMove(style));
80
81 return createRenderer<RenderLineBreak>(*this, WTFMove(style));
82}
