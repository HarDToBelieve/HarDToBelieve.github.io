63RenderPtr<RenderElement> HTMLButtonElement::createElementRenderer(RenderStyle&& style, const RenderTreePosition& position)
64{
65 // https://html.spec.whatwg.org/multipage/rendering.html#button-layout
66 DisplayType display = style.display();
67 if (display == DisplayType::InlineGrid || display == DisplayType::Grid || display == DisplayType::InlineFlex || display == DisplayType::Flex)
68 return HTMLFormControlElement::createElementRenderer(WTFMove(style), position);
69 return createRenderer<RenderButton>(*this, WTFMove(style));
70}

77const AtomString& HTMLButtonElement::formControlType() const
78{
79 switch (m_type) {
80 case SUBMIT: {
81 static MainThreadNeverDestroyed<const AtomString> submit("submit", AtomString::ConstructFromLiteral);
82 return submit;
83 }
84 case BUTTON: {
85 static MainThreadNeverDestroyed<const AtomString> button("button", AtomString::ConstructFromLiteral);
86 return button;
87 }
88 case RESET: {
89 static MainThreadNeverDestroyed<const AtomString> reset("reset", AtomString::ConstructFromLiteral);
90 return reset;
91 }
92 }
93
94 ASSERT_NOT_REACHED();
95 return emptyAtom();
96}

98bool HTMLButtonElement::hasPresentationalHintsForAttribute(const QualifiedName& name) const
99{
100 if (name == alignAttr) {
101 // Don't map 'align' attribute. This matches what Firefox and IE do, but not Opera.
102 // See http://bugs.webkit.org/show_bug.cgi?id=12071
103 return false;
104 }
105
106 return HTMLFormControlElement::hasPresentationalHintsForAttribute(name);
107}

109void HTMLButtonElement::parseAttribute(const QualifiedName& name, const AtomString& value)
110{
111 if (name == typeAttr) {
112 Type oldType = m_type;
113 if (equalLettersIgnoringASCIICase(value, "reset"))
114 m_type = RESET;
115 else if (equalLettersIgnoringASCIICase(value, "button"))
116 m_type = BUTTON;
117 else
118 m_type = SUBMIT;
119 if (oldType != m_type) {
120 updateWillValidateAndValidity();
121 if (form() && (oldType == SUBMIT || m_type == SUBMIT))
122 form()->resetDefaultButton();
123 }
124 } else
125 HTMLFormControlElement::parseAttribute(name, value);
126}

128void HTMLButtonElement::defaultEventHandler(Event& event)
129{
130 if (event.type() == eventNames().DOMActivateEvent && !isDisabledFormControl()) {
131 RefPtr<HTMLFormElement> protectedForm(form());
132
133 if (protectedForm) {
134 // Update layout before processing form actions in case the style changes
135 // the Form or button relationships.
136 document().updateLayoutIgnorePendingStylesheets();
137
138 if (auto currentForm = form()) {
139 if (m_type == SUBMIT) {
140 SetForScope<bool> activatedSubmitState(m_isActivatedSubmit, true);
141 currentForm->submitIfPossible(&event);
142 }
143
144 if (m_type == RESET)
145 currentForm->reset();
146 }
147
148 if (m_type == SUBMIT || m_type == RESET)
149 event.setDefaultHandled();
150 }
151 }
152
153 if (is<KeyboardEvent>(event)) {
154 KeyboardEvent& keyboardEvent = downcast<KeyboardEvent>(event);
155 if (keyboardEvent.type() == eventNames().keydownEvent && keyboardEvent.keyIdentifier() == "U+0020") {
156 setActive(true, true);
157 // No setDefaultHandled() - IE dispatches a keypress in this case.
158 return;
159 }
160 if (keyboardEvent.type() == eventNames().keypressEvent) {
161 switch (keyboardEvent.charCode()) {
162 case '\r':
163 dispatchSimulatedClick(&keyboardEvent);
164 keyboardEvent.setDefaultHandled();
165 return;
166 case ' ':
167 // Prevent scrolling down the page.
168 keyboardEvent.setDefaultHandled();
169 return;
170 }
171 }
172 if (keyboardEvent.type() == eventNames().keyupEvent && keyboardEvent.keyIdentifier() == "U+0020") {
173 if (active())
174 dispatchSimulatedClick(&keyboardEvent);
175 keyboardEvent.setDefaultHandled();
176 return;
177 }
178 }
179
180 HTMLFormControlElement::defaultEventHandler(event);
181}

183bool HTMLButtonElement::willRespondToMouseClickEvents()
184{
185 return !isDisabledFormControl();

188bool HTMLButtonElement::isSuccessfulSubmitButton() const
189{
190 // HTML spec says that buttons must have names to be considered successful.
191 // However, other browsers do not impose this constraint.
192 return m_type == SUBMIT && !isDisabledFormControl();
193}
194
195bool HTMLButtonElement::matchesDefaultPseudoClass() const
196{
197 return isSuccessfulSubmitButton() && form() && form()->defaultButton() == this;
198}

210bool HTMLButtonElement::appendFormData(DOMFormData& formData, bool)
211{
212 if (m_type != SUBMIT || name().isEmpty() || !m_isActivatedSubmit)
213 return false;
214 formData.append(name(), value());
215 return true;
216}

218bool HTMLButtonElement::accessKeyAction(bool sendMouseEvents)
219{
220 focus();
221
222 return dispatchSimulatedClick(0, sendMouseEvents ? SendMouseUpDownEvents : SendNoEvents);
223}
