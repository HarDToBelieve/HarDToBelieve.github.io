118// Firefox limits width/height to 32767 pixels, but slows down dramatically before it
119// reaches that limit. We limit by area instead, giving us larger maximum dimensions,
120// in exchange for a smaller maximum canvas size. The maximum canvas size is in device pixels.
121#if PLATFORM(IOS_FAMILY)
122const unsigned maxCanvasArea = 4096 * 4096;
123#else
124const unsigned maxCanvasArea = 16384 * 16384;

151HTMLCanvasElement::~HTMLCanvasElement()
152{
153 // FIXME: This has to be called here because CSSCanvasValue::CanvasObserverProxy::canvasDestroyed()
154 // downcasts the CanvasBase object to HTMLCanvasElement. That invokes virtual methods, which should be
155 // avoided in destructors, but works as long as it's done before HTMLCanvasElement destructs completely.
156 notifyObserversCanvasDestroyed();
157 document().clearCanvasPreparation(*this);
158
159 m_context = nullptr; // Ensure this goes away before the ImageBuffer.
160 setImageBuffer(nullptr);
161}

239ExceptionOr<std::optional<RenderingContext>> HTMLCanvasElement::getContext(JSC::JSGlobalObject& state, const String& contextId, Vector<JSC::Strong<JSC::Unknown>>&& arguments)
240{
241 if (m_context) {
242 if (m_context->isPlaceholder())
243 return Exception { InvalidStateError };
244
245 if (m_context->is2d()) {
246 if (!is2dType(contextId))
247 return std::optional<RenderingContext> { std::nullopt };
248 return std::optional<RenderingContext> { RefPtr<CanvasRenderingContext2D> { &downcast<CanvasRenderingContext2D>(*m_context) } };
249 }
250
251 if (m_context->isBitmapRenderer()) {
252 if (!isBitmapRendererType(contextId))
253 return std::optional<RenderingContext> { std::nullopt };
254 return std::optional<RenderingContext> { RefPtr<ImageBitmapRenderingContext> { &downcast<ImageBitmapRenderingContext>(*m_context) } };
255 }
256
257#if ENABLE(WEBGL)
258 if (m_context->isWebGL()) {
259 if (!isWebGLType(contextId))
260 return std::optional<RenderingContext> { std::nullopt };
261 auto version = toWebGLVersion(contextId);
262 if ((version == WebGLVersion::WebGL1) != m_context->isWebGL1())
263 return std::optional<RenderingContext> { std::nullopt };
264 if (is<WebGLRenderingContext>(*m_context))
265 return std::optional<RenderingContext> { RefPtr<WebGLRenderingContext> { &downcast<WebGLRenderingContext>(*m_context) } };
266#if ENABLE(WEBGL2)
267 ASSERT(is<WebGL2RenderingContext>(*m_context));
268 return std::optional<RenderingContext> { RefPtr<WebGL2RenderingContext> { &downcast<WebGL2RenderingContext>(*m_context) } };
269#endif
270 }
271#endif
272
273#if ENABLE(WEBGPU)
274 if (m_context->isWebGPU()) {
275 if (!isWebGPUType(contextId))
276 return std::optional<RenderingContext> { std::nullopt };
277 return std::optional<RenderingContext> { RefPtr<GPUCanvasContext> { &downcast<GPUCanvasContext>(*m_context) } };
278 }
279#endif
280
281 ASSERT_NOT_REACHED();
282 return std::optional<RenderingContext> { std::nullopt };
283 }
284
285 if (is2dType(contextId)) {
286 auto scope = DECLARE_THROW_SCOPE(state.vm());
287 auto settings = convert<IDLDictionary<CanvasRenderingContext2DSettings>>(state, !arguments.isEmpty() ? arguments[0].get() : JSC::jsUndefined());
288 RETURN_IF_EXCEPTION(scope, Exception { ExistingExceptionError });
289
290 auto context = createContext2d(contextId, WTFMove(settings));
291 if (!context)
292 return std::optional<RenderingContext> { std::nullopt };
293 return std::optional<RenderingContext> { RefPtr<CanvasRenderingContext2D> { context } };
294 }
295
296 if (isBitmapRendererType(contextId)) {
297 auto scope = DECLARE_THROW_SCOPE(state.vm());
298 auto settings = convert<IDLDictionary<ImageBitmapRenderingContextSettings>>(state, !arguments.isEmpty() ? arguments[0].get() : JSC::jsUndefined());
299 RETURN_IF_EXCEPTION(scope, Exception { ExistingExceptionError });
300
301 auto context = createContextBitmapRenderer(contextId, WTFMove(settings));
302 if (!context)
303 return std::optional<RenderingContext> { std::nullopt };
304 return std::optional<RenderingContext> { RefPtr<ImageBitmapRenderingContext> { context } };
305 }
306
307#if ENABLE(WEBGL)
308 if (isWebGLType(contextId)) {
309 auto scope = DECLARE_THROW_SCOPE(state.vm());
310 auto attributes = convert<IDLDictionary<WebGLContextAttributes>>(state, !arguments.isEmpty() ? arguments[0].get() : JSC::jsUndefined());
311 RETURN_IF_EXCEPTION(scope, Exception { ExistingExceptionError });
312
313 auto context = createContextWebGL(toWebGLVersion(contextId), WTFMove(attributes));
314 if (!context)
315 return std::optional<RenderingContext> { std::nullopt };
316
317 if (is<WebGLRenderingContext>(*context))
318 return std::optional<RenderingContext> { RefPtr<WebGLRenderingContext> { &downcast<WebGLRenderingContext>(*context) } };
319#if ENABLE(WEBGL2)
320 ASSERT(is<WebGL2RenderingContext>(*context));
321 return std::optional<RenderingContext> { RefPtr<WebGL2RenderingContext> { &downcast<WebGL2RenderingContext>(*context) } };
322#endif
323 }
324#endif
325
326#if ENABLE(WEBGPU)
327 if (isWebGPUType(contextId)) {
328 auto context = createContextWebGPU(contextId);
329 if (!context)
330 return std::optional<RenderingContext> { std::nullopt };
331 return std::optional<RenderingContext> { RefPtr<GPUCanvasContext> { context } };
332 }
333#endif
334
335 return std::optional<RenderingContext> { std::nullopt };
336}

364CanvasRenderingContext2D* HTMLCanvasElement::createContext2d(const String& type, CanvasRenderingContext2DSettings&& settings)
365{
366 ASSERT_UNUSED(HTMLCanvasElement::is2dType(type), type);
367 ASSERT(!m_context);
368
369 // Make sure we don't use more pixel memory than the system can support.
370 size_t requestedPixelMemory = 4 * width() * height();
371 if (activePixelMemory() + requestedPixelMemory > maxActivePixelMemory()) {
372 auto message = makeString("Total canvas memory use exceeds the maximum limit (", maxActivePixelMemory() / 1024 / 1024, " MB).");
373 document().addConsoleMessage(MessageSource::JS, MessageLevel::Warning, message);
374 return nullptr;
375 }
376
377 m_context = CanvasRenderingContext2D::create(*this, WTFMove(settings), document().inQuirksMode());
378
379#if USE(IOSURFACE_CANVAS_BACKING_STORE)
380 // Need to make sure a RenderLayer and compositing layer get created for the Canvas.
381 invalidateStyleAndLayerComposition();
382#endif
383
384 return static_cast<CanvasRenderingContext2D*>(m_context.get());
385}

387CanvasRenderingContext2D* HTMLCanvasElement::getContext2d(const String& type, CanvasRenderingContext2DSettings&& settings)
388{
389 ASSERT_UNUSED(HTMLCanvasElement::is2dType(type), type);
390
391 if (m_context && !m_context->is2d())
392 return nullptr;
393
394 if (!m_context)
395 return createContext2d(type, WTFMove(settings));
396 return static_cast<CanvasRenderingContext2D*>(m_context.get());
397}

443WebGLRenderingContextBase* HTMLCanvasElement::createContextWebGL(WebGLVersion type, WebGLContextAttributes&& attrs)
444{
445 ASSERT(!m_context);
446
447 if (!shouldEnableWebGL(document().settings()))
448 return nullptr;
449
450#if ENABLE(WEBXR)
451 // https://immersive-web.github.io/webxr/#xr-compatible
452 if (attrs.xrCompatible) {
453 if (auto* window = document().domWindow())
454 // FIXME: how to make this sync without blocking the main thread?
455 // For reference: https://immersive-web.github.io/webxr/#ref-for-dom-webglcontextattributes-xrcompatible
456 NavigatorWebXR::xr(window->navigator()).ensureImmersiveXRDeviceIsSelected([]() { });
457 }
458#endif
459
460 // TODO(WEBXR): ensure the context is created in a compatible graphics
461 // adapter when there is an active immersive device.
462 m_context = WebGLRenderingContextBase::create(*this, attrs, type);
463 if (m_context) {
464 // This new context needs to be observed by the Document, in order
465 // for it to be correctly preparedForRendering before it is composited.
466 addObserver(document());
467
468 // Need to make sure a RenderLayer and compositing layer get created for the Canvas.
469 invalidateStyleAndLayerComposition();
470#if ENABLE(WEBXR)
471 ASSERT(!attrs.xrCompatible || downcast<WebGLRenderingContextBase>(m_context.get())->isXRCompatible());
472#endif
473 }
474
475 return downcast<WebGLRenderingContextBase>(m_context.get());
476}

478WebGLRenderingContextBase* HTMLCanvasElement::getContextWebGL(WebGLVersion type, WebGLContextAttributes&& attrs)
479{
480 if (!shouldEnableWebGL(document().settings()))
481 return nullptr;
482
483 if (m_context) {
484 if (!m_context->isWebGL())
485 return nullptr;
486
487 if ((type == WebGLVersion::WebGL1) != m_context->isWebGL1())
488 return nullptr;
489 }
490
491 if (!m_context)
492 return createContextWebGL(type, WTFMove(attrs));
493 return &downcast<WebGLRenderingContextBase>(*m_context);
494}

567void HTMLCanvasElement::didDraw(const std::optional<FloatRect>& rect)
568{
569 clearCopiedImage();
570
571 if (!rect) {
572 notifyObserversCanvasChanged(std::nullopt);
573 return;
574 }
575
576 auto dirtyRect = rect.value();
577 if (auto* renderer = renderBox()) {
578 FloatRect destRect;
579 if (is<RenderReplaced>(renderer))
580 destRect = downcast<RenderReplaced>(renderer)->replacedContentRect();
581 else
582 destRect = renderer->contentBoxRect();
583
584 // Inflate dirty rect to cover antialiasing on image buffers.
585 if (drawingContext() && drawingContext()->shouldAntialias())
586 dirtyRect.inflate(1);
587
588 FloatRect r = mapRect(dirtyRect, FloatRect(0, 0, size().width(), size().height()), destRect);
589 r.intersect(destRect);
590
591 if (!r.isEmpty())
592 renderer->repaintRectangle(enclosingIntRect(r));
593 }
594 notifyObserversCanvasChanged(dirtyRect);
595}

597void HTMLCanvasElement::reset()
598{
599 if (m_ignoreReset || isControlledByOffscreen())
600 return;
601
602 bool hadImageBuffer = hasCreatedImageBuffer();
603
604 int w = limitToOnlyHTMLNonNegative(attributeWithoutSynchronization(widthAttr), defaultWidth);
605 int h = limitToOnlyHTMLNonNegative(attributeWithoutSynchronization(heightAttr), defaultHeight);
606
607 resetGraphicsContextState();
608 if (is<CanvasRenderingContext2D>(m_context.get()))
609 downcast<CanvasRenderingContext2D>(*m_context).reset();
610
611 IntSize oldSize = size();
612 IntSize newSize(w, h);
613 // If the size of an existing buffer matches, we can just clear it instead of reallocating.
614 // This optimization is only done for 2D canvases for now.
615 if (m_hasCreatedImageBuffer && oldSize == newSize && m_context && m_context->is2d() && buffer() && m_context->colorSpace() == buffer()->colorSpace() && m_context->pixelFormat() == buffer()->pixelFormat()) {
616 if (!m_didClearImageBuffer)
617 clearImageBuffer();
618 return;
619 }
620
621 setSurfaceSize(newSize);
622
623 if (isGPUBased() && oldSize != size())
624 downcast<GPUBasedCanvasRenderingContext>(*m_context).reshape(width(), height());
625
626 auto renderer = this->renderer();
627 if (is<RenderHTMLCanvas>(renderer)) {
628 auto& canvasRenderer = downcast<RenderHTMLCanvas>(*renderer);
629 if (oldSize != size()) {
630 canvasRenderer.canvasSizeChanged();
631 if (canvasRenderer.hasAcceleratedCompositing())
632 canvasRenderer.contentChanged(CanvasChanged);
633 }
634 if (hadImageBuffer)
635 canvasRenderer.repaint();
636 }
637
638 notifyObserversCanvasResized();
639}

641bool HTMLCanvasElement::paintsIntoCanvasBuffer() const
642{
643 ASSERT(m_context);
644#if USE(IOSURFACE_CANVAS_BACKING_STORE)
645 if (m_context->is2d() || m_context->isBitmapRenderer())
646 return true;
647#endif
648
649 if (!m_context->isAccelerated())
650 return true;
651
652 if (renderBox() && renderBox()->hasAcceleratedCompositing())
653 return false;
654
655 return true;
656}

659void HTMLCanvasElement::paint(GraphicsContext& context, const LayoutRect& r)
660{
661 if (m_context)
662 m_context->clearAccumulatedDirtyRect();
663
664 if (!context.paintingDisabled()) {
665 bool shouldPaint = true;
666
667 if (m_context) {
668 shouldPaint = paintsIntoCanvasBuffer() || document().printing() || m_isSnapshotting;
669 if (shouldPaint) {
670 m_context->prepareForDisplayWithPaint();
671 m_context->paintRenderingResultsToCanvas();
672 }
673 }
674
675 if (shouldPaint) {
676 if (hasCreatedImageBuffer()) {
677 if (ImageBuffer* imageBuffer = buffer())
678 context.drawImageBuffer(*imageBuffer, snappedIntRect(r));
679 }
680 }
681 }
682
683 if (UNLIKELY(m_context && m_context->hasActiveInspectorCanvasCallTracer()))
684 InspectorInstrumentation::didFinishRecordingCanvasFrame(*m_context);
685}

707// https://html.spec.whatwg.org/multipage/canvas.html#a-serialisation-of-the-bitmap-as-a-file
708static std::optional<double> qualityFromJSValue(JSC::JSValue qualityValue)
709{
710 if (!qualityValue.isNumber())
711 return std::nullopt;
712
713 double qualityNumber = qualityValue.asNumber();
714 if (qualityNumber < 0 || qualityNumber > 1)
715 return std::nullopt;
716
717 return qualityNumber;

923void HTMLCanvasElement::createImageBuffer() const
924{
925 ASSERT(!hasCreatedImageBuffer());
926
927 m_hasCreatedImageBuffer = true;
928 m_didClearImageBuffer = true;
929
930 // Perform multiplication as floating point to avoid overflow
931 if (float(width()) * height() > maxCanvasArea) {
932 auto message = makeString("Canvas area exceeds the maximum limit (width * height > ", maxCanvasArea, ").");
933 document().addConsoleMessage(MessageSource::JS, MessageLevel::Warning, message);
934 return;
935 }
936
937 // Make sure we don't use more pixel memory than the system can support.
938 size_t requestedPixelMemory = 4 * width() * height();
939 if (activePixelMemory() + requestedPixelMemory > maxActivePixelMemory()) {
940 auto message = makeString("Total canvas memory use exceeds the maximum limit (", maxActivePixelMemory() / 1024 / 1024, " MB).");
941 document().addConsoleMessage(MessageSource::JS, MessageLevel::Warning, message);
942 return;
943 }
944
945 if (!width() || !height())
946 return;
947
948 auto hostWindow = (document().view() && document().view()->root()) ? document().view()->root()->hostWindow() : nullptr;
949
950 auto renderingMode = shouldAccelerate(size()) ? RenderingMode::Accelerated : RenderingMode::Unaccelerated;
951 // FIXME: Add a new setting for DisplayList drawing on canvas.
952 auto useDisplayList = m_usesDisplayListDrawing.value_or(document().settings().displayListDrawingEnabled()) ? ShouldUseDisplayList::Yes : ShouldUseDisplayList::No;
953
954 auto [colorSpace, pixelFormat] = [&] {
955 if (m_context)
956 return std::pair { m_context->colorSpace(), m_context->pixelFormat() };
957 return std::pair { DestinationColorSpace::SRGB(), PixelFormat::BGRA8 };
958 }();
959
960 setImageBuffer(ImageBuffer::create(size(), renderingMode, useDisplayList, RenderingPurpose::Canvas, 1, colorSpace, pixelFormat, hostWindow));
961
962 if (buffer() && buffer()->drawingContext())
963 buffer()->drawingContext()->setTracksDisplayListReplay(m_tracksDisplayListReplay);
964
965#if USE(IOSURFACE_CANVAS_BACKING_STORE)
966 if (m_context && m_context->is2d()) {
967 // Recalculate compositing requirements if acceleration state changed.
968 const_cast<HTMLCanvasElement*>(this)->invalidateStyleAndLayerComposition();
969 }
970#endif
971
972 if (m_context && buffer() && buffer()->prefersPreparationForDisplay())
973 const_cast<HTMLCanvasElement*>(this)->addObserver(document());
974}
