204namespace WTF {
205template<typename ArgType> class TypeCastTraits<const WebCore::HTMLCanvasElement, ArgType, false /* isBaseType */> {
206public:
207 static bool isOfType(ArgType& node) { return checkTagName(node); }
208private:
209 static bool checkTagName(const WebCore::CanvasBase& base) { return base.isHTMLCanvasElement(); }
210 static bool checkTagName(const WebCore::HTMLElement& element) { return element.hasTagName(WebCore::HTMLNames::canvasTag); }
211 static bool checkTagName(const WebCore::Node& node) { return node.hasTagName(WebCore::HTMLNames::canvasTag); }
212 static bool checkTagName(const WebCore::EventTarget& target) { return is<WebCore::Node>(target) && checkTagName(downcast<WebCore::Node>(target)); }
213};
