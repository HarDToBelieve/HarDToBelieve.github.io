36inline auto HTMLCollection::rootTypeFromCollectionType(CollectionType type) -> RootType
37{
38 switch (type) {
39 case DocImages:
40 case DocApplets:
41 case DocEmbeds:
42 case DocForms:
43 case DocLinks:
44 case DocAnchors:
45 case DocScripts:
46 case DocAll:
47 case WindowNamedItems:
48 case DocumentNamedItems:
49 case DocumentAllNamedItems:
50 case FormControls:
51 return HTMLCollection::IsRootedAtDocument;
52 case AllDescendants:
53 case ByClass:
54 case ByTag:
55 case ByHTMLTag:
56 case FieldSetElements:
57 case NodeChildren:
58 case TableTBodies:
59 case TSectionRows:
60 case TableRows:
61 case TRCells:
62 case SelectOptions:
63 case SelectedOptions:
64 case DataListOptions:
65 case MapAreas:
66 return HTMLCollection::IsRootedAtNode;
67 }
68 ASSERT_NOT_REACHED();
69 return HTMLCollection::IsRootedAtNode;
70}

72static NodeListInvalidationType invalidationTypeExcludingIdAndNameAttributes(CollectionType type)
73{
74 switch (type) {
75 case ByTag:
76 case ByHTMLTag:
77 case AllDescendants:
78 case DocImages:
79 case DocEmbeds:
80 case DocForms:
81 case DocScripts:
82 case DocAll:
83 case NodeChildren:
84 case TableTBodies:
85 case TSectionRows:
86 case TableRows:
87 case TRCells:
88 case SelectOptions:
89 case MapAreas:
90 return DoNotInvalidateOnAttributeChanges;
91 case DocApplets:
92 case SelectedOptions:
93 case DataListOptions:
94 // FIXME: We can do better some day.
95 return InvalidateOnAnyAttrChange;
96 case ByClass:
97 return InvalidateOnClassAttrChange;
98 case DocAnchors:
99 return InvalidateOnNameAttrChange;
100 case DocLinks:
101 return InvalidateOnHRefAttrChange;
102 case WindowNamedItems:
103 case DocumentNamedItems:
104 case DocumentAllNamedItems:
105 return InvalidateOnIdNameAttrChange;
106 case FieldSetElements:
107 case FormControls:
108 return InvalidateForFormControls;
109 }
110 ASSERT_NOT_REACHED();
111 return DoNotInvalidateOnAttributeChanges;
112}

114HTMLCollection::HTMLCollection(ContainerNode& ownerNode, CollectionType type)
115 : m_collectionType(type)
116 , m_invalidationType(invalidationTypeExcludingIdAndNameAttributes(type))
117 , m_rootType(rootTypeFromCollectionType(type))
118 , m_ownerNode(ownerNode)
119{
120 ASSERT(m_rootType == static_cast<unsigned>(rootTypeFromCollectionType(type)));
121 ASSERT(m_invalidationType == static_cast<unsigned>(invalidationTypeExcludingIdAndNameAttributes(type)));
122 ASSERT(m_collectionType == static_cast<unsigned>(type));
123}

125HTMLCollection::~HTMLCollection()
126{
127 if (hasNamedElementCache())
128 document().collectionWillClearIdNameMap(*this);
129
130 // HTMLNameCollection & ClassCollection remove cache by themselves.
131 // FIXME: We need a cleaner way to handle this.
132 switch (type()) {
133 case ByClass:
134 case ByTag:
135 case ByHTMLTag:
136 case WindowNamedItems:
137 case DocumentNamedItems:
138 case DocumentAllNamedItems:
139 break;
140 default:
141 ownerNode().nodeLists()->removeCachedCollection(this);
142 }
143}

145void HTMLCollection::invalidateCacheForDocument(Document& document)

151void HTMLCollection::invalidateNamedElementCache(Document& document) const
152{
153 ASSERT(hasNamedElementCache());
154 document.collectionWillClearIdNameMap(*this);
155 {
156 Locker locker { m_namedElementCacheAssignmentLock };
157 m_namedElementCache = nullptr;
158 }
159}

161Element* HTMLCollection::namedItemSlow(const AtomString& name) const
162{
163 // The pathological case. We need to walk the entire subtree.
164 updateNamedElementCache();
165 ASSERT(m_namedElementCache);
166
167 if (const Vector<Element*>* idResults = m_namedElementCache->findElementsWithId(name)) {
168 if (idResults->size())
169 return idResults->at(0);
170 }
171
172 if (const Vector<Element*>* nameResults = m_namedElementCache->findElementsWithName(name)) {
173 if (nameResults->size())
174 return nameResults->at(0);
175 }
176
177 return nullptr;
178}

189bool HTMLCollection::isSupportedPropertyName(const String& name)
190{
191 updateNamedElementCache();
192 ASSERT(m_namedElementCache);
193
194 if (m_namedElementCache->findElementsWithId(name))
195 return true;
196 if (m_namedElementCache->findElementsWithName(name))
197 return true;
198
199 return false;
200}
201
202void HTMLCollection::updateNamedElementCache() const
203{
204 if (hasNamedElementCache())
205 return;
206
207 auto cache = makeUnique<CollectionNamedElementCache>();
208
209 unsigned size = length();
210 for (unsigned i = 0; i < size; ++i) {
211 Element& element = *item(i);
212 const AtomString& id = element.getIdAttribute();
213 if (!id.isEmpty())
214 cache->appendToIdCache(id, element);
215 if (!is<HTMLElement>(element))
216 continue;
217 const AtomString& name = element.getNameAttribute();
218 if (!name.isEmpty() && id != name && (type() != DocAll || nameShouldBeVisibleInDocumentAll(downcast<HTMLElement>(element))))
219 cache->appendToNameCache(name, element);
220 }
221
222 setNamedItemCache(WTFMove(cache));
223}

225Vector<Ref<Element>> HTMLCollection::namedItems(const AtomString& name) const
226{
227 // FIXME: This non-virtual function can't possibly be doing the correct thing for
228 // any derived class that overrides the virtual namedItem function.
229
230 Vector<Ref<Element>> elements;
231
232 if (name.isEmpty())
233 return elements;
234
235 updateNamedElementCache();
236 ASSERT(m_namedElementCache);
237
238 auto* elementsWithId = m_namedElementCache->findElementsWithId(name);
239 auto* elementsWithName = m_namedElementCache->findElementsWithName(name);
240
241 elements.reserveInitialCapacity((elementsWithId ? elementsWithId->size() : 0) + (elementsWithName ? elementsWithName->size() : 0));
242
243 if (elementsWithId) {
244 for (auto& element : *elementsWithId)
245 elements.uncheckedAppend(*element);
246 }
247 if (elementsWithName) {
248 for (auto& element : *elementsWithName)
249 elements.uncheckedAppend(*element);
250 }
251
252 return elements;
253}
