47WTF_MAKE_ISO_ALLOCATED_IMPL(HTMLDataListElement);
48
49inline HTMLDataListElement::HTMLDataListElement(const QualifiedName& tagName, Document& document)
50 : HTMLElement(tagName, document)
51{
52 document.incrementDataListElementCount();
53}
54
55HTMLDataListElement::~HTMLDataListElement()
56{
57 document().decrementDataListElementCount();
58}
59
60Ref<HTMLDataListElement> HTMLDataListElement::create(const QualifiedName& tagName, Document& document)
61{
62 return adoptRef(*new HTMLDataListElement(tagName, document));
63}
64
65void HTMLDataListElement::didMoveToNewDocument(Document& oldDocument, Document& newDocument)
66{
67 oldDocument.decrementDataListElementCount();
68 newDocument.incrementDataListElementCount();
69 HTMLElement::didMoveToNewDocument(oldDocument, newDocument);
70}
71
72Ref<HTMLCollection> HTMLDataListElement::options()
73{
74 return ensureRareData().ensureNodeLists().addCachedCollection<GenericCachedHTMLCollection<CollectionTypeTraits<DataListOptions>::traversalType>>(*this, DataListOptions);
75}
76
77void HTMLDataListElement::optionElementChildrenChanged()
78{
79 treeScope().idTargetObserverRegistry().notifyObservers(getIdAttribute());
80}
81
82bool HTMLDataListElement::isSuggestion(const HTMLOptionElement& descendant)
83{
84 return !descendant.isDisabledFormControl() && !descendant.value().isEmpty();
85}
