164enum class ContentEditableType {
165 Inherit,
166 True,
167 False,
168 PlaintextOnly
169};
170
171static inline ContentEditableType contentEditableType(const AtomString& value)
172{
173 if (value.isNull())
174 return ContentEditableType::Inherit;
175 if (value.isEmpty() || equalLettersIgnoringASCIICase(value, "true"))
176 return ContentEditableType::True;
177 if (equalLettersIgnoringASCIICase(value, "false"))
178 return ContentEditableType::False;
179 if (equalLettersIgnoringASCIICase(value, "plaintext-only"))
180 return ContentEditableType::PlaintextOnly;
181
182 return ContentEditableType::Inherit;
183}

186{
187 return contentEditableType(element.attributeWithoutSynchronization(contenteditableAttr));
188}

190void HTMLElement::collectPresentationalHintsForAttribute(const QualifiedName& name, const AtomString& value, MutableStyleProperties& style)
191{
192 if (name == alignAttr) {
193 if (equalLettersIgnoringASCIICase(value, "middle"))
194 addPropertyToPresentationalHintStyle(style, CSSPropertyTextAlign, CSSValueCenter);
195 else
196 addPropertyToPresentationalHintStyle(style, CSSPropertyTextAlign, value);
197 } else if (name == contenteditableAttr) {
198 CSSValueID userModifyValue = CSSValueReadWrite;
199 switch (contentEditableType(value)) {
200 case ContentEditableType::Inherit:
201 return;
202 case ContentEditableType::False:
203 userModifyValue = CSSValueReadOnly;
204 break;
205 case ContentEditableType::PlaintextOnly:
206 userModifyValue = CSSValueReadWritePlaintextOnly;
207 FALLTHROUGH;
208 case ContentEditableType::True:
209 addPropertyToPresentationalHintStyle(style, CSSPropertyWordWrap, CSSValueBreakWord);
210 addPropertyToPresentationalHintStyle(style, CSSPropertyWebkitNbspMode, CSSValueSpace);
211 addPropertyToPresentationalHintStyle(style, CSSPropertyLineBreak, CSSValueAfterWhiteSpace);
212#if PLATFORM(IOS_FAMILY)
213 addPropertyToPresentationalHintStyle(style, CSSPropertyWebkitTextSizeAdjust, CSSValueNone);
214#endif
215 break;
216 }
217 addPropertyToPresentationalHintStyle(style, CSSPropertyWebkitUserModify, userModifyValue);
218 } else if (name == hiddenAttr) {
219 addPropertyToPresentationalHintStyle(style, CSSPropertyDisplay, CSSValueNone);
220 } else if (name == draggableAttr) {
221 if (equalLettersIgnoringASCIICase(value, "true")) {
222 addPropertyToPresentationalHintStyle(style, CSSPropertyWebkitUserDrag, CSSValueElement);
223 if (!isDraggableIgnoringAttributes())
224 addPropertyToPresentationalHintStyle(style, CSSPropertyWebkitUserSelect, CSSValueNone);
225 } else if (equalLettersIgnoringASCIICase(value, "false"))
226 addPropertyToPresentationalHintStyle(style, CSSPropertyWebkitUserDrag, CSSValueNone);
227 } else if (name == dirAttr) {
228 if (equalLettersIgnoringASCIICase(value, "auto"))
229 addPropertyToPresentationalHintStyle(style, CSSPropertyUnicodeBidi, unicodeBidiAttributeForDirAuto(*this));
230 else {
231 auto unicodeBidiValue = CSSValueEmbed;
232
233 if (isLTROrRTLIgnoringCase(value)) {
234 addPropertyToPresentationalHintStyle(style, CSSPropertyDirection, value);
235 unicodeBidiValue = CSSValueIsolate;
236 }
237
238 if (!hasTagName(bdiTag) && !hasTagName(bdoTag) && !hasTagName(outputTag))
239 addPropertyToPresentationalHintStyle(style, CSSPropertyUnicodeBidi, unicodeBidiValue);
240 }
241 } else if (name.matches(XMLNames::langAttr))
242 mapLanguageAttributeToLocale(value, style);
243 else if (name == langAttr) {
244 // xml:lang has a higher priority than lang.
245 if (!hasAttributeWithoutSynchronization(XMLNames::langAttr))
246 mapLanguageAttributeToLocale(value, style);
247 } else
248 StyledElement::collectPresentationalHintsForAttribute(name, value, style);
249}

395void HTMLElement::populateEventHandlerNameMap(EventHandlerNameMap& map, const QualifiedName* const table[], size_t tableSize)
396{
397 for (size_t i = 0; i < tableSize; ++i) {
398 auto* entry = table[i];
399
400 // FIXME: Would be nice to check these against the actual event names in eventNames().
401 // Not obvious how to do that simply, though.
402 auto& attributeName = entry->localName();
403
404 // Remove the "on" prefix. Requires some memory allocation and computing a hash, but by not
405 // using pointers from eventNames(), the passed-in table can be initialized at compile time.
406 AtomString eventName = attributeName.string().substring(2);
407
408 map.add(attributeName.impl(), WTFMove(eventName));
409 }
410}

412const AtomString& HTMLElement::eventNameForEventHandlerAttribute(const QualifiedName& attributeName, const EventHandlerNameMap& map)
413{
414 ASSERT(!attributeName.localName().isNull());
415
416 // Event handler attributes have no namespace.
417 if (!attributeName.namespaceURI().isNull())
418 return nullAtom();
419
420 // Fast early return for names that don't start with "on".
421 AtomStringImpl& localName = *attributeName.localName().impl();
422 if (localName.length() < 3 || localName[0] != 'o' || localName[1] != 'n')
423 return nullAtom();
424
425 auto it = map.find(&localName);
426 return it == map.end() ? nullAtom() : it->value;
427}

435Node::Editability HTMLElement::editabilityFromContentEditableAttr(const Node& node)
436{
437 if (auto* startElement = is<Element>(node) ? &downcast<Element>(node) : node.parentElement()) {
438 for (auto& element : lineageOfType<HTMLElement>(*startElement)) {
439 switch (contentEditableType(element)) {
440 case ContentEditableType::True:
441 return Editability::CanEditRichly;
442 case ContentEditableType::PlaintextOnly:
443 return Editability::CanEditPlainText;
444 case ContentEditableType::False:
445 return Editability::ReadOnly;
446 case ContentEditableType::Inherit:
447 break;
448 }
449 }
450 }
451
452 auto containingShadowRoot = makeRefPtr(node.containingShadowRoot());
453 if (containingShadowRoot && containingShadowRoot->mode() == ShadowRootMode::UserAgent)
454 return Editability::ReadOnly;
455
456 auto& document = node.document();
457 if (is<HTMLDocument>(document))
458 return downcast<HTMLDocument>(document).inDesignMode() ? Editability::CanEditRichly : Editability::ReadOnly;
459
460 return Editability::ReadOnly;
461}

468void HTMLElement::parseAttribute(const QualifiedName& name, const AtomString& value)
469{
470 if (name == dirAttr) {
471 dirAttributeChanged(value);
472 return;
473 }
474
475 if (name == tabindexAttr) {
476 if (auto optionalTabIndex = parseHTMLInteger(value))
477 setTabIndexExplicitly(optionalTabIndex.value());
478 else
479 setTabIndexExplicitly(std::nullopt);
480 return;
481 }
482
483 if (name == inputmodeAttr) {
484 auto& document = this->document();
485 if (this == document.focusedElement()) {
486 if (auto* page = document.page())
487 page->chrome().client().focusedElementDidChangeInputMode(*this, canonicalInputMode());
488 }
489 }
490
491 auto& eventName = eventNameForEventHandlerAttribute(name);
492 if (!eventName.isNull())
493 setAttributeEventListener(eventName, name, value);
494}

496static Ref<DocumentFragment> textToFragment(Document& document, const String& text)
497{
498 auto fragment = DocumentFragment::create(document);
499
500 // It's safe to dispatch events on the new fragment since author scripts have no access to it yet.
501 ScriptDisallowedScope::EventAllowedScope allowedScope(fragment);
502
503 for (unsigned start = 0, length = text.length(); start < length; ) {
504 // Find next line break.
505 UChar c = 0;
506 unsigned i;
507 for (i = start; i < length; i++) {
508 c = text[i];
509 if (c == '\r' || c == '\n')
510 break;
511 }
512
513 // If text is not the empty string, then append a new Text node whose data is text and node document is document to fragment.
514 if (i > start)
515 fragment->appendChild(Text::create(document, text.substring(start, i - start)));
516
517 if (i == length)
518 break;
519
520 fragment->appendChild(HTMLBRElement::create(document));
521 // Make sure \r\n doesn't result in two line breaks.
522 if (c == '\r' && i + 1 < length && text[i + 1] == '\n')
523 ++i;
524
525 start = i + 1; // Character after line break.
526 }
527
528 return fragment;
529}

531// Returns the conforming 'dir' value associated with the state the attribute is in (in its canonical case), if any,
532// or the empty string if the attribute is in a state that has no associated keyword value or if the attribute is
533// not in a defined state (e.g. the attribute is missing and there is no missing value default).
534// http://www.whatwg.org/specs/web-apps/current-work/multipage/common-dom-interfaces.html#limited-to-only-known-values
535static inline const AtomString& toValidDirValue(const AtomString& value)
536{
537 static MainThreadNeverDestroyed<const AtomString> ltrValue("ltr", AtomString::ConstructFromLiteral);
538 static MainThreadNeverDestroyed<const AtomString> rtlValue("rtl", AtomString::ConstructFromLiteral);
539 static MainThreadNeverDestroyed<const AtomString> autoValue("auto", AtomString::ConstructFromLiteral);
540 if (equalLettersIgnoringASCIICase(value, "ltr"))
541 return ltrValue;
542 if (equalLettersIgnoringASCIICase(value, "rtl"))
543 return rtlValue;
544 if (equalLettersIgnoringASCIICase(value, "auto"))
545 return autoValue;
546 return nullAtom();
547}

559ExceptionOr<void> HTMLElement::setInnerText(const String& text)
560{
561 // FIXME: This doesn't take whitespace collapsing into account at all.
562
563 if (!text.contains('\n') && !text.contains('\r')) {
564 stringReplaceAll(text);
565 return { };
566 }
567
568 // FIXME: Do we need to be able to detect preserveNewline style even when there's no renderer?
569 // FIXME: Can the renderer be out of date here? Do we need to call updateStyleIfNeeded?
570 // For example, for the contents of textarea elements that are display:none?
571 auto* r = renderer();
572 if ((r && r->style().preserveNewline()) || (isConnected() && isTextControlInnerTextElement())) {
573 if (!text.contains('\r')) {
574 stringReplaceAll(text);
575 return { };
576 }
577 String textWithConsistentLineBreaks = text;
578 textWithConsistentLineBreaks.replace("\r\n", "\n");
579 textWithConsistentLineBreaks.replace('\r', '\n');
580 stringReplaceAll(textWithConsistentLineBreaks);
581 return { };
582 }
583
584 // FIXME: This should use replaceAll(), after we fix that to work properly for DocumentFragment.
585 // Add text nodes and <br> elements.
586 auto fragment = textToFragment(document(), text);
587 // It's safe to dispatch events on the new fragment since author scripts have no access to it yet.
588 ScriptDisallowedScope::EventAllowedScope allowedScope(fragment.get());
589 return replaceChildrenWithFragment(*this, WTFMove(fragment));
590}

592ExceptionOr<void> HTMLElement::setOuterText(const String& text)
593{
594 RefPtr<ContainerNode> parent = parentNode();
595 if (!parent)
596 return Exception { NoModificationAllowedError };
597
598 RefPtr<Node> prev = previousSibling();
599 RefPtr<Node> next = nextSibling();
600 RefPtr<Node> newChild;
601
602 // Convert text to fragment with <br> tags instead of linebreaks if needed.
603 if (text.contains('\r') || text.contains('\n'))
604 newChild = textToFragment(document(), text);
605 else
606 newChild = Text::create(document(), text);
607
608 if (!parentNode())
609 return Exception { HierarchyRequestError };
610
611 auto replaceResult = parent->replaceChild(*newChild, *this);
612 if (replaceResult.hasException())
613 return replaceResult.releaseException();
614
615 RefPtr<Node> node = next ? next->previousSibling() : nullptr;
616 if (is<Text>(node)) {
617 auto result = mergeWithNextTextNode(downcast<Text>(*node));
618 if (result.hasException())
619 return result.releaseException();
620 }
621 if (is<Text>(prev)) {
622 auto result = mergeWithNextTextNode(downcast<Text>(*prev));
623 if (result.hasException())
624 return result.releaseException();
625 }
626 return { };
627}

629void HTMLElement::applyAspectRatioFromWidthAndHeightAttributesToStyle(StringView widthAttribute, StringView heightAttribute, MutableStyleProperties& style)
630{
631 if (!document().settings().aspectRatioOfImgFromWidthAndHeightEnabled())
632 return;
633
634 auto dimensionWidth = parseHTMLDimension(widthAttribute);
635 if (!dimensionWidth || dimensionWidth->type != HTMLDimension::Type::Pixel)
636 return;
637 auto dimensionHeight = parseHTMLDimension(heightAttribute);
638 if (!dimensionHeight || dimensionHeight->type != HTMLDimension::Type::Pixel)
639 return;
640
641 auto ratioList = CSSValueList::createSlashSeparated();
642 ratioList->append(CSSValuePool::singleton().createValue(dimensionWidth->number, CSSUnitType::CSS_NUMBER));
643 ratioList->append(CSSValuePool::singleton().createValue(dimensionHeight->number, CSSUnitType::CSS_NUMBER));
644 auto list = CSSValueList::createSpaceSeparated();
645 list->append(CSSValuePool::singleton().createIdentifierValue(CSSValueAuto));
646 list->append(ratioList);
647
648 style.setProperty(CSSPropertyAspectRatio, RefPtr<CSSValue>(WTFMove(list)));
649}

651void HTMLElement::applyAlignmentAttributeToStyle(const AtomString& alignment, MutableStyleProperties& style)
652{
653 // Vertical alignment with respect to the current baseline of the text
654 // right or left means floating images.
655 CSSValueID floatValue = CSSValueInvalid;
656 CSSValueID verticalAlignValue = CSSValueInvalid;
657
658 if (equalLettersIgnoringASCIICase(alignment, "absmiddle"))
659 verticalAlignValue = CSSValueMiddle;
660 else if (equalLettersIgnoringASCIICase(alignment, "absbottom"))
661 verticalAlignValue = CSSValueBottom;
662 else if (equalLettersIgnoringASCIICase(alignment, "left")) {
663 floatValue = CSSValueLeft;
664 verticalAlignValue = CSSValueTop;
665 } else if (equalLettersIgnoringASCIICase(alignment, "right")) {

915TextDirection HTMLElement::directionality(Node** strongDirectionalityTextNode) const
916{
917 if (isTextField()) {
918 HTMLTextFormControlElement& textElement = downcast<HTMLTextFormControlElement>(const_cast<HTMLElement&>(*this));
919 bool hasStrongDirectionality;
920 UCharDirection textDirection = textElement.value().defaultWritingDirection(&hasStrongDirectionality);
921 if (strongDirectionalityTextNode)
922 *strongDirectionalityTextNode = hasStrongDirectionality ? &textElement : nullptr;
923 return (textDirection == U_LEFT_TO_RIGHT) ? TextDirection::LTR : TextDirection::RTL;
924 }
925
926 RefPtr<Node> node = firstChild();
927 while (node) {
928 // Skip bdi, script, style and text form controls.
929 if (equalLettersIgnoringASCIICase(node->nodeName(), "bdi") || node->hasTagName(scriptTag) || node->hasTagName(styleTag)
930 || (is<Element>(*node) && downcast<Element>(*node).isTextField())) {
931 node = NodeTraversal::nextSkippingChildren(*node, this);
932 continue;
933 }
934
935 // Skip elements with valid dir attribute
936 if (is<Element>(*node)) {
937 auto& dirAttributeValue = downcast<Element>(*node).attributeWithoutSynchronization(dirAttr);
938 if (isLTROrRTLIgnoringCase(dirAttributeValue) || equalLettersIgnoringASCIICase(dirAttributeValue, "auto")) {
939 node = NodeTraversal::nextSkippingChildren(*node, this);
940 continue;
941 }
942 }
943
944 if (node->isTextNode()) {
945 bool hasStrongDirectionality;
946 UCharDirection textDirection = node->textContent(true).defaultWritingDirection(&hasStrongDirectionality);
947 if (hasStrongDirectionality) {
948 if (strongDirectionalityTextNode)
949 *strongDirectionalityTextNode = node.get();
950 return (textDirection == U_LEFT_TO_RIGHT) ? TextDirection::LTR : TextDirection::RTL;
951 }
952 }
953 node = NodeTraversal::next(*node, this);
954 }
955 if (strongDirectionalityTextNode)
956 *strongDirectionalityTextNode = nullptr;
957 return TextDirection::LTR;
958}

1060// Color parsing that matches HTML's "rules for parsing a legacy color value"
1061// https://html.spec.whatwg.org/#rules-for-parsing-a-legacy-colour-value
1062static std::optional<SRGBA<uint8_t>> parseLegacyColorValue(StringView string)
1063{
1064 // An empty string doesn't apply a color.
1065 if (string.isEmpty())
1066 return std::nullopt;
1067
1068 string = string.stripLeadingAndTrailingMatchedCharacters(isHTMLSpace<UChar>);
1069 if (string.isEmpty())
1070 return Color::black;
1071
1072 // "transparent" doesn't apply a color either.
1073 if (equalLettersIgnoringASCIICase(string, "transparent"))
1074 return std::nullopt;
1075
1076 if (auto namedColor = CSSParser::parseNamedColor(string))
1077 return namedColor;
1078
1079 if (string.length() == 4 && string[0] == '#' && isASCIIHexDigit(string[1]) && isASCIIHexDigit(string[2]) && isASCIIHexDigit(string[3]))
1080 return { { static_cast<uint8_t>(toASCIIHexValue(string[1]) * 0x11), static_cast<uint8_t>(toASCIIHexValue(string[2]) * 0x11), static_cast<uint8_t>(toASCIIHexValue(string[3]) * 0x11) } };
1081
1082 // Per spec, only look at the first 128 digits of the string.
1083 constexpr unsigned maxColorLength = 128;
1084
1085 // We'll pad the buffer with two extra 0s later, so reserve two more than the max.
1086 Vector<char, maxColorLength + 2> digitBuffer;
1087
1088 // Grab the first 128 characters, replacing non-hex characters with 0.
1089 // Non-BMP characters are replaced with "00" due to them appearing as two "characters" in the String.
1090 unsigned i = 0;
1091 if (string[0] == '#') // Skip a leading #.
1092 i = 1;
1093 for (; i < string.length() && digitBuffer.size() < maxColorLength; i++) {
1094 if (!isASCIIHexDigit(string[i]))
1095 digitBuffer.append('0');
1096 else
1097 digitBuffer.append(string[i]);
1098 }
1099
1100 if (digitBuffer.isEmpty())
1101 return Color::black;
1102
1103 // Pad the buffer out to at least the next multiple of three in size.
1104 digitBuffer.append('0');
1105 digitBuffer.append('0');
1106
1107 if (digitBuffer.size() < 6)
1108 return { { toASCIIHexValue(digitBuffer[0]), toASCIIHexValue(digitBuffer[1]), toASCIIHexValue(digitBuffer[2]) } };
1109
1110 // Split the digits into three components, then search the last 8 digits of each component.
1111 ASSERT(digitBuffer.size() >= 6);
1112 unsigned componentLength = digitBuffer.size() / 3;
1113 unsigned componentSearchWindowLength = std::min(componentLength, 8U);
1114 unsigned redIndex = componentLength - componentSearchWindowLength;
1115 unsigned greenIndex = componentLength * 2 - componentSearchWindowLength;
1116 unsigned blueIndex = componentLength * 3 - componentSearchWindowLength;
1117 // Skip digits until one of them is non-zero, or we've only got two digits left in the component.
1118 while (digitBuffer[redIndex] == '0' && digitBuffer[greenIndex] == '0' && digitBuffer[blueIndex] == '0' && (componentLength - redIndex) > 2) {
1119 redIndex++;
1120 greenIndex++;
1121 blueIndex++;
1122 }
1123 ASSERT(redIndex + 1 < componentLength);
1124 ASSERT(greenIndex >= componentLength);
1125 ASSERT(greenIndex + 1 < componentLength * 2);
1126 ASSERT(blueIndex >= componentLength * 2);
1127 ASSERT_WITH_SECURITY_IMPLICATION(blueIndex + 1 < digitBuffer.size());
1128
1129 uint8_t redValue = toASCIIHexValue(digitBuffer[redIndex], digitBuffer[redIndex + 1]);
1130 uint8_t greenValue = toASCIIHexValue(digitBuffer[greenIndex], digitBuffer[greenIndex + 1]);
1131 uint8_t blueValue = toASCIIHexValue(digitBuffer[blueIndex], digitBuffer[blueIndex + 1]);
1132 return { { redValue, greenValue, blueValue } };
1133}

1359 bool hadExistingTextRecognitionElements = false;

1362 if (hasImageOverlay()) {
1363 for (auto& child : childrenOfType<HTMLDivElement>(*userAgentShadowRoot())) {

1375 continue;

1380 lineElements.children.append(text);
1381 textRecognitionElements.lines.append(WTFMove(lineElements));

1388 return false;
1389
1390 if (result.lines.size() != textRecognitionElements.lines.size())
1391 return false;

1394 auto& childResults = result.lines[lineIndex].children;

1396 if (childResults.size() != childTextElements.size())
1397 return false;

1399 for (size_t childIndex = 0; childIndex < childResults.size(); ++childIndex) {
1400 if (childResults[childIndex].text != childTextElements[childIndex]->textContent().stripWhiteSpace())
1401 return false;
1402 }

1406 })();

1428 auto lineContainer = HTMLDivElement::create(document());

1430 rootContainer->appendChild(lineContainer);
1431 TextRecognitionLineElements lineElements { lineContainer, { } };
1432 lineElements.children.reserveInitialCapacity(line.children.size());
1433 for (size_t childIndex = 0; childIndex < line.children.size(); ++childIndex) {
1434 auto& child = line.children[childIndex];
1435 auto textContainer = HTMLDivElement::create(document());
1436 textContainer->classList().add(imageOverlayTextClass);
1437 lineContainer->appendChild(textContainer);

1439 lineElements.children.uncheckedAppend(WTFMove(textContainer));
1440 }
1441
1442 lineContainer->appendChild(HTMLBRElement::create(document()));
1443 textRecognitionElements.lines.uncheckedAppend(WTFMove(lineElements));
1444 }
1445
1446#if ENABLE(DATA_DETECTION)
1447 textRecognitionElements.dataDetectors.reserveInitialCapacity(result.dataDetectors.size());

1449 auto dataDetectorContainer = DataDetection::createElementForImageOverlay(document(), dataDetector);
1450 dataDetectorContainer->classList().add(imageOverlayDataDetectorClassName());

1452 textRecognitionElements.dataDetectors.uncheckedAppend(WTFMove(dataDetectorContainer));
1453 }
1454#endif // ENABLE(DATA_DETECTION)
1455
1456 if (document().quirks().needsToForceUserSelectWhenInstallingImageOverlay())
1457 setInlineStyleProperty(CSSPropertyWebkitUserSelect, CSSValueText);
1458 }
1459
1460 if (!hadExistingTextRecognitionElements) {
1461 static MainThreadNeverDestroyed<const String> shadowStyle(StringImpl::createWithoutCopying(imageOverlayUserAgentStyleSheet, sizeof(imageOverlayUserAgentStyleSheet)));
1462 auto style = HTMLStyleElement::create(HTMLNames::styleTag, document(), false);
1463 style->setTextContent(shadowStyle);
1464 shadowRoot->appendChild(WTFMove(style));

1467 document().updateLayoutIgnorePendingStylesheets();
1468
1469 auto* renderer = this->renderer();
1470 if (!is<RenderImage>(renderer))
1471 return;
1472
1473 downcast<RenderImage>(*renderer).setHasImageOverlay();
1474
1475 auto containerRect = enclosingIntRect(downcast<RenderImage>(*renderer).replacedContentRect());
1476 auto convertToContainerCoordinates = [&](const FloatQuad& normalizedQuad) {
1477 auto quad = normalizedQuad;
1478 quad.scale(containerRect.width(), containerRect.height());
1479 quad.move(containerRect.x(), containerRect.y());
1480 return quad;
1481 };
1482
1483 for (size_t lineIndex = 0; lineIndex < result.lines.size(); ++lineIndex) {
1484 auto& lineElements = textRecognitionElements.lines[lineIndex];
1485 auto& lineContainer = lineElements.line;
1486 auto& line = result.lines[lineIndex];
1487 auto lineQuad = convertToContainerCoordinates(line.normalizedQuad);
1488 if (lineQuad.isEmpty())
1489 continue;

1491 auto lineBounds = rotatedBoundingRectWithMinimumAngleOfRotation(lineQuad, 0.01);
1492 lineContainer->setInlineStyleProperty(CSSPropertyWidth, lineBounds.size.width(), CSSUnitType::CSS_PX);
1493 lineContainer->setInlineStyleProperty(CSSPropertyHeight, lineBounds.size.height(), CSSUnitType::CSS_PX);
1494 lineContainer->setInlineStyleProperty(CSSPropertyTransform, makeString(
1495 "translate("_s,
1496 std::round(lineBounds.center.x() - (lineBounds.size.width() / 2)), "px, "_s,
1497 std::round(lineBounds.center.y() - (lineBounds.size.height() / 2)), "px) "_s,
1498 lineBounds.angleInRadians ? makeString("rotate("_s, lineBounds.angleInRadians, "rad) "_s) : emptyString()
1499 ));

1502 auto intervalLength = lineBounds.size.width();

1504 mid.moveBy(-lineBounds.center);
1505 mid.rotate(-lineBounds.angleInRadians);

1507 };

1510 auto textQuad = convertToContainerCoordinates(child.normalizedQuad);
1511 return {
1512 offsetAlongHorizontalAxis(textQuad.p1(), textQuad.p4()),
1513 offsetAlongHorizontalAxis(textQuad.p2(), textQuad.p3())
1514 };
1515 });
1516
1517 for (size_t childIndex = 0; childIndex < line.children.size(); ++childIndex) {
1518 auto& textContainer = lineElements.children[childIndex];
1519 bool lineHasOneChild = line.children.size() == 1;
1520 float horizontalMarginToMinimizeSelectionGaps = lineHasOneChild ? 0 : 0.125;
1521 float horizontalOffset = lineHasOneChild ? 0 : -horizontalMarginToMinimizeSelectionGaps;
1522 float horizontalExtent = lineHasOneChild ? 0 : horizontalMarginToMinimizeSelectionGaps;
1523

1525 horizontalOffset += offsetsAlongHorizontalAxis[childIndex].begin();
1526 horizontalExtent += offsetsAlongHorizontalAxis[childIndex].end();
1527 } else if (!childIndex) {
1528 horizontalOffset += offsetsAlongHorizontalAxis[childIndex].begin();
1529 horizontalExtent += (offsetsAlongHorizontalAxis[childIndex].end() + offsetsAlongHorizontalAxis[childIndex + 1].begin()) / 2;
1530 } else if (childIndex == line.children.size() - 1) {
1531 horizontalOffset += (offsetsAlongHorizontalAxis[childIndex - 1].end() + offsetsAlongHorizontalAxis[childIndex].begin()) / 2;
1532 horizontalExtent += offsetsAlongHorizontalAxis[childIndex].end();
1533 } else {
1534 horizontalOffset += (offsetsAlongHorizontalAxis[childIndex - 1].end() + offsetsAlongHorizontalAxis[childIndex].begin()) / 2;
1535 horizontalExtent += (offsetsAlongHorizontalAxis[childIndex].end() + offsetsAlongHorizontalAxis[childIndex + 1].begin()) / 2;
1536 }

1538 FloatSize targetSize { horizontalExtent - horizontalOffset, lineBounds.size.height() };
1539 if (targetSize.isEmpty()) {
1540 textContainer->setInlineStyleProperty(CSSPropertyTransform, "scale(0, 0)");

1560 "translate("_s,
1561 horizontalOffset + (targetSize.width() - sizeBeforeTransform.width()) / 2, "px, "_s,

1563 "scale("_s, targetSize.width() / sizeBeforeTransform.width(), ", "_s, targetSize.height() / sizeBeforeTransform.height(), ") "_s
1564 ));

1572 if (document().isImageDocument())
1573 lineContainer->setInlineStyleProperty(CSSPropertyCursor, CSSValueText);
1574 }

1592 targetBounds.angleInRadians ? makeString("rotate("_s, targetBounds.angleInRadians, "rad) "_s) : emptyString()
