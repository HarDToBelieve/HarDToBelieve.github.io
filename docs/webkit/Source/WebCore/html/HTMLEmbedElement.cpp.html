66static inline RenderWidget* findWidgetRenderer(const Node* node)
67{
68 if (!node->renderer())
69 node = ancestorsOfType<HTMLObjectElement>(*node).first();
70 if (node && is<RenderWidget>(node->renderer()))
71 return downcast<RenderWidget>(node->renderer());
72 return nullptr;
73}

82void HTMLEmbedElement::collectPresentationalHintsForAttribute(const QualifiedName& name, const AtomString& value, MutableStyleProperties& style)
83{
84 if (name == hiddenAttr) {
85 if (equalLettersIgnoringASCIICase(value, "yes") || equalLettersIgnoringASCIICase(value, "true")) {
86 addPropertyToPresentationalHintStyle(style, CSSPropertyWidth, 0, CSSUnitType::CSS_PX);
87 addPropertyToPresentationalHintStyle(style, CSSPropertyHeight, 0, CSSUnitType::CSS_PX);
88 }
89 } else
90 HTMLPlugInImageElement::collectPresentationalHintsForAttribute(name, value, style);
91}

94{
95 return embed.hasAttributeWithoutSynchronization(typeAttr) || embed.hasAttributeWithoutSynchronization(srcAttr);

98void HTMLEmbedElement::parseAttribute(const QualifiedName& name, const AtomString& value)
99{
100 if (name == typeAttr) {
101 m_serviceType = value.string().left(value.find(';')).convertToASCIILowercase();
102 // FIXME: The only difference between this and HTMLObjectElement's corresponding
103 // code is that HTMLObjectElement does setNeedsWidgetUpdate(true). Consider moving
104 // this up to the HTMLPlugInImageElement to be shared.
105 if (renderer() && !hasTypeOrSrc(*this))
106 invalidateStyle();
107 } else if (name == codeAttr) {
108 m_url = stripLeadingAndTrailingHTMLSpaces(value);
109 // FIXME: Why no call to updateImageLoaderWithNewURLSoon?
110 // FIXME: If both code and src attributes are specified, last one parsed/changed wins. That can't be right!
111 } else if (name == srcAttr) {
112 m_url = stripLeadingAndTrailingHTMLSpaces(value);
113 updateImageLoaderWithNewURLSoon();
114 if (renderer() && !hasTypeOrSrc(*this))
115 invalidateStyle();
116 // FIXME: If both code and src attributes are specified, last one parsed/changed wins. That can't be right!
117 } else
118 HTMLPlugInImageElement::parseAttribute(name, value);
119}

121void HTMLEmbedElement::parametersForPlugin(Vector<String>& paramNames, Vector<String>& paramValues)
122{
123 if (!hasAttributes())
124 return;
125
126 for (const Attribute& attribute : attributesIterator()) {
127 paramNames.append(attribute.localName().string());
128 paramValues.append(attribute.value().string());
129 }
130}

132// FIXME: This should be unified with HTMLObjectElement::updateWidget and
133// moved down into HTMLPluginImageElement.cpp
134void HTMLEmbedElement::updateWidget(CreatePlugins createPlugins)
135{
136 ASSERT(!renderEmbeddedObject()->isPluginUnavailable());
137 ASSERT(needsWidgetUpdate());
138
139 if (m_url.isEmpty() && m_serviceType.isEmpty()) {
140 setNeedsWidgetUpdate(false);
141 return;
142 }
143
144 // Note these pass m_url and m_serviceType to allow better code sharing with
145 // <object> which modifies url and serviceType before calling these.
146 if (!canLoadURL(m_url)) {
147 setNeedsWidgetUpdate(false);
148 return;
149 }
150
151 // FIXME: It's unfortunate that we have this special case here.
152 // See http://trac.webkit.org/changeset/25128 and the plugins/netscape-plugin-setwindow-size.html test.
153 if (createPlugins == CreatePlugins::No && wouldLoadAsPlugIn(m_url, m_serviceType))
154 return;
155
156 setNeedsWidgetUpdate(false);
157
158 // FIXME: These should be joined into a PluginParameters class.
159 Vector<String> paramNames;
160 Vector<String> paramValues;
161 parametersForPlugin(paramNames, paramValues);
162
163 Ref<HTMLEmbedElement> protectedThis(*this); // Loading the plugin might remove us from the document.
164 bool beforeLoadAllowedLoad = guardedDispatchBeforeLoadEvent(m_url);
165 if (!beforeLoadAllowedLoad) {
166 if (is<PluginDocument>(document())) {
167 // Plugins inside plugin documents load differently than other plugins. By the time
168 // we are here in a plugin document, the load of the plugin (which is the plugin document's
169 // main resource) has already started. We need to explicitly cancel the main resource load here.
170 downcast<PluginDocument>(document()).cancelManualPluginLoad();
171 }
172 return;
173 }
174 if (!renderer()) // Do not load the plugin if beforeload removed this element or its renderer.
175 return;
176
177 // Dispatching a beforeLoad event could have executed code that changed the document.
178 // Make sure the URL is still safe to load.
179 if (!canLoadURL(m_url))
180 return;
181
182 // FIXME: beforeLoad could have detached the renderer! Just like in the <object> case above.
183 requestObject(m_url, m_serviceType, paramNames, paramValues);
184}

186bool HTMLEmbedElement::rendererIsNeeded(const RenderStyle& style)
187{
188 if (!hasTypeOrSrc(*this))
189 return false;
190
191 if (isImageType())
192 return HTMLPlugInImageElement::rendererIsNeeded(style);
193
194 // If my parent is an <object> and is not set to use fallback content, I
195 // should be ignored and not get a renderer.
196 RefPtr<ContainerNode> parent = parentNode();
197 if (is<HTMLObjectElement>(parent)) {
198 if (!parent->renderer())
199 return false;
200 if (!downcast<HTMLObjectElement>(*parent).useFallbackContent()) {
201 ASSERT(!parent->renderer()->isEmbeddedObject());
202 return false;
203 }
204 }
205
206 return HTMLPlugInImageElement::rendererIsNeeded(style);
207}
