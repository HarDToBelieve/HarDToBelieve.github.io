26#include "HTMLFormControlElement.h"

28namespace WebCore {
29
30using FormControlState = Vector<String>;
31
32class HTMLFormControlElementWithState : public HTMLFormControlElement {
33 WTF_MAKE_ISO_ALLOCATED(HTMLFormControlElementWithState);
34public:
35 virtual ~HTMLFormControlElementWithState();
36
37 virtual bool shouldSaveAndRestoreFormControlState() const;
38 virtual FormControlState saveFormControlState() const;
39 virtual void restoreFormControlState(const FormControlState&) { } // Called only if state is not empty.
40
41protected:
42 HTMLFormControlElementWithState(const QualifiedName& tagName, Document&, HTMLFormElement*);
43
44 virtual bool shouldAutocomplete() const;
45
46 bool canContainRangeEndPoint() const override { return false; }
47 void finishParsingChildren() override;
48 InsertedIntoAncestorResult insertedIntoAncestor(InsertionType, ContainerNode&) override;
49 void removedFromAncestor(RemovalType, ContainerNode&) override;
50
51private:
52 bool isFormControlElementWithState() const final;
53};
54
55} // namespace WebCore

57SPECIALIZE_TYPE_TRAITS_BEGIN(WebCore::HTMLFormControlElementWithState)

59SPECIALIZE_TYPE_TRAITS_END()
