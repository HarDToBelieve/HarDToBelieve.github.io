57std::optional<Variant<RefPtr<RadioNodeList>, RefPtr<Element>>> HTMLFormControlsCollection::namedItemOrItems(const String& name) const
58{
59 auto namedItems = this->namedItems(name);
60
61 if (namedItems.isEmpty())
62 return std::nullopt;
63 if (namedItems.size() == 1)
64 return Variant<RefPtr<RadioNodeList>, RefPtr<Element>> { RefPtr<Element> { WTFMove(namedItems[0]) } };
65
66 return Variant<RefPtr<RadioNodeList>, RefPtr<Element>> { RefPtr<RadioNodeList> { ownerNode().radioNodeList(name) } };
67}

69static unsigned findFormAssociatedElement(const Vector<WeakPtr<HTMLElement>>& elements, const Element& element)
70{
71 for (unsigned i = 0; i < elements.size(); ++i) {
72 auto currentElement = makeRefPtr(elements[i].get());
73 ASSERT(currentElement);
74 auto* associatedElement = currentElement->asFormAssociatedElement();
75 ASSERT(associatedElement);
76 if (associatedElement->isEnumeratable() && currentElement == &element)
77 return i;
78 }
79 return elements.size();
80}

82HTMLElement* HTMLFormControlsCollection::customElementAfter(Element* current) const
83{
84 ScriptDisallowedScope::InMainThread scriptDisallowedScope;
85 auto& elements = ownerNode().unsafeAssociatedElements();
86 unsigned start;
87 if (!current)
88 start = 0;
89 else if (m_cachedElement == current)
90 start = m_cachedElementOffsetInArray + 1;
91 else
92 start = findFormAssociatedElement(elements, *current) + 1;
93
94 for (unsigned i = start; i < elements.size(); ++i) {
95 auto element = makeRefPtr(elements[i].get());
96 ASSERT(element);
97 ASSERT(element->asFormAssociatedElement());
98 if (element->asFormAssociatedElement()->isEnumeratable()) {
99 m_cachedElement = element.get();
100 m_cachedElementOffsetInArray = i;
101 return element.get();
102 }
103 }
104 return nullptr;
105}

112void HTMLFormControlsCollection::updateNamedElementCache() const
113{
114 if (hasNamedElementCache())
115 return;
116
117 auto cache = makeUnique<CollectionNamedElementCache>();
118
119 HashSet<AtomStringImpl*> foundInputElements;
120
121 ScriptDisallowedScope::InMainThread scriptDisallowedScope;
122 for (auto& weakElement : ownerNode().unsafeAssociatedElements()) {
123 auto element = makeRefPtr(weakElement.get());
124 ASSERT(element);
125 auto* associatedElement = element->asFormAssociatedElement();
126 ASSERT(associatedElement);
127 if (associatedElement->isEnumeratable()) {
128 const AtomString& id = element->getIdAttribute();
129 if (!id.isEmpty()) {
130 cache->appendToIdCache(id, *element);
131 foundInputElements.add(id.impl());
132 }
133 const AtomString& name = element->getNameAttribute();
134 if (!name.isEmpty() && id != name) {
135 cache->appendToNameCache(name, *element);
136 foundInputElements.add(name.impl());
137 }
138 }
139 }
140
141 for (auto& elementPtr : ownerNode().imageElements()) {
142 if (!elementPtr)
143 continue;
144 HTMLImageElement& element = *elementPtr;
145 const AtomString& id = element.getIdAttribute();
146 if (!id.isEmpty() && !foundInputElements.contains(id.impl()))
147 cache->appendToIdCache(id, element);
148 const AtomString& name = element.getNameAttribute();
149 if (!name.isEmpty() && id != name && !foundInputElements.contains(name.impl()))
150 cache->appendToNameCache(name, element);
151 }
152
153 setNamedItemCache(WTFMove(cache));
154}
