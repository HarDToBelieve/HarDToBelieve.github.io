40inline HTMLFrameElement::HTMLFrameElement(const QualifiedName& tagName, Document& document)
41 : HTMLFrameElementBase(tagName, document)
42{
43 ASSERT(hasTagName(frameTag));
44 setHasCustomStyleResolveCallbacks();
45}
46
47Ref<HTMLFrameElement> HTMLFrameElement::create(const QualifiedName& tagName, Document& document)
48{
49 return adoptRef(*new HTMLFrameElement(tagName, document));
50}
51
52bool HTMLFrameElement::rendererIsNeeded(const RenderStyle&)
53{
54 // For compatibility, frames render even when display: none is set.
55 return canLoad();
56}
57
58RenderPtr<RenderElement> HTMLFrameElement::createElementRenderer(RenderStyle&& style, const RenderTreePosition&)
59{
60 return createRenderer<RenderFrame>(*this, WTFMove(style));
61}
62
63bool HTMLFrameElement::noResize() const
64{
65 return hasAttributeWithoutSynchronization(noresizeAttr);
66}
67
68void HTMLFrameElement::didAttachRenderers()
69{
70 HTMLFrameElementBase::didAttachRenderers();
71 const auto containingFrameSet = HTMLFrameSetElement::findContaining(this);
72 if (!containingFrameSet)
73 return;
74 if (!m_frameBorderSet)
75 m_frameBorder = containingFrameSet->hasFrameBorder();
76}
77
78int HTMLFrameElement::defaultTabIndex() const
79{
80 return 0;
81}
82
83void HTMLFrameElement::parseAttribute(const QualifiedName& name, const AtomString& value)
84{
85 if (name == frameborderAttr) {
86 m_frameBorder = parseHTMLInteger(value).value_or(0);
87 m_frameBorderSet = !value.isNull();
88 // FIXME: If we are already attached, this has no effect.
89 } else if (name == noresizeAttr) {
90 if (auto* renderer = this->renderer())
91 renderer->updateFromElement();
92 } else
93 HTMLFrameElementBase::parseAttribute(name, value);
94}
