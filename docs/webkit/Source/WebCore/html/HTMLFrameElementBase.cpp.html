60bool HTMLFrameElementBase::canLoad() const
61{
62 // FIXME: Why is it valuable to return true when m_frameURL is empty?
63 // FIXME: After openURL replaces an empty URL with the blank URL, this may no longer necessarily return true.
64 return m_frameURL.isEmpty() || canLoadURL(m_frameURL);
65}

67bool HTMLFrameElementBase::canLoadURL(const String& relativeURL) const
68{
69 return canLoadURL(document().completeURL(relativeURL));
70}

72// Note that unlike HTMLPlugInImageElement::canLoadURL this uses ScriptController::canAccessFromCurrentOrigin.
73bool HTMLFrameElementBase::canLoadURL(const URL& completeURL) const
74{
75 if (completeURL.protocolIsJavaScript()) {
76 RefPtr<Document> contentDocument = this->contentDocument();
77 if (contentDocument && !ScriptController::canAccessFromCurrentOrigin(contentDocument->frame(), document()))
78 return false;
79 }
80
81 return !isProhibitedSelfReference(completeURL);
82}
83
84void HTMLFrameElementBase::openURL(LockHistory lockHistory, LockBackForwardList lockBackForwardList)
85{
86 if (!canLoad())
87 return;
88
89 if (m_frameURL.isEmpty())
90 m_frameURL = aboutBlankURL().string();
91
92 if (shouldLoadFrameLazily())
93 return;
94
95 RefPtr<Frame> parentFrame = document().frame();
96 if (!parentFrame)
97 return;
98
99 document().willLoadFrameElement(document().completeURL(m_frameURL));
100
101 String frameName = getNameAttribute();
102 if (frameName.isNull() && UNLIKELY(document().settings().needsFrameNameFallbackToIdQuirk()))
103 frameName = getIdAttribute();
104
105 parentFrame->loader().subframeLoader().requestFrame(*this, m_frameURL, frameName, lockHistory, lockBackForwardList);
106}

108void HTMLFrameElementBase::parseAttribute(const QualifiedName& name, const AtomString& value)
109{
110 if (name == srcdocAttr) {
111 if (value.isNull()) {
112 const AtomString& srcValue = attributeWithoutSynchronization(srcAttr);
113 if (!srcValue.isNull())
114 setLocation(stripLeadingAndTrailingHTMLSpaces(srcValue));
115 } else
116 setLocation("about:srcdoc");
117 } else if (name == srcAttr && !hasAttributeWithoutSynchronization(srcdocAttr))
118 setLocation(stripLeadingAndTrailingHTMLSpaces(value));
119 else
120 HTMLFrameOwnerElement::parseAttribute(name, value);
121}

131void HTMLFrameElementBase::didFinishInsertingNode()
132{
133 if (!isConnected())
134 return;
135
136 // DocumentFragments don't kick off any loads.
137 if (!document().frame())
138 return;
139
140 if (!SubframeLoadingDisabler::canLoadFrame(*this))
141 return;
142
143 if (!renderer())
144 invalidateStyleAndRenderersForSubtree();
145 openURL();
146}

148void HTMLFrameElementBase::didAttachRenderers()
149{
150 if (RenderWidget* part = renderWidget()) {
151 if (RefPtr<Frame> frame = contentFrame())
152 part->setWidget(frame->view());
153 }
154}

156URL HTMLFrameElementBase::location() const
157{
158 if (hasAttributeWithoutSynchronization(srcdocAttr))
159 return aboutSrcDocURL();
160 return document().completeURL(attributeWithoutSynchronization(srcAttr));
161}
162
163void HTMLFrameElementBase::setLocation(const String& str)
164{
165 if (document().settings().needsAcrobatFrameReloadingQuirk() && m_frameURL == str)
166 return;
167
168 m_frameURL = AtomString(str);
169
170 if (isConnected())
171 openURL(LockHistory::No, LockBackForwardList::No);
172}

174void HTMLFrameElementBase::setLocation(JSC::JSGlobalObject& state, const String& newLocation)
175{
176 if (WTF::protocolIsJavaScript(stripLeadingAndTrailingHTMLSpaces(newLocation))) {
177 if (!BindingSecurity::shouldAllowAccessToNode(state, contentDocument()))
178 return;
179 }
180
181 setLocation(newLocation);
182}

184bool HTMLFrameElementBase::supportsFocus() const
185{
186 return true;
187}
188
189void HTMLFrameElementBase::setFocus(bool received, FocusVisibility visibility)
190{
191 HTMLFrameOwnerElement::setFocus(received, visibility);
192 if (Page* page = document().page()) {
193 if (received)
194 page->focusController().setFocusedFrame(contentFrame());
195 else if (page->focusController().focusedFrame() == contentFrame()) // Focus may have already been given to another frame, don't take it away.
196 page->focusController().setFocusedFrame(0);
197 }
198}

226ScrollbarMode HTMLFrameElementBase::scrollingMode() const
227{
228 auto scrollingAttribute = attributeWithoutSynchronization(scrollingAttr);
229 return equalLettersIgnoringASCIICase(scrollingAttribute, "no")
230 || equalLettersIgnoringASCIICase(scrollingAttribute, "noscroll")
231 || equalLettersIgnoringASCIICase(scrollingAttribute, "off")
232 ? ScrollbarAlwaysOff : ScrollbarAuto;
233}
