44RenderWidget* HTMLFrameOwnerElement::renderWidget() const
45{
46 // HTMLObjectElement and HTMLEmbedElement may return arbitrary renderers
47 // when using fallback content.
48 if (!is<RenderWidget>(renderer()))
49 return nullptr;
50 return downcast<RenderWidget>(renderer());
51}

53void HTMLFrameOwnerElement::setContentFrame(Frame& frame)
54{
55 // Make sure we will not end up with two frames referencing the same owner element.
56 ASSERT(!m_contentFrame || m_contentFrame->ownerElement() != this);
57 // Disconnected frames should not be allowed to load.
58 ASSERT(isConnected());
59 m_contentFrame = makeWeakPtr(frame);
60
61 for (RefPtr<ContainerNode> node = this; node; node = node->parentOrShadowHostNode())
62 node->incrementConnectedSubframeCount();
63}
64
65void HTMLFrameOwnerElement::clearContentFrame()
66{
67 if (!m_contentFrame)
68 return;
69
70 m_contentFrame = nullptr;
71
72 for (RefPtr<ContainerNode> node = this; node; node = node->parentOrShadowHostNode())
73 node->decrementConnectedSubframeCount();
74}
75
76void HTMLFrameOwnerElement::disconnectContentFrame()
77{
78 if (RefPtr<Frame> frame = m_contentFrame.get()) {
79 frame->loader().frameDetached();
80 frame->disconnectOwnerElement();
81 }
82}
83
84HTMLFrameOwnerElement::~HTMLFrameOwnerElement()
85{
86 if (m_contentFrame)
87 m_contentFrame->disconnectOwnerElement();
88}

110ExceptionOr<Document&> HTMLFrameOwnerElement::getSVGDocument() const
111{
112 auto* document = contentDocument();
113 if (is<SVGDocument>(document))
114 return *document;
115 // Spec: http://www.w3.org/TR/SVG/struct.html#InterfaceGetSVGDocument
116 return Exception { NotSupportedError };
117}

119void HTMLFrameOwnerElement::scheduleInvalidateStyleAndLayerComposition()
120{
121 if (Style::postResolutionCallbacksAreSuspended()) {
122 RefPtr<HTMLFrameOwnerElement> element = this;
123 Style::queuePostResolutionCallback([element] {
124 element->invalidateStyleAndLayerComposition();
125 });
126 } else
127 invalidateStyleAndLayerComposition();
128}

130bool HTMLFrameOwnerElement::isProhibitedSelfReference(const URL& completeURL) const
131{
132 // We allow one level of self-reference because some websites depend on that, but we don't allow more than one.
133 bool foundOneSelfReference = false;
134 for (auto* frame = document().frame(); frame; frame = frame->tree().parent()) {
135 if (equalIgnoringFragmentIdentifier(frame->document()->url(), completeURL)) {
136 if (foundOneSelfReference)
137 return true;
138 foundOneSelfReference = true;
139 }
140 }
141 return false;
142}

144bool SubframeLoadingDisabler::canLoadFrame(HTMLFrameOwnerElement& owner)
145{
146 for (RefPtr<ContainerNode> node = &owner; node; node = node->parentOrShadowHostNode()) {
147 if (disabledSubtreeRoots().contains(node.get()))
148 return false;
149 }
150 return true;
151}
