52HTMLFrameSetElement::HTMLFrameSetElement(const QualifiedName& tagName, Document& document)
53 : HTMLElement(tagName, document)
54 , m_totalRows(1)
55 , m_totalCols(1)
56 , m_border(6)
57 , m_borderSet(false)
58 , m_borderColorSet(false)
59 , m_frameborder(true)
60 , m_frameborderSet(false)
61 , m_noresize(false)
62{
63 ASSERT(hasTagName(framesetTag));
64 setHasCustomStyleResolveCallbacks();
65}

87void HTMLFrameSetElement::parseAttribute(const QualifiedName& name, const AtomString& value)
88{
89 if (name == rowsAttr) {
90 // FIXME: What is the right thing to do when removing this attribute?
91 // Why not treat it the same way we treat setting it to the empty string?
92 if (!value.isNull()) {
93 m_rowLengths = newLengthArray(value.string(), m_totalRows);
94 // FIXME: Would be nice to optimize the case where m_rowLengths did not change.
95 invalidateStyleForSubtree();
96 }
97 return;
98 }
99
100 if (name == colsAttr) {
101 // FIXME: What is the right thing to do when removing this attribute?
102 // Why not treat it the same way we treat setting it to the empty string?
103 if (!value.isNull()) {
104 m_colLengths = newLengthArray(value.string(), m_totalCols);
105 // FIXME: Would be nice to optimize the case where m_colLengths did not change.
106 invalidateStyleForSubtree();
107 }
108 return;
109 }
110
111 if (name == frameborderAttr) {
112 if (!value.isNull()) {
113 if (equalLettersIgnoringASCIICase(value, "no") || value == "0") {
114 m_frameborder = false;
115 m_frameborderSet = true;
116 } else if (equalLettersIgnoringASCIICase(value, "yes") || value == "1") {
117 m_frameborderSet = true;
118 }
119 } else {
120 m_frameborder = false;
121 m_frameborderSet = false;
122 }
123 // FIXME: Do we need to trigger repainting?
124 return;
125 }
126
127 if (name == noresizeAttr) {
128 // FIXME: This should set m_noresize to false if the value is null.
129 m_noresize = true;
130 return;
131 }
132
133 if (name == borderAttr) {
134 if (!value.isNull()) {
135 m_border = parseHTMLInteger(value).value_or(0);
136 m_borderSet = true;
137 } else
138 m_borderSet = false;
139 // FIXME: Do we need to trigger repainting?
140 return;
141 }
142
143 if (name == bordercolorAttr) {
144 m_borderColorSet = !value.isEmpty();
145 // FIXME: Clearly wrong: This can overwrite the value inherited from the parent frameset.
146 // FIXME: Do we need to trigger repainting?
147 return;
148 }
149
150 auto& eventName = HTMLBodyElement::eventNameForWindowEventHandlerAttribute(name);
151 if (!eventName.isNull()) {
152 document().setWindowAttributeEventListener(eventName, name, value, mainThreadNormalWorld());
153 return;
154 }
155
156 HTMLElement::parseAttribute(name, value);
157}

159bool HTMLFrameSetElement::rendererIsNeeded(const RenderStyle& style)
160{
161 // For compatibility, frames render even when display: none is set.
162 // However, we delay creating a renderer until stylesheets have loaded.
163 return !style.isNotFinal();
164}
165
166RenderPtr<RenderElement> HTMLFrameSetElement::createElementRenderer(RenderStyle&& style, const RenderTreePosition&)
167{
168 if (style.hasContent())
169 return RenderElement::createFor(*this, WTFMove(style));
170
171 return createRenderer<RenderFrameSet>(*this, WTFMove(style));
172}

179void HTMLFrameSetElement::willAttachRenderers()
180{
181 // Inherit default settings from parent frameset.
182 // FIXME: This is not dynamic.
183 const auto containingFrameSet = findContaining(this);
184 if (!containingFrameSet)
185 return;
186 if (!m_frameborderSet)
187 m_frameborder = containingFrameSet->hasFrameBorder();
188 if (m_frameborder) {
189 if (!m_borderSet)
190 m_border = containingFrameSet->border();
191 if (!m_borderColorSet)
192 m_borderColorSet = containingFrameSet->hasBorderColor();
193 }
194 if (!m_noresize)
195 m_noresize = containingFrameSet->noResize();
196}

198void HTMLFrameSetElement::defaultEventHandler(Event& event)
199{
200 if (is<MouseEvent>(event) && !m_noresize && is<RenderFrameSet>(renderer())) {
201 if (downcast<RenderFrameSet>(*renderer()).userResize(downcast<MouseEvent>(event))) {
202 event.setDefaultHandled();
203 return;
204 }
205 }
206 HTMLElement::defaultEventHandler(event);
207}

212 renderer()->setNeedsLayout();

226WindowProxy* HTMLFrameSetElement::namedItem(const AtomString& name)
227{
228 auto frameElement = makeRefPtr(children()->namedItem(name));
229 if (!is<HTMLFrameElement>(frameElement))
230 return nullptr;
231
232 return downcast<HTMLFrameElement>(*frameElement).contentWindow();
233}
