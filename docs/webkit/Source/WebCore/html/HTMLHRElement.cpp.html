56bool HTMLHRElement::hasPresentationalHintsForAttribute(const QualifiedName& name) const
57{
58 if (name == widthAttr || name == colorAttr || name == noshadeAttr || name == sizeAttr)
59 return true;
60 return HTMLElement::hasPresentationalHintsForAttribute(name);
61}
62
63void HTMLHRElement::collectPresentationalHintsForAttribute(const QualifiedName& name, const AtomString& value, MutableStyleProperties& style)
64{
65 if (name == alignAttr) {
66 if (equalLettersIgnoringASCIICase(value, "left")) {
67 addPropertyToPresentationalHintStyle(style, CSSPropertyMarginLeft, 0, CSSUnitType::CSS_PX);
68 addPropertyToPresentationalHintStyle(style, CSSPropertyMarginRight, CSSValueAuto);
69 } else if (equalLettersIgnoringASCIICase(value, "right")) {
70 addPropertyToPresentationalHintStyle(style, CSSPropertyMarginLeft, CSSValueAuto);
71 addPropertyToPresentationalHintStyle(style, CSSPropertyMarginRight, 0, CSSUnitType::CSS_PX);
72 } else {
73 addPropertyToPresentationalHintStyle(style, CSSPropertyMarginLeft, CSSValueAuto);
74 addPropertyToPresentationalHintStyle(style, CSSPropertyMarginRight, CSSValueAuto);
75 }
76 } else if (name == widthAttr) {
77 if (auto valueInteger = parseHTMLInteger(value); valueInteger && !*valueInteger)
78 addPropertyToPresentationalHintStyle(style, CSSPropertyWidth, 1, CSSUnitType::CSS_PX);
79 else
80 addHTMLLengthToStyle(style, CSSPropertyWidth, value);
81 } else if (name == colorAttr) {
82 addPropertyToPresentationalHintStyle(style, CSSPropertyBorderStyle, CSSValueSolid);
83 addHTMLColorToStyle(style, CSSPropertyBorderColor, value);
84 addHTMLColorToStyle(style, CSSPropertyBackgroundColor, value);
85 } else if (name == noshadeAttr) {
86 if (!hasAttributeWithoutSynchronization(colorAttr)) {
87 addPropertyToPresentationalHintStyle(style, CSSPropertyBorderStyle, CSSValueSolid);
88
89 RefPtr<CSSPrimitiveValue> darkGrayValue = CSSValuePool::singleton().createColorValue(Color::darkGray);
90 style.setProperty(CSSPropertyBorderColor, darkGrayValue);
91 style.setProperty(CSSPropertyBackgroundColor, darkGrayValue);
92 }
93 } else if (name == sizeAttr) {
94 int size = parseHTMLInteger(value).value_or(0);
95 if (size <= 1)
96 addPropertyToPresentationalHintStyle(style, CSSPropertyBorderBottomWidth, 0, CSSUnitType::CSS_PX);
97 else
98 addPropertyToPresentationalHintStyle(style, CSSPropertyHeight, size - 2, CSSUnitType::CSS_PX);
99 } else
100 HTMLElement::collectPresentationalHintsForAttribute(name, value, style);
101}
