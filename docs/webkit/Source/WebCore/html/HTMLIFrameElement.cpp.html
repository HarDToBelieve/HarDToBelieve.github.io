62DOMTokenList& HTMLIFrameElement::sandbox()
63{

72bool HTMLIFrameElement::hasPresentationalHintsForAttribute(const QualifiedName& name) const
73{
74 if (name == widthAttr || name == heightAttr || name == frameborderAttr)
75 return true;
76 return HTMLFrameElementBase::hasPresentationalHintsForAttribute(name);
77}

79void HTMLIFrameElement::collectPresentationalHintsForAttribute(const QualifiedName& name, const AtomString& value, MutableStyleProperties& style)
80{
81 if (name == widthAttr)
82 addHTMLLengthToStyle(style, CSSPropertyWidth, value);
83 else if (name == heightAttr)
84 addHTMLLengthToStyle(style, CSSPropertyHeight, value);
85 else if (name == alignAttr)
86 applyAlignmentAttributeToStyle(value, style);
87 else if (name == frameborderAttr) {
88 // Frame border doesn't really match the HTML4 spec definition for iframes. It simply adds
89 // a presentational hint that the border should be off if set to zero.
90 if (!parseHTMLInteger(value).value_or(0)) {
91 // Add a rule that nulls out our border width.
92 addPropertyToPresentationalHintStyle(style, CSSPropertyBorderWidth, 0, CSSUnitType::CSS_PX);
93 }
94 } else
95 HTMLFrameElementBase::collectPresentationalHintsForAttribute(name, value, style);
96}

98void HTMLIFrameElement::parseAttribute(const QualifiedName& name, const AtomString& value)
99{
100 if (name == sandboxAttr) {
101 if (m_sandbox)
102 m_sandbox->associatedAttributeValueChanged(value);
103
104 String invalidTokens;
105 setSandboxFlags(value.isNull() ? SandboxNone : SecurityContext::parseSandboxPolicy(value, invalidTokens));
106 if (!invalidTokens.isNull())
107 document().addConsoleMessage(MessageSource::Other, MessageLevel::Error, "Error while parsing the 'sandbox' attribute: " + invalidTokens);
108 } else if (name == allowAttr || name == allowfullscreenAttr || name == webkitallowfullscreenAttr) {
109 m_featurePolicy = std::nullopt;
110 } else if (name == loadingAttr) {
111 // Allow loading=eager to load the frame immediately if the lazy load was started, but
112 // do not allow the reverse situation since the eager load is already started.
113 if (m_lazyLoadFrameObserver && !equalLettersIgnoringASCIICase(value, "lazy")) {
114 m_lazyLoadFrameObserver->unobserve();
115 loadDeferredFrame();
116 }
117 } else
118 HTMLFrameElementBase::parseAttribute(name, value);
119}

141ReferrerPolicy HTMLIFrameElement::referrerPolicy() const
142{
143 if (m_lazyLoadFrameObserver)
144 return m_lazyLoadFrameObserver->referrerPolicy();
145 if (document().settings().referrerPolicyAttributeEnabled())
146 return parseReferrerPolicy(attributeWithoutSynchronization(referrerpolicyAttr), ReferrerPolicySource::ReferrerPolicyAttribute).value_or(ReferrerPolicy::EmptyString);
147 return ReferrerPolicy::EmptyString;
148}

157const AtomString& HTMLIFrameElement::loadingForBindings() const
158{
159 static MainThreadNeverDestroyed<const AtomString> eager("eager", AtomString::ConstructFromLiteral);
160 static MainThreadNeverDestroyed<const AtomString> lazy("lazy", AtomString::ConstructFromLiteral);
161 return equalLettersIgnoringASCIICase(attributeWithoutSynchronization(HTMLNames::loadingAttr), "lazy") ? lazy : eager;
162}

169static bool isFrameLazyLoadable(const Document& document, const URL& completeURL, const AtomString& loadingAttributeValue)
170{
171 if (!completeURL.protocolIsInHTTPFamily())
172 return false;
173
174 if (!document.frame() || !document.frame()->script().canExecuteScripts(NotAboutToExecuteScript))
175 return false;
176
177 return equalLettersIgnoringASCIICase(loadingAttributeValue, "lazy");
178}
179
180bool HTMLIFrameElement::shouldLoadFrameLazily()
181{
182 if (!m_lazyLoadFrameObserver && document().settings().lazyIframeLoadingEnabled()) {
183 URL completeURL = document().completeURL(frameURL());
184 if (isFrameLazyLoadable(document(), completeURL, attributeWithoutSynchronization(HTMLNames::loadingAttr))) {
185 auto currentReferrerPolicy = referrerPolicy();
186 lazyLoadFrameObserver().observe(completeURL.string(), currentReferrerPolicy);
187 return true;
188 }
189 }
190 return false;
191}
192
193bool HTMLIFrameElement::isLazyLoadObserverActive() const
194{
195 return !!m_lazyLoadFrameObserver;
196}
197
198void HTMLIFrameElement::loadDeferredFrame()
199{
200 AtomString currentURL = frameURL();
201 setFrameURL(m_lazyLoadFrameObserver->frameURL());
202 openURL();
203 setFrameURL(currentURL);
204 m_lazyLoadFrameObserver = nullptr;
205}
206
207LazyLoadFrameObserver& HTMLIFrameElement::lazyLoadFrameObserver()
208{
209 if (!m_lazyLoadFrameObserver)
210 m_lazyLoadFrameObserver = makeUnique<LazyLoadFrameObserver>(*this);
211 return *m_lazyLoadFrameObserver;
212}
