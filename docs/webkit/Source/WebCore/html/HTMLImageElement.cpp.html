116void HTMLImageElement::collectPresentationalHintsForAttribute(const QualifiedName& name, const AtomString& value, MutableStyleProperties& style)
117{
118 if (name == widthAttr) {
119 addHTMLMultiLengthToStyle(style, CSSPropertyWidth, value);
120 applyAspectRatioFromWidthAndHeightAttributesToStyle(value, attributeWithoutSynchronization(heightAttr), style);
121 } else if (name == heightAttr) {
122 addHTMLMultiLengthToStyle(style, CSSPropertyHeight, value);
123 applyAspectRatioFromWidthAndHeightAttributesToStyle(attributeWithoutSynchronization(widthAttr), value, style);
124 } else if (name == borderAttr)
125 applyBorderAttributeToStyle(value, style);
126 else if (name == vspaceAttr) {
127 addHTMLLengthToStyle(style, CSSPropertyMarginTop, value);
128 addHTMLLengthToStyle(style, CSSPropertyMarginBottom, value);
129 } else if (name == hspaceAttr) {
130 addHTMLLengthToStyle(style, CSSPropertyMarginLeft, value);
131 addHTMLLengthToStyle(style, CSSPropertyMarginRight, value);
132 } else if (name == alignAttr)
133 applyAlignmentAttributeToStyle(value, style);
134 else if (name == valignAttr)
135 addPropertyToPresentationalHintStyle(style, CSSPropertyVerticalAlign, value);
136 else
137 HTMLElement::collectPresentationalHintsForAttribute(name, value, style);
138}

140void HTMLImageElement::collectExtraStyleForPresentationalHints(MutableStyleProperties& style)
141{
142 if (!sourceElement())
143 return;
144 auto& widthAttrFromSource = sourceElement()->attributeWithoutSynchronization(widthAttr);
145 auto& heightAttrFromSource = sourceElement()->attributeWithoutSynchronization(heightAttr);
146 // If both width and height attributes of <source> is undefined, the style's value should not
147 // be overwritten. Otherwise, <souce> will overwrite it. I.e., if <source> only has one attribute
148 // defined, the other one and aspect-ratio shouldn't be set to auto.
149 if (widthAttrFromSource.isNull() && heightAttrFromSource.isNull())
150 return;
151
152 if (!widthAttrFromSource.isNull())
153 addHTMLLengthToStyle(style, CSSPropertyWidth, widthAttrFromSource);
154 else
155 addPropertyToPresentationalHintStyle(style, CSSPropertyWidth, CSSValueAuto);
156
157 if (!heightAttrFromSource.isNull())
158 addHTMLLengthToStyle(style, CSSPropertyHeight, heightAttrFromSource);
159 else
160 addPropertyToPresentationalHintStyle(style, CSSPropertyHeight, CSSValueAuto);
161
162 if (!widthAttrFromSource.isNull() && !heightAttrFromSource.isNull())
163 applyAspectRatioFromWidthAndHeightAttributesToStyle(widthAttrFromSource, heightAttrFromSource, style);
164 else
165 addPropertyToPresentationalHintStyle(style, CSSPropertyAspectRatio, CSSValueAuto);
166}

173void HTMLImageElement::setBestFitURLAndDPRFromImageCandidate(const ImageCandidate& candidate)
174{
175 m_bestFitImageURL = candidate.string.toAtomString();

177 if (candidate.density >= 0)

183ImageCandidate HTMLImageElement::bestFitSourceFromPictureElement()
184{
185 auto picture = makeRefPtr(pictureElement());
186 if (!picture)
187 return { };
188
189 ImageCandidate candidate;
190
191 for (RefPtr<Node> child = picture->firstChild(); child && child != this; child = child->nextSibling()) {
192 if (!is<HTMLSourceElement>(*child))
193 continue;
194 auto& source = downcast<HTMLSourceElement>(*child);
195
196 auto& srcset = source.attributeWithoutSynchronization(srcsetAttr);
197 if (srcset.isEmpty())
198 continue;
199
200 auto& typeAttribute = source.attributeWithoutSynchronization(typeAttr);
201 if (!typeAttribute.isNull()) {
202 String type = typeAttribute.string();
203 type.truncate(type.find(';'));
204 type = stripLeadingAndTrailingHTMLSpaces(type);
205 if (!type.isEmpty() && !MIMETypeRegistry::isSupportedImageVideoOrSVGMIMEType(type))
206 continue;
207 }
208
209 auto documentElement = makeRefPtr(document().documentElement());
210 MediaQueryEvaluator evaluator { document().printing() ? "print" : "screen", document(), documentElement ? documentElement->computedStyle() : nullptr };
211 auto* queries = source.parsedMediaAttribute(document());
212 LOG(MediaQueries, "HTMLImageElement %p bestFitSourceFromPictureElement evaluating media queries", this);
213
214 auto evaluation = !queries || evaluator.evaluate(*queries, &m_mediaQueryDynamicResults);
215 if (!evaluation)
216 continue;
217
218 SizesAttributeParser sizesParser(source.attributeWithoutSynchronization(sizesAttr).string(), document(), &m_mediaQueryDynamicResults);
219 auto sourceSize = sizesParser.length();
220
221 candidate = bestFitSourceForImageAttributes(document().deviceScaleFactor(), nullAtom(), srcset, sourceSize);
222 if (!candidate.isEmpty()) {
223 setSourceElement(&source);
224 break;
225 }
226 }
227
228 return candidate;
229}

231void HTMLImageElement::evaluateDynamicMediaQueryDependencies()
232{
233 auto documentElement = makeRefPtr(document().documentElement());
234 MediaQueryEvaluator evaluator { document().printing() ? "print" : "screen", document(), documentElement ? documentElement->computedStyle() : nullptr };
235
236 if (!evaluator.evaluateForChanges(m_mediaQueryDynamicResults))
237 return;
238
239 selectImageSource(RelevantMutation::No);
240}

242void HTMLImageElement::selectImageSource(RelevantMutation relevantMutation)
243{
244 m_mediaQueryDynamicResults = { };
245 document().removeDynamicMediaQueryDependentImage(*this);
246
247 // First look for the best fit source from our <picture> parent if we have one.
248 ImageCandidate candidate = bestFitSourceFromPictureElement();
249 if (candidate.isEmpty()) {
250 setSourceElement(nullptr);
251 // If we don't have a <picture> or didn't find a source, then we use our own attributes.
252 SizesAttributeParser sizesParser(attributeWithoutSynchronization(sizesAttr).string(), document(), &m_mediaQueryDynamicResults);
253 auto sourceSize = sizesParser.length();
254 candidate = bestFitSourceForImageAttributes(document().deviceScaleFactor(), attributeWithoutSynchronization(srcAttr), attributeWithoutSynchronization(srcsetAttr), sourceSize);
255 }
256 setBestFitURLAndDPRFromImageCandidate(candidate);
257 m_imageLoader->updateFromElementIgnoringPreviousError(relevantMutation);
258
259 if (!m_mediaQueryDynamicResults.isEmpty())
260 document().addDynamicMediaQueryDependentImage(*this);
261}

276void HTMLImageElement::attributeChanged(const QualifiedName& name, const AtomString& oldValue, const AtomString& newValue, AttributeModificationReason reason)
277{
278 HTMLElement::attributeChanged(name, oldValue, newValue, reason);
279
280 if (name == referrerpolicyAttr && document().settings().referrerPolicyAttributeEnabled()) {
281 auto oldReferrerPolicy = parseReferrerPolicy(oldValue, ReferrerPolicySource::ReferrerPolicyAttribute).value_or(ReferrerPolicy::EmptyString);
282 auto newReferrerPolicy = parseReferrerPolicy(newValue, ReferrerPolicySource::ReferrerPolicyAttribute).value_or(ReferrerPolicy::EmptyString);
283 if (oldReferrerPolicy != newReferrerPolicy)
284 m_imageLoader->updateFromElementIgnoringPreviousError(RelevantMutation::Yes);
285 } else if (name == crossoriginAttr) {
286 if (parseCrossoriginState(oldValue) != parseCrossoriginState(newValue))
287 m_imageLoader->updateFromElementIgnoringPreviousError(RelevantMutation::Yes);
288 }
289}

291void HTMLImageElement::parseAttribute(const QualifiedName& name, const AtomString& value)
292{
293 if (name == altAttr) {
294 if (is<RenderImage>(renderer()))
295 downcast<RenderImage>(*renderer()).updateAltText();
296 } else if (name == srcAttr || name == srcsetAttr || name == sizesAttr)
297 selectImageSource(RelevantMutation::Yes);
298 else if (name == usemapAttr) {
299 if (isInTreeScope() && !m_parsedUsemap.isNull())
300 treeScope().removeImageElementByUsemap(*m_parsedUsemap.impl(), *this);
301
302 m_parsedUsemap = parseHTMLHashNameReference(value);
303
304 if (isInTreeScope() && !m_parsedUsemap.isNull())
305 treeScope().addImageElementByUsemap(*m_parsedUsemap.impl(), *this);
306 } else if (name == compositeAttr) {
307 // FIXME: images don't support blend modes in their compositing attribute.
308 BlendMode blendOp = BlendMode::Normal;
309 if (!parseCompositeAndBlendOperator(value, m_compositeOperator, blendOp))
310 m_compositeOperator = CompositeOperator::SourceOver;
311 } else if (name == loadingAttr) {
312 // No action needed for eager to lazy transition.
313 if (!hasLazyLoadableAttributeValue(value))
314 loadDeferredImage();
315 } else {
316 if (name == nameAttr) {
317 bool willHaveName = !value.isEmpty();
318 if (m_hadNameBeforeAttributeChanged != willHaveName && isConnected() && !isInShadowTree() && is<HTMLDocument>(document())) {
319 HTMLDocument& document = downcast<HTMLDocument>(this->document());
320 const AtomString& id = getIdAttribute();
321 if (!id.isEmpty() && id != getNameAttribute()) {
322 if (willHaveName)
323 document.addDocumentNamedItem(*id.impl(), *this);
324 else
325 document.removeDocumentNamedItem(*id.impl(), *this);
326 }
327 }
328 m_hadNameBeforeAttributeChanged = willHaveName;
329 }
330 HTMLElement::parseAttribute(name, value);
331 }

372void HTMLImageElement::didAttachRenderers()
373{
374 if (!is<RenderImage>(renderer()))
375 return;
376 if (m_imageLoader->hasPendingBeforeLoadEvent())
377 return;
378
379 auto& renderImage = downcast<RenderImage>(*renderer());
380 RenderImageResource& renderImageResource = renderImage.imageResource();
381 if (renderImageResource.cachedImage())
382 return;
383 renderImageResource.setCachedImage(m_imageLoader->image());
384
385 // If we have no image at all because we have no src attribute, set
386 // image height and width for the alt text instead.
387 if (!m_imageLoader->image() && !renderImageResource.cachedImage())
388 renderImage.setImageSizeForAltText();
389}

391Node::InsertedIntoAncestorResult HTMLImageElement::insertedIntoAncestor(InsertionType insertionType, ContainerNode& parentOfInsertedTree)
392{
393 if (m_formSetByParser) {
394 m_form = WTFMove(m_formSetByParser);
395 m_form->registerImgElement(this);
396 }
397
398 if (m_form && rootElement() != m_form->rootElement()) {
399 m_form->removeImgElement(this);
400 m_form = nullptr;
401 }
402
403 if (!m_form) {
404 if (auto* newForm = HTMLFormElement::findClosestFormAncestor(*this)) {
405 m_form = makeWeakPtr(newForm);
406 newForm->registerImgElement(this);
407 }
408 }
409
410 // Insert needs to complete first, before we start updating the loader. Loader dispatches events which could result
411 // in callbacks back to this node.
412 Node::InsertedIntoAncestorResult insertNotificationRequest = HTMLElement::insertedIntoAncestor(insertionType, parentOfInsertedTree);
413
414 if (insertionType.treeScopeChanged && !m_parsedUsemap.isNull())
415 treeScope().addImageElementByUsemap(*m_parsedUsemap.impl(), *this);
416
417 if (is<HTMLPictureElement>(&parentOfInsertedTree) && &parentOfInsertedTree == parentElement()) {
418 // FIXME: When the hack in HTMLConstructionSite::createHTMLElementOrFindCustomElementInterface to eagerly call setPictureElement is removed, we can just assert !pictureElement().
419 ASSERT(!pictureElement() || pictureElement() == &parentOfInsertedTree);
420 setPictureElement(&downcast<HTMLPictureElement>(parentOfInsertedTree));
421 selectImageSource(RelevantMutation::Yes);
422 return insertNotificationRequest;
423 }
424
425 // If we have been inserted from a renderer-less document,
426 // our loader may have not fetched the image, so do it now.
427 if (insertionType.connectedToDocument && !m_imageLoader->image())
428 m_imageLoader->updateFromElement();
429
430 return insertNotificationRequest;
431}

433void HTMLImageElement::removedFromAncestor(RemovalType removalType, ContainerNode& oldParentOfRemovedTree)
434{
435 if (m_form)
436 m_form->removeImgElement(this);
437
438 if (removalType.treeScopeChanged && !m_parsedUsemap.isNull())
439 oldParentOfRemovedTree.treeScope().removeImageElementByUsemap(*m_parsedUsemap.impl(), *this);
440
441 if (is<HTMLPictureElement>(oldParentOfRemovedTree) && !parentElement()) {
442 ASSERT(pictureElement() == &oldParentOfRemovedTree);
443 setPictureElement(nullptr);
444 selectImageSource(RelevantMutation::Yes);
445 }
446
447 m_form = nullptr;
448 HTMLElement::removedFromAncestor(removalType, oldParentOfRemovedTree);
449}

461unsigned HTMLImageElement::width(bool ignorePendingStylesheets)
462{
463 if (!renderer()) {
464 // check the attribute first for an explicit pixel value
465 auto optionalWidth = parseHTMLNonNegativeInteger(attributeWithoutSynchronization(widthAttr));
466 if (optionalWidth)
467 return optionalWidth.value();
468
469 // if the image is available, use its width
470 if (m_imageLoader->image())
471 return m_imageLoader->image()->imageSizeForRenderer(renderer(), 1.0f).width().toUnsigned();
472 }
473
474 if (ignorePendingStylesheets)
475 document().updateLayoutIgnorePendingStylesheets();
476 else
477 document().updateLayout();
478
479 RenderBox* box = renderBox();
480 if (!box)
481 return 0;
482 LayoutRect contentRect = box->contentBoxRect();
483 return adjustForAbsoluteZoom(snappedIntRect(contentRect).width(), *box);
484}

555String HTMLImageElement::completeURLsInAttributeValue(const URL& base, const Attribute& attribute) const
556{
557 if (attribute.name() == srcsetAttr) {
558 Vector<ImageCandidate> imageCandidates = parseImageCandidatesFromSrcsetAttribute(StringView(attribute.value()));
559 StringBuilder result;
560 for (const auto& candidate : imageCandidates) {
561 if (&candidate != &imageCandidates[0])
562 result.append(", ");
563 result.append(URL(base, candidate.string.toString()).string());
564 if (candidate.density != UninitializedDescriptor)
565 result.append(' ', candidate.density, 'x');
566 if (candidate.resourceWidth != UninitializedDescriptor)
567 result.append(' ', candidate.resourceWidth, 'w');
568 }
569 return result.toString();
570 }
571 return HTMLElement::completeURLsInAttributeValue(base, attribute);
572}

668bool HTMLImageElement::isServerMap() const
669{
670 if (!hasAttributeWithoutSynchronization(ismapAttr))
671 return false;
672
673 const AtomString& usemap = attributeWithoutSynchronization(usemapAttr);
674
675 // If the usemap attribute starts with '#', it refers to a map element in the document.
676 if (usemap.string()[0] == '#')
677 return false;
678
679 return document().completeURL(stripLeadingAndTrailingHTMLSpaces(usemap)).isEmpty();
680}

687String HTMLImageElement::crossOrigin() const
