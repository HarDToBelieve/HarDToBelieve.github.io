168HTMLInputElement::~HTMLInputElement()
169{
170 if (needsSuspensionCallback())
171 document().unregisterForDocumentSuspensionCallbacks(*this);
172
173 // Need to remove form association while this is still an HTMLInputElement
174 // so that virtual functions are called correctly.
175 setForm(nullptr);
176
177 // This is needed for a radio button that was not in a form, and also for
178 // a radio button that was in a form. The call to setForm(nullptr) above
179 // actually adds the button to the document groups in the latter case.
180 // That is inelegant, but harmless since we remove it here.
181 if (isRadioButton())
182 treeScope().radioButtonGroups().removeButton(*this);
183
184#if ENABLE(TOUCH_EVENTS)
185 if (m_hasTouchEventHandler)
186 document().didRemoveEventTargetNode(*this);
187#endif
188}

269bool HTMLInputElement::isValidValue(const String& value) const
270{
271 if (!m_inputType->canSetStringValue()) {
272 ASSERT_NOT_REACHED();
273 return false;
274 }
275 return !m_inputType->typeMismatchFor(value)
276 && !m_inputType->stepMismatch(value)
277 && !m_inputType->rangeUnderflow(value)
278 && !m_inputType->rangeOverflow(value)
279 && !tooShort(value, IgnoreDirtyFlag)
280 && !tooLong(value, IgnoreDirtyFlag)
281 && !m_inputType->patternMismatch(value)
282 && !m_inputType->valueMissing(value);
283}

315bool HTMLInputElement::tooShort(StringView value, NeedsToCheckDirtyFlag check) const
316{
317 if (!supportsMinLength())
318 return false;
319
320 int min = minLength();
321 if (min <= 0)
322 return false;
323
324 if (check == CheckDirtyFlag) {
325 // Return false for the default value or a value set by a script even if
326 // it is shorter than minLength.
327 if (!hasDirtyValue() || !m_wasModifiedByUser)
328 return false;
329 }
330
331 // The empty string is excluded from tooShort validation.
332 if (value.isEmpty())
333 return false;
334
335 // FIXME: The HTML specification says that the "number of characters" is measured using code-unit length.
336 return numGraphemeClusters(value) < static_cast<unsigned>(min);
337}
338
339bool HTMLInputElement::tooLong(StringView value, NeedsToCheckDirtyFlag check) const
340{
341 if (!supportsMaxLength())
342 return false;
343 unsigned max = effectiveMaxLength();
344 if (check == CheckDirtyFlag) {
345 // Return false for the default value or a value set by a script even if
346 // it is longer than maxLength.
347 if (!hasDirtyValue() || !m_wasModifiedByUser)
348 return false;
349 }
350 // FIXME: The HTML specification says that the "number of characters" is measured using code-unit length.
351 return numGraphemeClusters(value) > max;
352}

497void HTMLInputElement::setDefaultSelectionAfterFocus(SelectionRestorationMode restorationMode, SelectionRevealMode revealMode)
498{
499 ASSERT(isTextField());
500 int start = 0;
501 auto direction = SelectionHasNoDirection;
502 auto* frame = document().frame();
503 if (frame && frame->editor().behavior().shouldMoveSelectionToEndWhenFocusingTextInput()) {
504 start = std::numeric_limits<int>::max();
505 direction = SelectionHasForwardDirection;
506 }
507 int end = restorationMode == SelectionRestorationMode::PlaceCaretAtStart ? start : std::numeric_limits<int>::max();
508 setSelectionRange(start, end, direction, revealMode, Element::defaultFocusTextStateChangeIntent());
509}

555void HTMLInputElement::updateType()
556{
557 ASSERT(m_inputType);
558 auto newType = InputType::create(*this, attributeWithoutSynchronization(typeAttr));
559 m_hasType = true;
560 if (m_inputType->formControlType() == newType->formControlType())
561 return;
562
563 removeFromRadioButtonGroup();
564 resignStrongPasswordAppearance();
565
566 bool didStoreValue = m_inputType->storesValueSeparateFromAttribute();
567 bool willStoreValue = newType->storesValueSeparateFromAttribute();
568 bool neededSuspensionCallback = needsSuspensionCallback();
569 bool didRespectHeightAndWidth = m_inputType->shouldRespectHeightAndWidthAttributes();
570 bool wasSuccessfulSubmitButtonCandidate = m_inputType->canBeSuccessfulSubmitButton();
571
572 if (didStoreValue && !willStoreValue && hasDirtyValue()) {
573 setAttributeWithoutSynchronization(valueAttr, m_valueIfDirty);
574 m_valueIfDirty = String();
575 }
576
577 m_inputType->destroyShadowSubtree();
578 m_inputType->detachFromElement();
579
580 m_inputType = WTFMove(newType);
581 if (m_inputType->needsShadowSubtree()) {
582 ensureUserAgentShadowRoot();
583 createShadowSubtreeAndUpdateInnerTextElementEditability();
584 }
585
586 updateWillValidateAndValidity();
587
588 if (!didStoreValue && willStoreValue)
589 m_valueIfDirty = sanitizeValue(attributeWithoutSynchronization(valueAttr));
590 else
591 updateValueIfNeeded();
592
593 setFormControlValueMatchesRenderer(false);
594 m_inputType->updateInnerTextValue();
595
596 m_wasModifiedByUser = false;
597
598 if (neededSuspensionCallback)
599 unregisterForSuspensionCallbackIfNeeded();
600 else
601 registerForSuspensionCallbackIfNeeded();
602
603 if (didRespectHeightAndWidth != m_inputType->shouldRespectHeightAndWidthAttributes()) {
604 ASSERT(elementData());
605 // FIXME: We don't have the old attribute values so we pretend that we didn't have the old values.
606 if (const Attribute* height = findAttributeByName(heightAttr))
607 attributeChanged(heightAttr, nullAtom(), height->value());
608 if (const Attribute* width = findAttributeByName(widthAttr))
609 attributeChanged(widthAttr, nullAtom(), width->value());
610 if (const Attribute* align = findAttributeByName(alignAttr))
611 attributeChanged(alignAttr, nullAtom(), align->value());
612 }
613
614 if (form() && wasSuccessfulSubmitButtonCandidate != m_inputType->canBeSuccessfulSubmitButton())
615 form()->resetDefaultButton();
616
617 runPostTypeUpdateTasks();
618}

620inline void HTMLInputElement::runPostTypeUpdateTasks()
621{
622 ASSERT(m_inputType);
623#if ENABLE(TOUCH_EVENTS)
624 bool hasTouchEventHandler = m_inputType->hasTouchEventHandler();
625 if (hasTouchEventHandler != m_hasTouchEventHandler) {
626 if (hasTouchEventHandler)
627 document().didAddTouchEventHandler(*this);
628 else
629 document().didRemoveTouchEventHandler(*this);
630 m_hasTouchEventHandler = hasTouchEventHandler;
631 }
632#endif
633
634 if (renderer())
635 invalidateStyleAndRenderersForSubtree();
636
637 if (document().focusedElement() == this)
638 updateFocusAppearance(SelectionRestorationMode::RestoreOrSelectAll, SelectionRevealMode::Reveal);
639
640 setChangedSinceLastFormControlChangeEvent(false);
641
642 addToRadioButtonGroup();
643
644 updateValidity();
645}

702void HTMLInputElement::collectPresentationalHintsForAttribute(const QualifiedName& name, const AtomString& value, MutableStyleProperties& style)
703{
704 if (name == vspaceAttr) {
705 addHTMLLengthToStyle(style, CSSPropertyMarginTop, value);
706 addHTMLLengthToStyle(style, CSSPropertyMarginBottom, value);
707 } else if (name == hspaceAttr) {
708 addHTMLLengthToStyle(style, CSSPropertyMarginLeft, value);
709 addHTMLLengthToStyle(style, CSSPropertyMarginRight, value);
710 } else if (name == alignAttr) {
711 if (m_inputType->shouldRespectAlignAttribute())
712 applyAlignmentAttributeToStyle(value, style);
713 } else if (name == widthAttr) {
714 if (m_inputType->shouldRespectHeightAndWidthAttributes())
715 addHTMLLengthToStyle(style, CSSPropertyWidth, value);
716 if (isImageButton())
717 applyAspectRatioFromWidthAndHeightAttributesToStyle(value, attributeWithoutSynchronization(heightAttr), style);
718 } else if (name == heightAttr) {
719 if (m_inputType->shouldRespectHeightAndWidthAttributes())
720 addHTMLLengthToStyle(style, CSSPropertyHeight, value);
721 if (isImageButton())
722 applyAspectRatioFromWidthAndHeightAttributesToStyle(attributeWithoutSynchronization(widthAttr), value, style);
723 } else if (name == borderAttr && isImageButton())
724 applyBorderAttributeToStyle(value, style);
725 else
726 HTMLTextFormControlElement::collectPresentationalHintsForAttribute(name, value, style);
727}

729inline void HTMLInputElement::initializeInputType()
730{
731 ASSERT(m_parsingInProgress);
732 ASSERT(!m_inputType);
733
734 const AtomString& type = attributeWithoutSynchronization(typeAttr);
735 if (type.isNull()) {
736 m_inputType = InputType::createText(*this);
737 ASSERT(m_inputType->needsShadowSubtree());
738 createUserAgentShadowRoot();
739 createShadowSubtreeAndUpdateInnerTextElementEditability();
740 updateWillValidateAndValidity();
741 return;
742 }
743
744 m_hasType = true;
745 m_inputType = InputType::create(*this, type);
746 if (m_inputType->needsShadowSubtree()) {
747 createUserAgentShadowRoot();
748 createShadowSubtreeAndUpdateInnerTextElementEditability();
749 }
750 updateWillValidateAndValidity();
751 registerForSuspensionCallbackIfNeeded();
752 runPostTypeUpdateTasks();
753}

755void HTMLInputElement::parseAttribute(const QualifiedName& name, const AtomString& value)
756{
757 ASSERT(m_inputType);
758 Ref<InputType> protectedInputType(*m_inputType);
759
760 if (name == nameAttr) {
761 removeFromRadioButtonGroup();
762 m_name = value;
763 addToRadioButtonGroup();
764 HTMLTextFormControlElement::parseAttribute(name, value);
765 } else if (name == autocompleteAttr) {
766 if (equalLettersIgnoringASCIICase(value, "off")) {
767 m_autocomplete = Off;
768 registerForSuspensionCallbackIfNeeded();
769 } else {
770 bool needsToUnregister = m_autocomplete == Off;
771
772 if (value.isEmpty())
773 m_autocomplete = Uninitialized;
774 else
775 m_autocomplete = On;
776
777 if (needsToUnregister)
778 unregisterForSuspensionCallbackIfNeeded();
779 }
780 } else if (name == typeAttr)
781 updateType();
782 else if (name == valueAttr) {
783 // Changes to the value attribute may change whether or not this element has a default value.
784 // If this field is autocomplete=off that might affect the return value of needsSuspensionCallback.
785 if (m_autocomplete == Off) {
786 unregisterForSuspensionCallbackIfNeeded();
787 registerForSuspensionCallbackIfNeeded();
788 }
789 // We only need to setChanged if the form is looking at the default value right now.
790 if (!hasDirtyValue()) {
791 updatePlaceholderVisibility();
792 invalidateStyleForSubtree();
793 setFormControlValueMatchesRenderer(false);
794 }
795 updateValidity();
796 m_valueAttributeWasUpdatedAfterParsing = !m_parsingInProgress;
797 } else if (name == checkedAttr) {
798 if (m_inputType->isCheckable())
799 invalidateStyleForSubtree();
800
801 // Another radio button in the same group might be checked by state
802 // restore. We shouldn't call setChecked() even if this has the checked
803 // attribute. So, delay the setChecked() call until
804 // finishParsingChildren() is called if parsing is in progress.
805 if ((!m_parsingInProgress || !document().formController().hasFormStateToRestore()) && !m_dirtyCheckednessFlag) {
806 setChecked(!value.isNull());
807 // setChecked() above sets the dirty checkedness flag so we need to reset it.
808 m_dirtyCheckednessFlag = false;
809 }
810 } else if (name == maxlengthAttr)
811 maxLengthAttributeChanged(value);
812 else if (name == minlengthAttr)
813 minLengthAttributeChanged(value);
814 else if (name == sizeAttr) {
815 unsigned oldSize = m_size;
816 m_size = limitToOnlyHTMLNonNegativeNumbersGreaterThanZero(value, defaultSize);
817 if (m_size != oldSize && renderer())
818 renderer()->setNeedsLayoutAndPrefWidthsRecalc();
819 } else if (name == resultsAttr)
820 m_maxResults = value.isNull() ? -1 : std::min(parseHTMLInteger(value).value_or(0), maxSavedResults);
821 else if (name == autosaveAttr || name == incrementalAttr)
822 invalidateStyleForSubtree();
823 else if (name == maxAttr || name == minAttr || name == multipleAttr || name == patternAttr || name == precisionAttr || name == stepAttr)
824 updateValidity();
825#if ENABLE(DATALIST_ELEMENT)
826 else if (name == listAttr) {
827 m_hasNonEmptyList = !value.isEmpty();
828 if (m_hasNonEmptyList) {
829 resetListAttributeTargetObserver();
830 dataListMayHaveChanged();
831 }
832 }
833#endif
834 else
835 HTMLTextFormControlElement::parseAttribute(name, value);
836
837 m_inputType->attributeChanged(name);
838}

1183void HTMLInputElement::defaultEventHandler(Event& event)
1184{
1185 if (is<MouseEvent>(event) && event.type() == eventNames().clickEvent && downcast<MouseEvent>(event).button() == LeftButton) {
1186 m_inputType->handleClickEvent(downcast<MouseEvent>(event));
1187 if (event.defaultHandled())
1188 return;
1189 }
1190
1191#if ENABLE(TOUCH_EVENTS)
1192 if (is<TouchEvent>(event)) {
1193 m_inputType->handleTouchEvent(downcast<TouchEvent>(event));
1194 if (event.defaultHandled())
1195 return;
1196 }
1197#endif
1198
1199 if (is<KeyboardEvent>(event) && event.type() == eventNames().keydownEvent) {
1200 auto shouldCallBaseEventHandler = m_inputType->handleKeydownEvent(downcast<KeyboardEvent>(event));
1201 if (event.defaultHandled() || shouldCallBaseEventHandler == InputType::ShouldCallBaseEventHandler::No)
1202 return;
1203 }
1204
1205 // Call the base event handler before any of our own event handling for almost all events in text fields.
1206 // Makes editing keyboard handling take precedence over the keydown and keypress handling in this function.
1207 bool callBaseClassEarly = isTextField() && (event.type() == eventNames().keydownEvent || event.type() == eventNames().keypressEvent);
1208 if (callBaseClassEarly) {
1209 HTMLTextFormControlElement::defaultEventHandler(event);
1210 if (event.defaultHandled())
1211 return;
1212 }
1213
1214 // DOMActivate events cause the input to be "activated" - in the case of image and submit inputs, this means
1215 // actually submitting the form. For reset inputs, the form is reset. These events are sent when the user clicks
1216 // on the element, or presses enter while it is the active element. JavaScript code wishing to activate the element
1217 // must dispatch a DOMActivate event - a click event will not do the job.
1218 if (event.type() == eventNames().DOMActivateEvent) {
1219 m_inputType->handleDOMActivateEvent(event);
1220 if (event.defaultHandled())
1221 return;
1222 }
1223
1224 // Use key press event here since sending simulated mouse events
1225 // on key down blocks the proper sending of the key press event.
1226 if (is<KeyboardEvent>(event)) {
1227 KeyboardEvent& keyboardEvent = downcast<KeyboardEvent>(event);
1228 if (keyboardEvent.type() == eventNames().keypressEvent) {
1229 m_inputType->handleKeypressEvent(keyboardEvent);
1230 if (keyboardEvent.defaultHandled())
1231 return;
1232 } else if (keyboardEvent.type() == eventNames().keyupEvent) {
1233 m_inputType->handleKeyupEvent(keyboardEvent);
1234 if (keyboardEvent.defaultHandled())
1235 return;
1236 }
1237 }
1238
1239 if (m_inputType->shouldSubmitImplicitly(event)) {
1240 if (isSearchField()) {
1241 addSearchResult();
1242 onSearch();
1243 }
1244 // Form submission finishes editing, just as loss of focus does.
1245 // If there was a change, send the event now.
1246 if (wasChangedSinceLastFormControlChangeEvent())
1247 dispatchFormControlChangeEvent();
1248
1249 // Form may never have been present, or may have been destroyed by code responding to the change event.
1250 if (auto formElement = makeRefPtr(form()))
1251 formElement->submitImplicitly(event, canTriggerImplicitSubmission());
1252
1253 event.setDefaultHandled();
1254 return;
1255 }
1256
1257 if (is<BeforeTextInsertedEvent>(event))
1258 m_inputType->handleBeforeTextInsertedEvent(downcast<BeforeTextInsertedEvent>(event));
1259
1260 if (is<MouseEvent>(event) && event.type() == eventNames().mousedownEvent) {
1261 m_inputType->handleMouseDownEvent(downcast<MouseEvent>(event));
1262 if (event.defaultHandled())
1263 return;
1264 }
1265
1266 m_inputType->forwardEvent(event);
1267
1268 if (!callBaseClassEarly && !event.defaultHandled())
1269 HTMLTextFormControlElement::defaultEventHandler(event);
1270}

1481bool HTMLInputElement::needsSuspensionCallback()
1482{
1483 if (m_inputType->shouldResetOnDocumentActivation())
1484 return true;
1485
1486 // Sensitive input elements are marked with autocomplete=off, and we want to wipe them out
1487 // when going back; returning true here arranges for us to call reset at the time
1488 // the page is restored. Non-empty textual default values indicate that the field
