44static LabelableElement* firstElementWithIdIfLabelable(TreeScope& treeScope, const AtomString& id)
45{
46 auto element = makeRefPtr(treeScope.getElementById(id));
47 if (!is<LabelableElement>(element))
48 return nullptr;
49
50 auto& labelableElement = downcast<LabelableElement>(*element);
51 return labelableElement.supportLabels() ? &labelableElement : nullptr;
52}

65RefPtr<LabelableElement> HTMLLabelElement::control() const
66{
67 auto& controlId = attributeWithoutSynchronization(forAttr);
68 if (controlId.isNull()) {
69 // Search the children and descendants of the label element for a form element.
70 // per http://dev.w3.org/html5/spec/Overview.html#the-label-element
71 // the form element must be "labelable form-associated element".
72 for (auto& labelableElement : descendantsOfType<LabelableElement>(*this)) {
73 if (labelableElement.supportLabels())
74 return const_cast<LabelableElement*>(&labelableElement);
75 }
76 return nullptr;
77 }
78 return isConnected() ? firstElementWithIdIfLabelable(treeScope(), controlId) : nullptr;
79}

81HTMLFormElement* HTMLLabelElement::form() const
82{
83 auto control = this->control();
84 if (!is<HTMLFormControlElement>(control))
85 return nullptr;
86 return downcast<HTMLFormControlElement>(control.get())->form();

89void HTMLLabelElement::setActive(bool down, bool pause, Style::InvalidationScope invalidationScope)
90{
91 if (down == active())

115bool HTMLLabelElement::isEventTargetedAtInteractiveDescendants(Event& event) const
116{
117 if (!is<Node>(event.target()))
118 return false;
119
120 auto& node = downcast<Node>(*event.target());
121 if (!containsIncludingShadowDOM(&node))
122 return false;
123
124 for (const auto* it = &node; it && it != this; it = it->parentElementInComposedTree()) {
125 if (is<HTMLElement>(it) && downcast<HTMLElement>(*it).isInteractiveContent())
126 return true;
127 }
128
129 return false;
130}
131void HTMLLabelElement::defaultEventHandler(Event& event)
132{
133 static bool processingClick = false;
134
135 if (event.type() == eventNames().clickEvent && !processingClick) {
136 auto control = this->control();
137
138 // If we can't find a control or if the control received the click
139 // event, then there's no need for us to do anything.
140 if (!control || (is<Node>(event.target()) && control->containsIncludingShadowDOM(&downcast<Node>(*event.target())))) {
141 HTMLElement::defaultEventHandler(event);
142 return;
143 }
144
145 // The activation behavior of a label element for events targeted at interactive
146 // content descendants of a label element, and any descendants of those interactive
147 // content descendants, must be to do nothing.
148 // https://html.spec.whatwg.org/#the-label-element
149 if (isEventTargetedAtInteractiveDescendants(event)) {
150 HTMLElement::defaultEventHandler(event);
151 return;
152 }
153
154 processingClick = true;
155
156 control->dispatchSimulatedClick(&event);
157
158 document().updateLayoutIgnorePendingStylesheets();
159 if (control->isMouseFocusable())
160 control->focus();
161
162 processingClick = false;
163
164 event.setDefaultHandled();
165 }
166
167 HTMLElement::defaultEventHandler(event);
168}

170bool HTMLLabelElement::willRespondToMouseClickEvents()
171{
172 auto element = control();

176void HTMLLabelElement::focus(const FocusOptions& options)
177{
178 Ref<HTMLLabelElement> protectedThis(*this);
179 if (document().haveStylesheetsLoaded()) {
180 document().updateLayout();
181 if (isFocusable()) {
182 // The value of restorationMode is not used for label elements as it doesn't override updateFocusAppearance.
183 Element::focus(options);
184 return;
185 }
186 }
187
188 // To match other browsers, always restore previous selection.
189 if (auto element = control())
190 element->focus({ SelectionRestorationMode::RestoreOrSelectAll, options.direction });
191}
