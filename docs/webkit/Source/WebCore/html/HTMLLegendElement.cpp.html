38inline HTMLLegendElement::HTMLLegendElement(const QualifiedName& tagName, Document& document)
39 : HTMLElement(tagName, document)
40{
41 ASSERT(hasTagName(HTMLNames::legendTag));
42}

44Ref<HTMLLegendElement> HTMLLegendElement::create(const QualifiedName& tagName, Document& document)
45{
46 return adoptRef(*new HTMLLegendElement(tagName, document));
47}
48
49RefPtr<HTMLFormControlElement> HTMLLegendElement::associatedControl()
50{
51 // Check if there's a fieldset belonging to this legend.
52 auto enclosingFieldset = ancestorsOfType<HTMLFieldSetElement>(*this).first();
53 if (!enclosingFieldset)
54 return nullptr;
55
56 // Find first form element inside the fieldset that is not a legend element.
57 // FIXME: Should we consider tabindex?
58 return descendantsOfType<HTMLFormControlElement>(*enclosingFieldset).first();
59}
60
61void HTMLLegendElement::focus(const FocusOptions& options)
62{
63 if (document().haveStylesheetsLoaded()) {
64 document().updateLayoutIgnorePendingStylesheets();
65 if (isFocusable()) {
66 Element::focus({ options.selectionRestorationMode, options.direction });
67 return;
68 }
69 }
70
71 // To match other browsers' behavior, never restore previous selection.
72 if (auto control = associatedControl())
73 control->focus({ SelectionRestorationMode::SelectAll, options.direction });
74}
75
76bool HTMLLegendElement::accessKeyAction(bool sendMouseEvents)
77{
78 if (auto control = associatedControl())
79 return control->accessKeyAction(sendMouseEvents);
80 return false;
81}
82
83HTMLFormElement* HTMLLegendElement::form() const
84{
85 // According to the specification, If the legend has a fieldset element as
86 // its parent, then the form attribute must return the same value as the
87 // form attribute on that fieldset element. Otherwise, it must return null.
88 auto fieldset = makeRefPtr(parentNode());
89 if (!is<HTMLFieldSetElement>(fieldset))
90 return nullptr;
91 return downcast<HTMLFieldSetElement>(*fieldset).form();
92}
