44bool WindowNameCollection::elementMatchesIfNameAttributeMatch(const Element& element)
45{
46 return is<HTMLEmbedElement>(element)
47 || is<HTMLFormElement>(element)
48 || is<HTMLImageElement>(element)
49 || is<HTMLObjectElement>(element);
50}
51
52bool WindowNameCollection::elementMatches(const Element& element, const AtomStringImpl* name)
53{
54 // Find only images, forms, applets, embeds and objects by name, but anything by id.
55 return (elementMatchesIfNameAttributeMatch(element) && element.getNameAttribute().impl() == name)
56 || element.getIdAttribute() == name;
57}
58
59static inline bool isObjectElementForDocumentNameCollection(const Element& element)
60{
61 return is<HTMLObjectElement>(element) && downcast<HTMLObjectElement>(element).isExposed();
62}
63
64bool DocumentNameCollection::elementMatchesIfIdAttributeMatch(const Element& element)
65{
66 // FIXME: We need to fix HTMLImageElement to update the hash map for us when the name attribute is removed.
67 return isObjectElementForDocumentNameCollection(element)
68 || (is<HTMLImageElement>(element) && element.hasName() && !element.getNameAttribute().isEmpty());
69}
70
71bool DocumentNameCollection::elementMatchesIfNameAttributeMatch(const Element& element)
72{
73 return isObjectElementForDocumentNameCollection(element)
74 || is<HTMLEmbedElement>(element)
75 || is<HTMLFormElement>(element)
76 || is<HTMLIFrameElement>(element)
77 || is<HTMLImageElement>(element);
78}
79
80bool DocumentNameCollection::elementMatches(const Element& element, const AtomStringImpl* name)
81{
82 // Find images, forms, applets, embeds, objects and iframes by name, applets and object by id, and images by id
83 // but only if they have a name attribute (this very strange rule matches IE).
84 return (elementMatchesIfNameAttributeMatch(element) && element.getNameAttribute().impl() == name)
85 || (elementMatchesIfIdAttributeMatch(element) && element.getIdAttribute().impl() == name);
86}
