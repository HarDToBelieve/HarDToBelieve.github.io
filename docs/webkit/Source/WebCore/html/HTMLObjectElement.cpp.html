103void HTMLObjectElement::parseAttribute(const QualifiedName& name, const AtomString& value)
104{
105 bool invalidateRenderer = false;
106
107 if (name == formAttr)
108 formAttributeChanged();
109 else if (name == typeAttr) {
110 m_serviceType = value.string().left(value.find(';')).convertToASCIILowercase();
111 invalidateRenderer = !hasAttributeWithoutSynchronization(classidAttr);
112 setNeedsWidgetUpdate(true);
113 } else if (name == dataAttr) {
114 m_url = stripLeadingAndTrailingHTMLSpaces(value);
115 invalidateRenderer = !hasAttributeWithoutSynchronization(classidAttr);
116 setNeedsWidgetUpdate(true);
117 updateImageLoaderWithNewURLSoon();
118 } else if (name == classidAttr) {
119 invalidateRenderer = true;
120 setNeedsWidgetUpdate(true);
121 } else
122 HTMLPlugInImageElement::parseAttribute(name, value);
123
124 if (!invalidateRenderer || !isConnected() || !renderer())
125 return;
126
127 m_useFallbackContent = false;
128 scheduleUpdateForAfterStyleResolution();
129 invalidateStyleAndRenderersForSubtree();
130}

132static void mapDataParamToSrc(Vector<String>& paramNames, Vector<String>& paramValues)
133{
134 // Some plugins don't understand the "data" attribute of the OBJECT tag (i.e. Real and WMP require "src" attribute).
135 bool foundSrcParam = false;
136 String dataParamValue;
137 for (unsigned i = 0; i < paramNames.size(); ++i) {
138 if (equalLettersIgnoringASCIICase(paramNames[i], "src"))
139 foundSrcParam = true;
140 else if (equalLettersIgnoringASCIICase(paramNames[i], "data"))
141 dataParamValue = paramValues[i];
142 }
143 if (!foundSrcParam && !dataParamValue.isNull()) {
144 paramNames.append("src"_s);
145 paramValues.append(WTFMove(dataParamValue));
146 }
147}

149// FIXME: This function should not deal with url or serviceType!
150void HTMLObjectElement::parametersForPlugin(Vector<String>& paramNames, Vector<String>& paramValues, String& url, String& serviceType)
151{
152 HashSet<StringImpl*, ASCIICaseInsensitiveHash> uniqueParamNames;
153 String urlParameter;
154
155 // Scan the PARAM children and store their name/value pairs.
156 // Get the URL and type from the params if we don't already have them.
157 for (auto& param : childrenOfType<HTMLParamElement>(*this)) {
158 String name = param.name();
159 if (name.isEmpty())
160 continue;
161
162 uniqueParamNames.add(name.impl());
163 paramNames.append(param.name());
164 paramValues.append(param.value());
165
166 // FIXME: url adjustment does not belong in this function.
167 if (url.isEmpty() && urlParameter.isEmpty() && (equalLettersIgnoringASCIICase(name, "src") || equalLettersIgnoringASCIICase(name, "movie") || equalLettersIgnoringASCIICase(name, "code") || equalLettersIgnoringASCIICase(name, "url")))
168 urlParameter = stripLeadingAndTrailingHTMLSpaces(param.value());
169 // FIXME: serviceType calculation does not belong in this function.
170 if (serviceType.isEmpty() && equalLettersIgnoringASCIICase(name, "type")) {
171 serviceType = param.value();
172 size_t pos = serviceType.find(';');
173 if (pos != notFound)
174 serviceType = serviceType.left(pos);
175 }
176 }
177
178 // When OBJECT is used for an applet via Sun's Java plugin, the CODEBASE attribute in the tag
179 // points to the Java plugin itself (an ActiveX component) while the actual applet CODEBASE is
180 // in a PARAM tag. See <http://java.sun.com/products/plugin/1.2/docs/tags.html>. This means
181 // we have to explicitly suppress the tag's CODEBASE attribute if there is none in a PARAM,
182 // else our Java plugin will misinterpret it. [4004531]
183 String codebase;
184 if (MIMETypeRegistry::isJavaAppletMIMEType(serviceType)) {
185 codebase = "codebase";
186 uniqueParamNames.add(codebase.impl()); // pretend we found it in a PARAM already
187 }
188
189 // Turn the attributes of the <object> element into arrays, but don't override <param> values.
190 if (hasAttributes()) {
191 for (const Attribute& attribute : attributesIterator()) {
192 const AtomString& name = attribute.name().localName();
193 if (!uniqueParamNames.contains(name.impl())) {
194 paramNames.append(name.string());
195 paramValues.append(attribute.value().string());
196 }
197 }
198 }
199
200 mapDataParamToSrc(paramNames, paramValues);
201
202 // HTML5 says that an object resource's URL is specified by the object's data
203 // attribute, not by a param element. However, for compatibility, allow the
204 // resource's URL to be given by a param named "src", "movie", "code" or "url"
205 // if we know that resource points to a plug-in.
206
207 if (url.isEmpty() && !urlParameter.isEmpty()) {
208 auto& loader = document().frame()->loader().subframeLoader();
209 if (loader.resourceWillUsePlugin(urlParameter, serviceType))
210 url = urlParameter;
211 }
212}

214bool HTMLObjectElement::hasFallbackContent() const
215{
216 for (RefPtr<Node> child = firstChild(); child; child = child->nextSibling()) {
217 // Ignore whitespace-only text, and <param> tags, any other content is fallback content.
218 if (is<Text>(*child)) {
219 if (!downcast<Text>(*child).data().isAllSpecialCharacters<isHTMLSpace>())
220 return true;
221 } else if (!is<HTMLParamElement>(*child))
222 return true;
223 }
224 return false;
225}

227bool HTMLObjectElement::hasValidClassId()
228{
229 if (MIMETypeRegistry::isJavaAppletMIMEType(serviceType()) && protocolIs(attributeWithoutSynchronization(classidAttr), "java"))
230 return true;
231
232 // HTML5 says that fallback content should be rendered if a non-empty
233 // classid is specified for which the UA can't find a suitable plug-in.
234 return attributeWithoutSynchronization(classidAttr).isEmpty();
235}

237// FIXME: This should be unified with HTMLEmbedElement::updateWidget and
238// moved down into HTMLPluginImageElement.cpp
239void HTMLObjectElement::updateWidget(CreatePlugins createPlugins)
240{
241 ASSERT(!renderEmbeddedObject()->isPluginUnavailable());
242 ASSERT(needsWidgetUpdate());
243
244 // FIXME: This should ASSERT isFinishedParsingChildren() instead.
245 if (!isFinishedParsingChildren()) {
246 setNeedsWidgetUpdate(false);
247 return;
248 }
249
250 // FIXME: I'm not sure it's ever possible to get into updateWidget during a
251 // removal, but just in case we should avoid loading the frame to prevent
252 // security bugs.
253 if (!SubframeLoadingDisabler::canLoadFrame(*this)) {
254 setNeedsWidgetUpdate(false);
255 return;
256 }
257
258 String url = this->url();
259 String serviceType = this->serviceType();
260
261 // FIXME: These should be joined into a PluginParameters class.
262 Vector<String> paramNames;
263 Vector<String> paramValues;
264 parametersForPlugin(paramNames, paramValues, url, serviceType);
265
266 // Note: url is modified above by parametersForPlugin.
267 if (!canLoadURL(url)) {
268 setNeedsWidgetUpdate(false);
269 return;
270 }
271
272 // FIXME: It's unfortunate that we have this special case here.
273 // See http://trac.webkit.org/changeset/25128 and the plugins/netscape-plugin-setwindow-size.html test.
274 if (createPlugins == CreatePlugins::No && wouldLoadAsPlugIn(url, serviceType))
275 return;
276
277 setNeedsWidgetUpdate(false);
278
279 Ref<HTMLObjectElement> protectedThis(*this); // beforeload and plugin loading can make arbitrary DOM mutations.
280 bool beforeLoadAllowedLoad = guardedDispatchBeforeLoadEvent(url);
281 if (!renderer()) // Do not load the plugin if beforeload removed this element or its renderer.
282 return;
283
284 // Dispatching a beforeLoad event could have executed code that changed the document.
285 // Make sure the URL is still safe to load.
286 bool success = beforeLoadAllowedLoad && hasValidClassId() && canLoadURL(url);
287 if (success)
288 success = requestObject(url, serviceType, paramNames, paramValues);
289 if (!success && hasFallbackContent())
290 renderFallbackContent();
291}

337void HTMLObjectElement::renderFallbackContent()
338{
339 if (m_useFallbackContent)
340 return;
341
342 if (!isConnected())
343 return;
344
345 scheduleUpdateForAfterStyleResolution();
346 invalidateStyleAndRenderersForSubtree();
347
348 // Before we give up and use fallback content, check to see if this is a MIME type issue.
349 auto* loader = imageLoader();
350 if (loader && loader->image() && loader->image()->status() != CachedResource::LoadError) {
351 m_serviceType = loader->image()->response().mimeType();
352 if (!isImageType()) {
353 // If we don't think we have an image type anymore, then clear the image from the loader.
354 loader->clearImage();
355 return;
356 }
357 }
358
359 m_useFallbackContent = true;
360}

362static inline bool preventsParentObjectFromExposure(const Element& child)
363{
364 static const auto mostKnownTags = makeNeverDestroyed([] {
365 MemoryCompactLookupOnlyRobinHoodHashSet<QualifiedName> set;
366 auto* tags = HTMLNames::getHTMLTags();
367 for (size_t i = 0; i < HTMLNames::HTMLTagsCount; i++) {
368 auto& tag = *tags[i];
369 // Only the param element was explicitly mentioned in the HTML specification rule
370 // we were trying to implement, but these are other known HTML elements that we
371 // have decided, over the years, to treat as children that do not prevent object
372 // names from being exposed.
373 if (tag == bgsoundTag
374 || tag == commandTag
375 || tag == detailsTag
376 || tag == figcaptionTag
377 || tag == figureTag
378 || tag == paramTag
379 || tag == summaryTag
380 || tag == trackTag)
381 continue;
382 set.add(tag);
383 }
384 return set;
385 }());
386 return mostKnownTags.get().contains(child.tagQName());
387}

389static inline bool preventsParentObjectFromExposure(const Node& child)
390{
391 if (is<Element>(child))
392 return preventsParentObjectFromExposure(downcast<Element>(child));
393 if (is<Text>(child))
394 return !downcast<Text>(child).data().isAllSpecialCharacters<isHTMLSpace>();
395 return true;
396}

398static inline bool shouldBeExposed(const HTMLObjectElement& element)
399{
400 // FIXME: This should be redone to use the concept of an exposed object element,
401 // as documented in the HTML specification section describing DOM tree accessors.
402
403 // The rule we try to implement here, from older HTML specifications, is "object elements
404 // with no children other than param elements, unknown elements and whitespace can be found
405 // by name in a document, and other object elements cannot".
406
407 for (auto child = makeRefPtr(element.firstChild()); child; child = child->nextSibling()) {
408 if (preventsParentObjectFromExposure(*child))
409 return false;
410 }
411 return true;
412}

414void HTMLObjectElement::updateExposedState()
415{
416 bool wasExposed = std::exchange(m_isExposed, shouldBeExposed(*this));
417
418 if (m_isExposed != wasExposed && isConnected() && !isInShadowTree() && is<HTMLDocument>(document())) {
419 auto& document = downcast<HTMLDocument>(this->document());
420
421 auto& id = getIdAttribute();
422 if (!id.isEmpty()) {
423 if (m_isExposed)
424 document.addDocumentNamedItem(*id.impl(), *this);
425 else
426 document.removeDocumentNamedItem(*id.impl(), *this);
427 }
428
429 auto& name = getNameAttribute();
430 if (!name.isEmpty() && id != name) {
431 if (m_isExposed)
432 document.addDocumentNamedItem(*name.impl(), *this);
433 else
434 document.removeDocumentNamedItem(*name.impl(), *this);
435 }
436 }
437}

439bool HTMLObjectElement::containsJavaApplet() const
440{
441 if (MIMETypeRegistry::isJavaAppletMIMEType(attributeWithoutSynchronization(typeAttr)))
442 return true;
443
444 for (auto& child : childrenOfType<Element>(*this)) {
445 if (child.hasTagName(paramTag) && equalLettersIgnoringASCIICase(child.getNameAttribute(), "type")
446 && MIMETypeRegistry::isJavaAppletMIMEType(child.attributeWithoutSynchronization(valueAttr).string()))
447 return true;
448 if (child.hasTagName(objectTag) && downcast<HTMLObjectElement>(child).containsJavaApplet())
449 return true;
450 if (child.hasTagName(appletTag))
451 return true;
452 }
453
454 return false;
455}

476bool HTMLObjectElement::appendFormData(DOMFormData& formData, bool)
477{
478 if (name().isEmpty())
479 return false;
480
481 // Use PluginLoadingPolicy::DoNotLoad here or it would fire JS events synchronously
482 // which would not be safe here.
483 auto widget = makeRefPtr(pluginWidget(PluginLoadingPolicy::DoNotLoad));
484 if (!is<PluginViewBase>(widget))
485 return false;
486 String value;
487 if (!downcast<PluginViewBase>(*widget).getFormValue(value))
488 return false;
489 formData.append(name(), value);
490 return true;
491}
