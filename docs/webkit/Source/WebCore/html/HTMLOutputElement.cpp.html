41namespace WebCore {
42
43WTF_MAKE_ISO_ALLOCATED_IMPL(HTMLOutputElement);
44
45inline HTMLOutputElement::HTMLOutputElement(const QualifiedName& tagName, Document& document, HTMLFormElement* form)
46 : HTMLFormControlElement(tagName, document, form)
47{
48}
49
50Ref<HTMLOutputElement> HTMLOutputElement::create(const QualifiedName& tagName, Document& document, HTMLFormElement* form)
51{
52 return adoptRef(*new HTMLOutputElement(tagName, document, form));
53}
54
55Ref<HTMLOutputElement> HTMLOutputElement::create(Document& document)
56{
57 return create(HTMLNames::outputTag, document, nullptr);
58}
59
60const AtomString& HTMLOutputElement::formControlType() const
61{
62 static MainThreadNeverDestroyed<const AtomString> output("output", AtomString::ConstructFromLiteral);
63 return output;
64}
65
66bool HTMLOutputElement::supportsFocus() const
67{
68 return HTMLElement::supportsFocus();
69}
70
71void HTMLOutputElement::parseAttribute(const QualifiedName& name, const AtomString& value)
72{
73 if (name == HTMLNames::forAttr && m_forTokens)
74 m_forTokens->associatedAttributeValueChanged(value);
75 HTMLFormControlElement::parseAttribute(name, value);
76}
77
78void HTMLOutputElement::reset()
79{
80 stringReplaceAll(defaultValue());
81 m_defaultValueOverride = { };
82}
83
84String HTMLOutputElement::value() const
85{
86 return textContent();
87}
88
89void HTMLOutputElement::setValue(const String& value)
90{
91 m_defaultValueOverride = defaultValue();
92 stringReplaceAll(value);
93}
94
95String HTMLOutputElement::defaultValue() const
96{
97 return m_defaultValueOverride.isNull() ? textContent() : m_defaultValueOverride;
98}
99
100void HTMLOutputElement::setDefaultValue(const String& value)
101{
102 if (m_defaultValueOverride.isNull())
103 stringReplaceAll(value);
104 else
105 m_defaultValueOverride = value;
106}
107
108DOMTokenList& HTMLOutputElement::htmlFor()
109{
110 if (!m_forTokens)
111 m_forTokens = makeUnique<DOMTokenList>(*this, HTMLNames::forAttr);
112 return *m_forTokens;
113}
