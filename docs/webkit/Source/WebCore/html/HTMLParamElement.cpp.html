36inline HTMLParamElement::HTMLParamElement(const QualifiedName& tagName, Document& document)
37 : HTMLElement(tagName, document)
38{
39 ASSERT(hasTagName(paramTag));
40}
41
42Ref<HTMLParamElement> HTMLParamElement::create(const QualifiedName& tagName, Document& document)
43{
44 return adoptRef(*new HTMLParamElement(tagName, document));
45}
46
47String HTMLParamElement::name() const
48{
49 if (hasName())
50 return getNameAttribute();
51 return document().isHTMLDocument() ? emptyAtom() : getIdAttribute();
52}
53
54String HTMLParamElement::value() const
55{
56 return attributeWithoutSynchronization(valueAttr);
57}
58
59bool HTMLParamElement::isURLParameter(const String& name)
60{
61 return equalLettersIgnoringASCIICase(name, "data") || equalLettersIgnoringASCIICase(name, "movie") || equalLettersIgnoringASCIICase(name, "src");
62}
63
64bool HTMLParamElement::isURLAttribute(const Attribute& attribute) const
65{
66 if (attribute.name() == valueAttr && isURLParameter(name()))
67 return true;
68 return HTMLElement::isURLAttribute(attribute);
69}
70
71void HTMLParamElement::addSubresourceAttributeURLs(ListHashSet<URL>& urls) const
72{
73 HTMLElement::addSubresourceAttributeURLs(urls);
74
75 if (!isURLParameter(name()))
76 return;
77
78 addSubresourceURL(urls, document().completeURL(value()));
79}
