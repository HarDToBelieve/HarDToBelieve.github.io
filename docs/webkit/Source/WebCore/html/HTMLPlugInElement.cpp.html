104JSC::Bindings::Instance* HTMLPlugInElement::bindingsInstance()
105{
106 auto frame = makeRefPtr(document().frame());
107 if (!frame)
108 return nullptr;
109
110 // If the host dynamically turns off JavaScript (or Java) we will still return
111 // the cached allocated Bindings::Instance. Not supporting this edge-case is OK.
112
113 if (!m_instance) {
114 if (auto widget = makeRefPtr(pluginWidget()))
115 m_instance = frame->script().createScriptInstanceForWidget(widget.get());
116 }
117 return m_instance.get();
118}

120bool HTMLPlugInElement::guardedDispatchBeforeLoadEvent(const String& sourceURL)
121{
122 // FIXME: Our current plug-in loading design can't guarantee the following
123 // assertion is true, since plug-in loading can be initiated during layout,
124 // and synchronous layout can be initiated in a beforeload event handler!
125 // See <http://webkit.org/b/71264>.
126 // ASSERT(!m_inBeforeLoadEventHandler);
127 m_inBeforeLoadEventHandler = true;
128 // static_cast is used to avoid a compile error since dispatchBeforeLoadEvent
129 // is intentionally undefined on this class.
130 bool beforeLoadAllowedLoad = static_cast<HTMLFrameOwnerElement*>(this)->dispatchBeforeLoadEvent(sourceURL);
131 m_inBeforeLoadEventHandler = false;
132 return beforeLoadAllowedLoad;
133}

135Widget* HTMLPlugInElement::pluginWidget(PluginLoadingPolicy loadPolicy) const
136{
137 if (m_inBeforeLoadEventHandler) {
138 // The plug-in hasn't loaded yet, and it makes no sense to try to load if beforeload handler happened to touch the plug-in element.
139 // That would recursively call beforeload for the same element.
140 return nullptr;
141 }
142
143 RenderWidget* renderWidget = loadPolicy == PluginLoadingPolicy::Load ? renderWidgetLoadingPlugin() : this->renderWidget();
144 if (!renderWidget)
145 return nullptr;
146
147 return renderWidget->widget();
148}

150RenderWidget* HTMLPlugInElement::renderWidgetLoadingPlugin() const
151{
152 RefPtr<FrameView> view = document().view();
153 if (!view || (!view->inUpdateEmbeddedObjects() && !view->layoutContext().isInLayout() && !view->isPainting())) {
154 // Needs to load the plugin immediatedly because this function is called
155 // when JavaScript code accesses the plugin.
156 // FIXME: <rdar://16893708> Check if dispatching events here is safe.
157 document().updateLayoutIgnorePendingStylesheets(Document::RunPostLayoutTasks::Synchronously);
158 }
159 return renderWidget(); // This will return nullptr if the renderer is not a RenderWidget.
160}

169void HTMLPlugInElement::collectPresentationalHintsForAttribute(const QualifiedName& name, const AtomString& value, MutableStyleProperties& style)
170{
171 if (name == widthAttr)
172 addHTMLLengthToStyle(style, CSSPropertyWidth, value);
173 else if (name == heightAttr)
174 addHTMLLengthToStyle(style, CSSPropertyHeight, value);
175 else if (name == vspaceAttr) {
176 addHTMLLengthToStyle(style, CSSPropertyMarginTop, value);
177 addHTMLLengthToStyle(style, CSSPropertyMarginBottom, value);
178 } else if (name == hspaceAttr) {
179 addHTMLLengthToStyle(style, CSSPropertyMarginLeft, value);
180 addHTMLLengthToStyle(style, CSSPropertyMarginRight, value);
181 } else if (name == alignAttr)
182 applyAlignmentAttributeToStyle(value, style);
183 else
184 HTMLFrameOwnerElement::collectPresentationalHintsForAttribute(name, value, style);
185}

187void HTMLPlugInElement::defaultEventHandler(Event& event)
188{
189 // Firefox seems to use a fake event listener to dispatch events to plug-in (tested with mouse events only).
190 // This is observable via different order of events - in Firefox, event listeners specified in HTML attributes fires first, then an event
191 // gets dispatched to plug-in, and only then other event listeners fire. Hopefully, this difference does not matter in practice.
192
193 // FIXME: Mouse down and scroll events are passed down to plug-in via custom code in EventHandler; these code paths should be united.
194
195 auto renderer = this->renderer();
196 if (!is<RenderWidget>(renderer))
197 return;
198
199 if (is<RenderEmbeddedObject>(*renderer) && downcast<RenderEmbeddedObject>(*renderer).isPluginUnavailable())
200 downcast<RenderEmbeddedObject>(*renderer).handleUnavailablePluginIndicatorEvent(&event);
201
202 // Don't keep the widget alive over the defaultEventHandler call, since that can do things like navigate.
203 {
204 RefPtr<Widget> widget = downcast<RenderWidget>(*renderer).widget();
205 if (!widget)
206 return;
207 widget->handleEvent(event);
208 if (event.defaultHandled())
209 return;
210 }
211 HTMLFrameOwnerElement::defaultEventHandler(event);
212}

232bool HTMLPlugInElement::isUserObservable() const
233{
234 // No widget - can't be anything to see or hear here.
235 RefPtr<Widget> widget = pluginWidget(PluginLoadingPolicy::DoNotLoad);
236 if (!is<PluginViewBase>(widget))
237 return false;
238
239 PluginViewBase& pluginView = downcast<PluginViewBase>(*widget);
240
241 // If audio is playing (or might be) then the plugin is detectable.
242 if (pluginView.audioHardwareActivity() != AudioHardwareActivityType::IsInactive)
243 return true;
244
245 // If the plugin is visible and not vanishingly small in either dimension it is detectable.
246 return pluginView.isVisible() && pluginView.width() > 2 && pluginView.height() > 2;
247}

267void HTMLPlugInElement::swapRendererTimerFired()
268{
269 ASSERT(displayState() == PreparingPluginReplacement);
270 if (userAgentShadowRoot())
271 return;
272
273 // Create a shadow root, which will trigger the code to add a snapshot container
274 // and reattach, thus making a new Renderer.
275 Ref<HTMLPlugInElement> protectedThis(*this);
276 ensureUserAgentShadowRoot();
277}

291void HTMLPlugInElement::didAddUserAgentShadowRoot(ShadowRoot& root)
292{
293 if (!m_pluginReplacement || !document().page() || displayState() != PreparingPluginReplacement)
294 return;
295
296 root.setResetStyleInheritance(true);
297 auto result = m_pluginReplacement->installReplacement(root);
298
299#if PLATFORM(COCOA)
300 RELEASE_ASSERT(result.success || !result.scriptObject);
301 m_pluginReplacementScriptObject = result.scriptObject;
302#endif
303
304 if (result.success) {
305 setDisplayState(DisplayingPluginReplacement);
306 invalidateStyleAndRenderersForSubtree();
307 }
308}

314static Vector<ReplacementPlugin*>& registeredPluginReplacements()
315{
316 static NeverDestroyed<Vector<ReplacementPlugin*>> registeredReplacements;
317 static bool enginesQueried = false;
318
319 if (enginesQueried)
320 return registeredReplacements;
321 enginesQueried = true;
322
323#if PLATFORM(COCOA)

338static ReplacementPlugin* pluginReplacementForType(const URL& url, const String& mimeType)
339{
340 Vector<ReplacementPlugin*>& replacements = registeredPluginReplacements();
341 if (replacements.isEmpty())
342 return nullptr;
343
344 String extension;
345 auto lastPathComponent = url.lastPathComponent();
346 size_t dotOffset = lastPathComponent.reverseFind('.');
347 if (dotOffset != notFound)
348 extension = lastPathComponent.substring(dotOffset + 1).toString();
349
350 String type = mimeType;
351 if (type.isEmpty() && url.protocolIsData())
352 type = mimeTypeFromDataURL(url.string());
353
354 if (type.isEmpty() && !extension.isEmpty()) {
355 for (auto* replacement : replacements) {
356 if (replacement->supportsFileExtension(extension) && replacement->supportsURL(url))
357 return replacement;
358 }
359 }
360
361 if (type.isEmpty()) {
362 if (extension.isEmpty())
363 return nullptr;
364 type = MIMETypeRegistry::mediaMIMETypeForExtension(extension);
365 }
366
367 if (type.isEmpty())
368 return nullptr;
369
370 for (auto* replacement : replacements) {
371 if (replacement->supportsType(type) && replacement->supportsURL(url))
372 return replacement;
373 }
374
375 return nullptr;
376}

378bool HTMLPlugInElement::requestObject(const String& relativeURL, const String& mimeType, const Vector<String>& paramNames, const Vector<String>& paramValues)
379{
380 if (m_pluginReplacement)
381 return true;
382
383 URL completedURL;
384 if (!relativeURL.isEmpty())
385 completedURL = document().completeURL(relativeURL);
386
387 ReplacementPlugin* replacement = pluginReplacementForType(completedURL, mimeType);
388 if (!replacement || !replacement->isEnabledBySettings(document().settings()))
389 return false;
390
391 LOG(Plugins, "%p - Found plug-in replacement for %s.", this, completedURL.string().utf8().data());
392
393 m_pluginReplacement = replacement->create(*this, paramNames, paramValues);
394 setDisplayState(PreparingPluginReplacement);
395 return true;
396}

419bool HTMLPlugInElement::setReplacement(RenderEmbeddedObject::PluginUnavailabilityReason reason, const String& unavailabilityDescription)
420{
421 if (!is<RenderEmbeddedObject>(renderer()))
422 return false;
423
424 if (reason == RenderEmbeddedObject::UnsupportedPlugin)
425 document().addConsoleMessage(MessageSource::JS, MessageLevel::Warning, "Tried to use an unsupported plug-in."_s);
426
427 Ref<HTMLPlugInElement> protectedThis(*this);
428 downcast<RenderEmbeddedObject>(*renderer()).setPluginUnavailabilityReasonWithDescription(reason, unavailabilityDescription);
429 bool replacementIsObscured = isReplacementObscured();
430 // hittest in isReplacementObscured() method could destroy the renderer. Let's refetch it.
431 if (is<RenderEmbeddedObject>(renderer()))
432 downcast<RenderEmbeddedObject>(*renderer()).setUnavailablePluginIndicatorIsHidden(replacementIsObscured);
433 return replacementIsObscured;
434}

436bool HTMLPlugInElement::isReplacementObscured()
437{
438 auto topDocument = makeRef(document().topDocument());
439 auto topFrameView = makeRefPtr(topDocument->view());
440 if (!topFrameView)
441 return false;
442
443 topFrameView->updateLayoutAndStyleIfNeededRecursive();
444
445 // Updating the layout may have detached this document from the top document.
446 auto* renderView = topDocument->renderView();
447 if (!renderView || !document().view() || &document().topDocument() != topDocument.ptr())
448 return false;
449
450 if (!renderer() || !is<RenderEmbeddedObject>(*renderer()))
451 return false;
452 auto& pluginRenderer = downcast<RenderEmbeddedObject>(*renderer());
453 // Check the opacity of each layer containing the element or its ancestors.
454 float opacity = 1.0;
455 for (auto* layer = pluginRenderer.enclosingLayer(); layer; layer = layer->parent()) {
456 opacity *= layer->renderer().style().opacity();
457 if (opacity < 0.1)
458 return true;
459 }
460 // Calculate the absolute rect for the blocked plugin replacement text.
461 LayoutPoint absoluteLocation(pluginRenderer.absoluteBoundingBoxRect().location());
462 LayoutRect rect = pluginRenderer.unavailablePluginIndicatorBounds(absoluteLocation);
463 if (rect.isEmpty())
464 return true;
465 auto viewRect = document().view()->convertToRootView(snappedIntRect(rect));
466 auto x = viewRect.x();
467 auto y = viewRect.y();
468 auto width = viewRect.width();
469 auto height = viewRect.height();
470 // Hit test the center and near the corners of the replacement text to ensure
471 // it is visible and is not masked by other elements.
472 constexpr OptionSet<HitTestRequest::Type> hitType { HitTestRequest::Type::ReadOnly, HitTestRequest::Type::Active, HitTestRequest::Type::IgnoreClipping, HitTestRequest::Type::DisallowUserAgentShadowContent, HitTestRequest::Type::AllowChildFrameContent };
473 HitTestResult result;
474 HitTestLocation location { LayoutPoint { viewRect.center() } };
475 ASSERT(!renderView->needsLayout());
476 ASSERT(!renderView->document().needsStyleRecalc());
477 bool hit = topDocument->hitTest(hitType, location, result);
478 if (!hit || result.innerNode() != &pluginRenderer.frameOwnerElement())
479 return true;
480
481 location = LayoutPoint(x, y);
482 hit = topDocument->hitTest(hitType, location, result);
483 if (!hit || result.innerNode() != &pluginRenderer.frameOwnerElement())
484 return true;
485
486 location = LayoutPoint(x + width, y);
487 hit = topDocument->hitTest(hitType, location, result);
488 if (!hit || result.innerNode() != &pluginRenderer.frameOwnerElement())
489 return true;
490
491 location = LayoutPoint(x + width, y + height);
492 hit = topDocument->hitTest(hitType, location, result);
493 if (!hit || result.innerNode() != &pluginRenderer.frameOwnerElement())
494 return true;
495
496 location = LayoutPoint(x, y + height);
497 hit = topDocument->hitTest(hitType, location, result);
498 if (!hit || result.innerNode() != &pluginRenderer.frameOwnerElement())
499 return true;
500 return false;
501}
