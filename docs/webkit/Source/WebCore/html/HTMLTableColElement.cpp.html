40const unsigned defaultSpan { 1 };
41const unsigned minSpan { 1 };
42const unsigned maxSpan { 1000 };

44using namespace HTMLNames;

46inline HTMLTableColElement::HTMLTableColElement(const QualifiedName& tagName, Document& document)
47 : HTMLTablePartElement(tagName, document)
48 , m_span(1)
49{
50}
51
52Ref<HTMLTableColElement> HTMLTableColElement::create(const QualifiedName& tagName, Document& document)
53{
54 return adoptRef(*new HTMLTableColElement(tagName, document));
55}
56
57bool HTMLTableColElement::hasPresentationalHintsForAttribute(const QualifiedName& name) const
58{
59 if (name == widthAttr)
60 return true;
61 return HTMLTablePartElement::hasPresentationalHintsForAttribute(name);
62}
63
64void HTMLTableColElement::collectPresentationalHintsForAttribute(const QualifiedName& name, const AtomString& value, MutableStyleProperties& style)
65{
66 if (name == widthAttr)
67 addHTMLMultiLengthToStyle(style, CSSPropertyWidth, value);
68 else if (name == heightAttr)
69 addHTMLMultiLengthToStyle(style, CSSPropertyHeight, value);
70 else
71 HTMLTablePartElement::collectPresentationalHintsForAttribute(name, value, style);
72}
73
74void HTMLTableColElement::parseAttribute(const QualifiedName& name, const AtomString& value)
75{
76 if (name == spanAttr) {
77 m_span = clampHTMLNonNegativeIntegerToRange(value, minSpan, maxSpan, defaultSpan);
78 if (is<RenderTableCol>(renderer()))
79 downcast<RenderTableCol>(*renderer()).updateFromElement();
80 } else if (name == widthAttr) {
81 if (!value.isEmpty()) {
82 if (is<RenderTableCol>(renderer())) {
83 auto& col = downcast<RenderTableCol>(*renderer());
84 int newWidth = parseHTMLInteger(value).value_or(0);
85 if (newWidth != col.width())
86 col.setNeedsLayoutAndPrefWidthsRecalc();
87 }
88 }
89 } else
90 HTMLTablePartElement::parseAttribute(name, value);
91}
92
93const StyleProperties* HTMLTableColElement::additionalPresentationalHintStyle() const
94{
95 if (!hasTagName(colgroupTag))
96 return nullptr;
97 if (auto table = findParentTable())
98 return table->additionalGroupStyle(false);
99 return nullptr;
100}
101
102void HTMLTableColElement::setSpan(unsigned span)
103{
104 setUnsignedIntegralAttribute(spanAttr, limitToOnlyHTMLNonNegative(span, defaultSpan));
105}
106
107String HTMLTableColElement::width() const
108{
109 return attributeWithoutSynchronization(widthAttr);
110}
