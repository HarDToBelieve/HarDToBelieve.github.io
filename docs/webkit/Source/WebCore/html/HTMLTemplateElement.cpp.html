45inline HTMLTemplateElement::HTMLTemplateElement(const QualifiedName& tagName, Document& document)
46 : HTMLElement(tagName, document)
47{
48}
49
50HTMLTemplateElement::~HTMLTemplateElement()
51{
52 if (m_content)
53 m_content->clearHost();
54}
55
56Ref<HTMLTemplateElement> HTMLTemplateElement::create(const QualifiedName& tagName, Document& document)
57{
58 return adoptRef(*new HTMLTemplateElement(tagName, document));
59}
60
61DocumentFragment* HTMLTemplateElement::contentIfAvailable() const
62{
63 return m_content.get();
64}
65
66DocumentFragment& HTMLTemplateElement::content() const
67{
68 if (!m_content)
69 m_content = TemplateContentDocumentFragment::create(document().ensureTemplateDocument(), this);
70
71 return *m_content;
72}
73
74Ref<Node> HTMLTemplateElement::cloneNodeInternal(Document& targetDocument, CloningOperation type)
75{
76 RefPtr<Node> clone;
77 switch (type) {
78 case CloningOperation::OnlySelf:
79 return cloneElementWithoutChildren(targetDocument);
80 case CloningOperation::SelfWithTemplateContent:
81 clone = cloneElementWithoutChildren(targetDocument);
82 break;
83 case CloningOperation::Everything:
84 clone = cloneElementWithChildren(targetDocument);
85 break;
86 }
87 if (m_content)
88 content().cloneChildNodes(downcast<HTMLTemplateElement>(clone.get())->content());
89 return clone.releaseNonNull();
90}
91
92void HTMLTemplateElement::didMoveToNewDocument(Document& oldDocument, Document& newDocument)
93{
94 HTMLElement::didMoveToNewDocument(oldDocument, newDocument);
95 if (!m_content)
96 return;
97 ASSERT_WITH_SECURITY_IMPLICATION(&document() == &newDocument);
98 m_content->setTreeScopeRecursively(newDocument.ensureTemplateDocument());
99}
