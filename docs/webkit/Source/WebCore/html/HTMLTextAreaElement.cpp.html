62// On submission, LF characters are converted into CRLF.
63// This function returns number of characters considering this.
64static inline unsigned computeLengthForSubmission(StringView text, unsigned numberOfLineBreaks)
65{
66 return numGraphemeClusters(text) + numberOfLineBreaks;
67}
68
69static unsigned numberOfLineBreaks(StringView text)
70{
71 unsigned length = text.length();
72 unsigned count = 0;
73 for (unsigned i = 0; i < length; i++) {
74 if (text[i] == '\n')
75 count++;
76 }
77 return count;
78}
79
80static inline unsigned computeLengthForSubmission(StringView text)
81{
82 return numGraphemeClusters(text) + numberOfLineBreaks(text);
83}
84
85static inline unsigned upperBoundForLengthForSubmission(StringView text, unsigned numberOfLineBreaks)
86{
87 return text.length() + numberOfLineBreaks;
88}

143{

156{

169void HTMLTextAreaElement::parseAttribute(const QualifiedName& name, const AtomString& value)
170{
171 if (name == rowsAttr) {
172 unsigned rows = limitToOnlyHTMLNonNegativeNumbersGreaterThanZero(value, defaultRows);
173 if (m_rows != rows) {
174 m_rows = rows;
175 if (renderer())
176 renderer()->setNeedsLayoutAndPrefWidthsRecalc();
177 }
178 } else if (name == colsAttr) {
179 unsigned cols = limitToOnlyHTMLNonNegativeNumbersGreaterThanZero(value, defaultCols);
180 if (m_cols != cols) {
181 m_cols = cols;
182 if (renderer())
183 renderer()->setNeedsLayoutAndPrefWidthsRecalc();
184 }
185 } else if (name == wrapAttr) {
186 // The virtual/physical values were a Netscape extension of HTML 3.0, now deprecated.
187 // The soft/hard /off values are a recommendation for HTML 4 extension by IE and NS 4.
188 WrapMethod wrap;
189 if (equalLettersIgnoringASCIICase(value, "physical") || equalLettersIgnoringASCIICase(value, "hard") || equalLettersIgnoringASCIICase(value, "on"))
190 wrap = HardWrap;
191 else if (equalLettersIgnoringASCIICase(value, "off"))
192 wrap = NoWrap;
193 else
194 wrap = SoftWrap;
195 if (wrap != m_wrap) {
196 m_wrap = wrap;
197 if (renderer())
198 renderer()->setNeedsLayoutAndPrefWidthsRecalc();
199 }
200 } else if (name == maxlengthAttr)
201 maxLengthAttributeChanged(value);
202 else if (name == minlengthAttr)
203 minLengthAttributeChanged(value);
204 else
205 HTMLTextFormControlElement::parseAttribute(name, value);
206}

225bool HTMLTextAreaElement::appendFormData(DOMFormData& formData, bool)
226{
227 if (name().isEmpty())
228 return false;
229
230 Ref<HTMLTextAreaElement> protectedThis(*this);
231 document().updateLayout();
232
233 formData.append(name(), m_wrap == HardWrap ? valueWithHardLineBreaks() : value());
234
235 auto& dirnameAttrValue = attributeWithoutSynchronization(dirnameAttr);
236 if (!dirnameAttrValue.isNull())
237 formData.append(dirnameAttrValue, directionForFormData());
238
239 return true;
240}

268void HTMLTextAreaElement::updateFocusAppearance(SelectionRestorationMode restorationMode, SelectionRevealMode revealMode)
269{

271 restoreCachedSelection(revealMode, Element::defaultFocusTextStateChangeIntent());

306void HTMLTextAreaElement::handleBeforeTextInsertedEvent(BeforeTextInsertedEvent& event) const
307{
308 ASSERT(renderer());
309 int signedMaxLength = effectiveMaxLength();
310 if (signedMaxLength < 0)
311 return;
312 unsigned unsignedMaxLength = static_cast<unsigned>(signedMaxLength);
313
314 const String& currentValue = innerTextValue();
315 unsigned numberOfLineBreaksInCurrentValue = numberOfLineBreaks(currentValue);
316 if (upperBoundForLengthForSubmission(currentValue, numberOfLineBreaksInCurrentValue)
317 + upperBoundForLengthForSubmission(event.text(), numberOfLineBreaks(event.text())) < unsignedMaxLength)
318 return;
319
320 unsigned currentLength = computeLengthForSubmission(currentValue, numberOfLineBreaksInCurrentValue);
321 // selectionLength represents the selection length of this text field to be
322 // removed by this insertion.
323 // If the text field has no focus, we don't need to take account of the
324 // selection length. The selection is the source of text drag-and-drop in
325 // that case, and nothing in the text field will be removed.
326 auto selectionRange = focused() ? document().frame()->selection().selection().toNormalizedRange() : std::nullopt;
327 unsigned selectionLength = selectionRange ? computeLengthForSubmission(plainText(*selectionRange)) : 0;
328 ASSERT(currentLength >= selectionLength);
329 unsigned baseLength = currentLength - selectionLength;
330 unsigned appendableLength = unsignedMaxLength > baseLength ? unsignedMaxLength - baseLength : 0;
331 event.setText(sanitizeUserInputValue(event.text(), appendableLength));
332}

334String HTMLTextAreaElement::sanitizeUserInputValue(const String& proposedValue, unsigned maxLength)
335{
336 return proposedValue.left(numCodeUnitsInGraphemeClusters(proposedValue, maxLength));
337}

386void HTMLTextAreaElement::setValueCommon(const String& newValue)
387{
388 m_wasModifiedByUser = false;
389 // Code elsewhere normalizes line endings added by the user via the keyboard or pasting.
390 // We normalize line endings coming from JavaScript here.
391 String normalizedValue = newValue.isNull() ? emptyString() : newValue;
392 normalizedValue.replace("\r\n", "\n");
393 normalizedValue.replace('\r', '\n');
394
395 // Return early because we don't want to move the caret or trigger other side effects
396 // when the value isn't changing. This matches Firefox behavior, at least.
397 if (normalizedValue == value())
398 return;
399
400 m_value = normalizedValue;
401 setInnerTextValue(m_value);
402 setLastChangeWasNotUserEdit();
403 updatePlaceholderVisibility();
404 invalidateStyleForSubtree();
405 setFormControlValueMatchesRenderer(true);
406
407 // Set the caret to the end of the text value.
408 if (document().focusedElement() == this) {
409 unsigned endOfString = m_value.length();
410 setSelectionRange(endOfString, endOfString);
411 }
412
413 setTextAsOfLastFormControlChangeEvent(normalizedValue);
414}

426String HTMLTextAreaElement::validationMessage() const
427{
428 if (!willValidate())
429 return String();
430
431 if (customError())
432 return customValidationMessage();
433
434 if (valueMissing())
435 return validationMessageValueMissingText();
436
437 if (tooShort())

456bool HTMLTextAreaElement::tooShort(StringView value, NeedsToCheckDirtyFlag check) const
457{
458 // Return false for the default value or value set by script even if it is
459 // shorter than minLength.
460 if (check == CheckDirtyFlag && !m_wasModifiedByUser)
461 return false;
462
463 int min = minLength();
464 if (min <= 0)
465 return false;
466
467 // The empty string is excluded from tooShort validation.
468 if (value.isEmpty())
469 return false;
470
471 // FIXME: The HTML specification says that the "number of characters" is measured using code-unit length and,
472 // in the case of textarea elements, with all line breaks normalized to a single character (as opposed to CRLF pairs).
473 unsigned unsignedMin = static_cast<unsigned>(min);
474 unsigned numberOfLineBreaksInValue = numberOfLineBreaks(value);
475 return upperBoundForLengthForSubmission(value, numberOfLineBreaksInValue) < unsignedMin
476 && computeLengthForSubmission(value, numberOfLineBreaksInValue) < unsignedMin;
477}

479bool HTMLTextAreaElement::tooLong() const

483
484bool HTMLTextAreaElement::tooLong(StringView value, NeedsToCheckDirtyFlag check) const
485{
486 // Return false for the default value or value set by script even if it is
487 // longer than maxLength.
488 if (check == CheckDirtyFlag && !m_wasModifiedByUser)
489 return false;
490
491 int max = effectiveMaxLength();
492 if (max < 0)
493 return false;
494
495 // FIXME: The HTML specification says that the "number of characters" is measured using code-unit length and,
496 // in the case of textarea elements, with all line breaks normalized to a single character (as opposed to CRLF pairs).
497 unsigned unsignedMax = static_cast<unsigned>(max);
498 unsigned numberOfLineBreaksInValue = numberOfLineBreaks(value);
499 return upperBoundForLengthForSubmission(value, numberOfLineBreaksInValue) > unsignedMax
500 && computeLengthForSubmission(value, numberOfLineBreaksInValue) > unsignedMax;
501}

539void HTMLTextAreaElement::updatePlaceholderText()
540{
541 String placeholderText = strippedPlaceholder();
542 if (placeholderText.isEmpty()) {
543 if (m_placeholder) {
544 userAgentShadowRoot()->removeChild(*m_placeholder);
545 m_placeholder = nullptr;
546 }
547 return;
548 }
549 if (!m_placeholder) {
550 m_placeholder = TextControlPlaceholderElement::create(document());
551 userAgentShadowRoot()->insertBefore(*m_placeholder, innerTextElement()->nextSibling());
552 }
553 m_placeholder->setInnerText(placeholderText);
554}
