114void HTMLVideoElement::collectPresentationalHintsForAttribute(const QualifiedName& name, const AtomString& value, MutableStyleProperties& style)
115{
116 if (name == widthAttr) {
117 addHTMLLengthToStyle(style, CSSPropertyWidth, value);
118 applyAspectRatioFromWidthAndHeightAttributesToStyle(value, attributeWithoutSynchronization(heightAttr), style);
119 } else if (name == heightAttr) {
120 addHTMLLengthToStyle(style, CSSPropertyHeight, value);
121 applyAspectRatioFromWidthAndHeightAttributesToStyle(attributeWithoutSynchronization(widthAttr), value, style);
122 } else
123 HTMLMediaElement::collectPresentationalHintsForAttribute(name, value, style);
124}

133void HTMLVideoElement::parseAttribute(const QualifiedName& name, const AtomString& value)
134{
135 if (name == posterAttr) {
136 if (shouldDisplayPosterImage()) {
137 if (!m_imageLoader)
138 m_imageLoader = makeUnique<HTMLImageLoader>(*this);
139 m_imageLoader->updateFromElementIgnoringPreviousError();
140 } else {
141 if (auto* renderer = this->renderer()) {
142 renderer->imageResource().setCachedImage(nullptr);
143 renderer->updateFromElement();
144 }
145 }
146 }
147#if ENABLE(WIRELESS_PLAYBACK_TARGET)
148 else if (name == webkitwirelessvideoplaybackdisabledAttr)
149 mediaSession().setWirelessVideoPlaybackDisabled(true);
150#endif
151 else {
152 HTMLMediaElement::parseAttribute(name, value);
153
154#if PLATFORM(IOS_FAMILY) && ENABLE(WIRELESS_PLAYBACK_TARGET)
155 if (name == webkitairplayAttr) {
156 bool disabled = false;
157 if (equalLettersIgnoringASCIICase(attributeWithoutSynchronization(HTMLNames::webkitairplayAttr), "deny"))
158 disabled = true;
159 mediaSession().setWirelessVideoPlaybackDisabled(disabled);
160 }
161#endif
162 }
163}

165bool HTMLVideoElement::supportsFullscreen(HTMLMediaElementEnums::VideoFullscreenMode videoFullscreenMode) const
166{
167 if (!player())
168 return false;
169
170 if (videoFullscreenMode == HTMLMediaElementEnums::VideoFullscreenModePictureInPicture) {
171 if (!mediaSession().allowsPictureInPicture())
172 return false;
173 if (!player()->supportsPictureInPicture())
174 return false;
175 }
176
177 Page* page = document().page();
178 if (!page)
179 return false;
180
181 if (!player()->supportsFullscreen())
182 return false;
183
184#if PLATFORM(IOS_FAMILY)
185 UNUSED_PARAM(videoFullscreenMode);
186 // Fullscreen implemented by player.
187 return true;
188#else
189
190#if ENABLE(FULLSCREEN_API)
191 if (videoFullscreenMode == HTMLMediaElementEnums::VideoFullscreenModeStandard && !document().settings().fullScreenEnabled())
192 return false;
193
194 // If the full screen API is enabled and is supported for the current element
195 // do not require that the player has a video track to enter full screen.
196 if (videoFullscreenMode == HTMLMediaElementEnums::VideoFullscreenModeStandard && page->chrome().client().supportsFullScreenForElement(*this, false))
197 return true;
198#endif
199
200 if (!player()->hasVideo())
201 return false;
202
203 return page->chrome().client().supportsVideoFullscreen(videoFullscreenMode);
204#endif // PLATFORM(IOS_FAMILY)
205}

255bool HTMLVideoElement::shouldDisplayPosterImage() const
256{
257 if (!showPosterFlag())
258 return false;
259
260 if (posterImageURL().isEmpty())
261 return false;
262
263 auto* renderer = this->renderer();
264 if (renderer && renderer->failedToLoadPosterImage())
265 return false;
266
267 return true;
268}

319ExceptionOr<void> HTMLVideoElement::webkitEnterFullscreen()
320{
321 ALWAYS_LOG(LOGIDENTIFIER);
322 if (isFullscreen())
323 return { };
324
325 // Generate an exception if this isn't called in response to a user gesture, or if the
326 // element does not support fullscreen, or the element is changing fullscreen mode.
327 if (!mediaSession().fullscreenPermitted() || !supportsFullscreen(HTMLMediaElementEnums::VideoFullscreenModeStandard) || isChangingVideoFullscreenMode())
328 return Exception { InvalidStateError };
329
330 enterFullscreen();
331 return { };
332}

413bool HTMLVideoElement::webkitSupportsPresentationMode(VideoPresentationMode mode) const
414{
415 if (mode == VideoPresentationMode::Fullscreen)
416 return supportsFullscreen(HTMLMediaElementEnums::VideoFullscreenModeStandard);
417
418 if (mode == VideoPresentationMode::PictureInPicture) {
419 if (!supportsPictureInPicture())
420 return false;
421
422 return supportsFullscreen(HTMLMediaElementEnums::VideoFullscreenModePictureInPicture);
423 }
424
425 if (mode == VideoPresentationMode::Inline)
426 return !mediaSession().requiresFullscreenForVideoPlayback();
427
428 return false;
429}

431static inline HTMLMediaElementEnums::VideoFullscreenMode toFullscreenMode(HTMLVideoElement::VideoPresentationMode mode)
432{
433 switch (mode) {
434 case HTMLVideoElement::VideoPresentationMode::Fullscreen:
435 return HTMLMediaElementEnums::VideoFullscreenModeStandard;
436 case HTMLVideoElement::VideoPresentationMode::PictureInPicture:
437 return HTMLMediaElementEnums::VideoFullscreenModePictureInPicture;
438 case HTMLVideoElement::VideoPresentationMode::Inline:
439 return HTMLMediaElementEnums::VideoFullscreenModeNone;
440 }
441 ASSERT_NOT_REACHED();
442 return HTMLMediaElementEnums::VideoFullscreenModeNone;
443}
444
445HTMLVideoElement::VideoPresentationMode HTMLVideoElement::toPresentationMode(HTMLMediaElementEnums::VideoFullscreenMode mode)
446{
447 if (mode == HTMLMediaElementEnums::VideoFullscreenModeStandard)
448 return HTMLVideoElement::VideoPresentationMode::Fullscreen;
449
450 if (mode & HTMLMediaElementEnums::VideoFullscreenModePictureInPicture)
451 return HTMLVideoElement::VideoPresentationMode::PictureInPicture;
452
453 if (mode == HTMLMediaElementEnums::VideoFullscreenModeNone)
454 return HTMLVideoElement::VideoPresentationMode::Inline;
455
456 ASSERT_NOT_REACHED();
457 return HTMLVideoElement::VideoPresentationMode::Inline;
458}

467void HTMLVideoElement::setPresentationMode(VideoPresentationMode mode)
468{
469 if (toPresentationMode(fullscreenMode()) == mode)
470 return;
471
472 auto videoFullscreenMode = toFullscreenMode(mode);
473 INFO_LOG(LOGIDENTIFIER, ", videoFullscreenMode = ", videoFullscreenMode);
474
475 if (videoFullscreenMode == VideoFullscreenModeNone) {
476 if (isFullscreen()) {
477 if (toPresentationMode(fullscreenMode()) == VideoPresentationMode::PictureInPicture)
478 m_exitingPictureInPicture = true;
479
480 exitFullscreen();
481 }
482
483 return;
484 }
485
486 if (!mediaSession().fullscreenPermitted() || !supportsFullscreen(videoFullscreenMode))
487 return;
488
489 if (videoFullscreenMode == VideoFullscreenModePictureInPicture)
490 m_enteringPictureInPicture = true;
491 else if (fullscreenMode() == VideoFullscreenModePictureInPicture)
492 m_exitingPictureInPicture = true;
493
494 enterFullscreen(videoFullscreenMode);
495}

502void HTMLVideoElement::didEnterFullscreenOrPictureInPicture(const FloatSize& size)
503{
504 if (m_enteringPictureInPicture) {
505 m_enteringPictureInPicture = false;
506 setChangingVideoFullscreenMode(false);
507
508#if ENABLE(PICTURE_IN_PICTURE_API)
509 if (m_pictureInPictureObserver)
510 m_pictureInPictureObserver->didEnterPictureInPicture(flooredIntSize(size));
511#else
512 UNUSED_PARAM(size);
513#endif
514 return;
515 }
516
517 if (m_exitingPictureInPicture) {
518 m_exitingPictureInPicture = false;
519#if ENABLE(PICTURE_IN_PICTURE_API)
520 if (m_pictureInPictureObserver)
521 m_pictureInPictureObserver->didExitPictureInPicture();
522#endif
523 }
524
525 HTMLMediaElement::didBecomeFullscreenElement();
526}

528void HTMLVideoElement::didExitFullscreenOrPictureInPicture()
529{
530 if (m_exitingPictureInPicture) {
531 m_exitingPictureInPicture = false;
532 setChangingVideoFullscreenMode(false);
533
534#if ENABLE(PICTURE_IN_PICTURE_API)
535 if (m_pictureInPictureObserver)
536 m_pictureInPictureObserver->didExitPictureInPicture();
537#endif
538 return;
539 }

541 HTMLMediaElement::didStopBeingFullscreenElement();
