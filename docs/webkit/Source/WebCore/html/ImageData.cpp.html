38static CheckedUint32 computeDataSize(const IntSize& size)
39{
40 CheckedUint32 checkedDataSize = 4;
41 checkedDataSize *= static_cast<unsigned>(size.width());
42 checkedDataSize *= static_cast<unsigned>(size.height());
43 return checkedDataSize;
44}

46PredefinedColorSpace ImageData::computeColorSpace(std::optional<ImageDataSettings> settings, PredefinedColorSpace defaultColorSpace)
47{
48 if (settings && settings->colorSpace)
49 return *settings->colorSpace;
50 return defaultColorSpace;
51}

59RefPtr<ImageData> ImageData::create(std::optional<PixelBuffer>&& pixelBuffer)
60{
61 if (!pixelBuffer)
62 return nullptr;
63 return create(WTFMove(*pixelBuffer));

66RefPtr<ImageData> ImageData::create(const IntSize& size)
67{
68 auto dataSize = computeDataSize(size);
69 if (dataSize.hasOverflowed())
70 return nullptr;
71 auto byteArray = Uint8ClampedArray::tryCreateUninitialized(dataSize);
72 if (!byteArray)
73 return nullptr;
74 return adoptRef(*new ImageData(size, byteArray.releaseNonNull(), PredefinedColorSpace::SRGB));
75}
76
77RefPtr<ImageData> ImageData::create(const IntSize& size, Ref<Uint8ClampedArray>&& byteArray, PredefinedColorSpace colorSpace)
78{
79 auto dataSize = computeDataSize(size);
80 if (dataSize.hasOverflowed() || dataSize != byteArray->length())
81 return nullptr;
82
83 return adoptRef(*new ImageData(size, WTFMove(byteArray), colorSpace));
84}

86ExceptionOr<Ref<ImageData>> ImageData::createUninitialized(unsigned rows, unsigned pixelsPerRow, PredefinedColorSpace defaultColorSpace, std::optional<ImageDataSettings> settings)
87{
88 IntSize size(rows, pixelsPerRow);
89 auto dataSize = computeDataSize(size);
90 if (dataSize.hasOverflowed())
91 return Exception { RangeError, "Cannot allocate a buffer of this size"_s };
92
93 auto byteArray = Uint8ClampedArray::tryCreateUninitialized(dataSize);
94 if (!byteArray) {
95 // FIXME: Does this need to be a "real" out of memory error with setOutOfMemoryError called on it?
96 return Exception { RangeError, "Out of memory"_s };
97 }
98
99 auto colorSpace = computeColorSpace(settings, defaultColorSpace);
100 return adoptRef(*new ImageData(size, byteArray.releaseNonNull(), colorSpace));
101}

103ExceptionOr<Ref<ImageData>> ImageData::create(unsigned sw, unsigned sh, std::optional<ImageDataSettings> settings)
104{
105 if (!sw || !sh)
106 return Exception { IndexSizeError };
107
108 IntSize size(sw, sh);
109 auto dataSize = computeDataSize(size);
110 if (dataSize.hasOverflowed())
111 return Exception { RangeError, "Cannot allocate a buffer of this size"_s };
112
113 auto byteArray = Uint8ClampedArray::tryCreateUninitialized(dataSize);
114 if (!byteArray) {
115 // FIXME: Does this need to be a "real" out of memory error with setOutOfMemoryError called on it?
116 return Exception { RangeError, "Out of memory"_s };
117 }
118 byteArray->zeroFill();
119
120 auto colorSpace = computeColorSpace(settings);
121 return adoptRef(*new ImageData(size, byteArray.releaseNonNull(), colorSpace));
122}
123
124ExceptionOr<Ref<ImageData>> ImageData::create(Ref<Uint8ClampedArray>&& byteArray, unsigned sw, std::optional<unsigned> sh, std::optional<ImageDataSettings> settings)
125{
126 unsigned length = byteArray->length();
127 if (!length || length % 4)
128 return Exception { InvalidStateError, "Length is not a non-zero multiple of 4"_s };
129
130 length /= 4;
131 if (!sw || length % sw)
132 return Exception { IndexSizeError, "Length is not a multiple of sw"_s };
133
134 unsigned height = length / sw;
135 if (sh && sh.value() != height)
136 return Exception { IndexSizeError, "sh value is not equal to height"_s };
137
138 IntSize size(sw, height);
139 auto dataSize = computeDataSize(size);
140 if (dataSize.hasOverflowed() || dataSize != byteArray->length())
141 return Exception { RangeError };
142
143 auto colorSpace = computeColorSpace(settings);
144 return adoptRef(*new ImageData(size, WTFMove(byteArray), colorSpace));
145}

156PixelBuffer ImageData::pixelBuffer() const
157{
158 PixelBufferFormat format { AlphaPremultiplication::Unpremultiplied, PixelFormat::RGBA8, toDestinationColorSpace(m_colorSpace) };
159 return { format, m_size, m_data.get() };
