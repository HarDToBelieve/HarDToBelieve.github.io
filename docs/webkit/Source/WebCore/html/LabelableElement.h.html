31#pragma once
32
33#include "HTMLElement.h"
34
35namespace WebCore {
36
37// LabelableElement represents "labelable element" defined in the HTML specification.
38class LabelableElement : public HTMLElement {
39 WTF_MAKE_ISO_ALLOCATED(LabelableElement);
40public:
41 virtual ~LabelableElement();
42
43 WEBCORE_EXPORT RefPtr<NodeList> labels();
44 virtual bool supportLabels() const { return false; }
45
46protected:
47 LabelableElement(const QualifiedName& tagName, Document&);
48
49private:
50 bool isLabelable() const final { return true; }
51};
52
53} // namespace WebCore
54
55SPECIALIZE_TYPE_TRAITS_BEGIN(WebCore::LabelableElement)
56 static bool isType(const WebCore::HTMLElement& element) { return element.isLabelable(); }
57 static bool isType(const WebCore::Node& node) { return is<WebCore::HTMLElement>(node) && isType(downcast<WebCore::HTMLElement>(node)); }
58SPECIALIZE_TYPE_TRAITS_END()
