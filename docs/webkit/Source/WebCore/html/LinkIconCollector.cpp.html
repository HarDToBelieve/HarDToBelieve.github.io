38const unsigned defaultTouchIconWidth = 60;
39
40static unsigned iconSize(const LinkIcon& icon)
41{
42 if (icon.size)
43 return *icon.size;
44
45 if (icon.type == LinkIconType::TouchIcon || icon.type == LinkIconType::TouchPrecomposedIcon)
46 return defaultTouchIconWidth;
47
48 return 0;
49}
50
51static int compareIcons(const LinkIcon& a, const LinkIcon& b)
52{
53 // Apple Touch icons always come first.
54 if (a.type == LinkIconType::Favicon && b.type != LinkIconType::Favicon)
55 return 1;
56 if (a.type == LinkIconType::Favicon && b.type != LinkIconType::Favicon)
57 return -1;
58
59 unsigned aSize = iconSize(a);
60 unsigned bSize = iconSize(b);
61
62 if (bSize > aSize)
63 return 1;
64 if (bSize < aSize)
65 return -1;
66
67 // A Precomposed icon should come first if both icons have the same size.
68 if (a.type != LinkIconType::TouchPrecomposedIcon && b.type == LinkIconType::TouchPrecomposedIcon)
69 return 1;
70 if (b.type != LinkIconType::TouchPrecomposedIcon && a.type == LinkIconType::TouchPrecomposedIcon)
71 return -1;
72
73 return 0;
74}

76auto LinkIconCollector::iconsOfTypes(OptionSet<LinkIconType> iconTypes) -> Vector<LinkIcon>
77{
78 auto head = makeRefPtr(m_document.head());
79 if (!head)
80 return { };
81
82 Vector<LinkIcon> icons;
83
84 for (auto& linkElement : childrenOfType<HTMLLinkElement>(*head)) {
85 if (!linkElement.iconType())
86 continue;
87
88 auto iconType = *linkElement.iconType();
89 if (!iconTypes.contains(iconType))
90 continue;
91
92 auto url = linkElement.href();
93 if (!url.protocolIsInHTTPFamily())
94 continue;
95
96 // This icon size parsing is a little wonky - it only parses the first
97 // part of the size, "60x70" becomes "60". This is for compatibility reasons
98 // and is probably good enough for now.
99 std::optional<unsigned> iconSize;
100 if (linkElement.sizes().length())
101 iconSize = parseIntegerAllowingTrailingJunk<unsigned>(linkElement.sizes().item(0));
102
103 Vector<std::pair<String, String>> attributes;
104 if (linkElement.hasAttributes()) {
105 attributes.reserveCapacity(linkElement.attributeCount());
106 for (const Attribute& attribute : linkElement.attributesIterator())
107 attributes.uncheckedAppend({ attribute.localName(), attribute.value() });
108 }
109
110 icons.append({ url, iconType, linkElement.type(), iconSize, WTFMove(attributes) });
111 }
112
113 std::sort(icons.begin(), icons.end(), [](auto& a, auto& b) {
114 return compareIcons(a, b) < 0;
115 });
116
117 return icons;
118}
