42
43struct LinkRelAttribute {
44 Markable<LinkIconType, EnumMarkableTraits<LinkIconType>> iconType;
45 bool isStyleSheet : 1;
46 bool isAlternate : 1;
47 bool isDNSPrefetch : 1;
48 bool isLinkPreload : 1;
49 bool isLinkPreconnect : 1;
50 bool isLinkPrefetch : 1;
51#if ENABLE(APPLICATION_MANIFEST)
52 bool isApplicationManifest : 1;
53#endif
54
55 LinkRelAttribute();
56 LinkRelAttribute(Document&, const String&);
57
58 static bool isSupported(Document&, StringView);
59};
