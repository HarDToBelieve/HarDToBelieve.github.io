85void MediaDocumentParser::createDocumentStructure()
86{
87 auto& document = *this->document();
88
89 auto rootElement = HTMLHtmlElement::create(document);
90 document.appendChild(rootElement);
91 document.setCSSTarget(rootElement.ptr());
92 rootElement->insertedByParser();
93
94 if (document.frame())
95 document.frame()->injectUserScripts(UserScriptInjectionTime::DocumentStart);
96
97#if PLATFORM(IOS_FAMILY)
98 auto headElement = HTMLHeadElement::create(document);
99 rootElement->appendChild(headElement);
100
101 auto metaElement = HTMLMetaElement::create(document);
102 metaElement->setAttributeWithoutSynchronization(nameAttr, AtomString("viewport", AtomString::ConstructFromLiteral));
103 metaElement->setAttributeWithoutSynchronization(contentAttr, AtomString("width=device-width,initial-scale=1", AtomString::ConstructFromLiteral));
104 headElement->appendChild(metaElement);
105#endif
106
107 auto body = HTMLBodyElement::create(document);
108 rootElement->appendChild(body);
109
110 auto videoElement = HTMLVideoElement::create(document);
111 m_mediaElement = videoElement.ptr();
112 videoElement->setAttributeWithoutSynchronization(controlsAttr, emptyAtom());
113 videoElement->setAttributeWithoutSynchronization(autoplayAttr, emptyAtom());
114 videoElement->setAttributeWithoutSynchronization(srcAttr, document.url().string());
115 if (auto loader = makeRefPtr(document.loader()))
116 videoElement->setAttributeWithoutSynchronization(typeAttr, loader->responseMIMEType());
117
118#if !ENABLE(MODERN_MEDIA_CONTROLS)
119 videoElement->setAttribute(styleAttr, AtomString("max-width: 100%; max-height: 100%;", AtomString::ConstructFromLiteral));
120#endif // !ENABLE(MODERN_MEDIA_CONTROLS)
121
122 body->appendChild(videoElement);
123 document.setHasVisuallyNonEmptyCustomContent();
124
125 RefPtr<Frame> frame = document.frame();
126 if (!frame)
127 return;
128
129 frame->loader().activeDocumentLoader()->setMainResourceDataBufferingPolicy(DataBufferingPolicy::DoNotBufferData);
130 frame->loader().setOutgoingReferrer(document.completeURL(m_outgoingReferrer));
131}

133void MediaDocumentParser::appendBytes(DocumentWriter&, const uint8_t*, size_t)
134{
135 if (m_mediaElement)
136 return;
137
138 createDocumentStructure();
139 finish();
140}

160static inline HTMLVideoElement* descendantVideoElement(ContainerNode& node)
161{
162 if (is<HTMLVideoElement>(node))
163 return downcast<HTMLVideoElement>(&node);
164
165 return descendantsOfType<HTMLVideoElement>(node).first();
166}

168static inline HTMLVideoElement* ancestorVideoElement(Node* node)
169{
170 while (node && !is<HTMLVideoElement>(*node))
171 node = node->parentOrShadowHostNode();
172
173 return downcast<HTMLVideoElement>(node);
174}

176void MediaDocument::defaultEventHandler(Event& event)
177{
178#if !ENABLE(MODERN_MEDIA_CONTROLS)
179 // Match the default Quicktime plugin behavior to allow
180 // clicking and double-clicking to pause and play the media.
181 if (!is<Node>(event.target()))
182 return;
183 auto& targetNode = downcast<Node>(*event.target());
184
185 if (auto video = makeRefPtr(ancestorVideoElement(&targetNode))) {
186 if (event.type() == eventNames().clickEvent) {
187 if (!video->canPlay()) {
188 video->pause();
189 event.setDefaultHandled();
190 }
191 } else if (event.type() == eventNames().dblclickEvent) {
192 if (video->canPlay()) {
193 video->play();
194 event.setDefaultHandled();
195 }
196 }
197 }
198
199 if (!is<ContainerNode>(targetNode))
200 return;
201 auto& targetContainer = downcast<ContainerNode>(targetNode);
202
203 if (event.type() == eventNames().keydownEvent && is<KeyboardEvent>(event)) {
204 auto video = makeRefPtr(descendantVideoElement(targetContainer));
205 if (!video)
206 return;
207
208 auto& keyboardEvent = downcast<KeyboardEvent>(event);
209 if (keyboardEvent.keyIdentifier() == "U+0020") { // space
210 if (video->paused()) {
211 if (video->canPlay())
212 video->play();
213 } else
214 video->pause();
215 keyboardEvent.setDefaultHandled();
216 }
217 }
218#else // !ENABLE(MODERN_MEDIA_CONTROLS)
219 UNUSED_PARAM(event);
220#endif
221}

223void MediaDocument::replaceMediaElementTimerFired()
224{
225 auto htmlBody = makeRefPtr(bodyOrFrameset());
226 if (!htmlBody)
227 return;
228
229 // Set body margin width and height to 0 as that is what a PluginDocument uses.
230 htmlBody->setAttributeWithoutSynchronization(marginwidthAttr, AtomString("0", AtomString::ConstructFromLiteral));
231 htmlBody->setAttributeWithoutSynchronization(marginheightAttr, AtomString("0", AtomString::ConstructFromLiteral));
232
233 if (auto videoElement = makeRefPtr(descendantVideoElement(*htmlBody))) {
234 auto embedElement = HTMLEmbedElement::create(*this);
235
236 embedElement->setAttributeWithoutSynchronization(widthAttr, AtomString("100%", AtomString::ConstructFromLiteral));
237 embedElement->setAttributeWithoutSynchronization(heightAttr, AtomString("100%", AtomString::ConstructFromLiteral));
238 embedElement->setAttributeWithoutSynchronization(nameAttr, AtomString("plugin", AtomString::ConstructFromLiteral));
239 embedElement->setAttributeWithoutSynchronization(srcAttr, url().string());
240
241 ASSERT(loader());
242 if (auto loader = makeRefPtr(this->loader()))
243 embedElement->setAttributeWithoutSynchronization(typeAttr, loader->writer().mimeType());
244
245 videoElement->parentNode()->replaceChild(embedElement, *videoElement);
246 }

255 return;

259}
