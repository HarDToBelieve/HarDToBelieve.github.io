82static String restrictionNames(MediaElementSession::BehaviorRestrictions restriction)
83{
84 StringBuilder restrictionBuilder;
85#define CASE(restrictionType) \
86 if (restriction & MediaElementSession::restrictionType) { \
87 if (!restrictionBuilder.isEmpty()) \
88 restrictionBuilder.append(", "); \
89 restrictionBuilder.append(#restrictionType); \
90 } \
91
92 CASE(NoRestrictions)
93 CASE(RequireUserGestureForLoad)
94 CASE(RequireUserGestureForVideoRateChange)
95 CASE(RequireUserGestureForAudioRateChange)
96 CASE(RequireUserGestureForFullscreen)
97 CASE(RequirePageConsentToLoadMedia)
98 CASE(RequirePageConsentToResumeMedia)
99 CASE(RequireUserGestureToShowPlaybackTargetPicker)
100 CASE(WirelessVideoPlaybackDisabled)
101 CASE(RequireUserGestureToAutoplayToExternalDevice)
102 CASE(AutoPreloadingNotPermitted)
103 CASE(InvisibleAutoplayNotPermitted)
104 CASE(OverrideUserGestureRequirementForMainContent)
105 CASE(RequireUserGestureToControlControlsManager)
106 CASE(RequirePlaybackToControlControlsManager)
107 CASE(RequireUserGestureForVideoDueToLowPowerMode)
108
109 return restrictionBuilder.toString();
110}

304void MediaElementSession::addBehaviorRestriction(BehaviorRestrictions restrictions)
305{
306 if (restrictions & ~m_restrictions)
307 INFO_LOG(LOGIDENTIFIER, "adding ", restrictionNames(restrictions & ~m_restrictions));
308
309 m_restrictions |= restrictions;
310
311 if (restrictions & OverrideUserGestureRequirementForMainContent)
312 m_mainContentCheckTimer.startRepeating(elementMainContentCheckInterval);
313}
314
315void MediaElementSession::removeBehaviorRestriction(BehaviorRestrictions restriction)
316{
317 if (restriction & RequireUserGestureToControlControlsManager) {
318 m_mostRecentUserInteractionTime = MonotonicTime::now();
319 if (auto page = m_element.document().page())
320 page->setAllowsPlaybackControlsForAutoplayingAudio(true);
321 }
322
323 if (!(m_restrictions & restriction))
324 return;
325
326 INFO_LOG(LOGIDENTIFIER, "removed ", restrictionNames(m_restrictions & restriction));
327 m_restrictions &= ~restriction;
328}

330Expected<void, MediaPlaybackDenialReason> MediaElementSession::playbackStateChangePermitted(MediaPlaybackState state) const
331{
332 INFO_LOG(LOGIDENTIFIER, "state = ", state);
333 if (m_element.isSuspended()) {
334 ALWAYS_LOG(LOGIDENTIFIER, "Returning FALSE because element is suspended");
335 return makeUnexpected(MediaPlaybackDenialReason::InvalidState);
336 }
337
338 auto& document = m_element.document();
339 auto* page = document.page();
340 if (!page || page->mediaPlaybackIsSuspended()) {
341 ALWAYS_LOG(LOGIDENTIFIER, "Returning FALSE because media playback is suspended");
342 return makeUnexpected(MediaPlaybackDenialReason::PageConsentRequired);
343 }
344
345 if (document.isMediaDocument() && !document.ownerElement())
346 return { };
347
348 if (pageExplicitlyAllowsElementToAutoplayInline(m_element))
349 return { };
350
351 if (requiresFullscreenForVideoPlayback() && !fullscreenPermitted()) {
352 ALWAYS_LOG(LOGIDENTIFIER, "Returning FALSE because of fullscreen restriction");
353 return makeUnexpected(MediaPlaybackDenialReason::FullscreenRequired);
354 }
355
356 if (m_restrictions & OverrideUserGestureRequirementForMainContent && updateIsMainContent())
357 return { };
358
359#if ENABLE(MEDIA_STREAM)
360 if (m_element.hasMediaStreamSrcObject()) {
361 if (document.isCapturing())
362 return { };
363 if (document.mediaState() & MediaProducer::MediaState::IsPlayingAudio)
364 return { };
365 }
366#endif
367
368 // FIXME: Why are we checking top-level document only for PerDocumentAutoplayBehavior?
369 const auto& topDocument = document.topDocument();
370 if (topDocument.quirks().requiresUserGestureToPauseInPictureInPicture()
371 && m_element.fullscreenMode() & HTMLMediaElementEnums::VideoFullscreenModePictureInPicture
372 && !m_element.paused() && state == MediaPlaybackState::Paused
373 && !document.processingUserGestureForMedia()) {
374 ALWAYS_LOG(LOGIDENTIFIER, "Returning FALSE because a quirk requires a user gesture to pause while in Picture-in-Picture");
375 return makeUnexpected(MediaPlaybackDenialReason::UserGestureRequired);
376 }
377
378 if (topDocument.mediaState() & MediaProducer::MediaState::HasUserInteractedWithMediaElement && topDocument.quirks().needsPerDocumentAutoplayBehavior())
379 return { };
380
381 if (topDocument.hasHadUserInteraction() && document.quirks().shouldAutoplayForArbitraryUserGesture())
382 return { };
383
384 if (m_restrictions & RequireUserGestureForVideoRateChange && m_element.isVideo() && !document.processingUserGestureForMedia()) {
385 ALWAYS_LOG(LOGIDENTIFIER, "Returning FALSE because a user gesture is required for video rate change restriction");
386 return makeUnexpected(MediaPlaybackDenialReason::UserGestureRequired);
387 }
388
389 if (m_restrictions & RequireUserGestureForAudioRateChange && (!m_element.isVideo() || m_element.hasAudio()) && !m_element.muted() && m_element.volume() && !document.processingUserGestureForMedia()) {
390 ALWAYS_LOG(LOGIDENTIFIER, "Returning FALSE because a user gesture is required for audio rate change restriction");
391 return makeUnexpected(MediaPlaybackDenialReason::UserGestureRequired);
392 }
393
394 if (m_restrictions & RequireUserGestureForVideoDueToLowPowerMode && m_element.isVideo() && !document.processingUserGestureForMedia()) {
395 ALWAYS_LOG(LOGIDENTIFIER, "Returning FALSE because of video low power mode restriction");
396 return makeUnexpected(MediaPlaybackDenialReason::UserGestureRequired);
397 }
398
399 return { };
400}

402bool MediaElementSession::autoplayPermitted() const
403{
404 const Document& document = m_element.document();
405 if (document.backForwardCacheState() != Document::NotInBackForwardCache)
406 return false;
407 if (document.activeDOMObjectsAreSuspended())
408 return false;
409
410 if (!hasBehaviorRestriction(MediaElementSession::InvisibleAutoplayNotPermitted))
411 return true;
412
413 // If the media element is audible, allow autoplay even when not visible as pausing it would be observable by the user.
414 if ((!m_element.isVideo() || m_element.hasAudio()) && !m_element.muted() && m_element.volume())
415 return true;
416
417 auto* renderer = m_element.renderer();
418 if (!renderer) {
419 ALWAYS_LOG(LOGIDENTIFIER, "Returning FALSE because element has no renderer");
420 return false;
421 }
422 if (renderer->style().visibility() != Visibility::Visible) {
423 ALWAYS_LOG(LOGIDENTIFIER, "Returning FALSE because element is not visible");
424 return false;
425 }
426 if (renderer->view().frameView().isOffscreen()) {
427 ALWAYS_LOG(LOGIDENTIFIER, "Returning FALSE because frame is offscreen");
428 return false;
429 }
430 if (renderer->visibleInViewportState() != VisibleInViewportState::Yes) {
431 ALWAYS_LOG(LOGIDENTIFIER, "Returning FALSE because element is not visible in the viewport");
432 return false;
433 }
434 return true;
435}

451{
452 if (isSuspended())
453 return MediaPlayer::BufferingPolicy::MakeResourcesPurgeable;
454
455 if (bufferingSuspended())
456 return MediaPlayer::BufferingPolicy::LimitReadAhead;
457
458 if (state() == PlatformMediaSession::Playing)
459 return MediaPlayer::BufferingPolicy::Default;
460
461 if (shouldOverrideBackgroundLoadingRestriction())
462 return MediaPlayer::BufferingPolicy::Default;
463
464#if ENABLE(WIRELESS_PLAYBACK_TARGET)
465 if (m_shouldPlayToPlaybackTarget)
466 return MediaPlayer::BufferingPolicy::Default;
467#endif
468
469 if (m_elementIsHiddenUntilVisibleInViewport || m_elementIsHiddenBecauseItWasRemovedFromDOM || m_element.elementIsHidden())
470 return MediaPlayer::BufferingPolicy::MakeResourcesPurgeable;
471
472 return MediaPlayer::BufferingPolicy::Default;
473}

507bool MediaElementSession::canShowControlsManager(PlaybackControlsPurpose purpose) const
508{
509 if (m_element.isSuspended() || !m_element.inActiveDocument()) {
510 INFO_LOG(LOGIDENTIFIER, "returning FALSE: isSuspended()");
511 return false;
512 }
513
514 if (m_element.isFullscreen()) {
515 INFO_LOG(LOGIDENTIFIER, "returning TRUE: is fullscreen");
516 return true;
517 }
518
519 if (m_element.muted()) {
520 INFO_LOG(LOGIDENTIFIER, "returning FALSE: muted");
521 return false;
522 }
523
524 if (m_element.document().isMediaDocument() && (m_element.document().frame() && m_element.document().frame()->isMainFrame())) {
525 INFO_LOG(LOGIDENTIFIER, "returning TRUE: is media document");
526 return true;
527 }
528
529 if (client().presentationType() == MediaType::Audio && (purpose == PlaybackControlsPurpose::ControlsManager || purpose == PlaybackControlsPurpose::MediaSession)) {
530 if (!hasBehaviorRestriction(RequireUserGestureToControlControlsManager) || m_element.document().processingUserGestureForMedia()) {
531 INFO_LOG(LOGIDENTIFIER, "returning TRUE: audio element with user gesture");
532 return true;
533 }
534
535 if (m_element.isPlaying() && allowsPlaybackControlsForAutoplayingAudio()) {
536 INFO_LOG(LOGIDENTIFIER, "returning TRUE: user has played media before");
537 return true;
538 }
539
540 INFO_LOG(LOGIDENTIFIER, "returning FALSE: audio element is not suitable");
541 return false;
542 }
543
544 if (purpose == PlaybackControlsPurpose::ControlsManager && !isElementRectMostlyInMainFrame(m_element)) {
545 INFO_LOG(LOGIDENTIFIER, "returning FALSE: not in main frame");
546 return false;
547 }
548
549 if (!m_element.hasAudio() && !m_element.hasEverHadAudio()) {
550 INFO_LOG(LOGIDENTIFIER, "returning FALSE: no audio");
551 return false;
552 }
553
554 if (!playbackStateChangePermitted(MediaPlaybackState::Playing)) {
555 INFO_LOG(LOGIDENTIFIER, "returning FALSE: playback not permitted");
556 return false;
557 }
558
559 if (!hasBehaviorRestriction(RequireUserGestureToControlControlsManager) || m_element.document().processingUserGestureForMedia()) {
560 INFO_LOG(LOGIDENTIFIER, "returning TRUE: no user gesture required");
561 return true;
562 }
563
564 if (purpose == PlaybackControlsPurpose::ControlsManager && hasBehaviorRestriction(RequirePlaybackToControlControlsManager) && !m_element.isPlaying()) {
565 INFO_LOG(LOGIDENTIFIER, "returning FALSE: needs to be playing");
566 return false;
567 }
568
569 if (purpose != PlaybackControlsPurpose::MediaSession && !m_element.hasEverNotifiedAboutPlaying()) {
570 INFO_LOG(LOGIDENTIFIER, "returning FALSE: hasn't fired playing notification");
571 return false;
572 }
573
574#if ENABLE(FULLSCREEN_API)
575 // Elements which are not descendants of the current fullscreen element cannot be main content.
576 auto* fullscreenElement = m_element.document().fullscreenManager().currentFullscreenElement();
577 if (fullscreenElement && !m_element.isDescendantOf(*fullscreenElement)) {
578 INFO_LOG(LOGIDENTIFIER, "returning FALSE: outside of full screen");
579 return false;
580 }
581#endif
582
583 // Only allow the main content heuristic to forbid videos from showing up if our purpose is the controls manager.
584 if (purpose == PlaybackControlsPurpose::ControlsManager && m_element.isVideo()) {
585 if (!m_element.renderer()) {
586 INFO_LOG(LOGIDENTIFIER, "returning FALSE: no renderer");
587 return false;
588 }
589
590 if (!m_element.hasVideo() && !m_element.hasEverHadVideo()) {
591 INFO_LOG(LOGIDENTIFIER, "returning FALSE: no video");
592 return false;
593 }
594
595 if (isLargeEnoughForMainContent(MediaSessionMainContentPurpose::MediaControls)) {
596 INFO_LOG(LOGIDENTIFIER, "returning TRUE: is main content");
597 return true;
598 }
599 }
600
601 if (purpose == PlaybackControlsPurpose::NowPlaying || purpose == PlaybackControlsPurpose::MediaSession) {
602 INFO_LOG(LOGIDENTIFIER, "returning TRUE: potentially plays audio");
603 return true;
604 }
605
606 INFO_LOG(LOGIDENTIFIER, "returning FALSE: no user gesture");
607 return false;
608}

669bool MediaElementSession::wirelessVideoPlaybackDisabled() const
670{
671 if (!m_element.document().settings().allowsAirPlayForMediaPlayback()) {
672 INFO_LOG(LOGIDENTIFIER, "returning TRUE because of settings");
673 return true;
674 }
675
676 if (m_element.hasAttributeWithoutSynchronization(HTMLNames::webkitwirelessvideoplaybackdisabledAttr)) {
677 INFO_LOG(LOGIDENTIFIER, "returning TRUE because of attribute");
678 return true;
679 }
680
681#if PLATFORM(IOS_FAMILY)
682 auto& legacyAirplayAttributeValue = m_element.attributeWithoutSynchronization(HTMLNames::webkitairplayAttr);
683 if (equalLettersIgnoringASCIICase(legacyAirplayAttributeValue, "deny")) {
684 INFO_LOG(LOGIDENTIFIER, "returning TRUE because of legacy attribute");
685 return true;
686 }
687 if (equalLettersIgnoringASCIICase(legacyAirplayAttributeValue, "allow")) {
688 INFO_LOG(LOGIDENTIFIER, "returning FALSE because of legacy attribute");
689 return false;
690 }
691#endif
692
693 if (m_element.document().settings().remotePlaybackEnabled() && m_element.hasAttributeWithoutSynchronization(HTMLNames::disableremoteplaybackAttr)) {
694 LOG(Media, "MediaElementSession::wirelessVideoPlaybackDisabled - returning TRUE because of RemotePlayback attribute");
695 return true;
696 }
697
698 auto player = m_element.player();
699 if (!player)
700 return true;
701
702 bool disabled = player->wirelessVideoPlaybackDisabled();
703 INFO_LOG(LOGIDENTIFIER, "returning ", disabled, " because media engine says so");
704
705 return disabled;
706}

708void MediaElementSession::setWirelessVideoPlaybackDisabled(bool disabled)

902size_t MediaElementSession::maximumMediaSourceBufferSize(const SourceBuffer& buffer) const
903{
904 // A good quality 1080p video uses 8,000 kbps and stereo audio uses 384 kbps, so assume 95% for video and 5% for audio.
905 const float bufferBudgetPercentageForVideo = .95;
906 const float bufferBudgetPercentageForAudio = .05;
907
908 size_t maximum = buffer.document().settings().maximumSourceBufferSize();
909
910 // Allow a SourceBuffer to buffer as though it is audio-only even if it doesn't have any active tracks (yet).
911 size_t bufferSize = static_cast<size_t>(maximum * bufferBudgetPercentageForAudio);
912 if (buffer.hasVideo())
913 bufferSize += static_cast<size_t>(maximum * bufferBudgetPercentageForVideo);
914
915 // FIXME: we might want to modify this algorithm to:
916 // - decrease the maximum size for background tabs
917 // - decrease the maximum size allowed for inactive elements when a process has more than one
918 // element, eg. so a page with many elements which are played one at a time doesn't keep
919 // everything buffered after an element has finished playing.
920
921 return bufferSize;
922}

925static bool isElementMainContentForPurposesOfAutoplay(const HTMLMediaElement& element, bool shouldHitTestMainFrame)
926{
927 Document& document = element.document();
928 if (!document.hasLivingRenderTree() || document.activeDOMObjectsAreStopped() || element.isSuspended() || !element.hasAudio() || !element.hasVideo())
929 return false;
930
931 // Elements which have not yet been laid out, or which are not yet in the DOM, cannot be main content.
932 auto* renderer = element.renderer();
933 if (!renderer)
934 return false;
935
936 if (!isElementLargeEnoughForMainContent(element, MediaSessionMainContentPurpose::Autoplay))
937 return false;
938
939 // Elements which are hidden by style, or have been scrolled out of view, cannot be main content.
940 // But elements which have audio & video and are already playing should not stop playing because
941 // they are scrolled off the page.
942 if (renderer->style().visibility() != Visibility::Visible)
943 return false;
944 if (renderer->visibleInViewportState() != VisibleInViewportState::Yes && !element.isPlaying())
945 return false;
946
947 // Main content elements must be in the main frame.
948 if (!document.frame() || !document.frame()->isMainFrame())
949 return false;
950
951 auto& mainFrame = document.frame()->mainFrame();
952 if (!mainFrame.view() || !mainFrame.view()->renderView())
953 return false;
954
955 if (!shouldHitTestMainFrame)
956 return true;
957
958 if (!mainFrame.document())
959 return false;
960
961 // Hit test the area of the main frame where the element appears, to determine if the element is being obscured.
962 // Elements which are obscured by other elements cannot be main content.
963 IntRect rectRelativeToView = element.clientRect();
964 ScrollPosition scrollPosition = mainFrame.view()->documentScrollPositionRelativeToViewOrigin();
965 IntRect rectRelativeToTopDocument(rectRelativeToView.location() + scrollPosition, rectRelativeToView.size());
966 OptionSet<HitTestRequest::Type> hitType { HitTestRequest::Type::ReadOnly, HitTestRequest::Type::Active, HitTestRequest::Type::AllowChildFrameContent, HitTestRequest::Type::IgnoreClipping, HitTestRequest::Type::DisallowUserAgentShadowContent };
967 HitTestResult result(rectRelativeToTopDocument.center());
968
969 mainFrame.document()->hitTest(hitType, result);
970 result.setToNonUserAgentShadowAncestor();
971 RefPtr<Element> hitElement = result.targetElement();
972 if (hitElement != &element)
973 return false;
974
975 return true;
976}

978static bool isElementRectMostlyInMainFrame(const HTMLMediaElement& element)
979{
980 if (!element.renderer())
981 return false;
982
983 auto documentFrame = makeRefPtr(element.document().frame());
984 if (!documentFrame)
985 return false;
986
987 auto mainFrameView = documentFrame->mainFrame().view();
988 if (!mainFrameView)
989 return false;
990
991 IntRect mainFrameRectAdjustedForScrollPosition = IntRect(-mainFrameView->documentScrollPositionRelativeToViewOrigin(), mainFrameView->contentsSize());
992 IntRect elementRectInMainFrame = element.clientRect();
993 auto totalElementArea = elementRectInMainFrame.area<RecordOverflow>();
994 if (totalElementArea.hasOverflowed())
995 return false;
996
997 elementRectInMainFrame.intersect(mainFrameRectAdjustedForScrollPosition);
998
999 return elementRectInMainFrame.area() > totalElementArea / 2;
1000}

1002static bool isElementLargeRelativeToMainFrame(const HTMLMediaElement& element)
1003{
1004 static const double minimumPercentageOfMainFrameAreaForMainContent = 0.9;
1005 auto* renderer = element.renderer();
1006 if (!renderer)
1007 return false;
1008
1009 auto documentFrame = makeRefPtr(element.document().frame());
1010 if (!documentFrame)
1011 return false;
1012
1013 if (!documentFrame->mainFrame().view())
1014 return false;
1015
1016 auto& mainFrameView = *documentFrame->mainFrame().view();
1017 auto maxVisibleClientWidth = std::min(renderer->clientWidth().toInt(), mainFrameView.visibleWidth());
1018 auto maxVisibleClientHeight = std::min(renderer->clientHeight().toInt(), mainFrameView.visibleHeight());
1019
1020 return maxVisibleClientWidth * maxVisibleClientHeight > minimumPercentageOfMainFrameAreaForMainContent * mainFrameView.visibleWidth() * mainFrameView.visibleHeight();
1021}

1023static bool isElementLargeEnoughForMainContent(const HTMLMediaElement& element, MediaSessionMainContentPurpose purpose)
1024{
1025 static const double elementMainContentAreaMinimum = 400 * 300;
1026 static const double maximumAspectRatio = purpose == MediaSessionMainContentPurpose::MediaControls ? 3 : 1.8;
1027 static const double minimumAspectRatio = .5; // Slightly smaller than 9:16.
1028
1029 // Elements which have not yet been laid out, or which are not yet in the DOM, cannot be main content.
1030 auto* renderer = element.renderer();
1031 if (!renderer)
1032 return false;
1033
1034 double width = renderer->clientWidth();
1035 double height = renderer->clientHeight();
1036 double area = width * height;
1037 double aspectRatio = width / height;
1038
1039 if (area < elementMainContentAreaMinimum)
1040 return false;
1041
1042 if (aspectRatio >= minimumAspectRatio && aspectRatio <= maximumAspectRatio)
1043 return true;
1044
1045 return isElementLargeRelativeToMainFrame(element);
1046}

1092 case PauseCommand:
1093 actionDetails.action = MediaSessionAction::Pause;

1096 actionDetails.action = MediaSessionAction::Stop;

1102 m_isScrubbing = true;
1103 return;
1104 case EndScrubbingCommand:
1105 m_isScrubbing = false;
1106 return;
1107 case SeekToPlaybackPositionCommand:
1108 ASSERT(argument.time);
1109 if (!argument.time)
1110 return;
1111 actionDetails.action = MediaSessionAction::Seekto;
1112 actionDetails.seekTime = argument.time.value();
1113 actionDetails.fastSeek = m_isScrubbing;
1114 break;
1115 case SkipForwardCommand:
1116 if (argument.time)
1117 actionDetails.seekOffset = argument.time.value();
1118 actionDetails.action = MediaSessionAction::Seekforward;
1119 break;
1120 case SkipBackwardCommand:
1121 if (argument.time)
1122 actionDetails.seekOffset = argument.time.value();
1123 actionDetails.action = MediaSessionAction::Seekbackward;
1124 break;
1125 case NextTrackCommand:
1126 actionDetails.action = MediaSessionAction::Nexttrack;
1127 break;
1128 case PreviousTrackCommand:
1129 actionDetails.action = MediaSessionAction::Previoustrack;
1130 break;

1132 case EndSeekingBackwardCommand:
1133 case BeginSeekingForwardCommand:
1134 case EndSeekingForwardCommand:
1135 ASSERT_NOT_REACHED();

1139 session->callActionHandler(actionDetails);

1170void MediaElementSession::updateMediaUsageIfChanged()
1171{
1172 auto& document = m_element.document();
1173 auto* page = document.page();
1174 if (!page || page->sessionID().isEphemeral())
1175 return;
1176
1177 bool isOutsideOfFullscreen = false;

1184 bool processingUserGesture = document.processingUserGestureForMedia();
1185 bool isPlaying = m_element.isPlaying();

1187 MediaUsageInfo usage = {
1188 m_element.currentSrc(),
1189 state() == PlatformMediaSession::Playing,
1190 canShowControlsManager(PlaybackControlsPurpose::ControlsManager),
1191 !page->isVisibleAndActive(),
1192 m_element.isSuspended(),

1194 m_element.isFullscreen(),

1200 m_element.hasAudio(),
1201 m_element.hasRenderer(),
1202 isAudio && hasBehaviorRestriction(RequireUserGestureToControlControlsManager) && !processingUserGesture,
