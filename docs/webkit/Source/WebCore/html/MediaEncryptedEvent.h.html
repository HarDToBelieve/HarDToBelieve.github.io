40namespace WebCore {
41
42class MediaEncryptedEvent final : public Event {
43 WTF_MAKE_ISO_ALLOCATED(MediaEncryptedEvent);
44public:
45 using Init = MediaEncryptedEventInit;
46
47 static Ref<MediaEncryptedEvent> create(const AtomString& type, const MediaEncryptedEventInit& initializer, IsTrusted isTrusted = IsTrusted::No)
48 {
49 return adoptRef(*new MediaEncryptedEvent(type, initializer, isTrusted));
50 }
51
52 virtual ~MediaEncryptedEvent();
53
54 String initDataType() { return m_initDataType; }
55 JSC::ArrayBuffer* initData() { return m_initData.get(); }
56
57private:
58 MediaEncryptedEvent(const AtomString&, const MediaEncryptedEventInit&, IsTrusted);
59
60 // Event
61 EventInterface eventInterface() const override;
62
63 String m_initDataType;
64 RefPtr<JSC::ArrayBuffer> m_initData;
65};
