42constexpr int secondsPerHour = 3600;
43constexpr int secondsPerMinute = 60;
44constexpr unsigned nptIdentifierLength = 4; // "npt:"

46static String collectDigits(const LChar* input, unsigned length, unsigned& position)
47{
48 StringBuilder digits;
49
50 // http://www.ietf.org/rfc/rfc2326.txt
51 // DIGIT ; any positive number
52 while (position < length && isASCIIDigit(input[position]))
53 digits.append(input[position++]);
54 return digits.toString();
55}
56
57static String collectFraction(const LChar* input, unsigned length, unsigned& position)
58{
59 StringBuilder digits;
60
61 // http://www.ietf.org/rfc/rfc2326.txt
62 // [ "." *DIGIT ]
63 if (input[position] != '.')
64 return String();
65
66 digits.append(input[position++]);
67 while (position < length && isASCIIDigit(input[position]))
68 digits.append(input[position++]);
69 return digits.toString();
70}

98void MediaFragmentURIParser::parseFragments()
99{
100 auto fragmentString = m_url.fragmentIdentifier();
101 if (fragmentString.isEmpty())
102 return;
103
104 unsigned offset = 0;
105 unsigned end = fragmentString.length();
106 while (offset < end) {
107 // http://www.w3.org/2008/WebVideo/Fragments/WD-media-fragments-spec/#processing-name-value-components
108 // 1. Parse the octet string according to the namevalues syntax, yielding a list of
109 // name-value pairs, where name and value are both octet string. In accordance
110 // with RFC 3986, the name and value components must be parsed and separated before
111 // percent-encoded octets are decoded.
112 size_t parameterStart = offset;
113 size_t parameterEnd = fragmentString.find('&', offset);
114 if (parameterEnd == notFound)
115 parameterEnd = end;
116
117 size_t equalOffset = fragmentString.find('=', offset);
118 if (equalOffset == notFound || equalOffset > parameterEnd) {
119 offset = parameterEnd + 1;
120 continue;
121 }
122
123 // 2. For each name-value pair:
124 // a. Decode percent-encoded octets in name and value as defined by RFC 3986. If either
125 // name or value are not valid percent-encoded strings, then remove the name-value pair
126 // from the list.
127 String name = decodeURLEscapeSequences(fragmentString.substring(parameterStart, equalOffset - parameterStart));
128 String value;
129 if (equalOffset != parameterEnd)
130 value = decodeURLEscapeSequences(fragmentString.substring(equalOffset + 1, parameterEnd - equalOffset - 1));
131
132 // b. Convert name and value to Unicode strings by interpreting them as UTF-8. If either
133 // name or value are not valid UTF-8 strings, then remove the name-value pair from the list.
134 bool validUTF8 = false;
135 if (!name.isEmpty() && !value.isEmpty()) {
136 name = name.utf8(StrictConversion).data();
137 validUTF8 = !name.isEmpty();
138
139 if (validUTF8) {
140 value = value.utf8(StrictConversion).data();
141 validUTF8 = !value.isEmpty();
142 }
143 }
144
145 if (validUTF8)
146 m_fragments.append(std::make_pair(name, value));
147
148 offset = parameterEnd + 1;
149 }
150}

152void MediaFragmentURIParser::parseTimeFragment()
153{
154 ASSERT(m_timeFormat == None);
155
156 if (m_fragments.isEmpty())
157 parseFragments();
158
159 m_timeFormat = Invalid;
160
161 for (auto& fragment : m_fragments) {
162 ASSERT(fragment.first.is8Bit());
163 ASSERT(fragment.second.is8Bit());
164
165 // http://www.w3.org/2008/WebVideo/Fragments/WD-media-fragments-spec/#naming-time
166 // Temporal clipping is denoted by the name t, and specified as an interval with a begin
167 // time and an end time
168 if (fragment.first != "t")
169 continue;
170
171 // http://www.w3.org/2008/WebVideo/Fragments/WD-media-fragments-spec/#npt-time
172 // Temporal clipping can be specified either as Normal Play Time (npt) RFC 2326, as SMPTE timecodes,
173 // SMPTE, or as real-world clock time (clock) RFC 2326. Begin and end times are always specified
174 // in the same format. The format is specified by name, followed by a colon (:), with npt: being
175 // the default.
176
177 MediaTime start = MediaTime::invalidTime();
178 MediaTime end = MediaTime::invalidTime();
179 if (parseNPTFragment(fragment.second.characters8(), fragment.second.length(), start, end)) {
180 m_startTime = start;
181 m_endTime = end;
182 m_timeFormat = NormalPlayTime;
183
184 // Although we have a valid fragment, don't return yet because when a fragment dimensions
185 // occurs multiple times, only the last occurrence of that dimension is used:
186 // http://www.w3.org/2008/WebVideo/Fragments/WD-media-fragments-spec/#error-uri-general
187 // Multiple occurrences of the same dimension: only the last valid occurrence of a dimension
188 // (e.g., t=10 in #t=2&t=10) is interpreted, all previous occurrences (valid or invalid)
189 // SHOULD be ignored by the UA.
190 }
191 }
192 m_fragments.clear();
193}

195bool MediaFragmentURIParser::parseNPTFragment(const LChar* timeString, unsigned length, MediaTime& startTime, MediaTime& endTime)
196{
197 unsigned offset = 0;
198 if (length >= nptIdentifierLength && timeString[0] == 'n' && timeString[1] == 'p' && timeString[2] == 't' && timeString[3] == ':')
199 offset += nptIdentifierLength;
200
201 if (offset == length)
202 return false;
203
204 // http://www.w3.org/2008/WebVideo/Fragments/WD-media-fragments-spec/#naming-time
205 // If a single number only is given, this corresponds to the begin time except if it is preceded
206 // by a comma that would in this case indicate the end time.
207 if (timeString[offset] == ',')
208 startTime = MediaTime::zeroTime();
209 else {
210 if (!parseNPTTime(timeString, length, offset, startTime))
211 return false;
212 }
213
214 if (offset == length)
215 return true;
216
217 if (timeString[offset] != ',')
218 return false;
219 if (++offset == length)
220 return false;
221
222 if (!parseNPTTime(timeString, length, offset, endTime))
223 return false;
224
225 if (offset != length)
226 return false;
227
228 if (startTime >= endTime)
229 return false;
230
231 return true;
232}

234bool MediaFragmentURIParser::parseNPTTime(const LChar* timeString, unsigned length, unsigned& offset, MediaTime& time)
235{
236 enum Mode { minutes, hours };
237 Mode mode = minutes;
238
239 if (offset >= length || !isASCIIDigit(timeString[offset]))
240 return false;
241
242 // http://www.w3.org/2008/WebVideo/Fragments/WD-media-fragments-spec/#npttimedef
243 // Normal Play Time can either be specified as seconds, with an optional
244 // fractional part to indicate milliseconds, or as colon-separated hours,
245 // minutes and seconds (again with an optional fraction). Minutes and
246 // seconds must be specified as exactly two digits, hours and fractional
247 // seconds can be any number of digits. The hours, minutes and seconds
248 // specification for NPT is a convenience only, it does not signal frame
249 // accuracy. The specification of the "npt:" identifier is optional since
250 // NPT is the default time scheme. This specification builds on the RTSP
251 // specification of NPT RFC 2326.
252 //
253 // ; defined in RFC 2326
254 // npt-sec = 1*DIGIT [ "." *DIGIT ] ; definitions taken
255 // npt-hhmmss = npt-hh ":" npt-mm ":" npt-ss [ "." *DIGIT] ; from RFC 2326
256 // npt-mmss = npt-mm ":" npt-ss [ "." *DIGIT]
257 // npt-hh = 1*DIGIT ; any positive number
258 // npt-mm = 2DIGIT ; 0-59
259 // npt-ss = 2DIGIT ; 0-59
260
261 String digits1 = collectDigits(timeString, length, offset);
262 int value1 = parseInteger<int>(digits1).value_or(0);
263 if (offset >= length || timeString[offset] == ',') {
264 time = MediaTime::createWithDouble(value1);
265 return true;
266 }
267
268 MediaTime fraction;
269 if (timeString[offset] == '.') {
270 if (offset == length)
271 return true;
272 String digits = collectFraction(timeString, length, offset);
273 fraction = MediaTime::createWithDouble(digits.toDouble());
274 time = MediaTime::createWithDouble(value1) + fraction;
275 return true;
276 }
277
278 if (digits1.length() < 2)
279 return false;
280 if (digits1.length() > 2)
281 mode = hours;
282
283 // Collect the next sequence of 0-9 after ':'
284 if (offset >= length || timeString[offset++] != ':')
285 return false;
286 if (offset >= length || !isASCIIDigit(timeString[(offset)]))
287 return false;
288 String digits2 = collectDigits(timeString, length, offset);
289 if (digits2.length() != 2)
290 return false;
291 int value2 = parseInteger<int>(digits2).value();
292
293 // Detect whether this timestamp includes hours.
294 int value3;
295 if (mode == hours || (offset < length && timeString[offset] == ':')) {
296 if (offset >= length || timeString[offset++] != ':')
297 return false;
298 if (offset >= length || !isASCIIDigit(timeString[offset]))
299 return false;
300 String digits3 = collectDigits(timeString, length, offset);
301 if (digits3.length() != 2)
302 return false;
303 value3 = parseInteger<int>(digits3).value();
304 } else {
305 value3 = value2;
306 value2 = value1;
307 value1 = 0;
308 }
309
310 if (offset < length && timeString[offset] == '.')
311 fraction = MediaTime::createWithDouble(collectFraction(timeString, length, offset).toDouble());
312
313 time = MediaTime::createWithDouble((value1 * secondsPerHour) + (value2 * secondsPerMinute) + value3) + fraction;
314 return true;
315}
