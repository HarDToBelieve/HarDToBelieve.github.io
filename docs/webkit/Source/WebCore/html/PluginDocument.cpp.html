67void PluginDocumentParser::createDocumentStructure()
68{
69 auto& document = downcast<PluginDocument>(*this->document());
70
71 auto rootElement = HTMLHtmlElement::create(document);
72 document.appendChild(rootElement);
73 rootElement->insertedByParser();
74
75 if (document.frame())
76 document.frame()->injectUserScripts(UserScriptInjectionTime::DocumentStart);
77
78#if PLATFORM(IOS_FAMILY)
79 // Should not be able to zoom into standalone plug-in documents.
80 document.processViewport("user-scalable=no"_s, ViewportArguments::PluginDocument);
81#endif
82
83 auto body = HTMLBodyElement::create(document);
84 body->setAttributeWithoutSynchronization(marginwidthAttr, AtomString("0", AtomString::ConstructFromLiteral));
85 body->setAttributeWithoutSynchronization(marginheightAttr, AtomString("0", AtomString::ConstructFromLiteral));
86#if PLATFORM(IOS_FAMILY)
87 body->setAttribute(styleAttr, AtomString("background-color: rgb(217,224,233)", AtomString::ConstructFromLiteral));
88#else
89 body->setAttribute(styleAttr, AtomString("background-color: rgb(38,38,38)", AtomString::ConstructFromLiteral));
90#endif
91
92 rootElement->appendChild(body);
93
94 auto embedElement = HTMLEmbedElement::create(document);
95
96 m_embedElement = embedElement.ptr();
97 embedElement->setAttributeWithoutSynchronization(widthAttr, AtomString("100%", AtomString::ConstructFromLiteral));
98 embedElement->setAttributeWithoutSynchronization(heightAttr, AtomString("100%", AtomString::ConstructFromLiteral));
99
100 embedElement->setAttributeWithoutSynchronization(nameAttr, AtomString("plugin", AtomString::ConstructFromLiteral));
101 embedElement->setAttributeWithoutSynchronization(srcAttr, document.url().string());
102
103 ASSERT(document.loader());
104 if (auto loader = makeRefPtr(document.loader()))
105 m_embedElement->setAttributeWithoutSynchronization(typeAttr, loader->writer().mimeType());
106
107 document.setPluginElement(*m_embedElement);
108
109 body->appendChild(embedElement);
110 document.setHasVisuallyNonEmptyCustomContent();
111}

113void PluginDocumentParser::appendBytes(DocumentWriter&, const uint8_t*, size_t)
114{
115 if (m_embedElement)
116 return;
117
118 createDocumentStructure();
119
120 auto frame = makeRefPtr(document()->frame());
121 if (!frame)
122 return;
123
124 document()->updateLayout();
125
126 // Below we assume that renderer->widget() to have been created by
127 // document()->updateLayout(). However, in some cases, updateLayout() will
128 // recurse too many times and delay its post-layout tasks (such as creating
129 // the widget). Here we kick off the pending post-layout tasks so that we
130 // can synchronously redirect data to the plugin.
131 frame->view()->flushAnyPendingPostLayoutTasks();
132
133 if (auto renderer = m_embedElement->renderWidget()) {
134 if (auto widget = makeRefPtr(renderer->widget())) {
135 frame->loader().client().redirectDataToPlugin(*widget);
136
137 // In a plugin document, the main resource is the plugin. If we have a null widget, that means
138 // the loading of the plugin was cancelled, which gives us a null mainResourceLoader(), so we
139 // need to have this call in a null check of the widget or of mainResourceLoader().
140 if (auto loader = frame->loader().activeDocumentLoader())
141 loader->setMainResourceDataBufferingPolicy(DataBufferingPolicy::DoNotBufferData);
142 }
143 }
144}

153Ref<DocumentParser> PluginDocument::createParser()
154{
155 return PluginDocumentParser::create(*this);
156}
157
158Widget* PluginDocument::pluginWidget()
159{
160 if (!m_pluginElement)
161 return nullptr;
162 auto* renderer = m_pluginElement->renderer();
163 if (!renderer)
164 return nullptr;
165 return downcast<RenderEmbeddedObject>(*m_pluginElement->renderer()).widget();
166}

168void PluginDocument::setPluginElement(HTMLPlugInElement& element)
169{
170 m_pluginElement = &element;
171}

173void PluginDocument::detachFromPluginElement()
174{
175 // Release the plugin Element so that we don't have a circular reference.
176 m_pluginElement = nullptr;
177}

179void PluginDocument::cancelManualPluginLoad()
180{
181 // PluginDocument::cancelManualPluginLoad should only be called once, but there are issues
182 // with how many times we call beforeload on object elements. <rdar://problem/8441094>.
183 if (!shouldLoadPluginManually())
184 return;
185
186 auto& frameLoader = frame()->loader();
187 if (auto documentLoader = frameLoader.activeDocumentLoader())
188 documentLoader->cancelMainResourceLoad(frameLoader.cancelledError(documentLoader->request()));
189 m_shouldLoadPluginManually = false;
190}
