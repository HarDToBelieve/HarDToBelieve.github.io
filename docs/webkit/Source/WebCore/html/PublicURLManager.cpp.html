36std::unique_ptr<PublicURLManager> PublicURLManager::create(ScriptExecutionContext* context)
37{
38 auto publicURLManager = makeUnique<PublicURLManager>(context);
39 publicURLManager->suspendIfNeeded();
40 return publicURLManager;
41}
42
43PublicURLManager::PublicURLManager(ScriptExecutionContext* context)
44 : ActiveDOMObject(context)
45 , m_isStopped(false)
46{
47}
48
49void PublicURLManager::registerURL(const URL& url, URLRegistrable& registrable)
50{
51 if (m_isStopped)
52 return;
53
54 RegistryURLMap::iterator found = m_registryToURL.add(&registrable.registry(), URLSet()).iterator;
55 found->key->registerURL(*scriptExecutionContext(), url, registrable);
56 found->value.add(url.string());
57}
58
59void PublicURLManager::revoke(const URL& url)
60{
61 for (auto& registry : m_registryToURL) {
62 if (registry.value.contains(url.string())) {
63 registry.key->unregisterURL(url);
64 registry.value.remove(url.string());
65 break;
66 }
67 }
68}
69
70void PublicURLManager::stop()
71{
72 if (m_isStopped)
73 return;
74
75 m_isStopped = true;
76 for (auto& registry : m_registryToURL) {
77 for (auto& url : registry.value)
78 registry.key->unregisterURL(URL({ }, url));
79 }
80
81 m_registryToURL.clear();
82}
83
84const char* PublicURLManager::activeDOMObjectName() const
85{
86 return "PublicURLManager";
87}
