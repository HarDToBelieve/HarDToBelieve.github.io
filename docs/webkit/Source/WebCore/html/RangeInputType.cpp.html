70static const int rangeDefaultMinimum = 0;
71static const int rangeDefaultMaximum = 100;
72static const int rangeDefaultStep = 1;
73static const int rangeDefaultStepBase = 0;
74static const int rangeStepScaleFactor = 1;
75static const StepRange::StepDescription rangeStepDescription { rangeDefaultStep, rangeDefaultStepBase, rangeStepScaleFactor };
76
77static Decimal ensureMaximum(const Decimal& proposedValue, const Decimal& minimum, const Decimal& fallbackValue)
78{
79 return proposedValue >= minimum ? proposedValue : std::max(minimum, fallbackValue);
80}

116StepRange RangeInputType::createStepRange(AnyStepHandling anyStepHandling) const
117{
118 ASSERT(element());
119 const Decimal minimum = parseToNumber(element()->attributeWithoutSynchronization(minAttr), rangeDefaultMinimum);
120 const Decimal maximum = ensureMaximum(parseToNumber(element()->attributeWithoutSynchronization(maxAttr), rangeDefaultMaximum), minimum, rangeDefaultMaximum);
121
122 const AtomString& precisionValue = element()->attributeWithoutSynchronization(precisionAttr);
123 if (!precisionValue.isNull()) {
124 const Decimal step = equalLettersIgnoringASCIICase(precisionValue, "float") ? Decimal::nan() : 1;
125 return StepRange(minimum, RangeLimitations::Valid, minimum, maximum, step, rangeStepDescription);
126 }
127
128 const Decimal step = StepRange::parseStep(anyStepHandling, rangeStepDescription, element()->attributeWithoutSynchronization(stepAttr));
129 return StepRange(minimum, RangeLimitations::Valid, minimum, maximum, step, rangeStepDescription);
130}

132void RangeInputType::handleMouseDownEvent(MouseEvent& event)
133{
134 ASSERT(element());
135 if (element()->isDisabledFormControl())
136 return;
137
138 if (event.button() != LeftButton || !is<Node>(event.target()))
139 return;
140 ASSERT(element()->shadowRoot());
141 auto& targetNode = downcast<Node>(*event.target());
142 if (&targetNode != element() && !targetNode.isDescendantOf(element()->userAgentShadowRoot().get()))
143 return;
144 auto& thumb = typedSliderThumbElement();
145 if (&targetNode == &thumb)
146 return;
147 thumb.dragFrom(event.absoluteLocation());
148}

188auto RangeInputType::handleKeydownEvent(KeyboardEvent& event) -> ShouldCallBaseEventHandler
189{
190 ASSERT(element());
191 if (element()->isDisabledFormControl())
192 return ShouldCallBaseEventHandler::Yes;
193
194 const String& key = event.keyIdentifier();
195
196 const Decimal current = parseToNumberOrNaN(element()->value());
197 ASSERT(current.isFinite());
198
199 auto stepRange = createStepRange(AnyStepHandling::Reject);
200
201 // FIXME: We can't use stepUp() for the step value "any". So, we increase
202 // or decrease the value by 1/100 of the value range. Is it reasonable?
203 const Decimal step = equalLettersIgnoringASCIICase(element()->attributeWithoutSynchronization(stepAttr), "any") ? (stepRange.maximum() - stepRange.minimum()) / 100 : stepRange.step();
204 const Decimal bigStep = std::max((stepRange.maximum() - stepRange.minimum()) / 10, step);
205
206 bool isVertical = false;
207 if (auto* renderer = element()->renderer()) {
208 ControlPart part = renderer->style().appearance();
209 isVertical = part == SliderVerticalPart || part == MediaVolumeSliderPart;
210 }
211
212 Decimal newValue;
213 if (key == "Up")
214 newValue = current + step;
215 else if (key == "Down")
216 newValue = current - step;
217 else if (key == "Left")
218 newValue = isVertical ? current + step : current - step;
219 else if (key == "Right")
220 newValue = isVertical ? current - step : current + step;
221 else if (key == "PageUp")
222 newValue = current + bigStep;
223 else if (key == "PageDown")
224 newValue = current - bigStep;
225 else if (key == "Home")
226 newValue = isVertical ? stepRange.maximum() : stepRange.minimum();
227 else if (key == "End")
228 newValue = isVertical ? stepRange.minimum() : stepRange.maximum();
229 else
230 return ShouldCallBaseEventHandler::Yes; // Did not match any key binding.
231
232 newValue = stepRange.clampValue(newValue);
233
234 if (newValue != current) {
235 EventQueueScope scope;
236 setValueAsDecimal(newValue, DispatchInputAndChangeEvent);
237
238 if (AXObjectCache* cache = element()->document().existingAXObjectCache())
239 cache->postNotification(element(), AXObjectCache::AXValueChanged);
240 }
241
242 event.setDefaultHandled();
243 return ShouldCallBaseEventHandler::Yes;
244}

246void RangeInputType::createShadowSubtreeAndUpdateInnerTextElementEditability(ContainerNode::ChildChange::Source source, bool)
247{
248 ASSERT(needsShadowSubtree());
249 ASSERT(element());
250 ASSERT(element()->userAgentShadowRoot());
251
252 static MainThreadNeverDestroyed<const AtomString> webkitSliderRunnableTrackName("-webkit-slider-runnable-track", AtomString::ConstructFromLiteral);
253 Document& document = element()->document();
254 auto track = HTMLDivElement::create(document);
255 track->setPseudo(webkitSliderRunnableTrackName);
256 track->appendChild(source, SliderThumbElement::create(document));
257 auto container = SliderContainerElement::create(document);
258 container->appendChild(source, track);
259 element()->userAgentShadowRoot()->appendChild(source, container);
260}

262HTMLElement* RangeInputType::sliderTrackElement() const
263{
264 ASSERT(element());
265 ASSERT(element()->userAgentShadowRoot());
266 ASSERT(element()->userAgentShadowRoot()->firstChild()); // container
267 ASSERT(element()->userAgentShadowRoot()->firstChild()->isHTMLElement());
268 ASSERT(element()->userAgentShadowRoot()->firstChild()->firstChild()); // track
269
270 RefPtr<ShadowRoot> root = element()->userAgentShadowRoot();
271 if (!root)
272 return nullptr;
273
274 auto* container = childrenOfType<SliderContainerElement>(*root).first();
275 if (!container)
276 return nullptr;
277
278 return childrenOfType<HTMLElement>(*container).first();
279}

378void RangeInputType::updateTickMarkValues()
379{
380 if (!m_tickMarkValuesDirty)
381 return;
382 m_tickMarkValues.clear();
383 m_tickMarkValuesDirty = false;
384 ASSERT(element());
385 auto dataList = element()->dataList();
386 if (!dataList)
387 return;
388 Ref<HTMLCollection> options = dataList->options();
389 m_tickMarkValues.reserveCapacity(options->length());
390 for (unsigned i = 0; i < options->length(); ++i) {
391 RefPtr<Node> node = options->item(i);
392 HTMLOptionElement& optionElement = downcast<HTMLOptionElement>(*node);
393 String optionValue = optionElement.value();
394 if (!element()->isValidValue(optionValue))
395 continue;
396 m_tickMarkValues.append(parseToNumber(optionValue, Decimal::nan()));
397 }
398 m_tickMarkValues.shrinkToFit();
399 std::sort(m_tickMarkValues.begin(), m_tickMarkValues.end());
400}

402std::optional<Decimal> RangeInputType::findClosestTickMarkValue(const Decimal& value)
403{
404 updateTickMarkValues();
405 if (!m_tickMarkValues.size())
406 return std::nullopt;
407
408 size_t left = 0;
409 size_t right = m_tickMarkValues.size();
410 size_t middle;
411 while (true) {
412 ASSERT(left <= right);
413 middle = left + (right - left) / 2;
414 if (!middle)
415 break;
416 if (middle == m_tickMarkValues.size() - 1 && m_tickMarkValues[middle] < value) {
417 middle++;
418 break;
419 }
420 if (m_tickMarkValues[middle - 1] <= value && m_tickMarkValues[middle] >= value)
421 break;
422
423 if (m_tickMarkValues[middle] < value)
424 left = middle;
425 else
426 right = middle;
427 }
428
429 std::optional<Decimal> closestLeft = middle ? std::make_optional(m_tickMarkValues[middle - 1]) : std::nullopt;
430 std::optional<Decimal> closestRight = middle != m_tickMarkValues.size() ? std::make_optional(m_tickMarkValues[middle]) : std::nullopt;
431
432 if (!closestLeft)
433 return closestRight;
434 if (!closestRight)
435 return closestLeft;
436
437 if (*closestRight - value < value - *closestLeft)
438 return closestRight;
439
440 return closestLeft;
441}
