56RenderPtr<RenderElement> RubyTextElement::createElementRenderer(RenderStyle&& style, const RenderTreePosition& insertionPosition)
57{
58 // RenderRubyText requires its parent to be RenderRubyRun.
59 if (isRuby(insertionPosition.parent()) && style.display() == DisplayType::Block)
60 return createRenderer<RenderRubyText>(*this, WTFMove(style));
61 return HTMLElement::createElementRenderer(WTFMove(style), insertionPosition);
62}
