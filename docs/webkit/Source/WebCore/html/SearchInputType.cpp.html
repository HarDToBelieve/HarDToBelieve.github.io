54void SearchInputType::addSearchResult()
55{
56#if !PLATFORM(IOS_FAMILY)
57 // Normally we've got the correct renderer by the time we get here. However when the input type changes
58 // we don't update the associated renderers until after the next tree update, so we could actually end up here
59 // with a mismatched renderer (e.g. through form submission).
60 ASSERT(element());
61 if (is<RenderSearchField>(element()->renderer()))
62 downcast<RenderSearchField>(*element()->renderer()).addSearchResult();
63#endif
64}

66static void updateResultButtonPseudoType(SearchFieldResultsButtonElement& resultButton, int maxResults)
67{
68 static MainThreadNeverDestroyed<const AtomString> webkitSearchDecorationName("-webkit-search-decoration", AtomString::ConstructFromLiteral);
69 static MainThreadNeverDestroyed<const AtomString> webkitSearchResultsDecorationName("-webkit-search-results-decoration", AtomString::ConstructFromLiteral);
70 static MainThreadNeverDestroyed<const AtomString> webkitSearchResultsButtonName("-webkit-search-results-button", AtomString::ConstructFromLiteral);
71 if (!maxResults)
72 resultButton.setPseudo(webkitSearchResultsDecorationName);
73 else if (maxResults < 0)
74 resultButton.setPseudo(webkitSearchDecorationName);
75 else
76 resultButton.setPseudo(webkitSearchResultsButtonName);
77}

79void SearchInputType::attributeChanged(const QualifiedName& name)
80{
81 if (name == resultsAttr) {
82 if (m_resultsButton) {
83 if (auto* element = this->element())
84 updateResultButtonPseudoType(*m_resultsButton, element->maxResults());
85 }
86 }
87 BaseTextInputType::attributeChanged(name);
88}

106void SearchInputType::createShadowSubtreeAndUpdateInnerTextElementEditability(ContainerNode::ChildChange::Source source, bool isInnerTextElementEditable)
107{
108 ASSERT(needsShadowSubtree());
109 ASSERT(!m_resultsButton);
110 ASSERT(!m_cancelButton);
111
112 TextFieldInputType::createShadowSubtreeAndUpdateInnerTextElementEditability(source, isInnerTextElementEditable);
113 RefPtr<HTMLElement> container = containerElement();
114 RefPtr<HTMLElement> textWrapper = innerBlockElement();
115 ASSERT(container);
116 ASSERT(textWrapper);
117
118 ASSERT(element());
119 m_resultsButton = SearchFieldResultsButtonElement::create(element()->document());
120 updateResultButtonPseudoType(*m_resultsButton, element()->maxResults());
121 container->insertBefore(*m_resultsButton, textWrapper.get());
122
123 m_cancelButton = SearchFieldCancelButtonElement::create(element()->document());
124 container->insertBefore(*m_cancelButton, textWrapper->nextSibling());
125}

137auto SearchInputType::handleKeydownEvent(KeyboardEvent& event) -> ShouldCallBaseEventHandler
138{
139 ASSERT(element());
140 if (element()->isDisabledOrReadOnly())
141 return TextFieldInputType::handleKeydownEvent(event);
142
143 const String& key = event.keyIdentifier();
144 if (key == "U+001B") {
145 Ref<HTMLInputElement> protectedInputElement(*element());
146 protectedInputElement->setValueForUser(emptyString());
147 protectedInputElement->onSearch();
148 event.setDefaultHandled();
149 return ShouldCallBaseEventHandler::Yes;
150 }
151 return TextFieldInputType::handleKeydownEvent(event);
152}

161void SearchInputType::startSearchEventTimer()
162{
163 ASSERT(element());
164 ASSERT(element()->renderer());
165 unsigned length = element()->innerTextValue().length();
166
167 if (!length) {
168 m_searchEventTimer.startOneShot(0_ms);
169 return;
170 }
171
172 // After typing the first key, we wait 0.5 seconds.
173 // After the second key, 0.4 seconds, then 0.3, then 0.2 from then on.
174 m_searchEventTimer.startOneShot(std::max(200_ms, 600_ms - 100_ms * length));
175}

193
194void SearchInputType::didSetValueByUserEdit()
195{
196 ASSERT(element());
197 if (m_cancelButton && is<RenderSearchField>(element()->renderer()))
198 downcast<RenderSearchField>(*element()->renderer()).updateCancelButtonVisibility();
199 // If the incremental attribute is set, then dispatch the search event
200 if (searchEventsShouldBeDispatched())
201 startSearchEventTimer();
202
203 TextFieldInputType::didSetValueByUserEdit();
204}

213float SearchInputType::decorationWidth() const
214{
215 float width = 0;
216 if (m_resultsButton)
217 width += m_resultsButton->computedStyle()->logicalWidth().value();
218 if (m_cancelButton)
219 width += m_cancelButton->computedStyle()->logicalWidth().value();
220 return width;
221}
