21#pragma once
22
23#include "Decimal.h"
24#include <wtf/Forward.h>
25
26namespace WebCore {
27
28enum class AnyStepHandling : bool { Reject, Default };
29
30enum class RangeLimitations : bool { Valid, Invalid };
31
32class StepRange {
33public:
34 enum StepValueShouldBe {
35 StepValueShouldBeReal,
36 ParsedStepValueShouldBeInteger,
37 ScaledStepValueShouldBeInteger,
38 };
39
40 struct StepDescription {
41 WTF_MAKE_FAST_ALLOCATED;
42 public:
43 int defaultStep { 1 };
44 int defaultStepBase { 0 };
45 int stepScaleFactor { 1 };
46 StepValueShouldBe stepValueShouldBe { StepValueShouldBeReal };
47
48 constexpr StepDescription(int defaultStep, int defaultStepBase, int stepScaleFactor, StepValueShouldBe stepValueShouldBe = StepValueShouldBeReal)
49 : defaultStep(defaultStep)
50 , defaultStepBase(defaultStepBase)
51 , stepScaleFactor(stepScaleFactor)
52 , stepValueShouldBe(stepValueShouldBe)
53 {
54 }
55
56 StepDescription() = default;
57
58 Decimal defaultValue() const
59 {
60 return defaultStep * stepScaleFactor;
61 }
62 };
63
64 StepRange();
65 StepRange(const StepRange&);
66 StepRange(const Decimal& stepBase, RangeLimitations, const Decimal& minimum, const Decimal& maximum, const Decimal& step, const StepDescription&);
67 Decimal acceptableError() const;
68 Decimal alignValueForStep(const Decimal& currentValue, const Decimal& newValue) const;
69 Decimal clampValue(const Decimal& value) const;
70 bool hasStep() const { return m_hasStep; }
71 bool hasRangeLimitations() const { return m_hasRangeLimitations; }
72 Decimal maximum() const { return m_maximum; }
73 Decimal minimum() const { return m_minimum; }
74 static Decimal parseStep(AnyStepHandling, const StepDescription&, const String&);
75 Decimal step() const { return m_step; }
76 Decimal stepBase() const { return m_stepBase; }
77 int stepScaleFactor() const { return m_stepDescription.stepScaleFactor; }
78 bool stepMismatch(const Decimal&) const;
79
80 // Clamp the middle value according to the step
81 Decimal defaultValue() const
82 {
83 return clampValue((m_minimum + m_maximum) / 2);
84 }
85
86 // Map value into 0-1 range
87 Decimal proportionFromValue(const Decimal& value) const
88 {
89 if (m_minimum == m_maximum)
90 return 0;
91
92 return (value - m_minimum) / (m_maximum - m_minimum);
93 }
94
95 // Map from 0-1 range to value
96 Decimal valueFromProportion(const Decimal& proportion) const
97 {
98 return m_minimum + proportion * (m_maximum - m_minimum);
99 }
100
101private:
102 StepRange& operator =(const StepRange&);
103 Decimal roundByStep(const Decimal& value, const Decimal& base) const;
104
105 const Decimal m_maximum; // maximum must be >= minimum.
106 const Decimal m_minimum;
107 const Decimal m_step;
108 const Decimal m_stepBase;
109 const StepDescription m_stepDescription;
110 const bool m_hasRangeLimitations { false };
111 const bool m_hasStep { false };
112};
113
114} // namespace WebCore
