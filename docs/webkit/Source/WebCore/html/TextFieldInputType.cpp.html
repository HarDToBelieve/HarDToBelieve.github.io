106bool TextFieldInputType::isEmptyValue() const
107{
108 auto innerText = innerTextElement();
109 ASSERT(innerText);
110
111 for (Text* text = TextNodeTraversal::firstWithin(*innerText); text; text = TextNodeTraversal::next(*text, innerText.get())) {
112 if (text->length())
113 return false;
114 }
115 return true;
116}

124void TextFieldInputType::setValue(const String& sanitizedValue, bool valueChanged, TextFieldEventBehavior eventBehavior)
125{
126 ASSERT(element());
127
128 // Grab this input element to keep reference even if JS event handler
129 // changes input type.
130 Ref<HTMLInputElement> input(*element());
131
132 // We don't ask InputType::setValue to dispatch events because
133 // TextFieldInputType dispatches events different way from InputType.
134 InputType::setValue(sanitizedValue, valueChanged, DispatchNoEvent);
135
136 if (valueChanged)
137 updateInnerTextValue();
138
139 unsigned max = visibleValue().length();
140 if (input->focused())
141 input->setSelectionRange(max, max);
142 else
143 input->cacheSelectionInResponseToSetValue(max);
144
145 if (!valueChanged)
146 return;
147
148 switch (eventBehavior) {
149 case DispatchChangeEvent:
150 // If the user is still editing this field, dispatch an input event rather than a change event.
151 // The change event will be dispatched when editing finishes.
152 if (input->focused())
153 input->dispatchFormControlInputEvent();
154 else
155 input->dispatchFormControlChangeEvent();
156 break;
157
158 case DispatchInputAndChangeEvent: {
159 input->dispatchFormControlInputEvent();
160 input->dispatchFormControlChangeEvent();
161 break;
162 }
163
164 case DispatchNoEvent:
165 break;
166 }
167
168 // FIXME: Why do we do this when eventBehavior == DispatchNoEvent
169 if (!input->focused() || eventBehavior == DispatchNoEvent)
170 input->setTextAsOfLastFormControlChangeEvent(sanitizedValue);
171
172 if (UserTypingGestureIndicator::processingUserTypingGesture())
173 didSetValueByUserEdit();
174}

184auto TextFieldInputType::handleKeydownEvent(KeyboardEvent& event) -> ShouldCallBaseEventHandler
185{
186 ASSERT(element());
187 if (!element()->focused())
188 return ShouldCallBaseEventHandler::Yes;
189#if ENABLE(DATALIST_ELEMENT)
190 const String& key = event.keyIdentifier();
191 if (m_suggestionPicker && (key == "Enter" || key == "Up" || key == "Down")) {
192 m_suggestionPicker->handleKeydownWithIdentifier(key);
193 event.setDefaultHandled();
194 }
195#endif
196 RefPtr<Frame> frame = element()->document().frame();
197 if (frame && frame->editor().doTextFieldCommandFromEvent(element(), &event))
198 event.setDefaultHandled();
199 return ShouldCallBaseEventHandler::Yes;
200}

202void TextFieldInputType::handleKeydownEventForSpinButton(KeyboardEvent& event)
203{
204 ASSERT(element());
205 if (element()->isDisabledOrReadOnly())
206 return;
207#if ENABLE(DATALIST_ELEMENT)
208 if (m_suggestionPicker)
209 return;
210#endif
211 const String& key = event.keyIdentifier();
212 if (key == "Up")
213 spinButtonStepUp();
214 else if (key == "Down")
215 spinButtonStepDown();
216 else
217 return;
218 event.setDefaultHandled();
219}

221void TextFieldInputType::forwardEvent(Event& event)
222{
223 if (m_innerSpinButton) {
224 m_innerSpinButton->forwardEvent(event);
225 if (event.defaultHandled())
226 return;
227 }
228
229 bool isFocusEvent = event.type() == eventNames().focusEvent;
230 bool isBlurEvent = event.type() == eventNames().blurEvent;
231 if (isFocusEvent || isBlurEvent)
232 capsLockStateMayHaveChanged();
233 if (event.isMouseEvent() || isFocusEvent || isBlurEvent) {
234 ASSERT(element());
235 element()->forwardEvent(event);
236 }
237}

239void TextFieldInputType::elementDidBlur()
240{
241 ASSERT(element());
242 auto* renderer = element()->renderer();
243 if (!renderer)
244 return;
245
246 auto* innerTextRenderer = innerTextElement()->renderer();
247 if (!innerTextRenderer)
248 return;
249
250 auto* innerLayer = innerTextRenderer->layer();
251 if (!innerLayer)
252 return;
253
254 auto* innerLayerScrollable = innerLayer->ensureLayerScrollableArea();
255
256 bool isLeftToRightDirection = downcast<RenderTextControlSingleLine>(*renderer).style().isLeftToRightDirection();
257 ScrollOffset scrollOffset(isLeftToRightDirection ? 0 : innerLayerScrollable->scrollWidth(), 0);
258 innerLayerScrollable->scrollToOffset(scrollOffset);
259
260#if ENABLE(DATALIST_ELEMENT)
261 closeSuggestions();
262#endif
263}

318void TextFieldInputType::createShadowSubtreeAndUpdateInnerTextElementEditability(ContainerNode::ChildChange::Source source, bool isInnerTextElementEditable)
319{
320 ASSERT(needsShadowSubtree());
321 ASSERT(element());
322 ASSERT(element()->shadowRoot());
323
324 ASSERT(!m_innerText);
325 ASSERT(!m_innerBlock);
326 ASSERT(!m_innerSpinButton);
327 ASSERT(!m_capsLockIndicator);
328 ASSERT(!m_autoFillButton);
329
330 Document& document = element()->document();
331 bool shouldHaveSpinButton = this->shouldHaveSpinButton();
332 bool shouldHaveCapsLockIndicator = this->shouldHaveCapsLockIndicator();
333 bool createsContainer = shouldHaveSpinButton || shouldHaveCapsLockIndicator || needsContainer();
334
335 m_innerText = TextControlInnerTextElement::create(document, isInnerTextElementEditable);
336
337 if (!createsContainer) {
338 element()->userAgentShadowRoot()->appendChild(source, *m_innerText);
339 updatePlaceholderText();
340 return;
341 }
342
343 createContainer();
344 updatePlaceholderText();
345
346 if (shouldHaveSpinButton) {
347 m_innerSpinButton = SpinButtonElement::create(document, *this);
348 m_container->appendChild(source, *m_innerSpinButton);
349 }
350
351 if (shouldHaveCapsLockIndicator) {
352 static MainThreadNeverDestroyed<const AtomString> webkitCapsLockIndicatorName("-webkit-caps-lock-indicator", AtomString::ConstructFromLiteral);
353 m_capsLockIndicator = HTMLDivElement::create(document);
354 m_capsLockIndicator->setPseudo(webkitCapsLockIndicatorName);
355
356 bool shouldDrawCapsLockIndicator = this->shouldDrawCapsLockIndicator();
357 m_capsLockIndicator->setInlineStyleProperty(CSSPropertyDisplay, shouldDrawCapsLockIndicator ? CSSValueBlock : CSSValueNone, true);
358
359 m_container->appendChild(source, *m_capsLockIndicator);
360 }
361 updateAutoFillButton();
362}

483static String limitLength(const String& string, unsigned maxNumGraphemeClusters)
484{
485 StringView stringView { string };
486
487 if (!stringView.is8Bit())
488 maxNumGraphemeClusters = numCodeUnitsInGraphemeClusters(stringView, maxNumGraphemeClusters);
489
490 return string.left(maxNumGraphemeClusters);
491}

566void TextFieldInputType::handleBeforeTextInsertedEvent(BeforeTextInsertedEvent& event)
567{
568 ASSERT(element());
569 // Make sure that the text to be inserted will not violate the maxLength.
570
571 // We use RenderTextControlSingleLine::text() instead of InputElement::value()
572 // because they can be mismatched by sanitizeValue() in
573 // HTMLInputElement::subtreeHasChanged() in some cases.
574 String innerText = element()->innerTextValue();
575 unsigned oldLength = numGraphemeClusters(innerText);
576
577 // selectionLength represents the selection length of this text field to be
578 // removed by this insertion.
579 // If the text field has no focus, we don't need to take account of the
580 // selection length. The selection is the source of text drag-and-drop in
581 // that case, and nothing in the text field will be removed.
582 unsigned selectionLength = 0;
583 if (element()->focused()) {
584 ASSERT(enclosingTextFormControl(element()->document().frame()->selection().selection().start()) == element());
585 int selectionStart = element()->selectionStart();
586 ASSERT(selectionStart <= element()->selectionEnd());
587 int selectionCodeUnitCount = element()->selectionEnd() - selectionStart;
588 selectionLength = selectionCodeUnitCount ? numGraphemeClusters(StringView(innerText).substring(selectionStart, selectionCodeUnitCount)) : 0;
589 }
590 ASSERT(oldLength >= selectionLength);
591
592 // Selected characters will be removed by the next text event.
593 unsigned baseLength = oldLength - selectionLength;
594 unsigned maxLength = isTextType() ? element()->effectiveMaxLength() : HTMLInputElement::maxEffectiveLength;
595 unsigned appendableLength = maxLength > baseLength ? maxLength - baseLength : 0;
596
597 // Truncate the inserted text to avoid violating the maxLength and other constraints.
598 String eventText = event.text();
599 unsigned textLength = eventText.length();
600 while (textLength > 0 && isHTMLLineBreak(eventText[textLength - 1]))
601 textLength--;
602 eventText.truncate(textLength);
603 eventText.replace("\r\n", " ");
604 eventText.replace('\r', ' ');
605 eventText.replace('\n', ' ');
606 event.setText(limitLength(eventText, appendableLength));
607}

618void TextFieldInputType::updatePlaceholderText()
619{
620 if (!supportsPlaceholder())
621 return;
622 ASSERT(element());
623 String placeholderText = element()->strippedPlaceholder();
624 if (placeholderText.isEmpty()) {
625 if (m_placeholder) {
626 m_placeholder->parentNode()->removeChild(*m_placeholder);
627 m_placeholder = nullptr;
628 }
629 return;
630 }
631 if (!m_placeholder) {
632 m_placeholder = TextControlPlaceholderElement::create(element()->document());
633 element()->userAgentShadowRoot()->insertBefore(*m_placeholder, m_container ? m_container.get() : innerTextElement().get());
634 }
635 m_placeholder->setInnerText(placeholderText);
636}

653void TextFieldInputType::subtreeHasChanged()
654{
655 ASSERT(element());
656 element()->setChangedSinceLastFormControlChangeEvent(true);
657
658 // We don't need to call sanitizeUserInputValue() function here because
659 // HTMLInputElement::handleBeforeTextInsertedEvent() has already called
660 // sanitizeUserInputValue().
661 // ---
662 // sanitizeValue() is needed because IME input doesn't dispatch BeforeTextInsertedEvent.
663 // ---
664 // Input types that support the selection API do *not* sanitize their
665 // user input in order to retain parity between what's in the model and
666 // what's on the screen. Otherwise, we retain the sanitization process for
667 // backward compatibility. https://bugs.webkit.org/show_bug.cgi?id=150346
668 String innerText = convertFromVisibleValue(element()->innerTextValue());
669 if (!supportsSelectionAPI())
670 innerText = sanitizeValue(innerText);
671 element()->setValueFromRenderer(innerText);
672 element()->updatePlaceholderVisibility();
673 // Recalc for :invalid change.
674 element()->invalidateStyleForSubtree();
675
676 didSetValueByUserEdit();
677}

679void TextFieldInputType::didSetValueByUserEdit()
680{
681 ASSERT(element());
682 if (!element()->focused())
683 return;
684 if (RefPtr<Frame> frame = element()->document().frame())
685 frame->editor().textDidChangeInTextField(element());
686#if ENABLE(DATALIST_ELEMENT)
687 if (shouldOnlyShowDataListDropdownButtonWhenFocusedOrEdited() && element()->list() && m_dataListDropdownIndicator)
688 m_dataListDropdownIndicator->setInlineStyleProperty(CSSPropertyDisplay, suggestions().size() ? CSSValueBlock : CSSValueNone, true);
689
690 if (element()->list())
691 displaySuggestions(DataListSuggestionActivationType::TextChanged);
692#endif
693}

736bool TextFieldInputType::shouldDrawCapsLockIndicator() const
737{
738 ASSERT(element());
739 if (element()->document().focusedElement() != element())
740 return false;
741
742 if (element()->isDisabledOrReadOnly())
743 return false;
744
745 if (element()->hasAutoFillStrongPasswordButton())
746 return false;
747
748 RefPtr<Frame> frame = element()->document().frame();
749 if (!frame)
750 return false;
751
752 if (!frame->selection().isFocusedAndActive())
753 return false;
754
755 return PlatformKeyboardEvent::currentCapsLockState();
756}

884Vector<DataListSuggestion> TextFieldInputType::suggestions()
885{
886 // FIXME: Suggestions are "typing completions" and so should probably use the findPlainText algorithm rather than the simplistic "ignoring ASCII case" rules.
887
888 Vector<DataListSuggestion> suggestions;
889 Vector<DataListSuggestion> matchesContainingValue;
890
891 String elementValue = element()->value();
892
893 if (!m_cachedSuggestions.first.isNull() && equalIgnoringASCIICase(m_cachedSuggestions.first, elementValue))
894 return m_cachedSuggestions.second;
895
896 auto* page = element()->document().page();
897 bool canShowLabels = page && page->chrome().client().canShowDataListSuggestionLabels();
898 if (auto dataList = element()->dataList()) {
899 for (auto& option : dataList->suggestions()) {
900 DataListSuggestion suggestion;
901 suggestion.value = option.value();
902 if (!element()->isValidValue(suggestion.value))
903 continue;
904 suggestion.value = sanitizeValue(suggestion.value);
905 suggestion.label = option.label();
906 if (suggestion.value == suggestion.label)
907 suggestion.label = { };
908
909 if (elementValue.isEmpty() || suggestion.value.startsWithIgnoringASCIICase(elementValue))
910 suggestions.append(WTFMove(suggestion));
911 else if (suggestion.value.containsIgnoringASCIICase(elementValue) || (canShowLabels && suggestion.label.containsIgnoringASCIICase(elementValue)))
912 matchesContainingValue.append(WTFMove(suggestion));
913 }
914 }
915
916 suggestions.appendVector(WTFMove(matchesContainingValue));
917 m_cachedSuggestions = std::make_pair(elementValue, suggestions);
918
919 return suggestions;
920}
