70Decimal TimeInputType::defaultValueForStepUp() const
71{
72 double current = WallTime::now().secondsSinceEpoch().milliseconds();
73 int offset = calculateLocalTimeOffset(current).offset / msPerMinute;
74 current += offset * msPerMinute;
75
76 auto date = DateComponents::fromMillisecondsSinceMidnight(current);
77 if (!date)
78 return { };
79
80 double milliseconds = date->millisecondsSinceEpoch();
81 ASSERT(std::isfinite(milliseconds));
82 return Decimal::fromDouble(milliseconds);
83}

85StepRange TimeInputType::createStepRange(AnyStepHandling anyStepHandling) const
86{
87 ASSERT(element());
88 const Decimal stepBase = parseToNumber(element()->attributeWithoutSynchronization(minAttr), 0);
89 const Decimal minimum = parseToNumber(element()->attributeWithoutSynchronization(minAttr), Decimal::fromDouble(DateComponents::minimumTime()));
90 const Decimal maximum = parseToNumber(element()->attributeWithoutSynchronization(maxAttr), Decimal::fromDouble(DateComponents::maximumTime()));
91 const Decimal step = StepRange::parseStep(anyStepHandling, timeStepDescription, element()->attributeWithoutSynchronization(stepAttr));
92 return StepRange(stepBase, RangeLimitations::Valid, minimum, maximum, step, timeStepDescription);
93}

114String TimeInputType::formatDateTimeFieldsState(const DateTimeFieldsState& state) const
115{
116 if (!state.hour || !state.minute || !state.meridiem)
117 return emptyString();
118
119 auto hourMinuteString = makeString(pad('0', 2, *state.hour23()), ':', pad('0', 2, *state.minute));
120
121 if (state.millisecond)
122 return makeString(hourMinuteString, ':', pad('0', 2, state.second ? *state.second : 0), '.', pad('0', 3, *state.millisecond));
123
124 if (state.second)
125 return makeString(hourMinuteString, ':', pad('0', 2, *state.second));
126
127 return hourMinuteString;
128}

130void TimeInputType::setupLayoutParameters(DateTimeEditElement::LayoutParameters& layoutParameters, const DateComponents& date) const
131{
132 layoutParameters.shouldHaveMillisecondField = shouldHaveMillisecondField(date);
133
134 if (layoutParameters.shouldHaveMillisecondField || shouldHaveSecondField(date)) {
135 layoutParameters.dateTimeFormat = layoutParameters.locale.timeFormat();
136 layoutParameters.fallbackDateTimeFormat = "HH:mm:ss"_s;
137 } else {
138 layoutParameters.dateTimeFormat = layoutParameters.locale.shortTimeFormat();
139 layoutParameters.fallbackDateTimeFormat = "HH:mm"_s;
140 }
141}
