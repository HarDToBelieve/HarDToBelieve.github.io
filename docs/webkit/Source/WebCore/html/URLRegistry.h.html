31#pragma once
32
33#include <wtf/text/WTFString.h>
34
35namespace WebCore {
36
37class ScriptExecutionContext;
38class URLRegistry;
39
40class URLRegistrable {
41public:
42 virtual ~URLRegistrable() = default;
43 virtual URLRegistry& registry() const = 0;
44};
45
46class URLRegistry {
47 WTF_MAKE_FAST_ALLOCATED;
48public:
49 virtual ~URLRegistry() = default;
50 virtual void registerURL(ScriptExecutionContext&, const URL&, URLRegistrable&) = 0;
51 virtual void unregisterURL(const URL&) = 0;
52
53 // This is an optional API
54 virtual URLRegistrable* lookup(const String&) const { ASSERT_NOT_REACHED(); return 0; }
55};
56
57} // namespace WebCore
