27#include "ExceptionOr.h"
28#include <wtf/Variant.h>
29#include <wtf/Vector.h>
30#include <wtf/text/WTFString.h>

32namespace WebCore {
33
34class DOMURL;
35
36class URLSearchParams : public RefCounted<URLSearchParams> {
37public:
38 static ExceptionOr<Ref<URLSearchParams>> create(Variant<Vector<Vector<String>>, Vector<WTF::KeyValuePair<String, String>>, String>&&);
39 static Ref<URLSearchParams> create(const String& string, DOMURL* associatedURL)
40 {
41 return adoptRef(*new URLSearchParams(string, associatedURL));
42 }
43
44 void associatedURLDestroyed() { m_associatedURL = nullptr; }
45 void append(const String& name, const String& value);
46 void remove(const String& name);
47 String get(const String& name) const;
48 Vector<String> getAll(const String& name) const;
49 bool has(const String& name) const;
50 void set(const String& name, const String& value);
51 String toString() const;
52 void updateFromAssociatedURL();
53 void sort();
54
55 class Iterator {
56 public:
57 explicit Iterator(URLSearchParams&);
58 std::optional<WTF::KeyValuePair<String, String>> next();
59
60 private:
61 Ref<URLSearchParams> m_target;
62 size_t m_index { 0 };
63 };
64 Iterator createIterator() { return Iterator { *this }; }
65
66private:
67 const Vector<WTF::KeyValuePair<String, String>>& pairs() const { return m_pairs; }
68 URLSearchParams(const String&, DOMURL*);
69 URLSearchParams(const Vector<WTF::KeyValuePair<String, String>>&);
70 void updateURL();
71
72 DOMURL* m_associatedURL { nullptr };
73 Vector<WTF::KeyValuePair<String, String>> m_pairs;
74};

76} // namespace WebCore
