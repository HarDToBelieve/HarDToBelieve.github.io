55ValidationMessage::ValidationMessage(HTMLFormControlElement* element)
56 : m_element(element)
57{
58 ASSERT(m_element);

61ValidationMessage::~ValidationMessage()
62{
63 if (ValidationMessageClient* client = validationMessageClient()) {
64 client->hideValidationMessage(*m_element);
65 return;
66 }
67
68 deleteBubbleTree();
69}

71ValidationMessageClient* ValidationMessage::validationMessageClient() const
72{
73 if (Page* page = m_element->document().page())
74 return page->validationMessageClient();
75 return 0;
76}

78void ValidationMessage::updateValidationMessage(const String& message)
79{
80 // We want to hide the validation message as soon as the user starts
81 // typing, even if a constraint is still violated. Thefore, we hide the message instead
82 // of updating it if it is already visible.
83 if (isVisible()) {
84 requestToHideMessage();
85 return;
86 }
87
88 String updatedMessage = message;
89 if (!validationMessageClient()) {
90 // HTML5 specification doesn't ask UA to show the title attribute value
91 // with the validationMessage. However, this behavior is same as Opera
92 // and the specification describes such behavior as an example.
93 if (!updatedMessage.isEmpty()) {
94 const AtomString& title = m_element->attributeWithoutSynchronization(titleAttr);
95 if (!title.isEmpty())
96 updatedMessage = updatedMessage + '\n' + title;
97 }
98 }
99
100 if (updatedMessage.isEmpty()) {
101 requestToHideMessage();
102 return;
103 }
104 setMessage(updatedMessage);
105}

107void ValidationMessage::setMessage(const String& message)
108{
109 if (ValidationMessageClient* client = validationMessageClient()) {
110 client->showValidationMessage(*m_element, message);
111 return;
112 }
113
114 // Don't modify the DOM tree in this context.
115 // If so, an assertion in Element::isFocusable() fails.
116 ASSERT(!message.isEmpty());
117 m_message = message;
118 if (!m_bubble)
119 m_timer = makeUnique<Timer>(*this, &ValidationMessage::buildBubbleTree);
120 else
121 m_timer = makeUnique<Timer>(*this, &ValidationMessage::setMessageDOMAndStartTimer);
122 m_timer->startOneShot(0_s);
123}
124
125void ValidationMessage::setMessageDOMAndStartTimer()
126{
127 ASSERT(!validationMessageClient());
128 ASSERT(m_messageHeading);
129 ASSERT(m_messageBody);
130 m_messageHeading->removeChildren();
131 m_messageBody->removeChildren();
132 Vector<String> lines = m_message.split('\n');
133 Document& document = m_messageHeading->document();
134 for (unsigned i = 0; i < lines.size(); ++i) {
135 if (i) {
136 m_messageBody->appendChild(Text::create(document, lines[i]));
137 if (i < lines.size() - 1)
138 m_messageBody->appendChild(HTMLBRElement::create(document));
139 } else
140 m_messageHeading->setInnerText(lines[i]);
141 }
142
143 int magnification = document.page() ? document.page()->settings().validationMessageTimerMagnification() : -1;
144 if (magnification <= 0)
145 m_timer = nullptr;
146 else {
147 m_timer = makeUnique<Timer>(*this, &ValidationMessage::deleteBubbleTree);
148 m_timer->startOneShot(std::max(5_s, 1_ms * static_cast<double>(m_message.length()) * magnification));
149 }
150}

152void ValidationMessage::adjustBubblePosition()
153{
154 if (!m_bubble)
155 return;
156 if (!m_element->renderer())
157 return;
158 LayoutRect hostRect = m_element->renderer()->absoluteBoundingBoxRect();
159 if (hostRect.isEmpty())
160 return;
161 double hostX = hostRect.x();
162 double hostY = hostRect.y();
163 if (RenderObject* renderer = m_bubble->renderer()) {
164 if (RenderBox* container = renderer->containingBlock()) {
165 FloatPoint containerLocation = container->localToAbsolute();
166 hostX -= containerLocation.x() + container->borderLeft();
167 hostY -= containerLocation.y() + container->borderTop();
168 }
169 }
170
171 m_bubble->setInlineStyleProperty(CSSPropertyTop, hostY + hostRect.height(), CSSUnitType::CSS_PX);
172 // The 'left' value of ::-webkit-validation-bubble-arrow.
173 const int bubbleArrowTopOffset = 32;
174 double bubbleX = hostX;
175 if (hostRect.width() / 2 < bubbleArrowTopOffset)
176 bubbleX = std::max(hostX + hostRect.width() / 2 - bubbleArrowTopOffset, 0.0);
177 m_bubble->setInlineStyleProperty(CSSPropertyLeft, bubbleX, CSSUnitType::CSS_PX);
178}

180void ValidationMessage::buildBubbleTree()
181{
182 ASSERT(!validationMessageClient());
183
184 if (!m_element->renderer())
185 return;
186
187 auto shadowRoot = makeRef(m_element->ensureUserAgentShadowRoot());
188
189 ScriptDisallowedScope::InMainThread scriptDisallowedScope;
190 ScriptDisallowedScope::EventAllowedScope allowedScope(shadowRoot);
191
192 static MainThreadNeverDestroyed<const AtomString> webkitValidationBubbleName("-webkit-validation-bubble", AtomString::ConstructFromLiteral);
193 static MainThreadNeverDestroyed<const AtomString> webkitValidationBubbleArrowClipperName("-webkit-validation-bubble-arrow-clipper", AtomString::ConstructFromLiteral);
194 static MainThreadNeverDestroyed<const AtomString> webkitValidationBubbleArrowName("-webkit-validation-bubble-arrow", AtomString::ConstructFromLiteral);
195 static MainThreadNeverDestroyed<const AtomString> webkitValidationBubbleMessageName("-webkit-validation-bubble-message", AtomString::ConstructFromLiteral);
196 static MainThreadNeverDestroyed<const AtomString> webkitValidationBubbleIconName("-webkit-validation-bubble-icon", AtomString::ConstructFromLiteral);
197 static MainThreadNeverDestroyed<const AtomString> webkitValidationBubbleTextBlockName("-webkit-validation-bubble-text-block", AtomString::ConstructFromLiteral);
198 static MainThreadNeverDestroyed<const AtomString> webkitValidationBubbleHeadingName("-webkit-validation-bubble-heading", AtomString::ConstructFromLiteral);
199 static MainThreadNeverDestroyed<const AtomString> webkitValidationBubbleBodyName("-webkit-validation-bubble-body", AtomString::ConstructFromLiteral);
200
201 Document& document = m_element->document();
202 m_bubble = HTMLDivElement::create(document);
203 shadowRoot->appendChild(*m_bubble);
204 m_bubble->setPseudo(webkitValidationBubbleName);
205 // Need to force position:absolute because RenderMenuList doesn't assume it
206 // contains non-absolute or non-fixed renderers as children.
207 m_bubble->setInlineStyleProperty(CSSPropertyPosition, CSSValueAbsolute);
208
209 auto clipper = HTMLDivElement::create(document);
210 m_bubble->appendChild(clipper);
211 clipper->setPseudo(webkitValidationBubbleArrowClipperName);
212 auto bubbleArrow = HTMLDivElement::create(document);
213 clipper->appendChild(bubbleArrow);
214 bubbleArrow->setPseudo(webkitValidationBubbleArrowName);
215
216 auto message = HTMLDivElement::create(document);
217 m_bubble->appendChild(message);
218 message->setPseudo(webkitValidationBubbleMessageName);
219 auto icon = HTMLDivElement::create(document);
220 message->appendChild(icon);
221 icon->setPseudo(webkitValidationBubbleIconName);
222 auto textBlock = HTMLDivElement::create(document);
223 message->appendChild(textBlock);
224 textBlock->setPseudo(webkitValidationBubbleTextBlockName);
225 m_messageHeading = HTMLDivElement::create(document);
226 textBlock->appendChild(*m_messageHeading);
227 m_messageHeading->setPseudo(webkitValidationBubbleHeadingName);
228 m_messageBody = HTMLDivElement::create(document);
229 textBlock->appendChild(*m_messageBody);
230 m_messageBody->setPseudo(webkitValidationBubbleBodyName);
231
232 setMessageDOMAndStartTimer();
233
234 // FIXME: Use transition to show the bubble.
235
236 if (!document.view())
237 return;
238 document.view()->queuePostLayoutCallback([weakThis = makeWeakPtr(*this)] {
239 if (!weakThis)
240 return;
241 weakThis->adjustBubblePosition();
242 });
243}

245void ValidationMessage::requestToHideMessage()
246{
247 if (ValidationMessageClient* client = validationMessageClient()) {
248 client->hideValidationMessage(*m_element);
249 return;
250 }
251
252 // We must not modify the DOM tree in this context by the same reason as setMessage().
253 m_timer = makeUnique<Timer>(*this, &ValidationMessage::deleteBubbleTree);
254 m_timer->startOneShot(0_s);
255}

257bool ValidationMessage::shadowTreeContains(const Node& node) const
258{
259 if (validationMessageClient() || !m_bubble)
260 return false;
261 return &m_bubble->treeScope() == &node.treeScope();
262}
263
264void ValidationMessage::deleteBubbleTree()
265{
266 ASSERT(!validationMessageClient());
267 if (m_bubble) {
268 ScriptDisallowedScope::EventAllowedScope allowedScope(*m_element->userAgentShadowRoot());
269 m_messageHeading = nullptr;
270 m_messageBody = nullptr;
271 m_element->userAgentShadowRoot()->removeChild(*m_bubble);
272 m_bubble = nullptr;
273 }
274 m_message = String();
275}
276
277bool ValidationMessage::isVisible() const
278{
279 if (ValidationMessageClient* client = validationMessageClient())
280 return client->isValidationMessageVisible(*m_element);
281 return !m_message.isEmpty();
282}
