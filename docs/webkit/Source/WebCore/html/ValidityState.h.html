23#pragma once
24
25#include "FormAssociatedElement.h"
26
27namespace WebCore {
28
29// ValidityState is not a separate object, but rather an interface of FormAssociatedElement that
30// is published as part of the DOM. We could implement this as a base class of FormAssociatedElement,
31// but that would have a small runtime cost, and no significant benefit. We'd prefer to implement this
32// as a typedef of FormAssociatedElement, but that would require changes to bindings generation.
33class ValidityState : public FormAssociatedElement {
34};
35
36inline ValidityState* FormAssociatedElement::validity()
37{
38 // Because ValidityState adds nothing to FormAssociatedElement, we rely on it being safe
39 // to cast a FormAssociatedElement like this, even though it's not actually a ValidityState.
40 return static_cast<ValidityState*>(this);
41}

43} // namespace WebCore
