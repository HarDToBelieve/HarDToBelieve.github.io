46static const int weekDefaultStepBase = -259200000; // The first day of 1970-W01.
47static const int weekDefaultStep = 1;
48static const int weekStepScaleFactor = 604800000;
49static const StepRange::StepDescription weekStepDescription { weekDefaultStep, weekDefaultStepBase, weekStepScaleFactor, StepRange::ParsedStepValueShouldBeInteger };
50
51const AtomString& WeekInputType::formControlType() const
52{
53 return InputTypeNames::week();
54}
55
56DateComponentsType WeekInputType::dateType() const
57{
58 return DateComponentsType::Week;
59}

61StepRange WeekInputType::createStepRange(AnyStepHandling anyStepHandling) const
62{
63 ASSERT(element());
64 const Decimal stepBase = parseToNumber(element()->attributeWithoutSynchronization(minAttr), weekDefaultStepBase);
65 const Decimal minimum = parseToNumber(element()->attributeWithoutSynchronization(minAttr), Decimal::fromDouble(DateComponents::minimumWeek()));
66 const Decimal maximum = parseToNumber(element()->attributeWithoutSynchronization(maxAttr), Decimal::fromDouble(DateComponents::maximumWeek()));
67 const Decimal step = StepRange::parseStep(anyStepHandling, weekStepDescription, element()->attributeWithoutSynchronization(stepAttr));
68 return StepRange(stepBase, RangeLimitations::Valid, minimum, maximum, step, weekStepDescription);
69}
