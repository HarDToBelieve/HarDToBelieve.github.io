41ANGLEInstancedArrays::ANGLEInstancedArrays(WebGLRenderingContextBase& context)
42 : WebGLExtension(context)
43{
44#if USE(ANGLE)
45 context.graphicsContextGL()->getExtensions().ensureEnabled("GL_ANGLE_instanced_arrays");
46#endif
47}
48
49ANGLEInstancedArrays::~ANGLEInstancedArrays() = default;
50
51WebGLExtension::ExtensionName ANGLEInstancedArrays::getName() const
52{
53 return ANGLEInstancedArraysName;
54}
55
56bool ANGLEInstancedArrays::supported(WebGLRenderingContextBase& context)
57{
58#if USE(ANGLE) || PLATFORM(GTK) || PLATFORM(WPE)
59 return context.graphicsContextGL()->getExtensions().supports("GL_ANGLE_instanced_arrays");
60#else
61 UNUSED_PARAM(context);
62 return false;
63#endif
64}

66void ANGLEInstancedArrays::drawArraysInstancedANGLE(GCGLenum mode, GCGLint first, GCGLsizei count, GCGLsizei primcount)
67{
68 if (!m_context || m_context->isContextLost())
69 return;
70 m_context->drawArraysInstanced(mode, first, count, primcount);
71}
72
73void ANGLEInstancedArrays::drawElementsInstancedANGLE(GCGLenum mode, GCGLsizei count, GCGLenum type, long long offset, GCGLsizei primcount)
74{
75 if (!m_context || m_context->isContextLost())
76 return;
77 m_context->drawElementsInstanced(mode, count, type, offset, primcount);
78}

80void ANGLEInstancedArrays::vertexAttribDivisorANGLE(GCGLuint index, GCGLuint divisor)
81{
82 if (!m_context || m_context->isContextLost())
83 return;
84 m_context->vertexAttribDivisor(index, divisor);
85}
