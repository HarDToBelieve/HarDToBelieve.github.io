110bool CanvasRenderingContext::wouldTaintOrigin(const HTMLImageElement* element)
111{
112 if (!element || !m_canvas.originClean())
113 return false;
114
115 auto* cachedImage = element->cachedImage();
116 if (!cachedImage)
117 return false;
118
119 auto image = makeRefPtr(cachedImage->image());
120 if (!image)
121 return false;
122
123 if (image->sourceURL().protocolIsData())
124 return false;
125
126 if (!image->hasSingleSecurityOrigin())
127 return true;
128
129 if (!cachedImage->isCORSSameOrigin())
130 return true;
131
132 ASSERT(m_canvas.securityOrigin());
133 ASSERT(cachedImage->origin());
134 ASSERT(m_canvas.securityOrigin()->toString() == cachedImage->origin()->toString());
135 return false;
136}

138bool CanvasRenderingContext::wouldTaintOrigin(const HTMLVideoElement* video)
139{
140#if ENABLE(VIDEO)
141 // FIXME: This check is likely wrong when a redirect is involved. We need
142 // to test the finalURL. Please be careful when fixing this issue not to
143 // make currentSrc be the final URL because then the
144 // HTMLMediaElement.currentSrc DOM API would leak redirect destinations!
145 if (!video || !m_canvas.originClean())
146 return false;
147
148 if (!video->hasSingleSecurityOrigin())
149 return true;
150
151 if (!(video->player() && video->player()->didPassCORSAccessCheck()) && video->wouldTaintOrigin(*m_canvas.securityOrigin()))
152 return true;
153
154#else
155 UNUSED_PARAM(video);
156#endif
157
158 return false;
159}

161bool CanvasRenderingContext::wouldTaintOrigin(const ImageBitmap* imageBitmap)
162{
163 if (!imageBitmap || !m_canvas.originClean())
164 return false;
165
166 return !imageBitmap->originClean();
167}

169bool CanvasRenderingContext::wouldTaintOrigin(const URL& url)
170{
171 if (!m_canvas.originClean())
172 return false;
173
174 if (url.protocolIsData())
175 return false;
176
177 return !m_canvas.securityOrigin()->canRequest(url);
178}
179
180void CanvasRenderingContext::checkOrigin(const URL& url)
181{
182 if (wouldTaintOrigin(url))
183 m_canvas.setOriginTainted();
184}
