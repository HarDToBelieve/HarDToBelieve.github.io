37namespace WebCore {
38
39class CanvasPattern;
40class DestinationColorSpace;
41class HTMLCanvasElement;
42class HTMLImageElement;
43class HTMLVideoElement;
44class ImageBitmap;
45class TypedOMCSSImageValue;
46class WebGLObject;
47enum class PixelFormat : uint8_t;
48
49class CanvasRenderingContext : public ScriptWrappable {
50 WTF_MAKE_NONCOPYABLE(CanvasRenderingContext);
51 WTF_MAKE_ISO_ALLOCATED(CanvasRenderingContext);
52public:
53 virtual ~CanvasRenderingContext();
54
55 static HashSet<CanvasRenderingContext*>& instances() WTF_REQUIRES_LOCK(instancesLock());
56 static Lock& instancesLock() WTF_RETURNS_LOCK(s_instancesLock);
57
58 void ref();
59 WEBCORE_EXPORT void deref();
60
61 CanvasBase& canvasBase() const { return m_canvas; }
62
63 virtual bool is2d() const { return false; }
64 virtual bool isWebGL1() const { return false; }
65 virtual bool isWebGL2() const { return false; }
66 bool isWebGL() const { return isWebGL1() || isWebGL2(); }
67#if ENABLE(WEBGPU)
68 virtual bool isWebGPU() const { return false; }
69#endif
70 virtual bool isGPUBased() const { return false; }
71 virtual bool isAccelerated() const { return false; }
72 virtual bool isBitmapRenderer() const { return false; }
73 virtual bool isPlaceholder() const { return false; }
74 virtual bool isOffscreen2d() const { return false; }
75 virtual bool isPaint() const { return false; }
76
77 virtual void clearAccumulatedDirtyRect() { }
78
79 // Called before paintRenderingResultsToCanvas if paintRenderingResultsToCanvas is
80 // used for compositing purposes.
81 virtual void prepareForDisplayWithPaint() { }
82 virtual void paintRenderingResultsToCanvas() { }
83 virtual PlatformLayer* platformLayer() const { return 0; }
84
85 bool hasActiveInspectorCanvasCallTracer() const { return m_hasActiveInspectorCanvasCallTracer; }
86 void setHasActiveInspectorCanvasCallTracer(bool hasActiveInspectorCanvasCallTracer) { m_hasActiveInspectorCanvasCallTracer = hasActiveInspectorCanvasCallTracer; }
87
88 virtual bool compositingResultsNeedUpdating() const { return false; }
89 virtual bool needsPreparationForDisplay() const { return false; }
90 virtual void prepareForDisplay() { }
91
92 virtual PixelFormat pixelFormat() const;
93 virtual DestinationColorSpace colorSpace() const;
94
95protected:
96 explicit CanvasRenderingContext(CanvasBase&);
97 bool wouldTaintOrigin(const CanvasPattern*);
98 bool wouldTaintOrigin(const CanvasBase*);
99 bool wouldTaintOrigin(const HTMLImageElement*);
100 bool wouldTaintOrigin(const HTMLVideoElement*);
101 bool wouldTaintOrigin(const ImageBitmap*);
102 bool wouldTaintOrigin(const URL&);
103
104 template<class T> void checkOrigin(const T* arg)
105 {
106 if (wouldTaintOrigin(arg))
107 m_canvas.setOriginTainted();
108 }
109 void checkOrigin(const URL&);
110 void checkOrigin(const TypedOMCSSImageValue&);
111
112 bool m_hasActiveInspectorCanvasCallTracer { false };
113
114private:
115 static Lock s_instancesLock;
116
117 CanvasBase& m_canvas;
118};
119
120} // namespace WebCore
