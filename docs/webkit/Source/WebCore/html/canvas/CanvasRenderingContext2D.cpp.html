97void CanvasRenderingContext2D::drawFocusIfNeededInternal(const Path& path, Element& element)
98{
99 auto* context = drawingContext();
100 if (!element.focused() || !state().hasInvertibleTransform || path.isEmpty() || !element.isDescendantOf(canvas()) || !context)
101 return;
102 context->drawFocusRing(path, 1, 1, RenderTheme::singleton().focusRingColor(element.document().styleColorOptions(canvas().computedStyle())));
103}

113void CanvasRenderingContext2D::setFontWithoutUpdatingStyle(const String& newFont)
114{
115 // Intentionally don't update style here, because updating style can cause JS to run synchronously.
116 // This function is called in the middle of processing, and running arbitrary JS in the middle of processing can cause unexpected behavior.
117 // Instead, the relevant canvas entry points update style once when they begin running, and we won't touch the style after that.
118 // This means that the style may end up being stale here, but that's at least better than running arbitrary JS in the middle of processing.
119
120 ScriptDisallowedScope::InMainThread scriptDisallowedScope;
121
122 if (newFont.isEmpty())
123 return;
124
125 if (newFont == state().unparsedFont && state().font.realized())
126 return;
127
128 // According to http://lists.w3.org/Archives/Public/public-html/2009Jul/0947.html,
129 // the "inherit" and "initial" values must be ignored. CSSPropertyParserWorkerSafe::parseFont() ignores these.
130 auto fontRaw = CSSPropertyParserWorkerSafe::parseFont(newFont, strictToCSSParserMode(!usesCSSCompatibilityParseMode()));
131 if (!fontRaw)
132 return;
133
134 FontCascadeDescription fontDescription;
135 if (auto* computedStyle = canvas().computedStyle())
136 fontDescription = FontCascadeDescription { computedStyle->fontDescription() };
137 else {
138 static NeverDestroyed<AtomString> family = DefaultFontFamily;
139 fontDescription.setOneFamily(family.get());
140 fontDescription.setSpecifiedSize(DefaultFontSize);
141 fontDescription.setComputedSize(DefaultFontSize);
142 }
143
144 // Map the <canvas> font into the text style. If the font uses keywords like larger/smaller, these will work
145 // relative to the canvas.
146 Document& document = canvas().document();
147 auto fontCascade = Style::resolveForFontRaw(*fontRaw, WTFMove(fontDescription), document);
148 if (!fontCascade)
149 return;
150
151 String newFontSafeCopy(newFont); // Create a string copy since newFont can be deleted inside realizeSaves.
152 realizeSaves();
153 modifiableState().unparsedFont = newFontSafeCopy;
154
155 modifiableState().font.initialize(document.fontSelector(), *fontCascade);
156 ASSERT(state().font.realized());
157 ASSERT(state().font.isPopulated());
158}

160inline TextDirection CanvasRenderingContext2D::toTextDirection(Direction direction, const RenderStyle** computedStyle) const
161{
162 auto* style = computedStyle || direction == Direction::Inherit ? canvas().existingComputedStyle() : nullptr;
163 if (computedStyle)
164 *computedStyle = style;
165 switch (direction) {
166 case Direction::Inherit:
167 return style ? style->direction() : TextDirection::LTR;
168 case Direction::Rtl:
169 return TextDirection::RTL;
170 case Direction::Ltr:
171 return TextDirection::LTR;
172 }
173 ASSERT_NOT_REACHED();
174 return TextDirection::LTR;
175}
176
177CanvasDirection CanvasRenderingContext2D::direction() const
178{
179 if (state().direction == Direction::Inherit)
180 canvas().document().updateStyleIfNeeded();
181 return toTextDirection(state().direction) == TextDirection::RTL ? CanvasDirection::Rtl : CanvasDirection::Ltr;
182}

194Ref<TextMetrics> CanvasRenderingContext2D::measureText(const String& text)
195{
196 downcast<HTMLCanvasElement>(canvasBase()).document().updateStyleIfNeeded();
197
198 ScriptDisallowedScope::InMainThread scriptDisallowedScope;
199
200 if (RuntimeEnabledFeatures::sharedFeatures().webAPIStatisticsEnabled()) {
201 auto& canvas = this->canvas();
202 ResourceLoadObserver::shared().logCanvasWriteOrMeasure(canvas.document(), text);
203 ResourceLoadObserver::shared().logCanvasRead(canvas.document());
204 }
205
206 String normalizedText = normalizeSpaces(text);
207 const RenderStyle* computedStyle;
208 auto direction = toTextDirection(state().direction, &computedStyle);
209 bool override = computedStyle && isOverride(computedStyle->unicodeBidi());
210 TextRun textRun(normalizedText, 0, 0, AllowRightExpansion, direction, override, true);
211 return measureTextInternal(textRun);
212}

214auto CanvasRenderingContext2D::fontProxy() -> const FontProxy*
215{
216 // Intentionally don't update style here, because updating style can cause JS to run synchronously.
217 // This function is called in the middle of processing, and running arbitrary JS in the middle of processing can cause unexpected behavior.
218 // Instead, the relevant canvas entry points update style once when they begin running, and we won't touch the style after that.
219 // This means that the style may end up being stale here, but that's at least better than running arbitrary JS in the middle of processing.
220 ScriptDisallowedScope::InMainThread scriptDisallowedScope;
221
222 if (!state().font.realized())
223 setFontWithoutUpdatingStyle(state().unparsedFont);
224 return &state().font;
225}

227void CanvasRenderingContext2D::drawTextInternal(const String& text, double x, double y, bool fill, std::optional<double> maxWidth)
228{
229 downcast<HTMLCanvasElement>(canvasBase()).document().updateStyleIfNeeded();
230
231 ScriptDisallowedScope::InMainThread scriptDisallowedScope;
232
233 if (RuntimeEnabledFeatures::sharedFeatures().webAPIStatisticsEnabled())
234 ResourceLoadObserver::shared().logCanvasWriteOrMeasure(this->canvas().document(), text);
235
236 if (!canDrawText(x, y, fill, maxWidth))
237 return;
238
239 String normalizedText = normalizeSpaces(text);
240 const RenderStyle* computedStyle;
241 auto direction = toTextDirection(state().direction, &computedStyle);
242 bool override = computedStyle && isOverride(computedStyle->unicodeBidi());
243 TextRun textRun(normalizedText, 0, 0, AllowRightExpansion, direction, override, true);
244 drawTextUnchecked(textRun, x, y, fill, maxWidth);
245}
