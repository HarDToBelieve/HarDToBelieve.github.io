53bool isCurrentColorString(const String& colorString)
54{
55 return equalLettersIgnoringASCIICase(colorString, "currentcolor");
56}
57
58Color parseColor(const String& colorString, CanvasBase& canvasBase)
59{
60#if ENABLE(OFFSCREEN_CANVAS)
61 if (canvasBase.isOffscreenCanvas())
62 return CSSPropertyParserWorkerSafe::parseColor(colorString);
63#else
64 UNUSED_PARAM(canvasBase);
65#endif
66 Color color = CSSParser::parseColor(colorString);
67 if (color.isValid())
68 return color;
69 return CSSParser::parseSystemColor(colorString);
70}
71
72Color currentColor(CanvasBase& canvasBase)
73{
74 if (!is<HTMLCanvasElement>(canvasBase))
75 return Color::black;
76
77 auto& canvas = downcast<HTMLCanvasElement>(canvasBase);
78 if (!canvas.isConnected() || !canvas.inlineStyle())
79 return Color::black;
80 Color color = CSSParser::parseColor(canvas.inlineStyle()->getPropertyValue(CSSPropertyColor));
81 if (!color.isValid())
82 return Color::black;
83 return color;
84}
85
86Color parseColorOrCurrentColor(const String& colorString, CanvasBase& canvasBase)
87{
88 if (isCurrentColorString(colorString))
89 return currentColor(canvasBase);
90
91 return parseColor(colorString, canvasBase);
92}

119CanvasStyle CanvasStyle::createFromString(const String& colorString, CanvasBase& canvasBase)
120{
121 if (isCurrentColorString(colorString))
122 return CurrentColor { std::nullopt };
123
124 Color color = parseColor(colorString, canvasBase);
125 if (!color.isValid())
126 return { };
127
128 return color;
129}
130
131CanvasStyle CanvasStyle::createFromStringWithOverrideAlpha(const String& colorString, float alpha, CanvasBase& canvasBase)
132{
133 if (isCurrentColorString(colorString))
134 return CurrentColor { alpha };
135
136 Color color = parseColor(colorString, canvasBase);
137 if (!color.isValid())
138 return { };
139
140 return color.colorWithAlpha(alpha);
141}

143bool CanvasStyle::isEquivalentColor(const CanvasStyle& other) const
144{
145 if (WTF::holds_alternative<Color>(m_style) && WTF::holds_alternative<Color>(other.m_style))
146 return WTF::get<Color>(m_style) == WTF::get<Color>(other.m_style);
147
148 return false;
149}
150
151bool CanvasStyle::isEquivalent(const SRGBA<float>& components) const
152{
153 return WTF::holds_alternative<Color>(m_style) && WTF::get<Color>(m_style) == convertColor<SRGBA<uint8_t>>(components);
154}

156void CanvasStyle::applyStrokeColor(GraphicsContext& context) const
157{
158 WTF::switchOn(m_style,
159 [&context] (const Color& color) {
160 context.setStrokeColor(color);
161 },
162 [&context] (const RefPtr<CanvasGradient>& gradient) {
163 context.setStrokeGradient(gradient->gradient());
164 },
165 [&context] (const RefPtr<CanvasPattern>& pattern) {
166 context.setStrokePattern(pattern->pattern());
167 },
168 [] (const CurrentColor&) {
169 ASSERT_NOT_REACHED();
170 },
171 [] (const Invalid&) {
172 ASSERT_NOT_REACHED();
173 }
174 );
175}
