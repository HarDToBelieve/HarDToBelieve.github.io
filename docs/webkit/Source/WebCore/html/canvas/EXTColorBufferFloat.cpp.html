34namespace WebCore {
35
36WTF_MAKE_ISO_ALLOCATED_IMPL(EXTColorBufferFloat);
37
38EXTColorBufferFloat::EXTColorBufferFloat(WebGLRenderingContextBase& context)
39 : WebGLExtension(context)
40{
41 context.graphicsContextGL()->getExtensions().ensureEnabled("GL_EXT_color_buffer_float"_s);
42 // https://github.com/KhronosGroup/WebGL/pull/2830
43 // Spec requires EXT_float_blend to be turned on implicitly here.
44 context.graphicsContextGL()->getExtensions().ensureEnabled("GL_EXT_float_blend"_s);
45}
46
47EXTColorBufferFloat::~EXTColorBufferFloat() = default;
48
49WebGLExtension::ExtensionName EXTColorBufferFloat::getName() const
50{
51 return EXTColorBufferFloatName;
52}
53
54bool EXTColorBufferFloat::supported(const WebGLRenderingContextBase& context)
55{
56 return context.graphicsContextGL()->getExtensions().supports("GL_EXT_color_buffer_float"_s);
57}
