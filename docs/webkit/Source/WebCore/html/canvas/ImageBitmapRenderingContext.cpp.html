45std::unique_ptr<ImageBitmapRenderingContext> ImageBitmapRenderingContext::create(CanvasBase& canvas, ImageBitmapRenderingContextSettings&& settings)
46{
47 auto renderingContext = std::unique_ptr<ImageBitmapRenderingContext>(new ImageBitmapRenderingContext(canvas, WTFMove(settings)));
48
49 InspectorInstrumentation::didCreateCanvasRenderingContext(*renderingContext);
50
51 return renderingContext;
52}

54ImageBitmapRenderingContext::ImageBitmapRenderingContext(CanvasBase& canvas, ImageBitmapRenderingContextSettings&& settings)
55 : CanvasRenderingContext(canvas)

63HTMLCanvasElement* ImageBitmapRenderingContext::canvas() const
64{
65 auto& base = canvasBase();
66 if (!is<HTMLCanvasElement>(base))
67 return nullptr;
68 return &downcast<HTMLCanvasElement>(base);
69}

76void ImageBitmapRenderingContext::setOutputBitmap(RefPtr<ImageBitmap> imageBitmap)
77{
78 // 1. If a bitmap argument was not provided, then:
79
80 if (!imageBitmap) {
81
82 // 1.1. Set context's bitmap mode to blank.
83
84 m_bitmapMode = BitmapMode::Blank;
85
86 // 1.2. Let canvas be the canvas element to which context is bound.
87
88 // 1.3. Set context's output bitmap to be transparent black with an
89 // intrinsic width equal to the numeric value of canvas's width attribute
90 // and an intrinsic height equal to the numeric value of canvas's height
91 // attribute, those values being interpreted in CSS pixels.
92
93 // FIXME: What is the point of creating a full size transparent buffer that
94 // can never be changed? Wouldn't a 1x1 buffer give the same rendering? The
95 // only reason I can think of is toDataURL(), but that doesn't seem like
96 // a good enough argument to waste memory.
97
98 canvas()->setImageBufferAndMarkDirty(ImageBuffer::create(FloatSize(canvas()->width(), canvas()->height()), bufferRenderingMode, 1, DestinationColorSpace::SRGB(), PixelFormat::BGRA8));
99
100 // 1.4. Set the output bitmap's origin-clean flag to true.
101
102 canvas()->setOriginClean();
103 return;
104 }
105
106 // 2. If a bitmap argument was provided, then:
107
108 // 2.1. Set context's bitmap mode to valid.
109
110 m_bitmapMode = BitmapMode::Valid;
111
112 // 2.2. Set context's output bitmap to refer to the same underlying
113 // bitmap data as bitmap, without making a copy.
114 // Note: the origin-clean flag of bitmap is included in the
115 // bitmap data to be referenced by context's output bitmap.
116
117 if (imageBitmap->originClean())
118 canvas()->setOriginClean();
119 else
120 canvas()->setOriginTainted();
121 canvas()->setImageBufferAndMarkDirty(imageBitmap->takeImageBuffer());
122}

124ExceptionOr<void> ImageBitmapRenderingContext::transferFromImageBitmap(RefPtr<ImageBitmap> imageBitmap)
125{
126 // 1. Let bitmapContext be the ImageBitmapRenderingContext object on which
127 // the transferFromImageBitmap() method was called.
128
129 // 2. If imageBitmap is null, then run the steps to set an ImageBitmapRenderingContext's
130 // output bitmap, with bitmapContext as the context argument and no bitmap argument,
131 // then abort these steps.
132
133 if (!imageBitmap) {
134 setOutputBitmap(nullptr);
135 return { };
136 }
137
138 // 3. If the value of imageBitmap's [[Detached]] internal slot is set to true,
139 // then throw an "InvalidStateError" DOMException and abort these steps.
140
141 if (imageBitmap->isDetached())
142 return Exception { InvalidStateError };
143
144 // 4. Run the steps to set an ImageBitmapRenderingContext's output bitmap,
145 // with the context argument equal to bitmapContext, and the bitmap
146 // argument referring to imageBitmap's underlying bitmap data.
147
148 setOutputBitmap(imageBitmap);
149
150 // 5. Set the value of imageBitmap's [[Detached]] internal slot to true.
151 // 6. Unset imageBitmap's bitmap data.
152
153 // Note that the algorithm in the specification is currently a bit
154 // muddy here. The setOutputBitmap step above had to transfer ownership
155 // from the imageBitmap to this object, which requires a detach and unset,
156 // so this step isn't necessary, but we'll do it anyway.
157
158 imageBitmap->close();
159
160 return { };
161}
