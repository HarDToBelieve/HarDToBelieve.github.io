51RefPtr<WebGLVertexArrayObjectOES> OESVertexArrayObject::createVertexArrayOES()
52{
53 if (!m_context || m_context->isContextLost())
54 return nullptr;
55
56 auto object = WebGLVertexArrayObjectOES::create(*m_context, WebGLVertexArrayObjectOES::Type::User);
57 m_context->addContextObject(object.get());
58 return object;
59}
60
61void OESVertexArrayObject::deleteVertexArrayOES(WebGLVertexArrayObjectOES* arrayObject)
62{
63 if (!m_context)
64 return;
65
66 Locker locker { m_context->objectGraphLock() };
67
68 if (!arrayObject || m_context->isContextLost())
69 return;
70
71 if (!arrayObject->isDefaultObject() && arrayObject == static_cast<WebGLRenderingContext*>(m_context)->m_boundVertexArrayObject)
72 static_cast<WebGLRenderingContext*>(m_context)->setBoundVertexArrayObject(locker, nullptr);
73
74 arrayObject->deleteObject(locker, m_context->graphicsContextGL());
75}
76
77GCGLboolean OESVertexArrayObject::isVertexArrayOES(WebGLVertexArrayObjectOES* arrayObject)
78{
79 if (!m_context)
80 return false;
81
82 return arrayObject && !m_context->isContextLost() && arrayObject->hasEverBeenBound()
83 && m_context->graphicsContextGL()->isVertexArray(arrayObject->object());
84}
85
86void OESVertexArrayObject::bindVertexArrayOES(WebGLVertexArrayObjectOES* arrayObject)
87{
88 if (!m_context)
89 return;
90
91 Locker locker { m_context->objectGraphLock() };
92
93 if (m_context->isContextLost())
94 return;
95
96 if (arrayObject && (arrayObject->isDeleted() || !arrayObject->validate(nullptr, *context()))) {
97 m_context->graphicsContextGL()->synthesizeGLError(GraphicsContextGL::INVALID_OPERATION);
98 return;
99 }
100
101 auto* contextGL = m_context->graphicsContextGL();
102 auto& context = downcast<WebGLRenderingContext>(*m_context);
103 if (arrayObject && !arrayObject->isDefaultObject() && arrayObject->object()) {
104 contextGL->bindVertexArray(arrayObject->object());
105 arrayObject->setHasEverBeenBound();
106 context.setBoundVertexArrayObject(locker, arrayObject);
107 } else {
108 contextGL->bindVertexArray(0);
109 context.setBoundVertexArrayObject(locker, nullptr);
110 }
111}
