64bool WebGLBuffer::associateBufferDataImpl(const void* data, GCGLsizeiptr byteLength)
65{
66 if (byteLength < 0)
67 return false;
68
69 switch (m_target) {
70 case GraphicsContextGL::ELEMENT_ARRAY_BUFFER:
71 if (byteLength > std::numeric_limits<unsigned>::max())
72 return false;
73 m_byteLength = byteLength;
74 clearCachedMaxIndices();
75 if (byteLength) {
76 m_elementArrayBuffer = ArrayBuffer::tryCreate(byteLength, 1);
77 if (!m_elementArrayBuffer) {
78 m_byteLength = 0;
79 return false;
80 }
81 if (data) {
82 // We must always clone the incoming data because client-side
83 // modifications without calling bufferData or bufferSubData
84 // must never be able to change the validation results.
85 memcpy(m_elementArrayBuffer->data(), data, byteLength);
86 }
87 } else
88 m_elementArrayBuffer = nullptr;
89 return true;
90 case GraphicsContextGL::ARRAY_BUFFER:
91 m_byteLength = byteLength;
92 return true;
93 default:
94#if ENABLE(WEBGL2)
95 switch (m_target) {
96 case GraphicsContextGL::COPY_READ_BUFFER:
97 case GraphicsContextGL::COPY_WRITE_BUFFER:
98 case GraphicsContextGL::PIXEL_PACK_BUFFER:
99 case GraphicsContextGL::PIXEL_UNPACK_BUFFER:
100 case GraphicsContextGL::TRANSFORM_FEEDBACK_BUFFER:
101 case GraphicsContextGL::UNIFORM_BUFFER:
102 m_byteLength = byteLength;
103 return true;
104 }
105#endif
106 return false;
107 }
108}

129bool WebGLBuffer::associateBufferSubDataImpl(GCGLintptr offset, const void* data, GCGLsizeiptr byteLength)
130{
131 if (!data || offset < 0 || byteLength < 0)
132 return false;
133
134 if (byteLength) {
135 Checked<GCGLintptr, RecordOverflow> checkedBufferOffset(offset);
136 Checked<GCGLsizeiptr, RecordOverflow> checkedDataLength(byteLength);
137 Checked<GCGLintptr, RecordOverflow> checkedBufferMax = checkedBufferOffset + checkedDataLength;
138 if (checkedBufferMax.hasOverflowed() || offset > m_byteLength || checkedBufferMax > m_byteLength)
139 return false;
140 }
141
142 switch (m_target) {
143 case GraphicsContextGL::ELEMENT_ARRAY_BUFFER:
144 clearCachedMaxIndices();
145 if (byteLength) {
146 if (!m_elementArrayBuffer)
147 return false;
148 memcpy(static_cast<unsigned char*>(m_elementArrayBuffer->data()) + offset, data, byteLength);
149 }
150 return true;
151 case GraphicsContextGL::ARRAY_BUFFER:
152 return true;
153 default:
154#if ENABLE(WEBGL2)
155 switch (m_target) {
156 case GraphicsContextGL::COPY_READ_BUFFER:
157 case GraphicsContextGL::COPY_WRITE_BUFFER:
158 case GraphicsContextGL::PIXEL_PACK_BUFFER:
159 case GraphicsContextGL::PIXEL_UNPACK_BUFFER:
160 case GraphicsContextGL::TRANSFORM_FEEDBACK_BUFFER:
161 case GraphicsContextGL::UNIFORM_BUFFER:
162 return true;
163 }
164#endif
165 return false;
166 }
167}

183bool WebGLBuffer::associateCopyBufferSubData(const WebGLBuffer& readBuffer, GCGLintptr readOffset, GCGLintptr writeOffset, GCGLsizeiptr size)
184{
185 if (readOffset < 0 || writeOffset < 0 || size < 0)
186 return false;
187
188 if (size) {
189 Checked<GCGLintptr, RecordOverflow> checkedReadBufferOffset(readOffset);
190 Checked<GCGLsizeiptr, RecordOverflow> checkedDataLength(size);
191 Checked<GCGLintptr, RecordOverflow> checkedReadBufferMax = checkedReadBufferOffset + checkedDataLength;
192 if (checkedReadBufferMax.hasOverflowed() || readOffset > readBuffer.byteLength() || checkedReadBufferMax > readBuffer.byteLength())
193 return false;
194
195 Checked<GCGLintptr, RecordOverflow> checkedWriteBufferOffset(writeOffset);
196 Checked<GCGLintptr, RecordOverflow> checkedWriteBufferMax = checkedWriteBufferOffset + checkedDataLength;
197 if (checkedWriteBufferMax.hasOverflowed() || writeOffset > m_byteLength || checkedWriteBufferMax > m_byteLength)
198 return false;
199 }
200
201 switch (m_target) {
202 case GraphicsContextGL::ELEMENT_ARRAY_BUFFER:
203 clearCachedMaxIndices();
204 if (size) {
205 if (!m_elementArrayBuffer)
206 return false;
207 memcpy(static_cast<unsigned char*>(m_elementArrayBuffer->data()) + writeOffset, static_cast<const unsigned char*>(readBuffer.elementArrayBuffer()->data()) + readOffset, size);
208 }
209 return true;
210 case GraphicsContextGL::ARRAY_BUFFER:
211 return true;
212 default:
213#if ENABLE(WEBGL2)
214 switch (m_target) {
215 case GraphicsContextGL::COPY_READ_BUFFER:
216 case GraphicsContextGL::COPY_WRITE_BUFFER:
217 case GraphicsContextGL::PIXEL_PACK_BUFFER:
218 case GraphicsContextGL::PIXEL_UNPACK_BUFFER:
219 case GraphicsContextGL::TRANSFORM_FEEDBACK_BUFFER:
220 case GraphicsContextGL::UNIFORM_BUFFER:
221 return true;
222 }
223#endif
224 return false;
225 }

239std::optional<unsigned> WebGLBuffer::getCachedMaxIndex(GCGLenum type)
240{
241 for (auto& cache : m_maxIndexCache) {
242 if (cache.type == type)
243 return cache.maxIndex;
244 }
245 return std::nullopt;
246}
247
248void WebGLBuffer::setCachedMaxIndex(GCGLenum type, unsigned value)
249{
250 for (auto& cache : m_maxIndexCache) {
251 if (cache.type == type) {
252 cache.maxIndex = value;
253 return;
254 }
255 }
256 m_maxIndexCache[m_nextAvailableCacheEntry].type = type;
257 m_maxIndexCache[m_nextAvailableCacheEntry].maxIndex = value;
258 m_nextAvailableCacheEntry = (m_nextAvailableCacheEntry + 1) % WTF_ARRAY_LENGTH(m_maxIndexCache);
259}
