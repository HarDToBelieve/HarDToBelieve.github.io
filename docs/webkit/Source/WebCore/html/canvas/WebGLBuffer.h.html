26#pragma once
27
28#if ENABLE(WEBGL)
29
30#include "WebGLSharedObject.h"
31#include <wtf/RefPtr.h>
32
33namespace JSC {
34class ArrayBuffer;
35class ArrayBufferView;
36}
37
38namespace WebCore {
39
40class WebGLBuffer final : public WebGLSharedObject {
41public:
42 static Ref<WebGLBuffer> create(WebGLRenderingContextBase&);
43 virtual ~WebGLBuffer();
44
45 bool associateBufferData(GCGLsizeiptr size);
46 bool associateBufferData(JSC::ArrayBuffer*);
47 bool associateBufferData(JSC::ArrayBufferView*);
48 bool associateBufferSubData(GCGLintptr offset, JSC::ArrayBuffer*);
49 bool associateBufferSubData(GCGLintptr offset, JSC::ArrayBufferView*);
50 bool associateCopyBufferSubData(const WebGLBuffer& readBuffer, GCGLintptr readOffset, GCGLintptr writeOffset, GCGLsizeiptr);
51
52 void disassociateBufferData();
53
54 GCGLsizeiptr byteLength() const;
55 const RefPtr<JSC::ArrayBuffer> elementArrayBuffer() const { return m_elementArrayBuffer; }
56
57 // Gets the cached max index for the given type if one has been set.
58 std::optional<unsigned> getCachedMaxIndex(GCGLenum type);
59 // Sets the cached max index for the given type.
60 void setCachedMaxIndex(GCGLenum type, unsigned value);
61
62 GCGLenum getTarget() const { return m_target; }
63 void setTarget(GCGLenum);
64
65 bool hasEverBeenBound() const { return object() && m_target; }
66
67private:
68 WebGLBuffer(WebGLRenderingContextBase&);
69
70 void deleteObjectImpl(const WTF::AbstractLocker&, GraphicsContextGL*, PlatformGLObject) override;
71
72 GCGLenum m_target { 0 };
73
74 RefPtr<JSC::ArrayBuffer> m_elementArrayBuffer;
75 GCGLsizeiptr m_byteLength { 0 };
76
77 // Optimization for index validation. For each type of index
78 // (i.e., UNSIGNED_SHORT), cache the maximum index in the
79 // entire buffer.
80 //
81 // This is sufficient to eliminate a lot of work upon each
82 // draw call as long as all bound array buffers are at least
83 // that size.
84 struct MaxIndexCacheEntry {
85 GCGLenum type;
86 unsigned maxIndex;
87 };
88 // OpenGL ES 2.0 only has two valid index types (UNSIGNED_BYTE
89 // and UNSIGNED_SHORT) plus one extension (UNSIGNED_INT).
90 MaxIndexCacheEntry m_maxIndexCache[4];
91 unsigned m_nextAvailableCacheEntry { 0 };
92
93 // Clears all of the cached max indices.
94 void clearCachedMaxIndices();
95
96 // Helper function called by the three associateBufferData().
97 bool associateBufferDataImpl(const void* data, GCGLsizeiptr byteLength);
98 // Helper function called by the two associateBufferSubData().
99 bool associateBufferSubDataImpl(GCGLintptr offset, const void* data, GCGLsizeiptr byteLength);
100};
101
102} // namespace WebCore
103
104#endif
