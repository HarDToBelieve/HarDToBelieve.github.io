37Ref<WebGLContextGroup> WebGLContextGroup::create()
38{
39 return adoptRef(*new WebGLContextGroup);
40}
41
42WebGLContextGroup::~WebGLContextGroup()
43{
44 detachAndRemoveAllObjects();
45}
46
47bool WebGLContextGroup::hasAContext() const
48{
49 return !m_contexts.isEmpty();
50}
51
52GraphicsContextGL& WebGLContextGroup::getAGraphicsContextGL()
53{
54 ASSERT(!m_contexts.isEmpty());
55 return *(*m_contexts.begin())->graphicsContextGL();
56}
57
58Lock& WebGLContextGroup::objectGraphLockForAContext()
59{
60 ASSERT(!m_contexts.isEmpty());
61 // Since the WEBGL_shared_objects extension never shipped, and is
62 // unlikely to, this is equivalent to WebGLContextObject's
63 // implementation. If there were the possibility of WebGL objects
64 // being shared between multiple contexts, this would have to be
65 // rethought.
66 return (*m_contexts.begin())->objectGraphLock();
67}
68
69void WebGLContextGroup::addContext(WebGLRenderingContextBase& context)
70{
71 m_contexts.add(&context);
72}
73
74void WebGLContextGroup::removeContext(WebGLRenderingContextBase& context)
75{
76 // We must call detachAndRemoveAllObjects before removing the last context.
77 if (m_contexts.size() == 1 && m_contexts.contains(&context))
78 detachAndRemoveAllObjects();
79
80 m_contexts.remove(&context);
81}
82
83void WebGLContextGroup::removeObject(WebGLSharedObject& object)
84{
85 m_groupObjects.remove(&object);
86}
87
88void WebGLContextGroup::addObject(WebGLSharedObject& object)
89{
90 m_groupObjects.add(&object);
91}
92
93void WebGLContextGroup::detachAndRemoveAllObjects()
94{
95 if (m_contexts.isEmpty()) {
96 // No objectGraphLock is available to cover the calls to
97 // WebGLObject::deleteObject.
98 while (!m_groupObjects.isEmpty())
99 (*m_groupObjects.begin())->detachContextGroupWithoutDeletingObject();
100 return;
101 }
102
103 Locker locker { objectGraphLockForAContext() };
104 while (!m_groupObjects.isEmpty())
105 (*m_groupObjects.begin())->detachContextGroup(locker);
106}
107
108void WebGLContextGroup::loseContextGroup(WebGLRenderingContextBase::LostContextMode mode)
109{
110 for (auto& context : m_contexts)
111 context->loseContextImpl(mode);
112
113 detachAndRemoveAllObjects();
114}
