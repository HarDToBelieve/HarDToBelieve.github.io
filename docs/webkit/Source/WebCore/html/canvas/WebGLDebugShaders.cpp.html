41WebGLDebugShaders::WebGLDebugShaders(WebGLRenderingContextBase& context)
42 : WebGLExtension(context)
43{
44 context.graphicsContextGL()->getExtensions().ensureEnabled("GL_ANGLE_translated_shader_source");
45}
46
47WebGLDebugShaders::~WebGLDebugShaders() = default;
48
49WebGLExtension::ExtensionName WebGLDebugShaders::getName() const
50{
51 return WebGLDebugShadersName;
52}
53
54String WebGLDebugShaders::getTranslatedShaderSource(WebGLShader& shader)
55{
56 if (!m_context || m_context->isContextLost())
57 return String();
58 if (!m_context->validateWebGLObject("getTranslatedShaderSource", &shader))
59 return emptyString();
60 return m_context->graphicsContextGL()->getExtensions().getTranslatedShaderSourceANGLE(shader.object());
61}
