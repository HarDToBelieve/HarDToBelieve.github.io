51bool WebGLDrawBuffers::supported(WebGLRenderingContextBase& context)
52{
53#if USE(ANGLE)
54 return context.graphicsContextGL()->getExtensions().supports("GL_EXT_draw_buffers");
55#else
56 return context.graphicsContextGL()->getExtensions().supports("GL_EXT_draw_buffers")
57 && satisfiesWebGLRequirements(context);
58#endif
59}

61void WebGLDrawBuffers::drawBuffersWEBGL(const Vector<GCGLenum>& buffers)
62{
63 if (!m_context || m_context->isContextLost())
64 return;
65 GCGLsizei n = buffers.size();
66 const GCGLenum* bufs = buffers.data();
67 if (!m_context->m_framebufferBinding) {
68 if (n != 1) {
69 m_context->synthesizeGLError(GraphicsContextGL::INVALID_VALUE, "drawBuffersWEBGL", "more than one buffer");
70 return;
71 }
72 if (bufs[0] != GraphicsContextGL::BACK && bufs[0] != GraphicsContextGL::NONE) {
73 m_context->synthesizeGLError(GraphicsContextGL::INVALID_OPERATION, "drawBuffersWEBGL", "BACK or NONE");
74 return;
75 }
76 // Because the backbuffer is simulated on all current WebKit ports, we need to change BACK to COLOR_ATTACHMENT0.
77 GCGLenum value[1] { bufs[0] == GraphicsContextGL::BACK ? GraphicsContextGL::COLOR_ATTACHMENT0 : GraphicsContextGL::NONE };
78 m_context->graphicsContextGL()->getExtensions().drawBuffersEXT(value);
79 m_context->setBackDrawBuffer(bufs[0]);
80 } else {
81 if (n > m_context->getMaxDrawBuffers()) {
82 m_context->synthesizeGLError(GraphicsContextGL::INVALID_VALUE, "drawBuffersWEBGL", "more than max draw buffers");
83 return;
84 }
85 for (GCGLsizei i = 0; i < n; ++i) {
86 if (bufs[i] != GraphicsContextGL::NONE && bufs[i] != ExtensionsGL::COLOR_ATTACHMENT0_EXT + i) {
87 m_context->synthesizeGLError(GraphicsContextGL::INVALID_OPERATION, "drawBuffersWEBGL", "COLOR_ATTACHMENTi_EXT or NONE");
88 return;
89 }
90 }
91 m_context->m_framebufferBinding->drawBuffers(buffers);
92 }
93}

95// static
96bool WebGLDrawBuffers::satisfiesWebGLRequirements(WebGLRenderingContextBase& webglContext)
97{
98 GraphicsContextGL* context = webglContext.graphicsContextGL();
99
100 // This is called after we make sure GL_EXT_draw_buffers is supported.
101 GCGLint maxDrawBuffers = context->getInteger(ExtensionsGL::MAX_DRAW_BUFFERS_EXT);
102 GCGLint maxColorAttachments = context->getInteger(ExtensionsGL::MAX_COLOR_ATTACHMENTS_EXT);
103 if (maxDrawBuffers < 4 || maxColorAttachments < 4)
104 return false;
105
106 PlatformGLObject fbo = context->createFramebuffer();
107 context->bindFramebuffer(GraphicsContextGL::FRAMEBUFFER, fbo);
108
109 const unsigned char buffer[4] = { 0, 0, 0, 0 }; // textures are required to be initialized for other ports.
110 bool supportsDepth = context->getExtensions().supports("GL_OES_depth_texture")
111 || context->getExtensions().supports("GL_ARB_depth_texture");
112 bool supportsDepthStencil = (context->getExtensions().supports("GL_EXT_packed_depth_stencil")
113 || context->getExtensions().supports("GL_OES_packed_depth_stencil"));
114 PlatformGLObject depthStencil = 0;
115 if (supportsDepthStencil) {
116 depthStencil = context->createTexture();
117 context->bindTexture(GraphicsContextGL::TEXTURE_2D, depthStencil);
118 context->texImage2D(GraphicsContextGL::TEXTURE_2D, 0, GraphicsContextGL::DEPTH_STENCIL, 1, 1, 0, GraphicsContextGL::DEPTH_STENCIL, GraphicsContextGL::UNSIGNED_INT_24_8, buffer);
119 }
120 PlatformGLObject depth = 0;
121 if (supportsDepth) {
122 depth = context->createTexture();
123 context->bindTexture(GraphicsContextGL::TEXTURE_2D, depth);
124 context->texImage2D(GraphicsContextGL::TEXTURE_2D, 0, GraphicsContextGL::DEPTH_COMPONENT, 1, 1, 0, GraphicsContextGL::DEPTH_COMPONENT, GraphicsContextGL::UNSIGNED_INT, buffer);
125 }
126
127 Vector<PlatformGLObject> colors;
128 bool ok = true;
129 GCGLint maxAllowedBuffers = std::min(maxDrawBuffers, maxColorAttachments);
130 for (GCGLint i = 0; i < maxAllowedBuffers; ++i) {
131 PlatformGLObject color = context->createTexture();
132 colors.append(color);
133 context->bindTexture(GraphicsContextGL::TEXTURE_2D, color);
134 context->texImage2D(GraphicsContextGL::TEXTURE_2D, 0, GraphicsContextGL::RGBA, 1, 1, 0, GraphicsContextGL::RGBA, GraphicsContextGL::UNSIGNED_BYTE, buffer);
135 context->framebufferTexture2D(GraphicsContextGL::FRAMEBUFFER, GraphicsContextGL::COLOR_ATTACHMENT0 + i, GraphicsContextGL::TEXTURE_2D, color, 0);
136 if (context->checkFramebufferStatus(GraphicsContextGL::FRAMEBUFFER) != GraphicsContextGL::FRAMEBUFFER_COMPLETE) {
137 ok = false;
138 break;
139 }
140 if (supportsDepth) {
141 context->framebufferTexture2D(GraphicsContextGL::FRAMEBUFFER, GraphicsContextGL::DEPTH_ATTACHMENT, GraphicsContextGL::TEXTURE_2D, depth, 0);
142 if (context->checkFramebufferStatus(GraphicsContextGL::FRAMEBUFFER) != GraphicsContextGL::FRAMEBUFFER_COMPLETE) {
143 ok = false;
144 break;
145 }
146 context->framebufferTexture2D(GraphicsContextGL::FRAMEBUFFER, GraphicsContextGL::DEPTH_ATTACHMENT, GraphicsContextGL::TEXTURE_2D, 0, 0);
147 }
148 if (supportsDepthStencil) {
149 context->framebufferTexture2D(GraphicsContextGL::FRAMEBUFFER, GraphicsContextGL::DEPTH_ATTACHMENT, GraphicsContextGL::TEXTURE_2D, depthStencil, 0);
150 context->framebufferTexture2D(GraphicsContextGL::FRAMEBUFFER, GraphicsContextGL::STENCIL_ATTACHMENT, GraphicsContextGL::TEXTURE_2D, depthStencil, 0);
151 if (context->checkFramebufferStatus(GraphicsContextGL::FRAMEBUFFER) != GraphicsContextGL::FRAMEBUFFER_COMPLETE) {
152 ok = false;
153 break;
154 }
155 context->framebufferTexture2D(GraphicsContextGL::FRAMEBUFFER, GraphicsContextGL::DEPTH_ATTACHMENT, GraphicsContextGL::TEXTURE_2D, 0, 0);
156 context->framebufferTexture2D(GraphicsContextGL::FRAMEBUFFER, GraphicsContextGL::STENCIL_ATTACHMENT, GraphicsContextGL::TEXTURE_2D, 0, 0);
157 }
158 }
159
160 webglContext.restoreCurrentFramebuffer();
161 context->deleteFramebuffer(fbo);
162 webglContext.restoreCurrentTexture2D();
163 if (supportsDepth)
164 context->deleteTexture(depth);
165 if (supportsDepthStencil)
166 context->deleteTexture(depthStencil);
167 for (auto& color : colors)
168 context->deleteTexture(color);
169 return ok;
170}
