30#include "WebGLRenderingContextBase.h"
31#include <wtf/RefCounted.h>

33namespace WebCore {
34
35class WebGLExtension : public RefCounted<WebGLExtension> {
36 WTF_MAKE_ISO_ALLOCATED(WebGLExtension);
37public:
38 // Extension names are needed to properly wrap instances in JavaScript objects.
39 enum ExtensionName {
40 WebGLLoseContextName,
41 EXTBlendMinMaxName,
42 EXTFragDepthName,
43 EXTShaderTextureLODName,
44 EXTTextureCompressionRGTCName,
45 EXTTextureFilterAnisotropicName,
46 EXTsRGBName,
47 KHRParallelShaderCompileName,
48 OESTextureFloatName,
49 OESTextureFloatLinearName,
50 OESTextureHalfFloatName,
51 OESTextureHalfFloatLinearName,
52 OESStandardDerivativesName,
53 OESVertexArrayObjectName,
54 WebGLDebugRendererInfoName,
55 WebGLDebugShadersName,
56 WebGLCompressedTextureS3TCName,
57 WebGLCompressedTextureS3TCsRGBName,
58 WebGLDepthTextureName,
59 WebGLDrawBuffersName,
60 OESElementIndexUintName,
61 OESFBORenderMipmapName,
62 WebGLCompressedTextureATCName,
63 WebGLCompressedTextureETCName,
64 WebGLCompressedTextureETC1Name,
65 WebGLCompressedTexturePVRTCName,
66 WebGLCompressedTextureASTCName,
67 ANGLEInstancedArraysName,
68 EXTColorBufferHalfFloatName,
69 EXTFloatBlendName,
70 WebGLColorBufferFloatName,
71 EXTColorBufferFloatName,
72 WebGLMultiDrawName,
73 };
74
75 WebGLRenderingContextBase* context() { return m_context; }
76
77 virtual ~WebGLExtension();
78 virtual ExtensionName getName() const = 0;
79
80 // Lose the parent WebGL context. The context loss mode changes
81 // the behavior specifically of WEBGL_lose_context, which does not
82 // lose its connection to its parent context when it forces a
83 // context loss. However, all extensions must be lost when
84 // destroying their WebGLRenderingContextBase.
85 virtual void loseParentContext(WebGLRenderingContextBase::LostContextMode);
86 bool isLost() { return !m_context; }
87
88protected:
89 WebGLExtension(WebGLRenderingContextBase&);
90 WebGLRenderingContextBase* m_context;
91};
92
93} // namespace WebCore
