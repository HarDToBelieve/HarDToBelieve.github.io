30#include "WebGLContextObject.h"

32#include <wtf/HashMap.h>

34#include <wtf/Vector.h>

36namespace JSC {
37class AbstractSlotVisitor;
38}
39
40namespace WTF {
41class AbstractLocker;
42}
43
44namespace WebCore {
45
46class WebGLRenderbuffer;
47class WebGLTexture;
48
49class WebGLFramebuffer final : public WebGLContextObject {
50public:
51 class WebGLAttachment : public RefCounted<WebGLAttachment> {
52 public:
53 virtual ~WebGLAttachment();
54
55#if !USE(ANGLE)
56 virtual GCGLsizei getWidth() const = 0;
57 virtual GCGLsizei getHeight() const = 0;
58 virtual GCGLenum getFormat() const = 0;
59#endif
60 virtual WebGLSharedObject* getObject() const = 0;
61 virtual bool isSharedObject(WebGLSharedObject*) const = 0;
62 virtual bool isValid() const = 0;
63 virtual bool isInitialized() const = 0;
64 virtual void setInitialized() = 0;
65 virtual void onDetached(const WTF::AbstractLocker&, GraphicsContextGL*) = 0;
66 virtual void attach(GraphicsContextGL*, GCGLenum target, GCGLenum attachment) = 0;
67 virtual void unattach(GraphicsContextGL*, GCGLenum target, GCGLenum attachment) = 0;
68 virtual void addMembersToOpaqueRoots(const WTF::AbstractLocker&, JSC::AbstractSlotVisitor&) = 0;
69
70 protected:
71 WebGLAttachment();
72 };
73
74 virtual ~WebGLFramebuffer();
75
76 static Ref<WebGLFramebuffer> create(WebGLRenderingContextBase&);
77#if ENABLE(WEBXR)
78 static Ref<WebGLFramebuffer> createOpaque(WebGLRenderingContextBase&);
79#endif
80
81 void setAttachmentForBoundFramebuffer(GCGLenum target, GCGLenum attachment, GCGLenum texTarget, WebGLTexture*, GCGLint level, GCGLint layer);
82 void setAttachmentForBoundFramebuffer(GCGLenum target, GCGLenum attachment, WebGLRenderbuffer*);
83 // If an object is attached to the currently bound framebuffer, remove it.
84 void removeAttachmentFromBoundFramebuffer(const WTF::AbstractLocker&, GCGLenum target, WebGLSharedObject*);
85 // If a given attachment point for the currently bound framebuffer is not null, remove the attached object.
86 void removeAttachmentFromBoundFramebuffer(const WTF::AbstractLocker&, GCGLenum target, GCGLenum attachment);
87 WebGLSharedObject* getAttachmentObject(GCGLenum) const;
88
89#if !USE(ANGLE)
90 GCGLenum getColorBufferFormat() const;
91 GCGLsizei getColorBufferWidth() const;
92 GCGLsizei getColorBufferHeight() const;
93
94 // This should always be called before drawArray, drawElements, clear,
95 // readPixels, copyTexImage2D, copyTexSubImage2D if this framebuffer is
96 // currently bound.
97 // Return false if the framebuffer is incomplete; otherwise initialize
98 // the buffers if they haven't been initialized and
99 // needToInitializeAttachments is true.
100 bool onAccess(GraphicsContextGL*, const char** reason);
101
102 // Software version of glCheckFramebufferStatus(), except that when
103 // FRAMEBUFFER_COMPLETE is returned, it is still possible for
104 // glCheckFramebufferStatus() to return FRAMEBUFFER_UNSUPPORTED,
105 // depending on hardware implementation.
106 GCGLenum checkStatus(const char** reason) const;
107#endif
108
109 bool hasEverBeenBound() const { return object() && m_hasEverBeenBound; }
110
111 void setHasEverBeenBound() { m_hasEverBeenBound = true; }
112
113 bool hasStencilBuffer() const;
114
115 // Wrapper for drawBuffersEXT/drawBuffersARB to work around a driver bug.
116 void drawBuffers(const Vector<GCGLenum>& bufs);
117
118 GCGLenum getDrawBuffer(GCGLenum);
119
120 void addMembersToOpaqueRoots(const WTF::AbstractLocker&, JSC::AbstractSlotVisitor&);
121
122#if ENABLE(WEBXR)
123 bool isOpaque() const { return m_opaque; }
124 void setOpaqueActive(bool active) { m_opaqueActive = active; }
125#endif
126
127private:
128 WebGLFramebuffer(WebGLRenderingContextBase&);
129
130 void deleteObjectImpl(const WTF::AbstractLocker&, GraphicsContextGL*, PlatformGLObject) override;
131
132 WebGLAttachment* getAttachment(GCGLenum) const;
133
134#if !USE(ANGLE)
135 // Return false if framebuffer is incomplete.
136 bool initializeAttachments(GraphicsContextGL*, const char** reason);
137#endif
138
139 // Check if the framebuffer is currently bound to the given target.
140 bool isBound(GCGLenum target) const;
141
142 // attach 'attachment' at 'attachmentPoint'.
143 void attach(GCGLenum target, GCGLenum attachment, GCGLenum attachmentPoint);
144
145 // Check if a new drawBuffers call should be issued. This is called when we add or remove an attachment.
146 void drawBuffersIfNecessary(bool force);
147
148 void setAttachmentInternal(GCGLenum attachment, GCGLenum texTarget, WebGLTexture*, GCGLint level, GCGLint layer);
149 void setAttachmentInternal(GCGLenum attachment, WebGLRenderbuffer*);
150 // If a given attachment point for the currently bound framebuffer is not
151 // null, remove the attached object.
152 void removeAttachmentInternal(const WTF::AbstractLocker&, GCGLenum attachment);
153
154 typedef WTF::HashMap<GCGLenum, RefPtr<WebGLAttachment>> AttachmentMap;
155
156 AttachmentMap m_attachments;
157
158 bool m_hasEverBeenBound;
159
160 Vector<GCGLenum> m_drawBuffers;
161 Vector<GCGLenum> m_filteredDrawBuffers;
162
163#if ENABLE(WEBXR)
164 bool m_opaque { false };
165 bool m_opaqueActive { false };
166#endif
167};
168
169} // namespace WebCore
170
171#endif
