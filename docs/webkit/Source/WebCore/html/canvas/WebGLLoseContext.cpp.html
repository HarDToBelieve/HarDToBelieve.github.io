35namespace WebCore {
36
37WTF_MAKE_ISO_ALLOCATED_IMPL(WebGLLoseContext);
38
39WebGLLoseContext::WebGLLoseContext(WebGLRenderingContextBase& context)
40 : WebGLExtension(context)
41{
42}
43
44WebGLLoseContext::~WebGLLoseContext() = default;
45
46WebGLExtension::ExtensionName WebGLLoseContext::getName() const
47{
48 return WebGLLoseContextName;
49}
50
51void WebGLLoseContext::loseContext()
52{
53 if (!m_context)
54 return;
55
56 m_context->forceLostContext(WebGLRenderingContextBase::SyntheticLostContext);
57}
58
59void WebGLLoseContext::restoreContext()
60{
61 if (!m_context)
62 return;
63
64 m_context->forceRestoreContext();
65}
66
67void WebGLLoseContext::loseParentContext(WebGLRenderingContextBase::LostContextMode mode)
68{
69 if (mode == WebGLRenderingContextBase::LostContextMode::RealLostContext)
70 WebGLExtension::loseParentContext(mode);
71}
