38static GCGLSpan<const int> makeSpanWithOffset(WebGLMultiDraw::Int32List& list, GCGLuint offset)
39{
40 return makeGCGLSpan(list.data() + offset, list.length() - offset);
41}

43WebGLMultiDraw::WebGLMultiDraw(WebGLRenderingContextBase& context)
44 : WebGLExtension(context)
45{
46 context.graphicsContextGL()->getExtensions().ensureEnabled("GL_ANGLE_multi_draw"_s);
47 context.graphicsContextGL()->getExtensions().ensureEnabled("GL_ANGLE_instanced_arrays"_s);

57bool WebGLMultiDraw::supported(const WebGLRenderingContextBase& context)
58{
59 return context.graphicsContextGL()->getExtensions().supports("GL_ANGLE_multi_draw"_s)
60 && context.graphicsContextGL()->getExtensions().supports("GL_ANGLE_instanced_arrays"_s);
61}

63void WebGLMultiDraw::multiDrawArraysWEBGL(GCGLenum mode, Int32List firstsList, GCGLuint firstsOffset, Int32List countsList, GCGLuint countsOffset, GCGLsizei drawcount)
64{
65 if (!m_context || m_context->isContextLost())
66 return;
67
68 if (!validateDrawcount("multiDrawArraysWEBGL", drawcount)
69 || !validateOffset("multiDrawArraysWEBGL", "firstsOffset out of bounds", firstsList.length(), firstsOffset, drawcount)
70 || !validateOffset("multiDrawArraysWEBGL", "countsOffset out of bounds", countsList.length(), countsOffset, drawcount)) {
71 return;
72 }
73
74 m_context->graphicsContextGL()->multiDrawArraysANGLE(mode, makeSpanWithOffset(firstsList, firstsOffset), makeSpanWithOffset(countsList, countsOffset), drawcount);
75}

121bool WebGLMultiDraw::validateDrawcount(const char* functionName, GCGLsizei drawcount)
122{
123 if (drawcount < 0) {
124 m_context->synthesizeGLError(GraphicsContextGL::INVALID_VALUE, functionName, "negative drawcount");
125 return false;
126 }
127
128 return true;
129}
130
131bool WebGLMultiDraw::validateOffset(const char* functionName, const char* outOfBoundsDescription, GCGLsizei size, GCGLuint offset, GCGLsizei drawcount)
132{
133 if (drawcount > size) {
134 m_context->synthesizeGLError(GraphicsContextGL::INVALID_OPERATION, functionName, "drawcount out of bounds");
135 return false;
136 }
137
138 if (offset >= static_cast<GCGLuint>(size)) {
139 m_context->synthesizeGLError(GraphicsContextGL::INVALID_OPERATION, functionName, outOfBoundsDescription);
140 return false;
141 }
142
143 return true;
144}
