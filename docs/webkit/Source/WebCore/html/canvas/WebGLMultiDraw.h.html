30namespace WebCore {

32class WebGLMultiDraw final : public WebGLExtension {
33 WTF_MAKE_ISO_ALLOCATED(WebGLMultiDraw);
34
35private:
36 template <class TypedArray, class DataType>
37 class TypedList {
38 public:
39 using ListTypeOptions = Variant<RefPtr<TypedArray>, Vector<DataType>>;
40
41 TypedList(ListTypeOptions&& variant)
42 : m_variant(WTFMove(variant))
43 {
44 }
45
46 const DataType* data() const
47 {
48 return WTF::switchOn(m_variant,
49 [] (const RefPtr<TypedArray>& typedArray) -> const DataType* { return typedArray->data(); },
50 [] (const Vector<DataType>& vector) -> const DataType* { return vector.data(); }
51 );
52 }
53
54 GCGLsizei length() const
55 {
56 return WTF::switchOn(m_variant,
57 [] (const RefPtr<TypedArray>& typedArray) -> GCGLsizei { return typedArray->length(); },
58 [] (const Vector<DataType>& vector) -> GCGLsizei { return vector.size(); }
59 );
60 }
61
62 private:
63 ListTypeOptions m_variant;
64 };
65
66public:
67 using Int32List = TypedList<Int32Array, int32_t>;
68
69 explicit WebGLMultiDraw(WebGLRenderingContextBase&);
70 virtual ~WebGLMultiDraw();
71
72 ExtensionName getName() const override;
73
74 static bool supported(const WebGLRenderingContextBase&);
75
76 void multiDrawArraysWEBGL(GCGLenum mode, Int32List firstsList, GCGLuint firstsOffset, Int32List countsList, GCGLuint countsOffset, GCGLsizei drawcount);
77
78 void multiDrawElementsWEBGL(GCGLenum mode, Int32List countsList, GCGLuint countsOffset, GCGLenum type, Int32List offsetsList, GCGLuint offsetsOffset, GCGLsizei drawcount);
79
80 void multiDrawArraysInstancedWEBGL(GCGLenum mode, Int32List firstsList, GCGLuint firstsOffset, Int32List countsList, GCGLuint countsOffset, Int32List instanceCountsList, GCGLuint instanceCountsOffset, GCGLsizei drawcount);
81
82 void multiDrawElementsInstancedWEBGL(GCGLenum mode, Int32List countsList, GCGLuint countsOffset, GCGLenum type, Int32List offsetsList, GCGLuint offsetsOffset, Int32List instanceCountsList, GCGLuint instanceCountsOffset, GCGLsizei drawcount);
83
84private:
85 bool validateDrawcount(const char* functionName, GCGLsizei drawcount);
86 bool validateOffset(const char* functionName, const char* outOfBoundsDescription, GCGLsizei, GCGLuint offset, GCGLsizei drawcount);
87};
