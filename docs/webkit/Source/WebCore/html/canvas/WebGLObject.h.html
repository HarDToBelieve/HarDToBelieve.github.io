26#pragma once
27
28#if ENABLE(WEBGL)
29
30#include "GraphicsTypesGL.h"
31#include <wtf/Forward.h>
32#include <wtf/RefCounted.h>
33
34namespace WebCore {
35
36class GraphicsContextGL;
37class WebGLContextGroup;
38class WebGLRenderingContextBase;
39
40class WebGLObject : public RefCounted<WebGLObject> {
41public:
42 virtual ~WebGLObject() = default;
43
44 PlatformGLObject object() const { return m_object; }
45
46 // deleteObject may not always delete the OpenGL resource. For programs and
47 // shaders, deletion is delayed until they are no longer attached.
48 // FIXME: revisit this when resource sharing between contexts are implemented.
49 // The AbstractLocker argument enforces at compile time that the objectGraphLock
50 // is held. This isn't necessary for all object types, but enough of them that
51 // it's done for all of them.
52 void deleteObject(const WTF::AbstractLocker&, GraphicsContextGL*);
53
54 void onAttached() { ++m_attachmentCount; }
55 void onDetached(const WTF::AbstractLocker&, GraphicsContextGL*);
56
57 // This indicates whether the client side issue a delete call already, not
58 // whether the OpenGL resource is deleted.
59 // object()==0 indicates the OpenGL resource is deleted.
60 bool isDeleted() { return m_deleted; }
61
62 // True if this object belongs to the group or context.
63 virtual bool validate(const WebGLContextGroup*, const WebGLRenderingContextBase&) const = 0;
64
65 // Returns the object graph lock associated with the context most
66 // closely associated with this object. Since the
67 // WEBGL_shared_objects extension specification never shipped (and
68 // is unlikely to), this basically returns the same result for
69 // both context objects and shared objects.
70 virtual Lock& objectGraphLockForContext() = 0;
71
72protected:
73 WebGLObject() = default;
74
75 // setObject should be only called once right after creating a WebGLObject.
76 void setObject(PlatformGLObject);
77
78 void runDestructor();
79
80 // deleteObjectImpl should be only called once to delete the OpenGL resource.
81 virtual void deleteObjectImpl(const WTF::AbstractLocker&, GraphicsContextGL*, PlatformGLObject) = 0;
82
83 virtual bool hasGroupOrContext() const = 0;
84
85 virtual void detach();
86
87 virtual GraphicsContextGL* getAGraphicsContextGL() const = 0;
88
89private:
90 PlatformGLObject m_object { 0 };
91 unsigned m_attachmentCount { 0 };
92 bool m_deleted { false };
93};
94
95inline PlatformGLObject objectOrZero(WebGLObject* object)
96{
97 return object ? object->object() : 0;
98}
99
100} // namespace WebCore
101
102#endif
