45namespace WebCore {
46
47class ScriptExecutionContext;
48class WebGLRenderingContextBase;
49class WebGLShader;
50
51class WebGLProgram final : public WebGLSharedObject, public ContextDestructionObserver {
52public:
53 static Ref<WebGLProgram> create(WebGLRenderingContextBase&);
54 virtual ~WebGLProgram();
55
56 static HashMap<WebGLProgram*, WebGLRenderingContextBase*>& instances() WTF_REQUIRES_LOCK(instancesLock());
57 static Lock& instancesLock() WTF_RETURNS_LOCK(s_instancesLock);
58
59 void contextDestroyed() final;
60
61 unsigned numActiveAttribLocations();
62 GCGLint getActiveAttribLocation(GCGLuint index);
63
64 bool isUsingVertexAttrib0();
65
66 bool getLinkStatus();
67 void setLinkStatus(bool);
68
69 unsigned getLinkCount() const { return m_linkCount; }
70
71 // This is to be called everytime after the program is successfully linked.
72 // We don't deal with integer overflow here, assuming in reality a program
73 // will never be linked so many times.
74 // Also, we invalidate the cached program info.
75 void increaseLinkCount();
76
77 WebGLShader* getAttachedShader(GCGLenum);
78 bool attachShader(const WTF::AbstractLocker&, WebGLShader*);
79 bool detachShader(const WTF::AbstractLocker&, WebGLShader*);
80
81 void setRequiredTransformFeedbackBufferCount(int count)
82 {
83 m_requiredTransformFeedbackBufferCountAfterNextLink = count;
84 }
85 int requiredTransformFeedbackBufferCount()
86 {
87 cacheInfoIfNeeded();
88 return m_requiredTransformFeedbackBufferCount;
89 }
90
91 void addMembersToOpaqueRoots(const WTF::AbstractLocker&, JSC::AbstractSlotVisitor&);
92
93private:
94 WebGLProgram(WebGLRenderingContextBase&);
95
96 void deleteObjectImpl(const WTF::AbstractLocker&, GraphicsContextGL*, PlatformGLObject) override;
97
98 void cacheActiveAttribLocations(GraphicsContextGL*);
99 void cacheInfoIfNeeded();
100
101 static Lock s_instancesLock;
102
103 Vector<GCGLint> m_activeAttribLocations;
104
105 GCGLint m_linkStatus { 0 };
106
107 // This is used to track whether a WebGLUniformLocation belongs to this program or not.
108 unsigned m_linkCount { 0 };
109
110 RefPtr<WebGLShader> m_vertexShader;
111 RefPtr<WebGLShader> m_fragmentShader;
112
113 bool m_infoValid { true };
114 int m_requiredTransformFeedbackBufferCountAfterNextLink { 0 };
115 int m_requiredTransformFeedbackBufferCount { 0 };
116};

118} // namespace WebCore
