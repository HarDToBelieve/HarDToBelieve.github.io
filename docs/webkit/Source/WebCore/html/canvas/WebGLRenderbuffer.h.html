26#pragma once
27
28#if ENABLE(WEBGL)
29
30#include "WebGLSharedObject.h"
31
32namespace WebCore {
33
34class WebGLRenderbuffer final : public WebGLSharedObject {
35public:
36 virtual ~WebGLRenderbuffer();
37
38 static Ref<WebGLRenderbuffer> create(WebGLRenderingContextBase&);
39
40 void setInternalFormat(GCGLenum internalformat)
41 {
42 m_internalFormat = internalformat;
43 m_initialized = false;
44 }
45 GCGLenum getInternalFormat() const { return m_internalFormat; }
46
47 void setSize(GCGLsizei width, GCGLsizei height)
48 {
49 m_width = width;
50 m_height = height;
51 }
52 GCGLsizei getWidth() const { return m_width; }
53 GCGLsizei getHeight() const { return m_height; }
54
55 void setIsValid(bool isValid) { m_isValid = isValid; }
56 bool isValid() const { return m_isValid; }
57
58 bool isInitialized() const { return m_initialized; }
59 void setInitialized() { m_initialized = true; }
60
61 bool hasEverBeenBound() const { return object() && m_hasEverBeenBound; }
62
63 void setHasEverBeenBound() { m_hasEverBeenBound = true; }
64
65private:
66 WebGLRenderbuffer(WebGLRenderingContextBase&);
67
68 void deleteObjectImpl(const WTF::AbstractLocker&, GraphicsContextGL*, PlatformGLObject) override;
69
70 bool isRenderbuffer() const override { return true; }
71
72 GCGLenum m_internalFormat;
73 bool m_initialized;
74 GCGLsizei m_width, m_height;
75 bool m_isValid; // This is only false if internalFormat is DEPTH_STENCIL and packed_depth_stencil is not supported.
76
77 bool m_hasEverBeenBound;
78};
79
80} // namespace WebCore
81
82#endif
