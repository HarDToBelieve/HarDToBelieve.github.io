103{
104 auto renderingContext = std::unique_ptr<WebGLRenderingContext>(new WebGLRenderingContext(canvas, WTFMove(context), attributes));

107

298WebGLAny WebGLRenderingContext::getFramebufferAttachmentParameter(GCGLenum target, GCGLenum attachment, GCGLenum pname)
299{
300 if (isContextLostOrPending() || !validateFramebufferFuncParameters("getFramebufferAttachmentParameter", target, attachment))
301 return nullptr;
302
303 if (!m_framebufferBinding || !m_framebufferBinding->object()) {
304 synthesizeGLError(GraphicsContextGL::INVALID_OPERATION, "getFramebufferAttachmentParameter", "no framebuffer bound");
305 return nullptr;
306 }
307
308#if ENABLE(WEBXR)
309 if (m_framebufferBinding->isOpaque()) {
310 synthesizeGLError(GraphicsContextGL::INVALID_OPERATION, "getFramebufferAttachmentParameter", "An opaque framebuffer's attachments cannot be inspected or changed");
311 return nullptr;
312 }
313#endif
314
315 auto object = makeRefPtr(m_framebufferBinding->getAttachmentObject(attachment));
316 if (!object) {
317 if (pname == GraphicsContextGL::FRAMEBUFFER_ATTACHMENT_OBJECT_TYPE)
318 return static_cast<unsigned>(GraphicsContextGL::NONE);
319 // OpenGL ES 2.0 specifies INVALID_ENUM in this case, while desktop GL
320 // specifies INVALID_OPERATION.
321 synthesizeGLError(GraphicsContextGL::INVALID_ENUM, "getFramebufferAttachmentParameter", "invalid parameter name");
322 return nullptr;
323 }
324
325 if (object->isTexture()) {
326 switch (pname) {
327 case GraphicsContextGL::FRAMEBUFFER_ATTACHMENT_OBJECT_TYPE:
328 return static_cast<unsigned>(GraphicsContextGL::TEXTURE);
329 case GraphicsContextGL::FRAMEBUFFER_ATTACHMENT_OBJECT_NAME:
330 return makeRefPtr(reinterpret_cast<WebGLTexture&>(*object));
331 case GraphicsContextGL::FRAMEBUFFER_ATTACHMENT_TEXTURE_LEVEL:
332 case GraphicsContextGL::FRAMEBUFFER_ATTACHMENT_TEXTURE_CUBE_MAP_FACE:
333 case ExtensionsGL::FRAMEBUFFER_ATTACHMENT_COLOR_ENCODING_EXT:
334 return m_context->getFramebufferAttachmentParameteri(target, attachment, pname);
335 default:
336 synthesizeGLError(GraphicsContextGL::INVALID_ENUM, "getFramebufferAttachmentParameter", "invalid parameter name for texture attachment");
337 return nullptr;
338 }
339 } else {
340 ASSERT(object->isRenderbuffer());
341 switch (pname) {
342 case GraphicsContextGL::FRAMEBUFFER_ATTACHMENT_OBJECT_TYPE:
343 return static_cast<unsigned>(GraphicsContextGL::RENDERBUFFER);
344 case GraphicsContextGL::FRAMEBUFFER_ATTACHMENT_OBJECT_NAME:
345 return makeRefPtr(reinterpret_cast<WebGLRenderbuffer&>(*object));
346 case ExtensionsGL::FRAMEBUFFER_ATTACHMENT_COLOR_ENCODING_EXT: {
347 if (!m_extsRGB) {
348 synthesizeGLError(GraphicsContextGL::INVALID_ENUM, "getFramebufferAttachmentParameter", "invalid parameter name for renderbuffer attachment");
349 return nullptr;
350 }
351 RefPtr<WebGLRenderbuffer> renderBuffer = reinterpret_cast<WebGLRenderbuffer*>(object.get());
352 GCGLenum renderBufferFormat = renderBuffer->getInternalFormat();
353 ASSERT(renderBufferFormat != ExtensionsGL::SRGB_EXT && renderBufferFormat != ExtensionsGL::SRGB_ALPHA_EXT);
354 if (renderBufferFormat == ExtensionsGL::SRGB8_ALPHA8_EXT)
355 return static_cast<unsigned>(ExtensionsGL::SRGB_EXT);
356 return static_cast<unsigned>(GraphicsContextGL::LINEAR);
357 }
358 default:
359 synthesizeGLError(GraphicsContextGL::INVALID_ENUM, "getFramebufferAttachmentParameter", "invalid parameter name for renderbuffer attachment");
360 return nullptr;
361 }
362 }
363}

365GCGLint WebGLRenderingContext::getMaxDrawBuffers()
366{
367 if (!supportsDrawBuffers())
368 return 0;
369 if (!m_maxDrawBuffers)
370 m_maxDrawBuffers = m_context->getInteger(ExtensionsGL::MAX_DRAW_BUFFERS_EXT);

386bool WebGLRenderingContext::validateIndexArrayConservative(GCGLenum type, unsigned& numElementsRequired)
387{
388 // Performs conservative validation by caching a maximum index of
389 // the given type per element array buffer. If all of the bound
390 // array buffers have enough elements to satisfy that maximum
391 // index, skips the expensive per-draw-call iteration in
392 // validateIndexArrayPrecise.
393
394 RefPtr<WebGLBuffer> elementArrayBuffer = m_boundVertexArrayObject->getElementArrayBuffer();
395
396 if (!elementArrayBuffer)
397 return false;
398
399 GCGLsizeiptr numElements = elementArrayBuffer->byteLength();
400 // The case count==0 is already dealt with in drawElements before validateIndexArrayConservative.
401 if (!numElements)
402 return false;
403 auto buffer = elementArrayBuffer->elementArrayBuffer();
404 ASSERT(buffer);
405
406 std::optional<unsigned> maxIndex = elementArrayBuffer->getCachedMaxIndex(type);
407 if (!maxIndex) {
408 // Compute the maximum index in the entire buffer for the given type of index.
409 switch (type) {
410 case GraphicsContextGL::UNSIGNED_BYTE: {
411 const GCGLubyte* p = static_cast<const GCGLubyte*>(buffer->data());
412 for (GCGLsizeiptr i = 0; i < numElements; i++)
413 maxIndex = maxIndex ? std::max(maxIndex.value(), static_cast<unsigned>(p[i])) : static_cast<unsigned>(p[i]);
414 break;
415 }
416 case GraphicsContextGL::UNSIGNED_SHORT: {
417 numElements /= sizeof(GCGLushort);
418 const GCGLushort* p = static_cast<const GCGLushort*>(buffer->data());
419 for (GCGLsizeiptr i = 0; i < numElements; i++)
420 maxIndex = maxIndex ? std::max(maxIndex.value(), static_cast<unsigned>(p[i])) : static_cast<unsigned>(p[i]);
421 break;
422 }
423 case GraphicsContextGL::UNSIGNED_INT: {
424 if (!m_oesElementIndexUint)
425 return false;
426 numElements /= sizeof(GCGLuint);
427 const GCGLuint* p = static_cast<const GCGLuint*>(buffer->data());
428 for (GCGLsizeiptr i = 0; i < numElements; i++)
429 maxIndex = maxIndex ? std::max(maxIndex.value(), static_cast<unsigned>(p[i])) : static_cast<unsigned>(p[i]);
430 break;
431 }
432 default:
433 return false;
434 }
435 if (maxIndex)
436 elementArrayBuffer->setCachedMaxIndex(type, maxIndex.value());
437 }
438
439 if (!maxIndex)
440 return false;
441
442 // The number of required elements is one more than the maximum index that will be accessed.
443 auto checkedNumElementsRequired = checkedAddAndMultiply<unsigned>(maxIndex.value(), 1, 1);
444 if (!checkedNumElementsRequired)
445 return false;
446 numElementsRequired = checkedNumElementsRequired.value();
447
448 return true;
449}

451bool WebGLRenderingContext::validateBlendEquation(const char* functionName, GCGLenum mode)
452{
453 switch (mode) {
454 case GraphicsContextGL::FUNC_ADD:
455 case GraphicsContextGL::FUNC_SUBTRACT:
456 case GraphicsContextGL::FUNC_REVERSE_SUBTRACT:
457 case ExtensionsGL::MIN_EXT:
458 case ExtensionsGL::MAX_EXT:
459 if ((mode == ExtensionsGL::MIN_EXT || mode == ExtensionsGL::MAX_EXT) && !m_extBlendMinMax) {
460 synthesizeGLError(GraphicsContextGL::INVALID_ENUM, functionName, "invalid mode");
461 return false;
462 }
463 return true;
464 break;
465 default:
466 synthesizeGLError(GraphicsContextGL::INVALID_ENUM, functionName, "invalid mode");
467 return false;
468 }
469}
