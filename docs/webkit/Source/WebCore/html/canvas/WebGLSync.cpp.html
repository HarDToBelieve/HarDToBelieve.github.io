39Ref<WebGLSync> WebGLSync::create(WebGLRenderingContextBase& ctx)
40{
41 return adoptRef(*new WebGLSync(ctx));
42}
43
44WebGLSync::~WebGLSync()
45{
46 if (!hasGroupOrContext())
47 return;
48
49 runDestructor();
50}
51
52WebGLSync::WebGLSync(WebGLRenderingContextBase& ctx)
53 : WebGLSharedObject(ctx)
54 , m_sync(ctx.graphicsContextGL()->fenceSync(GraphicsContextGL::SYNC_GPU_COMMANDS_COMPLETE, 0))
55{
56 // This value is unused because the sync object is a pointer type, but it needs to be non-zero
57 // or other parts of the code will assume the object is invalid.
58 setObject(-1);
59}
60
61void WebGLSync::deleteObjectImpl(const AbstractLocker&, GraphicsContextGL* context3d, PlatformGLObject object)
62{
63 UNUSED_PARAM(object);
64 context3d->deleteSync(m_sync);
65 m_sync = nullptr;
66}
67
68void WebGLSync::updateCache(WebGLRenderingContextBase& context)
69{
70 if (m_syncStatus == GraphicsContextGL::SIGNALED || !m_allowCacheUpdate)
71 return;
72
73 m_allowCacheUpdate = false;
74 m_syncStatus = context.graphicsContextGL()->getSynci(m_sync, GraphicsContextGL::SYNC_STATUS);
75 if (m_syncStatus == GraphicsContextGL::UNSIGNALED)
76 scheduleAllowCacheUpdate(context);
77}
78
79GCGLint WebGLSync::getCachedResult(GCGLenum pname) const
80{
81 switch (pname) {
82 case GraphicsContextGL::OBJECT_TYPE:
83 return GraphicsContextGL::SYNC_FENCE;
84 case GraphicsContextGL::SYNC_STATUS:
85 return m_syncStatus;
86 case GraphicsContextGL::SYNC_CONDITION:
87 return GraphicsContextGL::SYNC_GPU_COMMANDS_COMPLETE;
88 case GraphicsContextGL::SYNC_FLAGS:
89 return 0;
90 }
91 ASSERT_NOT_REACHED();
92 return 0;
93}
94
95bool WebGLSync::isSignaled() const
96{
97 return m_syncStatus == GraphicsContextGL::SIGNALED;
98}
99
100void WebGLSync::scheduleAllowCacheUpdate(WebGLRenderingContextBase& context)
101{
102 auto* canvas = context.htmlCanvas();
103 if (!canvas)
104 return;
105
106 context.queueTaskKeepingObjectAlive(*canvas, TaskSource::WebGL, [protectedThis = makeRef(*this)]() {
107 protectedThis->m_allowCacheUpdate = true;
108 });
109}
