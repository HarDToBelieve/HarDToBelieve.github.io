39Ref<WebGLTransformFeedback> WebGLTransformFeedback::create(WebGL2RenderingContext& ctx)
40{
41 return adoptRef(*new WebGLTransformFeedback(ctx));
42}
43
44WebGLTransformFeedback::~WebGLTransformFeedback()
45{
46 if (!hasGroupOrContext())
47 return;
48
49 runDestructor();
50}
51
52WebGLTransformFeedback::WebGLTransformFeedback(WebGL2RenderingContext& ctx)
53 : WebGLSharedObject(ctx)
54{
55 setObject(ctx.graphicsContextGL()->createTransformFeedback());
56 m_boundIndexedTransformFeedbackBuffers.resize(ctx.maxTransformFeedbackSeparateAttribs());
57}
58
59void WebGLTransformFeedback::deleteObjectImpl(const AbstractLocker&, GraphicsContextGL* context3d, PlatformGLObject object)
60{
61 context3d->deleteTransformFeedback(object);
62}
63
64void WebGLTransformFeedback::setProgram(const AbstractLocker&, WebGLProgram& program)
65{
66 m_program = &program;
67 m_programLinkCount = program.getLinkCount();
68}
69
70void WebGLTransformFeedback::setBoundIndexedTransformFeedbackBuffer(const AbstractLocker&, GCGLuint index, WebGLBuffer* buffer)
71{
72 ASSERT(index < m_boundIndexedTransformFeedbackBuffers.size());
73 m_boundIndexedTransformFeedbackBuffers[index] = buffer;
74}
75
76bool WebGLTransformFeedback::getBoundIndexedTransformFeedbackBuffer(GCGLuint index, WebGLBuffer** outBuffer)
77{
78 if (index >= m_boundIndexedTransformFeedbackBuffers.size())
79 return false;
80 *outBuffer = m_boundIndexedTransformFeedbackBuffers[index].get();
81 return true;
82}
83
84bool WebGLTransformFeedback::hasEnoughBuffers(GCGLuint numRequired) const
85{
86 if (numRequired > m_boundIndexedTransformFeedbackBuffers.size())
87 return false;
88 for (GCGLuint i = 0; i < numRequired; i++) {
89 if (!m_boundIndexedTransformFeedbackBuffers[i].get())
90 return false;
91 }
92 return true;
93}
94
95void WebGLTransformFeedback::addMembersToOpaqueRoots(const AbstractLocker& locker, JSC::AbstractSlotVisitor& visitor)
96{
97 for (auto& buffer : m_boundIndexedTransformFeedbackBuffers)
98 visitor.addOpaqueRoot(buffer.get());
99
100 visitor.addOpaqueRoot(m_program.get());
101 if (m_program)
102 m_program->addMembersToOpaqueRoots(locker, visitor);
103}
104
105void WebGLTransformFeedback::unbindBuffer(const AbstractLocker&, WebGLBuffer& buffer)
106{
107 for (auto& boundBuffer : m_boundIndexedTransformFeedbackBuffers) {
108 if (boundBuffer == &buffer)
109 boundBuffer = nullptr;
110 }
111}
112
113bool WebGLTransformFeedback::validateProgramForResume(WebGLProgram* program) const
114{
115 return program && m_program == program && program->getLinkCount() == m_programLinkCount;
116}
