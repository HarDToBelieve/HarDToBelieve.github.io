28#if ENABLE(WEBGL2)
29
30#include "WebGLSharedObject.h"

32namespace JSC {
33class AbstractSlotVisitor;
34}
35
36namespace WTF {
37class AbstractLocker;
38}
39
40namespace WebCore {
41
42class WebGL2RenderingContext;
43class WebGLBuffer;
44class WebGLProgram;
45
46class WebGLTransformFeedback final : public WebGLSharedObject {
47public:
48 virtual ~WebGLTransformFeedback();
49
50 static Ref<WebGLTransformFeedback> create(WebGL2RenderingContext&);
51
52 bool isActive() const { return m_active; }
53 bool isPaused() const { return m_paused; }
54
55 void setActive(bool active) { m_active = active; }
56 void setPaused(bool paused) { m_paused = paused; }
57
58 // These are the indexed bind points for transform feedback buffers.
59 // Returns false if index is out of range and the caller should
60 // synthesize a GL error.
61 void setBoundIndexedTransformFeedbackBuffer(const WTF::AbstractLocker&, GCGLuint index, WebGLBuffer*);
62 bool getBoundIndexedTransformFeedbackBuffer(GCGLuint index, WebGLBuffer** outBuffer);
63
64 bool validateProgramForResume(WebGLProgram*) const;
65
66 bool hasEverBeenBound() const { return object() && m_hasEverBeenBound; }
67 void setHasEverBeenBound() { m_hasEverBeenBound = true; }
68
69 WebGLProgram* program() const { return m_program.get(); }
70 void setProgram(const WTF::AbstractLocker&, WebGLProgram&);
71
72 void unbindBuffer(const AbstractLocker&, WebGLBuffer&);
73
74 bool hasEnoughBuffers(GCGLuint numRequired) const;
75
76 void addMembersToOpaqueRoots(const AbstractLocker&, JSC::AbstractSlotVisitor&);
77
78private:
79 WebGLTransformFeedback(WebGL2RenderingContext&);
80
81 void deleteObjectImpl(const WTF::AbstractLocker&, GraphicsContextGL*, PlatformGLObject) override;
82
83 bool m_active { false };
84 bool m_paused { false };
85 bool m_hasEverBeenBound { false };
86 unsigned m_programLinkCount { 0 };
87 Vector<RefPtr<WebGLBuffer>> m_boundIndexedTransformFeedbackBuffers;
88 RefPtr<WebGLProgram> m_program;
89};
90
91} // namespace WebCore
