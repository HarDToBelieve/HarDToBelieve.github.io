38Ref<WebGLVertexArrayObject> WebGLVertexArrayObject::create(WebGLRenderingContextBase& context, Type type)
39{
40 return adoptRef(*new WebGLVertexArrayObject(context, type));
41}
42
43WebGLVertexArrayObject::~WebGLVertexArrayObject()
44{
45 if (!context())
46 return;
47
48 runDestructor();
49}
50
51WebGLVertexArrayObject::WebGLVertexArrayObject(WebGLRenderingContextBase& context, Type type)
52 : WebGLVertexArrayObjectBase(context, type)
53{
54#if USE(OPENGL_ES)
55 if (m_type != Type::User)
56 return;
57#endif
58 setObject(this->context()->graphicsContextGL()->createVertexArray());
59}
60
61void WebGLVertexArrayObject::deleteObjectImpl(const AbstractLocker& locker, GraphicsContextGL* context3d, PlatformGLObject object)
62{
63 switch (m_type) {
64 case Type::Default:
65 break;
66 case Type::User:
67 context3d->deleteVertexArray(object);
68 break;
69 }
70
71 if (m_boundElementArrayBuffer)
72 m_boundElementArrayBuffer->onDetached(locker, context3d);
73
74 for (auto& state : m_vertexAttribState) {
75 if (state.bufferBinding)
76 state.bufferBinding->onDetached(locker, context3d);
77 }
78}
79
80}
