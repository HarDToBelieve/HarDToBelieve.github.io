37WebGLVertexArrayObjectBase::WebGLVertexArrayObjectBase(WebGLRenderingContextBase& context, Type type)
38 : WebGLContextObject(context)
39 , m_type(type)
40{
41 m_vertexAttribState.resize(context.getMaxVertexAttribs());
42}

44void WebGLVertexArrayObjectBase::setElementArrayBuffer(const AbstractLocker& locker, WebGLBuffer* buffer)
45{
46 if (buffer)
47 buffer->onAttached();
48 if (m_boundElementArrayBuffer)
49 m_boundElementArrayBuffer->onDetached(locker, context()->graphicsContextGL());
50 m_boundElementArrayBuffer = buffer;
51
52}
53
54void WebGLVertexArrayObjectBase::setVertexAttribState(const AbstractLocker& locker, GCGLuint index, GCGLsizei bytesPerElement, GCGLint size, GCGLenum type, GCGLboolean normalized, GCGLsizei stride, GCGLintptr offset, bool isInteger, WebGLBuffer* buffer)
55{
56 GCGLsizei validatedStride = stride ? stride : bytesPerElement;
57
58 auto& state = m_vertexAttribState[index];
59
60 if (buffer)
61 buffer->onAttached();
62 if (state.bufferBinding)
63 state.bufferBinding->onDetached(locker, context()->graphicsContextGL());
64
65 state.bufferBinding = buffer;
66 state.bytesPerElement = bytesPerElement;
67 state.size = size;
68 state.type = type;
69 state.normalized = normalized;
70 state.stride = validatedStride;
71 state.originalStride = stride;
72 state.offset = offset;
73 state.isInteger = isInteger;
74}
75
76void WebGLVertexArrayObjectBase::unbindBuffer(const AbstractLocker& locker, WebGLBuffer& buffer)
77{
78 if (m_boundElementArrayBuffer == &buffer) {
79 m_boundElementArrayBuffer->onDetached(locker, context()->graphicsContextGL());
80 m_boundElementArrayBuffer = nullptr;
81 }
82
83 for (size_t i = 0; i < m_vertexAttribState.size(); ++i) {
84 auto& state = m_vertexAttribState[i];
85 if (state.bufferBinding == &buffer) {
86 buffer.onDetached(locker, context()->graphicsContextGL());
87
88#if !USE(ANGLE)
89 if (!i && !context()->isGLES2Compliant()) {
90 state.bufferBinding = context()->m_vertexAttrib0Buffer;
91 state.bufferBinding->onAttached();
92 state.bytesPerElement = 0;
93 state.size = 4;
94 state.type = GraphicsContextGL::FLOAT;
95 state.normalized = false;
96 state.stride = 16;
97 state.originalStride = 0;
98 state.offset = 0;
99 } else
100#endif
101 state.bufferBinding = nullptr;
102 }
103 }
104}

106void WebGLVertexArrayObjectBase::setVertexAttribDivisor(GCGLuint index, GCGLuint divisor)
107{
108 m_vertexAttribState[index].divisor = divisor;
109}

111void WebGLVertexArrayObjectBase::addMembersToOpaqueRoots(const AbstractLocker&, JSC::AbstractSlotVisitor& visitor)
112{
113 visitor.addOpaqueRoot(m_boundElementArrayBuffer.get());
114 for (auto& state : m_vertexAttribState)
115 visitor.addOpaqueRoot(state.bufferBinding.get());
116}
