42namespace WebCore {
43
44class WebGLVertexArrayObjectBase : public WebGLContextObject {
45public:
46 enum class Type { Default, User };
47
48 // Cached values for vertex attrib range checks
49 struct VertexAttribState {
50 bool isBound() const { return bufferBinding && bufferBinding->object(); }
51 bool validateBinding() const { return !enabled || isBound(); }
52
53 bool enabled { false };
54 RefPtr<WebGLBuffer> bufferBinding;
55 GCGLsizei bytesPerElement { 0 };
56 GCGLint size { 4 };
57 GCGLenum type { GraphicsContextGL::FLOAT };
58 bool normalized { false };
59 GCGLsizei stride { 16 };
60 GCGLsizei originalStride { 0 };
61 GCGLintptr offset { 0 };
62 GCGLuint divisor { 0 };
63 bool isInteger { false };
64 };
65
66 bool isDefaultObject() const { return m_type == Type::Default; }
67
68 bool hasEverBeenBound() const { return object() && m_hasEverBeenBound; }
69 void setHasEverBeenBound() { m_hasEverBeenBound = true; }
70
71 WebGLBuffer* getElementArrayBuffer() const { return m_boundElementArrayBuffer.get(); }
72 void setElementArrayBuffer(const WTF::AbstractLocker&, WebGLBuffer*);
73
74 VertexAttribState& getVertexAttribState(int index) { return m_vertexAttribState[index]; }
75 void setVertexAttribState(const WTF::AbstractLocker&, GCGLuint, GCGLsizei, GCGLint, GCGLenum, GCGLboolean, GCGLsizei, GCGLintptr, bool, WebGLBuffer*);
76 void unbindBuffer(const WTF::AbstractLocker&, WebGLBuffer&);
77
78 void setVertexAttribDivisor(GCGLuint index, GCGLuint divisor);
79
80 void addMembersToOpaqueRoots(const WTF::AbstractLocker&, JSC::AbstractSlotVisitor&);
81
82protected:
83 WebGLVertexArrayObjectBase(WebGLRenderingContextBase&, Type);
84 void deleteObjectImpl(const WTF::AbstractLocker&, GraphicsContextGL*, PlatformGLObject) override = 0;
85
86 Type m_type;
87 bool m_hasEverBeenBound { false };
88 RefPtr<WebGLBuffer> m_boundElementArrayBuffer;
89 Vector<VertexAttribState> m_vertexAttribState;
90};

92} // namespace WebCore
