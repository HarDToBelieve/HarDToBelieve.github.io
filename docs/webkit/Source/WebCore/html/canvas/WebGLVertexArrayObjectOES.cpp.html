37Ref<WebGLVertexArrayObjectOES> WebGLVertexArrayObjectOES::create(WebGLRenderingContextBase& context, Type type)
38{
39 return adoptRef(*new WebGLVertexArrayObjectOES(context, type));
40}
41
42WebGLVertexArrayObjectOES::WebGLVertexArrayObjectOES(WebGLRenderingContextBase& context, Type type)
43 : WebGLVertexArrayObjectBase(context, type)
44{
45 switch (type) {
46 case Type::Default:
47 break;
48 case Type::User:
49 setObject(this->context()->graphicsContextGL()->createVertexArray());
50 break;
51 }
52}
53
54WebGLVertexArrayObjectOES::~WebGLVertexArrayObjectOES()
55{
56 if (!context())
57 return;
58
59 runDestructor();
60}
61
62void WebGLVertexArrayObjectOES::deleteObjectImpl(const WTF::AbstractLocker& locker, GraphicsContextGL* context3d, PlatformGLObject object)
63{
64 switch (m_type) {
65 case Type::Default:
66 break;
67 case Type::User:
68 context3d->deleteVertexArray(object);
69 break;
70 }
71
72 if (m_boundElementArrayBuffer)
73 m_boundElementArrayBuffer->onDetached(locker, context3d);
74
75 for (auto& state : m_vertexAttribState) {
76 if (state.bufferBinding)
77 state.bufferBinding->onDetached(locker, context3d);
78 }
79}
80}
