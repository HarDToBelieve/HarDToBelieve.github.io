35class CSSPreloadScanner {

37public:

39 ~CSSPreloadScanner();
40
41 void reset();
42
43 void scan(const HTMLToken::DataVector&, PreloadRequestStream&);
44
45private:

47 Initial,
48 MaybeComment,
49 Comment,
50 MaybeCommentEnd,
51 RuleStart,
52 Rule,
53 AfterRule,
54 RuleValue,
55 AfterRuleValue,
56 DoneParsingImportRules,
57 };
58
59 inline void tokenize(UChar);
60 void emitRule();
61
62 State m_state;
63 Vector<UChar> m_rule;
64 Vector<UChar> m_ruleValue;
65
66 // Only non-zero during scan()
67 PreloadRequestStream* m_requests;
68};
