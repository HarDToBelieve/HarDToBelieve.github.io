124// This kicks off "Once the user agent stops parsing" as described by:
125// https://html.spec.whatwg.org/multipage/syntax.html#the-end
126void HTMLDocumentParser::prepareToStopParsing()
127{
128 ASSERT(!hasInsertionPoint());
129
130 // pumpTokenizer can cause this parser to be detached from the Document,
131 // but we need to ensure it isn't deleted yet.
132 Ref<HTMLDocumentParser> protectedThis(*this);
133
134 // NOTE: This pump should only ever emit buffered character tokens,
135 // so ForceSynchronous vs. AllowYield should be meaningless.
136 pumpTokenizerIfPossible(ForceSynchronous);
137
138 if (isStopped())
139 return;
140
141 DocumentParser::prepareToStopParsing();
142
143 // We will not have a scriptRunner when parsing a DocumentFragment.
144 if (m_scriptRunner)
145 document()->setReadyState(Document::Interactive);
146
147 // Setting the ready state above can fire mutation event and detach us
148 // from underneath. In that case, just bail out.
149 if (isDetached())
150 return;
151
152 attemptToRunDeferredScriptsAndEnd();
153}

185void HTMLDocumentParser::pumpTokenizerIfPossible(SynchronousMode mode)
186{
187 if (isStopped() || isWaitingForScripts())
188 return;
189
190 // Once a resume is scheduled, HTMLParserScheduler controls when we next pump.
191 if (isScheduledForResume()) {
192 ASSERT(mode == AllowYield);
193 return;
194 }
195
196 pumpTokenizer(mode);
197}

206{
207 // pumpTokenizer can cause this parser to be detached from the Document,
208 // but we need to ensure it isn't deleted yet.

217void HTMLDocumentParser::runScriptsForPausedTreeBuilder()
218{
219 ASSERT(scriptingContentIsAllowed(parserContentPolicy()));
220
221 if (std::unique_ptr<CustomElementConstructionData> constructionData = m_treeBuilder->takeCustomElementConstructionData()) {
222 ASSERT(!m_treeBuilder->hasParserBlockingScriptWork());
223
224 // https://html.spec.whatwg.org/#create-an-element-for-the-token
225 {
226 // Prevent document.open/write during reactions by allocating the incrementer before the reactions stack.
227 ThrowOnDynamicMarkupInsertionCountIncrementer incrementer(*document());
228
229 document()->eventLoop().performMicrotaskCheckpoint();
230
231 CustomElementReactionStack reactionStack(document()->globalObject());
232 auto& elementInterface = constructionData->elementInterface.get();
233 auto newElement = elementInterface.constructElementWithFallback(*document(), constructionData->name);
234 m_treeBuilder->didCreateCustomOrFallbackElement(WTFMove(newElement), *constructionData);
235 }
236 return;
237 }
238
239 TextPosition scriptStartPosition = TextPosition::belowRangePosition();
240 if (auto scriptElement = m_treeBuilder->takeScriptToProcess(scriptStartPosition)) {
241 ASSERT(!m_treeBuilder->hasParserBlockingScriptWork());
242 // We will not have a scriptRunner when parsing a DocumentFragment.
243 if (m_scriptRunner)
244 m_scriptRunner->execute(scriptElement.releaseNonNull(), scriptStartPosition);
245 }
246}

257bool HTMLDocumentParser::pumpTokenizerLoop(SynchronousMode mode, bool parsingFragment, PumpSession& session)
258{
259 do {
260 if (UNLIKELY(isWaitingForScripts())) {
261 if (mode == AllowYield && m_parserScheduler->shouldYieldBeforeExecutingScript(m_treeBuilder->scriptToProcess(), session))
262 return true;
263
264 runScriptsForPausedTreeBuilder();
265 // If we're paused waiting for a script, we try to execute scripts before continuing.
266 if (isWaitingForScripts() || isStopped())
267 return false;
268 }
269
270 // FIXME: It's wrong for the HTMLDocumentParser to reach back to the Frame, but this approach is
271 // how the parser has always handled stopping when the page assigns window.location. What should
272 // happen instead is that assigning window.location causes the parser to stop parsing cleanly.
273 // The problem is we're not prepared to do that at every point where we run JavaScript.
274 if (UNLIKELY(!parsingFragment && document()->frame() && document()->frame()->navigationScheduler().locationChangePending()))
275 return false;
276
277 if (UNLIKELY(mode == AllowYield && m_parserScheduler->shouldYieldBeforeToken(session)))
278 return true;
279
280 if (!parsingFragment)
281 m_sourceTracker.startToken(m_input.current(), m_tokenizer);
282
283 auto token = m_tokenizer.nextToken(m_input.current());
284 if (!token)
285 return false;
286
287 if (!parsingFragment) {
288 m_sourceTracker.endToken(m_input.current(), m_tokenizer);
289
290 // We do not XSS filter innerHTML, which means we (intentionally) fail
291 // http/tests/security/xssAuditor/dom-write-innerHTML.html
292 if (auto xssInfo = m_xssAuditor.filterToken(FilterTokenRequest(*token, m_sourceTracker, m_tokenizer.shouldAllowCDATA())))
293 m_xssAuditorDelegate.didBlockScript(*xssInfo);
294 }
295
296 constructTreeFromHTMLToken(token);
297 } while (!isStopped());
298
299 return false;
300}

302void HTMLDocumentParser::pumpTokenizer(SynchronousMode mode)
303{
304 ASSERT(!isStopped());
305 ASSERT(!isScheduledForResume());
306
307 // This is an attempt to check that this object is both attached to the Document and protected by something.
308 ASSERT(refCount() >= 2);
309
310 PumpSession session(m_pumpSessionNestingLevel, contextForParsingSession());
311
312 m_xssAuditor.init(document(), &m_xssAuditorDelegate);
313
314 auto emitTracePoint = [this](TracePointCode code) {
315 if (!m_shouldEmitTracePoints)
316 return;
317
318 auto position = textPosition();
319 tracePoint(code, position.m_line.oneBasedInt(), position.m_column.oneBasedInt());
320 };
321
322 emitTracePoint(ParseHTMLStart);
323 bool shouldResume = pumpTokenizerLoop(mode, isParsingFragment(), session);
324 emitTracePoint(ParseHTMLEnd);
325
326 // Ensure we haven't been totally deref'ed after pumping. Any caller of this
327 // function should be holding a RefPtr to this to ensure we weren't deleted.
328 ASSERT(refCount() >= 1);
329
330 if (isStopped() || isParsingFragment())
331 return;
332
333 if (shouldResume)
334 m_parserScheduler->scheduleForResume();
335
336 if (isWaitingForScripts() && !isDetached()) {
337 ASSERT(m_tokenizer.isInDataState());
338 if (!m_preloadScanner) {
339 m_preloadScanner = makeUnique<HTMLPreloadScanner>(m_options, document()->url(), document()->deviceScaleFactor());
340 m_preloadScanner->appendToEnd(m_input.current());
341 }
342 m_preloadScanner->scan(*m_preloader, *document());
343 }
344 // The viewport definition is known here, so we can load link preloads with media attributes.
345 if (document()->loader())
346 LinkLoader::loadLinksFromHeader(document()->loader()->response().httpHeaderField(HTTPHeaderName::Link), document()->url(), *document(), LinkLoader::MediaAttributeCheck::MediaAttributeNotEmpty);
347}

349void HTMLDocumentParser::constructTreeFromHTMLToken(HTMLTokenizer::TokenPtr& rawToken)
350{
351 AtomicHTMLToken token(*rawToken);
352
353 // We clear the rawToken in case constructTree
354 // synchronously re-enters the parser. We don't clear the token immedately
355 // for Character tokens because the AtomicHTMLToken avoids copying the
356 // characters by keeping a pointer to the underlying buffer in the
357 // HTMLToken. Fortunately, Character tokens can't cause us to re-enter
358 // the parser.
359 //
360 // FIXME: Stop clearing the rawToken once we start running the parser off
361 // the main thread or once we stop allowing synchronous JavaScript
362 // execution from parseAttribute.
363 if (rawToken->type() != HTMLToken::Character) {
364 // Clearing the TokenPtr makes sure we don't clear the HTMLToken a second time
365 // later when the TokenPtr is destroyed.
366 rawToken.clear();
367 }
368
369 m_treeBuilder->constructTree(WTFMove(token));
370}

372bool HTMLDocumentParser::hasInsertionPoint()
373{
374 // FIXME: The wasCreatedByScript() branch here might not be fully correct.
375 // Our model of the EOF character differs slightly from the one in the spec
376 // because our treatment is uniform between network-sourced and script-sourced
377 // input streams whereas the spec treats them differently.
378 return m_input.hasInsertionPoint() || (wasCreatedByScript() && !m_input.haveSeenEndOfFile());
379}

381void HTMLDocumentParser::insert(SegmentedString&& source)
382{
383 if (isStopped())
384 return;
385
386 // pumpTokenizer can cause this parser to be detached from the Document,
387 // but we need to ensure it isn't deleted yet.
388 Ref<HTMLDocumentParser> protectedThis(*this);
389
390 source.setExcludeLineNumbers();
391 m_input.insertAtCurrentInsertionPoint(WTFMove(source));
392 pumpTokenizerIfPossible(ForceSynchronous);
393
394 if (isWaitingForScripts() && !isDetached()) {
395 // Check the document.write() output with a separate preload scanner as
396 // the main scanner can't deal with insertions.
397 if (!m_insertionPreloadScanner)
398 m_insertionPreloadScanner = makeUnique<HTMLPreloadScanner>(m_options, document()->url(), document()->deviceScaleFactor());
399 m_insertionPreloadScanner->appendToEnd(source);
400 m_insertionPreloadScanner->scan(*m_preloader, *document());
401 }
402
403 endIfDelayed();
404}

406void HTMLDocumentParser::append(RefPtr<StringImpl>&& inputSource)
407{
408 if (isStopped())
409 return;
410
411 // pumpTokenizer can cause this parser to be detached from the Document,
412 // but we need to ensure it isn't deleted yet.
413 Ref<HTMLDocumentParser> protectedThis(*this);
414
415 String source { WTFMove(inputSource) };
416
417 if (m_preloadScanner) {
418 if (m_input.current().isEmpty() && !isWaitingForScripts()) {
419 // We have parsed until the end of the current input and so are now moving ahead of the preload scanner.
420 // Clear the scanner so we know to scan starting from the current input point if we block again.
421 m_preloadScanner = nullptr;
422 } else {
423 m_preloadScanner->appendToEnd(source);
424 if (isWaitingForScripts())
425 m_preloadScanner->scan(*m_preloader, *document());
426 }
427 }
428
429 m_input.appendToEnd(source);
430
431 if (inPumpSession()) {
432 // We've gotten data off the network in a nested write.
433 // We don't want to consume any more of the input stream now. Do
434 // not worry. We'll consume this data in a less-nested write().
435 return;
436 }
437
438 pumpTokenizerIfPossible(AllowYield);
439
440 endIfDelayed();
441}

452void HTMLDocumentParser::attemptToRunDeferredScriptsAndEnd()
453{
454 ASSERT(isStopping());
455 ASSERT(!hasInsertionPoint());
456 if (m_scriptRunner && !m_scriptRunner->executeScriptsWaitingForParsing())
457 return;
458 end();
459}

461void HTMLDocumentParser::attemptToEnd()
462{
463 // finish() indicates we will not receive any more data. If we are waiting on
464 // an external script to load, we can't finish parsing quite yet.
465
466 if (shouldDelayEnd()) {
467 m_endWasDelayed = true;
468 return;
469 }
470 prepareToStopParsing();
471}
472
473void HTMLDocumentParser::endIfDelayed()
474{
475 // If we've already been detached, don't bother ending.
476 if (isDetached())
477 return;
478
479 if (!m_endWasDelayed || shouldDelayEnd())
480 return;
481
482 m_endWasDelayed = false;
483 prepareToStopParsing();
484}

486void HTMLDocumentParser::finish()
487{
488 // FIXME: We should ASSERT(!m_parserStopped) here, since it does not
489 // makes sense to call any methods on DocumentParser once it's been stopped.
490 // However, FrameLoader::stop calls DocumentParser::finish unconditionally.
491
492 // We're not going to get any more data off the network, so we tell the
493 // input stream we've reached the end of file. finish() can be called more
494 // than once, if the first time does not call end().
495 if (!m_input.haveSeenEndOfFile())
496 m_input.markEndOfFile();
497
498 attemptToEnd();
499}

517bool HTMLDocumentParser::isWaitingForScripts() const
518{
519 if (isParsingFragment()) {
520 // HTMLTreeBuilder may have a parser blocking script element but we ignore them during fragment parsing.
521 ASSERT(!m_scriptRunner || !m_scriptRunner->hasParserBlockingScript());
522 return false;
523 }
524 // When the TreeBuilder encounters a </script> tag, it returns to the HTMLDocumentParser
525 // where the script is transfered from the treebuilder to the script runner.
526 // The script runner will hold the script until its loaded and run. During
527 // any of this time, we want to count ourselves as "waiting for a script" and thus
528 // run the preload scanner, as well as delay completion of parsing.
529 bool treeBuilderHasBlockingScript = m_treeBuilder->hasParserBlockingScriptWork();
530 bool scriptRunnerHasBlockingScript = m_scriptRunner && m_scriptRunner->hasParserBlockingScript();
531 // Since the parser is paused while a script runner has a blocking script, it should
532 // never be possible to end up with both objects holding a blocking script.
533 ASSERT(!(treeBuilderHasBlockingScript && scriptRunnerHasBlockingScript));
534 // If either object has a blocking script, the parser should be paused.
535 return treeBuilderHasBlockingScript || scriptRunnerHasBlockingScript;
536}

573void HTMLDocumentParser::notifyFinished(PendingScript& pendingScript)
574{
575 // pumpTokenizer can cause this parser to be detached from the Document,
576 // but we need to ensure it isn't deleted yet.
577 Ref<HTMLDocumentParser> protectedThis(*this);
578
579 // After Document parser is stopped or detached, the parser-inserted deferred script execution should be ignored.
580 if (isStopped())
581 return;
582
583 ASSERT(m_scriptRunner);
584 ASSERT(!isExecutingScript());
585 if (isStopping()) {
586 attemptToRunDeferredScriptsAndEnd();
587 return;
588 }
589
590 m_scriptRunner->executeScriptsWaitingForLoad(pendingScript);
591 if (!isWaitingForScripts())
592 resumeParsingAfterScriptExecution();
593}

600void HTMLDocumentParser::executeScriptsWaitingForStylesheets()
601{
602 // Document only calls this when the Document owns the DocumentParser
603 // so this will not be called in the DocumentFragment case.
604 ASSERT(m_scriptRunner);
605 // Ignore calls unless we have a script blocking the parser waiting on a
606 // stylesheet load. Otherwise we are currently parsing and this
607 // is a re-entrant call from encountering a </ style> tag.
608 if (!m_scriptRunner->hasScriptsWaitingForStylesheets())
609 return;
610
611 // pumpTokenizer can cause this parser to be detached from the Document,
612 // but we need to ensure it isn't deleted yet.
613 Ref<HTMLDocumentParser> protectedThis(*this);
614 m_scriptRunner->executeScriptsWaitingForStylesheets();
615 if (!isWaitingForScripts())
616 resumeParsingAfterScriptExecution();
617}

619void HTMLDocumentParser::parseDocumentFragment(const String& source, DocumentFragment& fragment, Element& contextElement, ParserContentPolicy parserContentPolicy)
620{
621 auto parser = create(fragment, contextElement, parserContentPolicy);
622 parser->insert(source); // Use insert() so that the parser will not yield.
623 parser->finish();
624 ASSERT(!parser->processingData());
625 parser->detach();
626}
