98// Inline implementations of some of the functions declared above.
99
100template<typename CharacterType> inline bool isHTMLSpace(CharacterType character)
101{
102 // Histogram from Apple's page load test combined with some ad hoc browsing some other test suites.
103 //
104 // 82%: 216330 non-space characters, all > U+0020
105 // 11%: 30017 plain space characters, U+0020
106 // 5%: 12099 newline characters, U+000A
107 // 2%: 5346 tab characters, U+0009
108 //
109 // No other characters seen. No U+000C or U+000D, and no other control characters.
110 // Accordingly, we check for non-spaces first, then space, then newline, then tab, then the other characters.
111
112 return character <= ' ' && (character == ' ' || character == '\n' || character == '\t' || character == '\r' || character == '\f');
113}
114
115template<typename CharacterType> inline bool isNotHTMLSpace(CharacterType character)
116{
117 return !isHTMLSpace(character);
118}
119
120inline bool isHTMLLineBreak(UChar character)
121{
122 return character <= '\r' && (character == '\n' || character == '\r');
123}
124
125template<typename CharacterType> inline bool isComma(CharacterType character)
126{
127 return character == ',';
128}
129
130template<typename CharacterType> inline bool isHTMLSpaceOrComma(CharacterType character)
131{
132 return isComma(character) || isHTMLSpace(character);
133}
134
135inline bool isHTMLSpaceButNotLineBreak(UChar character)
136{
137 return isHTMLSpace(character) && !isHTMLLineBreak(character);
138}

140// https://html.spec.whatwg.org/multipage/infrastructure.html#limited-to-only-non-negative-numbers-greater-than-zero
141inline unsigned limitToOnlyHTMLNonNegativeNumbersGreaterThanZero(unsigned value, unsigned defaultValue = 1)
142{
143 return (value > 0 && value <= maxHTMLNonNegativeInteger) ? value : defaultValue;
144}
145
146inline unsigned limitToOnlyHTMLNonNegativeNumbersGreaterThanZero(StringView stringValue, unsigned defaultValue = 1)
147{
148 ASSERT(defaultValue > 0);
149 ASSERT(defaultValue <= maxHTMLNonNegativeInteger);
150 auto optionalValue = parseHTMLNonNegativeInteger(stringValue);
151 unsigned value = optionalValue && optionalValue.value() ? optionalValue.value() : defaultValue;
152 ASSERT(value > 0);
153 ASSERT(value <= maxHTMLNonNegativeInteger);
154 return value;
155}
156
157// https://html.spec.whatwg.org/#reflecting-content-attributes-in-idl-attributes:idl-unsigned-long
158inline unsigned limitToOnlyHTMLNonNegative(unsigned value, unsigned defaultValue = 0)
159{
160 ASSERT(defaultValue <= maxHTMLNonNegativeInteger);
161 return value <= maxHTMLNonNegativeInteger ? value : defaultValue;
162}
163
164inline unsigned limitToOnlyHTMLNonNegative(StringView stringValue, unsigned defaultValue = 0)
165{
166 ASSERT(defaultValue <= maxHTMLNonNegativeInteger);
167 unsigned value = parseHTMLNonNegativeInteger(stringValue).value_or(defaultValue);
168 ASSERT(value <= maxHTMLNonNegativeInteger);
169 return value;
170}
171
172// https://html.spec.whatwg.org/#clamped-to-the-range
173inline unsigned clampHTMLNonNegativeIntegerToRange(StringView stringValue, unsigned min, unsigned max, unsigned defaultValue = 0)
174{
175 ASSERT(defaultValue >= min);
176 ASSERT(defaultValue <= max);
177 auto optionalValue = parseHTMLNonNegativeInteger(stringValue);
178 if (optionalValue)
179 return std::min(std::max(optionalValue.value(), min), max);
180
181 return optionalValue.error() == HTMLIntegerParsingError::PositiveOverflow ? max : defaultValue;
182}
