38HTMLParserOptions::HTMLParserOptions()
39 : scriptingFlag(false)
40 , usePreHTML5ParserQuirks(false)
41 , maximumDOMTreeDepth(Settings::defaultMaximumHTMLParserDOMTreeDepth)
42{
43}
44
45HTMLParserOptions::HTMLParserOptions(Document& document)
46{
47 RefPtr<Frame> frame = document.frame();
48 if (document.settings().htmlParserScriptingFlagPolicy() == HTMLParserScriptingFlagPolicy::Enabled)
49 scriptingFlag = true;
50 else
51 scriptingFlag = frame && frame->script().canExecuteScripts(NotAboutToExecuteScript);
52
53 usePreHTML5ParserQuirks = document.settings().usePreHTML5ParserQuirks();
54 maximumDOMTreeDepth = document.settings().maximumHTMLParserDOMTreeDepth();
55}
