38static Seconds parserTimeLimit(Page* page)
39{
40 // Always yield after exceeding this.
41 constexpr auto defaultParserTimeLimit = 500_ms;
42
43 // We're using the poorly named customHTMLTokenizerTimeDelay setting.
44 if (page && page->hasCustomHTMLTokenizerTimeDelay())
45 return Seconds(page->customHTMLTokenizerTimeDelay());
46 return defaultParserTimeLimit;
47}

49ActiveParserSession::ActiveParserSession(Document* document)
50 : m_document(document)
51{
52 if (!m_document)
53 return;
54 m_document->incrementActiveParserCount();

57ActiveParserSession::~ActiveParserSession()
58{
59 if (!m_document)
60 return;

72HTMLParserScheduler::HTMLParserScheduler(HTMLDocumentParser& parser)
73 : m_parser(parser)

88void HTMLParserScheduler::continueNextChunkTimerFired()
89{
90 ASSERT(!m_suspended);
91
92 // FIXME: The timer class should handle timer priorities instead of this code.
93 // If a layout is scheduled, wait again to let the layout timer run first.
94 if (m_parser.document()->isLayoutTimerActive()) {
95 m_continueNextChunkTimer.startOneShot(0_s);
96 return;
97 }
98 m_parser.resumeParsingAfterYield();
99}

101bool HTMLParserScheduler::shouldYieldBeforeExecutingScript(const ScriptElement* scriptElement, PumpSession& session)
102{
103 // If we've never painted before and a layout is pending, yield prior to running
104 // scripts to give the page a chance to paint earlier.
105 RefPtr<Document> document = m_parser.document();
106
107 session.didSeeScript = true;
108
109 if (!document->body())
110 return false;
111
112 if (!document->haveStylesheetsLoaded())
113 return false;
114
115 if (UNLIKELY(m_documentHasActiveParserYieldTokens))
116 return true;
117
118 auto elapsedTime = MonotonicTime::now() - session.startTime;
119
120 constexpr auto elapsedTimeLimit = 16_ms;
121 // Require at least some new parsed content before yielding.
122 constexpr auto tokenLimit = 256;
123 // Don't yield on very short inline scripts. This is an imperfect way to try to guess the execution cost.
124 constexpr auto inlineScriptLengthLimit = 1024;
125
126 if (elapsedTime < elapsedTimeLimit)
127 return false;
128 if (session.processedTokens < tokenLimit)
129 return false;
130
131 if (scriptElement) {
132 // Async and deferred scripts are not executed by the parser.
133 if (scriptElement->hasAsyncAttribute() || scriptElement->hasDeferAttribute())
134 return false;
135 if (!scriptElement->hasSourceAttribute() && scriptElement->scriptContent().length() < inlineScriptLengthLimit)
136 return false;
137 }
138
139 return true;
140}

142void HTMLParserScheduler::scheduleForResume()
143{
144 ASSERT(!m_suspended);
145 m_continueNextChunkTimer.startOneShot(0_s);
146}
147
148void HTMLParserScheduler::suspend()
149{
150 ASSERT(!m_suspended);
151 ASSERT(!m_isSuspendedWithActiveTimer);
152#if ASSERT_ENABLED
153 m_suspended = true;
154#endif
155
156 if (!m_continueNextChunkTimer.isActive())
157 return;
158 m_isSuspendedWithActiveTimer = true;
159 m_continueNextChunkTimer.stop();
160}
161
162void HTMLParserScheduler::resume()
163{
164 ASSERT(m_suspended);
165 ASSERT(!m_continueNextChunkTimer.isActive());
166#if ASSERT_ENABLED
167 m_suspended = false;
168#endif
169
170 if (!m_isSuspendedWithActiveTimer)
171 return;
172 m_isSuspendedWithActiveTimer = false;
173 m_continueNextChunkTimer.startOneShot(0_s);
174}
