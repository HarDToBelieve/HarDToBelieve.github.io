38class Document;

42class ActiveParserSession {
43public:
44 explicit ActiveParserSession(Document*);
45 ~ActiveParserSession();
46
47private:
48 RefPtr<Document> m_document;
49};

51class PumpSession : public NestingLevelIncrementer, public ActiveParserSession {
52public:
53 PumpSession(unsigned& nestingLevel, Document*);
54 ~PumpSession();
55
56 unsigned processedTokens { 0 };
57 unsigned processedTokensOnLastCheck { 0 };
58 MonotonicTime startTime { MonotonicTime::now() };
59 bool didSeeScript { false };
60};
61
62class HTMLParserScheduler {
63 WTF_MAKE_NONCOPYABLE(HTMLParserScheduler); WTF_MAKE_FAST_ALLOCATED;
64public:
65 explicit HTMLParserScheduler(HTMLDocumentParser&);
66 ~HTMLParserScheduler();
67
68 bool shouldYieldBeforeToken(PumpSession& session)
69 {
70#if PLATFORM(IOS_FAMILY)
71 if (WebThreadShouldYield())
72 return true;
73#endif
74 if (UNLIKELY(m_documentHasActiveParserYieldTokens))
75 return true;
76
77 if (UNLIKELY(session.processedTokens > session.processedTokensOnLastCheck + numberOfTokensBeforeCheckingForYield || session.didSeeScript))
78 return checkForYield(session);
79
80 ++session.processedTokens;
81 return false;
82 }
83 bool shouldYieldBeforeExecutingScript(const ScriptElement*, PumpSession&);
84
85 void scheduleForResume();
86 bool isScheduledForResume() const { return m_isSuspendedWithActiveTimer || m_continueNextChunkTimer.isActive() || m_documentHasActiveParserYieldTokens; }
87
88 void suspend();
89 void resume();
90
91 void didBeginYieldingParser()
92 {
93 ASSERT(!m_documentHasActiveParserYieldTokens);
94 m_documentHasActiveParserYieldTokens = true;
95 }
96
97 void didEndYieldingParser()
98 {
99 ASSERT(m_documentHasActiveParserYieldTokens);
100 m_documentHasActiveParserYieldTokens = false;
101
102 if (!isScheduledForResume())
103 scheduleForResume();
104 }
105
106private:
107 static const unsigned numberOfTokensBeforeCheckingForYield = 4096; // Performance optimization
108
109 void continueNextChunkTimerFired();
110
111 bool checkForYield(PumpSession& session)
112 {
113 session.processedTokensOnLastCheck = session.processedTokens;
114 session.didSeeScript = false;
115
116 Seconds elapsedTime = MonotonicTime::now() - session.startTime;
117 return elapsedTime > m_parserTimeLimit;
118 }
119
120 HTMLDocumentParser& m_parser;
121
122 Seconds m_parserTimeLimit;
123 Timer m_continueNextChunkTimer;
124 bool m_isSuspendedWithActiveTimer;
125#if ASSERT_ENABLED
126 bool m_suspended;
127#endif
128 bool m_documentHasActiveParserYieldTokens { false };
129};
