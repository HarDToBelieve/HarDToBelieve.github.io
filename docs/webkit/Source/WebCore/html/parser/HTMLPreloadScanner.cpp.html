54TokenPreloadScanner::TagId TokenPreloadScanner::tagIdFor(const HTMLToken::DataVector& data)
55{
56 AtomString tagName(data);
57 if (tagName == imgTag)
58 return TagId::Img;
59 if (tagName == inputTag)
60 return TagId::Input;
61 if (tagName == linkTag)
62 return TagId::Link;
63 if (tagName == scriptTag)
64 return TagId::Script;
65 if (tagName == styleTag)
66 return TagId::Style;
67 if (tagName == baseTag)
68 return TagId::Base;
69 if (tagName == templateTag)
70 return TagId::Template;
71 if (tagName == metaTag)
72 return TagId::Meta;
73 if (tagName == pictureTag)
74 return TagId::Picture;
75 if (tagName == sourceTag)
76 return TagId::Source;
77 return TagId::Unknown;
78}

80String TokenPreloadScanner::initiatorFor(TagId tagId)
81{
82 switch (tagId) {
83 case TagId::Source:
84 case TagId::Img:
85 return "img"_s;
86 case TagId::Input:
87 return "input"_s;
88 case TagId::Link:
89 return "link"_s;
90 case TagId::Script:
91 return "script"_s;
92 case TagId::Unknown:
93 case TagId::Style:
94 case TagId::Base:
95 case TagId::Template:
96 case TagId::Meta:
97 case TagId::Picture:
98 ASSERT_NOT_REACHED();
99 return "unknown"_s;
100 }
101 ASSERT_NOT_REACHED();
102 return "unknown"_s;
103}

119 void processAttributes(const HTMLToken::AttributeList& attributes, Vector<bool>& pictureState)
120 {
121 ASSERT(isMainThread());
122 if (m_tagId >= TagId::Unknown)
123 return;
124
125 for (auto& attribute : attributes) {
126 AtomString attributeName(attribute.name);
127 String attributeValue = StringImpl::create8BitIfPossible(attribute.value);
128 processAttribute(attributeName, attributeValue, pictureState);
129 }
130
131 if (m_tagId == TagId::Source && !pictureState.isEmpty() && !pictureState.last() && m_mediaMatched && m_typeMatched && !m_srcSetAttribute.isEmpty()) {
132
133 auto sourceSize = SizesAttributeParser(m_sizesAttribute, m_document).length();
134 ImageCandidate imageCandidate = bestFitSourceForImageAttributes(m_deviceScaleFactor, m_urlToLoad, m_srcSetAttribute, sourceSize);
135 if (!imageCandidate.isEmpty()) {
136 pictureState.last() = true;
137 setUrlToLoad(imageCandidate.string.toString(), true);
138 }
139 }
140
141 // Resolve between src and srcSet if we have them and the tag is img.
142 if (m_tagId == TagId::Img && !m_srcSetAttribute.isEmpty()) {
143 auto sourceSize = SizesAttributeParser(m_sizesAttribute, m_document).length();
144 ImageCandidate imageCandidate = bestFitSourceForImageAttributes(m_deviceScaleFactor, m_urlToLoad, m_srcSetAttribute, sourceSize);
145 setUrlToLoad(imageCandidate.string.toString(), true);
146 }
147
148 if (m_metaIsViewport && !m_metaContent.isNull())
149 m_document.processViewport(m_metaContent, ViewportArguments::ViewportMeta);
150
151 if (m_metaIsDisabledAdaptations && !m_metaContent.isNull())
152 m_document.processDisabledAdaptations(m_metaContent);
153 }

155 std::unique_ptr<PreloadRequest> createPreloadRequest(const URL& predictedBaseURL)
156 {
157 if (!shouldPreload())
158 return nullptr;
159
160 auto type = resourceType();
161 if (!type)
162 return nullptr;
163
164 if (!LinkLoader::isSupportedType(type.value(), m_typeAttribute, m_document))
165 return nullptr;
166
167 // Do not preload if lazyload is possible but metadata fetch is disabled.
168 if (HTMLImageElement::hasLazyLoadableAttributeValue(m_lazyloadAttribute))
169 return nullptr;
170
171 auto request = makeUnique<PreloadRequest>(initiatorFor(m_tagId), m_urlToLoad, predictedBaseURL, type.value(), m_mediaAttribute, m_moduleScript, m_referrerPolicy);
172 request->setCrossOriginMode(m_crossOriginMode);
173 request->setNonce(m_nonceAttribute);
174 request->setScriptIsAsync(m_scriptIsAsync);
175
176 // According to the spec, the module tag ignores the "charset" attribute as the same to the worker's
177 // importScript. But WebKit supports the "charset" for importScript intentionally. So to be consistent,
178 // even for the module tags, we handle the "charset" attribute.
179 request->setCharset(charset());
180 return request;
181 }

183 static bool match(const AtomString& name, const QualifiedName& qName)
184 {
185 ASSERT(isMainThread());
186 return qName.localName() == name;
187 }

189private:
190 void processImageAndScriptAttribute(const AtomString& attributeName, const String& attributeValue)
191 {
192 if (match(attributeName, srcAttr))
193 setUrlToLoad(attributeValue);
194 else if (match(attributeName, crossoriginAttr))
195 m_crossOriginMode = stripLeadingAndTrailingHTMLSpaces(attributeValue);
196 else if (match(attributeName, charsetAttr))
197 m_charset = attributeValue;
198 }

200 void processAttribute(const AtomString& attributeName, const String& attributeValue, const Vector<bool>& pictureState)
201 {
202 bool inPicture = !pictureState.isEmpty();
203 bool alreadyMatchedSource = inPicture && pictureState.last();
204
205 switch (m_tagId) {
206 case TagId::Img:
207 if (inPicture && alreadyMatchedSource)
208 break;
209 if (match(attributeName, srcsetAttr) && m_srcSetAttribute.isNull()) {
210 m_srcSetAttribute = attributeValue;
211 break;
212 }
213 if (match(attributeName, sizesAttr) && m_sizesAttribute.isNull()) {
214 m_sizesAttribute = attributeValue;
215 break;
216 }
217 if (m_document.settings().lazyImageLoadingEnabled()) {
218 if (match(attributeName, loadingAttr) && m_lazyloadAttribute.isNull()) {
219 m_lazyloadAttribute = attributeValue;
220 break;
221 }
222 }
223 processImageAndScriptAttribute(attributeName, attributeValue);
224 break;
225 case TagId::Source:
226 if (inPicture && alreadyMatchedSource)
227 break;
228 if (match(attributeName, srcsetAttr) && m_srcSetAttribute.isNull()) {
229 m_srcSetAttribute = attributeValue;
230 break;
231 }
232 if (match(attributeName, sizesAttr) && m_sizesAttribute.isNull()) {
233 m_sizesAttribute = attributeValue;
234 break;
235 }
236 if (match(attributeName, mediaAttr) && m_mediaAttribute.isNull()) {
237 m_mediaAttribute = attributeValue;
238 auto mediaSet = MediaQuerySet::create(attributeValue, MediaQueryParserContext(m_document));
239 auto documentElement = makeRefPtr(m_document.documentElement());
240 LOG(MediaQueries, "HTMLPreloadScanner %p processAttribute evaluating media queries", this);
241 m_mediaMatched = MediaQueryEvaluator { m_document.printing() ? "print" : "screen", m_document, documentElement ? documentElement->computedStyle() : nullptr }.evaluate(mediaSet.get());
242 }
243 if (match(attributeName, typeAttr) && m_typeAttribute.isNull()) {
244 // when multiple type attributes present: first value wins, ignore subsequent (to match ImageElement parser and Blink behaviours)
245 m_typeAttribute = attributeValue;
246 m_typeMatched &= MIMETypeRegistry::isSupportedImageVideoOrSVGMIMEType(m_typeAttribute);
247 }
248 break;
249 case TagId::Script:
250 if (match(attributeName, typeAttr)) {
251 m_moduleScript = equalLettersIgnoringASCIICase(attributeValue, "module") ? PreloadRequest::ModuleScript::Yes : PreloadRequest::ModuleScript::No;
252 break;
253 } else if (match(attributeName, nonceAttr)) {
254 m_nonceAttribute = attributeValue;
255 break;
256 } else if (match(attributeName, referrerpolicyAttr)) {
257 m_referrerPolicy = parseReferrerPolicy(attributeValue, ReferrerPolicySource::ReferrerPolicyAttribute).value_or(ReferrerPolicy::EmptyString);
258 break;
259 } else if (match(attributeName, nomoduleAttr)) {
260 m_scriptIsNomodule = true;
261 break;
262 } else if (match(attributeName, asyncAttr)) {
263 m_scriptIsAsync = true;
264 break;
265 }
266 processImageAndScriptAttribute(attributeName, attributeValue);
267 break;
268 case TagId::Link:
269 if (match(attributeName, hrefAttr))
270 setUrlToLoad(attributeValue);
271 else if (match(attributeName, relAttr)) {
272 LinkRelAttribute parsedAttribute { m_document, attributeValue };
273 m_linkIsStyleSheet = relAttributeIsStyleSheet(parsedAttribute);
274 m_linkIsPreload = parsedAttribute.isLinkPreload;
275 } else if (match(attributeName, mediaAttr))
276 m_mediaAttribute = attributeValue;
277 else if (match(attributeName, charsetAttr))
278 m_charset = attributeValue;
279 else if (match(attributeName, crossoriginAttr))
280 m_crossOriginMode = stripLeadingAndTrailingHTMLSpaces(attributeValue);
281 else if (match(attributeName, nonceAttr))
282 m_nonceAttribute = attributeValue;
283 else if (match(attributeName, asAttr))
284 m_asAttribute = attributeValue;
285 else if (match(attributeName, typeAttr))
286 m_typeAttribute = attributeValue;
287 else if (match(attributeName, referrerpolicyAttr))
288 m_referrerPolicy = parseReferrerPolicy(attributeValue, ReferrerPolicySource::ReferrerPolicyAttribute).value_or(ReferrerPolicy::EmptyString);
289 break;
290 case TagId::Input:
291 if (match(attributeName, srcAttr))
292 setUrlToLoad(attributeValue);
293 else if (match(attributeName, typeAttr))
294 m_inputIsImage = equalLettersIgnoringASCIICase(attributeValue, "image");
295 break;
296 case TagId::Meta:
297 if (match(attributeName, contentAttr))
298 m_metaContent = attributeValue;
299 else if (match(attributeName, nameAttr))
300 m_metaIsViewport = equalLettersIgnoringASCIICase(attributeValue, "viewport");
301 else if (m_document.settings().disabledAdaptationsMetaTagEnabled() && match(attributeName, nameAttr))
302 m_metaIsDisabledAdaptations = equalLettersIgnoringASCIICase(attributeValue, "disabled-adaptations");
303 break;
304 case TagId::Base:
305 case TagId::Style:
306 case TagId::Template:
307 case TagId::Picture:
308 case TagId::Unknown:
309 break;
310 }
311 }

313 static bool relAttributeIsStyleSheet(const LinkRelAttribute& parsedAttribute)
314 {
315 return parsedAttribute.isStyleSheet && !parsedAttribute.isAlternate && !parsedAttribute.iconType && !parsedAttribute.isDNSPrefetch;

318 void setUrlToLoad(const String& value, bool allowReplacement = false)
319 {
320 // We only respect the first src/href, per HTML5:
321 // http://www.whatwg.org/specs/web-apps/current-work/multipage/tokenization.html#attribute-name-state
322 if (!allowReplacement && !m_urlToLoad.isEmpty())
323 return;
324 String url = stripLeadingAndTrailingHTMLSpaces(value);
325 if (url.isEmpty())
326 return;
327 m_urlToLoad = url;
328 }

335 std::optional<CachedResource::Type> resourceType() const
336 {
337 switch (m_tagId) {
338 case TagId::Script:
339 return CachedResource::Type::Script;
340 case TagId::Img:
341 case TagId::Input:
342 case TagId::Source:
343 ASSERT(m_tagId != TagId::Input || m_inputIsImage);
344 return CachedResource::Type::ImageResource;
345 case TagId::Link:
346 if (m_linkIsStyleSheet)
347 return CachedResource::Type::CSSStyleSheet;
348 if (m_linkIsPreload)
349 return LinkLoader::resourceTypeFromAsAttribute(m_asAttribute, m_document);
350 break;
351 case TagId::Meta:
352 case TagId::Unknown:
353 case TagId::Style:
354 case TagId::Base:
355 case TagId::Template:
356 case TagId::Picture:
357 break;
358 }
359 ASSERT_NOT_REACHED();
360 return CachedResource::Type::RawResource;
361 }
362
363 bool shouldPreload()
364 {
365 if (m_urlToLoad.isEmpty())
366 return false;
367
368 if (protocolIs(m_urlToLoad, "data") || protocolIs(m_urlToLoad, "about"))
369 return false;
370
371 if (m_tagId == TagId::Link && !m_linkIsStyleSheet && !m_linkIsPreload)
372 return false;
373
374 if (m_tagId == TagId::Input && !m_inputIsImage)
375 return false;
376
377 if (m_tagId == TagId::Script && m_moduleScript == PreloadRequest::ModuleScript::No && m_scriptIsNomodule)
378 return false;
379
380 return true;
381 }

416void TokenPreloadScanner::scan(const HTMLToken& token, Vector<std::unique_ptr<PreloadRequest>>& requests, Document& document)
417{
418 switch (token.type()) {
419 case HTMLToken::Character:
420 if (!m_inStyle)
421 return;
422 m_cssScanner.scan(token.characters(), requests);
423 return;
424
425 case HTMLToken::EndTag: {
426 TagId tagId = tagIdFor(token.name());
427 if (tagId == TagId::Template) {
428 if (m_templateCount)
429 --m_templateCount;
430 return;
431 }
432 if (tagId == TagId::Style) {
433 if (m_inStyle)
434 m_cssScanner.reset();
435 m_inStyle = false;
436 } else if (tagId == TagId::Picture && !m_pictureSourceState.isEmpty())
437 m_pictureSourceState.removeLast();
438
439 return;
440 }
441
442 case HTMLToken::StartTag: {
443 if (m_templateCount)
444 return;
445 TagId tagId = tagIdFor(token.name());
446 if (tagId == TagId::Template) {
447 ++m_templateCount;
448 return;
449 }
450 if (tagId == TagId::Style) {
451 m_inStyle = true;
452 return;
453 }
454 if (tagId == TagId::Base) {
455 // The first <base> element is the one that wins.
456 if (!m_predictedBaseElementURL.isEmpty())
457 return;
458 updatePredictedBaseURL(token, document.settings().shouldRestrictBaseURLSchemes());
459 return;
460 }
461 if (tagId == TagId::Picture) {
462 m_pictureSourceState.append(false);
463 return;
464 }
465
466 StartTagScanner scanner(document, tagId, m_deviceScaleFactor);
467 scanner.processAttributes(token.attributes(), m_pictureSourceState);
468 if (auto request = scanner.createPreloadRequest(m_predictedBaseElementURL))
469 requests.append(WTFMove(request));
470 return;
471 }
472
473 default:
474 return;
475 }
476}

478void TokenPreloadScanner::updatePredictedBaseURL(const HTMLToken& token, bool shouldRestrictBaseURLSchemes)
479{
480 ASSERT(m_predictedBaseElementURL.isEmpty());
481 auto* hrefAttribute = findAttribute(token.attributes(), hrefAttr->localName().string());
482 if (!hrefAttribute)
483 return;
484 URL temp { m_documentURL, stripLeadingAndTrailingHTMLSpaces(StringImpl::create8BitIfPossible(hrefAttribute->value)) };
485 if (!shouldRestrictBaseURLSchemes || SecurityPolicy::isBaseURLSchemeAllowed(temp))
486 m_predictedBaseElementURL = temp.isolatedCopy();
487}

500void HTMLPreloadScanner::scan(HTMLResourcePreloader& preloader, Document& document)
501{
502 ASSERT(isMainThread()); // HTMLTokenizer::updateStateFor only works on the main thread.
503
504 const URL& startingBaseElementURL = document.baseElementURL();
505
506 // When we start scanning, our best prediction of the baseElementURL is the real one!
507 if (!startingBaseElementURL.isEmpty())
508 m_scanner.setPredictedBaseElementURL(startingBaseElementURL);
509
510 PreloadRequestStream requests;
511
512 while (auto token = m_tokenizer.nextToken(m_source)) {
513 if (token->type() == HTMLToken::StartTag)
514 m_tokenizer.updateStateFor(AtomString(token->name()));
515 m_scanner.scan(*token, requests, document);
516 }
517
518 preloader.preload(WTFMove(requests));
519}

521bool testPreloadScannerViewportSupport(Document* document)
522{
523 ASSERT(document);
524 HTMLParserOptions options(*document);
525 HTMLPreloadScanner scanner(options, document->url());
526 HTMLResourcePreloader preloader(*document);
527 scanner.appendToEnd(String("<meta name=viewport content='width=400'>"));
528 scanner.scan(preloader, *document);
529 return (document->viewportArguments().width == 400);
530}
