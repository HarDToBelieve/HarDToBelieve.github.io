28#include <wtf/Noncopyable.h>
29
30namespace WebCore {
31
32class NestingLevelIncrementer {
33 WTF_MAKE_NONCOPYABLE(NestingLevelIncrementer);
34public:
35 explicit NestingLevelIncrementer(unsigned& nestingLevel)
36 : m_nestingLevel(&nestingLevel)
37 {
38 ++(*m_nestingLevel);
39 }
40
41 ~NestingLevelIncrementer()
42 {
43 --(*m_nestingLevel);
44 }
45
46private:
47 unsigned* m_nestingLevel;
48};

50} // namespace WebCore
