46XSSAuditorDelegate::XSSAuditorDelegate(Document& document)
47 : m_document(document)
48{
49 ASSERT(isMainThread());
50}
51
52static inline String buildConsoleError(const XSSInfo& xssInfo)
53{
54 return makeString("The XSS Auditor ",
55 xssInfo.m_didBlockEntirePage ? "blocked access to" : "refused to execute a script in", " '",
56 xssInfo.m_originalURL, "' because ",
57 xssInfo.m_didBlockEntirePage ? "the source code of a script" : "its source code",
58 " was found within the request.",
59 xssInfo.m_didSendXSSProtectionHeader ? " The server sent an 'X-XSS-Protection' header requesting this behavior."
60 : " The auditor was enabled because the server did not send an 'X-XSS-Protection' header.");
61}

63Ref<FormData> XSSAuditorDelegate::generateViolationReport(const XSSInfo& xssInfo)
64{
65 ASSERT(isMainThread());
66
67 auto& frameLoader = m_document.frame()->loader();
68 String httpBody;
69 if (frameLoader.documentLoader()) {
70 if (auto formData = makeRefPtr(frameLoader.documentLoader()->originalRequest().httpBody()))
71 httpBody = formData->flattenToString();
72 }
73
74 auto reportDetails = JSON::Object::create();
75 reportDetails->setString("request-url", xssInfo.m_originalURL);
76 reportDetails->setString("request-body", httpBody);
77
78 auto reportObject = JSON::Object::create();
79 reportObject->setObject("xss-report", WTFMove(reportDetails));
80
81 return FormData::create(reportObject->toJSONString().utf8().data());
82}
83
84void XSSAuditorDelegate::didBlockScript(const XSSInfo& xssInfo)
85{
86 ASSERT(isMainThread());
87
88 m_document.addConsoleMessage(MessageSource::JS, MessageLevel::Error, buildConsoleError(xssInfo));
89
90 FrameLoader& frameLoader = m_document.frame()->loader();
91 if (xssInfo.m_didBlockEntirePage)
92 frameLoader.stopAllLoaders();
93
94 if (!m_didSendNotifications) {
95 m_didSendNotifications = true;
96
97 frameLoader.client().didDetectXSS(m_document.url(), xssInfo.m_didBlockEntirePage);
98
99 if (!m_reportURL.isEmpty())
100 PingLoader::sendViolationReport(*m_document.frame(), m_reportURL, generateViolationReport(xssInfo), ViolationReportType::XSSAuditor);
101 }
102
103 if (xssInfo.m_didBlockEntirePage)
104 m_document.frame()->navigationScheduler().schedulePageBlock(m_document);
105}
