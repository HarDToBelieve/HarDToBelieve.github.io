53void AutoFillButtonElement::defaultEventHandler(Event& event)
54{
55 if (!is<MouseEvent>(event)) {
56 if (!event.defaultHandled())
57 HTMLDivElement::defaultEventHandler(event);
58 return;
59 }
60
61 MouseEvent& mouseEvent = downcast<MouseEvent>(event);
62
63 if (mouseEvent.type() == eventNames().clickEvent) {
64 m_owner.autoFillButtonElementWasClicked();
65 event.setDefaultHandled();
66 }
67
68 if (!event.defaultHandled())
69 HTMLDivElement::defaultEventHandler(event);
70}
