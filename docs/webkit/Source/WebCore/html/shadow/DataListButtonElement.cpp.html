56void DataListButtonElement::defaultEventHandler(Event& event)
57{
58 if (!is<MouseEvent>(event)) {
59 if (!event.defaultHandled())
60 HTMLDivElement::defaultEventHandler(event);
61 return;
62 }
63
64 MouseEvent& mouseEvent = downcast<MouseEvent>(event);
65
66 if (mouseEvent.type() == eventNames().clickEvent) {
67 m_owner.dataListButtonElementWasClicked();
68 event.setDefaultHandled();
69 }
70
71 if (!event.defaultHandled())
72 HTMLDivElement::defaultEventHandler(event);
73}
