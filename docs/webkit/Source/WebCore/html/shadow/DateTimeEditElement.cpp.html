78void DateTimeEditBuilder::visitField(DateTimeFormat::FieldType fieldType, int count)
79{
80 Document& document = m_editElement.document();
81
82 switch (fieldType) {
83 case DateTimeFormat::FieldTypeDayOfMonth: {
84 m_editElement.addField(DateTimeDayFieldElement::create(document, m_editElement));
85 return;
86 }
87
88 case DateTimeFormat::FieldTypeFractionalSecond: {
89 m_editElement.addField(DateTimeMillisecondFieldElement::create(document, m_editElement));
90 return;
91 }
92
93 case DateTimeFormat::FieldTypeHour11: {
94 m_editElement.addField(DateTimeHourFieldElement::create(document, m_editElement, 0, 11));
95 return;
96 }
97
98 case DateTimeFormat::FieldTypeHour12: {
99 m_editElement.addField(DateTimeHourFieldElement::create(document, m_editElement, 1, 12));
100 return;
101 }
102
103 case DateTimeFormat::FieldTypeHour23: {
104 m_editElement.addField(DateTimeHourFieldElement::create(document, m_editElement, 0, 23));
105 return;
106 }
107
108 case DateTimeFormat::FieldTypeHour24: {
109 m_editElement.addField(DateTimeHourFieldElement::create(document, m_editElement, 1, 24));
110 return;
111 }
112
113 case DateTimeFormat::FieldTypeMinute: {
114 m_editElement.addField(DateTimeMinuteFieldElement::create(document, m_editElement));
115 return;
116 }
117
118 case DateTimeFormat::FieldTypeMonth:
119 case DateTimeFormat::FieldTypeMonthStandAlone: {
120 constexpr int countForAbbreviatedMonth = 3;
121 constexpr int countForFullMonth = 4;
122 constexpr int countForNarrowMonth = 5;
123
124 switch (count) {
125 case countForNarrowMonth:
126 case countForAbbreviatedMonth: {
127 auto field = DateTimeSymbolicMonthFieldElement::create(document, m_editElement, fieldType == DateTimeFormat::FieldTypeMonth ? m_parameters.locale.shortMonthLabels() : m_parameters.locale.shortStandAloneMonthLabels());
128 m_editElement.addField(field);
129 return;
130 }
131 case countForFullMonth: {
132 auto field = DateTimeSymbolicMonthFieldElement::create(document, m_editElement, fieldType == DateTimeFormat::FieldTypeMonth ? m_parameters.locale.monthLabels() : m_parameters.locale.standAloneMonthLabels());
133 m_editElement.addField(field);
134 return;
135 }
136 default:
137 m_editElement.addField(DateTimeMonthFieldElement::create(document, m_editElement));
138 return;
139 }
140 }
141
142 case DateTimeFormat::FieldTypePeriod: {
143 m_editElement.addField(DateTimeMeridiemFieldElement::create(document, m_editElement, m_parameters.locale.timeAMPMLabels()));
144 return;
145 }
146
147 case DateTimeFormat::FieldTypeSecond: {
148 m_editElement.addField(DateTimeSecondFieldElement::create(document, m_editElement));
149
150 if (m_parameters.shouldHaveMillisecondField) {
151 visitLiteral(m_parameters.locale.localizedDecimalSeparator());
152 visitField(DateTimeFormat::FieldTypeFractionalSecond, 3);
153 }
154 return;
155 }
156
157 case DateTimeFormat::FieldTypeYear: {
158 m_editElement.addField(DateTimeYearFieldElement::create(document, m_editElement));
159 return;
160 }
161
162 default:
163 return;
164 }
165}

167void DateTimeEditBuilder::visitLiteral(const String& text)
168{
169 ASSERT(text.length());
170 static MainThreadNeverDestroyed<const AtomString> textPseudoId("-webkit-datetime-edit-text", AtomString::ConstructFromLiteral);
171 auto element = HTMLDivElement::create(m_editElement.document());
172 element->setPseudo(textPseudoId);
173
174 // If the literal begins/ends with a space, the gap between two fields will appear
175 // exaggerated due to the presence of a 1px padding around each field. This can
176 // make spaces appear up to 2px larger between fields. This padding is necessary to
177 // prevent selected fields from appearing squished. To fix, pull fields closer
178 // together by applying a negative margin.
179 if (text.startsWith(' '))
180 element->setInlineStyleProperty(CSSPropertyMarginInlineStart, -1, CSSUnitType::CSS_PX);
181 if (text.endsWith(' '))
182 element->setInlineStyleProperty(CSSPropertyMarginInlineEnd, -1, CSSUnitType::CSS_PX);
183
184 element->appendChild(Text::create(m_editElement.document(), text));
185 m_editElement.fieldsWrapperElement().appendChild(element);
186}

220DateTimeFieldElement* DateTimeEditElement::focusedFieldElement() const
221{
222 auto* focusedElement = document().focusedElement();
223 auto fieldIndex = m_fields.findMatching([&] (auto& field) {
224 return field.ptr() == focusedElement;
225 });
226
227 if (fieldIndex == notFound)
228 return nullptr;
229
230 return m_fields[fieldIndex].ptr();
231}

241void DateTimeEditElement::layout(const LayoutParameters& layoutParameters)
242{
243 static MainThreadNeverDestroyed<const AtomString> fieldsWrapperPseudoId("-webkit-datetime-edit-fields-wrapper", AtomString::ConstructFromLiteral);
244
245 if (!firstChild()) {
246 auto element = HTMLDivElement::create(document());
247 element->setPseudo(fieldsWrapperPseudoId);
248 appendChild(element);
249 }
250
251 Element& fieldsWrapper = fieldsWrapperElement();
252 auto* focusedField = focusedFieldElement();
253
254 DateTimeEditBuilder builder(*this, layoutParameters);
255 Node* lastChildToBeRemoved = fieldsWrapper.lastChild();
256 if (!builder.build(layoutParameters.dateTimeFormat) || m_fields.isEmpty()) {
257 lastChildToBeRemoved = fieldsWrapper.lastChild();
258 builder.build(layoutParameters.fallbackDateTimeFormat);
259 }
260
261 if (focusedField) {
262 auto& focusedFieldId = focusedField->shadowPseudoId();
263
264 auto foundFieldToFocus = false;
265 for (auto& field : m_fields) {
266 if (field->shadowPseudoId() == focusedFieldId) {
267 foundFieldToFocus = true;
268 field->focus();
269 break;
270 }
271 }
272
273 if (!foundFieldToFocus)
274 focusOnNextFocusableField(0);
275 }
276
277 if (lastChildToBeRemoved) {
278 while (auto* childNode = fieldsWrapper.firstChild()) {
279 fieldsWrapper.removeChild(*childNode);
280 if (childNode == lastChildToBeRemoved)
281 break;
282 }
283 }
284}

286void DateTimeEditElement::didBlurFromField(Event& event)
287{
288 if (!m_editControlOwner)
289 return;
290
291 if (auto* newFocusedElement = event.relatedTarget()) {
292 bool didFocusSiblingField = notFound != m_fields.findMatching([&] (auto& field) {
293 return field.ptr() == newFocusedElement;
294 });
295
296 if (didFocusSiblingField)
297 return;
298 }
299
300 m_editControlOwner->didBlurFromControl();
301}

309bool DateTimeEditElement::focusOnNextFocusableField(size_t startIndex)
310{
311 for (size_t i = startIndex; i < m_fields.size(); ++i) {
312 if (m_fields[i]->isFocusable()) {
313 m_fields[i]->focus();
314 return true;
315 }
316 }
317 return false;
318}
319
320void DateTimeEditElement::focusByOwner()
321{
322 focusOnNextFocusableField(0);
323}

325bool DateTimeEditElement::focusOnNextField(const DateTimeFieldElement& field)
326{
327 auto startFieldIndex = fieldIndexOf(field);
328 if (startFieldIndex == notFound)
329 return false;
330
331 return focusOnNextFocusableField(startFieldIndex + 1);
332}
333
334bool DateTimeEditElement::focusOnPreviousField(const DateTimeFieldElement& field)
335{
336 auto startFieldIndex = fieldIndexOf(field);
337 if (startFieldIndex == notFound)
338 return false;
339
340 auto fieldIndex = startFieldIndex;
341 while (fieldIndex > 0) {
342 --fieldIndex;
343 if (m_fields[fieldIndex]->isFocusable()) {
344 m_fields[fieldIndex]->focus();
345 return true;
346 }
347 }
348
349 return false;
350}
