66std::optional<Style::ElementStyle> DateTimeFieldElement::resolveCustomStyle(const RenderStyle& parentStyle, const RenderStyle* shadowHostStyle)
67{
68 auto elementStyle = resolveStyle(&parentStyle);
69 if (!elementStyle.renderStyle)
70 return std::nullopt;
71
72 auto& style = *elementStyle.renderStyle;
73 adjustMinWidth(style);
74
75 if (!hasValue() && shadowHostStyle) {
76 auto textColor = shadowHostStyle->visitedDependentColorWithColorFilter(CSSPropertyColor);
77 auto backgroundColor = shadowHostStyle->visitedDependentColorWithColorFilter(CSSPropertyBackgroundColor);
78 style.setColor(RenderTheme::singleton().datePlaceholderTextColor(textColor, backgroundColor));
79 }
80
81 return elementStyle;
82}

84void DateTimeFieldElement::defaultEventHandler(Event& event)
85{
86 if (event.type() == eventNames().blurEvent)
87 handleBlurEvent(event);
88
89 if (is<KeyboardEvent>(event)) {
90 auto& keyboardEvent = downcast<KeyboardEvent>(event);
91 if (!isFieldOwnerDisabled() && !isFieldOwnerReadOnly()) {
92 handleKeyboardEvent(keyboardEvent);
93 if (keyboardEvent.defaultHandled())
94 return;
95 }
96
97 defaultKeyboardEventHandler(keyboardEvent);
98 if (keyboardEvent.defaultHandled())
99 return;
100 }
101
102 HTMLDivElement::defaultEventHandler(event);
103}
104
105void DateTimeFieldElement::defaultKeyboardEventHandler(KeyboardEvent& keyboardEvent)
106{
107 if (isFieldOwnerDisabled())
108 return;
109
110 if (keyboardEvent.type() != eventNames().keydownEvent)
111 return;
112
113 auto key = keyboardEvent.keyIdentifier();
114 auto code = keyboardEvent.code();
115
116 if (key == "Left" && m_fieldOwner && m_fieldOwner->focusOnPreviousField(*this)) {
117 keyboardEvent.setDefaultHandled();
118 return;
119 }
120
121 if ((key == "Right" || code == "Comma" || code == "Minus" || code == "Period" || code == "Slash" || code == "Semicolon")
122 && m_fieldOwner && m_fieldOwner->focusOnNextField(*this)) {
123 keyboardEvent.setDefaultHandled();
124 return;
125 }
126
127 if (isFieldOwnerReadOnly())
128 return;
129
130 if (key == "Up") {
131 stepUp();
132 keyboardEvent.setDefaultHandled();
133 return;
134 }
135
136 if (key == "Down") {
137 stepDown();
138 keyboardEvent.setDefaultHandled();
139 return;
140 }
141
142 // Clear value when pressing backspace or delete.
143 if (key == "U+0008" || key == "U+007F") {
144 setEmptyValue(DispatchInputAndChangeEvents);
145 keyboardEvent.setDefaultHandled();
146 return;
147 }
148}

188void DateTimeFieldElement::updateVisibleValue(EventBehavior eventBehavior)
189{
190 if (!firstChild())
191 appendChild(Text::create(document(), emptyString()));
192
193 auto& textNode = downcast<Text>(*firstChild());
194 String newVisibleValue = visibleValue();
195 if (textNode.wholeText() != newVisibleValue)
196 textNode.replaceWholeText(newVisibleValue);
197
198 if (eventBehavior == DispatchInputAndChangeEvents && m_fieldOwner)
199 m_fieldOwner->fieldValueChanged();
200}
