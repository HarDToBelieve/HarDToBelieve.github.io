36namespace WebCore {
37
38class DateComponents;
39class RenderStyle;
40
41struct DateTimeFieldsState;
42
43class DateTimeFieldElement : public HTMLDivElement {
44 WTF_MAKE_ISO_ALLOCATED(DateTimeFieldElement);
45public:
46 enum EventBehavior : bool { DispatchNoEvent, DispatchInputAndChangeEvents };
47
48 class FieldOwner : public CanMakeWeakPtr<FieldOwner> {
49 public:
50 virtual ~FieldOwner();
51 virtual void didBlurFromField(Event&) = 0;
52 virtual void fieldValueChanged() = 0;
53 virtual bool focusOnNextField(const DateTimeFieldElement&) = 0;
54 virtual bool focusOnPreviousField(const DateTimeFieldElement&) = 0;
55 virtual bool isFieldOwnerDisabled() const = 0;
56 virtual bool isFieldOwnerReadOnly() const = 0;
57 virtual AtomString localeIdentifier() const = 0;
58 virtual const GregorianDateTime& placeholderDate() const = 0;
59 };
60
61 void defaultEventHandler(Event&) override;
62 bool isFocusable() const final;
63
64 String visibleValue() const;
65
66 virtual bool hasValue() const = 0;
67 virtual void populateDateTimeFieldsState(DateTimeFieldsState&) = 0;
68 virtual void setEmptyValue(EventBehavior = DispatchNoEvent) = 0;
69 virtual void setValueAsDate(const DateComponents&) = 0;
70 virtual void setValueAsInteger(int, EventBehavior = DispatchNoEvent) = 0;
71 virtual void stepDown() = 0;
72 virtual void stepUp() = 0;
73 virtual String value() const = 0;
74 virtual String placeholderValue() const = 0;
75
76protected:
77 DateTimeFieldElement(Document&, FieldOwner&);
78 void initialize(const AtomString& pseudo);
79 Locale& localeForOwner() const;
80 AtomString localeIdentifier() const;
81 void updateVisibleValue(EventBehavior);
82 virtual void adjustMinWidth(RenderStyle&) const = 0;
83 virtual int valueAsInteger() const = 0;
84 virtual void handleKeyboardEvent(KeyboardEvent&) = 0;
85 virtual void handleBlurEvent(Event&);
86
87private:
88 std::optional<Style::ElementStyle> resolveCustomStyle(const RenderStyle&, const RenderStyle*) final;
89
90 bool supportsFocus() const override;
91
92 void defaultKeyboardEventHandler(KeyboardEvent&);
93 bool isFieldOwnerDisabled() const;
94 bool isFieldOwnerReadOnly() const;
95
96 WeakPtr<FieldOwner> m_fieldOwner;
97};

99} // namespace WebCore
