64void DateTimeNumericFieldElement::adjustMinWidth(RenderStyle& style) const
65{
66 auto& font = style.fontCascade();
67
68 unsigned length = 2;
69 if (m_range.maximum > 999)
70 length = 4;
71 else if (m_range.maximum > 99)
72 length = 3;
73
74 auto& locale = localeForOwner();
75
76 float width = 0;
77 for (char c = '0'; c <= '9'; ++c) {
78 auto numberString = locale.convertToLocalizedNumber(makeString(pad(c, length, makeString(c))));
79 width = std::max(width, font.width(RenderBlock::constructTextRun(numberString, style)));
80 }
81
82 style.setMinWidth({ width, LengthType::Fixed });
83}

90String DateTimeNumericFieldElement::formatValue(int value) const
91{
92 Locale& locale = localeForOwner();
93
94 if (m_range.maximum > 999)
95 return locale.convertToLocalizedNumber(makeString(pad('0', 4, value)));
96 if (m_range.maximum > 99)
97 return locale.convertToLocalizedNumber(makeString(pad('0', 3, value)));
98 return locale.convertToLocalizedNumber(makeString(pad('0', 2, value)));
99}

132void DateTimeNumericFieldElement::stepDown()
133{
134 int newValue = m_hasValue ? m_value - 1 : m_range.maximum;
135 if (!m_range.isInRange(newValue))
136 newValue = m_range.maximum;
137 setValueAsIntegerByStepping(newValue);
138}
139
140void DateTimeNumericFieldElement::stepUp()
141{
142 int newValue = m_hasValue ? m_value + 1 : m_range.minimum;
143 if (!m_range.isInRange(newValue))
144 newValue = m_range.minimum;
145 setValueAsIntegerByStepping(newValue);
146}

163void DateTimeNumericFieldElement::handleKeyboardEvent(KeyboardEvent& keyboardEvent)
164{
165 if (keyboardEvent.type() != eventNames().keypressEvent)
166 return;
167
168 auto charCode = static_cast<UChar>(keyboardEvent.charCode());
169 String number = localeForOwner().convertFromLocalizedNumber(String(&charCode, 1));
170 int digit = number[0] - '0';
171 if (digit < 0 || digit > 9)
172 return;
173
174 Seconds timeSinceLastDigitChar = keyboardEvent.timeStamp() - m_lastDigitCharTime;
175 m_lastDigitCharTime = keyboardEvent.timeStamp();
176
177 if (timeSinceLastDigitChar > typeAheadTimeout) {
178 m_typeAheadBuffer.clear();
179 } else if (auto length = m_typeAheadBuffer.length()) {
180 unsigned maxLength = formatValue(m_range.maximum).length();
181 if (length == maxLength)
182 m_typeAheadBuffer.clear();
183 }
184
185 m_typeAheadBuffer.append(number);
186 setValueAsInteger(parseIntegerAllowingTrailingJunk<int>(m_typeAheadBuffer).value_or(0), DispatchInputAndChangeEvents);
187
188 keyboardEvent.setDefaultHandled();
189}
