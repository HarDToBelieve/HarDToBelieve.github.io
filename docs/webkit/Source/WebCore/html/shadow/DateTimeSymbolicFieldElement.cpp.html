45DateTimeSymbolicFieldElement::DateTimeSymbolicFieldElement(Document& document, FieldOwner& fieldOwner, const Vector<String>& symbols, int placeholderIndex)
46 : DateTimeFieldElement(document, fieldOwner)
47 , m_symbols(symbols)
48 , m_placeholderIndex(placeholderIndex)
49 , m_typeAhead(this)
50{
51 ASSERT(!m_symbols.isEmpty());
52}
53
54void DateTimeSymbolicFieldElement::adjustMinWidth(RenderStyle& style) const
55{
56 auto& font = style.fontCascade();
57
58 float width = 0;
59 for (auto& symbol : m_symbols)
60 width = std::max(width, font.width(RenderBlock::constructTextRun(symbol, style)));
61
62 style.setMinWidth({ width, LengthType::Fixed });
63}
64
65bool DateTimeSymbolicFieldElement::hasValue() const
66{
67 return m_selectedIndex >= 0;
68}
69
70void DateTimeSymbolicFieldElement::initialize(const AtomString& pseudo)
71{
72 DateTimeFieldElement::initialize(pseudo);
73}
74
75void DateTimeSymbolicFieldElement::setEmptyValue(EventBehavior eventBehavior)
76{
77 m_selectedIndex = invalidIndex;
78 updateVisibleValue(eventBehavior);
79}
80
81void DateTimeSymbolicFieldElement::setValueAsInteger(int newSelectedIndex, EventBehavior eventBehavior)
82{
83 m_selectedIndex = std::max(0, std::min(newSelectedIndex, static_cast<int>(m_symbols.size() - 1)));
84 updateVisibleValue(eventBehavior);
85}
86
87void DateTimeSymbolicFieldElement::stepDown()
88{
89 int newValue = hasValue() ? m_selectedIndex - 1 : m_symbols.size() - 1;
90 if (newValue < 0)
91 newValue = m_symbols.size() - 1;
92 setValueAsInteger(newValue, DispatchInputAndChangeEvents);
93}
94
95void DateTimeSymbolicFieldElement::stepUp()
96{
97 int newValue = hasValue() ? m_selectedIndex + 1 : 0;
98 if (newValue >= static_cast<int>(m_symbols.size()))
99 newValue = 0;
100 setValueAsInteger(newValue, DispatchInputAndChangeEvents);
101}
102
103String DateTimeSymbolicFieldElement::value() const
104{
105 return hasValue() ? m_symbols[m_selectedIndex] : emptyString();
106}
107
108String DateTimeSymbolicFieldElement::placeholderValue() const
109{
110 return m_symbols[m_placeholderIndex];
111}
112
113int DateTimeSymbolicFieldElement::valueAsInteger() const
114{
115 return m_selectedIndex;
116}
117
118void DateTimeSymbolicFieldElement::handleKeyboardEvent(KeyboardEvent& keyboardEvent)
119{
120 if (keyboardEvent.type() != eventNames().keypressEvent)
121 return;
122
123 keyboardEvent.setDefaultHandled();
124
125 int index = m_typeAhead.handleEvent(&keyboardEvent, TypeAhead::MatchPrefix | TypeAhead::CycleFirstChar | TypeAhead::MatchIndex);
126 if (index < 0)
127 return;
128 setValueAsInteger(index, DispatchInputAndChangeEvents);
129}

131int DateTimeSymbolicFieldElement::indexOfSelectedOption() const
