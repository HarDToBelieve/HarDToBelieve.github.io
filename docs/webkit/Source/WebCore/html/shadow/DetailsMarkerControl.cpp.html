44Ref<DetailsMarkerControl> DetailsMarkerControl::create(Document& document)
45{
46 auto control = adoptRef(*new DetailsMarkerControl(document));
47 static MainThreadNeverDestroyed<const AtomString> webkitDetailsMarkerName("-webkit-details-marker", AtomString::ConstructFromLiteral);
48 control->setPseudo(webkitDetailsMarkerName);
49 return control;
50}
51
52DetailsMarkerControl::DetailsMarkerControl(Document& document)
53 : HTMLDivElement(HTMLNames::divTag, document)
54{
55}
56
57RenderPtr<RenderElement> DetailsMarkerControl::createElementRenderer(RenderStyle&& style, const RenderTreePosition&)
58{
59 return createRenderer<RenderDetailsMarker>(*this, WTFMove(style));
60}
61
62bool DetailsMarkerControl::rendererIsNeeded(const RenderStyle& style)
63{
64 return downcast<HTMLSummaryElement>(shadowHost())->isActiveSummary() && HTMLDivElement::rendererIsNeeded(style);
65}
