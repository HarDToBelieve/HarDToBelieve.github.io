67inline static Decimal sliderPosition(HTMLInputElement& element)
68{
69 const StepRange stepRange(element.createStepRange(AnyStepHandling::Reject));
70 const Decimal oldValue = parseToDecimalForNumberType(element.value(), stepRange.defaultValue());
71 return stepRange.proportionFromValue(stepRange.clampValue(oldValue));
72}

74inline static bool hasVerticalAppearance(HTMLInputElement& input)
75{
76 ASSERT(input.renderer());
77 const RenderStyle& sliderStyle = input.renderer()->style();
78
79#if ENABLE(VIDEO)
80 if (sliderStyle.appearance() == MediaVolumeSliderPart && input.renderer()->theme().usesVerticalVolumeSlider())
81 return true;
82#endif
83
84 return sliderStyle.appearance() == SliderVerticalPart;
85}

94void RenderSliderThumb::updateAppearance(const RenderStyle* parentStyle)
95{
96 if (parentStyle->appearance() == SliderVerticalPart)
97 mutableStyle().setAppearance(SliderThumbVerticalPart);
98 else if (parentStyle->appearance() == SliderHorizontalPart)
99 mutableStyle().setAppearance(SliderThumbHorizontalPart);
100 else if (parentStyle->appearance() == MediaSliderPart)
101 mutableStyle().setAppearance(MediaSliderThumbPart);
102 else if (parentStyle->appearance() == MediaVolumeSliderPart)
103 mutableStyle().setAppearance(MediaVolumeSliderThumbPart);
104 else if (parentStyle->appearance() == MediaFullScreenVolumeSliderPart)
105 mutableStyle().setAppearance(MediaFullScreenVolumeSliderThumbPart);
106 if (style().hasAppearance()) {
107 ASSERT(element());
108 theme().adjustSliderThumbSize(mutableStyle(), element());
109 }

137RenderBox::LogicalExtentComputedValues RenderSliderContainer::computeLogicalHeight(LayoutUnit logicalHeight, LayoutUnit logicalTop) const
138{
139 ASSERT(element()->shadowHost());
140 auto& input = downcast<HTMLInputElement>(*element()->shadowHost());
141 bool isVertical = hasVerticalAppearance(input);
142
143#if ENABLE(DATALIST_ELEMENT)
144 if (input.renderer()->isSlider() && !isVertical && input.list()) {
145 int offsetFromCenter = theme().sliderTickOffsetFromTrackCenter();
146 LayoutUnit trackHeight;
147 if (offsetFromCenter < 0)
148 trackHeight = -2 * offsetFromCenter;
149 else {
150 int tickLength = theme().sliderTickSize().height();
151 trackHeight = 2 * (offsetFromCenter + tickLength);
152 }
153 float zoomFactor = style().effectiveZoom();
154 if (zoomFactor != 1.0)
155 trackHeight *= zoomFactor;
156
157 return RenderBox::computeLogicalHeight(trackHeight, logicalTop);
158 }
159#endif
160 if (isVertical)
161 logicalHeight = RenderSlider::defaultTrackLength;
162 return RenderBox::computeLogicalHeight(logicalHeight, logicalTop);
163}

165void RenderSliderContainer::layout()
166{
167 ASSERT(element()->shadowHost());
168 auto& input = downcast<HTMLInputElement>(*element()->shadowHost());
169 bool isVertical = hasVerticalAppearance(input);
170 mutableStyle().setFlexDirection(isVertical ? FlexDirection::Column : FlexDirection::Row);
171 TextDirection oldTextDirection = style().direction();
172 if (isVertical) {
173 // FIXME: Work around rounding issues in RTL vertical sliders. We want them to
174 // render identically to LTR vertical sliders. We can remove this work around when
175 // subpixel rendering is enabled on all ports.
176 mutableStyle().setDirection(TextDirection::LTR);
177 }
178
179 RenderBox* thumb = input.sliderThumbElement() ? input.sliderThumbElement()->renderBox() : nullptr;
180 RenderBox* track = input.sliderTrackElement() ? input.sliderTrackElement()->renderBox() : nullptr;
181 // Force a layout to reset the position of the thumb so the code below doesn't move the thumb to the wrong place.
182 // FIXME: Make a custom Render class for the track and move the thumb positioning code there.
183 if (track)
184 track->setChildNeedsLayout(MarkOnlyThis);
185
186 RenderFlexibleBox::layout();
187
188 mutableStyle().setDirection(oldTextDirection);
189 // These should always exist, unless someone mutates the shadow DOM (e.g., in the inspector).
190 if (!thumb || !track)
191 return;
192
193 double percentageOffset = sliderPosition(input).toDouble();
194 LayoutUnit availableExtent = isVertical ? track->contentHeight() : track->contentWidth();
195 availableExtent -= isVertical ? thumb->height() : thumb->width();
196 LayoutUnit offset { percentageOffset * availableExtent };
197 LayoutPoint thumbLocation = thumb->location();
198 if (isVertical)
199 thumbLocation.setY(thumbLocation.y() + track->contentHeight() - thumb->height() - offset);
200 else if (style().isLeftToRightDirection())
201 thumbLocation.setX(thumbLocation.x() + offset);
202 else
203 thumbLocation.setX(thumbLocation.x() - offset);
204 thumb->setLocation(thumbLocation);
205 thumb->repaint();
206}

254void SliderThumbElement::setPositionFromPoint(const LayoutPoint& absolutePoint)
255{
256 auto input = hostInput();
257 if (!input)
258 return;
259
260 auto* inputRenderer = input->renderBox();
261 if (!inputRenderer)
262 return;
263
264 auto* thumbRenderer = renderBox();
265 if (!thumbRenderer)
266 return;
267
268 ASSERT(input->sliderTrackElement());
269 auto* trackRenderer = input->sliderTrackElement()->renderBox();
270 if (!trackRenderer)
271 return;
272
273 // Do all the tracking math relative to the input's renderer's box.
274
275 bool isVertical = hasVerticalAppearance(*input);
276 bool isLeftToRightDirection = thumbRenderer->style().isLeftToRightDirection();
277
278 auto offset = inputRenderer->absoluteToLocal(absolutePoint, UseTransforms);
279 auto trackBoundingBox = trackRenderer->localToContainerQuad(FloatRect { { }, trackRenderer->size() }, inputRenderer).enclosingBoundingBox();
280
281 LayoutUnit trackLength;
282 LayoutUnit position;
283 if (isVertical) {
284 trackLength = trackRenderer->contentHeight() - thumbRenderer->height();
285 position = offset.y() - thumbRenderer->height() / 2 - trackBoundingBox.y() - thumbRenderer->marginBottom();
286 } else {
287 trackLength = trackRenderer->contentWidth() - thumbRenderer->width();
288 position = offset.x() - thumbRenderer->width() / 2 - trackBoundingBox.x();
289 position -= isLeftToRightDirection ? thumbRenderer->marginLeft() : thumbRenderer->marginRight();
290 }
291
292 position = std::max<LayoutUnit>(0, std::min(position, trackLength));
293 auto ratio = Decimal::fromDouble(static_cast<double>(position) / trackLength);
294 auto fraction = isVertical || !isLeftToRightDirection ? Decimal(1) - ratio : ratio;
295 auto stepRange = input->createStepRange(AnyStepHandling::Reject);
296 auto value = stepRange.clampValue(stepRange.valueFromProportion(fraction));
297
298#if ENABLE(DATALIST_ELEMENT)
299 const LayoutUnit snappingThreshold = renderer()->theme().sliderTickSnappingThreshold();
300 if (snappingThreshold > 0) {
301 if (std::optional<Decimal> closest = input->findClosestTickMarkValue(value)) {
302 double closestFraction = stepRange.proportionFromValue(*closest).toDouble();
303 double closestRatio = isVertical || !isLeftToRightDirection ? 1.0 - closestFraction : closestFraction;
304 LayoutUnit closestPosition { trackLength * closestRatio };
305 if ((closestPosition - position).abs() <= snappingThreshold)
306 value = *closest;
307 }
308 }
309#endif
310
311 String valueString = serializeForNumberType(value);
312 if (valueString == input->value())
313 return;
314
315 // FIXME: This is no longer being set from renderer. Consider updating the method name.
316 input->setValueFromRenderer(valueString);
317 if (renderer())
318 renderer()->setNeedsLayout();
319}

329void SliderThumbElement::stopDragging()
330{

341void SliderThumbElement::defaultEventHandler(Event& event)
342{
343 if (!is<MouseEvent>(event)) {
344 HTMLDivElement::defaultEventHandler(event);
345 return;
346 }
347
348 // FIXME: Should handle this readonly/disabled check in more general way.
349 // Missing this kind of check is likely to occur elsewhere if adding it in each shadow element.
350 auto input = hostInput();
351 if (!input || input->isDisabledFormControl()) {
352 HTMLDivElement::defaultEventHandler(event);
353 return;
354 }
355
356 MouseEvent& mouseEvent = downcast<MouseEvent>(event);
357 bool isLeftButton = mouseEvent.button() == LeftButton;
358 const AtomString& eventType = mouseEvent.type();
359
360 // We intentionally do not call event->setDefaultHandled() here because
361 // MediaControlTimelineElement::defaultEventHandler() wants to handle these
362 // mouse events.
363 if (eventType == eventNames().mousedownEvent && isLeftButton) {
364 startDragging();
365 return;
366 } else if (eventType == eventNames().mouseupEvent && isLeftButton) {
367 input->dispatchFormControlChangeEvent();
368 stopDragging();
369 return;
370 } else if (eventType == eventNames().mousemoveEvent) {
371 if (m_inDragMode)
372 setPositionFromPoint(mouseEvent.absoluteLocation());
373 return;
374 }
375
376 HTMLDivElement::defaultEventHandler(mouseEvent);
377}

427static Touch* findTouchWithIdentifier(TouchList& list, unsigned identifier)
428{
429 unsigned length = list.length();
430 for (unsigned i = 0; i < length; ++i) {
431 RefPtr<Touch> touch = list.item(i);
432 if (touch->identifier() == identifier)
433 return touch.get();
434 }
435 return nullptr;
436}

438void SliderThumbElement::handleTouchStart(TouchEvent& touchEvent)
439{
440 RefPtr<TouchList> targetTouches = touchEvent.targetTouches();
441 if (!targetTouches)
442 return;
443
444 if (targetTouches->length() != 1)
445 return;
446
447 RefPtr<Touch> touch = targetTouches->item(0);
448 if (!renderer())
449 return;
450 IntRect boundingBox = renderer()->absoluteBoundingBoxRect();
451 // Ignore the touch if it is not really inside the thumb.
452 if (!boundingBox.contains(touch->pageX(), touch->pageY()))
453 return;
454
455 setExclusiveTouchIdentifier(touch->identifier());
456
457 startDragging();
458 touchEvent.setDefaultHandled();
459}
460
461void SliderThumbElement::handleTouchMove(TouchEvent& touchEvent)
462{
463 unsigned identifier = exclusiveTouchIdentifier();
464 if (identifier == NoIdentifier)
465 return;
466
467 RefPtr<TouchList> targetTouches = touchEvent.targetTouches();
468 if (!targetTouches)
469 return;
470
471 RefPtr<Touch> touch = findTouchWithIdentifier(*targetTouches, identifier);
472 if (!touch)
473 return;
474
475 if (m_inDragMode)
476 setPositionFromPoint(IntPoint(touch->pageX(), touch->pageY()));
477 touchEvent.setDefaultHandled();
478}
479
480void SliderThumbElement::handleTouchEndAndCancel(TouchEvent& touchEvent)
481{
482 unsigned identifier = exclusiveTouchIdentifier();
483 if (identifier == NoIdentifier)
484 return;
485
486 RefPtr<TouchList> targetTouches = touchEvent.targetTouches();
487 if (!targetTouches)
488 return;
489 // If our exclusive touch still exists, it was not the touch
490 // that ended, so we should not stop dragging.
491 RefPtr<Touch> exclusiveTouch = findTouchWithIdentifier(*targetTouches, identifier);
492 if (exclusiveTouch)
493 return;
494
495 clearExclusiveTouchIdentifier();
496
497 auto input = hostInput();
498 if (input)
499 input->dispatchFormControlChangeEvent();
500 stopDragging();
501}

509void SliderThumbElement::handleTouchEvent(TouchEvent& touchEvent)
510{
511 auto input = hostInput();
512 ASSERT(input);
513 if (input->isReadOnly() || input->isDisabledFormControl()) {
514 clearExclusiveTouchIdentifier();
515 stopDragging();
516 touchEvent.setDefaultHandled();
517 HTMLDivElement::defaultEventHandler(touchEvent);
518 return;
519 }
520
521 const AtomString& eventType = touchEvent.type();
522 if (eventType == eventNames().touchstartEvent) {
523 handleTouchStart(touchEvent);
524 return;
525 }
526 if (eventType == eventNames().touchendEvent || eventType == eventNames().touchcancelEvent) {
527 handleTouchEndAndCancel(touchEvent);
528 return;
529 }
530 if (eventType == eventNames().touchmoveEvent) {
531 handleTouchMove(touchEvent);
532 return;
533 }
534
535 HTMLDivElement::defaultEventHandler(touchEvent);
536}

538bool SliderThumbElement::shouldAcceptTouchEvents()
539{
540 return renderer() && !isDisabledFormControl();
541}
542
543void SliderThumbElement::registerForTouchEvents()
544{
545 if (m_isRegisteredAsTouchEventListener)
546 return;
547
548 ASSERT(shouldAcceptTouchEvents());
549
550 document().addTouchEventHandler(*this);
551 m_isRegisteredAsTouchEventListener = true;
552}
553
554void SliderThumbElement::unregisterForTouchEvents()
555{
556 if (!m_isRegisteredAsTouchEventListener)
557 return;
558
559 clearExclusiveTouchIdentifier();
560 stopDragging();
561
562 document().removeTouchEventHandler(*this);
563 m_isRegisteredAsTouchEventListener = false;
564}

588std::optional<Style::ElementStyle> SliderThumbElement::resolveCustomStyle(const RenderStyle&, const RenderStyle* hostStyle)
589{
590 // This doesn't actually compute style. This is just a hack to pick shadow pseudo id when host style is known.
591
592 static MainThreadNeverDestroyed<const AtomString> sliderThumbShadowPseudoId("-webkit-slider-thumb", AtomString::ConstructFromLiteral);
593 static MainThreadNeverDestroyed<const AtomString> mediaSliderThumbShadowPseudoId("-webkit-media-slider-thumb", AtomString::ConstructFromLiteral);
594
595 if (!hostStyle)
596 return std::nullopt;
597
598 switch (hostStyle->appearance()) {
599 case MediaSliderPart:
600 case MediaSliderThumbPart:
601 case MediaVolumeSliderPart:
602 case MediaVolumeSliderThumbPart:
603 case MediaFullScreenVolumeSliderPart:
604 case MediaFullScreenVolumeSliderThumbPart:
605 m_shadowPseudoId = mediaSliderThumbShadowPseudoId;
606 break;
607 default:
608 m_shadowPseudoId = sliderThumbShadowPseudoId;
609 }
610
611 return std::nullopt;
612}
