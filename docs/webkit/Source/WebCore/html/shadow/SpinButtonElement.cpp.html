50inline SpinButtonElement::SpinButtonElement(Document& document, SpinButtonOwner& spinButtonOwner)
51 : HTMLDivElement(divTag, document)
52 , m_spinButtonOwner(&spinButtonOwner)

55 , m_pressStartingState(Indeterminate)
56 , m_repeatingTimer(*this, &SpinButtonElement::repeatingTimerFired)

61Ref<SpinButtonElement> SpinButtonElement::create(Document& document, SpinButtonOwner& spinButtonOwner)
62{
63 auto element = adoptRef(*new SpinButtonElement(document, spinButtonOwner));
64 static MainThreadNeverDestroyed<const AtomString> webkitInnerSpinButtonName("-webkit-inner-spin-button", AtomString::ConstructFromLiteral);
65 element->setPseudo(webkitInnerSpinButtonName);
66 return element;

74void SpinButtonElement::defaultEventHandler(Event& event)
75{
76 if (!is<MouseEvent>(event)) {
77 if (!event.defaultHandled())
78 HTMLDivElement::defaultEventHandler(event);
79 return;
80 }
81
82 RenderBox* box = renderBox();
83 if (!box) {
84 if (!event.defaultHandled())
85 HTMLDivElement::defaultEventHandler(event);
86 return;
87 }
88
89 if (!shouldRespondToMouseEvents()) {
90 if (!event.defaultHandled())
91 HTMLDivElement::defaultEventHandler(event);
92 return;
93 }
94
95 MouseEvent& mouseEvent = downcast<MouseEvent>(event);
96 IntPoint local = roundedIntPoint(box->absoluteToLocal(mouseEvent.absoluteLocation(), UseTransforms));
97 if (mouseEvent.type() == eventNames().mousedownEvent && mouseEvent.button() == LeftButton) {
98 if (box->borderBoxRect().contains(local)) {
99 // The following functions of HTMLInputElement may run JavaScript
100 // code which detaches this shadow node. We need to take a reference
101 // and check renderer() after such function calls.
102 Ref<SpinButtonElement> protectedThis(*this);
103 if (m_spinButtonOwner)
104 m_spinButtonOwner->focusAndSelectSpinButtonOwner();
105 if (renderer()) {
106 if (m_upDownState != Indeterminate) {
107 // A JavaScript event handler called in doStepAction() below
108 // might change the element state and we might need to
109 // cancel the repeating timer by the state change. If we
110 // started the timer after doStepAction(), we would have no
111 // chance to cancel the timer.
112 startRepeatingTimer();
113 doStepAction(m_upDownState == Up ? 1 : -1);
114 }
115 }
116 mouseEvent.setDefaultHandled();
117 }
118 } else if (mouseEvent.type() == eventNames().mouseupEvent && mouseEvent.button() == LeftButton)
119 stopRepeatingTimer();
120 else if (mouseEvent.type() == eventNames().mousemoveEvent) {
121 if (box->borderBoxRect().contains(local)) {
122 if (!m_capturing) {
123 if (RefPtr<Frame> frame = document().frame()) {
124 frame->eventHandler().setCapturingMouseEventsElement(this);
125 m_capturing = true;
126 if (Page* page = document().page())
127 page->chrome().registerPopupOpeningObserver(*this);
128 }
129 }
130 UpDownState oldUpDownState = m_upDownState;
131 switch (renderer()->theme().innerSpinButtonLayout(*renderer())) {
132 case RenderTheme::InnerSpinButtonLayout::Vertical:
133 m_upDownState = local.y() < box->height() / 2 ? Up : Down;
134 break;
135 case RenderTheme::InnerSpinButtonLayout::HorizontalUpLeft:
136 m_upDownState = local.x() < box->width() / 2 ? Up : Down;
137 break;
138 case RenderTheme::InnerSpinButtonLayout::HorizontalUpRight:
139 m_upDownState = local.x() > box->width() / 2 ? Up : Down;
140 break;
141 }
142 if (m_upDownState != oldUpDownState)
143 renderer()->repaint();
144 } else {
145 releaseCapture();
146 m_upDownState = Indeterminate;
147 }
148 }
149
150 if (!mouseEvent.defaultHandled())
151 HTMLDivElement::defaultEventHandler(mouseEvent);
152}

160void SpinButtonElement::forwardEvent(Event& event)
161{
162 if (!is<WheelEvent>(event))
163 return;
164
165 if (!m_spinButtonOwner)
166 return;
167
168 if (!m_spinButtonOwner->shouldSpinButtonRespondToWheelEvents())
169 return;
170
171 doStepAction(downcast<WheelEvent>(event).wheelDeltaY());
172 event.setDefaultHandled();
173}

178 return true;

191void SpinButtonElement::doStepAction(int amount)
192{
193 if (!m_spinButtonOwner)
194 return;
195
196 if (amount > 0)
197 m_spinButtonOwner->spinButtonStepUp();
198 else if (amount < 0)
199 m_spinButtonOwner->spinButtonStepDown();
200}
201
202void SpinButtonElement::releaseCapture()
203{
204 stopRepeatingTimer();
205 if (m_capturing) {
206 if (RefPtr<Frame> frame = document().frame()) {
207 frame->eventHandler().setCapturingMouseEventsElement(nullptr);
208 m_capturing = false;
209 if (Page* page = document().page())
210 page->chrome().unregisterPopupOpeningObserver(*this);
211 }
212 }
213}

220void SpinButtonElement::startRepeatingTimer()
221{
222 m_pressStartingState = m_upDownState;
223 ScrollbarTheme& theme = ScrollbarTheme::theme();
224 m_repeatingTimer.start(theme.initialAutoscrollTimerDelay(), theme.autoscrollTimerDelay());
225}
226
227void SpinButtonElement::stopRepeatingTimer()
228{
229 m_repeatingTimer.stop();
230}
231
232void SpinButtonElement::step(int amount)
233{
234 if (!shouldRespondToMouseEvents())
235 return;
236 // On Mac OS, NSStepper updates the value for the button under the mouse
237 // cursor regardless of the button pressed at the beginning. So the
238 // following check is not needed for Mac OS.
239#if !OS(MAC_OS_X)
240 if (m_upDownState != m_pressStartingState)
241 return;
242#endif
243 doStepAction(amount);
244}
245
246void SpinButtonElement::repeatingTimerFired()
247{
248 if (m_upDownState != Indeterminate)
249 step(m_upDownState == Up ? 1 : -1);
250}

259bool SpinButtonElement::shouldRespondToMouseEvents()
260{
