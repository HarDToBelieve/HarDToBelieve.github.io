78AudioTrack::AudioTrack(AudioTrackClient& client, AudioTrackPrivate& trackPrivate)
79 : MediaTrackBase(MediaTrackBase::AudioTrack, trackPrivate.id(), trackPrivate.label(), trackPrivate.language())
80 , m_client(&client)
81 , m_private(trackPrivate)
82 , m_enabled(trackPrivate.enabled())
83{
84 m_private->setClient(this);
85 updateKindFromPrivate();
86}
87
88AudioTrack::~AudioTrack()
89{
90 m_private->setClient(nullptr);
91}
92
93void AudioTrack::setPrivate(AudioTrackPrivate& trackPrivate)
94{
95 if (m_private.ptr() == &trackPrivate)
96 return;
97
98 m_private->setClient(nullptr);
99 m_private = trackPrivate;
100 m_private->setEnabled(m_enabled);
101 m_private->setClient(this);
102#if !RELEASE_LOG_DISABLED
103 m_private->setLogger(logger(), logIdentifier());
104#endif
105
106 updateKindFromPrivate();
107 setId(m_private->id());
108}

110bool AudioTrack::isValidKind(const AtomString& value) const
111{
112 return value == alternativeKeyword()
113 || value == commentaryKeyword()
114 || value == descriptionKeyword()
115 || value == mainKeyword()
116 || value == mainDescKeyword()
117 || value == translationKeyword();
118}

120void AudioTrack::setEnabled(bool enabled)
121{

168void AudioTrack::updateKindFromPrivate()
169{
170 switch (m_private->kind()) {
171 case AudioTrackPrivate::Alternative:
172 setKind(AudioTrack::alternativeKeyword());
173 break;
174 case AudioTrackPrivate::Description:
175 setKind(AudioTrack::descriptionKeyword());
176 break;
177 case AudioTrackPrivate::Main:
178 setKind(AudioTrack::mainKeyword());
179 break;
180 case AudioTrackPrivate::MainDesc:
181 setKind(AudioTrack::mainDescKeyword());
182 break;
183 case AudioTrackPrivate::Translation:
184 setKind(AudioTrack::translationKeyword());
185 break;
186 case AudioTrackPrivate::Commentary:
187 setKind(AudioTrack::commentaryKeyword());
188 break;
189 case AudioTrackPrivate::None:
190 setKind(emptyString());
191 break;
192 default:
193 ASSERT_NOT_REACHED();
194 break;
195 }
196}
