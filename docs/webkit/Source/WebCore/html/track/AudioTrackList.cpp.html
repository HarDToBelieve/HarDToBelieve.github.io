44void AudioTrackList::append(Ref<AudioTrack>&& track)
45{
46 // Insert tracks in the media file order.
47 size_t index = track->inbandTrackIndex();
48 size_t insertionIndex;
49 for (insertionIndex = 0; insertionIndex < m_inbandTracks.size(); ++insertionIndex) {
50 auto& otherTrack = downcast<AudioTrack>(*m_inbandTracks[insertionIndex]);
51 if (otherTrack.inbandTrackIndex() > index)
52 break;
53 }
54 m_inbandTracks.insert(insertionIndex, track.ptr());
55
56
57 ASSERT(!track->mediaElement() || track->mediaElement() == mediaElement());
58 track->setMediaElement(mediaElement());
59
60 scheduleAddTrackEvent(WTFMove(track));
61}
62
63AudioTrack* AudioTrackList::item(unsigned index) const
64{
65 if (index < m_inbandTracks.size())
66 return downcast<AudioTrack>(m_inbandTracks[index].get());
67 return nullptr;
68}
69
70AudioTrack* AudioTrackList::getTrackById(const AtomString& id) const
71{
72 for (auto& inbandTrack : m_inbandTracks) {
73 auto& track = downcast<AudioTrack>(*inbandTrack);
74 if (track.id() == id)
75 return &track;
76 }
77 return nullptr;
78}
