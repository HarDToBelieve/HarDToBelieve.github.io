62void InbandDataTextTrack::addDataCue(const MediaTime& start, const MediaTime& end, Ref<SerializedPlatformDataCue>&& platformValue, const String& type)
63{
64 if (findIncompleteCue(platformValue))
65 return;
66
67 auto cue = DataCue::create(document(), start, end, platformValue.copyRef(), type);
68 if (hasCue(cue, TextTrackCue::IgnoreDuration)) {
69 INFO_LOG(LOGIDENTIFIER, "ignoring already added cue: ", cue.get());
70 return;
71 }
72
73 if (end.isPositiveInfinite() && mediaElement()) {
74 cue->setEndTime(mediaElement()->durationMediaTime());
75 m_incompleteCueMap.append(&cue.get());
76 }
77
78 INFO_LOG(LOGIDENTIFIER, cue.get());
79
80 addCue(WTFMove(cue));
81}
82
83RefPtr<DataCue> InbandDataTextTrack::findIncompleteCue(const SerializedPlatformDataCue& cueToFind)
84{
85 auto index = m_incompleteCueMap.findMatching([&](const auto& cue) {
86 return cueToFind.isEqual(*cue->platformValue());
87 });
88
89 if (index == notFound)
90 return nullptr;
91
92 return m_incompleteCueMap[index];
93}
94
95void InbandDataTextTrack::updateDataCue(const MediaTime& start, const MediaTime& inEnd, SerializedPlatformDataCue& platformValue)
96{
97 auto cue = findIncompleteCue(platformValue);
98 if (!cue)
99 return;
100
101 cue->willChange();
102
103 MediaTime end = inEnd;
104 if (end.isPositiveInfinite() && mediaElement())
105 end = mediaElement()->durationMediaTime();
106 else
107 m_incompleteCueMap.removeFirst(cue);
108
109 INFO_LOG(LOGIDENTIFIER, "was start = ", cue->startMediaTime(), ", end = ", cue->endMediaTime(), ", will be start = ", start, ", end = ", end);
110
111 cue->setStartTime(start);
112 cue->setEndTime(end);
113
114 cue->didChange();
115}
116
117void InbandDataTextTrack::removeDataCue(const MediaTime&, const MediaTime&, SerializedPlatformDataCue& platformValue)
118{
119 if (auto cue = findIncompleteCue(platformValue)) {
120 INFO_LOG(LOGIDENTIFIER, "removing: ", *cue);
121 m_incompleteCueMap.removeFirst(cue);
122 InbandTextTrack::removeCue(*cue);
123 }
124}
125
126ExceptionOr<void> InbandDataTextTrack::removeCue(TextTrackCue& cue)
127{
128 ASSERT(cue.cueType() == TextTrackCue::Data);
129
130 if (auto platformValue = const_cast<SerializedPlatformDataCue*>(downcast<DataCue>(cue).platformValue()))
131 removeDataCue({ }, { }, *platformValue);
132
133 return InbandTextTrack::removeCue(cue);
134}
