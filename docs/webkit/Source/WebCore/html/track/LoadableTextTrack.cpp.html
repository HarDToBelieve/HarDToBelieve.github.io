49Ref<LoadableTextTrack> LoadableTextTrack::create(HTMLTrackElement& track, const String& kind, const String& label, const String& language)

56void LoadableTextTrack::scheduleLoad(const URL& url)
57{
58 if (url == m_url)
59 return;
60
61 // When src attribute is changed we need to flush all collected track data
62 removeAllCues();
63
64 // 4.8.10.12.3 Sourcing out-of-band text tracks (continued)
65
66 // 2. Let URL be the track URL of the track element.
67 m_url = url;
68
69 // 3. Asynchronously run the remaining steps, while continuing with whatever task
70 // was responsible for creating the text track or changing the text track mode.
71 if (!m_loadTimer.isActive())
72 m_loadTimer.startOneShot(0_s);
73}

80void LoadableTextTrack::loadTimerFired()
81{
82 if (m_loader)
83 m_loader->cancelLoad();
84
85 if (!m_trackElement)
86 return;
87
88 // 4.8.10.12.3 Sourcing out-of-band text tracks (continued)
89
90 // 4. Download: If URL is not the empty string, perform a potentially CORS-enabled fetch of URL, with the
91 // mode being the state of the media element's crossorigin content attribute, the origin being the
92 // origin of the media element's Document, and the default origin behaviour set to fail.
93 m_loader = makeUnique<TextTrackLoader>(static_cast<TextTrackLoaderClient&>(*this), m_trackElement->document());
94 if (!m_loader->load(m_url, *m_trackElement))
95 m_trackElement->didCompleteLoad(HTMLTrackElement::Failure);
96}

98void LoadableTextTrack::newCuesAvailable(TextTrackLoader& loader)
99{
100 ASSERT_UNUSED(loader, m_loader.get() == &loader);
101
102 if (!m_cues)
103 m_cues = TextTrackCueList::create();
104
105 for (auto& newCue : m_loader->getNewCues()) {
106 newCue->setTrack(this);
107 INFO_LOG(LOGIDENTIFIER, newCue.get());
108 m_cues->add(WTFMove(newCue));
109 }
110
111 if (client())
112 client()->textTrackAddCues(*this, *m_cues);
113}
114
115void LoadableTextTrack::cueLoadingCompleted(TextTrackLoader& loader, bool loadingFailed)
116{
117 ASSERT_UNUSED(loader, m_loader.get() == &loader);
118
119 if (!m_trackElement)
120 return;
121
122 INFO_LOG(LOGIDENTIFIER);
123
124 m_trackElement->didCompleteLoad(loadingFailed ? HTMLTrackElement::Failure : HTMLTrackElement::Success);
125}

127void LoadableTextTrack::newRegionsAvailable(TextTrackLoader& loader)
128{
129 ASSERT_UNUSED(loader, m_loader.get() == &loader);
130 for (auto& newRegion : m_loader->getNewRegions()) {
131 newRegion->setTrack(this);
132 regions()->add(WTFMove(newRegion));
133 }
134}

142AtomString LoadableTextTrack::id() const
143{
144 if (!m_trackElement)

146 return m_trackElement->attributeWithoutSynchronization(idAttr);
147}

149size_t LoadableTextTrack::trackElementIndex()
150{
151 ASSERT(m_trackElement);
152 ASSERT(m_trackElement->parentNode());
153
154 size_t index = 0;
155 for (RefPtr<Node> node = m_trackElement->parentNode()->firstChild(); node; node = node->nextSibling()) {
156 if (!node->hasTagName(trackTag) || !node->parentNode())
157 continue;
158 if (node == m_trackElement)
159 return index;
160 ++index;
161 }
162 ASSERT_NOT_REACHED();
163
164 return 0;
165}

167bool LoadableTextTrack::isDefault() const
168{
169 return m_trackElement && m_trackElement->hasAttributeWithoutSynchronization(defaultAttr);
