120static inline bool isLegalNode(Node& node)
121{
122 return node.hasTagName(HTMLNames::brTag)
123 || node.hasTagName(HTMLNames::divTag)
124 || node.hasTagName(HTMLNames::imgTag)
125 || node.hasTagName(HTMLNames::pTag)
126 || node.hasTagName(HTMLNames::rbTag)
127 || node.hasTagName(HTMLNames::rtTag)
128 || node.hasTagName(HTMLNames::rtcTag)
129 || node.hasTagName(HTMLNames::rubyTag)
130 || node.hasTagName(HTMLNames::spanTag)
131 || node.nodeType() == Node::TEXT_NODE;
132}
133
134static Exception invalidNodeException(Node& node)
135{
136 return Exception { InvalidNodeTypeError, makeString("Invalid node type: ", node.nodeName()) };
137}
138
139static ExceptionOr<void> checkForInvalidNodeTypes(Node& root)
140{
141 if (!isLegalNode(root))
142 return invalidNodeException(root);
143
144 for (auto* child = root.firstChild(); child; child = child->nextSibling()) {
145 if (!isLegalNode(*child))
146 return invalidNodeException(*child);
147
148 if (is<ContainerNode>(*child)) {
149 auto result = checkForInvalidNodeTypes(*child);
150 if (result.hasException())
151 return result.releaseException();
152 }
153 }
154
155 return { };
156}
157
158enum RequiredNodes {
159 Cue = 1 << 0,
160 CueBackground = 1 << 1,
161};
162
163static OptionSet<RequiredNodes> tagPseudoObjects(Node& node)
164{
165 if (!is<Element>(node))
166 return { };
167
168 OptionSet<RequiredNodes> nodeTypes = { };
169
170 auto& element = downcast<Element>(node);
171 if (element.hasAttributeWithoutSynchronization(cueAttributName())) {
172 element.setPseudo(TextTrackCue::cueShadowPseudoId());
173 nodeTypes = { RequiredNodes::Cue };
174 } else if (element.hasAttributeWithoutSynchronization(cueBackgroundAttributName())) {
175 element.setPseudo(TextTrackCue::cueBackdropShadowPseudoId());
176 nodeTypes = { RequiredNodes::CueBackground };
177 }
178
179 for (auto* child = element.firstChild(); child; child = child->nextSibling())
180 nodeTypes.add(tagPseudoObjects(*child));
181
182 return nodeTypes;
183}
184
185static void removePseudoAttributes(Node& node)
186{
187 if (!is<Element>(node))
188 return;
189
190 auto& element = downcast<Element>(node);
191 if (element.hasAttributeWithoutSynchronization(cueAttributName()) || element.hasAttributeWithoutSynchronization(cueBackgroundAttributName()))
192 element.removeAttribute(HTMLNames::pseudoAttr);
193
194 for (auto* child = element.firstChild(); child; child = child->nextSibling())
195 removePseudoAttributes(*child);
196}

198ExceptionOr<Ref<TextTrackCue>> TextTrackCue::create(Document& document, double start, double end, DocumentFragment& cueFragment)
199{
200 if (!cueFragment.firstChild())
201 return Exception { InvalidNodeTypeError, "Empty cue fragment" };
202
203 for (Node* node = cueFragment.firstChild(); node; node = node->nextSibling()) {
204 auto result = checkForInvalidNodeTypes(*node);
205 if (result.hasException())
206 return result.releaseException();
207 }
208
209 auto fragment = DocumentFragment::create(document);
210 for (Node* node = cueFragment.firstChild(); node; node = node->nextSibling()) {
211 auto result = fragment->ensurePreInsertionValidity(*node, nullptr);
212 if (result.hasException())
213 return result.releaseException();
214 }
215 cueFragment.cloneChildNodes(fragment);
216
217 OptionSet<RequiredNodes> nodeTypes = { };
218 for (Node* node = fragment->firstChild(); node; node = node->nextSibling())
219 nodeTypes.add(tagPseudoObjects(*node));
220
221 if (!nodeTypes.contains(RequiredNodes::Cue))
222 return Exception { InvalidStateError, makeString("Missing required attribute: ", cueAttributName().toString()) };
223 if (!nodeTypes.contains(RequiredNodes::CueBackground))
224 return Exception { InvalidStateError, makeString("Missing required attribute: ", cueBackgroundAttributName().toString()) };
225
226 auto textTrackCue = adoptRef(*new TextTrackCue(document, MediaTime::createWithDouble(start), MediaTime::createWithDouble(end), WTFMove(fragment)));
227 textTrackCue->suspendIfNeeded();
228 return textTrackCue;
229}

369bool TextTrackCue::isOrderedBefore(const TextTrackCue* other) const
370{
371 // ... cues must be sorted by their start time, earliest first;
372 if (startMediaTime() != other->startMediaTime())
373 return startMediaTime() < other->startMediaTime();
374
375 // then, any cues with the same start time must be sorted by their end time, latest first;
376 if (endMediaTime() != other->endMediaTime())
377 return endMediaTime() > other->endMediaTime();
378
379 // and finally, any cues with identical end times must be sorted in the order they were last added to
380 // their respective text track list of cues, oldest first (so e.g. for cues from a WebVTT file, that
381 // would initially be the order in which the cues were listed in the file)
382 return cueIndex() < other->cueIndex();
383}

390bool TextTrackCue::isEqual(const TextTrackCue& other, TextTrackCue::CueMatchRules match) const
391{
392 if (match != IgnoreDuration && endMediaTime() != other.endMediaTime())
393 return false;
394 return cueType() == other.cueType() && hasEquivalentStartTime(other) && cueContentsMatch(other);
395}

397bool TextTrackCue::hasEquivalentStartTime(const TextTrackCue& cue) const
398{
399 MediaTime startTimeVariance = MediaTime::zeroTime();
400 if (track())
401 startTimeVariance = track()->startTimeVariance();
402 else if (cue.track())
403 startTimeVariance = cue.track()->startTimeVariance();
404
405 return abs(abs(startMediaTime()) - abs(cue.startMediaTime())) <= startTimeVariance;
406}

408void TextTrackCue::toJSON(JSON::Object& value) const
409{
410 ASCIILiteral type = "Generic"_s;
411 switch (cueType()) {
412 case TextTrackCue::ConvertedToWebVTT:
413 type = "ConvertedToWebVTT"_s;
414 break;
415 case TextTrackCue::WebVTT:
416 type = "WebVTT"_s;
417 break;
418 case TextTrackCue::Data:
419 type = "Data"_s;
420 break;
421 case TextTrackCue::Generic:
422 type = "Generic"_s;
423 break;
424 }
425
426 value.setString("type"_s, type);
427 value.setDouble("startTime"_s, startTime());
428 value.setDouble("endTime"_s, endTime());
429}

499void TextTrackCue::rebuildDisplayTree()
500{
501 if (!m_cueNode)
502 return;
503
504 ScriptDisallowedScope::EventAllowedScope allowedScopeForReferenceTree(*m_cueNode);
505
506 if (!m_displayTree) {
507 static MainThreadNeverDestroyed<const AtomString> webkitGenericCueRootName("-webkit-generic-cue-root", AtomString::ConstructFromLiteral);
508 m_displayTree = TextTrackCueBox::create(ownerDocument(), *this);
509 m_displayTree->setPseudo(webkitGenericCueRootName);
510 }
511
512 m_displayTree->removeChildren();
513 auto clonedFragment = DocumentFragment::create(ownerDocument());
514 m_cueNode->cloneChildNodes(clonedFragment);
515 m_displayTree->appendChild(clonedFragment);
516
517 if (m_fontSize) {
518 if (auto page = ownerDocument().page()) {
519 auto style = HTMLStyleElement::create(HTMLNames::styleTag, ownerDocument(), false);
520 style->setTextContent(makeString(page->captionUserPreferencesStyleSheet(),
521 " ::", TextTrackCue::cueShadowPseudoId(), "{font-size:", m_fontSize, m_fontSizeIsImportant ? "px !important}" : "px}"));
522 m_displayTree->appendChild(style);
523 }
524 }
525
526 if (track()) {
527 if (const auto& styleSheets = track()->styleSheets()) {
528 for (const auto& cssString : *styleSheets) {
529 auto style = HTMLStyleElement::create(HTMLNames::styleTag, m_displayTree->document(), false);
530 style->setTextContent(cssString);
531 m_displayTree->appendChild(style);
532 }
533 }
534 }
535
536 m_displayTreeNeedsUpdate = false;
537}
