36#ifdef CHECK_SORTING
37#define ASSERT_SORTED(begin, end) ASSERT(std::is_sorted(begin, end, cueSortsBefore))

42namespace WebCore {

44static inline bool cueSortsBefore(const RefPtr<TextTrackCue>& a, const RefPtr<TextTrackCue>& b)
45{
46 if (a->startMediaTime() < b->startMediaTime())
47 return true;
48
49 return a->startMediaTime() == b->startMediaTime() && a->endMediaTime() > b->endMediaTime();
50}

52Ref<TextTrackCueList> TextTrackCueList::create()
53{
54 return adoptRef(*new TextTrackCueList);
55}

57unsigned TextTrackCueList::cueIndex(const TextTrackCue& cue) const
58{
59 ASSERT(m_vector.contains(&cue));
60 return m_vector.find(&cue);
61}
62
63TextTrackCue* TextTrackCueList::item(unsigned index) const
64{
65 if (index >= m_vector.size())
66 return nullptr;
67 return m_vector[index].get();
68}

70TextTrackCue* TextTrackCueList::getCueById(const String& id) const
71{
72 for (auto& cue : m_vector) {
73 if (cue->id() == id)
74 return cue.get();
75 }
76 return nullptr;
77}

79TextTrackCueList& TextTrackCueList::activeCues()
80{
81 if (!m_activeCues)
82 m_activeCues = create();
83
84 Vector<RefPtr<TextTrackCue>> activeCuesVector;
85 for (auto& cue : m_vector) {
86 if (cue->isActive())
87 activeCuesVector.append(cue);
88 }
89 ASSERT_SORTED(activeCuesVector.begin(), activeCuesVector.end());
90 m_activeCues->m_vector = WTFMove(activeCuesVector);
91
92 // FIXME: This list of active cues is not updated as cues are added, removed, become active, and become inactive.
93 // Instead it is only updated each time this function is called again. That is not consistent with other dynamic DOM lists.
94 return *m_activeCues;
95}

97void TextTrackCueList::add(Ref<TextTrackCue>&& cue)
98{
99 ASSERT(!m_vector.contains(cue.ptr()));
100 ASSERT(cue->startMediaTime() >= MediaTime::zeroTime());
101 ASSERT(cue->endMediaTime() >= MediaTime::zeroTime());
102
103 RefPtr<TextTrackCue> cueRefPtr { WTFMove(cue) };
104 unsigned insertionPosition = std::upper_bound(m_vector.begin(), m_vector.end(), cueRefPtr, cueSortsBefore) - m_vector.begin();
105 ASSERT_SORTED(m_vector.begin(), m_vector.end());
106 m_vector.insert(insertionPosition, WTFMove(cueRefPtr));
107 ASSERT_SORTED(m_vector.begin(), m_vector.end());
108}
109
110void TextTrackCueList::remove(TextTrackCue& cue)
111{
112 ASSERT_SORTED(m_vector.begin(), m_vector.end());
113 m_vector.remove(cueIndex(cue));
114 ASSERT_SORTED(m_vector.begin(), m_vector.end());
115}
116
117void TextTrackCueList::clear()
118{
119 m_vector.clear();
120 if (m_activeCues)
121 m_activeCues->m_vector.clear();
122}

124void TextTrackCueList::updateCueIndex(const TextTrackCue& cue)
125{
126 auto cuePosition = m_vector.begin() + cueIndex(cue);
127 auto afterCuePosition = cuePosition + 1;
128
129 ASSERT_SORTED(m_vector.begin(), cuePosition);
130 ASSERT_SORTED(afterCuePosition, m_vector.end());
131
132 auto reinsertionPosition = std::upper_bound(m_vector.begin(), cuePosition, *cuePosition, cueSortsBefore);
133 if (reinsertionPosition != cuePosition)
134 std::rotate(reinsertionPosition, cuePosition, afterCuePosition);
135 else {
136 reinsertionPosition = std::upper_bound(afterCuePosition, m_vector.end(), *cuePosition, cueSortsBefore);
137 if (reinsertionPosition != afterCuePosition)
138 std::rotate(cuePosition, afterCuePosition, reinsertionPosition);
139 }
140
141 ASSERT_SORTED(m_vector.begin(), m_vector.end());
142}
