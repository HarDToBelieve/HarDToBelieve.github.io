54void TrackListBase::clearElement()
55{
56 m_element = nullptr;
57 for (auto& track : m_inbandTracks) {
58 track->setMediaElement(nullptr);
59 track->clearClient();
60 }

73void TrackListBase::remove(TrackBase& track, bool scheduleEvent)
74{
75 size_t index = m_inbandTracks.find(&track);
76 if (index == notFound)
77 return;
78
79 if (track.mediaElement()) {
80 ASSERT(track.mediaElement() == m_element);
81 track.setMediaElement(nullptr);
82 }
83
84 Ref<TrackBase> trackRef = *m_inbandTracks[index];
85
86 m_inbandTracks.remove(index);
87
88 if (scheduleEvent)
89 scheduleRemoveTrackEvent(WTFMove(trackRef));
90}
91
92bool TrackListBase::contains(TrackBase& track) const
93{
94 return m_inbandTracks.find(&track) != notFound;
95}

97void TrackListBase::scheduleTrackEvent(const AtomString& eventName, Ref<TrackBase>&& track)
98{
99 queueTaskToDispatchEvent(*this, TaskSource::MediaElement, TrackEvent::create(eventName, Event::CanBubble::No, Event::IsCancelable::No, WTFMove(track)));
100}

102void TrackListBase::scheduleAddTrackEvent(Ref<TrackBase>&& track)
103{
104 // 4.8.10.5 Loading the media resource
105 // ...
106 // Fire a trusted event with the name addtrack, that does not bubble and is
107 // not cancelable, and that uses the TrackEvent interface, with the track
108 // attribute initialized to the new AudioTrack object, at this
109 // AudioTrackList object.
110 // ...
111 // Fire a trusted event with the name addtrack, that does not bubble and is
112 // not cancelable, and that uses the TrackEvent interface, with the track
113 // attribute initialized to the new VideoTrack object, at this
114 // VideoTrackList object.
115
116 // 4.8.10.12.3 Sourcing out-of-band text tracks
117 // 4.8.10.12.4 Text track API
118 // ... then queue a task to fire an event with the name addtrack, that does not
119 // bubble and is not cancelable, and that uses the TrackEvent interface, with
120 // the track attribute initialized to the text track's TextTrack object, at
121 // the media element's textTracks attribute's TextTrackList object.
122 scheduleTrackEvent(eventNames().addtrackEvent, WTFMove(track));
123}
124
125void TrackListBase::scheduleRemoveTrackEvent(Ref<TrackBase>&& track)
126{
127 // 4.8.10.6 Offsets into the media resource
128 // If at any time the user agent learns that an audio or video track has
129 // ended and all media data relating to that track corresponds to parts of
130 // the media timeline that are before the earliest possible position, the
131 // user agent may queue a task to remove the track from the audioTracks
132 // attribute's AudioTrackList object or the videoTracks attribute's
133 // VideoTrackList object as appropriate and then fire a trusted event
134 // with the name removetrack, that does not bubble and is not cancelable,
135 // and that uses the TrackEvent interface, with the track attribute
136 // initialized to the AudioTrack or VideoTrack object representing the
137 // track, at the media element's aforementioned AudioTrackList or
138 // VideoTrackList object.
139
140 // 4.8.10.12.3 Sourcing out-of-band text tracks
141 // When a track element's parent element changes and the old parent was a
142 // media element, then the user agent must remove the track element's
143 // corresponding text track from the media element's list of text tracks,
144 // and then queue a task to fire a trusted event with the name removetrack,
145 // that does not bubble and is not cancelable, and that uses the TrackEvent
146 // interface, with the track attribute initialized to the text track's
147 // TextTrack object, at the media element's textTracks attribute's
148 // TextTrackList object.
149 scheduleTrackEvent(eventNames().removetrackEvent, WTFMove(track));
150}

152void TrackListBase::scheduleChangeEvent()
153{
154 // 4.8.10.6 Offsets into the media resource
155 // Whenever an audio track in an AudioTrackList is enabled or disabled, the
156 // user agent must queue a task to fire a simple event named change at the
157 // AudioTrackList object.
158 // ...
159 // Whenever a track in a VideoTrackList that was previously not selected is
160 // selected, the user agent must queue a task to fire a simple event named
161 // change at the VideoTrackList object.
162 m_isChangeEventScheduled = true;
163 queueTaskKeepingObjectAlive(*this, TaskSource::MediaElement, [this] {
164 m_isChangeEventScheduled = false;
165 dispatchEvent(Event::create(eventNames().changeEvent, Event::CanBubble::No, Event::IsCancelable::No));
166 });
167}

169bool TrackListBase::isAnyTrackEnabled() const
170{
171 for (auto& track : m_inbandTracks) {
172 if (track->enabled())
173 return true;
174 }
175 return false;
176}
