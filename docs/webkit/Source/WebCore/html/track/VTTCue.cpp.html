158void VTTCueBox::applyCSSProperties(const IntSize& videoSize)
159{
160 auto textTrackCue = getCue();
161 ASSERT(!textTrackCue || is<VTTCue>(textTrackCue));
162 if (!is<VTTCue>(textTrackCue))
163 return;
164
165 auto cue = makeRef(downcast<VTTCue>(*textTrackCue));
166
167 // FIXME: Apply all the initial CSS positioning properties. http://wkb.ug/79916
168 if (!cue->regionId().isEmpty()) {
169 setInlineStyleProperty(CSSPropertyPosition, CSSValueRelative);
170 return;
171 }
172
173 // 3.5.1 On the (root) List of WebVTT Node Objects:
174
175 // the 'position' property must be set to 'absolute'
176 setInlineStyleProperty(CSSPropertyPosition, CSSValueAbsolute);
177
178 // the 'unicode-bidi' property must be set to 'plaintext'
179 setInlineStyleProperty(CSSPropertyUnicodeBidi, CSSValuePlaintext);
180
181 // the 'direction' property must be set to direction
182 setInlineStyleProperty(CSSPropertyDirection, cue->getCSSWritingDirection());
183
184 // the 'writing-mode' property must be set to writing-mode
185 setInlineStyleProperty(CSSPropertyWritingMode, cue->getCSSWritingMode(), false);
186
187 auto position = cue->getCSSPosition();
188
189 // the 'top' property must be set to top,
190 setInlineStyleProperty(CSSPropertyTop, position.second, CSSUnitType::CSS_PERCENTAGE);
191
192 // the 'left' property must be set to left
193 if (cue->vertical() == horizontalKeyword())
194 setInlineStyleProperty(CSSPropertyLeft, position.first, CSSUnitType::CSS_PERCENTAGE);
195 else if (cue->vertical() == verticalGrowingRightKeyword()) {
196 // FIXME: Why use calc to do the math instead of doing the subtraction here?
197 setInlineStyleProperty(CSSPropertyLeft, makeString("calc(-", videoSize.width(), "px - ", cue->getCSSSize(), "px)"));
198 }
199
200 double authorFontSize = std::min(videoSize.width(), videoSize.height()) * DEFAULTCAPTIONFONTSIZEPERCENTAGE / 100.0;
201 double multiplier = 1.0;
202 if (authorFontSize)
203 multiplier = m_fontSizeFromCaptionUserPrefs / authorFontSize;
204
205 double textPosition = cue->calculateComputedTextPosition();
206 double maxSize = 100.0;
207 CSSValueID alignment = cue->getCSSAlignment();
208 if (alignment == CSSValueEnd || alignment == CSSValueRight)
209 maxSize = textPosition;
210 else if (alignment == CSSValueStart || alignment == CSSValueLeft)
211 maxSize = 100.0 - textPosition;
212
213 double newCueSize = std::min(cue->getCSSSize() * multiplier, 100.0);
214 // the 'width' property must be set to width, and the 'height' property must be set to height
215 if (cue->vertical() == horizontalKeyword()) {
216 setInlineStyleProperty(CSSPropertyWidth, newCueSize, CSSUnitType::CSS_PERCENTAGE);
217 setInlineStyleProperty(CSSPropertyHeight, CSSValueAuto);
218 setInlineStyleProperty(CSSPropertyMinWidth, "min-content");
219 setInlineStyleProperty(CSSPropertyMaxWidth, maxSize, CSSUnitType::CSS_PERCENTAGE);
220 if ((alignment == CSSValueMiddle || alignment == CSSValueCenter) && multiplier != 1.0)
221 setInlineStyleProperty(CSSPropertyLeft, static_cast<double>(position.first - (newCueSize - cue->getCSSSize()) / 2), CSSUnitType::CSS_PERCENTAGE);
222 } else {
223 setInlineStyleProperty(CSSPropertyWidth, CSSValueAuto);
224 setInlineStyleProperty(CSSPropertyHeight, newCueSize, CSSUnitType::CSS_PERCENTAGE);
225 setInlineStyleProperty(CSSPropertyMinHeight, "min-content");
226 setInlineStyleProperty(CSSPropertyMaxHeight, maxSize, CSSUnitType::CSS_PERCENTAGE);
227 if ((alignment == CSSValueMiddle || alignment == CSSValueCenter) && multiplier != 1.0)
228 setInlineStyleProperty(CSSPropertyTop, static_cast<double>(position.second - (newCueSize - cue->getCSSSize()) / 2), CSSUnitType::CSS_PERCENTAGE);
229 }
230
231 // The 'text-align' property on the (root) List of WebVTT Node Objects must
232 // be set to the value in the second cell of the row of the table below
233 // whose first cell is the value of the corresponding cue's text track cue
234 // alignment:
235 setInlineStyleProperty(CSSPropertyTextAlign, cue->getCSSAlignment());
236
237 if (!cue->snapToLines()) {
238 // 10.13.1 Set up x and y:
239 // Note: x and y are set through the CSS left and top above.
240
241 // 10.13.2 Position the boxes in boxes such that the point x% along the
242 // width of the bounding box of the boxes in boxes is x% of the way
243 // across the width of the video's rendering area, and the point y%
244 // along the height of the bounding box of the boxes in boxes is y%
245 // of the way across the height of the video's rendering area, while
246 // maintaining the relative positions of the boxes in boxes to each
247 // other.
248 setInlineStyleProperty(CSSPropertyTransform, makeString("translate(", -position.first, "%, ", -position.second, "%)"));
249
250 setInlineStyleProperty(CSSPropertyWhiteSpace, CSSValuePre);
251 }
252
253 // Make sure shadow or stroke is not clipped.
254 setInlineStyleProperty(CSSPropertyOverflow, CSSValueVisible);
255 cue->element().setInlineStyleProperty(CSSPropertyOverflow, CSSValueVisible);
256}

345ExceptionOr<void> VTTCue::setVertical(const String& value)
346{
347 // http://www.whatwg.org/specs/web-apps/current-work/multipage/the-video-element.html#dom-texttrackcue-vertical
348 // On setting, the text track cue writing direction must be set to the value given
349 // in the first cell of the row in the table above whose second cell is a
350 // case-sensitive match for the new value, if any. If none of the values match, then
351 // the user agent must instead throw a SyntaxError exception.
352
353 WritingDirection direction = m_writingDirection;
354 if (value == horizontalKeyword())
355 direction = Horizontal;
356 else if (value == verticalGrowingLeftKeyword())
357 direction = VerticalGrowingLeft;
358 else if (value == verticalGrowingRightKeyword())
359 direction = VerticalGrowingRight;
360 else
361 return { };
362
363 if (direction == m_writingDirection)
364 return { };
365
366 willChange();
367 m_writingDirection = direction;
368 didChange();
369
370 return { };
371}

459ExceptionOr<void> VTTCue::setPosition(const LineAndPositionSetting& position)
460{
461 // http://dev.w3.org/html5/webvtt/#dfn-vttcue-position
462 // On setting, if the new value is negative or greater than 100, then an
463 // IndexSizeError exception must be thrown. Otherwise, the WebVTT cue
464 // position must be set to the new value; if the new value is the string
465 // "auto", then it must be interpreted as the special value auto.
466 double textPosition = 0;
467 if (WTF::holds_alternative<AutoKeyword>(position)) {
468 if (textPositionIsAuto())
469 return { };
470 textPosition = std::numeric_limits<double>::quiet_NaN();
471 } else {
472 textPosition = WTF::get<double>(position);
473 if (!(textPosition >= 0 && textPosition <= 100))
474 return Exception { IndexSizeError };
475
476 // Otherwise, set the text track cue line position to the new value.
477 if (m_textPosition == textPosition)
478 return { };
479 }
480
481 willChange();
482 m_textPosition = textPosition;
483 didChange();
484
485 return { };
486}

618static void copyWebVTTNodeToDOMTree(ContainerNode& webVTTNode, Node& parent)
619{
620 for (RefPtr<Node> node = webVTTNode.firstChild(); node; node = node->nextSibling()) {
621 RefPtr<Node> clonedNode;
622 if (is<WebVTTElement>(*node))
623 clonedNode = downcast<WebVTTElement>(*node).createEquivalentHTMLElement(parent.document());
624 else
625 clonedNode = node->cloneNode(false);
626 parent.appendChild(*clonedNode);
627 if (is<ContainerNode>(*node))
628 copyWebVTTNodeToDOMTree(downcast<ContainerNode>(*node), *clonedNode);
629 }
630}

715int VTTCue::calculateComputedLinePosition()
716{
717 // http://www.whatwg.org/specs/web-apps/current-work/multipage/the-video-element.html#text-track-cue-computed-line-position
718
719 // If the text track cue line position is numeric, then that is the text
720 // track cue computed line position.
721 if (!std::isnan(m_linePosition))
722 return m_linePosition;
723
724 // If the text track cue snap-to-lines flag of the text track cue is not
725 // set, the text track cue computed line position is the value 100;
726 if (!m_snapToLines)
727 return 100;
728
729 // Otherwise, it is the value returned by the following algorithm:
730
731 // If cue is not associated with a text track, return -1 and abort these
732 // steps.
733 if (!track())
734 return -1;
735
736 // Let n be the number of text tracks whose text track mode is showing or
737 // showing by default and that are in the media element's list of text
738 // tracks before track.
739 int n = track()->trackIndexRelativeToRenderedTracks();
740
741 // Increment n by one.
742 n++;
743
744 // Negate n.
745 n = -n;
746
747 return n;
748}

750static bool isCueParagraphSeparator(UChar character)
751{
752 // Within a cue, paragraph boundaries are only denoted by Type B characters,
753 // such as U+000A LINE FEED (LF), U+0085 NEXT LINE (NEL), and U+2029 PARAGRAPH SEPARATOR.
754 return u_charType(character) == U_PARAGRAPH_SEPARATOR;
755}

762void VTTCue::determineTextDirection()
763{
764 static NeverDestroyed<const String> rtTag(MAKE_STATIC_STRING_IMPL("rt"));
765 createWebVTTNodeTree();
766 if (!m_webVTTNodeTree)
767 return;
768
769 // Apply the Unicode Bidirectional Algorithm's Paragraph Level steps to the
770 // concatenation of the values of each WebVTT Text Object in nodes, in a
771 // pre-order, depth-first traversal, excluding WebVTT Ruby Text Objects and
772 // their descendants.
773 StringBuilder paragraphBuilder;
774 for (RefPtr<Node> node = m_webVTTNodeTree->firstChild(); node; node = NodeTraversal::next(*node, m_webVTTNodeTree.get())) {
775 // FIXME: The code does not match the comment above. This does not actually exclude Ruby Text Object descendant.
776 if (!node->isTextNode() || node->localName() == rtTag)
777 continue;
778
779 paragraphBuilder.append(node->nodeValue());
780 }
781
782 String paragraph = paragraphBuilder.toString();
783 if (!paragraph.length())
784 return;
785
786 for (size_t i = 0; i < paragraph.length(); ++i) {
787 UChar current = paragraph[i];
788 if (!current || isCueParagraphSeparator(current))
789 return;
790
791 if (UChar current = paragraph[i]) {
792 UCharDirection charDirection = u_charDirection(current);
793 if (charDirection == U_LEFT_TO_RIGHT) {
794 m_displayDirection = CSSValueLtr;
795 return;
796 }
797 if (charDirection == U_RIGHT_TO_LEFT || charDirection == U_RIGHT_TO_LEFT_ARABIC) {
798 m_displayDirection = CSSValueRtl;
799 return;
800 }
801 }
802 }

805double VTTCue::calculateComputedTextPosition() const
806{
807 // http://dev.w3.org/html5/webvtt/#dfn-cue-computed-position
808
809 // 1. If the position is numeric, then return the value of the position and
810 // abort these steps. (Otherwise, the position is the special value auto.)
811 if (!textPositionIsAuto())
812 return m_textPosition;
813
814 switch (m_cueAlignment) {
815 case Start:
816 case Left:
817 // 2. If the cue text alignment is start or left, return 0 and abort these
818 // steps.
819 return 0;
820 case End:
821 case Right:
822 // 3. If the cue text alignment is end or right, return 100 and abort these
823 // steps.
824 return 100;
825 case Center:
826 // 4. If the cue text alignment is center, return 50 and abort these steps.
827 return 50;
828 default:
829 ASSERT_NOT_REACHED();
830 return 0;
831 }

834void VTTCue::calculateDisplayParameters()
835{
836 // Steps 10.2, 10.3
837 determineTextDirection();
838
839 // 10.4 If the text track cue writing direction is horizontal, then let
840 // block-flow be 'tb'. Otherwise, if the text track cue writing direction is
841 // vertical growing left, then let block-flow be 'lr'. Otherwise, the text
842 // track cue writing direction is vertical growing right; let block-flow be
843 // 'rl'.
844
845 // The above step is done through the writing direction static map.
846
847 // 10.5 Determine the value of maximum size for cue as per the appropriate
848 // rules from the following list:
849 double computedTextPosition = calculateComputedTextPosition();
850 int maximumSize = computedTextPosition;
851 if ((m_writingDirection == Horizontal && m_cueAlignment == Start && m_displayDirection == CSSValueLtr)
852 || (m_writingDirection == Horizontal && m_cueAlignment == End && m_displayDirection == CSSValueRtl)
853 || (m_writingDirection == Horizontal && m_cueAlignment == Left)
854 || (m_writingDirection == VerticalGrowingLeft && (m_cueAlignment == Start || m_cueAlignment == Left))
855 || (m_writingDirection == VerticalGrowingRight && (m_cueAlignment == Start || m_cueAlignment == Left))) {
856 maximumSize = 100 - computedTextPosition;
857 } else if ((m_writingDirection == Horizontal && m_cueAlignment == End && m_displayDirection == CSSValueLtr)
858 || (m_writingDirection == Horizontal && m_cueAlignment == Start && m_displayDirection == CSSValueRtl)
859 || (m_writingDirection == Horizontal && m_cueAlignment == Right)
860 || (m_writingDirection == VerticalGrowingLeft && (m_cueAlignment == End || m_cueAlignment == Right))
861 || (m_writingDirection == VerticalGrowingRight && (m_cueAlignment == End || m_cueAlignment == Right))) {
862 maximumSize = computedTextPosition;
863 } else if (m_cueAlignment == Center) {
864 maximumSize = computedTextPosition <= 50 ? computedTextPosition : (100 - computedTextPosition);
865 maximumSize = maximumSize * 2;
866 } else
867 ASSERT_NOT_REACHED();
868
869 // 10.6 If the text track cue size is less than maximum size, then let size
870 // be text track cue size. Otherwise, let size be maximum size.
871 m_displaySize = std::min(m_cueSize, maximumSize);
872
873 // FIXME: Understand why step 10.7 is missing (just a copy/paste error?)
874 // Could be done within a spec implementation check - http://crbug.com/301580
875
876 // 10.8 Determine the value of x-position or y-position for cue as per the
877 // appropriate rules from the following list:
878 if (m_writingDirection == Horizontal) {
879 switch (m_cueAlignment) {
880 case Start:
881 if (m_displayDirection == CSSValueLtr)
882 m_displayPosition.first = computedTextPosition;
883 else
884 m_displayPosition.first = 100 - computedTextPosition - m_displaySize;
885 break;
886 case End:
887 if (m_displayDirection == CSSValueRtl)
888 m_displayPosition.first = 100 - computedTextPosition;
889 else
890 m_displayPosition.first = computedTextPosition - m_displaySize;
891 break;
892 case Left:
893 if (m_displayDirection == CSSValueLtr)
894 m_displayPosition.first = computedTextPosition;
895 else
896 m_displayPosition.first = 100 - computedTextPosition;
897 break;
898 case Right:
899 if (m_displayDirection == CSSValueLtr)
900 m_displayPosition.first = computedTextPosition - m_displaySize;
901 else
902 m_displayPosition.first = 100 - computedTextPosition - m_displaySize;
903 break;
904 case Center:
905 if (m_displayDirection == CSSValueLtr)
906 m_displayPosition.first = computedTextPosition - m_displaySize / 2;
907 else
908 m_displayPosition.first = 100 - computedTextPosition - m_displaySize / 2;
909 break;
910 case NumberOfAlignments:
911 ASSERT_NOT_REACHED();
912 }
913 }
914
915 // A text track cue has a text track cue computed line position whose value
916 // is defined in terms of the other aspects of the cue.
917 m_computedLinePosition = calculateComputedLinePosition();
918
919 // 10.9 Determine the value of whichever of x-position or y-position is not
920 // yet calculated for cue as per the appropriate rules from the following
921 // list:
922 if (m_snapToLines && m_displayPosition.second == undefinedPosition && m_writingDirection == Horizontal)
923 m_displayPosition.second = 0;
924
925 if (!m_snapToLines && m_displayPosition.second == undefinedPosition && m_writingDirection == Horizontal)
926 m_displayPosition.second = m_computedLinePosition;
927
928 if (m_snapToLines && m_displayPosition.first == undefinedPosition
929 && (m_writingDirection == VerticalGrowingLeft || m_writingDirection == VerticalGrowingRight))
930 m_displayPosition.first = 0;
931
932 if (!m_snapToLines && (m_writingDirection == VerticalGrowingLeft || m_writingDirection == VerticalGrowingRight))
933 m_displayPosition.first = m_computedLinePosition;
934}

1134void VTTCue::setCueSettings(const String& inputString)
1135{
1136 if (inputString.isEmpty())
1137 return;
1138
1139 VTTScanner input(inputString);
1140
1141 while (!input.isAtEnd()) {
1142
1143 // The WebVTT cue settings part of a WebVTT cue consists of zero or more of the following components, in any order,
1144 // separated from each other by one or more U+0020 SPACE characters or U+0009 CHARACTER TABULATION (tab) characters.
1145 input.skipWhile<WebVTTParser::isValidSettingDelimiter>();
1146 if (input.isAtEnd())
1147 break;
1148
1149 // When the user agent is to parse the WebVTT settings given by a string input for a text track cue cue,
1150 // the user agent must run the following steps:
1151 // 1. Let settings be the result of splitting input on spaces.
1152 // 2. For each token setting in the list settings, run the following substeps:
1153 // 1. If setting does not contain a U+003A COLON character (:), or if the first U+003A COLON character (:)
1154 // in setting is either the first or last character of setting, then jump to the step labeled next setting.
1155 // 2. Let name be the leading substring of setting up to and excluding the first U+003A COLON character (:) in that string.
1156 CueSetting name = settingName(input);
1157
1158 // 3. Let value be the trailing substring of setting starting from the character immediately after the first U+003A COLON character (:) in that string.
1159 VTTScanner::Run valueRun = input.collectUntil<WebVTTParser::isValidSettingDelimiter>();
1160
1161 // 4. Run the appropriate substeps that apply for the value of name, as follows:
1162 switch (name) {
1163 case Vertical: {
1164 // If name is a case-sensitive match for "vertical"
1165 // 1. If value is a case-sensitive match for the string "rl", then let cue's text track cue writing direction
1166 // be vertical growing left.
1167 if (input.scanRun(valueRun, verticalGrowingLeftKeyword()))
1168 m_writingDirection = VerticalGrowingLeft;
1169
1170 // 2. Otherwise, if value is a case-sensitive match for the string "lr", then let cue's text track cue writing
1171 // direction be vertical growing right.
1172 else if (input.scanRun(valueRun, verticalGrowingRightKeyword()))
1173 m_writingDirection = VerticalGrowingRight;
1174
1175 else
1176 LOG(Media, "VTTCue::setCueSettings, invalid Vertical");
1177 break;
1178 }
1179 case Line: {
1180 bool isValid = false;
1181 do {
1182 // 1-2 - Collect chars that are either '-', '%', or a digit.
1183 // 1. If value contains any characters other than U+002D HYPHEN-MINUS characters (-), U+0025 PERCENT SIGN
1184 // characters (%), and characters in the range U+0030 DIGIT ZERO (0) to U+0039 DIGIT NINE (9), then jump
1185 // to the step labeled next setting.
1186 float linePosition;
1187 bool isNegative;
1188 if (!input.scanFloat(linePosition, &isNegative))
1189 break;
1190
1191 bool isPercentage = input.scan('%');
1192 if (!input.isAt(valueRun.end())) {
1193 if (!input.scan(','))
1194 break;
1195 // FIXME: implement handling of line setting alignment.
1196 if (!input.scan(startKeyword().characters8(), startKeyword().length())
1197 && !input.scan(centerKeyword().characters8(), centerKeyword().length())
1198 && !input.scan(endKeyword().characters8(), endKeyword().length())) {
1199 LOG(Media, "VTTCue::setCueSettings, invalid line setting alignment");
1200 break;
1201 }
1202 }
1203
1204 // 2. If value does not contain at least one character in the range U+0030 DIGIT ZERO (0) to U+0039 DIGIT
1205 // NINE (9), then jump to the step labeled next setting.
1206 // 3. If any character in value other than the first character is a U+002D HYPHEN-MINUS character (-), then
1207 // jump to the step labeled next setting.
1208 // 4. If any character in value other than the last character is a U+0025 PERCENT SIGN character (%), then
1209 // jump to the step labeled next setting.
1210 // 5. If the first character in value is a U+002D HYPHEN-MINUS character (-) and the last character in value is a
1211 // U+0025 PERCENT SIGN character (%), then jump to the step labeled next setting.
1212 if (isPercentage && isNegative)
1213 break;
1214
1215 // 6. Ignoring the trailing percent sign, if any, interpret value as a (potentially signed) integer, and
1216 // let number be that number.
1217 // 7. If the last character in value is a U+0025 PERCENT SIGN character (%), but number is not in the range
1218 // 0 number 100, then jump to the step labeled next setting.
1219 // 8. Let cue's text track cue line position be number.
1220 // 9. If the last character in value is a U+0025 PERCENT SIGN character (%), then let cue's text track cue
1221 // snap-to-lines flag be false. Otherwise, let it be true.
1222 if (isPercentage) {
1223 if (linePosition < 0 || linePosition > 100)
1224 break;
1225
1226 // 10 - If '%' then set snap-to-lines flag to false.
1227 m_snapToLines = false;
1228 } else {
1229 if (linePosition - static_cast<int>(linePosition))
1230 break;
1231
1232 m_snapToLines = true;
1233 }
1234
1235 m_linePosition = linePosition;
1236 isValid = true;
1237 } while (0);
1238
1239 if (!isValid)
1240 LOG(Media, "VTTCue::setCueSettings, invalid Line");
1241
1242 break;
1243 }
1244 case Position: {
1245 float position;
1246 if (WebVTTParser::parseFloatPercentageValue(input, position) && input.isAt(valueRun.end()))
1247 m_textPosition = position;
1248 else
1249 LOG(Media, "VTTCue::setCueSettings, invalid Position");
1250 break;
1251 }
1252 case Size: {
1253 float cueSize;
1254 if (WebVTTParser::parseFloatPercentageValue(input, cueSize) && input.isAt(valueRun.end()))
1255 m_cueSize = cueSize;
1256 else
1257 LOG(Media, "VTTCue::setCueSettings, invalid Size");
1258 break;
1259 }
1260 case Align: {
1261 // 1. If value is a case-sensitive match for the string "start", then let cue's text track cue alignment be start alignment.
1262 if (input.scanRun(valueRun, startKeyword()))
1263 m_cueAlignment = Start;
1264
1265 // 2. If value is a case-sensitive match for the string "center", then let cue's text track cue alignment be center alignment.
1266 else if (input.scanRun(valueRun, centerKeyword()))
1267 m_cueAlignment = Center;
1268
1269 // 3. If value is a case-sensitive match for the string "end", then let cue's text track cue alignment be end alignment.
1270 else if (input.scanRun(valueRun, endKeyword()))
1271 m_cueAlignment = End;
1272
1273 // 4. If value is a case-sensitive match for the string "left", then let cue's text track cue alignment be left alignment.
1274 else if (input.scanRun(valueRun, leftKeyword()))
1275 m_cueAlignment = Left;
1276
1277 // 5. If value is a case-sensitive match for the string "right", then let cue's text track cue alignment be right alignment.
1278 else if (input.scanRun(valueRun, rightKeyword()))
1279 m_cueAlignment = Right;
1280
1281 else
1282 LOG(Media, "VTTCue::setCueSettings, invalid Align");
1283
1284 break;
1285 }
1286 case Region: {
1287 m_parsedRegionId = input.extractString(valueRun);
1288 break;
1289 }
1290 case None:
1291 break;
1292 }
1293
1294 // Make sure the entire run is consumed.
1295 input.skipRun(valueRun);
1296 }
1297
