32#pragma once
33
34#if ENABLE(VIDEO)
35
36#include "ContextDestructionObserver.h"
37#include "FloatPoint.h"
38#include "TextTrack.h"
39
40namespace WebCore {
41
42class HTMLDivElement;
43class VTTCueBox;
44class VTTScanner;
45
46class VTTRegion final : public RefCounted<VTTRegion>, public ContextDestructionObserver {
47public:
48 static Ref<VTTRegion> create(ScriptExecutionContext& context)
49 {
50 return adoptRef(*new VTTRegion(context));
51 }
52
53 virtual ~VTTRegion();
54
55 TextTrack* track() const { return m_track; }
56 void setTrack(TextTrack*);
57
58 const String& id() const { return m_id; }
59 void setId(const String&);
60
61 double width() const { return m_width; }
62 ExceptionOr<void> setWidth(double);
63
64 int lines() const { return m_lines; }
65 ExceptionOr<void> setLines(int);
66
67 double regionAnchorX() const { return m_regionAnchor.x(); }
68 ExceptionOr<void> setRegionAnchorX(double);
69
70 double regionAnchorY() const { return m_regionAnchor.y(); }
71 ExceptionOr<void> setRegionAnchorY(double);
72
73 double viewportAnchorX() const { return m_viewportAnchor.x(); }
74 ExceptionOr<void> setViewportAnchorX(double);
75
76 double viewportAnchorY() const { return m_viewportAnchor.y(); }
77 ExceptionOr<void> setViewportAnchorY(double);
78
79 const AtomString& scroll() const;
80 ExceptionOr<void> setScroll(const AtomString&);
81
82 void updateParametersFromRegion(const VTTRegion&);
83
84 const String& regionSettings() const { return m_settings; }
85 void setRegionSettings(const String&);
86
87 bool isScrollingRegion() { return m_scroll; }
88
89 HTMLDivElement& getDisplayTree();
90
91 void appendTextTrackCueBox(Ref<TextTrackCueBox>&&);
92 void displayLastTextTrackCueBox();
93 void willRemoveTextTrackCueBox(VTTCueBox*);
94
95 void cueStyleChanged() { m_recalculateStyles = true; }
96
97private:
98 VTTRegion(ScriptExecutionContext&);
99
100 void prepareRegionDisplayTree();
101
102 // The timer is needed to continue processing when cue scrolling ended.
103 void startTimer();
104 void stopTimer();
105 void scrollTimerFired();
106
107 enum RegionSetting {
108 None,
109 Id,
110 Width,
111 Lines,
112 RegionAnchor,
113 ViewportAnchor,
114 Scroll
115 };
116
117 RegionSetting scanSettingName(VTTScanner&);
118
119 void parseSettingValue(RegionSetting, VTTScanner&);
120
121 static const AtomString& textTrackCueContainerShadowPseudoId();
122 static const AtomString& textTrackCueContainerScrollingClass();
123 static const AtomString& textTrackRegionShadowPseudoId();
124
125 String m_id;
126 String m_settings;
127
128 double m_width { 100 };
129 unsigned m_lines { 3 };
130
131 FloatPoint m_regionAnchor { 0, 100 };
132 FloatPoint m_viewportAnchor { 0, 100 };
133
134 bool m_scroll { false };
135
136 // The cue container is the container that is scrolled up to obtain the
137 // effect of scrolling cues when this is enabled for the regions.
138 RefPtr<HTMLDivElement> m_cueContainer;
139 RefPtr<HTMLDivElement> m_regionDisplayTree;
140
141 // The member variable track can be a raw pointer as it will never
142 // reference a destroyed TextTrack, as this member variable
143 // is cleared in the TextTrack destructor and it is generally
144 // set/reset within the addRegion and removeRegion methods.
145 TextTrack* m_track { nullptr };
146
147 // Keep track of the current numeric value of the css "top" property.
148 double m_currentTop { 0 };
149
150 // The timer is used to display the next cue line after the current one has
151 // been displayed. It's main use is for scrolling regions and it triggers as
152 // soon as the animation for rolling out one line has finished, but
153 // currently it is used also for non-scrolling regions to use a single
154 // code path.
155 Timer m_scrollTimer;
156
157 bool m_recalculateStyles { true };
158};
159
160} // namespace WebCore
161
162#endif
