34VTTRegion* VTTRegionList::item(unsigned index) const
35{
36 if (index >= m_vector.size())
37 return nullptr;
38 return const_cast<VTTRegion*>(m_vector[index].ptr());
39}
40
41VTTRegion* VTTRegionList::getRegionById(const String& id) const
42{
43 // FIXME: Why is this special case needed?
44 if (id.isEmpty())
45 return nullptr;
46 for (auto& region : m_vector) {
47 if (region->id() == id)
48 return const_cast<VTTRegion*>(region.ptr());
49 }
50 return nullptr;
51}
52
53void VTTRegionList::add(Ref<VTTRegion>&& region)
54{
55 m_vector.append(WTFMove(region));
56}
57
58void VTTRegionList::remove(VTTRegion& region)
59{
60 for (unsigned i = 0, size = m_vector.size(); i < size; ++i) {
61 if (m_vector[i].ptr() == &region) {
62 m_vector.remove(i);
63 return;
64 }
65 }
66 ASSERT_NOT_REACHED();
67}
