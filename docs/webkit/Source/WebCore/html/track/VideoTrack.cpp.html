83VideoTrack::VideoTrack(VideoTrackClient& client, VideoTrackPrivate& trackPrivate)
84 : MediaTrackBase(MediaTrackBase::VideoTrack, trackPrivate.id(), trackPrivate.label(), trackPrivate.language())
85 , m_client(&client)
86 , m_private(trackPrivate)
87 , m_selected(trackPrivate.selected())
88{
89 m_private->setClient(this);
90 updateKindFromPrivate();
91}
92
93VideoTrack::~VideoTrack()
94{
95 m_private->setClient(nullptr);
96}
97
98void VideoTrack::setPrivate(VideoTrackPrivate& trackPrivate)
99{
100 if (m_private.ptr() == &trackPrivate)
101 return;
102
103 m_private->setClient(nullptr);
104 m_private = trackPrivate;
105 m_private->setClient(this);
106#if !RELEASE_LOG_DISABLED
107 m_private->setLogger(logger(), logIdentifier());
108#endif
109
110 m_private->setSelected(m_selected);
111 updateKindFromPrivate();
112 setId(m_private->id());
113}

115bool VideoTrack::isValidKind(const AtomString& value) const
116{
117 return value == alternativeKeyword()
118 || value == commentaryKeyword()
119 || value == captionsKeyword()
120 || value == mainKeyword()
121 || value == signKeyword()
122 || value == subtitlesKeyword();
123}

125void VideoTrack::setSelected(const bool selected)
126{
127 if (m_selected == selected)
128 return;
129
130 m_selected = selected;
131 m_private->setSelected(selected);
132
133 if (m_client)
134 m_client->videoTrackSelectedChanged(*this);
135}

172void VideoTrack::setKind(const AtomString& kind)
173{
174 // 10.1 kind, on setting:
175 // 1. If the value being assigned to this attribute does not match one of the video track kinds,
176 // then abort these steps.
177 if (!isValidKind(kind))
178 return;
179
180 // 2. Update this attribute to the new value.
181 setKindInternal(kind);
182
183 // 3. If the sourceBuffer attribute on this track is not null, then queue a task to fire a simple
184 // event named change at sourceBuffer.videoTracks.
185 if (m_sourceBuffer)
186 m_sourceBuffer->videoTracks().scheduleChangeEvent();
187
188 // 4. Queue a task to fire a simple event named change at the VideoTrackList object referenced by
189 // the videoTracks attribute on the HTMLMediaElement.
190 mediaElement()->ensureVideoTracks().scheduleChangeEvent();
191}

193void VideoTrack::setLanguage(const AtomString& language)
194{
195 // 10.1 language, on setting:
196 // 1. If the value being assigned to this attribute is not an empty string or a BCP 47 language
197 // tag[BCP47], then abort these steps.
198 // BCP 47 validation is done in TrackBase::setLanguage() which is
199 // shared between all tracks that support setting language.
200
201 // 2. Update this attribute to the new value.
202 MediaTrackBase::setLanguage(language);
203
204 // 3. If the sourceBuffer attribute on this track is not null, then queue a task to fire a simple
205 // event named change at sourceBuffer.videoTracks.
206 if (m_sourceBuffer)
207 m_sourceBuffer->videoTracks().scheduleChangeEvent();
208
209 // 4. Queue a task to fire a simple event named change at the VideoTrackList object referenced by
210 // the videoTracks attribute on the HTMLMediaElement.
211 if (mediaElement())
212 mediaElement()->ensureVideoTracks().scheduleChangeEvent();
213}

217void VideoTrack::updateKindFromPrivate()
218{
219 switch (m_private->kind()) {
220 case VideoTrackPrivate::Alternative:
221 setKindInternal(VideoTrack::alternativeKeyword());
222 return;
223 case VideoTrackPrivate::Captions:
224 setKindInternal(VideoTrack::captionsKeyword());
225 return;
226 case VideoTrackPrivate::Main:
227 setKindInternal(VideoTrack::mainKeyword());
228 return;
229 case VideoTrackPrivate::Sign:
230 setKindInternal(VideoTrack::signKeyword());
231 return;
232 case VideoTrackPrivate::Subtitles:
233 setKindInternal(VideoTrack::subtitlesKeyword());
234 return;
235 case VideoTrackPrivate::Commentary:
236 setKindInternal(VideoTrack::commentaryKeyword());
237 return;
238 case VideoTrackPrivate::None:
239 setKindInternal(emptyString());
240 return;
241 }
242 ASSERT_NOT_REACHED();
243}
