28#if ENABLE(VIDEO)
29
30#include "TrackListBase.h"

32namespace WebCore {
33
34class VideoTrack;
35
36class VideoTrackList final : public TrackListBase {
37public:
38 static Ref<VideoTrackList> create(WeakPtr<HTMLMediaElement> owner, ScriptExecutionContext* context)
39 {
40 auto list = adoptRef(*new VideoTrackList(owner, context));
41 list->suspendIfNeeded();
42 return list;
43 }
44 virtual ~VideoTrackList();
45
46 VideoTrack* getTrackById(const AtomString&) const;
47 int selectedIndex() const;
48
49 VideoTrack* item(unsigned) const;
50 VideoTrack* lastItem() const { return item(length() - 1); }
51 void append(Ref<VideoTrack>&&);
52
53 // EventTarget
54 EventTargetInterface eventTargetInterface() const override;
55
56private:
57 VideoTrackList(WeakPtr<HTMLMediaElement>, ScriptExecutionContext*);
58
59 const char* activeDOMObjectName() const final;
60};
61static_assert(sizeof(VideoTrackList) == sizeof(TrackListBase), "");
62
63} // namespace WebCore
