36namespace WebCore {
37
38class WebVTTTokenTypes {
39public:
40 enum Type {
41 Uninitialized,
42 Character,
43 StartTag,
44 EndTag,
45 TimestampTag,
46 };
47};
48
49class WebVTTToken {
50public:
51 typedef WebVTTTokenTypes Type;
52
53 WebVTTToken()
54 : m_type(Type::Uninitialized) { }
55
56 static WebVTTToken StringToken(const String& characterData)
57 {
58 return WebVTTToken(Type::Character, characterData);
59 }
60
61 static WebVTTToken StartTag(const String& tagName, const AtomString& classes = emptyAtom(), const AtomString& annotation = emptyAtom())
62 {
63 WebVTTToken token(Type::StartTag, tagName);
64 token.m_classes = classes;
65 token.m_annotation = annotation;
66 return token;
67 }
68
69 static WebVTTToken EndTag(const String& tagName)
70 {
71 return WebVTTToken(Type::EndTag, tagName);
72 }
73
74 static WebVTTToken TimestampTag(const String& timestampData)
75 {
76 return WebVTTToken(Type::TimestampTag, timestampData);
77 }
78
79 Type::Type type() const { return m_type; }
80 const String& name() const { return m_data; }
81 const String& characters() const { return m_data; }
82 const AtomString& classes() const { return m_classes; }
83 const AtomString& annotation() const { return m_annotation; }
84
85private:
86 WebVTTToken(Type::Type type, const String& data)
87 : m_type(type)
88 , m_data(data) { }
89
90 Type::Type m_type;
91 String m_data;
92 AtomString m_annotation;
93 AtomString m_classes;
94};
95
96} // namespace WebCore

98#endif // ENABLE(VIDEO)
