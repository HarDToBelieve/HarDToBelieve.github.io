104#if ENABLE(WEBGPU)
105static HTMLCanvasElement* canvasIfContextMatchesDevice(CanvasRenderingContext& context, WebGPUDevice& device)
106{
107 if (is<GPUCanvasContext>(context)) {
108 auto& contextGPU = downcast<GPUCanvasContext>(context);
109 if (auto* webGPUSwapChain = contextGPU.swapChain()) {
110 if (auto* gpuSwapChain = webGPUSwapChain->swapChain()) {
111 if (gpuSwapChain == device.device().swapChain()) {
112 if (is<HTMLCanvasElement>(contextGPU.canvasBase()))
113 return &downcast<HTMLCanvasElement>(contextGPU.canvasBase());
114 }
115 }
116 }
117 }
118 return nullptr;
119}
120#endif

159HTMLCanvasElement* InspectorCanvas::canvasElement() const
160{
161 return WTF::switchOn(m_context,
162 [] (std::reference_wrapper<CanvasRenderingContext> contextWrapper) -> HTMLCanvasElement* {
163 auto& context = contextWrapper.get();
164 if (is<HTMLCanvasElement>(context.canvasBase()))
165 return &downcast<HTMLCanvasElement>(context.canvasBase());
166 return nullptr;
167 },
168#if ENABLE(WEBGPU)
169 [&] (std::reference_wrapper<WebGPUDevice> deviceWrapper) -> HTMLCanvasElement* {
170 auto& device = deviceWrapper.get();
171 {
172 Locker locker { CanvasRenderingContext::instancesLock() };
173 for (auto* canvasRenderingContext : CanvasRenderingContext::instances()) {
174 if (auto* canvasElement = canvasIfContextMatchesDevice(*canvasRenderingContext, device))
175 return canvasElement;
176 }
177 }
178 return nullptr;
179 },
180#endif
181 [] (Monostate) {
182 ASSERT_NOT_REACHED();
183 return nullptr;
184 }
185 );
186 return nullptr;
187}

225JSC::JSValue InspectorCanvas::resolveContext(JSC::JSGlobalObject* exec) const
226{
227 JSC::JSLockHolder lock(exec);
228
229 auto* globalObject = deprecatedGlobalObjectForPrototype(exec);
230
231 return WTF::switchOn(m_context,
232 [&] (std::reference_wrapper<CanvasRenderingContext> contextWrapper) {
233 auto& context = contextWrapper.get();
234 if (is<CanvasRenderingContext2D>(context))
235 return toJS(exec, globalObject, downcast<CanvasRenderingContext2D>(context));
236 if (is<ImageBitmapRenderingContext>(context))
237 return toJS(exec, globalObject, downcast<ImageBitmapRenderingContext>(context));
238#if ENABLE(WEBGL)
239 if (is<WebGLRenderingContext>(context))
240 return toJS(exec, globalObject, downcast<WebGLRenderingContext>(context));
241#endif
242#if ENABLE(WEBGL2)
243 if (is<WebGL2RenderingContext>(context))
244 return toJS(exec, globalObject, downcast<WebGL2RenderingContext>(context));
245#endif
246 return JSC::JSValue();
247 },
248#if ENABLE(WEBGPU)
249 [&] (std::reference_wrapper<WebGPUDevice> deviceWrapper) {
250 return toJS(exec, globalObject, deviceWrapper.get());
251 },
252#endif
253 [] (Monostate) {
254 ASSERT_NOT_REACHED();
255 return JSC::JSValue();
256 }
257 );
258}

260HashSet<Element*> InspectorCanvas::clientNodes() const
261{
262 return WTF::switchOn(m_context,
263 [] (std::reference_wrapper<CanvasRenderingContext> contextWrapper) {
264 auto& context = contextWrapper.get();
265 return context.canvasBase().cssCanvasClients();
266 },
267#if ENABLE(WEBGPU)
268 [&] (std::reference_wrapper<WebGPUDevice> deviceWrapper) {
269 auto& device = deviceWrapper.get();
270
271 HashSet<Element*> canvasElementClients;
272 {
273 Locker locker { CanvasRenderingContext::instancesLock() };
274 for (auto* canvasRenderingContext : CanvasRenderingContext::instances()) {
275 if (auto* canvasElement = canvasIfContextMatchesDevice(*canvasRenderingContext, device))
276 canvasElementClients.add(canvasElement);
277 }
278 }
279 return canvasElementClients;
280 },
281#endif
282 [] (Monostate) {
283 ASSERT_NOT_REACHED();
284 return HashSet<Element*>();
285 }
286 );
287}

289void InspectorCanvas::canvasChanged()
290{
291 auto* context = canvasContext();
292 ASSERT(context);
293
294 if (!context->hasActiveInspectorCanvasCallTracer())
295 return;
296
297 // Since 2D contexts are able to be fully reproduced in the frontend, we don't need snapshots.
298 if (is<CanvasRenderingContext2D>(context))
299 return;
300
301 m_contentChanged = true;
302}

304void InspectorCanvas::resetRecordingData()
305{
306 m_initialState = nullptr;
307 m_frames = nullptr;
308 m_currentActions = nullptr;
309 m_serializedDuplicateData = nullptr;
310 m_indexedDuplicateData.clear();
311 m_recordingName = { };
312 m_bufferLimit = 100 * 1024 * 1024;
313 m_bufferUsed = 0;
314 m_frameCount = std::nullopt;
315 m_framesCaptured = 0;
316 m_contentChanged = false;
317
318 auto* context = canvasContext();
319 ASSERT(context);
320 // FIXME: <https://webkit.org/b/201651> Web Inspector: Canvas: support canvas recordings for WebGPUDevice
321
322 context->setHasActiveInspectorCanvasCallTracer(false);
323}

335template<typename T> static Ref<JSON::ArrayOf<JSON::Value>> buildArrayForVector(const Vector<T>& vector)
336{
337 auto array = JSON::ArrayOf<JSON::Value>::create();
338 for (auto& item : vector)
339 array->addItem(item);
340 return array;
341}

817void InspectorCanvas::recordAction(String&& name, InspectorCanvasCallTracer::ProcessedArguments&& arguments)
818{
819 if (!m_initialState) {
820 // We should only construct the initial state for the first action of the recording.
821 ASSERT(!m_frames && !m_currentActions);
822
823 m_initialState = buildInitialState();
824 m_bufferUsed += m_initialState->memoryCost();
825 }
826
827 if (!m_frames)
828 m_frames = JSON::ArrayOf<Protocol::Recording::Frame>::create();
829
830 if (!m_currentActions) {
831 m_currentActions = JSON::ArrayOf<JSON::Value>::create();
832
833 auto frame = Protocol::Recording::Frame::create()
834 .setActions(*m_currentActions)
835 .release();
836
837 m_frames->addItem(WTFMove(frame));
838 ++m_framesCaptured;
839
840 m_currentFrameStartTime = MonotonicTime::now();
841 }
842
843 appendActionSnapshotIfNeeded();
844
845 auto* context = canvasContext();
846 ASSERT(context);
847 // FIXME: <https://webkit.org/b/201651> Web Inspector: Canvas: support canvas recordings for WebGPUDevice
848
849 if (is<ImageBitmapRenderingContext>(context) && shouldSnapshotBitmapRendererAction(name))
850 m_contentChanged = true;
851#if ENABLE(WEBGL)
852 else if (is<WebGLRenderingContext>(context) && shouldSnapshotWebGLAction(name))
853 m_contentChanged = true;
854#endif
855#if ENABLE(WEBGL2)
856 else if (is<WebGL2RenderingContext>(context) && shouldSnapshotWebGL2Action(name))
857 m_contentChanged = true;
858#endif
859
860 m_lastRecordedAction = buildAction(WTFMove(name), WTFMove(arguments));
861 m_bufferUsed += m_lastRecordedAction->memoryCost();
862 m_currentActions->addItem(*m_lastRecordedAction);
863}

865void InspectorCanvas::finalizeFrame()
866{
867 appendActionSnapshotIfNeeded();
868
869 if (m_frames && m_frames->length() && !std::isnan(m_currentFrameStartTime)) {
870 auto currentFrame = static_reference_cast<Protocol::Recording::Frame>(m_frames->get(m_frames->length() - 1));
871 currentFrame->setDuration((MonotonicTime::now() - m_currentFrameStartTime).milliseconds());
872
873 m_currentFrameStartTime = MonotonicTime::nan();
874 }
875
876 m_currentActions = nullptr;
877}

885 currentFrame->setIncomplete(true);

911Ref<Protocol::Canvas::Canvas> InspectorCanvas::buildObjectForCanvas(bool captureBacktrace)
912{
913 using ContextTypeType = std::optional<Protocol::Canvas::ContextType>;
914 auto contextType = WTF::switchOn(m_context,
915 [] (std::reference_wrapper<CanvasRenderingContext> contextWrapper) -> ContextTypeType {
916 auto& context = contextWrapper.get();
917 if (is<CanvasRenderingContext2D>(context))
918 return Protocol::Canvas::ContextType::Canvas2D;
919 if (is<ImageBitmapRenderingContext>(context))
920 return Protocol::Canvas::ContextType::BitmapRenderer;
921#if ENABLE(WEBGL)
922 if (is<WebGLRenderingContext>(context))
923 return Protocol::Canvas::ContextType::WebGL;
924#endif
925#if ENABLE(WEBGL2)
926 if (is<WebGL2RenderingContext>(context))
927 return Protocol::Canvas::ContextType::WebGL2;
928#endif
929 return std::nullopt;
930 },
931#if ENABLE(WEBGPU)
932 [] (std::reference_wrapper<WebGPUDevice>) {
933 return Protocol::Canvas::ContextType::WebGPU;
934 },
935#endif
936 [] (Monostate) {
937 ASSERT_NOT_REACHED();
938 return std::nullopt;
939 }
940 );
941 if (!contextType) {
942 ASSERT_NOT_REACHED();
943 contextType = Protocol::Canvas::ContextType::Canvas2D;
944 }
945
946 auto canvas = Protocol::Canvas::Canvas::create()
947 .setCanvasId(m_identifier)
948 .setContextType(contextType.value())
949 .release();
950
951 if (auto* node = canvasElement()) {
952 String cssCanvasName = node->document().nameForCSSCanvasElement(*node);
953 if (!cssCanvasName.isEmpty())
954 canvas->setCssCanvasName(cssCanvasName);
955
956 // FIXME: <https://webkit.org/b/178282> Web Inspector: send a DOM node with each Canvas payload and eliminate Canvas.requestNode
957 }
958
959 using ContextAttributesType = RefPtr<Protocol::Canvas::ContextAttributes>;
960 auto contextAttributes = WTF::switchOn(m_context,
961 [] (std::reference_wrapper<CanvasRenderingContext> contextWrapper) -> ContextAttributesType {
962 auto& context = contextWrapper.get();
963 if (is<ImageBitmapRenderingContext>(context)) {
964 auto contextAttributesPayload = Protocol::Canvas::ContextAttributes::create()
965 .release();
966 contextAttributesPayload->setAlpha(downcast<ImageBitmapRenderingContext>(context).hasAlpha());
967 return contextAttributesPayload;
968 }
969
970#if ENABLE(WEBGL)
971 if (is<WebGLRenderingContextBase>(context)) {
972 if (const auto& attributes = downcast<WebGLRenderingContextBase>(context).getContextAttributes()) {
973 auto contextAttributesPayload = Protocol::Canvas::ContextAttributes::create()
974 .release();
975 contextAttributesPayload->setAlpha(attributes->alpha);
976 contextAttributesPayload->setDepth(attributes->depth);
977 contextAttributesPayload->setStencil(attributes->stencil);
978 contextAttributesPayload->setAntialias(attributes->antialias);
979 contextAttributesPayload->setPremultipliedAlpha(attributes->premultipliedAlpha);
980 contextAttributesPayload->setPreserveDrawingBuffer(attributes->preserveDrawingBuffer);
981 switch (attributes->powerPreference) {
982 case WebGLPowerPreference::Default:
983 contextAttributesPayload->setPowerPreference("default");
984 break;
985 case WebGLPowerPreference::LowPower:
986 contextAttributesPayload->setPowerPreference("low-power");
987 break;
988 case WebGLPowerPreference::HighPerformance:
989 contextAttributesPayload->setPowerPreference("high-performance");
990 break;
991 }
992 contextAttributesPayload->setFailIfMajorPerformanceCaveat(attributes->failIfMajorPerformanceCaveat);
993 return contextAttributesPayload;
994 }
995 }
996#endif
997 return nullptr;
998 },
999#if ENABLE(WEBGPU)
1000 [] (std::reference_wrapper<WebGPUDevice> deviceWrapper) -> ContextAttributesType {
1001 auto& device = deviceWrapper.get();
1002 if (const auto& options = device.adapter().options()) {
1003 auto contextAttributesPayload = Protocol::Canvas::ContextAttributes::create()
1004 .release();
1005 if (const auto& powerPreference = options->powerPreference) {
1006 switch (powerPreference.value()) {
1007 case GPUPowerPreference::LowPower:
1008 contextAttributesPayload->setPowerPreference("low-power");
1009 break;
1010
1011 case GPUPowerPreference::HighPerformance:
1012 contextAttributesPayload->setPowerPreference("high-performance");
1013 break;
1014 }
1015 }
1016 return WTFMove(contextAttributesPayload);
1017 }
1018 return nullptr;
1019 },
1020#endif
1021 [] (Monostate) {
1022 ASSERT_NOT_REACHED();
1023 return nullptr;
1024 }
1025 );
1026 if (contextAttributes)
1027 canvas->setContextAttributes(contextAttributes.releaseNonNull());
1028
1029 // FIXME: <https://webkit.org/b/180833> Web Inspector: support OffscreenCanvas for Canvas related operations
1030
1031 if (auto* node = canvasElement()) {
1032 if (size_t memoryCost = node->memoryCost())
1033 canvas->setMemoryCost(memoryCost);
1034 }
1035
1036 if (captureBacktrace) {
1037 auto stackTrace = Inspector::createScriptCallStack(JSExecState::currentState(), Inspector::ScriptCallStack::maxCallStackSizeToCapture);
1038 canvas->setBacktrace(stackTrace->buildInspectorArray());
1039 }
1040
1041 return canvas;
1042}

1047 ASSERT(!m_lastRecordedAction);

1051 ASSERT(context);

1055 if (is<CanvasRenderingContext2D>(context))
1056 type = Protocol::Recording::Type::Canvas2D;

1134int InspectorCanvas::indexForData(DuplicateDataVariant data)
1135{
1136 size_t index = m_indexedDuplicateData.findMatching([&] (auto item) {
1137 if (data == item)
1138 return true;
1139
1140 auto traceA = WTF::get_if<RefPtr<ScriptCallStack>>(data);
1141 auto traceB = WTF::get_if<RefPtr<ScriptCallStack>>(item);
1142 if (traceA && *traceA && traceB && *traceB)
1143 return (*traceA)->isEqual((*traceB).get());
1144
1145 return false;
1146 });
1147 if (index != notFound) {
1148 ASSERT(index < static_cast<size_t>(std::numeric_limits<int>::max()));
1149 return static_cast<int>(index);
1150 }
1151
1152 if (!m_serializedDuplicateData)
1153 m_serializedDuplicateData = JSON::ArrayOf<JSON::Value>::create();
1154
1155 RefPtr<JSON::Value> item;
1156 WTF::switchOn(data,
1157 [&] (const RefPtr<HTMLImageElement>& imageElement) {
1158 String dataURL = "data:,"_s;
1159
1160 if (CachedImage* cachedImage = imageElement->cachedImage()) {
1161 Image* image = cachedImage->image();
1162 if (image && image != &Image::nullImage()) {
1163 auto imageBuffer = ImageBuffer::create(image->size(), RenderingMode::Unaccelerated, 1, DestinationColorSpace::SRGB(), PixelFormat::BGRA8);
1164 imageBuffer->context().drawImage(*image, FloatPoint(0, 0));
1165 dataURL = imageBuffer->toDataURL("image/png");
1166 }
1167 }
1168
1169 index = indexForData(dataURL);
1170 },
1171#if ENABLE(VIDEO)
1172 [&] (RefPtr<HTMLVideoElement>& videoElement) {
1173 String dataURL = "data:,"_s;
1174
1175 unsigned videoWidth = videoElement->videoWidth();
1176 unsigned videoHeight = videoElement->videoHeight();
1177 auto imageBuffer = ImageBuffer::create(FloatSize(videoWidth, videoHeight), RenderingMode::Unaccelerated, 1, DestinationColorSpace::SRGB(), PixelFormat::BGRA8);
1178 if (imageBuffer) {
1179 videoElement->paintCurrentFrameInContext(imageBuffer->context(), FloatRect(0, 0, videoWidth, videoHeight));
1180 dataURL = imageBuffer->toDataURL("image/png");
1181 }
1182
1183 index = indexForData(dataURL);
1184 },
1185#endif
1186 [&] (RefPtr<HTMLCanvasElement>& canvasElement) {
1187 String dataURL = "data:,"_s;
1188
1189 ExceptionOr<UncachedString> result = canvasElement->toDataURL("image/png"_s);
1190 if (!result.hasException())
1191 dataURL = result.releaseReturnValue().string;
1192
1193 index = indexForData(dataURL);
1194 },
1195 [&] (const RefPtr<CanvasGradient>& canvasGradient) { item = buildArrayForCanvasGradient(*canvasGradient); },
1196 [&] (const RefPtr<CanvasPattern>& canvasPattern) { item = buildArrayForCanvasPattern(*canvasPattern); },
1197 [&] (const RefPtr<ImageData>& imageData) { item = buildArrayForImageData(*imageData); },
1198 [&] (RefPtr<ImageBitmap>& imageBitmap) {
1199 index = indexForData(imageBitmap->buffer()->toDataURL("image/png"));
1200 },
1201 [&] (const RefPtr<ScriptCallStack>& scriptCallStack) {
1202 auto array = JSON::ArrayOf<double>::create();
1203 for (size_t i = 0; i < scriptCallStack->size(); ++i)
1204 array->addItem(indexForData(scriptCallStack->at(i)));
1205 item = WTFMove(array);
1206 },
1207#if ENABLE(CSS_TYPED_OM)
1208 [&] (const RefPtr<TypedOMCSSImageValue>& cssImageValue) {
1209 String dataURL = "data:,"_s;
1210
1211 if (auto* cachedImage = cssImageValue->image()) {
1212 auto* image = cachedImage->image();
1213 if (image && image != &Image::nullImage()) {
1214 auto imageBuffer = ImageBuffer::create(image->size(), RenderingMode::Unaccelerated, 1, DestinationColorSpace::SRGB(), PixelFormat::BGRA8);
1215 imageBuffer->context().drawImage(*image, FloatPoint(0, 0));
1216 dataURL = imageBuffer->toDataURL("image/png");
1217 }
1218 }
1219
1220 index = indexForData(dataURL);
1221 },
1222#endif
1223 [&] (const ScriptCallFrame& scriptCallFrame) {
1224 auto array = JSON::ArrayOf<double>::create();
1225 array->addItem(indexForData(scriptCallFrame.functionName()));
1226 array->addItem(indexForData(scriptCallFrame.sourceURL()));
1227 array->addItem(static_cast<int>(scriptCallFrame.lineNumber()));
1228 array->addItem(static_cast<int>(scriptCallFrame.columnNumber()));
1229 item = WTFMove(array);
1230 },
1231#if ENABLE(OFFSCREEN_CANVAS)
1232 [&] (const RefPtr<OffscreenCanvas> offscreenCanvas) {
1233 String dataURL = "data:,"_s;
1234
1235 if (offscreenCanvas->originClean() && offscreenCanvas->hasCreatedImageBuffer()) {
1236 if (auto *buffer = offscreenCanvas->buffer())
1237 dataURL = buffer->toDataURL("image/png");
1238 }
1239
1240 index = indexForData(dataURL);
1241 },
1242#endif
1243 [&] (const String& value) { item = JSON::Value::create(value); }
1244 );
1245
1246 if (item) {
1247 m_bufferUsed += item->memoryCost();
1248 m_serializedDuplicateData->addItem(item.releaseNonNull());
1249
1250 m_indexedDuplicateData.append(data);
1251 index = m_indexedDuplicateData.size() - 1;
1252 }
1253
1254 ASSERT(index < static_cast<size_t>(std::numeric_limits<int>::max()));
1255 return static_cast<int>(index);
1256}

1280Ref<Protocol::Recording::InitialState> InspectorCanvas::buildInitialState()
1281{
1282 auto* context = canvasContext();
1283 ASSERT(context);
1284 // FIXME: <https://webkit.org/b/201651> Web Inspector: Canvas: support canvas recordings for WebGPUDevice
1285
1286 auto initialStatePayload = Protocol::Recording::InitialState::create().release();
1287
1288 auto attributesPayload = JSON::Object::create();
1289 attributesPayload->setInteger("width"_s, context->canvasBase().width());
1290 attributesPayload->setInteger("height"_s, context->canvasBase().height());
1291
1292 auto statesPayload = JSON::ArrayOf<JSON::Object>::create();
1293
1294 auto parametersPayload = JSON::ArrayOf<JSON::Value>::create();
1295
1296 if (is<CanvasRenderingContext2D>(context)) {
1297 auto& context2d = downcast<CanvasRenderingContext2D>(*context);
1298 for (auto& state : context2d.stateStack()) {
1299 auto statePayload = JSON::Object::create();
1300
1301 statePayload->setArray(stringIndexForKey("setTransform"_s), buildArrayForAffineTransform(state.transform));
1302 statePayload->setDouble(stringIndexForKey("globalAlpha"_s), state.globalAlpha);
1303 statePayload->setInteger(stringIndexForKey("globalCompositeOperation"_s), indexForData(state.globalCompositeOperationString()));
1304 statePayload->setDouble(stringIndexForKey("lineWidth"_s), state.lineWidth);
1305 statePayload->setInteger(stringIndexForKey("lineCap"_s), indexForData(convertEnumerationToString(state.canvasLineCap())));
1306 statePayload->setInteger(stringIndexForKey("lineJoin"_s), indexForData(convertEnumerationToString(state.canvasLineJoin())));
1307 statePayload->setDouble(stringIndexForKey("miterLimit"_s), state.miterLimit);
1308 statePayload->setDouble(stringIndexForKey("shadowOffsetX"_s), state.shadowOffset.width());
1309 statePayload->setDouble(stringIndexForKey("shadowOffsetY"_s), state.shadowOffset.height());
1310 statePayload->setDouble(stringIndexForKey("shadowBlur"_s), state.shadowBlur);
1311 statePayload->setInteger(stringIndexForKey("shadowColor"_s), indexForData(serializationForHTML(state.shadowColor)));
1312
1313 // The parameter to `setLineDash` is itself an array, so we need to wrap the parameters
1314 // list in an array to allow spreading.
1315 auto setLineDash = JSON::ArrayOf<JSON::Value>::create();
1316 setLineDash->addItem(buildArrayForVector(state.lineDash));
1317 statePayload->setArray(stringIndexForKey("setLineDash"_s), WTFMove(setLineDash));
1318
1319 statePayload->setDouble(stringIndexForKey("lineDashOffset"_s), state.lineDashOffset);
1320 statePayload->setInteger(stringIndexForKey("font"_s), indexForData(state.fontString()));
1321 statePayload->setInteger(stringIndexForKey("textAlign"_s), indexForData(convertEnumerationToString(state.canvasTextAlign())));
1322 statePayload->setInteger(stringIndexForKey("textBaseline"_s), indexForData(convertEnumerationToString(state.canvasTextBaseline())));
1323 statePayload->setInteger(stringIndexForKey("direction"_s), indexForData(convertEnumerationToString(state.direction)));
1324
1325 int strokeStyleIndex;
1326 if (auto canvasGradient = state.strokeStyle.canvasGradient())
1327 strokeStyleIndex = indexForData(canvasGradient);
1328 else if (auto canvasPattern = state.strokeStyle.canvasPattern())
1329 strokeStyleIndex = indexForData(canvasPattern);
1330 else
1331 strokeStyleIndex = indexForData(state.strokeStyle.color());
1332 statePayload->setInteger(stringIndexForKey("strokeStyle"_s), strokeStyleIndex);
1333
1334 int fillStyleIndex;
1335 if (auto canvasGradient = state.fillStyle.canvasGradient())
1336 fillStyleIndex = indexForData(canvasGradient);
1337 else if (auto canvasPattern = state.fillStyle.canvasPattern())
1338 fillStyleIndex = indexForData(canvasPattern);
