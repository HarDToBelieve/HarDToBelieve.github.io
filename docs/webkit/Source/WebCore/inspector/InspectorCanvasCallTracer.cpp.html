72static InspectorCanvasAgent* enabledCanvasAgent(CanvasRenderingContext& canvasRenderingContext)
73{
74 ASSERT(InspectorInstrumentationPublic::hasFrontends());
75
76 auto* agents = InspectorInstrumentation::instrumentingAgents(canvasRenderingContext.canvasBase().scriptExecutionContext());
77 ASSERT(agents);
78 if (!agents)
79 return nullptr;
80
81 ASSERT(agents->enabledCanvasAgent());
82 return agents->enabledCanvasAgent();
83}

85#define PROCESS_ARGUMENT_DEFINITION(ArgumentType) \
86std::optional<InspectorCanvasCallTracer::ProcessedArgument> InspectorCanvasCallTracer::processArgument(CanvasRenderingContext& canvasRenderingContext, ArgumentType argument) \
87{ \
88 if (auto* canvasAgent = enabledCanvasAgent(canvasRenderingContext)) \
89 return canvasAgent->processArgument(canvasRenderingContext, argument); \
90 return std::nullopt; \
91} \
92// end of PROCESS_ARGUMENT_DEFINITION
93 FOR_EACH_INSPECTOR_CANVAS_CALL_TRACER_ARGUMENT(PROCESS_ARGUMENT_DEFINITION)
94#undef PROCESS_ARGUMENT_DEFINITION
95
96void InspectorCanvasCallTracer::recordAction(CanvasRenderingContext& canvasRenderingContext, String&& name, InspectorCanvasCallTracer::ProcessedArguments&& arguments)
97{
98 if (auto* canvasAgent = enabledCanvasAgent(canvasRenderingContext))
99 canvasAgent->recordAction(canvasRenderingContext, WTFMove(name), WTFMove(arguments));
100}
101
102std::optional<InspectorCanvasCallTracer::ProcessedArgument> InspectorCanvasCallTracer::processArgument(const HTMLCanvasElement& canvasElement, uint32_t argument)
103{
104 ASSERT(canvasElement.renderingContext());
105 return processArgument(*canvasElement.renderingContext(), argument);
106}
107
108void InspectorCanvasCallTracer::recordAction(const HTMLCanvasElement& canvasElement, String&& name, InspectorCanvasCallTracer::ProcessedArguments&& arguments)
109{
110 ASSERT(canvasElement.renderingContext());
111 recordAction(*canvasElement.renderingContext(), WTFMove(name), WTFMove(arguments));
112}
