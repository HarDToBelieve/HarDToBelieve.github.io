96InspectorController::InspectorController(Page& page, InspectorClient* inspectorClient)
97 : m_instrumentingAgents(InstrumentingAgents::create(*this))
98 , m_injectedScriptManager(makeUnique<WebInjectedScriptManager>(*this, WebInjectedScriptHost::create()))
99 , m_frontendRouter(FrontendRouter::create())
100 , m_backendDispatcher(BackendDispatcher::create(m_frontendRouter.copyRef()))
101 , m_overlay(makeUnique<InspectorOverlay>(page, inspectorClient))
102 , m_executionStopwatch(Stopwatch::create())
103 , m_debugger(page)
104 , m_page(page)
105 , m_inspectorClient(inspectorClient)
106{
107 ASSERT_ARG(inspectorClient, inspectorClient);
108
109 auto pageContext = pageAgentContext();
110
111 auto consoleAgent = makeUnique<PageConsoleAgent>(pageContext);
112 m_instrumentingAgents->setWebConsoleAgent(consoleAgent.get());
113 m_agents.append(WTFMove(consoleAgent));
114}

144void InspectorController::createLazyAgents()
145{
146 if (m_didCreateLazyAgents)
147 return;
148
149 m_didCreateLazyAgents = true;
150
151 m_injectedScriptManager->connect();
152
153 auto pageContext = pageAgentContext();
154
155 ensureInspectorAgent();
156 ensurePageAgent();
157
158 m_agents.append(makeUnique<PageRuntimeAgent>(pageContext));
159
160 auto debuggerAgent = makeUnique<PageDebuggerAgent>(pageContext);
161 auto debuggerAgentPtr = debuggerAgent.get();
162 m_agents.append(WTFMove(debuggerAgent));
163
164 m_agents.append(makeUnique<PageNetworkAgent>(pageContext));
165 m_agents.append(makeUnique<InspectorCSSAgent>(pageContext));
166 ensureDOMAgent();
167 m_agents.append(makeUnique<PageDOMDebuggerAgent>(pageContext, debuggerAgentPtr));
168 m_agents.append(makeUnique<InspectorApplicationCacheAgent>(pageContext));
169 m_agents.append(makeUnique<InspectorLayerTreeAgent>(pageContext));
170 m_agents.append(makeUnique<InspectorWorkerAgent>(pageContext));
171 m_agents.append(makeUnique<InspectorDOMStorageAgent>(pageContext));
172 m_agents.append(makeUnique<InspectorDatabaseAgent>(pageContext));
173 m_agents.append(makeUnique<InspectorIndexedDBAgent>(pageContext));
174
175 auto scriptProfilerAgentPtr = makeUnique<InspectorScriptProfilerAgent>(pageContext);
176 m_instrumentingAgents->setPersistentScriptProfilerAgent(scriptProfilerAgentPtr.get());
177 m_agents.append(WTFMove(scriptProfilerAgentPtr));
178
179#if ENABLE(RESOURCE_USAGE)
180 m_agents.append(makeUnique<InspectorCPUProfilerAgent>(pageContext));
181 m_agents.append(makeUnique<InspectorMemoryAgent>(pageContext));
182#endif
183 m_agents.append(makeUnique<PageHeapAgent>(pageContext));
184 m_agents.append(makeUnique<PageAuditAgent>(pageContext));
185 m_agents.append(makeUnique<InspectorCanvasAgent>(pageContext));
186 m_agents.append(makeUnique<InspectorTimelineAgent>(pageContext));
187 m_agents.append(makeUnique<InspectorAnimationAgent>(pageContext));
188
189 if (auto& commandLineAPIHost = m_injectedScriptManager->commandLineAPIHost())
190 commandLineAPIHost->init(m_instrumentingAgents.copyRef());
191}

193void InspectorController::inspectedPageDestroyed()

199 m_inspectorClient->inspectedPageDestroyed();
200 m_inspectorClient = nullptr;
201
202 m_agents.discardValues();
203}

225void InspectorController::didClearWindowObjectInWorld(Frame& frame, DOMWrapperWorld& world)
226{
227 if (&world != &mainThreadNormalWorld())
228 return;
229
230 if (frame.isMainFrame())
231 m_injectedScriptManager->discardInjectedScripts();
232
233 // If the page is supposed to serve as InspectorFrontend notify inspector frontend
234 // client that it's cleared so that the client can expose inspector bindings.
235 if (m_inspectorFrontendClient && frame.isMainFrame())
236 m_inspectorFrontendClient->windowObjectCleared();
237}

239void InspectorController::connectFrontend(Inspector::FrontendChannel& frontendChannel, bool isAutomaticInspection, bool immediatelyPause)
240{
241 ASSERT(m_inspectorClient);
242
243 // If a frontend has connected enable the developer extras and keep them enabled.
244 m_page.settings().setDeveloperExtrasEnabled(true);
245
246 createLazyAgents();
247
248 bool connectedFirstFrontend = !m_frontendRouter->hasFrontends();
249 m_isAutomaticInspection = isAutomaticInspection;
250 m_pauseAfterInitialization = immediatelyPause;
251
252 m_frontendRouter->connectFrontend(frontendChannel);
253
254 InspectorInstrumentation::frontendCreated();
255
256 if (connectedFirstFrontend) {
257 InspectorInstrumentation::registerInstrumentingAgents(m_instrumentingAgents.get());
258 m_agents.didCreateFrontendAndBackend(&m_frontendRouter.get(), &m_backendDispatcher.get());
259 }
260
261 m_inspectorClient->frontendCountChanged(m_frontendRouter->frontendCount());
262
263#if ENABLE(REMOTE_INSPECTOR)
264 if (hasLocalFrontend())
265 m_page.remoteInspectorInformationDidChange();
266#endif
267}
268
269void InspectorController::disconnectFrontend(FrontendChannel& frontendChannel)
270{
271 m_frontendRouter->disconnectFrontend(frontendChannel);
272
273 m_isAutomaticInspection = false;
274 m_pauseAfterInitialization = false;
275
276 InspectorInstrumentation::frontendDeleted();
277
278 bool disconnectedLastFrontend = !m_frontendRouter->hasFrontends();
279 if (disconnectedLastFrontend) {
280 // Notify agents first.
281 m_agents.willDestroyFrontendAndBackend(DisconnectReason::InspectorDestroyed);
282
283 // Clean up inspector resources.
284 m_injectedScriptManager->discardInjectedScripts();
285
286 // Unplug all instrumentations since they aren't needed now.
287 InspectorInstrumentation::unregisterInstrumentingAgents(m_instrumentingAgents.get());
288 }
289
290 m_inspectorClient->frontendCountChanged(m_frontendRouter->frontendCount());
291
292#if ENABLE(REMOTE_INSPECTOR)
293 if (disconnectedLastFrontend)
294 m_page.remoteInspectorInformationDidChange();
295#endif
296}

298void InspectorController::disconnectAllFrontends()
299{
300 // If the local frontend page was destroyed, close the window.
301 if (m_inspectorFrontendClient)
302 m_inspectorFrontendClient->closeWindow();
303
304 // The frontend should call setInspectorFrontendClient(nullptr) under closeWindow().
305 ASSERT(!m_inspectorFrontendClient);
306
307 if (!m_frontendRouter->hasFrontends())
308 return;
309
310 for (unsigned i = 0; i < m_frontendRouter->frontendCount(); ++i)
311 InspectorInstrumentation::frontendDeleted();
312
313 // Unplug all instrumentations to prevent further agent callbacks.
314 InspectorInstrumentation::unregisterInstrumentingAgents(m_instrumentingAgents.get());
315
316 // Notify agents first, since they may need to use InspectorClient.
317 m_agents.willDestroyFrontendAndBackend(DisconnectReason::InspectedTargetDestroyed);
318
319 // Clean up inspector resources.
320 m_injectedScriptManager->disconnect();
321
322 // Disconnect any remaining remote frontends.
323 m_frontendRouter->disconnectAllFrontends();
324 m_isAutomaticInspection = false;
325 m_pauseAfterInitialization = false;
326
327 m_inspectorClient->frontendCountChanged(m_frontendRouter->frontendCount());
328
329#if ENABLE(REMOTE_INSPECTOR)
330 m_page.remoteInspectorInformationDidChange();
331#endif
332}

334void InspectorController::show()
335{
336 ASSERT(!hasRemoteFrontend());
337
338 if (!enabled())
339 return;
340
341 if (m_frontendRouter->hasLocalFrontend())
342 m_inspectorClient->bringFrontendToFront();
343 else if (Inspector::FrontendChannel* frontendChannel = m_inspectorClient->openLocalFrontend(this))
344 connectFrontend(*frontendChannel);
345}

420InspectorAgent& InspectorController::ensureInspectorAgent()
421{
422 if (!m_inspectorAgent) {
423 auto pageContext = pageAgentContext();
424 auto inspectorAgent = makeUnique<InspectorAgent>(pageContext);
425 m_inspectorAgent = inspectorAgent.get();
426 m_instrumentingAgents->setPersistentInspectorAgent(m_inspectorAgent);
427 m_agents.append(WTFMove(inspectorAgent));
428 }
429 return *m_inspectorAgent;
430}

459bool InspectorController::canAccessInspectedScriptState(JSC::JSGlobalObject* lexicalGlobalObject) const
460{
461 JSLockHolder lock(lexicalGlobalObject);
462
463 auto* inspectedWindow = jsDynamicCast<JSDOMWindow*>(lexicalGlobalObject->vm(), lexicalGlobalObject);
464 if (!inspectedWindow)
465 return false;
466
467 return BindingSecurity::shouldAllowAccessToDOMWindow(lexicalGlobalObject, inspectedWindow->wrapped(), DoNotReportSecurityError);
468}

480void InspectorController::frontendInitialized()
481{
482 if (m_pauseAfterInitialization) {
483 m_pauseAfterInitialization = false;
484 if (auto* debuggerAgent = m_instrumentingAgents->enabledPageDebuggerAgent())
485 debuggerAgent->pause();
486 }
487
488#if ENABLE(REMOTE_INSPECTOR)
489 if (m_isAutomaticInspection)
490 m_page.inspectorDebuggable().unpauseForInitializedInspector();
491#endif
492}
