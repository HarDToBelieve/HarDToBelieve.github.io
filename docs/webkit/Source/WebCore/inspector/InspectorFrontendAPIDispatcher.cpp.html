56void InspectorFrontendAPIDispatcher::reset()
57{
58 m_frontendLoaded = false;
59 m_suspended = false;
60
61 invalidateQueuedExpressions();
62 invalidatePendingResponses();
63}

65void InspectorFrontendAPIDispatcher::frontendLoaded()
66{
67 ASSERT(m_frontendPage);
68 m_frontendLoaded = true;
69
70 // In some convoluted WebKitLegacy-only scenarios, the backend may try to dispatch events to the frontend
71 // underneath InspectorFrontendHost::loaded() when it is unsafe to execute script, causing suspend() to
72 // be called before the frontend has fully loaded. See <https://bugs.webkit.org/show_bug.cgi?id=218840>.
73 if (!m_suspended)
74 evaluateQueuedExpressions();
75}
76
77void InspectorFrontendAPIDispatcher::suspend(UnsuspendSoon unsuspendSoon)
78{
79 if (m_suspended)
80 return;
81
82 m_suspended = true;
83
84 if (unsuspendSoon == UnsuspendSoon::Yes) {
85 RunLoop::main().dispatch([protectedThis = makeRef(*this)] {
86 // If the frontend page has been deallocated, there's nothing to do.
87 if (!protectedThis->m_frontendPage)
88 return;
89
90 protectedThis->unsuspend();
91 });
92 }
93}
94
95void InspectorFrontendAPIDispatcher::unsuspend()
96{
97 if (!m_suspended)
98 return;
99
100 m_suspended = false;
101
102 if (m_frontendLoaded)
103 evaluateQueuedExpressions();
104}

106JSDOMGlobalObject* InspectorFrontendAPIDispatcher::frontendGlobalObject()
107{
108 if (!m_frontendPage)
109 return nullptr;
110
111 return m_frontendPage->mainFrame().script().globalObject(mainThreadNormalWorld());
112}

114static String expressionForEvaluatingCommand(const String& command, Vector<Ref<JSON::Value>>&& arguments)
115{
116 StringBuilder expression;
117 expression.append("InspectorFrontendAPI.dispatch([\"", command, '"');
118 for (auto& argument : arguments) {
119 expression.append(", ");
120 argument->writeJSON(expression);
121 }
122 expression.append("])");
123 return expression.toString();
124}

126InspectorFrontendAPIDispatcher::EvaluationResult InspectorFrontendAPIDispatcher::dispatchCommandWithResultSync(const String& command, Vector<Ref<JSON::Value>>&& arguments)
127{
128 if (m_suspended)
129 return makeUnexpected(EvaluationError::ExecutionSuspended);
130
131 return evaluateExpression(expressionForEvaluatingCommand(command, WTFMove(arguments)));
132}

134void InspectorFrontendAPIDispatcher::dispatchCommandWithResultAsync(const String& command, Vector<Ref<JSON::Value>>&& arguments, EvaluationResultHandler&& resultHandler)
135{
136 evaluateOrQueueExpression(expressionForEvaluatingCommand(command, WTFMove(arguments)), WTFMove(resultHandler));

144void InspectorFrontendAPIDispatcher::evaluateOrQueueExpression(const String& expression, EvaluationResultHandler&& optionalResultHandler)
145{
146 // If the frontend page has been deallocated, then there is nothing to do.
147 if (!m_frontendPage) {
148 if (optionalResultHandler)
149 optionalResultHandler(makeUnexpected(EvaluationError::ContextDestroyed));
150
151 return;
152 }
153
154 // Sometimes we get here by sending messages for events triggered by DOM mutations earlier in the call stack.
155 // If this is the case, then it's not safe to evaluate script synchronously, so do it later. This only affects
156 // WebKit1 and some layout tests that use a single web process for both the inspector and inspected page.
157 if (!ScriptDisallowedScope::InMainThread::isScriptAllowed())
158 suspend(UnsuspendSoon::Yes);
159
160 if (!m_frontendLoaded || m_suspended) {
161 m_queuedEvaluations.append(std::make_pair(expression, WTFMove(optionalResultHandler)));
162 return;
163 }
164
165 ValueOrException result = evaluateExpression(expression);
166 if (!optionalResultHandler)
167 return;
168
169 if (!result.has_value()) {
170 optionalResultHandler(result);
171 return;
172 }
173
174 JSDOMGlobalObject* globalObject = frontendGlobalObject();
175 if (!globalObject) {
176 optionalResultHandler(makeUnexpected(EvaluationError::ContextDestroyed));
177 return;
178 }
179
180 auto& vm = globalObject->vm();
181 auto* castedPromise = JSC::jsDynamicCast<JSC::JSPromise*>(vm, result.value());
182 if (!castedPromise) {
183 // Simple case: result is NOT a promise, just return the JSValue.
184 optionalResultHandler(result);
185 return;
186 }
187
188 // If the result is a promise, call the result handler when the promise settles.
189 Ref<DOMPromise> promise = DOMPromise::create(*globalObject, *castedPromise);
190 m_pendingResponses.set(promise.copyRef(), WTFMove(optionalResultHandler));
191 auto isRegistered = promise->whenSettled([promise = promise.copyRef(), weakThis = makeWeakPtr(*this)] {
192 // If `this` is cleared or the responses map is empty, then the promise settled
193 // beyond the time when we care about its result. Ignore late-settled promises.
194 // We clear out completion handlers for pending responses during teardown.
195 if (!weakThis)
196 return;
197
198 auto strongThis = makeRef(*weakThis);
199 if (!strongThis->m_pendingResponses.size())
200 return;
201
202 EvaluationResultHandler resultHandler = strongThis->m_pendingResponses.take(promise);
203 ASSERT(resultHandler);
204
205 JSDOMGlobalObject* globalObject = strongThis->frontendGlobalObject();
206 if (!globalObject) {
207 resultHandler(makeUnexpected(EvaluationError::ContextDestroyed));
208 return;
209 }
210
211 resultHandler({ promise->promise()->result(globalObject->vm()) });
212 });
213
214 if (isRegistered == DOMPromise::IsCallbackRegistered::No)
215 optionalResultHandler(makeUnexpected(EvaluationError::InternalError));
216}

218void InspectorFrontendAPIDispatcher::invalidateQueuedExpressions()
219{
220 auto queuedEvaluations = std::exchange(m_queuedEvaluations, { });
221 for (auto& pair : queuedEvaluations) {
222 auto resultHandler = WTFMove(pair.second);
223 if (resultHandler)
224 resultHandler(makeUnexpected(EvaluationError::ContextDestroyed));
225 }
226}
227
228void InspectorFrontendAPIDispatcher::invalidatePendingResponses()
229{
230 auto pendingResponses = std::exchange(m_pendingResponses, { });
231 for (auto& callback : pendingResponses.values())
232 callback(makeUnexpected(EvaluationError::ContextDestroyed));
233
234 // No more pending responses should have been added while erroring out the callbacks.
235 ASSERT(m_pendingResponses.isEmpty());
236}
237
238void InspectorFrontendAPIDispatcher::evaluateQueuedExpressions()
239{
240 // If the frontend page has been deallocated, then there is nothing to do.
241 if (!m_frontendPage)
242 return;
243
244 if (m_queuedEvaluations.isEmpty())
245 return;
246
247 auto queuedEvaluations = std::exchange(m_queuedEvaluations, { });
248 for (auto& pair : queuedEvaluations) {
249 auto result = evaluateExpression(pair.first);
250 if (auto resultHandler = WTFMove(pair.second))
251 resultHandler(result);
252 }
253}
254
255ValueOrException InspectorFrontendAPIDispatcher::evaluateExpression(const String& expression)
256{
257 ASSERT(m_frontendPage);
258 ASSERT(!m_suspended);
259 ASSERT(m_queuedEvaluations.isEmpty());
260
261 JSC::SuspendExceptionScope scope(&m_frontendPage->inspectorController().vm());
262 return m_frontendPage->mainFrame().script().evaluateInWorld(ScriptSourceCode(expression), mainThreadNormalWorld());
263}

265void InspectorFrontendAPIDispatcher::evaluateExpressionForTesting(const String& expression)
266{
