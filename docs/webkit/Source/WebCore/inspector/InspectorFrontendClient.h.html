50
51class FloatRect;
52class InspectorFrontendAPIDispatcher;
53class Page;

55class InspectorFrontendClient : public CanMakeWeakPtr<InspectorFrontendClient> {
56public:
57 enum class DockSide {
58 Undocked = 0,
59 Right,
60 Left,
61 Bottom,
62 };
63
64 virtual ~InspectorFrontendClient() = default;
65
66 WEBCORE_EXPORT virtual void windowObjectCleared() = 0;
67 virtual void frontendLoaded() = 0;
68
69 virtual void pagePaused() = 0;
70 virtual void pageUnpaused() = 0;
71
72 virtual void startWindowDrag() = 0;
73 virtual void moveWindowBy(float x, float y) = 0;
74
75 // Information about the debuggable.
76 virtual bool isRemote() const = 0;
77 virtual String localizedStringsURL() const = 0;
78 virtual String backendCommandsURL() const = 0;
79 virtual Inspector::DebuggableType debuggableType() const = 0;
80 virtual String targetPlatformName() const = 0;
81 virtual String targetBuildVersion() const = 0;
82 virtual String targetProductVersion() const = 0;
83 virtual bool targetIsSimulator() const = 0;
84 virtual unsigned inspectionLevel() const = 0;
85
86 virtual void bringToFront() = 0;
87 virtual void closeWindow() = 0;
88 virtual void reopen() = 0;
89 virtual void resetState() = 0;
90
91 enum class Appearance {
92 System,
93 Light,
94 Dark,
95 };
96 WEBCORE_EXPORT virtual void setForcedAppearance(Appearance) = 0;
97
98 virtual UserInterfaceLayoutDirection userInterfaceLayoutDirection() const = 0;
99
100 WEBCORE_EXPORT virtual bool supportsDockSide(DockSide) = 0;
101 WEBCORE_EXPORT virtual void requestSetDockSide(DockSide) = 0;
102 WEBCORE_EXPORT virtual void changeAttachedWindowHeight(unsigned) = 0;
103 WEBCORE_EXPORT virtual void changeAttachedWindowWidth(unsigned) = 0;
104
105 WEBCORE_EXPORT virtual void changeSheetRect(const FloatRect&) = 0;
106
107 WEBCORE_EXPORT virtual void openURLExternally(const String& url) = 0;
108 virtual bool canSave() = 0;
109 virtual void save(const WTF::String& url, const WTF::String& content, bool base64Encoded, bool forceSaveAs) = 0;
110 virtual void append(const WTF::String& url, const WTF::String& content) = 0;
111
112 virtual void inspectedURLChanged(const String&) = 0;
113 virtual void showCertificate(const CertificateInfo&) = 0;
114
115#if ENABLE(INSPECTOR_TELEMETRY)
116 virtual bool supportsDiagnosticLogging() { return false; }
117 virtual bool diagnosticLoggingAvailable() { return false; }
118 virtual void logDiagnosticEvent(const String& /* eventName */, const DiagnosticLoggingClient::ValueDictionary&) { }
119#endif
120
121#if ENABLE(INSPECTOR_EXTENSIONS)
122 virtual bool supportsWebExtensions() { return false; }
123 virtual void didShowExtensionTab(const Inspector::ExtensionID&, const Inspector::ExtensionTabID&) { }
124 virtual void didHideExtensionTab(const Inspector::ExtensionID&, const Inspector::ExtensionTabID&) { }
125#endif
126
127 WEBCORE_EXPORT virtual void sendMessageToBackend(const String&) = 0;
128 WEBCORE_EXPORT virtual InspectorFrontendAPIDispatcher& frontendAPIDispatcher() = 0;
129 WEBCORE_EXPORT virtual Page* frontendPage() = 0;
130
131 WEBCORE_EXPORT virtual bool isUnderTest() = 0;
132};
