73#if ENABLE(CONTEXT_MENUS)
74class FrontendMenuProvider : public ContextMenuProvider {
75public:
76 static Ref<FrontendMenuProvider> create(InspectorFrontendHost* frontendHost, Deprecated::ScriptObject frontendApiObject, const Vector<ContextMenuItem>& items)
77 {
78 return adoptRef(*new FrontendMenuProvider(frontendHost, frontendApiObject, items));
79 }
80
81 void disconnect()
82 {
83 m_frontendApiObject = { };
84 m_frontendHost = nullptr;
85 }
86
87private:
88 FrontendMenuProvider(InspectorFrontendHost* frontendHost, Deprecated::ScriptObject frontendApiObject, const Vector<ContextMenuItem>& items)
89 : m_frontendHost(frontendHost)
90 , m_frontendApiObject(frontendApiObject)
91 , m_items(items)
92 {
93 }
94
95 ~FrontendMenuProvider() override
96 {
97 contextMenuCleared();
98 }
99
100 void populateContextMenu(ContextMenu* menu) override
101 {
102 for (auto& item : m_items)
103 menu->appendItem(item);
104 }
105
106 void contextMenuItemSelected(ContextMenuAction action, const String&) override
107 {
108 if (m_frontendHost) {
109 UserGestureIndicator gestureIndicator(ProcessingUserGesture);
110 int itemNumber = action - ContextMenuItemBaseCustomTag;
111
112 Deprecated::ScriptFunctionCall function(m_frontendApiObject, "contextMenuItemSelected", WebCore::functionCallHandlerFromAnyThread);
113 function.appendArgument(itemNumber);
114 function.call();
115 }
116 }
117
118 void contextMenuCleared() override
119 {
120 if (m_frontendHost) {
121 Deprecated::ScriptFunctionCall function(m_frontendApiObject, "contextMenuCleared", WebCore::functionCallHandlerFromAnyThread);
122 function.call();
123
124 m_frontendHost->m_menuProvider = nullptr;
125 }
126 m_items.clear();
127 }
128
129 InspectorFrontendHost* m_frontendHost;
130 Deprecated::ScriptObject m_frontendApiObject;
131 Vector<ContextMenuItem> m_items;
132};
133#endif

159void InspectorFrontendHost::addSelfToGlobalObjectInWorld(DOMWrapperWorld& world)
160{
161 auto& lexicalGlobalObject = *globalObject(world, m_frontendPage ? &m_frontendPage->mainFrame() : nullptr);
162 auto& vm = lexicalGlobalObject.vm();
163 JSC::JSLockHolder lock(vm);
164 auto scope = DECLARE_CATCH_SCOPE(vm);
165
166 auto& globalObject = *JSC::jsCast<JSDOMGlobalObject*>(&lexicalGlobalObject);
167 globalObject.putDirect(vm, JSC::Identifier::fromString(vm, "InspectorFrontendHost"), toJS<IDLInterface<InspectorFrontendHost>>(lexicalGlobalObject, globalObject, *this));
168 if (UNLIKELY(scope.exception()))
169 reportException(&lexicalGlobalObject, scope.exception());
170}

262void InspectorFrontendHost::setForcedAppearance(String appearance)
263{
264 if (appearance == "light"_s) {
265 if (m_frontendPage)
266 m_frontendPage->setUseDarkAppearanceOverride(false);
267 if (m_client)
268 m_client->setForcedAppearance(InspectorFrontendClient::Appearance::Light);
269 return;
270 }
271
272 if (appearance == "dark"_s) {
273 if (m_frontendPage)
274 m_frontendPage->setUseDarkAppearanceOverride(true);
275 if (m_client)
276 m_client->setForcedAppearance(InspectorFrontendClient::Appearance::Dark);
277 return;
278 }
279
280 if (m_frontendPage)
281 m_frontendPage->setUseDarkAppearanceOverride(std::nullopt);
282 if (m_client)
283 m_client->setForcedAppearance(InspectorFrontendClient::Appearance::System);
284}

471#if ENABLE(CONTEXT_MENUS)
472
473static void populateContextMenu(Vector<InspectorFrontendHost::ContextMenuItem>&& items, ContextMenu& menu)
474{
475 for (auto& item : items) {
476 if (item.type == "separator") {
477 menu.appendItem({ SeparatorType, ContextMenuItemTagNoAction, { } });
478 continue;
479 }
480
481 if (item.type == "subMenu" && item.subItems) {
482 ContextMenu subMenu;
483 populateContextMenu(WTFMove(*item.subItems), subMenu);
484
485 menu.appendItem({ SubmenuType, ContextMenuItemTagNoAction, item.label, &subMenu });
486 continue;
487 }
488
489 auto type = item.type == "checkbox" ? CheckableActionType : ActionType;
490 auto action = static_cast<ContextMenuAction>(ContextMenuItemBaseCustomTag + item.id.value_or(0));
491 ContextMenuItem menuItem = { type, action, item.label };
492 if (item.enabled)
493 menuItem.setEnabled(*item.enabled);
494 if (item.checked)
495 menuItem.setChecked(*item.checked);
496 menu.appendItem(menuItem);
497 }
498}
499#endif

501void InspectorFrontendHost::showContextMenu(Event& event, Vector<ContextMenuItem>&& items)
502{
503#if ENABLE(CONTEXT_MENUS)
504 ASSERT(m_frontendPage);
505
506 auto& lexicalGlobalObject = *globalObject(debuggerWorld(), &m_frontendPage->mainFrame());
507 auto& vm = lexicalGlobalObject.vm();
508 auto value = lexicalGlobalObject.get(&lexicalGlobalObject, JSC::Identifier::fromString(vm, "InspectorFrontendAPI"));
509 ASSERT(value);
510 ASSERT(value.isObject());
511 auto* frontendAPIObject = asObject(value);
512
513 ContextMenu menu;
514 populateContextMenu(WTFMove(items), menu);
515
516 auto menuProvider = FrontendMenuProvider::create(this, { &lexicalGlobalObject, frontendAPIObject }, menu.items());
517 m_menuProvider = menuProvider.ptr();
518 m_frontendPage->contextMenuController().showContextMenu(event, menuProvider);
519#else
520 UNUSED_PARAM(event);
521 UNUSED_PARAM(items);
522#endif
523}

525void InspectorFrontendHost::dispatchEventAsContextMenuEvent(Event& event)
526{
527#if ENABLE(CONTEXT_MENUS) && USE(ACCESSIBILITY_CONTEXT_MENUS)
528 if (!is<MouseEvent>(event))
529 return;
530
531 auto& mouseEvent = downcast<MouseEvent>(event);
532 auto& frame = *downcast<Node>(mouseEvent.target())->document().frame();
533 m_frontendPage->contextMenuController().showContextMenuAt(frame, roundedIntPoint(mouseEvent.absoluteLocation()));
534#else
535 UNUSED_PARAM(event);
536#endif
537}

596bool InspectorFrontendHost::showCertificate(const String& serializedCertificate)
597{
598 if (!m_client)
599 return false;
600
601 auto data = base64Decode(serializedCertificate);
602 if (!data)
603 return false;
604
605 WTF::Persistence::Decoder decoder(data->data(), data->size());
606 std::optional<CertificateInfo> certificateInfo;
607 decoder >> certificateInfo;
608 if (!certificateInfo)
609 return false;
610
611 if (certificateInfo->isEmpty())
612 return false;
613
614 m_client->showCertificate(*certificateInfo);
615 return true;
616}

633static std::optional<DiagnosticLoggingClient::ValuePayload> valuePayloadFromJSONValue(Ref<JSON::Value>&& value)
634{
635 switch (value->type()) {
636 case JSON::Value::Type::Array:
637 case JSON::Value::Type::Null:
638 case JSON::Value::Type::Object:
639 ASSERT_NOT_REACHED();
640 return std::nullopt;
641
642 case JSON::Value::Type::Boolean:
643 return DiagnosticLoggingClient::ValuePayload(value->asBoolean().value_or(false));
644
645 case JSON::Value::Type::Double:
646 return DiagnosticLoggingClient::ValuePayload(value->asDouble().value_or(0));
647
648 case JSON::Value::Type::Integer:
649 return DiagnosticLoggingClient::ValuePayload(static_cast<long long>(value->asInteger().value_or(0)));
650
651 case JSON::Value::Type::String:
652 return DiagnosticLoggingClient::ValuePayload(value->asString());
653 }
654
655 ASSERT_NOT_REACHED();
656 return std::nullopt;
657}
658
659void InspectorFrontendHost::logDiagnosticEvent(const String& eventName, const String& payloadString)
660{
661 if (!supportsDiagnosticLogging())
662 return;
663
664 auto payloadValue = JSON::Value::parseJSON(payloadString);
665 if (!payloadValue)
666 return;
667
668 auto payloadObject = payloadValue->asObject();
669 if (!payloadObject)
670 return;
671
672 DiagnosticLoggingClient::ValueDictionary dictionary;
673 for (auto& [key, value] : *payloadObject) {
674 if (auto valuePayload = valuePayloadFromJSONValue(WTFMove(value)))
675 dictionary.set(key, WTFMove(valuePayload.value()));
676 }
677
678 m_client->logDiagnosticEvent(makeString("WebInspector."_s, eventName), dictionary);
679}
