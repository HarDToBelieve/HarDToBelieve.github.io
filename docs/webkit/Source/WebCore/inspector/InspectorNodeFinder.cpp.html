47static String stripCharacters(const String& string, const char startCharacter, const char endCharacter, bool& startCharFound, bool& endCharFound)
48{
49 startCharFound = string.startsWith(startCharacter);
50 endCharFound = string.endsWith(endCharacter);
51
52 unsigned start = startCharFound ? 1 : 0;
53 unsigned end = string.length() - (endCharFound ? 1 : 0);
54 return string.substring(start, end - start);
55}

57InspectorNodeFinder::InspectorNodeFinder(const String& query, bool caseSensitive)
58 : m_query(query)
59 , m_caseSensitive(caseSensitive)
60{
61 m_tagNameQuery = stripCharacters(query, '<', '>', m_startTagFound, m_endTagFound);
62
63 bool startQuoteFound, endQuoteFound;
64 m_attributeQuery = stripCharacters(query, '"', '"', startQuoteFound, endQuoteFound);
65 m_exactAttributeMatch = startQuoteFound && endQuoteFound;
66}

68void InspectorNodeFinder::performSearch(Node* parentNode)
69{
70 if (!parentNode)
71 return;
72
73 searchUsingXPath(*parentNode);
74 searchUsingCSSSelectors(*parentNode);
75
76 // Keep the DOM tree traversal last. This way iframe content will come after their parents.
77 searchUsingDOMTreeTraversal(*parentNode);
78}
79
80void InspectorNodeFinder::searchUsingDOMTreeTraversal(Node& parentNode)
81{
82 // Manual plain text search.
83 for (auto* node = &parentNode; node; node = NodeTraversal::next(*node, &parentNode)) {
84 switch (node->nodeType()) {
85 case Node::TEXT_NODE:
86 case Node::COMMENT_NODE:
87 case Node::CDATA_SECTION_NODE:
88 if (checkContains(node->nodeValue(), m_query))
89 m_results.add(node);
90 break;
91 case Node::ELEMENT_NODE:
92 if (matchesElement(downcast<Element>(*node)))
93 m_results.add(node);
94 if (is<HTMLFrameOwnerElement>(downcast<Element>(*node)))
95 performSearch(downcast<HTMLFrameOwnerElement>(*node).contentDocument());
96 break;
97 default:
98 break;
99 }
100 }
101}

103bool InspectorNodeFinder::checkEquals(const String& a, const String& b)
104{
105 if (m_caseSensitive)
106 return a == b;
107 return equalIgnoringASCIICase(a, b);

131bool InspectorNodeFinder::matchesAttribute(const Attribute& attribute)
132{
133 if (checkContains(attribute.localName().string(), m_query))
134 return true;
135
136 auto value = attribute.value().string();
137 return m_exactAttributeMatch ? checkEquals(value, m_attributeQuery) : checkContains(value, m_attributeQuery);
138}
139
140bool InspectorNodeFinder::matchesElement(const Element& element)
141{
142 String nodeName = element.nodeName();
143 if ((!m_startTagFound && !m_endTagFound && checkContains(nodeName, m_tagNameQuery))
144 || (m_startTagFound && m_endTagFound && checkEquals(nodeName, m_tagNameQuery))
145 || (m_startTagFound && !m_endTagFound && checkStartsWith(nodeName, m_tagNameQuery))
146 || (!m_startTagFound && m_endTagFound && checkEndsWith(nodeName, m_tagNameQuery)))
147 return true;
148
149 if (!element.hasAttributes())
150 return false;
151
152 for (const Attribute& attribute : element.attributesIterator()) {
153 if (matchesAttribute(attribute))
154 return true;
155 }
156
157 return false;
158}

160void InspectorNodeFinder::searchUsingXPath(Node& parentNode)
161{
162 auto evaluateResult = parentNode.document().evaluate(m_query, parentNode, nullptr, XPathResult::ORDERED_NODE_SNAPSHOT_TYPE, nullptr);
163 if (evaluateResult.hasException())
164 return;
165 auto result = evaluateResult.releaseReturnValue();
166
167 auto snapshotLengthResult = result->snapshotLength();
168 if (snapshotLengthResult.hasException())
169 return;
170 unsigned size = snapshotLengthResult.releaseReturnValue();
171
172 for (unsigned i = 0; i < size; ++i) {
173 auto snapshotItemResult = result->snapshotItem(i);
174 if (snapshotItemResult.hasException())
175 return;
176 Node* node = snapshotItemResult.releaseReturnValue();
177
178 if (is<Attr>(*node))
179 node = downcast<Attr>(*node).ownerElement();
180
181 // XPath can get out of the context node that we pass as the starting point to evaluate, so we need to filter for just the nodes we care about.
182 if (parentNode.contains(node))
183 m_results.add(node);
184 }
185}
186
187void InspectorNodeFinder::searchUsingCSSSelectors(Node& parentNode)
188{
189 if (!is<ContainerNode>(parentNode))
190 return;
191
192 auto queryResult = downcast<ContainerNode>(parentNode).querySelectorAll(m_query);
193 if (queryResult.hasException())
194 return;
195
196 auto nodeList = queryResult.releaseReturnValue();
197 unsigned size = nodeList->length();
198 for (unsigned i = 0; i < size; ++i)
199 m_results.add(nodeList->item(i));
200}
