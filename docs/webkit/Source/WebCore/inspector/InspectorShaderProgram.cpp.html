110#if ENABLE(WEBGL)
111static WebGLShader* shaderForType(WebGLProgram& program, Inspector::Protocol::Canvas::ShaderType shaderType)
112{
113 switch (shaderType) {
114 case Inspector::Protocol::Canvas::ShaderType::Fragment:
115 return program.getAttachedShader(GraphicsContextGL::FRAGMENT_SHADER);
116
117 case Inspector::Protocol::Canvas::ShaderType::Vertex:
118 return program.getAttachedShader(GraphicsContextGL::VERTEX_SHADER);
119
120 // Compute shaders are a WebGPU concept.
121 case Inspector::Protocol::Canvas::ShaderType::Compute:
122 return nullptr;
123 }
124
125 ASSERT_NOT_REACHED();
126 return nullptr;
127}
128#endif
129
130#if ENABLE(WEBGPU)
131static RefPtr<WebGPUShaderModule> shaderForType(WebGPUPipeline& pipeline, Inspector::Protocol::Canvas::ShaderType shaderType)
132{
133 switch (shaderType) {
134 case Inspector::Protocol::Canvas::ShaderType::Compute:
135 if (is<WebGPUComputePipeline>(pipeline))
136 return downcast<WebGPUComputePipeline>(pipeline).computeShader();
137 return nullptr;
138
139 case Inspector::Protocol::Canvas::ShaderType::Fragment:
140 if (is<WebGPURenderPipeline>(pipeline))
141 return downcast<WebGPURenderPipeline>(pipeline).fragmentShader();
142 return nullptr;
143
144 case Inspector::Protocol::Canvas::ShaderType::Vertex:
145 if (is<WebGPURenderPipeline>(pipeline))
146 return downcast<WebGPURenderPipeline>(pipeline).vertexShader();
147 return nullptr;
148 }
149
150 ASSERT_NOT_REACHED();
151 return nullptr;
152}
153#endif

155String InspectorShaderProgram::requestShaderSource(Inspector::Protocol::Canvas::ShaderType shaderType)
156{
157 return WTF::switchOn(m_program,
158#if ENABLE(WEBGL)
159 [&] (std::reference_wrapper<WebGLProgram> programWrapper) {
160 auto& program = programWrapper.get();
161 if (auto* shader = shaderForType(program, shaderType))
162 return shader->getSource();
163 return String();
164 },
165#endif
166#if ENABLE(WEBGPU)
167 [&] (std::reference_wrapper<WebGPUPipeline> pipelineWrapper) {
168 auto& pipeline = pipelineWrapper.get();
169 if (auto shader = shaderForType(pipeline, shaderType))
170 return shader->source();
171 return String();
172 },
173#endif
174 [&] (Monostate) {
175#if ENABLE(WEBGL) || ENABLE(WEBGPU)
176 ASSERT_NOT_REACHED();
177#endif
178 return String();
179 }
180 );
181}

183bool InspectorShaderProgram::updateShader(Inspector::Protocol::Canvas::ShaderType shaderType, const String& source)
184{
185 return WTF::switchOn(m_program,
186#if ENABLE(WEBGL)
187 [&] (std::reference_wrapper<WebGLProgram> programWrapper) {
188 auto& program = programWrapper.get();
189 if (auto* shader = shaderForType(program, shaderType)) {
190 if (auto* context = m_canvas.canvasContext()) {
191 if (is<WebGLRenderingContextBase>(context)) {
192 auto& contextWebGLBase = downcast<WebGLRenderingContextBase>(*context);
193 contextWebGLBase.shaderSource(*shader, source);
194 contextWebGLBase.compileShader(*shader);
195 if (shader->isValid()) {
196 contextWebGLBase.linkProgramWithoutInvalidatingAttribLocations(&program);
197 return true;
198 }
199 }
200 }
201 }
202 return false;
203 },
204#endif
205#if ENABLE(WEBGPU)
206 [&] (std::reference_wrapper<WebGPUPipeline> pipelineWrapper) {
207 auto& pipeline = pipelineWrapper.get();
208 if (auto* device = m_canvas.deviceContext()) {
209 if (pipeline.cloneShaderModules(*device)) {
210 if (auto shader = shaderForType(pipeline, shaderType)) {
211 shader->update(*device, source);
212 if (pipeline.recompile(*device))
213 return true;
214 }
215 }
216 }
217 return false;
218 },
219#endif
220 [&] (Monostate) {
221#if ENABLE(WEBGL) || ENABLE(WEBGPU)
222 ASSERT_NOT_REACHED();
223#endif
224 return false;
225 }
226 );
227}

229Ref<Inspector::Protocol::Canvas::ShaderProgram> InspectorShaderProgram::buildObjectForShaderProgram()
230{
231 bool sharesVertexFragmentShader = false;
232
233 using ProgramTypeType = std::optional<Inspector::Protocol::Canvas::ProgramType>;
234 auto programType = WTF::switchOn(m_program,
235#if ENABLE(WEBGL)
236 [&] (std::reference_wrapper<WebGLProgram>) -> ProgramTypeType {
237 return Inspector::Protocol::Canvas::ProgramType::Render;
238 },
239#endif
240#if ENABLE(WEBGPU)
241 [&] (std::reference_wrapper<WebGPUPipeline> pipelineWrapper) -> ProgramTypeType {
242 auto& pipeline = pipelineWrapper.get();
243 if (is<WebGPUComputePipeline>(pipeline))
244 return Inspector::Protocol::Canvas::ProgramType::Compute;
245 if (is<WebGPURenderPipeline>(pipeline)) {
246 auto& renderPipeline = downcast<WebGPURenderPipeline>(pipeline);
247 if (renderPipeline.vertexShader() == renderPipeline.fragmentShader())
248 sharesVertexFragmentShader = true;
249 return Inspector::Protocol::Canvas::ProgramType::Render;
250 }
251 return std::nullopt;
252 },
253#endif
254 [&] (Monostate) -> ProgramTypeType {
255#if ENABLE(WEBGL) || ENABLE(WEBGPU)
256 ASSERT_NOT_REACHED();
257#endif
258 return std::nullopt;
259 }
260 );
261 if (!programType) {
262 ASSERT_NOT_REACHED();
263 programType = Inspector::Protocol::Canvas::ProgramType::Render;
264 }
265
266 auto payload = Inspector::Protocol::Canvas::ShaderProgram::create()
267 .setProgramId(m_identifier)
268 .setProgramType(programType.value())
269 .setCanvasId(m_canvas.identifier())
270 .release();
271 if (sharesVertexFragmentShader)
272 payload->setSharesVertexFragmentShader(true);
273 return payload;
274}
