137class InstrumentingAgents : public RefCounted<InstrumentingAgents> {
138 WTF_MAKE_NONCOPYABLE(InstrumentingAgents);
139 WTF_MAKE_FAST_ALLOCATED;
140public:
141 // FIXME: InstrumentingAgents could be uniquely owned by InspectorController if instrumentation
142 // cookies kept only a weak reference to InstrumentingAgents. Then, reset() would be unnecessary.
143 static Ref<InstrumentingAgents> create(Inspector::InspectorEnvironment& environment)
144 {
145 return adoptRef(*new InstrumentingAgents(environment));
146 }
147 ~InstrumentingAgents() = default;
148 void reset();
149
150 Inspector::InspectorEnvironment& inspectorEnvironment() const { return m_environment; }
151
152#define DECLARE_GETTER_SETTER_FOR_INSPECTOR_AGENT(Class, Name, Getter, Setter) \
153 Class* Getter##Name() const { return m_##Getter##Name; } \
154 void set##Setter##Name(Class* agent) { m_##Getter##Name = agent; } \
155
156FOR_EACH_INSPECTOR_AGENT(DECLARE_GETTER_SETTER_FOR_INSPECTOR_AGENT)
157#undef DECLARE_GETTER_SETTER_FOR_INSPECTOR_AGENT
158
159private:
160 InstrumentingAgents(Inspector::InspectorEnvironment&);
161
162 Inspector::InspectorEnvironment& m_environment;
163
164#define DECLARE_MEMBER_VARIABLE_FOR_INSPECTOR_AGENT(Class, Name, Getter, Setter) \
165 Class* m_##Getter##Name { nullptr }; \
166
167FOR_EACH_INSPECTOR_AGENT(DECLARE_MEMBER_VARIABLE_FOR_INSPECTOR_AGENT)
168#undef DECLARE_MEMBER_VARIABLE_FOR_INSPECTOR_AGENT
169};
