51JSValue WebInjectedScriptHost::subtype(JSGlobalObject* exec, JSValue value)
52{
53 VM& vm = exec->vm();
54 if (value.inherits<JSNode>(vm))
55 return jsNontrivialString(vm, "node"_s);
56 if (value.inherits<JSNodeList>(vm))
57 return jsNontrivialString(vm, "array"_s);
58 if (value.inherits<JSHTMLCollection>(vm))
59 return jsNontrivialString(vm, "array"_s);
60 if (value.inherits<JSDOMException>(vm))
61 return jsNontrivialString(vm, "error"_s);
62
63 return jsUndefined();
64}

66static JSObject* constructInternalProperty(VM& vm, JSGlobalObject* exec, const String& name, JSValue value)
67{
68 auto* object = constructEmptyObject(exec);
69 object->putDirect(vm, Identifier::fromString(vm, "name"), jsString(vm, name));
70 object->putDirect(vm, Identifier::fromString(vm, "value"), value);
71 return object;
72}

113static JSObject* objectForPaymentDetailsModifier(VM& vm, JSGlobalObject* exec, const PaymentDetailsModifier& modifier)
114{
115 auto* object = constructEmptyObject(exec);
116 object->putDirect(vm, Identifier::fromString(vm, "supportedMethods"), jsString(vm, modifier.supportedMethods));
117 if (modifier.total)
118 object->putDirect(vm, Identifier::fromString(vm, "total"), objectForPaymentItem(vm, exec, *modifier.total));
119 if (!modifier.additionalDisplayItems.isEmpty()) {
120 auto* additionalDisplayItems = constructEmptyArray(exec, nullptr);
121 for (unsigned i = 0; i < modifier.additionalDisplayItems.size(); ++i)
122 additionalDisplayItems->putDirectIndex(exec, i, objectForPaymentItem(vm, exec, modifier.additionalDisplayItems[i]));
123 object->putDirect(vm, Identifier::fromString(vm, "additionalDisplayItems"), additionalDisplayItems);
124 }
125 return object;
126}

128static JSObject* objectForPaymentDetails(VM& vm, JSGlobalObject* exec, const PaymentDetailsInit& paymentDetails)
129{
130 auto* object = constructEmptyObject(exec);
131 object->putDirect(vm, Identifier::fromString(vm, "id"), jsString(vm, paymentDetails.id));
132 object->putDirect(vm, Identifier::fromString(vm, "total"), objectForPaymentItem(vm, exec, paymentDetails.total));
133 if (paymentDetails.displayItems) {
134 auto* displayItems = constructEmptyArray(exec, nullptr);
135 for (unsigned i = 0; i < paymentDetails.displayItems->size(); ++i)
136 displayItems->putDirectIndex(exec, i, objectForPaymentItem(vm, exec, paymentDetails.displayItems->at(i)));
137 object->putDirect(vm, Identifier::fromString(vm, "displayItems"), displayItems);
138 }
139 if (paymentDetails.shippingOptions) {
140 auto* shippingOptions = constructEmptyArray(exec, nullptr);
141 for (unsigned i = 0; i < paymentDetails.shippingOptions->size(); ++i)
142 shippingOptions->putDirectIndex(exec, i, objectForPaymentShippingOption(vm, exec, paymentDetails.shippingOptions->at(i)));
143 object->putDirect(vm, Identifier::fromString(vm, "shippingOptions"), shippingOptions);
144 }
145 if (paymentDetails.modifiers) {
146 auto* modifiers = constructEmptyArray(exec, nullptr);
147 for (unsigned i = 0; i < paymentDetails.modifiers->size(); ++i)
148 modifiers->putDirectIndex(exec, i, objectForPaymentDetailsModifier(vm, exec, paymentDetails.modifiers->at(i)));
149 object->putDirect(vm, Identifier::fromString(vm, "modifiers"), modifiers);
150 }
151 return object;
152}

154static JSString* jsStringForPaymentRequestState(VM& vm, PaymentRequest::State state)
155{
156 switch (state) {
157 case PaymentRequest::State::Created:
158 return jsNontrivialString(vm, "created"_s);
159 case PaymentRequest::State::Interactive:
160 return jsNontrivialString(vm, "interactive"_s);
161 case PaymentRequest::State::Closed:
162 return jsNontrivialString(vm, "closed"_s);
163 }
164
165 ASSERT_NOT_REACHED();
166 return jsEmptyString(vm);
167}

170static JSObject* objectForEventTargetListeners(VM& vm, JSGlobalObject* exec, EventTarget* eventTarget)
171{
172 auto* scriptExecutionContext = eventTarget->scriptExecutionContext();
173 if (!scriptExecutionContext)
174 return nullptr;
175
176 JSObject* listeners = nullptr;
177
178 for (auto& eventType : eventTarget->eventTypes()) {
179 unsigned listenersForEventIndex = 0;
180 auto* listenersForEvent = constructEmptyArray(exec, nullptr);
181
182 for (auto& eventListener : eventTarget->eventListeners(eventType)) {
183 if (!is<JSEventListener>(eventListener->callback()))
184 continue;
185
186 auto& jsListener = downcast<JSEventListener>(eventListener->callback());
187 if (&jsListener.isolatedWorld() != &currentWorld(*exec))
188 continue;
189
190 auto* jsFunction = jsListener.ensureJSFunction(*scriptExecutionContext);
191 if (!jsFunction)
192 continue;
193
194 auto* propertiesForListener = constructEmptyObject(exec);
195 propertiesForListener->putDirect(vm, Identifier::fromString(vm, "callback"), jsFunction);
196 propertiesForListener->putDirect(vm, Identifier::fromString(vm, "capture"), jsBoolean(eventListener->useCapture()));
197 propertiesForListener->putDirect(vm, Identifier::fromString(vm, "passive"), jsBoolean(eventListener->isPassive()));
198 propertiesForListener->putDirect(vm, Identifier::fromString(vm, "once"), jsBoolean(eventListener->isOnce()));
199 listenersForEvent->putDirectIndex(exec, listenersForEventIndex++, propertiesForListener);
200 }
201
202 if (listenersForEventIndex) {
203 if (!listeners)
204 listeners = constructEmptyObject(exec);
205 listeners->putDirect(vm, Identifier::fromString(vm, eventType), listenersForEvent);
206 }
207 }
208
209 return listeners;
210}

212JSValue WebInjectedScriptHost::getInternalProperties(VM& vm, JSGlobalObject* exec, JSC::JSValue value)
213{
214 auto scope = DECLARE_THROW_SCOPE(vm);
215
216 if (auto* worker = JSWorker::toWrapped(vm, value)) {
217 unsigned index = 0;
218 auto* array = constructEmptyArray(exec, nullptr);
219
220 String name = worker->name();
221 if (!name.isEmpty())
222 array->putDirectIndex(exec, index++, constructInternalProperty(vm, exec, "name"_s, jsString(vm, name)));
223
224 array->putDirectIndex(exec, index++, constructInternalProperty(vm, exec, "terminated"_s, jsBoolean(worker->wasTerminated())));
225
226 if (auto* listeners = objectForEventTargetListeners(vm, exec, worker))
227 array->putDirectIndex(exec, index++, constructInternalProperty(vm, exec, "listeners"_s, listeners));
228
229 RETURN_IF_EXCEPTION(scope, { });
230 return array;
231 }
232
233#if ENABLE(PAYMENT_REQUEST)
234 if (PaymentRequest* paymentRequest = JSPaymentRequest::toWrapped(vm, value)) {
235 unsigned index = 0;
236 auto* array = constructEmptyArray(exec, nullptr);
237
238 array->putDirectIndex(exec, index++, constructInternalProperty(vm, exec, "options"_s, objectForPaymentOptions(vm, exec, paymentRequest->paymentOptions())));
239 array->putDirectIndex(exec, index++, constructInternalProperty(vm, exec, "details"_s, objectForPaymentDetails(vm, exec, paymentRequest->paymentDetails())));
240 array->putDirectIndex(exec, index++, constructInternalProperty(vm, exec, "state"_s, jsStringForPaymentRequestState(vm, paymentRequest->state())));
241
242 if (auto* listeners = objectForEventTargetListeners(vm, exec, paymentRequest))
243 array->putDirectIndex(exec, index++, constructInternalProperty(vm, exec, "listeners"_s, listeners));
244
245 RETURN_IF_EXCEPTION(scope, { });
246 return array;
247 }
248#endif
249
250 if (auto* eventTarget = JSEventTarget::toWrapped(vm, value)) {
251 unsigned index = 0;
252 auto* array = constructEmptyArray(exec, nullptr);
253
254 if (auto* listeners = objectForEventTargetListeners(vm, exec, eventTarget))
255 array->putDirectIndex(exec, index++, constructInternalProperty(vm, exec, "listeners"_s, listeners));
256
257 RETURN_IF_EXCEPTION(scope, { });
258 return array;
259 }
260
261 return { };
262}
