67static std::optional<double> protocolValueForSeconds(const Seconds& seconds)
68{
69 if (seconds == Seconds::infinity() || seconds == Seconds::nan())
70 return std::nullopt;
71 return seconds.milliseconds();
72}

74static std::optional<Protocol::Animation::PlaybackDirection> protocolValueForPlaybackDirection(PlaybackDirection playbackDirection)
75{
76 switch (playbackDirection) {
77 case PlaybackDirection::Normal:
78 return Protocol::Animation::PlaybackDirection::Normal;
79 case PlaybackDirection::Reverse:
80 return Protocol::Animation::PlaybackDirection::Reverse;
81 case PlaybackDirection::Alternate:
82 return Protocol::Animation::PlaybackDirection::Alternate;
83 case PlaybackDirection::AlternateReverse:
84 return Protocol::Animation::PlaybackDirection::AlternateReverse;
85 }
86
87 ASSERT_NOT_REACHED();
88 return std::nullopt;

110static Ref<JSON::ArrayOf<Protocol::Animation::Keyframe>> buildObjectForKeyframes(KeyframeEffect& keyframeEffect)
111{
112 auto keyframesPayload = JSON::ArrayOf<Protocol::Animation::Keyframe>::create();
113
114 const auto& blendingKeyframes = keyframeEffect.blendingKeyframes();
115 const auto& parsedKeyframes = keyframeEffect.parsedKeyframes();
116
117 if (is<DeclarativeAnimation>(keyframeEffect.animation())) {
118 auto& declarativeAnimation = downcast<DeclarativeAnimation>(*keyframeEffect.animation());
119
120 auto* target = keyframeEffect.target();
121 auto* renderer = keyframeEffect.renderer();
122
123 // Synthesize CSS style declarations for each keyframe so the frontend can display them.
124 ComputedStyleExtractor computedStyleExtractor(target, false, target->pseudoId());
125
126 for (size_t i = 0; i < blendingKeyframes.size(); ++i) {
127 auto& blendingKeyframe = blendingKeyframes[i];
128
129 ASSERT(blendingKeyframe.style());
130 auto& style = *blendingKeyframe.style();
131
132 auto keyframePayload = Protocol::Animation::Keyframe::create()
133 .setOffset(blendingKeyframe.key())
134 .release();
135
136 RefPtr<TimingFunction> timingFunction;
137 if (!parsedKeyframes.isEmpty())
138 timingFunction = parsedKeyframes[i].timingFunction;
139 if (!timingFunction)
140 timingFunction = blendingKeyframe.timingFunction();
141 if (!timingFunction)
142 timingFunction = declarativeAnimation.backingAnimation().timingFunction();
143 if (timingFunction)
144 keyframePayload->setEasing(timingFunction->cssText());
145
146 StringBuilder stylePayloadBuilder;
147 auto& cssPropertyIds = blendingKeyframe.properties();
148 size_t count = cssPropertyIds.size();
149 for (auto cssPropertyId : cssPropertyIds) {
150 --count;
151 if (cssPropertyId == CSSPropertyCustom)
152 continue;
153
154 stylePayloadBuilder.append(getPropertyNameString(cssPropertyId));
155 stylePayloadBuilder.append(": ");
156 if (auto value = computedStyleExtractor.valueForPropertyInStyle(style, cssPropertyId, renderer))
157 stylePayloadBuilder.append(value->cssText());
158 stylePayloadBuilder.append(';');
159 if (count > 0)
160 stylePayloadBuilder.append(' ');
161 }
162 if (!stylePayloadBuilder.isEmpty())
163 keyframePayload->setStyle(stylePayloadBuilder.toString());
164
165 keyframesPayload->addItem(WTFMove(keyframePayload));
166 }
167 } else {
168 for (const auto& parsedKeyframe : parsedKeyframes) {
169 auto keyframePayload = Protocol::Animation::Keyframe::create()
170 .setOffset(parsedKeyframe.computedOffset)
171 .release();
172
173 if (!parsedKeyframe.easing.isEmpty())
174 keyframePayload->setEasing(parsedKeyframe.easing);
175 else if (const auto& timingFunction = parsedKeyframe.timingFunction)
176 keyframePayload->setEasing(timingFunction->cssText());
177
178 if (!parsedKeyframe.style->isEmpty())
179 keyframePayload->setStyle(parsedKeyframe.style->asText());
180
181 keyframesPayload->addItem(WTFMove(keyframePayload));
182 }
183 }
184
185 return keyframesPayload;
186}

188static Ref<Protocol::Animation::Effect> buildObjectForEffect(AnimationEffect& effect)
189{
190 auto effectPayload = Protocol::Animation::Effect::create()
191 .release();
192
193 if (auto startDelay = protocolValueForSeconds(effect.delay()))
194 effectPayload->setStartDelay(startDelay.value());
195
196 if (auto endDelay = protocolValueForSeconds(effect.endDelay()))
197 effectPayload->setEndDelay(endDelay.value());
198
199 effectPayload->setIterationCount(effect.iterations() == std::numeric_limits<double>::infinity() ? -1 : effect.iterations());
200 effectPayload->setIterationStart(effect.iterationStart());
201
202 if (auto iterationDuration = protocolValueForSeconds(effect.iterationDuration()))
203 effectPayload->setIterationDuration(iterationDuration.value());
204
205 if (auto* timingFunction = effect.timingFunction())
206 effectPayload->setTimingFunction(timingFunction->cssText());
207
208 if (auto playbackDirection = protocolValueForPlaybackDirection(effect.direction()))
209 effectPayload->setPlaybackDirection(playbackDirection.value());
210
211 if (auto fillMode = protocolValueForFillMode(effect.fill()))
212 effectPayload->setFillMode(fillMode.value());
213
214 if (is<KeyframeEffect>(effect))
215 effectPayload->setKeyframes(buildObjectForKeyframes(downcast<KeyframeEffect>(effect)));
216
217 return effectPayload;
218}

247Protocol::ErrorStringOr<void> InspectorAnimationAgent::enable()
248{
249 if (m_instrumentingAgents.enabledAnimationAgent() == this)
250 return makeUnexpected("Animation domain already enabled"_s);
251
252 m_instrumentingAgents.setEnabledAnimationAgent(this);
253
254 const auto existsInCurrentPage = [&] (ScriptExecutionContext* scriptExecutionContext) {
255 if (!is<Document>(scriptExecutionContext))
256 return false;
257
258 // FIXME: <https://webkit.org/b/168475> Web Inspector: Correctly display iframe's WebSockets
259 auto* document = downcast<Document>(scriptExecutionContext);
260 return document->page() == &m_inspectedPage;
261 };
262
263 {
264 for (auto* animation : WebAnimation::instances()) {
265 if (existsInCurrentPage(animation->scriptExecutionContext()))
266 bindAnimation(*animation, false);
267 }
268 }
269
270 return { };
271}

279 return { };

282Protocol::ErrorStringOr<Protocol::DOM::NodeId> InspectorAnimationAgent::requestEffectTarget(const Protocol::Animation::AnimationId& animationId)
283{
284 Protocol::ErrorString errorString;
285
286 auto* animation = assertAnimation(errorString, animationId);
287 if (!animation)
288 return makeUnexpected(errorString);
289
290 auto* domAgent = m_instrumentingAgents.persistentDOMAgent();
291 if (!domAgent)
292 return makeUnexpected("DOM domain must be enabled"_s);
293
294 auto* effect = animation->effect();
295 if (!is<KeyframeEffect>(effect))
296 return makeUnexpected("Animation for given animationId does not have an effect"_s);
297
298 auto& keyframeEffect = downcast<KeyframeEffect>(*effect);
299
300 auto* target = keyframeEffect.targetElementOrPseudoElement();
301 if (!target)
302 return makeUnexpected("Animation for given animationId does not have a target"_s);
303
304 return domAgent->pushNodePathToFrontend(errorString, target);
305}

307Protocol::ErrorStringOr<Ref<Protocol::Runtime::RemoteObject>> InspectorAnimationAgent::resolveAnimation(const Protocol::Animation::AnimationId& animationId, const String& objectGroup)
308{
309 Protocol::ErrorString errorString;
310
311 auto* animation = assertAnimation(errorString, animationId);
312 if (!animation)
313 return makeUnexpected(errorString);
314
315 auto* state = animation->scriptExecutionContext()->globalObject();
316 auto injectedScript = m_injectedScriptManager.injectedScriptFor(state);
317 ASSERT(!injectedScript.hasNoValue());
318
319 JSC::JSValue value;
320 {
321 JSC::JSLockHolder lock(state);
322
323 auto* globalObject = deprecatedGlobalObjectForPrototype(state);
324 value = toJS(state, globalObject, animation);
325 }
326
327 if (!value) {
328 ASSERT_NOT_REACHED();
329 return makeUnexpected("Internal error: unknown Animation for given animationId"_s);
330 }
331
332 auto object = injectedScript.wrapObject(value, objectGroup);
333 if (!object)
334 return makeUnexpected("Internal error: unable to cast Animation");
335
336 return object.releaseNonNull();
337}

367static bool isDelayed(ComputedEffectTiming& computedTiming)
368{
369 if (!computedTiming.localTime)
370 return false;
371 return computedTiming.localTime.value() < (computedTiming.endTime - computedTiming.activeDuration);
372}

374void InspectorAnimationAgent::willApplyKeyframeEffect(Element& target, KeyframeEffect& keyframeEffect, ComputedEffectTiming computedTiming)
375{
376 auto* animation = keyframeEffect.animation();
377 if (!is<DeclarativeAnimation>(animation))
378 return;
379
380 auto ensureResult = m_trackedDeclarativeAnimationData.ensure(downcast<DeclarativeAnimation>(animation), [&] () -> UniqueRef<TrackedDeclarativeAnimationData> {
381 return makeUniqueRef<TrackedDeclarativeAnimationData>(TrackedDeclarativeAnimationData { makeString("animation:"_s, IdentifiersFactory::createIdentifier()), computedTiming });
382 });
383 auto& trackingData = ensureResult.iterator->value.get();
384
385 std::optional<Protocol::Animation::AnimationState> animationAnimationState;
386
387 if ((ensureResult.isNewEntry || !isDelayed(trackingData.lastComputedTiming)) && isDelayed(computedTiming))
388 animationAnimationState = Protocol::Animation::AnimationState::Delayed;
389 else if (ensureResult.isNewEntry || trackingData.lastComputedTiming.phase != computedTiming.phase) {
390 switch (computedTiming.phase) {
391 case AnimationEffectPhase::Before:
392 animationAnimationState = Protocol::Animation::AnimationState::Ready;
393 break;
394
395 case AnimationEffectPhase::Active:
396 animationAnimationState = Protocol::Animation::AnimationState::Active;
397 break;
398
399 case AnimationEffectPhase::After:
400 animationAnimationState = Protocol::Animation::AnimationState::Done;
401 break;
402
403 case AnimationEffectPhase::Idle:
404 animationAnimationState = Protocol::Animation::AnimationState::Canceled;
405 break;
406 }
407 } else if (trackingData.lastComputedTiming.currentIteration != computedTiming.currentIteration) {
408 // Iterations are represented by sequential "active" state events.
409 animationAnimationState = Protocol::Animation::AnimationState::Active;
410 }
411
412 trackingData.lastComputedTiming = computedTiming;
413
414 if (!animationAnimationState)
415 return;
416
417 auto event = Protocol::Animation::TrackingUpdate::create()
418 .setTrackingAnimationId(trackingData.trackingAnimationId)
419 .setAnimationState(animationAnimationState.value())
420 .release();
421
422 if (ensureResult.isNewEntry) {
423 if (auto* domAgent = m_instrumentingAgents.persistentDOMAgent()) {
424 if (auto nodeId = domAgent->pushNodeToFrontend(&target))
425 event->setNodeId(nodeId);
426 }
427
428 if (is<CSSAnimation>(animation))
429 event->setAnimationName(downcast<CSSAnimation>(*animation).animationName());
430 else if (is<CSSTransition>(animation))
431 event->setTransitionProperty(downcast<CSSTransition>(*animation).transitionProperty());
432 else
433 ASSERT_NOT_REACHED();
434 }
435
436 m_frontendDispatcher->trackingUpdate(m_environment.executionStopwatch().elapsedTime().seconds(), WTFMove(event));
437}

502void InspectorAnimationAgent::frameNavigated(Frame& frame)
503{
504 if (frame.isMainFrame()) {
505 reset();
506 return;
507 }
508
509 Vector<String> animationIdsToRemove;
510 for (auto& [animationId, animation] : m_animationIdMap) {
511 if (auto* scriptExecutionContext = animation->scriptExecutionContext()) {
512 if (is<Document>(scriptExecutionContext) && downcast<Document>(*scriptExecutionContext).frame() == &frame)
513 animationIdsToRemove.append(animationId);
514 }
515 }
516 for (const auto& animationId : animationIdsToRemove)
517 unbindAnimation(animationId);
518}

520String InspectorAnimationAgent::findAnimationId(WebAnimation& animation)
521{
522 for (auto& [animationId, existingAnimation] : m_animationIdMap) {
523 if (existingAnimation == &animation)
524 return animationId;
525 }
526 return nullString();

537void InspectorAnimationAgent::bindAnimation(WebAnimation& animation, bool captureBacktrace)
538{
539 auto animationId = makeString("animation:" + IdentifiersFactory::createIdentifier());
540 m_animationIdMap.set(animationId, &animation);
541
542 auto animationPayload = Protocol::Animation::Animation::create()
543 .setAnimationId(animationId)
544 .release();
545
546 auto name = animation.id();
547 if (!name.isEmpty())
548 animationPayload->setName(name);
549
550 if (is<CSSAnimation>(animation))
551 animationPayload->setCssAnimationName(downcast<CSSAnimation>(animation).animationName());
552 else if (is<CSSTransition>(animation))
553 animationPayload->setCssTransitionProperty(downcast<CSSTransition>(animation).transitionProperty());
554
555 if (auto* effect = animation.effect())
556 animationPayload->setEffect(buildObjectForEffect(*effect));
557
558 if (captureBacktrace) {
559 auto stackTrace = Inspector::createScriptCallStack(JSExecState::currentState(), Inspector::ScriptCallStack::maxCallStackSizeToCapture);
560 animationPayload->setBacktrace(stackTrace->buildInspectorArray());
561 }
562
563 m_frontendDispatcher->animationCreated(WTFMove(animationPayload));
564}
565
566void InspectorAnimationAgent::unbindAnimation(const String& animationId)
567{
568 m_animationIdMap.remove(animationId);
569
570 // This can be called in response to GC. Due to the single-process model used in WebKit1, the
571 // event must be dispatched from a timer to prevent the frontend from making JS allocations
572 // while the GC is still active.
573 m_removedAnimationIds.append(animationId);
574
575 if (!m_animationDestroyedTimer.isActive())
576 m_animationDestroyedTimer.startOneShot(0_s);
577}
578
579void InspectorAnimationAgent::animationDestroyedTimerFired()
580{
581 if (!m_removedAnimationIds.size())
582 return;
583
584 for (auto& identifier : m_removedAnimationIds)
585 m_frontendDispatcher->animationDestroyed(identifier);
586
587 m_removedAnimationIds.clear();
588}
589
590void InspectorAnimationAgent::reset()
591{
592 m_animationIdMap.clear();
593
594 m_removedAnimationIds.clear();
595
596 if (m_animationDestroyedTimer.isActive())
597 m_animationDestroyedTimer.stop();
598}

600void InspectorAnimationAgent::stopTrackingDeclarativeAnimation(DeclarativeAnimation& animation)
601{
602 auto data = m_trackedDeclarativeAnimationData.take(&animation);
603 if (!data)
604 return;
605
606 if (data->lastComputedTiming.phase != AnimationEffectPhase::After && data->lastComputedTiming.phase != AnimationEffectPhase::Idle) {
607 auto event = Protocol::Animation::TrackingUpdate::create()
608 .setTrackingAnimationId(data->trackingAnimationId)
609 .setAnimationState(Protocol::Animation::AnimationState::Canceled)
610 .release();
611 m_frontendDispatcher->trackingUpdate(m_environment.executionStopwatch().elapsedTime().seconds(), WTFMove(event));
612 }
