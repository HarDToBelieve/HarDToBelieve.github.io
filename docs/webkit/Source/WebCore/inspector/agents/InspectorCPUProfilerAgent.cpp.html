40InspectorCPUProfilerAgent::InspectorCPUProfilerAgent(PageAgentContext& context)
41 : InspectorAgentBase("CPUProfiler"_s, context)
42 , m_frontendDispatcher(makeUnique<Inspector::CPUProfilerFrontendDispatcher>(context.frontendRouter))
43 , m_backendDispatcher(Inspector::CPUProfilerBackendDispatcher::create(context.backendDispatcher, this))

49void InspectorCPUProfilerAgent::didCreateFrontendAndBackend(FrontendRouter*, BackendDispatcher*)
50{
51 m_instrumentingAgents.setPersistentCPUProfilerAgent(this);

54void InspectorCPUProfilerAgent::willDestroyFrontendAndBackend(DisconnectReason)
55{
56 stopTracking();
57
58 m_instrumentingAgents.setPersistentCPUProfilerAgent(nullptr);
59}

61Protocol::ErrorStringOr<void> InspectorCPUProfilerAgent::startTracking()
62{
63 if (m_tracking)
64 return { };
65
66 ResourceUsageThread::addObserver(this, CPU, [this] (const ResourceUsageData& data) {
67 collectSample(data);
68 });
69
70 m_tracking = true;
71
72 m_frontendDispatcher->trackingStart(m_environment.executionStopwatch().elapsedTime().seconds());
73
74 return { };
75}
76
77Protocol::ErrorStringOr<void> InspectorCPUProfilerAgent::stopTracking()
78{
79 if (!m_tracking)
80 return { };
81
82 ResourceUsageThread::removeObserver(this);
83
84 m_tracking = false;
85
86 m_frontendDispatcher->trackingComplete(m_environment.executionStopwatch().elapsedTime().seconds());
87
88 return { };
89}

91static Ref<Protocol::CPUProfiler::ThreadInfo> buildThreadInfo(const ThreadCPUInfo& thread)
92{
93 ASSERT(thread.cpu <= 100);
94
95 auto threadInfo = Protocol::CPUProfiler::ThreadInfo::create()
96 .setName(thread.name)
97 .setUsage(thread.cpu)
98 .release();
99
100 if (thread.type == ThreadCPUInfo::Type::Main)
101 threadInfo->setType(Protocol::CPUProfiler::ThreadInfo::Type::Main);
102 else if (thread.type == ThreadCPUInfo::Type::WebKit)
103 threadInfo->setType(Protocol::CPUProfiler::ThreadInfo::Type::WebKit);
104
105 if (!thread.identifier.isEmpty())
106 threadInfo->setTargetId(thread.identifier);
107
108 return threadInfo;
109}
110
111void InspectorCPUProfilerAgent::collectSample(const ResourceUsageData& data)
112{
113 auto event = Protocol::CPUProfiler::Event::create()
114 .setTimestamp(m_environment.executionStopwatch().elapsedTimeSince(data.timestamp).seconds())
115 .setUsage(data.cpuExcludingDebuggerThreads)
116 .release();
117
118 if (!data.cpuThreads.isEmpty()) {
119 auto threads = JSON::ArrayOf<Protocol::CPUProfiler::ThreadInfo>::create();
120 for (auto& threadInfo : data.cpuThreads)
121 threads->addItem(buildThreadInfo(threadInfo));
122 event->setThreads(WTFMove(threads));
123 }
124
125 m_frontendDispatcher->trackingUpdate(WTFMove(event));
126}
