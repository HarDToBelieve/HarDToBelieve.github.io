81enum ForcePseudoClassFlags {
82 PseudoClassNone = 0,
83 PseudoClassHover = 1 << 0,
84 PseudoClassFocus = 1 << 1,
85 PseudoClassActive = 1 << 2,
86 PseudoClassVisited = 1 << 3
87};
88
89static unsigned computePseudoClassMask(const JSON::Array& pseudoClassArray)
90{
91 static NeverDestroyed<String> active(MAKE_STATIC_STRING_IMPL("active"));
92 static NeverDestroyed<String> hover(MAKE_STATIC_STRING_IMPL("hover"));
93 static NeverDestroyed<String> focus(MAKE_STATIC_STRING_IMPL("focus"));
94 static NeverDestroyed<String> visited(MAKE_STATIC_STRING_IMPL("visited"));
95 if (!pseudoClassArray.length())
96 return PseudoClassNone;
97
98 unsigned result = PseudoClassNone;
99 for (auto& pseudoClassValue : pseudoClassArray) {
100 auto pseudoClass = pseudoClassValue->asString();
101 if (pseudoClass == active)
102 result |= PseudoClassActive;
103 else if (pseudoClass == hover)
104 result |= PseudoClassHover;
105 else if (pseudoClass == focus)
106 result |= PseudoClassFocus;
107 else if (pseudoClass == visited)
108 result |= PseudoClassVisited;
109 }
110
111 return result;
112}

127class InspectorCSSAgent::SetStyleSheetTextAction final : public InspectorCSSAgent::StyleSheetAction {
128 WTF_MAKE_NONCOPYABLE(SetStyleSheetTextAction);
129public:

131 : InspectorCSSAgent::StyleSheetAction(styleSheet)
132 , m_text(text)
133 {
134 }
135
136private:
137 ExceptionOr<void> perform() final
138 {
139 auto result = m_styleSheet->text();
140 if (result.hasException())
141 return result.releaseException();
142 m_oldText = result.releaseReturnValue();
143 return redo();
144 }
145
146 ExceptionOr<void> undo() final
147 {
148 auto result = m_styleSheet->setText(m_oldText);
149 if (result.hasException())
150 return result.releaseException();
151 m_styleSheet->reparseStyleSheet(m_oldText);
152 return { };
153 }
154
155 ExceptionOr<void> redo() final
156 {
157 auto result = m_styleSheet->setText(m_text);
158 if (result.hasException())
159 return result.releaseException();
160 m_styleSheet->reparseStyleSheet(m_text);
161 return { };
162 }
163
164 String mergeId() final
165 {
166 return "SetStyleSheetText " + m_styleSheet->id();
167 }
168
169 void merge(std::unique_ptr<Action> action) override
170 {
171 ASSERT(action->mergeId() == mergeId());
172 m_text = static_cast<SetStyleSheetTextAction&>(*action).m_text;
173 }
174
175 String m_text;
176 String m_oldText;
177};

336Protocol::ErrorStringOr<void> InspectorCSSAgent::enable()
337{
338 if (m_instrumentingAgents.enabledCSSAgent() == this)
339 return { };
340
341 m_instrumentingAgents.setEnabledCSSAgent(this);
342
343 if (auto* domAgent = m_instrumentingAgents.persistentDOMAgent()) {
344 for (auto* document : domAgent->documents())

348 return { };

383void InspectorCSSAgent::setActiveStyleSheetsForDocument(Document& document, Vector<CSSStyleSheet*>& activeStyleSheets)
384{
385 HashSet<CSSStyleSheet*>& previouslyKnownActiveStyleSheets = m_documentToKnownCSSStyleSheets.add(&document, HashSet<CSSStyleSheet*>()).iterator->value;
386
387 HashSet<CSSStyleSheet*> removedStyleSheets(previouslyKnownActiveStyleSheets);
388 Vector<CSSStyleSheet*> addedStyleSheets;
389 for (auto& activeStyleSheet : activeStyleSheets) {
390 if (removedStyleSheets.contains(activeStyleSheet))
391 removedStyleSheets.remove(activeStyleSheet);
392 else
393 addedStyleSheets.append(activeStyleSheet);
394 }
395
396 for (auto* cssStyleSheet : removedStyleSheets) {
397 previouslyKnownActiveStyleSheets.remove(cssStyleSheet);
398 RefPtr<InspectorStyleSheet> inspectorStyleSheet = m_cssStyleSheetToInspectorStyleSheet.get(cssStyleSheet);
399 if (m_idToInspectorStyleSheet.contains(inspectorStyleSheet->id())) {
400 auto id = unbindStyleSheet(inspectorStyleSheet.get());
401 m_frontendDispatcher->styleSheetRemoved(id);
402 }
403 }
404
405 for (auto* cssStyleSheet : addedStyleSheets) {
406 previouslyKnownActiveStyleSheets.add(cssStyleSheet);
407 if (!m_cssStyleSheetToInspectorStyleSheet.contains(cssStyleSheet)) {
408 InspectorStyleSheet* inspectorStyleSheet = bindStyleSheet(cssStyleSheet);
409 if (auto header = inspectorStyleSheet->buildObjectForStyleSheetInfo())
410 m_frontendDispatcher->styleSheetAdded(header.releaseNonNull());
411 }
412 }

415bool InspectorCSSAgent::forcePseudoState(const Element& element, CSSSelector::PseudoClassType pseudoClassType)
416{
417 if (m_nodeIdToForcedPseudoState.isEmpty())
418 return false;
419
420 auto* domAgent = m_instrumentingAgents.persistentDOMAgent();
421 if (!domAgent)
422 return false;
423
424 auto nodeId = domAgent->boundNodeId(&element);
425 if (!nodeId)
426 return false;
427
428 unsigned forcedPseudoState = m_nodeIdToForcedPseudoState.get(nodeId);
429 switch (pseudoClassType) {
430 case CSSSelector::PseudoClassActive:
431 return forcedPseudoState & PseudoClassActive;
432 case CSSSelector::PseudoClassFocus:
433 return forcedPseudoState & PseudoClassFocus;
434 case CSSSelector::PseudoClassHover:
435 return forcedPseudoState & PseudoClassHover;
436 case CSSSelector::PseudoClassVisited:
437 return forcedPseudoState & PseudoClassVisited;
438 default:
439 return false;
440 }
441}

443static std::optional<Protocol::CSS::PseudoId> protocolValueForPseudoId(PseudoId pseudoId)
444{
445 switch (pseudoId) {
446 case PseudoId::FirstLine:
447 return Protocol::CSS::PseudoId::FirstLine;
448 case PseudoId::FirstLetter:
449 return Protocol::CSS::PseudoId::FirstLetter;
450 case PseudoId::Marker:
451 return Protocol::CSS::PseudoId::Marker;
452 case PseudoId::Before:
453 return Protocol::CSS::PseudoId::Before;
454 case PseudoId::After:
455 return Protocol::CSS::PseudoId::After;
456 case PseudoId::Selection:
457 return Protocol::CSS::PseudoId::Selection;
458 case PseudoId::Highlight:
459 return Protocol::CSS::PseudoId::Highlight;
460 case PseudoId::Scrollbar:
461 return Protocol::CSS::PseudoId::Scrollbar;
462 case PseudoId::ScrollbarThumb:
463 return Protocol::CSS::PseudoId::ScrollbarThumb;
464 case PseudoId::ScrollbarButton:
465 return Protocol::CSS::PseudoId::ScrollbarButton;
466 case PseudoId::ScrollbarTrack:
467 return Protocol::CSS::PseudoId::ScrollbarTrack;
468 case PseudoId::ScrollbarTrackPiece:
469 return Protocol::CSS::PseudoId::ScrollbarTrackPiece;
470 case PseudoId::ScrollbarCorner:
471 return Protocol::CSS::PseudoId::ScrollbarCorner;
472 case PseudoId::Resizer:
473 return Protocol::CSS::PseudoId::Resizer;
474
475 default:
476 ASSERT_NOT_REACHED();
477 return { };
478 }
479}

481Protocol::ErrorStringOr<std::tuple<RefPtr<JSON::ArrayOf<Protocol::CSS::RuleMatch>>, RefPtr<JSON::ArrayOf<Protocol::CSS::PseudoIdMatches>>, RefPtr<JSON::ArrayOf<Protocol::CSS::InheritedStyleEntry>>>> InspectorCSSAgent::getMatchedStylesForNode(Protocol::DOM::NodeId nodeId, std::optional<bool>&& includePseudo, std::optional<bool>&& includeInherited)
482{
483 Protocol::ErrorString errorString;
484
485 Element* element = elementForId(errorString, nodeId);
486 if (!element)
487 return makeUnexpected(errorString);
488
489 Element* originalElement = element;
490 PseudoId elementPseudoId = element->pseudoId();
491 if (elementPseudoId != PseudoId::None) {
492 element = downcast<PseudoElement>(*element).hostElement();
493 if (!element)
494 return makeUnexpected("Missing parent of pseudo-element node for given nodeId"_s);
495 }
496
497 // Matched rules.
498 auto& styleResolver = element->styleResolver();
499 auto matchedRules = styleResolver.pseudoStyleRulesForElement(element, elementPseudoId, Style::Resolver::AllCSSRules);
500 auto matchedCSSRules = buildArrayForMatchedRuleList(matchedRules, styleResolver, *element, elementPseudoId);
501 RefPtr<JSON::ArrayOf<Protocol::CSS::PseudoIdMatches>> pseudoElements;
502 RefPtr<JSON::ArrayOf<Protocol::CSS::InheritedStyleEntry>> inherited;
503
504 if (!originalElement->isPseudoElement()) {
505 if (!includePseudo || *includePseudo) {
506 pseudoElements = JSON::ArrayOf<Protocol::CSS::PseudoIdMatches>::create();
507 for (PseudoId pseudoId = PseudoId::FirstPublicPseudoId; pseudoId < PseudoId::AfterLastInternalPseudoId; pseudoId = static_cast<PseudoId>(static_cast<unsigned>(pseudoId) + 1)) {
508 // `*::marker` selectors are only applicable to elements with `display: list-item`.
509 if (pseudoId == PseudoId::Marker && element->computedStyle()->display() != DisplayType::ListItem)
510 continue;
511 if (auto protocolPseudoId = protocolValueForPseudoId(pseudoId)) {
512 auto matchedRules = styleResolver.pseudoStyleRulesForElement(element, pseudoId, Style::Resolver::AllCSSRules);
513 if (!matchedRules.isEmpty()) {
514 auto matches = Protocol::CSS::PseudoIdMatches::create()
515 .setPseudoId(protocolPseudoId.value())
516 .setMatches(buildArrayForMatchedRuleList(matchedRules, styleResolver, *element, pseudoId))
517 .release();
518 pseudoElements->addItem(WTFMove(matches));
519 }
520 }
521 }
522 }
523
524 if (!includeInherited || *includeInherited) {
525 inherited = JSON::ArrayOf<Protocol::CSS::InheritedStyleEntry>::create();
526 for (auto& ancestor : ancestorsOfType<Element>(*element)) {
527 auto& parentStyleResolver = ancestor.styleResolver();
528 auto parentMatchedRules = parentStyleResolver.styleRulesForElement(&ancestor, Style::Resolver::AllCSSRules);
529 auto entry = Protocol::CSS::InheritedStyleEntry::create()
530 .setMatchedCSSRules(buildArrayForMatchedRuleList(parentMatchedRules, styleResolver, ancestor, PseudoId::None))
531 .release();
532 if (is<StyledElement>(ancestor) && downcast<StyledElement>(ancestor).cssomStyle().length()) {
533 auto& styleSheet = asInspectorStyleSheet(downcast<StyledElement>(ancestor));
534 entry->setInlineStyle(styleSheet.buildObjectForStyle(styleSheet.styleForId(InspectorCSSId(styleSheet.id(), 0))));
535 }
536 inherited->addItem(WTFMove(entry));
537 }
538 }
539 }
540
541 return { { WTFMove(matchedCSSRules), WTFMove(pseudoElements), WTFMove(inherited) } };
542}

573static Ref<Protocol::CSS::Font> buildObjectForFont(const Font& font)
574{
575 auto resultVariationAxes = JSON::ArrayOf<Protocol::CSS::FontVariationAxis>::create();
576 for (auto& variationAxis : font.platformData().variationAxes()) {
577 auto axis = Protocol::CSS::FontVariationAxis::create()
578 .setTag(variationAxis.tag())
579 .setMinimumValue(variationAxis.minimumValue())
580 .setMaximumValue(variationAxis.maximumValue())
581 .setDefaultValue(variationAxis.defaultValue())
582 .release();
583
584 if (variationAxis.name().length() && variationAxis.name() != variationAxis.tag())
585 axis->setName(variationAxis.name());
586
587 resultVariationAxes->addItem(WTFMove(axis));
588 }
589
590 return Protocol::CSS::Font::create()
591 .setDisplayName(font.platformData().familyName())
592 .setVariationAxes(WTFMove(resultVariationAxes))
593 .release();
594}

598 Protocol::ErrorString errorString;

643void InspectorCSSAgent::collectStyleSheets(CSSStyleSheet* styleSheet, Vector<CSSStyleSheet*>& result)
644{
645 result.append(styleSheet);
646
647 for (unsigned i = 0, size = styleSheet->length(); i < size; ++i) {
648 CSSRule* rule = styleSheet->item(i);
649 if (is<CSSImportRule>(*rule)) {
650 if (CSSStyleSheet* importedStyleSheet = downcast<CSSImportRule>(*rule).styleSheet())
651 collectStyleSheets(importedStyleSheet, result);
652 }
653 }

686Protocol::ErrorStringOr<void> InspectorCSSAgent::setStyleSheetText(const Protocol::CSS::StyleSheetId& styleSheetId, const String& text)
687{
688 Protocol::ErrorString errorString;
689
690 InspectorStyleSheet* inspectorStyleSheet = assertStyleSheetForId(errorString, styleSheetId);
691 if (!inspectorStyleSheet)
692 return makeUnexpected(errorString);
693
694 auto* domAgent = m_instrumentingAgents.persistentDOMAgent();
695 if (!domAgent)
696 return makeUnexpected("DOM domain must be enabled"_s);
697
698 auto result = domAgent->history()->perform(makeUnique<SetStyleSheetTextAction>(inspectorStyleSheet, text));
699 if (result.hasException())
700 return makeUnexpected(InspectorDOMAgent::toErrorString(result.releaseException()));
701
702 return { };

776InspectorStyleSheet* InspectorCSSAgent::createInspectorStyleSheetForDocument(Document& document)
777{
778 if (!document.isHTMLDocument() && !document.isSVGDocument())
779 return nullptr;
780
781 auto styleElement = HTMLStyleElement::create(document);
782 styleElement->setAttributeWithoutSynchronization(HTMLNames::typeAttr, AtomString("text/css", AtomString::ConstructFromLiteral));
783
784 ContainerNode* targetNode;
785 // HEAD is absent in ImageDocuments, for example.
786 if (auto* head = document.head())
787 targetNode = head;
788 else if (auto* body = document.bodyOrFrameset())
789 targetNode = body;
790 else
791 return nullptr;
792
793 // Inserting this <style> into the document will trigger activeStyleSheetsUpdated
794 // and we will create an InspectorStyleSheet for this <style>'s CSSStyleSheet.
795 // Set this flag, so when we create it, we put it into the via inspector map.
796 m_creatingViaInspectorStyleSheet = true;
797 InlineStyleOverrideScope overrideScope(document);
798 auto appendResult = targetNode->appendChild(styleElement);
799 document.styleScope().flushPendingUpdate();
800 m_creatingViaInspectorStyleSheet = false;
801 if (appendResult.hasException())
802 return nullptr;
803
804 auto iterator = m_documentToInspectorStyleSheet.find(&document);
805 ASSERT(iterator != m_documentToInspectorStyleSheet.end());
806 if (iterator == m_documentToInspectorStyleSheet.end())
807 return nullptr;
808
809 auto& inspectorStyleSheetsForDocument = iterator->value;
810 ASSERT(!inspectorStyleSheetsForDocument.isEmpty());
811 if (inspectorStyleSheetsForDocument.isEmpty())
812 return nullptr;
813
814 return inspectorStyleSheetsForDocument.last().get();
815}

842Protocol::ErrorStringOr<Ref<JSON::ArrayOf<Protocol::CSS::CSSPropertyInfo>>> InspectorCSSAgent::getSupportedCSSProperties()
843{
844 auto cssProperties = JSON::ArrayOf<Protocol::CSS::CSSPropertyInfo>::create();
845
846 for (int i = firstCSSProperty; i <= lastCSSProperty; ++i) {
847 CSSPropertyID propertyID = convertToCSSPropertyID(i);
848 // FIXME: Should take account for flags in settings().
849 if (isInternalCSSProperty(propertyID) || !isEnabledCSSProperty(propertyID))
850 continue;
851
852 auto property = Protocol::CSS::CSSPropertyInfo::create()
853 .setName(getPropertyNameString(propertyID))
854 .release();
855
856 auto aliases = CSSProperty::aliasesForProperty(propertyID);
857 if (!aliases.isEmpty()) {
858 auto aliasesArray = JSON::ArrayOf<String>::create();
859 for (auto& alias : aliases)
860 aliasesArray->addItem(alias);
861 property->setAliases(WTFMove(aliasesArray));
862 }
863
864 auto shorthand = shorthandForProperty(propertyID);
865 if (shorthand.length()) {
866 auto longhands = JSON::ArrayOf<String>::create();
867 for (auto longhand : shorthand) {
868 if (isEnabledCSSProperty(longhand))
869 longhands->addItem(getPropertyNameString(longhand));
870 }
871 property->setLonghands(WTFMove(longhands));
872 }
873
874 if (CSSParserFastPaths::isKeywordPropertyID(propertyID)) {
875 auto values = JSON::ArrayOf<String>::create();
876 for (int j = firstCSSValueKeyword; j <= lastCSSValueKeyword; ++j) {
877 CSSValueID valueID = convertToCSSValueID(j);
878 if (CSSParserFastPaths::isValidKeywordPropertyAndValue(propertyID, valueID, strictCSSParserContext()))
879 values->addItem(getValueNameString(valueID));
880 }
881 if (values->length())
882 property->setValues(WTFMove(values));
883 }
884
885 if (CSSProperty::isInheritedProperty(propertyID))
886 property->setInherited(true);
887
888 cssProperties->addItem(WTFMove(property));
889 }
890
891 return cssProperties;
892}

906{

1087RefPtr<Protocol::CSS::CSSRule> InspectorCSSAgent::buildObjectForRule(const StyleRule* styleRule, Style::Resolver& styleResolver, Element& element)
1088{
1089 if (!styleRule)
1090 return nullptr;
1091
1092 // StyleRules returned by Style::Resolver::styleRulesForElement lack parent pointers since that infomation is not cheaply available.
1093 // Since the inspector wants to walk the parent chain, we construct the full wrappers here.
1094 styleResolver.inspectorCSSOMWrappers().collectDocumentWrappers(styleResolver.document().extensionStyleSheets());
1095 styleResolver.inspectorCSSOMWrappers().collectScopeWrappers(Style::Scope::forNode(element));
1096
1097 // Possiblity of :host styles if this element has a shadow root.
1098 if (ShadowRoot* shadowRoot = element.shadowRoot())
1099 styleResolver.inspectorCSSOMWrappers().collectScopeWrappers(shadowRoot->styleScope());
1100
1101 CSSStyleRule* cssomWrapper = styleResolver.inspectorCSSOMWrappers().getWrapperForRuleInSheets(styleRule);
1102 return buildObjectForRule(cssomWrapper);

1115Ref<JSON::ArrayOf<Protocol::CSS::RuleMatch>> InspectorCSSAgent::buildArrayForMatchedRuleList(const Vector<RefPtr<const StyleRule>>& matchedRules, Style::Resolver& styleResolver, Element& element, PseudoId pseudoId)
1116{
1117 auto result = JSON::ArrayOf<Protocol::CSS::RuleMatch>::create();
1118
1119 SelectorChecker::CheckingContext context(SelectorChecker::Mode::CollectingRules);
1120 context.pseudoId = pseudoId != PseudoId::None ? pseudoId : element.pseudoId();
1121 SelectorChecker selectorChecker(element.document());
1122
1123 for (auto& matchedRule : matchedRules) {
1124 RefPtr<Protocol::CSS::CSSRule> ruleObject = buildObjectForRule(matchedRule.get(), styleResolver, element);
1125 if (!ruleObject)
1126 continue;
1127
1128 auto matchingSelectors = JSON::ArrayOf<int>::create();
1129 const CSSSelectorList& selectorList = matchedRule->selectorList();
1130 int index = 0;
1131 for (const CSSSelector* selector = selectorList.first(); selector; selector = CSSSelectorList::next(selector)) {
1132 bool matched = selectorChecker.match(*selector, element, context);
1133 if (matched)
1134 matchingSelectors->addItem(index);
1135 ++index;
1136 }
1137
1138 auto match = Protocol::CSS::RuleMatch::create()
1139 .setRule(ruleObject.releaseNonNull())
1140 .setMatchingSelectors(WTFMove(matchingSelectors))
1141 .release();
1142 result->addItem(WTFMove(match));
1143 }
1144
1145 return result;
1146}
