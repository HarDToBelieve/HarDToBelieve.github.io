125Protocol::ErrorStringOr<void> InspectorCanvasAgent::enable()
126{
127 if (m_instrumentingAgents.enabledCanvasAgent() == this)
128 return { };
129
130 m_instrumentingAgents.setEnabledCanvasAgent(this);
131
132 const auto existsInCurrentPage = [&] (ScriptExecutionContext* scriptExecutionContext) {
133 if (!is<Document>(scriptExecutionContext))
134 return false;
135
136 // FIXME: <https://webkit.org/b/168475> Web Inspector: Correctly display iframe's WebSockets
137 auto* document = downcast<Document>(scriptExecutionContext);
138 return document->page() == &m_inspectedPage;
139 };
140
141 {
142 Locker locker { CanvasRenderingContext::instancesLock() };
143 for (auto* context : CanvasRenderingContext::instances()) {
144#if ENABLE(WEBGPU)
145 // The actual "context" for WebGPU is the `WebGPUDevice`, not the <canvas>.
146 if (is<GPUCanvasContext>(context))
147 continue;
148#endif // ENABLE(WEBGPU)
149
150 if (existsInCurrentPage(context->canvasBase().scriptExecutionContext()))
151 bindCanvas(*context, false);
152 }
153 }
154
155#if ENABLE(WEBGPU)
156 {
157 Locker locker { WebGPUDevice::instancesLock() };
158 for (auto* device : WebGPUDevice::instances()) {
159 if (existsInCurrentPage(device->scriptExecutionContext()))
160 bindCanvas(*device, false);
161 }
162 }
163#endif // ENABLE(WEBGPU)
164
165#if ENABLE(WEBGL)
166 {
167 Locker locker { WebGLProgram::instancesLock() };
168 for (auto& [program, contextWebGLBase] : WebGLProgram::instances()) {
169 if (contextWebGLBase && existsInCurrentPage(contextWebGLBase->canvasBase().scriptExecutionContext()))
170 didCreateWebGLProgram(*contextWebGLBase, *program);
171 }
172 }
173#endif // ENABLE(WEBGL)
174
175#if ENABLE(WEBGPU)
176 {
177 Locker locker { WebGPUPipeline::instancesLock() };
178 for (auto& [pipeline, device] : WebGPUPipeline::instances()) {
179 if (device && existsInCurrentPage(device->scriptExecutionContext()) && pipeline->isValid())
180 didCreateWebGPUPipeline(*device, *pipeline);
181 }
182 }
183#endif // ENABLE(WEBGPU)
184
185 return { };
186}

199Protocol::ErrorStringOr<Protocol::DOM::NodeId> InspectorCanvasAgent::requestNode(const Protocol::Canvas::CanvasId& canvasId)
200{
201 Protocol::ErrorString errorString;
202
203 auto inspectorCanvas = assertInspectorCanvas(errorString, canvasId);
204 if (!inspectorCanvas)
205 return makeUnexpected(errorString);
206
207 auto* node = inspectorCanvas->canvasElement();
208 if (!node)
209 makeUnexpected("Missing element of canvas for given canvasId"_s);
210
211 // FIXME: <https://webkit.org/b/213499> Web Inspector: allow DOM nodes to be instrumented at any point, regardless of whether the main document has also been instrumented
212 int documentNodeId = m_instrumentingAgents.persistentDOMAgent()->boundNodeId(&node->document());
213 if (!documentNodeId)
214 makeUnexpected("Document must have been requested"_s);
215
216 return m_instrumentingAgents.persistentDOMAgent()->pushNodeToFrontend(errorString, documentNodeId, node);
217}

234Protocol::ErrorStringOr<Ref<JSON::ArrayOf<Protocol::DOM::NodeId>>> InspectorCanvasAgent::requestClientNodes(const Protocol::Canvas::CanvasId& canvasId)
235{
236 Protocol::ErrorString errorString;
237
238 auto* domAgent = m_instrumentingAgents.persistentDOMAgent();
239 if (!domAgent)
240 return makeUnexpected("DOM domain must be enabled"_s);
241
242 auto inspectorCanvas = assertInspectorCanvas(errorString, canvasId);
243 if (!inspectorCanvas)
244 return makeUnexpected(errorString);
245
246 auto clientNodeIds = JSON::ArrayOf<Protocol::DOM::NodeId>::create();
247 for (auto& clientNode : inspectorCanvas->clientNodes()) {
248 // FIXME: <https://webkit.org/b/213499> Web Inspector: allow DOM nodes to be instrumented at any point, regardless of whether the main document has also been instrumented
249 if (auto documentNodeId = domAgent->boundNodeId(&clientNode->document()))
250 clientNodeIds->addItem(domAgent->pushNodeToFrontend(errorString, documentNodeId, clientNode));
251 }
252 return clientNodeIds;
253}

255Protocol::ErrorStringOr<Ref<Protocol::Runtime::RemoteObject>> InspectorCanvasAgent::resolveContext(const Protocol::Canvas::CanvasId& canvasId, const String& objectGroup)
256{
257 Protocol::ErrorString errorString;
258
259 auto inspectorCanvas = assertInspectorCanvas(errorString, canvasId);
260 if (!inspectorCanvas)
261 return makeUnexpected(errorString);
262
263 auto* state = inspectorCanvas->scriptExecutionContext()->globalObject();
264 auto injectedScript = m_injectedScriptManager.injectedScriptFor(state);
265 ASSERT(!injectedScript.hasNoValue());
266
267 JSC::JSValue value = inspectorCanvas->resolveContext(state);
268
269 if (!value) {
270 ASSERT_NOT_REACHED();
271 return makeUnexpected("Internal error: unknown context of canvas for given canvasId"_s);
272 }
273
274 auto result = injectedScript.wrapObject(value, objectGroup);
275 if (!result)
276 return makeUnexpected("Internal error: unable to cast Context");
277
278 return result.releaseNonNull();
279}

290Protocol::ErrorStringOr<void> InspectorCanvasAgent::startRecording(const Protocol::Canvas::CanvasId& canvasId, std::optional<int>&& frameCount, std::optional<int>&& memoryLimit)
291{
292 Protocol::ErrorString errorString;
293
294 auto inspectorCanvas = assertInspectorCanvas(errorString, canvasId);
295 if (!inspectorCanvas)
296 return makeUnexpected(errorString);
297
298 auto* context = inspectorCanvas->canvasContext();
299
300 // FIXME: <https://webkit.org/b/201651> Web Inspector: Canvas: support canvas recordings for WebGPUDevice
301 if (!context)
302 return makeUnexpected("Not supported"_s);
303
304 if (context->hasActiveInspectorCanvasCallTracer())
305 return makeUnexpected("Already recording canvas"_s);
306
307 RecordingOptions recordingOptions;
308 if (frameCount)
309 recordingOptions.frameCount = *frameCount;
310 if (memoryLimit)
311 recordingOptions.memoryLimit = *memoryLimit;
312 startRecording(*inspectorCanvas, Protocol::Recording::Initiator::Frontend, WTFMove(recordingOptions));
313
314 return { };
315}

398void InspectorCanvasAgent::frameNavigated(Frame& frame)
399{
400 if (frame.isMainFrame()) {
401 reset();
402 return;
403 }
404
405 Vector<InspectorCanvas*> inspectorCanvases;
406 for (auto& inspectorCanvas : m_identifierToInspectorCanvas.values()) {
407 if (auto* canvasElement = inspectorCanvas->canvasElement()) {
408 if (canvasElement->document().frame() == &frame)
409 inspectorCanvases.append(inspectorCanvas.get());
410 }
411 }
412
413 for (auto* inspectorCanvas : inspectorCanvases)
414 unbindCanvas(*inspectorCanvas);
415}

505void InspectorCanvasAgent::didFinishRecordingCanvasFrame(CanvasRenderingContext& context, bool forceDispatch)
506{
507 if (!context.hasActiveInspectorCanvasCallTracer())
508 return;
509
510 auto inspectorCanvas = findInspectorCanvas(context);
511 ASSERT(inspectorCanvas);
512 if (!inspectorCanvas)
513 return;
514
515 if (!inspectorCanvas->hasRecordingData()) {
516 if (forceDispatch) {
517 m_frontendDispatcher->recordingFinished(inspectorCanvas->identifier(), nullptr);
518 inspectorCanvas->resetRecordingData();
519 ASSERT(!m_recordingCanvasIdentifiers.contains(inspectorCanvas->identifier()));
520 }
521 return;
522 }
523
524 if (forceDispatch)
525 inspectorCanvas->markCurrentFrameIncomplete();
526
527 inspectorCanvas->finalizeFrame();
528 if (inspectorCanvas->currentFrameHasData())
529 m_frontendDispatcher->recordingProgress(inspectorCanvas->identifier(), inspectorCanvas->releaseFrames(), inspectorCanvas->bufferUsed());
530
531 if (!forceDispatch && !inspectorCanvas->overFrameCount())
532 return;
533
534 m_frontendDispatcher->recordingFinished(inspectorCanvas->identifier(), inspectorCanvas->releaseObjectForRecording());
535
536 m_recordingCanvasIdentifiers.remove(inspectorCanvas->identifier());
537}

539void InspectorCanvasAgent::consoleStartRecordingCanvas(CanvasRenderingContext& context, JSC::JSGlobalObject& exec, JSC::JSObject* options)
540{
541 auto inspectorCanvas = findInspectorCanvas(context);
542 ASSERT(inspectorCanvas);
543 if (!inspectorCanvas)
544 return;
545
546 RecordingOptions recordingOptions;
547 if (options) {
548 JSC::VM& vm = exec.vm();
549 if (JSC::JSValue optionSingleFrame = options->get(&exec, JSC::Identifier::fromString(vm, "singleFrame")))
550 recordingOptions.frameCount = optionSingleFrame.toBoolean(&exec) ? 1 : 0;
551 if (JSC::JSValue optionFrameCount = options->get(&exec, JSC::Identifier::fromString(vm, "frameCount")))
552 recordingOptions.frameCount = optionFrameCount.toNumber(&exec);
553 if (JSC::JSValue optionMemoryLimit = options->get(&exec, JSC::Identifier::fromString(vm, "memoryLimit")))
554 recordingOptions.memoryLimit = optionMemoryLimit.toNumber(&exec);
555 if (JSC::JSValue optionName = options->get(&exec, JSC::Identifier::fromString(vm, "name")))
556 recordingOptions.name = optionName.toWTFString(&exec);
557 }
558 startRecording(*inspectorCanvas, Protocol::Recording::Initiator::Console, WTFMove(recordingOptions));
559}

694void InspectorCanvasAgent::recordAction(CanvasRenderingContext& canvasRenderingContext, String&& name, InspectorCanvasCallTracer::ProcessedArguments&& arguments)
695{
696 ASSERT(canvasRenderingContext.hasActiveInspectorCanvasCallTracer());
697
698 auto inspectorCanvas = findInspectorCanvas(canvasRenderingContext);
699 ASSERT(inspectorCanvas);
700
701 // Only enqueue one microtask for all actively recording canvases.
702 if (m_recordingCanvasIdentifiers.isEmpty()) {
703 if (auto* scriptExecutionContext = inspectorCanvas->scriptExecutionContext()) {
704 scriptExecutionContext->eventLoop().queueMicrotask([weakThis = makeWeakPtr(*this)] {
705 if (!weakThis)
706 return;
707
708 auto& canvasAgent = *weakThis;
709
710 auto identifiers = copyToVector(canvasAgent.m_recordingCanvasIdentifiers);
711 for (auto& identifier : identifiers) {
712 auto inspectorCanvas = canvasAgent.m_identifierToInspectorCanvas.get(identifier);
713 if (!inspectorCanvas)
714 continue;
715
716 auto* canvasRenderingContext = inspectorCanvas->canvasContext();
717 ASSERT(canvasRenderingContext);
718 // FIXME: <https://webkit.org/b/201651> Web Inspector: Canvas: support canvas recordings for WebGPUDevice
719
720 if (canvasRenderingContext->hasActiveInspectorCanvasCallTracer())
721 canvasAgent.didFinishRecordingCanvasFrame(*canvasRenderingContext);
722 }
723
724 canvasAgent.m_recordingCanvasIdentifiers.clear();
725 });
726 }
727 }
728
729 m_recordingCanvasIdentifiers.add(inspectorCanvas->identifier());
730
731 inspectorCanvas->recordAction(WTFMove(name), WTFMove(arguments));
732
733 if (!inspectorCanvas->hasBufferSpace())
734 didFinishRecordingCanvasFrame(canvasRenderingContext, true);
735}

737void InspectorCanvasAgent::startRecording(InspectorCanvas& inspectorCanvas, Protocol::Recording::Initiator initiator, RecordingOptions&& recordingOptions)
738{
739 auto* context = inspectorCanvas.canvasContext();
740 ASSERT(context);
741 // FIXME: <https://webkit.org/b/201651> Web Inspector: Canvas: support canvas recordings for WebGPUDevice
742
743 if (!is<CanvasRenderingContext2D>(context)
744 && !is<ImageBitmapRenderingContext>(context)
745#if ENABLE(WEBGL)
746 && !is<WebGLRenderingContext>(context)
747#endif // ENABLE(WEBGL)
748#if ENABLE(WEBGL2)
749 && !is<WebGL2RenderingContext>(context)
750#endif // ENABLE(WEBGL2)
751 )
752 return;
753
754 if (context->hasActiveInspectorCanvasCallTracer())
755 return;
756
757 inspectorCanvas.resetRecordingData();
758 if (recordingOptions.frameCount)
759 inspectorCanvas.setFrameCount(recordingOptions.frameCount.value());
760 if (recordingOptions.memoryLimit)
761 inspectorCanvas.setBufferLimit(recordingOptions.memoryLimit.value());
762 if (recordingOptions.name)
763 inspectorCanvas.setRecordingName(recordingOptions.name.value());
764 context->setHasActiveInspectorCanvasCallTracer(true);
765
766 m_frontendDispatcher->recordingStarted(inspectorCanvas.identifier(), initiator);
767}

851void InspectorCanvasAgent::unbindCanvas(InspectorCanvas& inspectorCanvas)
852{
853#if ENABLE(WEBGL)
854 Vector<InspectorShaderProgram*> programsToRemove;
855 for (auto& inspectorProgram : m_identifierToInspectorProgram.values()) {
856 if (&inspectorProgram->canvas() == &inspectorCanvas)
857 programsToRemove.append(inspectorProgram.get());
858 }
859
860 for (auto* inspectorProgram : programsToRemove)
861 unbindProgram(*inspectorProgram);
862#endif // ENABLE(WEBGL)
863
864 if (auto* context = inspectorCanvas.canvasContext())
865 context->canvasBase().removeObserver(*this);
866
867 String identifier = inspectorCanvas.identifier();
868 m_identifierToInspectorCanvas.remove(identifier);
869
870 // This can be called in response to GC. Due to the single-process model used in WebKit1, the
871 // event must be dispatched from a timer to prevent the frontend from making JS allocations
872 // while the GC is still active.
873 m_removedCanvasIdentifiers.append(identifier);
874
875 if (!m_canvasDestroyedTimer.isActive())
876 m_canvasDestroyedTimer.startOneShot(0_s);
877}
