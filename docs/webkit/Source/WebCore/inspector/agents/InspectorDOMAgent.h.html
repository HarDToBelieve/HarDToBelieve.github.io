289 struct InspectorEventListener {
290 Inspector::Protocol::DOM::EventListenerId identifier { 1 };
291 RefPtr<EventTarget> eventTarget;
292 RefPtr<EventListener> eventListener;
293 AtomString eventType;
294 bool useCapture { false };
295 bool disabled { false };
296 RefPtr<JSC::Breakpoint> breakpoint;
297
298 InspectorEventListener() { }
299
300 InspectorEventListener(Inspector::Protocol::DOM::EventListenerId identifier, EventTarget& target, const AtomString& type, EventListener& listener, bool capture)
301 : identifier(identifier)
302 , eventTarget(&target)
303 , eventListener(&listener)
304 , eventType(type)
305 , useCapture(capture)
306 {
307 }
308
309 bool matches(EventTarget& target, const AtomString& type, EventListener& listener, bool capture)
310 {
311 if (eventTarget.get() != &target)
312 return false;
313 if (eventListener.get() != &listener)
314 return false;
315 if (eventType != type)
316 return false;
317 if (useCapture != capture)
318 return false;
319 return true;
320 }
321 };
