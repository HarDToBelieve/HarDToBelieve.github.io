53InspectorDOMDebuggerAgent::InspectorDOMDebuggerAgent(WebAgentContext& context, InspectorDebuggerAgent* debuggerAgent)

74void InspectorDOMDebuggerAgent::disable()
75{
76 m_instrumentingAgents.setEnabledDOMDebuggerAgent(nullptr);
77
78 m_listenerBreakpoints.clear();
79 m_pauseOnAllIntervalsBreakpoint = nullptr;
80 m_pauseOnAllListenersBreakpoint = nullptr;
81 m_pauseOnAllTimeoutsBreakpoint = nullptr;
82
83 m_urlTextBreakpoints.clear();
84 m_urlRegexBreakpoints.clear();
85 m_pauseOnAllURLsBreakpoint = nullptr;
86}

116void InspectorDOMDebuggerAgent::mainFrameNavigated()
117{
118 for (auto& breakpoint : m_listenerBreakpoints.values())
119 breakpoint->resetHitCount();
120
121 if (m_pauseOnAllIntervalsBreakpoint)
122 m_pauseOnAllIntervalsBreakpoint->resetHitCount();
123
124 if (m_pauseOnAllListenersBreakpoint)
125 m_pauseOnAllListenersBreakpoint->resetHitCount();
126
127 if (m_pauseOnAllTimeoutsBreakpoint)
128 m_pauseOnAllTimeoutsBreakpoint->resetHitCount();

131Protocol::ErrorStringOr<void> InspectorDOMDebuggerAgent::setEventBreakpoint(Protocol::DOMDebugger::EventBreakpointType breakpointType, const String& eventName, RefPtr<JSON::Object>&& options)
132{
133 Protocol::ErrorString errorString;
134
135 auto breakpoint = InspectorDebuggerAgent::debuggerBreakpointFromPayload(errorString, WTFMove(options));
136 if (!breakpoint)
137 return makeUnexpected(errorString);
138
139 if (!eventName.isEmpty()) {
140 if (breakpointType == Protocol::DOMDebugger::EventBreakpointType::Listener) {
141 if (!m_listenerBreakpoints.add(eventName, breakpoint.releaseNonNull()))
142 return makeUnexpected("Breakpoint with eventName already exists"_s);
143 return { };
144 }
145
146 return makeUnexpected("Unexpected eventName"_s);
147 }
148
149 switch (breakpointType) {
150 case Protocol::DOMDebugger::EventBreakpointType::AnimationFrame:
151 if (!setAnimationFrameBreakpoint(errorString, WTFMove(breakpoint)))
152 return makeUnexpected(errorString);
153 return { };
154
155 case Protocol::DOMDebugger::EventBreakpointType::Interval:
156 if (m_pauseOnAllIntervalsBreakpoint)
157 return makeUnexpected("Breakpoint for Interval already exists"_s);
158 m_pauseOnAllIntervalsBreakpoint = WTFMove(breakpoint);
159 return { };
160
161 case Protocol::DOMDebugger::EventBreakpointType::Listener:
162 if (m_pauseOnAllListenersBreakpoint)
163 return makeUnexpected("Breakpoint for Listener already exists"_s);
164 m_pauseOnAllListenersBreakpoint = WTFMove(breakpoint);
165 return { };
166
167 case Protocol::DOMDebugger::EventBreakpointType::Timeout:
168 if (m_pauseOnAllTimeoutsBreakpoint)
169 return makeUnexpected("Breakpoint for Timeout already exists"_s);
170 m_pauseOnAllTimeoutsBreakpoint = WTFMove(breakpoint);
171 return { };
172 }
173
174 ASSERT_NOT_REACHED();
175 return makeUnexpected("Not supported");
176}

178Protocol::ErrorStringOr<void> InspectorDOMDebuggerAgent::removeEventBreakpoint(Protocol::DOMDebugger::EventBreakpointType breakpointType, const String& eventName)
179{
180 Protocol::ErrorString errorString;
181
182 if (!eventName.isEmpty()) {
183 if (breakpointType == Protocol::DOMDebugger::EventBreakpointType::Listener) {
184 if (!m_listenerBreakpoints.remove(eventName))
185 return makeUnexpected("Breakpoint for given eventName missing"_s);
186 return { };
187 }
188
189 return makeUnexpected("Unexpected eventName"_s);
190 }
191
192 switch (breakpointType) {
193 case Protocol::DOMDebugger::EventBreakpointType::AnimationFrame:
194 if (!setAnimationFrameBreakpoint(errorString, nullptr))
195 return makeUnexpected(errorString);
196 return { };
197
198 case Protocol::DOMDebugger::EventBreakpointType::Interval:
199 if (!m_pauseOnAllIntervalsBreakpoint)
200 return makeUnexpected("Breakpoint for Intervals missing"_s);
201 m_pauseOnAllIntervalsBreakpoint = nullptr;
202 return { };
203
204 case Protocol::DOMDebugger::EventBreakpointType::Listener:
205 if (!m_pauseOnAllListenersBreakpoint)
206 return makeUnexpected("Breakpoint for Listeners missing"_s);
207 m_pauseOnAllListenersBreakpoint = nullptr;
208 return { };
209
210 case Protocol::DOMDebugger::EventBreakpointType::Timeout:
211 if (!m_pauseOnAllTimeoutsBreakpoint)
212 return makeUnexpected("Breakpoint for Timeouts missing"_s);
213 m_pauseOnAllTimeoutsBreakpoint = nullptr;
214 return { };
215 }
216
217 ASSERT_NOT_REACHED();
218 return makeUnexpected("Not supported");
219}

221void InspectorDOMDebuggerAgent::willHandleEvent(Event& event, const RegisteredEventListener& registeredEventListener)
222{
223 auto state = event.target()->scriptExecutionContext()->globalObject();
224 auto injectedScript = m_injectedScriptManager.injectedScriptFor(state);
225 if (injectedScript.hasNoValue())
226 return;
227
228 // Set Web Inspector console command line `$event` getter.
229 {
230 JSC::JSLockHolder lock(state);
231
232 injectedScript.setEventValue(toJS(state, deprecatedGlobalObjectForPrototype(state), event));
233 }
234
235 if (!m_debuggerAgent->breakpointsActive())
236 return;
237
238 auto* domAgent = m_instrumentingAgents.persistentDOMAgent();
239
240 auto breakpoint = m_pauseOnAllListenersBreakpoint;
241 if (!breakpoint) {
242 auto it = m_listenerBreakpoints.find(event.type());
243 if (it != m_listenerBreakpoints.end())
244 breakpoint = it->value.copyRef();
245 }
246 if (!breakpoint && domAgent)
247 breakpoint = domAgent->breakpointForEventListener(*event.currentTarget(), event.type(), registeredEventListener.callback(), registeredEventListener.useCapture());
248 if (!breakpoint)
249 return;
250
251 Ref<JSON::Object> eventData = JSON::Object::create();
252 eventData->setString("eventName"_s, event.type());
253 if (domAgent) {
254 int eventListenerId = domAgent->idForEventListener(*event.currentTarget(), event.type(), registeredEventListener.callback(), registeredEventListener.useCapture());
255 if (eventListenerId)
256 eventData->setInteger("eventListenerId"_s, eventListenerId);
257 }
258
259 m_debuggerAgent->schedulePauseForSpecialBreakpoint(*breakpoint, Inspector::DebuggerFrontendDispatcher::Reason::Listener, WTFMove(eventData));
260}

262void InspectorDOMDebuggerAgent::didHandleEvent(Event& event, const RegisteredEventListener& registeredEventListener)
263{
264 auto state = event.target()->scriptExecutionContext()->globalObject();
265 auto injectedScript = m_injectedScriptManager.injectedScriptFor(state);
266 if (injectedScript.hasNoValue())
267 return;
268
269 // Clear Web Inspector console command line `$event` getter.
270 {
271 JSC::JSLockHolder lock(state);
272
273 injectedScript.clearEventValue();
274 }
275
276 if (!m_debuggerAgent->breakpointsActive())
277 return;
278
279 auto breakpoint = m_pauseOnAllListenersBreakpoint;
280 if (!breakpoint)
281 breakpoint = m_listenerBreakpoints.get(event.type());
282 if (!breakpoint) {
283 if (auto* domAgent = m_instrumentingAgents.persistentDOMAgent())
284 breakpoint = domAgent->breakpointForEventListener(*event.currentTarget(), event.type(), registeredEventListener.callback(), registeredEventListener.useCapture());
285 }
286 if (!breakpoint)
287 return;
288
289 m_debuggerAgent->cancelPauseForSpecialBreakpoint(*breakpoint);
290}

292void InspectorDOMDebuggerAgent::willFireTimer(bool oneShot)
293{
294 if (!m_debuggerAgent->breakpointsActive())
295 return;
296
297 auto breakpoint = oneShot ? m_pauseOnAllTimeoutsBreakpoint : m_pauseOnAllIntervalsBreakpoint;
298 if (!breakpoint)
299 return;
300
301 auto breakReason = oneShot ? Inspector::DebuggerFrontendDispatcher::Reason::Timeout : Inspector::DebuggerFrontendDispatcher::Reason::Interval;
302 m_debuggerAgent->schedulePauseForSpecialBreakpoint(*breakpoint, breakReason);
303}

305void InspectorDOMDebuggerAgent::didFireTimer(bool oneShot)
306{
307 if (!m_debuggerAgent->breakpointsActive())
308 return;

311 if (!breakpoint)
312 return;
313
314 m_debuggerAgent->cancelPauseForSpecialBreakpoint(*breakpoint);

317Protocol::ErrorStringOr<void> InspectorDOMDebuggerAgent::setURLBreakpoint(const String& url, std::optional<bool>&& isRegex, RefPtr<JSON::Object>&& options)
318{
319 Protocol::ErrorString errorString;
320
321 auto breakpoint = InspectorDebuggerAgent::debuggerBreakpointFromPayload(errorString, WTFMove(options));
322 if (!breakpoint)
323 return makeUnexpected(errorString);
324
325 if (url.isEmpty()) {
326 if (m_pauseOnAllURLsBreakpoint)
327 return makeUnexpected("Breakpoint for all URLs already exists"_s);
328 m_pauseOnAllURLsBreakpoint = WTFMove(breakpoint);
329 return { };
330 }
331
332 if (isRegex && *isRegex) {
333 if (!m_urlRegexBreakpoints.add(url, breakpoint.releaseNonNull()))
334 return makeUnexpected("Breakpoint for given regex already exists"_s);
335 } else {
336 if (!m_urlTextBreakpoints.add(url, breakpoint.releaseNonNull()))
337 return makeUnexpected("Breakpoint for given URL already exists"_s);
338 }
339
340 return { };
341}

343Protocol::ErrorStringOr<void> InspectorDOMDebuggerAgent::removeURLBreakpoint(const String& url, std::optional<bool>&& isRegex)
344{
345 if (url.isEmpty()) {
346 if (!m_pauseOnAllURLsBreakpoint)
347 return makeUnexpected("Breakpoint for all URLs missing"_s);
348 m_pauseOnAllURLsBreakpoint = nullptr;
349 return { };
350 }
351
352 if (isRegex && *isRegex) {
353 if (!m_urlRegexBreakpoints.remove(url))
354 return makeUnexpected("Missing breakpoint for given regex"_s);
355 } else {
356 if (!m_urlTextBreakpoints.remove(url))
357 return makeUnexpected("Missing breakpoint for given URL"_s);
358 }
359
360 return { };
361}

363void InspectorDOMDebuggerAgent::breakOnURLIfNeeded(const String& url, URLBreakpointSource source)
364{
365 if (!m_debuggerAgent->breakpointsActive())
366 return;
367
368 constexpr bool caseSensitive = false;
369
370 auto breakpointURL = emptyString();
371 auto breakpoint = m_pauseOnAllURLsBreakpoint.copyRef();
372 if (!breakpoint) {
373 for (auto& [query, textBreakpoint] : m_urlTextBreakpoints) {
374 auto regex = ContentSearchUtilities::createRegularExpressionForSearchString(query, caseSensitive, ContentSearchUtilities::SearchStringType::ContainsString);
375 if (regex.match(url) != -1) {
376 breakpoint = textBreakpoint.copyRef();
377 breakpointURL = query;
378 break;
379 }
380 }
381 }
382 if (!breakpoint) {
383 for (auto& [query, regexBreakpoint] : m_urlRegexBreakpoints) {
384 auto regex = ContentSearchUtilities::createRegularExpressionForSearchString(query, caseSensitive, ContentSearchUtilities::SearchStringType::Regex);
385 if (regex.match(url) != -1) {
386 breakpoint = regexBreakpoint.copyRef();
387 breakpointURL = query;
388 break;
389 }
390 }
391 }
392 if (!breakpoint)
393 return;
394
395 auto breakReason = Inspector::DebuggerFrontendDispatcher::Reason::Other;
396 switch (source) {
397 case URLBreakpointSource::Fetch:
398 breakReason = Inspector::DebuggerFrontendDispatcher::Reason::Fetch;
399 break;
400
401 case URLBreakpointSource::XHR:
402 breakReason = Inspector::DebuggerFrontendDispatcher::Reason::XHR;
403 break;
404 }
405
406 Ref<JSON::Object> eventData = JSON::Object::create();
407 eventData->setString("breakpointURL", breakpointURL);
408 eventData->setString("url", url);
409 m_debuggerAgent->breakProgram(breakReason, WTFMove(eventData), WTFMove(breakpoint));
410}
