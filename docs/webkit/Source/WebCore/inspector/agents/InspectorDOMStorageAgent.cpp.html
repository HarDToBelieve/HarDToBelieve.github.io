76Protocol::ErrorStringOr<void> InspectorDOMStorageAgent::enable()
77{
78 if (m_instrumentingAgents.enabledDOMStorageAgent() == this)
79 return makeUnexpected("DOMStorage domain already enabled"_s);
80
81 m_instrumentingAgents.setEnabledDOMStorageAgent(this);
82
83 return { };
84}

86Protocol::ErrorStringOr<void> InspectorDOMStorageAgent::disable()
87{
88 if (m_instrumentingAgents.enabledDOMStorageAgent() != this)
89 return makeUnexpected("DOMStorage domain already disabled"_s);
90
91 m_instrumentingAgents.setEnabledDOMStorageAgent(nullptr);
92
93 return { };
94}

96Protocol::ErrorStringOr<Ref<JSON::ArrayOf<Protocol::DOMStorage::Item>>> InspectorDOMStorageAgent::getDOMStorageItems(Ref<JSON::Object>&& storageId)
97{
98 Protocol::ErrorString errorString;
99
100 Frame* frame;
101 RefPtr<StorageArea> storageArea = findStorageArea(errorString, WTFMove(storageId), frame);
102 if (!storageArea)
103 return makeUnexpected(errorString);
104
105 auto storageItems = JSON::ArrayOf<JSON::ArrayOf<String>>::create();
106 for (unsigned i = 0; i < storageArea->length(); ++i) {
107 String key = storageArea->key(i);
108 String value = storageArea->item(key);
109
110 auto entry = JSON::ArrayOf<String>::create();
111 entry->addItem(key);
112 entry->addItem(value);
113 storageItems->addItem(WTFMove(entry));
114 }
115 return storageItems;
116}

118Protocol::ErrorStringOr<void> InspectorDOMStorageAgent::setDOMStorageItem(Ref<JSON::Object>&& storageId, const String& key, const String& value)
119{
120 Protocol::ErrorString errorString;
121
122 Frame* frame;
123 RefPtr<StorageArea> storageArea = findStorageArea(errorString, WTFMove(storageId), frame);
124 if (!storageArea)
125 return makeUnexpected(errorString);
126
127 bool quotaException = false;
128 storageArea->setItem(frame, key, value, quotaException);
129 if (quotaException)
130 return makeUnexpected(DOMException::name(QuotaExceededError));
131
132 return { };
133}

163String InspectorDOMStorageAgent::storageId(Storage& storage)
164{
165 Document* document = storage.frame()->document();
166 ASSERT(document);
167 DOMWindow* window = document->domWindow();
168 ASSERT(window);
169 Ref<SecurityOrigin> securityOrigin = document->securityOrigin();
170 bool isLocalStorage = window->optionalLocalStorage() == &storage;
171 return InspectorDOMStorageAgent::storageId(securityOrigin.ptr(), isLocalStorage)->toJSONString();
172}

174Ref<Protocol::DOMStorage::StorageId> InspectorDOMStorageAgent::storageId(SecurityOrigin* securityOrigin, bool isLocalStorage)
175{
176 return Protocol::DOMStorage::StorageId::create()
177 .setSecurityOrigin(securityOrigin->toRawString())
178 .setIsLocalStorage(isLocalStorage)
179 .release();
180}

182void InspectorDOMStorageAgent::didDispatchDOMStorageEvent(const String& key, const String& oldValue, const String& newValue, StorageType storageType, SecurityOrigin* securityOrigin)
183{
184 auto id = InspectorDOMStorageAgent::storageId(securityOrigin, storageType == StorageType::Local);
185
186 if (key.isNull())
187 m_frontendDispatcher->domStorageItemsCleared(WTFMove(id));
188 else if (newValue.isNull())
189 m_frontendDispatcher->domStorageItemRemoved(WTFMove(id), key);
190 else if (oldValue.isNull())
191 m_frontendDispatcher->domStorageItemAdded(WTFMove(id), key, newValue);
192 else
193 m_frontendDispatcher->domStorageItemUpdated(WTFMove(id), key, oldValue, newValue);
194}

196RefPtr<StorageArea> InspectorDOMStorageAgent::findStorageArea(Protocol::ErrorString& errorString, Ref<JSON::Object>&& storageId, Frame*& targetFrame)
197{
198 auto securityOrigin = storageId->getString(Protocol::DOMStorage::StorageId::securityOriginKey);
199 if (!securityOrigin) {
200 errorString = "Missing securityOrigin in given storageId";
201 return nullptr;
202 }
203
204 auto isLocalStorage = storageId->getBoolean(Protocol::DOMStorage::StorageId::isLocalStorageKey);
205 if (!isLocalStorage) {
206 errorString = "Missing isLocalStorage in given storageId";
207 return nullptr;
208 }
209
210 targetFrame = InspectorPageAgent::findFrameWithSecurityOrigin(m_inspectedPage, securityOrigin);
211 if (!targetFrame) {
212 errorString = "Missing frame for given securityOrigin"_s;
213 return nullptr;
214 }
215
216 if (!*isLocalStorage)
217 return m_inspectedPage.sessionStorage()->storageArea(targetFrame->document()->securityOrigin().data());
218 return m_inspectedPage.storageNamespaceProvider().localStorageArea(*targetFrame->document());
219}
