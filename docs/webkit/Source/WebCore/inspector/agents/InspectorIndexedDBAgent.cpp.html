91class OpenDatabaseCallback final : public EventListener {
92public:
93 static Ref<OpenDatabaseCallback> create(ExecutableWithDatabase& executableWithDatabase)
94 {
95 return adoptRef(*new OpenDatabaseCallback(executableWithDatabase));
96 }
97
98 bool operator==(const EventListener& other) const final
99 {
100 return this == &other;
101 }
102
103 void handleEvent(ScriptExecutionContext&, Event& event) final
104 {
105 if (event.type() != eventNames().successEvent) {
106 m_executableWithDatabase->requestCallback().sendFailure("Unexpected event type.");
107 return;
108 }
109
110 auto& request = static_cast<IDBOpenDBRequest&>(*event.target());
111
112 auto result = request.result();
113 if (result.hasException()) {
114 m_executableWithDatabase->requestCallback().sendFailure("Could not get result in callback.");
115 return;
116 }
117
118 auto resultValue = result.releaseReturnValue();
119 if (!WTF::holds_alternative<RefPtr<IDBDatabase>>(resultValue)) {
120 m_executableWithDatabase->requestCallback().sendFailure("Unexpected result type.");
121 return;
122 }
123
124 auto databaseResult = WTF::get<RefPtr<IDBDatabase>>(resultValue);
125 m_executableWithDatabase->execute(*databaseResult);
126 databaseResult->close();
127 }
128
129private:
130 OpenDatabaseCallback(ExecutableWithDatabase& executableWithDatabase)
131 : EventListener(EventListener::CPPEventListenerType)
132 , m_executableWithDatabase(executableWithDatabase) { }
133 Ref<ExecutableWithDatabase> m_executableWithDatabase;
134};

136void ExecutableWithDatabase::start(IDBFactory* idbFactory, SecurityOrigin*, const String& databaseName)
137{
138 if (!context()) {
139 requestCallback().sendFailure("Could not open database.");
140 return;
141 }
142
143 auto result = idbFactory->open(*context(), databaseName, std::nullopt);
144 if (result.hasException()) {
145 requestCallback().sendFailure("Could not open database.");
146 return;
147 }
148
149 result.releaseReturnValue()->addEventListener(eventNames().successEvent, OpenDatabaseCallback::create(*this), false);
150}

153static Ref<Protocol::IndexedDB::KeyPath> keyPathFromIDBKeyPath(const std::optional<IDBKeyPath>& idbKeyPath)
154{
155 if (!idbKeyPath)
156 return Protocol::IndexedDB::KeyPath::create().setType(Protocol::IndexedDB::KeyPath::Type::Null).release();
157
158 auto visitor = WTF::makeVisitor([](const String& string) {
159 auto keyPath = Protocol::IndexedDB::KeyPath::create().setType(Protocol::IndexedDB::KeyPath::Type::String).release();
160 keyPath->setString(string);
161 return keyPath;
162 }, [](const Vector<String>& vector) {
163 auto array = JSON::ArrayOf<String>::create();
164 for (auto& string : vector)
165 array->addItem(string);
166 auto keyPath = Protocol::IndexedDB::KeyPath::create().setType(Protocol::IndexedDB::KeyPath::Type::Array).release();
167 keyPath->setArray(WTFMove(array));
168 return keyPath;
169 });
170 return WTF::visit(visitor, idbKeyPath.value());
171}

197class DatabaseLoader final : public ExecutableWithDatabase {
198public:
199 static Ref<DatabaseLoader> create(ScriptExecutionContext* context, Ref<IndexedDBBackendDispatcherHandler::RequestDatabaseCallback>&& requestCallback)
200 {
201 return adoptRef(*new DatabaseLoader(context, WTFMove(requestCallback)));
202 }
203
204 ~DatabaseLoader() override = default;
205
206 void execute(IDBDatabase& database) override
207 {
208 if (!requestCallback().isActive())
209 return;
210
211 auto& databaseInfo = database.info();
212 auto objectStores = JSON::ArrayOf<Protocol::IndexedDB::ObjectStore>::create();
213 auto objectStoreNames = databaseInfo.objectStoreNames();
214 for (auto& name : objectStoreNames) {
215 auto* objectStoreInfo = databaseInfo.infoForExistingObjectStore(name);
216 if (!objectStoreInfo)
217 continue;
218
219 auto indexes = JSON::ArrayOf<Protocol::IndexedDB::ObjectStoreIndex>::create();
220
221 for (auto& indexInfo : objectStoreInfo->indexMap().values()) {
222 auto objectStoreIndex = Protocol::IndexedDB::ObjectStoreIndex::create()
223 .setName(indexInfo.name())
224 .setKeyPath(keyPathFromIDBKeyPath(indexInfo.keyPath()))
225 .setUnique(indexInfo.unique())
226 .setMultiEntry(indexInfo.multiEntry())
227 .release();
228 indexes->addItem(WTFMove(objectStoreIndex));
229 }
230
231 auto objectStore = Protocol::IndexedDB::ObjectStore::create()
232 .setName(objectStoreInfo->name())
233 .setKeyPath(keyPathFromIDBKeyPath(objectStoreInfo->keyPath()))
234 .setAutoIncrement(objectStoreInfo->autoIncrement())
235 .setIndexes(WTFMove(indexes))
236 .release();
237 objectStores->addItem(WTFMove(objectStore));
238 }
239
240 auto result = Protocol::IndexedDB::DatabaseWithObjectStores::create()
241 .setName(databaseInfo.name())
242 .setVersion(databaseInfo.version())
243 .setObjectStores(WTFMove(objectStores))
244 .release();
245 m_requestCallback->sendSuccess(WTFMove(result));
246 }

248 BackendDispatcher::CallbackBase& requestCallback() override { return m_requestCallback.get(); }
249private:
250 DatabaseLoader(ScriptExecutionContext* context, Ref<IndexedDBBackendDispatcherHandler::RequestDatabaseCallback>&& requestCallback)
251 : ExecutableWithDatabase(context)
252 , m_requestCallback(WTFMove(requestCallback)) { }
253 Ref<IndexedDBBackendDispatcherHandler::RequestDatabaseCallback> m_requestCallback;
254};

256static RefPtr<IDBKey> idbKeyFromInspectorObject(Ref<JSON::Object>&& key)
257{
258 auto typeString = key->getString(Protocol::IndexedDB::Key::typeKey);
259 if (!typeString)
260 return nullptr;
261
262 auto type = Protocol::Helpers::parseEnumValueFromString<Protocol::IndexedDB::Key::Type>(typeString);
263 if (!type)
264 return nullptr;
265
266 switch (*type) {
267 case Protocol::IndexedDB::Key::Type::Number: {
268 auto number = key->getDouble(Protocol::IndexedDB::Key::numberKey);
269 if (!number)
270 return nullptr;
271 return IDBKey::createNumber(*number);
272 }
273
274 case Protocol::IndexedDB::Key::Type::String: {
275 auto string = key->getString(Protocol::IndexedDB::Key::stringKey);
276 if (!string)
277 return nullptr;
278 return IDBKey::createString(string);
279 }
280
281 case Protocol::IndexedDB::Key::Type::Date: {
282 auto date = key->getDouble(Protocol::IndexedDB::Key::dateKey);
283 if (!date)
284 return nullptr;
285 return IDBKey::createDate(*date);
286 }
287
288 case Protocol::IndexedDB::Key::Type::Array: {
289 auto array = key->getArray(Protocol::IndexedDB::Key::arrayKey);
290 if (!array)
291 return nullptr;
292
293 Vector<RefPtr<IDBKey>> keyArray;
294 for (size_t i = 0; i < array->length(); ++i) {
295 auto object = array->get(i)->asObject();
296 if (!object)
297 return nullptr;
298 keyArray.append(idbKeyFromInspectorObject(object.releaseNonNull()));
299 }
300 return IDBKey::createArray(keyArray);
301 }
302 }
303
304 ASSERT_NOT_REACHED();
305 return nullptr;
306}

308static RefPtr<IDBKeyRange> idbKeyRangeFromKeyRange(JSON::Object& keyRange)
309{
310 RefPtr<IDBKey> idbLower;
311 if (auto lower = keyRange.getObject(Protocol::IndexedDB::KeyRange::lowerKey)) {
312 idbLower = idbKeyFromInspectorObject(lower.releaseNonNull());
313 if (!idbLower)
314 return nullptr;
315 }
316
317 RefPtr<IDBKey> idbUpper;
318 if (auto upper = keyRange.getObject(Protocol::IndexedDB::KeyRange::upperKey)) {
319 idbUpper = idbKeyFromInspectorObject(upper.releaseNonNull());
320 if (!idbUpper)
321 return nullptr;
322 }
323
324 auto lowerOpen = keyRange.getBoolean(Protocol::IndexedDB::KeyRange::lowerOpenKey);
325 if (!lowerOpen)
326 return nullptr;
327
328 auto upperOpen = keyRange.getBoolean(Protocol::IndexedDB::KeyRange::upperOpenKey);
329 if (!upperOpen)
330 return nullptr;
331
332 return IDBKeyRange::create(WTFMove(idbLower), WTFMove(idbUpper), *lowerOpen, *upperOpen);
333}

335class OpenCursorCallback final : public EventListener {
336public:
337 static Ref<OpenCursorCallback> create(InjectedScript injectedScript, Ref<IndexedDBBackendDispatcherHandler::RequestDataCallback>&& requestCallback, int skipCount, unsigned pageSize)
338 {
339 return adoptRef(*new OpenCursorCallback(injectedScript, WTFMove(requestCallback), skipCount, pageSize));
340 }
341
342 ~OpenCursorCallback() override = default;
343
344 bool operator==(const EventListener& other) const override
345 {
346 return this == &other;
347 }
348
349 void handleEvent(ScriptExecutionContext& context, Event& event) override
350 {
351 if (event.type() != eventNames().successEvent) {
352 m_requestCallback->sendFailure("Unexpected event type.");
353 return;
354 }
355
356 auto& request = static_cast<IDBRequest&>(*event.target());
357
358 auto result = request.result();
359 if (result.hasException()) {
360 m_requestCallback->sendFailure("Could not get result in callback.");
361 return;
362 }
363
364 auto resultValue = result.releaseReturnValue();
365 if (!WTF::holds_alternative<RefPtr<IDBCursor>>(resultValue)) {
366 end(false);
367 return;
368 }
369
370 auto cursor = WTF::get<RefPtr<IDBCursor>>(resultValue);
371
372 if (m_skipCount) {
373 if (cursor->advance(m_skipCount).hasException())
374 m_requestCallback->sendFailure("Could not advance cursor.");
375 m_skipCount = 0;
376 return;
377 }
378
379 if (m_result->length() == m_pageSize) {
380 end(true);
381 return;
382 }
383
384 // Continue cursor before making injected script calls, otherwise transaction might be finished.
385 if (cursor->continueFunction(nullptr).hasException()) {
386 m_requestCallback->sendFailure("Could not continue cursor.");
387 return;
388 }
389
390 auto* lexicalGlobalObject = context.globalObject();
391
392 auto key = m_injectedScript.wrapObject(toJS(*lexicalGlobalObject, *lexicalGlobalObject, cursor->key()), String(), true);
393 if (!key)
394 return;
395
396 auto primaryKey = m_injectedScript.wrapObject(toJS(*lexicalGlobalObject, *lexicalGlobalObject, cursor->primaryKey()), String(), true);
397 if (!primaryKey)
398 return;
399
400 auto value = m_injectedScript.wrapObject(deserializeIDBValueToJSValue(*lexicalGlobalObject, cursor->value()), String(), true);
401 if (!value)
402 return;
403
404 auto dataEntry = Protocol::IndexedDB::DataEntry::create()
405 .setKey(key.releaseNonNull())
406 .setPrimaryKey(primaryKey.releaseNonNull())
407 .setValue(value.releaseNonNull())
408 .release();
409 m_result->addItem(WTFMove(dataEntry));
410 }
411
412 void end(bool hasMore)
413 {
414 if (!m_requestCallback->isActive())
415 return;
416 m_requestCallback->sendSuccess(WTFMove(m_result), hasMore);
417 }
418
419private:
420 OpenCursorCallback(InjectedScript injectedScript, Ref<IndexedDBBackendDispatcherHandler::RequestDataCallback>&& requestCallback, int skipCount, unsigned pageSize)
421 : EventListener(EventListener::CPPEventListenerType)
422 , m_injectedScript(injectedScript)
423 , m_requestCallback(WTFMove(requestCallback))
424 , m_result(JSON::ArrayOf<Protocol::IndexedDB::DataEntry>::create())
425 , m_skipCount(skipCount)
426 , m_pageSize(pageSize)
427 {
428 }
429 InjectedScript m_injectedScript;
430 Ref<IndexedDBBackendDispatcherHandler::RequestDataCallback> m_requestCallback;
431 Ref<JSON::ArrayOf<Protocol::IndexedDB::DataEntry>> m_result;
432 int m_skipCount;
433 unsigned m_pageSize;
434};

436class DataLoader final : public ExecutableWithDatabase {
437public:
438 static Ref<DataLoader> create(ScriptExecutionContext* context, Ref<IndexedDBBackendDispatcherHandler::RequestDataCallback>&& requestCallback, const InjectedScript& injectedScript, const String& objectStoreName, const String& indexName, RefPtr<IDBKeyRange>&& idbKeyRange, int skipCount, unsigned pageSize)
439 {
440 return adoptRef(*new DataLoader(context, WTFMove(requestCallback), injectedScript, objectStoreName, indexName, WTFMove(idbKeyRange), skipCount, pageSize));
441 }
442
443 ~DataLoader() override = default;
444
445 void execute(IDBDatabase& database) override
446 {
447 if (!requestCallback().isActive())
448 return;
449
450 auto idbTransaction = transactionForDatabase(&database, m_objectStoreName);
451 if (!idbTransaction) {
452 m_requestCallback->sendFailure("Could not get transaction");
453 return;
454 }
455
456 auto idbObjectStore = objectStoreForTransaction(idbTransaction.get(), m_objectStoreName);
457 if (!idbObjectStore) {
458 m_requestCallback->sendFailure("Could not get object store");
459 return;
460 }
461
462 TransactionActivator activator(idbTransaction.get());
463 RefPtr<IDBRequest> idbRequest;
464 auto* exec = context() ? context()->globalObject() : nullptr;
465 if (!m_indexName.isEmpty()) {
466 auto idbIndex = indexForObjectStore(idbObjectStore.get(), m_indexName);
467 if (!idbIndex) {
468 m_requestCallback->sendFailure("Could not get index");
469 return;
470 }
471
472 if (exec) {
473 auto result = idbIndex->openCursor(*exec, m_idbKeyRange.get(), IDBCursorDirection::Next);
474 if (!result.hasException())
475 idbRequest = result.releaseReturnValue();
476 }
477 } else {
478 if (exec) {
479 auto result = idbObjectStore->openCursor(*exec, m_idbKeyRange.get(), IDBCursorDirection::Next);
480 if (!result.hasException())
481 idbRequest = result.releaseReturnValue();
482 }
483 }
484
485 if (!idbRequest) {
486 m_requestCallback->sendFailure("Could not open cursor to populate database data");
487 return;
488 }
489
490 auto openCursorCallback = OpenCursorCallback::create(m_injectedScript, m_requestCallback.copyRef(), m_skipCount, m_pageSize);
491 idbRequest->addEventListener(eventNames().successEvent, WTFMove(openCursorCallback), false);
492 }
493
494 BackendDispatcher::CallbackBase& requestCallback() override { return m_requestCallback.get(); }
495 DataLoader(ScriptExecutionContext* scriptExecutionContext, Ref<IndexedDBBackendDispatcherHandler::RequestDataCallback>&& requestCallback, const InjectedScript& injectedScript, const String& objectStoreName, const String& indexName, RefPtr<IDBKeyRange> idbKeyRange, int skipCount, unsigned pageSize)
496 : ExecutableWithDatabase(scriptExecutionContext)
497 , m_requestCallback(WTFMove(requestCallback))
498 , m_injectedScript(injectedScript)
499 , m_objectStoreName(objectStoreName)
500 , m_indexName(indexName)
501 , m_idbKeyRange(WTFMove(idbKeyRange))
502 , m_skipCount(skipCount)
503 , m_pageSize(pageSize) { }
504 Ref<IndexedDBBackendDispatcherHandler::RequestDataCallback> m_requestCallback;
505 InjectedScript m_injectedScript;
506 String m_objectStoreName;
507 String m_indexName;
508 RefPtr<IDBKeyRange> m_idbKeyRange;
509 int m_skipCount;
510 unsigned m_pageSize;
511};

544static Protocol::ErrorStringOr<Document*> documentFromFrame(Frame* frame)
545{
546 Document* document = frame ? frame->document() : nullptr;
547 if (!document)
548 return makeUnexpected("Missing document for given frame"_s);
549

553static Protocol::ErrorStringOr<IDBFactory*> IDBFactoryFromDocument(Document* document)
554{
555 DOMWindow* domWindow = document->domWindow();
556 if (!domWindow)
557 return makeUnexpected("Missing window for given document"_s);
558
559 IDBFactory* idbFactory = WindowOrWorkerGlobalScopeIndexedDatabase::indexedDB(*domWindow);
560 if (!idbFactory)
561 makeUnexpected("Missing IndexedDB factory of window for given document"_s);
562
563 return idbFactory;
564}

566static bool getDocumentAndIDBFactoryFromFrameOrSendFailure(Frame* frame, Document*& out_document, IDBFactory*& out_idbFactory, BackendDispatcher::CallbackBase& callback)
567{
568 Protocol::ErrorStringOr<Document*> document = documentFromFrame(frame);
569 if (!document.has_value()) {
570 callback.sendFailure(document.error());
571 return false;
572 }
573
574 Protocol::ErrorStringOr<IDBFactory*> idbFactory = IDBFactoryFromDocument(document.value());
575 if (!idbFactory.has_value()) {
576 callback.sendFailure(idbFactory.error());
577 return false;
578 }
579
580 out_document = document.value();
581 out_idbFactory = idbFactory.value();
582 return true;
583}

585void InspectorIndexedDBAgent::requestDatabaseNames(const String& securityOrigin, Ref<RequestDatabaseNamesCallback>&& callback)
586{
587 auto* frame = InspectorPageAgent::findFrameWithSecurityOrigin(m_inspectedPage, securityOrigin);
588 Document* document;
589 IDBFactory* idbFactory;
590 if (!getDocumentAndIDBFactoryFromFrameOrSendFailure(frame, document, idbFactory, callback))
591 return;
592
593 idbFactory->getAllDatabaseNames(*document, [callback = WTFMove(callback)](auto& databaseNames) {
594 if (!callback->isActive())
595 return;
596
597 Ref<JSON::ArrayOf<String>> databaseNameArray = JSON::ArrayOf<String>::create();
598 for (auto& databaseName : databaseNames)
599 databaseNameArray->addItem(databaseName);
600
601 callback->sendSuccess(WTFMove(databaseNameArray));
602 });
603}

617void InspectorIndexedDBAgent::requestData(const String& securityOrigin, const String& databaseName, const String& objectStoreName, const String& indexName, int skipCount, int pageSize, RefPtr<JSON::Object>&& keyRange, Ref<RequestDataCallback>&& callback)
618{

631
632 Ref<DataLoader> dataLoader = DataLoader::create(document, WTFMove(callback), injectedScript, objectStoreName, indexName, WTFMove(idbKeyRange), skipCount, pageSize);
633 dataLoader->start(idbFactory, &document->securityOrigin(), databaseName);
634}

697 }
698
699 auto idbObjectStore = objectStoreForTransaction(idbTransaction.get(), m_objectStoreName);
700 if (!idbObjectStore) {
701 m_requestCallback->sendFailure("Could not get object store");
702 return;
703 }
704
705 TransactionActivator activator(idbTransaction.get());
706 RefPtr<IDBRequest> idbRequest;
707 if (auto* exec = context() ? context()->globalObject() : nullptr) {
708 auto result = idbObjectStore->clear(*exec);
709 ASSERT(!result.hasException());
710 if (result.hasException()) {
711 m_requestCallback->sendFailure(makeString("Could not clear object store '", m_objectStoreName, "': ", static_cast<int>(result.releaseException().code())));
712 return;
713 }
714 idbRequest = result.releaseReturnValue();
715 }
716
717 idbTransaction->addEventListener(eventNames().completeEvent, ClearObjectStoreListener::create(m_requestCallback.copyRef()), false);
718 }
719
720 BackendDispatcher::CallbackBase& requestCallback() override { return m_requestCallback.get(); }
721private:
722 const String m_objectStoreName;
723 Ref<IndexedDBBackendDispatcherHandler::ClearObjectStoreCallback> m_requestCallback;
724};
