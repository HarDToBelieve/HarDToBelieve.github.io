112Protocol::ErrorStringOr<Ref<JSON::ArrayOf<Protocol::LayerTree::Layer>>> InspectorLayerTreeAgent::layersForNode(Protocol::DOM::NodeId nodeId)
113{
114 auto* node = m_instrumentingAgents.persistentDOMAgent()->nodeForId(nodeId);
115 if (!node)
116 return makeUnexpected("Missing node for given nodeId"_s);
117
118 auto* renderer = node->renderer();
119 if (!renderer)
120 return makeUnexpected("Missing renderer of node for given nodeId"_s);
121
122 if (!is<RenderElement>(*renderer))
123 return makeUnexpected("Missing renderer of element for given nodeId"_s);
124
125 auto layers = JSON::ArrayOf<Protocol::LayerTree::Layer>::create();
126
127 gatherLayersUsingRenderObjectHierarchy(downcast<RenderElement>(*renderer), layers);
128
129 m_suppressLayerChangeEvents = false;
130
131 return layers;
132}
133
134void InspectorLayerTreeAgent::gatherLayersUsingRenderObjectHierarchy(RenderElement& renderer, JSON::ArrayOf<Protocol::LayerTree::Layer>& layers)
135{
136 if (renderer.hasLayer()) {
137 gatherLayersUsingRenderLayerHierarchy(downcast<RenderLayerModelObject>(renderer).layer(), layers);
138 return;
139 }
140
141 for (auto& child : childrenOfType<RenderElement>(renderer))
142 gatherLayersUsingRenderObjectHierarchy(child, layers);
143}
144
145void InspectorLayerTreeAgent::gatherLayersUsingRenderLayerHierarchy(RenderLayer* renderLayer, JSON::ArrayOf<Protocol::LayerTree::Layer>& layers)
146{
147 if (renderLayer->isComposited())
148 layers.addItem(buildObjectForLayer(renderLayer));
149
150 for (renderLayer = renderLayer->firstChild(); renderLayer; renderLayer = renderLayer->nextSibling())
151 gatherLayersUsingRenderLayerHierarchy(renderLayer, layers);
152}

154Ref<Protocol::LayerTree::Layer> InspectorLayerTreeAgent::buildObjectForLayer(RenderLayer* renderLayer)
155{
156 RenderObject* renderer = &renderLayer->renderer();
157 RenderLayerBacking* backing = renderLayer->backing();
158 Node* node = renderer->node();
159
160 bool isReflection = renderLayer->isReflection();
161 bool isGenerated = (isReflection ? renderer->parent() : renderer)->isBeforeOrAfterContent();
162 bool isAnonymous = renderer->isAnonymous();
163
164 if (renderer->isRenderView())
165 node = &renderer->document();
166 else if (isReflection && isGenerated)
167 node = renderer->parent()->generatingElement();
168 else if (isGenerated)
169 node = renderer->generatingNode();
170 else if (isReflection || isAnonymous)
171 node = renderer->parent()->element();
172
173 // Basic set of properties.
174 auto layerObject = Protocol::LayerTree::Layer::create()
175 .setLayerId(bind(renderLayer))
176 .setNodeId(idForNode(node))
177 .setBounds(buildObjectForIntRect(renderer->absoluteBoundingBoxRect()))
178 .setMemory(backing->backingStoreMemoryEstimate())
179 .setCompositedBounds(buildObjectForIntRect(enclosingIntRect(backing->compositedBounds())))
180 .setPaintCount(backing->graphicsLayer()->repaintCount())
181 .release();
182
183 if (node && node->shadowHost())
184 layerObject->setIsInShadowTree(true);
185
186 if (isReflection)
187 layerObject->setIsReflection(true);
188
189 if (isGenerated) {
190 if (isReflection)
191 renderer = renderer->parent();
192 layerObject->setIsGeneratedContent(true);
193 layerObject->setPseudoElementId(bindPseudoElement(downcast<PseudoElement>(renderer->node())));
194 if (renderer->isBeforeContent())
195 layerObject->setPseudoElement("before");
196 else if (renderer->isAfterContent())
197 layerObject->setPseudoElement("after");
198 }
199
200 // FIXME: RenderView is now really anonymous but don't tell about it to the frontend before making sure it can handle it.
201 if (isAnonymous && !renderer->isRenderView()) {
202 layerObject->setIsAnonymous(true);
203 const RenderStyle& style = renderer->style();
204 if (style.styleType() == PseudoId::FirstLetter)
205 layerObject->setPseudoElement("first-letter");
206 else if (style.styleType() == PseudoId::FirstLine)
207 layerObject->setPseudoElement("first-line");
208 }
209
210 return layerObject;
211}

213Protocol::DOM::NodeId InspectorLayerTreeAgent::idForNode(Node* node)
214{
215 if (!node)
216 return 0;
217
218 InspectorDOMAgent* domAgent = m_instrumentingAgents.persistentDOMAgent();
219
220 auto nodeId = domAgent->boundNodeId(node);
221 if (!nodeId) {
222 // FIXME: <https://webkit.org/b/213499> Web Inspector: allow DOM nodes to be instrumented at any point, regardless of whether the main document has also been instrumented
223 nodeId = domAgent->pushNodeToFrontend(node);
224 }
225
226 return nodeId;
227}

239Protocol::ErrorStringOr<Ref<Protocol::LayerTree::CompositingReasons>> InspectorLayerTreeAgent::reasonsForCompositingLayer(const Protocol::LayerTree::LayerId& layerId)
240{
241 const RenderLayer* renderLayer = m_idToLayer.get(layerId);
242
243 if (!renderLayer)
244 return makeUnexpected("Missing render layer for given layerId"_s);
245
246 OptionSet<CompositingReason> reasons = renderLayer->compositor().reasonsForCompositing(*renderLayer);
247 auto compositingReasons = Protocol::LayerTree::CompositingReasons::create().release();
248
249 if (reasons.contains(CompositingReason::Transform3D))
250 compositingReasons->setTransform3D(true);
251
252 if (reasons.contains(CompositingReason::Video))
253 compositingReasons->setVideo(true);
254 else if (reasons.contains(CompositingReason::Canvas))
255 compositingReasons->setCanvas(true);
256 else if (reasons.contains(CompositingReason::Plugin))
257 compositingReasons->setPlugin(true);
258 else if (reasons.contains(CompositingReason::IFrame))
259 compositingReasons->setIFrame(true);
260 else if (reasons.contains(CompositingReason::Model))
261 compositingReasons->setModel(true);
262
263 if (reasons.contains(CompositingReason::BackfaceVisibilityHidden))
264 compositingReasons->setBackfaceVisibilityHidden(true);
265
266 if (reasons.contains(CompositingReason::ClipsCompositingDescendants))
267 compositingReasons->setClipsCompositingDescendants(true);
268
269 if (reasons.contains(CompositingReason::Animation))
270 compositingReasons->setAnimation(true);
271
272 if (reasons.contains(CompositingReason::Filters))
273 compositingReasons->setFilters(true);
274
275 if (reasons.contains(CompositingReason::PositionFixed))
276 compositingReasons->setPositionFixed(true);
277
278 if (reasons.contains(CompositingReason::PositionSticky))
279 compositingReasons->setPositionSticky(true);
280
281 if (reasons.contains(CompositingReason::OverflowScrolling))
282 compositingReasons->setOverflowScrollingTouch(true);
283
284 // FIXME: handle OverflowScrollPositioning (webkit.org/b/195985).
285
286 if (reasons.contains(CompositingReason::Stacking))
287 compositingReasons->setStacking(true);
288
289 if (reasons.contains(CompositingReason::Overlap))
290 compositingReasons->setOverlap(true);
291
292 if (reasons.contains(CompositingReason::NegativeZIndexChildren))
293 compositingReasons->setNegativeZIndexChildren(true);
294
295 if (reasons.contains(CompositingReason::TransformWithCompositedDescendants))
296 compositingReasons->setTransformWithCompositedDescendants(true);
297
298 if (reasons.contains(CompositingReason::OpacityWithCompositedDescendants))
299 compositingReasons->setOpacityWithCompositedDescendants(true);
300
301 if (reasons.contains(CompositingReason::MaskWithCompositedDescendants))
302 compositingReasons->setMaskWithCompositedDescendants(true);
303
304 if (reasons.contains(CompositingReason::ReflectionWithCompositedDescendants))
305 compositingReasons->setReflectionWithCompositedDescendants(true);
306
307 if (reasons.contains(CompositingReason::FilterWithCompositedDescendants))
308 compositingReasons->setFilterWithCompositedDescendants(true);
309
310 if (reasons.contains(CompositingReason::BlendingWithCompositedDescendants))
311 compositingReasons->setBlendingWithCompositedDescendants(true);
312
313 if (reasons.contains(CompositingReason::IsolatesCompositedBlendingDescendants))
314 compositingReasons->setIsolatesCompositedBlendingDescendants(true);
315
316 if (reasons.contains(CompositingReason::Perspective))
317 compositingReasons->setPerspective(true);
318
319 if (reasons.contains(CompositingReason::Preserve3D))
320 compositingReasons->setPreserve3D(true);
321
322 if (reasons.contains(CompositingReason::WillChange))
323 compositingReasons->setWillChange(true);
324
325 if (reasons.contains(CompositingReason::Root))
326 compositingReasons->setRoot(true);
327
328 return compositingReasons;
329}

331String InspectorLayerTreeAgent::bind(const RenderLayer* layer)
332{
333 if (!layer)
334 return emptyString();
335 return m_documentLayerToIdMap.ensure(layer, [this, layer] {
336 auto identifier = IdentifiersFactory::createIdentifier();
337 m_idToLayer.set(identifier, layer);
338 return identifier;
339 }).iterator->value;
340}
341
342void InspectorLayerTreeAgent::unbind(const RenderLayer* layer)
343{
344 auto identifier = m_documentLayerToIdMap.take(layer);
345 if (identifier.isNull())
346 return;
347 m_idToLayer.remove(identifier);
348}

350String InspectorLayerTreeAgent::bindPseudoElement(PseudoElement* pseudoElement)
351{
352 if (!pseudoElement)
353 return emptyString();
354 return m_pseudoElementToIdMap.ensure(*pseudoElement, [this, pseudoElement] {
355 auto identifier = IdentifiersFactory::createIdentifier();
356 m_idToPseudoElement.set(identifier, makeWeakPtr(pseudoElement));
357 return identifier;
358 }).iterator->value;
359}
360
361void InspectorLayerTreeAgent::unbindPseudoElement(PseudoElement* pseudoElement)
362{
363 if (!pseudoElement)
364 return;
365 auto identifier = m_pseudoElementToIdMap.take(*pseudoElement);
366 if (identifier.isNull())
367 return;
368 m_idToPseudoElement.remove(identifier);
369}
