41InspectorMemoryAgent::InspectorMemoryAgent(PageAgentContext& context)
42 : InspectorAgentBase("Memory"_s, context)
43 , m_frontendDispatcher(makeUnique<Inspector::MemoryFrontendDispatcher>(context.frontendRouter))
44 , m_backendDispatcher(Inspector::MemoryBackendDispatcher::create(context.backendDispatcher, this))
45{
46}

48InspectorMemoryAgent::~InspectorMemoryAgent() = default;
49
50void InspectorMemoryAgent::didCreateFrontendAndBackend(FrontendRouter*, BackendDispatcher*)
51{
52 m_instrumentingAgents.setPersistentMemoryAgent(this);
53}
54
55void InspectorMemoryAgent::willDestroyFrontendAndBackend(DisconnectReason)
56{
57 disable();
58
59 m_instrumentingAgents.setPersistentMemoryAgent(nullptr);
60}

62Protocol::ErrorStringOr<void> InspectorMemoryAgent::enable()
63{
64 if (m_instrumentingAgents.enabledMemoryAgent() == this)
65 return makeUnexpected("Memory domain already enabled"_s);
66
67 m_instrumentingAgents.setEnabledMemoryAgent(this);
68
69 return { };
70}
71
72Protocol::ErrorStringOr<void> InspectorMemoryAgent::disable()
73{
74 if (m_instrumentingAgents.enabledMemoryAgent() != this)
75 return makeUnexpected("Memory domain already disabled"_s);
76
77 m_instrumentingAgents.setEnabledMemoryAgent(nullptr);
78
79 m_tracking = false;
80
81 ResourceUsageThread::removeObserver(this);
82
83 return { };
84}
85
86Protocol::ErrorStringOr<void> InspectorMemoryAgent::startTracking()
87{
88 if (m_tracking)
89 return { };
90
91 ResourceUsageThread::addObserver(this, Memory, [this] (const ResourceUsageData& data) {
92 collectSample(data);
93 });
94
95 m_tracking = true;
96
97 m_frontendDispatcher->trackingStart(m_environment.executionStopwatch().elapsedTime().seconds());
98
99 return { };
100}
101
102Protocol::ErrorStringOr<void> InspectorMemoryAgent::stopTracking()
103{
104 if (!m_tracking)
105 return { };
106
107 ResourceUsageThread::removeObserver(this);
108
109 m_tracking = false;
110
111 m_frontendDispatcher->trackingComplete(m_environment.executionStopwatch().elapsedTime().seconds());
112
113 return { };
114}

116void InspectorMemoryAgent::didHandleMemoryPressure(Critical critical)
117{
118 MemoryFrontendDispatcher::Severity severity = critical == Critical::Yes ? MemoryFrontendDispatcher::Severity::Critical : MemoryFrontendDispatcher::Severity::NonCritical;
119 m_frontendDispatcher->memoryPressure(m_environment.executionStopwatch().elapsedTime().seconds(), Protocol::Helpers::getEnumConstantValue(severity));
120}

122void InspectorMemoryAgent::collectSample(const ResourceUsageData& data)
123{
124 auto javascriptCategory = Protocol::Memory::CategoryData::create()
125 .setType(Protocol::Memory::CategoryData::Type::JavaScript)
126 .setSize(data.categories[MemoryCategory::GCHeap].totalSize() + data.categories[MemoryCategory::GCOwned].totalSize())
127 .release();
128
129 auto jitCategory = Protocol::Memory::CategoryData::create()
130 .setType(Protocol::Memory::CategoryData::Type::JIT)
131 .setSize(data.categories[MemoryCategory::JSJIT].totalSize())
132 .release();
133
134 auto imagesCategory = Protocol::Memory::CategoryData::create()
135 .setType(Protocol::Memory::CategoryData::Type::Images)
136 .setSize(data.categories[MemoryCategory::Images].totalSize())
137 .release();
138
139 auto layersCategory = Protocol::Memory::CategoryData::create()
140 .setType(Protocol::Memory::CategoryData::Type::Layers)
141 .setSize(data.categories[MemoryCategory::Layers].totalSize())
142 .release();
143
144 auto pageCategory = Protocol::Memory::CategoryData::create()
145 .setType(Protocol::Memory::CategoryData::Type::Page)
146 .setSize(data.categories[MemoryCategory::bmalloc].totalSize() + data.categories[MemoryCategory::LibcMalloc].totalSize())
147 .release();
148
149 auto otherCategory = Protocol::Memory::CategoryData::create()
150 .setType(Protocol::Memory::CategoryData::Type::Other)
151 .setSize(data.categories[MemoryCategory::Other].totalSize())
152 .release();
153
154 auto categories = JSON::ArrayOf<Protocol::Memory::CategoryData>::create();
155 categories->addItem(WTFMove(javascriptCategory));
156 categories->addItem(WTFMove(jitCategory));
157 categories->addItem(WTFMove(imagesCategory));
158 categories->addItem(WTFMove(layersCategory));
159 categories->addItem(WTFMove(pageCategory));
160 categories->addItem(WTFMove(otherCategory));
161
162 auto event = Protocol::Memory::Event::create()
163 .setTimestamp(m_environment.executionStopwatch().elapsedTimeSince(data.timestamp).seconds())
164 .setCategories(WTFMove(categories))
165 .release();
166
167 m_frontendDispatcher->trackingUpdate(WTFMove(event));
168}
