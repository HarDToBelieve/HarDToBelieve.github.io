101class InspectorThreadableLoaderClient final : public ThreadableLoaderClient {
102 WTF_MAKE_NONCOPYABLE(InspectorThreadableLoaderClient);
103public:
104 InspectorThreadableLoaderClient(RefPtr<LoadResourceCallback>&& callback)
105 : m_callback(WTFMove(callback))
106 {
107 }
108
109 ~InspectorThreadableLoaderClient() override = default;
110
111 void didReceiveResponse(unsigned long, const ResourceResponse& response) override
112 {
113 m_mimeType = response.mimeType();
114 m_statusCode = response.httpStatusCode();
115
116 // FIXME: This assumes text only responses. We should support non-text responses as well.
117 TextEncoding textEncoding(response.textEncodingName());
118 bool useDetector = false;
119 if (!textEncoding.isValid()) {
120 textEncoding = UTF8Encoding();
121 useDetector = true;
122 }
123
124 m_decoder = TextResourceDecoder::create("text/plain"_s, textEncoding, useDetector);
125 }
126
127 void didReceiveData(const uint8_t* data, int dataLength) override
128 {
129 if (!dataLength)
130 return;
131
132 if (dataLength == -1)
133 dataLength = strlen(reinterpret_cast<const char*>(data));
134
135 m_responseText.append(m_decoder->decode(data, dataLength));
136 }
137
138 void didFinishLoading(unsigned long) override
139 {
140 if (m_decoder)
141 m_responseText.append(m_decoder->flush());
142
143 m_callback->sendSuccess(m_responseText.toString(), m_mimeType, m_statusCode);
144 dispose();
145 }
146
147 void didFail(const ResourceError& error) override
148 {
149 m_callback->sendFailure(error.isAccessControl() ? "Loading resource for inspector failed access control check"_s : "Loading resource for inspector failed"_s);
150 dispose();
151 }
152
153 void setLoader(RefPtr<ThreadableLoader>&& loader)
154 {
155 m_loader = WTFMove(loader);
156 }
157
158private:
159 void dispose()
160 {
161 m_loader = nullptr;
162 delete this;
163 }
164
165 RefPtr<LoadResourceCallback> m_callback;
166 RefPtr<ThreadableLoader> m_loader;
167 RefPtr<TextResourceDecoder> m_decoder;
168 String m_mimeType;
169 StringBuilder m_responseText;
170 int m_statusCode;
171};

205static Ref<Protocol::Network::Headers> buildObjectForHeaders(const HTTPHeaderMap& headers)
206{
207 auto headersValue = Protocol::Network::Headers::create().release();
208
209 auto headersObject = headersValue->asObject();
210 for (const auto& header : headers)
211 headersObject->setString(header.key, header.value);
212
213 return headersValue;
214}

216Ref<Protocol::Network::ResourceTiming> InspectorNetworkAgent::buildObjectForTiming(const NetworkLoadMetrics& timing, ResourceLoader& resourceLoader)
217{
218 auto elapsedTimeSince = [&] (const MonotonicTime& time) {
219 return m_environment.executionStopwatch().elapsedTimeSince(time).seconds();
220 };
221 auto millisecondsSinceFetchStart = [&] (const MonotonicTime& time) {
222 if (!time)
223 return 0.0;
224 return (time - timing.fetchStart).milliseconds();
225 };
226
227 return Protocol::Network::ResourceTiming::create()
228 .setStartTime(elapsedTimeSince(resourceLoader.loadTiming().startTime()))
229 .setRedirectStart(elapsedTimeSince(timing.redirectStart))
230 .setRedirectEnd(elapsedTimeSince(timing.fetchStart))
231 .setFetchStart(elapsedTimeSince(timing.fetchStart))
232 .setDomainLookupStart(millisecondsSinceFetchStart(timing.domainLookupStart))
233 .setDomainLookupEnd(millisecondsSinceFetchStart(timing.domainLookupEnd))
234 .setConnectStart(millisecondsSinceFetchStart(timing.connectStart))
235 .setConnectEnd(millisecondsSinceFetchStart(timing.connectEnd))
236 .setSecureConnectionStart(millisecondsSinceFetchStart(timing.secureConnectionStart))
237 .setRequestStart(millisecondsSinceFetchStart(timing.requestStart))
238 .setResponseStart(millisecondsSinceFetchStart(timing.responseStart))
239 .setResponseEnd(millisecondsSinceFetchStart(timing.responseEnd))
240 .release();
241}

243static Protocol::Network::Metrics::Priority toProtocol(NetworkLoadPriority priority)
244{
245 switch (priority) {
246 case NetworkLoadPriority::Low:
247 return Protocol::Network::Metrics::Priority::Low;
248 case NetworkLoadPriority::Medium:
249 return Protocol::Network::Metrics::Priority::Medium;
250 case NetworkLoadPriority::High:
251 return Protocol::Network::Metrics::Priority::High;
252 case NetworkLoadPriority::Unknown:
253 break;
254 }
255
256 ASSERT_NOT_REACHED();
257 return Protocol::Network::Metrics::Priority::Medium;

260Ref<Protocol::Network::Metrics> InspectorNetworkAgent::buildObjectForMetrics(const NetworkLoadMetrics& networkLoadMetrics)
261{
262 auto metrics = Protocol::Network::Metrics::create().release();
263
264 if (!networkLoadMetrics.protocol.isNull())
265 metrics->setProtocol(networkLoadMetrics.protocol);
266 if (networkLoadMetrics.priority != NetworkLoadPriority::Unknown)
267 metrics->setPriority(toProtocol(networkLoadMetrics.priority));
268 if (!networkLoadMetrics.remoteAddress.isNull())
269 metrics->setRemoteAddress(networkLoadMetrics.remoteAddress);
270 if (!networkLoadMetrics.connectionIdentifier.isNull())
271 metrics->setConnectionIdentifier(networkLoadMetrics.connectionIdentifier);
272 if (!networkLoadMetrics.requestHeaders.isEmpty())
273 metrics->setRequestHeaders(buildObjectForHeaders(networkLoadMetrics.requestHeaders));
274
275 if (networkLoadMetrics.requestHeaderBytesSent != std::numeric_limits<uint64_t>::max())
276 metrics->setRequestHeaderBytesSent(networkLoadMetrics.requestHeaderBytesSent);
277 if (networkLoadMetrics.requestBodyBytesSent != std::numeric_limits<uint64_t>::max())
278 metrics->setRequestBodyBytesSent(networkLoadMetrics.requestBodyBytesSent);
279 if (networkLoadMetrics.responseHeaderBytesReceived != std::numeric_limits<uint64_t>::max())
280 metrics->setResponseHeaderBytesReceived(networkLoadMetrics.responseHeaderBytesReceived);
281 if (networkLoadMetrics.responseBodyBytesReceived != std::numeric_limits<uint64_t>::max())
282 metrics->setResponseBodyBytesReceived(networkLoadMetrics.responseBodyBytesReceived);
283 if (networkLoadMetrics.responseBodyDecodedSize != std::numeric_limits<uint64_t>::max())
284 metrics->setResponseBodyDecodedSize(networkLoadMetrics.responseBodyDecodedSize);
285
286 auto connectionPayload = Protocol::Security::Connection::create()
287 .release();
288
289 if (!networkLoadMetrics.tlsProtocol.isEmpty())
290 connectionPayload->setProtocol(networkLoadMetrics.tlsProtocol);
291
292 if (!networkLoadMetrics.tlsCipher.isEmpty())
293 connectionPayload->setCipher(networkLoadMetrics.tlsCipher);
294
295 metrics->setSecurityConnection(WTFMove(connectionPayload));
296
297 return metrics;
298}

300static Ref<Protocol::Network::Request> buildObjectForResourceRequest(const ResourceRequest& request)
301{
302 auto requestObject = Protocol::Network::Request::create()
303 .setUrl(request.url().string())
304 .setMethod(request.httpMethod())
305 .setHeaders(buildObjectForHeaders(request.httpHeaderFields()))
306 .release();
307 if (request.httpBody() && !request.httpBody()->isEmpty()) {
308 auto bytes = request.httpBody()->flatten();
309 requestObject->setPostData(String::fromUTF8WithLatin1Fallback(bytes.data(), bytes.size()));
310 }
311 return requestObject;
312}

315{
316 switch (source) {
317 case ResourceResponse::Source::DOMCache:
318 case ResourceResponse::Source::ApplicationCache:

320 case ResourceResponse::Source::Unknown:
321 return Protocol::Network::Response::Source::Unknown;
322 case ResourceResponse::Source::Network:
323 return Protocol::Network::Response::Source::Network;
324 case ResourceResponse::Source::MemoryCache:
325 case ResourceResponse::Source::MemoryCacheAfterValidation:
326 return Protocol::Network::Response::Source::MemoryCache;
327 case ResourceResponse::Source::DiskCache:
328 case ResourceResponse::Source::DiskCacheAfterValidation:
329 return Protocol::Network::Response::Source::DiskCache;

340RefPtr<Protocol::Network::Response> InspectorNetworkAgent::buildObjectForResourceResponse(const ResourceResponse& response, ResourceLoader* resourceLoader)
341{
342 if (response.isNull())
343 return nullptr;
344
345 auto responseObject = Protocol::Network::Response::create()
346 .setUrl(response.url().string())
347 .setStatus(response.httpStatusCode())
348 .setStatusText(response.httpStatusText())
349 .setHeaders(buildObjectForHeaders(response.httpHeaderFields()))
350 .setMimeType(response.mimeType())
351 .setSource(responseSource(response.source()))
352 .release();
353
354 if (resourceLoader) {
355 auto* metrics = response.deprecatedNetworkLoadMetricsOrNull();
356 responseObject->setTiming(buildObjectForTiming(metrics ? *metrics : NetworkLoadMetrics { }, *resourceLoader));
357 }
358
359 if (auto& certificateInfo = response.certificateInfo()) {
360 auto securityPayload = Protocol::Security::Security::create()
361 .release();
362
363 if (auto certificateSummaryInfo = certificateInfo.value().summary()) {
364 auto certificatePayload = Protocol::Security::Certificate::create()
365 .release();
366
367 certificatePayload->setSubject(certificateSummaryInfo.value().subject);
368
369 if (auto validFrom = certificateSummaryInfo.value().validFrom)
370 certificatePayload->setValidFrom(validFrom.seconds());
371
372 if (auto validUntil = certificateSummaryInfo.value().validUntil)
373 certificatePayload->setValidUntil(validUntil.seconds());
374
375 auto dnsNamesPayload = JSON::ArrayOf<String>::create();
376 for (auto& dnsName : certificateSummaryInfo.value().dnsNames)
377 dnsNamesPayload->addItem(dnsName);
378 if (dnsNamesPayload->length())
379 certificatePayload->setDnsNames(WTFMove(dnsNamesPayload));
380
381 auto ipAddressesPayload = JSON::ArrayOf<String>::create();
382 for (auto& ipAddress : certificateSummaryInfo.value().ipAddresses)
383 ipAddressesPayload->addItem(ipAddress);
384 if (ipAddressesPayload->length())
385 certificatePayload->setIpAddresses(WTFMove(ipAddressesPayload));
386
387 securityPayload->setCertificate(WTFMove(certificatePayload));
388 }
389
390 responseObject->setSecurity(WTFMove(securityPayload));
391 }
392
393 return responseObject;
394}

419void InspectorNetworkAgent::willSendRequest(unsigned long identifier, DocumentLoader* loader, ResourceRequest& request, const ResourceResponse& redirectResponse, InspectorPageAgent::ResourceType type)
420{
421 if (request.hiddenFromInspector()) {
422 m_hiddenRequestIdentifiers.add(identifier);
423 return;
424 }
425
426 double sendTimestamp = timestamp();
427 WallTime walltime = WallTime::now();
428
429 auto requestId = IdentifiersFactory::requestId(identifier);
430 auto frameId = frameIdentifier(loader);
431 auto loaderId = loaderIdentifier(loader);
432 String targetId = request.initiatorIdentifier();
433
434 if (type == InspectorPageAgent::OtherResource) {
435 if (m_loadingXHRSynchronously || request.requester() == ResourceRequest::Requester::XHR)
436 type = InspectorPageAgent::XHRResource;
437 else if (request.requester() == ResourceRequest::Requester::Fetch)
438 type = InspectorPageAgent::FetchResource;
439 else if (loader && equalIgnoringFragmentIdentifier(request.url(), loader->url()) && !loader->isCommitted())
440 type = InspectorPageAgent::DocumentResource;
441 else if (loader) {
442 for (auto& linkIcon : loader->linkIcons()) {
443 if (equalIgnoringFragmentIdentifier(request.url(), linkIcon.url)) {
444 type = InspectorPageAgent::ImageResource;
445 break;
446 }
447 }
448 }
449 }
450
451 m_resourcesData->resourceCreated(requestId, loaderId, type);
452
453 for (auto& entry : m_extraRequestHeaders)
454 request.setHTTPHeaderField(entry.key, entry.value);
455
456 auto protocolResourceType = InspectorPageAgent::resourceTypeJSON(type);
457
458 Document* document = loader && loader->frame() ? loader->frame()->document() : nullptr;
459 auto initiatorObject = buildInitiatorObject(document, &request);
460
461 String url = loader ? loader->url().string() : request.url().string();
462 std::optional<Protocol::Page::ResourceType> typePayload;
463 if (type != InspectorPageAgent::OtherResource)
464 typePayload = protocolResourceType;
465 m_frontendDispatcher->requestWillBeSent(requestId, frameId, loaderId, url, buildObjectForResourceRequest(request), sendTimestamp, walltime.secondsSinceEpoch().seconds(), WTFMove(initiatorObject), buildObjectForResourceResponse(redirectResponse, nullptr), WTFMove(typePayload), targetId);
466}

501void InspectorNetworkAgent::didReceiveResponse(unsigned long identifier, DocumentLoader* loader, const ResourceResponse& response, ResourceLoader* resourceLoader)
502{
503 if (m_hiddenRequestIdentifiers.contains(identifier))
504 return;
505
506 String requestId = IdentifiersFactory::requestId(identifier);
507
508 std::optional<ResourceResponse> realResponse;
509 if (platformStrategies()->loaderStrategy()->havePerformedSecurityChecks(response)) {
510 callOnMainThreadAndWait([&] {
511 // We do not need to isolate response since it comes straight from IPC, but we might want to isolate it for extra safety.
512 auto response = platformStrategies()->loaderStrategy()->responseFromResourceLoadIdentifier(identifier);
513 if (!response.isNull())
514 realResponse = WTFMove(response);
515 });
516 }
517
518 auto resourceResponse = buildObjectForResourceResponse(realResponse ? *realResponse : response, resourceLoader);
519 ASSERT(resourceResponse);
520
521 bool isNotModified = response.httpStatusCode() == 304;
522
523 CachedResource* cachedResource = nullptr;
524 if (is<SubresourceLoader>(resourceLoader) && !isNotModified)
525 cachedResource = downcast<SubresourceLoader>(resourceLoader)->cachedResource();
526 if (!cachedResource && loader)
527 cachedResource = InspectorPageAgent::cachedResource(loader->frame(), response.url());
528
529 if (cachedResource) {
530 // Use mime type from cached resource in case the one in response is empty.
531 if (resourceResponse && response.mimeType().isEmpty())
532 resourceResponse->setString(Protocol::Network::Response::mimeTypeKey, cachedResource->response().mimeType());
533 m_resourcesData->addCachedResource(requestId, cachedResource);
534 }
535
536 InspectorPageAgent::ResourceType type = m_resourcesData->resourceType(requestId);
537 InspectorPageAgent::ResourceType newType = cachedResource ? InspectorPageAgent::inspectorResourceType(*cachedResource) : type;
538
539 // FIXME: XHRResource is returned for CachedResource::Type::RawResource, it should be OtherResource unless it truly is an XHR.
540 // RawResource is used for loading worker scripts, and those should stay as ScriptResource and not change to XHRResource.
541 if (type != newType && newType != InspectorPageAgent::XHRResource && newType != InspectorPageAgent::OtherResource)
542 type = newType;
543
544 // FIXME: <webkit.org/b/216125> 304 Not Modified responses for XHR/Fetch do not have all their information from the cache.
545 if (isNotModified && (type == InspectorPageAgent::XHRResource || type == InspectorPageAgent::FetchResource) && (!cachedResource || !cachedResource->encodedSize())) {
546 if (auto previousResourceData = m_resourcesData->dataForURL(response.url().string())) {
547 if (previousResourceData->hasContent())
548 m_resourcesData->setResourceContent(requestId, previousResourceData->content(), previousResourceData->base64Encoded());
549 else if (previousResourceData->hasBufferedData()) {
550 auto previousBuffer = previousResourceData->buffer();
551 previousBuffer->forEachSegment([&](auto& segment) {
552 m_resourcesData->maybeAddResourceData(requestId, segment.data(), segment.size());
553 });
554 }
555
556 resourceResponse->setString(Protocol::Network::Response::mimeTypeKey, previousResourceData->mimeType());
557
558 resourceResponse->setInteger(Protocol::Network::Response::statusKey, previousResourceData->httpStatusCode());
559 resourceResponse->setString(Protocol::Network::Response::statusTextKey, previousResourceData->httpStatusText());
560
561 resourceResponse->setString(Protocol::Network::Response::sourceKey, Protocol::Helpers::getEnumConstantValue(Protocol::Network::Response::Source::DiskCache));
562 }
563 }
564
565 Protocol::Network::FrameId frameId = frameIdentifier(loader);
566 Protocol::Network::LoaderId loaderId = loaderIdentifier(loader);
567
568 m_resourcesData->responseReceived(requestId, frameId, response, type, shouldForceBufferingNetworkResourceData());
569
570 m_frontendDispatcher->responseReceived(requestId, frameId, loaderId, timestamp(), InspectorPageAgent::resourceTypeJSON(type), resourceResponse.releaseNonNull());
571
572 // If we revalidated the resource and got Not modified, send content length following didReceiveResponse
573 // as there will be no calls to didReceiveData from the network stack.
574 if (isNotModified && cachedResource && cachedResource->encodedSize())
575 didReceiveData(identifier, nullptr, cachedResource->encodedSize(), 0);
576}
577
578void InspectorNetworkAgent::didReceiveData(unsigned long identifier, const uint8_t* data, int dataLength, int encodedDataLength)
579{
580 if (m_hiddenRequestIdentifiers.contains(identifier))
581 return;
582
583 String requestId = IdentifiersFactory::requestId(identifier);
584
585 if (data) {
586 NetworkResourcesData::ResourceData const* resourceData = m_resourcesData->maybeAddResourceData(requestId, data, dataLength);
587
588 // For a synchronous XHR, if we didn't add data then we can apply it here as base64 encoded content.
589 // Often the data is text and we would have a decoder, but for non-text we won't have a decoder.
590 // Sync XHRs may not have a cached resource, while non-sync XHRs usually transfer data over on completion.
591 if (m_loadingXHRSynchronously && resourceData && !resourceData->hasBufferedData() && !resourceData->cachedResource())
592 m_resourcesData->setResourceContent(requestId, base64EncodeToString(data, dataLength), true);
593 }
594
595 m_frontendDispatcher->dataReceived(requestId, timestamp(), dataLength, encodedDataLength);
596}

598void InspectorNetworkAgent::didFinishLoading(unsigned long identifier, DocumentLoader* loader, const NetworkLoadMetrics& networkLoadMetrics, ResourceLoader*)
599{
600 if (m_hiddenRequestIdentifiers.remove(identifier))
601 return;
602
603 double elapsedFinishTime;
604 if (networkLoadMetrics.responseEnd)
605 elapsedFinishTime = m_environment.executionStopwatch().elapsedTimeSince(networkLoadMetrics.responseEnd).seconds();
606 else
607 elapsedFinishTime = timestamp();
608
609 String requestId = IdentifiersFactory::requestId(identifier);
610 if (loader && m_resourcesData->resourceType(requestId) == InspectorPageAgent::DocumentResource)
611 m_resourcesData->addResourceSharedBuffer(requestId, loader->frameLoader()->documentLoader()->mainResourceData(), loader->frame()->document()->encoding());
612
613 m_resourcesData->maybeDecodeDataToContent(requestId);
614
615 String sourceMappingURL;
616 NetworkResourcesData::ResourceData const* resourceData = m_resourcesData->data(requestId);
617 if (resourceData && resourceData->cachedResource())
618 sourceMappingURL = InspectorPageAgent::sourceMapURLForResource(resourceData->cachedResource());
619
620 std::optional<NetworkLoadMetrics> realMetrics;
621 if (platformStrategies()->loaderStrategy()->shouldPerformSecurityChecks() && !networkLoadMetrics.isComplete()) {
622 callOnMainThreadAndWait([&] {
623 realMetrics = platformStrategies()->loaderStrategy()->networkMetricsFromResourceLoadIdentifier(identifier).isolatedCopy();
624 });
625 }
626 auto metrics = buildObjectForMetrics(realMetrics ? *realMetrics : networkLoadMetrics);
627
628 m_frontendDispatcher->loadingFinished(requestId, elapsedFinishTime, sourceMappingURL, WTFMove(metrics));
629}

631void InspectorNetworkAgent::didFailLoading(unsigned long identifier, DocumentLoader* loader, const ResourceError& error)
632{
633 if (m_hiddenRequestIdentifiers.remove(identifier))
634 return;
635
636 String requestId = IdentifiersFactory::requestId(identifier);
637
638 if (loader && m_resourcesData->resourceType(requestId) == InspectorPageAgent::DocumentResource) {
639 Frame* frame = loader->frame();
640 if (frame && frame->loader().documentLoader() && frame->document()) {
641 m_resourcesData->addResourceSharedBuffer(requestId,
642 frame->loader().documentLoader()->mainResourceData(),
643 frame->document()->encoding());
644 }
645 }
646
647 m_frontendDispatcher->loadingFailed(requestId, timestamp(), error.localizedDescription(), error.isCancellation());
648}

650void InspectorNetworkAgent::didLoadResourceFromMemoryCache(DocumentLoader* loader, CachedResource& resource)

733Ref<Protocol::Network::Initiator> InspectorNetworkAgent::buildInitiatorObject(Document* document, const ResourceRequest* resourceRequest)
734{
735 // FIXME: Worker support.
736 if (!isMainThread()) {
737 return Protocol::Network::Initiator::create()
738 .setType(Protocol::Network::Initiator::Type::Other)
739 .release();
740 }
741
742 RefPtr<Protocol::Network::Initiator> initiatorObject;
743
744 Ref<ScriptCallStack> stackTrace = createScriptCallStack(JSExecState::currentState());
745 if (stackTrace->size() > 0) {
746 initiatorObject = Protocol::Network::Initiator::create()
747 .setType(Protocol::Network::Initiator::Type::Script)
748 .release();
749 initiatorObject->setStackTrace(stackTrace->buildInspectorArray());
750 } else if (document && document->scriptableDocumentParser()) {
751 initiatorObject = Protocol::Network::Initiator::create()
752 .setType(Protocol::Network::Initiator::Type::Parser)
753 .release();
754 initiatorObject->setUrl(document->url().string());
755 initiatorObject->setLineNumber(document->scriptableDocumentParser()->textPosition().m_line.oneBasedInt());
756 }
757
758 auto domAgent = m_instrumentingAgents.persistentDOMAgent();
759 if (domAgent && resourceRequest) {
760 if (auto inspectorInitiatorNodeIdentifier = resourceRequest->inspectorInitiatorNodeIdentifier()) {
761 if (!initiatorObject) {
762 initiatorObject = Protocol::Network::Initiator::create()
763 .setType(Protocol::Network::Initiator::Type::Other)
764 .release();
765 }
766
767 initiatorObject->setNodeId(*inspectorInitiatorNodeIdentifier);
768 }
769 }
770
771 if (initiatorObject)
772 return initiatorObject.releaseNonNull();
773
774 if (m_isRecalculatingStyle && m_styleRecalculationInitiator)
775 return *m_styleRecalculationInitiator;
776
777 return Protocol::Network::Initiator::create()
778 .setType(Protocol::Network::Initiator::Type::Other)
779 .release();
780}

824Protocol::ErrorStringOr<void> InspectorNetworkAgent::enable()
825{
826 m_enabled = true;
827 m_instrumentingAgents.setEnabledNetworkAgent(this);
828
829 {
830 Locker locker { WebSocket::allActiveWebSocketsLock() };
831
832 for (auto* webSocket : activeWebSockets()) {
833 if (!is<Document>(webSocket->scriptExecutionContext()))
834 continue;
835
836 auto& document = downcast<Document>(*webSocket->scriptExecutionContext());
837 auto channel = webSocket->channel();
838
839 auto identifier = channel->progressIdentifier();
840 didCreateWebSocket(identifier, webSocket->url());
841
842 auto cookieRequestHeaderFieldValue = [document = makeWeakPtr(document)](const URL& url) -> String {
843 if (!document || !document->page())
844 return { };
845 return document->page()->cookieJar().cookieRequestHeaderFieldValue(*document, url);
846 };
847 willSendWebSocketHandshakeRequest(identifier, channel->clientHandshakeRequest(WTFMove(cookieRequestHeaderFieldValue)));
848
849 if (channel->isConnected())
850 didReceiveWebSocketHandshakeResponse(identifier, channel->serverHandshakeResponse());
851
852 if (webSocket->readyState() == WebSocket::CLOSED)
853 didCloseWebSocket(identifier);
854 }
855 }
856
857 return { };
858}

877bool InspectorNetworkAgent::shouldIntercept(URL url, Protocol::Network::NetworkStage networkStage)
878{
879 url.removeFragmentIdentifier();
880
881 String urlString = url.string();
882 if (urlString.isEmpty())
883 return false;
884
885 for (auto& intercept : m_intercepts) {
886 if (intercept.networkStage != networkStage)
887 continue;
888 if (intercept.url.isEmpty())
889 return true;
890
891 auto searchStringType = intercept.isRegex ? ContentSearchUtilities::SearchStringType::Regex : ContentSearchUtilities::SearchStringType::ExactString;
892 auto regex = ContentSearchUtilities::createRegularExpressionForSearchString(intercept.url, intercept.caseSensitive, searchStringType);
893 if (regex.match(urlString) != -1)
894 return true;
895 }
896
897 return false;
898}

925Protocol::ErrorStringOr<std::tuple<String, bool /* base64Encoded */>> InspectorNetworkAgent::getResponseBody(const Protocol::Network::RequestId& requestId)
926{
927 NetworkResourcesData::ResourceData const* resourceData = m_resourcesData->data(requestId);
928 if (!resourceData)
929 return makeUnexpected("Missing resource for given requestId"_s);
930
931 if (resourceData->hasContent())
932 return { { resourceData->content(), resourceData->base64Encoded() } };
933
934 if (resourceData->isContentEvicted())
935 return makeUnexpected("Resource content was evicted from inspector cache"_s);
936
937 if (resourceData->buffer() && !resourceData->textEncodingName().isNull()) {
938 String body;
939 if (InspectorPageAgent::sharedBufferContent(resourceData->buffer(), resourceData->textEncodingName(), false, &body))
940 return { { body, false } };
941 }
942
943 if (resourceData->cachedResource()) {
944 String body;
945 bool base64Encoded;
946 if (InspectorNetworkAgent::cachedResourceContent(*resourceData->cachedResource(), &body, &base64Encoded))
947 return { { body, base64Encoded } };

949
950 return makeUnexpected("Missing content of resource for given requestId"_s);

960void InspectorNetworkAgent::loadResource(const Protocol::Network::FrameId& frameId, const String& urlString, Ref<LoadResourceCallback>&& callback)
961{
962 Protocol::ErrorString errorString;
963 auto* context = scriptExecutionContext(errorString, frameId);
964 if (!context) {
965 callback->sendFailure(errorString);
966 return;
967 }
968
969 URL url = context->completeURL(urlString);
970 ResourceRequest request(url);
971 request.setHTTPMethod("GET"_s);
972 request.setHiddenFromInspector(true);
973
974 ThreadableLoaderOptions options;
975 options.sendLoadCallbacks = SendCallbackPolicy::SendCallbacks; // So we remove this from m_hiddenRequestIdentifiers on completion.
976 options.defersLoadingPolicy = DefersLoadingPolicy::DisallowDefersLoading; // So the request is never deferred.
977 options.mode = FetchOptions::Mode::NoCors;
978 options.credentials = FetchOptions::Credentials::SameOrigin;
979 options.contentSecurityPolicyEnforcement = ContentSecurityPolicyEnforcement::DoNotEnforce;
980
981 // InspectorThreadableLoaderClient deletes itself when the load completes or fails.
982 InspectorThreadableLoaderClient* inspectorThreadableLoaderClient = new InspectorThreadableLoaderClient(callback.copyRef());
983 auto loader = ThreadableLoader::create(*context, *inspectorThreadableLoaderClient, WTFMove(request), options);
984 if (!loader) {
985 callback->sendFailure("Could not load requested resource."_s);
986 return;
987 }
988
989 // If the load already completed, inspectorThreadableLoaderClient will have been deleted and we will have already called the callback.
990 if (!callback->isActive())
991 return;
992
993 inspectorThreadableLoaderClient->setLoader(WTFMove(loader));
994}

1029Protocol::ErrorStringOr<Ref<Protocol::Runtime::RemoteObject>> InspectorNetworkAgent::resolveWebSocket(const Protocol::Network::RequestId& requestId, const String& objectGroup)
1030{
1031 WebSocket* webSocket = webSocketForRequestId(requestId);
1032 if (!webSocket)
1033 return makeUnexpected("Missing web socket for given requestId"_s);
1034
1035 // FIXME: <https://webkit.org/b/168475> Web Inspector: Correctly display iframe's and worker's WebSockets
1036 if (!is<Document>(webSocket->scriptExecutionContext()))
1037 return makeUnexpected("Not supported");
1038
1039 auto* document = downcast<Document>(webSocket->scriptExecutionContext());
1040 auto* frame = document->frame();
1041 if (!frame)
1042 return makeUnexpected("Missing frame of web socket for given requestId"_s);
1043
1044 auto& state = *mainWorldExecState(frame);
1045 auto injectedScript = m_injectedScriptManager.injectedScriptFor(&state);
1046 ASSERT(!injectedScript.hasNoValue());
1047
1048 auto object = injectedScript.wrapObject(webSocketAsScriptValue(state, webSocket), objectGroup);
1049 if (!object)
1050 return makeUnexpected("Internal error: unable to cast WebSocket");
1051
1052 return object.releaseNonNull();
1053}

1184Protocol::ErrorStringOr<void> InspectorNetworkAgent::interceptWithRequest(const Protocol::Network::RequestId& requestId, const String& url, const String& method, RefPtr<JSON::Object>&& headers, const String& postData)
1185{
1186 auto pendingRequest = m_pendingInterceptRequests.take(requestId);
1187 if (!pendingRequest)
1188 return makeUnexpected("Missing pending intercept request for given requestId"_s);
1189
1190 auto& loader = *pendingRequest->m_loader;
1191 ResourceRequest request = loader.request();
1192 if (!!url)
1193 request.setURL(URL({ }, url));
1194 if (!!method)
1195 request.setHTTPMethod(method);
1196 if (headers) {
1197 HTTPHeaderMap explicitHeaders;
1198 for (auto& header : *headers) {
1199 auto headerValue = header.value->asString();
1200 if (!!headerValue)
1201 explicitHeaders.add(header.key, headerValue);
1202 }
1203 request.setHTTPHeaderFields(WTFMove(explicitHeaders));
1204 }
1205 if (!!postData) {
1206 auto buffer = base64Decode(postData);
1207 if (!buffer)
1208 return makeUnexpected("Unable to decode given postData"_s);
1209
1210 request.setHTTPBody(FormData::create(WTFMove(*buffer)));
1211 }
1212 // FIXME: figure out how to identify when a request has been overridden when we add this to the frontend.
1213 pendingRequest->continueWithRequest(request);
1214
1215 return { };
1216}

1218Protocol::ErrorStringOr<void> InspectorNetworkAgent::interceptWithResponse(const Protocol::Network::RequestId& requestId, const String& content, bool base64Encoded, const String& mimeType, std::optional<int>&& status, const String& statusText, RefPtr<JSON::Object>&& headers)
1219{
1220 auto pendingInterceptResponse = m_pendingInterceptResponses.take(requestId);
1221 if (!pendingInterceptResponse)
1222 return makeUnexpected("Missing pending intercept response for given requestId"_s);
1223
1224 ResourceResponse overrideResponse(pendingInterceptResponse->originalResponse());
1225 overrideResponse.setSource(ResourceResponse::Source::InspectorOverride);
1226
1227 if (status)
1228 overrideResponse.setHTTPStatusCode(*status);
1229 if (!!statusText)
1230 overrideResponse.setHTTPStatusText(statusText);
1231 if (!!mimeType)
1232 overrideResponse.setMimeType(mimeType);
1233 if (headers) {
1234 HTTPHeaderMap explicitHeaders;
1235 for (auto& header : *headers) {
1236 auto headerValue = header.value->asString();
1237 if (!!headerValue)
1238 explicitHeaders.add(header.key, headerValue);
1239 }
1240 overrideResponse.setHTTPHeaderFields(WTFMove(explicitHeaders));
1241 overrideResponse.setHTTPHeaderField(HTTPHeaderName::ContentType, overrideResponse.mimeType());
1242 }
1243
1244 RefPtr<SharedBuffer> overrideData;
1245 if (base64Encoded) {
1246 auto buffer = base64Decode(content);
1247 if (!buffer)
1248 return makeUnexpected("Unable to decode given content"_s);
1249
1250 overrideData = SharedBuffer::create(WTFMove(*buffer));
1251 } else {
1252 auto utf8Content = content.utf8();
1253 overrideData = SharedBuffer::create(utf8Content.data(), utf8Content.length());
1254 }
1255
1256 pendingInterceptResponse->respond(overrideResponse, overrideData);
1257
1258 return { };
1259}

1261Protocol::ErrorStringOr<void> InspectorNetworkAgent::interceptRequestWithResponse(const Protocol::Network::RequestId& requestId, const String& content, bool base64Encoded, const String& mimeType, int status, const String& statusText, Ref<JSON::Object>&& headers)
1262{
1263 auto pendingRequest = m_pendingInterceptRequests.take(requestId);
1264 if (!pendingRequest)
1265 return makeUnexpected("Missing pending intercept request for given requestId"_s);
1266
1267 // Loader will be retained in the didReceiveResponse lambda below.
1268 RefPtr<ResourceLoader> loader = pendingRequest->m_loader.get();
1269 if (loader->reachedTerminalState())
1270 return makeUnexpected("Unable to fulfill request, it has already been processed"_s);
1271
1272 RefPtr<SharedBuffer> data;
1273 if (base64Encoded) {
1274 auto buffer = base64Decode(content);
1275 if (!buffer)
1276 return makeUnexpected("Unable to decode given content"_s);
1277
1278 data = SharedBuffer::create(WTFMove(*buffer));
1279 } else {
1280 auto utf8Content = content.utf8();
1281 data = SharedBuffer::create(utf8Content.data(), utf8Content.length());
1282 }
1283
1284 // Mimic data URL load behavior - report didReceiveResponse & didFinishLoading.
1285 ResourceResponse response(pendingRequest->m_loader->url(), mimeType, data->size(), String());
1286 response.setSource(ResourceResponse::Source::InspectorOverride);
1287 response.setHTTPStatusCode(status);
1288 response.setHTTPStatusText(statusText);
1289 HTTPHeaderMap explicitHeaders;
1290 for (auto& header : headers.get()) {
1291 auto headerValue = header.value->asString();
1292 if (!!headerValue)
1293 explicitHeaders.add(header.key, headerValue);
1294 }
1295 response.setHTTPHeaderFields(WTFMove(explicitHeaders));
1296 response.setHTTPHeaderField(HTTPHeaderName::ContentType, response.mimeType());
1297 loader->didReceiveResponse(response, [loader, buffer = data.releaseNonNull()]() mutable {
1298 if (buffer->size())
1299 loader->didReceiveBuffer(WTFMove(buffer), buffer->size(), DataPayloadWholeResource);
1300 loader->didFinishLoading(NetworkLoadMetrics());
1301 });
1302
1303 return { };
1304}

1306Protocol::ErrorStringOr<void> InspectorNetworkAgent::interceptRequestWithError(const Protocol::Network::RequestId& requestId, Protocol::Network::ResourceErrorType errorType)
1307{
1308 auto pendingRequest = m_pendingInterceptRequests.take(requestId);
1309 if (!pendingRequest)
1310 return makeUnexpected("Missing pending intercept request for given requestId"_s);
1311
1312 auto& loader = *pendingRequest->m_loader;
1313 if (loader.reachedTerminalState())
1314 return makeUnexpected("Unable to abort request, it has already been processed"_s);
1315
1316 switch (errorType) {
1317 case Protocol::Network::ResourceErrorType::General:
1318 loader.didFail(ResourceError(errorDomainWebKitInternal, 0, loader.url(), "Request intercepted"_s, ResourceError::Type::General));
1319 return { };
1320
1321 case Protocol::Network::ResourceErrorType::AccessControl:
1322 loader.didFail(ResourceError(errorDomainWebKitInternal, 0, loader.url(), "Access denied"_s, ResourceError::Type::AccessControl));
1323 return { };
1324
1325 case Protocol::Network::ResourceErrorType::Cancellation:
1326 loader.didFail(ResourceError(errorDomainWebKitInternal, 0, loader.url(), "Request canceled"_s, ResourceError::Type::Cancellation));
1327 return { };
1328
1329 case Protocol::Network::ResourceErrorType::Timeout:
1330 loader.didFail(ResourceError(errorDomainWebKitInternal, 0, loader.url(), "Request timed out"_s, ResourceError::Type::Timeout));

1334 ASSERT_NOT_REACHED();

1379bool InspectorNetworkAgent::cachedResourceContent(CachedResource& resource, String* result, bool* base64Encoded)
1380{
1381 ASSERT(result);
1382 ASSERT(base64Encoded);
1383
1384 if (!resource.encodedSize()) {
1385 *base64Encoded = false;
1386 *result = String();
1387 return true;
1388 }
1389
1390 switch (resource.type()) {
1391 case CachedResource::Type::CSSStyleSheet:
1392 *base64Encoded = false;
1393 *result = downcast<CachedCSSStyleSheet>(resource).sheetText();
1394 // The above can return a null String if the MIME type is invalid.
1395 return !result->isNull();
1396 case CachedResource::Type::Script:
1397 *base64Encoded = false;
1398 *result = downcast<CachedScript>(resource).script().toString();
1399 return true;
1400 default:
1401 auto* buffer = resource.resourceBuffer();
1402 if (!buffer)
1403 return false;
1404
1405 if (InspectorNetworkAgent::shouldTreatAsText(resource.mimeType())) {
1406 auto decoder = InspectorNetworkAgent::createTextDecoder(resource.mimeType(), resource.response().textEncodingName());
1407 *base64Encoded = false;
1408 *result = decoder->decodeAndFlush(buffer->data(), buffer->size());
1409 return true;
1410 }
1411
1412 *base64Encoded = true;
1413 *result = base64EncodeToString(buffer->data(), buffer->size());
1414 return true;
1415 }
1416}
