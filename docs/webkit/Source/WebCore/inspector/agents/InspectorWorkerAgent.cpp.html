37InspectorWorkerAgent::InspectorWorkerAgent(PageAgentContext& context)
38 : InspectorAgentBase("Worker"_s, context)
39 , m_frontendDispatcher(makeUnique<Inspector::WorkerFrontendDispatcher>(context.frontendRouter))
40 , m_backendDispatcher(Inspector::WorkerBackendDispatcher::create(context.backendDispatcher, this))
41 , m_page(context.inspectedPage)
42{
43}

45InspectorWorkerAgent::~InspectorWorkerAgent() = default;

47void InspectorWorkerAgent::didCreateFrontendAndBackend(FrontendRouter*, BackendDispatcher*)
48{
49 m_instrumentingAgents.setPersistentWorkerAgent(this);
50}

52void InspectorWorkerAgent::willDestroyFrontendAndBackend(DisconnectReason)
53{
54 m_instrumentingAgents.setPersistentWorkerAgent(nullptr);
55
56 disable();

59Protocol::ErrorStringOr<void> InspectorWorkerAgent::enable()
60{
61 if (m_enabled)
62 return { };
63
64 m_enabled = true;
65
66 connectToAllWorkerInspectorProxiesForPage();
67
68 return { };
69}
70
71Protocol::ErrorStringOr<void> InspectorWorkerAgent::disable()
72{
73 if (!m_enabled)
74 return { };
75
76 m_enabled = false;
77
78 disconnectFromAllWorkerInspectorProxies();
79
80 return { };
81}

83Protocol::ErrorStringOr<void> InspectorWorkerAgent::initialized(const String& workerId)
84{
85 WorkerInspectorProxy* proxy = m_connectedProxies.get(workerId);
86 if (!proxy)
87 return makeUnexpected("Missing worker for given workerId"_s);
88
89 proxy->resumeWorkerIfPaused();
90
91 return { };
92}
93
94Protocol::ErrorStringOr<void> InspectorWorkerAgent::sendMessageToWorker(const String& workerId, const String& message)
95{
96 if (!m_enabled)
97 return makeUnexpected("Worker domain must be enabled"_s);
98
99 WorkerInspectorProxy* proxy = m_connectedProxies.get(workerId);
100 if (!proxy)
101 return makeUnexpected("Missing worker for given workerId"_s);
102
103 proxy->sendMessageToWorkerInspectorController(message);
104
105 return { };
106}

133
134void InspectorWorkerAgent::connectToAllWorkerInspectorProxiesForPage()
135{
136 ASSERT(m_connectedProxies.isEmpty());
137
138 for (auto* proxy : WorkerInspectorProxy::allWorkerInspectorProxies()) {
139 if (!is<Document>(proxy->scriptExecutionContext()))
140 continue;
141
142 Document& document = downcast<Document>(*proxy->scriptExecutionContext());
143 if (document.page() != &m_page)
144 continue;
145
146 connectToWorkerInspectorProxy(*proxy);
147 }
148}
149
150void InspectorWorkerAgent::disconnectFromAllWorkerInspectorProxies()
151{
152 for (auto* proxy : copyToVector(m_connectedProxies.values()))
153 proxy->disconnectFromWorkerInspectorController();
154
155 m_connectedProxies.clear();
156}
157
158void InspectorWorkerAgent::connectToWorkerInspectorProxy(WorkerInspectorProxy& proxy)
159{
160 proxy.connectToWorkerInspectorController(*this);
161
162 m_connectedProxies.set(proxy.identifier(), &proxy);
163
164 m_frontendDispatcher->workerCreated(proxy.identifier(), proxy.url().string(), proxy.name());
165}
166
167void InspectorWorkerAgent::disconnectFromWorkerInspectorProxy(WorkerInspectorProxy& proxy)
168{
169 m_frontendDispatcher->workerTerminated(proxy.identifier());
170
171 m_connectedProxies.remove(proxy.identifier());
172
173 proxy.disconnectFromWorkerInspectorController();
174}
