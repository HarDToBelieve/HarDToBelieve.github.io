66void WebDebuggerAgent::didAddEventListener(EventTarget& target, const AtomString& eventType, EventListener& listener, bool capture)
67{
68 if (!breakpointsActive())
69 return;
70
71 auto& eventListeners = target.eventListeners(eventType);
72 auto position = eventListeners.findMatching([&](auto& registeredListener) {
73 return &registeredListener->callback() == &listener && registeredListener->useCapture() == capture;
74 });
75 if (position == notFound)
76 return;
77
78 auto& registeredListener = eventListeners.at(position);
79 if (m_registeredEventListeners.contains(registeredListener.get()))
80 return;
81
82 auto* globalObject = target.scriptExecutionContext()->globalObject();
83 if (!globalObject)
84 return;
85
86 int identifier = m_nextEventListenerIdentifier++;
87 m_registeredEventListeners.set(registeredListener.get(), identifier);
88
89 didScheduleAsyncCall(globalObject, InspectorDebuggerAgent::AsyncCallType::EventListener, identifier, registeredListener->isOnce());
90}
91
92void WebDebuggerAgent::willRemoveEventListener(EventTarget& target, const AtomString& eventType, EventListener& listener, bool capture)
93{
94 auto& eventListeners = target.eventListeners(eventType);
95 size_t listenerIndex = eventListeners.findMatching([&](auto& registeredListener) {
96 return &registeredListener->callback() == &listener && registeredListener->useCapture() == capture;
97 });
98
99 if (listenerIndex == notFound)
100 return;
101
102 int identifier = m_registeredEventListeners.take(eventListeners[listenerIndex].get());
103 didCancelAsyncCall(InspectorDebuggerAgent::AsyncCallType::EventListener, identifier);
104}
105
106void WebDebuggerAgent::willHandleEvent(const RegisteredEventListener& listener)
107{
108 auto it = m_registeredEventListeners.find(&listener);
109 if (it == m_registeredEventListeners.end())
110 return;
111
112 willDispatchAsyncCall(InspectorDebuggerAgent::AsyncCallType::EventListener, it->value);
113}
114
115int WebDebuggerAgent::willPostMessage()
116{
117 if (!breakpointsActive())
118 return 0;
119
120 auto postMessageIdentifier = m_nextPostMessageIdentifier++;
121 m_postMessageTasks.add(postMessageIdentifier);
122 return postMessageIdentifier;
123}
124
125void WebDebuggerAgent::didPostMessage(int postMessageIdentifier, JSC::JSGlobalObject& state)
126{
127 if (!breakpointsActive())
128 return;
129
130 if (!postMessageIdentifier || !m_postMessageTasks.contains(postMessageIdentifier))
131 return;
132
133 didScheduleAsyncCall(&state, InspectorDebuggerAgent::AsyncCallType::PostMessage, postMessageIdentifier, true);
134}
135
136void WebDebuggerAgent::didFailPostMessage(int postMessageIdentifier)
137{
138 if (!postMessageIdentifier)
139 return;
140
141 auto it = m_postMessageTasks.find(postMessageIdentifier);
142 if (it == m_postMessageTasks.end())
143 return;
144
145 didCancelAsyncCall(InspectorDebuggerAgent::AsyncCallType::PostMessage, postMessageIdentifier);
146
147 m_postMessageTasks.remove(it);
148}
149
150void WebDebuggerAgent::willDispatchPostMessage(int postMessageIdentifier)
151{
152 if (!postMessageIdentifier || !m_postMessageTasks.contains(postMessageIdentifier))
153 return;
154
155 willDispatchAsyncCall(InspectorDebuggerAgent::AsyncCallType::PostMessage, postMessageIdentifier);
156}
157
158void WebDebuggerAgent::didDispatchPostMessage(int postMessageIdentifier)
159{
160 if (!postMessageIdentifier)
161 return;
162
163 auto it = m_postMessageTasks.find(postMessageIdentifier);
164 if (it == m_postMessageTasks.end())
165 return;
166
167 didDispatchAsyncCall();
168
169 m_postMessageTasks.remove(it);
170}
171
172void WebDebuggerAgent::didClearAsyncStackTraceData()
173{
174 m_registeredEventListeners.clear();
175 m_postMessageTasks.clear();
176 m_nextEventListenerIdentifier = 1;
177 m_nextPostMessageIdentifier = 1;
178}
