56Protocol::ErrorStringOr<Ref<JSON::ArrayOf<Protocol::Console::Channel>>> PageConsoleAgent::getLoggingChannels()
57{
58 auto channels = JSON::ArrayOf<Protocol::Console::Channel>::create();
59
60 auto addLogChannel = [&] (Protocol::Console::ChannelSource source) {
61 auto* logChannel = getLogChannel(Protocol::Helpers::getEnumConstantValue(source));
62 if (!logChannel)
63 return;
64
65 auto level = Protocol::Console::ChannelLevel::Off;
66 if (logChannel->state != WTFLogChannelState::Off) {
67 switch (logChannel->level) {
68 case WTFLogLevel::Always:
69 case WTFLogLevel::Error:
70 case WTFLogLevel::Warning:
71 case WTFLogLevel::Info:
72 level = Protocol::Console::ChannelLevel::Basic;
73 break;
74
75 case WTFLogLevel::Debug:
76 level = Protocol::Console::ChannelLevel::Verbose;
77 break;
78 }
79 }
80
81 auto channel = Protocol::Console::Channel::create()
82 .setSource(source)
83 .setLevel(level)
84 .release();
85 channels->addItem(WTFMove(channel));
86 };
87 addLogChannel(Protocol::Console::ChannelSource::XML);
88 addLogChannel(Protocol::Console::ChannelSource::JavaScript);
89 addLogChannel(Protocol::Console::ChannelSource::Network);
90 addLogChannel(Protocol::Console::ChannelSource::ConsoleAPI);
91 addLogChannel(Protocol::Console::ChannelSource::Storage);
92 addLogChannel(Protocol::Console::ChannelSource::Appcache);
93 addLogChannel(Protocol::Console::ChannelSource::Rendering);
94 addLogChannel(Protocol::Console::ChannelSource::CSS);
95 addLogChannel(Protocol::Console::ChannelSource::Security);
96 addLogChannel(Protocol::Console::ChannelSource::ContentBlocker);
97 addLogChannel(Protocol::Console::ChannelSource::Media);
98 addLogChannel(Protocol::Console::ChannelSource::MediaSource);
99 addLogChannel(Protocol::Console::ChannelSource::WebRTC);
100 addLogChannel(Protocol::Console::ChannelSource::ITPDebug);
101 addLogChannel(Protocol::Console::ChannelSource::PrivateClickMeasurement);
102 addLogChannel(Protocol::Console::ChannelSource::PaymentRequest);
103 addLogChannel(Protocol::Console::ChannelSource::Other);
104
105 return channels;
106}

108Protocol::ErrorStringOr<void> PageConsoleAgent::setLoggingChannelLevel(Protocol::Console::ChannelSource source, Protocol::Console::ChannelLevel level)
109{
110 switch (level) {
111 case Protocol::Console::ChannelLevel::Off:
112 m_inspectedPage.configureLoggingChannel(Protocol::Helpers::getEnumConstantValue(source), WTFLogChannelState::Off, WTFLogLevel::Error);
113 return { };
114
115 case Protocol::Console::ChannelLevel::Basic:
116 m_inspectedPage.configureLoggingChannel(Protocol::Helpers::getEnumConstantValue(source), WTFLogChannelState::On, WTFLogLevel::Info);
117 return { };
118
119 case Protocol::Console::ChannelLevel::Verbose:
120 m_inspectedPage.configureLoggingChannel(Protocol::Helpers::getEnumConstantValue(source), WTFLogChannelState::On, WTFLogLevel::Debug);
121 return { };
122 }
123
124 ASSERT_NOT_REACHED();
125 return { };
126}
