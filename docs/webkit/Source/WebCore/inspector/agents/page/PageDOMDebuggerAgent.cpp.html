71Protocol::ErrorStringOr<void> PageDOMDebuggerAgent::setDOMBreakpoint(Protocol::DOM::NodeId nodeId, Protocol::DOMDebugger::DOMBreakpointType type, RefPtr<JSON::Object>&& options)
72{
73 Protocol::ErrorString errorString;
74
75 auto* domAgent = m_instrumentingAgents.persistentDOMAgent();
76 if (!domAgent)
77 return makeUnexpected("DOM domain must be enabled"_s);
78
79 Node* node = domAgent->assertNode(errorString, nodeId);
80 if (!node)
81 return makeUnexpected(errorString);
82
83 auto breakpoint = InspectorDebuggerAgent::debuggerBreakpointFromPayload(errorString, WTFMove(options));
84 if (!breakpoint)
85 return makeUnexpected(errorString);
86
87 switch (type) {
88 case Protocol::DOMDebugger::DOMBreakpointType::SubtreeModified:
89 if (!m_domSubtreeModifiedBreakpoints.add(node, breakpoint.releaseNonNull()))
90 return makeUnexpected("Breakpoint for given node and given type already exists"_s);
91 return { };
92
93 case Protocol::DOMDebugger::DOMBreakpointType::AttributeModified:
94 if (!m_domAttributeModifiedBreakpoints.add(node, breakpoint.releaseNonNull()))
95 return makeUnexpected("Breakpoint for given node and given type already exists"_s);
96 return { };
97
98 case Protocol::DOMDebugger::DOMBreakpointType::NodeRemoved:
99 if (!m_domNodeRemovedBreakpoints.add(node, breakpoint.releaseNonNull()))
100 return makeUnexpected("Breakpoint for given node and given type already exists"_s);
101 return { };
102 }
103
104 ASSERT_NOT_REACHED();
105 return makeUnexpected("Not supported");
106}

108Protocol::ErrorStringOr<void> PageDOMDebuggerAgent::removeDOMBreakpoint(Protocol::DOM::NodeId nodeId, Protocol::DOMDebugger::DOMBreakpointType type)
109{
110 Protocol::ErrorString errorString;
111
112 auto* domAgent = m_instrumentingAgents.persistentDOMAgent();
113 if (!domAgent)
114 return makeUnexpected("DOM domain must be enabled"_s);
115
116 Node* node = domAgent->assertNode(errorString, nodeId);
117 if (!node)
118 return makeUnexpected(errorString);
119
120 switch (type) {
121 case Protocol::DOMDebugger::DOMBreakpointType::SubtreeModified:
122 if (!m_domSubtreeModifiedBreakpoints.remove(node))
123 return makeUnexpected("Breakpoint for given node and given type missing"_s);
124 return { };
125
126 case Protocol::DOMDebugger::DOMBreakpointType::AttributeModified:
127 if (!m_domAttributeModifiedBreakpoints.remove(node))
128 return makeUnexpected("Breakpoint for given node and given type missing"_s);
129 return { };
130
131 case Protocol::DOMDebugger::DOMBreakpointType::NodeRemoved:
132 if (!m_domNodeRemovedBreakpoints.remove(node))
133 return makeUnexpected("Breakpoint for given node and given type missing"_s);
134 return { };
135 }
136
137 ASSERT_NOT_REACHED();
138 return makeUnexpected("Not supported");
139}

160static std::optional<size_t> calculateDistance(Node& child, Node& ancestor)
161{
162 size_t distance = 0;
163
164 auto* current = &child;
165 while (current != &ancestor) {
166 ++distance;
167
168 current = InspectorDOMAgent::innerParentNode(current);
169 if (!current)
170 return std::nullopt;
171 }
172
173 return distance;
174}

176void PageDOMDebuggerAgent::willInsertDOMNode(Node& parent)
177{
178 if (!m_debuggerAgent->breakpointsActive())
179 return;
180
181 if (m_domSubtreeModifiedBreakpoints.isEmpty())
182 return;
183
184 std::optional<size_t> closestDistance;
185 RefPtr<JSC::Breakpoint> closestBreakpoint;
186 Node* closestBreakpointOwner = nullptr;
187
188 for (auto [breakpointOwner, breakpoint] : m_domSubtreeModifiedBreakpoints) {
189 auto distance = calculateDistance(parent, *breakpointOwner);
190 if (!distance)
191 continue;
192
193 if (!closestDistance || distance < closestDistance) {
194 closestDistance = distance;
195 closestBreakpoint = breakpoint.copyRef();
196 closestBreakpointOwner = breakpointOwner;
197 }
198 }
199
200 if (!closestBreakpoint)
201 return;
202
203 ASSERT(closestBreakpointOwner);
204
205 auto pauseData = buildPauseDataForDOMBreakpoint(Protocol::DOMDebugger::DOMBreakpointType::SubtreeModified, *closestBreakpointOwner);
206 pauseData->setBoolean("insertion", true);
207 // FIXME: <https://webkit.org/b/213499> Web Inspector: allow DOM nodes to be instrumented at any point, regardless of whether the main document has also been instrumented
208 // Include the new child node ID so the frontend can show the node that's about to be inserted.
209 m_debuggerAgent->breakProgram(Inspector::DebuggerFrontendDispatcher::Reason::DOM, WTFMove(pauseData), WTFMove(closestBreakpoint));
210}

212void PageDOMDebuggerAgent::willRemoveDOMNode(Node& node)
213{
214 if (!m_debuggerAgent->breakpointsActive())
215 return;
216
217 if (m_domNodeRemovedBreakpoints.isEmpty() && m_domSubtreeModifiedBreakpoints.isEmpty())
218 return;
219
220 std::optional<size_t> closestDistance;
221 RefPtr<JSC::Breakpoint> closestBreakpoint;
222 std::optional<Protocol::DOMDebugger::DOMBreakpointType> closestBreakpointType;
223 Node* closestBreakpointOwner = nullptr;
224
225 for (auto [breakpointOwner, breakpoint] : m_domNodeRemovedBreakpoints) {
226 auto distance = calculateDistance(*breakpointOwner, node);
227 if (!distance)
228 continue;
229
230 if (!closestDistance || distance < closestDistance) {
231 closestDistance = distance;
232 closestBreakpoint = breakpoint.copyRef();
233 closestBreakpointType = Protocol::DOMDebugger::DOMBreakpointType::NodeRemoved;
234 closestBreakpointOwner = breakpointOwner;
235 }
236 }
237
238 if (!closestBreakpoint) {
239 for (auto [breakpointOwner, breakpoint] : m_domSubtreeModifiedBreakpoints) {
240 auto distance = calculateDistance(node, *breakpointOwner);
241 if (!distance)
242 continue;
243
244 if (!closestDistance || distance < closestDistance) {
245 closestDistance = distance;
246 closestBreakpoint = breakpoint.copyRef();
247 closestBreakpointType = Protocol::DOMDebugger::DOMBreakpointType::SubtreeModified;
248 closestBreakpointOwner = breakpointOwner;
249 }
250 }
251 }
252
253 if (!closestBreakpoint)
254 return;
255
256 ASSERT(closestBreakpointType);
257 ASSERT(closestBreakpointOwner);
258
259 auto pauseData = buildPauseDataForDOMBreakpoint(*closestBreakpointType, *closestBreakpointOwner);
260 if (auto* domAgent = m_instrumentingAgents.persistentDOMAgent()) {
261 if (&node != closestBreakpointOwner) {
262 if (auto targetNodeId = domAgent->pushNodeToFrontend(&node))
263 pauseData->setInteger("targetNodeId", targetNodeId);
264 }
265 }
266 m_debuggerAgent->breakProgram(Inspector::DebuggerFrontendDispatcher::Reason::DOM, WTFMove(pauseData), WTFMove(closestBreakpoint));
267}

269void PageDOMDebuggerAgent::didRemoveDOMNode(Node& node)
270{
271 auto nodeContainsBreakpointOwner = [&] (auto& entry) {
272 return node.contains(entry.key);
273 };
274 m_domSubtreeModifiedBreakpoints.removeIf(nodeContainsBreakpointOwner);
275 m_domAttributeModifiedBreakpoints.removeIf(nodeContainsBreakpointOwner);
276 m_domNodeRemovedBreakpoints.removeIf(nodeContainsBreakpointOwner);
277}

279void PageDOMDebuggerAgent::willDestroyDOMNode(Node& node)
280{
281 // This can be called in response to GC.
282 // DOM Node destruction should be treated as if the node was removed from the DOM tree.
283 didRemoveDOMNode(node);
284}

286void PageDOMDebuggerAgent::willModifyDOMAttr(Element& element)
287{
288 if (!m_debuggerAgent->breakpointsActive())
289 return;
290
291 auto it = m_domAttributeModifiedBreakpoints.find(&element);
292 if (it == m_domAttributeModifiedBreakpoints.end())
293 return;
294
295 auto pauseData = buildPauseDataForDOMBreakpoint(Protocol::DOMDebugger::DOMBreakpointType::AttributeModified, element);
296 m_debuggerAgent->breakProgram(Inspector::DebuggerFrontendDispatcher::Reason::DOM, WTFMove(pauseData), it->value.copyRef());
297}

299void PageDOMDebuggerAgent::willFireAnimationFrame()
300{
301 if (!m_debuggerAgent->breakpointsActive())
302 return;
303
304 auto breakpoint = m_pauseOnAllAnimationFramesBreakpoint;
305 if (!breakpoint)
306 return;
307
308 m_debuggerAgent->schedulePauseForSpecialBreakpoint(*breakpoint, Inspector::DebuggerFrontendDispatcher::Reason::AnimationFrame);
309}

311void PageDOMDebuggerAgent::didFireAnimationFrame()
312{
313 if (!m_debuggerAgent->breakpointsActive())
314 return;
315
316 auto breakpoint = m_pauseOnAllAnimationFramesBreakpoint;
317 if (!breakpoint)
318 return;
319
320 m_debuggerAgent->cancelPauseForSpecialBreakpoint(*breakpoint);
321}

336bool PageDOMDebuggerAgent::setAnimationFrameBreakpoint(Protocol::ErrorString& errorString, RefPtr<JSC::Breakpoint>&& breakpoint)
337{
338 if (!m_pauseOnAllAnimationFramesBreakpoint == !breakpoint) {
339 errorString = m_pauseOnAllAnimationFramesBreakpoint ? "Breakpoint for AnimationFrame already exists"_s : "Breakpoint for AnimationFrame missing"_s;
340 return false;
341 }
342
343 m_pauseOnAllAnimationFramesBreakpoint = WTFMove(breakpoint);
344 return true;
345}

347Ref<JSON::Object> PageDOMDebuggerAgent::buildPauseDataForDOMBreakpoint(Protocol::DOMDebugger::DOMBreakpointType breakpointType, Node& breakpointOwner)
348{
349 ASSERT(m_debuggerAgent->breakpointsActive());
350 ASSERT(m_domSubtreeModifiedBreakpoints.contains(&breakpointOwner) || m_domAttributeModifiedBreakpoints.contains(&breakpointOwner) || m_domNodeRemovedBreakpoints.contains(&breakpointOwner));
351
352 auto pauseData = JSON::Object::create();
353 pauseData->setString("type", Protocol::Helpers::getEnumConstantValue(breakpointType));
354 if (auto* domAgent = m_instrumentingAgents.persistentDOMAgent()) {
355 if (auto breakpointOwnerNodeId = domAgent->pushNodeToFrontend(&breakpointOwner))
356 pauseData->setInteger("nodeId", breakpointOwnerNodeId);
357 }
358 return pauseData;
359}
