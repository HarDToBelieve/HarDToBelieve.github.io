91String PageDebuggerAgent::sourceMapURLForScript(const JSC::Debugger::Script& script)
92{
93 static NeverDestroyed<String> sourceMapHTTPHeader(MAKE_STATIC_STRING_IMPL("SourceMap"));
94 static NeverDestroyed<String> sourceMapHTTPHeaderDeprecated(MAKE_STATIC_STRING_IMPL("X-SourceMap"));
95
96 if (!script.url.isEmpty()) {
97 CachedResource* resource = InspectorPageAgent::cachedResource(&m_inspectedPage.mainFrame(), URL({ }, script.url));
98 if (resource) {
99 String sourceMapHeader = resource->response().httpHeaderField(sourceMapHTTPHeader);
100 if (!sourceMapHeader.isEmpty())
101 return sourceMapHeader;
102
103 sourceMapHeader = resource->response().httpHeaderField(sourceMapHTTPHeaderDeprecated);
104 if (!sourceMapHeader.isEmpty())
105 return sourceMapHeader;
106 }
107 }
108
109 return InspectorDebuggerAgent::sourceMapURLForScript(script);
110}

123{

128void PageDebuggerAgent::debuggerDidEvaluate(JSC::Debugger&, const JSC::Breakpoint::Action& action)
129{
130 if (action.emulateUserGesture)
131 m_breakpointActionUserGestureEmulationScopeStack.removeLast();

139InjectedScript PageDebuggerAgent::injectedScriptForEval(Protocol::ErrorString& errorString, std::optional<Protocol::Runtime::ExecutionContextId>&& executionContextId)
140{
141 if (!executionContextId) {
142 JSC::JSGlobalObject* scriptState = mainWorldExecState(&m_inspectedPage.mainFrame());
143 return injectedScriptManager().injectedScriptFor(scriptState);
144 }
145
146 InjectedScript injectedScript = injectedScriptManager().injectedScriptForId(*executionContextId);
147 if (injectedScript.hasNoValue())
148 errorString = "Missing injected script for given executionContextId."_s;
149
150 return injectedScript;
151}

180void PageDebuggerAgent::didRequestAnimationFrame(int callbackId, Document& document)
181{
182 if (!breakpointsActive())
183 return;
184
185 auto* globalObject = document.globalObject();
186 if (!globalObject)
187 return;
188
189 didScheduleAsyncCall(globalObject, InspectorDebuggerAgent::AsyncCallType::RequestAnimationFrame, callbackId, true);
190}
