34PageHeapAgent::PageHeapAgent(PageAgentContext& context)
35 : WebHeapAgent(context)
36 , m_instrumentingAgents(context.instrumentingAgents)
37{
38}
39
40PageHeapAgent::~PageHeapAgent() = default;
41
42Protocol::ErrorStringOr<void> PageHeapAgent::enable()
43{
44 auto result = WebHeapAgent::enable();
45
46 m_instrumentingAgents.setEnabledPageHeapAgent(this);
47
48 return result;
49}
50
51Protocol::ErrorStringOr<void> PageHeapAgent::disable()
52{
53 m_instrumentingAgents.setEnabledPageHeapAgent(nullptr);
54
55 return WebHeapAgent::disable();
56}

58void PageHeapAgent::mainFrameNavigated()
59{
60 clearHeapSnapshots();
61}
