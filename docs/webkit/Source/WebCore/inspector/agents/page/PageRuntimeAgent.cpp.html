66Protocol::ErrorStringOr<void> PageRuntimeAgent::enable()
67{
68 if (m_instrumentingAgents.enabledPageRuntimeAgent() == this)
69 return { };
70
71 auto result = InspectorRuntimeAgent::enable();
72 if (!result)
73 return result;
74
75 // Report initial contexts before enabling instrumentation as the reporting
76 // can force creation of script state which could result in duplicate notifications.
77 reportExecutionContextCreation();
78
79 m_instrumentingAgents.setEnabledPageRuntimeAgent(this);
80
81 return result;
82}
83
84Protocol::ErrorStringOr<void> PageRuntimeAgent::disable()
85{
86 m_instrumentingAgents.setEnabledPageRuntimeAgent(nullptr);
87
88 return InspectorRuntimeAgent::disable();
89}

97void PageRuntimeAgent::didClearWindowObjectInWorld(Frame& frame, DOMWrapperWorld& world)
98{
99 auto* pageAgent = m_instrumentingAgents.enabledPageAgent();
100 if (!pageAgent)
101 return;
102
103 notifyContextCreated(pageAgent->frameId(&frame), frame.script().globalObject(world), world);
104}

106InjectedScript PageRuntimeAgent::injectedScriptForEval(Protocol::ErrorString& errorString, std::optional<Protocol::Runtime::ExecutionContextId>&& executionContextId)
107{
108 if (!executionContextId) {
109 JSC::JSGlobalObject* scriptState = mainWorldExecState(&m_inspectedPage.mainFrame());
110 InjectedScript result = injectedScriptManager().injectedScriptFor(scriptState);
111 if (result.hasNoValue())
112 errorString = "Internal error: main world execution context not found"_s;
113 return result;
114 }
115
116 InjectedScript injectedScript = injectedScriptManager().injectedScriptForId(*executionContextId);
117 if (injectedScript.hasNoValue())
118 errorString = "Missing injected script for given executionContextId"_s;
119 return injectedScript;
120}

132void PageRuntimeAgent::reportExecutionContextCreation()
133{
134 auto* pageAgent = m_instrumentingAgents.enabledPageAgent();
135 if (!pageAgent)
136 return;
137
138 for (auto* frame = &m_inspectedPage.mainFrame(); frame; frame = frame->tree().traverseNext()) {
139 if (!frame->script().canExecuteScripts(NotAboutToExecuteScript))
140 continue;
141
142 auto frameId = pageAgent->frameId(frame);
143
144 // Always send the main world first.
145 auto* mainGlobalObject = mainWorldExecState(frame);
146 notifyContextCreated(frameId, mainGlobalObject, mainThreadNormalWorld());
147
148 for (auto& jsWindowProxy : frame->windowProxy().jsWindowProxiesAsVector()) {
149 auto* globalObject = jsWindowProxy->window();
150 if (globalObject == mainGlobalObject)
151 continue;
152
153 auto& securityOrigin = downcast<DOMWindow>(jsWindowProxy->wrapped()).document()->securityOrigin();
154 notifyContextCreated(frameId, globalObject, jsWindowProxy->world(), &securityOrigin);
155 }
156 }
157}

159static Protocol::Runtime::ExecutionContextType toProtocol(DOMWrapperWorld::Type type)
160{
161 switch (type) {
162 case DOMWrapperWorld::Type::Normal:
163 return Protocol::Runtime::ExecutionContextType::Normal;
164 case DOMWrapperWorld::Type::User:
165 return Protocol::Runtime::ExecutionContextType::User;
166 case DOMWrapperWorld::Type::Internal:
167 return Protocol::Runtime::ExecutionContextType::Internal;
168 }
169
170 ASSERT_NOT_REACHED();
171 return Protocol::Runtime::ExecutionContextType::Internal;
172}

174void PageRuntimeAgent::notifyContextCreated(const Protocol::Network::FrameId& frameId, JSC::JSGlobalObject* globalObject, const DOMWrapperWorld& world, SecurityOrigin* securityOrigin)
175{
176 auto injectedScript = injectedScriptManager().injectedScriptFor(globalObject);
177 if (injectedScript.hasNoValue())
178 return;
179
180 auto name = world.name();
181 if (name.isEmpty() && securityOrigin)
182 name = securityOrigin->toRawString();
183
184 m_frontendDispatcher->executionContextCreated(Protocol::Runtime::ExecutionContextDescription::create()
185 .setId(injectedScriptManager().injectedScriptIdFor(globalObject))
186 .setType(toProtocol(world.type()))
187 .setName(name)
188 .setFrameId(frameId)
189 .release());
190}
