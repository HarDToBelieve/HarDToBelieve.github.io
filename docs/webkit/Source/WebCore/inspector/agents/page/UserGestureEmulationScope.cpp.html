39namespace WebCore {
40
41UserGestureEmulationScope::UserGestureEmulationScope(Page& inspectedPage, bool emulateUserGesture)
42 : m_pageChromeClient(inspectedPage.chrome().client())
43 , m_gestureIndicator(emulateUserGesture ? std::optional<ProcessingUserGestureState>(ProcessingUserGesture) : std::nullopt)
44 , m_emulateUserGesture(emulateUserGesture)
45 , m_userWasInteracting(false)
46{
47 if (m_emulateUserGesture) {
48 m_userWasInteracting = m_pageChromeClient.userIsInteracting();
49 if (!m_userWasInteracting)
50 m_pageChromeClient.setUserIsInteracting(true);
51 }
52}
53
54UserGestureEmulationScope::~UserGestureEmulationScope()
55{
56 if (m_emulateUserGesture && !m_userWasInteracting && m_pageChromeClient.userIsInteracting())
57 m_pageChromeClient.setUserIsInteracting(false);
58}
