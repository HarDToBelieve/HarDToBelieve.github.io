39ServiceWorkerAgent::ServiceWorkerAgent(WorkerAgentContext& context)
40 : InspectorAgentBase("ServiceWorker"_s, context)
41 , m_serviceWorkerGlobalScope(downcast<ServiceWorkerGlobalScope>(context.globalScope))
42 , m_backendDispatcher(Inspector::ServiceWorkerBackendDispatcher::create(context.backendDispatcher, this))
43{
44 ASSERT(context.globalScope.isContextThread());
45}
46
47ServiceWorkerAgent::~ServiceWorkerAgent() = default;

51}

55}

57Protocol::ErrorStringOr<Ref<Protocol::ServiceWorker::Configuration>> ServiceWorkerAgent::getInitializationInfo()
58{
59 return Protocol::ServiceWorker::Configuration::create()
60 .setTargetId(m_serviceWorkerGlobalScope.identifier())
61 .setSecurityOrigin(m_serviceWorkerGlobalScope.securityOrigin()->toRawString())
62 .setUrl(m_serviceWorkerGlobalScope.contextData().scriptURL.string())
63 .setContent(m_serviceWorkerGlobalScope.contextData().script.toString())
64 .release();
65}
