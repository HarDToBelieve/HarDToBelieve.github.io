51static bool areEssentiallyEqual(LayoutUnit a, LayoutUnit b)
52{
53 if (a == b)
54 return true;
55 // 1/4th CSS pixel.
56 constexpr float epsilon = kFixedPointDenominator / 4;
57 return abs(a.rawValue() - b.rawValue()) <= epsilon;
58}
59
60static bool areEssentiallyEqual(float a, InlineLayoutUnit b)
61{
62 return areEssentiallyEqual(LayoutUnit { a }, LayoutUnit { b });
63}
64
65static bool areEssentiallyEqual(LayoutRect a, LayoutRect b)
66{
67 return areEssentiallyEqual(a.x(), b.x())
68 && areEssentiallyEqual(a.y(), b.y())
69 && areEssentiallyEqual(a.width(), b.width())
70 && areEssentiallyEqual(a.height(), b.height());
71}

73static bool checkForMatchingNonTextRuns(const LineRun& lineRun, const WebCore::LegacyInlineBox& inlineBox)
74{
75 return areEssentiallyEqual(inlineBox.left(), lineRun.logicalLeft())
76 && areEssentiallyEqual(inlineBox.right(), lineRun.logicalRight())
77 && areEssentiallyEqual(inlineBox.top(), lineRun.logicalTop())
78 && areEssentiallyEqual(inlineBox.bottom(), lineRun.logicalBottom());
79}

81
82static bool checkForMatchingTextRuns(const LineRun& lineRun, const WebCore::LegacyInlineTextBox& inlineTextBox)
83{
84 if (!lineRun.text())
85 return false;
86 return areEssentiallyEqual(inlineTextBox.left(), lineRun.logicalLeft())
87 && areEssentiallyEqual(inlineTextBox.right(), lineRun.logicalRight())
88 && areEssentiallyEqual(inlineTextBox.top(), lineRun.logicalTop())
89 && areEssentiallyEqual(inlineTextBox.bottom(), lineRun.logicalBottom())
90 && (inlineTextBox.isLineBreak() || (inlineTextBox.start() == lineRun.text()->start() && inlineTextBox.end() == lineRun.text()->end()));
91}

93static void collectFlowBoxSubtree(const LegacyInlineFlowBox& flowbox, Vector<WebCore::LegacyInlineBox*>& inlineBoxes)
94{
95 auto* inlineBox = flowbox.firstLeafDescendant();
96 auto* lastLeafDescendant = flowbox.lastLeafDescendant();
97 while (inlineBox) {
98 inlineBoxes.append(inlineBox);
99 if (inlineBox == lastLeafDescendant)
100 break;
101 inlineBox = inlineBox->nextLeafOnLine();
102 }
103}
104
105static void collectInlineBoxes(const RenderBlockFlow& root, Vector<WebCore::LegacyInlineBox*>& inlineBoxes)
106{
107 for (auto* rootLine = root.firstRootBox(); rootLine; rootLine = rootLine->nextRootBox()) {
108 for (auto* inlineBox = rootLine->firstChild(); inlineBox; inlineBox = inlineBox->nextOnLine()) {
109 if (!is<LegacyInlineFlowBox>(inlineBox)) {
110 inlineBoxes.append(inlineBox);
111 continue;
112 }
113 collectFlowBoxSubtree(downcast<LegacyInlineFlowBox>(*inlineBox), inlineBoxes);
114 }
115 }
116}

118static bool outputMismatchingComplexLineInformationIfNeeded(TextStream& stream, const LayoutState& layoutState, const RenderBlockFlow& blockFlow, const ContainerBox& inlineFormattingRoot)
119{
120 auto& inlineFormattingState = layoutState.establishedFormattingState(inlineFormattingRoot);
121 auto& lineRuns = downcast<InlineFormattingState>(inlineFormattingState).lineRuns();
122 // Collect inlineboxes.
123 Vector<WebCore::LegacyInlineBox*> inlineBoxes;
124 collectInlineBoxes(blockFlow, inlineBoxes);
125
126 auto mismatched = false;
127 unsigned runIndex = 0;
128
129 if (inlineBoxes.size() != lineRuns.size()) {
130 stream << "Warning: mismatching number of runs: inlineboxes(" << inlineBoxes.size() << ") vs. inline runs(" << lineRuns.size() << ")";
131 stream.nextLine();
132 }
133
134 for (unsigned inlineBoxIndex = 0; inlineBoxIndex < inlineBoxes.size() && runIndex < lineRuns.size(); ++inlineBoxIndex) {
135 auto& lineRun = lineRuns[runIndex];
136 auto* inlineBox = inlineBoxes[inlineBoxIndex];
137 auto* inlineTextBox = is<WebCore::LegacyInlineTextBox>(inlineBox) ? downcast<WebCore::LegacyInlineTextBox>(inlineBox) : nullptr;
138 bool matchingRuns = inlineTextBox ? checkForMatchingTextRuns(lineRun, *inlineTextBox) : checkForMatchingNonTextRuns(lineRun, *inlineBox);
139
140 if (!matchingRuns) {
141
142 if (is<RenderLineBreak>(inlineBox->renderer())) {
143 // <br> positioning is weird at this point. It needs proper baseline.
144 matchingRuns = true;
145 ++runIndex;
146 continue;
147 }
148
149 stream << "Mismatching: run";
150
151 if (inlineTextBox)
152 stream << " (" << inlineTextBox->start() << ", " << inlineTextBox->end() << ")";
153 stream << " (" << inlineBox->logicalLeft() << ", " << inlineBox->logicalTop() << ") (" << inlineBox->logicalWidth() << "x" << inlineBox->logicalHeight() << ")";
154
155 stream << " inline run";
156 if (lineRun.text())
157 stream << " (" << lineRun.text()->start() << ", " << lineRun.text()->end() << ")";
158 stream << " (" << lineRun.logicalLeft() << ", " << lineRun.logicalTop() << ") (" << lineRun.logicalWidth() << "x" << lineRun.logicalHeight() << ")";
159 stream.nextLine();
160 mismatched = true;
161 }
162 ++runIndex;
163 }
164 return mismatched;
165}

167static bool outputMismatchingBlockBoxInformationIfNeeded(TextStream& stream, const LayoutState& layoutState, const RenderBox& renderer, const Box& layoutBox)
168{
169 bool firstMismatchingRect = true;
170 auto outputRect = [&] (const String& prefix, const LayoutRect& rendererRect, const LayoutRect& layoutRect) {
171 if (firstMismatchingRect) {
172 stream << (renderer.element() ? renderer.element()->nodeName().utf8().data() : "") << " " << renderer.renderName() << "(" << &renderer << ") layoutBox(" << &layoutBox << ")";
173 stream.nextLine();
174 firstMismatchingRect = false;
175 }
176
177 stream << prefix.utf8().data() << "\trenderer->(" << rendererRect.x() << "," << rendererRect.y() << ") (" << rendererRect.width() << "x" << rendererRect.height() << ")"
178 << "\tlayout->(" << layoutRect.x() << "," << layoutRect.y() << ") (" << layoutRect.width() << "x" << layoutRect.height() << ")";
179 stream.nextLine();
180 };
181
182 auto renderBoxLikeMarginBox = [&] (const auto& boxGeometry) {
183 if (layoutBox.isInitialContainingBlock())
184 return BoxGeometry::borderBoxRect(boxGeometry);
185
186 // Produce a RenderBox matching margin box.
187 auto containingBlockWidth = layoutState.geometryForBox(layoutBox.containingBlock()).contentBoxWidth();
188 auto marginStart = LayoutUnit { };
189 auto& marginStartStyle = layoutBox.style().marginStart();
190 if (marginStartStyle.isFixed() || marginStartStyle.isPercent() || marginStartStyle.isCalculated())
191 marginStart = valueForLength(marginStartStyle, containingBlockWidth);
192
193 auto marginEnd = LayoutUnit { };
194 auto& marginEndStyle = layoutBox.style().marginEnd();
195 if (marginEndStyle.isFixed() || marginEndStyle.isPercent() || marginEndStyle.isCalculated())
196 marginEnd = valueForLength(marginEndStyle, containingBlockWidth);
197
198 auto marginBefore = boxGeometry.marginBefore();
199 auto marginAfter = boxGeometry.marginAfter();
200 if (layoutBox.formattingContextRoot().establishesBlockFormattingContext()) {
201 auto& formattingState = downcast<BlockFormattingState>(layoutState.formattingStateForBox(layoutBox));
202 auto verticalMargin = formattingState.usedVerticalMargin(layoutBox);
203 marginBefore = verticalMargin.nonCollapsedValues.before;
204 marginAfter = verticalMargin.nonCollapsedValues.after;
205 }
206 auto borderBox = boxGeometry.borderBox();
207 return Rect {
208 borderBox.top() - marginBefore,
209 borderBox.left() - marginStart,
210 marginStart + borderBox.width() + marginEnd,
211 marginBefore + borderBox.height() + marginAfter
212 };
213 };
214
215 // rendering does not offset for relative positioned boxes.
216 auto frameRect = renderer.frameRect();
217 if (renderer.isInFlowPositioned())
218 frameRect.move(renderer.offsetForInFlowPosition());
219
220 auto boxGeometry = BoxGeometry { layoutState.geometryForBox(layoutBox) };
221 if (layoutBox.isTableBox()) {
222 // When the <table> is out-of-flow positioned, the wrapper table box has the offset
223 // while the actual table box is static, inflow.
224 auto& tableWrapperBoxGeometry = layoutState.geometryForBox(layoutBox.containingBlock());
225 boxGeometry.moveBy(BoxGeometry::borderBoxTopLeft(tableWrapperBoxGeometry));
226 // Table wrapper box has the margin values for the table.
227 boxGeometry.setHorizontalMargin(tableWrapperBoxGeometry.horizontalMargin());
228 boxGeometry.setVerticalMargin(tableWrapperBoxGeometry.verticalMargin());
229 }
230
231 if (is<RenderTableRow>(renderer) || is<RenderTableSection>(renderer)) {
232 // Table rows and tbody have 0 width for some reason when border collapsing is on.
233 if (is<RenderTableRow>(renderer) && downcast<RenderTableRow>(renderer).table()->collapseBorders())
234 return false;
235 // Section borders are either collapsed or ignored. However they may produce negative padding boxes.
236 if (is<RenderTableSection>(renderer) && (downcast<RenderTableSection>(renderer).table()->collapseBorders() || renderer.style().hasBorder()))
237 return false;
238 }
239 if (!areEssentiallyEqual(frameRect, BoxGeometry::borderBoxRect(boxGeometry))) {
240 outputRect("frameBox", renderer.frameRect(), BoxGeometry::borderBoxRect(boxGeometry));
241 return true;
242 }
243
244 if (!areEssentiallyEqual(renderer.borderBoxRect(), boxGeometry.borderBox())) {
245 outputRect("borderBox", renderer.borderBoxRect(), boxGeometry.borderBox());
246 return true;
247 }
248
249 // When the table row border overflows the row, padding box becomes negative and content box is incorrect.
250 auto shouldCheckPaddingAndContentBox = !is<RenderTableRow>(renderer) || renderer.paddingBoxRect().width() >= 0;
251 if (shouldCheckPaddingAndContentBox && !areEssentiallyEqual(renderer.paddingBoxRect(), boxGeometry.paddingBox())) {
252 outputRect("paddingBox", renderer.paddingBoxRect(), boxGeometry.paddingBox());
253 return true;
254 }
255
256 auto shouldCheckContentBox = [&] {
257 if (!shouldCheckPaddingAndContentBox)
258 return false;
259 // FIXME: Figure out why trunk/rendering comes back with odd values for <tbody> and <td> content box.
260 if (is<RenderTableCell>(renderer) || is<RenderTableSection>(renderer))
261 return false;
262 // Tables have 0 content box size for some reason when border collapsing is on.
263 return !is<RenderTable>(renderer) || !downcast<RenderTable>(renderer).collapseBorders();
264 }();
265 if (shouldCheckContentBox && !areEssentiallyEqual(renderer.contentBoxRect(), boxGeometry.contentBox())) {
266 outputRect("contentBox", renderer.contentBoxRect(), boxGeometry.contentBox());
267 return true;
268 }
269
270 if (!areEssentiallyEqual(renderer.marginBoxRect(), renderBoxLikeMarginBox(boxGeometry))) {
271 // In certain cases, like out-of-flow boxes with margin auto, marginBoxRect() returns 0. It's clearly incorrect,
272 // so let's check the individual margin values instead (and at this point we know that all other boxes match).
273 auto marginsMatch = boxGeometry.marginBefore() == renderer.marginBefore()
274 && boxGeometry.marginAfter() == renderer.marginAfter()
275 && boxGeometry.marginStart() == renderer.marginStart()
276 && boxGeometry.marginEnd() == renderer.marginEnd();
277
278 if (!marginsMatch) {
279 outputRect("marginBox", renderer.marginBoxRect(), renderBoxLikeMarginBox(boxGeometry));
280 return true;
281 }
282 }
283
284 return false;
285}

287static bool verifyAndOutputSubtree(TextStream& stream, const LayoutState& context, const RenderBox& renderer, const Box& layoutBox)
288{
289 // Rendering code does not have the concept of table wrapper box. Skip it by verifying the first child(table box) instead.
290 if (layoutBox.isTableWrapperBox())
291 return verifyAndOutputSubtree(stream, context, renderer, *downcast<ContainerBox>(layoutBox).firstChild());
292
293 auto mismtachingGeometry = outputMismatchingBlockBoxInformationIfNeeded(stream, context, renderer, layoutBox);
294
295 if (!is<ContainerBox>(layoutBox))
296 return mismtachingGeometry;
297
298 auto& containerBox = downcast<ContainerBox>(layoutBox);
299 auto* childLayoutBox = containerBox.firstChild();
300 auto* childRenderer = renderer.firstChild();
301
302 while (childRenderer) {
303 if (!is<RenderBox>(*childRenderer)) {
304 childRenderer = childRenderer->nextSibling();
305 continue;
306 }
307
308 if (!childLayoutBox) {
309 stream << "Trees are out of sync!";
310 stream.nextLine();
311 return true;
312 }
313
314 if (is<RenderBlockFlow>(*childRenderer) && childLayoutBox->establishesInlineFormattingContext()) {
315 ASSERT(childRenderer->childrenInline());
316 auto mismtachingGeometry = outputMismatchingBlockBoxInformationIfNeeded(stream, context, downcast<RenderBox>(*childRenderer), *childLayoutBox);
317 if (mismtachingGeometry)
318 return true;
319
320 auto& blockFlow = downcast<RenderBlockFlow>(*childRenderer);
321 auto& formattingRoot = downcast<ContainerBox>(*childLayoutBox);
322 mismtachingGeometry |= outputMismatchingComplexLineInformationIfNeeded(stream, context, blockFlow, formattingRoot);
323 } else {
324 auto mismatchingSubtreeGeometry = verifyAndOutputSubtree(stream, context, downcast<RenderBox>(*childRenderer), *childLayoutBox);
325 mismtachingGeometry |= mismatchingSubtreeGeometry;
326 }
327
328 childLayoutBox = childLayoutBox->nextSibling();
329 childRenderer = childRenderer->nextSibling();
330 }
331
332 return mismtachingGeometry;
333}

335void LayoutContext::verifyAndOutputMismatchingLayoutTree(const LayoutState& layoutState, const RenderView& rootRenderer)
336{
337 TextStream stream;
338 auto& layoutRoot = layoutState.root();
339 auto mismatchingGeometry = verifyAndOutputSubtree(stream, layoutState, rootRenderer, layoutRoot);
340 if (!mismatchingGeometry)
341 return;
342#if ENABLE(TREE_DEBUGGING)
343 showRenderTree(&rootRenderer);
344 showLayoutTree(layoutRoot, &layoutState);
345#endif
346 WTFLogAlways("%s", stream.release().utf8().data());
347 ASSERT_NOT_REACHED();
348}
