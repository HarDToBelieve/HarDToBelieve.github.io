37namespace WebCore {
38
39namespace Layout {
40
41class FloatAvoider {
42 WTF_MAKE_ISO_ALLOCATED(FloatAvoider);
43public:
44 FloatAvoider(LayoutPoint absoluteTopLeft, LayoutUnit borderBoxWidth, const Edges& margin, HorizontalEdges containingBlockAbsoluteContentBox, bool isFloatingPositioned, bool isLeftAligned);
45 virtual ~FloatAvoider() = default;
46
47 void setHorizontalPosition(LayoutUnit);
48 void setVerticalPosition(LayoutUnit);
49 void resetHorizontalPosition() { m_absoluteTopLeft.setX(initialHorizontalPosition()); }
50
51 bool overflowsContainingBlock() const;
52
53 LayoutUnit top() const;
54 LayoutUnit left() const;
55 LayoutUnit right() const;
56
57 bool isLeftAligned() const { return m_isLeftAligned; }
58
59private:
60 LayoutUnit borderBoxWidth() const { return m_borderBoxWidth; }
61 LayoutUnit initialHorizontalPosition() const;
62
63 LayoutUnit marginBefore() const { return m_margin.vertical.top; }
64 LayoutUnit marginAfter() const { return m_margin.vertical.bottom; }
65 LayoutUnit marginStart() const { return m_margin.horizontal.left; }
66 LayoutUnit marginEnd() const { return m_margin.horizontal.right; }
67 LayoutUnit marginBoxWidth() const { return marginStart() + borderBoxWidth() + marginEnd(); }
68
69 bool isFloatingBox() const { return m_isFloatingPositioned; }
70
71private:
72 // These coordinate values are relative to the formatting root's border box.
73 LayoutPoint m_absoluteTopLeft;
74 // Note that float avoider should work with no height value.
75 LayoutUnit m_borderBoxWidth;
76 Edges m_margin;
77 HorizontalEdges m_containingBlockAbsoluteContentBox;
78 bool m_isFloatingPositioned { true };
79 bool m_isLeftAligned { true };
80};
81
82inline LayoutUnit FloatAvoider::top() const
83{
84 auto top = m_absoluteTopLeft.y();
85 if (isFloatingBox())
86 top -= marginBefore();
87 return top;
88}
89
90inline LayoutUnit FloatAvoider::left() const
91{
92 auto left = m_absoluteTopLeft.x();
93 if (isFloatingBox())
94 left -= marginStart();
95 return left;
96}
97
98inline LayoutUnit FloatAvoider::right() const
99{
100 auto right = left() + borderBoxWidth();
101 if (isFloatingBox())
102 right += marginAfter();
103 return right;
104}
105
106}
