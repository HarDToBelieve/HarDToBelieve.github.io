44// Finding the top/left position for a new floating(F)
45// ____ ____ _____ _______
46// | || L2 || | <-----1---->| |
47// | ||____|| L3 | | R1 |
48// | L1 | |_____| | |
49// |____| <-------------2--------->| |
50// | |
51// |_______|
52//
53// 1. Compute the initial vertical position for (F) -> (1)
54// 2. Find the corresponding floating pair (L3-R1)
55// 3. Align (F) horizontally with (L3-R1) depending whether (F) is left/right positioned
56// 4. Intersect (F) with (L3-R1)
57// 5. If (F) does not fit, find the next floating pair (L1-R1)
58// 6. Repeat until either (F) fits/no more floats.
59// Note that all coordinates are in the coordinate system of the formatting root.
60// The formatting root here is always the one that establishes the floating context (see inherited floating context).
61// (It simply means that the float box's formatting root is not necessarily the same as the FormattingContext's root.)
62

125#if ASSERT_ENABLED
126static bool areFloatsHorizontallySorted(const FloatingState& floatingState)
127{
128 auto& floats = floatingState.floats();
129 auto rightEdgeOfLeftFloats = LayoutUnit::min();
130 auto leftEdgeOfRightFloats = LayoutUnit::max();
131 std::optional<LayoutUnit> leftBottom;
132 std::optional<LayoutUnit> rightBottom;
133
134 for (auto& floatItem : floats) {
135 if (floatItem.isLeftPositioned()) {
136 auto rightEdge = floatItem.rectWithMargin().right();
137 if (rightEdge < rightEdgeOfLeftFloats) {
138 if (leftBottom && floatItem.rectWithMargin().top() < *leftBottom)
139 return false;
140 }
141 leftBottom = floatItem.rectWithMargin().bottom();
142 rightEdgeOfLeftFloats = rightEdge;
143 } else {
144 auto leftEdge = floatItem.rectWithMargin().left();
145 if (leftEdge > leftEdgeOfRightFloats) {
146 if (rightBottom && floatItem.rectWithMargin().top() < *rightBottom)
147 return false;
148 }
149 rightBottom = floatItem.rectWithMargin().bottom();
150 leftEdgeOfRightFloats = leftEdge;
151 }
152 }
153 return true;
154}
155#endif

157static ComputedHorizontalMargin computedHorizontalMargin(const Box& layoutBox, LayoutUnit horizontalConstraint)
158{
159 auto& style = layoutBox.style();
160 auto computedValue = [&] (const auto& margin) {
161 if (margin.isFixed() || margin.isPercent() || margin.isCalculated())
162 return valueForLength(margin, horizontalConstraint);
163 return 0_lu;
164 };
165 return { computedValue(style.marginStart()), computedValue(style.marginEnd()) };
166}

168static FloatPair::LeftRightIndex findAvailablePosition(FloatAvoider& floatAvoider, const FloatingState::FloatList& floats)
169{
170 std::optional<PositionInContextRoot> bottomMost;
171 std::optional<FloatPair::LeftRightIndex> innerMostLeftAndRight;
172 auto end = Layout::end(floats);
173 for (auto iterator = begin(floats, { floatAvoider.top() }); iterator != end; ++iterator) {
174 ASSERT(!(*iterator).isEmpty());
175 auto leftRightFloatPair = *iterator;
176 innerMostLeftAndRight = innerMostLeftAndRight.value_or(*leftRightFloatPair);
177
178 // Move the box horizontally so that it either
179 // 1. aligns with the current floating pair
180 // 2. or with the containing block's content box if there's no float to align with at this vertical position.
181 auto leftRightEdge = leftRightFloatPair.horizontalConstraints();
182 if (auto horizontalConstraint = floatAvoider.isLeftAligned() ? leftRightEdge.left : leftRightEdge.right)
183 floatAvoider.setHorizontalPosition(*horizontalConstraint);
184 else
185 floatAvoider.resetHorizontalPosition();
186 floatAvoider.setVerticalPosition(leftRightFloatPair.verticalConstraint());
187
188 // Ensure that the float avoider
189 // 1. does not "overflow" its containing block with the current horiztonal constraints. It simply means that the float avoider's
190 // containing block could push the candidate position beyond the current float horizontally (too far to the left/right)
191 // 2. avoids floats on both sides.
192 if (!floatAvoider.overflowsContainingBlock() && !leftRightFloatPair.intersects(floatAvoider))
193 return *innerMostLeftAndRight;
194
195 bottomMost = leftRightFloatPair.bottom();
196 // Move to the next floating pair.
197 }
198
199 // The candidate box is already below of all the floats.
200 if (!bottomMost)
201 return { };
202
203 // Passed all the floats and still does not fit? Push it below the last float.
204 floatAvoider.setVerticalPosition(*bottomMost);
205 floatAvoider.resetHorizontalPosition();
206 ASSERT(innerMostLeftAndRight);
207 return *innerMostLeftAndRight;
208}

222LayoutPoint FloatingContext::positionForFloat(const Box& layoutBox, const HorizontalConstraints& horizontalConstraints) const
223{
224 ASSERT(layoutBox.isFloatingPositioned());
225 ASSERT(areFloatsHorizontallySorted(m_floatingState));
226
227 if (isEmpty()) {
228 auto& boxGeometry = formattingContext().geometryForBox(layoutBox);
229 auto alignWithContainingBlock = [&]() -> Position {
230 // If there is no floating to align with, push the box to the left/right edge of its containing block's content box.
231 if (layoutBox.isLeftFloatingPositioned())
232 return { horizontalConstraints.logicalLeft + boxGeometry.marginStart() };
233 return { horizontalConstraints.logicalRight() - boxGeometry.marginEnd() - boxGeometry.borderBoxWidth() };
234 };
235 // No float box on the context yet -> align it with the containing block's left/right edge.
236 return { alignWithContainingBlock(), BoxGeometry::borderBoxTop(boxGeometry) };
237 }
238
239 // Find the top most position where the float box fits.
240 ASSERT(!isEmpty());
241 auto absoluteCoordinates = this->absoluteCoordinates(layoutBox);
242 auto absoluteTopLeft = absoluteCoordinates.topLeft;
243 auto verticalPositionCandidate = absoluteTopLeft.y();
244
245 auto& boxGeometry = formattingContext().geometryForBox(layoutBox);
246 if (layoutBox.hasFloatClear()) {
247 // The vertical position candidate needs to clear the existing floats in this context.
248 auto floatBottom = [&]() -> std::optional<LayoutUnit> {
249 switch (layoutBox.style().clear()) {
250 case Clear::Left:
251 return leftBottom();
252 case Clear::Right:
253 return rightBottom();
254 case Clear::Both:
255 return bottom();
256 default:
257 ASSERT_NOT_REACHED();
258 }
259 return { };
260 };
261 if (auto bottomWithClear = floatBottom())
262 verticalPositionCandidate = *bottomWithClear + boxGeometry.marginBefore();
263 } else {
264 // Incoming float cannot be placed higher than existing floats (margin box of the last float).
265 // Take the static position (where the box would go if it wasn't floating) and adjust it with the last float.
266 auto previousFloatAbsoluteTop = floatingState().floats().last().rectWithMargin().top();
267 if (verticalPositionCandidate - boxGeometry.marginBefore() < previousFloatAbsoluteTop)
268 verticalPositionCandidate = previousFloatAbsoluteTop + boxGeometry.marginBefore();
269 }
270 absoluteTopLeft.setY(verticalPositionCandidate);
271 auto horizontalMargin = computedHorizontalMargin(layoutBox, horizontalConstraints.logicalWidth);
272 auto margins = Edges { { *horizontalMargin.start, *horizontalMargin.end }, { boxGeometry.marginBefore(), boxGeometry.marginAfter() } };
273 auto isLeftAligned = layoutBox.isFloatingPositioned() ? layoutBox.isLeftFloatingPositioned() : layoutBox.style().isLeftToRightDirection();
274 auto floatBox = FloatAvoider { absoluteTopLeft, boxGeometry.borderBoxWidth(), margins, absoluteCoordinates.containingBlockContentBox, layoutBox.isFloatingPositioned(), isLeftAligned };
275 findAvailablePosition(floatBox, m_floatingState.floats());
276 // Convert box coordinates from formatting root back to containing block.
277 auto containingBlockTopLeft = absoluteCoordinates.containingBlockTopLeft;
278 return { floatBox.left() + margins.horizontal.left - containingBlockTopLeft.x(), floatBox.top() + margins.vertical.top - containingBlockTopLeft.y() };
279}

281LayoutPoint FloatingContext::positionForNonFloatingFloatAvoider(const Box& layoutBox, const HorizontalConstraints& horizontalConstraints) const
282{
283 ASSERT(layoutBox.establishesBlockFormattingContext());
284 ASSERT(!layoutBox.isFloatingPositioned());
285 ASSERT(!layoutBox.hasFloatClear());
286 ASSERT(areFloatsHorizontallySorted(m_floatingState));
287
288 if (isEmpty())
289 return BoxGeometry::borderBoxTopLeft(formattingContext().geometryForBox(layoutBox));
290
291 auto absoluteCoordinates = this->absoluteCoordinates(layoutBox);
292 auto& boxGeometry = formattingContext().geometryForBox(layoutBox);
293 auto horizontalMargin = computedHorizontalMargin(layoutBox, horizontalConstraints.logicalWidth);
294 auto margins = Edges { { *horizontalMargin.start, *horizontalMargin.end }, { boxGeometry.marginBefore(), boxGeometry.marginAfter() } };
295 auto isLeftAligned = layoutBox.isFloatingPositioned() ? layoutBox.isLeftFloatingPositioned() : layoutBox.style().isLeftToRightDirection();
296 auto floatAvoider = FloatAvoider { absoluteCoordinates.topLeft, boxGeometry.borderBoxWidth(), margins, absoluteCoordinates.containingBlockContentBox, layoutBox.isFloatingPositioned(), isLeftAligned };
297 findPositionForFormattingContextRoot(floatAvoider);
298 auto containingBlockTopLeft = absoluteCoordinates.containingBlockTopLeft;
299 return { floatAvoider.left() - containingBlockTopLeft.x(), floatAvoider.top() - containingBlockTopLeft.y() };
300}

302std::optional<FloatingContext::PositionWithClearance> FloatingContext::verticalPositionWithClearance(const Box& layoutBox) const
303{
304 ASSERT(layoutBox.hasFloatClear());
305 ASSERT(areFloatsHorizontallySorted(m_floatingState));
306
307 if (isEmpty())
308 return { };
309
310 auto bottom = [&](auto floatBottom) -> std::optional<PositionWithClearance> {
311 if (!floatBottom)
312 return { };
313 // 9.5.2 Controlling flow next to floats: the 'clear' property
314 // Then the amount of clearance is set to the greater of:
315 //
316 // 1. The amount necessary to place the border edge of the block even with the bottom outer edge of the lowest float that is to be cleared.
317 // 2. The amount necessary to place the top border edge of the block at its hypothetical position.
318 auto logicalTopRelativeToFloatingStateRoot = mapTopLeftToFloatingStateRoot(layoutBox).y();
319 auto clearance = *floatBottom - logicalTopRelativeToFloatingStateRoot;
320 if (clearance <= 0)
321 return { };
322
323 if (layoutBox.isBlockLevelBox()) {
324 // Clearance inhibits margin collapsing in block formatting context.
325 if (auto* previousInFlowSibling = layoutBox.previousInFlowSibling()) {
326 // Does this box with clearance actually collapse its margin before with the previous inflow box's margin after?
327 auto& formattingState = downcast<BlockFormattingState>(layoutState().formattingStateForBox(layoutBox));
328 auto verticalMargin = formattingState.usedVerticalMargin(layoutBox);
329 if (verticalMargin.collapsedValues.before) {
330 auto previousVerticalMarginAfter = formattingContext().geometryForBox(*previousInFlowSibling).marginAfter();
331 auto collapsedMargin = *verticalMargin.collapsedValues.before;
332 auto nonCollapsedMargin = previousVerticalMarginAfter + marginBefore(verticalMargin);
333 auto marginDifference = nonCollapsedMargin - collapsedMargin;
334 // Move the box to the position where it would be with non-collapsed margins.
335 logicalTopRelativeToFloatingStateRoot += marginDifference;
336 // Having negative clearance is also normal. It just means that the box with the non-collapsed margins is now lower than it needs to be.
337 clearance -= marginDifference;
338 }
339 }
340 }
341 // Now adjust the box's position with the clearance.
342 logicalTopRelativeToFloatingStateRoot += clearance;
343 ASSERT(*floatBottom == logicalTopRelativeToFloatingStateRoot);
344
345 // The return vertical position needs to be in the containing block's coordinate system.
346 if (&layoutBox.containingBlock() == &m_floatingState.root())
347 return PositionWithClearance { logicalTopRelativeToFloatingStateRoot, clearance };
348
349 auto containingBlockRootRelativeTop = mapTopLeftToFloatingStateRoot(layoutBox.containingBlock()).y();
350 return PositionWithClearance { logicalTopRelativeToFloatingStateRoot - containingBlockRootRelativeTop, clearance };
351 };
352
353 auto clear = layoutBox.style().clear();
354 if (clear == Clear::Left)
355 return bottom(leftBottom());
356
357 if (clear == Clear::Right)
358 return bottom(rightBottom());
359
360 if (clear == Clear::Both)
361 return bottom(this->bottom());
362
363 ASSERT_NOT_REACHED();
364 return { };
365}

367std::optional<LayoutUnit> FloatingContext::bottom(Clear type) const
368{
369 // TODO: Currently this is only called once for each formatting context root with floats per layout.
370 // Cache the value if we end up calling it more frequently (and update it at append/remove).
371 auto bottom = std::optional<LayoutUnit> { };
372 for (auto& floatItem : floatingState().floats()) {
373 if ((type == Clear::Left && !floatItem.isLeftPositioned())
374 || (type == Clear::Right && floatItem.isLeftPositioned()))
375 continue;
376 bottom = !bottom ? floatItem.rectWithMargin().bottom() : std::max(*bottom, floatItem.rectWithMargin().bottom());
377 }
378 return bottom;
379}
380
381std::optional<LayoutUnit> FloatingContext::top() const
382{
383 auto top = std::optional<LayoutUnit> { };
384 for (auto& floatItem : floatingState().floats())
385 top = !top ? floatItem.rectWithMargin().top() : std::min(*top, floatItem.rectWithMargin().top());
386 return top;
387}

389FloatingContext::Constraints FloatingContext::constraints(LayoutUnit candidateTop, LayoutUnit candidateBottom) const
390{
391 if (isEmpty())
392 return { };
393 // 1. Convert vertical position if this floating context is inherited.
394 // 2. Find the inner left/right floats at candidateTop/candidateBottom.
395 // 3. Convert left/right positions back to formattingContextRoot's cooridnate system.
396 auto coordinateMappingIsRequired = &floatingState().root() != &root();
397 auto adjustedCandidateTop = candidateTop;
398 LayoutSize adjustingDelta;
399 if (coordinateMappingIsRequired) {
400 auto adjustedCandidatePosition = mapPointFromFormattingContextRootToFloatingStateRoot({ 0, candidateTop });
401 adjustedCandidateTop = adjustedCandidatePosition.y;
402 adjustingDelta = { adjustedCandidatePosition.x, adjustedCandidateTop - candidateTop };
403 }
404 auto adjustedCandidateBottom = adjustedCandidateTop + (candidateBottom - candidateTop);
405 auto isCandidateEmpty = adjustedCandidateTop == adjustedCandidateBottom;
406
407 Constraints constraints;
408 auto& floats = floatingState().floats();
409 for (auto index = floats.size(); index--;) {
410 auto& floatItem = floats[index];
411
412 if (constraints.left && floatItem.isLeftPositioned())
413 continue;
414
415 if (constraints.right && !floatItem.isLeftPositioned())
416 continue;
417
418 auto floatBoxRect = floatItem.rectWithMargin();
419 auto contains = [&] {
420 if (isCandidateEmpty)
421 return floatBoxRect.top() <= adjustedCandidateTop && floatBoxRect.bottom() > adjustedCandidateTop;
422 return floatBoxRect.top() < adjustedCandidateBottom && floatBoxRect.bottom() > adjustedCandidateTop;
423 };
424 if (!contains())
425 continue;
426
427 if (floatItem.isLeftPositioned())
428 constraints.left = PointInContextRoot { floatBoxRect.right(), floatBoxRect.bottom() };
429 else
430 constraints.right = PointInContextRoot { floatBoxRect.left(), floatBoxRect.bottom() };
431
432 if (constraints.left && constraints.right)
433 break;
434 }
435
436 if (coordinateMappingIsRequired) {
437 if (constraints.left)
438 constraints.left->move(-adjustingDelta);
439
440 if (constraints.right)
441 constraints.right->move(-adjustingDelta);
442 }
443 return constraints;
444}

453void FloatingContext::findPositionForFormattingContextRoot(FloatAvoider& floatAvoider) const
454{
455 // A non-floating formatting root's initial vertical position is its static position.
456 // It means that such boxes can end up vertically placed in-between existing floats (which is
457 // never the case for floats, since they cannot be placed above existing floats).
458 // ____ ____
459 // | || F1 |
460 // | L1 | ----
461 // | | ________
462 // ---- | R1 |
463 // --------
464 // Document order: 1. float: left (L1) 2. float: right (R1) 3. formatting root (F1)
465 //
466 // 1. Probe for available placement at initial position (note it runs a backward probing algorithm at a specific vertical position)
467 // 2. Check if there's any intersecing float below (forward seaching)
468 // 3. Align the box with the intersected float and probe for placement again (#1).
469 auto& floats = m_floatingState.floats();
470 while (true) {
471 auto innerMostLeftAndRight = findAvailablePosition(floatAvoider, floats);
472 if (innerMostLeftAndRight.isEmpty())
473 return;
474
475 auto overlappingFloatBox = [&floats](auto startFloatIndex, auto& floatAvoider) -> const FloatingState::FloatItem* {
476 for (auto i = startFloatIndex; i < floats.size(); ++i) {
477 auto& floatBox = floats[i];
478
479 auto intersects = [&] {
480 auto floatingRect = floatBox.rectWithMargin();
481 if (floatAvoider.left() >= floatingRect.right() || floatAvoider.right() <= floatingRect.left())
482 return false;
483 return floatAvoider.top() >= floatingRect.top() && floatAvoider.top() < floatingRect.bottom();
484 }();
485 if (intersects)
486 return &floatBox;
487 }
488 return nullptr;
489 };
490
491 auto startIndex = std::max(innerMostLeftAndRight.left.value_or(0), innerMostLeftAndRight.right.value_or(0)) + 1;
492 auto* intersectedFloatBox = overlappingFloatBox(startIndex, floatAvoider);
493 if (!intersectedFloatBox)
494 return;
495 floatAvoider.setVerticalPosition({ intersectedFloatBox->rectWithMargin().top() });
496 }
497}

499FloatingContext::AbsoluteCoordinateValuesForFloatAvoider FloatingContext::absoluteCoordinates(const Box& floatAvoider) const
500{
501 auto& containingBlock = floatAvoider.containingBlock();
502 auto& containingBlockGeometry = formattingContext().geometryForBox(containingBlock, FormattingContext::EscapeReason::FloatBoxIsAlwaysRelativeToFloatStateRoot);
503 auto absoluteTopLeft = mapTopLeftToFloatingStateRoot(floatAvoider);
504
505 if (&containingBlock == &floatingState().root())
506 return { absoluteTopLeft, { }, { containingBlockGeometry.contentBoxLeft(), containingBlockGeometry.contentBoxRight() } };
507
508 auto containingBlockAbsoluteTopLeft = mapTopLeftToFloatingStateRoot(containingBlock);
509 return { absoluteTopLeft, containingBlockAbsoluteTopLeft, { containingBlockAbsoluteTopLeft.x() + containingBlockGeometry.contentBoxLeft(), containingBlockAbsoluteTopLeft.x() + containingBlockGeometry.contentBoxRight() } };
510}
511
512LayoutPoint FloatingContext::mapTopLeftToFloatingStateRoot(const Box& floatBox) const
513{
514 auto& floatingStateRoot = floatingState().root();
515 auto topLeft = BoxGeometry::borderBoxTopLeft(formattingContext().geometryForBox(floatBox, FormattingContext::EscapeReason::FloatBoxIsAlwaysRelativeToFloatStateRoot));
516 for (auto& containingBlock : containingBlockChain(floatBox, floatingStateRoot))
517 topLeft.moveBy(BoxGeometry::borderBoxTopLeft(formattingContext().geometryForBox(containingBlock, FormattingContext::EscapeReason::FloatBoxIsAlwaysRelativeToFloatStateRoot)));
518 return topLeft;
519}
520
521Point FloatingContext::mapPointFromFormattingContextRootToFloatingStateRoot(Point position) const
522{
523 auto& from = root();
524 auto& to = floatingState().root();
525 if (&from == &to)
526 return position;
527 auto mappedPosition = position;
528 for (auto* containingBlock = &from; containingBlock != &to; containingBlock = &containingBlock->containingBlock())
529 mappedPosition.moveBy(BoxGeometry::borderBoxTopLeft(formattingContext().geometryForBox(*containingBlock, FormattingContext::EscapeReason::FloatBoxIsAlwaysRelativeToFloatStateRoot)));
530 return mappedPosition;
531}

556bool FloatPair::intersects(const FloatAvoider& floatAvoider) const
557{
558 auto intersects = [&](auto* floating) {
559 if (!floating)
560 return false;
561 auto floatingRect = floating->rectWithMargin();
562 if (floatAvoider.left() >= floatingRect.right() || floatAvoider.right() <= floatingRect.left())
563 return false;
564 return floatAvoider.top() >= floatingRect.top() && floatAvoider.top() < floatingRect.bottom();
565 };
566
567 ASSERT(!m_floatPair.isEmpty());
568 return intersects(left()) || intersects(right());
569}

571bool FloatPair::operator ==(const FloatPair& other) const
572{
573 return m_floatPair.left == other.m_floatPair.left && m_floatPair.right == other.m_floatPair.right;
574}

576FloatPair::HorizontalConstraints FloatPair::horizontalConstraints() const
577{
578 std::optional<PositionInContextRoot> leftEdge;
579 std::optional<PositionInContextRoot> rightEdge;
580
581 if (left())
582 leftEdge = PositionInContextRoot { left()->rectWithMargin().right() };
583
584 if (right())
585 rightEdge = PositionInContextRoot { right()->rectWithMargin().left() };
586
587 return { leftEdge, rightEdge };
588}
589
590PositionInContextRoot FloatPair::bottom() const
591{
592 auto* left = this->left();
593 auto* right = this->right();
594 ASSERT(left || right);
595
596 auto leftBottom = left ? std::optional<PositionInContextRoot>(PositionInContextRoot { left->rectWithMargin().bottom() }) : std::nullopt;
597 auto rightBottom = right ? std::optional<PositionInContextRoot>(PositionInContextRoot { right->rectWithMargin().bottom() }) : std::nullopt;
598
599 if (leftBottom && rightBottom)
600 return std::max(*leftBottom, *rightBottom);
601
602 if (leftBottom)
603 return *leftBottom;
604
605 return *rightBottom;
606}

616inline static std::optional<unsigned> previousFloatingIndex(Float floatingType, const FloatingState::FloatList& floats, unsigned currentIndex)
617{
618 RELEASE_ASSERT(currentIndex <= floats.size());
619
620 while (currentIndex) {
621 auto& floating = floats[--currentIndex];
622 if ((floatingType == Float::Left && floating.isLeftPositioned()) || (floatingType == Float::Right && !floating.isLeftPositioned()))
623 return currentIndex;
624 }
625
626 return { };
627}
628
629Iterator& Iterator::operator++()
630{
631 if (m_current.isEmpty()) {
632 ASSERT_NOT_REACHED();
633 return *this;
634 }
635
636 auto findPreviousFloatingWithLowerBottom = [&](Float floatingType, unsigned currentIndex) -> std::optional<unsigned> {
637
638 RELEASE_ASSERT(currentIndex < m_floats.size());
639
640 // Last floating? There's certainly no previous floating at this point.
641 if (!currentIndex)
642 return { };
643
644 auto currentBottom = m_floats[currentIndex].rectWithMargin().bottom();
645
646 std::optional<unsigned> index = currentIndex;
647 while (true) {
648 index = previousFloatingIndex(floatingType, m_floats, *index);
649 if (!index)
650 return { };
651
652 if (m_floats[*index].rectWithMargin().bottom() > currentBottom)
653 return index;
654 }
655
656 ASSERT_NOT_REACHED();
657 return { };
658 };
659
660 // 1. Take the current floating from left and right and check which one's bottom edge is positioned higher (they could be on the same vertical position too).
661 // The current floats from left and right are considered the inner-most pair for the current vertical position.
662 // 2. Move away from inner-most pair by picking one of the previous floats in the list(#1)
663 // Ensure that the new floating's bottom edge is positioned lower than the current one -which essentially means skipping in-between floats that are positioned higher).
664 // 3. Reset the vertical position and align it with the new left-right pair. These floats are now the inner-most boxes for the current vertical position.
665 // As the result we have more horizontal space on the current vertical position.
666 auto leftBottom = m_current.left() ? std::optional<PositionInContextRoot>(m_current.left()->bottom()) : std::nullopt;
667 auto rightBottom = m_current.right() ? std::optional<PositionInContextRoot>(m_current.right()->bottom()) : std::nullopt;
668
669 auto updateLeft = (leftBottom == rightBottom) || (!rightBottom || (leftBottom && *leftBottom < *rightBottom));
670 auto updateRight = (leftBottom == rightBottom) || (!leftBottom || (rightBottom && *leftBottom > *rightBottom));
671
672 if (updateLeft) {
673 ASSERT(m_current.m_floatPair.left);
674 m_current.m_verticalPosition = *leftBottom;
675 m_current.m_floatPair.left = findPreviousFloatingWithLowerBottom(Float::Left, *m_current.m_floatPair.left);
676 }
677
678 if (updateRight) {
679 ASSERT(m_current.m_floatPair.right);
680 m_current.m_verticalPosition = *rightBottom;
681 m_current.m_floatPair.right = findPreviousFloatingWithLowerBottom(Float::Right, *m_current.m_floatPair.right);
682 }
683
684 return *this;
685}

687void Iterator::set(PositionInContextRoot verticalPosition)
688{
689 // Move the iterator to the initial vertical position by starting at the inner-most floating pair (last floats on left/right).
690 // 1. Check if the inner-most pair covers the vertical position.
691 // 2. Move outwards from the inner-most pair until the vertical postion intersects.
692 m_current.m_verticalPosition = verticalPosition;
693 // No floats at all?
694 if (m_floats.isEmpty()) {
695 ASSERT_NOT_REACHED();
696 m_current.m_floatPair = { };
697 return;
698 }
699
700 auto findFloatingBelow = [&](Float floatingType) -> std::optional<unsigned> {
701
702 ASSERT(!m_floats.isEmpty());
703
704 auto index = floatingType == Float::Left ? m_current.m_floatPair.left : m_current.m_floatPair.right;
705 // Start from the end if we don't have current yet.
706 index = index.value_or(m_floats.size());
707 while (true) {
708 index = previousFloatingIndex(floatingType, m_floats, *index);
709 if (!index)
710 return { };
711
712 // Is this floating intrusive on this position?
713 auto rect = m_floats[*index].rectWithMargin();
714 if (rect.top() <= verticalPosition && rect.bottom() > verticalPosition)
715 return index;
716 }
717
718 return { };
719 };
720
721 m_current.m_floatPair.left = findFloatingBelow(Float::Left);
722 m_current.m_floatPair.right = findFloatingBelow(Float::Right);
723
724 ASSERT(!m_current.m_floatPair.left || (*m_current.m_floatPair.left < m_floats.size() && m_floats[*m_current.m_floatPair.left].isLeftPositioned()));
725 ASSERT(!m_current.m_floatPair.right || (*m_current.m_floatPair.right < m_floats.size() && !m_floats[*m_current.m_floatPair.right].isLeftPositioned()));
726}
