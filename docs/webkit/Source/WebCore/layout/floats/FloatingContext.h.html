35namespace WebCore {
36namespace Layout {

38class FloatAvoider;
39class Box;
40class LayoutState;
41
42// FloatingContext is responsible for adjusting the position of a box in the current formatting context
43// by taking the floating boxes into account.
44class FloatingContext {
45 WTF_MAKE_ISO_ALLOCATED(FloatingContext);
46public:
47 FloatingContext(const FormattingContext&, const FloatingState&);
48
49 const FloatingState& floatingState() const { return m_floatingState; }
50
51 LayoutPoint positionForFloat(const Box&, const HorizontalConstraints&) const;
52 LayoutPoint positionForNonFloatingFloatAvoider(const Box&, const HorizontalConstraints&) const;
53
54 struct PositionWithClearance {
55 LayoutUnit position;
56 std::optional<LayoutUnit> clearance;
57 };
58 std::optional<PositionWithClearance> verticalPositionWithClearance(const Box&) const;
59
60 std::optional<LayoutUnit> top() const;
61 std::optional<LayoutUnit> leftBottom() const { return bottom(Clear::Left); }
62 std::optional<LayoutUnit> rightBottom() const { return bottom(Clear::Right); }
63 std::optional<LayoutUnit> bottom() const { return bottom(Clear::Both); }
64
65 bool isEmpty() const { return m_floatingState.floats().isEmpty(); }
66
67 struct Constraints {
68 std::optional<PointInContextRoot> left;
69 std::optional<PointInContextRoot> right;
70 };
71 Constraints constraints(LayoutUnit candidateTop, LayoutUnit candidateBottom) const;
72
73 FloatingState::FloatItem toFloatItem(const Box& floatBox) const;
74
75private:
76 std::optional<LayoutUnit> bottom(Clear) const;
77
78 const LayoutState& layoutState() const { return m_floatingState.layoutState(); }
79 const FormattingContext& formattingContext() const { return m_formattingContext; }
80 const ContainerBox& root() const { return m_formattingContext.root(); }
81
82 void findPositionForFormattingContextRoot(FloatAvoider&) const;
83
84 struct AbsoluteCoordinateValuesForFloatAvoider;
85 AbsoluteCoordinateValuesForFloatAvoider absoluteCoordinates(const Box&) const;
86 LayoutPoint mapTopLeftToFloatingStateRoot(const Box&) const;
87 Point mapPointFromFormattingContextRootToFloatingStateRoot(Point) const;
88
89 const FormattingContext& m_formattingContext;
90 const FloatingState& m_floatingState;
91};

93}
