47static inline bool isHeightAuto(const Box& layoutBox)
48{
49 // 10.5 Content height: the 'height' property
50 //
51 // The percentage is calculated with respect to the height of the generated box's containing block.
52 // If the height of the containing block is not specified explicitly (i.e., it depends on content height),
53 // and this element is not absolutely positioned, the used height is calculated as if 'auto' was specified.
54
55 auto height = layoutBox.style().logicalHeight();
56 if (height.isAuto())
57 return true;
58
59 if (height.isPercent()) {
60 if (layoutBox.isOutOfFlowPositioned())
61 return false;
62
63 return !layoutBox.containingBlock().style().logicalHeight().isFixed();
64 }
65
66 return false;
67}

74std::optional<LayoutUnit> FormattingGeometry::computedHeightValue(const Box& layoutBox, HeightType heightType, std::optional<LayoutUnit> containingBlockHeight) const
75{
76 auto& style = layoutBox.style();
77 auto height = heightType == HeightType::Normal ? style.logicalHeight() : heightType == HeightType::Min ? style.logicalMinHeight() : style.logicalMaxHeight();
78 if (height.isUndefined() || height.isAuto() || height.isMaxContent() || height.isMinContent() || height.isFitContent())
79 return { };
80
81 if (height.isFixed())
82 return LayoutUnit { height.value() };
83
84 if (!containingBlockHeight) {
85 if (layoutState().inQuirksMode())
86 containingBlockHeight = formattingContext().formattingQuirks().heightValueOfNearestContainingBlockWithFixedHeight(layoutBox);
87 else {
88 auto nonAnonymousContainingBlockLogicalHeight = [&]() -> Length {
89 // When the block level box is a direct child of an inline level box (<span><div></div></span>) and we wrap it into a continuation,
90 // the containing block (anonymous wrapper) is not the box we need to check for fixed height.
91 for (auto& containingBlock : containingBlockChain(layoutBox)) {
92 if (containingBlock.isAnonymous())
93 continue;
94 return containingBlock.style().logicalHeight();
95 }
96 ASSERT_NOT_REACHED();
97 return { };
98 };
99 containingBlockHeight = fixedValue(nonAnonymousContainingBlockLogicalHeight());
100 }
101 }
102
103 if (!containingBlockHeight)
104 return { };
105
106 return valueForLength(height, *containingBlockHeight);
107}

109std::optional<LayoutUnit> FormattingGeometry::computedHeight(const Box& layoutBox, std::optional<LayoutUnit> containingBlockHeight) const
110{
111 if (auto height = computedHeightValue(layoutBox, HeightType::Normal, containingBlockHeight)) {
112 if (layoutBox.style().boxSizing() == BoxSizing::ContentBox)
113 return height;
114 auto& boxGeometry = formattingContext().geometryForBox(layoutBox);
115 return *height - (boxGeometry.verticalBorder() + boxGeometry.verticalPadding().value_or(0));
116 }
117 return { };
118}

120std::optional<LayoutUnit> FormattingGeometry::computedWidthValue(const Box& layoutBox, WidthType widthType, LayoutUnit containingBlockWidth) const
121{
122 // Applies to: all elements except non-replaced inlines (out-of-flow check is required for positioned <br> as for some reason we don't blockify them).
123 ASSERT(!layoutBox.isInlineBox() || layoutBox.isOutOfFlowPositioned());
124
125 auto width = [&] {
126 auto& style = layoutBox.style();
127 switch (widthType) {
128 case WidthType::Normal:
129 return style.logicalWidth();
130 case WidthType::Min:
131 return style.logicalMinWidth();
132 case WidthType::Max:
133 return style.logicalMaxWidth();
134 }
135 ASSERT_NOT_REACHED();
136 return style.logicalWidth();
137 }();
138 if (auto computedValue = this->computedValue(width, containingBlockWidth))
139 return computedValue;
140
141 if (width.isMinContent() || width.isMaxContent() || width.isFitContent()) {
142 if (!is<ContainerBox>(layoutBox))
143 return { };
144 auto& containerBox = downcast<ContainerBox>(layoutBox);
145 // FIXME: Consider splitting up computedIntrinsicWidthConstraints so that we could computed the min and max values separately.
146 auto intrinsicWidthConstraints = [&] {
147 if (!containerBox.hasInFlowOrFloatingChild())
148 return IntrinsicWidthConstraints { 0_lu, containingBlockWidth };
149 ASSERT(containerBox.establishesFormattingContext());
150 auto& layoutState = this->layoutState();
151 if (layoutState.hasFormattingState(containerBox)) {
152 if (auto intrinsicWidthConstraints = layoutState.establishedFormattingState(containerBox).intrinsicWidthConstraints())
153 return *intrinsicWidthConstraints;
154 }
155 return LayoutContext::createFormattingContext(containerBox, const_cast<LayoutState&>(layoutState))->computedIntrinsicWidthConstraints();
156 }();
157 if (width.isMinContent())
158 return intrinsicWidthConstraints.minimum;
159 if (width.isMaxContent())
160 return intrinsicWidthConstraints.maximum;
161 ASSERT(width.isFitContent());
162 // If the available space in a given axis is definite, equal to min(max-content size,
163 // max(min-content size, stretch-fit size)). Otherwise, equal to the max-content size in that axis.
164 // FIXME: We don't yet have indefinite available size.
165 return std::min(intrinsicWidthConstraints.maximum, std::max(intrinsicWidthConstraints.minimum, containingBlockWidth));
166 }
167 return { };
168}

170std::optional<LayoutUnit> FormattingGeometry::computedWidth(const Box& layoutBox, LayoutUnit containingBlockWidth) const
171{
172 if (auto computedWidth = computedWidthValue(layoutBox, WidthType::Normal, containingBlockWidth)) {
173 auto& style = layoutBox.style();
174 // Non-quantitative values such as auto and min-content are not influenced by the box-sizing property.
175 if (style.boxSizing() == BoxSizing::ContentBox || style.width().isIntrinsicOrAuto())
176 return computedWidth;
177 auto& boxGeometry = formattingContext().geometryForBox(layoutBox);
178 return *computedWidth - (boxGeometry.horizontalBorder() + boxGeometry.horizontalPadding().value_or(0));
179 }
180 return { };

235LayoutUnit FormattingGeometry::staticVerticalPositionForOutOfFlowPositioned(const Box& layoutBox, const VerticalConstraints& verticalConstraints) const
236{
237 ASSERT(layoutBox.isOutOfFlowPositioned());
238
239 // For the purposes of this section and the next, the term "static position" (of an element) refers, roughly, to the position an element would have
240 // had in the normal flow. More precisely, the static position for 'top' is the distance from the top edge of the containing block to the top margin
241 // edge of a hypothetical box that would have been the first box of the element if its specified 'position' value had been 'static' and its specified
242 // 'float' had been 'none' and its specified 'clear' had been 'none'. (Note that due to the rules in section 9.7 this might require also assuming a different
243 // computed value for 'display'.) The value is negative if the hypothetical box is above the containing block.
244
245 // Start with this box's border box offset from the parent's border box.
246 auto& formattingContext = this->formattingContext();
247 LayoutUnit top;
248 if (layoutBox.previousInFlowSibling() && layoutBox.previousInFlowSibling()->isBlockLevelBox()) {
249 // Add sibling offset
250 auto& previousInFlowSibling = *layoutBox.previousInFlowSibling();
251 auto& previousInFlowBoxGeometry = formattingContext.geometryForBox(previousInFlowSibling, FormattingContext::EscapeReason::OutOfFlowBoxNeedsInFlowGeometry);
252 auto& formattingState = downcast<BlockFormattingState>(layoutState().formattingStateForBox(previousInFlowSibling));
253 auto usedVerticalMarginForPreviousBox = formattingState.usedVerticalMargin(previousInFlowSibling);
254
255 top += BoxGeometry::borderBoxRect(previousInFlowBoxGeometry).bottom() + usedVerticalMarginForPreviousBox.nonCollapsedValues.after;
256 } else
257 top = formattingContext.geometryForBox(layoutBox.parent(), FormattingContext::EscapeReason::OutOfFlowBoxNeedsInFlowGeometry).contentBoxTop();
258
259 // Resolve top all the way up to the containing block.
260 auto& containingBlock = layoutBox.containingBlock();
261 // Start with the parent since we pretend that this box is normal flow.
262 for (auto* ancestor = &layoutBox.parent(); ancestor != &containingBlock; ancestor = &ancestor->containingBlock()) {
263 auto& boxGeometry = formattingContext.geometryForBox(*ancestor, FormattingContext::EscapeReason::OutOfFlowBoxNeedsInFlowGeometry);
264 // BoxGeometry::top is the border box top position in its containing block's coordinate system.
265 top += BoxGeometry::borderBoxTop(boxGeometry);
266 ASSERT(!ancestor->isPositioned() || layoutBox.isFixedPositioned());
267 }
268 // Move the static position relative to the padding box. This is very specific to abolutely positioned boxes.
269 return top - verticalConstraints.logicalTop;
270}

272LayoutUnit FormattingGeometry::staticHorizontalPositionForOutOfFlowPositioned(const Box& layoutBox, const HorizontalConstraints& horizontalConstraints) const
273{
274 ASSERT(layoutBox.isOutOfFlowPositioned());
275 // See staticVerticalPositionForOutOfFlowPositioned for the definition of the static position.
276
277 // Start with this box's border box offset from the parent's border box.
278 auto& formattingContext = this->formattingContext();
279 auto left = formattingContext.geometryForBox(layoutBox.parent(), FormattingContext::EscapeReason::OutOfFlowBoxNeedsInFlowGeometry).contentBoxLeft();
280
281 // Resolve left all the way up to the containing block.
282 auto& containingBlock = layoutBox.containingBlock();
283 // Start with the parent since we pretend that this box is normal flow.
284 for (auto* ancestor = &layoutBox.parent(); ancestor != &containingBlock; ancestor = &ancestor->containingBlock()) {
285 auto& boxGeometry = formattingContext.geometryForBox(*ancestor, FormattingContext::EscapeReason::OutOfFlowBoxNeedsInFlowGeometry);
286 // BoxGeometry::left is the border box left position in its containing block's coordinate system.
287 left += BoxGeometry::borderBoxLeft(boxGeometry);
288 ASSERT(!ancestor->isPositioned() || layoutBox.isFixedPositioned());
289 }
290 // Move the static position relative to the padding box. This is very specific to abolutely positioned boxes.
291 return left - horizontalConstraints.logicalLeft;
292}

294LayoutUnit FormattingGeometry::shrinkToFitWidth(const Box& formattingContextRoot, LayoutUnit availableWidth) const
295{
296 LOG_WITH_STREAM(FormattingContextLayout, stream << "[Width] -> shrink to fit -> unsupported -> width(" << LayoutUnit { } << "px) layoutBox: " << &formattingContextRoot << ")");
297 ASSERT(formattingContextRoot.establishesFormattingContext());
298
299 // Calculation of the shrink-to-fit width is similar to calculating the width of a table cell using the automatic table layout algorithm.
300 // Roughly: calculate the preferred width by formatting the content without breaking lines other than where explicit line breaks occur,
301 // and also calculate the preferred minimum width, e.g., by trying all possible line breaks. CSS 2.2 does not define the exact algorithm.
302 // Thirdly, find the available width: in this case, this is the width of the containing block minus the used values of 'margin-left', 'border-left-width',
303 // 'padding-left', 'padding-right', 'border-right-width', 'margin-right', and the widths of any relevant scroll bars.
304
305 // Then the shrink-to-fit width is: min(max(preferred minimum width, available width), preferred width).
306 auto hasContent = is<ContainerBox>(formattingContextRoot) && downcast<ContainerBox>(formattingContextRoot).hasInFlowOrFloatingChild();
307 // The used width of the containment box is determined as if performing a normal layout of the box, except that it is treated as having no content.
308 auto shouldIgnoreContent = formattingContextRoot.isSizeContainmentBox();
309 if (!hasContent || shouldIgnoreContent)
310 return { };
311
312 auto computedIntrinsicWidthConstraints = [&] {
313 auto& layoutState = this->layoutState();
314 auto& root = downcast<ContainerBox>(formattingContextRoot);
315 if (layoutState.hasFormattingState(root)) {
316 if (auto intrinsicWidthConstraints = layoutState.establishedFormattingState(root).intrinsicWidthConstraints())
317 return *intrinsicWidthConstraints;
318 }
319 return LayoutContext::createFormattingContext(root, const_cast<LayoutState&>(layoutState))->computedIntrinsicWidthConstraints();
320 }();
321 return std::min(std::max(computedIntrinsicWidthConstraints.minimum, availableWidth), computedIntrinsicWidthConstraints.maximum);
322}

324VerticalGeometry FormattingGeometry::outOfFlowNonReplacedVerticalGeometry(const ContainerBox& layoutBox, const HorizontalConstraints& horizontalConstraints, const VerticalConstraints& verticalConstraints, const OverriddenVerticalValues& overriddenVerticalValues) const
325{
326 ASSERT(layoutBox.isOutOfFlowPositioned() && !layoutBox.isReplacedBox());
327
328 // 10.6.4 Absolutely positioned, non-replaced elements
329 //
330 // For absolutely positioned elements, the used values of the vertical dimensions must satisfy this constraint:
331 // 'top' + 'margin-top' + 'border-top-width' + 'padding-top' + 'height' + 'padding-bottom' + 'border-bottom-width' + 'margin-bottom' + 'bottom'
332 // = height of containing block
333
334 // If all three of 'top', 'height', and 'bottom' are auto, set 'top' to the static position and apply rule number three below.
335
336 // If none of the three are 'auto': If both 'margin-top' and 'margin-bottom' are 'auto', solve the equation under the extra
337 // constraint that the two margins get equal values. If one of 'margin-top' or 'margin-bottom' is 'auto', solve the equation for that value.
338 // If the values are over-constrained, ignore the value for 'bottom' and solve for that value.
339
340 // Otherwise, pick the one of the following six rules that applies.
341
342 // 1. 'top' and 'height' are 'auto' and 'bottom' is not 'auto', then the height is based on the content per 10.6.7,
343 // set 'auto' values for 'margin-top' and 'margin-bottom' to 0, and solve for 'top'
344 // 2. 'top' and 'bottom' are 'auto' and 'height' is not 'auto', then set 'top' to the static position, set 'auto' values for
345 // 'margin-top' and 'margin-bottom' to 0, and solve for 'bottom'
346 // 3. 'height' and 'bottom' are 'auto' and 'top' is not 'auto', then the height is based on the content per 10.6.7, set 'auto'
347 // values for 'margin-top' and 'margin-bottom' to 0, and solve for 'bottom'
348 // 4. 'top' is 'auto', 'height' and 'bottom' are not 'auto', then set 'auto' values for 'margin-top' and 'margin-bottom' to 0, and solve for 'top'
349 // 5. 'height' is 'auto', 'top' and 'bottom' are not 'auto', then 'auto' values for 'margin-top' and 'margin-bottom' are set to 0 and solve for 'height'
350 // 6. 'bottom' is 'auto', 'top' and 'height' are not 'auto', then set 'auto' values for 'margin-top' and 'margin-bottom' to 0 and solve for 'bottom'
351
352 auto& formattingContext = this->formattingContext();
353 auto& style = layoutBox.style();
354 auto& boxGeometry = formattingContext.geometryForBox(layoutBox);
355 auto containingBlockHeight = verticalConstraints.logicalHeight;
356 auto containingBlockWidth = horizontalConstraints.logicalWidth;
357
358 auto top = computedValue(style.logicalTop(), containingBlockWidth);
359 auto bottom = computedValue(style.logicalBottom(), containingBlockWidth);
360 auto height = overriddenVerticalValues.height ? overriddenVerticalValues.height.value() : computedHeight(layoutBox, containingBlockHeight);
361 auto computedVerticalMargin = FormattingGeometry::computedVerticalMargin(layoutBox, horizontalConstraints);
362 UsedVerticalMargin::NonCollapsedValues usedVerticalMargin;
363 auto paddingTop = boxGeometry.paddingTop().value_or(0);
364 auto paddingBottom = boxGeometry.paddingBottom().value_or(0);
365 auto borderTop = boxGeometry.borderTop();
366 auto borderBottom = boxGeometry.borderBottom();
367
368 if (!top && !height && !bottom)
369 top = staticVerticalPositionForOutOfFlowPositioned(layoutBox, verticalConstraints);
370
371 if (top && height && bottom) {
372 if (!computedVerticalMargin.before && !computedVerticalMargin.after) {
373 auto marginBeforeAndAfter = containingBlockHeight - (*top + borderTop + paddingTop + *height + paddingBottom + borderBottom + *bottom);
374 usedVerticalMargin = { marginBeforeAndAfter / 2, marginBeforeAndAfter / 2 };
375 } else if (!computedVerticalMargin.before) {
376 usedVerticalMargin.after = *computedVerticalMargin.after;
377 usedVerticalMargin.before = containingBlockHeight - (*top + borderTop + paddingTop + *height + paddingBottom + borderBottom + usedVerticalMargin.after + *bottom);
378 } else if (!computedVerticalMargin.after) {
379 usedVerticalMargin.before = *computedVerticalMargin.before;
380 usedVerticalMargin.after = containingBlockHeight - (*top + usedVerticalMargin.before + borderTop + paddingTop + *height + paddingBottom + borderBottom + *bottom);
381 } else
382 usedVerticalMargin = { *computedVerticalMargin.before, *computedVerticalMargin.after };
383 // Over-constrained?
384 auto boxHeight = *top + usedVerticalMargin.before + borderTop + paddingTop + *height + paddingBottom + borderBottom + usedVerticalMargin.after + *bottom;
385 if (boxHeight != containingBlockHeight)
386 bottom = containingBlockHeight - (*top + usedVerticalMargin.before + borderTop + paddingTop + *height + paddingBottom + borderBottom + usedVerticalMargin.after);
387 }
388
389 if (!top && !height && bottom) {
390 // #1
391 height = contentHeightForFormattingContextRoot(layoutBox);
392 usedVerticalMargin = { computedVerticalMargin.before.value_or(0), computedVerticalMargin.after.value_or(0) };
393 top = containingBlockHeight - (usedVerticalMargin.before + borderTop + paddingTop + *height + paddingBottom + borderBottom + usedVerticalMargin.after + *bottom);
394 }
395
396 if (!top && !bottom && height) {
397 // #2
398 top = staticVerticalPositionForOutOfFlowPositioned(layoutBox, verticalConstraints);
399 usedVerticalMargin = { computedVerticalMargin.before.value_or(0), computedVerticalMargin.after.value_or(0) };
400 bottom = containingBlockHeight - (*top + usedVerticalMargin.before + borderTop + paddingTop + *height + paddingBottom + borderBottom + usedVerticalMargin.after);
401 }
402
403 if (!height && !bottom && top) {
404 // #3
405 height = contentHeightForFormattingContextRoot(layoutBox);
406 usedVerticalMargin = { computedVerticalMargin.before.value_or(0), computedVerticalMargin.after.value_or(0) };
407 bottom = containingBlockHeight - (*top + usedVerticalMargin.before + borderTop + paddingTop + *height + paddingBottom + borderBottom + usedVerticalMargin.after);
408 }
409
410 if (!top && height && bottom) {
411 // #4
412 usedVerticalMargin = { computedVerticalMargin.before.value_or(0), computedVerticalMargin.after.value_or(0) };
413 top = containingBlockHeight - (usedVerticalMargin.before + borderTop + paddingTop + *height + paddingBottom + borderBottom + usedVerticalMargin.after + *bottom);
414 }
415
416 if (!height && top && bottom) {
417 // #5
418 usedVerticalMargin = { computedVerticalMargin.before.value_or(0), computedVerticalMargin.after.value_or(0) };
419 height = containingBlockHeight - (*top + usedVerticalMargin.before + borderTop + paddingTop + paddingBottom + borderBottom + usedVerticalMargin.after + *bottom);
420 }
421
422 if (!bottom && top && height) {
423 // #6
424 usedVerticalMargin = { computedVerticalMargin.before.value_or(0), computedVerticalMargin.after.value_or(0) };
425 bottom = containingBlockHeight - (*top + usedVerticalMargin.before + borderTop + paddingTop + *height + paddingBottom + borderBottom + usedVerticalMargin.after);
426 }
427
428 ASSERT(top);
429 ASSERT(bottom);
430 ASSERT(height);
431
432 // For out-of-flow elements the containing block is formed by the padding edge of the ancestor.
433 // At this point the positioned value is in the coordinate system of the padding box. Let's convert it to border box coordinate system.
434 auto containingBlockPaddingVerticalEdge = verticalConstraints.logicalTop;
435 *top += containingBlockPaddingVerticalEdge;
436 *bottom += containingBlockPaddingVerticalEdge;
437
438 LOG_WITH_STREAM(FormattingContextLayout, stream << "[Position][Height][Margin] -> out-of-flow non-replaced -> top(" << *top << "px) bottom(" << *bottom << "px) height(" << *height << "px) margin(" << usedVerticalMargin.before << "px, " << usedVerticalMargin.after << "px) layoutBox(" << &layoutBox << ")");
439 return { *top, *bottom, { *height, usedVerticalMargin } };
440}

442HorizontalGeometry FormattingGeometry::outOfFlowNonReplacedHorizontalGeometry(const ContainerBox& layoutBox, const HorizontalConstraints& horizontalConstraints, const OverriddenHorizontalValues& overriddenHorizontalValues) const
443{
444 ASSERT(layoutBox.isOutOfFlowPositioned() && !layoutBox.isReplacedBox());
445
446 // 10.3.7 Absolutely positioned, non-replaced elements
447 //
448 // 'left' + 'margin-left' + 'border-left-width' + 'padding-left' + 'width' + 'padding-right' + 'border-right-width' + 'margin-right' + 'right'
449 // = width of containing block
450
451 // If all three of 'left', 'width', and 'right' are 'auto': First set any 'auto' values for 'margin-left' and 'margin-right' to 0.
452 // Then, if the 'direction' property of the element establishing the static-position containing block is 'ltr' set 'left' to the static
453 // position and apply rule number three below; otherwise, set 'right' to the static position and apply rule number one below.
454 //
455 // If none of the three is 'auto': If both 'margin-left' and 'margin-right' are 'auto', solve the equation under the extra constraint that the two margins get equal values,
456 // unless this would make them negative, in which case when direction of the containing block is 'ltr' ('rtl'), set 'margin-left' ('margin-right') to zero and
457 // solve for 'margin-right' ('margin-left'). If one of 'margin-left' or 'margin-right' is 'auto', solve the equation for that value.
458 // If the values are over-constrained, ignore the value for 'left' (in case the 'direction' property of the containing block is 'rtl') or 'right'
459 // (in case 'direction' is 'ltr') and solve for that value.
460 //
461 // Otherwise, set 'auto' values for 'margin-left' and 'margin-right' to 0, and pick the one of the following six rules that applies.
462 //
463 // 1. 'left' and 'width' are 'auto' and 'right' is not 'auto', then the width is shrink-to-fit. Then solve for 'left'
464 // 2. 'left' and 'right' are 'auto' and 'width' is not 'auto', then if the 'direction' property of the element establishing the static-position
465 // containing block is 'ltr' set 'left' to the static position, otherwise set 'right' to the static position.
466 // Then solve for 'left' (if 'direction is 'rtl') or 'right' (if 'direction' is 'ltr').
467 // 3. 'width' and 'right' are 'auto' and 'left' is not 'auto', then the width is shrink-to-fit . Then solve for 'right'
468 // 4. 'left' is 'auto', 'width' and 'right' are not 'auto', then solve for 'left'
469 // 5. 'width' is 'auto', 'left' and 'right' are not 'auto', then solve for 'width'
470 // 6. 'right' is 'auto', 'left' and 'width' are not 'auto', then solve for 'right'
471
472 auto& formattingContext = this->formattingContext();
473 auto& style = layoutBox.style();
474 auto& boxGeometry = formattingContext.geometryForBox(layoutBox);
475 auto containingBlockWidth = horizontalConstraints.logicalWidth;
476 auto isLeftToRightDirection = layoutBox.containingBlock().style().isLeftToRightDirection();
477
478 auto left = computedValue(style.logicalLeft(), containingBlockWidth);
479 auto right = computedValue(style.logicalRight(), containingBlockWidth);
480 auto width = overriddenHorizontalValues.width ? overriddenHorizontalValues.width : computedWidth(layoutBox, containingBlockWidth);
481 auto computedHorizontalMargin = FormattingGeometry::computedHorizontalMargin(layoutBox, horizontalConstraints);
482 UsedHorizontalMargin usedHorizontalMargin;
483 auto paddingLeft = boxGeometry.paddingLeft().value_or(0);
484 auto paddingRight = boxGeometry.paddingRight().value_or(0);
485 auto borderLeft = boxGeometry.borderLeft();
486 auto borderRight = boxGeometry.borderRight();
487 if (!left && !width && !right) {
488 // If all three of 'left', 'width', and 'right' are 'auto': First set any 'auto' values for 'margin-left' and 'margin-right' to 0.
489 // Then, if the 'direction' property of the element establishing the static-position containing block is 'ltr' set 'left' to the static
490 // position and apply rule number three below; otherwise, set 'right' to the static position and apply rule number one below.
491 usedHorizontalMargin = { computedHorizontalMargin.start.value_or(0), computedHorizontalMargin.end.value_or(0) };
492
493 auto staticHorizontalPosition = staticHorizontalPositionForOutOfFlowPositioned(layoutBox, horizontalConstraints);
494 if (isLeftToRightDirection)
495 left = staticHorizontalPosition;
496 else
497 right = staticHorizontalPosition;
498 } else if (left && width && right) {
499 // If none of the three is 'auto': If both 'margin-left' and 'margin-right' are 'auto', solve the equation under the extra constraint that the two margins get equal values,
500 // unless this would make them negative, in which case when direction of the containing block is 'ltr' ('rtl'), set 'margin-left' ('margin-right') to zero and
501 // solve for 'margin-right' ('margin-left'). If one of 'margin-left' or 'margin-right' is 'auto', solve the equation for that value.
502 // If the values are over-constrained, ignore the value for 'left' (in case the 'direction' property of the containing block is 'rtl') or 'right'
503 // (in case 'direction' is 'ltr') and solve for that value.
504 if (!computedHorizontalMargin.start && !computedHorizontalMargin.end) {
505 auto marginStartAndEnd = containingBlockWidth - (*left + borderLeft + paddingLeft + *width + paddingRight + borderRight + *right);
506 if (marginStartAndEnd >= 0)
507 usedHorizontalMargin = { marginStartAndEnd / 2, marginStartAndEnd / 2 };
508 else {
509 if (isLeftToRightDirection) {
510 usedHorizontalMargin.start = 0_lu;
511 usedHorizontalMargin.end = containingBlockWidth - (*left + usedHorizontalMargin.start + borderLeft + paddingLeft + *width + paddingRight + borderRight + *right);
512 } else {
513 usedHorizontalMargin.end = 0_lu;
514 usedHorizontalMargin.start = containingBlockWidth - (*left + borderLeft + paddingLeft + *width + paddingRight + borderRight + usedHorizontalMargin.end + *right);
515 }
516 }
517 } else if (!computedHorizontalMargin.start) {
518 usedHorizontalMargin.end = *computedHorizontalMargin.end;
519 usedHorizontalMargin.start = containingBlockWidth - (*left + borderLeft + paddingLeft + *width + paddingRight + borderRight + usedHorizontalMargin.end + *right);
520 } else if (!computedHorizontalMargin.end) {
521 usedHorizontalMargin.start = *computedHorizontalMargin.start;
522 usedHorizontalMargin.end = containingBlockWidth - (*left + usedHorizontalMargin.start + borderLeft + paddingLeft + *width + paddingRight + borderRight + *right);
523 } else {
524 usedHorizontalMargin = { *computedHorizontalMargin.start, *computedHorizontalMargin.end };
525 // Overconstrained? Ignore right (left).
526 if (isLeftToRightDirection)
527 right = containingBlockWidth - (usedHorizontalMargin.start + *left + borderLeft + paddingLeft + *width + paddingRight + borderRight + usedHorizontalMargin.end);
528 else
529 left = containingBlockWidth - (usedHorizontalMargin.start + borderLeft + paddingLeft + *width + paddingRight + borderRight + usedHorizontalMargin.end + *right);
530 }
531 } else {
532 // Otherwise, set 'auto' values for 'margin-left' and 'margin-right' to 0, and pick the one of the following six rules that applies.
533 usedHorizontalMargin = { computedHorizontalMargin.start.value_or(0), computedHorizontalMargin.end.value_or(0) };
534 }
535
536 if (!left && !width && right) {
537 // #1
538 // Calculate the available width by solving for 'width' after setting 'left' (in case 1) to 0
539 left = LayoutUnit { 0 };
540 auto availableWidth = containingBlockWidth - (*left + usedHorizontalMargin.start + borderLeft + paddingLeft + paddingRight + borderRight + usedHorizontalMargin.end + *right);
541 width = shrinkToFitWidth(layoutBox, availableWidth);
542 left = containingBlockWidth - (usedHorizontalMargin.start + borderLeft + paddingLeft + *width + paddingRight + borderRight + usedHorizontalMargin.end + *right);
543 } else if (!left && !right && width) {
544 // #2
545 auto staticHorizontalPosition = staticHorizontalPositionForOutOfFlowPositioned(layoutBox, horizontalConstraints);
546 if (isLeftToRightDirection) {
547 left = staticHorizontalPosition;
548 right = containingBlockWidth - (*left + usedHorizontalMargin.start + borderLeft + paddingLeft + *width + paddingRight + borderRight + usedHorizontalMargin.end);
549 } else {
550 right = staticHorizontalPosition;
551 left = containingBlockWidth - (usedHorizontalMargin.start + borderLeft + paddingLeft + *width + paddingRight + borderRight + usedHorizontalMargin.end + *right);
552 }
553 } else if (!width && !right && left) {
554 // #3
555 // Calculate the available width by solving for 'width' after setting 'right' (in case 3) to 0
556 right = LayoutUnit { 0 };
557 auto availableWidth = containingBlockWidth - (*left + usedHorizontalMargin.start + borderLeft + paddingLeft + paddingRight + borderRight + usedHorizontalMargin.end + *right);
558 width = shrinkToFitWidth(layoutBox, availableWidth);
559 right = containingBlockWidth - (*left + usedHorizontalMargin.start + borderLeft + paddingLeft + *width + paddingRight + borderRight + usedHorizontalMargin.end);
560 } else if (!left && width && right) {
561 // #4
562 left = containingBlockWidth - (usedHorizontalMargin.start + borderLeft + paddingLeft + *width + paddingRight + borderRight + usedHorizontalMargin.end + *right);
563 } else if (!width && left && right) {
564 // #5
565 width = containingBlockWidth - (*left + usedHorizontalMargin.start + borderLeft + paddingLeft + paddingRight + borderRight + usedHorizontalMargin.end + *right);
566 } else if (!right && left && width) {
567 // #6
568 right = containingBlockWidth - (*left + usedHorizontalMargin.start + borderLeft + paddingLeft + *width + paddingRight + borderRight + usedHorizontalMargin.end);
569 }
570
571 ASSERT(left);
572 ASSERT(right);
573 ASSERT(width);
574
575 // For out-of-flow elements the containing block is formed by the padding edge of the ancestor.
576 // At this point the positioned value is in the coordinate system of the padding box. Let's convert it to border box coordinate system.
577 auto containingBlockPaddingVerticalEdge = horizontalConstraints.logicalLeft;
578 *left += containingBlockPaddingVerticalEdge;
579 *right += containingBlockPaddingVerticalEdge;
580
581 LOG_WITH_STREAM(FormattingContextLayout, stream << "[Position][Width][Margin] -> out-of-flow non-replaced -> left(" << *left << "px) right(" << *right << "px) width(" << *width << "px) margin(" << usedHorizontalMargin.start << "px, " << usedHorizontalMargin.end << "px) layoutBox(" << &layoutBox << ")");
582 return { *left, *right, { *width, usedHorizontalMargin } };
583}

585VerticalGeometry FormattingGeometry::outOfFlowReplacedVerticalGeometry(const ReplacedBox& replacedBox, const HorizontalConstraints& horizontalConstraints, const VerticalConstraints& verticalConstraints, const OverriddenVerticalValues& overriddenVerticalValues) const
586{
587 ASSERT(replacedBox.isOutOfFlowPositioned());
588
589 // 10.6.5 Absolutely positioned, replaced elements
590 //
591 // The used value of 'height' is determined as for inline replaced elements.
592 // If 'margin-top' or 'margin-bottom' is specified as 'auto' its used value is determined by the rules below.
593 // 1. If both 'top' and 'bottom' have the value 'auto', replace 'top' with the element's static position.
594 // 2. If 'bottom' is 'auto', replace any 'auto' on 'margin-top' or 'margin-bottom' with '0'.
595 // 3. If at this point both 'margin-top' and 'margin-bottom' are still 'auto', solve the equation under the extra constraint that the two margins must get equal values.
596 // 4. If at this point there is only one 'auto' left, solve the equation for that value.
597 // 5. If at this point the values are over-constrained, ignore the value for 'bottom' and solve for that value.
598
599 auto& formattingContext = this->formattingContext();
600 auto& style = replacedBox.style();
601 auto& boxGeometry = formattingContext.geometryForBox(replacedBox);
602 auto containingBlockHeight = verticalConstraints.logicalHeight;
603 auto containingBlockWidth = horizontalConstraints.logicalWidth;
604
605 auto top = computedValue(style.logicalTop(), containingBlockWidth);
606 auto bottom = computedValue(style.logicalBottom(), containingBlockWidth);
607 auto height = inlineReplacedContentHeightAndMargin(replacedBox, horizontalConstraints, verticalConstraints, overriddenVerticalValues).contentHeight;
608 auto computedVerticalMargin = FormattingGeometry::computedVerticalMargin(replacedBox, horizontalConstraints);
609 std::optional<LayoutUnit> usedMarginBefore = computedVerticalMargin.before;
610 std::optional<LayoutUnit> usedMarginAfter = computedVerticalMargin.after;
611 auto paddingTop = boxGeometry.paddingTop().value_or(0);
612 auto paddingBottom = boxGeometry.paddingBottom().value_or(0);
613 auto borderTop = boxGeometry.borderTop();
614 auto borderBottom = boxGeometry.borderBottom();
615
616 if (!top && !bottom) {
617 // #1
618 top = staticVerticalPositionForOutOfFlowPositioned(replacedBox, verticalConstraints);
619 }
620
621 if (!bottom) {
622 // #2
623 usedMarginBefore = computedVerticalMargin.before.value_or(0);
624 usedMarginAfter = usedMarginBefore;
625 }
626
627 if (!usedMarginBefore && !usedMarginAfter) {
628 // #3
629 auto marginBeforeAndAfter = containingBlockHeight - (*top + borderTop + paddingTop + height + paddingBottom + borderBottom + *bottom);
630 usedMarginBefore = marginBeforeAndAfter / 2;
631 usedMarginAfter = usedMarginBefore;
632 }
633
634 // #4
635 if (!top)
636 top = containingBlockHeight - (*usedMarginBefore + borderTop + paddingTop + height + paddingBottom + borderBottom + *usedMarginAfter + *bottom);
637
638 if (!bottom)
639 bottom = containingBlockHeight - (*top + *usedMarginBefore + borderTop + paddingTop + height + paddingBottom + borderBottom + *usedMarginAfter);
640
641 if (!usedMarginBefore)
642 usedMarginBefore = containingBlockHeight - (*top + borderTop + paddingTop + height + paddingBottom + borderBottom + *usedMarginAfter + *bottom);
643
644 if (!usedMarginAfter)
645 usedMarginAfter = containingBlockHeight - (*top + *usedMarginBefore + borderTop + paddingTop + height + paddingBottom + borderBottom + *bottom);
646
647 // #5
648 auto boxHeight = *top + *usedMarginBefore + borderTop + paddingTop + height + paddingBottom + borderBottom + *usedMarginAfter + *bottom;
649 if (boxHeight > containingBlockHeight)
650 bottom = containingBlockHeight - (*top + *usedMarginBefore + borderTop + paddingTop + height + paddingBottom + borderBottom + *usedMarginAfter);
651
652 // For out-of-flow elements the containing block is formed by the padding edge of the ancestor.
653 // At this point the positioned value is in the coordinate system of the padding box. Let's convert it to border box coordinate system.
654 auto containingBlockPaddingVerticalEdge = verticalConstraints.logicalTop;
655 *top += containingBlockPaddingVerticalEdge;
656 *bottom += containingBlockPaddingVerticalEdge;
657
658 ASSERT(top);
659 ASSERT(bottom);
660 ASSERT(usedMarginBefore);
661 ASSERT(usedMarginAfter);
662 LOG_WITH_STREAM(FormattingContextLayout, stream << "[Position][Height][Margin] -> out-of-flow replaced -> top(" << *top << "px) bottom(" << *bottom << "px) height(" << height << "px) margin(" << *usedMarginBefore << "px, " << *usedMarginAfter << "px) layoutBox(" << &replacedBox << ")");
663 return { *top, *bottom, { height, { *usedMarginBefore, *usedMarginAfter } } };
664}

666HorizontalGeometry FormattingGeometry::outOfFlowReplacedHorizontalGeometry(const ReplacedBox& replacedBox, const HorizontalConstraints& horizontalConstraints, const VerticalConstraints& verticalConstraints, const OverriddenHorizontalValues& overriddenHorizontalValues) const
667{
668 ASSERT(replacedBox.isOutOfFlowPositioned());
669
670 // 10.3.8 Absolutely positioned, replaced elements
671 // In this case, section 10.3.7 applies up through and including the constraint equation, but the rest of section 10.3.7 is replaced by the following rules:
672 //
673 // The used value of 'width' is determined as for inline replaced elements. If 'margin-left' or 'margin-right' is specified as 'auto' its used value is determined by the rules below.
674 // 1. If both 'left' and 'right' have the value 'auto', then if the 'direction' property of the element establishing the static-position containing block is 'ltr',
675 // set 'left' to the static position; else if 'direction' is 'rtl', set 'right' to the static position.
676 // 2. If 'left' or 'right' are 'auto', replace any 'auto' on 'margin-left' or 'margin-right' with '0'.
677 // 3. If at this point both 'margin-left' and 'margin-right' are still 'auto', solve the equation under the extra constraint that the two margins must get equal values,
678 // unless this would make them negative, in which case when the direction of the containing block is 'ltr' ('rtl'), set 'margin-left' ('margin-right') to zero and
679 // solve for 'margin-right' ('margin-left').
680 // 4. If at this point there is an 'auto' left, solve the equation for that value.
681 // 5. If at this point the values are over-constrained, ignore the value for either 'left' (in case the 'direction' property of the containing block is 'rtl') or
682 // 'right' (in case 'direction' is 'ltr') and solve for that value.
683
684 auto& formattingContext = this->formattingContext();
685 auto& style = replacedBox.style();
686 auto& boxGeometry = formattingContext.geometryForBox(replacedBox);
687 auto containingBlockWidth = horizontalConstraints.logicalWidth;
688 auto isLeftToRightDirection = replacedBox.containingBlock().style().isLeftToRightDirection();
689
690 auto left = computedValue(style.logicalLeft(), containingBlockWidth);
691 auto right = computedValue(style.logicalRight(), containingBlockWidth);
692 auto computedHorizontalMargin = FormattingGeometry::computedHorizontalMargin(replacedBox, horizontalConstraints);
693 std::optional<LayoutUnit> usedMarginStart = computedHorizontalMargin.start;
694 std::optional<LayoutUnit> usedMarginEnd = computedHorizontalMargin.end;
695 auto width = inlineReplacedContentWidthAndMargin(replacedBox, horizontalConstraints, verticalConstraints, overriddenHorizontalValues).contentWidth;
696 auto paddingLeft = boxGeometry.paddingLeft().value_or(0);
697 auto paddingRight = boxGeometry.paddingRight().value_or(0);
698 auto borderLeft = boxGeometry.borderLeft();
699 auto borderRight = boxGeometry.borderRight();
700
701 if (!left && !right) {
702 // #1
703 auto staticHorizontalPosition = staticHorizontalPositionForOutOfFlowPositioned(replacedBox, horizontalConstraints);
704 if (isLeftToRightDirection)
705 left = staticHorizontalPosition;
706 else
707 right = staticHorizontalPosition;
708 }
709
710 if (!left || !right) {
711 // #2
712 usedMarginStart = computedHorizontalMargin.start.value_or(0);
713 usedMarginEnd = computedHorizontalMargin.end.value_or(0);
714 }
715
716 if (!usedMarginStart && !usedMarginEnd) {
717 // #3
718 auto marginStartAndEnd = containingBlockWidth - (*left + borderLeft + paddingLeft + width + paddingRight + borderRight + *right);
719 if (marginStartAndEnd >= 0) {
720 usedMarginStart = marginStartAndEnd / 2;
721 usedMarginEnd = usedMarginStart;
722 } else {
723 if (isLeftToRightDirection) {
724 usedMarginStart = 0_lu;
725 usedMarginEnd = containingBlockWidth - (*left + *usedMarginStart + borderLeft + paddingLeft + width + paddingRight + borderRight + *right);
726 } else {
727 usedMarginEnd = 0_lu;
728 usedMarginStart = containingBlockWidth - (*left + borderLeft + paddingLeft + width + paddingRight + borderRight + *usedMarginEnd + *right);
729 }
730 }
731 }
732
733 // #4
734 if (!left)
735 left = containingBlockWidth - (*usedMarginStart + borderLeft + paddingLeft + width + paddingRight + borderRight + *usedMarginEnd + *right);
736
737 if (!right)
738 right = containingBlockWidth - (*left + *usedMarginStart + borderLeft + paddingLeft + width + paddingRight + borderRight + *usedMarginEnd);
739
740 if (!usedMarginStart)
741 usedMarginStart = containingBlockWidth - (*left + borderLeft + paddingLeft + width + paddingRight + borderRight + *usedMarginEnd + *right);
742
743 if (!usedMarginEnd)
744 usedMarginEnd = containingBlockWidth - (*left + *usedMarginStart + borderLeft + paddingLeft + width + paddingRight + borderRight + *right);
745
746 auto boxWidth = (*left + *usedMarginStart + borderLeft + paddingLeft + width + paddingRight + borderRight + *usedMarginEnd + *right);
747 if (boxWidth > containingBlockWidth) {
748 // #5 Over-constrained?
749 if (isLeftToRightDirection)
750 right = containingBlockWidth - (*left + *usedMarginStart + borderLeft + paddingLeft + width + paddingRight + borderRight + *usedMarginEnd);
751 else
752 left = containingBlockWidth - (*usedMarginStart + borderLeft + paddingLeft + width + paddingRight + borderRight + *usedMarginEnd + *right);
753 }
754
755 ASSERT(left);
756 ASSERT(right);
757 ASSERT(usedMarginStart);
758 ASSERT(usedMarginEnd);
759
760 // For out-of-flow elements the containing block is formed by the padding edge of the ancestor.
761 // At this point the positioned value is in the coordinate system of the padding box. Let's convert it to border box coordinate system.
762 auto containingBlockPaddingVerticalEdge = horizontalConstraints.logicalLeft;
763 *left += containingBlockPaddingVerticalEdge;
764 *right += containingBlockPaddingVerticalEdge;
765
766 LOG_WITH_STREAM(FormattingContextLayout, stream << "[Position][Width][Margin] -> out-of-flow replaced -> left(" << *left << "px) right(" << *right << "px) width(" << width << "px) margin(" << *usedMarginStart << "px, " << *usedMarginEnd << "px) layoutBox(" << &replacedBox << ")");
767 return { *left, *right, { width, { *usedMarginStart, *usedMarginEnd } } };
768}

770ContentHeightAndMargin FormattingGeometry::complicatedCases(const Box& layoutBox, const HorizontalConstraints& horizontalConstraints, const OverriddenVerticalValues& overriddenVerticalValues) const
771{
772 ASSERT(!layoutBox.isReplacedBox());
773 // TODO: Use complicated-case for document renderer for now (see BlockFormattingGeometry::inFlowHeightAndMargin).
774 ASSERT((layoutBox.isBlockLevelBox() && layoutBox.isInFlow() && !layoutBox.isOverflowVisible()) || layoutBox.isInlineBlockBox() || layoutBox.isFloatingPositioned() || layoutBox.isDocumentBox() || layoutBox.isTableBox());
775
776 // 10.6.6 Complicated cases
777 //
778 // Block-level, non-replaced elements in normal flow when 'overflow' does not compute to 'visible' (except if the 'overflow' property's value has been propagated to the viewport).
779 // 'Inline-block', non-replaced elements.
780 // Floating, non-replaced elements.
781 //
782 // 1. If 'margin-top', or 'margin-bottom' are 'auto', their used value is 0.
783 // 2. If 'height' is 'auto', the height depends on the element's descendants per 10.6.7.
784
785 auto height = overriddenVerticalValues.height ? overriddenVerticalValues.height.value() : computedHeight(layoutBox);
786 auto computedVerticalMargin = FormattingGeometry::computedVerticalMargin(layoutBox, horizontalConstraints);
787 // #1
788 auto usedVerticalMargin = UsedVerticalMargin::NonCollapsedValues { computedVerticalMargin.before.value_or(0), computedVerticalMargin.after.value_or(0) };
789 // #2
790 if (!height) {
791 ASSERT(isHeightAuto(layoutBox));
792 if (!is<ContainerBox>(layoutBox) || !downcast<ContainerBox>(layoutBox).hasInFlowOrFloatingChild())
793 height = 0_lu;
794 else if (layoutBox.isDocumentBox() && !layoutBox.establishesFormattingContext()) {
795 auto& documentBox = downcast<ContainerBox>(layoutBox);
796 auto top = BoxGeometry::marginBoxRect(formattingContext().geometryForBox(*documentBox.firstInFlowChild())).top();
797 auto bottom = BoxGeometry::marginBoxRect(formattingContext().geometryForBox(*documentBox.lastInFlowChild())).bottom();
798 // This is a special (quirk?) behavior since the document box is not a formatting context root and
799 // all the float boxes end up at the ICB level.
800 auto& initialContainingBlock = documentBox.formattingContextRoot();
801 auto floatingContext = FloatingContext { formattingContext(), layoutState().establishedFormattingState(initialContainingBlock).floatingState() };
802 if (auto floatBottom = floatingContext.bottom()) {
803 bottom = std::max<LayoutUnit>(*floatBottom, bottom);
804 auto floatTop = floatingContext.top();
805 ASSERT(floatTop);
806 top = std::min<LayoutUnit>(*floatTop, top);
807 }
808 height = bottom - top;
809 } else {
810 ASSERT(layoutBox.establishesFormattingContext());
811 height = contentHeightForFormattingContextRoot(downcast<ContainerBox>(layoutBox));
812 }
813 }
814
815 ASSERT(height);
816
817 LOG_WITH_STREAM(FormattingContextLayout, stream << "[Height][Margin] -> floating non-replaced -> height(" << *height << "px) margin(" << usedVerticalMargin.before << "px, " << usedVerticalMargin.after << "px) -> layoutBox(" << &layoutBox << ")");
818 return ContentHeightAndMargin { *height, usedVerticalMargin };

904ContentHeightAndMargin FormattingGeometry::inlineReplacedContentHeightAndMargin(const ReplacedBox& replacedBox, const HorizontalConstraints& horizontalConstraints, std::optional<VerticalConstraints> verticalConstraints, const OverriddenVerticalValues& overriddenVerticalValues) const
905{
906 // 10.6.2 Inline replaced elements, block-level replaced elements in normal flow, 'inline-block' replaced elements in normal flow and floating replaced elements
907 //
908 // 1. If 'margin-top', or 'margin-bottom' are 'auto', their used value is 0.
909 // 2. If 'height' and 'width' both have computed values of 'auto' and the element also has an intrinsic height, then that intrinsic height is the used value of 'height'.
910 // 3. Otherwise, if 'height' has a computed value of 'auto', and the element has an intrinsic ratio then the used value of 'height' is:
911 // (used width) / (intrinsic ratio)
912 // 4. Otherwise, if 'height' has a computed value of 'auto', and the element has an intrinsic height, then that intrinsic height is the used value of 'height'.
913 // 5. Otherwise, if 'height' has a computed value of 'auto', but none of the conditions above are met, then the used value of 'height' must be set to
914 // the height of the largest rectangle that has a 2:1 ratio, has a height not greater than 150px, and has a width not greater than the device width.
915
916 // #1
917 auto& formattingContext = this->formattingContext();
918 auto computedVerticalMargin = FormattingGeometry::computedVerticalMargin(replacedBox, horizontalConstraints);
919 auto usedVerticalMargin = UsedVerticalMargin::NonCollapsedValues { computedVerticalMargin.before.value_or(0), computedVerticalMargin.after.value_or(0) };
920 auto& style = replacedBox.style();
921
922 auto height = overriddenVerticalValues.height ? overriddenVerticalValues.height.value() : computedHeight(replacedBox, verticalConstraints ? std::optional<LayoutUnit>(verticalConstraints->logicalHeight) : std::nullopt);
923 auto heightIsAuto = !overriddenVerticalValues.height && isHeightAuto(replacedBox);
924 auto widthIsAuto = style.logicalWidth().isAuto();
925
926 if (heightIsAuto && widthIsAuto && replacedBox.hasIntrinsicHeight()) {
927 // #2
928 height = replacedBox.intrinsicHeight();
929 } else if (heightIsAuto && replacedBox.hasIntrinsicRatio()) {
930 // #3
931 auto usedWidth = formattingContext.geometryForBox(replacedBox).contentBoxWidth();
932 height = usedWidth / replacedBox.intrinsicRatio();
933 } else if (heightIsAuto && replacedBox.hasIntrinsicHeight()) {
934 // #4
935 height = replacedBox.intrinsicHeight();
936 } else if (heightIsAuto) {
937 // #5
938 height = { 150 };
939 }
940
941 ASSERT(height);
942
943 LOG_WITH_STREAM(FormattingContextLayout, stream << "[Height][Margin] -> inflow replaced -> height(" << *height << "px) margin(" << usedVerticalMargin.before << "px, " << usedVerticalMargin.after << "px) -> layoutBox(" << &replacedBox << ")");
944 return { *height, usedVerticalMargin };

947ContentWidthAndMargin FormattingGeometry::inlineReplacedContentWidthAndMargin(const ReplacedBox& replacedBox, const HorizontalConstraints& horizontalConstraints, std::optional<VerticalConstraints> verticalConstraints, const OverriddenHorizontalValues& overriddenHorizontalValues) const
948{
949 // 10.3.2 Inline, replaced elements
950 //
951 // A computed value of 'auto' for 'margin-left' or 'margin-right' becomes a used value of '0'.
952 //
953 // 1. If 'height' and 'width' both have computed values of 'auto' and the element also has an intrinsic width, then that intrinsic width is the used value of 'width'.
954 //
955 // 2. If 'height' and 'width' both have computed values of 'auto' and the element has no intrinsic width, but does have an intrinsic height and intrinsic ratio;
956 // or if 'width' has a computed value of 'auto', 'height' has some other computed value, and the element does have an intrinsic ratio;
957 // then the used value of 'width' is: (used height) * (intrinsic ratio)
958 //
959 // 3. If 'height' and 'width' both have computed values of 'auto' and the element has an intrinsic ratio but no intrinsic height or width,
960 // then the used value of 'width' is undefined in CSS 2.2. However, it is suggested that, if the containing block's width does not itself depend on the replaced
961 // element's width, then the used value of 'width' is calculated from the constraint equation used for block-level, non-replaced elements in normal flow.
962 //
963 // 4. Otherwise, if 'width' has a computed value of 'auto', and the element has an intrinsic width, then that intrinsic width is the used value of 'width'.
964 //
965 // 5. Otherwise, if 'width' has a computed value of 'auto', but none of the conditions above are met, then the used value of 'width' becomes 300px.
966 // If 300px is too wide to fit the device, UAs should use the width of the largest rectangle that has a 2:1 ratio and fits the device instead.
967
968 auto computedHorizontalMargin = FormattingGeometry::computedHorizontalMargin(replacedBox, horizontalConstraints);
969
970 auto usedMarginStart = [&] {
971 if (overriddenHorizontalValues.margin)
972 return overriddenHorizontalValues.margin->start;
973 return computedHorizontalMargin.start.value_or(0_lu);
974 };
975
976 auto usedMarginEnd = [&] {
977 if (overriddenHorizontalValues.margin)
978 return overriddenHorizontalValues.margin->end;
979 return computedHorizontalMargin.end.value_or(0_lu);
980 };
981
982 auto width = overriddenHorizontalValues.width ? overriddenHorizontalValues.width : computedWidth(replacedBox, horizontalConstraints.logicalWidth);
983 auto heightIsAuto = isHeightAuto(replacedBox);
984 auto height = computedHeight(replacedBox, verticalConstraints ? std::optional<LayoutUnit>(verticalConstraints->logicalHeight) : std::nullopt);
985
986 if (!width && heightIsAuto && replacedBox.hasIntrinsicWidth()) {
987 // #1
988 width = replacedBox.intrinsicWidth();
989 } else if ((!width && heightIsAuto && !replacedBox.hasIntrinsicWidth() && replacedBox.hasIntrinsicHeight() && replacedBox.hasIntrinsicRatio())
990 || (!width && height && replacedBox.hasIntrinsicRatio())) {
991 // #2
992 width = height.value_or(replacedBox.hasIntrinsicHeight()) * replacedBox.intrinsicRatio();
993 } else if (!width && heightIsAuto && replacedBox.hasIntrinsicRatio() && !replacedBox.hasIntrinsicWidth() && !replacedBox.hasIntrinsicHeight()) {
994 // #3
995 // FIXME: undefined but surely doable.
996 ASSERT_NOT_IMPLEMENTED_YET();
997 } else if (!width && replacedBox.hasIntrinsicWidth()) {
998 // #4
999 width = replacedBox.intrinsicWidth();
1000 } else if (!width) {
1001 // #5
1002 width = { 300 };
1003 }
1004
1005 ASSERT(width);
1006
1007 LOG_WITH_STREAM(FormattingContextLayout, stream << "[Width][Margin] -> inflow replaced -> width(" << *width << "px) margin(" << usedMarginStart() << "px, " << usedMarginEnd() << "px) -> layoutBox(" << &replacedBox << ")");
1008 return { *width, { usedMarginStart(), usedMarginEnd() } };
1009}

1011LayoutSize FormattingGeometry::inFlowPositionedPositionOffset(const Box& layoutBox, const HorizontalConstraints& horizontalConstraints) const
1012{
1013 ASSERT(layoutBox.isInFlowPositioned());
1014
1015 // 9.4.3 Relative positioning
1016 //
1017 // The 'top' and 'bottom' properties move relatively positioned element(s) up or down without changing their size.
1018 // Top' moves the boxes down, and 'bottom' moves them up. Since boxes are not split or stretched as a result of 'top' or 'bottom', the used values are always: top = -bottom.
1019 //
1020 // 1. If both are 'auto', their used values are both '0'.
1021 // 2. If one of them is 'auto', it becomes the negative of the other.
1022 // 3. If neither is 'auto', 'bottom' is ignored (i.e., the used value of 'bottom' will be minus the value of 'top').
1023
1024 auto& style = layoutBox.style();
1025 auto containingBlockWidth = horizontalConstraints.logicalWidth;
1026
1027 auto top = computedValue(style.logicalTop(), containingBlockWidth);
1028 auto bottom = computedValue(style.logicalBottom(), containingBlockWidth);
1029
1030 if (!top && !bottom) {
1031 // #1
1032 top = bottom = { 0 };
1033 } else if (!top) {
1034 // #2
1035 top = -*bottom;
1036 } else if (!bottom) {
1037 // #3
1038 bottom = -*top;
1039 } else {
1040 // #4
1041 bottom = std::nullopt;
1042 }
1043
1044 // For relatively positioned elements, 'left' and 'right' move the box(es) horizontally, without changing their size.
1045 // 'Left' moves the boxes to the right, and 'right' moves them to the left.
1046 // Since boxes are not split or stretched as a result of 'left' or 'right', the used values are always: left = -right.
1047 //
1048 // 1. If both 'left' and 'right' are 'auto' (their initial values), the used values are '0' (i.e., the boxes stay in their original position).
1049 // 2. If 'left' is 'auto', its used value is minus the value of 'right' (i.e., the boxes move to the left by the value of 'right').
1050 // 3. If 'right' is specified as 'auto', its used value is minus the value of 'left'.
1051 // 4. If neither 'left' nor 'right' is 'auto', the position is over-constrained, and one of them has to be ignored.
1052 // If the 'direction' property of the containing block is 'ltr', the value of 'left' wins and 'right' becomes -'left'.
1053 // If 'direction' of the containing block is 'rtl', 'right' wins and 'left' is ignored.
1054
1055 auto left = computedValue(style.logicalLeft(), containingBlockWidth);
1056 auto right = computedValue(style.logicalRight(), containingBlockWidth);
1057
1058 if (!left && !right) {
1059 // #1
1060 left = right = { 0 };
1061 } else if (!left) {
1062 // #2
1063 left = -*right;
1064 } else if (!right) {
1065 // #3
1066 right = -*left;
1067 } else {
1068 // #4
1069 auto isLeftToRightDirection = layoutBox.containingBlock().style().isLeftToRightDirection();
1070 if (isLeftToRightDirection)
1071 right = -*left;
1072 else
1073 left = std::nullopt;
1074 }
1075
1076 ASSERT(!bottom || *top == -*bottom);
1077 ASSERT(!left || *left == -*right);
1078
1079 auto topPositionOffset = *top;
1080 auto leftPositionOffset = left.value_or(-*right);
1081
1082 LOG_WITH_STREAM(FormattingContextLayout, stream << "[Position] -> positioned inflow -> top offset(" << topPositionOffset << "px) left offset(" << leftPositionOffset << "px) layoutBox(" << &layoutBox << ")");
1083 return { leftPositionOffset, topPositionOffset };
1084}
