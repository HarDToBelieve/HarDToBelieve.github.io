43class Box;

47// This class implements the layout logic for block formatting contexts.
48// https://www.w3.org/TR/CSS22/visuren.html#block-formatting
49class BlockFormattingContext : public FormattingContext {
50 WTF_MAKE_ISO_ALLOCATED(BlockFormattingContext);
51public:
52 BlockFormattingContext(const ContainerBox& formattingContextRoot, BlockFormattingState&);
53
54 void layoutInFlowContent(InvalidationState&, const ConstraintsForInFlowContent&) override;
55 LayoutUnit usedContentHeight() const override;
56
57 const BlockFormattingState& formattingState() const { return downcast<BlockFormattingState>(FormattingContext::formattingState()); }
58 const BlockFormattingGeometry& formattingGeometry() const final { return m_blockFormattingGeometry; }
59 const BlockFormattingQuirks& formattingQuirks() const override { return m_blockFormattingQuirks; }
60
61protected:
62 struct ConstraintsPair {
63 ConstraintsForInFlowContent formattingContextRoot;
64 ConstraintsForInFlowContent containingBlock;
65 };
66 void placeInFlowPositionedChildren(const ContainerBox&, const HorizontalConstraints&);
67
68 void computeWidthAndMargin(const FloatingContext&, const Box&, const ConstraintsPair&);
69 void computeHeightAndMargin(const Box&, const ConstraintsForInFlowContent&);
70
71 void computeStaticHorizontalPosition(const Box&, const HorizontalConstraints&);
72 void computeStaticVerticalPosition(const Box&, LayoutUnit containingBlockContentBoxTop);
73 void computePositionToAvoidFloats(const FloatingContext&, const Box&, const ConstraintsPair&);
74 void computeVerticalPositionForFloatClear(const FloatingContext&, const Box&);
75
76 void precomputeVerticalPositionForBoxAndAncestors(const Box&, const ConstraintsPair&);
77
78 IntrinsicWidthConstraints computedIntrinsicWidthConstraints() override;
79
80 LayoutUnit verticalPositionWithMargin(const Box&, const UsedVerticalMargin&, LayoutUnit containingBlockContentBoxTop) const;
81
82 std::optional<LayoutUnit> usedAvailableWidthForFloatAvoider(const FloatingContext&, const Box&, const ConstraintsPair&);
83 void updateMarginAfterForPreviousSibling(const Box&);
84
85 BlockFormattingState& formattingState() { return downcast<BlockFormattingState>(FormattingContext::formattingState()); }
86 BlockMarginCollapse marginCollapse() const;
87
88#if ASSERT_ENABLED
89 void setPrecomputedMarginBefore(const Box& layoutBox, const PrecomputedMarginBefore& precomputedMarginBefore) { m_precomputedMarginBeforeList.set(&layoutBox, precomputedMarginBefore); }
90 PrecomputedMarginBefore precomputedMarginBefore(const Box& layoutBox) const { return m_precomputedMarginBeforeList.get(&layoutBox); }
91 bool hasPrecomputedMarginBefore(const Box& layoutBox) const { return m_precomputedMarginBeforeList.contains(&layoutBox); }
92#endif
93
94private:
95 HashMap<const Box*, PrecomputedMarginBefore> m_precomputedMarginBeforeList;
96 const BlockFormattingGeometry m_blockFormattingGeometry;
97 const BlockFormattingQuirks m_blockFormattingQuirks;
98};
99
100}
