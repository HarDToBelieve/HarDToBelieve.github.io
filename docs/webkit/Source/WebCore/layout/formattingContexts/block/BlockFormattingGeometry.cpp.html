51ContentHeightAndMargin BlockFormattingGeometry::inFlowNonReplacedContentHeightAndMargin(const Box& layoutBox, const HorizontalConstraints& horizontalConstraints, const OverriddenVerticalValues& overriddenVerticalValues) const
52{
53 ASSERT(layoutBox.isInFlow() && !layoutBox.isReplacedBox());
54 ASSERT(layoutBox.isOverflowVisible());
55
56 auto compute = [&](const auto& overriddenVerticalValues) -> ContentHeightAndMargin {
57
58 // 10.6.3 Block-level non-replaced elements in normal flow when 'overflow' computes to 'visible'
59 //
60 // If 'margin-top', or 'margin-bottom' are 'auto', their used value is 0.
61 // If 'height' is 'auto', the height depends on whether the element has any block-level children and whether it has padding or borders:
62 // The element's height is the distance from its top content edge to the first applicable of the following:
63 // 1. the bottom edge of the last line box, if the box establishes a inline formatting context with one or more lines
64 // 2. the bottom edge of the bottom (possibly collapsed) margin of its last in-flow child, if the child's bottom margin
65 // does not collapse with the element's bottom margin
66 // 3. the bottom border edge of the last in-flow child whose top margin doesn't collapse with the element's bottom margin
67 // 4. zero, otherwise
68 // Only children in the normal flow are taken into account (i.e., floating boxes and absolutely positioned boxes are ignored,
69 // and relatively positioned boxes are considered without their offset). Note that the child box may be an anonymous block box.
70
71 auto& boxGeometry = formattingContext().geometryForBox(layoutBox);
72 auto computedVerticalMargin = FormattingGeometry::computedVerticalMargin(layoutBox, horizontalConstraints);
73 auto nonCollapsedMargin = UsedVerticalMargin::NonCollapsedValues { computedVerticalMargin.before.value_or(0), computedVerticalMargin.after.value_or(0) };
74 auto borderAndPaddingTop = boxGeometry.borderTop() + boxGeometry.paddingTop().value_or(0);
75 auto height = overriddenVerticalValues.height ? overriddenVerticalValues.height.value() : computedHeight(layoutBox);
76
77 if (height)
78 return { *height, nonCollapsedMargin };
79
80 if (!is<ContainerBox>(layoutBox) || !downcast<ContainerBox>(layoutBox).hasInFlowChild())
81 return { 0, nonCollapsedMargin };
82
83 // 1. the bottom edge of the last line box, if the box establishes a inline formatting context with one or more lines
84 auto& layoutContainer = downcast<ContainerBox>(layoutBox);
85 if (layoutContainer.establishesInlineFormattingContext()) {
86 auto& inlineFormattingState = layoutState().establishedInlineFormattingState(layoutContainer);
87 auto& lines = inlineFormattingState.lines();
88 // Even empty containers generate one line.
89 ASSERT(!lines.isEmpty());
90 return { toLayoutUnit(lines.last().lineBoxLogicalRect().bottom() + inlineFormattingState.clearGapAfterLastLine()) - borderAndPaddingTop, nonCollapsedMargin };
91 }
92
93 // 2. the bottom edge of the bottom (possibly collapsed) margin of its last in-flow child, if the child's bottom margin...
94 auto marginCollapse = BlockMarginCollapse { layoutState(), formattingContext().formattingState() };
95 auto& lastInFlowChild = *layoutContainer.lastInFlowChild();
96 if (!marginCollapse.marginAfterCollapsesWithParentMarginAfter(lastInFlowChild)) {
97 auto& lastInFlowBoxGeometry = formattingContext().geometryForBox(lastInFlowChild);
98 auto bottomEdgeOfBottomMargin = BoxGeometry::borderBoxRect(lastInFlowBoxGeometry).bottom() + lastInFlowBoxGeometry.marginAfter();
99 return { bottomEdgeOfBottomMargin - borderAndPaddingTop, nonCollapsedMargin };
100 }
101
102 // 3. the bottom border edge of the last in-flow child whose top margin doesn't collapse with the element's bottom margin
103 auto* inFlowChild = &lastInFlowChild;
104 while (inFlowChild && marginCollapse.marginBeforeCollapsesWithParentMarginAfter(*inFlowChild))
105 inFlowChild = inFlowChild->previousInFlowSibling();
106 if (inFlowChild) {
107 auto& inFlowBoxGeometry = formattingContext().geometryForBox(*inFlowChild);
108 return { BoxGeometry::borderBoxTop(inFlowBoxGeometry) + inFlowBoxGeometry.borderBox().height() - borderAndPaddingTop, nonCollapsedMargin };
109 }
110
111 // 4. zero, otherwise
112 return { 0, nonCollapsedMargin };
113 };
114
115 // 10.6.7 'Auto' heights for block-level formatting context boxes.
116 auto isAutoHeight = !overriddenVerticalValues.height && !computedHeight(layoutBox);
117 if (isAutoHeight && (layoutBox.establishesFormattingContext() && !layoutBox.establishesInlineFormattingContext()))
118 return compute( OverriddenVerticalValues { contentHeightForFormattingContextRoot(downcast<ContainerBox>(layoutBox)) });
119 return compute(overriddenVerticalValues);
120}

122ContentWidthAndMargin BlockFormattingGeometry::inFlowNonReplacedContentWidthAndMargin(const Box& layoutBox, const HorizontalConstraints& horizontalConstraints, const OverriddenHorizontalValues& overriddenHorizontalValues) const
123{
124 ASSERT(layoutBox.isInFlow());
125
126 auto compute = [&]() {
127
128 // 10.3.3 Block-level, non-replaced elements in normal flow
129 //
130 // The following constraints must hold among the used values of the other properties:
131 // 'margin-left' + 'border-left-width' + 'padding-left' + 'width' + 'padding-right' + 'border-right-width' + 'margin-right' = width of containing block
132 //
133 // 1. If 'width' is not 'auto' and 'border-left-width' + 'padding-left' + 'width' + 'padding-right' + 'border-right-width'
134 // (plus any of 'margin-left' or 'margin-right' that are not 'auto') is larger than the width of the containing block, then
135 // any 'auto' values for 'margin-left' or 'margin-right' are, for the following rules, treated as zero.
136 //
137 // 2. If all of the above have a computed value other than 'auto', the values are said to be "over-constrained" and one of the used values will
138 // have to be different from its computed value. If the 'direction' property of the containing block has the value 'ltr', the specified value
139 // of 'margin-right' is ignored and the value is calculated so as to make the equality true. If the value of 'direction' is 'rtl',
140 // this happens to 'margin-left' instead.
141 //
142 // 3. If there is exactly one value specified as 'auto', its used value follows from the equality.
143 //
144 // 4. If 'width' is set to 'auto', any other 'auto' values become '0' and 'width' follows from the resulting equality.
145 //
146 // 5. If both 'margin-left' and 'margin-right' are 'auto', their used values are equal. This horizontally centers the element with respect to the
147 // edges of the containing block.
148
149 auto containingBlockWidth = horizontalConstraints.logicalWidth;
150 auto& containingBlockStyle = layoutBox.containingBlock().style();
151 auto& boxGeometry = formattingContext().geometryForBox(layoutBox);
152
153 auto width = overriddenHorizontalValues.width ? overriddenHorizontalValues.width : computedWidth(layoutBox, containingBlockWidth);
154 auto computedHorizontalMargin = FormattingGeometry::computedHorizontalMargin(layoutBox, horizontalConstraints);
155 UsedHorizontalMargin usedHorizontalMargin;
156 auto borderLeft = boxGeometry.borderLeft();
157 auto borderRight = boxGeometry.borderRight();
158 auto paddingLeft = boxGeometry.paddingLeft().value_or(0);
159 auto paddingRight = boxGeometry.paddingRight().value_or(0);
160
161 // #1
162 if (width) {
163 auto horizontalSpaceForMargin = containingBlockWidth - (computedHorizontalMargin.start.value_or(0) + borderLeft + paddingLeft + *width + paddingRight + borderRight + computedHorizontalMargin.end.value_or(0));
164 if (horizontalSpaceForMargin < 0)
165 usedHorizontalMargin = { computedHorizontalMargin.start.value_or(0), computedHorizontalMargin.end.value_or(0) };
166 }
167
168 // #2
169 if (width && computedHorizontalMargin.start && computedHorizontalMargin.end) {
170 if (containingBlockStyle.isLeftToRightDirection()) {
171 usedHorizontalMargin.start = *computedHorizontalMargin.start;
172 usedHorizontalMargin.end = containingBlockWidth - (usedHorizontalMargin.start + borderLeft + paddingLeft + *width + paddingRight + borderRight);
173 } else {
174 usedHorizontalMargin.end = *computedHorizontalMargin.end;
175 usedHorizontalMargin.start = containingBlockWidth - (borderLeft + paddingLeft + *width + paddingRight + borderRight + usedHorizontalMargin.end);
176 }
177 }
178
179 // #3
180 if (!computedHorizontalMargin.start && width && computedHorizontalMargin.end) {
181 usedHorizontalMargin.end = *computedHorizontalMargin.end;
182 usedHorizontalMargin.start = containingBlockWidth - (borderLeft + paddingLeft + *width + paddingRight + borderRight + usedHorizontalMargin.end);
183 } else if (computedHorizontalMargin.start && !width && computedHorizontalMargin.end) {
184 usedHorizontalMargin = { *computedHorizontalMargin.start, *computedHorizontalMargin.end };
185 width = containingBlockWidth - (usedHorizontalMargin.start + borderLeft + paddingLeft + paddingRight + borderRight + usedHorizontalMargin.end);
186 } else if (computedHorizontalMargin.start && width && !computedHorizontalMargin.end) {
187 usedHorizontalMargin.start = *computedHorizontalMargin.start;
188 usedHorizontalMargin.end = containingBlockWidth - (usedHorizontalMargin.start + borderLeft + paddingLeft + *width + paddingRight + borderRight);
189 }
190
191 // #4
192 if (!width) {
193 usedHorizontalMargin = { computedHorizontalMargin.start.value_or(0), computedHorizontalMargin.end.value_or(0) };
194 width = containingBlockWidth - (usedHorizontalMargin.start + borderLeft + paddingLeft + paddingRight + borderRight + usedHorizontalMargin.end);
195 }
196
197 // #5
198 if (!computedHorizontalMargin.start && !computedHorizontalMargin.end) {
199 auto horizontalSpaceForMargin = containingBlockWidth - (borderLeft + paddingLeft + *width + paddingRight + borderRight);
200 usedHorizontalMargin = { horizontalSpaceForMargin / 2, horizontalSpaceForMargin / 2 };
201 }
202
203 auto shouldApplyCenterAlignForBlockContent = containingBlockStyle.textAlign() == TextAlignMode::WebKitCenter && (computedHorizontalMargin.start || computedHorizontalMargin.end);
204 if (shouldApplyCenterAlignForBlockContent) {
205 auto borderBoxWidth = (borderLeft + paddingLeft + *width + paddingRight + borderRight);
206 auto marginStart = computedHorizontalMargin.start.value_or(0);
207 auto marginEnd = computedHorizontalMargin.end.value_or(0);
208 auto centeredLogicalLeftForMarginBox = std::max((containingBlockWidth - borderBoxWidth - marginStart - marginEnd) / 2, 0_lu);
209 usedHorizontalMargin.start = centeredLogicalLeftForMarginBox + marginStart;
210 usedHorizontalMargin.end = containingBlockWidth - borderBoxWidth - marginStart + marginEnd;
211 }
212 ASSERT(width);
213
214 return ContentWidthAndMargin { *width, usedHorizontalMargin };
215 };
216
217 auto contentWidthAndMargin = compute();
218 LOG_WITH_STREAM(FormattingContextLayout, stream << "[Width][Margin] -> inflow non-replaced -> width(" << contentWidthAndMargin.contentWidth << "px) margin(" << contentWidthAndMargin.usedMargin.start << "px, " << contentWidthAndMargin.usedMargin.end << "px) -> layoutBox(" << &layoutBox << ")");
219 return contentWidthAndMargin;
220}

222ContentWidthAndMargin BlockFormattingGeometry::inFlowReplacedContentWidthAndMargin(const ReplacedBox& replacedBox, const HorizontalConstraints& horizontalConstraints, const OverriddenHorizontalValues& overriddenHorizontalValues) const
223{
224 ASSERT(replacedBox.isInFlow());
225
226 // 10.3.4 Block-level, replaced elements in normal flow
227 //
228 // 1. The used value of 'width' is determined as for inline replaced elements.
229 // 2. Then the rules for non-replaced block-level elements are applied to determine the margins.
230
231 // #1
232 auto usedWidth = inlineReplacedContentWidthAndMargin(replacedBox, horizontalConstraints, { }, overriddenHorizontalValues).contentWidth;
233 // #2
234 auto nonReplacedWidthAndMargin = inFlowNonReplacedContentWidthAndMargin(replacedBox, horizontalConstraints, OverriddenHorizontalValues { usedWidth, overriddenHorizontalValues.margin });
235
236 LOG_WITH_STREAM(FormattingContextLayout, stream << "[Width][Margin] -> inflow replaced -> width(" << usedWidth << "px) margin(" << nonReplacedWidthAndMargin.usedMargin.start << "px, " << nonReplacedWidthAndMargin.usedMargin.end << "px) -> layoutBox(" << &replacedBox << ")");
237 return { usedWidth, nonReplacedWidthAndMargin.usedMargin };
238}

240LayoutUnit BlockFormattingGeometry::staticVerticalPosition(const Box& layoutBox, LayoutUnit containingBlockContentBoxTop) const
241{
242 // https://www.w3.org/TR/CSS22/visuren.html#block-formatting
243 // In a block formatting context, boxes are laid out one after the other, vertically, beginning at the top of a containing block.
244 // The vertical distance between two sibling boxes is determined by the 'margin' properties.
245 // Vertical margins between adjacent block-level boxes in a block formatting context collapse.
246 if (auto* previousInFlowSibling = layoutBox.previousInFlowSibling()) {
247 auto& previousInFlowBoxGeometry = formattingContext().geometryForBox(*previousInFlowSibling);
248 return BoxGeometry::borderBoxRect(previousInFlowBoxGeometry).bottom() + previousInFlowBoxGeometry.marginAfter();
249 }
250 return containingBlockContentBoxTop;
251}

253LayoutUnit BlockFormattingGeometry::staticHorizontalPosition(const Box& layoutBox, const HorizontalConstraints& horizontalConstraints) const
254{
255 // https://www.w3.org/TR/CSS22/visuren.html#block-formatting
256 // In a block formatting context, each box's left outer edge touches the left edge of the containing block (for right-to-left formatting, right edges touch).
257 return horizontalConstraints.logicalLeft + formattingContext().geometryForBox(layoutBox).marginStart();
258}

260ContentHeightAndMargin BlockFormattingGeometry::inFlowContentHeightAndMargin(const Box& layoutBox, const HorizontalConstraints& horizontalConstraints, const OverriddenVerticalValues& overriddenVerticalValues) const
261{
262 ASSERT(layoutBox.isInFlow());
263
264 // 10.6.2 Inline replaced elements, block-level replaced elements in normal flow, 'inline-block'
265 // replaced elements in normal flow and floating replaced elements
266 if (layoutBox.isReplacedBox())
267 return inlineReplacedContentHeightAndMargin(downcast<ReplacedBox>(layoutBox), horizontalConstraints, { }, overriddenVerticalValues);
268
269 ContentHeightAndMargin contentHeightAndMargin;
270 if (layoutBox.isOverflowVisible() && !layoutBox.isDocumentBox()) {
271 // TODO: Figure out the case for the document element. Let's just complicated-case it for now.
272 contentHeightAndMargin = inFlowNonReplacedContentHeightAndMargin(layoutBox, horizontalConstraints, overriddenVerticalValues);
273 } else {
274 // 10.6.6 Complicated cases
275 // Block-level, non-replaced elements in normal flow when 'overflow' does not compute to 'visible' (except if the 'overflow' property's value has been propagated to the viewport).
276 contentHeightAndMargin = complicatedCases(layoutBox, horizontalConstraints, overriddenVerticalValues);
277 }
278
279 if (layoutState().inQuirksMode()) {
280 if (auto stretchedInFlowHeight = formattingContext().formattingQuirks().stretchedInFlowHeightIfApplicable(layoutBox, contentHeightAndMargin))
281 contentHeightAndMargin.contentHeight = *stretchedInFlowHeight;
282 }
283
284 LOG_WITH_STREAM(FormattingContextLayout, stream << "[Height][Margin] -> inflow non-replaced -> streched to viewport -> height(" << contentHeightAndMargin.contentHeight << "px) margin(" << contentHeightAndMargin.nonCollapsedMargin.before << "px, " << contentHeightAndMargin.nonCollapsedMargin.after << "px) -> layoutBox(" << &layoutBox << ")");
285 return contentHeightAndMargin;
286}

297ContentWidthAndMargin BlockFormattingGeometry::computedContentWidthAndMargin(const Box& layoutBox, const HorizontalConstraints& horizontalConstraints, std::optional<LayoutUnit> availableWidthFloatAvoider) const
298{
299 auto compute = [&] (auto constraintsForWidth, std::optional<LayoutUnit> usedWidth) {
300 if (layoutBox.isFloatingPositioned())
301 return floatingContentWidthAndMargin(layoutBox, constraintsForWidth, { usedWidth, { } });
302
303 if (layoutBox.isInFlow())
304 return inFlowContentWidthAndMargin(layoutBox, constraintsForWidth, { usedWidth, { } });
305
306 ASSERT_NOT_REACHED();
307 return ContentWidthAndMargin { };
308 };
309
310 auto horizontalConstraintsForWidth = horizontalConstraints;
311 if (layoutBox.style().logicalWidth().isAuto() && availableWidthFloatAvoider) {
312 // While the non-auto width values should all be resolved against the containing block's width, when
313 // the width is auto the available horizontal space is shrunk by neighboring floats.
314 horizontalConstraintsForWidth.logicalWidth = *availableWidthFloatAvoider;
315 }
316 auto contentWidthAndMargin = compute(horizontalConstraintsForWidth, { });
317 auto availableWidth = horizontalConstraints.logicalWidth;
318 if (auto maxWidth = computedMaxWidth(layoutBox, availableWidth)) {
319 auto maxWidthAndMargin = compute(horizontalConstraints, maxWidth);
320 if (contentWidthAndMargin.contentWidth > maxWidthAndMargin.contentWidth)
321 contentWidthAndMargin = maxWidthAndMargin;
322 }
323
324 auto minWidth = computedMinWidth(layoutBox, availableWidth).value_or(0);
325 auto minWidthAndMargin = compute(horizontalConstraints, minWidth);
326 if (contentWidthAndMargin.contentWidth < minWidthAndMargin.contentWidth)
327 contentWidthAndMargin = minWidthAndMargin;
328 return contentWidthAndMargin;
329}

331IntrinsicWidthConstraints BlockFormattingGeometry::intrinsicWidthConstraints(const Box& layoutBox) const
332{
333 auto fixedMarginBorderAndPadding = [&](auto& layoutBox) {
334 auto& style = layoutBox.style();
335 return fixedValue(style.marginStart()).value_or(0)
336 + LayoutUnit { style.borderLeftWidth() }
337 + fixedValue(style.paddingLeft()).value_or(0)
338 + fixedValue(style.paddingRight()).value_or(0)
339 + LayoutUnit { style.borderRightWidth() }
340 + fixedValue(style.marginEnd()).value_or(0);
341 };
342
343 auto computedIntrinsicWidthConstraints = [&]() -> IntrinsicWidthConstraints {
344 auto logicalWidth = layoutBox.style().logicalWidth();
345 // Minimum/maximum width can't be depending on the containing block's width.
346 auto needsResolvedContainingBlockWidth = logicalWidth.isCalculated() || logicalWidth.isPercent() || logicalWidth.isRelative();
347 if (needsResolvedContainingBlockWidth)
348 return { };
349
350 if (auto width = fixedValue(logicalWidth))
351 return { *width, *width };
352
353 if (layoutBox.isReplacedBox()) {
354 auto& replacedBox = downcast<ReplacedBox>(layoutBox);
355 if (replacedBox.hasIntrinsicWidth()) {
356 auto replacedWidth = replacedBox.intrinsicWidth();
357 return { replacedWidth, replacedWidth };
358 }
359 return { };
360 }
361
362 if (!is<ContainerBox>(layoutBox) || !downcast<ContainerBox>(layoutBox).hasInFlowOrFloatingChild())
363 return { };
364
365 if (layoutBox.isSizeContainmentBox()) {
366 // The intrinsic sizes of the size containment box are determined as if the element had no content,
367 // following the same logic as when sizing as if empty.
368 return { };
369 }
370
371 auto& layoutState = this->layoutState();
372 if (layoutBox.establishesFormattingContext()) {
373 auto intrinsicWidthConstraints = LayoutContext::createFormattingContext(downcast<ContainerBox>(layoutBox), const_cast<LayoutState&>(layoutState))->computedIntrinsicWidthConstraints();
374 if (logicalWidth.isMinContent())
375 return { intrinsicWidthConstraints.minimum, intrinsicWidthConstraints.minimum };
376 if (logicalWidth.isMaxContent())
377 return { intrinsicWidthConstraints.maximum, intrinsicWidthConstraints.maximum };
378 return intrinsicWidthConstraints;
379 }
380
381 auto intrinsicWidthConstraints = IntrinsicWidthConstraints { };
382 auto& formattingState = layoutState.formattingStateForBox(layoutBox);
383 for (auto& child : childrenOfType<Box>(downcast<ContainerBox>(layoutBox))) {
384 if (child.isOutOfFlowPositioned() || (child.isFloatAvoider() && !child.hasFloatClear()))
385 continue;
386 auto childIntrinsicWidthConstraints = formattingState.intrinsicWidthConstraintsForBox(child);
387 ASSERT(childIntrinsicWidthConstraints);
388
389 intrinsicWidthConstraints.minimum = std::max(intrinsicWidthConstraints.minimum, childIntrinsicWidthConstraints->minimum);
390 intrinsicWidthConstraints.maximum = std::max(intrinsicWidthConstraints.maximum, childIntrinsicWidthConstraints->maximum);
391 }
392 return intrinsicWidthConstraints;
393 };
394 // FIXME Check for box-sizing: border-box;
395 auto intrinsicWidthConstraints = constrainByMinMaxWidth(layoutBox, computedIntrinsicWidthConstraints());
396 intrinsicWidthConstraints.expand(fixedMarginBorderAndPadding(layoutBox));
397 return intrinsicWidthConstraints;
398}
