43static bool isQuirkContainer(const Box& layoutBox)
44{

53static bool needsStretching(const Box& layoutBox)
54{
55 ASSERT(layoutBox.isInFlow());
56 // In quirks mode, in-flow body and html stretch to the initial containing block (height: auto only).
57 if (!layoutBox.isDocumentBox() && !layoutBox.isBodyBox())
58 return false;
59 return layoutBox.style().logicalHeight().isAuto();
60}

62std::optional<LayoutUnit> BlockFormattingQuirks::stretchedInFlowHeightIfApplicable(const Box& layoutBox, ContentHeightAndMargin contentHeightAndMargin) const
63{
64 ASSERT(layoutState().inQuirksMode());
65 if (!needsStretching(layoutBox))
66 return { };
67 auto& formattingContext = downcast<BlockFormattingContext>(this->formattingContext());
68 auto nonCollapsedVerticalMargin = contentHeightAndMargin.nonCollapsedMargin.before + contentHeightAndMargin.nonCollapsedMargin.after;
69
70 if (layoutBox.isDocumentBox()) {
71 // Let's stretch the inflow document box(<html>) to the height of the initial containing block (view).
72 auto documentBoxContentHeight = formattingContext.geometryForBox(layoutBox.initialContainingBlock(), FormattingContext::EscapeReason::DocumentBoxStretchesToViewportQuirk).contentBoxHeight();
73 // Document box's own vertical margin/border/padding values always shrink the content height.
74 auto& documentBoxGeometry = formattingContext.geometryForBox(layoutBox);
75 documentBoxContentHeight -= nonCollapsedVerticalMargin + documentBoxGeometry.verticalBorder() + documentBoxGeometry.verticalPadding().value_or(0);
76 return std::max(contentHeightAndMargin.contentHeight, documentBoxContentHeight);
77 }
78
79 // Here is the quirky part for body box when it stretches all the way to the ICB even when the document box does not (e.g. out-of-flow positioned).
80 ASSERT(layoutBox.isBodyBox());
81 auto& initialContainingBlock = layoutBox.initialContainingBlock();
82 auto& initialContainingBlockGeometry = formattingContext.geometryForBox(initialContainingBlock, FormattingContext::EscapeReason::BodyStretchesToViewportQuirk);
83 // Start the content height with the ICB.
84 auto bodyBoxContentHeight = initialContainingBlockGeometry.contentBoxHeight();
85 // Body box's own border and padding shrink the content height.
86 auto& bodyBoxGeometry = formattingContext.geometryForBox(layoutBox);
87 bodyBoxContentHeight -= bodyBoxGeometry.verticalBorder() + bodyBoxGeometry.verticalPadding().value_or(0);
88 // Body box never collapses its vertical margins with the document box but it might collapse its margin with its descendants.
89 auto nonCollapsedMargin = contentHeightAndMargin.nonCollapsedMargin;
90 auto marginCollapse = BlockMarginCollapse { formattingContext.layoutState(), formattingContext.formattingState() };
91 auto collapsedMargin = marginCollapse.collapsedVerticalValues(layoutBox, nonCollapsedMargin).collapsedValues;
92 auto usedVerticalMargin = collapsedMargin.before.value_or(nonCollapsedMargin.before);
93 usedVerticalMargin += collapsedMargin.isCollapsedThrough ? nonCollapsedMargin.after : collapsedMargin.after.value_or(nonCollapsedMargin.after);
94 bodyBoxContentHeight -= usedVerticalMargin;
95 // Document box's padding and border also shrink the body box's content height.
96 auto& documentBox = layoutBox.parent();
97 auto& documentBoxGeometry = formattingContext.geometryForBox(documentBox, FormattingContext::EscapeReason::BodyStretchesToViewportQuirk);
98 bodyBoxContentHeight -= documentBoxGeometry.verticalBorder() + documentBoxGeometry.verticalPadding().value_or(0);
99 // However the non-in-flow document box's vertical margins are ignored. They don't affect the body box's content height.
100 if (documentBox.isInFlow()) {
101 auto& formattingGeometry = formattingContext.formattingGeometry();
102 auto precomputeDocumentBoxVerticalMargin = formattingGeometry.computedVerticalMargin(documentBox, formattingGeometry.constraintsForInFlowContent(initialContainingBlock, FormattingContext::EscapeReason::BodyStretchesToViewportQuirk).horizontal());
103 bodyBoxContentHeight -= precomputeDocumentBoxVerticalMargin.before.value_or(0) + precomputeDocumentBoxVerticalMargin.after.value_or(0);
104 }
105 return std::max(contentHeightAndMargin.contentHeight, bodyBoxContentHeight);
106}

108bool BlockFormattingQuirks::shouldIgnoreCollapsedQuirkMargin(const Box& layoutBox)
109{
110 return isQuirkContainer(layoutBox);
111}

113enum class VerticalMargin { Before, After };
114static inline bool hasQuirkMarginToCollapse(const Box& layoutBox, VerticalMargin verticalMargin)
115{
116 if (!layoutBox.isInFlow())
117 return false;
118 auto& style = layoutBox.style();
119 return (verticalMargin == VerticalMargin::Before && style.hasMarginBeforeQuirk()) || (verticalMargin == VerticalMargin::After && style.hasMarginAfterQuirk());
120}
121
122bool BlockFormattingQuirks::shouldCollapseMarginBeforeWithParentMarginBefore(const Box& layoutBox)
123{
124 return hasQuirkMarginToCollapse(layoutBox, VerticalMargin::Before) && isQuirkContainer(layoutBox.containingBlock());
125}
126
127bool BlockFormattingQuirks::shouldCollapseMarginAfterWithParentMarginAfter(const Box& layoutBox)
128{
129 return hasQuirkMarginToCollapse(layoutBox, VerticalMargin::After) && isQuirkContainer(layoutBox.containingBlock());
130}
