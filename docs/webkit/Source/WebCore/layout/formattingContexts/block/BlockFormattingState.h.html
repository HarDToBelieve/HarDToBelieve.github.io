32#include <wtf/WeakHashSet.h>

34namespace WebCore {
35namespace Layout {
36
37// BlockFormattingState holds the state for a particular block formatting context tree.
38class BlockFormattingState : public FormattingState {
39 WTF_MAKE_ISO_ALLOCATED(BlockFormattingState);
40public:
41 BlockFormattingState(Ref<FloatingState>&&, LayoutState&);
42 ~BlockFormattingState();
43
44 void setUsedVerticalMargin(const Box& layoutBox, const UsedVerticalMargin& usedVerticalMargin) { m_usedVerticalMargins.set(&layoutBox, usedVerticalMargin); }
45 UsedVerticalMargin usedVerticalMargin(const Box& layoutBox) const { return m_usedVerticalMargins.get(&layoutBox); }
46 bool hasUsedVerticalMargin(const Box& layoutBox) const { return m_usedVerticalMargins.contains(&layoutBox); }
47
48 void setHasClearance(const Box& layoutBox) { m_clearanceSet.add(layoutBox); }
49 void clearHasClearance(const Box& layoutBox) { m_clearanceSet.remove(layoutBox); }
50 bool hasClearance(const Box& layoutBox) const { return m_clearanceSet.contains(layoutBox); }
51
52private:
53 HashMap<const Box*, UsedVerticalMargin> m_usedVerticalMargins;
54 WeakHashSet<const Box> m_clearanceSet;
55};
56
57}
58}
