101bool BlockMarginCollapse::marginBeforeCollapsesWithParentMarginAfter(const Box& layoutBox) const
102{
103 // 1. This is the last in-flow child and its margins collapse through and the margin after collapses with parent's margin after or
104 // 2. This box's margin after collapses with the next sibling's margin before and that sibling collapses through and
105 // we can get to the last in-flow child like that.
106 auto* lastInFlowChild = layoutBox.containingBlock().lastInFlowChild();
107 for (auto* currentBox = &layoutBox; currentBox; currentBox = currentBox->nextInFlowSibling()) {
108 if (!marginsCollapseThrough(*currentBox))
109 return false;
110 if (currentBox == lastInFlowChild)
111 return marginAfterCollapsesWithParentMarginAfter(*currentBox);
112 if (!marginAfterCollapsesWithNextSiblingMarginBefore(*currentBox))
113 return false;
114 }
115 ASSERT_NOT_REACHED();
116 return false;
117}

119bool BlockMarginCollapse::marginBeforeCollapsesWithParentMarginBefore(const Box& layoutBox) const
120{
121 // The first inflow child could propagate its top margin to parent.
122 // https://www.w3.org/TR/CSS21/box.html#collapsing-margins
123 ASSERT(layoutBox.isBlockLevelBox());
124
125 if (inQuirksMode() && BlockFormattingQuirks::shouldCollapseMarginBeforeWithParentMarginBefore(layoutBox))
126 return true;
127
128 // Margins between a floated box and any other box do not collapse.
129 if (layoutBox.isFloatingPositioned())
130 return false;
131
132 // Margins of absolutely positioned boxes do not collapse.
133 if (layoutBox.isOutOfFlowPositioned())
134 return false;
135
136 // Margins of inline-block boxes do not collapse.
137 if (layoutBox.isInlineBlockBox())
138 return false;
139
140 // Only the first inlflow child collapses with parent.
141 if (layoutBox.previousInFlowSibling())
142 return false;
143
144 auto& containingBlock = layoutBox.containingBlock();
145 // Margins of elements that establish new block formatting contexts do not collapse with their in-flow children
146 if (establishesBlockFormattingContext(containingBlock))
147 return false;
148
149 if (hasBorderBefore(containingBlock))
150 return false;
151
152 if (hasPaddingBefore(containingBlock))
153 return false;
154
155 // ...and the child has no clearance.
156 if (hasClearance(layoutBox))
157 return false;
158
159 return true;
160}

162bool BlockMarginCollapse::marginBeforeCollapsesWithPreviousSiblingMarginAfter(const Box& layoutBox) const
163{
164 ASSERT(layoutBox.isBlockLevelBox());
165
166 if (!layoutBox.previousInFlowSibling())
167 return false;
168
169 auto& previousInFlowSibling = *layoutBox.previousInFlowSibling();
170 // Margins between a floated box and any other box do not collapse.
171 if (layoutBox.isFloatingPositioned() || previousInFlowSibling.isFloatingPositioned())
172 return false;
173
174 // Margins of absolutely positioned boxes do not collapse.
175 if ((layoutBox.isOutOfFlowPositioned() && !layoutBox.style().top().isAuto())
176 || (previousInFlowSibling.isOutOfFlowPositioned() && !previousInFlowSibling.style().bottom().isAuto()))
177 return false;
178
179 // Margins of inline-block boxes do not collapse.
180 if (layoutBox.isInlineBlockBox() || previousInFlowSibling.isInlineBlockBox())
181 return false;
182
183 // The bottom margin of an in-flow block-level element always collapses with the top margin of
184 // its next in-flow block-level sibling, unless that sibling has clearance.
185 if (hasClearance(layoutBox))
186 return false;
187
188 return true;
189}

191bool BlockMarginCollapse::marginBeforeCollapsesWithFirstInFlowChildMarginBefore(const Box& layoutBox) const
192{
193 ASSERT(layoutBox.isBlockLevelBox());
194 // Margins of elements that establish new block formatting contexts do not collapse with their in-flow children.
195 if (establishesBlockFormattingContext(layoutBox))
196 return false;
197
198 // The top margin of an in-flow block element collapses with its first in-flow block-level
199 // child's top margin if the element has no top border...
200 if (hasBorderBefore(layoutBox))
201 return false;
202
203 // ...no top padding
204 if (hasPaddingBefore(layoutBox))
205 return false;
206
207 if (!is<ContainerBox>(layoutBox) || !downcast<ContainerBox>(layoutBox).hasInFlowChild())
208 return false;
209
210 auto& firstInFlowChild = *downcast<ContainerBox>(layoutBox).firstInFlowChild();
211 if (!firstInFlowChild.isBlockLevelBox())
212 return false;
213
214 // ...and the child has no clearance.
215 if (hasClearance(firstInFlowChild))
216 return false;
217
218 // Margins of inline-block boxes do not collapse.
219 if (firstInFlowChild.isInlineBlockBox())
220 return false;
221
222 return true;
223}

225bool BlockMarginCollapse::marginAfterCollapsesWithSiblingMarginBeforeWithClearance(const Box& layoutBox) const
226{
227 // If the top and bottom margins of an element with clearance are adjoining, its margins collapse with the adjoining margins
228 // of following siblings but that resulting margin does not collapse with the bottom margin of the parent block.
229 if (!marginsCollapseThrough(layoutBox))
230 return false;
231
232 for (auto* previousSibling = layoutBox.previousInFlowSibling(); previousSibling; previousSibling = previousSibling->previousInFlowSibling()) {
233 if (!marginsCollapseThrough(*previousSibling))
234 return false;
235 if (hasClearance(*previousSibling))
236 return true;
237 }
238 return false;
239}

241bool BlockMarginCollapse::marginAfterCollapsesWithParentMarginBefore(const Box& layoutBox) const
242{
243 // 1. This is the first in-flow child and its margins collapse through and the margin before collapses with parent's margin before or
244 // 2. This box's margin before collapses with the previous sibling's margin after and that sibling collapses through and
245 // we can get to the first in-flow child like that.
246 auto* firstInFlowChild = layoutBox.containingBlock().firstInFlowChild();
247 for (auto* currentBox = &layoutBox; currentBox; currentBox = currentBox->previousInFlowSibling()) {
248 if (!marginsCollapseThrough(*currentBox))
249 return false;
250 if (currentBox == firstInFlowChild)
251 return marginBeforeCollapsesWithParentMarginBefore(*currentBox);
252 if (!marginBeforeCollapsesWithPreviousSiblingMarginAfter(*currentBox))
253 return false;
254 }
255 ASSERT_NOT_REACHED();
256 return false;
257}

259bool BlockMarginCollapse::marginAfterCollapsesWithParentMarginAfter(const Box& layoutBox) const
260{
261 ASSERT(layoutBox.isBlockLevelBox());
262
263 if (inQuirksMode() && BlockFormattingQuirks::shouldCollapseMarginAfterWithParentMarginAfter(layoutBox))
264 return true;
265
266 // Margins between a floated box and any other box do not collapse.
267 if (layoutBox.isFloatingPositioned())
268 return false;
269
270 // Margins of absolutely positioned boxes do not collapse.
271 if (layoutBox.isOutOfFlowPositioned())
272 return false;
273
274 // Margins of inline-block boxes do not collapse.
275 if (layoutBox.isInlineBlockBox())
276 return false;
277
278 // Only the last inlflow child collapses with parent.
279 if (layoutBox.nextInFlowSibling())
280 return false;
281
282 auto& containingBlock = layoutBox.containingBlock();
283 // Margins of elements that establish new block formatting contexts do not collapse with their in-flow children.
284 if (establishesBlockFormattingContext(containingBlock))
285 return false;
286
287 // The bottom margin of an in-flow block box with a 'height' of 'auto' collapses with its last in-flow block-level child's bottom margin, if:
288 if (!containingBlock.style().height().isAuto())
289 return false;
290
291 // the box has no bottom padding, and
292 if (hasPaddingAfter(containingBlock))
293 return false;
294
295 // the box has no bottom border, and
296 if (hasBorderAfter(containingBlock))
297 return false;
298
299 // the child's bottom margin neither collapses with a top margin that has clearance...
300 if (marginAfterCollapsesWithSiblingMarginBeforeWithClearance(layoutBox))
301 return false;
302
303 // nor (if the box's min-height is non-zero) with the box's top margin.
304 auto computedMinHeight = containingBlock.style().logicalMinHeight();
305 if (!computedMinHeight.isAuto() && computedMinHeight.value() && marginAfterCollapsesWithParentMarginBefore(layoutBox))
306 return false;
307
308 return true;
309}

311bool BlockMarginCollapse::marginAfterCollapsesWithLastInFlowChildMarginAfter(const Box& layoutBox) const
312{
313 ASSERT(layoutBox.isBlockLevelBox());
314
315 // Margins of elements that establish new block formatting contexts do not collapse with their in-flow children.
316 if (establishesBlockFormattingContext(layoutBox))
317 return false;
318
319 if (!is<ContainerBox>(layoutBox) || !downcast<ContainerBox>(layoutBox).hasInFlowChild())
320 return false;
321
322 auto& lastInFlowChild = *downcast<ContainerBox>(layoutBox).lastInFlowChild();
323 if (!lastInFlowChild.isBlockLevelBox())
324 return false;
325
326 // The bottom margin of an in-flow block box with a 'height' of 'auto' collapses with its last in-flow block-level child's bottom margin, if:
327 if (!layoutBox.style().height().isAuto())
328 return false;
329
330 // the box has no bottom padding, and
331 if (hasPaddingAfter(layoutBox))
332 return false;
333
334 // the box has no bottom border, and
335 if (hasBorderAfter(layoutBox))
336 return false;
337
338 // the child's bottom margin neither collapses with a top margin that has clearance...
339 if (marginAfterCollapsesWithSiblingMarginBeforeWithClearance(lastInFlowChild))
340 return false;
341
342 // nor (if the box's min-height is non-zero) with the box's top margin.
343 auto computedMinHeight = layoutBox.style().logicalMinHeight();
344 if (!computedMinHeight.isAuto() && computedMinHeight.value()
345 && (marginAfterCollapsesWithParentMarginBefore(lastInFlowChild) || hasClearance(lastInFlowChild)))
346 return false;
347
348 // Margins of inline-block boxes do not collapse.
349 if (lastInFlowChild.isInlineBlockBox())
350 return false;
351
352 // This is a quirk behavior: When the margin after of the last inflow child (or a previous sibling with collapsed through margins)
353 // collapses with a quirk parent's the margin before, then the same margin after does not collapses with the parent's margin after.
354 auto shouldIgnoreCollapsedMargin = inQuirksMode() && BlockFormattingQuirks::shouldIgnoreCollapsedQuirkMargin(layoutBox);
355 if (shouldIgnoreCollapsedMargin && marginAfterCollapsesWithParentMarginBefore(lastInFlowChild))
356 return false;
357
358 return true;
359}

361bool BlockMarginCollapse::marginAfterCollapsesWithNextSiblingMarginBefore(const Box& layoutBox) const
362{
363 ASSERT(layoutBox.isBlockLevelBox());
364
365 if (!layoutBox.nextInFlowSibling())
366 return false;
367
368 return marginBeforeCollapsesWithPreviousSiblingMarginAfter(*layoutBox.nextInFlowSibling());

371bool BlockMarginCollapse::marginsCollapseThrough(const Box& layoutBox) const
372{
373 ASSERT(layoutBox.isBlockLevelBox());
374
375 // A box's own margins collapse if the 'min-height' property is zero, and it has neither top or bottom borders nor top or bottom padding,
376 // and it has a 'height' of either 0 or 'auto', and it does not contain a line box, and all of its in-flow children's margins (if any) collapse.
377 if (hasBorderBefore(layoutBox) || hasBorderAfter(layoutBox))
378 return false;
379
380 if (hasPaddingBefore(layoutBox) || hasPaddingAfter(layoutBox))
381 return false;
382
383 // Margins are not adjoining when the box has clearance.
384 if (hasClearance(layoutBox))
385 return false;
386
387 auto& style = layoutBox.style();
388 auto computedHeightValueIsZero = style.height().isFixed() && !style.height().value();
389 if (!(style.height().isAuto() || computedHeightValueIsZero))
390 return false;
391
392 // FIXME: Check for computed 0 height.
393 if (!style.minHeight().isAuto())
394 return false;
395
396 // FIXME: Block replaced boxes clearly don't collapse through their margins, but I couldn't find it in the spec yet (and no, it's not a quirk).
397 if (layoutBox.isReplacedBox())
398 return false;
399
400 if (!is<ContainerBox>(layoutBox))
401 return true;
402
403 if (!downcast<ContainerBox>(layoutBox).hasInFlowChild())
404 return !establishesBlockFormattingContext(layoutBox);
405
406 if (layoutBox.establishesFormattingContext()) {
407 if (layoutBox.establishesInlineFormattingContext()) {
408 auto& layoutState = this->layoutState();
409 // If we get here through margin estimation, we don't necessarily have an actual state for this layout box since
410 // we haven't started laying it out yet.
411 auto& containerBox = downcast<ContainerBox>(layoutBox);
412 if (!layoutState.hasInlineFormattingState(containerBox))
413 return false;
414
415 auto isConsideredEmpty = [&] {
416 auto& inlineFormattingState = layoutState.establishedInlineFormattingState(containerBox);
417 if (!inlineFormattingState.lines().isEmpty())
418 return false;
419 // Any float box in this formatting context prevents collapsing through.
420 auto& floats = inlineFormattingState.floatingState().floats();
421 for (auto& floatItem : floats) {
422 if (floatItem.isInFormattingContextOf(containerBox))
423 return false;
424 }
425 return true;
426 };
427 return isConsideredEmpty();
428 }
429
430 // A root of a non-inline formatting context (table, flex etc) with inflow descendants should not collapse through.
431 return false;
432 }
433
434 for (auto* inflowChild = downcast<ContainerBox>(layoutBox).firstInFlowOrFloatingChild(); inflowChild; inflowChild = inflowChild->nextInFlowOrFloatingSibling()) {
435 if (establishesBlockFormattingContext(*inflowChild))
436 return false;
437 if (!marginsCollapseThrough(*inflowChild))
438 return false;
439 }
440 return true;
441}

443UsedVerticalMargin::PositiveAndNegativePair::Values BlockMarginCollapse::computedPositiveAndNegativeMargin(UsedVerticalMargin::PositiveAndNegativePair::Values a, UsedVerticalMargin::PositiveAndNegativePair::Values b) const
444{
445 UsedVerticalMargin::PositiveAndNegativePair::Values computedValues;
446 if (a.positive && b.positive)
447 computedValues.positive = std::max(*a.positive, *b.positive);
448 else
449 computedValues.positive = a.positive ? a.positive : b.positive;
450
451 if (a.negative && b.negative)
452 computedValues.negative = std::min(*a.negative, *b.negative);
453 else
454 computedValues.negative = a.negative ? a.negative : b.negative;
455
456 if (a.isNonZero() && b.isNonZero())
457 computedValues.isQuirk = a.isQuirk || b.isQuirk;
458 else if (a.isNonZero())
459 computedValues.isQuirk = a.isQuirk;
460 else
461 computedValues.isQuirk = b.isQuirk;
462
463 return computedValues;
464}

466std::optional<LayoutUnit> BlockMarginCollapse::marginValue(UsedVerticalMargin::PositiveAndNegativePair::Values marginValues) const
467{
468 // When two or more margins collapse, the resulting margin width is the maximum of the collapsing margins' widths.
469 // In the case of negative margins, the maximum of the absolute values of the negative adjoining margins is deducted from the maximum
470 // of the positive adjoining margins. If there are no positive margins, the maximum of the absolute values of the adjoining margins is deducted from zero.
471 if (!marginValues.negative)
472 return marginValues.positive;
473
474 if (!marginValues.positive)
475 return marginValues.negative;
476
477 return *marginValues.positive + *marginValues.negative;
478}

480UsedVerticalMargin::PositiveAndNegativePair::Values BlockMarginCollapse::positiveNegativeValues(const Box& layoutBox, MarginType marginType) const
481{
482 // By the time we get here in BFC layout to gather positive and negative margin values for either a previous sibling or a child box,
483 // we mush have computed and cached those values.
484 ASSERT(formattingState().hasUsedVerticalMargin(layoutBox));
485 auto positiveAndNegativeVerticalMargin = formattingState().usedVerticalMargin(layoutBox).positiveAndNegativeValues;
486 return marginType == MarginType::Before ? positiveAndNegativeVerticalMargin.before : positiveAndNegativeVerticalMargin.after;
487}

489UsedVerticalMargin::PositiveAndNegativePair::Values BlockMarginCollapse::positiveNegativeMarginBefore(const Box& layoutBox, UsedVerticalMargin::NonCollapsedValues nonCollapsedValues) const
490{
491 auto firstChildCollapsedMarginBefore = [&]() -> UsedVerticalMargin::PositiveAndNegativePair::Values {
492 if (!marginBeforeCollapsesWithFirstInFlowChildMarginBefore(layoutBox))
493 return { };
494 return positiveNegativeValues(*downcast<ContainerBox>(layoutBox).firstInFlowChild(), MarginType::Before);
495 };
496
497 auto previouSiblingCollapsedMarginAfter = [&]() -> UsedVerticalMargin::PositiveAndNegativePair::Values {
498 if (!marginBeforeCollapsesWithPreviousSiblingMarginAfter(layoutBox))
499 return { };
500 return positiveNegativeValues(*layoutBox.previousInFlowSibling(), MarginType::After);
501 };
502
503 // 1. Gather positive and negative margin values from first child if margins are adjoining.
504 // 2. Gather positive and negative margin values from previous inflow sibling if margins are adjoining.
505 // 3. Compute min/max positive and negative collapsed margin values using non-collpased computed margin before.
506 auto collapsedMarginBefore = computedPositiveAndNegativeMargin(firstChildCollapsedMarginBefore(), previouSiblingCollapsedMarginAfter());
507 auto shouldIgnoreCollapsedMargin = collapsedMarginBefore.isQuirk && inQuirksMode() && BlockFormattingQuirks::shouldIgnoreCollapsedQuirkMargin(layoutBox);
508 if (shouldIgnoreCollapsedMargin)
509 collapsedMarginBefore = { };
510
511 UsedVerticalMargin::PositiveAndNegativePair::Values nonCollapsedBefore;
512 if (nonCollapsedValues.before > 0)
513 nonCollapsedBefore = { nonCollapsedValues.before, { }, layoutBox.style().hasMarginBeforeQuirk() };
514 else if (nonCollapsedValues.before < 0)
515 nonCollapsedBefore = { { }, nonCollapsedValues.before, layoutBox.style().hasMarginBeforeQuirk() };
516
517 return computedPositiveAndNegativeMargin(collapsedMarginBefore, nonCollapsedBefore);
518}

520UsedVerticalMargin::PositiveAndNegativePair::Values BlockMarginCollapse::positiveNegativeMarginAfter(const Box& layoutBox, UsedVerticalMargin::NonCollapsedValues nonCollapsedValues) const
521{
522 auto lastChildCollapsedMarginAfter = [&]() -> UsedVerticalMargin::PositiveAndNegativePair::Values {
523 if (!marginAfterCollapsesWithLastInFlowChildMarginAfter(layoutBox))
524 return { };
525 return positiveNegativeValues(*downcast<ContainerBox>(layoutBox).lastInFlowChild(), MarginType::After);
526 };
527
528 // We don't know yet the margin before value of the next sibling. Let's just pretend it does not have one and
529 // update it later when we compute the next sibling's margin before. See updateMarginAfterForPreviousSibling.
530 UsedVerticalMargin::PositiveAndNegativePair::Values nonCollapsedAfter;
531 if (nonCollapsedValues.after > 0)
532 nonCollapsedAfter = { nonCollapsedValues.after, { }, layoutBox.style().hasMarginAfterQuirk() };
533 else if (nonCollapsedValues.after < 0)
534 nonCollapsedAfter = { { }, nonCollapsedValues.after, layoutBox.style().hasMarginAfterQuirk() };
535
536 return computedPositiveAndNegativeMargin(lastChildCollapsedMarginAfter(), nonCollapsedAfter);
537}

539LayoutUnit BlockMarginCollapse::marginBeforeIgnoringCollapsingThrough(const Box& layoutBox, UsedVerticalMargin::NonCollapsedValues nonCollapsedValues)
540{
541 ASSERT(layoutBox.isBlockLevelBox());
542 return marginValue(positiveNegativeMarginBefore(layoutBox, nonCollapsedValues)).value_or(nonCollapsedValues.before);
543}

545UsedVerticalMargin BlockMarginCollapse::collapsedVerticalValues(const Box& layoutBox, UsedVerticalMargin::NonCollapsedValues nonCollapsedValues)
546{
547 ASSERT(layoutBox.isBlockLevelBox());
548 // 1. Get min/max margin top values from the first in-flow child if we are collapsing margin top with it.
549 // 2. Get min/max margin top values from the previous in-flow sibling, if we are collapsing margin top with it.
550 // 3. Get this layout box's computed margin top value.
551 // 4. Update the min/max value and compute the final margin.
552 auto positiveAndNegativeVerticalMargin = UsedVerticalMargin::PositiveAndNegativePair { this->positiveNegativeMarginBefore(layoutBox, nonCollapsedValues), this->positiveNegativeMarginAfter(layoutBox, nonCollapsedValues) };
553
554 auto marginsCollapseThrough = this->marginsCollapseThrough(layoutBox);
555 if (marginsCollapseThrough) {
556 positiveAndNegativeVerticalMargin.before = computedPositiveAndNegativeMargin(positiveAndNegativeVerticalMargin.before, positiveAndNegativeVerticalMargin.after);
557 positiveAndNegativeVerticalMargin.after = positiveAndNegativeVerticalMargin.before;
558 }
559
560 auto hasCollapsedMarginBefore = marginBeforeCollapsesWithFirstInFlowChildMarginBefore(layoutBox) || marginBeforeCollapsesWithPreviousSiblingMarginAfter(layoutBox);
561 auto hasCollapsedMarginAfter = marginAfterCollapsesWithLastInFlowChildMarginAfter(layoutBox);
562 auto usedVerticalMargin = UsedVerticalMargin { nonCollapsedValues, { }, positiveAndNegativeVerticalMargin };
563
564 if ((hasCollapsedMarginBefore && hasCollapsedMarginAfter) || marginsCollapseThrough)
565 usedVerticalMargin.collapsedValues = { marginValue(positiveAndNegativeVerticalMargin.before), marginValue(positiveAndNegativeVerticalMargin.after), marginsCollapseThrough };
566 else if (hasCollapsedMarginBefore)
567 usedVerticalMargin.collapsedValues = { marginValue(positiveAndNegativeVerticalMargin.before), { }, false };
568 else if (hasCollapsedMarginAfter)
569 usedVerticalMargin.collapsedValues = { { }, marginValue(positiveAndNegativeVerticalMargin.after), false };
570 return usedVerticalMargin;
571}
