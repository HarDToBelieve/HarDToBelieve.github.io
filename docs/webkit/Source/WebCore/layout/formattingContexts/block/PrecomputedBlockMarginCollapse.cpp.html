43UsedVerticalMargin::PositiveAndNegativePair::Values BlockMarginCollapse::precomputedPositiveNegativeValues(const Box& layoutBox, const BlockFormattingGeometry& formattingGeometry) const
44{
45 auto& blockFormattingState = downcast<BlockFormattingState>(layoutState().formattingStateForBox(layoutBox));
46 if (blockFormattingState.hasUsedVerticalMargin(layoutBox))
47 return blockFormattingState.usedVerticalMargin(layoutBox).positiveAndNegativeValues.before;
48
49 auto horizontalConstraints = formattingGeometry.constraintsForInFlowContent(layoutBox.containingBlock()).horizontal();
50 auto computedVerticalMargin = formattingGeometry.computedVerticalMargin(layoutBox, horizontalConstraints);
51 auto nonCollapsedMargin = UsedVerticalMargin::NonCollapsedValues { computedVerticalMargin.before.value_or(0), computedVerticalMargin.after.value_or(0) };
52 return precomputedPositiveNegativeMarginBefore(layoutBox, nonCollapsedMargin, formattingGeometry);
53}
54
55UsedVerticalMargin::PositiveAndNegativePair::Values BlockMarginCollapse::precomputedPositiveNegativeMarginBefore(const Box& layoutBox, UsedVerticalMargin::NonCollapsedValues nonCollapsedValues, const BlockFormattingGeometry& formattingGeometry) const
56{
57 auto firstChildCollapsedMarginBefore = [&]() -> UsedVerticalMargin::PositiveAndNegativePair::Values {
58 if (!marginBeforeCollapsesWithFirstInFlowChildMarginBefore(layoutBox))
59 return { };
60 return precomputedPositiveNegativeValues(*downcast<ContainerBox>(layoutBox).firstInFlowChild(), formattingGeometry);
61 };
62
63 auto previouSiblingCollapsedMarginAfter = [&]() -> UsedVerticalMargin::PositiveAndNegativePair::Values {
64 if (!marginBeforeCollapsesWithPreviousSiblingMarginAfter(layoutBox))
65 return { };
66 auto& previousInFlowSibling = *layoutBox.previousInFlowSibling();
67 auto& blockFormattingState = downcast<BlockFormattingState>(layoutState().formattingStateForBox(previousInFlowSibling));
68 return blockFormattingState.usedVerticalMargin(previousInFlowSibling).positiveAndNegativeValues.after;
69 };
70
71 // 1. Gather positive and negative margin values from first child if margins are adjoining.
72 // 2. Gather positive and negative margin values from previous inflow sibling if margins are adjoining.
73 // 3. Compute min/max positive and negative collapsed margin values using non-collpased computed margin before.
74 auto collapsedMarginBefore = computedPositiveAndNegativeMargin(firstChildCollapsedMarginBefore(), previouSiblingCollapsedMarginAfter());
75 auto shouldIgnoreCollapsedMargin = collapsedMarginBefore.isQuirk && inQuirksMode() && BlockFormattingQuirks::shouldIgnoreCollapsedQuirkMargin(layoutBox);
76 if (shouldIgnoreCollapsedMargin)
77 collapsedMarginBefore = { };
78
79 auto nonCollapsedBefore = UsedVerticalMargin::PositiveAndNegativePair::Values { };
80 if (nonCollapsedValues.before > 0)
81 nonCollapsedBefore = { nonCollapsedValues.before, { }, layoutBox.style().hasMarginBeforeQuirk() };
82 else if (nonCollapsedValues.before < 0)
83 nonCollapsedBefore = { { }, nonCollapsedValues.before, layoutBox.style().hasMarginBeforeQuirk() };
84
85 return computedPositiveAndNegativeMargin(collapsedMarginBefore, nonCollapsedBefore);
86}
87
88PrecomputedMarginBefore BlockMarginCollapse::precomputedMarginBefore(const Box& layoutBox, UsedVerticalMargin::NonCollapsedValues usedNonCollapsedMargin, const BlockFormattingGeometry& formattingGeometry)
89{
90 ASSERT(layoutBox.isBlockLevelBox());
91 // Don't pre-compute vertical margins for out of flow boxes.
92 ASSERT(layoutBox.isInFlow() || layoutBox.isFloatingPositioned());
93 ASSERT(!layoutBox.isReplacedBox());
94
95 auto positiveNegativeMarginBefore = precomputedPositiveNegativeMarginBefore(layoutBox, usedNonCollapsedMargin, formattingGeometry);
96 auto collapsedMarginBefore = marginValue(positiveNegativeMarginBefore);
97 return { usedNonCollapsedMargin.before, collapsedMarginBefore, positiveNegativeMarginBefore };
98}
