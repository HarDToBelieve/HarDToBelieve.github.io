36namespace Layout {

38// This class implements the special block formatting context layout logic for the table wrapper.
39// https://www.w3.org/TR/CSS22/tables.html#model
40class TableWrapperBlockFormattingContext final : public BlockFormattingContext {
41 WTF_MAKE_ISO_ALLOCATED(TableWrapperBlockFormattingContext);
42public:
43 TableWrapperBlockFormattingContext(const ContainerBox& formattingContextRoot, BlockFormattingState&);
44
45 void layoutInFlowContent(InvalidationState&, const ConstraintsForInFlowContent&) final;
46
47 void setHorizontalConstraintsIgnoringFloats(const HorizontalConstraints& horizontalConstraints) { m_horizontalConstraintsIgnoringFloats = horizontalConstraints; }
48
49 const TableWrapperQuirks& formattingQuirks() const final { return m_tableWrapperFormattingQuirks; }
50
51private:
52 void layoutTableBox(const ContainerBox& tableBox, const ConstraintsForInFlowContent&);
53
54 void computeBorderAndPaddingForTableBox(const ContainerBox&, const HorizontalConstraints&);
55 void computeWidthAndMarginForTableBox(const ContainerBox&, const HorizontalConstraints&);
56 void computeHeightAndMarginForTableBox(const ContainerBox&, const ConstraintsForInFlowContent&);
57
58 HorizontalConstraints m_horizontalConstraintsIgnoringFloats;
59 const TableWrapperQuirks m_tableWrapperFormattingQuirks;
60};
61
62}
63}
