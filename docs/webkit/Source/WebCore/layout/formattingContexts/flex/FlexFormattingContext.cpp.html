52void FlexFormattingContext::layoutInFlowContent(InvalidationState&, const ConstraintsForInFlowContent& constraints)
53{
54 computeIntrinsicWidthConstraintsForFlexItems();
55 sizeAndPlaceFlexItems(constraints);

69void FlexFormattingContext::sizeAndPlaceFlexItems(const ConstraintsForInFlowContent& constraints)
70{
71 auto& formattingState = this->formattingState();
72 auto& formattingGeometry = this->formattingGeometry();
73 auto flexItemMainAxisStart = constraints.horizontal().logicalLeft;
74 auto flexItemMainAxisEnd = flexItemMainAxisStart;
75 auto flexItemCrosAxisStart = constraints.logicalTop();
76 auto flexItemCrosAxisEnd = flexItemCrosAxisStart;
77 for (auto& flexItem : childrenOfType<ContainerBox>(root())) {
78 ASSERT(flexItem.establishesFormattingContext());
79 // FIXME: This is just a simple, let's layout the flex items and place them next to each other setup.
80 auto intrinsicWidths = formattingState.intrinsicWidthConstraintsForBox(flexItem);
81 auto flexItemLogicalWidth = std::min(std::max(intrinsicWidths->minimum, constraints.horizontal().logicalWidth), intrinsicWidths->maximum);
82 auto flexItemConstraints = ConstraintsForInFlowContent { { { }, flexItemLogicalWidth }, { } };
83
84 auto invalidationState = InvalidationState { };
85 LayoutContext::createFormattingContext(flexItem, layoutState())->layoutInFlowContent(invalidationState, flexItemConstraints);
86
87 auto computeFlexItemGeometry = [&] {
88 auto& flexItemGeometry = formattingState.boxGeometry(flexItem);
89
90 flexItemGeometry.setLogicalTopLeft(LayoutPoint { flexItemMainAxisEnd, flexItemCrosAxisStart });
91
92 flexItemGeometry.setBorder(formattingGeometry.computedBorder(flexItem));
93 flexItemGeometry.setPadding(formattingGeometry.computedPadding(flexItem, constraints.horizontal().logicalWidth));
94
95 auto computedHorizontalMargin = formattingGeometry.computedHorizontalMargin(flexItem, constraints.horizontal());
96 flexItemGeometry.setHorizontalMargin({ computedHorizontalMargin.start.value_or(0_lu), computedHorizontalMargin.end.value_or(0_lu) });
97
98 auto computedVerticalMargin = formattingGeometry.computedVerticalMargin(flexItem, constraints.horizontal());
99 flexItemGeometry.setVerticalMargin({ computedVerticalMargin.before.value_or(0_lu), computedVerticalMargin.after.value_or(0_lu) });
100
101 flexItemGeometry.setContentBoxHeight(formattingGeometry.contentHeightForFormattingContextRoot(flexItem));
102 flexItemGeometry.setContentBoxWidth(flexItemLogicalWidth);
103 flexItemMainAxisEnd= BoxGeometry::borderBoxRect(flexItemGeometry).right();
104 flexItemCrosAxisEnd = std::max(flexItemCrosAxisEnd, BoxGeometry::borderBoxRect(flexItemGeometry).bottom());
105 };
106 computeFlexItemGeometry();
107 }
108 auto flexLine = InlineRect { flexItemCrosAxisStart, flexItemMainAxisStart, flexItemMainAxisEnd - flexItemMainAxisStart, flexItemCrosAxisEnd - flexItemCrosAxisStart };
109 formattingState.addLine(flexLine);
110}

112void FlexFormattingContext::computeIntrinsicWidthConstraintsForFlexItems()
113{
114 auto& formattingState = this->formattingState();
115 auto& formattingGeometry = this->formattingGeometry();
116 for (auto& flexItem : childrenOfType<ContainerBox>(root())) {
117 if (formattingState.intrinsicWidthConstraintsForBox(flexItem))
118 continue;
119 formattingState.setIntrinsicWidthConstraintsForBox(flexItem, formattingGeometry.intrinsicWidthConstraints(flexItem));
120 }
121}
