41class InlineContentBreaker {
42public:
43 struct PartialRun {
44 size_t length { 0 };
45 InlineLayoutUnit logicalWidth { 0 };
46 std::optional<InlineLayoutUnit> hyphenWidth { };
47 };
48 enum class IsEndOfLine { No, Yes };
49 struct Result {
50 enum class Action {
51 Keep, // Keep content on the current line.
52 Break, // Partial content is on the current line.
53 Wrap, // Content is wrapped to the next line.
54 WrapWithHyphen, // Content is wrapped to the next line and the current line ends with a visible hyphen.
55 // The current content overflows and can't get broken up into smaller bits.
56 RevertToLastWrapOpportunity, // The content needs to be reverted back to the last wrap opportunity.
57 RevertToLastNonOverflowingWrapOpportunity // The content needs to be reverted back to a wrap opportunity that still fits the line.
58 };
59 struct PartialTrailingContent {
60 size_t trailingRunIndex { 0 };
61 std::optional<PartialRun> partialRun; // nullopt partial run means the trailing run is a complete run.
62 };
63 Action action { Action::Keep };
64 IsEndOfLine isEndOfLine { IsEndOfLine::No };
65 std::optional<PartialTrailingContent> partialTrailingContent { };
66 const InlineItem* lastWrapOpportunityItem { nullptr };
67 };
68
69 // This struct represents the amount of continuous content committed to content breaking at a time (no in-between wrap opportunities).
70 // e.g.
71 // <div>text content <span>span1</span>between<span>span2</span></div>
72 // [text][ ][content][ ][inline box start][span1][inline box end][between][inline box start][span2][inline box end]
73 // continuous candidate content at a time:
74 // 1. [text]
75 // 2. [ ]
76 // 3. [content]
77 // 4. [ ]
78 // 5. [inline box start][span1][inline box end][between][inline box start][span2][inline box end]
79 // see https://drafts.csswg.org/css-text-3/#line-break-details
80 struct ContinuousContent {
81 InlineLayoutUnit logicalWidth() const { return m_logicalWidth; }
82 InlineLayoutUnit collapsibleLogicalWidth() const { return m_collapsibleLogicalWidth; }
83 InlineLayoutUnit nonCollapsibleLogicalWidth() const { return logicalWidth() - collapsibleLogicalWidth(); }
84 bool hasTrailingCollapsibleContent() const { return !!collapsibleLogicalWidth(); }
85 bool isFullyCollapsible() const { return logicalWidth() == collapsibleLogicalWidth(); }
86
87 void append(const InlineItem&, InlineLayoutUnit logicalWidth, std::optional<InlineLayoutUnit> collapsibleWidth);
88 void reset();
89
90 struct Run {
91 Run(const InlineItem&, InlineLayoutUnit logicalWidth);
92 Run(const Run&);
93 Run& operator=(const Run&);
94
95 const InlineItem& inlineItem;
96 InlineLayoutUnit logicalWidth { 0 };
97 };
98 using RunList = Vector<Run, 3>;
99 const RunList& runs() const { return m_runs; }
100
101 private:
102 RunList m_runs;
103 InlineLayoutUnit m_logicalWidth { 0 };
104 InlineLayoutUnit m_collapsibleLogicalWidth { 0 };
105 };

112 bool hasFullyCollapsibleTrailingRun { false };
113 bool hasContent { false };
114 bool hasWrapOpportunityAtPreviousPosition { false };
115 };
116 Result processInlineContent(const ContinuousContent&, const LineStatus&);
117 void setHyphenationDisabled() { n_hyphenationIsDisabled = true; }

119 static bool isWrappingAllowed(const InlineItem&);
120
121private:
122 Result processOverflowingContent(const ContinuousContent&, const LineStatus&) const;

124 std::optional<PartialRun> tryBreakingTextRun(const ContinuousContent::Run& overflowRun, InlineLayoutUnit logicalLeft, std::optional<InlineLayoutUnit> availableWidth, bool hasWrapOpportunityAtPreviousPosition) const;

132
133 bool n_hyphenationIsDisabled { false };
134};
