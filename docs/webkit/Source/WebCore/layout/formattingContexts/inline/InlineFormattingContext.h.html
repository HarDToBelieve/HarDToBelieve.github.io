37namespace Layout {
38
39class InlineFormattingState;
40class InvalidationState;
41class LineBox;
42
43// This class implements the layout logic for inline formatting contexts.
44// https://www.w3.org/TR/CSS22/visuren.html#inline-formatting
45class InlineFormattingContext final : public FormattingContext {
46 WTF_MAKE_ISO_ALLOCATED(InlineFormattingContext);
47public:
48 InlineFormattingContext(const ContainerBox& formattingContextRoot, InlineFormattingState&);
49 void layoutInFlowContent(InvalidationState&, const ConstraintsForInFlowContent&) override;
50 LayoutUnit usedContentHeight() const override;
51
52 const InlineFormattingState& formattingState() const { return downcast<InlineFormattingState>(FormattingContext::formattingState()); }
53 InlineFormattingState& formattingState() { return downcast<InlineFormattingState>(FormattingContext::formattingState()); }
54
55 void lineLayoutForIntergration(InvalidationState&, const ConstraintsForInFlowContent&);
56
57 const InlineFormattingGeometry& formattingGeometry() const final { return m_inlineFormattingGeometry; }
58 const InlineFormattingQuirks& formattingQuirks() const final { return m_inlineFormattingQuirks; }
59
60private:
61 IntrinsicWidthConstraints computedIntrinsicWidthConstraints() override;
62
63 void lineLayout(InlineItems&, LineBuilder::InlineItemRange, const ConstraintsForInFlowContent&);
64
65 void computeIntrinsicWidthForFormattingRoot(const Box&);
66 InlineLayoutUnit computedIntrinsicWidthForConstraint(InlineLayoutUnit availableWidth) const;
67
68 void computeHorizontalMargin(const Box&, const HorizontalConstraints&);
69 void computeHeightAndMargin(const Box&, const HorizontalConstraints&);
70 void computeWidthAndMargin(const Box&, const HorizontalConstraints&);
71
72 void collectInlineContentIfNeeded();
73 InlineRect computeGeometryForLineContent(const LineBuilder::LineContent&, const HorizontalConstraints&);
74 void invalidateFormattingState(const InvalidationState&);
75
76 const InlineFormattingGeometry m_inlineFormattingGeometry;
77 const InlineFormattingQuirks m_inlineFormattingQuirks;
78};
79
80}
81}
