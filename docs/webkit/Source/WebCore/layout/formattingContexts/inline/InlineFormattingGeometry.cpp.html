107static HangingTrailingWhitespaceContent collectHangingTrailingWhitespaceContent(const Line::RunList& runs, bool isLastLineWithInlineContent)
108{
109 auto hangingContent = HangingTrailingWhitespaceContent { };
110 if (isLastLineWithInlineContent)
111 hangingContent.setIsConditional();
112 for (auto& run : WTF::makeReversedRange(runs)) {
113 if (run.isInlineBoxStart() || run.isInlineBoxEnd())
114 continue;
115 if (run.isLineBreak()) {
116 hangingContent.setIsConditional();
117 continue;
118 }
119 if (!run.hasTrailingWhitespace())
120 break;
121 // Check if we have a preserved or hung whitespace.
122 if (run.style().whiteSpace() != WhiteSpace::PreWrap)
123 break;
124 // This is either a normal or conditionally hanging trailing whitespace.
125 hangingContent.expand(run.trailingWhitespaceWidth());
126 }
127 return hangingContent;
128}

130static std::optional<InlineLayoutUnit> horizontalAlignmentOffset(const Line::RunList& runs, TextAlignMode textAlign, InlineLayoutUnit lineLogicalWidth, InlineLayoutUnit contentLogicalWidth, bool isLastLine)
131{
132 auto availableWidth = lineLogicalWidth - contentLogicalWidth;
133 auto hangingTrailingWhitespaceContent = collectHangingTrailingWhitespaceContent(runs, isLastLine);
134 availableWidth += hangingTrailingWhitespaceContent.width();
135 if (availableWidth <= 0)
136 return { };
137
138 auto computedHorizontalAlignment = [&] {
139 if (textAlign != TextAlignMode::Justify)
140 return textAlign;
141 // Text is justified according to the method specified by the text-justify property,
142 // in order to exactly fill the line box. Unless otherwise specified by text-align-last,
143 // the last line before a forced break or the end of the block is start-aligned.
144 if (isLastLine || (!runs.isEmpty() && runs.last().isLineBreak()))
145 return TextAlignMode::Start;
146 return TextAlignMode::Justify;
147 };
148
149 switch (computedHorizontalAlignment()) {
150 case TextAlignMode::Left:
151 case TextAlignMode::WebKitLeft:
152 case TextAlignMode::Start:
153 return { };
154 case TextAlignMode::Right:
155 case TextAlignMode::WebKitRight:
156 case TextAlignMode::End:
157 return availableWidth;
158 case TextAlignMode::Center:
159 case TextAlignMode::WebKitCenter:
160 return availableWidth / 2;
161 case TextAlignMode::Justify:
162 // TextAlignMode::Justify is a run alignment (and we only do inline box alignment here)
163 return { };
164 default:
165 ASSERT_NOT_IMPLEMENTED_YET();
166 return { };
167 }
168 ASSERT_NOT_REACHED();
169 return { };
170}

177LineBox LineBoxBuilder::build(const LineBuilder::LineContent& lineContent)
178{
179 auto& runs = lineContent.runs;
180 auto lineLogicalWidth = lineContent.lineLogicalWidth;
181 auto contentLogicalWidth = lineContent.contentLogicalWidth;
182 auto horizontalAlignmentOffset = Layout::horizontalAlignmentOffset(runs, rootBox().style().textAlign(), lineLogicalWidth, contentLogicalWidth, lineContent.isLastLineWithInlineContent);
183 auto lineBox = LineBox { rootBox(), lineContent.logicalTopLeft, lineLogicalWidth, horizontalAlignmentOffset.value_or(InlineLayoutUnit { }), contentLogicalWidth, lineContent.nonSpanningInlineLevelBoxCount };
184 constructAndAlignInlineLevelBoxes(lineBox, runs);
185 return lineBox;
186}

188void LineBoxBuilder::setVerticalGeometryForInlineBox(LineBox::InlineLevelBox& inlineLevelBox) const
189{
190 ASSERT(inlineLevelBox.isInlineBox() || inlineLevelBox.isLineBreakBox());
191 auto& fontMetrics = inlineLevelBox.style().fontMetrics();
192 InlineLayoutUnit ascent = fontMetrics.ascent();
193 InlineLayoutUnit descent = fontMetrics.descent();
194 auto logicalHeight = ascent + descent;
195 // We need floor/ceil to match legacy layout integral positioning.
196 inlineLevelBox.setBaseline(floorf(ascent));
197 inlineLevelBox.setDescent(ceil(descent));
198 inlineLevelBox.setLogicalHeight(logicalHeight);
199
200 // FIXME: Adjust layout bounds with fallback font when applicable.
201 auto& style = inlineLevelBox.layoutBox().style();
202 auto lineHeight = style.lineHeight();
203 if (lineHeight.isNegative()) {
204 // If line-height computes to normal and either text-edge is leading or this is the root inline box,
205 // the font’s line gap metric may also be incorporated into A and D by adding half to each side as half-leading.
206 // https://www.w3.org/TR/css-inline-3/#inline-height
207 // Since text-edge is not supported yet and the initial value is leading, we should just apply it to
208 // all inline boxes.
209 auto halfLineGap = (fontMetrics.lineSpacing() - logicalHeight) / 2;
210 ascent += halfLineGap;
211 descent += halfLineGap;
212 } else {
213 InlineLayoutUnit lineHeight = style.computedLineHeight();
214 InlineLayoutUnit halfLeading = (lineHeight - (ascent + descent)) / 2;
215 ascent += halfLeading;
216 descent += halfLeading;
217 }
218 // We need floor/ceil to match legacy layout integral positioning.
219 inlineLevelBox.setLayoutBounds(LineBox::InlineLevelBox::LayoutBounds { floorf(ascent), ceil(descent) });
220}

222void LineBoxBuilder::constructAndAlignInlineLevelBoxes(LineBox& lineBox, const Line::RunList& runs)
223{
224 auto& rootInlineBox = lineBox.rootInlineBox();
225 setVerticalGeometryForInlineBox(rootInlineBox);
226
227 auto simplifiedVerticalAlignment = SimplifiedVerticalAlignment { rootInlineBox };
228 // FIXME: Add fast path support for line-height content.
229 simplifiedVerticalAlignment.setEnabled(layoutState().inStandardsMode() && rootBox().style().lineHeight().isNegative());
230
231 auto simplifiedAlignVerticallyIfApplicable = [&](auto& inlineLevelBox, std::optional<const BoxGeometry> boxGeometry) {
232 if (!simplifiedVerticalAlignment.isEnabled())
233 return;
234 if (!SimplifiedVerticalAlignment::canUseSimplifiedAlignment(rootInlineBox, inlineLevelBox, boxGeometry)) {
235 simplifiedVerticalAlignment.setEnabled(false);
236 return;
237 }
238 simplifiedVerticalAlignment.align(inlineLevelBox);
239 };
240
241 auto createWrappedInlineBoxes = [&] {
242 if (runs.isEmpty())
243 return;
244 // An inline box may not necessarily start on the current line:
245 // <span id=outer>line break<br>this content's parent inline box('outer') <span id=inner>starts on the previous line</span></span>
246 // We need to make sure that there's an LineBox::InlineLevelBox for every inline box that's present on the current line.
247 // In nesting case we need to create LineBox::InlineLevelBoxes for the inline box ancestors.
248 // We only have to do it on the first run as any subsequent inline content is either at the same/higher nesting level or
249 // nested with a [inline box start] run.
250 auto& firstRun = runs[0];
251 auto& firstRunParentLayoutBox = firstRun.layoutBox().parent();
252 // If the parent is the formatting root, we can stop here. This is root inline box content, there's no nesting inline box from the previous line(s)
253 // unless the inline box closing is forced over to the current line.
254 // e.g.
255 // <span>normally the inline box closing forms a continuous content</span>
256 // <span>unless it's forced to the next line<br></span>
257 auto firstRunNeedsInlineBox = firstRun.isInlineBoxEnd();
258 if (!firstRunNeedsInlineBox && isRootLayoutBox(firstRunParentLayoutBox))
259 return;
260 Vector<const Box*> layoutBoxesWithoutInlineBoxes;
261 if (firstRunNeedsInlineBox)
262 layoutBoxesWithoutInlineBoxes.append(&firstRun.layoutBox());
263 auto* ancestor = &firstRunParentLayoutBox;
264 while (!isRootLayoutBox(*ancestor)) {
265 layoutBoxesWithoutInlineBoxes.append(ancestor);
266 ancestor = &ancestor->parent();
267 }
268 // Construct the missing LineBox::InlineBoxes starting with the topmost layout box.
269 for (auto* layoutBox : WTF::makeReversedRange(layoutBoxesWithoutInlineBoxes)) {
270 auto inlineBox = LineBox::InlineLevelBox::createInlineBox(*layoutBox, rootInlineBox.logicalLeft(), lineBox.contentLogicalWidth());
271 setVerticalGeometryForInlineBox(inlineBox);
272 simplifiedAlignVerticallyIfApplicable(inlineBox, { });
273 lineBox.addInlineLevelBox(WTFMove(inlineBox));
274 }
275 };
276 createWrappedInlineBoxes();
277
278 auto lineHasContent = false;
279 for (auto& run : runs) {
280 auto& layoutBox = run.layoutBox();
281 auto runHasContent = [&] () -> bool {
282 ASSERT(!lineHasContent);
283 if (run.isText() || run.isBox() || run.isSoftLineBreak() || run.isHardLineBreak())
284 return true;
285 auto& inlineBoxGeometry = formattingContext().geometryForBox(layoutBox);
286 // Even negative horizontal margin makes the line "contentful".
287 if (run.isInlineBoxStart())
288 return inlineBoxGeometry.marginStart() || inlineBoxGeometry.borderLeft() || inlineBoxGeometry.paddingLeft().value_or(0_lu);
289 if (run.isInlineBoxEnd())
290 return inlineBoxGeometry.marginEnd() || inlineBoxGeometry.borderRight() || inlineBoxGeometry.paddingRight().value_or(0_lu);
291 if (run.isWordBreakOpportunity())
292 return false;
293 ASSERT_NOT_REACHED();
294 return true;
295 };
296 lineHasContent = lineHasContent || runHasContent();
297 auto logicalLeft = rootInlineBox.logicalLeft() + run.logicalLeft();
298 if (run.isBox()) {
299 auto& inlineLevelBoxGeometry = formattingContext().geometryForBox(layoutBox);
300 auto marginBoxHeight = inlineLevelBoxGeometry.marginBoxHeight();
301 auto ascent = InlineLayoutUnit { };
302 if (layoutState().shouldNotSynthesizeInlineBlockBaseline()) {
303 // Integration codepath constructs replaced boxes for inline-block content.
304 ASSERT(layoutBox.isReplacedBox());
305 ascent = *downcast<ReplacedBox>(layoutBox).baseline();
306 } else if (layoutBox.isInlineBlockBox()) {
307 // The baseline of an 'inline-block' is the baseline of its last line box in the normal flow, unless it has either no in-flow line boxes or
308 // if its 'overflow' property has a computed value other than 'visible', in which case the baseline is the bottom margin edge.
309 auto synthesizeBaseline = !layoutBox.establishesInlineFormattingContext() || !layoutBox.style().isOverflowVisible();
310 if (synthesizeBaseline)
311 ascent = marginBoxHeight;
312 else {
313 auto& formattingState = layoutState().establishedInlineFormattingState(downcast<ContainerBox>(layoutBox));
314 auto& lastLine = formattingState.lines().last();
315 auto inlineBlockBaseline = lastLine.lineBoxLogicalRect().top() + lastLine.baseline();
316 ascent = inlineLevelBoxGeometry.marginBefore() + inlineLevelBoxGeometry.borderTop() + inlineLevelBoxGeometry.paddingTop().value_or(0) + inlineBlockBaseline;
317 }
318 } else if (layoutBox.isReplacedBox())
319 ascent = downcast<ReplacedBox>(layoutBox).baseline().value_or(marginBoxHeight);
320 else
321 ascent = marginBoxHeight;
322 logicalLeft += std::max(0_lu, inlineLevelBoxGeometry.marginStart());
323 auto atomicInlineLevelBox = LineBox::InlineLevelBox::createAtomicInlineLevelBox(layoutBox, logicalLeft, { inlineLevelBoxGeometry.borderBoxWidth(), marginBoxHeight });
324 atomicInlineLevelBox.setBaseline(ascent);
325 atomicInlineLevelBox.setLayoutBounds(LineBox::InlineLevelBox::LayoutBounds { ascent, marginBoxHeight - ascent });
326 simplifiedAlignVerticallyIfApplicable(atomicInlineLevelBox, inlineLevelBoxGeometry);
327 lineBox.addInlineLevelBox(WTFMove(atomicInlineLevelBox));
328 continue;
329 }
330 if (run.isInlineBoxStart()) {
331 // At this point we don't know yet how wide this inline box is. Let's assume it's as long as the line is
332 // and adjust it later if we come across an inlineBoxEnd run (see below).
333 auto initialLogicalWidth = lineBox.contentLogicalWidth() - run.logicalLeft();
334 ASSERT(initialLogicalWidth >= 0);
335 // Inline box run is based on margin box. Let's convert it to border box.
336 auto marginStart = std::max(0_lu, formattingContext().geometryForBox(layoutBox).marginStart());
337 logicalLeft += marginStart;
338 initialLogicalWidth -= marginStart;
339 auto inlineBox = LineBox::InlineLevelBox::createInlineBox(layoutBox, logicalLeft, initialLogicalWidth);
340 setVerticalGeometryForInlineBox(inlineBox);
341 simplifiedAlignVerticallyIfApplicable(inlineBox, { });
342 lineBox.addInlineLevelBox(WTFMove(inlineBox));
343 continue;
344 }
345 if (run.isInlineBoxEnd()) {
346 // Adjust the logical width when the inline box closes on this line.
347 auto& inlineBox = lineBox.inlineLevelBoxForLayoutBox(layoutBox);
348 ASSERT(inlineBox.isInlineBox());
349 // Inline box run is based on margin box. Let's convert it to border box.
350 auto marginEnd = std::max(0_lu, formattingContext().geometryForBox(layoutBox).marginEnd());
351 auto inlineBoxLogicalRight = logicalLeft + run.logicalWidth() - marginEnd;
352 inlineBox.setLogicalWidth(inlineBoxLogicalRight - inlineBox.logicalLeft());
353 simplifiedAlignVerticallyIfApplicable(inlineBox, { });
354 continue;
355 }
356 if (run.isText() || run.isSoftLineBreak()) {
357 // FIXME: Adjust non-empty inline box height when glyphs from the non-primary font stretch the box.
358 lineBox.inlineLevelBoxForLayoutBox(layoutBox.parent()).setHasContent();
359 continue;
360 }
361 if (run.isHardLineBreak()) {
362 auto lineBreakBox = LineBox::InlineLevelBox::createLineBreakBox(layoutBox, logicalLeft);
363 setVerticalGeometryForInlineBox(lineBreakBox);
364 simplifiedAlignVerticallyIfApplicable(lineBreakBox, { });
365 lineBox.addInlineLevelBox(WTFMove(lineBreakBox));
366 continue;
367 }
368 if (run.isWordBreakOpportunity()) {
369 lineBox.addInlineLevelBox(LineBox::InlineLevelBox::createGenericInlineLevelBox(layoutBox, logicalLeft));
370 continue;
371 }
372 ASSERT_NOT_REACHED();
373 }
374
375 lineBox.setHasContent(lineHasContent);
376 if (simplifiedVerticalAlignment.isEnabled() || !lineHasContent) {
377 // We should always be able to exercise the fast path when the line has no content at all, even in non-standards mode or with line-height set.
378 rootInlineBox.setLogicalTop(lineHasContent ? simplifiedVerticalAlignment.rootInlineBoxLogicalTop() : -rootInlineBox.baseline());
379 lineBox.setLogicalHeight(lineHasContent ? simplifiedVerticalAlignment.lineBoxHeight() : InlineLayoutUnit());
380 } else
381 computeLineBoxHeightAndAlignInlineLevelBoxesVertically(lineBox);
382}

384void LineBoxBuilder::computeLineBoxHeightAndAlignInlineLevelBoxesVertically(LineBox& lineBox)
385{
386 ASSERT(lineBox.hasContent());
387 // This function (partially) implements:
388 // 2.2. Layout Within Line Boxes
389 // https://www.w3.org/TR/css-inline-3/#line-layout
390 // 1. Compute the line box height using the layout bounds geometry. This height computation strictly uses layout bounds and not normal inline level box geometries.
391 // 2. Compute the baseline/logical top position of the root inline box. Aligned boxes push the root inline box around inside the line box.
392 // 3. Finally align the inline level boxes using (mostly) normal inline level box geometries.
393 auto& rootInlineBox = lineBox.rootInlineBox();
394 auto& formattingGeometry = formattingContext().formattingGeometry();
395
396 auto computeLineBoxLogicalHeight = [&] {
397 // Line box height computation is based on the layout bounds of the inline boxes and not their logical (ascent/descent) dimensions.
398 struct AbsoluteTopAndBottom {
399 InlineLayoutUnit top { 0 };
400 InlineLayoutUnit bottom { 0 };
401 };
402 HashMap<const LineBox::InlineLevelBox*, AbsoluteTopAndBottom> inlineLevelBoxAbsoluteTopAndBottomMap;
403
404 auto minimumLogicalTop = std::optional<InlineLayoutUnit> { };
405 auto maximumLogicalBottom = std::optional<InlineLayoutUnit> { };
406 if (formattingGeometry.inlineLevelBoxAffectsLineBox(rootInlineBox, lineBox)) {
407 minimumLogicalTop = InlineLayoutUnit { };
408 maximumLogicalBottom = rootInlineBox.layoutBounds().height();
409 inlineLevelBoxAbsoluteTopAndBottomMap.add(&rootInlineBox, AbsoluteTopAndBottom { *minimumLogicalTop, *maximumLogicalBottom });
410 } else
411 inlineLevelBoxAbsoluteTopAndBottomMap.add(&rootInlineBox, AbsoluteTopAndBottom { });
412
413 Vector<LineBox::InlineLevelBox*> lineBoxRelativeInlineLevelBoxes;
414 for (auto& inlineLevelBox : lineBox.nonRootInlineLevelBoxes()) {
415 auto& layoutBox = inlineLevelBox.layoutBox();
416 if (inlineLevelBox.hasLineBoxRelativeAlignment()) {
417 lineBoxRelativeInlineLevelBoxes.append(&inlineLevelBox);
418 continue;
419 }
420 auto& parentInlineBox = lineBox.inlineLevelBoxForLayoutBox(layoutBox.parent());
421 // Logical top is relative to the parent inline box's layout bounds.
422 // Note that this logical top is not the final logical top of the inline level box.
423 // This is the logical top in the context of the layout bounds geometry which may be very different from the inline box's normal geometry.
424 auto logicalTop = InlineLayoutUnit { };
425 switch (inlineLevelBox.verticalAlign()) {
426 case VerticalAlign::Baseline: {
427 auto logicalTopOffsetFromParentBaseline = inlineLevelBox.layoutBounds().ascent;
428 logicalTop = parentInlineBox.layoutBounds().ascent - logicalTopOffsetFromParentBaseline;
429 break;
430 }
431 case VerticalAlign::Middle: {
432 auto logicalTopOffsetFromParentBaseline = inlineLevelBox.layoutBounds().height() / 2 + parentInlineBox.style().fontMetrics().xHeight() / 2;
433 logicalTop = parentInlineBox.layoutBounds().ascent - logicalTopOffsetFromParentBaseline;
434 break;
435 }
436 case VerticalAlign::BaselineMiddle: {
437 auto logicalTopOffsetFromParentBaseline = inlineLevelBox.layoutBounds().height() / 2;
438 logicalTop = parentInlineBox.layoutBounds().ascent - logicalTopOffsetFromParentBaseline;
439 break;
440 }
441 case VerticalAlign::Length: {
442 auto& style = inlineLevelBox.style();
443 auto logicalTopOffsetFromParentBaseline = floatValueForLength(style.verticalAlignLength(), style.computedLineHeight()) + inlineLevelBox.layoutBounds().ascent;
444 logicalTop = parentInlineBox.layoutBounds().ascent - logicalTopOffsetFromParentBaseline;
445 break;
446 }
447 case VerticalAlign::TextTop: {
448 // Note that text-top aligns with the inline box's font metrics top (ascent) and not the layout bounds top.
449 logicalTop = parentInlineBox.layoutBounds().ascent - parentInlineBox.baseline();
450 break;
451 }
452 case VerticalAlign::TextBottom: {
453 // Note that text-bottom aligns with the inline box's font metrics bottom (descent) and not the layout bounds bottom.
454 auto parentInlineBoxLayoutBounds = parentInlineBox.layoutBounds();
455 auto parentInlineBoxLogicalBottom = parentInlineBoxLayoutBounds.height() - parentInlineBoxLayoutBounds.descent + parentInlineBox.descent().value_or(InlineLayoutUnit());
456 logicalTop = parentInlineBoxLogicalBottom - inlineLevelBox.layoutBounds().height();
457 break;
458 }
459 default:
460 ASSERT_NOT_IMPLEMENTED_YET();
461 break;
462 }
463 auto parentInlineBoxAbsoluteTopAndBottom = inlineLevelBoxAbsoluteTopAndBottomMap.get(&parentInlineBox);
464 auto absoluteLogicalTop = parentInlineBoxAbsoluteTopAndBottom.top + logicalTop;
465 auto absoluteLogicalBottom = absoluteLogicalTop + inlineLevelBox.layoutBounds().height();
466 inlineLevelBoxAbsoluteTopAndBottomMap.add(&inlineLevelBox, AbsoluteTopAndBottom { absoluteLogicalTop, absoluteLogicalBottom });
467 // Stretch the min/max absolute values if applicable.
468 if (formattingGeometry.inlineLevelBoxAffectsLineBox(inlineLevelBox, lineBox)) {
469 minimumLogicalTop = std::min(minimumLogicalTop.value_or(absoluteLogicalTop), absoluteLogicalTop);
470 maximumLogicalBottom = std::max(maximumLogicalBottom.value_or(absoluteLogicalBottom), absoluteLogicalBottom);
471 }
472 }
473 // The line box height computation is as follows:
474 // 1. Stretch the line box with the non-line-box relative aligned inline box absolute top and bottom values.
475 // 2. Check if the line box relative aligned inline boxes (top, bottom etc) have enough room and stretch the line box further if needed.
476 auto lineBoxLogicalHeight = maximumLogicalBottom.value_or(InlineLayoutUnit()) - minimumLogicalTop.value_or(InlineLayoutUnit());
477 for (auto* lineBoxRelativeInlineLevelBox : lineBoxRelativeInlineLevelBoxes) {
478 if (!formattingGeometry.inlineLevelBoxAffectsLineBox(*lineBoxRelativeInlineLevelBox, lineBox))
479 continue;
480 lineBoxLogicalHeight = std::max(lineBoxLogicalHeight, lineBoxRelativeInlineLevelBox->layoutBounds().height());
481 }
482 lineBox.setLogicalHeight(lineBoxLogicalHeight);
483 };
484 computeLineBoxLogicalHeight();
485
486 auto computeRootInlineBoxVerticalPosition = [&] {
487 HashMap<const LineBox::InlineLevelBox*, InlineLayoutUnit> inlineLevelBoxAbsoluteBaselineOffsetMap;
488 inlineLevelBoxAbsoluteBaselineOffsetMap.add(&rootInlineBox, InlineLayoutUnit { });
489
490 auto maximumTopOffsetFromRootInlineBoxBaseline = std::optional<InlineLayoutUnit> { };
491 if (formattingGeometry.inlineLevelBoxAffectsLineBox(rootInlineBox, lineBox))
492 maximumTopOffsetFromRootInlineBoxBaseline = rootInlineBox.layoutBounds().ascent;
493
494 for (auto& inlineLevelBox : lineBox.nonRootInlineLevelBoxes()) {
495 auto absoluteBaselineOffset = InlineLayoutUnit { };
496 if (!inlineLevelBox.hasLineBoxRelativeAlignment()) {
497 auto& layoutBox = inlineLevelBox.layoutBox();
498 auto& parentInlineBox = lineBox.inlineLevelBoxForLayoutBox(layoutBox.parent());
499 auto baselineOffsetFromParentBaseline = InlineLayoutUnit { };
500
501 switch (inlineLevelBox.verticalAlign()) {
502 case VerticalAlign::Baseline:
503 baselineOffsetFromParentBaseline = { };
504 break;
505 case VerticalAlign::Middle: {
506 auto logicalTopOffsetFromParentBaseline = (inlineLevelBox.layoutBounds().height() / 2 + parentInlineBox.style().fontMetrics().xHeight() / 2);
507 baselineOffsetFromParentBaseline = logicalTopOffsetFromParentBaseline - inlineLevelBox.layoutBounds().ascent;
508 break;
509 }
510 case VerticalAlign::BaselineMiddle: {
511 auto logicalTopOffsetFromParentBaseline = inlineLevelBox.layoutBounds().height() / 2;
512 baselineOffsetFromParentBaseline = logicalTopOffsetFromParentBaseline - inlineLevelBox.layoutBounds().ascent;
513 break;
514 }
515 case VerticalAlign::Length: {
516 auto& style = inlineLevelBox.style();
517 auto verticalAlignOffset = floatValueForLength(style.verticalAlignLength(), style.computedLineHeight());
518 auto logicalTopOffsetFromParentBaseline = verticalAlignOffset + inlineLevelBox.baseline();
519 baselineOffsetFromParentBaseline = logicalTopOffsetFromParentBaseline - inlineLevelBox.baseline();
520 break;
521 }
522 case VerticalAlign::TextTop:
523 baselineOffsetFromParentBaseline = parentInlineBox.baseline() - inlineLevelBox.layoutBounds().ascent;
524 break;
525 case VerticalAlign::TextBottom:
526 baselineOffsetFromParentBaseline = inlineLevelBox.layoutBounds().descent - *parentInlineBox.descent();
527 break;
528 default:
529 ASSERT_NOT_IMPLEMENTED_YET();
530 break;
531 }
532 absoluteBaselineOffset = inlineLevelBoxAbsoluteBaselineOffsetMap.get(&parentInlineBox) + baselineOffsetFromParentBaseline;
533 } else {
534 switch (inlineLevelBox.verticalAlign()) {
535 case VerticalAlign::Top: {
536 absoluteBaselineOffset = rootInlineBox.layoutBounds().ascent - inlineLevelBox.layoutBounds().ascent;
537 break;
538 }
539 case VerticalAlign::Bottom: {
540 absoluteBaselineOffset = inlineLevelBox.layoutBounds().descent - rootInlineBox.layoutBounds().descent;
541 break;
542 }
543 default:
544 ASSERT_NOT_IMPLEMENTED_YET();
545 break;
546 }
547 }
548 inlineLevelBoxAbsoluteBaselineOffsetMap.add(&inlineLevelBox, absoluteBaselineOffset);
549 auto affectsRootInlineBoxVerticalPosition = formattingGeometry.inlineLevelBoxAffectsLineBox(inlineLevelBox, lineBox);
550 if (affectsRootInlineBoxVerticalPosition) {
551 auto topOffsetFromRootInlineBoxBaseline = absoluteBaselineOffset + inlineLevelBox.layoutBounds().ascent;
552 if (maximumTopOffsetFromRootInlineBoxBaseline)
553 maximumTopOffsetFromRootInlineBoxBaseline = std::max(*maximumTopOffsetFromRootInlineBoxBaseline, topOffsetFromRootInlineBoxBaseline);
554 else {
555 // We are is quirk mode and the root inline box has no content. The root inline box's baseline is anchored at 0.
556 // However negative ascent (e.g negative top margin) can "push" the root inline box upwards and have a negative value.
557 maximumTopOffsetFromRootInlineBoxBaseline = inlineLevelBox.layoutBounds().ascent >= 0
558 ? std::max(0.0f, topOffsetFromRootInlineBoxBaseline)
559 : topOffsetFromRootInlineBoxBaseline;
560 }
561 }
562 }
563 auto rootInlineBoxLogicalTop = maximumTopOffsetFromRootInlineBoxBaseline.value_or(0.f) - rootInlineBox.baseline();
564 rootInlineBox.setLogicalTop(rootInlineBoxLogicalTop);
565 };
566 computeRootInlineBoxVerticalPosition();
567
568 auto alignInlineLevelBoxes = [&] {
569 for (auto& inlineLevelBox : lineBox.nonRootInlineLevelBoxes()) {
570 auto& layoutBox = inlineLevelBox.layoutBox();
571 auto& parentInlineBox = lineBox.inlineLevelBoxForLayoutBox(layoutBox.parent());
572 auto logicalTop = InlineLayoutUnit { };
573 switch (inlineLevelBox.verticalAlign()) {
574 case VerticalAlign::Baseline:
575 logicalTop = parentInlineBox.baseline() - inlineLevelBox.baseline();
576 break;
577 case VerticalAlign::Middle: {
578 auto logicalTopOffsetFromParentBaseline = (inlineLevelBox.logicalHeight() / 2 + parentInlineBox.style().fontMetrics().xHeight() / 2);
579 logicalTop = parentInlineBox.baseline() - logicalTopOffsetFromParentBaseline;
580 break;
581 }
582 case VerticalAlign::BaselineMiddle: {
583 auto logicalTopOffsetFromParentBaseline = inlineLevelBox.logicalHeight() / 2;
584 logicalTop = parentInlineBox.baseline() - logicalTopOffsetFromParentBaseline;
585 break;
586 }
587 case VerticalAlign::Length: {
588 auto& style = inlineLevelBox.style();
589 auto verticalAlignOffset = floatValueForLength(style.verticalAlignLength(), style.computedLineHeight());
590 auto logicalTopOffsetFromParentBaseline = verticalAlignOffset + inlineLevelBox.baseline();
591 logicalTop = parentInlineBox.baseline() - logicalTopOffsetFromParentBaseline;
592 break;
593 }
594 // Note that (text)top/bottom align their layout bounds.
595 case VerticalAlign::TextTop:
596 logicalTop = inlineLevelBox.layoutBounds().ascent - inlineLevelBox.baseline();
597 break;
598 case VerticalAlign::TextBottom:
599 logicalTop = parentInlineBox.logicalHeight() - inlineLevelBox.layoutBounds().descent - inlineLevelBox.baseline();
600 break;
601 case VerticalAlign::Top:
602 // Note that this logical top is not relative to the parent inline box.
603 logicalTop = inlineLevelBox.layoutBounds().ascent - inlineLevelBox.baseline();
604 break;
605 case VerticalAlign::Bottom:
606 // Note that this logical top is not relative to the parent inline box.
607 logicalTop = lineBox.logicalHeight() - inlineLevelBox.layoutBounds().descent - inlineLevelBox.baseline();
608 break;
609 default:
610 ASSERT_NOT_IMPLEMENTED_YET();
611 break;
612 }
613 inlineLevelBox.setLogicalTop(logicalTop);
614 }
615 };
616 alignInlineLevelBoxes();
617}

625bool LineBoxBuilder::SimplifiedVerticalAlignment::canUseSimplifiedAlignment(const LineBox::InlineLevelBox& rootInlineBox, const LineBox::InlineLevelBox& inlineLevelBox, std::optional<const BoxGeometry> inlineLevelBoxGeometry)
626{
627 if (inlineLevelBox.isAtomicInlineLevelBox()) {
628 ASSERT(inlineLevelBoxGeometry);
629 // Baseline aligned, non-stretchy direct children are considered to be simple for now.
630 auto& layoutBox = inlineLevelBox.layoutBox();
631 return &layoutBox.parent() == &rootInlineBox.layoutBox()
632 && layoutBox.style().verticalAlign() == VerticalAlign::Baseline
633 && !inlineLevelBoxGeometry->marginBefore()
634 && !inlineLevelBoxGeometry->marginAfter()
635 && inlineLevelBoxGeometry->marginBoxHeight() <= rootInlineBox.baseline();
636 }
637 if (inlineLevelBox.isLineBreakBox()) {
638 // Baseline aligned, non-stretchy line breaks e.g. <div><span><br></span></div> but not <div><span style="font-size: 100px;"><br></span></div>.
639 return inlineLevelBox.layoutBox().style().verticalAlign() == VerticalAlign::Baseline && inlineLevelBox.baseline() <= rootInlineBox.baseline();
640 }
641 if (inlineLevelBox.isInlineBox()) {
642 // Baseline aligned, non-stretchy inline boxes e.g. <div><span></span></div> but not <div><span style="font-size: 100px;"></span></div>.
643 return inlineLevelBox.layoutBox().style().verticalAlign() == VerticalAlign::Baseline && inlineLevelBox.layoutBounds() == rootInlineBox.layoutBounds();
644 }
645 return false;
646}

648void LineBoxBuilder::SimplifiedVerticalAlignment::align(LineBox::InlineLevelBox& inlineLevelBox)
649{
650 if (inlineLevelBox.isAtomicInlineLevelBox() || inlineLevelBox.isLineBreakBox() || inlineLevelBox.isInlineBox()) {
651 // Only baseline alignment for now.
652 inlineLevelBox.setLogicalTop(m_rootInlineBox.baseline() - inlineLevelBox.baseline());
653 adjust(inlineLevelBox);
654 return;
655 }
656 ASSERT_NOT_IMPLEMENTED_YET();

659void LineBoxBuilder::SimplifiedVerticalAlignment::adjust(const LineBox::InlineLevelBox& inlineLevelBox)
660{
661 auto layoutBoundsLogicalTop = m_rootInlineBox.layoutBounds().ascent - inlineLevelBox.layoutBounds().ascent;
662 m_lineBoxLogicalTop = std::min(m_lineBoxLogicalTop, layoutBoundsLogicalTop);
663 m_lineBoxLogicalBottom = std::max(m_lineBoxLogicalBottom, layoutBoundsLogicalTop + inlineLevelBox.layoutBounds().height());
664 m_rootInlineBoxLogicalTop = std::max(m_rootInlineBoxLogicalTop, inlineLevelBox.layoutBounds().ascent - m_rootInlineBox.baseline());
665}

677InlineLayoutUnit InlineFormattingGeometry::logicalTopForNextLine(const LineBuilder::LineContent& lineContent, InlineLayoutUnit previousLineLogicalBottom, const FloatingContext& floatingContext) const
678{
679 // Normally the next line's logical top is the previous line's logical bottom, but when the line ends
680 // with the clear property set, the next line needs to clear the existing floats.
681 if (lineContent.runs.isEmpty())
682 return previousLineLogicalBottom;
683 auto& lastRunLayoutBox = lineContent.runs.last().layoutBox();
684 if (lastRunLayoutBox.style().clear() == Clear::None)
685 return previousLineLogicalBottom;
686 auto positionWithClearance = floatingContext.verticalPositionWithClearance(lastRunLayoutBox);
687 if (!positionWithClearance)
688 return previousLineLogicalBottom;
689 return std::max(previousLineLogicalBottom, InlineLayoutUnit(positionWithClearance->position));
690}

692ContentWidthAndMargin InlineFormattingGeometry::inlineBlockContentWidthAndMargin(const Box& formattingContextRoot, const HorizontalConstraints& horizontalConstraints, const OverriddenHorizontalValues& overriddenHorizontalValues) const
693{
694 ASSERT(formattingContextRoot.isInFlow());
695
696 // 10.3.10 'Inline-block', replaced elements in normal flow
697
698 // Exactly as inline replaced elements.
699 if (formattingContextRoot.isReplacedBox())
700 return inlineReplacedContentWidthAndMargin(downcast<ReplacedBox>(formattingContextRoot), horizontalConstraints, { }, overriddenHorizontalValues);
701
702 // 10.3.9 'Inline-block', non-replaced elements in normal flow
703
704 // If 'width' is 'auto', the used value is the shrink-to-fit width as for floating elements.
705 // A computed value of 'auto' for 'margin-left' or 'margin-right' becomes a used value of '0'.
706 // #1
707 auto width = computedValue(formattingContextRoot.style().logicalWidth(), horizontalConstraints.logicalWidth);
708 if (!width)
709 width = shrinkToFitWidth(formattingContextRoot, horizontalConstraints.logicalWidth);
710
711 // #2
712 auto computedHorizontalMargin = FormattingGeometry::computedHorizontalMargin(formattingContextRoot, horizontalConstraints);
713
714 return ContentWidthAndMargin { *width, { computedHorizontalMargin.start.value_or(0_lu), computedHorizontalMargin.end.value_or(0_lu) } };
715}

717ContentHeightAndMargin InlineFormattingGeometry::inlineBlockContentHeightAndMargin(const Box& layoutBox, const HorizontalConstraints& horizontalConstraints, const OverriddenVerticalValues& overriddenVerticalValues) const
718{
719 ASSERT(layoutBox.isInFlow());
720
721 // 10.6.2 Inline replaced elements, block-level replaced elements in normal flow, 'inline-block' replaced elements in normal flow and floating replaced elements
722 if (layoutBox.isReplacedBox())
723 return inlineReplacedContentHeightAndMargin(downcast<ReplacedBox>(layoutBox), horizontalConstraints, { }, overriddenVerticalValues);
724
725 // 10.6.6 Complicated cases
726 // - 'Inline-block', non-replaced elements.
727 return complicatedCases(layoutBox, horizontalConstraints, overriddenVerticalValues);
728}

730bool InlineFormattingGeometry::inlineLevelBoxAffectsLineBox(const LineBox::InlineLevelBox& inlineLevelBox, const LineBox& lineBox) const
731{
732 if (inlineLevelBox.isInlineBox() || inlineLevelBox.isLineBreakBox())
733 return layoutState().inStandardsMode() ? true : formattingContext().formattingQuirks().inlineLevelBoxAffectsLineBox(inlineLevelBox, lineBox);
734 if (inlineLevelBox.isAtomicInlineLevelBox()) {
735 if (inlineLevelBox.layoutBounds().height())
736 return true;
737 // While in practice when the negative vertical margin makes the layout bounds empty (e.g: height: 100px; margin-top: -100px;), and this inline
738 // level box contributes 0px to the line box height, it still needs to be taken into account while computing line box geometries.
739 auto& boxGeometry = formattingContext().geometryForBox(inlineLevelBox.layoutBox());
740 return boxGeometry.marginBefore() || boxGeometry.marginAfter();
741 }
742 return false;
743}
