42InlineLayoutUnit InlineFormattingQuirks::initialLineHeight() const
43{
44 ASSERT(!layoutState().inStandardsMode());
45 // Negative lineHeight value means the line-height is not set
46 auto& rootStyle = formattingContext().root().style();
47 return rootStyle.lineHeight().isNegative() ? rootStyle.fontMetrics().floatHeight() : rootStyle.computedLineHeight();
48}

50bool InlineFormattingQuirks::inlineLevelBoxAffectsLineBox(const LineBox::InlineLevelBox& inlineLevelBox, const LineBox& lineBox) const
51{
52 ASSERT(!layoutState().inStandardsMode());
53 if (inlineLevelBox.isLineBreakBox()) {
54 // In quirks mode linebreak boxes (<br>) stop affecting the line box when (assume <br> is nested e.g. <span style="font-size: 100px"><br></span>)
55 // 1. the root inline box has content <div>content<br>/div>
56 // 2. there's at least one atomic inline level box on the line e.g <div><img><br></div> or <div><span><img></span><br></div>
57 // 3. there's at least one inline box with content e.g. <div><span>content</span><br></div>
58 if (lineBox.rootInlineBox().hasContent())
59 return false;
60 if (lineBox.hasAtomicInlineLevelBox())
61 return false;
62 // At this point we either have only the <br> on the line or inline boxes with or without content.
63 auto& inlineLevelBoxes = lineBox.nonRootInlineLevelBoxes();
64 ASSERT(!inlineLevelBoxes.isEmpty());
65 if (inlineLevelBoxes.size() == 1)
66 return true;
67 for (auto& inlineLevelBox : lineBox.nonRootInlineLevelBoxes()) {
68 // Filter out empty inline boxes e.g. <div><span></span><span></span><br></div>
69 if (inlineLevelBox.isInlineBox() && inlineLevelBox.hasContent())
70 return false;
71 }
72 return true;
73 }
74 if (inlineLevelBox.isInlineBox()) {
75 // Inline boxes (e.g. root inline box or <span>) affects line boxes either through the strut or actual content.
76 if (inlineLevelBox.hasContent())
77 return true;
78 if (inlineLevelBox.isRootInlineBox()) {
79 // This root inline box has no direct text content and we are in non-standards mode.
80 // Now according to legacy line layout, we need to apply the following list-item specific quirk:
81 // We do not create markers for list items when the list-style-type is none, while other browsers do.
82 // The side effect of having no marker is that in quirks mode we have to specifically check for list-item
83 // and make sure it is treated as if it had content and stretched the line.
84 // see LegacyInlineFlowBox c'tor.
85 return inlineLevelBox.layoutBox().style().isOriginalDisplayListItemType();
86 }
87 // Non-root inline boxes (e.g. <span>).
88 auto& boxGeometry = formattingContext().geometryForBox(inlineLevelBox.layoutBox());
89 if (boxGeometry.horizontalBorder() || boxGeometry.horizontalPadding().value_or(0_lu)) {
90 // Horizontal border and padding make the inline box stretch the line (e.g. <span style="padding: 10px;"></span>).
91 return true;
92 }
93 return false;
94 }
95 ASSERT_NOT_REACHED();
96 return false;
97}

99bool InlineFormattingQuirks::hasSoftWrapOpportunityAtImage() const
100{
101 ASSERT(layoutState().inQuirksMode());
102 return !formattingContext().root().isTableCell();
103}
