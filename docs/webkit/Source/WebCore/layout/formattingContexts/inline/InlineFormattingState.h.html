30#include "FormattingState.h"
31#include "InlineItem.h"
32#include "InlineLineBox.h"
33#include "InlineLineGeometry.h"
34#include "InlineLineRun.h"
35#include <wtf/IsoMalloc.h>
36
37namespace WebCore {
38namespace Layout {
39
40using InlineItems = Vector<InlineItem>;
41using InlineLines = Vector<InlineLineGeometry>;
42using InlineLineBoxes = Vector<LineBox>;
43using InlineLineRuns = Vector<LineRun>;
44
45// InlineFormattingState holds the state for a particular inline formatting context tree.
46class InlineFormattingState : public FormattingState {
47 WTF_MAKE_ISO_ALLOCATED(InlineFormattingState);
48public:
49 InlineFormattingState(Ref<FloatingState>&&, LayoutState&);
50 ~InlineFormattingState();
51
52 InlineItems& inlineItems() { return m_inlineItems; }
53 const InlineItems& inlineItems() const { return m_inlineItems; }
54 void addInlineItem(InlineItem&& inlineItem) { m_inlineItems.append(WTFMove(inlineItem)); }
55
56 const InlineLines& lines() const { return m_lines; }
57 InlineLines& lines() { return m_lines; }
58 void addLine(const InlineLineGeometry& line) { m_lines.append(line); }
59
60 const InlineLineBoxes& lineBoxes() const { return m_lineBoxes; }
61 void addLineBox(LineBox&& lineBox) { m_lineBoxes.append(WTFMove(lineBox)); }
62
63 const InlineLineRuns& lineRuns() const { return m_lineRuns; }
64 InlineLineRuns& lineRuns() { return m_lineRuns; }
65 void addLineRun(LineRun&& run) { m_lineRuns.append(WTFMove(run)); }
66
67 void setClearGapAfterLastLine(InlineLayoutUnit verticalGap);
68 InlineLayoutUnit clearGapAfterLastLine() const { return m_clearGapAfterLastLine; }
69
70 void clearLineAndRuns();
71 void shrinkToFit();
72
73private:
74 // Cacheable input to line layout.
75 InlineItems m_inlineItems;
76 InlineLines m_lines;
77 InlineLineBoxes m_lineBoxes;
78 InlineLineRuns m_lineRuns;
79 InlineLayoutUnit m_clearGapAfterLastLine { 0 };
80};
81
82inline void InlineFormattingState::setClearGapAfterLastLine(InlineLayoutUnit verticalGap)
83{
84 ASSERT(verticalGap >= 0);
85 m_clearGapAfterLastLine = verticalGap;
86}
87
88inline void InlineFormattingState::clearLineAndRuns()
89{
90 m_lines.clear();
91 m_lineBoxes.clear();
92 m_lineRuns.clear();
93 m_clearGapAfterLastLine = { };
94}
95
96inline void InlineFormattingState::shrinkToFit()
97{
98 m_inlineItems.shrinkToFit();
99 m_lines.shrinkToFit();
100 m_lineBoxes.shrinkToFit();
101 m_lineRuns.shrinkToFit();
102}
103
104}
105}
