68void Line::applyRunExpansion(InlineLayoutUnit extraHorizontalSpace)
69{
70 ASSERT(formattingContext().root().style().textAlign() == TextAlignMode::Justify);
71 // Text is justified according to the method specified by the text-justify property,
72 // in order to exactly fill the line box. Unless otherwise specified by text-align-last,
73 // the last line before a forced break or the end of the block is start-aligned.
74 if (m_runs.isEmpty() || m_runs.last().isLineBreak())
75 return;
76
77 auto expansionOpportunityCount = 0;
78 Run* lastRunWithContent = nullptr;
79 // Collect and distribute the expansion opportunities.
80 for (auto& run : m_runs) {
81 expansionOpportunityCount += run.expansionOpportunityCount();
82 if (run.isText() || run.isBox())
83 lastRunWithContent = &run;
84 }
85 // Need to fix up the last run's trailing expansion.
86 if (lastRunWithContent && lastRunWithContent->hasExpansionOpportunity()) {
87 // Turn off the trailing bits first and add the forbid trailing expansion.
88 auto leadingExpansion = lastRunWithContent->expansionBehavior() & LeftExpansionMask;
89 lastRunWithContent->setExpansionBehavior(leadingExpansion | ForbidRightExpansion);
90 }
91 // Anything to distribute?
92 if (!expansionOpportunityCount || !extraHorizontalSpace)
93 return;
94 // Distribute the extra space.
95 auto expansionToDistribute = extraHorizontalSpace / expansionOpportunityCount;
96 auto accumulatedExpansion = InlineLayoutUnit { };
97 for (auto& run : m_runs) {
98 // Expand and move runs by the accumulated expansion.
99 run.moveHorizontally(accumulatedExpansion);
100 if (!run.hasExpansionOpportunity())
101 continue;
102 ASSERT(run.expansionOpportunityCount());
103 auto computedExpansion = expansionToDistribute * run.expansionOpportunityCount();
104 // FIXME: Check why we need to set both.
105 run.setHorizontalExpansion(computedExpansion);
106 run.shrinkHorizontally(-computedExpansion);
107 accumulatedExpansion += computedExpansion;
108 }
109 // Content grows as runs expand.
110 m_contentLogicalWidth += accumulatedExpansion;
111}

113void Line::removeTrailingTrimmableContent()
114{
115 if (m_trimmableTrailingContent.isEmpty() || m_runs.isEmpty())
116 return;
117
118 // Complex line layout quirk: keep the trailing whitespace around when it is followed by a line break, unless the content overflows the line.
119 if (RuntimeEnabledFeatures::sharedFeatures().layoutFormattingContextIntegrationEnabled()) {
120 auto isTextAlignRight = [&] {
121 auto textAlign = formattingContext().root().style().textAlign();
122 return textAlign == TextAlignMode::Right
123 || textAlign == TextAlignMode::WebKitRight
124 || textAlign == TextAlignMode::End;
125 }();
126
127 if (m_runs.last().isLineBreak() && !isTextAlignRight) {
128 m_trimmableTrailingContent.reset();
129 return;
130 }
131 }
132
133 m_contentLogicalWidth -= m_trimmableTrailingContent.remove();
134}

136void Line::visuallyCollapsePreWrapOverflowContent(InlineLayoutUnit extraHorizontalSpace)
137{
138 ASSERT(m_trimmableTrailingContent.isEmpty());
139 // If white-space is set to pre-wrap, the UA must
140 // ...
141 // It may also visually collapse the character advance widths of any that would otherwise overflow.
142 auto overflowWidth = -extraHorizontalSpace;
143 if (overflowWidth <= 0)
144 return;
145 // Let's just find the trailing pre-wrap whitespace content for now (e.g check if there are multiple trailing runs with
146 // different set of white-space values and decide if the in-between pre-wrap content should be collapsed as well.)
147 InlineLayoutUnit trimmedContentWidth = 0;
148 for (auto& run : WTF::makeReversedRange(m_runs)) {
149 if (run.style().whiteSpace() != WhiteSpace::PreWrap) {
150 // We are only interested in pre-wrap trailing content.
151 break;
152 }
153 auto preWrapVisuallyCollapsibleInlineItem = run.isInlineBoxStart() || run.isInlineBoxEnd() || run.hasTrailingWhitespace();
154 if (!preWrapVisuallyCollapsibleInlineItem)
155 break;
156 ASSERT(!run.hasCollapsibleTrailingWhitespace());
157 InlineLayoutUnit trimmableWidth = { };
158 if (run.isText()) {
159 // FIXME: We should always collapse the run at a glyph boundary as the spec indicates: "collapse the character advance widths of any that would otherwise overflow"
160 // and the trimmed width should be capped at std::min(run.trailingWhitespaceWidth(), overflowWidth) for text runs. Both FF and Chrome agree.
161 trimmableWidth = run.trailingWhitespaceWidth();
162 run.visuallyCollapseTrailingWhitespace();
163 } else {
164 trimmableWidth = run.logicalWidth();
165 run.shrinkHorizontally(trimmableWidth);
166 }
167 trimmedContentWidth += trimmableWidth;
168 overflowWidth -= trimmableWidth;
169 if (overflowWidth <= 0)
170 break;
171 }
172 m_contentLogicalWidth -= trimmedContentWidth;
173}

175void Line::append(const InlineItem& inlineItem, InlineLayoutUnit logicalWidth)
176{
177 if (inlineItem.isText())
178 appendTextContent(downcast<InlineTextItem>(inlineItem), logicalWidth);
179 else if (inlineItem.isLineBreak())
180 appendLineBreak(inlineItem);
181 else if (inlineItem.isWordBreakOpportunity())
182 appendWordBreakOpportunity(inlineItem);
183 else if (inlineItem.isInlineBoxStart())
184 appendInlineBoxStart(inlineItem, logicalWidth);
185 else if (inlineItem.isInlineBoxEnd())
186 appendInlineBoxEnd(inlineItem, logicalWidth);
187 else if (inlineItem.layoutBox().isReplacedBox())
188 appendReplacedInlineLevelBox(inlineItem, logicalWidth);
189 else if (inlineItem.isBox())
190 appendNonReplacedInlineLevelBox(inlineItem, logicalWidth);
191 else
192 ASSERT_NOT_REACHED();
193}

201void Line::appendInlineBoxStart(const InlineItem& inlineItem, InlineLayoutUnit logicalWidth)
202{
203 // This is really just a placeholder to mark the start of the inline box <span>.
204 auto& boxGeometry = formattingContext().geometryForBox(inlineItem.layoutBox());
205 auto adjustedRunStart = contentLogicalRight() + std::min(boxGeometry.marginStart(), 0_lu);
206 ++m_nonSpanningInlineLevelBoxCount;
207 appendNonBreakableSpace(inlineItem, adjustedRunStart, logicalWidth);
208}

210void Line::appendInlineBoxEnd(const InlineItem& inlineItem, InlineLayoutUnit logicalWidth)
211{
212 // This is really just a placeholder to mark the end of the inline box </span>.
213 auto removeTrailingLetterSpacing = [&] {
214 if (!m_trimmableTrailingContent.isTrailingRunPartiallyTrimmable())
215 return;
216 m_contentLogicalWidth -= m_trimmableTrailingContent.removePartiallyTrimmableContent();
217 };
218 // Prevent trailing letter-spacing from spilling out of the inline box.
219 // https://drafts.csswg.org/css-text-3/#letter-spacing-property See example 21.
220 removeTrailingLetterSpacing();
221 appendNonBreakableSpace(inlineItem, contentLogicalWidth(), logicalWidth);
222}

224void Line::appendTextContent(const InlineTextItem& inlineTextItem, InlineLayoutUnit logicalWidth)
225{
226 auto& style = inlineTextItem.style();
227 auto willCollapseCompletely = [&] {
228 if (inlineTextItem.isEmptyContent())
229 return true;
230 if (!inlineTextItem.isWhitespace())
231 return false;
232 if (InlineTextItem::shouldPreserveSpacesAndTabs(inlineTextItem))
233 return false;
234 // Check if the last item is collapsed as well.
235 for (auto& run : WTF::makeReversedRange(m_runs)) {
236 if (run.isBox())
237 return false;
238 // https://drafts.csswg.org/css-text-3/#white-space-phase-1
239 // Any collapsible space immediately following another collapsible spaceeven one outside the boundary of the inline containing that space,
240 // provided both spaces are within the same inline formatting contextis collapsed to have zero advance width.
241 // : "<span> </span> " <- the trailing whitespace collapses completely.
242 // Not that when the inline box has preserve whitespace style, "<span style="white-space: pre"> </span> " <- this whitespace stays around.
243 if (run.isText())
244 return run.hasCollapsibleTrailingWhitespace();
245 ASSERT(run.isInlineBoxStart() || run.isInlineBoxEnd() || run.isWordBreakOpportunity());
246 }
247 // Leading whitespace.
248 return true;
249 };
250
251 if (willCollapseCompletely())
252 return;
253
254 auto needsNewRun = [&] {
255 if (m_runs.isEmpty())
256 return true;
257 auto& lastRun = m_runs.last();
258 if (&lastRun.layoutBox() != &inlineTextItem.layoutBox())
259 return true;
260 if (!lastRun.isText())
261 return true;
262 if (lastRun.hasCollapsedTrailingWhitespace())
263 return true;
264 if (inlineTextItem.isWordSeparator() && style.fontCascade().wordSpacing())
265 return true;
266 return false;
267 }();
268 auto oldContentLogicalWidth = contentLogicalWidth();
269 if (needsNewRun) {
270 // Note, negative words spacing may cause glyph overlap.
271 auto runLogicalLeft = contentLogicalRight() + (inlineTextItem.isWordSeparator() ? style.fontCascade().wordSpacing() : 0.0f);
272 m_runs.append({ inlineTextItem, runLogicalLeft, logicalWidth });
273 m_contentLogicalWidth = std::max(oldContentLogicalWidth, runLogicalLeft + logicalWidth);
274 } else {
275 m_runs.last().expand(inlineTextItem, logicalWidth);
276 // Do not let negative letter spacing make the content shorter than it already is.
277 m_contentLogicalWidth += std::max(0.0f, logicalWidth);
278 }
279 // Set the trailing trimmable content.
280 if (inlineTextItem.isWhitespace() && !InlineTextItem::shouldPreserveSpacesAndTabs(inlineTextItem)) {
281 m_trimmableTrailingContent.addFullyTrimmableContent(m_runs.size() - 1, contentLogicalWidth() - oldContentLogicalWidth);
282 return;
283 }
284 // Any non-whitespace, no-trimmable content resets the existing trimmable.
285 m_trimmableTrailingContent.reset();
286 if (!formattingContext().layoutState().shouldIgnoreTrailingLetterSpacing() && !inlineTextItem.isWhitespace() && style.letterSpacing() > 0)
287 m_trimmableTrailingContent.addPartiallyTrimmableContent(m_runs.size() - 1, style.letterSpacing());
288 m_trailingSoftHyphenWidth = inlineTextItem.hasTrailingSoftHyphen() ? std::make_optional(style.fontCascade().width(TextRun { StringView { style.hyphenString() } })) : std::nullopt;
289}

291void Line::appendNonReplacedInlineLevelBox(const InlineItem& inlineItem, InlineLayoutUnit marginBoxLogicalWidth)
292{
293 m_trimmableTrailingContent.reset();
294 m_trailingSoftHyphenWidth = { };
295 m_contentLogicalWidth += marginBoxLogicalWidth;
296 ++m_nonSpanningInlineLevelBoxCount;
297 auto marginStart = formattingContext().geometryForBox(inlineItem.layoutBox()).marginStart();
298 if (marginStart >= 0) {
299 m_runs.append({ inlineItem, contentLogicalRight(), marginBoxLogicalWidth });
300 return;
301 }
302 // Negative margin-start pulls the content to the logical left direction.
303 // Negative margin also squeezes the margin box, we need to stretch it to make sure the subsequent content won't overlap.
304 // e.g. <img style="width: 100px; margin-left: -100px;"> pulls the replaced box to -100px with the margin box width of 0px.
305 // Instead we need to position it at -100px and size it to 100px so the subsequent content starts at 0px.
306 m_runs.append({ inlineItem, contentLogicalRight() + marginStart, marginBoxLogicalWidth - marginStart });
307}

316void Line::appendLineBreak(const InlineItem& inlineItem)
317{
318 m_trailingSoftHyphenWidth = { };
319 if (inlineItem.isHardLineBreak()) {
320 ++m_nonSpanningInlineLevelBoxCount;
321 return m_runs.append({ inlineItem, contentLogicalRight(), 0_lu });
322 }
323 // Soft line breaks (preserved new line characters) require inline text boxes for compatibility reasons.
324 ASSERT(inlineItem.isSoftLineBreak());
325 m_runs.append({ downcast<InlineSoftLineBreakItem>(inlineItem), contentLogicalRight() });
326}

355void Line::TrimmableTrailingContent::addFullyTrimmableContent(size_t runIndex, InlineLayoutUnit trimmableWidth)
356{
357 // Any subsequent trimmable whitespace should collapse to zero advanced width and ignored at ::appendTextContent().
358 ASSERT(!m_hasFullyTrimmableContent);
359 m_fullyTrimmableWidth = trimmableWidth;
360 // Note that just because the trimmable width is 0 (font-size: 0px), it does not mean we don't have a trimmable trailing content.
361 m_hasFullyTrimmableContent = true;
362 m_firstTrimmableRunIndex = m_firstTrimmableRunIndex.value_or(runIndex);
363}

365void Line::TrimmableTrailingContent::addPartiallyTrimmableContent(size_t runIndex, InlineLayoutUnit trimmableWidth)
366{
367 // Do not add trimmable letter spacing after a fully trimmable whitespace.
368 ASSERT(!m_firstTrimmableRunIndex);
369 ASSERT(!m_hasFullyTrimmableContent);
370 ASSERT(!m_partiallyTrimmableWidth);
371 ASSERT(trimmableWidth);
372 m_partiallyTrimmableWidth = trimmableWidth;
373 m_firstTrimmableRunIndex = runIndex;
374}

376InlineLayoutUnit Line::TrimmableTrailingContent::remove()
377{
378 // Remove trimmable trailing content and move all the subsequent trailing runs.
379 // <span> </span><span></span>
380 // [trailing whitespace][inline box end][inline box start][inline box end]
381 // Trim the whitespace run and move the trailing inline box runs to the logical left.
382 ASSERT(!isEmpty());
383 auto& trimmableRun = m_runs[*m_firstTrimmableRunIndex];
384 ASSERT(trimmableRun.isText());
385
386 if (m_hasFullyTrimmableContent)
387 trimmableRun.removeTrailingWhitespace();
388 if (m_partiallyTrimmableWidth)
389 trimmableRun.removeTrailingLetterSpacing();
390
391 auto trimmableWidth = width();
392 // When the trimmable run is followed by some non-content runs, we need to adjust their horizontal positions.
393 // e.g. <div>text is followed by trimmable content <span> </span></div>
394 // When the [text...] run is trimmed (trailing whitespace is removed), both "<span>" and "</span>" runs
395 // need to be moved horizontally to catch up with the [text...] run. Note that the whitespace inside the <span> does
396 // not produce a run since in ::appendText() we see it as a fully collapsible run.
397 for (auto index = *m_firstTrimmableRunIndex + 1; index < m_runs.size(); ++index) {
398 auto& run = m_runs[index];
399 ASSERT(run.isWordBreakOpportunity() || run.isInlineBoxStart() || run.isInlineBoxEnd() || run.isLineBreak());
400 run.moveHorizontally(-trimmableWidth);
401 }
402 if (!trimmableRun.textContent()->length()) {
403 // This trimmable run is fully collapsed now (e.g. <div><img> <span></span></div>).
404 // We don't need to keep it around anymore.
405 m_runs.remove(*m_firstTrimmableRunIndex);
406 }
407 reset();
408 return trimmableWidth;
409}

411InlineLayoutUnit Line::TrimmableTrailingContent::removePartiallyTrimmableContent()
412{
413 // Partially trimmable content is always gated by a fully trimmable content.
414 // We can't just trim spacing in the middle.
415 ASSERT(!m_fullyTrimmableWidth);
416 return remove();
417}

451void Line::Run::expand(const InlineTextItem& inlineTextItem, InlineLayoutUnit logicalWidth)
452{
453 // FIXME: This is a very simple expansion merge. We should eventually switch over to FontCascade::expansionOpportunityCount.
454 ASSERT(!hasCollapsedTrailingWhitespace());
455 ASSERT(isText() && inlineTextItem.isText());
456 ASSERT(m_layoutBox == &inlineTextItem.layoutBox());
457
458 m_logicalWidth += logicalWidth;
459 m_trailingWhitespaceType = trailingWhitespaceType(inlineTextItem);
460
461 if (m_trailingWhitespaceType == TrailingWhitespace::None) {
462 m_trailingWhitespaceWidth = { };
463 setExpansionBehavior(AllowLeftExpansion | AllowRightExpansion);
464 m_textContent->expand(inlineTextItem.length());
465 return;
466 }
467 m_trailingWhitespaceWidth += logicalWidth;
468 if (m_whitespaceIsExpansionOpportunity)
469 ++m_expansionOpportunityCount;
470 setExpansionBehavior(DefaultExpansion);
471 m_textContent->expand(m_trailingWhitespaceType == TrailingWhitespace::Collapsed ? 1 : inlineTextItem.length());
472}

493void Line::Run::removeTrailingWhitespace()
494{
495 // According to https://www.w3.org/TR/css-text-3/#white-space-property matrix
496 // Trimmable whitespace is always collapsible so the length of the trailing trimmable whitespace is always 1 (or non-existent).
497 ASSERT(m_textContent->length());
498 constexpr size_t trailingTrimmableContentLength = 1;
499 m_textContent->shrink(trailingTrimmableContentLength);
500 visuallyCollapseTrailingWhitespace();
501}
502
503void Line::Run::visuallyCollapseTrailingWhitespace()
504{
505 // This is just a visual adjustment, the text length should remain the same.
506 shrinkHorizontally(m_trailingWhitespaceWidth);
507 m_trailingWhitespaceWidth = { };
508 m_trailingWhitespaceType = TrailingWhitespace::None;
509
510 if (m_whitespaceIsExpansionOpportunity) {
511 ASSERT(m_expansionOpportunityCount);
512 m_expansionOpportunityCount--;
513 }
514 setExpansionBehavior(AllowLeftExpansion | AllowRightExpansion);
515}
