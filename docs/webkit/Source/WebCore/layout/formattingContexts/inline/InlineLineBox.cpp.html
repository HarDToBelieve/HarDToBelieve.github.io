102InlineRect LineBox::logicalRectForTextRun(const Line::Run& run) const
103{
104 ASSERT(run.isText() || run.isSoftLineBreak());
105 auto* parentInlineBox = &inlineLevelBoxForLayoutBox(run.layoutBox().parent());
106 ASSERT(parentInlineBox->isInlineBox());
107 auto& fontMetrics = parentInlineBox->style().fontMetrics();
108 auto runlogicalTop = parentInlineBox->logicalTop() + parentInlineBox->baseline() - fontMetrics.ascent();
109
110 while (parentInlineBox != &m_rootInlineBox && !parentInlineBox->hasLineBoxRelativeAlignment()) {
111 parentInlineBox = &inlineLevelBoxForLayoutBox(parentInlineBox->layoutBox().parent());
112 ASSERT(parentInlineBox->isInlineBox());
113 runlogicalTop += parentInlineBox->logicalTop();
114 }
115 InlineLayoutUnit logicalHeight = fontMetrics.height();
116 return { runlogicalTop, m_rootInlineBox.logicalLeft() + run.logicalLeft(), run.logicalWidth(), logicalHeight };
117}

125InlineRect LineBox::logicalRectForInlineLevelBox(const Box& layoutBox) const
126{
127 ASSERT(layoutBox.isInlineLevelBox() || layoutBox.isLineBreakBox());
128 // Inline level boxes are relative to their parent unless the vertical alignment makes them relative to the line box (e.g. top, bottom).
129 auto* inlineBox = &inlineLevelBoxForLayoutBox(layoutBox);
130 auto inlineBoxLogicalRect = inlineBox->logicalRect();
131 if (inlineBox->hasLineBoxRelativeAlignment())
132 return inlineBoxLogicalRect;
133
134 // Fast path for inline level boxes on the root inline box (e.g <div><img></div>).
135 if (&layoutBox.parent() == &m_rootInlineBox.layoutBox()) {
136 inlineBoxLogicalRect.moveVertically(m_rootInlineBox.logicalTop());
137 return inlineBoxLogicalRect;
138 }
139
140 // e.g <div><span><img></span></div>
141 auto inlineBoxAbsolutelogicalTop = inlineBoxLogicalRect.top();
142 while (inlineBox != &m_rootInlineBox && !inlineBox->hasLineBoxRelativeAlignment()) {
143 inlineBox = &inlineLevelBoxForLayoutBox(inlineBox->layoutBox().parent());
144 ASSERT(inlineBox->isInlineBox());
145 inlineBoxAbsolutelogicalTop += inlineBox->logicalTop();
146 }
147 return InlineRect { inlineBoxAbsolutelogicalTop, inlineBoxLogicalRect.left(), inlineBoxLogicalRect.width(), inlineBoxLogicalRect.height() };
148}

150InlineRect LineBox::logicalBorderBoxForAtomicInlineLevelBox(const Box& layoutBox, const BoxGeometry& boxGeometry) const
151{
152 ASSERT(layoutBox.isAtomicInlineLevelBox());
153 auto logicalRect = logicalRectForInlineLevelBox(layoutBox);
154 // Inline level boxes use their margin box for vertical alignment. Let's covert them to border boxes.
155 logicalRect.moveVertically(boxGeometry.marginBefore());
156 auto verticalMargin = boxGeometry.marginBefore() + boxGeometry.marginAfter();
157 logicalRect.expandVertically(-verticalMargin);
158 return logicalRect;
159}
160
161InlineRect LineBox::logicalBorderBoxForInlineBox(const Box& layoutBox, const BoxGeometry& boxGeometry) const
162{
163 auto logicalRect = logicalRectForInlineLevelBox(layoutBox);
164 // This logical rect is as tall as the "text" content is. Let's adjust with vertical border and padding.
165 auto verticalBorderAndPadding = boxGeometry.verticalBorder() + boxGeometry.verticalPadding().value_or(0_lu);
166 logicalRect.expandVertically(verticalBorderAndPadding);
167 logicalRect.moveVertically(-(boxGeometry.borderTop() + boxGeometry.paddingTop().value_or(0_lu)));
168 return logicalRect;
169}
