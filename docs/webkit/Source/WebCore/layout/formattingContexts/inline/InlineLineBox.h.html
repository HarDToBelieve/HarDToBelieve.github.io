44// ____________________________________________________________ Line Box
45// | --------------------
46// | | |
47// | ----------------------------------|--------------------|---------- Root Inline Box
48// || _____ ___ ___ | |
49// || | / \ / \ | Inline Level Box |
50// || |_____ | | | | | | ascent
51// || | | | | | | |
52// ||__|________\___/____\___/_________|____________________|_______ alignment_baseline
53// ||
54// || descent
55// ||_______________________________________________________________
56// |________________________________________________________________
57// The resulting rectangular area that contains the boxes that form a single line of inline-level content is called a line box.
58// https://www.w3.org/TR/css-inline-3/#model
59class LineBox {
60 WTF_MAKE_FAST_ALLOCATED;
61public:
62 struct InlineLevelBox {
63 public:
64 static LineBox::InlineLevelBox createInlineBox(const Box&, InlineLayoutUnit logicalLeft, InlineLayoutUnit logicalWidth);
65 static LineBox::InlineLevelBox createAtomicInlineLevelBox(const Box&, InlineLayoutUnit logicalLeft, InlineLayoutSize);
66 static LineBox::InlineLevelBox createLineBreakBox(const Box&, InlineLayoutUnit logicalLeft);
67 static LineBox::InlineLevelBox createGenericInlineLevelBox(const Box&, InlineLayoutUnit logicalLeft);
68
69 InlineLayoutUnit baseline() const { return m_baseline; }
70 std::optional<InlineLayoutUnit> descent() const { return m_descent; }
71 // See https://www.w3.org/TR/css-inline-3/#layout-bounds
72 struct LayoutBounds {
73 InlineLayoutUnit height() const { return ascent + descent; }
74 bool operator==(const LayoutBounds& other) const { return ascent == other.ascent && descent == other.descent; }
75
76 InlineLayoutUnit ascent { 0 };
77 InlineLayoutUnit descent { 0 };
78 };
79 LayoutBounds layoutBounds() const { return m_layoutBounds; }
80
81 bool hasContent() const { return m_hasContent; }
82 void setHasContent();
83
84 VerticalAlign verticalAlign() const { return layoutBox().style().verticalAlign(); }
85 const Box& layoutBox() const { return *m_layoutBox; }
86 const RenderStyle& style() const { return m_layoutBox->style(); }
87
88 bool isInlineBox() const { return m_type == Type::InlineBox || m_type == Type::RootInlineBox; }
89 bool isRootInlineBox() const { return m_type == Type::RootInlineBox; }
90 bool isAtomicInlineLevelBox() const { return m_type == Type::AtomicInlineLevelBox; }
91 bool isLineBreakBox() const { return m_type == Type::LineBreakBox; }
92 bool hasLineBoxRelativeAlignment() const;
93
94 enum class Type : uint8_t {
95 InlineBox = 1 << 0,
96 RootInlineBox = 1 << 1,
97 AtomicInlineLevelBox = 1 << 2,
98 LineBreakBox = 1 << 3,
99 GenericInlineLevelBox = 1 << 4
100 };
101 Type type() const { return m_type; }
102
103 InlineLevelBox(const Box&, InlineLayoutUnit logicalLeft, InlineLayoutSize, Type);
104 InlineLevelBox() = default;
105
106 private:
107 friend struct SimplifiedVerticalAlignment;
108 friend class LineBoxBuilder;
109 friend class LineBox;
110
111 const InlineRect& logicalRect() const { return m_logicalRect; }
112 InlineLayoutUnit logicalTop() const { return m_logicalRect.top(); }
113 InlineLayoutUnit logicalBottom() const { return m_logicalRect.bottom(); }
114 InlineLayoutUnit logicalLeft() const { return m_logicalRect.left(); }
115 InlineLayoutUnit logicalWidth() const { return m_logicalRect.width(); }
116 InlineLayoutUnit logicalHeight() const { return m_logicalRect.height(); }
117
118 void setLogicalWidth(InlineLayoutUnit logicalWidth) { m_logicalRect.setWidth(logicalWidth); }
119 void setLogicalHeight(InlineLayoutUnit);
120 void setLogicalTop(InlineLayoutUnit);
121 void setBaseline(InlineLayoutUnit);
122 void setDescent(InlineLayoutUnit);
123 void setLayoutBounds(const LayoutBounds&);
124
125 private:
126 WeakPtr<const Box> m_layoutBox;
127 // This is the combination of margin and border boxes. Inline level boxes are vertically aligned using their margin boxes.
128 InlineRect m_logicalRect;
129 LayoutBounds m_layoutBounds;
130 InlineLayoutUnit m_baseline { 0 };
131 std::optional<InlineLayoutUnit> m_descent;
132 bool m_hasContent { false };
133 Type m_type { Type::InlineBox };
134 };
135
136 LineBox(const Box& rootLayoutBox, const InlineLayoutPoint& logicalTopLeft, InlineLayoutUnit contentLogicalLeft, InlineLayoutUnit lineLogicalWidth, InlineLayoutUnit contentLogicalWidth, size_t nonSpanningInlineLevelBoxCount);
137
138 const InlineRect& logicalRect() const { return m_logicalRect; }
139 InlineLayoutUnit logicalWidth() const { return logicalSize().width(); }
140 InlineLayoutUnit logicalHeight() const { return logicalSize().height(); }
141 InlineLayoutPoint logicalTopLeft() const { return logicalRect().topLeft(); }
142 InlineLayoutSize logicalSize() const { return logicalRect().size(); }
143 InlineLayoutUnit contentLogicalWidth() const { return m_contentLogicalWidth; }
144
145 // Note that the line can have many inline boxes and be "empty" the same time e.g. <div><span></span><span></span></div>
146 bool hasContent() const { return m_hasContent; }
147 bool hasInlineBox() const { return m_boxTypes.contains(InlineLevelBox::Type::InlineBox); }
148 bool hasNonInlineBox() const { return m_boxTypes.containsAny({ InlineLevelBox::Type::AtomicInlineLevelBox, InlineLevelBox::Type::LineBreakBox, InlineLevelBox::Type::GenericInlineLevelBox }); }
149 bool hasAtomicInlineLevelBox() const { return m_boxTypes.contains(InlineLevelBox::Type::AtomicInlineLevelBox); }
150
151 const InlineLevelBox& inlineLevelBoxForLayoutBox(const Box& layoutBox) const { return const_cast<LineBox&>(*this).inlineLevelBoxForLayoutBox(layoutBox); }
152
153 InlineRect logicalRectForTextRun(const Line::Run&) const;
154 InlineRect logicalRectForLineBreakBox(const Box&) const;
155 InlineRect logicalRectForRootInlineBox() const { return m_rootInlineBox.logicalRect(); }
156 InlineRect logicalBorderBoxForAtomicInlineLevelBox(const Box&, const BoxGeometry&) const;
157 InlineRect logicalBorderBoxForInlineBox(const Box&, const BoxGeometry&) const;
158
159 const InlineLevelBox& rootInlineBox() const { return m_rootInlineBox; }
160 using InlineLevelBoxList = Vector<InlineLevelBox>;
161 const InlineLevelBoxList& nonRootInlineLevelBoxes() const { return m_nonRootInlineLevelBoxList; }
162
163 InlineLayoutUnit alignmentBaseline() const { return m_rootInlineBox.logicalTop() + m_rootInlineBox.baseline(); }
164
165private:
166 friend class LineBoxBuilder;
167
168 void setLogicalHeight(InlineLayoutUnit logicalHeight) { m_logicalRect.setHeight(logicalHeight); }
169
170 void addInlineLevelBox(InlineLevelBox&&);
171 InlineLevelBoxList& nonRootInlineLevelBoxes() { return m_nonRootInlineLevelBoxList; }
172
173 InlineLevelBox& rootInlineBox() { return m_rootInlineBox; }
174
175 InlineLevelBox& inlineLevelBoxForLayoutBox(const Box& layoutBox) { return &layoutBox == &m_rootInlineBox.layoutBox() ? m_rootInlineBox : m_nonRootInlineLevelBoxList[m_nonRootInlineLevelBoxMap.get(&layoutBox)]; }
176 InlineRect logicalRectForInlineLevelBox(const Box& layoutBox) const;
177
178 void setHasContent(bool hasContent) { m_hasContent = hasContent; }
179
180private:
181 InlineRect m_logicalRect;
182 InlineLayoutUnit m_contentLogicalWidth { 0 };
183 bool m_hasContent { false };
184 OptionSet<InlineLevelBox::Type> m_boxTypes;
185
186 InlineLevelBox m_rootInlineBox;
187 InlineLevelBoxList m_nonRootInlineLevelBoxList;
188
189 HashMap<const Box*, size_t> m_nonRootInlineLevelBoxMap;
190};
