26#pragma once
27
28#if ENABLE(LAYOUT_FORMATTING_CONTEXT)
29
30#include "InlineRect.h"
31
32namespace WebCore {
33namespace Layout {
34
35class InlineLineGeometry {
36 WTF_MAKE_FAST_ALLOCATED;
37public:
38 struct EnclosingTopAndBottom {
39 // This values encloses the root inline box and any other inline level box's border box.
40 InlineLayoutUnit top { 0 };
41 InlineLayoutUnit bottom { 0 };
42 };
43 InlineLineGeometry(const InlineRect& lineBoxLogicalRect, EnclosingTopAndBottom, InlineLayoutUnit aligmentBaseline, InlineLayoutUnit contentLogicalLeft, InlineLayoutUnit contentLogicalWidth);
44
45 const InlineRect& lineBoxLogicalRect() const { return m_lineBoxLogicalRect; }
46
47 EnclosingTopAndBottom enclosingTopAndBottom() const { return m_enclosingTopAndBottom; }
48
49 InlineLayoutUnit baseline() const { return m_aligmentBaseline; }
50
51 InlineLayoutUnit contentLogicalLeft() const { return m_contentLogicalLeft; }
52 InlineLayoutUnit contentLogicalWidth() const { return m_contentLogicalWidth; }
53
54 void moveVertically(InlineLayoutUnit offset) { m_lineBoxLogicalRect.moveVertically(offset); }
55
56private:
57 // This is line box geometry (see https://www.w3.org/TR/css-inline-3/#line-box).
58 InlineRect m_lineBoxLogicalRect;
59 // Enclosing top and bottom includes all inline level boxes (border box) vertically.
60 // While the line box usually enclose them as well, its vertical geometry is based on
61 // the layout bounds of the inline level boxes which may be different when line-height is present.
62 EnclosingTopAndBottom m_enclosingTopAndBottom;
63 InlineLayoutUnit m_aligmentBaseline { 0 };
64 InlineLayoutUnit m_contentLogicalLeft { 0 };
65 InlineLayoutUnit m_contentLogicalWidth { 0 };
66};
67
68inline InlineLineGeometry::InlineLineGeometry(const InlineRect& lineBoxLogicalRect, EnclosingTopAndBottom enclosingTopAndBottom, InlineLayoutUnit aligmentBaseline, InlineLayoutUnit contentLogicalLeft, InlineLayoutUnit contentLogicalWidth)
69 : m_lineBoxLogicalRect(lineBoxLogicalRect)
70 , m_enclosingTopAndBottom(enclosingTopAndBottom)
71 , m_aligmentBaseline(aligmentBaseline)
72 , m_contentLogicalLeft(contentLogicalLeft)
73 , m_contentLogicalWidth(contentLogicalWidth)
74{
75}
76
77}
78}
