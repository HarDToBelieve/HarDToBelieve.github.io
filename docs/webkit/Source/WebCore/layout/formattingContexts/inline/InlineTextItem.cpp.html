41struct WhitespaceContent {
42 size_t length { 0 };
43 bool isWordSeparator { true };
44};
45static std::optional<WhitespaceContent> moveToNextNonWhitespacePosition(const StringView& textContent, size_t startPosition, bool preserveNewline, bool preserveTab)
46{
47 auto hasWordSeparatorCharacter = false;
48 auto isWhitespaceCharacter = [&](auto character) {
49 // white space processing in CSS affects only the document white space characters: spaces (U+0020), tabs (U+0009), and segment breaks.
50 auto isTreatedAsSpaceCharacter = character == space || (character == newlineCharacter && !preserveNewline) || (character == tabCharacter && !preserveTab);
51 hasWordSeparatorCharacter = hasWordSeparatorCharacter || isTreatedAsSpaceCharacter;
52 return isTreatedAsSpaceCharacter || character == tabCharacter;
53 };
54 auto nextNonWhiteSpacePosition = startPosition;
55 while (nextNonWhiteSpacePosition < textContent.length() && isWhitespaceCharacter(textContent[nextNonWhiteSpacePosition]))
56 ++nextNonWhiteSpacePosition;
57 return nextNonWhiteSpacePosition == startPosition ? std::nullopt : std::make_optional(WhitespaceContent { nextNonWhiteSpacePosition - startPosition, hasWordSeparatorCharacter });
58}

60static unsigned moveToNextBreakablePosition(unsigned startPosition, LazyLineBreakIterator& lineBreakIterator, const RenderStyle& style)
61{
62 auto textLength = lineBreakIterator.stringView().length();
63 auto startPositionForNextBreakablePosition = startPosition;
64 while (startPositionForNextBreakablePosition < textLength) {
65 auto nextBreakablePosition = TextUtil::findNextBreakablePosition(lineBreakIterator, startPositionForNextBreakablePosition, style);
66 // Oftentimes the next breakable position comes back as the start position (most notably hyphens).
67 if (nextBreakablePosition != startPosition)
68 return nextBreakablePosition - startPosition;
69 ++startPositionForNextBreakablePosition;
70 }
71 return textLength - startPosition;
72}

74void InlineTextItem::createAndAppendTextItems(InlineItems& inlineContent, const InlineTextBox& inlineTextBox)
75{
76 auto text = inlineTextBox.content();
77 if (!text.length())
78 return inlineContent.append(InlineTextItem::createEmptyItem(inlineTextBox));
79
80 auto& style = inlineTextBox.style();
81 auto& font = style.fontCascade();
82 auto whitespaceContentIsTreatedAsSingleSpace = !TextUtil::shouldPreserveSpacesAndTabs(inlineTextBox);
83 auto shouldPreserveNewline = TextUtil::shouldPreserveNewline(inlineTextBox);
84 auto lineBreakIterator = LazyLineBreakIterator { text, style.computedLocale(), TextUtil::lineBreakIteratorMode(style.lineBreak()) };
85 unsigned currentPosition = 0;
86
87 auto inlineItemWidth = [&](auto startPosition, auto length) -> std::optional<InlineLayoutUnit> {
88 if (!inlineTextBox.canUseSimplifiedContentMeasuring())
89 return { };
90 return TextUtil::width(inlineTextBox, startPosition, startPosition + length, { });
91 };
92
93 while (currentPosition < text.length()) {
94 auto isSegmentBreakCandidate = [](auto character) {
95 return character == '\n';
96 };
97
98 // Segment breaks with preserve new line style (white-space: pre, pre-wrap, break-spaces and pre-line) compute to forced line break.
99 if (isSegmentBreakCandidate(text[currentPosition]) && shouldPreserveNewline) {
100 inlineContent.append(InlineSoftLineBreakItem::createSoftLineBreakItem(inlineTextBox, currentPosition));
101 ++currentPosition;
102 continue;
103 }
104
105 if (auto whitespaceContent = moveToNextNonWhitespacePosition(text, currentPosition, shouldPreserveNewline, !whitespaceContentIsTreatedAsSingleSpace)) {
106 ASSERT(whitespaceContent->length);
107 auto appendWhitespaceItem = [&] (auto startPosition, auto itemLength) {
108 auto simpleSingleWhitespaceContent = inlineTextBox.canUseSimplifiedContentMeasuring() && (itemLength == 1 || whitespaceContentIsTreatedAsSingleSpace);
109 auto width = simpleSingleWhitespaceContent ? std::make_optional(InlineLayoutUnit { font.spaceWidth() }) : inlineItemWidth(startPosition, itemLength);
110 inlineContent.append(InlineTextItem::createWhitespaceItem(inlineTextBox, startPosition, itemLength, whitespaceContent->isWordSeparator, width));
111 };
112 if (style.whiteSpace() == WhiteSpace::BreakSpaces) {
113 // https://www.w3.org/TR/css-text-3/#white-space-phase-1
114 // For break-spaces, a soft wrap opportunity exists after every space and every tab.
115 // FIXME: if this turns out to be a perf hit with too many individual whitespace inline items, we should transition this logic to line breaking.
116 for (size_t i = 0; i < whitespaceContent->length; ++i)
117 appendWhitespaceItem(currentPosition + i, 1);
118 } else
119 appendWhitespaceItem(currentPosition, whitespaceContent->length);
120 currentPosition += whitespaceContent->length;
121 continue;
122 }
123
124 auto hasTrailingSoftHyphen = false;
125 auto initialNonWhitespacePosition = currentPosition;
126 auto isAtSoftHyphen = [&](auto position) {
127 return text[position] == softHyphen;
128 };
129 if (style.hyphens() == Hyphens::None) {
130 // Let's merge candidate InlineTextItems separated by soft hyphen when the style says so.
131 while (currentPosition < text.length()) {
132 auto nonWhiteSpaceLength = moveToNextBreakablePosition(currentPosition, lineBreakIterator, style);
133 ASSERT(nonWhiteSpaceLength);
134 currentPosition += nonWhiteSpaceLength;
135 if (!isAtSoftHyphen(currentPosition - 1))
136 break;
137 }
138 } else {
139 auto nonWhiteSpaceLength = moveToNextBreakablePosition(initialNonWhitespacePosition, lineBreakIterator, style);
140 ASSERT(nonWhiteSpaceLength);
141 currentPosition += nonWhiteSpaceLength;
142 hasTrailingSoftHyphen = isAtSoftHyphen(currentPosition - 1);
143 }
144 ASSERT(initialNonWhitespacePosition < currentPosition);
145 ASSERT_IMPLIES(style.hyphens() == Hyphens::None, !hasTrailingSoftHyphen);
146 auto length = currentPosition - initialNonWhitespacePosition;
147 inlineContent.append(InlineTextItem::createNonWhitespaceItem(inlineTextBox, initialNonWhitespacePosition, length, hasTrailingSoftHyphen, inlineItemWidth(initialNonWhitespacePosition, length)));
148 }
149}

151bool InlineTextItem::isEmptyContent() const
152{
153 // FIXME: We should check for more zero width content and not just U+200B.
154 return !m_length || (m_length == 1 && inlineTextBox().content()[start()] == zeroWidthSpace);
155}
156
157bool InlineTextItem::shouldPreserveSpacesAndTabs(const InlineTextItem& inlineTextItem)
158{
159 ASSERT(inlineTextItem.isWhitespace());
160 return TextUtil::shouldPreserveSpacesAndTabs(inlineTextItem.layoutBox());
161}
