47LayoutUnit TableFormattingGeometry::cellBoxContentHeight(const ContainerBox& cellBox) const
48{
49 ASSERT(cellBox.isInFlow());
50 if (layoutState().inQuirksMode() && TableFormattingQuirks::shouldIgnoreChildContentVerticalMargin(cellBox)) {
51 ASSERT(cellBox.firstInFlowChild());
52 auto formattingContext = this->formattingContext();
53 auto& firstInFlowChild = *cellBox.firstInFlowChild();
54 auto& lastInFlowChild = *cellBox.lastInFlowChild();
55 auto& firstInFlowChildGeometry = formattingContext.geometryForBox(firstInFlowChild, FormattingContext::EscapeReason::TableQuirkNeedsGeometryFromEstablishedFormattingContext);
56 auto& lastInFlowChildGeometry = formattingContext.geometryForBox(lastInFlowChild, FormattingContext::EscapeReason::TableQuirkNeedsGeometryFromEstablishedFormattingContext);
57
58 auto top = firstInFlowChild.style().hasMarginBeforeQuirk() ? BoxGeometry::borderBoxRect(firstInFlowChildGeometry).top() : BoxGeometry::marginBoxRect(firstInFlowChildGeometry).top();
59 auto bottom = lastInFlowChild.style().hasMarginAfterQuirk() ? BoxGeometry::borderBoxRect(lastInFlowChildGeometry).bottom() : BoxGeometry::marginBoxRect(lastInFlowChildGeometry).bottom();
60 return bottom - top;
61 }
62 return contentHeightForFormattingContextRoot(cellBox);
63}

65Edges TableFormattingGeometry::computedCellBorder(const TableGrid::Cell& cell) const
66{
67 auto& grid = formattingContext().formattingState().tableGrid();
68 auto& cellBox = cell.box();
69 auto border = computedBorder(cellBox);
70 auto collapsedBorder = grid.collapsedBorder();
71 if (!collapsedBorder)
72 return border;
73
74 // We might want to cache these collapsed borders on the grid.
75 auto cellPosition = cell.position();
76 // Collapsed border left from table and adjacent cells.
77 if (!cellPosition.column)
78 border.horizontal.left = collapsedBorder->horizontal.left / 2;
79 else {
80 auto adjacentBorderRight = computedBorder(grid.slot({ cellPosition.column - 1, cellPosition.row })->cell().box()).horizontal.right;
81 border.horizontal.left = std::max(border.horizontal.left, adjacentBorderRight) / 2;
82 }
83 // Collapsed border right from table and adjacent cells.
84 if (cellPosition.column == grid.columns().size() - 1)
85 border.horizontal.right = collapsedBorder->horizontal.right / 2;
86 else {
87 auto adjacentBorderLeft = computedBorder(grid.slot({ cellPosition.column + 1, cellPosition.row })->cell().box()).horizontal.left;
88 border.horizontal.right = std::max(border.horizontal.right, adjacentBorderLeft) / 2;
89 }
90 // Collapsed border top from table, row and adjacent cells.
91 auto& rows = grid.rows().list();
92 if (!cellPosition.row)
93 border.vertical.top = collapsedBorder->vertical.top / 2;
94 else {
95 auto adjacentBorderBottom = computedBorder(grid.slot({ cellPosition.column, cellPosition.row - 1 })->cell().box()).vertical.bottom;
96 auto adjacentRowBottom = computedBorder(rows[cellPosition.row - 1].box()).vertical.bottom;
97 auto adjacentCollapsedBorder = std::max(adjacentBorderBottom, adjacentRowBottom);
98 border.vertical.top = std::max(border.vertical.top, adjacentCollapsedBorder) / 2;
99 }
100 // Collapsed border bottom from table, row and adjacent cells.
101 if (cellPosition.row == grid.rows().size() - 1)
102 border.vertical.bottom = collapsedBorder->vertical.bottom / 2;
103 else {
104 auto adjacentBorderTop = computedBorder(grid.slot({ cellPosition.column, cellPosition.row + 1 })->cell().box()).vertical.top;
105 auto adjacentRowTop = computedBorder(rows[cellPosition.row + 1].box()).vertical.top;
106 auto adjacentCollapsedBorder = std::max(adjacentBorderTop, adjacentRowTop);
107 border.vertical.bottom = std::max(border.vertical.bottom, adjacentCollapsedBorder) / 2;
108 }
109 return border;
110}

122{

130InlineLayoutUnit TableFormattingGeometry::usedBaselineForCell(const ContainerBox& cellBox) const
131{
132 // The baseline of a cell is defined as the baseline of the first in-flow line box in the cell,
133 // or the first in-flow table-row in the cell, whichever comes first.
134 // If there is no such line box, the baseline is the bottom of content edge of the cell box.
135 if (cellBox.establishesInlineFormattingContext())
136 return layoutState().establishedInlineFormattingState(cellBox).lines()[0].baseline();
137 for (auto& cellDescendant : descendantsOfType<ContainerBox>(cellBox)) {
138 if (cellDescendant.establishesInlineFormattingContext()) {
139 auto& inlineFormattingStateForCell = layoutState().establishedInlineFormattingState(cellDescendant);
140 if (!inlineFormattingStateForCell.lines().isEmpty())
141 return inlineFormattingStateForCell.lines()[0].baseline();
142 }
143 if (cellDescendant.establishesTableFormattingContext())
144 return layoutState().establishedTableFormattingState(cellDescendant).tableGrid().rows().list()[0].baseline();
145 }
146 return formattingContext().geometryForBox(cellBox).contentBoxBottom();
147}

149LayoutUnit TableFormattingGeometry::horizontalSpaceForCellContent(const TableGrid::Cell& cell) const
150{
151 auto& grid = formattingContext().formattingState().tableGrid();
152 auto& columnList = grid.columns().list();
153 auto logicalWidth = LayoutUnit { };
154 for (auto columnIndex = cell.startColumn(); columnIndex < cell.endColumn(); ++columnIndex)
155 logicalWidth += columnList.at(columnIndex).usedLogicalWidth();
156 // No column spacing when spanning.
157 logicalWidth += (cell.columnSpan() - 1) * grid.horizontalSpacing();
158 auto& cellBoxGeometry = formattingContext().geometryForBox(cell.box());
159 logicalWidth -= (cellBoxGeometry.horizontalBorder() + cellBoxGeometry.horizontalPadding().value_or(0));
160 return logicalWidth;
161}
162
163LayoutUnit TableFormattingGeometry::verticalSpaceForCellContent(const TableGrid::Cell& cell, std::optional<LayoutUnit> availableVerticalSpace) const
164{
165 auto& cellBox = cell.box();
166 auto contentHeight = cellBoxContentHeight(cellBox);
167 auto computedHeight = this->computedHeight(cellBox, availableVerticalSpace);
168 if (!computedHeight)
169 return contentHeight;
170 auto heightUsesBorderBox = layoutState().inQuirksMode() || cellBox.style().boxSizing() == BoxSizing::BorderBox;
171 if (heightUsesBorderBox) {
172 auto& cellBoxGeometry = formattingContext().geometryForBox(cell.box());
173 *computedHeight -= (cellBoxGeometry.verticalBorder() + cellBoxGeometry.verticalPadding().value_or(0));
174 }
175 return std::max(contentHeight, *computedHeight);
176}
