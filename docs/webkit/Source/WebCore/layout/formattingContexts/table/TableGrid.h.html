65 // Column represents a vertical set of slots in the grid. A column has horizontal position and width.
66 class Column {
67 public:
68 Column(const ContainerBox*);
69
70 void setUsedLogicalLeft(LayoutUnit);
71 LayoutUnit usedLogicalLeft() const;
72 LayoutUnit usedLogicalRight() const { return usedLogicalLeft() + usedLogicalWidth(); }
73 void setUsedLogicalWidth(LayoutUnit);
74 LayoutUnit usedLogicalWidth() const;
75
76 void setComputedLogicalWidth(Length&&);
77 const Length& computedLogicalWidth() const { return m_computedLogicalWidth; }
78
79 const ContainerBox* box() const { return m_layoutBox.get(); }
80
81 private:
82 LayoutUnit m_usedLogicalWidth;
83 LayoutUnit m_usedLogicalLeft;
84 Length m_computedLogicalWidth;
85 WeakPtr<const ContainerBox> m_layoutBox;

91 };

146 // Cell represents a <td> or <th>. It can span multiple slots in the grid.
147 class Cell : public CanMakeWeakPtr<Cell> {
148 WTF_MAKE_ISO_ALLOCATED_INLINE(Cell);
149 public:
150 Cell(const ContainerBox&, SlotPosition, CellSpan);
151
152 size_t startColumn() const { return m_position.column; }
153 size_t endColumn() const { return m_position.column + m_span.column; }
154
155 size_t startRow() const { return m_position.row; }
156 size_t endRow() const { return m_position.row + m_span.row; }
157
158 size_t columnSpan() const { return m_span.column; }
159 size_t rowSpan() const { return m_span.row; }
160
161 SlotPosition position() const { return m_position; }
162 CellSpan span() const { return m_span; }
163
164 void setBaseline(InlineLayoutUnit baseline) { m_baseline = baseline; }
165 InlineLayoutUnit baseline() const { return m_baseline; }
166
167 const ContainerBox& box() const { return *m_layoutBox.get(); }
168
169 private:
170 WeakPtr<const ContainerBox> m_layoutBox;
171 SlotPosition m_position;
172 CellSpan m_span;
173 InlineLayoutUnit m_baseline { 0 };
174 };

176 class Slot {
177 public:
178 WTF_MAKE_STRUCT_FAST_ALLOCATED;
179 Slot() = default;
180 Slot(Cell&, bool isColumnSpanned, bool isRowSpanned);
181
182 const Cell& cell() const { return *m_cell; }
183 Cell& cell() { return *m_cell; }
184
185 const IntrinsicWidthConstraints& widthConstraints() const { return m_widthConstraints; }
186 void setWidthConstraints(const IntrinsicWidthConstraints& widthConstraints) { m_widthConstraints = widthConstraints; }
187
188 // Initial slot position for a spanning cell.
189 // <td></td><td colspan=2></td> [1, 0] slot has column span of 2.
190 bool hasColumnSpan() const { return m_cell->columnSpan() > 1 && !isColumnSpanned(); }
191 bool hasRowSpan() const { return m_cell->rowSpan() > 1 && !isRowSpanned(); }
192
193 // Non-initial spanned slot.
194 // <td></td><td colspan=2></td> [2, 0] slot is column spanned by [1, 0].
195 // <td></td><td></td><td></td>
196 bool isColumnSpanned() const { return m_isColumnSpanned; }
197 bool isRowSpanned() const { return m_isRowSpanned; }
198
199 private:
200 WeakPtr<Cell> m_cell;
201 bool m_isColumnSpanned { false };
202 bool m_isRowSpanned { false };
203 IntrinsicWidthConstraints m_widthConstraints;
204 };

233inline void TableGrid::Column::setComputedLogicalWidth(Length&& computedLogicalWidth)
234{
235 ASSERT(computedLogicalWidth.type() == LengthType::Fixed || computedLogicalWidth.type() == LengthType::Percent || computedLogicalWidth.type() == LengthType::Relative);
236 m_computedLogicalWidth = WTFMove(computedLogicalWidth);
237}

239inline void TableGrid::Column::setUsedLogicalWidth(LayoutUnit usedLogicalWidth)
240{
241#if ASSERT_ENABLED
242 m_hasUsedWidth = true;
243#endif
244 m_usedLogicalWidth = usedLogicalWidth;
245}
246
247inline LayoutUnit TableGrid::Column::usedLogicalWidth() const
248{
249 ASSERT(m_hasUsedWidth);
250 return m_usedLogicalWidth;
251}

259}

261inline LayoutUnit TableGrid::Column::usedLogicalLeft() const

263 ASSERT(m_hasUsedLeft);
264 return m_usedLogicalLeft;
