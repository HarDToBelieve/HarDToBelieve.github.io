82class LineIterator {
83public:
84 LineIterator() : m_line(LineIteratorLegacyPath { nullptr }) { };
85 LineIterator(const LegacyRootInlineBox* rootInlineBox) : m_line(LineIteratorLegacyPath { rootInlineBox }) { };
86 LineIterator(PathLine::PathVariant&&);
87
88 LineIterator& operator++() { return traverseNext(); }
89 LineIterator& traverseNext();
90 LineIterator& traversePrevious();
91
92 LineIterator next() const;
93 LineIterator previous() const;
94
95 explicit operator bool() const { return !atEnd(); }
96
97 bool operator==(const LineIterator&) const;
98 bool operator!=(const LineIterator& other) const { return !(*this == other); }
99
100 bool operator==(EndLineIterator) const { return atEnd(); }
101 bool operator!=(EndLineIterator) const { return !atEnd(); }
102
103 const PathLine& operator*() const { return m_line; }
104 const PathLine* operator->() const { return &m_line; }
105
106 bool atEnd() const;
107
108 RunIterator firstRun() const;
109 RunIterator lastRun() const;
110
111 RunIterator closestRunForPoint(const IntPoint& pointInContents, bool editableOnly);
112 RunIterator closestRunForLogicalLeftPosition(int position, bool editableOnly = false);
113
114 RunIterator logicalStartRun() const;
115 RunIterator logicalEndRun() const;
116 RunIterator logicalStartRunWithNode() const;
117 RunIterator logicalEndRunWithNode() const;
118
119private:
120 PathLine m_line;
121};
