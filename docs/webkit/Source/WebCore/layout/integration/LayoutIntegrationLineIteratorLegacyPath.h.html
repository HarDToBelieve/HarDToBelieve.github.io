28#include "LayoutIntegrationInlineContent.h"
29#include "LayoutIntegrationRunIteratorLegacyPath.h"
30#include "LegacyRootInlineBox.h"

32namespace WebCore {
33
34namespace LayoutIntegration {
35
36class LineIteratorLegacyPath {
37public:
38 LineIteratorLegacyPath(const LegacyRootInlineBox* rootInlineBox)
39 : m_rootInlineBox(rootInlineBox)
40 {
41 }
42 LineIteratorLegacyPath(LineIteratorLegacyPath&&) = default;
43 LineIteratorLegacyPath(const LineIteratorLegacyPath&) = default;
44 LineIteratorLegacyPath& operator=(const LineIteratorLegacyPath&) = default;
45 LineIteratorLegacyPath& operator=(LineIteratorLegacyPath&&) = default;
46
47 LayoutUnit top() const { return m_rootInlineBox->lineTop(); }
48 LayoutUnit bottom() const { return m_rootInlineBox->lineBottom(); }
49 LayoutUnit selectionTop() const { return m_rootInlineBox->selectionTop(); }
50 LayoutUnit selectionTopForHitTesting() const { return m_rootInlineBox->selectionTop(LegacyRootInlineBox::ForHitTesting::Yes); }
51 LayoutUnit selectionBottom() const { return m_rootInlineBox->selectionBottom(); }
52 LayoutUnit lineBoxTop() const { return m_rootInlineBox->lineBoxTop(); }
53 LayoutUnit lineBoxBottom() const { return m_rootInlineBox->lineBoxBottom(); }
54
55 float y() const { return m_rootInlineBox->y(); }
56 float contentLogicalLeft() const { return m_rootInlineBox->logicalLeft(); }
57 float contentLogicalRight() const { return m_rootInlineBox->logicalRight(); }
58 float logicalHeight() const { return m_rootInlineBox->logicalHeight(); }
59 bool isHorizontal() const { return m_rootInlineBox->isHorizontal(); }
60
61 const RenderBlockFlow& containingBlock() const { return m_rootInlineBox->blockFlow(); }
62 const LegacyRootInlineBox* legacyRootInlineBox() const { return m_rootInlineBox; }
63
64 void traverseNext()
65 {
66 m_rootInlineBox = m_rootInlineBox->nextRootBox();
67 }
68
69 void traversePrevious()
70 {
71 m_rootInlineBox = m_rootInlineBox->prevRootBox();
72 }
73
74 bool operator==(const LineIteratorLegacyPath& other) const { return m_rootInlineBox == other.m_rootInlineBox; }
75
76 bool atEnd() const { return !m_rootInlineBox; }
77
78 RunIteratorLegacyPath firstRun() const
79 {
80 return { m_rootInlineBox->firstLeafDescendant() };
81 }
82
83 RunIteratorLegacyPath lastRun() const
84 {
85 return { m_rootInlineBox->lastLeafDescendant() };
86 }
87
88
89 RunIteratorLegacyPath logicalStartRun() const
90 {
91 return { *m_rootInlineBox, RunIteratorLegacyPath::LogicalOrder::Start };
92 }
93
94 RunIteratorLegacyPath logicalEndRun() const
95 {
96 return { *m_rootInlineBox, RunIteratorLegacyPath::LogicalOrder::End };
97 }
98
99private:
100 const LegacyRootInlineBox* m_rootInlineBox;
101};
