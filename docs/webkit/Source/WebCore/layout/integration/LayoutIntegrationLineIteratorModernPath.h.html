34namespace WebCore {
35
36namespace LayoutIntegration {
37
38class RunIteratorModernPath;
39
40class LineIteratorModernPath {
41public:
42 LineIteratorModernPath(const InlineContent& inlineContent, size_t lineIndex)
43 : m_inlineContent(&inlineContent)
44 , m_lineIndex(lineIndex)
45 {
46 ASSERT(lineIndex <= lines().size());
47 }
48 LineIteratorModernPath(LineIteratorModernPath&&) = default;
49 LineIteratorModernPath(const LineIteratorModernPath&) = default;
50 LineIteratorModernPath& operator=(const LineIteratorModernPath&) = default;
51 LineIteratorModernPath& operator=(LineIteratorModernPath&&) = default;
52
53 LayoutUnit top() const { return LayoutUnit::fromFloatRound(line().enclosingContentTop()); }
54 LayoutUnit bottom() const { return LayoutUnit::fromFloatRound(line().enclosingContentBottom()); }
55 LayoutUnit lineBoxTop() const { return LayoutUnit::fromFloatRound(line().lineBoxTop()); }
56 LayoutUnit lineBoxBottom() const { return LayoutUnit::fromFloatRound(line().lineBoxBottom()); }
57
58 LayoutUnit selectionTop() const { return !m_lineIndex ? top() : LineIteratorModernPath(*m_inlineContent, m_lineIndex - 1).selectionBottom(); }
59 // FIXME: Remove the containingBlock().borderAndPaddingBefore() offset after retiring legacy line layout. It also requires changes in RenderText::positionForPoint to find the first line with offset.
60 // - the "before" value is already factored in to the line offset
61 // - this logic negates the first line's natural offset (e.g. block has no border/padding but the first line has a computed offset).
62 LayoutUnit selectionTopForHitTesting() const { return !m_lineIndex ? containingBlock().borderAndPaddingBefore() : selectionTop(); };
63 LayoutUnit selectionBottom() const { return bottom(); }
64
65 float contentLogicalLeft() const { return line().lineBoxLeft() + line().contentLeft(); }
66 float contentLogicalRight() const { return contentLogicalLeft() + line().contentWidth(); }
67 float y() const { return lineBoxTop(); }
68 float logicalHeight() const { return lineBoxBottom() - lineBoxTop(); }
69 bool isHorizontal() const { return true; }
70
71 const RenderBlockFlow& containingBlock() const { return m_inlineContent->containingBlock(); }
72 const LegacyRootInlineBox* legacyRootInlineBox() const { return nullptr; }
73
74 void traverseNext()
75 {
76 ASSERT(!atEnd());
77
78 ++m_lineIndex;
79 }
80
81 void traversePrevious()
82 {
83 ASSERT(!atEnd());
84
85 if (!m_lineIndex) {
86 setAtEnd();
87 return;
88 }
89
90 --m_lineIndex;
91 }
92
93 bool operator==(const LineIteratorModernPath& other) const { return m_inlineContent == other.m_inlineContent && m_lineIndex == other.m_lineIndex; }
94
95 bool atEnd() const { return m_lineIndex == lines().size(); }
96
97 RunIteratorModernPath firstRun() const
98 {
99 if (!line().runCount())
100 return { *m_inlineContent };
101 return { *m_inlineContent, line().firstRunIndex() };
102 }
103
104 RunIteratorModernPath lastRun() const
105 {
106 auto runCount = line().runCount();
107 if (!runCount)
108 return { *m_inlineContent };
109 return { *m_inlineContent, line().firstRunIndex() + runCount - 1 };
110 }
111
112 RunIteratorModernPath logicalStartRun() const
113 {
114 return firstRun();
115 }
116
117 RunIteratorModernPath logicalEndRun() const
118 {
119 return lastRun();
120 }
121
122private:
123 void setAtEnd() { m_lineIndex = lines().size(); }
124
125 const InlineContent::Lines& lines() const { return m_inlineContent->lines; }
126 const Line& line() const { return lines()[m_lineIndex]; }
127
128 RefPtr<const InlineContent> m_inlineContent;
129 size_t m_lineIndex { 0 };
130};
131
132}
133}
