71RenderBlockFlow* LineLayout::blockContainer(RenderObject& renderer)
72{
73 // FIXME: These fake renderers have their parent set but are not actually in the tree.
74 if (renderer.isReplica() || renderer.isRenderScrollbarPart())
75 return nullptr;
76
77 for (auto* parent = renderer.parent(); parent; parent = parent->parent()) {
78 if (!parent->childrenInline())
79 return nullptr;
80 if (is<RenderBlockFlow>(*parent))
81 return downcast<RenderBlockFlow>(parent);
82 }
83
84 return nullptr;
85}

140void LineLayout::updateLayoutBoxDimensions(const RenderBox& replacedOrInlineBlock)
141{
142 auto& layoutBox = m_boxTree.layoutBoxForRenderer(replacedOrInlineBlock);
143 // Internally both replaced and inline-box content use replaced boxes.
144 auto& replacedBox = downcast<Layout::ReplacedBox>(layoutBox);
145
146 // Always use the physical size here for inline level boxes (this is where the logical vs. physical coords flip happens).
147 auto& replacedBoxGeometry = m_layoutState.ensureGeometryForBox(replacedBox);
148
149 // Scrollbars eat into the padding box area. They never stretch the border box but they may shrink the padding box.
150 // In legacy render tree, RenderBox::contentWidth/contentHeight values are adjusted to accomodate the scrollbar width/height.
151 // e.g. <div style="width: 10px; overflow: scroll;">content</div>, RenderBox::contentWidth() won't be returning the value of 10px but instead 0px (10px - 15px).
152 auto horizontalSpaceReservedForScrollbar = replacedOrInlineBlock.paddingBoxRectIncludingScrollbar().width() - replacedOrInlineBlock.paddingBoxWidth();
153 replacedBoxGeometry.setHorizontalSpaceForScrollbar(horizontalSpaceReservedForScrollbar);
154
155 auto verticalSpaceReservedForScrollbar = replacedOrInlineBlock.paddingBoxRectIncludingScrollbar().height() - replacedOrInlineBlock.paddingBoxHeight();
156 replacedBoxGeometry.setVerticalSpaceForScrollbar(verticalSpaceReservedForScrollbar);
157
158 replacedBoxGeometry.setContentBoxWidth(replacedOrInlineBlock.contentWidth());
159 replacedBoxGeometry.setContentBoxHeight(replacedOrInlineBlock.contentHeight());
160
161 replacedBoxGeometry.setBorder({ { replacedOrInlineBlock.borderLeft(), replacedOrInlineBlock.borderRight() }, { replacedOrInlineBlock.borderTop(), replacedOrInlineBlock.borderBottom() } });
162 replacedBoxGeometry.setPadding(Layout::Edges { { replacedOrInlineBlock.paddingLeft(), replacedOrInlineBlock.paddingRight() }, { replacedOrInlineBlock.paddingTop(), replacedOrInlineBlock.paddingBottom() } });
163
164 replacedBoxGeometry.setHorizontalMargin({ replacedOrInlineBlock.marginLeft(), replacedOrInlineBlock.marginRight() });
165 replacedBoxGeometry.setVerticalMargin({ replacedOrInlineBlock.marginTop(), replacedOrInlineBlock.marginBottom() });
166
167 auto baseline = replacedOrInlineBlock.baselinePosition(AlphabeticBaseline, false /* firstLine */, HorizontalLine, PositionOnContainingLine);
168 replacedBox.setBaseline(roundToInt(baseline));
169}

183void LineLayout::updateInlineBoxDimensions(const RenderInline& renderInline)
184{
185 auto& boxGeometry = m_layoutState.ensureGeometryForBox(m_boxTree.layoutBoxForRenderer(renderInline));
186
187 // Check if this renderer is part of a continuation and adjust horizontal margin/border/padding accordingly.
188 auto shouldNotRetainBorderPaddingAndMarginStart = renderInline.parent()->isAnonymousBlock() && renderInline.isContinuation();
189 auto shouldNotRetainBorderPaddingAndMarginEnd = renderInline.parent()->isAnonymousBlock() && !renderInline.isContinuation() && renderInline.inlineContinuation();
190
191 auto horizontalMargin = Layout::BoxGeometry::HorizontalMargin { shouldNotRetainBorderPaddingAndMarginStart ? 0_lu : renderInline.marginLeft(), shouldNotRetainBorderPaddingAndMarginEnd ? 0_lu : renderInline.marginRight() };
192 auto horizontalBorder = Layout::HorizontalEdges { shouldNotRetainBorderPaddingAndMarginStart ? 0_lu : renderInline.borderLeft(), shouldNotRetainBorderPaddingAndMarginEnd ? 0_lu : renderInline.borderRight() };
193 auto horizontalPadding = Layout::HorizontalEdges { shouldNotRetainBorderPaddingAndMarginStart ? 0_lu : renderInline.paddingLeft(), shouldNotRetainBorderPaddingAndMarginEnd ? 0_lu : renderInline.paddingRight() };
194
195 boxGeometry.setPadding(Layout::Edges { horizontalPadding, { renderInline.paddingTop(), renderInline.paddingBottom() } });
196 boxGeometry.setBorder({ horizontalBorder, { renderInline.borderTop(), renderInline.borderBottom() } });
197 boxGeometry.setHorizontalMargin(horizontalMargin);
198 boxGeometry.setVerticalMargin({ });
199}

206void LineLayout::layout()
207{
208 if (!rootLayoutBox().hasInFlowOrFloatingChild())
209 return;
210
211 prepareLayoutState();
212 prepareFloatingState();
213
214 m_inlineContent = nullptr;
215 auto inlineFormattingContext = Layout::InlineFormattingContext { rootLayoutBox(), m_inlineFormattingState };
216
217 auto invalidationState = Layout::InvalidationState { };
218 auto horizontalConstraints = Layout::HorizontalConstraints { flow().borderAndPaddingStart(), flow().contentSize().width() };
219
220 inlineFormattingContext.lineLayoutForIntergration(invalidationState, { horizontalConstraints, flow().borderAndPaddingBefore() });
221
222 constructContent();
223}

225void LineLayout::constructContent()
226{
227 auto inlineFormattingContext = Layout::InlineFormattingContext { rootLayoutBox(), m_inlineFormattingState };
228
229 auto inlineContentBuilder = InlineContentBuilder { m_layoutState, flow(), m_boxTree };
230 inlineContentBuilder.build(inlineFormattingContext, ensureInlineContent());
231 ASSERT(m_inlineContent);
232
233 for (auto& run : m_inlineContent->runs) {
234 auto& layoutBox = run.layoutBox();
235 if (!layoutBox.isReplacedBox())
236 continue;
237
238 auto& renderer = downcast<RenderBox>(m_boxTree.rendererForLayoutBox(layoutBox));
239 renderer.setLocation(flooredLayoutPoint(run.rect().location()));
240 }
241
242 m_inlineContent->clearGapAfterLastLine = m_inlineFormattingState.clearGapAfterLastLine();
243 m_inlineContent->shrinkToFit();
244 m_inlineFormattingState.shrinkToFit();
245}

259void LineLayout::prepareFloatingState()
260{
261 auto& floatingState = m_inlineFormattingState.floatingState();
262 floatingState.clear();
263
264 if (!flow().containsFloats())
265 return;
266
267 for (auto& floatingObject : *flow().floatingObjectSet()) {
268 auto& rect = floatingObject->frameRect();
269 auto position = floatingObject->type() == FloatingObject::FloatRight
270 ? Layout::FloatingState::FloatItem::Position::Right
271 : Layout::FloatingState::FloatItem::Position::Left;
272 auto boxGeometry = Layout::BoxGeometry { };
273 // FIXME: We are flooring here for legacy compatibility.
274 // See FloatingObjects::intervalForFloatingObject and RenderBlockFlow::clearFloats.
275 auto y = rect.y().floor();
276 auto maxY = rect.maxY().floor();
277 boxGeometry.setLogicalTopLeft({ rect.x(), y });
278 boxGeometry.setContentBoxWidth(rect.width());
279 boxGeometry.setContentBoxHeight(maxY - y);
280 boxGeometry.setBorder({ });
281 boxGeometry.setPadding({ });
282 boxGeometry.setHorizontalMargin({ });
283 boxGeometry.setVerticalMargin({ });
284 floatingState.append({ position, boxGeometry });
285 }

361TextRunIterator LineLayout::textRunsFor(const RenderText& renderText) const
362{
363 if (!m_inlineContent)
364 return { };
365 auto& layoutBox = m_boxTree.layoutBoxForRenderer(renderText);
366
367 auto firstIndex = [&]() -> std::optional<size_t> {
368 for (size_t i = 0; i < m_inlineContent->runs.size(); ++i) {
369 if (&m_inlineContent->runs[i].layoutBox() == &layoutBox)
370 return i;
371 }
372 return { };
373 }();
374
375 if (!firstIndex)
376 return { };
377
378 return { RunIteratorModernPath(*m_inlineContent, *firstIndex) };
379}

444void LineLayout::paint(PaintInfo& paintInfo, const LayoutPoint& paintOffset)
445{
446 if (!m_inlineContent)
447 return;
448
449 if (paintInfo.phase != PaintPhase::Foreground && paintInfo.phase != PaintPhase::EventRegion)
450 return;
451
452 auto& inlineContent = *m_inlineContent;
453 float deviceScaleFactor = flow().document().deviceScaleFactor();
454
455 auto paintRect = paintInfo.rect;
456 paintRect.moveBy(-paintOffset);
457
458 for (auto& run : inlineContent.runsForRect(paintRect)) {
459 if (!run.textContent()) {
460 auto& renderer = m_boxTree.rendererForLayoutBox(run.layoutBox());
461 if (renderer.isReplaced() && is<RenderBox>(renderer)) {
462 auto& renderBox = downcast<RenderBox>(renderer);
463 if (renderBox.hasSelfPaintingLayer())
464 continue;
465 if (!paintInfo.shouldPaintWithinRoot(renderBox))
466 continue;
467 renderBox.paintAsInlineBlock(paintInfo, paintOffset);
468 }
469 continue;
470 }
471
472 auto& textContent = *run.textContent();
473 if (!textContent.length())
474 continue;
475
476 auto& style = run.style();
477 if (style.visibility() != Visibility::Visible)
478 continue;
479
480 auto rect = FloatRect { run.rect() };
481 auto visualOverflowRect = FloatRect { run.inkOverflow() };
482 if (paintRect.y() > visualOverflowRect.maxY() || paintRect.maxY() < visualOverflowRect.y())
483 continue;
484
485 if (paintInfo.eventRegionContext) {
486 if (style.pointerEvents() != PointerEvents::None) {
487 visualOverflowRect.moveBy(paintOffset);
488 paintInfo.eventRegionContext->unite(enclosingIntRect(visualOverflowRect), style);
489 }
490 continue;
491 }
492
493 auto& line = inlineContent.lineForRun(run);
494 auto expansion = run.expansion();
495 // TextRun expects the xPos to be adjusted with the aligment offset (e.g. when the line is center aligned
496 // and the run starts at 100px, due to the horizontal aligment, the xpos is supposed to be at 0px).
497 auto& fontCascade = style.fontCascade();
498 auto xPos = rect.x() - (line.lineBoxLeft() + line.contentLeft());
499 WebCore::TextRun textRun { textContent.renderedContent(), xPos, expansion.horizontalExpansion, expansion.behavior };
500 textRun.setTabSize(!style.collapseWhiteSpace(), style.tabSize());
501
502 TextPainter textPainter(paintInfo.context());
503 textPainter.setFont(fontCascade);
504 textPainter.setStyle(computeTextPaintStyle(flow().frame(), style, paintInfo));
505 textPainter.setGlyphDisplayListIfNeeded(run, paintInfo, fontCascade, paintInfo.context(), textRun);
506
507 auto textOrigin = FloatPoint { paintOffset.x() + rect.x(), roundToDevicePixel(paintOffset.y() + rect.y() + fontCascade.fontMetrics().ascent(), deviceScaleFactor) };
508 textPainter.paint(textRun, rect, textOrigin);
509
510 if (!style.textDecorationsInEffect().isEmpty()) {
511 auto& textRenderer = downcast<RenderText>(m_boxTree.rendererForLayoutBox(run.layoutBox()));
512 auto painter = TextDecorationPainter { paintInfo.context(), style.textDecorationsInEffect(), textRenderer, false, fontCascade };
513 painter.setWidth(rect.width());
514 painter.paintTextDecoration(textRun, textOrigin, rect.location() + paintOffset);
515 }
516 }
517}

519bool LineLayout::hitTest(const HitTestRequest& request, HitTestResult& result, const HitTestLocation& locationInContainer, const LayoutPoint& accumulatedOffset, HitTestAction hitTestAction)
520{
521 if (hitTestAction != HitTestForeground)
522 return false;
523
524 if (!m_inlineContent)
525 return false;
526
527 auto& inlineContent = *m_inlineContent;
528
529 // FIXME: This should do something efficient to find the run range.
530 for (auto& run : WTF::makeReversedRange(inlineContent.runs)) {
531 auto& renderer = m_boxTree.rendererForLayoutBox(run.layoutBox());
532
533 if (is<RenderText>(renderer)) {
534 auto runRect = Layout::toLayoutRect(run.rect());
535 runRect.moveBy(accumulatedOffset);
536
537 if (!locationInContainer.intersects(runRect))
538 continue;
539
540 auto& style = run.style();
541 if (style.visibility() != Visibility::Visible || style.pointerEvents() == PointerEvents::None)
542 continue;
543
544 renderer.updateHitTestResult(result, locationInContainer.point() - toLayoutSize(accumulatedOffset));
545 if (result.addNodeToListBasedTestResult(renderer.nodeForHitTest(), request, locationInContainer, runRect) == HitTestProgress::Stop)
546 return true;
547 continue;
548 }
549
550 if (is<RenderBox>(renderer)) {
551 auto& renderBox = downcast<RenderBox>(renderer);
552 if (renderBox.hasSelfPaintingLayer())
553 continue;
554
555 if (renderBox.hitTest(request, result, locationInContainer, accumulatedOffset)) {
556 renderBox.updateHitTestResult(result, locationInContainer.point() - toLayoutSize(accumulatedOffset));
557 return true;
558 }
559 }
560 }
561
562 for (auto& inlineBox : WTF::makeReversedRange(inlineContent.nonRootInlineBoxes)) {
563 auto inlineBoxRect = Layout::toLayoutRect(inlineBox.rect());
564 inlineBoxRect.moveBy(accumulatedOffset);
565
566 if (!locationInContainer.intersects(inlineBoxRect))
567 continue;
568
569 auto& style = inlineBox.style();
570 if (style.visibility() != Visibility::Visible || style.pointerEvents() == PointerEvents::None)
571 continue;
572
573 auto& renderer = m_boxTree.rendererForLayoutBox(inlineBox.layoutBox());
574
575 renderer.updateHitTestResult(result, locationInContainer.point() - toLayoutSize(accumulatedOffset));
576 if (result.addNodeToListBasedTestResult(renderer.nodeForHitTest(), request, locationInContainer, inlineBoxRect) == HitTestProgress::Stop)
577 return true;
578 }
579
580 return false;
581}
