157TextRunIterator firstTextRunFor(const RenderText& text)
158{
159#if ENABLE(LAYOUT_FORMATTING_CONTEXT)
160 if (auto* lineLayout = LineLayout::containing(text))
161 return lineLayout->textRunsFor(text);
162#endif
163
164 return { RunIteratorLegacyPath { text.firstTextBox() } };
165}
166
167TextRunIterator firstTextRunInTextOrderFor(const RenderText& text)
168{
169 if (text.firstTextBox() && text.containsReversedText()) {
170 Vector<const LegacyInlineBox*> sortedTextBoxes;
171 for (auto* textBox = text.firstTextBox(); textBox; textBox = textBox->nextTextBox())
172 sortedTextBoxes.append(textBox);
173 std::sort(sortedTextBoxes.begin(), sortedTextBoxes.end(), [](auto* a, auto* b) {
174 return LegacyInlineTextBox::compareByStart(downcast<LegacyInlineTextBox>(a), downcast<LegacyInlineTextBox>(b));
175 });
176 auto* first = sortedTextBoxes[0];
177 return { RunIteratorLegacyPath { first, WTFMove(sortedTextBoxes), 0 } };
178 }
179
180 return firstTextRunFor(text);
181}
182
183TextRunRange textRunsFor(const RenderText& text)
184{
185 return { firstTextRunFor(text) };
186}
