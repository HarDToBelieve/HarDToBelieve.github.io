38class RunIteratorModernPath {
39public:
40 RunIteratorModernPath(const InlineContent& inlineContent)
41 : m_inlineContent(&inlineContent)
42 {
43 setAtEnd();
44 }
45 RunIteratorModernPath(const InlineContent& inlineContent, size_t startIndex)
46 : m_inlineContent(&inlineContent)
47 , m_runIndex(startIndex)
48 {
49 }
50
51 bool isText() const { return !!run().textContent(); }
52
53 FloatRect rect() const { return run().rect(); }
54
55 bool isHorizontal() const { return true; }
56 bool dirOverride() const { return false; }
57 bool isLineBreak() const { return run().isLineBreak(); }
58
59 unsigned minimumCaretOffset() const { return isText() ? start() : 0; }
60 unsigned maximumCaretOffset() const { return isText() ? end() : 1; }
61
62 unsigned char bidiLevel() const { return 0; }
63
64 bool hasHyphen() const { return run().textContent()->hasHyphen(); }
65 StringView text() const { return run().textContent()->originalContent(); }
66 unsigned start() const { return run().textContent()->start(); }
67 unsigned end() const { return run().textContent()->end(); }
68 unsigned length() const { return run().textContent()->length(); }
69
70 // FIXME: Make a shared generic version of this.
71 inline unsigned offsetForPosition(float x) const
72 {
73 if (isLineBreak())
74 return 0;
75 auto rect = this->rect();
76 auto localX = x - rect.x();
77 if (localX > rect.width())
78 return length();
79 if (localX < 0)
80 return 0;
81
82 bool includePartialGlyphs = true;
83 return run().style().fontCascade().offsetForPosition(createTextRun(HyphenMode::Ignore), localX, includePartialGlyphs);
84 }
85
86 // FIXME: Make a shared generic version of this.
87 float positionForOffset(unsigned offset) const
88 {
89 ASSERT(offset >= start());
90 ASSERT(offset <= end());
91
92 if (isLineBreak())
93 return rect().x();
94
95 auto endOffset = selectableRange().clamp(offset);
96
97 LayoutRect selectionRect = LayoutRect(rect().x(), 0, 0, 0);
98 TextRun textRun = createTextRun(HyphenMode::Ignore);
99 run().style().fontCascade().adjustSelectionRectForText(textRun, selectionRect, 0, endOffset);
100 return snapRectToDevicePixelsWithWritingDirection(selectionRect, renderer().document().deviceScaleFactor(), textRun.ltr()).maxX();
101 }
102
103 bool isSelectable(unsigned start, unsigned end) const
104 {
105 return selectableRange().intersects(start, end);
106 }
107
108 LayoutRect selectionRect(unsigned rangeStart, unsigned rangeEnd) const
109 {
110 auto [clampedStart, clampedEnd] = selectableRange().clamp(rangeStart, rangeEnd);
111
112 if (clampedStart >= clampedEnd && !(rangeStart == rangeEnd && rangeStart >= start() && rangeStart <= end()))
113 return { };
114
115 auto logicalLeft = LayoutUnit(isHorizontal() ? rect().x() : rect().y());
116 auto logicalRight = LayoutUnit(isHorizontal() ? rect().maxX() : rect().maxY());
117 auto logicalWidth = logicalRight - logicalLeft;
118
119 // FIXME: These should share implementation with the line iterator.
120 auto selectionTop = LayoutUnit::fromFloatRound(line().enclosingContentTop());
121 auto selectionHeight = LayoutUnit::fromFloatRound(line().enclosingContentBottom() - line().enclosingContentTop());
122
123 LayoutRect selectionRect { logicalLeft, selectionTop, logicalWidth, selectionHeight };
124
125 TextRun textRun = createTextRun(HyphenMode::Include);
126 if (clampedStart || clampedEnd != textRun.length())
127 run().style().fontCascade().adjustSelectionRectForText(textRun, selectionRect, clampedStart, clampedEnd);
128
129 return snappedSelectionRect(selectionRect, logicalRight, selectionTop, selectionHeight, isHorizontal());
130 }
131
132 const RenderObject& renderer() const
133 {
134 return m_inlineContent->rendererForLayoutBox(run().layoutBox());
135 }
136
137 void traverseNextTextRun()
138 {
139 ASSERT(!atEnd());
140 ASSERT(run().textContent());
141
142 auto& layoutBox = run().layoutBox();
143
144 ++m_runIndex;
145
146 if (!atEnd() && &layoutBox != &run().layoutBox())
147 setAtEnd();
148 }
149
150 void traverseNextTextRunInTextOrder()
151 {
152 // FIXME: No RTL in LFC.
153 traverseNextTextRun();
154 }
155
156 void traverseNextOnLine()
157 {
158 ASSERT(!atEnd());
159
160 auto oldLineIndex = run().lineIndex();
161
162 ++m_runIndex;
163
164 if (!atEnd() && oldLineIndex != run().lineIndex())
165 setAtEnd();
166 }
167
168 void traversePreviousOnLine()
169 {
170 ASSERT(!atEnd());
171
172 if (!m_runIndex) {
173 setAtEnd();
174 return;
175 }
176
177 auto oldLineIndex = run().lineIndex();
178
179 --m_runIndex;
180
181 if (oldLineIndex != run().lineIndex())
182 setAtEnd();
183 }
184
185 void traverseNextOnLineInLogicalOrder()
186 {
187 traverseNextOnLine();
188 }
189
190 void traversePreviousOnLineInLogicalOrder()
191 {
192 traversePreviousOnLine();
193 }
194
195 bool operator==(const RunIteratorModernPath& other) const { return m_inlineContent == other.m_inlineContent && m_runIndex == other.m_runIndex; }
196
197 bool atEnd() const { return m_runIndex == runs().size() || !run().hasUnderlyingLayout(); }
198
199 LegacyInlineBox* legacyInlineBox() const
200 {
201 return nullptr;
202 }
203
204private:
205 friend class RunIterator;
206
207 TextBoxSelectableRange selectableRange() const
208 {
209 return {
210 start(),
211 length(),
212 run().style().hyphenString().length(),
213 run().isLineBreak()
214 };
215 }
216
217 enum class HyphenMode { Include, Ignore };
218 TextRun createTextRun(HyphenMode hyphenMode) const
219 {
220 auto& style = run().style();
221 auto expansion = run().expansion();
222 auto rect = this->rect();
223 auto xPos = rect.x() - (line().lineBoxLeft() + line().contentLeft());
224
225 auto textForRun = [&] {
226 if (hyphenMode == HyphenMode::Ignore || !hasHyphen())
227 return text().toStringWithoutCopying();
228
229 return makeString(text(), style.hyphenString());
230 }();
231
232 TextRun textRun { textForRun, xPos, expansion.horizontalExpansion, expansion.behavior };
233 textRun.setTabSize(!style.collapseWhiteSpace(), style.tabSize());
234 return textRun;
235 };
236
237 void setAtEnd() { m_runIndex = runs().size(); }
238
239 const InlineContent::Runs& runs() const { return m_inlineContent->runs; }
240 const Run& run() const { return runs()[m_runIndex]; }
241 const Line& line() const { return m_inlineContent->lineForRun(run()); }
242
243 RefPtr<const InlineContent> m_inlineContent;
244 size_t m_runIndex { 0 };
245};
