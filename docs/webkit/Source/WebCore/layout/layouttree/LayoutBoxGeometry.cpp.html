66Rect BoxGeometry::marginBox() const
67{
68 auto borderBox = this->borderBox();
69
70 Rect marginBox;
71 marginBox.setTop(borderBox.top() - marginBefore());
72 marginBox.setLeft(borderBox.left() - marginStart());
73 marginBox.setHeight(borderBox.height() + marginBefore() + marginAfter());
74 marginBox.setWidth(borderBox.width() + marginStart() + marginEnd());
75 return marginBox;
76}
77
78Rect BoxGeometry::borderBox() const
79{
80 Rect borderBox;
81 borderBox.setTopLeft({ });
82 borderBox.setSize({ borderBoxWidth(), borderBoxHeight() });
83 return borderBox;
84}
85
86Rect BoxGeometry::paddingBox() const
87{
88 auto borderBox = this->borderBox();
89
90 Rect paddingBox;
91 paddingBox.setTop(borderBox.top() + borderTop());
92 paddingBox.setLeft(borderBox.left() + borderLeft());
93 paddingBox.setHeight(borderBox.bottom() - verticalSpaceForScrollbar() - borderBottom() - borderTop());
94 paddingBox.setWidth(borderBox.width() - borderRight() - horizontalSpaceForScrollbar() - borderLeft());
95 return paddingBox;
96}
97
98Rect BoxGeometry::contentBox() const
99{
100 Rect contentBox;
101 contentBox.setTop(contentBoxTop());
102 contentBox.setLeft(contentBoxLeft());
103 contentBox.setWidth(contentBoxWidth());
104 contentBox.setHeight(contentBoxHeight());
105 return contentBox;
106}
