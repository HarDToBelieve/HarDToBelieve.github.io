37class BoxGeometry {
38 WTF_MAKE_ISO_ALLOCATED(BoxGeometry);
39public:
40 BoxGeometry(const BoxGeometry&);
41 BoxGeometry() = default;
42 ~BoxGeometry();
43
44 static LayoutUnit borderBoxTop(const BoxGeometry& box) { return box.logicalTop(); }
45 static LayoutUnit borderBoxLeft(const BoxGeometry& box) { return box.logicalLeft(); }
46 static LayoutPoint borderBoxTopLeft(const BoxGeometry& box) { return box.logicalTopLeft(); }
47 static Rect borderBoxRect(const BoxGeometry& box) { return { box.logicalTop(), box.logicalLeft(), box.borderBoxWidth(), box.borderBoxHeight() }; }
48 static Rect marginBoxRect(const BoxGeometry& box) { return { box.logicalTop() - box.marginBefore(), box.logicalLeft() - box.marginStart(), box.marginBoxWidth(), box.marginBoxHeight() }; }

84 LayoutUnit contentBoxHeight() const;

87 LayoutUnit paddingBoxTop() const { return borderTop(); }

90 LayoutUnit paddingBoxRight() const { return paddingBoxLeft() + paddingBoxWidth(); }
91 LayoutUnit paddingBoxHeight() const { return paddingTop().value_or(0) + contentBoxHeight() + paddingBottom().value_or(0); }
92 LayoutUnit paddingBoxWidth() const { return paddingLeft().value_or(0) + contentBoxWidth() + paddingRight().value_or(0); }
93
94 LayoutUnit borderBoxHeight() const { return borderTop() + paddingBoxHeight() + verticalSpaceForScrollbar() + borderBottom(); }
95 LayoutUnit borderBoxWidth() const { return borderLeft() + paddingBoxWidth() + horizontalSpaceForScrollbar() + borderRight(); }
96 LayoutUnit marginBoxHeight() const { return marginBefore() + borderBoxHeight() + marginAfter(); }
97 LayoutUnit marginBoxWidth() const { return marginStart() + borderBoxWidth() + marginEnd(); }
98
99 LayoutUnit verticalMarginBorderAndPadding() const { return marginBefore() + verticalBorder() + verticalPadding().value_or(0) + marginAfter(); }
100 LayoutUnit horizontalMarginBorderAndPadding() const { return marginStart() + horizontalBorder() + horizontalPadding().value_or(0) + marginEnd(); }

193inline LayoutUnit BoxGeometry::logicalTop() const
194{
195 ASSERT(m_hasValidTop && (m_hasPrecomputedMarginBefore || m_hasValidVerticalMargin));
196 return m_topLeft.y();
197}
198
199inline LayoutUnit BoxGeometry::logicalLeft() const
200{
201 ASSERT(m_hasValidLeft && m_hasValidHorizontalMargin);
202 return m_topLeft.x();
203}
204
205inline LayoutPoint BoxGeometry::logicalTopLeft() const
206{
207 ASSERT(m_hasValidTop && (m_hasPrecomputedMarginBefore || m_hasValidVerticalMargin));
208 ASSERT(m_hasValidLeft && m_hasValidHorizontalMargin);
209 return m_topLeft;
210}

374inline std::optional<LayoutUnit> BoxGeometry::verticalPadding() const
375{
376 auto paddingTop = this->paddingTop();
377 auto paddingBottom = this->paddingBottom();
378 if (!paddingTop && !paddingBottom)
379 return { };
380 return paddingTop.value_or(0) + paddingBottom.value_or(0);
381}
382
383inline std::optional<LayoutUnit> BoxGeometry::horizontalPadding() const
384{
385 auto paddingLeft = this->paddingLeft();
386 auto paddingRight = this->paddingRight();
387 if (!paddingLeft && !paddingRight)
388 return { };
389 return paddingLeft.value_or(0) + paddingRight.value_or(0);
390}
