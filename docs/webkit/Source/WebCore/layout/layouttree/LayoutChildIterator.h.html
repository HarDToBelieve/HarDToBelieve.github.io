32namespace WebCore {
33namespace Layout {
34
35template <typename T>
36class LayoutChildIterator : public LayoutIterator<T> {
37public:
38 LayoutChildIterator(const ContainerBox& parent);
39 LayoutChildIterator(const ContainerBox& parent, const T* current);
40 LayoutChildIterator& operator++();
41};
42
43template <typename T>
44class LayoutChildIteratorAdapter {
45public:
46 LayoutChildIteratorAdapter(const ContainerBox& parent);
47 LayoutChildIterator<T> begin() const;
48 LayoutChildIterator<T> end() const;
49 const T* first() const;
50 const T* last() const;
51
52private:
53 const ContainerBox& m_parent;
54};
55
56template <typename T> LayoutChildIteratorAdapter<T> childrenOfType(const ContainerBox&);
57
58// LayoutChildIterator
59
60template <typename T>
61inline LayoutChildIterator<T>::LayoutChildIterator(const ContainerBox& parent)
62 : LayoutIterator<T>(&parent)
63{
64}
65
66template <typename T>
67inline LayoutChildIterator<T>::LayoutChildIterator(const ContainerBox& parent, const T* current)
68 : LayoutIterator<T>(&parent, current)
69{
70}
71
72template <typename T>
73inline LayoutChildIterator<T>& LayoutChildIterator<T>::operator++()
74{
75 return static_cast<LayoutChildIterator<T>&>(LayoutIterator<T>::traverseNextSibling());
76}
77
78// LayoutChildIteratorAdapter
79
80template <typename T>
81inline LayoutChildIteratorAdapter<T>::LayoutChildIteratorAdapter(const ContainerBox& parent)
82 : m_parent(parent)
83{
84}
85
86template <typename T>
87inline LayoutChildIterator<T> LayoutChildIteratorAdapter<T>::begin() const
88{
89 return LayoutChildIterator<T>(m_parent, Traversal::firstChild<T>(m_parent));
90}
91
92template <typename T>
93inline LayoutChildIterator<T> LayoutChildIteratorAdapter<T>::end() const
94{
95 return LayoutChildIterator<T>(m_parent);
96}
97
98template <typename T>
99inline const T* LayoutChildIteratorAdapter<T>::first() const
100{
101 return Traversal::firstChild<T>(m_parent);
102}
103
104template <typename T>
105inline LayoutChildIteratorAdapter<T> childrenOfType(const ContainerBox& parent)
106{
107 return LayoutChildIteratorAdapter<T>(parent);
108}

110}
