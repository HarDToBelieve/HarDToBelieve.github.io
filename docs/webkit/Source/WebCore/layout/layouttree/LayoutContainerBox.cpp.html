39ContainerBox::ContainerBox(std::optional<ElementAttributes> attributes, RenderStyle&& style, OptionSet<BaseTypeFlag> baseTypeFlags)
40 : Box(attributes, WTFMove(style), baseTypeFlags | ContainerBoxFlag)
41{
42}
43
44const Box* ContainerBox::firstInFlowChild() const
45{
46 if (auto* firstChild = this->firstChild()) {
47 if (firstChild->isInFlow())
48 return firstChild;
49 return firstChild->nextInFlowSibling();
50 }
51 return nullptr;
52}
53
54const Box* ContainerBox::firstInFlowOrFloatingChild() const
55{
56 if (auto* firstChild = this->firstChild()) {
57 if (firstChild->isInFlow() || firstChild->isFloatingPositioned())
58 return firstChild;
59 return firstChild->nextInFlowOrFloatingSibling();
60 }
61 return nullptr;
62}
63
64const Box* ContainerBox::lastInFlowChild() const
65{
66 if (auto* lastChild = this->lastChild()) {
67 if (lastChild->isInFlow())
68 return lastChild;
69 return lastChild->previousInFlowSibling();
70 }
71 return nullptr;
72}
73
74const Box* ContainerBox::lastInFlowOrFloatingChild() const
75{
76 if (auto* lastChild = this->lastChild()) {
77 if (lastChild->isInFlow() || lastChild->isFloatingPositioned())
78 return lastChild;
79 return lastChild->previousInFlowOrFloatingSibling();
80 }
81 return nullptr;
82}
83
84void ContainerBox::setFirstChild(Box& childBox)
85{
86 m_firstChild = &childBox;
87}
88
89void ContainerBox::setLastChild(Box& childBox)
90{
91 m_lastChild = &childBox;
92}
93
94void ContainerBox::appendChild(Box& childBox)
95{
96 childBox.setParent(*this);
97
98 if (m_lastChild) {
99 m_lastChild->setNextSibling(childBox);
100 childBox.setPreviousSibling(*m_lastChild);
101 } else
102 m_firstChild = &childBox;
103
104 m_lastChild = &childBox;
105}
