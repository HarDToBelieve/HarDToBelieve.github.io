32namespace WebCore {
33namespace Layout {
34
35class LayoutContainingBlockChainIterator {
36public:
37 LayoutContainingBlockChainIterator() = default;
38 LayoutContainingBlockChainIterator(const ContainerBox*);
39 const ContainerBox& operator*() const { return *m_current; }
40 const ContainerBox* operator->() const { return m_current; }
41
42 LayoutContainingBlockChainIterator& operator++();
43 bool operator==(const LayoutContainingBlockChainIterator& other) const { return m_current == other.m_current; }
44 bool operator!=(const LayoutContainingBlockChainIterator& other) const { return !(*this == other); }
45
46private:
47 const ContainerBox* m_current { nullptr };
48};
49
50class LayoutContainingBlockChainIteratorAdapter {
51public:
52 LayoutContainingBlockChainIteratorAdapter(const ContainerBox&, const ContainerBox* stayWithin = nullptr);
53 auto begin() { return LayoutContainingBlockChainIterator(&m_containingBlock); }
54 auto end() { return LayoutContainingBlockChainIterator(m_stayWithin); }
55
56private:
57 const ContainerBox& m_containingBlock;
58 const ContainerBox* m_stayWithin { nullptr };
59};
60
61LayoutContainingBlockChainIteratorAdapter containingBlockChain(const Box&);
62LayoutContainingBlockChainIteratorAdapter containingBlockChain(const Box&, const ContainerBox& stayWithin);
63LayoutContainingBlockChainIteratorAdapter containingBlockChainWithinFormattingContext(const Box&);
64
65inline LayoutContainingBlockChainIterator::LayoutContainingBlockChainIterator(const ContainerBox* current)
66 : m_current(current)
67{
68}
69
70inline LayoutContainingBlockChainIterator& LayoutContainingBlockChainIterator::operator++()
71{
72 ASSERT(m_current);
73 m_current = m_current->isInitialContainingBlock() ? nullptr : &m_current->containingBlock();
74 return *this;
75}
76
77inline LayoutContainingBlockChainIteratorAdapter::LayoutContainingBlockChainIteratorAdapter(const ContainerBox& containingBlock, const ContainerBox* stayWithin)
78 : m_containingBlock(containingBlock)
79 , m_stayWithin(stayWithin)
80{
81}
82
83inline LayoutContainingBlockChainIteratorAdapter containingBlockChain(const Box& layoutBox)
84{
85 return LayoutContainingBlockChainIteratorAdapter(layoutBox.containingBlock());
86}
87
88inline LayoutContainingBlockChainIteratorAdapter containingBlockChain(const Box& layoutBox, const ContainerBox& stayWithin)
89{
90 ASSERT(layoutBox.isDescendantOf(stayWithin));
91 return LayoutContainingBlockChainIteratorAdapter(layoutBox.containingBlock(), &stayWithin);
92}
93
94inline LayoutContainingBlockChainIteratorAdapter containingBlockChainWithinFormattingContext(const Box& layoutBox)
95{
96 return containingBlockChain(layoutBox, layoutBox.formattingContextRoot());
97}
