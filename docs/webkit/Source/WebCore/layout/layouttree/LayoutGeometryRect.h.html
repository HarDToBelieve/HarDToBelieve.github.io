33namespace WebCore {
34namespace Layout {
35
36class Rect {
37public:
38 Rect() = default;
39 Rect(LayoutUnit top, LayoutUnit left, LayoutUnit width, LayoutUnit height);
40 Rect(LayoutPoint topLeft, LayoutUnit width, LayoutUnit height);
41 Rect(const LayoutPoint& topLeft, const LayoutSize&);
42
43 LayoutUnit top() const;
44 LayoutUnit left() const;
45 LayoutPoint topLeft() const;
46
47 LayoutUnit bottom() const;
48 LayoutUnit right() const;
49 LayoutPoint bottomRight() const;
50
51 LayoutUnit width() const;
52 LayoutUnit height() const;
53 LayoutSize size() const;
54
55 void setTop(LayoutUnit);
56 void setLeft(LayoutUnit);
57 void setTopLeft(const LayoutPoint&);
58 void setWidth(LayoutUnit);
59 void setHeight(LayoutUnit);
60 void setSize(const LayoutSize&);
61
62 void shiftLeftTo(LayoutUnit);
63 void shiftRightTo(LayoutUnit);
64 void shiftTopTo(LayoutUnit);
65 void shiftBottomTo(LayoutUnit);
66
67 void moveHorizontally(LayoutUnit);
68 void moveVertically(LayoutUnit);
69 void moveBy(LayoutPoint);
70
71 void expand(std::optional<LayoutUnit>, std::optional<LayoutUnit>);
72 void expandHorizontally(LayoutUnit delta) { expand(delta, { }); }
73 void expandVertically(LayoutUnit delta) { expand({ }, delta); }
74 void expandToContain(const Rect&);
75 bool intersects(const Rect& rect) const { return m_rect.intersects(rect); }
76
77 Rect clone() const;
78 operator LayoutRect() const;
79
80private:
81#if ASSERT_ENABLED
82 void invalidateTop() { m_hasValidTop = false; }
83 void invalidateLeft() { m_hasValidLeft = false; }
84 void invalidateWidth() { m_hasValidWidth = false; }
85 void invalidateHeight() { m_hasValidHeight = false; }
86 void invalidatePosition();
87
88 bool hasValidPosition() const { return m_hasValidTop && m_hasValidLeft; }
89 bool hasValidSize() const { return m_hasValidWidth && m_hasValidHeight; }
90 bool hasValidGeometry() const { return hasValidPosition() && hasValidSize(); }
91
92 void setHasValidPosition();
93 void setHasValidSize();
94
95 bool m_hasValidTop { false };
96 bool m_hasValidLeft { false };
97 bool m_hasValidWidth { false };
98 bool m_hasValidHeight { false };
99#endif // ASSERT_ENABLED
100 LayoutRect m_rect;
101};
102
103inline Rect::Rect(LayoutUnit top, LayoutUnit left, LayoutUnit width, LayoutUnit height)
104 : m_rect(left, top, width, height)
105{
106#if ASSERT_ENABLED
107 m_hasValidTop = true;
108 m_hasValidLeft = true;
109 m_hasValidWidth = true;
110 m_hasValidHeight = true;
111#endif
112}
113
114inline Rect::Rect(LayoutPoint topLeft, LayoutUnit width, LayoutUnit height)
115 : Rect(topLeft.y(), topLeft.x(), width, height)
116{
117}
118
119inline Rect::Rect(const LayoutPoint& topLeft, const LayoutSize& size)
120 : Rect(topLeft.y(), topLeft.x(), size.width(), size.height())
121{
122}
123
124#if ASSERT_ENABLED
125inline void Rect::invalidatePosition()
126{
127 invalidateTop();
128 invalidateLeft();
129}
130
131inline void Rect::setHasValidPosition()
132{
133 m_hasValidTop = true;
134 m_hasValidLeft = true;
135}
136
137inline void Rect::setHasValidSize()
138{
139 m_hasValidWidth = true;
140 m_hasValidHeight = true;
141}
142#endif
143
144inline LayoutUnit Rect::top() const
145{
146 ASSERT(m_hasValidTop);
147 return m_rect.y();
148}
149
150inline LayoutUnit Rect::left() const
151{
152 ASSERT(m_hasValidLeft);
153 return m_rect.x();
154}
155
156inline LayoutUnit Rect::bottom() const
157{
158 ASSERT(m_hasValidTop && m_hasValidHeight);
159 return m_rect.maxY();
160}
161
162inline LayoutUnit Rect::right() const
163{
164 ASSERT(m_hasValidLeft && m_hasValidWidth);
165 return m_rect.maxX();
166}
167
168inline LayoutPoint Rect::topLeft() const
169{
170 ASSERT(hasValidPosition());
171 return m_rect.minXMinYCorner();
172}
173
174inline LayoutPoint Rect::bottomRight() const
175{
176 ASSERT(hasValidGeometry());
177 return m_rect.maxXMaxYCorner();
178}
179
180inline LayoutSize Rect::size() const
181{
182 ASSERT(hasValidSize());
183 return m_rect.size();
184}
185
186inline LayoutUnit Rect::width() const
187{
188 ASSERT(m_hasValidWidth);
189 return m_rect.width();
190}
191
192inline LayoutUnit Rect::height() const
193{
194 ASSERT(m_hasValidHeight);
195 return m_rect.height();
196}
197
198inline void Rect::setTopLeft(const LayoutPoint& topLeft)
199{
200#if ASSERT_ENABLED
201 setHasValidPosition();
202#endif
203 m_rect.setLocation(topLeft);
204}
205
206inline void Rect::setTop(LayoutUnit top)
207{
208#if ASSERT_ENABLED
209 m_hasValidTop = true;
210#endif
211 m_rect.setY(top);
212}
213
214inline void Rect::setLeft(LayoutUnit left)
215{
216#if ASSERT_ENABLED
217 m_hasValidLeft = true;
218#endif
219 m_rect.setX(left);
220}
221
222inline void Rect::setWidth(LayoutUnit width)
223{
224#if ASSERT_ENABLED
225 m_hasValidWidth = true;
226#endif
227 m_rect.setWidth(width);
228}
229
230inline void Rect::setHeight(LayoutUnit height)
231{
232#if ASSERT_ENABLED
233 m_hasValidHeight = true;
234#endif
235 m_rect.setHeight(height);
236}
237
238inline void Rect::setSize(const LayoutSize& size)
239{
240#if ASSERT_ENABLED
241 setHasValidSize();
242#endif
243 m_rect.setSize(size);
244}
245
246inline void Rect::shiftLeftTo(LayoutUnit left)
247{
248 ASSERT(m_hasValidLeft);
249 m_rect.shiftXEdgeTo(left);
250}
251
252inline void Rect::shiftRightTo(LayoutUnit right)
253{
254 ASSERT(m_hasValidLeft && m_hasValidWidth);
255 m_rect.shiftMaxXEdgeTo(right);
256}
257
258inline void Rect::shiftTopTo(LayoutUnit top)
259{
260 ASSERT(m_hasValidTop);
261 m_rect.shiftYEdgeTo(top);
262}
263
264inline void Rect::shiftBottomTo(LayoutUnit bottom)
265{
266 ASSERT(m_hasValidTop && m_hasValidHeight);
267 m_rect.shiftMaxYEdgeTo(bottom);
268}
269
270inline void Rect::moveHorizontally(LayoutUnit offset)
271{
272 ASSERT(m_hasValidLeft);
273 m_rect.move(LayoutSize { offset, 0 });
274}
275
276inline void Rect::moveVertically(LayoutUnit offset)
277{
278 ASSERT(m_hasValidTop);
279 m_rect.move(LayoutSize { 0, offset });
280}
281
282inline void Rect::moveBy(LayoutPoint offset)
283{
284 ASSERT(m_hasValidTop);
285 ASSERT(m_hasValidLeft);
286 m_rect.moveBy(offset);
287}
288
289inline void Rect::expand(std::optional<LayoutUnit> width, std::optional<LayoutUnit> height)
290{
291 ASSERT(!width || m_hasValidWidth);
292 ASSERT(!height || m_hasValidHeight);
293 m_rect.expand(width.value_or(0), height.value_or(0));
294}
295
296inline void Rect::expandToContain(const Rect& rect)
297{
298 ASSERT(m_hasValidWidth);
299 ASSERT(m_hasValidHeight);
300 m_rect.uniteEvenIfEmpty(rect);
301}
302
303inline Rect Rect::clone() const
304{
305 Rect rect;
306#if ASSERT_ENABLED
307 rect.m_hasValidTop = m_hasValidTop;
308 rect.m_hasValidLeft = m_hasValidLeft;
309 rect.m_hasValidWidth = m_hasValidWidth;
310 rect.m_hasValidHeight = m_hasValidHeight;
311#endif
312 rect.m_rect = m_rect;
313 return rect;
314}
315
316inline Rect::operator LayoutRect() const
317{
318 ASSERT(hasValidGeometry());
319 return m_rect;
320}
