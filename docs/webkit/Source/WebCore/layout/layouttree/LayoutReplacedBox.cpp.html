34namespace WebCore {

37WTF_MAKE_ISO_ALLOCATED_IMPL(ReplacedBox);
38
39ReplacedBox::ReplacedBox(std::optional<ElementAttributes> elementAttributes, RenderStyle&& style)
40 : Box(elementAttributes, WTFMove(style), Box::ReplacedBoxFlag)
41{
42}
43
44bool ReplacedBox::hasIntrinsicWidth() const
45{
46 return m_intrinsicSize || style().logicalWidth().isIntrinsic();
47}
48
49bool ReplacedBox::hasIntrinsicHeight() const
50{
51 return m_intrinsicSize || style().logicalHeight().isIntrinsic();
52}
53
54bool ReplacedBox::hasIntrinsicRatio() const
55{
56 if (!hasAspectRatio())
57 return false;
58 return m_intrinsicSize || m_intrinsicRatio;
59}
60
61LayoutUnit ReplacedBox::intrinsicWidth() const
62{
63 ASSERT(hasIntrinsicWidth());
64 if (m_intrinsicSize)
65 return m_intrinsicSize->width();
66 return LayoutUnit { style().logicalWidth().value() };
67}
68
69LayoutUnit ReplacedBox::intrinsicHeight() const
70{
71 ASSERT(hasIntrinsicHeight());
72 if (m_intrinsicSize)
73 return m_intrinsicSize->height();
74 return LayoutUnit { style().logicalHeight().value() };
75}
76
77LayoutUnit ReplacedBox::intrinsicRatio() const
78{
79 ASSERT(hasIntrinsicRatio() || (hasIntrinsicWidth() && hasIntrinsicHeight()));
80 if (m_intrinsicRatio)
81 return *m_intrinsicRatio;
82 if (m_intrinsicSize && m_intrinsicSize->height())
83 return m_intrinsicSize->width() / m_intrinsicSize->height();
84 return 1;
85}
86
87bool ReplacedBox::hasAspectRatio() const
88{
89 return isImage();
90}
