75static void appendChild(ContainerBox& parent, Box& newChild)
76{
77 if (!parent.hasChild()) {
78 parent.setFirstChild(newChild);
79 parent.setLastChild(newChild);
80 newChild.setParent(parent);
81 return;
82 }
83
84 auto& lastChild = const_cast<Box&>(*parent.lastChild());
85 lastChild.setNextSibling(newChild);
86 newChild.setPreviousSibling(lastChild);
87 newChild.setParent(parent);
88 parent.setLastChild(newChild);
89}

91static std::optional<LayoutSize> accumulatedOffsetForInFlowPositionedContinuation(const RenderBox& block)
92{
93 // FIXE: This is a workaround of the continuation logic when the relatively positioned parent inline box
94 // becomes a sibling box of this block and only reachable through the continuation link which we don't have here.
95 if (!block.isAnonymous() || !block.isInFlowPositioned() || !block.isContinuation())
96 return { };
97 return block.relativePositionOffset();
98}

100static bool canUseSimplifiedTextMeasuring(const StringView& content, const FontCascade& font, bool whitespaceIsCollapsed)
101{
102 if (font.codePath(TextRun(content)) == FontCascade::CodePath::Complex)
103 return false;
104
105 if (font.wordSpacing() || font.letterSpacing())
106 return false;
107
108 for (unsigned i = 0; i < content.length(); ++i) {
109 if ((!whitespaceIsCollapsed && content[i] == '\t') || content[i] == noBreakSpace || content[i] == softHyphen || content[i] >= HiraganaLetterSmallA)
110 return false;
111 }
112 return true;
113}

115std::unique_ptr<Layout::LayoutTree> TreeBuilder::buildLayoutTree(const RenderView& renderView)
116{
117 PhaseScope scope(Phase::Type::TreeBuilding);
118
119 auto rootStyle = RenderStyle::clone(renderView.style());
120 rootStyle.setLogicalWidth(Length(renderView.width(), LengthType::Fixed));
121 rootStyle.setLogicalHeight(Length(renderView.height(), LengthType::Fixed));
122
123 auto rootLayoutBox = makeUnique<InitialContainingBlock>(WTFMove(rootStyle));
124 auto& rootContainer = *rootLayoutBox;
125 auto layoutTree = makeUnique<LayoutTree>();
126 layoutTree->append(WTFMove(rootLayoutBox));
127 TreeBuilder(*layoutTree).buildSubTree(renderView, rootContainer);
128 return layoutTree;
129}

168Box* TreeBuilder::createLayoutBox(const ContainerBox& parentContainer, const RenderObject& childRenderer)
169{
170 auto elementAttributes = [] (const RenderElement& renderer) -> std::optional<Box::ElementAttributes> {
171 if (renderer.isDocumentElementRenderer())
172 return Box::ElementAttributes { Box::ElementType::Document };
173 if (auto* element = renderer.element()) {
174 if (element->hasTagName(HTMLNames::bodyTag))
175 return Box::ElementAttributes { Box::ElementType::Body };
176 if (element->hasTagName(HTMLNames::imgTag))
177 return Box::ElementAttributes { Box::ElementType::Image };
178 if (element->hasTagName(HTMLNames::iframeTag))
179 return Box::ElementAttributes { Box::ElementType::IFrame };
180 return Box::ElementAttributes { Box::ElementType::GenericElement };
181 }
182 return std::nullopt;
183 };
184
185 Box* childLayoutBox = nullptr;
186 if (is<RenderText>(childRenderer)) {
187 auto& textRenderer = downcast<RenderText>(childRenderer);
188 // RenderText::text() has already applied text-transform and text-security properties.
189 String text = textRenderer.text();
190 auto useSimplifiedTextMeasuring = canUseSimplifiedTextMeasuring(text, parentContainer.style().fontCascade(), parentContainer.style().collapseWhiteSpace());
191 if (parentContainer.style().display() == DisplayType::Inline)
192 childLayoutBox = &createTextBox(text, useSimplifiedTextMeasuring, RenderStyle::clone(parentContainer.style()));
193 else
194 childLayoutBox = &createTextBox(text, useSimplifiedTextMeasuring, RenderStyle::createAnonymousStyleWithDisplay(parentContainer.style(), DisplayType::Inline));
195 } else {
196 auto& renderer = downcast<RenderElement>(childRenderer);
197 auto displayType = renderer.style().display();
198
199 auto clonedStyle = RenderStyle::clone(renderer.style());
200
201 if (is<RenderLineBreak>(renderer)) {
202 clonedStyle.setDisplay(DisplayType::Inline);
203 clonedStyle.setFloating(Float::None);
204 clonedStyle.setPosition(PositionType::Static);
205 childLayoutBox = &createLineBreakBox(downcast<RenderLineBreak>(childRenderer).isWBR(), WTFMove(clonedStyle));
206 } else if (is<RenderTable>(renderer)) {
207 // Construct the principal table wrapper box (and not the table box itself).
208 // The computed values of properties 'position', 'float', 'margin-*', 'top', 'right', 'bottom', and 'left' on the table element
209 // are used on the table wrapper box and not the table box; all other values of non-inheritable properties are used
210 // on the table box and not the table wrapper box.
211 auto tableWrapperBoxStyle = RenderStyle::createAnonymousStyleWithDisplay(parentContainer.style(), renderer.style().display() == DisplayType::Table ? DisplayType::Block : DisplayType::Inline);
212 tableWrapperBoxStyle.setPosition(renderer.style().position());
213 tableWrapperBoxStyle.setFloating(renderer.style().floating());
214
215 tableWrapperBoxStyle.setTop(Length { renderer.style().top() });
216 tableWrapperBoxStyle.setLeft(Length { renderer.style().left() });
217 tableWrapperBoxStyle.setBottom(Length { renderer.style().bottom() });
218 tableWrapperBoxStyle.setRight(Length { renderer.style().right() });
219
220 tableWrapperBoxStyle.setMarginTop(Length { renderer.style().marginTop() });
221 tableWrapperBoxStyle.setMarginLeft(Length { renderer.style().marginLeft() });
222 tableWrapperBoxStyle.setMarginBottom(Length { renderer.style().marginBottom() });
223 tableWrapperBoxStyle.setMarginRight(Length { renderer.style().marginRight() });
224
225 childLayoutBox = &createContainer(Box::ElementAttributes { Box::ElementType::TableWrapperBox }, WTFMove(tableWrapperBoxStyle));
226 childLayoutBox->setIsAnonymous();
227 } else if (is<RenderReplaced>(renderer)) {
228 childLayoutBox = &createReplacedBox(elementAttributes(renderer), WTFMove(clonedStyle));
229 // FIXME: We don't yet support all replaced elements and this is temporary anyway.
230 downcast<ReplacedBox>(*childLayoutBox).setIntrinsicSize(downcast<RenderReplaced>(renderer).intrinsicSize());
231 if (is<RenderImage>(renderer)) {
232 auto& imageRenderer = downcast<RenderImage>(renderer);
233 if (imageRenderer.shouldDisplayBrokenImageIcon())
234 downcast<ReplacedBox>(*childLayoutBox).setIntrinsicRatio(1);
235 if (imageRenderer.cachedImage())
236 downcast<ReplacedBox>(*childLayoutBox).setCachedImage(*imageRenderer.cachedImage());
237 }
238 } else {
239 if (displayType == DisplayType::Block) {
240 if (auto offset = accumulatedOffsetForInFlowPositionedContinuation(downcast<RenderBox>(renderer))) {
241 clonedStyle.setTop({ offset->height(), LengthType::Fixed });
242 clonedStyle.setLeft({ offset->width(), LengthType::Fixed });
243 childLayoutBox = &createContainer(elementAttributes(renderer), WTFMove(clonedStyle));
244 } else
245 childLayoutBox = &createContainer(elementAttributes(renderer), WTFMove(clonedStyle));
246 } else if (displayType == DisplayType::Flex)
247 childLayoutBox = &createContainer(elementAttributes(renderer), WTFMove(clonedStyle));
248 else if (displayType == DisplayType::Inline)
249 childLayoutBox = &createContainer(elementAttributes(renderer), WTFMove(clonedStyle));
250 else if (displayType == DisplayType::InlineBlock)
251 childLayoutBox = &createContainer(elementAttributes(renderer), WTFMove(clonedStyle));
252 else if (displayType == DisplayType::TableCaption || displayType == DisplayType::TableCell) {
253 childLayoutBox = &createContainer(elementAttributes(renderer), WTFMove(clonedStyle));
254 } else if (displayType == DisplayType::TableRowGroup || displayType == DisplayType::TableHeaderGroup || displayType == DisplayType::TableFooterGroup
255 || displayType == DisplayType::TableRow || displayType == DisplayType::TableColumnGroup) {
256 childLayoutBox = &createContainer(elementAttributes(renderer), WTFMove(clonedStyle));
257 } else if (displayType == DisplayType::TableColumn) {
258 childLayoutBox = &createContainer(elementAttributes(renderer), WTFMove(clonedStyle));
259 auto& tableColElement = static_cast<HTMLTableColElement&>(*renderer.element());
260 auto columnWidth = tableColElement.width();
261 if (!columnWidth.isEmpty())
262 childLayoutBox->setColumnWidth(parseHTMLInteger(columnWidth).value_or(0));
263 if (tableColElement.span() > 1)
264 childLayoutBox->setColumnSpan(tableColElement.span());
265 } else {
266 ASSERT_NOT_IMPLEMENTED_YET();
267 // Let's fall back to a regular block level container when the renderer type is not yet supported.
268 clonedStyle.setDisplay(DisplayType::Block);
269 childLayoutBox = &createContainer(elementAttributes(renderer), WTFMove(clonedStyle));
270 }
271 }
272
273 if (is<RenderTableCell>(renderer)) {
274 auto* tableCellElement = renderer.element();
275 if (is<HTMLTableCellElement>(tableCellElement)) {
276 auto& cellElement = downcast<HTMLTableCellElement>(*tableCellElement);
277 auto rowSpan = cellElement.rowSpan();
278 if (rowSpan > 1)
279 childLayoutBox->setRowSpan(rowSpan);
280 auto columnSpan = cellElement.colSpan();
281 if (columnSpan > 1)
282 childLayoutBox->setColumnSpan(columnSpan);
283 }
284 }
285
286 if (childRenderer.isAnonymous())
287 childLayoutBox->setIsAnonymous();
288 }
289 return childLayoutBox;
290}

292void TreeBuilder::buildTableStructure(const RenderTable& tableRenderer, ContainerBox& tableWrapperBox)
293{
294 // Create caption and table box.
295 auto* tableChild = tableRenderer.firstChild();
296 while (is<RenderTableCaption>(tableChild)) {
297 auto& captionRenderer = *tableChild;
298 auto* captionBox = createLayoutBox(tableWrapperBox, captionRenderer);
299 appendChild(tableWrapperBox, *captionBox);
300 auto& captionContainer = downcast<ContainerBox>(*captionBox);
301 buildSubTree(downcast<RenderElement>(captionRenderer), captionContainer);
302 tableChild = tableChild->nextSibling();
303 }
304
305 auto tableBoxStyle = RenderStyle::clone(tableRenderer.style());
306 tableBoxStyle.setPosition(PositionType::Static);
307 tableBoxStyle.setFloating(Float::None);
308 tableBoxStyle.resetMargin();
309 // FIXME: Figure out where the spec says table width is like box-sizing: border-box;
310 if (is<HTMLTableElement>(tableRenderer.element()))
311 tableBoxStyle.setBoxSizing(BoxSizing::BorderBox);
312 auto& tableBox = createContainer(Box::ElementAttributes { Box::ElementType::TableBox }, WTFMove(tableBoxStyle));
313 appendChild(tableWrapperBox, tableBox);
314 auto* sectionRenderer = tableChild;
315 while (sectionRenderer) {
316 auto* sectionBox = createLayoutBox(tableBox, *sectionRenderer);
317 appendChild(tableBox, *sectionBox);
318 auto& sectionContainer = downcast<ContainerBox>(*sectionBox);
319 buildSubTree(downcast<RenderElement>(*sectionRenderer), sectionContainer);
320 sectionRenderer = sectionRenderer->nextSibling();
321 }
322 auto addMissingTableCells = [&] (auto& tableBody) {
323 // A "missing cell" is a cell in the row/column grid that is not occupied by an element or pseudo-element.
324 // Missing cells are rendered as if an anonymous table-cell box occupied their position in the grid.
325
326 // Find the max number of columns and fill in the gaps.
327 size_t maximumColumns = 0;
328 size_t currentRow = 0;
329 Vector<size_t> numberOfCellsPerRow;
330 for (auto& rowBox : childrenOfType<ContainerBox>(tableBody)) {
331 if (numberOfCellsPerRow.size() <= currentRow) {
332 // Ensure we always have a vector entry for the current row -even when the row is empty.
333 numberOfCellsPerRow.append({ });
334 }
335 for (auto& cellBox : childrenOfType<ContainerBox>(rowBox)) {
336 auto numberOfSpannedColumns = cellBox.columnSpan();
337 for (size_t rowSpan = 0; rowSpan < cellBox.rowSpan(); ++rowSpan) {
338 auto rowIndexWithSpan = currentRow + rowSpan;
339 if (numberOfCellsPerRow.size() <= rowIndexWithSpan) {
340 // This is where we advance from the current row by having a row spanner.
341 numberOfCellsPerRow.append(numberOfSpannedColumns);
342 continue;
343 }
344 numberOfCellsPerRow[rowIndexWithSpan] += numberOfSpannedColumns;
345 }
346 }
347 maximumColumns = std::max(maximumColumns, numberOfCellsPerRow[currentRow]);
348 ++currentRow;
349 }
350 // Fill in the gaps.
351 size_t rowIndex = 0;
352 for (auto& rowBox : childrenOfType<ContainerBox>(tableBody)) {
353 ASSERT(maximumColumns >= numberOfCellsPerRow[rowIndex]);
354 auto numberOfMissingCells = maximumColumns - numberOfCellsPerRow[rowIndex++];
355 for (size_t i = 0; i < numberOfMissingCells; ++i)
356 appendChild(const_cast<ContainerBox&>(rowBox), createContainer({ }, RenderStyle::createAnonymousStyleWithDisplay(rowBox.style(), DisplayType::TableCell)));
357 }
358 };
359
360 for (auto& section : childrenOfType<ContainerBox>(tableBox)) {
361 // FIXME: Check if headers and footers need the same treatment.
362 if (!section.isTableBody())
363 continue;
364 addMissingTableCells(section);
365 }
366}

368void TreeBuilder::buildSubTree(const RenderElement& parentRenderer, ContainerBox& parentContainer)
369{
370 for (auto& childRenderer : childrenOfType<RenderObject>(parentRenderer)) {
371 auto* childLayoutBox = createLayoutBox(parentContainer, childRenderer);
372 appendChild(parentContainer, *childLayoutBox);
373 if (childLayoutBox->isTableWrapperBox())
374 buildTableStructure(downcast<RenderTable>(childRenderer), downcast<ContainerBox>(*childLayoutBox));
375 else if (is<ContainerBox>(*childLayoutBox))
376 buildSubTree(downcast<RenderElement>(childRenderer), downcast<ContainerBox>(*childLayoutBox));
377 }
378}

381void showInlineTreeAndRuns(TextStream& stream, const LayoutState& layoutState, const ContainerBox& inlineFormattingRoot, size_t depth)
382{
383 auto& inlineFormattingState = layoutState.establishedInlineFormattingState(inlineFormattingRoot);
384 auto& lines = inlineFormattingState.lines();
385 auto& lineBoxes = inlineFormattingState.lineBoxes();
386
387 for (size_t lineIndex = 0; lineIndex < lines.size(); ++lineIndex) {
388 auto addSpacing = [&] {
389 size_t printedCharacters = 0;
390 stream << "-------- --";
391 while (++printedCharacters <= depth * 2)
392 stream << " ";
393
394 };
395 addSpacing();
396 auto& line = lines[lineIndex];
397 auto& lineBoxLogicalRect = line.lineBoxLogicalRect();
398 auto enclosingTopAndBottom = line.enclosingTopAndBottom();
399 stream << "line at (" << lineBoxLogicalRect.left() << "," << lineBoxLogicalRect.top() << ") size (" << lineBoxLogicalRect.width() << "x" << lineBoxLogicalRect.height() << ") baseline (" << line.baseline() << ") enclosing top (" << enclosingTopAndBottom.top << ") bottom (" << enclosingTopAndBottom.bottom << ")";
400 stream.nextLine();
401
402 addSpacing();
403 stream << " Inline level boxes:";
404 stream.nextLine();
405
406 auto& lineBox = lineBoxes[lineIndex];
407 auto outputInlineLevelBox = [&](const auto& inlineLevelBox) {
408 addSpacing();
409 stream << " ";
410 auto logicalRect = InlineRect { };
411 auto& layoutBox = inlineLevelBox.layoutBox();
412 if (inlineLevelBox.isRootInlineBox()) {
413 stream << "Root inline box";
414 logicalRect = lineBox.logicalRectForRootInlineBox();
415 } else if (inlineLevelBox.isAtomicInlineLevelBox()) {
416 stream << "Atomic inline level box";
417 logicalRect = lineBox.logicalBorderBoxForAtomicInlineLevelBox(layoutBox, layoutState.geometryForBox(layoutBox));
418 } else if (inlineLevelBox.isLineBreakBox()) {
419 stream << "Line break box";
420 logicalRect = lineBox.logicalRectForLineBreakBox(layoutBox);
421 } else if (inlineLevelBox.isInlineBox()) {
422 stream << "Inline box";
423 logicalRect = lineBox.logicalBorderBoxForInlineBox(layoutBox, layoutState.geometryForBox(layoutBox));
424 } else
425 stream << "Generic inline level box";
426 stream
427 << " at (" << logicalRect.left() << "," << logicalRect.top() << ")"
428 << " size (" << logicalRect.width() << "x" << logicalRect.height() << ")"
429 << " baseline (" << logicalRect.top() + inlineLevelBox.baseline() << ")"
430 << " ascent (" << inlineLevelBox.baseline() << "/" << inlineLevelBox.layoutBounds().ascent << ")"
431 << " descent (" << inlineLevelBox.descent().value_or(0.0f) << "/" << inlineLevelBox.layoutBounds().descent << ")";
432 stream.nextLine();
433 };
434 outputInlineLevelBox(lineBox.rootInlineBox());
435 for (auto& inlineLevelBox : lineBox.nonRootInlineLevelBoxes())
436 outputInlineLevelBox(inlineLevelBox);
437
438 addSpacing();
439 stream << " Runs:";
440 stream.nextLine();
441 for (auto& run : inlineFormattingState.lineRuns()) {
442 if (run.lineIndex() != lineIndex)
443 continue;
444 addSpacing();
445 stream << " ";
446 if (run.text())
447 stream << "text run";
448 else
449 stream << "box run";
450 stream << " at (" << run.logicalLeft() << "," << run.logicalTop() << ") size " << run.logicalWidth() << "x" << run.logicalHeight();
451 if (run.text())
452 stream << " run(" << run.text()->start() << ", " << run.text()->end() << ")";
453 stream.nextLine();
454 }
455
456 }
457}

459static void outputLayoutBox(TextStream& stream, const Box& layoutBox, const BoxGeometry* boxGeometry, unsigned depth)
460{
461 unsigned printedCharacters = 0;
462 while (++printedCharacters <= depth * 2)
463 stream << " ";
464
465 if (layoutBox.isFloatingPositioned())
466 stream << "[float] ";
467
468 if (is<InitialContainingBlock>(layoutBox))
469 stream << "Initial containing block";
470 else if (layoutBox.isDocumentBox())
471 stream << "HTML";
472 else if (layoutBox.isBodyBox())
473 stream << "BODY";
474 else if (layoutBox.isTableWrapperBox())
475 stream << "TABLE wrapper box";
476 else if (layoutBox.isTableBox())
477 stream << "TABLE";
478 else if (layoutBox.isTableCaption())
479 stream << "CAPTION";
480 else if (layoutBox.isTableHeader())
481 stream << "THEAD";
482 else if (layoutBox.isTableBody())
483 stream << "TBODY";
484 else if (layoutBox.isTableFooter())
485 stream << "TFOOT";
486 else if (layoutBox.isTableColumnGroup())
487 stream << "COL GROUP";
488 else if (layoutBox.isTableColumn())
489 stream << "COL";
490 else if (layoutBox.isTableCell())
491 stream << "TD";
492 else if (layoutBox.isTableRow())
493 stream << "TR";
494 else if (layoutBox.isFlexBox())
495 stream << "Flex box";
496 else if (layoutBox.isFlexItem())
497 stream << "Flex item";
498 else if (layoutBox.isInlineLevelBox()) {
499 if (layoutBox.isAnonymous())
500 stream << "anonymous inline box";
501 else if (layoutBox.isInlineBlockBox())
502 stream << "inline-block box";
503 else if (layoutBox.isLineBreakBox())
504 stream << (downcast<LineBreakBox>(layoutBox).isOptional() ? "word break opportunity" : "line break");
505 else if (layoutBox.isAtomicInlineLevelBox())
506 stream << "atomic inline level box";
507 else if (layoutBox.isReplacedBox())
508 stream << "replaced inline box";
509 else if (layoutBox.isInlineBox())
510 stream << "inline box";
511 else
512 stream << "other inline level box";
513 } else if (layoutBox.isBlockLevelBox())
514 stream << "block box";
515 else
516 stream << "unknown box";
517
518 if (boxGeometry) {
519 auto borderBox = BoxGeometry::borderBoxRect(*boxGeometry);
520 stream << " at (" << borderBox.left() << "," << borderBox.top() << ") size " << borderBox.width() << "x" << borderBox.height();
521 }
522 stream << " (" << &layoutBox << ")";
523 if (is<InlineTextBox>(layoutBox)) {
524 auto textContent = downcast<InlineTextBox>(layoutBox).content();
525 stream << " length->(" << textContent.length() << ")";
526
527 textContent.replaceWithLiteral('\\', "\\\\");
528 textContent.replaceWithLiteral('\n', "\\n");
529
530 const size_t maxPrintedLength = 80;
531 if (textContent.length() > maxPrintedLength) {
532 auto substring = textContent.substring(0, maxPrintedLength);
533 stream << " \"" << substring.utf8().data() << "\"...";
534 } else
535 stream << " \"" << textContent.utf8().data() << "\"";
536 }
537 stream.nextLine();
538}

540static void outputLayoutTree(const LayoutState* layoutState, TextStream& stream, const ContainerBox& rootContainer, unsigned depth)
541{
542 for (auto& child : childrenOfType<Box>(rootContainer)) {
543 if (layoutState) {
544 // Not all boxes generate display boxes.
545 if (layoutState->hasBoxGeometry(child))
546 outputLayoutBox(stream, child, &layoutState->geometryForBox(child), depth);
547 else
548 outputLayoutBox(stream, child, nullptr, depth);
549 if (child.establishesInlineFormattingContext())
550 showInlineTreeAndRuns(stream, *layoutState, downcast<ContainerBox>(child), depth + 1);
551 } else
552 outputLayoutBox(stream, child, nullptr, depth);
553
554 if (is<ContainerBox>(child))
555 outputLayoutTree(layoutState, stream, downcast<ContainerBox>(child), depth + 1);
556 }
557}

559String layoutTreeAsText(const Box& layoutBox, const LayoutState* layoutState)
560{
561 TextStream stream(TextStream::LineMode::MultipleLine, TextStream::Formatting::SVGStyleRect);
562
563 auto& initialContainingBlock = layoutBox.initialContainingBlock();
564 outputLayoutBox(stream, initialContainingBlock, layoutState ? &layoutState->geometryForBox(initialContainingBlock) : nullptr, 0);
565 outputLayoutTree(layoutState, stream, initialContainingBlock, 1);
566
567 return stream.release();
568}

581void printLayoutTreeForLiveDocuments()
582{
583 for (const auto* document : Document::allDocuments()) {
584 if (!document->renderView())
585 continue;
586 if (document->frame() && document->frame()->isMainFrame())
587 fprintf(stderr, "----------------------main frame--------------------------\n");
588 fprintf(stderr, "%s\n", document->url().string().utf8().data());
589 // FIXME: Need to find a way to output geometry without layout context.
590 auto& renderView = *document->renderView();
591 auto layoutTree = TreeBuilder::buildLayoutTree(renderView);
592 auto layoutState = LayoutState { *document, layoutTree->root() };
