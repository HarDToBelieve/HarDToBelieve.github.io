26#pragma once
27
28#include "FrameIdentifier.h"
29#include "PageIdentifier.h"
30#include "SameSiteInfo.h"
31#include <wtf/Forward.h>
32#include <wtf/text/WTFString.h>
33
34namespace WebCore {
35
36enum class IncludeSecureCookies : bool { No, Yes };
37enum class IncludeHttpOnlyCookies : bool { No, Yes };
38enum class SecureCookiesAccessed : bool { No, Yes };
39
40class Document;
41struct Cookie;
42struct CookieRequestHeaderFieldProxy;
43class NetworkStorageSession;
44class StorageSessionProvider;
45struct SameSiteInfo;
46
47class WEBCORE_EXPORT CookieJar : public RefCounted<CookieJar> {
48public:
49 static Ref<CookieJar> create(Ref<StorageSessionProvider>&&);
50
51 static CookieRequestHeaderFieldProxy cookieRequestHeaderFieldProxy(const Document&, const URL&);
52
53 String cookieRequestHeaderFieldValue(Document&, const URL&) const;
54
55 // These two functions implement document.cookie API, with special rules for HttpOnly cookies.
56 virtual String cookies(Document&, const URL&) const;
57 virtual void setCookies(Document&, const URL&, const String& cookieString);
58
59 virtual bool cookiesEnabled(const Document&) const;
60 virtual std::pair<String, SecureCookiesAccessed> cookieRequestHeaderFieldValue(const URL& firstParty, const SameSiteInfo&, const URL&, std::optional<FrameIdentifier>, std::optional<PageIdentifier>, IncludeSecureCookies) const;
61 virtual bool getRawCookies(const Document&, const URL&, Vector<Cookie>&) const;
62 virtual void setRawCookie(const Document&, const Cookie&);
63 virtual void deleteCookie(const Document&, const URL&, const String& cookieName);
64
65 // Cookie Cache.
66 virtual void clearCache() { }
67 virtual void clearCacheForHost(const String&) { }
68
69 virtual ~CookieJar();
70protected:
71 static SameSiteInfo sameSiteInfo(const Document&, IsForDOMCookieAccess = IsForDOMCookieAccess::No);
72 static IncludeSecureCookies shouldIncludeSecureCookies(const Document&, const URL&);
73 CookieJar(Ref<StorageSessionProvider>&&);
74
75private:
76 Ref<StorageSessionProvider> m_storageSessionProvider;
77};
78
79} // namespace WebCore
