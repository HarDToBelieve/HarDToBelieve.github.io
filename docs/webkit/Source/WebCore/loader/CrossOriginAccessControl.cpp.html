48bool isOnAccessControlSimpleRequestMethodAllowlist(const String& method)
49{
50 return method == "GET" || method == "HEAD" || method == "POST";
51}
52
53bool isSimpleCrossOriginAccessRequest(const String& method, const HTTPHeaderMap& headerMap)
54{
55 if (!isOnAccessControlSimpleRequestMethodAllowlist(method))
56 return false;
57
58 for (const auto& header : headerMap) {
59 if (!header.keyAsHTTPHeaderName || !isCrossOriginSafeRequestHeader(header.keyAsHTTPHeaderName.value(), header.value))
60 return false;
61 }
62
63 return true;
64}

66void updateRequestReferrer(ResourceRequest& request, ReferrerPolicy referrerPolicy, const String& outgoingReferrer)
67{
68 String newOutgoingReferrer = SecurityPolicy::generateReferrerHeader(referrerPolicy, request.url(), outgoingReferrer);
69 if (newOutgoingReferrer.isEmpty())
70 request.clearHTTPReferrer();
71 else
72 request.setHTTPReferrer(newOutgoingReferrer);
73}

75void updateRequestForAccessControl(ResourceRequest& request, SecurityOrigin& securityOrigin, StoredCredentialsPolicy storedCredentialsPolicy)
76{
77 request.removeCredentials();
78 request.setAllowCookies(storedCredentialsPolicy == StoredCredentialsPolicy::Use);
79 request.setHTTPOrigin(securityOrigin.toString());
80}

82ResourceRequest createAccessControlPreflightRequest(const ResourceRequest& request, SecurityOrigin& securityOrigin, const String& referrer)
83{
84 ResourceRequest preflightRequest(request.url());
85 static const double platformDefaultTimeout = 0;
86 preflightRequest.setTimeoutInterval(platformDefaultTimeout);
87 updateRequestForAccessControl(preflightRequest, securityOrigin, StoredCredentialsPolicy::DoNotUse);
88 preflightRequest.setHTTPMethod("OPTIONS");
89 preflightRequest.setHTTPHeaderField(HTTPHeaderName::AccessControlRequestMethod, request.httpMethod());
90 preflightRequest.setPriority(request.priority());
91 preflightRequest.setFirstPartyForCookies(request.firstPartyForCookies());
92 preflightRequest.setIsAppBound(request.isAppBound());
93 if (!referrer.isNull())
94 preflightRequest.setHTTPReferrer(referrer);
95
96 const HTTPHeaderMap& requestHeaderFields = request.httpHeaderFields();
97
98 if (!requestHeaderFields.isEmpty()) {
99 Vector<String> unsafeHeaders;
100 for (auto& headerField : requestHeaderFields) {
101 if (!headerField.keyAsHTTPHeaderName || !isCrossOriginSafeRequestHeader(*headerField.keyAsHTTPHeaderName, headerField.value))
102 unsafeHeaders.append(headerField.key.convertToASCIILowercase());
103 }
104
105 std::sort(unsafeHeaders.begin(), unsafeHeaders.end(), WTF::codePointCompareLessThan);
106
107 StringBuilder headerBuffer;
108
109 bool appendComma = false;
110 for (const auto& headerField : unsafeHeaders) {
111 if (appendComma)
112 headerBuffer.append(',');
113 else
114 appendComma = true;
115
116 headerBuffer.append(headerField);
117 }
118 if (!headerBuffer.isEmpty())
119 preflightRequest.setHTTPHeaderField(HTTPHeaderName::AccessControlRequestHeaders, headerBuffer.toString());
120 }
121
122 return preflightRequest;
123}

125// https://html.spec.whatwg.org/multipage/urls-and-fetching.html#create-a-potential-cors-request
126CachedResourceRequest createPotentialAccessControlRequest(ResourceRequest&& request, ResourceLoaderOptions&& options, Document& document, const String& crossOriginAttribute, SameOriginFlag sameOriginFlag)
127{
128 ASSERT(options.mode == FetchOptions::Mode::NoCors);
129 if (!crossOriginAttribute.isNull())
130 options.mode = FetchOptions::Mode::Cors;
131 else if (sameOriginFlag == SameOriginFlag::Yes)
132 options.mode = FetchOptions::Mode::SameOrigin;
133
134 if (options.mode != FetchOptions::Mode::NoCors) {
135 if (auto* page = document.page()) {
136 if (page->shouldDisableCorsForRequestTo(request.url()))
137 options.mode = FetchOptions::Mode::NoCors;
138 }
139 }
140
141 if (crossOriginAttribute.isNull()) {
142 CachedResourceRequest cachedRequest { WTFMove(request), WTFMove(options) };
143 cachedRequest.setOrigin(document.securityOrigin());
144 return cachedRequest;
145 }
146
147 FetchOptions::Credentials credentials = equalLettersIgnoringASCIICase(crossOriginAttribute, "omit")
148 ? FetchOptions::Credentials::Omit : equalLettersIgnoringASCIICase(crossOriginAttribute, "use-credentials")
149 ? FetchOptions::Credentials::Include : FetchOptions::Credentials::SameOrigin;
150 options.credentials = credentials;
151 switch (credentials) {
152 case FetchOptions::Credentials::Include:
153 options.storedCredentialsPolicy = StoredCredentialsPolicy::Use;
154 break;
155 case FetchOptions::Credentials::SameOrigin:
156 options.storedCredentialsPolicy = document.securityOrigin().canRequest(request.url()) ? StoredCredentialsPolicy::Use : StoredCredentialsPolicy::DoNotUse;
157 break;
158 case FetchOptions::Credentials::Omit:
159 options.storedCredentialsPolicy = StoredCredentialsPolicy::DoNotUse;
160 }
161
162 CachedResourceRequest cachedRequest { WTFMove(request), WTFMove(options) };
163 updateRequestForAccessControl(cachedRequest.resourceRequest(), document.securityOrigin(), options.storedCredentialsPolicy);
164 return cachedRequest;
165}

167String validateCrossOriginRedirectionURL(const URL& redirectURL)
168{
169 if (!LegacySchemeRegistry::shouldTreatURLSchemeAsCORSEnabled(redirectURL.protocol().toStringWithoutCopying()))
170 return makeString("not allowed to follow a cross-origin CORS redirection with non CORS scheme");
171
172 if (redirectURL.hasCredentials())
173 return makeString("redirection URL ", redirectURL.string(), " has credentials");
174
175 return { };

178OptionSet<HTTPHeadersToKeepFromCleaning> httpHeadersToKeepFromCleaning(const HTTPHeaderMap& headers)
179{
180 OptionSet<HTTPHeadersToKeepFromCleaning> headersToKeep;
181 if (headers.contains(HTTPHeaderName::ContentType))
182 headersToKeep.add(HTTPHeadersToKeepFromCleaning::ContentType);
183 if (headers.contains(HTTPHeaderName::Referer))
184 headersToKeep.add(HTTPHeadersToKeepFromCleaning::Referer);
185 if (headers.contains(HTTPHeaderName::Origin))
186 headersToKeep.add(HTTPHeadersToKeepFromCleaning::Origin);
187 if (headers.contains(HTTPHeaderName::UserAgent))
188 headersToKeep.add(HTTPHeadersToKeepFromCleaning::UserAgent);
189 if (headers.contains(HTTPHeaderName::AcceptEncoding))
190 headersToKeep.add(HTTPHeadersToKeepFromCleaning::AcceptEncoding);
191 return headersToKeep;
192}

194void cleanHTTPRequestHeadersForAccessControl(ResourceRequest& request, OptionSet<HTTPHeadersToKeepFromCleaning> headersToKeep)
195{
196 // Remove headers that may have been added by the network layer that cause access control to fail.
197 if (!headersToKeep.contains(HTTPHeadersToKeepFromCleaning::ContentType)) {
198 auto contentType = request.httpContentType();
199 if (!contentType.isNull() && !isCrossOriginSafeRequestHeader(HTTPHeaderName::ContentType, contentType))
200 request.clearHTTPContentType();
201 }
202 if (!headersToKeep.contains(HTTPHeadersToKeepFromCleaning::Referer))
203 request.clearHTTPReferrer();
204 if (!headersToKeep.contains(HTTPHeadersToKeepFromCleaning::Origin))
205 request.clearHTTPOrigin();
206 if (!headersToKeep.contains(HTTPHeadersToKeepFromCleaning::UserAgent))
207 request.clearHTTPUserAgent();
208 if (!headersToKeep.contains(HTTPHeadersToKeepFromCleaning::AcceptEncoding))
209 request.clearHTTPAcceptEncoding();
210}

231Expected<void, String> passesAccessControlCheck(const ResourceResponse& response, StoredCredentialsPolicy storedCredentialsPolicy, const SecurityOrigin& securityOrigin, const CrossOriginAccessControlCheckDisabler* checkDisabler)
232{
233 // A wildcard Access-Control-Allow-Origin can not be used if credentials are to be sent,
234 // even with Access-Control-Allow-Credentials set to true.
235 const String& accessControlOriginString = response.httpHeaderField(HTTPHeaderName::AccessControlAllowOrigin);
236 bool starAllowed = storedCredentialsPolicy == StoredCredentialsPolicy::DoNotUse;
237 if (!starAllowed)
238 starAllowed = checkDisabler && !checkDisabler->crossOriginAccessControlCheckEnabled();
239 if (accessControlOriginString == "*" && starAllowed)
240 return { };
241
242 String securityOriginString = securityOrigin.toString();
243 if (accessControlOriginString != securityOriginString) {
244 if (accessControlOriginString == "*")
245 return makeUnexpected("Cannot use wildcard in Access-Control-Allow-Origin when credentials flag is true."_s);
246 if (accessControlOriginString.find(',') != notFound)
247 return makeUnexpected("Access-Control-Allow-Origin cannot contain more than one origin."_s);
248 return makeUnexpected(makeString("Origin ", securityOriginString, " is not allowed by Access-Control-Allow-Origin."));
249 }
250
251 if (storedCredentialsPolicy == StoredCredentialsPolicy::Use) {
252 const String& accessControlCredentialsString = response.httpHeaderField(HTTPHeaderName::AccessControlAllowCredentials);
253 if (accessControlCredentialsString != "true")
254 return makeUnexpected("Credentials flag is true, but Access-Control-Allow-Credentials is not \"true\"."_s);
255 }
256
257 return { };
258}

260Expected<void, String> validatePreflightResponse(PAL::SessionID sessionID, const ResourceRequest& request, const ResourceResponse& response, StoredCredentialsPolicy storedCredentialsPolicy, const SecurityOrigin& securityOrigin, const CrossOriginAccessControlCheckDisabler* checkDisabler)
261{
262 if (!response.isSuccessful())
263 return makeUnexpected("Preflight response is not successful"_s);
264
265 auto accessControlCheckResult = passesAccessControlCheck(response, storedCredentialsPolicy, securityOrigin, checkDisabler);
266 if (!accessControlCheckResult)
267 return accessControlCheckResult;
268
269 auto result = CrossOriginPreflightResultCacheItem::create(storedCredentialsPolicy, response);
270 if (!result.has_value())
271 return makeUnexpected(WTFMove(result.error()));
272
273 auto entry = WTFMove(result.value());
274 auto errorDescription = entry->validateMethodAndHeaders(request.httpMethod(), request.httpHeaderFields());
275 CrossOriginPreflightResultCache::singleton().appendEntry(sessionID, securityOrigin.toString(), request.url(), entry.moveToUniquePtr());
276
277 if (errorDescription)
278 return makeUnexpected(WTFMove(*errorDescription));
279
280 return { };
281}

283static inline bool shouldCrossOriginResourcePolicyCancelLoad(const SecurityOrigin& origin, const ResourceResponse& response)
284{
285 if (origin.canRequest(response.url()))
286 return false;
287
288 auto policy = parseCrossOriginResourcePolicyHeader(response.httpHeaderField(HTTPHeaderName::CrossOriginResourcePolicy));
289
290 if (policy == CrossOriginResourcePolicy::SameOrigin)
291 return true;
292
293 if (policy == CrossOriginResourcePolicy::SameSite) {
294 if (origin.isUnique())
295 return true;
296#if ENABLE(PUBLIC_SUFFIX_LIST)
297 if (!RegistrableDomain::uncheckedCreateFromHost(origin.host()).matches(response.url()))
298 return true;
299#endif
300 if (origin.protocol() == "http" && response.url().protocol() == "https")
301 return true;
302 }
303
304 return false;
305}

307std::optional<ResourceError> validateCrossOriginResourcePolicy(const SecurityOrigin& origin, const URL& requestURL, const ResourceResponse& response)
308{
309 if (shouldCrossOriginResourcePolicyCancelLoad(origin, response))
310 return ResourceError { errorDomainWebKitInternal, 0, requestURL, makeString("Cancelled load to ", response.url().stringCenterEllipsizedToLength(), " because it violates the resource's Cross-Origin-Resource-Policy response header."), ResourceError::Type::AccessControl };
311 return std::nullopt;
312}

315{
316 if (response.isRangeRequested() && response.httpStatusCode() == 206 && response.type() == ResourceResponse::Type::Opaque && !request.hasHTTPHeaderField(HTTPHeaderName::Range))
317 return ResourceError({ }, 0, response.url(), { }, ResourceError::Type::General);
318 return std::nullopt;
