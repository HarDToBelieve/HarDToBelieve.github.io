40// These values are at the discretion of the user agent.
41static const auto defaultPreflightCacheTimeout = 5_s;
42static const auto maxPreflightCacheTimeout = 600_s; // Should be short enough to minimize the risk of using a poisoned cache after switching to a secure network.

44CrossOriginPreflightResultCache::CrossOriginPreflightResultCache()
45{
46}

48static bool parseAccessControlMaxAge(const String& string, Seconds& expiryDelta)
49{
50 // FIXME: This should probably reject strings that have a leading "+".
51 auto parsedInteger = parseInteger<uint64_t>(string);
52 expiryDelta = Seconds(static_cast<double>(parsedInteger.value_or(0)));
53 return parsedInteger.has_value();
54}

56Expected<UniqueRef<CrossOriginPreflightResultCacheItem>, String> CrossOriginPreflightResultCacheItem::create(StoredCredentialsPolicy policy, const ResourceResponse& response)
57{
58 auto methods = parseAccessControlAllowList(response.httpHeaderField(HTTPHeaderName::AccessControlAllowMethods));
59 if (!methods)
60 return makeUnexpected(makeString("Header Access-Control-Allow-Methods has an invalid value: ", response.httpHeaderField(HTTPHeaderName::AccessControlAllowMethods)));
61
62 auto headers = parseAccessControlAllowList<ASCIICaseInsensitiveHash>(response.httpHeaderField(HTTPHeaderName::AccessControlAllowHeaders));
63 if (!headers)
64 return makeUnexpected(makeString("Header Access-Control-Allow-Headers has an invalid value: ", response.httpHeaderField(HTTPHeaderName::AccessControlAllowHeaders)));
65
66 Seconds expiryDelta = 0_s;
67 if (parseAccessControlMaxAge(response.httpHeaderField(HTTPHeaderName::AccessControlMaxAge), expiryDelta)) {
68 if (expiryDelta > maxPreflightCacheTimeout)
69 expiryDelta = maxPreflightCacheTimeout;
70 } else
71 expiryDelta = defaultPreflightCacheTimeout;
72
73 return makeUniqueRef<CrossOriginPreflightResultCacheItem>(MonotonicTime::now() + expiryDelta, policy, WTFMove(*methods), WTFMove(*headers));
74}

76std::optional<String> CrossOriginPreflightResultCacheItem::validateMethodAndHeaders(const String& method, const HTTPHeaderMap& requestHeaders) const
77{
78 if (!allowsCrossOriginMethod(method, m_storedCredentialsPolicy))
79 return makeString("Method ", method, " is not allowed by Access-Control-Allow-Methods.");
80
81 if (auto badHeader = validateCrossOriginHeaders(requestHeaders, m_storedCredentialsPolicy))
82 return makeString("Request header field ", *badHeader, " is not allowed by Access-Control-Allow-Headers.");
83 return { };
84}
85
86bool CrossOriginPreflightResultCacheItem::allowsCrossOriginMethod(const String& method, StoredCredentialsPolicy storedCredentialsPolicy) const
87{
88 return m_methods.contains(method) || (m_methods.contains("*") && storedCredentialsPolicy != StoredCredentialsPolicy::Use) || isOnAccessControlSimpleRequestMethodAllowlist(method);
89}
90
91std::optional<String> CrossOriginPreflightResultCacheItem::validateCrossOriginHeaders(const HTTPHeaderMap& requestHeaders, StoredCredentialsPolicy storedCredentialsPolicy) const
92{
93 bool validWildcard = m_headers.contains("*") && storedCredentialsPolicy != StoredCredentialsPolicy::Use;
94 for (const auto& header : requestHeaders) {
95 if (header.keyAsHTTPHeaderName && isCrossOriginSafeRequestHeader(header.keyAsHTTPHeaderName.value(), header.value))
96 continue;
97 if (!m_headers.contains(header.key) && !validWildcard)
98 return header.key;
99 }
100 return { };
101}

103bool CrossOriginPreflightResultCacheItem::allowsRequest(StoredCredentialsPolicy storedCredentialsPolicy, const String& method, const HTTPHeaderMap& requestHeaders) const
104{
105 if (m_absoluteExpiryTime < MonotonicTime::now())
106 return false;
107 if (storedCredentialsPolicy == StoredCredentialsPolicy::Use && m_storedCredentialsPolicy == StoredCredentialsPolicy::DoNotUse)
108 return false;
109 if (!allowsCrossOriginMethod(method, storedCredentialsPolicy))
110 return false;
111 if (auto badHeader = validateCrossOriginHeaders(requestHeaders, storedCredentialsPolicy))
112 return false;
113 return true;
114}

124void CrossOriginPreflightResultCache::appendEntry(PAL::SessionID sessionID, const String& origin, const URL& url, std::unique_ptr<CrossOriginPreflightResultCacheItem> preflightResult)
125{
126 ASSERT(isMainThread());
127 m_preflightHashMap.set(std::make_tuple(sessionID, origin, url), WTFMove(preflightResult));
128}
129
130bool CrossOriginPreflightResultCache::canSkipPreflight(PAL::SessionID sessionID, const String& origin, const URL& url, StoredCredentialsPolicy storedCredentialsPolicy, const String& method, const HTTPHeaderMap& requestHeaders)
131{
132 ASSERT(isMainThread());
133 auto it = m_preflightHashMap.find(std::make_tuple(sessionID, origin, url));
134 if (it == m_preflightHashMap.end())
135 return false;
136
137 if (it->value->allowsRequest(storedCredentialsPolicy, method, requestHeaders))
138 return true;
139
140 m_preflightHashMap.remove(it);
141 return false;
142}

144void CrossOriginPreflightResultCache::clear()
145{
146 ASSERT(isMainThread());
147 m_preflightHashMap.clear();
