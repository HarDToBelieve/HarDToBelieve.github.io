26#pragma once
27
28#include "HTTPHeaderField.h"
29#include <wtf/Vector.h>
30#include <wtf/text/WTFString.h>
31
32namespace WebCore {
33
34struct WEBCORE_EXPORT CustomHeaderFields {
35 Vector<HTTPHeaderField> fields;
36 Vector<String> thirdPartyDomains;
37
38 bool thirdPartyDomainsMatch(const URL&) const;
39
40 template<class Encoder> void encode(Encoder&) const;
41 template<class Decoder> static std::optional<CustomHeaderFields> decode(Decoder&);
42};
43
44template<class Encoder>
45void CustomHeaderFields::encode(Encoder& encoder) const
46{
47 encoder << fields;
48 encoder << thirdPartyDomains;
49}
50
51template<class Decoder>
52std::optional<CustomHeaderFields> CustomHeaderFields::decode(Decoder& decoder)
53{
54 std::optional<Vector<HTTPHeaderField>> fields;
55 decoder >> fields;
56 if (!fields)
57 return std::nullopt;
58
59 std::optional<Vector<String>> thirdPartyDomains;
60 decoder >> thirdPartyDomains;
61 if (!thirdPartyDomains)
62 return std::nullopt;
63
64 return {{ WTFMove(*fields), WTFMove(*thirdPartyDomains) }};
65}
66
67} // namespace WebCore
