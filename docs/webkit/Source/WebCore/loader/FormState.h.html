29#pragma once
30
31#include "FrameDestructionObserver.h"
32#include <wtf/WeakPtr.h>
33#include <wtf/text/WTFString.h>
34
35namespace WebCore {
36
37class Document;
38class HTMLFormElement;
39
40enum FormSubmissionTrigger { SubmittedByJavaScript, NotSubmittedByJavaScript };
41
42using StringPairVector = Vector<std::pair<String, String>>;
43
44class FormState : public RefCounted<FormState>, public CanMakeWeakPtr<FormState>, public FrameDestructionObserver {
45public:
46 static Ref<FormState> create(HTMLFormElement&, StringPairVector&& textFieldValues, Document&, FormSubmissionTrigger);
47
48 HTMLFormElement& form() const { return m_form; }
49 const StringPairVector& textFieldValues() const { return m_textFieldValues; }
50 Document& sourceDocument() const { return m_sourceDocument; }
51 FormSubmissionTrigger formSubmissionTrigger() const { return m_formSubmissionTrigger; }
52
53private:
54 FormState(HTMLFormElement&, StringPairVector&& textFieldValues, Document&, FormSubmissionTrigger);
55 void willDetachPage() override;
56
57 Ref<HTMLFormElement> m_form;
58 StringPairVector m_textFieldValues;
59 Ref<Document> m_sourceDocument;
60 FormSubmissionTrigger m_formSubmissionTrigger;
61};
62
63} // namespace WebCore
