40FrameLoadRequest::FrameLoadRequest(Document& requester, SecurityOrigin& requesterSecurityOrigin, ResourceRequest&& resourceRequest, const String& frameName, InitiatedByMainFrame initiatedByMainFrame, const AtomString& downloadAttribute, const SystemPreviewInfo& systemPreviewInfo)
41 : m_requester { makeRef(requester) }
42 , m_requesterSecurityOrigin { makeRef(requesterSecurityOrigin) }
43 , m_resourceRequest { WTFMove(resourceRequest) }
44 , m_frameName { frameName }
45 , m_downloadAttribute { downloadAttribute }
46 , m_initiatedByMainFrame { initiatedByMainFrame }
47 , m_systemPreviewInfo { systemPreviewInfo }
48{
49}
50
51FrameLoadRequest::FrameLoadRequest(Frame& frame, const ResourceRequest& resourceRequest, const SubstituteData& substituteData)
52 : m_requester { makeRef(*frame.document()) }
53 , m_requesterSecurityOrigin { makeRef(frame.document()->securityOrigin()) }
54 , m_resourceRequest { resourceRequest }
55 , m_substituteData { substituteData }
56{
57}

59FrameLoadRequest::~FrameLoadRequest() = default;

62FrameLoadRequest& FrameLoadRequest::operator=(FrameLoadRequest&&) = default;

64Document& FrameLoadRequest::requester()
65{
66 return m_requester.get();
67}

69const SecurityOrigin& FrameLoadRequest::requesterSecurityOrigin() const

71 return m_requesterSecurityOrigin.get();
72}
