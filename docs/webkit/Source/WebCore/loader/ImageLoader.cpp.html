110ImageLoader::~ImageLoader()
111{
112 if (m_image)
113 m_image->removeClient(*this);
114
115 ASSERT(m_hasPendingBeforeLoadEvent || !beforeLoadEventSender().hasPendingEvents(*this));
116 if (m_hasPendingBeforeLoadEvent)
117 beforeLoadEventSender().cancelEvent(*this);
118
119 ASSERT(m_hasPendingLoadEvent || !loadEventSender().hasPendingEvents(*this));
120 if (m_hasPendingLoadEvent)
121 loadEventSender().cancelEvent(*this);
122
123 ASSERT(m_hasPendingErrorEvent || !errorEventSender().hasPendingEvents(*this));
124 if (m_hasPendingErrorEvent)
125 errorEventSender().cancelEvent(*this);
126}

137void ImageLoader::clearImageWithoutConsideringPendingLoadEvent()
138{
139 ASSERT(m_failedLoadURL.isEmpty());
140 CachedImage* oldImage = m_image.get();
141 if (oldImage) {
142 m_image = nullptr;
143 if (m_hasPendingBeforeLoadEvent) {
144 beforeLoadEventSender().cancelEvent(*this);
145 m_hasPendingBeforeLoadEvent = false;
146 }
147 if (m_hasPendingLoadEvent) {
148 loadEventSender().cancelEvent(*this);
149 m_hasPendingLoadEvent = false;
150 }
151 if (m_hasPendingErrorEvent) {
152 errorEventSender().cancelEvent(*this);
153 m_hasPendingErrorEvent = false;
154 }
155 m_imageComplete = true;
156 if (oldImage)
157 oldImage->removeClient(*this);
158 }
159
160 if (RenderImageResource* imageResource = renderImageResource())
161 imageResource->resetAnimation();
162}

164void ImageLoader::updateFromElement(RelevantMutation relevantMutation)
165{
166 // If we're not making renderers for the page, then don't load images. We don't want to slow
167 // down the raw HTML parsing case by loading images we don't intend to display.
168 Document& document = element().document();
169 if (!document.hasLivingRenderTree())
170 return;
171
172 AtomString attr = element().imageSourceURL();
173
174 // Avoid loading a URL we already failed to load.
175 if (!m_failedLoadURL.isEmpty() && attr == m_failedLoadURL)
176 return;
177
178 // Do not load any image if the 'src' attribute is missing or if it is
179 // an empty string.
180 CachedResourceHandle<CachedImage> newImage = nullptr;
181 if (!attr.isNull() && !stripLeadingAndTrailingHTMLSpaces(attr).isEmpty()) {
182 ResourceLoaderOptions options = CachedResourceLoader::defaultCachedResourceOptions();
183 options.contentSecurityPolicyImposition = element().isInUserAgentShadowTree() ? ContentSecurityPolicyImposition::SkipPolicyCheck : ContentSecurityPolicyImposition::DoPolicyCheck;
184 options.sameOriginDataURLFlag = SameOriginDataURLFlag::Set;
185 bool isImageElement = is<HTMLImageElement>(element());
186 if (isImageElement)
187 options.referrerPolicy = downcast<HTMLImageElement>(element()).referrerPolicy();
188
189 auto crossOriginAttribute = element().attributeWithoutSynchronization(HTMLNames::crossoriginAttr);
190
191 // Use url from original request for lazy loads that are no longer in deferred state.
192 URL imageURL = m_lazyImageLoadState == LazyImageLoadState::LoadImmediately
193 ? m_image->url() : document.completeURL(sourceURI(attr));
194 ResourceRequest resourceRequest(imageURL);
195 resourceRequest.setInspectorInitiatorNodeIdentifier(InspectorInstrumentation::identifierForNode(m_element));
196
197 auto request = createPotentialAccessControlRequest(WTFMove(resourceRequest), WTFMove(options), document, crossOriginAttribute);
198 request.setInitiator(element());
199
200 if (m_loadManually) {
201 bool autoLoadOtherImages = document.cachedResourceLoader().autoLoadImages();
202 document.cachedResourceLoader().setAutoLoadImages(false);
203 auto* page = m_element.document().page();
204 newImage = new CachedImage(WTFMove(request), page->sessionID(), &page->cookieJar());
205 newImage->setStatus(CachedResource::Pending);
206 newImage->setLoading(true);
207 document.cachedResourceLoader().m_documentResources.set(newImage->url().string(), newImage.get());
208 document.cachedResourceLoader().setAutoLoadImages(autoLoadOtherImages);
209 } else {
210 if (m_lazyImageLoadState == LazyImageLoadState::None && isImageElement) {
211 auto& imageElement = downcast<HTMLImageElement>(element());
212 if (imageElement.isLazyLoadable() && document.settings().lazyImageLoadingEnabled()) {
213 m_lazyImageLoadState = LazyImageLoadState::Deferred;
214 request.setIgnoreForRequestCount(true);
215 }
216 }
217 auto imageLoading = (m_lazyImageLoadState == LazyImageLoadState::Deferred) ? ImageLoading::DeferredUntilVisible : ImageLoading::Immediate;
218 newImage = document.cachedResourceLoader().requestImage(WTFMove(request), imageLoading).value_or(nullptr);
219 }
220
221 // If we do not have an image here, it means that a cross-site
222 // violation occurred, or that the image was blocked via Content
223 // Security Policy, or the page is being dismissed. Trigger an
224 // error event if the page is not being dismissed.
225 if (!newImage && !pageIsBeingDismissed(document)) {
226 m_failedLoadURL = attr;
227 m_hasPendingErrorEvent = true;
228 errorEventSender().dispatchEventSoon(*this);
229 } else
230 clearFailedLoadURL();
231 } else if (!attr.isNull()) {
232 // Fire an error event if the url is empty.
233 m_failedLoadURL = attr;
234 m_hasPendingErrorEvent = true;
235 errorEventSender().dispatchEventSoon(*this);
236 }
237
238 CachedImage* oldImage = m_image.get();
239 if (newImage != oldImage || relevantMutation == RelevantMutation::Yes) {
240 if (m_hasPendingBeforeLoadEvent) {
241 beforeLoadEventSender().cancelEvent(*this);
242 m_hasPendingBeforeLoadEvent = false;
243 }
244 if (m_hasPendingLoadEvent) {
245 loadEventSender().cancelEvent(*this);
246 m_hasPendingLoadEvent = false;
247 }
248
249 // Cancel error events that belong to the previous load, which is now cancelled by changing the src attribute.
250 // If newImage is null and m_hasPendingErrorEvent is true, we know the error event has been just posted by
251 // this load and we should not cancel the event.
252 // FIXME: If both previous load and this one got blocked with an error, we can receive one error event instead of two.
253 if (m_hasPendingErrorEvent && newImage) {
254 errorEventSender().cancelEvent(*this);
255 m_hasPendingErrorEvent = false;
256 }
257
258 m_image = newImage;
259 m_hasPendingBeforeLoadEvent = !document.isImageDocument() && newImage;
260 m_hasPendingLoadEvent = newImage;
261 m_imageComplete = !newImage;
262
263 if (newImage) {
264 if (!document.isImageDocument()) {
265 if (!document.hasListenerType(Document::BEFORELOAD_LISTENER))
266 dispatchPendingBeforeLoadEvent();
267 else
268 beforeLoadEventSender().dispatchEventSoon(*this);
269 } else
270 updateRenderer();
271
272 if (m_lazyImageLoadState == LazyImageLoadState::Deferred)
273 LazyLoadImageObserver::observe(element());
274
275 // If newImage is cached, addClient() will result in the load event
276 // being queued to fire. Ensure this happens after beforeload is
277 // dispatched.
278 newImage->addClient(*this);
279 } else
280 resetLazyImageLoading(element().document());
281 if (oldImage) {
282 oldImage->removeClient(*this);
283 updateRenderer();
284 }
285 }
286
287 if (RenderImageResource* imageResource = renderImageResource())
288 imageResource->resetAnimation();
289
290 // Only consider updating the protection ref-count of the Element immediately before returning
291 // from this function as doing so might result in the destruction of this ImageLoader.
292 updatedHasPendingEvent();
293}

301static inline void resolvePromises(Vector<RefPtr<DeferredPromise>>& promises)
302{
303 ASSERT(!promises.isEmpty());
304 auto promisesToBeResolved = std::exchange(promises, { });
305 for (auto& promise : promisesToBeResolved)
306 promise->resolve();
307}
308
309static inline void rejectPromises(Vector<RefPtr<DeferredPromise>>& promises, const char* message)
310{
311 ASSERT(!promises.isEmpty());
312 auto promisesToBeRejected = std::exchange(promises, { });
313 for (auto& promise : promisesToBeRejected)
314 promise->reject(Exception { EncodingError, message });
315}

327void ImageLoader::notifyFinished(CachedResource& resource, const NetworkLoadMetrics&)
328{
329 ASSERT(m_failedLoadURL.isEmpty());
330 ASSERT_UNUSED(resource, &resource == m_image.get());
331
332 if (isDeferred()) {
333 LazyLoadImageObserver::unobserve(element(), element().document());
334 m_lazyImageLoadState = LazyImageLoadState::FullImage;
335 }
336
337 m_imageComplete = true;
338 if (!hasPendingBeforeLoadEvent())
339 updateRenderer();
340
341 if (!m_hasPendingLoadEvent)
342 return;
343
344 if (m_image->resourceError().isAccessControl()) {
345 URL imageURL = m_image->url();
346
347 clearImageWithoutConsideringPendingLoadEvent();
348
349 m_hasPendingErrorEvent = true;
350 errorEventSender().dispatchEventSoon(*this);
351
352 auto message = makeString("Cannot load image ", imageURL.string(), " due to access control checks.");
353 element().document().addConsoleMessage(MessageSource::Security, MessageLevel::Error, message);
354
355 if (hasPendingDecodePromises())
356 rejectDecodePromises("Access control error.");
357
358 ASSERT(!m_hasPendingLoadEvent);
359
360 // Only consider updating the protection ref-count of the Element immediately before returning
361 // from this function as doing so might result in the destruction of this ImageLoader.
362 updatedHasPendingEvent();
363 return;
364 }
365
366 if (m_image->wasCanceled()) {
367 if (hasPendingDecodePromises())
368 rejectDecodePromises("Loading was canceled.");
369 m_hasPendingLoadEvent = false;
370 // Only consider updating the protection ref-count of the Element immediately before returning
371 // from this function as doing so might result in the destruction of this ImageLoader.
372 updatedHasPendingEvent();
373 return;
374 }
375
376 if (hasPendingDecodePromises())
377 decode();
378 loadEventSender().dispatchEventSoon(*this);
379}

381RenderImageResource* ImageLoader::renderImageResource()
382{
383 auto* renderer = element().renderer();
384 if (!renderer)
385 return nullptr;
386
387 // We don't return style generated image because it doesn't belong to the ImageLoader.
388 // See <https://bugs.webkit.org/show_bug.cgi?id=42840>
389 if (is<RenderImage>(*renderer) && !downcast<RenderImage>(*renderer).isGeneratedContent())
390 return &downcast<RenderImage>(*renderer).imageResource();
391
392 if (is<RenderSVGImage>(*renderer))
393 return &downcast<RenderSVGImage>(*renderer).imageResource();
394
395#if ENABLE(VIDEO)
396 if (is<RenderVideo>(*renderer))
397 return &downcast<RenderVideo>(*renderer).imageResource();
398#endif
399
400 return nullptr;
401}

403void ImageLoader::updateRenderer()
404{
405 RenderImageResource* imageResource = renderImageResource();
406
407 if (!imageResource)
408 return;
409
410 // Only update the renderer if it doesn't have an image or if what we have
411 // is a complete image. This prevents flickering in the case where a dynamic
412 // change is happening between two images.
413 CachedImage* cachedImage = imageResource->cachedImage();
414 if (m_image != cachedImage && (m_imageComplete || !cachedImage))
415 imageResource->setCachedImage(m_image.get());
416}

418void ImageLoader::updatedHasPendingEvent()
419{
420 // If an Element that does image loading is removed from the DOM the load/error event for the image is still observable.
421 // As long as the ImageLoader is actively loading, the Element itself needs to be ref'ed to keep it from being
422 // destroyed by DOM manipulation or garbage collection.
423 // If such an Element wishes for the load to stop when removed from the DOM it needs to stop the ImageLoader explicitly.
424 bool wasProtected = m_elementIsProtected;
425 m_elementIsProtected = m_hasPendingLoadEvent || m_hasPendingErrorEvent;
426 if (wasProtected == m_elementIsProtected)
427 return;
428
429 if (m_elementIsProtected) {
430 if (m_derefElementTimer.isActive())
431 m_derefElementTimer.stop();
432 else
433 m_protectedElement = &element();
434 } else {
435 ASSERT(!m_derefElementTimer.isActive());
436 m_derefElementTimer.startOneShot(0_s);
437 }
438}

440void ImageLoader::decode(Ref<DeferredPromise>&& promise)
441{
442 m_decodingPromises.append(WTFMove(promise));
443
444 if (!element().document().domWindow()) {
445 rejectDecodePromises("Inactive document.");
446 return;
447 }
448
449 AtomString attr = element().imageSourceURL();
450 if (stripLeadingAndTrailingHTMLSpaces(attr).isEmpty()) {
451 rejectDecodePromises("Missing source URL.");
452 return;
453 }
454
455 if (m_imageComplete)
456 decode();
457}

459void ImageLoader::decode()
460{
461 ASSERT(hasPendingDecodePromises());
462
463 if (!element().document().domWindow()) {
464 rejectDecodePromises("Inactive document.");
465 return;
466 }
467
468 if (!m_image || !m_image->image() || m_image->errorOccurred()) {
469 rejectDecodePromises("Loading error.");
470 return;
471 }
472
473 Image* image = m_image->image();
474 if (!is<BitmapImage>(image)) {
475 resolveDecodePromises();
476 return;
477 }
478
479 auto& bitmapImage = downcast<BitmapImage>(*image);
480 bitmapImage.decode([promises = WTFMove(m_decodingPromises)]() mutable {
481 resolvePromises(promises);
482 });
483}

490void ImageLoader::dispatchPendingEvent(ImageEventSender* eventSender)
491{
492 ASSERT(eventSender == &beforeLoadEventSender() || eventSender == &loadEventSender() || eventSender == &errorEventSender());
493 const AtomString& eventType = eventSender->eventType();
494 if (eventType == eventNames().beforeloadEvent)
495 dispatchPendingBeforeLoadEvent();
496 if (eventType == eventNames().loadEvent)
497 dispatchPendingLoadEvent();
498 if (eventType == eventNames().errorEvent)
499 dispatchPendingErrorEvent();
500}
501
502void ImageLoader::dispatchPendingBeforeLoadEvent()
503{
504 if (!m_hasPendingBeforeLoadEvent)
505 return;
506 if (!m_image)
507 return;
508 if (!element().document().hasLivingRenderTree())
509 return;
510 m_hasPendingBeforeLoadEvent = false;
511 Ref<Document> originalDocument = element().document();
512 if (element().dispatchBeforeLoadEvent(m_image->url().string())) {
513 bool didEventListenerDisconnectThisElement = !element().isConnected() || &element().document() != originalDocument.ptr();
514 if (didEventListenerDisconnectThisElement)
515 return;
516
517 updateRenderer();
518 return;
519 }
520 if (m_image) {
521 m_image->removeClient(*this);
522 m_image = nullptr;
523 }
524
525 loadEventSender().cancelEvent(*this);
526 m_hasPendingLoadEvent = false;
527
528 if (is<HTMLObjectElement>(element()))
529 downcast<HTMLObjectElement>(element()).renderFallbackContent();
530
531 // Only consider updating the protection ref-count of the Element immediately before returning
532 // from this function as doing so might result in the destruction of this ImageLoader.
533 updatedHasPendingEvent();
534}
535
536void ImageLoader::dispatchPendingLoadEvent()
537{
538 if (!m_hasPendingLoadEvent)
539 return;
540 if (!m_image)
541 return;
542 m_hasPendingLoadEvent = false;
543 if (element().document().hasLivingRenderTree())
544 dispatchLoadEvent();
545
546 // Only consider updating the protection ref-count of the Element immediately before returning
547 // from this function as doing so might result in the destruction of this ImageLoader.
548 updatedHasPendingEvent();
549}
550
551void ImageLoader::dispatchPendingErrorEvent()
552{
553 if (!m_hasPendingErrorEvent)
554 return;
555 m_hasPendingErrorEvent = false;
556 errorEventSender().cancelEvent(*this);
557 if (element().document().hasLivingRenderTree())
558 element().dispatchEvent(Event::create(eventNames().errorEvent, Event::CanBubble::No, Event::IsCancelable::No));
559
560 // Only consider updating the protection ref-count of the Element immediately before returning
561 // from this function as doing so might result in the destruction of this ImageLoader.
562 updatedHasPendingEvent();
563}

592void ImageLoader::loadDeferredImage()
593{
594 if (m_lazyImageLoadState != LazyImageLoadState::Deferred)
595 return;
596 m_lazyImageLoadState = LazyImageLoadState::LoadImmediately;
597 updateFromElement(RelevantMutation::No);
598}

600void ImageLoader::resetLazyImageLoading(Document& document)
601{
602 if (isDeferred())
603 LazyLoadImageObserver::unobserve(element(), document);
604 m_lazyImageLoadState = LazyImageLoadState::None;

607VisibleInViewportState ImageLoader::imageVisibleInViewport(const Document& document) const
608{
609 if (&element().document() != &document)
610 return VisibleInViewportState::No;
611
612 auto* renderer = element().renderer();
613 if (!is<RenderReplaced>(renderer))
614 return VisibleInViewportState::No;
615
616 return downcast<RenderReplaced>(*renderer).isContentLikelyVisibleInViewport() ? VisibleInViewportState::Yes : VisibleInViewportState::No;
617}
