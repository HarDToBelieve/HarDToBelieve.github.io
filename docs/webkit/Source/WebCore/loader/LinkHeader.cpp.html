35// LWSP definition in https://www.ietf.org/rfc/rfc0822.txt
36template<typename CharacterType> static bool isSpaceOrTab(CharacterType character)
37{
38 return character == ' ' || character == '\t';
39}
40
41template<typename CharacterType> static bool isNotURLTerminatingChar(CharacterType character)
42{
43 return character != '>';
44}
45
46template<typename CharacterType> static bool isValidParameterNameChar(CharacterType character)
47{
48 // A separator, CTL or '%', '*' or '\'' means the char is not valid.
49 // Definition as attr-char at https://tools.ietf.org/html/rfc5987
50 // CTL and separators are defined in https://tools.ietf.org/html/rfc2616#section-2.2
51 // Valid chars are alpha-numeric and any of !#$&+-.^_`|~"
52 if ((character >= '^' && character <= 'z') || (character >= 'A' && character <= 'Z') || (character >= '0' && character <= '9') || (character >= '!' && character <= '$') || character == '&' || character == '+' || character == '-' || character == '.')
53 return true;
54 return false;
55}
56
57template<typename CharacterType> static bool isParameterValueEnd(CharacterType character)
58{
59 return character == ';' || character == ',';
60}
61
62template<typename CharacterType> static bool isParameterValueChar(CharacterType character)
63{
64 return !isSpaceOrTab(character) && !isParameterValueEnd(character);
65}

67// Verify that the parameter is a link-extension which according to spec doesn't have to have a value.
68static bool isExtensionParameter(LinkHeader::LinkParameterName name)
69{
70 return name >= LinkHeader::LinkParameterUnknown;
71}

73// Before:
74//
75// <cat.jpg>; rel=preload
76// ^ ^
77// position end
78//
79// After (if successful: otherwise the method returns false)
80//
81// <cat.jpg>; rel=preload
82// ^ ^
83// position end
84template<typename CharacterType> static std::optional<String> findURLBoundaries(StringParsingBuffer<CharacterType>& buffer)
85{
86 skipWhile<isSpaceOrTab>(buffer);
87 if (!skipExactly(buffer, '<'))
88 return std::nullopt;
89 skipWhile<isSpaceOrTab>(buffer);
90
91 auto urlStart = buffer.position();
92 skipWhile<isNotURLTerminatingChar>(buffer);
93 auto urlEnd = buffer.position();
94 skipUntil(buffer, '>');
95 if (!skipExactly(buffer, '>'))
96 return std::nullopt;
97
98 return String(urlStart, urlEnd - urlStart);
99}

101template<typename CharacterType> static bool invalidParameterDelimiter(StringParsingBuffer<CharacterType>& buffer)
102{
103 return !skipExactly(buffer, ';') && buffer.hasCharactersRemaining() && *buffer != ',';
104}
105
106template<typename CharacterType> static bool validFieldEnd(StringParsingBuffer<CharacterType>& buffer)
107{
108 return buffer.atEnd() || *buffer == ',';
109}
110
111// Before:
112//
113// <cat.jpg>; rel=preload
114// ^ ^
115// position end
116//
117// After (if successful: otherwise the method returns false, and modifies the isValid boolean accordingly)
118//
119// <cat.jpg>; rel=preload
120// ^ ^
121// position end
122template<typename CharacterType> static bool parseParameterDelimiter(StringParsingBuffer<CharacterType>& buffer, bool& isValid)
123{
124 isValid = true;
125 skipWhile<isSpaceOrTab>(buffer);
126 if (invalidParameterDelimiter(buffer)) {
127 isValid = false;
128 return false;
129 }
130 skipWhile<isSpaceOrTab>(buffer);
131 if (validFieldEnd(buffer))
132 return false;
133 return true;
134}

136static LinkHeader::LinkParameterName paramterNameFromString(StringView name)
137{
138 if (equalLettersIgnoringASCIICase(name, "rel"))
139 return LinkHeader::LinkParameterRel;
140 if (equalLettersIgnoringASCIICase(name, "anchor"))
141 return LinkHeader::LinkParameterAnchor;
142 if (equalLettersIgnoringASCIICase(name, "crossorigin"))
143 return LinkHeader::LinkParameterCrossOrigin;
144 if (equalLettersIgnoringASCIICase(name, "title"))
145 return LinkHeader::LinkParameterTitle;
146 if (equalLettersIgnoringASCIICase(name, "media"))
147 return LinkHeader::LinkParameterMedia;
148 if (equalLettersIgnoringASCIICase(name, "type"))
149 return LinkHeader::LinkParameterType;
150 if (equalLettersIgnoringASCIICase(name, "rev"))
151 return LinkHeader::LinkParameterRev;
152 if (equalLettersIgnoringASCIICase(name, "hreflang"))
153 return LinkHeader::LinkParameterHreflang;
154 if (equalLettersIgnoringASCIICase(name, "as"))
155 return LinkHeader::LinkParameterAs;
156 if (equalLettersIgnoringASCIICase(name, "imagesrcset"))
157 return LinkHeader::LinkParameterImageSrcSet;
158 if (equalLettersIgnoringASCIICase(name, "imagesizes"))
159 return LinkHeader::LinkParameterImageSizes;
160 return LinkHeader::LinkParameterUnknown;
161}

163// Before:
164//
165// <cat.jpg>; rel=preload
166// ^ ^
167// position end
168//
169// After (if successful: otherwise the method returns false)
170//
171// <cat.jpg>; rel=preload
172// ^ ^
173// position end
174template<typename CharacterType> static std::optional<LinkHeader::LinkParameterName> parseParameterName(StringParsingBuffer<CharacterType>& buffer)
175{
176 auto nameStart = buffer.position();
177 skipWhile<isValidParameterNameChar>(buffer);
178 auto nameEnd = buffer.position();
179 skipWhile<isSpaceOrTab>(buffer);
180 bool hasEqual = skipExactly(buffer, '=');
181 skipWhile<isSpaceOrTab>(buffer);
182 auto name = paramterNameFromString(StringView { nameStart, static_cast<unsigned>(nameEnd - nameStart) });
183 if (hasEqual)
184 return name;
185 bool validParameterValueEnd = buffer.atEnd() || isParameterValueEnd(*buffer);
186 if (validParameterValueEnd && isExtensionParameter(name))
187 return name;
188 return std::nullopt;
189}

191// Before:
192//
193// <cat.jpg>; rel="preload"; type="image/jpeg";
194// ^ ^
195// position end
196//
197// After (if the parameter starts with a quote, otherwise the method returns false)
198//
199// <cat.jpg>; rel="preload"; type="image/jpeg";
200// ^ ^
201// position end
202template<typename CharacterType> static bool skipQuotesIfNeeded(StringParsingBuffer<CharacterType>& buffer, bool& completeQuotes)
203{
204 unsigned char quote;
205 if (skipExactly(buffer, '\''))
206 quote = '\'';
207 else if (skipExactly(buffer, '"'))
208 quote = '"';
209 else
210 return false;
211
212 while (!completeQuotes && buffer.hasCharactersRemaining()) {
213 skipUntil(buffer, static_cast<CharacterType>(quote));
214 if (*(buffer.position() - 1) != '\\')
215 completeQuotes = true;
216 completeQuotes = skipExactly(buffer, static_cast<CharacterType>(quote)) && completeQuotes;
217 }
218 return true;
219}

221// Before:
222//
223// <cat.jpg>; rel=preload; foo=bar
224// ^ ^
225// position end
226//
227// After (if successful: otherwise the method returns false)
228//
229// <cat.jpg>; rel=preload; foo=bar
230// ^ ^
231// position end
232template<typename CharacterType> static bool parseParameterValue(StringParsingBuffer<CharacterType>& buffer, String& value)
233{
234 auto valueStart = buffer.position();
235 auto valueEnd = buffer.position();
236 bool completeQuotes = false;
237 bool hasQuotes = skipQuotesIfNeeded(buffer, completeQuotes);
238 if (!hasQuotes)
239 skipWhile<isParameterValueChar>(buffer);
240 valueEnd = buffer.position();
241 skipWhile<isSpaceOrTab>(buffer);
242 if ((!completeQuotes && valueStart == valueEnd) || (!buffer.atEnd() && !isParameterValueEnd(*buffer))) {
243 value = emptyString();
244 return false;
245 }
246 if (hasQuotes)
247 ++valueStart;
248 if (completeQuotes)
249 --valueEnd;
250 ASSERT(valueEnd >= valueStart);
251 value = String(valueStart, valueEnd - valueStart);
252 return !hasQuotes || completeQuotes;
253}

255void LinkHeader::setValue(LinkParameterName name, String&& value)
256{
257 switch (name) {
258 case LinkParameterRel:
259 if (!m_rel)
260 m_rel = WTFMove(value);
261 break;
262 case LinkParameterAnchor:
263 m_isValid = false;
264 break;
265 case LinkParameterCrossOrigin:
266 m_crossOrigin = WTFMove(value);
267 break;
268 case LinkParameterAs:
269 m_as = WTFMove(value);
270 break;
271 case LinkParameterType:
272 m_mimeType = WTFMove(value);
273 break;
274 case LinkParameterMedia:
275 m_media = WTFMove(value);
276 break;
277 case LinkParameterImageSrcSet:
278 m_imageSrcSet = WTFMove(value);
279 break;
280 case LinkParameterImageSizes:
281 m_imageSizes = WTFMove(value);
282 break;
283 case LinkParameterTitle:
284 case LinkParameterRev:
285 case LinkParameterHreflang:
286 case LinkParameterUnknown:
287 // These parameters are not yet supported, so they are currently ignored.
288 break;
289 }
290 // FIXME: Add support for more header parameters as neccessary.
291}

293template<typename CharacterType> static void findNextHeader(StringParsingBuffer<CharacterType>& buffer)
294{
295 skipUntil(buffer, ',');
296 skipExactly(buffer, ',');
297}
298
299template<typename CharacterType> LinkHeader::LinkHeader(StringParsingBuffer<CharacterType>& buffer)
300{
301 auto urlResult = findURLBoundaries(buffer);
302 if (urlResult == std::nullopt) {
303 m_isValid = false;
304 findNextHeader(buffer);
305 return;
306 }
307 m_url = urlResult.value();
308
309 while (m_isValid && buffer.hasCharactersRemaining()) {
310 if (!parseParameterDelimiter(buffer, m_isValid)) {
311 findNextHeader(buffer);
312 return;
313 }
314
315 auto parameterName = parseParameterName(buffer);
316 if (!parameterName) {
317 findNextHeader(buffer);
318 m_isValid = false;
319 return;
320 }
321
322 String parameterValue;
323 if (!parseParameterValue(buffer, parameterValue) && !isExtensionParameter(*parameterName)) {
324 findNextHeader(buffer);
325 m_isValid = false;
326 return;
327 }
328
329 setValue(*parameterName, WTFMove(parameterValue));
330 }
331 findNextHeader(buffer);
332}

334LinkHeaderSet::LinkHeaderSet(const String& header)
335{
336 readCharactersForParsing(header, [&](auto buffer) {
337 while (buffer.hasCharactersRemaining())
338 m_headerSet.append(LinkHeader { buffer });
339 });
