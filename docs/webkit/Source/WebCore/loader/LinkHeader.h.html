33namespace WebCore {
34
35class LinkHeader {
36public:
37 template<typename CharacterType> LinkHeader(StringParsingBuffer<CharacterType>&);
38
39 const String& url() const { return m_url; }
40 const String& rel() const { return m_rel; }
41 const String& as() const { return m_as; }
42 const String& mimeType() const { return m_mimeType; }
43 const String& media() const { return m_media; }
44 const String& crossOrigin() const { return m_crossOrigin; }
45 const String& imageSrcSet() const { return m_imageSrcSet; }
46 const String& imageSizes() const { return m_imageSizes; }
47 bool valid() const { return m_isValid; }
48 bool isViewportDependent() const { return !media().isEmpty() || !imageSrcSet().isEmpty() || !imageSizes().isEmpty(); }
49
50 enum LinkParameterName {
51 LinkParameterRel,
52 LinkParameterAnchor,
53 LinkParameterTitle,
54 LinkParameterMedia,
55 LinkParameterType,
56 LinkParameterRev,
57 LinkParameterHreflang,
58 // Beyond this point, only link-extension parameters
59 LinkParameterUnknown,
60 LinkParameterCrossOrigin,
61 LinkParameterAs,
62 LinkParameterImageSrcSet,
63 LinkParameterImageSizes,
64 };
65
66private:
67 void setValue(LinkParameterName, String&& value);
68
69 String m_url;
70 String m_rel;
71 String m_as;
72 String m_mimeType;
73 String m_media;
74 String m_crossOrigin;
75 String m_imageSrcSet;
76 String m_imageSizes;
77 bool m_isValid { true };
78};
79
80class LinkHeaderSet {
81public:
82 LinkHeaderSet(const String& header);
83
84 Vector<LinkHeader>::const_iterator begin() const { return m_headerSet.begin(); }
85 Vector<LinkHeader>::const_iterator end() const { return m_headerSet.end(); }
86
87private:
88 Vector<LinkHeader> m_headerSet;
89};
