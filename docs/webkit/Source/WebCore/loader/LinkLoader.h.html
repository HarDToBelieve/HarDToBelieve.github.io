41#include <wtf/WeakPtr.h>

43namespace WebCore {
44
45class Document;
46class LinkPreloadResourceClient;
47
48struct LinkLoadParameters {
49 LinkRelAttribute relAttribute;
50 URL href;
51 String as;
52 String media;
53 String mimeType;
54 String crossOrigin;
55 String imageSrcSet;
56 String imageSizes;
57 ReferrerPolicy referrerPolicy { ReferrerPolicy::EmptyString };
58};
59
60class LinkLoader : private CachedResourceClient, public CanMakeWeakPtr<LinkLoader> {
61public:
62 explicit LinkLoader(LinkLoaderClient&);
63 virtual ~LinkLoader();
64
65 void loadLink(const LinkLoadParameters&, Document&);
66 static std::optional<CachedResource::Type> resourceTypeFromAsAttribute(const String&, Document&);
67
68 enum class MediaAttributeCheck { MediaAttributeEmpty, MediaAttributeNotEmpty, SkipMediaAttributeCheck };
69 static void loadLinksFromHeader(const String& headerValue, const URL& baseURL, Document&, MediaAttributeCheck);
70 static bool isSupportedType(CachedResource::Type, const String& mimeType, Document&);
71
72 void triggerEvents(const CachedResource&);
73 void cancelLoad();
74
75private:
76 void notifyFinished(CachedResource&, const NetworkLoadMetrics&) override;
77 static void preconnectIfNeeded(const LinkLoadParameters&, Document&);
78 static std::unique_ptr<LinkPreloadResourceClient> preloadIfNeeded(const LinkLoadParameters&, Document&, LinkLoader*);
79 void prefetchIfNeeded(const LinkLoadParameters&, Document&);
80
81 LinkLoaderClient& m_client;
82 CachedResourceHandle<CachedResource> m_cachedLinkResource;
83 std::unique_ptr<LinkPreloadResourceClient> m_preloadResourceClient;
84};
