39class ContentSecurityPolicy;

56struct FetchOptions;

58class WEBCORE_EXPORT LoaderStrategy {
59public:
60 virtual void loadResource(Frame&, CachedResource&, ResourceRequest&&, const ResourceLoaderOptions&, CompletionHandler<void(RefPtr<SubresourceLoader>&&)>&&) = 0;
61 virtual void loadResourceSynchronously(FrameLoader&, unsigned long identifier, const ResourceRequest&, ClientCredentialPolicy, const FetchOptions&, const HTTPHeaderMap&, ResourceError&, ResourceResponse&, Vector<uint8_t>& data) = 0;
62 virtual void pageLoadCompleted(Page&) = 0;
63 virtual void browsingContextRemoved(Frame&) = 0;
64
65 virtual void remove(ResourceLoader*) = 0;
66 virtual void setDefersLoading(ResourceLoader&, bool) = 0;
67 virtual void crossOriginRedirectReceived(ResourceLoader*, const URL& redirectURL) = 0;
68
69 virtual void servePendingRequests(ResourceLoadPriority minimumPriority = ResourceLoadPriority::VeryLow) = 0;
70 virtual void suspendPendingRequests() = 0;
71 virtual void resumePendingRequests() = 0;
72
73 virtual void setResourceLoadSchedulingMode(Page&, LoadSchedulingMode);
74 virtual void prioritizeResourceLoads(const Vector<SubresourceLoader*>&);
75
76 virtual bool usePingLoad() const { return true; }
77 using PingLoadCompletionHandler = WTF::Function<void(const ResourceError&, const ResourceResponse&)>;
78 virtual void startPingLoad(Frame&, ResourceRequest&, const HTTPHeaderMap& originalRequestHeaders, const FetchOptions&, ContentSecurityPolicyImposition, PingLoadCompletionHandler&& = { }) = 0;
79
80 using PreconnectCompletionHandler = WTF::Function<void(const ResourceError&)>;
81 virtual void preconnectTo(FrameLoader&, const URL&, StoredCredentialsPolicy, PreconnectCompletionHandler&&) = 0;
82
83 virtual void setCaptureExtraNetworkLoadMetricsEnabled(bool) = 0;
84
85 virtual bool isOnLine() const = 0;
86 virtual void addOnlineStateChangeListener(WTF::Function<void(bool)>&&) = 0;
87
88 virtual bool shouldPerformSecurityChecks() const { return false; }
89 virtual bool havePerformedSecurityChecks(const ResourceResponse&) const { return false; }
90
91 virtual ResourceResponse responseFromResourceLoadIdentifier(uint64_t resourceLoadIdentifier);
92 virtual Vector<NetworkTransactionInformation> intermediateLoadInformationFromResourceLoadIdentifier(uint64_t resourceLoadIdentifier);
93 virtual NetworkLoadMetrics networkMetricsFromResourceLoadIdentifier(uint64_t resourceLoadIdentifier);
94
95 virtual void isResourceLoadFinished(CachedResource&, CompletionHandler<void(bool)>&& callback) = 0;
96
97 // Used for testing only.
98 virtual Vector<uint64_t> ongoingLoads() const { return { }; }
99
100protected:
101 virtual ~LoaderStrategy();
102};
103
104} // namespace WebCore
