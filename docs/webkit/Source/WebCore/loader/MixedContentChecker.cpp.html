45// static
46bool MixedContentChecker::isMixedContent(SecurityOrigin& securityOrigin, const URL& url)
47{
48 if (securityOrigin.protocol() != "https")
49 return false; // We only care about HTTPS security origins.
50
51 // We're in a secure context, so |url| is mixed content if it's insecure.
52 return !SecurityOrigin::isSecure(url);
53}
54
55static void logWarning(const Frame& frame, bool allowed, const String& action, const URL& target)
56{
57 const char* errorString = allowed ? " was allowed to " : " was not allowed to ";
58 auto message = makeString((allowed ? String() : "[blocked] "), "The page at ", frame.document()->url().stringCenterEllipsizedToLength(), errorString, action, " insecure content from ", target.stringCenterEllipsizedToLength(), ".\n");
59 frame.document()->addConsoleMessage(MessageSource::Security, MessageLevel::Warning, message);
60}

62bool MixedContentChecker::canDisplayInsecureContent(Frame& frame, SecurityOrigin& securityOrigin, ContentType type, const URL& url, AlwaysDisplayInNonStrictMode alwaysDisplayInNonStrictMode)
63{
64 if (!isMixedContent(securityOrigin, url))
65 return true;
66
67 if (!frame.document()->contentSecurityPolicy()->allowRunningOrDisplayingInsecureContent(url))
68 return false;
69
70 bool isStrictMode = frame.document()->isStrictMixedContentMode();
71 if (!isStrictMode && alwaysDisplayInNonStrictMode == AlwaysDisplayInNonStrictMode::Yes)
72 return true;
73
74 bool allowed = !isStrictMode && (frame.settings().allowDisplayOfInsecureContent() || type == ContentType::ActiveCanWarn) && !frame.document()->geolocationAccessed();
75 logWarning(frame, allowed, "display", url);
76
77 if (allowed) {
78 frame.document()->setFoundMixedContent(SecurityContext::MixedContentType::Inactive);
79 frame.loader().client().didDisplayInsecureContent();
80 }
81
82 return allowed;
83}
84
85bool MixedContentChecker::canRunInsecureContent(Frame& frame, SecurityOrigin& securityOrigin, const URL& url)
86{
87 if (!isMixedContent(securityOrigin, url))
88 return true;
89
90 if (!frame.document()->contentSecurityPolicy()->allowRunningOrDisplayingInsecureContent(url))
91 return false;
92
93 bool allowed = !frame.document()->isStrictMixedContentMode() && frame.settings().allowRunningOfInsecureContent() && !frame.document()->geolocationAccessed() && !frame.document()->secureCookiesAccessed();
94 logWarning(frame, allowed, "run", url);
95
96 if (allowed) {
97 frame.document()->setFoundMixedContent(SecurityContext::MixedContentType::Active);
98 frame.loader().client().didRunInsecureContent(securityOrigin, url);
99 }
100
101 return allowed;
102}

104void MixedContentChecker::checkFormForMixedContent(Frame& frame, SecurityOrigin& securityOrigin, const URL& url)
105{
106 // Unconditionally allow javascript: URLs as form actions as some pages do this and it does not introduce
107 // a mixed content issue.
108 if (url.protocolIsJavaScript())
109 return;
110
111 if (!isMixedContent(securityOrigin, url))
112 return;
113
114 auto message = makeString("The page at ", frame.document()->url().stringCenterEllipsizedToLength(), " contains a form which targets an insecure URL ", url.stringCenterEllipsizedToLength(), ".\n");
115 frame.document()->addConsoleMessage(MessageSource::Security, MessageLevel::Warning, message);
116
117 frame.loader().client().didDisplayInsecureContent();
118}

120std::optional<String> MixedContentChecker::checkForMixedContentInFrameTree(const Frame& frame, const URL& url)
121{
122 auto* document = frame.document();
123
124 while (document) {
125 RELEASE_ASSERT_WITH_MESSAGE(document->frame(), "An unparented document tried to connect to a websocket with url: %s", url.string().utf8().data());
126
127 auto* frame = document->frame();
128 if (isMixedContent(document->securityOrigin(), url))
129 return makeString("The page at ", document->url().stringCenterEllipsizedToLength(), " was blocked from connecting insecurely to ", url.stringCenterEllipsizedToLength(), " either because the protocol is insecure or the page is embedded from an insecure page.");
130
131 if (frame->isMainFrame())
132 break;
133
134 frame = frame->tree().parent();
135 RELEASE_ASSERT_WITH_MESSAGE(frame, "Should never have a parentless non main frame");
136 document = frame->document();
137 }
138
139 return std::nullopt;
140}
