63static bool isAllowedByContentSecurityPolicy(const URL& url, const Element* ownerElement, bool didReceiveRedirectResponse)
64{
65 if (!ownerElement)
66 return true;
67 // Elements in user agent show tree should load whatever the embedding document policy is.
68 if (ownerElement->isInUserAgentShadowTree())
69 return true;
70
71 auto redirectResponseReceived = didReceiveRedirectResponse ? ContentSecurityPolicy::RedirectResponseReceived::Yes : ContentSecurityPolicy::RedirectResponseReceived::No;
72
73 ASSERT(ownerElement->document().contentSecurityPolicy());
74 if (is<HTMLPlugInElement>(ownerElement))
75 return ownerElement->document().contentSecurityPolicy()->allowObjectFromSource(url, redirectResponseReceived);
76 return ownerElement->document().contentSecurityPolicy()->allowChildFrameFromSource(url, redirectResponseReceived);
77}

79PolicyCheckIdentifier PolicyCheckIdentifier::create()
80{
81 static uint64_t identifier = 0;
82 identifier++;
83 return PolicyCheckIdentifier { Process::identifier(), identifier };

86bool PolicyCheckIdentifier::isValidFor(PolicyCheckIdentifier expectedIdentifier)
87{
88 RELEASE_ASSERT_WITH_MESSAGE(m_policyCheck, "Received 0 as the policy check identifier");
89 RELEASE_ASSERT_WITH_MESSAGE(m_process == expectedIdentifier.m_process, "Received a policy check response for a wrong process");
90 RELEASE_ASSERT_WITH_MESSAGE(m_policyCheck <= expectedIdentifier.m_policyCheck, "Received a policy check response from the future");
91 return m_policyCheck == expectedIdentifier.m_policyCheck;
92}

107CompletionHandlerCallingScope FrameLoader::PolicyChecker::extendBlobURLLifetimeIfNecessary(ResourceRequest& request, DocumentLoader* loader) const
108{
109 if (!request.url().protocolIsBlob())
110 return { };
111
112 // Create a new temporary blobURL in case this one gets revoked during the asynchronous navigation policy decision.
113 auto origin = SecurityOrigin::create(BlobURL::getOriginURL(request.url()));
114 URL temporaryBlobURL = BlobURL::createPublicURL(origin.ptr());
115 ThreadableBlobRegistry::registerBlobURL(origin.ptr(), temporaryBlobURL, request.url());
116 request.setURL(temporaryBlobURL);
117 if (loader)
118 loader->request().setURL(temporaryBlobURL);
119 return CompletionHandler<void()>([temporaryBlobURL = WTFMove(temporaryBlobURL)] {
120 ThreadableBlobRegistry::unregisterBlobURL(temporaryBlobURL);
121 });
122}

124void FrameLoader::PolicyChecker::checkNavigationPolicy(ResourceRequest&& request, const ResourceResponse& redirectResponse, DocumentLoader* loader, RefPtr<FormState>&& formState, NavigationPolicyDecisionFunction&& function, PolicyDecisionMode policyDecisionMode)
125{
126 NavigationAction action = loader->triggeringAction();
127 if (action.isEmpty()) {
128 action = NavigationAction { *m_frame.document(), request, InitiatedByMainFrame::Unknown, NavigationType::Other, loader->shouldOpenExternalURLsPolicyToPropagate() };
129 loader->setTriggeringAction(NavigationAction { action });
130 }
131
132 if (m_frame.page() && m_frame.page()->openedByDOMWithOpener())
133 action.setOpenedByDOMWithOpener();
134 action.setHasOpenedFrames(m_frame.loader().hasOpenedFrames());
135
136 // Don't ask more than once for the same request or if we are loading an empty URL.
137 // This avoids confusion on the part of the client.
138 if (equalIgnoringHeaderFields(request, loader->lastCheckedRequest()) || (!request.isNull() && request.url().isEmpty())) {
139 if (!request.isNull() && request.url().isEmpty())
140 POLICYCHECKER_RELEASE_LOG("checkNavigationPolicy: continuing because the URL is empty");
141 else
142 POLICYCHECKER_RELEASE_LOG("checkNavigationPolicy: continuing because the URL is the same as the last request");
143 function(ResourceRequest(request), { }, NavigationPolicyDecision::ContinueLoad);
144 loader->setLastCheckedRequest(WTFMove(request));
145 return;
146 }
147
148 // We are always willing to show alternate content for unreachable URLs;
149 // treat it like a reload so it maintains the right state for b/f list.
150 auto& substituteData = loader->substituteData();
151 if (substituteData.isValid() && !substituteData.failingURL().isEmpty()) {
152 bool shouldContinue = true;
153#if ENABLE(CONTENT_FILTERING)
154 if (auto loader = m_frame.loader().activeDocumentLoader())
155 shouldContinue = ContentFilter::continueAfterSubstituteDataRequest(*loader, substituteData);
156#endif
157 if (isBackForwardLoadType(m_loadType))
158 m_loadType = FrameLoadType::Reload;
159 if (shouldContinue)
160 POLICYCHECKER_RELEASE_LOG("checkNavigationPolicy: continuing because we have valid substitute data");
161 else
162 POLICYCHECKER_RELEASE_LOG("checkNavigationPolicy: not continuing with substitute data because the content filter told us not to");
163
164 function(WTFMove(request), { }, shouldContinue ? NavigationPolicyDecision::ContinueLoad : NavigationPolicyDecision::IgnoreLoad);
165 return;
166 }
167
168 if (!isAllowedByContentSecurityPolicy(request.url(), m_frame.ownerElement(), !redirectResponse.isNull())) {
169 if (m_frame.ownerElement()) {
170 // Fire a load event (even though we were blocked by CSP) as timing attacks would otherwise
171 // reveal that the frame was blocked. This way, it looks like any other cross-origin page load.
172 m_frame.ownerElement()->dispatchEvent(Event::create(eventNames().loadEvent, Event::CanBubble::No, Event::IsCancelable::No));
173 }
174 POLICYCHECKER_RELEASE_LOG("checkNavigationPolicy: ignoring because disallowed by content security policy");
175 function(WTFMove(request), { }, NavigationPolicyDecision::IgnoreLoad);
176 return;
177 }
178
179 loader->setLastCheckedRequest(ResourceRequest(request));
180
181#if USE(QUICK_LOOK)
182 // Always allow QuickLook-generated URLs based on the protocol scheme.
183 if (!request.isNull() && isQuickLookPreviewURL(request.url())) {
184 POLICYCHECKER_RELEASE_LOG("checkNavigationPolicy: continuing because quicklook-generated URL");
185 return function(WTFMove(request), makeWeakPtr(formState.get()), NavigationPolicyDecision::ContinueLoad);
186 }
187#endif
188
189#if ENABLE(CONTENT_FILTERING)
190 if (m_contentFilterUnblockHandler.canHandleRequest(request)) {
191 RefPtr<Frame> frame { &m_frame };
192 m_contentFilterUnblockHandler.requestUnblockAsync([frame](bool unblocked) {
193 if (unblocked)
194 frame->loader().reload();
195 });
196 POLICYCHECKER_RELEASE_LOG("checkNavigationPolicy: ignoring because ContentFilterUnblockHandler can handle the request");
197 return function({ }, nullptr, NavigationPolicyDecision::IgnoreLoad);
198 }
199 m_contentFilterUnblockHandler = { };
200#endif
201
202 m_frame.loader().clearProvisionalLoadForPolicyCheck();
203
204 auto blobURLLifetimeExtension = policyDecisionMode == PolicyDecisionMode::Asynchronous ? extendBlobURLLifetimeIfNecessary(request, loader) : CompletionHandlerCallingScope { };
205
206 bool isInitialEmptyDocumentLoad = !m_frame.loader().stateMachine().committedFirstRealDocumentLoad() && request.url().protocolIsAbout() && !substituteData.isValid();
207 auto requestIdentifier = PolicyCheckIdentifier::create();
208 m_delegateIsDecidingNavigationPolicy = true;
209 String suggestedFilename = action.downloadAttribute().isEmpty() ? nullAtom() : action.downloadAttribute();
210 FramePolicyFunction decisionHandler = [this, function = WTFMove(function), request = ResourceRequest(request), formState = std::exchange(formState, nullptr), suggestedFilename = WTFMove(suggestedFilename),
211 blobURLLifetimeExtension = WTFMove(blobURLLifetimeExtension), requestIdentifier, isInitialEmptyDocumentLoad] (PolicyAction policyAction, PolicyCheckIdentifier responseIdentifier) mutable {
212 if (!responseIdentifier.isValidFor(requestIdentifier)) {
213 POLICYCHECKER_RELEASE_LOG("checkNavigationPolicy: ignoring because response is not valid for request");
214 return function({ }, nullptr, NavigationPolicyDecision::IgnoreLoad);
215 }
216
217 m_delegateIsDecidingNavigationPolicy = false;
218
219 switch (policyAction) {
220 case PolicyAction::Download:
221 m_frame.loader().setOriginalURLForDownloadRequest(request);
222 m_frame.loader().client().startDownload(request, suggestedFilename);
223 FALLTHROUGH;
224 case PolicyAction::Ignore:
225 POLICYCHECKER_RELEASE_LOG("checkNavigationPolicy: ignoring because policyAction from dispatchDecidePolicyForNavigationAction is Ignore");
226 return function({ }, nullptr, NavigationPolicyDecision::IgnoreLoad);
227 case PolicyAction::StopAllLoads:
228 POLICYCHECKER_RELEASE_LOG("checkNavigationPolicy: stopping because policyAction from dispatchDecidePolicyForNavigationAction is StopAllLoads");
229 function({ }, nullptr, NavigationPolicyDecision::StopAllLoads);
230 return;
231 case PolicyAction::Use:
232 if (!m_frame.loader().client().canHandleRequest(request)) {
233 handleUnimplementablePolicy(m_frame.loader().client().cannotShowURLError(request));
234 POLICYCHECKER_RELEASE_LOG("checkNavigationPolicy: ignoring because frame loader client can't handle the request");
235 return function({ }, { }, NavigationPolicyDecision::IgnoreLoad);
236 }
237 if (isInitialEmptyDocumentLoad)
238 POLICYCHECKER_RELEASE_LOG("checkNavigationPolicy: continuing because this is an initial empty document");
239 else
240 POLICYCHECKER_RELEASE_LOG("checkNavigationPolicy: continuing because this policyAction from dispatchDecidePolicyForNavigationAction is Use");
241 return function(WTFMove(request), makeWeakPtr(formState.get()), NavigationPolicyDecision::ContinueLoad);
242 }
243 ASSERT_NOT_REACHED();
244 };
245
246 if (isInitialEmptyDocumentLoad) {
247 // We ignore the response from the client for initial empty document loads and proceed with the load synchronously.
248 m_frame.loader().client().dispatchDecidePolicyForNavigationAction(action, request, redirectResponse, formState.get(), policyDecisionMode, requestIdentifier, [](PolicyAction, PolicyCheckIdentifier) { });
249 decisionHandler(PolicyAction::Use, requestIdentifier);
250 } else
251 m_frame.loader().client().dispatchDecidePolicyForNavigationAction(action, request, redirectResponse, formState.get(), policyDecisionMode, requestIdentifier, WTFMove(decisionHandler));
252}

254void FrameLoader::PolicyChecker::checkNewWindowPolicy(NavigationAction&& navigationAction, ResourceRequest&& request, RefPtr<FormState>&& formState, const String& frameName, NewWindowPolicyDecisionFunction&& function)
255{
256 if (m_frame.document() && m_frame.document()->isSandboxed(SandboxPopups))
257 return function({ }, nullptr, { }, { }, ShouldContinuePolicyCheck::No);
258
259 if (!DOMWindow::allowPopUp(m_frame))
260 return function({ }, nullptr, { }, { }, ShouldContinuePolicyCheck::No);
261
262 auto blobURLLifetimeExtension = extendBlobURLLifetimeIfNecessary(request, nullptr);
263
264 auto requestIdentifier = PolicyCheckIdentifier::create();
265 m_frame.loader().client().dispatchDecidePolicyForNewWindowAction(navigationAction, request, formState.get(), frameName, requestIdentifier, [frame = makeRef(m_frame), request,
266 formState = WTFMove(formState), frameName, navigationAction, function = WTFMove(function), blobURLLifetimeExtension = WTFMove(blobURLLifetimeExtension),
267 requestIdentifier] (PolicyAction policyAction, PolicyCheckIdentifier responseIdentifier) mutable {
268
269 if (!responseIdentifier.isValidFor(requestIdentifier))
270 return function({ }, nullptr, { }, { }, ShouldContinuePolicyCheck::No);
271
272 switch (policyAction) {
273 case PolicyAction::Download:
274 frame->loader().client().startDownload(request);
275 FALLTHROUGH;
276 case PolicyAction::Ignore:
277 function({ }, nullptr, { }, { }, ShouldContinuePolicyCheck::No);
278 return;
279 case PolicyAction::StopAllLoads:
280 ASSERT_NOT_REACHED();
281 function({ }, nullptr, { }, { }, ShouldContinuePolicyCheck::No);
282 return;
283 case PolicyAction::Use:
284 function(request, makeWeakPtr(formState.get()), frameName, navigationAction, ShouldContinuePolicyCheck::Yes);
285 return;
286 }
287 ASSERT_NOT_REACHED();
288 });
289}

296void FrameLoader::PolicyChecker::cannotShowMIMEType(const ResourceResponse& response)

301void FrameLoader::PolicyChecker::handleUnimplementablePolicy(const ResourceError& error)
302{
303 m_delegateIsHandlingUnimplementablePolicy = true;
304 m_frame.loader().client().dispatchUnableToImplementPolicy(error);
305 m_delegateIsHandlingUnimplementablePolicy = false;
