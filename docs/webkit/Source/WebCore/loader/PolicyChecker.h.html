30#pragma once
31
32#include "FrameLoader.h"
33#include "ResourceRequest.h"
34#include <wtf/WeakPtr.h>
35#include <wtf/text/WTFString.h>
36
37#if ENABLE(CONTENT_FILTERING)
38#include "ContentFilterUnblockHandler.h"
39#endif
40
41namespace WTF {
42template<typename> class CompletionHandler;
43class CompletionHandlerCallingScope;
44}
45
46namespace WebCore {
47
48class DocumentLoader;
49class FormState;
50class Frame;
51class NavigationAction;
52class ResourceError;
53class ResourceResponse;
54
55enum class NavigationPolicyDecision : uint8_t {
56 ContinueLoad,
57 IgnoreLoad,
58 StopAllLoads,
59};
60
61enum class PolicyDecisionMode { Synchronous, Asynchronous };
62
63class FrameLoader::PolicyChecker {
64 WTF_MAKE_NONCOPYABLE(PolicyChecker);
65 WTF_MAKE_FAST_ALLOCATED;
66public:
67 explicit PolicyChecker(Frame&);
68
69 using NavigationPolicyDecisionFunction = CompletionHandler<void(ResourceRequest&&, WeakPtr<FormState>&&, NavigationPolicyDecision)>;
70 using NewWindowPolicyDecisionFunction = CompletionHandler<void(const ResourceRequest&, WeakPtr<FormState>&&, const String& frameName, const NavigationAction&, ShouldContinuePolicyCheck)>;
71
72 void checkNavigationPolicy(ResourceRequest&&, const ResourceResponse& redirectResponse, DocumentLoader*, RefPtr<FormState>&&, NavigationPolicyDecisionFunction&&, PolicyDecisionMode = PolicyDecisionMode::Asynchronous);
73 void checkNavigationPolicy(ResourceRequest&&, const ResourceResponse& redirectResponse, NavigationPolicyDecisionFunction&&);
74 void checkNewWindowPolicy(NavigationAction&&, ResourceRequest&&, RefPtr<FormState>&&, const String& frameName, NewWindowPolicyDecisionFunction&&);
75
76 void stopCheck();
77
78 void cannotShowMIMEType(const ResourceResponse&);
79
80 FrameLoadType loadType() const { return m_loadType; }
81 void setLoadType(FrameLoadType loadType) { m_loadType = loadType; }
82
83 bool delegateIsDecidingNavigationPolicy() const { return m_delegateIsDecidingNavigationPolicy; }
84 bool delegateIsHandlingUnimplementablePolicy() const { return m_delegateIsHandlingUnimplementablePolicy; }
85
86#if ENABLE(CONTENT_FILTERING)
87 void setContentFilterUnblockHandler(ContentFilterUnblockHandler unblockHandler) { m_contentFilterUnblockHandler = WTFMove(unblockHandler); }
88#endif
89
90private:
91 void handleUnimplementablePolicy(const ResourceError&);
92 WTF::CompletionHandlerCallingScope extendBlobURLLifetimeIfNecessary(ResourceRequest&, DocumentLoader*) const;
93
94 Frame& m_frame;
95
96 bool m_delegateIsDecidingNavigationPolicy;
97 bool m_delegateIsHandlingUnimplementablePolicy;
98
99 // This identifies the type of navigation action which prompted this load. Note
100 // that WebKit conveys this value as the WebActionNavigationTypeKey value
101 // on navigation action delegate callbacks.
102 FrameLoadType m_loadType;
103
104#if ENABLE(CONTENT_FILTERING)
105 ContentFilterUnblockHandler m_contentFilterUnblockHandler;
106#endif
107};
108
109} // namespace WebCore
