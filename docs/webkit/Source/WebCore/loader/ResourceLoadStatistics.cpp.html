89void ResourceLoadStatistics::encode(KeyedEncoder& encoder) const
90{
91 encoder.encodeString("PrevalentResourceDomain"_s, registrableDomain.string());
92
93 encoder.encodeDouble("lastSeen"_s, lastSeen.secondsSinceEpoch().value());
94
95 // User interaction
96 encoder.encodeBool("hadUserInteraction"_s, hadUserInteraction);
97 encoder.encodeDouble("mostRecentUserInteraction"_s, mostRecentUserInteractionTime.secondsSinceEpoch().value());
98 encoder.encodeBool("grandfathered"_s, grandfathered);
99
100 // Storage access
101 encodeHashSet(encoder, "storageAccessUnderTopFrameDomains"_s, "domain"_s, storageAccessUnderTopFrameDomains);
102
103 // Top frame stats
104 encodeHashSet(encoder, "topFrameUniqueRedirectsTo"_s, "domain"_s, topFrameUniqueRedirectsTo);
105 encodeHashSet(encoder, "topFrameUniqueRedirectsFrom"_s, "domain"_s, topFrameUniqueRedirectsFrom);
106 encodeHashSet(encoder, "topFrameLinkDecorationsFrom"_s, "domain", topFrameLinkDecorationsFrom);
107 encoder.encodeBool("gotLinkDecorationFromPrevalentResource"_s, gotLinkDecorationFromPrevalentResource);
108 encodeHashSet(encoder, "topFrameLoadedThirdPartyScripts"_s, "domain", topFrameLoadedThirdPartyScripts);
109
110 // Subframe stats
111 encodeHashSet(encoder, "subframeUnderTopFrameDomains"_s, "domain"_s, subframeUnderTopFrameDomains);
112
113 // Subresource stats
114 encodeHashSet(encoder, "subresourceUnderTopFrameDomains"_s, "domain"_s, subresourceUnderTopFrameDomains);
115 encodeHashSet(encoder, "subresourceUniqueRedirectsTo"_s, "domain"_s, subresourceUniqueRedirectsTo);
116 encodeHashSet(encoder, "subresourceUniqueRedirectsFrom"_s, "domain"_s, subresourceUniqueRedirectsFrom);
117
118 // Prevalent Resource
119 encoder.encodeBool("isPrevalentResource"_s, isPrevalentResource);
120 encoder.encodeBool("isVeryPrevalentResource"_s, isVeryPrevalentResource);
121 encoder.encodeUInt32("dataRecordsRemoved"_s, dataRecordsRemoved);
122
123 encoder.encodeUInt32("timesAccessedAsFirstPartyDueToUserInteraction"_s, timesAccessedAsFirstPartyDueToUserInteraction);
124 encoder.encodeUInt32("timesAccessedAsFirstPartyDueToStorageAccessAPI"_s, timesAccessedAsFirstPartyDueToStorageAccessAPI);
125
126#if ENABLE(WEB_API_STATISTICS)
127 encodeFontHashSet(encoder, "fontsFailedToLoad", fontsFailedToLoad);
128 encodeFontHashSet(encoder, "fontsSuccessfullyLoaded", fontsSuccessfullyLoaded);
129 encodeHashSet(encoder, "topFrameRegistrableDomainsWhichAccessedWebAPIs", "domain", topFrameRegistrableDomainsWhichAccessedWebAPIs);
130 encodeCanvasActivityRecord(encoder, "canvasActivityRecord", canvasActivityRecord);
131 encodeOptionSet(encoder, "navigatorFunctionsAccessedBitMask", navigatorFunctionsAccessed);
132 encodeOptionSet(encoder, "screenFunctionsAccessedBitMask", screenFunctionsAccessed);
133#endif
134}

136static void decodeHashCountedSet(KeyedDecoder& decoder, const String& label, HashCountedSet<RegistrableDomain>& hashCountedSet)
137{
138 Vector<String> ignored;
139IGNORE_WARNINGS_BEGIN("unused-result")
140 decoder.decodeObjects(label, ignored, [&hashCountedSet](KeyedDecoder& decoderInner, String& domain) {
141 if (!decoderInner.decodeString("origin", domain))
142 return false;
143
144 unsigned count;
145 if (!decoderInner.decodeUInt32("count", count))
146 return false;
147
148 hashCountedSet.add(RegistrableDomain::uncheckedCreateFromRegistrableDomainString(domain), count);
149 return true;
150 });
151IGNORE_WARNINGS_END
152}

154static void decodeHashSet(KeyedDecoder& decoder, const String& label, const String& key, HashSet<RegistrableDomain>& hashSet)
155{
156 Vector<String> ignored;
157IGNORE_WARNINGS_BEGIN("unused-result")
158 decoder.decodeObjects(label, ignored, [&hashSet, &key](KeyedDecoder& decoderInner, String& domain) {
159 if (!decoderInner.decodeString(key, domain))
160 return false;
161
162 hashSet.add(RegistrableDomain::uncheckedCreateFromRegistrableDomainString(domain));
163 return true;
164 });
165IGNORE_WARNINGS_END
166}

196static void decodeCanvasActivityRecord(KeyedDecoder& decoder, const String& label, CanvasActivityRecord& canvasActivityRecord)
197{
198IGNORE_WARNINGS_BEGIN("unused-result")
199 decoder.decodeObject(label, canvasActivityRecord, [] (KeyedDecoder& decoderInner, CanvasActivityRecord& canvasActivityRecord) {
200 if (!decoderInner.decodeBool("wasDataRead", canvasActivityRecord.wasDataRead))
201 return false;
202 Vector<String> ignore;
203 decoderInner.decodeObjects("textWritten", ignore, [&canvasActivityRecord] (KeyedDecoder& decoderInner2, String& text) {
204 if (!decoderInner2.decodeString("text", text))
205 return false;
206 canvasActivityRecord.textWritten.add(text);
207 return true;
208 });
209 return true;
210 });
211IGNORE_WARNINGS_END
212}

215bool ResourceLoadStatistics::decode(KeyedDecoder& decoder, unsigned modelVersion)
216{
217 String registrableDomainAsString;
218 if (modelVersion >= 15) {
219 if (!decoder.decodeString("PrevalentResourceDomain", registrableDomainAsString))
220 return false;
221 } else {
222 if (!decoder.decodeString("PrevalentResourceOrigin", registrableDomainAsString))
223 return false;
224 }
225 registrableDomain = RegistrableDomain::uncheckedCreateFromRegistrableDomainString(registrableDomainAsString);
226
227 // User interaction
228 if (!decoder.decodeBool("hadUserInteraction", hadUserInteraction))
229 return false;
230
231 // Storage access
232 if (modelVersion >= 15)
233 decodeHashSet(decoder, "storageAccessUnderTopFrameDomains", "domain", storageAccessUnderTopFrameDomains);
234 else
235 decodeHashSet(decoder, "storageAccessUnderTopFrameOrigins", "origin", storageAccessUnderTopFrameDomains);
236
237 // Top frame stats
238 if (modelVersion >= 15) {
239 decodeHashSet(decoder, "topFrameUniqueRedirectsTo", "domain", topFrameUniqueRedirectsTo);
240 decodeHashSet(decoder, "topFrameUniqueRedirectsFrom", "domain", topFrameUniqueRedirectsFrom);
241 } else if (modelVersion >= 11) {
242 HashCountedSet<RegistrableDomain> topFrameUniqueRedirectsToCounted;
243 decodeHashCountedSet(decoder, "topFrameUniqueRedirectsTo", topFrameUniqueRedirectsToCounted);
244 for (auto& domain : topFrameUniqueRedirectsToCounted.values())
245 topFrameUniqueRedirectsTo.add(domain);
246
247 HashCountedSet<RegistrableDomain> topFrameUniqueRedirectsFromCounted;
248 decodeHashCountedSet(decoder, "topFrameUniqueRedirectsFrom", topFrameUniqueRedirectsFromCounted);
249 for (auto& domain : topFrameUniqueRedirectsFromCounted.values())
250 topFrameUniqueRedirectsFrom.add(domain);
251 }
252
253 if (modelVersion >= 16) {
254 decodeHashSet(decoder, "topFrameLinkDecorationsFrom", "domain", topFrameLinkDecorationsFrom);
255 if (!decoder.decodeBool("gotLinkDecorationFromPrevalentResource", gotLinkDecorationFromPrevalentResource))
256 return false;
257 }
258
259 if (modelVersion >= 17) {
260 HashCountedSet<RegistrableDomain> topFrameLoadedThirdPartyScriptsCounted;
261 decodeHashCountedSet(decoder, "topFrameLoadedThirdPartyScripts", topFrameLoadedThirdPartyScriptsCounted);
262 for (auto& domain : topFrameLoadedThirdPartyScriptsCounted.values())
263 topFrameLoadedThirdPartyScripts.add(domain);
264 }
265
266 // Subframe stats
267 if (modelVersion >= 15)
268 decodeHashSet(decoder, "subframeUnderTopFrameDomains", "domain", subframeUnderTopFrameDomains);
269 else if (modelVersion >= 14) {
270 HashCountedSet<RegistrableDomain> subframeUnderTopFrameDomainsCounted;
271 decodeHashCountedSet(decoder, "subframeUnderTopFrameOrigins", subframeUnderTopFrameDomainsCounted);
272 for (auto& domain : subframeUnderTopFrameDomainsCounted.values())
273 subframeUnderTopFrameDomains.add(domain);
274 }
275
276 // Subresource stats
277 if (modelVersion >= 15) {
278 decodeHashSet(decoder, "subresourceUnderTopFrameDomains", "domain", subresourceUnderTopFrameDomains);
279 decodeHashSet(decoder, "subresourceUniqueRedirectsTo", "domain", subresourceUniqueRedirectsTo);
280 decodeHashSet(decoder, "subresourceUniqueRedirectsFrom", "domain", subresourceUniqueRedirectsFrom);
281 } else {
282 HashCountedSet<RegistrableDomain> subresourceUnderTopFrameDomainsCounted;
283 decodeHashCountedSet(decoder, "subresourceUnderTopFrameOrigins", subresourceUnderTopFrameDomainsCounted);
284 for (auto& domain : subresourceUnderTopFrameDomainsCounted.values())
285 subresourceUnderTopFrameDomains.add(domain);
286
287 HashCountedSet<RegistrableDomain> subresourceUniqueRedirectsToCounted;
288 decodeHashCountedSet(decoder, "subresourceUniqueRedirectsTo", subresourceUniqueRedirectsToCounted);
289 for (auto& domain : subresourceUniqueRedirectsToCounted.values())
290 subresourceUniqueRedirectsTo.add(domain);
291 if (modelVersion >= 11) {
292 HashCountedSet<RegistrableDomain> subresourceUniqueRedirectsFromCounted;
293 decodeHashCountedSet(decoder, "subresourceUniqueRedirectsFrom", subresourceUniqueRedirectsFromCounted);
294 for (auto& domain : subresourceUniqueRedirectsFromCounted.values())
295 subresourceUniqueRedirectsFrom.add(domain);
296 }
297 }
298
299
300 // Prevalent Resource
301 if (!decoder.decodeBool("isPrevalentResource", isPrevalentResource))
302 return false;
303
304 if (modelVersion >= 12) {
305 if (!decoder.decodeBool("isVeryPrevalentResource", isVeryPrevalentResource))
306 return false;
307 }
308
309 // Trigger re-classification based on model 14.
310 if (modelVersion < 14) {
311 isPrevalentResource = false;
312 isVeryPrevalentResource = false;
313 }
314
315 if (!decoder.decodeUInt32("dataRecordsRemoved", dataRecordsRemoved))
316 return false;
317
318 double mostRecentUserInteractionTimeAsDouble;
319 if (!decoder.decodeDouble("mostRecentUserInteraction", mostRecentUserInteractionTimeAsDouble))
320 return false;
321 mostRecentUserInteractionTime = WallTime::fromRawSeconds(mostRecentUserInteractionTimeAsDouble);
322
323 if (!decoder.decodeBool("grandfathered", grandfathered))
324 return false;
325
326 double lastSeenTimeAsDouble;
327 if (!decoder.decodeDouble("lastSeen", lastSeenTimeAsDouble))
328 return false;
329 lastSeen = WallTime::fromRawSeconds(lastSeenTimeAsDouble);
330
331 if (modelVersion >= 11) {
332 if (!decoder.decodeUInt32("timesAccessedAsFirstPartyDueToUserInteraction", timesAccessedAsFirstPartyDueToUserInteraction))
333 timesAccessedAsFirstPartyDueToUserInteraction = 0;
334 if (!decoder.decodeUInt32("timesAccessedAsFirstPartyDueToStorageAccessAPI", timesAccessedAsFirstPartyDueToStorageAccessAPI))
335 timesAccessedAsFirstPartyDueToStorageAccessAPI = 0;
336 }
337
338#if ENABLE(WEB_API_STATISTICS)
339 if (modelVersion >= 13) {
340 decodeFontHashSet(decoder, "fontsFailedToLoad", fontsFailedToLoad);
341 decodeFontHashSet(decoder, "fontsSuccessfullyLoaded", fontsSuccessfullyLoaded);
342 decodeHashSet(decoder, "topFrameRegistrableDomainsWhichAccessedWebAPIs", "domain", topFrameRegistrableDomainsWhichAccessedWebAPIs);
343 decodeCanvasActivityRecord(decoder, "canvasActivityRecord", canvasActivityRecord);
344 decodeOptionSet(decoder, "navigatorFunctionsAccessedBitMask", navigatorFunctionsAccessed);
345 decodeOptionSet(decoder, "screenFunctionsAccessedBitMask", screenFunctionsAccessed);
346 }
347#endif
348
349 return true;
350}

442static bool hasHadRecentUserInteraction(WTF::Seconds interactionTimeSeconds)
443{
444 return interactionTimeSeconds > Seconds(0) && WallTime::now().secondsSinceEpoch() - interactionTimeSeconds < 24_h;

447String ResourceLoadStatistics::toString() const
448{
449 StringBuilder builder;
450 builder.append("Registrable domain: ", registrableDomain.string(), '\n');
451
452 // User interaction
453 appendBoolean(builder, "hadUserInteraction", hadUserInteraction);
454 builder.append('\n');
455 builder.append(" mostRecentUserInteraction: ", hasHadRecentUserInteraction(mostRecentUserInteractionTime.secondsSinceEpoch()) ? "within 24 hours" : "-1");
456 builder.append('\n');
457 appendBoolean(builder, "grandfathered", grandfathered);
458 builder.append('\n');
459
460 // Storage access
461 appendHashSet(builder, "storageAccessUnderTopFrameDomains", storageAccessUnderTopFrameDomains);
462
463 // Top frame stats
464 appendHashSet(builder, "topFrameUniqueRedirectsTo", topFrameUniqueRedirectsTo);
465 appendHashSet(builder, "topFrameUniqueRedirectsFrom", topFrameUniqueRedirectsFrom);
466 appendHashSet(builder, "topFrameLinkDecorationsFrom", topFrameLinkDecorationsFrom);
467 appendBoolean(builder, "gotLinkDecorationFromPrevalentResource", gotLinkDecorationFromPrevalentResource);
468 builder.append('\n');
469 appendHashSet(builder, "topFrameLoadedThirdPartyScripts", topFrameLoadedThirdPartyScripts);
470
471 // Subframe stats
472 appendHashSet(builder, "subframeUnderTopFrameDomains", subframeUnderTopFrameDomains);
473
474 // Subresource stats
475 appendHashSet(builder, "subresourceUnderTopFrameDomains", subresourceUnderTopFrameDomains);
476 appendHashSet(builder, "subresourceUniqueRedirectsTo", subresourceUniqueRedirectsTo);
477 appendHashSet(builder, "subresourceUniqueRedirectsFrom", subresourceUniqueRedirectsFrom);
478
479 // Prevalent Resource
480 appendBoolean(builder, "isPrevalentResource", isPrevalentResource);
481 builder.append('\n');
482 appendBoolean(builder, "isVeryPrevalentResource", isVeryPrevalentResource);
483 builder.append('\n');
484 builder.append(" dataRecordsRemoved: ", dataRecordsRemoved);
485 builder.append('\n');
486
487#if ENABLE(WEB_API_STATISTICS)

494 appendBoolean(builder, "canvasReadData", canvasActivityRecord.wasDataRead);
495 builder.append('\n');
496 builder.append('\n');

516void ResourceLoadStatistics::merge(const ResourceLoadStatistics& other)
517{
518 ASSERT(other.registrableDomain == registrableDomain);
519
520 if (lastSeen < other.lastSeen)
521 lastSeen = other.lastSeen;
522
523 if (!other.hadUserInteraction) {
524 // If user interaction has been reset do so here too.
525 // Else, do nothing.
526 if (!other.mostRecentUserInteractionTime) {
527 hadUserInteraction = false;
528 mostRecentUserInteractionTime = { };
529 }
530 } else {
531 hadUserInteraction = true;
532 if (mostRecentUserInteractionTime < other.mostRecentUserInteractionTime)
533 mostRecentUserInteractionTime = other.mostRecentUserInteractionTime;
534 }
535 grandfathered |= other.grandfathered;
536
537 // Storage access
538 mergeHashSet(storageAccessUnderTopFrameDomains, other.storageAccessUnderTopFrameDomains);
539
540 // Top frame stats
541 mergeHashSet(topFrameUniqueRedirectsTo, other.topFrameUniqueRedirectsTo);
542 mergeHashSet(topFrameUniqueRedirectsFrom, other.topFrameUniqueRedirectsFrom);
543 mergeHashSet(topFrameLinkDecorationsFrom, other.topFrameLinkDecorationsFrom);
544 gotLinkDecorationFromPrevalentResource |= other.gotLinkDecorationFromPrevalentResource;
545 mergeHashSet(topFrameLoadedThirdPartyScripts, other.topFrameLoadedThirdPartyScripts);
546
547 // Subframe stats
548 mergeHashSet(subframeUnderTopFrameDomains, other.subframeUnderTopFrameDomains);
549
550 // Subresource stats
551 mergeHashSet(subresourceUnderTopFrameDomains, other.subresourceUnderTopFrameDomains);
552 mergeHashSet(subresourceUniqueRedirectsTo, other.subresourceUniqueRedirectsTo);
553 mergeHashSet(subresourceUniqueRedirectsFrom, other.subresourceUniqueRedirectsFrom);
554
555 // Prevalent resource stats
556 isPrevalentResource |= other.isPrevalentResource;
557 isVeryPrevalentResource |= other.isVeryPrevalentResource;
558 dataRecordsRemoved = std::max(dataRecordsRemoved, other.dataRecordsRemoved);
559
560#if ENABLE(WEB_API_STATISTICS)
561 mergeHashSet(fontsFailedToLoad, other.fontsFailedToLoad);
562 mergeHashSet(fontsSuccessfullyLoaded, other.fontsSuccessfullyLoaded);
563 mergeHashSet(topFrameRegistrableDomainsWhichAccessedWebAPIs, other.topFrameRegistrableDomainsWhichAccessedWebAPIs);
564 canvasActivityRecord.mergeWith(other.canvasActivityRecord);
565 navigatorFunctionsAccessed.add(other.navigatorFunctionsAccessed);
566 screenFunctionsAccessed.add(other.screenFunctionsAccessed);
567#endif
568}

570WallTime ResourceLoadStatistics::reduceTimeResolution(WallTime time)
571{
572 return WallTime::fromRawSeconds(std::floor(time.secondsSinceEpoch() / timestampResolution) * timestampResolution.seconds());
573}
