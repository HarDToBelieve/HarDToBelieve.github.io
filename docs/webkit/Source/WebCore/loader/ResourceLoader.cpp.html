100void ResourceLoader::releaseResources()
101{
102 ASSERT(!m_reachedTerminalState);
103
104 // It's possible that when we release the handle, it will be
105 // deallocated and release the last reference to this object.
106 // We need to retain to avoid accessing the object after it
107 // has been deallocated and also to avoid reentering this method.
108 Ref<ResourceLoader> protectedThis(*this);
109
110 m_frame = nullptr;
111 m_documentLoader = nullptr;
112
113 // We need to set reachedTerminalState to true before we release
114 // the resources to prevent a double dealloc of WebView <rdar://problem/4372628>
115 m_reachedTerminalState = true;
116
117 finishNetworkLoad();
118
119 m_identifier = 0;
120
121 m_resourceData = nullptr;
122 m_deferredRequest = ResourceRequest();
123}

125void ResourceLoader::init(ResourceRequest&& clientRequest, CompletionHandler<void(bool)>&& completionHandler)
126{
127 if (!m_documentLoader || !m_documentLoader->frame()) {
128 if (!m_documentLoader)
129 RESOURCELOADER_RELEASE_LOG("init: Cancelling because there is no document loader.");
130 else
131 RESOURCELOADER_RELEASE_LOG("init: Cancelling because the document loader has no frame.");
132 cancel();
133 return completionHandler(false);
134 }
135
136 ASSERT(!m_handle);
137 ASSERT(m_request.isNull());
138 ASSERT(m_deferredRequest.isNull());
139 ASSERT(!m_documentLoader->isSubstituteLoadPending(this));
140
141 m_loadTiming.markStartTime();
142
143 m_defersLoading = m_options.defersLoadingPolicy == DefersLoadingPolicy::AllowDefersLoading && m_frame->page()->defersLoading();
144
145 if (m_options.securityCheck == SecurityCheckPolicy::DoSecurityCheck && !m_frame->document()->securityOrigin().canDisplay(clientRequest.url())) {
146 RESOURCELOADER_RELEASE_LOG("init: Cancelling load because it violates security policy.");
147 FrameLoader::reportLocalLoadFailed(m_frame.get(), clientRequest.url().string());
148 releaseResources();
149 return completionHandler(false);
150 }
151
152 if (!portAllowed(clientRequest.url())) {
153 RESOURCELOADER_RELEASE_LOG("init: Cancelling load to a blocked port.");
154 FrameLoader::reportBlockedLoadFailed(*m_frame, clientRequest.url());
155 releaseResources();
156 return completionHandler(false);
157 }
158
159 // The various plug-in implementations call directly to ResourceLoader::load() instead of piping requests
160 // through FrameLoader. As a result, they miss the FrameLoader::updateRequestAndAddExtraFields() step which sets
161 // up the 1st party for cookies URL and Same-Site info. Until plug-in implementations can be reigned in
162 // to pipe through that method, we need to make sure there is always both a 1st party for cookies set and
163 // Same-Site info. See <https://bugs.webkit.org/show_bug.cgi?id=26391>.
164 if (clientRequest.firstPartyForCookies().isNull()) {
165 if (Document* document = m_frame->document())
166 clientRequest.setFirstPartyForCookies(document->firstPartyForCookies());
167 }
168 FrameLoader::addSameSiteInfoToRequestIfNeeded(clientRequest, m_frame->document());
169
170 willSendRequestInternal(WTFMove(clientRequest), ResourceResponse(), [this, protectedThis = makeRef(*this), completionHandler = WTFMove(completionHandler)](ResourceRequest&& request) mutable {
171
172#if PLATFORM(IOS_FAMILY)
173 // If this ResourceLoader was stopped as a result of willSendRequest, bail out.
174 if (m_reachedTerminalState) {
175 RESOURCELOADER_RELEASE_LOG("init: Cancelling load because it was stopped as a result of willSendRequest.");
176 return completionHandler(false);
177 }
178#endif
179
180 if (request.isNull()) {
181 RESOURCELOADER_RELEASE_LOG("init: Cancelling load because the request is null.");
182 cancel();
183 return completionHandler(false);
184 }
185
186 m_request = WTFMove(request);
187 m_originalRequest = m_request;
188 completionHandler(true);
189 });
190}

192void ResourceLoader::deliverResponseAndData(const ResourceResponse& response, RefPtr<SharedBuffer>&& buffer)
193{
194 didReceiveResponse(response, [this, protectedThis = makeRef(*this), buffer = WTFMove(buffer)]() mutable {
195 if (reachedTerminalState())
196 return;
197
198 if (buffer) {
199 unsigned size = buffer->size();
200 didReceiveBuffer(buffer.releaseNonNull(), size, DataPayloadWholeResource);
201 if (reachedTerminalState())
202 return;
203 }
204
205 NetworkLoadMetrics emptyMetrics;
206 didFinishLoading(emptyMetrics);
207 });
208}

210void ResourceLoader::start()
211{
212 ASSERT(!m_handle);
213 ASSERT(!m_request.isNull());
214 ASSERT(m_deferredRequest.isNull());
215 ASSERT(frameLoader());
216
217#if ENABLE(WEB_ARCHIVE) || ENABLE(MHTML)
218 if (m_documentLoader && m_documentLoader->scheduleArchiveLoad(*this, m_request))
219 return;
220#endif
221
222 if (m_documentLoader && m_documentLoader->applicationCacheHost().maybeLoadResource(*this, m_request, m_request.url()))
223 return;
224
225 if (m_defersLoading) {
226 m_deferredRequest = m_request;
227 return;
228 }
229
230 if (m_reachedTerminalState)
231 return;
232
233 if (m_request.url().protocolIsData()) {
234 loadDataURL();
235 return;
236 }
237
238#if USE(SOUP)
239 if (m_request.url().protocolIs("resource")) {
240 loadGResource();
241 return;
242 }
243#endif
244
245 RefPtr<SecurityOrigin> sourceOrigin = is<SubresourceLoader>(*this) ? downcast<SubresourceLoader>(*this).origin() : nullptr;
246 if (!sourceOrigin && frameLoader()) {
247 auto* document = frameLoader()->frame().document();
248 sourceOrigin = document ? &document->securityOrigin() : nullptr;
249 }
250
251 bool isMainFrameNavigation = frame() && frame()->isMainFrame() && options().mode == FetchOptions::Mode::Navigate;
252
253 m_handle = ResourceHandle::create(frameLoader()->networkingContext(), m_request, this, m_defersLoading, m_options.sniffContent == ContentSniffingPolicy::SniffContent, m_options.sniffContentEncoding == ContentEncodingSniffingPolicy::Sniff, WTFMove(sourceOrigin), isMainFrameNavigation);
254}

265 platformStrategies()->loaderStrategy()->setDefersLoading(*this, defers);

275void ResourceLoader::loadDataURL()
276{
277 auto url = m_request.url();
278 ASSERT(url.protocolIsData());
279
280 DataURLDecoder::ScheduleContext scheduleContext;
281#if USE(COCOA_EVENT_LOOP)
282 if (auto page = m_frame->page())
283 scheduleContext.scheduledPairs = *page->scheduledRunLoopPairs();
284#endif
285 auto mode = DataURLDecoder::Mode::Legacy;
286 if (m_request.requester() == ResourceRequest::Requester::Fetch)
287 mode = DataURLDecoder::Mode::ForgivingBase64;
288 DataURLDecoder::decode(url, scheduleContext, mode, [this, protectedThis = makeRef(*this), url](auto decodeResult) mutable {
289 if (this->reachedTerminalState())
290 return;
291 if (!decodeResult) {
292 RESOURCELOADER_RELEASE_LOG("loadDataURL: decoding of data failed");
293 protectedThis->didFail(ResourceError(errorDomainWebKitInternal, 0, url, "Data URL decoding failed"));
294 return;
295 }
296 if (this->wasCancelled()) {
297 RESOURCELOADER_RELEASE_LOG("loadDataURL: Load was cancelled");
298 return;
299 }
300
301 auto dataSize = decodeResult->data.size();
302 ResourceResponse dataResponse = ResourceResponse::dataURLResponse(url, decodeResult.value());
303 this->didReceiveResponse(dataResponse, [this, protectedThis = WTFMove(protectedThis), dataSize, data = SharedBuffer::create(WTFMove(decodeResult->data))]() mutable {
304 if (!this->reachedTerminalState() && dataSize && m_request.httpMethod() != "HEAD")
305 this->didReceiveBuffer(WTFMove(data), dataSize, DataPayloadWholeResource);
306
307 if (!this->reachedTerminalState()) {
308 NetworkLoadMetrics emptyMetrics;
309 this->didFinishLoading(emptyMetrics);
310 }
311 });
312 });
313}

332void ResourceLoader::addDataOrBuffer(const uint8_t* data, unsigned length, SharedBuffer* buffer, DataPayloadType dataPayloadType)
333{
334 if (m_options.dataBufferingPolicy == DataBufferingPolicy::DoNotBufferData)
335 return;
336
337 if (!m_resourceData || dataPayloadType == DataPayloadWholeResource) {
338 if (buffer)
339 m_resourceData = buffer;
340 else
341 m_resourceData = SharedBuffer::create(data, length);
342 return;
343 }
344
345 if (buffer)
346 m_resourceData->append(*buffer);
347 else
348 m_resourceData->append(data, length);
349}

362void ResourceLoader::willSendRequestInternal(ResourceRequest&& request, const ResourceResponse& redirectResponse, CompletionHandler<void(ResourceRequest&&)>&& completionHandler)
363{
364 // Protect this in this delegate method since the additional processing can do
365 // anything including possibly derefing this; one example of this is Radar 3266216.
366 Ref<ResourceLoader> protectedThis(*this);
367
368 ASSERT(!m_reachedTerminalState);
369#if ENABLE(CONTENT_EXTENSIONS)
370 ASSERT(!m_resourceType.isEmpty());
371#endif
372
373 // We need a resource identifier for all requests, even if FrameLoader is never going to see it (such as with CORS preflight requests).
374 bool createdResourceIdentifier = false;
375 if (!m_identifier) {
376 m_identifier = m_frame->page()->progress().createUniqueIdentifier();
377 createdResourceIdentifier = true;
378 }
379
380#if ENABLE(CONTENT_EXTENSIONS)
381 if (!redirectResponse.isNull() && frameLoader()) {
382 Page* page = frameLoader()->frame().page();
383 if (page && m_documentLoader) {
384 auto results = page->userContentProvider().processContentRuleListsForLoad(*page, request.url(), m_resourceType, *m_documentLoader, redirectResponse.url());
385 bool blockedLoad = results.summary.blockedLoad;
386 ContentExtensions::applyResultsToRequest(WTFMove(results), page, request);
387 if (blockedLoad) {
388 RESOURCELOADER_RELEASE_LOG("willSendRequestInternal: resource load canceled because of content blocker");
389 didFail(blockedByContentBlockerError());
390 completionHandler({ });
391 return;
392 }
393 }
394 }
395#endif
396
397 if (request.isNull()) {
398 RESOURCELOADER_RELEASE_LOG("willSendRequestInternal: resource load canceled because of empty request");
399 didFail(cannotShowURLError());
400 completionHandler({ });
401 return;
402 }
403
404 if (m_options.sendLoadCallbacks == SendCallbackPolicy::SendCallbacks) {
405 if (createdResourceIdentifier)
406 frameLoader()->notifier().assignIdentifierToInitialRequest(m_identifier, documentLoader(), request);
407
408#if PLATFORM(IOS_FAMILY)
409 // If this ResourceLoader was stopped as a result of assignIdentifierToInitialRequest, bail out
410 if (m_reachedTerminalState) {
411 RESOURCELOADER_RELEASE_LOG("willSendRequestInternal: resource load reached terminal state after calling assignIdentifierToInitialRequest()");
412 completionHandler(WTFMove(request));
413 return;
414 }
415#endif
416
417 frameLoader()->notifier().willSendRequest(this, request, redirectResponse);
418 }
419 else
420 InspectorInstrumentation::willSendRequest(m_frame.get(), m_identifier, m_frame->loader().documentLoader(), request, redirectResponse, cachedResource());
421
422#if USE(QUICK_LOOK)
423 if (m_documentLoader) {
424 if (auto previewConverter = m_documentLoader->previewConverter())
425 request = previewConverter->safeRequest(request);
426 }
427#endif
428
429 bool isRedirect = !redirectResponse.isNull();
430 if (isRedirect) {
431 RESOURCELOADER_RELEASE_LOG("willSendRequestInternal: Processing cross-origin redirect");
432 platformStrategies()->loaderStrategy()->crossOriginRedirectReceived(this, request.url());
433#if ENABLE(RESOURCE_LOAD_STATISTICS)
434 frameLoader()->client().didLoadFromRegistrableDomain(RegistrableDomain(request.url()));
435#endif
436 }
437 m_request = request;
438
439 if (isRedirect) {
440 auto& redirectURL = request.url();
441 if (m_documentLoader && !m_documentLoader->isCommitted())
442 frameLoader()->client().dispatchDidReceiveServerRedirectForProvisionalLoad();
443
444 if (redirectURL.protocolIsData()) {
445 // Handle data URL decoding locally.
446 RESOURCELOADER_RELEASE_LOG("willSendRequestInternal: Redirected to a data URL. Processing locally");
447 finishNetworkLoad();
448 loadDataURL();
449 }
450 }
451
452 RESOURCELOADER_RELEASE_LOG("willSendRequestInternal: calling completion handler");
453 completionHandler(WTFMove(request));
454}

465static void logResourceResponseSource(Frame* frame, ResourceResponse::Source source)
466{
467 if (!frame || !frame->page())
468 return;
469
470 String sourceKey;
471 switch (source) {
472 case ResourceResponse::Source::Network:
473 sourceKey = DiagnosticLoggingKeys::networkKey();
474 break;
475 case ResourceResponse::Source::DiskCache:
476 sourceKey = DiagnosticLoggingKeys::diskCacheKey();
477 break;
478 case ResourceResponse::Source::DiskCacheAfterValidation:
479 sourceKey = DiagnosticLoggingKeys::diskCacheAfterValidationKey();
480 break;
481 case ResourceResponse::Source::ServiceWorker:
482 sourceKey = DiagnosticLoggingKeys::serviceWorkerKey();
483 break;
484 case ResourceResponse::Source::MemoryCache:
485 sourceKey = DiagnosticLoggingKeys::memoryCacheKey();
486 break;
487 case ResourceResponse::Source::MemoryCacheAfterValidation:
488 sourceKey = DiagnosticLoggingKeys::memoryCacheAfterValidationKey();
489 break;
490 case ResourceResponse::Source::DOMCache:
491 case ResourceResponse::Source::ApplicationCache:
492 case ResourceResponse::Source::InspectorOverride:
493 case ResourceResponse::Source::Unknown:
494 return;
495 }
496
497 frame->page()->diagnosticLoggingClient().logDiagnosticMessage(DiagnosticLoggingKeys::resourceResponseSourceKey(), sourceKey, ShouldSample::Yes);
498}

516void ResourceLoader::didReceiveResponse(const ResourceResponse& r, CompletionHandler<void()>&& policyCompletionHandler)
517{
518 ASSERT(!m_reachedTerminalState);
519 CompletionHandlerCallingScope completionHandlerCaller(WTFMove(policyCompletionHandler));
520
521 // Protect this in this delegate method since the additional processing can do
522 // anything including possibly derefing this; one example of this is Radar 3266216.
523 Ref<ResourceLoader> protectedThis(*this);
524
525 if (r.usedLegacyTLS() && m_frame) {
526 if (auto* document = m_frame->document()) {
527 if (!document->usedLegacyTLS()) {
528 if (auto* page = document->page()) {
529 RESOURCELOADER_RELEASE_LOG("usedLegacyTLS:");
530 page->console().addMessage(MessageSource::Network, MessageLevel::Warning, makeString("Loaded resource from ", r.url().host(), " using TLS 1.0 or 1.1, which are deprecated protocols that will be removed. Please use TLS 1.2 or newer instead."), 0, document);
531 }
532 document->setUsedLegacyTLS(true);
533 }
534 }
535 }
536
537 logResourceResponseSource(m_frame.get(), r.source());
538
539 m_response = r;
540
541 if (m_options.sendLoadCallbacks == SendCallbackPolicy::SendCallbacks)
542 frameLoader()->notifier().didReceiveResponse(this, m_response);
543}

561void ResourceLoader::didReceiveDataOrBuffer(const uint8_t* data, unsigned length, RefPtr<SharedBuffer>&& buffer, long long encodedDataLength, DataPayloadType dataPayloadType)
562{
563 // This method should only get data+length *OR* a SharedBuffer.
564 ASSERT(!buffer || (!data && !length));
565
566 // Protect this in this delegate method since the additional processing can do
567 // anything including possibly derefing this; one example of this is Radar 3266216.
568 Ref<ResourceLoader> protectedThis(*this);
569
570 addDataOrBuffer(data, length, buffer.get(), dataPayloadType);
571
572 // FIXME: If we get a resource with more than 2B bytes, this code won't do the right thing.
573 // However, with today's computers and networking speeds, this won't happen in practice.
574 // Could be an issue with a giant local file.
575 if (m_options.sendLoadCallbacks == SendCallbackPolicy::SendCallbacks && m_frame)
576 frameLoader()->notifier().didReceiveData(this, buffer ? buffer->data() : data, buffer ? buffer->size() : length, static_cast<int>(encodedDataLength));
577}

579void ResourceLoader::didFinishLoading(const NetworkLoadMetrics& networkLoadMetrics)
580{
581 RESOURCELOADER_RELEASE_LOG("didFinishLoading:");
582
583 didFinishLoadingOnePart(networkLoadMetrics);
584
585 // If the load has been cancelled by a delegate in response to didFinishLoad(), do not release
586 // the resources a second time, they have been released by cancel.
587 if (wasCancelled())
588 return;
589 releaseResources();
590}
591
592void ResourceLoader::didFinishLoadingOnePart(const NetworkLoadMetrics& networkLoadMetrics)
593{
594 // If load has been cancelled after finishing (which could happen with a
595 // JavaScript that changes the window location), do nothing.
596 if (wasCancelled()) {
597 RESOURCELOADER_RELEASE_LOG("didFinishLoadingOnePart: Load was cancelled after finishing.");
598 return;
599 }
600 ASSERT(!m_reachedTerminalState);
601
602 if (m_notifiedLoadComplete)
603 return;
604 m_notifiedLoadComplete = true;
605 if (m_options.sendLoadCallbacks == SendCallbackPolicy::SendCallbacks)
606 frameLoader()->notifier().didFinishLoad(this, networkLoadMetrics);
607}

609void ResourceLoader::didFail(const ResourceError& error)
610{
611 RESOURCELOADER_RELEASE_LOG("didFail:");
612
613 if (wasCancelled())
614 return;
615 ASSERT(!m_reachedTerminalState);
616
617 // Protect this in this delegate method since the additional processing can do
618 // anything including possibly derefing this; one example of this is Radar 3266216.
619 Ref<ResourceLoader> protectedThis(*this);
620
621 cleanupForError(error);
622 releaseResources();
623}

625void ResourceLoader::cleanupForError(const ResourceError& error)

638
639void ResourceLoader::cancel(const ResourceError& error)
640{
641 // If the load has already completed - succeeded, failed, or previously cancelled - do nothing.
642 if (m_reachedTerminalState)
643 return;
644
645 ResourceError nonNullError = error.isNull() ? cancelledError() : error;
646
647 // willCancel() and didFailToLoad() both call out to clients that might do
648 // something causing the last reference to this object to go away.
649 Ref<ResourceLoader> protectedThis(*this);
650
651 // If we re-enter cancel() from inside willCancel(), we want to pick up from where we left
652 // off without re-running willCancel()
653 if (m_cancellationStatus == NotCancelled) {
654 m_cancellationStatus = CalledWillCancel;
655
656 willCancel(nonNullError);
657 }
658
659 // If we re-enter cancel() from inside didFailToLoad(), we want to pick up from where we
660 // left off without redoing any of this work.
661 if (m_cancellationStatus == CalledWillCancel) {
662 m_cancellationStatus = Cancelled;
663
664 if (m_handle)
665 m_handle->clearAuthentication();
666
667 if (m_documentLoader)
668 m_documentLoader->cancelPendingSubstituteLoad(this);
669 if (m_handle) {
670 m_handle->cancel();
671 m_handle = nullptr;
672 }
673 cleanupForError(nonNullError);
674 }
675
676 // If cancel() completed from within the call to willCancel() or didFailToLoad(),
677 // we don't want to redo didCancel() or releasesResources().
678 if (m_reachedTerminalState)
679 return;
680
681 didCancel(nonNullError);
682
683 if (m_cancellationStatus == FinishedCancel)
684 return;
685 m_cancellationStatus = FinishedCancel;
686
687 releaseResources();
688}

801void ResourceLoader::didReceiveAuthenticationChallenge(ResourceHandle* handle, const AuthenticationChallenge& challenge)
802{
803 ASSERT_UNUSED(handle, handle == m_handle);
804 ASSERT(m_handle->hasAuthenticationChallenge());
805
806 // Protect this in this delegate method since the additional processing can do
807 // anything including possibly derefing this; one example of this is Radar 3266216.
808 Ref<ResourceLoader> protectedThis(*this);
809
810 if (m_options.storedCredentialsPolicy == StoredCredentialsPolicy::Use) {
811 if (isAllowedToAskUserForCredentials()) {
812 frameLoader()->notifier().didReceiveAuthenticationChallenge(this, challenge);
813 return;
814 }
815 didBlockAuthenticationChallenge();
816 }
817 challenge.authenticationClient()->receivedRequestToContinueWithoutCredential(challenge);
818 ASSERT(!m_handle || !m_handle->hasAuthenticationChallenge());
819}
