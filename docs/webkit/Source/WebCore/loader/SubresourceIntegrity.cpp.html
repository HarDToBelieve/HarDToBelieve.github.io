40template<typename CharacterType>
41static bool isVCHAR(CharacterType c)
42{
43 return c >= 0x21 && c <= 0x7e;
44}
45
46template<typename CharacterType>
47struct IntegrityMetadataParser {
48public:
49 IntegrityMetadataParser(std::optional<Vector<EncodedResourceCryptographicDigest>>& digests)
50 : m_digests(digests)
51 {
52 }
53
54 bool operator()(StringParsingBuffer<CharacterType>& buffer)
55 {
56 // Initialize hashes to be something other std::nullopt, to indicate
57 // that at least one token was seen, and thus setting the empty flag
58 // from section 3.3.3 Parse metadata, to false.
59 if (!m_digests)
60 m_digests = Vector<EncodedResourceCryptographicDigest> { };
61
62 auto digest = parseEncodedCryptographicDigest(buffer);
63 if (!digest)
64 return false;
65
66 // The spec allows for options following the digest, but so far, no
67 // specific options have been specified. Thus, we just parse and ignore
68 // them. Their syntax is a '?' follow by any number of VCHARs.
69 if (skipExactly(buffer, '?'))
70 skipWhile<isVCHAR>(buffer);
71
72 // After the base64 value and options, the current character pointed to by position
73 // should either be the end or a space.
74 if (!buffer.atEnd() && !isHTMLSpace(*buffer))
75 return false;
76
77 m_digests->append(WTFMove(*digest));
78 return true;
79 }
80
81private:
82 std::optional<Vector<EncodedResourceCryptographicDigest>>& m_digests;
83};

87template <typename CharacterType, typename Functor>
88static inline void splitOnSpaces(StringParsingBuffer<CharacterType> buffer, Functor&& functor)
89{
90 skipWhile<isHTMLSpace>(buffer);
91
92 while (buffer.hasCharactersRemaining()) {
93 if (!functor(buffer))
94 skipWhile<isNotHTMLSpace>(buffer);
95 skipWhile<isHTMLSpace>(buffer);
96 }
97}

99static std::optional<Vector<EncodedResourceCryptographicDigest>> parseIntegrityMetadata(const String& integrityMetadata)
100{
101 if (integrityMetadata.isEmpty())
102 return std::nullopt;
103
104 std::optional<Vector<EncodedResourceCryptographicDigest>> result;
105
106 readCharactersForParsing(integrityMetadata, [&result] (auto buffer) {
107 using CharacterType = typename decltype(buffer)::CharacterType;
108 splitOnSpaces(buffer, IntegrityMetadataParser<CharacterType> { result });
109 });
110
111 return result;
112}

120static std::optional<EncodedResourceCryptographicDigest::Algorithm> prioritizedHashFunction(EncodedResourceCryptographicDigest::Algorithm a, EncodedResourceCryptographicDigest::Algorithm b)
121{
122 if (a == b)
123 return std::nullopt;
124 return (a > b) ? a : b;
125}

127static Vector<EncodedResourceCryptographicDigest> strongestMetadataFromSet(Vector<EncodedResourceCryptographicDigest>&& set)
128{
129 // 1. Let result be the empty set and strongest be the empty string.
130 Vector<EncodedResourceCryptographicDigest> result;
131 auto strongest = EncodedResourceCryptographicDigest::Algorithm::SHA256;
132
133 // 2. For each item in set:
134 for (auto& item : set) {
135 // 1. If result is the empty set, add item to result and set strongest to item, skip to the next item.
136 if (result.isEmpty()) {
137 strongest = item.algorithm;
138 result.append(WTFMove(item));
139 continue;
140 }
141
142 // 2. Let currentAlgorithm be the alg component of strongest.
143 auto currentAlgorithm = strongest;
144
145 // 3. Let newAlgorithm be the alg component of item.
146 auto newAlgorithm = item.algorithm;
147
148 // 4. If the result of getPrioritizedHashFunction(currentAlgorithm, newAlgorithm) is
149 // the empty string, add item to result. If the result is newAlgorithm, set strongest
150 // to item, set result to the empty set, and add item to result.
151 auto priority = prioritizedHashFunction(currentAlgorithm, newAlgorithm);
152 if (!priority)
153 result.append(WTFMove(item));
154 else if (priority.value() == newAlgorithm) {
155 strongest = item.algorithm;
156
157 result.clear();
158 result.append(WTFMove(item));
159 }
160 }
161
162 return result;
163}

165bool matchIntegrityMetadata(const CachedResource& resource, const String& integrityMetadataList)
166{
167 // FIXME: Consider caching digests on the CachedResource rather than always recomputing it.
168
169 // 1. Let parsedMetadata be the result of parsing metadataList.
170 auto parsedMetadata = parseIntegrityMetadata(integrityMetadataList);
171
172 // 2. If parsedMetadata is no metadata, return true.
173 if (!parsedMetadata)
174 return true;
175
176 // 3. If response is not eligible for integrity validation, return false.
177 if (!isResponseEligible(resource))
178 return false;
179
180 // 4. If parsedMetadata is the empty set, return true.
181 if (parsedMetadata->isEmpty())
182 return true;
183
184 // 5. Let metadata be the result of getting the strongest metadata from parsedMetadata.
185 auto metadata = strongestMetadataFromSet(WTFMove(*parsedMetadata));
186
187 const auto* sharedBuffer = resource.resourceBuffer();
188
189 // 6. For each item in metadata:
190 for (auto& item : metadata) {
191 // 1. Let algorithm be the alg component of item.
192 auto algorithm = item.algorithm;
193
194 // 2. Let expectedValue be the val component of item.
195 auto expectedValue = decodeEncodedResourceCryptographicDigest(item);
196
197 // 3. Let actualValue be the result of applying algorithm to response.
198 auto actualValue = cryptographicDigestForSharedBuffer(algorithm, sharedBuffer);
199
200 // 4. If actualValue is a case-sensitive match for expectedValue, return true.
201 if (expectedValue && actualValue.value == expectedValue->value)
202 return true;
203 }
204
205 return false;
206}

208String integrityMismatchDescription(const CachedResource& resource, const String& integrityMetadata)
209{
210 auto resourceURL = resource.url().stringCenterEllipsizedToLength();
211 if (auto resourceBuffer = resource.resourceBuffer()) {
212 return makeString(resourceURL, ". Failed integrity metadata check. Content length: ", resourceBuffer->size(), ", Expected content length: ",
213 resource.response().expectedContentLength(), ", Expected metadata: ", integrityMetadata);
214 }
215 return makeString(resourceURL, ". Failed integrity metadata check. Content length: (no content), Expected content length: ",
216 resource.response().expectedContentLength(), ", Expected metadata: ", integrityMetadata);
217}
