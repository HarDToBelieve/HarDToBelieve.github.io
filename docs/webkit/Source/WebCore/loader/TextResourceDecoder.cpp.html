40static constexpr bool bytesEqual(const char* p, char b)
41{
42 return *p == b;
43}
44
45template<typename... T>
46static constexpr bool bytesEqual(const char* p, char b, T... bs)
47{
48 return *p == b && bytesEqual(p + 1, bs...);
49}

51// You might think we should put these find functions elsewhere, perhaps with the
52// similar functions that operate on UChar, but arguably only the decoder has
53// a reason to process strings of char rather than UChar.
54
55static int find(const char* subject, size_t subjectLength, const char* target)
56{
57 size_t targetLength = strlen(target);
58 if (targetLength > subjectLength)
59 return -1;
60 for (size_t i = 0; i <= subjectLength - targetLength; ++i) {
61 bool match = true;
62 for (size_t j = 0; j < targetLength; ++j) {
63 if (subject[i + j] != target[j]) {
64 match = false;
65 break;
66 }
67 }
68 if (match)
69 return i;
70 }
71 return -1;
72}

74static TextEncoding findTextEncoding(const char* encodingName, int length)
75{
76 Vector<char, 64> buffer(length + 1);
77 memcpy(buffer.data(), encodingName, length);
78 buffer[length] = '\0';
79 return buffer.data();
80}

84 enum Type { ASCII, JIS, EUC, SJIS, UTF16, UTF8 };

90 if (code >= 0x100)
91 return 0;
92 return sjisMap[code & 0xff] & 1;

94 static int ISkana(int code)
95 {
96 if (code >= 0x100)
97 return 0;
98 return sjisMap[code & 0xff] & 2;
99 }

122 * EUC-JP is
123 * [0xa1 - 0xfe][0xa1 - 0xfe]
124 * 0x8e[0xa1 - 0xfe](SS2)
125 * 0x8f[0xa1 - 0xfe][0xa1 - 0xfe](SS3)

127 * Shift_Jis is
128 * [0x81 - 0x9f, 0xe0 - 0xef(0xfe?)][0x40 - 0x7e, 0x80 - 0xfc]

130 * Shift_Jis Hankaku Kana is
131 * [0xa1 - 0xdf]

141enum KanjiCode::Type KanjiCode::judge(const char* str, int size)
142{
143 enum Type code;
144 int i;
145 int bfr = false; /* Kana Moji */
146 int bfk = 0; /* EUC Kana */
147 int sjis = 0;
148 int euc = 0;
149
150 const unsigned char* ptr = reinterpret_cast<const unsigned char*>(str);
151
152 code = ASCII;
153
154 i = 0;
155 while (i < size) {
156 if (ptr[i] == ESC && (size - i >= 3)) {
157 if (bytesEqual(str + i + 1, '$', 'B')
158 || bytesEqual(str + i + 1, '(', 'B')
159 || bytesEqual(str + i + 1, '$', '@')
160 || bytesEqual(str + i + 1, '(', 'J')) {
161 code = JIS;
162 goto breakBreak;
163 }
164 if (bytesEqual(str + i + 1, '(', 'I') || bytesEqual(str + i + 1, ')', 'I')) {
165 code = JIS;
166 i += 3;
167 } else {
168 i++;
169 }
170 bfr = false;
171 bfk = 0;
172 } else {
173 if (ptr[i] < 0x20) {
174 bfr = false;
175 bfk = 0;
176 /* ?? check kudokuten ?? && ?? hiragana ?? */
177 if ((i >= 2) && (ptr[i - 2] == 0x81)
178 && (0x41 <= ptr[i - 1] && ptr[i - 1] <= 0x49)) {
179 code = SJIS;
180 sjis += 100; /* kudokuten */
181 } else if ((i >= 2) && (ptr[i - 2] == 0xa1)
182 && (0xa2 <= ptr[i - 1] && ptr[i - 1] <= 0xaa)) {
183 code = EUC;
184 euc += 100; /* kudokuten */
185 } else if ((i >= 2) && (ptr[i - 2] == 0x82) && (0xa0 <= ptr[i - 1])) {
186 sjis += 40; /* hiragana */
187 } else if ((i >= 2) && (ptr[i - 2] == 0xa4) && (0xa0 <= ptr[i - 1])) {
188 euc += 40; /* hiragana */
189 }
190 } else {
191 /* ?? check hiragana or katana ?? */
192 if ((size - i > 1) && (ptr[i] == 0x82) && (0xa0 <= ptr[i + 1])) {
193 sjis++; /* hiragana */
194 } else if ((size - i > 1) && (ptr[i] == 0x83)
195 && (0x40 <= ptr[i + 1] && ptr[i + 1] <= 0x9f)) {
196 sjis++; /* katakana */
197 } else if ((size - i > 1) && (ptr[i] == 0xa4) && (0xa0 <= ptr[i + 1])) {
198 euc++; /* hiragana */
199 } else if ((size - i > 1) && (ptr[i] == 0xa5) && (0xa0 <= ptr[i + 1])) {
200 euc++; /* katakana */
201 }
202 if (bfr) {
203 if ((i >= 1) && (0x40 <= ptr[i] && ptr[i] <= 0xa0) && ISkanji(ptr[i - 1])) {
204 code = SJIS;
205 goto breakBreak;
206 } else if ((i >= 1) && (0x81 <= ptr[i - 1] && ptr[i - 1] <= 0x9f) && ((0x40 <= ptr[i] && ptr[i] < 0x7e) || (0x7e < ptr[i] && ptr[i] <= 0xfc))) {
207 code = SJIS;
208 goto breakBreak;
209 } else if ((i >= 1) && (0xfd <= ptr[i] && ptr[i] <= 0xfe) && (0xa1 <= ptr[i - 1] && ptr[i - 1] <= 0xfe)) {
210 code = EUC;
211 goto breakBreak;
212 } else if ((i >= 1) && (0xfd <= ptr[i - 1] && ptr[i - 1] <= 0xfe) && (0xa1 <= ptr[i] && ptr[i] <= 0xfe)) {
213 code = EUC;
214 goto breakBreak;
215 } else if ((i >= 1) && (ptr[i] < 0xa0 || 0xdf < ptr[i]) && (0x8e == ptr[i - 1])) {
216 code = SJIS;
217 goto breakBreak;
218 } else if (ptr[i] <= 0x7f) {
219 code = SJIS;
220 goto breakBreak;
221 } else {
222 if (0xa1 <= ptr[i] && ptr[i] <= 0xa6) {
223 euc++; /* sjis hankaku kana kigo */
224 } else if (0xa1 <= ptr[i] && ptr[i] <= 0xdf) {
225 ; /* sjis hankaku kana */
226 } else if (0xa1 <= ptr[i] && ptr[i] <= 0xfe) {
227 euc++;
228 } else if (0x8e == ptr[i]) {
229 euc++;
230 } else if (0x20 <= ptr[i] && ptr[i] <= 0x7f) {
231 sjis++;
232 }
233 bfr = false;
234 bfk = 0;
235 }
236 } else if (0x8e == ptr[i]) {
237 if (size - i <= 1) {
238 ;
239 } else if (0xa1 <= ptr[i + 1] && ptr[i + 1] <= 0xdf) {
240 /* EUC KANA or SJIS KANJI */
241 if (bfk == 1) {
242 euc += 100;
243 }
244 bfk++;
245 i++;
246 } else {
247 /* SJIS only */
248 code = SJIS;
249 goto breakBreak;
250 }
251 } else if (0x81 <= ptr[i] && ptr[i] <= 0x9f) {
252 /* SJIS only */
253 code = SJIS;
254 if ((size - i >= 1)
255 && ((0x40 <= ptr[i + 1] && ptr[i + 1] <= 0x7e)
256 || (0x80 <= ptr[i + 1] && ptr[i + 1] <= 0xfc))) {
257 goto breakBreak;
258 }
259 } else if (0xfd <= ptr[i] && ptr[i] <= 0xfe) {
260 /* EUC only */
261 code = EUC;
262 if ((size - i >= 1)
263 && (0xa1 <= ptr[i + 1] && ptr[i + 1] <= 0xfe)) {
264 goto breakBreak;
265 }
266 } else if (ptr[i] <= 0x7f) {
267 ;
268 } else {
269 bfr = true;
270 bfk = 0;
271 }
272 }
273 i++;
274 }
275 }
276 if (code == ASCII) {
277 if (sjis > euc) {
278 code = SJIS;
279 } else if (sjis < euc) {
280 code = EUC;
281 }
282 }
283breakBreak:
284 return (code);
285}

287TextResourceDecoder::ContentType TextResourceDecoder::determineContentType(const String& mimeType)
288{
289 if (equalLettersIgnoringASCIICase(mimeType, "text/css"))
290 return CSS;
291 if (equalLettersIgnoringASCIICase(mimeType, "text/html"))
292 return HTML;
293 if (MIMETypeRegistry::isXMLMIMEType(mimeType))
294 return XML;
295 return PlainText;
296}

298const TextEncoding& TextResourceDecoder::defaultEncoding(ContentType contentType, const TextEncoding& specifiedDefaultEncoding)
299{
300 // Despite 8.5 "Text/xml with Omitted Charset" of RFC 3023, we assume UTF-8 instead of US-ASCII
301 // for text/xml. This matches Firefox.
302 if (contentType == XML)
303 return UTF8Encoding();
304 if (!specifiedDefaultEncoding.isValid())
305 return Latin1Encoding();
306 return specifiedDefaultEncoding;
307}

323static inline bool shouldPrependBOM(const unsigned char* data, unsigned length)
324{
325 if (length < 3)
326 return true;
327 return data[0] != 0xef || data[1] != 0xbb || data[2] != 0xbf;

330// https://encoding.spec.whatwg.org/#utf-8-decode
331String TextResourceDecoder::textFromUTF8(const unsigned char* data, unsigned length)
332{
333 auto decoder = TextResourceDecoder::create("text/plain", "UTF-8");
334 if (shouldPrependBOM(data, length))
335 decoder->decode("\xef\xbb\xbf", 3);
336 return decoder->decodeAndFlush(data, length);
337}

339void TextResourceDecoder::setEncoding(const TextEncoding& encoding, EncodingSource source)
340{
341 // In case the encoding didn't exist, we keep the old one (helps some sites specifying invalid encodings).
342 if (!encoding.isValid())
343 return;
344
345 // When encoding comes from meta tag (i.e. it cannot be XML files sent via XHR),
346 // treat x-user-defined as windows-1252 (bug 18270)
347 if (source == EncodingFromMetaTag && equalLettersIgnoringASCIICase(encoding.name(), "x-user-defined"))
348 m_encoding = "windows-1252";
349 else if (source == EncodingFromMetaTag || source == EncodingFromXMLHeader || source == EncodingFromCSSCharset)
350 m_encoding = encoding.closestByteBasedEquivalent();
351 else
352 m_encoding = encoding;
353
354 m_codec = nullptr;
355 m_source = source;
356}

363// Returns the position of the encoding string.
364static int findXMLEncoding(const char* str, int len, int& encodingLength)
365{
366 int pos = find(str, len, "encoding");
367 if (pos == -1)
368 return -1;
369 pos += 8;
370
371 // Skip spaces and stray control characters.
372 while (pos < len && str[pos] <= ' ')
373 ++pos;
374
375 // Skip equals sign.
376 if (pos >= len || str[pos] != '=')
377 return -1;
378 ++pos;
379
380 // Skip spaces and stray control characters.
381 while (pos < len && str[pos] <= ' ')
382 ++pos;
383
384 // Skip quotation mark.
385 if (pos >= len)
386 return - 1;
387 char quoteMark = str[pos];
388 if (quoteMark != '"' && quoteMark != '\'')
389 return -1;
390 ++pos;
391
392 // Find the trailing quotation mark.
393 int end = pos;
394 while (end < len && str[end] != quoteMark)
395 ++end;
396 if (end >= len)
397 return -1;
398
399 encodingLength = end - pos;
400 return pos;
401}

403size_t TextResourceDecoder::checkForBOM(const char* data, size_t len)
404{
405 // Check for UTF-16 or UTF-8 BOM mark at the beginning, which is a sure sign of a Unicode encoding.
406 // We let it override even a user-chosen encoding.
407 const size_t maximumBOMLength = 3;
408
409 ASSERT(!m_checkedForBOM);
410
411 size_t lengthOfBOM = 0;
412
413 size_t bufferLength = m_buffer.size();
414
415 size_t buf1Len = bufferLength;
416 size_t buf2Len = len;
417 const unsigned char* buf1 = reinterpret_cast<const unsigned char*>(m_buffer.data());
418 const unsigned char* buf2 = reinterpret_cast<const unsigned char*>(data);
419 unsigned char c1 = buf1Len ? (static_cast<void>(--buf1Len), *buf1++) : buf2Len ? (static_cast<void>(--buf2Len), *buf2++) : 0;
420 unsigned char c2 = buf1Len ? (static_cast<void>(--buf1Len), *buf1++) : buf2Len ? (static_cast<void>(--buf2Len), *buf2++) : 0;
421 unsigned char c3 = buf1Len ? (static_cast<void>(--buf1Len), *buf1++) : buf2Len ? (static_cast<void>(--buf2Len), *buf2++) : 0;
422
423 // Check for the BOM.
424 if (c1 == 0xFF && c2 == 0xFE) {
425 ASSERT(UTF16LittleEndianEncoding().isValid());
426 setEncoding(UTF16LittleEndianEncoding(), AutoDetectedEncoding);
427 lengthOfBOM = 2;
428 } else if (c1 == 0xFE && c2 == 0xFF) {
429 ASSERT(UTF16BigEndianEncoding().isValid());
430 setEncoding(UTF16BigEndianEncoding(), AutoDetectedEncoding);
431 lengthOfBOM = 2;
432 } else if (c1 == 0xEF && c2 == 0xBB && c3 == 0xBF) {
433 ASSERT(UTF8Encoding().isValid());
434 setEncoding(UTF8Encoding(), AutoDetectedEncoding);
435 lengthOfBOM = 3;
436 }
437
438 if (lengthOfBOM || bufferLength + len >= maximumBOMLength)
439 m_checkedForBOM = true;
440
441 ASSERT(lengthOfBOM <= maximumBOMLength);
442 return lengthOfBOM;
443}

445bool TextResourceDecoder::checkForCSSCharset(const char* data, size_t len, bool& movedDataToBuffer)
446{
447 if (m_source != DefaultEncoding && m_source != EncodingFromParentFrame) {
448 m_checkedForCSSCharset = true;
449 return true;
450 }
451
452 size_t oldSize = m_buffer.size();
453 m_buffer.grow(oldSize + len);
454 memcpy(m_buffer.data() + oldSize, data, len);
455
456 movedDataToBuffer = true;
457
458 if (m_buffer.size() <= 13) // strlen('@charset "x";') == 13
459 return false;
460
461 const char* dataStart = m_buffer.data();
462 const char* dataEnd = dataStart + m_buffer.size();
463
464 if (bytesEqual(dataStart, '@', 'c', 'h', 'a', 'r', 's', 'e', 't', ' ', '"')) {
465 dataStart += 10;
466 const char* pos = dataStart;
467
468 while (pos < dataEnd && *pos != '"')
469 ++pos;
470 if (pos == dataEnd)
471 return false;
472
473 int encodingNameLength = pos - dataStart;
474
475 ++pos;
476 if (pos == dataEnd)
477 return false;
478
479 if (*pos == ';')
480 setEncoding(findTextEncoding(dataStart, encodingNameLength), EncodingFromCSSCharset);
481 }
482
483 m_checkedForCSSCharset = true;
484 return true;
485}

487bool TextResourceDecoder::checkForHeadCharset(const char* data, size_t len, bool& movedDataToBuffer)
488{
489 if (m_source != DefaultEncoding && m_source != EncodingFromParentFrame) {
490 m_checkedForHeadCharset = true;
491 return true;
492 }
493
494 // This is not completely efficient, since the function might go
495 // through the HTML head several times.
496
497 size_t oldSize = m_buffer.size();
498 m_buffer.grow(oldSize + len);
499 memcpy(m_buffer.data() + oldSize, data, len);
500
501 movedDataToBuffer = true;
502
503 // Continue with checking for an HTML meta tag if we were already doing so.
504 if (m_charsetParser)
505 return checkForMetaCharset(data, len);
506
507 const char* ptr = m_buffer.data();
508 const char* pEnd = ptr + m_buffer.size();
509
510 // Is there enough data available to check for XML declaration?
511 if (m_buffer.size() < 8)
512 return false;
513
514 // Handle XML declaration, which can have encoding in it. This encoding is honored even for HTML documents.
515 // It is an error for an XML declaration not to be at the start of an XML document, and it is ignored in HTML documents in such case.
516 if (bytesEqual(ptr, '<', '?', 'x', 'm', 'l')) {
517 const char* xmlDeclarationEnd = ptr;
518 while (xmlDeclarationEnd != pEnd && *xmlDeclarationEnd != '>')
519 ++xmlDeclarationEnd;
520 if (xmlDeclarationEnd == pEnd)
521 return false;
522 // No need for +1, because we have an extra "?" to lose at the end of XML declaration.
523 int len = 0;
524 int pos = findXMLEncoding(ptr, xmlDeclarationEnd - ptr, len);
525 if (pos != -1)
526 setEncoding(findTextEncoding(ptr + pos, len), EncodingFromXMLHeader);
527 // continue looking for a charset - it may be specified in an HTTP-Equiv meta
528 } else if (bytesEqual(ptr, '<', 0, '?', 0, 'x', 0)) {
529 setEncoding(UTF16LittleEndianEncoding(), AutoDetectedEncoding);
530 return true;
531 } else if (bytesEqual(ptr, 0, '<', 0, '?', 0, 'x')) {
532 setEncoding(UTF16BigEndianEncoding(), AutoDetectedEncoding);
533 return true;
534 }
535
536 // The HTTP-EQUIV meta has no effect on XHTML.
537 if (m_contentType == XML)
538 return true;
539
540 m_charsetParser = makeUnique<HTMLMetaCharsetParser>();
541 return checkForMetaCharset(data, len);
542}
543
544bool TextResourceDecoder::checkForMetaCharset(const char* data, size_t length)
545{
546 if (!m_charsetParser->checkForMetaCharset(data, length))
547 return false;
548
549 setEncoding(m_charsetParser->encoding(), EncodingFromMetaTag);
550 m_charsetParser = nullptr;
551 m_checkedForHeadCharset = true;
552 return true;
553}

555void TextResourceDecoder::detectJapaneseEncoding(const char* data, size_t len)
556{
557 switch (KanjiCode::judge(data, len)) {
558 case KanjiCode::JIS:
559 setEncoding("ISO-2022-JP", AutoDetectedEncoding);
560 break;
561 case KanjiCode::EUC:
562 setEncoding("EUC-JP", AutoDetectedEncoding);
563 break;
564 case KanjiCode::SJIS:
565 setEncoding("Shift_JIS", AutoDetectedEncoding);
566 break;
567 case KanjiCode::ASCII:
568 case KanjiCode::UTF16:
569 case KanjiCode::UTF8:
570 break;
571 }
572}

574// We use the encoding detector in two cases:
575// 1. Encoding detector is turned ON and no other encoding source is
576// available (that is, it's DefaultEncoding).
577// 2. Encoding detector is turned ON and the encoding is set to
578// the encoding of the parent frame, which is also auto-detected.
579// Note that condition #2 is NOT satisfied unless parent-child frame
580// relationship is compliant to the same-origin policy. If they're from
581// different domains, |m_source| would not be set to EncodingFromParentFrame
582// in the first place.
583bool TextResourceDecoder::shouldAutoDetect() const
584{
585 return m_usesEncodingDetector
586 && (m_source == DefaultEncoding || (m_source == EncodingFromParentFrame && m_parentFrameAutoDetectedEncoding));
587}
588
589String TextResourceDecoder::decode(const char* data, size_t length)
590{
591 size_t lengthOfBOM = 0;
592 if (!m_checkedForBOM)
593 lengthOfBOM = checkForBOM(data, length);
594
595 bool movedDataToBuffer = false;
596
597 if (m_contentType == CSS && !m_checkedForCSSCharset)
598 if (!checkForCSSCharset(data, length, movedDataToBuffer))
599 return emptyString();
600
601 if ((m_contentType == HTML || m_contentType == XML) && !m_checkedForHeadCharset) // HTML and XML
602 if (!checkForHeadCharset(data, length, movedDataToBuffer))
603 return emptyString();
604
605 // FIXME: It is wrong to change the encoding downstream after we have already done some decoding.
606 if (shouldAutoDetect()) {
607 if (m_encoding.isJapanese())
608 detectJapaneseEncoding(data, length); // FIXME: We should use detectTextEncoding() for all languages.
609 else {
610 TextEncoding detectedEncoding;
611 if (detectTextEncoding(data, length, m_parentFrameAutoDetectedEncoding, &detectedEncoding))
612 setEncoding(detectedEncoding, AutoDetectedEncoding);
613 }
614 }
615
616 ASSERT(m_encoding.isValid());
617
618 if (!m_codec)
619 m_codec = newTextCodec(m_encoding);
620
621 if (m_buffer.isEmpty())
622 return m_codec->decode(data + lengthOfBOM, length - lengthOfBOM, false, m_contentType == XML, m_sawError);
623
624 if (!movedDataToBuffer) {
625 size_t oldSize = m_buffer.size();
626 m_buffer.grow(oldSize + length);
627 memcpy(m_buffer.data() + oldSize, data, length);
628 }
629
630 String result = m_codec->decode(m_buffer.data() + lengthOfBOM, m_buffer.size() - lengthOfBOM, false, m_contentType == XML && !m_useLenientXMLDecoding, m_sawError);
631 m_buffer.clear();
632 return result;
633}

635String TextResourceDecoder::flush()
636{
637 // If we can not identify the encoding even after a document is completely
638 // loaded, we need to detect the encoding if other conditions for
639 // autodetection is satisfied.
640 if (m_buffer.size() && shouldAutoDetect()
641 && ((!m_checkedForHeadCharset && (m_contentType == HTML || m_contentType == XML)) || (!m_checkedForCSSCharset && (m_contentType == CSS)))) {
642 TextEncoding detectedEncoding;
643 if (detectTextEncoding(m_buffer.data(), m_buffer.size(), m_parentFrameAutoDetectedEncoding, &detectedEncoding))
644 setEncoding(detectedEncoding, AutoDetectedEncoding);
645 }
646
647 if (!m_codec)
648 m_codec = newTextCodec(m_encoding);
649
650 String result = m_codec->decode(m_buffer.data(), m_buffer.size(), true, m_contentType == XML && !m_useLenientXMLDecoding, m_sawError);
651 m_buffer.clear();
652 m_codec = nullptr;
653 m_checkedForBOM = false; // Skip BOM again when re-decoding.
654 return result;
655}

657String TextResourceDecoder::decodeAndFlush(const char* data, size_t length)
658{
659 String decoded = decode(data, length);
660 return decoded + flush();
661}

663const TextEncoding* TextResourceDecoder::encodingForURLParsing()
664{
665 // For UTF-{7,16,32}, we want to use UTF-8 for the query part as
666 // we do when submitting a form. A form with GET method
667 // has its contents added to a URL as query params and it makes sense
668 // to be consistent.
669 auto& encoding = m_encoding.encodingForFormSubmissionOrURLParsing();
670 if (encoding == UTF8Encoding())
671 return nullptr;
672 return &encoding;
673}
