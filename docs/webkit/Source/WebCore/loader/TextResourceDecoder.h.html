23#pragma once
24
25#include "TextEncoding.h"
26#include <wtf/RefCounted.h>
27
28namespace WebCore {
29
30class HTMLMetaCharsetParser;
31class TextCodec;
32
33class TextResourceDecoder : public RefCounted<TextResourceDecoder> {
34public:
35 enum EncodingSource {
36 DefaultEncoding,
37 AutoDetectedEncoding,
38 EncodingFromXMLHeader,
39 EncodingFromMetaTag,
40 EncodingFromCSSCharset,
41 EncodingFromHTTPHeader,
42 UserChosenEncoding,
43 EncodingFromParentFrame
44 };
45
46 WEBCORE_EXPORT static Ref<TextResourceDecoder> create(const String& mimeType, const TextEncoding& defaultEncoding = { }, bool usesEncodingDetector = false);
47 WEBCORE_EXPORT ~TextResourceDecoder();
48
49 static String textFromUTF8(const unsigned char* data, unsigned length);
50
51 void setEncoding(const TextEncoding&, EncodingSource);
52 const TextEncoding& encoding() const { return m_encoding; }
53 const TextEncoding* encodingForURLParsing();
54
55 bool hasEqualEncodingForCharset(const String& charset) const;
56
57 WEBCORE_EXPORT String decode(const char* data, size_t length);
58 String decode(const uint8_t* data, size_t length) { return decode(reinterpret_cast<const char*>(data), length); }
59 WEBCORE_EXPORT String flush();
60
61 WEBCORE_EXPORT String decodeAndFlush(const char* data, size_t length);
62 String decodeAndFlush(const uint8_t* data, size_t length) { return decodeAndFlush(reinterpret_cast<const char*>(data), length); }
63
64 void setHintEncoding(const TextResourceDecoder* parentFrameDecoder);
65
66 void useLenientXMLDecoding() { m_useLenientXMLDecoding = true; }
67 bool sawError() const { return m_sawError; }
68
69private:
70 TextResourceDecoder(const String& mimeType, const TextEncoding& defaultEncoding, bool usesEncodingDetector);
71
72 enum ContentType { PlainText, HTML, XML, CSS }; // PlainText only checks for BOM.
73 static ContentType determineContentType(const String& mimeType);
74 static const TextEncoding& defaultEncoding(ContentType, const TextEncoding& defaultEncoding);
75
76 size_t checkForBOM(const char*, size_t);
77 bool checkForCSSCharset(const char*, size_t, bool& movedDataToBuffer);
78 bool checkForHeadCharset(const char*, size_t, bool& movedDataToBuffer);
79 bool checkForMetaCharset(const char*, size_t);
80 void detectJapaneseEncoding(const char*, size_t);
81 bool shouldAutoDetect() const;
82
83 ContentType m_contentType;
84 TextEncoding m_encoding;
85 std::unique_ptr<TextCodec> m_codec;
86 std::unique_ptr<HTMLMetaCharsetParser> m_charsetParser;
87 EncodingSource m_source { DefaultEncoding };
88 const char* m_parentFrameAutoDetectedEncoding { nullptr };
89 Vector<char> m_buffer;
90 bool m_checkedForBOM { false };
91 bool m_checkedForCSSCharset { false };
92 bool m_checkedForHeadCharset { false };
93 bool m_useLenientXMLDecoding { false }; // Don't stop on XML decoding errors.
94 bool m_sawError { false };
95 bool m_usesEncodingDetector { false };
96};
97
98inline void TextResourceDecoder::setHintEncoding(const TextResourceDecoder* parentFrameDecoder)
99{
100 if (parentFrameDecoder && parentFrameDecoder->m_source == AutoDetectedEncoding)
101 m_parentFrameAutoDetectedEncoding = parentFrameDecoder->encoding().name();
102}
103
104} // namespace WebCore
