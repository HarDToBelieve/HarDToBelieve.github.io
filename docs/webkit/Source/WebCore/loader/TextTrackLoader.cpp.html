62void TextTrackLoader::cueLoadTimerFired()
63{
64 if (m_newCuesAvailable) {
65 m_newCuesAvailable = false;
66 m_client.newCuesAvailable(*this);
67 }
68
69 if (m_state >= Finished)
70 m_client.cueLoadingCompleted(*this, m_state == Failed);
71}

73void TextTrackLoader::cancelLoad()
74{
75 if (m_resource) {
76 m_resource->removeClient(*this);
77 m_resource = nullptr;
78 }

81void TextTrackLoader::processNewCueData(CachedResource& resource)
82{
83 ASSERT_UNUSED(resource, m_resource == &resource);
84
85 if (m_state == Failed || !m_resource->resourceBuffer())
86 return;
87
88 auto* buffer = m_resource->resourceBuffer();
89 if (m_parseOffset == buffer->size())
90 return;
91
92 if (!m_cueParser)
93 m_cueParser = makeUnique<WebVTTParser>(static_cast<WebVTTParserClient&>(*this), m_document);
94
95 while (m_parseOffset < buffer->size()) {
96 auto data = buffer->getSomeData(m_parseOffset);
97 m_cueParser->parseBytes(data.data(), data.size());
98 m_parseOffset += data.size();
99 }
100}

113void TextTrackLoader::corsPolicyPreventedLoad()
114{
115 static NeverDestroyed<String> consoleMessage(MAKE_STATIC_STRING_IMPL("Cross-origin text track load denied by Cross-Origin Resource Sharing policy."));
116 m_document.addConsoleMessage(MessageSource::Security, MessageLevel::Error, consoleMessage);
117 m_state = Failed;
118}

120void TextTrackLoader::notifyFinished(CachedResource& resource, const NetworkLoadMetrics&)
121{
122 ASSERT_UNUSED(resource, m_resource == &resource);
123
124 if (m_resource->resourceError().isAccessControl())
125 corsPolicyPreventedLoad();
126
127 if (m_state != Failed) {
128 processNewCueData(*m_resource);
129 if (m_cueParser)
130 m_cueParser->fileFinished();
131 if (m_state != Failed)
132 m_state = m_resource->errorOccurred() ? Failed : Finished;
133 }
134
135 if (m_state == Finished && m_cueParser)
136 m_cueParser->flush();
137
138 if (!m_cueLoadTimer.isActive())
139 m_cueLoadTimer.startOneShot(0_s);
140
141 cancelLoad();
142}

144bool TextTrackLoader::load(const URL& url, HTMLTrackElement& element)
145{
146 cancelLoad();
147
148 ResourceLoaderOptions options = CachedResourceLoader::defaultCachedResourceOptions();
149 options.contentSecurityPolicyImposition = element.isInUserAgentShadowTree() ? ContentSecurityPolicyImposition::SkipPolicyCheck : ContentSecurityPolicyImposition::DoPolicyCheck;
150
151 // FIXME: Do we really need to call completeURL here?
152 ResourceRequest resourceRequest(m_document.completeURL(url.string()));
153
154 if (auto mediaElement = element.mediaElement())
155 resourceRequest.setInspectorInitiatorNodeIdentifier(InspectorInstrumentation::identifierForNode(*mediaElement));
156
157 auto cueRequest = createPotentialAccessControlRequest(WTFMove(resourceRequest), WTFMove(options), m_document, element.mediaElementCrossOriginAttribute());
158 m_resource = m_document.cachedResourceLoader().requestTextTrack(WTFMove(cueRequest)).value_or(nullptr);
159 if (!m_resource)
160 return false;
161 m_resource->addClient(*this);
162 return true;
163}

165void TextTrackLoader::newCuesParsed()
166{
167 if (m_cueLoadTimer.isActive())
168 return;
169
170 m_newCuesAvailable = true;
171 m_cueLoadTimer.startOneShot(0_s);

184void TextTrackLoader::fileFailedToParse()
185{
186 LOG(Media, "TextTrackLoader::fileFailedToParse");
187
188 m_state = Failed;

190 if (!m_cueLoadTimer.isActive())
191 m_cueLoadTimer.startOneShot(0_s);

193 cancelLoad();

196Vector<Ref<VTTCue>> TextTrackLoader::getNewCues()
197{
198 ASSERT(m_cueParser);
199 if (!m_cueParser)
200 return { };
201
202 auto cues = m_cueParser->takeCues();
203 Vector<Ref<VTTCue>> result;
204 result.reserveInitialCapacity(cues.size());
205 for (auto& cueData : cues)
206 result.uncheckedAppend(VTTCue::create(m_document, cueData));
207 return result;
208}
