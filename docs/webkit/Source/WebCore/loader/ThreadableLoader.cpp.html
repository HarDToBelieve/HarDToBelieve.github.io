66ThreadableLoaderOptions ThreadableLoaderOptions::isolatedCopy() const
67{
68 ThreadableLoaderOptions copy;
69
70 // FetchOptions
71 copy.destination = this->destination;
72 copy.mode = this->mode;
73 copy.credentials = this->credentials;
74 copy.cache = this->cache;
75 copy.redirect = this->redirect;
76 copy.referrerPolicy = this->referrerPolicy;
77 copy.integrity = this->integrity.isolatedCopy();
78
79 // ResourceLoaderOptions
80 copy.sendLoadCallbacks = this->sendLoadCallbacks;
81 copy.sniffContent = this->sniffContent;
82 copy.dataBufferingPolicy = this->dataBufferingPolicy;
83 copy.storedCredentialsPolicy = this->storedCredentialsPolicy;
84 copy.securityCheck = this->securityCheck;
85 copy.certificateInfoPolicy = this->certificateInfoPolicy;
86 copy.contentSecurityPolicyImposition = this->contentSecurityPolicyImposition;
87 copy.defersLoadingPolicy = this->defersLoadingPolicy;
88 copy.cachingPolicy = this->cachingPolicy;
89 copy.sameOriginDataURLFlag = this->sameOriginDataURLFlag;
90 copy.initiatorContext = this->initiatorContext;
91 copy.clientCredentialPolicy = this->clientCredentialPolicy;
92 copy.maxRedirectCount = this->maxRedirectCount;
93 copy.preflightPolicy = this->preflightPolicy;
94
95 // ThreadableLoaderOptions
96 copy.contentSecurityPolicyEnforcement = this->contentSecurityPolicyEnforcement;
97 copy.initiator = this->initiator.isolatedCopy();
98 copy.filteringPolicy = this->filteringPolicy;
99
100 return copy;
101}

104RefPtr<ThreadableLoader> ThreadableLoader::create(ScriptExecutionContext& context, ThreadableLoaderClient& client, ResourceRequest&& request, const ThreadableLoaderOptions& options, String&& referrer, String&& taskMode)
105{
106 if (is<WorkerGlobalScope>(context) || (is<WorkletGlobalScope>(context) && downcast<WorkletGlobalScope>(context).workerOrWorkletThread()))
107 return WorkerThreadableLoader::create(static_cast<WorkerOrWorkletGlobalScope&>(context), client, WTFMove(taskMode), WTFMove(request), options, WTFMove(referrer));
108
109 Document* document = nullptr;
110 if (is<WorkletGlobalScope>(context))
111 document = downcast<WorkletGlobalScope>(context).responsibleDocument();
112 else
113 document = &downcast<Document>(context);
114
115 return DocumentThreadableLoader::create(*document, client, WTFMove(request), options, WTFMove(referrer));
116}

118void ThreadableLoader::loadResourceSynchronously(ScriptExecutionContext& context, ResourceRequest&& request, ThreadableLoaderClient& client, const ThreadableLoaderOptions& options)
119{
120 auto resourceURL = request.url();
121 if (is<WorkerGlobalScope>(context))
122 WorkerThreadableLoader::loadResourceSynchronously(downcast<WorkerGlobalScope>(context), WTFMove(request), client, options);
123 else
124 DocumentThreadableLoader::loadResourceSynchronously(downcast<Document>(context), WTFMove(request), client, options);
125 context.didLoadResourceSynchronously(resourceURL);
126}

128void ThreadableLoader::logError(ScriptExecutionContext& context, const ResourceError& error, const String& initiator)
129{
130 if (error.isCancellation())
131 return;
132
133 // FIXME: Some errors are returned with null URLs. This leads to poor console messages. We should do better for these errors.
134 if (error.failingURL().isNull())
135 return;
136
137 // We further reduce logging to some errors.
138 // FIXME: Log more errors when making so do not make some layout tests flaky.
139 if (error.domain() != errorDomainWebKitInternal && error.domain() != errorDomainWebKitServiceWorker && !error.isAccessControl())
140 return;
141
142 const char* messageStart;
143 if (initiator == cachedResourceRequestInitiators().eventsource)
144 messageStart = "EventSource cannot load ";
145 else if (initiator == cachedResourceRequestInitiators().fetch)
146 messageStart = "Fetch API cannot load ";
147 else if (initiator == cachedResourceRequestInitiators().xmlhttprequest)
148 messageStart = "XMLHttpRequest cannot load ";
149 else
150 messageStart = "Cannot load ";
151
152 String messageEnd = error.isAccessControl() ? " due to access control checks."_s : "."_s;
153 context.addConsoleMessage(MessageSource::JS, MessageLevel::Error, makeString(messageStart, error.failingURL().string(), messageEnd));
154}
