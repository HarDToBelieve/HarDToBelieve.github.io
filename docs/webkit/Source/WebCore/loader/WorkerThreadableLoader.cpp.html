68void WorkerThreadableLoader::loadResourceSynchronously(WorkerOrWorkletGlobalScope& workerOrWorkletGlobalScope, ResourceRequest&& request, ThreadableLoaderClient& client, const ThreadableLoaderOptions& options)
69{
70 WorkerRunLoop& runLoop = workerOrWorkletGlobalScope.workerOrWorkletThread()->runLoop();
71
72 // Create a unique mode just for this synchronous resource load.
73 auto mode = makeString("loadResourceSynchronouslyMode", runLoop.createUniqueId());
74
75 auto loader = WorkerThreadableLoader::create(workerOrWorkletGlobalScope, client, mode, WTFMove(request), options, String());
76 MessageQueueWaitResult result = MessageQueueMessageReceived;
77 while (!loader->done() && result != MessageQueueTerminated)
78 result = runLoop.runInMode(&workerOrWorkletGlobalScope, mode);
79
80 if (!loader->done() && result == MessageQueueTerminated)
81 loader->cancel();
82}

110WorkerThreadableLoader::MainThreadBridge::MainThreadBridge(ThreadableLoaderClientWrapper& workerClientWrapper, WorkerLoaderProxy& loaderProxy, const String& taskMode,
111 ResourceRequest&& request, const ThreadableLoaderOptions& options, const String& outgoingReferrer, WorkerOrWorkletGlobalScope& globalScope)
112 : m_workerClientWrapper(&workerClientWrapper)
113 , m_loaderProxy(loaderProxy)
114 , m_taskMode(taskMode.isolatedCopy())
115 , m_workerRequestIdentifier(globalScope.createUniqueIdentifier())
116{
117 auto* securityOrigin = globalScope.securityOrigin();
118 auto* contentSecurityPolicy = globalScope.contentSecurityPolicy();
119
120 ASSERT(securityOrigin);
121 ASSERT(contentSecurityPolicy);
122
123 auto securityOriginCopy = securityOrigin->isolatedCopy();
124 auto contentSecurityPolicyCopy = makeUnique<ContentSecurityPolicy>(globalScope.url().isolatedCopy());
125 contentSecurityPolicyCopy->copyStateFrom(contentSecurityPolicy);
126 contentSecurityPolicyCopy->copyUpgradeInsecureRequestStateFrom(*contentSecurityPolicy);
127
128 auto optionsCopy = makeUnique<LoaderTaskOptions>(options, request.httpReferrer().isNull() ? outgoingReferrer : request.httpReferrer(), WTFMove(securityOriginCopy));
129
130 // All loads start out as Document. Inside WorkerThreadableLoader we upgrade this to a Worker load.
131 ASSERT(optionsCopy->options.initiatorContext == InitiatorContext::Document);
132 optionsCopy->options.initiatorContext = InitiatorContext::Worker;
133
134#if ENABLE(SERVICE_WORKER)
135 optionsCopy->options.serviceWorkersMode = is<ServiceWorkerGlobalScope>(globalScope) ? ServiceWorkersMode::None : ServiceWorkersMode::All;
136 if (auto* activeServiceWorker = globalScope.activeServiceWorker())
137 optionsCopy->options.serviceWorkerRegistrationIdentifier = activeServiceWorker->registrationIdentifier();
138#endif
139
140 if (is<WorkerGlobalScope>(globalScope))
141 InspectorInstrumentation::willSendRequest(downcast<WorkerGlobalScope>(globalScope), m_workerRequestIdentifier, request);
142
143 // Can we benefit from request being an r-value to create more efficiently its isolated copy?
144 m_loaderProxy.postTaskToLoader([this, request = request.isolatedCopy(), options = WTFMove(optionsCopy), contentSecurityPolicyCopy = WTFMove(contentSecurityPolicyCopy)](ScriptExecutionContext& context) mutable {
145 ASSERT(isMainThread());
146 Document& document = downcast<Document>(context);
147
148 // FIXME: If the site requests a local resource, then this will return a non-zero value but the sync path will return a 0 value.
149 // Either this should return 0 or the other code path should call a failure callback.
150 m_mainThreadLoader = DocumentThreadableLoader::create(document, *this, WTFMove(request), options->options, WTFMove(options->origin), WTFMove(contentSecurityPolicyCopy), WTFMove(options->referrer), DocumentThreadableLoader::ShouldLogError::No);
151 ASSERT(m_mainThreadLoader || m_loadingFinished);
152 });
153}

155void WorkerThreadableLoader::MainThreadBridge::destroy()
156{
157 // Ensure that no more client callbacks are done in the worker context's thread.
158 clearClientWrapper();
159
160 // "delete this" and m_mainThreadLoader::deref() on the worker object's thread.
161 m_loaderProxy.postTaskToLoader([self = std::unique_ptr<WorkerThreadableLoader::MainThreadBridge>(this)] (ScriptExecutionContext& context) {
162 ASSERT(isMainThread());
163 ASSERT_UNUSED(context, context.isDocument());
164 if (self->m_mainThreadLoader)
165 self->m_mainThreadLoader->clearClient();
166 });
167}

169void WorkerThreadableLoader::MainThreadBridge::cancel()
170{
171 m_loaderProxy.postTaskToLoader([this] (ScriptExecutionContext& context) {
172 ASSERT(isMainThread());
173 ASSERT_UNUSED(context, context.isDocument());
174
175 if (!m_mainThreadLoader)
176 return;
177 m_mainThreadLoader->cancel();
178 m_mainThreadLoader = nullptr;
179 });
180
181 // Taking a ref of client wrapper as call to didFail may take out the last reference of it.
182 Ref<ThreadableLoaderClientWrapper> protectedWorkerClientWrapper(*m_workerClientWrapper);
183 // If the client hasn't reached a termination state, then transition it by sending a cancellation error.
184 // Note: no more client callbacks will be done after this method -- we clear the client wrapper to ensure that.
185 ResourceError error(ResourceError::Type::Cancellation);
186 protectedWorkerClientWrapper->didFail(error);
187 protectedWorkerClientWrapper->clearClient();
188}

190void WorkerThreadableLoader::MainThreadBridge::computeIsDone()
191{
192 m_loaderProxy.postTaskToLoader([this](auto&) {
193 if (!m_mainThreadLoader) {
194 notifyIsDone(true);
195 return;
196 }
197 m_mainThreadLoader->computeIsDone();
198 });
199}

201void WorkerThreadableLoader::MainThreadBridge::notifyIsDone(bool isDone)
202{
203 m_loaderProxy.postTaskForModeToWorkerOrWorkletGlobalScope([protectedWorkerClientWrapper = makeRef(*m_workerClientWrapper), isDone](auto&) {
204 protectedWorkerClientWrapper->notifyIsDone(isDone);
205 }, m_taskMode);

221void WorkerThreadableLoader::MainThreadBridge::didReceiveResponse(unsigned long identifier, const ResourceResponse& response)
222{
223 m_loaderProxy.postTaskForModeToWorkerOrWorkletGlobalScope([protectedWorkerClientWrapper = makeRef(*m_workerClientWrapper), workerRequestIdentifier = m_workerRequestIdentifier, identifier, responseData = response.crossThreadData()] (ScriptExecutionContext& context) mutable {
224 ASSERT(context.isWorkerGlobalScope() || context.isWorkletGlobalScope());
225 auto response = ResourceResponse::fromCrossThreadData(WTFMove(responseData));
226 protectedWorkerClientWrapper->didReceiveResponse(identifier, response);
227 if (is<WorkerGlobalScope>(context))
228 InspectorInstrumentation::didReceiveResourceResponse(downcast<WorkerGlobalScope>(context), workerRequestIdentifier, response);
229 }, m_taskMode);

243void WorkerThreadableLoader::MainThreadBridge::didFinishLoading(unsigned long identifier)
244{
245 m_loadingFinished = true;
246 m_loaderProxy.postTaskForModeToWorkerOrWorkletGlobalScope([protectedWorkerClientWrapper = makeRef(*m_workerClientWrapper), workerRequestIdentifier = m_workerRequestIdentifier, networkLoadMetrics = m_networkLoadMetrics.isolatedCopy(), identifier] (ScriptExecutionContext& context) mutable {
247 ASSERT(context.isWorkerGlobalScope() || context.isWorkletGlobalScope());
248 protectedWorkerClientWrapper->didFinishLoading(identifier);

251 }, m_taskMode);

254void WorkerThreadableLoader::MainThreadBridge::didFail(const ResourceError& error)
255{
256 m_loadingFinished = true;
257 m_loaderProxy.postTaskForModeToWorkerOrWorkletGlobalScope([protectedWorkerClientWrapper = makeRef(*m_workerClientWrapper), workerRequestIdentifier = m_workerRequestIdentifier, error = error.isolatedCopy()] (ScriptExecutionContext& context) mutable {
258 ASSERT(context.isWorkerGlobalScope() || context.isWorkletGlobalScope());
259 ThreadableLoader::logError(context, error, protectedWorkerClientWrapper->initiator());
260 protectedWorkerClientWrapper->didFail(error);
261 if (is<WorkerGlobalScope>(context))
262 InspectorInstrumentation::didFailLoading(downcast<WorkerGlobalScope>(context), workerRequestIdentifier, error);
263 }, m_taskMode);
264}

266void WorkerThreadableLoader::MainThreadBridge::didFinishTiming(const ResourceTiming& resourceTiming)
267{
268 m_networkLoadMetrics = resourceTiming.networkLoadMetrics();
269 m_loaderProxy.postTaskForModeToWorkerOrWorkletGlobalScope([protectedWorkerClientWrapper = makeRef(*m_workerClientWrapper), resourceTiming = resourceTiming.isolatedCopy()] (ScriptExecutionContext& context) mutable {
270 ASSERT(context.isWorkerGlobalScope() || context.isWorkletGlobalScope());
271 ASSERT(!resourceTiming.initiator().isEmpty());
272
273 // No need to notify clients, just add the performance timing entry.
274 if (is<WorkerGlobalScope>(context))
275 downcast<WorkerGlobalScope>(context).performance().addResourceTiming(WTFMove(resourceTiming));
276 }, m_taskMode);
