39static inline bool fallbackURLLongerThan(const std::pair<URL, URL>& lhs, const std::pair<URL, URL>& rhs)
40{
41 return lhs.first.string().length() > rhs.first.string().length();
42}

75void ApplicationCache::setManifestResource(Ref<ApplicationCacheResource>&& manifest)
76{
77 ASSERT(!m_manifest);
78 ASSERT(manifest->type() & ApplicationCacheResource::Manifest);
79
80 m_manifest = makeWeakPtr(manifest.get());
81
82 addResource(WTFMove(manifest));
83}
84
85void ApplicationCache::addResource(Ref<ApplicationCacheResource>&& resource)
86{
87 auto& url = resource->url();
88
89 ASSERT(!url.hasFragmentIdentifier());
90 ASSERT(!m_resources.contains(url.string()));
91
92 if (m_storageID) {
93 ASSERT(!resource->storageID());
94 ASSERT(resource->type() & ApplicationCacheResource::Master);
95
96 // Add the resource to the storage.
97 m_group->storage().store(resource.ptr(), this);
98 }
99
100 m_estimatedSizeInStorage += resource->estimatedSizeInStorage();
101
102 m_resources.set(url.string(), WTFMove(resource));
103}

105ApplicationCacheResource* ApplicationCache::resourceForURL(const String& url)
106{
107 ASSERT(!URL({ }, url).hasFragmentIdentifier());
108 return m_resources.get(url);
109}

111bool ApplicationCache::requestIsHTTPOrHTTPSGet(const ResourceRequest& request)
112{
113 return request.url().protocolIsInHTTPFamily() && equalLettersIgnoringASCIICase(request.httpMethod(), "get");
114}
115
116ApplicationCacheResource* ApplicationCache::resourceForRequest(const ResourceRequest& request)
117{
118 // We only care about HTTP/HTTPS GET requests.
119 if (!requestIsHTTPOrHTTPSGet(request))
120 return nullptr;
121
122 URL url(request.url());
123 url.removeFragmentIdentifier();
124 return resourceForURL(url.string());
125}

127void ApplicationCache::setOnlineAllowlist(const Vector<URL>& onlineAllowlist)
128{
129 ASSERT(m_onlineAllowlist.isEmpty());
130 m_onlineAllowlist = onlineAllowlist;

133bool ApplicationCache::isURLInOnlineAllowlist(const URL& url)
134{
135 for (auto& allowlistURL : m_onlineAllowlist) {
136 if (protocolHostAndPortAreEqual(url, allowlistURL) && url.string().startsWith(allowlistURL.string()))
137 return true;
138 }
139 return false;
140}
141
142void ApplicationCache::setFallbackURLs(const FallbackURLVector& fallbackURLs)
143{
144 ASSERT(m_fallbackURLs.isEmpty());
145 m_fallbackURLs = fallbackURLs;
146 // FIXME: What's the right behavior if we have 2 or more identical namespace URLs?
147 std::stable_sort(m_fallbackURLs.begin(), m_fallbackURLs.end(), fallbackURLLongerThan);
148}
149
150bool ApplicationCache::urlMatchesFallbackNamespace(const URL& url, URL* fallbackURL)
151{
152 for (auto& fallback : m_fallbackURLs) {
153 if (protocolHostAndPortAreEqual(url, fallback.first) && url.string().startsWith(fallback.first.string())) {
154 if (fallbackURL)
155 *fallbackURL = fallback.second;
156 return true;
157 }
158 }
159 return false;
160}

162void ApplicationCache::clearStorageID()
163{
164 m_storageID = 0;
165
166 for (const auto& resource : m_resources.values())
167 resource->clearStorageID();
168}
