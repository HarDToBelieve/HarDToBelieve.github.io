82bool ApplicationCacheHost::canLoadMainResource(const ResourceRequest& request)
83{
84 if (!isApplicationCacheEnabled() || isApplicationCacheBlockedForRequest(request))

88
89void ApplicationCacheHost::maybeLoadMainResource(const ResourceRequest& request, SubstituteData& substituteData)
90{
91 // Check if this request should be loaded from the application cache
92 if (!substituteData.isValid() && isApplicationCacheEnabled() && !isApplicationCacheBlockedForRequest(request)) {
93 ASSERT(!m_mainResourceApplicationCache);
94
95 m_mainResourceApplicationCache = ApplicationCacheGroup::cacheForMainRequest(request, &m_documentLoader);
96
97 if (m_mainResourceApplicationCache) {
98 // Get the resource from the application cache. By definition, cacheForMainRequest() returns a cache that contains the resource.
99 ApplicationCacheResource* resource = m_mainResourceApplicationCache->resourceForRequest(request);
100
101 // ApplicationCache resources have fragment identifiers stripped off of their URLs,
102 // but we'll need to restore that for the SubstituteData.
103 ResourceResponse responseToUse = resource->response();
104 if (request.url().hasFragmentIdentifier()) {
105 URL url = responseToUse.url();
106 url.setFragmentIdentifier(request.url().fragmentIdentifier());
107 responseToUse.setURL(url);
108 }
109
110 substituteData = SubstituteData(&resource->data(),
111 URL(),
112 responseToUse,
113 SubstituteData::SessionHistoryVisibility::Visible);
114 }
115 }
116}

124bool ApplicationCacheHost::maybeLoadFallbackForMainResponse(const ResourceRequest& request, const ResourceResponse& r)
125{
126 if (r.httpStatusCode() / 100 == 4 || r.httpStatusCode() / 100 == 5) {
127 ASSERT(!m_mainResourceApplicationCache);
128 if (isApplicationCacheEnabled() && !isApplicationCacheBlockedForRequest(request)) {
129 m_mainResourceApplicationCache = ApplicationCacheGroup::fallbackCacheForMainRequest(request, &m_documentLoader);
130
131 if (scheduleLoadFallbackResourceFromApplicationCache(m_documentLoader.mainResourceLoader(), m_mainResourceApplicationCache.get()))
132 return true;
133 }
134 }
135 return false;
136}

138bool ApplicationCacheHost::maybeLoadFallbackForMainError(const ResourceRequest& request, const ResourceError& error)

140 if (!error.isCancellation()) {

143 m_mainResourceApplicationCache = ApplicationCacheGroup::fallbackCacheForMainRequest(request, &m_documentLoader);
144
145 if (scheduleLoadFallbackResourceFromApplicationCache(m_documentLoader.mainResourceLoader(), m_mainResourceApplicationCache.get()))
146 return true;
147 }
148 }
149 return false;

156void ApplicationCacheHost::failedLoadingMainResource()
157{
158 auto* group = m_candidateApplicationCacheGroup.get();
159 if (!group && m_applicationCache) {
160 if (mainResourceApplicationCache()) {
161 // Even when the main resource is being loaded from an application cache, loading can fail if aborted.
162 return;
163 }
164 group = m_applicationCache->group();
165 }
166
167 if (group)
168 group->failedLoadingMainResource(m_documentLoader);
169}

171void ApplicationCacheHost::finishedLoadingMainResource()
172{
173 auto* group = candidateApplicationCacheGroup();
174 if (!group && applicationCache() && !mainResourceApplicationCache())
175 group = applicationCache()->group();
176
177 if (group)
178 group->finishedLoadingMainResource(m_documentLoader);

181bool ApplicationCacheHost::maybeLoadResource(ResourceLoader& loader, const ResourceRequest& request, const URL& originalURL)
182{
183 if (loader.options().applicationCacheMode != ApplicationCacheMode::Use)
184 return false;
185
186 if (!isApplicationCacheEnabled() && !isApplicationCacheBlockedForRequest(request))
187 return false;
188
189 if (request.url() != originalURL)
190 return false;
191
192#if ENABLE(SERVICE_WORKER)
193 if (loader.options().serviceWorkerRegistrationIdentifier)
194 return false;
195#endif
196
197 ApplicationCacheResource* resource;
198 if (!shouldLoadResourceFromApplicationCache(request, resource))
199 return false;
200
201 if (resource)
202 m_documentLoader.scheduleSubstituteResourceLoad(loader, *resource);
203 else
204 m_documentLoader.scheduleCannotShowURLError(loader);
205 return true;
206}

208bool ApplicationCacheHost::maybeLoadFallbackForRedirect(ResourceLoader* resourceLoader, ResourceRequest& request, const ResourceResponse& redirectResponse)
209{
210 if (resourceLoader && resourceLoader->options().applicationCacheMode != ApplicationCacheMode::Use)
211 return false;
212
213 if (!redirectResponse.isNull() && !protocolHostAndPortAreEqual(request.url(), redirectResponse.url())) {
214 if (scheduleLoadFallbackResourceFromApplicationCache(resourceLoader))
215 return true;
216 }
217 return false;
218}
219
220bool ApplicationCacheHost::maybeLoadFallbackForResponse(ResourceLoader* resourceLoader, const ResourceResponse& response)
221{
222 if (resourceLoader && resourceLoader->options().applicationCacheMode != ApplicationCacheMode::Use)
223 return false;
224
225 if (response.httpStatusCode() / 100 == 4 || response.httpStatusCode() / 100 == 5) {
226 if (scheduleLoadFallbackResourceFromApplicationCache(resourceLoader))
227 return true;
228 }
229 return false;
230}
231
232bool ApplicationCacheHost::maybeLoadFallbackForError(ResourceLoader* resourceLoader, const ResourceError& error)
233{
234 if (resourceLoader && resourceLoader->options().applicationCacheMode != ApplicationCacheMode::Use)
235 return false;
236
237 if (!error.isCancellation()) {
238 if (resourceLoader == m_documentLoader.mainResourceLoader())
239 return maybeLoadFallbackForMainError(resourceLoader->request(), error);
240 if (scheduleLoadFallbackResourceFromApplicationCache(resourceLoader))
241 return true;
242 }
243 return false;
244}

256static inline RefPtr<SharedBuffer> bufferFromResource(ApplicationCacheResource& resource)
257{
258 // FIXME: Clients probably do not need a copy of the SharedBuffer.

265bool ApplicationCacheHost::maybeLoadSynchronously(ResourceRequest& request, ResourceError& error, ResourceResponse& response, RefPtr<SharedBuffer>& data)
266{
267 ApplicationCacheResource* resource;
268 if (!shouldLoadResourceFromApplicationCache(request, resource))
269 return false;
270
271 auto responseData = resource ? bufferFromResource(*resource) : nullptr;
272 if (!responseData) {
273 error = m_documentLoader.frameLoader()->client().cannotShowURLError(request);
274 return true;
275 }
276
277 response = resource->response();
278 data = WTFMove(responseData);
279 return true;
280}

282void ApplicationCacheHost::maybeLoadFallbackSynchronously(const ResourceRequest& request, ResourceError& error, ResourceResponse& response, RefPtr<SharedBuffer>& data)
283{
284 // If normal loading results in a redirect to a resource with another origin (indicative of a captive portal), or a 4xx or 5xx status code or equivalent,
285 // or if there were network errors (but not if the user canceled the download), then instead get, from the cache, the resource of the fallback entry
286 // corresponding to the matched namespace.
287 if ((!error.isNull() && !error.isCancellation())
288 || response.httpStatusCode() / 100 == 4 || response.httpStatusCode() / 100 == 5
289 || !protocolHostAndPortAreEqual(request.url(), response.url())) {
290 ApplicationCacheResource* resource;
291 if (getApplicationCacheFallbackResource(request, resource)) {
292 response = resource->response();
293 // FIXME: Clients proably do not need a copy of the SharedBuffer.
294 // Remove the call to copy() once we ensure SharedBuffer will not be modified.
295 data = resource->data().copy();
296 }
297 }

311void ApplicationCacheHost::notifyDOMApplicationCache(const AtomString& eventType, int total, int done)
312{
313 if (eventType != eventNames().progressEvent)
314 InspectorInstrumentation::updateApplicationCacheStatus(m_documentLoader.frame());
315
316 if (m_defersEvents) {
317 // Event dispatching is deferred until document.onload has fired.
318 m_deferredEvents.append({ eventType, total, done });
319 return;
320 }
321
322 dispatchDOMEvent(eventType, total, done);
323}

335void ApplicationCacheHost::stopDeferringEvents()
336{
337 Ref<DocumentLoader> protect(m_documentLoader);
338
339 // Note, do not cache the size in a local variable.
340 // This code needs to properly handle the case where more events are added to
341 // m_deferredEvents while iterating it. This is why we don't use a modern for loop.
342 for (size_t i = 0; i < m_deferredEvents.size(); ++i) {
343 auto& event = m_deferredEvents[i];
344 dispatchDOMEvent(event.eventType, event.progressTotal, event.progressDone);
345 }
346
347 m_deferredEvents.clear();
348 m_defersEvents = false;
349}

351Vector<ApplicationCacheHost::ResourceInfo> ApplicationCacheHost::resourceList()
352{
353 auto* cache = applicationCache();
354 if (!cache || !cache->isComplete())
355 return { };
356
357 return WTF::map(cache->resources(), [] (auto& urlAndResource) -> ApplicationCacheHost::ResourceInfo {
358 ASSERT(urlAndResource.value);
359 auto& resource = *urlAndResource.value;
360
361 unsigned type = resource.type();
362 bool isMaster = type & ApplicationCacheResource::Master;
363 bool isManifest = type & ApplicationCacheResource::Manifest;
364 bool isExplicit = type & ApplicationCacheResource::Explicit;
365 bool isForeign = type & ApplicationCacheResource::Foreign;
366 bool isFallback = type & ApplicationCacheResource::Fallback;
367
368 return { resource.url(), isMaster, isManifest, isFallback, isForeign, isExplicit, resource.estimatedSizeInStorage() };
369 });
370}

412bool ApplicationCacheHost::shouldLoadResourceFromApplicationCache(const ResourceRequest& originalRequest, ApplicationCacheResource*& resource)
413{
414 auto* cache = applicationCache();
415 if (!cache || !cache->isComplete())
416 return false;
417
418 ResourceRequest request(originalRequest);
419 if (auto* loaderFrame = m_documentLoader.frame()) {
420 if (auto* document = loaderFrame->document())
421 document->contentSecurityPolicy()->upgradeInsecureRequestIfNeeded(request, ContentSecurityPolicy::InsecureRequestType::Load);
422 }
423
424 // If the resource is not to be fetched using the HTTP GET mechanism or equivalent, or if its URL has a different
425 // <scheme> component than the application cache's manifest, then fetch the resource normally.
426 if (!ApplicationCache::requestIsHTTPOrHTTPSGet(request) || !equalIgnoringASCIICase(request.url().protocol(), cache->manifestResource()->url().protocol()))
427 return false;
428
429 // If the resource's URL is an master entry, the manifest, an explicit entry, or a fallback entry
430 // in the application cache, then get the resource from the cache (instead of fetching it).
431 resource = cache->resourceForURL(request.url().string());
432
433 // Resources that match fallback namespaces or online allowlist entries are fetched from the network,
434 // unless they are also cached.
435 if (!resource && (cache->allowsAllNetworkRequests() || cache->urlMatchesFallbackNamespace(request.url()) || cache->isURLInOnlineAllowlist(request.url())))
436 return false;
437
438 // Resources that are not present in the manifest will always fail to load (at least, after the
439 // cache has been primed the first time), making the testing of offline applications simpler.
440 return true;
441}

443bool ApplicationCacheHost::getApplicationCacheFallbackResource(const ResourceRequest& request, ApplicationCacheResource*& resource, ApplicationCache* cache)
444{
445 if (!cache) {
446 cache = applicationCache();
447 if (!cache)
448 return false;
449 }
450 if (!cache->isComplete())
451 return false;
452
453 // If the resource is not a HTTP/HTTPS GET, then abort
454 if (!ApplicationCache::requestIsHTTPOrHTTPSGet(request))
455 return false;
456
457 URL fallbackURL;
458 if (cache->isURLInOnlineAllowlist(request.url()))
459 return false;
460 if (!cache->urlMatchesFallbackNamespace(request.url(), &fallbackURL))
461 return false;
462
463 resource = cache->resourceForURL(fallbackURL.string());
464 ASSERT(resource);
465 return true;
466}

468bool ApplicationCacheHost::scheduleLoadFallbackResourceFromApplicationCache(ResourceLoader* loader, ApplicationCache* cache)
469{
470 if (!loader)
471 return false;
472
473 if (!isApplicationCacheEnabled() && !isApplicationCacheBlockedForRequest(loader->request()))
474 return false;
475
476#if ENABLE(SERVICE_WORKER)
477 if (loader->options().serviceWorkerRegistrationIdentifier)
478 return false;
479#endif
480
481 ApplicationCacheResource* resource;
482 if (!getApplicationCacheFallbackResource(loader->request(), resource, cache))
483 return false;
484
485 loader->willSwitchToSubstituteResource();
486 m_documentLoader.scheduleSubstituteResourceLoad(*loader, *resource);
487 return true;
488}

490ApplicationCacheHost::Status ApplicationCacheHost::status() const
491{
492 auto* cache = applicationCache();
493 if (!cache)
494 return UNCACHED;
495
496 switch (cache->group()->updateStatus()) {
497 case ApplicationCacheGroup::Checking:
498 return CHECKING;
499 case ApplicationCacheGroup::Downloading:
500 return DOWNLOADING;
501 case ApplicationCacheGroup::Idle:
502 if (cache->group()->isObsolete())
503 return OBSOLETE;
504 if (cache != cache->group()->newestCache())
505 return UPDATEREADY;
506 return IDLE;
507 }
508
509 ASSERT_NOT_REACHED();
510 return UNCACHED;
511}

513bool ApplicationCacheHost::update()
514{
515 auto* cache = applicationCache();
516 if (!cache)
517 return false;
518 auto* frame = m_documentLoader.frame();
519 if (!frame)
520 return false;
521 cache->group()->update(*frame, ApplicationCacheUpdateWithoutBrowsingContext);
522 return true;
523}

525bool ApplicationCacheHost::swapCache()
526{
527 auto* cache = applicationCache();
528 if (!cache)
529 return false;
530
531 auto* group = cache->group();
532 if (!group)
533 return false;
534
535 // If the group of application caches to which cache belongs has the lifecycle status obsolete, unassociate document from cache.
536 if (group->isObsolete()) {
537 group->disassociateDocumentLoader(m_documentLoader);
538 return true;
539 }
540
541 // If there is no newer cache, raise an InvalidStateError exception.
542 auto* newestCache = group->newestCache();
543 if (!newestCache || cache == newestCache)
544 return false;
545
546 ASSERT(group == newestCache->group());
547 setApplicationCache(newestCache);
548 InspectorInstrumentation::updateApplicationCacheStatus(m_documentLoader.frame());
549 return true;
550}

552void ApplicationCacheHost::abort()
553{
554 auto* frame = m_documentLoader.frame();
555 if (!frame)
556 return;
557 if (auto* cacheGroup = candidateApplicationCacheGroup())
558 cacheGroup->abort(*frame);
559 else if (auto* cache = applicationCache())
560 cache->group()->abort(*frame);
561}

563bool ApplicationCacheHost::isApplicationCacheEnabled()
564{
565 return m_documentLoader.frame() && m_documentLoader.frame()->settings().offlineWebApplicationCacheEnabled() && !m_documentLoader.frame()->page()->usesEphemeralSession();
566}
567
568bool ApplicationCacheHost::isApplicationCacheBlockedForRequest(const ResourceRequest& request)
569{
570 auto* frame = m_documentLoader.frame();
571 if (!frame)
572 return false;
573 if (frame->isMainFrame())
574 return false;
575 return !SecurityOrigin::create(request.url())->canAccessApplicationCache(frame->document()->topOrigin());
576}
