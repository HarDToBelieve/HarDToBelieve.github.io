27
28#include "DOMWindowProperty.h"
29#include "EventTarget.h"
30#include <wtf/WeakPtr.h>
31
32namespace WebCore {
33
34class ApplicationCacheHost;
35class Frame;
36
37class DOMApplicationCache final : public RefCounted<DOMApplicationCache>, public EventTargetWithInlineData, public DOMWindowProperty {
38 WTF_MAKE_ISO_ALLOCATED(DOMApplicationCache);
39public:
40 static Ref<DOMApplicationCache> create(DOMWindow& window) { return adoptRef(*new DOMApplicationCache(window)); }
41 virtual ~DOMApplicationCache() { ASSERT(!frame()); }
42
43 unsigned short status() const;
44 ExceptionOr<void> update();
45 ExceptionOr<void> swapCache();
46 void abort();
47
48 using RefCounted::ref;
49 using RefCounted::deref;
50
51private:
52 explicit DOMApplicationCache(DOMWindow&);
53
54 void refEventTarget() final { ref(); }
55 void derefEventTarget() final { deref(); }
56
57 EventTargetInterface eventTargetInterface() const final { return DOMApplicationCacheEventTargetInterfaceType; }
58 ScriptExecutionContext* scriptExecutionContext() const final;
59
60 ApplicationCacheHost* applicationCacheHost() const;
61};

63} // namespace WebCore
