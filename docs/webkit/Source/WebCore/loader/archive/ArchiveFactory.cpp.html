49typedef RefPtr<Archive> RawDataCreationFunction(const URL&, SharedBuffer&);
50typedef HashMap<String, RawDataCreationFunction*, ASCIICaseInsensitiveHash> ArchiveMIMETypesMap;
51
52// The create functions in the archive classes return RefPtr to concrete subclasses
53// of Archive. This adaptor makes the functions have a uniform return type.
54template<typename ArchiveClass> static RefPtr<Archive> archiveFactoryCreate(const URL& url, SharedBuffer& buffer)
55{
56 return ArchiveClass::create(url, buffer);
57}
58
59static ArchiveMIMETypesMap createArchiveMIMETypesMap()
60{
61 ArchiveMIMETypesMap map;
62
63#if ENABLE(WEB_ARCHIVE) && USE(CF)
64 map.add("application/x-webarchive"_s, archiveFactoryCreate<LegacyWebArchive>);
65#endif
66
67#if ENABLE(MHTML)
68 map.add("multipart/related"_s, archiveFactoryCreate<MHTMLArchive>);
69 map.add("application/x-mimearchive"_s, archiveFactoryCreate<MHTMLArchive>);
70#endif
71
72 return map;
73}
74
75static ArchiveMIMETypesMap& archiveMIMETypes()
76{
77 static NeverDestroyed<ArchiveMIMETypesMap> map = createArchiveMIMETypesMap();
78 return map;
79}
80
81bool ArchiveFactory::isArchiveMIMEType(const String& mimeType)
82{
83 return !mimeType.isEmpty() && archiveMIMETypes().contains(mimeType);
84}
85
86RefPtr<Archive> ArchiveFactory::create(const URL& url, SharedBuffer* data, const String& mimeType)
87{
88 if (!data)
89 return nullptr;
90 if (mimeType.isEmpty())
91 return nullptr;
92 auto* function = archiveMIMETypes().get(mimeType);
93 if (!function)
94 return nullptr;
95 return function(url, *data);
96}
97
98void ArchiveFactory::registerKnownArchiveMIMETypes(HashSet<String, ASCIICaseInsensitiveHash>& mimeTypes)
99{
100 for (auto& mimeType : archiveMIMETypes().keys())
101 mimeTypes.add(mimeType);
102}
