36inline ArchiveResource::ArchiveResource(Ref<SharedBuffer>&& data, const URL& url, const String& mimeType, const String& textEncoding, const String& frameName, const ResourceResponse& response)
37 : SubstituteResource(URL { url }, ResourceResponse { response }, WTFMove(data))
38 , m_mimeType(mimeType)
39 , m_textEncoding(textEncoding)
40 , m_frameName(frameName)
41 , m_shouldIgnoreWhenUnarchiving(false)
42{
43}
44
45RefPtr<ArchiveResource> ArchiveResource::create(RefPtr<SharedBuffer>&& data, const URL& url, const String& mimeType, const String& textEncoding, const String& frameName, const ResourceResponse& response)
46{
47 if (!data)
48 return nullptr;
49 if (response.isNull()) {
50 unsigned dataSize = data->size();
51 return adoptRef(*new ArchiveResource(data.releaseNonNull(), url, mimeType, textEncoding, frameName,
52 ResourceResponse(url, mimeType, dataSize, textEncoding)));
53 }
54 return adoptRef(*new ArchiveResource(data.releaseNonNull(), url, mimeType, textEncoding, frameName, response));
55}
56
57RefPtr<ArchiveResource> ArchiveResource::create(RefPtr<SharedBuffer>&& data, const URL& url, const ResourceResponse& response)
58{
59 return create(WTFMove(data), url, response.mimeType(), response.textEncodingName(), String(), response);
60}
