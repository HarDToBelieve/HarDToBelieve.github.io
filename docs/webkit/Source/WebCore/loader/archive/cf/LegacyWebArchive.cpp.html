75RetainPtr<CFDictionaryRef> LegacyWebArchive::createPropertyListRepresentation(ArchiveResource* resource, MainResourceStatus isMainResource)
76{
77 if (!resource) {
78 // The property list representation of a null/empty WebResource has the following 3 objects stored as nil.
79 // FIXME: 0 is not serializable. Presumably we need to use kCFNull here instead for compatibility.
80 // FIXME: But why do we need to support a resource of 0? Who relies on that?
81 RetainPtr<CFMutableDictionaryRef> propertyList = adoptCF(CFDictionaryCreateMutable(0, 3, 0, 0));
82 CFDictionarySetValue(propertyList.get(), LegacyWebArchiveResourceDataKey, 0);
83 CFDictionarySetValue(propertyList.get(), LegacyWebArchiveResourceURLKey, 0);
84 CFDictionarySetValue(propertyList.get(), LegacyWebArchiveResourceMIMETypeKey, 0);
85 return propertyList;
86 }
87
88 auto propertyList = adoptCF(CFDictionaryCreateMutable(0, 6, 0, &kCFTypeDictionaryValueCallBacks));
89
90 // Resource data can be empty, but must be represented by an empty CFDataRef
91 auto& data = resource->data();
92
93 CFDictionarySetValue(propertyList.get(), LegacyWebArchiveResourceDataKey, data.createCFData().get());
94
95 // Resource URL cannot be null
96 if (auto cfURL = resource->url().string().createCFString())
97 CFDictionarySetValue(propertyList.get(), LegacyWebArchiveResourceURLKey, cfURL.get());
98 else {
99 LOG(Archives, "LegacyWebArchive - NULL resource URL is invalid - returning null property list");
100 return nullptr;
101 }
102
103 // FrameName should be left out if empty for subresources, but always included for main resources
104 auto& frameName = resource->frameName();
105 if (!frameName.isEmpty() || isMainResource)
106 CFDictionarySetValue(propertyList.get(), LegacyWebArchiveResourceFrameNameKey, frameName.createCFString().get());
107
108 // Set MIMEType, TextEncodingName, and ResourceResponse only if they actually exist
109 auto& mimeType = resource->mimeType();
110 if (!mimeType.isEmpty())
111 CFDictionarySetValue(propertyList.get(), LegacyWebArchiveResourceMIMETypeKey, mimeType.createCFString().get());
112
113 auto& textEncoding = resource->textEncoding();
114 if (!textEncoding.isEmpty())
115 CFDictionarySetValue(propertyList.get(), LegacyWebArchiveResourceTextEncodingNameKey, textEncoding.createCFString().get());
116
117 // Don't include the resource response for the main resource
118 if (!isMainResource) {
119 if (auto resourceResponseData = createPropertyListRepresentation(resource->response()))
120 CFDictionarySetValue(propertyList.get(), LegacyWebArchiveResourceResponseKey, resourceResponseData.get());
121 }
122
123 return propertyList;
124}

126RetainPtr<CFDictionaryRef> LegacyWebArchive::createPropertyListRepresentation(Archive& archive)
127{
128 auto propertyList = adoptCF(CFDictionaryCreateMutable(0, 3, 0, &kCFTypeDictionaryValueCallBacks));
129
130 auto mainResourceDict = createPropertyListRepresentation(archive.mainResource(), MainResource);
131 ASSERT(mainResourceDict);
132 if (!mainResourceDict)
133 return nullptr;
134 CFDictionarySetValue(propertyList.get(), LegacyWebArchiveMainResourceKey, mainResourceDict.get());
135
136 auto subresourcesArray = adoptCF(CFArrayCreateMutable(0, archive.subresources().size(), &kCFTypeArrayCallBacks));
137 for (auto& resource : archive.subresources()) {
138 if (auto subresource = createPropertyListRepresentation(resource.ptr(), Subresource))
139 CFArrayAppendValue(subresourcesArray.get(), subresource.get());
140 else
141 LOG(Archives, "LegacyWebArchive - Failed to create property list for subresource");
142 }
143 if (CFArrayGetCount(subresourcesArray.get()))
144 CFDictionarySetValue(propertyList.get(), LegacyWebArchiveSubresourcesKey, subresourcesArray.get());
145
146 auto subframesArray = adoptCF(CFArrayCreateMutable(0, archive.subframeArchives().size(), &kCFTypeArrayCallBacks));
147 for (auto& subframe : archive.subframeArchives()) {
148 if (auto subframeArchive = createPropertyListRepresentation(subframe.get()))
149 CFArrayAppendValue(subframesArray.get(), subframeArchive.get());
150 else
151 LOG(Archives, "LegacyWebArchive - Failed to create property list for subframe archive");
152 }
153 if (CFArrayGetCount(subframesArray.get()))
154 CFDictionarySetValue(propertyList.get(), LegacyWebArchiveSubframeArchivesKey, subframesArray.get());
155
156 return propertyList;
157}

159ResourceResponse LegacyWebArchive::createResourceResponseFromPropertyListData(CFDataRef data, CFStringRef responseDataType)
160{
161 ASSERT(data);
162 if (!data)
163 return ResourceResponse();
164
165 // If the ResourceResponseVersion (passed in as responseDataType) exists at all, this is a "new" web archive that we
166 // can parse well in a cross platform manner If it doesn't exist, we will assume this is an "old" web archive with,
167 // NSURLResponse objects in it and parse the ResourceResponse as such.
168 if (!responseDataType)
169 return createResourceResponseFromMacArchivedData(data);
170
171 // FIXME: Parse the "new" format that the above comment references here. This format doesn't exist yet.
172 return ResourceResponse();
173}

175RefPtr<ArchiveResource> LegacyWebArchive::createResource(CFDictionaryRef dictionary)
176{
177 ASSERT(dictionary);
178 if (!dictionary)
179 return nullptr;
180
181 auto resourceData = static_cast<CFDataRef>(CFDictionaryGetValue(dictionary, LegacyWebArchiveResourceDataKey));
182 if (resourceData && CFGetTypeID(resourceData) != CFDataGetTypeID()) {
183 LOG(Archives, "LegacyWebArchive - Resource data is not of type CFData, cannot create invalid resource");
184 return nullptr;
185 }
186
187 auto frameName = static_cast<CFStringRef>(CFDictionaryGetValue(dictionary, LegacyWebArchiveResourceFrameNameKey));
188 if (frameName && CFGetTypeID(frameName) != CFStringGetTypeID()) {
189 LOG(Archives, "LegacyWebArchive - Frame name is not of type CFString, cannot create invalid resource");
190 return nullptr;
191 }
192
193 auto mimeType = static_cast<CFStringRef>(CFDictionaryGetValue(dictionary, LegacyWebArchiveResourceMIMETypeKey));
194 if (mimeType && CFGetTypeID(mimeType) != CFStringGetTypeID()) {
195 LOG(Archives, "LegacyWebArchive - MIME type is not of type CFString, cannot create invalid resource");
196 return nullptr;
197 }
198
199 auto url = static_cast<CFStringRef>(CFDictionaryGetValue(dictionary, LegacyWebArchiveResourceURLKey));
200 if (url && CFGetTypeID(url) != CFStringGetTypeID()) {
201 LOG(Archives, "LegacyWebArchive - URL is not of type CFString, cannot create invalid resource");
202 return nullptr;
203 }
204
205 auto textEncoding = static_cast<CFStringRef>(CFDictionaryGetValue(dictionary, LegacyWebArchiveResourceTextEncodingNameKey));
206 if (textEncoding && CFGetTypeID(textEncoding) != CFStringGetTypeID()) {
207 LOG(Archives, "LegacyWebArchive - Text encoding is not of type CFString, cannot create invalid resource");
208 return nullptr;
209 }
210
211 ResourceResponse response;
212
213 if (auto resourceResponseData = static_cast<CFDataRef>(CFDictionaryGetValue(dictionary, LegacyWebArchiveResourceResponseKey))) {
214 if (CFGetTypeID(resourceResponseData) != CFDataGetTypeID()) {
215 LOG(Archives, "LegacyWebArchive - Resource response data is not of type CFData, cannot create invalid resource");
216 return nullptr;
217 }
218
219 auto resourceResponseVersion = static_cast<CFStringRef>(CFDictionaryGetValue(dictionary, LegacyWebArchiveResourceResponseVersionKey));
220 if (resourceResponseVersion && CFGetTypeID(resourceResponseVersion) != CFStringGetTypeID()) {
221 LOG(Archives, "LegacyWebArchive - Resource response version is not of type CFString, cannot create invalid resource");
222 return nullptr;
223 }
224
225 response = createResourceResponseFromPropertyListData(resourceResponseData, resourceResponseVersion);
226 }
227
228 return ArchiveResource::create(SharedBuffer::create(resourceData), URL(URL(), url), mimeType, textEncoding, frameName, response);
229}

256RefPtr<LegacyWebArchive> LegacyWebArchive::create(const URL&, SharedBuffer& data)
257{
258 LOG(Archives, "LegacyWebArchive - Creating from raw data");
259
260 Ref<LegacyWebArchive> archive = create();
261
262 RetainPtr<CFDataRef> cfData = data.createCFData();
263 if (!cfData)
264 return nullptr;
265
266 CFErrorRef error = nullptr;
267
268 RetainPtr<CFDictionaryRef> plist = adoptCF(static_cast<CFDictionaryRef>(CFPropertyListCreateWithData(0, cfData.get(), kCFPropertyListImmutable, 0, &error)));
269 if (!plist) {
270#ifndef NDEBUG
271 RetainPtr<CFStringRef> errorString = error ? adoptCF(CFErrorCopyDescription(error)) : 0;
272 const char* cError = errorString ? CFStringGetCStringPtr(errorString.get(), kCFStringEncodingUTF8) : "unknown error";
273 LOG(Archives, "LegacyWebArchive - Error parsing PropertyList from archive data - %s", cError);
274#endif
275 if (error)
276 CFRelease(error);
277 return nullptr;
278 }
279
280 if (CFGetTypeID(plist.get()) != CFDictionaryGetTypeID()) {
281 LOG(Archives, "LegacyWebArchive - Archive property list is not the expected CFDictionary, aborting invalid WebArchive");
282 return nullptr;
283 }
284
285 if (!archive->extract(plist.get()))
286 return nullptr;
287
288 return WTFMove(archive);
289}

291bool LegacyWebArchive::extract(CFDictionaryRef dictionary)
292{
293 ASSERT(dictionary);
294 if (!dictionary) {
295 LOG(Archives, "LegacyWebArchive - Null root CFDictionary, aborting invalid WebArchive");
296 return false;
297 }
298
299 CFDictionaryRef mainResourceDict = static_cast<CFDictionaryRef>(CFDictionaryGetValue(dictionary, LegacyWebArchiveMainResourceKey));
300 if (!mainResourceDict) {
301 LOG(Archives, "LegacyWebArchive - No main resource in archive, aborting invalid WebArchive");
302 return false;
303 }
304 if (CFGetTypeID(mainResourceDict) != CFDictionaryGetTypeID()) {
305 LOG(Archives, "LegacyWebArchive - Main resource is not the expected CFDictionary, aborting invalid WebArchive");
306 return false;
307 }
308
309 auto mainResource = createResource(mainResourceDict);
310 if (!mainResource) {
311 LOG(Archives, "LegacyWebArchive - Failed to parse main resource from CFDictionary or main resource does not exist, aborting invalid WebArchive");
312 return false;
313 }
314
315 if (mainResource->mimeType().isNull()) {
316 LOG(Archives, "LegacyWebArchive - Main resource MIME type is required, but was null.");
317 return false;
318 }
319
320 setMainResource(mainResource.releaseNonNull());
321
322 auto subresourceArray = static_cast<CFArrayRef>(CFDictionaryGetValue(dictionary, LegacyWebArchiveSubresourcesKey));
323 if (subresourceArray && CFGetTypeID(subresourceArray) != CFArrayGetTypeID()) {
324 LOG(Archives, "LegacyWebArchive - Subresources is not the expected Array, aborting invalid WebArchive");
325 return false;
326 }
327
328 if (subresourceArray) {
329 auto count = CFArrayGetCount(subresourceArray);
330 for (CFIndex i = 0; i < count; ++i) {
331 auto subresourceDict = static_cast<CFDictionaryRef>(CFArrayGetValueAtIndex(subresourceArray, i));
332 if (CFGetTypeID(subresourceDict) != CFDictionaryGetTypeID()) {
333 LOG(Archives, "LegacyWebArchive - Subresource is not expected CFDictionary, aborting invalid WebArchive");
334 return false;
335 }
336
337 if (auto subresource = createResource(subresourceDict))
338 addSubresource(subresource.releaseNonNull());
339 }
340 }
341
342 auto subframeArray = static_cast<CFArrayRef>(CFDictionaryGetValue(dictionary, LegacyWebArchiveSubframeArchivesKey));
343 if (subframeArray && CFGetTypeID(subframeArray) != CFArrayGetTypeID()) {
344 LOG(Archives, "LegacyWebArchive - Subframe archives is not the expected Array, aborting invalid WebArchive");
345 return false;
346 }
347
348 if (subframeArray) {
349 auto count = CFArrayGetCount(subframeArray);
350 for (CFIndex i = 0; i < count; ++i) {
351 auto subframeDict = static_cast<CFDictionaryRef>(CFArrayGetValueAtIndex(subframeArray, i));
352 if (CFGetTypeID(subframeDict) != CFDictionaryGetTypeID()) {
353 LOG(Archives, "LegacyWebArchive - Subframe array is not expected CFDictionary, aborting invalid WebArchive");
354 return false;
355 }
356
357 auto subframeArchive = create();
358 if (subframeArchive->extract(subframeDict))
359 addSubframeArchive(WTFMove(subframeArchive));
360 else
361 LOG(Archives, "LegacyWebArchive - Invalid subframe archive skipped");
362 }
363 }
364
365 return true;
366}

368RetainPtr<CFDataRef> LegacyWebArchive::rawDataRepresentation()
369{
370 auto propertyList = createPropertyListRepresentation(*this);
371 ASSERT(propertyList);
372 if (!propertyList) {
373 LOG(Archives, "LegacyWebArchive - Failed to create property list for archive, returning no data");
374 return nullptr;
375 }
376
377 auto stream = adoptCF(CFWriteStreamCreateWithAllocatedBuffers(0, 0));
378
379 CFWriteStreamOpen(stream.get());
380 CFPropertyListWrite(propertyList.get(), stream.get(), kCFPropertyListBinaryFormat_v1_0, 0, 0);
381
382 auto plistData = adoptCF(static_cast<CFDataRef>(CFWriteStreamCopyProperty(stream.get(), kCFStreamPropertyDataWritten)));
383 ASSERT(plistData);
384
385 CFWriteStreamClose(stream.get());
386
387 if (!plistData) {
388 LOG(Archives, "LegacyWebArchive - Failed to convert property list into raw data, returning no data");
389 return nullptr;
390 }
391
392 return plistData;
393}

413RefPtr<LegacyWebArchive> LegacyWebArchive::create(Node& node, WTF::Function<bool (Frame&)>&& frameFilter)
414{
415 Frame* frame = node.document().frame();
416 if (!frame)
417 return create();
418
419 // If the page was loaded with JavaScript enabled, we don't want to archive <noscript> tags
420 // In practice we don't actually know whether scripting was enabled when the page was originally loaded
421 // but we can approximate that by checking if scripting is enabled right now.
422 std::unique_ptr<Vector<QualifiedName>> tagNamesToFilter;
423 if (frame->page() && frame->page()->settings().isScriptEnabled()) {
424 tagNamesToFilter = makeUnique<Vector<QualifiedName>>();
425 tagNamesToFilter->append(HTMLNames::noscriptTag);
426 }
427
428 Vector<Node*> nodeList;
429 String markupString = serializeFragment(node, SerializedNodes::SubtreeIncludingNode, &nodeList, ResolveURLs::No, tagNamesToFilter.get());
430 auto nodeType = node.nodeType();
431 if (nodeType != Node::DOCUMENT_NODE && nodeType != Node::DOCUMENT_TYPE_NODE)
432 markupString = documentTypeString(node.document()) + markupString;
433
434 return create(markupString, *frame, nodeList, WTFMove(frameFilter));
435}

437RefPtr<LegacyWebArchive> LegacyWebArchive::create(Frame& frame)
438{
439 auto* documentLoader = frame.loader().documentLoader();
440 if (!documentLoader)
441 return nullptr;
442
443 auto mainResource = documentLoader->mainResource();
444 if (!mainResource)
445 return nullptr;
446
447 Vector<Ref<LegacyWebArchive>> subframeArchives;
448 for (unsigned i = 0; i < frame.tree().childCount(); ++i) {
449 if (auto childFrameArchive = create(*frame.tree().child(i)))
450 subframeArchives.append(childFrameArchive.releaseNonNull());
451 }
452
453 return create(mainResource.releaseNonNull(), documentLoader->subresources(), WTFMove(subframeArchives));
454}

471static void addSubresourcesForAttachmentElementsIfNecessary(Frame& frame, const Vector<Node*>& nodes, Vector<Ref<ArchiveResource>>& subresources)
472{
473 if (!RuntimeEnabledFeatures::sharedFeatures().attachmentElementEnabled())
474 return;
475
476 Vector<String> identifiers;
477 for (auto* node : nodes) {
478 if (!is<HTMLAttachmentElement>(node))
479 continue;
480
481 auto uniqueIdentifier = downcast<HTMLAttachmentElement>(*node).uniqueIdentifier();
482 if (uniqueIdentifier.isEmpty())
483 continue;
484
485 identifiers.append(WTFMove(uniqueIdentifier));
486 }
487
488 if (identifiers.isEmpty())
489 return;
490
491 auto* editorClient = frame.editor().client();
492 if (!editorClient)
493 return;
494
495 auto frameName = frame.tree().uniqueName();
496 for (auto& data : editorClient->serializedAttachmentDataForIdentifiers(WTFMove(identifiers))) {
497 auto resourceURL = HTMLAttachmentElement::archiveResourceURL(data.identifier);
498 if (auto resource = ArchiveResource::create(data.data.ptr(), WTFMove(resourceURL), data.mimeType, { }, frameName))
499 subresources.append(resource.releaseNonNull());
500 }
501}

505RefPtr<LegacyWebArchive> LegacyWebArchive::create(const String& markupString, Frame& frame, const Vector<Node*>& nodes, WTF::Function<bool (Frame&)>&& frameFilter)
506{
507 auto& response = frame.loader().documentLoader()->response();
508 URL responseURL = response.url();
509
510 // it's possible to have a response without a URL here
511 // <rdar://problem/5454935>
512 if (responseURL.isNull())
513 responseURL = URL({ }, emptyString());
514
515 auto mainResource = ArchiveResource::create(utf8Buffer(markupString), responseURL, response.mimeType(), "UTF-8", frame.tree().uniqueName());
516 if (!mainResource)
517 return nullptr;
518
519 Vector<Ref<LegacyWebArchive>> subframeArchives;
520 Vector<Ref<ArchiveResource>> subresources;
521 HashSet<URL> uniqueSubresources;
522
523 for (auto& nodePtr : nodes) {
524 Node& node = *nodePtr;
525 Frame* childFrame;
526 if ((is<HTMLFrameElementBase>(node) || is<HTMLObjectElement>(node))
527 && (childFrame = downcast<HTMLFrameOwnerElement>(node).contentFrame())) {
528 if (frameFilter && !frameFilter(*childFrame))
529 continue;
530 if (auto subframeArchive = create(*childFrame->document(), WTFMove(frameFilter)))
531 subframeArchives.append(subframeArchive.releaseNonNull());
532 else
533 LOG_ERROR("Unabled to archive subframe %s", childFrame->tree().uniqueName().string().utf8().data());
534
535 } else {
536 ListHashSet<URL> subresourceURLs;
537 node.getSubresourceURLs(subresourceURLs);
538
539 ASSERT(frame.loader().documentLoader());
540 auto& documentLoader = *frame.loader().documentLoader();
541
542 for (auto& subresourceURL : subresourceURLs) {
543 if (uniqueSubresources.contains(subresourceURL))
544 continue;
545
546 uniqueSubresources.add(subresourceURL);
547
548 if (auto resource = documentLoader.subresource(subresourceURL)) {
549 subresources.append(resource.releaseNonNull());
550 continue;
551 }
552
553 ResourceRequest request(subresourceURL);
554 request.setDomainForCachePartition(frame.document()->domainForCachePartition());
555
556 if (auto* cachedResource = MemoryCache::singleton().resourceForRequest(request, frame.page()->sessionID())) {
557 if (auto resource = ArchiveResource::create(cachedResource->resourceBuffer(), subresourceURL, cachedResource->response())) {
558 subresources.append(resource.releaseNonNull());
559 continue;
560 }
561 }
562
563 // FIXME: should do something better than spew to console here
564 LOG_ERROR("Failed to archive subresource for %s", subresourceURL.string().utf8().data());
565 }
566 }
567 }
568
569#if ENABLE(ATTACHMENT_ELEMENT)
570 addSubresourcesForAttachmentElementsIfNecessary(frame, nodes, subresources);
571#endif
572
573 // If we are archiving the entire page, add any link icons that we have data for.
574 if (!nodes.isEmpty() && nodes[0]->isDocumentNode()) {
575 auto* documentLoader = frame.loader().documentLoader();
576 ASSERT(documentLoader);
577 for (auto& icon : documentLoader->linkIcons()) {
578 if (auto resource = documentLoader->subresource(icon.url))
579 subresources.append(resource.releaseNonNull());
580 }
581 }
582
583 return create(mainResource.releaseNonNull(), WTFMove(subresources), WTFMove(subframeArchives));
584}

586RefPtr<LegacyWebArchive> LegacyWebArchive::createFromSelection(Frame* frame)
587{
588 if (!frame)
589 return nullptr;
590
591 auto* document = frame->document();
592 if (!document)
593 return nullptr;
594
595 StringBuilder builder;
596 builder.append(documentTypeString(*document));
597
598 Vector<Node*> nodeList;
599 auto serializeComposedTree = frame->settings().selectionAcrossShadowBoundariesEnabled() ? SerializeComposedTree::Yes : SerializeComposedTree::No;
600 builder.append(serializePreservingVisualAppearance(frame->selection().selection(), ResolveURLs::No, serializeComposedTree, &nodeList));
601
602 auto archive = create(builder.toString(), *frame, nodeList, nullptr);
603 if (!archive)
604 return nullptr;
605
606 if (!document->isFrameSet())
607 return archive;
608
609 // Wrap the frameset document in an iframe so it can be pasted into
610 // another document (which will have a body or frameset of its own).
611 String iframeMarkup = "<iframe frameborder=\"no\" marginwidth=\"0\" marginheight=\"0\" width=\"98%%\" height=\"98%%\" src=\"" + frame->loader().documentLoader()->response().url().string() + "\"></iframe>";
612 auto iframeResource = ArchiveResource::create(utf8Buffer(iframeMarkup), aboutBlankURL(), "text/html", "UTF-8", String());
613
614 Vector<Ref<LegacyWebArchive>> subframeArchives;
615 subframeArchives.reserveInitialCapacity(1);
616 subframeArchives.uncheckedAppend(archive.releaseNonNull());
617
618 return create(iframeResource.releaseNonNull(), { }, WTFMove(subframeArchives));
619}
