33bool PrivateClickMeasurement::calculateAndUpdateSourceUnlinkableToken(const String& serverPublicKeyBase64URL)
34{
35#if HAVE(RSA_BSSA)
36 {
37 auto serverPublicKeyData = base64URLDecode(serverPublicKeyBase64URL);
38 if (!serverPublicKeyData)
39 return false;
40 auto serverPublicKey = adoptNS([[NSData alloc] initWithBytes:serverPublicKeyData->data() length:serverPublicKeyData->size()]);
41
42 // FIXME(222018): Check error.
43 m_sourceUnlinkableToken.blinder = adoptNS([PAL::allocRSABSSATokenBlinderInstance() initWithPublicKey:serverPublicKey.get() error:nullptr]);
44 if (!m_sourceUnlinkableToken.blinder)
45 return false;
46 }
47
48 // FIXME(222018): Check error.
49 m_sourceUnlinkableToken.waitingToken = [m_sourceUnlinkableToken.blinder tokenWaitingActivationWithContent:nullptr error:nullptr];
50 if (!m_sourceUnlinkableToken.waitingToken)
51 return false;
52
53 m_sourceUnlinkableToken.valueBase64URL = base64URLEncodeToString([m_sourceUnlinkableToken.waitingToken blindedMessage].bytes, [m_sourceUnlinkableToken.waitingToken blindedMessage].length);
54 return true;
55#else
56 UNUSED_PARAM(serverPublicKeyBase64URL);
57 return false;
58#endif // HAVE(RSA_BSSA)
59}

61bool PrivateClickMeasurement::calculateAndUpdateSourceSecretToken(const String& serverResponseBase64URL)
62{
63#if HAVE(RSA_BSSA)
64 if (!m_sourceUnlinkableToken.waitingToken)
65 return false;
66
67 {
68 auto serverResponseData = base64URLDecode(serverResponseBase64URL);
69 if (!serverResponseData)
70 return false;
71 auto serverResponse = adoptNS([[NSData alloc] initWithBytes:serverResponseData->data() length:serverResponseData->size()]);
72
73 // FIXME(222018): Check error.
74 m_sourceUnlinkableToken.readyToken = [m_sourceUnlinkableToken.waitingToken activateTokenWithServerResponse:serverResponse.get() error:nullptr];
75 if (!m_sourceUnlinkableToken.readyToken)
76 return false;
77 }
78
79 SourceSecretToken token;
80 token.tokenBase64URL = base64URLEncodeToString([m_sourceUnlinkableToken.readyToken tokenContent].bytes, [m_sourceUnlinkableToken.readyToken tokenContent].length);
81 token.keyIDBase64URL = base64URLEncodeToString([m_sourceUnlinkableToken.readyToken keyId].bytes, [m_sourceUnlinkableToken.readyToken keyId].length);
82 token.signatureBase64URL = base64URLEncodeToString([m_sourceUnlinkableToken.readyToken signature].bytes, [m_sourceUnlinkableToken.readyToken signature].length);
83
84 m_sourceSecretToken = WTFMove(token);
85 return true;
86#else
87 UNUSED_PARAM(serverResponseBase64URL);
88 return false;
89#endif // HAVE(RSA_BSSA)
90}
