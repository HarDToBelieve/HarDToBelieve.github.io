35typedef std::pair<UChar32, Form> Key;
36struct Entry {
37 UChar32 character;
38 unsigned form : 2;
39 unsigned lspace : 3;
40 unsigned rspace : 3;
41 unsigned flags : 8;
42};
43static inline Key ExtractKey(const Entry* entry) { return Key(entry->character, static_cast<Form>(entry->form)); }
44static inline UChar32 ExtractChar(const Entry* entry) { return entry->character; }
45static inline Property ExtractProperty(const Entry& entry)
46{
47 Property property;
48 property.form = static_cast<Form>(entry.form);
49 property.leadingSpaceInMathUnit = entry.lspace;
50 property.trailingSpaceInMathUnit = entry.rspace;
51 property.flags = entry.flags;
52 return property;
53}

1128std::optional<Property> MathMLOperatorDictionary::search(UChar32 character, Form form, bool explicitForm)
1129{
1130 if (!character)
1131 return std::nullopt;
1132
1133 // We try and find the default values from the operator dictionary.
1134 if (auto* entry = tryBinarySearch<const Entry, Key>(dictionary, dictionarySize, Key(character, form), ExtractKey))
1135 return ExtractProperty(*entry);
1136
1137 if (explicitForm)
1138 return std::nullopt;
1139
1140 // If we did not find the desired operator form and if it was not set explicitly, we use the first one in the following order: Infix, Prefix, Postfix.
1141 // This is to handle bad MathML markup without explicit <mrow> delimiters like "<mo>(</mo><mi>a</mi><mo>)</mo><mo>(</mo><mi>b</mi><mo>)</mo>" where innerfences should not be considered infix.
1142 if (auto* entry = tryBinarySearch<const Entry, UChar32>(dictionary, dictionarySize, character, ExtractChar)) {
1143 // There are at most two other entries before the one found.
1144 if (entry != dictionary && (entry - 1)->character == character)
1145 entry--;
1146 if (entry != dictionary && (entry - 1)->character == character)
1147 entry--;
1148 return ExtractProperty(*entry);
1149 }

1151 return std::nullopt;
1152}
