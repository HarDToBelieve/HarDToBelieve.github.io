34namespace WebCore {
35
36namespace MathMLOperatorDictionary {
37enum Form { Infix, Prefix, Postfix };
38enum Flag {
39 Accent = 0x1,
40 Fence = 0x2, // This has no visual effect but allows to expose semantic information via the accessibility tree.
41 LargeOp = 0x4,
42 MovableLimits = 0x8,
43 Separator = 0x10, // This has no visual effect but allows to expose semantic information via the accessibility tree.
44 Stretchy = 0x20,
45 Symmetric = 0x40
46};
47const unsigned allFlags = Accent | Fence | LargeOp | MovableLimits | Separator | Stretchy | Symmetric;
48struct Property {
49 MathMLOperatorDictionary::Form form;
50 // Default leading and trailing spaces are "thickmathspace".
51 unsigned short leadingSpaceInMathUnit { 5 };
52 unsigned short trailingSpaceInMathUnit { 5 };
53 // Default operator properties are all set to "false".
54 unsigned short flags { 0 };
55};
56std::optional<Property> search(UChar32, Form, bool explicitForm);
57bool isVertical(UChar32);
58}
59
60}
61#endif // ENABLE(MATHML)
