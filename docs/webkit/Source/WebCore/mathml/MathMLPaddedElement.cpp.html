76void MathMLPaddedElement::parseAttribute(const QualifiedName& name, const AtomString& value)
77{
78 if (name == widthAttr)
79 m_width = std::nullopt;
80 else if (name == heightAttr)
81 m_height = std::nullopt;
82 else if (name == depthAttr)
83 m_depth = std::nullopt;
84 else if (name == lspaceAttr)
85 m_lspace = std::nullopt;
86 else if (name == voffsetAttr)
87 m_voffset = std::nullopt;
88
89 MathMLElement::parseAttribute(name, value);
90}

92RenderPtr<RenderElement> MathMLPaddedElement::createElementRenderer(RenderStyle&& style, const RenderTreePosition&)
93{
94 ASSERT(hasTagName(MathMLNames::mpaddedTag));
95 return createRenderer<RenderMathMLPadded>(*this, WTFMove(style));
96}
