49Ref<MathMLRowElement> MathMLRowElement::create(const QualifiedName& tagName, Document& document)

54void MathMLRowElement::childrenChanged(const ChildChange& change)
55{
56 for (auto child = firstChild(); child; child = child->nextSibling()) {
57 if (child->hasTagName(moTag))
58 static_cast<MathMLOperatorElement*>(child)->setOperatorFormDirty();
59 }
60
61 MathMLPresentationElement::childrenChanged(change);
62}
63
64RenderPtr<RenderElement> MathMLRowElement::createElementRenderer(RenderStyle&& style, const RenderTreePosition&)
65{
66 if (hasTagName(mfencedTag))
67 return createRenderer<RenderMathMLFenced>(*this, WTFMove(style));
68
69 ASSERT(hasTagName(merrorTag) || hasTagName(mphantomTag) || hasTagName(mrowTag) || hasTagName(mstyleTag));
70 return createRenderer<RenderMathMLRow>(*this, WTFMove(style));
71}
72
73bool MathMLRowElement::acceptsMathVariantAttribute()
74{
75 return hasTagName(mstyleTag);
76}
