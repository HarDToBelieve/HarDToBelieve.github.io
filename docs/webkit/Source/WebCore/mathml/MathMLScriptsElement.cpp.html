41static MathMLScriptsElement::ScriptType scriptTypeOf(const QualifiedName& tagName)
42{
43 if (tagName.matches(msubTag))
44 return MathMLScriptsElement::ScriptType::Sub;
45 if (tagName.matches(msupTag))
46 return MathMLScriptsElement::ScriptType::Super;
47 if (tagName.matches(msubsupTag))
48 return MathMLScriptsElement::ScriptType::SubSup;
49 if (tagName.matches(munderTag))
50 return MathMLScriptsElement::ScriptType::Under;
51 if (tagName.matches(moverTag))
52 return MathMLScriptsElement::ScriptType::Over;
53 if (tagName.matches(munderoverTag))
54 return MathMLScriptsElement::ScriptType::UnderOver;
55 ASSERT(tagName.matches(mmultiscriptsTag));
56 return MathMLScriptsElement::ScriptType::Multiscripts;
57}

59MathMLScriptsElement::MathMLScriptsElement(const QualifiedName& tagName, Document& document)
60 : MathMLPresentationElement(tagName, document)
61 , m_scriptType(scriptTypeOf(tagName))
62{
63}

65Ref<MathMLScriptsElement> MathMLScriptsElement::create(const QualifiedName& tagName, Document& document)
66{
67 return adoptRef(*new MathMLScriptsElement(tagName, document));
68}

70const MathMLElement::Length& MathMLScriptsElement::subscriptShift()
71{
72 return cachedMathMLLength(subscriptshiftAttr, m_subscriptShift);
73}
74
75const MathMLElement::Length& MathMLScriptsElement::superscriptShift()
76{
77 return cachedMathMLLength(superscriptshiftAttr, m_superscriptShift);
78}
79
80void MathMLScriptsElement::parseAttribute(const QualifiedName& name, const AtomString& value)
81{
82 if (name == subscriptshiftAttr)
83 m_subscriptShift = std::nullopt;
84 else if (name == superscriptshiftAttr)
85 m_superscriptShift = std::nullopt;
86
87 MathMLElement::parseAttribute(name, value);
88}

90RenderPtr<RenderElement> MathMLScriptsElement::createElementRenderer(RenderStyle&& style, const RenderTreePosition&)
91{
92 ASSERT(hasTagName(msubTag) || hasTagName(msupTag) || hasTagName(msubsupTag) || hasTagName(mmultiscriptsTag));
93 return createRenderer<RenderMathMLScripts>(*this, WTFMove(style));
94}
