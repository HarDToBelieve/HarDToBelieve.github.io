66void MathMLSpaceElement::parseAttribute(const QualifiedName& name, const AtomString& value)
67{
68 if (name == widthAttr)
69 m_width = std::nullopt;
70 else if (name == heightAttr)
71 m_height = std::nullopt;
72 else if (name == depthAttr)
73 m_depth = std::nullopt;
74
75 MathMLPresentationElement::parseAttribute(name, value);
76}

78RenderPtr<RenderElement> MathMLSpaceElement::createElementRenderer(RenderStyle&& style, const RenderTreePosition&)
79{
80 ASSERT(hasTagName(MathMLNames::mspaceTag));
81 return createRenderer<RenderMathMLSpace>(*this, WTFMove(style));
82}
