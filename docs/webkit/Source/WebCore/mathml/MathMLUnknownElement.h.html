35 WTF_MAKE_ISO_ALLOCATED(MathMLUnknownElement);
36public:
37 static Ref<MathMLUnknownElement> create(const QualifiedName& tagName, Document& document)
38 {
39 return adoptRef(*new MathMLUnknownElement(tagName, document));
40 }
41
42private:
43 MathMLUnknownElement(const QualifiedName& tagName, Document& document)
44 : MathMLElement(tagName, document, CreateMathMLElement | NodeFlag::IsUnknownElement)
45 {
46 }
47
48 bool rendererIsNeeded(const RenderStyle&) final { return false; }
49};
