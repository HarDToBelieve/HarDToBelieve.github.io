37namespace WebCore {
38
39WTF_MAKE_ISO_ALLOCATED_IMPL(BarProp);
40
41BarProp::BarProp(DOMWindow& window, Type type)
42 : DOMWindowProperty(&window)
43 , m_type(type)
44{
45}
46
47BarProp::Type BarProp::type() const
48{
49 return m_type;
50}
51
52bool BarProp::visible() const
53{
54 auto* frame = this->frame();
55 if (!frame)
56 return false;
57 auto* page = frame->page();
58 if (!page)
59 return false;
60
61 switch (m_type) {
62 case Locationbar:
63 case Personalbar:
64 case Toolbar:
65 return page->chrome().toolbarsVisible();
66 case Menubar:
67 return page->chrome().menubarVisible();
68 case Scrollbars:
69 return page->chrome().scrollbarsVisible();
70 case Statusbar:
71 return page->chrome().statusbarVisible();
72 }
73
74 ASSERT_NOT_REACHED();
75 return false;
76}
