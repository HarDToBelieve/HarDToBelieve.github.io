36namespace WebCore {
37
38class HTMLMediaElement;
39class PageGroup;
40class AudioTrackList;
41class TextTrackList;
42struct MediaSelectionOption;
43
44class CaptionUserPreferences {
45 WTF_MAKE_FAST_ALLOCATED;
46public:
47 CaptionUserPreferences(PageGroup&);
48 virtual ~CaptionUserPreferences();
49
50 enum CaptionDisplayMode {
51 Automatic,
52 ForcedOnly,
53 AlwaysOn,
54 Manual,
55 };
56 virtual CaptionDisplayMode captionDisplayMode() const;
57 virtual void setCaptionDisplayMode(CaptionDisplayMode);
58
59 virtual int textTrackSelectionScore(TextTrack*, HTMLMediaElement*) const;
60 virtual int textTrackLanguageSelectionScore(TextTrack*, const Vector<String>&) const;
61
62 virtual bool userPrefersCaptions() const;
63 virtual void setUserPrefersCaptions(bool);
64
65 virtual bool userPrefersSubtitles() const;
66 virtual void setUserPrefersSubtitles(bool preference);
67
68 virtual bool userPrefersTextDescriptions() const;
69 virtual void setUserPrefersTextDescriptions(bool preference);
70
71 virtual float captionFontSizeScaleAndImportance(bool& important) const { important = false; return 0.05f; }
72
73 virtual bool captionStrokeWidthForFont(float, const String&, float&, bool&) const { return false; }
74
75 virtual String captionsStyleSheetOverride() const { return m_captionsStyleSheetOverride; }
76 virtual void setCaptionsStyleSheetOverride(const String&);
77
78 virtual void setInterestedInCaptionPreferenceChanges() { }
79
80 virtual void captionPreferencesChanged();
81
82 virtual void setPreferredLanguage(const String&);
83 virtual Vector<String> preferredLanguages() const;
84
85 virtual void setPreferredAudioCharacteristic(const String&);
86 virtual Vector<String> preferredAudioCharacteristics() const;
87
88 virtual String displayNameForTrack(TextTrack*) const;
89 MediaSelectionOption mediaSelectionOptionForTrack(TextTrack*) const;
90 virtual Vector<RefPtr<TextTrack>> sortedTrackListForMenu(TextTrackList*, HashSet<TextTrack::Kind>);
91
92 virtual String displayNameForTrack(AudioTrack*) const;
93 MediaSelectionOption mediaSelectionOptionForTrack(AudioTrack*) const;
94 virtual Vector<RefPtr<AudioTrack>> sortedTrackListForMenu(AudioTrackList*);
95
96 void setPrimaryAudioTrackLanguageOverride(const String& language) { m_primaryAudioTrackLanguageOverride = language; }
97 String primaryAudioTrackLanguageOverride() const;
98
99 virtual bool testingMode() const { return m_testingMode; }
100 void setTestingMode(bool override) { m_testingMode = override; }
101
102 PageGroup& pageGroup() const { return m_pageGroup; }
103
104protected:
105 void updateCaptionStyleSheetOverride();
106 void beginBlockingNotifications();
107 void endBlockingNotifications();
108
109private:
110 void timerFired();
111 void notify();
112 Page* currentPage() const;
113
114 PageGroup& m_pageGroup;
115 mutable CaptionDisplayMode m_displayMode;
116 Timer m_timer;
117 String m_userPreferredLanguage;
118 String m_userPreferredAudioCharacteristic;
119 String m_captionsStyleSheetOverride;
120 String m_primaryAudioTrackLanguageOverride;
121 unsigned m_blockNotificationsCounter { 0 };
122 bool m_testingMode { false };
123 bool m_havePreferences { false };
124};
125
126}

134 WebCore::CaptionUserPreferences::CaptionDisplayMode::ForcedOnly,
135 WebCore::CaptionUserPreferences::CaptionDisplayMode::AlwaysOn,
136 WebCore::CaptionUserPreferences::CaptionDisplayMode::Manual
137 >;
138};
