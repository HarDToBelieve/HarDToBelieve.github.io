99void ContextMenuController::handleContextMenuEvent(Event& event)
100{
101 if (m_isHandlingContextMenuEvent)
102 return;
103
104 SetForScope<bool> isHandlingContextMenuEventForScope(m_isHandlingContextMenuEvent, true);
105
106 constexpr OptionSet<HitTestRequest::Type> hitType { HitTestRequest::Type::ReadOnly, HitTestRequest::Type::Active, HitTestRequest::Type::DisallowUserAgentShadowContent, HitTestRequest::Type::AllowChildFrameContent };
107 m_contextMenu = maybeCreateContextMenu(event, hitType, ContextMenuContext::Type::ContextMenu);
108 if (!m_contextMenu)
109 return;
110
111 populate();
112
113 showContextMenu(event);
114}

121void ContextMenuController::showContextMenu(Event& event, ContextMenuProvider& provider)
122{
123 m_menuProvider = &provider;
124
125 auto contextType = provider.contextMenuContextType();
126
127 OptionSet<HitTestRequest::Type> hitType { HitTestRequest::Type::ReadOnly, HitTestRequest::Type::Active, HitTestRequest::Type::AllowChildFrameContent };
128 if (contextType == ContextMenuContext::Type::ContextMenu)
129 hitType.add(HitTestRequest::Type::DisallowUserAgentShadowContent);
130
131 m_contextMenu = maybeCreateContextMenu(event, WTFMove(hitType), contextType);
132 if (!m_contextMenu) {
133 clearContextMenu();
134 return;
135 }
136
137 provider.populateContextMenu(m_contextMenu.get());
138 if (m_context.hitTestResult().isSelected()) {
139 appendItem(*separatorItem(), m_contextMenu.get());
140 populate();
141 }
142 showContextMenu(event);
143}

145std::unique_ptr<ContextMenu> ContextMenuController::maybeCreateContextMenu(Event& event, OptionSet<HitTestRequest::Type> hitType, ContextMenuContext::Type contextType)
146{
147 if (!is<MouseEvent>(event))
148 return nullptr;
149
150 auto& mouseEvent = downcast<MouseEvent>(event);
151 if (!is<Node>(mouseEvent.target()))
152 return nullptr;
153 auto& node = downcast<Node>(*mouseEvent.target());
154 auto* frame = node.document().frame();
155 if (!frame)
156 return nullptr;
157
158 auto result = frame->eventHandler().hitTestResultAtPoint(mouseEvent.absoluteLocation(), WTFMove(hitType));
159 if (!result.innerNonSharedNode())
160 return nullptr;
161
162 m_context = ContextMenuContext(contextType, result);
163
164 return makeUnique<ContextMenu>();
165}

175static void openNewWindow(const URL& urlToLoad, Frame& frame, ShouldOpenExternalURLsPolicy shouldOpenExternalURLsPolicy)
176{
177 Page* oldPage = frame.page();
178 if (!oldPage)
179 return;
180
181 FrameLoadRequest frameLoadRequest { *frame.document(), frame.document()->securityOrigin(), ResourceRequest(urlToLoad, frame.loader().outgoingReferrer()), { }, InitiatedByMainFrame::Unknown };
182 frameLoadRequest.setShouldOpenExternalURLsPolicy(shouldOpenExternalURLsPolicy);
183 frameLoadRequest.setNewFrameOpenerPolicy(NewFrameOpenerPolicy::Suppress);
184
185 Page* newPage = oldPage->chrome().createWindow(frame, { }, { *frame.document(), frameLoadRequest.resourceRequest(), frameLoadRequest.initiatedByMainFrame() });
186 if (!newPage)
187 return;
188 newPage->chrome().show();
189 newPage->mainFrame().loader().loadFrameRequest(WTFMove(frameLoadRequest), nullptr, { });
190}

206void ContextMenuController::contextMenuItemSelected(ContextMenuAction action, const String& title)
207{
208 if (action >= ContextMenuItemBaseCustomTag) {
209 ASSERT(m_menuProvider);
210 m_menuProvider->contextMenuItemSelected(action, title);
211 return;
212 }
213
214 auto& document = m_context.hitTestResult().innerNonSharedNode()->document();
215
216 auto* frame = document.frame();
217 if (!frame)
218 return;
219
220 Ref<Frame> protector(*frame);
221
222 switch (action) {
223 case ContextMenuItemTagOpenLinkInNewWindow:
224 openNewWindow(m_context.hitTestResult().absoluteLinkURL(), *frame, ShouldOpenExternalURLsPolicy::ShouldAllowExternalSchemesButNotAppLinks);
225 break;
226 case ContextMenuItemTagDownloadLinkToDisk:
227 // FIXME: Some day we should be able to do this from within WebCore. (Bug 117709)
228 m_client.downloadURL(m_context.hitTestResult().absoluteLinkURL());
229 break;
230 case ContextMenuItemTagCopyLinkToClipboard:
231 frame->editor().copyURL(m_context.hitTestResult().absoluteLinkURL(), m_context.hitTestResult().textContent());
232 break;
233 case ContextMenuItemTagOpenImageInNewWindow:
234 openNewWindow(m_context.hitTestResult().absoluteImageURL(), *frame, ShouldOpenExternalURLsPolicy::ShouldNotAllow);
235 break;
236 case ContextMenuItemTagDownloadImageToDisk:
237 // FIXME: Some day we should be able to do this from within WebCore. (Bug 117709)
238 m_client.downloadURL(m_context.hitTestResult().absoluteImageURL());
239 break;
240 case ContextMenuItemTagCopyImageToClipboard:
241 // FIXME: The Pasteboard class is not written yet
242 // For now, call into the client. This is temporary!
243 frame->editor().copyImage(m_context.hitTestResult());
244 break;
245#if PLATFORM(GTK)
246 case ContextMenuItemTagCopyImageUrlToClipboard:
247 frame->editor().copyURL(m_context.hitTestResult().absoluteImageURL(), m_context.hitTestResult().textContent());
248 break;
249#endif
250 case ContextMenuItemTagOpenMediaInNewWindow:
251 openNewWindow(m_context.hitTestResult().absoluteMediaURL(), *frame, ShouldOpenExternalURLsPolicy::ShouldNotAllow);
252 break;
253 case ContextMenuItemTagDownloadMediaToDisk:
254 // FIXME: Some day we should be able to do this from within WebCore. (Bug 117709)
255 m_client.downloadURL(m_context.hitTestResult().absoluteMediaURL());
256 break;
257 case ContextMenuItemTagCopyMediaLinkToClipboard:
258 frame->editor().copyURL(m_context.hitTestResult().absoluteMediaURL(), m_context.hitTestResult().textContent());
259 break;
260 case ContextMenuItemTagToggleMediaControls:
261 m_context.hitTestResult().toggleMediaControlsDisplay();
262 break;
263 case ContextMenuItemTagToggleMediaLoop:
264 m_context.hitTestResult().toggleMediaLoopPlayback();
265 break;
266 case ContextMenuItemTagToggleVideoFullscreen:
267 m_context.hitTestResult().toggleMediaFullscreenState();
268 break;
269 case ContextMenuItemTagEnterVideoFullscreen:
270 m_context.hitTestResult().enterFullscreenForVideo();
271 break;
272 case ContextMenuItemTagMediaPlayPause:
273 m_context.hitTestResult().toggleMediaPlayState();
274 break;
275 case ContextMenuItemTagMediaMute:
276 m_context.hitTestResult().toggleMediaMuteState();
277 break;
278 case ContextMenuItemTagToggleVideoEnhancedFullscreen:
279 m_context.hitTestResult().toggleEnhancedFullscreenForVideo();
280 break;
281 case ContextMenuItemTagOpenFrameInNewWindow: {
282 DocumentLoader* loader = frame->loader().documentLoader();
283 if (!loader->unreachableURL().isEmpty())
284 openNewWindow(loader->unreachableURL(), *frame, ShouldOpenExternalURLsPolicy::ShouldNotAllow);
285 else
286 openNewWindow(loader->url(), *frame, ShouldOpenExternalURLsPolicy::ShouldNotAllow);
287 break;
288 }
289 case ContextMenuItemTagCopy:
290 frame->editor().copy();
291 break;
292 case ContextMenuItemTagGoBack:
293 if (Page* page = frame->page())
294 page->backForward().goBackOrForward(-1);
295 break;
296 case ContextMenuItemTagGoForward:
297 if (Page* page = frame->page())
298 page->backForward().goBackOrForward(1);
299 break;
300 case ContextMenuItemTagStop:
301 frame->loader().stop();
302 break;
303 case ContextMenuItemTagReload:
304 frame->loader().reload();
305 break;
306 case ContextMenuItemTagCut:
307 frame->editor().command("Cut").execute();
308 break;
309 case ContextMenuItemTagPaste:
310 frame->editor().command("Paste").execute();
311 break;
312#if PLATFORM(GTK)
313 case ContextMenuItemTagPasteAsPlainText:
314 frame->editor().command("PasteAsPlainText").execute();
315 break;
316 case ContextMenuItemTagDelete:
317 frame->editor().performDelete();
318 break;
319 case ContextMenuItemTagUnicodeInsertLRMMark:
320 insertUnicodeCharacter(leftToRightMark, *frame);
321 break;
322 case ContextMenuItemTagUnicodeInsertRLMMark:
323 insertUnicodeCharacter(rightToLeftMark, *frame);
324 break;
325 case ContextMenuItemTagUnicodeInsertLREMark:
326 insertUnicodeCharacter(leftToRightEmbed, *frame);
327 break;
328 case ContextMenuItemTagUnicodeInsertRLEMark:
329 insertUnicodeCharacter(rightToLeftEmbed, *frame);
330 break;
331 case ContextMenuItemTagUnicodeInsertLROMark:
332 insertUnicodeCharacter(leftToRightOverride, *frame);
333 break;
334 case ContextMenuItemTagUnicodeInsertRLOMark:
335 insertUnicodeCharacter(rightToLeftOverride, *frame);
336 break;
337 case ContextMenuItemTagUnicodeInsertPDFMark:
338 insertUnicodeCharacter(popDirectionalFormatting, *frame);
339 break;
340 case ContextMenuItemTagUnicodeInsertZWSMark:
341 insertUnicodeCharacter(zeroWidthSpace, *frame);
342 break;
343 case ContextMenuItemTagUnicodeInsertZWJMark:
344 insertUnicodeCharacter(zeroWidthJoiner, *frame);
345 break;
346 case ContextMenuItemTagUnicodeInsertZWNJMark:
347 insertUnicodeCharacter(zeroWidthNonJoiner, *frame);
348 break;
349 case ContextMenuItemTagSelectAll:
350 frame->editor().command("SelectAll").execute();
351 break;
352 case ContextMenuItemTagInsertEmoji:
353 m_client.insertEmoji(*frame);
354 break;
355#endif
356 case ContextMenuItemTagSpellingGuess: {
357 VisibleSelection selection = frame->selection().selection();
358 if (frame->editor().shouldInsertText(title, selection.toNormalizedRange(), EditorInsertAction::Pasted)) {
359 OptionSet<ReplaceSelectionCommand::CommandOption> replaceOptions { ReplaceSelectionCommand::MatchStyle, ReplaceSelectionCommand::PreventNesting };
360
361 if (frame->editor().behavior().shouldAllowSpellingSuggestionsWithoutSelection()) {
362 ASSERT(selection.isCaretOrRange());
363 VisibleSelection wordSelection(selection.base());
364 wordSelection.expandUsingGranularity(TextGranularity::WordGranularity);
365 frame->selection().setSelection(wordSelection);
366 } else {
367 ASSERT(frame->editor().selectedText().length());
368 replaceOptions.add(ReplaceSelectionCommand::SelectReplacement);
369 }
370
371 Document* document = frame->document();
372 ASSERT(document);
373 auto command = ReplaceSelectionCommand::create(*document, createFragmentFromMarkup(*document, title, emptyString()), replaceOptions);
374 command->apply();
375 frame->selection().revealSelection(SelectionRevealMode::Reveal, ScrollAlignment::alignToEdgeIfNeeded);
376 }
377 break;
378 }
379 case ContextMenuItemTagIgnoreSpelling:
380 frame->editor().ignoreSpelling();
381 break;
382 case ContextMenuItemTagLearnSpelling:
383 frame->editor().learnSpelling();
384 break;
385 case ContextMenuItemTagSearchWeb:
386 m_client.searchWithGoogle(frame);
387 break;
388 case ContextMenuItemTagLookUpInDictionary:
389 // FIXME: Some day we may be able to do this from within WebCore.
390 m_client.lookUpInDictionary(frame);
391 break;
392 case ContextMenuItemTagOpenLink:
393 if (Frame* targetFrame = m_context.hitTestResult().targetFrame()) {
394 ResourceRequest resourceRequest { m_context.hitTestResult().absoluteLinkURL(), frame->loader().outgoingReferrer() };
395 FrameLoadRequest frameLoadRequest { *frame->document(), frame->document()->securityOrigin(), WTFMove(resourceRequest), { }, InitiatedByMainFrame::Unknown };
396 frameLoadRequest.setNewFrameOpenerPolicy(NewFrameOpenerPolicy::Suppress);
397 if (targetFrame->isMainFrame())
398 frameLoadRequest.setShouldOpenExternalURLsPolicy(ShouldOpenExternalURLsPolicy::ShouldAllow);
399 targetFrame->loader().loadFrameRequest(WTFMove(frameLoadRequest), nullptr, { });
400 } else
401 openNewWindow(m_context.hitTestResult().absoluteLinkURL(), *frame, ShouldOpenExternalURLsPolicy::ShouldAllow);
402 break;
403 case ContextMenuItemTagBold:
404 frame->editor().command("ToggleBold").execute();
405 break;
406 case ContextMenuItemTagItalic:
407 frame->editor().command("ToggleItalic").execute();
408 break;
409 case ContextMenuItemTagUnderline:
410 frame->editor().toggleUnderline();
411 break;
412 case ContextMenuItemTagOutline:
413 // We actually never enable this because CSS does not have a way to specify an outline font,
414 // which may make this difficult to implement. Maybe a special case of text-shadow?
415 break;
416 case ContextMenuItemTagStartSpeaking: {
417 auto selectedRange = frame->selection().selection().toNormalizedRange();
418 if (!selectedRange || selectedRange->collapsed())
419 selectedRange = makeRangeSelectingNodeContents(document);
420 m_client.speak(plainText(*selectedRange));
421 break;
422 }
423 case ContextMenuItemTagStopSpeaking:
424 m_client.stopSpeaking();
425 break;
426 case ContextMenuItemTagDefaultDirection:
427 frame->editor().setBaseWritingDirection(WritingDirection::Natural);
428 break;
429 case ContextMenuItemTagLeftToRight:
430 frame->editor().setBaseWritingDirection(WritingDirection::LeftToRight);
431 break;
432 case ContextMenuItemTagRightToLeft:
433 frame->editor().setBaseWritingDirection(WritingDirection::RightToLeft);
434 break;
435 case ContextMenuItemTagTextDirectionDefault:
436 frame->editor().command("MakeTextWritingDirectionNatural").execute();
437 break;
438 case ContextMenuItemTagTextDirectionLeftToRight:
439 frame->editor().command("MakeTextWritingDirectionLeftToRight").execute();
440 break;
441 case ContextMenuItemTagTextDirectionRightToLeft:
442 frame->editor().command("MakeTextWritingDirectionRightToLeft").execute();
443 break;
444#if PLATFORM(COCOA)
445 case ContextMenuItemTagSearchInSpotlight:
446 m_client.searchWithSpotlight();
447 break;
448#endif
449 case ContextMenuItemTagShowSpellingPanel:
450 frame->editor().showSpellingGuessPanel();
451 break;
452 case ContextMenuItemTagCheckSpelling:
453 frame->editor().advanceToNextMisspelling();
454 break;
455 case ContextMenuItemTagCheckSpellingWhileTyping:
456 frame->editor().toggleContinuousSpellChecking();
457 break;
458 case ContextMenuItemTagCheckGrammarWithSpelling:
459 frame->editor().toggleGrammarChecking();
460 break;
461#if PLATFORM(COCOA)
462 case ContextMenuItemTagShowFonts:
463 frame->editor().showFontPanel();
464 break;
465 case ContextMenuItemTagStyles:
466 frame->editor().showStylesPanel();
467 break;
468 case ContextMenuItemTagShowColors:
469 frame->editor().showColorPanel();
470 break;
471#endif
472#if USE(APPKIT)
473 case ContextMenuItemTagMakeUpperCase:
474 frame->editor().uppercaseWord();
475 break;
476 case ContextMenuItemTagMakeLowerCase:
477 frame->editor().lowercaseWord();
478 break;
479 case ContextMenuItemTagCapitalize:
480 frame->editor().capitalizeWord();
481 break;
482#endif
483#if PLATFORM(COCOA)
484 case ContextMenuItemTagChangeBack:
485 frame->editor().changeBackToReplacedString(m_context.hitTestResult().replacedString());
486 break;
487#endif
488#if USE(AUTOMATIC_TEXT_REPLACEMENT)
489 case ContextMenuItemTagShowSubstitutions:
490 frame->editor().showSubstitutionsPanel();
491 break;
492 case ContextMenuItemTagSmartCopyPaste:
493 frame->editor().toggleSmartInsertDelete();
494 break;
495 case ContextMenuItemTagSmartQuotes:

510#endif

512 if (Page* page = frame->page())
513 page->inspectorController().inspect(m_context.hitTestResult().innerNonSharedNode());
514 break;
515 case ContextMenuItemTagDictationAlternative:

1204void ContextMenuController::checkOrEnableIfNeeded(ContextMenuItem& item) const
1205{
1206 if (item.type() == SeparatorType)
1207 return;
1208
1209 Frame* frame = m_context.hitTestResult().innerNonSharedNode()->document().frame();
1210 if (!frame)
1211 return;
1212
1213 // Custom items already have proper checked and enabled values.
1214 if (ContextMenuItemBaseCustomTag <= item.action() && item.action() <= ContextMenuItemLastCustomTag)
1215 return;
1216
1217 bool shouldEnable = true;
1218 bool shouldCheck = false;
1219
1220 switch (item.action()) {
1221 case ContextMenuItemTagCheckSpelling:
1222 shouldEnable = frame->editor().canEdit();
1223 break;
1224 case ContextMenuItemTagDefaultDirection:
1225 shouldCheck = false;
1226 shouldEnable = false;
1227 break;
1228 case ContextMenuItemTagLeftToRight:
1229 case ContextMenuItemTagRightToLeft: {
1230 String direction = item.action() == ContextMenuItemTagLeftToRight ? "ltr" : "rtl";
1231 shouldCheck = frame->editor().selectionHasStyle(CSSPropertyDirection, direction) != TriState::False;
1232 shouldEnable = true;
1233 break;
1234 }
1235 case ContextMenuItemTagTextDirectionDefault: {
1236 Editor::Command command = frame->editor().command("MakeTextWritingDirectionNatural");
1237 shouldCheck = command.state() == TriState::True;
1238 shouldEnable = command.isEnabled();
1239 break;
1240 }
1241 case ContextMenuItemTagTextDirectionLeftToRight: {
1242 Editor::Command command = frame->editor().command("MakeTextWritingDirectionLeftToRight");
1243 shouldCheck = command.state() == TriState::True;
1244 shouldEnable = command.isEnabled();
1245 break;
1246 }
1247 case ContextMenuItemTagTextDirectionRightToLeft: {
1248 Editor::Command command = frame->editor().command("MakeTextWritingDirectionRightToLeft");
1249 shouldCheck = command.state() == TriState::True;
1250 shouldEnable = command.isEnabled();
1251 break;
1252 }
1253 case ContextMenuItemTagCopy:
1254 shouldEnable = frame->editor().canDHTMLCopy() || frame->editor().canCopy();
1255 break;
1256 case ContextMenuItemTagCut:
1257 shouldEnable = frame->editor().canDHTMLCut() || frame->editor().canCut();
1258 break;
1259 case ContextMenuItemTagIgnoreSpelling:
1260 case ContextMenuItemTagLearnSpelling:
1261 shouldEnable = frame->selection().isRange();
1262 break;
1263 case ContextMenuItemTagPaste:
1264 shouldEnable = frame->editor().canDHTMLPaste() || frame->editor().canPaste();
1265 break;
1266#if PLATFORM(GTK)
1267 case ContextMenuItemTagPasteAsPlainText:
1268 shouldEnable = frame->editor().canDHTMLPaste() || frame->editor().canPaste();
1269 break;
1270 case ContextMenuItemTagDelete:
1271 shouldEnable = frame->editor().canDelete();
