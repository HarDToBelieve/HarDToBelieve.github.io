43static RefPtr<Node> selectionShadowAncestor(Frame& frame)
44{
45 ASSERT(!frame.settings().liveRangeSelectionEnabled());
46 auto* node = frame.selection().selection().base().anchorNode();
47 if (!node || !node->isInShadowTree())
48 return nullptr;
49 return node->document().ancestorNodeInThisScope(node);
50}

67std::optional<SimpleRange> DOMSelection::range() const
68{
69 auto frame = this->frame();
70 if (!frame)
71 return std::nullopt;
72 auto range = frame->settings().liveRangeSelectionEnabled()
73 ? frame->selection().selection().range()
74 : frame->selection().selection().firstRange();
75 if (!range || range->start.container->isInShadowTree())
76 return std::nullopt;
77 return range;
78}

164bool DOMSelection::isCollapsed() const
165{
166 auto frame = this->frame();
167 if (!frame)
168 return true;
169 auto range = this->range();
170 return !range || range->collapsed();
171}

173String DOMSelection::type() const
174{
175 auto frame = this->frame();
176 if (!frame)
177 return "None"_s;
178 auto& selection = frame->selection();
179 if (frame->settings().liveRangeSelectionEnabled())
180 return !selection.isInDocumentTree() ? "None"_s : range()->collapsed() ? "Caret"_s : "Range"_s;
181 if (selection.isNone())
182 return "None"_s;
183 if (selection.isCaret())
184 return "Caret"_s;
185 return "Range"_s;
186}

188unsigned DOMSelection::rangeCount() const
189{
190 auto frame = this->frame();
191 if (frame && frame->settings().liveRangeSelectionEnabled())
192 return frame->selection().isInDocumentTree();
193 return !frame || frame->selection().isNone() ? 0 : 1;

196ExceptionOr<void> DOMSelection::collapse(Node* node, unsigned offset)
197{
198 auto frame = this->frame();
199 if (!frame)
200 return { };
201 if (frame->settings().liveRangeSelectionEnabled()) {
202 if (!node) {
203 removeAllRanges();
204 return { };
205 }
206 if (auto result = Range::checkNodeOffsetPair(*node, offset); result.hasException())
207 return result.releaseException();
208 if (!frame->document()->contains(*node))
209 return { };
210 } else {
211 if (!isValidForPosition(node))
212 return { };
213 }
214 auto& selection = frame->selection();
215 selection.disassociateLiveRange();
216 selection.moveTo(makeContainerOffsetPosition(node, offset), Affinity::Downstream);
217 return { };
218}

220ExceptionOr<void> DOMSelection::collapseToEnd()
221{
222 auto frame = this->frame();
223 if (!frame)
224 return { };
225 auto& selection = frame->selection();
226 if (selection.isNone())
227 return Exception { InvalidStateError };
228 if (frame->settings().liveRangeSelectionEnabled()) {
229 selection.disassociateLiveRange();
230 selection.moveTo(selection.selection().uncanonicalizedEnd(), Affinity::Downstream);
231 } else
232 selection.moveTo(selection.selection().end(), Affinity::Downstream);
233 return { };
234}

257ExceptionOr<void> DOMSelection::setBaseAndExtent(Node* baseNode, unsigned baseOffset, Node* extentNode, unsigned extentOffset)
258{
259 auto frame = this->frame();
260 if (!frame)
261 return { };
262 if (frame->settings().liveRangeSelectionEnabled()) {
263 // FIXME: We should do this by making the arguments non-nullable in the IDL file, once liveRangeSelectionEnabled is always true.
264 if (!baseNode || !extentNode)
265 return Exception { TypeError };
266 if (auto result = Range::checkNodeOffsetPair(*baseNode, baseOffset); result.hasException())
267 return result.releaseException();
268 if (auto result = Range::checkNodeOffsetPair(*extentNode, extentOffset); result.hasException())
269 return result.releaseException();
270 auto& document = *frame->document();
271 if (!document.contains(*baseNode) || !document.contains(*extentNode))
272 return { };
273 } else {
274 if (!isValidForPosition(baseNode) || !isValidForPosition(extentNode))
275 return { };
276 }
277 auto& selection = frame->selection();
278 selection.disassociateLiveRange();
279 selection.moveTo(makeContainerOffsetPosition(baseNode, baseOffset), makeContainerOffsetPosition(extentNode, extentOffset), Affinity::Downstream);
280 return { };
281}

288void DOMSelection::modify(const String& alterString, const String& directionString, const String& granularityString)
289{
290 FrameSelection::EAlteration alter;
291 if (equalLettersIgnoringASCIICase(alterString, "extend"))
292 alter = FrameSelection::AlterationExtend;
293 else if (equalLettersIgnoringASCIICase(alterString, "move"))
294 alter = FrameSelection::AlterationMove;
295 else
296 return;
297
298 SelectionDirection direction;
299 if (equalLettersIgnoringASCIICase(directionString, "forward"))
300 direction = SelectionDirection::Forward;
301 else if (equalLettersIgnoringASCIICase(directionString, "backward"))
302 direction = SelectionDirection::Backward;
303 else if (equalLettersIgnoringASCIICase(directionString, "left"))
304 direction = SelectionDirection::Left;
305 else if (equalLettersIgnoringASCIICase(directionString, "right"))
306 direction = SelectionDirection::Right;
307 else
308 return;
309
310 TextGranularity granularity;
311 if (equalLettersIgnoringASCIICase(granularityString, "character"))
312 granularity = TextGranularity::CharacterGranularity;
313 else if (equalLettersIgnoringASCIICase(granularityString, "word"))
314 granularity = TextGranularity::WordGranularity;
315 else if (equalLettersIgnoringASCIICase(granularityString, "sentence"))
316 granularity = TextGranularity::SentenceGranularity;
317 else if (equalLettersIgnoringASCIICase(granularityString, "line"))
318 granularity = TextGranularity::LineGranularity;
319 else if (equalLettersIgnoringASCIICase(granularityString, "paragraph"))
320 granularity = TextGranularity::ParagraphGranularity;
321 else if (equalLettersIgnoringASCIICase(granularityString, "lineboundary"))
322 granularity = TextGranularity::LineBoundary;
323 else if (equalLettersIgnoringASCIICase(granularityString, "sentenceboundary"))
324 granularity = TextGranularity::SentenceBoundary;
325 else if (equalLettersIgnoringASCIICase(granularityString, "paragraphboundary"))
326 granularity = TextGranularity::ParagraphBoundary;
327 else if (equalLettersIgnoringASCIICase(granularityString, "documentboundary"))
328 granularity = TextGranularity::DocumentBoundary;
329 else
330 return;
331
332 if (auto frame = this->frame())
333 frame->selection().modify(alter, direction, granularity);
334}

336ExceptionOr<void> DOMSelection::extend(Node& node, unsigned offset)
337{
338 auto frame = this->frame();
339 if (!frame)
340 return { };
341 if (frame->settings().liveRangeSelectionEnabled()) {
342 if (!frame->document()->contains(node))
343 return { };
344 if (auto result = Range::checkNodeOffsetPair(node, offset); result.hasException())
345 return result.releaseException();
346 auto& selection = frame->selection();
347 auto newSelection = selection.selection();
348 newSelection.setExtent(makeContainerOffsetPosition(&node, offset));
349 selection.disassociateLiveRange();
350 selection.setSelection(newSelection);
351 } else {
352 if (offset > node.length())
353 return Exception { IndexSizeError };
354 if (!isValidForPosition(&node))
355 return { };
356 frame->selection().setExtent(makeContainerOffsetPosition(&node, offset), Affinity::Downstream);
357 }
358 return { };
359}

361ExceptionOr<Ref<Range>> DOMSelection::getRangeAt(unsigned index)
362{
363 if (index >= rangeCount())
364 return Exception { IndexSizeError };
365 auto frame = this->frame().releaseNonNull();
366 if (frame->settings().liveRangeSelectionEnabled())
367 return frame->selection().associatedLiveRange().releaseNonNull();
368 if (auto shadowAncestor = selectionShadowAncestor(frame))
369 return createLiveRange(makeSimpleRange(*makeBoundaryPointBeforeNode(*shadowAncestor)));
370 return createLiveRange(*frame->selection().selection().firstRange());
371}

381void DOMSelection::addRange(Range& liveRange)
382{
383 auto frame = this->frame();
384 if (!frame)
385 return;
386 auto& selection = frame->selection();
387 if (frame->settings().liveRangeSelectionEnabled()) {
388 if (selection.isNone())
389 selection.associateLiveRange(liveRange);
390 return;
391 }
392 auto range = makeSimpleRange(liveRange);
393 if (auto selectedRange = selection.selection().toNormalizedRange()) {
394 if (!selectedRange->start.container->containingShadowRoot() && intersects(*selectedRange, range))
395 selection.setSelection(unionRange(*selectedRange, range));
396 return;
397 }
398 selection.setSelection(range);
399}

401ExceptionOr<void> DOMSelection::removeRange(Range& liveRange)
402{
403 auto frame = this->frame();
404 if (!frame)
405 return { };
406 ASSERT(frame->settings().liveRangeSelectionEnabled());
407 if (&liveRange != frame->selection().associatedLiveRange())
408 return Exception { NotFoundError };
409 removeAllRanges();
410 return { };
411}

413void DOMSelection::deleteFromDocument()
414{
415 auto frame = this->frame();
416 if (!frame)
417 return;
418 if (frame->settings().liveRangeSelectionEnabled()) {
419 if (auto range = frame->selection().associatedLiveRange())
420 range->deleteContents();
421 return;
422 }
423 auto selectedRange = frame->selection().selection().toNormalizedRange();
424 if (!selectedRange || selectedRange->start.container->containingShadowRoot())
425 return;
426 createLiveRange(*selectedRange)->deleteContents();
427 frame->selection().setSelectedRange(makeSimpleRange(selectedRange->start), Affinity::Downstream, FrameSelection::ShouldCloseTyping::No);
428}

430bool DOMSelection::containsNode(Node& node, bool allowPartial) const
431{
432 // FIXME: The rule implemented here for text nodes is wrong, and was added to work around anomalies caused when we canonicalize selection endpoints.
433 if (node.isTextNode() && !node.document().settings().liveRangeSelectionEnabled())
434 allowPartial = true;
435 auto range = this->range();
436 return range && (allowPartial ? intersects(*range, node) : contains(*range, node));
437}

439ExceptionOr<void> DOMSelection::selectAllChildren(Node& node)
440{
441 // This doesn't (and shouldn't) select the characters in a node if passed a text node.
442 // Selection API specification seems to have this wrong: https://github.com/w3c/selection-api/issues/125
443 return setBaseAndExtent(&node, 0, &node, node.countChildNodes());
444}

446String DOMSelection::toString() const
447{
448 auto frame = this->frame();
449 if (!frame)
450 return String();
451 if (frame->settings().liveRangeSelectionEnabled()) {
452 auto range = this->range();
453 return range ? plainText(*range) : emptyString();
454 }
455 auto range = frame->selection().selection().firstRange();
456 return range ? plainText(*range) : emptyString();
457}

459RefPtr<Node> DOMSelection::shadowAdjustedNode(const Position& position) const
460{
461 if (position.isNull())
462 return nullptr;
463
464 if (frame()->settings().liveRangeSelectionEnabled()) {
465 auto node = position.containerNode();
466 return !node || node->isInShadowTree() ? nullptr : node;
467 }
468
469 auto* containerNode = position.containerNode();
470 auto* adjustedNode = frame()->document()->ancestorNodeInThisScope(containerNode);
471 if (!adjustedNode)
472 return nullptr;
473
474 if (containerNode == adjustedNode)
475 return containerNode;
476
477 return adjustedNode->parentNodeGuaranteedHostFree();
478}
479
480unsigned DOMSelection::shadowAdjustedOffset(const Position& position) const
481{
482 if (position.isNull())
483 return 0;
484
485 if (frame()->settings().liveRangeSelectionEnabled())
486 return shadowAdjustedNode(position) ? position.computeOffsetInContainerNode() : 0;
487
488 auto* containerNode = position.containerNode();
489 auto* adjustedNode = frame()->document()->ancestorNodeInThisScope(containerNode);
490 if (!adjustedNode)
491 return 0;
492
493 if (containerNode == adjustedNode)
494 return position.computeOffsetInContainerNode();
495
496 return adjustedNode->computeNodeIndex();
497}

499bool DOMSelection::isValidForPosition(Node* node) const
500{
501 auto frame = this->frame();
502 ASSERT(!frame->settings().liveRangeSelectionEnabled());
503 return frame && (!node || &node->document() == frame->document());
504}
