30#pragma once
31
32#include "DOMWindowProperty.h"
33#include "ExceptionOr.h"
34#include <wtf/Forward.h>
35#include <wtf/RefCounted.h>
36#include <wtf/RefPtr.h>
37
38namespace WebCore {
39
40class Node;
41class Position;
42class Range;
43class VisibleSelection;
44
45struct SimpleRange;
46
47class DOMSelection : public RefCounted<DOMSelection>, public DOMWindowProperty {
48public:
49 static Ref<DOMSelection> create(DOMWindow&);
50
51 RefPtr<Node> baseNode() const;
52 RefPtr<Node> extentNode() const;
53 unsigned baseOffset() const;
54 unsigned extentOffset() const;
55 String type() const;
56 ExceptionOr<void> setBaseAndExtent(Node* baseNode, unsigned baseOffset, Node* extentNode, unsigned extentOffset);
57 ExceptionOr<void> setPosition(Node*, unsigned offset);
58 void modify(const String& alter, const String& direction, const String& granularity);
59
60 // The anchor and focus are the start and end of the selection, and
61 // reflect the direction in which the selection was made by the user.
62 // The base and extent are different, because they don't reflect expansion.
63 RefPtr<Node> anchorNode() const;
64 unsigned anchorOffset() const;
65 RefPtr<Node> focusNode() const;
66 unsigned focusOffset() const;
67 bool isCollapsed() const;
68 unsigned rangeCount() const;
69 ExceptionOr<void> collapse(Node*, unsigned offset);
70 ExceptionOr<void> collapseToEnd();
71 ExceptionOr<void> collapseToStart();
72 ExceptionOr<void> extend(Node&, unsigned offset);
73 ExceptionOr<Ref<Range>> getRangeAt(unsigned);
74 void removeAllRanges();
75 void addRange(Range&);
76 ExceptionOr<void> removeRange(Range&);
77 void deleteFromDocument();
78 bool containsNode(Node&, bool partlyContained) const;
79 ExceptionOr<void> selectAllChildren(Node&);
80
81 String toString() const;
82
83 void empty();
84
85private:
86 explicit DOMSelection(DOMWindow&);
87
88 // FIXME: Change DOMWindowProperty::frame to return RefPtr and then delete this.
89 RefPtr<Frame> frame() const;
90 std::optional<SimpleRange> range() const;
91
92 Position anchorPosition() const;
93 Position focusPosition() const;
94 Position basePosition() const;
95 Position extentPosition() const;
96
97 RefPtr<Node> shadowAdjustedNode(const Position&) const;
98 unsigned shadowAdjustedOffset(const Position&) const;
99
100 bool isValidForPosition(Node*) const;
101};
102
103} // namespace WebCore
