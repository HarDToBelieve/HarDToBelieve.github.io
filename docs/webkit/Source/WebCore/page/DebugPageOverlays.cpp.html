93bool MouseWheelRegionOverlay::updateRegion()
94{
95#if ENABLE(WHEEL_EVENT_REGIONS)
96 // Wheel event regions are painted via RenderLayerBacking::paintDebugOverlays().
97 return false;
98#else
99 auto region = makeUnique<Region>();
100
101 for (const Frame* frame = &m_page.mainFrame(); frame; frame = frame->tree().traverseNext()) {
102 if (!frame->view() || !frame->document())
103 continue;
104
105 auto frameRegion = frame->document()->absoluteRegionForEventTargets(frame->document()->wheelEventTargets());
106 frameRegion.first.translate(toIntSize(frame->view()->contentsToRootView(IntPoint())));
107 region->unite(frameRegion.first);
108 }
109
110 region->translate(m_overlay->viewToOverlayOffset());
111
112 bool regionChanged = !m_region || !(*m_region == *region);
113 m_region = WTFMove(region);
114 return regionChanged;
115#endif
116}

137bool NonFastScrollableRegionOverlay::updateRegion()
138{
139 bool regionChanged = false;
140
141 if (ScrollingCoordinator* scrollingCoordinator = m_page.scrollingCoordinator()) {
142 EventTrackingRegions eventTrackingRegions = scrollingCoordinator->absoluteEventTrackingRegions();
143
144 if (eventTrackingRegions != m_eventTrackingRegions) {
145 m_eventTrackingRegions = eventTrackingRegions;
146 regionChanged = true;
147 }
148 }
149
150 return regionChanged;
151}

170static void drawRightAlignedText(const String& text, GraphicsContext& context, const FontCascade& font, const FloatPoint& boxLocation)
171{
172 float textGap = 10;
173 float textBaselineFromTop = 14;
174
175 TextRun textRun = TextRun(text);
176 float textWidth = font.width(textRun);
177 context.setFillColor(Color::black);
178 context.drawText(font, textRun, boxLocation + FloatSize(-(textWidth + textGap), textBaselineFromTop));
179}

181void NonFastScrollableRegionOverlay::drawRect(PageOverlay& pageOverlay, GraphicsContext& context, const IntRect&)
182{
183 IntRect bounds = pageOverlay.bounds();
184
185 context.clearRect(bounds);
186
187 FloatRect legendRect = { bounds.maxX() - 30.0f, 10, 20, 20 };
188
189 FontCascadeDescription fontDescription;
190 fontDescription.setOneFamily("Helvetica");
191 fontDescription.setSpecifiedSize(12);
192 fontDescription.setComputedSize(12);
193 fontDescription.setWeight(FontSelectionValue(500));
194 FontCascade font(WTFMove(fontDescription), 0, 0);
195 font.update(nullptr);
196
197#if ENABLE(TOUCH_EVENTS)
198 context.setFillColor(touchEventRegionColors().get("touchstart"_s));
199 context.fillRect(legendRect);
200 drawRightAlignedText("touchstart"_s, context, font, legendRect.location());
201
202 legendRect.move(0, 30);
203 context.setFillColor(touchEventRegionColors().get("touchmove"_s));
204 context.fillRect(legendRect);
205 drawRightAlignedText("touchmove"_s, context, font, legendRect.location());
206
207 legendRect.move(0, 30);
208 context.setFillColor(touchEventRegionColors().get("touchend"_s));
209 context.fillRect(legendRect);
210 drawRightAlignedText("touchend"_s, context, font, legendRect.location());
211
212 legendRect.move(0, 30);
213 context.setFillColor(touchEventRegionColors().get("touchforcechange"_s));
214 context.fillRect(legendRect);
215 drawRightAlignedText("touchforcechange"_s, context, font, legendRect.location());
216
217 legendRect.move(0, 30);
218 context.setFillColor(m_color);
219 context.fillRect(legendRect);
220 drawRightAlignedText("passive listeners"_s, context, font, legendRect.location());
221
222 legendRect.move(0, 30);
223 context.setFillColor(touchEventRegionColors().get("mousedown"_s));
224 context.fillRect(legendRect);
225 drawRightAlignedText("mousedown"_s, context, font, legendRect.location());
226
227 legendRect.move(0, 30);
228 context.setFillColor(touchEventRegionColors().get("mousemove"_s));
229 context.fillRect(legendRect);
230 drawRightAlignedText("mousemove"_s, context, font, legendRect.location());
231
232 legendRect.move(0, 30);
233 context.setFillColor(touchEventRegionColors().get("mouseup"_s));
234 context.fillRect(legendRect);
235 drawRightAlignedText("mouseup"_s, context, font, legendRect.location());
236#else
237 // On desktop platforms, the "wheel" region includes the non-fast scrollable region.
238 context.setFillColor(touchEventRegionColors().get("wheel"_s));
239 context.fillRect(legendRect);
240 drawRightAlignedText("non-fast region"_s, context, font, legendRect.location());
241#endif
242
243 for (const auto& synchronousEventRegion : m_eventTrackingRegions.eventSpecificSynchronousDispatchRegions) {
244 auto regionColor = Color::black.colorWithAlphaByte(64);
245 auto it = touchEventRegionColors().find(synchronousEventRegion.key);
246 if (it != touchEventRegionColors().end())
247 regionColor = it->value;
248 drawRegion(context, synchronousEventRegion.value, regionColor, bounds);
249 }
250
251 drawRegion(context, m_eventTrackingRegions.asynchronousDispatchRegion, m_color, bounds);
252}

254Ref<RegionOverlay> RegionOverlay::create(Page& page, DebugPageOverlays::RegionType regionType)
255{
256 switch (regionType) {
257 case DebugPageOverlays::RegionType::WheelEventHandlers:
258 return MouseWheelRegionOverlay::create(page);
259 case DebugPageOverlays::RegionType::NonFastScrollableRegion:
260 return NonFastScrollableRegionOverlay::create(page);
261 }
262 ASSERT_NOT_REACHED();
263 return MouseWheelRegionOverlay::create(page);
264}

320void RegionOverlay::recomputeRegion()
321{
322 if (!m_regionChanged)
323 return;
324

339static inline size_t indexOf(DebugPageOverlays::RegionType regionType)
340{
341 return static_cast<size_t>(regionType);
342}

344RegionOverlay& DebugPageOverlays::ensureRegionOverlayForPage(Page& page, RegionType regionType)
345{
346 auto it = m_pageRegionOverlays.find(&page);
347 if (it != m_pageRegionOverlays.end()) {
348 auto& visualizer = it->value[indexOf(regionType)];
349 if (!visualizer)
350 visualizer = RegionOverlay::create(page, regionType);
351 return *visualizer;
352 }
353
354 Vector<RefPtr<RegionOverlay>> visualizers(NumberOfRegionTypes);
355 auto visualizer = RegionOverlay::create(page, regionType);
356 visualizers[indexOf(regionType)] = visualizer.copyRef();
357 m_pageRegionOverlays.add(&page, WTFMove(visualizers));
358 return visualizer;
359}
360
361void DebugPageOverlays::showRegionOverlay(Page& page, RegionType regionType)
362{
363 auto& visualizer = ensureRegionOverlayForPage(page, regionType);
364 page.pageOverlayController().installPageOverlay(visualizer.overlay(), PageOverlay::FadeMode::DoNotFade);
365}
366
367void DebugPageOverlays::hideRegionOverlay(Page& page, RegionType regionType)
368{
369 auto it = m_pageRegionOverlays.find(&page);
370 if (it == m_pageRegionOverlays.end())
371 return;
372 auto& visualizer = it->value[indexOf(regionType)];
373 if (!visualizer)
374 return;
375 page.pageOverlayController().uninstallPageOverlay(visualizer->overlay(), PageOverlay::FadeMode::DoNotFade);
376 visualizer = nullptr;
377}
378
379void DebugPageOverlays::regionChanged(Frame& frame, RegionType regionType)
380{
381 auto* page = frame.page();
382 if (!page)
383 return;
384
385 if (auto* visualizer = regionOverlayForPage(*page, regionType))
386 visualizer->setRegionChanged();
387}
388
389void DebugPageOverlays::updateRegionIfNecessary(Page& page, RegionType regionType)
390{
391 if (auto* visualizer = regionOverlayForPage(page, regionType))
392 visualizer->recomputeRegion();
393}
394
395RegionOverlay* DebugPageOverlays::regionOverlayForPage(Page& page, RegionType regionType) const
396{
397 auto it = m_pageRegionOverlays.find(&page);
398 if (it == m_pageRegionOverlays.end())
399 return nullptr;
400 return it->value.at(indexOf(regionType)).get();
401}

403void DebugPageOverlays::updateOverlayRegionVisibility(Page& page, OptionSet<DebugOverlayRegions> visibleRegions)
404{
405 if (visibleRegions.contains(DebugOverlayRegions::NonFastScrollableRegion))
406 showRegionOverlay(page, RegionType::NonFastScrollableRegion);
407 else
408 hideRegionOverlay(page, RegionType::NonFastScrollableRegion);
409
410 if (visibleRegions.contains(DebugOverlayRegions::WheelEventHandlerRegion))
411 showRegionOverlay(page, RegionType::WheelEventHandlers);
412 else
413 hideRegionOverlay(page, RegionType::WheelEventHandlers);
414}

416void DebugPageOverlays::settingsChanged(Page& page)
417{
418 auto activeOverlayRegions = OptionSet<DebugOverlayRegions>::fromRaw(page.settings().visibleDebugOverlayRegions());
419 if (!activeOverlayRegions && !hasOverlays(page))
420 return;
421
422 DebugPageOverlays::singleton().updateOverlayRegionVisibility(page, activeOverlayRegions);
423}
