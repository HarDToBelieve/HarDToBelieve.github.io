28#include "DebugOverlayRegions.h"

34namespace WebCore {
35
36class Page;
37class RegionOverlay;
38
39class DebugPageOverlays {
40public:
41 static DebugPageOverlays& singleton();
42
43 enum class RegionType {
44 WheelEventHandlers,
45 NonFastScrollableRegion,
46 };
47 static constexpr unsigned NumberOfRegionTypes = static_cast<unsigned>(RegionType::NonFastScrollableRegion) + 1;
48
49 static void didLayout(Frame&);
50 static void didChangeEventHandlers(Frame&);
51 static void doAfterUpdateRendering(Page&);
52
53 WEBCORE_EXPORT static void settingsChanged(Page&);
54
55private:
56 static bool hasOverlays(Page&);
57
58 void showRegionOverlay(Page&, RegionType);
59 void hideRegionOverlay(Page&, RegionType);
60
61 void updateRegionIfNecessary(Page&, RegionType);
62
63 void regionChanged(Frame&, RegionType);
64
65 bool hasOverlaysForPage(Page& page) const
66 {
67 return m_pageRegionOverlays.contains(&page);
68 }
69
70 void updateOverlayRegionVisibility(Page&, OptionSet<DebugOverlayRegions>);
71
72 RegionOverlay* regionOverlayForPage(Page&, RegionType) const;
73 RegionOverlay& ensureRegionOverlayForPage(Page&, RegionType);
74
75 HashMap<Page*, Vector<RefPtr<RegionOverlay>>> m_pageRegionOverlays;
76
77 static DebugPageOverlays* sharedDebugOverlays;
78};
79
80#define FAST_RETURN_IF_NO_OVERLAYS(page) if (LIKELY(!page || !hasOverlays(*page))) return;
81
82inline bool DebugPageOverlays::hasOverlays(Page& page)
83{
84 if (!sharedDebugOverlays)
85 return false;
86
87 return sharedDebugOverlays->hasOverlaysForPage(page);
88}
89
90inline void DebugPageOverlays::didLayout(Frame& frame)
91{
92 FAST_RETURN_IF_NO_OVERLAYS(frame.page());
93
94 sharedDebugOverlays->regionChanged(frame, RegionType::WheelEventHandlers);
95 sharedDebugOverlays->regionChanged(frame, RegionType::NonFastScrollableRegion);
96}
97
98inline void DebugPageOverlays::didChangeEventHandlers(Frame& frame)
99{
100 FAST_RETURN_IF_NO_OVERLAYS(frame.page());
101
102 sharedDebugOverlays->regionChanged(frame, RegionType::WheelEventHandlers);
103 sharedDebugOverlays->regionChanged(frame, RegionType::NonFastScrollableRegion);
104}
105
106inline void DebugPageOverlays::doAfterUpdateRendering(Page& page)
107{
108 if (LIKELY(!hasOverlays(page)))
109 return;
110
111 sharedDebugOverlays->updateRegionIfNecessary(page, RegionType::WheelEventHandlers);
112 sharedDebugOverlays->updateRegionIfNecessary(page, RegionType::NonFastScrollableRegion);
113}
