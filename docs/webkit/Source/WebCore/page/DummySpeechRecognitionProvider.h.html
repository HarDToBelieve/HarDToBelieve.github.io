27#pragma once
28
29#include "SpeechRecognitionProvider.h"
30
31namespace WebCore {
32
33class DummySpeechRecognitionProvider final : public SpeechRecognitionProvider {
34public:
35 class DummySpeechRecognitionConnection final : public SpeechRecognitionConnection {
36 public:
37 static Ref<DummySpeechRecognitionConnection> create()
38 {
39 return adoptRef(*new DummySpeechRecognitionConnection());
40 }
41 void registerClient(SpeechRecognitionConnectionClient&) final { }
42 void unregisterClient(SpeechRecognitionConnectionClient&) final { }
43 void start(SpeechRecognitionConnectionClientIdentifier, const String&, bool, bool, uint64_t, ClientOrigin&&, FrameIdentifier) final { }
44 void stop(SpeechRecognitionConnectionClientIdentifier) final { }
45 void abort(SpeechRecognitionConnectionClientIdentifier) final { }
46 void didReceiveUpdate(SpeechRecognitionUpdate&&) final { }
47 };
48 DummySpeechRecognitionProvider() = default;
49 SpeechRecognitionConnection& speechRecognitionConnection()
50 {
51 if (!m_connection)
52 m_connection = DummySpeechRecognitionConnection::create();
53 return *m_connection;
54 }
55private:
56 RefPtr<DummySpeechRecognitionConnection> m_connection;
57};

59} // namespace WebCore
