57inline EventSource::EventSource(ScriptExecutionContext& context, const URL& url, const Init& eventSourceInit)
58 : ActiveDOMObject(&context)

61 , m_decoder(TextResourceDecoder::create("text/plain"_s, "UTF-8"))
62 , m_connectTimer(&context, *this, &EventSource::connect)
63{

67ExceptionOr<Ref<EventSource>> EventSource::create(ScriptExecutionContext& context, const String& url, const Init& eventSourceInit)
68{
69 URL fullURL = context.completeURL(url);
70 if (!fullURL.isValid())
71 return Exception { SyntaxError };
72
73 // FIXME: Convert this to check the isolated world's Content Security Policy once webkit.org/b/104520 is resolved.
74 if (!context.shouldBypassMainWorldContentSecurityPolicy() && !context.contentSecurityPolicy()->allowConnectToSource(fullURL)) {
75 // FIXME: Should this be throwing an exception?
76 return Exception { SecurityError };
77 }
78
79 auto source = adoptRef(*new EventSource(context, fullURL, eventSourceInit));
80 source->scheduleInitialConnect();
81 source->suspendIfNeeded();
82 return source;
83}

91void EventSource::connect()
92{
93 ASSERT(m_state == CONNECTING);
94 ASSERT(!m_requestInFlight);
95
96 ResourceRequest request { m_url };
97 request.setHTTPMethod("GET");
98 request.setHTTPHeaderField(HTTPHeaderName::Accept, "text/event-stream");
99 request.setHTTPHeaderField(HTTPHeaderName::CacheControl, "no-cache");
100 if (!m_lastEventId.isEmpty())
101 request.setHTTPHeaderField(HTTPHeaderName::LastEventID, m_lastEventId);
102
103 ThreadableLoaderOptions options;
104 options.sendLoadCallbacks = SendCallbackPolicy::SendCallbacks;
105 options.credentials = m_withCredentials ? FetchOptions::Credentials::Include : FetchOptions::Credentials::SameOrigin;
106 options.preflightPolicy = PreflightPolicy::Prevent;
107 options.mode = FetchOptions::Mode::Cors;
108 options.cache = FetchOptions::Cache::NoStore;
109 options.dataBufferingPolicy = DataBufferingPolicy::DoNotBufferData;
110 options.contentSecurityPolicyEnforcement = scriptExecutionContext()->shouldBypassMainWorldContentSecurityPolicy() ? ContentSecurityPolicyEnforcement::DoNotEnforce : ContentSecurityPolicyEnforcement::EnforceConnectSrcDirective;
111 options.initiator = cachedResourceRequestInitiators().eventsource;
112
113 ASSERT(scriptExecutionContext());
114 m_loader = ThreadableLoader::create(*scriptExecutionContext(), *this, WTFMove(request), options);
115
116 // FIXME: Can we just use m_loader for this, null it out when it's no longer in flight, and eliminate the m_requestInFlight member?
117 if (m_loader)
118 m_requestInFlight = true;
119}

121void EventSource::networkRequestEnded()
122{
123 ASSERT(m_requestInFlight);
124
125 m_requestInFlight = false;
126
127 if (m_state != CLOSED)
128 scheduleReconnect();

131void EventSource::scheduleInitialConnect()
132{
133 ASSERT(m_state == CONNECTING);
134 ASSERT(!m_requestInFlight);
135
136 m_connectTimer.startOneShot(0_s);
137}
138
139void EventSource::scheduleReconnect()
140{
141 RELEASE_ASSERT_WITH_SECURITY_IMPLICATION(!m_isSuspendedForBackForwardCache);
142 m_state = CONNECTING;
143 m_connectTimer.startOneShot(1_ms * m_reconnectDelay);
144 dispatchErrorEvent();
145}
146
147void EventSource::close()
148{
149 if (m_state == CLOSED) {
150 ASSERT(!m_requestInFlight);
151 return;
152 }
153
154 // Stop trying to connect/reconnect if EventSource was explicitly closed or if ActiveDOMObject::stop() was called.
155 if (m_connectTimer.isActive())
156 m_connectTimer.cancel();
157
158 if (m_requestInFlight)
159 doExplicitLoadCancellation();
160 else
161 m_state = CLOSED;
162}

164bool EventSource::responseIsValid(const ResourceResponse& response) const
165{
166 // Logs to the console as a side effect.
167
168 // To keep the signal-to-noise ratio low, we don't log anything if the status code is not 200.
169 if (response.httpStatusCode() != 200)
170 return false;
171
172 if (!equalLettersIgnoringASCIICase(response.mimeType(), "text/event-stream")) {
173 auto message = makeString("EventSource's response has a MIME type (\"", response.mimeType(), "\") that is not \"text/event-stream\". Aborting the connection.");
174 // FIXME: Console message would be better with a source code location; where would we get that?
175 scriptExecutionContext()->addConsoleMessage(MessageSource::JS, MessageLevel::Error, WTFMove(message));
176 return false;
177 }
178
179 // The specification states we should always decode as UTF-8. If there is a provided charset and it is not UTF-8, then log a warning
180 // message but keep going anyway.
181 auto& charset = response.textEncodingName();
182 if (!charset.isEmpty() && !equalLettersIgnoringASCIICase(charset, "utf-8")) {
183 auto message = makeString("EventSource's response has a charset (\"", charset, "\") that is not UTF-8. The response will be decoded as UTF-8.");
184 // FIXME: Console message would be better with a source code location; where would we get that?
185 scriptExecutionContext()->addConsoleMessage(MessageSource::JS, MessageLevel::Error, WTFMove(message));
186 }
187
188 return true;
189}

191void EventSource::didReceiveResponse(unsigned long, const ResourceResponse& response)
192{
193 ASSERT(m_state == CONNECTING);
194 ASSERT(m_requestInFlight);
195 RELEASE_ASSERT_WITH_SECURITY_IMPLICATION(!m_isSuspendedForBackForwardCache);
196
197 if (!responseIsValid(response)) {
198 doExplicitLoadCancellation();
199 dispatchErrorEvent();
200 return;
201 }
202
203 m_eventStreamOrigin = SecurityOriginData::fromURL(response.url()).toString();
204 m_state = OPEN;
205 dispatchEvent(Event::create(eventNames().openEvent, Event::CanBubble::No, Event::IsCancelable::No));
206}

213void EventSource::didReceiveData(const uint8_t* data, int length)
214{
215 ASSERT(m_state == OPEN);
216 ASSERT(m_requestInFlight);
217 RELEASE_ASSERT_WITH_SECURITY_IMPLICATION(!m_isSuspendedForBackForwardCache);
218
219 append(m_receiveBuffer, m_decoder->decode(data, length));
220 parseEventStream();
221}

223void EventSource::didFinishLoading(unsigned long)
224{
225 ASSERT(m_state == OPEN);
226 ASSERT(m_requestInFlight);
227 RELEASE_ASSERT_WITH_SECURITY_IMPLICATION(!m_isSuspendedForBackForwardCache);
228
229 append(m_receiveBuffer, m_decoder->flush());
230 parseEventStream();
231
232 // Discard everything that has not been dispatched by now.
233 // FIXME: Why does this need to be done?
234 // If this is important, why isn't it important to clear other data members: m_decoder, m_lastEventId, m_loader?
235 m_receiveBuffer.clear();
236 m_data.clear();
237 m_eventName = { };
238 m_currentlyParsedEventId = { };
239
240 networkRequestEnded();
241}

243void EventSource::didFail(const ResourceError& error)
244{
245 ASSERT(m_state != CLOSED);
246
247 if (error.isAccessControl()) {
248 abortConnectionAttempt();
249 return;
250 }
251
252 ASSERT(m_requestInFlight);
253
254 // This is the case where the load gets cancelled on navigating away. We only fire an error event and attempt to reconnect
255 // if we end up getting resumed from back/forward cache.
256 if (error.isCancellation() && !m_isDoingExplicitCancellation) {
257 m_shouldReconnectOnResume = true;
258 m_requestInFlight = false;
259 return;
260 }
261
262 if (error.isCancellation())
263 m_state = CLOSED;
264
265 // FIXME: Why don't we need to clear data members here as in didFinishLoading?
266
267 networkRequestEnded();
268}

270void EventSource::abortConnectionAttempt()
271{
272 ASSERT(m_state == CONNECTING);
273 RELEASE_ASSERT_WITH_SECURITY_IMPLICATION(!m_isSuspendedForBackForwardCache);
274
275 auto jsWrapperProtector = makePendingActivity(*this);
276 if (m_requestInFlight)
277 doExplicitLoadCancellation();
278 else
279 m_state = CLOSED;
280
281 ASSERT(m_state == CLOSED);
282 dispatchEvent(Event::create(eventNames().errorEvent, Event::CanBubble::No, Event::IsCancelable::No));
283}

297void EventSource::parseEventStream()
298{
299 unsigned position = 0;
300 unsigned size = m_receiveBuffer.size();
301 while (position < size) {
302 if (m_discardTrailingNewline) {
303 if (m_receiveBuffer[position] == '\n')
304 ++position;
305 m_discardTrailingNewline = false;
306 }
307
308 std::optional<unsigned> lineLength;
309 std::optional<unsigned> fieldLength;
310 for (unsigned i = position; !lineLength && i < size; ++i) {
311 switch (m_receiveBuffer[i]) {
312 case ':':
313 if (!fieldLength)
314 fieldLength = i - position;
315 break;
316 case '\r':
317 m_discardTrailingNewline = true;
318 FALLTHROUGH;
319 case '\n':
320 lineLength = i - position;
321 break;
322 }
323 }
324
325 if (!lineLength)
326 break;
327
328 parseEventStreamLine(position, fieldLength, lineLength.value());
329 position += lineLength.value() + 1;
330
331 // EventSource.close() might've been called by one of the message event handlers.
332 // Per spec, no further messages should be fired after that.
333 if (m_state == CLOSED)
334 break;
335 }
336
337 // FIXME: The following operation makes it clear that m_receiveBuffer should be some other type,
338 // perhaps a Deque or a circular buffer of some sort.
339 if (position == size)
340 m_receiveBuffer.clear();
341 else if (position)
342 m_receiveBuffer.remove(0, position);
343}

345void EventSource::parseEventStreamLine(unsigned position, std::optional<unsigned> fieldLength, unsigned lineLength)
346{
347 if (!lineLength) {
348 if (!m_data.isEmpty())
349 dispatchMessageEvent();
350 m_eventName = { };
351 return;
352 }
353
354 if (fieldLength && !fieldLength.value())
355 return;
356
357 StringView field { &m_receiveBuffer[position], fieldLength ? fieldLength.value() : lineLength };
358
359 unsigned step;
360 if (!fieldLength)
361 step = lineLength;
362 else if (m_receiveBuffer[position + fieldLength.value() + 1] != ' ')
363 step = fieldLength.value() + 1;
364 else
365 step = fieldLength.value() + 2;
366 position += step;
367 unsigned valueLength = lineLength - step;
368
369 if (field == "data") {
370 m_data.append(&m_receiveBuffer[position], valueLength);
371 m_data.append('\n');
372 } else if (field == "event")
373 m_eventName = { &m_receiveBuffer[position], valueLength };
374 else if (field == "id") {
375 StringView parsedEventId = { &m_receiveBuffer[position], valueLength };
376 constexpr UChar nullCharacter = '\0';
377 if (!parsedEventId.contains(nullCharacter))
378 m_currentlyParsedEventId = parsedEventId.toString();
379 } else if (field == "retry") {
380 if (!valueLength)
381 m_reconnectDelay = defaultReconnectDelay;
382 else {
383 // FIXME: Do we really want to ignore trailing junk here?
384 // FIXME: When we can't parse the value, should we really leave m_reconnectDelay alone? Shouldn't we set it to defaultReconnectDelay?
385 if (auto reconnectDelay = parseIntegerAllowingTrailingJunk<uint64_t>({ &m_receiveBuffer[position], valueLength }))
386 m_reconnectDelay = *reconnectDelay;
387 }
388 }
389}

401void EventSource::suspend(ReasonForSuspension reason)
402{
403 if (reason != ReasonForSuspension::BackForwardCache)
404 return;
405
406 m_isSuspendedForBackForwardCache = true;
407 RELEASE_ASSERT_WITH_MESSAGE(!m_requestInFlight, "Loads get cancelled before entering the BackForwardCache.");
408}
409
410void EventSource::resume()
411{
412 if (!m_isSuspendedForBackForwardCache)
413 return;
414
415 m_isSuspendedForBackForwardCache = false;
416 if (std::exchange(m_shouldReconnectOnResume, false)) {
417 scriptExecutionContext()->postTask([this, pendingActivity = makePendingActivity(*this)](ScriptExecutionContext&) {
418 if (!isContextStopped())
419 scheduleReconnect();
420 });
421 }
422}
423
424void EventSource::dispatchMessageEvent()
425{
426 RELEASE_ASSERT_WITH_SECURITY_IMPLICATION(!m_isSuspendedForBackForwardCache);
427
428 if (!m_currentlyParsedEventId.isNull())
429 m_lastEventId = WTFMove(m_currentlyParsedEventId);
430
431 auto& name = m_eventName.isEmpty() ? eventNames().messageEvent : m_eventName;
432
433 // Omit the trailing "\n" character.
434 ASSERT(!m_data.isEmpty());
435 unsigned size = m_data.size() - 1;
436 auto data = SerializedScriptValue::create({ m_data.data(), size });
437 RELEASE_ASSERT(data);
438 m_data = { };
439
440 dispatchEvent(MessageEvent::create(name, data.releaseNonNull(), m_eventStreamOrigin, m_lastEventId));
441}
