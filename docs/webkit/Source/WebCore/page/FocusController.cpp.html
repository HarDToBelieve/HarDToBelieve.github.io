72static inline bool hasCustomFocusLogic(const Element& element)
73{
74 return is<HTMLElement>(element) && downcast<HTMLElement>(element).hasCustomFocusLogic();
75}
76
77static inline bool isFocusScopeOwner(const Element& element)
78{
79 if (element.shadowRoot() && !hasCustomFocusLogic(element))
80 return true;
81 if (is<HTMLSlotElement>(element)) {
82 ShadowRoot* root = element.containingShadowRoot();
83 if (!root || !root->host() || !hasCustomFocusLogic(*root->host()))
84 return true;
85 }
86 return false;
87}

89class FocusNavigationScope {
90public:
91 Element* owner() const;
92 WEBCORE_EXPORT static FocusNavigationScope scopeOf(Node&);
93 static FocusNavigationScope scopeOwnedByScopeOwner(Element&);
94 static FocusNavigationScope scopeOwnedByIFrame(HTMLFrameOwnerElement&);

96 Node* firstNodeInScope() const;
97 Node* lastNodeInScope() const;
98 Node* nextInScope(const Node*) const;

103 enum class SlotKind : uint8_t { Assigned, Fallback };

105 Node* firstChildInScope(const Node&) const;

107 Node* parentInScope(const Node&) const;
108
109 Node* nextSiblingInScope(const Node&) const;

112 explicit FocusNavigationScope(TreeScope&);
113 explicit FocusNavigationScope(HTMLSlotElement&, SlotKind);

115 RefPtr<ContainerNode> m_treeScopeRootNode;
116 RefPtr<HTMLSlotElement> m_slotElement;
117 SlotKind m_slotKind { SlotKind::Assigned };

120// FIXME: Focus navigation should work with shadow trees that have slots.
121Node* FocusNavigationScope::firstChildInScope(const Node& node) const
122{
123 if (is<Element>(node) && isFocusScopeOwner(downcast<Element>(node)))
124 return nullptr;
125 return node.firstChild();
126}
127
128Node* FocusNavigationScope::lastChildInScope(const Node& node) const
129{
130 if (is<Element>(node) && isFocusScopeOwner(downcast<Element>(node)))
131 return nullptr;
132 return node.lastChild();
133}
134
135Node* FocusNavigationScope::parentInScope(const Node& node) const
136{
137 if (m_treeScopeRootNode == &node)
138 return nullptr;
139
140 if (UNLIKELY(m_slotElement)) {
141 if (m_slotKind == SlotKind::Assigned) {
142 if (m_slotElement == node.assignedSlot())
143 return nullptr;
144 } else {
145 ASSERT(m_slotKind == SlotKind::Fallback);
146 auto* parentNode = node.parentNode();
147 if (parentNode == m_slotElement)
148 return nullptr;
149 }
150 }
151
152 return node.parentNode();
153}
154
155Node* FocusNavigationScope::nextSiblingInScope(const Node& node) const
156{
157 if (UNLIKELY(m_slotElement && m_slotElement == node.assignedSlot())) {
158 for (Node* current = node.nextSibling(); current; current = current->nextSibling()) {
159 if (current->assignedSlot() == m_slotElement)
160 return current;
161 }
162 return nullptr;
163 }
164 return node.nextSibling();
165}
166
167Node* FocusNavigationScope::previousSiblingInScope(const Node& node) const
168{
169 if (UNLIKELY(m_slotElement && m_slotElement == node.assignedSlot())) {
170 for (Node* current = node.previousSibling(); current; current = current->previousSibling()) {
171 if (current->assignedSlot() == m_slotElement)
172 return current;
173 }
174 return nullptr;
175 }
176 return node.previousSibling();
177}
178
179Node* FocusNavigationScope::firstNodeInScope() const
180{
181 if (UNLIKELY(m_slotElement)) {
182 auto* assigneNodes = m_slotElement->assignedNodes();
183 if (m_slotKind == SlotKind::Assigned) {
184 ASSERT(assigneNodes);
185 return assigneNodes->first().get();
186 }
187 ASSERT(m_slotKind == SlotKind::Fallback);
188 return m_slotElement->firstChild();
189 }
190 ASSERT(m_treeScopeRootNode);
191 return m_treeScopeRootNode.get();
192}
193
194Node* FocusNavigationScope::lastNodeInScope() const
195{
196 if (UNLIKELY(m_slotElement)) {
197 auto* assigneNodes = m_slotElement->assignedNodes();
198 if (m_slotKind == SlotKind::Assigned) {
199 ASSERT(assigneNodes);
200 return assigneNodes->last().get();
201 }
202 ASSERT(m_slotKind == SlotKind::Fallback);
203 return m_slotElement->lastChild();
204 }
205 ASSERT(m_treeScopeRootNode);
206 return m_treeScopeRootNode.get();
207}
208
209Node* FocusNavigationScope::nextInScope(const Node* node) const
210{
211 ASSERT(node);
212 if (Node* next = firstChildInScope(*node))
213 return next;
214 if (Node* next = nextSiblingInScope(*node))
215 return next;
216 const Node* current = node;
217 while (current && !nextSiblingInScope(*current))
218 current = parentInScope(*current);
219 return current ? nextSiblingInScope(*current) : nullptr;
220}
221
222Node* FocusNavigationScope::previousInScope(const Node* node) const
223{
224 ASSERT(node);
225 if (Node* current = previousSiblingInScope(*node)) {
226 while (Node* child = lastChildInScope(*current))
227 current = child;
228 return current;
229 }
230 return parentInScope(*node);
231}

244Element* FocusNavigationScope::owner() const
245{
246 if (m_slotElement)
247 return m_slotElement.get();
248
249 ASSERT(m_treeScopeRootNode);
250 if (is<ShadowRoot>(*m_treeScopeRootNode))
251 return downcast<ShadowRoot>(*m_treeScopeRootNode).host();
252 if (Frame* frame = m_treeScopeRootNode->document().frame())
253 return frame->ownerElement();
254 return nullptr;
255}

257FocusNavigationScope FocusNavigationScope::scopeOf(Node& startingNode)
258{
259 ASSERT(startingNode.isInTreeScope());
260 RefPtr<Node> root;
261 RefPtr<Node> parentNode;
262 for (RefPtr<Node> currentNode = &startingNode; currentNode; currentNode = parentNode) {
263 root = currentNode;
264 if (HTMLSlotElement* slot = currentNode->assignedSlot()) {
265 if (isFocusScopeOwner(*slot))
266 return FocusNavigationScope(*slot, SlotKind::Assigned);
267 }
268 if (is<ShadowRoot>(currentNode))
269 return FocusNavigationScope(downcast<ShadowRoot>(*currentNode));
270 parentNode = currentNode->parentNode();
271 // The scope of a fallback content of a HTMLSlotElement is the slot element
272 // but the scope of a HTMLSlotElement is its parent scope.
273 if (parentNode && is<HTMLSlotElement>(parentNode) && !downcast<HTMLSlotElement>(*parentNode).assignedNodes())
274 return FocusNavigationScope(downcast<HTMLSlotElement>(*parentNode), SlotKind::Fallback);
275 }
276 ASSERT(root);
277 return FocusNavigationScope(root->treeScope());
278}

280FocusNavigationScope FocusNavigationScope::scopeOwnedByScopeOwner(Element& element)
281{
282 ASSERT(element.shadowRoot() || is<HTMLSlotElement>(element));
283 if (is<HTMLSlotElement>(element)) {
284 auto& slot = downcast<HTMLSlotElement>(element);
285 return FocusNavigationScope(slot, slot.assignedNodes() ? SlotKind::Assigned : SlotKind::Fallback);
286 }
287 return FocusNavigationScope(*element.shadowRoot());
288}

297static inline void dispatchEventsOnWindowAndFocusedElement(Document* document, bool focused)
298{
299 // If we have a focused node we should dispatch blur on it before we blur the window.
300 // If we have a focused node we should dispatch focus on it after we focus the window.
301 // https://bugs.webkit.org/show_bug.cgi?id=27105
302
303 // Do not fire events while modal dialogs are up. See https://bugs.webkit.org/show_bug.cgi?id=33962
304 if (Page* page = document->page()) {
305 if (page->defersLoading())
306 return;
307 }
308
309 if (!focused && document->focusedElement())
310 document->focusedElement()->dispatchBlurEvent(nullptr);
311 document->dispatchWindowEvent(Event::create(focused ? eventNames().focusEvent : eventNames().blurEvent, Event::CanBubble::No, Event::IsCancelable::No));
312 if (focused && document->focusedElement())
313 document->focusedElement()->dispatchFocusEvent(nullptr, FocusDirection::None);
314}

331static inline int shadowAdjustedTabIndex(Element& element, KeyboardEvent* event)
332{
333 if (isNonFocusableScopeOwner(element, event)) {
334 if (!element.tabIndexSetExplicitly())
335 return 0; // Treat a shadow host without tabindex if it has tabindex=0 even though HTMLElement::tabIndex returns -1 on such an element.
336 }
337 return element.shouldBeIgnoredInSequentialFocusNavigation() ? -1 : element.tabIndexSetExplicitly().value_or(0);
338}

348void FocusController::setFocusedFrame(Frame* frame)
349{
350 ASSERT(!frame || frame->page() == &m_page);
351 if (m_focusedFrame == frame || m_isChangingFocusedFrame)
352 return;
353
354 m_isChangingFocusedFrame = true;
355
356 RefPtr<Frame> oldFrame = m_focusedFrame;
357 RefPtr<Frame> newFrame = frame;
358
359 m_focusedFrame = newFrame;
360
361 // Now that the frame is updated, fire events and update the selection focused states of both frames.
362 if (oldFrame && oldFrame->view()) {
363 oldFrame->selection().setFocused(false);
364 oldFrame->document()->dispatchWindowEvent(Event::create(eventNames().blurEvent, Event::CanBubble::No, Event::IsCancelable::No));
365 }
366
367 if (newFrame && newFrame->view() && isFocused()) {
368 newFrame->selection().setFocused(true);
369 newFrame->document()->dispatchWindowEvent(Event::create(eventNames().focusEvent, Event::CanBubble::No, Event::IsCancelable::No));
370 }
371
372 m_page.chrome().focusedFrameChanged(newFrame.get());
373
374 m_isChangingFocusedFrame = false;
375}

389void FocusController::setFocusedInternal(bool focused)
390{
391 if (!isFocused())

403Element* FocusController::findFocusableElementDescendingIntoSubframes(FocusDirection direction, Element* element, KeyboardEvent* event)
404{
405 // The node we found might be a HTMLFrameOwnerElement, so descend down the tree until we find either:
406 // 1) a focusable node, or
407 // 2) the deepest-nested HTMLFrameOwnerElement.
408 while (is<HTMLFrameOwnerElement>(element)) {
409 HTMLFrameOwnerElement& owner = downcast<HTMLFrameOwnerElement>(*element);
410 if (!owner.contentFrame() || !owner.contentFrame()->document())
411 break;
412 owner.contentFrame()->document()->updateLayoutIgnorePendingStylesheets();
413 Element* foundElement = findFocusableElementWithinScope(direction, FocusNavigationScope::scopeOwnedByIFrame(owner), nullptr, event);
414 if (!foundElement)
415 break;
416 ASSERT(element != foundElement);
417 element = foundElement;
418 }
419 return element;
420}

422bool FocusController::setInitialFocus(FocusDirection direction, KeyboardEvent* providedEvent)
423{
424 bool didAdvanceFocus = advanceFocus(direction, providedEvent, true);
425
426 // If focus is being set initially, accessibility needs to be informed that system focus has moved
427 // into the web area again, even if focus did not change within WebCore. PostNotification is called instead
428 // of handleFocusedUIElementChanged, because this will send the notification even if the element is the same.
429 if (auto* cache = focusedOrMainFrame().document()->existingAXObjectCache())
430 cache->postNotification(focusedOrMainFrame().document(), AXObjectCache::AXFocusedUIElementChanged);
431
432 return didAdvanceFocus;
433}

435bool FocusController::advanceFocus(FocusDirection direction, KeyboardEvent* event, bool initialFocus)
436{
437 switch (direction) {
438 case FocusDirection::Forward:
439 case FocusDirection::Backward:
440 return advanceFocusInDocumentOrder(direction, event, initialFocus);
441 case FocusDirection::Left:
442 case FocusDirection::Right:
443 case FocusDirection::Up:
444 case FocusDirection::Down:
445 return advanceFocusDirectionally(direction, event);
446 default:
447 ASSERT_NOT_REACHED();
448 }
449
450 return false;
451}

453bool FocusController::relinquishFocusToChrome(FocusDirection direction)
454{
455 RefPtr<Document> document = focusedOrMainFrame().document();
456 if (!document)
457 return false;
458
459 if (!m_page.chrome().canTakeFocus(direction) || m_page.isControlledByAutomation())
460 return false;
461
462 document->setFocusedElement(nullptr);
463 setFocusedFrame(nullptr);
464 m_page.chrome().takeFocus(direction);
465 return true;
466}

468bool FocusController::advanceFocusInDocumentOrder(FocusDirection direction, KeyboardEvent* event, bool initialFocus)
469{
470 Frame& frame = focusedOrMainFrame();
471 Document* document = frame.document();
472
473 Node* currentNode = document->focusNavigationStartingNode(direction);
474 // FIXME: Not quite correct when it comes to focus transitions leaving/entering the WebView itself
475 bool caretBrowsing = frame.settings().caretBrowsingEnabled();
476
477 if (caretBrowsing && !currentNode)
478 currentNode = frame.selection().selection().start().deprecatedNode();
479
480 document->updateLayoutIgnorePendingStylesheets();
481
482 RefPtr<Element> element = findFocusableElementAcrossFocusScope(direction, FocusNavigationScope::scopeOf(currentNode ? *currentNode : *document), currentNode, event);
483
484 if (!element) {
485 // We didn't find a node to focus, so we should try to pass focus to Chrome.
486 if (!initialFocus) {
487 if (relinquishFocusToChrome(direction))
488 return true;
489 }
490
491 // Chrome doesn't want focus, so we should wrap focus.
492 element = findFocusableElementAcrossFocusScope(direction, FocusNavigationScope::scopeOf(*m_page.mainFrame().document()), nullptr, event);
493
494 if (!element)
495 return false;
496 }
497
498 ASSERT(element);
499
500 if (element == document->focusedElement()) {
501 // Focus wrapped around to the same element.
502 return true;
503 }
504
505 if (is<HTMLFrameOwnerElement>(*element) && (!is<HTMLPlugInElement>(*element) || !element->isKeyboardFocusable(event))) {
506 // We focus frames rather than frame owners.
507 // FIXME: We should not focus frames that have no scrollbars, as focusing them isn't useful to the user.
508 HTMLFrameOwnerElement& owner = downcast<HTMLFrameOwnerElement>(*element);
509 if (!owner.contentFrame())
510 return false;
511
512 document->setFocusedElement(nullptr);
513 setFocusedFrame(owner.contentFrame());
514 return true;
515 }
516
517 // FIXME: It would be nice to just be able to call setFocusedElement(node) here, but we can't do
518 // that because some elements (e.g. HTMLInputElement and HTMLTextAreaElement) do extra work in
519 // their focus() methods.
520
521 Document& newDocument = element->document();
522
523 if (&newDocument != document) {
524 // Focus is going away from this document, so clear the focused node.
525 document->setFocusedElement(nullptr);
526 }
527
528 setFocusedFrame(newDocument.frame());
529
530 if (caretBrowsing) {
531 VisibleSelection newSelection(firstPositionInOrBeforeNode(element.get()), Affinity::Downstream);
532 if (frame.selection().shouldChangeSelection(newSelection)) {
533 AXTextStateChangeIntent intent(AXTextStateChangeTypeSelectionMove, AXTextSelection { AXTextSelectionDirectionDiscontiguous, AXTextSelectionGranularityUnknown, true });
534 frame.selection().setSelection(newSelection, FrameSelection::defaultSetSelectionOptions(UserTriggered), intent);
535 }
536 }
537
538 element->focus({ SelectionRestorationMode::SelectAll, direction, { }, { }, FocusVisibility::Visible });
539 return true;
540}

542Element* FocusController::findFocusableElementAcrossFocusScope(FocusDirection direction, const FocusNavigationScope& scope, Node* currentNode, KeyboardEvent* event)
543{
544 ASSERT(!is<Element>(currentNode) || !isNonFocusableScopeOwner(downcast<Element>(*currentNode), event));
545
546 if (currentNode && direction == FocusDirection::Forward && is<Element>(currentNode) && isFocusableScopeOwner(downcast<Element>(*currentNode), event)) {
547 if (Element* candidateInInnerScope = findFocusableElementWithinScope(direction, FocusNavigationScope::scopeOwnedByScopeOwner(downcast<Element>(*currentNode)), 0, event))
548 return candidateInInnerScope;
549 }
550
551 if (Element* candidateInCurrentScope = findFocusableElementWithinScope(direction, scope, currentNode, event))
552 return candidateInCurrentScope;
553
554 // If there's no focusable node to advance to, move up the focus scopes until we find one.
555 Element* owner = scope.owner();
556 while (owner) {
557 if (direction == FocusDirection::Backward && isFocusableScopeOwner(*owner, event))
558 return findFocusableElementDescendingIntoSubframes(direction, owner, event);
559
560 auto outerScope = FocusNavigationScope::scopeOf(*owner);
561 if (Element* candidateInOuterScope = findFocusableElementWithinScope(direction, outerScope, owner, event))
562 return candidateInOuterScope;
563 owner = outerScope.owner();
564 }
565 return nullptr;
566}

568Element* FocusController::findFocusableElementWithinScope(FocusDirection direction, const FocusNavigationScope& scope, Node* start, KeyboardEvent* event)
569{
570 // Starting node is exclusive.
571 Element* candidate = direction == FocusDirection::Forward
572 ? nextFocusableElementWithinScope(scope, start, event)
573 : previousFocusableElementWithinScope(scope, start, event);
574 return findFocusableElementDescendingIntoSubframes(direction, candidate, event);
575}

577Element* FocusController::nextFocusableElementWithinScope(const FocusNavigationScope& scope, Node* start, KeyboardEvent* event)
578{
579 Element* found = nextFocusableElementOrScopeOwner(scope, start, event);
580 if (!found)
581 return nullptr;
582 if (isNonFocusableScopeOwner(*found, event)) {
583 if (Element* foundInInnerFocusScope = nextFocusableElementWithinScope(FocusNavigationScope::scopeOwnedByScopeOwner(*found), 0, event))
584 return foundInInnerFocusScope;
585 return nextFocusableElementWithinScope(scope, found, event);
586 }
587 return found;
588}
589
590Element* FocusController::previousFocusableElementWithinScope(const FocusNavigationScope& scope, Node* start, KeyboardEvent* event)
591{
592 Element* found = previousFocusableElementOrScopeOwner(scope, start, event);
593 if (!found)
594 return nullptr;
595 if (isFocusableScopeOwner(*found, event)) {
596 // Search an inner focusable element in the shadow tree from the end.
597 if (Element* foundInInnerFocusScope = previousFocusableElementWithinScope(FocusNavigationScope::scopeOwnedByScopeOwner(*found), 0, event))
598 return foundInInnerFocusScope;
599 return found;
600 }
601 if (isNonFocusableScopeOwner(*found, event)) {
602 if (Element* foundInInnerFocusScope = previousFocusableElementWithinScope(FocusNavigationScope::scopeOwnedByScopeOwner(*found), 0, event))
603 return foundInInnerFocusScope;
604 return previousFocusableElementWithinScope(scope, found, event);
605 }
606 return found;
607}

609Element* FocusController::findFocusableElementOrScopeOwner(FocusDirection direction, const FocusNavigationScope& scope, Node* node, KeyboardEvent* event)
610{
611 return (direction == FocusDirection::Forward)
612 ? nextFocusableElementOrScopeOwner(scope, node, event)
613 : previousFocusableElementOrScopeOwner(scope, node, event);

616Element* FocusController::findElementWithExactTabIndex(const FocusNavigationScope& scope, Node* start, int tabIndex, KeyboardEvent* event, FocusDirection direction)
617{
618 // Search is inclusive of start
619 for (Node* node = start; node; node = direction == FocusDirection::Forward ? scope.nextInScope(node) : scope.previousInScope(node)) {
620 if (!is<Element>(*node))
621 continue;
622 Element& element = downcast<Element>(*node);
623 if (isFocusableElementOrScopeOwner(element, event) && shadowAdjustedTabIndex(element, event) == tabIndex)
624 return &element;
625 }
626 return nullptr;
627}
628
629static Element* nextElementWithGreaterTabIndex(const FocusNavigationScope& scope, int tabIndex, KeyboardEvent* event)
630{
631 // Search is inclusive of start
632 int winningTabIndex = std::numeric_limits<int>::max();
633 Element* winner = nullptr;
634 for (Node* node = scope.firstNodeInScope(); node; node = scope.nextInScope(node)) {
635 if (!is<Element>(*node))
636 continue;
637 Element& candidate = downcast<Element>(*node);
638 int candidateTabIndex = shadowAdjustedTabIndex(candidate, event);
639 if (isFocusableElementOrScopeOwner(candidate, event) && candidateTabIndex > tabIndex && (!winner || candidateTabIndex < winningTabIndex)) {
640 winner = &candidate;
641 winningTabIndex = candidateTabIndex;
642 }
643 }
644
645 return winner;
646}
647
648static Element* previousElementWithLowerTabIndex(const FocusNavigationScope& scope, Node* start, int tabIndex, KeyboardEvent* event)
649{
650 // Search is inclusive of start
651 int winningTabIndex = 0;
652 Element* winner = nullptr;
653 for (Node* node = start; node; node = scope.previousInScope(node)) {
654 if (!is<Element>(*node))
655 continue;
656 Element& element = downcast<Element>(*node);
657 int currentTabIndex = shadowAdjustedTabIndex(element, event);
658 if (isFocusableElementOrScopeOwner(element, event) && currentTabIndex < tabIndex && currentTabIndex > winningTabIndex) {
659 winner = &element;
660 winningTabIndex = currentTabIndex;
661 }
662 }
663 return winner;
664}

680Element* FocusController::nextFocusableElementOrScopeOwner(const FocusNavigationScope& scope, Node* start, KeyboardEvent* event)
681{
682 int startTabIndex = 0;
683 if (start && is<Element>(*start))
684 startTabIndex = shadowAdjustedTabIndex(downcast<Element>(*start), event);
685
686 if (start) {
687 // If a node is excluded from the normal tabbing cycle, the next focusable node is determined by tree order
688 if (startTabIndex < 0) {
689 for (Node* node = scope.nextInScope(start); node; node = scope.nextInScope(node)) {
690 if (!is<Element>(*node))
691 continue;
692 Element& element = downcast<Element>(*node);
693 if (isFocusableElementOrScopeOwner(element, event) && shadowAdjustedTabIndex(element, event) >= 0)
694 return &element;
695 }
696 }
697
698 // First try to find a node with the same tabindex as start that comes after start in the scope.
699 if (Element* winner = findElementWithExactTabIndex(scope, scope.nextInScope(start), startTabIndex, event, FocusDirection::Forward))
700 return winner;
701
702 if (!startTabIndex)
703 return nullptr; // We've reached the last node in the document with a tabindex of 0. This is the end of the tabbing order.
704 }
705
706 // Look for the first Element in the scope that:
707 // 1) has the lowest tabindex that is higher than start's tabindex (or 0, if start is null), and
708 // 2) comes first in the scope, if there's a tie.
709 if (Element* winner = nextElementWithGreaterTabIndex(scope, startTabIndex, event))
710 return winner;
711
712 // There are no nodes with a tabindex greater than start's tabindex,
713 // so find the first node with a tabindex of 0.
714 return findElementWithExactTabIndex(scope, scope.firstNodeInScope(), 0, event, FocusDirection::Forward);
715}

717Element* FocusController::previousFocusableElementOrScopeOwner(const FocusNavigationScope& scope, Node* start, KeyboardEvent* event)
718{
719 Node* last = nullptr;
720 for (Node* node = scope.lastNodeInScope(); node; node = scope.lastChildInScope(*node))
721 last = node;
722 ASSERT(last);
723
724 // First try to find the last node in the scope that comes before start and has the same tabindex as start.
725 // If start is null, find the last node in the scope with a tabindex of 0.
726 Node* startingNode;
727 int startingTabIndex = 0;
728 if (start) {
729 startingNode = scope.previousInScope(start);
730 if (is<Element>(*start))
731 startingTabIndex = shadowAdjustedTabIndex(downcast<Element>(*start), event);
732 } else
733 startingNode = last;
734
735 // However, if a node is excluded from the normal tabbing cycle, the previous focusable node is determined by tree order
736 if (startingTabIndex < 0) {
737 for (Node* node = startingNode; node; node = scope.previousInScope(node)) {
738 if (!is<Element>(*node))
739 continue;
740 Element& element = downcast<Element>(*node);
741 if (isFocusableElementOrScopeOwner(element, event) && shadowAdjustedTabIndex(element, event) >= 0)
742 return &element;
743 }
744 }
745
746 if (Element* winner = findElementWithExactTabIndex(scope, startingNode, startingTabIndex, event, FocusDirection::Backward))
747 return winner;
748
749 // There are no nodes before start with the same tabindex as start, so look for a node that:
750 // 1) has the highest non-zero tabindex (that is less than start's tabindex), and
751 // 2) comes last in the scope, if there's a tie.
752 startingTabIndex = (start && startingTabIndex) ? startingTabIndex : std::numeric_limits<int>::max();
753 return previousElementWithLowerTabIndex(scope, last, startingTabIndex, event);
754}

768static void clearSelectionIfNeeded(Frame* oldFocusedFrame, Frame* newFocusedFrame, Node* newFocusedNode)
769{
770 if (!oldFocusedFrame || !newFocusedFrame)
771 return;
772
773 if (oldFocusedFrame->document() != newFocusedFrame->document())
774 return;
775
776 const VisibleSelection& selection = oldFocusedFrame->selection().selection();
777 if (selection.isNone())
778 return;
779
780 bool caretBrowsing = oldFocusedFrame->settings().caretBrowsingEnabled();
781 if (caretBrowsing)
782 return;
783
784 if (newFocusedNode) {
785 Node* selectionStartNode = selection.start().deprecatedNode();
786 if (newFocusedNode->contains(selectionStartNode) || selectionStartNode->shadowHost() == newFocusedNode)
787 return;
788 }
789
790 if (Node* mousePressNode = newFocusedFrame->eventHandler().mousePressNode()) {
791 if (!mousePressNode->canStartSelection()) {
792 // Don't clear the selection for contentEditable elements, but do clear it for input and textarea. See bug 38696.
793 auto* root = selection.rootEditableElement();
794 if (!root)
795 return;
796 auto* host = root->shadowHost();
797 // FIXME: Seems likely we can just do the check on "host" here instead of "rootOrHost".
798 auto* rootOrHost = host ? host : root;
799 if (!is<HTMLInputElement>(*rootOrHost) && !is<HTMLTextAreaElement>(*rootOrHost))
800 return;
801 }
802 }
803
804 oldFocusedFrame->selection().clear();
805}

830bool FocusController::setFocusedElement(Element* element, Frame& newFocusedFrame, const FocusOptions& options)
831{
832 Ref<Frame> protectedNewFocusedFrame = newFocusedFrame;
833 RefPtr<Frame> oldFocusedFrame = focusedFrame();
834 RefPtr<Document> oldDocument = oldFocusedFrame ? oldFocusedFrame->document() : nullptr;
835
836 Element* oldFocusedElement = oldDocument ? oldDocument->focusedElement() : nullptr;
837 if (oldFocusedElement == element) {
838 if (element)
839 m_page.chrome().client().elementDidRefocus(*element);
840 return true;
841 }
842
843 // FIXME: Might want to disable this check for caretBrowsing
844 if (oldFocusedElement && oldFocusedElement->isRootEditableElement() && !relinquishesEditingFocus(*oldFocusedElement))
845 return false;
846
847 m_page.editorClient().willSetInputMethodState();
848
849 if (shouldClearSelectionWhenChangingFocusedElement(m_page, oldFocusedElement, element))
850 clearSelectionIfNeeded(oldFocusedFrame.get(), &newFocusedFrame, element);
851
852 if (!element) {
853 if (oldDocument)
854 oldDocument->setFocusedElement(nullptr);
855 m_page.editorClient().setInputMethodState(nullptr);
856 return true;
857 }
858
859 Ref<Document> newDocument(element->document());
860
861 if (newDocument->focusedElement() == element) {
862 m_page.editorClient().setInputMethodState(element);
863 return true;
864 }
865
866 if (oldDocument && oldDocument != newDocument.ptr())
867 oldDocument->setFocusedElement(nullptr);
868
869 if (!newFocusedFrame.page()) {
870 setFocusedFrame(nullptr);
871 return false;
872 }
873 setFocusedFrame(&newFocusedFrame);
874
875 Ref<Element> protect(*element);
876
877 bool successfullyFocused = newDocument->setFocusedElement(element, options);
878 if (!successfullyFocused)
879 return false;
880
881 if (newDocument->focusedElement() == element)
882 m_page.editorClient().setInputMethodState(element);
883
884 m_focusSetTime = MonotonicTime::now();
885 m_focusRepaintTimer.stop();
886
887 return true;
888}

957static void updateFocusCandidateIfNeeded(FocusDirection direction, const FocusCandidate& current, FocusCandidate& candidate, FocusCandidate& closest)
958{
959 ASSERT(candidate.visibleNode->isElementNode());
960 ASSERT(candidate.visibleNode->renderer());
961
962 // Ignore iframes that don't have a src attribute
963 if (frameOwnerElement(candidate) && (!frameOwnerElement(candidate)->contentFrame() || candidate.rect.isEmpty()))
964 return;
965
966 // Ignore off screen child nodes of containers that do not scroll (overflow:hidden)
967 if (candidate.isOffscreen && !canBeScrolledIntoView(direction, candidate))
968 return;
969
970 distanceDataForNode(direction, current, candidate);
971 if (candidate.distance == maxDistance())
972 return;
973
974 if (candidate.isOffscreenAfterScrolling && candidate.alignment < Full)
975 return;
976
977 if (closest.isNull()) {
978 closest = candidate;
979 return;
980 }
981
982 LayoutRect intersectionRect = intersection(candidate.rect, closest.rect);
983 if (!intersectionRect.isEmpty() && !areElementsOnSameLine(closest, candidate)) {
984 // If 2 nodes are intersecting, do hit test to find which node in on top.
985 auto center = flooredIntPoint(intersectionRect.center()); // FIXME: Would roundedIntPoint be better?
986 constexpr OptionSet<HitTestRequest::Type> hitType { HitTestRequest::Type::ReadOnly, HitTestRequest::Type::Active, HitTestRequest::Type::IgnoreClipping, HitTestRequest::Type::DisallowUserAgentShadowContent, HitTestRequest::Type::AllowChildFrameContent };
987 HitTestResult result = candidate.visibleNode->document().page()->mainFrame().eventHandler().hitTestResultAtPoint(center, hitType);
988 if (candidate.visibleNode->contains(result.innerNode())) {
989 closest = candidate;
990 return;
991 }
992 if (closest.visibleNode->contains(result.innerNode()))
993 return;
994 }
995
996 if (candidate.alignment == closest.alignment) {
997 if (candidate.distance < closest.distance)
998 closest = candidate;
999 return;
1000 }
1001
1002 if (candidate.alignment > closest.alignment)
1003 closest = candidate;
1004}

1006void FocusController::findFocusCandidateInContainer(Node& container, const LayoutRect& startingRect, FocusDirection direction, KeyboardEvent* event, FocusCandidate& closest)
1007{
1008 Node* focusedNode = (focusedFrame() && focusedFrame()->document()) ? focusedFrame()->document()->focusedElement() : 0;
1009
1010 Element* element = ElementTraversal::firstWithin(container);
1011 FocusCandidate current;
1012 current.rect = startingRect;
1013 current.focusableNode = focusedNode;
1014 current.visibleNode = focusedNode;
1015
1016 unsigned candidateCount = 0;
1017 for (; element; element = (element->isFrameOwnerElement() || canScrollInDirection(element, direction))
1018 ? ElementTraversal::nextSkippingChildren(*element, &container)
1019 : ElementTraversal::next(*element, &container)) {
1020 if (element == focusedNode)
1021 continue;
1022
1023 if (!element->isKeyboardFocusable(event) && !element->isFrameOwnerElement() && !canScrollInDirection(element, direction))
1024 continue;
1025
1026 FocusCandidate candidate = FocusCandidate(element, direction);
1027 if (candidate.isNull())
1028 continue;
1029
1030 if (!isValidCandidate(direction, current, candidate))
1031 continue;
1032
1033 candidateCount++;
1034 candidate.enclosingScrollableBox = &container;
1035 updateFocusCandidateIfNeeded(direction, current, candidate, closest);
1036 }
1037
1038 // The variable 'candidateCount' keeps track of the number of nodes traversed in a given container.
1039 // If we have more than one container in a page then the total number of nodes traversed is equal to the sum of nodes traversed in each container.
1040 if (focusedFrame() && focusedFrame()->document()) {
1041 candidateCount += focusedFrame()->document()->page()->lastSpatialNavigationCandidateCount();
1042 focusedFrame()->document()->page()->setLastSpatialNavigationCandidateCount(candidateCount);
1043 }
1044}

1046bool FocusController::advanceFocusDirectionallyInContainer(Node* container, const LayoutRect& startingRect, FocusDirection direction, KeyboardEvent* event)
1047{
1048 if (!container)
1049 return false;
1050
1051 LayoutRect newStartingRect = startingRect;
1052
1053 if (startingRect.isEmpty())
1054 newStartingRect = virtualRectForDirection(direction, nodeRectInAbsoluteCoordinates(container));
1055
1056 // Find the closest node within current container in the direction of the navigation.
1057 FocusCandidate focusCandidate;
1058 findFocusCandidateInContainer(*container, newStartingRect, direction, event, focusCandidate);
1059
1060 if (focusCandidate.isNull()) {
1061 // Nothing to focus, scroll if possible.
1062 // NOTE: If no scrolling is performed (i.e. scrollInDirection returns false), the
1063 // spatial navigation algorithm will skip this container.
1064 return scrollInDirection(container, direction);
1065 }
1066
1067 if (HTMLFrameOwnerElement* frameElement = frameOwnerElement(focusCandidate)) {
1068 // If we have an iframe without the src attribute, it will not have a contentFrame().
1069 // We ASSERT here to make sure that
1070 // updateFocusCandidateIfNeeded() will never consider such an iframe as a candidate.
1071 ASSERT(frameElement->contentFrame());
1072
1073 if (focusCandidate.isOffscreenAfterScrolling) {
1074 scrollInDirection(&focusCandidate.visibleNode->document(), direction);
1075 return true;
1076 }
1077 // Navigate into a new frame.
1078 LayoutRect rect;
1079 Element* focusedElement = focusedOrMainFrame().document()->focusedElement();
1080 if (focusedElement && !hasOffscreenRect(focusedElement))
1081 rect = nodeRectInAbsoluteCoordinates(focusedElement, true /* ignore border */);
1082 frameElement->contentFrame()->document()->updateLayoutIgnorePendingStylesheets();
1083 if (!advanceFocusDirectionallyInContainer(frameElement->contentFrame()->document(), rect, direction, event)) {
1084 // The new frame had nothing interesting, need to find another candidate.
1085 return advanceFocusDirectionallyInContainer(container, nodeRectInAbsoluteCoordinates(focusCandidate.visibleNode, true), direction, event);
1086 }
1087 return true;
1088 }
1089
1090 if (canScrollInDirection(focusCandidate.visibleNode, direction)) {
1091 if (focusCandidate.isOffscreenAfterScrolling) {
1092 scrollInDirection(focusCandidate.visibleNode, direction);
1093 return true;
1094 }
1095 // Navigate into a new scrollable container.
1096 LayoutRect startingRect;
1097 Element* focusedElement = focusedOrMainFrame().document()->focusedElement();
1098 if (focusedElement && !hasOffscreenRect(focusedElement))
1099 startingRect = nodeRectInAbsoluteCoordinates(focusedElement, true);
1100 return advanceFocusDirectionallyInContainer(focusCandidate.visibleNode, startingRect, direction, event);
1101 }
1102 if (focusCandidate.isOffscreenAfterScrolling) {
1103 Node* container = focusCandidate.enclosingScrollableBox;
1104 scrollInDirection(container, direction);
1105 return true;
1106 }
1107
1108 // We found a new focus node, navigate to it.
1109 Element* element = downcast<Element>(focusCandidate.focusableNode);
1110 ASSERT(element);
1111
1112 element->focus({ SelectionRestorationMode::SelectAll, direction });
1113 return true;
1114}

1116bool FocusController::advanceFocusDirectionally(FocusDirection direction, KeyboardEvent* event)
1117{
1118 Document* focusedDocument = focusedOrMainFrame().document();
1119 if (!focusedDocument)
1120 return false;
1121
1122 focusedDocument->updateLayoutIgnorePendingStylesheets();
1123
1124 // Figure out the starting rect.
1125 Node* container = focusedDocument;
1126 LayoutRect startingRect;
1127 if (auto* focusedElement = focusedDocument->focusedElement()) {
1128 if (!hasOffscreenRect(focusedElement)) {
1129 container = scrollableEnclosingBoxOrParentFrameForNodeInDirection(direction, focusedElement);
1130 startingRect = nodeRectInAbsoluteCoordinates(focusedElement, true /* ignore border */);
1131 } else if (is<HTMLAreaElement>(*focusedElement)) {
1132 HTMLAreaElement& area = downcast<HTMLAreaElement>(*focusedElement);
1133 container = scrollableEnclosingBoxOrParentFrameForNodeInDirection(direction, area.imageElement());
1134 startingRect = virtualRectForAreaElementAndDirection(&area, direction);
1135 }
1136 }
1137
1138 if (focusedFrame() && focusedFrame()->document())
1139 focusedDocument->page()->setLastSpatialNavigationCandidateCount(0);
1140
1141 bool consumed = false;
1142 do {
1143 consumed = advanceFocusDirectionallyInContainer(container, startingRect, direction, event);
1144 focusedDocument->updateLayoutIgnorePendingStylesheets();
1145 startingRect = nodeRectInAbsoluteCoordinates(container, true /* ignore border */);
1146 container = scrollableEnclosingBoxOrParentFrameForNodeInDirection(direction, container);
1147 } while (!consumed && container);
1148
1149 return consumed;
1150}
