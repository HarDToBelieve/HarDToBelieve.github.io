26#pragma once
27
28#include "FrameIdentifier.h"
29#include "PageIdentifier.h"
30
31namespace WebCore {
32
33// Frame identifier that is unique across all WebContent processes.
34struct GlobalFrameIdentifier {
35 PageIdentifier pageID;
36 FrameIdentifier frameID;
37
38 template<class Encoder> void encode(Encoder&) const;
39 template<class Decoder> static std::optional<GlobalFrameIdentifier> decode(Decoder&);
40};
41
42template<class Encoder>
43void GlobalFrameIdentifier::encode(Encoder& encoder) const
44{
45 encoder << pageID << frameID;
46}
47
48template<class Decoder>
49std::optional<GlobalFrameIdentifier> GlobalFrameIdentifier::decode(Decoder& decoder)
50{
51 std::optional<PageIdentifier> pageID;
52 decoder >> pageID;
53 if (!pageID)
54 return std::nullopt;
55
56 std::optional<FrameIdentifier> frameID;
57 decoder >> frameID;
58 if (!frameID)
59 return std::nullopt;
60
61 return { { WTFMove(*pageID), WTFMove(*frameID) } };
62}

64} // namespace WebCore
