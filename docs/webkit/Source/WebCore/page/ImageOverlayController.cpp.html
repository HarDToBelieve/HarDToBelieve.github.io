56void ImageOverlayController::selectionQuadsDidChange(Frame& frame, const Vector<FloatQuad>& quads)
57{
58 if (!m_page || !m_page->chrome().client().needsImageOverlayControllerForSelectionPainting())
59 return;
60
61 if (frame.editor().ignoreSelectionChanges() || frame.editor().isGettingDictionaryPopupInfo())
62 return;
63
64 m_hostElementForSelection = nullptr;
65 m_selectionQuads.clear();
66 m_selectionBackgroundColor = Color::transparentBlack;
67 m_selectionClipRect = { };
68
69 auto overlayHost = ([&] () -> RefPtr<HTMLElement> {
70 auto selectedRange = frame.selection().selection().range();
71 if (!selectedRange)
72 return nullptr;
73
74 if (!HTMLElement::isInsideImageOverlay(*selectedRange))
75 return nullptr;
76
77 if (auto host = makeRefPtr(selectedRange->startContainer().shadowHost()); is<HTMLElement>(host))
78 return makeRefPtr(downcast<HTMLElement>(*host));
79
80 return nullptr;
81 })();
82
83 if (!overlayHost) {
84 uninstallPageOverlayIfNeeded();
85 return;
86 }
87
88 auto overlayHostRenderer = overlayHost->renderer();
89 if (!overlayHostRenderer) {
90 uninstallPageOverlayIfNeeded();
91 return;
92 }
93
94 if (!shouldUsePageOverlayToPaintSelection(*overlayHostRenderer)) {
95 uninstallPageOverlayIfNeeded();
96 return;
97 }
98
99 m_hostElementForSelection = makeWeakPtr(*overlayHost);
100 m_selectionQuads = quads;
101 m_selectionBackgroundColor = overlayHostRenderer->selectionBackgroundColor();
102 m_selectionClipRect = overlayHostRenderer->absoluteBoundingBoxRect();
103
104 installPageOverlayIfNeeded().setNeedsDisplay();
105}
106
107bool ImageOverlayController::shouldUsePageOverlayToPaintSelection(const RenderElement& renderer)
108{
109 // If the selection is already painted (with nonzero opacity) in the overlay host's renderer,
110 // then we don't need to fall back to a page overlay to paint the selection.
111 return renderer.style().opacity() <= 0.01;
112}

114void ImageOverlayController::documentDetached(const Document& document)
115{
116 if (m_hostElementForSelection && &document == &m_hostElementForSelection->document())
117 m_hostElementForSelection = nullptr;
118
119#if PLATFORM(MAC)
120 if (m_hostElementForDataDetectors && &document == &m_hostElementForDataDetectors->document())
121 m_hostElementForDataDetectors = nullptr;
122#endif
123
124 uninstallPageOverlayIfNeeded();
125}
126
127PageOverlay& ImageOverlayController::installPageOverlayIfNeeded()
128{
129 if (m_overlay)
130 return *m_overlay;
131
132 m_overlay = PageOverlay::create(*this, PageOverlay::OverlayType::Document);
133 m_page->pageOverlayController().installPageOverlay(*m_overlay, PageOverlay::FadeMode::DoNotFade);
134 return *m_overlay;
135}
136
137void ImageOverlayController::uninstallPageOverlay()
138{
139 m_hostElementForSelection = nullptr;
140 m_selectionQuads.clear();
141 m_selectionBackgroundColor = Color::transparentBlack;
142 m_selectionClipRect = { };
143
144#if PLATFORM(MAC)
145 clearDataDetectorHighlights();
146#endif
147
148 auto overlayToUninstall = std::exchange(m_overlay, nullptr);
149 if (!m_page || !overlayToUninstall)
150 return;
151
152 m_page->pageOverlayController().uninstallPageOverlay(*overlayToUninstall, PageOverlay::FadeMode::DoNotFade);
153}
154
155void ImageOverlayController::uninstallPageOverlayIfNeeded()
156{
157 if (m_hostElementForSelection)
158 return;
159
160#if PLATFORM(MAC)
161 if (m_hostElementForDataDetectors)
162 return;
163#endif
164
165 uninstallPageOverlay();
166}

168void ImageOverlayController::willMoveToPage(PageOverlay&, Page* page)

170 if (!page)
171 uninstallPageOverlay();

174void ImageOverlayController::drawRect(PageOverlay& pageOverlay, GraphicsContext& context, const IntRect& dirtyRect)
175{
176 if (&pageOverlay != m_overlay) {
177 ASSERT_NOT_REACHED();
178 return;
179 }
180
181 GraphicsContextStateSaver stateSaver(context);
182 context.clearRect(dirtyRect);
183
184 if (m_selectionQuads.isEmpty())
185 return;
186
187 Path coalescedSelectionPath;
188 for (auto& quad : m_selectionQuads) {
189 coalescedSelectionPath.moveTo(quad.p1());
190 coalescedSelectionPath.addLineTo(quad.p2());
191 coalescedSelectionPath.addLineTo(quad.p3());
192 coalescedSelectionPath.addLineTo(quad.p4());
193 coalescedSelectionPath.addLineTo(quad.p1());
194 coalescedSelectionPath.closeSubpath();
195 }
196
197 context.setFillColor(m_selectionBackgroundColor);
198 context.clip(m_selectionClipRect);
199 context.fillPath(coalescedSelectionPath);
200}
