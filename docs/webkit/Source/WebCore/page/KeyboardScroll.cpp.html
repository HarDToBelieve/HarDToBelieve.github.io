31FloatSize unitVectorForScrollDirection(ScrollDirection direction)
32{
33 switch (direction) {
34 case ScrollDirection::ScrollUp:
35 return { 0, -1 };
36 case ScrollDirection::ScrollDown:
37 return { 0, 1 };
38 case ScrollDirection::ScrollLeft:
39 return { -1, 0 };
40 case ScrollDirection::ScrollRight:
41 return { 1, 0 };
42 }
43
44 RELEASE_ASSERT_NOT_REACHED();
45}
