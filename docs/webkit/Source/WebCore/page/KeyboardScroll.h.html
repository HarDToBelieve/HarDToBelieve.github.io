32WEBCORE_EXPORT FloatSize unitVectorForScrollDirection(ScrollDirection);

34struct KeyboardScroll {
35 FloatSize offset; // Points per increment.
36 FloatSize maximumVelocity; // Points per second.
37 FloatSize force;
38
39 ScrollGranularity granularity;
40 ScrollDirection direction;
41};

43struct KeyboardScrollParameters {
44 float springMass { 1 };
45 float springStiffness { 109 };
46 float springDamping { 20 };
47
48 float maximumVelocityMultiplier { 25 };
49 float timeToMaximumVelocity { 1 };
50
51 float rubberBandForce { 5000 };
52
53 static const KeyboardScrollParameters& parameters()
54 {
55 static const KeyboardScrollParameters parameters;
56 return parameters;
57 }
58};
