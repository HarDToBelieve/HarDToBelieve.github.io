30
31#include "DOMStringList.h"
32#include "DOMWindowProperty.h"
33#include "ExceptionOr.h"
34#include "ScriptWrappable.h"
35
36namespace WebCore {
37
38class DOMWindow;
39
40class Location final : public ScriptWrappable, public RefCounted<Location>, public DOMWindowProperty {
41 WTF_MAKE_ISO_ALLOCATED(Location);
42public:
43 static Ref<Location> create(DOMWindow& window) { return adoptRef(*new Location(window)); }
44
45 ExceptionOr<void> setHref(DOMWindow& activeWindow, DOMWindow& firstWindow, const String&);
46 String href() const;
47
48 ExceptionOr<void> assign(DOMWindow& activeWindow, DOMWindow& firstWindow, const String&);
49 ExceptionOr<void> replace(DOMWindow& activeWindow, DOMWindow& firstWindow, const String&);
50 void reload(DOMWindow& activeWindow);
51
52 ExceptionOr<void> setProtocol(DOMWindow& activeWindow, DOMWindow& firstWindow, const String&);
53 String protocol() const;
54 ExceptionOr<void> setHost(DOMWindow& activeWindow, DOMWindow& firstWindow, const String&);
55 WEBCORE_EXPORT String host() const;
56 ExceptionOr<void> setHostname(DOMWindow& activeWindow, DOMWindow& firstWindow, const String&);
57 String hostname() const;
58 ExceptionOr<void> setPort(DOMWindow& activeWindow, DOMWindow& firstWindow, const String&);
59 String port() const;
60 ExceptionOr<void> setPathname(DOMWindow& activeWindow, DOMWindow& firstWindow, const String&);
61 String pathname() const;
62 ExceptionOr<void> setSearch(DOMWindow& activeWindow, DOMWindow& firstWindow, const String&);
63 String search() const;
64 ExceptionOr<void> setHash(DOMWindow& activeWindow, DOMWindow& firstWindow, const String&);
65 String hash() const;
66 String origin() const;
67
68 String toString() const { return href(); }
69
70 Ref<DOMStringList> ancestorOrigins() const;
71
72private:
73 explicit Location(DOMWindow&);
74
75 ExceptionOr<void> setLocation(DOMWindow& activeWindow, DOMWindow& firstWindow, const String&);
76
77 const URL& url() const;
78};
79
80} // namespace WebCore
