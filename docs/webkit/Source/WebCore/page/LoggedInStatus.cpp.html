39ExceptionOr<UniqueRef<LoggedInStatus>> LoggedInStatus::create(const RegistrableDomain& domain, const String& username, CredentialTokenType tokenType, AuthenticationType authType)
40{
41 return create(domain, username, tokenType, authType, authType == AuthenticationType::Unmanaged ? TimeToLiveShort : TimeToLiveLong);

43
44ExceptionOr<UniqueRef<LoggedInStatus>> LoggedInStatus::create(const RegistrableDomain& domain, const String& username, CredentialTokenType tokenType, AuthenticationType authType, Seconds timeToLive)
45{
46 if (domain.isEmpty())
47 return Exception { SecurityError, "IsLoggedIn status can only be set for origins with a registrable domain." };
48
49 if (username.isEmpty())
50 return Exception { SyntaxError, "IsLoggedIn requires a non-empty username." };
51
52 unsigned length = username.length();
53 if (length > UsernameMaxLength)
54 return Exception { SyntaxError, makeString("IsLoggedIn usernames cannot be longer than ", UsernameMaxLength) };
55
56 auto spaceOrNewline = username.find([](UChar ch) {
57 return isSpaceOrNewline(ch);
58 });
59 if (spaceOrNewline != notFound)
60 return Exception { InvalidCharacterError, "IsLoggedIn usernames cannot contain whitespace or newlines." };
61
62 return makeUniqueRef<LoggedInStatus>(*new LoggedInStatus(domain, username, tokenType, authType, timeToLive));
63}
64
65LoggedInStatus::LoggedInStatus(const RegistrableDomain& domain, const String& username, CredentialTokenType tokenType, AuthenticationType authType, Seconds timeToLive)
66 : m_domain { domain }
67 , m_username { username }
68 , m_tokenType { tokenType }
69 , m_authType { authType }
70 , m_loggedInTime { WallTime::now() }
71{
72 setTimeToLive(timeToLive);
73}
74
75void LoggedInStatus::setTimeToLive(Seconds timeToLive)
76{
77 m_timeToLive = std::min(timeToLive, m_authType == AuthenticationType::Unmanaged ? TimeToLiveShort : TimeToLiveLong);
78}
79
80bool LoggedInStatus::hasExpired() const
81{
82 ASSERT(!m_domain.isEmpty() && !m_username.isEmpty());
83 return WallTime::now() > m_loggedInTime + m_timeToLive;
84}
85
86WallTime LoggedInStatus::expiry() const
87{
88 return WallTime::now() + m_timeToLive;
89}
