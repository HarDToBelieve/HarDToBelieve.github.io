32namespace WebCore {
33
34class MediaProducer : public CanMakeWeakPtr<MediaProducer> {
35public:
36 enum class MediaState {
37 IsPlayingAudio = 1 << 0,
38 IsPlayingVideo = 1 << 1,
39 IsPlayingToExternalDevice = 1 << 2,
40 RequiresPlaybackTargetMonitoring = 1 << 3,
41 ExternalDeviceAutoPlayCandidate = 1 << 4,
42 DidPlayToEnd = 1 << 5,
43 IsSourceElementPlaying = 1 << 6,
44 IsNextTrackControlEnabled = 1 << 7,
45 IsPreviousTrackControlEnabled = 1 << 8,
46 HasPlaybackTargetAvailabilityListener = 1 << 9,
47 HasAudioOrVideo = 1 << 10,
48 HasActiveAudioCaptureDevice = 1 << 11,
49 HasActiveVideoCaptureDevice = 1 << 12,
50 HasMutedAudioCaptureDevice = 1 << 13,
51 HasMutedVideoCaptureDevice = 1 << 14,
52 HasInterruptedAudioCaptureDevice = 1 << 15,
53 HasInterruptedVideoCaptureDevice = 1 << 16,
54 HasUserInteractedWithMediaElement = 1 << 17,
55 HasActiveDisplayCaptureDevice = 1 << 18,
56 HasMutedDisplayCaptureDevice = 1 << 19,
57 HasInterruptedDisplayCaptureDevice = 1 << 20,
58 };
59 using MediaStateFlags = OptionSet<MediaState>;
60 static constexpr MediaStateFlags IsNotPlaying = { };
61 static constexpr MediaStateFlags AudioCaptureMask = { MediaState::HasActiveAudioCaptureDevice, MediaState::HasMutedAudioCaptureDevice, MediaState::HasInterruptedAudioCaptureDevice };
62 static constexpr MediaStateFlags VideoCaptureMask = { MediaState::HasActiveVideoCaptureDevice, MediaState::HasMutedVideoCaptureDevice, MediaState::HasInterruptedVideoCaptureDevice };
63 static constexpr MediaStateFlags DisplayCaptureMask = { MediaState::HasActiveDisplayCaptureDevice, MediaState::HasMutedDisplayCaptureDevice, MediaState::HasInterruptedDisplayCaptureDevice };
64 static constexpr MediaStateFlags ActiveCaptureMask = { MediaState::HasActiveAudioCaptureDevice, MediaState::HasActiveVideoCaptureDevice, MediaState::HasActiveDisplayCaptureDevice };
65 static constexpr MediaStateFlags MutedCaptureMask = { MediaState::HasMutedAudioCaptureDevice, MediaState::HasMutedVideoCaptureDevice, MediaState::HasMutedDisplayCaptureDevice };
66 static constexpr MediaStateFlags MediaCaptureMask = { MediaState::HasActiveAudioCaptureDevice, MediaState::HasMutedAudioCaptureDevice, MediaState::HasInterruptedAudioCaptureDevice, MediaState::HasActiveVideoCaptureDevice, MediaState::HasMutedVideoCaptureDevice, MediaState::HasInterruptedVideoCaptureDevice, MediaState::HasActiveDisplayCaptureDevice, MediaState::HasMutedDisplayCaptureDevice, MediaState::HasInterruptedDisplayCaptureDevice };
67
68 static bool isCapturing(MediaStateFlags state) { return state.containsAny(ActiveCaptureMask) || state.containsAny(MutedCaptureMask); }
69
70 virtual MediaStateFlags mediaState() const = 0;
71
72 enum class MediaCaptureKind {
73 Audio,
74 Video,
75 AudioVideo
76 };
77
78 enum class MutedState {
79 AudioIsMuted = 1 << 0,
80 AudioCaptureIsMuted = 1 << 1,
81 VideoCaptureIsMuted = 1 << 2,
82 ScreenCaptureIsMuted = 1 << 3,
83 };
84
85 using MutedStateFlags = OptionSet<MutedState>;
86 static constexpr MutedStateFlags AudioAndVideoCaptureIsMuted = { MutedState::AudioCaptureIsMuted, MutedState::VideoCaptureIsMuted };
87 static constexpr MutedStateFlags MediaStreamCaptureIsMuted = { MutedState::AudioCaptureIsMuted, MutedState::VideoCaptureIsMuted, MutedState::ScreenCaptureIsMuted };
88
89 virtual void pageMutedStateDidChange() = 0;
90
91protected:
92 virtual ~MediaProducer() = default;
93};
94
95} // namespace WebCore
