38OriginAccessEntry::OriginAccessEntry(const String& protocol, const String& host, SubdomainSetting subdomainSetting, IPAddressSetting ipAddressSetting)
39 : m_protocol(protocol.convertToASCIILowercase())
40 , m_host(host.convertToASCIILowercase())
41 , m_subdomainSettings(subdomainSetting)
42 , m_ipAddressSettings(ipAddressSetting)
43 , m_hostIsIPAddress(URL::hostIsIPAddress(m_host))
44{
45 ASSERT(subdomainSetting == AllowSubdomains || subdomainSetting == DisallowSubdomains);
46}
47
48bool OriginAccessEntry::matchesOrigin(const SecurityOrigin& origin) const
49{
50 ASSERT(origin.host() == origin.host().convertToASCIILowercase());
51 ASSERT(origin.protocol() == origin.protocol().convertToASCIILowercase());
52
53 if (m_protocol != origin.protocol())
54 return false;
55
56 // Special case: Include subdomains and empty host means "all hosts, including ip addresses".
57 if (m_subdomainSettings == AllowSubdomains && m_host.isEmpty())
58 return true;
59
60 // Exact match.
61 if (m_host == origin.host())
62 return true;
63
64 // Otherwise we can only match if we're matching subdomains.
65 if (m_subdomainSettings == DisallowSubdomains)
66 return false;
67
68 // IP addresses are not domains: https://url.spec.whatwg.org/#concept-domain
69 // Don't try to do subdomain matching on IP addresses.
70 if (m_ipAddressSettings == TreatIPAddressAsIPAddress && (m_hostIsIPAddress || URL::hostIsIPAddress(origin.host())))
71 return false;
72
73 // Match subdomains.
74 if (origin.host().length() > m_host.length() && origin.host()[origin.host().length() - m_host.length() - 1] == '.' && origin.host().endsWith(m_host))
75 return true;
76
77 return false;
78}
