31#pragma once
32
33#include <wtf/text/WTFString.h>
34
35namespace WebCore {
36
37class SecurityOrigin;
38
39class OriginAccessEntry {
40public:
41 enum SubdomainSetting {
42 AllowSubdomains,
43 DisallowSubdomains
44 };
45
46 enum IPAddressSetting {
47 TreatIPAddressAsDomain,
48 TreatIPAddressAsIPAddress
49 };
50
51 // If host is empty string and SubdomainSetting is AllowSubdomains, the entry will match all domains in the specified protocol.
52 OriginAccessEntry(const String& protocol, const String& host, SubdomainSetting, IPAddressSetting);
53 bool matchesOrigin(const SecurityOrigin&) const;
54
55 const String& protocol() const { return m_protocol; }
56 const String& host() const { return m_host; }
57 SubdomainSetting subdomainSettings() const { return m_subdomainSettings; }
58 IPAddressSetting ipAddressSettings() const { return m_ipAddressSettings; }
59
60private:
61 String m_protocol;
62 String m_host;
63 SubdomainSetting m_subdomainSettings;
64 IPAddressSetting m_ipAddressSettings;
65 bool m_hostIsIPAddress;
66};
67
68inline bool operator==(const OriginAccessEntry& a, const OriginAccessEntry& b)
69{
70 return equalIgnoringASCIICase(a.protocol(), b.protocol())
71 && equalIgnoringASCIICase(a.host(), b.host())
72 && a.subdomainSettings() == b.subdomainSettings()
73 && a.ipAddressSettings() == b.ipAddressSettings();
74}
75
76inline bool operator!=(const OriginAccessEntry& a, const OriginAccessEntry& b)
77{
78 return !(a == b);
79}
