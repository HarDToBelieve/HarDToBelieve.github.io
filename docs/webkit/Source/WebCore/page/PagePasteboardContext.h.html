26#pragma once
27
28#include "PageIdentifier.h"
29#include "PasteboardContext.h"
30#include <wtf/TypeCasts.h>
31
32namespace WebCore {
33
34class PagePasteboardContext final : public PasteboardContext {
35 WTF_MAKE_FAST_ALLOCATED;
36public:
37 PagePasteboardContext(std::optional<PageIdentifier>&& pageID = std::nullopt)
38 : PasteboardContext()
39 , m_pageID(WTFMove(pageID))
40 {
41 }
42
43 ~PagePasteboardContext() { }
44
45 static std::unique_ptr<PasteboardContext> create(std::optional<PageIdentifier>&& pageID = std::nullopt)
46 {
47 return makeUnique<PagePasteboardContext>(WTFMove(pageID));
48 }
49
50 std::optional<PageIdentifier> pageID() const { return m_pageID; }
51 bool isPagePasteboardContext() const override { return true; }
52
53private:
54 std::optional<PageIdentifier> m_pageID;
55};
56
57} // namespace WebCore
58
59SPECIALIZE_TYPE_TRAITS_BEGIN(WebCore::PagePasteboardContext)
60 static bool isType(const WebCore::PasteboardContext& context) { return context.isPagePasteboardContext(); }
61SPECIALIZE_TYPE_TRAITS_END()
