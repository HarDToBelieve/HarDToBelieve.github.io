82DOMHighResTimeStamp Performance::timeOrigin() const
83{
84 return reduceTimeResolution(m_timeOrigin.approximateWallTime().secondsSinceEpoch()).milliseconds();

87ReducedResolutionSeconds Performance::nowInReducedResolutionSeconds() const
88{
89 Seconds now = MonotonicTime::now() - m_timeOrigin;
90 return reduceTimeResolution(now);
91}
92
93Seconds Performance::reduceTimeResolution(Seconds seconds)
94{
95 double resolution = (1000_us).seconds();
96 double reduced = std::floor(seconds.seconds() / resolution) * resolution;
97 return Seconds(reduced);
98}

100DOMHighResTimeStamp Performance::relativeTimeFromTimeOriginInReducedResolution(MonotonicTime timestamp) const
101{
102 Seconds seconds = timestamp - m_timeOrigin;
103 return reduceTimeResolution(seconds).milliseconds();

128Vector<RefPtr<PerformanceEntry>> Performance::getEntries() const
129{
130 Vector<RefPtr<PerformanceEntry>> entries;
131
132 if (m_navigationTiming)
133 entries.append(m_navigationTiming);
134
135 entries.appendVector(m_resourceTimingBuffer);
136
137 if (m_userTiming) {
138 entries.appendVector(m_userTiming->getMarks());
139 entries.appendVector(m_userTiming->getMeasures());
140 }
141
142 if (m_firstContentfulPaint)
143 entries.append(m_firstContentfulPaint);
144
145 std::sort(entries.begin(), entries.end(), PerformanceEntry::startTimeCompareLessThan);
146 return entries;
147}

149Vector<RefPtr<PerformanceEntry>> Performance::getEntriesByType(const String& entryType) const
150{
151 Vector<RefPtr<PerformanceEntry>> entries;
152
153 if (m_navigationTiming && entryType == "navigation")
154 entries.append(m_navigationTiming);
155
156 if (entryType == "resource")
157 entries.appendVector(m_resourceTimingBuffer);
158
159 if (m_firstContentfulPaint && entryType == "paint")
160 entries.append(m_firstContentfulPaint);
161
162 if (m_userTiming) {
163 if (entryType == "mark")
164 entries.appendVector(m_userTiming->getMarks());
165 else if (entryType == "measure")
166 entries.appendVector(m_userTiming->getMeasures());
167 }
168
169 std::sort(entries.begin(), entries.end(), PerformanceEntry::startTimeCompareLessThan);
170 return entries;
171}

173Vector<RefPtr<PerformanceEntry>> Performance::getEntriesByName(const String& name, const String& entryType) const
174{
175 Vector<RefPtr<PerformanceEntry>> entries;
176
177 if (m_navigationTiming && (entryType.isNull() || entryType == "navigation") && name == m_navigationTiming->name())
178 entries.append(m_navigationTiming);
179
180 if (entryType.isNull() || entryType == "resource") {
181 for (auto& resource : m_resourceTimingBuffer) {
182 if (resource->name() == name)
183 entries.append(resource);
184 }
185 }
186
187 if (m_firstContentfulPaint && (entryType.isNull() || entryType == "paint") && name == "first-contentful-paint")
188 entries.append(m_firstContentfulPaint);
189
190 if (m_userTiming) {
191 if (entryType.isNull() || entryType == "mark")
192 entries.appendVector(m_userTiming->getMarks(name));
193 if (entryType.isNull() || entryType == "measure")
194 entries.appendVector(m_userTiming->getMeasures(name));
195 }
196
197 std::sort(entries.begin(), entries.end(), PerformanceEntry::startTimeCompareLessThan);
198 return entries;
199}

247void Performance::addResourceTiming(ResourceTiming&& resourceTiming)
248{
249 ASSERT(scriptExecutionContext());
250
251 auto entry = PerformanceResourceTiming::create(m_timeOrigin, WTFMove(resourceTiming));
252
253 if (m_waitingForBackupBufferToBeProcessed) {
254 m_backupResourceTimingBuffer.append(WTFMove(entry));
255 return;
256 }
257
258 if (m_resourceTimingBufferFullFlag) {
259 // We fired resourcetimingbufferfull event but the author script didn't clear the buffer.
260 // Notify performance observers but don't add it to the buffer.
261 queueEntry(entry.get());
262 return;
263 }
264
265 if (isResourceTimingBufferFull()) {
266 ASSERT(!m_resourceTimingBufferFullTimer.isActive());
267 m_backupResourceTimingBuffer.append(WTFMove(entry));
268 m_waitingForBackupBufferToBeProcessed = true;
269 m_resourceTimingBufferFullTimer.startOneShot(0_s);
270 return;
271 }
272
273 queueEntry(entry.get());
274 m_resourceTimingBuffer.append(WTFMove(entry));
275}

282void Performance::resourceTimingBufferFullTimerFired()
283{
284 ASSERT(scriptExecutionContext());
285
286 while (!m_backupResourceTimingBuffer.isEmpty()) {
287 auto beforeCount = m_backupResourceTimingBuffer.size();
288
289 auto backupBuffer = WTFMove(m_backupResourceTimingBuffer);
290 ASSERT(m_backupResourceTimingBuffer.isEmpty());
291
292 if (isResourceTimingBufferFull()) {
293 m_resourceTimingBufferFullFlag = true;
294 dispatchEvent(Event::create(eventNames().resourcetimingbufferfullEvent, Event::CanBubble::No, Event::IsCancelable::No));
295 }
296
297 if (m_resourceTimingBufferFullFlag) {
298 for (auto& entry : backupBuffer)
299 queueEntry(*entry);
300 // Dispatching resourcetimingbufferfull event may have inserted more entries.
301 for (auto& entry : m_backupResourceTimingBuffer)
302 queueEntry(*entry);
303 m_backupResourceTimingBuffer.clear();
304 break;
305 }
306
307 // More entries may have added while dispatching resourcetimingbufferfull event.
308 backupBuffer.appendVector(m_backupResourceTimingBuffer);
309 m_backupResourceTimingBuffer.clear();
310
311 for (auto& entry : backupBuffer) {
312 if (!isResourceTimingBufferFull()) {
313 m_resourceTimingBuffer.append(entry.copyRef());
314 queueEntry(*entry);
315 } else
316 m_backupResourceTimingBuffer.append(entry.copyRef());
317 }
318
319 auto afterCount = m_backupResourceTimingBuffer.size();
320
321 if (beforeCount <= afterCount) {
322 m_backupResourceTimingBuffer.clear();
323 break;
324 }
325 }
326 m_waitingForBackupBufferToBeProcessed = false;
327}

329ExceptionOr<Ref<PerformanceMark>> Performance::mark(JSC::JSGlobalObject& globalObject, const String& markName, std::optional<PerformanceMarkOptions>&& markOptions)
330{
331 if (!m_userTiming)
332 m_userTiming = makeUnique<PerformanceUserTiming>(*this);
333
334 auto mark = m_userTiming->mark(globalObject, markName, WTFMove(markOptions));
335 if (mark.hasException())
336 return mark.releaseException();
337
338 queueEntry(mark.returnValue().get());
339 return mark.releaseReturnValue();
340}

376void Performance::registerPerformanceObserver(PerformanceObserver& observer)
377{
378 m_observers.add(&observer);
379
380 if (m_navigationTiming
381 && observer.typeFilter().contains(PerformanceEntry::Type::Navigation)
382 && !observer.hasNavigationTiming()) {
383 observer.queueEntry(*m_navigationTiming);
384 observer.addedNavigationTiming();
385 }
386}

388void Performance::unregisterPerformanceObserver(PerformanceObserver& observer)
389{
390 m_observers.remove(&observer);
391}

399void Performance::queueEntry(PerformanceEntry& entry)
400{
401 bool shouldScheduleTask = false;
402 for (auto& observer : m_observers) {
403 if (observer->typeFilter().contains(entry.performanceEntryType())) {
404 observer->queueEntry(entry);
405 shouldScheduleTask = true;
406 }
407 }
408
409 if (!shouldScheduleTask)
410 return;
411
412 scheduleTaskIfNeeded();
413}
414
415void Performance::scheduleTaskIfNeeded()
416{
417 if (m_hasScheduledTimingBufferDeliveryTask)
418 return;
419
420 auto* context = scriptExecutionContext();
421 if (!context)
422 return;
423
424 m_hasScheduledTimingBufferDeliveryTask = true;
425 context->eventLoop().queueTask(TaskSource::PerformanceTimeline, [protectedThis = makeRef(*this), this] {
426 auto* context = scriptExecutionContext();
427 if (!context)
428 return;
429
430 m_hasScheduledTimingBufferDeliveryTask = false;
431 for (auto& observer : copyToVector(m_observers))
432 observer->deliver();
433 });
434}
