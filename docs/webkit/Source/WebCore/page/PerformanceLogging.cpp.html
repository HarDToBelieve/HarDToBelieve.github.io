43static const char* toString(PerformanceLogging::PointOfInterest poi)
44{

46 case PerformanceLogging::MainFrameLoadStarted:

56HashMap<const char*, size_t> PerformanceLogging::memoryUsageStatistics(ShouldIncludeExpensiveComputations includeExpensive)
57{
58 HashMap<const char*, size_t> stats;
59
60 auto& vm = commonVM();
61 JSC::JSLockHolder locker(vm);
62 stats.add("javascript_gc_heap_capacity", vm.heap.capacity());
63 stats.add("javascript_gc_heap_extra_memory_size", vm.heap.extraMemorySize());
64
65 auto& backForwardCache = BackForwardCache::singleton();
66 stats.add("backforward_cache_page_count", backForwardCache.pageCount());
67
68 stats.add("document_count", Document::allDocuments().size());
69
70 if (includeExpensive == ShouldIncludeExpensiveComputations::Yes) {
71 stats.add("javascript_gc_heap_size", vm.heap.size());
72 stats.add("javascript_gc_object_count", vm.heap.objectCount());
73 stats.add("javascript_gc_protected_object_count", vm.heap.protectedObjectCount());
74 stats.add("javascript_gc_global_object_count", vm.heap.globalObjectCount());
75 stats.add("javascript_gc_protected_global_object_count", vm.heap.protectedGlobalObjectCount());
76 }
77
78 return stats;
79}

91void PerformanceLogging::didReachPointOfInterest(PointOfInterest poi)
92{
93#if RELEASE_LOG_DISABLED
94 UNUSED_PARAM(poi);
95 UNUSED_VARIABLE(m_page);
96#else
97 // Ignore synthetic main frames used internally by SVG and web inspector.
98 if (m_page.mainFrame().loader().client().isEmptyFrameLoaderClient())
99 return;
100
101 auto stats = memoryUsageStatistics(ShouldIncludeExpensiveComputations::No);
102 getPlatformMemoryUsageStatistics(stats);
103
104 RELEASE_LOG(PerformanceLogging, "Memory usage info dump at %s:", toString(poi));
105 for (auto& it : stats)
106 RELEASE_LOG(PerformanceLogging, " %s: %zu", it.key, it.value);
107#endif
108}
