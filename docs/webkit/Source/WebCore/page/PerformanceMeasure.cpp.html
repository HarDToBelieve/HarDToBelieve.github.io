33ExceptionOr<Ref<PerformanceMeasure>> PerformanceMeasure::create(const String& name, double startTime, double endTime, Ref<SerializedScriptValue>&& serializedDetail)
34{
35 return adoptRef(*new PerformanceMeasure(name, startTime, endTime, WTFMove(serializedDetail)));

38PerformanceMeasure::PerformanceMeasure(const String& name, double startTime, double endTime, Ref<SerializedScriptValue>&& serializedDetail)

41{
42}

48 return m_serializedDetail->deserialize(globalObject, &globalObject);
