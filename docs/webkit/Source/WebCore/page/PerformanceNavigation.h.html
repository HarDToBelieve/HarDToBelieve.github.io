33#include "DOMWindowProperty.h"
34#include <wtf/RefCounted.h>
35#include <wtf/Ref.h>
36
37namespace WebCore {
38
39class PerformanceNavigation : public RefCounted<PerformanceNavigation>, public DOMWindowProperty {
40public:
41 static Ref<PerformanceNavigation> create(DOMWindow* window) { return adoptRef(*new PerformanceNavigation(window)); }
42
43 enum PerformanceNavigationType {
44 TYPE_NAVIGATE,
45 TYPE_RELOAD,
46 TYPE_BACK_FORWARD,
47 TYPE_RESERVED = 255
48 };
49
50 unsigned short type() const;
51 unsigned short redirectCount() const;
52
53private:
54 explicit PerformanceNavigation(DOMWindow*);
55};
56
57} // namespace WebCore
