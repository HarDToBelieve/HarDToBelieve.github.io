36static PerformanceNavigationTiming::NavigationType toPerformanceNavigationTimingNavigationType(NavigationType navigationType)
37{
38 switch (navigationType) {
39 case NavigationType::BackForward:
40 return PerformanceNavigationTiming::NavigationType::Back_forward;
41 case NavigationType::Reload:
42 return PerformanceNavigationTiming::NavigationType::Reload;
43 case NavigationType::LinkClicked:
44 case NavigationType::FormSubmitted:
45 case NavigationType::FormResubmitted:
46 case NavigationType::Other:
47 return PerformanceNavigationTiming::NavigationType::Navigate;
48 }
49 ASSERT_NOT_REACHED();
50 return PerformanceNavigationTiming::NavigationType::Navigate;
51}

53PerformanceNavigationTiming::PerformanceNavigationTiming(MonotonicTime timeOrigin, CachedResource& resource, const DocumentLoadTiming& documentLoadTiming, const NetworkLoadMetrics& metrics, const DocumentEventTiming& documentEventTiming, const SecurityOrigin& origin, WebCore::NavigationType navigationType)
54 : PerformanceResourceTiming(timeOrigin, ResourceTiming::fromLoad(resource, resource.response().url(), "navigation"_s, documentLoadTiming, metrics, origin))
55 , m_documentEventTiming(documentEventTiming)
56 , m_documentLoadTiming(documentLoadTiming)
57 , m_navigationType(toPerformanceNavigationTimingNavigationType(navigationType)) { }
58
59PerformanceNavigationTiming::~PerformanceNavigationTiming() = default;
60
61double PerformanceNavigationTiming::millisecondsSinceOrigin(MonotonicTime time) const
62{
63 if (!time)
64 return 0;
65 return Performance::reduceTimeResolution(time - m_timeOrigin).milliseconds();
66}
67
68bool PerformanceNavigationTiming::sameOriginCheckFails() const
69{
70 // https://www.w3.org/TR/navigation-timing-2/#dfn-same-origin-check
71 return m_resourceTiming.networkLoadMetrics().hasCrossOriginRedirect
72 || !m_documentLoadTiming.hasSameOriginAsPreviousDocument();
73}
74
75double PerformanceNavigationTiming::unloadEventStart() const
76{
77 if (sameOriginCheckFails())
78 return 0.0;
79 return millisecondsSinceOrigin(m_documentLoadTiming.unloadEventStart());
80}
81
82double PerformanceNavigationTiming::unloadEventEnd() const
83{
84 if (sameOriginCheckFails())
85 return 0.0;
86 return millisecondsSinceOrigin(m_documentLoadTiming.unloadEventEnd());
87}

124unsigned short PerformanceNavigationTiming::redirectCount() const
125{
126 if (m_resourceTiming.networkLoadMetrics().hasCrossOriginRedirect)
127 return 0;
128
129 return m_resourceTiming.networkLoadMetrics().redirectCount;
130}

132double PerformanceNavigationTiming::startTime() const
133{
134 // https://www.w3.org/TR/navigation-timing-2/#dom-PerformanceNavigationTiming-startTime
135 return 0.0;
136}
137
138double PerformanceNavigationTiming::duration() const
139{
140 // https://www.w3.org/TR/navigation-timing-2/#dom-PerformanceNavigationTiming-duration
141 return loadEventEnd() - startTime();
142}
