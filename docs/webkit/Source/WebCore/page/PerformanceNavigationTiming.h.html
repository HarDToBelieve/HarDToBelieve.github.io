35namespace WebCore {
36
37class CachedResource;
38class NetworkLoadMetrics;
39class SecurityOrigin;
40enum class NavigationType : uint8_t;
41
42class PerformanceNavigationTiming final : public PerformanceResourceTiming {
43public:
44 enum class NavigationType : uint8_t {
45 Navigate,
46 Reload,
47 Back_forward,
48 Prerender,
49 };
50
51 template<typename... Args> static Ref<PerformanceNavigationTiming> create(Args&&... args) { return adoptRef(*new PerformanceNavigationTiming(std::forward<Args>(args)...)); }
52 ~PerformanceNavigationTiming();
53
54 Type performanceEntryType() const final { return Type::Navigation; }
55 ASCIILiteral entryType() const final { return "navigation"_s; }
56
57 double unloadEventStart() const;
58 double unloadEventEnd() const;
59 double domInteractive() const;
60 double domContentLoadedEventStart() const;
61 double domContentLoadedEventEnd() const;
62 double domComplete() const;
63 double loadEventStart() const;
64 double loadEventEnd() const;
65 NavigationType type() const;
66 unsigned short redirectCount() const;
67
68 double startTime() const final;
69 double duration() const final;
70
71private:
72 PerformanceNavigationTiming(MonotonicTime timeOrigin, CachedResource&, const DocumentLoadTiming&, const NetworkLoadMetrics&, const DocumentEventTiming&, const SecurityOrigin&, WebCore::NavigationType);
73
74 double millisecondsSinceOrigin(MonotonicTime) const;
75 bool sameOriginCheckFails() const;
76
77 DocumentEventTiming m_documentEventTiming;
78 DocumentLoadTiming m_documentLoadTiming;
79 NavigationType m_navigationType;
80};
