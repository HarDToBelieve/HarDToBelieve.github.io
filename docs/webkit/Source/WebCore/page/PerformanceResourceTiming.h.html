34#include "NetworkLoadMetrics.h"
35#include "PerformanceEntry.h"
36#include "ResourceTiming.h"
37#include <wtf/Ref.h>
38#include <wtf/text/WTFString.h>

40namespace WebCore {
41
42class PerformanceServerTiming;
43class ResourceTiming;
44
45class PerformanceResourceTiming : public PerformanceEntry {
46public:
47 static Ref<PerformanceResourceTiming> create(MonotonicTime timeOrigin, ResourceTiming&&);
48
49 const String& initiatorType() const { return m_resourceTiming.initiator(); }
50 const String& nextHopProtocol() const;
51
52 double workerStart() const;
53 double redirectStart() const;
54 double redirectEnd() const;
55 double fetchStart() const;
56 double domainLookupStart() const;
57 double domainLookupEnd() const;
58 double connectStart() const;
59 double connectEnd() const;
60 double secureConnectionStart() const;
61 double requestStart() const;
62 double responseStart() const;
63 double responseEnd() const;
64 uint64_t transferSize() const;
65 uint64_t encodedBodySize() const;
66 uint64_t decodedBodySize() const;
67
68 const Vector<Ref<PerformanceServerTiming>>& serverTiming() const { return m_serverTiming; }
69
70 Type performanceEntryType() const override { return Type::Resource; }
71 ASCIILiteral entryType() const override { return "resource"_s; }
72
73protected:
74 PerformanceResourceTiming(MonotonicTime timeOrigin, ResourceTiming&&);
75 ~PerformanceResourceTiming();
76
77 MonotonicTime m_timeOrigin;
78 const ResourceTiming m_resourceTiming;
79 Vector<Ref<PerformanceServerTiming>> m_serverTiming;
80};

82} // namespace WebCore
