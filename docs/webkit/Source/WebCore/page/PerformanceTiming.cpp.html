51unsigned long long PerformanceTiming::navigationStart() const
52{
53 auto* timing = documentLoadTiming();
54 if (!timing)
55 return 0;
56
57 return monotonicTimeToIntegerMilliseconds(timing->startTime());
58}

60unsigned long long PerformanceTiming::unloadEventStart() const
61{
62 auto* timing = documentLoadTiming();
63 if (!timing)
64 return 0;
65
66 auto* metrics = networkLoadMetrics();
67 if (!metrics)
68 return 0;
69
70 if (metrics->failsTAOCheck || !timing->hasSameOriginAsPreviousDocument())
71 return 0;
72
73 return monotonicTimeToIntegerMilliseconds(timing->unloadEventStart());
74}

141unsigned long long PerformanceTiming::connectStart() const
142{
143 auto* metrics = networkLoadMetrics();
144 if (!metrics || !metrics->connectStart
145 || metrics->domainLookupEnd.secondsSinceEpoch() > metrics->connectStart.secondsSinceEpoch())
146 return domainLookupEnd();
147
148 return monotonicTimeToIntegerMilliseconds(metrics->connectStart);
149}

156
157 return monotonicTimeToIntegerMilliseconds(metrics->connectEnd);
158}

160unsigned long long PerformanceTiming::secureConnectionStart() const
161{
162 auto* metrics = networkLoadMetrics();
163 if (!metrics)
164 return connectEnd();
165
166 if (!metrics->secureConnectionStart
167 || metrics->secureConnectionStart == reusedTLSConnectionSentinel)
168 return 0;
169
170 return monotonicTimeToIntegerMilliseconds(metrics->secureConnectionStart);
171}

265 auto* frame = this->frame();

272const DocumentEventTiming* PerformanceTiming::documentEventTiming() const
273{
274 auto* frame = this->frame();
275 if (!frame)
276 return nullptr;
277
278 auto* document = frame->document();
279 if (!document)
280 return nullptr;
281
282 return &document->eventTiming();
283}

285const DocumentLoadTiming* PerformanceTiming::documentLoadTiming() const
286{
287 auto* loader = documentLoader();
288 if (!loader)
289 return nullptr;
290
291 return &loader->timing();
292}

294const NetworkLoadMetrics* PerformanceTiming::networkLoadMetrics() const
295{
296 auto* loader = documentLoader();
297 if (!loader)
298 return nullptr;
299 return loader->response().deprecatedNetworkLoadMetricsOrNull();
300}

302unsigned long long PerformanceTiming::monotonicTimeToIntegerMilliseconds(MonotonicTime timeStamp) const
303{
304 ASSERT(timeStamp.secondsSinceEpoch().seconds() >= 0);
305 if (!timeStamp)
306 return 0;
307 Seconds reduced = Performance::reduceTimeResolution(timeStamp.approximateWallTime().secondsSinceEpoch());
308 return static_cast<unsigned long long>(reduced.milliseconds());
309}
