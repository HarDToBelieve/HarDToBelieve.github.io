42using NavigationTimingFunction = unsigned long long (PerformanceTiming::*)() const;
43static const MemoryCompactLookupOnlyRobinHoodHashMap<String, NavigationTimingFunction>& restrictedMarkNamesToNavigationTimingFunctionMap()
44{
45 ASSERT(isMainThread());
46
47 static auto map = makeNeverDestroyed<MemoryCompactLookupOnlyRobinHoodHashMap<String, NavigationTimingFunction>>({
48 { "connectEnd"_s, &PerformanceTiming::connectEnd },
49 { "connectStart"_s, &PerformanceTiming::connectStart },
50 { "domComplete"_s, &PerformanceTiming::domComplete },
51 { "domContentLoadedEventEnd"_s, &PerformanceTiming::domContentLoadedEventEnd },
52 { "domContentLoadedEventStart"_s, &PerformanceTiming::domContentLoadedEventStart },
53 { "domInteractive"_s, &PerformanceTiming::domInteractive },
54 { "domLoading"_s, &PerformanceTiming::domLoading },
55 { "domainLookupEnd"_s, &PerformanceTiming::domainLookupEnd },
56 { "domainLookupStart"_s, &PerformanceTiming::domainLookupStart },
57 { "fetchStart"_s, &PerformanceTiming::fetchStart },
58 { "loadEventEnd"_s, &PerformanceTiming::loadEventEnd },
59 { "loadEventStart"_s, &PerformanceTiming::loadEventStart },
60 { "navigationStart"_s, &PerformanceTiming::navigationStart },
61 { "redirectEnd"_s, &PerformanceTiming::redirectEnd },
62 { "redirectStart"_s, &PerformanceTiming::redirectStart },
63 { "requestStart"_s, &PerformanceTiming::requestStart },
64 { "responseEnd"_s, &PerformanceTiming::responseEnd },
65 { "responseStart"_s, &PerformanceTiming::responseStart },
66 { "secureConnectionStart"_s, &PerformanceTiming::secureConnectionStart },
67 { "unloadEventEnd"_s, &PerformanceTiming::unloadEventEnd },
68 { "unloadEventStart"_s, &PerformanceTiming::unloadEventStart },
69 });
70
71 return map;
72}

74static NavigationTimingFunction restrictedMarkFunction(const String& markName)
75{
76 ASSERT(isMainThread());
77 return restrictedMarkNamesToNavigationTimingFunctionMap().get(markName);
78}

80static bool isRestrictedMarkNameNonMainThread(const String& markName)
81{
82 ASSERT(!isMainThread());
83
84 bool isRestricted;
85 callOnMainThreadAndWait([&isRestricted, markName = markName.isolatedCopy()] {
86 isRestricted = restrictedMarkNamesToNavigationTimingFunctionMap().contains(markName);
87 });
88 return isRestricted;
89}

91bool PerformanceUserTiming::isRestrictedMarkName(const String& markName)
92{
93 ASSERT(isMainThread());
94 return restrictedMarkNamesToNavigationTimingFunctionMap().contains(markName);

131ExceptionOr<double> PerformanceUserTiming::convertMarkToTimestamp(const Variant<String, double>& mark) const
132{
133 return WTF::switchOn(mark, [&](auto& value) {
134 return convertMarkToTimestamp(value);
135 });
136}

138ExceptionOr<double> PerformanceUserTiming::convertMarkToTimestamp(const String& mark) const
139{
140 if (!is<Document>(m_performance.scriptExecutionContext())) {
141 if (isRestrictedMarkNameNonMainThread(mark))
142 return Exception { TypeError };
143 } else {
144 if (auto function = restrictedMarkFunction(mark)) {
145 if (function == &PerformanceTiming::navigationStart)
146 return 0.0;
147
148 // PerformanceTiming should always be non-null for the Document ScriptExecutionContext.
149 ASSERT(m_performance.timing());
150 auto timing = m_performance.timing();
151 auto startTime = timing->navigationStart();
152 auto endTime = ((*timing).*(function))();
153 if (!endTime)
154 return Exception { InvalidAccessError };
155 return endTime - startTime;
156 }
157 }
158
159 auto iterator = m_marksMap.find(mark);
160 if (iterator != m_marksMap.end())
161 return iterator->value.last()->startTime();
162
163 return Exception { SyntaxError, makeString("No mark named '", mark, "' exists") };
164}

166ExceptionOr<double> PerformanceUserTiming::convertMarkToTimestamp(double mark) const
167{
168 if (mark < 0)
169 return Exception { TypeError };
170 return mark;

173ExceptionOr<Ref<PerformanceMeasure>> PerformanceUserTiming::measure(const String& measureName, const String& startMark, const String& endMark)
174{
175 double endTime;
176 if (!endMark.isNull()) {
177 auto end = convertMarkToTimestamp(endMark);
178 if (end.hasException())
179 return end.releaseException();
180 endTime = end.returnValue();
181 } else
182 endTime = m_performance.now();
183
184 double startTime;
185 if (!startMark.isNull()) {
186 auto start = convertMarkToTimestamp(startMark);
187 if (start.hasException())
188 return start.releaseException();
189 startTime = start.returnValue();
190 } else
191 startTime = 0.0;
192
193 auto measure = PerformanceMeasure::create(measureName, startTime, endTime, SerializedScriptValue::nullValue());
194 if (measure.hasException())
195 return measure.releaseException();
196
197 addPerformanceEntry(m_measuresMap, measureName, measure.returnValue().get());
198 return measure.releaseReturnValue();
199}

201ExceptionOr<Ref<PerformanceMeasure>> PerformanceUserTiming::measure(JSC::JSGlobalObject& globalObject, const String& measureName, const PerformanceMeasureOptions& measureOptions)
202{
203 double endTime;
204 if (measureOptions.end) {
205 auto end = convertMarkToTimestamp(*measureOptions.end);
206 if (end.hasException())
207 return end.releaseException();
208 endTime = end.returnValue();
209 } else if (measureOptions.start && measureOptions.duration) {
210 auto start = convertMarkToTimestamp(*measureOptions.start);
211 if (start.hasException())
212 return start.releaseException();
213 auto duration = convertMarkToTimestamp(*measureOptions.duration);
214 if (duration.hasException())
215 return duration.releaseException();
216 endTime = start.returnValue() + duration.returnValue();
217 } else
218 endTime = m_performance.now();
219
220 double startTime;
221 if (measureOptions.start) {
222 auto start = convertMarkToTimestamp(*measureOptions.start);
223 if (start.hasException())
224 return start.releaseException();
225 startTime = start.returnValue();
226 } else if (measureOptions.duration && measureOptions.end) {
227 auto duration = convertMarkToTimestamp(*measureOptions.duration);
228 if (duration.hasException())
229 return duration.releaseException();
230 auto end = convertMarkToTimestamp(*measureOptions.end);
231 if (end.hasException())
232 return end.releaseException();
233 startTime = end.returnValue() - duration.returnValue();
234 } else
235 startTime = 0;
236
237
238 JSC::JSValue detail = measureOptions.detail;
239 if (detail.isUndefined())
240 detail = JSC::jsNull();
241
242 Vector<RefPtr<MessagePort>> ignoredMessagePorts;
243 auto serializedDetail = SerializedScriptValue::create(globalObject, detail, { }, ignoredMessagePorts);
244 if (serializedDetail.hasException())
245 return serializedDetail.releaseException();
246
247 auto measure = PerformanceMeasure::create(measureName, startTime, endTime, serializedDetail.releaseReturnValue());
248 if (measure.hasException())
249 return measure.releaseException();
250
251 addPerformanceEntry(m_measuresMap, measureName, measure.returnValue().get());
252 return measure.releaseReturnValue();
253}

255static bool isNonEmptyDictionary(const PerformanceMeasureOptions& measureOptions)
256{
257 return !measureOptions.detail.isUndefined() || measureOptions.start || measureOptions.duration || measureOptions.end;
258}

260ExceptionOr<Ref<PerformanceMeasure>> PerformanceUserTiming::measure(JSC::JSGlobalObject& globalObject, const String& measureName, std::optional<StartOrMeasureOptions>&& startOrMeasureOptions, const String& endMark)
261{
262 if (startOrMeasureOptions) {
263 return WTF::switchOn(*startOrMeasureOptions,
264 [&] (const PerformanceMeasureOptions& measureOptions) -> ExceptionOr<Ref<PerformanceMeasure>> {
265 if (isNonEmptyDictionary(measureOptions)) {
266 if (!endMark.isNull())
267 return Exception { TypeError };
268 if (!measureOptions.start && !measureOptions.end)
269 return Exception { TypeError };
270 if (measureOptions.start && measureOptions.duration && measureOptions.end)
271 return Exception { TypeError };
272 }
273
274 return measure(globalObject, measureName, measureOptions);
275 },
276 [&] (const String& startMark) {
277 return measure(measureName, startMark, endMark);
278 }
279 );
280 }
281
282 return measure(measureName, { }, endMark);
283}
