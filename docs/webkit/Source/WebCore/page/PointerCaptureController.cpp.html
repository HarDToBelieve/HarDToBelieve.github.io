50Element* PointerCaptureController::pointerCaptureElement(Document* document, PointerID pointerId) const
51{
52 auto iterator = m_activePointerIdsToCapturingData.find(pointerId);
53 if (iterator != m_activePointerIdsToCapturingData.end()) {
54 auto pointerCaptureElement = iterator->value.targetOverride;
55 if (pointerCaptureElement && &pointerCaptureElement->document() == document)
56 return pointerCaptureElement.get();
57 }
58 return nullptr;
59}

61ExceptionOr<void> PointerCaptureController::setPointerCapture(Element* capturingTarget, PointerID pointerId)
62{
63 // https://w3c.github.io/pointerevents/#setting-pointer-capture
64
65 // 1. If the pointerId provided as the method's argument does not match any of the active pointers, then throw a DOMException with the name NotFoundError.
66 auto iterator = m_activePointerIdsToCapturingData.find(pointerId);
67 if (iterator == m_activePointerIdsToCapturingData.end())
68 return Exception { NotFoundError };
69
70 // 2. If the Element on which this method is invoked is not connected, throw an exception with the name InvalidStateError.
71 if (!capturingTarget->isConnected())
72 return Exception { InvalidStateError };
73
74#if ENABLE(POINTER_LOCK)
75 // 3. If this method is invoked while the document has a locked element, throw an exception with the name InvalidStateError.
76 if (auto* page = capturingTarget->document().page()) {
77 if (page->pointerLockController().isLocked())
78 return Exception { InvalidStateError };
79 }
80#endif
81
82 // 4. If the pointer is not in the active buttons state, then terminate these steps.
83 // 5. For the specified pointerId, set the pending pointer capture target override to the Element on which this method was invoked.
84 auto& capturingData = iterator->value;
85 if (capturingData.pointerIsPressed)
86 capturingData.pendingTargetOverride = capturingTarget;
87
88 updateHaveAnyCapturingElement();
89 return { };
90}

92ExceptionOr<void> PointerCaptureController::releasePointerCapture(Element* capturingTarget, PointerID pointerId)
93{
94 // https://w3c.github.io/pointerevents/#releasing-pointer-capture
95
96 // Pointer capture is released on an element explicitly by calling the element.releasePointerCapture(pointerId) method.
97 // When this method is called, a user agent MUST run the following steps:
98
99 // 1. If the pointerId provided as the method's argument does not match any of the active pointers and these steps are not
100 // being invoked as a result of the implicit release of pointer capture, then throw a DOMException with the name NotFoundError.
101 auto iterator = m_activePointerIdsToCapturingData.find(pointerId);
102 if (iterator == m_activePointerIdsToCapturingData.end())
103 return Exception { NotFoundError };
104
105 // 2. If hasPointerCapture is false for the Element with the specified pointerId, then terminate these steps.
106 if (!hasPointerCapture(capturingTarget, pointerId))
107 return { };
108
109 // 3. For the specified pointerId, clear the pending pointer capture target override, if set.
110 iterator->value.pendingTargetOverride = nullptr;
111
112 // FIXME: This leaves value.targetOverride set: webkit.org/b/221342.
113
114 updateHaveAnyCapturingElement();
115 return { };
116}

118bool PointerCaptureController::hasPointerCapture(Element* capturingTarget, PointerID pointerId)
119{
120 // https://w3c.github.io/pointerevents/#dom-element-haspointercapture
121
122 // Indicates whether the element on which this method is invoked has pointer capture for the pointer identified by the argument pointerId.
123 // In particular, returns true if the pending pointer capture target override for pointerId is set to the element on which this method is
124 // invoked, and false otherwise.
125
126 if (!m_haveAnyCapturingElement)
127 return false;
128
129 auto iterator = m_activePointerIdsToCapturingData.find(pointerId);
130 return iterator != m_activePointerIdsToCapturingData.end() && iterator->value.pendingTargetOverride == capturingTarget;
131}

133void PointerCaptureController::pointerLockWasApplied()
134{
135 // https://w3c.github.io/pointerevents/#implicit-release-of-pointer-capture
136
137 // When a pointer lock is successfully applied on an element, a user agent MUST run the steps as if the releasePointerCapture()
138 // method has been called if any element is set to be captured or pending to be captured.
139 for (auto& capturingData : m_activePointerIdsToCapturingData.values()) {
140 capturingData.pendingTargetOverride = nullptr;
141 capturingData.targetOverride = nullptr;
142 }
143
144 updateHaveAnyCapturingElement();
145}

147void PointerCaptureController::elementWasRemoved(Element& element)
148{
149 if (!m_haveAnyCapturingElement)
150 return;
151
152 for (auto& keyAndValue : m_activePointerIdsToCapturingData) {
153 auto& capturingData = keyAndValue.value;
154 if (capturingData.pendingTargetOverride == &element || capturingData.targetOverride == &element) {
155 // https://w3c.github.io/pointerevents/#implicit-release-of-pointer-capture
156 // When the pointer capture target override is no longer connected, the pending pointer capture target override and pointer capture target
157 // override nodes SHOULD be cleared and also a PointerEvent named lostpointercapture corresponding to the captured pointer SHOULD be fired
158 // at the document.
159 ASSERT(isInBounds<PointerID>(keyAndValue.key));
160 auto pointerId = static_cast<PointerID>(keyAndValue.key);
161 auto pointerType = capturingData.pointerType;
162 releasePointerCapture(&element, pointerId);
163 // FIXME: Spec doesn't specify which task source to use.
164 element.document().queueTaskToDispatchEvent(TaskSource::UserInteraction, PointerEvent::create(eventNames().lostpointercaptureEvent, pointerId, pointerType));
165 return;
166 }
167 }

206static bool hierarchyHasCapturingEventListeners(Element* target, const AtomString& eventName)
207{
208 for (ContainerNode* curr = target; curr; curr = curr->parentInComposedTree()) {
209 if (curr->hasCapturingEventListeners(eventName))
210 return true;
211 }

215void PointerCaptureController::dispatchEventForTouchAtIndex(EventTarget& target, const PlatformTouchEvent& platformTouchEvent, unsigned index, bool isPrimary, WindowProxy& view)
216{
217 ASSERT(is<Element>(target));
218
219 auto dispatchOverOrOutEvent = [&](const String& type, EventTarget* target) {
220 dispatchEvent(PointerEvent::create(type, platformTouchEvent, index, isPrimary, view), target);
221 };
222
223 auto dispatchEnterOrLeaveEvent = [&](const String& type) {
224 auto* targetElement = &downcast<Element>(target);
225
226 bool hasCapturingListenerInHierarchy = false;
227 for (ContainerNode* curr = targetElement; curr; curr = curr->parentInComposedTree()) {
228 if (curr->hasCapturingEventListeners(type)) {
229 hasCapturingListenerInHierarchy = true;
230 break;
231 }
232 }
233
234 Vector<Ref<Element>, 32> targetChain;
235 for (Element* element = targetElement; element; element = element->parentElementInComposedTree()) {
236 if (hasCapturingListenerInHierarchy || element->hasEventListeners(type))
237 targetChain.append(*element);
238 }
239
240 if (type == eventNames().pointerenterEvent) {
241 for (auto& element : WTF::makeReversedRange(targetChain))
242 dispatchEvent(PointerEvent::create(type, platformTouchEvent, index, isPrimary, view), element.ptr());
243 } else {
244 for (auto& element : targetChain)
245 dispatchEvent(PointerEvent::create(type, platformTouchEvent, index, isPrimary, view), element.ptr());
246 }
247 };
248
249 auto pointerEvent = PointerEvent::create(platformTouchEvent, index, isPrimary, view);
250
251 auto& capturingData = ensureCapturingDataForPointerEvent(pointerEvent);
252
253 // Check if the target changed, which would require dispatching boundary events.
254 RefPtr<Element> previousTarget = capturingData.previousTarget;
255 RefPtr<Element> currentTarget = downcast<Element>(&target);
256
257 capturingData.previousTarget = currentTarget;
258
259 if (pointerEvent->type() == eventNames().pointermoveEvent && previousTarget != currentTarget) {
260 // The pointerenter and pointerleave events are only dispatched if there is a capturing event listener on an ancestor
261 // or a normal event listener on the element itself since those events do not bubble.
262 // This optimization is necessary since these events can cause O(n^2) capturing event-handler checks. This follows the
263 // code for similar mouse events in EventHandler::updateMouseEventTargetNode().
264 bool hasCapturingPointerEnterListener = hierarchyHasCapturingEventListeners(currentTarget.get(), eventNames().pointerenterEvent);
265 bool hasCapturingPointerLeaveListener = hierarchyHasCapturingEventListeners(previousTarget.get(), eventNames().pointerleaveEvent);
266
267 Vector<Ref<Element>, 32> leftElementsChain;
268 for (Element* element = previousTarget.get(); element; element = element->parentElementInComposedTree())
269 leftElementsChain.append(*element);
270 Vector<Ref<Element>, 32> enteredElementsChain;
271 for (Element* element = currentTarget.get(); element; element = element->parentElementInComposedTree())
272 enteredElementsChain.append(*element);
273
274 if (!leftElementsChain.isEmpty() && !enteredElementsChain.isEmpty() && leftElementsChain.last().ptr() == enteredElementsChain.last().ptr()) {
275 size_t minHeight = std::min(leftElementsChain.size(), enteredElementsChain.size());
276 size_t i;
277 for (i = 0; i < minHeight; ++i) {
278 if (leftElementsChain[leftElementsChain.size() - i - 1].ptr() != enteredElementsChain[enteredElementsChain.size() - i - 1].ptr())
279 break;
280 }
281 leftElementsChain.shrink(leftElementsChain.size() - i);
282 enteredElementsChain.shrink(enteredElementsChain.size() - i);
283 }
284
285 if (previousTarget)
286 dispatchOverOrOutEvent(eventNames().pointeroutEvent, previousTarget.get());
287
288 for (auto& chain : leftElementsChain) {
289 if (hasCapturingPointerLeaveListener || chain->hasEventListeners(eventNames().pointerleaveEvent))
290 dispatchEvent(PointerEvent::create(eventNames().pointerleaveEvent, platformTouchEvent, index, isPrimary, view), chain.ptr());
291 }
292
293 if (currentTarget)
294 dispatchOverOrOutEvent(eventNames().pointeroverEvent, currentTarget.get());
295
296 for (auto& chain : WTF::makeReversedRange(enteredElementsChain)) {
297 if (hasCapturingPointerEnterListener || chain->hasEventListeners(eventNames().pointerenterEvent))
298 dispatchEvent(PointerEvent::create(eventNames().pointerenterEvent, platformTouchEvent, index, isPrimary, view), chain.ptr());
299 }
300 }
301
302 if (pointerEvent->type() == eventNames().pointerdownEvent) {
303 // https://w3c.github.io/pointerevents/#the-pointerdown-event
304 // For input devices that do not support hover, a user agent MUST also fire a pointer event named pointerover followed by a pointer event named
305 // pointerenter prior to dispatching the pointerdown event.
306 dispatchOverOrOutEvent(eventNames().pointeroverEvent, currentTarget.get());
307 dispatchEnterOrLeaveEvent(eventNames().pointerenterEvent);
308 }
309
310 dispatchEvent(pointerEvent, &target);
311
312 if (pointerEvent->type() == eventNames().pointerupEvent) {
313 // https://w3c.github.io/pointerevents/#the-pointerup-event
314 // For input devices that do not support hover, a user agent MUST also fire a pointer event named pointerout followed by a
315 // pointer event named pointerleave after dispatching the pointerup event.
316 dispatchOverOrOutEvent(eventNames().pointeroutEvent, currentTarget.get());
317 dispatchEnterOrLeaveEvent(eventNames().pointerleaveEvent);
318 capturingData.previousTarget = nullptr;
319 capturingData.state = CapturingData::State::Finished;
320 }
321}

324RefPtr<PointerEvent> PointerCaptureController::pointerEventForMouseEvent(const MouseEvent& mouseEvent, PointerID pointerId, const String& pointerType)
325{
326 // If we already have known touches then we cannot dispatch a mouse event,
327 // for instance in the case of a long press to initiate a system drag.
328 for (auto& capturingData : m_activePointerIdsToCapturingData.values()) {
329 if (capturingData.pointerType == touchPointerEventType() && capturingData.state == CapturingData::State::Ready)
330 return nullptr;
331 }
332
333 const auto& type = mouseEvent.type();
334 const auto& names = eventNames();
335
336 auto iterator = m_activePointerIdsToCapturingData.find(pointerId);
337 bool pointerIsPressed = iterator != m_activePointerIdsToCapturingData.end() ? iterator->value.pointerIsPressed : false;
338
339 short newButton = mouseEvent.button();
340 short previousMouseButton = iterator != m_activePointerIdsToCapturingData.end() ? iterator->value.previousMouseButton : -1;
341 short button = (type == names.mousemoveEvent && newButton == previousMouseButton) ? -1 : newButton;
342
343 // https://w3c.github.io/pointerevents/#chorded-button-interactions
344 // Some pointer devices, such as mouse or pen, support multiple buttons. In the Mouse Event model, each button
345 // press produces a mousedown and mouseup event. To better abstract this hardware difference and simplify
346 // cross-device input authoring, Pointer Events do not fire overlapping pointerdown and pointerup events
347 // for chorded button presses (depressing an additional button while another button on the pointer device is
348 // already depressed).
349 if (type == names.mousedownEvent || type == names.mouseupEvent) {
350 // We're already active and getting another mousedown, this means that we should dispatch
351 // a pointermove event and let the button state show the newly depressed button.
352 if (type == names.mousedownEvent && pointerIsPressed)
353 return PointerEvent::create(names.pointermoveEvent, button, mouseEvent, pointerId, pointerType);
354
355 // We're active and the mouseup still has some pressed button, this means we should dispatch
356 // a pointermove event.
357 if (type == names.mouseupEvent && pointerIsPressed && mouseEvent.buttons() > 0)
358 return PointerEvent::create(names.pointermoveEvent, button, mouseEvent, pointerId, pointerType);
359 }
360
361 auto pointerEvent = PointerEvent::create(button, mouseEvent, pointerId, pointerType);
362 if (iterator != m_activePointerIdsToCapturingData.end())
363 iterator->value.previousMouseButton = newButton;
364 else if (pointerEvent)
365 ensureCapturingDataForPointerEvent(*pointerEvent).previousMouseButton = newButton;
366
367 return pointerEvent;
368}

370void PointerCaptureController::dispatchEvent(PointerEvent& event, EventTarget* target)
371{
372 if (!target || event.target())
373 return;
374
375 // https://w3c.github.io/pointerevents/#firing-events-using-the-pointerevent-interface
376 // If the event is not gotpointercapture or lostpointercapture, run Process Pending Pointer Capture steps for this PointerEvent.
377 // We only need to do this for non-mouse type since for mouse events this method will be called in Document::prepareMouseEvent().
378 if (event.pointerType() != mousePointerEventType())
379 processPendingPointerCapture(event.pointerId());
380
381 pointerEventWillBeDispatched(event, target);
382 target->dispatchEvent(event);
383 pointerEventWasDispatched(event);
384}
385
386void PointerCaptureController::pointerEventWillBeDispatched(const PointerEvent& event, EventTarget* target)
387{
388 if (!is<Element>(target))
389 return;
390
391 bool isPointerdown = event.type() == eventNames().pointerdownEvent;
392 bool isPointerup = event.type() == eventNames().pointerupEvent;
393 if (!isPointerdown && !isPointerup)
394 return;
395
396 auto pointerId = event.pointerId();
397
398 if (event.pointerType() != touchPointerEventType()) {
399 auto iterator = m_activePointerIdsToCapturingData.find(pointerId);
400 if (iterator != m_activePointerIdsToCapturingData.end())
401 iterator->value.pointerIsPressed = isPointerdown;
402 return;
403 }
404
405 if (!isPointerdown)
406 return;
407
408 // https://w3c.github.io/pointerevents/#implicit-pointer-capture
409
410 // Some input devices (such as touchscreens) implement a "direct manipulation" metaphor where a pointer is intended to act primarily on the UI
411 // element it became active upon (providing a physical illusion of direct contact, instead of indirect contact via a cursor that conceptually
412 // floats above the UI). Such devices are identified by the InputDeviceCapabilities.pointerMovementScrolls property and should have "implicit
413 // pointer capture" behavior as follows.
414
415 // Direct manipulation devices should behave exactly as if setPointerCapture was called on the target element just before the invocation of any
416 // pointerdown listeners. The hasPointerCapture API may be used (eg. within any pointerdown listener) to determine whether this has occurred. If
417 // releasePointerCapture is not called for the pointer before the next pointer event is fired, then a gotpointercapture event will be dispatched
418 // to the target (as normal) indicating that capture is active.
419
420 auto& capturingData = ensureCapturingDataForPointerEvent(event);
421 capturingData.pointerIsPressed = true;
422 setPointerCapture(downcast<Element>(target), pointerId);
423}

425PointerCaptureController::CapturingData& PointerCaptureController::ensureCapturingDataForPointerEvent(const PointerEvent& event)
426{
427 return m_activePointerIdsToCapturingData.ensure(event.pointerId(), [&event] {
428 CapturingData capturingData;
429 capturingData.pointerType = event.pointerType();
430 return capturingData;
431 }).iterator->value;
432}

434void PointerCaptureController::pointerEventWasDispatched(const PointerEvent& event)
435{
436 auto iterator = m_activePointerIdsToCapturingData.find(event.pointerId());
437 if (iterator != m_activePointerIdsToCapturingData.end()) {
438 auto& capturingData = iterator->value;
439 capturingData.isPrimary = event.isPrimary();
440
441 // Immediately after firing the pointerup or pointercancel events, a user agent MUST clear the pending pointer capture target
442 // override for the pointerId of the pointerup or pointercancel event that was just dispatched, and then run Process Pending
443 // Pointer Capture steps to fire lostpointercapture if necessary.
444 // https://w3c.github.io/pointerevents/#implicit-release-of-pointer-capture
445 if (event.type() == eventNames().pointerupEvent) {
446 capturingData.pendingTargetOverride = nullptr;
447 processPendingPointerCapture(event.pointerId());
448 }
449
450 // If a mouse pointer has moved while it isn't pressed, make sure we reset the preventsCompatibilityMouseEvents flag since
451 // we could otherwise prevent compatibility mouse events while those are only supposed to be prevented while the pointer is pressed.
452 if (event.type() == eventNames().pointermoveEvent && capturingData.pointerType == mousePointerEventType() && !capturingData.pointerIsPressed)
453 capturingData.preventsCompatibilityMouseEvents = false;
454
455 // If the pointer event dispatched was pointerdown and the event was canceled, then set the PREVENT MOUSE EVENT flag for this pointerType.
456 // https://www.w3.org/TR/pointerevents/#mapping-for-devices-that-support-hover
457 if (event.type() == eventNames().pointerdownEvent)
458 capturingData.preventsCompatibilityMouseEvents = event.defaultPrevented();
459 }

462void PointerCaptureController::cancelPointer(PointerID pointerId, const IntPoint& documentPoint)
463{
464 // https://w3c.github.io/pointerevents/#the-pointercancel-event
465
466 // A user agent MUST fire a pointer event named pointercancel in the following circumstances:
467 //
468 // The user agent has determined that a pointer is unlikely to continue to produce events (for example, because of a hardware event).
469 // After having fired the pointerdown event, if the pointer is subsequently used to manipulate the page viewport (e.g. panning or zooming).
470 // Immediately before drag operation starts [HTML], for the pointer that caused the drag operation.
471 // After firing the pointercancel event, a user agent MUST also fire a pointer event named pointerout followed by firing a pointer event named pointerleave.
472
473 // https://w3c.github.io/pointerevents/#implicit-release-of-pointer-capture
474
475 // Immediately after firing the pointerup or pointercancel events, a user agent MUST clear the pending pointer capture target
476 // override for the pointerId of the pointerup or pointercancel event that was just dispatched, and then run Process Pending
477 // Pointer Capture steps to fire lostpointercapture if necessary. After running Process Pending Pointer Capture steps, if the
478 // pointer supports hover, user agent MUST also send corresponding boundary events necessary to reflect the current position of
479 // the pointer with no capture.
480
481 auto iterator = m_activePointerIdsToCapturingData.find(pointerId);
482 if (iterator == m_activePointerIdsToCapturingData.end())
483 return;
484
485 auto& capturingData = iterator->value;
486 if (capturingData.state == CapturingData::State::Cancelled)
487 return;
488
489 capturingData.pendingTargetOverride = nullptr;
490 capturingData.state = CapturingData::State::Cancelled;
491
492#if ENABLE(TOUCH_EVENTS) && PLATFORM(IOS_FAMILY)
493 capturingData.previousTarget = nullptr;
494#endif
495
496 auto target = [&]() -> RefPtr<Element> {
497 if (capturingData.targetOverride)
498 return capturingData.targetOverride;
499 constexpr OptionSet<HitTestRequest::Type> hitType { HitTestRequest::Type::ReadOnly, HitTestRequest::Type::Active, HitTestRequest::Type::DisallowUserAgentShadowContent, HitTestRequest::Type::AllowChildFrameContent };
500 return m_page.mainFrame().eventHandler().hitTestResultAtPoint(documentPoint, hitType).innerNonSharedElement();
501 }();
502
503 if (!target)
504 return;
505
506 // After firing the pointercancel event, a user agent MUST also fire a pointer event named pointerout
507 // followed by firing a pointer event named pointerleave.
508 auto isPrimary = capturingData.isPrimary ? PointerEvent::IsPrimary::Yes : PointerEvent::IsPrimary::No;
509 auto cancelEvent = PointerEvent::create(eventNames().pointercancelEvent, pointerId, capturingData.pointerType, isPrimary);
510 target->dispatchEvent(cancelEvent);
511 target->dispatchEvent(PointerEvent::create(eventNames().pointeroutEvent, pointerId, capturingData.pointerType, isPrimary));
512 target->dispatchEvent(PointerEvent::create(eventNames().pointerleaveEvent, pointerId, capturingData.pointerType, isPrimary));
513 processPendingPointerCapture(pointerId);
514}

516void PointerCaptureController::processPendingPointerCapture(PointerID pointerId)
517{
518 auto iterator = m_activePointerIdsToCapturingData.find(pointerId);
519 if (iterator == m_activePointerIdsToCapturingData.end())
520 return;
521
522 if (m_processingPendingPointerCapture)
523 return;
524
525 m_processingPendingPointerCapture = true;
526
527 auto& capturingData = iterator->value;
528
529 // Cache the pending target override since it could be modified during the dispatch of events in this function.
530 auto pendingTargetOverride = capturingData.pendingTargetOverride;
531
532 // https://w3c.github.io/pointerevents/#process-pending-pointer-capture
533 // 1. If the pointer capture target override for this pointer is set and is not equal to the pending pointer capture target override,
534 // then fire a pointer event named lostpointercapture at the pointer capture target override node.
535 if (capturingData.targetOverride && capturingData.targetOverride != pendingTargetOverride) {
536 if (capturingData.targetOverride->isConnected())
537 capturingData.targetOverride->dispatchEvent(PointerEvent::createForPointerCapture(eventNames().lostpointercaptureEvent, pointerId, capturingData.isPrimary, capturingData.pointerType));
538 if (capturingData.pointerType == mousePointerEventType()) {
539 if (auto* frame = capturingData.targetOverride->document().frame())
540 frame->eventHandler().pointerCaptureElementDidChange(nullptr);
541 }
542 }
543
544 // 2. If the pending pointer capture target override for this pointer is set and is not equal to the pointer capture target override,
545 // then fire a pointer event named gotpointercapture at the pending pointer capture target override.
546 if (capturingData.pendingTargetOverride && capturingData.targetOverride != pendingTargetOverride) {
547 if (capturingData.pointerType == mousePointerEventType()) {
548 if (auto* frame = pendingTargetOverride->document().frame())
549 frame->eventHandler().pointerCaptureElementDidChange(pendingTargetOverride.get());
550 }
551 pendingTargetOverride->dispatchEvent(PointerEvent::createForPointerCapture(eventNames().gotpointercaptureEvent, pointerId, capturingData.isPrimary, capturingData.pointerType));
552 }
553
554 // 3. Set the pointer capture target override to the pending pointer capture target override, if set. Otherwise, clear the pointer
555 // capture target override.
556 capturingData.targetOverride = pendingTargetOverride;
557
558 m_processingPendingPointerCapture = false;
559}
