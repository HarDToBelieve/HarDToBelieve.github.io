37Ref<ResizeObservation> ResizeObservation::create(Element* target)

42ResizeObservation::ResizeObservation(Element* element)

45}

49}

51void ResizeObservation::updateObservationSize(const LayoutSize& size)
52{
53 m_lastObservationSize = size;
54}
55
56LayoutSize ResizeObservation::computeObservedSize() const
57{
58 if (m_target->isSVGElement()) {
59 if (auto svgRect = downcast<SVGElement>(*m_target).getBoundingBox())
60 return LayoutSize(svgRect->width(), svgRect->height());
61 }
62 auto* box = m_target->renderBox();
63 if (box) {
64 auto contentSize = box->contentSize();
65 return LayoutSize(adjustLayoutUnitForAbsoluteZoom(contentSize.width(), *box), adjustLayoutUnitForAbsoluteZoom(contentSize.height(), *box));
66 }
67
68 return LayoutSize();
69}
70
71LayoutPoint ResizeObservation::computeTargetLocation() const
72{
73 if (!m_target->isSVGElement()) {
74 if (auto box = m_target->renderBox())
75 return LayoutPoint(box->paddingLeft(), box->paddingTop());
76 }
77
78 return { };
79}
80
81FloatRect ResizeObservation::computeContentRect() const
82{
83 return FloatRect(FloatPoint(computeTargetLocation()), FloatSize(m_lastObservationSize));
84}
85
86bool ResizeObservation::elementSizeChanged(LayoutSize& currentSize) const
87{
88 currentSize = computeObservedSize();
89 return m_lastObservationSize != currentSize;
90}
91
92size_t ResizeObservation::targetElementDepth() const
93{
94 unsigned depth = 0;
95 for (Element* ownerElement = m_target; ownerElement; ownerElement = ownerElement->document().ownerElement()) {
96 for (Element* parent = ownerElement; parent; parent = parent->parentElement())
97 ++depth;
98 }
99
100 return depth;
101}
