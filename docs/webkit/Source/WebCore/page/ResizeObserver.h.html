30#include "ActiveDOMObject.h"
31#include "GCReachableRef.h"
32#include "ResizeObservation.h"

34#include <wtf/RefCounted.h>
35#include <wtf/WeakPtr.h>

37namespace WebCore {
38
39class Document;
40class Element;
41
42struct ResizeObserverData {
43 WTF_MAKE_STRUCT_FAST_ALLOCATED;
44 Vector<WeakPtr<ResizeObserver>> observers;
45};
46
47class ResizeObserver : public RefCounted<ResizeObserver>, public ActiveDOMObject, public CanMakeWeakPtr<ResizeObserver> {
48 WTF_MAKE_FAST_ALLOCATED;
49public:
50 static Ref<ResizeObserver> create(Document&, Ref<ResizeObserverCallback>&&);
51 ~ResizeObserver();
52
53 bool hasObservations() const { return m_observations.size(); }
54 bool hasActiveObservations() const { return m_activeObservations.size(); }
55
56 void observe(Element&);
57 void unobserve(Element&);
58 void disconnect();
59 void targetDestroyed(Element&);
60
61 static size_t maxElementDepth() { return SIZE_MAX; }
62 size_t gatherObservations(size_t depth);
63 void deliverObservations();
64 bool hasSkippedObservations() const { return m_hasSkippedObservations; }
65 void setHasSkippedObservations(bool skipped) { m_hasSkippedObservations = skipped; }
66
67 ResizeObserverCallback* callbackConcurrently() { return m_callback.get(); }
68
69private:
70 ResizeObserver(Document&, Ref<ResizeObserverCallback>&&);
71
72 // ActiveDOMObject.
73 bool virtualHasPendingActivity() const override;
74 const char* activeDOMObjectName() const override;
75 void stop() override;
76
77 bool removeTarget(Element&);
78 void removeAllTargets();
79 bool removeObservation(const Element&);
80
81 WeakPtr<Document> m_document;
82 RefPtr<ResizeObserverCallback> m_callback;
83 Vector<Ref<ResizeObservation>> m_observations;
84
85 Vector<Ref<ResizeObservation>> m_activeObservations;
86 Vector<GCReachableRef<Element>> m_pendingTargets;
87 bool m_hasSkippedObservations { false };
88};

90} // namespace WebCore
