53unsigned Screen::height() const
54{
55 auto* frame = this->frame();
56 if (!frame)
57 return 0;
58 if (RuntimeEnabledFeatures::sharedFeatures().webAPIStatisticsEnabled())
59 ResourceLoadObserver::shared().logScreenAPIAccessed(*frame->document(), ResourceLoadStatistics::ScreenAPI::Height);
60 return static_cast<unsigned>(frame->screenSize().height());
61}

83unsigned Screen::pixelDepth() const
84{
85 auto* frame = this->frame();
86 if (!frame)
87 return 0;
88 if (RuntimeEnabledFeatures::sharedFeatures().webAPIStatisticsEnabled())
89 ResourceLoadObserver::shared().logScreenAPIAccessed(*frame->document(), ResourceLoadStatistics::ScreenAPI::PixelDepth);
90
91 auto* document = window()->document();
92 if (!document || !document->quirks().needsHDRPixelDepthQuirk() || !screenSupportsHighDynamicRange(frame->view()))
93 return static_cast<unsigned>(screenDepth(frame->view()));
94
95 return static_cast<unsigned>(screenDepth(frame->view())) + 1;
96}
