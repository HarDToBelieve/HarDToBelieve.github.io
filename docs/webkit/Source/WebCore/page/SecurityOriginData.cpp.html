39String SecurityOriginData::toString() const
40{
41 if (protocol == "file")
42 return "file://"_s;
43
44 if (protocol.isEmpty() && host.isEmpty())
45 return { };
46
47 if (!port)
48 return makeString(protocol, "://", host);
49 return makeString(protocol, "://", host, ':', static_cast<uint32_t>(*port));
50}

52SecurityOriginData SecurityOriginData::fromFrame(Frame* frame)
53{
54 if (!frame)
55 return SecurityOriginData { };
56
57 auto* document = frame->document();
58 if (!document)
59 return SecurityOriginData { };
60
61 return document->securityOrigin().data();
62}

64Ref<SecurityOrigin> SecurityOriginData::securityOrigin() const

69static const char separatorCharacter = '_';
70
71String SecurityOriginData::databaseIdentifier() const
72{
73 // Historically, we've used the following (somewhat nonsensical) string
74 // for the databaseIdentifier of local files. We used to compute this
75 // string because of a bug in how we handled the scheme for file URLs.
76 // Now that we've fixed that bug, we produce this string for compatibility
77 // with existing persistent state.
78 if (equalIgnoringASCIICase(protocol, "file"))
79 return "file__0"_s;
80
81 return makeString(protocol, separatorCharacter, FileSystem::encodeForFileName(host), separatorCharacter, port.value_or(0));
82}
83
84std::optional<SecurityOriginData> SecurityOriginData::fromDatabaseIdentifier(const String& databaseIdentifier)
85{
86 // Make sure there's a first separator
87 size_t separator1 = databaseIdentifier.find(separatorCharacter);
88 if (separator1 == notFound)
89 return std::nullopt;
90
91 // Make sure there's a second separator
92 size_t separator2 = databaseIdentifier.reverseFind(separatorCharacter);
93 if (separator2 == notFound)
94 return std::nullopt;
95
96 // Ensure there were at least 2 separator characters. Some hostnames on intranets have
97 // underscores in them, so we'll assume that any additional underscores are part of the host.
98 if (separator1 == separator2)
99 return std::nullopt;
100
101 // Make sure the port section is a valid port number or doesn't exist.
102 auto portLength = databaseIdentifier.length() - separator2 - 1;
103 auto port = parseIntegerAllowingTrailingJunk<uint16_t>(StringView { databaseIdentifier }.right(portLength));
104
105 // Nothing after the colon is fine. Failure to parse after the colon is not.
106 if (!port && portLength)
107 return std::nullopt;
108
109 // Treat port 0 like there is was no port specified.
110 if (port && !*port)
111 port = std::nullopt;
112
113 auto protocol = databaseIdentifier.substring(0, separator1);
114 auto host = databaseIdentifier.substring(separator1 + 1, separator2 - separator1 - 1);
115 return SecurityOriginData { protocol, host, port };
116}

118SecurityOriginData SecurityOriginData::isolatedCopy() const
119{
120 SecurityOriginData result;
121
122 result.protocol = protocol.isolatedCopy();
123 result.host = host.isolatedCopy();
124 result.port = port;
125
126 return result;
127}
128
129bool operator==(const SecurityOriginData& a, const SecurityOriginData& b)
130{
131 if (&a == &b)
132 return true;
133
134 return a.protocol == b.protocol
135 && a.host == b.host
136 && a.port == b.port;
137}
