31#include <wtf/URL.h>
32#include "SecurityOrigin.h"
33#include <wtf/RefPtr.h>

35namespace WebCore {
36
37struct SecurityOriginHash {
38 static unsigned hash(SecurityOrigin* origin)
39 {
40 unsigned hashCodes[3] = {
41 origin->protocol().impl() ? origin->protocol().impl()->hash() : 0,
42 origin->host().impl() ? origin->host().impl()->hash() : 0,
43 origin->port().value_or(0)
44 };
45 return StringHasher::hashMemory<sizeof(hashCodes)>(hashCodes);
46 }
47 static unsigned hash(const RefPtr<SecurityOrigin>& origin)
48 {
49 return hash(origin.get());
50 }
51
52 static bool equal(SecurityOrigin* a, SecurityOrigin* b)
53 {
54 if (!a || !b)
55 return a == b;
56 return a->isSameSchemeHostPort(*b);
57 }
58 static bool equal(SecurityOrigin* a, const RefPtr<SecurityOrigin>& b)
59 {
60 return equal(a, b.get());
61 }
62 static bool equal(const RefPtr<SecurityOrigin>& a, SecurityOrigin* b)
63 {
64 return equal(a.get(), b);
65 }
66 static bool equal(const RefPtr<SecurityOrigin>& a, const RefPtr<SecurityOrigin>& b)
67 {
68 return equal(a.get(), b.get());
69 }
70
71 static const bool safeToCompareToEmptyOrDeleted = false;
72};
73
74} // namespace WebCore
75
76namespace WTF {
77
78template<typename> struct DefaultHash;
79template<> struct DefaultHash<RefPtr<WebCore::SecurityOrigin>> : WebCore::SecurityOriginHash { };

81} // namespace WTF
