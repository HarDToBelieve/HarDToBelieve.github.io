66bool SecurityPolicy::shouldHideReferrer(const URL& url, const String& referrer)
67{
68 bool referrerIsSecureURL = protocolIs(referrer, "https");
69 bool referrerIsWebURL = referrerIsSecureURL || protocolIs(referrer, "http");
70
71 if (!referrerIsWebURL)
72 return true;
73
74 if (!referrerIsSecureURL)
75 return false;
76
77 bool URLIsSecureURL = url.protocolIs("https");
78
79 return !URLIsSecureURL;
80}

82String SecurityPolicy::referrerToOriginString(const String& referrer)
83{
84 String originString = SecurityOrigin::createFromString(referrer)->toString();
85 if (originString == "null")
86 return String();
87 // A security origin is not a canonical URL as it lacks a path. Add /
88 // to turn it into a canonical URL we can use as referrer.
89 return originString + "/";
90}

92String SecurityPolicy::generateReferrerHeader(ReferrerPolicy referrerPolicy, const URL& url, const String& referrer)
93{
94 ASSERT(referrer == URL(URL(), referrer).strippedForUseAsReferrer()
95 || referrer == SecurityOrigin::create(URL(URL(), referrer))->toString());
96
97 if (referrer.isEmpty())
98 return String();
99
100 if (!protocolIsInHTTPFamily(referrer))
101 return String();
102
103 switch (referrerPolicy) {
104 case ReferrerPolicy::EmptyString:
105 ASSERT_NOT_REACHED();
106 break;
107 case ReferrerPolicy::NoReferrer:
108 return String();
109 case ReferrerPolicy::NoReferrerWhenDowngrade:
110 break;
111 case ReferrerPolicy::SameOrigin: {
112 auto origin = SecurityOrigin::createFromString(referrer);
113 if (!origin->canRequest(url))
114 return String();
115 break;
116 }
117 case ReferrerPolicy::Origin:
118 return referrerToOriginString(referrer);
119 case ReferrerPolicy::StrictOrigin:
120 if (shouldHideReferrer(url, referrer))
121 return String();
122 return referrerToOriginString(referrer);
123 case ReferrerPolicy::OriginWhenCrossOrigin: {
124 auto origin = SecurityOrigin::createFromString(referrer);
125 if (!origin->canRequest(url))
126 return referrerToOriginString(referrer);
127 break;
128 }
129 case ReferrerPolicy::StrictOriginWhenCrossOrigin: {
130 auto origin = SecurityOrigin::createFromString(referrer);
131 if (!origin->canRequest(url)) {
132 if (shouldHideReferrer(url, referrer))
133 return String();
134 return referrerToOriginString(referrer);
135 }
136 break;
137 }
138 case ReferrerPolicy::UnsafeUrl:
139 return referrer;
140 }
141
142 return shouldHideReferrer(url, referrer) ? String() : referrer;
143}

145String SecurityPolicy::generateOriginHeader(ReferrerPolicy referrerPolicy, const URL& url, const SecurityOrigin& securityOrigin)
146{
147 switch (referrerPolicy) {
148 case ReferrerPolicy::NoReferrer:
149 return "null"_s;
150 case ReferrerPolicy::NoReferrerWhenDowngrade:
151 case ReferrerPolicy::StrictOrigin:
152 case ReferrerPolicy::StrictOriginWhenCrossOrigin:
153 if (protocolIs(securityOrigin.protocol(), "https") && !url.protocolIs("https"))
154 return "null"_s;
155 break;
156 case ReferrerPolicy::SameOrigin:
157 if (!securityOrigin.canRequest(url))
158 return "null"_s;
159 break;
160 case ReferrerPolicy::EmptyString:
161 case ReferrerPolicy::Origin:
162 case ReferrerPolicy::OriginWhenCrossOrigin:
163 case ReferrerPolicy::UnsafeUrl:
164 break;
165 }
166
167 return securityOrigin.toString();
168}

170bool SecurityPolicy::shouldInheritSecurityOriginFromOwner(const URL& url)
171{
172 // Paraphrased from <https://html.spec.whatwg.org/multipage/browsers.html#origin> (8 July 2016)
173 //
174 // If a Document has the address "about:blank"
175 // The origin of the document is the origin it was assigned when its browsing context was created.
176 // If a Document has the address "about:srcdoc"
177 // The origin of the document is the origin of its parent document.
178 //
179 // Note: We generalize this to invalid URLs because we treat such URLs as about:blank.
180 // FIXME: We also allow some URLs like "about:BLANK". We should probably block navigation to these URLs, see rdar://problem/57966056.
181 return url.isEmpty() || url.isAboutBlank() || url.isAboutSrcDoc() || equalIgnoringASCIICase(url.string(), aboutBlankURL().string());
182}

184bool SecurityPolicy::isBaseURLSchemeAllowed(const URL& url)
185{
186 // See <https://github.com/whatwg/html/issues/2249>.
187 return !url.protocolIsData() && !url.protocolIsJavaScript();
188}

205bool SecurityPolicy::isAccessAllowed(const SecurityOrigin& activeOrigin, const SecurityOrigin& targetOrigin, const URL& targetURL)
206{
207 ASSERT(targetOrigin.equal(SecurityOrigin::create(targetURL).ptr()));
208 {
209 Locker locker { originAccessMapLock };
210 if (auto* list = originAccessMap().get(activeOrigin.data())) {
211 for (auto& entry : *list) {
212 if (entry.matchesOrigin(targetOrigin))
213 return true;
214 }
215 }
216 }
217 Locker locker { originAccessPatternLock };
218 for (const auto& pattern : originAccessPatterns()) {
219 if (pattern.matches(targetURL))
220 return true;
221 }
222 return false;
223}
224
225bool SecurityPolicy::isAccessAllowed(const SecurityOrigin& activeOrigin, const URL& url)
226{
227 return isAccessAllowed(activeOrigin, SecurityOrigin::create(url).get(), url);
228}

230void SecurityPolicy::addOriginAccessAllowlistEntry(const SecurityOrigin& sourceOrigin, const String& destinationProtocol, const String& destinationDomain, bool allowDestinationSubdomains)
231{
232 ASSERT(!sourceOrigin.isUnique());
233 if (sourceOrigin.isUnique())
234 return;
235
236 Locker locker { originAccessMapLock };
237 auto* allowList = originAccessMap().ensure(sourceOrigin.data(), [] { return makeUnique<OriginAccessAllowlist>(); }).iterator->value.get();
238 allowList->append(OriginAccessEntry(destinationProtocol, destinationDomain, allowDestinationSubdomains ? OriginAccessEntry::AllowSubdomains : OriginAccessEntry::DisallowSubdomains, OriginAccessEntry::TreatIPAddressAsIPAddress));
239}
240
241void SecurityPolicy::removeOriginAccessAllowlistEntry(const SecurityOrigin& sourceOrigin, const String& destinationProtocol, const String& destinationDomain, bool allowDestinationSubdomains)
242{
243 ASSERT(!sourceOrigin.isUnique());
244 if (sourceOrigin.isUnique())
245 return;
246
247 Locker locker { originAccessMapLock };
248 auto& map = originAccessMap();
249 auto it = map.find(sourceOrigin.data());
250 if (it == map.end())
251 return;
252
253 auto& list = *it->value;
254 OriginAccessEntry originAccessEntry(destinationProtocol, destinationDomain, allowDestinationSubdomains ? OriginAccessEntry::AllowSubdomains : OriginAccessEntry::DisallowSubdomains, OriginAccessEntry::TreatIPAddressAsIPAddress);
255 if (!list.removeFirst(originAccessEntry))
256 return;
257
258 if (list.isEmpty())
259 map.remove(it);
260}
