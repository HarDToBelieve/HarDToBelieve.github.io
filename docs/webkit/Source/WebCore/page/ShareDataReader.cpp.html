46void ShareDataReader::start(Document* document, ShareDataWithParsedURL&& shareData)
47{
48 m_filesReadSoFar = 0;
49 m_shareData = WTFMove(shareData);
50 int count = 0;
51 m_pendingFileLoads.reserveInitialCapacity(m_shareData.shareData.files.size());
52 for (auto& blob : m_shareData.shareData.files) {
53 m_pendingFileLoads.uncheckedAppend(makeUniqueRef<BlobLoader>([this, count, fileName = blob->name()](BlobLoader&) {
54 this->didFinishLoading(count, fileName);
55 }));
56 m_pendingFileLoads.last()->start(*blob, document, FileReaderLoader::ReadAsArrayBuffer);
57 ++count;
58 }
59}
60
61void ShareDataReader::didFinishLoading(int loadIndex, const String& fileName)
62{
63 if (m_pendingFileLoads.isEmpty()) {
64 // cancel() was called.
65 return;
66 }
67
68 if (m_pendingFileLoads[loadIndex]->errorCode()) {
69 if (auto completionHandler = std::exchange(m_completionHandler, { }))
70 completionHandler(Exception { AbortError, "Abort due to error while reading files."_s });
71 cancel();
72 return;
73 }
74
75 auto arrayBuffer = m_pendingFileLoads[loadIndex]->arrayBufferResult();
76
77 RawFile file;
78 file.fileName = fileName;
79 file.fileData = SharedBuffer::create(static_cast<const unsigned char*>(arrayBuffer->data()), arrayBuffer->byteLength());
80 m_shareData.files.append(WTFMove(file));
81 m_filesReadSoFar++;
82
83 if (m_filesReadSoFar == static_cast<int>(m_pendingFileLoads.size())) {
84 m_pendingFileLoads.clear();
85 if (auto completionHandler = std::exchange(m_completionHandler, { }))
86 completionHandler(m_shareData);
87 }
88}
89
90void ShareDataReader::cancel()
91{
92 // Don't call m_pendingFileLoads.clear() here since destroying a BlobLoader will cause its completion handler
93 // to get called, which will call didFinishLoading() and try to access m_pendingFileLoads.
94 std::exchange(m_pendingFileLoads, { });
95}
