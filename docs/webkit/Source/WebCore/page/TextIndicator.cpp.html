70RefPtr<TextIndicator> TextIndicator::createWithRange(const SimpleRange& range, OptionSet<TextIndicatorOption> options, TextIndicatorPresentationTransition presentationTransition, FloatSize margin)
71{
72 auto frame = makeRefPtr(range.startContainer().document().frame());
73 if (!frame)
74 return nullptr;
75
76 auto document = makeRefPtr(frame->document());
77 if (!document)
78 return nullptr;
79
80 bool indicatesCurrentSelection = range == document->selection().selection().toNormalizedRange();
81
82 OptionSet<TemporarySelectionOption> temporarySelectionOptions;
83 temporarySelectionOptions.add(TemporarySelectionOption::DoNotSetFocus);
84#if PLATFORM(IOS_FAMILY)
85 temporarySelectionOptions.add(TemporarySelectionOption::IgnoreSelectionChanges);
86 temporarySelectionOptions.add(TemporarySelectionOption::EnableAppearanceUpdates);
87#endif
88 TemporarySelectionChange selectionChange(*document, { range }, temporarySelectionOptions);
89
90 TextIndicatorData data;
91
92 data.presentationTransition = presentationTransition;
93 data.options = options;
94
95 if (!initializeIndicator(data, *frame, range, margin, indicatesCurrentSelection))
96 return nullptr;
97
98 return TextIndicator::create(data);
99}

101RefPtr<TextIndicator> TextIndicator::createWithSelectionInFrame(Frame& frame, OptionSet<TextIndicatorOption> options, TextIndicatorPresentationTransition presentationTransition, FloatSize margin)
102{
103 auto range = frame.selection().selection().toNormalizedRange();
104 if (!range)
105 return nullptr;
106
107 TextIndicatorData data;
108
109 data.presentationTransition = presentationTransition;
110 data.options = options;
111
112 if (!initializeIndicator(data, frame, *range, margin, true))
113 return nullptr;
114
115 return TextIndicator::create(data);
116}

118static bool hasNonInlineOrReplacedElements(const SimpleRange& range)
119{
120 for (auto& node : intersectingNodes(range)) {
121 auto renderer = node.renderer();
122 if (renderer && (!renderer->isInline() || renderer->isReplaced()))
123 return true;
124 }
125 return false;
126}

128static SnapshotOptions snapshotOptionsForTextIndicatorOptions(OptionSet<TextIndicatorOption> options)
129{
130 SnapshotOptions snapshotOptions { { SnapshotFlags::PaintWithIntegralScaleFactor }, PixelFormat::BGRA8, DestinationColorSpace::SRGB() };
131
132 if (!options.contains(TextIndicatorOption::PaintAllContent)) {
133 if (options.contains(TextIndicatorOption::PaintBackgrounds))
134 snapshotOptions.flags.add(SnapshotFlags::PaintSelectionAndBackgroundsOnly);
135 else {
136 snapshotOptions.flags.add(SnapshotFlags::PaintSelectionOnly);
137
138 if (!options.contains(TextIndicatorOption::RespectTextColor))
139 snapshotOptions.flags.add(SnapshotFlags::ForceBlackText);
140 }
141 } else
142 snapshotOptions.flags.add(SnapshotFlags::ExcludeSelectionHighlighting);
143
144 return snapshotOptions;
145}

147static RefPtr<Image> takeSnapshot(Frame& frame, IntRect rect, SnapshotOptions&& options, float& scaleFactor, const Vector<FloatRect>& clipRectsInDocumentCoordinates)
148{
149 auto buffer = snapshotFrameRectWithClip(frame, rect, clipRectsInDocumentCoordinates, WTFMove(options));
150 if (!buffer)
151 return nullptr;
152 scaleFactor = buffer->resolutionScale();
153 return ImageBuffer::sinkIntoImage(WTFMove(buffer), PreserveResolution::Yes);
154}

156static bool takeSnapshots(TextIndicatorData& data, Frame& frame, IntRect snapshotRect, const Vector<FloatRect>& clipRectsInDocumentCoordinates)
157{
158 data.contentImage = takeSnapshot(frame, snapshotRect, snapshotOptionsForTextIndicatorOptions(data.options), data.contentImageScaleFactor, clipRectsInDocumentCoordinates);
159 if (!data.contentImage)
160 return false;
161
162 if (data.options.contains(TextIndicatorOption::IncludeSnapshotWithSelectionHighlight)) {
163 float snapshotScaleFactor;
164 data.contentImageWithHighlight = takeSnapshot(frame, snapshotRect, { { }, PixelFormat::BGRA8, DestinationColorSpace::SRGB() }, snapshotScaleFactor, clipRectsInDocumentCoordinates);
165 ASSERT(!data.contentImageWithHighlight || data.contentImageScaleFactor >= snapshotScaleFactor);
166 }
167
168 if (data.options.contains(TextIndicatorOption::IncludeSnapshotOfAllVisibleContentWithoutSelection)) {
169 float snapshotScaleFactor;
170 auto snapshotRect = frame.view()->visibleContentRect();
171 data.contentImageWithoutSelection = takeSnapshot(frame, snapshotRect, { { SnapshotFlags::PaintEverythingExcludingSelection }, PixelFormat::BGRA8, DestinationColorSpace::SRGB() }, snapshotScaleFactor, { });
172 data.contentImageWithoutSelectionRectInRootViewCoordinates = frame.view()->contentsToRootView(snapshotRect);
173 }
174
175 return true;
176}

178static bool styleContainsComplexBackground(const RenderStyle& style)
179{
180 return style.hasBlendMode() || style.hasBackgroundImage() || style.hasBackdropFilter();
181}

183static HashSet<Color> estimatedTextColorsForRange(const SimpleRange& range)
184{
185 HashSet<Color> colors;
186 for (TextIterator iterator(range); !iterator.atEnd(); iterator.advance()) {
187 auto node = iterator.node();
188 if (!node)
189 continue;
190 auto renderer = node->renderer();
191 if (is<RenderText>(renderer))
192 colors.add(renderer->style().color());
193 }
194 return colors;
195}

197static FloatRect absoluteBoundingRectForRange(const SimpleRange& range)

199 return unionRectIgnoringZeroRects(RenderObject::absoluteBorderAndTextRects(range, {

201 RenderObject::BoundingRectBehavior::UseVisibleBounds,

206static Color estimatedBackgroundColorForRange(const SimpleRange& range, const Frame& frame)
207{
208 auto estimatedBackgroundColor = frame.view() ? frame.view()->documentBackgroundColor() : Color::transparentBlack;
209
210 RenderElement* renderer = nullptr;
211 auto commonAncestor = commonInclusiveAncestor<ComposedTree>(range);
212 while (commonAncestor) {
213 if (is<RenderElement>(commonAncestor->renderer())) {
214 renderer = downcast<RenderElement>(commonAncestor->renderer());
215 break;
216 }
217 commonAncestor = commonAncestor->parentOrShadowHostElement();
218 }
219
220 auto boundingRectForRange = enclosingIntRect(absoluteBoundingRectForRange(range));
221 Vector<Color> parentRendererBackgroundColors;
222 for (; !!renderer; renderer = renderer->parent()) {
223 auto absoluteBoundingBox = renderer->absoluteBoundingBoxRect();
224 auto& style = renderer->style();
225 if (!absoluteBoundingBox.contains(boundingRectForRange) || !style.hasBackground())
226 continue;
227
228 if (styleContainsComplexBackground(style))
229 return estimatedBackgroundColor;
230
231 auto visitedDependentBackgroundColor = style.visitedDependentColor(CSSPropertyBackgroundColor);
232 if (visitedDependentBackgroundColor != Color::transparentBlack)
233 parentRendererBackgroundColors.append(visitedDependentBackgroundColor);
234 }
235 parentRendererBackgroundColors.reverse();
236 for (const auto& backgroundColor : parentRendererBackgroundColors)
237 estimatedBackgroundColor = blendSourceOver(estimatedBackgroundColor, backgroundColor);
238
239 return estimatedBackgroundColor;
240}

242static bool hasAnyIllegibleColors(TextIndicatorData& data, const Color& backgroundColor, HashSet<Color>&& textColors)
243{
244 if (data.options.contains(TextIndicatorOption::PaintAllContent))
245 return false;
246
247 if (!data.options.contains(TextIndicatorOption::UseBoundingRectAndPaintAllContentForComplexRanges))
248 return false;
249
250 if (!data.options.contains(TextIndicatorOption::ComputeEstimatedBackgroundColor))
251 return false;
252
253 bool hasOnlyLegibleTextColors = true;
254 if (data.options.contains(TextIndicatorOption::RespectTextColor)) {
255 for (auto& textColor : textColors) {
256 hasOnlyLegibleTextColors = textColorIsLegibleAgainstBackgroundColor(textColor, backgroundColor);
257 if (!hasOnlyLegibleTextColors)
258 break;
259 }
260 } else
261 hasOnlyLegibleTextColors = textColorIsLegibleAgainstBackgroundColor(Color::black, backgroundColor);
262
263 return !hasOnlyLegibleTextColors || textColors.isEmpty();
264}

266static bool containsOnlyWhiteSpaceText(const SimpleRange& range)
267{
268 for (auto& node : intersectingNodes(range)) {
269 if (!is<RenderText>(node.renderer()))
270 return false;
271 }
272 return plainTextReplacingNoBreakSpace(range).stripWhiteSpace().isEmpty();
273}

275static bool initializeIndicator(TextIndicatorData& data, Frame& frame, const SimpleRange& range, FloatSize margin, bool indicatesCurrentSelection)
276{
277 if (auto* document = frame.document())
278 document->updateLayoutIgnorePendingStylesheets();
279
280 bool treatRangeAsComplexDueToIllegibleTextColors = false;
281 if (data.options.contains(TextIndicatorOption::ComputeEstimatedBackgroundColor)) {
282 data.estimatedBackgroundColor = estimatedBackgroundColorForRange(range, frame);
283 treatRangeAsComplexDueToIllegibleTextColors = hasAnyIllegibleColors(data, data.estimatedBackgroundColor, estimatedTextColorsForRange(range));
284 }
285
286 // FIXME (138888): Ideally we wouldn't remove the margin in this case, but we need to
287 // ensure that the indicator and indicator-with-highlight overlap precisely, and
288 // we can't add a margin to the indicator-with-highlight.
289 if (indicatesCurrentSelection && !data.options.contains(TextIndicatorOption::IncludeMarginIfRangeMatchesSelection))
290 margin = FloatSize();
291
292 Vector<FloatRect> textRects;
293
294 bool useBoundingRectAndPaintAllContentForComplexRanges = data.options.contains(TextIndicatorOption::UseBoundingRectAndPaintAllContentForComplexRanges);
295 if (useBoundingRectAndPaintAllContentForComplexRanges && containsOnlyWhiteSpaceText(range)) {
296 if (auto* containerRenderer = commonInclusiveAncestor<ComposedTree>(range)->renderer()) {
297 data.options.add(TextIndicatorOption::PaintAllContent);
298 textRects.append(containerRenderer->absoluteBoundingBoxRect());
299 }
300 } else if (useBoundingRectAndPaintAllContentForComplexRanges && (treatRangeAsComplexDueToIllegibleTextColors || hasNonInlineOrReplacedElements(range)))
301 data.options.add(TextIndicatorOption::PaintAllContent);
302#if PLATFORM(IOS_FAMILY)
303 else if (data.options.contains(TextIndicatorOption::UseSelectionRectForSizing)) {
304 textRects = RenderObject::collectSelectionGeometries(range).map([&](auto& geometry) -> FloatRect {
305 return geometry.rect();
306 });
307 }
308#endif
309 else {
310 OptionSet<RenderObject::BoundingRectBehavior> behavior { RenderObject::BoundingRectBehavior::RespectClipping };
311 if (!data.options.contains(TextIndicatorOption::TightlyFitContent))
312 behavior.add(RenderObject::BoundingRectBehavior::UseSelectionHeight);
313 textRects = RenderObject::absoluteTextRects(range, behavior).map([&](auto& rect) -> FloatRect {
314 return rect;
315 });
316 }
317
318 if (textRects.isEmpty())
319 textRects.append(absoluteBoundingRectForRange(range));
320
321 auto frameView = frame.view();
322
323 // Use the exposedContentRect/viewExposedRect instead of visibleContentRect to avoid creating a huge indicator for a large view inside a scroll view.
324 IntRect contentsClipRect;
325#if PLATFORM(IOS_FAMILY)
326 contentsClipRect = enclosingIntRect(frameView->exposedContentRect());
327#else
328 if (auto viewExposedRect = frameView->viewExposedRect())
329 contentsClipRect = enclosingIntRect(*viewExposedRect);
330 else
331 contentsClipRect = frameView->visibleContentRect();
332#endif
333
334 if (data.options.contains(TextIndicatorOption::ExpandClipBeyondVisibleRect)) {
335 contentsClipRect.inflateX(contentsClipRect.width() / 2);
336 contentsClipRect.inflateY(contentsClipRect.height() / 2);
337 }
338
339 FloatRect textBoundingRectInRootViewCoordinates;
340 FloatRect textBoundingRectInDocumentCoordinates;
341 Vector<FloatRect> clippedTextRectsInDocumentCoordinates;
342 Vector<FloatRect> textRectsInRootViewCoordinates;
343 for (const FloatRect& textRect : textRects) {
344 FloatRect clippedTextRect;
345 if (data.options.contains(TextIndicatorOption::DoNotClipToVisibleRect))
346 clippedTextRect = textRect;
347 else
348 clippedTextRect = intersection(textRect, contentsClipRect);
349 if (clippedTextRect.isEmpty())
350 continue;
351
352 clippedTextRectsInDocumentCoordinates.append(clippedTextRect);
353
354 FloatRect textRectInDocumentCoordinatesIncludingMargin = clippedTextRect;
355 textRectInDocumentCoordinatesIncludingMargin.inflateX(margin.width());
356 textRectInDocumentCoordinatesIncludingMargin.inflateY(margin.height());
357 textBoundingRectInDocumentCoordinates.unite(textRectInDocumentCoordinatesIncludingMargin);
358
359 FloatRect textRectInRootViewCoordinates = frame.view()->contentsToRootView(enclosingIntRect(textRectInDocumentCoordinatesIncludingMargin));
360 textRectsInRootViewCoordinates.append(textRectInRootViewCoordinates);
361 textBoundingRectInRootViewCoordinates.unite(textRectInRootViewCoordinates);
362 }
363
364 Vector<FloatRect> textRectsInBoundingRectCoordinates;
365 for (auto rect : textRectsInRootViewCoordinates) {
366 rect.moveBy(-textBoundingRectInRootViewCoordinates.location());
367 textRectsInBoundingRectCoordinates.append(rect);
368 }
369
370 // Store the selection rect in window coordinates, to be used subsequently
371 // to determine if the indicator and selection still precisely overlap.
372 data.selectionRectInRootViewCoordinates = frame.view()->contentsToRootView(enclosingIntRect(frame.selection().selectionBounds(FrameSelection::ClipToVisibleContent::No)));
373 data.textBoundingRectInRootViewCoordinates = textBoundingRectInRootViewCoordinates;
374 data.textRectsInBoundingRectCoordinates = textRectsInBoundingRectCoordinates;
375
376 return takeSnapshots(data, frame, enclosingIntRect(textBoundingRectInDocumentCoordinates), clippedTextRectsInDocumentCoordinates);
377}
