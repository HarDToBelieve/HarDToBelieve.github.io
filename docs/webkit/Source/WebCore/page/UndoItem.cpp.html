26#include "config.h"
27#include "UndoItem.h"
28
29#include "Document.h"
30#include "UndoManager.h"
31#include <wtf/IsoMallocInlines.h>
32
33namespace WebCore {
34
35WTF_MAKE_ISO_ALLOCATED_IMPL(UndoItem);
36
37UndoManager* UndoItem::undoManager() const
38{
39 return m_undoManager.get();
40}
41
42void UndoItem::setUndoManager(UndoManager* undoManager)
43{
44 m_undoManager = makeWeakPtr(undoManager);
45 m_document = makeWeakPtr(undoManager ? &undoManager->document() : nullptr);
46}
47
48void UndoItem::invalidate()
49{
50 if (m_undoManager)
51 m_undoManager->removeItem(*this);
52 m_undoManager.clear();
53 m_document.clear();
54}
55
56bool UndoItem::isValid() const
57{
58 return !!m_undoManager;
59}
60
61Document* UndoItem::document() const
62{
63 return m_document.get();
64}
