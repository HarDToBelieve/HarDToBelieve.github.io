35bool UserContentURLPattern::matchesPatterns(const URL& url, const Vector<String>& allowlist, const Vector<String>& blocklist)
36{
37 // In order for a URL to be a match it has to be present in the allowlist and not present in the blocklist.
38 // If there is no allowlist at all, then all URLs are assumed to be in the allowlist.
39 bool matchesAllowlist = allowlist.isEmpty();
40 if (!matchesAllowlist) {
41 for (auto& entry : allowlist) {
42 UserContentURLPattern contentPattern(entry);
43 if (contentPattern.matches(url)) {
44 matchesAllowlist = true;
45 break;
46 }
47 }
48 }
49
50 bool matchesBlocklist = false;
51 if (!blocklist.isEmpty()) {
52 for (auto& entry : blocklist) {
53 UserContentURLPattern contentPattern(entry);
54 if (contentPattern.matches(url)) {
55 matchesBlocklist = true;
56 break;
57 }
58 }
59 }
60
61 return matchesAllowlist && !matchesBlocklist;
62}

64bool UserContentURLPattern::parse(const String& pattern)
65{
66 static NeverDestroyed<const String> schemeSeparator(MAKE_STATIC_STRING_IMPL("://"));
67
68 size_t schemeEndPos = pattern.find(schemeSeparator);
69 if (schemeEndPos == notFound)
70 return false;
71
72 m_scheme = pattern.left(schemeEndPos);
73
74 unsigned hostStartPos = schemeEndPos + schemeSeparator.get().length();
75 if (hostStartPos >= pattern.length())
76 return false;
77
78 int pathStartPos = 0;
79
80 if (equalLettersIgnoringASCIICase(m_scheme, "file"))
81 pathStartPos = hostStartPos;
82 else {
83 size_t hostEndPos = pattern.find('/', hostStartPos);
84 if (hostEndPos == notFound)
85 return false;
86
87 m_host = pattern.substring(hostStartPos, hostEndPos - hostStartPos);
88 m_matchSubdomains = false;
89
90 if (m_host == "*") {
91 // The pattern can be just '*', which means match all domains.
92 m_host = emptyString();
93 m_matchSubdomains = true;
94 } else if (m_host.startsWith("*.")) {
95 // The first component can be '*', which means to match all subdomains.
96 m_host = m_host.substring(2); // Length of "*."
97 m_matchSubdomains = true;
98 }
99
100 // No other '*' can occur in the host.
101 if (m_host.find('*') != notFound)
102 return false;
103
104 pathStartPos = hostEndPos;
105 }
106
107 m_path = pattern.right(pattern.length() - pathStartPos);
108
109 return true;
110}

112bool UserContentURLPattern::matches(const URL& test) const
113{
114 if (m_invalid)
115 return false;
116
117 if (m_scheme != "*" && !equalIgnoringASCIICase(test.protocol(), m_scheme))
118 return false;
119
120 if (!equalLettersIgnoringASCIICase(m_scheme, "file") && !matchesHost(test))
121 return false;
122
123 return matchesPath(test);
124}

126bool UserContentURLPattern::matchesHost(const URL& test) const
127{
128 auto host = test.host();
129 if (equalIgnoringASCIICase(host, m_host))
130 return true;
131
132 if (!m_matchSubdomains)
133 return false;
134
135 // If we're matching subdomains, and we have no host, that means the pattern
136 // was <scheme>://*/<whatever>, so we match anything.
137 if (!m_host.length())
138 return true;
139
140 // Check if the domain is a subdomain of our host.
141 if (!host.endsWithIgnoringASCIICase(m_host))
142 return false;
143
144 ASSERT(host.length() > m_host.length());
145
146 // Check that the character before the suffix is a period.
147 return host[host.length() - m_host.length() - 1] == '.';
148}

150struct MatchTester {
151 StringView m_pattern;
152 unsigned m_patternIndex { 0 };
153
154 StringView m_test;
155 unsigned m_testIndex { 0 };
156
157 MatchTester(StringView pattern, StringView test)
158 : m_pattern(pattern)
159 , m_test(test)
160 {
161 }
162
163 bool testStringFinished() const { return m_testIndex >= m_test.length(); }
164 bool patternStringFinished() const { return m_patternIndex >= m_pattern.length(); }
165
166 void eatWildcard()
167 {
168 while (!patternStringFinished()) {
169 if (m_pattern[m_patternIndex] != '*')
170 return;
171 m_patternIndex++;
172 }
173 }
174
175 void eatSameChars()
176 {
177 while (!patternStringFinished() && !testStringFinished()) {
178 if (m_pattern[m_patternIndex] == '*')
179 return;
180 if (m_pattern[m_patternIndex] != m_test[m_testIndex])
181 return;
182 m_patternIndex++;
183 m_testIndex++;
184 }
185 }
186
187 bool test()
188 {
189 // Eat all the matching chars.
190 eatSameChars();
191
192 // If the string is finished, then the pattern must be empty too, or contains
193 // only wildcards.
194 if (testStringFinished()) {
195 eatWildcard();
196 if (patternStringFinished())
197 return true;
198 return false;
199 }
200
201 // Pattern is empty but not string, this is not a match.
202 if (patternStringFinished())
203 return false;
204
205 // If we don't encounter a *, then we're hosed.
206 if (m_pattern[m_patternIndex] != '*')
207 return false;
208
209 while (!testStringFinished()) {
210 MatchTester nextMatch(*this);
211 nextMatch.m_patternIndex++;
212 if (nextMatch.test())
213 return true;
214 m_testIndex++;
215 }
216
217 // We reached the end of the string. Let's see if the pattern contains only
218 // wildcards.
219 eatWildcard();
220 return patternStringFinished();
221 }
222};
223
224bool UserContentURLPattern::matchesPath(const URL& test) const
225{
226 return MatchTester(m_path, test.path()).test();
227}
