66static bool needsUpdateAfterChangingDisabledAdaptations(const OptionSet<DisabledAdaptations>& oldDisabledAdaptations, const OptionSet<DisabledAdaptations>& newDisabledAdaptations)
67{
68 if (oldDisabledAdaptations == newDisabledAdaptations)
69 return false;
70
71#if PLATFORM(WATCHOS)
72 if (oldDisabledAdaptations.contains(DisabledAdaptations::Watch) != newDisabledAdaptations.contains(DisabledAdaptations::Watch))
73 return true;
74#endif
75
76 return false;
77}

79ViewportConfiguration::ViewportConfiguration()
80 : m_minimumLayoutSize(1024, 768)
81 , m_viewLayoutSize(1024, 768)
82 , m_canIgnoreScalingConstraints(false)
83 , m_forceAlwaysUserScalable(false)
84{
85 // Setup a reasonable default configuration to avoid computing infinite scale/sizes.
86 // Those are the original iPhone configuration.
87 m_defaultConfiguration = ViewportConfiguration::webpageParameters();
88 updateConfiguration();
89}

91void ViewportConfiguration::setDefaultConfiguration(const ViewportConfiguration::Parameters& defaultConfiguration)
92{
93 ASSERT(!constraintsAreAllRelative(m_configuration));
94 ASSERT(!defaultConfiguration.initialScaleIsSet || defaultConfiguration.initialScale > 0);
95 ASSERT(defaultConfiguration.minimumScale > 0);
96 ASSERT(defaultConfiguration.maximumScale >= defaultConfiguration.minimumScale);
97
98 if (m_defaultConfiguration == defaultConfiguration)
99 return;
100
101 m_defaultConfiguration = defaultConfiguration;
102 updateConfiguration();
103}

117bool ViewportConfiguration::setViewLayoutSize(const FloatSize& viewLayoutSize, std::optional<double>&& scaleFactor, std::optional<double>&& minimumEffectiveDeviceWidth)
118{
119 double newScaleFactor = scaleFactor.value_or(m_layoutSizeScaleFactor);
120 double newEffectiveWidth = minimumEffectiveDeviceWidth.value_or(m_minimumEffectiveDeviceWidth);
121 if (m_viewLayoutSize == viewLayoutSize && m_layoutSizeScaleFactor == newScaleFactor && newEffectiveWidth == m_minimumEffectiveDeviceWidth)
122 return false;
123
124 m_layoutSizeScaleFactor = newScaleFactor;
125 m_viewLayoutSize = viewLayoutSize;
126 m_minimumEffectiveDeviceWidth = newEffectiveWidth;
127
128 updateMinimumLayoutSize();
129 updateConfiguration();
130 return true;
131}

190bool ViewportConfiguration::shouldOverrideDeviceWidthAndShrinkToFit() const
191{
192 if (m_disabledAdaptations.contains(DisabledAdaptations::Watch))
193 return false;
194
195 auto viewWidth = m_viewLayoutSize.width();
196 return 0 < viewWidth && viewWidth < platformDeviceWidthOverride();
197}

199bool ViewportConfiguration::shouldIgnoreHorizontalScalingConstraints() const
200{
201 if (!m_canIgnoreScalingConstraints)
202 return false;
203
204 if (shouldOverrideDeviceWidthAndShrinkToFit())
205 return true;
206
207 if (!m_configuration.allowsShrinkToFit)
208 return false;
209
210 bool laidOutWiderThanViewport = m_contentSize.width() > layoutWidth();
211 if (m_viewportArguments.width == ViewportArguments::ValueDeviceWidth)
212 return laidOutWiderThanViewport;
213
214 if (m_configuration.initialScaleIsSet && m_configuration.initialScaleIgnoringLayoutScaleFactor == 1)
215 return laidOutWiderThanViewport;
216
217 return false;
218}

220bool ViewportConfiguration::shouldIgnoreVerticalScalingConstraints() const
221{
222 if (!m_canIgnoreScalingConstraints)
223 return false;
224
225 if (!m_configuration.allowsShrinkToFit)
226 return false;
227
228 bool laidOutTallerThanViewport = m_contentSize.height() > layoutHeight();
229 if (m_viewportArguments.height == ViewportArguments::ValueDeviceHeight && m_viewportArguments.width == ViewportArguments::ValueAuto)
230 return laidOutTallerThanViewport;
231
232 return false;
233}

245double ViewportConfiguration::initialScaleFromSize(double width, double height, bool shouldIgnoreScalingConstraints) const
246{
247 ASSERT(!constraintsAreAllRelative(m_configuration));
248
249 auto clampToMinimumAndMaximumScales = [&] (double initialScale) {
250 return clampTo<double>(initialScale, shouldIgnoreScalingConstraints ? m_defaultConfiguration.minimumScale : m_configuration.minimumScale, m_configuration.maximumScale);
251 };
252
253 if (layoutSizeIsExplicitlyScaled()) {
254 if (m_configuration.initialScaleIsSet)
255 return clampToMinimumAndMaximumScales(m_configuration.initialScale);
256
257 if (m_configuration.width > 0)
258 return clampToMinimumAndMaximumScales(m_viewLayoutSize.width() / m_configuration.width);
259 }
260
261 // If the document has specified its own initial scale, use it regardless.
262 // This is guaranteed to be sanity checked already, so no need for MIN/MAX.
263 if (m_configuration.initialScaleIsSet && !shouldIgnoreScalingConstraints)
264 return m_configuration.initialScale;
265
266 // If not, it is up to us to determine the initial scale.
267 // We want a scale small enough to fit the document width-wise.
268 double initialScale = 0;
269 if (!shouldIgnoreVerticalScalingConstraints()) {
270 static const double maximumContentWidthBeforePreferringExplicitWidthToAvoidExcessiveScaling = 1920;
271 if (width > maximumContentWidthBeforePreferringExplicitWidthToAvoidExcessiveScaling && m_configuration.widthIsSet && 0 < m_configuration.width && m_configuration.width < width)
272 initialScale = m_viewLayoutSize.width() / m_configuration.width;
273 else if (shouldShrinkToFitMinimumEffectiveDeviceWidthWhenIgnoringScalingConstraints())
274 initialScale = effectiveLayoutSizeScaleFactor();
275 else if (width > 0)
276 initialScale = m_viewLayoutSize.width() / width;
277 }
278
279 // Prevent the initial scale from shrinking to a height smaller than our view's minimum height.
280 if (height > 0 && height * initialScale < m_viewLayoutSize.height() && !shouldIgnoreHorizontalScalingConstraints())
281 initialScale = m_viewLayoutSize.height() / height;
282
283 return clampToMinimumAndMaximumScales(initialScale);
284}

296double ViewportConfiguration::minimumScale() const
297{
298 // If we scale to fit, then this is our minimum scale as well.
299 if (!m_configuration.initialScaleIsSet || shouldIgnoreScalingConstraints())
300 return initialScale();
301
302 // If not, we still need to sanity check our value.
303 double minimumScale = m_configuration.minimumScale;
304
305 if (m_forceAlwaysUserScalable)
306 minimumScale = std::min(minimumScale, forceAlwaysUserScalableMinimumScale());
307
308 auto scaleForFittingContentIsApproximatelyEqualToMinimumScale = [] (double viewLength, double contentLength, double minimumScale) {
309 if (contentLength <= 1 || viewLength <= 1)
310 return false;
311
312 if (minimumScale < (viewLength - 0.5) / (contentLength + 0.5))
313 return false;
314
315 if (minimumScale > (viewLength + 0.5) / (contentLength - 0.5))
316 return false;
317
318 return true;
319 };
320
321 double contentWidth = m_contentSize.width();
322 if (contentWidth > 0 && contentWidth * minimumScale < m_viewLayoutSize.width() && !shouldIgnoreVerticalScalingConstraints()) {
323 if (!scaleForFittingContentIsApproximatelyEqualToMinimumScale(m_viewLayoutSize.width(), contentWidth, minimumScale))
324 minimumScale = m_viewLayoutSize.width() / contentWidth;
325 }
326
327 double contentHeight = m_contentSize.height();
328 if (contentHeight > 0 && contentHeight * minimumScale < m_viewLayoutSize.height() && !shouldIgnoreHorizontalScalingConstraints()) {
329 if (!scaleForFittingContentIsApproximatelyEqualToMinimumScale(m_viewLayoutSize.height(), contentHeight, minimumScale))
330 minimumScale = m_viewLayoutSize.height() / contentHeight;
331 }
332
333 minimumScale = std::min(std::max(minimumScale, m_configuration.minimumScale), m_configuration.maximumScale);
334
335 return minimumScale;
336}

442static inline bool viewportArgumentValueIsValid(float value)
443{
444 return value > 0;
445}
446
447template<typename ValueType, typename ViewportArgumentsType>
448static inline void applyViewportArgument(ValueType& value, ViewportArgumentsType viewportArgumentValue, ValueType minimum, ValueType maximum)
449{
450 if (viewportArgumentValueIsValid(viewportArgumentValue))
451 value = std::min(maximum, std::max(minimum, static_cast<ValueType>(viewportArgumentValue)));
452}
453
454template<typename ValueType, typename ViewportArgumentsType>
455static inline void applyViewportArgument(ValueType& value, bool& valueIsSet, ViewportArgumentsType viewportArgumentValue, ValueType minimum, ValueType maximum)
456{
457 if (viewportArgumentValueIsValid(viewportArgumentValue)) {
458 value = std::min(maximum, std::max(minimum, static_cast<ValueType>(viewportArgumentValue)));
459 valueIsSet = true;
460 } else
461 valueIsSet = false;
462}

464static inline bool booleanViewportArgumentIsSet(float value)
465{
466 return !value || value == 1;
467}

469void ViewportConfiguration::updateConfiguration()
470{
471 m_configuration = m_defaultConfiguration;
472
473 const double minimumViewportArgumentsScaleFactor = 0.1;
474 const double maximumViewportArgumentsScaleFactor = 10.0;
475
476 bool viewportArgumentsOverridesInitialScale;
477 bool viewportArgumentsOverridesWidth;
478 bool viewportArgumentsOverridesHeight;
479
480 auto effectiveLayoutScale = effectiveLayoutSizeScaleFactor();
481
482 if (layoutSizeIsExplicitlyScaled())
483 m_configuration.width /= effectiveLayoutScale;
484
485 applyViewportArgument(m_configuration.minimumScale, m_viewportArguments.minZoom, minimumViewportArgumentsScaleFactor, maximumViewportArgumentsScaleFactor);
486 applyViewportArgument(m_configuration.maximumScale, m_viewportArguments.maxZoom, m_configuration.minimumScale, maximumViewportArgumentsScaleFactor);
487 applyViewportArgument(m_configuration.initialScale, viewportArgumentsOverridesInitialScale, m_viewportArguments.zoom, m_configuration.minimumScale, m_configuration.maximumScale);
488
489 double minimumViewportArgumentsDimension = 10;
490 double maximumViewportArgumentsDimension = 10000;
491 applyViewportArgument(m_configuration.width, viewportArgumentsOverridesWidth, viewportArgumentsLength(m_viewportArguments.width), minimumViewportArgumentsDimension, maximumViewportArgumentsDimension);
492 applyViewportArgument(m_configuration.height, viewportArgumentsOverridesHeight, viewportArgumentsLength(m_viewportArguments.height), minimumViewportArgumentsDimension, maximumViewportArgumentsDimension);
493
494 if (viewportArgumentsOverridesInitialScale || viewportArgumentsOverridesWidth || viewportArgumentsOverridesHeight) {
495 m_configuration.initialScaleIsSet = viewportArgumentsOverridesInitialScale;
496 m_configuration.widthIsSet = viewportArgumentsOverridesWidth;
497 m_configuration.heightIsSet = viewportArgumentsOverridesHeight;
498 }
499
500 if (booleanViewportArgumentIsSet(m_viewportArguments.userZoom))
501 m_configuration.allowsUserScaling = m_viewportArguments.userZoom != 0.;
502
503 if (shouldOverrideShrinkToFitArgument())
504 m_configuration.allowsShrinkToFit = shouldOverrideDeviceWidthAndShrinkToFit();
505 else if (booleanViewportArgumentIsSet(m_viewportArguments.shrinkToFit))
506 m_configuration.allowsShrinkToFit = m_viewportArguments.shrinkToFit != 0.;
507
508 if (canOverrideConfigurationParameters() && !viewportArgumentsOverridesWidth)
509 m_configuration.width = m_minimumLayoutSize.width();
510
511 m_configuration.avoidsUnsafeArea = m_viewportArguments.viewportFit != ViewportFit::Cover;
512 m_configuration.initialScaleIgnoringLayoutScaleFactor = m_configuration.initialScale;
513 m_configuration.initialScale *= effectiveLayoutScale;
514 m_configuration.minimumScale *= effectiveLayoutScale;
515 m_configuration.maximumScale *= effectiveLayoutScale;
516
517 LOG_WITH_STREAM(Viewports, stream << "ViewportConfiguration " << this << " updateConfiguration " << *this << " gives initial scale " << initialScale() << " based on contentSize " << m_contentSize << " and layout size " << layoutWidth() << "x" << layoutHeight());
518}

520void ViewportConfiguration::updateMinimumLayoutSize()
521{
522 m_minimumLayoutSize = m_viewLayoutSize / effectiveLayoutSizeScaleFactor();
523
524 if (!shouldOverrideDeviceWidthAndShrinkToFit())
525 return;
526
527 float minDeviceWidth = platformDeviceWidthOverride();
528 m_minimumLayoutSize = FloatSize(minDeviceWidth, std::roundf(m_minimumLayoutSize.height() * (minDeviceWidth / m_minimumLayoutSize.width())));
529}

540int ViewportConfiguration::layoutWidth() const
541{
542 ASSERT(!constraintsAreAllRelative(m_configuration));
543
544 const FloatSize& minimumLayoutSize = m_minimumLayoutSize;
545 if (m_configuration.widthIsSet) {
546 // If we scale to fit, then accept the viewport width with sanity checking.
547 if (!m_configuration.initialScaleIsSet) {
548 double maximumScale = this->maximumScale();
549 double maximumContentWidthInViewportCoordinate = maximumScale * m_configuration.width;
550 if (maximumContentWidthInViewportCoordinate < minimumLayoutSize.width()) {
551 // The content zoomed to maxScale does not fit the view. Return the minimum width
552 // satisfying the constraint maximumScale.
553 return std::round(minimumLayoutSize.width() / maximumScale);
554 }
555 return std::round(m_configuration.width);
556 }
557
558 // If not, make sure the viewport width and initial scale can co-exist.
559 double initialContentWidthInViewportCoordinate = m_configuration.width * m_configuration.initialScaleIgnoringLayoutScaleFactor;
560 if (initialContentWidthInViewportCoordinate < minimumLayoutSize.width()) {
561 // The specified width does not fit in viewport. Return the minimum width that satisfy the initialScale constraint.
562 return std::round(minimumLayoutSize.width() / m_configuration.initialScaleIgnoringLayoutScaleFactor);
563 }
564 return std::round(m_configuration.width);
565 }
566
567 // If the page has a real scale, then just return the minimum size over the initial scale.
568 if (m_configuration.initialScaleIsSet && !m_configuration.heightIsSet)
569 return std::round(minimumLayoutSize.width() / m_configuration.initialScaleIgnoringLayoutScaleFactor);
570
571 if (minimumLayoutSize.height() > 0)
572 return std::round(minimumLayoutSize.width() * layoutHeight() / minimumLayoutSize.height());
573 return minimumLayoutSize.width();
574}
575
576int ViewportConfiguration::layoutHeight() const
577{
578 ASSERT(!constraintsAreAllRelative(m_configuration));
579
580 const FloatSize& minimumLayoutSize = m_minimumLayoutSize;
581 if (m_configuration.heightIsSet) {
582 // If we scale to fit, then accept the viewport height with sanity checking.
583 if (!m_configuration.initialScaleIsSet) {
584 double maximumScale = this->maximumScale();
585 double maximumContentHeightInViewportCoordinate = maximumScale * m_configuration.height;
586 if (maximumContentHeightInViewportCoordinate < minimumLayoutSize.height()) {
587 // The content zoomed to maxScale does not fit the view. Return the minimum height that
588 // satisfy the constraint maximumScale.
589 return std::round(minimumLayoutSize.height() / maximumScale);
590 }
591 return std::round(m_configuration.height);
592 }
593
594 // If not, make sure the viewport width and initial scale can co-exist.
595 double initialContentHeightInViewportCoordinate = m_configuration.height * m_configuration.initialScaleIgnoringLayoutScaleFactor;
596 if (initialContentHeightInViewportCoordinate < minimumLayoutSize.height()) {
597 // The specified width does not fit in viewport. Return the minimum height that satisfy the initialScale constraint.
598 return std::round(minimumLayoutSize.height() / m_configuration.initialScaleIgnoringLayoutScaleFactor);
599 }
600 return std::round(m_configuration.height);
601 }
602
603 // If the page has a real scale, then just return the minimum size over the initial scale.
604 if (m_configuration.initialScaleIsSet && !m_configuration.widthIsSet)
605 return std::round(minimumLayoutSize.height() / m_configuration.initialScaleIgnoringLayoutScaleFactor);
606
607 if (minimumLayoutSize.width() > 0)
608 return std::round(minimumLayoutSize.height() * layoutWidth() / minimumLayoutSize.width());
609 return minimumLayoutSize.height();
610}

612bool ViewportConfiguration::setMinimumEffectiveDeviceWidth(double width)
613{
614 if (WTF::areEssentiallyEqual(m_minimumEffectiveDeviceWidth, width))
615 return false;
616
617 m_minimumEffectiveDeviceWidth = width;
618
619 if (shouldIgnoreMinimumEffectiveDeviceWidth())
620 return false;
621
622 updateMinimumLayoutSize();
623 updateConfiguration();
624 return true;
625}
626
627bool ViewportConfiguration::setMinimumEffectiveDeviceWidthWhenIgnoringScalingConstraints(double width)
628{
629 if (WTF::areEssentiallyEqual(m_minimumEffectiveDeviceWidthWhenIgnoringScalingConstraints, width))
630 return false;
631
632 bool wasShrinkingToFitMinimumEffectiveDeviceWidth = shouldShrinkToFitMinimumEffectiveDeviceWidthWhenIgnoringScalingConstraints();
633 m_minimumEffectiveDeviceWidthWhenIgnoringScalingConstraints = width;
634 if (wasShrinkingToFitMinimumEffectiveDeviceWidth == shouldShrinkToFitMinimumEffectiveDeviceWidthWhenIgnoringScalingConstraints())
635 return false;
636
637 updateMinimumLayoutSize();
638 updateConfiguration();
639 return true;
640}

684String ViewportConfiguration::description() const
685{
686 TextStream ts;
687
688 ts.startGroup();
689 ts << "viewport-configuration " << (void*)this;
690 {
691 TextStream::GroupScope scope(ts);
692 ts << "viewport arguments";
693 ts << m_viewportArguments;
694 }
695 {
696 TextStream::GroupScope scope(ts);
697 ts << "configuration";
698 ts << m_configuration;
699 }
700 {
701 TextStream::GroupScope scope(ts);
702 ts << "default configuration";
703 ts << m_defaultConfiguration;
704 }
705
706 ts.dumpProperty("contentSize", m_contentSize);
707 ts.dumpProperty("minimumLayoutSize", m_minimumLayoutSize);
708 ts.dumpProperty("layoutSizeScaleFactor", m_layoutSizeScaleFactor);
709 ts.dumpProperty("computed initial scale", initialScale());
710 ts.dumpProperty("computed minimum scale", minimumScale());
711 ts.dumpProperty("computed layout size", layoutSize());
712 ts.dumpProperty("ignoring horizontal scaling constraints", shouldIgnoreHorizontalScalingConstraints() ? "true" : "false");
713 ts.dumpProperty("ignoring vertical scaling constraints", shouldIgnoreVerticalScalingConstraints() ? "true" : "false");
714 ts.dumpProperty("avoids unsafe area", avoidsUnsafeArea() ? "true" : "false");
715 ts.dumpProperty("minimum effective device width", m_minimumEffectiveDeviceWidth);
716 ts.dumpProperty("known to lay out wider than viewport", m_isKnownToLayOutWiderThanViewport ? "true" : "false");
717
718 ts.endGroup();
719
720 return ts.release();
721}
