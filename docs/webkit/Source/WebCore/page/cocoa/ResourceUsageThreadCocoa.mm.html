45static unsigned categoryForVMTag(unsigned tag)
46{
47 switch (tag) {
48 case VM_MEMORY_IOKIT:
49 case VM_MEMORY_LAYERKIT:
50 return MemoryCategory::Layers;
51 case VM_MEMORY_IMAGEIO:
52 case VM_MEMORY_CGIMAGE:
53 return MemoryCategory::Images;
54 case VM_MEMORY_JAVASCRIPT_JIT_REGISTER_FILE:
55 return MemoryCategory::IsoHeap;
56 case VM_MEMORY_JAVASCRIPT_JIT_EXECUTABLE_ALLOCATOR:
57 return MemoryCategory::JSJIT;
58 case VM_MEMORY_JAVASCRIPT_CORE:
59 return MemoryCategory::Gigacage;
60 case VM_MEMORY_MALLOC:
61 case VM_MEMORY_MALLOC_HUGE:
62 case VM_MEMORY_MALLOC_LARGE:
63 case VM_MEMORY_MALLOC_SMALL:
64 case VM_MEMORY_MALLOC_TINY:
65 case VM_MEMORY_MALLOC_NANO:
66 return MemoryCategory::LibcMalloc;
67 case VM_MEMORY_TCMALLOC:
68 return MemoryCategory::bmalloc;
69 default:
70 return MemoryCategory::Other;
71 }
72}

74struct ThreadInfo {
75 MachSendRight sendRight;
76 float usage { 0 };
77 String threadName;
78 String dispatchQueueName;
79};
80
81static Vector<ThreadInfo> threadInfos()
82{
83 thread_array_t threadList = nullptr;
84 mach_msg_type_number_t threadCount = 0;
85 kern_return_t kr = task_threads(mach_task_self(), &threadList, &threadCount);
86 ASSERT(kr == KERN_SUCCESS);
87 if (kr != KERN_SUCCESS)
88 return { };
89
90 // Since we collect thread usage without suspending threads, threads retrieved by task_threads can have gone while iterating
91 // them to get usage information. If a thread has gone, a system call returns non KERN_SUCCESS, and we just ignore these threads.
92 Vector<ThreadInfo> infos;
93 for (mach_msg_type_number_t i = 0; i < threadCount; ++i) {
94 MachSendRight sendRight = MachSendRight::adopt(threadList[i]);
95
96 thread_info_data_t threadInfo;
97 mach_msg_type_number_t threadInfoCount = THREAD_INFO_MAX;
98 kr = thread_info(sendRight.sendRight(), THREAD_BASIC_INFO, reinterpret_cast<thread_info_t>(&threadInfo), &threadInfoCount);
99 if (kr != KERN_SUCCESS)
100 continue;
101
102 thread_identifier_info_data_t threadIdentifierInfo;
103 mach_msg_type_number_t threadIdentifierInfoCount = THREAD_IDENTIFIER_INFO_COUNT;
104 kr = thread_info(sendRight.sendRight(), THREAD_IDENTIFIER_INFO, reinterpret_cast<thread_info_t>(&threadIdentifierInfo), &threadIdentifierInfoCount);
105 if (kr != KERN_SUCCESS)
106 continue;
107
108 thread_extended_info_data_t threadExtendedInfo;
109 mach_msg_type_number_t threadExtendedInfoCount = THREAD_EXTENDED_INFO_COUNT;
110 kr = thread_info(sendRight.sendRight(), THREAD_EXTENDED_INFO, reinterpret_cast<thread_info_t>(&threadExtendedInfo), &threadExtendedInfoCount);
111 if (kr != KERN_SUCCESS)
112 continue;
113
114 float usage = 0;
115 auto threadBasicInfo = reinterpret_cast<thread_basic_info_t>(threadInfo);
116 if (!(threadBasicInfo->flags & TH_FLAGS_IDLE))
117 usage = threadBasicInfo->cpu_usage / static_cast<float>(TH_USAGE_SCALE) * 100.0;
118
119 // FIXME: dispatch_queue_t can be destroyed concurrently while we are accessing to it here. We should not use it.
120 String threadName = String(threadExtendedInfo.pth_name);
121 String dispatchQueueName;
122 if (threadIdentifierInfo.dispatch_qaddr) {
123 dispatch_queue_t queue = *reinterpret_cast<dispatch_queue_t*>(threadIdentifierInfo.dispatch_qaddr);
124 dispatchQueueName = String(dispatch_queue_get_label(queue));
125 }
126
127 infos.append(ThreadInfo { WTFMove(sendRight), usage, threadName, dispatchQueueName });
128 }
129
130 kr = vm_deallocate(mach_task_self(), (vm_offset_t)threadList, threadCount * sizeof(thread_t));
131 ASSERT(kr == KERN_SUCCESS);
132
133 return infos;
134}

148void ResourceUsageThread::platformCollectCPUData(JSC::VM*, ResourceUsageData& data)
149{
150 Vector<ThreadInfo> threads = threadInfos();
151 if (threads.isEmpty()) {
152 ASSERT_NOT_REACHED();
153 return;
154 }
155
156 // Main thread is always first.
157 ASSERT(threads[0].dispatchQueueName == "com.apple.main-thread");
158
159 mach_port_t resourceUsageMachThread = mach_thread_self();
160 mach_port_t mainThreadMachThread = threads[0].sendRight.sendRight();
161
162 HashSet<mach_port_t> knownWebKitThreads;
163 {
164 Locker locker { Thread::allThreadsLock() };
165 for (auto* thread : Thread::allThreads()) {
166 mach_port_t machThread = thread->machThread();
167 if (MACH_PORT_VALID(machThread))
168 knownWebKitThreads.add(machThread);
169 }
170 }
171
172 HashMap<mach_port_t, String> knownWorkerThreads;
173 {
174 Locker locker { WorkerOrWorkletThread::workerOrWorkletThreadsLock() };
175 for (auto* thread : WorkerOrWorkletThread::workerOrWorkletThreads()) {
176 // Ignore worker threads that have not been fully started yet.
177 if (!thread->thread())
178 continue;
179 mach_port_t machThread = thread->thread()->machThread();
180 if (MACH_PORT_VALID(machThread))
181 knownWorkerThreads.set(machThread, thread->identifier().isolatedCopy());
182 }
183 }
184
185 auto isDebuggerThread = [&](const ThreadInfo& thread) -> bool {
186 mach_port_t machThread = thread.sendRight.sendRight();
187 if (machThread == resourceUsageMachThread)
188 return true;
189#if ENABLE(SAMPLING_PROFILER)
190 if (machThread == m_samplingProfilerMachThread)
191 return true;
192#endif
193 return false;
194 };
195
196 auto isWebKitThread = [&](const ThreadInfo& thread) -> bool {
197 mach_port_t machThread = thread.sendRight.sendRight();
198 if (knownWebKitThreads.contains(machThread))
199 return true;
200
201 // The bmalloc scavenger thread is below WTF. Detect it by its name.
202 if (thread.threadName == "JavaScriptCore bmalloc scavenger")
203 return true;
204
205 // WebKit uses many WorkQueues with common prefixes.
206 if (thread.dispatchQueueName.startsWith("com.apple.IPC.")
207 || thread.dispatchQueueName.startsWith("com.apple.WebKit.")
208 || thread.dispatchQueueName.startsWith("org.webkit."))
209 return true;
210
211 return false;
212 };
213
214 for (auto& thread : threads) {
215 data.cpu += thread.usage;
216 if (isDebuggerThread(thread))
217 continue;
218
219 data.cpuExcludingDebuggerThreads += thread.usage;
220
221 mach_port_t machThread = thread.sendRight.sendRight();
222 if (machThread == mainThreadMachThread) {
223 data.cpuThreads.append(ThreadCPUInfo { "Main Thread"_s, String(), thread.usage, ThreadCPUInfo::Type::Main});
224 continue;
225 }
226
227 String threadIdentifier = knownWorkerThreads.get(machThread);
228 bool isWorkerThread = !threadIdentifier.isEmpty();
229 ThreadCPUInfo::Type type = (isWorkerThread || isWebKitThread(thread)) ? ThreadCPUInfo::Type::WebKit : ThreadCPUInfo::Type::Unknown;
230 data.cpuThreads.append(ThreadCPUInfo { thread.threadName, threadIdentifier, thread.usage, type });
231 }
232}

234void ResourceUsageThread::platformCollectMemoryData(JSC::VM* vm, ResourceUsageData& data)
235{
236 auto tags = pagesPerVMTag();
237 std::array<TagInfo, MemoryCategory::NumberOfCategories> pagesPerCategory;
238 size_t totalDirtyPages = 0;
239 for (unsigned i = 0; i < 256; ++i) {
240 pagesPerCategory[categoryForVMTag(i)].dirty += tags[i].dirty;
241 pagesPerCategory[categoryForVMTag(i)].reclaimable += tags[i].reclaimable;
242 totalDirtyPages += tags[i].dirty;
243 }
244
245 for (auto& category : data.categories) {
246 if (category.isSubcategory) // Only do automatic tallying for top-level categories.
247 continue;
248 category.dirtySize = pagesPerCategory[category.type].dirty * vmPageSize();
249 category.reclaimableSize = pagesPerCategory[category.type].reclaimable * vmPageSize();
250 }
251 data.totalDirtySize = totalDirtyPages * vmPageSize();
252
253 size_t currentGCHeapCapacity = vm->heap.blockBytesAllocated();
254 size_t currentGCOwnedExtra = vm->heap.extraMemorySize();
255 size_t currentGCOwnedExternal = vm->heap.externalMemorySize();
256 ASSERT(currentGCOwnedExternal <= currentGCOwnedExtra);
257
258 data.categories[MemoryCategory::GCHeap].dirtySize = currentGCHeapCapacity;
259 data.categories[MemoryCategory::GCOwned].dirtySize = currentGCOwnedExtra - currentGCOwnedExternal;
260 data.categories[MemoryCategory::GCOwned].externalSize = currentGCOwnedExternal;
261
262 auto& mallocBucket = isFastMallocEnabled() ? data.categories[MemoryCategory::bmalloc] : data.categories[MemoryCategory::LibcMalloc];
263
264 // First subtract memory allocated by the GC heap, since we track that separately.
265 mallocBucket.dirtySize -= currentGCHeapCapacity;
266
267 // It would be nice to assert that the "GC owned" amount is smaller than the total dirty malloc size,
268 // but since the "GC owned" accounting is inexact, it's not currently feasible.
269 size_t currentGCOwnedGenerallyInMalloc = currentGCOwnedExtra - currentGCOwnedExternal;
270 if (currentGCOwnedGenerallyInMalloc < mallocBucket.dirtySize)
271 mallocBucket.dirtySize -= currentGCOwnedGenerallyInMalloc;
272
273 data.totalExternalSize = currentGCOwnedExternal;
274
275 data.timeOfNextEdenCollection = data.timestamp + vm->heap.edenActivityCallback()->timeUntilFire().value_or(Seconds(std::numeric_limits<double>::infinity()));
276 data.timeOfNextFullCollection = data.timestamp + vm->heap.fullActivityCallback()->timeUntilFire().value_or(Seconds(std::numeric_limits<double>::infinity()));
277}
