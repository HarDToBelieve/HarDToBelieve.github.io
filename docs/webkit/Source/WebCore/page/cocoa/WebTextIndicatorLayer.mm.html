59static bool indicatorWantsContentCrossfade(const WebCore::TextIndicator& indicator)
60{
61 if (!indicator.data().contentImageWithHighlight)
62 return false;
63
64 switch (indicator.presentationTransition()) {
65 case WebCore::TextIndicatorPresentationTransition::BounceAndCrossfade:
66 return true;
67
68 case WebCore::TextIndicatorPresentationTransition::Bounce:
69 case WebCore::TextIndicatorPresentationTransition::FadeIn:
70 case WebCore::TextIndicatorPresentationTransition::None:
71 return false;
72 }
73
74 ASSERT_NOT_REACHED();
75 return false;
76}

78static bool indicatorWantsFadeIn(const WebCore::TextIndicator& indicator)
79{
80 switch (indicator.presentationTransition()) {
81 case WebCore::TextIndicatorPresentationTransition::FadeIn:
82 return true;
83
84 case WebCore::TextIndicatorPresentationTransition::Bounce:
85 case WebCore::TextIndicatorPresentationTransition::BounceAndCrossfade:
86 case WebCore::TextIndicatorPresentationTransition::None:
87 return false;
88 }

90 ASSERT_NOT_REACHED();
91 return false;
92}

126- (instancetype)initWithFrame:(CGRect)frame textIndicator:(WebCore::TextIndicator&)textIndicator margin:(CGSize)margin offset:(CGPoint)offset
127{
128 if (!(self = [super init]))
129 return nil;
130
131 self.anchorPoint = CGPointZero;
132 self.frame = frame;
133
134 _textIndicator = &textIndicator;
135 _margin = margin;
136
137 RefPtr<WebCore::NativeImage> contentsImage;
138 WebCore::FloatSize contentsImageLogicalSize { 1, 1 };
139 if (auto* contentImage = _textIndicator->contentImage()) {
140 contentsImageLogicalSize = contentImage->size();
141 contentsImageLogicalSize.scale(1 / _textIndicator->contentImageScaleFactor());
142 if (indicatorWantsContentCrossfade(*_textIndicator) && _textIndicator->contentImageWithHighlight())
143 contentsImage = _textIndicator->contentImageWithHighlight()->nativeImage();
144 else
145 contentsImage = contentImage->nativeImage();
146 }
147
148 auto bounceLayers = adoptNS([[NSMutableArray alloc] init]);
149
150 RetainPtr<CGColorRef> highlightColor;
151 auto rimShadowColor = adoptCF(CGColorCreateGenericGray(0, 0.35));
152 auto dropShadowColor = adoptCF(CGColorCreateGenericGray(0, 0.2));
153 auto borderColor = adoptCF(CGColorCreateSRGB(0.96, 0.9, 0, 1));
154#if PLATFORM(MAC)
155 highlightColor = [NSColor findHighlightColor].CGColor;
156#else
157 highlightColor = adoptCF(CGColorCreateSRGB(.99, .89, 0.22, 1.0));
158#endif
159
160 auto textRectsInBoundingRectCoordinates = _textIndicator->textRectsInBoundingRectCoordinates();
161
162 auto paths = WebCore::PathUtilities::pathsWithShrinkWrappedRects(textRectsInBoundingRectCoordinates, cornerRadius);
163
164 for (const auto& path : paths) {
165 WebCore::FloatRect pathBoundingRect = path.boundingRect();
166
167 WebCore::Path translatedPath;
168 WebCore::AffineTransform transform;
169 transform.translate(-pathBoundingRect.location());
170 translatedPath.addPath(path, transform);
171
172 WebCore::FloatRect offsetTextRect = pathBoundingRect;
173 offsetTextRect.move(offset.x, offset.y);
174
175 WebCore::FloatRect bounceLayerRect = offsetTextRect;
176 bounceLayerRect.move(_margin.width, _margin.height);
177
178 RetainPtr<CALayer> bounceLayer = adoptNS([[CALayer alloc] init]);
179 [bounceLayer setDelegate:[WebActionDisablingCALayerDelegate shared]];
180 [bounceLayer setFrame:bounceLayerRect];
181 [bounceLayer setOpacity:0];
182 [bounceLayers addObject:bounceLayer.get()];
183
184 WebCore::FloatRect yellowHighlightRect(WebCore::FloatPoint(), bounceLayerRect.size());
185
186#if PLATFORM(MAC)
187 RetainPtr<CALayer> dropShadowLayer = adoptNS([[CALayer alloc] init]);
188 [dropShadowLayer setDelegate:[WebActionDisablingCALayerDelegate shared]];
189 [dropShadowLayer setShadowColor:dropShadowColor.get()];
190 [dropShadowLayer setShadowRadius:WebCore::dropShadowBlurRadius];
191 [dropShadowLayer setShadowOffset:CGSizeMake(dropShadowOffsetX, dropShadowOffsetY)];
192 [dropShadowLayer setShadowPath:translatedPath.platformPath()];
193 [dropShadowLayer setShadowOpacity:1];
194 [dropShadowLayer setFrame:yellowHighlightRect];
195 [bounceLayer addSublayer:dropShadowLayer.get()];
196 [bounceLayer setValue:dropShadowLayer.get() forKey:dropShadowLayerKey];
197
198 RetainPtr<CALayer> rimShadowLayer = adoptNS([[CALayer alloc] init]);
199 [rimShadowLayer setDelegate:[WebActionDisablingCALayerDelegate shared]];
200 [rimShadowLayer setFrame:yellowHighlightRect];
201 [rimShadowLayer setShadowColor:rimShadowColor.get()];
202 [rimShadowLayer setShadowRadius:WebCore::rimShadowBlurRadius];
203 [rimShadowLayer setShadowPath:translatedPath.platformPath()];
204 [rimShadowLayer setShadowOffset:CGSizeZero];
205 [rimShadowLayer setShadowOpacity:1];
206 [rimShadowLayer setFrame:yellowHighlightRect];
207 [bounceLayer addSublayer:rimShadowLayer.get()];
208 [bounceLayer setValue:rimShadowLayer.get() forKey:rimShadowLayerKey];
209#endif // PLATFORM(MAC)
210
211 RetainPtr<CALayer> textLayer = adoptNS([[CALayer alloc] init]);
212 [textLayer setBackgroundColor:highlightColor.get()];
213 [textLayer setBorderColor:borderColor.get()];
214 [textLayer setBorderWidth:borderWidth];
215 [textLayer setDelegate:[WebActionDisablingCALayerDelegate shared]];
216 if (contentsImage)
217 [textLayer setContents:(__bridge id)contentsImage->platformImage().get()];
218
219 RetainPtr<CAShapeLayer> maskLayer = adoptNS([[CAShapeLayer alloc] init]);
220 [maskLayer setPath:translatedPath.platformPath()];
221 [textLayer setMask:maskLayer.get()];
222
223 WebCore::FloatRect imageRect = pathBoundingRect;
224 [textLayer setContentsRect:CGRectMake(imageRect.x() / contentsImageLogicalSize.width(), imageRect.y() / contentsImageLogicalSize.height(), imageRect.width() / contentsImageLogicalSize.width(), imageRect.height() / contentsImageLogicalSize.height())];
225 [textLayer setContentsGravity:kCAGravityCenter];
226 [textLayer setContentsScale:_textIndicator->contentImageScaleFactor()];
227 [textLayer setFrame:yellowHighlightRect];
228 [bounceLayer setValue:textLayer.get() forKey:textLayerKey];
229 [bounceLayer addSublayer:textLayer.get()];
230 }
231
232 self.sublayers = bounceLayers.get();
233 _bounceLayers = bounceLayers;
234
235 return self;
236}

238static RetainPtr<CAKeyframeAnimation> createBounceAnimation(CFTimeInterval duration)
239{
240 RetainPtr<CAKeyframeAnimation> bounceAnimation = [CAKeyframeAnimation animationWithKeyPath:@"transform"];
241 [bounceAnimation setValues:@[
242 [NSValue valueWithCATransform3D:CATransform3DIdentity],
243 [NSValue valueWithCATransform3D:CATransform3DMakeScale(WebCore::midBounceScale, WebCore::midBounceScale, 1)],
244 [NSValue valueWithCATransform3D:CATransform3DIdentity]
245 ]];
246 [bounceAnimation setDuration:duration];
247
248 return bounceAnimation;
249}

252{
253 RetainPtr<CABasicAnimation> crossfadeAnimation = [CABasicAnimation animationWithKeyPath:@"contents"];
254 auto contentsImage = textIndicator.contentImage()->nativeImage();
255 [crossfadeAnimation setToValue:(__bridge id)contentsImage->platformImage().get()];
256 [crossfadeAnimation setFillMode:kCAFillModeForwards];
257 [crossfadeAnimation setRemovedOnCompletion:NO];
258 [crossfadeAnimation setDuration:duration];

260 return crossfadeAnimation;

287- (CFTimeInterval)_animationDuration
288{
289 if ([self indicatorWantsBounce:*_textIndicator]) {
290 if (indicatorWantsContentCrossfade(*_textIndicator))
291 return bounceWithCrossfadeAnimationDuration;
292 return WebCore::bounceAnimationDuration.value();
293 }
294
295 return fadeInAnimationDuration;
296}

303- (void)present
304{
305 bool wantsBounce = [self indicatorWantsBounce:*_textIndicator];
306 bool wantsCrossfade = indicatorWantsContentCrossfade(*_textIndicator);
307 bool wantsFadeIn = indicatorWantsFadeIn(*_textIndicator);
308 CFTimeInterval animationDuration = [self _animationDuration];
309
310 _hasCompletedAnimation = false;
311
312 RetainPtr<CAAnimation> presentationAnimation;
313 if (wantsBounce)
314 presentationAnimation = createBounceAnimation(animationDuration);
315 else if (wantsFadeIn)
316 presentationAnimation = createFadeInAnimation(animationDuration);
317
318 RetainPtr<CABasicAnimation> crossfadeAnimation;
319 RetainPtr<CABasicAnimation> fadeShadowInAnimation;
320 if (wantsCrossfade) {
321 crossfadeAnimation = createContentCrossfadeAnimation(animationDuration, *_textIndicator);
322 fadeShadowInAnimation = createShadowFadeAnimation(animationDuration);
323 }
324
325 [CATransaction begin];
326 for (CALayer *bounceLayer in _bounceLayers.get()) {
327 if ([self indicatorWantsManualAnimation:*_textIndicator])
328 bounceLayer.speed = 0;
329
330 if (!wantsFadeIn)
331 bounceLayer.opacity = 1;
332
333 if (presentationAnimation)
334 [bounceLayer addAnimation:presentationAnimation.get() forKey:@"presentation"];
335
336 if (wantsCrossfade) {
337 [[bounceLayer valueForKey:textLayerKey] addAnimation:crossfadeAnimation.get() forKey:@"contentTransition"];
338 [[bounceLayer valueForKey:dropShadowLayerKey] addAnimation:fadeShadowInAnimation.get() forKey:@"fadeShadowIn"];
339 [[bounceLayer valueForKey:rimShadowLayerKey] addAnimation:fadeShadowInAnimation.get() forKey:@"fadeShadowIn"];
340 }
341 }
342 [CATransaction commit];
343}

345- (void)hideWithCompletionHandler:(void(^)(void))completionHandler
346{
347 RetainPtr<CABasicAnimation> fadeAnimation = [CABasicAnimation animationWithKeyPath:@"opacity"];
348 [fadeAnimation setFromValue:@1];
349 [fadeAnimation setToValue:@0];
350 [fadeAnimation setFillMode:kCAFillModeForwards];
351 [fadeAnimation setRemovedOnCompletion:NO];
352 [fadeAnimation setDuration:fadeOutAnimationDuration];
353
354 [CATransaction begin];
355 [CATransaction setCompletionBlock:completionHandler];
356 [self addAnimation:fadeAnimation.get() forKey:@"fadeOut"];
357 [CATransaction commit];
358}

360- (void)setAnimationProgress:(float)progress
361{
362 if (_hasCompletedAnimation)
363 return;
364
365 if (progress == 1) {
366 _hasCompletedAnimation = true;
367
368 for (CALayer *bounceLayer in _bounceLayers.get()) {
369 // Continue the animation from wherever it had manually progressed to.
370 CFTimeInterval beginTime = bounceLayer.timeOffset;
371 bounceLayer.speed = 1;
372 beginTime = [bounceLayer convertTime:CACurrentMediaTime() fromLayer:nil] - beginTime;
373 bounceLayer.beginTime = beginTime;
374 }
375 } else {
376 CFTimeInterval animationDuration = [self _animationDuration];
377 for (CALayer *bounceLayer in _bounceLayers.get())
378 bounceLayer.timeOffset = progress * animationDuration;
379 }
380}
