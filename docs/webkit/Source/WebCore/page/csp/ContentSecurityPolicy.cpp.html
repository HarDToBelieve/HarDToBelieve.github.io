69static String consoleMessageForViolation(const char* effectiveViolatedDirective, const ContentSecurityPolicyDirective& violatedDirective, const URL& blockedURL, const char* prefix, const char* subject = "it")
70{
71 return makeString(violatedDirective.directiveList().isReportOnly() ? "[Report Only] " : "",
72 prefix, blockedURL.isEmpty() ? "" : " ", blockedURL.stringCenterEllipsizedToLength(), " because ", subject,
73 violatedDirective.isDefaultSrc() ? " appears in neither the " : " does not appear in the ",
74 effectiveViolatedDirective,
75 violatedDirective.isDefaultSrc() ? " directive nor the default-src directive of the Content Security Policy." : " directive of the Content Security Policy.");
76}

123bool ContentSecurityPolicy::allowRunningOrDisplayingInsecureContent(const URL& url)
124{
125 bool allow = true;
126 for (auto& policy : m_policies) {
127 if (!policy->hasBlockAllMixedContentDirective())
128 continue;
129 bool isReportOnly = policy->isReportOnly();
130 auto message = makeString(isReportOnly ? "[Report Only] " : "", "Blocked mixed content ",
131 url.stringCenterEllipsizedToLength(), " because 'block-all-mixed-content' appears in the Content Security Policy.");
132 reportViolation(ContentSecurityPolicyDirectiveNames::blockAllMixedContent,
133 ContentSecurityPolicyDirectiveNames::blockAllMixedContent, *policy, url, message);
134 if (!isReportOnly)
135 allow = false;
136 }
137 return allow;
138}

154ContentSecurityPolicyResponseHeaders ContentSecurityPolicy::responseHeaders() const
155{
156 if (!m_cachedResponseHeaders) {
157 ContentSecurityPolicyResponseHeaders result;
158 result.m_headers.reserveInitialCapacity(m_policies.size());
159 for (auto& policy : m_policies)
160 result.m_headers.uncheckedAppend({ policy->header(), policy->headerType() });
161 result.m_httpStatusCode = m_httpStatusCode;
162 m_cachedResponseHeaders = WTFMove(result);
163 }
164 return *m_cachedResponseHeaders;
165}

167void ContentSecurityPolicy::didReceiveHeaders(const ContentSecurityPolicyResponseHeaders& headers, String&& referrer, ReportParsingErrors reportParsingErrors)
168{
169 SetForScope<bool> isReportingEnabled(m_isReportingEnabled, reportParsingErrors == ReportParsingErrors::Yes);
170 for (auto& header : headers.m_headers)
171 didReceiveHeader(header.first, header.second, ContentSecurityPolicy::PolicyFrom::HTTPHeader, String { });
172 m_referrer = WTFMove(referrer);
173 m_httpStatusCode = headers.m_httpStatusCode;
174}
175
176void ContentSecurityPolicy::didReceiveHeader(const String& header, ContentSecurityPolicyHeaderType type, ContentSecurityPolicy::PolicyFrom policyFrom, String&& referrer, int httpStatusCode)
177{
178 if (m_hasAPIPolicy)
179 return;
180
181 m_referrer = WTFMove(referrer);
182 m_httpStatusCode = httpStatusCode;
183
184 if (policyFrom == PolicyFrom::API) {
185 ASSERT(m_policies.isEmpty());
186 m_hasAPIPolicy = true;
187 }
188
189 m_cachedResponseHeaders = std::nullopt;
190
191 // RFC2616, section 4.2 specifies that headers appearing multiple times can
192 // be combined with a comma. Walk the header string, and parse each comma
193 // separated chunk as a separate header.
194 readCharactersForParsing(header, [&](auto buffer) {
195 auto begin = buffer.position();
196
197 while (buffer.hasCharactersRemaining()) {
198 skipUntil(buffer, ',');
199
200 // header1,header2 OR header1
201 // ^ ^
202 m_policies.append(ContentSecurityPolicyDirectiveList::create(*this, String(begin, buffer.position() - begin), type, policyFrom));
203
204 // Skip the comma, and begin the next header from the current position.
205 ASSERT(buffer.atEnd() || *buffer == ',');
206 skipExactly(buffer, ',');
207 begin = buffer.position();
208 }
209 });
210
211
212 if (m_scriptExecutionContext)
213 applyPolicyToScriptExecutionContext();
214}

222void ContentSecurityPolicy::applyPolicyToScriptExecutionContext()
223{
224 ASSERT(m_scriptExecutionContext);
225
226 // Update source self as the security origin may have changed between the time we were created and now.
227 // For instance, we may have been initially created for an about:blank iframe that later inherited the
228 // security origin of its owner document.
229 ASSERT(m_scriptExecutionContext->securityOrigin());
230 updateSourceSelf(*m_scriptExecutionContext->securityOrigin());
231
232 bool enableStrictMixedContentMode = false;
233 for (auto& policy : m_policies) {
234 const ContentSecurityPolicyDirective* violatedDirective = policy->violatedDirectiveForUnsafeEval();
235 if (violatedDirective && !violatedDirective->directiveList().isReportOnly()) {
236 m_lastPolicyEvalDisabledErrorMessage = policy->evalDisabledErrorMessage();
237 m_lastPolicyWebAssemblyDisabledErrorMessage = policy->webAssemblyDisabledErrorMessage();
238 }
239 if (policy->hasBlockAllMixedContentDirective() && !policy->isReportOnly())
240 enableStrictMixedContentMode = true;
241 }
242
243 if (!m_lastPolicyEvalDisabledErrorMessage.isNull())
244 m_scriptExecutionContext->disableEval(m_lastPolicyEvalDisabledErrorMessage);
245 if (!m_lastPolicyWebAssemblyDisabledErrorMessage.isNull())
246 m_scriptExecutionContext->disableWebAssembly(m_lastPolicyWebAssemblyDisabledErrorMessage);
247 if (m_sandboxFlags != SandboxNone && is<Document>(m_scriptExecutionContext))
248 m_scriptExecutionContext->enforceSandboxFlags(m_sandboxFlags);
249 if (enableStrictMixedContentMode)
250 m_scriptExecutionContext->setStrictMixedContentMode(true);
251}

258bool ContentSecurityPolicy::urlMatchesSelf(const URL& url, bool forFrameSrc) const
259{
260 // As per https://w3c.github.io/webappsec-csp/#match-url-to-source-expression, we compare the URL origin with the policy origin.
261 // We get origin using https://url.spec.whatwg.org/#concept-url-origin which has specific blob URLs treatment as follow.
262 if (forFrameSrc && url.protocolIsBlob())
263 return m_selfSource->matches(BlobURL::getOriginURL(url));
264 return m_selfSource->matches(url);
265}

281template<typename Predicate, typename... Args>
282typename std::enable_if<!std::is_convertible<Predicate, ContentSecurityPolicy::ViolatedDirectiveCallback>::value, bool>::type ContentSecurityPolicy::allPoliciesWithDispositionAllow(Disposition disposition, Predicate&& predicate, Args&&... args) const
283{
284 bool isReportOnly = disposition == ContentSecurityPolicy::Disposition::ReportOnly;
285 for (auto& policy : m_policies) {
286 if (policy->isReportOnly() != isReportOnly)
287 continue;
288 if ((policy.get()->*predicate)(std::forward<Args>(args)...))
289 return false;
290 }
291 return true;
292}
293
294template<typename Predicate, typename... Args>
295bool ContentSecurityPolicy::allPoliciesWithDispositionAllow(Disposition disposition, ViolatedDirectiveCallback&& callback, Predicate&& predicate, Args&&... args) const
296{
297 bool isReportOnly = disposition == ContentSecurityPolicy::Disposition::ReportOnly;
298 bool isAllowed = true;
299 for (auto& policy : m_policies) {
300 if (policy->isReportOnly() != isReportOnly)
301 continue;
302 if (const ContentSecurityPolicyDirective* violatedDirective = (policy.get()->*predicate)(std::forward<Args>(args)...)) {
303 isAllowed = false;
304 callback(*violatedDirective);
305 }
306 }
307 return isAllowed;
308}
309
310template<typename Predicate, typename... Args>
311bool ContentSecurityPolicy::allPoliciesAllow(ViolatedDirectiveCallback&& callback, Predicate&& predicate, Args&&... args) const
312{
313 bool isAllowed = true;
314 for (auto& policy : m_policies) {
315 if (const ContentSecurityPolicyDirective* violatedDirective = (policy.get()->*predicate)(std::forward<Args>(args)...)) {
316 if (!violatedDirective->directiveList().isReportOnly())
317 isAllowed = false;
318 callback(*violatedDirective);
319 }
320 }
321 return isAllowed;
322}

324template<typename Predicate>
325ContentSecurityPolicy::HashInEnforcedAndReportOnlyPoliciesPair ContentSecurityPolicy::findHashOfContentInPolicies(Predicate&& predicate, StringView content, OptionSet<ContentSecurityPolicyHashAlgorithm> algorithms) const
326{
327 if (algorithms.isEmpty() || content.isEmpty())
328 return { false, false };
329
330 // FIXME: We should compute the document encoding once and cache it instead of computing it on each invocation.
331 TextEncoding documentEncoding;
332 if (is<Document>(m_scriptExecutionContext))
333 documentEncoding = downcast<Document>(*m_scriptExecutionContext).textEncoding();
334 const TextEncoding& encodingToUse = documentEncoding.isValid() ? documentEncoding : UTF8Encoding();
335
336 // FIXME: Compute the digest with respect to the raw bytes received from the page.
337 // See <https://bugs.webkit.org/show_bug.cgi?id=155184>.
338 auto encodedContent = encodingToUse.encode(content, UnencodableHandling::Entities);
339 bool foundHashInEnforcedPolicies = false;
340 bool foundHashInReportOnlyPolicies = false;
341 for (auto algorithm : algorithms) {
342 ContentSecurityPolicyHash hash = cryptographicDigestForBytes(algorithm, encodedContent.data(), encodedContent.size());
343 if (!foundHashInEnforcedPolicies && allPoliciesWithDispositionAllow(ContentSecurityPolicy::Disposition::Enforce, std::forward<Predicate>(predicate), hash))
344 foundHashInEnforcedPolicies = true;
345 if (!foundHashInReportOnlyPolicies && allPoliciesWithDispositionAllow(ContentSecurityPolicy::Disposition::ReportOnly, std::forward<Predicate>(predicate), hash))
346 foundHashInReportOnlyPolicies = true;
347 if (foundHashInEnforcedPolicies && foundHashInReportOnlyPolicies)
348 break;
349 }
350 return { foundHashInEnforcedPolicies, foundHashInReportOnlyPolicies };
351}

407bool ContentSecurityPolicy::allowInlineScript(const String& contextURL, const WTF::OrdinalNumber& contextLine, StringView scriptContent, bool overrideContentSecurityPolicy) const
408{
409 if (overrideContentSecurityPolicy)
410 return true;
411 bool didNotifyInspector = false;
412 auto [foundHashInEnforcedPolicies, foundHashInReportOnlyPolicies] = findHashOfContentInPolicies(&ContentSecurityPolicyDirectiveList::violatedDirectiveForScriptHash, scriptContent, m_hashAlgorithmsForInlineScripts);
413 if (foundHashInEnforcedPolicies && foundHashInReportOnlyPolicies)
414 return true;
415 auto handleViolatedDirective = [&] (const ContentSecurityPolicyDirective& violatedDirective) {
416 String consoleMessage = consoleMessageForViolation(ContentSecurityPolicyDirectiveNames::scriptSrc, violatedDirective, URL(), "Refused to execute a script", "its hash, its nonce, or 'unsafe-inline'");
417 reportViolation(ContentSecurityPolicyDirectiveNames::scriptSrc, violatedDirective, URL(), consoleMessage, contextURL, TextPosition(contextLine, WTF::OrdinalNumber()));
418 if (!didNotifyInspector && !violatedDirective.directiveList().isReportOnly()) {
419 reportBlockedScriptExecutionToInspector(violatedDirective.text());
420 didNotifyInspector = true;
421 }
422 };
423 // FIXME: We should not report that the inline script violated a policy when its hash matched a source
424 // expression in the policy and the page has more than one policy. See <https://bugs.webkit.org/show_bug.cgi?id=159832>.
425 if (!foundHashInReportOnlyPolicies)
426 allPoliciesWithDispositionAllow(ContentSecurityPolicy::Disposition::ReportOnly, handleViolatedDirective, &ContentSecurityPolicyDirectiveList::violatedDirectiveForUnsafeInlineScript);
427 return foundHashInEnforcedPolicies || allPoliciesWithDispositionAllow(ContentSecurityPolicy::Disposition::Enforce, handleViolatedDirective, &ContentSecurityPolicyDirectiveList::violatedDirectiveForUnsafeInlineScript);
428}

483{
484 // If a resource is delivered with an policy that includes a directive named frame-ancestors and whose disposition
485 // is "enforce", then the X-Frame-Options header MUST be ignored.
486 // https://www.w3.org/TR/CSP3/#frame-ancestors-and-frame-options
487 for (auto& policy : m_policies) {
488 if (!policy->isReportOnly() && policy->hasFrameAncestorsDirective())
489 return true;
490 }
491 return false;
492}

523bool ContentSecurityPolicy::allowObjectFromSource(const URL& url, RedirectResponseReceived redirectResponseReceived) const
524{
525 if (LegacySchemeRegistry::schemeShouldBypassContentSecurityPolicy(url.protocol().toStringWithoutCopying()))
526 return true;
527 // As per section object-src of the Content Security Policy Level 3 spec., <http://w3c.github.io/webappsec-csp> (Editor's Draft, 29 February 2016),
528 // "If plugin content is loaded without an associated URL (perhaps an object element lacks a data attribute, but loads some default plugin based
529 // on the specified type), it MUST be blocked if object-src's value is 'none', but will otherwise be allowed".
530 String sourceURL;

667void ContentSecurityPolicy::reportViolation(const String& effectiveViolatedDirective, const String& violatedDirective, const ContentSecurityPolicyDirectiveList& violatedDirectiveList, const URL& blockedURL, const String& consoleMessage, const String& sourceURL, const TextPosition& sourcePosition, JSC::JSGlobalObject* state) const
668{
669 logToConsole(consoleMessage, sourceURL, sourcePosition.m_line, sourcePosition.m_column, state);
670
671 if (!m_isReportingEnabled)
672 return;
673
674 // FIXME: Support sending reports from worker.
675 CSPInfo info;
676
677 info.documentURI = m_documentURL ? m_documentURL.value().strippedForUseAsReferrer() : deprecatedURLForReporting(blockedURL);
678
679 if (m_client)
680 m_client->willSendCSPViolationReport(info);
681 else {
682 if (!is<Document>(m_scriptExecutionContext))
683 return;
684
685 auto& document = downcast<Document>(*m_scriptExecutionContext);
686 auto* frame = document.frame();
687 if (!frame)
688 return;
689
690 info.documentURI = shouldReportProtocolOnly(document.url()) ? document.url().protocol().toString() : document.url().strippedForUseAsReferrer();
691
692 auto stack = createScriptCallStack(JSExecState::currentState(), 2);
693 auto* callFrame = stack->firstNonNativeCallFrame();
694 if (callFrame && callFrame->lineNumber()) {
695 info.sourceFile = deprecatedURLForReporting(URL { URL { }, callFrame->sourceURL() });
696 info.lineNumber = callFrame->lineNumber();
697 info.columnNumber = callFrame->columnNumber();
698 }
699 }
700 ASSERT(m_client || is<Document>(m_scriptExecutionContext));
701
702 String blockedURI = deprecatedURLForReporting(blockedURL);
703 // FIXME: Is it policy to not use the status code for HTTPS, or is that a bug?
704 unsigned short httpStatusCode = m_selfSourceProtocol == "http" ? m_httpStatusCode : 0;
705
706 // 1. Dispatch violation event.
707 SecurityPolicyViolationEvent::Init violationEventInit;
708 violationEventInit.documentURI = info.documentURI;
709 violationEventInit.referrer = m_referrer;
710 violationEventInit.blockedURI = blockedURI;
711 violationEventInit.violatedDirective = violatedDirective;
712 violationEventInit.effectiveDirective = effectiveViolatedDirective;
713 violationEventInit.originalPolicy = violatedDirectiveList.header();
714 violationEventInit.sourceFile = info.sourceFile;
715 violationEventInit.disposition = violatedDirectiveList.isReportOnly() ? SecurityPolicyViolationEvent::Disposition::Report : SecurityPolicyViolationEvent::Disposition::Enforce;
716 violationEventInit.statusCode = httpStatusCode;
717 violationEventInit.lineNumber = info.lineNumber;
718 violationEventInit.columnNumber = info.columnNumber;
719 if (m_client)
720 m_client->enqueueSecurityPolicyViolationEvent(WTFMove(violationEventInit));
721 else
722 downcast<Document>(*m_scriptExecutionContext).enqueueSecurityPolicyViolationEvent(WTFMove(violationEventInit));
723
724 // 2. Send violation report (if applicable).
725 auto& reportURIs = violatedDirectiveList.reportURIs();
726 if (reportURIs.isEmpty())
727 return;
728
729 // We need to be careful here when deciding what information to send to the
730 // report-uri. Currently, we send only the current document's URL and the
731 // directive that was violated. The document's URL is safe to send because
732 // it's the document itself that's requesting that it be sent. You could
733 // make an argument that we shouldn't send HTTPS document URLs to HTTP
734 // report-uris (for the same reasons that we suppress the Referer in that
735 // case), but the Referer is sent implicitly whereas this request is only
736 // sent explicitly. As for which directive was violated, that's pretty
737 // harmless information.
738
739 auto cspReport = JSON::Object::create();
740 cspReport->setString("document-uri"_s, info.documentURI);
741 cspReport->setString("referrer"_s, m_referrer);
742 cspReport->setString("violated-directive"_s, violatedDirective);
743 cspReport->setString("effective-directive"_s, effectiveViolatedDirective);
744 cspReport->setString("original-policy"_s, violatedDirectiveList.header());
745 cspReport->setString("blocked-uri"_s, blockedURI);
746 cspReport->setInteger("status-code"_s, httpStatusCode);
747 if (!info.sourceFile.isNull()) {
748 cspReport->setString("source-file"_s, info.sourceFile);
749 cspReport->setInteger("line-number"_s, info.lineNumber);
750 cspReport->setInteger("column-number"_s, info.columnNumber);
751 }
752
753 auto reportObject = JSON::Object::create();
754 reportObject->setObject("csp-report"_s, WTFMove(cspReport));
755
756 auto report = FormData::create(reportObject->toJSONString().utf8());
757
758 if (m_client) {
759 for (const auto& url : reportURIs)
760 m_client->sendCSPViolationReport(URL { m_protectedURL, url }, report.copyRef());
761 } else {
762 auto& document = downcast<Document>(*m_scriptExecutionContext);
763 for (const auto& url : reportURIs)
764 PingLoader::sendViolationReport(*document.frame(), URL { m_protectedURL, url }, report.copyRef(), ViolationReportType::ContentSecurityPolicy);
765 }
766}

870void ContentSecurityPolicy::upgradeInsecureRequestIfNeeded(URL& url, InsecureRequestType requestType) const
871{
872 if (!url.protocolIs("http") && !url.protocolIs("ws"))
873 return;
874
875 bool upgradeRequest = m_insecureNavigationRequestsToUpgrade.contains(SecurityOriginData::fromURL(url));
876 if (requestType == InsecureRequestType::Load || requestType == InsecureRequestType::FormSubmission)
877 upgradeRequest |= m_upgradeInsecureRequests;
878
879 if (!upgradeRequest)
880 return;
881
882 if (url.protocolIs("http"))
883 url.setProtocol("https");
884 else {
885 ASSERT(url.protocolIs("ws"));
886 url.setProtocol("wss");
887 }
888
889 if (url.port() && url.port().value() == 80)
890 url.setPort(443);
891}

893void ContentSecurityPolicy::setUpgradeInsecureRequests(bool upgradeInsecureRequests)
894{
895 m_upgradeInsecureRequests = upgradeInsecureRequests;
896 if (!m_upgradeInsecureRequests)
897 return;
898
899 if (!m_scriptExecutionContext)
900 return;
901
902 // Store the upgrade domain as an 'insecure' protocol so we can quickly identify
903 // origins we should upgrade.
904 URL upgradeURL = m_scriptExecutionContext->url();
905 if (upgradeURL.protocolIs("https"))
906 upgradeURL.setProtocol("http");
907 else if (upgradeURL.protocolIs("wss"))
908 upgradeURL.setProtocol("ws");
909
910 m_insecureNavigationRequestsToUpgrade.add(SecurityOriginData::fromURL(upgradeURL));
911}
