26#pragma once
27
28#include "SecurityPolicyViolationEvent.h"
29#include <JavaScriptCore/ConsoleTypes.h>
30#include <wtf/text/WTFString.h>

32namespace WebCore {
33
34class FormData;
35
36struct CSPInfo {
37 String documentURI;
38 String sourceFile;
39 int lineNumber { 0 };
40 int columnNumber { 0 };
41};
42
43struct WEBCORE_EXPORT ContentSecurityPolicyClient {
44 // An inline function cannot be the first non-abstract virtual function declared
45 // in the class as it results in the vtable being generated as a weak symbol.
46 // This hurts performance (in Mac OS X at least, when loading frameworks), so we
47 // don't want to do it in WebKit.
48 virtual void willSendCSPViolationReport(CSPInfo&);
49
50 virtual ~ContentSecurityPolicyClient() = default;
51
52 virtual void addConsoleMessage(MessageSource, MessageLevel, const String&, unsigned long requestIdentifier = 0) = 0;
53 virtual void sendCSPViolationReport(URL&&, Ref<FormData>&&) = 0;
54 virtual void enqueueSecurityPolicyViolationEvent(SecurityPolicyViolationEvent::Init&&) = 0;
55};
56
57} // namespace WebCore
