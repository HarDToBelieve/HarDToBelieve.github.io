74// Used to compute the comparison URL when checking frame-ancestors. We do this weird conversion so that child
75// frames of a page with a unique origin (e.g. about:blank) are not blocked due to their frame-ancestors policy
76// and do not need to add the parent's URL to their policy. The latter could allow the child page to be framed
77// by anyone. See <https://github.com/w3c/webappsec/issues/311> for more details.
78static inline URL urlFromOrigin(const SecurityOrigin& origin)
79{
80 return { URL { }, origin.toString() };
81}
82
83static inline bool checkFrameAncestors(ContentSecurityPolicySourceListDirective* directive, const Frame& frame)
84{
85 if (!directive)
86 return true;
87 bool didReceiveRedirectResponse = false;
88 for (Frame* current = frame.tree().parent(); current; current = current->tree().parent()) {
89 URL origin = urlFromOrigin(current->document()->securityOrigin());
90 if (!origin.isValid() || !directive->allows(origin, didReceiveRedirectResponse, ContentSecurityPolicySourceListDirective::ShouldAllowEmptyURLIfSourceListIsNotNone::No))
91 return false;
92 }
93 return true;
94}
95
96static inline bool checkFrameAncestors(ContentSecurityPolicySourceListDirective* directive, const Vector<RefPtr<SecurityOrigin>>& ancestorOrigins)
97{
98 if (!directive)
99 return true;
100 bool didReceiveRedirectResponse = false;
101 for (auto& origin : ancestorOrigins) {
102 URL originURL = urlFromOrigin(*origin);
103 if (!originURL.isValid() || !directive->allows(originURL, didReceiveRedirectResponse, ContentSecurityPolicySourceListDirective::ShouldAllowEmptyURLIfSourceListIsNotNone::No))
104 return false;
105 }
106 return true;
107}

109static inline bool checkMediaType(ContentSecurityPolicyMediaListDirective* directive, const String& type, const String& typeAttribute)
110{
111 if (!directive)
112 return true;
113 if (typeAttribute.isEmpty() || typeAttribute.stripWhiteSpace() != type)
114 return false;
115 return directive->allows(type);

125std::unique_ptr<ContentSecurityPolicyDirectiveList> ContentSecurityPolicyDirectiveList::create(ContentSecurityPolicy& policy, const String& header, ContentSecurityPolicyHeaderType type, ContentSecurityPolicy::PolicyFrom from)
126{
127 auto directives = makeUnique<ContentSecurityPolicyDirectiveList>(policy, type);
128 directives->parse(header, from);
129
130 if (!checkEval(directives->operativeDirective(directives->m_scriptSrc.get()))) {
131 directives->setEvalDisabledErrorMessage(makeString("Refused to evaluate a string as JavaScript because 'unsafe-eval' is not an allowed source of script in the following Content Security Policy directive: \"", directives->operativeDirective(directives->m_scriptSrc.get())->text(), "\".\n"));
132 directives->setWebAssemblyDisabledErrorMessage(makeString("Refused to create a WebAssembly object because 'unsafe-eval' is not an allowed source of script in the following Content Security Policy directive: \"", directives->operativeDirective(directives->m_scriptSrc.get())->text(), "\".\n"));
133 }
134
135 if (directives->isReportOnly() && directives->reportURIs().isEmpty())
136 policy.reportMissingReportURI(header);
137
138 return directives;
139}

326// policy = directive-list
327// directive-list = [ directive *( ";" [ directive ] ) ]
328//
329void ContentSecurityPolicyDirectiveList::parse(const String& policy, ContentSecurityPolicy::PolicyFrom policyFrom)
330{
331 m_header = policy;
332 if (policy.isEmpty())
333 return;
334
335 readCharactersForParsing(policy, [&](auto buffer) {
336 while (buffer.hasCharactersRemaining()) {
337 auto directiveBegin = buffer.position();
338 skipUntil(buffer, ';');
339
340 if (auto directive = parseDirective(StringParsingBuffer { directiveBegin, buffer.position() })) {
341 ASSERT(!directive->name.isEmpty());
342 if (policyFrom == ContentSecurityPolicy::PolicyFrom::Inherited) {
343 if (equalIgnoringASCIICase(directive->name, ContentSecurityPolicyDirectiveNames::upgradeInsecureRequests))
344 continue;
345 } else if (policyFrom == ContentSecurityPolicy::PolicyFrom::HTTPEquivMeta) {
346 if (equalIgnoringASCIICase(directive->name, ContentSecurityPolicyDirectiveNames::sandbox)
347 || equalIgnoringASCIICase(directive->name, ContentSecurityPolicyDirectiveNames::reportURI)
348 || equalIgnoringASCIICase(directive->name, ContentSecurityPolicyDirectiveNames::frameAncestors)) {
349 m_policy.reportInvalidDirectiveInHTTPEquivMeta(directive->name);
350 continue;
351 }
352 } else if (policyFrom == ContentSecurityPolicy::PolicyFrom::InheritedForPluginDocument) {
353 if (!equalIgnoringASCIICase(directive->name, ContentSecurityPolicyDirectiveNames::pluginTypes)
354 && !equalIgnoringASCIICase(directive->name, ContentSecurityPolicyDirectiveNames::reportURI))
355 continue;
356 }
357 addDirective(WTFMove(*directive));
358 }
359
360 ASSERT(buffer.atEnd() || *buffer == ';');
361 skipExactly(buffer, ';');
362 }
363 });
364}

366// directive = *WSP [ directive-name [ WSP directive-value ] ]
367// directive-name = 1*( ALPHA / DIGIT / "-" )
368// directive-value = *( WSP / <VCHAR except ";"> )
369//
370template<typename CharacterType> auto ContentSecurityPolicyDirectiveList::parseDirective(StringParsingBuffer<CharacterType> buffer) -> std::optional<ParsedDirective>
371{
372 skipWhile<isASCIISpace>(buffer);
373
374 // Empty directive (e.g. ";;;"). Exit early.
375 if (buffer.atEnd())
376 return std::nullopt;
377
378 auto nameBegin = buffer.position();
379 skipWhile<isDirectiveNameCharacter>(buffer);
380
381 // The directive-name must be non-empty.
382 if (nameBegin == buffer.position()) {
383 skipWhile<isNotASCIISpace>(buffer);
384 m_policy.reportUnsupportedDirective(String(nameBegin, buffer.position() - nameBegin));
385 return std::nullopt;
386 }
387
388 auto name = String(nameBegin, buffer.position() - nameBegin);
389
390 if (buffer.atEnd())
391 return ParsedDirective { WTFMove(name), { } };
392
393 if (!skipExactly<isASCIISpace>(buffer)) {
394 skipWhile<isNotASCIISpace>(buffer);
395 m_policy.reportUnsupportedDirective(String(nameBegin, buffer.position() - nameBegin));
396 return std::nullopt;
397 }
398
399 skipWhile<isASCIISpace>(buffer);
400
401 auto valueBegin = buffer.position();
402 skipWhile<isDirectiveValueCharacter>(buffer);
403
404 if (!buffer.atEnd()) {
405 m_policy.reportInvalidDirectiveValueCharacter(name, String(valueBegin, buffer.end() - valueBegin));
406 return std::nullopt;
407 }
408
409 // The directive-value may be empty.
410 if (valueBegin == buffer.position())
411 return ParsedDirective { WTFMove(name), { } };
412
413 auto value = String(valueBegin, buffer.position() - valueBegin);
414 return ParsedDirective { WTFMove(name), WTFMove(value) };
415}

417void ContentSecurityPolicyDirectiveList::parseReportURI(ParsedDirective&& directive)
418{
419 if (!m_reportURIs.isEmpty()) {
420 m_policy.reportDuplicateDirective(directive.name);
421 return;
422 }
423
424 readCharactersForParsing(directive.value, [&](auto buffer) {
425 auto begin = buffer.position();
426 while (buffer.hasCharactersRemaining()) {
427 skipWhile<isASCIISpace>(buffer);
428
429 auto urlBegin = buffer.position();
430 skipWhile<isNotASCIISpace>(buffer);
431
432 if (urlBegin < buffer.position())
433 m_reportURIs.append(directive.value.substring(urlBegin - begin, buffer.position() - urlBegin));
434 }
435 });
436}

449void ContentSecurityPolicyDirectiveList::applySandboxPolicy(ParsedDirective&& directive)
450{
451 if (m_reportOnly) {
452 m_policy.reportInvalidDirectiveInReportOnlyMode(directive.name);
453 return;
454 }
455 if (m_haveSandboxPolicy) {
456 m_policy.reportDuplicateDirective(directive.name);
457 return;
458 }
459 m_haveSandboxPolicy = true;
460 String invalidTokens;
461 m_policy.enforceSandboxFlags(SecurityContext::parseSandboxPolicy(WTFMove(directive.value), invalidTokens));
462 if (!invalidTokens.isNull())
463 m_policy.reportInvalidSandboxFlags(invalidTokens);
464}

489void ContentSecurityPolicyDirectiveList::addDirective(ParsedDirective&& directive)
490{
491 ASSERT(!directive.name.isEmpty());
492
493 if (equalIgnoringASCIICase(directive.name, ContentSecurityPolicyDirectiveNames::defaultSrc)) {
494 setCSPDirective<ContentSecurityPolicySourceListDirective>(WTFMove(directive), m_defaultSrc);
495 m_policy.addHashAlgorithmsForInlineScripts(m_defaultSrc->hashAlgorithmsUsed());
496 m_policy.addHashAlgorithmsForInlineStylesheets(m_defaultSrc->hashAlgorithmsUsed());
497 } else if (equalIgnoringASCIICase(directive.name, ContentSecurityPolicyDirectiveNames::scriptSrc)) {
498 setCSPDirective<ContentSecurityPolicySourceListDirective>(WTFMove(directive), m_scriptSrc);
499 m_policy.addHashAlgorithmsForInlineScripts(m_scriptSrc->hashAlgorithmsUsed());
500 } else if (equalIgnoringASCIICase(directive.name, ContentSecurityPolicyDirectiveNames::styleSrc)) {
501 setCSPDirective<ContentSecurityPolicySourceListDirective>(WTFMove(directive), m_styleSrc);
502 m_policy.addHashAlgorithmsForInlineStylesheets(m_styleSrc->hashAlgorithmsUsed());
503 } else if (equalIgnoringASCIICase(directive.name, ContentSecurityPolicyDirectiveNames::objectSrc))
504 setCSPDirective<ContentSecurityPolicySourceListDirective>(WTFMove(directive), m_objectSrc);
505 else if (equalIgnoringASCIICase(directive.name, ContentSecurityPolicyDirectiveNames::frameSrc)) {
506 // FIXME: Log to console "The frame-src directive is deprecated. Use the child-src directive instead."
507 // See <https://bugs.webkit.org/show_bug.cgi?id=155773>.
508 setCSPDirective<ContentSecurityPolicySourceListDirective>(WTFMove(directive), m_frameSrc);
509 } else if (equalIgnoringASCIICase(directive.name, ContentSecurityPolicyDirectiveNames::imgSrc))
510 setCSPDirective<ContentSecurityPolicySourceListDirective>(WTFMove(directive), m_imgSrc);
511 else if (equalIgnoringASCIICase(directive.name, ContentSecurityPolicyDirectiveNames::fontSrc))
512 setCSPDirective<ContentSecurityPolicySourceListDirective>(WTFMove(directive), m_fontSrc);
513#if ENABLE(APPLICATION_MANIFEST)
514 else if (equalIgnoringASCIICase(directive.name, ContentSecurityPolicyDirectiveNames::manifestSrc))
515 setCSPDirective<ContentSecurityPolicySourceListDirective>(WTFMove(directive), m_manifestSrc);
516#endif
517 else if (equalIgnoringASCIICase(directive.name, ContentSecurityPolicyDirectiveNames::mediaSrc))
518 setCSPDirective<ContentSecurityPolicySourceListDirective>(WTFMove(directive), m_mediaSrc);
519 else if (equalIgnoringASCIICase(directive.name, ContentSecurityPolicyDirectiveNames::connectSrc))
520 setCSPDirective<ContentSecurityPolicySourceListDirective>(WTFMove(directive), m_connectSrc);
521 else if (equalIgnoringASCIICase(directive.name, ContentSecurityPolicyDirectiveNames::childSrc))
522 setCSPDirective<ContentSecurityPolicySourceListDirective>(WTFMove(directive), m_childSrc);
523 else if (equalIgnoringASCIICase(directive.name, ContentSecurityPolicyDirectiveNames::formAction))
524 setCSPDirective<ContentSecurityPolicySourceListDirective>(WTFMove(directive), m_formAction);
525 else if (equalIgnoringASCIICase(directive.name, ContentSecurityPolicyDirectiveNames::baseURI))
526 setCSPDirective<ContentSecurityPolicySourceListDirective>(WTFMove(directive), m_baseURI);
527 else if (equalIgnoringASCIICase(directive.name, ContentSecurityPolicyDirectiveNames::frameAncestors)) {
528 if (m_reportOnly) {
529 m_policy.reportInvalidDirectiveInReportOnlyMode(directive.name);
530 return;
531 }
532 setCSPDirective<ContentSecurityPolicySourceListDirective>(WTFMove(directive), m_frameAncestors);
533 } else if (equalIgnoringASCIICase(directive.name, ContentSecurityPolicyDirectiveNames::pluginTypes))
534 setCSPDirective<ContentSecurityPolicyMediaListDirective>(WTFMove(directive), m_pluginTypes);
535 else if (equalIgnoringASCIICase(directive.name, ContentSecurityPolicyDirectiveNames::sandbox))
536 applySandboxPolicy(WTFMove(directive));
537 else if (equalIgnoringASCIICase(directive.name, ContentSecurityPolicyDirectiveNames::reportURI))
538 parseReportURI(WTFMove(directive));
539 else if (equalIgnoringASCIICase(directive.name, ContentSecurityPolicyDirectiveNames::upgradeInsecureRequests))
540 setUpgradeInsecureRequests(WTFMove(directive));
541 else if (equalIgnoringASCIICase(directive.name, ContentSecurityPolicyDirectiveNames::blockAllMixedContent))
542 setBlockAllMixedContentEnabled(WTFMove(directive));
543 else
544 m_policy.reportUnsupportedDirective(WTFMove(directive.name));
545}
