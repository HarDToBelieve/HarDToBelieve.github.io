48#if ENABLE(FULLSCREEN_API)
49static bool isHiddenBehindFullscreenElement(const Node& descendantCandidate)
50{
51 // Fullscreen status is propagated on the ancestor document chain all the way to the top document.
52 auto& document = descendantCandidate.document();
53 auto* topMostFullScreenElement = document.topDocument().fullscreenManager().fullscreenElement();
54 if (!topMostFullScreenElement)
55 return false;
56
57 // If the document where the node lives does not have an active fullscreen element, it is a sibling/nephew document -> not a descendant.
58 auto* fullscreenElement = document.fullscreenManager().fullscreenElement();
59 if (!fullscreenElement)
60 return true;
61 return !descendantCandidate.isDescendantOf(*fullscreenElement);
62}
63#endif

65bool ContentChangeObserver::isVisuallyHidden(const Node& node)
66{
67 if (!node.renderStyle())
68 return true;
69
70 auto& style = *node.renderStyle();
71 if (style.display() == DisplayType::None)
72 return true;
73
74 if (style.visibility() == Visibility::Hidden)
75 return true;
76
77 if (!style.opacity())
78 return true;
79
80 auto width = style.logicalWidth();
81 auto height = style.logicalHeight();
82 if ((width.isFixed() && !width.value()) || (height.isFixed() && !height.value()))
83 return true;
84
85 auto top = style.logicalTop();
86 auto left = style.logicalLeft();
87 // FIXME: This is trying to check if the element is outside of the viewport. This is incorrect for many reasons.
88 if (left.isFixed() && width.isFixed() && -left.value() >= width.value())
89 return true;
90 if (top.isFixed() && height.isFixed() && -top.value() >= height.value())
91 return true;
92
93 // It's a common technique used to position content offscreen.
94 if (style.hasOutOfFlowPosition() && left.isFixed() && left.value() <= -999)
95 return true;
96
97 // FIXME: Check for other cases like zero height with overflow hidden.
98 auto maxHeight = style.maxHeight();
99 if (maxHeight.isFixed() && !maxHeight.value())
100 return true;
101
102 // Special case opacity, because a descendant with non-zero opacity should still be considered hidden when one of its ancetors has opacity: 0;
103 // YouTube.com has this setup with the bottom control bar.
104 constexpr static unsigned numberOfAncestorsToCheckForOpacity = 4;
105 unsigned i = 0;
106 for (auto* parent = node.parentNode(); parent && i < numberOfAncestorsToCheckForOpacity; parent = parent->parentNode(), ++i) {
107 if (!parent->renderStyle() || !parent->renderStyle()->opacity())
108 return true;
109 }
110
111#if ENABLE(FULLSCREEN_API)
112 if (isHiddenBehindFullscreenElement(node))
113 return true;
114#endif
115 return false;
116}

118bool ContentChangeObserver::isConsideredVisible(const Node& node)
119{
120 if (isVisuallyHidden(node))
121 return false;
122
123 auto& style = *node.renderStyle();
124 auto width = style.logicalWidth();
125 // 1px width or height content is not considered visible.
126 if (width.isFixed() && width.value() <= 1)
127 return false;
128
129 auto height = style.logicalHeight();
130 if (height.isFixed() && height.value() <= 1)
131 return false;
132
133 return true;
134}

136bool ContentChangeObserver::isConsideredActionableContent(const Element& candidateElement, ElementHadRenderer hadRenderer) const
137{
138 if (m_document.quirks().shouldTooltipPreventFromProceedingWithClick(candidateElement))
139 return true;
140
141 auto isConsideredClickable = [&] {
142 auto& element = const_cast<Element&>(candidateElement);
143 if (element.isInUserAgentShadowTree())
144 return false;
145
146 if (is<HTMLIFrameElement>(element))
147 return true;
148
149 if (is<HTMLImageElement>(element)) {
150 // This is required to avoid HTMLImageElement's touch callout override logic. See rdar://problem/48937767.
151 return element.Element::willRespondToMouseClickEvents();
152 }
153 bool hasRenderer = element.renderer();
154 auto willRespondToMouseClickEvents = element.willRespondToMouseClickEvents();
155 if (willRespondToMouseClickEvents || !hasRenderer || hadRenderer == ElementHadRenderer::No)
156 return willRespondToMouseClickEvents;
157
158 // In case when the content already had renderers it's not sufficient to check the candidate element only since it might just be the container for the clickable content.
159 for (auto& descendant : descendantsOfType<RenderElement>(*element.renderer())) {
160 if (!descendant.element())
161 continue;
162 if (descendant.element()->willRespondToMouseClickEvents())
163 return true;
164 }
165 return false;
166 };
167 return isConsideredClickable();
168}

218void ContentChangeObserver::didAddTransition(const Element& element, const Animation& transition)
219{
220 if (!m_document.settings().contentChangeObserverEnabled())
221 return;
222 if (hasVisibleChangeState())
223 return;
224 if (!isObservingContentChanges())
225 return;
226 if (!isObservingTransitions())
227 return;
228 if (!transition.isDurationSet() || !transition.isPropertySet())
229 return;
230 if (!isObservedPropertyForTransition(transition.property().id))
231 return;
232 auto transitionEnd = Seconds { transition.duration() + std::max<double>(0, transition.isDelaySet() ? transition.delay() : 0) };
233 if (transitionEnd > maximumDelayForTransitions)
234 return;
235 if (!isVisuallyHidden(element))
236 return;
237 // In case of multiple transitions, the first tranistion wins (and it has to produce a visible content change in order to show up as hover).
238 if (m_elementsWithTransition.contains(&element))
239 return;
240 LOG_WITH_STREAM(ContentObservation, stream << "didAddTransition: transition created on " << &element << " (" << transitionEnd.milliseconds() << "ms).");
241
242 m_elementsWithTransition.add(&element);
243 adjustObservedState(Event::AddedTransition);
244}

246void ContentChangeObserver::didFinishTransition(const Element& element, CSSPropertyID propertyID)
247{
248 if (!isObservedPropertyForTransition(propertyID))
249 return;
250 if (!m_elementsWithTransition.take(&element))
251 return;
252 LOG_WITH_STREAM(ContentObservation, stream << "didFinishTransition: transition finished (" << &element << ").");
253
254 // isConsideredActionableContent may trigger style update through Node::computeEditability. Let's adjust the state in the next runloop.
255 callOnMainThread([weakThis = makeWeakPtr(*this), targetElement = makeWeakPtr(element)] {
256 if (!weakThis || !targetElement)
257 return;
258 if (isVisuallyHidden(*targetElement)) {
259 weakThis->adjustObservedState(Event::EndedTransitionButFinalStyleIsNotDefiniteYet);
260 return;
261 }
262 if (weakThis->isConsideredActionableContent(*targetElement, ElementHadRenderer::Yes))
263 weakThis->elementDidBecomeVisible(*targetElement);
264 weakThis->adjustObservedState(Event::CompletedTransition);
265 });
266}

279void ContentChangeObserver::didInstallDOMTimer(const DOMTimer& timer, Seconds timeout, bool singleShot)
280{
281 if (!m_document.settings().contentChangeObserverEnabled())
282 return;
283 if (!isObservingContentChanges())
284 return;
285 if (!isObservingDOMTimerScheduling())
286 return;
287 if (hasVisibleChangeState())
288 return;
289 if (m_document.activeDOMObjectsAreSuspended())
290 return;
291 if (timeout > maximumDelayForTimers || !singleShot)
292 return;
293 LOG_WITH_STREAM(ContentObservation, stream << "didInstallDOMTimer: register this timer: (" << &timer << ") and observe when it fires.");
294
295 registerDOMTimer(timer);
296 adjustObservedState(Event::InstalledDOMTimer);
297}

420void ContentChangeObserver::elementDidBecomeHidden(const Element& element)
421{
422 LOG_WITH_STREAM(ContentObservation, stream << "elementDidBecomeHidden: element went from visible to hidden: " << &element);
423 // Candidate element is no longer visible.

490void ContentChangeObserver::adjustObservedState(Event event)
491{
492 auto resetToStartObserving = [&] {
493 setHasNoChangeState();
494 clearObservedDOMTimers();
495 clearObservedTransitions();
496 setIsBetweenTouchEndAndMouseMoved(false);
497 setShouldObserveNextStyleRecalc(false);
498 setShouldObserveDOMTimerSchedulingAndTransitions(false);
499 ASSERT(!m_isInObservedStyleRecalc);
500 ASSERT(!m_observedDomTimerIsBeingExecuted);
501 };
502
503 auto notifyClientIfNeeded = [&] {
504 if (isTouchEventBeingDispatched()) {
505 LOG(ContentObservation, "notifyClientIfNeeded: Touch event is being dispatched. No need to notify the client.");
506 return;
507 }
508 if (isBetweenTouchEndAndMouseMoved()) {
509 LOG(ContentObservation, "notifyClientIfNeeded: Not reached mouseMoved yet. No need to notify the client.");
510 return;
511 }
512 if (isMouseMovedEventBeingDispatched()) {
513 LOG(ContentObservation, "notifyClientIfNeeded: in mouseMoved call. No need to notify the client.");
514 return;
515 }
516 if (isObservationTimeWindowActive()) {
517 LOG(ContentObservation, "notifyClientIfNeeded: Inside the fixed window observation. No need to notify the client.");
518 return;
519 }
520
521 // The fixed observation window (which is the final step in content observation) is closed and now we check if are still waiting for timers or animations to finish.
522 if (hasPendingActivity()) {
523 LOG(ContentObservation, "notifyClientIfNeeded: We are still waiting on some events.");
524 return;
525 }
526
527 // First demote to "no change" because we've got no pending activity anymore.
528 if (observedContentChange() == WKContentIndeterminateChange)
529 setHasNoChangeState();
530
531 LOG_WITH_STREAM(ContentObservation, stream << "notifyClientIfNeeded: sending observedContentChange ->" << observedContentChange());
532 ASSERT(m_document.page());
533 ASSERT(m_document.frame());
534 m_document.page()->chrome().client().didFinishContentChangeObserving(*m_document.frame(), observedContentChange());
535 stopContentObservation();
536 };
537
538 // These user initiated events trigger content observation (touchStart and mouseMove).
539 {
540 if (event == Event::StartedTouchStartEventDispatching) {
541 resetToStartObserving();
542 setShouldObserveDOMTimerSchedulingAndTransitions(true);
543 return;
544 }
545 if (event == Event::EndedTouchStartEventDispatching) {
546 setShouldObserveDOMTimerSchedulingAndTransitions(false);
547 setIsBetweenTouchEndAndMouseMoved(true);
548 return;
549 }
550 if (event == Event::StartedMouseMovedEventDispatching) {
551 ASSERT(!m_document.hasPendingStyleRecalc());
552 if (!isBetweenTouchEndAndMouseMoved())
553 resetToStartObserving();
554 setIsBetweenTouchEndAndMouseMoved(false);
555 setShouldObserveDOMTimerSchedulingAndTransitions(!hasVisibleChangeState());
556 return;
557 }
558 if (event == Event::EndedMouseMovedEventDispatching) {
559 setShouldObserveDOMTimerSchedulingAndTransitions(false);
560 return;
561 }
562 }
563 // Fixed window observation starts soon after mouseMove when we don't have a definite answer to whether we should proceed with hover or click.
564 {
565 if (event == Event::StartedFixedObservationTimeWindow) {
566 ASSERT(!hasVisibleChangeState());
567 setHasIndeterminateState();
568 return;
569 }
570 if (event == Event::EndedFixedObservationTimeWindow) {
571 notifyClientIfNeeded();
572 return;
573 }
574 if (event == Event::WillNotProceedWithFixedObservationTimeWindow) {
575 notifyClientIfNeeded();
576 return;
577 }
578 }
579 // These events (DOM timer, transition and style recalc) could trigger style changes that are candidates to visibility checking.
580 {
581 if (event == Event::InstalledDOMTimer || event == Event::AddedTransition) {
582 ASSERT(!hasVisibleChangeState());
583 setHasIndeterminateState();
584 return;
585 }
586 if (event == Event::RemovedDOMTimer || event == Event::CanceledTransition) {
587 notifyClientIfNeeded();
588 return;
589 }
590 if (event == Event::StartedDOMTimerExecution) {
591 ASSERT(isObservationTimeWindowActive() || observedContentChange() == WKContentIndeterminateChange);
592 return;
593 }
594 if (event == Event::EndedDOMTimerExecution) {
595 if (m_document.hasPendingStyleRecalc()) {
596 setShouldObserveNextStyleRecalc(true);
597 return;
598 }
599 notifyClientIfNeeded();
600 return;
601 }
602 if (event == Event::EndedTransitionButFinalStyleIsNotDefiniteYet) {
603 // onAnimationEnd can be called while in the middle of resolving the document (synchronously) or
604 // asynchronously right before the style update is issued. It also means we don't know whether this animation ends up producing visible content yet.
605 if (m_document.inStyleRecalc()) {
606 // We need to start observing this style change synchronously.
607 m_isInObservedStyleRecalc = true;
608 return;
609 }
610 setShouldObserveNextStyleRecalc(true);
611 return;
612 }
613 if (event == Event::CompletedTransition) {
614 if (m_document.inStyleRecalc()) {
615 m_isInObservedStyleRecalc = true;
616 return;
617 }
618 notifyClientIfNeeded();
619 return;
620 }
621 if (event == Event::StartedStyleRecalc) {
622 setShouldObserveNextStyleRecalc(false);
623 ASSERT(isObservationTimeWindowActive() || observedContentChange() == WKContentIndeterminateChange);
624 return;
625 }
626 if (event == Event::EndedStyleRecalc) {
627 notifyClientIfNeeded();
628 return;
629 }
630 }
631 // Either the page decided to call preventDefault on the touch action or the tap gesture evolved to some other gesture (long press, double tap).
632 if (event == Event::WillNotProceedWithClick) {
633 stopContentObservation();
634 return;
635 }
636 // The page produced an visible change on an actionable content.
637 if (event == Event::ElementDidBecomeVisible) {
638 setHasVisibleChangeState();
639 // Stop pending activities. We don't need to observe them anymore.
640 stopObservingPendingActivities();
641 return;
642 }
643}

650ContentChangeObserver::StyleChangeScope::StyleChangeScope(Document& document, const Element& element)
651 : m_contentChangeObserver(document.contentChangeObserver())
652 , m_element(element)
653 , m_hadRenderer(element.renderer())
654{
655 if (m_contentChangeObserver.shouldObserveVisibilityChangeForElement(element))
656 m_wasHidden = isVisuallyHidden(m_element);
657}
658
659ContentChangeObserver::StyleChangeScope::~StyleChangeScope()
660{
661 // Do we track this element?
662 if (!m_wasHidden.has_value())
663 return;
664
665 if (!m_contentChangeObserver.isConsideredActionableContent(m_element, m_hadRenderer ? ElementHadRenderer::Yes : ElementHadRenderer::No))
666 return;
667
668 auto wasVisible = !m_wasHidden.value();
669 auto isVisible = isConsideredVisible(m_element);
670 if (!wasVisible && isVisible)
671 m_contentChangeObserver.elementDidBecomeVisible(m_element);
672 else if (wasVisible && !isVisible)
673 m_contentChangeObserver.elementDidBecomeHidden(m_element);
674}
