82// Create <html><body (style="...")></body></html> doing minimal amount of work.
83void Frame::initWithSimpleHTMLDocument(const String& style, const URL& url)
84{
85 m_loader->initForSynthesizedDocument(url);
86
87 auto document = HTMLDocument::createSynthesizedDocument(*this, url);
88 document->setCompatibilityMode(DocumentCompatibilityMode::LimitedQuirksMode);
89 document->createDOMWindow();
90 setDocument(document.copyRef());
91
92 auto rootElement = HTMLHtmlElement::create(document);
93
94 auto body = HTMLBodyElement::create(document);
95 if (!style.isEmpty())
96 body->setAttribute(HTMLNames::styleAttr, style);
97
98 rootElement->appendChild(body);
99 document->appendChild(rootElement);
100}

112NSArray *Frame::wordsInCurrentParagraph() const
113{
114 document()->updateLayout();
115
116 if (!page() || !page()->selection().isCaret())
117 return nil;
118
119 VisiblePosition position(page()->selection().start(), page()->selection().affinity());
120 VisiblePosition end(position);
121
122 if (!isStartOfParagraph(end)) {
123 VisiblePosition previous = end.previous();
124 UChar c(previous.characterAfter());
125 // FIXME: Should use something from ICU or ASCIICType that is not subject to POSIX current language rather than iswpunct.
126 if (!iswpunct(c) && !isSpaceOrNewline(c) && c != noBreakSpace)
127 end = startOfWord(end);
128 }
129 VisiblePosition start(startOfParagraph(end));
130
131 auto searchRange = makeSimpleRange(start, end);
132 if (!searchRange || searchRange->collapsed())
133 return nil;
134
135 NSMutableArray *words = [NSMutableArray array];
136
137 WordAwareIterator it(*searchRange);
138 while (!it.atEnd()) {
139 StringView text = it.text();
140 int length = text.length();
141 if (length > 1 || !isSpaceOrNewline(text[0])) {
142 int startOfWordBoundary = 0;
143 for (int i = 1; i < length; i++) {
144 if (isSpaceOrNewline(text[i]) || text[i] == noBreakSpace) {
145 int wordLength = i - startOfWordBoundary;
146 if (wordLength > 0) {
147 RetainPtr<NSString> chunk = text.substring(startOfWordBoundary, wordLength).createNSString();
148 [words addObject:chunk.get()];
149 }
150 startOfWordBoundary += wordLength + 1;
151 }
152 }
153 if (startOfWordBoundary < length) {
154 RetainPtr<NSString> chunk = text.substring(startOfWordBoundary, length - startOfWordBoundary).createNSString();
155 [words addObject:chunk.get()];
156 }
157 }
158 it.advance();
159 }
160
161 if ([words count] > 0 && isEndOfParagraph(position) && !isStartOfParagraph(position)) {
162 VisiblePosition previous = position.previous();
163 UChar c(previous.characterAfter());
164 if (!isSpaceOrNewline(c) && c != noBreakSpace)
165 [words removeLastObject];
166 }
167
168 return words;
169}

174CGRect Frame::renderRectForPoint(CGPoint point, bool* isReplaced, float* fontSize) const
175{
176 *isReplaced = false;
177 *fontSize = 0;
178
179 if (!m_doc || !m_doc->renderBox())
180 return CGRectZero;
181
182 // FIXME: Why this layer check?
183 RenderLayer* layer = m_doc->renderBox()->layer();
184 if (!layer)
185 return CGRectZero;
186
187 constexpr OptionSet<HitTestRequest::Type> hitType { HitTestRequest::Type::ReadOnly, HitTestRequest::Type::Active, HitTestRequest::Type::DisallowUserAgentShadowContent, HitTestRequest::Type::AllowChildFrameContent };
188 auto result = eventHandler().hitTestResultAtPoint(IntPoint(roundf(point.x), roundf(point.y)), hitType);
189
190 Node* node = result.innerNode();
191 if (!node)
192 return CGRectZero;
193
194 RenderObject* hitRenderer = node->renderer();
195 RenderObject* renderer = hitRenderer;
196#if RECT_LOGGING
197 printf("\n%f %f\n", point.x, point.y);
198#endif
199 while (renderer && !renderer->isBody() && !renderer->isDocumentElementRenderer()) {
200#if RECT_LOGGING
201 CGRect rect = renderer->absoluteBoundingBoxRect(true);
202 if (renderer->node()) {
203 const char *nodeName = renderer->node()->nodeName().ascii().data();
204 printf("%s %f %f %f %f\n", nodeName, rect.origin.x, rect.origin.y, rect.size.width, rect.size.height);
205 }
206#endif
207 if (renderer->isRenderBlock() || renderer->isInlineBlockOrInlineTable() || renderer->isReplaced()) {
208 *isReplaced = renderer->isReplaced();
209#if CHECK_FONT_SIZE
210 for (RenderObject* textRenderer = hitRenderer; textRenderer; textRenderer = textRenderer->traverseNext(hitRenderer)) {
211 if (textRenderer->isText()) {
212 *fontSize = textRenderer->font(true).pixelSize();
213 break;
214 }
215 }
216#endif
217 IntRect targetRect = renderer->absoluteBoundingBoxRect(true);
218 for (Widget* currView = &(renderer->view().frameView()); currView && currView != view(); currView = currView->parent())
219 targetRect = currView->convertToContainingView(targetRect);
220
221 return targetRect;
222 }
223 renderer = renderer->parent();
224 }
225
226 return CGRectZero;
227}

229void Frame::betterApproximateNode(const IntPoint& testPoint, const NodeQualifier& nodeQualifierFunction, Node*& best, Node* failedNode, IntPoint& bestPoint, IntRect& bestRect, const IntRect& testRect)
230{
231 IntRect candidateRect;
232 constexpr OptionSet<HitTestRequest::Type> hitType { HitTestRequest::Type::ReadOnly, HitTestRequest::Type::Active, HitTestRequest::Type::DisallowUserAgentShadowContent, HitTestRequest::Type::AllowVisibleChildFrameContentOnly };
233 auto* candidate = nodeQualifierFunction(eventHandler().hitTestResultAtPoint(testPoint, hitType), failedNode, &candidateRect);
234
235 // Bail if we have no candidate, or the candidate is already equal to our current best node,
236 // or our candidate is the avoidedNode and there is a current best node.
237 if (!candidate || candidate == best)
238 return;
239
240 // The document should never be considered the best alternative.
241 if (candidate->isDocumentNode())
242 return;
243
244 if (best) {
245 IntRect bestIntersect = intersection(testRect, bestRect);
246 IntRect candidateIntersect = intersection(testRect, candidateRect);
247 // if the candidate intersection is smaller than the current best intersection, bail.
248 if (candidateIntersect.width() * candidateIntersect.height() <= bestIntersect.width() * bestIntersect.height())
249 return;
250 }
251
252 // At this point we either don't have a previous best, or our current candidate has a better intersection.
253 best = candidate;
254 bestPoint = testPoint;
255 bestRect = candidateRect;
256}

258bool Frame::hitTestResultAtViewportLocation(const FloatPoint& viewportLocation, HitTestResult& hitTestResult, IntPoint& center)
259{
260 if (!m_doc || !m_doc->renderView())
261 return false;
262
263 FrameView* view = m_view.get();
264 if (!view)
265 return false;
266
267 center = view->windowToContents(roundedIntPoint(viewportLocation));
268 constexpr OptionSet<HitTestRequest::Type> hitType { HitTestRequest::Type::ReadOnly, HitTestRequest::Type::Active, HitTestRequest::Type::DisallowUserAgentShadowContent, HitTestRequest::Type::AllowVisibleChildFrameContentOnly };
269 hitTestResult = eventHandler().hitTestResultAtPoint(center, hitType);
270 return true;
271}

273Node* Frame::qualifyingNodeAtViewportLocation(const FloatPoint& viewportLocation, FloatPoint& adjustedViewportLocation, const NodeQualifier& nodeQualifierFunction, ShouldApproximate shouldApproximate, ShouldFindRootEditableElement shouldFindRootEditableElement)
274{
275 adjustedViewportLocation = viewportLocation;
276
277 IntPoint testCenter;
278 HitTestResult candidateInfo;
279 if (!hitTestResultAtViewportLocation(viewportLocation, candidateInfo, testCenter))
280 return nullptr;
281
282 IntPoint bestPoint = testCenter;
283
284 // We have the candidate node at the location, check whether it or one of its ancestors passes
285 // the qualifier function, which typically checks if the node responds to a particular event type.
286 Node* approximateNode = nodeQualifierFunction(candidateInfo, 0, 0);
287
288 if (shouldFindRootEditableElement == ShouldFindRootEditableElement::Yes && approximateNode && approximateNode->isContentEditable()) {
289 // If we are in editable content, we look for the root editable element.
290 approximateNode = approximateNode->rootEditableElement();
291 // If we have a focusable node, there is no need to approximate.
292 if (approximateNode)
293 shouldApproximate = ShouldApproximate::No;
294 }
295
296 float scale = page() ? page()->pageScaleFactor() : 1;
297 float ppiFactor = screenPPIFactor();
298
299 static const float unscaledSearchRadius = 15;
300 int searchRadius = static_cast<int>(unscaledSearchRadius * ppiFactor / scale);
301
302 if (approximateNode && shouldApproximate == ShouldApproximate::Yes) {
303 const float testOffsets[] = {
304 -.3f, -.3f,
305 -.6f, -.6f,
306 +.3f, +.3f,
307 -.9f, -.9f,
308 };
309
310 Node* originalApproximateNode = approximateNode;
311 for (unsigned n = 0; n < WTF_ARRAY_LENGTH(testOffsets); n += 2) {
312 IntSize testOffset(testOffsets[n] * searchRadius, testOffsets[n + 1] * searchRadius);
313 IntPoint testPoint = testCenter + testOffset;
314
315 constexpr OptionSet<HitTestRequest::Type> hitType { HitTestRequest::Type::ReadOnly, HitTestRequest::Type::Active, HitTestRequest::Type::DisallowUserAgentShadowContent, HitTestRequest::Type::AllowChildFrameContent };
316 auto candidateInfo = eventHandler().hitTestResultAtPoint(testPoint, hitType);
317 Node* candidateNode = nodeQualifierFunction(candidateInfo, 0, 0);
318 if (candidateNode && candidateNode->isDescendantOf(originalApproximateNode)) {
319 approximateNode = candidateNode;
320 bestPoint = testPoint;
321 break;
322 }
323 }
324 } else if (!approximateNode && shouldApproximate == ShouldApproximate::Yes) {
325 // Grab the closest parent element of our failed candidate node.
326 Node* candidate = candidateInfo.innerNode();
327 Node* failedNode = candidate;
328
329 while (candidate && !candidate->isElementNode())
330 candidate = candidate->parentInComposedTree();
331
332 if (candidate)
333 failedNode = candidate;
334
335 // The center point was tested earlier.
336 const float testOffsets[] = {
337 -.3f, -.3f,
338 +.3f, -.3f,
339 -.3f, +.3f,
340 +.3f, +.3f,
341 -.6f, -.6f,
342 +.6f, -.6f,
343 -.6f, +.6f,
344 +.6f, +.6f,
345 -1.f, 0,
346 +1.f, 0,
347 0, +1.f,
348 0, -1.f,
349 };
350 IntRect bestFrame;
351 IntRect testRect(testCenter, IntSize());
352 testRect.inflate(searchRadius);
353 int currentTestRadius = 0;
354 for (unsigned n = 0; n < WTF_ARRAY_LENGTH(testOffsets); n += 2) {
355 IntSize testOffset(testOffsets[n] * searchRadius, testOffsets[n + 1] * searchRadius);
356 IntPoint testPoint = testCenter + testOffset;
357 int testRadius = std::max(abs(testOffset.width()), abs(testOffset.height()));
358 if (testRadius > currentTestRadius) {
359 // Bail out with the best match within a radius
360 currentTestRadius = testRadius;
361 if (approximateNode)
362 break;
363 }
364 betterApproximateNode(testPoint, nodeQualifierFunction, approximateNode, failedNode, bestPoint, bestFrame, testRect);
365 }
366 }
367
368 if (approximateNode) {
369 IntPoint p = m_view->contentsToWindow(bestPoint);
370 adjustedViewportLocation = p;
371 if (shouldFindRootEditableElement == ShouldFindRootEditableElement::Yes && approximateNode->isContentEditable()) {
372 // When in editable content, look for the root editable node again,
373 // since this could be the node found with the approximation.
374 approximateNode = approximateNode->rootEditableElement();
375 }
376 }
377
378 return approximateNode;
379}

391static bool nodeIsMouseFocusable(Node& node)
392{
393 if (!is<Element>(node))
394 return false;
395
396 auto& element = downcast<Element>(node);
397 if (element.isMouseFocusable())
398 return true;
399
400 if (auto shadowRoot = makeRefPtr(element.shadowRoot())) {
401 if (shadowRoot->delegatesFocus()) {
402 for (auto& node : composedTreeDescendants(element)) {
403 if (is<Element>(node) && downcast<Element>(node).isMouseFocusable())
404 return true;
405 }
406 }
407 }
408
409 return false;
410}

414 return node.willRespondToMouseClickEvents() || node.willRespondToMouseMoveEvents() || nodeIsMouseFocusable(node);

417Node* Frame::approximateNodeAtViewportLocationLegacy(const FloatPoint& viewportLocation, FloatPoint& adjustedViewportLocation)
418{
419 // This function is only used for UIWebView.
420 auto&& ancestorRespondingToClickEvents = [](const HitTestResult& hitTestResult, Node* terminationNode, IntRect* nodeBounds) -> Node* {
421 bool bodyHasBeenReached = false;
422 bool pointerCursorStillValid = true;
423
424 if (nodeBounds)
425 *nodeBounds = IntRect();
426
427 auto node = hitTestResult.innerNode();
428 if (!node)
429 return nullptr;
430
431 Node* pointerCursorNode = nullptr;
432 for (; node && node != terminationNode; node = node->parentInComposedTree()) {
433 // We only accept pointer nodes before reaching the body tag.
434 if (node->hasTagName(HTMLNames::bodyTag)) {
435 // Make sure we cover the case of an empty editable body.
436 if (!pointerCursorNode && node->isContentEditable())
437 pointerCursorNode = node;
438 bodyHasBeenReached = true;
439 pointerCursorStillValid = false;
440 }
441
442 // If we already have a pointer, and we reach a table, don't accept it.
443 if (pointerCursorNode && (node->hasTagName(HTMLNames::tableTag) || node->hasTagName(HTMLNames::tbodyTag)))
444 pointerCursorStillValid = false;
445
446 // If we haven't reached the body, and we are still paying attention to pointer cursors, and the node has a pointer cursor.
447 if (pointerCursorStillValid && node->renderStyle() && node->renderStyle()->cursor() == CursorType::Pointer)
448 pointerCursorNode = node;
449 else if (pointerCursorNode) {
450 // We want the lowest unbroken chain of pointer cursors.
451 pointerCursorStillValid = false;
452 }
453
454 if (nodeWillRespondToMouseEvents(*node)) {
455 // If we're at the body or higher, use the pointer cursor node (which may be null).
456 if (bodyHasBeenReached)
457 node = pointerCursorNode;
458
459 // If we are interested about the frame, use it.
460 if (nodeBounds) {
461 // This is a check to see whether this node is an area element. The only way this can happen is if this is the first check.
462 if (node == hitTestResult.innerNode() && node != hitTestResult.innerNonSharedNode() && is<HTMLAreaElement>(*node))
463 *nodeBounds = snappedIntRect(downcast<HTMLAreaElement>(*node).computeRect(hitTestResult.innerNonSharedNode()->renderer()));
464 else if (node && node->renderer())
465 *nodeBounds = node->renderer()->absoluteBoundingBoxRect(true);
466 }
467
468 return node;
469 }
470 }
471
472 return nullptr;
473 };
474
475 return qualifyingNodeAtViewportLocation(viewportLocation, adjustedViewportLocation, WTFMove(ancestorRespondingToClickEvents), ShouldApproximate::Yes);
476}

478static inline NodeQualifier ancestorRespondingToClickEventsNodeQualifier(SecurityOrigin* securityOrigin = nullptr)
479{
480 return [securityOrigin](const HitTestResult& hitTestResult, Node* terminationNode, IntRect* nodeBounds) -> Node* {
481 if (nodeBounds)
482 *nodeBounds = IntRect();
483
484 auto node = hitTestResult.innerNode();
485 if (!node || (securityOrigin && !securityOrigin->isSameOriginAs(node->document().securityOrigin())))
486 return nullptr;
487
488 for (; node && node != terminationNode; node = node->parentInComposedTree()) {
489 if (nodeWillRespondToMouseEvents(*node)) {
490 // If we are interested about the frame, use it.
491 if (nodeBounds) {
492 // This is a check to see whether this node is an area element. The only way this can happen is if this is the first check.
493 if (node == hitTestResult.innerNode() && node != hitTestResult.innerNonSharedNode() && is<HTMLAreaElement>(*node))
494 *nodeBounds = snappedIntRect(downcast<HTMLAreaElement>(*node).computeRect(hitTestResult.innerNonSharedNode()->renderer()));
495 else if (node && node->renderer())
496 *nodeBounds = node->renderer()->absoluteBoundingBoxRect(true);
497 }
498
499 return node;
500 }
501 }
502
503 return nullptr;
504 };
505}

512Node* Frame::nodeRespondingToDoubleClickEvent(const FloatPoint& viewportLocation, FloatPoint& adjustedViewportLocation)
513{
514 auto&& ancestorRespondingToDoubleClickEvent = [](const HitTestResult& hitTestResult, Node* terminationNode, IntRect* nodeBounds) -> Node* {
515 if (nodeBounds)
516 *nodeBounds = IntRect();
517
518 auto* node = hitTestResult.innerNode();
519 if (!node)
520 return nullptr;
521
522 for (; node && node != terminationNode; node = node->parentInComposedTree()) {
523 if (!node->hasEventListeners(eventNames().dblclickEvent))
524 continue;
525#if ENABLE(TOUCH_EVENTS)
526 if (!node->allowsDoubleTapGesture())
527 continue;
528#endif
529 if (nodeBounds && node->renderer())
530 *nodeBounds = node->renderer()->absoluteBoundingBoxRect(true);
531 return node;
532 }
533 return nullptr;
534 };
535
536 return qualifyingNodeAtViewportLocation(viewportLocation, adjustedViewportLocation, WTFMove(ancestorRespondingToDoubleClickEvent), ShouldApproximate::Yes);
537}

544Node* Frame::nodeRespondingToScrollWheelEvents(const FloatPoint& viewportLocation)
545{
546 auto&& ancestorRespondingToScrollWheelEvents = [](const HitTestResult& hitTestResult, Node* terminationNode, IntRect* nodeBounds) -> Node* {
547 if (nodeBounds)
548 *nodeBounds = IntRect();
549
550 Node* scrollingAncestor = nullptr;
551 for (Node* node = hitTestResult.innerNode(); node && node != terminationNode && !node->hasTagName(HTMLNames::bodyTag); node = node->parentNode()) {
552 RenderObject* renderer = node->renderer();
553 if (!renderer)
554 continue;
555
556 if ((renderer->isTextField() || renderer->isTextArea()) && downcast<RenderTextControl>(*renderer).canScroll()) {
557 scrollingAncestor = node;
558 continue;
559 }
560
561 auto& style = renderer->style();
562
563 if (renderer->hasOverflowClip()
564 && (style.overflowY() == Overflow::Auto || style.overflowY() == Overflow::Scroll
565 || style.overflowX() == Overflow::Auto || style.overflowX() == Overflow::Scroll)) {
566 scrollingAncestor = node;
567 }
568 }
569
570 return scrollingAncestor;
571 };
572
573 FloatPoint adjustedViewportLocation;
574 return qualifyingNodeAtViewportLocation(viewportLocation, adjustedViewportLocation, WTFMove(ancestorRespondingToScrollWheelEvents), ShouldApproximate::No);
575}

577int Frame::preferredHeight() const
578{
579 Document* document = this->document();
580 if (!document)
581 return 0;
582
583 document->updateLayout();
584
585 auto* body = document->bodyOrFrameset();
586 if (!body)
587 return 0;
588
589 RenderObject* renderer = body->renderer();
590 if (!is<RenderBlock>(renderer))
591 return 0;
592
593 RenderBlock& block = downcast<RenderBlock>(*renderer);
594 return block.height() + block.marginTop() + block.marginBottom();
595}

720// FIXME: Break this function up into pieces with descriptive function names so that it's easier to follow.
721NSArray *Frame::interpretationsForCurrentRoot() const
722{
723 if (!document())
724 return nil;
725
726 auto* root = selection().isNone() ? document()->bodyOrFrameset() : selection().selection().rootEditableElement();
727 auto rangeOfRootContents = makeRangeSelectingNodeContents(*root);
728
729 auto markersInRoot = document()->markers().markersInRange(rangeOfRootContents, DocumentMarker::DictationPhraseWithAlternatives);
730
731 // There are no phrases with alternatives, so there is just one interpretation.
732 if (markersInRoot.isEmpty())
733 return @[plainText(rangeOfRootContents)];
734
735 // The number of interpretations will be i1 * i2 * ... * iN, where iX is the number of interpretations for the Xth phrase with alternatives.
736 size_t interpretationsCount = 1;
737
738 for (auto* marker : markersInRoot)
739 interpretationsCount *= WTF::get<Vector<String>>(marker->data()).size() + 1;
740
741 Vector<Vector<UChar>> interpretations;
742 interpretations.grow(interpretationsCount);
743
744 Position precedingTextStartPosition = makeDeprecatedLegacyPosition(root, 0);
745
746 unsigned combinationsSoFar = 1;
747
748 for (auto& node : intersectingNodes(rangeOfRootContents)) {
749 for (auto* marker : document()->markers().markersFor(node, DocumentMarker::DictationPhraseWithAlternatives)) {
750 auto& alternatives = WTF::get<Vector<String>>(marker->data());
751
752 auto rangeForMarker = makeSimpleRange(node, *marker);
753
754 if (auto precedingTextRange = makeSimpleRange(precedingTextStartPosition, rangeForMarker.start)) {
755 String precedingText = plainText(*precedingTextRange);
756 if (!precedingText.isEmpty()) {
757 for (auto& interpretation : interpretations)
758 append(interpretation, precedingText);
759 }
760 }
761
762 String visibleTextForMarker = plainText(rangeForMarker);
763 size_t interpretationsCountForCurrentMarker = alternatives.size() + 1;
764 for (size_t i = 0; i < interpretationsCount; ++i) {
765 size_t indexOfInterpretationForCurrentMarker = (i / combinationsSoFar) % interpretationsCountForCurrentMarker;
766 if (!indexOfInterpretationForCurrentMarker)
767 append(interpretations[i], visibleTextForMarker);
768 else
769 append(interpretations[i], alternatives[i % alternatives.size()]);
770 }
771
772 combinationsSoFar *= interpretationsCountForCurrentMarker;
773
774 precedingTextStartPosition = makeDeprecatedLegacyPosition(rangeForMarker.end);
775 }
776 }
777
778 // Finally, add any text after the last marker.
779 if (auto range = makeSimpleRange(precedingTextStartPosition, rangeOfRootContents.end)) {
780 String textAfterLastMarker = plainText(*range);
781 if (!textAfterLastMarker.isEmpty()) {
782 for (auto& interpretation : interpretations)
783 append(interpretation, textAfterLastMarker);
784 }
785 }
786
787 return createNSArray(interpretations, [] (auto& interpretation) {
788 return adoptNS([[NSString alloc] initWithCharacters:reinterpret_cast<const unichar*>(interpretation.data()) length:interpretation.size()]);
789 }).autorelease();
790}

815void Frame::overflowScrollPositionChangedForNode(const IntPoint& position, Node* node, bool isUserScroll)
816{
817 LOG_WITH_STREAM(Scrolling, stream << "Frame::overflowScrollPositionChangedForNode " << node << " position " << position);
818
819 RenderObject* renderer = node->renderer();
820 if (!renderer || !renderer->hasLayer())
821 return;
822
823 auto* layer = downcast<RenderBoxModelObject>(*renderer).layer();
824 if (!layer)
825 return;
826 auto* scrollableArea = layer->ensureLayerScrollableArea();
827
828 auto oldScrollType = scrollableArea->currentScrollType();
829 scrollableArea->setCurrentScrollType(isUserScroll ? ScrollType::User : ScrollType::Programmatic);
830 scrollableArea->scrollToOffsetWithoutAnimation(position);
831 scrollableArea->setCurrentScrollType(oldScrollType);
832
833 scrollableArea->didEndScroll(); // FIXME: Should we always call this?
834}
