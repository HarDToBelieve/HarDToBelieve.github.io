144bool EventHandler::wheelEvent(NSEvent *event)
145{
146 Page* page = m_frame.page();
147 if (!page)
148 return false;
149
150 CurrentEventScope scope(event, nil);
151 auto wheelEvent = PlatformEventFactory::createPlatformWheelEvent(event, page->chrome().platformPageClient());
152 OptionSet<WheelEventProcessingSteps> processingSteps = { WheelEventProcessingSteps::MainThreadForScrolling, WheelEventProcessingSteps::MainThreadForBlockingDOMEventDispatch };
153
154 if (wheelEvent.phase() == PlatformWheelEventPhase::Changed || wheelEvent.momentumPhase() == PlatformWheelEventPhase::Changed) {
155 if (m_frame.settings().wheelEventGesturesBecomeNonBlocking() && m_wheelScrollGestureState.value_or(WheelScrollGestureState::Blocking) == WheelScrollGestureState::NonBlocking)
156 processingSteps = { WheelEventProcessingSteps::MainThreadForScrolling, WheelEventProcessingSteps::MainThreadForNonBlockingDOMEventDispatch };
157 }
158 return handleWheelEvent(wheelEvent, processingSteps);
159}

176{

179 return;
180
181 if (auto frameView = makeRefPtr(m_frame.view())) {
182 if (NSView *documentView = frameView->documentView()) {

194bool EventHandler::passWidgetMouseDownEventToWidget(const MouseEventWithHitTestResults& event)
195{
196 // Figure out which view to send the event to.
197 auto* target = event.targetNode() ? event.targetNode()->renderer() : nullptr;
198 if (!is<RenderWidget>(target))
199 return false;
200
201 // Double-click events don't exist in Cocoa. Since passWidgetMouseDownEventToWidget() will
202 // just pass currentEvent down to the widget, we don't want to call it for events that
203 // don't correspond to Cocoa events. The mousedown/ups will have already been passed on as
204 // part of the pressed/released handling.
205 return passMouseDownEventToWidget(downcast<RenderWidget>(*target).widget());
206}

208bool EventHandler::passWidgetMouseDownEventToWidget(RenderWidget* renderWidget)
209{
210 return passMouseDownEventToWidget(renderWidget->widget());
211}

213static bool lastEventIsMouseUp()
214{
215 // Many AppKit widgets run their own event loops and consume events while the mouse is down.
216 // When they finish, currentEvent is the mouseUp that they exited on. We need to update
217 // the WebCore state with this mouseUp, which we never saw. This method lets us detect
218 // that state. Handling this was critical when we used AppKit widgets for form elements.
219 // It's not clear in what cases this is helpful now -- it's possible it can be removed.
220
221 ASSERT([NSApp isRunning]);
222 ASSERT(hasProcessPrivilege(ProcessPrivilege::CanCommunicateWithWindowServer));
223
224 BEGIN_BLOCK_OBJC_EXCEPTIONS
225 NSEvent *currentEventAfterHandlingMouseDown = [NSApp currentEvent];
226 return EventHandler::currentNSEvent() != currentEventAfterHandlingMouseDown
227 && [currentEventAfterHandlingMouseDown type] == NSEventTypeLeftMouseUp
228 && [currentEventAfterHandlingMouseDown timestamp] >= [EventHandler::currentNSEvent() timestamp];
229 END_BLOCK_OBJC_EXCEPTIONS
230
231 return false;
232}

234bool EventHandler::passMouseDownEventToWidget(Widget* pWidget)
235{
236 // FIXME: This function always returns true. It should be changed either to return
237 // false in some cases or the return value should be removed.
238
239 RefPtr<Widget> widget = pWidget;
240
241 if (!widget) {
242 LOG_ERROR("hit a RenderWidget without a corresponding Widget, means a frame is half-constructed");
243 return true;
244 }
245
246 // In WebKit2 we will never have a native widget. Just return early and let the regular event handler machinery take care of
247 // dispatching the event.
248 if (!widget->platformWidget())
249 return false;
250
251 BEGIN_BLOCK_OBJC_EXCEPTIONS
252
253 NSView *nodeView = widget->platformWidget();
254 ASSERT([nodeView superview]);
255 NSView *view = [nodeView hitTest:[[nodeView superview] convertPoint:[currentNSEvent() locationInWindow] fromView:nil]];
256 if (!view) {
257 // We probably hit the border of a RenderWidget
258 return true;
259 }
260
261 Page* page = m_frame.page();
262 if (!page)
263 return true;
264
265 if (page->chrome().client().firstResponder() != view) {
266 // Normally [NSWindow sendEvent:] handles setting the first responder.
267 // But in our case, the event was sent to the view representing the entire web page.
268 if ([currentNSEvent() clickCount] <= 1 && [view acceptsFirstResponder] && [view needsPanelToBecomeKey])
269 page->chrome().client().makeFirstResponder(view);
270 }
271
272 // We need to "defer loading" while tracking the mouse, because tearing down the
273 // page while an AppKit control is tracking the mouse can cause a crash.
274
275 // FIXME: In theory, WebCore now tolerates tear-down while tracking the
276 // mouse. We should confirm that, and then remove the deferrsLoading
277 // hack entirely.
278
279 bool wasDeferringLoading = page->defersLoading();
280 if (!wasDeferringLoading)
281 page->setDefersLoading(true);
282
283 ASSERT(!m_sendingEventToSubview);
284 m_sendingEventToSubview = true;
285
286 {
287 WidgetHierarchyUpdatesSuspensionScope suspendWidgetHierarchyUpdates;
288 [view mouseDown:currentNSEvent()];
289 }
290
291 m_sendingEventToSubview = false;
292
293 if (!wasDeferringLoading)
294 page->setDefersLoading(false);
295
296 // Remember which view we sent the event to, so we can direct the release event properly.
297 m_mouseDownView = view;
298 m_mouseDownWasInSubframe = false;
299
300 // Many AppKit widgets run their own event loops and consume events while the mouse is down.
301 // When they finish, currentEvent is the mouseUp that they exited on. We need to update
302 // the EventHandler state with this mouseUp, which we never saw.
303 // If this event isn't a mouseUp, we assume that the mouseUp will be coming later. There
304 // is a hole here if the widget consumes both the mouseUp and subsequent events.
305 if (lastEventIsMouseUp())
306 m_mousePressed = false;
307
308 END_BLOCK_OBJC_EXCEPTIONS
309
310 return true;
311}

313// Note that this does the same kind of check as [target isDescendantOf:superview].
314// There are two differences: This is a lot slower because it has to walk the whole
315// tree, and this works in cases where the target has already been deallocated.
316static bool findViewInSubviews(NSView *superview, NSView *target)
317{
318 BEGIN_BLOCK_OBJC_EXCEPTIONS
319 NSEnumerator *e = [[superview subviews] objectEnumerator];
320 NSView *subview;
321 while ((subview = [e nextObject])) {
322 if (subview == target || findViewInSubviews(subview, target)) {
323 return true;
324 }
325 }
326 END_BLOCK_OBJC_EXCEPTIONS
327
328 return false;
329}

331NSView *EventHandler::mouseDownViewIfStillGood()
332{
333 // Since we have no way of tracking the lifetime of m_mouseDownView, we have to assume that
334 // it could be deallocated already. We search for it in our subview tree; if we don't find
335 // it, we set it to nil.
336 NSView *mouseDownView = m_mouseDownView;
337 if (!mouseDownView) {
338 return nil;
339 }
340 FrameView* topFrameView = m_frame.view();
341 NSView *topView = topFrameView ? topFrameView->platformWidget() : nil;
342 if (!topView || !findViewInSubviews(topView, mouseDownView)) {
343 m_mouseDownView = nil;
344 return nil;
345 }
346 return mouseDownView;
347}

388bool EventHandler::passSubframeEventToSubframe(MouseEventWithHitTestResults& event, Frame& subframe, HitTestResult* hitTestResult)
389{
390 BEGIN_BLOCK_OBJC_EXCEPTIONS
391
392 switch ([currentNSEvent() type]) {
393 case NSEventTypeLeftMouseDragged:
394 case NSEventTypeOtherMouseDragged:
395 case NSEventTypeRightMouseDragged:
396 // This check is bogus and results in <rdar://6813830>, but removing it breaks a number of
397 // layout tests.
398 if (!m_mouseDownWasInSubframe)
399 return false;
400#if ENABLE(DRAG_SUPPORT)
401 if (subframe.page()->dragController().didInitiateDrag())
402 return false;
403#endif
404 case NSEventTypeMouseMoved:
405 // Since we're passing in currentNSEvent() here, we can call
406 // handleMouseMoveEvent() directly, since the save/restore of
407 // currentNSEvent() that mouseMoved() does would have no effect.
408 ASSERT(!m_sendingEventToSubview);
409 m_sendingEventToSubview = true;
410 subframe.eventHandler().handleMouseMoveEvent(currentPlatformMouseEvent(), hitTestResult);
411 m_sendingEventToSubview = false;
412 return true;
413
414 case NSEventTypeLeftMouseDown: {
415 Node* node = event.targetNode();
416 if (!node)
417 return false;
418 auto* renderer = node->renderer();
419 if (!is<RenderWidget>(renderer))
420 return false;
421 Widget* widget = downcast<RenderWidget>(*renderer).widget();
422 if (!widget || !widget->isFrameView())
423 return false;
424 if (!passWidgetMouseDownEventToWidget(downcast<RenderWidget>(renderer)))
425 return false;
426 m_mouseDownWasInSubframe = true;
427 return true;
428 }
429 case NSEventTypeLeftMouseUp: {
430 if (!m_mouseDownWasInSubframe)
431 return false;
432 ASSERT(!m_sendingEventToSubview);
433 m_sendingEventToSubview = true;
434 subframe.eventHandler().handleMouseReleaseEvent(currentPlatformMouseEvent());
435 m_sendingEventToSubview = false;
436 return true;
437 }
438 default:
439 return false;
440 }
441 END_BLOCK_OBJC_EXCEPTIONS
442
443 return false;
444}

478bool EventHandler::passWheelEventToWidget(const PlatformWheelEvent& wheelEvent, Widget& widget, OptionSet<WheelEventProcessingSteps> processingSteps)
479{
480 BEGIN_BLOCK_OBJC_EXCEPTIONS
481
482 NSView* nodeView = widget.platformWidget();
483 if (!nodeView) {
484 // WebKit2 code path.
485 if (!is<FrameView>(widget))
486 return false;
487
488 return downcast<FrameView>(widget).frame().eventHandler().handleWheelEvent(wheelEvent, processingSteps);
489 }
490
491 if ([currentNSEvent() type] != NSEventTypeScrollWheel || m_sendingEventToSubview)
492 return false;
493
494 ASSERT(nodeView);
495 ASSERT([nodeView superview]);
496 NSView *view = [nodeView hitTest:[[nodeView superview] convertPoint:[currentNSEvent() locationInWindow] fromView:nil]];
497 if (!view) {
498 // We probably hit the border of a RenderWidget
499 return false;
500 }
501
502 ASSERT(!m_sendingEventToSubview);
503 m_sendingEventToSubview = true;
504 // Work around <rdar://problem/6806810> which can cause -[NSScrollView scrollWheel:] to
505 // crash if the NSScrollView is released during timer or network callback dispatch
506 // in the nested tracking runloop that -[NSScrollView scrollWheel:] runs.
507 setNSScrollViewScrollWheelShouldRetainSelf(true);
508 [view scrollWheel:currentNSEvent()];
509 setNSScrollViewScrollWheelShouldRetainSelf(false);
510 m_sendingEventToSubview = false;
511 return true;
512
513 END_BLOCK_OBJC_EXCEPTIONS
514 return false;
515}

548void EventHandler::mouseUp(NSEvent *event, NSEvent *correspondingPressureEvent)
549{
550 FrameView* v = m_frame.view();
551 if (!v || m_sendingEventToSubview)
552 return;
553
554 BEGIN_BLOCK_OBJC_EXCEPTIONS
555
556 CurrentEventScope scope(event, correspondingPressureEvent);
557
558 // Our behavior here is a little different that Qt. Qt always sends
559 // a mouse release event, even for a double click. To correct problems
560 // in khtml's DOM click event handling we do not send a release here
561 // for a double click. Instead we send that event from FrameView's
562 // handleMouseDoubleClickEvent. Note also that the third click of
563 // a triple click is treated as a single click, but the fourth is then
564 // treated as another double click. Hence the "% 2" below.
565 int clickCount = [event clickCount];
566 if (clickCount > 0 && clickCount % 2 == 0)
567 handleMouseDoubleClickEvent(currentPlatformMouseEvent());
568 else
569 handleMouseReleaseEvent(currentPlatformMouseEvent());
570
571 m_mouseDownView = nil;
572
573 END_BLOCK_OBJC_EXCEPTIONS
574}

576/*
577 A hack for the benefit of AK's PopUpButton, which uses the Carbon menu manager, which thus
578 eats all subsequent events after it is starts its modal tracking loop. After the interaction
579 is done, this routine is used to fix things up. When a mouse down started us tracking in
580 the widget, we post a fake mouse up to balance the mouse down we started with. When a
581 key down started us tracking in the widget, we post a fake key up to balance things out.
582 In addition, we post a fake mouseMoved to get the cursor in sync with whatever we happen to
583 be over after the tracking is done.
584 */
585void EventHandler::sendFakeEventsAfterWidgetTracking(NSEvent *initiatingEvent)
586{
587 FrameView* view = m_frame.view();
588 if (!view)
589 return;
590
591 ASSERT(hasProcessPrivilege(ProcessPrivilege::CanCommunicateWithWindowServer));
592
593 BEGIN_BLOCK_OBJC_EXCEPTIONS
594
595 m_sendingEventToSubview = false;
596 int eventType = [initiatingEvent type];
597 if (eventType == NSEventTypeLeftMouseDown || eventType == NSEventTypeKeyDown) {
598 ASSERT([NSApp isRunning]);
599 NSEvent *fakeEvent = nil;
600 if (eventType == NSEventTypeLeftMouseDown) {
601 fakeEvent = [NSEvent mouseEventWithType:NSEventTypeLeftMouseUp
602 location:[initiatingEvent locationInWindow]
603 modifierFlags:[initiatingEvent modifierFlags]
604 timestamp:[initiatingEvent timestamp]
605 windowNumber:[initiatingEvent windowNumber]
606 context:nullptr
607 eventNumber:[initiatingEvent eventNumber]
608 clickCount:[initiatingEvent clickCount]
609 pressure:[initiatingEvent pressure]];
610
611 [NSApp postEvent:fakeEvent atStart:YES];
612 } else { // eventType == NSEventTypeKeyDown
613 fakeEvent = [NSEvent keyEventWithType:NSEventTypeKeyUp
614 location:[initiatingEvent locationInWindow]
615 modifierFlags:[initiatingEvent modifierFlags]
616 timestamp:[initiatingEvent timestamp]
617 windowNumber:[initiatingEvent windowNumber]
618 context:nullptr
619 characters:[initiatingEvent characters]
620 charactersIgnoringModifiers:[initiatingEvent charactersIgnoringModifiers]
621 isARepeat:[initiatingEvent isARepeat]
622 keyCode:[initiatingEvent keyCode]];
623 [NSApp postEvent:fakeEvent atStart:YES];
624 }
625
626 // FIXME: We should really get the current modifierFlags here, but there's no way to poll
627 // them in Cocoa, and because the event stream was stolen by the Carbon menu code we have
628 // no up-to-date cache of them anywhere.
629 fakeEvent = [NSEvent mouseEventWithType:NSEventTypeMouseMoved
630 location:[[view->platformWidget() window]
631 ALLOW_DEPRECATED_DECLARATIONS_BEGIN
632 convertScreenToBase:[NSEvent mouseLocation]]
633 ALLOW_DEPRECATED_DECLARATIONS_END
634 modifierFlags:[initiatingEvent modifierFlags]
635 timestamp:[initiatingEvent timestamp]
636 windowNumber:[initiatingEvent windowNumber]
637 context:nullptr
638 eventNumber:0
639 clickCount:0
640 pressure:0];
641 [NSApp postEvent:fakeEvent atStart:YES];
642 }
643
644 END_BLOCK_OBJC_EXCEPTIONS
645}

694bool EventHandler::passMousePressEventToSubframe(MouseEventWithHitTestResults& mouseEventAndResult, Frame& subframe)
695{
696 // WebKit1 code path.
697 if (frameHasPlatformWidget(m_frame))
698 return passSubframeEventToSubframe(mouseEventAndResult, subframe);
699

799// FIXME: This could be written in terms of ScrollableArea::enclosingScrollableArea().
800static ContainerNode* findEnclosingScrollableContainer(ContainerNode* node, const PlatformWheelEvent& wheelEvent)
801{
802 auto biasedDelta = ScrollController::wheelDeltaBiasingTowardsVertical(wheelEvent);
803
804 // Find the first node with a valid scrollable area starting with the current
805 // node and traversing its parents (or shadow hosts).
806 for (ContainerNode* candidate = node; candidate; candidate = candidate->parentOrShadowHostNode()) {
807 if (is<HTMLIFrameElement>(*candidate))
808 continue;
809
810 if (is<HTMLHtmlElement>(*candidate) || is<HTMLDocument>(*candidate))
811 return nullptr;
812
813 RenderBox* box = candidate->renderBox();
814 if (!box || !box->canBeScrolledAndHasScrollableArea())
815 continue;
816
817 auto* scrollableArea = scrollableAreaForBox(*box);
818 if (!scrollableArea)
819 continue;
820
821 if (wheelEvent.phase() == PlatformWheelEventPhase::MayBegin || wheelEvent.phase() == PlatformWheelEventPhase::Cancelled)
822 return candidate;
823
824 if (biasedDelta.height() && !scrollableArea->isPinnedForScrollDeltaOnAxis(-biasedDelta.height(), ScrollEventAxis::Vertical))
825 return candidate;
826
827 if (biasedDelta.width() && !scrollableArea->isPinnedForScrollDeltaOnAxis(-biasedDelta.width(), ScrollEventAxis::Horizontal))
828 return candidate;
829 }
830
831 return nullptr;
832}

865void EventHandler::determineWheelEventTarget(const PlatformWheelEvent& wheelEvent, RefPtr<Element>& wheelEventTarget, WeakPtr<ScrollableArea>& scrollableArea, bool& isOverWidget)
866{
867 auto* page = m_frame.page();
868 if (!page)
869 return;
870
871 auto* view = m_frame.view();
872 if (!view)
873 return;
874
875 if (eventTargetIsPlatformWidget(wheelEventTarget.get()))
876 scrollableArea = scrollableAreaForEventTarget(wheelEventTarget.get());
877 else {
878 auto* scrollableContainer = findEnclosingScrollableContainer(wheelEventTarget.get(), wheelEvent);
879 if (scrollableContainer)
880 scrollableArea = scrollableAreaForContainerNode(*scrollableContainer);
881 else
882 scrollableArea = makeWeakPtr(static_cast<ScrollableArea&>(*view));
883 }
884
885 LOG_WITH_STREAM(ScrollLatching, stream << "EventHandler::determineWheelEventTarget() - event " << wheelEvent << " found scrollableArea " << ValueOrNull(scrollableArea.get()) << ", latching state is " << page->scrollLatchingController());
886
887 if (scrollableArea && page->isMonitoringWheelEvents())
888 scrollableArea->scrollAnimator().setWheelEventTestMonitor(page->wheelEventTestMonitor());
889
890 if (wheelEvent.shouldResetLatching() || wheelEvent.isNonGestureEvent())
891 return;
892
893 if (m_frame.isMainFrame() && wheelEvent.isGestureStart())
894 page->wheelEventDeltaFilter()->beginFilteringDeltas();
895
896 page->scrollLatchingController().updateAndFetchLatchingStateForFrame(m_frame, wheelEvent, wheelEventTarget, scrollableArea, isOverWidget);
897}

918bool EventHandler::processWheelEventForScrolling(const PlatformWheelEvent& wheelEvent, const WeakPtr<ScrollableArea>& scrollableArea, OptionSet<EventHandling> eventHandling)
919{
920 LOG_WITH_STREAM(ScrollLatching, stream << "EventHandler::processWheelEventForScrolling " << wheelEvent << " - scrollableArea " << ValueOrNull(scrollableArea.get()) << " use latched element " << wheelEvent.useLatchedEventElement());
921
922#if ASSERT_ENABLED
923 {
924 // FIXME: Clean up processWheelEventForScrollSnap() and then turn this into an early return.
925 WeakPtr<ScrollableArea> latchedScrollableArea;
926 ASSERT(m_frame.page()->scrollLatchingController().latchingAllowsScrollingInFrame(m_frame, latchedScrollableArea));
927 }
928#endif
929
930 Ref<Frame> protectedFrame(m_frame);
931
932 if (!m_frame.page())
933 return false;
934
935 FrameView* view = m_frame.view();
936 // We do another check on the frame view because the event handler can run JS which results in the frame getting destroyed.
937 if (!view)
938 return false;
939
940 // We handle non-view scrollableAreas elsewhere.
941 if (wheelEvent.useLatchedEventElement() && scrollableArea) {
942 m_isHandlingWheelEvent = false;
943
944 LOG_WITH_STREAM(ScrollLatching, stream << " latching state " << m_frame.page()->scrollLatchingController());
945
946 if (!frameHasPlatformWidget(m_frame) && scrollableArea != view) {
947 LOG_WITH_STREAM(Scrolling, stream << " latched to non-view scroller " << scrollableArea << " and not propagating");
948 return true;
949 }
950
951 LOG_WITH_STREAM(ScrollLatching, stream << " sending to view " << *view);
952
953 bool didHandleWheelEvent = handleWheelEventInScrollableArea(wheelEvent, *view, eventHandling);
954 // If the platform widget is handling the event, we always want to return false.
955 if (view->platformWidget())
956 didHandleWheelEvent = false;
957
958 LOG_WITH_STREAM(ScrollLatching, stream << " EventHandler::processWheelEventForScrolling returning " << didHandleWheelEvent);
959 return didHandleWheelEvent;
960 }
961
962 bool didHandleEvent = handleWheelEventInScrollableArea(wheelEvent, *view, eventHandling);
963 m_isHandlingWheelEvent = false;
964 return didHandleEvent;
965}

1030static IntSize autoscrollAdjustmentFactorForScreenBoundaries(const IntPoint& screenPoint, const FloatRect& screenRect)
1031{
1032 // If the window is at the edge of the screen, and the mouse position is also at that edge of the screen,
1033 // we need to adjust the autoscroll amount in order for the user to be able to autoscroll in that direction.
1034 // We can pretend that the mouse position is slightly beyond the edge of the screen, and then autoscrolling
1035 // will occur as excpected. This function figures out just how much to adjust the autoscroll amount by
1036 // in order to get autoscrolling to feel natural in this situation.
1037
1038 constexpr float edgeDistanceThreshold = 50;
1039 constexpr float pixelsMultiplier = 20;
1040
1041 IntSize adjustmentFactor;
1042
1043 float screenLeftEdge = screenRect.x();
1044 float insetScreenLeftEdge = screenLeftEdge + edgeDistanceThreshold;
1045 float screenRightEdge = screenRect.maxX();
1046 float insetScreenRightEdge = screenRightEdge - edgeDistanceThreshold;
1047 if (screenPoint.x() >= screenLeftEdge && screenPoint.x() < insetScreenLeftEdge) {
1048 float distanceFromEdge = screenPoint.x() - screenLeftEdge - edgeDistanceThreshold;
1049 if (distanceFromEdge < 0)
1050 adjustmentFactor.setWidth((distanceFromEdge / edgeDistanceThreshold) * pixelsMultiplier);
1051 } else if (screenPoint.x() >= insetScreenRightEdge && screenPoint.x() < screenRightEdge) {
1052 float distanceFromEdge = edgeDistanceThreshold - (screenRightEdge - screenPoint.x());
1053 if (distanceFromEdge > 0)
1054 adjustmentFactor.setWidth((distanceFromEdge / edgeDistanceThreshold) * pixelsMultiplier);
1055 }
1056
1057 float screenTopEdge = screenRect.y();
1058 float insetScreenTopEdge = screenTopEdge + edgeDistanceThreshold;
1059 float screenBottomEdge = screenRect.maxY();
1060 float insetScreenBottomEdge = screenBottomEdge - edgeDistanceThreshold;
1061
1062 if (screenPoint.y() >= screenTopEdge && screenPoint.y() < insetScreenTopEdge) {
1063 float distanceFromEdge = screenPoint.y() - screenTopEdge - edgeDistanceThreshold;
1064 if (distanceFromEdge < 0)
1065 adjustmentFactor.setHeight((distanceFromEdge / edgeDistanceThreshold) * pixelsMultiplier);
1066 } else if (screenPoint.y() >= insetScreenBottomEdge && screenPoint.y() < screenBottomEdge) {
1067 float distanceFromEdge = edgeDistanceThreshold - (screenBottomEdge - screenPoint.y());
1068 if (distanceFromEdge > 0)
1069 adjustmentFactor.setHeight((distanceFromEdge / edgeDistanceThreshold) * pixelsMultiplier);
1070 }
1071
1072 return adjustmentFactor;
1073}

1075IntPoint EventHandler::targetPositionInWindowForSelectionAutoscroll() const
1076{
1077 Page* page = m_frame.page();
1078 if (!page)
1079 return m_lastKnownMousePosition.value_or(IntPoint());
