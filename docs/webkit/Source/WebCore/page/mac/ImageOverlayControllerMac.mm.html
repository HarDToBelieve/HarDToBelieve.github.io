50void ImageOverlayController::updateDataDetectorHighlights(const HTMLElement& overlayHost)
51{
52 if (!overlayHost.hasImageOverlay()) {
53 ASSERT_NOT_REACHED();
54 clearDataDetectorHighlights();
55 return;
56 }
57
58 Vector<Ref<HTMLElement>> dataDetectorResultElements;
59 for (auto& child : descendantsOfType<HTMLElement>(*overlayHost.userAgentShadowRoot())) {
60 if (child.isImageOverlayDataDetectorResult() && child.renderer())
61 dataDetectorResultElements.append(makeRef(child));
62 }
63
64 HashSet<Ref<HTMLElement>> dataDetectorResultElementsWithHighlights;
65 for (auto& containerAndHighlight : m_dataDetectorContainersAndHighlights) {
66 if (containerAndHighlight.first)
67 dataDetectorResultElementsWithHighlights.add(*containerAndHighlight.first);
68 }
69
70 if (dataDetectorResultElementsWithHighlights == dataDetectorResultElements)
71 return;
72
73 auto mainFrameView = makeRefPtr(m_page->mainFrame().view());
74 if (!mainFrameView)
75 return;
76
77 auto frameView = makeRefPtr(overlayHost.document().view());
78 if (!frameView)
79 return;
80
81 m_activeDataDetectorHighlight = nullptr;
82 m_dataDetectorContainersAndHighlights.clear();
83 m_dataDetectorContainersAndHighlights.reserveInitialCapacity(dataDetectorResultElements.size());
84
85 for (auto& element : dataDetectorResultElements) {
86 CGRect elementBounds = element->renderer()->absoluteBoundingBoxRect();
87 elementBounds.origin = mainFrameView->windowToContents(frameView->contentsToWindow(roundedIntPoint(elementBounds.origin)));
88
89 // FIXME: We should teach DataDetectorHighlight to render quads instead of always falling back to axis-aligned bounding rects.
90#if HAVE(DD_HIGHLIGHT_CREATE_WITH_SCALE)
91 auto highlight = adoptCF(PAL::softLink_DataDetectors_DDHighlightCreateWithRectsInVisibleRectWithStyleScaleAndDirection(nullptr, &elementBounds, 1, mainFrameView->visibleContentRect(), DDHighlightStyleBubbleStandard | DDHighlightStyleStandardIconArrow, YES, NSWritingDirectionNatural, NO, YES, 0));
92#else
93 auto highlight = adoptCF(PAL::softLink_DataDetectors_DDHighlightCreateWithRectsInVisibleRectWithStyleAndDirection(nullptr, &elementBounds, 1, mainFrameView->visibleContentRect(), DDHighlightStyleBubbleStandard | DDHighlightStyleStandardIconArrow, YES, NSWritingDirectionNatural, NO, YES));
94#endif
95 m_dataDetectorContainersAndHighlights.append({ makeWeakPtr(element.get()), DataDetectorHighlight::createForImageOverlay(*m_page, *this, WTFMove(highlight), *makeRangeSelectingNode(element.get())) });
96 }
97}

99bool ImageOverlayController::platformHandleMouseEvent(const PlatformMouseEvent& event)
100{
101 auto mainFrameView = makeRefPtr(m_page->mainFrame().view());
102 if (!mainFrameView)
103 return false;
104
105 RefPtr<HTMLElement> activeDataDetectorElement;
106 auto previousActiveHighlight = std::exchange(m_activeDataDetectorHighlight, nullptr);
107 auto mousePositionInContents = mainFrameView->windowToContents(event.position());
108 bool mouseIsOverActiveDataDetectorHighlightButton = false;
109
110 for (auto& [element, highlight] : m_dataDetectorContainersAndHighlights) {
111 if (!element)
112 continue;
113
114 Boolean isOverButton = NO;
115 if (!PAL::softLink_DataDetectors_DDHighlightPointIsOnHighlight(highlight->highlight(), mousePositionInContents, &isOverButton))
116 continue;
117
118 mouseIsOverActiveDataDetectorHighlightButton = isOverButton;
119 m_activeDataDetectorHighlight = highlight.copyRef();
120 activeDataDetectorElement = makeRefPtr(*element);
121 break;
122 }
123
124 if (previousActiveHighlight != m_activeDataDetectorHighlight) {
125 if (previousActiveHighlight)
126 previousActiveHighlight->fadeOut();
127
128 if (m_activeDataDetectorHighlight) {
129 m_overlay->layer().addChild(m_activeDataDetectorHighlight->layer());
130 m_activeDataDetectorHighlight->fadeIn();
131 }
132 }
133
134 if (event.type() == PlatformEvent::MousePressed && mouseIsOverActiveDataDetectorHighlightButton)
135 return handleDataDetectorAction(*activeDataDetectorElement, mousePositionInContents);
136
137 return false;
138}

140bool ImageOverlayController::handleDataDetectorAction(const HTMLElement& element, const IntPoint& locationInContents)
141{
142 if (!m_page)
143 return false;
144
145 auto frame = makeRefPtr(element.document().frame());
146 if (!frame)
147 return false;
148
149 auto frameView = makeRefPtr(element.document().view());
150 if (!frameView)
151 return false;
152
153 auto identifierValue = parseInteger<uint64_t>(element.attributeWithoutSynchronization(HTMLNames::x_apple_data_detectors_resultAttr));
154 if (!identifierValue)
155 return false;
156
157 auto identifier = makeObjectIdentifier<ImageOverlayDataDetectionResultIdentifierType>(*identifierValue);
158 if (!identifier.isValid())
159 return false;
160
161 auto* dataDetectionResults = frame->dataDetectionResultsIfExists();
162 if (!dataDetectionResults)
163 return false;
164
165 auto dataDetectionResult = retainPtr(dataDetectionResults->imageOverlayDataDetectionResult(identifier));
166 if (!dataDetectionResult)
167 return false;
168
169 auto* renderer = element.renderer();
170 if (!renderer)
171 return false;
172
173 m_page->chrome().client().handleClickForDataDetectionResult({ WTFMove(dataDetectionResult), frameView->contentsToWindow(renderer->absoluteBoundingBoxRect()) }, frameView->contentsToWindow(locationInContents));
174 return true;
175}

177void ImageOverlayController::clearDataDetectorHighlights()
178{
179 m_hostElementForDataDetectors = nullptr;
180 m_dataDetectorContainersAndHighlights.clear();
181 m_activeDataDetectorHighlight = nullptr;
182}

184void ImageOverlayController::elementUnderMouseDidChange(Frame& frame, Element* elementUnderMouse)
185{
186 if (m_activeDataDetectorHighlight)
187 return;
188
189 if (!elementUnderMouse && m_hostElementForDataDetectors && frame.document() != &m_hostElementForDataDetectors->document())
190 return;
191
192 if (!elementUnderMouse || !HTMLElement::isInsideImageOverlay(*elementUnderMouse)) {
193 m_hostElementForDataDetectors = nullptr;
194 uninstallPageOverlayIfNeeded();
195 return;
196 }
197
198 auto shadowHost = elementUnderMouse->shadowHost();
199 if (!is<HTMLElement>(shadowHost)) {
200 ASSERT_NOT_REACHED();
201 m_hostElementForDataDetectors = nullptr;
202 uninstallPageOverlayIfNeeded();
203 return;
204 }
205
206 auto imageOverlayHost = makeRef(downcast<HTMLElement>(*shadowHost));
207 if (!imageOverlayHost->hasImageOverlay()) {
208 ASSERT_NOT_REACHED();
209 m_hostElementForDataDetectors = nullptr;
210 uninstallPageOverlayIfNeeded();
211 return;
212 }
213
214 if (m_hostElementForDataDetectors == imageOverlayHost.ptr())
215 return;
216
217 updateDataDetectorHighlights(imageOverlayHost.get());
218
219 if (m_dataDetectorContainersAndHighlights.isEmpty()) {
220 m_hostElementForDataDetectors = nullptr;
221 uninstallPageOverlayIfNeeded();
222 return;
223 }
224
225 m_hostElementForDataDetectors = makeWeakPtr(imageOverlayHost.get());
226 installPageOverlayIfNeeded();
227}
