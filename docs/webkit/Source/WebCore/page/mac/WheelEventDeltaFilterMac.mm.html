36WheelEventDeltaFilterMac::WheelEventDeltaFilterMac()
37 : WheelEventDeltaFilter()
38 , m_predominantAxisFilter(adoptNS([[_NSScrollingPredominantAxisFilter alloc] init]))
39{
40}
41
42void WheelEventDeltaFilterMac::beginFilteringDeltas()
43{
44 m_beginFilteringDeltasTime = MonotonicTime::now();
45 m_isFilteringDeltas = true;
46}
47
48void WheelEventDeltaFilterMac::updateFromDelta(const FloatSize& delta)
49{
50 if (!m_isFilteringDeltas)
51 return;
52
53 NSPoint filteredDeltaResult;
54 NSPoint filteredVelocityResult;
55 [m_predominantAxisFilter filterInputDelta:NSPoint(FloatPoint(delta.width(), delta.height())) timestamp:(MonotonicTime::now() - m_beginFilteringDeltasTime).seconds() outputDelta:&filteredDeltaResult velocity:&filteredVelocityResult];
56 m_currentFilteredVelocity = FloatSize(filteredVelocityResult.x, filteredVelocityResult.y);
57 m_currentFilteredDelta = FloatSize(filteredDeltaResult.x, filteredDeltaResult.y);
58}
59
60void WheelEventDeltaFilterMac::endFilteringDeltas()
61{
62 m_currentFilteredDelta = FloatSize(0, 0);
63 m_beginFilteringDeltasTime = MonotonicTime();
64 [m_predominantAxisFilter reset];
65 m_isFilteringDeltas = false;
66}
