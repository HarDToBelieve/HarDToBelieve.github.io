135void AsyncScrollingCoordinator::frameViewLayoutUpdated(FrameView& frameView)
136{
137 ASSERT(isMainThread());
138 ASSERT(m_page);
139
140 // If there isn't a root node yet, don't do anything. We'll be called again after creating one.
141 if (!m_scrollingStateTree->rootStateNode())
142 return;
143
144 setEventTrackingRegionsDirty();
145
146#if PLATFORM(COCOA)
147 if (!coordinatesScrollingForFrameView(frameView))
148 return;
149
150 auto* page = frameView.frame().page();
151 if (page && page->isMonitoringWheelEvents()) {
152 auto* node = m_scrollingStateTree->stateNodeForID(frameView.scrollingNodeID());
153 if (!is<ScrollingStateFrameScrollingNode>(node))
154 return;
155
156 auto& frameScrollingNode = downcast<ScrollingStateFrameScrollingNode>(*node);
157 frameScrollingNode.setIsMonitoringWheelEvents(page->isMonitoringWheelEvents());
158 }
159#else
160 UNUSED_PARAM(frameView);
161#endif
162}

172 // If the root layer does not have a ScrollingStateNode, then we should create one.

209void AsyncScrollingCoordinator::frameViewRootLayerDidChange(FrameView& frameView)
210{
211 ASSERT(isMainThread());
212 ASSERT(m_page);
213
214 if (!coordinatesScrollingForFrameView(frameView))
215 return;
216
217 // FIXME: In some navigation scenarios, the FrameView has no RenderView or that RenderView has not been composited.
218 // This needs cleaning up: https://bugs.webkit.org/show_bug.cgi?id=132724
219 if (!frameView.scrollingNodeID())
220 return;
221
222 // If the root layer does not have a ScrollingStateNode, then we should create one.
223 ensureRootStateNodeForFrameView(frameView);
224 ASSERT(m_scrollingStateTree->stateNodeForID(frameView.scrollingNodeID()));
225
226 ScrollingCoordinator::frameViewRootLayerDidChange(frameView);
227
228 auto* node = downcast<ScrollingStateFrameScrollingNode>(m_scrollingStateTree->stateNodeForID(frameView.scrollingNodeID()));
229 node->setScrollContainerLayer(scrollContainerLayerForFrameView(frameView));
230 node->setScrolledContentsLayer(scrolledContentsLayerForFrameView(frameView));
231 node->setRootContentsLayer(rootContentsLayerForFrameView(frameView));
232 node->setCounterScrollingLayer(counterScrollingLayerForFrameView(frameView));
233 node->setInsetClipLayer(insetClipLayerForFrameView(frameView));
234 node->setContentShadowLayer(contentShadowLayerForFrameView(frameView));
235 node->setHeaderLayer(headerLayerForFrameView(frameView));
236 node->setFooterLayer(footerLayerForFrameView(frameView));
237 node->setScrollBehaviorForFixedElements(frameView.scrollBehaviorForFixedElements());
238 node->setVerticalScrollbarLayer(frameView.layerForVerticalScrollbar());
239 node->setHorizontalScrollbarLayer(frameView.layerForHorizontalScrollbar());
240}

242bool AsyncScrollingCoordinator::requestScrollPositionUpdate(ScrollableArea& scrollableArea, const IntPoint& scrollPosition, ScrollType scrollType, ScrollClamping clamping)
243{
244 ASSERT(isMainThread());
245 ASSERT(m_page);
246 auto scrollingNodeID = scrollableArea.scrollingNodeID();
247 if (!scrollingNodeID)
248 return false;
249
250 auto* frameView = frameViewForScrollingNode(scrollingNodeID);
251 if (!frameView)
252 return false;
253
254 if (!coordinatesScrollingForFrameView(*frameView))
255 return false;
256
257 setScrollingNodeScrollableAreaGeometry(scrollingNodeID, scrollableArea);
258
259 bool inBackForwardCache = frameView->frame().document()->backForwardCacheState() != Document::NotInBackForwardCache;
260 bool isSnapshotting = m_page->isTakingSnapshotsForApplicationSuspension();
261 bool inProgrammaticScroll = scrollableArea.currentScrollType() == ScrollType::Programmatic;
262 if (inProgrammaticScroll || inBackForwardCache)
263 applyScrollUpdate(scrollingNodeID, scrollPosition, { }, ScrollType::Programmatic, ScrollingLayerPositionAction::Set);
264
265 ASSERT(inProgrammaticScroll == (scrollType == ScrollType::Programmatic));
266
267 // If this frame view's document is being put into the back/forward cache, we don't want to update our
268 // main frame scroll position. Just let the FrameView think that we did.
269 if (inBackForwardCache || isSnapshotting)
270 return true;
271
272 auto* stateNode = downcast<ScrollingStateScrollingNode>(m_scrollingStateTree->stateNodeForID(scrollingNodeID));
273 if (!stateNode)
274 return false;
275
276 stateNode->setRequestedScrollData({ scrollPosition, scrollType, clamping });
277 commitTreeStateIfNeeded();
278 return true;
279}

286void AsyncScrollingCoordinator::synchronizeStateFromScrollingTree()
287{
288 ASSERT(isMainThread());
289 applyPendingScrollUpdates();
290
291 m_scrollingTree->traverseScrollingTree([&](ScrollingNodeID nodeID, ScrollingNodeType, std::optional<FloatPoint> scrollPosition, std::optional<FloatPoint> layoutViewportOrigin, bool scrolledSinceLastCommit) {
292 if (scrollPosition && scrolledSinceLastCommit) {
293 LOG_WITH_STREAM(Scrolling, stream << "AsyncScrollingCoordinator::synchronizeStateFromScrollingTree - node " << nodeID << " scroll position " << scrollPosition);
294 updateScrollPositionAfterAsyncScroll(nodeID, scrollPosition.value(), layoutViewportOrigin, ScrollType::User, ScrollingLayerPositionAction::Set);
295 }
296 });
297}

299void AsyncScrollingCoordinator::applyPendingScrollUpdates()
300{
301 if (!m_scrollingTree)
302 return;
303
304 auto scrollUpdates = m_scrollingTree->takePendingScrollUpdates();
305 for (auto& update : scrollUpdates) {
306 LOG_WITH_STREAM(Scrolling, stream << "AsyncScrollingCoordinator::applyPendingScrollUpdates - node " << update.nodeID << " scroll position " << update.scrollPosition);
307 updateScrollPositionAfterAsyncScroll(update.nodeID, update.scrollPosition, update.layoutViewportOrigin, ScrollType::User, update.updateLayerPositionAction);
308 }

316FrameView* AsyncScrollingCoordinator::frameViewForScrollingNode(ScrollingNodeID scrollingNodeID) const
317{
318 if (!m_scrollingStateTree->rootStateNode())
319 return nullptr;
320
321 if (scrollingNodeID == m_scrollingStateTree->rootStateNode()->scrollingNodeID())
322 return m_page->mainFrame().view();
323
324 auto* stateNode = m_scrollingStateTree->stateNodeForID(scrollingNodeID);
325 if (!stateNode)
326 return nullptr;
327
328 // Find the enclosing frame scrolling node.
329 auto* parentNode = stateNode;
330 while (parentNode && !parentNode->isFrameScrollingNode())
331 parentNode = parentNode->parent();
332
333 if (!parentNode)
334 return nullptr;
335
336 // Walk the frame tree to find the matching FrameView. This is not ideal, but avoids back pointers to FrameViews
337 // from ScrollingTreeStateNodes.
338 for (Frame* frame = &m_page->mainFrame(); frame; frame = frame->tree().traverseNext()) {
339 if (auto* view = frame->view()) {
340 if (view->scrollingNodeID() == parentNode->scrollingNodeID())
341 return view;
342 }
343 }
344
345 return nullptr;
346}

354void AsyncScrollingCoordinator::updateScrollPositionAfterAsyncScroll(ScrollingNodeID scrollingNodeID, const FloatPoint& scrollPosition, std::optional<FloatPoint> layoutViewportOrigin, ScrollType scrollType, ScrollingLayerPositionAction scrollingLayerPositionAction)
355{
356 ASSERT(isMainThread());
357
358 if (!m_page)
359 return;
360
361 auto* frameViewPtr = frameViewForScrollingNode(scrollingNodeID);
362 if (!frameViewPtr)
363 return;
364
365 LOG_WITH_STREAM(Scrolling, stream << "AsyncScrollingCoordinator::updateScrollPositionAfterAsyncScroll node " << scrollingNodeID << " " << scrollType << " scrollPosition " << scrollPosition << " action " << scrollingLayerPositionAction);
366
367 auto& frameView = *frameViewPtr;
368
369 if (!frameViewPtr->frame().isMainFrame()) {
370 if (scrollingLayerPositionAction == ScrollingLayerPositionAction::Set)
371 m_page->editorClient().subFrameScrollPositionChanged();
372 }
373
374 if (scrollingNodeID == frameView.scrollingNodeID()) {
375 reconcileScrollingState(frameView, scrollPosition, layoutViewportOrigin, scrollType, ViewportRectStability::Stable, scrollingLayerPositionAction);
376 return;
377 }
378
379 // Overflow-scroll area.
380 if (auto* scrollableArea = frameView.scrollableAreaForScrollingNodeID(scrollingNodeID)) {
381 auto previousScrollType = scrollableArea->currentScrollType();
382 scrollableArea->setCurrentScrollType(scrollType);
383 scrollableArea->notifyScrollPositionChanged(roundedIntPoint(scrollPosition));
384 scrollableArea->setCurrentScrollType(previousScrollType);
385
386 if (scrollingLayerPositionAction == ScrollingLayerPositionAction::Set)
387 m_page->editorClient().overflowScrollPositionChanged();
388 }
389}

391void AsyncScrollingCoordinator::reconcileScrollingState(FrameView& frameView, const FloatPoint& scrollPosition, const LayoutViewportOriginOrOverrideRect& layoutViewportOriginOrOverrideRect, ScrollType scrollType, ViewportRectStability viewportRectStability, ScrollingLayerPositionAction scrollingLayerPositionAction)
392{
393 auto previousScrollType = frameView.currentScrollType();
394 frameView.setCurrentScrollType(scrollType);
395
396 LOG_WITH_STREAM(Scrolling, stream << getCurrentProcessID() << " AsyncScrollingCoordinator " << this << " reconcileScrollingState scrollPosition " << scrollPosition << " type " << scrollType << " stability " << viewportRectStability << " " << scrollingLayerPositionAction);
397
398 std::optional<FloatRect> layoutViewportRect;
399
400 WTF::switchOn(layoutViewportOriginOrOverrideRect,
401 [&frameView](std::optional<FloatPoint> origin) {
402 if (origin)
403 frameView.setBaseLayoutViewportOrigin(LayoutPoint(origin.value()), FrameView::TriggerLayoutOrNot::No);
404 }, [&frameView, &layoutViewportRect, viewportRectStability](std::optional<FloatRect> overrideRect) {
405 if (!overrideRect)
406 return;
407
408 layoutViewportRect = overrideRect;
409 if (viewportRectStability != ViewportRectStability::ChangingObscuredInsetsInteractively)
410 frameView.setLayoutViewportOverrideRect(LayoutRect(overrideRect.value()), viewportRectStability == ViewportRectStability::Stable ? FrameView::TriggerLayoutOrNot::Yes : FrameView::TriggerLayoutOrNot::No);
411 }
412 );
413
414 frameView.setConstrainsScrollingToContentEdge(false);
415 frameView.notifyScrollPositionChanged(roundedIntPoint(scrollPosition));
416 frameView.setConstrainsScrollingToContentEdge(true);
417
418 frameView.setCurrentScrollType(previousScrollType);
419
420 if (scrollType == ScrollType::User && scrollingLayerPositionAction != ScrollingLayerPositionAction::Set) {
421 auto scrollingNodeID = frameView.scrollingNodeID();
422 if (viewportRectStability == ViewportRectStability::Stable)
423 reconcileViewportConstrainedLayerPositions(scrollingNodeID, frameView.rectForFixedPositionLayout(), scrollingLayerPositionAction);
424 else if (layoutViewportRect)
425 reconcileViewportConstrainedLayerPositions(scrollingNodeID, LayoutRect(layoutViewportRect.value()), scrollingLayerPositionAction);
426 }
427
428 if (!scrolledContentsLayerForFrameView(frameView))
429 return;
430
431 auto* counterScrollingLayer = counterScrollingLayerForFrameView(frameView);
432 auto* insetClipLayer = insetClipLayerForFrameView(frameView);
433 auto* contentShadowLayer = contentShadowLayerForFrameView(frameView);
434 auto* rootContentsLayer = rootContentsLayerForFrameView(frameView);
435 auto* headerLayer = headerLayerForFrameView(frameView);
436 auto* footerLayer = footerLayerForFrameView(frameView);
437
438 ASSERT(frameView.scrollPosition() == roundedIntPoint(scrollPosition));
439 LayoutPoint scrollPositionForFixed = frameView.scrollPositionForFixedPosition();
440 float topContentInset = frameView.topContentInset();
441
442 FloatPoint positionForInsetClipLayer;
443 if (insetClipLayer)
444 positionForInsetClipLayer = FloatPoint(insetClipLayer->position().x(), FrameView::yPositionForInsetClipLayer(scrollPosition, topContentInset));
445 FloatPoint positionForContentsLayer = frameView.positionForRootContentLayer();
446
447 FloatPoint positionForHeaderLayer = FloatPoint(scrollPositionForFixed.x(), FrameView::yPositionForHeaderLayer(scrollPosition, topContentInset));
448 FloatPoint positionForFooterLayer = FloatPoint(scrollPositionForFixed.x(),
449 FrameView::yPositionForFooterLayer(scrollPosition, topContentInset, frameView.totalContentsSize().height(), frameView.footerHeight()));
450
451 if (scrollType == ScrollType::Programmatic || scrollingLayerPositionAction == ScrollingLayerPositionAction::Set) {
452 reconcileScrollPosition(frameView, ScrollingLayerPositionAction::Set);
453
454 if (counterScrollingLayer)
455 counterScrollingLayer->setPosition(scrollPositionForFixed);
456 if (insetClipLayer)
457 insetClipLayer->setPosition(positionForInsetClipLayer);
458 if (contentShadowLayer)
459 contentShadowLayer->setPosition(positionForContentsLayer);
460 if (rootContentsLayer)
461 rootContentsLayer->setPosition(positionForContentsLayer);
462 if (headerLayer)
463 headerLayer->setPosition(positionForHeaderLayer);
464 if (footerLayer)
465 footerLayer->setPosition(positionForFooterLayer);
466 } else {
467 reconcileScrollPosition(frameView, ScrollingLayerPositionAction::Sync);
468
469 if (counterScrollingLayer)
470 counterScrollingLayer->syncPosition(scrollPositionForFixed);
471 if (insetClipLayer)
472 insetClipLayer->syncPosition(positionForInsetClipLayer);
473 if (contentShadowLayer)
474 contentShadowLayer->syncPosition(positionForContentsLayer);
475 if (rootContentsLayer)
476 rootContentsLayer->syncPosition(positionForContentsLayer);
477 if (headerLayer)
478 headerLayer->syncPosition(positionForHeaderLayer);
479 if (footerLayer)
480 footerLayer->syncPosition(positionForFooterLayer);
481 }
482}

484void AsyncScrollingCoordinator::reconcileScrollPosition(FrameView& frameView, ScrollingLayerPositionAction scrollingLayerPositionAction)
485{
486#if PLATFORM(IOS_FAMILY)
487 // Doing all scrolling like this (UIScrollView style) would simplify code.
488 auto* scrollContainerLayer = scrollContainerLayerForFrameView(frameView);
489 if (!scrollContainerLayer)
490 return;
491 if (scrollingLayerPositionAction == ScrollingLayerPositionAction::Set)
492 scrollContainerLayer->setBoundsOrigin(frameView.scrollPosition());
493 else
494 scrollContainerLayer->syncBoundsOrigin(frameView.scrollPosition());
495#else
496 // This uses scrollPosition because the root content layer accounts for scrollOrigin (see FrameView::positionForRootContentLayer()).
497 auto* scrolledContentsLayer = scrolledContentsLayerForFrameView(frameView);
498 if (!scrolledContentsLayer)
499 return;
500 if (scrollingLayerPositionAction == ScrollingLayerPositionAction::Set)
501 scrolledContentsLayer->setPosition(-frameView.scrollPosition());
502 else
503 scrolledContentsLayer->syncPosition(-frameView.scrollPosition());
504#endif
505}

515 ASSERT(isMainThread());
516 ASSERT(m_page);
517

525 }
526
527 if (orientation == VerticalScrollbar)
528 scrollableArea.verticalScrollbarLayerDidChange();
529 else
530 scrollableArea.horizontalScrollbarLayerDidChange();
531}

574Vector<ScrollingNodeID> AsyncScrollingCoordinator::childrenOfNode(ScrollingNodeID nodeID) const
575{
576 auto* scrollingNode = m_scrollingStateTree->stateNodeForID(nodeID);
577 if (!scrollingNode)
578 return { };
579
580 auto* children = scrollingNode->children();
581 if (!children || children->isEmpty())
582 return { };
583
584 Vector<ScrollingNodeID> childNodeIDs;
585 childNodeIDs.reserveInitialCapacity(children->size());
586 for (const auto& childNode : *children)
587 childNodeIDs.uncheckedAppend(childNode->scrollingNodeID());
588
589 return childNodeIDs;
590}

599void AsyncScrollingCoordinator::ensureRootStateNodeForFrameView(FrameView& frameView)
600{
601 ASSERT(frameView.scrollingNodeID());
602 if (m_scrollingStateTree->stateNodeForID(frameView.scrollingNodeID()))

608 insertNode(ScrollingNodeType::MainFrame, frameView.scrollingNodeID(), 0, 0);

611void AsyncScrollingCoordinator::setNodeLayers(ScrollingNodeID nodeID, const NodeLayers& nodeLayers)
612{
613 auto* node = m_scrollingStateTree->stateNodeForID(nodeID);
614 ASSERT(node);
615 if (!node)
616 return;
617
618 node->setLayer(nodeLayers.layer);
619
620 if (is<ScrollingStateScrollingNode>(node)) {
621 auto& scrollingNode = downcast<ScrollingStateScrollingNode>(*node);
622 scrollingNode.setScrollContainerLayer(nodeLayers.scrollContainerLayer);
623 scrollingNode.setScrolledContentsLayer(nodeLayers.scrolledContentsLayer);
624 scrollingNode.setHorizontalScrollbarLayer(nodeLayers.horizontalScrollbarLayer);
625 scrollingNode.setVerticalScrollbarLayer(nodeLayers.verticalScrollbarLayer);
626
627 if (is<ScrollingStateFrameScrollingNode>(node)) {
628 auto& frameScrollingNode = downcast<ScrollingStateFrameScrollingNode>(*node);
629 frameScrollingNode.setInsetClipLayer(nodeLayers.insetClipLayer);
630 frameScrollingNode.setCounterScrollingLayer(nodeLayers.counterScrollingLayer);

636void AsyncScrollingCoordinator::setFrameScrollingNodeState(ScrollingNodeID nodeID, const FrameView& frameView)

647 frameScrollingNode.setHeaderHeight(frameView.headerHeight());
648 frameScrollingNode.setFooterHeight(frameView.footerHeight());

651 frameScrollingNode.setAsyncFrameOrOverflowScrollingEnabled(settings.asyncFrameScrollingEnabled() || settings.asyncOverflowScrollingEnabled());
652 frameScrollingNode.setScrollingPerformanceTestingEnabled(settings.scrollingPerformanceTestingEnabled());

659 frameScrollingNode.setOverrideVisualViewportSize(FloatSize(visualOverrideRect.value().size()));

668 return visualViewport.width() < layoutViewport.width() || visualViewport.height() < layoutViewport.height();
669 };
670 frameScrollingNode.setVisualViewportIsSmallerThanLayoutViewport(visualViewportIsSmallerThanLayoutViewport(frameView));
671
672 frameScrollingNode.setScrollBehaviorForFixedElements(frameView.scrollBehaviorForFixedElements());
673}

762void AsyncScrollingCoordinator::setSynchronousScrollingReasons(ScrollingNodeID nodeID, OptionSet<SynchronousScrollingReason> reasons)
763{
764 auto* node = m_scrollingStateTree->stateNodeForID(nodeID);
765 if (!is<ScrollingStateScrollingNode>(node))
766 return;
767
768 auto& scrollingStateNode = downcast<ScrollingStateScrollingNode>(*node);
769 if (reasons && is<ScrollingStateFrameScrollingNode>(scrollingStateNode)) {
770 // The FrameView's GraphicsLayer is likely to be out-of-synch with the PlatformLayer
771 // at this point. So we'll update it before we switch back to main thread scrolling
772 // in order to avoid layer positioning bugs.
773 if (auto* frameView = frameViewForScrollingNode(nodeID))
774 reconcileScrollPosition(*frameView, ScrollingLayerPositionAction::Set);
775 }
776
777 // FIXME: Ideally all the "synchronousScrollingReasons" functions should be #ifdeffed.
778#if ENABLE(SCROLLING_THREAD)
779 scrollingStateNode.setSynchronousScrollingReasons(reasons);
780#endif
781}

828ScrollingNodeID AsyncScrollingCoordinator::scrollableContainerNodeID(const RenderObject& renderer) const
829{
830 if (auto overflowScrollingNodeID = renderer.view().compositor().asyncScrollableContainerNodeID(renderer))
831 return overflowScrollingNodeID;
832
833 // If we're in a scrollable frame, return that.
834 auto* frameView = renderer.frame().view();
835 if (!frameView)
836 return 0;
837
838 if (auto scrollingNodeID = frameView->scrollingNodeID())
839 return scrollingNodeID;
840
841 // Otherwise, look for a scrollable element in the containing frame.
842 if (auto* ownerElement = renderer.document().ownerElement()) {
843 if (auto* frameRenderer = ownerElement->renderer())
844 return scrollableContainerNodeID(*frameRenderer);
845 }
846
847 return 0;
848}
