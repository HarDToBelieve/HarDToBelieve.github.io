51void ScrollLatchingController::clear()
52{
53 LOG_WITH_STREAM(ScrollLatching, stream << "ScrollLatchingController::clear()");
54 m_cumulativeEventDelta = { };
55 m_frameStateStack.clear();
56}

58// FIXME: This logic is different from ScrollingTreeLatchingController, which simply lets the latching state elapse after 100ms.
59void ScrollLatchingController::clearOrScheduleClearIfNeeded(const PlatformWheelEvent& wheelEvent)
60{
61 if (wheelEvent.shouldResetLatching() || wheelEvent.isNonGestureEvent()) {
62 clear();
63 LOG_WITH_STREAM(ScrollLatching, stream << "ScrollLatchingController::clearOrScheduleClearingLatchedStateIfNeeded() - event" << wheelEvent << ", resetting latching");
64 return;
65 }
66
67 if (m_clearLatchingStateTimer.isActive()) {
68 // If another wheel event scrolling starts, stop the timer manually, and reset the latched state immediately.
69 if (wheelEvent.isGestureStart()) {
70 LOG_WITH_STREAM(ScrollLatching, stream << "ScrollLatchingController::clearOrScheduleClearingLatchedStateIfNeeded() - event" << wheelEvent << ", timer pending, another scroll starting");
71 clear();
72 m_clearLatchingStateTimer.stop();
73 } else if (wheelEvent.isTransitioningToMomentumScroll()) {
74 // Wheel events machinery is transitioning to momentum scrolling, so no need to reset latched state. Stop the timer.
75 m_clearLatchingStateTimer.stop();
76 }
77 return;
78 }
79
80 if (wheelEvent.isEndOfNonMomentumScroll()) {
81 LOG_WITH_STREAM(ScrollLatching, stream << "ScrollLatchingController::clearOrScheduleClearingLatchedStateIfNeeded() - event" << wheelEvent << ", scheduling clear timer");
82 m_clearLatchingStateTimer.startOneShot(resetLatchedStateTimeout);
83 }
84}
85
86void ScrollLatchingController::clearTimerFired()
87{
88 LOG_WITH_STREAM(ScrollLatching, stream << "ScrollLatchingController::clearTimerFired() - clearing state");
89 clear();
90}

92void ScrollLatchingController::receivedWheelEvent(const PlatformWheelEvent& wheelEvent)
93{
94 clearOrScheduleClearIfNeeded(wheelEvent);
95
96 if (wheelEvent.isGestureStart() || wheelEvent.isNonGestureEvent())
97 m_cumulativeEventDelta = wheelEvent.delta();
98 else
99 m_cumulativeEventDelta += wheelEvent.delta();
100}

102bool ScrollLatchingController::latchingAllowsScrollingInFrame(const Frame& frame, WeakPtr<ScrollableArea>& latchedScroller) const
103{
104 if (m_frameStateStack.isEmpty())
105 return true;
106
107 if (auto* frameState = stateForFrame(frame)) {
108 latchedScroller = frameState->scrollableArea;
109 return !!frameState->scrollableArea;
110 }
111 return false;
112}

114void ScrollLatchingController::updateAndFetchLatchingStateForFrame(Frame& frame, const PlatformWheelEvent& wheelEvent, RefPtr<Element>& latchedElement, WeakPtr<ScrollableArea>& scrollableArea, bool& isOverWidget)
115{
116 if (wheelEvent.isGestureStart()) {
117 // We can have existing state here because state is cleared on a timer.
118 if (!hasStateForFrame(frame)) {
119 FrameState state;
120 state.frame = &frame;
121 state.wheelEventElement = makeWeakPtr(latchedElement.get());
122 if (shouldLatchToScrollableArea(frame, scrollableArea.get(), m_cumulativeEventDelta))
123 state.scrollableArea = scrollableArea;
124 state.isOverWidget = isOverWidget;
125
126 LOG_WITH_STREAM(ScrollLatching, stream << "ScrollLatchingController::updateAndFetchLatchingStateForFrame() - pushing state for Frame " << &frame << " element " << state.wheelEventElement.get() << " scrollableArea " << state.scrollableArea);
127 m_frameStateStack.append(WTFMove(state));
128 return;
129 }
130 }
131
132 if (wheelEvent.isGestureContinuation()) {
133 auto* state = stateForFrame(frame);
134 if (!state)
135 return;
136
137 // We may not have latched at gesture start because of small deltas. Re-evaluate latching based on accumulated delta.
138 if (!state->scrollableArea && shouldLatchToScrollableArea(frame, scrollableArea.get(), m_cumulativeEventDelta))
139 state->scrollableArea = scrollableArea;
140 }
141
142 if (!wheelEvent.useLatchedEventElement())
143 return;
144
145 for (const auto& state : m_frameStateStack) {
146 if (state.frame == &frame) {
147 latchedElement = state.wheelEventElement.get();
148 scrollableArea = state.scrollableArea;
149 isOverWidget = state.isOverWidget;
150
151 LOG_WITH_STREAM(ScrollLatching, stream << "ScrollLatchingController::updateAndFetchLatchingStateForFrame() - using state for Frame " << &frame << " element " << latchedElement.get() << " scrollableArea " << scrollableArea);
152 }
153 }
154}

156void ScrollLatchingController::removeLatchingStateForTarget(const Element& element)
157{
158 if (m_frameStateStack.isEmpty())
159 return;
160
161 auto findResult = m_frameStateStack.findMatching([&element] (const auto& state) {
162 auto* wheelElement = state.wheelEventElement.get();
163 return wheelElement && element.isEqualNode(wheelElement);
164 });
165
166 // If this element is found in the latching stack, just clear the whole stack. We can't just remove one entry,
167 // since the stack has to match the frame hierarchy.
168 if (findResult != notFound)
169 m_frameStateStack.clear();
170}
171
172void ScrollLatchingController::removeLatchingStateForFrame(const Frame& frame)
173{
174 if (m_frameStateStack.isEmpty())
175 return;
176
177 // If the frame was in the latching stack, just clear state.
178 if (auto* frameState = stateForFrame(frame))
179 clear();
180}

182static bool deltaIsPredominantlyVertical(FloatSize delta)
183{
184 return std::abs(delta.height()) > std::abs(delta.width());
185}
186
187bool ScrollLatchingController::shouldLatchToScrollableArea(const Frame& frame, ScrollableArea* scrollableArea, FloatSize scrollDelta) const
188{
189 if (!scrollableArea)
190 return false;
191
192 // We always allow the main frame to receive wheel events to permit rubber-banding.
193 if (frame.isMainFrame() && scrollableArea == frame.view())
194 return true;
195
196 if (!scrollableArea->canHaveScrollbars())
197 return false;
198
199 if (scrollDelta.isZero())
200 return false;
201
202 if (!deltaIsPredominantlyVertical(scrollDelta) && scrollDelta.width()) {
203 if (!scrollableArea->horizontalScrollbar())
204 return false;
205
206 if (scrollDelta.width() < 0)
207 return !scrollableArea->scrolledToRight();
208
209 return !scrollableArea->scrolledToLeft();
210 }
211
212 if (!scrollableArea->verticalScrollbar())
213 return false;
214
215 if (scrollDelta.height() < 0)
216 return !scrollableArea->scrolledToBottom();
217
218 return !scrollableArea->scrolledToTop();
219}

221bool ScrollLatchingController::hasStateForFrame(const Frame& frame) const
222{
223 for (const auto& state : m_frameStateStack) {
224 if (state.frame == &frame)
225 return true;
226 }
227 return false;
228}

230ScrollLatchingController::FrameState* ScrollLatchingController::stateForFrame(const Frame& frame)
231{
232 for (auto& state : m_frameStateStack) {
233 if (state.frame == &frame)
234 return &state;
235 }
236 return nullptr;
237}
238
239const ScrollLatchingController::FrameState* ScrollLatchingController::stateForFrame(const Frame& frame) const
240{
241 for (const auto& state : m_frameStateStack) {
242 if (state.frame == &frame)
243 return &state;
244 }
245 return nullptr;
246}

248void ScrollLatchingController::dump(WTF::TextStream& ts) const
249{
250 TextStream multilineStream;
251 multilineStream.setIndent(ts.indent() + 2);
252
253 for (const auto& state : m_frameStateStack) {
254 TextStream::GroupScope groupScope(multilineStream);
255 multilineStream.dumpProperty("frame", ValueOrNull(state.frame));
256 multilineStream.dumpProperty("element", ValueOrNull(state.wheelEventElement.get()));
257 multilineStream.dumpProperty("scrollable area", ValueOrNull(state.scrollableArea.get()));
258 multilineStream.dumpProperty("is over widget", state.isOverWidget);
259 }
260
261 ts << "ScrollLatchingController state " << multilineStream.release();
262}
263
264TextStream& operator<<(TextStream& ts, const ScrollLatchingController& controller)
265{
266 controller.dump(ts);
267 return ts;
268}
