43template <typename UnitType>
44struct PotentialSnapPointSearchResult {
45 std::optional<std::pair<UnitType, unsigned>> previous;
46 std::optional<std::pair<UnitType, unsigned>> next;
47 std::optional<std::pair<UnitType, unsigned>> snapStop;
48 bool landedInsideSnapAreaThatConsumesViewport;
49};
50
51template <typename InfoType, typename UnitType>
52static PotentialSnapPointSearchResult<UnitType> searchForPotentialSnapPoints(const InfoType& info, ScrollEventAxis axis, UnitType viewportLength, UnitType destinationOffset, std::optional<UnitType> originalOffset)
53{
54 const auto& snapOffsets = info.offsetsForAxis(axis);
55 std::optional<std::pair<UnitType, unsigned>> previous, next, exact, snapStop;
56 bool landedInsideSnapAreaThatConsumesViewport = false;
57
58 // A particular snap stop is better if it's between the original offset and destination offset and closer original
59 // offset than the previously selected snap stop. We always want to stop at the snap stop closest to the original offset.
60 auto isBetterSnapStop = [&](UnitType candidate) {
61 if (!originalOffset)
62 return false;
63 auto original = *originalOffset;
64 if (candidate <= std::min(destinationOffset, original) || candidate >= std::max(destinationOffset, original))
65 return false;
66 return !snapStop || std::abs(float { candidate - original }) < std::abs(float { (*snapStop).first - original });
67 };
68
69 for (unsigned i = 0; i < snapOffsets.size(); i++) {
70 UnitType potentialSnapOffset = snapOffsets[i].offset;
71
72 const auto& snapArea = info.snapAreas[snapOffsets[i].snapAreaIndex];
73 auto snapAreaMin = axis == ScrollEventAxis::Horizontal ? snapArea.x() : snapArea.y();
74 auto snapAreaMax = axis == ScrollEventAxis::Horizontal ? snapArea.maxX() : snapArea.maxY();
75 landedInsideSnapAreaThatConsumesViewport |= snapAreaMin <= destinationOffset && snapAreaMax >= (destinationOffset + viewportLength);
76
77 if (potentialSnapOffset == destinationOffset)
78 exact = std::make_pair(potentialSnapOffset, i);
79 else if (potentialSnapOffset < destinationOffset)
80 previous = std::make_pair(potentialSnapOffset, i);
81 else if (!next && potentialSnapOffset > destinationOffset)
82 next = std::make_pair(potentialSnapOffset, i);
83
84 if (snapOffsets[i].stop == ScrollSnapStop::Always && isBetterSnapStop(potentialSnapOffset))
85 snapStop = std::make_pair(potentialSnapOffset, i);
86 }
87
88 if (exact)
89 return { exact, exact, snapStop, landedInsideSnapAreaThatConsumesViewport };
90 return { previous, next, snapStop, landedInsideSnapAreaThatConsumesViewport };
91}

93template <typename InfoType, typename SizeType, typename LayoutType>
94static std::pair<LayoutType, std::optional<unsigned>> closestSnapOffsetWithInfoAndAxis(const InfoType& info, ScrollEventAxis axis, const SizeType& viewportSize, LayoutType scrollDestinationOffset, float velocity, std::optional<LayoutType> originalOffsetForDirectionalSnapping)
95{
96 const auto& snapOffsets = info.offsetsForAxis(axis);
97 auto pairForNoSnapping = std::make_pair(scrollDestinationOffset, std::nullopt);
98 if (snapOffsets.isEmpty())
99 return pairForNoSnapping;
100
101 auto viewportLength = axis == ScrollEventAxis::Horizontal ? viewportSize.width() : viewportSize.height();
102 auto [previous, next, snapStop, landedInsideSnapAreaThatConsumesViewport] = searchForPotentialSnapPoints(info, axis, viewportLength, scrollDestinationOffset, originalOffsetForDirectionalSnapping);
103 if (snapStop)
104 return *snapStop;
105
106 // From https://www.w3.org/TR/css-scroll-snap-1/#snap-overflow
107 // "If the snap area is larger than the snapport in a particular axis, then any scroll position
108 // in which the snap area covers the snapport, and the distance between the geometrically
109 // previous and subsequent snap positions in that axis is larger than size of the snapport in
110 // that axis, is a valid snap position in that axis. The UA may use the specified alignment as a
111 // more precise target for certain scroll operations (e.g. explicit paging)."
112 if (landedInsideSnapAreaThatConsumesViewport && (!previous || !next || ((*next).first - (*previous).first) >= viewportLength))
113 return pairForNoSnapping;
114
115 auto isNearEnoughToOffsetForProximity = [&](LayoutType candidateSnapOffset) {
116 if (info.strictness != ScrollSnapStrictness::Proximity)
117 return true;
118
119 // This is an arbitrary choice for what it means to be "in proximity" of a snap offset. We should play around with
120 // this and see what feels best.
121 static const float ratioOfScrollPortAxisLengthToBeConsideredForProximity = 0.3;
122 return std::abs(float {candidateSnapOffset - scrollDestinationOffset}) <= (viewportLength * ratioOfScrollPortAxisLengthToBeConsideredForProximity);
123 };
124
125 if (scrollDestinationOffset <= snapOffsets.first().offset)
126 return isNearEnoughToOffsetForProximity(snapOffsets.first().offset) ? std::make_pair(snapOffsets.first().offset, std::make_optional(0u)) : pairForNoSnapping;
127
128 if (scrollDestinationOffset >= snapOffsets.last().offset) {
129 unsigned lastIndex = static_cast<unsigned>(snapOffsets.size() - 1);
130 return isNearEnoughToOffsetForProximity(snapOffsets.last().offset) ? std::make_pair(snapOffsets.last().offset, std::make_optional(lastIndex)) : pairForNoSnapping;
131 }
132
133 if (previous && !isNearEnoughToOffsetForProximity((*previous).first))
134 previous.reset();
135 if (next && !isNearEnoughToOffsetForProximity((*next).first))
136 next.reset();
137
138 if (originalOffsetForDirectionalSnapping) {
139 // From https://www.w3.org/TR/css-scroll-snap-1/#choosing
140 // "User agents must ensure that a user can “escape” a snap position, regardless of the scroll
141 // method. For example, if the snap type is mandatory and the next snap position is more than
142 // two screen-widths away, a naïve “always snap to nearest” selection algorithm might “trap” the
143 //
144 // For a directional scroll, we never snap back to the original scroll position or before it,
145 // always preferring the snap offset in the scroll direction.
146 auto& originalOffset = *originalOffsetForDirectionalSnapping;
147 if (originalOffset < scrollDestinationOffset && previous && (*previous).first <= originalOffset)
148 previous.reset();
149 if (originalOffset > scrollDestinationOffset && next && (*next).first >= originalOffset)
150 next.reset();
151 }
152
153 if (!previous && !next)
154 return pairForNoSnapping;
155 if (!previous)
156 return *next;
157 if (!next)
158 return *previous;
159
160 // If this scroll isn't directional, then choose whatever snap point is closer, otherwise pick the offset in the scroll direction.
161 if (!std::abs(velocity))
162 return (scrollDestinationOffset - (*previous).first) <= ((*next).first - scrollDestinationOffset) ? *previous : *next;
163 return velocity < 0 ? *previous : *next;
164}

166enum class InsetOrOutset {
167 Inset,
168 Outset
169};

171static LayoutRect computeScrollSnapPortOrAreaRect(const LayoutRect& rect, const LengthBox& insetOrOutsetBox, InsetOrOutset insetOrOutset)
172{
173 // We are using minimumValueForLength here for insetOrOutset box, because if this value is defined by scroll-padding then the
174 // Length of any side may be "auto." In that case, we want to use 0, because that is how WebKit currently interprets an "auto"
175 // value for scroll-padding. See: https://drafts.csswg.org/css-scroll-snap-1/#propdef-scroll-padding
176 LayoutBoxExtent extents(
177 minimumValueForLength(insetOrOutsetBox.top(), rect.height()), minimumValueForLength(insetOrOutsetBox.right(), rect.width()),
178 minimumValueForLength(insetOrOutsetBox.bottom(), rect.height()), minimumValueForLength(insetOrOutsetBox.left(), rect.width()));
179 auto snapPortOrArea(rect);
180 if (insetOrOutset == InsetOrOutset::Inset)
181 snapPortOrArea.contract(extents);
182 else
183 snapPortOrArea.expand(extents);
184 return snapPortOrArea;
185}

187static LayoutUnit computeScrollSnapAlignOffset(LayoutUnit minLocation, LayoutUnit maxLocation, ScrollSnapAxisAlignType alignment, bool axisIsFlipped)
188{
189 switch (alignment) {
190 case ScrollSnapAxisAlignType::Start:
191 return axisIsFlipped ? maxLocation : minLocation;
192 case ScrollSnapAxisAlignType::Center:
193 return (minLocation + maxLocation) / 2;
194 case ScrollSnapAxisAlignType::End:
195 return axisIsFlipped ? minLocation : maxLocation;
196 default:
197 ASSERT_NOT_REACHED();
198 return 0;
199 }
200}

202void updateSnapOffsetsForScrollableArea(ScrollableArea& scrollableArea, const RenderBox& scrollingElementBox, const RenderStyle& scrollingElementStyle, LayoutRect viewportRectInBorderBoxCoordinates, WritingMode writingMode, TextDirection textDirection)
203{
204 auto scrollSnapType = scrollingElementStyle.scrollSnapType();
205 const auto& boxesWithScrollSnapPositions = scrollingElementBox.view().boxesWithScrollSnapPositions();
206 if (scrollSnapType.strictness == ScrollSnapStrictness::None || boxesWithScrollSnapPositions.isEmpty()) {
207 scrollableArea.clearSnapOffsets();
208 return;
209 }
210
211 auto addOrUpdateStopForSnapOffset = [](HashMap<float, SnapOffset<LayoutUnit>>& offsets, SnapOffset<LayoutUnit> newOffset)
212 {
213 // If the offset already exists, we ensure that it has ScrollSnapStop::Always, when appropriate.
214 auto addResult = offsets.add(newOffset.offset, newOffset);
215 if (newOffset.stop == ScrollSnapStop::Always)
216 addResult.iterator->value.stop = ScrollSnapStop::Always;
217 };
218
219 HashMap<float, SnapOffset<LayoutUnit>> verticalSnapOffsetsMap;
220 HashMap<float, SnapOffset<LayoutUnit>> horizontalSnapOffsetsMap;
221 Vector<LayoutRect> snapAreas;
222
223 auto maxScrollOffset = scrollableArea.maximumScrollOffset();
224 auto scrollPosition = LayoutPoint { scrollableArea.scrollPosition() };
225
226 // text-direction flips the inline axis and writing-mode can flip the block axis. Whether or
227 // not the writing-mode is vertical determines the physical orientation of the block and inline axes.
228 bool scrollerHasVerticalWritingMode = isVerticalWritingMode(writingMode);
229 bool blockAxisFlipped = isFlippedWritingMode(writingMode);
230 bool inlineAxisFlipped = textDirection == TextDirection::RTL;
231 bool xAxisFlipped = scrollerHasVerticalWritingMode ? blockAxisFlipped : inlineAxisFlipped;
232 bool yAxisFlipped = scrollerHasVerticalWritingMode ? inlineAxisFlipped : blockAxisFlipped;
233
234 bool hasHorizontalSnapOffsets = scrollSnapType.axis == ScrollSnapAxis::Both || scrollSnapType.axis == ScrollSnapAxis::XAxis;
235 bool hasVerticalSnapOffsets = scrollSnapType.axis == ScrollSnapAxis::Both || scrollSnapType.axis == ScrollSnapAxis::YAxis;
236 if (scrollSnapType.axis == ScrollSnapAxis::Block) {
237 hasHorizontalSnapOffsets = scrollerHasVerticalWritingMode;
238 hasVerticalSnapOffsets = !scrollerHasVerticalWritingMode;
239 }
240 if (scrollSnapType.axis == ScrollSnapAxis::Inline) {
241 hasHorizontalSnapOffsets = !scrollerHasVerticalWritingMode;
242 hasVerticalSnapOffsets = scrollerHasVerticalWritingMode;
243 }
244
245 // The bounds of the scrolling container's snap port, where the top left of the scrolling container's border box is the origin.
246 auto scrollSnapPort = computeScrollSnapPortOrAreaRect(viewportRectInBorderBoxCoordinates, scrollingElementStyle.scrollPadding(), InsetOrOutset::Inset);
247 LOG_WITH_STREAM(ScrollSnap, stream << "Computing scroll snap offsets for " << scrollableArea << " in snap port " << scrollSnapPort);
248 for (auto* child : boxesWithScrollSnapPositions) {
249 if (child->enclosingScrollableContainerForSnapping() != &scrollingElementBox)
250 continue;
251
252 // The bounds of the child element's snap area, where the top left of the scrolling container's border box is the origin.
253 // The snap area is the bounding box of the child element's border box, after applying transformations.
254 auto scrollSnapArea = LayoutRect(child->localToContainerQuad(FloatQuad(child->borderBoundingBox()), &scrollingElementBox).boundingBox());
255
256 // localToContainerQuad will transform the scroll snap area by the scroll position, except in the case that this position is
257 // coming from a ScrollView. We want the transformed area, but without scroll position taken into account.
258 if (!scrollableArea.isScrollView())
259 scrollSnapArea.moveBy(scrollPosition);
260
261 scrollSnapArea = computeScrollSnapPortOrAreaRect(scrollSnapArea, child->style().scrollMargin(), InsetOrOutset::Outset);
262 LOG_WITH_STREAM(ScrollSnap, stream << " Considering scroll snap target area " << scrollSnapArea);
263 auto alignment = child->style().scrollSnapAlign();
264 auto stop = child->style().scrollSnapStop();
265
266 ScrollSnapAxisAlignType xAlign = scrollerHasVerticalWritingMode ? alignment.blockAlign : alignment.inlineAlign;
267 ScrollSnapAxisAlignType yAlign = scrollerHasVerticalWritingMode ? alignment.inlineAlign : alignment.blockAlign;
268 bool snapsHorizontally = hasHorizontalSnapOffsets && xAlign != ScrollSnapAxisAlignType::None;
269 bool snapsVertically = hasVerticalSnapOffsets && yAlign != ScrollSnapAxisAlignType::None;
270
271 if (!snapsHorizontally && !snapsVertically)
272 continue;
273
274 // The scroll snap area is defined via its scroll position, so convert the snap area rectangle to be relative to scroll offsets.
275 auto snapAreaOriginRelativeToBorderEdge = scrollSnapArea.location() - scrollSnapPort.location();
276 LayoutRect scrollSnapAreaAsOffsets(scrollableArea.scrollOffsetFromPosition(roundedIntPoint(snapAreaOriginRelativeToBorderEdge)), scrollSnapArea.size());
277 snapAreas.append(scrollSnapAreaAsOffsets);
278
279 if (snapsHorizontally) {
280 auto absoluteScrollXPosition = computeScrollSnapAlignOffset(scrollSnapArea.x(), scrollSnapArea.maxX(), xAlign, xAxisFlipped) - computeScrollSnapAlignOffset(scrollSnapPort.x(), scrollSnapPort.maxX(), xAlign, xAxisFlipped);
281 auto absoluteScrollOffset = clampTo<int>(scrollableArea.scrollOffsetFromPosition({ roundToInt(absoluteScrollXPosition), 0 }).x(), 0, maxScrollOffset.x());
282 addOrUpdateStopForSnapOffset(horizontalSnapOffsetsMap, { absoluteScrollOffset, stop, snapAreas.size() - 1 });
283 }
284 if (snapsVertically) {
285 auto absoluteScrollYPosition = computeScrollSnapAlignOffset(scrollSnapArea.y(), scrollSnapArea.maxY(), yAlign, yAxisFlipped) - computeScrollSnapAlignOffset(scrollSnapPort.y(), scrollSnapPort.maxY(), yAlign, yAxisFlipped);
286 auto absoluteScrollOffset = clampTo<int>(scrollableArea.scrollOffsetFromPosition({ 0, roundToInt(absoluteScrollYPosition) }).y(), 0, maxScrollOffset.y());
287 addOrUpdateStopForSnapOffset(verticalSnapOffsetsMap, { absoluteScrollOffset, stop, snapAreas.size() - 1 });
288 }
289
290 if (!snapAreas.isEmpty())
291 LOG_WITH_STREAM(ScrollSnap, stream << " => Computed snap areas: " << snapAreas);
292 }
293
294 auto compareSnapOffsets = [](const SnapOffset<LayoutUnit>& a, const SnapOffset<LayoutUnit>& b)
295 {
296 return a.offset < b.offset;
297 };
298
299 Vector<SnapOffset<LayoutUnit>> horizontalSnapOffsets = copyToVector(horizontalSnapOffsetsMap.values());
300 if (!horizontalSnapOffsets.isEmpty()) {
301 std::sort(horizontalSnapOffsets.begin(), horizontalSnapOffsets.end(), compareSnapOffsets);
302 LOG_WITH_STREAM(ScrollSnap, stream << " => Computed horizontal scroll snap offsets: " << horizontalSnapOffsets);
303 }
304
305 Vector<SnapOffset<LayoutUnit>> verticalSnapOffsets = copyToVector(verticalSnapOffsetsMap.values());
306 if (!verticalSnapOffsets.isEmpty()) {
307 std::sort(verticalSnapOffsets.begin(), verticalSnapOffsets.end(), compareSnapOffsets);
308 LOG_WITH_STREAM(ScrollSnap, stream << " => Computed vertical scroll snap offsets: " << verticalSnapOffsets);
309 }
310
311 scrollableArea.setScrollSnapOffsetInfo({
312 scrollSnapType.strictness,
313 horizontalSnapOffsets,
314 verticalSnapOffsets,
315 snapAreas
316 });
317}

329template <typename InputType, typename InputRectType, typename OutputType, typename OutputRectType>
330static ScrollSnapOffsetsInfo<OutputType, OutputRectType> convertOffsetInfo(const ScrollSnapOffsetsInfo<InputType, InputRectType>& input, float scaleFactor = 0.0)
331{
332 auto convertOffsets = [scaleFactor](const Vector<SnapOffset<InputType>>& input)
333 {
334 Vector<SnapOffset<OutputType>> output;
335 output.reserveInitialCapacity(input.size());
336 for (auto& offset : input)
337 output.uncheckedAppend({ convertOffsetUnit(offset.offset, scaleFactor), offset.stop, offset.snapAreaIndex });
338 return output;
339 };
340
341 auto convertRects = [scaleFactor](const Vector<InputRectType>& input)
342 {
343 Vector<OutputRectType> output;
344 output.reserveInitialCapacity(input.size());
345 for (auto& rect : input) {
346 OutputRectType outputRect(
347 convertOffsetUnit(rect.x(), scaleFactor), convertOffsetUnit(rect.y(), scaleFactor),
348 convertOffsetUnit(rect.width(), scaleFactor), convertOffsetUnit(rect.height(), scaleFactor));
349 output.uncheckedAppend(outputRect);
350 }
351
352 return output;
353 };
354
355 return {
356 input.strictness,
357 convertOffsets(input.horizontalSnapOffsets),
358 convertOffsets(input.verticalSnapOffsets),
359 convertRects(input.snapAreas),
360 };
361}
