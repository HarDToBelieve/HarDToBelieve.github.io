42template <typename T>
43struct SnapOffset {
44 T offset;
45 ScrollSnapStop stop;
46 size_t snapAreaIndex;
47};
48
49template <typename UnitType, typename RectType>
50struct ScrollSnapOffsetsInfo {
51 WTF_MAKE_STRUCT_FAST_ALLOCATED;
52 ScrollSnapStrictness strictness;
53 Vector<SnapOffset<UnitType>> horizontalSnapOffsets;
54 Vector<SnapOffset<UnitType>> verticalSnapOffsets;
55 Vector<RectType> snapAreas;
56
57 bool isEqual(const ScrollSnapOffsetsInfo<UnitType, RectType>& other) const
58 {
59 return strictness == other.strictness && horizontalSnapOffsets == other.horizontalSnapOffsets && verticalSnapOffsets == other.verticalSnapOffsets && snapAreas == other.snapAreas;
60 }
61
62 bool isEmpty() const
63 {
64 return horizontalSnapOffsets.isEmpty() && verticalSnapOffsets.isEmpty();
65 }
66
67 Vector<SnapOffset<UnitType>> offsetsForAxis(ScrollEventAxis axis) const
68 {
69 return axis == ScrollEventAxis::Vertical ? verticalSnapOffsets : horizontalSnapOffsets;
70 }

75};
