71bool ScrollingCoordinator::coordinatesScrollingForFrameView(const FrameView& frameView) const
72{
73 ASSERT(isMainThread());
74 ASSERT(m_page);
75
76 if (!frameView.frame().isMainFrame() && !m_page->settings().scrollingTreeIncludesFrames()
77#if PLATFORM(MAC) || USE(NICOSIA)
78 && !m_page->settings().asyncFrameScrollingEnabled()
79#endif
80 )
81 return false;
82
83 auto* renderView = frameView.frame().contentRenderer();
84 if (!renderView)
85 return false;
86 return renderView->usesCompositing();
87}

102EventTrackingRegions ScrollingCoordinator::absoluteEventTrackingRegionsForFrame(const Frame& frame) const
103{
104 auto* renderView = frame.contentRenderer();
105 if (!renderView || renderView->renderTreeBeingDestroyed())
106 return EventTrackingRegions();
107
108#if ENABLE(IOS_TOUCH_EVENTS)
109 // On iOS, we use nonFastScrollableRegion to represent the region covered by elements with touch event handlers.
110 ASSERT(frame.isMainFrame());
111 auto* document = frame.document();
112 if (!document)
113 return { };
114 return document->eventTrackingRegions();
115#else
116 auto* frameView = frame.view();
117 if (!frameView)
118 return EventTrackingRegions();
119
120 Region nonFastScrollableRegion;
121
122 // FIXME: should ASSERT(!frameView->needsLayout()) here, but need to fix DebugPageOverlays
123 // to not ask for regions at bad times.
124
125 if (auto* scrollableAreas = frameView->scrollableAreas()) {
126 for (auto& scrollableArea : *scrollableAreas) {
127 // Composited scrollable areas can be scrolled off the main thread.
128 if (scrollableArea->usesAsyncScrolling())
129 continue;
130
131 bool isInsideFixed;
132 IntRect box = scrollableArea->scrollableAreaBoundingBox(&isInsideFixed);
133 if (isInsideFixed)
134 box = IntRect(frameView->fixedScrollableAreaBoundsInflatedForScrolling(LayoutRect(box)));
135
136 nonFastScrollableRegion.unite(box);
137 }
138 }
139
140 for (auto& widget : frameView->widgetsInRenderTree()) {
141 if (!is<PluginViewBase>(*widget))
142 continue;
143 if (!downcast<PluginViewBase>(*widget).wantsWheelEvents())
144 continue;
145 auto* renderWidget = RenderWidget::find(*widget);
146 if (!renderWidget)
147 continue;
148 nonFastScrollableRegion.unite(renderWidget->absoluteBoundingBoxRect());
149 }
150
151 EventTrackingRegions eventTrackingRegions;
152
153 // FIXME: if we've already accounted for this subframe as a scrollable area, we can avoid recursing into it here.
154 for (auto* subframe = frame.tree().firstChild(); subframe; subframe = subframe->tree().nextSibling()) {
155 auto* subframeView = subframe->view();
156 if (!subframeView)
157 continue;
158
159 EventTrackingRegions subframeRegion = absoluteEventTrackingRegionsForFrame(*subframe);
160 // Map from the frame document to our document.
161 IntPoint offset = subframeView->contentsToContainingViewContents(IntPoint());
162
163 // FIXME: this translation ignores non-trival transforms on the frame.
164 subframeRegion.translate(toIntSize(offset));
165 eventTrackingRegions.unite(subframeRegion);
166 }
167
168#if !ENABLE(WHEEL_EVENT_REGIONS)
169 auto wheelHandlerRegion = frame.document()->absoluteRegionForEventTargets(frame.document()->wheelEventTargets());
170 bool wheelHandlerInFixedContent = wheelHandlerRegion.second;
171 if (wheelHandlerInFixedContent) {
172 // FIXME: need to handle position:sticky here too.
173 LayoutRect inflatedWheelHandlerBounds = frameView->fixedScrollableAreaBoundsInflatedForScrolling(LayoutRect(wheelHandlerRegion.first.bounds()));
174 wheelHandlerRegion.first.unite(enclosingIntRect(inflatedWheelHandlerBounds));
175 }
176
177 nonFastScrollableRegion.unite(wheelHandlerRegion.first);
178#endif
179
180 // FIXME: If this is not the main frame, we could clip the region to the frame's bounds.
181 eventTrackingRegions.uniteSynchronousRegion(eventNames().wheelEvent, nonFastScrollableRegion);
182
183 return eventTrackingRegions;
184#endif
185}

287bool ScrollingCoordinator::hasVisibleSlowRepaintViewportConstrainedObjects(const FrameView& frameView) const
288{
289 auto* viewportConstrainedObjects = frameView.viewportConstrainedObjects();
290 if (!viewportConstrainedObjects)
291 return false;
292
293 for (auto& viewportConstrainedObject : *viewportConstrainedObjects) {
294 if (!is<RenderBoxModelObject>(viewportConstrainedObject) || !viewportConstrainedObject.hasLayer())
295 return true;
296 auto& layer = *downcast<RenderBoxModelObject>(viewportConstrainedObject).layer();
297 // Any explicit reason that a fixed position element is not composited shouldn't cause slow scrolling.
298 if (!layer.isComposited() && layer.viewportConstrainedNotCompositedReason() == RenderLayer::NoNotCompositedReason)
299 return true;
300 }
301 return false;
302}

304void ScrollingCoordinator::updateSynchronousScrollingReasons(FrameView& frameView)
305{
306 ASSERT(coordinatesScrollingForFrameView(frameView));
307
308 OptionSet<SynchronousScrollingReason> newSynchronousScrollingReasons;
309
310 // RenderLayerCompositor::updateSynchronousScrollingReasons maintains this bit, so maintain its current value.
311 if (synchronousScrollingReasons(frameView.scrollingNodeID()).contains(SynchronousScrollingReason::HasSlowRepaintObjects))
312 newSynchronousScrollingReasons.add(SynchronousScrollingReason::HasSlowRepaintObjects);
313
314 if (m_forceSynchronousScrollLayerPositionUpdates)
315 newSynchronousScrollingReasons.add(SynchronousScrollingReason::ForcedOnMainThread);
316
317 if (hasVisibleSlowRepaintViewportConstrainedObjects(frameView))
318 newSynchronousScrollingReasons.add(SynchronousScrollingReason::HasNonLayerViewportConstrainedObjects);
319
320 if (frameView.frame().mainFrame().document() && frameView.frame().document()->isImageDocument())
321 newSynchronousScrollingReasons.add(SynchronousScrollingReason::IsImageDocument);
322
323 setSynchronousScrollingReasons(frameView.scrollingNodeID(), newSynchronousScrollingReasons);
324}

326void ScrollingCoordinator::updateSynchronousScrollingReasonsForAllFrames()
327{
328 for (Frame* frame = &m_page->mainFrame(); frame; frame = frame->tree().traverseNext()) {
329 if (auto* frameView = frame->view()) {
330 if (coordinatesScrollingForFrameView(*frameView))
331 updateSynchronousScrollingReasons(*frameView);
332 }
333 }
334}

336void ScrollingCoordinator::setForceSynchronousScrollLayerPositionUpdates(bool forceSynchronousScrollLayerPositionUpdates)
337{
338 if (m_forceSynchronousScrollLayerPositionUpdates == forceSynchronousScrollLayerPositionUpdates)
339 return;
340
341 m_forceSynchronousScrollLayerPositionUpdates = forceSynchronousScrollLayerPositionUpdates;
342 updateSynchronousScrollingReasonsForAllFrames();
343}

345bool ScrollingCoordinator::shouldUpdateScrollLayerPositionSynchronously(const FrameView& frameView) const
346{
347 if (&frameView == m_page->mainFrame().view())
348 return !synchronousScrollingReasons(frameView.scrollingNodeID()).isEmpty();
349
350 return true;

369String ScrollingCoordinator::synchronousScrollingReasonsAsText(OptionSet<SynchronousScrollingReason> reasons)
370{
371 auto string = makeString(reasons.contains(SynchronousScrollingReason::ForcedOnMainThread) ? "Forced on main thread, " : "",
372 reasons.contains(SynchronousScrollingReason::HasSlowRepaintObjects) ? "Has slow repaint objects, " : "",
373 reasons.contains(SynchronousScrollingReason::HasViewportConstrainedObjectsWithoutSupportingFixedLayers) ? "Has viewport constrained objects without supporting fixed layers, " : "",
374 reasons.contains(SynchronousScrollingReason::HasNonLayerViewportConstrainedObjects) ? "Has non-layer viewport-constrained objects, " : "",
375 reasons.contains(SynchronousScrollingReason::IsImageDocument) ? "Is image document, " : "",
376 reasons.contains(SynchronousScrollingReason::DescendantScrollersHaveSynchronousScrolling) ? "Has slow repaint descendant scrollers, " : "");
377 return string.isEmpty() ? string : string.left(string.length() - 2);
378}

388TextStream& operator<<(TextStream& ts, ScrollableAreaParameters scrollableAreaParameters)
389{
390 ts.dumpProperty("horizontal scroll elasticity", scrollableAreaParameters.horizontalScrollElasticity);
391 ts.dumpProperty("vertical scroll elasticity", scrollableAreaParameters.verticalScrollElasticity);
392 ts.dumpProperty("horizontal scrollbar mode", scrollableAreaParameters.horizontalScrollbarMode);
393 ts.dumpProperty("vertical scrollbar mode", scrollableAreaParameters.verticalScrollbarMode);
394
395 if (scrollableAreaParameters.allowsHorizontalScrolling)
396 ts.dumpProperty("allows horizontal scrolling", scrollableAreaParameters.allowsHorizontalScrolling);
397 if (scrollableAreaParameters.allowsVerticalScrolling)
398 ts.dumpProperty("allows vertical scrolling", scrollableAreaParameters.allowsVerticalScrolling);
399
400 if (scrollableAreaParameters.horizontalScrollbarHiddenByStyle)
401 ts.dumpProperty("horizontal scrollbar hidden by style", scrollableAreaParameters.horizontalScrollbarHiddenByStyle);
402 if (scrollableAreaParameters.verticalScrollbarHiddenByStyle)
403 ts.dumpProperty("vertical scrollbar hidden by style", scrollableAreaParameters.verticalScrollbarHiddenByStyle);
404
405 return ts;
406}
