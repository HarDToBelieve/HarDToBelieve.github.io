70void ScrollingStateFixedNode::updateConstraints(const FixedPositionViewportConstraints& constraints)
71{
72 if (m_constraints == constraints)
73 return;
74
75 LOG_WITH_STREAM(Scrolling, stream << "ScrollingStateFixedNode " << scrollingNodeID() << " updateConstraints with viewport rect " << constraints.viewportRectAtLastLayout() << " layer pos at last layout " << constraints.layerPositionAtLastLayout() << " offset from top " << (constraints.layerPositionAtLastLayout().y() - constraints.viewportRectAtLastLayout().y()));
76
77 m_constraints = constraints;
78 setPropertyChanged(Property::ViewportConstraints);
79}

81void ScrollingStateFixedNode::reconcileLayerPositionForViewportRect(const LayoutRect& viewportRect, ScrollingLayerPositionAction action)
82{
83 FloatPoint position = m_constraints.layerPositionForViewportRect(viewportRect);
84 if (layer().representsGraphicsLayer()) {
85 auto* graphicsLayer = static_cast<GraphicsLayer*>(layer());
86
87 LOG_WITH_STREAM(Scrolling, stream << "ScrollingStateFixedNode " << scrollingNodeID() <<" reconcileLayerPositionForViewportRect " << action << " position of layer " << graphicsLayer->primaryLayerID() << " to " << position);
88
89 switch (action) {
90 case ScrollingLayerPositionAction::Set:
91 graphicsLayer->setPosition(position);
92 break;
93
94 case ScrollingLayerPositionAction::SetApproximate:
95 graphicsLayer->setApproximatePosition(position);
96 break;
97
98 case ScrollingLayerPositionAction::Sync:
99 graphicsLayer->syncPosition(position);
100 break;
101 }
102 }
103}

105void ScrollingStateFixedNode::dumpProperties(TextStream& ts, ScrollingStateTreeAsTextBehavior behavior) const
106{
107 ts << "Fixed node";
108 ScrollingStateNode::dumpProperties(ts, behavior);
109
110 if (m_constraints.anchorEdges()) {
111 TextStream::GroupScope scope(ts);
112 ts << "anchor edges: ";
113 if (m_constraints.hasAnchorEdge(ViewportConstraints::AnchorEdgeLeft))
114 ts << "AnchorEdgeLeft ";
115 if (m_constraints.hasAnchorEdge(ViewportConstraints::AnchorEdgeRight))
116 ts << "AnchorEdgeRight ";
117 if (m_constraints.hasAnchorEdge(ViewportConstraints::AnchorEdgeTop))
118 ts << "AnchorEdgeTop";
119 if (m_constraints.hasAnchorEdge(ViewportConstraints::AnchorEdgeBottom))
120 ts << "AnchorEdgeBottom";
121 }
122
123 if (!m_constraints.alignmentOffset().isEmpty())
124 ts.dumpProperty("alignment offset", m_constraints.alignmentOffset());
125
126 if (!m_constraints.viewportRectAtLastLayout().isEmpty())
127 ts.dumpProperty("viewport rect at last layout", m_constraints.viewportRectAtLastLayout());
128
129 if (m_constraints.layerPositionAtLastLayout() != FloatPoint())
130 ts.dumpProperty("layer position at last layout", m_constraints.layerPositionAtLastLayout());
131}
