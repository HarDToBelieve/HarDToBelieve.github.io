39ScrollingStateNode::ScrollingStateNode(ScrollingNodeType nodeType, ScrollingStateTree& scrollingStateTree, ScrollingNodeID nodeID)
40 : m_nodeType(nodeType)
41 , m_nodeID(nodeID)
42 , m_scrollingStateTree(scrollingStateTree)

44}

46// This copy constructor is used for cloning nodes in the tree, and it doesn't make sense
47// to clone the relationship pointers, so don't copy that information from the original node.
48ScrollingStateNode::ScrollingStateNode(const ScrollingStateNode& stateNode, ScrollingStateTree& adoptiveTree)
49 : m_nodeType(stateNode.nodeType())
50 , m_nodeID(stateNode.scrollingNodeID())
51 , m_changedProperties(stateNode.changedProperties())
52 , m_scrollingStateTree(adoptiveTree)
53{
54 if (hasChangedProperty(Property::Layer))
55 setLayer(stateNode.layer().toRepresentation(adoptiveTree.preferredLayerRepresentation()));
56
57 scrollingStateTree().addNode(*this);
58}

62void ScrollingStateNode::setPropertyChanged(Property property)
63{
64 if (hasChangedProperty(property))
65 return;
66
67 setPropertyChangedInternal(property);
68 m_scrollingStateTree.setHasChangedProperties();

83Ref<ScrollingStateNode> ScrollingStateNode::cloneAndReset(ScrollingStateTree& adoptiveTree)
84{
85 auto clone = this->clone(adoptiveTree);
86
87 // Now that this node is cloned, reset our change properties.
88 resetChangedProperties();
89
90 cloneAndResetChildren(clone.get(), adoptiveTree);
91
92 return clone;
93}
94
95void ScrollingStateNode::cloneAndResetChildren(ScrollingStateNode& clone, ScrollingStateTree& adoptiveTree)
96{
97 if (!m_children)
98 return;
99
100 for (auto& child : *m_children)
101 clone.appendChild(child->cloneAndReset(adoptiveTree));
102}
103
104void ScrollingStateNode::appendChild(Ref<ScrollingStateNode>&& childNode)
105{
106 childNode->setParent(this);
107
108 if (!m_children)
109 m_children = makeUnique<Vector<RefPtr<ScrollingStateNode>>>();
110 m_children->append(WTFMove(childNode));
111 setPropertyChanged(Property::ChildNodes);
112}
113
114void ScrollingStateNode::insertChild(Ref<ScrollingStateNode>&& childNode, size_t index)
115{
116 childNode->setParent(this);
117
118 if (!m_children) {
119 ASSERT(!index);
120 m_children = makeUnique<Vector<RefPtr<ScrollingStateNode>>>();
121 }
122
123 if (index > m_children->size()) {
124 ASSERT_NOT_REACHED(); // Crash data suggest we can get here.
125 m_children->append(WTFMove(childNode));
126 } else
127 m_children->insert(index, WTFMove(childNode));
128
129 setPropertyChanged(Property::ChildNodes);
130}
131
132void ScrollingStateNode::removeFromParent()
133{
134 if (!m_parent)
135 return;
136
137 m_parent->removeChild(*this);
138 m_parent = nullptr;
139}
140
141void ScrollingStateNode::removeChild(ScrollingStateNode& childNode)
142{
143 auto childIndex = indexOfChild(childNode);
144 if (childIndex != notFound)
145 removeChildAtIndex(childIndex);
146}
147
148void ScrollingStateNode::removeChildAtIndex(size_t index)
149{
150 ASSERT(m_children && index < m_children->size());
151 if (m_children && index < m_children->size()) {
152 m_children->remove(index);
153 setPropertyChanged(Property::ChildNodes);
154 }
155}
156
157size_t ScrollingStateNode::indexOfChild(ScrollingStateNode& childNode) const
158{
159 if (!m_children)
160 return notFound;
161
162 return m_children->find(&childNode);
163}

165void ScrollingStateNode::setLayer(const LayerRepresentation& layerRepresentation)
166{
167 if (layerRepresentation == m_layer)
168 return;
169
170 m_layer = layerRepresentation;
171
172 setPropertyChanged(Property::Layer);
173}

175void ScrollingStateNode::dumpProperties(TextStream& ts, ScrollingStateTreeAsTextBehavior behavior) const
176{
177 if (behavior & ScrollingStateTreeAsTextBehaviorIncludeNodeIDs)
178 ts.dumpProperty("nodeID", scrollingNodeID());
179
180 if (behavior & ScrollingStateTreeAsTextBehaviorIncludeLayerIDs)
181 ts.dumpProperty("layerID", layer().layerID());
182}
183
184void ScrollingStateNode::dump(TextStream& ts, ScrollingStateTreeAsTextBehavior behavior) const
185{
186 ts << "\n";
187 ts << indent << "(";
188 ts.increaseIndent();
189 dumpProperties(ts, behavior);
190
191 if (m_children) {
192 ts << "\n";
193 ts << indent <<"(";
194 {
195 TextStream::IndentScope indentScope(ts);
196 ts << "children " << children()->size();
197 for (auto& child : *m_children)
198 child->dump(ts, behavior);
199 ts << "\n";
200 }
201 ts << indent << ")";
202 }
203 ts << "\n";
204 ts.decreaseIndent();
205 ts << indent << ")";
206}
207
208String ScrollingStateNode::scrollingStateTreeAsText(ScrollingStateTreeAsTextBehavior behavior) const
209{
210 TextStream ts(TextStream::LineMode::MultipleLine, TextStream::Formatting::SVGStyleRect);
211
212 dump(ts, behavior);
213 ts << "\n";
214 return ts.release();
215}
