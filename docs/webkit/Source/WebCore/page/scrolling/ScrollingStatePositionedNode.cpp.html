38Ref<ScrollingStatePositionedNode> ScrollingStatePositionedNode::create(ScrollingStateTree& stateTree, ScrollingNodeID nodeID)
39{
40 return adoptRef(*new ScrollingStatePositionedNode(stateTree, nodeID));
41}

43ScrollingStatePositionedNode::ScrollingStatePositionedNode(ScrollingStateTree& tree, ScrollingNodeID nodeID)
44 : ScrollingStateNode(ScrollingNodeType::Positioned, tree, nodeID)
45{
46}

48ScrollingStatePositionedNode::ScrollingStatePositionedNode(const ScrollingStatePositionedNode& node, ScrollingStateTree& adoptiveTree)
49 : ScrollingStateNode(node, adoptiveTree)
50 , m_relatedOverflowScrollingNodes(node.relatedOverflowScrollingNodes())
51 , m_constraints(node.layoutConstraints())
52{
53}

55ScrollingStatePositionedNode::~ScrollingStatePositionedNode() = default;

57Ref<ScrollingStateNode> ScrollingStatePositionedNode::clone(ScrollingStateTree& adoptiveTree)
58{
59 return adoptRef(*new ScrollingStatePositionedNode(*this, adoptiveTree));

62OptionSet<ScrollingStateNode::Property> ScrollingStatePositionedNode::applicableProperties() const
63{
64 constexpr OptionSet<Property> nodeProperties = { Property::RelatedOverflowScrollingNodes, Property::LayoutConstraintData };
65
66 auto properties = ScrollingStateNode::applicableProperties();
67 properties.add(nodeProperties);
68 return properties;
69}
70
71void ScrollingStatePositionedNode::setRelatedOverflowScrollingNodes(Vector<ScrollingNodeID>&& nodes)
72{
73 if (nodes == m_relatedOverflowScrollingNodes)
74 return;
75
76 m_relatedOverflowScrollingNodes = WTFMove(nodes);
77 setPropertyChanged(Property::RelatedOverflowScrollingNodes);
78}
79
80void ScrollingStatePositionedNode::updateConstraints(const AbsolutePositionConstraints& constraints)
81{
82 if (m_constraints == constraints)
83 return;
84
85 LOG_WITH_STREAM(Scrolling, stream << "ScrollingStatePositionedNode " << scrollingNodeID() << " updateConstraints " << constraints);
86
87 m_constraints = constraints;
88 setPropertyChanged(Property::LayoutConstraintData);
89}
90
91void ScrollingStatePositionedNode::dumpProperties(TextStream& ts, ScrollingStateTreeAsTextBehavior behavior) const
92{
93 ts << "Positioned node";
94 ScrollingStateNode::dumpProperties(ts, behavior);
95
96 ts.dumpProperty("layout constraints", m_constraints);
97 ts.dumpProperty("related overflow nodes", m_relatedOverflowScrollingNodes.size());
98
99 if (behavior & ScrollingStateTreeAsTextBehaviorIncludeNodeIDs) {
100 if (!m_relatedOverflowScrollingNodes.isEmpty()) {
101 TextStream::GroupScope scope(ts);
102 ts << "overflow nodes";
103 for (auto nodeID : m_relatedOverflowScrollingNodes)
104 ts << "\n" << indent << "nodeID " << nodeID;
105 }
106 }
107}
