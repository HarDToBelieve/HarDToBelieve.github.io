34class PlatformWheelEvent;

37class WEBCORE_EXPORT ScrollingTreeFrameScrollingNode : public ScrollingTreeScrollingNode {
38public:
39 virtual ~ScrollingTreeFrameScrollingNode();
40
41 void commitStateBeforeChildren(const ScrollingStateNode&) override;
42
43 bool fixedElementsLayoutRelativeToFrame() const { return m_fixedElementsLayoutRelativeToFrame; }
44 bool visualViewportIsSmallerThanLayoutViewport() const { return m_visualViewportIsSmallerThanLayoutViewport; }
45
46 FloatSize viewToContentsOffset(const FloatPoint& scrollPosition) const;
47
48 FloatRect layoutViewport() const { return m_layoutViewport; };
49 void setLayoutViewport(const FloatRect& r) { m_layoutViewport = r; };
50
51 FloatRect layoutViewportRespectingRubberBanding() const;
52
53 float frameScaleFactor() const { return m_frameScaleFactor; }
54
55protected:
56 ScrollingTreeFrameScrollingNode(ScrollingTree&, ScrollingNodeType, ScrollingNodeID);
57
58 int headerHeight() const { return m_headerHeight; }
59 int footerHeight() const { return m_footerHeight; }
60 float topContentInset() const { return m_topContentInset; }
61
62 FloatPoint minLayoutViewportOrigin() const { return m_minLayoutViewportOrigin; }
63 FloatPoint maxLayoutViewportOrigin() const { return m_maxLayoutViewportOrigin; }
64
65 ScrollBehaviorForFixedElements scrollBehaviorForFixedElements() const { return m_behaviorForFixed; }
66
67private:
68 void updateViewportForCurrentScrollPosition(std::optional<FloatRect>) override;
69 bool scrollPositionAndLayoutViewportMatch(const FloatPoint& position, std::optional<FloatRect> overrideLayoutViewport) override;
70 FloatRect layoutViewportForScrollPosition(const FloatPoint&, float scale, ScrollBehaviorForFixedElements = StickToDocumentBounds) const;
71
72 void dumpProperties(WTF::TextStream&, ScrollingStateTreeAsTextBehavior) const override;
73
74 FloatRect m_layoutViewport;
75 FloatPoint m_minLayoutViewportOrigin;
76 FloatPoint m_maxLayoutViewportOrigin;
77 std::optional<FloatSize> m_overrideVisualViewportSize;
78
79 float m_frameScaleFactor { 1 };
80 float m_topContentInset { 0 };
81
82 int m_headerHeight { 0 };
83 int m_footerHeight { 0 };
84
85 ScrollBehaviorForFixedElements m_behaviorForFixed { StickToDocumentBounds };
86
87 bool m_fixedElementsLayoutRelativeToFrame { false };
88 bool m_visualViewportIsSmallerThanLayoutViewport { false };
89};

91} // namespace WebCore
