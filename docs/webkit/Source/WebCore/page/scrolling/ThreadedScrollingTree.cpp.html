67bool ThreadedScrollingTree::handleWheelEventAfterMainThread(const PlatformWheelEvent& wheelEvent, ScrollingNodeID targetNodeID, std::optional<WheelScrollGestureState> gestureState)
68{
69 ASSERT(ScrollingThread::isCurrentThread());
70
71 LOG_WITH_STREAM(Scrolling, stream << "ThreadedScrollingTree::handleWheelEventAfterMainThread " << wheelEvent << " node " << targetNodeID << " gestureState " << gestureState);
72
73 Locker locker { m_treeLock };
74
75 bool allowLatching = false;
76 OptionSet<WheelEventProcessingSteps> processingSteps;
77 if (gestureState.value_or(WheelScrollGestureState::Blocking) == WheelScrollGestureState::NonBlocking) {
78 allowLatching = true;
79 processingSteps = { WheelEventProcessingSteps::ScrollingThread, WheelEventProcessingSteps::MainThreadForNonBlockingDOMEventDispatch };
80 }
81
82 SetForScope<bool> disallowLatchingScope(m_allowLatching, allowLatching);
83 RefPtr<ScrollingTreeNode> targetNode = nodeForID(targetNodeID);
84 auto result = handleWheelEventWithNode(wheelEvent, processingSteps, targetNode.get(), EventTargeting::NodeOnly);
85 return result.wasHandled;
86}

88void ThreadedScrollingTree::wheelEventWasProcessedByMainThread(const PlatformWheelEvent& wheelEvent, std::optional<WheelScrollGestureState> gestureState)
89{
90 LOG_WITH_STREAM(Scrolling, stream << "ThreadedScrollingTree::wheelEventWasProcessedByMainThread - gestureState " << gestureState);
91
92 ASSERT(isMainThread());
93
94 Locker locker { m_treeLock };
95 if (m_receivedBeganEventFromMainThread || !wheelEvent.isGestureStart())
96 return;
97
98 setGestureState(gestureState);
99
100 m_receivedBeganEventFromMainThread = true;
101 m_waitingForBeganEventCondition.notifyOne();
102}

112void ThreadedScrollingTree::waitForEventToBeProcessedByMainThread(const PlatformWheelEvent& wheelEvent)
113{
114 ASSERT(ScrollingThread::isCurrentThread());
115
116 if (!wheelEvent.isGestureStart())
117 return;
118
119 Locker locker { m_treeLock };
120
121 static constexpr auto maxAllowableMainThreadDelay = 50_ms;
122 auto startTime = MonotonicTime::now();
123 auto timeoutTime = startTime + maxAllowableMainThreadDelay;
124
125 bool receivedEvent = m_waitingForBeganEventCondition.waitUntil(m_treeLock, timeoutTime, [&] {
126 assertIsHeld(m_treeLock);
127 return m_receivedBeganEventFromMainThread;
128 });
129
130 if (!receivedEvent) {
131 // Timed out, go asynchronous.
132 setGestureState(WheelScrollGestureState::NonBlocking);
133 }
134
135 LOG_WITH_STREAM(Scrolling, stream << "ThreadedScrollingTree::waitForBeganEventFromMainThread done - timed out " << !receivedEvent << " gesture state is " << gestureState());
136}

138void ThreadedScrollingTree::invalidate()
139{
140 // Invalidate is dispatched by the ScrollingCoordinator class on the ScrollingThread
141 // to break the reference cycle between ScrollingTree and ScrollingCoordinator when the
142 // ScrollingCoordinator's page is destroyed.
143 ASSERT(ScrollingThread::isCurrentThread());
144
145 Locker locker { m_treeLock };
146
147 removeAllNodes();
148 m_delayedRenderingUpdateDetectionTimer = nullptr;
149
150 // Since this can potentially be the last reference to the scrolling coordinator,
151 // we need to release it on the main thread since it has member variables (such as timers)
152 // that expect to be destroyed from the main thread.
153 RunLoop::main().dispatch([scrollingCoordinator = WTFMove(m_scrollingCoordinator)] {
154 });
155}

157void ThreadedScrollingTree::propagateSynchronousScrollingReasons(const HashSet<ScrollingNodeID>& synchronousScrollingNodes)
158{
159 auto propagateStateToAncestors = [&](ScrollingTreeNode& node) {
160 ASSERT(is<ScrollingTreeScrollingNode>(node) && !downcast<ScrollingTreeScrollingNode>(node).synchronousScrollingReasons().isEmpty());
161
162 if (is<ScrollingTreeFrameScrollingNode>(node))
163 return;
164
165 auto currNode = node.parent();
166
167 while (currNode) {
168 if (is<ScrollingTreeScrollingNode>(currNode))
169 downcast<ScrollingTreeScrollingNode>(*currNode).addSynchronousScrollingReason(SynchronousScrollingReason::DescendantScrollersHaveSynchronousScrolling);
170
171 if (is<ScrollingTreeOverflowScrollProxyNode>(currNode)) {
172 currNode = nodeForID(downcast<ScrollingTreeOverflowScrollProxyNode>(*currNode).overflowScrollingNodeID());
173 continue;
174 }
175
176 if (is<ScrollingTreeFrameScrollingNode>(currNode))
177 break;
178
179 currNode = currNode->parent();
180 }
181 };
182
183 m_hasNodesWithSynchronousScrollingReasons = !synchronousScrollingNodes.isEmpty();
184
185 for (auto nodeID : synchronousScrollingNodes) {
186 if (auto node = nodeForID(nodeID))
187 propagateStateToAncestors(*node);
188 }
189}

191bool ThreadedScrollingTree::canUpdateLayersOnScrollingThread() const

196void ThreadedScrollingTree::scrollingTreeNodeDidScroll(ScrollingTreeScrollingNode& node, ScrollingLayerPositionAction scrollingLayerPositionAction)
197{
198 if (!m_scrollingCoordinator)
199 return;
200
201 auto scrollPosition = node.currentScrollPosition();
202 if (node.isRootNode())
203 setMainFrameScrollPosition(scrollPosition);
204
205 if (isHandlingProgrammaticScroll())
206 return;
207
208 std::optional<FloatPoint> layoutViewportOrigin;
209 if (is<ScrollingTreeFrameScrollingNode>(node))
210 layoutViewportOrigin = downcast<ScrollingTreeFrameScrollingNode>(node).layoutViewport().location();
211
212 if (RunLoop::isMain()) {
213 m_scrollingCoordinator->applyScrollUpdate(node.scrollingNodeID(), scrollPosition, layoutViewportOrigin, ScrollType::User, scrollingLayerPositionAction);
214 return;
215 }
216
217 LOG_WITH_STREAM(Scrolling, stream << "ThreadedScrollingTree::scrollingTreeNodeDidScroll " << node.scrollingNodeID() << " to " << scrollPosition << " triggering main thread rendering update");
218
219 auto scrollUpdate = ScrollUpdate { node.scrollingNodeID(), scrollPosition, layoutViewportOrigin, scrollingLayerPositionAction };
220 addPendingScrollUpdate(WTFMove(scrollUpdate));
221
222 auto deferrer = WheelEventTestMonitorCompletionDeferrer { wheelEventTestMonitor(), reinterpret_cast<WheelEventTestMonitor::ScrollableAreaIdentifier>(node.scrollingNodeID()), WheelEventTestMonitor::ScrollingThreadSyncNeeded };
223 RunLoop::main().dispatch([strongThis = makeRef(*this), deferrer = WTFMove(deferrer)] {
224 if (auto* scrollingCoordinator = strongThis->m_scrollingCoordinator.get())
225 scrollingCoordinator->scrollingThreadAddedPendingUpdate();
226 });
227}

277void ThreadedScrollingTree::willStartRenderingUpdate()
278{
279 ASSERT(isMainThread());
280
281 if (!hasProcessedWheelEventsRecently())
282 return;
283
284 tracePoint(ScrollingThreadRenderUpdateSyncStart);
285
286 // Wait for the scrolling thread to acquire m_treeLock. This ensures that any pending wheel events are processed.
287 BinarySemaphore semaphore;
288 ScrollingThread::dispatch([protectedThis = makeRef(*this), &semaphore]() {
289 Locker treeLocker { protectedThis->m_treeLock };
290 semaphore.signal();
291 protectedThis->waitForRenderingUpdateCompletionOrTimeout();
292 });
293 semaphore.wait();
294
295 Locker locker { m_treeLock };
296 m_state = SynchronizationState::InRenderingUpdate;
297}

299Seconds ThreadedScrollingTree::maxAllowableRenderingUpdateDurationForSynchronization()
300{
301 constexpr double allowableFrameFraction = 0.5;
302 auto displayFPS = nominalFramesPerSecond().value_or(60);
303 Seconds frameDuration = 1_s / (double)displayFPS;
304 return allowableFrameFraction * frameDuration;
305}
306
307// This code allows the main thread about half a frame to complete its rendering udpate. If the main thread
308// is responsive (i.e. managing to render every frame), then we expect to get a didCompleteRenderingUpdate()
309// within 8ms of willStartRenderingUpdate(). We time this via m_stateCondition, which blocks the scrolling
310// thread (with m_treeLock locked at the start and end) so that we don't handle wheel events while waiting.
311// If the condition times out, we know the main thread is being slow, and allow the scrolling thread to
312// commit layer positions.
313void ThreadedScrollingTree::waitForRenderingUpdateCompletionOrTimeout()
314{
315 ASSERT(ScrollingThread::isCurrentThread());
316 ASSERT(m_treeLock.isLocked());
317
318 if (m_delayedRenderingUpdateDetectionTimer)
319 m_delayedRenderingUpdateDetectionTimer->stop();
320
321 auto startTime = MonotonicTime::now();
322 auto timeoutTime = startTime + maxAllowableRenderingUpdateDurationForSynchronization();
323
324 bool becameIdle = m_stateCondition.waitUntil(m_treeLock, timeoutTime, [&] {
325 assertIsHeld(m_treeLock);
326 return m_state == SynchronizationState::Idle;
327 });
328
329 ASSERT(m_treeLock.isLocked());
330
331 if (!becameIdle) {
332 m_state = SynchronizationState::Desynchronized;
333 // At this point we know the main thread is taking too long in the rendering update,
334 // so we give up trying to sync with the main thread and update layers here on the scrolling thread.
335 if (canUpdateLayersOnScrollingThread())
336 applyLayerPositionsInternal();
337 tracePoint(ScrollingThreadRenderUpdateSyncEnd, 1);
338 } else
339 tracePoint(ScrollingThreadRenderUpdateSyncEnd);
340}
341
342void ThreadedScrollingTree::didCompleteRenderingUpdate()
343{
344 ASSERT(isMainThread());
345 Locker locker { m_treeLock };
346
347 if (m_state == SynchronizationState::InRenderingUpdate)
348 m_stateCondition.notifyOne();
349
350 m_state = SynchronizationState::Idle;
351}

353void ThreadedScrollingTree::scheduleDelayedRenderingUpdateDetectionTimer(Seconds delay)
354{
355 ASSERT(ScrollingThread::isCurrentThread());
356 ASSERT(m_treeLock.isLocked());
357
358 if (!m_delayedRenderingUpdateDetectionTimer)
359 m_delayedRenderingUpdateDetectionTimer = makeUnique<RunLoop::Timer<ThreadedScrollingTree>>(RunLoop::current(), this, &ThreadedScrollingTree::delayedRenderingUpdateDetectionTimerFired);
360
361 m_delayedRenderingUpdateDetectionTimer->startOneShot(delay);
362}
363
364void ThreadedScrollingTree::delayedRenderingUpdateDetectionTimerFired()
365{
366 ASSERT(ScrollingThread::isCurrentThread());
367
368 Locker locker { m_treeLock };
369 if (canUpdateLayersOnScrollingThread())
370 applyLayerPositionsInternal();
371 m_state = SynchronizationState::Desynchronized;
372}

374void ThreadedScrollingTree::displayDidRefreshOnScrollingThread()
375{
376 TraceScope tracingScope(ScrollingThreadDisplayDidRefreshStart, ScrollingThreadDisplayDidRefreshEnd, displayID());
377 ASSERT(ScrollingThread::isCurrentThread());
378
379 Locker locker { m_treeLock };
380
381 if (m_state != SynchronizationState::Idle && canUpdateLayersOnScrollingThread())
382 applyLayerPositionsInternal();
383
384 switch (m_state) {
385 case SynchronizationState::Idle: {
386 m_state = SynchronizationState::WaitingForRenderingUpdate;
387 constexpr auto maxStartRenderingUpdateDelay = 1_ms;
388 scheduleDelayedRenderingUpdateDetectionTimer(maxStartRenderingUpdateDelay);
389 break;
390 }
391 case SynchronizationState::WaitingForRenderingUpdate:
392 case SynchronizationState::InRenderingUpdate:
393 case SynchronizationState::Desynchronized:
394 break;
395 }
396}

398void ThreadedScrollingTree::displayDidRefresh(PlatformDisplayID displayID)
399{
400 bool hasProcessedWheelEventsRecently = this->hasProcessedWheelEventsRecently();
401
402 // We're on the EventDispatcher thread or in the ThreadedCompositor thread here.
403 tracePoint(ScrollingTreeDisplayDidRefresh, displayID, hasProcessedWheelEventsRecently);
404
405 if (displayID != this->displayID())
406 return;
407
408#if !PLATFORM(WPE) && !PLATFORM(GTK)
409 if (!hasProcessedWheelEventsRecently)
410 return;
411#endif
412
413 ScrollingThread::dispatch([protectedThis = makeRef(*this)]() {
414 protectedThis->displayDidRefreshOnScrollingThread();
415 });
416}
