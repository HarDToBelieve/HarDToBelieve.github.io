44Ref<ScrollingTreeFixedNode> ScrollingTreeFixedNode::create(ScrollingTree& scrollingTree, ScrollingNodeID nodeID)

46 return adoptRef(*new ScrollingTreeFixedNode(scrollingTree, nodeID));

49ScrollingTreeFixedNode::ScrollingTreeFixedNode(ScrollingTree& scrollingTree, ScrollingNodeID nodeID)
50 : ScrollingTreeNode(scrollingTree, ScrollingNodeType::Fixed, nodeID)
51{
52 scrollingTree.fixedOrStickyNodeAdded();
53}

55ScrollingTreeFixedNode::~ScrollingTreeFixedNode()
56{
57 scrollingTree().fixedOrStickyNodeRemoved();
58}

60void ScrollingTreeFixedNode::commitStateBeforeChildren(const ScrollingStateNode& stateNode)
61{
62 const ScrollingStateFixedNode& fixedStateNode = downcast<ScrollingStateFixedNode>(stateNode);
63
64 if (fixedStateNode.hasChangedProperty(ScrollingStateNode::Property::Layer))
65 m_layer = static_cast<CALayer*>(fixedStateNode.layer());
66
67 if (stateNode.hasChangedProperty(ScrollingStateNode::Property::ViewportConstraints))
68 m_constraints = fixedStateNode.viewportConstraints();
69}

71void ScrollingTreeFixedNode::applyLayerPositions()
72{
73 auto computeLayerPosition = [&] {
74 FloatSize overflowScrollDelta;
75 ScrollingTreeStickyNode* lastStickyNode = nullptr;
76 for (auto* ancestor = parent(); ancestor; ancestor = ancestor->parent()) {
77 if (is<ScrollingTreeFrameScrollingNode>(*ancestor)) {
78 // Fixed nodes are positioned relative to the containing frame scrolling node.
79 // We bail out after finding one.
80 auto layoutViewport = downcast<ScrollingTreeFrameScrollingNode>(*ancestor).layoutViewport();
81 return m_constraints.layerPositionForViewportRect(layoutViewport) - overflowScrollDelta;
82 }
83
84 if (is<ScrollingTreeOverflowScrollingNode>(*ancestor)) {
85 // To keep the layer still during async scrolling we adjust by how much the position has changed since layout.
86 auto& overflowNode = downcast<ScrollingTreeOverflowScrollingNode>(*ancestor);
87 overflowScrollDelta -= overflowNode.scrollDeltaSinceLastCommit();
88 continue;
89 }
90
91 if (is<ScrollingTreeOverflowScrollProxyNode>(*ancestor)) {
92 // To keep the layer still during async scrolling we adjust by how much the position has changed since layout.
93 auto& overflowNode = downcast<ScrollingTreeOverflowScrollProxyNode>(*ancestor);
94 overflowScrollDelta -= overflowNode.scrollDeltaSinceLastCommit();
95 continue;
96 }
97
98 if (is<ScrollingTreePositionedNode>(*ancestor)) {
99 auto& positioningAncestor = downcast<ScrollingTreePositionedNode>(*ancestor);
100 // See if sticky node already handled this positioning node.
101 // FIXME: Include positioning node information to sticky/fixed node to avoid these tests.
102 if (lastStickyNode && lastStickyNode->layer() == positioningAncestor.layer())
103 continue;
104 if (positioningAncestor.layer() != m_layer)
105 overflowScrollDelta -= positioningAncestor.scrollDeltaSinceLastCommit();
106 continue;
107 }
108
109 if (is<ScrollingTreeStickyNode>(*ancestor)) {
110 auto& stickyNode = downcast<ScrollingTreeStickyNode>(*ancestor);
111 overflowScrollDelta += stickyNode.scrollDeltaSinceLastCommit();
112 lastStickyNode = &stickyNode;
113 continue;
114 }
115
116 if (is<ScrollingTreeFixedNode>(*ancestor)) {
117 // The ancestor fixed node has already applied the needed corrections to say put.
118 return m_constraints.layerPositionAtLastLayout() - overflowScrollDelta;
119 }
120 }
121 ASSERT_NOT_REACHED();
122 return FloatPoint();
123 };
124
125 auto layerPosition = computeLayerPosition();
126
127 LOG_WITH_STREAM(Scrolling, stream << "ScrollingTreeFixedNode " << scrollingNodeID() << " relatedNodeScrollPositionDidChange: viewportRectAtLastLayout " << m_constraints.viewportRectAtLastLayout() << " last layer pos " << m_constraints.layerPositionAtLastLayout() << " layerPosition " << layerPosition);
128
129 [m_layer _web_setLayerTopLeftPosition:layerPosition - m_constraints.alignmentOffset()];
130}

132void ScrollingTreeFixedNode::dumpProperties(TextStream& ts, ScrollingStateTreeAsTextBehavior behavior) const
133{
134 ts << "fixed node";
135 ScrollingTreeNode::dumpProperties(ts, behavior);
136 ts.dumpProperty("fixed constraints", m_constraints);
137
138 if (behavior & ScrollingStateTreeAsTextBehaviorIncludeLayerPositions) {
139 FloatRect layerBounds = [m_layer bounds];
140 FloatPoint anchorPoint = [m_layer anchorPoint];
141 FloatPoint position = [m_layer position];
142 FloatPoint layerTopLeft = position - toFloatSize(anchorPoint) * layerBounds.size() + m_constraints.alignmentOffset();
143 ts.dumpProperty("layer top left", layerTopLeft);
144 }
145}
