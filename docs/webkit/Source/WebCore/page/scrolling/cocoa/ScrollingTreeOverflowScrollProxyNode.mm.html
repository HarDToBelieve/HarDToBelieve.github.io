50void ScrollingTreeOverflowScrollProxyNode::commitStateBeforeChildren(const ScrollingStateNode& stateNode)
51{
52 const ScrollingStateOverflowScrollProxyNode& overflowProxyStateNode = downcast<ScrollingStateOverflowScrollProxyNode>(stateNode);
53
54 if (stateNode.hasChangedProperty(ScrollingStateNode::Property::Layer))
55 m_layer = static_cast<CALayer*>(stateNode.layer());
56
57 if (overflowProxyStateNode.hasChangedProperty(ScrollingStateNode::Property::OverflowScrollingNode))
58 m_overflowScrollingNodeID = overflowProxyStateNode.overflowScrollingNode();
59
60 if (m_overflowScrollingNodeID) {
61 auto& relatedNodes = scrollingTree().overflowRelatedNodes();
62 relatedNodes.ensure(m_overflowScrollingNodeID, [] {
63 return Vector<ScrollingNodeID>();
64 }).iterator->value.append(scrollingNodeID());
65
66 scrollingTree().activeOverflowScrollProxyNodes().add(*this);
67 }
68}

70FloatSize ScrollingTreeOverflowScrollProxyNode::scrollDeltaSinceLastCommit() const
71{
72 if (auto* node = scrollingTree().nodeForID(m_overflowScrollingNodeID)) {
73 if (is<ScrollingTreeOverflowScrollingNode>(node))
74 return downcast<ScrollingTreeOverflowScrollingNode>(*node).scrollDeltaSinceLastCommit();
75 }
76
77 return { };
78}
79
80void ScrollingTreeOverflowScrollProxyNode::applyLayerPositions()
81{
82 FloatPoint scrollOffset;
83 if (auto* node = scrollingTree().nodeForID(m_overflowScrollingNodeID)) {
84 if (is<ScrollingTreeOverflowScrollingNode>(node)) {
85 auto& overflowNode = downcast<ScrollingTreeOverflowScrollingNode>(*node);
86 scrollOffset = overflowNode.currentScrollOffset();
87 }
88 }
89
90 LOG_WITH_STREAM(ScrollingTree, stream << "ScrollingTreeOverflowScrollProxyNode " << scrollingNodeID() << " applyLayerPositions: setting bounds origin to " << scrollOffset);
91 [m_layer _web_setLayerBoundsOrigin:scrollOffset];
92}
93
94void ScrollingTreeOverflowScrollProxyNode::dumpProperties(TextStream& ts, ScrollingStateTreeAsTextBehavior behavior) const
95{
96 ts << "overflow scroll proxy node";
97 ScrollingTreeNode::dumpProperties(ts, behavior);
98
99 if (auto* relatedOverflowNode = scrollingTree().nodeForID(m_overflowScrollingNodeID)) {
100 auto scrollPosition = downcast<ScrollingTreeOverflowScrollingNode>(*relatedOverflowNode).currentScrollPosition();
101 ts.dumpProperty("related overflow scrolling node scroll position", scrollPosition);
102 }
103
104 if (behavior & ScrollingStateTreeAsTextBehaviorIncludeNodeIDs)
105 ts.dumpProperty("overflow scrolling node", m_overflowScrollingNodeID);
106}
