53void ScrollingTreePositionedNode::commitStateBeforeChildren(const ScrollingStateNode& stateNode)
54{
55 const ScrollingStatePositionedNode& positionedStateNode = downcast<ScrollingStatePositionedNode>(stateNode);
56
57 if (positionedStateNode.hasChangedProperty(ScrollingStateNode::Property::Layer))
58 m_layer = static_cast<CALayer*>(positionedStateNode.layer());
59
60 if (positionedStateNode.hasChangedProperty(ScrollingStateNode::Property::RelatedOverflowScrollingNodes))
61 m_relatedOverflowScrollingNodes = positionedStateNode.relatedOverflowScrollingNodes();
62
63 if (positionedStateNode.hasChangedProperty(ScrollingStateNode::Property::LayoutConstraintData))
64 m_constraints = positionedStateNode.layoutConstraints();
65
66 if (!m_relatedOverflowScrollingNodes.isEmpty())
67 scrollingTree().activePositionedNodes().add(*this);
68}

70FloatSize ScrollingTreePositionedNode::scrollDeltaSinceLastCommit() const
71{
72 FloatSize delta;
73 for (auto nodeID : m_relatedOverflowScrollingNodes) {
74 if (auto* node = scrollingTree().nodeForID(nodeID)) {
75 if (is<ScrollingTreeOverflowScrollingNode>(node)) {
76 auto& overflowNode = downcast<ScrollingTreeOverflowScrollingNode>(*node);
77 delta += overflowNode.scrollDeltaSinceLastCommit();
78 }
79 }
80 }
81
82 // Positioned nodes compensate for scrolling, so negate the scroll delta.
83 return -delta;
84}
85
86void ScrollingTreePositionedNode::applyLayerPositions()
87{
88 auto delta = scrollDeltaSinceLastCommit();
89 auto layerPosition = m_constraints.layerPositionAtLastLayout() - delta;
90
91 LOG_WITH_STREAM(Scrolling, stream << "ScrollingTreePositionedNode " << scrollingNodeID() << " applyLayerPositions: overflow delta " << delta << " moving layer to " << layerPosition);
92
93 [m_layer _web_setLayerTopLeftPosition:layerPosition - m_constraints.alignmentOffset()];
94}

96void ScrollingTreePositionedNode::dumpProperties(TextStream& ts, ScrollingStateTreeAsTextBehavior behavior) const
97{
98 ts << "positioned node";
99 ScrollingTreeNode::dumpProperties(ts, behavior);
100
101 ts.dumpProperty("layout constraints", m_constraints);
102 ts.dumpProperty("related overflow nodes", m_relatedOverflowScrollingNodes.size());
103
104 if (behavior & ScrollingStateTreeAsTextBehaviorIncludeNodeIDs) {
105 if (!m_relatedOverflowScrollingNodes.isEmpty()) {
106 TextStream::GroupScope scope(ts);
107 ts << "overflow nodes";
108 for (auto nodeID : m_relatedOverflowScrollingNodes)
109 ts << "\n" << indent << "nodeID " << nodeID;
110 }
111 }
