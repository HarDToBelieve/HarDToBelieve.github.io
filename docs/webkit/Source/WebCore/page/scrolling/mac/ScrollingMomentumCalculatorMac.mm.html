37std::unique_ptr<ScrollingMomentumCalculator> ScrollingMomentumCalculator::create(const FloatSize& viewportSize, const FloatSize& contentSize, const FloatPoint& initialOffset, const FloatSize& initialDelta, const FloatSize& initialVelocity)

42void ScrollingMomentumCalculator::setPlatformMomentumScrollingPredictionEnabled(bool enabled)

44 gEnablePlatformMomentumScrollingPrediction = enabled;

47ScrollingMomentumCalculatorMac::ScrollingMomentumCalculatorMac(const FloatSize& viewportSize, const FloatSize& contentSize, const FloatPoint& initialOffset, const FloatSize& initialDelta, const FloatSize& initialVelocity)
48 : ScrollingMomentumCalculator(viewportSize, contentSize, initialOffset, initialDelta, initialVelocity)

52FloatPoint ScrollingMomentumCalculatorMac::scrollOffsetAfterElapsedTime(Seconds elapsedTime)
53{
54 if (!requiresMomentumScrolling())
55 return { retargetedScrollOffset().width(), retargetedScrollOffset().height() };
56
57 return [ensurePlatformMomentumCalculator() positionAfterDuration:elapsedTime.value()];
58}
59
60FloatSize ScrollingMomentumCalculatorMac::predictedDestinationOffset()
61{
62 if (!gEnablePlatformMomentumScrollingPrediction)
63 return ScrollingMomentumCalculator::predictedDestinationOffset();
64
65 ensurePlatformMomentumCalculator();
66 return { m_initialDestinationOrigin.x(), m_initialDestinationOrigin.y() };
67}
68
69void ScrollingMomentumCalculatorMac::retargetedScrollOffsetDidChange()
70{
71 _NSScrollingMomentumCalculator *calculator = ensurePlatformMomentumCalculator();
72 calculator.destinationOrigin = NSMakePoint(retargetedScrollOffset().width(), retargetedScrollOffset().height());
73 [calculator calculateToReachDestination];
74}
75
76Seconds ScrollingMomentumCalculatorMac::animationDuration()
77{
78 if (!requiresMomentumScrolling())
79 return 0_s;
80
81 return Seconds([ensurePlatformMomentumCalculator() durationUntilStop]);
82}
83
84bool ScrollingMomentumCalculatorMac::requiresMomentumScrolling()
85{
86 if (m_requiresMomentumScrolling == std::nullopt)
87 m_requiresMomentumScrolling = m_initialScrollOffset != retargetedScrollOffset() || m_initialVelocity.area();
88 return m_requiresMomentumScrolling.value();
89}
90
91_NSScrollingMomentumCalculator *ScrollingMomentumCalculatorMac::ensurePlatformMomentumCalculator()
92{
93 if (m_platformMomentumCalculator)
94 return m_platformMomentumCalculator.get();
95
96 NSPoint origin = NSMakePoint(m_initialScrollOffset.width(), m_initialScrollOffset.height());
97 NSRect contentFrame = NSMakeRect(0, 0, m_contentSize.width(), m_contentSize.height());
98 NSPoint velocity = NSMakePoint(m_initialVelocity.width(), m_initialVelocity.height());
99 m_platformMomentumCalculator = adoptNS([[_NSScrollingMomentumCalculator alloc] initWithInitialOrigin:origin velocity:velocity documentFrame:contentFrame constrainedClippingOrigin:NSZeroPoint clippingSize:m_viewportSize tolerance:NSMakeSize(1, 1)]);
100 m_initialDestinationOrigin = [m_platformMomentumCalculator destinationOrigin];
101 return m_platformMomentumCalculator.get();
102}
