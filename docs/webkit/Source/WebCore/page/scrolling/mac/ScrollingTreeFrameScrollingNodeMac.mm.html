66void ScrollingTreeFrameScrollingNodeMac::commitStateBeforeChildren(const ScrollingStateNode& stateNode)
67{
68 ScrollingTreeFrameScrollingNode::commitStateBeforeChildren(stateNode);
69 const auto& scrollingStateNode = downcast<ScrollingStateFrameScrollingNode>(stateNode);
70
71 if (scrollingStateNode.hasChangedProperty(ScrollingStateNode::Property::RootContentsLayer))
72 m_rootContentsLayer = static_cast<CALayer*>(scrollingStateNode.rootContentsLayer());
73
74 if (scrollingStateNode.hasChangedProperty(ScrollingStateNode::Property::CounterScrollingLayer))
75 m_counterScrollingLayer = static_cast<CALayer*>(scrollingStateNode.counterScrollingLayer());
76
77 if (scrollingStateNode.hasChangedProperty(ScrollingStateNode::Property::InsetClipLayer))
78 m_insetClipLayer = static_cast<CALayer*>(scrollingStateNode.insetClipLayer());
79
80 if (scrollingStateNode.hasChangedProperty(ScrollingStateNode::Property::ContentShadowLayer))
81 m_contentShadowLayer = static_cast<CALayer*>(scrollingStateNode.contentShadowLayer());
82
83 if (scrollingStateNode.hasChangedProperty(ScrollingStateNode::Property::HeaderLayer))
84 m_headerLayer = static_cast<CALayer*>(scrollingStateNode.headerLayer());
85
86 if (scrollingStateNode.hasChangedProperty(ScrollingStateNode::Property::FooterLayer))
87 m_footerLayer = static_cast<CALayer*>(scrollingStateNode.footerLayer());
88
89 bool logScrollingMode = !m_hadFirstUpdate;
90 if (scrollingStateNode.hasChangedProperty(ScrollingStateNode::Property::ReasonsForSynchronousScrolling))
91 logScrollingMode = true;
92
93 if (logScrollingMode && isRootNode() && scrollingTree().scrollingPerformanceTestingEnabled())
94 scrollingTree().reportSynchronousScrollingReasonsChanged(MonotonicTime::now(), synchronousScrollingReasons());
95
96 m_delegate.updateFromStateNode(scrollingStateNode);
97
98 m_hadFirstUpdate = true;
99}

101void ScrollingTreeFrameScrollingNodeMac::commitStateAfterChildren(const ScrollingStateNode& stateNode)
102{
103 ScrollingTreeFrameScrollingNode::commitStateAfterChildren(stateNode);
104
105 const auto& scrollingStateNode = downcast<ScrollingStateScrollingNode>(stateNode);
106
107 // Update the scroll position after child nodes have been updated, because they need to have updated their constraints before any scrolling happens.
108 if (scrollingStateNode.hasChangedProperty(ScrollingStateNode::Property::RequestedScrollPosition)) {
109 const auto& requestedScrollData = scrollingStateNode.requestedScrollData();
110 scrollTo(requestedScrollData.scrollPosition, requestedScrollData.scrollType, requestedScrollData.clamping);
111 }
112
113 if (isRootNode()
114 && (scrollingStateNode.hasChangedProperty(ScrollingStateNode::Property::ScrolledContentsLayer)
115 || scrollingStateNode.hasChangedProperty(ScrollingStateNode::Property::TotalContentsSize)
116 || scrollingStateNode.hasChangedProperty(ScrollingStateNode::Property::ScrollableAreaSize)))
117 updateMainFramePinAndRubberbandState();
118}

120WheelEventHandlingResult ScrollingTreeFrameScrollingNodeMac::handleWheelEvent(const PlatformWheelEvent& wheelEvent, EventTargeting eventTargeting)
121{
122 if (!canHandleWheelEvent(wheelEvent, eventTargeting))
123 return WheelEventHandlingResult::unhandled();
124
125 bool handled = m_delegate.handleWheelEvent(wheelEvent);
126
127 setScrollSnapInProgress(m_delegate.isScrollSnapInProgress());
128
129 if (m_delegate.activeScrollSnapIndexDidChange())
130 scrollingTree().setActiveScrollSnapIndices(scrollingNodeID(), m_delegate.activeScrollSnapIndexForAxis(ScrollEventAxis::Horizontal), m_delegate.activeScrollSnapIndexForAxis(ScrollEventAxis::Vertical));
131
132 return WheelEventHandlingResult::result(handled);
133}

146void ScrollingTreeFrameScrollingNodeMac::currentScrollPositionChanged(ScrollType scrollType, ScrollingLayerPositionAction action)
147{
148 LOG_WITH_STREAM(Scrolling, stream << "ScrollingTreeFrameScrollingNodeMac " << scrollingNodeID() << " currentScrollPositionChanged to " << currentScrollPosition() << " min: " << minimumScrollPosition() << " max: " << maximumScrollPosition() << " sync: " << hasSynchronousScrollingReasons());
149
150 m_delegate.currentScrollPositionChanged();
151
152 if (isRootNode())
153 updateMainFramePinAndRubberbandState();
154
155 ScrollingTreeFrameScrollingNode::currentScrollPositionChanged(scrollType, hasSynchronousScrollingReasons() ? ScrollingLayerPositionAction::Set : action);
156
157 if (scrollingTree().scrollingPerformanceTestingEnabled()) {
158 unsigned unfilledArea = exposedUnfilledArea();
159 if (unfilledArea || m_lastScrollHadUnfilledPixels)
160 scrollingTree().reportExposedUnfilledArea(MonotonicTime::now(), unfilledArea);
161
162 m_lastScrollHadUnfilledPixels = unfilledArea;
163 }
164}

166void ScrollingTreeFrameScrollingNodeMac::repositionScrollingLayers()
167{
168 BEGIN_BLOCK_OBJC_EXCEPTIONS
169 // We use scroll position here because the root content layer is offset to account for scrollOrigin (see FrameView::positionForRootContentLayer).
170 static_cast<CALayer*>(scrolledContentsLayer()).position = -currentScrollPosition();
171 END_BLOCK_OBJC_EXCEPTIONS
172}

174void ScrollingTreeFrameScrollingNodeMac::repositionRelatedLayers()
175{
176 BEGIN_BLOCK_OBJC_EXCEPTIONS
177 auto scrollPosition = currentScrollPosition();
178 auto layoutViewport = this->layoutViewport();
179
180 FloatRect visibleContentRect(scrollPosition, scrollableAreaSize());
181
182 if (m_counterScrollingLayer)
183 m_counterScrollingLayer.get().position = layoutViewport.location();
184
185 float topContentInset = this->topContentInset();
186 if (m_insetClipLayer && m_rootContentsLayer) {
187 m_insetClipLayer.get().position = FloatPoint(m_insetClipLayer.get().position.x, FrameView::yPositionForInsetClipLayer(scrollPosition, topContentInset));
188 m_rootContentsLayer.get().position = FrameView::positionForRootContentLayer(scrollPosition, scrollOrigin(), topContentInset, headerHeight());
189 if (m_contentShadowLayer)
190 m_contentShadowLayer.get().position = m_rootContentsLayer.get().position;
191 }
192
193 if (m_headerLayer || m_footerLayer) {
194 // Generally the banners should have the same horizontal-position computation as a fixed element. However,
195 // the banners are not affected by the frameScaleFactor(), so if there is currently a non-1 frameScaleFactor()
196 // then we should recompute layoutViewport.x() for the banner with a scale factor of 1.
197 float horizontalScrollOffsetForBanner = layoutViewport.x();
198 if (m_headerLayer)
199 m_headerLayer.get().position = FloatPoint(horizontalScrollOffsetForBanner, FrameView::yPositionForHeaderLayer(scrollPosition, topContentInset));
200
201 if (m_footerLayer)
202 m_footerLayer.get().position = FloatPoint(horizontalScrollOffsetForBanner, FrameView::yPositionForFooterLayer(scrollPosition, topContentInset, totalContentsSize().height(), footerHeight()));
203 }
204 END_BLOCK_OBJC_EXCEPTIONS
205
206 m_delegate.updateScrollbarPainters();
207}

209FloatPoint ScrollingTreeFrameScrollingNodeMac::minimumScrollPosition() const
210{
211 FloatPoint position = ScrollableArea::scrollPositionFromOffset(FloatPoint(), toFloatSize(scrollOrigin()));
212
213 if (isRootNode() && scrollingTree().scrollPinningBehavior() == PinToBottom)
214 position.setY(maximumScrollPosition().y());
215
216 return position;
217}
218
219FloatPoint ScrollingTreeFrameScrollingNodeMac::maximumScrollPosition() const
220{
221 FloatPoint position = ScrollableArea::scrollPositionFromOffset(FloatPoint(totalContentsSizeForRubberBand() - scrollableAreaSize()), toFloatSize(scrollOrigin()));
222 position = position.expandedTo(FloatPoint());
223
224 if (isRootNode() && scrollingTree().scrollPinningBehavior() == PinToTop)
225 position.setY(minimumScrollPosition().y());
226
227 return position;
228}

236unsigned ScrollingTreeFrameScrollingNodeMac::exposedUnfilledArea() const
237{
238 Region paintedVisibleTiles;
239
240 Deque<CALayer*> layerQueue;
241 layerQueue.append(scrolledContentsLayer());
242 PlatformLayerList tiles;
243
244 while (!layerQueue.isEmpty() && tiles.isEmpty()) {
245 CALayer* layer = layerQueue.takeFirst();
246 auto sublayers = adoptNS([[layer sublayers] copy]);
247
248 // If this layer is the parent of a tile, it is the parent of all of the tiles and nothing else.
249 if ([[[sublayers objectAtIndex:0] valueForKey:@"isTile"] boolValue]) {
250 for (CALayer* sublayer in sublayers.get())
251 tiles.append(sublayer);
252 } else {
253 for (CALayer* sublayer in sublayers.get())
254 layerQueue.append(sublayer);
255 }
256 }
257
258 FloatPoint clampedScrollPosition = clampScrollPosition(currentScrollPosition());
259 FloatRect viewportRect({ }, scrollableAreaSize());
260 return TileController::blankPixelCountForTiles(tiles, viewportRect, IntPoint(-clampedScrollPosition.x(), -clampedScrollPosition.y()));
261}
