57Ref<ScrollingTreeNode> ScrollingTreeMac::createScrollingTreeNode(ScrollingNodeType nodeType, ScrollingNodeID nodeID)
58{
59 switch (nodeType) {
60 case ScrollingNodeType::MainFrame:
61 case ScrollingNodeType::Subframe:
62 return ScrollingTreeFrameScrollingNodeMac::create(*this, nodeType, nodeID);
63 case ScrollingNodeType::FrameHosting:
64 return ScrollingTreeFrameHostingNode::create(*this, nodeID);
65 case ScrollingNodeType::Overflow:
66 return ScrollingTreeOverflowScrollingNodeMac::create(*this, nodeID);
67 case ScrollingNodeType::OverflowProxy:
68 return ScrollingTreeOverflowScrollProxyNode::create(*this, nodeID);
69 case ScrollingNodeType::Fixed:
70 return ScrollingTreeFixedNode::create(*this, nodeID);
71 case ScrollingNodeType::Sticky:
72 return ScrollingTreeStickyNode::create(*this, nodeID);
73 case ScrollingNodeType::Positioned:
74 return ScrollingTreePositionedNode::create(*this, nodeID);
75 }
76 ASSERT_NOT_REACHED();
77 return ScrollingTreeFixedNode::create(*this, nodeID);
78}

80using LayerAndPoint = std::pair<CALayer *, FloatPoint>;
81
82static void collectDescendantLayersAtPoint(Vector<LayerAndPoint, 16>& layersAtPoint, CALayer *parent, CGPoint point)
83{
84 if (parent.masksToBounds && ![parent containsPoint:point])
85 return;
86
87 if (parent.mask && ![parent _web_maskContainsPoint:point])
88 return;
89
90 for (CALayer *layer in [parent sublayers]) {
91 CALayer *layerWithResolvedAnimations = layer;
92
93 if ([[layer animationKeys] count])
94 layerWithResolvedAnimations = [layer presentationLayer];
95
96 auto transform = TransformationMatrix { [layerWithResolvedAnimations transform] };
97 if (!transform.isInvertible())
98 continue;
99
100 CGPoint subviewPoint = [layerWithResolvedAnimations convertPoint:point fromLayer:parent];
101
102 auto handlesEvent = [&] {
103 if (CGRectIsEmpty([layerWithResolvedAnimations frame]))
104 return false;
105
106 if (![layerWithResolvedAnimations containsPoint:subviewPoint])
107 return false;
108
109 auto platformCALayer = PlatformCALayer::platformCALayerForLayer((__bridge void*)layer);
110 if (platformCALayer) {
111 // Scrolling changes boundsOrigin on the scroll container layer, but we computed its event region ignoring scroll position, so factor out bounds origin.
112 FloatPoint boundsOrigin = layer.bounds.origin;
113 FloatPoint localPoint = subviewPoint - toFloatSize(boundsOrigin);
114 auto* eventRegion = platformCALayer->eventRegion();
115 return eventRegion && eventRegion->contains(roundedIntPoint(localPoint));
116 }
117
118 return false;
119 }();
120
121 if (handlesEvent)
122 layersAtPoint.append(std::make_pair(layer, subviewPoint));
123
124 if ([layer sublayers])
125 collectDescendantLayersAtPoint(layersAtPoint, layer, subviewPoint);
126 };
127}

129static ScrollingNodeID scrollingNodeIDForLayer(CALayer *layer)
130{
131 auto platformCALayer = PlatformCALayer::platformCALayerForLayer((__bridge void*)layer);
132 return platformCALayer ? platformCALayer->scrollingNodeID() : 0;
133}

135static bool isScrolledBy(const ScrollingTree& tree, ScrollingNodeID scrollingNodeID, CALayer *hitLayer)
136{
137 for (CALayer *layer = hitLayer; layer; layer = [layer superlayer]) {
138 auto nodeID = scrollingNodeIDForLayer(layer);
139 if (nodeID == scrollingNodeID)
140 return true;
141
142 auto* scrollingNode = tree.nodeForID(nodeID);
143 if (is<ScrollingTreeOverflowScrollProxyNode>(scrollingNode)) {
144 ScrollingNodeID actingOverflowScrollingNodeID = downcast<ScrollingTreeOverflowScrollProxyNode>(*scrollingNode).overflowScrollingNodeID();
145 if (actingOverflowScrollingNodeID == scrollingNodeID)
146 return true;
147 }
148
149 if (is<ScrollingTreePositionedNode>(scrollingNode)) {
150 if (downcast<ScrollingTreePositionedNode>(*scrollingNode).relatedOverflowScrollingNodes().contains(scrollingNodeID))
151 return false;
152 }
153 }
154
155 return false;
156}

158RefPtr<ScrollingTreeNode> ScrollingTreeMac::scrollingNodeForPoint(FloatPoint point)
159{
160 auto* rootScrollingNode = rootNode();
161 if (!rootScrollingNode)
162 return nullptr;
163
164 Locker locker { m_layerHitTestMutex };
165
166 auto rootContentsLayer = static_cast<ScrollingTreeFrameScrollingNodeMac*>(rootScrollingNode)->rootContentsLayer();
167 FloatPoint scrollOrigin = rootScrollingNode->scrollOrigin();
168 auto pointInContentsLayer = point;
169 pointInContentsLayer.moveBy(scrollOrigin);
170
171 Vector<LayerAndPoint, 16> layersAtPoint;
172 collectDescendantLayersAtPoint(layersAtPoint, rootContentsLayer.get(), pointInContentsLayer);
173
174 LOG_WITH_STREAM(Scrolling, stream << "ScrollingTreeMac " << this << " scrollingNodeForPoint " << point << " found " << layersAtPoint.size() << " layers");
175#if !LOG_DISABLED
176 for (auto [layer, point] : WTF::makeReversedRange(layersAtPoint))
177 LOG_WITH_STREAM(Scrolling, stream << " layer " << [layer description] << " scrolling node " << scrollingNodeIDForLayer(layer));
178#endif
179
180 if (layersAtPoint.size()) {
181 auto* frontmostLayer = layersAtPoint.last().first;
182 for (auto [layer, point] : WTF::makeReversedRange(layersAtPoint)) {
183 auto nodeID = scrollingNodeIDForLayer(layer);
184
185 auto* scrollingNode = nodeForID(nodeID);
186 if (!is<ScrollingTreeScrollingNode>(scrollingNode))
187 continue;
188
189 if (isScrolledBy(*this, nodeID, frontmostLayer))
190 return scrollingNode;
191
192 // FIXME: Hit-test scroll indicator layers.
193 }
194 }
195
196 LOG_WITH_STREAM(Scrolling, stream << "ScrollingTreeMac " << this << " scrollingNodeForPoint " << point << " found no scrollable layers; using root node");
197 return rootScrollingNode;
198}

200#if ENABLE(WHEEL_EVENT_REGIONS)
201OptionSet<EventListenerRegionType> ScrollingTreeMac::eventListenerRegionTypesForPoint(FloatPoint point) const
202{
203 auto* rootScrollingNode = rootNode();
204 if (!rootScrollingNode)
205 return { };
206
207 Locker locker { m_layerHitTestMutex };
208
209 auto rootContentsLayer = static_cast<ScrollingTreeFrameScrollingNodeMac*>(rootScrollingNode)->rootContentsLayer();
210
211 Vector<LayerAndPoint, 16> layersAtPoint;
212 collectDescendantLayersAtPoint(layersAtPoint, rootContentsLayer.get(), point);
213
214 if (layersAtPoint.isEmpty())
215 return { };
216
217 auto [hitLayer, localPoint] = layersAtPoint.last();
218 if (!hitLayer)
219 return { };
220
221 auto platformCALayer = PlatformCALayer::platformCALayerForLayer((__bridge void*)hitLayer);
222 if (!platformCALayer)
223 return { };
224
225 auto* eventRegion = platformCALayer->eventRegion();
226 if (!eventRegion)
227 return { };
228
229 return eventRegion->eventListenerRegionTypesForPoint(roundedIntPoint(localPoint));
230}
