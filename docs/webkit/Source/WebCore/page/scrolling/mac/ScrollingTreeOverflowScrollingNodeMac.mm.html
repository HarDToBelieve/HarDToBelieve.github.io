58void ScrollingTreeOverflowScrollingNodeMac::commitStateBeforeChildren(const ScrollingStateNode& stateNode)
59{
60 ScrollingTreeOverflowScrollingNode::commitStateBeforeChildren(stateNode);
61 m_delegate.updateFromStateNode(downcast<ScrollingStateOverflowScrollingNode>(stateNode));
62}
63
64void ScrollingTreeOverflowScrollingNodeMac::commitStateAfterChildren(const ScrollingStateNode& stateNode)
65{
66 ScrollingTreeOverflowScrollingNode::commitStateAfterChildren(stateNode);
67
68 const auto& overflowStateNode = downcast<ScrollingStateOverflowScrollingNode>(stateNode);
69
70 if (overflowStateNode.hasChangedProperty(ScrollingStateNode::Property::RequestedScrollPosition)) {
71 const auto& requestedScrollData = overflowStateNode.requestedScrollData();
72 scrollTo(requestedScrollData.scrollPosition, requestedScrollData.scrollType, requestedScrollData.clamping);
73 }
74}

76WheelEventHandlingResult ScrollingTreeOverflowScrollingNodeMac::handleWheelEvent(const PlatformWheelEvent& wheelEvent, EventTargeting eventTargeting)
77{
78#if ENABLE(SCROLLING_THREAD)
79 if (hasSynchronousScrollingReasons() && eventTargeting != EventTargeting::NodeOnly)
80 return { { WheelEventProcessingSteps::MainThreadForScrolling, WheelEventProcessingSteps::MainThreadForNonBlockingDOMEventDispatch }, false };
81#endif
82
83 if (!canHandleWheelEvent(wheelEvent, eventTargeting))
84 return WheelEventHandlingResult::unhandled();
85
86 return WheelEventHandlingResult::result(m_delegate.handleWheelEvent(wheelEvent));
87}
