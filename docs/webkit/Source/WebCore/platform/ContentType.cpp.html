58String ContentType::parameter(const String& parameterName) const
59{
60 // A MIME type can have one or more "param=value" after a semicolon, separated from each other by semicolons.
61
62 // FIXME: This will ignore a quotation mark if it comes before the semicolon. Is that the desired behavior?
63 auto semicolonPosition = m_type.find(';');
64 if (semicolonPosition == notFound)
65 return { };
66
67 // FIXME: This matches parameters that have parameterName as a suffix; that is not the desired behavior.
68 auto nameStart = m_type.findIgnoringASCIICase(parameterName, semicolonPosition + 1);
69 if (nameStart == notFound)
70 return { };
71
72 auto equalSignPosition = m_type.find('=', nameStart + parameterName.length());
73 if (equalSignPosition == notFound)
74 return { };
75
76 // FIXME: This skips over any characters that come before a quotation mark; that is not the desired behavior.
77 auto quotePosition = m_type.find('"', equalSignPosition + 1);
78 // FIXME: This does not work if there is an escaped quotation mark in the quoted string. Is that the desired behavior?
79 auto secondQuotePosition = m_type.find('"', quotePosition + 1);
80 size_t start;
81 size_t end;
82 if (quotePosition != notFound && secondQuotePosition != notFound) {
83 start = quotePosition + 1;
84 end = secondQuotePosition;
85 } else {
86 // FIXME: If there is only one quotation mark, this will treat it as part of the string; that is not the desired behavior.
87 start = equalSignPosition + 1;
88 end = m_type.find(';', start);
89 }
90 return StringView { m_type }.substring(start, end - start).stripLeadingAndTrailingMatchedCharacters(isHTMLSpace<UChar>).toString();
91}
92
93String ContentType::containerType() const
94{
95 // Strip parameters that come after a semicolon.
96 // FIXME: This will ignore a quotation mark if it comes before the semicolon. Is that the desired behavior?
97 return stripLeadingAndTrailingHTMLSpaces(m_type.left(m_type.find(';')));
98}

100static inline Vector<String> splitParameters(StringView parametersView)
101{
102 Vector<String> result;
103 for (auto view : parametersView.split(','))
104 result.append(view.stripLeadingAndTrailingMatchedCharacters(isHTMLSpace<UChar>).toString());
105 return result;
106}
107
108Vector<String> ContentType::codecs() const
109{
110 return splitParameters(parameter(codecsParameter()));
111}
112
113Vector<String> ContentType::profiles() const
114{
115 return splitParameters(parameter(profilesParameter()));
116}
117
118String ContentType::toJSONString() const
119{
120 auto object = JSON::Object::create();
121
122 object->setString("containerType"_s, containerType());
123
124 auto codecs = codecsParameter();
125 if (!codecs.isEmpty())
126 object->setString("codecs"_s, codecs);
127
128 auto profiles = profilesParameter();
129 if (!profiles.isEmpty())
130 object->setString("profiles"_s, profiles);
131
132 return object->toJSONString();
133}
