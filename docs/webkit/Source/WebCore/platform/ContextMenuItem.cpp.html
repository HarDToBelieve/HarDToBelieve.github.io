35ContextMenuItem::ContextMenuItem(ContextMenuItemType type, ContextMenuAction action, const String& title, ContextMenu* subMenu)

79bool ContextMenuItem::isNull() const
80{
81 // FIXME: This is a bit of a hack. Cross-platform ContextMenuItem users need a concrete way to track "isNull".
82 return m_action == ContextMenuItemTagNoAction && m_title.isNull() && m_subMenuItems.isEmpty();
83}
84
85void ContextMenuItem::setSubMenu(ContextMenu* subMenu)
86{
87 if (subMenu) {
88 m_type = SubmenuType;
89 m_subMenuItems = subMenu->items();
90 } else {
91 m_type = ActionType;
92 m_subMenuItems.clear();
93 }
94}

146bool isValidContextMenuAction(ContextMenuAction action)
147{
148 switch (action) {
149 case ContextMenuAction::ContextMenuItemTagNoAction:
150 case ContextMenuAction::ContextMenuItemTagOpenLinkInNewWindow:
151 case ContextMenuAction::ContextMenuItemTagDownloadLinkToDisk:
152 case ContextMenuAction::ContextMenuItemTagCopyLinkToClipboard:
153 case ContextMenuAction::ContextMenuItemTagOpenImageInNewWindow:
154 case ContextMenuAction::ContextMenuItemTagDownloadImageToDisk:
155 case ContextMenuAction::ContextMenuItemTagCopyImageToClipboard:
156#if PLATFORM(GTK)
157 case ContextMenuAction::ContextMenuItemTagCopyImageUrlToClipboard:
158#endif
159 case ContextMenuAction::ContextMenuItemTagOpenFrameInNewWindow:
160 case ContextMenuAction::ContextMenuItemTagCopy:
161 case ContextMenuAction::ContextMenuItemTagGoBack:
162 case ContextMenuAction::ContextMenuItemTagGoForward:
163 case ContextMenuAction::ContextMenuItemTagStop:
164 case ContextMenuAction::ContextMenuItemTagReload:
165 case ContextMenuAction::ContextMenuItemTagCut:
166 case ContextMenuAction::ContextMenuItemTagPaste:
167#if PLATFORM(GTK)
168 case ContextMenuAction::ContextMenuItemTagPasteAsPlainText:
169 case ContextMenuAction::ContextMenuItemTagDelete:
170 case ContextMenuAction::ContextMenuItemTagSelectAll:
171 case ContextMenuAction::ContextMenuItemTagInputMethods:
172 case ContextMenuAction::ContextMenuItemTagUnicode:
173 case ContextMenuAction::ContextMenuItemTagUnicodeInsertLRMMark:
174 case ContextMenuAction::ContextMenuItemTagUnicodeInsertRLMMark:
175 case ContextMenuAction::ContextMenuItemTagUnicodeInsertLREMark:
176 case ContextMenuAction::ContextMenuItemTagUnicodeInsertRLEMark:
177 case ContextMenuAction::ContextMenuItemTagUnicodeInsertLROMark:
178 case ContextMenuAction::ContextMenuItemTagUnicodeInsertRLOMark:
179 case ContextMenuAction::ContextMenuItemTagUnicodeInsertPDFMark:
180 case ContextMenuAction::ContextMenuItemTagUnicodeInsertZWSMark:
181 case ContextMenuAction::ContextMenuItemTagUnicodeInsertZWJMark:
182 case ContextMenuAction::ContextMenuItemTagUnicodeInsertZWNJMark:
183 case ContextMenuAction::ContextMenuItemTagInsertEmoji:
184#endif
185 case ContextMenuAction::ContextMenuItemTagSpellingGuess:
186 case ContextMenuAction::ContextMenuItemTagNoGuessesFound:
187 case ContextMenuAction::ContextMenuItemTagIgnoreSpelling:
188 case ContextMenuAction::ContextMenuItemTagLearnSpelling:
189 case ContextMenuAction::ContextMenuItemTagOther:
190 case ContextMenuAction::ContextMenuItemTagSearchInSpotlight:
191 case ContextMenuAction::ContextMenuItemTagSearchWeb:
192 case ContextMenuAction::ContextMenuItemTagLookUpInDictionary:
193 case ContextMenuAction::ContextMenuItemTagOpenWithDefaultApplication:
194 case ContextMenuAction::ContextMenuItemPDFActualSize:
195 case ContextMenuAction::ContextMenuItemPDFZoomIn:
196 case ContextMenuAction::ContextMenuItemPDFZoomOut:
197 case ContextMenuAction::ContextMenuItemPDFAutoSize:
198 case ContextMenuAction::ContextMenuItemPDFSinglePage:
199 case ContextMenuAction::ContextMenuItemPDFFacingPages:
200 case ContextMenuAction::ContextMenuItemPDFContinuous:
201 case ContextMenuAction::ContextMenuItemPDFNextPage:
202 case ContextMenuAction::ContextMenuItemPDFPreviousPage:
203 case ContextMenuAction::ContextMenuItemTagOpenLink:
204 case ContextMenuAction::ContextMenuItemTagIgnoreGrammar:
205 case ContextMenuAction::ContextMenuItemTagSpellingMenu:
206 case ContextMenuAction::ContextMenuItemTagShowSpellingPanel:
207 case ContextMenuAction::ContextMenuItemTagCheckSpelling:
208 case ContextMenuAction::ContextMenuItemTagCheckSpellingWhileTyping:
209 case ContextMenuAction::ContextMenuItemTagCheckGrammarWithSpelling:
210 case ContextMenuAction::ContextMenuItemTagFontMenu:
211 case ContextMenuAction::ContextMenuItemTagShowFonts:
212 case ContextMenuAction::ContextMenuItemTagBold:
213 case ContextMenuAction::ContextMenuItemTagItalic:
214 case ContextMenuAction::ContextMenuItemTagUnderline:
215 case ContextMenuAction::ContextMenuItemTagOutline:
216 case ContextMenuAction::ContextMenuItemTagStyles:
217 case ContextMenuAction::ContextMenuItemTagShowColors:
218 case ContextMenuAction::ContextMenuItemTagSpeechMenu:
219 case ContextMenuAction::ContextMenuItemTagStartSpeaking:
220 case ContextMenuAction::ContextMenuItemTagStopSpeaking:
221 case ContextMenuAction::ContextMenuItemTagWritingDirectionMenu:
222 case ContextMenuAction::ContextMenuItemTagDefaultDirection:
223 case ContextMenuAction::ContextMenuItemTagLeftToRight:
224 case ContextMenuAction::ContextMenuItemTagRightToLeft:
225 case ContextMenuAction::ContextMenuItemTagPDFSinglePageScrolling:
226 case ContextMenuAction::ContextMenuItemTagPDFFacingPagesScrolling:
227 case ContextMenuAction::ContextMenuItemTagInspectElement:
228 case ContextMenuAction::ContextMenuItemTagTextDirectionMenu:
229 case ContextMenuAction::ContextMenuItemTagTextDirectionDefault:
230 case ContextMenuAction::ContextMenuItemTagTextDirectionLeftToRight:
231 case ContextMenuAction::ContextMenuItemTagTextDirectionRightToLeft:
232 case ContextMenuAction::ContextMenuItemTagAddHighlightToCurrentQuickNote:
233 case ContextMenuAction::ContextMenuItemTagAddHighlightToNewQuickNote:
234#if PLATFORM(COCOA)
235 case ContextMenuAction::ContextMenuItemTagCorrectSpellingAutomatically:
236 case ContextMenuAction::ContextMenuItemTagSubstitutionsMenu:
237 case ContextMenuAction::ContextMenuItemTagShowSubstitutions:
238 case ContextMenuAction::ContextMenuItemTagSmartCopyPaste:
239 case ContextMenuAction::ContextMenuItemTagSmartQuotes:
240 case ContextMenuAction::ContextMenuItemTagSmartDashes:
241 case ContextMenuAction::ContextMenuItemTagSmartLinks:
242 case ContextMenuAction::ContextMenuItemTagTextReplacement:
243 case ContextMenuAction::ContextMenuItemTagTransformationsMenu:
244 case ContextMenuAction::ContextMenuItemTagMakeUpperCase:
245 case ContextMenuAction::ContextMenuItemTagMakeLowerCase:
246 case ContextMenuAction::ContextMenuItemTagCapitalize:
247 case ContextMenuAction::ContextMenuItemTagChangeBack:
248#endif
249 case ContextMenuAction::ContextMenuItemTagOpenMediaInNewWindow:
250 case ContextMenuAction::ContextMenuItemTagDownloadMediaToDisk:
251 case ContextMenuAction::ContextMenuItemTagCopyMediaLinkToClipboard:
252 case ContextMenuAction::ContextMenuItemTagToggleMediaControls:
253 case ContextMenuAction::ContextMenuItemTagToggleMediaLoop:
254 case ContextMenuAction::ContextMenuItemTagEnterVideoFullscreen:
255 case ContextMenuAction::ContextMenuItemTagMediaPlayPause:
256 case ContextMenuAction::ContextMenuItemTagMediaMute:
257 case ContextMenuAction::ContextMenuItemTagDictationAlternative:
258 case ContextMenuAction::ContextMenuItemTagToggleVideoFullscreen:
259 case ContextMenuAction::ContextMenuItemTagShareMenu:
260 case ContextMenuAction::ContextMenuItemTagToggleVideoEnhancedFullscreen:
261 case ContextMenuAction::ContextMenuItemTagQuickLookImage:
262 case ContextMenuAction::ContextMenuItemTagTranslate:
263 case ContextMenuAction::ContextMenuItemBaseCustomTag:
264 case ContextMenuAction::ContextMenuItemLastCustomTag:
265 case ContextMenuAction::ContextMenuItemBaseApplicationTag:
266 return true;
267 }
268
269 if (action > ContextMenuAction::ContextMenuItemBaseCustomTag && action < ContextMenuAction::ContextMenuItemLastCustomTag)
270 return true;
271
272 if (action > ContextMenuAction::ContextMenuItemBaseApplicationTag)
273 return true;
274
275 return false;
276}
