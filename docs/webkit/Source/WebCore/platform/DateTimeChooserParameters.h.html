34namespace WebCore {
35
36struct DateTimeChooserParameters {
37 AtomString type;
38 IntRect anchorRectInRootView;
39 // Locale name for which the chooser should be localized.
40 // This might be an invalid name because it comes from HTML lang attributes.
41 AtomString locale;
42 String currentValue;
43 Vector<String> suggestionValues;
44 Vector<String> localizedSuggestionValues;
45 Vector<String> suggestionLabels;
46 double minimum { 0 };
47 double maximum { 0 };
48 double step { 0 };
49 double stepBase { 0 };
50 bool required { false };
51 bool isAnchorElementRTL { false };
52 bool useDarkAppearance { false };
53 bool hasSecondField { false };
54 bool hasMillisecondField { false };
55
56 template<class Encoder> void encode(Encoder&) const;
57 template<class Decoder> static std::optional<DateTimeChooserParameters> decode(Decoder&);
58};
59
60template<class Encoder>
61void DateTimeChooserParameters::encode(Encoder& encoder) const
62{
63 encoder << type;
64 encoder << anchorRectInRootView;
65 encoder << locale;
66 encoder << currentValue;
67 encoder << suggestionValues;
68 encoder << localizedSuggestionValues;
69 encoder << suggestionLabels;
70 encoder << minimum;
71 encoder << maximum;
72 encoder << step;
73 encoder << stepBase;
74 encoder << required;
75 encoder << isAnchorElementRTL;
76 encoder << useDarkAppearance;
77 encoder << hasSecondField;
78 encoder << hasMillisecondField;
79}
80
81template<class Decoder>
82std::optional<DateTimeChooserParameters> DateTimeChooserParameters::decode(Decoder& decoder)
83{
84 AtomString type;
85 if (!decoder.decode(type))
86 return std::nullopt;
87
88 IntRect anchorRectInRootView;
89 if (!decoder.decode(anchorRectInRootView))
90 return std::nullopt;
91
92 AtomString locale;
93 if (!decoder.decode(locale))
94 return std::nullopt;
95
96 String currentValue;
97 if (!decoder.decode(currentValue))
98 return std::nullopt;
99
100 Vector<String> suggestionValues;
101 if (!decoder.decode(suggestionValues))
102 return std::nullopt;
103
104 Vector<String> localizedSuggestionValues;
105 if (!decoder.decode(localizedSuggestionValues))
106 return std::nullopt;
107
108 Vector<String> suggestionLabels;
109 if (!decoder.decode(suggestionLabels))
110 return std::nullopt;
111
112 double minimum;
113 if (!decoder.decode(minimum))
114 return std::nullopt;
115
116 double maximum;
117 if (!decoder.decode(maximum))
118 return std::nullopt;
119
120 double step;
121 if (!decoder.decode(step))
122 return std::nullopt;
123
124 double stepBase;
125 if (!decoder.decode(stepBase))
126 return std::nullopt;
127
128 bool required;
129 if (!decoder.decode(required))
130 return std::nullopt;
131
132 bool isAnchorElementRTL;
133 if (!decoder.decode(isAnchorElementRTL))
134 return std::nullopt;
135
136 bool useDarkAppearance;
137 if (!decoder.decode(useDarkAppearance))
138 return std::nullopt;
139
140 bool hasSecondField;
141 if (!decoder.decode(hasSecondField))
142 return std::nullopt;
143
144 bool hasMillisecondField;
145 if (!decoder.decode(hasMillisecondField))
146 return std::nullopt;
147
148 return {{ WTFMove(type), anchorRectInRootView, WTFMove(locale), WTFMove(currentValue), WTFMove(suggestionValues), WTFMove(localizedSuggestionValues), WTFMove(suggestionLabels), minimum, maximum, step, stepBase, required, isAnchorElementRTL, useDarkAppearance, hasSecondField, hasMillisecondField }};
149}
