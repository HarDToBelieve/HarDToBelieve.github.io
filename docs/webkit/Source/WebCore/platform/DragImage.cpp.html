49DragImageRef fitDragImageToMaxSize(DragImageRef image, const IntSize& layoutSize, const IntSize& maxSize)
50{
51 float heightResizeRatio = 0.0f;
52 float widthResizeRatio = 0.0f;
53 float resizeRatio = -1.0f;
54 IntSize originalSize = dragImageSize(image);
55
56 if (layoutSize.width() > maxSize.width()) {
57 widthResizeRatio = maxSize.width() / (float)layoutSize.width();
58 resizeRatio = widthResizeRatio;
59 }
60
61 if (layoutSize.height() > maxSize.height()) {
62 heightResizeRatio = maxSize.height() / (float)layoutSize.height();
63 if ((resizeRatio < 0.0f) || (resizeRatio > heightResizeRatio))
64 resizeRatio = heightResizeRatio;
65 }
66
67 if (layoutSize == originalSize)
68 return resizeRatio > 0.0f ? scaleDragImage(image, FloatSize(resizeRatio, resizeRatio)) : image;
69
70 // The image was scaled in the webpage so at minimum we must account for that scaling.
71 float scaleX = layoutSize.width() / (float)originalSize.width();
72 float scaleY = layoutSize.height() / (float)originalSize.height();
73 if (resizeRatio > 0.0f) {
74 scaleX *= resizeRatio;
75 scaleY *= resizeRatio;
76 }
77
78 return scaleDragImage(image, FloatSize(scaleX, scaleY));
79}

81struct ScopedNodeDragEnabler {
82 ScopedNodeDragEnabler(Frame& frame, Node& node)
83 : element(is<Element>(node) ? &downcast<Element>(node) : nullptr)
84 {
85 if (element)
86 element->setBeingDragged(true);
87 frame.document()->updateLayout();
88 }
89
90 ~ScopedNodeDragEnabler()
91 {
92 if (element)
93 element->setBeingDragged(false);
94 }
95
96 RefPtr<Element> element;
97};

99static DragImageRef createDragImageFromSnapshot(RefPtr<ImageBuffer> snapshot, Node* node)
100{
101 if (!snapshot)
102 return nullptr;
103
104 ImageOrientation orientation;
105 if (node) {
106 RenderObject* renderer = node->renderer();
107 if (!renderer || !is<RenderElement>(renderer))
108 return nullptr;
109
110 orientation = downcast<RenderElement>(*renderer).imageOrientation();
111 }
112
113 RefPtr<Image> image = ImageBuffer::sinkIntoImage(WTFMove(snapshot), PreserveResolution::Yes);
114 if (!image)
115 return nullptr;
116 return createDragImageFromImage(image.get(), orientation);
117}

137struct ScopedFrameSelectionState {
138 ScopedFrameSelectionState(Frame& frame)
139 : frame(frame)
140 {
141 if (auto* renderView = frame.contentRenderer())
142 selection = renderView->selection().get();
143 }
144
145 ~ScopedFrameSelectionState()
146 {
147 if (auto* renderView = frame.contentRenderer()) {
148 ASSERT(selection);
149 renderView->selection().set(selection.value(), SelectionRangeData::RepaintMode::Nothing);
150 }
151 }
152
153 const Frame& frame;
154 std::optional<RenderRange> selection;
155};

159DragImageRef createDragImageForRange(Frame& frame, const SimpleRange& range, bool forceBlackText)
160{
161 frame.document()->updateLayout();
162 RenderView* view = frame.contentRenderer();
163 if (!view)
164 return nullptr;
165
166 // To snapshot the range, temporarily select it and take selection snapshot.
167 Position start = makeDeprecatedLegacyPosition(range.start);
168 Position candidate = start.downstream();
169 if (candidate.deprecatedNode() && candidate.deprecatedNode()->renderer())
170 start = candidate;
171
172 Position end = makeDeprecatedLegacyPosition(range.end);
173 candidate = end.upstream();
174 if (candidate.deprecatedNode() && candidate.deprecatedNode()->renderer())
175 end = candidate;
176
177 if (start.isNull() || end.isNull() || start == end)
178 return nullptr;
179
180 const ScopedFrameSelectionState selectionState(frame);
181
182 RenderObject* startRenderer = start.deprecatedNode()->renderer();
183 RenderObject* endRenderer = end.deprecatedNode()->renderer();
184 if (!startRenderer || !endRenderer)
185 return nullptr;
186
187 SnapshotOptions options { { SnapshotFlags::PaintSelectionOnly }, PixelFormat::BGRA8, DestinationColorSpace::SRGB() };
188 if (forceBlackText)
189 options.flags.add(SnapshotFlags::ForceBlackText);
190
191 int startOffset = start.deprecatedEditingOffset();
192 int endOffset = end.deprecatedEditingOffset();
193 ASSERT(startOffset >= 0 && endOffset >= 0);
194 view->selection().set({ startRenderer, endRenderer, static_cast<unsigned>(startOffset), static_cast<unsigned>(endOffset) }, SelectionRangeData::RepaintMode::Nothing);
195 // We capture using snapshotFrameRect() because we fake up the selection using
196 // FrameView but snapshotSelection() uses the selection from the Frame itself.
197 return createDragImageFromSnapshot(snapshotFrameRect(frame, view->selection().boundsClippedToVisibleContent(), WTFMove(options)), nullptr);
198}

202DragImageRef createDragImageForImage(Frame& frame, Node& node, IntRect& imageRect, IntRect& elementRect)
203{
204 ScopedNodeDragEnabler enableDrag(frame, node);
205
206 RenderObject* renderer = node.renderer();
207 if (!renderer)
208 return nullptr;
209
210 // Calculate image and element metrics for the client, then create drag image.
211 LayoutRect topLevelRect;
212 IntRect paintingRect = snappedIntRect(renderer->paintingRootRect(topLevelRect));
213
214 if (paintingRect.isEmpty())
215 return nullptr;
216
217 elementRect = snappedIntRect(topLevelRect);
218 imageRect = paintingRect;
219
220 return createDragImageFromSnapshot(snapshotNode(frame, node, { { }, PixelFormat::BGRA8, DestinationColorSpace::SRGB() }), &node);
221}

265{

271 m_visiblePath = other.m_visiblePath;

274}
