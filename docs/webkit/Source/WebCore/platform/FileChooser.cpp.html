54void FileChooser::chooseFile(const String& filename)
55{

57 filenames.append(filename);

61void FileChooser::chooseFiles(const Vector<String>& filenames, const Vector<String>& replacementNames)
62{
63 // FIXME: This is inelegant. We should not be looking at settings here.
64 if (m_settings.selectedFiles == filenames)
65 return;
66
67 if (!m_client)
68 return;
69
70 Vector<FileChooserFileInfo> files;
71 for (size_t i = 0, size = filenames.size(); i < size; ++i)
72 files.append({ filenames[i], i < replacementNames.size() ? replacementNames[i] : nullString(), { } });
73 m_client->filesChosen(files);
74}

97void FileChooser::chooseFiles(const Vector<FileChooserFileInfo>& files)
98{
99 Vector<String> paths;
100 for (auto& file : files)
101 paths.append(file.path);
102
103 // FIXME: This is inelegant. We should not be looking at settings here.
104 if (m_settings.selectedFiles == paths)
105 return;
106
107 if (m_client)
108 m_client->filesChosen(files);
109}
