29namespace WebCore {

31static constexpr Seconds MinimumAverageDuration = 1_s;
32static constexpr Seconds MaxQueueDuration = 2_s;
33static constexpr unsigned MaxFrameDelayCount = 3;
34
35void FrameRateMonitor::update()
36{
37 ++m_frameCount;
38
39 auto frameTime = MonotonicTime::now().secondsSinceEpoch().value();
40 auto lastFrameTime = m_observedFrameTimeStamps.isEmpty() ? frameTime : m_observedFrameTimeStamps.last();
41
42 if (m_observedFrameRate) {
43 auto maxDelay = MaxFrameDelayCount / m_observedFrameRate;
44 if ((frameTime - lastFrameTime) > maxDelay)
45 m_lateFrameCallback({ MonotonicTime::fromRawSeconds(frameTime), MonotonicTime::fromRawSeconds(lastFrameTime) });
46 }
47 m_observedFrameTimeStamps.append(frameTime);
48 m_observedFrameTimeStamps.removeAllMatching([&](auto time) {
49 return time <= frameTime - MaxQueueDuration.value();
50 });
51
52 auto queueDuration = m_observedFrameTimeStamps.last() - m_observedFrameTimeStamps.first();
53 if (queueDuration > MinimumAverageDuration.value())
54 m_observedFrameRate = (m_observedFrameTimeStamps.size() / queueDuration);
55}
