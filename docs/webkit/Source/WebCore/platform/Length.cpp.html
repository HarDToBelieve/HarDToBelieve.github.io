44static Length parseLength(const UChar* data, unsigned length)
45{
46 if (length == 0)
47 return Length(1, LengthType::Relative);
48
49 unsigned i = 0;
50 while (i < length && isSpaceOrNewline(data[i]))
51 ++i;
52 if (i < length && (data[i] == '+' || data[i] == '-'))
53 ++i;
54 while (i < length && isASCIIDigit(data[i]))
55 ++i;
56 unsigned intLength = i;
57 while (i < length && (isASCIIDigit(data[i]) || data[i] == '.'))
58 ++i;
59 unsigned doubleLength = i;
60
61 // IE quirk: Skip whitespace between the number and the % character (20 % => 20%).
62 while (i < length && isSpaceOrNewline(data[i]))
63 ++i;
64
65 bool ok;
66 UChar next = (i < length) ? data[i] : ' ';
67 if (next == '%') {
68 // IE quirk: accept decimal fractions for percentages.
69 double r = charactersToDouble(data, doubleLength, &ok);
70 if (ok)
71 return Length(r, LengthType::Percent);
72 return Length(1, LengthType::Relative);
73 }
74 auto r = parseInteger<int>({ data, intLength });
75 if (next == '*')
76 return Length(r.value_or(1), LengthType::Relative);
77 if (r)
78 return Length(*r, LengthType::Fixed);
79 return Length(0, LengthType::Relative);
80}

82static unsigned countCharacter(StringImpl& string, UChar character)
83{
84 unsigned count = 0;
85 unsigned length = string.length();
86 for (unsigned i = 0; i < length; ++i)
87 count += string[i] == character;
88 return count;
89}

91UniqueArray<Length> newCoordsArray(const String& string, int& len)
92{
93 unsigned length = string.length();
94 LChar* spacifiedCharacters;
95 auto str = StringImpl::createUninitialized(length, spacifiedCharacters);
96 for (unsigned i = 0; i < length; i++) {
97 UChar cc = string[i];
98 if (cc > '9' || (cc < '0' && cc != '-' && cc != '*' && cc != '.'))
99 spacifiedCharacters[i] = ' ';
100 else
101 spacifiedCharacters[i] = cc;
102 }
103 str = str->simplifyWhiteSpace();
104
105 len = countCharacter(str, ' ') + 1;
106 auto r = makeUniqueArray<Length>(len);
107
108 int i = 0;
109 unsigned pos = 0;
110 size_t pos2;
111
112 while ((pos2 = str->find(' ', pos)) != notFound) {
113 r[i++] = parseLength(str->characters16() + pos, pos2 - pos);
114 pos = pos2+1;
115 }
116 r[i] = parseLength(str->characters16() + pos, str->length() - pos);
117
118 ASSERT(i == len - 1);
119
120 return r;
121}

123UniqueArray<Length> newLengthArray(const String& string, int& len)
124{
125 RefPtr<StringImpl> str = string.impl()->simplifyWhiteSpace();
126 if (!str->length()) {
127 len = 1;
128 return nullptr;
129 }
130
131 len = countCharacter(*str, ',') + 1;
132 auto r = makeUniqueArray<Length>(len);
133
134 int i = 0;
135 unsigned pos = 0;
136 size_t pos2;
137
138 auto upconvertedCharacters = StringView(str.get()).upconvertedCharacters();
139 while ((pos2 = str->find(',', pos)) != notFound) {
140 r[i++] = parseLength(upconvertedCharacters + pos, pos2 - pos);
141 pos = pos2+1;
142 }
143
144 ASSERT(i == len - 1);
145
146 // IE Quirk: If the last comma is the last char skip it and reduce len by one.
147 if (str->length()-pos > 0)
148 r[i] = parseLength(upconvertedCharacters + pos, str->length() - pos);
149 else
150 len--;
151
152 return r;
153}

155class CalculationValueMap {
156public:
157 CalculationValueMap();
158
159 unsigned insert(Ref<CalculationValue>&&);
160 void ref(unsigned handle);
161 void deref(unsigned handle);
162
163 CalculationValue& get(unsigned handle) const;
164
165private:
166 struct Entry {
167 uint64_t referenceCountMinusOne;
168 CalculationValue* value;
169 Entry();
170 Entry(CalculationValue&);
171 };
172
173 unsigned m_nextAvailableHandle;
174 HashMap<unsigned, Entry> m_map;
175};

177inline CalculationValueMap::Entry::Entry()
178 : referenceCountMinusOne(0)
179 , value(nullptr)
180{
181}

183inline CalculationValueMap::Entry::Entry(CalculationValue& value)
184 : referenceCountMinusOne(0)
185 , value(&value)
186{
187}

189inline CalculationValueMap::CalculationValueMap()
190 : m_nextAvailableHandle(1)
191{
192}

194inline unsigned CalculationValueMap::insert(Ref<CalculationValue>&& value)
195{
196 ASSERT(m_nextAvailableHandle);
197
198 // The leakRef below is balanced by the adoptRef in the deref member function.
199 Entry leakedValue = value.leakRef();
200
201 // FIXME: This monotonically increasing handle generation scheme is potentially wasteful
202 // of the handle space. Consider reusing empty handles. https://bugs.webkit.org/show_bug.cgi?id=80489
203 while (!m_map.isValidKey(m_nextAvailableHandle) || !m_map.add(m_nextAvailableHandle, leakedValue).isNewEntry)
204 ++m_nextAvailableHandle;
205
206 return m_nextAvailableHandle++;
207}
208
209inline CalculationValue& CalculationValueMap::get(unsigned handle) const
210{
211 ASSERT(m_map.contains(handle));
212
213 return *m_map.find(handle)->value.value;
214}
215
216inline void CalculationValueMap::ref(unsigned handle)
217{
218 ASSERT(m_map.contains(handle));
219
220 ++m_map.find(handle)->value.referenceCountMinusOne;
221}
222
223inline void CalculationValueMap::deref(unsigned handle)
224{
225 ASSERT(m_map.contains(handle));
226
227 auto it = m_map.find(handle);
228 if (it->value.referenceCountMinusOne) {
229 --it->value.referenceCountMinusOne;
230 return;
231 }
232
233 // The adoptRef here is balanced by the leakRef in the insert member function.
234 Ref<CalculationValue> value { adoptRef(*it->value.value) };
235
236 m_map.remove(it);
237}

285Length convertTo100PercentMinusLength(const Length& length)
286{
287 if (length.isPercent())
288 return Length(100 - length.value(), LengthType::Percent);
289
290 // Turn this into a calc expression: calc(100% - length)
291 Vector<std::unique_ptr<CalcExpressionNode>> lengths;
292 lengths.reserveInitialCapacity(2);
293 lengths.uncheckedAppend(makeUnique<CalcExpressionLength>(Length(100, LengthType::Percent)));
294 lengths.uncheckedAppend(makeUnique<CalcExpressionLength>(length));
295 auto op = makeUnique<CalcExpressionOperation>(WTFMove(lengths), CalcOperator::Subtract);
296 return Length(CalculationValue::create(WTFMove(op), ValueRange::All));
297}

299static Length blendMixedTypes(const Length& from, const Length& to, const BlendingContext& context)
300{
301 if (!to.isCalculated() && !from.isPercent() && (context.progress == 1 || from.isZero()))
302 return blend(Length(0, to.type()), to, context);
303
304 if (!from.isCalculated() && !to.isPercent() && (!context.progress || to.isZero()))
305 return blend(from, Length(0, from.type()), context);
306
307 auto blend = makeUnique<CalcExpressionBlendLength>(from, to, context.progress);
308 return Length(CalculationValue::create(WTFMove(blend), ValueRange::All));
309}

311Length blend(const Length& from, const Length& to, const BlendingContext& context)
312{
313 if ((from.isAuto() || to.isAuto()) || (from.isUndefined() || to.isUndefined()))
314 return context.progress < 0.5 ? from : to;
315
316 if (from.isCalculated() || to.isCalculated() || (from.type() != to.type()))
317 return blendMixedTypes(from, to, context);
318
319 if (!context.progress)
320 return from;
321
322 if (context.progress == 1)
323 return to;
324
325 LengthType resultType = to.type();
326 if (to.isZero())
327 resultType = from.type();
328
329 if (resultType == LengthType::Percent) {
330 float fromPercent = from.isZero() ? 0 : from.percent();
331 float toPercent = to.isZero() ? 0 : to.percent();
332 return Length(WebCore::blend(fromPercent, toPercent, context), LengthType::Percent);
333 }
334
335 float fromValue = from.isZero() ? 0 : from.value();
336 float toValue = to.isZero() ? 0 : to.value();
337 return Length(WebCore::blend(fromValue, toValue, context), resultType);
338}

340Length blend(const Length& from, const Length& to, const BlendingContext& context, ValueRange valueRange)
341{
342 auto blended = blend(from, to, context);
343 if (valueRange == ValueRange::NonNegative && blended.isNegative())
344 return { 0, from.isZero () ? to.type() : from.type() };
345 return blended;
346}

354static TextStream& operator<<(TextStream& ts, LengthType type)
355{
356 switch (type) {
357 case LengthType::Auto: ts << "auto"; break;
358 case LengthType::Relative: ts << "relative"; break;
359 case LengthType::Percent: ts << "percent"; break;
360 case LengthType::Fixed: ts << "fixed"; break;
361 case LengthType::Intrinsic: ts << "intrinsic"; break;
362 case LengthType::MinIntrinsic: ts << "min-intrinsic"; break;
363 case LengthType::MinContent: ts << "min-content"; break;
364 case LengthType::MaxContent: ts << "max-content"; break;
365 case LengthType::FillAvailable: ts << "fill-available"; break;
366 case LengthType::FitContent: ts << "fit-content"; break;
367 case LengthType::Calculated: ts << "calc"; break;
368 case LengthType::Undefined: ts << "undefined"; break;
369 }
370 return ts;
371}
372
373TextStream& operator<<(TextStream& ts, Length length)
374{
375 switch (length.type()) {
376 case LengthType::Auto:
377 case LengthType::Undefined:
378 ts << length.type();
379 break;
380 case LengthType::Fixed:
381 ts << TextStream::FormatNumberRespectingIntegers(length.value()) << "px";
382 break;
383 case LengthType::Relative:
384 case LengthType::Intrinsic:
385 case LengthType::MinIntrinsic:
386 case LengthType::MinContent:
387 case LengthType::MaxContent:
388 case LengthType::FillAvailable:
389 case LengthType::FitContent:
390 ts << length.type() << " " << TextStream::FormatNumberRespectingIntegers(length.value());
391 break;
392 case LengthType::Percent:
393 ts << TextStream::FormatNumberRespectingIntegers(length.percent()) << "%";
394 break;
395 case LengthType::Calculated:
396 ts << length.calculationValue();
397 break;
398 }
399
400 if (length.hasQuirk())
401 ts << " has-quirk";
402
403 return ts;
404}
