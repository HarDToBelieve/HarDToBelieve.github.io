177inline Length::Length(const Length& other)
178{
179 if (other.isCalculated())
180 other.ref();
181
182 memcpy(static_cast<void*>(this), static_cast<void*>(const_cast<Length*>(&other)), sizeof(Length));
183}
184
185inline Length::Length(Length&& other)
186{
187 memcpy(static_cast<void*>(this), static_cast<void*>(&other), sizeof(Length));
188 other.m_type = LengthType::Auto;
189}
190
191inline Length& Length::operator=(const Length& other)
192{
193 if (this == &other)
194 return *this;
195
196 if (other.isCalculated())
197 other.ref();
198 if (isCalculated())
199 deref();
200
201 memcpy(static_cast<void*>(this), static_cast<void*>(const_cast<Length*>(&other)), sizeof(Length));
202 return *this;
203}
204
205inline Length& Length::operator=(Length&& other)
206{
207 if (this == &other)
208 return *this;
209
210 if (isCalculated())
211 deref();
212
213 memcpy(static_cast<void*>(this), static_cast<void*>(&other), sizeof(Length));
214 other.m_type = LengthType::Auto;
215 return *this;
216}
217
218inline Length::~Length()
219{
220 if (isCalculated())
221 deref();
222}
223
224inline bool Length::operator==(const Length& other) const
225{
226 // FIXME: This might be too long to be inline.
227 if (type() != other.type() || hasQuirk() != other.hasQuirk())
228 return false;
229 if (isUndefined())
230 return true;
231 if (isCalculated())
232 return isCalculatedEqual(other);
233 return value() == other.value();
234}
235
236inline bool Length::operator!=(const Length& other) const
237{
238 return !(*this == other);
239}
240
241inline Length& Length::operator*=(float value)
242{
243 ASSERT(!isCalculated());
244 if (isCalculated())
245 return *this;
246
247 if (m_isFloat)
248 m_floatValue *= value;
249 else
250 m_intValue *= value;
251
252 return *this;
253}

255inline float Length::value() const
256{
257 ASSERT(!isUndefined());
258 ASSERT(!isCalculated());
259 return m_isFloat ? m_floatValue : m_intValue;
260}
261
262inline int Length::intValue() const
263{
264 ASSERT(!isUndefined());
265 ASSERT(!isCalculated());
266 // FIXME: Makes no sense to return 0 here but not in the value() function above.
267 if (isCalculated())
268 return 0;
269 return m_isFloat ? static_cast<int>(m_floatValue) : m_intValue;
270}

376inline bool Length::isZero() const
377{
378 ASSERT(!isUndefined());
379 if (isCalculated() || isAuto())
380 return false;
381 return m_isFloat ? !m_floatValue : !m_intValue;
382}

394inline bool Length::isIntrinsic() const
395{
396 // FIXME: This is misleadingly named. One would expect this function does "return type() == LengthType::Intrinsic;".
397 return type() == LengthType::MinContent || type() == LengthType::MaxContent || type() == LengthType::FillAvailable || type() == LengthType::FitContent;
398}
