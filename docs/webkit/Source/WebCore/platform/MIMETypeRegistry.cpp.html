151template<ASCIISubset subset, unsigned size> static FixedVector<const char*> makeFixedVector(const ComparableASCIISubsetLiteral<subset> (&array)[size])
152{
153 FixedVector<const char*> result(std::size(array));
154 std::transform(std::begin(array), std::end(array), result.begin(), [] (auto literal) {
155 return literal.literal;
156 });
157 return result;
158}

378static const Vector<String>* typesForCommonExtension(const String& extension)
379{
380 auto mapEntry = commonMimeTypesMap().find(extension);
381 if (mapEntry == commonMimeTypesMap().end())
382 return nullptr;
383 return &mapEntry->value;
384}

386String MIMETypeRegistry::mediaMIMETypeForExtension(const String& extension)

394String MIMETypeRegistry::mimeTypeForPath(const String& path)
395{
396 size_t pos = path.reverseFind('.');
397 if (pos != notFound) {
398 String extension = path.substring(pos + 1);
399 String result = mimeTypeForExtension(extension);
400 if (result.length())
401 return result;
402 }
403 return defaultMIMEType();
404}

406bool MIMETypeRegistry::isSupportedImageMIMEType(const String& mimeType)
407{
408 if (mimeType.isEmpty())
409 return false;
410 static constexpr SortedArraySet supportedImageMIMETypeSet { supportedImageMIMETypeArray };
411#if USE(CG) && ASSERT_ENABLED
412 // Esnure supportedImageMIMETypeArray matches defaultSupportedImageTypes().
413 static std::once_flag onceFlag;
414 std::call_once(onceFlag, [] {
415 for (auto& imageType : defaultSupportedImageTypes()) {
416 auto mimeType = MIMETypeForImageType(imageType);
417 ASSERT_IMPLIES(!mimeType.isEmpty(), supportedImageMIMETypeSet.contains(mimeType));
418 }
419 });
420#endif
421 if (supportedImageMIMETypeSet.contains(mimeType))
422 return true;
423 return additionalSupportedImageMIMETypes().contains(normalizedImageMIMEType(mimeType));
424}

449 CFIndex count = CFArrayGetCount(supportedTypes.get());
450 for (CFIndex i = 0; i < count; i++) {
451 CFStringRef supportedType = reinterpret_cast<CFStringRef>(CFArrayGetValueAtIndex(supportedTypes.get(), i));
452 if (!isSupportedImageType(supportedType))
453 continue;

455 if (mimeType.isEmpty())
456 continue;

458 }
459#else

461#if USE(CG) || USE(DIRECT2D)

463 // For now, only support PNG, JPEG and GIF. See <rdar://problem/6095286>.
464 "image/png"_s,
465 "image/jpeg"_s,
466 "image/gif"_s,
467#elif PLATFORM(GTK)
468 "image/png"_s,
469 "image/jpeg"_s,
470 "image/tiff"_s,
471 "image/bmp"_s,
472 "image/ico"_s,
473#elif USE(CAIRO)

477#endif
478 return makeUnique<MIMETypeRegistryThreadGlobalData>(WTFMove(supportedImageMIMETypesForEncoding));

533{

543 if (slashPosition != notFound && slashPosition > 0 && slashPosition <= mimeType.length() - 6)
544 return true;
545 }
546
547 return false;

585static inline bool isValidXMLMIMETypeChar(UChar c)
586{
587 // Valid characters per RFCs 3023 and 2045: 0-9a-zA-Z_-+~!$^{}|.%'`#&*
588 return isASCIIAlphanumeric(c) || c == '!' || c == '#' || c == '$' || c == '%' || c == '&' || c == '\'' || c == '*' || c == '+'
589 || c == '-' || c == '.' || c == '^' || c == '_' || c == '`' || c == '{' || c == '|' || c == '}' || c == '~';
590}
591
592bool MIMETypeRegistry::isXMLMIMEType(const String& mimeType)
593{
594 if (equalLettersIgnoringASCIICase(mimeType, "text/xml") || equalLettersIgnoringASCIICase(mimeType, "application/xml") || equalLettersIgnoringASCIICase(mimeType, "text/xsl"))
595 return true;
596
597 if (!mimeType.endsWithIgnoringASCIICase("+xml"))
598 return false;
599
600 size_t slashPosition = mimeType.find('/');
601 // Take into account the '+xml' ending of mimeType.
602 if (slashPosition == notFound || !slashPosition || slashPosition == mimeType.length() - 5)
603 return false;
604
605 // Again, mimeType ends with '+xml', no need to check the validity of that substring.
606 size_t mimeLength = mimeType.length();
607 for (size_t i = 0; i < mimeLength - 4; ++i) {
608 if (!isValidXMLMIMETypeChar(mimeType[i]) && i != slashPosition)
609 return false;
610 }
611
612 return true;
613}

649{

752String MIMETypeRegistry::appendFileExtensionIfNecessary(const String& filename, const String& mimeType)
753{
754 if (filename.isEmpty() || filename.contains('.') || equalIgnoringASCIICase(mimeType, defaultMIMEType()))
755 return filename;
756
757 auto preferredExtension = preferredExtensionForMIMEType(mimeType);
758 if (preferredExtension.isEmpty())
759 return filename;
760
761 return makeString(filename, '.', preferredExtension);
762}

765{
766 return extension.startsWith('.') ? extension.right(extension.length() - 1) : extension;

769String MIMETypeRegistry::preferredImageMIMETypeForEncoding(const Vector<String>& mimeTypes, const Vector<String>& extensions)
770{
771 auto allowedMIMETypes = MIMETypeRegistry::allowedMIMETypes(mimeTypes, extensions);
772

785Vector<String> MIMETypeRegistry::allowedMIMETypes(const Vector<String>& mimeTypes, const Vector<String>& extensions)
786{
787 Vector<String> allowedMIMETypes;
788
789 for (auto& mimeType : mimeTypes)
790 allowedMIMETypes.appendIfNotContains(mimeType.convertToASCIILowercase());
791
792 for (auto& extension : extensions) {
793 auto mimeType = MIMETypeRegistry::mimeTypeForExtension(trimmedExtension(extension));
794 if (mimeType.isEmpty())
795 continue;
796 allowedMIMETypes.appendIfNotContains(mimeType.convertToASCIILowercase());
797 }
798
799 return allowedMIMETypes;
800}

802Vector<String> MIMETypeRegistry::allowedFileExtensions(const Vector<String>& mimeTypes, const Vector<String>& extensions)
803{
804 Vector<String> allowedFileExtensions;
805
806 for (auto& mimeType : mimeTypes) {
807 for (auto& extension : MIMETypeRegistry::extensionsForMIMEType(mimeType))
808 allowedFileExtensions.appendIfNotContains(extension);
809 }
810
811 for (auto& extension : extensions)
812 allowedFileExtensions.appendIfNotContains(trimmedExtension(extension));
813
814 return allowedFileExtensions;
815}
