37struct PODIntervalNodeUpdater;

39// An interval tree, which is a form of augmented red-black tree. It
40// supports efficient (O(lg n)) insertion, removal and querying of
41// intervals in the tree.
42template<typename T, typename UserData> class PODIntervalTree final : public PODRedBlackTree<PODInterval<T, UserData>, PODIntervalNodeUpdater> {
43 WTF_MAKE_FAST_ALLOCATED;
44public:
45 using IntervalType = PODInterval<T, UserData>;
46 class OverlapsSearchAdapter;
47
48 // Returns all intervals in the tree which overlap the given query interval, sorted by the < operator.
49 Vector<IntervalType> allOverlaps(const IntervalType& interval) const
50 {
51 Vector<IntervalType> result;
52 OverlapsSearchAdapter adapter(result, interval);
53 allOverlapsWithAdapter(adapter);
54 return result;
55 }
56
57 template<typename AdapterType> void allOverlapsWithAdapter(AdapterType& adapter) const
58 {
59 searchForOverlapsFrom(this->root(), adapter);
60 }
61
62 std::optional<IntervalType> nextIntervalAfter(const T& point)
63 {
64 auto next = smallestNodeGreaterThanFrom(point, this->root());
65 if (!next)
66 return std::nullopt;
67 return next->data();
68 }
69
70#ifndef NDEBUG
71
72 bool checkInvariants() const
73 {
74 if (!Base::checkInvariants())
75 return false;
76 if (!this->root())
77 return true;
78 return checkInvariantsFromNode(this->root(), nullptr);
79 }
80
81#endif
82
83private:
84 using Base = PODRedBlackTree<PODInterval<T, UserData>, PODIntervalNodeUpdater>;
85 using IntervalNode = typename Base::Node;
86
87 // Starting from the given node, adds all overlaps with the given
88 // interval to the result vector. The intervals are sorted by
89 // increasing low endpoint.
90 template<typename AdapterType> void searchForOverlapsFrom(IntervalNode* node, AdapterType& adapter) const
91 {
92 if (!node)
93 return;
94
95 // Because the intervals are sorted by left endpoint, inorder
96 // traversal produces results sorted as desired.
97
98 // See whether we need to traverse the left subtree.
99 IntervalNode* left = node->left();
100 if (left
101 // This is phrased this way to avoid the need for operator
102 // <= on type T.
103 && !(left->data().maxHigh() < adapter.lowValue()))
104 searchForOverlapsFrom<AdapterType>(left, adapter);
105
106 // Check for overlap with current node.
107 adapter.collectIfNeeded(node->data());
108
109 // See whether we need to traverse the right subtree.
110 // This is phrased this way to avoid the need for operator <=
111 // on type T.
112 if (!(adapter.highValue() < node->data().low()))
113 searchForOverlapsFrom<AdapterType>(node->right(), adapter);
114 }
115
116 IntervalNode* smallestNodeGreaterThanFrom(const T& point, IntervalNode* node) const
117 {
118 if (!node)
119 return nullptr;
120
121 if (!(point < node->data().low()))
122 return smallestNodeGreaterThanFrom(point, node->right());
123
124 if (auto left = smallestNodeGreaterThanFrom(point, node->right()))
125 return left;
126
127 return node;
128 }
129
130#ifndef NDEBUG
131
132 bool checkInvariantsFromNode(IntervalNode* node, T* currentMaxValue) const
133 {
134 // These assignments are only done in order to avoid requiring
135 // a default constructor on type T.
136 T leftMaxValue(node->data().maxHigh());
137 T rightMaxValue(node->data().maxHigh());
138 IntervalNode* left = node->left();
139 IntervalNode* right = node->right();
140 if (left) {
141 if (!checkInvariantsFromNode(left, &leftMaxValue))
142 return false;
143 }
144 if (right) {
145 if (!checkInvariantsFromNode(right, &rightMaxValue))
146 return false;
147 }
148 if (!left && !right) {
149 // Base case.
150 if (currentMaxValue)
151 *currentMaxValue = node->data().high();
152 return (node->data().high() == node->data().maxHigh());
153 }
154 T localMaxValue(node->data().maxHigh());
155 if (!left || !right) {
156 if (left)
157 localMaxValue = leftMaxValue;
158 else
159 localMaxValue = rightMaxValue;
160 } else
161 localMaxValue = (leftMaxValue < rightMaxValue) ? rightMaxValue : leftMaxValue;
162 if (localMaxValue < node->data().high())
163 localMaxValue = node->data().high();
164 if (!(localMaxValue == node->data().maxHigh())) {
165 TextStream stream;
166 stream << "localMaxValue=" << localMaxValue << "and data =" << node->data();
167 LOG_ERROR("PODIntervalTree verification failed at node 0x%p: %s",
168 node, stream.release().utf8().data());
169 return false;
170 }
171 if (currentMaxValue)
172 *currentMaxValue = localMaxValue;
173 return true;
174 }
175
176#endif
177
178};
179
180template<typename T, typename UserData> class PODIntervalTree<T, UserData>::OverlapsSearchAdapter {
181public:
182 using IntervalType = PODInterval<T, UserData>;
183
184 OverlapsSearchAdapter(Vector<IntervalType>& result, const IntervalType& interval)
185 : m_result(result)
186 , m_interval(interval)
187 {
188 }
189
190 const T& lowValue() const { return m_interval.low(); }
191 const T& highValue() const { return m_interval.high(); }
192 void collectIfNeeded(const IntervalType& data) const
193 {
194 if (data.overlaps(m_interval))
195 m_result.append(data);
196 }
197
198private:
199 Vector<IntervalType>& m_result;
200 const IntervalType& m_interval;
201};
202
203struct PODIntervalNodeUpdater {
204 template<typename Node> static bool update(Node& node)
205 {
206 auto* curMax = &node.data().high();
207 auto* left = node.left();
208 if (left) {
209 if (*curMax < left->data().maxHigh())
210 curMax = &left->data().maxHigh();
211 }
212 auto* right = node.right();
213 if (right) {
214 if (*curMax < right->data().maxHigh())
215 curMax = &right->data().maxHigh();
216 }
217 // This is phrased like this to avoid needing operator!= on type T.
218 if (!(*curMax == node.data().maxHigh())) {
219 node.data().setMaxHigh(*curMax);
220 return true;
221 }
222 return false;
223 }
224};
