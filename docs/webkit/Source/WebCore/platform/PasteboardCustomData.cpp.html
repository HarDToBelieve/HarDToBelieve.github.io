36static Variant<String, Ref<SharedBuffer>> copyPlatformData(const Variant<String, Ref<SharedBuffer>>& other)
37{
38 if (WTF::holds_alternative<String>(other))
39 return { WTF::get<String>(other) };
40
41 if (WTF::holds_alternative<Ref<SharedBuffer>>(other))
42 return { WTF::get<Ref<SharedBuffer>>(other).copyRef() };
43
44 return { };
45}

83Ref<SharedBuffer> PasteboardCustomData::createSharedBuffer() const
84{
85 constexpr unsigned currentCustomDataSerializationVersion = 1;
86
87 WTF::Persistence::Encoder encoder;
88 encoder << currentCustomDataSerializationVersion;
89 encoder << m_origin;
90 encoder << sameOriginCustomStringData();
91 encoder << orderedTypes();
92 return SharedBuffer::create(encoder.buffer(), encoder.bufferSize());
93}

95PasteboardCustomData PasteboardCustomData::fromPersistenceDecoder(WTF::Persistence::Decoder&& decoder)
96{
97 constexpr unsigned maxSupportedDataSerializationVersionNumber = 1;
98
99 PasteboardCustomData result;
100 std::optional<unsigned> version;
101 decoder >> version;
102 if (!version || *version > maxSupportedDataSerializationVersionNumber)
103 return { };
104
105 std::optional<String> origin;
106 decoder >> origin;
107 if (!origin)
108 return { };
109 result.m_origin = WTFMove(*origin);
110
111 std::optional<HashMap<String, String>> sameOriginCustomStringData;
112 decoder >> sameOriginCustomStringData;
113 if (!sameOriginCustomStringData)
114 return { };
115
116 std::optional<Vector<String>> orderedTypes;
117 decoder >> orderedTypes;
118 if (!orderedTypes)
119 return { };
120
121 for (auto& type : *orderedTypes)
122 result.writeStringInCustomData(type, sameOriginCustomStringData->get(type));
123
124 return result;
125}

127PasteboardCustomData PasteboardCustomData::fromSharedBuffer(const SharedBuffer& buffer)

147PasteboardCustomData::Entry& PasteboardCustomData::addOrMoveEntryToEnd(const String& type)
148{
149 auto index = m_data.findMatching([&] (auto& entry) {
150 return entry.type == type;
151 });
152 auto entry = index == notFound ? Entry(type) : m_data[index];
153 if (index != notFound)
154 m_data.remove(index);
155 m_data.append(WTFMove(entry));
156 return m_data.last();
157}

197HashMap<String, String> PasteboardCustomData::sameOriginCustomStringData() const
198{
199 HashMap<String, String> customData;
200 for (auto& entry : m_data)
201 customData.set(entry.type, entry.customData);
202 return customData;

205RefPtr<SharedBuffer> PasteboardCustomData::readBuffer(const String& type) const
206{
207 for (auto& entry : m_data) {
208 if (entry.type != type)
209 continue;
210
211 if (WTF::holds_alternative<Ref<SharedBuffer>>(entry.platformData))
212 return makeRefPtr(WTF::get<Ref<SharedBuffer>>(entry.platformData).get());
213
214 return nullptr;
215 }
216 return nullptr;
217}

233String PasteboardCustomData::readStringInCustomData(const String& type) const

248void PasteboardCustomData::forEachPlatformString(Function<void(const String& type, const String& data)>&& function) const
249{
250 for (auto& entry : m_data) {
251 if (!WTF::holds_alternative<String>(entry.platformData))
252 continue;
253
254 auto string = WTF::get<String>(entry.platformData);
255 if (!string.isNull())
256 function(entry.type, string);
257 }
258}

260void PasteboardCustomData::forEachCustomString(Function<void(const String& type, const String& data)>&& function) const
261{
262 for (auto& entry : m_data) {
263 if (!entry.customData.isNull())
264 function(entry.type, entry.customData);
265 }
266}

268void PasteboardCustomData::forEachPlatformStringOrBuffer(Function<void(const String& type, const Variant<String, Ref<SharedBuffer>>& data)>&& function) const
269{
270 for (auto& entry : m_data) {
271 auto& data = entry.platformData;
272 if ((WTF::holds_alternative<String>(data) && !WTF::get<String>(data).isNull()) || WTF::holds_alternative<Ref<SharedBuffer>>(data))
273 function(entry.type, data);
274 }
