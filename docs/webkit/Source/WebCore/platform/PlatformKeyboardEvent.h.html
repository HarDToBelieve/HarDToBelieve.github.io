44namespace WebCore {
45
46 class PlatformKeyboardEvent : public PlatformEvent {
47 WTF_MAKE_FAST_ALLOCATED;
48 public:
49 PlatformKeyboardEvent()
50 : PlatformEvent(PlatformEvent::KeyDown)
51 {
52 }
53
54 PlatformKeyboardEvent(Type type, const String& text, const String& unmodifiedText, const String& key, const String& code,
55 const String& keyIdentifier, int windowsVirtualKeyCode, bool isAutoRepeat, bool isKeypad, bool isSystemKey, OptionSet<Modifier> modifiers, WallTime timestamp)
56 : PlatformEvent(type, modifiers, timestamp)
57 , m_autoRepeat(isAutoRepeat)
58 , m_isKeypad(isKeypad)
59 , m_isSystemKey(isSystemKey)
60 , m_text(text)
61 , m_unmodifiedText(unmodifiedText)
62 , m_key(key)
63 , m_code(code)
64 , m_keyIdentifier(keyIdentifier)
65 , m_windowsVirtualKeyCode(windowsVirtualKeyCode)
66 {
67 }
68
69 WEBCORE_EXPORT void disambiguateKeyDownEvent(Type, bool backwardCompatibilityMode = false); // Only used on platforms that need it, i.e. those that generate KeyDown events.
70
71 // Text as as generated by processing a virtual key code with a keyboard layout
72 // (in most cases, just a character code, but the layout can emit several
73 // characters in a single keypress event on some platforms).
74 // This may bear no resemblance to the ultimately inserted text if an input method
75 // processes the input.
76 // Will be null for KeyUp and RawKeyDown events.
77 String text() const { return m_text; }
78
79 // Text that would have been generated by the keyboard if no modifiers were pressed
80 // (except for Shift); useful for shortcut (accelerator) key handling.
81 // Otherwise, same as text().
82 String unmodifiedText() const { return m_unmodifiedText; }
83
84 String keyIdentifier() const { return m_keyIdentifier; }
85 const String& key() const { return m_key; }
86 const String& code() const { return m_code; }
87
88 // Most compatible Windows virtual key code associated with the event.
89 // Zero for Char events.
90 int windowsVirtualKeyCode() const { return m_windowsVirtualKeyCode; }
91 void setWindowsVirtualKeyCode(int code) { m_windowsVirtualKeyCode = code; }
92
93#if USE(APPKIT) || USE(UIKIT_KEYBOARD_ADDITIONS) || PLATFORM(GTK) || USE(LIBWPE)
94 bool handledByInputMethod() const { return m_handledByInputMethod; }
95#endif
96#if PLATFORM(GTK) || USE(LIBWPE)
97 const std::optional<Vector<WebCore::CompositionUnderline>>& preeditUnderlines() const { return m_preeditUnderlines; }
98 const std::optional<uint64_t>& preeditSelectionRangeStart() const { return m_preeditSelectionRangeStart; }
99 const std::optional<uint64_t>& preeditSelectionRangeLength() const { return m_preeditSelectionRangeLength; }
100#endif
101#if USE(APPKIT)
102 const Vector<KeypressCommand>& commands() const { return m_commands; }
103#elif PLATFORM(GTK)
104 const Vector<String>& commands() const { return m_commands; }
105#endif
106
107 bool isAutoRepeat() const { return m_autoRepeat; }
108 bool isKeypad() const { return m_isKeypad; }
109 bool isSystemKey() const { return m_isSystemKey; }
110
111 bool isSyntheticEvent() const { return m_isSyntheticEvent; }
112 void setIsSyntheticEvent() { m_isSyntheticEvent = true; }

114 WEBCORE_EXPORT static bool currentCapsLockState();
115 WEBCORE_EXPORT static void getCurrentModifierState(bool& shiftKey, bool& ctrlKey, bool& altKey, bool& metaKey);
116 WEBCORE_EXPORT static void setCurrentModifierState(OptionSet<Modifier>);
