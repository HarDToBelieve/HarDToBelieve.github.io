26#pragma once
27
28#if ENABLE(SPEECH_SYNTHESIS)
29
30#include "PlatformSpeechSynthesisVoice.h"
31#include <wtf/MonotonicTime.h>
32
33namespace WebCore {
34
35class PlatformSpeechSynthesisUtteranceClient {
36};
37
38class PlatformSpeechSynthesisUtterance : public RefCounted<PlatformSpeechSynthesisUtterance> {
39public:
40 WEBCORE_EXPORT static Ref<PlatformSpeechSynthesisUtterance> create(PlatformSpeechSynthesisUtteranceClient&);
41
42 const String& text() const { return m_text; }
43 void setText(const String& text) { m_text = text; }
44
45 const String& lang() const { return m_lang; }
46 void setLang(const String& lang) { m_lang = lang; }
47
48 PlatformSpeechSynthesisVoice* voice() const { return m_voice.get(); }
49 void setVoice(PlatformSpeechSynthesisVoice* voice) { m_voice = voice; }
50
51 // Range = [0, 1] where 1 is the default.
52 float volume() const { return m_volume; }
53 void setVolume(float volume) { m_volume = std::max(std::min(1.0f, volume), 0.0f); }
54
55 // Range = [0.1, 10] where 1 is the default.
56 float rate() const { return m_rate; }
57 void setRate(float rate) { m_rate = std::max(std::min(10.0f, rate), 0.1f); }
58
59 // Range = [0, 2] where 1 is the default.
60 float pitch() const { return m_pitch; }
61 void setPitch(float pitch) { m_pitch = std::max(std::min(2.0f, pitch), 0.0f); }
62
63 MonotonicTime startTime() const { return m_startTime; }
64 void setStartTime(MonotonicTime startTime) { m_startTime = startTime; }
65
66 PlatformSpeechSynthesisUtteranceClient* client() const { return m_client; }
67 void setClient(PlatformSpeechSynthesisUtteranceClient* client) { m_client = client; }
68
69private:
70 explicit PlatformSpeechSynthesisUtterance(PlatformSpeechSynthesisUtteranceClient&);
71
72 PlatformSpeechSynthesisUtteranceClient* m_client;
73 String m_text;
74 String m_lang;
75 RefPtr<PlatformSpeechSynthesisVoice> m_voice;
76 float m_volume { 1 };
77 float m_rate { 1 };
78 float m_pitch { 1 };
79 MonotonicTime m_startTime;
80};
81
82} // namespace WebCore
83
84#endif // ENABLE(SPEECH_SYNTHESIS)
