41enum class WheelEventProcessingSteps : uint8_t {
42 ScrollingThread = 1 << 0,
43 MainThreadForScrolling = 1 << 1,
44 MainThreadForNonBlockingDOMEventDispatch = 1 << 2,
45 MainThreadForBlockingDOMEventDispatch = 1 << 3,
46};

48enum class WheelScrollGestureState : uint8_t {
49 Blocking,
50 NonBlocking
51};

53// The ScrollByPixelWheelEvent is a fine-grained event that specifies the precise number of pixels to scroll.
54// It is sent directly by touch pads on macOS, or synthesized when platforms generate line-by-line scrolling events.
55//
56// The ScrollByPageWheelEvent indicates that the wheel event should scroll an entire page.
57// In this case, WebKit built in paging behavior is used to page up and down.
58// This yields the same behavior as clicking in a scrollbar track to page up and down.
59
60enum PlatformWheelEventGranularity : uint8_t {
61 ScrollByPageWheelEvent,
62 ScrollByPixelWheelEvent,
63};
64
65#if ENABLE(KINETIC_SCROLLING)

67enum class PlatformWheelEventPhase : uint8_t {
68 None = 0,
69 Began = 1 << 0,
70 Stationary = 1 << 1,
71 Changed = 1 << 2,
72 Ended = 1 << 3,
73 Cancelled = 1 << 4,
74 MayBegin = 1 << 5,
75};
76
77WTF::TextStream& operator<<(WTF::TextStream&, PlatformWheelEventPhase);

79#endif

81#if PLATFORM(WIN)
82// How many pixels should we scroll per line? Gecko uses the height of the
83// current line, which means scroll distance changes as you go through the
84// page or go to different pages. IE 7 is ~50 px/line, although the value
85// seems to vary slightly by page and zoom level. Since IE 7 has a
86// smoothing algorithm on scrolling, it can get away with slightly larger
87// scroll values without feeling jerky. Here we use 100 px per three lines
88// (the default scroll amount on Windows is three lines per wheel tick).
89const float cScrollbarPixelsPerLine = 100.0f / 3.0f;
90#endif

92class PlatformWheelEvent : public PlatformEvent {
93public:
94 PlatformWheelEvent()
95 : PlatformEvent(PlatformEvent::Wheel)
96 {
97 }
98
99 PlatformWheelEvent(IntPoint position, IntPoint globalPosition, float deltaX, float deltaY, float wheelTicksX, float wheelTicksY, PlatformWheelEventGranularity granularity, bool shiftKey, bool ctrlKey, bool altKey, bool metaKey)
100 : PlatformEvent(PlatformEvent::Wheel, shiftKey, ctrlKey, altKey, metaKey, { })
101 , m_granularity(granularity)
102 , m_position(position)
103 , m_globalPosition(globalPosition)
104 , m_deltaX(deltaX)
105 , m_deltaY(deltaY)
106 , m_wheelTicksX(wheelTicksX)
107 , m_wheelTicksY(wheelTicksY)
108 {
109 }
110
111#if ENABLE(MAC_GESTURE_EVENTS)
112 static PlatformWheelEvent createFromGesture(const PlatformGestureEvent&, double deltaY);
113#endif
114
115 PlatformWheelEvent copySwappingDirection() const
116 {
117 PlatformWheelEvent copy = *this;
118 std::swap(copy.m_deltaX, copy.m_deltaY);
119 std::swap(copy.m_wheelTicksX, copy.m_wheelTicksY);
120 return copy;
121 }
122
123 PlatformWheelEvent copyWithDeltasAndVelocity(float deltaX, float deltaY, const FloatSize& velocity) const
124 {
125 PlatformWheelEvent copy = *this;
126 copy.m_deltaX = deltaX;
127 copy.m_deltaY = deltaY;
128 copy.m_scrollingVelocity = velocity;
129 return copy;
130 }
131
132 const IntPoint& position() const { return m_position; } // PlatformWindow coordinates.
133 const IntPoint& globalPosition() const { return m_globalPosition; } // Screen coordinates.
134
135 float deltaX() const { return m_deltaX; }
136 float deltaY() const { return m_deltaY; }
137 FloatSize delta() const { return { m_deltaX, m_deltaY}; }
138
139 float wheelTicksX() const { return m_wheelTicksX; }
140 float wheelTicksY() const { return m_wheelTicksY; }
141
142 PlatformWheelEventGranularity granularity() const { return m_granularity; }
143
144 bool directionInvertedFromDevice() const { return m_directionInvertedFromDevice; }
145
146 const FloatSize& scrollingVelocity() const { return m_scrollingVelocity; }
147
148 bool hasPreciseScrollingDeltas() const { return m_hasPreciseScrollingDeltas; }
149 void setHasPreciseScrollingDeltas(bool hasPreciseScrollingDeltas) { m_hasPreciseScrollingDeltas = hasPreciseScrollingDeltas; }
150
151#if PLATFORM(COCOA)
152 unsigned scrollCount() const { return m_scrollCount; }
153 float unacceleratedScrollingDeltaX() const { return m_unacceleratedScrollingDeltaX; }
154 float unacceleratedScrollingDeltaY() const { return m_unacceleratedScrollingDeltaY; }
155#endif
156
157#if ENABLE(ASYNC_SCROLLING)
158 bool useLatchedEventElement() const;
159 bool isGestureContinuation() const; // The fingers-down part of the gesture excluding momentum.
160 bool shouldResetLatching() const;
161 bool isNonGestureEvent() const;
162 bool isEndOfMomentumScroll() const;
163#else
164 bool useLatchedEventElement() const { return false; }
165#endif
166
167#if ENABLE(KINETIC_SCROLLING)
168 PlatformWheelEventPhase phase() const { return m_phase; }
169 PlatformWheelEventPhase momentumPhase() const { return m_momentumPhase; }
170
171 bool isGestureStart() const;
172 bool isGestureCancel() const;
173
174 bool isEndOfNonMomentumScroll() const;
175 bool isTransitioningToMomentumScroll() const;
176 FloatPoint swipeVelocity() const;
177#endif
178
179#if PLATFORM(WIN)
180 PlatformWheelEvent(HWND, WPARAM, LPARAM, bool isMouseHWheel);
181#endif
182
183protected:
184 PlatformWheelEventGranularity m_granularity { ScrollByPixelWheelEvent };
185 bool m_directionInvertedFromDevice { false };
186 bool m_hasPreciseScrollingDeltas { false };
187
188 IntPoint m_position;
189 IntPoint m_globalPosition;
190 float m_deltaX { 0 };
191 float m_deltaY { 0 };
192 float m_wheelTicksX { 0 };
193 float m_wheelTicksY { 0 };
194
195 // Scrolling velocity in pixels per second.
196 FloatSize m_scrollingVelocity;
197
198#if ENABLE(KINETIC_SCROLLING)
199 PlatformWheelEventPhase m_phase { PlatformWheelEventPhase::None };
200 PlatformWheelEventPhase m_momentumPhase { PlatformWheelEventPhase::None };
201#endif
202#if PLATFORM(COCOA)
203 unsigned m_scrollCount { 0 };
204 float m_unacceleratedScrollingDeltaX { 0 };
205 float m_unacceleratedScrollingDeltaY { 0 };
206#endif
207};

211inline bool PlatformWheelEvent::useLatchedEventElement() const
212{
213 return m_phase == PlatformWheelEventPhase::Began
214 || m_phase == PlatformWheelEventPhase::Changed
215 || m_momentumPhase == PlatformWheelEventPhase::Began
216 || m_momentumPhase == PlatformWheelEventPhase::Changed
217 || (m_phase == PlatformWheelEventPhase::Ended && m_momentumPhase == PlatformWheelEventPhase::None)
218 || (m_phase == PlatformWheelEventPhase::None && m_momentumPhase == PlatformWheelEventPhase::Ended);
219}
220
221inline bool PlatformWheelEvent::isGestureContinuation() const
222{
223 return m_phase == PlatformWheelEventPhase::Changed;
224}
225
226inline bool PlatformWheelEvent::shouldResetLatching() const
227{
228 return m_phase == PlatformWheelEventPhase::Cancelled || m_phase == PlatformWheelEventPhase::MayBegin || (m_phase == PlatformWheelEventPhase::None && m_momentumPhase == PlatformWheelEventPhase::None) || isEndOfMomentumScroll();
229}
230
231inline bool PlatformWheelEvent::isNonGestureEvent() const
232{
233 return m_phase == PlatformWheelEventPhase::None && m_momentumPhase == PlatformWheelEventPhase::None;
234}
235
236inline bool PlatformWheelEvent::isEndOfMomentumScroll() const
237{
238 return m_phase == PlatformWheelEventPhase::None && m_momentumPhase == PlatformWheelEventPhase::Ended;
239}

241#endif // ENABLE(ASYNC_SCROLLING)

243#if ENABLE(KINETIC_SCROLLING)
244
245inline bool PlatformWheelEvent::isGestureStart() const
246{
247 return m_phase == PlatformWheelEventPhase::Began || m_phase == PlatformWheelEventPhase::MayBegin;
248}
249
250inline bool PlatformWheelEvent::isGestureCancel() const
251{
252 return m_phase == PlatformWheelEventPhase::Cancelled;
253}
254
255inline bool PlatformWheelEvent::isEndOfNonMomentumScroll() const
256{
257 return m_phase == PlatformWheelEventPhase::Ended && m_momentumPhase == PlatformWheelEventPhase::None;
258}
259
260inline bool PlatformWheelEvent::isTransitioningToMomentumScroll() const
261{
262 return m_phase == PlatformWheelEventPhase::None && m_momentumPhase == PlatformWheelEventPhase::Began;
263}
264
265inline FloatPoint PlatformWheelEvent::swipeVelocity() const
266{
267 // The swiping velocity is stored in the deltas of the event declaring it.
268 return isTransitioningToMomentumScroll() ? FloatPoint(m_wheelTicksX, m_wheelTicksY) : FloatPoint();
269}
