26#pragma once
27
28#include "Color.h"
29#include "FontCascade.h"
30#include "Length.h"
31#include <wtf/EnumTraits.h>
32
33namespace WebCore {
34
35class PopupMenuStyle {
36public:
37 enum PopupMenuType { SelectPopup, AutofillPopup };
38 enum BackgroundColorType { DefaultBackgroundColor, CustomBackgroundColor };
39 enum PopupMenuSize {
40 PopupMenuSizeNormal,
41 PopupMenuSizeSmall,
42 PopupMenuSizeMini,
43#if HAVE(LARGE_CONTROL_SIZE)
44 PopupMenuSizeLarge,
45#endif
46 };
47
48 PopupMenuStyle(const Color& foreground, const Color& background, const FontCascade& font, bool visible, bool isDisplayNone, bool hasDefaultAppearance, Length textIndent, TextDirection textDirection, bool hasTextDirectionOverride, BackgroundColorType backgroundColorType = DefaultBackgroundColor, PopupMenuType menuType = SelectPopup, PopupMenuSize menuSize = PopupMenuSizeNormal)
49 : m_foregroundColor(foreground)
50 , m_backgroundColor(background)
51 , m_font(font)
52 , m_visible(visible)
53 , m_isDisplayNone(isDisplayNone)
54 , m_hasDefaultAppearance(hasDefaultAppearance)
55 , m_textIndent(textIndent)
56 , m_textDirection(textDirection)
57 , m_hasTextDirectionOverride(hasTextDirectionOverride)
58 , m_backgroundColorType(backgroundColorType)
59 , m_menuType(menuType)
60 , m_menuSize(menuSize)
61 {
62 }
63
64 const Color& foregroundColor() const { return m_foregroundColor; }
65 const Color& backgroundColor() const { return m_backgroundColor; }
66 const FontCascade& font() const { return m_font; }
67 bool isVisible() const { return m_visible; }
68 bool isDisplayNone() const { return m_isDisplayNone; }
69 bool hasDefaultAppearance() const { return m_hasDefaultAppearance; }
70 Length textIndent() const { return m_textIndent; }
71 TextDirection textDirection() const { return m_textDirection; }
72 bool hasTextDirectionOverride() const { return m_hasTextDirectionOverride; }
73 BackgroundColorType backgroundColorType() const { return m_backgroundColorType; }
74 PopupMenuType menuType() const { return m_menuType; }
75 PopupMenuSize menuSize() const { return m_menuSize; }
76
77private:
78 Color m_foregroundColor;
79 Color m_backgroundColor;
80 FontCascade m_font;
81 bool m_visible;
82 bool m_isDisplayNone;
83 bool m_hasDefaultAppearance;
84 Length m_textIndent;
85 TextDirection m_textDirection;
86 bool m_hasTextDirectionOverride;
87 BackgroundColorType m_backgroundColorType;
88 PopupMenuType m_menuType;
89 PopupMenuSize m_menuSize;
90};
91
92} // namespace WebCore

102#if HAVE(LARGE_CONTROL_SIZE)
103 , WebCore::PopupMenuStyle::PopupMenuSize::PopupMenuSizeLarge
104#endif
105 >;
