35static RemoteCommandListener::CreationFunction& remoteCommandListenerCreationFunction()
36{
37 static NeverDestroyed<RemoteCommandListener::CreationFunction> creationFunction;
38 return creationFunction;
39}
40
41void RemoteCommandListener::setCreationFunction(CreationFunction&& function)
42{
43 remoteCommandListenerCreationFunction() = WTFMove(function);
44}
45
46void RemoteCommandListener::resetCreationFunction()
47{
48 remoteCommandListenerCreationFunction() = [] (RemoteCommandListenerClient& client) {
49#if PLATFORM(COCOA)
50 return RemoteCommandListenerCocoa::create(client);
51#else
52 return RemoteCommandListener::create(client);
53#endif
54 };
55}
56
57std::unique_ptr<RemoteCommandListener> RemoteCommandListener::create(RemoteCommandListenerClient& client)
58{
59 if (!remoteCommandListenerCreationFunction())
60 resetCreationFunction();
61 return remoteCommandListenerCreationFunction()(client);
62}

64RemoteCommandListener::RemoteCommandListener(RemoteCommandListenerClient& client)
65 : m_client(client)
66{
67}

69RemoteCommandListener::~RemoteCommandListener() = default;

72void RemoteCommandListener::scheduleSupportedCommandsUpdate()
73{
74 if (!m_updateCommandsTask.isPending()) {
75 m_updateCommandsTask.scheduleTask([this] () {
76 updateSupportedCommands();
77 });
78 }
79}
80
81void RemoteCommandListener::setSupportsSeeking(bool supports)
82{
83 if (m_supportsSeeking == supports)
84 return;
85
86 m_supportsSeeking = supports;
87 scheduleSupportedCommandsUpdate();
88}
89
90void RemoteCommandListener::addSupportedCommand(PlatformMediaSession::RemoteControlCommandType command)
91{
92 m_supportedCommands.add(command);
93 scheduleSupportedCommandsUpdate();
94}
95
96void RemoteCommandListener::removeSupportedCommand(PlatformMediaSession::RemoteControlCommandType command)
97{
98 m_supportedCommands.remove(command);
99 scheduleSupportedCommandsUpdate();
100}

102void RemoteCommandListener::setSupportedCommands(const RemoteCommandsSet& commands)
103{
104 m_supportedCommands = commands;
105 scheduleSupportedCommandsUpdate();
106}
107
108void RemoteCommandListener::updateSupportedCommands()

110 ASSERT_NOT_REACHED();
111}

113const RemoteCommandListener::RemoteCommandsSet& RemoteCommandListener::supportedCommands() const
