33namespace WebCore {
34
35#if !ASSERT_MSG_DISABLED
36static bool presentingApplicationPIDOverrideWasQueried;
37#endif
38
39static std::optional<int>& presentingApplicationPIDOverride()
40{
41 static NeverDestroyed<std::optional<int>> pid;
42#if !ASSERT_MSG_DISABLED
43 presentingApplicationPIDOverrideWasQueried = true;
44#endif
45 return pid;
46}
47
48int presentingApplicationPID()
49{
50 const auto& pid = presentingApplicationPIDOverride();
51 ASSERT(!pid || RunLoop::isMain());
52 return pid ? pid.value() : getCurrentProcessID();
53}
54
55void setPresentingApplicationPID(int pid)
56{
57 ASSERT(RunLoop::isMain());
58 ASSERT_WITH_MESSAGE(!presentingApplicationPIDOverrideWasQueried, "presentingApplicationPID() should not be called before setPresentingApplicationPID()");
59 presentingApplicationPIDOverride() = pid;
60}
61
62static std::optional<AuxiliaryProcessType>& auxiliaryProcessType()
63{
64 static std::optional<AuxiliaryProcessType> auxiliaryProcessType;
65 return auxiliaryProcessType;
66}
67
68bool isInAuxiliaryProcess()
69{
70 return !!auxiliaryProcessType();
71}
72
73void setAuxiliaryProcessType(AuxiliaryProcessType type)
74{
75 auxiliaryProcessType() = type;
76}
77
78bool checkAuxiliaryProcessType(AuxiliaryProcessType type)
79{
80 auto currentType = auxiliaryProcessType();
81 if (!currentType)
82 return false;
83 return *currentType == type;
84}
