40static const double frameRate = 60;
41static const Seconds tickTime = 1_s / frameRate;
42static const Seconds minimumTimerInterval { 1_ms };
43static const double smoothFactorForProgrammaticScroll = 1;

45ScrollAnimationSmooth::PerAxisData::PerAxisData(ScrollbarOrientation orientation, const FloatPoint& position, ScrollExtentsCallback& extentsCallback)
46{
47 auto extents = extentsCallback();
48 switch (orientation) {
49 case HorizontalScrollbar:
50 currentPosition = position.x();
51 visibleLength = extents.visibleSize.width();
52 break;
53 case VerticalScrollbar:
54 currentPosition = position.y();
55 visibleLength = extents.visibleSize.height();
56 break;
57 }
58
59 desiredPosition = currentPosition;
60}

75bool ScrollAnimationSmooth::scroll(ScrollbarOrientation orientation, ScrollGranularity granularity, float step, float multiplier)
76{
77 float minScrollPosition;
78 float maxScrollPosition;
79 auto extents = m_scrollExtentsFunction();
80 if (orientation == HorizontalScrollbar) {
81 minScrollPosition = extents.minimumScrollPosition.x();
82 maxScrollPosition = extents.maximumScrollPosition.x();
83 } else {
84 minScrollPosition = extents.minimumScrollPosition.y();
85 maxScrollPosition = extents.maximumScrollPosition.y();
86 }
87 bool needToScroll = updatePerAxisData(orientation == HorizontalScrollbar ? m_horizontalData : m_verticalData, granularity, step * multiplier, minScrollPosition, maxScrollPosition);
88 if (needToScroll && !isActive()) {
89 m_startTime = orientation == HorizontalScrollbar ? m_horizontalData.startTime : m_verticalData.startTime;
90 animationTimerFired();
91 }
92 return needToScroll;
93}
94
95void ScrollAnimationSmooth::scroll(const FloatPoint& position)
96{
97 ScrollGranularity granularity = ScrollByPage;
98 auto extents = m_scrollExtentsFunction();
99 bool needToScroll = updatePerAxisData(m_horizontalData, granularity, position.x() - m_horizontalData.currentPosition, extents.minimumScrollPosition.x(), extents.maximumScrollPosition.x(), smoothFactorForProgrammaticScroll);
100 needToScroll |=
101 updatePerAxisData(m_verticalData, granularity, position.y() - m_verticalData.currentPosition, extents.minimumScrollPosition.y(), extents.maximumScrollPosition.y(), smoothFactorForProgrammaticScroll);
102 if (needToScroll && !isActive()) {
103 m_startTime = m_horizontalData.startTime;
104 animationTimerFired();
105 }
106};

130static inline double curveAt(ScrollAnimationSmooth::Curve curve, double t)
131{
132 switch (curve) {
133 case ScrollAnimationSmooth::Curve::Linear:
134 return t;
135 case ScrollAnimationSmooth::Curve::Quadratic:
136 return t * t;
137 case ScrollAnimationSmooth::Curve::Cubic:
138 return t * t * t;
139 case ScrollAnimationSmooth::Curve::Quartic:
140 return t * t * t * t;
141 case ScrollAnimationSmooth::Curve::Bounce:
142 // Time base is chosen to keep the bounce points simpler:
143 // 1 (half bounce coming in) + 1 + .5 + .25
144 static const double timeBase = 2.75;
145 static const double timeBaseSquared = timeBase * timeBase;
146 if (t < 1 / timeBase)
147 return timeBaseSquared * t * t;
148 if (t < 2 / timeBase) {
149 // Invert a [-.5,.5] quadratic parabola, center it in [1,2].
150 double t1 = t - 1.5 / timeBase;
151 const double parabolaAtEdge = 1 - .5 * .5;
152 return timeBaseSquared * t1 * t1 + parabolaAtEdge;
153 }
154 if (t < 2.5 / timeBase) {
155 // Invert a [-.25,.25] quadratic parabola, center it in [2,2.5].
156 double t2 = t - 2.25 / timeBase;
157 const double parabolaAtEdge = 1 - .25 * .25;
158 return timeBaseSquared * t2 * t2 + parabolaAtEdge;
159 }
160 // Invert a [-.125,.125] quadratic parabola, center it in [2.5,2.75].
161 const double parabolaAtEdge = 1 - .125 * .125;
162 t -= 2.625 / timeBase;
163 return timeBaseSquared * t * t + parabolaAtEdge;
164 }
165 ASSERT_NOT_REACHED();
166 return 0;
167}

169static inline double attackCurve(ScrollAnimationSmooth::Curve curve, double deltaTime, double curveT, double startPosition, double attackPosition)
170{
171 double t = deltaTime / curveT;
172 double positionFactor = curveAt(curve, t);
173 return startPosition + positionFactor * (attackPosition - startPosition);
174}
175
176static inline double releaseCurve(ScrollAnimationSmooth::Curve curve, double deltaTime, double curveT, double releasePosition, double desiredPosition)
177{
178 double t = deltaTime / curveT;
179 double positionFactor = 1 - curveAt(curve, 1 - t);
180 return releasePosition + (positionFactor * (desiredPosition - releasePosition));
181}
182
183static inline double coastCurve(ScrollAnimationSmooth::Curve curve, double factor)
184{
185 return 1 - curveAt(curve, 1 - factor);
186}

188static inline double curveIntegralAt(ScrollAnimationSmooth::Curve curve, double t)
189{
190 switch (curve) {
191 case ScrollAnimationSmooth::Curve::Linear:
192 return t * t / 2;
193 case ScrollAnimationSmooth::Curve::Quadratic:
194 return t * t * t / 3;
195 case ScrollAnimationSmooth::Curve::Cubic:
196 return t * t * t * t / 4;
197 case ScrollAnimationSmooth::Curve::Quartic:
198 return t * t * t * t * t / 5;
199 case ScrollAnimationSmooth::Curve::Bounce:
200 static const double timeBase = 2.75;
201 static const double timeBaseSquared = timeBase * timeBase;
202 static const double timeBaseSquaredOverThree = timeBaseSquared / 3;
203 double area;
204 double t1 = std::min(t, 1 / timeBase);
205 area = timeBaseSquaredOverThree * t1 * t1 * t1;
206 if (t < 1 / timeBase)
207 return area;
208
209 t1 = std::min(t - 1 / timeBase, 1 / timeBase);
210 // The integral of timeBaseSquared * (t1 - .5 / timeBase) * (t1 - .5 / timeBase) + parabolaAtEdge
211 static const double secondInnerOffset = timeBaseSquared * .5 / timeBase;
212 double bounceArea = t1 * (t1 * (timeBaseSquaredOverThree * t1 - secondInnerOffset) + 1);
213 area += bounceArea;
214 if (t < 2 / timeBase)
215 return area;
216
217 t1 = std::min(t - 2 / timeBase, 0.5 / timeBase);
218 // The integral of timeBaseSquared * (t1 - .25 / timeBase) * (t1 - .25 / timeBase) + parabolaAtEdge
219 static const double thirdInnerOffset = timeBaseSquared * .25 / timeBase;
220 bounceArea = t1 * (t1 * (timeBaseSquaredOverThree * t1 - thirdInnerOffset) + 1);
221 area += bounceArea;
222 if (t < 2.5 / timeBase)
223 return area;
224
225 t1 = t - 2.5 / timeBase;
226 // The integral of timeBaseSquared * (t1 - .125 / timeBase) * (t1 - .125 / timeBase) + parabolaAtEdge
227 static const double fourthInnerOffset = timeBaseSquared * .125 / timeBase;
228 bounceArea = t1 * (t1 * (timeBaseSquaredOverThree * t1 - fourthInnerOffset) + 1);
229 area += bounceArea;
230 return area;
231 }
232 ASSERT_NOT_REACHED();
233 return 0;
234}

236static inline double attackArea(ScrollAnimationSmooth::Curve curve, double startT, double endT)
237{
238 double startValue = curveIntegralAt(curve, startT);
239 double endValue = curveIntegralAt(curve, endT);
240 return endValue - startValue;
241}
242
243static inline double releaseArea(ScrollAnimationSmooth::Curve curve, double startT, double endT)
244{
245 double startValue = curveIntegralAt(curve, 1 - endT);
246 double endValue = curveIntegralAt(curve, 1 - startT);
247 return endValue - startValue;
248}

250static inline void getAnimationParametersForGranularity(ScrollGranularity granularity, Seconds& animationTime, Seconds& repeatMinimumSustainTime, Seconds& attackTime, Seconds& releaseTime, ScrollAnimationSmooth::Curve& coastTimeCurve, Seconds& maximumCoastTime)
251{
252 switch (granularity) {
253 case ScrollByDocument:
254 animationTime = tickTime * 10;
255 repeatMinimumSustainTime = tickTime * 10;
256 attackTime = tickTime * 10;
257 releaseTime = tickTime * 10;
258 coastTimeCurve = ScrollAnimationSmooth::Curve::Linear;
259 maximumCoastTime = 1_s;
260 break;
261 case ScrollByLine:
262 animationTime = tickTime * 10;
263 repeatMinimumSustainTime = tickTime * 7;
264 attackTime = tickTime * 3;
265 releaseTime = tickTime * 3;
266 coastTimeCurve = ScrollAnimationSmooth::Curve::Linear;
267 maximumCoastTime = 1_s;
268 break;
269 case ScrollByPage:
270 animationTime = tickTime * 15;
271 repeatMinimumSustainTime = tickTime * 10;
272 attackTime = tickTime * 5;
273 releaseTime = tickTime * 5;
274 coastTimeCurve = ScrollAnimationSmooth::Curve::Linear;
275 maximumCoastTime = 1_s;
276 break;
277 case ScrollByPixel:
278 animationTime = tickTime * 11;
279 repeatMinimumSustainTime = tickTime * 2;
280 attackTime = tickTime * 3;
281 releaseTime = tickTime * 3;
282 coastTimeCurve = ScrollAnimationSmooth::Curve::Quadratic;
283 maximumCoastTime = 1250_ms;
284 break;
285 default:
286 ASSERT_NOT_REACHED();
287 }

290bool ScrollAnimationSmooth::updatePerAxisData(PerAxisData& data, ScrollGranularity granularity, float delta, float minScrollPosition, float maxScrollPosition, double smoothFactor)
291{
292 if (!data.startTime || !delta || (delta < 0) != (data.desiredPosition - data.currentPosition < 0)) {
293 data.desiredPosition = data.currentPosition;
294 data.startTime = { };
295 }
296 float newPosition = data.desiredPosition + delta;
297
298 newPosition = std::max(std::min(newPosition, maxScrollPosition), minScrollPosition);
299
300 if (newPosition == data.desiredPosition)
301 return false;
302
303 Seconds animationTime, repeatMinimumSustainTime, attackTime, releaseTime, maximumCoastTime;
304 Curve coastTimeCurve;
305 getAnimationParametersForGranularity(granularity, animationTime, repeatMinimumSustainTime, attackTime, releaseTime, coastTimeCurve, maximumCoastTime);
306
307 animationTime *= smoothFactor;
308 repeatMinimumSustainTime *= smoothFactor;
309 attackTime *= smoothFactor;
310 releaseTime *= smoothFactor;
311 maximumCoastTime *= smoothFactor;
312
313 data.desiredPosition = newPosition;
314 if (!data.startTime)
315 data.attackTime = attackTime;
316 data.animationTime = animationTime;
317 data.releaseTime = releaseTime;
318
319 // Prioritize our way out of over constraint.
320 if (data.attackTime + data.releaseTime > data.animationTime) {
321 if (data.releaseTime > data.animationTime)
322 data.releaseTime = data.animationTime;
323 data.attackTime = data.animationTime - data.releaseTime;
324 }
325
326 if (!data.startTime) {
327 // FIXME: This should be the time from the event that got us here.
328 data.startTime = MonotonicTime::now() - tickTime / 2.;
329 data.startPosition = data.currentPosition;
330 data.lastAnimationTime = data.startTime;
331 }
332 data.startVelocity = data.currentVelocity;
333
334 double remainingDelta = data.desiredPosition - data.currentPosition;
335 double attackAreaLeft = 0;
336 Seconds deltaTime = data.lastAnimationTime - data.startTime;
337 Seconds attackTimeLeft = std::max(0_s, data.attackTime - deltaTime);
338 Seconds timeLeft = data.animationTime - deltaTime;
339 Seconds minTimeLeft = data.releaseTime + std::min(repeatMinimumSustainTime, data.animationTime - data.releaseTime - attackTimeLeft);
340 if (timeLeft < minTimeLeft) {
341 data.animationTime = deltaTime + minTimeLeft;
342 timeLeft = minTimeLeft;
343 }
344
345 if (maximumCoastTime > (repeatMinimumSustainTime + releaseTime)) {
346 double targetMaxCoastVelocity = data.visibleLength * .25 * frameRate;
347 // This needs to be as minimal as possible while not being intrusive to page up/down.
348 double minCoastDelta = data.visibleLength;
349
350 if (fabs(remainingDelta) > minCoastDelta) {
351 double maxCoastDelta = maximumCoastTime.value() * targetMaxCoastVelocity;
352 double coastFactor = std::min(1., (fabs(remainingDelta) - minCoastDelta) / (maxCoastDelta - minCoastDelta));
353
354 // We could play with the curve here - linear seems a little soft. Initial testing makes me want to feed into the sustain time more aggressively.
355 Seconds coastMinTimeLeft = std::min(maximumCoastTime, minTimeLeft + (maximumCoastTime - minTimeLeft) * coastCurve(coastTimeCurve, coastFactor));
356
357 if (Seconds additionalTime = std::max(0_s, coastMinTimeLeft - minTimeLeft)) {
358 Seconds additionalReleaseTime = std::min(additionalTime, additionalTime * (releaseTime / (releaseTime + repeatMinimumSustainTime)));
359 data.releaseTime = releaseTime + additionalReleaseTime;
360 data.animationTime = deltaTime + coastMinTimeLeft;
361 timeLeft = coastMinTimeLeft;
362 }
363 }
364 }
365
366 Seconds releaseTimeLeft = std::min(timeLeft, data.releaseTime);
367 Seconds sustainTimeLeft = std::max(0_s, timeLeft - releaseTimeLeft - attackTimeLeft);
368 if (attackTimeLeft) {
369 double attackSpot = deltaTime / data.attackTime;
370 attackAreaLeft = attackArea(Curve::Cubic, attackSpot, 1) * data.attackTime.value();
371 }
372
373 double releaseSpot = (data.releaseTime - releaseTimeLeft) / data.releaseTime;
374 double releaseAreaLeft = releaseArea(Curve::Cubic, releaseSpot, 1) * data.releaseTime.value();
375
376 data.desiredVelocity = remainingDelta / (attackAreaLeft + sustainTimeLeft.value() + releaseAreaLeft);
377 data.releasePosition = data.desiredPosition - data.desiredVelocity * releaseAreaLeft;
378 if (attackAreaLeft)
379 data.attackPosition = data.startPosition + data.desiredVelocity * attackAreaLeft;
380 else
381 data.attackPosition = data.releasePosition - (data.animationTime - data.releaseTime - data.attackTime).value() * data.desiredVelocity;
382
383 if (sustainTimeLeft) {
384 double roundOff = data.releasePosition - ((attackAreaLeft ? data.attackPosition : data.currentPosition) + data.desiredVelocity * sustainTimeLeft.value());
385 data.desiredVelocity += roundOff / sustainTimeLeft.value();
386 }
387
388 return true;
389}

391bool ScrollAnimationSmooth::animateScroll(PerAxisData& data, MonotonicTime currentTime)
392{
393 if (!data.startTime)
394 return false;
395
396 Seconds lastScrollInterval = currentTime - data.lastAnimationTime;
397 if (lastScrollInterval < minimumTimerInterval)
398 return true;
399
400 data.lastAnimationTime = currentTime;
401
402 Seconds deltaTime = currentTime - data.startTime;
403 double newPosition = data.currentPosition;
404
405 if (deltaTime > data.animationTime) {
406 data = PerAxisData(data.desiredPosition, data.visibleLength);
407 return false;
408 }
409 if (deltaTime < data.attackTime)
410 newPosition = attackCurve(Curve::Cubic, deltaTime.value(), data.attackTime.value(), data.startPosition, data.attackPosition);
411 else if (deltaTime < (data.animationTime - data.releaseTime))
412 newPosition = data.attackPosition + (deltaTime - data.attackTime).value() * data.desiredVelocity;
413 else {
414 // release is based on targeting the exact final position.
415 Seconds releaseDeltaT = deltaTime - (data.animationTime - data.releaseTime);
416 newPosition = releaseCurve(Curve::Cubic, releaseDeltaT.value(), data.releaseTime.value(), data.releasePosition, data.desiredPosition);
417 }
418
419 // Normalize velocity to a per second amount. Could be used to check for jank.
420 if (lastScrollInterval > 0_s)
421 data.currentVelocity = (newPosition - data.currentPosition) / lastScrollInterval.value();
422 data.currentPosition = newPosition;
423
424 return true;
425}

427void ScrollAnimationSmooth::animationTimerFired()
428{
429 MonotonicTime currentTime = MonotonicTime::now();
430 Seconds deltaToNextFrame = 1_s * ceil((currentTime - m_startTime).value() * frameRate) / frameRate - (currentTime - m_startTime);
431 currentTime += deltaToNextFrame;
432
433 bool continueAnimation = false;
434 if (animateScroll(m_horizontalData, currentTime))
435 continueAnimation = true;
436 if (animateScroll(m_verticalData, currentTime))
437 continueAnimation = true;
438
439 if (continueAnimation)
440 startNextTimer(std::max(minimumTimerInterval, deltaToNextFrame));
441 else
442 m_notifyAnimationStoppedFunction();
443
444 m_notifyPositionChangedFunction(FloatPoint(m_horizontalData.currentPosition, m_verticalData.currentPosition));
445}
