32namespace WebCore {
33
34class FloatPoint;
35class ScrollableArea;
36enum class ScrollClamping : bool;
37
38class ScrollAnimationSmooth final: public ScrollAnimation {
39public:
40 using ScrollExtentsCallback = WTF::Function<ScrollExtents(void)>;
41 using NotifyPositionChangedCallback = WTF::Function<void(FloatPoint&&)>;
42 using NotifyAnimationStoppedCallback = WTF::Function<void(void)>;
43
44 ScrollAnimationSmooth(ScrollExtentsCallback&&, const FloatPoint& position, NotifyPositionChangedCallback&&, NotifyAnimationStoppedCallback&&);
45 virtual ~ScrollAnimationSmooth();
46
47 enum class Curve {
48 Linear,
49 Quadratic,
50 Cubic,
51 Quartic,
52 Bounce
53 };
54
55private:
56 bool scroll(ScrollbarOrientation, ScrollGranularity, float step, float multiplier) override;
57 void scroll(const FloatPoint&) override;
58 void stop() override;
59 void updateVisibleLengths() override;
60 void setCurrentPosition(const FloatPoint&) override;
61 bool isActive() const override;
62
63 struct PerAxisData {
64 PerAxisData() = delete;
65
66 PerAxisData(ScrollbarOrientation, const FloatPoint& position, ScrollExtentsCallback&);
67
68 PerAxisData(float position, int length)
69 : currentPosition(position)
70 , desiredPosition(position)
71 , visibleLength(length)
72 {
73 }
74
75 float currentPosition { 0 };
76 double currentVelocity { 0 };
77
78 double desiredPosition { 0 };
79 double desiredVelocity { 0 };
80
81 double startPosition { 0 };
82 MonotonicTime startTime;
83 double startVelocity { 0 };
84
85 Seconds animationTime;
86 MonotonicTime lastAnimationTime;
87
88 double attackPosition { 0 };
89 Seconds attackTime;
90 Curve attackCurve { Curve::Quadratic };
91
92 double releasePosition { 0 };
93 Seconds releaseTime;
94 Curve releaseCurve { Curve::Quadratic };
95
96 int visibleLength { 0 };
97 };
98
99 bool updatePerAxisData(PerAxisData&, ScrollGranularity, float delta, float minScrollPosition, float maxScrollPosition, double smoothFactor = 1);
100 bool animateScroll(PerAxisData&, MonotonicTime currentTime);
101
102 void requestAnimationTimerFired();
103 void startNextTimer(Seconds delay);
104 void animationTimerFired();
105
106 ScrollExtentsCallback m_scrollExtentsFunction;
107 NotifyPositionChangedCallback m_notifyPositionChangedFunction;
108 NotifyAnimationStoppedCallback m_notifyAnimationStoppedFunction;
109
110 PerAxisData m_horizontalData;
111 PerAxisData m_verticalData;
112
113 MonotonicTime m_startTime;
114 RunLoop::Timer<ScrollAnimationSmooth> m_animationTimer;
115};
