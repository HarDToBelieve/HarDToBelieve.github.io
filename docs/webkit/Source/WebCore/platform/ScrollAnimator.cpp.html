52ScrollAnimator::ScrollAnimator(ScrollableArea& scrollableArea)
53 : m_scrollableArea(scrollableArea)
54 , m_scrollController(*this)
55 , m_scrollControllerAnimationTimer(*this, &ScrollAnimator::scrollControllerAnimationTimerFired)
56 , m_scrollAnimation(makeUnique<ScrollAnimationSmooth>(
57 [this]() -> ScrollExtents {
58 return { m_scrollableArea.minimumScrollPosition(), m_scrollableArea.maximumScrollPosition(), m_scrollableArea.visibleSize() };
59 },
60 m_currentPosition,
61 [this](FloatPoint&& position) {
62 FloatSize delta = position - m_currentPosition;
63 m_currentPosition = WTFMove(position);
64 notifyPositionChanged(delta);
65 updateActiveScrollSnapIndexForOffset();
66 },
67 [this] {
68 m_scrollableArea.setScrollBehaviorStatus(ScrollBehaviorStatus::NotInAnimation);
69 }))
70{
71}

78bool ScrollAnimator::scroll(ScrollbarOrientation orientation, ScrollGranularity granularity, float step, float multiplier, OptionSet<ScrollBehavior> behavior)
79{
80 auto delta = deltaFromStep(orientation, step, multiplier);
81 if (behavior.contains(ScrollBehavior::DoDirectionalSnapping)) {
82 behavior.remove(ScrollBehavior::DoDirectionalSnapping);
83 if (!m_scrollController.usesScrollSnap())
84 return scroll(orientation, granularity, step, multiplier, behavior);
85
86 auto currentOffset = offsetFromPosition(currentPosition());
87 auto newOffset = currentOffset + delta;
88 if (orientation == HorizontalScrollbar)
89 newOffset.setX(m_scrollController.adjustScrollDestination(ScrollEventAxis::Horizontal, newOffset.x(), multiplier, currentOffset.x()));
90 else
91 newOffset.setY(m_scrollController.adjustScrollDestination(ScrollEventAxis::Vertical, newOffset.y(), multiplier, currentOffset.y()));
92 auto newDelta = newOffset - currentOffset;
93
94 if (orientation == HorizontalScrollbar)
95 return scroll(HorizontalScrollbar, granularity, newDelta.width(), 1.0, behavior);
96 return scroll(VerticalScrollbar, granularity, newDelta.height(), 1.0, behavior);
97 }
98
99#if ENABLE(SMOOTH_SCROLLING) && !PLATFORM(IOS_FAMILY)
100 if (m_scrollableArea.scrollAnimatorEnabled() && !behavior.contains(ScrollBehavior::NeverAnimate)) {
101 if (!m_scrollAnimation->isActive())
102 m_scrollAnimation->setCurrentPosition(m_currentPosition);
103 return m_scrollAnimation->scroll(orientation, granularity, step, multiplier);
104 }
105#endif
106
107 return scrollToPositionWithoutAnimation(currentPosition() + delta);
108}

115bool ScrollAnimator::scrollToPositionWithoutAnimation(const FloatPoint& position, ScrollClamping clamping)
116{
117 FloatPoint currentPosition = this->currentPosition();
118 auto adjustedPosition = clamping == ScrollClamping::Clamped ? position.constrainedBetween(scrollableArea().minimumScrollPosition(), scrollableArea().maximumScrollPosition()) : position;
119
120 // FIXME: In some cases on iOS the ScrollableArea position is out of sync with the ScrollAnimator position.
121 // When these cases are fixed, this extra check against the ScrollableArea position can be removed.
122 if (adjustedPosition == currentPosition && adjustedPosition == scrollableArea().scrollPosition() && !scrollableArea().scrollOriginChanged())
123 return false;
124
125 m_scrollAnimation->setCurrentPosition(adjustedPosition);
126 m_currentPosition = adjustedPosition;
127 notifyPositionChanged(adjustedPosition - currentPosition);
128 updateActiveScrollSnapIndexForOffset();
129 return true;
130}

137bool ScrollAnimator::scrollToPositionWithAnimation(const FloatPoint& newPosition)
138{
139 bool positionChanged = newPosition != currentPosition();
140 if (!positionChanged && !scrollableArea().scrollOriginChanged())
141 return false;
142
143 m_scrollAnimation->setCurrentPosition(m_currentPosition);
144 m_scrollAnimation->scroll(newPosition);
145 scrollableArea().setScrollBehaviorStatus(ScrollBehaviorStatus::InNonNativeAnimation);
146 return true;
147}

159FloatSize ScrollAnimator::deltaFromStep(ScrollbarOrientation orientation, float step, float multiplier)
160{
161 FloatSize delta;
162 if (orientation == HorizontalScrollbar)
163 delta.setWidth(step * multiplier);
164 else
165 delta.setHeight(step * multiplier);
166 return delta;
167}

189bool ScrollAnimator::handleWheelEvent(const PlatformWheelEvent& e)
190{
191 if (processWheelEventForScrollSnap(e))
192 return false;
193
194#if PLATFORM(COCOA)
195 // Events in the PlatformWheelEventPhase::MayBegin phase have no deltas, and therefore never passes through the scroll handling logic below.
196 // This causes us to return with an 'unhandled' return state, even though this event was successfully processed.
197 //
198 // We receive at least one PlatformWheelEventPhase::MayBegin when starting main-thread scrolling (see FrameView::wheelEvent), which can
199 // fool the scrolling thread into attempting to handle the scroll, unless we treat the event as handled here.
200 if (e.phase() == PlatformWheelEventPhase::MayBegin)
201 return true;
202#endif
203
204 Scrollbar* horizontalScrollbar = m_scrollableArea.horizontalScrollbar();
205 Scrollbar* verticalScrollbar = m_scrollableArea.verticalScrollbar();
206
207 // Accept the event if we have a scrollbar in that direction and can still
208 // scroll any further.
209 float deltaX = horizontalScrollbar ? e.deltaX() : 0;
210 float deltaY = verticalScrollbar ? e.deltaY() : 0;
211
212 bool handled = false;
213
214 IntSize maxForwardScrollDelta = m_scrollableArea.maximumScrollPosition() - m_scrollableArea.scrollPosition();
215 IntSize maxBackwardScrollDelta = m_scrollableArea.scrollPosition() - m_scrollableArea.minimumScrollPosition();
216 if ((deltaX < 0 && maxForwardScrollDelta.width() > 0)
217 || (deltaX > 0 && maxBackwardScrollDelta.width() > 0)
218 || (deltaY < 0 && maxForwardScrollDelta.height() > 0)
219 || (deltaY > 0 && maxBackwardScrollDelta.height() > 0)) {
220 handled = true;
221
222 OptionSet<ScrollBehavior> behavior(ScrollBehavior::DoDirectionalSnapping);
223 if (e.hasPreciseScrollingDeltas())
224 behavior.add(ScrollBehavior::NeverAnimate);
225
226 if (deltaY) {
227 if (e.granularity() == ScrollByPageWheelEvent) {
228 bool negative = deltaY < 0;
229 deltaY = Scrollbar::pageStepDelta(m_scrollableArea.visibleHeight());
230 if (negative)
231 deltaY = -deltaY;
232 }
233 scroll(VerticalScrollbar, ScrollByPixel, verticalScrollbar->pixelStep(), -deltaY, behavior);
234 }
235
236 if (deltaX) {
237 if (e.granularity() == ScrollByPageWheelEvent) {
238 bool negative = deltaX < 0;
239 deltaX = Scrollbar::pageStepDelta(m_scrollableArea.visibleWidth());
240 if (negative)
241 deltaX = -deltaX;
242 }
243 scroll(HorizontalScrollbar, ScrollByPixel, horizontalScrollbar->pixelStep(), -deltaX, behavior);
244 }
245 }
246 return handled;
247}

289void ScrollAnimator::immediateScrollOnAxis(ScrollEventAxis axis, float delta)
290{
291 FloatSize deltaSize;
292 if (axis == ScrollEventAxis::Horizontal)

297 scrollToOffsetWithoutAnimation(currentPosition() + deltaSize);

387FloatPoint ScrollAnimator::adjustScrollOffsetForSnappingIfNeeded(const FloatPoint& offset, ScrollSnapPointSelectionMethod method)
388{
389 if (!m_scrollController.usesScrollSnap())
390 return offset;
391
392 FloatPoint newOffset = offset;
393 newOffset.setX(adjustScrollOffsetForSnappingIfNeeded(ScrollEventAxis::Horizontal, newOffset.x(), method));
394 newOffset.setY(adjustScrollOffsetForSnappingIfNeeded(ScrollEventAxis::Vertical, newOffset.y(), method));
395 return newOffset;
396}
397
398float ScrollAnimator::adjustScrollOffsetForSnappingIfNeeded(ScrollEventAxis axis, float newOffset, ScrollSnapPointSelectionMethod method)
399{
400 if (!m_scrollController.usesScrollSnap())
401 return newOffset;
402
403 std::optional<float> originalOffset;
404 float velocity = 0.;
405 if (method == ScrollSnapPointSelectionMethod::Directional) {
406 FloatSize scrollOrigin = toFloatSize(m_scrollableArea.scrollOrigin());
407 auto currentOffset = ScrollableArea::scrollOffsetFromPosition(this->currentPosition(), scrollOrigin);
408 originalOffset = axis == ScrollEventAxis::Horizontal ? currentOffset.x() : currentOffset.y();
409 velocity = newOffset - (axis == ScrollEventAxis::Horizontal ? currentOffset.x() : currentOffset.y());
410 }
411
412 return m_scrollController.adjustScrollDestination(axis, newOffset, velocity, originalOffset);
413}
