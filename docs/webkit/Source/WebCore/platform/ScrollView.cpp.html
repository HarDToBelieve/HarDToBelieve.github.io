78bool ScrollView::setHasScrollbarInternal(RefPtr<Scrollbar>& scrollbar, ScrollbarOrientation orientation, bool hasBar, bool* contentSizeAffected)
79{
80 ASSERT(!hasBar || !avoidScrollbarCreation());
81
82 if (hasBar && !scrollbar) {
83 scrollbar = createScrollbar(orientation);
84 addChild(*scrollbar);
85 didAddScrollbar(scrollbar.get(), orientation);
86 scrollbar->styleChanged();
87 if (contentSizeAffected)
88 *contentSizeAffected = !scrollbar->isOverlayScrollbar();
89 return true;
90 }
91
92 if (!hasBar && scrollbar) {
93 bool wasOverlayScrollbar = scrollbar->isOverlayScrollbar();
94 willRemoveScrollbar(scrollbar.get(), orientation);
95 removeChild(*scrollbar);
96 scrollbar = nullptr;
97 if (contentSizeAffected)
98 *contentSizeAffected = !wasOverlayScrollbar;
99 return true;
100 }
101
102 return false;
103}

110void ScrollView::setScrollbarModes(ScrollbarMode horizontalMode, ScrollbarMode verticalMode,
111 bool horizontalLock, bool verticalLock)
112{
113 bool needsUpdate = false;
114
115 if (horizontalMode != horizontalScrollbarMode() && !m_horizontalScrollbarLock) {
116 m_horizontalScrollbarMode = horizontalMode;
117 needsUpdate = true;
118 }
119
120 if (verticalMode != verticalScrollbarMode() && !m_verticalScrollbarLock) {
121 m_verticalScrollbarMode = verticalMode;
122 needsUpdate = true;
123 }
124
125 if (horizontalLock)
126 setHorizontalScrollbarLock();
127
128 if (verticalLock)
129 setVerticalScrollbarLock();
130
131 if (!needsUpdate)
132 return;
133
134 if (platformWidget())
135 platformSetScrollbarModes();
136 else
137 updateScrollbars(scrollPosition());
138}

150void ScrollView::setCanHaveScrollbars(bool canScroll)
151{
152 ScrollbarMode newHorizontalMode;
153 ScrollbarMode newVerticalMode;
154
155 scrollbarModes(newHorizontalMode, newVerticalMode);
156
157 if (canScroll && newVerticalMode == ScrollbarAlwaysOff)
158 newVerticalMode = ScrollbarAuto;
159 else if (!canScroll)
160 newVerticalMode = ScrollbarAlwaysOff;
161
162 if (canScroll && newHorizontalMode == ScrollbarAlwaysOff)
163 newHorizontalMode = ScrollbarAuto;
164 else if (!canScroll)
165 newHorizontalMode = ScrollbarAlwaysOff;
166
167 setScrollbarModes(newHorizontalMode, newVerticalMode);
168}

450void ScrollView::scrollOffsetChangedViaPlatformWidget(const ScrollOffset& oldOffset, const ScrollOffset& newOffset)
451{
452 // We should not attempt to actually modify (paint) platform widgets if the layout phase
453 // is not complete. Instead, defer the scroll event until the layout finishes.
454 if (shouldDeferScrollUpdateAfterContentSizeChange()) {
455 // We only care about the most recent scroll position change request
456 m_deferredScrollOffsets = std::make_pair(oldOffset, newOffset);
457 return;
458 }
459
460 scrollOffsetChangedViaPlatformWidgetImpl(oldOffset, newOffset);
461}
462
463void ScrollView::handleDeferredScrollUpdateAfterContentSizeChange()
464{
465 ASSERT(!shouldDeferScrollUpdateAfterContentSizeChange());
466
467 if (!m_deferredScrollDelta && !m_deferredScrollOffsets)
468 return;
469
470 ASSERT(static_cast<bool>(m_deferredScrollDelta) != static_cast<bool>(m_deferredScrollOffsets));
471
472 if (m_deferredScrollDelta)
473 completeUpdatesAfterScrollTo(m_deferredScrollDelta.value());
474 else if (m_deferredScrollOffsets)
475 scrollOffsetChangedViaPlatformWidgetImpl(m_deferredScrollOffsets.value().first, m_deferredScrollOffsets.value().second);
476
477 m_deferredScrollDelta = std::nullopt;
478 m_deferredScrollOffsets = std::nullopt;
479}

481void ScrollView::scrollTo(const ScrollPosition& newPosition)
482{
483 LOG_WITH_STREAM(Scrolling, stream << "ScrollView::scrollTo " << newPosition << " min: " << minimumScrollPosition() << " max: " << maximumScrollPosition());
484
485 IntSize scrollDelta = newPosition - m_scrollPosition;
486 if (scrollDelta.isZero())
487 return;
488
489 m_scrollPosition = newPosition;
490
491 if (scrollbarsSuppressed())
492 return;
493
494#if USE(COORDINATED_GRAPHICS)
495 if (delegatesScrolling()) {
496 requestScrollPositionUpdate(newPosition);
497 return;
498 }
499#endif
500 // We should not attempt to actually modify layer contents if the layout phase
501 // is not complete. Instead, defer the scroll event until the layout finishes.
502 if (shouldDeferScrollUpdateAfterContentSizeChange()) {
503 ASSERT(!m_deferredScrollDelta);
504 m_deferredScrollDelta = scrollDelta;
505 return;
506 }
507
508 completeUpdatesAfterScrollTo(scrollDelta);
509}
510
511void ScrollView::completeUpdatesAfterScrollTo(const IntSize& scrollDelta)
512{
513 updateLayerPositionsAfterScrolling();
514 scrollContents(scrollDelta);
515 updateCompositingLayersAfterScrolling();
516}

518void ScrollView::setScrollPosition(const ScrollPosition& scrollPosition, const ScrollPositionChangeOptions& options)
519{
520 LOG_WITH_STREAM(Scrolling, stream << "ScrollView::setScrollPosition " << scrollPosition);
521
522 if (prohibitsScrolling())
523 return;
524
525 if (platformWidget()) {
526 platformSetScrollPosition(scrollPosition);
527 return;
528 }
529
530 if (currentScrollBehaviorStatus() == ScrollBehaviorStatus::InNonNativeAnimation)
531 scrollAnimator().cancelAnimations();
532
533 ScrollPosition newScrollPosition = (!delegatesScrolling() && options.clamping == ScrollClamping::Clamped) ? adjustScrollPositionWithinRange(scrollPosition) : scrollPosition;
534 if ((!delegatesScrolling() || currentScrollType() == ScrollType::User) && currentScrollBehaviorStatus() == ScrollBehaviorStatus::NotInAnimation && newScrollPosition == this->scrollPosition())
535 return;
536
537 if (!requestScrollPositionUpdate(newScrollPosition, currentScrollType(), options.clamping))
538 updateScrollbars(newScrollPosition);
539
540 setScrollBehaviorStatus(ScrollBehaviorStatus::NotInAnimation);
541}

556IntSize ScrollView::overhangAmount() const
557{
558 IntSize stretch;
559
560 // FIXME: use maximumScrollOffset()
561 ScrollOffset scrollOffset = this->scrollOffset();
562 if (scrollOffset.y() < 0)
563 stretch.setHeight(scrollOffset.y());
564 else if (totalContentsSize().height() && scrollOffset.y() > totalContentsSize().height() - visibleHeight())
565 stretch.setHeight(scrollOffset.y() - (totalContentsSize().height() - visibleHeight()));
566
567 if (scrollOffset.x() < 0)
568 stretch.setWidth(scrollOffset.x());
569 else if (contentsWidth() && scrollOffset.x() > contentsWidth() - visibleWidth())
570 stretch.setWidth(scrollOffset.x() - (contentsWidth() - visibleWidth()));
571
572 return stretch;
573}

588void ScrollView::updateScrollbars(const ScrollPosition& desiredPosition)
589{
590 LOG_WITH_STREAM(Scrolling, stream << "ScrollView::updateScrollbars " << desiredPosition << " isRubberBandInProgress " << isRubberBandInProgress());
591
592 if (m_inUpdateScrollbars || prohibitsScrolling() || platformWidget())
593 return;
594
595 auto scrollToPosition = [&](ScrollPosition desiredPosition) {
596 auto adjustedScrollPosition = desiredPosition;
597 if (!isRubberBandInProgress())
598 adjustedScrollPosition = adjustScrollPositionWithinRange(adjustedScrollPosition);
599
600 if (adjustedScrollPosition != scrollPosition() || scrollOriginChanged()) {
601 ScrollableArea::scrollToPositionWithoutAnimation(adjustedScrollPosition);
602 resetScrollOriginChanged();
603 }
604 };
605
606 if (!managesScrollbars()) {
607 scrollToPosition(desiredPosition);
608 return;
609 }
610
611 bool hasOverlayScrollbars = (!m_horizontalScrollbar || m_horizontalScrollbar->isOverlayScrollbar()) && (!m_verticalScrollbar || m_verticalScrollbar->isOverlayScrollbar());
612
613 // If we came in here with the view already needing a layout then do that first.
614 // (This will be the common case, e.g., when the page changes due to window resizing for example).
615 // This layout will not re-enter updateScrollbars and does not count towards our max layout pass total.
616 if (!m_scrollbarsSuppressed && !hasOverlayScrollbars) {
617 m_inUpdateScrollbars = true;
618 updateContentsSize();
619 m_inUpdateScrollbars = false;
620 }
621
622 IntRect oldScrollCornerRect = scrollCornerRect();
623
624 bool hasHorizontalScrollbar = m_horizontalScrollbar;
625 bool hasVerticalScrollbar = m_verticalScrollbar;
626
627 bool newHasHorizontalScrollbar = hasHorizontalScrollbar;
628 bool newHasVerticalScrollbar = hasVerticalScrollbar;
629
630 ScrollbarMode hScroll = m_horizontalScrollbarMode;
631 ScrollbarMode vScroll = m_verticalScrollbarMode;
632
633 if (hScroll != ScrollbarAuto)
634 newHasHorizontalScrollbar = (hScroll == ScrollbarAlwaysOn);
635 if (vScroll != ScrollbarAuto)
636 newHasVerticalScrollbar = (vScroll == ScrollbarAlwaysOn);
637
638 bool scrollbarAddedOrRemoved = false;
639
640 if (m_scrollbarsSuppressed || (hScroll != ScrollbarAuto && vScroll != ScrollbarAuto)) {
641 if (hasHorizontalScrollbar != newHasHorizontalScrollbar && (hasHorizontalScrollbar || !avoidScrollbarCreation())) {
642 if (setHasHorizontalScrollbar(newHasHorizontalScrollbar))
643 scrollbarAddedOrRemoved = true;
644 }
645
646 if (hasVerticalScrollbar != newHasVerticalScrollbar && (hasVerticalScrollbar || !avoidScrollbarCreation())) {
647 if (setHasVerticalScrollbar(newHasVerticalScrollbar))
648 scrollbarAddedOrRemoved = true;
649 }
650 } else {
651 bool sendContentResizedNotification = false;
652
653 IntSize docSize = totalContentsSize();
654 IntSize fullVisibleSize = unobscuredContentRectIncludingScrollbars().size();
655
656 if (hScroll == ScrollbarAuto)
657 newHasHorizontalScrollbar = docSize.width() > visibleWidth();
658 if (vScroll == ScrollbarAuto)
659 newHasVerticalScrollbar = docSize.height() > visibleHeight();
660
661 bool needAnotherPass = false;
662 if (!hasOverlayScrollbars) {
663 // If we ever turn one scrollbar off, do not turn the other one on. Never ever
664 // try to both gain/lose a scrollbar in the same pass.
665 if (!m_updateScrollbarsPass && docSize.width() <= fullVisibleSize.width() && docSize.height() <= fullVisibleSize.height()) {
666 if (hScroll == ScrollbarAuto)
667 newHasHorizontalScrollbar = false;
668 if (vScroll == ScrollbarAuto)
669 newHasVerticalScrollbar = false;
670 }
671 if (!newHasHorizontalScrollbar && hasHorizontalScrollbar && vScroll != ScrollbarAlwaysOn && !hasVerticalScrollbar) {
672 newHasVerticalScrollbar = false;
673 needAnotherPass = true;
674 }
675 if (!newHasVerticalScrollbar && hasVerticalScrollbar && hScroll != ScrollbarAlwaysOn && !hasHorizontalScrollbar) {
676 newHasHorizontalScrollbar = false;
677 needAnotherPass = true;
678 }
679 }
680
681 if (hasHorizontalScrollbar != newHasHorizontalScrollbar && (hasHorizontalScrollbar || !avoidScrollbarCreation())) {
682 if (scrollOrigin().y() && !newHasHorizontalScrollbar)
683 ScrollableArea::setScrollOrigin(IntPoint(scrollOrigin().x(), scrollOrigin().y() - m_horizontalScrollbar->occupiedHeight()));
684 if (m_horizontalScrollbar)
685 m_horizontalScrollbar->invalidate();
686
687 bool changeAffectsContentSize = false;
688 if (setHasHorizontalScrollbar(newHasHorizontalScrollbar, &changeAffectsContentSize)) {
689 scrollbarAddedOrRemoved = true;
690 sendContentResizedNotification |= changeAffectsContentSize;
691 }
692 }
693
694 if (hasVerticalScrollbar != newHasVerticalScrollbar && (hasVerticalScrollbar || !avoidScrollbarCreation())) {
695 if (scrollOrigin().x() && !newHasVerticalScrollbar)
696 ScrollableArea::setScrollOrigin(IntPoint(scrollOrigin().x() - m_verticalScrollbar->occupiedWidth(), scrollOrigin().y()));
697 if (m_verticalScrollbar)
698 m_verticalScrollbar->invalidate();
699
700 bool changeAffectsContentSize = false;
701 if (setHasVerticalScrollbar(newHasVerticalScrollbar, &changeAffectsContentSize)) {
702 scrollbarAddedOrRemoved = true;
703 sendContentResizedNotification |= changeAffectsContentSize;
704 }
705 }
706
707 const unsigned cMaxUpdateScrollbarsPass = 2;
708 if ((sendContentResizedNotification || needAnotherPass) && m_updateScrollbarsPass < cMaxUpdateScrollbarsPass) {
709 m_updateScrollbarsPass++;
710 availableContentSizeChanged(AvailableSizeChangeReason::ScrollbarsChanged);
711 updateContentsSize();
712 IntSize newDocSize = totalContentsSize();
713 if (newDocSize == docSize) {
714 // The layout with the new scroll state had no impact on
715 // the document's overall size, so updateScrollbars didn't get called.
716 // Recur manually.
717 updateScrollbars(desiredPosition);
718 }
719 m_updateScrollbarsPass--;
720 }
721 }
722
723 if (scrollbarAddedOrRemoved)
724 addedOrRemovedScrollbar();
725
726 // Set up the range (and page step/line step), but only do this if we're not in a nested call (to avoid
727 // doing it multiple times).
728 if (m_updateScrollbarsPass)
729 return;
730
731 m_inUpdateScrollbars = true;
732
733 if (m_horizontalScrollbar) {
734 int clientWidth = visibleWidth();
735 IntRect oldRect(m_horizontalScrollbar->frameRect());
736 IntRect hBarRect(shouldPlaceVerticalScrollbarOnLeft() && m_verticalScrollbar ? m_verticalScrollbar->occupiedWidth() : 0,
737 height() - m_horizontalScrollbar->height(),
738 width() - (m_verticalScrollbar ? m_verticalScrollbar->occupiedWidth() : 0),
739 m_horizontalScrollbar->height());
740 m_horizontalScrollbar->setFrameRect(hBarRect);
741 if (!m_scrollbarsSuppressed && oldRect != m_horizontalScrollbar->frameRect())
742 m_horizontalScrollbar->invalidate();
743
744 if (m_scrollbarsSuppressed)
745 m_horizontalScrollbar->setSuppressInvalidation(true);
746 m_horizontalScrollbar->setEnabled(contentsWidth() > clientWidth);
747 m_horizontalScrollbar->setProportion(clientWidth, contentsWidth());
748 if (m_scrollbarsSuppressed)
749 m_horizontalScrollbar->setSuppressInvalidation(false);
750 }
751
752 if (m_verticalScrollbar) {
753 int clientHeight = visibleHeight();
754 IntRect oldRect(m_verticalScrollbar->frameRect());
755 IntRect vBarRect(shouldPlaceVerticalScrollbarOnLeft() ? 0 : width() - m_verticalScrollbar->width(),
756 topContentInset(),
757 m_verticalScrollbar->width(),
758 height() - topContentInset() - (m_horizontalScrollbar ? m_horizontalScrollbar->occupiedHeight() : 0));
759 m_verticalScrollbar->setFrameRect(vBarRect);
760 if (!m_scrollbarsSuppressed && oldRect != m_verticalScrollbar->frameRect())
761 m_verticalScrollbar->invalidate();
762
763 if (m_scrollbarsSuppressed)
764 m_verticalScrollbar->setSuppressInvalidation(true);
765 m_verticalScrollbar->setEnabled(totalContentsSize().height() > clientHeight);
766 m_verticalScrollbar->setProportion(clientHeight, totalContentsSize().height());
767 if (m_scrollbarsSuppressed)
768 m_verticalScrollbar->setSuppressInvalidation(false);
769 }
770
771 updateScrollbarSteps();
772
773 if (hasHorizontalScrollbar != newHasHorizontalScrollbar || hasVerticalScrollbar != newHasVerticalScrollbar) {
774 // FIXME: Is frameRectsChanged really necessary here? Have any frame rects changed?
775 frameRectsChanged();
776 positionScrollbarLayers();
777 updateScrollCorner();
778 if (!m_horizontalScrollbar && !m_verticalScrollbar)
779 invalidateScrollCornerRect(oldScrollCornerRect);
780 }
781
782 scrollToPosition(desiredPosition);
783
784 // Make sure the scrollbar offsets are up to date.
785 if (m_horizontalScrollbar)
786 m_horizontalScrollbar->offsetDidChange();
787 if (m_verticalScrollbar)
788 m_verticalScrollbar->offsetDidChange();
789
790 m_inUpdateScrollbars = false;
791}

816void ScrollView::scrollContents(const IntSize& scrollDelta)
817{
818 HostWindow* window = hostWindow();
819 if (!window)
820 return;
821
822 // Since scrolling is double buffered, we will be blitting the scroll view's intersection
823 // with the clip rect every time to keep it smooth.
824 IntRect clipRect = windowClipRect();
825 IntRect scrollViewRect = rectToCopyOnScroll();
826 IntRect updateRect = clipRect;
827 updateRect.intersect(scrollViewRect);
828
829 // Invalidate the root view (not the backing store).
830 window->invalidateRootView(updateRect);
831
832 if (m_drawPanScrollIcon) {
833 // FIXME: the pan icon is broken when accelerated compositing is on, since it will draw under the compositing layers.
834 // https://bugs.webkit.org/show_bug.cgi?id=47837
835 int panIconDirtySquareSizeLength = 2 * (panIconSizeLength + std::max(abs(scrollDelta.width()), abs(scrollDelta.height()))); // We only want to repaint what's necessary
836 IntPoint panIconDirtySquareLocation = IntPoint(m_panScrollIconPoint.x() - (panIconDirtySquareSizeLength / 2), m_panScrollIconPoint.y() - (panIconDirtySquareSizeLength / 2));
837 IntRect panScrollIconDirtyRect = IntRect(panIconDirtySquareLocation, IntSize(panIconDirtySquareSizeLength, panIconDirtySquareSizeLength));
838 panScrollIconDirtyRect.intersect(clipRect);
839 window->invalidateContentsAndRootView(panScrollIconDirtyRect);
840 }
841
842 if (canBlitOnScroll()) { // The main frame can just blit the WebView window
843 // FIXME: Find a way to scroll subframes with this faster path
844 if (!scrollContentsFastPath(-scrollDelta, scrollViewRect, clipRect))
845 scrollContentsSlowPath(updateRect);
846 } else {
847 // We need to repaint the entire backing store. Do it now before moving the windowed plugins.
848 scrollContentsSlowPath(updateRect);
849 }
850
851 // Invalidate the overhang areas if they are visible.
852 updateOverhangAreas();
853
854 // This call will move children with native widgets (plugins) and invalidate them as well.
855 frameRectsChanged();
856
857 // Now blit the backingstore into the window which should be very fast.
858 window->invalidateRootView(IntRect());
859}

863 hostWindow()->invalidateContentsForSlowScroll(updateRect);
864}

1275void ScrollView::paint(GraphicsContext& context, const IntRect& rect, SecurityOriginPaintPolicy securityOriginPaintPolicy, EventRegionContext* eventRegionContext)
1276{
1277 if (platformWidget()) {
1278 Widget::paint(context, rect);
1279 return;
1280 }
1281
1282 if (context.paintingDisabled() && !context.performingPaintInvalidation() && !eventRegionContext)
1283 return;
1284
1285 notifyPageThatContentAreaWillPaint();
1286
1287 IntRect documentDirtyRect = rect;
1288 if (!paintsEntireContents()) {
1289 IntRect visibleAreaWithoutScrollbars(locationOfContents(), visibleContentRect(LegacyIOSDocumentVisibleRect).size());
1290 documentDirtyRect.intersect(visibleAreaWithoutScrollbars);
1291 }
1292
1293 if (!documentDirtyRect.isEmpty()) {
1294 GraphicsContextStateSaver stateSaver(context);
1295
1296 IntPoint locationOfContents = this->locationOfContents();
1297 context.translate(locationOfContents.x(), locationOfContents.y());
1298 documentDirtyRect.moveBy(-locationOfContents);
1299
1300 if (!paintsEntireContents()) {
1301 context.translate(-scrollX(), -scrollY());
1302 documentDirtyRect.moveBy(scrollPosition());
1303
1304 context.clip(visibleContentRect(LegacyIOSDocumentVisibleRect));
1305 }
1306
1307 paintContents(context, documentDirtyRect, securityOriginPaintPolicy, eventRegionContext);
1308 }
1309
1310#if ENABLE(RUBBER_BANDING)
1311 if (!layerForOverhangAreas())
1312 calculateAndPaintOverhangAreas(context, rect);
1313#else
1314 calculateAndPaintOverhangAreas(context, rect);
1315#endif
1316
1317 // Now paint the scrollbars.
1318 if (!m_scrollbarsSuppressed && (m_horizontalScrollbar || m_verticalScrollbar)) {
1319 GraphicsContextStateSaver stateSaver(context);
1320 IntRect scrollViewDirtyRect = rect;
1321 IntRect visibleAreaWithScrollbars(location(), unobscuredContentRectIncludingScrollbars().size());
1322 scrollViewDirtyRect.intersect(visibleAreaWithScrollbars);
1323 context.translate(x(), y());
1324 scrollViewDirtyRect.moveBy(-location());
1325 context.clip(IntRect(IntPoint(), visibleAreaWithScrollbars.size()));
1326
1327 paintScrollbars(context, scrollViewDirtyRect);
1328 }
1329
1330 // Paint the panScroll Icon
1331 if (m_drawPanScrollIcon)
1332 paintPanScrollIcon(context);
1333}

1335void ScrollView::calculateOverhangAreasForPainting(IntRect& horizontalOverhangRect, IntRect& verticalOverhangRect)
1336{
1337 IntSize scrollbarSpace = scrollbarIntrusion();
1338
1339 // FIXME: use maximumScrollOffset().
1340 ScrollOffset scrollOffset = scrollOffsetFromPosition(scrollPosition());
1341 if (scrollOffset.y() < 0) {
1342 horizontalOverhangRect = frameRect();
1343 horizontalOverhangRect.setHeight(-scrollOffset.y());
1344 horizontalOverhangRect.setWidth(horizontalOverhangRect.width() - scrollbarSpace.width());
1345 } else if (totalContentsSize().height() && scrollOffset.y() > totalContentsSize().height() - visibleHeight()) {
1346 int height = scrollOffset.y() - (totalContentsSize().height() - visibleHeight());
1347 horizontalOverhangRect = frameRect();
1348 horizontalOverhangRect.setY(frameRect().maxY() - height - scrollbarSpace.height());
1349 horizontalOverhangRect.setHeight(height);
1350 horizontalOverhangRect.setWidth(horizontalOverhangRect.width() - scrollbarSpace.width());
1351 }
1352
1353 if (scrollOffset.x() < 0) {
1354 verticalOverhangRect.setWidth(-scrollOffset.x());
1355 verticalOverhangRect.setHeight(frameRect().height() - horizontalOverhangRect.height() - scrollbarSpace.height());
1356 verticalOverhangRect.setX(frameRect().x());
1357 if (horizontalOverhangRect.y() == frameRect().y())
1358 verticalOverhangRect.setY(frameRect().y() + horizontalOverhangRect.height());
1359 else
1360 verticalOverhangRect.setY(frameRect().y());
1361 } else if (contentsWidth() && scrollOffset.x() > contentsWidth() - visibleWidth()) {
1362 int width = scrollOffset.x() - (contentsWidth() - visibleWidth());
1363 verticalOverhangRect.setWidth(width);
1364 verticalOverhangRect.setHeight(frameRect().height() - horizontalOverhangRect.height() - scrollbarSpace.height());
1365 verticalOverhangRect.setX(frameRect().maxX() - width - scrollbarSpace.width());
1366 if (horizontalOverhangRect.y() == frameRect().y())
1367 verticalOverhangRect.setY(frameRect().y() + horizontalOverhangRect.height());
1368 else
1369 verticalOverhangRect.setY(frameRect().y());
1370 }
1371}

1403bool ScrollView::isPointInScrollbarCorner(const IntPoint& windowPoint)
1404{
1405 if (!scrollbarCornerPresent())
1406 return false;
1407
1408 IntPoint viewPoint = convertFromContainingWindow(windowPoint);
1409
1410 if (m_horizontalScrollbar) {
1411 int horizontalScrollbarYMin = m_horizontalScrollbar->frameRect().y();
1412 int horizontalScrollbarYMax = m_horizontalScrollbar->frameRect().y() + m_horizontalScrollbar->frameRect().height();
1413 int horizontalScrollbarXMin = m_horizontalScrollbar->frameRect().x() + m_horizontalScrollbar->frameRect().width();
1414
1415 return viewPoint.y() > horizontalScrollbarYMin && viewPoint.y() < horizontalScrollbarYMax && viewPoint.x() > horizontalScrollbarXMin;
1416 }
1417
1418 int verticalScrollbarXMin = m_verticalScrollbar->frameRect().x();
1419 int verticalScrollbarXMax = m_verticalScrollbar->frameRect().x() + m_verticalScrollbar->frameRect().width();
1420 int verticalScrollbarYMin = m_verticalScrollbar->frameRect().y() + m_verticalScrollbar->frameRect().height();
