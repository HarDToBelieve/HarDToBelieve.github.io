103bool ScrollableArea::scroll(ScrollDirection direction, ScrollGranularity granularity, float multiplier)
104{
105 ScrollbarOrientation orientation;
106 Scrollbar* scrollbar;
107 if (direction == ScrollUp || direction == ScrollDown) {
108 orientation = VerticalScrollbar;
109 scrollbar = verticalScrollbar();
110 } else {
111 orientation = HorizontalScrollbar;
112 scrollbar = horizontalScrollbar();
113 }
114
115 if (!scrollbar)
116 return false;
117
118 float step = 0;
119 switch (granularity) {
120 case ScrollByLine:
121 step = scrollbar->lineStep();
122 break;
123 case ScrollByPage:
124 step = scrollbar->pageStep();
125 break;
126 case ScrollByDocument:
127 step = scrollbar->totalSize();
128 break;
129 case ScrollByPixel:
130 step = scrollbar->pixelStep();
131 break;
132 }
133
134 if (direction == ScrollUp || direction == ScrollLeft)
135 multiplier = -multiplier;
136
137 step = adjustScrollStepForFixedContent(step, orientation, granularity);
138 return scrollAnimator().scroll(orientation, granularity, step, multiplier, ScrollAnimator::ScrollBehavior::DoDirectionalSnapping);
139}

174void ScrollableArea::scrollPositionChanged(const ScrollPosition& position)
175{
176 IntPoint oldPosition = scrollPosition();
177 // Tell the derived class to scroll its contents.
178 setScrollOffset(scrollOffsetFromPosition(position));
179
180 Scrollbar* verticalScrollbar = this->verticalScrollbar();
181
182 // Tell the scrollbars to update their thumb postions.
183 if (Scrollbar* horizontalScrollbar = this->horizontalScrollbar()) {
184 horizontalScrollbar->offsetDidChange();
185 if (horizontalScrollbar->isOverlayScrollbar() && !hasLayerForHorizontalScrollbar()) {
186 if (!verticalScrollbar)
187 horizontalScrollbar->invalidate();
188 else {
189 // If there is both a horizontalScrollbar and a verticalScrollbar,
190 // then we must also invalidate the corner between them.
191 IntRect boundsAndCorner = horizontalScrollbar->boundsRect();
192 boundsAndCorner.setWidth(boundsAndCorner.width() + verticalScrollbar->width());
193 horizontalScrollbar->invalidateRect(boundsAndCorner);
194 }
195 }
196 }
197 if (verticalScrollbar) {
198 verticalScrollbar->offsetDidChange();
199 if (verticalScrollbar->isOverlayScrollbar() && !hasLayerForVerticalScrollbar())
200 verticalScrollbar->invalidate();
201 }
202
203 if (scrollPosition() != oldPosition)
204 scrollAnimator().notifyContentAreaScrolled(scrollPosition() - oldPosition);
205}

516void ScrollableArea::resnapAfterLayout()
517{
518 LOG_WITH_STREAM(ScrollSnap, stream << *this << " updateScrollSnapState: isScrollSnapInProgress " << isScrollSnapInProgress() << " isUserScrollInProgress " << isUserScrollInProgress());
519
520 ScrollAnimator* scrollAnimator = existingScrollAnimator();
521 if (!scrollAnimator || isScrollSnapInProgress() || isUserScrollInProgress())
522 return;
523
524 scrollAnimator->resnapAfterLayout();
525
526 const auto* info = snapOffsetsInfo();
527 if (!info)
528 return;
529
530 auto currentOffset = scrollOffset();
531 auto correctedOffset = currentOffset;
532 const auto& horizontal = info->horizontalSnapOffsets;
533 auto activeHorizontalIndex = currentHorizontalSnapPointIndex();
534 if (activeHorizontalIndex)
535 correctedOffset.setX(horizontal[*activeHorizontalIndex].offset.toInt());
536
537 const auto& vertical = info->verticalSnapOffsets;
538 auto activeVerticalIndex = currentVerticalSnapPointIndex();
539 if (activeVerticalIndex)
540 correctedOffset.setY(vertical[*activeVerticalIndex].offset.toInt());
541
542 if (correctedOffset != currentOffset) {
543 LOG_WITH_STREAM(ScrollSnap, stream << " adjusting offset from " << currentOffset << " to " << correctedOffset);
544 scrollToOffsetWithoutAnimation(correctedOffset);
545 }
546}

548void ScrollableArea::doPostThumbMoveSnapping(ScrollbarOrientation orientation)
549{
550 auto* scrollAnimator = existingScrollAnimator();
551 if (!scrollAnimator)
552 return;
553
554 auto currentOffset = scrollOffset();
555 auto newOffset = currentOffset;
556 if (orientation == HorizontalScrollbar)
557 newOffset.setX(scrollAnimator->adjustScrollOffsetForSnappingIfNeeded(ScrollEventAxis::Horizontal, currentOffset.x(), ScrollSnapPointSelectionMethod::Closest));
558 else
559 newOffset.setY(scrollAnimator->adjustScrollOffsetForSnappingIfNeeded(ScrollEventAxis::Vertical, currentOffset.y(), ScrollSnapPointSelectionMethod::Closest));
560 if (newOffset == currentOffset)
561 return;
562
563 scrollAnimator->scrollToOffsetWithAnimation(newOffset);
564}

566bool ScrollableArea::isPinnedForScrollDeltaOnAxis(float scrollDelta, ScrollEventAxis axis) const
567{
568 auto scrollPosition = this->scrollPosition();
569 switch (axis) {
570 case ScrollEventAxis::Vertical:
571 if (!allowsVerticalScrolling())
572 return true;
573
574 if (scrollDelta < 0) // top
575 return scrollPosition.y() <= minimumScrollPosition().y();
576
577 if (scrollDelta > 0) // bottom
578 return scrollPosition.y() >= maximumScrollPosition().y();
579
580 break;
581 case ScrollEventAxis::Horizontal:
582 if (!allowsHorizontalScrolling())
583 return true;
584
585 if (scrollDelta < 0) // left
586 return scrollPosition.x() <= minimumScrollPosition().x();
587
588 if (scrollDelta > 0) // right
589 return scrollPosition.x() >= maximumScrollPosition().x();
590
591 break;
592 }
593
594 return false;
595}

597bool ScrollableArea::isPinnedForScrollDelta(const FloatSize& scrollDelta) const
598{
599 return (!scrollDelta.width() || isPinnedForScrollDeltaOnAxis(scrollDelta.width(), ScrollEventAxis::Horizontal))
600 && (!scrollDelta.height() || isPinnedForScrollDeltaOnAxis(scrollDelta.height(), ScrollEventAxis::Vertical));
601}

603RectEdges<bool> ScrollableArea::edgePinnedState() const
604{
605 auto scrollPosition = this->scrollPosition();
606 auto minScrollPosition = minimumScrollPosition();
607 auto maxScrollPosition = maximumScrollPosition();
608
609 bool horizontallyUnscrollable = !allowsHorizontalScrolling();
610 bool verticallyUnscrollable = !allowsVerticalScrolling();
611
612 // Top, right, bottom, left.
613 return {
614 verticallyUnscrollable || scrollPosition.y() <= minScrollPosition.y(),
615 horizontallyUnscrollable || scrollPosition.x() >= maxScrollPosition.x(),
616 verticallyUnscrollable || scrollPosition.y() >= maxScrollPosition.y(),
617 horizontallyUnscrollable || scrollPosition.x() <= minScrollPosition.x()
618 };
619}

731LayoutPoint ScrollableArea::constrainScrollPositionForOverhang(const LayoutRect& visibleContentRect, const LayoutSize& totalContentsSize, const LayoutPoint& scrollPosition, const LayoutPoint& scrollOrigin, int headerHeight, int footerHeight)
732{
733 // The viewport rect that we're scrolling shouldn't be larger than our document.
734 LayoutSize idealScrollRectSize(std::min(visibleContentRect.width(), totalContentsSize.width()), std::min(visibleContentRect.height(), totalContentsSize.height()));
735
736 LayoutRect scrollRect(scrollPosition + scrollOrigin - LayoutSize(0, headerHeight), idealScrollRectSize);
737 LayoutRect documentRect(LayoutPoint(), LayoutSize(totalContentsSize.width(), totalContentsSize.height() - headerHeight - footerHeight));
738
739 // Use intersection to constrain our ideal scroll rect by the document rect.
740 scrollRect.intersect(documentRect);
741
742 if (scrollRect.size() != idealScrollRectSize) {
743 // If the rect was clipped, restore its size, effectively pushing it "down" from the top left.
744 scrollRect.setSize(idealScrollRectSize);
745
746 // If we still clip, push our rect "up" from the bottom right.
747 scrollRect.intersect(documentRect);
748 if (scrollRect.width() < idealScrollRectSize.width())
749 scrollRect.move(-(idealScrollRectSize.width() - scrollRect.width()), 0_lu);
750 if (scrollRect.height() < idealScrollRectSize.height())
751 scrollRect.move(0_lu, -(idealScrollRectSize.height() - scrollRect.height()));
752 }
753
754 return scrollRect.location() - toLayoutSize(scrollOrigin);
755}

757LayoutPoint ScrollableArea::constrainScrollPositionForOverhang(const LayoutPoint& scrollPosition)

760}

762void ScrollableArea::computeScrollbarValueAndOverhang(float currentPosition, float totalSize, float visibleSize, float& doubleValue, float& overhangAmount)
763{
764 doubleValue = 0;
765 overhangAmount = 0;
766 float maximum = totalSize - visibleSize;
767
768 if (currentPosition < 0) {
769 // Scrolled past the top.
770 doubleValue = 0;
771 overhangAmount = -currentPosition;
772 } else if (visibleSize + currentPosition > totalSize) {
773 // Scrolled past the bottom.
774 doubleValue = 1;
775 overhangAmount = currentPosition + visibleSize - totalSize;
776 } else {
777 // Within the bounds of the scrollable area.
778 if (maximum > 0)
779 doubleValue = currentPosition / maximum;
780 else
781 doubleValue = 0;
782 }
783}
