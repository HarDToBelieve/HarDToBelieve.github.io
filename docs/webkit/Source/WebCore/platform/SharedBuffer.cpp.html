89static Vector<uint8_t> combineSegmentsData(const SharedBuffer::DataSegmentVector& segments, size_t size)
90{
91 Vector<uint8_t> combinedData;
92 combinedData.reserveInitialCapacity(size);
93 for (auto& segment : segments)
94 combinedData.append(segment.segment->data(), segment.segment->size());
95 ASSERT(combinedData.size() == size);
96 return combinedData;
97}

99void SharedBuffer::combineIntoOneSegment() const
100{
101#if ASSERT_ENABLED
102 // FIXME: We ought to be able to set this to true and have no assertions fire.
103 // Remove all instances of appending after calling this, because they are all O(n^2) algorithms since r215686.
104 // m_hasBeenCombinedIntoOneSegment = true;
105#endif
106 if (m_segments.size() <= 1)
107 return;
108
109 auto combinedData = combineSegmentsData(m_segments, m_size);
110 m_segments.clear();
111 m_segments.append({0, DataSegment::create(WTFMove(combinedData))});
112 ASSERT(m_segments.size() == 1);
113 ASSERT(internallyConsistent());
114}
115
116const uint8_t* SharedBuffer::data() const
117{
118 if (m_segments.isEmpty())
119 return nullptr;
120 combineIntoOneSegment();
121 ASSERT(internallyConsistent());
122 return m_segments[0].segment->data();
123}
124
125Vector<uint8_t> SharedBuffer::takeData()
126{
127 if (m_segments.isEmpty())
128 return { };
129
130 Vector<uint8_t> combinedData;
131 if (hasOneSegment() && WTF::holds_alternative<Vector<uint8_t>>(m_segments[0].segment->m_immutableData))
132 combinedData = std::exchange(WTF::get<Vector<uint8_t>>(m_segments[0].segment->m_immutableData), Vector<uint8_t>());
133 else
134 combinedData = combineSegmentsData(m_segments, m_size);
135
136 clear();
137 return combinedData;
138}

140SharedBufferDataView SharedBuffer::getSomeData(size_t position) const
141{
142 RELEASE_ASSERT(position < m_size);
143 auto comparator = [](const size_t& position, const DataSegmentVectorEntry& entry) {
144 return position < entry.beginPosition;
145 };
146 const DataSegmentVectorEntry* element = std::upper_bound(m_segments.begin(), m_segments.end(), position, comparator);
147 element--; // std::upper_bound gives a pointer to the element that is greater than position. We want the element just before that.
148 return { element->segment.copyRef(), position - element->beginPosition };
149}

151String SharedBuffer::toHexString() const
152{
153 StringBuilder stringBuilder;
154 forEachSegment([&](auto& segment) {
155 for (unsigned i = 0; i < segment.size(); ++i)
156 stringBuilder.append(pad('0', 2, hex(segment[i])));
157 });
158 return stringBuilder.toString();
159}

161RefPtr<ArrayBuffer> SharedBuffer::tryCreateArrayBuffer() const
162{
163 auto arrayBuffer = ArrayBuffer::tryCreateUninitialized(static_cast<unsigned>(size()), 1);
164 if (!arrayBuffer) {
165 WTFLogAlways("SharedBuffer::tryCreateArrayBuffer Unable to create buffer. Requested size was %zu\n", size());
166 return nullptr;
167 }
168
169 size_t position = 0;
170 for (const auto& segment : m_segments) {
171 memcpy(static_cast<uint8_t*>(arrayBuffer->data()) + position, segment.segment->data(), segment.segment->size());
172 position += segment.segment->size();
173 }
174
175 ASSERT(position == m_size);
176 ASSERT(internallyConsistent());
177 return arrayBuffer;
178}

180void SharedBuffer::append(const SharedBuffer& data)
181{
182 ASSERT(!m_hasBeenCombinedIntoOneSegment);
183 m_segments.reserveCapacity(m_segments.size() + data.m_segments.size());
184 for (const auto& element : data.m_segments) {
185 m_segments.uncheckedAppend({m_size, element.segment.copyRef()});
186 m_size += element.segment->size();
187 }
188 ASSERT(internallyConsistent());
189}
190
191void SharedBuffer::append(const uint8_t* data, size_t length)
192{
193 ASSERT(!m_hasBeenCombinedIntoOneSegment);
194 Vector<uint8_t> vector;
195 vector.append(data, length);
196 m_segments.append({m_size, DataSegment::create(WTFMove(vector))});
197 m_size += length;
198 ASSERT(internallyConsistent());
199}
200
201void SharedBuffer::append(Vector<uint8_t>&& data)
202{
203 ASSERT(!m_hasBeenCombinedIntoOneSegment);
204 auto dataSize = data.size();
205 m_segments.append({m_size, DataSegment::create(WTFMove(data))});
206 m_size += dataSize;
207 ASSERT(internallyConsistent());
208}

210void SharedBuffer::clear()
211{
212 m_size = 0;
213 m_segments.clear();
214 ASSERT(internallyConsistent());

217Ref<SharedBuffer> SharedBuffer::copy() const
218{
219 Ref<SharedBuffer> clone = adoptRef(*new SharedBuffer);
220 clone->m_size = m_size;
221 clone->m_segments.reserveInitialCapacity(m_segments.size());
222 for (const auto& element : m_segments)
223 clone->m_segments.uncheckedAppend({element.beginPosition, element.segment.copyRef()});
224 ASSERT(clone->internallyConsistent());
225 ASSERT(internallyConsistent());
226 return clone;
227}

235bool SharedBuffer::startsWith(const Span<const uint8_t>& prefix) const
236{
237 if (prefix.empty())
238 return true;
239
240 if (size() < prefix.size())
241 return false;
242
243 const uint8_t* prefixPtr = prefix.data();
244 size_t remaining = prefix.size();
245 for (auto& segment : m_segments) {
246 size_t amountToCompareThisTime = std::min(remaining, segment.segment->size());
247 if (memcmp(prefixPtr, segment.segment->data(), amountToCompareThisTime))
248 return false;
249 remaining -= amountToCompareThisTime;
250 if (!remaining)
251 return true;
252 prefixPtr += amountToCompareThisTime;
253 }
254 return false;
255}
256
257void SharedBuffer::copyTo(void* destination, size_t length) const
258{
259 ASSERT(length <= size());
260 auto destinationPtr = static_cast<uint8_t*>(destination);
261 auto remaining = std::min(length, size());
262 for (auto& segment : m_segments) {
263 size_t amountToCopyThisTime = std::min(remaining, segment.segment->size());
264 memcpy(destinationPtr, segment.segment->data(), amountToCopyThisTime);
265 remaining -= amountToCopyThisTime;
266 if (!remaining)
267 return;
268 destinationPtr += amountToCopyThisTime;
269 }
270}

272bool SharedBuffer::hasOneSegment() const
273{
274 auto it = begin();
275 return it != end() && ++it == end();

278#if ASSERT_ENABLED
279bool SharedBuffer::internallyConsistent() const
280{
281 size_t position = 0;
282 for (const auto& element : m_segments) {
283 if (element.beginPosition != position)
284 return false;
285 position += element.segment->size();
286 }
287 return position == m_size;
288}

325bool SharedBuffer::operator==(const SharedBuffer& other) const
326{
327 if (this == &other)
328 return true;
329
330 if (m_size != other.m_size)
331 return false;
332
333 auto thisIterator = begin();
334 size_t thisOffset = 0;
335 auto otherIterator = other.begin();
336 size_t otherOffset = 0;
337
338 while (thisIterator != end() && otherIterator != other.end()) {
339 auto& thisSegment = thisIterator->segment.get();
340 auto& otherSegment = otherIterator->segment.get();
341
342 if (&thisSegment == &otherSegment && !thisOffset && !otherOffset) {
343 ++thisIterator;
344 ++otherIterator;
345 continue;
346 }
347
348 ASSERT(thisOffset <= thisSegment.size());
349 ASSERT(otherOffset <= otherSegment.size());
350
351 size_t thisRemaining = thisSegment.size() - thisOffset;
352 size_t otherRemaining = otherSegment.size() - otherOffset;
353 size_t remaining = std::min(thisRemaining, otherRemaining);
354
355 if (memcmp(thisSegment.data() + thisOffset, otherSegment.data() + otherOffset, remaining))
356 return false;
357
358 thisOffset += remaining;
359 otherOffset += remaining;
360
361 if (thisOffset == thisSegment.size()) {
362 ++thisIterator;
363 thisOffset = 0;
364 }
365
366 if (otherOffset == otherSegment.size()) {
367 ++otherIterator;
368 otherOffset = 0;
369 }
370 }
371 return true;
372}

409RefPtr<SharedBuffer> utf8Buffer(const String& string)
410{
411 // Allocate a buffer big enough to hold all the characters.
412 const size_t length = string.length();
413 if constexpr (String::MaxLength > std::numeric_limits<size_t>::max() / 3) {
414 if (length > std::numeric_limits<size_t>::max() / 3)
415 return nullptr;
416 }
417
418 Vector<uint8_t> buffer(length * 3);
419
420 // Convert to runs of 8-bit characters.
421 char* p = reinterpret_cast<char*>(buffer.data());
422 if (length) {
423 if (string.is8Bit()) {
424 const LChar* d = string.characters8();
425 if (!WTF::Unicode::convertLatin1ToUTF8(&d, d + length, &p, p + buffer.size()))
426 return nullptr;
427 } else {
428 const UChar* d = string.characters16();
429 if (WTF::Unicode::convertUTF16ToUTF8(&d, d + length, &p, p + buffer.size()) != WTF::Unicode::ConversionOK)
430 return nullptr;
431 }
432 }
433
434 buffer.shrink(p - reinterpret_cast<char*>(buffer.data()));
435 return SharedBuffer::create(WTFMove(buffer));
436}
