26#pragma once
27
28#include "Pasteboard.h"
29#include <wtf/HashMap.h>
30#include <wtf/Vector.h>
31#include <wtf/text/StringHash.h>
32
33namespace WebCore {
34
35class SharedBuffer;
36
37class StaticPasteboard final : public Pasteboard {
38public:
39 StaticPasteboard();
40 ~StaticPasteboard();
41
42 PasteboardCustomData takeCustomData();
43
44 bool hasNonDefaultData() const;
45 bool isStatic() const final { return true; }
46
47 bool hasData() final;
48 Vector<String> typesSafeForBindings(const String&) final;
49 Vector<String> typesForLegacyUnsafeBindings() final;
50 String readOrigin() final { return { }; }
51 String readString(const String& type) final;
52 String readStringInCustomData(const String& type) final;
53
54 void writeString(const String& type, const String& data) final;
55 void writeData(const String& type, Ref<SharedBuffer>&& data);
56 void writeStringInCustomData(const String& type, const String& data);
57 void clear() final;
58 void clear(const String& type) final;
59
60 void read(PasteboardPlainText&, PlainTextURLReadingPolicy = PlainTextURLReadingPolicy::AllowURL, std::optional<size_t> = std::nullopt) final { }
61 void read(PasteboardWebContentReader&, WebContentReadingPolicy, std::optional<size_t> = std::nullopt) final { }
62
63 void write(const PasteboardURL&) final;
64 void write(const PasteboardImage&) final;
65 void write(const PasteboardWebContent&) final;
66 void writeMarkup(const String&) final;
67 void writePlainText(const String&, SmartReplaceOption) final;
68
69 void writeCustomData(const Vector<PasteboardCustomData>&) final { }
70
71 Pasteboard::FileContentState fileContentState() final { return m_fileContentState; }
72 bool canSmartReplace() final { return false; }
73
74#if ENABLE(DRAG_SUPPORT)
75 void setDragImage(DragImage, const IntPoint&) final { }
76#endif
77
78private:
79 PasteboardCustomData m_customData;
80 HashSet<String> m_nonDefaultDataTypes;
81 Pasteboard::FileContentState m_fileContentState { Pasteboard::FileContentState::NoFileOrImageData };
82};
83
84}
85
86SPECIALIZE_TYPE_TRAITS_BEGIN(WebCore::StaticPasteboard)
87 static bool isType(const WebCore::Pasteboard& pasteboard) { return pasteboard.isStatic(); }
88SPECIALIZE_TYPE_TRAITS_END()
