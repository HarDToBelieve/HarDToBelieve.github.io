33namespace WebCore {
34
35SystemSoundManager& SystemSoundManager::singleton()
36{
37 static NeverDestroyed<SystemSoundManager> systemSoundManager;
38 return systemSoundManager;
39}
40
41void SystemSoundManager::setSystemSoundDelegate(std::unique_ptr<SystemSoundDelegate>&& delegate)
42{
43 m_delegate = WTFMove(delegate);
44}
45
46void SystemSoundManager::systemBeep()
47{
48 if (m_delegate) {
49 m_delegate->systemBeep();
50 return;
51 }
52 PAL::systemBeep();
53}
