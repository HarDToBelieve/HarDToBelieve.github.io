42// Fire timers for this length of time, and then quit to let the run loop process user input events.
43static constexpr auto maxDurationOfFiringTimers { 16_ms };
44
45// Timers are created, started and fired on the same thread, and each thread has its own ThreadTimers
46// copy to keep the heap and a set of currently firing timers.
47
48ThreadTimers::ThreadTimers()
49{
50 if (isUIThread())
51 setSharedTimer(&MainThreadSharedTimer::singleton());
52}
53
54// A worker thread may initialize SharedTimer after some timers are created.
55// Also, SharedTimer can be replaced with nullptr before all timers are destroyed.
56void ThreadTimers::setSharedTimer(SharedTimer* sharedTimer)
57{
58 if (m_sharedTimer) {
59 m_sharedTimer->setFiredFunction(nullptr);
60 m_sharedTimer->stop();
61 m_pendingSharedTimerFireTime = MonotonicTime { };
62 }
63
64 m_sharedTimer = sharedTimer;
65
66 if (sharedTimer) {
67 m_sharedTimer->setFiredFunction([] { threadGlobalData().threadTimers().sharedTimerFiredInternal(); });
68 updateSharedTimer();
69 }
70}
71
72void ThreadTimers::updateSharedTimer()
73{
74 if (!m_sharedTimer)
75 return;
76
77 while (!m_timerHeap.isEmpty() && !m_timerHeap.first()->hasTimer()) {
78 ASSERT_NOT_REACHED();
79 TimerBase::heapDeleteNullMin(m_timerHeap);
80 }
81 ASSERT(m_timerHeap.isEmpty() || m_timerHeap.first()->hasTimer());
82
83 if (m_firingTimers || m_timerHeap.isEmpty()) {
84 m_pendingSharedTimerFireTime = MonotonicTime { };
85 m_sharedTimer->stop();
86 } else {
87 MonotonicTime nextFireTime = m_timerHeap.first()->time;
88 MonotonicTime currentMonotonicTime = MonotonicTime::now();
89 if (m_pendingSharedTimerFireTime) {
90 // No need to restart the timer if both the pending fire time and the new fire time are in the past.
91 if (m_pendingSharedTimerFireTime <= currentMonotonicTime && nextFireTime <= currentMonotonicTime)
92 return;
93 }
94 m_pendingSharedTimerFireTime = nextFireTime;
95 m_sharedTimer->setFireInterval(std::max(nextFireTime - currentMonotonicTime, 0_s));
96 }
97}
98
99void ThreadTimers::sharedTimerFiredInternal()
100{
101 ASSERT(isMainThread() || (!isWebThread() && !isUIThread()));
102 // Do a re-entrancy check.
103 if (m_firingTimers)
104 return;
105 m_firingTimers = true;
106 m_pendingSharedTimerFireTime = MonotonicTime { };
107
108 MonotonicTime fireTime = MonotonicTime::now();
109 MonotonicTime timeToQuit = fireTime + maxDurationOfFiringTimers;
110
111 while (!m_timerHeap.isEmpty()) {
112 Ref<ThreadTimerHeapItem> item = *m_timerHeap.first();
113 ASSERT(item->hasTimer());
114 if (!item->hasTimer()) {
115 TimerBase::heapDeleteNullMin(m_timerHeap);
116 continue;
117 }
118
119 if (item->time > fireTime)
120 break;
121
122 auto& timer = item->timer();
123 Seconds interval = timer.repeatInterval();
124 timer.setNextFireTime(interval ? fireTime + interval : MonotonicTime { });
125
126 // Once the timer has been fired, it may be deleted, so do nothing else with it after this point.
127 item->timer().fired();
128
129 // Catch the case where the timer asked timers to fire in a nested event loop, or we are over time limit.
130 if (!m_firingTimers || timeToQuit < MonotonicTime::now())
131 break;
132
133 if (m_shouldBreakFireLoopForRenderingUpdate)
134 break;
135 }
136
137 m_firingTimers = false;
138 m_shouldBreakFireLoopForRenderingUpdate = false;
139
140 updateSharedTimer();
141}
142
143void ThreadTimers::fireTimersInNestedEventLoop()
144{
145 // Reset the reentrancy guard so the timers can fire again.
146 m_firingTimers = false;
147
148 if (m_sharedTimer) {
149 m_sharedTimer->invalidate();
150 m_pendingSharedTimerFireTime = MonotonicTime { };
151 }
152
153 updateSharedTimer();
154}
155
156void ThreadTimers::breakFireLoopForRenderingUpdate()
157{
158 if (!m_firingTimers)
159 return;
160 m_shouldBreakFireLoopForRenderingUpdate = true;
161}
