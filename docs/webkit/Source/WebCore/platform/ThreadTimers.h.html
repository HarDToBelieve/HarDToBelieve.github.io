27#ifndef ThreadTimers_h
28#define ThreadTimers_h
29
30#include <wtf/MonotonicTime.h>
31#include <wtf/Noncopyable.h>
32#include <wtf/RefCounted.h>
33#include <wtf/ThreadSafeRefCounted.h>
34#include <wtf/Vector.h>
35
36namespace WebCore {
37
38class SharedTimer;
39class ThreadTimers;
40class TimerBase;
41
42struct ThreadTimerHeapItem;
43typedef Vector<RefPtr<ThreadTimerHeapItem>> ThreadTimerHeap;
44
45// A collection of timers per thread. Kept in ThreadGlobalData.
46class ThreadTimers {
47 WTF_MAKE_NONCOPYABLE(ThreadTimers); WTF_MAKE_FAST_ALLOCATED;
48public:
49 ThreadTimers();
50
51 // On a thread different then main, we should set the thread's instance of the SharedTimer.
52 void setSharedTimer(SharedTimer*);
53
54 ThreadTimerHeap& timerHeap() { return m_timerHeap; }
55
56 void updateSharedTimer();
57 void fireTimersInNestedEventLoop();
58 void breakFireLoopForRenderingUpdate();
59
60 unsigned nextHeapInsertionCount() { return m_currentHeapInsertionOrder++; }
61
62private:
63 void sharedTimerFiredInternal();
64 void fireTimersInNestedEventLoopInternal();
65
66 ThreadTimerHeap m_timerHeap;
67 SharedTimer* m_sharedTimer { nullptr }; // External object, can be a run loop on a worker thread. Normally set/reset by worker thread.
68 bool m_firingTimers { false };
69 bool m_shouldBreakFireLoopForRenderingUpdate { false };
70 unsigned m_currentHeapInsertionOrder { 0 };
71 MonotonicTime m_pendingSharedTimerFireTime;
72};
73
74struct ThreadTimerHeapItem : ThreadSafeRefCounted<ThreadTimerHeapItem> {
75 static RefPtr<ThreadTimerHeapItem> create(TimerBase&, MonotonicTime, unsigned);
76
77 bool hasTimer() const { return m_timer; }
78 TimerBase& timer();
79 void clearTimer();
80
81 ThreadTimerHeap& timerHeap() const;
82
83 unsigned heapIndex() const;
84 void setHeapIndex(unsigned newIndex);
85 void setNotInHeap() { m_heapIndex = invalidHeapIndex; }
86
87 bool isInHeap() const { return m_heapIndex != invalidHeapIndex; }
88 bool isFirstInHeap() const { return !m_heapIndex; }
89
90 MonotonicTime time;
91 unsigned insertionOrder { 0 };
92
93private:
94 ThreadTimers& m_threadTimers;
95 TimerBase* m_timer { nullptr };
96 unsigned m_heapIndex { invalidHeapIndex };
97
98 static const unsigned invalidHeapIndex = static_cast<unsigned>(-1);
99
100 ThreadTimerHeapItem(TimerBase&, MonotonicTime, unsigned);
101};
102
103inline TimerBase& ThreadTimerHeapItem::timer()
104{
105 ASSERT(m_timer);
106 return *m_timer;
107}
108
109inline void ThreadTimerHeapItem::clearTimer()
110{
111 ASSERT(!isInHeap());
112 m_timer = nullptr;
113}
114
115inline unsigned ThreadTimerHeapItem::heapIndex() const
116{
117 ASSERT(m_heapIndex != invalidHeapIndex);
118 return static_cast<unsigned>(m_heapIndex);
119}
120
121inline void ThreadTimerHeapItem::setHeapIndex(unsigned newIndex)
122{
123 ASSERT(newIndex != invalidHeapIndex);
124 m_heapIndex = newIndex;
125}
126
127inline ThreadTimerHeap& ThreadTimerHeapItem::timerHeap() const
128{
129 return m_threadTimers.timerHeap();
130}
131
132}
