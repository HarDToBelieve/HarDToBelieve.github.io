38namespace WebCore {
39
40// The UserActivity type is used to indicate to the operating system that
41// a user initiated or visible action is taking place, and as such that
42// resources should be allocated to the process accordingly.
43class UserActivity : public PAL::HysteresisActivity {
44public:
45 class Impl {
46 public:
47 explicit Impl(const char* description);
48
49 void beginActivity();
50 void endActivity();
51
52#if HAVE(NS_ACTIVITY)
53 private:
54 RetainPtr<id> m_activity;
55 RetainPtr<NSString> m_description;
56#endif
57 };
58
59 WEBCORE_EXPORT explicit UserActivity(const char* description);
60
61private:
62 void hysteresisUpdated(PAL::HysteresisState);
63
64 Impl m_impl;
65};
66
67} // namespace WebCore
