56namespace WebCore {
57
58class ValidationBubble : public RefCounted<ValidationBubble> {
59public:
60 struct Settings {
61 double minimumFontSize { 0 };
62 };
63
64 static Ref<ValidationBubble> create(PlatformView* view, const String& message, const Settings& settings)
65 {
66 return adoptRef(*new ValidationBubble(view, message, settings));
67 }
68
69 WEBCORE_EXPORT ~ValidationBubble();
70
71 const String& message() const { return m_message; }
72 double fontSize() const { return m_fontSize; }
73
74#if PLATFORM(IOS_FAMILY)
75 WEBCORE_EXPORT void setAnchorRect(const IntRect& anchorRect, UIViewController* presentingViewController = nullptr);
76 WEBCORE_EXPORT void show();
77#else
78 WEBCORE_EXPORT void showRelativeTo(const IntRect& anchorRect);
79#endif
80
81private:
82 WEBCORE_EXPORT ValidationBubble(PlatformView*, const String& message, const Settings&);
83
84 PlatformView* m_view;
85 String m_message;
86 double m_fontSize { 0 };
87#if PLATFORM(MAC)
88 RetainPtr<NSPopover> m_popover;
89#elif PLATFORM(IOS_FAMILY)
90 RetainPtr<WebValidationBubbleViewController> m_popoverController;
91 RetainPtr<WebValidationBubbleTapRecognizer> m_tapRecognizer;
92 RetainPtr<WebValidationBubbleDelegate> m_popoverDelegate;
93 WeakObjCPtr<UIViewController> m_presentingViewController;
94#endif
95};
