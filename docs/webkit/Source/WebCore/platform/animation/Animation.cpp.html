106bool Animation::animationsMatch(const Animation& other, bool matchProperties) const
107{
108 bool result = m_name.string == other.m_name.string
109 && m_playState == other.m_playState
110 && m_playStateSet == other.m_playStateSet
111 && m_iterationCount == other.m_iterationCount
112 && m_delay == other.m_delay
113 && m_duration == other.m_duration
114 && *(m_timingFunction.get()) == *(other.m_timingFunction.get())
115 && m_nameStyleScopeOrdinal == other.m_nameStyleScopeOrdinal
116 && m_direction == other.m_direction
117 && m_fillMode == other.m_fillMode
118 && m_delaySet == other.m_delaySet
119 && m_directionSet == other.m_directionSet
120 && m_durationSet == other.m_durationSet
121 && m_fillModeSet == other.m_fillModeSet
122 && m_iterationCountSet == other.m_iterationCountSet
123 && m_nameSet == other.m_nameSet
124 && m_timingFunctionSet == other.m_timingFunctionSet
125 && m_isNone == other.m_isNone;
126
127 if (!result)
128 return false;
129
130 return !matchProperties || (m_property.mode == other.m_property.mode && m_property.id == other.m_property.id && m_propertySet == other.m_propertySet);
131}

139TextStream& operator<<(TextStream& ts, Animation::TransitionProperty transitionProperty)
140{
141 switch (transitionProperty.mode) {
142 case Animation::TransitionMode::All: ts << "all"; break;
143 case Animation::TransitionMode::None: ts << "none"; break;
144 case Animation::TransitionMode::SingleProperty: ts << getPropertyName(transitionProperty.id); break;
145 case Animation::TransitionMode::UnknownProperty: ts << "unknown property"; break;
146 }
147 return ts;
148}
149
150TextStream& operator<<(TextStream& ts, Animation::AnimationDirection direction)
151{
152 switch (direction) {
153 case Animation::AnimationDirectionNormal: ts << "normal"; break;
154 case Animation::AnimationDirectionAlternate: ts << "alternate"; break;
155 case Animation::AnimationDirectionReverse: ts << "reverse"; break;
156 case Animation::AnimationDirectionAlternateReverse: ts << "alternate-reverse"; break;
157 }
158 return ts;
159}
160
161TextStream& operator<<(TextStream& ts, const Animation& animation)
162{
163 ts.dumpProperty("property", animation.property());
164 ts.dumpProperty("name", animation.name().string);
165 ts.dumpProperty("iteration count", animation.iterationCount());
166 ts.dumpProperty("delay", animation.iterationCount());
167 ts.dumpProperty("duration", animation.duration());
168 if (animation.timingFunction())
169 ts.dumpProperty("timing function", *animation.timingFunction());
170 ts.dumpProperty("direction", animation.direction());
171 ts.dumpProperty("fill-mode", animation.fillMode());
172 ts.dumpProperty("play-state", animation.playState());
173
174 return ts;
175}
