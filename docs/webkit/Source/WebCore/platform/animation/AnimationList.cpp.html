29#define FILL_UNSET_PROPERTY(test, propGet, propSet) \
30for (i = 0; i < size() && animation(i).test(); ++i) { } \
31if (i) { \
32 for (size_t j = 0; i < size(); ++i, ++j) \
33 animation(i).propSet(animation(j).propGet()); \
34}

36AnimationList::AnimationList() = default;
37
38AnimationList::AnimationList(const AnimationList& other)
39 : RefCounted()
40{
41 m_animations.reserveInitialCapacity(other.size());
42 for (auto& animation : other.m_animations)
43 m_animations.uncheckedAppend(Animation::create(animation.get()));
44}

46void AnimationList::fillUnsetProperties()
47{
48 size_t i;
49 FILL_UNSET_PROPERTY(isDelaySet, delay, setDelay);
50 FILL_UNSET_PROPERTY(isDirectionSet, direction, setDirection);
51 FILL_UNSET_PROPERTY(isDurationSet, duration, setDuration);
52 FILL_UNSET_PROPERTY(isFillModeSet, fillMode, setFillMode);
53 FILL_UNSET_PROPERTY(isIterationCountSet, iterationCount, setIterationCount);
54 FILL_UNSET_PROPERTY(isPlayStateSet, playState, setPlayState);
55 FILL_UNSET_PROPERTY(isNameSet, name, setName);
56 FILL_UNSET_PROPERTY(isTimingFunctionSet, timingFunction, setTimingFunction);
57 FILL_UNSET_PROPERTY(isPropertySet, property, setProperty);
58}

60bool AnimationList::operator==(const AnimationList& other) const
61{
62 if (size() != other.size())
63 return false;
64 for (size_t i = 0; i < size(); ++i) {
65 if (animation(i) != other.animation(i))
66 return false;
67 }
68 return true;
69}
70
71TextStream& operator<<(TextStream& ts, const AnimationList& animationList)
72{
73 ts << "[";
74 for (size_t i = 0; i < animationList.size(); ++i) {
75 if (i > 0)
76 ts << ", ";
77 ts << animationList.animation(i);
78 }
79 ts << "]";
80 return ts;
81}
