29#ifndef AudioArray_h
30#define AudioArray_h
31
32#include <string.h>
33#include <wtf/CheckedArithmetic.h>
34#include <wtf/FastMalloc.h>
35
36namespace WebCore {
37
38template<typename T>
39class AudioArray {
40 WTF_MAKE_FAST_ALLOCATED;
41public:
42 AudioArray() = default;
43 explicit AudioArray(size_t n)
44 {
45 resize(n);
46 }
47
48 ~AudioArray()
49 {
50 fastAlignedFree(m_allocation);
51 }
52
53 // It's OK to call resize() multiple times, but data will *not* be copied from an initial allocation
54 // if re-allocated. Allocations are zero-initialized.
55 void resize(Checked<size_t> n)
56 {
57 if (n == m_size)
58 return;
59
60 Checked<size_t> initialSize = sizeof(T) * n;
61 const size_t alignment = 16;
62
63 fastAlignedFree(m_allocation);
64
65 m_allocation = static_cast<T*>(fastAlignedMalloc(alignment, initialSize));
66 if (!m_allocation)
67 CRASH();
68 m_size = n;
69 zero();
70 }
71
72 T* data() { return m_allocation; }
73 const T* data() const { return m_allocation; }
74 size_t size() const { return m_size; }
75 bool isEmpty() const { return !m_size; }
76
77 T& at(size_t i)
78 {
79 // Note that although it is a size_t, m_size is now guaranteed to be
80 // no greater than max unsigned. This guarantee is enforced in resize().
81 ASSERT_WITH_SECURITY_IMPLICATION(i < size());
82 return data()[i];
83 }
84
85 const T& at(size_t i) const
86 {
87 // Note that although it is a size_t, m_size is now guaranteed to be
88 // no greater than max unsigned. This guarantee is enforced in resize().
89 ASSERT_WITH_SECURITY_IMPLICATION(i < size());
90 return data()[i];
91 }
92
93 T& operator[](size_t i) { return at(i); }
94 const T& operator[](size_t i) const { return at(i); }
95
96 void zero()
97 {
98 // This multiplication is made safe by the check in resize().
99 memset(this->data(), 0, sizeof(T) * this->size());
100 }
101
102 void zeroRange(unsigned start, unsigned end)
103 {
104 bool isSafe = (start <= end) && (end <= this->size());
105 ASSERT(isSafe);
106 if (!isSafe)
107 return;
108
109 // This expression cannot overflow because end - start cannot be
110 // greater than m_size, which is safe due to the check in resize().
111 memset(this->data() + start, 0, sizeof(T) * (end - start));
112 }
113
114 void copyToRange(const T* sourceData, unsigned start, unsigned end)
115 {
116 bool isSafe = (start <= end) && (end <= this->size());
117 ASSERT(isSafe);
118 if (!isSafe)
119 return;
120
121 // This expression cannot overflow because end - start cannot be
122 // greater than m_size, which is safe due to the check in resize().
123 memcpy(this->data() + start, sourceData, sizeof(T) * (end - start));
124 }
125
126 bool containsConstantValue() const
127 {
128 if (m_size <= 1)
129 return true;
130 float constant = data()[0];
131 for (unsigned i = 1; i < m_size; ++i) {
132 if (data()[i] != constant)
133 return false;
134 }
135 return true;
136 }
137
138private:
139 T* m_allocation { nullptr };
140 size_t m_size { 0 };
141};
142
143typedef AudioArray<float> AudioFloatArray;
144typedef AudioArray<double> AudioDoubleArray;
145
146} // WebCore
147
148#endif // AudioArray_h
