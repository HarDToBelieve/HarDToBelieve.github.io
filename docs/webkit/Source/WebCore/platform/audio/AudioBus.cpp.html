46RefPtr<AudioBus> AudioBus::create(unsigned numberOfChannels, size_t length, bool allocate)
47{
48 ASSERT(numberOfChannels <= MaxBusChannels);
49 if (numberOfChannels > MaxBusChannels)
50 return nullptr;
51
52 return adoptRef(*new AudioBus(numberOfChannels, length, allocate));
53}

55AudioBus::AudioBus(unsigned numberOfChannels, size_t length, bool allocate)
56 : m_length(length)
57{
58 m_channels.reserveInitialCapacity(numberOfChannels);
59
60 for (unsigned i = 0; i < numberOfChannels; ++i) {
61 auto channel = allocate ? makeUnique<AudioChannel>(length) : makeUnique<AudioChannel>(nullptr, length);
62 m_channels.uncheckedAppend(WTFMove(channel));
63 }
64
65 m_layout = LayoutCanonical; // for now this is the only layout we define
66}

82AudioChannel* AudioBus::channelByType(unsigned channelType)
83{
84 // For now we only support canonical channel layouts...
85 if (m_layout != LayoutCanonical)
86 return 0;
87
88 switch (numberOfChannels()) {
89 case 1: // mono
90 if (channelType == ChannelMono || channelType == ChannelLeft)
91 return channel(0);
92 return 0;
93
94 case 2: // stereo
95 switch (channelType) {
96 case ChannelLeft: return channel(0);
97 case ChannelRight: return channel(1);
98 default: return 0;
99 }
100
101 case 4: // quad
102 switch (channelType) {
103 case ChannelLeft: return channel(0);
104 case ChannelRight: return channel(1);
105 case ChannelSurroundLeft: return channel(2);
106 case ChannelSurroundRight: return channel(3);
107 default: return 0;
108 }
109
110 case 5: // 5.0
111 switch (channelType) {
112 case ChannelLeft: return channel(0);
113 case ChannelRight: return channel(1);
114 case ChannelCenter: return channel(2);
115 case ChannelSurroundLeft: return channel(3);
116 case ChannelSurroundRight: return channel(4);
117 default: return 0;
118 }
119
120 case 6: // 5.1
121 switch (channelType) {
122 case ChannelLeft: return channel(0);
123 case ChannelRight: return channel(1);
124 case ChannelCenter: return channel(2);
125 case ChannelLFE: return channel(3);
126 case ChannelSurroundLeft: return channel(4);
127 case ChannelSurroundRight: return channel(5);
128 default: return 0;
129 }
130 }
131
132 ASSERT_NOT_REACHED();
133 return 0;
134}

141// Returns true if the channel count and frame-size match.
142bool AudioBus::topologyMatches(const AudioBus& bus) const
143{
144 if (numberOfChannels() != bus.numberOfChannels())
145 return false; // channel mismatch
146
147 // Make sure source bus has enough frames.
148 if (length() > bus.length())
149 return false; // frame-size mismatch
150
151 return true;
152}

154RefPtr<AudioBus> AudioBus::createBufferFromRange(const AudioBus* sourceBuffer, unsigned startFrame, unsigned endFrame)
155{
156 size_t numberOfSourceFrames = sourceBuffer->length();
157 unsigned numberOfChannels = sourceBuffer->numberOfChannels();
158
159 // Sanity checking
160 bool isRangeSafe = startFrame < endFrame && endFrame <= numberOfSourceFrames;
161 ASSERT(isRangeSafe);
162 if (!isRangeSafe)
163 return nullptr;
164
165 size_t rangeLength = endFrame - startFrame;
166
167 RefPtr<AudioBus> audioBus = create(numberOfChannels, rangeLength);
168 audioBus->setSampleRate(sourceBuffer->sampleRate());
169
170 for (unsigned i = 0; i < numberOfChannels; ++i)
171 audioBus->channel(i)->copyFromRange(sourceBuffer->channel(i), startFrame, endFrame);
172
173 return audioBus;
174}

176float AudioBus::maxAbsValue() const
177{
178 float max = 0.0f;
179 for (unsigned i = 0; i < numberOfChannels(); ++i) {
180 const AudioChannel* channel = this->channel(i);
181 max = std::max(max, channel->maxAbsValue());
182 }
183
184 return max;
185}

200void AudioBus::copyFromRange(const AudioBus& sourceBus, unsigned startFrame, unsigned endFrame)
201{
202 if (!topologyMatches(sourceBus)) {
203 ASSERT_NOT_REACHED();
204 zero();
205 return;
206 }
207
208 size_t numberOfSourceFrames = sourceBus.length();
209 bool isRangeSafe = startFrame < endFrame && endFrame <= numberOfSourceFrames;
210 ASSERT(isRangeSafe);
211 if (!isRangeSafe) {
212 zero();
213 return;
214 }
215
216 unsigned numberOfChannels = this->numberOfChannels();
217 ASSERT(numberOfChannels <= MaxBusChannels);
218 if (numberOfChannels > MaxBusChannels) {
219 zero();
220 return;
221 }
222
223 for (unsigned i = 0; i < numberOfChannels; ++i)
224 channel(i)->copyFromRange(sourceBus.channel(i), startFrame, endFrame);
225}

227void AudioBus::copyFrom(const AudioBus& sourceBus, ChannelInterpretation channelInterpretation)
228{
229 if (&sourceBus == this)
230 return;
231
232 // Copying bus is equivalent to zeroing and then summing.
233 zero();
234 sumFrom(sourceBus, channelInterpretation);
235}
236
237void AudioBus::sumFrom(const AudioBus& sourceBus, ChannelInterpretation channelInterpretation)
238{
239 if (&sourceBus == this)
240 return;
241
242 unsigned numberOfSourceChannels = sourceBus.numberOfChannels();
243 unsigned numberOfDestinationChannels = numberOfChannels();
244
245 if (numberOfDestinationChannels == numberOfSourceChannels) {
246 for (unsigned i = 0; i < numberOfSourceChannels; ++i)
247 channel(i)->sumFrom(sourceBus.channel(i));
248 } else {
249 switch (channelInterpretation) {
250 case ChannelInterpretation::Speakers:
251 if (numberOfSourceChannels < numberOfDestinationChannels)
252 speakersSumFromByUpMixing(sourceBus);
253 else
254 speakersSumFromByDownMixing(sourceBus);
255 break;
256 case ChannelInterpretation::Discrete:
257 discreteSumFrom(sourceBus);
258 break;
259 default:
260 ASSERT_NOT_REACHED();
261 }
262 }
263}

265void AudioBus::speakersSumFromByUpMixing(const AudioBus& sourceBus)
266{
267 unsigned numberOfSourceChannels = sourceBus.numberOfChannels();
268 unsigned numberOfDestinationChannels = numberOfChannels();
269
270 if ((numberOfSourceChannels == 1 && numberOfDestinationChannels == 2)
271 || (numberOfSourceChannels == 1 && numberOfDestinationChannels == 4)) {
272 // Handle mono -> stereo case (summing mono channel into both left and right).
273 auto* sourceL = sourceBus.channelByType(ChannelLeft);
274 channelByType(ChannelLeft)->sumFrom(sourceL);
275 channelByType(ChannelRight)->sumFrom(sourceL);
276 } else if (numberOfSourceChannels == 1 && numberOfDestinationChannels == 6) {
277 // Handle mono -> 5.1 case, sum mono channel into center.
278 channelByType(ChannelCenter)->sumFrom(sourceBus.channelByType(ChannelLeft));
279 } else if ((numberOfSourceChannels == 2 && numberOfDestinationChannels == 4)
280 || (numberOfSourceChannels == 2 && numberOfDestinationChannels == 6)) {
281 // Up-mixing: 2 -> 4, 2 -> 5.1
282 channelByType(ChannelLeft)->sumFrom(sourceBus.channelByType(ChannelLeft));
283 channelByType(ChannelRight)->sumFrom(sourceBus.channelByType(ChannelRight));
284 } else if (numberOfSourceChannels == 4 && numberOfDestinationChannels == 6) {
285 // Up-mixing: 4 -> 5.1
286 channelByType(ChannelLeft)->sumFrom(sourceBus.channelByType(ChannelLeft));
287 channelByType(ChannelRight)->sumFrom(sourceBus.channelByType(ChannelRight));
288 channelByType(ChannelSurroundLeft)->sumFrom(sourceBus.channelByType(ChannelSurroundLeft));
289 channelByType(ChannelSurroundRight)->sumFrom(sourceBus.channelByType(ChannelSurroundRight));
290 } else {
291 // Fallback for unknown combinations.
292 discreteSumFrom(sourceBus);
293 }
294}

296void AudioBus::speakersSumFromByDownMixing(const AudioBus& sourceBus)
297{
298 unsigned numberOfSourceChannels = sourceBus.numberOfChannels();
299 unsigned numberOfDestinationChannels = numberOfChannels();
300
301 if (numberOfSourceChannels == 2 && numberOfDestinationChannels == 1) {
302 // Handle stereo -> mono case. output += 0.5 * (input.L + input.R).
303 AudioBus& sourceBusSafe = const_cast<AudioBus&>(sourceBus);
304
305 const float* sourceL = sourceBusSafe.channelByType(ChannelLeft)->data();
306 const float* sourceR = sourceBusSafe.channelByType(ChannelRight)->data();
307
308 float* destination = channelByType(ChannelLeft)->mutableData();
309 VectorMath::multiplyByScalarThenAddToOutput(sourceL, 0.5, destination, length());
310 VectorMath::multiplyByScalarThenAddToOutput(sourceR, 0.5, destination, length());
311 } else if (numberOfSourceChannels == 4 && numberOfDestinationChannels == 1) {
312 // Down-mixing: 4 -> 1
313 // output = 0.25 * (input.L + input.R + input.SL + input.SR)
314 auto* sourceL = sourceBus.channelByType(ChannelLeft)->data();
315 auto* sourceR = sourceBus.channelByType(ChannelRight)->data();
316 auto* sourceSL = sourceBus.channelByType(ChannelSurroundLeft)->data();
317 auto* sourceSR = sourceBus.channelByType(ChannelSurroundRight)->data();
318
319 auto* destination = channelByType(ChannelLeft)->mutableData();
320
321 VectorMath::multiplyByScalarThenAddToOutput(sourceL, 0.25, destination, length());
322 VectorMath::multiplyByScalarThenAddToOutput(sourceR, 0.25, destination, length());
323 VectorMath::multiplyByScalarThenAddToOutput(sourceSL, 0.25, destination, length());
324 VectorMath::multiplyByScalarThenAddToOutput(sourceSR, 0.25, destination, length());
325 } else if (numberOfSourceChannels == 6 && numberOfDestinationChannels == 1) {
326 // Down-mixing: 5.1 -> 1
327 // output = sqrt(1/2) * (input.L + input.R) + input.C + 0.5 * (input.SL + input.SR)
328 auto* sourceL = sourceBus.channelByType(ChannelLeft)->data();
329 auto* sourceR = sourceBus.channelByType(ChannelRight)->data();
330 auto* sourceC = sourceBus.channelByType(ChannelCenter)->data();
331 auto* sourceSL = sourceBus.channelByType(ChannelSurroundLeft)->data();
332 auto* sourceSR = sourceBus.channelByType(ChannelSurroundRight)->data();
333
334 auto* destination = channelByType(ChannelLeft)->mutableData();
335 float scaleSqrtHalf = sqrtf(0.5);
336
337 VectorMath::multiplyByScalarThenAddToOutput(sourceL, scaleSqrtHalf, destination, length());
338 VectorMath::multiplyByScalarThenAddToOutput(sourceR, scaleSqrtHalf, destination, length());
339 VectorMath::add(sourceC, destination, destination, length());
340 VectorMath::multiplyByScalarThenAddToOutput(sourceSL, 0.5, destination, length());
341 VectorMath::multiplyByScalarThenAddToOutput(sourceSR, 0.5, destination, length());
342 } else if (numberOfSourceChannels == 4 && numberOfDestinationChannels == 2) {
343 // Down-mixing: 4 -> 2
344 // output.L = 0.5 * (input.L + input.SL)
345 // output.R = 0.5 * (input.R + input.SR)
346 auto* sourceL = sourceBus.channelByType(ChannelLeft)->data();
347 auto* sourceR = sourceBus.channelByType(ChannelRight)->data();
348 auto* sourceSL = sourceBus.channelByType(ChannelSurroundLeft)->data();
349 auto* sourceSR = sourceBus.channelByType(ChannelSurroundRight)->data();
350
351 auto* destinationL = channelByType(ChannelLeft)->mutableData();
352 auto* destinationR = channelByType(ChannelRight)->mutableData();
353
354 VectorMath::multiplyByScalarThenAddToOutput(sourceL, 0.5, destinationL, length());
355 VectorMath::multiplyByScalarThenAddToOutput(sourceSL, 0.5, destinationL, length());
356 VectorMath::multiplyByScalarThenAddToOutput(sourceR, 0.5, destinationR, length());
357 VectorMath::multiplyByScalarThenAddToOutput(sourceSR, 0.5, destinationR, length());
358 } else if (numberOfSourceChannels == 6 && numberOfDestinationChannels == 2) {
359 // Down-mixing: 5.1 -> 2
360 // output.L = input.L + sqrt(1/2) * (input.C + input.SL)
361 // output.R = input.R + sqrt(1/2) * (input.C + input.SR)
362 auto* sourceL = sourceBus.channelByType(ChannelLeft)->data();
363 auto* sourceR = sourceBus.channelByType(ChannelRight)->data();
364 auto* sourceC = sourceBus.channelByType(ChannelCenter)->data();
365 auto* sourceSL = sourceBus.channelByType(ChannelSurroundLeft)->data();
366 auto* sourceSR = sourceBus.channelByType(ChannelSurroundRight)->data();
367
368 float* destinationL = channelByType(ChannelLeft)->mutableData();
369 float* destinationR = channelByType(ChannelRight)->mutableData();
370 float scaleSqrtHalf = sqrtf(0.5);
371
372 VectorMath::add(sourceL, destinationL, destinationL, length());
373 VectorMath::multiplyByScalarThenAddToOutput(sourceC, scaleSqrtHalf, destinationL, length());
374 VectorMath::multiplyByScalarThenAddToOutput(sourceSL, scaleSqrtHalf, destinationL, length());
375 VectorMath::add(sourceR, destinationR, destinationR, length());
376 VectorMath::multiplyByScalarThenAddToOutput(sourceC, scaleSqrtHalf, destinationR, length());
377 VectorMath::multiplyByScalarThenAddToOutput(sourceSR, scaleSqrtHalf, destinationR, length());
378 } else if (numberOfSourceChannels == 6 && numberOfDestinationChannels == 4) {
379 // Down-mixing: 5.1 -> 4
380 // output.L = input.L + sqrt(1/2) * input.C
381 // output.R = input.R + sqrt(1/2) * input.C
382 // output.SL = input.SL
383 // output.SR = input.SR
384 auto* sourceL = sourceBus.channelByType(ChannelLeft)->data();
385 auto* sourceR = sourceBus.channelByType(ChannelRight)->data();
386 auto* sourceC = sourceBus.channelByType(ChannelCenter)->data();
387
388 auto* destinationL = channelByType(ChannelLeft)->mutableData();
389 auto* destinationR = channelByType(ChannelRight)->mutableData();
390 auto scaleSqrtHalf = sqrtf(0.5);
391
392 VectorMath::add(sourceL, destinationL, destinationL, length());
393 VectorMath::multiplyByScalarThenAddToOutput(sourceC, scaleSqrtHalf, destinationL, length());
394 VectorMath::add(sourceR, destinationR, destinationR, length());
395 VectorMath::multiplyByScalarThenAddToOutput(sourceC, scaleSqrtHalf, destinationR, length());
396 channel(2)->sumFrom(sourceBus.channel(4));
397 channel(3)->sumFrom(sourceBus.channel(5));
398 } else {
399 // Fallback for unknown combinations.
400 discreteSumFrom(sourceBus);
401 }
402}

404void AudioBus::discreteSumFrom(const AudioBus& sourceBus)
405{
406 unsigned numberOfSourceChannels = sourceBus.numberOfChannels();
407 unsigned numberOfDestinationChannels = numberOfChannels();
408
409 if (numberOfDestinationChannels < numberOfSourceChannels) {
410 // Down-mix by summing channels and dropping the remaining.
411 for (unsigned i = 0; i < numberOfDestinationChannels; ++i)
412 channel(i)->sumFrom(sourceBus.channel(i));
413 } else if (numberOfDestinationChannels > numberOfSourceChannels) {
414 // Up-mix by summing as many channels as we have.
415 for (unsigned i = 0; i < numberOfSourceChannels; ++i)
416 channel(i)->sumFrom(sourceBus.channel(i));
417 }
418}

420void AudioBus::copyWithGainFrom(const AudioBus& sourceBus, float gain)
421{
422 if (!topologyMatches(sourceBus)) {
423 ASSERT_NOT_REACHED();
424 zero();
425 return;
426 }
427
428 if (sourceBus.isSilent()) {
429 zero();
430 return;
431 }
432
433 unsigned numberOfChannels = this->numberOfChannels();
434 ASSERT(numberOfChannels <= MaxBusChannels);
435 if (numberOfChannels > MaxBusChannels)
436 return;
437
438 // If it is copying from the same bus and no need to change gain, just return.
439 if (this == &sourceBus && gain == 1)
440 return;
441
442 AudioBus& sourceBusSafe = const_cast<AudioBus&>(sourceBus);
443 const float* sources[MaxBusChannels];
444 float* destinations[MaxBusChannels];
445
446 for (unsigned i = 0; i < numberOfChannels; ++i) {
447 sources[i] = sourceBusSafe.channel(i)->data();
448 destinations[i] = channel(i)->mutableData();
449 }
450
451 unsigned framesToProcess = length();
452
453 // Handle gains of 0 and 1 (exactly) specially.
454 if (gain == 1) {
455 for (unsigned channelIndex = 0; channelIndex < numberOfChannels; ++channelIndex)
456 memcpy(destinations[channelIndex], sources[channelIndex], framesToProcess * sizeof(*destinations[channelIndex]));
457 } else if (!gain) {
458 for (unsigned channelIndex = 0; channelIndex < numberOfChannels; ++channelIndex)
459 memset(destinations[channelIndex], 0, framesToProcess * sizeof(*destinations[channelIndex]));
460 } else {
461 for (unsigned channelIndex = 0; channelIndex < numberOfChannels; ++channelIndex)
462 VectorMath::multiplyByScalar(sources[channelIndex], gain, destinations[channelIndex], framesToProcess);
463 }
464}

466void AudioBus::copyWithSampleAccurateGainValuesFrom(const AudioBus &sourceBus, float* gainValues, unsigned numberOfGainValues)
467{
468 // Make sure we're processing from the same type of bus.
469 // We *are* able to process from mono -> stereo
470 if (sourceBus.numberOfChannels() != 1 && !topologyMatches(sourceBus)) {
471 ASSERT_NOT_REACHED();
472 return;
473 }
474
475 if (!gainValues || numberOfGainValues > sourceBus.length()) {
476 ASSERT_NOT_REACHED();
477 return;
478 }
479
480 if (sourceBus.length() == numberOfGainValues && sourceBus.length() == length() && sourceBus.isSilent()) {
481 zero();
482 return;
483 }
484
485 // We handle both the 1 -> N and N -> N case here.
486 const float* source = sourceBus.channel(0)->data();
487 for (unsigned channelIndex = 0; channelIndex < numberOfChannels(); ++channelIndex) {
488 if (sourceBus.numberOfChannels() == numberOfChannels())
489 source = sourceBus.channel(channelIndex)->data();
490 float* destination = channel(channelIndex)->mutableData();
491 VectorMath::multiply(source, gainValues, destination, numberOfGainValues);
492 }
493}

495RefPtr<AudioBus> AudioBus::createBySampleRateConverting(const AudioBus* sourceBus, bool mixToMono, double newSampleRate)
496{
497 // sourceBus's sample-rate must be known.
498 ASSERT(sourceBus && sourceBus->sampleRate());
499 if (!sourceBus || !sourceBus->sampleRate())
500 return nullptr;
501
502 double sourceSampleRate = sourceBus->sampleRate();
503 double destinationSampleRate = newSampleRate;
504 double sampleRateRatio = sourceSampleRate / destinationSampleRate;
505 unsigned numberOfSourceChannels = sourceBus->numberOfChannels();
506
507 if (numberOfSourceChannels == 1)
508 mixToMono = false; // already mono
509
510 if (sourceSampleRate == destinationSampleRate) {
511 // No sample-rate conversion is necessary.
512 if (mixToMono)
513 return AudioBus::createByMixingToMono(sourceBus);
514
515 // Return exact copy.
516 return AudioBus::createBufferFromRange(sourceBus, 0, sourceBus->length());
517 }
518
519 if (sourceBus->isSilent()) {
520 RefPtr<AudioBus> silentBus = create(numberOfSourceChannels, sourceBus->length() / sampleRateRatio);
521 silentBus->setSampleRate(newSampleRate);
522 return silentBus;
523 }
524
525 // First, mix to mono (if necessary) then sample-rate convert.
526 const AudioBus* resamplerSourceBus;
527 RefPtr<AudioBus> mixedMonoBus;
528 if (mixToMono) {
529 mixedMonoBus = AudioBus::createByMixingToMono(sourceBus);
530 resamplerSourceBus = mixedMonoBus.get();
531 } else {
532 // Directly resample without down-mixing.
533 resamplerSourceBus = sourceBus;
534 }
535
536 // Calculate destination length based on the sample-rates.
537 int sourceLength = resamplerSourceBus->length();
538 int destinationLength = sourceLength / sampleRateRatio;
539
540 // Create destination bus with same number of channels.
541 unsigned numberOfDestinationChannels = resamplerSourceBus->numberOfChannels();
542 RefPtr<AudioBus> destinationBus = create(numberOfDestinationChannels, destinationLength);
543
544 // Sample-rate convert each channel.
545 for (unsigned i = 0; i < numberOfDestinationChannels; ++i) {
546 const float* source = resamplerSourceBus->channel(i)->data();
547 float* destination = destinationBus->channel(i)->mutableData();
548
549 SincResampler::processBuffer(source, destination, sourceLength, sampleRateRatio);
550 }
551
552 destinationBus->clearSilentFlag();
553 destinationBus->setSampleRate(newSampleRate);
554 return destinationBus;
555}

557RefPtr<AudioBus> AudioBus::createByMixingToMono(const AudioBus* sourceBus)
558{
559 if (sourceBus->isSilent())
560 return create(1, sourceBus->length());
561
562 switch (sourceBus->numberOfChannels()) {
563 case 1:
564 // Simply create an exact copy.
565 return AudioBus::createBufferFromRange(sourceBus, 0, sourceBus->length());
566 case 2:
567 {
568 unsigned n = sourceBus->length();
569 RefPtr<AudioBus> destinationBus = create(1, n);
570
571 const float* sourceL = sourceBus->channel(0)->data();
572 const float* sourceR = sourceBus->channel(1)->data();
573 float* destination = destinationBus->channel(0)->mutableData();
574
575 // Do the mono mixdown.
576 VectorMath::addVectorsThenMultiplyByScalar(sourceL, sourceR, 0.5, destination, n);
577
578 destinationBus->clearSilentFlag();
579 destinationBus->setSampleRate(sourceBus->sampleRate());
580 return destinationBus;
581 }
582 }
583
584 ASSERT_NOT_REACHED();
585 return nullptr;
586}
