31#ifndef AudioDSPKernel_h
32#define AudioDSPKernel_h
33
34#include "AudioDSPKernelProcessor.h"
35
36namespace WebCore {
37
38// AudioDSPKernel does the processing for one channel of an AudioDSPKernelProcessor.
39
40class AudioDSPKernel {
41 WTF_MAKE_NONCOPYABLE(AudioDSPKernel);
42 WTF_MAKE_FAST_ALLOCATED;
43public:
44 AudioDSPKernel(AudioDSPKernelProcessor* kernelProcessor)
45 : m_kernelProcessor(kernelProcessor)
46 , m_sampleRate(kernelProcessor->sampleRate())
47 {
48 }
49
50 AudioDSPKernel(float sampleRate)
51 : m_kernelProcessor(0)
52 , m_sampleRate(sampleRate)
53 {
54 }
55
56 virtual ~AudioDSPKernel() { };
57
58 // Subclasses must override process() to do the processing and reset() to reset DSP state.
59 virtual void process(const float* source, float* destination, size_t framesToProcess) = 0;
60
61 // Subclasses that have AudioParams must override this to process the AudioParams.
62 virtual void processOnlyAudioParams(size_t) { }
63
64 virtual void reset() = 0;
65
66 float sampleRate() const { return m_sampleRate; }
67 double nyquist() const { return 0.5 * sampleRate(); }
68
69 AudioDSPKernelProcessor* processor() { return m_kernelProcessor; }
70 const AudioDSPKernelProcessor* processor() const { return m_kernelProcessor; }
71
72 virtual double tailTime() const = 0;
73 virtual double latencyTime() const = 0;
74 virtual bool requiresTailProcessing() const = 0;
75
76protected:
77 AudioDSPKernelProcessor* m_kernelProcessor;
78 float m_sampleRate;
79};
80
81} // namespace WebCore
82
83#endif // AudioDSPKernel_h
