29#ifndef AudioDestination_h
30#define AudioDestination_h
31
32#include "AudioBus.h"
33#include "AudioIOCallback.h"
34#include <memory>
35#include <wtf/CompletionHandler.h>
36#include <wtf/Lock.h>
37#include <wtf/ThreadSafeRefCounted.h>
38#include <wtf/text/WTFString.h>
39
40namespace WebCore {
41
42// AudioDestination is an abstraction for audio hardware I/O.
43// The audio hardware periodically calls the AudioIOCallback render() method asking it to render/output the next render quantum of audio.
44// It optionally will pass in local/live audio input when it calls render().
45
46class AudioDestination : public ThreadSafeRefCounted<AudioDestination, WTF::DestructionThread::Main> {
47 WTF_MAKE_FAST_ALLOCATED;
48public:
49 // Pass in (numberOfInputChannels > 0) if live/local audio input is desired.
50 // Port-specific device identification information for live/local input streams can be passed in the inputDeviceId.
51 WEBCORE_EXPORT static Ref<AudioDestination> create(AudioIOCallback&, const String& inputDeviceId, unsigned numberOfInputChannels, unsigned numberOfOutputChannels, float sampleRate);
52
53 virtual ~AudioDestination() = default;
54
55 void clearCallback();
56
57 virtual void start(Function<void(Function<void()>&&)>&& dispatchToRenderThread, CompletionHandler<void(bool)>&& = [](bool) { }) = 0;
58 virtual void stop(CompletionHandler<void(bool)>&& = [](bool) { }) = 0;
59 virtual bool isPlaying() = 0;
60
61 // Sample-rate conversion may happen in AudioDestination to the hardware sample-rate
62 virtual float sampleRate() const = 0;
63 WEBCORE_EXPORT static float hardwareSampleRate();
64
65 virtual unsigned framesPerBuffer() const = 0;
66
67 // maxChannelCount() returns the total number of output channels of the audio hardware.
68 // A value of 0 indicates that the number of channels cannot be configured and
69 // that only stereo (2-channel) destinations can be created.
70 // The numberOfOutputChannels parameter of AudioDestination::create() is allowed to
71 // be a value: 1 <= numberOfOutputChannels <= maxChannelCount(),
72 // or if maxChannelCount() equals 0, then numberOfOutputChannels must be 2.
73 static unsigned long maxChannelCount();
74
75 void callRenderCallback(AudioBus* sourceBus, AudioBus* destinationBus, size_t framesToProcess, const AudioIOPosition& outputPosition);
76
77protected:
78 explicit AudioDestination(AudioIOCallback&);
79
80 Lock m_callbackLock;
81 AudioIOCallback* m_callback WTF_GUARDED_BY_LOCK(m_callbackLock) { nullptr };
82};
83
84inline AudioDestination::AudioDestination(AudioIOCallback& callback)
85{
86 Locker locker { m_callbackLock };
87 m_callback = &callback;
88}
89
90inline void AudioDestination::clearCallback()
91{
92 Locker locker { m_callbackLock };
93 m_callback = nullptr;
94}
95
96inline void AudioDestination::callRenderCallback(AudioBus* sourceBus, AudioBus* destinationBus, size_t framesToProcess, const AudioIOPosition& outputPosition)
97{
98 if (m_callbackLock.tryLock()) {
99 Locker locker { AdoptLock, m_callbackLock };
100 if (m_callback) {
101 m_callback->render(sourceBus, destinationBus, framesToProcess, outputPosition);
102 return;
103 }
104 }
105 destinationBus->zero();
106}

108} // namespace WebCore

110#endif // AudioDestination_h
