36namespace WebCore {
37
38static AudioHardwareListener::CreationFunction& audioHardwareListenerCreationFunction()
39{
40 static NeverDestroyed<AudioHardwareListener::CreationFunction> creationFunction;
41 return creationFunction;
42}
43
44void AudioHardwareListener::setCreationFunction(CreationFunction&& function)
45{
46 audioHardwareListenerCreationFunction() = WTFMove(function);
47}
48
49void AudioHardwareListener::resetCreationFunction()
50{
51 audioHardwareListenerCreationFunction() = [] (AudioHardwareListener::Client& client) {
52#if PLATFORM(MAC)
53 return AudioHardwareListenerMac::create(client);
54#else
55 return adoptRef(*new AudioHardwareListener(client));
56#endif
57 };
58}
59
60Ref<AudioHardwareListener> AudioHardwareListener::create(Client& client)
61{
62 if (!audioHardwareListenerCreationFunction())
63 resetCreationFunction();
64
65 return audioHardwareListenerCreationFunction()(client);
66}
67
68AudioHardwareListener::AudioHardwareListener(Client& client)
69 : m_client(client)
70{
71#if PLATFORM(IOS_FAMILY)
72 m_supportedBufferSizes = { 32, 4096 };
73#endif
74}
