31#ifndef AudioProcessor_h
32#define AudioProcessor_h
33
34namespace WebCore {
35
36class AudioBus;
37
38// AudioProcessor is an abstract base class representing an audio signal processing object with a single input and a single output,
39// where the number of input channels equals the number of output channels. It can be used as one part of a complex DSP algorithm,
40// or as the processor for a basic (one input - one output) AudioNode.
41
42class AudioProcessor {
43 WTF_MAKE_NONCOPYABLE(AudioProcessor);
44 WTF_MAKE_FAST_ALLOCATED;
45public:
46 AudioProcessor(float sampleRate, unsigned numberOfChannels)
47 : m_initialized(false)
48 , m_numberOfChannels(numberOfChannels)
49 , m_sampleRate(sampleRate)
50 {
51 }
52
53 virtual ~AudioProcessor() = default;
54
55 // Full initialization can be done here instead of in the constructor.
56 virtual void initialize() = 0;
57 virtual void uninitialize() = 0;
58
59 // Processes the source to destination bus. The number of channels must match in source and destination.
60 virtual void process(const AudioBus* source, AudioBus* destination, size_t framesToProcess) = 0;
61
62 // Forces all AudioParams in the processor to run the timeline, bypassing any other processing the processor
63 // would do in process().
64 virtual void processOnlyAudioParams(size_t) { }
65
66 // Resets filter state
67 virtual void reset() = 0;
68
69 virtual void setNumberOfChannels(unsigned) = 0;
70 virtual unsigned numberOfChannels() const = 0;
71
72 bool isInitialized() const { return m_initialized; }
73
74 float sampleRate() const { return m_sampleRate; }
75
76 virtual double tailTime() const = 0;
77 virtual double latencyTime() const = 0;
78 virtual bool requiresTailProcessing() const = 0;
79
80protected:
81 bool m_initialized;
82 unsigned m_numberOfChannels;
83 float m_sampleRate;
84};
85
86} // namespace WebCore
87
88#endif // AudioProcessor_h
