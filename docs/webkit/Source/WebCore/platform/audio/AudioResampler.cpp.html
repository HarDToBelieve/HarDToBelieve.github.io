37AudioResampler::AudioResampler()
38{
39 m_kernels.append(makeUnique<AudioResamplerKernel>(this));
40 m_sourceBus = AudioBus::create(1, 0, false);
41}
42
43AudioResampler::AudioResampler(unsigned numberOfChannels)
44{
45 for (unsigned i = 0; i < numberOfChannels; ++i)
46 m_kernels.append(makeUnique<AudioResamplerKernel>(this));
47
48 m_sourceBus = AudioBus::create(numberOfChannels, 0, false);
49}

51void AudioResampler::configureChannels(unsigned numberOfChannels)
52{
53 unsigned currentSize = m_kernels.size();
54 if (numberOfChannels == currentSize)
55 return; // already setup
56
57 // First deal with adding or removing kernels.
58 if (numberOfChannels > currentSize) {
59 for (unsigned i = currentSize; i < numberOfChannels; ++i)
60 m_kernels.append(makeUnique<AudioResamplerKernel>(this));
61 } else
62 m_kernels.resize(numberOfChannels);
63
64 // Reconfigure our source bus to the new channel size.
65 m_sourceBus = AudioBus::create(numberOfChannels, 0, false);
66}
67
68void AudioResampler::process(AudioSourceProvider* provider, AudioBus* destinationBus, size_t framesToProcess)
69{
70 ASSERT(provider);
71 if (!provider)
72 return;
73
74 unsigned numberOfChannels = m_kernels.size();
75
76 // Make sure our configuration matches the bus we're rendering to.
77 bool channelsMatch = (destinationBus && destinationBus->numberOfChannels() == numberOfChannels);
78 ASSERT(channelsMatch);
79 if (!channelsMatch)
80 return;
81
82 // Setup the source bus.
83 for (unsigned i = 0; i < numberOfChannels; ++i) {
84 // Figure out how many frames we need to get from the provider, and a pointer to the buffer.
85 size_t framesNeeded;
86 float* fillPointer = m_kernels[i]->getSourcePointer(framesToProcess, &framesNeeded);
87 ASSERT(fillPointer);
88 if (!fillPointer)
89 return;
90
91 m_sourceBus->setChannelMemory(i, fillPointer, framesNeeded);
92 }
93
94 // Ask the provider to supply the desired number of source frames.
95 provider->provideInput(m_sourceBus.get(), m_sourceBus->length());
96
97 // Now that we have the source data, resample each channel into the destination bus.
98 // FIXME: optimize for the common stereo case where it's faster to process both left/right channels in the same inner loop.
99 for (unsigned i = 0; i < numberOfChannels; ++i) {
100 float* destination = destinationBus->channel(i)->mutableData();
101 m_kernels[i]->process(destination, framesToProcess);
102 }
103}

105void AudioResampler::setRate(double rate)
106{
107 if (std::isnan(rate) || std::isinf(rate) || rate <= 0.0)
108 return;
109
110 m_rate = std::min(AudioResampler::MaxRate, rate);
111}
112
113void AudioResampler::reset()
114{
115 unsigned numberOfChannels = m_kernels.size();
116 for (unsigned i = 0; i < numberOfChannels; ++i)
117 m_kernels[i]->reset();
118}
