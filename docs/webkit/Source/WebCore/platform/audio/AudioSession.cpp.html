70AudioSession& AudioSession::sharedSession()
71{
72 if (!sharedAudioSession())
73 setSharedSession(AudioSession::create());
74 return *sharedAudioSession();
75}
76
77void AudioSession::setSharedSession(UniqueRef<AudioSession>&& session)
78{
79 sharedAudioSession() = WTFMove(session);
80
81 audioSessionChangedObservers().forEach([] (auto& observer) {
82 observer(*sharedAudioSession());
83 });
84}
85
86void AudioSession::addAudioSessionChangedObserver(const ChangedObserver& observer)
87{
88 ASSERT(!audioSessionChangedObservers().contains(observer));
89 audioSessionChangedObservers().add(observer);
90
91 if (sharedAudioSession())
92 observer(*sharedAudioSession());
93}

207String convertEnumerationToString(RouteSharingPolicy enumerationValue)
208{
209 static const NeverDestroyed<String> values[] = {
210 MAKE_STATIC_STRING_IMPL("Default"),
211 MAKE_STATIC_STRING_IMPL("LongFormAudio"),
212 MAKE_STATIC_STRING_IMPL("Independent"),
213 MAKE_STATIC_STRING_IMPL("LongFormVideo"),
214 };
215 static_assert(!static_cast<size_t>(RouteSharingPolicy::Default), "RouteSharingPolicy::Default is not 0 as expected");
216 static_assert(static_cast<size_t>(RouteSharingPolicy::LongFormAudio) == 1, "RouteSharingPolicy::LongFormAudio is not 1 as expected");
217 static_assert(static_cast<size_t>(RouteSharingPolicy::Independent) == 2, "RouteSharingPolicy::Independent is not 2 as expected");
218 static_assert(static_cast<size_t>(RouteSharingPolicy::LongFormVideo) == 3, "RouteSharingPolicy::LongFormVideo is not 3 as expected");
219 ASSERT(static_cast<size_t>(enumerationValue) < WTF_ARRAY_LENGTH(values));
220 return values[static_cast<size_t>(enumerationValue)];
221}
