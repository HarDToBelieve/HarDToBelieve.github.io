34// https://www.w3.org/TR/webaudio/#render-quantum-size
35constexpr size_t renderQuantumSize = 128;
36
37// Standard functions for converting to and from decibel values from linear.
38float linearToDecibels(float);
39float decibelsToLinear(float);
40
41// timeConstant is the time it takes a first-order linear time-invariant system
42// to reach the value 1 - 1/e (around 63.2%) given a step input response.
43// discreteTimeConstantForSampleRate() will return the discrete time-constant for the specific sampleRate.
44double discreteTimeConstantForSampleRate(double timeConstant, double sampleRate);
45
46// How to do rounding when converting time to sample frame.
47enum class SampleFrameRounding {
48 Nearest,
49 Down,
50 Up
51};
52
53// Convert the time to a sample frame at the given sample rate.
54size_t timeToSampleFrame(double time, double sampleRate, SampleFrameRounding = SampleFrameRounding::Nearest);
55
56inline float linearToDecibels(float linear)
57{
58 ASSERT(linear >= 0);
59 return 20 * log10f(linear);
60}
61
62inline float decibelsToLinear(float decibels)
63{
64 return powf(10, 0.05f * decibels);
65}
