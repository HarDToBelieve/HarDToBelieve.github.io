36// Use a 50ms smoothing / de-zippering time-constant.
37constexpr float SmoothingTimeConstant = 0.050f;

41EqualPowerPanner::EqualPowerPanner(float sampleRate)
42 : Panner(PanningModelType::Equalpower)
43{
44 m_smoothingConstant = AudioUtilities::discreteTimeConstantForSampleRate(SmoothingTimeConstant, sampleRate);
45}

47void EqualPowerPanner::calculateDesiredGain(double& desiredGainL, double& desiredGainR, double azimuth, unsigned numberOfChannels)
48{
49 // Clamp azimuth to allowed range of -180 -> +180.
50 azimuth = std::clamp(azimuth, -180.0, 180.0);
51
52 // Alias the azimuth ranges behind us to in front of us:
53 // -90 -> -180 to -90 -> 0 and 90 -> 180 to 90 -> 0
54 if (azimuth < -90)
55 azimuth = -180 - azimuth;
56 else if (azimuth > 90)
57 azimuth = 180 - azimuth;
58
59 double desiredPanPosition;
60
61 if (numberOfChannels == 1) { // For mono source case.
62 // Pan smoothly from left to right with azimuth going from -90 -> +90
63 // degrees.
64 desiredPanPosition = (azimuth + 90) / 180;
65 } else { // For stereo source case.
66 if (azimuth <= 0) { // from -90 -> 0
67 // sourceL -> destL and "equal-power pan" sourceR as in mono case
68 // by transforming the "azimuth" value from -90 -> 0 degrees into the
69 // range -90 -> +90.
70 desiredPanPosition = (azimuth + 90) / 90;
71 } else { // from 0 -> +90
72 // sourceR -> destR and "equal-power pan" sourceL as in mono case
73 // by transforming the "azimuth" value from 0 -> +90 degrees into the
74 // range -90 -> +90.
75 desiredPanPosition = azimuth / 90;
76 }
77 }
78
79 desiredGainL = std::cos(piOverTwoDouble * desiredPanPosition);
80 desiredGainR = std::sin(piOverTwoDouble * desiredPanPosition);
81}

83void EqualPowerPanner::pan(double azimuth, double /*elevation*/, const AudioBus* inputBus, AudioBus* outputBus, size_t framesToProcess)
84{
85 bool isInputSafe = inputBus && (inputBus->numberOfChannels() == 1 || inputBus->numberOfChannels() == 2) && framesToProcess <= inputBus->length();
86 ASSERT(isInputSafe);
87 if (!isInputSafe)
88 return;
89
90 unsigned numberOfInputChannels = inputBus->numberOfChannels();
91
92 bool isOutputSafe = outputBus && outputBus->numberOfChannels() == 2 && framesToProcess <= outputBus->length();
93 ASSERT(isOutputSafe);
94 if (!isOutputSafe)
95 return;
96
97 const float* sourceL = inputBus->channel(0)->data();
98 const float* sourceR = numberOfInputChannels > 1 ? inputBus->channel(1)->data() : sourceL;
99 float* destinationL = outputBus->channelByType(AudioBus::ChannelLeft)->mutableData();
100 float* destinationR = outputBus->channelByType(AudioBus::ChannelRight)->mutableData();
101
102 if (!sourceL || !sourceR || !destinationL || !destinationR)
103 return;
104
105 // Clamp azimuth to allowed range of -180 -> +180.
106 azimuth = std::max(-180.0, azimuth);
107 azimuth = std::min(180.0, azimuth);
108
109 // Alias the azimuth ranges behind us to in front of us:
110 // -90 -> -180 to -90 -> 0 and 90 -> 180 to 90 -> 0
111 if (azimuth < -90)
112 azimuth = -180 - azimuth;
113 else if (azimuth > 90)
114 azimuth = 180 - azimuth;
115
116 double desiredPanPosition;
117 double desiredGainL;
118 double desiredGainR;
119
120 if (numberOfInputChannels == 1) { // For mono source case.
121 // Pan smoothly from left to right with azimuth going from -90 -> +90 degrees.
122 desiredPanPosition = (azimuth + 90) / 180;
123 } else { // For stereo source case.
124 if (azimuth <= 0) { // from -90 -> 0
125 // sourceL -> destL and "equal-power pan" sourceR as in mono case
126 // by transforming the "azimuth" value from -90 -> 0 degrees into the range -90 -> +90.
127 desiredPanPosition = (azimuth + 90) / 90;
128 } else { // from 0 -> +90
129 // sourceR -> destR and "equal-power pan" sourceL as in mono case
130 // by transforming the "azimuth" value from 0 -> +90 degrees into the range -90 -> +90.
131 desiredPanPosition = azimuth / 90;
132 }
133 }
134
135 desiredGainL = cos(0.5 * piDouble * desiredPanPosition);
136 desiredGainR = sin(0.5 * piDouble * desiredPanPosition);
137
138 // Don't de-zipper on first render call.
139 if (m_isFirstRender) {
140 m_isFirstRender = false;
141 m_gainL = desiredGainL;
142 m_gainR = desiredGainR;
143 }
144
145 // Cache in local variables.
146 double gainL = m_gainL;
147 double gainR = m_gainR;
148
149 // Get local copy of smoothing constant.
150 const double SmoothingConstant = m_smoothingConstant;
151
152 int n = framesToProcess;
153
154 if (numberOfInputChannels == 1) { // For mono source case.
155 while (n--) {
156 float inputL = *sourceL++;
157 gainL += (desiredGainL - gainL) * SmoothingConstant;
158 gainR += (desiredGainR - gainR) * SmoothingConstant;
159 *destinationL++ = static_cast<float>(inputL * gainL);
160 *destinationR++ = static_cast<float>(inputL * gainR);
161 }
162 } else { // For stereo source case.
163 if (azimuth <= 0) { // from -90 -> 0
164 while (n--) {
165 float inputL = *sourceL++;
166 float inputR = *sourceR++;
167 gainL += (desiredGainL - gainL) * SmoothingConstant;
168 gainR += (desiredGainR - gainR) * SmoothingConstant;
169 *destinationL++ = static_cast<float>(inputL + inputR * gainL);
170 *destinationR++ = static_cast<float>(inputR * gainR);
171 }
172 } else { // from 0 -> +90
173 while (n--) {
174 float inputL = *sourceL++;
175 float inputR = *sourceR++;
176 gainL += (desiredGainL - gainL) * SmoothingConstant;
177 gainR += (desiredGainR - gainR) * SmoothingConstant;
178 *destinationL++ = static_cast<float>(inputL * gainL);
179 *destinationR++ = static_cast<float>(inputR + inputL * gainR);
180 }
181 }
182 }
183
184 m_gainL = gainL;
185 m_gainR = gainR;
186}

188void EqualPowerPanner::panWithSampleAccurateValues(double* azimuth, double*, const AudioBus* inputBus, AudioBus* outputBus, size_t framesToProcess)
189{
190 ASSERT(inputBus);
191 ASSERT(framesToProcess <= inputBus->length());
192 ASSERT(inputBus->numberOfChannels() >= 1u);
193 ASSERT(inputBus->numberOfChannels() <= 2u);
194
195 unsigned numberOfInputChannels = inputBus->numberOfChannels();
196
197 ASSERT(outputBus);
198 ASSERT(outputBus->numberOfChannels() == 2u);
199 ASSERT(framesToProcess <= outputBus->length());
200
201 const float* sourceL = inputBus->channel(0)->data();
202 const float* sourceR = numberOfInputChannels > 1 ? inputBus->channel(1)->data() : sourceL;
203 float* destinationL = outputBus->channelByType(AudioBus::ChannelLeft)->mutableData();
204 float* destinationR = outputBus->channelByType(AudioBus::ChannelRight)->mutableData();
205
206 ASSERT(sourceL);
207 ASSERT(sourceR);
208 ASSERT(destinationL);
209 ASSERT(destinationR);
210
211 int n = framesToProcess;
212
213 if (numberOfInputChannels == 1) { // For mono source case.
214 for (int k = 0; k < n; ++k) {
215 double desiredGainL;
216 double desiredGainR;
217 float inputL = *sourceL++;
218
219 calculateDesiredGain(desiredGainL, desiredGainR, azimuth[k], numberOfInputChannels);
220 *destinationL++ = static_cast<float>(inputL * desiredGainL);
221 *destinationR++ = static_cast<float>(inputL * desiredGainR);
222 }
223 } else { // For stereo source case.
224 for (int k = 0; k < n; ++k) {
225 double desiredGainL;
226 double desiredGainR;
227
228 calculateDesiredGain(desiredGainL, desiredGainR, azimuth[k], numberOfInputChannels);
229 if (azimuth[k] <= 0) { // from -90 -> 0
230 float inputL = *sourceL++;
231 float inputR = *sourceR++;
232 *destinationL++ = static_cast<float>(inputL + inputR * desiredGainL);
233 *destinationR++ = static_cast<float>(inputR * desiredGainR);
234 } else { // from 0 -> +90
235 float inputL = *sourceL++;
236 float inputR = *sourceR++;
237 *destinationL++ = static_cast<float>(inputL * desiredGainL);
238 *destinationR++ = static_cast<float>(inputR + inputL * desiredGainR);
239 }
240 }
241 }
242}
