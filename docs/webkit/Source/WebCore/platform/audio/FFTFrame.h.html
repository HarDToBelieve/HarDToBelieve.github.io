46namespace WebCore {
47
48// Defines the interface for an "FFT frame", an object which is able to perform a forward
49// and reverse FFT, internally storing the resultant frequency-domain data.
50
51class FFTFrame {
52 WTF_MAKE_FAST_ALLOCATED;
53public:
54 // The constructors, destructor, and methods up to the CROSS-PLATFORM section have platform-dependent implementations.
55
56 FFTFrame(unsigned fftSize);
57 FFTFrame(); // creates a blank/empty frame for later use with createInterpolatedFrame()
58 FFTFrame(const FFTFrame& frame);
59 ~FFTFrame();
60
61 static void initialize();
62 void doFFT(const float* data);
63 void doInverseFFT(float* data);
64 void multiply(const FFTFrame& frame); // multiplies ourself with frame : effectively operator*=()
65 void scaleFFT(float factor);
66
67 AudioFloatArray& realData() { return m_realData; }
68 AudioFloatArray& imagData() { return m_imagData; }
69 const AudioFloatArray& realData() const { return m_realData; }
70 const AudioFloatArray& imagData() const { return m_imagData; }
71
72 static int minFFTSize();
73 static int maxFFTSize();
74
75 void print(); // for debugging
76
77 // CROSS-PLATFORM
78 // The remaining public methods have cross-platform implementations:
79
80 // Interpolates from frame1 -> frame2 as x goes from 0.0 -> 1.0
81 static std::unique_ptr<FFTFrame> createInterpolatedFrame(const FFTFrame& frame1, const FFTFrame& frame2, double x);
82
83 void doPaddedFFT(const float* data, size_t dataSize); // zero-padding with dataSize <= fftSize
84 double extractAverageGroupDelay();
85 void addConstantGroupDelay(double sampleFrameDelay);
86
87 unsigned fftSize() const { return m_FFTSize; }
88 unsigned log2FFTSize() const { return m_log2FFTSize; }
89
90private:
91 unsigned m_FFTSize;
92 unsigned m_log2FFTSize;
93
94 void interpolateFrequencyComponents(const FFTFrame& frame1, const FFTFrame& frame2, double x);
95
96#if USE(ACCELERATE)
97 DSPSplitComplex& dspSplitComplex() { return m_frame; }
98 DSPSplitComplex dspSplitComplex() const { return m_frame; }
99
100 static FFTSetup fftSetupForSize(unsigned fftSize);
101
102 FFTSetup m_FFTSetup;
103
104 DSPSplitComplex m_frame;
105#endif
106
107#if USE(GSTREAMER)
108 GstFFTF32* m_fft;
109 GstFFTF32* m_inverseFft;
110 UniqueArray<GstFFTF32Complex> m_complexData;
111#endif // USE(GSTREAMER)
112
113 AudioFloatArray m_realData;
114 AudioFloatArray m_imagData;
115};
116
117} // namespace WebCore
118
119#endif // FFTFrame_h
