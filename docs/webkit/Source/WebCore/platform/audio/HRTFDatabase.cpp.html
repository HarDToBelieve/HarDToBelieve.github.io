39HRTFDatabase::HRTFDatabase(float sampleRate)
40 : m_sampleRate(sampleRate)
41{
42 unsigned elevationIndex = 0;
43 for (int elevation = MinElevation; elevation <= MaxElevation; elevation += RawElevationAngleSpacing) {
44 std::unique_ptr<HRTFElevation> hrtfElevation = HRTFElevation::createForSubject("Composite", elevation, sampleRate);
45 ASSERT(hrtfElevation.get());
46 if (!hrtfElevation.get())
47 return;
48
49 m_elevations[elevationIndex] = WTFMove(hrtfElevation);
50 elevationIndex += InterpolationFactor;
51 }
52
53 // Now, go back and interpolate elevations.
54 if (InterpolationFactor > 1) {
55 for (unsigned i = 0; i < NumberOfTotalElevations; i += InterpolationFactor) {
56 unsigned j = (i + InterpolationFactor);
57 if (j >= NumberOfTotalElevations)
58 j = i; // for last elevation interpolate with itself
59
60 // Create the interpolated convolution kernels and delays.
61 for (unsigned jj = 1; jj < InterpolationFactor; ++jj) {
62 float x = static_cast<float>(jj) / static_cast<float>(InterpolationFactor);
63 m_elevations[i + jj] = HRTFElevation::createByInterpolatingSlices(m_elevations[i].get(), m_elevations[j].get(), x, sampleRate);
64 ASSERT(m_elevations[i + jj].get());
65 }
66 }
67 }
68}

70void HRTFDatabase::getKernelsFromAzimuthElevation(double azimuthBlend, unsigned azimuthIndex, double elevationAngle, HRTFKernel* &kernelL, HRTFKernel* &kernelR,
71 double& frameDelayL, double& frameDelayR)
72{
73 unsigned elevationIndex = indexFromElevationAngle(elevationAngle);
74 ASSERT_WITH_SECURITY_IMPLICATION(elevationIndex < m_elevations.size() && m_elevations.size() > 0);
75
76 if (!m_elevations.size()) {
77 kernelL = 0;
78 kernelR = 0;
79 return;
80 }
81
82 if (elevationIndex > m_elevations.size() - 1)
83 elevationIndex = m_elevations.size() - 1;
84
85 HRTFElevation* hrtfElevation = m_elevations[elevationIndex].get();
86 ASSERT(hrtfElevation);
87 if (!hrtfElevation) {
88 kernelL = 0;
89 kernelR = 0;
90 return;
91 }
92
93 hrtfElevation->getKernelsFromAzimuth(azimuthBlend, azimuthIndex, kernelL, kernelR, frameDelayL, frameDelayR);
94}
95
96unsigned HRTFDatabase::indexFromElevationAngle(double elevationAngle)
97{
98 // Clamp to allowed range.
99 elevationAngle = std::max(static_cast<double>(MinElevation), elevationAngle);
100 elevationAngle = std::min(static_cast<double>(MaxElevation), elevationAngle);
101
102 unsigned elevationIndex = static_cast<int>(InterpolationFactor * (elevationAngle - MinElevation) / RawElevationAngleSpacing);
103 return elevationIndex;
104}
