25#ifndef HRTFPanner_h
26#define HRTFPanner_h
27
28#include "DelayDSPKernel.h"
29#include "FFTConvolver.h"
30#include "Panner.h"
31
32namespace WebCore {
33
34class HRTFDatabaseLoader;
35
36class HRTFPanner final : public Panner {
37public:
38 explicit HRTFPanner(float sampleRate, HRTFDatabaseLoader*);
39 virtual ~HRTFPanner();
40
41 // Panner
42 void pan(double azimuth, double elevation, const AudioBus* inputBus, AudioBus* outputBus, size_t framesToProcess) final;
43 void panWithSampleAccurateValues(double* azimuth, double* elevation, const AudioBus* inputBus, AudioBus* outputBus, size_t framesToProcess) final;
44 void reset() override;
45
46 size_t fftSize() const { return fftSizeForSampleRate(m_sampleRate); }
47 static size_t fftSizeForSampleRate(float sampleRate);
48
49 float sampleRate() const { return m_sampleRate; }
50
51 double tailTime() const override;
52 double latencyTime() const override;
53 bool requiresTailProcessing() const final;
54
55private:
56 // Given an azimuth angle in the range -180 -> +180, returns the corresponding azimuth index for the database,
57 // and azimuthBlend which is an interpolation value from 0 -> 1.
58 int calculateDesiredAzimuthIndexAndBlend(double azimuth, double& azimuthBlend);
59
60 RefPtr<HRTFDatabaseLoader> m_databaseLoader;
61
62 float m_sampleRate;
63
64 // We maintain two sets of convolvers for smooth cross-faded interpolations when
65 // then azimuth and elevation are dynamically changing.
66 // When the azimuth and elevation are not changing, we simply process with one of the two sets.
67 // Initially we use CrossfadeSelection1 corresponding to m_convolverL1 and m_convolverR1.
68 // Whenever the azimuth or elevation changes, a crossfade is initiated to transition
69 // to the new position. So if we're currently processing with CrossfadeSelection1, then
70 // we transition to CrossfadeSelection2 (and vice versa).
71 // If we're in the middle of a transition, then we wait until it is complete before
72 // initiating a new transition.
73
74 // Selects either the convolver set (m_convolverL1, m_convolverR1) or (m_convolverL2, m_convolverR2).
75 enum CrossfadeSelection {
76 CrossfadeSelection1,
77 CrossfadeSelection2
78 };
79
80 CrossfadeSelection m_crossfadeSelection { CrossfadeSelection1 };
81
82 // azimuth/elevation for CrossfadeSelection1.
83 std::optional<int> m_azimuthIndex1;
84 double m_elevation1 { 0 };
85
86 // azimuth/elevation for CrossfadeSelection2.
87 std::optional<int> m_azimuthIndex2;
88 double m_elevation2 { 0 };
89
90 // A crossfade value 0 <= m_crossfadeX <= 1.
91 float m_crossfadeX { 0 };
92
93 // Per-sample-frame crossfade value increment.
94 float m_crossfadeIncr { 0 };
95
96 FFTConvolver m_convolverL1;
97 FFTConvolver m_convolverR1;
98 FFTConvolver m_convolverL2;
99 FFTConvolver m_convolverR2;
100
101 DelayDSPKernel m_delayLineL;
102 DelayDSPKernel m_delayLineR;
103
104 AudioFloatArray m_tempL1;
105 AudioFloatArray m_tempR1;
106 AudioFloatArray m_tempL2;
107 AudioFloatArray m_tempR2;
108};
109
110} // namespace WebCore
111
112#endif // HRTFPanner_h
