26#pragma once
27
28#include <wtf/Vector.h>
29
30namespace WebCore {
31
32class IIRFilter {
33 WTF_MAKE_FAST_ALLOCATED;
34public:
35 static constexpr size_t maxOrder { 20 };
36 IIRFilter(const Vector<double>& feedforward, const Vector<double>& feedback);
37
38 void reset();
39
40 void process(const float* source, float* destination, size_t framesToProcess);
41 void getFrequencyResponse(unsigned length, const float* frequency, float* magResponse, float* phaseResponse);
42 double tailTime(double sampleRate, bool isFilterStable);
43
44 const Vector<double>& feedforward() const { return m_feedforward; }
45 const Vector<double>& feedback() const { return m_feedback; }
46
47private:
48 // Filter memory
49 //
50 // For simplicity, we assume |m_xBuffer| and |m_yBuffer| have the same length,
51 // and the length is a power of two. Since the number of coefficients has a
52 // fixed upper length, the size of xBuffer and yBuffer is fixed. |m_xBuffer|
53 // holds the old input values and |m_yBuffer| holds the old output values
54 // needed to compute the new output value.
55 //
56 // m_yBuffer[m_bufferIndex] holds the most recent output value, say, y[n].
57 // Then m_yBuffer[m_bufferIndex - k] is y[n - k]. Similarly for m_xBuffer.
58 //
59 // To minimize roundoff, these arrays are double's instead of floats.
60 Vector<double> m_xBuffer;
61 Vector<double> m_yBuffer;
62
63 // Index into the xBuffer and yBuffer arrays where the most current x and y
64 // values should be stored. xBuffer[bufferIndex] corresponds to x[n], the
65 // current x input value and yBuffer[bufferIndex] is where y[n], the current
66 // output value.
67 size_t m_bufferIndex { 0 };
68
69 // Those Vectors are owned by the IIRProcess, which owns the IIRFilters via
70 // the IIRDSPKernels. This is a memory optimization to avoid having copies
71 // of these vectors in each IIRDSPKernel / IIRFilter.
72 const Vector<double>& m_feedforward;
73 const Vector<double>& m_feedback;
74};
75
76} // namespace WebCore
