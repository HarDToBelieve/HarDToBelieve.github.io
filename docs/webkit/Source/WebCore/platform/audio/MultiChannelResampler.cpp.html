40MultiChannelResampler::MultiChannelResampler(double scaleFactor, unsigned numberOfChannels, unsigned requestFrames, Function<void(AudioBus*, size_t framesToProcess)>&& provideInput)
41 : m_numberOfChannels(numberOfChannels)
42 , m_provideInput(WTFMove(provideInput))
43 , m_multiChannelBus(AudioBus::create(numberOfChannels, requestFrames, false))
44{
45 // As an optimization, we will use the buffer passed to provideInputForChannel() as channel memory for the first channel so we
46 // only need to allocate memory if there is more than one channel.
47 if (numberOfChannels > 1) {
48 m_channelsMemory.reserveInitialCapacity(numberOfChannels - 1);
49 for (unsigned channelIndex = 1; channelIndex < numberOfChannels; ++channelIndex) {
50 m_channelsMemory.uncheckedAppend(makeUnique<AudioFloatArray>(requestFrames));
51 m_multiChannelBus->setChannelMemory(channelIndex, m_channelsMemory.last()->data(), requestFrames);
52 }
53 }
54
55 // Create each channel's resampler.
56 for (unsigned channelIndex = 0; channelIndex < numberOfChannels; ++channelIndex)
57 m_kernels.append(makeUnique<SincResampler>(scaleFactor, requestFrames, std::bind(&MultiChannelResampler::provideInputForChannel, this, std::placeholders::_1, std::placeholders::_2, channelIndex)));
58}

60MultiChannelResampler::~MultiChannelResampler() = default;

62void MultiChannelResampler::process(AudioBus* destination, size_t framesToProcess)
63{
64 ASSERT(m_numberOfChannels == destination->numberOfChannels());
65 if (destination->numberOfChannels() == 1) {
66 // Fast path when the bus is mono to avoid the chunking below.
67 m_kernels[0]->process(destination->channel(0)->mutableData(), framesToProcess);
68 return;
69 }
70
71 // We need to ensure that SincResampler only calls provideInputForChannel() once for each channel or it will confuse the logic
72 // inside provideInputForChannel(). To ensure this, we chunk the number of requested frames into SincResampler::chunkSize()
73 // sized chunks. SincResampler guarantees it will only call provideInputForChannel() once once we resample this way.
74 m_outputFramesReady = 0;
75 while (m_outputFramesReady < framesToProcess) {
76 size_t chunkSize = m_kernels[0]->chunkSize();
77 size_t framesThisTime = std::min(framesToProcess - m_outputFramesReady, chunkSize);
78
79 for (unsigned channelIndex = 0; channelIndex < m_numberOfChannels; ++channelIndex) {
80 ASSERT(chunkSize == m_kernels[channelIndex]->chunkSize());
81 m_kernels[channelIndex]->process(destination->channel(channelIndex)->mutableData() + m_outputFramesReady, framesThisTime);
82 }
83
84 m_outputFramesReady += framesThisTime;
85 }
86}
87
88void MultiChannelResampler::provideInputForChannel(float* buffer, size_t framesToProcess, unsigned channelIndex)
89{
90 ASSERT(channelIndex < m_multiChannelBus->numberOfChannels());
91 ASSERT(framesToProcess == m_multiChannelBus->length());
92
93 if (!channelIndex) {
94 // As an optimization, we use the provided buffer as memory for the first channel in the AudioBus. This avoids
95 // having to memcpy() for the first channel.
96 m_multiChannelBus->setChannelMemory(0, buffer, framesToProcess);
97 m_provideInput(m_multiChannelBus.get(), framesToProcess);
98 return;
99 }
100
101 // Copy the channel data from what we received from m_multiChannelProvider.
102 memcpy(buffer, m_multiChannelBus->channel(channelIndex)->data(), sizeof(float) * framesToProcess);
103}
