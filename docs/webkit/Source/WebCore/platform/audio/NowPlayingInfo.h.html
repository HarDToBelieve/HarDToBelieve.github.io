36struct NowPlayingInfoArtwork {
37 String src;
38 String mimeType;
39 RefPtr<SharedBuffer> imageData;
40
41 bool operator==(const NowPlayingInfoArtwork& other) const
42 {
43 return src == other.src && mimeType == other.mimeType;
44 }
45
46 bool operator!=(const NowPlayingInfoArtwork& other) const
47 {
48 return !(*this == other);
49 }
50
51 template<class Encoder> void encode(Encoder&) const;
52 template<class Decoder> static std::optional<NowPlayingInfoArtwork> decode(Decoder&);
53};
54
55template<class Encoder> inline void NowPlayingInfoArtwork::encode(Encoder& encoder) const
56{
57 encoder << src << mimeType << imageData;
58}
59
60template<class Decoder> inline std::optional<NowPlayingInfoArtwork> NowPlayingInfoArtwork::decode(Decoder& decoder)
61{
62 String src;
63 if (!decoder.decode(src))
64 return { };
65
66 String mimeType;
67 if (!decoder.decode(mimeType))
68 return { };
69
70 RefPtr<SharedBuffer> imageData;
71 if (!decoder.decode(imageData))
72 return { };
73
74 return NowPlayingInfoArtwork { WTFMove(src), WTFMove(mimeType), WTFMove(imageData) };
75}

77struct NowPlayingInfo {
78 String title;
79 String artist;
80 String album;
81 String sourceApplicationIdentifier;
82 double duration { 0 };
83 double currentTime { 0 };
84 bool supportsSeeking { false };
85 MediaUniqueIdentifier uniqueIdentifier;
86 bool isPlaying { false };
87 bool allowsNowPlayingControlsVisibility { false };
88 std::optional<NowPlayingInfoArtwork> artwork;
89
90 bool operator==(const NowPlayingInfo& other) const
91 {
92 return title == other.title
93 && artist == other.artist
94 && album == other.album
95 && sourceApplicationIdentifier == other.sourceApplicationIdentifier
96 && duration == other.duration
97 && currentTime == other.currentTime
98 && supportsSeeking == other.supportsSeeking
99 && uniqueIdentifier == other.uniqueIdentifier
100 && isPlaying == other.isPlaying
101 && allowsNowPlayingControlsVisibility == other.allowsNowPlayingControlsVisibility
102 && artwork == other.artwork;
103 }
104
105 bool operator!=(const NowPlayingInfo& other) const
106 {
107 return !(*this == other);
108 }
109
110 template<class Encoder> void encode(Encoder&) const;
111 template<class Decoder> static std::optional<NowPlayingInfo> decode(Decoder&);
112};
113
114template<class Encoder> inline void NowPlayingInfo::encode(Encoder& encoder) const
115{
116 encoder << title << artist << album << sourceApplicationIdentifier << duration << currentTime << supportsSeeking << uniqueIdentifier << isPlaying << allowsNowPlayingControlsVisibility << artwork;
117}
118
119template<class Decoder> inline std::optional<NowPlayingInfo> NowPlayingInfo::decode(Decoder& decoder)
120{
121 String title;
122 if (!decoder.decode(title))
123 return { };
124
125 String artist;
126 if (!decoder.decode(artist))
127 return { };
128
129 String album;
130 if (!decoder.decode(album))
131 return { };
132
133 String sourceApplicationIdentifier;
134 if (!decoder.decode(sourceApplicationIdentifier))
135 return { };
136
137 double duration;
138 if (!decoder.decode(duration))
139 return { };
140
141 double currentTime;
142 if (!decoder.decode(currentTime))
143 return { };
144
145 bool supportsSeeking;
146 if (!decoder.decode(supportsSeeking))
147 return { };
148
149 MediaUniqueIdentifier uniqueIdentifier;
150 if (!decoder.decode(uniqueIdentifier))
151 return { };
152
153 bool isPlaying;
154 if (!decoder.decode(isPlaying))
155 return { };
156
157 bool allowsNowPlayingControlsVisibility;
158 if (!decoder.decode(allowsNowPlayingControlsVisibility))
159 return { };
160
161 std::optional<NowPlayingInfoArtwork> artwork;
162 if (!decoder.decode(artwork))
163 return { };
164
165 return NowPlayingInfo { WTFMove(title), WTFMove(artist), WTFMove(album), WTFMove(sourceApplicationIdentifier), duration, currentTime, supportsSeeking, uniqueIdentifier, isPlaying, allowsNowPlayingControlsVisibility, WTFMove(artwork) };
166}
