192void PlatformMediaSessionManager::removeSession(PlatformMediaSession& session)
193{
194 ALWAYS_LOG(LOGIDENTIFIER, session.logIdentifier());
195
196 size_t index = m_sessions.find(&session);
197 if (index == notFound)
198 return;
199
200 m_sessions.remove(index);
201
202#if USE(AUDIO_SESSION)
203 if (hasNoSession())
204 maybeDeactivateAudioSession();
205#endif
206
207#if !RELEASE_LOG_DISABLED
208 m_logger->removeLogger(session.logger());
209#endif
210
211 scheduleUpdateSessionState();
212}

229bool PlatformMediaSessionManager::sessionWillBeginPlayback(PlatformMediaSession& session)
230{
231 setCurrentSession(session);
232
233 auto sessionType = session.mediaType();
234 auto restrictions = this->restrictions(sessionType);
235 if (session.state() == PlatformMediaSession::Interrupted && restrictions & InterruptedPlaybackNotPermitted) {
236 ALWAYS_LOG(LOGIDENTIFIER, session.logIdentifier(), " returning false because session.state() is Interrupted, and InterruptedPlaybackNotPermitted");
237 return false;
238 }
239
240#if USE(AUDIO_SESSION)
241 if (activeAudioSessionRequired()) {
242 if (!AudioSession::sharedSession().tryToSetActive(true)) {
243 ALWAYS_LOG(LOGIDENTIFIER, session.logIdentifier(), " returning false failed to set active AudioSession");
244 return false;
245 }
246
247 ALWAYS_LOG(LOGIDENTIFIER, session.logIdentifier(), " sucessfully activated AudioSession");
248 m_becameActive = true;
249 }
250#endif
251
252 if (m_interrupted)
253 endInterruption(PlatformMediaSession::NoFlags);
254
255 if (restrictions & ConcurrentPlaybackNotPermitted) {
256 forEachMatchingSession([&session](auto& oneSession) {
257 return &oneSession != &session
258 && oneSession.state() == PlatformMediaSession::Playing
259 && !oneSession.canPlayConcurrently(session);
260 }, [](auto& oneSession) {
261 oneSession.pauseSession();
262 });
263 }
264 ALWAYS_LOG(LOGIDENTIFIER, session.logIdentifier(), " returning true");
265 return true;
266}

268void PlatformMediaSessionManager::sessionWillEndPlayback(PlatformMediaSession& session, DelayCallingUpdateNowPlaying)
269{
270 ALWAYS_LOG(LOGIDENTIFIER, session.logIdentifier());
271
272 if (m_sessions.size() < 2)
273 return;
274
275 size_t pausingSessionIndex = notFound;
276 size_t lastPlayingSessionIndex = notFound;
277 for (size_t i = 0, size = m_sessions.size(); i < size; ++i) {
278 const auto& oneSession = *m_sessions[i];
279 if (&oneSession == &session)
280 pausingSessionIndex = i;
281 else if (oneSession.state() == PlatformMediaSession::Playing)
282 lastPlayingSessionIndex = i;
283 else
284 break;
285 }
286
287 if (lastPlayingSessionIndex == notFound || pausingSessionIndex == notFound)
288 return;
289
290 if (pausingSessionIndex > lastPlayingSessionIndex)
291 return;
292
293 m_sessions.remove(pausingSessionIndex);
294 m_sessions.append(makeWeakPtr(session));
295
296 ALWAYS_LOG(LOGIDENTIFIER, "session moved from index ", pausingSessionIndex, " to ", lastPlayingSessionIndex);
297}

304void PlatformMediaSessionManager::setCurrentSession(PlatformMediaSession& session)
305{
306 ALWAYS_LOG(LOGIDENTIFIER, session.logIdentifier());
307
308 if (m_sessions.size() < 2)
309 return;
310
311 size_t index = m_sessions.find(&session);
312 ASSERT(index != notFound);
313 if (!index || index == notFound)
314 return;
315
316 m_sessions.remove(index);
317 m_sessions.insert(0, makeWeakPtr(session));
318
319 ALWAYS_LOG(LOGIDENTIFIER, "session moved from index ", index, " to 0");
320}

330void PlatformMediaSessionManager::applicationWillBecomeInactive()
331{
332 ALWAYS_LOG(LOGIDENTIFIER);
333
334 forEachMatchingSession([&](auto& session) {
335 return restrictions(session.mediaType()) & InactiveProcessPlaybackRestricted;
336 }, [](auto& session) {

341void PlatformMediaSessionManager::applicationDidBecomeActive()
342{
343 ALWAYS_LOG(LOGIDENTIFIER);
344
345 forEachMatchingSession([&](auto& session) {
346 return restrictions(session.mediaType()) & InactiveProcessPlaybackRestricted;
347 }, [](auto& session) {
348 session.endInterruption(PlatformMediaSession::MayResumePlaying);
349 });
350}

352void PlatformMediaSessionManager::applicationDidEnterBackground(bool suspendedUnderLock)
353{
354 ALWAYS_LOG(LOGIDENTIFIER, "suspendedUnderLock: ", suspendedUnderLock);
355
356 if (m_isApplicationInBackground)
357 return;
358
359 m_isApplicationInBackground = true;
360
361 forEachSession([&] (auto& session) {
362 if (suspendedUnderLock && restrictions(session.mediaType()) & SuspendedUnderLockPlaybackRestricted)
363 session.beginInterruption(PlatformMediaSession::SuspendedUnderLock);
364 else if (restrictions(session.mediaType()) & BackgroundProcessPlaybackRestricted)
365 session.beginInterruption(PlatformMediaSession::EnteringBackground);
366 });
367}
368
369void PlatformMediaSessionManager::applicationWillEnterForeground(bool suspendedUnderLock)
370{
371 ALWAYS_LOG(LOGIDENTIFIER, "suspendedUnderLock: ", suspendedUnderLock);
372
373 if (!m_isApplicationInBackground)
374 return;
375
376 m_isApplicationInBackground = false;
377
378 forEachMatchingSession([&](auto& session) {
379 return (suspendedUnderLock && restrictions(session.mediaType()) & SuspendedUnderLockPlaybackRestricted) || restrictions(session.mediaType()) & BackgroundProcessPlaybackRestricted;
380 }, [](auto& session) {
381 session.endInterruption(PlatformMediaSession::MayResumePlaying);
382 });
383}

385void PlatformMediaSessionManager::processWillSuspend()
386{
387 if (m_processIsSuspended)
388 return;
389 m_processIsSuspended = true;
390
391 ALWAYS_LOG(LOGIDENTIFIER);
392
393 forEachSession([&] (auto& session) {
394 session.client().processIsSuspendedChanged();
395 });
396
397#if USE(AUDIO_SESSION)
398 maybeDeactivateAudioSession();
399#endif
400}
401
402void PlatformMediaSessionManager::processDidResume()
403{
404 if (!m_processIsSuspended)
405 return;
406 m_processIsSuspended = false;
407
408 forEachSession([&] (auto& session) {
409 session.client().processIsSuspendedChanged();
410 });
411
412#if USE(AUDIO_SESSION)
413 if (!m_becameActive && activeAudioSessionRequired()) {
414 m_becameActive = AudioSession::sharedSession().tryToSetActive(true);
415 ALWAYS_LOG(LOGIDENTIFIER, "tried to set active AudioSession, ", m_becameActive ? "succeeded" : "failed");
416 }
417#endif
418}
