29#ifndef ZeroPole_h
30#define ZeroPole_h
31
32namespace WebCore {
33
34// ZeroPole is a simple filter with one zero and one pole.
35
36class ZeroPole final {
37 WTF_MAKE_FAST_ALLOCATED;
38public:
39 ZeroPole()
40 : m_zero(0)
41 , m_pole(0)
42 , m_lastX(0)
43 , m_lastY(0)
44 {
45 }
46
47 void process(const float *source, float *destination, unsigned framesToProcess);
48
49 // Reset filter state.
50 void reset() { m_lastX = 0; m_lastY = 0; }
51
52 void setZero(float zero) { m_zero = zero; }
53 void setPole(float pole) { m_pole = pole; }
54
55 float zero() const { return m_zero; }
56 float pole() const { return m_pole; }
57
58private:
59 float m_zero;
60 float m_pole;
61 float m_lastX;
62 float m_lastY;
63};
64
65} // namespace WebCore
66
67#endif // ZeroPole_h
