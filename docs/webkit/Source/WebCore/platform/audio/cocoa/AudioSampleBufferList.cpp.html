61void AudioSampleBufferList::applyGain(AudioBufferList& bufferList, float gain, AudioStreamDescription::PCMFormat format)
62{
63 for (uint32_t i = 0; i < bufferList.mNumberBuffers; ++i) {
64 switch (format) {
65 case AudioStreamDescription::Int16: {
66 int16_t* buffer = static_cast<int16_t*>(bufferList.mBuffers[i].mData);
67 int frameCount = bufferList.mBuffers[i].mDataByteSize / sizeof(int16_t);
68 for (int i = 0; i < frameCount; i++)
69 buffer[i] *= gain;
70 break;
71 }
72 case AudioStreamDescription::Int32: {
73 int32_t* buffer = static_cast<int32_t*>(bufferList.mBuffers[i].mData);
74 int frameCount = bufferList.mBuffers[i].mDataByteSize / sizeof(int32_t);
75 for (int i = 0; i < frameCount; i++)
76 buffer[i] *= gain;
77 break;
78 break;
79 }
80 case AudioStreamDescription::Float32: {
81 float* buffer = static_cast<float*>(bufferList.mBuffers[i].mData);
82 vDSP_vsmul(buffer, 1, &gain, buffer, 1, bufferList.mBuffers[i].mDataByteSize / sizeof(float));
83 break;
84 }
85 case AudioStreamDescription::Float64: {
86 double* buffer = static_cast<double*>(bufferList.mBuffers[i].mData);
87 double gainAsDouble = gain;
88 vDSP_vsmulD(buffer, 1, &gainAsDouble, buffer, 1, bufferList.mBuffers[i].mDataByteSize / sizeof(double));
89 break;
90 }
91 case AudioStreamDescription::None:
92 ASSERT_NOT_REACHED();
93 break;
94 }
95 }
96}

103static void mixBuffers(WebAudioBufferList& destinationBuffer, const AudioBufferList& sourceBuffer, AudioStreamDescription::PCMFormat format, size_t frameCount)
104{
105 for (uint32_t i = 0; i < destinationBuffer.bufferCount(); i++) {
106 switch (format) {
107 case AudioStreamDescription::Int16: {
108 ASSERT(frameCount <= destinationBuffer.buffer(i)->mDataByteSize / 2);
109 ASSERT(frameCount <= sourceBuffer.mBuffers[i].mDataByteSize / 2);
110
111 auto* destination = static_cast<int16_t*>(destinationBuffer.buffer(i)->mData);
112 auto* source = static_cast<int16_t*>(sourceBuffer.mBuffers[i].mData);
113 for (size_t i = 0; i < frameCount; i++)
114 destination[i] += source[i];
115 break;
116 }
117 case AudioStreamDescription::Int32: {
118 ASSERT(frameCount <= destinationBuffer.buffer(i)->mDataByteSize / 4);
119 ASSERT(frameCount <= sourceBuffer.mBuffers[i].mDataByteSize / 4);
120
121 auto* destination = static_cast<int32_t*>(destinationBuffer.buffer(i)->mData);
122 vDSP_vaddi(destination, 1, reinterpret_cast<int32_t*>(sourceBuffer.mBuffers[i].mData), 1, destination, 1, frameCount);
123 break;
124 }
125 case AudioStreamDescription::Float32: {
126 ASSERT(frameCount <= destinationBuffer.buffer(i)->mDataByteSize / 4);
127 ASSERT(frameCount <= sourceBuffer.mBuffers[i].mDataByteSize / 4);
128
129 auto* destination = static_cast<float*>(destinationBuffer.buffer(i)->mData);
130 vDSP_vadd(destination, 1, reinterpret_cast<float*>(sourceBuffer.mBuffers[i].mData), 1, destination, 1, frameCount);
131 break;
132 }
133 case AudioStreamDescription::Float64: {
134 ASSERT(frameCount <= destinationBuffer.buffer(i)->mDataByteSize / 8);
135 ASSERT(frameCount <= sourceBuffer.mBuffers[i].mDataByteSize / 8);
136
137 auto* destination = static_cast<double*>(destinationBuffer.buffer(i)->mData);
138 vDSP_vaddD(destination, 1, reinterpret_cast<double*>(sourceBuffer.mBuffers[i].mData), 1, destination, 1, frameCount);
139 break;
140 }
141 case AudioStreamDescription::None:
142 ASSERT_NOT_REACHED();
143 break;
144 }
145 }
146}

148OSStatus AudioSampleBufferList::mixFrom(const AudioSampleBufferList& source, size_t frameCount)
149{
150 ASSERT(source.streamDescription() == streamDescription());
151
152 if (source.streamDescription() != streamDescription())
153 return kAudio_ParamError;
154
155 if (frameCount > source.sampleCount())
156 frameCount = source.sampleCount();
157
158 if (frameCount > m_sampleCapacity)
159 return kAudio_ParamError;
160
161 m_sampleCount = frameCount;
162
163 mixBuffers(bufferList(), source.bufferList(), m_internalFormat->format(), frameCount);
164 return 0;
165}

167OSStatus AudioSampleBufferList::copyFrom(const AudioSampleBufferList& source, size_t frameCount)
168{
169 ASSERT(source.streamDescription() == streamDescription());
170
171 if (source.streamDescription() != streamDescription())
172 return kAudio_ParamError;
173
174 if (frameCount > source.sampleCount())
175 frameCount = source.sampleCount();
176
177 if (frameCount > m_sampleCapacity)
178 return kAudio_ParamError;
179
180 m_sampleCount = frameCount;
181
182 for (uint32_t i = 0; i < m_bufferList->bufferCount(); i++) {
183 uint8_t* sourceData = static_cast<uint8_t*>(source.bufferList().buffer(i)->mData);
184 uint8_t* destination = static_cast<uint8_t*>(m_bufferList->buffer(i)->mData);
185 memcpy(destination, sourceData, frameCount * m_internalFormat->bytesPerPacket());
186 }
187
188 return 0;
189}

191OSStatus AudioSampleBufferList::copyTo(AudioBufferList& buffer, size_t frameCount)
192{
193 if (frameCount > m_sampleCount)
194 return kAudio_ParamError;
195 if (buffer.mNumberBuffers > m_bufferList->bufferCount())
196 return kAudio_ParamError;
197
198 for (uint32_t i = 0; i < buffer.mNumberBuffers; i++) {
199 uint8_t* sourceData = static_cast<uint8_t*>(m_bufferList->buffer(i)->mData);
200 uint8_t* destination = static_cast<uint8_t*>(buffer.mBuffers[i].mData);
201 memcpy(destination, sourceData, frameCount * m_internalFormat->bytesPerPacket());
202 }
203
204 return 0;
205}

238struct AudioConverterFromABLContext {
239 const AudioBufferList& buffer;
240 size_t packetsAvailableToConvert;
241 size_t bytesPerPacket;
242};

244static const OSStatus kRanOutOfInputDataStatus = '!mor';
245
246static OSStatus audioConverterFromABLCallback(AudioConverterRef, UInt32* ioNumberDataPackets, AudioBufferList* ioData, AudioStreamPacketDescription**, void* inRefCon)
247{
248 if (!ioNumberDataPackets || !ioData || !inRefCon) {
249 LOG_ERROR("AudioSampleBufferList::audioConverterCallback() invalid input to AudioConverterInput");
250 return kAudioConverterErr_UnspecifiedError;
251 }
252
253 auto& context = *static_cast<AudioConverterFromABLContext*>(inRefCon);
254 if (!context.packetsAvailableToConvert) {
255 *ioNumberDataPackets = 0;
256 return kRanOutOfInputDataStatus;
257 }
258
259 *ioNumberDataPackets = static_cast<UInt32>(context.packetsAvailableToConvert);
260
261 for (uint32_t i = 0; i < ioData->mNumberBuffers; ++i) {
262 ioData->mBuffers[i].mData = context.buffer.mBuffers[i].mData;
263 ioData->mBuffers[i].mDataByteSize = context.packetsAvailableToConvert * context.bytesPerPacket;
264 }
265 context.packetsAvailableToConvert = 0;
266
267 return 0;
268}

270OSStatus AudioSampleBufferList::copyFrom(const AudioBufferList& source, size_t frameCount, AudioConverterRef converter)
271{
272 reset();
273
274 AudioStreamBasicDescription inputFormat;
275 UInt32 propertyDataSize = sizeof(inputFormat);
276 PAL::AudioConverterGetProperty(converter, kAudioConverterCurrentInputStreamDescription, &propertyDataSize, &inputFormat);
277 ASSERT(frameCount <= source.mBuffers[0].mDataByteSize / inputFormat.mBytesPerPacket);
278
279 AudioConverterFromABLContext context { source, frameCount, inputFormat.mBytesPerPacket };
280
281#if !LOG_DISABLED
282 AudioStreamBasicDescription outputFormat;
283 propertyDataSize = sizeof(outputFormat);
284 PAL::AudioConverterGetProperty(converter, kAudioConverterCurrentOutputStreamDescription, &propertyDataSize, &outputFormat);
285
286 ASSERT(CAAudioStreamDescription(outputFormat).numberOfChannelStreams() == m_bufferList->bufferCount());
287 for (uint32_t i = 0; i < m_bufferList->bufferCount(); ++i) {
288 ASSERT(m_bufferList->buffer(i)->mData);
289 ASSERT(m_bufferList->buffer(i)->mDataByteSize);
290 }
291#endif
292
293 UInt32 samplesConverted = m_sampleCapacity;
294 OSStatus err = PAL::AudioConverterFillComplexBuffer(converter, audioConverterFromABLCallback, &context, &samplesConverted, m_bufferList->list(), nullptr);
295 if (!err || err == kRanOutOfInputDataStatus) {
296 m_sampleCount = samplesConverted;
297 return 0;
298 }
299
300 LOG_ERROR("AudioSampleBufferList::copyFrom(%p) AudioConverterFillComplexBuffer returned error %d (%.4s)", this, (int)err, (char*)&err);
301 m_sampleCount = std::min(m_sampleCapacity, static_cast<size_t>(samplesConverted));
302 zero();
303 return err;
304}
