73OSStatus AudioSampleDataSource::setupConverter()
74{
75 ASSERT(m_inputDescription && m_outputDescription);
76
77 if (m_converter) {
78 AudioConverterDispose(m_converter);
79 m_converter = nullptr;
80 }
81
82 if (*m_inputDescription == *m_outputDescription)
83 return 0;
84
85 OSStatus err = AudioConverterNew(&m_inputDescription->streamDescription(), &m_outputDescription->streamDescription(), &m_converter);
86 if (err) {
87 RunLoop::main().dispatch([this, protectedThis = makeRefPtr(*this), err] {
88 ERROR_LOG("AudioConverterNew returned error ", err);
89 });
90 }
91
92 return err;
93
94}

96OSStatus AudioSampleDataSource::setInputFormat(const CAAudioStreamDescription& format)
97{
98 ASSERT(format.sampleRate() >= 0);
99
100 m_inputDescription = CAAudioStreamDescription { format };
101 if (m_outputDescription)
102 return setupConverter();
103
104 return 0;

107OSStatus AudioSampleDataSource::setOutputFormat(const CAAudioStreamDescription& format)
108{
109 ASSERT(m_inputDescription);
110 ASSERT(format.sampleRate() >= 0);
111
112 m_outputDescription = CAAudioStreamDescription { format };
113
114 {
115 // Heap allocations are forbidden on the audio thread for performance reasons so we need to
116 // explicitly allow the following allocation(s).
117 DisableMallocRestrictionsForCurrentThreadScope disableMallocRestrictions;
118 m_ringBuffer->allocate(format, static_cast<size_t>(m_maximumSampleCount));
119 m_scratchBuffer = AudioSampleBufferList::create(m_outputDescription->streamDescription(), m_maximumSampleCount);
120 }
121
122 return setupConverter();
123}

125MediaTime AudioSampleDataSource::hostTime() const
126{
127 // Based on listing #2 from Apple Technical Q&A QA1398, modified to be thread-safe.
128 static double frequency;
129 static mach_timebase_info_data_t timebaseInfo;
130 static std::once_flag initializeTimerOnceFlag;
131 std::call_once(initializeTimerOnceFlag, [] {
132 kern_return_t kr = mach_timebase_info(&timebaseInfo);
133 frequency = 1e-9 * static_cast<double>(timebaseInfo.numer) / static_cast<double>(timebaseInfo.denom);
134 ASSERT_UNUSED(kr, kr == KERN_SUCCESS);
135 ASSERT(timebaseInfo.denom);
136 });
137
138 return MediaTime::createWithDouble(mach_absolute_time() * frequency);
139}

141void AudioSampleDataSource::pushSamplesInternal(const AudioBufferList& bufferList, const MediaTime& presentationTime, size_t sampleCount)
142{
143 MediaTime sampleTime = presentationTime;
144
145 const AudioBufferList* sampleBufferList;
146 if (m_converter) {
147 m_scratchBuffer->reset();
148 OSStatus err = m_scratchBuffer->copyFrom(bufferList, sampleCount, m_converter);
149 if (err)
150 return;
151
152 sampleBufferList = m_scratchBuffer->bufferList().list();
153 sampleCount = m_scratchBuffer->sampleCount();
154 sampleTime = presentationTime.toTimeScale(m_outputDescription->sampleRate(), MediaTime::RoundingFlags::TowardZero);
155 } else
156 sampleBufferList = &bufferList;
157
158 if (m_expectedNextPushedSampleTime.isValid() && abs(m_expectedNextPushedSampleTime - sampleTime).timeValue() == 1)
159 sampleTime = m_expectedNextPushedSampleTime;
160 m_expectedNextPushedSampleTime = sampleTime + MediaTime(sampleCount, sampleTime.timeScale());
161
162 if (m_inputSampleOffset == MediaTime::invalidTime())
163 m_inputSampleOffset = MediaTime(1 - sampleTime.timeValue(), sampleTime.timeScale());
164 sampleTime += m_inputSampleOffset;
165
166#if !LOG_DISABLED
167 uint64_t startFrame1 = 0;
168 uint64_t endFrame1 = 0;
169 m_ringBuffer->getCurrentFrameBounds(startFrame1, endFrame1);
170#endif
171
172 if (m_isInNeedOfMoreData) {
173 m_isInNeedOfMoreData = false;
174 DisableMallocRestrictionsForCurrentThreadScope disableMallocRestrictions;
175 RunLoop::main().dispatch([logIdentifier = LOGIDENTIFIER, sampleCount, this, protectedThis = makeRefPtr(*this)] {
176 ALWAYS_LOG(logIdentifier, "needed more data, pushing ", sampleCount, " samples");
177 });
178 }
179 m_ringBuffer->store(sampleBufferList, sampleCount, sampleTime.timeValue());
180 m_lastPushedSampleCount = sampleCount;
181}

197static inline int64_t computeOffsetDelay(double sampleRate, uint64_t lastPushedSampleCount)
198{
199 const double twentyMS = .02;
200 const double tenMS = .01;
201 const double fiveMS = .005;
202
203 if (lastPushedSampleCount > sampleRate * twentyMS)
204 return sampleRate * twentyMS;
205 if (lastPushedSampleCount > sampleRate * tenMS)
206 return sampleRate * tenMS;
207 if (lastPushedSampleCount > sampleRate * fiveMS)
208 return sampleRate * fiveMS;
209 return 0;
210}

212bool AudioSampleDataSource::pullSamplesInternal(AudioBufferList& buffer, size_t sampleCount, uint64_t timeStamp, double /*hostTime*/, PullMode mode)
213{
214 size_t byteCount = sampleCount * m_outputDescription->bytesPerFrame();
215
216 ASSERT(buffer.mNumberBuffers == m_ringBuffer->channelCount());
217 if (buffer.mNumberBuffers != m_ringBuffer->channelCount()) {
218 if (mode != AudioSampleDataSource::Mix)
219 AudioSampleBufferList::zeroABL(buffer, byteCount);
220 return false;
221 }
222
223 if (m_muted || m_inputSampleOffset == MediaTime::invalidTime()) {
224 if (mode != AudioSampleDataSource::Mix)
225 AudioSampleBufferList::zeroABL(buffer, byteCount);
226 return false;
227 }
228
229 uint64_t startFrame = 0;
230 uint64_t endFrame = 0;
231 m_ringBuffer->getCurrentFrameBounds(startFrame, endFrame);
232
233 if (m_shouldComputeOutputSampleOffset) {
234 uint64_t buffered = endFrame - startFrame;
235 if (m_isFirstPull) {
236 if (buffered >= m_waitToStartForPushCount * m_lastPushedSampleCount) {
237 m_outputSampleOffset = startFrame - timeStamp;
238 m_shouldComputeOutputSampleOffset = false;
239 m_endFrameWhenNotEnoughData = 0;
240 } else {
241 // We wait for one chunk of value before starting to play.
242 if (mode != AudioSampleDataSource::Mix)
243 AudioSampleBufferList::zeroABL(buffer, byteCount);
244 return false;
245 }
246 } else {
247 if (buffered < sampleCount * 2 || (m_endFrameWhenNotEnoughData && m_endFrameWhenNotEnoughData == endFrame)) {
248 if (mode != AudioSampleDataSource::Mix)
249 AudioSampleBufferList::zeroABL(buffer, byteCount);
250 return false;
251 }
252
253 m_shouldComputeOutputSampleOffset = false;
254 m_endFrameWhenNotEnoughData = 0;
255
256 m_outputSampleOffset = (endFrame - sampleCount) - timeStamp;
257 m_outputSampleOffset -= computeOffsetDelay(m_outputDescription->sampleRate(), m_lastPushedSampleCount);
258 }
259 }
260
261 timeStamp += m_outputSampleOffset;
262
263 if (timeStamp < startFrame || timeStamp + sampleCount > endFrame) {
264 if (!m_isInNeedOfMoreData) {
265 m_isInNeedOfMoreData = true;
266 DisableMallocRestrictionsForCurrentThreadScope disableMallocRestrictions;
267 RunLoop::main().dispatch([logIdentifier = LOGIDENTIFIER, timeStamp, startFrame, endFrame, sampleCount, outputSampleOffset = m_outputSampleOffset, this, protectedThis = makeRefPtr(*this)] {
268 ERROR_LOG(logIdentifier, "need more data, sample ", timeStamp, " with offset ", outputSampleOffset, ", trying to get ", sampleCount, " samples, but not completely in range [", startFrame, " .. ", endFrame, "]");
269 });
270 }
271 if (timeStamp < startFrame || timeStamp >= endFrame) {
272 // We are out of the window, let's restart the offset computation.
273 m_shouldComputeOutputSampleOffset = true;
274
275 if (timeStamp >= endFrame)
276 m_endFrameWhenNotEnoughData = endFrame;
277 } else {
278 // We are too close from endFrame, let's wait for more data to be pushed.
279 m_outputSampleOffset -= sampleCount;
280 }
281 if (mode != AudioSampleDataSource::Mix)
282 AudioSampleBufferList::zeroABL(buffer, byteCount);
283 return false;
284 }
285
286 m_isFirstPull = false;
287
288 if (mode == Copy) {
289 m_ringBuffer->fetch(&buffer, sampleCount, timeStamp, CARingBuffer::Copy);
290 if (m_volume < EquivalentToMaxVolume)
291 AudioSampleBufferList::applyGain(buffer, m_volume, m_outputDescription->format());
292 return true;
293 }
294
295 if (m_volume >= EquivalentToMaxVolume) {
296 m_ringBuffer->fetch(&buffer, sampleCount, timeStamp, CARingBuffer::Mix);
297 return true;
298 }
299
300 if (m_scratchBuffer->copyFrom(m_ringBuffer.get(), sampleCount, timeStamp, CARingBuffer::Copy))
301 return false;
302
303 m_scratchBuffer->applyGain(m_volume);
304 m_scratchBuffer->mixFrom(buffer, sampleCount);
305 if (m_scratchBuffer->copyTo(buffer, sampleCount))
306 return false;
307
308 return true;
309}

311bool AudioSampleDataSource::pullAvailableSamplesAsChunks(AudioBufferList& buffer, size_t sampleCountPerChunk, uint64_t timeStamp, Function<void()>&& consumeFilledBuffer)
312{
313 ASSERT(buffer.mNumberBuffers == m_ringBuffer->channelCount());
314 if (buffer.mNumberBuffers != m_ringBuffer->channelCount())
315 return false;
316
317 uint64_t startFrame = 0;
318 uint64_t endFrame = 0;
319 m_ringBuffer->getCurrentFrameBounds(startFrame, endFrame);
320 if (m_shouldComputeOutputSampleOffset) {
321 m_outputSampleOffset = timeStamp + (endFrame - sampleCountPerChunk);
322 m_shouldComputeOutputSampleOffset = false;
323 }
324
325 timeStamp += m_outputSampleOffset;
326
327 if (timeStamp < startFrame)
328 timeStamp = startFrame;
329
330 startFrame = timeStamp;
331
332 if (m_muted) {
333 AudioSampleBufferList::zeroABL(buffer, sampleCountPerChunk * m_outputDescription->bytesPerFrame());
334 while (endFrame - startFrame >= sampleCountPerChunk) {
335 consumeFilledBuffer();
336 startFrame += sampleCountPerChunk;
337 }
338 return true;
339 }
340
341 while (endFrame - startFrame >= sampleCountPerChunk) {
342 m_ringBuffer->fetch(&buffer, sampleCountPerChunk, startFrame, CARingBuffer::Copy);
343 consumeFilledBuffer();
344 startFrame += sampleCountPerChunk;
345 }
346 return true;
347}

349bool AudioSampleDataSource::pullSamples(AudioBufferList& buffer, size_t sampleCount, uint64_t timeStamp, double hostTime, PullMode mode)
350{
351 return pullSamplesInternal(buffer, sampleCount, timeStamp, hostTime, mode);
352}
353
354bool AudioSampleDataSource::pullSamples(AudioSampleBufferList& buffer, size_t sampleCount, uint64_t timeStamp, double hostTime, PullMode mode)
355{
356 if (!pullSamplesInternal(buffer.bufferList(), sampleCount, timeStamp, hostTime, mode))
357 return false;
358
359 buffer.setTimes(timeStamp, hostTime);
360 buffer.setSampleCount(sampleCount);
361
362 return true;
363}
