41class CARingBufferStorage {
42 WTF_MAKE_FAST_ALLOCATED;
43public:
44 virtual ~CARingBufferStorage() = default;
45 virtual bool allocate(size_t, const CAAudioStreamDescription& format, size_t frameCount) = 0;
46 virtual void deallocate() = 0;
47 virtual void* data() = 0;
48 virtual void getCurrentFrameBounds(uint64_t& startTime, uint64_t& endTime) = 0;
49 virtual void setCurrentFrameBounds(uint64_t startFrame, uint64_t endFrame) = 0;
50 virtual void updateFrameBounds() { }
51 virtual uint64_t currentStartFrame() const = 0;
52 virtual uint64_t currentEndFrame() const = 0;
53 virtual void flush() = 0;
54 virtual size_t size() const = 0;
55};

57class CARingBufferStorageVector final : public CARingBufferStorage {
58public:
59 CARingBufferStorageVector();
60 ~CARingBufferStorageVector() = default;
61
62private:
63 bool allocate(size_t byteCount, const CAAudioStreamDescription&, size_t) final;
64 void deallocate() final { m_buffer.clear(); }
65 void* data() final { return m_buffer.data(); }
66 void getCurrentFrameBounds(uint64_t& startTime, uint64_t& endTime) final;
67 void setCurrentFrameBounds(uint64_t startFrame, uint64_t endFrame) final;
68 uint64_t currentStartFrame() const final;
69 uint64_t currentEndFrame() const final;
70 void flush() final;
71 size_t size() const final { return m_buffer.size(); }
72
73 struct TimeBounds {
74 TimeBounds()
75 : m_startFrame(0)
76 , m_endFrame(0)
77 , m_updateCounter(0)
78 {
79 }
80 volatile uint64_t m_startFrame;
81 volatile uint64_t m_endFrame;
82 volatile uint32_t m_updateCounter;
83 };
84
85 Vector<uint8_t> m_buffer;
86 Vector<TimeBounds> m_timeBoundsQueue;
87 Lock m_currentFrameBoundsLock;
88 std::atomic<int32_t> m_timeBoundsQueuePtr { 0 };
89};

91class CARingBuffer {
92 WTF_MAKE_FAST_ALLOCATED;
93public:
94 WEBCORE_EXPORT CARingBuffer();
95 WEBCORE_EXPORT explicit CARingBuffer(UniqueRef<CARingBufferStorage>&&);
96 WEBCORE_EXPORT ~CARingBuffer();
97
98 WEBCORE_EXPORT static UniqueRef<CARingBuffer> adoptStorage(UniqueRef<CARingBufferStorage>&&, const CAAudioStreamDescription&, size_t frameCount);
99
100 enum Error {
101 Ok,
102 TooMuch, // fetch start time is earlier than buffer start time and fetch end time is later than buffer end time
103 };
104
105 WEBCORE_EXPORT bool allocate(const CAAudioStreamDescription&, size_t frameCount);
106 WEBCORE_EXPORT void deallocate();
107
108 WEBCORE_EXPORT Error store(const AudioBufferList*, size_t frameCount, uint64_t startFrame);
109
110 enum FetchMode { Copy, Mix };
111 WEBCORE_EXPORT bool fetchIfHasEnoughData(AudioBufferList*, size_t frameCount, uint64_t startFrame, FetchMode = Copy);
112
113 // Fills buffer with silence if there is not enough data.
114 WEBCORE_EXPORT void fetch(AudioBufferList*, size_t frameCount, uint64_t startFrame, FetchMode = Copy);
115
116 WEBCORE_EXPORT void flush();
117
118 WEBCORE_EXPORT void getCurrentFrameBounds(uint64_t& startFrame, uint64_t& endFrame);
119
120 uint32_t channelCount() const { return m_channelCount; }
121 CARingBufferStorage& storage() { return m_buffers; }
122
123private:
124 void updateFrameBounds();
125 size_t frameOffset(uint64_t frameNumber) const { return (frameNumber % m_frameCount) * m_bytesPerFrame; }
126
127 void clipTimeBounds(uint64_t& startRead, uint64_t& endRead);
128 void setCurrentFrameBounds(uint64_t startFrame, uint64_t endFrame);
129
130 void getCurrentFrameBoundsWithoutUpdate(uint64_t& startFrame, uint64_t& endFrame);
131 void fetchInternal(AudioBufferList*, size_t frameCount, uint64_t startFrame, FetchMode);
132
133 void initializeAfterAllocation(const CAAudioStreamDescription& format, size_t frameCount);
134
135 uint64_t currentStartFrame() const;
136 uint64_t currentEndFrame() const;
137
138 UniqueRef<CARingBufferStorage> m_buffers;
139 Vector<Byte*> m_pointers;
140 uint32_t m_channelCount { 0 };
141 size_t m_bytesPerFrame { 0 };
142 uint32_t m_frameCount { 0 };
143 size_t m_capacityBytes { 0 };
144
145 CAAudioStreamDescription m_description;
146};
