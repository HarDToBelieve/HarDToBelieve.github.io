36WebAudioBufferList::WebAudioBufferList(const CAAudioStreamDescription& format)
37 : m_bytesPerFrame(format.bytesPerFrame())
38 , m_channelCount(format.numberOfInterleavedChannels())
39{
40 // AudioBufferList is a variable-length struct, so create on the heap with a generic new() operator
41 // with a custom size, and initialize the struct manually.
42 uint32_t bufferCount = format.numberOfChannelStreams();
43
44 CheckedSize bufferListSize = offsetof(AudioBufferList, mBuffers);
45 bufferListSize += CheckedSize { sizeof(AudioBuffer) } * std::max(1U, bufferCount);
46 m_listBufferSize = bufferListSize;
47 m_canonicalList = std::unique_ptr<AudioBufferList>(static_cast<AudioBufferList*>(::operator new (m_listBufferSize)));
48 memset(m_canonicalList.get(), 0, m_listBufferSize);
49 m_canonicalList->mNumberBuffers = bufferCount;
50 for (uint32_t buffer = 0; buffer < bufferCount; ++buffer)
51 m_canonicalList->mBuffers[buffer].mNumberChannels = m_channelCount;
52
53 reset();
54}

62static inline std::optional<std::pair<size_t, size_t>> computeBufferSizes(uint32_t numberOfInterleavedChannels, uint32_t bytesPerFrame, uint32_t numberOfChannelStreams, uint32_t sampleCount)
63{
64 size_t totalSampleCount;
65 bool result = WTF::safeMultiply(sampleCount, numberOfInterleavedChannels, totalSampleCount);
66 if (!result)
67 return { };
68
69 size_t bytesPerBuffer;
70 result = WTF::safeMultiply(bytesPerFrame, totalSampleCount, bytesPerBuffer);
71 if (!result)
72 return { };
73
74 size_t flatBufferSize;
75 result = WTF::safeMultiply(numberOfChannelStreams, bytesPerBuffer, flatBufferSize);
76 if (!result)
77 return { };
78
79 return std::make_pair(bytesPerBuffer, flatBufferSize);
80}

82bool WebAudioBufferList::isSupportedDescription(const CAAudioStreamDescription& format, uint32_t sampleCount)
83{
84 return !!computeBufferSizes(format.numberOfInterleavedChannels(), format.bytesPerFrame(), format.numberOfChannelStreams(), sampleCount);
85}

87void WebAudioBufferList::setSampleCount(uint32_t sampleCount)
88{
89 if (!sampleCount || m_sampleCount == sampleCount)
90 return;
91
92 auto bufferSizes = computeBufferSizes(m_channelCount, m_bytesPerFrame, m_canonicalList->mNumberBuffers, sampleCount);
93 ASSERT(bufferSizes);
94 if (!bufferSizes)
95 return;
96
97 m_sampleCount = sampleCount;
98
99 m_flatBuffer.resize(bufferSizes->second);
100 auto* data = m_flatBuffer.data();
101
102 for (uint32_t buffer = 0; buffer < m_canonicalList->mNumberBuffers; ++buffer) {
103 m_canonicalList->mBuffers[buffer].mData = data;
104 m_canonicalList->mBuffers[buffer].mDataByteSize = bufferSizes->first;
105 data += bufferSizes->first;
106 }
107
108 reset();
109}

111WebAudioBufferList::WebAudioBufferList(const CAAudioStreamDescription& format, CMSampleBufferRef sampleBuffer)
112 : WebAudioBufferList(format)
113{
114 if (!sampleBuffer)
115 return;
116
117 CMBlockBufferRef buffer = nullptr;
118 if (noErr == PAL::CMSampleBufferGetAudioBufferListWithRetainedBlockBuffer(sampleBuffer, nullptr, m_canonicalList.get(), m_listBufferSize, kCFAllocatorSystemDefault, kCFAllocatorSystemDefault, kCMSampleBufferFlag_AudioBufferList_Assure16ByteAlignment, &buffer))
119 m_blockBuffer = adoptCF(buffer);
120
121 reset();
122}

124void WebAudioBufferList::reset()
125{
126 if (!m_list)
127 m_list = std::unique_ptr<AudioBufferList>(static_cast<AudioBufferList*>(::operator new (m_listBufferSize)));
128 memcpy(m_list.get(), m_canonicalList.get(), m_listBufferSize);
129}

131WTF::IteratorRange<AudioBuffer*> WebAudioBufferList::buffers() const
132{
133 return WTF::makeIteratorRange(&m_list->mBuffers[0], &m_list->mBuffers[m_list->mNumberBuffers]);
134}

136uint32_t WebAudioBufferList::bufferCount() const
137{

141AudioBuffer* WebAudioBufferList::buffer(uint32_t index) const
142{
143 ASSERT(index < m_list->mNumberBuffers);
144 if (index < m_list->mNumberBuffers)
145 return &m_list->mBuffers[index];
146 return nullptr;
147}
