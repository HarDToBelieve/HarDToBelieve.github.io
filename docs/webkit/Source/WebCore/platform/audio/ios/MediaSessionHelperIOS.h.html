34class MediaPlaybackTarget;

36class MediaSessionHelperClient : public CanMakeWeakPtr<MediaSessionHelperClient> {
37public:
38 virtual ~MediaSessionHelperClient() = default;
39
40 enum class SuspendedUnderLock : bool { No, Yes };
41 virtual void applicationWillEnterForeground(SuspendedUnderLock) = 0;
42 virtual void applicationDidEnterBackground(SuspendedUnderLock) = 0;
43 virtual void applicationWillBecomeInactive() = 0;
44 virtual void applicationDidBecomeActive() = 0;
45 virtual void mediaServerConnectionDied() { };
46
47 enum class HasAvailableTargets : bool { No, Yes };
48 virtual void externalOutputDeviceAvailableDidChange(HasAvailableTargets) = 0;
49
50 enum class PlayingToAutomotiveHeadUnit : bool { No, Yes };
51 virtual void isPlayingToAutomotiveHeadUnitDidChange(PlayingToAutomotiveHeadUnit) = 0;
52
53 enum class ShouldPause : bool { No, Yes };
54 virtual void activeAudioRouteDidChange(ShouldPause) = 0;
55
56 enum class SupportsAirPlayVideo : bool { No, Yes };
57 virtual void activeVideoRouteDidChange(SupportsAirPlayVideo, Ref<MediaPlaybackTarget>&&) = 0;
58};

60class WEBCORE_EXPORT MediaSessionHelper {
61 WTF_MAKE_FAST_ALLOCATED;
62public:
63 static MediaSessionHelper& sharedHelper();
64 static void setSharedHelper(UniqueRef<MediaSessionHelper>&&);
65 static void resetSharedHelper();
66
67 MediaSessionHelper() = default;
68 explicit MediaSessionHelper(bool isExternalOutputDeviceAvailable);
69 virtual ~MediaSessionHelper() = default;
70
71 void addClient(MediaSessionHelperClient&);
72 void removeClient(MediaSessionHelperClient&);
73
74 void startMonitoringWirelessRoutes();
75 void stopMonitoringWirelessRoutes();
76 virtual void providePresentingApplicationPID(int) = 0;
77
78 void setIsExternalOutputDeviceAvailable(bool);
79
80 bool isMonitoringWirelessRoutes() const { return m_monitoringWirelessRoutesCount; }
81 bool isExternalOutputDeviceAvailable() const { return m_isExternalOutputDeviceAvailable; }
82 bool activeVideoRouteSupportsAirPlayVideo() const { return m_activeVideoRouteSupportsAirPlayVideo; }
83 bool isPlayingToAutomotiveHeadUnit() const { return m_isPlayingToAutomotiveHeadUnit; }
84
85 MediaPlaybackTarget* playbackTarget() const { return m_playbackTarget.get(); }
86
87 using HasAvailableTargets = MediaSessionHelperClient::HasAvailableTargets;
88 using PlayingToAutomotiveHeadUnit = MediaSessionHelperClient::PlayingToAutomotiveHeadUnit;
89 using ShouldPause = MediaSessionHelperClient::ShouldPause;
90 using SupportsAirPlayVideo = MediaSessionHelperClient::SupportsAirPlayVideo;
91 using SuspendedUnderLock = MediaSessionHelperClient::SuspendedUnderLock;
92
93 void activeAudioRouteDidChange(ShouldPause);
94 void applicationWillEnterForeground(SuspendedUnderLock);
95 void applicationDidEnterBackground(SuspendedUnderLock);
96 void applicationWillBecomeInactive();
97 void applicationDidBecomeActive();
98 void mediaServerConnectionDied();
99
100protected:
101 void externalOutputDeviceAvailableDidChange(HasAvailableTargets);
102 void isPlayingToAutomotiveHeadUnitDidChange(PlayingToAutomotiveHeadUnit);
103 void activeVideoRouteDidChange(SupportsAirPlayVideo, Ref<MediaPlaybackTarget>&&);
104
105private:
106 virtual void startMonitoringWirelessRoutesInternal() = 0;
107 virtual void stopMonitoringWirelessRoutesInternal() = 0;
108
109 WeakHashSet<MediaSessionHelperClient> m_clients;
110 bool m_isExternalOutputDeviceAvailable { false };
111 uint32_t m_monitoringWirelessRoutesCount { 0 };
112 bool m_activeVideoRouteSupportsAirPlayVideo { false };
113 bool m_isPlayingToAutomotiveHeadUnit { false };
114 RefPtr<MediaPlaybackTarget> m_playbackTarget;
115};

125}
