39static AudioHardwareActivityType isAudioHardwareProcessRunning()
40{
41 AudioObjectPropertyAddress propertyAddress = {
42 kAudioHardwarePropertyProcessIsRunning,
43 kAudioObjectPropertyScopeGlobal,
44#if HAVE(AUDIO_OBJECT_PROPERTY_ELEMENT_MAIN)
45 kAudioObjectPropertyElementMain
46#else
47 ALLOW_DEPRECATED_DECLARATIONS_BEGIN
48 kAudioObjectPropertyElementMaster
49 ALLOW_DEPRECATED_DECLARATIONS_END
50#endif
51 };
52
53 if (!AudioObjectHasProperty(kAudioObjectSystemObject, &propertyAddress))
54 return AudioHardwareActivityType::Unknown;
55
56 UInt32 result = 0;
57 UInt32 resultSize = sizeof(UInt32);
58
59 if (AudioObjectGetPropertyData(kAudioObjectSystemObject, &propertyAddress, 0, 0, &resultSize, &result))
60 return AudioHardwareActivityType::Unknown;
61
62 if (result)
63 return AudioHardwareActivityType::IsActive;
64 else
65 return AudioHardwareActivityType::IsInactive;
66}

68static AudioHardwareListener::BufferSizeRange currentDeviceSupportedBufferSizes()
69{
70 AudioDeviceID deviceID = kAudioDeviceUnknown;
71 UInt32 descriptorSize = sizeof(deviceID);
72 AudioObjectPropertyAddress defaultOutputDeviceDescriptor = {
73 kAudioHardwarePropertyDefaultOutputDevice,
74 kAudioObjectPropertyScopeGlobal,
75#if HAVE(AUDIO_OBJECT_PROPERTY_ELEMENT_MAIN)
76 kAudioObjectPropertyElementMain
77#else
78 ALLOW_DEPRECATED_DECLARATIONS_BEGIN
79 kAudioObjectPropertyElementMaster
80 ALLOW_DEPRECATED_DECLARATIONS_END
81#endif
82 };
83
84 if (AudioObjectGetPropertyData(kAudioObjectSystemObject, &defaultOutputDeviceDescriptor, 0, 0, &descriptorSize, (void*)&deviceID))
85 return { };
86
87 AudioValueRange bufferSizes;
88 descriptorSize = sizeof(bufferSizes);
89
90 AudioObjectPropertyAddress bufferSizeDescriptor = {
91 kAudioDevicePropertyBufferFrameSizeRange,
92 kAudioObjectPropertyScopeGlobal,
93#if HAVE(AUDIO_OBJECT_PROPERTY_ELEMENT_MAIN)
94 kAudioObjectPropertyElementMain
95#else
96 ALLOW_DEPRECATED_DECLARATIONS_BEGIN
97 kAudioObjectPropertyElementMaster,
98 ALLOW_DEPRECATED_DECLARATIONS_END
99#endif
100 };
101
102 if (AudioObjectGetPropertyData(deviceID, &bufferSizeDescriptor, 0, 0, &descriptorSize, &bufferSizes))
103 return { };
104
105 return { static_cast<size_t>(bufferSizes.mMinimum), static_cast<size_t>(bufferSizes.mMaximum) };
106}

148AudioHardwareListenerMac::AudioHardwareListenerMac(Client& client)
149 : AudioHardwareListener(client)
150{
151 setHardwareActivity(isAudioHardwareProcessRunning());
152 setSupportedBufferSizes(currentDeviceSupportedBufferSizes());
153
154 auto weakThis = makeWeakPtr(*this);
155 m_block = Block_copy(^(UInt32 count, const AudioObjectPropertyAddress properties[]) {
156 if (weakThis)
157 weakThis->propertyChanged(count, properties);
158 });
159
160 AudioObjectAddPropertyListenerBlock(kAudioObjectSystemObject, &processIsRunningPropertyDescriptor(), dispatch_get_main_queue(), m_block);
161 AudioObjectAddPropertyListenerBlock(kAudioObjectSystemObject, &outputDevicePropertyDescriptor(), dispatch_get_main_queue(), m_block);
162}
163
164AudioHardwareListenerMac::~AudioHardwareListenerMac()
165{
166 AudioObjectRemovePropertyListenerBlock(kAudioObjectSystemObject, &processIsRunningPropertyDescriptor(), dispatch_get_main_queue(), m_block);
167 AudioObjectRemovePropertyListenerBlock(kAudioObjectSystemObject, &outputDevicePropertyDescriptor(), dispatch_get_main_queue(), m_block);
168 Block_release(m_block);
169}
170
171void AudioHardwareListenerMac::propertyChanged(UInt32 propertyCount, const AudioObjectPropertyAddress properties[])
172{
173 const AudioObjectPropertyAddress& deviceRunning = processIsRunningPropertyDescriptor();
174 const AudioObjectPropertyAddress& outputDevice = outputDevicePropertyDescriptor();
175
176 for (UInt32 i = 0; i < propertyCount; ++i) {
177 const AudioObjectPropertyAddress& property = properties[i];
178
179 if (!memcmp(&property, &deviceRunning, sizeof(AudioObjectPropertyAddress)))
180 processIsRunningChanged();
181 else if (!memcmp(&property, &outputDevice, sizeof(AudioObjectPropertyAddress)))
182 outputDeviceChanged();
183 }
184}
185
186void AudioHardwareListenerMac::processIsRunningChanged()
187{
188 AudioHardwareActivityType activity = isAudioHardwareProcessRunning();
189 if (activity == hardwareActivity())
190 return;
191 setHardwareActivity(activity);
192
193 if (hardwareActivity() == AudioHardwareActivityType::IsActive)
194 m_client.audioHardwareDidBecomeActive();
195 else if (hardwareActivity() == AudioHardwareActivityType::IsInactive)
196 m_client.audioHardwareDidBecomeInactive();
197}
198
199void AudioHardwareListenerMac::outputDeviceChanged()
200{
201 setSupportedBufferSizes(currentDeviceSupportedBufferSizes());
202 m_client.audioOutputDeviceChanged();
203}
