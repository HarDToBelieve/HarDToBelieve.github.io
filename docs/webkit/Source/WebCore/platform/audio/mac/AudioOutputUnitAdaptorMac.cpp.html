41void AudioOutputUnitAdaptor::configure(float hardwareSampleRate, unsigned numberOfOutputChannels)
42{
43 // Open and initialize DefaultOutputUnit
44 AudioComponent comp;
45 AudioComponentDescription desc;
46
47 desc.componentType = kAudioUnitType_Output;
48 desc.componentSubType = kAudioUnitSubType_DefaultOutput;
49 desc.componentManufacturer = kAudioUnitManufacturer_Apple;
50 desc.componentFlags = 0;
51 desc.componentFlagsMask = 0;
52 comp = PAL::AudioComponentFindNext(0, &desc);
53
54 ASSERT(comp);
55
56 OSStatus result = PAL::AudioComponentInstanceNew(comp, &m_outputUnit);
57 ASSERT(!result);
58
59 result = PAL::AudioUnitInitialize(m_outputUnit);
60 ASSERT(!result);
61
62 // Set render callback
63 AURenderCallbackStruct input;
64 input.inputProc = inputProc;
65 input.inputProcRefCon = this;
66 result = PAL::AudioUnitSetProperty(m_outputUnit, kAudioUnitProperty_SetRenderCallback, kAudioUnitScope_Global, 0, &input, sizeof(input));
67 ASSERT(!result);
68
69 // Set stream format
70 constexpr int bytesPerFloat = sizeof(Float32);
71 constexpr int bitsPerByte = 8;
72
73 AudioStreamBasicDescription streamFormat;
74 streamFormat.mSampleRate = hardwareSampleRate;
75 streamFormat.mFormatID = kAudioFormatLinearPCM;
76 streamFormat.mFormatFlags = kAudioFormatFlagsNativeFloatPacked | kAudioFormatFlagIsNonInterleaved;
77 streamFormat.mBytesPerPacket = bytesPerFloat;
78 streamFormat.mFramesPerPacket = 1;
79 streamFormat.mBytesPerFrame = bytesPerFloat;
80 streamFormat.mChannelsPerFrame = numberOfOutputChannels;
81 streamFormat.mBitsPerChannel = bitsPerByte * bytesPerFloat;
82
83 result = PAL::AudioUnitSetProperty(m_outputUnit, kAudioUnitProperty_StreamFormat, kAudioUnitScope_Input, 0, (void*)&streamFormat, sizeof(AudioStreamBasicDescription));
84 ASSERT(!result);
85}
