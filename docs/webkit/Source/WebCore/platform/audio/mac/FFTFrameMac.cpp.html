46constexpr unsigned kMinFFTPow2Size = 2;
47constexpr unsigned kMaxFFTPow2Size = 24;
48
49static Lock fftSetupsLock;
50
51static Vector<FFTSetup>& fftSetups() WTF_REQUIRES_LOCK(fftSetupsLock)
52{
53 ASSERT(fftSetupsLock.isHeld());
54 static NeverDestroyed<Vector<FFTSetup>> fftSetups(kMaxFFTPow2Size, nullptr);
55 return fftSetups;
56}

58// Normal constructor: allocates for a given fftSize
59FFTFrame::FFTFrame(unsigned fftSize)
60 : m_realData(fftSize)
61 , m_imagData(fftSize)
62{
63 m_FFTSize = fftSize;
64 m_log2FFTSize = static_cast<unsigned>(log2(fftSize));
65
66 // We only allow power of two
67 ASSERT(1UL << m_log2FFTSize == m_FFTSize);
68
69 // Lazily create and share fftSetup with other frames
70 m_FFTSetup = fftSetupForSize(fftSize);
71
72 // Setup frame data
73 m_frame.realp = m_realData.data();
74 m_frame.imagp = m_imagData.data();
75}
76
77// Creates a blank/empty frame (interpolate() must later be called)
78FFTFrame::FFTFrame()
79{
80 // Later will be set to correct values when interpolate() is called
81 m_frame.realp = 0;
82 m_frame.imagp = 0;
83
84 m_FFTSize = 0;
85 m_log2FFTSize = 0;
86}
87
88// Copy constructor
89FFTFrame::FFTFrame(const FFTFrame& frame)
90 : m_FFTSize(frame.m_FFTSize)
91 , m_log2FFTSize(frame.m_log2FFTSize)
92 , m_FFTSetup(frame.m_FFTSetup)
93 , m_realData(frame.m_FFTSize)
94 , m_imagData(frame.m_FFTSize)
95{
96 // Setup frame data
97 m_frame.realp = m_realData.data();
98 m_frame.imagp = m_imagData.data();
99
100 // Copy/setup frame data
101 memcpy(realData().data(), frame.m_frame.realp, sizeof(float) * realData().size());
102 memcpy(imagData().data(), frame.m_frame.imagp, sizeof(float) * imagData().size());
103}

107void FFTFrame::doFFT(const float* data)
108{
109 unsigned halfSize = m_FFTSize / 2;
110 vDSP_ctoz(reinterpret_cast<const DSPComplex*>(data), 2, &m_frame, 1, halfSize);
111 vDSP_fft_zrip(m_FFTSetup, &m_frame, 1, m_log2FFTSize, FFT_FORWARD);
112
113 RELEASE_ASSERT(realData().size() >= halfSize);
114 RELEASE_ASSERT(imagData().size() >= halfSize);
115
116 // To provide the best possible execution speeds, the vDSP library's functions don't always adhere strictly
117 // to textbook formulas for Fourier transforms, and must be scaled accordingly.
118 // (See https://developer.apple.com/library/archive/documentation/Performance/Conceptual/vDSP_Programming_Guide/UsingFourierTransforms/UsingFourierTransforms.html#//apple_ref/doc/uid/TP40005147-CH3-SW5)
119 // In the case of a Real forward Transform like above: RFimp = RFmath * 2 so we need to divide the output
120 // by 2 to get the correct value.
121 VectorMath::multiplyByScalar(realData().data(), 0.5, realData().data(), halfSize);
122 VectorMath::multiplyByScalar(imagData().data(), 0.5, imagData().data(), halfSize);
123}
124
125void FFTFrame::doInverseFFT(float* data)
126{
127 vDSP_fft_zrip(m_FFTSetup, &m_frame, 1, m_log2FFTSize, FFT_INVERSE);
128 vDSP_ztoc(&m_frame, 1, (DSPComplex*)data, 2, m_FFTSize / 2);
129
130 // Do final scaling so that x == IFFT(FFT(x))
131 VectorMath::multiplyByScalar(data, 1.0f / m_FFTSize, data, m_FFTSize);
132}

134FFTSetup FFTFrame::fftSetupForSize(unsigned fftSize)
135{
136 auto pow2size = static_cast<size_t>(log2(fftSize));
137 ASSERT(pow2size < kMaxFFTPow2Size);
138
139 Locker locker { fftSetupsLock };
140 auto& fftSetup = fftSetups().at(pow2size);
141 if (!fftSetup)
142 fftSetup = vDSP_create_fftsetup(pow2size, FFT_RADIX2);
143
144 return fftSetup;
145}

147int FFTFrame::minFFTSize()
148{
149 return 1 << kMinFFTPow2Size;
150}
151
152int FFTFrame::maxFFTSize()
153{
154 return 1 << kMaxFFTPow2Size;
155}
