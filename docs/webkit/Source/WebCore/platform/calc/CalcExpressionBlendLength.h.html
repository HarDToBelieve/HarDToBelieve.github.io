28#include "CalcExpressionNode.h"
29#include "Length.h"

31namespace WebCore {
32
33class CalcExpressionBlendLength final : public CalcExpressionNode {
34public:
35 CalcExpressionBlendLength(Length from, Length to, float progress);
36
37 const Length& from() const { return m_from; }
38 const Length& to() const { return m_to; }
39 float progress() const { return m_progress; }
40
41private:
42 float evaluate(float maxValue) const final;
43 bool operator==(const CalcExpressionNode&) const final;
44 void dump(TextStream&) const final;
45
46 Length m_from;
47 Length m_to;
48 float m_progress;
49};
50
51inline bool operator==(const CalcExpressionBlendLength& a, const CalcExpressionBlendLength& b)
52{
53 return a.progress() == b.progress() && a.from() == b.from() && a.to() == b.to();
54}
55
56}
