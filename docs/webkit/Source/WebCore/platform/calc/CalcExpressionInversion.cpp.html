33float CalcExpressionInversion::evaluate(float maxValue) const
34{
35 return 1.0f / m_child->evaluate(maxValue);
36}
37
38void CalcExpressionInversion::dump(TextStream& ts) const
39{
40 ts << "1.0 / " << "(" << *m_child << ")";
41}
42
43bool CalcExpressionInversion::operator==(const CalcExpressionNode& other) const
44{
45 return is<CalcExpressionInversion>(other) && *this == downcast<CalcExpressionInversion>(other);
46}
47
48bool operator==(const CalcExpressionInversion& a, const CalcExpressionInversion& b)
49{
50 return *a.child() == *b.child();
51}
