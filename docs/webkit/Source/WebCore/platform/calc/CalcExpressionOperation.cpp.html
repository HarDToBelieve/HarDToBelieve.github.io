33float CalcExpressionOperation::evaluate(float maxValue) const
34{
35 switch (m_operator) {
36 case CalcOperator::Add: {
37 float sum = 0;
38 for (auto& child : m_children)
39 sum += child->evaluate(maxValue);
40 return sum;
41 }
42 case CalcOperator::Subtract: {
43 // FIXME
44 ASSERT(m_children.size() == 2);
45 float left = m_children[0]->evaluate(maxValue);
46 float right = m_children[1]->evaluate(maxValue);
47 return left - right;
48 }
49 case CalcOperator::Multiply: {
50 float product = 1;
51 for (auto& child : m_children)
52 product *= child->evaluate(maxValue);
53 return product;
54 }
55 case CalcOperator::Divide: {
56 // FIXME
57 ASSERT(m_children.size() == 1 || m_children.size() == 2);
58 if (m_children.size() == 1)
59 return std::numeric_limits<float>::quiet_NaN();
60 float left = m_children[0]->evaluate(maxValue);
61 float right = m_children[1]->evaluate(maxValue);
62 return left / right;
63 }
64 case CalcOperator::Min: {
65 if (m_children.isEmpty())
66 return std::numeric_limits<float>::quiet_NaN();
67 float minimum = m_children[0]->evaluate(maxValue);
68 for (auto& child : m_children)
69 minimum = std::min(minimum, child->evaluate(maxValue));
70 return minimum;
71 }
72 case CalcOperator::Max: {
73 if (m_children.isEmpty())
74 return std::numeric_limits<float>::quiet_NaN();
75 float maximum = m_children[0]->evaluate(maxValue);
76 for (auto& child : m_children)
77 maximum = std::max(maximum, child->evaluate(maxValue));
78 return maximum;
79 }
80 case CalcOperator::Clamp: {
81 if (m_children.size() != 3)
82 return std::numeric_limits<float>::quiet_NaN();
83
84 float min = m_children[0]->evaluate(maxValue);
85 float value = m_children[1]->evaluate(maxValue);
86 float max = m_children[2]->evaluate(maxValue);
87 return std::max(min, std::min(value, max));
88 }
89 }
90 ASSERT_NOT_REACHED();
91 return std::numeric_limits<float>::quiet_NaN();
92}

94bool CalcExpressionOperation::operator==(const CalcExpressionNode& other) const
95{
96 return is<CalcExpressionOperation>(other) && *this == downcast<CalcExpressionOperation>(other);
97}
98
99bool operator==(const CalcExpressionOperation& a, const CalcExpressionOperation& b)
100{
101 if (a.getOperator() != b.getOperator() || a.destinationCategory() != b.destinationCategory())
102 return false;
103 if (a.children().size() != b.children().size())
104 return false;
105 for (unsigned i = 0; i < a.children().size(); ++i) {
106 if (!(*a.children()[i] == *b.children()[i]))
107 return false;
108 }
109 return true;
110}

112void CalcExpressionOperation::dump(TextStream& ts) const
113{
114 if (m_operator == CalcOperator::Min || m_operator == CalcOperator::Max) {
115 ts << m_operator << "(";
116 size_t childrenCount = m_children.size();
117 for (size_t i = 0; i < childrenCount; i++) {
118 ts << m_children[i].get();
119 if (i < childrenCount - 1)
120 ts << ", ";
121 }
122 ts << ")";
123 } else
124 ts << m_children[0].get() << " " << m_operator << " " << m_children[1].get();
125}
