71void ContentFilterUnblockHandler::wrapWithDecisionHandler(const DecisionHandlerFunction& decisionHandler)
72{
73 ContentFilterUnblockHandler wrapped { *this };
74 UnblockRequesterFunction wrappedRequester { [wrapped, decisionHandler](DecisionHandlerFunction wrappedDecisionHandler) {
75 wrapped.requestUnblockAsync([wrappedDecisionHandler, decisionHandler](bool unblocked) {
76 wrappedDecisionHandler(unblocked);
77 decisionHandler(unblocked);
78 });
79 }};
80#if HAVE(PARENTAL_CONTROLS_WITH_UNBLOCK_HANDLER)
81 m_webFilterEvaluator = nullptr;
82#endif
83 std::swap(m_unblockRequester, wrappedRequester);
84}

95void ContentFilterUnblockHandler::encode(NSCoder *coder) const
96{
97 ASSERT_ARG(coder, coder.allowsKeyedCoding && coder.requiresSecureCoding);
98 BEGIN_BLOCK_OBJC_EXCEPTIONS
99 [coder encodeObject:m_unblockURLHost forKey:unblockURLHostKey];
100 [coder encodeObject:(NSURL *)m_unreachableURL forKey:unreachableURLKey];
101#if HAVE(PARENTAL_CONTROLS_WITH_UNBLOCK_HANDLER)
102 [coder encodeObject:m_webFilterEvaluator.get() forKey:webFilterEvaluatorKey];
103#endif
104 END_BLOCK_OBJC_EXCEPTIONS
105}
106
107bool ContentFilterUnblockHandler::decode(NSCoder *coder, ContentFilterUnblockHandler& unblockHandler)
108{
109 ASSERT_ARG(coder, coder.allowsKeyedCoding && coder.requiresSecureCoding);
110 BEGIN_BLOCK_OBJC_EXCEPTIONS
111 unblockHandler.m_unblockURLHost = [coder decodeObjectOfClass:[NSString class] forKey:unblockURLHostKey];
112 unblockHandler.m_unreachableURL = [coder decodeObjectOfClass:[NSURL class] forKey:unreachableURLKey];
113#if HAVE(PARENTAL_CONTROLS_WITH_UNBLOCK_HANDLER)
114 unblockHandler.m_webFilterEvaluator = [coder decodeObjectOfClass:getWebFilterEvaluatorClass() forKey:webFilterEvaluatorKey];
115#endif
116 return true;
117 END_BLOCK_OBJC_EXCEPTIONS
118 return false;
119}

121bool ContentFilterUnblockHandler::canHandleRequest(const ResourceRequest& request) const
122{
123 if (!m_unblockRequester) {
124#if HAVE(PARENTAL_CONTROLS_WITH_UNBLOCK_HANDLER)
125 if (!m_webFilterEvaluator)
126 return false;
127#else
128 return false;
129#endif
130 }
131
132 bool isUnblockRequest = request.url().protocolIs(ContentFilter::urlScheme()) && equalIgnoringASCIICase(request.url().host(), m_unblockURLHost);
133#if !LOG_DISABLED
134 if (isUnblockRequest)
135 LOG(ContentFiltering, "ContentFilterUnblockHandler will handle <%s> as an unblock request.\n", request.url().string().ascii().data());
136#endif
137 return isUnblockRequest;
138}
139
140void ContentFilterUnblockHandler::requestUnblockAsync(DecisionHandlerFunction decisionHandler) const
141{
142#if HAVE(PARENTAL_CONTROLS_WITH_UNBLOCK_HANDLER)
143 if (m_webFilterEvaluator) {
144 [m_webFilterEvaluator unblockWithCompletion:[decisionHandler](BOOL unblocked, NSError *) {
145 callOnMainThread([decisionHandler, unblocked] {
146 LOG(ContentFiltering, "WebFilterEvaluator %s the unblock request.\n", unblocked ? "allowed" : "did not allow");
147 decisionHandler(unblocked);
148 });
149 }];
150 return;
151 }
152#endif
153
154 if (m_unblockRequester) {
155 m_unblockRequester([decisionHandler](bool unblocked) {
156 callOnMainThread([decisionHandler, unblocked] {
157 decisionHandler(unblocked);
158 });
159 });
160 }
161}
