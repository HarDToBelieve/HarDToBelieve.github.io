194Vector<String> DragData::asFilenames() const
195{
196 auto context = createPasteboardContext();
197#if PLATFORM(MAC)
198 Vector<String> types;
199 platformStrategies()->pasteboardStrategy()->getTypes(types, m_pasteboardName, context.get());
200 if (types.contains(String(legacyFilesPromisePasteboardType())))
201 return fileNames();
202
203 Vector<String> results;
204 platformStrategies()->pasteboardStrategy()->getPathnamesForType(results, String(legacyFilenamesPasteboardType()), m_pasteboardName, context.get());
205 return results;
206#else
207 return fileNames();
208#endif
209}

211bool DragData::containsPlainText() const
212{

215 platformStrategies()->pasteboardStrategy()->getTypes(types, m_pasteboardName, context.get());
216
217 return types.contains(stringPasteboardType())
218 || types.contains(rtfdPasteboardType())
219 || types.contains(rtfPasteboardType())
220#if PLATFORM(MAC)
221 || types.contains(String(legacyFilenamesPasteboardType()))
222#endif
223 || platformStrategies()->pasteboardStrategy()->containsStringSafeForDOMToReadForType(urlPasteboardType(), m_pasteboardName, context.get());
224}

226String DragData::asPlainText() const
227{
228 Pasteboard pasteboard(createPasteboardContext(), m_pasteboardName);
229 PasteboardPlainText text;
230 pasteboard.read(text);
231 String string = text.text;
232
233 // FIXME: It's not clear this is 100% correct since we know -[NSURL URLWithString:] does not handle
234 // all the same cases we handle well in the URL code for creating an NSURL.
235 if (text.isURL)
236 return WTF::userVisibleString([NSURL URLWithString:string]);
237
238 // FIXME: WTF should offer a non-Mac-specific way to convert string to precomposed form so we can do it for all platforms.
239 return [(NSString *)string precomposedStringWithCanonicalMapping];
240}

248bool DragData::containsCompatibleContent(DraggingPurpose purpose) const
249{
250 if (purpose == DraggingPurpose::ForFileUpload)
251 return containsFiles();
252
253 if (purpose == DraggingPurpose::ForColorControl)
254 return containsColor();
255
256 if (purpose == DraggingPurpose::ForEditing && RuntimeEnabledFeatures::sharedFeatures().attachmentElementEnabled() && containsFiles())
257 return true;
258
259 auto context = createPasteboardContext();
260 Vector<String> types;
261 platformStrategies()->pasteboardStrategy()->getTypes(types, m_pasteboardName, context.get());
262ALLOW_DEPRECATED_DECLARATIONS_BEGIN
263 return types.contains(String(WebArchivePboardType))
264 || types.contains(htmlPasteboardType())
265 || types.contains(String(kUTTypeWebArchive))
266#if PLATFORM(MAC)
267 || types.contains(String(legacyFilenamesPasteboardType()))
268 || types.contains(String(legacyFilesPromisePasteboardType()))
269#endif
270 || types.contains(tiffPasteboardType())
271 || types.contains(pdfPasteboardType())
272 || types.contains(urlPasteboardType())
273 || types.contains(rtfdPasteboardType())
274 || types.contains(rtfPasteboardType())
275 || types.contains(String(kUTTypeUTF8PlainText))
276 || types.contains(stringPasteboardType())
277 || types.contains(colorPasteboardType())
278 || types.contains(String(kUTTypeJPEG))
279 || types.contains(String(kUTTypePNG));
280ALLOW_DEPRECATED_DECLARATIONS_END
281}

283bool DragData::containsPromise() const
284{
285 auto context = createPasteboardContext();
286 // FIXME: legacyFilesPromisePasteboardType() contains UTIs, not path names. Also, why do we
287 // think promises should only contain one file (or UTI)?
288 Vector<String> files;
289#if PLATFORM(MAC)
290 platformStrategies()->pasteboardStrategy()->getPathnamesForType(files, String(legacyFilesPromisePasteboardType()), m_pasteboardName, context.get());
291#endif
292 return files.size() == 1;
293}

295bool DragData::containsURL(FilenameConversionPolicy) const
296{
297 auto context = createPasteboardContext();
298 if (platformStrategies()->pasteboardStrategy()->containsURLStringSuitableForLoading(m_pasteboardName, context.get()))
299 return true;
300
301#if PLATFORM(MAC)
302 Vector<String> types;
303 platformStrategies()->pasteboardStrategy()->getTypes(types, m_pasteboardName, context.get());
304 if (types.contains(String(legacyFilesPromisePasteboardType())) && fileNames().size() == 1)
305 return !![NSURL fileURLWithPath:fileNames().first()];
306#endif
307
308 return false;
309}

311String DragData::asURL(FilenameConversionPolicy, String* title) const
312{
313 auto context = createPasteboardContext();
314 // FIXME: Use filenamePolicy.
315
316 String urlTitle;
317 auto urlString = platformStrategies()->pasteboardStrategy()->urlStringSuitableForLoading(m_pasteboardName, urlTitle, context.get());
318 if (title)
319 *title = urlTitle;
320
321 if (!urlString.isEmpty())
322 return urlString;
323
324#if PLATFORM(MAC)
325 Vector<String> types;
326 platformStrategies()->pasteboardStrategy()->getTypes(types, m_pasteboardName, context.get());
327 if (types.contains(String(legacyFilesPromisePasteboardType())) && fileNames().size() == 1)
328 return [URLByCanonicalizingURL([NSURL fileURLWithPath:fileNames()[0]]) absoluteString];
329#endif
330
331 return { };
332}
