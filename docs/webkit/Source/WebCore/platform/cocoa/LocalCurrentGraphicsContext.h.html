31namespace WebCore {
32
33// This class automatically saves and restores the current NSGraphicsContext for
34// functions which call out into AppKit and rely on the currentContext being set
35class LocalCurrentGraphicsContext {
36 WTF_MAKE_NONCOPYABLE(LocalCurrentGraphicsContext);
37public:
38 WEBCORE_EXPORT LocalCurrentGraphicsContext(GraphicsContext&);
39 WEBCORE_EXPORT ~LocalCurrentGraphicsContext();
40 CGContextRef cgContext() { return m_savedGraphicsContext.platformContext(); }
41private:
42 GraphicsContext& m_savedGraphicsContext;
43#if USE(APPKIT)
44 RetainPtr<NSGraphicsContext> m_savedNSGraphicsContext;
45#endif
46 bool m_didSetGraphicsContext { false };
47};
48
49class ContextContainer {
50 WTF_MAKE_NONCOPYABLE(ContextContainer);
51public:
52 ContextContainer(GraphicsContext& graphicsContext)
53 : m_graphicsContext(graphicsContext.platformContext())
54 {
55 }
56
57 CGContextRef context() { return m_graphicsContext; }
58private:
59 CGContextRef m_graphicsContext;
60};
