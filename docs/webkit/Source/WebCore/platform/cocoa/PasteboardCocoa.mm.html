54enum class ImageType {
55 Invalid = 0,
56 TIFF,
57 PNG,
58 JPEG,
59 GIF,
60};

62static ImageType cocoaTypeToImageType(const String& cocoaType)
63{
64ALLOW_DEPRECATED_DECLARATIONS_BEGIN
65#if PLATFORM(MAC)
66 if (cocoaType == String(legacyTIFFPasteboardType()))
67 return ImageType::TIFF;
68#endif
69 if (cocoaType == String(kUTTypeTIFF))
70 return ImageType::TIFF;
71#if PLATFORM(MAC)
72 if (cocoaType == String(legacyPNGPasteboardType())) // NSPNGPboardType
73 return ImageType::PNG;
74#endif
75 if (cocoaType == String(kUTTypePNG))
76 return ImageType::PNG;
77 if (cocoaType == String(kUTTypeJPEG))
78 return ImageType::JPEG;
79 if (cocoaType == String(kUTTypeGIF))
80 return ImageType::GIF;
81ALLOW_DEPRECATED_DECLARATIONS_END
82 return ImageType::Invalid;
83}

85// String literals returned by this function must be defined exactly once
86// since read(PasteboardFileReader&) uses HashMap<const char*> to check uniqueness.
87static const char* imageTypeToMIMEType(ImageType type)
88{
89 switch (type) {
90 case ImageType::Invalid:
91 return nullptr;
92 case ImageType::TIFF:
93#if PLATFORM(MAC)
94 return "image/png"; // For Web compatibility, we pretend to have PNG instead.
95#else
96 return nullptr; // Don't support pasting TIFF on iOS for now.
97#endif
98 case ImageType::PNG:
99 return "image/png";
100 case ImageType::JPEG:
101 return "image/jpeg";
102 case ImageType::GIF:
103 return "image/gif";
104 }
105}

134Pasteboard::FileContentState Pasteboard::fileContentState()
135{
136 bool mayContainFilePaths = platformStrategies()->pasteboardStrategy()->getNumberOfFiles(m_pasteboardName, context());
137
138#if PLATFORM(IOS_FAMILY)
139 if (mayContainFilePaths) {
140 // On iOS, files are not written to the pasteboard using file URLs, so we need a heuristic to determine
141 // whether or not the pasteboard contains items that represent files. An example of when this gets tricky
142 // is differentiating between cases where the user is dragging a plain text file, versus selected text.
143 // Also, the presence of any other declared non-text data in the same item indicates that the content
144 // being dropped can take on another non-text format, which could be a file.
145 // If the item can't be treated as an attachment, it's very likely that the content being dropped is just
146 // an inline piece of text, with no files in the pasteboard (and therefore, no risk of leaking file paths
147 // to web content). In cases such as these, we should not suppress DataTransfer access.
148 auto items = allPasteboardItemInfo();
149 if (!items)
150 return FileContentState::NoFileOrImageData;
151
152 mayContainFilePaths = items->size() != 1 || notFound != items->findMatching([] (auto& item) {
153 return item.canBeTreatedAsAttachmentOrFile() || item.isNonTextType || item.containsFileURLAndFileUploadContent;
154 });
155 }
156#endif
157
158 if (!mayContainFilePaths) {
159 Vector<String> cocoaTypes;
160 platformStrategies()->pasteboardStrategy()->getTypes(cocoaTypes, m_pasteboardName, context());
161 if (cocoaTypes.findMatching([](const String& cocoaType) { return shouldTreatCocoaTypeAsFile(cocoaType); }) == notFound)
162 return FileContentState::NoFileOrImageData;
163
164 auto indexOfURL = cocoaTypes.findMatching([](auto& cocoaType) {
165#if PLATFORM(MAC)
166 if (cocoaType == String(legacyURLPasteboardType()))
167 return true;
168#endif
169ALLOW_DEPRECATED_DECLARATIONS_BEGIN
170 return cocoaType == String(kUTTypeURL);
171ALLOW_DEPRECATED_DECLARATIONS_END
172 });
173 mayContainFilePaths = indexOfURL != notFound && !platformStrategies()->pasteboardStrategy()->containsStringSafeForDOMToReadForType(cocoaTypes[indexOfURL], m_pasteboardName, context());
174 }
175
176 // Enforce changeCount ourselves for security. We check after reading instead of before to be
177 // sure it doesn't change between our testing the change count and accessing the data.
178 if (m_changeCount != platformStrategies()->pasteboardStrategy()->changeCount(m_pasteboardName, context()))
179 return FileContentState::NoFileOrImageData;
180
181 // Even when there's only image data in the pasteboard and no file representations, we still run the risk of exposing file paths
182 // to the page if the app has written image data to the pasteboard with a corresponding file path as plain text. An example of
183 // this is copying an image with a local `src` in Safari. To mitigate this, we additionally require that the app has not also
184 // written URLs to the pasteboard, as this would suggest that the plain text data might contain file paths.
185 return mayContainFilePaths ? FileContentState::MayContainFilePaths : FileContentState::InMemoryImage;
186}

188Vector<String> Pasteboard::typesSafeForBindings(const String& origin)
189{
190 Vector<String> types = platformStrategies()->pasteboardStrategy()->typesSafeForDOMToReadAndWrite(m_pasteboardName, origin, context());
191
192 // Enforce changeCount ourselves for security. We check after reading instead of before to be
193 // sure it doesn't change between our testing the change count and accessing the data.
194 if (m_changeCount != platformStrategies()->pasteboardStrategy()->changeCount(m_pasteboardName, context()))
195 return { };
196
197 return types;
198}

200Vector<String> Pasteboard::typesForLegacyUnsafeBindings()
201{
202 auto cocoaTypes = readTypesWithSecurityCheck();
203 if (cocoaTypes.isEmpty())
204 return cocoaTypes;
205
206 ListHashSet<String> result;
207 for (auto& cocoaType : cocoaTypes)
208 addHTMLClipboardTypesForCocoaType(result, cocoaType);
209
210 return copyToVector(result);
211}

215{
216 auto image = adoptNS([[NSBitmapImageRep alloc] initWithData: tiffBuffer.createNSData().get()]);
217 NSData *pngData = [image representationUsingType:bitmapPNGFileType() properties:@{ }];
218 return SharedBuffer::create(pngData);

222void Pasteboard::read(PasteboardFileReader& reader, std::optional<size_t> itemIndex)
223{
224 if (!itemIndex) {
225 auto filenames = readFilePaths();
226 if (!filenames.isEmpty()) {
227 for (auto& filename : filenames)
228 reader.readFilename(filename);
229 return;
230 }
231 }
232
233 auto readBufferAtIndex = [&](const PasteboardItemInfo& info, size_t itemIndex) {
234 for (auto cocoaType : info.platformTypesByFidelity) {
235 auto imageType = cocoaTypeToImageType(cocoaType);
236 auto* mimeType = imageTypeToMIMEType(imageType);
237 if (!mimeType || !reader.shouldReadBuffer(mimeType))
238 continue;
239 auto buffer = readBuffer(itemIndex, cocoaType);
240#if PLATFORM(MAC)
241 if (buffer && imageType == ImageType::TIFF)
242 buffer = convertTIFFToPNG(buffer.releaseNonNull());
243#endif
244 if (buffer) {
245 reader.readBuffer(imageTypeToFakeFilename(imageType), mimeType, buffer.releaseNonNull());
246 break;
247 }
248 }
249 };
250
251 if (itemIndex) {
252 if (auto info = pasteboardItemInfo(*itemIndex))
253 readBufferAtIndex(*info, *itemIndex);
254 return;
255 }
256
257 if (auto allInfo = allPasteboardItemInfo()) {
258 for (size_t itemIndex = 0; itemIndex < allInfo->size(); ++itemIndex)
259 readBufferAtIndex(allInfo->at(itemIndex), itemIndex);
260 }
261}

284const PasteboardCustomData& Pasteboard::readCustomData()
285{
286 if (m_customDataCache)
287 return *m_customDataCache;
288
289 if (auto buffer = readBufferForTypeWithSecurityCheck(PasteboardCustomData::cocoaType()))
290 m_customDataCache = PasteboardCustomData::fromSharedBuffer(*buffer);
291 else
292 m_customDataCache = PasteboardCustomData { };
293 return *m_customDataCache;
294}
