41std::optional<Vector<PasteboardItemInfo>> PlatformPasteboard::allPasteboardItemInfo(int64_t changeCount)
42{
43 if (changeCount != [m_pasteboard changeCount])
44 return std::nullopt;
45
46 Vector<PasteboardItemInfo> itemInfo;
47 int numberOfItems = count();
48 itemInfo.reserveInitialCapacity(numberOfItems);
49 for (NSInteger itemIndex = 0; itemIndex < numberOfItems; ++itemIndex) {
50 auto item = informationForItemAtIndex(itemIndex, changeCount);
51 if (!item)
52 return std::nullopt;
53
54 itemInfo.uncheckedAppend(WTFMove(*item));
55 }
56 return itemInfo;
57}

64String PlatformPasteboard::urlStringSuitableForLoading(String& title)
65{
66#if PLATFORM(MAC)
67 String URLTitleString = stringForType(String(WebURLNamePboardType));
68 if (!URLTitleString.isEmpty())
69 title = URLTitleString;
70#endif
71
72 Vector<String> types;
73 getTypes(types);
74
75#if PLATFORM(IOS_FAMILY)
76 UNUSED_PARAM(title);
77ALLOW_DEPRECATED_DECLARATIONS_BEGIN
78 String urlPasteboardType = kUTTypeURL;
79 String stringPasteboardType = kUTTypeText;
80ALLOW_DEPRECATED_DECLARATIONS_END
81#else
82 String urlPasteboardType = legacyURLPasteboardType();
83 String stringPasteboardType = legacyStringPasteboardType();
84#endif
85
86 if (types.contains(urlPasteboardType)) {
87 NSURL *URLFromPasteboard = [NSURL URLWithString:stringForType(urlPasteboardType)];
88 // Cannot drop other schemes unless <rdar://problem/10562662> and <rdar://problem/11187315> are fixed.
89 if (URL { URLFromPasteboard }.protocolIsInHTTPFamily())
90 return [URLByCanonicalizingURL(URLFromPasteboard) absoluteString];
91 }
92
93 if (types.contains(stringPasteboardType)) {
94 NSURL *URLFromPasteboard = [NSURL URLWithString:stringForType(stringPasteboardType)];
95 // Pasteboard content is not trusted, because JavaScript code can modify it. We can sanitize it for URLs and other typed content, but not for strings.
96 // The result of this function is used to initiate navigation, so we shouldn't allow arbitrary file URLs.
97 // FIXME: Should we allow only http family schemes, or anything non-local?
98 if (URL { URLFromPasteboard }.protocolIsInHTTPFamily())
99 return [URLByCanonicalizingURL(URLFromPasteboard) absoluteString];
100 }
101
102#if PLATFORM(MAC)
103 if (types.contains(String(legacyFilenamesPasteboardType()))) {
104 Vector<String> files;
105 getPathnamesForType(files, String(legacyFilenamesPasteboardType()));
106 if (files.size() == 1) {
107 BOOL isDirectory;
108 if ([[NSFileManager defaultManager] fileExistsAtPath:files[0] isDirectory:&isDirectory] && isDirectory)
109 return String();
110 return [URLByCanonicalizingURL([NSURL fileURLWithPath:files[0]]) absoluteString];
111 }
112 }
113#endif
114
115 return { };
116}
