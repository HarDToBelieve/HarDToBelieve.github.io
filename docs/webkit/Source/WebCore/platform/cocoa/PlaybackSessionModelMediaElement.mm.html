59void PlaybackSessionModelMediaElement::setMediaElement(HTMLMediaElement* mediaElement)
60{
61 if (m_mediaElement == mediaElement) {
62 if (m_mediaElement) {
63 for (auto client : m_clients)
64 client->isPictureInPictureSupportedChanged(isPictureInPictureSupported());
65 }
66 return;
67 }
68
69 auto& events = eventNames();
70
71 if (m_mediaElement && m_isListening) {
72 for (auto& eventName : observedEventNames())
73 m_mediaElement->removeEventListener(eventName, *this, false);
74 if (auto* audioTracks = m_mediaElement->audioTracks()) {
75 audioTracks->removeEventListener(events.addtrackEvent, *this, false);
76 audioTracks->removeEventListener(events.changeEvent, *this, false);
77 audioTracks->removeEventListener(events.removetrackEvent, *this, false);
78 }
79
80 if (auto* textTracks = m_mediaElement->audioTracks()) {
81 textTracks->removeEventListener(events.addtrackEvent, *this, false);
82 textTracks->removeEventListener(events.changeEvent, *this, false);
83 textTracks->removeEventListener(events.removetrackEvent, *this, false);
84 }
85 }
86 m_isListening = false;
87
88 if (m_mediaElement)
89 m_mediaElement->resetPlaybackSessionState();
90
91 m_mediaElement = mediaElement;
92
93 if (m_mediaElement) {
94 for (auto& eventName : observedEventNames())
95 m_mediaElement->addEventListener(eventName, *this, false);
96
97 auto& audioTracks = m_mediaElement->ensureAudioTracks();
98 audioTracks.addEventListener(events.addtrackEvent, *this, false);
99 audioTracks.addEventListener(events.changeEvent, *this, false);
100 audioTracks.addEventListener(events.removetrackEvent, *this, false);
101
102 auto& textTracks = m_mediaElement->ensureTextTracks();
103 textTracks.addEventListener(events.addtrackEvent, *this, false);
104 textTracks.addEventListener(events.changeEvent, *this, false);
105 textTracks.addEventListener(events.removetrackEvent, *this, false);
106 }
107
108 updateForEventName(eventNameAll());
109
110 for (auto client : m_clients)
111 client->isPictureInPictureSupportedChanged(isPictureInPictureSupported());
112}

126void PlaybackSessionModelMediaElement::updateForEventName(const WTF::AtomString& eventName)
127{
128 if (m_clients.isEmpty())
129 return;
130
131 bool all = eventName == eventNameAll();
132
133 if (all
134 || eventName == eventNames().durationchangeEvent) {
135 double duration = this->duration();
136 for (auto client : m_clients)
137 client->durationChanged(duration);
138 // These is no standard event for minFastReverseRateChange; duration change is a reasonable proxy for it.
139 // It happens every time a new item becomes ready to play.
140 bool canPlayFastReverse = this->canPlayFastReverse();
141 for (auto client : m_clients)
142 client->canPlayFastReverseChanged(canPlayFastReverse);
143 }
144
145 if (all
146 || eventName == eventNames().playEvent
147 || eventName == eventNames().playingEvent) {
148 for (auto client : m_clients)
149 client->playbackStartedTimeChanged(playbackStartedTime());
150 }
151
152 if (all
153 || eventName == eventNames().pauseEvent
154 || eventName == eventNames().playEvent
155 || eventName == eventNames().ratechangeEvent
156 || eventName == eventNames().waitingEvent
157 || eventName == eventNames().canplayEvent) {
158 OptionSet<PlaybackSessionModel::PlaybackState> playbackState;
159 if (isPlaying())
160 playbackState.add(PlaybackSessionModel::PlaybackState::Playing);
161 if (isStalled())
162 playbackState.add(PlaybackSessionModel::PlaybackState::Stalled);
163
164 double playbackRate = this->playbackRate();
165 double defaultPlaybackRate = this->defaultPlaybackRate();
166 for (auto client : m_clients)
167 client->rateChanged(playbackState, playbackRate, defaultPlaybackRate);
168 }
169
170 if (all
171 || eventName == eventNames().timeupdateEvent) {
172 auto currentTime = this->currentTime();
173 auto anchorTime = [[NSProcessInfo processInfo] systemUptime];
174 for (auto client : m_clients)
175 client->currentTimeChanged(currentTime, anchorTime);
176 }
177
178 if (all
179 || eventName == eventNames().progressEvent) {
180 auto bufferedTime = this->bufferedTime();
181 auto seekableRanges = this->seekableRanges();
182 auto seekableTimeRangesLastModifiedTime = this->seekableTimeRangesLastModifiedTime();
183 auto liveUpdateInterval = this->liveUpdateInterval();
184 for (auto client : m_clients) {
185 client->bufferedTimeChanged(bufferedTime);
186 client->seekableRangesChanged(seekableRanges, seekableTimeRangesLastModifiedTime, liveUpdateInterval);
187 }
188 }
189
190 if (all
191 || eventName == eventNames().addtrackEvent
192 || eventName == eventNames().removetrackEvent)
193 updateMediaSelectionOptions();
194
195 if (all
196 || eventName == eventNames().webkitcurrentplaybacktargetiswirelesschangedEvent) {
197 bool enabled = externalPlaybackEnabled();
198 ExternalPlaybackTargetType targetType = externalPlaybackTargetType();
199 String localizedDeviceName = externalPlaybackLocalizedDeviceName();
200
201 bool wirelessVideoPlaybackDisabled = this->wirelessVideoPlaybackDisabled();
202
203 for (auto client : m_clients) {
204 client->externalPlaybackChanged(enabled, targetType, localizedDeviceName);
205 client->wirelessVideoPlaybackDisabledChanged(wirelessVideoPlaybackDisabled);
206 }
207 }
208
209 if (all
210 || eventName == eventNames().webkitpresentationmodechangedEvent) {
211 bool isPictureInPictureActive = this->isPictureInPictureActive();
212
213 for (auto client : m_clients)
214 client->pictureInPictureActiveChanged(isPictureInPictureActive);
215 }
216
217
218 // We don't call updateMediaSelectionIndices() in the all case, since
219 // updateMediaSelectionOptions() will also update the selection indices.
220 if (eventName == eventNames().changeEvent)
221 updateMediaSelectionIndices();
222
223 if (all
224 || eventName == eventNames().volumechangeEvent) {
225 for (auto client : m_clients) {
226 client->mutedChanged(isMuted());
227 client->volumeChanged(volume());
228 }
229 }

382void PlaybackSessionModelMediaElement::updateMediaSelectionOptions()
383{
384 if (!m_mediaElement)
385 return;
386
387 if (!m_mediaElement->document().page())
388 return;
389
390 auto& captionPreferences = m_mediaElement->document().page()->group().captionPreferences();
391 auto* textTracks = m_mediaElement->textTracks();
392 if (textTracks && textTracks->length())
393 m_legibleTracksForMenu = captionPreferences.sortedTrackListForMenu(textTracks, { TextTrack::Kind::Subtitles, TextTrack::Kind::Captions, TextTrack::Kind::Descriptions });
394 else
395 m_legibleTracksForMenu.clear();
396
397 auto* audioTracks = m_mediaElement->audioTracks();
398 if (audioTracks && audioTracks->length() > 1)
399 m_audioTracksForMenu = captionPreferences.sortedTrackListForMenu(audioTracks);
400 else
401 m_audioTracksForMenu.clear();
402
403 auto audioOptions = audioMediaSelectionOptions();
404 auto audioIndex = audioMediaSelectedIndex();
405 auto legibleOptions = legibleMediaSelectionOptions();
406 auto legibleIndex = legibleMediaSelectedIndex();
407
408 for (auto client : m_clients) {
409 client->audioMediaSelectionOptionsChanged(audioOptions, audioIndex);
410 client->legibleMediaSelectionOptionsChanged(legibleOptions, legibleIndex);
411 }
412}

512Vector<MediaSelectionOption> PlaybackSessionModelMediaElement::audioMediaSelectionOptions() const
513{
514 Vector<MediaSelectionOption> audioOptions;
515
516 if (!m_mediaElement || !m_mediaElement->document().page())
517 return audioOptions;
518
519 auto& captionPreferences = m_mediaElement->document().page()->group().captionPreferences();
520
521 audioOptions.reserveInitialCapacity(m_audioTracksForMenu.size());
522 for (auto& audioTrack : m_audioTracksForMenu)
523 audioOptions.uncheckedAppend(captionPreferences.mediaSelectionOptionForTrack(audioTrack.get()));
524
525 return audioOptions;
526}

552uint64_t PlaybackSessionModelMediaElement::legibleMediaSelectedIndex() const
553{
554 auto host = m_mediaElement ? m_mediaElement->mediaControlsHost() : nullptr;
555 if (!host)
556 return std::numeric_limits<uint64_t>::max();
557
558 AtomString displayMode = host->captionDisplayMode();
559 TextTrack& offItem = TextTrack::captionMenuOffItem();
560 TextTrack& automaticItem = TextTrack::captionMenuAutomaticItem();
561
562 std::optional<uint64_t> selectedIndex;
563 std::optional<uint64_t> offIndex;
564
565 for (size_t index = 0; index < m_legibleTracksForMenu.size(); index++) {
566 auto& track = m_legibleTracksForMenu[index];
567
568 if (track == &offItem)
569 offIndex = index;
570
571 if (displayMode == MediaControlsHost::automaticKeyword()) {
572 if (track == &automaticItem)
573 selectedIndex = index;
574 } else {
575 if (track->mode() == TextTrack::Mode::Showing)
576 selectedIndex = index;
577 }
578 }
579
580 if (!selectedIndex && displayMode == MediaControlsHost::forcedOnlyKeyword())
581 selectedIndex = offIndex;
582
583 return selectedIndex.value_or(std::numeric_limits<uint64_t>::max());
584}

591PlaybackSessionModel::ExternalPlaybackTargetType PlaybackSessionModelMediaElement::externalPlaybackTargetType() const
592{
593 if (!m_mediaElement || !m_mediaElement->mediaControlsHost())
594 return TargetTypeNone;
595
596 switch (m_mediaElement->mediaControlsHost()->externalDeviceType()) {
597 default:
598 ASSERT_NOT_REACHED();
599 return TargetTypeNone;
600 case MediaControlsHost::DeviceType::None:
601 return TargetTypeNone;
602 case MediaControlsHost::DeviceType::Airplay:
603 return TargetTypeAirPlay;
604 case MediaControlsHost::DeviceType::Tvout:
605 return TargetTypeTVOut;
606 }
607}
