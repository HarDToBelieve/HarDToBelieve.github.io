34namespace WebCore {

36PowerSourceNotifier::PowerSourceNotifier(PowerSourceNotifierCallback&& callback)
37 : m_callback(WTFMove(callback))
38{
39 int token = 0;
40 auto status = notify_register_dispatch(kIOPSNotifyPowerSource, &token, dispatch_get_main_queue(), [weakThis = makeWeakPtr(*this)] (int) {
41 if (weakThis)
42 weakThis->notifyPowerSourceChanged();
43 });
44 if (status == NOTIFY_STATUS_OK)
45 m_tokenID = token;
46
47 // If the current value of systemHasAC() is uncached, force a notification.
48 if (!cachedSystemHasAC()) {
49 RunLoop::main().dispatch([weakThis = makeWeakPtr(*this)] {
50 if (weakThis)
51 weakThis->notifyPowerSourceChanged();
52 });
53 }
54}
55
56PowerSourceNotifier::~PowerSourceNotifier()
57{
58 if (m_tokenID)
59 notify_cancel(*m_tokenID);
60}
61
62void PowerSourceNotifier::notifyPowerSourceChanged()
63{
64 resetSystemHasAC();
65 if (m_callback)
66 m_callback(systemHasAC());
67}
