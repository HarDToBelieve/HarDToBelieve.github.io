38static std::optional<MRMediaRemoteCommand> mediaRemoteCommandForPlatformCommand(PlatformMediaSession::RemoteControlCommandType command)
39{
40 static const auto commandMap = makeNeverDestroyed([] {
41 using CommandToActionMap = HashMap<PlatformMediaSession::RemoteControlCommandType, MRMediaRemoteCommand, WTF::IntHash<PlatformMediaSession::RemoteControlCommandType>, WTF::StrongEnumHashTraits<PlatformMediaSession::RemoteControlCommandType>>;
42
43 return CommandToActionMap {
44 { PlatformMediaSession::PlayCommand, MRMediaRemoteCommandPlay },
45 { PlatformMediaSession::PauseCommand, MRMediaRemoteCommandPause },
46 { PlatformMediaSession::StopCommand, MRMediaRemoteCommandStop },
47 { PlatformMediaSession::TogglePlayPauseCommand, MRMediaRemoteCommandTogglePlayPause },
48 { PlatformMediaSession::BeginSeekingBackwardCommand, MRMediaRemoteCommandBeginRewind },
49 { PlatformMediaSession::EndSeekingBackwardCommand, MRMediaRemoteCommandEndRewind },
50 { PlatformMediaSession::BeginSeekingForwardCommand, MRMediaRemoteCommandBeginFastForward },
51 { PlatformMediaSession::EndSeekingForwardCommand, MRMediaRemoteCommandEndFastForward },
52 { PlatformMediaSession::SeekToPlaybackPositionCommand, MRMediaRemoteCommandSeekToPlaybackPosition },
53 { PlatformMediaSession::SkipForwardCommand, MRMediaRemoteCommandSkipForward },
54 { PlatformMediaSession::SkipBackwardCommand, MRMediaRemoteCommandSkipBackward },
55 { PlatformMediaSession::NextTrackCommand, MRMediaRemoteCommandNextTrack },
56 { PlatformMediaSession::PreviousTrackCommand, MRMediaRemoteCommandPreviousTrack },
57 };
58 }());
59
60 auto it = commandMap.get().find(command);
61 if (it != commandMap.get().end())
62 return { it->value };
63
64 return { };
65}

72const RemoteCommandListener::RemoteCommandsSet& RemoteCommandListenerCocoa::defaultCommands()
73{
74 static NeverDestroyed<RemoteCommandsSet> commands(std::initializer_list<PlatformMediaSession::RemoteControlCommandType> {
75 PlatformMediaSession::PlayCommand,
76 PlatformMediaSession::PauseCommand,
77 PlatformMediaSession::TogglePlayPauseCommand,
78 PlatformMediaSession::BeginSeekingForwardCommand,
79 PlatformMediaSession::EndSeekingForwardCommand,
80 PlatformMediaSession::BeginSeekingBackwardCommand,
81 PlatformMediaSession::EndSeekingBackwardCommand,
82 PlatformMediaSession::SeekToPlaybackPositionCommand,
83 PlatformMediaSession::SkipForwardCommand,
84 PlatformMediaSession::SkipBackwardCommand,
85 });
86
87 return commands;
88}

90static bool isSeekCommand(PlatformMediaSession::RemoteControlCommandType command)
91{
92 return command == PlatformMediaSession::SeekToPlaybackPositionCommand
93 || command == PlatformMediaSession::SkipForwardCommand
94 || command == PlatformMediaSession::SkipBackwardCommand
95 || command == PlatformMediaSession::BeginSeekingForwardCommand
96 || command == PlatformMediaSession::BeginSeekingBackwardCommand;
97}

99void RemoteCommandListenerCocoa::updateSupportedCommands()
100{
101 if (!isMediaRemoteFrameworkAvailable())
102 return;
103
104 auto& currentCommands = !supportedCommands().isEmpty() ? supportedCommands() : defaultCommands();
105 if (m_currentCommands == currentCommands)
106 return;
107
108 auto commandInfoArray = adoptCF(CFArrayCreateMutable(kCFAllocatorDefault, currentCommands.size(), &kCFTypeArrayCallBacks));
109 for (auto platformCommand : currentCommands) {
110 if (isSeekCommand(platformCommand) && !supportsSeeking())
111 continue;
112
113 auto command = mediaRemoteCommandForPlatformCommand(platformCommand);
114 ASSERT(command);
115 if (!command)
116 continue;
117
118 auto commandInfo = adoptCF(MRMediaRemoteCommandInfoCreate(kCFAllocatorDefault));
119 MRMediaRemoteCommandInfoSetCommand(commandInfo.get(), command.value());
120 MRMediaRemoteCommandInfoSetEnabled(commandInfo.get(), true);
121 if (platformCommand == PlatformMediaSession::SkipForwardCommand || platformCommand == PlatformMediaSession::SkipBackwardCommand)
122 MRMediaRemoteCommandInfoSetOptions(commandInfo.get(), (__bridge CFDictionaryRef)(@{(__bridge NSString *)kMRMediaRemoteCommandInfoPreferredIntervalsKey : @[@(15.0)]}));
123 CFArrayAppendValue(commandInfoArray.get(), commandInfo.get());
124 }
125
126 MRMediaRemoteSetSupportedCommands(commandInfoArray.get(), MRMediaRemoteGetLocalOrigin(), nullptr, nullptr);
127 m_currentCommands = currentCommands;
128}

130RemoteCommandListenerCocoa::RemoteCommandListenerCocoa(RemoteCommandListenerClient& client)
131 : RemoteCommandListener(client)
132{
133 if (!isMediaRemoteFrameworkAvailable())
134 return;
135
136 scheduleSupportedCommandsUpdate();
137
138 auto weakThis = makeWeakPtr(*this);
139 m_commandHandler = MRMediaRemoteAddAsyncCommandHandlerBlock(^(MRMediaRemoteCommand command, CFDictionaryRef options, void(^completion)(CFArrayRef)) {
140
141 LOG(Media, "RemoteCommandListenerCocoa::RemoteCommandListenerCocoa - received command %u", command);
142
143 PlatformMediaSession::RemoteControlCommandType platformCommand { PlatformMediaSession::NoCommand };
144 PlatformMediaSession::RemoteCommandArgument argument;
145 MRMediaRemoteCommandHandlerStatus status = MRMediaRemoteCommandHandlerStatusSuccess;
146
147 switch (command) {
148 case MRMediaRemoteCommandPlay:
149 platformCommand = PlatformMediaSession::PlayCommand;
150 break;
151 case MRMediaRemoteCommandPause:
152 platformCommand = PlatformMediaSession::PauseCommand;
153 break;
154 case MRMediaRemoteCommandStop:
155 platformCommand = PlatformMediaSession::StopCommand;
156 break;
157 case MRMediaRemoteCommandTogglePlayPause:
158 platformCommand = PlatformMediaSession::TogglePlayPauseCommand;
159 break;
160 case MRMediaRemoteCommandBeginFastForward:
161 platformCommand = PlatformMediaSession::BeginSeekingForwardCommand;
162 break;
163 case MRMediaRemoteCommandEndFastForward:
164 platformCommand = PlatformMediaSession::EndSeekingForwardCommand;
165 break;
166 case MRMediaRemoteCommandBeginRewind:
167 platformCommand = PlatformMediaSession::BeginSeekingBackwardCommand;
168 break;
169 case MRMediaRemoteCommandEndRewind:
170 platformCommand = PlatformMediaSession::EndSeekingBackwardCommand;
171 break;
172 case MRMediaRemoteCommandSeekToPlaybackPosition: {
173 if (!supportsSeeking()) {
174 status = MRMediaRemoteCommandHandlerStatusCommandFailed;
175 break;
176 }
177
178 CFNumberRef positionRef = static_cast<CFNumberRef>(CFDictionaryGetValue(options, kMRMediaRemoteOptionPlaybackPosition));
179 if (!positionRef) {
180 status = MRMediaRemoteCommandHandlerStatusCommandFailed;
181 break;
182 }
183
184 double position = 0;
185 CFNumberGetValue(positionRef, kCFNumberDoubleType, &position);
186 argument.time = position;
187 platformCommand = PlatformMediaSession::SeekToPlaybackPositionCommand;
188 break;
189 }
190 case MRMediaRemoteCommandSkipForward:
191 case MRMediaRemoteCommandSkipBackward:
192 if (!supportsSeeking()) {
193 status = MRMediaRemoteCommandHandlerStatusCommandFailed;
194 break;
195 }
196
197 if (auto positionRef = static_cast<CFNumberRef>(CFDictionaryGetValue(options, kMRMediaRemoteOptionSkipInterval))) {
198 double position = 0;
199 CFNumberGetValue(positionRef, kCFNumberDoubleType, &position);
200 argument.time = position;
201 }
202
203 platformCommand = (command == MRMediaRemoteCommandSkipForward) ? PlatformMediaSession::SkipForwardCommand : PlatformMediaSession::SkipBackwardCommand;
204 break;
205 case MRMediaRemoteCommandNextTrack:
206 platformCommand = PlatformMediaSession::NextTrackCommand;
207 break;
208 case MRMediaRemoteCommandPreviousTrack:
209 platformCommand = PlatformMediaSession::PreviousTrackCommand;
210 break;
211 default:
212 LOG(Media, "RemoteCommandListenerCocoa::RemoteCommandListenerCocoa - command %u not supported!", command);
213 status = MRMediaRemoteCommandHandlerStatusCommandFailed;
214 };
215
216 if (weakThis && status != MRMediaRemoteCommandHandlerStatusCommandFailed)
217 weakThis->client().didReceiveRemoteControlCommand(platformCommand, argument);
218
219 completion((__bridge CFArrayRef)@[@(status)]);
220 });
221}

223RemoteCommandListenerCocoa::~RemoteCommandListenerCocoa()
224{
225 if (m_commandHandler)
226 MRMediaRemoteRemoveCommandHandlerBlock(m_commandHandler);
227}
