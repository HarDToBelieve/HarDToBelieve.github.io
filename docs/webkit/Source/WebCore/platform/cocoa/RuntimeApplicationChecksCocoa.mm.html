43// The application bundle identifier gets set to the UIProcess bundle identifier by the WebProcess and
44// the Networking upon initialization. It is unset otherwise.
45static String& bundleIdentifierOverride()
46{
47 static NeverDestroyed<String> identifierOverride;
48#if !ASSERT_MSG_DISABLED
49 applicationBundleIdentifierOverrideWasQueried = true;
50#endif
51 return identifierOverride;
52}
53
54static String& bundleIdentifier()
55{
56 static NeverDestroyed<String> identifier;
57 return identifier;
58}
59
60String applicationBundleIdentifier()
61{
62 // The override only gets set in WebKitTestRunner. If unset, we use the bundle identifier set
63 // in WebKit2's WebProcess and NetworkProcess. If that is also unset, we use the main bundle identifier.
64 auto identifier = bundleIdentifierOverride();
65 ASSERT(identifier.isNull() || RunLoop::isMain());
66 if (identifier.isNull())
67 identifier = bundleIdentifier();
68 ASSERT_WITH_MESSAGE(!isInAuxiliaryProcess() || !!identifier, "applicationBundleIsEqualTo() and applicationBundleStartsWith() must not be called before setApplicationBundleIdentifier() in auxiliary processes");
69 return identifier.isNull() ? String([[NSBundle mainBundle] bundleIdentifier]) : identifier;
70}
71
72void setApplicationBundleIdentifier(const String& identifier)
73{
74 ASSERT(RunLoop::isMain());
75 bundleIdentifier() = identifier;
76}
77
78void setApplicationBundleIdentifierOverride(const String& identifier)
79{
80 ASSERT(RunLoop::isMain());
81 ASSERT_WITH_MESSAGE(!applicationBundleIdentifierOverrideWasQueried, "applicationBundleIsEqualTo() and applicationBundleStartsWith() must not be called before setApplicationBundleIdentifierOverride()");
82 bundleIdentifierOverride() = identifier;
83}
84
85void clearApplicationBundleIdentifierTestingOverride()
86{
87 ASSERT(RunLoop::isMain());
88 bundleIdentifierOverride() = String();
89#if !ASSERT_MSG_DISABLED
90 applicationBundleIdentifierOverrideWasQueried = false;
91#endif
92}
93
94static bool applicationBundleIsEqualTo(const String& bundleIdentifierString)
95{
96 return applicationBundleIdentifier() == bundleIdentifierString;
97}
