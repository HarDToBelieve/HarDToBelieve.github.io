36@interface WebCoreSharedBufferData : NSData
37- (instancetype)initWithDataSegment:(const WebCore::SharedBuffer::DataSegment&)dataSegment position:(NSUInteger)position;
38@end
39
40@implementation WebCoreSharedBufferData {
41 RefPtr<const WebCore::SharedBuffer::DataSegment> _dataSegment;
42 NSUInteger _position;
43}
44
45+ (void)initialize
46{
47#if !USE(WEB_THREAD)
48 JSC::initialize();
49 WTF::initializeMainThread();
50 WebCore::populateJITOperations();
51#endif // !USE(WEB_THREAD)
52}
53
54- (void)dealloc
55{
56 if (WebCoreObjCScheduleDeallocateOnMainThread([WebCoreSharedBufferData class], self))
57 return;
58
59 [super dealloc];
60}
61
62- (instancetype)initWithDataSegment:(const WebCore::SharedBuffer::DataSegment&)dataSegment position:(NSUInteger)position
63{
64 if (!(self = [super init]))
65 return nil;
66
67 RELEASE_ASSERT(!position || position < dataSegment.size());
68 _dataSegment = &dataSegment;
69 _position = position;
70 return self;
71}
72
73- (NSUInteger)length
74{
75 return _dataSegment->size() - _position;
76}
77
78- (const void *)bytes
79{
80 return _dataSegment->data() + _position;
81}
82
83@end

87Ref<SharedBuffer> SharedBuffer::create(NSData *nsData)
88{
89 return adoptRef(*new SharedBuffer((__bridge CFDataRef)nsData));
90}
91
92void SharedBuffer::append(NSData *nsData)
93{
94 return append((__bridge CFDataRef)nsData);
95}
96
97RetainPtr<NSData> SharedBuffer::createNSData() const
98{
99 return adoptNS((NSData *)createCFData().leakRef());
100}
101
102RetainPtr<CFDataRef> SharedBuffer::createCFData() const
103{
104 combineIntoOneSegment();
105 if (!m_segments.size())
106 return adoptCF(CFDataCreate(nullptr, nullptr, 0));
107 ASSERT(m_segments.size() == 1);
108 return adoptCF((__bridge CFDataRef)m_segments[0].segment->createNSData().leakRef());
109}
110
111RefPtr<SharedBuffer> SharedBuffer::createFromReadingFile(const String& filePath)
112{
113 NSData *resourceData = [NSData dataWithContentsOfFile:filePath];
114 if (resourceData)
115 return SharedBuffer::create(resourceData);
116 return nullptr;
117}
118
119RetainPtr<NSArray> SharedBuffer::createNSDataArray() const
120{
121 return createNSArray(m_segments, [] (auto& segment) {
122 return segment.segment->createNSData();
123 });
124}
125
126RetainPtr<NSData> SharedBuffer::DataSegment::createNSData() const
127{
128 return adoptNS([[WebCoreSharedBufferData alloc] initWithDataSegment:*this position:0]);
129}
130
131RetainPtr<NSData> SharedBufferDataView::createNSData() const
132{
133 return adoptNS([[WebCoreSharedBufferData alloc] initWithDataSegment:m_segment.get() position:m_positionWithinSegment]);
134}
