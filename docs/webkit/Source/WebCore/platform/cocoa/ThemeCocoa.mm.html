35static void fitContextToBox(GraphicsContext& context, const FloatSize& srcImageSize, const FloatSize& dstSize)
36{
37 float srcRatio = srcImageSize.aspectRatio();
38 float dstRatio = dstSize.aspectRatio();
39
40 float scale;
41 float translationX = 0;
42 float translationY = 0;
43 if (srcRatio > dstRatio) {
44 scale = dstSize.width() / srcImageSize.width();
45 translationY = (dstSize.height() - scale * srcImageSize.height()) / 2;
46 } else {
47 scale = dstSize.height() / srcImageSize.height();
48 translationX = (dstSize.width() - scale * srcImageSize.width()) / 2;
49 }
50 context.translate(translationX, translationY);
51 context.scale(scale);
52}

56static NSBundle *passKitBundle()
57{
58 static NSBundle *passKitBundle;
59 static dispatch_once_t onceToken;
60 dispatch_once(&onceToken, ^{
61#if PLATFORM(MAC)
62 passKitBundle = [NSBundle bundleWithURL:[NSURL fileURLWithPath:@"/System/Library/PrivateFrameworks/PassKit.framework" isDirectory:YES]];
63#else
64 dlopen("/System/Library/Frameworks/PassKit.framework/PassKit", RTLD_NOW);
65 passKitBundle = [NSBundle bundleForClass:NSClassFromString(@"PKPaymentAuthorizationViewController")];
66#endif
67 });
68
69 return passKitBundle;
70}

72static RetainPtr<CGPDFPageRef> loadPassKitPDFPage(NSString *imageName)
73{
74 NSURL *url = [passKitBundle() URLForResource:imageName withExtension:@"pdf"];
75 if (!url)
76 return nullptr;
77
78 auto document = adoptCF(CGPDFDocumentCreateWithURL((CFURLRef)url));
79 if (!document)
80 return nullptr;
81
82 if (!CGPDFDocumentGetNumberOfPages(document.get()))
83 return nullptr;
84
85 return CGPDFDocumentGetPage(document.get(), 1);
86};

110static void drawApplePayButton(GraphicsContext& context, CGPDFPageRef page, const FloatRect& rect)
111{
112 CGSize pdfSize = CGPDFPageGetBoxRect(page, kCGPDFMediaBox).size;
113 GraphicsContextStateSaver stateSaver(context);
114 fitContextToBox(context, FloatSize(pdfSize), rect.size());
115
116 CGContextTranslateCTM(context.platformContext(), 0, pdfSize.height);
117 CGContextScaleCTM(context.platformContext(), 1, -1);
118
119 CGContextDrawPDFPage(context.platformContext(), page);
120};

124void ThemeCocoa::drawNamedImage(const String& name, GraphicsContext& context, const FloatRect& rect) const
125{
126 if (name == "wireless-playback") {
127 GraphicsContextStateSaver stateSaver(context);
128 context.setFillColor(Color::black);
129
130 FloatSize wirelessPlaybackSrcSize(32, 24.016);
131 fitContextToBox(context, wirelessPlaybackSrcSize, rect.size());
132
133 Path outline;
134 outline.moveTo(FloatPoint(24.066, 18));
135 outline.addLineTo(FloatPoint(22.111, 16));
136 outline.addLineTo(FloatPoint(30, 16));
137 outline.addLineTo(FloatPoint(30, 2));
138 outline.addLineTo(FloatPoint(2, 2));
139 outline.addLineTo(FloatPoint(2, 16));
140 outline.addLineTo(FloatPoint(9.908, 16));
141 outline.addLineTo(FloatPoint(7.953, 18));
142 outline.addLineTo(FloatPoint(0, 18));
143 outline.addLineTo(FloatPoint(0, 0));
144 outline.addLineTo(FloatPoint(32, 0));
145 outline.addLineTo(FloatPoint(32, 18));
146 outline.addLineTo(FloatPoint(24.066, 18));
147 outline.closeSubpath();
148 outline.moveTo(FloatPoint(26.917, 24.016));
149 outline.addLineTo(FloatPoint(5.040, 24.016));
150 outline.addLineTo(FloatPoint(15.978, 12.828));
151 outline.addLineTo(FloatPoint(26.917, 24.016));
152 outline.closeSubpath();
153
154 context.fillPath(outline);
155 return;
156 }
157
158#if ENABLE(APPLE_PAY)
159 if (name == "apple-pay-logo-black") {
160 if (auto logo = applePayButtonLogoBlack()) {
161 drawApplePayButton(context, logo, rect);
162 return;
163 }
164 }
165
166 if (name == "apple-pay-logo-white") {
167 if (auto logo = applePayButtonLogoWhite()) {
168 drawApplePayButton(context, logo, rect);
169 return;
170 }
171 }
172#endif
173
174 Theme::drawNamedImage(name, context, rect);
175}
