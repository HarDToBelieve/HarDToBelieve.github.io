29#pragma once
30
31#if ENABLE(ENCRYPTED_MEDIA)
32
33#include "CDMEncryptionScheme.h"
34#include <wtf/text/WTFString.h>
35
36namespace WebCore {
37
38struct CDMMediaCapability {
39 String contentType;
40 String robustness;
41 std::optional<CDMEncryptionScheme> encryptionScheme;
42
43 template<class Encoder>
44 void encode(Encoder& encoder) const
45 {
46 encoder << contentType;
47 encoder << robustness;
48 encoder << encryptionScheme;
49 }
50
51 template <class Decoder>
52 static std::optional<CDMMediaCapability> decode(Decoder& decoder)
53 {
54 std::optional<String> contentType;
55 decoder >> contentType;
56 if (!contentType)
57 return std::nullopt;
58
59 std::optional<String> robustness;
60 decoder >> robustness;
61 if (!robustness)
62 return std::nullopt;
63
64 std::optional<std::optional<CDMEncryptionScheme>> encryptionScheme;
65 decoder >> encryptionScheme;
66 if (!encryptionScheme)
67 return std::nullopt;
68
69 return {{
70 WTFMove(*contentType),
71 WTFMove(*robustness),
72 WTFMove(*encryptionScheme),
73 }};
74 }
75};
76
77} // namespace WebCore
78
79#endif // ENABLE(ENCRYPTED_MEDIA)
