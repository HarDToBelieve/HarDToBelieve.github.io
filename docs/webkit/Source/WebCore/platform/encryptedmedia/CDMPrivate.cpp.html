47void CDMPrivate::getSupportedConfiguration(CDMKeySystemConfiguration&& candidateConfiguration, LocalStorageAccess access, SupportedConfigurationCallback&& callback)
48{
49 // https://w3c.github.io/encrypted-media/#get-supported-configuration
50 // W3C Editor's Draft 09 November 2016
51
52 // 3.1.1.1 Get Supported Configuration
53 // Given a Key Systems implementation implementation, CDMKeySystemConfiguration candidate configuration, and origin,
54 // this algorithm returns a supported configuration or NotSupported as appropriate.
55
56 // 1. Let supported configuration be ConsentDenied.
57 // 2. Initialize restrictions to indicate that no configurations have had user consent denied.
58 CDMRestrictions restrictions { };
59 doSupportedConfigurationStep(WTFMove(candidateConfiguration), WTFMove(restrictions), access, WTFMove(callback));
60}

62void CDMPrivate::doSupportedConfigurationStep(CDMKeySystemConfiguration&& candidateConfiguration, CDMRestrictions&& restrictions, LocalStorageAccess access, SupportedConfigurationCallback&& callback)
63{
64 // https://w3c.github.io/encrypted-media/#get-supported-configuration
65 // W3C Editor's Draft 09 November 2016, ctd.
66
67 // 3.1.1.1 Get Supported Configuration
68 // 3. Repeat the following step while supported configuration is ConsentDenied:
69 // 3.1. Let supported configuration and, if provided, restrictions be the result of executing the
70 // Get Supported Configuration and Consent algorithm with implementation, candidate configuration,
71 // restrictions and origin.
72 auto optionalConfiguration = getSupportedConfiguration(candidateConfiguration, restrictions, access);
73 if (!optionalConfiguration) {
74 callback(std::nullopt);
75 return;
76 }
77
78 auto consentCallback = [weakThis = makeWeakPtr(*this), callback = WTFMove(callback), access] (ConsentStatus status, CDMKeySystemConfiguration&& configuration, CDMRestrictions&& restrictions) mutable {
79 if (!weakThis) {
80 callback(std::nullopt);
81 return;
82 }
83 // 3.1.1.2 Get Supported Configuration and Consent, ctd.
84 // 22. Let consent status and updated restrictions be the result of running the Get Consent Status algorithm on accumulated configuration,
85 // restrictions and origin and follow the steps for the value of consent status from the following list:
86 switch (status) {
87 case ConsentStatus::ConsentDenied:
88 // ↳ ConsentDenied:
89 // Return ConsentDenied and updated restrictions.
90 weakThis->doSupportedConfigurationStep(WTFMove(configuration), WTFMove(restrictions), access, WTFMove(callback));
91 return;
92
93 case ConsentStatus::InformUser:
94 // ↳ InformUser
95 // Inform the user that accumulated configuration is in use in the origin including, specifically, the information that
96 // Distinctive Identifier(s) and/or Distinctive Permanent Identifier(s) as appropriate will be used if the
97 // distinctiveIdentifier member of accumulated configuration is "required". Continue to the next step.
98 // NOTE: Implement.
99 break;
100
101 case ConsentStatus::Allowed:
102 // ↳ Allowed:
103 // Continue to the next step.
104 break;
105 }
106 // 23. Return accumulated configuration.
107 callback(WTFMove(configuration));
108 };
109 getConsentStatus(WTFMove(optionalConfiguration.value()), WTFMove(restrictions), access, WTFMove(consentCallback));
110}

118 // 1. Let the session type be the specified CDMSessionType value.
119 // 2. Follow the steps for the value of session type from the following list:
120 switch (sessionType) {
121 case CDMSessionType::Temporary:
122 // ↳ "temporary"
123 return false;
124 case CDMSessionType::PersistentLicense:
125 case CDMSessionType::PersistentUsageRecord:
126 // ↳ "persistent-license"

128 }
129
130 ASSERT_NOT_REACHED();
131 return false;

134std::optional<CDMKeySystemConfiguration> CDMPrivate::getSupportedConfiguration(const CDMKeySystemConfiguration& candidateConfiguration, CDMRestrictions& restrictions, LocalStorageAccess access)
135{
136 // https://w3c.github.io/encrypted-media/#get-supported-configuration-and-consent
137 // W3C Editor's Draft 09 November 2016
138
139 // 3.1.1.2 Get Supported Configuration and Consent
140 // Given a Key Systems implementation implementation, CDMKeySystemConfiguration candidate configuration,
141 // restrictions and origin, this algorithm returns a supported configuration, NotSupported, or ConsentDenied
142 // as appropriate and, in the ConsentDenied case, restrictions.
143
144 // 1. Let accumulated configuration be a new CDMKeySystemConfiguration dictionary.
145 CDMKeySystemConfiguration accumulatedConfiguration { };
146
147 // 2. Set the label member of accumulated configuration to equal the label member of candidate configuration.
148 accumulatedConfiguration.label = candidateConfiguration.label;
149
150 // 3. If the initDataTypes member of candidate configuration is non-empty, run the following steps:
151 if (!candidateConfiguration.initDataTypes.isEmpty()) {
152 // 3.1. Let supported types be an empty sequence of DOMStrings.
153 Vector<String> supportedTypes;
154
155 // 3.2. For each value in candidate configuration's initDataTypes member:
156 for (auto initDataType : candidateConfiguration.initDataTypes) {
157 // 3.2.1. Let initDataType be the value.
158 // 3.2.2. If the implementation supports generating requests based on initDataType, add initDataType
159 // to supported types. String comparison is case-sensitive. The empty string is never supported.
160 if (initDataType.isEmpty())
161 continue;
162
163 if (supportedInitDataTypes().contains(initDataType))
164 supportedTypes.append(initDataType);
165 }
166
167 // 3.3. If supported types is empty, return NotSupported.
168 if (supportedTypes.isEmpty())
169 return std::nullopt;
170
171 // 3.4. Set the initDataTypes member of accumulated configuration to supported types.
172 accumulatedConfiguration.initDataTypes = WTFMove(supportedTypes);
173 }
174
175 // 4. Let distinctive identifier requirement be the value of candidate configuration's distinctiveIdentifier member.
176 CDMRequirement distinctiveIdentifierRequirement = candidateConfiguration.distinctiveIdentifier;
177
178 // 5. If distinctive identifier requirement is "optional" and Distinctive Identifiers are not allowed according to
179 // restrictions, set distinctive identifier requirement to "not-allowed".
180 if (distinctiveIdentifierRequirement == CDMRequirement::Optional && restrictions.distinctiveIdentifierDenied)
181 distinctiveIdentifierRequirement = CDMRequirement::NotAllowed;
182
183 // 6. Follow the steps for distinctive identifier requirement from the following list:
184 switch (distinctiveIdentifierRequirement) {
185 case CDMRequirement::Required:
186 // ↳ "required"
187 // If the implementation does not support use of Distinctive Identifier(s) in combination
188 // with accumulated configuration and restrictions, return NotSupported.
189 if (distinctiveIdentifiersRequirement(accumulatedConfiguration, restrictions) == CDMRequirement::NotAllowed)
190 return std::nullopt;
191 break;
192
193 case CDMRequirement::Optional:
194 // ↳ "optional"
195 // Continue with the following steps.
196 break;
197
198 case CDMRequirement::NotAllowed:
199 // ↳ "not-allowed"
200 // If the implementation requires use Distinctive Identifier(s) or Distinctive Permanent Identifier(s)
201 // in combination with accumulated configuration and restrictions, return NotSupported.
202 if (distinctiveIdentifiersRequirement(accumulatedConfiguration, restrictions) == CDMRequirement::Required)
203 return std::nullopt;
204 break;
205 }
206
207 // 7. Set the distinctiveIdentifier member of accumulated configuration to equal distinctive identifier requirement.
208 accumulatedConfiguration.distinctiveIdentifier = distinctiveIdentifierRequirement;
209
210 // 8. Let persistent state requirement be equal to the value of candidate configuration's persistentState member.
211 CDMRequirement persistentStateRequirement = candidateConfiguration.persistentState;
212
213 // 9. If persistent state requirement is "optional" and persisting state is not allowed according to restrictions,
214 // set persistent state requirement to "not-allowed".
215 if (persistentStateRequirement == CDMRequirement::Optional && restrictions.persistentStateDenied)
216 persistentStateRequirement = CDMRequirement::NotAllowed;
217
218 // 10. Follow the steps for persistent state requirement from the following list:
219 switch (persistentStateRequirement) {
220 case CDMRequirement::Required:
221 // ↳ "required"
222 // If the implementation does not support persisting state in combination with accumulated configuration
223 // and restrictions, return NotSupported.
224 if (this->persistentStateRequirement(accumulatedConfiguration, restrictions) == CDMRequirement::NotAllowed)
225 return std::nullopt;
226 break;
227
228 case CDMRequirement::Optional:
229 // ↳ "optional"
230 // Continue with the following steps.
231 break;
232
233 case CDMRequirement::NotAllowed:
234 // ↳ "not-allowed"
235 // If the implementation requires persisting state in combination with accumulated configuration
236 // and restrictions, return NotSupported
237 if (this->persistentStateRequirement(accumulatedConfiguration, restrictions) == CDMRequirement::Required)
238 return std::nullopt;
239 break;
240 }
241
242 // 11. Set the persistentState member of accumulated configuration to equal the value of persistent state requirement.
243 accumulatedConfiguration.persistentState = persistentStateRequirement;
244
245 // 12. Follow the steps for the first matching condition from the following list:
246 Vector<CDMSessionType> sessionTypes;
247
248 if (!candidateConfiguration.sessionTypes.isEmpty()) {
249 // ↳ If the sessionTypes member is present [WebIDL] in candidate configuration
250 // Let session types be candidate configuration's sessionTypes member.
251 sessionTypes = candidateConfiguration.sessionTypes;
252 } else {
253 // ↳ Otherwise
254 // Let session types be [ "temporary" ].
255 sessionTypes = { CDMSessionType::Temporary };
256 }
257
258 // 13. For each value in session types:
259 for (auto& sessionType : sessionTypes) {
260 // 13.1. Let session type be the value.
261 // 13.2. If accumulated configuration's persistentState value is "not-allowed" and the
262 // Is persistent session type? algorithm returns true for session type return NotSupported.
263 if (accumulatedConfiguration.persistentState == CDMRequirement::NotAllowed && isPersistentType(sessionType))
264 return std::nullopt;
265
266 // 13.3. If the implementation does not support session type in combination with accumulated configuration
267 // and restrictions for other reasons, return NotSupported.
268 if (!supportsSessionTypeWithConfiguration(sessionType, accumulatedConfiguration))
269 return std::nullopt;
270
271 // 13.4 If accumulated configuration's persistentState value is "optional" and the result of running the Is
272 // persistent session type? algorithm on session type is true, change accumulated configuration's persistentState
273 // value to "required".
274 if (accumulatedConfiguration.persistentState == CDMRequirement::Optional && isPersistentType(sessionType))
275 accumulatedConfiguration.persistentState = CDMRequirement::Required;
276 }
277
278 // 14. Set the sessionTypes member of accumulated configuration to session types.
279 accumulatedConfiguration.sessionTypes = sessionTypes;
280
281 // 15. If the videoCapabilities and audioCapabilities members in candidate configuration are both empty, return NotSupported.
282 if (candidateConfiguration.videoCapabilities.isEmpty() && candidateConfiguration.audioCapabilities.isEmpty())
283 return std::nullopt;
284
285 // 16. ↳ If the videoCapabilities member in candidate configuration is non-empty:
286 if (!candidateConfiguration.videoCapabilities.isEmpty()) {
287 // 16.1. Let video capabilities be the result of executing the Get Supported Capabilities for Audio/Video Type algorithm on
288 // Video, candidate configuration's videoCapabilities member, accumulated configuration, and restrictions.
289 auto videoCapabilities = getSupportedCapabilitiesForAudioVideoType(AudioVideoType::Video, candidateConfiguration.videoCapabilities, accumulatedConfiguration, restrictions);
290
291 // 16.2. If video capabilities is null, return NotSupported.
292 if (!videoCapabilities)
293 return std::nullopt;
294
295 // 16.3 Set the videoCapabilities member of accumulated configuration to video capabilities.
296 accumulatedConfiguration.videoCapabilities = WTFMove(videoCapabilities.value());
297 } else {
298 // 16. ↳ Otherwise:
299 // Set the videoCapabilities member of accumulated configuration to an empty sequence.
300 accumulatedConfiguration.videoCapabilities = { };
301 }
302
303 // 17. ↳ If the audioCapabilities member in candidate configuration is non-empty:
304 if (!candidateConfiguration.audioCapabilities.isEmpty()) {
305 // 17.1. Let audio capabilities be the result of executing the Get Supported Capabilities for Audio/Video Type algorithm on
306 // Audio, candidate configuration's audioCapabilities member, accumulated configuration, and restrictions.
307 auto audioCapabilities = getSupportedCapabilitiesForAudioVideoType(AudioVideoType::Audio, candidateConfiguration.audioCapabilities, accumulatedConfiguration, restrictions);
308
309 // 17.2. If audio capabilities is null, return NotSupported.
310 if (!audioCapabilities)
311 return std::nullopt;
312
313 // 17.3 Set the audioCapabilities member of accumulated configuration to audio capabilities.
314 accumulatedConfiguration.audioCapabilities = WTFMove(audioCapabilities.value());
315 } else {
316 // 17. ↳ Otherwise:
317 // Set the audioCapabilities member of accumulated configuration to an empty sequence.
318 accumulatedConfiguration.audioCapabilities = { };
319 }
320
321 // 18. If accumulated configuration's distinctiveIdentifier value is "optional", follow the steps for the first matching
322 // condition from the following list:
323 if (accumulatedConfiguration.distinctiveIdentifier == CDMRequirement::Optional) {
324 // ↳ If the implementation requires use Distinctive Identifier(s) or Distinctive Permanent Identifier(s) for any of the
325 // combinations in accumulated configuration
326 if (distinctiveIdentifiersRequirement(accumulatedConfiguration, restrictions) == CDMRequirement::Required) {
327 // Change accumulated configuration's distinctiveIdentifier value to "required".
328 accumulatedConfiguration.distinctiveIdentifier = CDMRequirement::Required;
329 } else {
330 // ↳ Otherwise
331 // Change accumulated configuration's distinctiveIdentifier value to "not-allowed".
332 accumulatedConfiguration.distinctiveIdentifier = CDMRequirement::NotAllowed;
333 }
334 }
335
336 // 19. If accumulated configuration's persistentState value is "optional", follow the steps for the first matching
337 // condition from the following list:
338 if (accumulatedConfiguration.persistentState == CDMRequirement::Optional) {
339 // ↳ If the implementation requires persisting state for any of the combinations in accumulated configuration
340 if (this->persistentStateRequirement(accumulatedConfiguration, restrictions) == CDMRequirement::Required) {
341 // Change accumulated configuration's persistentState value to "required".
342 accumulatedConfiguration.persistentState = CDMRequirement::Required;
343 } else {
344 // ↳ Otherwise
345 // Change accumulated configuration's persistentState value to "not-allowed".
346 accumulatedConfiguration.persistentState = CDMRequirement::NotAllowed;
347 }
348 }
349
350 // 20. If implementation in the configuration specified by the combination of the values in accumulated configuration
351 // is not supported or not allowed in the origin, return NotSupported.
352 if (!supportsConfiguration(accumulatedConfiguration))
353 return std::nullopt;
354
355 if ((accumulatedConfiguration.distinctiveIdentifier == CDMRequirement::Required || accumulatedConfiguration.persistentState == CDMRequirement::Required) && access == LocalStorageAccess::NotAllowed)
356 return std::nullopt;
357
358 return accumulatedConfiguration;
359 // NOTE: Continued in getConsentStatus().
360}

362std::optional<Vector<CDMMediaCapability>> CDMPrivate::getSupportedCapabilitiesForAudioVideoType(CDMPrivate::AudioVideoType type, const Vector<CDMMediaCapability>& requestedCapabilities, const CDMKeySystemConfiguration& partialConfiguration, CDMRestrictions& restrictions)
363{
364 // https://w3c.github.io/encrypted-media/#get-supported-capabilities-for-audio-video-type
365 // W3C Editor's Draft 09 November 2016
366
367 // 3.1.1.3 Get Supported Capabilities for Audio/Video Type
368
369 // Given an audio/video type, CDMMediaCapability sequence requested media capabilities, CDMKeySystemConfiguration
370 // partial configuration, and restrictions, this algorithm returns a sequence of supported CDMMediaCapability values
371 // for this audio/video type or null as appropriate.
372
373 // 1. Let local accumulated configuration be a local copy of partial configuration.
374 CDMKeySystemConfiguration accumulatedConfiguration = partialConfiguration;
375
376 // 2. Let supported media capabilities be an empty sequence of CDMMediaCapability dictionaries.
377 Vector<CDMMediaCapability> supportedMediaCapabilities { };
378
379 // 3. For each requested media capability in requested media capabilities:
380 for (auto& requestedCapability : requestedCapabilities) {
381 // 3.1. Let content type be requested media capability's contentType member.
382 // 3.2. Let robustness be requested media capability's robustness member.
383 String robustness = requestedCapability.robustness;
384
385 // 3.3. If content type is the empty string, return null.
386 if (requestedCapability.contentType.isEmpty())
387 return std::nullopt;
388
389 // 3.4. If content type is an invalid or unrecognized MIME type, continue to the next iteration.
390 std::optional<ParsedContentType> contentType = ParsedContentType::create(requestedCapability.contentType, Mode::Rfc2045);
391 if (!contentType)
392 continue;
393
394 // 3.5. Let container be the container type specified by content type.
395 String container = contentType->mimeType();
396
397 // 3.6. If the user agent does not support container, continue to the next iteration. The case-sensitivity
398 // of string comparisons is determined by the appropriate RFC.
399 // 3.7. Let parameters be the RFC 6381 [RFC6381] parameters, if any, specified by content type.
400 // 3.8. If the user agent does not recognize one or more parameters, continue to the next iteration.
401 // 3.9. Let media types be the set of codecs and codec constraints specified by parameters. The case-sensitivity
402 // of string comparisons is determined by the appropriate RFC or other specification.
403 String codecs = contentType->parameterValueForName("codecs");
404 if (contentType->parameterCount() > (codecs.isEmpty() ? 0 : 1))
405 continue;
406
407 // 3.10. If media types is empty:
408 if (codecs.isEmpty()) {
409 // ↳ If container normatively implies a specific set of codecs and codec constraints:
410 // ↳ Otherwise:
411 notImplemented();
412 }
413
414 // 3.11. If content type is not strictly a audio/video type, continue to the next iteration.
415 // 3.12. If robustness is not the empty string and contains an unrecognized value or a value not supported by
416 // implementation, continue to the next iteration. String comparison is case-sensitive.
417 if (!robustness.isEmpty() && !supportedRobustnesses().contains(robustness))
418 continue;
419
420 // 3.13. If the user agent and implementation definitely support playback of encrypted media data for the
421 // combination of container, media types, robustness and local accumulated configuration in combination
422 // with restrictions:
423 MediaEngineSupportParameters parameters;
424 parameters.type = ContentType(contentType->mimeType());
425 if (MediaPlayer::supportsType(parameters) == MediaPlayer::SupportsType::IsNotSupported) {
426
427 // Try with Media Source:
428 parameters.isMediaSource = true;
429 if (MediaPlayer::supportsType(parameters) == MediaPlayer::SupportsType::IsNotSupported)
430 continue;
431 }
432
433 if (!supportsConfigurationWithRestrictions(accumulatedConfiguration, restrictions))
434 continue;
435
436 // 3.13.1. Add requested media capability to supported media capabilities.
437 supportedMediaCapabilities.append(requestedCapability);
438
439 // 3.13.2. ↳ If audio/video type is Video:
440 // Add requested media capability to the videoCapabilities member of local accumulated configuration.
441 if (type == AudioVideoType::Video)
442 accumulatedConfiguration.videoCapabilities.append(requestedCapability);
443 // 3.13.2. ↳ If audio/video type is Audio:
444 // Add requested media capability to the audioCapabilities member of local accumulated configuration.
445 else
446 accumulatedConfiguration.audioCapabilities.append(requestedCapability);
447 }
448
449 // 4. If supported media capabilities is empty, return null.
450 if (supportedMediaCapabilities.isEmpty())
451 return std::nullopt;
452
453 // 5. Return supported media capabilities.
454 return supportedMediaCapabilities;
455}

457void CDMPrivate::getConsentStatus(CDMKeySystemConfiguration&& accumulatedConfiguration, CDMRestrictions&& restrictions, LocalStorageAccess access, ConsentStatusCallback&& callback)
458{
459 // https://w3c.github.io/encrypted-media/#get-supported-configuration-and-consent
460 // W3C Editor's Draft 09 November 2016
461
462 // 3.1.1.2 Get Supported Configuration and Consent, ctd.
463 // 21. If accumulated configuration's distinctiveIdentifier value is "required" and the Distinctive Identifier(s) associated
464 // with accumulated configuration are not unique per origin and profile and clearable:
465 if (accumulatedConfiguration.distinctiveIdentifier == CDMRequirement::Required && !distinctiveIdentifiersAreUniquePerOriginAndClearable(accumulatedConfiguration)) {
466 // 21.1. Update restrictions to reflect that all configurations described by accumulated configuration do not have user consent.
467 restrictions.distinctiveIdentifierDenied = true;
468 callback(ConsentStatus::ConsentDenied, WTFMove(accumulatedConfiguration), WTFMove(restrictions));
469 return;
470 }
471
472 // https://w3c.github.io/encrypted-media/#get-consent-status
473 // 3.1.1.4 Get Consent Status
474 // Given an accumulated configuration, restrictions and origin, this algorithm returns the consent status for accumulated
475 // configuration and origin as one of ConsentDenied, InformUser or Allowed, together with an updated value for restrictions
476 // in the ConsentDenied case.
477
478 // 1. If there is persisted denial for origin indicating that accumulated configuration is not allowed, run the following steps:
479 // 1.1. Update restrictions to reflect the configurations for which consent has been denied.
480 // 1.2. Return ConsentDenied and restrictions.
481 // 2. If there is persisted consent for origin indicating accumulated configuration is allowed, return Allowed.
482 // NOTE: persisted denial / consent unimplemented.
483
484 // 3. If any of the following are true:
485 // ↳ The distinctiveIdentifier member of accumulated configuration is not "not-allowed" and the combination of the User Agent,
486 // implementation and accumulated configuration does not follow all the recommendations of Allow Persistent Data to Be Cleared
487 // with respect to Distinctive Identifier(s).
488 // NOTE: assume that implementations follow all recommendations.
489
490 // ↳ The user agent requires explicit user consent for the accumulated configuration for other reasons.
491 // NOTE: assume the user agent does not require explicit user consent.
492
493 // 3.1. Request user consent to use accumulated configuration in the origin and wait for the user response.
494 // The consent must include consent to use a Distinctive Identifier(s) and/or Distinctive Permanent Identifier(s) as appropriate
495 // if accumulated configuration's distinctiveIdentifier member is "required".
496 // 3.2. If consent was denied, run the following steps:
497 // 3.2.1. Update restrictions to reflect the configurations for which consent was denied.
498 // 3.2.1. Return ConsentDenied and restrictions.
499 // NOTE: assume implied consent if the combination of origin and topOrigin allows it.
500 if (accumulatedConfiguration.distinctiveIdentifier == CDMRequirement::Required && access == LocalStorageAccess::NotAllowed) {
501 restrictions.distinctiveIdentifierDenied = true;
502 callback(ConsentStatus::ConsentDenied, WTFMove(accumulatedConfiguration), WTFMove(restrictions));
503 return;
504 }
505
506 // 4. If the distinctiveIdentifier member of accumulated configuration is not "not-allowed", return InformUser.
507 if (accumulatedConfiguration.distinctiveIdentifier != CDMRequirement::NotAllowed) {
508 callback(ConsentStatus::InformUser, WTFMove(accumulatedConfiguration), WTFMove(restrictions));
509 return;
510 }
511
512 // 5. If the user agent requires informing the user for the accumulated configuration for other reasons, return InformUser.
513 // NOTE: assume the user agent does not require informing the user.
514
515 // 6. Return Allowed.
516 callback(ConsentStatus::Allowed, WTFMove(accumulatedConfiguration), WTFMove(restrictions));
517}
