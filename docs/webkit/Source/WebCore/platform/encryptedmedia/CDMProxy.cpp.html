60RefPtr<CDMProxy> CDMProxyFactory::createCDMProxyForKeySystem(const String& keySystem)
61{
62 RefPtr<CDMProxy> cdmProxy;
63 for (auto* factory : CDMProxyFactory::registeredFactories()) {
64 if (factory->supportsKeySystem(keySystem)) {
65 cdmProxy = factory->createCDMProxy(keySystem);
66 break;
67 }
68 }
69#if USE(GSTREAMER)
70 ASSERT(cdmProxy);
71#endif
72 return cdmProxy;
73}

85static String vectorToHexString(const Vector<uint8_t>& vec)
86{
87 StringBuilder stringBuilder;
88 for (auto byte : vec)
89 stringBuilder.append(pad('0', 2, hex(byte)));
90 return stringBuilder.toString();
91}

116void KeyStore::merge(const KeyStore& other)
117{
118 ASSERT(isMainThread());
119 LOG(EME, "EME - CDMProxy - merging %u new keys into a key store of %u keys", other.numKeys(), numKeys());
120 for (const auto& key : other)
121 add(key.copyRef());
122
123#if !LOG_DISABLED
124 LOG(EME, "EME - CDMProxy - key store now has %u keys", numKeys());
125 for (const auto& key : m_keys)
126 LOG(EME, "\tEME - CDMProxy - Key ID: %s", key->idAsString().ascii().data());
127#endif // !LOG_DISABLED
128}

138bool KeyStore::addKeys(Vector<RefPtr<KeyHandle>>&& newKeys)
139{
140 bool didKeyStoreChange = false;
141 for (auto& key : newKeys) {
142 if (add(WTFMove(key)))
143 didKeyStoreChange = true;
144 }
145 return didKeyStoreChange;
146}
147
148bool KeyStore::add(RefPtr<KeyHandle>&& key)
149{
150 bool didStoreChange = false;
151 size_t keyWithMatchingKeyIDIndex = m_keys.findMatching([&] (const RefPtr<KeyHandle>& storedKey) {
152 return *key == *storedKey;
153 });
154
155 addSessionReferenceTo(key);
156 if (keyWithMatchingKeyIDIndex != WTF::notFound) {
157 auto& keyWithMatchingKeyID = m_keys[keyWithMatchingKeyIDIndex];
158 didStoreChange = keyWithMatchingKeyID != key;
159 if (didStoreChange)
160 keyWithMatchingKeyID->mergeKeyInto(WTFMove(key));
161 } else {
162 LOG(EME, "EME - ClearKey - New key with ID %s getting added to key store", key->idAsString().ascii().data());
163 m_keys.append(WTFMove(key));
164 didStoreChange = true;
165 }
166
167 if (didStoreChange) {
168 // Sort the keys lexicographically.
169 // NOTE: This is not as pathological as it may seem, for all
170 // practical purposes the store has a maximum of 2 keys.
171 std::sort(m_keys.begin(), m_keys.end(),
172 [](const RefPtr<KeyHandle>& a, const RefPtr<KeyHandle>& b) {
173 return *a < *b;
174 });
175 }
176
177 return didStoreChange;
178}

192bool KeyStore::unref(const RefPtr<KeyHandle>& key)
193{
194 bool storeChanged = false;
195
196 size_t keyWithMatchingKeyIDIndex = m_keys.find(key);
197 LOG(EME, "EME - ClearKey - requested to unref key with ID %s and %d session references", key->idAsString().ascii().data(), key->numSessionReferences());
198
199 if (keyWithMatchingKeyIDIndex != WTF::notFound) {
200 auto& keyWithMatchingKeyID = m_keys[keyWithMatchingKeyIDIndex];
201 removeSessionReferenceFrom(keyWithMatchingKeyID);
202 if (!keyWithMatchingKeyID->hasReferences()) {
203 LOG(EME, "EME - ClearKey - unref key with ID %s", keyWithMatchingKeyID->idAsString().ascii().data());
204 m_keys.remove(keyWithMatchingKeyIDIndex);
205 storeChanged = true;
206 }
207 } else
208 LOG(EME, "EME - ClearKey - attempt to unref key with ID %s ignored, does not exist", key->idAsString().ascii().data());
209
210 return storeChanged;
211}

224CDMInstanceSession::KeyStatusVector KeyStore::convertToJSKeyStatusVector() const
225{
226 CDMInstanceSession::KeyStatusVector keyStatusVector;
227 keyStatusVector.reserveInitialCapacity(numKeys());
228 for (const auto& key : m_keys)
229 keyStatusVector.uncheckedAppend(std::pair<Ref<SharedBuffer>, CDMInstanceSession::KeyStatus> { key->idAsSharedBuffer(), key->status() });
230 return keyStatusVector;

233void CDMProxy::updateKeyStore(const KeyStore& newKeyStore)
234{

237 LOG(EME, "EME - CDMProxy - updating key store from a session update");
238 m_keysCondition.notifyAll();

290std::optional<Ref<KeyHandle>> CDMProxy::tryWaitForKeyHandle(const KeyIDType& keyID, WeakPtr<CDMProxyDecryptionClient>&& client) const
291{
292 startedWaitingForKey();
293 // Unconditionally saying we have stopped waiting for a key means that decryptors only get
294 // one shot at fetching a key. If MaxKeyWaitTimeSeconds expires, that's it, no more clear bytes
295 // for you.
296 auto stopWaitingForKeyOnReturn = makeScopeExit([this] {
297 stoppedWaitingForKey();
298 });
299 LOG(EME, "EME - CDMProxy - trying to wait for key ID %s", vectorToHexString(keyID).ascii().data());
300 bool wasKeyAvailable = false;
301 {
302 Locker locker { m_keysLock };
303
304 m_keysCondition.waitFor(m_keysLock, CDMProxy::MaxKeyWaitTimeSeconds, [this, keyID, client = WTFMove(client), &wasKeyAvailable]() {
305 assertIsHeld(m_keysLock);
306 if (!client || client->isAborting())
307 return true;
308 wasKeyAvailable = keyAvailableUnlocked(keyID);
309 return wasKeyAvailable;
310 });
311 }
312
313 if (wasKeyAvailable) {
314 RefPtr<KeyHandle> handle = keyHandle(keyID);
315 LOG(EME, "EME - CDMProxy - successfully waited for key ID %s", vectorToHexString(keyID).ascii().data());
316 return std::make_optional(handle.releaseNonNull());
317 }
318
319 LOG(EME, "EME - CDMProxy - key ID %s not available or operation aborted", vectorToHexString(keyID).ascii().data());
320 return std::nullopt;
321}

334std::optional<Ref<KeyHandle>> CDMProxy::getOrWaitForKeyHandle(const KeyIDType& keyID, WeakPtr<CDMProxyDecryptionClient>&& client) const
335{
336 if (!keyAvailable(keyID)) {
337 LOG(EME, "EME - CDMProxy key cache does not contain key ID %s", vectorToHexString(keyID).ascii().data());
338 return tryWaitForKeyHandle(keyID, WTFMove(client));
339 }
340
341 RefPtr<KeyHandle> handle = keyHandle(keyID);
342 return std::make_optional(handle.releaseNonNull());
343}

345std::optional<KeyHandleValueVariant> CDMProxy::getOrWaitForKeyValue(const KeyIDType& keyID, WeakPtr<CDMProxyDecryptionClient>&& client) const
346{
347 if (auto keyHandle = getOrWaitForKeyHandle(keyID, WTFMove(client)))
348 return std::make_optional((*keyHandle)->value());
349 return std::nullopt;

352void CDMInstanceProxy::startedWaitingForKey()
353{
354 ASSERT(!isMainThread());
355 ASSERT(m_player);
356
357 bool wasWaitingForKey = m_numDecryptorsWaitingForKey > 0;
358 m_numDecryptorsWaitingForKey++;
359
360 callOnMainThread([player = m_player, wasWaitingForKey] {
361 if (player && !wasWaitingForKey)
362 player->waitingForKeyChanged();
363 });
364}
365
366void CDMInstanceProxy::stoppedWaitingForKey()
367{
368 ASSERT(!isMainThread());
369 ASSERT(m_player);
370 ASSERT(m_numDecryptorsWaitingForKey > 0);
371 m_numDecryptorsWaitingForKey--;
372 bool isNobodyWaitingForKey = !m_numDecryptorsWaitingForKey;
373
374 callOnMainThread([player = m_player, isNobodyWaitingForKey] {
375 if (player && isNobodyWaitingForKey)
376 player->waitingForKeyChanged();
377 });
378}
