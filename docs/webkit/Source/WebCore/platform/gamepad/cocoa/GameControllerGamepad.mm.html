37GameControllerGamepad::GameControllerGamepad(GCController *controller, unsigned index)
38 : PlatformGamepad(index)
39 , m_gcController(controller)
40{
41 ASSERT(index < 4);
42 controller.playerIndex = (GCControllerPlayerIndex)(GCControllerPlayerIndex1 + index);
43
44 m_extendedGamepad = controller.extendedGamepad;
45
46 ALLOW_DEPRECATED_DECLARATIONS_BEGIN
47 if (!m_extendedGamepad)
48 m_gamepad = controller.gamepad;
49 ALLOW_DEPRECATED_DECLARATIONS_END
50
51 ASSERT(m_extendedGamepad || m_gamepad);
52
53 if (m_extendedGamepad)
54 setupAsExtendedGamepad();
55 else
56 setupAsGamepad();
57}
58
59static GCControllerButtonInput *homeButtonFromExtendedGamepad(GCExtendedGamepad *gamepad)
60{
61#if HAVE(GCEXTENDEDGAMEPAD_HOME_BUTTON)
62 if (gamepad.buttonHome)
63 return gamepad.buttonHome;
64#endif
65
66 id potentialButton = [gamepad valueForKey:@"_buttonHome"];
67 if (potentialButton && [potentialButton isKindOfClass:getGCControllerButtonInputClass()])
68 return potentialButton;
69
70 return nil;
71}

73void GameControllerGamepad::setupAsExtendedGamepad()
74{
75 ASSERT(m_extendedGamepad);
76
77 m_id = makeString(String(m_gcController.get().vendorName), " Extended Gamepad"_s);
78 m_mapping = standardGamepadMappingString();
79
80 auto *homeButton = homeButtonFromExtendedGamepad(m_extendedGamepad.get());
81 m_buttonValues.resize(homeButton ? numberOfStandardGamepadButtonsWithHomeButton : numberOfStandardGamepadButtonsWithoutHomeButton);
82
83 m_extendedGamepad.get().valueChangedHandler = ^(GCExtendedGamepad *, GCControllerElement *) {
84 m_lastUpdateTime = MonotonicTime::now();
85 GameControllerGamepadProvider::singleton().gamepadHadInput(*this, m_hadButtonPresses);
86 m_hadButtonPresses = false;
87 };
88
89 auto bindButton = ^(GCControllerButtonInput *button, GamepadButtonRole index) {
90 m_buttonValues[(size_t)index].setValue(button.value);
91 button.valueChangedHandler = ^(GCControllerButtonInput *, float value, BOOL pressed) {
92 m_buttonValues[(size_t)index].setValue(value);
93 if (pressed)
94 m_hadButtonPresses = true;
95 };
96 };
97
98 // Button Pad
99 bindButton(m_extendedGamepad.get().buttonA, GamepadButtonRole::RightClusterBottom);
100 bindButton(m_extendedGamepad.get().buttonB, GamepadButtonRole::RightClusterRight);
101 bindButton(m_extendedGamepad.get().buttonX, GamepadButtonRole::RightClusterLeft);
102 bindButton(m_extendedGamepad.get().buttonY, GamepadButtonRole::RightClusterTop);
103
104 // Shoulders, Triggers
105 bindButton(m_extendedGamepad.get().leftShoulder, GamepadButtonRole::LeftShoulderFront);
106 bindButton(m_extendedGamepad.get().rightShoulder, GamepadButtonRole::RightShoulderFront);
107 bindButton(m_extendedGamepad.get().leftTrigger, GamepadButtonRole::LeftShoulderBack);
108 bindButton(m_extendedGamepad.get().rightTrigger, GamepadButtonRole::RightShoulderBack);
109
110 // D Pad
111 bindButton(m_extendedGamepad.get().dpad.up, GamepadButtonRole::LeftClusterTop);
112 bindButton(m_extendedGamepad.get().dpad.down, GamepadButtonRole::LeftClusterBottom);
113 bindButton(m_extendedGamepad.get().dpad.left, GamepadButtonRole::LeftClusterLeft);
114 bindButton(m_extendedGamepad.get().dpad.right, GamepadButtonRole::LeftClusterRight);
115
116 if (homeButton)
117 bindButton(homeButton, GamepadButtonRole::CenterClusterCenter);
118
119 // Select, Start
120#if HAVE(GCEXTENDEDGAMEPAD_BUTTONS_OPTIONS_MENU)
121 bindButton(m_extendedGamepad.get().buttonOptions, GamepadButtonRole::CenterClusterLeft);
122 bindButton(m_extendedGamepad.get().buttonMenu, GamepadButtonRole::CenterClusterRight);
123#endif
124
125 // L3, R3
126#if HAVE(GCEXTENDEDGAMEPAD_BUTTONS_THUMBSTICK)
127 // Thumbstick buttons are only in macOS 10.14.1 / iOS 12.1
128#pragma clang diagnostic push
129#pragma clang diagnostic ignored "-Wunguarded-availability"
130 if ([m_extendedGamepad.get() respondsToSelector:@selector(leftThumbstickButton)]) {
131 bindButton(m_extendedGamepad.get().leftThumbstickButton, GamepadButtonRole::LeftStick);
132 bindButton(m_extendedGamepad.get().rightThumbstickButton, GamepadButtonRole::RightStick);
133 }
134#pragma clang diagnostic pop
135#endif
136
137 m_axisValues.resize(4);
138 m_axisValues[0].setValue(m_extendedGamepad.get().leftThumbstick.xAxis.value);
139 m_axisValues[1].setValue(-m_extendedGamepad.get().leftThumbstick.yAxis.value);
140 m_axisValues[2].setValue(m_extendedGamepad.get().rightThumbstick.xAxis.value);
141 m_axisValues[3].setValue(-m_extendedGamepad.get().rightThumbstick.yAxis.value);
142
143 m_extendedGamepad.get().leftThumbstick.xAxis.valueChangedHandler = ^(GCControllerAxisInput *, float value) {
144 m_axisValues[0].setValue(value);
145 };
146 m_extendedGamepad.get().leftThumbstick.yAxis.valueChangedHandler = ^(GCControllerAxisInput *, float value) {
147 m_axisValues[1].setValue(-value);
148 };
149 m_extendedGamepad.get().rightThumbstick.xAxis.valueChangedHandler = ^(GCControllerAxisInput *, float value) {
150 m_axisValues[2].setValue(value);
151 };
152 m_extendedGamepad.get().rightThumbstick.yAxis.valueChangedHandler = ^(GCControllerAxisInput *, float value) {
153 m_axisValues[3].setValue(-value);
154 };
155}

157void GameControllerGamepad::setupAsGamepad()
158{
159 ASSERT(m_gamepad);
160
161 m_id = makeString(String(m_gcController.get().vendorName), " Gamepad"_s);
162
163 ALLOW_DEPRECATED_DECLARATIONS_BEGIN // GCGamepad
164 m_gamepad.get().valueChangedHandler = ^(GCGamepad *, GCControllerElement *) {
165 m_lastUpdateTime = MonotonicTime::now();
166 GameControllerGamepadProvider::singleton().gamepadHadInput(*this, m_hadButtonPresses);
167 m_hadButtonPresses = false;
168 };
169 ALLOW_DEPRECATED_DECLARATIONS_END
170
171 m_buttonValues.resize(6);
172 m_buttonValues[0].setValue(m_extendedGamepad.get().buttonA.value);
173 m_buttonValues[1].setValue(m_extendedGamepad.get().buttonB.value);
174 m_buttonValues[2].setValue(m_extendedGamepad.get().buttonX.value);
175 m_buttonValues[3].setValue(m_extendedGamepad.get().buttonY.value);
176 m_buttonValues[4].setValue(m_extendedGamepad.get().leftShoulder.value);
177 m_buttonValues[5].setValue(m_extendedGamepad.get().rightShoulder.value);
178
179 m_axisValues.resize(2);
180 m_axisValues[0].setValue(m_extendedGamepad.get().dpad.xAxis.value);
181 m_axisValues[1].setValue(m_extendedGamepad.get().dpad.yAxis.value);
182
183 m_extendedGamepad.get().buttonA.valueChangedHandler = ^(GCControllerButtonInput *, float value, BOOL pressed) {
184 m_buttonValues[0].setValue(value);
185 if (pressed)
186 m_hadButtonPresses = true;
187 };
188 m_extendedGamepad.get().buttonB.valueChangedHandler = ^(GCControllerButtonInput *, float value, BOOL pressed) {
189 m_buttonValues[1].setValue(value);
190 if (pressed)
191 m_hadButtonPresses = true;
192 };
193 m_extendedGamepad.get().buttonX.valueChangedHandler = ^(GCControllerButtonInput *, float value, BOOL pressed) {
194 m_buttonValues[2].setValue(value);
195 if (pressed)
196 m_hadButtonPresses = true;
197 };
198 m_extendedGamepad.get().buttonY.valueChangedHandler = ^(GCControllerButtonInput *, float value, BOOL pressed) {
199 m_buttonValues[3].setValue(value);
200 if (pressed)
201 m_hadButtonPresses = true;
202 };
203 m_extendedGamepad.get().leftShoulder.valueChangedHandler = ^(GCControllerButtonInput *, float value, BOOL pressed) {
204 m_buttonValues[4].setValue(value);
205 if (pressed)
206 m_hadButtonPresses = true;
207 };
208 m_extendedGamepad.get().rightShoulder.valueChangedHandler = ^(GCControllerButtonInput *, float value, BOOL pressed) {
209 m_buttonValues[5].setValue(value);
210 if (pressed)
211 m_hadButtonPresses = true;
212 };
213
214 m_extendedGamepad.get().leftThumbstick.xAxis.valueChangedHandler = ^(GCControllerAxisInput *, float value) {
215 m_axisValues[0].setValue(value);
216 };
217 m_extendedGamepad.get().leftThumbstick.yAxis.valueChangedHandler = ^(GCControllerAxisInput *, float value) {
218 m_axisValues[1].setValue(value);
219 };
220}
