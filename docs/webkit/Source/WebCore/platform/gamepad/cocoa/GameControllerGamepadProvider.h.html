40class GameControllerGamepad;

43class GameControllerGamepadProvider : public GamepadProvider {

46public:
47 WEBCORE_EXPORT static GameControllerGamepadProvider& singleton();
48
49 WEBCORE_EXPORT void startMonitoringGamepads(GamepadProviderClient&) final;
50 WEBCORE_EXPORT void stopMonitoringGamepads(GamepadProviderClient&) final;
51 const Vector<PlatformGamepad*>& platformGamepads() final { return m_gamepadVector; }

53 WEBCORE_EXPORT void stopMonitoringInput();

55
56 void gamepadHadInput(GameControllerGamepad&, bool hadButtonPresses);

58
59 void makeInvisibleGamepadsVisible();
60
61 size_t numberOfConnectedGamepads() const { return m_gamepadMap.size(); };
62
63#if !HAVE(GCCONTROLLER_HID_DEVICE_CHECK)
64 static bool willHandleVendorAndProduct(uint16_t vendorID, uint16_t productID);

67private:

75 void controllerDidConnect(GCController *, ConnectionVisibility);
76 void controllerDidDisconnect(GCController *);

78 unsigned indexForNewlyConnectedDevice();

80 void inputNotificationTimerFired();

82 HashMap<CFTypeRef, std::unique_ptr<GameControllerGamepad>> m_gamepadMap;
