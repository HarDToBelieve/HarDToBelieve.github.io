35HIDGamepadElement::HIDGamepadElement(const HIDElement& element)

40void HIDGamepadElement::refreshCurrentValue()
41{
42 IOHIDValueRef value;
43 if (IOHIDDeviceGetValue(IOHIDElementGetDevice(rawElement()), rawElement(), &value) == kIOReturnSuccess)
44 gamepadValueChanged(value);
45}
46
47double HIDGamepadElement::normalizedValue()
48{
49 // Default normalization (and the normalization buttons will use) is 0 to 1.0
50 return (double)(physicalValue() - physicalMin()) / (double)(physicalMax() - physicalMin());
51}
52
53#pragma mark HIDGamepadButton
54
55HIDInputType HIDGamepadButton::gamepadValueChanged(IOHIDValueRef value)
56{
57 valueChanged(value);
58 m_value.setValue(normalizedValue());
59 return m_value.value() > 0.5 ? HIDInputType::ButtonPress : HIDInputType::NotAButtonPress;
60}
61
62#pragma mark HIDGamepadAxis
63
64HIDInputType HIDGamepadAxis::gamepadValueChanged(IOHIDValueRef value)
65{
66 valueChanged(value);
67 m_value.setValue(normalizedValue());
68 return HIDInputType::NotAButtonPress;
69}
70
71double HIDGamepadAxis::normalizedValue()
72{
73 // Web Gamepad axes have a value of -1.0 to 1.0
74 return (HIDGamepadElement::normalizedValue() * 2.0) - 1.0;
75}

77#pragma mark HIDGamepadHatswitch
78
79HIDInputType HIDGamepadHatswitch::gamepadValueChanged(IOHIDValueRef value)
80{
81 valueChanged(value);
82
83 for (size_t i = 0; i < 4; ++i)
84 m_buttonValues[i].setValue(0.0);
85
86 switch (physicalValue()) {
87 case 0:
88 m_buttonValues[0].setValue(1.0);
89 break;
90 case 45:
91 m_buttonValues[0].setValue(1.0);
92 m_buttonValues[1].setValue(1.0);
93 break;
94 case 90:
95 m_buttonValues[1].setValue(1.0);
96 break;
97 case 135:
98 m_buttonValues[1].setValue(1.0);
99 m_buttonValues[2].setValue(1.0);
100 break;
101 case 180:
102 m_buttonValues[2].setValue(1.0);
103 break;
104 case 225:
105 m_buttonValues[2].setValue(1.0);
106 m_buttonValues[3].setValue(1.0);
107 break;
108 case 270:
109 m_buttonValues[3].setValue(1.0);
110 break;
111 case 315:
112 m_buttonValues[3].setValue(1.0);
113 m_buttonValues[0].setValue(1.0);
114 break;
115 default:
116 break;
117 };
118
119 return HIDInputType::ButtonPress;
120}

124 // Hatswitch normalizedValue should never be accessed.
125 RELEASE_ASSERT_NOT_REACHED();
