40namespace WebCore {
41
42class GamepadProviderClient;
43
44class HIDGamepadProvider : public GamepadProvider {
45 WTF_MAKE_NONCOPYABLE(HIDGamepadProvider);
46 friend class NeverDestroyed<HIDGamepadProvider>;
47public:
48 WEBCORE_EXPORT static HIDGamepadProvider& singleton();
49
50 WEBCORE_EXPORT void startMonitoringGamepads(GamepadProviderClient&) final;
51 WEBCORE_EXPORT void stopMonitoringGamepads(GamepadProviderClient&) final;
52 const Vector<PlatformGamepad*>& platformGamepads() final { return m_gamepadVector; }
53
54 WEBCORE_EXPORT void stopMonitoringInput();
55 WEBCORE_EXPORT void startMonitoringInput();
56
57 void deviceAdded(IOHIDDeviceRef);
58 void deviceRemoved(IOHIDDeviceRef);
59 void valuesChanged(IOHIDValueRef);
60
61 void ignoreGameControllerFrameworkDevices() { m_ignoresGameControllerFrameworkDevices = true; }
62
63 size_t numberOfConnectedGamepads() const { return m_gamepadMap.size(); };
64
65private:
66 HIDGamepadProvider();
67
68 std::unique_ptr<HIDGamepad> removeGamepadForDevice(IOHIDDeviceRef);
69
70 void openAndScheduleManager();
71 void closeAndUnscheduleManager();
72
73 void initialGamepadsConnectedTimerFired();
74 void inputNotificationTimerFired();
75
76 unsigned indexForNewlyConnectedDevice();
77
78 Vector<PlatformGamepad*> m_gamepadVector;
79 HashMap<IOHIDDeviceRef, std::unique_ptr<HIDGamepad>> m_gamepadMap;
80
81 RetainPtr<IOHIDManagerRef> m_manager;
82
83 bool m_initialGamepadsConnected { false };
84 bool m_ignoresGameControllerFrameworkDevices { false };
85
86 Timer m_initialGamepadsConnectedTimer;
87 Timer m_inputNotificationTimer;
88
89#if HAVE(MULTIGAMEPADPROVIDER_SUPPORT)
90 HashSet<IOHIDDeviceRef> m_gameControllerManagedGamepads;
91#endif // HAVE(MULTIGAMEPADPROVIDER_SUPPORT)
92};
93
94} // namespace WebCore

96#endif // ENABLE(GAMEPAD) && PLATFORM(MAC)
