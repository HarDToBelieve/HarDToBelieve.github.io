39LogitechGamepad::LogitechGamepad(HIDDevice&& device, unsigned index)
40 : HIDGamepad(WTFMove(device), index)
41{
42 LOG(Gamepad, "Creating LogitechGamepad %p", this);
43
44 m_mapping = standardGamepadMappingString();
45
46 for (size_t i = 0; i < numberOfStandardGamepadButtonsWithoutHomeButton; ++i)
47 m_buttonValues.append(0.0);
48
49 static const size_t axisCount = 4;
50 for (size_t i = 0; i < axisCount; ++i)
51 m_axisValues.append(0.0);
52
53 auto inputElements = hidDevice().uniqueInputElementsInDeviceTreeOrder();
54
55 auto mapButton = [this] (HIDElement& element, GamepadButtonRole role) {
56 m_elementMap.set(element.cookie(), makeUnique<HIDGamepadButton>(element, m_buttonValues[(size_t)role]));
57 };
58
59 for (auto& element : inputElements) {
60 switch (element.fullUsage()) {
61 case hidHatswitchFullUsage: {
62 Vector<SharedGamepadValue> hatswitchValues;
63 hatswitchValues.append(m_buttonValues[(size_t)GamepadButtonRole::LeftClusterTop]);
64 hatswitchValues.append(m_buttonValues[(size_t)GamepadButtonRole::LeftClusterRight]);
65 hatswitchValues.append(m_buttonValues[(size_t)GamepadButtonRole::LeftClusterBottom]);
66 hatswitchValues.append(m_buttonValues[(size_t)GamepadButtonRole::LeftClusterLeft]);
67
68 m_elementMap.set(element.cookie(), makeUnique<HIDGamepadHatswitch>(element, WTFMove(hatswitchValues)));
69 break;
70 }
71 case hidXAxisFullUsage:
72 m_elementMap.set(element.cookie(), makeUnique<HIDGamepadAxis>(element, m_axisValues[0]));
73 break;
74 case hidYAxisFullUsage:
75 m_elementMap.set(element.cookie(), makeUnique<HIDGamepadAxis>(element, m_axisValues[1]));
76 break;
77 case hidZAxisFullUsage:
78 m_elementMap.set(element.cookie(), makeUnique<HIDGamepadAxis>(element, m_axisValues[2]));
79 break;
80 case hidRzAxisFullUsage:
81 m_elementMap.set(element.cookie(), makeUnique<HIDGamepadAxis>(element, m_axisValues[3]));
82 break;
83 case hidButton1FullUsage:
84 mapButton(element, GamepadButtonRole::RightClusterLeft);
85 break;
86 case hidButton2FullUsage:
87 mapButton(element, GamepadButtonRole::RightClusterBottom);
88 break;
89 case hidButton3FullUsage:
90 mapButton(element, GamepadButtonRole::RightClusterRight);
91 break;
92 case hidButton4FullUsage:
93 mapButton(element, GamepadButtonRole::RightClusterTop);
94 break;
95 case hidButton5FullUsage:
96 mapButton(element, GamepadButtonRole::LeftShoulderFront);
97 break;
98 case hidButton6FullUsage:
99 mapButton(element, GamepadButtonRole::RightShoulderFront);
100 break;
101 case hidButton7FullUsage:
102 mapButton(element, GamepadButtonRole::LeftShoulderBack);
103 break;
104 case hidButton8FullUsage:
105 mapButton(element, GamepadButtonRole::RightShoulderBack);
106 break;
107 case hidButton9FullUsage:
108 mapButton(element, GamepadButtonRole::CenterClusterLeft);
109 break;
110 case hidButton10FullUsage:
111 mapButton(element, GamepadButtonRole::CenterClusterRight);
112 break;
113 case hidButton11FullUsage:
114 mapButton(element, GamepadButtonRole::LeftStick);
115 break;
116 case hidButton12FullUsage:
117 mapButton(element, GamepadButtonRole::RightStick);
118 break;
119 default:
120 break;
121 }
122 }
123}
