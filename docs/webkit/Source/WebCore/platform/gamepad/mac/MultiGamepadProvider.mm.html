41MultiGamepadProvider& MultiGamepadProvider::singleton()
42{
43 static NeverDestroyed<MultiGamepadProvider> sharedProvider;
44 return sharedProvider;
45}

47void MultiGamepadProvider::startMonitoringGamepads(GamepadProviderClient& client)
48{
49 bool monitorOtherProviders = m_clients.isEmpty();
50
51 ASSERT(!m_clients.contains(&client));
52 m_clients.add(&client);
53
54 if (!m_usesOnlyHIDProvider) {
55 HIDGamepadProvider::singleton().ignoreGameControllerFrameworkDevices();
56 GameControllerGamepadProvider::singleton().prewarmGameControllerDevicesIfNecessary();
57 }
58
59 if (monitorOtherProviders) {
60 HIDGamepadProvider::singleton().startMonitoringGamepads(*this);
61 if (!m_usesOnlyHIDProvider)
62 GameControllerGamepadProvider::singleton().startMonitoringGamepads(*this);
63 }
64}
65
66void MultiGamepadProvider::stopMonitoringGamepads(GamepadProviderClient& client)
67{
68 ASSERT(m_clients.contains(&client));
69
70 bool shouldStopMonitoringOtherProviders = m_clients.remove(&client) && m_clients.isEmpty();
71
72 if (shouldStopMonitoringOtherProviders) {
73 HIDGamepadProvider::singleton().stopMonitoringGamepads(*this);
74 if (!m_usesOnlyHIDProvider)
75 GameControllerGamepadProvider::singleton().stopMonitoringGamepads(*this);
76 }
77}

79unsigned MultiGamepadProvider::indexForNewlyConnectedDevice()
80{
81 unsigned index = 0;
82 while (index < m_gamepadVector.size() && m_gamepadVector[index])
83 ++index;
84
85 ASSERT(index <= m_gamepadVector.size());
86
87 if (index == m_gamepadVector.size())
88 m_gamepadVector.resize(index + 1);
89
90 return index;
91}
92
93void MultiGamepadProvider::platformGamepadConnected(PlatformGamepad& gamepad, EventMakesGamepadsVisible eventVisibility)
94{
95 auto index = indexForNewlyConnectedDevice();
96
97 LOG(Gamepad, "MultiGamepadProvider adding new platform gamepad to index %i from a %s source", index, gamepad.source());
98
99 ASSERT(m_gamepadVector.size() > index);
100
101 auto addResult = m_gamepadMap.set(&gamepad, WTF::makeUnique<PlatformGamepadWrapper>(index, &gamepad));
102 ASSERT(addResult.isNewEntry);
103 m_gamepadVector[index] = addResult.iterator->value.get();
104
105 for (auto& client : m_clients)
106 client->platformGamepadConnected(*m_gamepadVector[index], eventVisibility);
107}
108
109void MultiGamepadProvider::platformGamepadDisconnected(PlatformGamepad& gamepad)
110{
111 LOG(Gamepad, "MultiGamepadProvider disconnecting gamepad from a %s source", gamepad.source());
112
113 auto gamepadWrapper = m_gamepadMap.take(&gamepad);
114
115 ASSERT(gamepadWrapper);
116 ASSERT(gamepadWrapper->index() < m_gamepadVector.size());
117 ASSERT(m_gamepadVector[gamepadWrapper->index()] == gamepadWrapper.get());
118
119 m_gamepadVector[gamepadWrapper->index()] = nullptr;
120
121 for (auto& client : m_clients)
122 client->platformGamepadDisconnected(*gamepadWrapper);
123}
124
125void MultiGamepadProvider::platformGamepadInputActivity(EventMakesGamepadsVisible eventVisibility)
126{
127 if (eventVisibility == EventMakesGamepadsVisible::Yes)
128 GameControllerGamepadProvider::singleton().makeInvisibleGamepadsVisible();
129
130 for (auto& client : m_clients)
131 client->platformGamepadInputActivity(eventVisibility);
132}
