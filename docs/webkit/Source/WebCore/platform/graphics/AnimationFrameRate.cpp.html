33static constexpr OptionSet<ThrottlingReason> halfSpeedThrottlingReasons { ThrottlingReason::LowPowerMode, ThrottlingReason::NonInteractedCrossOriginFrame, ThrottlingReason::VisuallyIdle };
34
35FramesPerSecond framesPerSecondNearestFullSpeed(FramesPerSecond nominalFramesPerSecond)
36{
37 if (nominalFramesPerSecond <= FullSpeedFramesPerSecond)
38 return nominalFramesPerSecond;
39
40 float fullSpeedRatio = nominalFramesPerSecond / FullSpeedFramesPerSecond;
41 FramesPerSecond floorSpeed = nominalFramesPerSecond / std::floor(fullSpeedRatio);
42 FramesPerSecond ceilSpeed = nominalFramesPerSecond / std::ceil(fullSpeedRatio);
43
44 return fullSpeedRatio - std::floor(fullSpeedRatio) <= 0.5 ? floorSpeed : ceilSpeed;
45}

47std::optional<FramesPerSecond> preferredFramesPerSecond(OptionSet<ThrottlingReason> reasons, std::optional<FramesPerSecond> nominalFramesPerSecond, bool preferFrameRatesNear60FPS)
48{
49 if (reasons.contains(ThrottlingReason::OutsideViewport))
50 return std::nullopt;
51
52 if (!nominalFramesPerSecond || *nominalFramesPerSecond == FullSpeedFramesPerSecond) {
53 // FIXME: handle ThrottlingReason::VisuallyIdle
54 if (reasons.containsAny(halfSpeedThrottlingReasons))
55 return HalfSpeedThrottlingFramesPerSecond;
56
57 return FullSpeedFramesPerSecond;
58 }
59
60 auto framesPerSecond = preferFrameRatesNear60FPS ? framesPerSecondNearestFullSpeed(*nominalFramesPerSecond) : *nominalFramesPerSecond;
61 if (reasons.containsAny(halfSpeedThrottlingReasons))
62 framesPerSecond /= IntervalThrottlingFactor;
63
64 return framesPerSecond;
65}
66
67Seconds preferredFrameInterval(OptionSet<ThrottlingReason> reasons, std::optional<FramesPerSecond> nominalFramesPerSecond, bool preferFrameRatesNear60FPS)
68{
69 if (reasons.contains(ThrottlingReason::OutsideViewport))
70 return AggressiveThrottlingAnimationInterval;
71
72 if (!nominalFramesPerSecond || *nominalFramesPerSecond == FullSpeedFramesPerSecond) {
73 // FIXME: handle ThrottlingReason::VisuallyIdle
74 if (reasons.containsAny(halfSpeedThrottlingReasons))
75 return HalfSpeedThrottlingAnimationInterval;
76 return FullSpeedAnimationInterval;
77 }
78
79 auto framesPerSecond = preferFrameRatesNear60FPS ? framesPerSecondNearestFullSpeed(*nominalFramesPerSecond) : *nominalFramesPerSecond;
80 auto interval = Seconds(1.0 / framesPerSecond);
81
82 if (reasons.containsAny(halfSpeedThrottlingReasons))
83 interval *= IntervalThrottlingFactor;
84
85 return interval;
86}
87
88FramesPerSecond preferredFramesPerSecondFromInterval(Seconds preferredFrameInterval)
89{
90 if (preferredFrameInterval == FullSpeedAnimationInterval)
91 return FullSpeedFramesPerSecond;
92
93 if (preferredFrameInterval == HalfSpeedThrottlingAnimationInterval)
94 return HalfSpeedThrottlingFramesPerSecond;
95
96 return std::round(1 / preferredFrameInterval.seconds());
97}

99TextStream& operator<<(TextStream& ts, const OptionSet<ThrottlingReason>& reasons)
100{
101 bool didAppend = false;
102
103 for (auto reason : reasons) {
104 if (didAppend)
105 ts << "|";
106 switch (reason) {
107 case ThrottlingReason::VisuallyIdle:
108 ts << "VisuallyIdle";
109 break;
110 case ThrottlingReason::OutsideViewport:
111 ts << "OutsideViewport";
112 break;
113 case ThrottlingReason::LowPowerMode:
114 ts << "LowPowerMode";
115 break;
116 case ThrottlingReason::NonInteractedCrossOriginFrame:
117 ts << "NonInteractiveCrossOriginFrame";
118 break;
119 }
120 didAppend = true;
121 }
122
123 if (reasons.isEmpty())
124 ts << "[Unthrottled]";
125 return ts;
126}
