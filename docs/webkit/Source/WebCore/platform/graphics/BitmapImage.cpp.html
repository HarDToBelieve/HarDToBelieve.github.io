76void BitmapImage::destroyDecodedData(bool destroyAll)
77{
78 LOG(Images, "BitmapImage::%s - %p - url: %s", __FUNCTION__, this, sourceURL().string().utf8().data());
79
80 if (!destroyAll)
81 m_source->destroyDecodedDataBeforeFrame(m_currentFrame);
82 else if (!canDestroyDecodedData())
83 m_source->destroyAllDecodedDataExcludeFrame(m_currentFrame);
84 else {
85 m_source->destroyAllDecodedData();
86 m_currentFrameDecodingStatus = DecodingStatus::Invalid;
87 }
88
89 // There's no need to throw away the decoder unless we're explicitly asked
90 // to destroy all of the frames.
91 if (!destroyAll || m_source->hasAsyncDecodingQueue())
92 m_source->clearFrameBufferCache(m_currentFrame);
93 else
94 m_source->resetData(data());
95
96 invalidatePlatformData();
97}

99void BitmapImage::destroyDecodedDataIfNecessary(bool destroyAll)
100{
101 // If we have decoded frames but there is no encoded data, we shouldn't destroy
102 // the decoded image since we won't be able to reconstruct it later.
103 if (!data() && frameCount())
104 return;
105
106 if (m_source->decodedSize() < LargeAnimationCutoff)
107 return;
108
109 destroyDecodedData(destroyAll);
110}

121void BitmapImage::setCurrentFrameDecodingStatusIfNecessary(DecodingStatus decodingStatus)
122{
123 // When new data is received, m_currentFrameDecodingStatus is set to DecodingStatus::Invalid
124 // to force decoding the frame when it's drawn. m_currentFrameDecodingStatus should not be
125 // changed in this case till draw() is called and sets its value to DecodingStatus::Decoding.
126 if (m_currentFrameDecodingStatus != DecodingStatus::Decoding)
127 return;
128 m_currentFrameDecodingStatus = decodingStatus;
129}

155RefPtr<NativeImage> BitmapImage::preTransformedNativeImageForCurrentFrame(bool respectOrientation, const GraphicsContext* targetContext)
156{
157 auto image = nativeImageForCurrentFrame(targetContext);
158 if (!image)
159 return image;
160
161 auto orientation = respectOrientation ? orientationForCurrentFrame() : ImageOrientation(ImageOrientation::None);
162 auto correctedSize = m_source->densityCorrectedSize(orientation);
163 if (orientation == ImageOrientation::None && !correctedSize)
164 return image;
165
166 auto correctedSizeFloat = correctedSize ? FloatSize(correctedSize.value()) : size();
167 auto buffer = ImageBuffer::create(correctedSizeFloat, RenderingMode::Unaccelerated, 1, DestinationColorSpace::SRGB(), PixelFormat::BGRA8);
168 if (!buffer)
169 return image;
170
171 auto sourceSize = this->sourceSize();
172
173 FloatRect destRect(FloatPoint(), correctedSizeFloat);
174 FloatRect sourceRect(FloatPoint(), sourceSize);
175
176 buffer->context().drawNativeImage(*image, sourceSize, destRect, sourceRect, { orientation });
177 return ImageBuffer::sinkIntoNativeImage(WTFMove(buffer));
178}

216static inline void drawNativeImage(NativeImage& image, GraphicsContext& context, const FloatRect& destRect, const FloatRect& srcRect, const IntSize& srcSize, const ImagePaintingOptions& options)
217{
218 // Subsampling may have given us an image that is smaller than size().
219 IntSize subsampledImageSize = image.size();
220 if (options.orientation().usesWidthAsHeight())
221 subsampledImageSize = subsampledImageSize.transposedSize();
222
223 // srcRect is in the coordinates of the unsubsampled image, so we have to map it to the subsampled image.
224 FloatRect adjustedSrcRect = srcRect;
225 if (subsampledImageSize != srcSize)
226 adjustedSrcRect = mapRect(srcRect, FloatRect({ }, srcSize), FloatRect({ }, subsampledImageSize));
227
228 context.drawNativeImage(image, subsampledImageSize, destRect, adjustedSrcRect, options);
229}

231ImageDrawResult BitmapImage::draw(GraphicsContext& context, const FloatRect& destRect, const FloatRect& requestedSrcRect, const ImagePaintingOptions& options)
232{
233 if (destRect.isEmpty() || requestedSrcRect.isEmpty())
234 return ImageDrawResult::DidNothing;
235
236
237 auto srcRect = requestedSrcRect;
238 auto preferredSize = size();
239 auto srcSize = sourceSize();
240
241 if (preferredSize != srcSize)
242 srcRect.scale(srcSize.width() / preferredSize.width(), srcSize.height() / preferredSize.height());
243
244 auto scaleFactorForDrawing = context.scaleFactorForDrawing(destRect, srcRect);
245 auto sizeForDrawing = expandedIntSize(srcSize * scaleFactorForDrawing);
246 ImageDrawResult result = ImageDrawResult::DidDraw;
247
248 m_currentSubsamplingLevel = m_allowSubsampling ? subsamplingLevelForScaleFactor(context, scaleFactorForDrawing) : SubsamplingLevel::Default;
249 LOG(Images, "BitmapImage::%s - %p - url: %s [subsamplingLevel = %d scaleFactorForDrawing = (%.4f, %.4f)]", __FUNCTION__, this, sourceURL().string().utf8().data(), static_cast<int>(m_currentSubsamplingLevel), scaleFactorForDrawing.width(), scaleFactorForDrawing.height());
250
251 RefPtr<NativeImage> image;
252 if (options.decodingMode() == DecodingMode::Asynchronous) {
253 ASSERT(!canAnimate());
254 ASSERT(!m_currentFrame || m_animationFinished);
255
256 bool frameIsCompatible = frameHasDecodedNativeImageCompatibleWithOptionsAtIndex(m_currentFrame, m_currentSubsamplingLevel, DecodingOptions(sizeForDrawing));
257 bool frameIsBeingDecoded = frameIsBeingDecodedAndIsCompatibleWithOptionsAtIndex(m_currentFrame, DecodingOptions(sizeForDrawing));
258
259 // If the current frame is incomplete, a new request for decoding this frame has to be made even if
260 // it is currently being decoded. New data may have been received since the previous request was made.
261 if ((!frameIsCompatible && !frameIsBeingDecoded) || m_currentFrameDecodingStatus == DecodingStatus::Invalid) {
262 LOG(Images, "BitmapImage::%s - %p - url: %s [requesting large async decoding]", __FUNCTION__, this, sourceURL().string().utf8().data());
263 m_source->requestFrameAsyncDecodingAtIndex(m_currentFrame, m_currentSubsamplingLevel, sizeForDrawing);
264 m_currentFrameDecodingStatus = DecodingStatus::Decoding;
265 }
266
267 if (m_currentFrameDecodingStatus == DecodingStatus::Decoding)
268 result = ImageDrawResult::DidRequestDecoding;
269
270 if (!frameHasDecodedNativeImageCompatibleWithOptionsAtIndex(m_currentFrame, m_currentSubsamplingLevel, DecodingOptions(DecodingMode::Asynchronous))) {
271 if (m_showDebugBackground)
272 fillWithSolidColor(context, destRect, Color::yellow.colorWithAlphaByte(128), options.compositeOperator());
273 return result;
274 }
275
276 image = frameImageAtIndex(m_currentFrame);
277 LOG(Images, "BitmapImage::%s - %p - url: %s [a decoded frame will be used for asynchronous drawing]", __FUNCTION__, this, sourceURL().string().utf8().data());
278 } else {
279 StartAnimationStatus status = internalStartAnimation();
280 ASSERT_IMPLIES(status == StartAnimationStatus::DecodingActive, (!m_currentFrame && !m_repetitionsComplete) || frameHasFullSizeNativeImageAtIndex(m_currentFrame, m_currentSubsamplingLevel));
281
282 if (status == StartAnimationStatus::DecodingActive && m_showDebugBackground) {
283 fillWithSolidColor(context, destRect, Color::yellow.colorWithAlphaByte(128), options.compositeOperator());
284 return result;
285 }
286
287 // If the decodingMode changes from asynchronous to synchronous and new data is received,
288 // the current incomplete decoded frame has to be destroyed.
289 if (m_currentFrameDecodingStatus == DecodingStatus::Invalid)
290 m_source->destroyIncompleteDecodedData();
291
292 bool frameIsCompatible = frameHasDecodedNativeImageCompatibleWithOptionsAtIndex(m_currentFrame, m_currentSubsamplingLevel, DecodingOptions(sizeForDrawing));
293 bool frameIsBeingDecoded = frameIsBeingDecodedAndIsCompatibleWithOptionsAtIndex(m_currentFrame, DecodingOptions(DecodingMode::Asynchronous));
294
295 if (frameIsCompatible) {
296 image = frameImageAtIndex(m_currentFrame);
297 LOG(Images, "BitmapImage::%s - %p - url: %s [a decoded frame will reused for synchronous drawing]", __FUNCTION__, this, sourceURL().string().utf8().data());
298 } else if (frameIsBeingDecoded) {
299 // FIXME: instead of showing the yellow rectangle and returning we need to wait for this frame to finish decoding.
300 if (m_showDebugBackground) {
301 fillWithSolidColor(context, destRect, Color::yellow.colorWithAlphaByte(128), options.compositeOperator());
302 LOG(Images, "BitmapImage::%s - %p - url: %s [waiting for async decoding to finish]", __FUNCTION__, this, sourceURL().string().utf8().data());
303 }
304 return ImageDrawResult::DidRequestDecoding;
305 } else {
306 image = frameImageAtIndexCacheIfNeeded(m_currentFrame, m_currentSubsamplingLevel, &context);
307 LOG(Images, "BitmapImage::%s - %p - url: %s [an image frame will be decoded synchronously]", __FUNCTION__, this, sourceURL().string().utf8().data());
308 }
309
310 if (!image) // If it's too early we won't have an image yet.
311 return ImageDrawResult::DidNothing;
312
313 if (m_currentFrameDecodingStatus != DecodingStatus::Complete)
314 ++m_decodeCountForTesting;
315 }
316
317 ASSERT(image);
318 Color color = singlePixelSolidColor();
319 if (color.isValid()) {
320 fillWithSolidColor(context, destRect, color, options.compositeOperator());
321 return result;
322 }
323
324 auto orientation = options.orientation();
325 if (orientation == ImageOrientation::FromImage) {
326 orientation = frameOrientationAtIndex(m_currentFrame);
327 drawNativeImage(*image, context, destRect, srcRect, IntSize(sourceSize(orientation)), { options, orientation });
328 } else
329 drawNativeImage(*image, context, destRect, srcRect, IntSize(sourceSize(orientation)), options);
330
331 m_currentFrameDecodingStatus = frameDecodingStatusAtIndex(m_currentFrame);
332
333 if (imageObserver())
334 imageObserver()->didDraw(*this);
335
336 return result;
337}

339void BitmapImage::drawPattern(GraphicsContext& ctxt, const FloatRect& destRect, const FloatRect& tileRect, const AffineTransform& transform, const FloatPoint& phase, const FloatSize& spacing, const ImagePaintingOptions& options)
340{
341 if (tileRect.isEmpty())
342 return;
343
344 if (!ctxt.drawLuminanceMask()) {
345 // If new data is received, the current incomplete decoded frame has to be destroyed.
346 if (m_currentFrameDecodingStatus == DecodingStatus::Invalid)
347 m_source->destroyIncompleteDecodedData();
348
349 Image::drawPattern(ctxt, destRect, tileRect, transform, phase, spacing, { options, ImageOrientation::FromImage });
350 m_currentFrameDecodingStatus = frameDecodingStatusAtIndex(m_currentFrame);
351 return;
352 }
353
354 if (!m_cachedImage) {
355 auto buffer = ImageBuffer::createCompatibleBuffer(expandedIntSize(tileRect.size()), DestinationColorSpace::SRGB(), ctxt);
356 if (!buffer)
357 return;
358
359 ImageObserver* observer = imageObserver();
360
361 // Temporarily reset image observer, we don't want to receive any changeInRect() calls due to this relayout.
362 setImageObserver(nullptr);
363
364 draw(buffer->context(), tileRect, tileRect, { options, DecodingMode::Synchronous, ImageOrientation::FromImage });
365
366 setImageObserver(observer);
367 buffer->convertToLuminanceMask();
368
369 m_cachedImage = ImageBuffer::sinkIntoImage(WTFMove(buffer), PreserveResolution::Yes);
370 if (!m_cachedImage)
371 return;
372 }
373
374 ctxt.setDrawLuminanceMask(false);
375 m_cachedImage->drawPattern(ctxt, destRect, tileRect, transform, phase, spacing, { options, ImageOrientation::FromImage });
376}

410SubsamplingLevel BitmapImage::subsamplingLevelForScaleFactor(GraphicsContext& context, const FloatSize& scaleFactor)
411{
412#if USE(CG)
413 // Never use subsampled images for drawing into PDF contexts.
414 if (context.hasPlatformContext() && CGContextGetType(context.platformContext()) == kCGContextTypePDF)
415 return SubsamplingLevel::Default;
416
417 float scale = std::min(float(1), std::max(scaleFactor.width(), scaleFactor.height()));
418 if (!(scale > 0 && scale <= 1))
419 return SubsamplingLevel::Default;
420
421 int result = std::ceil(std::log2(1 / scale));
422 return static_cast<SubsamplingLevel>(std::min(result, static_cast<int>(m_source->maximumSubsamplingLevel())));
423#else
424 UNUSED_PARAM(context);
425 UNUSED_PARAM(scaleFactor);
426 return SubsamplingLevel::Default;
427#endif
428}

430bool BitmapImage::canDestroyDecodedData()
431{
432 // Animated images should preserve the current frame till the next one finishes decoding.
433 if (m_source->hasAsyncDecodingQueue())
434 return false;
435
436 // Small image should be decoded synchronously. Deleting its decoded frame is fine.
437 if (!canUseAsyncDecodingForLargeImages())
438 return true;
439
440 return !imageObserver() || imageObserver()->canDestroyDecodedData(*this);
441}

443BitmapImage::StartAnimationStatus BitmapImage::internalStartAnimation()
444{
445 LOG_WITH_STREAM(Images, stream << "BitmapImage " << this << " internalStartAnimation");
446
447 if (!canAnimate())
448 return StartAnimationStatus::CannotStart;
449
450 if (m_frameTimer)
451 return StartAnimationStatus::TimerActive;
452
453 // Don't start a new animation until we draw the frame that is currently being decoded.
454 size_t nextFrame = (m_currentFrame + 1) % frameCount();
455 if (frameIsBeingDecodedAndIsCompatibleWithOptionsAtIndex(nextFrame, DecodingOptions(DecodingMode::Asynchronous))) {
456 LOG(Images, "BitmapImage::%s - %p - url: %s [nextFrame = %ld is being decoded]", __FUNCTION__, this, sourceURL().string().utf8().data(), nextFrame);
457 return StartAnimationStatus::DecodingActive;
458 }
459
460 if (m_currentFrame >= frameCount() - 1) {
461 // Don't advance past the last frame if we haven't decoded the whole image
462 // yet and our repetition count is potentially unset. The repetition count
463 // in a GIF can potentially come after all the rest of the image data, so
464 // wait on it.
465 if (!m_source->isAllDataReceived() && repetitionCount() == RepetitionCountOnce)
466 return StartAnimationStatus::IncompleteData;
467
468 ++m_repetitionsComplete;
469
470 // Check for the end of animation.
471 if (repetitionCount() != RepetitionCountInfinite && m_repetitionsComplete >= repetitionCount()) {
472 m_animationFinished = true;
473 destroyDecodedDataIfNecessary(false);
474 return StartAnimationStatus::CannotStart;
475 }
476
477 destroyDecodedDataIfNecessary(true);
478 }
479
480 // Don't advance the animation to an incomplete frame.
481 if (!m_source->isAllDataReceived() && !frameIsCompleteAtIndex(nextFrame))
482 return StartAnimationStatus::IncompleteData;
483
484 MonotonicTime time = MonotonicTime::now();
485
486 // Handle initial state.
487 if (!m_desiredFrameStartTime)
488 m_desiredFrameStartTime = time;
489
490 // Setting 'm_desiredFrameStartTime' to 'time' means we are late; otherwise we are early.
491 m_desiredFrameStartTime = std::max(time, m_desiredFrameStartTime + Seconds { frameDurationAtIndex(m_currentFrame) });
492
493 // Request async decoding for nextFrame only if this is required. If nextFrame is not in the frameCache,
494 // it will be decoded on a separate work queue. When decoding nextFrame finishes, we will be notified
495 // through the callback newFrameNativeImageAvailableAtIndex(). Otherwise, advanceAnimation() will be called
496 // when the timer fires and m_currentFrame will be advanced to nextFrame since it is not being decoded.
497 if (shouldUseAsyncDecodingForAnimatedImages()) {
498 if (frameHasDecodedNativeImageCompatibleWithOptionsAtIndex(nextFrame, m_currentSubsamplingLevel, DecodingOptions(std::optional<IntSize>())))
499 LOG(Images, "BitmapImage::%s - %p - url: %s [cachedFrameCount = %ld nextFrame = %ld]", __FUNCTION__, this, sourceURL().string().utf8().data(), ++m_cachedFrameCount, nextFrame);
500 else {
501 m_source->requestFrameAsyncDecodingAtIndex(nextFrame, m_currentSubsamplingLevel);
502 m_currentFrameDecodingStatus = DecodingStatus::Decoding;
503 LOG(Images, "BitmapImage::%s - %p - url: %s [requesting async decoding for nextFrame = %ld]", __FUNCTION__, this, sourceURL().string().utf8().data(), nextFrame);
504 }
505
506 if (m_clearDecoderAfterAsyncFrameRequestForTesting)
507 m_source->resetData(data());
508 }
509
510 ASSERT(!m_frameTimer);
511 startTimer(m_desiredFrameStartTime - time);
512 return StartAnimationStatus::Started;
513}
514
515void BitmapImage::advanceAnimation()
516{
517 clearTimer();
518
519 // Don't advance to nextFrame unless its decoding has finished or was not required.
520 size_t nextFrame = (m_currentFrame + 1) % frameCount();
521 if (!frameIsBeingDecodedAndIsCompatibleWithOptionsAtIndex(nextFrame, DecodingOptions(DecodingMode::Asynchronous)))
522 internalAdvanceAnimation();
523 else {
524 // Force repaint if showDebugBackground() is on.
525 if (m_showDebugBackground)
526 imageObserver()->changedInRect(*this);
527 LOG(Images, "BitmapImage::%s - %p - url: %s [lateFrameCount = %ld nextFrame = %ld]", __FUNCTION__, this, sourceURL().string().utf8().data(), ++m_lateFrameCount, nextFrame);
528 }
529}
530
531void BitmapImage::internalAdvanceAnimation()
532{
533 m_currentFrame = (m_currentFrame + 1) % frameCount();
534 ASSERT(!frameIsBeingDecodedAndIsCompatibleWithOptionsAtIndex(m_currentFrame, DecodingOptions(DecodingMode::Asynchronous)));
535
536 destroyDecodedDataIfNecessary(false);
537
538 DecodingStatus decodingStatus = frameDecodingStatusAtIndex(m_currentFrame);
539 setCurrentFrameDecodingStatusIfNecessary(decodingStatus);
540
541 callDecodingCallbacks();
542
543 if (imageObserver())
544 imageObserver()->imageFrameAvailable(*this, ImageAnimatingState::Yes, nullptr, decodingStatus);
545
546 LOG(Images, "BitmapImage::%s - %p - url: %s [m_currentFrame = %ld]", __FUNCTION__, this, sourceURL().string().utf8().data(), m_currentFrame);
547}

554void BitmapImage::stopAnimation()
555{
556 // This timer is used to animate all occurrences of this image. Don't invalidate
557 // the timer unless all renderers have stopped drawing.
558 clearTimer();
559 if (canAnimate())
560 m_source->stopAsyncDecodingQueue();
561}

563void BitmapImage::resetAnimation()
564{
565 stopAnimation();
566 m_currentFrame = 0;
567 m_repetitionsComplete = RepetitionCountNone;
568 m_desiredFrameStartTime = { };
569 m_animationFinished = false;
570
571 // For extremely large animations, when the animation is reset, we just throw everything away.
572 destroyDecodedDataIfNecessary(true);
573}

575void BitmapImage::decode(WTF::Function<void()>&& callback)
576{
577 if (!m_decodingCallbacks)
578 m_decodingCallbacks = makeUnique<Vector<Function<void()>, 1>>();
579
580 m_decodingCallbacks->append(WTFMove(callback));
581
582 if (canAnimate()) {
583 if (m_desiredFrameStartTime) {
584 internalStartAnimation();
585 return;
586 }
587
588 // The animated image has not been displayed. In this case, either the first frame has not been decoded yet or the animation has not started yet.
589 bool frameIsCompatible = frameHasDecodedNativeImageCompatibleWithOptionsAtIndex(m_currentFrame, m_currentSubsamplingLevel, std::optional<IntSize>());
590 bool frameIsBeingDecoded = frameIsBeingDecodedAndIsCompatibleWithOptionsAtIndex(m_currentFrame, std::optional<IntSize>());
591
592 if (frameIsCompatible)
593 internalStartAnimation();
594 else if (!frameIsBeingDecoded) {
595 m_source->requestFrameAsyncDecodingAtIndex(m_currentFrame, m_currentSubsamplingLevel, std::optional<IntSize>());
596 m_currentFrameDecodingStatus = DecodingStatus::Decoding;
597 }
598 return;
599 }
600
601 bool frameIsCompatible = frameHasDecodedNativeImageCompatibleWithOptionsAtIndex(m_currentFrame, m_currentSubsamplingLevel, std::optional<IntSize>());
602 bool frameIsBeingDecoded = frameIsBeingDecodedAndIsCompatibleWithOptionsAtIndex(m_currentFrame, std::optional<IntSize>());
603
604 if (frameIsCompatible)
605 callDecodingCallbacks();
606 else if (!frameIsBeingDecoded) {
607 m_source->requestFrameAsyncDecodingAtIndex(m_currentFrame, m_currentSubsamplingLevel, std::optional<IntSize>());
608 m_currentFrameDecodingStatus = DecodingStatus::Decoding;
609 }
610}
611
612void BitmapImage::callDecodingCallbacks()
613{
614 if (!m_decodingCallbacks)
615 return;
616 for (auto& decodingCallback : *m_decodingCallbacks)
617 decodingCallback();
618 m_decodingCallbacks = nullptr;
619}

621void BitmapImage::imageFrameAvailableAtIndex(size_t index)
622{
623 LOG(Images, "BitmapImage::%s - %p - url: %s [requested frame %ld is now available]", __FUNCTION__, this, sourceURL().string().utf8().data(), index);
624
625 if (canAnimate()) {
626 if (index == (m_currentFrame + 1) % frameCount()) {
627 // Don't advance to nextFrame unless the timer was fired before its decoding finishes.
628 if (!m_frameTimer)
629 internalAdvanceAnimation();
630 else
631 LOG(Images, "BitmapImage::%s - %p - url: %s [earlyFrameCount = %ld nextFrame = %ld]", __FUNCTION__, this, sourceURL().string().utf8().data(), ++m_earlyFrameCount, index);
632 return;
633 }
634
635 // Because of image partial loading, an image may start decoding as a large static image. But
636 // when more data is received, frameCount() changes to be > 1 so the image starts animating.
637 // The animation may even start before finishing the decoding of the first frame.
638 ASSERT(!m_repetitionsComplete);
639 LOG(Images, "BitmapImage::%s - %p - url: %s [More data makes frameCount() > 1]", __FUNCTION__, this, sourceURL().string().utf8().data());
640 }
641
642 ASSERT(index == m_currentFrame && !m_currentFrame);
643 if (m_source->isAsyncDecodingQueueIdle())
644 m_source->stopAsyncDecodingQueue();
645
646 DecodingStatus decodingStatus = frameDecodingStatusAtIndex(m_currentFrame);
647 setCurrentFrameDecodingStatusIfNecessary(decodingStatus);
648
649 if (m_currentFrameDecodingStatus == DecodingStatus::Complete)
650 ++m_decodeCountForTesting;
651
652 // Call m_decodingCallbacks only if the image frame was decoded with the native size.
653 if (frameHasDecodedNativeImageCompatibleWithOptionsAtIndex(m_currentFrame, m_currentSubsamplingLevel, std::optional<IntSize>()))
654 callDecodingCallbacks();
655
656 if (imageObserver())
657 imageObserver()->imageFrameAvailable(*this, ImageAnimatingState::No, nullptr, decodingStatus);
658}

661{
662 return m_decodeCountForTesting;
663}
