161 // Called to invalidate cached data. When |destroyAll| is true, we wipe out
162 // the entire frame buffer cache and tell the image source to destroy
163 // everything; this is used when e.g. we want to free some room in the image
164 // cache. If |destroyAll| is false, we only delete frames up to the current
165 // one; this is used while animating large images to keep memory footprint
166 // low without redecoding the whole image on every frame.
167 void destroyDecodedData(bool destroyAll = true) override;
168
169 // If the image is large enough, calls destroyDecodedData() and passes
170 // |destroyAll| along.
171 void destroyDecodedDataIfNecessary(bool destroyAll = true);
