33namespace WebCore {

35template<typename T, size_t N>
36struct ColorComponents {
37 constexpr static size_t Size = N;
38
39 constexpr ColorComponents()
40 : components { }
41 {
42 }
43
44 template<typename ...Ts>
45 constexpr ColorComponents(Ts ...input)
46 : components {{ input ... }}
47 {
48 static_assert(sizeof...(Ts) == N);
49 }
50
51 template<typename F>
52 constexpr auto map(F&& function) const -> ColorComponents<decltype(function(std::declval<T>())), N>;
53
54 constexpr ColorComponents& operator+=(const ColorComponents&);
55
56 constexpr ColorComponents operator+(T) const;
57 constexpr ColorComponents operator/(T) const;
58 constexpr ColorComponents operator*(T) const;
59
60 constexpr ColorComponents abs() const;
61
62 constexpr T& operator[](size_t i) { return components[i]; }
63 constexpr const T& operator[](size_t i) const { return components[i]; }
64
65 template<size_t I>
66 constexpr T get() const;
67
68 template<size_t Start, size_t End>
69 constexpr ColorComponents<T, End - Start> subset() const;
70
71 std::array<T, N> components;
72};
73
74template<typename T, typename ...Ts>
75ColorComponents(T, Ts...) -> ColorComponents<T, 1 + sizeof...(Ts)>;
76
77template<typename F, typename T, typename... Ts>
78constexpr auto mapColorComponents(F&& function, T component, Ts... components) -> ColorComponents<decltype(function(component[0], components[0]...)), T::Size>
79{
80 static_assert(std::conjunction_v<std::bool_constant<Ts::Size == T::Size>...>, "All ColorComponents passed to mapColorComponents must have the same size");
81
82 ColorComponents<decltype(function(component[0], components[0]...)), T::Size> result;
83 for (std::remove_const_t<decltype(T::Size)> i = 0; i < T::Size; ++i)
84 result[i] = function(component[i], components[i]...);
85 return result;
86}
87
88template<typename T, size_t N>
89template<typename F>
90constexpr auto ColorComponents<T, N>::map(F&& function) const -> ColorComponents<decltype(function(std::declval<T>())), N>
91{
92 return mapColorComponents(std::forward<F>(function), *this);
93}
94
95template<typename T, size_t N>
96constexpr ColorComponents<T, N>& ColorComponents<T, N>::operator+=(const ColorComponents& rhs)
97{
98 *this = mapColorComponents([](T c1, T c2) { return c1 + c2; }, *this, rhs);
99 return *this;
100}
101
102template<typename T, size_t N>
103constexpr ColorComponents<T, N> ColorComponents<T, N>::operator+(T rhs) const
104{
105 return map([rhs](T c) { return c + rhs; });
106}
107
108template<typename T, size_t N>
109constexpr ColorComponents<T, N> ColorComponents<T, N>::operator/(T denominator) const
110{
111 return map([denominator](T c) { return c / denominator; });
112}
113
114template<typename T, size_t N>
115constexpr ColorComponents<T, N> ColorComponents<T, N>::operator*(T factor) const
116{
117 return map([factor](T c) { return c * factor; });
118}
119
120template<typename T, size_t N>
121constexpr ColorComponents<T, N> ColorComponents<T, N>::abs() const
122{
123 return map([](T c) { return std::abs(c); });
124}
125
126template<typename T, size_t N>
127template<size_t I>
128constexpr T ColorComponents<T, N>::get() const
129{
130 return components[I];
131}
132
133template<typename T, size_t N>
134template<size_t Start, size_t End>
135constexpr ColorComponents<T, End - Start> ColorComponents<T, N>::subset() const
136{
137 ColorComponents<T, End - Start> result;
138 for (std::remove_const_t<decltype(T::Size)> i = Start; i < End; ++i)
139 result[i - Start] = components[i];
140 return result;
141}
142
143template<typename T, size_t N>
144constexpr ColorComponents<T, N> perComponentMax(const ColorComponents<T, N>& a, const ColorComponents<T, N>& b)
145{
146 return mapColorComponents([](T c1, T c2) { return std::max(c1, c2); }, a, b);
147}
148
149template<typename T, size_t N>
150constexpr ColorComponents<T, N> perComponentMin(const ColorComponents<T, N>& a, const ColorComponents<T, N>& b)
151{
152 return mapColorComponents([](T c1, T c2) { return std::min(c1, c2); }, a, b);
153}
154
155template<typename T, size_t N>
156constexpr bool operator==(const ColorComponents<T, N>& a, const ColorComponents<T, N>& b)
157{
158 return a.components == b.components;
159}
160
161template<typename T, size_t N>
162constexpr bool operator!=(const ColorComponents<T, N>& a, const ColorComponents<T, N>& b)
163{
164 return !(a == b);
165}
