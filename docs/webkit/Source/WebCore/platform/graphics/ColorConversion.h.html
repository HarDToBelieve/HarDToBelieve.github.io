44// All color types, other than XYZA or those inheriting from RGBType, must implement
45// the following conversions to and from their "Reference" color.
46//
47// template<> struct ColorConversion<`ColorType`<float>::Reference, `ColorType`<float>> {
48// WEBCORE_EXPORT static `ColorType`<float>::Reference convert(const `ColorType`<float>&);
49// };
50//
51// template<> struct ColorConversion<`ColorType`<float>, `ColorType`<float>::Reference> {
52// WEBCORE_EXPORT static `ColorType`<float> convert(const `ColorType`<float>::Reference&);
53// };
54//
55
56template<typename Output, typename Input, typename = void> struct ColorConversion;
57
58template<typename Output, typename Input> inline Output convertColor(const Input& color)
59{
60 return ColorConversion<Output, Input>::convert(color);
61}

131// │ │ XYZ (D50) │││ XYZ (D65) │ │
132// └─────▲─────┘│└─────▲─────┘
133// │ │ │ │ │
134// ┌─────────────────────────┬───────────┘ │ └───────────┬───────────────────────────────┬───────────────────────────────┬───────────────────────────────┐

136// │ │ │ │ │ │ │

139// │ │ │┌────────┐ ┌────────────────┐│ │ │┌────────┐ ┌────────────────┐│ │┌────────┐ ┌────────────────┐│ │┌────────┐ ┌────────────────┐│ │┌────────┐ ┌────────────────┐│ │

142// │ ─│─ ─ ─ ─ ─ ─ ─ ─ ─ ─ ─ ─ ─ ─ ─│─ ─│─ ─│─ ─ ─ ─ ─ ─ ─ ─ ─ ─ ─ ─ ─ ─ ─│─│─ ─ ─ ─ ─ ─ ─ ─ ─ ─ ─ ─ ─ ─ ─│─│─ ─ ─ ─ ─ ─ ─ ─ ─ ─ ─ ─ ─ ─ ─│─│─ ─ ─ ─ ─ ─ ─ ─ ─ ─ ─ ─ ─ ─ ─│─

147// │ │ ┌──┴──────────┐
148// │ │ │ │
149// ┌───────────┐ │┌───────────┐ ┌───────────┐
150// │ LCH │ ││ HSL │ │ HWB │
151// └───────────┘ │└───────────┘ └───────────┘

153template<typename Output, typename Input, typename> struct ColorConversion {
154public:
155 static constexpr Output convert(const Input& color)
156 {
157 // 1. Handle the special case of Input or Output with a uint8_t component type.
158 if constexpr (std::is_same_v<typename Input::ComponentType, uint8_t>)
159 return handleToFloatConversion(color);
160 else if constexpr (std::is_same_v<typename Output::ComponentType, uint8_t>)
161 return handleToByteConversion(color);
162
163 // 2. Handle all color types that are not IsRGBType<T> or IsXYZA<T> for Input and Output. For all
164 // these other color types, we can uncondtionally convert them to their "reference" color, as
165 // either they have already been handled by a ColorConversion specialization or this will
166 // get us closer to the final conversion.
167 else if constexpr (!IsRGBType<Input> && !IsXYZA<Input>)
168 return convertColor<Output>(convertColor<typename Input::Reference>(color));
169 else if constexpr (!IsRGBType<Output> && !IsXYZA<Output>)
170 return convertColor<Output>(convertColor<typename Output::Reference>(color));
171
172 // 3. Handle conversions within a RGBFamily (e.g. all have the same descriptor).
173 else if constexpr (IsSameRGBTypeFamily<Output, Input>)
174 return handleRGBFamilyConversion(color);
175
176 // 4. Handle any gamma conversions for the Input and Output.
177 else if constexpr (IsRGBGammaEncodedType<Input>)
178 return convertColor<Output>(convertColor<typename Input::LinearCounterpart>(color));
179 else if constexpr (IsRGBGammaEncodedType<Output>)
180 return convertColor<Output>(convertColor<typename Output::LinearCounterpart>(color));
181
182 // 5. At this point, Input and Output are each either Linear-RGB types (of different familes) or XYZA
183 // and therefore all additional conversion can happen via matrix transformation.
184 else
185 return handleMatrixConversion(color);
186 }
187
188private:
189 static inline constexpr Output handleToFloatConversion(const Input& color)
190 {
191 static_assert(IsRGBBoundedType<Input>, "Only bounded ([0..1]) RGB color types support conversion to/from bytes.");
192
193 using InputWithReplacement = ColorTypeWithReplacementComponent<Input, float>;
194 if constexpr (std::is_same_v<InputWithReplacement, Output>)
195 return makeFromComponents<InputWithReplacement>(asColorComponents(color).map([](uint8_t value) -> float { return value / 255.0f; }));
196 else
197 return convertColor<Output>(convertColor<InputWithReplacement>(color));
198 }
199
200 static inline constexpr Output handleToByteConversion(const Input& color)
201 {
202 static_assert(IsRGBBoundedType<Output>, "Only bounded ([0..1]) RGB color types support conversion to/from bytes.");
203
204 using OutputWithReplacement = ColorTypeWithReplacementComponent<Output, float>;
205 if constexpr (std::is_same_v<OutputWithReplacement, Input>)
206 return makeFromComponents<Output>(asColorComponents(color).map([](float value) -> uint8_t { return std::clamp(std::lround(value * 255.0f), 0l, 255l); }));
207 else
208 return convertColor<Output>(convertColor<OutputWithReplacement>(color));
209 }

211 template<typename ColorType> static inline constexpr auto toLinearEncoded(const ColorType& color) -> typename ColorType::LinearCounterpart
212 {
213 auto [c1, c2, c3, alpha] = color;
214 return { ColorType::TransferFunction::toLinear(c1), ColorType::TransferFunction::toLinear(c2), ColorType::TransferFunction::toLinear(c3), alpha };
215 }
216
217 template<typename ColorType> static inline constexpr auto toGammaEncoded(const ColorType& color) -> typename ColorType::GammaEncodedCounterpart
218 {
219 auto [c1, c2, c3, alpha] = color;
220 return { ColorType::TransferFunction::toGammaEncoded(c1), ColorType::TransferFunction::toGammaEncoded(c2), ColorType::TransferFunction::toGammaEncoded(c3), alpha };
221 }
222
223 template<typename ColorType> static inline constexpr auto toExtended(const ColorType& color) -> typename ColorType::ExtendedCounterpart
224 {
225 return makeFromComponents<typename ColorType::ExtendedCounterpart>(asColorComponents(color));
226 }
227
228 template<typename ColorType> static inline constexpr auto toBounded(const ColorType& color) -> typename ColorType::BoundedCounterpart
229 {
230 return makeFromComponentsClampingExceptAlpha<typename ColorType::BoundedCounterpart>(asColorComponents(color));
231 }
232
233 static inline constexpr Output handleRGBFamilyConversion(const Input& color)
234 {
235 static_assert(IsSameRGBTypeFamily<Output, Input>);
236
237 // RGB Family────────────────────┐
238 // │┌────────┐ ┌────────────────┐│
239 // ││ Linear │ │ LinearExtended ││
240 // │└────────┘ └────────────────┘│
241 // │┌────────┐ ┌────────────────┐│
242 // ││ Gamma │ │ GammaExtended ││
243 // │└────────┘ └────────────────┘│
244 // └─────────────────────────────┘
245
246 // This handles conversions between any two of these within the same family, so SRGBLinear -> SRGB, but not
247 // SRGBLinear -> A98RGB.
248
249 auto boundsConversion = [](auto color) {
250 if constexpr (IsRGBExtendedType<Output> && IsRGBBoundedType<Input>)
251 return toExtended(color);
252 else if constexpr (IsRGBBoundedType<Output> && IsRGBExtendedType<Input>)
253 return toBounded(color);
254 else
255 return color;
256 };
257
258 auto gammaConversion = [](auto color) {
259 if constexpr (IsRGBGammaEncodedType<Output> && IsRGBLinearEncodedType<Input>)
260 return toGammaEncoded(color);
261 else if constexpr (IsRGBLinearEncodedType<Output> && IsRGBGammaEncodedType<Input>)
262 return toLinearEncoded(color);
263 else
264 return color;
265 };
266
267 return boundsConversion(gammaConversion(color));
268 }
269
270 static inline constexpr Output handleMatrixConversion(const Input& color)
271 {
272 static_assert(IsRGBLinearEncodedType<Input> || IsXYZA<Input>);
273 static_assert(IsRGBLinearEncodedType<Output> || IsXYZA<Output>);
274
275 // ┌ ─ ─ ─ ─ ─ ─ ─ ─ ─ ─ ─ ─ ─ ─ ─ ─ ─ ┼ ─ ─ ─ ─ ─ ─ ─ ─ ─ ─ ─ ─ ─ ─ ─ ─ ─ ─ ─ ─ ─ ─ ─ ─ ─ ─ ─ ─ ─ ─ ─ ─ ─ ─ ─ ─ ─ ─ ─ ─ ─ ─ ─ ─ ─ ─ ─ ─ ─ ─ ─ ─ ─ ─ ─ ─ ─ ─ ─ ─ ─ ─ ─ ─ ─ ┐
276 // Matrix Conversions ┌───────────┐│┌───────────┐
277 // │ │ XYZ (D50) │││ XYZ (D65) │ │
278 // └─────▲─────┘│└─────▲─────┘
279 // │ │ │ │ │
280 // ┌───────────┘ │ └───────────┬───────────────────────────────┬───────────────────────────────┬───────────────────────────────┐
281 // │ │ │ │ │ │ │ │
282 // │ │ │ │ │ │
283 // │ │ │ │ │ │ │ │
284 // ProPhotoRGB───────────────────┐ │ SRGB──────────────────────────┐ DisplayP3─────────────────────┐ A98RGB────────────────────────┐ Rec2020───────────────────────┐
285 // │ │┌────────┐ ┌────────────────┐│ │ │┌────────┐ ┌────────────────┐│ │┌────────┐ ┌────────────────┐│ │┌────────┐ ┌────────────────┐│ │┌────────┐ ┌────────────────┐│ │
286 // ││ Linear │ │ LinearExtended ││ │ ││ Linear │ │ LinearExtended ││ ││ Linear │ │ LinearExtended ││ ││ Linear │ │ LinearExtended ││ ││ Linear │ │ LinearExtended ││
287 // │ │└────────┘ └────────────────┘│ │ │└────────┘ └────────────────┘│ │└────────┘ └────────────────┘│ │└────────┘ └────────────────┘│ │└────────┘ └────────────────┘│ │
288 // ─│─ ─ ─ ─ ─ ─ ─ ─ ─ ─ ─ ─ ─ ─ ─│─ ─│─ ─│─ ─ ─ ─ ─ ─ ─ ─ ─ ─ ─ ─ ─ ─ ─│─│─ ─ ─ ─ ─ ─ ─ ─ ─ ─ ─ ─ ─ ─ ─│─│─ ─ ─ ─ ─ ─ ─ ─ ─ ─ ─ ─ ─ ─ ─│─│─ ─ ─ ─ ─ ─ ─ ─ ─ ─ ─ ─ ─ ─ ─│─
289
290 // This handles conversions between linear color types that can be converted using pre-defined
291 // 3x3 matrices.
292
293 // FIXME: Pre-compute (using constexpr) the concatenation of the matrices prior to applying them
294 // to reduce number of matrix multiplications to a minimum. This will likely give subtly different
295 // results (due to floating point effects) so if this optimization is considered we should ensure we
296 // have sufficient testing coverage to notice any adverse effects.
297
298 auto applyMatrices = [](const Input& color, auto... matrices) {
299 return makeFromComponentsClampingExceptAlpha<Output>(applyMatricesToColorComponents(asColorComponents(color), matrices...));
300 };
301
302 if constexpr (Input::whitePoint == Output::whitePoint) {
303 if constexpr (IsXYZA<Input>)
304 return applyMatrices(color, Output::xyzToLinear);
305 else if constexpr (IsXYZA<Output>)
306 return applyMatrices(color, Input::linearToXYZ);
307 else
308 return applyMatrices(color, Input::linearToXYZ, Output::xyzToLinear);
309 } else {
310 if constexpr (IsXYZA<Input> && IsXYZA<Output>)
311 return applyMatrices(color, ChromaticAdapation<Input::whitePoint, Output::whitePoint>::matrix);
312 else if constexpr (IsXYZA<Input>)
313 return applyMatrices(color, ChromaticAdapation<Input::whitePoint, Output::whitePoint>::matrix, Output::xyzToLinear);
314 else if constexpr (IsXYZA<Output>)
315 return applyMatrices(color, Input::linearToXYZ, ChromaticAdapation<Input::whitePoint, Output::whitePoint>::matrix);
316 else
317 return applyMatrices(color, Input::linearToXYZ, ChromaticAdapation<Input::whitePoint, Output::whitePoint>::matrix, Output::xyzToLinear);
318 }
319 }
