26#include "config.h"
27#include "ColorSpace.h"
28
29#include <wtf/text/TextStream.h>
30
31namespace WebCore {
32
33TextStream& operator<<(TextStream& ts, ColorSpace colorSpace)
34{
35 switch (colorSpace) {
36 case ColorSpace::A98RGB:
37 ts << "a98-rgb";
38 break;
39 case ColorSpace::DisplayP3:
40 ts << "DisplayP3";
41 break;
42 case ColorSpace::LCH:
43 ts << "LCH";
44 break;
45 case ColorSpace::Lab:
46 ts << "L*a*b";
47 break;
48 case ColorSpace::LinearSRGB:
49 ts << "LinearSRGB";
50 break;
51 case ColorSpace::ProPhotoRGB:
52 ts << "ProPhotoRGB";
53 break;
54 case ColorSpace::Rec2020:
55 ts << "Rec2020";
56 break;
57 case ColorSpace::SRGB:
58 ts << "sRGB";
59 break;
60 case ColorSpace::XYZ_D50:
61 ts << "XYZ";
62 break;
63 }
64 return ts;
65}
