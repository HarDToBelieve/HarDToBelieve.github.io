33namespace WebCore {

35enum class ColorSpace : uint8_t {
36 A98RGB,
37 DisplayP3,
38 LCH,
39 Lab,
40 LinearSRGB,
41 ProPhotoRGB,
42 Rec2020,
43 SRGB,
44 XYZ_D50,
45};
46
47WEBCORE_EXPORT TextStream& operator<<(TextStream&, ColorSpace);
48
49
50template<typename> struct ColorSpaceMapping;
51template<typename T> struct ColorSpaceMapping<A98RGB<T>> { static constexpr auto colorSpace { ColorSpace::A98RGB }; };
52template<typename T> struct ColorSpaceMapping<DisplayP3<T>> { static constexpr auto colorSpace { ColorSpace::DisplayP3 }; };
53template<typename T> struct ColorSpaceMapping<LCHA<T>> { static constexpr auto colorSpace { ColorSpace::LCH }; };
54template<typename T> struct ColorSpaceMapping<Lab<T>> { static constexpr auto colorSpace { ColorSpace::Lab }; };
55template<typename T> struct ColorSpaceMapping<LinearSRGBA<T>> { static constexpr auto colorSpace { ColorSpace::LinearSRGB }; };
56template<typename T> struct ColorSpaceMapping<ProPhotoRGB<T>> { static constexpr auto colorSpace { ColorSpace::ProPhotoRGB }; };
57template<typename T> struct ColorSpaceMapping<Rec2020<T>> { static constexpr auto colorSpace { ColorSpace::Rec2020 }; };
58template<typename T> struct ColorSpaceMapping<SRGBA<T>> { static constexpr auto colorSpace { ColorSpace::SRGB }; };
59template<typename T> struct ColorSpaceMapping<XYZA<T, WhitePoint::D50>> { static constexpr auto colorSpace { ColorSpace::XYZ_D50 }; };
60
61template<typename ColorType> constexpr ColorSpace ColorSpaceFor = ColorSpaceMapping<ColorType>::colorSpace;
62
63
64template<typename T, typename Functor> constexpr decltype(auto) callWithColorType(const ColorComponents<T, 4>& components, ColorSpace colorSpace, Functor&& functor)
65{
66 switch (colorSpace) {
67 case ColorSpace::A98RGB:
68 return std::invoke(std::forward<Functor>(functor), makeFromComponents<A98RGB<T>>(components));
69 case ColorSpace::DisplayP3:
70 return std::invoke(std::forward<Functor>(functor), makeFromComponents<DisplayP3<T>>(components));
71 case ColorSpace::LCH:
72 return std::invoke(std::forward<Functor>(functor), makeFromComponents<LCHA<T>>(components));
73 case ColorSpace::Lab:
74 return std::invoke(std::forward<Functor>(functor), makeFromComponents<Lab<T>>(components));
75 case ColorSpace::LinearSRGB:
76 return std::invoke(std::forward<Functor>(functor), makeFromComponents<LinearSRGBA<T>>(components));
77 case ColorSpace::ProPhotoRGB:
78 return std::invoke(std::forward<Functor>(functor), makeFromComponents<ProPhotoRGB<T>>(components));
79 case ColorSpace::Rec2020:
80 return std::invoke(std::forward<Functor>(functor), makeFromComponents<Rec2020<T>>(components));
81 case ColorSpace::SRGB:
82 return std::invoke(std::forward<Functor>(functor), makeFromComponents<SRGBA<T>>(components));
83 case ColorSpace::XYZ_D50:
84 return std::invoke(std::forward<Functor>(functor), makeFromComponents<XYZA<T, WhitePoint::D50>>(components));
85 }
86
87 ASSERT_NOT_REACHED();
88 return std::invoke(std::forward<Functor>(functor), makeFromComponents<SRGBA<T>>(components));
89}
