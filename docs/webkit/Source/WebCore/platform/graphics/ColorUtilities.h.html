96template<typename ColorType, typename Functor> ColorType colorByModifingEachNonAlphaComponent(const ColorType& color, Functor&& functor)
97{
98 auto components = asColorComponents(color);
99 auto copy = components;
100 copy[0] = std::invoke(functor, components[0]);
101 copy[1] = std::invoke(functor, components[1]);
102 copy[2] = std::invoke(std::forward<Functor>(functor), components[2]);
103 return { copy[0], copy[1], copy[2], copy[3] };
104}

120template<typename ColorType> constexpr ColorType invertedColorWithOverriddenAlpha(const ColorType& color, uint8_t overrideAlpha)
121{
122 static_assert(ColorType::Model::isInvertible);
123
124 auto components = asColorComponents(color);
125 auto copy = components;
126
127 for (unsigned i = 0; i < 3; ++i)
128 copy[i] = ColorType::Model::componentInfo[i].max - components[i];
129 copy[3] = convertByteAlphaTo<typename ColorType::ComponentType>(overrideAlpha);
130
131 return makeFromComponents<ColorType>(copy);
132}

134template<typename ColorType> ColorType invertedColorWithOverriddenAlpha(const ColorType& color, float overrideAlpha)
135{
136 static_assert(ColorType::Model::isInvertible);
137
138 auto components = asColorComponents(color);
139 auto copy = components;
140
141 for (unsigned i = 0; i < 3; ++i)
142 copy[i] = ColorType::Model::componentInfo[i].max - components[i];
143 copy[3] = convertFloatAlphaTo<typename ColorType::ComponentType>(overrideAlpha);
144
145 return makeFromComponents<ColorType>(copy);
146}

163template<typename ColorType, typename std::enable_if_t<std::is_same_v<typename ColorType::Model, RGBModel<typename ColorType::ComponentType>>>*>
164constexpr bool isBlack(const ColorType& color)
165{
166 auto [c1, c2, c3, alpha] = color;
167 constexpr auto componentInfo = ColorType::Model::componentInfo;
168 return c1 == componentInfo[0].min && c2 == componentInfo[1].min && c3 == componentInfo[2].min && alpha == AlphaTraits<typename ColorType::ComponentType>::opaque;
169}

186template<typename ColorType, typename std::enable_if_t<std::is_same_v<typename ColorType::Model, RGBModel<typename ColorType::ComponentType>>>*>
187constexpr bool isWhite(const ColorType& color)
188{
189 auto [c1, c2, c3, alpha] = color;
190 constexpr auto componentInfo = ColorType::Model::componentInfo;
191 return c1 == componentInfo[0].max && c2 == componentInfo[1].max && c3 == componentInfo[2].max && alpha == AlphaTraits<typename ColorType::ComponentType>::opaque;
192}

194constexpr uint16_t fastMultiplyBy255(uint16_t value)
195{
196 return (value << 8) - value;
197}
198
199constexpr uint16_t fastDivideBy255(uint16_t value)
200{
201 // While this is an approximate algorithm for division by 255, it gives perfectly accurate results for 16-bit values.
202 // FIXME: Since this gives accurate results for 16-bit values, we should get this optimization into compilers like clang.
203 uint16_t approximation = value >> 8;
204 uint16_t remainder = value - (approximation * 255) + 1;
205 return approximation + (remainder >> 8);
206}
