39#if USE(CG)
40using KnownColorSpaceAccessor = CGColorSpaceRef();
41template<KnownColorSpaceAccessor accessor> static const DestinationColorSpace& knownColorSpace()
42{
43 static LazyNeverDestroyed<DestinationColorSpace> colorSpace;
44 static std::once_flag onceFlag;
45 std::call_once(onceFlag, [] {
46 colorSpace.construct(accessor());
47 });
48 return colorSpace.get();
49}
50#else
51template<PlatformColorSpace::Name name> static const DestinationColorSpace& knownColorSpace()
52{
53 static NeverDestroyed<DestinationColorSpace> colorSpace { name };
54 return colorSpace.get();
55}
56#endif
57
58const DestinationColorSpace& DestinationColorSpace::SRGB()
59{
60#if USE(CG)
61 return knownColorSpace<sRGBColorSpaceRef>();
62#else
63 return knownColorSpace<PlatformColorSpace::Name::SRGB>();
64#endif
65}
66
67#if ENABLE(DESTINATION_COLOR_SPACE_LINEAR_SRGB)
68const DestinationColorSpace& DestinationColorSpace::LinearSRGB()
69{
70#if USE(CG)
71 return knownColorSpace<linearSRGBColorSpaceRef>();
72#else
73 return knownColorSpace<PlatformColorSpace::Name::LinearSRGB>();
74#endif
75}
76#endif
77
78#if ENABLE(DESTINATION_COLOR_SPACE_DISPLAY_P3)
79const DestinationColorSpace& DestinationColorSpace::DisplayP3()
80{
81#if USE(CG)
82 return knownColorSpace<displayP3ColorSpaceRef>();
83#else
84 return knownColorSpace<PlatformColorSpace::Name::DisplayP3>();
85#endif
86}
87#endif
88
89DestinationColorSpace::DestinationColorSpace(PlatformColorSpace platformColorSpace)
90 : m_platformColorSpace { WTFMove(platformColorSpace) }
91{
92#if USE(CG)
93 ASSERT(m_platformColorSpace);
94#endif
95}
96
97bool operator==(const DestinationColorSpace& a, const DestinationColorSpace& b)
98{
99#if USE(CG)
100 return CGColorSpaceEqualToColorSpace(a.platformColorSpace(), b.platformColorSpace());
101#else
102 return a.platformColorSpace() == b.platformColorSpace();
103#endif
104}
105
106bool operator!=(const DestinationColorSpace& a, const DestinationColorSpace& b)
107{
108 return !(a == b);
109}
110
111TextStream& operator<<(TextStream& ts, const DestinationColorSpace& colorSpace)
112{
113 if (colorSpace == DestinationColorSpace::SRGB())
114 ts << "sRGB";
115#if ENABLE(DESTINATION_COLOR_SPACE_LINEAR_SRGB)
116 else if (colorSpace == DestinationColorSpace::LinearSRGB())
117 ts << "LinearSRGB";
118#endif
119#if ENABLE(DESTINATION_COLOR_SPACE_DISPLAY_P3)
120 else if (colorSpace == DestinationColorSpace::DisplayP3())
121 ts << "DisplayP3";
122#endif
123#if USE(CG)
124 else if (auto description = adoptCF(CGColorSpaceCopyICCProfileDescription(colorSpace.platformColorSpace())))
125 ts << String(description.get());
126#endif
127
128 return ts;
129}
