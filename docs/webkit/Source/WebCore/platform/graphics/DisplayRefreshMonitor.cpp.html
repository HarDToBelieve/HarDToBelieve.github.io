47constexpr unsigned maxUnscheduledFireCount { 1 };

49RefPtr<DisplayRefreshMonitor> DisplayRefreshMonitor::createDefaultDisplayRefreshMonitor(PlatformDisplayID displayID)
50{
51#if PLATFORM(MAC)
52 return LegacyDisplayRefreshMonitorMac::create(displayID);
53#endif
54#if PLATFORM(IOS_FAMILY)
55 return DisplayRefreshMonitorIOS::create(displayID);
56#endif
57#if PLATFORM(GTK) && !USE(GTK4)
58 return DisplayRefreshMonitorGtk::create(displayID);
59#endif
60#if PLATFORM(WIN)
61 return DisplayRefreshMonitorWin::create(displayID);
62#endif
63 UNUSED_PARAM(displayID);
64 return nullptr;
65}
66
67RefPtr<DisplayRefreshMonitor> DisplayRefreshMonitor::create(DisplayRefreshMonitorFactory* factory, PlatformDisplayID displayID)
68{
69 if (factory) {
70 auto monitor = factory->createDisplayRefreshMonitor(displayID);
71 if (monitor)
72 return monitor;
73 }
74
75 return createDefaultDisplayRefreshMonitor(displayID);
76}

91void DisplayRefreshMonitor::addClient(DisplayRefreshMonitorClient& client)
92{
93 auto addResult = m_clients.add(&client);
94 if (addResult.isNewEntry) {
95 LOG_WITH_STREAM(DisplayLink, stream << "[Web] DisplayRefreshMonitor " << this << " addedClient - displayID " << m_displayID << " client " << &client << " client preferred fps " << client.preferredFramesPerSecond());
96 computeMaxPreferredFramesPerSecond();
97 }
98}
99
100bool DisplayRefreshMonitor::removeClient(DisplayRefreshMonitorClient& client)
101{
102 if (m_clientsToBeNotified)
103 m_clientsToBeNotified->remove(&client);
104
105 bool removed = m_clients.remove(&client);
106 if (removed) {
107 LOG_WITH_STREAM(DisplayLink, stream << "[Web] DisplayRefreshMonitor " << this << " removedClient " << &client);
108 computeMaxPreferredFramesPerSecond();
109 }
110
111 return removed;
112}
113
114std::optional<FramesPerSecond> DisplayRefreshMonitor::maximumClientPreferredFramesPerSecond() const
115{
116 std::optional<FramesPerSecond> maxFramesPerSecond;
117 for (auto* client : m_clients)
118 maxFramesPerSecond = std::max<FramesPerSecond>(maxFramesPerSecond.value_or(0), client->preferredFramesPerSecond());
119
120 return maxFramesPerSecond;
121}
122
123void DisplayRefreshMonitor::computeMaxPreferredFramesPerSecond()
124{
125 auto maxFramesPerSecond = maximumClientPreferredFramesPerSecond();
126 LOG_WITH_STREAM(DisplayLink, stream << "[Web] DisplayRefreshMonitor " << this << " computeMaxPreferredFramesPerSecond - displayID " << m_displayID << " adjusting max fps to " << maxFramesPerSecond);
127 if (maxFramesPerSecond != m_maxClientPreferredFramesPerSecond) {
128 m_maxClientPreferredFramesPerSecond = maxFramesPerSecond;
129 if (m_maxClientPreferredFramesPerSecond)
130 adjustPreferredFramesPerSecond(*m_maxClientPreferredFramesPerSecond);
131 }
132}

134void DisplayRefreshMonitor::clientPreferredFramesPerSecondChanged(DisplayRefreshMonitorClient&)
135{
136 computeMaxPreferredFramesPerSecond();
137}

139bool DisplayRefreshMonitor::requestRefreshCallback()
140{
141 Locker locker { m_lock };
142
143 if (isScheduled())
144 return true;
145
146 if (!startNotificationMechanism())
147 return false;
148
149 setIsScheduled(true);
150 return true;
151}
152
153bool DisplayRefreshMonitor::firedAndReachedMaxUnscheduledFireCount()
154{
155 ASSERT(m_lock.isLocked());
156
157 if (isScheduled()) {
158 m_unscheduledFireCount = 0;
159 return false;
160 }
161
162 ++m_unscheduledFireCount;
163 return m_unscheduledFireCount > m_maxUnscheduledFireCount;
164}

166void DisplayRefreshMonitor::displayLinkFired(const DisplayUpdate& displayUpdate)
167{
168 {
169 Locker locker { m_lock };
170
171 // This may be off the main thread.
172 if (!isPreviousFrameDone())
173 return;
174
175 LOG_WITH_STREAM(DisplayLink, stream << "[Web] DisplayRefreshMonitor::displayLinkFired for display " << displayID() << " - scheduled " << isScheduled() << " unscheduledFireCount " << m_unscheduledFireCount << " of " << m_maxUnscheduledFireCount);
176 if (firedAndReachedMaxUnscheduledFireCount()) {
177 stopNotificationMechanism();
178 return;
179 }
180
181 setIsScheduled(false);
182 setIsPreviousFrameDone(false);
183 }
184 dispatchDisplayDidRefresh(displayUpdate);
185}
186
187void DisplayRefreshMonitor::dispatchDisplayDidRefresh(const DisplayUpdate& displayUpdate)
188{
189 ASSERT(isMainThread());
190 displayDidRefresh(displayUpdate);
191}
192
193void DisplayRefreshMonitor::displayDidRefresh(const DisplayUpdate& displayUpdate)
194{
195 ASSERT(isMainThread());
196
197 UNUSED_PARAM(displayUpdate);
198 LOG_WITH_STREAM(DisplayLink, stream << "DisplayRefreshMonitor::displayDidRefresh for display " << displayID() << " update " << displayUpdate);
199
200 // The call back can cause all our clients to be unregistered, so we need to protect
201 // against deletion until the end of the method.
202 Ref<DisplayRefreshMonitor> protectedThis(*this);
203
204 // Copy the hash table and remove clients from it one by one so we don't notify
205 // any client twice, but can respond to removal of clients during the delivery process.
206 HashSet<DisplayRefreshMonitorClient*> clientsToBeNotified = m_clients;
207 m_clientsToBeNotified = &clientsToBeNotified;
208 while (!clientsToBeNotified.isEmpty()) {
209 DisplayRefreshMonitorClient* client = clientsToBeNotified.takeAny();
210 client->fireDisplayRefreshIfNeeded(displayUpdate);
211
212 // This checks if this function was reentered. In that case, stop iterating
213 // since it's not safe to use the set any more.
214 if (m_clientsToBeNotified != &clientsToBeNotified)
215 break;
216 }
217
218 if (m_clientsToBeNotified == &clientsToBeNotified)
219 m_clientsToBeNotified = nullptr;
220
221 {
222 Locker locker { m_lock };
223 setIsPreviousFrameDone(true);
224 }
225
226 DisplayRefreshMonitorManager::sharedManager().displayDidRefresh(*this);
227}
