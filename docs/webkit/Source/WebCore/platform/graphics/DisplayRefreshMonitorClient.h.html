26#pragma once
27
28#include "AnimationFrameRate.h"
29#include "PlatformScreen.h"
30#include <optional>
31
32namespace WebCore {
33
34class DisplayRefreshMonitor;
35class DisplayRefreshMonitorFactory;
36
37struct DisplayUpdate;
38
39class DisplayRefreshMonitorClient {
40public:
41 DisplayRefreshMonitorClient();
42 virtual ~DisplayRefreshMonitorClient();
43
44 // Always called on the main thread.
45 virtual void displayRefreshFired() = 0;
46
47 virtual DisplayRefreshMonitorFactory* displayRefreshMonitorFactory() const = 0;
48
49 PlatformDisplayID displayID() const { return m_displayID.value(); }
50 bool hasDisplayID() const { return !!m_displayID; }
51 void setDisplayID(PlatformDisplayID displayID) { m_displayID = displayID; }
52
53 void setPreferredFramesPerSecond(FramesPerSecond);
54 FramesPerSecond preferredFramesPerSecond() const { return m_preferredFramesPerSecond; }
55
56 void setIsScheduled(bool isScheduled) { m_scheduled = isScheduled; }
57 bool isScheduled() const { return m_scheduled; }
58
59 void fireDisplayRefreshIfNeeded(const DisplayUpdate&);
60
61private:
62 std::optional<PlatformDisplayID> m_displayID;
63 FramesPerSecond m_preferredFramesPerSecond { FullSpeedFramesPerSecond };
64 bool m_scheduled { false };
65};
