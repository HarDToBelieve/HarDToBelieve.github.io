44DisplayRefreshMonitor* DisplayRefreshMonitorManager::ensureMonitorForDisplayID(PlatformDisplayID displayID, DisplayRefreshMonitorFactory* factory)
45{
46 if (auto* existingMonitor = monitorForDisplayID(displayID))
47 return existingMonitor;
48
49 auto monitor = DisplayRefreshMonitor::create(factory, displayID);
50 if (!monitor)
51 return nullptr;
52
53 LOG_WITH_STREAM(DisplayLink, stream << "[Web] DisplayRefreshMonitorManager::ensureMonitorForDisplayID() - created monitor " << monitor.get() << " for display " << displayID);
54 DisplayRefreshMonitor* result = monitor.get();
55 m_monitors.append({ WTFMove(monitor) });
56 return result;
57}

59void DisplayRefreshMonitorManager::unregisterClient(DisplayRefreshMonitorClient& client)
60{
61 if (!client.hasDisplayID())
62 return;
63
64 PlatformDisplayID clientDisplayID = client.displayID();
65 auto index = findMonitorForDisplayID(clientDisplayID);
66 if (index == notFound)
67 return;
68
69 RefPtr<DisplayRefreshMonitor> monitor = m_monitors[index].monitor;
70 monitor->removeClient(client);
71}

73void DisplayRefreshMonitorManager::clientPreferredFramesPerSecondChanged(DisplayRefreshMonitorClient& client)
74{
75 if (auto* monitor = monitorForClient(client))
76 monitor->clientPreferredFramesPerSecondChanged(client);
77}

79bool DisplayRefreshMonitorManager::scheduleAnimation(DisplayRefreshMonitorClient& client)
80{
81 if (auto* monitor = monitorForClient(client)) {
82 client.setIsScheduled(true);
83 return monitor->requestRefreshCallback();
84 }
85 return false;
86}

93void DisplayRefreshMonitorManager::windowScreenDidChange(PlatformDisplayID displayID, DisplayRefreshMonitorClient& client)
94{
95 if (client.hasDisplayID() && client.displayID() == displayID)
96 return;
97
98 unregisterClient(client);
99 client.setDisplayID(displayID);
100 if (client.isScheduled())
101 scheduleAnimation(client);
102}

104std::optional<FramesPerSecond> DisplayRefreshMonitorManager::nominalFramesPerSecondForDisplay(PlatformDisplayID displayID, DisplayRefreshMonitorFactory* factory)
105{
106 auto* monitor = ensureMonitorForDisplayID(displayID, factory);
107 if (monitor)
108 monitor->displayNominalFramesPerSecond();
109
110 return std::nullopt;
111}

113void DisplayRefreshMonitorManager::displayWasUpdated(PlatformDisplayID displayID, const DisplayUpdate& displayUpdate)
114{
115 auto* monitor = monitorForDisplayID(displayID);
116 if (monitor)
117 monitor->displayLinkFired(displayUpdate);
118}
119
120size_t DisplayRefreshMonitorManager::findMonitorForDisplayID(PlatformDisplayID displayID) const
121{
122 return m_monitors.findMatching([&](auto& monitorWrapper) {
123 return monitorWrapper.monitor->displayID() == displayID;
124 });
125}
126
127DisplayRefreshMonitor* DisplayRefreshMonitorManager::monitorForClient(DisplayRefreshMonitorClient& client)
128{
129 if (!client.hasDisplayID())
130 return nullptr;
131
132 auto* monitor = ensureMonitorForDisplayID(client.displayID(), client.displayRefreshMonitorFactory());
133 if (monitor)
134 monitor->addClient(client);
135
136 return monitor;
137}
138
139DisplayRefreshMonitor* DisplayRefreshMonitorManager::monitorForDisplayID(PlatformDisplayID displayID) const
140{
141 auto index = findMonitorForDisplayID(displayID);
142 return index == notFound ? nullptr : m_monitors[index].monitor.get();
143}
