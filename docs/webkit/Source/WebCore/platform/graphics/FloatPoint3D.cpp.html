28namespace WebCore {
29
30void FloatPoint3D::normalize()
31{
32 float tempLength = length();
33
34 if (tempLength) {
35 m_x /= tempLength;
36 m_y /= tempLength;
37 m_z /= tempLength;
38 }
39}
40
41TextStream& operator<<(TextStream& ts, const FloatPoint3D& point)
42{
43 return ts << point.x() << " " << point.y() << " " << point.z();
44}
