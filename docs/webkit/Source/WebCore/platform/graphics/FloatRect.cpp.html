50bool FloatRect::isExpressibleAsIntRect() const
51{
52 return isWithinIntRange(x()) && isWithinIntRange(y())
53 && isWithinIntRange(width()) && isWithinIntRange(height())
54 && isWithinIntRange(maxX()) && isWithinIntRange(maxY());
55}

57bool FloatRect::inclusivelyIntersects(const FloatRect& other) const
58{
59 return width() >= 0 && height() >= 0 && other.width() >= 0 && other.height() >= 0
60 && x() <= other.maxX() && other.x() <= maxX() && y() <= other.maxY() && other.y() <= maxY();
61}

63bool FloatRect::intersects(const FloatRect& other) const
64{
65 // Checking emptiness handles negative widths and heights as well as zero.
66 return !isEmpty() && !other.isEmpty()
67 && x() < other.maxX() && other.x() < maxX()
68 && y() < other.maxY() && other.y() < maxY();
69}
70
71bool FloatRect::contains(const FloatRect& other) const
72{
73 return x() <= other.x() && maxX() >= other.maxX()
74 && y() <= other.y() && maxY() >= other.maxY();
75}
76
77bool FloatRect::contains(const FloatPoint& point, ContainsMode containsMode) const
78{
79 if (containsMode == InsideOrOnStroke)
80 return contains(point.x(), point.y());
81 return x() < point.x() && maxX() > point.x() && y() < point.y() && maxY() > point.y();
82}

84void FloatRect::intersect(const FloatRect& other)
85{
86 float l = std::max(x(), other.x());
87 float t = std::max(y(), other.y());
88 float r = std::min(maxX(), other.maxX());
89 float b = std::min(maxY(), other.maxY());
90
91 // Return a clean empty rectangle for non-intersecting cases.
92 if (l >= r || t >= b) {
93 l = 0;
94 t = 0;
95 r = 0;
96 b = 0;
97 }
98
99 setLocationAndSizeFromEdges(l, t, r, b);
100}
101
102bool FloatRect::edgeInclusiveIntersect(const FloatRect& other)
103{
104 FloatPoint newLocation(std::max(x(), other.x()), std::max(y(), other.y()));
105 FloatPoint newMaxPoint(std::min(maxX(), other.maxX()), std::min(maxY(), other.maxY()));
106
107 bool intersects = true;
108
109 // Return a clean empty rectangle for non-intersecting cases.
110 if (newLocation.x() > newMaxPoint.x() || newLocation.y() > newMaxPoint.y()) {
111 newLocation = { };
112 newMaxPoint = { };
113 intersects = false;
114 }
115
116 m_location = newLocation;
117 m_size = newMaxPoint - newLocation;
118 return intersects;
119}

121void FloatRect::unite(const FloatRect& other)
122{
123 // Handle empty special cases first.
124 if (other.isEmpty())
125 return;
126 if (isEmpty()) {
127 *this = other;
128 return;
129 }
130
131 uniteEvenIfEmpty(other);
132}
133
134void FloatRect::uniteEvenIfEmpty(const FloatRect& other)
135{
136 float minX = std::min(x(), other.x());
137 float minY = std::min(y(), other.y());
138 float maxX = std::max(this->maxX(), other.maxX());
139 float maxY = std::max(this->maxY(), other.maxY());
140
141 setLocationAndSizeFromEdges(minX, minY, maxX, maxY);
142}
143
144void FloatRect::uniteIfNonZero(const FloatRect& other)
145{
146 // Handle empty special cases first.
147 if (other.isZero())
148 return;
149 if (isZero()) {
150 *this = other;
151 return;
152 }
153
154 uniteEvenIfEmpty(other);
155}

157void FloatRect::extend(const FloatPoint& p)
158{
159 float minX = std::min(x(), p.x());
160 float minY = std::min(y(), p.y());
161 float maxX = std::max(this->maxX(), p.x());
162 float maxY = std::max(this->maxY(), p.y());
163
164 setLocationAndSizeFromEdges(minX, minY, maxX, maxY);
165}

167void FloatRect::scale(float sx, float sy)
168{
169 m_location.setX(x() * sx);
170 m_location.setY(y() * sy);
171 m_size.setWidth(width() * sx);
172 m_size.setHeight(height() * sy);
173}

175void FloatRect::fitToPoints(const FloatPoint& p0, const FloatPoint& p1)
176{
177 float left = std::min(p0.x(), p1.x());
178 float top = std::min(p0.y(), p1.y());
179 float right = std::max(p0.x(), p1.x());
180 float bottom = std::max(p0.y(), p1.y());
181
182 setLocationAndSizeFromEdges(left, top, right, bottom);
183}

186// Helpers for 3- and 4-way max and min.
187
188template <typename T>
189T min3(const T& v1, const T& v2, const T& v3)
190{
191 return std::min(std::min(v1, v2), v3);
192}
193
194template <typename T>
195T max3(const T& v1, const T& v2, const T& v3)
196{
197 return std::max(std::max(v1, v2), v3);
198}
199
200template <typename T>
201T min4(const T& v1, const T& v2, const T& v3, const T& v4)
202{
203 return std::min(std::min(v1, v2), std::min(v3, v4));
204}
205
206template <typename T>
207T max4(const T& v1, const T& v2, const T& v3, const T& v4)
208{
209 return std::max(std::max(v1, v2), std::max(v3, v4));
210}

214void FloatRect::fitToPoints(const FloatPoint& p0, const FloatPoint& p1, const FloatPoint& p2)
215{
216 float left = min3(p0.x(), p1.x(), p2.x());
217 float top = min3(p0.y(), p1.y(), p2.y());
218 float right = max3(p0.x(), p1.x(), p2.x());
219 float bottom = max3(p0.y(), p1.y(), p2.y());
220
221 setLocationAndSizeFromEdges(left, top, right, bottom);
222}

224void FloatRect::fitToPoints(const FloatPoint& p0, const FloatPoint& p1, const FloatPoint& p2, const FloatPoint& p3)

227 float top = min4(p0.y(), p1.y(), p2.y(), p3.y());
228 float right = max4(p0.x(), p1.x(), p2.x(), p3.x());
229 float bottom = max4(p0.y(), p1.y(), p2.y(), p3.y());
230
231 setLocationAndSizeFromEdges(left, top, right, bottom);
232}

234FloatRect normalizeRect(const FloatRect& rect)
235{
236 return FloatRect(std::min(rect.x(), rect.maxX()),
237 std::min(rect.y(), rect.maxY()),
238 std::max(rect.width(), -rect.width()),
239 std::max(rect.height(), -rect.height()));
240}

242FloatRect encloseRectToDevicePixels(const FloatRect& rect, float deviceScaleFactor)
243{
244 FloatPoint location = floorPointToDevicePixels(rect.minXMinYCorner(), deviceScaleFactor);
245 FloatPoint maxPoint = ceilPointToDevicePixels(rect.maxXMaxYCorner(), deviceScaleFactor);
246 return FloatRect(location, maxPoint - location);
247}

249IntRect enclosingIntRect(const FloatRect& rect)
250{
251 FloatPoint location = flooredIntPoint(rect.minXMinYCorner());
252 FloatPoint maxPoint = ceiledIntPoint(rect.maxXMaxYCorner());
253 return IntRect(IntPoint(location), IntSize(maxPoint - location));
254}

256IntRect roundedIntRect(const FloatRect& rect)
257{
258 return IntRect(roundedIntPoint(rect.location()), roundedIntSize(rect.size()));

261TextStream& operator<<(TextStream& ts, const FloatRect &r)
262{
263 if (ts.hasFormattingFlag(TextStream::Formatting::SVGStyleRect)) {
264 // FIXME: callers should use the NumberRespectingIntegers flag.
265 return ts << "at (" << TextStream::FormatNumberRespectingIntegers(r.x()) << "," << TextStream::FormatNumberRespectingIntegers(r.y())
266 << ") size " << TextStream::FormatNumberRespectingIntegers(r.width()) << "x" << TextStream::FormatNumberRespectingIntegers(r.height());
267 }
268
269 return ts << r.location() << " " << r.size();
270}
