62class IntSize;

64class FloatSize {
65public:
66 FloatSize() { }
67 FloatSize(float width, float height) : m_width(width), m_height(height) { }
68 WEBCORE_EXPORT FloatSize(const IntSize&);
69
70 static FloatSize narrowPrecision(double width, double height);
71
72 float width() const { return m_width; }
73 float height() const { return m_height; }
74
75 bool hasIntrinsicWidth = false;
76 bool hasIntrinsicHeight = false;
77
78 void setWidth(float width) { m_width = width; }
79 void setHeight(float height) { m_height = height; }
80
81 bool isEmpty() const { return m_width <= 0 || m_height <= 0; }
82 WEBCORE_EXPORT bool isZero() const;
83 bool isExpressibleAsIntSize() const;
84
85 float aspectRatio() const { return m_width / m_height; }
86
87 void expand(float width, float height)
88 {
89 m_width += width;
90 m_height += height;
91 }
92
93 void scale(float s)
94 {
95 m_width *= s;
96 m_height *= s;
97 }
98
99 void scale(float scaleX, float scaleY)
100 {
101 m_width *= scaleX;
102 m_height *= scaleY;
103 }
104
105 FloatSize scaled(float s) const
106 {
107 return { m_width * s, m_height * s };
108 }
109
110 FloatSize scaled(float scaleX, float scaleY) const
111 {
112 return { m_width * scaleX, m_height * scaleY };
113 }
114
115 WEBCORE_EXPORT FloatSize constrainedBetween(const FloatSize& min, const FloatSize& max) const;
116
117 FloatSize expandedTo(const FloatSize& other) const
118 {
119 return FloatSize(m_width > other.m_width ? m_width : other.m_width,
120 m_height > other.m_height ? m_height : other.m_height);
121 }
122
123 FloatSize shrunkTo(const FloatSize& other) const
124 {
125 return FloatSize(m_width < other.m_width ? m_width : other.m_width,
126 m_height < other.m_height ? m_height : other.m_height);
127 }
128
129 float diagonalLength() const
130 {
131 return std::hypot(m_width, m_height);
132 }
133
134 float diagonalLengthSquared() const
135 {
136 return m_width * m_width + m_height * m_height;
137 }
138
139 float area() const
140 {
141 return m_width * m_height;
142 }
143
144 FloatSize transposedSize() const
145 {
146 return FloatSize(m_height, m_width);
147 }
148
149#if USE(CG)
150 WEBCORE_EXPORT explicit FloatSize(const CGSize&); // don't do this implicitly since it's lossy
151 WEBCORE_EXPORT operator CGSize() const;
152#endif
153
154#if PLATFORM(MAC) && !defined(NSGEOMETRY_TYPES_SAME_AS_CGGEOMETRY_TYPES)
155 WEBCORE_EXPORT explicit FloatSize(const NSSize&); // don't do this implicitly since it's lossy
156 operator NSSize() const;
157#endif
158
159#if PLATFORM(WIN)
160 WEBCORE_EXPORT FloatSize(const D2D1_SIZE_F&);
161 operator D2D1_SIZE_F() const;
162#endif
163
164 String toJSONString() const;
165 WEBCORE_EXPORT Ref<JSON::Object> toJSONObject() const;
166
167private:
168 float m_width { 0 };
169 float m_height { 0 };
170};

172inline FloatSize& operator+=(FloatSize& a, const FloatSize& b)
173{
174 a.setWidth(a.width() + b.width());
175 a.setHeight(a.height() + b.height());
176 return a;
177}
178
179inline FloatSize& operator-=(FloatSize& a, const FloatSize& b)
180{
181 a.setWidth(a.width() - b.width());
182 a.setHeight(a.height() - b.height());
183 return a;
184}
185
186inline FloatSize operator+(const FloatSize& a, const FloatSize& b)
187{
188 return FloatSize(a.width() + b.width(), a.height() + b.height());
189}
190
191inline FloatSize operator-(const FloatSize& a, const FloatSize& b)
192{
193 return FloatSize(a.width() - b.width(), a.height() - b.height());
194}
195
196inline FloatSize operator-(const FloatSize& size)
197{
198 return FloatSize(-size.width(), -size.height());
199}
200
201inline FloatSize operator*(const FloatSize& a, float b)
202{
203 return FloatSize(a.width() * b, a.height() * b);
204}
205
206inline FloatSize operator*(float a, const FloatSize& b)
207{
208 return FloatSize(a * b.width(), a * b.height());
209}
210
211inline FloatSize operator*(const FloatSize& a, const FloatSize& b)
212{
213 return FloatSize(a.width() * b.width(), a.height() * b.height());
214}
215
216inline FloatSize operator/(const FloatSize& a, const FloatSize& b)
217{
218 return FloatSize(a.width() / b.width(), a.height() / b.height());
219}
220
221inline FloatSize operator/(const FloatSize& a, float b)
222{
223 return FloatSize(a.width() / b, a.height() / b);
224}
225
226inline FloatSize operator/(float a, const FloatSize& b)
227{
228 return FloatSize(a / b.width(), a / b.height());
229}
230
231inline bool operator==(const FloatSize& a, const FloatSize& b)
232{
233 return a.width() == b.width() && a.height() == b.height();
234}
235
236inline bool areEssentiallyEqual(const FloatSize& a, const FloatSize& b)
237{
238 return WTF::areEssentiallyEqual(a.width(), b.width()) && WTF::areEssentiallyEqual(a.height(), b.height());
239}
240
241inline bool operator!=(const FloatSize& a, const FloatSize& b)
242{
243 return a.width() != b.width() || a.height() != b.height();
244}
245
246inline IntSize roundedIntSize(const FloatSize& p)
247{
248 return IntSize(clampToInteger(roundf(p.width())), clampToInteger(roundf(p.height())));
249}
250
251inline IntSize flooredIntSize(const FloatSize& p)
252{
253 return IntSize(clampToInteger(floorf(p.width())), clampToInteger(floorf(p.height())));
254}
255
256inline IntSize expandedIntSize(const FloatSize& p)
257{
258 return IntSize(clampToInteger(ceilf(p.width())), clampToInteger(ceilf(p.height())));
259}
260
261inline IntPoint flooredIntPoint(const FloatSize& p)
262{
263 return IntPoint(clampToInteger(floorf(p.width())), clampToInteger(floorf(p.height())));
264}
