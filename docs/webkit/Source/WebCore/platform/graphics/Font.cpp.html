109// Estimates of avgCharWidth and maxCharWidth for platforms that don't support accessing these values from the font.
110void Font::initCharWidths()
111{
112 auto* glyphPageZero = glyphPage(GlyphPage::pageNumberForCodePoint('0'));
113
114 // Treat the width of a '0' as the avgCharWidth.
115 if (m_avgCharWidth <= 0.f && glyphPageZero) {
116 Glyph digitZeroGlyph = glyphPageZero->glyphDataForCharacter('0').glyph;
117 if (digitZeroGlyph)
118 m_avgCharWidth = widthForGlyph(digitZeroGlyph);
119 }
120
121 // If we can't retrieve the width of a '0', fall back to the x height.
122 if (m_avgCharWidth <= 0.f)
123 m_avgCharWidth = m_fontMetrics.xHeight();
124
125 if (m_maxCharWidth <= 0.f)
126 m_maxCharWidth = std::max(m_avgCharWidth, m_fontMetrics.floatAscent());
127}

129void Font::platformGlyphInit()
130{
131#if USE(FREETYPE)
132 auto* glyphPageZeroWidthSpace = glyphPage(GlyphPage::pageNumberForCodePoint(zeroWidthSpace));
133 UChar32 zeroWidthSpaceCharacter = zeroWidthSpace;
134#else
135 // Ask for the glyph for 0 to avoid paging in ZERO WIDTH SPACE. Control characters, including 0,
136 // are mapped to the ZERO WIDTH SPACE glyph for non FreeType based ports.
137 auto* glyphPageZeroWidthSpace = glyphPage(0);
138 UChar32 zeroWidthSpaceCharacter = 0;
139#endif
140 auto* glyphPageCharacterZero = glyphPage(GlyphPage::pageNumberForCodePoint('0'));
141 auto* glyphPageSpace = glyphPage(GlyphPage::pageNumberForCodePoint(space));
142
143 if (glyphPageZeroWidthSpace)
144 m_zeroWidthSpaceGlyph = glyphPageZeroWidthSpace->glyphDataForCharacter(zeroWidthSpaceCharacter).glyph;
145
146 // Nasty hack to determine if we should round or ceil space widths.
147 // If the font is monospace or fake monospace we ceil to ensure that
148 // every character and the space are the same width. Otherwise we round.
149 if (glyphPageSpace)
150 m_spaceGlyph = glyphPageSpace->glyphDataForCharacter(space).glyph;
151 if (glyphPageCharacterZero)
152 m_zeroGlyph = glyphPageCharacterZero->glyphDataForCharacter('0').glyph;
153
154 // Force the glyph for ZERO WIDTH SPACE to have zero width, unless it is shared with SPACE.
155 // Helvetica is an example of a non-zero width ZERO WIDTH SPACE glyph.
156 // See <http://bugs.webkit.org/show_bug.cgi?id=13178> and Font::isZeroWidthSpaceGlyph()
157 if (m_zeroWidthSpaceGlyph == m_spaceGlyph)
158 m_zeroWidthSpaceGlyph = 0;
159
160 float width = widthForGlyph(m_spaceGlyph);
161 m_spaceWidth = width;
162 m_fontMetrics.setZeroWidth(widthForGlyph(m_zeroGlyph));
163 auto amountToAdjustLineGap = std::min(m_fontMetrics.floatLineGap(), 0.0f);
164 m_fontMetrics.setLineGap(m_fontMetrics.floatLineGap() - amountToAdjustLineGap);
165 m_fontMetrics.setLineSpacing(m_fontMetrics.floatLineSpacing() - amountToAdjustLineGap);
166 determinePitch();
167 m_adjustedSpaceWidth = m_treatAsFixedPitch ? ceilf(width) : roundf(width);
168}

182static bool fillGlyphPage(GlyphPage& pageToFill, UChar* buffer, unsigned bufferLength, const Font& font)
183{
184 bool hasGlyphs = pageToFill.fill(buffer, bufferLength);
185#if ENABLE(OPENTYPE_VERTICAL)
186 if (hasGlyphs && font.verticalData())
187 font.verticalData()->substituteWithVerticalGlyphs(&font, &pageToFill);
188#else
189 UNUSED_PARAM(font);
190#endif
191 return hasGlyphs;
192}

194static std::optional<size_t> codePointSupportIndex(UChar32 codePoint)
195{
196 // FIXME: Consider reordering these so the most common ones are at the front.
197 // Doing this could cause the BitVector to fit inside inline storage and therefore
198 // be both a performance and a memory progression.
199 if (codePoint < 0x20)
200 return codePoint;
201 if (codePoint >= 0x7F && codePoint < 0xA0)
202 return codePoint - 0x7F + 0x20;
203 std::optional<size_t> result;
204 switch (codePoint) {
205 case softHyphen:
206 result = 0x41;
207 break;
208 case newlineCharacter:
209 result = 0x42;
210 break;
211 case tabCharacter:
212 result = 0x43;
213 break;
214 case noBreakSpace:
215 result = 0x44;
216 break;
217 case narrowNoBreakSpace:
218 result = 0x45;
219 break;
220 case leftToRightMark:
221 result = 0x46;
222 break;
223 case rightToLeftMark:
224 result = 0x47;
225 break;
226 case leftToRightEmbed:
227 result = 0x48;
228 break;
229 case rightToLeftEmbed:
230 result = 0x49;
231 break;
232 case leftToRightOverride:
233 result = 0x4A;
234 break;
235 case rightToLeftOverride:
236 result = 0x4B;
237 break;
238 case leftToRightIsolate:
239 result = 0x4C;
240 break;
241 case rightToLeftIsolate:
242 result = 0x4D;
243 break;
244 case zeroWidthNonJoiner:
245 result = 0x4E;
246 break;
247 case zeroWidthJoiner:
248 result = 0x4F;
249 break;
250 case popDirectionalFormatting:
251 result = 0x50;
252 break;
253 case popDirectionalIsolate:
254 result = 0x51;
255 break;
256 case firstStrongIsolate:
257 result = 0x52;
258 break;
259 case objectReplacementCharacter:
260 result = 0x53;
261 break;
262 case zeroWidthNoBreakSpace:
263 result = 0x54;
264 break;
265 default:
266 result = std::nullopt;
267 }
268
269#ifndef NDEBUG
270 UChar32 codePointOrder[] = {
271 0x00, 0x01, 0x02, 0x03, 0x04, 0x05, 0x06, 0x07, 0x08, 0x09, 0x0A, 0x0B, 0x0C, 0x0D, 0x0E, 0x0F,
272 0x10, 0x11, 0x12, 0x13, 0x14, 0x15, 0x16, 0x17, 0x18, 0x19, 0x1A, 0x1B, 0x1C, 0x1D, 0x1E, 0x1F,
273 0x7F,
274 0x80, 0x81, 0x82, 0x83, 0x84, 0x85, 0x86, 0x87, 0x88, 0x89, 0x8A, 0x8B, 0x8C, 0x8D, 0x8E, 0x8F,
275 0x90, 0x91, 0x92, 0x93, 0x94, 0x95, 0x96, 0x97, 0x98, 0x99, 0x9A, 0x9B, 0x9C, 0x9D, 0x9E, 0x9F,
276 softHyphen,
277 newlineCharacter,
278 tabCharacter,
279 noBreakSpace,
280 narrowNoBreakSpace,
281 leftToRightMark,
282 rightToLeftMark,
283 leftToRightEmbed,
284 rightToLeftEmbed,
285 leftToRightOverride,
286 rightToLeftOverride,
287 leftToRightIsolate,
288 rightToLeftIsolate,
289 zeroWidthNonJoiner,
290 zeroWidthJoiner,
291 popDirectionalFormatting,
292 popDirectionalIsolate,
293 firstStrongIsolate,
294 objectReplacementCharacter,
295 zeroWidthNoBreakSpace
296 };
297 bool found = false;
298 for (size_t i = 0; i < WTF_ARRAY_LENGTH(codePointOrder); ++i) {
299 if (codePointOrder[i] == codePoint) {
300 ASSERT(i == result);
301 found = true;
302 }
303 }
304 ASSERT(found == static_cast<bool>(result));
305#endif
306 return result;
307}

309#if !USE(FREETYPE)
310static void overrideControlCharacters(Vector<UChar>& buffer, unsigned start, unsigned end)
311{
312 auto overwriteCodePoints = [&](unsigned minimum, unsigned maximum, UChar newCodePoint) {
313 unsigned begin = std::max(start, minimum);
314 unsigned complete = std::min(end, maximum);
315 for (unsigned i = begin; i < complete; ++i) {
316 ASSERT(codePointSupportIndex(i));
317 buffer[i - start] = newCodePoint;
318 }
319 };
320
321 auto overwriteCodePoint = [&](UChar codePoint, UChar newCodePoint) {
322 ASSERT(codePointSupportIndex(codePoint));
323 if (codePoint >= start && codePoint < end)
324 buffer[codePoint - start] = newCodePoint;
325 };
326
327 // Code points 0x0 - 0x20 and 0x7F - 0xA0 are control character and shouldn't render. Map them to ZERO WIDTH SPACE.
328 overwriteCodePoints(0x0, 0x20, zeroWidthSpace);
329 overwriteCodePoints(0x7F, 0xA0, zeroWidthSpace);
330 overwriteCodePoint(softHyphen, zeroWidthSpace);
331 overwriteCodePoint('\n', space);
332 overwriteCodePoint('\t', space);
333 overwriteCodePoint(noBreakSpace, space);
334 overwriteCodePoint(leftToRightMark, zeroWidthSpace);
335 overwriteCodePoint(rightToLeftMark, zeroWidthSpace);
336 overwriteCodePoint(leftToRightEmbed, zeroWidthSpace);
337 overwriteCodePoint(rightToLeftEmbed, zeroWidthSpace);
338 overwriteCodePoint(leftToRightOverride, zeroWidthSpace);
339 overwriteCodePoint(rightToLeftOverride, zeroWidthSpace);
340 overwriteCodePoint(leftToRightIsolate, zeroWidthSpace);
341 overwriteCodePoint(rightToLeftIsolate, zeroWidthSpace);
342 overwriteCodePoint(zeroWidthNonJoiner, zeroWidthSpace);
343 overwriteCodePoint(zeroWidthJoiner, zeroWidthSpace);
344 overwriteCodePoint(popDirectionalFormatting, zeroWidthSpace);
345 overwriteCodePoint(popDirectionalIsolate, zeroWidthSpace);
346 overwriteCodePoint(firstStrongIsolate, zeroWidthSpace);
347 overwriteCodePoint(objectReplacementCharacter, zeroWidthSpace);
348 overwriteCodePoint(zeroWidthNoBreakSpace, zeroWidthSpace);
349}
350#endif

352static RefPtr<GlyphPage> createAndFillGlyphPage(unsigned pageNumber, const Font& font)
353{
354#if PLATFORM(IOS_FAMILY)
355 // FIXME: Times New Roman contains Arabic glyphs, but Core Text doesn't know how to shape them. See <rdar://problem/9823975>.
356 // Once we have the fix for <rdar://problem/9823975> then remove this code together with Font::shouldNotBeUsedForArabic()
357 // in <rdar://problem/12096835>.
358 if (GlyphPage::pageNumberIsUsedForArabic(pageNumber) && font.shouldNotBeUsedForArabic())
359 return nullptr;
360#endif
361
362 unsigned glyphPageSize = GlyphPage::sizeForPageNumber(pageNumber);
363
364 unsigned start = GlyphPage::startingCodePointInPageNumber(pageNumber);
365 Vector<UChar> buffer(glyphPageSize * 2 + 2);
366 unsigned bufferLength;
367 // Fill in a buffer with the entire "page" of characters that we want to look up glyphs for.
368 if (U_IS_BMP(start)) {
369 bufferLength = glyphPageSize;
370 for (unsigned i = 0; i < bufferLength; i++)
371 buffer[i] = start + i;
372
373#if !USE(FREETYPE)
374 overrideControlCharacters(buffer, start, start + glyphPageSize);
375#endif
376 } else {
377 bufferLength = glyphPageSize * 2;
378 for (unsigned i = 0; i < glyphPageSize; i++) {
379 int c = i + start;
380 buffer[i * 2] = U16_LEAD(c);
381 buffer[i * 2 + 1] = U16_TRAIL(c);
382 }
383 }
384
385 // Now that we have a buffer full of characters, we want to get back an array
386 // of glyph indices. This part involves calling into the platform-specific
387 // routine of our glyph map for actually filling in the page with the glyphs.
388 // Success is not guaranteed. For example, Times fails to fill page 260, giving glyph data
389 // for only 128 out of 256 characters.
390 Ref<GlyphPage> glyphPage = GlyphPage::create(font);
391
392 bool haveGlyphs = fillGlyphPage(glyphPage, buffer.data(), bufferLength, font);
393 if (!haveGlyphs)
394 return nullptr;
395
396 return glyphPage;
397}

399const GlyphPage* Font::glyphPage(unsigned pageNumber) const
400{
401 if (!pageNumber) {
402 if (!m_glyphPageZero)
403 m_glyphPageZero = createAndFillGlyphPage(0, *this);
404 return m_glyphPageZero.get();
405 }
406 auto addResult = m_glyphPages.add(pageNumber, nullptr);
407 if (addResult.isNewEntry)
408 addResult.iterator->value = createAndFillGlyphPage(pageNumber, *this);
409
410 return addResult.iterator->value.get();
411}

550struct CharacterFallbackMapKey {
551 AtomString locale;
552 UChar32 character { 0 };
553 bool isForPlatformFont { false };
554};
555
556inline bool operator==(const CharacterFallbackMapKey& a, const CharacterFallbackMapKey& b)
557{
558 return a.locale == b.locale && a.character == b.character && a.isForPlatformFont == b.isForPlatformFont;
559}
560
561struct CharacterFallbackMapKeyHash {
562 static unsigned hash(const CharacterFallbackMapKey& key) { return computeHash(key.locale, key.character, key.isForPlatformFont); }
563 static bool equal(const CharacterFallbackMapKey& a, const CharacterFallbackMapKey& b) { return a == b; }
564 static const bool safeToCompareToEmptyOrDeleted = true;
565};
566
567struct CharacterFallbackMapKeyHashTraits : SimpleClassHashTraits<CharacterFallbackMapKey> {
568 static void constructDeletedValue(CharacterFallbackMapKey& slot) { new (NotNull, &slot) CharacterFallbackMapKey { { }, U_SENTINEL, { } }; }
569 static bool isDeletedValue(const CharacterFallbackMapKey& key) { return key.character == U_SENTINEL; }
570};
571
572// Fonts are not ref'd to avoid cycles.
573// FIXME: Consider changing these maps to use WeakPtr instead of raw pointers.
574using CharacterFallbackMap = HashMap<CharacterFallbackMapKey, Font*, CharacterFallbackMapKeyHash, CharacterFallbackMapKeyHashTraits>;
575using SystemFallbackCache = HashMap<const Font*, CharacterFallbackMap>;
576
577static SystemFallbackCache& systemFallbackCache()
578{
579 static NeverDestroyed<SystemFallbackCache> map;
580 return map.get();
581}
582
583RefPtr<Font> Font::systemFallbackFontForCharacter(UChar32 character, const FontDescription& description, IsForPlatformFont isForPlatformFont, FontCache& fontCache) const
584{
585 auto fontAddResult = systemFallbackCache().add(this, CharacterFallbackMap());
586
587 if (!character) {
588 UChar codeUnit = 0;
589 return fontCache.systemFallbackForCharacters(description, this, isForPlatformFont, FontCache::PreferColoredFont::No, &codeUnit, 1);
590 }
591
592 auto key = CharacterFallbackMapKey { description.computedLocale(), character, isForPlatformFont != IsForPlatformFont::No };
593 return fontAddResult.iterator->value.ensure(WTFMove(key), [&] {
594 UChar codeUnits[2];
595 unsigned codeUnitsLength;
596 if (U_IS_BMP(character)) {
597 codeUnits[0] = FontCascade::normalizeSpaces(character);
598 codeUnitsLength = 1;
599 } else {
600 codeUnits[0] = U16_LEAD(character);
601 codeUnits[1] = U16_TRAIL(character);
602 codeUnitsLength = 2;
603 }
604 auto font = fontCache.systemFallbackForCharacters(description, this, isForPlatformFont, FontCache::PreferColoredFont::No, codeUnits, codeUnitsLength).get();
605 if (font)
606 font->m_isUsedInSystemFallbackCache = true;
607 return font;
608 }).iterator->value;
609}
610
611void Font::removeFromSystemFallbackCache()
612{
613 systemFallbackCache().remove(this);
614
615 if (!m_isUsedInSystemFallbackCache)
616 return;
617
618 for (auto& characterMap : systemFallbackCache().values()) {
619 Vector<CharacterFallbackMapKey, 512> toRemove;
620 for (auto& entry : characterMap) {
621 if (entry.value == this)
622 toRemove.append(entry.key);
623 }
624 for (auto& key : toRemove)
625 characterMap.remove(key);
626 }
627}

645bool Font::supportsCodePoint(UChar32 character) const
646{
647 // This is very similar to static_cast<bool>(glyphForCharacter(character))
648 // except that glyphForCharacter() maps certain code points to ZWS (because they
649 // shouldn't be visible). This function doesn't do that mapping, and instead is
650 // as honest as possible about what code points the font supports. This is so
651 // that we can accurately determine which characters are supported by this font
652 // so we know which boundaries to break strings when we send them to the complex
653 // text codepath. The complex text codepath is totally separate from this ZWS
654 // replacement logic (because CoreText handles those characters instead of WebKit).
655 if (auto index = codePointSupportIndex(character)) {
656 m_codePointSupport.ensureSize(2 * (*index + 1));
657 bool hasBeenSet = m_codePointSupport.quickSet(2 * *index);
658 if (!hasBeenSet && platformSupportsCodePoint(character))
659 m_codePointSupport.quickSet(2 * *index + 1);
660 return m_codePointSupport.quickGet(2 * *index + 1);
661 }
662 return glyphForCharacter(character);
663}

665bool Font::canRenderCombiningCharacterSequence(const UChar* characters, size_t length) const
666{
667 ASSERT(isMainThread());
668
669 auto codePoints = StringView(characters, length).codePoints();
670 auto it = codePoints.begin();
671 auto end = codePoints.end();
672 while (it != end) {
673 auto codePoint = *it;
674 ++it;
675
676 if (it != end && isVariationSelector(*it)) {
677 if (!platformSupportsCodePoint(codePoint, *it)) {
678 // Try the characters individually.
679 if (!supportsCodePoint(codePoint) || !supportsCodePoint(*it))
680 return false;
681 }
682 ++it;
683 continue;
684 }
685
686 if (!supportsCodePoint(codePoint))
687 return false;
688 }
689 return true;
690}

698 m_glyphPathMap.setMetricsForGlyph(glyph, path);
