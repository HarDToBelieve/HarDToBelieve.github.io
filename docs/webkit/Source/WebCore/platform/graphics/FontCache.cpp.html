89 return a.descriptionKey == b.descriptionKey && a.family == b.family && a.features == b.features && a.capabilities == b.capabilities;

92struct FontPlatformDataCacheKeyHash {
93 static unsigned hash(const FontPlatformDataCacheKey& key) { return computeHash(key.descriptionKey, key.family, key.features, key.capabilities.tied()); }
94 static bool equal(const FontPlatformDataCacheKey& a, const FontPlatformDataCacheKey& b) { return a == b; }
95 static constexpr bool safeToCompareToEmptyOrDeleted = true;

169std::optional<ASCIILiteral> FontCache::alternateFamilyName(const String& familyName)
170{
171 if (auto platformSpecificAlternate = platformAlternateFamilyName(familyName))
172 return platformSpecificAlternate;
173
174 switch (familyName.length()) {
175 case 5:
176 if (equalLettersIgnoringASCIICase(familyName, "arial"))
177 return "Helvetica"_s;
178 if (equalLettersIgnoringASCIICase(familyName, "times"))
179 return "Times New Roman"_s;
180 break;
181 case 7:
182 if (equalLettersIgnoringASCIICase(familyName, "courier"))
183 return "Courier New"_s;
184 break;
185 case 9:
186 if (equalLettersIgnoringASCIICase(familyName, "helvetica"))
187 return "Arial"_s;
188 break;
189#if !OS(WINDOWS)
190 // On Windows, Courier New is a TrueType font that is always present and
191 // Courier is a bitmap font that we do not support. So, we don't want to map
192 // Courier New to Courier.
193 // FIXME: Not sure why this is harmful on Windows, since the alternative will
194 // only be tried if Courier New is not found.
195 case 11:
196 if (equalLettersIgnoringASCIICase(familyName, "courier new"))
197 return "Courier"_s;
198 break;
199#endif
200 case 15:
201 if (equalLettersIgnoringASCIICase(familyName, "times new roman"))
202 return "Times"_s;
203 break;
204 }
205
206 return std::nullopt;
207}

209FontPlatformData* FontCache::cachedFontPlatformData(const FontDescription& fontDescription, const String& passedFamilyName,
210 const FontFeatureSettings* features, FontSelectionSpecifiedCapabilities capabilities, bool checkingAlternateName)
211{
212#if PLATFORM(IOS_FAMILY)
213 Locker locker { m_fontLock };
214#endif
215
216#if OS(WINDOWS) && ENABLE(OPENTYPE_VERTICAL)
217 // Leading "@" in the font name enables Windows vertical flow flag for the font.
218 // Because we do vertical flow by ourselves, we don't want to use the Windows feature.
219 // IE disregards "@" regardless of the orientation, so we follow the behavior.
220 const String& familyName = passedFamilyName.substring(passedFamilyName[0] == '@' ? 1 : 0);
221#else
222 const String& familyName = passedFamilyName;
223#endif
224
225 static std::once_flag onceFlag;
226 std::call_once(onceFlag, [&]() {
227 platformInit();
228 });
229
230 FontPlatformDataCacheKey key { fontDescription, { familyName }, features ? *features : FontFeatureSettings { }, capabilities };
231
232 auto addResult = m_fontDataCaches->platformData.add(key, nullptr);
233 FontPlatformDataCache::iterator it = addResult.iterator;
234 if (addResult.isNewEntry) {
235 it->value = createFontPlatformData(fontDescription, familyName, features, capabilities);
236 if (!it->value && !checkingAlternateName) {
237 // We were unable to find a font. We have a small set of fonts that we alias to other names,
238 // e.g., Arial/Helvetica, Courier/Courier New, etc. Try looking up the font under the aliased name.
239 if (auto alternateName = alternateFamilyName(familyName)) {
240 auto* alternateData = cachedFontPlatformData(fontDescription, *alternateName, features, capabilities, true);
241 // Look up the key in the hash table again as the previous iterator may have
242 // been invalidated by the recursive call to cachedFontPlatformData().
243 it = m_fontDataCaches->platformData.find(key);
244 ASSERT(it != m_fontDataCaches->platformData.end());
245 if (alternateData)
246 it->value = makeUnique<FontPlatformData>(*alternateData);
247 }
248 }
249 }
250
251 return it->value.get();
252}

265RefPtr<Font> FontCache::fontForFamily(const FontDescription& fontDescription, const String& family, const FontFeatureSettings* features, FontSelectionSpecifiedCapabilities capabilities, bool checkingAlternateName)
266{
267 if (!m_purgeTimer.isActive())
268 m_purgeTimer.startOneShot(0_s);
269
270 if (auto* platformData = cachedFontPlatformData(fontDescription, family, features, capabilities, checkingAlternateName))
271 return fontForPlatformData(*platformData);
272
273 return nullptr;
274}

276Ref<Font> FontCache::fontForPlatformData(const FontPlatformData& platformData)
277{
278#if PLATFORM(IOS_FAMILY)
279 Locker locker { m_fontLock };
280#endif
281
282 auto addResult = m_fontDataCaches->data.ensure(platformData, [&] {
283 return Font::create(platformData, Font::Origin::Local, this);
284 });
285
286 ASSERT(addResult.iterator->value->platformData() == platformData);
287
288 return addResult.iterator->value.copyRef();
289}

291void FontCache::purgeInactiveFontDataIfNeeded()
292{
293 bool underMemoryPressure = MemoryPressureHandler::singleton().isUnderMemoryPressure();
294 unsigned inactiveFontDataLimit = underMemoryPressure ? cMaxUnderMemoryPressureInactiveFontData : cMaxInactiveFontData;
295
296 LOG(Fonts, "FontCache::purgeInactiveFontDataIfNeeded() - underMemoryPressure %d, inactiveFontDataLimit %u", underMemoryPressure, inactiveFontDataLimit);
297
298 if (m_fontDataCaches->data.size() < inactiveFontDataLimit)
299 return;
300 unsigned inactiveCount = inactiveFontCount();
301 if (inactiveCount <= inactiveFontDataLimit)
302 return;
303
304 unsigned targetFontDataLimit = underMemoryPressure ? cTargetUnderMemoryPressureInactiveFontData : cTargetInactiveFontData;
305 purgeInactiveFontData(inactiveCount - targetFontDataLimit);
306}

308void FontCache::purgeInactiveFontData(unsigned purgeCount)
309{
310 LOG(Fonts, "FontCache::purgeInactiveFontData(%u)", purgeCount);
311
312 pruneUnreferencedEntriesFromFontCascadeCache();
313 pruneSystemFallbackFonts();
314
315#if PLATFORM(IOS_FAMILY)
316 Locker locker { m_fontLock };
317#endif
318
319 while (purgeCount) {
320 Vector<Ref<Font>, 20> fontsToDelete;
321 for (auto& font : m_fontDataCaches->data.values()) {
322 LOG(Fonts, " trying to purge font %s (has one ref %d)", font->platformData().description().utf8().data(), font->hasOneRef());
323 if (!font->hasOneRef())
324 continue;
325 fontsToDelete.append(font.copyRef());
326 if (!--purgeCount)
327 break;
328 }
329 // Fonts may ref other fonts so we loop until there are no changes.
330 if (fontsToDelete.isEmpty())
331 break;
332 for (auto& font : fontsToDelete) {
333 bool success = m_fontDataCaches->data.remove(font->platformData());
334 ASSERT_UNUSED(success, success);
335#if ENABLE(OPENTYPE_VERTICAL)
336 m_fontDataCaches->verticalData.remove(font->platformData());
337#endif
338 }
339 };
340
341 Vector<FontPlatformDataCacheKey> keysToRemove;
342 keysToRemove.reserveInitialCapacity(m_fontDataCaches->platformData.size());
343 for (auto& entry : m_fontDataCaches->platformData) {
344 if (entry.value && !m_fontDataCaches->data.contains(*entry.value))
345 keysToRemove.uncheckedAppend(entry.key);
346 }
347
348 LOG(Fonts, " removing %lu keys", keysToRemove.size());
349
350 for (auto& key : keysToRemove)
351 m_fontDataCaches->platformData.remove(key);
352
353 platformPurgeInactiveFontData();
354}

390static FontCascadeCacheKey makeFontCascadeCacheKey(const FontCascadeDescription& description, FontSelector* fontSelector)
391{
392 FontCascadeCacheKey key;
393 key.fontDescriptionKey = FontDescriptionKey(description);
394 unsigned familyCount = description.familyCount();
395 key.families.reserveInitialCapacity(familyCount);
396 for (unsigned i = 0; i < familyCount; ++i)
397 key.families.uncheckedAppend(description.familyAt(i));
398 key.fontSelectorId = fontSelector ? fontSelector->uniqueId() : 0;
399 key.fontSelectorVersion = fontSelector ? fontSelector->version() : 0;
400 return key;
401}

416Ref<FontCascadeFonts> FontCache::retrieveOrAddCachedFonts(const FontCascadeDescription& fontDescription, RefPtr<FontSelector>&& fontSelector)
417{
418 auto key = makeFontCascadeCacheKey(fontDescription, fontSelector.get());
419 auto addResult = m_fontCascadeCache.add(key, nullptr);
420 if (!addResult.isNewEntry)
421 return addResult.iterator->value->fonts.get();
422
423 auto& newEntry = addResult.iterator->value;
424 newEntry = makeUnique<FontCascadeCacheEntry>(FontCascadeCacheEntry { WTFMove(key), FontCascadeFonts::create(WTFMove(fontSelector)) });
425 Ref<FontCascadeFonts> glyphs = newEntry->fonts.get();
426
427 static constexpr unsigned unreferencedPruneInterval = 50;
428 static constexpr int maximumEntries = 400;
429 static unsigned pruneCounter;
430 // Referenced FontCascadeFonts would exist anyway so pruning them saves little memory.
431 if (!(++pruneCounter % unreferencedPruneInterval))
432 pruneUnreferencedEntriesFromFontCascadeCache();
433 // Prevent pathological growth.
434 if (m_fontCascadeCache.size() > maximumEntries)
435 m_fontCascadeCache.remove(m_fontCascadeCache.random());
436 return glyphs;
437}

475void FontCache::invalidate()
476{
477 m_fontDataCaches->platformData.clear();
478#if ENABLE(OPENTYPE_VERTICAL)
479 m_fontDataCaches->verticalData.clear();
480#endif
481 invalidateFontCascadeCache();
482
483 ++m_generation;
484
485 for (auto& client : copyToVectorOf<RefPtr<FontSelector>>(m_clients))
486 client->fontCacheInvalidated();
487
488 purgeInactiveFontData();
489}
