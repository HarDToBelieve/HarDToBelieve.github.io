50static bool useBackslashAsYenSignForFamily(const AtomString& family)
51{
52 if (family.isEmpty())
53 return false;
54 static const auto set = makeNeverDestroyed([] {
55 MemoryCompactLookupOnlyRobinHoodHashSet<AtomString> set;
56 auto add = [&set] (const char* name, std::initializer_list<UChar> unicodeName) {
57 unsigned nameLength = strlen(name);
58 set.add(AtomString { name, nameLength, AtomString::ConstructFromLiteral });
59 unsigned unicodeNameLength = unicodeName.size();
60 set.add(AtomString { unicodeName.begin(), unicodeNameLength });
61 };
62 add("MS PGothic", { 0xFF2D, 0xFF33, 0x0020, 0xFF30, 0x30B4, 0x30B7, 0x30C3, 0x30AF });
63 add("MS PMincho", { 0xFF2D, 0xFF33, 0x0020, 0xFF30, 0x660E, 0x671D });
64 add("MS Gothic", { 0xFF2D, 0xFF33, 0x0020, 0x30B4, 0x30B7, 0x30C3, 0x30AF });
65 add("MS Mincho", { 0xFF2D, 0xFF33, 0x0020, 0x660E, 0x671D });
66 add("Meiryo", { 0x30E1, 0x30A4, 0x30EA, 0x30AA });
67 return set;
68 }());
69 return set.get().contains(family);
70}

115bool FontCascade::operator==(const FontCascade& other) const
116{
117 if (isLoadingCustomFonts() || other.isLoadingCustomFonts())
118 return false;
119
120 if (m_fontDescription != other.m_fontDescription || m_letterSpacing != other.m_letterSpacing || m_wordSpacing != other.m_wordSpacing)
121 return false;
122 if (m_fonts == other.m_fonts)
123 return true;
124 if (!m_fonts || !other.m_fonts)
125 return false;
126 if (m_fonts->fontSelector() != other.m_fonts->fontSelector())
127 return false;
128 // Can these cases actually somehow occur? All fonts should get wiped out by full style recalc.
129 if (m_fonts->fontSelectorVersion() != other.m_fonts->fontSelectorVersion())
130 return false;
131 if (m_fonts->generation() != other.m_fonts->generation())
132 return false;
133 return true;
134}

169FloatSize FontCascade::drawText(GraphicsContext& context, const TextRun& run, const FloatPoint& point, unsigned from, std::optional<unsigned> to, CustomFontNotReadyAction customFontNotReadyAction) const
170{
171 unsigned destination = to.value_or(run.length());
172 auto glyphBuffer = layoutText(codePath(run, from, to), run, from, destination);
173 glyphBuffer.flatten();
174
175 if (glyphBuffer.isEmpty())
176 return FloatSize();
177
178 FloatPoint startPoint = point + WebCore::size(glyphBuffer.initialAdvance());
179 drawGlyphBuffer(context, glyphBuffer, startPoint, customFontNotReadyAction);
180 return startPoint - point;
181}

200std::unique_ptr<DisplayList::InMemoryDisplayList> FontCascade::displayListForTextRun(GraphicsContext& context, const TextRun& run, unsigned from, std::optional<unsigned> to, CustomFontNotReadyAction customFontNotReadyAction) const
201{
202 ASSERT(!context.paintingDisabled());
203 unsigned destination = to.value_or(run.length());
204
205 // FIXME: Use the fast code path once it handles partial runs with kerning and ligatures. See http://webkit.org/b/100050
206 CodePath codePathToUse = codePath(run);
207 if (codePathToUse != CodePath::Complex && (enableKerning() || requiresShaping()) && (from || destination != run.length()))
208 codePathToUse = CodePath::Complex;
209
210 auto glyphBuffer = layoutText(codePathToUse, run, from, destination);
211 glyphBuffer.flatten();
212
213 if (glyphBuffer.isEmpty())
214 return nullptr;
215
216 std::unique_ptr<DisplayList::InMemoryDisplayList> displayList = makeUnique<DisplayList::InMemoryDisplayList>();
217 DisplayList::Recorder recordingContext(*displayList, context.state(), FloatRect(), AffineTransform(), nullptr, DisplayList::DrawGlyphsRecorder::DrawGlyphsDeconstruction::DontDeconstruct);
218
219 FloatPoint startPoint = toFloatPoint(WebCore::size(glyphBuffer.initialAdvance()));
220 drawGlyphBuffer(recordingContext, glyphBuffer, startPoint, customFontNotReadyAction);
221
222 displayList->shrinkToFit();
223 return displayList;
224}

226float FontCascade::widthOfTextRange(const TextRun& run, unsigned from, unsigned to, HashSet<const Font*>* fallbackFonts, float* outWidthBeforeRange, float* outWidthAfterRange) const
227{
228 ASSERT(from <= to);
229 ASSERT(to <= run.length());
230
231 if (!run.length())
232 return 0;
233
234 float offsetBeforeRange = 0;
235 float offsetAfterRange = 0;
236 float totalWidth = 0;
237
238 auto codePathToUse = codePath(run);
239 if (codePathToUse == CodePath::Complex) {
240 ComplexTextController complexIterator(*this, run, false, fallbackFonts);
241 complexIterator.advance(from, nullptr, IncludePartialGlyphs, fallbackFonts);
242 offsetBeforeRange = complexIterator.runWidthSoFar();
243 complexIterator.advance(to, nullptr, IncludePartialGlyphs, fallbackFonts);
244 offsetAfterRange = complexIterator.runWidthSoFar();
245 complexIterator.advance(run.length(), nullptr, IncludePartialGlyphs, fallbackFonts);
246 totalWidth = complexIterator.runWidthSoFar();
247 } else {
248 WidthIterator simpleIterator(*this, run, fallbackFonts);
249 GlyphBuffer glyphBuffer;
250 simpleIterator.advance(from, glyphBuffer);
251 offsetBeforeRange = simpleIterator.runWidthSoFar();
252 simpleIterator.advance(to, glyphBuffer);
253 offsetAfterRange = simpleIterator.runWidthSoFar();
254 simpleIterator.advance(run.length(), glyphBuffer);
255 totalWidth = simpleIterator.runWidthSoFar();
256 simpleIterator.finalize(glyphBuffer);
257 }
258
259 if (outWidthBeforeRange)
260 *outWidthBeforeRange = offsetBeforeRange;
261
262 if (outWidthAfterRange)
263 *outWidthAfterRange = totalWidth - offsetAfterRange;
264
265 return offsetAfterRange - offsetBeforeRange;
266}

268float FontCascade::width(const TextRun& run, HashSet<const Font*>* fallbackFonts, GlyphOverflow* glyphOverflow) const
269{
270 if (!run.length())
271 return 0;
272
273 CodePath codePathToUse = codePath(run);
274 if (codePathToUse != CodePath::Complex) {
275 // The complex path is more restrictive about returning fallback fonts than the simple path, so we need an explicit test to make their behaviors match.
276 if (!canReturnFallbackFontsForComplexText())
277 fallbackFonts = nullptr;
278 // The simple path can optimize the case where glyph overflow is not observable.
279 if (codePathToUse != CodePath::SimpleWithGlyphOverflow && (glyphOverflow && !glyphOverflow->computeBounds))
280 glyphOverflow = nullptr;
281 }
282
283 bool hasWordSpacingOrLetterSpacing = wordSpacing() || letterSpacing();
284 float* cacheEntry = m_fonts->widthCache().add(run, std::numeric_limits<float>::quiet_NaN(), enableKerning() || requiresShaping(), hasWordSpacingOrLetterSpacing, glyphOverflow);
285 if (cacheEntry && !std::isnan(*cacheEntry))
286 return *cacheEntry;
287
288 HashSet<const Font*> localFallbackFonts;
289 if (!fallbackFonts)
290 fallbackFonts = &localFallbackFonts;
291
292 float result;
293 if (codePathToUse == CodePath::Complex)
294 result = floatWidthForComplexText(run, fallbackFonts, glyphOverflow);
295 else
296 result = floatWidthForSimpleText(run, fallbackFonts, glyphOverflow);
297
298 if (cacheEntry && fallbackFonts->isEmpty())
299 *cacheEntry = result;
300 return result;
301}

303float FontCascade::widthForSimpleText(StringView text, TextDirection textDirection) const
304{
305 if (text.isNull() || text.isEmpty())
306 return 0;
307 ASSERT(codePath(TextRun(text)) != CodePath::Complex);
308 float* cacheEntry = m_fonts->widthCache().add(text, std::numeric_limits<float>::quiet_NaN());
309 if (cacheEntry && !std::isnan(*cacheEntry))
310 return *cacheEntry;
311
312 GlyphBuffer glyphBuffer;
313 float runWidth = 0;
314 auto& font = primaryFont();
315 for (unsigned i = 0; i < text.length(); ++i) {
316 auto glyph = glyphDataForCharacter(text[i], false).glyph;
317 auto glyphWidth = font.widthForGlyph(glyph);
318 runWidth += glyphWidth;
319 glyphBuffer.add(glyph, font, glyphWidth, i);
320 }
321
322 font.applyTransforms(glyphBuffer, 0, 0, enableKerning(), requiresShaping(), fontDescription().computedLocale(), text, textDirection);
323 // This is needed only to match the result of the slow path.
324 // Same glyph widths but different floating point arithmetic can produce different run width.
325 float runWidthDifferenceWithTransformApplied = -runWidth;
326 for (size_t i = 0; i < glyphBuffer.size(); ++i)
327 runWidthDifferenceWithTransformApplied += WebCore::width(glyphBuffer.advanceAt(i));
328 runWidth += runWidthDifferenceWithTransformApplied;
329
330 if (cacheEntry)
331 *cacheEntry = runWidth;
332 return runWidth;
333}

335GlyphData FontCascade::glyphDataForCharacter(UChar32 c, bool mirror, FontVariant variant) const
336{
337 if (variant == AutoVariant) {
338 if (m_fontDescription.variantCaps() == FontVariantCaps::Small) {
339 UChar32 upperC = u_toupper(c);
340 if (upperC != c) {
341 c = upperC;
342 variant = SmallCapsVariant;
343 } else
344 variant = NormalVariant;
345 } else
346 variant = NormalVariant;
347 }
348
349 if (mirror)
350 c = u_charMirror(c);
351
352 return m_fonts->glyphDataForCharacter(c, m_fontDescription, variant);
353}

488FontCascade::CodePath FontCascade::codePath(const TextRun& run, std::optional<unsigned> from, std::optional<unsigned> to) const
489{
490 if (s_codePath != CodePath::Auto)
491 return s_codePath;
492
493#if !USE(FREETYPE)
494 // FIXME: Use the fast code path once it handles partial runs with kerning and ligatures. See http://webkit.org/b/100050
495 if ((enableKerning() || requiresShaping()) && (from.value_or(0) || to.value_or(run.length()) != run.length()))
496 return CodePath::Complex;
497#else
498 UNUSED_PARAM(from);
499 UNUSED_PARAM(to);
500#endif
501
502#if PLATFORM(COCOA) || USE(FREETYPE)
503 // Because Font::applyTransforms() doesn't know which features to enable/disable in the simple code path, it can't properly handle feature or variant settings.
504 // FIXME: https://bugs.webkit.org/show_bug.cgi?id=150791: @font-face features should also cause this to be complex.
505 if (m_fontDescription.featureSettings().size() > 0 || !m_fontDescription.variantSettings().isAllNormal())
506 return CodePath::Complex;
507
508#else
509
510 if (run.length() > 1 && (enableKerning() || requiresShaping()))
511 return CodePath::Complex;
512#endif
513
514 if (!run.characterScanForCodePath())
515 return CodePath::Simple;
516
517 if (run.is8Bit())
518 return CodePath::Simple;
519
520 // Start from 0 since drawing and highlighting also measure the characters before run->from.
521 return characterRangeCodePath(run.characters16(), run.length());
522}

524FontCascade::CodePath FontCascade::characterRangeCodePath(const UChar* characters, unsigned len)
525{
526 // FIXME: Should use a UnicodeSet in ports where ICU is used. Note that we
527 // can't simply use UnicodeCharacter Property/class because some characters
528 // are not 'combining', but still need to go to the complex path.
529 // Alternatively, we may as well consider binary search over a sorted
530 // list of ranges.
531 CodePath result = CodePath::Simple;
532 bool previousCharacterIsEmojiGroupCandidate = false;
533 for (unsigned i = 0; i < len; i++) {
534 const UChar c = characters[i];
535 if (c == zeroWidthJoiner && previousCharacterIsEmojiGroupCandidate)
536 return CodePath::Complex;
537
538 previousCharacterIsEmojiGroupCandidate = false;
539 if (c < 0x2E5) // U+02E5 through U+02E9 (Modifier Letters : Tone letters)
540 continue;
541 if (c <= 0x2E9)
542 return CodePath::Complex;
543
544 if (c < 0x300) // U+0300 through U+036F Combining diacritical marks
545 continue;
546 if (c <= 0x36F)
547 return CodePath::Complex;
548
549 if (c < 0x0591 || c == 0x05BE) // U+0591 through U+05CF excluding U+05BE Hebrew combining marks, Hebrew punctuation Paseq, Sof Pasuq and Nun Hafukha
550 continue;
551 if (c <= 0x05CF)
552 return CodePath::Complex;
553
554 // U+0600 through U+109F Arabic, Syriac, Thaana, NKo, Samaritan, Mandaic,
555 // Devanagari, Bengali, Gurmukhi, Gujarati, Oriya, Tamil, Telugu, Kannada,
556 // Malayalam, Sinhala, Thai, Lao, Tibetan, Myanmar
557 if (c < 0x0600)
558 continue;
559 if (c <= 0x109F)
560 return CodePath::Complex;
561
562 // U+1100 through U+11FF Hangul Jamo (only Ancient Korean should be left here if you precompose;
563 // Modern Korean will be precomposed as a result of step A)
564 if (c < 0x1100)
565 continue;
566 if (c <= 0x11FF)
567 return CodePath::Complex;
568
569 if (c < 0x135D) // U+135D through U+135F Ethiopic combining marks
570 continue;
571 if (c <= 0x135F)
572 return CodePath::Complex;
573
574 if (c < 0x1700) // U+1780 through U+18AF Tagalog, Hanunoo, Buhid, Taghanwa,Khmer, Mongolian
575 continue;
576 if (c <= 0x18AF)
577 return CodePath::Complex;
578
579 if (c < 0x1900) // U+1900 through U+194F Limbu (Unicode 4.0)
580 continue;
581 if (c <= 0x194F)
582 return CodePath::Complex;
583
584 if (c < 0x1980) // U+1980 through U+19DF New Tai Lue
585 continue;
586 if (c <= 0x19DF)
587 return CodePath::Complex;
588
589 if (c < 0x1A00) // U+1A00 through U+1CFF Buginese, Tai Tham, Balinese, Batak, Lepcha, Vedic
590 continue;
591 if (c <= 0x1CFF)
592 return CodePath::Complex;
593
594 if (c < 0x1DC0) // U+1DC0 through U+1DFF Comining diacritical mark supplement
595 continue;
596 if (c <= 0x1DFF)
597 return CodePath::Complex;
598
599 // U+1E00 through U+2000 characters with diacritics and stacked diacritics
600 if (c <= 0x2000) {
601 result = CodePath::SimpleWithGlyphOverflow;
602 continue;
603 }
604
605 if (c < 0x20D0) // U+20D0 through U+20FF Combining marks for symbols
606 continue;
607 if (c <= 0x20FF)
608 return CodePath::Complex;
609
610 if (c < 0x26F9)
611 continue;
612 if (c < 0x26FA)
613 return CodePath::Complex;
614
615 if (c < 0x2CEF) // U+2CEF through U+2CF1 Combining marks for Coptic
616 continue;
617 if (c <= 0x2CF1)
618 return CodePath::Complex;
619
620 if (c < 0x302A) // U+302A through U+302F Ideographic and Hangul Tone marks
621 continue;
622 if (c <= 0x302F)
623 return CodePath::Complex;
624
625 if (c < 0x3099)
626 continue;
627 if (c < 0x309D)
628 return CodePath::Complex; // KATAKANA-HIRAGANA (SEMI-)VOICED SOUND MARKS require character composition
629
630 if (c < 0xA67C) // U+A67C through U+A67D Combining marks for old Cyrillic
631 continue;
632 if (c <= 0xA67D)
633 return CodePath::Complex;
634
635 if (c < 0xA6F0) // U+A6F0 through U+A6F1 Combining mark for Bamum
636 continue;
637 if (c <= 0xA6F1)
638 return CodePath::Complex;
639
640 // U+A800 through U+ABFF Nagri, Phags-pa, Saurashtra, Devanagari Extended,
641 // Hangul Jamo Ext. A, Javanese, Myanmar Extended A, Tai Viet, Meetei Mayek,
642 if (c < 0xA800)
643 continue;
644 if (c <= 0xABFF)
645 return CodePath::Complex;
646
647 if (c < 0xD7B0) // U+D7B0 through U+D7FF Hangul Jamo Ext. B
648 continue;
649 if (c <= 0xD7FF)
650 return CodePath::Complex;
651
652 if (c <= 0xDBFF) {
653 // High surrogate
654
655 if (i == len - 1)
656 continue;
657
658 UChar next = characters[++i];
659 if (!U16_IS_TRAIL(next))

962std::pair<unsigned, bool> FontCascade::expansionOpportunityCountInternal(const LChar* characters, unsigned length, TextDirection direction, ExpansionBehavior expansionBehavior)
963{
964 unsigned count = 0;
965 bool isAfterExpansion = (expansionBehavior & LeftExpansionMask) == ForbidLeftExpansion;
966 if ((expansionBehavior & LeftExpansionMask) == ForceLeftExpansion) {
967 ++count;
968 isAfterExpansion = true;
969 }
970 if (direction == TextDirection::LTR) {
971 for (unsigned i = 0; i < length; ++i) {
972 if (treatAsSpace(characters[i])) {
973 count++;
974 isAfterExpansion = true;
975 } else
976 isAfterExpansion = false;
977 }
978 } else {
979 for (unsigned i = length; i > 0; --i) {
980 if (treatAsSpace(characters[i - 1])) {
981 count++;
982 isAfterExpansion = true;
983 } else
984 isAfterExpansion = false;
985 }
986 }
987 if (!isAfterExpansion && (expansionBehavior & RightExpansionMask) == ForceRightExpansion) {
988 ++count;
989 isAfterExpansion = true;
990 } else if (isAfterExpansion && (expansionBehavior & RightExpansionMask) == ForbidRightExpansion) {
991 ASSERT(count);
992 --count;
993 isAfterExpansion = false;
994 }
995 return std::make_pair(count, isAfterExpansion);

999{

1003 if ((expansionBehavior & LeftExpansionMask) == ForceLeftExpansion) {
1004 ++count;
1005 isAfterExpansion = true;
1006 }

1010 if (treatAsSpace(character)) {
1011 count++;
1012 isAfterExpansion = true;
1013 continue;
1014 }

1016 character = U16_GET_SUPPLEMENTARY(character, characters[i + 1]);

1022 count++;
1023 isAfterExpansion = true;

1025 }

1027 }
1028 } else {

1033 isAfterExpansion = true;

1037 character = U16_GET_SUPPLEMENTARY(characters[i - 2], character);

1043 count++;
1044 isAfterExpansion = true;
1045 continue;
1046 }
1047 isAfterExpansion = false;
1048 }
1049 }

1134static GlyphUnderlineType computeUnderlineType(const TextRun& textRun, const GlyphBuffer& glyphBuffer, unsigned index)
1135{
1136 // In general, we want to skip descenders. However, skipping descenders on CJK characters leads to undesirable renderings,
1137 // so we want to draw through CJK characters (on a character-by-character basis).
1138 // FIXME: The CSS spec says this should instead be done by the user-agent stylesheet using the lang= attribute.
1139 UChar32 baseCharacter;
1140 auto offsetInString = glyphBuffer.checkedStringOffsetAt(index, textRun.length());
1141 if (!offsetInString)
1142 return GlyphUnderlineType::SkipDescenders;
1143
1144 if (textRun.is8Bit())
1145 baseCharacter = textRun.characters8()[offsetInString.value()];
1146 else
1147 U16_GET(textRun.characters16(), 0, static_cast<unsigned>(offsetInString.value()), textRun.length(), baseCharacter);
1148
1149 // u_getIntPropertyValue with UCHAR_IDEOGRAPHIC doesn't return true for Japanese or Korean codepoints.
1150 // Instead, we can use the "Unicode allocation block" for the character.
1151 UBlockCode blockCode = ublock_getCode(baseCharacter);
1152 switch (blockCode) {
1153 case UBLOCK_CJK_RADICALS_SUPPLEMENT:
1154 case UBLOCK_CJK_SYMBOLS_AND_PUNCTUATION:
1155 case UBLOCK_ENCLOSED_CJK_LETTERS_AND_MONTHS:

1157 case UBLOCK_CJK_UNIFIED_IDEOGRAPHS_EXTENSION_A:
1158 case UBLOCK_CJK_UNIFIED_IDEOGRAPHS:
1159 case UBLOCK_CJK_COMPATIBILITY_IDEOGRAPHS:
1160 case UBLOCK_CJK_COMPATIBILITY_FORMS:
1161 case UBLOCK_CJK_UNIFIED_IDEOGRAPHS_EXTENSION_B:
1162 case UBLOCK_CJK_COMPATIBILITY_IDEOGRAPHS_SUPPLEMENT:

1246GlyphBuffer FontCascade::layoutSimpleText(const TextRun& run, unsigned from, unsigned to, ForTextEmphasisOrNot forTextEmphasis) const
1247{
1248 GlyphBuffer glyphBuffer;
1249
1250 WidthIterator it(*this, run, 0, false, forTextEmphasis);
1251 // FIXME: Using separate glyph buffers for the prefix and the suffix is incorrect when kerning or
1252 // ligatures are enabled.
1253 GlyphBuffer localGlyphBuffer;
1254 it.advance(from, localGlyphBuffer);
1255 float beforeWidth = it.runWidthSoFar();
1256 it.advance(to, glyphBuffer);
1257
1258 if (glyphBuffer.isEmpty())
1259 return glyphBuffer;
1260
1261 float afterWidth = it.runWidthSoFar();
1262
1263 float initialAdvance = 0;
1264 if (run.rtl()) {
1265 it.advance(run.length(), localGlyphBuffer);
1266 it.finalize(localGlyphBuffer);
1267 initialAdvance = it.runWidthSoFar() - afterWidth;
1268 } else {
1269 it.finalize(localGlyphBuffer);
1270 initialAdvance = beforeWidth;
1271 }
1272 glyphBuffer.expandInitialAdvance(initialAdvance);
1273 if (!glyphBuffer.isEmpty()) {
1274 // The initial advance is supposed to point directly to the first glyph's paint position.
1275 // See the ascii-art diagram in ComplexTextController.h.
1276 glyphBuffer.expandInitialAdvance(makeGlyphBufferAdvance(x(glyphBuffer.originAt(0)), y(glyphBuffer.originAt(0))));
1277 }
1278
1279 // The glyph buffer is currently in logical order,
1280 // but we need to return the results in visual order.
1281 if (run.rtl())
1282 glyphBuffer.reverse(0, glyphBuffer.size());
1283
1284 return glyphBuffer;
1285}

1289 GlyphBuffer glyphBuffer;
1290
1291 ComplexTextController controller(*this, run, false, 0, forTextEmphasis);
1292 GlyphBuffer dummyGlyphBuffer;
1293 controller.advance(from, &dummyGlyphBuffer);
1294 controller.advance(to, &glyphBuffer);
1295
1296 if (glyphBuffer.isEmpty())
1297 return glyphBuffer;
1298
1299 if (run.rtl()) {
1300 // Exploit the fact that the sum of the paint advances is equal to
1301 // the sum of the layout advances.
1302 FloatSize initialAdvance = controller.totalAdvance();
1303 for (unsigned i = 0; i < dummyGlyphBuffer.size(); ++i)
1304 initialAdvance -= WebCore::size(dummyGlyphBuffer.advanceAt(i));
1305 for (unsigned i = 0; i < glyphBuffer.size(); ++i)
1306 initialAdvance -= WebCore::size(glyphBuffer.advanceAt(i));
1307 // FIXME: Shouldn't we subtract the other initial advance?
1308 glyphBuffer.reverse(0, glyphBuffer.size());
1309 glyphBuffer.setInitialAdvance(makeGlyphBufferAdvance(initialAdvance));
1310 } else {
1311 FloatSize initialAdvance = WebCore::size(dummyGlyphBuffer.initialAdvance());
1312 for (unsigned i = 0; i < dummyGlyphBuffer.size(); ++i)
1313 initialAdvance += WebCore::size(dummyGlyphBuffer.advanceAt(i));
1314 // FIXME: Shouldn't we add the other initial advance?
1315 glyphBuffer.setInitialAdvance(makeGlyphBufferAdvance(initialAdvance));
1316 }
1317
1318 return glyphBuffer;
1319}

1329void FontCascade::drawGlyphBuffer(GraphicsContext& context, const GlyphBuffer& glyphBuffer, FloatPoint& point, CustomFontNotReadyAction customFontNotReadyAction) const
1330{
1331 ASSERT(glyphBuffer.isFlattened());
1332 const Font* fontData = &glyphBuffer.fontAt(0);
1333 FloatPoint startPoint = point;
1334 float nextX = startPoint.x() + WebCore::width(glyphBuffer.advanceAt(0));
1335 float nextY = startPoint.y() + height(glyphBuffer.advanceAt(0));
1336 unsigned lastFrom = 0;
1337 unsigned nextGlyph = 1;
1338 while (nextGlyph < glyphBuffer.size()) {
1339 const Font& nextFontData = glyphBuffer.fontAt(nextGlyph);
1340
1341 if (&nextFontData != fontData) {
1342 if (shouldDrawIfLoading(*fontData, customFontNotReadyAction))
1343 context.drawGlyphs(*fontData, glyphBuffer.glyphs(lastFrom), glyphBuffer.advances(lastFrom), nextGlyph - lastFrom, startPoint, m_fontDescription.fontSmoothing());
1344
1345 lastFrom = nextGlyph;
1346 fontData = &nextFontData;
1347 startPoint.setX(nextX);
1348 startPoint.setY(nextY);
1349 }
1350 nextX += WebCore::width(glyphBuffer.advanceAt(nextGlyph));
1351 nextY += height(glyphBuffer.advanceAt(nextGlyph));
1352 nextGlyph++;
1353 }
1354
1355 if (shouldDrawIfLoading(*fontData, customFontNotReadyAction))
1356 context.drawGlyphs(*fontData, glyphBuffer.glyphs(lastFrom), glyphBuffer.advances(lastFrom), nextGlyph - lastFrom, startPoint, m_fontDescription.fontSmoothing());
1357 point.setX(nextX);
1358}

1375void FontCascade::drawEmphasisMarks(GraphicsContext& context, const GlyphBuffer& glyphBuffer, const AtomString& mark, const FloatPoint& point) const
1376{
1377 ASSERT(glyphBuffer.isFlattened());
1378 std::optional<GlyphData> markGlyphData = getEmphasisMarkGlyphData(mark);
1379 if (!markGlyphData)
1380 return;
1381
1382 const Font* markFontData = markGlyphData.value().font;
1383 ASSERT(markFontData);
1384 if (!markFontData)
1385 return;
1386
1387 Glyph markGlyph = markGlyphData.value().glyph;
1388 Glyph spaceGlyph = markFontData->spaceGlyph();
1389
1390 float middleOfLastGlyph = offsetToMiddleOfGlyphAtIndex(glyphBuffer, 0);
1391 FloatPoint startPoint(point.x() + middleOfLastGlyph - offsetToMiddleOfGlyph(*markFontData, markGlyph), point.y());
1392
1393 GlyphBuffer markBuffer;
1394 for (unsigned i = 0; i + 1 < glyphBuffer.size(); ++i) {
1395 float middleOfNextGlyph = offsetToMiddleOfGlyphAtIndex(glyphBuffer, i + 1);
1396 float advance = WebCore::width(glyphBuffer.advanceAt(i)) - middleOfLastGlyph + middleOfNextGlyph;
1397 markBuffer.add(glyphBuffer.glyphAt(i) ? markGlyph : spaceGlyph, *markFontData, advance);
1398 middleOfLastGlyph = middleOfNextGlyph;
1399 }
1400 markBuffer.add(glyphBuffer.glyphAt(glyphBuffer.size() - 1) ? markGlyph : spaceGlyph, *markFontData, 0);
1401
1402 drawGlyphBuffer(context, markBuffer, startPoint, CustomFontNotReadyAction::DoNotPaintIfFontNotReady);

1471{

1473

1475 GlyphBuffer localGlyphBuffer;

1483 break;
1484 delta += w;

1486 if (delta - w / 2 >= 0)
1487 break;
1488 } else {
1489 if (delta >= 0)
1490 break;
1491 }

1493 } else {
1494 while (1) {
1495 offset = it.currentCharacterIndex();
1496 float w;
1497 if (!it.advanceOneCharacter(w, localGlyphBuffer))
1498 break;
1499 delta -= w;
1500 if (includePartialGlyphs) {
1501 if (delta + w / 2 <= 0)
1502 break;
1503 } else {

1544static std::optional<float> findIntersectionPoint(float y, FloatPoint p1, FloatPoint p2)
1545{
1546 if ((p1.y() < y && p2.y() > y) || (p1.y() > y && p2.y() < y))
1547 return p1.x() + (y - p1.y()) * (p2.x() - p1.x()) / (p2.y() - p1.y());

1551static void updateX(GlyphIterationState& state, float x)
1552{

1557// This function is called by CGPathApply and is therefore invoked for each
1558// contour in a glyph. This function models each contours as a straight line
1559// and calculates the intersections between each pseudo-contour and
1560// two horizontal lines (the upper and lower bounds of an underline) found in
1561// GlyphIterationState::y1 and GlyphIterationState::y2. It keeps track of the
1562// leftmost and rightmost intersection in GlyphIterationState::minX and
1563// GlyphIterationState::maxX.
1564static void findPathIntersections(GlyphIterationState& state, const PathElement& element)
1565{
1566 bool doIntersection = false;
1567 FloatPoint point = FloatPoint();
1568 switch (element.type) {
1569 case PathElement::Type::MoveToPoint:
1570 state.startingPoint = element.points[0];
1571 state.currentPoint = element.points[0];
1572 break;
1573 case PathElement::Type::AddLineToPoint:
1574 doIntersection = true;
1575 point = element.points[0];
1576 break;
1577 case PathElement::Type::AddQuadCurveToPoint:
1578 doIntersection = true;
1579 point = element.points[1];
1580 break;
1581 case PathElement::Type::AddCurveToPoint:
1582 doIntersection = true;
1583 point = element.points[2];
1584 break;
1585 case PathElement::Type::CloseSubpath:
1586 doIntersection = true;
1587 point = state.startingPoint;
1588 break;
1589 }
1590 if (!doIntersection)
1591 return;
1592 if (auto intersectionPoint = findIntersectionPoint(state.y1, state.currentPoint, point))
1593 updateX(state, *intersectionPoint);
1594 if (auto intersectionPoint = findIntersectionPoint(state.y2, state.currentPoint, point))
1595 updateX(state, *intersectionPoint);
1596 if ((state.currentPoint.y() >= state.y1 && state.currentPoint.y() <= state.y2)
1597 || (state.currentPoint.y() <= state.y1 && state.currentPoint.y() >= state.y2))
1598 updateX(state, state.currentPoint.x());
1599 state.currentPoint = point;
1600}

1659DashArray FontCascade::dashesForIntersectionsWithRect(const TextRun& run, const FloatPoint& textOrigin, const FloatRect& lineExtents) const
1660{
1661 if (isLoadingCustomFonts())
1662 return DashArray();
1663
1664 auto glyphBuffer = layoutText(codePath(run), run, 0, run.length());
1665
1666 if (!glyphBuffer.size())
1667 return DashArray();
1668
1669 FloatPoint origin = textOrigin + WebCore::size(glyphBuffer.initialAdvance());
1670 GlyphToPathTranslator translator(run, glyphBuffer, origin);
1671 DashArray result;
1672 for (unsigned index = 0; translator.containsMorePaths(); ++index, translator.advance()) {

1674 switch (translator.underlineType()) {
1675 case GlyphUnderlineType::SkipDescenders: {
1676 Path path = translator.path();
1677 path.apply([&](const PathElement& element) {
1678 findPathIntersections(info, element);
1679 });
1680 if (info.minX < info.maxX) {
1681 result.append(info.minX - lineExtents.x());
1682 result.append(info.maxX - lineExtents.x());
1683 }
1684 break;
1685 }
1686 case GlyphUnderlineType::SkipGlyph: {
1687 std::pair<float, float> extents = translator.extents();
1688 result.append(extents.first - lineExtents.x());
1689 result.append(extents.second - lineExtents.x());
1690 break;
1691 }
1692 case GlyphUnderlineType::DrawOverGlyph:
1693 // Nothing to do
1694 break;
