56struct GlyphOverflow {
57 bool isEmpty() const
58 {
59 return !left && !right && !top && !bottom;
60 }
61
62 void extendTo(const GlyphOverflow& other)
63 {
64 left = std::max(left, other.left);
65 right = std::max(right, other.right);
66 top = std::max(top, other.top);
67 bottom = std::max(bottom, other.bottom);
68 }
69
70 bool operator!=(const GlyphOverflow& other)
71 {
72 // FIXME: Probably should name this rather than making it the != operator since it ignores the value of computeBounds.
73 return left != other.left || right != other.right || top != other.top || bottom != other.bottom;
74 }
75
76 int left { 0 };
77 int right { 0 };
78 int top { 0 };
79 int bottom { 0 };
80 bool computeBounds { false };
81};

254 static bool treatAsSpace(UChar c) { return c == ' ' || c == '\t' || c == '\n' || c == noBreakSpace; }
255 static bool treatAsZeroWidthSpace(UChar c) { return treatAsZeroWidthSpaceInComplexScript(c) || c == 0x200c || c == 0x200d; }
256 static bool treatAsZeroWidthSpaceInComplexScript(UChar c) { return c < 0x20 || (c >= 0x7F && c < 0xA0) || c == softHyphen || c == zeroWidthSpace || (c >= 0x200e && c <= 0x200f) || (c >= 0x202a && c <= 0x202e) || c == zeroWidthNoBreakSpace || c == objectReplacementCharacter; }

259 static inline UChar normalizeSpaces(UChar character)
260 {
261 if (treatAsSpace(character))
262 return space;
263
264 if (treatAsZeroWidthSpace(character))
265 return zeroWidthSpace;
266
267 return character;
268 }

279 bool advancedTextRenderingMode() const
280 {
281 auto textRenderingMode = m_fontDescription.textRenderingMode();
282 if (textRenderingMode == TextRenderingMode::GeometricPrecision || textRenderingMode == TextRenderingMode::OptimizeLegibility)
283 return true;
284 if (textRenderingMode == TextRenderingMode::OptimizeSpeed)
285 return false;
286#if PLATFORM(COCOA) || USE(FREETYPE)
287 return true;
288#else
289 return false;
290#endif
291 }
292
293 bool computeEnableKerning() const
294 {
295 auto kerning = m_fontDescription.kerning();
296 if (kerning == Kerning::Normal)
297 return true;
298 if (kerning == Kerning::NoShift)
299 return false;
300 return advancedTextRenderingMode();
301 }
302
303 bool computeRequiresShaping() const
304 {
305#if PLATFORM(COCOA) || USE(FREETYPE)
306 if (!m_fontDescription.variantSettings().isAllNormal())
307 return true;
308 if (m_fontDescription.featureSettings().size())
309 return true;
310#endif
311 return advancedTextRenderingMode();
312 }

355inline float FontCascade::tabWidth(const Font& font, const TabSize& tabSize, float position) const
356{
357 float baseTabWidth = tabSize.widthInPixels(font.spaceWidth());
358 if (!baseTabWidth)
359 return letterSpacing();
360 float tabDeltaWidth = baseTabWidth - fmodf(position, baseTabWidth);
361 return (tabDeltaWidth < font.spaceWidth() / 2) ? baseTabWidth : tabDeltaWidth;
362}
