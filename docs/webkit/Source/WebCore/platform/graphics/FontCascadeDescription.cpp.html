62FontSelectionValue FontCascadeDescription::lighterWeight(FontSelectionValue weight)
63{
64 if (weight < FontSelectionValue(100))
65 return weight;
66 if (weight < FontSelectionValue(550))
67 return FontSelectionValue(100);
68 if (weight < FontSelectionValue(750))
69 return FontSelectionValue(400);
70 return FontSelectionValue(700);
71}
72
73FontSelectionValue FontCascadeDescription::bolderWeight(FontSelectionValue weight)
74{
75 if (weight < FontSelectionValue(350))
76 return FontSelectionValue(400);
77 if (weight < FontSelectionValue(550))
78 return FontSelectionValue(700);
79 if (weight < FontSelectionValue(900))
80 return FontSelectionValue(900);
81 return weight;
82}

86bool FontCascadeDescription::familiesEqualForTextAutoSizing(const FontCascadeDescription& other) const
87{
88 unsigned thisFamilyCount = familyCount();
89 unsigned otherFamilyCount = other.familyCount();
90
91 if (thisFamilyCount != otherFamilyCount)
92 return false;
93
94 for (unsigned i = 0; i < thisFamilyCount; ++i) {
95 if (!familyNamesAreEqual(familyAt(i), other.familyAt(i)))
96 return false;
97 }
98
99 return true;
100}

102#endif // ENABLE(TEXT_AUTOSIZING)

104bool FontCascadeDescription::familyNamesAreEqual(const AtomString& family1, const AtomString& family2)
105{
106#if PLATFORM(COCOA)
107 if (family1.startsWith('.'))
108 return family1 == family2;
109#endif
110 return ASCIICaseInsensitiveHash::equal(family1, family2);
111}
112
113unsigned FontCascadeDescription::familyNameHash(const AtomString& family)
114{
115#if PLATFORM(COCOA)
116 if (family.startsWith('.'))
117 return family.existingHash();
118#endif
119 return ASCIICaseInsensitiveHash::hash(family);
120}
121
122String FontCascadeDescription::foldedFamilyName(const String& family)
123{
124#if PLATFORM(COCOA)
125 if (family.startsWith('.'))
126 return family;
127#endif
128 return family.convertToASCIILowercase();
129}
