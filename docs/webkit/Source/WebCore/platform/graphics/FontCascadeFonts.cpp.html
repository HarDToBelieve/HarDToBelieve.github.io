38class MixedFontGlyphPage {
39 WTF_MAKE_FAST_ALLOCATED;
40public:
41 MixedFontGlyphPage(const GlyphPage* initialPage)
42 {
43 if (initialPage) {
44 for (unsigned i = 0; i < GlyphPage::size; ++i)
45 setGlyphDataForIndex(i, initialPage->glyphDataForIndex(i));

49 GlyphData glyphDataForCharacter(UChar32 c) const
50 {
51 unsigned index = GlyphPage::indexForCodePoint(c);
52 ASSERT_WITH_SECURITY_IMPLICATION(index < GlyphPage::size);
53 return { m_glyphs[index], m_fonts[index] };
54 }

56 void setGlyphDataForCharacter(UChar32 c, GlyphData glyphData)
57 {
58 setGlyphDataForIndex(GlyphPage::indexForCodePoint(c), glyphData);
59 }
60
61private:
62 void setGlyphDataForIndex(unsigned index, const GlyphData& glyphData)
63 {
64 ASSERT_WITH_SECURITY_IMPLICATION(index < GlyphPage::size);
65 m_glyphs[index] = glyphData.glyph;
66 m_fonts[index] = glyphData.font;
67 }
68
69 Glyph m_glyphs[GlyphPage::size] { };
70 const Font* m_fonts[GlyphPage::size] { };

143static FontRanges realizeNextFallback(const FontCascadeDescription& description, unsigned& index, FontSelector* fontSelector)
144{
145 ASSERT(index < description.effectiveFamilyCount());
146
147 auto& fontCache = FontCache::fontCacheFallingBackToSingleton(fontSelector);
148 while (index < description.effectiveFamilyCount()) {
149 auto visitor = WTF::makeVisitor([&](const AtomString& family) -> FontRanges {
150 if (family.isEmpty())
151 return FontRanges();
152 if (fontSelector) {
153 auto ranges = fontSelector->fontRangesForFamily(description, family);
154 if (!ranges.isNull())
155 return ranges;
156 }
157 if (auto font = fontCache.fontForFamily(description, family))
158 return FontRanges(WTFMove(font));
159 return FontRanges();
160 }, [&](const FontFamilyPlatformSpecification& fontFamilySpecification) -> FontRanges {
161 return fontFamilySpecification.fontRanges(description);
162 });
163 const auto& currentFamily = description.effectiveFamilyAt(index++);
164 auto ranges = WTF::visit(visitor, currentFamily);
165 if (!ranges.isNull())
166 return ranges;
167 }
168 // We didn't find a font. Try to find a similar font using our own specific knowledge about our platform.
169 // For example on OS X, we know to map any families containing the words Arabic, Pashto, or Urdu to the
170 // Geeza Pro font.
171 for (auto& family : description.families()) {
172 if (auto font = fontCache.similarFont(description, family))
173 return FontRanges(WTFMove(font));
174 }
175 return { };
176}

178const FontRanges& FontCascadeFonts::realizeFallbackRangesAt(const FontCascadeDescription& description, unsigned index)
179{
180 if (index < m_realizedFallbackRanges.size())
181 return m_realizedFallbackRanges[index];
182
183 ASSERT(index == m_realizedFallbackRanges.size());
184 ASSERT(FontCache::fontCacheFallingBackToSingleton(m_fontSelector).generation() == m_generation);
185
186 m_realizedFallbackRanges.append(FontRanges());
187 auto& fontRanges = m_realizedFallbackRanges.last();
188
189 if (!index) {
190 fontRanges = realizeNextFallback(description, m_lastRealizedFallbackIndex, m_fontSelector.get());
191 if (fontRanges.isNull() && m_fontSelector)
192 fontRanges = m_fontSelector->fontRangesForFamily(description, familyNamesData->at(FamilyNamesIndex::StandardFamily));
193 if (fontRanges.isNull())
194 fontRanges = FontRanges(FontCache::fontCacheFallingBackToSingleton(m_fontSelector).lastResortFallbackFont(description));
195 return fontRanges;
196 }
197
198 if (m_lastRealizedFallbackIndex < description.effectiveFamilyCount())
199 fontRanges = realizeNextFallback(description, m_lastRealizedFallbackIndex, m_fontSelector.get());
200
201 if (fontRanges.isNull() && m_fontSelector) {
202 ASSERT(m_lastRealizedFallbackIndex >= description.effectiveFamilyCount());
203
204 unsigned fontSelectorFallbackIndex = m_lastRealizedFallbackIndex - description.effectiveFamilyCount();
205 if (fontSelectorFallbackIndex == m_fontSelector->fallbackFontCount())
206 return fontRanges;
207 ++m_lastRealizedFallbackIndex;
208 fontRanges = FontRanges(m_fontSelector->fallbackFontAt(description, fontSelectorFallbackIndex));
209 }
210
211 return fontRanges;
212}

297static GlyphData glyphDataForNonCJKCharacterWithGlyphOrientation(UChar32 character, NonCJKGlyphOrientation orientation, const GlyphData& data)
298{
299 bool syntheticOblique = data.font->platformData().syntheticOblique();
300 if (orientation == NonCJKGlyphOrientation::Upright || shouldIgnoreRotation(character)) {
301 GlyphData uprightData = data.font->uprightOrientationFont().glyphDataForCharacter(character);
302 // If the glyphs are the same, then we know we can just use the horizontal glyph rotated vertically
303 // to be upright. For synthetic oblique, however, we will always return the uprightData to ensure
304 // that non-CJK and CJK runs are broken up. This guarantees that vertical
305 // fonts without isTextOrientationFallback() set contain CJK characters only and thus we can get
306 // the oblique slant correct.
307 if (data.glyph == uprightData.glyph && !syntheticOblique)
308 return data;
309 // The glyphs are distinct, meaning that the font has a vertical-right glyph baked into it. We can't use that
310 // glyph, so we fall back to the upright data and use the horizontal glyph.
311 if (uprightData.font)
312 return uprightData;
313 } else if (orientation == NonCJKGlyphOrientation::Mixed) {
314 GlyphData verticalRightData = data.font->verticalRightOrientationFont().glyphDataForCharacter(character);
315
316 // If there is a baked-in rotated glyph, we will use it unless syntheticOblique is set. If
317 // synthetic oblique is set, we fall back to the horizontal glyph. This guarantees that vertical
318 // fonts without isTextOrientationFallback() set contain CJK characters only and thus we can get
319 // the oblique slant correct.
320 if (data.glyph != verticalRightData.glyph && !syntheticOblique)
321 return data;
322
323 // The glyphs are identical, meaning that we should just use the horizontal glyph.
324 if (verticalRightData.font)
325 return verticalRightData;
326 }
327 return data;
328}

330static const Font* findBestFallbackFont(FontCascadeFonts& fontCascadeFonts, const FontCascadeDescription& description, UChar32 character)
331{
332 for (unsigned fallbackIndex = 0; ; ++fallbackIndex) {
333 auto& fontRanges = fontCascadeFonts.realizeFallbackRangesAt(description, fallbackIndex);
334 if (fontRanges.isNull())
335 break;
336 auto* currentFont = fontRanges.glyphDataForCharacter(character, ExternalResourceDownloadPolicy::Forbid).font;
337 if (!currentFont)
338 currentFont = &fontRanges.fontForFirstRange();
339
340 if (!currentFont->isInterstitial())
341 return currentFont;
342 }
343
344 return nullptr;
345}

347GlyphData FontCascadeFonts::glyphDataForSystemFallback(UChar32 character, const FontCascadeDescription& description, FontVariant variant, bool systemFallbackShouldBeInvisible)
348{
349 const Font* font = findBestFallbackFont(*this, description, character);
350
351 if (!font)
352 font = &realizeFallbackRangesAt(description, 0).fontForFirstRange();
353
354 auto systemFallbackFont = font->systemFallbackFontForCharacter(character, description, m_isForPlatformFont ? IsForPlatformFont::Yes : IsForPlatformFont::No, FontCache::fontCacheFallingBackToSingleton(m_fontSelector));
355 if (!systemFallbackFont)
356 return GlyphData();
357
358 if (systemFallbackShouldBeInvisible)
359 systemFallbackFont = const_cast<Font*>(&systemFallbackFont->invisibleFont());
360
361 if (systemFallbackFont->platformData().orientation() == FontOrientation::Vertical && !systemFallbackFont->hasVerticalGlyphs() && FontCascade::isCJKIdeographOrSymbol(character))
362 variant = BrokenIdeographVariant;
363
364 GlyphData fallbackGlyphData;
365 if (variant == NormalVariant)
366 fallbackGlyphData = systemFallbackFont->glyphDataForCharacter(character);
367 else
368 fallbackGlyphData = systemFallbackFont->variantFont(description, variant)->glyphDataForCharacter(character);
369
370 if (fallbackGlyphData.font && fallbackGlyphData.font->platformData().orientation() == FontOrientation::Vertical && !fallbackGlyphData.font->isTextOrientationFallback()) {
371 if (variant == NormalVariant && !FontCascade::isCJKIdeographOrSymbol(character))
372 fallbackGlyphData = glyphDataForNonCJKCharacterWithGlyphOrientation(character, description.nonCJKGlyphOrientation(), fallbackGlyphData);
373 }
374
375 // Keep the system fallback fonts we use alive.
376 if (fallbackGlyphData.glyph)
377 m_systemFallbackFontSet.add(WTFMove(systemFallbackFont));
378
379 return fallbackGlyphData;
380}

388static void opportunisticallyStartFontDataURLLoading(const FontCascadeDescription& description, FontSelector* fontSelector)
389{
390 // It is a somewhat common practice for a font foundry to break up a single font into two fonts, each having a random half of
391 // the alphabet, and then encoding the two fonts as data: urls (with different font-family names).
392 // Therefore, if these two fonts don't get loaded at (nearly) the same time, there will be a flash of unintelligible text where
393 // only a random half of the letters are visible.
394 // This code attempts to pre-warm these data urls to make them load at closer to the same time. However, font loading is
395 // asynchronous, and this code doesn't actually fix the race - it just makes it more likely for the two fonts to tie in the race.
396 if (!fontSelector)
397 return;
398 for (unsigned i = 0; i < description.familyCount(); ++i)
399 fontSelector->opportunisticallyStartFontDataURLLoading(description, description.familyAt(i));
400}

402GlyphData FontCascadeFonts::glyphDataForVariant(UChar32 character, const FontCascadeDescription& description, FontVariant variant, unsigned fallbackIndex)
403{
404 FallbackVisibility fallbackVisibility = FallbackVisibility::Immaterial;
405 ExternalResourceDownloadPolicy policy = ExternalResourceDownloadPolicy::Allow;
406 GlyphData loadingResult;
407 opportunisticallyStartFontDataURLLoading(description, m_fontSelector.get());
408 for (; ; ++fallbackIndex) {
409 auto& fontRanges = realizeFallbackRangesAt(description, fallbackIndex);
410 if (fontRanges.isNull())
411 break;
412
413 GlyphData data = fontRanges.glyphDataForCharacter(character, policy);
414 if (!data.font)
415 continue;
416
417 if (data.font->isInterstitial()) {
418 policy = ExternalResourceDownloadPolicy::Forbid;
419 if (fallbackVisibility == FallbackVisibility::Immaterial)
420 fallbackVisibility = data.font->visibility() == Font::Visibility::Visible ? FallbackVisibility::Visible : FallbackVisibility::Invisible;
421 if (!loadingResult.font && data.glyph)
422 loadingResult = data;
423 continue;
424 }
425
426 if (fallbackVisibility == FallbackVisibility::Invisible && data.font->visibility() == Font::Visibility::Visible)
427 data.font = &data.font->invisibleFont();
428
429 if (variant == NormalVariant) {
430 if (data.font->platformData().orientation() == FontOrientation::Vertical && !data.font->isTextOrientationFallback()) {
431 if (!FontCascade::isCJKIdeographOrSymbol(character))
432 return glyphDataForNonCJKCharacterWithGlyphOrientation(character, description.nonCJKGlyphOrientation(), data);
433
434 if (!data.font->hasVerticalGlyphs()) {
435 // Use the broken ideograph font data. The broken ideograph font will use the horizontal width of glyphs
436 // to make sure you get a square (even for broken glyphs like symbols used for punctuation).
437 return glyphDataForVariant(character, description, BrokenIdeographVariant, fallbackIndex);
438 }
439 }
440 } else {
441 // The variantFont function should not normally return 0.
442 // But if it does, we will just render the capital letter big.
443 if (const Font* variantFont = data.font->variantFont(description, variant))
444 return variantFont->glyphDataForCharacter(character);
445 }
446
447 return data;
448 }
449
450 if (loadingResult.font)
451 return loadingResult;
452 return glyphDataForSystemFallback(character, description, variant, fallbackVisibility == FallbackVisibility::Invisible);
453}

455static RefPtr<GlyphPage> glyphPageFromFontRanges(unsigned pageNumber, const FontRanges& fontRanges)
456{
457 const Font* font = nullptr;
458 UChar32 pageRangeFrom = pageNumber * GlyphPage::size;
459 UChar32 pageRangeTo = pageRangeFrom + GlyphPage::size - 1;
460 auto policy = ExternalResourceDownloadPolicy::Allow;
461 FallbackVisibility desiredVisibility = FallbackVisibility::Immaterial;
462 for (unsigned i = 0; i < fontRanges.size(); ++i) {
463 auto& range = fontRanges.rangeAt(i);
464 if (range.from() <= pageRangeFrom && pageRangeTo <= range.to()) {
465 font = range.font(policy);
466 if (!font)
467 continue;
468 if (font->isInterstitial()) {
469 if (desiredVisibility == FallbackVisibility::Immaterial) {
470 auto fontVisibility = font->visibility();
471 if (fontVisibility == Font::Visibility::Visible)
472 desiredVisibility = FallbackVisibility::Visible;
473 else {
474 ASSERT(fontVisibility == Font::Visibility::Invisible);
475 desiredVisibility = FallbackVisibility::Invisible;
476 }
477 }
478 font = nullptr;
479 policy = ExternalResourceDownloadPolicy::Forbid;
480 continue;
481 }
482 }
483 break;
484 }
485 if (!font || font->platformData().orientation() == FontOrientation::Vertical)
486 return nullptr;
487
488 if (desiredVisibility == FallbackVisibility::Invisible && font->visibility() == Font::Visibility::Visible)
489 return const_cast<GlyphPage*>(font->invisibleFont().glyphPage(pageNumber));
490 return const_cast<GlyphPage*>(font->glyphPage(pageNumber));
491}

493GlyphData FontCascadeFonts::glyphDataForCharacter(UChar32 c, const FontCascadeDescription& description, FontVariant variant)
494{
495 ASSERT(m_thread ? m_thread->ptr() == &Thread::current() : isMainThread());
496 ASSERT(variant != AutoVariant);
497
498 if (variant != NormalVariant)
499 return glyphDataForVariant(c, description, variant);
500
501 const unsigned pageNumber = GlyphPage::pageNumberForCodePoint(c);
502
503 auto& cacheEntry = pageNumber ? m_cachedPages.add(pageNumber, GlyphPageCacheEntry()).iterator->value : m_cachedPageZero;
504
505 // Initialize cache with a full page of glyph mappings from a single font.
506 if (cacheEntry.isNull())
507 cacheEntry.setSingleFontPage(glyphPageFromFontRanges(pageNumber, realizeFallbackRangesAt(description, 0)));
508
509 GlyphData glyphData = cacheEntry.glyphDataForCharacter(c);
510 if (!glyphData.glyph) {
511 // No glyph, resolve per-character.
512 ASSERT(variant == NormalVariant);
513 glyphData = glyphDataForVariant(c, description, variant);
514 // Cache the results.
515 cacheEntry.setGlyphDataForCharacter(c, glyphData);
516 }
517
518 return glyphData;
519}

521void FontCascadeFonts::pruneSystemFallbacks()
522{
523 if (m_systemFallbackFontSet.isEmpty())
524 return;
525 // Mutable glyph pages may reference fallback fonts.
526 if (m_cachedPageZero.isMixedFont())
527 m_cachedPageZero = { };
528 m_cachedPages.removeIf([](auto& keyAndValue) {
529 return keyAndValue.value.isMixedFont();
530 });
531 m_systemFallbackFontSet.clear();
532}
