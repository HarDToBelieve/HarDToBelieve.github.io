122inline bool FontCascadeFonts::isFixedPitch(const FontCascadeDescription& description)
123{
124 if (m_pitch == UnknownPitch)
125 determinePitch(description);
126 return m_pitch == FixedPitch;
127};

129inline const Font& FontCascadeFonts::primaryFont(const FontCascadeDescription& description)
130{
131 ASSERT(m_thread ? m_thread->ptr() == &Thread::current() : isMainThread());
132 if (!m_cachedPrimaryFont) {
133 auto& primaryRanges = realizeFallbackRangesAt(description, 0);
134 m_cachedPrimaryFont = primaryRanges.glyphDataForCharacter(' ', ExternalResourceDownloadPolicy::Allow).font;
135 if (!m_cachedPrimaryFont)
136 m_cachedPrimaryFont = &primaryRanges.fontForFirstRange();
137 else if (m_cachedPrimaryFont->isInterstitial()) {
138 for (unsigned index = 1; ; ++index) {
139 auto& localRanges = realizeFallbackRangesAt(description, index);
140 if (localRanges.isNull())
141 break;
142 auto* font = localRanges.glyphDataForCharacter(' ', ExternalResourceDownloadPolicy::Forbid).font;
143 if (font && !font->isInterstitial()) {
144 m_cachedPrimaryFont = font;
145 break;
146 }
147 }
148 }
149 }
150 return *m_cachedPrimaryFont;
151}
