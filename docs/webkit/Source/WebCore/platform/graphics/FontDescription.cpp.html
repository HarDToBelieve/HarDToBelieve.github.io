70static AtomString computeSpecializedChineseLocale()
71{
72 for (auto& language : userPreferredLanguages()) {
73 if (startsWithLettersIgnoringASCIICase(language, "zh-"))
74 return language;
75 }
76 return AtomString("zh-hans", AtomString::ConstructFromLiteral); // We have no signal. Pick one option arbitrarily.
77}
78
79static AtomString& cachedSpecializedChineseLocale()
80{
81 static NeverDestroyed<AtomString> specializedChineseLocale;
82 return specializedChineseLocale.get();
83}
84
85static void fontDescriptionLanguageChanged(void*)
86{
87 cachedSpecializedChineseLocale() = computeSpecializedChineseLocale();
88}
89
90static const AtomString& specializedChineseLocale()
91{
92 auto& locale = cachedSpecializedChineseLocale();
93 if (cachedSpecializedChineseLocale().isNull()) {
94 static char forNonNullPointer;
95 addLanguageChangeObserver(&forNonNullPointer, &fontDescriptionLanguageChanged); // We will never remove the observer, so all we need is a non-null pointer.
96 fontDescriptionLanguageChanged(nullptr);
97 }
98 return locale;
99}

101void FontDescription::setSpecifiedLocale(const AtomString& locale)
102{
103 m_specifiedLocale = locale;
104 m_script = localeToScriptCodeForFontSelection(m_specifiedLocale);
105 m_locale = m_script == USCRIPT_HAN ? specializedChineseLocale() : m_specifiedLocale;
106}
