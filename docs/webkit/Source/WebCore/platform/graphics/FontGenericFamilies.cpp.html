35static bool setGenericFontFamilyForScript(ScriptFontFamilyMap& fontMap, const String& family, UScriptCode script)
36{
37 if (family.isEmpty())
38 return fontMap.remove(static_cast<int>(script));
39 auto& familyInMap = fontMap.add(static_cast<int>(script), String { }).iterator->value;
40 if (familyInMap == family)
41 return false;
42 familyInMap = family;
43 return true;
44}

46static inline bool computeUserPrefersSimplified()
47{
48 const Vector<String>& preferredLanguages = userPreferredLanguages();
49 for (auto& language : preferredLanguages) {
50 if (equalIgnoringASCIICase(language, "zh-tw"))
51 return false;
52 if (equalIgnoringASCIICase(language, "zh-cn"))
53 return true;
54 }
55 return true;
56}
57
58static bool& cachedUserPrefersSimplified()
59{
60 static bool cached = true;
61 return cached;
62}
63
64static void languageChanged(void*)
65{
66 cachedUserPrefersSimplified() = computeUserPrefersSimplified();
67}
68
69static const String& genericFontFamilyForScript(const ScriptFontFamilyMap& fontMap, UScriptCode script)
70{
71 ScriptFontFamilyMap::const_iterator it = fontMap.find(static_cast<int>(script));
72 if (it != fontMap.end())
73 return it->value;
74 // Content using USCRIPT_HAN doesn't tell us if we should be using Simplified Chinese or Traditional Chinese. In the
75 // absence of all other signals, we consult with the user's system preferences.
76 if (script == USCRIPT_HAN) {
77 it = fontMap.find(static_cast<int>(cachedUserPrefersSimplified() ? USCRIPT_SIMPLIFIED_HAN : USCRIPT_TRADITIONAL_HAN));
78 if (it != fontMap.end())
79 return it->value;
80 }
81 if (script != USCRIPT_COMMON)
82 return genericFontFamilyForScript(fontMap, USCRIPT_COMMON);
83 return emptyAtom();
84}

86FontGenericFamilies::FontGenericFamilies()
87{
88 static std::once_flag once;
89 std::call_once(once, [&] {
90 addLanguageChangeObserver(&once, &languageChanged);
91 languageChanged(nullptr);
92 });
93}

95FontGenericFamilies FontGenericFamilies::isolatedCopy() const
96{
97 FontGenericFamilies copy;
98 for (auto &keyValue : m_standardFontFamilyMap)
99 copy.m_standardFontFamilyMap.add(keyValue.key, keyValue.value.isolatedCopy());
100 for (auto &keyValue : m_serifFontFamilyMap)
101 copy.m_serifFontFamilyMap.add(keyValue.key, keyValue.value.isolatedCopy());
102 for (auto &keyValue : m_fixedFontFamilyMap)
103 copy.m_fixedFontFamilyMap.add(keyValue.key, keyValue.value.isolatedCopy());
104 for (auto &keyValue : m_sansSerifFontFamilyMap)
105 copy.m_sansSerifFontFamilyMap.add(keyValue.key, keyValue.value.isolatedCopy());
106 for (auto &keyValue : m_cursiveFontFamilyMap)
107 copy.m_cursiveFontFamilyMap.add(keyValue.key, keyValue.value.isolatedCopy());
108 for (auto &keyValue : m_fantasyFontFamilyMap)
109 copy.m_fantasyFontFamilyMap.add(keyValue.key, keyValue.value.isolatedCopy());
110 for (auto &keyValue : m_pictographFontFamilyMap)
111 copy.m_pictographFontFamilyMap.add(keyValue.key, keyValue.value.isolatedCopy());
112 return copy;
113}

185const String* FontGenericFamilies::fontFamily(FamilyNamesIndex family, UScriptCode script) const
186{
187 switch (family) {
188 case FamilyNamesIndex::CursiveFamily:
189 return &cursiveFontFamily(script);
190 case FamilyNamesIndex::FantasyFamily:
191 return &fantasyFontFamily(script);
192 case FamilyNamesIndex::MonospaceFamily:
193 return &fixedFontFamily(script);
194 case FamilyNamesIndex::PictographFamily:
195 return &pictographFontFamily(script);
196 case FamilyNamesIndex::SansSerifFamily:
197 return &sansSerifFontFamily(script);
198 case FamilyNamesIndex::SerifFamily:
199 return &serifFontFamily(script);
200 case FamilyNamesIndex::StandardFamily:
201 return &standardFontFamily(script);
202 case FamilyNamesIndex::SystemUiFamily:
203 return nullptr;
204 }
205
206 ASSERT_NOT_REACHED();
207 return nullptr;
208}
