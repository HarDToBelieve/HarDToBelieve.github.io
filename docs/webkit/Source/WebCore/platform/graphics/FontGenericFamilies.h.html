34namespace WebCore {
35
36// UScriptCode uses -1 and 0 for UScriptInvalidCode and UScriptCommon.
37// We need to use -2 and -3 for empty value and deleted value.
38struct UScriptCodeHashTraits : HashTraits<int> {
39 static const bool emptyValueIsZero = false;
40 static int emptyValue() { return -2; }
41 static void constructDeletedValue(int& slot) { slot = -3; }
42 static bool isDeletedValue(int value) { return value == -3; }
43};
44
45typedef HashMap<int, String, DefaultHash<int>, UScriptCodeHashTraits> ScriptFontFamilyMap;
46
47class FontGenericFamilies {
48 WTF_MAKE_FAST_ALLOCATED;
49public:
50 FontGenericFamilies();
51
52 FontGenericFamilies isolatedCopy() const;
53
54 const String& standardFontFamily(UScriptCode = USCRIPT_COMMON) const;
55 const String& fixedFontFamily(UScriptCode = USCRIPT_COMMON) const;
56 const String& serifFontFamily(UScriptCode = USCRIPT_COMMON) const;
57 const String& sansSerifFontFamily(UScriptCode = USCRIPT_COMMON) const;
58 const String& cursiveFontFamily(UScriptCode = USCRIPT_COMMON) const;
59 const String& fantasyFontFamily(UScriptCode = USCRIPT_COMMON) const;
60 const String& pictographFontFamily(UScriptCode = USCRIPT_COMMON) const;
61
62 const String* fontFamily(WebKitFontFamilyNames::FamilyNamesIndex, UScriptCode = USCRIPT_COMMON) const;
63
64 bool setStandardFontFamily(const String&, UScriptCode);
65 bool setFixedFontFamily(const String&, UScriptCode);
66 bool setSerifFontFamily(const String&, UScriptCode);
67 bool setSansSerifFontFamily(const String&, UScriptCode);
68 bool setCursiveFontFamily(const String&, UScriptCode);
69 bool setFantasyFontFamily(const String&, UScriptCode);
70 bool setPictographFontFamily(const String&, UScriptCode);
71
72private:
73 ScriptFontFamilyMap m_standardFontFamilyMap;
74 ScriptFontFamilyMap m_serifFontFamilyMap;
75 ScriptFontFamilyMap m_fixedFontFamilyMap;
76 ScriptFontFamilyMap m_sansSerifFontFamilyMap;
77 ScriptFontFamilyMap m_cursiveFontFamilyMap;
78 ScriptFontFamilyMap m_fantasyFontFamilyMap;
79 ScriptFontFamilyMap m_pictographFontFamilyMap;
80};
