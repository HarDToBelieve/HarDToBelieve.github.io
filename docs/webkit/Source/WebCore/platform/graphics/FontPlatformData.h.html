199 bool operator==(const FontPlatformData& other) const
200 {
201 return platformIsEqual(other)
202 && m_isHashTableDeletedValue == other.m_isHashTableDeletedValue
203 && m_size == other.m_size
204 && m_syntheticBold == other.m_syntheticBold
205 && m_syntheticOblique == other.m_syntheticOblique
206 && m_isColorBitmapFont == other.m_isColorBitmapFont
207 && m_orientation == other.m_orientation
208 && m_widthVariant == other.m_widthVariant
209 && m_textRenderingMode == other.m_textRenderingMode;
210 }

339 : m_context(context)

341 {
