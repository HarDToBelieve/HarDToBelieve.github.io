26#pragma once
27
28#include "FontRanges.h"
29#include <wtf/Forward.h>
30#include <wtf/RefCounted.h>
31
32namespace WebCore {
33
34class FontCache;
35class FontCascadeDescription;
36class FontDescription;
37class FontSelectorClient;
38
39class FontAccessor : public RefCounted<FontAccessor> {
40public:
41 virtual ~FontAccessor() = default;
42
43 virtual const Font* font(ExternalResourceDownloadPolicy) const = 0;
44 virtual bool isLoading() const = 0;
45};
46
47class FontSelector : public RefCounted<FontSelector> {
48public:
49 virtual ~FontSelector() = default;
50
51 virtual FontRanges fontRangesForFamily(const FontDescription&, const AtomString&) = 0;
52 virtual RefPtr<Font> fallbackFontAt(const FontDescription&, size_t) = 0;
53
54 virtual size_t fallbackFontCount() = 0;
55
56 virtual void opportunisticallyStartFontDataURLLoading(const FontCascadeDescription&, const AtomString& family) = 0;
57
58 virtual FontCache& fontCache() const = 0;
59 virtual void fontCacheInvalidated() { }
60
61 virtual void registerForInvalidationCallbacks(FontSelectorClient&) = 0;
62 virtual void unregisterForInvalidationCallbacks(FontSelectorClient&) = 0;
63
64 virtual unsigned uniqueId() const = 0;
65 virtual unsigned version() const = 0;
66};

68}
