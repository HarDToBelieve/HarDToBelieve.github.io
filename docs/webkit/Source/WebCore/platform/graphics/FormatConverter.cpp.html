189inline unsigned short convertFloatToHalfFloat(float f)
190{
191 unsigned temp = *(reinterpret_cast<unsigned *>(&f));
192 unsigned signexp = (temp >> 23) & 0x1ff;
193 return baseTable[signexp] + ((temp & 0x007fffff) >> shiftTable[signexp]);
194}

234template<> ALWAYS_INLINE void unpack<GraphicsContextGL::DataFormat::BGRA8, uint8_t, uint8_t>(const uint8_t* source, uint8_t* destination, unsigned pixelsPerRow)
235{
236 const uint32_t* source32 = reinterpret_cast_ptr<const uint32_t*>(source);
237 uint32_t* destination32 = reinterpret_cast_ptr<uint32_t*>(destination);
238 for (unsigned int i = 0; i < pixelsPerRow; ++i) {
239 uint32_t bgra = source32[i];
240#if CPU(BIG_ENDIAN)
241 uint32_t brMask = 0xff00ff00;
242 uint32_t gaMask = 0x00ff00ff;
243#else
244 uint32_t brMask = 0x00ff00ff;
245 uint32_t gaMask = 0xff00ff00;
246#endif
247 uint32_t rgba = (((bgra >> 16) | (bgra << 16)) & brMask) | (bgra & gaMask);
248 destination32[i] = rgba;
249 }

252template<> ALWAYS_INLINE void unpack<GraphicsContextGL::DataFormat::RGBA5551, uint16_t, uint8_t>(const uint16_t* source, uint8_t* destination, unsigned pixelsPerRow)
253{
254#if HAVE(ARM_NEON_INTRINSICS)
255 SIMD::unpackOneRowOfRGBA5551ToRGBA8(source, destination, pixelsPerRow);
256#endif
257 for (unsigned int i = 0; i < pixelsPerRow; ++i) {
258 uint16_t packedValue = source[0];
259 uint8_t r = packedValue >> 11;
260 uint8_t g = (packedValue >> 6) & 0x1F;
261 uint8_t b = (packedValue >> 1) & 0x1F;
262 destination[0] = (r << 3) | (r & 0x7);
263 destination[1] = (g << 3) | (g & 0x7);
264 destination[2] = (b << 3) | (b & 0x7);
265 destination[3] = (packedValue & 0x1) ? 0xFF : 0x0;
266 source += 1;
267 destination += 4;
268 }

1706// Visual Studio crashes with a C1063 Fatal Error if everything is inlined.
1707template<GraphicsContextGL::DataFormat SrcFormat, GraphicsContextGL::DataFormat DstFormat, GraphicsContextGL::AlphaOp alphaOp>
1708ALWAYS_INLINE_EXCEPT_MSVC void FormatConverter::convert()
1709{
1710 // Many instantiations of this template function will never be entered, so we
1711 // try to return immediately in these cases to avoid generating useless code.
1712 if (SrcFormat == DstFormat && alphaOp == GraphicsContextGL::AlphaOp::DoNothing) {
1713 ASSERT_NOT_REACHED();
1714 return;
1715 }
1716 if (!IsFloatFormat<DstFormat>::Value && IsFloatFormat<SrcFormat>::Value) {
1717 ASSERT_NOT_REACHED();
1718 return;
1719 }
1720
1721 // Only textures uploaded from DOM elements or ImageData can allow DstFormat != SrcFormat.
1722 const bool srcFormatComesFromDOMElementOrImageData = GraphicsContextGL::srcFormatComesFromDOMElementOrImageData(SrcFormat);
1723 if (!srcFormatComesFromDOMElementOrImageData && SrcFormat != DstFormat) {
1724 ASSERT_NOT_REACHED();
1725 return;
1726 }
1727 // Likewise, only textures uploaded from DOM elements or ImageData can possibly have to be unpremultiplied.
1728 if (!srcFormatComesFromDOMElementOrImageData && alphaOp == GraphicsContextGL::AlphaOp::DoUnmultiply) {
1729 ASSERT_NOT_REACHED();
1730 return;
1731 }
1732 if (srcFormatComesFromDOMElementOrImageData && alphaOp == GraphicsContextGL::AlphaOp::DoUnmultiply && !SupportsConversionFromDomElements<DstFormat>::Value) {
1733 ASSERT_NOT_REACHED();
1734 return;
1735 }
1736 if ((!GraphicsContextGL::hasAlpha(SrcFormat) || !GraphicsContextGL::hasColor(SrcFormat) || !GraphicsContextGL::hasColor(DstFormat)) && alphaOp != GraphicsContextGL::AlphaOp::DoNothing) {
1737 ASSERT_NOT_REACHED();
1738 return;
1739 }
1740 // If converting DOM element data to UNSIGNED_INT_5_9_9_9_REV or
1741 // UNSIGNED_INT_10F_11F_11F_REV, we should always switch to FLOAT instead to
1742 // avoid unpacking/packing these two types.
1743 if (srcFormatComesFromDOMElementOrImageData && SrcFormat != DstFormat
1744 && (DstFormat == GraphicsContextGL::DataFormat::RGB5999 || DstFormat == GraphicsContextGL::DataFormat::RGB10F11F11F)) {
1745 ASSERT_NOT_REACHED();
1746 return;
1747 }
1748
1749 typedef typename DataTypeForFormat<SrcFormat>::Type SrcType;
1750 typedef typename DataTypeForFormat<DstFormat>::Type DstType;
1751 const GraphicsContextGL::DataFormat IntermediateSrcFormat = IntermediateFormat<DstFormat>::Value;
1752 typedef typename DataTypeForFormat<IntermediateSrcFormat>::Type IntermediateSrcType;
1753 // Stride here could be negative.
1754 const ptrdiff_t srcStrideInElements = m_srcStride / sizeof(SrcType);
1755 const ptrdiff_t dstStrideInElements = m_dstStride / sizeof(DstType);
1756 const bool trivialUnpack = SrcFormat == IntermediateSrcFormat;
1757 const bool trivialPack = DstFormat == IntermediateSrcFormat && alphaOp == GraphicsContextGL::AlphaOp::DoNothing;
1758 ASSERT(!trivialUnpack || !trivialPack);
1759
1760 const SrcType* srcRowStart =
1761 static_cast<const SrcType*>(static_cast<const void*>(
1762 static_cast<const uint8_t*>(m_srcStart) +
1763 ((m_srcStride * m_srcSubRectangle.y()) + m_srcRowOffset)));
1764
1765 // If packing multiple images into a 3D texture, and flipY is true,
1766 // then the sub-rectangle is pointing at the start of the
1767 // "bottommost" of those images. Since the source pointer strides in
1768 // the positive direction, we need to back it up to point at the
1769 // last, or "topmost", of these images.
1770 if (m_dstStride < 0 && m_depth > 1)
1771 srcRowStart -= (m_depth - 1) * srcStrideInElements * m_unpackImageHeight;
1772
1773 DstType* dstRowStart = static_cast<DstType*>(m_dstStart);
1774 if (trivialUnpack) {
1775 for (int d = 0; d < m_depth; ++d) {
1776 for (int i = 0; i < m_srcSubRectangle.height(); ++i) {
1777 pack<DstFormat, alphaOp>(srcRowStart, dstRowStart,
1778 m_srcSubRectangle.width());
1779 srcRowStart += srcStrideInElements;
1780 dstRowStart += dstStrideInElements;
1781 }
1782 srcRowStart += srcStrideInElements * (m_unpackImageHeight - m_srcSubRectangle.height());
1783 }
1784 } else if (trivialPack) {
1785 for (int d = 0; d < m_depth; ++d) {
1786 for (int i = 0; i < m_srcSubRectangle.height(); ++i) {
1787 unpack<SrcFormat>(srcRowStart, dstRowStart,
1788 m_srcSubRectangle.width());
1789 srcRowStart += srcStrideInElements;
1790 dstRowStart += dstStrideInElements;
1791 }
1792 srcRowStart += srcStrideInElements * (m_unpackImageHeight - m_srcSubRectangle.height());
1793 }
1794 } else {
1795 for (int d = 0; d < m_depth; ++d) {
1796 for (int i = 0; i < m_srcSubRectangle.height(); ++i) {
1797 unpack<SrcFormat>(srcRowStart,
1798 reinterpret_cast<IntermediateSrcType*>(m_unpackedIntermediateSrcData.get()),
1799 m_srcSubRectangle.width());
1800 pack<DstFormat, alphaOp>(reinterpret_cast<IntermediateSrcType*>(m_unpackedIntermediateSrcData.get()),
1801 dstRowStart, m_srcSubRectangle.width());
1802 srcRowStart += srcStrideInElements;
1803 dstRowStart += dstStrideInElements;
1804 }
1805 srcRowStart += srcStrideInElements * (m_unpackImageHeight - m_srcSubRectangle.height());
1806 }
1807 }
1808 m_success = true;
1809}
