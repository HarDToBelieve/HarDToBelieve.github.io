28#pragma once
29
30#if ENABLE(WEBGL)
31
32#include "GraphicsContextGL.h"
33#include "IntRect.h"
34#include <wtf/StdLibExtras.h>
35#include <wtf/UniqueArray.h>
36
37namespace WebCore {
38
39class FormatConverter {
40public:
41 FormatConverter(
42 const IntRect& sourceDataSubRectangle,
43 int depth,
44 int unpackImageHeight,
45 const void* srcStart,
46 void* dstStart,
47 int srcStride,
48 int srcRowOffset,
49 int dstStride)
50 : m_srcSubRectangle(sourceDataSubRectangle)
51 , m_depth(depth)
52 , m_unpackImageHeight(unpackImageHeight)
53 , m_srcStart(srcStart)
54 , m_dstStart(dstStart)
55 , m_srcStride(srcStride)
56 , m_srcRowOffset(srcRowOffset)
57 , m_dstStride(dstStride)
58 , m_success(false)
59 {
60 const unsigned MaxNumberOfComponents = 4;
61 const unsigned MaxBytesPerComponent = 4;
62 m_unpackedIntermediateSrcData = makeUniqueArray<uint8_t>(Checked<size_t>(m_srcSubRectangle.width()) * MaxNumberOfComponents * MaxBytesPerComponent);
63
64 ASSERT(m_unpackedIntermediateSrcData.get());
65 }
66
67 void convert(GraphicsContextGL::DataFormat srcFormat, GraphicsContextGL::DataFormat dstFormat, GraphicsContextGL::AlphaOp);
68 bool success() const { return m_success; }
69
70private:
71 template<GraphicsContextGL::DataFormat SrcFormat>
72 ALWAYS_INLINE void convert(GraphicsContextGL::DataFormat dstFormat, GraphicsContextGL::AlphaOp);
73
74 template<GraphicsContextGL::DataFormat SrcFormat, GraphicsContextGL::DataFormat DstFormat>
75 ALWAYS_INLINE void convert(GraphicsContextGL::AlphaOp);
76
77 template<GraphicsContextGL::DataFormat SrcFormat, GraphicsContextGL::DataFormat DstFormat, GraphicsContextGL::AlphaOp alphaOp>
78 ALWAYS_INLINE void convert();
79
80 const IntRect& m_srcSubRectangle;
81 const int m_depth;
82 const int m_unpackImageHeight;
83 const void* const m_srcStart;
84 void* const m_dstStart;
85 const int m_srcStride, m_srcRowOffset, m_dstStride;
86 bool m_success;
87 UniqueArray<uint8_t> m_unpackedIntermediateSrcData;
88};
89
90} // namespace WebCore
91
92#endif // ENABLE(WEBGL)
