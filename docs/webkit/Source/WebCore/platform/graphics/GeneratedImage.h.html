26#pragma once
27
28#include "FloatSize.h"
29#include "Image.h"
30
31namespace WebCore {
32
33class GeneratedImage : public Image {
34public:
35 bool hasSingleSecurityOrigin() const override { return true; }
36
37 void setContainerSize(const FloatSize& size) override { m_size = size; }
38 bool usesContainerSize() const override { return true; }
39 bool hasRelativeWidth() const override { return true; }
40 bool hasRelativeHeight() const override { return true; }
41 void computeIntrinsicDimensions(Length& intrinsicWidth, Length& intrinsicHeight, FloatSize& intrinsicRatio) override;
42
43 FloatSize size(ImageOrientation = ImageOrientation::FromImage) const override { return m_size; }
44
45 // Assume that generated content has no decoded data we need to worry about
46 void destroyDecodedData(bool /*destroyAll*/ = true) override { }
47
48protected:
49 ImageDrawResult draw(GraphicsContext&, const FloatRect& dstRect, const FloatRect& srcRect, const ImagePaintingOptions& = { }) override = 0;
50 void drawPattern(GraphicsContext&, const FloatRect& destRect, const FloatRect& srcRect, const AffineTransform& patternTransform, const FloatPoint& phase, const FloatSize& spacing, const ImagePaintingOptions& = { }) override = 0;
51
52 // FIXME: Implement this to be less conservative.
53 bool currentFrameKnownToBeOpaque() const override { return false; }
54
55 GeneratedImage() { }
56
57private:
58 bool isGeneratedImage() const override { return true; }
59
60 FloatSize m_size;
61};
62
63}
