47void Gradient::adjustParametersForTiledDrawing(FloatSize& size, FloatRect& srcRect, const FloatSize& spacing)
48{
49 if (srcRect.isEmpty())
50 return;
51
52 if (!spacing.isZero())
53 return;
54
55 WTF::switchOn(m_data,
56 [&] (const LinearData& data) {
57 if (data.point0.x() == data.point1.x()) {
58 size.setWidth(1);
59 srcRect.setWidth(1);
60 srcRect.setX(0);
61 return;
62 }
63 if (data.point0.y() != data.point1.y())
64 return;
65
66 size.setHeight(1);
67 srcRect.setHeight(1);
68 srcRect.setY(0);
69 },
70 [] (const RadialData&) {
71 },
72 [] (const ConicData&) {
73 }
74 );
75}
76
77bool Gradient::isZeroSize() const
78{
79 return WTF::switchOn(m_data,
80 [] (const LinearData& data) {
81 return data.point0.x() == data.point1.x() && data.point0.y() == data.point1.y();
82 },
83 [] (const RadialData& data) {
84 return data.point0.x() == data.point1.x() && data.point0.y() == data.point1.y() && data.startRadius == data.endRadius;
85 },
86 [] (const ConicData&) {
87 return false;
88 }
89 );
90}

92void Gradient::addColorStop(Gradient::ColorStop&& stop)
93{
94 m_stops.append(WTFMove(stop));
95 m_stopsSorted = false;
96 m_cachedHash = 0;
97 stopsChanged();
98}
99
100void Gradient::setSortedColorStops(ColorStopVector&& stops)
101{
102 m_stops = WTFMove(stops);
103 m_stopsSorted = true;
104 m_cachedHash = 0;
105 stopsChanged();
106}
107
108void Gradient::sortStops() const
109{
110 if (m_stopsSorted)
111 return;
112 m_stopsSorted = true;
113 std::stable_sort(m_stops.begin(), m_stops.end(), [] (auto& a, auto& b) {
114 return a.offset < b.offset;
115 });
116}
117
118void Gradient::setSpreadMethod(GradientSpreadMethod spreadMethod)
119{
120 if (m_spreadMethod == spreadMethod)
121 return;
122 m_spreadMethod = spreadMethod;
123 m_cachedHash = 0;
124}

133
134static void add(Hasher& hasher, const FloatPoint& point)
135{
136 add(hasher, point.x(), point.y());
137}
138
139static void add(Hasher& hasher, const Gradient::ColorStop& stop)
140{
141 add(hasher, stop.offset, stop.color);
142}
143
144static void add(Hasher& hasher, const Gradient::LinearData& data)
145{
146 add(hasher, data.point0, data.point1);
147}
148
149static void add(Hasher& hasher, const Gradient::RadialData& data)
150{
151 add(hasher, data.point0, data.point1, data.startRadius, data.endRadius, data.aspectRatio);
152}
153
154static void add(Hasher& hasher, const Gradient::ConicData& data)
155{
156 add(hasher, data.point0, data.angleRadians);
157}
158
159unsigned Gradient::hash() const
160{
161 if (!m_cachedHash) {
162 sortStops();
163 m_cachedHash = computeHash(m_data, m_spreadMethod, m_stops);
164 }
165 return m_cachedHash;
166}
