44class TextRunIterator {
45public:
46 TextRunIterator()
47 : m_textRun(0)
48 , m_offset(0)
49 {
50 }
51
52 TextRunIterator(const TextRun* textRun, unsigned offset)
53 : m_textRun(textRun)
54 , m_offset(offset)
55 {
56 }
57
58 unsigned offset() const { return m_offset; }
59 void increment() { m_offset++; }
60 bool atEnd() const { return !m_textRun || m_offset >= m_textRun->length(); }
61 UChar current() const { return (*m_textRun)[m_offset]; }
62 UCharDirection direction() const { return atEnd() ? U_OTHER_NEUTRAL : u_charDirection(current()); }
63
64 bool operator==(const TextRunIterator& other)
65 {
66 return m_offset == other.m_offset && m_textRun == other.m_textRun;
67 }
68
69 bool operator!=(const TextRunIterator& other) { return !operator==(other); }
70
71private:
72 const TextRun* m_textRun;
73 unsigned m_offset;
74};

95GraphicsContextState::StateChangeFlags GraphicsContextStateChange::changesFromState(const GraphicsContextState& state) const
96{
97 GraphicsContextState::StateChangeFlags changeFlags;
98
99 CHECK_FOR_CHANGED_PROPERTY(StrokeGradientChange, strokeGradient);
100 CHECK_FOR_CHANGED_PROPERTY(StrokePatternChange, strokePattern);
101 CHECK_FOR_CHANGED_PROPERTY(FillGradientChange, fillGradient);
102 CHECK_FOR_CHANGED_PROPERTY(FillPatternChange, fillPattern);
103
104 if (m_changeFlags.contains(GraphicsContextState::ShadowChange)
105 && (m_state.shadowOffset != state.shadowOffset
106 || m_state.shadowBlur != state.shadowBlur
107 || m_state.shadowColor != state.shadowColor))
108 changeFlags.add(GraphicsContextState::ShadowChange);
109
110 CHECK_FOR_CHANGED_PROPERTY(StrokeThicknessChange, strokeThickness);
111 CHECK_FOR_CHANGED_PROPERTY(TextDrawingModeChange, textDrawingMode);
112 CHECK_FOR_CHANGED_PROPERTY(StrokeColorChange, strokeColor);
113 CHECK_FOR_CHANGED_PROPERTY(FillColorChange, fillColor);
114 CHECK_FOR_CHANGED_PROPERTY(StrokeStyleChange, strokeStyle);
115 CHECK_FOR_CHANGED_PROPERTY(FillRuleChange, fillRule);
116 CHECK_FOR_CHANGED_PROPERTY(AlphaChange, alpha);
117
118 if (m_changeFlags.containsAny({ GraphicsContextState::CompositeOperationChange, GraphicsContextState::BlendModeChange })
119 && (m_state.compositeOperator != state.compositeOperator || m_state.blendMode != state.blendMode)) {
120 changeFlags.add(GraphicsContextState::CompositeOperationChange);
121 changeFlags.add(GraphicsContextState::BlendModeChange);
122 }
123
124 CHECK_FOR_CHANGED_PROPERTY(ShouldAntialiasChange, shouldAntialias);
125 CHECK_FOR_CHANGED_PROPERTY(ShouldSmoothFontsChange, shouldSmoothFonts);
126 CHECK_FOR_CHANGED_PROPERTY(ShouldSubpixelQuantizeFontsChange, shouldSubpixelQuantizeFonts);
127 CHECK_FOR_CHANGED_PROPERTY(ShadowsIgnoreTransformsChange, shadowsIgnoreTransforms);
128 CHECK_FOR_CHANGED_PROPERTY(DrawLuminanceMaskChange, drawLuminanceMask);
129 CHECK_FOR_CHANGED_PROPERTY(ImageInterpolationQualityChange, imageInterpolationQuality);
130
131#if HAVE(OS_DARK_MODE_SUPPORT)
132 CHECK_FOR_CHANGED_PROPERTY(UseDarkAppearanceChange, useDarkAppearance);
133#endif
134
135 return changeFlags;
136}

138void GraphicsContextStateChange::accumulate(const GraphicsContextState& state, GraphicsContextState::StateChangeFlags flags)
139{
140 // FIXME: This code should move to GraphicsContextState.
141 auto strokeFlags = { GraphicsContextState::StrokeColorChange, GraphicsContextState::StrokeGradientChange, GraphicsContextState::StrokePatternChange };
142 if (flags.containsAny(strokeFlags)) {
143 m_state.strokeColor = state.strokeColor;
144 m_state.strokeGradient = state.strokeGradient;
145 m_state.strokePattern = state.strokePattern;
146 m_changeFlags.remove(strokeFlags);
147 }
148
149 auto fillFlags = { GraphicsContextState::FillColorChange, GraphicsContextState::FillGradientChange, GraphicsContextState::FillPatternChange };
150 if (flags.containsAny(fillFlags)) {
151 m_state.fillColor = state.fillColor;
152 m_state.fillGradient = state.fillGradient;
153 m_state.fillPattern = state.fillPattern;
154 m_changeFlags.remove(fillFlags);
155 }
156
157 if (flags.contains(GraphicsContextState::ShadowChange)) {
158 m_state.shadowOffset = state.shadowOffset;
159 m_state.shadowBlur = state.shadowBlur;
160 m_state.shadowColor = state.shadowColor;
161 m_state.shadowRadiusMode = state.shadowRadiusMode;
162 }
163
164 if (flags.contains(GraphicsContextState::StrokeThicknessChange))
165 m_state.strokeThickness = state.strokeThickness;
166
167 if (flags.contains(GraphicsContextState::TextDrawingModeChange))
168 m_state.textDrawingMode = state.textDrawingMode;
169
170 if (flags.contains(GraphicsContextState::StrokeStyleChange))
171 m_state.strokeStyle = state.strokeStyle;
172
173 if (flags.contains(GraphicsContextState::FillRuleChange))
174 m_state.fillRule = state.fillRule;
175
176 if (flags.contains(GraphicsContextState::AlphaChange))
177 m_state.alpha = state.alpha;
178
179 if (flags.containsAny({ GraphicsContextState::CompositeOperationChange, GraphicsContextState::BlendModeChange })) {
180 m_state.compositeOperator = state.compositeOperator;
181 m_state.blendMode = state.blendMode;
182 }
183
184 if (flags.contains(GraphicsContextState::ShouldAntialiasChange))
185 m_state.shouldAntialias = state.shouldAntialias;
186
187 if (flags.contains(GraphicsContextState::ShouldSmoothFontsChange))
188 m_state.shouldSmoothFonts = state.shouldSmoothFonts;
189
190 if (flags.contains(GraphicsContextState::ShouldSubpixelQuantizeFontsChange))
191 m_state.shouldSubpixelQuantizeFonts = state.shouldSubpixelQuantizeFonts;
192
193 if (flags.contains(GraphicsContextState::ShadowsIgnoreTransformsChange))
194 m_state.shadowsIgnoreTransforms = state.shadowsIgnoreTransforms;
195
196 if (flags.contains(GraphicsContextState::DrawLuminanceMaskChange))
197 m_state.drawLuminanceMask = state.drawLuminanceMask;
198
199 if (flags.contains(GraphicsContextState::ImageInterpolationQualityChange))
200 m_state.imageInterpolationQuality = state.imageInterpolationQuality;
201
202#if HAVE(OS_DARK_MODE_SUPPORT)
203 if (flags.contains(GraphicsContextState::UseDarkAppearanceChange))
204 m_state.useDarkAppearance = state.useDarkAppearance;
205#endif
206
207 m_changeFlags.add(flags);
208}

211{

225 context.setShadowsIgnoreTransforms(m_state.shadowsIgnoreTransforms);
226
227 if (m_changeFlags.contains(GraphicsContextState::ShadowChange))
228 context.setShadow(m_state.shadowOffset, m_state.shadowBlur, m_state.shadowColor, m_state.shadowRadiusMode);
229
230 if (m_changeFlags.contains(GraphicsContextState::StrokeThicknessChange))
231 context.setStrokeThickness(m_state.strokeThickness);
232
233 if (m_changeFlags.contains(GraphicsContextState::TextDrawingModeChange))
234 context.setTextDrawingMode(m_state.textDrawingMode);
235
236 if (m_changeFlags.contains(GraphicsContextState::StrokeColorChange))
237 context.setStrokeColor(m_state.strokeColor);

240 context.setFillColor(m_state.fillColor);
241
242 if (m_changeFlags.contains(GraphicsContextState::StrokeStyleChange))
243 context.setStrokeStyle(m_state.strokeStyle);
244
245 if (m_changeFlags.contains(GraphicsContextState::FillRuleChange))
246 context.setFillRule(m_state.fillRule);

248 if (m_changeFlags.contains(GraphicsContextState::AlphaChange))
249 context.setAlpha(m_state.alpha);

251 if (m_changeFlags.containsAny({ GraphicsContextState::CompositeOperationChange, GraphicsContextState::BlendModeChange }))

357void GraphicsContext::save()
358{
359 m_stack.append(m_state);
360}
361
362void GraphicsContext::restore()
363{
364 if (m_stack.isEmpty()) {
365 LOG_ERROR("ERROR void GraphicsContext::restore() stack is empty");
366 return;
367 }
368
369 m_state = m_stack.last();
370 m_stack.removeLast();
371
372 // Make sure we deallocate the state stack buffer when it goes empty.
373 // Canvas elements will immediately save() again, but that goes into inline capacity.
374 if (m_stack.isEmpty())
375 m_stack.clear();
376}

500void GraphicsContext::drawBidiText(const FontCascade& font, const TextRun& run, const FloatPoint& point, FontCascade::CustomFontNotReadyAction customFontNotReadyAction)
501{
502 BidiResolver<TextRunIterator, BidiCharacterRun> bidiResolver;
503 bidiResolver.setStatus(BidiStatus(run.direction(), run.directionalOverride()));
504 bidiResolver.setPositionIgnoringNestedIsolates(TextRunIterator(&run, 0));
505
506 // FIXME: This ownership should be reversed. We should pass BidiRunList
507 // to BidiResolver in createBidiRunsForLine.
508 BidiRunList<BidiCharacterRun>& bidiRuns = bidiResolver.runs();
509 bidiResolver.createBidiRunsForLine(TextRunIterator(&run, run.length()));
510
511 if (!bidiRuns.runCount())
512 return;
513
514 FloatPoint currPoint = point;
515 BidiCharacterRun* bidiRun = bidiRuns.firstRun();
516 while (bidiRun) {
517 TextRun subrun = run.subRun(bidiRun->start(), bidiRun->stop() - bidiRun->start());
518 bool isRTL = bidiRun->level() % 2;
519 subrun.setDirection(isRTL ? TextDirection::RTL : TextDirection::LTR);
520 subrun.setDirectionalOverride(bidiRun->dirOverride(false));
521
522 auto advance = font.drawText(*this, subrun, currPoint, 0, std::nullopt, customFontNotReadyAction);
523 currPoint.move(advance);
524
525 bidiRun = bidiRun->next();
526 }
527
528 bidiRuns.clear();
529}

699void GraphicsContext::adjustLineToPixelBoundaries(FloatPoint& p1, FloatPoint& p2, float strokeWidth, StrokeStyle penStyle)
700{
701 // For odd widths, we add in 0.5 to the appropriate x/y so that the float arithmetic
702 // works out. For example, with a border width of 3, WebKit will pass us (y1+y2)/2, e.g.,
703 // (50+53)/2 = 103/2 = 51 when we want 51.5. It is always true that an even width gave
704 // us a perfect position, but an odd width gave us a position that is off by exactly 0.5.
705 if (penStyle == DottedStroke || penStyle == DashedStroke) {
706 if (p1.x() == p2.x()) {
707 p1.setY(p1.y() + strokeWidth);
708 p2.setY(p2.y() - strokeWidth);
709 } else {
710 p1.setX(p1.x() + strokeWidth);
711 p2.setX(p2.x() - strokeWidth);
712 }
713 }
714
715 if (static_cast<int>(strokeWidth) % 2) { //odd
716 if (p1.x() == p2.x()) {
717 // We're a vertical line. Adjust our x.
718 p1.setX(p1.x() + 0.5f);
719 p2.setX(p2.x() + 0.5f);
720 } else {
721 // We're a horizontal line. Adjust our y.
722 p1.setY(p1.y() + 0.5f);
723 p2.setY(p2.y() + 0.5f);
724 }
725 }
726}

772FloatRect GraphicsContext::computeLineBoundsAndAntialiasingModeForText(const FloatRect& rect, bool printing, Color& color)
773{
774 FloatPoint origin = rect.location();
775 float thickness = std::max(rect.height(), 0.5f);
776 if (printing)
777 return FloatRect(origin, FloatSize(rect.width(), thickness));
778
779 AffineTransform transform = getCTM(GraphicsContext::DefinitelyIncludeDeviceScale);
780 // Just compute scale in x dimension, assuming x and y scales are equal.
781 float scale = transform.b() ? std::hypot(transform.a(), transform.b()) : transform.a();
782 if (scale < 1.0) {
783 // This code always draws a line that is at least one-pixel line high,
784 // which tends to visually overwhelm text at small scales. To counter this
785 // effect, an alpha is applied to the underline color when text is at small scales.
786 static const float minimumUnderlineAlpha = 0.4f;
787 float shade = scale > minimumUnderlineAlpha ? scale : minimumUnderlineAlpha;
788 color = color.colorWithAlphaMultipliedBy(shade);
789 }
790
791 FloatPoint devicePoint = transform.mapPoint(rect.location());
792 // Visual overflow might occur here due to integral roundf/ceilf. visualOverflowForDecorations adjusts the overflow value for underline decoration.
793 FloatPoint deviceOrigin = FloatPoint(roundf(devicePoint.x()), ceilf(devicePoint.y()));
794 if (auto inverse = transform.inverse())
795 origin = inverse.value().mapPoint(deviceOrigin);
796 return FloatRect(origin, FloatSize(rect.width(), thickness));
797}

811float GraphicsContext::dashedLinePatternOffsetForPatternAndStrokeWidth(float patternWidth, float strokeWidth) const
812{
813 // Pattern starts with full fill and ends with the empty fill.
814 // 1. Let's start with the empty phase after the corner.
815 // 2. Check if we've got odd or even number of patterns and whether they fully cover the line.
816 // 3. In case of even number of patterns and/or remainder, move the pattern start position
817 // so that the pattern is balanced between the corners.
818 float patternOffset = patternWidth;
819 int numberOfSegments = std::floor(strokeWidth / patternWidth);
820 bool oddNumberOfSegments = numberOfSegments % 2;
821 float remainder = strokeWidth - (numberOfSegments * patternWidth);
822 if (oddNumberOfSegments && remainder)
823 patternOffset -= remainder / 2.0f;
824 else if (!oddNumberOfSegments) {
825 if (remainder)
826 patternOffset += patternOffset - (patternWidth + remainder) / 2.0f;
827 else
828 patternOffset += patternWidth / 2.0f;
829 }
830
831 return patternOffset;
832}

834Vector<FloatPoint> GraphicsContext::centerLineAndCutOffCorners(bool isVerticalLine, float cornerWidth, FloatPoint point1, FloatPoint point2) const
835{
836 // Center line and cut off corners for pattern painting.
837 if (isVerticalLine) {
838 float centerOffset = (point2.x() - point1.x()) / 2.0f;
839 point1.move(centerOffset, cornerWidth);
840 point2.move(-centerOffset, -cornerWidth);
841 } else {
842 float centerOffset = (point2.y() - point1.y()) / 2.0f;
843 point1.move(cornerWidth, centerOffset);
844 point2.move(-cornerWidth, -centerOffset);
845 }
846
847 return { point1, point2 };
848}
