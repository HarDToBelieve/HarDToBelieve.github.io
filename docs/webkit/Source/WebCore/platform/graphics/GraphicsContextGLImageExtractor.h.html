34class GraphicsContextGLImageExtractor {
35public:
36 using DOMSource = GraphicsContextGL::DOMSource;
37 using DataFormat = GraphicsContextGL::DataFormat;
38 using AlphaOp = GraphicsContextGL::AlphaOp;
39 GraphicsContextGLImageExtractor(Image*, DOMSource, bool premultiplyAlpha, bool ignoreGammaAndColorProfile, bool ignoreNativeImageAlphaPremultiplication);
40
41 // Each platform must provide an implementation of this method to deallocate or release resources
42 // associated with the image if needed.
43 ~GraphicsContextGLImageExtractor();
44
45 bool extractSucceeded() { return m_extractSucceeded; }
46 const void* imagePixelData() { return m_imagePixelData; }
47 unsigned imageWidth() { return m_imageWidth; }
48 unsigned imageHeight() { return m_imageHeight; }
49 DataFormat imageSourceFormat() { return m_imageSourceFormat; }
50 AlphaOp imageAlphaOp() { return m_alphaOp; }
51 unsigned imageSourceUnpackAlignment() { return m_imageSourceUnpackAlignment; }
52 DOMSource imageHtmlDomSource() { return m_imageHtmlDomSource; }
53private:
54 // Each platform must provide an implementation of this method.
55 // Extracts the image and keeps track of its status, such as width, height, Source Alignment, format and AlphaOp etc,
56 // needs to lock the resources or relevant data if needed and returns true upon success
57 bool extractImage(bool premultiplyAlpha, bool ignoreGammaAndColorProfile, bool ignoreNativeImageAlphaPremultiplication);

59#if USE(CAIRO)
60 RefPtr<cairo_surface_t> m_imageSurface;

63 UniqueArray<uint8_t> m_formalizedRGBA8Data;
64#endif
65 Image* m_image;
66 DOMSource m_imageHtmlDomSource;
67 bool m_extractSucceeded;
68 const void* m_imagePixelData;
69 unsigned m_imageWidth;
70 unsigned m_imageHeight;
71 DataFormat m_imageSourceFormat;
72 AlphaOp m_alphaOp;
73 unsigned m_imageSourceUnpackAlignment;
74};
