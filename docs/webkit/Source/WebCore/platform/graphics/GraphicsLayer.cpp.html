54void KeyframeValueList::insert(std::unique_ptr<const AnimationValue> value)
55{
56 for (size_t i = 0; i < m_values.size(); ++i) {
57 const AnimationValue* curValue = m_values[i].get();
58 if (curValue->keyTime() == value->keyTime()) {
59 ASSERT_NOT_REACHED();
60 // insert after
61 m_values.insert(i + 1, WTFMove(value));
62 return;
63 }
64 if (curValue->keyTime() > value->keyTime()) {
65 // insert before
66 m_values.insert(i, WTFMove(value));
67 return;
68 }
69 }
70
71 m_values.append(WTFMove(value));
72}

240bool GraphicsLayer::setChildren(Vector<Ref<GraphicsLayer>>&& newChildren)
241{
242 // If the contents of the arrays are the same, nothing to do.
243 if (newChildren == m_children)
244 return false;
245
246 removeAllChildren();
247
248 size_t listSize = newChildren.size();
249 for (size_t i = 0; i < listSize; ++i)
250 addChild(WTFMove(newChildren[i]));
251
252 return true;
253}

307bool GraphicsLayer::replaceChild(GraphicsLayer* oldChild, Ref<GraphicsLayer>&& newChild)
308{
309 ASSERT(!newChild->parent());
310
311 GraphicsLayer* rawNewChild = newChild.ptr();
312
313 bool found = false;
314 for (unsigned i = 0; i < m_children.size(); i++) {
315 if (oldChild == m_children[i].ptr()) {
316 m_children[i] = WTFMove(newChild);
317 found = true;
318 break;
319 }
320 }
321 if (found) {
322 oldChild->setParent(nullptr);
323
324 rawNewChild->removeFromParent();
325 rawNewChild->setParent(this);
326 return true;
327 }
328 return false;
329}

331void GraphicsLayer::removeAllChildren()
332{
333 while (m_children.size()) {
334 GraphicsLayer* curLayer = m_children[0].ptr();
335 ASSERT(curLayer->parent());
336 curLayer->removeFromParent();
337 // curLayer may be destroyed here.
338 }
339}

341void GraphicsLayer::removeFromParentInternal()
342{
343 if (m_parent) {
344 GraphicsLayer* parent = m_parent;
345 setParent(nullptr);
346 parent->m_children.removeFirstMatching([this](auto& layer) {
347 return layer.ptr() == this;
348 });
349 // |this| may be destroyed here.
350 }

543FloatRect GraphicsLayer::adjustCoverageRectForMovement(const FloatRect& coverageRect, const FloatRect& previousVisibleRect, const FloatRect& currentVisibleRect)
544{
545 // If the old visible rect is empty, we have no information about how the visible area is changing
546 // (maybe the layer was just created), so don't attempt to expand. Also don't attempt to expand if the rects don't overlap.
547 if (previousVisibleRect.isEmpty() || !currentVisibleRect.intersects(previousVisibleRect))
548 return unionRect(coverageRect, currentVisibleRect);
549
550 const float paddingMultiplier = 2;
551
552 float leftEdgeDelta = paddingMultiplier * (currentVisibleRect.x() - previousVisibleRect.x());
553 float rightEdgeDelta = paddingMultiplier * (currentVisibleRect.maxX() - previousVisibleRect.maxX());
554
555 float topEdgeDelta = paddingMultiplier * (currentVisibleRect.y() - previousVisibleRect.y());
556 float bottomEdgeDelta = paddingMultiplier * (currentVisibleRect.maxY() - previousVisibleRect.maxY());
557
558 FloatRect expandedRect = currentVisibleRect;
559
560 // More exposed on left side.
561 if (leftEdgeDelta < 0) {
562 float newLeft = expandedRect.x() + leftEdgeDelta;
563 // Pad to the left, but don't reduce padding that's already in the backing store (since we're still exposing to the left).
564 if (newLeft < previousVisibleRect.x())
565 expandedRect.shiftXEdgeTo(newLeft);
566 else
567 expandedRect.shiftXEdgeTo(previousVisibleRect.x());
568 }
569
570 // More exposed on right.
571 if (rightEdgeDelta > 0) {
572 float newRight = expandedRect.maxX() + rightEdgeDelta;
573 // Pad to the right, but don't reduce padding that's already in the backing store (since we're still exposing to the right).
574 if (newRight > previousVisibleRect.maxX())
575 expandedRect.setWidth(newRight - expandedRect.x());
576 else
577 expandedRect.setWidth(previousVisibleRect.maxX() - expandedRect.x());
578 }
579
580 // More exposed at top.
581 if (topEdgeDelta < 0) {
582 float newTop = expandedRect.y() + topEdgeDelta;
583 if (newTop < previousVisibleRect.y())
584 expandedRect.shiftYEdgeTo(newTop);
585 else
586 expandedRect.shiftYEdgeTo(previousVisibleRect.y());
587 }
588
589 // More exposed on bottom.
590 if (bottomEdgeDelta > 0) {
591 float newBottom = expandedRect.maxY() + bottomEdgeDelta;
592 if (newBottom > previousVisibleRect.maxY())
593 expandedRect.setHeight(newBottom - expandedRect.y());
594 else
595 expandedRect.setHeight(previousVisibleRect.maxY() - expandedRect.y());
596 }
597
598 return unionRect(coverageRect, expandedRect);
599}

615void GraphicsLayer::getDebugBorderInfo(Color& color, float& width) const
616{
617 width = 2;
618
619 if (needsBackdrop()) {
620 color = Color::magenta.colorWithAlphaByte(128); // has backdrop: magenta
621 width = 12;
622 return;
623 }
624
625 if (drawsContent()) {
626 if (tiledBacking()) {
627 color = Color::orange.colorWithAlphaByte(128); // tiled layer: orange
628 return;
629 }
630
631 color = SRGBA<uint8_t> { 0, 128, 32, 128 }; // normal layer: green
632 return;
633 }
634
635 if (usesContentsLayer()) {
636 color = SRGBA<uint8_t> { 0, 64, 128, 150 }; // non-painting layer with contents: blue
637 width = 8;
638 return;
639 }
640
641 if (masksToBounds()) {
642 color = SRGBA<uint8_t> { 128, 255, 255, 48 }; // masking layer: pale blue
643 width = 16;
644 return;
645 }
646
647 color = Color::yellow.colorWithAlphaByte(192); // container: yellow
648}

666static inline const FilterOperations& filterOperationsAt(const KeyframeValueList& valueList, size_t index)
667{
668 return static_cast<const FilterAnimationValue&>(valueList.at(index)).value();
669}
670
671int GraphicsLayer::validateFilterOperations(const KeyframeValueList& valueList)
672{
673#if ENABLE(FILTERS_LEVEL_2)
674 ASSERT(valueList.property() == AnimatedPropertyFilter || valueList.property() == AnimatedPropertyWebkitBackdropFilter);
675#else
676 ASSERT(valueList.property() == AnimatedPropertyFilter);
677#endif
678
679 if (valueList.size() < 2)
680 return -1;
681
682 // Empty filters match anything, so find the first non-empty entry as the reference
683 size_t firstIndex = 0;
684 for ( ; firstIndex < valueList.size(); ++firstIndex) {
685 if (!filterOperationsAt(valueList, firstIndex).operations().isEmpty())
686 break;
687 }
688
689 if (firstIndex >= valueList.size())
690 return -1;
691
692 const FilterOperations& firstVal = filterOperationsAt(valueList, firstIndex);
693
694 for (size_t i = firstIndex + 1; i < valueList.size(); ++i) {
695 const FilterOperations& val = filterOperationsAt(valueList, i);
696
697 // An emtpy filter list matches anything.
698 if (val.operations().isEmpty())
699 continue;
700
701 if (!firstVal.operationsMatch(val))
702 return -1;
703 }
704
705 return firstIndex;
706}

708// An "invalid" list is one whose functions don't match, and therefore has to be animated as a Matrix
709// The hasBigRotation flag will always return false if isValid is false. Otherwise hasBigRotation is
710// true if the rotation between any two keyframes is >= 180 degrees.
711
712static inline const TransformOperations& operationsAt(const KeyframeValueList& valueList, size_t index)
713{
714 return static_cast<const TransformAnimationValue&>(valueList.at(index)).value();
715}
716
717int GraphicsLayer::validateTransformOperations(const KeyframeValueList& valueList, bool& hasBigRotation)
718{
719 ASSERT(animatedPropertyIsTransformOrRelated(valueList.property()));
720
721 hasBigRotation = false;
722
723 if (valueList.size() < 2)
724 return -1;
725
726 // Empty transforms match anything, so find the first non-empty entry as the reference.
727 size_t firstIndex = 0;
728 for ( ; firstIndex < valueList.size(); ++firstIndex) {
729 if (!operationsAt(valueList, firstIndex).operations().isEmpty())
730 break;
731 }
732
733 if (firstIndex >= valueList.size())
734 return -1;
735
736 const TransformOperations& firstVal = operationsAt(valueList, firstIndex);
737
738 // See if the keyframes are valid.
739 for (size_t i = firstIndex + 1; i < valueList.size(); ++i) {
740 const TransformOperations& val = operationsAt(valueList, i);
741
742 // An empty transform list matches anything.
743 if (val.operations().isEmpty())
744 continue;
745
746 if (!firstVal.operationsMatch(val))
747 return -1;
748 }
749
750 // Keyframes are valid, check for big rotations.
751 double lastRotationAngle = 0.0;
752 double maxRotationAngle = -1.0;
753
754 for (size_t j = 0; j < firstVal.operations().size(); ++j) {
755 TransformOperation::OperationType type = firstVal.operations().at(j)->type();
756
757 // if this is a rotation entry, we need to see if any angle differences are >= 180 deg
758 if (type == TransformOperation::ROTATE_X ||
759 type == TransformOperation::ROTATE_Y ||
760 type == TransformOperation::ROTATE_Z ||
761 type == TransformOperation::ROTATE_3D) {
762 lastRotationAngle = downcast<RotateTransformOperation>(*firstVal.operations().at(j)).angle();
763
764 if (maxRotationAngle < 0)
765 maxRotationAngle = fabs(lastRotationAngle);
766
767 for (size_t i = firstIndex + 1; i < valueList.size(); ++i) {
768 const TransformOperations& val = operationsAt(valueList, i);
769 double rotationAngle = val.operations().isEmpty() ? 0 : downcast<RotateTransformOperation>(*val.operations().at(j)).angle();
770 double diffAngle = fabs(rotationAngle - lastRotationAngle);
771 if (diffAngle > maxRotationAngle)
772 maxRotationAngle = diffAngle;
773 lastRotationAngle = rotationAngle;
774 }
775 }
776 }
777
778 hasBigRotation = maxRotationAngle >= 180.0;
779
780 return firstIndex;
781}

797void GraphicsLayer::addRepaintRect(const FloatRect& repaintRect)
798{
799 if (!client().isTrackingRepaints())
800 return;
801
802 FloatRect largestRepaintRect(FloatPoint(), m_size);
803 largestRepaintRect.intersect(repaintRect);
804 RepaintMap::iterator repaintIt = repaintRectMap().find(this);
805 if (repaintIt == repaintRectMap().end()) {
806 Vector<FloatRect> repaintRects;
807 repaintRects.append(largestRepaintRect);
808 repaintRectMap().set(this, repaintRects);
809 } else {
810 Vector<FloatRect>& repaintRects = repaintIt->value;
811 repaintRects.append(largestRepaintRect);
812 }
813}

815void GraphicsLayer::traverse(GraphicsLayer& layer, const WTF::Function<void (GraphicsLayer&)>& traversalFunc)
816{
817 traversalFunc(layer);
818
819 for (auto& childLayer : layer.children())
820 traverse(childLayer.get(), traversalFunc);
821
822 if (auto* replicaLayer = layer.replicaLayer())
823 traverse(*replicaLayer, traversalFunc);
824
825 if (auto* maskLayer = layer.maskLayer())
826 traverse(*maskLayer, traversalFunc);

843static void dumpChildren(TextStream& ts, const Vector<Ref<GraphicsLayer>>& children, unsigned& totalChildCount, LayerTreeAsTextBehavior behavior)

853 totalChildCount--;
