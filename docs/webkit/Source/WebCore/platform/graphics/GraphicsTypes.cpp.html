38static const char* const compositeOperatorNames[] = {
39 "clear",
40 "copy",
41 "source-over",
42 "source-in",
43 "source-out",
44 "source-atop",
45 "destination-over",
46 "destination-in",
47 "destination-out",
48 "destination-atop",
49 "xor",
50 "darker",
51 "lighter",
52 "difference"
53};

55static const char* const blendOperatorNames[] = {
56 "normal",
57 "multiply",
58 "screen",
59 "darken",
60 "lighten",
61 "overlay",
62 "color-dodge",
63 "color-burn",
64 "hard-light",
65 "soft-light",
66 "difference",
67 "exclusion",
68 "hue",
69 "saturation",
70 "color",
71 "luminosity",
72 "plus-darker",
73 "plus-lighter"
74};
75const uint8_t numCompositeOperatorNames = WTF_ARRAY_LENGTH(compositeOperatorNames);
76const uint8_t numBlendOperatorNames = WTF_ARRAY_LENGTH(blendOperatorNames);
77
78bool parseBlendMode(const String& s, BlendMode& blendMode)
79{
80 for (unsigned i = 0; i < numBlendOperatorNames; i++) {
81 if (s == blendOperatorNames[i]) {
82 blendMode = static_cast<BlendMode>(i + static_cast<unsigned>(BlendMode::Normal));
83 return true;
84 }
85 }
86
87 return false;
88}
89
90bool parseCompositeAndBlendOperator(const String& s, CompositeOperator& op, BlendMode& blendOp)
91{
92 for (int i = 0; i < numCompositeOperatorNames; i++) {
93 if (s == compositeOperatorNames[i]) {
94 op = static_cast<CompositeOperator>(i);
95 blendOp = BlendMode::Normal;
96 return true;
97 }
98 }
99
100 if (parseBlendMode(s, blendOp)) {
101 // For now, blending will always assume source-over. This will be fixed in the future
102 op = CompositeOperator::SourceOver;
103 return true;
104 }
105
106 return false;
107}
108
109// FIXME: when we support blend modes in combination with compositing other than source-over
110// this routine needs to be updated.
111String compositeOperatorName(CompositeOperator op, BlendMode blendOp)
112{
113 ASSERT(op >= CompositeOperator::Clear);
114 ASSERT(static_cast<uint8_t>(op) < numCompositeOperatorNames);
115 ASSERT(blendOp >= BlendMode::Normal);
116 ASSERT(static_cast<uint8_t>(blendOp) <= numBlendOperatorNames);
117 if (blendOp > BlendMode::Normal)
118 return blendOperatorNames[static_cast<unsigned>(blendOp) - static_cast<unsigned>(BlendMode::Normal)];
119 return compositeOperatorNames[static_cast<unsigned>(op)];
120}
121
122String blendModeName(BlendMode blendOp)
123{
124 ASSERT(blendOp >= BlendMode::Normal);
125 ASSERT(blendOp <= BlendMode::PlusLighter);
126 return blendOperatorNames[static_cast<unsigned>(blendOp) - static_cast<unsigned>(BlendMode::Normal)];
127}
128
129TextStream& operator<<(TextStream& ts, CompositeOperator op)
130{
131 return ts << compositeOperatorName(op, BlendMode::Normal);
132}
133
134TextStream& operator<<(TextStream& ts, BlendMode blendMode)
135{
136 return ts << blendModeName(blendMode);
137}

139TextStream& operator<<(TextStream& ts, WindRule rule)
140{
141 switch (rule) {
142 case WindRule::NonZero:
143 ts << "NON-ZERO";
144 break;
145 case WindRule::EvenOdd:
146 ts << "EVEN-ODD";
147 break;
148 }
149
150 return ts;
151}

153TextStream& operator<<(TextStream& ts, LineCap capStyle)
154{
155 switch (capStyle) {
156 case ButtCap:
157 ts << "BUTT";
158 break;
159 case RoundCap:
160 ts << "ROUND";
161 break;
162 case SquareCap:
163 ts << "SQUARE";
164 break;
165 }
166 return ts;
167}

180 break;
181 }
182 return ts;
