35std::optional<HEVCParameters> parseHEVCCodecParameters(StringView codecString)
36{
37 // The format of the 'hevc' codec string is specified in ISO/IEC 14496-15:2014, Annex E.3.
38 StringView codecView(codecString);
39 auto codecSplit = codecView.split('.');
40 auto nextElement = codecSplit.begin();
41 if (nextElement == codecSplit.end())
42 return std::nullopt;
43
44 HEVCParameters parameters;
45
46 // Codec identifier: legal values are specified in ISO/IEC 14496-15:2014, section 8:
47 auto codecName = *nextElement;
48 if (codecName != "hvc1" && codecName != "hev1")
49 return std::nullopt;
50
51 if (++nextElement == codecSplit.end())
52 return std::nullopt;
53
54 // First element: Optional General Profile Space parameter ['A', 'B', 'C'], mapping to [1, 2, 3]
55 // and [0] for absent, then General Profile IDC as a 5-bit decimal number.
56 auto profileSpace = *nextElement;
57 if (!profileSpace.length())
58 return std::nullopt;
59
60 auto firstCharacter = profileSpace[0];
61 bool hasProfileSpace = firstCharacter >= 'A' && firstCharacter <= 'C';
62 if (hasProfileSpace) {
63 parameters.generalProfileSpace = 1 + (firstCharacter - 'A');
64 profileSpace = profileSpace.substring(1);
65 }
66
67 auto profileIDC = parseInteger<uint8_t>(profileSpace);
68 if (!profileIDC)
69 return std::nullopt;
70 parameters.generalProfileIDC = *profileIDC;
71
72 if (++nextElement == codecSplit.end())
73 return std::nullopt;
74
75 // Second element: 32 bit of General Profile Compatibility Flags, in reverse bit order,
76 // in hex with leading zeros omitted.
77 auto compatibilityFlags = parseInteger<uint32_t>(*nextElement, 16);
78 if (!compatibilityFlags)
79 return std::nullopt;
80 parameters.generalProfileCompatibilityFlags = *compatibilityFlags;
81
82 if (++nextElement == codecSplit.end())
83 return std::nullopt;
84
85 // Third element: General Tier Flag ['L', 'H'], mapping to [false, true], followed by
86 // General Level IDC as a 8-bit decimal number.
87 auto generalTier = *nextElement;
88 firstCharacter = generalTier[0];
89 if (firstCharacter != 'L' && firstCharacter != 'H')
90 return std::nullopt;
91
92 auto generalLevelIDC = parseInteger<uint8_t>(generalTier.substring(1));
93 if (!generalLevelIDC)
94 return std::nullopt;
95 parameters.generalLevelIDC = *generalLevelIDC;
96
97 // Optional fourth and remaining elements: a sequence of 6 1-byte constraint flags, each byte encoded
98 // in hex, and separated by a period, with trailing zero bytes omitted.
99 for (unsigned i = 0; i < 6; ++i) {
100 if (++nextElement == codecSplit.end())
101 break;
102 if (!parseInteger<uint8_t>(*nextElement, 16))
103 return std::nullopt;
104 }
105
106 return parameters;
107}

109template<typename ValueType> inline std::optional<ValueType> makeOptionalFromPointer(const ValueType* pointer)
110{
111 if (!pointer)
112 return std::nullopt;
113 return *pointer;
114}

116static std::optional<DoViParameters::Codec> parseDoViCodecType(StringView string)
117{
118 static constexpr std::pair<PackedASCIILowerCodes<uint32_t>, DoViParameters::Codec> typesArray[] = {
119 { "dva1", DoViParameters::Codec::AVC1 },
120 { "dvav", DoViParameters::Codec::AVC3 },
121 { "dvh1", DoViParameters::Codec::HVC1 },
122 { "dvhe", DoViParameters::Codec::HEV1 },
123 };
124 static constexpr SortedArrayMap typesMap { typesArray };
125 return makeOptionalFromPointer(typesMap.tryGet(string));
126}

128static std::optional<uint16_t> profileIDForAlphabeticDoViProfile(StringView profile)
129{
130 // See Table 7 of "Dolby Vision Profiles and Levels Version 1.3.2"
131 static constexpr std::pair<PackedASCIILowerCodes<uint64_t>, uint16_t> profilesArray[] = {
132 { "dvav.se", 9 },
133 { "dvhe.dtb", 7 },
134 { "dvhe.dtr", 4 },
135 { "dvhe.st", 8 },
136 { "dvhe.stn", 5 },
137 };
138 static constexpr SortedArrayMap profilesMap { profilesArray };
139 return makeOptionalFromPointer(profilesMap.tryGet(profile));
140}

156static std::optional<uint16_t> maximumLevelIDForDoViProfileID(uint16_t profileID)
157{
158 // See Section 4.1 of "Dolby Vision Profiles and Levels Version 1.3.2"
159 switch (profileID) {
160 case 4: return 9;
161 case 5: return 13;
162 case 7: return 9;
163 case 8: return 13;
164 case 9: return 5;
165 default: return std::nullopt;
166 }
167}

169static bool isValidProfileIDForCodec(uint16_t profileID, DoViParameters::Codec codec)
170{
171 if (profileID == 9)
172 return codec == DoViParameters::Codec::AVC1 || codec == DoViParameters::Codec::AVC3;
173 return codec == DoViParameters::Codec::HVC1 || codec == DoViParameters::Codec::HEV1;
174}

176std::optional<DoViParameters> parseDoViCodecParameters(StringView codecView)
177{
178 // The format of the DoVi codec string is specified in "Dolby Vision Profiles and Levels Version 1.3.2"
179 auto codecSplit = codecView.split('.');
180 auto nextElement = codecSplit.begin();
181 if (nextElement == codecSplit.end())
182 return std::nullopt;
183
184 DoViParameters parameters;
185
186 auto codec = parseDoViCodecType(*nextElement);
187 if (!codec)
188 return std::nullopt;
189 parameters.codec = *codec;
190
191 if (++nextElement == codecSplit.end())
192 return std::nullopt;
193
194 auto profileID = *nextElement;
195 if (!profileID.length())
196 return std::nullopt;
197
198 auto firstCharacter = profileID[0];
199 // Profile definition can either be numeric or alpha:
200 if (firstCharacter == '0') {
201 auto bitstreamProfileID = parseInteger<uint8_t>(profileID);
202 if (!bitstreamProfileID)
203 return std::nullopt;
204 parameters.bitstreamProfileID = *bitstreamProfileID;
205 } else {
206 auto bitstreamProfileID = profileIDForAlphabeticDoViProfile(codecView.left(5 + profileID.length()));
207 if (!bitstreamProfileID)
208 return std::nullopt;
209 parameters.bitstreamProfileID = *bitstreamProfileID;
210 }
211
212 if (!isValidDoViProfileID(parameters.bitstreamProfileID))
213 return std::nullopt;
214
215 if (!isValidProfileIDForCodec(parameters.bitstreamProfileID, parameters.codec))
216 return std::nullopt;
217
218 if (++nextElement == codecSplit.end())
219 return std::nullopt;
220
221 auto bitstreamLevelID = parseInteger<uint8_t>(*nextElement);
222 if (!bitstreamLevelID)
223 return std::nullopt;
224 parameters.bitstreamLevelID = *bitstreamLevelID;
225
226 auto maximumLevelID = maximumLevelIDForDoViProfileID(parameters.bitstreamProfileID);
227 if (!maximumLevelID || parameters.bitstreamLevelID > *maximumLevelID)
228 return std::nullopt;
229
230 return parameters;
231}
