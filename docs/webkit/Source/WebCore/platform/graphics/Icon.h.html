38namespace WebCore {
39
40class GraphicsContext;
41class FloatRect;
42class NativeImage;
43
44class Icon : public RefCounted<Icon> {
45public:
46 WEBCORE_EXPORT static RefPtr<Icon> createIconForFiles(const Vector<String>& filenames);
47
48 WEBCORE_EXPORT ~Icon();
49
50 void paint(GraphicsContext&, const FloatRect&);
51
52#if PLATFORM(WIN)
53 static Ref<Icon> create(HICON hIcon) { return adoptRef(*new Icon(hIcon)); }
54#endif
55
56#if PLATFORM(IOS_FAMILY)
57 // FIXME: Make this work for non-iOS ports and remove the PLATFORM(IOS_FAMILY)-guard.
58 WEBCORE_EXPORT static RefPtr<Icon> createIconForImage(PlatformImagePtr&&);
59#endif
60
61#if PLATFORM(MAC)
62 static RefPtr<Icon> createIconForUTI(const String&);
63 static RefPtr<Icon> createIconForFileExtension(const String&);
64#endif
65
66private:
67#if PLATFORM(IOS_FAMILY)
68 Icon(RefPtr<NativeImage>&&);
69 RefPtr<NativeImage> m_cgImage;
70#elif PLATFORM(MAC)
71 Icon(NSImage*);
72 RetainPtr<NSImage> m_nsImage;
73#elif PLATFORM(WIN)
74 Icon(HICON);
75 HICON m_hIcon;
76#endif
77};
