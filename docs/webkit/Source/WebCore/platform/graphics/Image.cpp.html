65RefPtr<Image> Image::create(ImageObserver& observer)
66{
67 // SVGImage and PDFDocumentImage are not safe to use off the main thread.
68 // Workers can use BitmapImage directly.
69 ASSERT(isMainThread());
70
71 auto mimeType = observer.mimeType();
72 if (mimeType == "image/svg+xml")
73 return SVGImage::create(observer);
74
75 auto url = observer.sourceUrl();
76 if (isPDFResource(mimeType, url) || isPostScriptResource(mimeType, url)) {
77#if USE(CG) && !USE(WEBKIT_IMAGE_DECODERS)
78 return PDFDocumentImage::create(&observer);
79#else
80 return nullptr;
81#endif
82 }
83
84 return BitmapImage::create(&observer);
85}

100{
101 if (mimeType.isEmpty())

103 return MIMETypeRegistry::isPostScriptMIMEType(mimeType);
104}

133void Image::fillWithSolidColor(GraphicsContext& ctxt, const FloatRect& dstRect, const Color& color, CompositeOperator op)
134{
135 if (!color.isVisible())
136 return;
137
138 CompositeOperator previousOperator = ctxt.compositeOperation();
139 ctxt.setCompositeOperation(color.isOpaque() && op == CompositeOperator::SourceOver ? CompositeOperator::Copy : op);
140 ctxt.fillRect(dstRect, color);
141 ctxt.setCompositeOperation(previousOperator);
142}

144void Image::drawPattern(GraphicsContext& ctxt, const FloatRect& destRect, const FloatRect& tileRect, const AffineTransform& patternTransform, const FloatPoint& phase, const FloatSize& spacing, const ImagePaintingOptions& options)
145{
146 auto tileImage = preTransformedNativeImageForCurrentFrame(options.orientation() == ImageOrientation::FromImage);
147 if (!tileImage)
148 return;
149
150 ctxt.drawPattern(*tileImage, size(), destRect, tileRect, patternTransform, phase, spacing, options);
151
152 if (imageObserver())
153 imageObserver()->didDraw(*this);
154}

156ImageDrawResult Image::drawTiled(GraphicsContext& ctxt, const FloatRect& destRect, const FloatPoint& srcPoint, const FloatSize& scaledTileSize, const FloatSize& spacing, const ImagePaintingOptions& options)
157{
158 Color color = singlePixelSolidColor();
159 if (color.isValid()) {
160 fillWithSolidColor(ctxt, destRect, color, options.compositeOperator());
161 return ImageDrawResult::DidDraw;
162 }
163
164 ASSERT(!isBitmapImage() || notSolidColor());
165
166 FloatSize intrinsicTileSize = size();
167 if (hasRelativeWidth())
168 intrinsicTileSize.setWidth(scaledTileSize.width());
169 if (hasRelativeHeight())
170 intrinsicTileSize.setHeight(scaledTileSize.height());
171
172 FloatSize scale(scaledTileSize / intrinsicTileSize);
173
174 FloatRect oneTileRect;
175 FloatSize actualTileSize = scaledTileSize + spacing;
176 oneTileRect.setX(destRect.x() + fmodf(fmodf(-srcPoint.x(), actualTileSize.width()) - actualTileSize.width(), actualTileSize.width()));
177 oneTileRect.setY(destRect.y() + fmodf(fmodf(-srcPoint.y(), actualTileSize.height()) - actualTileSize.height(), actualTileSize.height()));
178 oneTileRect.setSize(scaledTileSize);
179
180 // Check and see if a single draw of the image can cover the entire area we are supposed to tile.
181 if (oneTileRect.contains(destRect) && !ctxt.drawLuminanceMask()) {
182 FloatRect visibleSrcRect;
183 visibleSrcRect.setX((destRect.x() - oneTileRect.x()) / scale.width());
184 visibleSrcRect.setY((destRect.y() - oneTileRect.y()) / scale.height());
185 visibleSrcRect.setWidth(destRect.width() / scale.width());
186 visibleSrcRect.setHeight(destRect.height() / scale.height());
187 return draw(ctxt, destRect, visibleSrcRect, options);
188 }
189
190#if PLATFORM(IOS_FAMILY)
191 // FIXME: We should re-test this and remove this iOS behavior difference if possible.
192 // When using accelerated drawing on iOS, it's faster to stretch an image than to tile it.
193 if (ctxt.renderingMode() == RenderingMode::Accelerated) {
194 if (size().width() == 1 && intersection(oneTileRect, destRect).height() == destRect.height()) {
195 FloatRect visibleSrcRect;
196 visibleSrcRect.setX(0);
197 visibleSrcRect.setY((destRect.y() - oneTileRect.y()) / scale.height());
198 visibleSrcRect.setWidth(1);
199 visibleSrcRect.setHeight(destRect.height() / scale.height());
200 return draw(ctxt, destRect, visibleSrcRect, { options, BlendMode::Normal });
201 }
202 if (size().height() == 1 && intersection(oneTileRect, destRect).width() == destRect.width()) {
203 FloatRect visibleSrcRect;
204 visibleSrcRect.setX((destRect.x() - oneTileRect.x()) / scale.width());
205 visibleSrcRect.setY(0);
206 visibleSrcRect.setWidth(destRect.width() / scale.width());
207 visibleSrcRect.setHeight(1);
208 return draw(ctxt, destRect, visibleSrcRect, { options, BlendMode::Normal });
209 }
210 }
211#endif
212
213 // Patterned images and gradients can use lots of memory for caching when the
214 // tile size is large (<rdar://problem/4691859>, <rdar://problem/6239505>).
215 // Memory consumption depends on the transformed tile size which can get
216 // larger than the original tile if user zooms in enough.
217#if PLATFORM(IOS_FAMILY)
218 const float maxPatternTilePixels = 512 * 512;
219#else
220 const float maxPatternTilePixels = 2048 * 2048;
221#endif
222 FloatRect transformedTileSize = ctxt.getCTM().mapRect(FloatRect(FloatPoint(), scaledTileSize));
223 float transformedTileSizePixels = transformedTileSize.width() * transformedTileSize.height();
224 FloatRect currentTileRect = oneTileRect;
225 if (transformedTileSizePixels > maxPatternTilePixels) {
226 GraphicsContextStateSaver stateSaver(ctxt);
227 ctxt.clip(destRect);
228
229 currentTileRect.shiftYEdgeTo(destRect.y());
230 float toY = currentTileRect.y();
231 ImageDrawResult result = ImageDrawResult::DidNothing;
232 while (toY < destRect.maxY()) {
233 currentTileRect.shiftXEdgeTo(destRect.x());
234 float toX = currentTileRect.x();
235 while (toX < destRect.maxX()) {
236 FloatRect toRect(toX, toY, currentTileRect.width(), currentTileRect.height());
237 FloatRect fromRect(toFloatPoint(currentTileRect.location() - oneTileRect.location()), currentTileRect.size());
238 fromRect.scale(1 / scale.width(), 1 / scale.height());
239
240 result = draw(ctxt, toRect, fromRect, { options, BlendMode::Normal });
241 if (result == ImageDrawResult::DidRequestDecoding)
242 return result;
243 toX += currentTileRect.width();
244 currentTileRect.shiftXEdgeTo(oneTileRect.x());
245 }
246 toY += currentTileRect.height();
247 currentTileRect.shiftYEdgeTo(oneTileRect.y());
248 }
249 return result;
250 }
251
252 AffineTransform patternTransform = AffineTransform().scaleNonUniform(scale.width(), scale.height());
253 FloatRect tileRect(FloatPoint(), intrinsicTileSize);
254 drawPattern(ctxt, destRect, tileRect, patternTransform, oneTileRect.location(), spacing, options);
255 startAnimation();
256 return ImageDrawResult::DidDraw;
257}

259// FIXME: Merge with the other drawTiled eventually, since we need a combination of both for some things.
260ImageDrawResult Image::drawTiled(GraphicsContext& ctxt, const FloatRect& dstRect, const FloatRect& srcRect, const FloatSize& tileScaleFactor, TileRule hRule, TileRule vRule, const ImagePaintingOptions& options)
261{
262 Color color = singlePixelSolidColor();
263 if (color.isValid()) {
264 fillWithSolidColor(ctxt, dstRect, color, options.compositeOperator());
265 return ImageDrawResult::DidDraw;
266 }
267
268 FloatSize tileScale = tileScaleFactor;
269 FloatSize spacing;
270
271 // FIXME: These rules follow CSS border-image rules, but they should not be down here in Image.
272 bool centerOnGapHorizonally = false;
273 bool centerOnGapVertically = false;
274 switch (hRule) {
275 case RoundTile: {
276 int numItems = std::max<int>(floorf(dstRect.width() / srcRect.width()), 1);
277 tileScale.setWidth(dstRect.width() / (srcRect.width() * numItems));
278 break;
279 }
280 case SpaceTile: {
281 int numItems = floorf(dstRect.width() / srcRect.width());
282 if (!numItems)
283 return ImageDrawResult::DidNothing;
284 spacing.setWidth((dstRect.width() - srcRect.width() * numItems) / (numItems + 1));
285 tileScale.setWidth(1);
286 centerOnGapHorizonally = !(numItems & 1);
287 break;
288 }
289 case StretchTile:
290 case RepeatTile:
291 break;
292 }
293
294 switch (vRule) {
295 case RoundTile: {
296 int numItems = std::max<int>(floorf(dstRect.height() / srcRect.height()), 1);
297 tileScale.setHeight(dstRect.height() / (srcRect.height() * numItems));
298 break;
299 }
300 case SpaceTile: {
301 int numItems = floorf(dstRect.height() / srcRect.height());
302 if (!numItems)
303 return ImageDrawResult::DidNothing;
304 spacing.setHeight((dstRect.height() - srcRect.height() * numItems) / (numItems + 1));
305 tileScale.setHeight(1);
306 centerOnGapVertically = !(numItems & 1);
307 break;
308 }
309 case StretchTile:
310 case RepeatTile:
311 break;
312 }
313
314 AffineTransform patternTransform = AffineTransform().scaleNonUniform(tileScale.width(), tileScale.height());
315
316 // We want to construct the phase such that the pattern is centered (when stretch is not
317 // set for a particular rule).
318 float hPhase = tileScale.width() * srcRect.x();
319 float vPhase = tileScale.height() * srcRect.y();
320 float scaledTileWidth = tileScale.width() * srcRect.width();
321 float scaledTileHeight = tileScale.height() * srcRect.height();
322
323 if (centerOnGapHorizonally)
324 hPhase -= spacing.width();
325 else if (hRule == Image::RepeatTile || hRule == Image::SpaceTile)
326 hPhase -= (dstRect.width() - scaledTileWidth) / 2;
327
328 if (centerOnGapVertically)
329 vPhase -= spacing.height();
330 else if (vRule == Image::RepeatTile || vRule == Image::SpaceTile)
331 vPhase -= (dstRect.height() - scaledTileHeight) / 2;
332
333 FloatPoint patternPhase(dstRect.x() - hPhase, dstRect.y() - vPhase);
334 drawPattern(ctxt, dstRect, srcRect, patternTransform, patternPhase, spacing, options);
335 startAnimation();
336 return ImageDrawResult::DidDraw;
337}

339void Image::computeIntrinsicDimensions(Length& intrinsicWidth, Length& intrinsicHeight, FloatSize& intrinsicRatio)
340{
341 intrinsicRatio = size();
342 intrinsicWidth = Length(intrinsicRatio.width(), LengthType::Fixed);
343 intrinsicHeight = Length(intrinsicRatio.height(), LengthType::Fixed);

346void Image::startAnimationAsynchronously()
347{
348 if (!m_animationStartTimer)
349 m_animationStartTimer = makeUnique<Timer>(*this, &Image::startAnimation);
350 if (m_animationStartTimer->isActive())
351 return;
352 m_animationStartTimer->startOneShot(0_s);
353}

355void Image::dump(TextStream& ts) const
356{
357 if (isAnimated())
358 ts.dumpProperty("animated", isAnimated());
359
360 if (isNull())
361 ts.dumpProperty("is-null-image", true);
362
363 ts.dumpProperty("size", size());
364}
365
366TextStream& operator<<(TextStream& ts, const Image& image)
367{
368 TextStream::GroupScope scope(ts);
369
370 if (image.isBitmapImage())
371 ts << "bitmap image";
372 else if (image.isCrossfadeGeneratedImage())
373 ts << "crossfade image";
374 else if (image.isNamedImageGeneratedImage())
375 ts << "named image";
376 else if (image.isGradientImage())
377 ts << "gradient image";
378 else if (image.isSVGImage())
379 ts << "svg image";
380 else if (image.isPDFDocumentImage())
381 ts << "pdf image";
382
383 image.dump(ts);
384 return ts;
385}
