76// This class gets notified when an image creates or destroys decoded frames and when it advances animation frames.
77class ImageObserver;
78
79class Image : public RefCounted<Image> {
80 friend class GraphicsContext;
81public:
82 virtual ~Image();
83
84 WEBCORE_EXPORT static Ref<Image> loadPlatformResource(const char* name);
85 WEBCORE_EXPORT static RefPtr<Image> create(ImageObserver&);
86 WEBCORE_EXPORT static bool supportsType(const String&);
87 static bool isPDFResource(const String& mimeType, const URL&);
88 static bool isPostScriptResource(const String& mimeType, const URL&);
89
90 virtual bool isBitmapImage() const { return false; }
91 virtual bool isGeneratedImage() const { return false; }
92 virtual bool isCrossfadeGeneratedImage() const { return false; }
93 virtual bool isNamedImageGeneratedImage() const { return false; }
94 virtual bool isGradientImage() const { return false; }
95 virtual bool isSVGImage() const { return false; }
96 virtual bool isPDFDocumentImage() const { return false; }
97 virtual bool isCustomPaintImage() const { return false; }
98
99 virtual bool currentFrameKnownToBeOpaque() const = 0;
100 virtual bool isAnimated() const { return false; }
101
102 // Derived classes should override this if they can assure that
103 // the image contains only resources from its own security origin.
104 virtual bool hasSingleSecurityOrigin() const { return false; }
105
106 WEBCORE_EXPORT static Image& nullImage();
107 bool isNull() const { return size().isEmpty(); }
108
109 virtual void setContainerSize(const FloatSize&) { }
110 virtual bool usesContainerSize() const { return false; }
111 virtual bool hasRelativeWidth() const { return false; }
112 virtual bool hasRelativeHeight() const { return false; }
113 virtual void computeIntrinsicDimensions(Length& intrinsicWidth, Length& intrinsicHeight, FloatSize& intrinsicRatio);
114
115 virtual FloatSize size(ImageOrientation = ImageOrientation::FromImage) const = 0;
116 virtual FloatSize sourceSize(ImageOrientation orientation = ImageOrientation::FromImage) const { return size(orientation); }
117 virtual bool hasDensityCorrectedSize() const { return false; }
118 FloatRect rect() const { return FloatRect(FloatPoint(), size()); }
119 float width() const { return size().width(); }
120 float height() const { return size().height(); }
121 virtual std::optional<IntPoint> hotSpot() const { return std::nullopt; }
122 virtual ImageOrientation orientation() const { return ImageOrientation::FromImage; }
123
124 WEBCORE_EXPORT EncodedDataStatus setData(RefPtr<SharedBuffer>&& data, bool allDataReceived);
125 virtual EncodedDataStatus dataChanged(bool /*allDataReceived*/) { return EncodedDataStatus::Unknown; }
126
127 virtual String uti() const { return String(); } // null string if unknown
128 virtual String filenameExtension() const { return String(); } // null string if unknown
129 virtual String accessibilityDescription() const { return String(); } // null string if unknown
130
131 virtual void destroyDecodedData(bool destroyAll = true) = 0;
132
133 SharedBuffer* data() { return m_encodedImageData.get(); }
134 const SharedBuffer* data() const { return m_encodedImageData.get(); }
135
136 // Animation begins whenever someone draws the image, so startAnimation() is not normally called.
137 // It will automatically pause once all observers no longer want to render the image anywhere.
138 virtual void startAnimation() { }
139 void startAnimationAsynchronously();
140 virtual void stopAnimation() {}
141 virtual void resetAnimation() {}
142 virtual bool isAnimating() const { return false; }
143 bool animationPending() const { return m_animationStartTimer && m_animationStartTimer->isActive(); }
144
145 // Typically the CachedImage that owns us.
146 ImageObserver* imageObserver() const { return m_imageObserver; }
147 void setImageObserver(ImageObserver* observer) { m_imageObserver = observer; }
148 URL sourceURL() const;
149 String mimeType() const;
150 long long expectedContentLength() const;
151
152 enum TileRule { StretchTile, RoundTile, SpaceTile, RepeatTile };
153
154 virtual RefPtr<NativeImage> nativeImage(const GraphicsContext* = nullptr) { return nullptr; }
155 virtual RefPtr<NativeImage> nativeImageForCurrentFrame(const GraphicsContext* = nullptr) { return nullptr; }
156 virtual RefPtr<NativeImage> preTransformedNativeImageForCurrentFrame(bool = true, const GraphicsContext* = nullptr) { return nullptr; }
157 virtual RefPtr<NativeImage> nativeImageOfSize(const IntSize&, const GraphicsContext* = nullptr) { return nullptr; }
158
159 // Accessors for native image formats.
160
161#if USE(APPKIT)
162 virtual NSImage *nsImage() { return nullptr; }
163 virtual RetainPtr<NSImage> snapshotNSImage() { return nullptr; }
164#endif
165
166#if PLATFORM(COCOA)
167 virtual CFDataRef tiffRepresentation() { return nullptr; }
168#endif
169
170#if PLATFORM(WIN)
171 virtual bool getHBITMAP(HBITMAP) { return false; }
172 virtual bool getHBITMAPOfSize(HBITMAP, const IntSize*) { return false; }
173#endif
174
175#if PLATFORM(GTK)
176 virtual GdkPixbuf* getGdkPixbuf() { return nullptr; }
177#if USE(GTK4)
178 virtual GdkTexture* gdkTexture() { return nullptr; }
179#endif
180#endif
181
182 virtual void drawPattern(GraphicsContext&, const FloatRect& destRect, const FloatRect& srcRect, const AffineTransform& patternTransform, const FloatPoint& phase, const FloatSize& spacing, const ImagePaintingOptions& = { });
183
184#if ASSERT_ENABLED
185 virtual bool notSolidColor() { return true; }
186#endif
187
188 virtual void dump(WTF::TextStream&) const;
189
190protected:
191 Image(ImageObserver* = nullptr);
192
193 static void fillWithSolidColor(GraphicsContext&, const FloatRect& dstRect, const Color&, CompositeOperator);
194
195#if PLATFORM(WIN)
196 virtual void drawFrameMatchingSourceSize(GraphicsContext&, const FloatRect& dstRect, const IntSize& srcSize, CompositeOperator) { }
197#endif
198 virtual ImageDrawResult draw(GraphicsContext&, const FloatRect& dstRect, const FloatRect& srcRect, const ImagePaintingOptions& = { }) = 0;
199 ImageDrawResult drawTiled(GraphicsContext&, const FloatRect& dstRect, const FloatPoint& srcPoint, const FloatSize& tileSize, const FloatSize& spacing, const ImagePaintingOptions& = { });
200 ImageDrawResult drawTiled(GraphicsContext&, const FloatRect& dstRect, const FloatRect& srcRect, const FloatSize& tileScaleFactor, TileRule hRule, TileRule vRule, const ImagePaintingOptions& = { });
201
202 // Supporting tiled drawing
203 virtual Color singlePixelSolidColor() const { return Color(); }
204
205private:
206 RefPtr<SharedBuffer> m_encodedImageData;
207 ImageObserver* m_imageObserver;
208 std::unique_ptr<Timer> m_animationStartTimer;
209};
