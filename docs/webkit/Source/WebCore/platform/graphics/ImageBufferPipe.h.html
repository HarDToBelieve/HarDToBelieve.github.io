29#include "PlatformLayer.h"
30#include <wtf/RefCounted.h>
31#include <wtf/RefPtr.h>
32#include <wtf/ThreadSafeRefCounted.h>
33
34namespace WebCore {
35
36class ImageBuffer;
37
38// Used to provide a PlatformLayer for an externally managed ImageBuffer; e.g. an ImageBuffer created and owned by a Worker thread
39class ImageBufferPipe : public RefCounted<ImageBufferPipe> {
40public:
41 class Source : public ThreadSafeRefCounted<Source> {
42 public:
43 virtual ~Source() = default;
44
45 virtual void handle(RefPtr<ImageBuffer>&&) = 0;
46 };
47
48 static RefPtr<ImageBufferPipe> create();
49
50 virtual ~ImageBufferPipe() = default;
51
52 virtual RefPtr<Source> source() const = 0;
53 virtual PlatformLayer* platformLayer() const = 0;
54};
55
56} // namespace WebCore
