26#pragma once
27
28#include "AffineTransform.h"
29#include "FloatSize.h"
30#include "IntSize.h"
31
32namespace WebCore {
33
34struct ImageResolution {
35 static constexpr unsigned DefaultResolution = 72;
36
37 // Since this class is to be used mainly with EXIF,
38 // this range intentionally matches the resolution values from the EXIF spec.
39 // See JEITA CP-3451, page 18. http://www.exif.org/Exif2-2.PDF
40
41 enum ResolutionUnit {
42 None = 1,
43 Inches = 2,
44 Centimeters = 3
45 };
46
47 struct ResolutionMetadata {
48 FloatSize preferredSize;
49 FloatSize resolution;
50 ResolutionUnit resolutionUnit;
51 };
52
53 static std::optional<IntSize> densityCorrectedSize(const FloatSize& sourceSize, const ResolutionMetadata&);
54};
55
56} // namespace WebCore
