191 // Asynchronous image decoding.
192 struct ImageFrameRequest {
193 size_t index;
194 SubsamplingLevel subsamplingLevel;
195 DecodingOptions decodingOptions;
196 DecodingStatus decodingStatus;
197 bool operator==(const ImageFrameRequest& other) const
198 {
199 return index == other.index && subsamplingLevel == other.subsamplingLevel && decodingOptions == other.decodingOptions && decodingStatus == other.decodingStatus;
200 }
201 };
202 using FrameRequestQueue = SynchronizedFixedQueue<ImageFrameRequest, BufferSize>;
203 using FrameCommitQueue = Deque<ImageFrameRequest, BufferSize>;
