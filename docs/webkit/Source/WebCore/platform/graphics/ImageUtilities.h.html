28#include <wtf/Forward.h>
29#include <wtf/WorkQueue.h>

31namespace WebCore {
32
33WEBCORE_EXPORT WorkQueue& sharedImageTranscodingQueue();
34
35// Given a list of files' 'paths' and 'allowedMIMETypes', the function returns a list
36// of strings whose size is the same as the size of 'paths' and its entries are all
37// null strings except the ones whose MIME types are not in 'allowedMIMETypes'.
38WEBCORE_EXPORT Vector<String> findImagesForTranscoding(const Vector<String>& paths, const Vector<String>& allowedMIMETypes);
39
40// Given a list of images' 'paths', this function transcodes these images to a new
41// format whose UTI is destinationUTI. The result of the transcoding will be written
42// to temporary files whose extensions are 'destinationExtension'. It returns a list
43// of paths to the result temporary files. If an entry in 'paths' is null or an error
44// happens while transcoding, a null string will be added to the returned list.
45WEBCORE_EXPORT Vector<String> transcodeImages(const Vector<String>& paths, const String& destinationUTI, const String& destinationExtension);

47} // namespace WebCore
