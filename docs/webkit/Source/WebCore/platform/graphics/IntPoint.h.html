60namespace WebCore {

62class FloatPoint;
63
64class IntPoint {
65public:
66 IntPoint() : m_x(0), m_y(0) { }
67 IntPoint(int x, int y) : m_x(x), m_y(y) { }
68 explicit IntPoint(const IntSize& size) : m_x(size.width()), m_y(size.height()) { }
69 WEBCORE_EXPORT explicit IntPoint(const FloatPoint&); // don't do this implicitly since it's lossy
70
71 static IntPoint zero() { return IntPoint(); }
72 bool isZero() const { return !m_x && !m_y; }
73
74 int x() const { return m_x; }
75 int y() const { return m_y; }
76
77 void setX(int x) { m_x = x; }
78 void setY(int y) { m_y = y; }
79
80 void move(const IntSize& s) { move(s.width(), s.height()); }
81 void moveBy(const IntPoint& offset) { move(offset.x(), offset.y()); }
82 void move(int dx, int dy) { m_x += dx; m_y += dy; }
83 void scale(float sx, float sy)
84 {
85 m_x = lroundf(static_cast<float>(m_x * sx));
86 m_y = lroundf(static_cast<float>(m_y * sy));
87 }
88
89 void scale(float scale)
90 {
91 this->scale(scale, scale);
92 }
93
94 IntPoint expandedTo(const IntPoint& other) const
95 {
96 return {
97 m_x > other.m_x ? m_x : other.m_x,
98 m_y > other.m_y ? m_y : other.m_y
99 };
100 }
101
102 IntPoint shrunkTo(const IntPoint& other) const
103 {
104 return {
105 m_x < other.m_x ? m_x : other.m_x,
106 m_y < other.m_y ? m_y : other.m_y
107 };
108 }
109
110 WEBCORE_EXPORT IntPoint constrainedBetween(const IntPoint& min, const IntPoint& max) const;
111
112 int distanceSquaredToPoint(const IntPoint&) const;
113
114 void clampNegativeToZero()
115 {
116 *this = expandedTo(zero());
117 }
118
119 IntPoint transposedPoint() const
120 {
121 return IntPoint(m_y, m_x);
122 }
123
124#if USE(CG)
125 WEBCORE_EXPORT explicit IntPoint(const CGPoint&); // don't do this implicitly since it's lossy
126 WEBCORE_EXPORT operator CGPoint() const;
127#endif
128
129#if !PLATFORM(IOS_FAMILY)
130#if OS(DARWIN) && !defined(NSGEOMETRY_TYPES_SAME_AS_CGGEOMETRY_TYPES)
131 WEBCORE_EXPORT explicit IntPoint(const NSPoint&); // don't do this implicitly since it's lossy
132 WEBCORE_EXPORT operator NSPoint() const;
133#endif
134#endif // !PLATFORM(IOS_FAMILY)
135
136#if PLATFORM(WIN)
137 IntPoint(const POINT&);
138 operator POINT() const;
139 IntPoint(const POINTS&);
140 operator POINTS() const;
141
142 IntPoint(const D2D1_POINT_2U&);
143 explicit IntPoint(const D2D1_POINT_2F&); // Don't do this implicitly, since it's lossy.
144 operator D2D1_POINT_2F() const;
145 operator D2D1_POINT_2U() const;
146#endif
147
148private:
149 int m_x, m_y;
150};
151
152inline IntPoint& operator+=(IntPoint& a, const IntSize& b)
153{
154 a.move(b.width(), b.height());
155 return a;
156}
157
158inline IntPoint& operator-=(IntPoint& a, const IntSize& b)
159{
160 a.move(-b.width(), -b.height());
161 return a;
162}
163
164inline IntPoint operator+(const IntPoint& a, const IntSize& b)
165{
166 return IntPoint(a.x() + b.width(), a.y() + b.height());
167}
168
169inline IntPoint operator+(const IntPoint& a, const IntPoint& b)
170{
171 return IntPoint(a.x() + b.x(), a.y() + b.y());
172}
173
174inline IntSize operator-(const IntPoint& a, const IntPoint& b)
175{
176 return IntSize(a.x() - b.x(), a.y() - b.y());
177}
178
179inline IntPoint operator-(const IntPoint& a, const IntSize& b)
180{
181 return IntPoint(a.x() - b.width(), a.y() - b.height());
182}
183
184inline IntPoint operator-(const IntPoint& point)
185{
186 return IntPoint(-point.x(), -point.y());
187}
188
189inline bool operator==(const IntPoint& a, const IntPoint& b)
190{
191 return a.x() == b.x() && a.y() == b.y();
192}
193
194inline bool operator!=(const IntPoint& a, const IntPoint& b)
195{
196 return a.x() != b.x() || a.y() != b.y();
197}
198
199inline IntSize toIntSize(const IntPoint& a)
200{
201 return IntSize(a.x(), a.y());
202}
203
204inline int IntPoint::distanceSquaredToPoint(const IntPoint& point) const
205{
206 return ((*this) - point).diagonalLengthSquared();
207}
