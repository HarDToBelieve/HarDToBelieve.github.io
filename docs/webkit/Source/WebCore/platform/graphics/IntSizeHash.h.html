20#ifndef IntSizeHash_h
21#define IntSizeHash_h
22
23#include "IntSize.h"
24#include <wtf/HashTraits.h>
25
26namespace WTF {
27
28 template<> struct IntHash<WebCore::IntSize> {
29 static unsigned hash(const WebCore::IntSize& key) { return pairIntHash(key.width(), key.height()); }
30 static bool equal(const WebCore::IntSize& a, const WebCore::IntSize& b) { return a == b; }
31 static const bool safeToCompareToEmptyOrDeleted = true;
32 };
33 template<> struct DefaultHash<WebCore::IntSize> : IntHash<WebCore::IntSize> { };
34
35 template<> struct HashTraits<WebCore::IntSize> : GenericHashTraits<WebCore::IntSize> {
36 static const bool emptyValueIsZero = true;
37 static void constructDeletedValue(WebCore::IntSize& slot) { new (NotNull, &slot) WebCore::IntSize(-1, -1); }
38 static bool isDeletedValue(const WebCore::IntSize& value) { return value.width() == -1 && value.height() == -1; }
39 };
40} // namespace WTF
41
42#endif
