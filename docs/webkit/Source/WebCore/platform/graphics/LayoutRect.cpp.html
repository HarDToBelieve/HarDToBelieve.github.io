39LayoutRect::LayoutRect(const FloatRect& r)
40 : m_location(LayoutPoint(r.location()))
41 , m_size(LayoutSize(r.size()))
42{
43}

45bool LayoutRect::intersects(const LayoutRect& other) const
46{
47 // Checking emptiness handles negative widths as well as zero.
48 return !isEmpty() && !other.isEmpty()
49 && x() < other.maxX() && other.x() < maxX()
50 && y() < other.maxY() && other.y() < maxY();
51}
52
53bool LayoutRect::contains(const LayoutRect& other) const
54{
55 return x() <= other.x() && maxX() >= other.maxX()
56 && y() <= other.y() && maxY() >= other.maxY();
57}
58
59void LayoutRect::intersect(const LayoutRect& other)
60{
61 LayoutPoint newLocation(std::max(x(), other.x()), std::max(y(), other.y()));
62 LayoutPoint newMaxPoint(std::min(maxX(), other.maxX()), std::min(maxY(), other.maxY()));
63
64 // Return a clean empty rectangle for non-intersecting cases.
65 if (newLocation.x() >= newMaxPoint.x() || newLocation.y() >= newMaxPoint.y()) {
66 newLocation = LayoutPoint(0, 0);
67 newMaxPoint = LayoutPoint(0, 0);
68 }
69
70 m_location = newLocation;
71 m_size = newMaxPoint - newLocation;
72}
73
74bool LayoutRect::edgeInclusiveIntersect(const LayoutRect& other)
75{
76 LayoutPoint newLocation(std::max(x(), other.x()), std::max(y(), other.y()));
77 LayoutPoint newMaxPoint(std::min(maxX(), other.maxX()), std::min(maxY(), other.maxY()));
78
79 bool intersects = true;
80
81 // Return a clean empty rectangle for non-intersecting cases.
82 if (newLocation.x() > newMaxPoint.x() || newLocation.y() > newMaxPoint.y()) {
83 newLocation = { };
84 newMaxPoint = { };
85 intersects = false;
86 }
87
88 m_location = newLocation;
89 m_size = newMaxPoint - newLocation;
90 return intersects;
91}

93void LayoutRect::unite(const LayoutRect& other)
94{
95 // Handle empty special cases first.
96 if (other.isEmpty())
97 return;
98 if (isEmpty()) {
99 *this = other;
100 return;
101 }
102
103 uniteEvenIfEmpty(other);
104}
105
106void LayoutRect::uniteEvenIfEmpty(const LayoutRect& other)
107{
108 auto minX = std::min(x(), other.x());
109 auto minY = std::min(y(), other.y());
110 auto maxX = std::max(this->maxX(), other.maxX());
111 auto maxY = std::max(this->maxY(), other.maxY());
112
113 setLocationAndSizeFromEdges(minX, minY, maxX, maxY);
114}
115
116bool LayoutRect::checkedUnite(const LayoutRect& other)
117{
118 if (other.isEmpty())
119 return true;
120 if (isEmpty()) {
121 *this = other;
122 return true;
123 }
124 if (!isMaxXMaxYRepresentable() || !other.isMaxXMaxYRepresentable())
125 return false;
126 FloatPoint topLeft = FloatPoint(std::min<float>(x(), other.x()), std::min<float>(y(), other.y()));
127 FloatPoint bottomRight = FloatPoint(std::max<float>(maxX(), other.maxX()), std::max<float>(maxY(), other.maxY()));
128 FloatSize size = bottomRight - topLeft;
129
130 if (size.width() >= LayoutUnit::nearlyMax() || size.height() >= LayoutUnit::nearlyMax())
131 return false;
132 m_location = LayoutPoint(topLeft);
133 m_size = LayoutSize(size);
134 return true;
135}

137void LayoutRect::uniteIfNonZero(const LayoutRect& other)
138{
139 // Handle empty special cases first.
140 if (!other.width() && !other.height())
141 return;
142 if (!width() && !height()) {
143 *this = other;
144 return;
145 }
146
147 LayoutPoint newLocation(std::min(x(), other.x()), std::min(y(), other.y()));
148 LayoutPoint newMaxPoint(std::max(maxX(), other.maxX()), std::max(maxY(), other.maxY()));
149
150 m_location = newLocation;
151 m_size = newMaxPoint - newLocation;
152}

159void LayoutRect::scale(float xScale, float yScale)
160{
161 if (isInfinite())
162 return;
163 m_location.scale(xScale, yScale);
164 m_size.scale(xScale, yScale);
165}

167LayoutRect unionRect(const Vector<LayoutRect>& rects)
168{
169 LayoutRect result;
170
171 size_t count = rects.size();
172 for (size_t i = 0; i < count; ++i)
173 result.unite(rects[i]);
174
175 return result;
176}

178IntRect enclosingIntRect(const LayoutRect& rect)
179{
180 // Empty rects with fractional x, y values turn into non-empty rects when converting to enclosing.
181 // We need to ensure that empty rects stay empty after the conversion, because the selection code expects them to be empty.
182 IntPoint location = flooredIntPoint(rect.minXMinYCorner());
183 IntPoint maxPoint = IntPoint(rect.width() ? rect.maxX().ceil() : location.x(), rect.height() ? rect.maxY().ceil() : location.y());
184 return IntRect(location, maxPoint - location);
185}
186
187LayoutRect enclosingLayoutRect(const FloatRect& rect)
188{
189 LayoutPoint location = flooredLayoutPoint(rect.minXMinYCorner());
190 LayoutPoint maxPoint = ceiledLayoutPoint(rect.maxXMaxYCorner());
191
192 return LayoutRect(location, maxPoint - location);
193}

195FloatRect encloseRectToDevicePixels(const LayoutRect& rect, float pixelSnappingFactor)
196{
197 FloatPoint location = floorPointToDevicePixels(rect.minXMinYCorner(), pixelSnappingFactor);
198 FloatPoint maxPoint = ceilPointToDevicePixels(rect.maxXMaxYCorner(), pixelSnappingFactor);
199
200 return FloatRect(location, maxPoint - location);
201}

203TextStream& operator<<(TextStream& ts, const LayoutRect& r)
204{
205 if (ts.hasFormattingFlag(TextStream::Formatting::LayoutUnitsAsIntegers))
206 return ts << snappedIntRect(r);
207
208 return ts << FloatRect(r);
209}
