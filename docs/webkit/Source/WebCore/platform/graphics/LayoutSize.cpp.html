31namespace WebCore {
32
33LayoutSize LayoutSize::constrainedBetween(const LayoutSize& min, const LayoutSize& max) const
34{
35 return {
36 std::max(min.width(), std::min(max.width(), m_width)),
37 std::max(min.height(), std::min(max.height(), m_height))
38 };
39}
40
41TextStream& operator<<(TextStream& ts, const LayoutSize& size)
42{
43 return ts << "width=" << size.width().toFloat() << " height=" << size.height().toFloat();
44}
