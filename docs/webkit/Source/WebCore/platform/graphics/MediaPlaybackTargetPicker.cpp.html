36static const Seconds pendingActionInterval { 100_ms };
37
38MediaPlaybackTargetPicker::MediaPlaybackTargetPicker(Client& client)
39 : m_client(&client)
40 , m_pendingActionTimer(RunLoop::main(), this, &MediaPlaybackTargetPicker::pendingActionTimerFired)
41{
42}
43
44MediaPlaybackTargetPicker::~MediaPlaybackTargetPicker()
45{
46 m_pendingActionTimer.stop();
47 m_client = nullptr;
48}
49
50void MediaPlaybackTargetPicker::pendingActionTimerFired()
51{
52 LOG(Media, "MediaPlaybackTargetPicker::pendingActionTimerFired - flags = 0x%x", m_pendingActionFlags);
53
54 PendingActionFlags pendingActions = m_pendingActionFlags;
55 m_pendingActionFlags = 0;
56
57 if (pendingActions & CurrentDeviceDidChange)
58 m_client->setPlaybackTarget(playbackTarget());
59
60 if (pendingActions & OutputDeviceAvailabilityChanged)
61 m_client->externalOutputDeviceAvailableDidChange(externalOutputDeviceAvailable());
62
63 if (pendingActions & PlaybackTargetPickerWasDismissed)
64 m_client->playbackTargetPickerWasDismissed();
65}
66
67void MediaPlaybackTargetPicker::addPendingAction(PendingActionFlags action)
68{
69 if (!m_client)
70 return;
71
72 m_pendingActionFlags |= action;
73 m_pendingActionTimer.startOneShot(pendingActionInterval);
74}
