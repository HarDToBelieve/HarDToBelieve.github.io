26#ifndef MediaPlaybackTargetPicker_h
27#define MediaPlaybackTargetPicker_h
28
29#if ENABLE(WIRELESS_PLAYBACK_TARGET)
30
31#include "PlatformView.h"
32#include <wtf/Ref.h>
33#include <wtf/RunLoop.h>
34
35namespace WebCore {
36
37class FloatRect;
38class MediaPlaybackTarget;
39
40class MediaPlaybackTargetPicker {
41public:
42 class Client {
43 protected:
44 virtual ~Client() = default;
45
46 public:
47 virtual void setPlaybackTarget(Ref<MediaPlaybackTarget>&&) = 0;
48 virtual void externalOutputDeviceAvailableDidChange(bool) = 0;
49 virtual void playbackTargetPickerWasDismissed() = 0;
50
51 void invalidate();
52 };
53
54 virtual ~MediaPlaybackTargetPicker();
55
56 virtual void showPlaybackTargetPicker(PlatformView*, const FloatRect&, bool checkActiveRoute, bool useDarkAppearance, bool useiTunesAVOutputContext);
57 virtual void startingMonitoringPlaybackTargets();
58 virtual void stopMonitoringPlaybackTargets();
59 virtual void invalidatePlaybackTargets();
60
61 void availableDevicesDidChange() { addPendingAction(OutputDeviceAvailabilityChanged); }
62 void currentDeviceDidChange() { addPendingAction(CurrentDeviceDidChange); }
63 void playbackTargetPickerWasDismissed() { addPendingAction(PlaybackTargetPickerWasDismissed); }
64
65protected:
66 explicit MediaPlaybackTargetPicker(Client&);
67
68 enum ActionType {
69 OutputDeviceAvailabilityChanged = 1 << 0,
70 CurrentDeviceDidChange = 1 << 1,
71 PlaybackTargetPickerWasDismissed = 1 << 2,
72 };
73 typedef unsigned PendingActionFlags;
74
75 void addPendingAction(PendingActionFlags);
76 void pendingActionTimerFired();
77 Client* client() const { return m_client; }
78 void setClient(Client* client) { m_client = client; }
79
80private:
81 virtual bool externalOutputDeviceAvailable() = 0;
82 virtual Ref<MediaPlaybackTarget> playbackTarget() = 0;
83
84 PendingActionFlags m_pendingActionFlags { 0 };
85 Client* m_client;
86 RunLoop::Timer<MediaPlaybackTargetPicker> m_pendingActionTimer;
87};
88
89} // namespace WebCore
90
91#endif // ENABLE(WIRELESS_PLAYBACK_TARGET)

93#endif // MediaPlaybackTargetPicker_h
