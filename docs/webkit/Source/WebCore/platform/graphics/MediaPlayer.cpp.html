321const MediaPlayerFactory* MediaPlayer::mediaEngine(MediaPlayerEnums::MediaEngineIdentifier identifier)
322{
323 auto& engines = installedMediaEngines();
324 auto currentIndex = engines.findMatching([identifier] (auto& engine) {
325 return engine->identifier() == identifier;
326 });
327
328 if (currentIndex == notFound) {
329#if PLATFORM(IOS_FAMILY_SIMULATOR)
330 ASSERT(identifier == MediaPlayerEnums::MediaEngineIdentifier::AVFoundationMSE);
331#else
332 ASSERT_NOT_REACHED();
333#endif
334 return nullptr;
335 }
336
337 return engines[currentIndex].get();
338}

340static const MediaPlayerFactory* bestMediaEngineForSupportParameters(const MediaEngineSupportParameters& parameters, const MediaPlayerFactory* current = nullptr)
341{
342 if (parameters.type.isEmpty() && !parameters.isMediaSource && !parameters.isMediaStream)
343 return nullptr;
344
345 // 4.8.10.3 MIME types - In the absence of a specification to the contrary, the MIME type "application/octet-stream"
346 // when used with parameters, e.g. "application/octet-stream;codecs=theora", is a type that the user agent knows
347 // it cannot render.
348 if (parameters.type.containerType() == applicationOctetStream()) {
349 if (!parameters.type.codecs().isEmpty())
350 return nullptr;
351 }
352
353 const MediaPlayerFactory* foundEngine = nullptr;
354 MediaPlayer::SupportsType supported = MediaPlayer::SupportsType::IsNotSupported;
355 for (auto& engine : installedMediaEngines()) {
356 if (current) {
357 if (current == engine.get())
358 current = nullptr;
359 continue;
360 }
361 MediaPlayer::SupportsType engineSupport = engine->supportsTypeAndCodecs(parameters);
362 if (engineSupport > supported) {
363 supported = engineSupport;
364 foundEngine = engine.get();
365 }
366 }
367
368 return foundEngine;
369}

371const MediaPlayerFactory* MediaPlayer::nextMediaEngine(const MediaPlayerFactory* current)
372{
373 if (m_activeEngineIdentifier) {
374 auto* engine = mediaEngine(m_activeEngineIdentifier.value());
375 return current != engine ? engine : nullptr;
376 }
377
378 auto& engines = installedMediaEngines();
379 if (engines.isEmpty())
380 return nullptr;
381
382 if (!current)
383 return engines.first().get();
384
385 auto currentIndex = engines.findMatching([current] (auto& engine) {
386 return engine.get() == current;
387 });
388 if (currentIndex == notFound) {
389 ASSERT_NOT_REACHED();
390 return nullptr;
391 }
392
393 if (currentIndex + 1 >= engines.size())
394 return nullptr;
395
396 return engines[currentIndex + 1].get();
397}

436bool MediaPlayer::load(const URL& url, const ContentType& contentType, const String& keySystem)
437{
438 ASSERT(!m_reloadTimer.isActive());
439
440 // Protect against MediaPlayer being destroyed during a MediaPlayerClient callback.
441 Ref<MediaPlayer> protectedThis(*this);
442
443 m_contentType = contentType;
444 m_url = url;
445 m_keySystem = keySystem.convertToASCIILowercase();
446 m_contentMIMETypeWasInferredFromExtension = false;
447
448#if ENABLE(MEDIA_SOURCE)
449 m_mediaSource = nullptr;
450#endif
451#if ENABLE(MEDIA_STREAM)
452 m_mediaStream = nullptr;
453#endif
454
455 // If the MIME type is missing or is not meaningful, try to figure it out from the URL.
456 AtomString containerType = m_contentType.containerType();
457 if (containerType.isEmpty() || containerType == applicationOctetStream() || containerType == textPlain()) {
458 if (m_url.protocolIsData())
459 m_contentType = ContentType(mimeTypeFromDataURL(m_url.string()));
460 else {
461 auto lastPathComponent = url.lastPathComponent();
462 size_t pos = lastPathComponent.reverseFind('.');
463 if (pos != notFound) {
464 String extension = lastPathComponent.substring(pos + 1).toString();
465 String mediaType = MIMETypeRegistry::mediaMIMETypeForExtension(extension);
466 if (!mediaType.isEmpty()) {
467 m_contentType = ContentType { WTFMove(mediaType) };
468 m_contentMIMETypeWasInferredFromExtension = true;
469 }
470 }
471 }
472 }
473
474 loadWithNextMediaEngine(nullptr);
475 return m_currentMediaEngine;
476}

508const MediaPlayerFactory* MediaPlayer::nextBestMediaEngine(const MediaPlayerFactory* current)
509{
510 MediaEngineSupportParameters parameters;
511 parameters.type = m_contentType;
512 parameters.url = m_url;
513#if ENABLE(MEDIA_SOURCE)
514 parameters.isMediaSource = !!m_mediaSource;
515#endif
516#if ENABLE(MEDIA_STREAM)
517 parameters.isMediaStream = !!m_mediaStream;
518#endif
519
520 if (m_activeEngineIdentifier) {
521 if (current)
522 return nullptr;
523
524 auto* engine = mediaEngine(m_activeEngineIdentifier.value());
525 if (engine && engine->supportsTypeAndCodecs(parameters) != SupportsType::IsNotSupported)
526 return engine;
527
528 return nullptr;
529 }
530
531 return bestMediaEngineForSupportParameters(parameters, current);
532}

539void MediaPlayer::loadWithNextMediaEngine(const MediaPlayerFactory* current)
540{
541#if ENABLE(MEDIA_SOURCE)
542#define MEDIASOURCE m_mediaSource
543#else
544#define MEDIASOURCE 0
545#endif
546
547#if ENABLE(MEDIA_STREAM)
548#define MEDIASTREAM m_mediaStream
549#else
550#define MEDIASTREAM 0
551#endif
552
553 ASSERT(!m_initializingMediaEngine);
554 m_initializingMediaEngine = true;
555 client().mediaPlayerWillInitializeMediaEngine();
556
557 const MediaPlayerFactory* engine = nullptr;
558
559 if (!m_contentType.isEmpty() || MEDIASTREAM || MEDIASOURCE)
560 engine = nextBestMediaEngine(current);
561
562 // If no MIME type is specified or the type was inferred from the file extension, just use the next engine.
563 if (!engine && (m_contentType.isEmpty() || m_contentMIMETypeWasInferredFromExtension))
564 engine = nextMediaEngine(current);
565
566 // Don't delete and recreate the player unless it comes from a different engine.
567 if (!engine) {
568 LOG(Media, "MediaPlayer::loadWithNextMediaEngine - no media engine found for type \"%s\"", m_contentType.raw().utf8().data());
569 m_currentMediaEngine = engine;
570 m_private = nullptr;
571 } else if (m_currentMediaEngine != engine) {
572 m_currentMediaEngine = engine;
573 m_private = engine->createMediaEnginePlayer(this);
574 if (m_private) {
575 client().mediaPlayerEngineUpdated();
576 if (m_visible)
577 m_private->setVisible(m_visible);
578 m_private->prepareForPlayback(m_privateBrowsing, m_preload, m_preservesPitch, m_shouldPrepareToRender);
579 }
580 }
581
582 if (m_private) {
583#if ENABLE(MEDIA_SOURCE)
584 if (m_mediaSource)
585 m_private->load(m_url, m_contentMIMETypeWasInferredFromExtension ? ContentType() : m_contentType, m_mediaSource.get());
586 else
587#endif
588#if ENABLE(MEDIA_STREAM)
589 if (m_mediaStream)
590 m_private->load(*m_mediaStream);
591 else
592#endif
593 m_private->load(m_url, m_contentMIMETypeWasInferredFromExtension ? ContentType() : m_contentType, m_keySystem);
594 } else {
595 m_private = makeUnique<NullMediaPlayerPrivate>(this);
596 if (!m_activeEngineIdentifier && installedMediaEngines().size() > 1 && nextBestMediaEngine(m_currentMediaEngine))
597 m_reloadTimer.startOneShot(0_s);
598 else {
599 client().mediaPlayerEngineUpdated();
600 client().mediaPlayerResourceNotSupported();
601 }
602 }
603
604 m_initializingMediaEngine = false;
605 client().mediaPlayerDidInitializeMediaEngine();
606}

1046MediaPlayer::SupportsType MediaPlayer::supportsType(const MediaEngineSupportParameters& parameters)
1047{
1048 // 4.8.10.3 MIME types - The canPlayType(type) method must return the empty string if type is a type that the
1049 // user agent knows it cannot render or is the type "application/octet-stream"
1050 AtomString containerType = parameters.type.containerType();
1051 if (containerType == applicationOctetStream())
1052 return SupportsType::IsNotSupported;
1053
1054 auto lowercaseType = containerType.convertToASCIILowercase();
1055 if (!lowercaseType.startsWith("video/") && !lowercaseType.startsWith("audio/") && !lowercaseType.startsWith("application/"))
1056 return SupportsType::IsNotSupported;
1057
1058 const MediaPlayerFactory* engine = bestMediaEngineForSupportParameters(parameters);
1059 if (!engine)
1060 return SupportsType::IsNotSupported;
1061
1062 return engine->supportsTypeAndCodecs(parameters);
1063}

1270// Client callbacks.
1271void MediaPlayer::networkStateChanged()
1272{
1273 // If more than one media engine is installed and this one failed before finding metadata,
1274 // let the next engine try.
1275 if (m_private->networkState() >= MediaPlayer::NetworkState::FormatError && m_private->readyState() < MediaPlayer::ReadyState::HaveMetadata) {
1276 client().mediaPlayerEngineFailedToLoad();
1277 if (!m_activeEngineIdentifier && installedMediaEngines().size() > 1 && (m_contentType.isEmpty() || nextBestMediaEngine(m_currentMediaEngine))) {
1278 m_reloadTimer.startOneShot(0_s);
1279 return;
1280 }
1281 }
1282 client().mediaPlayerNetworkStateChanged();
1283}

1289 seek(*std::exchange(m_pendingSeekRequest, std::nullopt));
1290}
