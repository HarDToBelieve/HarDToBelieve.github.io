270 virtual size_t extraMemoryCost() const
271 {
272 MediaTime duration = this->durationMediaTime();
273 if (!duration)
274 return 0;
275
276 unsigned long long extra = totalBytes() * buffered()->totalDuration().toDouble() / duration.toDouble();
277 return static_cast<unsigned>(extra);
278 }
