40namespace WebCore {
41
42class ContentType;
43class SourceBufferPrivate;
44
45class MediaSourcePrivate : public RefCounted<MediaSourcePrivate> {
46public:
47 typedef Vector<String> CodecsArray;
48
49 MediaSourcePrivate() = default;
50 virtual ~MediaSourcePrivate() = default;
51
52 enum class AddStatus : uint8_t {
53 Ok,
54 NotSupported,
55 ReachedIdLimit
56 };
57 virtual AddStatus addSourceBuffer(const ContentType&, bool webMParserEnabled, RefPtr<SourceBufferPrivate>&) = 0;
58 virtual void durationChanged(const MediaTime&) = 0;
59 virtual void bufferedChanged(const PlatformTimeRanges&) { }
60 enum EndOfStreamStatus { EosNoError, EosNetworkError, EosDecodeError };
61 virtual void markEndOfStream(EndOfStreamStatus) = 0;
62 virtual void unmarkEndOfStream() = 0;
63 virtual bool isEnded() const = 0;
64
65 virtual MediaPlayer::ReadyState readyState() const = 0;
66 virtual void setReadyState(MediaPlayer::ReadyState) = 0;
67
68 virtual void setIsSeeking(bool isSeeking) { m_isSeeking = isSeeking; }
69 virtual void waitForSeekCompleted() = 0;
70 virtual void seekCompleted() = 0;
71
72 virtual void setTimeFudgeFactor(const MediaTime& fudgeFactor) { m_timeFudgeFactor = fudgeFactor; }
73
74 MediaTime timeFudgeFactor() const { return m_timeFudgeFactor; }
75 bool isSeeking() const { return m_isSeeking; }
76
77private:
78 MediaTime m_timeFudgeFactor;
79 bool m_isSeeking { false };
80};

83String convertEnumerationToString(MediaSourcePrivate::EndOfStreamStatus);

85} // namespace WebCore
