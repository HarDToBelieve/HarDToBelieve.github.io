85void Path::addRoundedRect(const FloatRect& rect, const FloatSize& roundingRadii, RoundedRectStrategy strategy)
86{
87 if (rect.isEmpty())
88 return;
89
90 FloatSize radius(roundingRadii);
91 FloatSize halfSize = rect.size() / 2;
92
93 // Apply the SVG corner radius constraints, per the rect section of the SVG shapes spec: if
94 // one of rx,ry is negative, then the other corner radius value is used. If both values are
95 // negative then rx = ry = 0. If rx is greater than half of the width of the rectangle
96 // then set rx to half of the width; ry is handled similarly.
97
98 if (radius.width() < 0)
99 radius.setWidth((radius.height() < 0) ? 0 : radius.height());
100
101 if (radius.height() < 0)
102 radius.setHeight(radius.width());
103
104 if (radius.width() > halfSize.width())
105 radius.setWidth(halfSize.width());
106
107 if (radius.height() > halfSize.height())
108 radius.setHeight(halfSize.height());
109
110 addRoundedRect(FloatRoundedRect(rect, radius, radius, radius, radius), strategy);
111}

113void Path::addRoundedRect(const FloatRoundedRect& r, RoundedRectStrategy strategy)
114{
115 if (r.isEmpty())
116 return;
117
118 const FloatRoundedRect::Radii& radii = r.radii();
119 const FloatRect& rect = r.rect();
120
121 if (!r.isRenderable()) {
122 // If all the radii cannot be accommodated, return a rect.
123 addRect(rect);
124 return;
125 }
126
127 if (strategy == RoundedRectStrategy::PreferNative) {
128#if USE(CG) || USE(DIRECT2D)
129 platformAddPathForRoundedRect(rect, radii.topLeft(), radii.topRight(), radii.bottomLeft(), radii.bottomRight());
130 return;
131#endif
132 }
133
134 addBeziersForRoundedRect(rect, radii.topLeft(), radii.topRight(), radii.bottomLeft(), radii.bottomRight());
135}

142void Path::addBeziersForRoundedRect(const FloatRect& rect, const FloatSize& topLeftRadius, const FloatSize& topRightRadius, const FloatSize& bottomLeftRadius, const FloatSize& bottomRightRadius)
143{
144 moveTo(FloatPoint(rect.x() + topLeftRadius.width(), rect.y()));
145
146 addLineTo(FloatPoint(rect.maxX() - topRightRadius.width(), rect.y()));
147 if (topRightRadius.width() > 0 || topRightRadius.height() > 0)
148 addBezierCurveTo(FloatPoint(rect.maxX() - topRightRadius.width() * circleControlPoint(), rect.y()),
149 FloatPoint(rect.maxX(), rect.y() + topRightRadius.height() * circleControlPoint()),
150 FloatPoint(rect.maxX(), rect.y() + topRightRadius.height()));
151 addLineTo(FloatPoint(rect.maxX(), rect.maxY() - bottomRightRadius.height()));
152 if (bottomRightRadius.width() > 0 || bottomRightRadius.height() > 0)
153 addBezierCurveTo(FloatPoint(rect.maxX(), rect.maxY() - bottomRightRadius.height() * circleControlPoint()),
154 FloatPoint(rect.maxX() - bottomRightRadius.width() * circleControlPoint(), rect.maxY()),
155 FloatPoint(rect.maxX() - bottomRightRadius.width(), rect.maxY()));
156 addLineTo(FloatPoint(rect.x() + bottomLeftRadius.width(), rect.maxY()));
157 if (bottomLeftRadius.width() > 0 || bottomLeftRadius.height() > 0)
158 addBezierCurveTo(FloatPoint(rect.x() + bottomLeftRadius.width() * circleControlPoint(), rect.maxY()),
159 FloatPoint(rect.x(), rect.maxY() - bottomLeftRadius.height() * circleControlPoint()),
160 FloatPoint(rect.x(), rect.maxY() - bottomLeftRadius.height()));
161 addLineTo(FloatPoint(rect.x(), rect.y() + topLeftRadius.height()));
162 if (topLeftRadius.width() > 0 || topLeftRadius.height() > 0)
163 addBezierCurveTo(FloatPoint(rect.x(), rect.y() + topLeftRadius.height() * circleControlPoint()),
164 FloatPoint(rect.x() + topLeftRadius.width() * circleControlPoint(), rect.y()),
165 FloatPoint(rect.x() + topLeftRadius.width(), rect.y()));
166
167 closeSubpath();
168}

170void Path::apply(const PathApplierFunction& function) const
171{
172 if (isNull())
173 return;
174
175#if ENABLE(INLINE_PATH_DATA)
176 if (hasInlineData<MoveData>()) {
177 PathElement element;
178 element.type = PathElement::Type::MoveToPoint;
179 element.points[0] = WTF::get<MoveData>(m_inlineData).location;
180 function(element);
181 return;
182 }
183
184 if (hasInlineData<LineData>()) {
185 auto& line = WTF::get<LineData>(m_inlineData);
186 PathElement element;
187 element.type = PathElement::Type::MoveToPoint;
188 element.points[0] = line.start;
189 function(element);
190 element.type = PathElement::Type::AddLineToPoint;
191 element.points[0] = line.end;
192 function(element);
193 return;
194 }
195
196 if (hasInlineData<BezierCurveData>()) {
197 auto& curve = WTF::get<BezierCurveData>(m_inlineData);
198 PathElement element;
199 element.type = PathElement::Type::MoveToPoint;
200 element.points[0] = curve.startPoint;
201 function(element);
202 element.type = PathElement::Type::AddCurveToPoint;
203 element.points[0] = curve.controlPoint1;
204 element.points[1] = curve.controlPoint2;
205 element.points[2] = curve.endPoint;
206 function(element);
207 return;
208 }
209
210 if (hasInlineData<QuadCurveData>()) {
211 auto& curve = WTF::get<QuadCurveData>(m_inlineData);
212 PathElement element;
213 element.type = PathElement::Type::MoveToPoint;
214 element.points[0] = curve.startPoint;
215 function(element);
216 element.type = PathElement::Type::AddQuadCurveToPoint;
217 element.points[0] = curve.controlPoint;
218 element.points[1] = curve.endPoint;
219 function(element);
220 return;
221 }
222#endif
223
224 applySlowCase(function);
225}

245FloatPoint Path::currentPoint() const
246{
247 if (isNull())
248 return { };
249
250#if ENABLE(INLINE_PATH_DATA)
251 if (hasInlineData<MoveData>())
252 return inlineData<MoveData>().location;
253
254 if (hasInlineData<LineData>())
255 return inlineData<LineData>().end;
256
257 if (hasInlineData<BezierCurveData>())
258 return inlineData<BezierCurveData>().endPoint;
259
260 if (hasInlineData<QuadCurveData>())
261 return inlineData<QuadCurveData>().endPoint;
262
263 if (hasInlineData<ArcData>()) {
264 auto& arc = inlineData<ArcData>();
265 if (arc.type == ArcData::Type::ClosedLineAndArc)
266 return arc.start;
267
268 return {
269 arc.center.x() + arc.radius * std::acos(arc.endAngle),
270 arc.center.y() + arc.radius * std::asin(arc.endAngle)
271 };
272 }
273#endif
274
275 return currentPointSlowCase();
276}

291void Path::addArc(const FloatPoint& point, float radius, float startAngle, float endAngle, bool anticlockwise)
292{
293 // Workaround for <rdar://problem/5189233> CGPathAddArc hangs or crashes when passed inf as start or end angle,
294 // as well as http://bugs.webkit.org/show_bug.cgi?id=16449, since cairo_arc() functions hang or crash when
295 // passed inf as radius or start/end angle.
296 if (!std::isfinite(radius) || !std::isfinite(startAngle) || !std::isfinite(endAngle))
297 return;
298
299#if ENABLE(INLINE_PATH_DATA)
300 bool hasMoveData = hasInlineData<MoveData>();
301 if (isNull() || hasMoveData) {
302 ArcData arc;
303 if (hasMoveData) {
304 arc.type = ArcData::Type::LineAndArc;
305 arc.start = inlineData<MoveData>().location;
306 }
307 arc.center = point;
308 arc.radius = radius;
309 arc.startAngle = startAngle;
310 arc.endAngle = endAngle;
311 // FIXME: Either ArcData::clockwise needs to be renamed to anticlockwise, or the last argument to
312 // Path::addArc needs to be renamed to clockwise.
313 arc.clockwise = anticlockwise;
314 m_inlineData = { WTFMove(arc) };
315 return;
316 }
317#endif
318
319 addArcSlowCase(point, radius, startAngle, endAngle, anticlockwise);
320}

322void Path::addLineTo(const FloatPoint& point)
323{
324#if ENABLE(INLINE_PATH_DATA)
325 bool hasMoveData = hasInlineData<MoveData>();
326 if (isNull() || hasMoveData) {
327 LineData line;
328 line.start = hasMoveData ? inlineData<MoveData>().location : FloatPoint();
329 line.end = point;
330 m_inlineData = { WTFMove(line) };
331 return;
332 }
333
334 if (hasInlineData<ArcData>()) {
335 auto& arc = inlineData<ArcData>();
336 if (arc.type == ArcData::Type::LineAndArc && arc.start == point) {
337 arc.type = ArcData::Type::ClosedLineAndArc;
338 return;
339 }
340 }
341#endif
342
343 addLineToSlowCase(point);
344}

434static FloatRect computeArcBounds(const FloatPoint& center, float radius, float start, float end, bool clockwise)
435{
436 if (clockwise)
437 std::swap(start, end);
438
439 constexpr float fullCircle = 2 * piFloat;
440 if (end - start >= fullCircle) {
441 auto diameter = radius * 2;
442 return { center.x() - radius, center.y() - radius, diameter, diameter };
443 }
444
445 auto normalize = [&] (float radians) {
446 double circles = radians / fullCircle;
447 return fullCircle * (circles - floor(circles));
448 };
449
450 start = normalize(start);
451 end = normalize(end);
452
453 auto lengthInRadians = end - start;
454 if (start > end)
455 lengthInRadians += fullCircle;
456
457 FloatPoint startPoint { center.x() + radius * cos(start), center.y() + radius * sin(start) };
458 FloatPoint endPoint { center.x() + radius * cos(end), center.y() + radius * sin(end) };
459 FloatRect result;
460 result.fitToPoints(startPoint, endPoint);
461
462 auto contains = [&] (float angleToCheck) {
463 return (start < angleToCheck && start + lengthInRadians > angleToCheck)
464 || (start > angleToCheck && start + lengthInRadians > angleToCheck + fullCircle);
465 };
466
467 if (contains(0))
468 result.shiftMaxXEdgeTo(center.x() + radius);
469
470 if (contains(piOverTwoFloat))
471 result.shiftMaxYEdgeTo(center.y() + radius);
472
473 if (contains(piFloat))
474 result.shiftXEdgeTo(center.x() - radius);
475
476 if (contains(3 * piOverTwoFloat))
477 result.shiftYEdgeTo(center.y() - radius);
478
479 return result;
480}

482std::optional<FloatRect> Path::boundingRectFromInlineData() const
483{
484 if (hasInlineData<ArcData>()) {
485 auto& arc = inlineData<ArcData>();
486 auto bounds = computeArcBounds(arc.center, arc.radius, arc.startAngle, arc.endAngle, arc.clockwise);
487 if (arc.type == ArcData::Type::LineAndArc || arc.type == ArcData::Type::ClosedLineAndArc)
488 bounds.extend(arc.start);
489 return bounds;
490 }
491
492 if (hasInlineData<MoveData>())
493 return {{ inlineData<MoveData>().location, FloatSize { } }};
494
495 if (hasInlineData<LineData>()) {
496 FloatRect result;
497 auto& line = inlineData<LineData>();
498 result.fitToPoints(line.start, line.end);
499 return result;
500 }
501
502 return std::nullopt;
503}

507#if !USE(CG) && !USE(DIRECT2D)
508Path Path::polygonPathFromPoints(const Vector<FloatPoint>& points)
509{
510 Path path;
511 if (points.size() < 2)
512 return path;
513
514 path.moveTo(points[0]);
515 for (size_t i = 1; i < points.size(); ++i)
516 path.addLineTo(points[i]);
517
518 path.closeSubpath();
519 return path;
520}

531
532TextStream& operator<<(TextStream& stream, const Path& path)
533{
534 bool isFirst = true;
535 path.apply([&stream, &isFirst](const PathElement& element) {
536 if (!isFirst)
537 stream << ", ";
538 isFirst = false;
539 switch (element.type) {
540 case PathElement::Type::MoveToPoint: // The points member will contain 1 value.
541 stream << "move to " << element.points[0];
542 break;
543 case PathElement::Type::AddLineToPoint: // The points member will contain 1 value.
544 stream << "add line to " << element.points[0];
545 break;
546 case PathElement::Type::AddQuadCurveToPoint: // The points member will contain 2 values.
547 stream << "add quad curve to " << element.points[0] << " " << element.points[1];
548 break;
549 case PathElement::Type::AddCurveToPoint: // The points member will contain 3 values.
550 stream << "add curve to " << element.points[0] << " " << element.points[1] << " " << element.points[2];
551 break;
552 case PathElement::Type::CloseSubpath: // The points member will contain no values.
553 stream << "close subpath";
554 break;
555 }
556 });
557
558 return stream;
559}
