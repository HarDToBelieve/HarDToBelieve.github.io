30static const float kPathSegmentLengthTolerance = 0.00001f;
31
32static inline float distanceLine(const FloatPoint& start, const FloatPoint& end)
33{
34 return std::hypot(end.x() - start.x(), end.y() - start.y());
35}
36
37struct QuadraticBezier {
38 QuadraticBezier() { }
39 QuadraticBezier(const FloatPoint& s, const FloatPoint& c, const FloatPoint& e)
40 : start(s)
41 , control(c)
42 , end(e)
43 {
44 }
45
46 bool operator==(const QuadraticBezier& rhs) const
47 {
48 return start == rhs.start
49 && control == rhs.control
50 && end == rhs.end;
51 }
52
53 float approximateDistance() const
54 {
55 return distanceLine(start, control) + distanceLine(control, end);
56 }
57
58 bool split(QuadraticBezier& left, QuadraticBezier& right) const
59 {
60 left.control = midPoint(start, control);
61 right.control = midPoint(control, end);
62
63 FloatPoint leftControlToRightControl = midPoint(left.control, right.control);
64 left.end = leftControlToRightControl;
65 right.start = leftControlToRightControl;
66
67 left.start = start;
68 right.end = end;
69
70 return !(left == *this) && !(right == *this);
71 }
72
73 FloatPoint start;
74 FloatPoint control;
75 FloatPoint end;
76};
77
78struct CubicBezier {
79 CubicBezier() { }
80 CubicBezier(const FloatPoint& s, const FloatPoint& c1, const FloatPoint& c2, const FloatPoint& e)
81 : start(s)
82 , control1(c1)
83 , control2(c2)
84 , end(e)
85 {
86 }
87
88 bool operator==(const CubicBezier& rhs) const
89 {
90 return start == rhs.start
91 && control1 == rhs.control1
92 && control2 == rhs.control2
93 && end == rhs.end;
94 }
95
96 float approximateDistance() const
97 {
98 return distanceLine(start, control1) + distanceLine(control1, control2) + distanceLine(control2, end);
99 }
100
101 bool split(CubicBezier& left, CubicBezier& right) const
102 {
103 FloatPoint startToControl1 = midPoint(control1, control2);
104
105 left.start = start;
106 left.control1 = midPoint(start, control1);
107 left.control2 = midPoint(left.control1, startToControl1);
108
109 right.control2 = midPoint(control2, end);
110 right.control1 = midPoint(right.control2, startToControl1);
111 right.end = end;
112
113 FloatPoint leftControl2ToRightControl1 = midPoint(left.control2, right.control1);
114 left.end = leftControl2ToRightControl1;
115 right.start = leftControl2ToRightControl1;
116
117 return !(left == *this) && !(right == *this);
118 }
119
120 FloatPoint start;
121 FloatPoint control1;
122 FloatPoint control2;
123 FloatPoint end;
124};

126// FIXME: This function is possibly very slow due to the ifs required for proper path measuring
127// A simple speed-up would be to use an additional boolean template parameter to control whether
128// to use the "fast" version of this function with no PathTraversalState updating, vs. the slow
129// version which does update the PathTraversalState. We'll have to shark it to see if that's necessary.
130// Another check which is possible up-front (to send us down the fast path) would be to check if
131// approximateDistance() + current total distance > desired distance
132template<class CurveType>
133static float curveLength(const PathTraversalState& traversalState, const CurveType& originalCurve, FloatPoint& previous, FloatPoint& current)
134{
135 static const unsigned curveStackDepthLimit = 20;
136 CurveType curve = originalCurve;
137 Vector<CurveType, curveStackDepthLimit> curveStack;
138 float totalLength = 0;
139
140 while (true) {
141 float length = curve.approximateDistance();
142
143 CurveType leftCurve;
144 CurveType rightCurve;
145
146 if ((length - distanceLine(curve.start, curve.end)) > kPathSegmentLengthTolerance && curveStack.size() < curveStackDepthLimit && curve.split(leftCurve, rightCurve)) {
147 curve = leftCurve;
148 curveStack.append(rightCurve);
149 continue;
150 }
151
152 totalLength += length;
153 if (traversalState.action() == PathTraversalState::Action::VectorAtLength) {
154 previous = curve.start;
155 current = curve.end;
156 if (traversalState.totalLength() + totalLength > traversalState.desiredLength())
157 break;
158 }
159
160 if (curveStack.isEmpty())
161 break;
162
163 curve = curveStack.last();
164 curveStack.removeLast();
165 }
166
167 if (traversalState.action() != PathTraversalState::Action::VectorAtLength) {
168 ASSERT(curve.end == originalCurve.end);
169 previous = curve.start;
170 current = curve.end;
171 }
172
173 return totalLength;
174}

176PathTraversalState::PathTraversalState(Action action, float desiredLength)
177 : m_action(action)
178 , m_desiredLength(desiredLength)
179{
180 ASSERT(action != Action::TotalLength || !desiredLength);
181}

183void PathTraversalState::closeSubpath()
184{
185 m_totalLength += distanceLine(m_current, m_start);
186 m_current = m_start;
187}
188
189void PathTraversalState::moveTo(const FloatPoint& point)
190{
191 m_previous = m_current = m_start = point;
192}
193
194void PathTraversalState::lineTo(const FloatPoint& point)
195{
196 m_totalLength += distanceLine(m_current, point);
197 m_current = point;
198}

200void PathTraversalState::quadraticBezierTo(const FloatPoint& newControl, const FloatPoint& newEnd)
201{
202 m_totalLength += curveLength<QuadraticBezier>(*this, QuadraticBezier(m_current, newControl, newEnd), m_previous, m_current);
203}
204
205void PathTraversalState::cubicBezierTo(const FloatPoint& newControl1, const FloatPoint& newControl2, const FloatPoint& newEnd)
206{
207 m_totalLength += curveLength<CubicBezier>(*this, CubicBezier(m_current, newControl1, newControl2, newEnd), m_previous, m_current);
208}

210bool PathTraversalState::finalizeAppendPathElement()
211{
212 if (m_action == Action::TotalLength)
213 return false;
214
215 if (m_action == Action::SegmentAtLength) {
216 if (m_totalLength >= m_desiredLength)
217 m_success = true;
218 return m_success;
219 }
220
221 ASSERT(m_action == Action::VectorAtLength);
222
223 if (m_totalLength >= m_desiredLength) {
224 float slope = FloatPoint(m_current - m_previous).slopeAngleRadians();
225 float offset = m_desiredLength - m_totalLength;
226 m_current.move(offset * cosf(slope), offset * sinf(slope));
227
228 if (!m_isZeroVector && !m_desiredLength)
229 m_isZeroVector = true;
230 else {
231 m_success = true;
232 m_normalAngle = rad2deg(slope);
233 }
234 }
235
236 m_previous = m_current;
237 return m_success;
238}

240bool PathTraversalState::appendPathElement(PathElement::Type type, const FloatPoint* points)
241{
242 switch (type) {
243 case PathElement::Type::MoveToPoint:
244 moveTo(points[0]);
245 break;
246 case PathElement::Type::AddLineToPoint:
247 lineTo(points[0]);
248 break;
249 case PathElement::Type::AddQuadCurveToPoint:
250 quadraticBezierTo(points[0], points[1]);
251 break;
252 case PathElement::Type::AddCurveToPoint:
253 cubicBezierTo(points[0], points[1], points[2]);
254 break;
255 case PathElement::Type::CloseSubpath:
256 closeSubpath();
257 break;
258 }
259
260 return finalizeAppendPathElement();
261}
262
263bool PathTraversalState::processPathElement(PathElement::Type type, const FloatPoint* points)
264{
265 if (m_success)
266 return true;
267
268 if (m_isZeroVector) {
269 PathTraversalState traversalState(*this);
270 m_success = traversalState.appendPathElement(type, points);
271 m_normalAngle = traversalState.m_normalAngle;
272 return m_success;
273 }
274
275 return appendPathElement(type, points);
276}
