51namespace WebCore {
52
53class AffineTransform;
54class GraphicsContext;
55class Image;
56
57class Pattern final : public RefCounted<Pattern> {
58public:
59 struct Parameters {
60 Parameters(bool repeatX = true, bool repeatY = true, AffineTransform patternSpaceTransform = { })
61 : patternSpaceTransform(patternSpaceTransform)
62 , repeatX(repeatX)
63 , repeatY(repeatY)
64 {
65 }
66 template<class Encoder> void encode(Encoder&) const;
67 template<class Decoder> static std::optional<Parameters> decode(Decoder&);
68 AffineTransform patternSpaceTransform;
69 bool repeatX;
70 bool repeatY;
71 };
72
73 WEBCORE_EXPORT static Ref<Pattern> create(Ref<NativeImage>&& tileImage, const Parameters& = { });
74 WEBCORE_EXPORT ~Pattern();
75
76 NativeImage& tileImage() const { return m_tileImage.get(); }
77 const Parameters& parameters() const { return m_parameters; }
78
79 // Pattern space is an abstract space that maps to the default user space by the transformation 'userSpaceTransform'
80#if !USE(DIRECT2D)
81 PlatformPatternPtr createPlatformPattern(const AffineTransform& userSpaceTransform) const;
82#else
83 PlatformPatternPtr createPlatformPattern(const GraphicsContext&, float alpha, const AffineTransform& userSpaceTransform) const;
84#endif
85 void setPatternSpaceTransform(const AffineTransform&);
86 const AffineTransform& patternSpaceTransform() const { return m_parameters.patternSpaceTransform; };
87 bool repeatX() const { return m_parameters.repeatX; }
88 bool repeatY() const { return m_parameters.repeatY; }
89
90private:
91 Pattern(Ref<NativeImage>&&, const Parameters&);
92
93 Ref<NativeImage> m_tileImage;
94 Parameters m_parameters;
95};
96
97template<class Encoder>
98void Pattern::Parameters::encode(Encoder& encoder) const
99{
100 encoder << patternSpaceTransform;
101 encoder << repeatX;
102 encoder << repeatY;
103}
104
105template<class Decoder>
106std::optional<Pattern::Parameters> Pattern::Parameters::decode(Decoder& decoder)
107{
108 std::optional<AffineTransform> patternSpaceTransform;
109 decoder >> patternSpaceTransform;
110 if (!patternSpaceTransform)
111 return std::nullopt;
112
113 std::optional<bool> repeatX;
114 decoder >> repeatX;
115 if (!repeatX)
116 return std::nullopt;
117
118 std::optional<bool> repeatY;
119 decoder >> repeatY;
120 if (!repeatY)
121 return std::nullopt;
122
123 return {{ *repeatX, *repeatY, *patternSpaceTransform }};
124}
