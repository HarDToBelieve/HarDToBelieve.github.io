34bool PixelBuffer::supportedPixelFormat(PixelFormat pixelFormat)
35{
36 switch (pixelFormat) {
37 case PixelFormat::RGBA8:
38 case PixelFormat::BGRA8:
39 return true;
40
41 case PixelFormat::RGB10:
42 case PixelFormat::RGB10A8:
43 return false;
44 }
45
46 ASSERT_NOT_REACHED();
47 return false;
48}
49
50CheckedUint32 PixelBuffer::computeBufferSize(const PixelBufferFormat& format, const IntSize& size)
51{
52 ASSERT_UNUSED(format, supportedPixelFormat(format.pixelFormat));
53
54 constexpr unsigned bytesPerPixel = 4;
55
56 return size.area<RecordOverflow>() * bytesPerPixel;
57}
58
59std::optional<PixelBuffer> PixelBuffer::tryCreateForDecoding(const PixelBufferFormat& format, const IntSize& size, unsigned dataByteLength)
60{
61 ASSERT(supportedPixelFormat(format.pixelFormat));
62 ASSERT(computeBufferSize(format, size) == dataByteLength);
63
64 auto pixelArray = Uint8ClampedArray::tryCreateUninitialized(dataByteLength);
65 if (!pixelArray)
66 return std::nullopt;
67 return { { format, size, pixelArray.releaseNonNull() } };
68}
69
70std::optional<PixelBuffer> PixelBuffer::tryCreate(const PixelBufferFormat& format, const IntSize& size)
71{
72 ASSERT(supportedPixelFormat(format.pixelFormat));
73
74 auto bufferSize = computeBufferSize(format, size);
75 if (bufferSize.hasOverflowed())
76 return std::nullopt;
77 auto pixelArray = Uint8ClampedArray::tryCreateUninitialized(bufferSize);
78 if (!pixelArray)
79 return std::nullopt;
80 return { { format, size, pixelArray.releaseNonNull() } };
81}
82
83std::optional<PixelBuffer> PixelBuffer::tryCreate(const PixelBufferFormat& format, const IntSize& size, Ref<JSC::ArrayBuffer>&& arrayBuffer)
84{
85 ASSERT(supportedPixelFormat(format.pixelFormat));
86
87 auto bufferSize = computeBufferSize(format, size);
88 if (bufferSize.hasOverflowed())
89 return std::nullopt;
90 if (bufferSize != arrayBuffer->byteLength())
91 return std::nullopt;
92 auto pixelArray = Uint8ClampedArray::tryCreate(WTFMove(arrayBuffer), 0, bufferSize);
93 if (!pixelArray)
94 return std::nullopt;
95 return { { format, size, pixelArray.releaseNonNull() } };
96}

98PixelBuffer::PixelBuffer(const PixelBufferFormat& format, const IntSize& size, Ref<JSC::Uint8ClampedArray>&& data)
99 : m_format { format }
100 , m_size { size }
101 , m_data { WTFMove(data) }
102{
103 RELEASE_ASSERT_WITH_SECURITY_IMPLICATION((m_size.area() * 4) <= m_data->length());
104}

116PixelBuffer PixelBuffer::deepClone() const
117{
118 return { m_format, m_size, Uint8ClampedArray::create(m_data->data(), m_data->length()) };
119}
