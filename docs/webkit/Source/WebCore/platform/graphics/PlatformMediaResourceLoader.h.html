37namespace WebCore {
38
39class PlatformMediaResource;
40class ResourceError;
41class ResourceRequest;
42class ResourceResponse;
43
44class PlatformMediaResourceClient : public RefCounted<PlatformMediaResourceClient> {
45public:
46 virtual ~PlatformMediaResourceClient() = default;
47
48 virtual void responseReceived(PlatformMediaResource&, const ResourceResponse&, CompletionHandler<void(ShouldContinuePolicyCheck)>&& completionHandler) { completionHandler(ShouldContinuePolicyCheck::Yes); }
49 virtual void redirectReceived(PlatformMediaResource&, ResourceRequest&& request, const ResourceResponse&, CompletionHandler<void(ResourceRequest&&)>&& completionHandler) { completionHandler(WTFMove(request)); }
50 virtual bool shouldCacheResponse(PlatformMediaResource&, const ResourceResponse&) { return true; }
51 virtual void dataSent(PlatformMediaResource&, unsigned long long, unsigned long long) { }
52 virtual void dataReceived(PlatformMediaResource&, const uint8_t*, int) { }
53 virtual void accessControlCheckFailed(PlatformMediaResource&, const ResourceError&) { }
54 virtual void loadFailed(PlatformMediaResource&, const ResourceError&) { }
55 virtual void loadFinished(PlatformMediaResource&, const NetworkLoadMetrics&) { }
56};
57
58class PlatformMediaResourceLoader : public ThreadSafeRefCounted<PlatformMediaResourceLoader, WTF::DestructionThread::Main> {
59 WTF_MAKE_NONCOPYABLE(PlatformMediaResourceLoader); WTF_MAKE_FAST_ALLOCATED;
60public:
61 enum LoadOption {
62 BufferData = 1 << 0,
63 DisallowCaching = 1 << 1,
64 };
65 typedef unsigned LoadOptions;
66
67 virtual ~PlatformMediaResourceLoader() = default;
68
69 virtual RefPtr<PlatformMediaResource> requestResource(ResourceRequest&&, LoadOptions) = 0;
70 virtual void sendH2Ping(const URL&, CompletionHandler<void(Expected<Seconds, ResourceError>&&)>&&) = 0;
71
72protected:
73 PlatformMediaResourceLoader() = default;
74};
75
76class PlatformMediaResource : public ThreadSafeRefCounted<PlatformMediaResource, WTF::DestructionThread::Main> {
77 WTF_MAKE_NONCOPYABLE(PlatformMediaResource); WTF_MAKE_FAST_ALLOCATED;
78public:
79 PlatformMediaResource() = default;
80 virtual ~PlatformMediaResource() = default;
81 virtual void stop() { }
82 virtual bool didPassAccessControlCheck() const { return false; }
83
84 void setClient(RefPtr<PlatformMediaResourceClient>&& client) { m_client = WTFMove(client); }
85 PlatformMediaResourceClient* client() { return m_client.get(); }
86
87protected:
88 RefPtr<PlatformMediaResourceClient> m_client;
89};
