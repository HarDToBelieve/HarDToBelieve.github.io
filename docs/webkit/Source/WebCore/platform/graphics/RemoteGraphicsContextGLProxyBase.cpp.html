69void RemoteGraphicsContextGLProxyBase::markLayerComposited()
70{
71 m_layerComposited = true;
72 auto attrs = contextAttributes();
73 if (!attrs.preserveDrawingBuffer) {
74 m_buffersToAutoClear = GraphicsContextGL::COLOR_BUFFER_BIT;
75 if (attrs.depth)
76 m_buffersToAutoClear |= GraphicsContextGL::DEPTH_BUFFER_BIT;
77 if (attrs.stencil)
78 m_buffersToAutoClear |= GraphicsContextGL::STENCIL_BUFFER_BIT;
79 }
80 for (auto* client : copyToVector(m_clients))
81 client->didComposite();
82}

113void RemoteGraphicsContextGLProxyBase::ensureEnabled(const String& name)
114{
115 waitUntilInitialized();
116 if (m_requestableExtensions.contains(name) && !m_enabledExtensions.contains(name)) {
117 ensureExtensionEnabled(name);
118 m_enabledExtensions.add(name);
119 }
120}

122bool RemoteGraphicsContextGLProxyBase::isEnabled(const String& name)
123{
124 waitUntilInitialized();

126}

128void RemoteGraphicsContextGLProxyBase::initialize(const String& availableExtensions, const String& requestableExtensions)
129{
130 for (auto& extension : availableExtensions.split(' '))
