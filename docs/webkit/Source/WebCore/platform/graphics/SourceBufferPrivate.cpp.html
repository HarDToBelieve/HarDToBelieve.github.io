44// Do not enqueue samples spanning a significant unbuffered gap.
45// NOTE: one second is somewhat arbitrary. MediaSource::monitorSourceBuffers() is run
46// on the playbackTimer, which is effectively every 350ms. Allowing > 350ms gap between
47// enqueued samples allows for situations where we overrun the end of a buffered range
48// but don't notice for 350ms of playback time, and the client can enqueue data for the
49// new current time without triggering this early return.
50// FIXME(135867): Make this gap detection logic less arbitrary.
51static const MediaTime discontinuityTolerance = MediaTime(1, 1);

96void SourceBufferPrivate::updateHighestPresentationTimestamp()
97{
98 MediaTime highestTime;
99 for (auto& trackBuffer : m_trackBufferMap.values()) {
100 auto lastSampleIter = trackBuffer.get().samples.presentationOrder().rbegin();
101 if (lastSampleIter == trackBuffer.get().samples.presentationOrder().rend())
102 continue;
103 highestTime = std::max(highestTime, lastSampleIter->first);
104 }
105
106 if (m_highestPresentationTimestamp == highestTime)
107 return;
108
109 m_highestPresentationTimestamp = highestTime;
110 if (m_client)
111 m_client->sourceBufferPrivateHighestPresentationTimestampChanged(m_highestPresentationTimestamp);
112}

120void SourceBufferPrivate::updateBufferedFromTrackBuffers(bool sourceIsEnded)
121{
122 // 3.1 Attributes, buffered
123 // https://rawgit.com/w3c/media-source/45627646344eea0170dd1cbc5a3d508ca751abb8/media-source-respec.html#dom-sourcebuffer-buffered
124
125 // 2. Let highest end time be the largest track buffer ranges end time across all the track buffers managed by this SourceBuffer object.
126 MediaTime highestEndTime = MediaTime::negativeInfiniteTime();
127 for (auto& trackBuffer : m_trackBufferMap.values()) {
128 if (!trackBuffer.get().buffered.length())
129 continue;
130 highestEndTime = std::max(highestEndTime, trackBuffer.get().buffered.maximumBufferedTime());
131 }
132
133 // NOTE: Short circuit the following if none of the TrackBuffers have buffered ranges to avoid generating
134 // a single range of {0, 0}.
135 if (highestEndTime.isNegativeInfinite()) {
136 setBufferedRanges(PlatformTimeRanges());
137 return;
138 }
139
140 // 3. Let intersection ranges equal a TimeRange object containing a single range from 0 to highest end time.
141 PlatformTimeRanges intersectionRanges { MediaTime::zeroTime(), highestEndTime };
142
143 // 4. For each audio and video track buffer managed by this SourceBuffer, run the following steps:
144 for (auto& trackBuffer : m_trackBufferMap.values()) {
145 // 4.1 Let track ranges equal the track buffer ranges for the current track buffer.
146 PlatformTimeRanges trackRanges = trackBuffer.get().buffered;
147 if (!trackRanges.length())
148 continue;
149
150 // 4.2 If readyState is "ended", then set the end time on the last range in track ranges to highest end time.
151 if (sourceIsEnded)
152 trackRanges.add(trackRanges.maximumBufferedTime(), highestEndTime);
153
154 // 4.3 Let new intersection ranges equal the intersection between the intersection ranges and the track ranges.
155 // 4.4 Replace the ranges in intersection ranges with the new intersection ranges.
156 intersectionRanges.intersectWith(trackRanges);
157 }
158
159 // 5. If intersection ranges does not contain the exact same range information as the current value of this attribute,
160 // then update the current value of this attribute to intersection ranges.
161 setBufferedRanges(intersectionRanges);
162}

227MediaTime SourceBufferPrivate::fastSeekTimeForMediaTime(const MediaTime& targetTime, const MediaTime& negativeThreshold, const MediaTime& positiveThreshold)
228{
229 if (!m_client)
230 return targetTime;
231
232 MediaTime seekTime = targetTime;
233
234 for (auto& trackBuffer : m_trackBufferMap.values()) {
235 // Find the sample which contains the target time time.
236 auto futureSyncSampleIterator = trackBuffer.get().samples.decodeOrder().findSyncSampleAfterPresentationTime(targetTime, positiveThreshold);
237 auto pastSyncSampleIterator = trackBuffer.get().samples.decodeOrder().findSyncSamplePriorToPresentationTime(targetTime, negativeThreshold);
238 auto upperBound = trackBuffer.get().samples.decodeOrder().end();
239 auto lowerBound = trackBuffer.get().samples.decodeOrder().rend();
240
241 if (futureSyncSampleIterator == upperBound && pastSyncSampleIterator == lowerBound)
242 continue;
243
244 MediaTime futureSeekTime = MediaTime::positiveInfiniteTime();
245 if (futureSyncSampleIterator != upperBound) {
246 RefPtr<MediaSample>& sample = futureSyncSampleIterator->second;
247 futureSeekTime = sample->presentationTime();
248 }
249
250 MediaTime pastSeekTime = MediaTime::negativeInfiniteTime();
251 if (pastSyncSampleIterator != lowerBound) {
252 RefPtr<MediaSample>& sample = pastSyncSampleIterator->second;
253 pastSeekTime = sample->presentationTime();
254 }
255
256 MediaTime trackSeekTime = abs(targetTime - futureSeekTime) < abs(targetTime - pastSeekTime) ? futureSeekTime : pastSeekTime;
257 if (abs(targetTime - trackSeekTime) > abs(targetTime - seekTime))
258 seekTime = trackSeekTime;
259 }
260
261 return seekTime;
262}

264void SourceBufferPrivate::updateMinimumUpcomingPresentationTime(TrackBuffer& trackBuffer, const AtomString& trackID)
265{
266 if (!canSetMinimumUpcomingPresentationTime(trackID))
267 return;
268
269 if (trackBuffer.decodeQueue.empty()) {
270 trackBuffer.minimumEnqueuedPresentationTime = MediaTime::invalidTime();
271 clearMinimumUpcomingPresentationTime(trackID);
272 return;
273 }
274
275 auto minPts = std::min_element(trackBuffer.decodeQueue.begin(), trackBuffer.decodeQueue.end(), [](auto& left, auto& right) -> bool {
276 return left.second->presentationTime() < right.second->presentationTime();
277 });
278
279 if (minPts == trackBuffer.decodeQueue.end()) {
280 trackBuffer.minimumEnqueuedPresentationTime = MediaTime::invalidTime();
281 clearMinimumUpcomingPresentationTime(trackID);
282 return;
283 }
284
285 trackBuffer.minimumEnqueuedPresentationTime = minPts->second->presentationTime();
286 setMinimumUpcomingPresentationTime(trackID, trackBuffer.minimumEnqueuedPresentationTime);
287}

323void SourceBufferPrivate::provideMediaData(TrackBuffer& trackBuffer, const AtomString& trackID)
324{
325 if (!m_isAttached || isSeeking())
326 return;
327
328#if !RELEASE_LOG_DISABLED
329 unsigned enqueuedSamples = 0;
330#endif
331
332 if (trackBuffer.needsMinimumUpcomingPresentationTimeUpdating && canSetMinimumUpcomingPresentationTime(trackID)) {
333 trackBuffer.minimumEnqueuedPresentationTime = MediaTime::invalidTime();
334 clearMinimumUpcomingPresentationTime(trackID);
335 }
336
337 while (!trackBuffer.decodeQueue.empty()) {
338 if (!isReadyForMoreSamples(trackID)) {
339 DEBUG_LOG(LOGIDENTIFIER, "bailing early, track id ", trackID, " is not ready for more data");
340 notifyClientWhenReadyForMoreSamples(trackID);
341 break;
342 }
343
344 // FIXME(rdar://problem/20635969): Remove this re-entrancy protection when the aforementioned radar is resolved; protecting
345 // against re-entrancy introduces a small inefficency when removing appended samples from the decode queue one at a time
346 // rather than when all samples have been enqueued.
347 auto sample = trackBuffer.decodeQueue.begin()->second;
348
349 if (sample->decodeTime() > trackBuffer.enqueueDiscontinuityBoundary) {
350 DEBUG_LOG(LOGIDENTIFIER, "bailing early because of unbuffered gap, new sample: ", sample->decodeTime(), " >= the current discontinuity boundary: ", trackBuffer.enqueueDiscontinuityBoundary);
351 break;
352 }
353
354 // Remove the sample from the decode queue now.
355 trackBuffer.decodeQueue.erase(trackBuffer.decodeQueue.begin());
356
357 MediaTime samplePresentationEnd = sample->presentationTime() + sample->duration();
358 if (trackBuffer.highestEnqueuedPresentationTime.isInvalid() || samplePresentationEnd > trackBuffer.highestEnqueuedPresentationTime)
359 trackBuffer.highestEnqueuedPresentationTime = samplePresentationEnd;
360
361 trackBuffer.lastEnqueuedDecodeKey = {sample->decodeTime(), sample->presentationTime()};
362 trackBuffer.enqueueDiscontinuityBoundary = sample->decodeTime() + sample->duration() + discontinuityTolerance;
363
364 enqueueSample(sample.releaseNonNull(), trackID);
365#if !RELEASE_LOG_DISABLED
366 ++enqueuedSamples;
367#endif
368 }
369
370 updateMinimumUpcomingPresentationTime(trackBuffer, trackID);
371
372#if !RELEASE_LOG_DISABLED
373 DEBUG_LOG(LOGIDENTIFIER, "enqueued ", enqueuedSamples, " samples, ", static_cast<uint64_t>(trackBuffer.decodeQueue.size()), " remaining");
374#endif
375
376 trySignalAllSamplesInTrackEnqueued(trackBuffer, trackID);
377}

379void SourceBufferPrivate::reenqueueMediaForTime(TrackBuffer& trackBuffer, const AtomString& trackID, const MediaTime& time)
380{
381 flush(trackID);
382 trackBuffer.decodeQueue.clear();
383
384 trackBuffer.highestEnqueuedPresentationTime = MediaTime::invalidTime();
385 trackBuffer.lastEnqueuedDecodeKey = {MediaTime::invalidTime(), MediaTime::invalidTime()};
386 trackBuffer.enqueueDiscontinuityBoundary = time + discontinuityTolerance;
387
388 // Find the sample which contains the current presentation time.
389 auto currentSamplePTSIterator = trackBuffer.samples.presentationOrder().findSampleContainingPresentationTime(time);
390
391 if (currentSamplePTSIterator == trackBuffer.samples.presentationOrder().end())
392 currentSamplePTSIterator = trackBuffer.samples.presentationOrder().findSampleStartingOnOrAfterPresentationTime(time);
393
394 if (currentSamplePTSIterator == trackBuffer.samples.presentationOrder().end()
395 || (currentSamplePTSIterator->first - time) > timeFudgeFactor())
396 return;
397
398 // Seach backward for the previous sync sample.
399 DecodeOrderSampleMap::KeyType decodeKey(currentSamplePTSIterator->second->decodeTime(), currentSamplePTSIterator->second->presentationTime());
400 auto currentSampleDTSIterator = trackBuffer.samples.decodeOrder().findSampleWithDecodeKey(decodeKey);
401 ASSERT(currentSampleDTSIterator != trackBuffer.samples.decodeOrder().end());
402
403 auto reverseCurrentSampleIter = --DecodeOrderSampleMap::reverse_iterator(currentSampleDTSIterator);
404 auto reverseLastSyncSampleIter = trackBuffer.samples.decodeOrder().findSyncSamplePriorToDecodeIterator(reverseCurrentSampleIter);
405 if (reverseLastSyncSampleIter == trackBuffer.samples.decodeOrder().rend())
406 return;
407
408 // Fill the decode queue with the non-displaying samples.
409 for (auto iter = reverseLastSyncSampleIter; iter != reverseCurrentSampleIter; --iter) {
410 auto copy = iter->second->createNonDisplayingCopy();
411 DecodeOrderSampleMap::KeyType decodeKey(copy->decodeTime(), copy->presentationTime());
412 trackBuffer.decodeQueue.insert(DecodeOrderSampleMap::MapType::value_type(decodeKey, WTFMove(copy)));
413 }
414
415 // Fill the decode queue with the remaining samples.
416 for (auto iter = currentSampleDTSIterator; iter != trackBuffer.samples.decodeOrder().end(); ++iter)
417 trackBuffer.decodeQueue.insert(*iter);
418 provideMediaData(trackBuffer, trackID);
419
420 trackBuffer.needsReenqueueing = false;
421}

442#if !RELEASE_LOG_DISABLED
443static PlatformTimeRanges removeSamplesFromTrackBuffer(const DecodeOrderSampleMap::MapType& samples, SourceBufferPrivate::TrackBuffer& trackBuffer, const SourceBufferPrivate* sourceBufferPrivate, const char* logPrefix)
444#else
445static PlatformTimeRanges removeSamplesFromTrackBuffer(const DecodeOrderSampleMap::MapType& samples, SourceBufferPrivate::TrackBuffer& trackBuffer)
446#endif
447{
448#if !RELEASE_LOG_DISABLED
449 MediaTime earliestSample = MediaTime::positiveInfiniteTime();
450 MediaTime latestSample = MediaTime::zeroTime();
451 uint64_t bytesRemoved = 0;
452 auto logIdentifier = WTF::Logger::LogSiteIdentifier(sourceBufferPrivate->logClassName(), logPrefix, sourceBufferPrivate->logIdentifier());
453 auto& logger = sourceBufferPrivate->logger();
454 auto willLog = logger.willLog(sourceBufferPrivate->logChannel(), WTFLogLevel::Debug);
455#endif
456
457 PlatformTimeRanges erasedRanges;
458 for (const auto& sampleIt : samples) {
459 const DecodeOrderSampleMap::KeyType& decodeKey = sampleIt.first;
460#if !RELEASE_LOG_DISABLED
461 uint64_t startBufferSize = trackBuffer.samples.sizeInBytes();
462#endif
463
464 const RefPtr<MediaSample>& sample = sampleIt.second;
465
466#if !RELEASE_LOG_DISABLED
467 if (willLog)
468 logger.debug(sourceBufferPrivate->logChannel(), logIdentifier, "removing sample ", *sampleIt.second);
469#endif
470
471 // Remove the erased samples from the TrackBuffer sample map.
472 trackBuffer.samples.removeSample(sample.get());
473
474 // Also remove the erased samples from the TrackBuffer decodeQueue.
475 trackBuffer.decodeQueue.erase(decodeKey);
476
477 auto startTime = sample->presentationTime();
478 auto endTime = startTime + sample->duration();
479 erasedRanges.add(startTime, endTime);
480
481#if !RELEASE_LOG_DISABLED
482 bytesRemoved += startBufferSize - trackBuffer.samples.sizeInBytes();
483 if (startTime < earliestSample)
484 earliestSample = startTime;
485 if (endTime > latestSample)
486 latestSample = endTime;
487#endif
488 }
489
490 // Because we may have added artificial padding in the buffered ranges when adding samples, we may
491 // need to remove that padding when removing those same samples. Walk over the erased ranges looking
492 // for unbuffered areas and expand erasedRanges to encompass those areas.
493 PlatformTimeRanges additionalErasedRanges;
494 for (unsigned i = 0; i < erasedRanges.length(); ++i) {
495 auto erasedStart = erasedRanges.start(i);
496 auto erasedEnd = erasedRanges.end(i);
497 auto startIterator = trackBuffer.samples.presentationOrder().reverseFindSampleBeforePresentationTime(erasedStart);
498 if (startIterator == trackBuffer.samples.presentationOrder().rend())
499 additionalErasedRanges.add(MediaTime::zeroTime(), erasedStart);
500 else {
501 auto& previousSample = *startIterator->second;
502 if (previousSample.presentationTime() + previousSample.duration() < erasedStart)
503 additionalErasedRanges.add(previousSample.presentationTime() + previousSample.duration(), erasedStart);
504 }
505
506 auto endIterator = trackBuffer.samples.presentationOrder().findSampleStartingOnOrAfterPresentationTime(erasedEnd);
507 if (endIterator == trackBuffer.samples.presentationOrder().end())
508 additionalErasedRanges.add(erasedEnd, MediaTime::positiveInfiniteTime());
509 else {
510 auto& nextSample = *endIterator->second;
511 if (nextSample.presentationTime() > erasedEnd)
512 additionalErasedRanges.add(erasedEnd, nextSample.presentationTime());
513 }
514 }
515 if (additionalErasedRanges.length())
516 erasedRanges.unionWith(additionalErasedRanges);
517
518#if !RELEASE_LOG_DISABLED
519 if (bytesRemoved && willLog)
520 logger.debug(sourceBufferPrivate->logChannel(), logIdentifier, "removed ", bytesRemoved, ", start = ", earliestSample, ", end = ", latestSample);
521#endif
522
523 return erasedRanges;
524}

526void SourceBufferPrivate::removeCodedFrames(const MediaTime& start, const MediaTime& end, const MediaTime& currentTime, bool isEnded, CompletionHandler<void()>&& completionHandler)
527{
528 ASSERT(start < end);
529 if (start >= end) {
530 completionHandler();
531 return;
532 }
533
534 // 3.5.9 Coded Frame Removal Algorithm
535 // https://dvcs.w3.org/hg/html-media/raw-file/tip/media-source/media-source.html#sourcebuffer-coded-frame-removal
536
537 // 1. Let start be the starting presentation timestamp for the removal range.
538 // 2. Let end be the end presentation timestamp for the removal range.
539 // 3. For each track buffer in this source buffer, run the following steps:
540 for (auto& trackBufferKeyValue : m_trackBufferMap) {
541 TrackBuffer& trackBuffer = trackBufferKeyValue.value;
542 AtomString trackID = trackBufferKeyValue.key;
543
544 // 3.1. Let remove end timestamp be the current value of duration
545 // 3.2 If this track buffer has a random access point timestamp that is greater than or equal to end, then update
546 // remove end timestamp to that random access point timestamp.
547 // NOTE: Step 3.2 will be incorrect for any random access point timestamp whose decode time is later than the sample at end,
548 // but whose presentation time is less than the sample at end. Skip this step until step 3.3 below.
549
550 // NOTE: To handle MediaSamples which may be an amalgamation of multiple shorter samples, find samples whose presentation
551 // interval straddles the start and end times, and divide them if possible:
552 auto divideSampleIfPossibleAtPresentationTime = [&] (const MediaTime& time) {
553 auto sampleIterator = trackBuffer.samples.presentationOrder().findSampleContainingPresentationTime(time);
554 if (sampleIterator == trackBuffer.samples.presentationOrder().end())
555 return;
556 RefPtr<MediaSample> sample = sampleIterator->second;
557 if (!sample->isDivisable())
558 return;
559 std::pair<RefPtr<MediaSample>, RefPtr<MediaSample>> replacementSamples = sample->divide(time);
560 if (!replacementSamples.first || !replacementSamples.second)
561 return;
562 DEBUG_LOG(LOGIDENTIFIER, "splitting sample ", *sample, " into ", *replacementSamples.first, " and ", *replacementSamples.second);
563 trackBuffer.samples.removeSample(sample.get());
564 trackBuffer.samples.addSample(*replacementSamples.first);
565 trackBuffer.samples.addSample(*replacementSamples.second);
566 };
567 divideSampleIfPossibleAtPresentationTime(start);
568 divideSampleIfPossibleAtPresentationTime(end);
569
570 auto removePresentationStart = trackBuffer.samples.presentationOrder().findSampleContainingOrAfterPresentationTime(start);
571 auto removePresentationEnd = trackBuffer.samples.presentationOrder().findSampleStartingOnOrAfterPresentationTime(end);
572 if (removePresentationStart == removePresentationEnd)
573 continue;
574
575 // 3.3 Remove all media data, from this track buffer, that contain starting timestamps greater than or equal to
576 // start and less than the remove end timestamp.
577 // NOTE: frames must be removed in decode order, so that all dependant frames between the frame to be removed
578 // and the next sync sample frame are removed. But we must start from the first sample in decode order, not
579 // presentation order.
580 auto minmaxDecodeTimeIterPair = std::minmax_element(removePresentationStart, removePresentationEnd, decodeTimeComparator);
581 auto& firstSample = *minmaxDecodeTimeIterPair.first->second;
582 auto& lastSample = *minmaxDecodeTimeIterPair.second->second;
583 auto removeDecodeStart = trackBuffer.samples.decodeOrder().findSampleWithDecodeKey({firstSample.decodeTime(), firstSample.presentationTime()});
584 auto removeDecodeLast = trackBuffer.samples.decodeOrder().findSampleWithDecodeKey({lastSample.decodeTime(), lastSample.presentationTime()});
585 auto removeDecodeEnd = trackBuffer.samples.decodeOrder().findSyncSampleAfterDecodeIterator(removeDecodeLast);
586
587 DecodeOrderSampleMap::MapType erasedSamples(removeDecodeStart, removeDecodeEnd);
588#if !RELEASE_LOG_DISABLED
589 PlatformTimeRanges erasedRanges = removeSamplesFromTrackBuffer(erasedSamples, trackBuffer, this, "removeCodedFrames");
590#else
591 PlatformTimeRanges erasedRanges = removeSamplesFromTrackBuffer(erasedSamples, trackBuffer);
592#endif
593
594 // Only force the TrackBuffer to re-enqueue if the removed ranges overlap with enqueued and possibly
595 // not yet displayed samples.
596 if (trackBuffer.highestEnqueuedPresentationTime.isValid() && currentTime < trackBuffer.highestEnqueuedPresentationTime) {
597 PlatformTimeRanges possiblyEnqueuedRanges(currentTime, trackBuffer.highestEnqueuedPresentationTime);
598 possiblyEnqueuedRanges.intersectWith(erasedRanges);
599 if (possiblyEnqueuedRanges.length()) {
600 trackBuffer.needsReenqueueing = true;
601 DEBUG_LOG(LOGIDENTIFIER, "the range in removeCodedFrames() includes already enqueued samples, reenqueueing from ", currentTime);
602 reenqueueMediaForTime(trackBuffer, trackID, currentTime);
603 }
604 }
605
606 erasedRanges.invert();
607 trackBuffer.buffered.intersectWith(erasedRanges);
608 setBufferedDirty(true);
609
610 // 3.4 If this object is in activeSourceBuffers, the current playback position is greater than or equal to start
611 // and less than the remove end timestamp, and HTMLMediaElement.readyState is greater than HAVE_METADATA, then set
612 // the HTMLMediaElement.readyState attribute to HAVE_METADATA and stall playback.
613 if (isActive() && currentTime >= start && currentTime < end && readyState() > MediaPlayer::ReadyState::HaveMetadata)
614 setReadyState(MediaPlayer::ReadyState::HaveMetadata);
615 }
616
617 updateBufferedFromTrackBuffers(isEnded);
618
619 // 4. If buffer full flag equals true and this object is ready to accept more bytes, then set the buffer full flag to false.
620 // No-op
621
622 updateHighestPresentationTimestamp();
623
624 LOG(Media, "SourceBuffer::removeCodedFrames(%p) - buffered = %s", this, toString(m_buffered->ranges()).utf8().data());
625
626 completionHandler();
627}
628
629void SourceBufferPrivate::evictCodedFrames(uint64_t newDataSize, uint64_t maximumBufferSize, const MediaTime& currentTime, const MediaTime& duration, bool isEnded)
630{
631 // 3.5.13 Coded Frame Eviction Algorithm
632 // http://www.w3.org/TR/media-source/#sourcebuffer-coded-frame-eviction
633
634 if (!m_isAttached)
635 return;
636
637 // This algorithm is run to free up space in this source buffer when new data is appended.
638 // 1. Let new data equal the data that is about to be appended to this SourceBuffer.
639 // 2. If the buffer full flag equals false, then abort these steps.
640 if (!isBufferFullFor(newDataSize, maximumBufferSize))
641 return;
642
643 // 3. Let removal ranges equal a list of presentation time ranges that can be evicted from
644 // the presentation to make room for the new data.
645
646 // NOTE: begin by removing data from the beginning of the buffered ranges, 30 seconds at
647 // a time, up to 30 seconds before currentTime.
648 MediaTime thirtySeconds = MediaTime(30, 1);
649 MediaTime maximumRangeEnd = currentTime - thirtySeconds;
650
651#if !RELEASE_LOG_DISABLED
652 uint64_t initialBufferedSize = totalTrackBufferSizeInBytes();
653 DEBUG_LOG(LOGIDENTIFIER, "currentTime = ", currentTime, ", require ", initialBufferedSize + newDataSize, " bytes, maximum buffer size is ", maximumBufferSize);
654#endif
655
656 MediaTime rangeStart = MediaTime::invalidTime();
657
658 for (auto& trackBuffer : m_trackBufferMap.values()) {
659 auto iter = trackBuffer.get().samples.presentationOrder().findSampleContainingOrAfterPresentationTime(MediaTime::zeroTime());
660 if (iter != trackBuffer.get().samples.presentationOrder().end()) {
661 MediaTime startTime = iter->first;
662 if (rangeStart.isInvalid() || startTime < rangeStart)
663 rangeStart = startTime;
664 }
665 }
666
667 if (rangeStart.isInvalid())
668 rangeStart = MediaTime::zeroTime();
669
670 MediaTime rangeEnd = rangeStart + thirtySeconds;
671 while (rangeStart < maximumRangeEnd) {
672 // 4. For each range in removal ranges, run the coded frame removal algorithm with start and
673 // end equal to the removal range start and end timestamp respectively.
674 removeCodedFrames(rangeStart, std::min(rangeEnd, maximumRangeEnd), currentTime, isEnded);
675 if (!isBufferFullFor(newDataSize, maximumBufferSize)) {
676 break;
677 }
678
679 rangeStart += thirtySeconds;
680 rangeEnd += thirtySeconds;
681 }
682
683 if (!isBufferFullFor(newDataSize, maximumBufferSize)) {
684#if !RELEASE_LOG_DISABLED
685 DEBUG_LOG(LOGIDENTIFIER, "evicted ", initialBufferedSize - totalTrackBufferSizeInBytes());
686#endif
687 return;
688 }
689
690 // If there still isn't enough free space and there buffers in time ranges after the current range (ie. there is a gap after
691 // the current buffered range), delete 30 seconds at a time from duration back to the current time range or 30 seconds after
692 // currenTime whichever we hit first.
693 auto buffered = m_buffered->ranges();
694 uint64_t currentTimeRange = buffered.find(currentTime);
695 if (!buffered.length() || currentTimeRange == buffered.length() - 1) {
696#if !RELEASE_LOG_DISABLED
697 ERROR_LOG(LOGIDENTIFIER, "FAILED to free enough after evicting ", initialBufferedSize - totalTrackBufferSizeInBytes());
698#endif
699 return;
700 }
701
702 MediaTime minimumRangeStart =
703 currentTimeRange == notFound ? currentTime + thirtySeconds : std::max(currentTime + thirtySeconds, buffered.end(currentTimeRange));
704
705 rangeEnd = duration;
706 if (!rangeEnd.isFinite()) {
707 rangeEnd = buffered.maximumBufferedTime();
708#if !RELEASE_LOG_DISABLED
709 DEBUG_LOG(LOGIDENTIFIER, "MediaSource duration is not a finite value, using maximum buffered time: ", rangeEnd);
710#endif
711 }
712
713 rangeStart = rangeEnd - thirtySeconds;
714 while (rangeEnd > minimumRangeStart) {
715 // 4. For each range in removal ranges, run the coded frame removal algorithm with start and
716 // end equal to the removal range start and end timestamp respectively.
717 removeCodedFrames(std::max(minimumRangeStart, rangeStart), rangeEnd, currentTime, isEnded);
718
719 if (!isBufferFullFor(newDataSize, maximumBufferSize)) {
720 break;
721 }
722
723 rangeStart -= thirtySeconds;
724 rangeEnd -= thirtySeconds;
725 }
726
727#if !RELEASE_LOG_DISABLED
728 if (isBufferFullFor(newDataSize, maximumBufferSize))
729 ERROR_LOG(LOGIDENTIFIER, "FAILED to free enough after evicting ", initialBufferedSize - totalTrackBufferSizeInBytes());
730 else
731 DEBUG_LOG(LOGIDENTIFIER, "evicted ", initialBufferedSize - totalTrackBufferSizeInBytes());
732#endif
733}
734
735bool SourceBufferPrivate::isBufferFullFor(uint64_t requiredSize, uint64_t maximumBufferSize)
736{
737 auto totalRequired = checkedSum<uint64_t>(totalTrackBufferSizeInBytes(), requiredSize);
738 if (totalRequired.hasOverflowed())
739 return true;
740
741 return totalRequired >= maximumBufferSize;
742}

803{

805 // IDs match the ones in the first initialization segment.

810 }
811 }
812

814 for (auto& videoTrackInfo : segment.videoTracks) {
815 if (!m_trackBufferMap.contains(videoTrackInfo.track->id()))
816 return false;
817 }
818 }
819
820 if (segment.textTracks.size() >= 2) {
821 for (auto& textTrackInfo : segment.videoTracks) {
822 if (!m_trackBufferMap.contains(textTrackInfo.track->id()))
823 return false;
824 }
825 }
826
827 return true;

829
830void SourceBufferPrivate::didReceiveSample(Ref<MediaSample>&& originalSample)
831{
832 if (!m_isAttached)
833 return;
834
835 // 3.5.1 Segment Parser Loop
836 // 6.1 If the first initialization segment received flag is false, (Note: Issue # 155 & changeType()
837 // algorithm) or the pending initialization segment for changeType flag is true, (End note)
838 // then run the append error algorithm
839 // with the decode error parameter set to true and abort this algorithm.
840 // Note: current design makes SourceBuffer somehow ignorant of append state - it's more a thing
841 // of SourceBufferPrivate. That's why this check can't really be done in appendInternal.
842 // unless we force some kind of design with state machine switching.
843
844 if ((!m_receivedFirstInitializationSegment || m_pendingInitializationSegmentForChangeType) && m_client) {
845 m_client->sourceBufferPrivateAppendError(true);
846 return;
847 }
848
849 // 3.5.8 Coded Frame Processing
850 // http://www.w3.org/TR/media-source/#sourcebuffer-coded-frame-processing
851
852 // When complete coded frames have been parsed by the segment parser loop then the following steps
853 // are run:
854 // 1. For each coded frame in the media segment run the following steps:
855 // 1.1. Loop Top
856
857 Ref<MediaSample> sample = WTFMove(originalSample);
858
859 do {
860 MediaTime presentationTimestamp;
861 MediaTime decodeTimestamp;
862
863 // NOTE: this is out-of-order, but we need the timescale from the
864 // sample's duration for timestamp generation.
865 // 1.2 Let frame duration be a double precision floating point representation of the coded frame's
866 // duration in seconds.
867 MediaTime frameDuration = sample->duration();
868
869 if (m_shouldGenerateTimestamps) {
870 // If generate timestamps flag equals true:
871 // 1. Let presentation timestamp equal 0.
872 // NOTE: Use the duration timscale for the presentation timestamp, as this will eliminate
873 // timescale rounding when generating timestamps.
874 presentationTimestamp = { 0, frameDuration.timeScale() };
875
876 // 2. Let decode timestamp equal 0.
877 decodeTimestamp = { 0, frameDuration.timeScale() };
878 } else {
879 // Otherwise:
880 // 1. Let presentation timestamp be a double precision floating point representation of
881 // the coded frame's presentation timestamp in seconds.
882 presentationTimestamp = sample->presentationTime();
883
884 // 2. Let decode timestamp be a double precision floating point representation of the coded frame's
885 // decode timestamp in seconds.
886 decodeTimestamp = sample->decodeTime();
887 }
888
889 // 1.3 If mode equals "sequence" and group start timestamp is set, then run the following steps:
890 if (m_appendMode == SourceBufferAppendMode::Sequence && m_groupStartTimestamp.isValid()) {
891 // 1.3.1 Set timestampOffset equal to group start timestamp - presentation timestamp.
892 m_timestampOffset = m_groupStartTimestamp;
893
894 for (auto& trackBuffer : m_trackBufferMap.values()) {
895 trackBuffer.get().lastFrameTimescale = 0;
896 trackBuffer.get().roundedTimestampOffset = MediaTime::invalidTime();
897 }
898
899 // 1.3.2 Set group end timestamp equal to group start timestamp.
900 m_groupEndTimestamp = m_groupStartTimestamp;
901
902 // 1.3.3 Set the need random access point flag on all track buffers to true.
903 for (auto& trackBuffer : m_trackBufferMap.values())
904 trackBuffer.get().needRandomAccessFlag = true;
905
906 // 1.3.4 Unset group start timestamp.
907 m_groupStartTimestamp = MediaTime::invalidTime();
908 }
909
910 // NOTE: this is out-of-order, but we need TrackBuffer to be able to cache the results of timestamp offset rounding
911 // 1.5 Let track buffer equal the track buffer that the coded frame will be added to.
912 AtomString trackID = sample->trackID();
913 auto it = m_trackBufferMap.find(trackID);
914 if (it == m_trackBufferMap.end()) {
915 // The client managed to append a sample with a trackID not present in the initialization
916 // segment. This would be a good place to post an message to the developer console.
917 m_client->sourceBufferPrivateDidDropSample();
918 return;
919 }
920 TrackBuffer& trackBuffer = it->value;
921
922 MediaTime microsecond(1, 1000000);
923
924 auto roundTowardsTimeScaleWithRoundingMargin = [] (const MediaTime& time, uint32_t timeScale, const MediaTime& roundingMargin) {
925 while (true) {
926 MediaTime roundedTime = time.toTimeScale(timeScale);
927 if (abs(roundedTime - time) < roundingMargin || timeScale >= MediaTime::MaximumTimeScale)
928 return roundedTime;
929
930 if (!WTF::safeMultiply(timeScale, 2, timeScale) || timeScale > MediaTime::MaximumTimeScale)
931 timeScale = MediaTime::MaximumTimeScale;
932 }
933 };
934
935 // 1.4 If timestampOffset is not 0, then run the following steps:
936 if (m_timestampOffset) {
937 if (!trackBuffer.roundedTimestampOffset.isValid() || presentationTimestamp.timeScale() != trackBuffer.lastFrameTimescale) {
938 trackBuffer.lastFrameTimescale = presentationTimestamp.timeScale();
939 trackBuffer.roundedTimestampOffset = roundTowardsTimeScaleWithRoundingMargin(m_timestampOffset, trackBuffer.lastFrameTimescale, microsecond);
940 }
941
942 // 1.4.1 Add timestampOffset to the presentation timestamp.
943 presentationTimestamp += trackBuffer.roundedTimestampOffset;
944
945 // 1.4.2 Add timestampOffset to the decode timestamp.
946 decodeTimestamp += trackBuffer.roundedTimestampOffset;
947 }
948
949 // 1.6 If last decode timestamp for track buffer is set and decode timestamp is less than last
950 // decode timestamp:
951 // OR
952 // If last decode timestamp for track buffer is set and the difference between decode timestamp and
953 // last decode timestamp is greater than 2 times last frame duration:
954 MediaTime decodeDurationToCheck = trackBuffer.greatestDecodeDuration;
955
956 if (decodeDurationToCheck.isValid() && trackBuffer.lastFrameDuration.isValid()
957 && (trackBuffer.lastFrameDuration > decodeDurationToCheck))
958 decodeDurationToCheck = trackBuffer.lastFrameDuration;
959
960 if (trackBuffer.lastDecodeTimestamp.isValid() && (decodeTimestamp < trackBuffer.lastDecodeTimestamp
961 || (decodeDurationToCheck.isValid() && abs(decodeTimestamp - trackBuffer.lastDecodeTimestamp) > (decodeDurationToCheck * 2)))) {
962
963 // 1.6.1:
964 if (m_appendMode == SourceBufferAppendMode::Segments) {
965 // If mode equals "segments":
966 // Set group end timestamp to presentation timestamp.
967 m_groupEndTimestamp = presentationTimestamp;
968 } else {
969 // If mode equals "sequence":
970 // Set group start timestamp equal to the group end timestamp.
971 m_groupStartTimestamp = m_groupEndTimestamp;
972 }
973
974 for (auto& trackBuffer : m_trackBufferMap.values()) {
975 // 1.6.2 Unset the last decode timestamp on all track buffers.
976 trackBuffer.get().lastDecodeTimestamp = MediaTime::invalidTime();
977 // 1.6.3 Unset the last frame duration on all track buffers.
978 trackBuffer.get().greatestDecodeDuration = MediaTime::invalidTime();
979 trackBuffer.get().lastFrameDuration = MediaTime::invalidTime();
980 // 1.6.4 Unset the highest presentation timestamp on all track buffers.
981 trackBuffer.get().highestPresentationTimestamp = MediaTime::invalidTime();
982 // 1.6.5 Set the need random access point flag on all track buffers to true.
983 trackBuffer.get().needRandomAccessFlag = true;
984 }
985
986 // 1.6.6 Jump to the Loop Top step above to restart processing of the current coded frame.
987 continue;
988 }
989
990 if (m_appendMode == SourceBufferAppendMode::Sequence) {
991 // Use the generated timestamps instead of the sample's timestamps.
992 sample->setTimestamps(presentationTimestamp, decodeTimestamp);
993 } else if (trackBuffer.roundedTimestampOffset) {
994 // Reflect the timestamp offset into the sample.
995 sample->offsetTimestampsBy(trackBuffer.roundedTimestampOffset);
996 }
997
998 DEBUG_LOG(LOGIDENTIFIER, sample.get());
999
1000 // 1.7 Let frame end timestamp equal the sum of presentation timestamp and frame duration.
1001 MediaTime frameEndTimestamp = presentationTimestamp + frameDuration;
1002
1003 // 1.8 If presentation timestamp is less than appendWindowStart, then set the need random access
1004 // point flag to true, drop the coded frame, and jump to the top of the loop to start processing
1005 // the next coded frame.
1006 // 1.9 If frame end timestamp is greater than appendWindowEnd, then set the need random access
1007 // point flag to true, drop the coded frame, and jump to the top of the loop to start processing
1008 // the next coded frame.
1009 if (presentationTimestamp < m_appendWindowStart || frameEndTimestamp > m_appendWindowEnd) {
1010 // 1.8 Note.
1011 // Some implementations MAY choose to collect some of these coded frames with presentation
1012 // timestamp less than appendWindowStart and use them to generate a splice at the first coded
1013 // frame that has a presentation timestamp greater than or equal to appendWindowStart even if
1014 // that frame is not a random access point. Supporting this requires multiple decoders or
1015 // faster than real-time decoding so for now this behavior will not be a normative
1016 // requirement.
1017 // 1.9 Note.
1018 // Some implementations MAY choose to collect coded frames with presentation timestamp less
1019 // than appendWindowEnd and frame end timestamp greater than appendWindowEnd and use them to
1020 // generate a splice across the portion of the collected coded frames within the append
1021 // window at time of collection, and the beginning portion of later processed frames which
1022 // only partially overlap the end of the collected coded frames. Supporting this requires
1023 // multiple decoders or faster than real-time decoding so for now this behavior will not be a
1024 // normative requirement. In conjunction with collecting coded frames that span
1025 // appendWindowStart, implementations MAY thus support gapless audio splicing.
1026 // Audio MediaSamples are typically made of packed audio samples. Trim sample to make it fit within the appendWindow.
1027 if (sample->isDivisable()) {
1028 std::pair<RefPtr<MediaSample>, RefPtr<MediaSample>> replacementSamples = sample->divide(m_appendWindowStart);
1029 if (replacementSamples.second) {
1030 ASSERT(replacementSamples.second->presentationTime() >= m_appendWindowStart);
1031 replacementSamples = replacementSamples.second->divide(m_appendWindowEnd, MediaSample::UseEndTime::Use);
1032 if (replacementSamples.first) {
1033 sample = replacementSamples.first.releaseNonNull();
1034 ASSERT(sample->presentationTime() >= m_appendWindowStart && sample->presentationTime() + sample->duration() <= m_appendWindowEnd);
1035 if (m_appendMode != SourceBufferAppendMode::Sequence && trackBuffer.roundedTimestampOffset)
1036 sample->offsetTimestampsBy(-trackBuffer.roundedTimestampOffset);
1037 continue;
1038 }
1039 }
1040 }
1041 trackBuffer.needRandomAccessFlag = true;
1042 m_client->sourceBufferPrivateDidDropSample();
1043 return;
1044 }
1045
1046 // If the decode timestamp is less than the presentation start time, then run the end of stream
1047 // algorithm with the error parameter set to "decode", and abort these steps.
1048 // NOTE: Until <https://www.w3.org/Bugs/Public/show_bug.cgi?id=27487> is resolved, we will only check
1049 // the presentation timestamp.
1050 MediaTime presentationStartTime = MediaTime::zeroTime();
1051 if (presentationTimestamp < presentationStartTime) {
1052 ERROR_LOG(LOGIDENTIFIER, "failing because presentationTimestamp (", presentationTimestamp, ") < presentationStartTime (", presentationStartTime, ")");
1053 m_client->sourceBufferPrivateStreamEndedWithDecodeError();
1054 return;
1055 }
1056
1057 // 1.10 If the need random access point flag on track buffer equals true, then run the following steps:
1058 if (trackBuffer.needRandomAccessFlag) {
1059 // 1.11.1 If the coded frame is not a random access point, then drop the coded frame and jump
1060 // to the top of the loop to start processing the next coded frame.
1061 if (!sample->isSync()) {
1062 m_client->sourceBufferPrivateDidDropSample();
1063 return;
1064 }
1065
1066 // 1.11.2 Set the need random access point flag on track buffer to false.
1067 trackBuffer.needRandomAccessFlag = false;
1068 }
1069
1070 // 1.11 Let spliced audio frame be an unset variable for holding audio splice information
1071 // 1.12 Let spliced timed text frame be an unset variable for holding timed text splice information
1072 // FIXME: Add support for sample splicing.
1073
1074 SampleMap erasedSamples;
1075
1076 // 1.13 If last decode timestamp for track buffer is unset and presentation timestamp
1077 // falls within the presentation interval of a coded frame in track buffer, then run the
1078 // following steps:
1079 if (trackBuffer.lastDecodeTimestamp.isInvalid()) {
1080 auto iter = trackBuffer.samples.presentationOrder().findSampleContainingPresentationTime(presentationTimestamp);
1081 if (iter != trackBuffer.samples.presentationOrder().end()) {
1082 // 1.13.1 Let overlapped frame be the coded frame in track buffer that matches the condition above.
1083 RefPtr<MediaSample> overlappedFrame = iter->second;
1084
1085 // 1.13.2 If track buffer contains audio coded frames:
1086 // Run the audio splice frame algorithm and if a splice frame is returned, assign it to
1087 // spliced audio frame.
1088 // FIXME: Add support for sample splicing.
1089
1090 // If track buffer contains video coded frames:
1091 if (trackBuffer.description && trackBuffer.description->isVideo()) {
1092 // 1.13.2.1 Let overlapped frame presentation timestamp equal the presentation timestamp
1093 // of overlapped frame.
1094 MediaTime overlappedFramePresentationTimestamp = overlappedFrame->presentationTime();
1095
1096 // 1.13.2.2 Let remove window timestamp equal overlapped frame presentation timestamp
1097 // plus 1 microsecond.
1098 MediaTime removeWindowTimestamp = overlappedFramePresentationTimestamp + microsecond;
1099
1100 // 1.13.2.3 If the presentation timestamp is less than the remove window timestamp,
1101 // then remove overlapped frame and any coded frames that depend on it from track buffer.
1102 if (presentationTimestamp < removeWindowTimestamp)
1103 erasedSamples.addSample(*iter->second);
1104 }
1105
1106 // If track buffer contains timed text coded frames:
1107 // Run the text splice frame algorithm and if a splice frame is returned, assign it to spliced timed text frame.
1108 // FIXME: Add support for sample splicing.
1109 }
1110 }
1111
1112 // 1.14 Remove existing coded frames in track buffer:
1113 // If highest presentation timestamp for track buffer is not set:
1114 if (trackBuffer.highestPresentationTimestamp.isInvalid()) {
1115 // Remove all coded frames from track buffer that have a presentation timestamp greater than or
1116 // equal to presentation timestamp and less than frame end timestamp.
1117 auto iterPair = trackBuffer.samples.presentationOrder().findSamplesBetweenPresentationTimes(presentationTimestamp, frameEndTimestamp);
1118 if (iterPair.first != trackBuffer.samples.presentationOrder().end())
1119 erasedSamples.addRange(iterPair.first, iterPair.second);
1120 }
1121
1122 // When appending media containing B-frames (media whose samples' presentation timestamps
1123 // do not increase monotonically, the prior erase steps could leave a sample in the trackBuffer
1124 // which will be disconnected from its previous I-frame. If the incoming frame is an I-frame,
1125 // remove all samples in decode order between the incoming I-frame's decode timestamp and the
1126 // next I-frame. See <https://github.com/w3c/media-source/issues/187> for a discussion of what
1127 // the how the MSE specification should handlie this secnario.
1128 do {
1129 if (!sample->isSync())
1130 break;
1131
1132 DecodeOrderSampleMap::KeyType decodeKey(sample->decodeTime(), sample->presentationTime());
1133 auto nextSampleInDecodeOrder = trackBuffer.samples.decodeOrder().findSampleAfterDecodeKey(decodeKey);
1134 if (nextSampleInDecodeOrder == trackBuffer.samples.decodeOrder().end())
1135 break;
1136
1137 if (nextSampleInDecodeOrder->second->isSync())
1138 break;
1139
1140 auto nextSyncSample = trackBuffer.samples.decodeOrder().findSyncSampleAfterDecodeIterator(nextSampleInDecodeOrder);
1141 INFO_LOG(LOGIDENTIFIER, "Discovered out-of-order frames, from: ", *nextSampleInDecodeOrder->second, " to: ", (nextSyncSample == trackBuffer.samples.decodeOrder().end() ? "[end]"_s : toString(*nextSyncSample->second)));
1142 erasedSamples.addRange(nextSampleInDecodeOrder, nextSyncSample);
1143 } while (false);
1144
1145 // There are many files out there where the frame times are not perfectly contiguous and may have small overlaps
1146 // between the beginning of a frame and the end of the previous one; therefore a tolerance is needed whenever
1147 // durations are considered.
1148 // For instance, most WebM files are muxed rounded to the millisecond (the default TimecodeScale of the format)
1149 // but their durations use a finer timescale (causing a sub-millisecond overlap). More rarely, there are also
1150 // MP4 files with slightly off tfdt boxes, presenting a similar problem at the beginning of each fragment.
1151 const MediaTime contiguousFrameTolerance = MediaTime(1, 1000);
1152
1153 // If highest presentation timestamp for track buffer is set and less than or equal to presentation timestamp
1154 if (trackBuffer.highestPresentationTimestamp.isValid() && trackBuffer.highestPresentationTimestamp - contiguousFrameTolerance <= presentationTimestamp) {
1155 // Remove all coded frames from track buffer that have a presentation timestamp greater than highest
1156 // presentation timestamp and less than or equal to frame end timestamp.
1157 do {
1158 // NOTE: Searching from the end of the trackBuffer will be vastly more efficient if the search range is
1159 // near the end of the buffered range. Use a linear-backwards search if the search range is within one
1160 // frame duration of the end:
1161 unsigned bufferedLength = trackBuffer.buffered.length();
1162 if (!bufferedLength)
1163 break;
1164
1165 MediaTime highestBufferedTime = trackBuffer.buffered.maximumBufferedTime();
1166 MediaTime eraseBeginTime = trackBuffer.highestPresentationTimestamp - contiguousFrameTolerance;
1167 MediaTime eraseEndTime = frameEndTimestamp - contiguousFrameTolerance;
1168
1169 PresentationOrderSampleMap::iterator_range range;
1170 if (highestBufferedTime - trackBuffer.highestPresentationTimestamp < trackBuffer.lastFrameDuration) {
1171 // If the new frame is at the end of the buffered ranges, perform a sequential scan from end (O(1)).
1172 range = trackBuffer.samples.presentationOrder().findSamplesBetweenPresentationTimesFromEnd(eraseBeginTime, eraseEndTime);
1173 } else {
1174 // In any other case, perform a binary search (O(log(n)).
1175 range = trackBuffer.samples.presentationOrder().findSamplesBetweenPresentationTimes(eraseBeginTime, eraseEndTime);
1176 }
1177
1178 if (range.first != trackBuffer.samples.presentationOrder().end())
1179 erasedSamples.addRange(range.first, range.second);
1180 } while (false);
1181 }
1182
1183 // 1.15 Remove decoding dependencies of the coded frames removed in the previous step:
1184 DecodeOrderSampleMap::MapType dependentSamples;
1185 if (!erasedSamples.empty()) {
1186 // If detailed information about decoding dependencies is available:
1187 // FIXME: Add support for detailed dependency information
1188
1189 // Otherwise: Remove all coded frames between the coded frames removed in the previous step
1190 // and the next random access point after those removed frames.
1191 auto firstDecodeIter = trackBuffer.samples.decodeOrder().findSampleWithDecodeKey(erasedSamples.decodeOrder().begin()->first);
1192 auto lastDecodeIter = trackBuffer.samples.decodeOrder().findSampleWithDecodeKey(erasedSamples.decodeOrder().rbegin()->first);
1193 auto nextSyncIter = trackBuffer.samples.decodeOrder().findSyncSampleAfterDecodeIterator(lastDecodeIter);
1194 dependentSamples.insert(firstDecodeIter, nextSyncIter);
1195
1196 // NOTE: in the case of b-frames, the previous step may leave in place samples whose presentation
1197 // timestamp < presentationTime, but whose decode timestamp >= decodeTime. These will eventually cause
1198 // a decode error if left in place, so remove these samples as well.
1199 DecodeOrderSampleMap::KeyType decodeKey(sample->decodeTime(), sample->presentationTime());
1200 auto samplesWithHigherDecodeTimes = trackBuffer.samples.decodeOrder().findSamplesBetweenDecodeKeys(decodeKey, erasedSamples.decodeOrder().begin()->first);
1201 if (samplesWithHigherDecodeTimes.first != samplesWithHigherDecodeTimes.second)
1202 dependentSamples.insert(samplesWithHigherDecodeTimes.first, samplesWithHigherDecodeTimes.second);
1203
1204#if !RELEASE_LOG_DISABLED
1205 PlatformTimeRanges erasedRanges = removeSamplesFromTrackBuffer(dependentSamples, trackBuffer, this, "didReceiveSample");
1206#else
1207 PlatformTimeRanges erasedRanges = removeSamplesFromTrackBuffer(dependentSamples, trackBuffer);
1208#endif
1209
1210 // Only force the TrackBuffer to re-enqueue if the removed ranges overlap with enqueued and possibly
1211 // not yet displayed samples.
1212 MediaTime currentTime = currentMediaTime();
1213 if (trackBuffer.highestEnqueuedPresentationTime.isValid() && currentTime < trackBuffer.highestEnqueuedPresentationTime) {
1214 PlatformTimeRanges possiblyEnqueuedRanges(currentTime, trackBuffer.highestEnqueuedPresentationTime);
1215 possiblyEnqueuedRanges.intersectWith(erasedRanges);
1216 if (possiblyEnqueuedRanges.length())
1217 trackBuffer.needsReenqueueing = true;
1218 }
1219
1220 erasedRanges.invert();
1221 trackBuffer.buffered.intersectWith(erasedRanges);
1222 setBufferedDirty(true);
1223 }
1224
1225 // 1.16 If spliced audio frame is set:
1226 // Add spliced audio frame to the track buffer.
1227 // If spliced timed text frame is set:
1228 // Add spliced timed text frame to the track buffer.
1229 // FIXME: Add support for sample splicing.
1230
1231 // Otherwise:
1232 // Add the coded frame with the presentation timestamp, decode timestamp, and frame duration to the track buffer.
1233 trackBuffer.samples.addSample(sample);
1234
1235 // Note: The terminology here is confusing: "enqueuing" means providing a frame to the inner media framework.
1236 // First, frames are inserted in the decode queue; later, at the end of the append some of the frames in the
1237 // decode may be "enqueued" (sent to the inner media framework) in `provideMediaData()`.
1238 //
1239 // In order to check whether a frame should be added to the decode queue we check that it does not precede any
1240 // frame already enqueued.
1241 //
1242 // Note that adding a frame to the decode queue is no guarantee that it will be actually enqueued at that point.
1243 // If the frame is after the discontinuity boundary, the enqueueing algorithm will hold it there until samples
1244 // with earlier timestamps are enqueued. The decode queue is not FIFO, but rather an ordered map.
1245 DecodeOrderSampleMap::KeyType decodeKey(sample->decodeTime(), sample->presentationTime());
1246 if (trackBuffer.lastEnqueuedDecodeKey.first.isInvalid() || decodeKey > trackBuffer.lastEnqueuedDecodeKey) {
1247 trackBuffer.decodeQueue.insert(DecodeOrderSampleMap::MapType::value_type(decodeKey, &sample.get()));
1248
1249 if (trackBuffer.minimumEnqueuedPresentationTime.isValid() && sample->presentationTime() < trackBuffer.minimumEnqueuedPresentationTime)
1250 trackBuffer.needsMinimumUpcomingPresentationTimeUpdating = true;
1251 }
1252
1253 // NOTE: the spec considers "Coded Frame Duration" to be the presentation duration, but this is not necessarily equal
1254 // to the decoded duration. When comparing deltas between decode timestamps, the decode duration, not the presentation.
1255 if (trackBuffer.lastDecodeTimestamp.isValid()) {
1256 MediaTime lastDecodeDuration = decodeTimestamp - trackBuffer.lastDecodeTimestamp;
1257 if (!trackBuffer.greatestDecodeDuration.isValid() || lastDecodeDuration > trackBuffer.greatestDecodeDuration)
1258 trackBuffer.greatestDecodeDuration = lastDecodeDuration;
1259 }
1260
1261 // 1.17 Set last decode timestamp for track buffer to decode timestamp.
1262 trackBuffer.lastDecodeTimestamp = decodeTimestamp;
1263
1264 // 1.18 Set last frame duration for track buffer to frame duration.
1265 trackBuffer.lastFrameDuration = frameDuration;
1266
1267 // 1.19 If highest presentation timestamp for track buffer is unset or frame end timestamp is greater
1268 // than highest presentation timestamp, then set highest presentation timestamp for track buffer
1269 // to frame end timestamp.
1270 if (trackBuffer.highestPresentationTimestamp.isInvalid() || frameEndTimestamp > trackBuffer.highestPresentationTimestamp)
1271 trackBuffer.highestPresentationTimestamp = frameEndTimestamp;
1272
1273 // 1.20 If frame end timestamp is greater than group end timestamp, then set group end timestamp equal
1274 // to frame end timestamp.
1275 if (m_groupEndTimestamp.isInvalid() || frameEndTimestamp > m_groupEndTimestamp)
1276 m_groupEndTimestamp = frameEndTimestamp;
1277
1278 // 1.21 If generate timestamps flag equals true, then set timestampOffset equal to frame end timestamp.
1279 if (m_shouldGenerateTimestamps) {
1280 m_timestampOffset = frameEndTimestamp;
1281 for (auto& trackBuffer : m_trackBufferMap.values()) {
1282 trackBuffer.get().lastFrameTimescale = 0;
1283 trackBuffer.get().roundedTimestampOffset = MediaTime::invalidTime();
1284 }
1285 }
1286
1287 // Eliminate small gaps between buffered ranges by coalescing
1288 // disjoint ranges separated by less than a "fudge factor".
1289 auto presentationEndTime = presentationTimestamp + frameDuration;
1290 auto nearestToPresentationStartTime = trackBuffer.buffered.nearest(presentationTimestamp);
1291 if (nearestToPresentationStartTime.isValid() && (presentationTimestamp - nearestToPresentationStartTime).isBetween(MediaTime::zeroTime(), timeFudgeFactor()))
1292 presentationTimestamp = nearestToPresentationStartTime;
1293
1294 auto nearestToPresentationEndTime = trackBuffer.buffered.nearest(presentationEndTime);
1295 if (nearestToPresentationEndTime.isValid() && (nearestToPresentationEndTime - presentationEndTime).isBetween(MediaTime::zeroTime(), timeFudgeFactor()))
1296 presentationEndTime = nearestToPresentationEndTime;
1297
1298 trackBuffer.buffered.add(presentationTimestamp, presentationEndTime);
1299 m_client->sourceBufferPrivateDidParseSample(frameDuration.toDouble());
1300 setBufferedDirty(true);
1301
1302 break;
1303 } while (true);
1304
1305 // Steps 2-4 will be handled by MediaSource::monitorSourceBuffers()
1306
1307 // 5. If the media segment contains data beyond the current duration, then run the duration change algorithm with new
1308 // duration set to the maximum of the current duration and the group end timestamp.
1309 if (m_groupEndTimestamp > duration())
1310 m_client->sourceBufferPrivateDurationChanged(m_groupEndTimestamp);
1311
1312 updateHighestPresentationTimestamp();
1313}
