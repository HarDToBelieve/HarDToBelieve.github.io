21#ifndef SurrogatePairAwareTextIterator_h

24#include <wtf/unicode/CharacterNames.h>
25#include <wtf/text/WTFString.h>

27namespace WebCore {
28
29class SurrogatePairAwareTextIterator {
30public:
31 // The passed in UChar pointer starts at 'currentIndex'. The iterator operatoes on the range [currentIndex, lastIndex].
32 // 'endIndex' denotes the maximum length of the UChar array, which might exceed 'lastIndex'.
33 SurrogatePairAwareTextIterator(const UChar*, unsigned currentIndex, unsigned lastIndex, unsigned endIndex);
34
35 inline bool consume(UChar32& character, unsigned& clusterLength)
36 {
37 if (m_currentIndex >= m_lastIndex)
38 return false;
39
40 character = *m_characters;
41 clusterLength = 1;
42
43 if (!U16_IS_SURROGATE(character))
44 return true;
45
46 return consumeSlowCase(character, clusterLength);
47 }
48
49 void advance(unsigned advanceLength)
50 {
51 m_characters += advanceLength;
52 m_currentIndex += advanceLength;
53 }
54
55 unsigned currentIndex() const { return m_currentIndex; }
56 const UChar* characters() const { return m_characters; }
57
58private:
59 bool consumeSlowCase(UChar32&, unsigned&);
60 UChar32 normalizeVoicingMarks();
61
62 const UChar* m_characters;
63 unsigned m_currentIndex;
64 unsigned m_lastIndex;
65 unsigned m_endIndex;
66};
67
68}

70#endif
