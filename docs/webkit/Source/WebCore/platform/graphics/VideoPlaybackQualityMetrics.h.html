26#pragma once
27
28#if ENABLE(VIDEO)
29
30namespace WebCore {
31
32struct VideoPlaybackQualityMetrics {
33 uint32_t totalVideoFrames { 0 };
34 uint32_t droppedVideoFrames { 0 };
35 uint32_t corruptedVideoFrames { 0 };
36 double totalFrameDelay { 0 };
37 uint32_t displayCompositedVideoFrames { 0 };
38
39 template<class Encoder>
40 void encode(Encoder& encoder) const
41 {
42 encoder << totalVideoFrames;
43 encoder << droppedVideoFrames;
44 encoder << corruptedVideoFrames;
45 encoder << totalFrameDelay;
46 encoder << displayCompositedVideoFrames;
47 }
48
49 template <class Decoder>
50 static std::optional<VideoPlaybackQualityMetrics> decode(Decoder& decoder)
51 {
52 std::optional<uint32_t> totalVideoFrames;
53 decoder >> totalVideoFrames;
54 if (!totalVideoFrames)
55 return std::nullopt;
56
57 std::optional<uint32_t> droppedVideoFrames;
58 decoder >> droppedVideoFrames;
59 if (!droppedVideoFrames)
60 return std::nullopt;
61
62 std::optional<uint32_t> corruptedVideoFrames;
63 decoder >> corruptedVideoFrames;
64 if (!corruptedVideoFrames)
65 return std::nullopt;
66
67 std::optional<double> totalFrameDelay;
68 decoder >> totalFrameDelay;
69 if (!totalFrameDelay)
70 return std::nullopt;
71
72 std::optional<uint32_t> displayCompositedVideoFrames;
73 decoder >> displayCompositedVideoFrames;
74 if (!displayCompositedVideoFrames)
75 return std::nullopt;
76
77 return {{ *totalVideoFrames, *droppedVideoFrames, *corruptedVideoFrames, *totalFrameDelay, *displayCompositedVideoFrames }};
78 }
79};
80
81} // namespace WebCore
82
83#endif // ENABLE(VIDEO)
