33namespace WebCore {
34
35class VideoTrackPrivateClient : public TrackPrivateBaseClient {
36public:
37 virtual void selectedChanged(bool) = 0;
38};
39
40class VideoTrackPrivate : public TrackPrivateBase {
41public:
42 void setClient(VideoTrackPrivateClient* client) { m_client = client; }
43 VideoTrackPrivateClient* client() const override { return m_client; }
44
45 virtual void setSelected(bool selected)
46 {
47 if (m_selected == selected)
48 return;
49 m_selected = selected;
50 if (m_client)
51 m_client->selectedChanged(m_selected);
52 if (m_selectedChangedCallback)
53 m_selectedChangedCallback(*this, m_selected);
54 }
55 virtual bool selected() const { return m_selected; }
56
57 enum Kind { Alternative, Captions, Main, Sign, Subtitles, Commentary, None };
58 virtual Kind kind() const { return None; }
59
60#if !RELEASE_LOG_DISABLED
61 const char* logClassName() const final { return "VideoTrackPrivate"; }
62#endif
63
64 using SelectedChangedCallback = Function<void(VideoTrackPrivate&, bool selected)>;
65 void setSelectedChangedCallback(SelectedChangedCallback&& callback) { m_selectedChangedCallback = WTFMove(callback); }
66
67protected:
68 VideoTrackPrivate() = default;
69
70private:
71 VideoTrackPrivateClient* m_client { nullptr };
72 bool m_selected { false };
73 SelectedChangedCallback m_selectedChangedCallback;
74};
