31
32class FontCascade;
33class Font;
34class TextRun;
35struct GlyphData;
36struct GlyphIndexRange;
37struct OriginalAdvancesForCharacterTreatedAsSpace;

39using CharactersTreatedAsSpace = Vector<OriginalAdvancesForCharacterTreatedAsSpace, 64>;

41struct WidthIterator {
42 WTF_MAKE_FAST_ALLOCATED;
43public:
44 WidthIterator(const FontCascade&, const TextRun&, HashSet<const Font*>* fallbackFonts = 0, bool accountForGlyphBounds = false, bool forTextEmphasis = false);
45
46 void advance(unsigned to, GlyphBuffer&);
47 bool advanceOneCharacter(float& width, GlyphBuffer&);
48 void finalize(GlyphBuffer&);
49
50 float maxGlyphBoundingBoxY() const { ASSERT(m_accountForGlyphBounds); return m_maxGlyphBoundingBoxY; }
51 float minGlyphBoundingBoxY() const { ASSERT(m_accountForGlyphBounds); return m_minGlyphBoundingBoxY; }
52 float firstGlyphOverflow() const { ASSERT(m_accountForGlyphBounds); return m_firstGlyphOverflow; }
53 float lastGlyphOverflow() const { ASSERT(m_accountForGlyphBounds); return m_lastGlyphOverflow; }
54
55 const TextRun& run() const { return m_run; }
56 float runWidthSoFar() const { return m_runWidthSoFar; }
57 unsigned currentCharacterIndex() const { return m_currentCharacterIndex; }
58
59private:
60 GlyphData glyphDataForCharacter(UChar32, bool mirror);
61 template <typename TextIterator>
62 inline void advanceInternal(TextIterator&, GlyphBuffer&);
63
64 enum class TransformsType { None, Forced, NotForced };
65 TransformsType shouldApplyFontTransforms(const GlyphBuffer&, unsigned lastGlyphCount, unsigned currentCharacterIndex) const;
66 float applyFontTransforms(GlyphBuffer&, unsigned lastGlyphCount, unsigned currentCharacterIndex, const Font&, bool force, CharactersTreatedAsSpace&);
67 void commitCurrentFontRange(GlyphBuffer&, unsigned lastGlyphCount, unsigned currentCharacterIndex, const Font&, const Font& primaryFont, UChar32 character, float widthOfCurrentFontRange, CharactersTreatedAsSpace&);
68
69 bool hasExtraSpacing() const;
70 void applyExtraSpacingAfterShaping(GlyphBuffer&, unsigned characterStartIndex, unsigned glyphBufferStartIndex, unsigned characterDestinationIndex, float startingRunWidth);
71 struct AdditionalWidth {
72 float left;
73 float right;
74 float leftExpansion;
75 float rightExpansion;
76 };
77 AdditionalWidth calculateAdditionalWidth(GlyphBuffer&, GlyphBufferStringOffset currentCharacterIndex, unsigned leadingGlyphIndex, unsigned trailingGlyphIndex, float position) const;
78 void applyAdditionalWidth(GlyphBuffer&, GlyphIndexRange, float leftAdditionalWidth, float rightAdditionalWidth, float leftExpansionAdditionalWidth, float rightExpansionAdditionalWidth);
79
80 const FontCascade& m_font;
81 const TextRun& m_run;
82 HashSet<const Font*>* m_fallbackFonts { nullptr };
83
84 std::optional<unsigned> m_lastCharacterIndex;
85 unsigned m_currentCharacterIndex { 0 };
86 float m_leftoverJustificationWidth { 0 };
87 float m_runWidthSoFar { 0 };
88 float m_expansion { 0 };
89 float m_expansionPerOpportunity { 0 };
90 float m_maxGlyphBoundingBoxY { std::numeric_limits<float>::min() };
91 float m_minGlyphBoundingBoxY { std::numeric_limits<float>::max() };
92 float m_firstGlyphOverflow { 0 };
93 float m_lastGlyphOverflow { 0 };
94 bool m_containsTabs { false };
95 bool m_isAfterExpansion { false };
96 bool m_accountForGlyphBounds { false };
97 bool m_enableKerning { false };
98 bool m_requiresShaping { false };
99 bool m_forTextEmphasis { false };
100};
