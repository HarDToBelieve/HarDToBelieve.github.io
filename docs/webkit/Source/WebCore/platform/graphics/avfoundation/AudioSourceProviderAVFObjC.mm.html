86void AudioSourceProviderAVFObjC::provideInput(AudioBus* bus, size_t framesToProcess)
87{
88 // Protect access to m_ringBuffer by using tryLock(). If we failed
89 // to aquire, a re-configure is underway, and m_ringBuffer is unsafe to access.
90 // Emit silence.
91 if (!m_tapStorage) {
92 bus->zero();
93 return;
94 }
95
96 if (!m_tapStorage->lock.tryLock()) {
97 bus->zero();
98 return;
99 }
100 Locker locker { AdoptLock, m_tapStorage->lock };
101
102 if (!m_ringBuffer) {
103 bus->zero();
104 return;
105 }
106
107 uint64_t startFrame = 0;
108 uint64_t endFrame = 0;
109 uint64_t seekTo = std::exchange(m_seekTo, NoSeek);
110 if (seekTo != NoSeek)
111 m_readCount = seekTo;
112
113 m_ringBuffer->getCurrentFrameBounds(startFrame, endFrame);
114
115 if (!m_readCount || m_readCount == seekTo) {
116 // We have not started rendering yet. If there aren't enough frames in the buffer, then output
117 // silence until there is.
118 if (endFrame <= m_readCount + m_writeAheadCount + framesToProcess) {
119 bus->zero();
120 return;
121 }
122 } else {
123 // We've started rendering. Don't output silence unless we really have to.
124 size_t framesAvailable = static_cast<size_t>(endFrame - m_readCount);
125 if (framesAvailable < framesToProcess) {
126 bus->zero();
127 if (!framesAvailable)
128 return;
129 framesToProcess = framesAvailable;
130 }
131 }
132
133 ASSERT(bus->numberOfChannels() == m_ringBuffer->channelCount());
134
135 for (unsigned i = 0; i < m_list->mNumberBuffers; ++i) {
136 AudioChannel* channel = bus->channel(i);
137 m_list->mBuffers[i].mNumberChannels = 1;
138 m_list->mBuffers[i].mData = channel->mutableData();
139 m_list->mBuffers[i].mDataByteSize = channel->length() * sizeof(float);
140 }
141
142 m_ringBuffer->fetch(m_list.get(), framesToProcess, m_readCount);
143 m_readCount += framesToProcess;
144
145 if (m_converter)
146 PAL::AudioConverterConvertComplexBuffer(m_converter.get(), framesToProcess, m_list.get(), m_list.get());
147}

176void AudioSourceProviderAVFObjC::destroyMixIfNeeded()
177{
178 if (!m_avAudioMix)
179 return;
180 ASSERT(m_tapStorage);
181 {
182 Locker locker { m_tapStorage->lock };
183 if (m_avPlayerItem)
184 [m_avPlayerItem setAudioMix:nil];
185 [m_avAudioMix setInputParameters:@[ ]];
186 m_avAudioMix.clear();
187 m_tap.clear();
188 m_tapStorage->_this = nullptr;
189 // Call unprepare, since Tap cannot call it after clear.
190 unprepare();
191 m_weakFactory.revokeAll();
192 }
193 m_tapStorage = nullptr;
194}
195
196void AudioSourceProviderAVFObjC::createMixIfNeeded()
197{
198 if (!m_client || !m_avPlayerItem || !m_avAssetTrack)
199 return;
200
201 ASSERT(!m_avAudioMix);
202 ASSERT(!m_tapStorage);
203 ASSERT(!m_tap);
204
205 auto tapStorage = adoptRef(new TapStorage(this));
206 Locker locker { tapStorage->lock };
207
208 MTAudioProcessingTapCallbacks callbacks = {
209 0,
210 tapStorage.get(),
211 initCallback,
212 finalizeCallback,
213 prepareCallback,
214 unprepareCallback,
215 processCallback,
216 };
217
218 MTAudioProcessingTapRef tap = nullptr;
219 OSStatus status = PAL::MTAudioProcessingTapCreate(kCFAllocatorDefault, &callbacks, 1, &tap);
220 if (status != noErr) {
221 if (tap)
222 CFRelease(tap);
223 return;
224 }
225 m_tap = adoptCF(tap);
226 m_tapStorage = WTFMove(tapStorage);
227 m_avAudioMix = adoptNS([PAL::allocAVMutableAudioMixInstance() init]);
228
229 RetainPtr<AVMutableAudioMixInputParameters> parameters = adoptNS([PAL::allocAVMutableAudioMixInputParametersInstance() init]);
230 [parameters setAudioTapProcessor:m_tap.get()];
231
232 CMPersistentTrackID trackID = m_avAssetTrack.get().trackID;
233 [parameters setTrackID:trackID];
234
235 [m_avAudioMix setInputParameters:@[parameters.get()]];
236 [m_avPlayerItem setAudioMix:m_avAudioMix.get()];
237 m_weakFactory.initializeIfNeeded(*this);
238}

289void AudioSourceProviderAVFObjC::prepare(CMItemCount maxFrames, const AudioStreamBasicDescription *processingFormat)
290{
291 ASSERT(maxFrames >= 0);
292
293 m_tapDescription = makeUniqueWithoutFastMallocCheck<AudioStreamBasicDescription>(*processingFormat);
294 int numberOfChannels = processingFormat->mChannelsPerFrame;
295 double sampleRate = processingFormat->mSampleRate;
296 ASSERT(sampleRate >= 0);
297
298 m_outputDescription = makeUniqueWithoutFastMallocCheck<AudioStreamBasicDescription>();
299 m_outputDescription->mSampleRate = sampleRate;
300 m_outputDescription->mFormatID = kAudioFormatLinearPCM;
301 m_outputDescription->mFormatFlags = kAudioFormatFlagsNativeFloatPacked;
302 m_outputDescription->mBitsPerChannel = 8 * sizeof(Float32);
303 m_outputDescription->mChannelsPerFrame = numberOfChannels;
304 m_outputDescription->mFramesPerPacket = 1;
305 m_outputDescription->mBytesPerPacket = sizeof(Float32);
306 m_outputDescription->mBytesPerFrame = sizeof(Float32);
307 m_outputDescription->mFormatFlags |= kAudioFormatFlagIsNonInterleaved;
308
309 if (*m_tapDescription != *m_outputDescription) {
310 AudioConverterRef outConverter = nullptr;
311 PAL::AudioConverterNew(m_tapDescription.get(), m_outputDescription.get(), &outConverter);
312 m_converter = outConverter;
313 }
314
315 // Make the ringbuffer large enough to store at least two callbacks worth of audio, or 1s, whichever is larger.
316 size_t capacity = std::max(static_cast<size_t>(2 * maxFrames), static_cast<size_t>(kRingBufferDuration * sampleRate));
317
318 CAAudioStreamDescription description { *processingFormat };
319 if (!m_ringBuffer)
320 m_ringBuffer = m_ringBufferCreationCallback().moveToUniquePtr();
321 m_ringBuffer->allocate(description, capacity);
322
323 // AudioBufferList is a variable-length struct, so create on the heap with a generic new() operator
324 // with a custom size, and initialize the struct manually.
325 size_t bufferListSize = sizeof(AudioBufferList) + (sizeof(AudioBuffer) * std::max(1, numberOfChannels - 1));
326 m_list = std::unique_ptr<AudioBufferList>((AudioBufferList*) ::operator new (bufferListSize));
327 memset(m_list.get(), 0, bufferListSize);
328 m_list->mNumberBuffers = numberOfChannels;
329
330 callOnMainThread([weakThis = m_weakFactory.createWeakPtr(*this), numberOfChannels, sampleRate] {
331 auto* self = weakThis.get();
332 if (self && self->m_client)
333 self->m_client->setFormat(numberOfChannels, sampleRate);
334 });
335}

341 m_ringBuffer = nullptr;
342 m_list = nullptr;

345void AudioSourceProviderAVFObjC::process(MTAudioProcessingTapRef tap, CMItemCount numberOfFrames, MTAudioProcessingTapFlags flags, AudioBufferList* bufferListInOut, CMItemCount* numberFramesOut, MTAudioProcessingTapFlags* flagsOut)
346{
347 UNUSED_PARAM(flags);
348
349 CMItemCount itemCount = 0;
350 CMTimeRange rangeOut;
351 OSStatus status = PAL::MTAudioProcessingTapGetSourceAudio(tap, numberOfFrames, bufferListInOut, flagsOut, &rangeOut, &itemCount);
352 if (status != noErr || !itemCount)
353 return;
354
355 MediaTime rangeStart = PAL::toMediaTime(rangeOut.start);
356 MediaTime rangeDuration = PAL::toMediaTime(rangeOut.duration);
357
358 if (rangeStart.isInvalid())
359 return;
360
361 MediaTime currentTime = PAL::toMediaTime(PAL::CMTimebaseGetTime([m_avPlayerItem timebase]));
362 if (currentTime.isInvalid())
363 return;
364
365 // The audio tap will generate silence when the media is paused, and will not advance the
366 // tap currentTime.
367 if (rangeStart == m_startTimeAtLastProcess || rangeDuration == MediaTime::zeroTime()) {
368 m_paused = true;
369 return;
370 }
371
372 if (m_paused) {
373 // Only check the write-ahead time when playback begins.
374 m_paused = false;
375 MediaTime earlyBy = rangeStart - currentTime;
376 m_writeAheadCount = m_tapDescription->mSampleRate * earlyBy.toDouble();
377 }
378
379 uint64_t startFrame = 0;
380 uint64_t endFrame = 0;
381 m_ringBuffer->getCurrentFrameBounds(startFrame, endFrame);
382
383 // Check to see if the underlying media has seeked, which would require us to "flush"
384 // our outstanding buffers.
385 if (rangeStart != m_endTimeAtLastProcess)
386 m_seekTo = endFrame;
387
388 m_startTimeAtLastProcess = rangeStart;
389 m_endTimeAtLastProcess = rangeStart + rangeDuration;
390
391 // StartOfStream indicates a discontinuity, such as when an AVPlayerItem is re-added
392 // to an AVPlayer, so "flush" outstanding buffers.
393 if (flagsOut && *flagsOut & kMTAudioProcessingTapFlag_StartOfStream)
394 m_seekTo = endFrame;
395
396 m_ringBuffer->store(bufferListInOut, itemCount, endFrame);
397
398 // Mute the default audio playback by zeroing the tap-owned buffers.
399 for (uint32_t i = 0; i < bufferListInOut->mNumberBuffers; ++i) {
400 AudioBuffer& buffer = bufferListInOut->mBuffers[i];
401 memset(buffer.mData, 0, buffer.mDataByteSize);
402 }
403 *numberFramesOut = 0;
404
405 if (m_audioCallback)
406 m_audioCallback(endFrame, itemCount);
407}
