89static Vector<Ref<SharedBuffer>> extractSinfData(const SharedBuffer& buffer)
90{
91 // JSON of the format: "{ sinf: [ <base64-encoded-string> ] }"
92 if (buffer.size() > std::numeric_limits<unsigned>::max())
93 return { };
94 String json { buffer.data(), static_cast<unsigned>(buffer.size()) };
95
96 auto value = JSON::Value::parseJSON(json);
97 if (!value)
98 return { };
99
100 auto object = value->asObject();
101 if (!object)
102 return { };
103
104 auto sinfArray = object->getArray(CDMPrivateFairPlayStreaming::sinfName());
105 if (!sinfArray)
106 return { };
107
108 Vector<Ref<SharedBuffer>> sinfs;
109 sinfs.reserveInitialCapacity(sinfArray->length());
110
111 for (auto& value : *sinfArray) {
112 auto keyID = value->asString();
113 if (!keyID)
114 continue;
115
116 auto sinfData = base64Decode(keyID, Base64DecodeOptions::IgnoreSpacesAndNewLines);
117 if (!sinfData)
118 continue;
119
120 sinfs.uncheckedAppend(SharedBuffer::create(WTFMove(*sinfData)));
121 }
122
123 return sinfs;
124}

126using SchemeAndKeyResult = Vector<std::pair<FourCC, Vector<uint8_t>>>;
127static SchemeAndKeyResult extractSchemeAndKeyIdFromSinf(const SharedBuffer& buffer)
128{
129 auto buffers = extractSinfData(buffer);
130 if (!buffers.size())
131 return { };
132
133 SchemeAndKeyResult result;
134 for (auto& buffer : buffers) {
135 unsigned offset = 0;
136 std::optional<FourCC> scheme;
137 std::optional<Vector<uint8_t>> keyID;
138
139 auto view = JSC::DataView::create(buffer->tryCreateArrayBuffer(), offset, buffer->size());
140 while (auto optionalBoxType = ISOBox::peekBox(view, offset)) {
141 auto& boxTypeName = optionalBoxType.value().first;
142 auto& boxSize = optionalBoxType.value().second;
143
144 if (boxTypeName == ISOSchemeTypeBox::boxTypeName()) {
145 ISOSchemeTypeBox schemeTypeBox;
146 if (!schemeTypeBox.read(view, offset))
147 break;
148
149 scheme = schemeTypeBox.schemeType();
150 continue;
151 }
152
153 if (boxTypeName == ISOSchemeInformationBox::boxTypeName()) {
154 ISOSchemeInformationBox schemeInfoBox;
155 if (!schemeInfoBox.read(view, offset))
156 break;
157
158 auto trackEncryptionBox = downcast<ISOTrackEncryptionBox>(schemeInfoBox.schemeSpecificData());
159 if (trackEncryptionBox)
160 keyID = trackEncryptionBox->defaultKID();
161 continue;
162 }
163
164 offset += boxSize;
165 }
166 if (scheme && keyID)
167 result.append(std::make_pair(scheme.value(), WTFMove(keyID.value())));
168 }
169
170 return result;
171}

173std::optional<Vector<Ref<SharedBuffer>>> CDMPrivateFairPlayStreaming::extractKeyIDsSinf(const SharedBuffer& buffer)
174{
175 Vector<Ref<SharedBuffer>> keyIDs;
176 auto results = extractSchemeAndKeyIdFromSinf(buffer);
177
178 for (auto& result : results) {
179 if (validFairPlayStreamingSchemes().contains(result.first))
180 keyIDs.append(SharedBuffer::create(WTFMove(result.second)));
181 }
182
183 return keyIDs;
184}

251{
252 // https://w3c.github.io/encrypted-media/#key-system

273bool CDMPrivateFairPlayStreaming::supportsConfiguration(const CDMKeySystemConfiguration& configuration) const
274{
275 if (!WTF::anyOf(configuration.initDataTypes, [] (auto& initDataType) { return validInitDataTypes().contains(initDataType); })) {
276 DEBUG_LOG_IF_POSSIBLE(LOGIDENTIFIER, " false, no initDataType supported");
277 return false;
278 }
279
280#if HAVE(AVCONTENTKEYSESSION)
281 // FIXME: verify that FairPlayStreaming does not (and cannot) expose a distinctive identifier to the client
282 if (configuration.distinctiveIdentifier == CDMRequirement::Required) {
283 DEBUG_LOG_IF_POSSIBLE(LOGIDENTIFIER, "false, requried distinctiveIdentifier not supported");
284 return false;
285 }
286
287 if (configuration.persistentState == CDMRequirement::Required && !CDMInstanceFairPlayStreamingAVFObjC::supportsPersistableState()) {
288 DEBUG_LOG_IF_POSSIBLE(LOGIDENTIFIER, "false, required persistentState not supported");
289 return false;
290 }
291
292 if (configuration.sessionTypes.contains(CDMSessionType::PersistentLicense)
293 && !configuration.sessionTypes.contains(CDMSessionType::Temporary)
294 && !CDMInstanceFairPlayStreamingAVFObjC::supportsPersistentKeys()) {
295 DEBUG_LOG_IF_POSSIBLE(LOGIDENTIFIER, "false, sessionType PersistentLicense not supported");
296 return false;
297 }
298
299 if (!configuration.audioCapabilities.isEmpty()
300 && !WTF::anyOf(configuration.audioCapabilities, [](auto& capability) {
301 return CDMInstanceFairPlayStreamingAVFObjC::supportsMediaCapability(capability);
302 })) {
303 DEBUG_LOG_IF_POSSIBLE(LOGIDENTIFIER, "false, no audio configuration supported");
304 return false;
305 }
306
307 if (!configuration.videoCapabilities.isEmpty()
308 && !WTF::anyOf(configuration.videoCapabilities, [](auto& capability) {
309 return CDMInstanceFairPlayStreamingAVFObjC::supportsMediaCapability(capability);
310 })) {
311 DEBUG_LOG_IF_POSSIBLE(LOGIDENTIFIER, "false, no video configuration supported");
312 return false;
313 }
314
315 DEBUG_LOG_IF_POSSIBLE(LOGIDENTIFIER, "true, supported");
316 return true;
317#else
318 return false;
319#endif
320}

322bool CDMPrivateFairPlayStreaming::supportsConfigurationWithRestrictions(const CDMKeySystemConfiguration& configuration, const CDMRestrictions& restrictions) const
323{
324 if (restrictions.persistentStateDenied
325 && !configuration.sessionTypes.isEmpty()
326 && !configuration.sessionTypes.contains(CDMSessionType::Temporary))
327 return false;
328
329 if (restrictions.persistentStateDenied && configuration.persistentState == CDMRequirement::Required)
330 return false;
331
332 if (WTF::allOf(configuration.sessionTypes, [restrictions](auto& sessionType) {
333 return restrictions.deniedSessionTypes.contains(sessionType);
334 }))
335 return false;
336
337 return supportsConfiguration(configuration);
338}

340bool CDMPrivateFairPlayStreaming::supportsSessionTypeWithConfiguration(const CDMSessionType& sessionType, const CDMKeySystemConfiguration& configuration) const
341{
342 if (sessionType == CDMSessionType::Temporary) {
343 if (configuration.persistentState == CDMRequirement::Required)
344 return false;
345 } else if (configuration.persistentState == CDMRequirement::NotAllowed)
346 return false;
347
348 return supportsConfiguration(configuration);
349}

396bool CDMPrivateFairPlayStreaming::supportsInitData(const AtomString& initDataType, const SharedBuffer& initData) const
397{
398 if (!validInitDataTypes().contains(initDataType))
399 return false;
400
401 if (initDataType == sinfName()) {
402 return WTF::anyOf(extractSchemeAndKeyIdFromSinf(initData), [](auto& result) {
403 return validFairPlayStreamingSchemes().contains(result.first);
404 });
405 }
406
407#if HAVE(FAIRPLAYSTREAMING_CENC_INITDATA)
408 if (initDataType == InitDataRegistry::cencName()) {
409 auto psshBoxes = InitDataRegistry::extractPsshBoxesFromCenc(initData);
410 if (!psshBoxes)
411 return false;
412
413 return WTF::anyOf(psshBoxes.value(), [](auto& psshBox) {
414 return is<ISOFairPlayStreamingPsshBox>(*psshBox);
415 });
416 }
417#endif
418
419 if (initDataType == skdName())
420 return true;
421
422 ASSERT_NOT_REACHED();
423 return false;
424}
