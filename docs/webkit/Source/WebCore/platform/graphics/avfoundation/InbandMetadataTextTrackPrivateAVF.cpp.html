56
57void InbandMetadataTextTrackPrivateAVF::addDataCue(const MediaTime& start, const MediaTime& end, Ref<SerializedPlatformDataCue>&& cueData, const String& type)
58{
59 ASSERT(cueFormat() == CueFormat::Data);
60 ASSERT(start >= MediaTime::zeroTime());
61
62 if (!client())
63 return;
64
65 m_currentCueStartTime = start;
66 if (end.isPositiveInfinite())
67 m_incompleteCues.append(IncompleteMetaDataCue { cueData.ptr(), start });
68 client()->addDataCue(start, end, WTFMove(cueData), type);
69}
70
71void InbandMetadataTextTrackPrivateAVF::updatePendingCueEndTimes(const MediaTime& time)
72{
73 ASSERT(time >= MediaTime::zeroTime());
74
75 if (time >= m_currentCueStartTime) {
76 if (client()) {
77 for (auto& partialCue : m_incompleteCues) {
78 INFO_LOG(LOGIDENTIFIER, "updating cue: start = ", partialCue.startTime, ", end = ", time);
79 client()->updateDataCue(partialCue.startTime, time, *partialCue.cueData);
80 }
81 }
82 } else
83 WARNING_LOG(LOGIDENTIFIER, "negative length cue(s) ignored: start = ", m_currentCueStartTime, ", end = ", time);
84
85 m_incompleteCues.shrink(0);
86 m_currentCueStartTime = MediaTime::zeroTime();
87}
88
89#endif
90
91void InbandMetadataTextTrackPrivateAVF::flushPartialCues()
92{
93 if (m_currentCueStartTime && m_incompleteCues.size())
94 INFO_LOG(LOGIDENTIFIER, "flushing incomplete data for cues: start = ", m_currentCueStartTime);
95
96 if (client()) {
97 for (auto& partialCue : m_incompleteCues)
98 client()->removeDataCue(partialCue.startTime, MediaTime::positiveInfiniteTime(), *partialCue.cueData);
99 }
100
101 m_incompleteCues.shrink(0);
102 m_currentCueStartTime = MediaTime::zeroTime();
103}
