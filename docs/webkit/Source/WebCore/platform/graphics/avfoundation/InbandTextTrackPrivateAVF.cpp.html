67static std::optional<SRGBA<uint8_t>> makeSimpleColorFromARGBCFArray(CFArrayRef colorArray)
68{
69 if (CFArrayGetCount(colorArray) < 4)
70 return std::nullopt;
71
72 float componentArray[4];
73 for (int i = 0; i < 4; i++) {
74 CFNumberRef value = static_cast<CFNumberRef>(CFArrayGetValueAtIndex(colorArray, i));
75 if (CFGetTypeID(value) != CFNumberGetTypeID())
76 return std::nullopt;
77
78 float component;
79 CFNumberGetValue(value, kCFNumberFloatType, &component);
80 componentArray[i] = component;
81 }
82
83 return convertColor<SRGBA<uint8_t>>(SRGBA<float> { componentArray[1], componentArray[2], componentArray[3], componentArray[0] });
84}

86Ref<InbandGenericCue> InbandTextTrackPrivateAVF::processCueAttributes(CFAttributedStringRef attributedString)
87{
88 using namespace PAL;
89 // Some of the attributes we translate into per-cue WebVTT settings are repeated on each part of an attributed string so only
90 // process the first instance of each.
91 enum AttributeFlags {
92 Line = 1 << 0,
93 Position = 1 << 1,
94 Size = 1 << 2,
95 Vertical = 1 << 3,
96 Align = 1 << 4,
97 FontName = 1 << 5
98 };
99 unsigned processed = 0;
100
101 auto cueData = InbandGenericCue::create();
102 StringBuilder content;
103 String attributedStringValue = CFAttributedStringGetString(attributedString);
104 CFIndex length = attributedStringValue.length();
105 if (!length)
106 return cueData;
107
108
109 CFRange effectiveRange = CFRangeMake(0, 0);
110 while ((effectiveRange.location + effectiveRange.length) < length) {
111
112 CFDictionaryRef attributes = CFAttributedStringGetAttributes(attributedString, effectiveRange.location + effectiveRange.length, &effectiveRange);
113 if (!attributes)
114 continue;
115
116 StringBuilder tagStart;
117 CFStringRef valueString;
118 String tagEnd;
119 CFIndex attributeCount = CFDictionaryGetCount(attributes);
120 Vector<const void*> keys(attributeCount);
121 Vector<const void*> values(attributeCount);
122 CFDictionaryGetKeysAndValues(attributes, keys.data(), values.data());
123
124 for (CFIndex i = 0; i < attributeCount; ++i) {
125 CFStringRef key = static_cast<CFStringRef>(keys[i]);
126 CFTypeRef value = values[i];
127 if (CFGetTypeID(key) != CFStringGetTypeID() || !CFStringGetLength(key))
128 continue;
129
130 if (CFStringCompare(key, kCMTextMarkupAttribute_Alignment, 0) == kCFCompareEqualTo) {
131 valueString = static_cast<CFStringRef>(value);
132 if (CFGetTypeID(valueString) != CFStringGetTypeID() || !CFStringGetLength(valueString))
133 continue;
134 if (processed & Align)
135 continue;
136 processed |= Align;
137
138 if (CFStringCompare(valueString, kCMTextMarkupAlignmentType_Start, 0) == kCFCompareEqualTo)
139 cueData->setAlign(GenericCueData::Alignment::Start);
140 else if (CFStringCompare(valueString, kCMTextMarkupAlignmentType_Middle, 0) == kCFCompareEqualTo)
141 cueData->setAlign(GenericCueData::Alignment::Middle);
142 else if (CFStringCompare(valueString, kCMTextMarkupAlignmentType_End, 0) == kCFCompareEqualTo)
143 cueData->setAlign(GenericCueData::Alignment::End);
144 else
145 ASSERT_NOT_REACHED();
146
147 continue;
148 }
149
150 if (CFStringCompare(key, kCMTextMarkupAttribute_BoldStyle, 0) == kCFCompareEqualTo) {
151 if (static_cast<CFBooleanRef>(value) != kCFBooleanTrue)
152 continue;
153
154 tagStart.append("<b>");
155 tagEnd = "</b>" + tagEnd;
156 continue;
157 }
158
159 if (CFStringCompare(key, kCMTextMarkupAttribute_ItalicStyle, 0) == kCFCompareEqualTo) {
160 if (static_cast<CFBooleanRef>(value) != kCFBooleanTrue)
161 continue;
162
163 tagStart.append("<i>");
164 tagEnd = "</i>" + tagEnd;
165 continue;
166 }
167
168 if (CFStringCompare(key, kCMTextMarkupAttribute_UnderlineStyle, 0) == kCFCompareEqualTo) {
169 if (static_cast<CFBooleanRef>(value) != kCFBooleanTrue)
170 continue;
171
172 tagStart.append("<u>");
173 tagEnd = "</u>" + tagEnd;
174 continue;
175 }
176
177 if (CFStringCompare(key, kCMTextMarkupAttribute_OrthogonalLinePositionPercentageRelativeToWritingDirection, 0) == kCFCompareEqualTo) {
178 if (CFGetTypeID(value) != CFNumberGetTypeID())
179 continue;
180 if (processed & Line)
181 continue;
182 processed |= Line;
183
184 CFNumberRef valueNumber = static_cast<CFNumberRef>(value);
185 double line;
186 CFNumberGetValue(valueNumber, kCFNumberFloat64Type, &line);
187 cueData->setLine(line);
188 continue;
189 }
190
191 if (CFStringCompare(key, kCMTextMarkupAttribute_TextPositionPercentageRelativeToWritingDirection, 0) == kCFCompareEqualTo) {
192 if (CFGetTypeID(value) != CFNumberGetTypeID())
193 continue;
194 if (processed & Position)
195 continue;
196 processed |= Position;
197
198 CFNumberRef valueNumber = static_cast<CFNumberRef>(value);
199 double position;
200 CFNumberGetValue(valueNumber, kCFNumberFloat64Type, &position);
201 cueData->setPosition(position);
202 continue;
203 }
204
205 if (CFStringCompare(key, kCMTextMarkupAttribute_WritingDirectionSizePercentage, 0) == kCFCompareEqualTo) {
206 if (CFGetTypeID(value) != CFNumberGetTypeID())
207 continue;
208 if (processed & Size)
209 continue;
210 processed |= Size;
211
212 CFNumberRef valueNumber = static_cast<CFNumberRef>(value);
213 double size;
214 CFNumberGetValue(valueNumber, kCFNumberFloat64Type, &size);
215 cueData->setSize(size);
216 continue;
217 }
218
219 if (CFStringCompare(key, kCMTextMarkupAttribute_VerticalLayout, 0) == kCFCompareEqualTo) {
220 valueString = static_cast<CFStringRef>(value);
221 if (CFGetTypeID(valueString) != CFStringGetTypeID() || !CFStringGetLength(valueString))
222 continue;
223
224 if (CFStringCompare(valueString, kCMTextVerticalLayout_LeftToRight, 0) == kCFCompareEqualTo)
225 tagStart.append(leftToRightMark);
226 else if (CFStringCompare(valueString, kCMTextVerticalLayout_RightToLeft, 0) == kCFCompareEqualTo)
227 tagStart.append(rightToLeftMark);
228 continue;
229 }
230
231 if (CFStringCompare(key, kCMTextMarkupAttribute_BaseFontSizePercentageRelativeToVideoHeight, 0) == kCFCompareEqualTo) {
232 if (CFGetTypeID(value) != CFNumberGetTypeID())
233 continue;
234
235 CFNumberRef valueNumber = static_cast<CFNumberRef>(value);
236 double baseFontSize;
237 CFNumberGetValue(valueNumber, kCFNumberFloat64Type, &baseFontSize);
238 cueData->setBaseFontSize(baseFontSize);
239 continue;
240 }
241
242 if (CFStringCompare(key, kCMTextMarkupAttribute_RelativeFontSize, 0) == kCFCompareEqualTo) {
243 if (CFGetTypeID(value) != CFNumberGetTypeID())
244 continue;
245
246 CFNumberRef valueNumber = static_cast<CFNumberRef>(value);
247 double relativeFontSize;
248 CFNumberGetValue(valueNumber, kCFNumberFloat64Type, &relativeFontSize);
249 cueData->setRelativeFontSize(relativeFontSize);
250 continue;
251 }
252
253 if (CFStringCompare(key, kCMTextMarkupAttribute_FontFamilyName, 0) == kCFCompareEqualTo) {
254 valueString = static_cast<CFStringRef>(value);
255 if (CFGetTypeID(valueString) != CFStringGetTypeID() || !CFStringGetLength(valueString))
256 continue;
257 if (processed & FontName)
258 continue;
259 processed |= FontName;
260
261 cueData->setFontName(valueString);
262 continue;
263 }
264
265 if (CFStringCompare(key, kCMTextMarkupAttribute_ForegroundColorARGB, 0) == kCFCompareEqualTo) {
266 CFArrayRef arrayValue = static_cast<CFArrayRef>(value);
267 if (CFGetTypeID(arrayValue) != CFArrayGetTypeID())
268 continue;
269
270 auto color = makeSimpleColorFromARGBCFArray(arrayValue);
271 if (!color)
272 continue;
273 cueData->setForegroundColor(*color);
274 continue;
275 }
276
277 if (CFStringCompare(key, kCMTextMarkupAttribute_BackgroundColorARGB, 0) == kCFCompareEqualTo) {
278 CFArrayRef arrayValue = static_cast<CFArrayRef>(value);
279 if (CFGetTypeID(arrayValue) != CFArrayGetTypeID())
280 continue;
281
282 auto color = makeSimpleColorFromARGBCFArray(arrayValue);
283 if (!color)
284 continue;
285 cueData->setBackgroundColor(*color);
286 continue;
287 }
288
289 if (CFStringCompare(key, kCMTextMarkupAttribute_CharacterBackgroundColorARGB, 0) == kCFCompareEqualTo) {
290 CFArrayRef arrayValue = static_cast<CFArrayRef>(value);
291 if (CFGetTypeID(arrayValue) != CFArrayGetTypeID())
292 continue;
293
294 auto color = makeSimpleColorFromARGBCFArray(arrayValue);
295 if (!color)
296 continue;
297 cueData->setHighlightColor(*color);
298 continue;
299 }
300 }
301
302 content.append(tagStart);
303 content.append(attributedStringValue.substring(effectiveRange.location, effectiveRange.length));
304 content.append(tagEnd);
305 }
306
307 if (content.length())
308 cueData->setContent(content.toString());
309
310 return cueData;
311}

322void InbandTextTrackPrivateAVF::processAttributedStrings(CFArrayRef attributedStrings, const MediaTime& time)
323{
324 CFIndex count = attributedStrings ? CFArrayGetCount(attributedStrings) : 0;
325
326 if (count)
327 INFO_LOG(LOGIDENTIFIER, "added ", count, count > 1 ? " cues" : " cue", " at time ", time);
328
329 Vector<Ref<InbandGenericCue>> arrivingCues;
330 if (count) {
331 for (CFIndex i = 0; i < count; i++) {
332 CFAttributedStringRef attributedString = static_cast<CFAttributedStringRef>(CFArrayGetValueAtIndex(attributedStrings, i));
333
334 if (!attributedString || !CFAttributedStringGetLength(attributedString))
335 continue;
336
337 auto cueData = processCueAttributes(attributedString);
338 if (!cueData->content().length())
339 continue;
340
341 cueData->setStartTime(time);
342 cueData->setEndTime(MediaTime::positiveInfiniteTime());
343
344 // AVFoundation cue "position" is to the center of the text so adjust relative to the edge because we will use it to
345 // set CSS "left".
346 if (cueData->position() >= 0 && cueData->size() > 0)
347 cueData->setPosition(cueData->position() - cueData->size() / 2);
348
349 cueData->setStatus(GenericCueData::Status::Partial);
350
351 arrivingCues.append(WTFMove(cueData));
352 }
353 }
354
355 if (m_pendingCueStatus != None) {
356 // Cues do not have an explicit duration, they are displayed until the next "cue" (which might be empty) is emitted.
357 m_currentCueEndTime = time;
358
359 if (m_currentCueEndTime >= m_currentCueStartTime) {
360 for (auto& cueData : m_cues) {
361 // See if one of the newly-arrived cues is an extension of this cue.
362 Vector<Ref<InbandGenericCue>> nonExtensionCues;
363 for (auto& arrivingCue : arrivingCues) {
364 if (!arrivingCue->doesExtendCueData(cueData))
365 nonExtensionCues.append(WTFMove(arrivingCue));
366 else
367 INFO_LOG(LOGIDENTIFIER, "found an extension cue ", cueData.get());
368 }
369
370 bool currentCueIsExtended = (arrivingCues.size() != nonExtensionCues.size());
371
372 arrivingCues = WTFMove(nonExtensionCues);
373
374 if (currentCueIsExtended)
375 continue;
376
377 if (m_pendingCueStatus == Valid) {
378 cueData->setEndTime(m_currentCueEndTime);
379 cueData->setStatus(GenericCueData::Status::Complete);
380
381 INFO_LOG(LOGIDENTIFIER, "updating cue ", cueData.get());
382
383 client()->updateGenericCue(cueData);
384 } else {
385 // We have to assume that the implicit duration is invalid for cues delivered during a seek because the AVF decode pipeline may not
386 // see every cue, so DO NOT update cue duration while seeking.
387 INFO_LOG(LOGIDENTIFIER, "ignoring cue delivered during seek ", cueData.get());
388 }
389 }
390 } else
391 ERROR_LOG(LOGIDENTIFIER, "negative length cue(s): ", MediaTimeRange { m_currentCueStartTime, m_currentCueEndTime });
392
393 removeCompletedCues();
394 }
395
396 if (arrivingCues.isEmpty())
397 return;
398
399 m_currentCueStartTime = time;
400
401 for (auto& cueData : arrivingCues) {
402 m_cues.append(cueData.get());
403 INFO_LOG(LOGIDENTIFIER, "adding cue ", cueData.get());
404 client()->addGenericCue(cueData);
405 }
406
407 m_pendingCueStatus = seeking() ? DeliveredDuringSeek : Valid;
408}

410void InbandTextTrackPrivateAVF::beginSeeking()
411{
412 // Forget any partially accumulated cue data as the seek could be to a time outside of the cue's
413 // range, which will mean that the next cue delivered will result in the current cue getting the
414 // incorrect duration.
415 resetCueValues();
416 m_seeking = true;
417}

425void InbandTextTrackPrivateAVF::removeCompletedCues()
426{
427 if (client()) {
428 long currentCue = m_cues.size() - 1;
429 for (; currentCue >= 0; --currentCue) {
430 auto& cue = m_cues[currentCue];
431 if (cue->status() != GenericCueData::Status::Complete)
432 continue;
433
434 INFO_LOG(LOGIDENTIFIER, "removing cue ", cue.get());
435
436 m_cues.remove(currentCue);
437 }
438 }
439
440 if (m_cues.isEmpty())
441 m_pendingCueStatus = None;
442
443 m_currentCueStartTime = MediaTime::zeroTime();
444 m_currentCueEndTime = MediaTime::zeroTime();
445}
446
447void InbandTextTrackPrivateAVF::resetCueValues()
448{
449 if (m_currentCueEndTime && m_cues.size())
450 INFO_LOG(LOGIDENTIFIER, "flushing data for cues: start = ", m_currentCueStartTime);
451
452 if (auto* client = this->client()) {
453 for (auto& cue : m_cues)
454 client->removeGenericCue(cue);
455 }
456
457 m_cues.shrink(0);
458 m_pendingCueStatus = None;
459 m_currentCueStartTime = MediaTime::zeroTime();
460 m_currentCueEndTime = MediaTime::zeroTime();
461}

477void InbandTextTrackPrivateAVF::processNativeSamples(CFArrayRef nativeSamples, const MediaTime& presentationTime)
478{
479 using namespace PAL;
480
481 if (!nativeSamples)
482 return;
483
484 CFIndex count = CFArrayGetCount(nativeSamples);
485 if (!count)
486 return;
487
488 INFO_LOG(LOGIDENTIFIER, count, " sample buffers at time ", presentationTime);
489
490 for (CFIndex i = 0; i < count; i++) {
491 RefPtr<ArrayBuffer> buffer;
492 MediaTime duration;
493 CMFormatDescriptionRef formatDescription;
494 if (!readNativeSampleBuffer(nativeSamples, i, buffer, duration, formatDescription))
495 continue;
496
497 auto view = JSC::DataView::create(WTFMove(buffer), 0, buffer->byteLength());
498 auto peekResult = ISOBox::peekBox(view, 0);
499 if (!peekResult)
500 continue;
501
502 auto type = peekResult.value().first;
503 auto boxLength = peekResult.value().second;
504 if (boxLength > view->byteLength()) {
505 ERROR_LOG(LOGIDENTIFIER, "chunk type = '", type.toString(), "', size = ", (size_t)boxLength, " larger than buffer length!");
506 continue;
507 }
508
509 INFO_LOG(LOGIDENTIFIER, "chunk type = '", type.toString(), "', size = ", (size_t)boxLength);
510
511 do {
512 if (m_haveReportedVTTHeader || !formatDescription)
513 break;
514
515 CFDictionaryRef extensions = CMFormatDescriptionGetExtensions(formatDescription);
516 if (!extensions)
517 break;
518
519 CFDictionaryRef sampleDescriptionExtensions = static_cast<CFDictionaryRef>(CFDictionaryGetValue(extensions, kCMFormatDescriptionExtension_SampleDescriptionExtensionAtoms));
520 if (!sampleDescriptionExtensions)
521 break;
522
523 CFDataRef webvttHeaderData = static_cast<CFDataRef>(CFDictionaryGetValue(sampleDescriptionExtensions, CFSTR("vttC")));
524 if (!webvttHeaderData)
525 break;
526
527 unsigned length = CFDataGetLength(webvttHeaderData);
528 if (!length)
529 break;
530
531 // A WebVTT header is terminated by "One or more WebVTT line terminators" so append two line feeds to make sure the parser
532 // reccognized this string as a full header.
533 auto header = makeString(StringView { CFDataGetBytePtr(webvttHeaderData), length }, "\n\n");
534
535 INFO_LOG(LOGIDENTIFIER, "VTT header ", header);
536 client()->parseWebVTTFileHeader(WTFMove(header));
537 m_haveReportedVTTHeader = true;
538 } while (0);
539
540 if (type == ISOWebVTTCue::boxTypeName()) {
541 ISOWebVTTCue cueData = ISOWebVTTCue(presentationTime, duration);
542 cueData.read(view);
543 INFO_LOG(LOGIDENTIFIER, "VTT cue data ", cueData);
544 client()->parseWebVTTCueData(WTFMove(cueData));
545 }
546
547 m_sampleInputBuffer.remove(0, (size_t)boxLength);
548 }
549}

551bool InbandTextTrackPrivateAVF::readNativeSampleBuffer(CFArrayRef nativeSamples, CFIndex index, RefPtr<ArrayBuffer>& buffer, MediaTime& duration, CMFormatDescriptionRef& formatDescription)
552{
553 using namespace PAL;
554#if OS(WINDOWS) && HAVE(AVCFPLAYERITEM_CALLBACK_VERSION_2)
555 return false;
556#else
557 CMSampleBufferRef sampleBuffer = reinterpret_cast<CMSampleBufferRef>(const_cast<void*>(CFArrayGetValueAtIndex(nativeSamples, index)));
558 if (!sampleBuffer)
559 return false;
560
561 CMSampleTimingInfo timingInfo;
562 OSStatus status = CMSampleBufferGetSampleTimingInfo(sampleBuffer, index, &timingInfo);
563 if (status) {
564 ERROR_LOG(LOGIDENTIFIER, "CMSampleBufferGetSampleTimingInfo returned error ", status, "' for sample ", index);
565 return false;
566 }
567
568 duration = PAL::toMediaTime(timingInfo.duration);
569
570 CMBlockBufferRef blockBuffer = CMSampleBufferGetDataBuffer(sampleBuffer);
571 size_t bufferLength = CMBlockBufferGetDataLength(blockBuffer);
572 if (bufferLength < ISOBox::minimumBoxSize()) {
573 ERROR_LOG(LOGIDENTIFIER, "CMSampleBuffer size length unexpectedly small ", bufferLength);
574 return false;
575 }
576
577 m_sampleInputBuffer.grow(m_sampleInputBuffer.size() + bufferLength);
578 CMBlockBufferCopyDataBytes(blockBuffer, 0, bufferLength, m_sampleInputBuffer.data() + m_sampleInputBuffer.size() - bufferLength);
579
580 buffer = ArrayBuffer::create(m_sampleInputBuffer.data(), m_sampleInputBuffer.size());
581
582 formatDescription = CMSampleBufferGetFormatDescription(sampleBuffer);
583
584 return true;
585#endif
586}
