34typedef const struct opaqueCMFormatDescription* CMFormatDescriptionRef;

36namespace JSC {
37class ArrayBuffer;
38}

40namespace WebCore {
41
42class AVFInbandTrackParent {
43public:
44 virtual ~AVFInbandTrackParent();
45
46 virtual void trackModeChanged() = 0;
47};
48
49class InbandTextTrackPrivateAVF : public InbandTextTrackPrivate {
50public:
51 virtual ~InbandTextTrackPrivateAVF();
52
53 void setMode(InbandTextTrackPrivate::Mode) override;
54
55 int trackIndex() const override { return m_index; }
56 void setTextTrackIndex(int index) { m_index = index; }
57
58 virtual void disconnect();
59
60 bool hasBeenReported() const { return m_hasBeenReported; }
61 void setHasBeenReported(bool reported) { m_hasBeenReported = reported; }
62
63 virtual void processCue(CFArrayRef attributedStrings, CFArrayRef nativeSamples, const MediaTime&);
64 virtual void resetCueValues();
65
66 void beginSeeking();
67 void endSeeking() { m_seeking = false; }
68 bool seeking() const { return m_seeking; }
69
70 enum Category {
71 LegacyClosedCaption,
72 OutOfBand,
73 InBand
74 };
75 virtual Category textTrackCategory() const = 0;
76
77 MediaTime startTimeVariance() const override { return MediaTime(1, 4); }
78
79 virtual bool readNativeSampleBuffer(CFArrayRef nativeSamples, CFIndex, RefPtr<JSC::ArrayBuffer>&, MediaTime&, CMFormatDescriptionRef&);
80
81protected:
82 InbandTextTrackPrivateAVF(AVFInbandTrackParent*, CueFormat);
83
84 Ref<InbandGenericCue> processCueAttributes(CFAttributedStringRef);
85 void processAttributedStrings(CFArrayRef, const MediaTime&);
86 void processNativeSamples(CFArrayRef, const MediaTime&);
87 void removeCompletedCues();
88
89 Vector<uint8_t> m_sampleInputBuffer;
90
91private:
92#if !RELEASE_LOG_DISABLED
93 const char* logClassName() const final { return "InbandTextTrackPrivateAVF"; }
94#endif
95
96 MediaTime m_currentCueStartTime;
97 MediaTime m_currentCueEndTime;
98
99 Vector<Ref<InbandGenericCue>> m_cues;
100 AVFInbandTrackParent* m_owner;
101
102 enum PendingCueStatus {
103 None,
104 DeliveredDuringSeek,
105 Valid
106 };
107 PendingCueStatus m_pendingCueStatus;
108
109 int m_index;
110 bool m_hasBeenReported;
111 bool m_seeking;
112 bool m_haveReportedVTTHeader;
113};
