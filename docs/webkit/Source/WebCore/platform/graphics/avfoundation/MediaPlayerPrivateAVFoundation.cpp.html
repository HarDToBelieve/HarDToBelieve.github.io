117void MediaPlayerPrivateAVFoundation::setUpVideoRendering()
118{
119 if (!isReadyForVideoSetup())
120 return;
121
122 MediaRenderingMode currentMode = currentRenderingMode();
123 MediaRenderingMode preferredMode = preferredRenderingMode();
124
125 if (preferredMode == MediaRenderingNone)
126 preferredMode = MediaRenderingToContext;
127
128 if (currentMode == preferredMode && currentMode != MediaRenderingNone)
129 return;
130
131 if (currentMode != MediaRenderingNone)
132 tearDownVideoRendering();
133
134 switch (preferredMode) {
135 case MediaRenderingNone:
136 case MediaRenderingToContext:
137 createContextVideoRenderer();
138 break;
139
140 case MediaRenderingToLayer:
141 createVideoLayer();
142 break;
143 }
144
145 // If using a movie layer, inform the client so the compositing tree is updated.
146 if (currentMode == MediaRenderingToLayer || preferredMode == MediaRenderingToLayer)
147 m_player->renderingModeChanged();
148}

165void MediaPlayerPrivateAVFoundation::load(const String& url)
166{
167 ALWAYS_LOG(LOGIDENTIFIER);
168
169 setNetworkState(m_preload == MediaPlayer::Preload::None ? MediaPlayer::NetworkState::Idle : MediaPlayer::NetworkState::Loading);
170 setReadyState(MediaPlayer::ReadyState::HaveNothing);
171
172 m_assetURL = URL({ }, url);
173 m_requestedOrigin = SecurityOrigin::create(m_assetURL);
174
175 // Don't do any more work if the url is empty.
176 if (!url.length())
177 return;
178
179 setPreload(m_preload);
180}

190void MediaPlayerPrivateAVFoundation::playabilityKnown()
191{
192 INFO_LOG(LOGIDENTIFIER, "metadata loaded = ", assetStatus() > MediaPlayerAVAssetStatusLoading);
193
194 if (m_assetIsPlayable)
195 return;
196
197 // Nothing more to do if we already have all of the item's metadata.
198 if (assetStatus() > MediaPlayerAVAssetStatusLoading)
199 return;
200
201 // At this point we are supposed to load metadata. It is OK to ask the asset to load the same
202 // information multiple times, because if it has already been loaded the completion handler
203 // will just be called synchronously.
204 m_loadingMetadata = true;
205 beginLoadingMetadata();

245void MediaPlayerPrivateAVFoundation::seekWithTolerance(const MediaTime& mediaTime, const MediaTime& negativeTolerance, const MediaTime& positiveTolerance)
246{
247 MediaTime time = mediaTime;
248
249 if (m_seeking) {
250 ALWAYS_LOG(LOGIDENTIFIER, "saving pending seek");
251 m_pendingSeek = [this, time, negativeTolerance, positiveTolerance]() {
252 seekWithTolerance(time, negativeTolerance, positiveTolerance);
253 };
254 return;
255 }
256 m_seeking = true;
257
258 if (!metaDataAvailable())
259 return;
260
261 if (time > durationMediaTime())
262 time = durationMediaTime();
263
264 if (currentTextTrack())
265 currentTextTrack()->beginSeeking();
266
267 ALWAYS_LOG(LOGIDENTIFIER, "seeking to ", time);
268
269 seekToTime(time, negativeTolerance, positiveTolerance);
270}

474void MediaPlayerPrivateAVFoundation::updateStates()
475{
476 if (m_ignoreLoadStateChanges)
477 return;
478
479 ScriptDisallowedScope::InMainThread scriptDisallowedScope;
480
481 MediaPlayer::NetworkState newNetworkState = m_networkState;
482 MediaPlayer::ReadyState newReadyState = m_readyState;
483
484 if (m_loadingMetadata)
485 newNetworkState = MediaPlayer::NetworkState::Loading;
486 else {
487 // -loadValuesAsynchronouslyForKeys:completionHandler: has invoked its handler; test status of keys and determine state.
488 AssetStatus assetStatus = this->assetStatus();
489 ItemStatus itemStatus = playerItemStatus();
490
491 m_assetIsPlayable = (assetStatus == MediaPlayerAVAssetStatusPlayable);
492 if (m_readyState < MediaPlayer::ReadyState::HaveMetadata && assetStatus > MediaPlayerAVAssetStatusLoading) {
493 if (m_assetIsPlayable) {
494 if (assetStatus >= MediaPlayerAVAssetStatusLoaded)
495 newReadyState = MediaPlayer::ReadyState::HaveMetadata;
496 if (itemStatus <= MediaPlayerAVPlayerItemStatusUnknown) {
497 if (assetStatus == MediaPlayerAVAssetStatusFailed || m_preload > MediaPlayer::Preload::MetaData || isLiveStream()) {
498 // The asset is playable but doesn't support inspection prior to playback (eg. streaming files),
499 // or we are supposed to prepare for playback immediately, so create the player item now.
500 newNetworkState = MediaPlayer::NetworkState::Loading;
501 prepareToPlay();
502 } else
503 newNetworkState = MediaPlayer::NetworkState::Idle;
504 }
505 } else {
506 // FIX ME: fetch the error associated with the @"playable" key to distinguish between format
507 // and network errors.
508 newNetworkState = MediaPlayer::NetworkState::FormatError;
509 }
510 }
511
512 if (assetStatus >= MediaPlayerAVAssetStatusLoaded && itemStatus > MediaPlayerAVPlayerItemStatusUnknown) {
513 switch (itemStatus) {
514 case MediaPlayerAVPlayerItemStatusDoesNotExist:
515 case MediaPlayerAVPlayerItemStatusUnknown:
516 case MediaPlayerAVPlayerItemStatusFailed:
517 break;
518
519 case MediaPlayerAVPlayerItemStatusPlaybackLikelyToKeepUp:
520 case MediaPlayerAVPlayerItemStatusPlaybackBufferFull:
521 // If the status becomes PlaybackBufferFull, loading stops and the status will not
522 // progress to LikelyToKeepUp. Set the readyState to HAVE_ENOUGH_DATA, on the
523 // presumption that if the playback buffer is full, playback will probably not stall.
524 newReadyState = MediaPlayer::ReadyState::HaveEnoughData;
525 break;
526
527 case MediaPlayerAVPlayerItemStatusReadyToPlay:
528 if (m_readyState != MediaPlayer::ReadyState::HaveEnoughData && maxTimeLoaded() > currentMediaTime())
529 newReadyState = MediaPlayer::ReadyState::HaveFutureData;
530 break;
531
532 case MediaPlayerAVPlayerItemStatusPlaybackBufferEmpty:
533 newReadyState = MediaPlayer::ReadyState::HaveCurrentData;
534 break;
535 }
536
537 if (itemStatus == MediaPlayerAVPlayerItemStatusPlaybackBufferFull)
538 newNetworkState = MediaPlayer::NetworkState::Idle;
539 else if (itemStatus == MediaPlayerAVPlayerItemStatusFailed)
540 newNetworkState = MediaPlayer::NetworkState::DecodeError;
541 else if (itemStatus != MediaPlayerAVPlayerItemStatusPlaybackBufferFull && itemStatus >= MediaPlayerAVPlayerItemStatusReadyToPlay)
542 newNetworkState = (maxTimeLoaded() == durationMediaTime()) ? MediaPlayer::NetworkState::Loaded : MediaPlayer::NetworkState::Loading;
543 }
544 }
545
546 if (isReadyForVideoSetup() && currentRenderingMode() != preferredRenderingMode())
547 setUpVideoRendering();
548
549 if (!m_haveReportedFirstVideoFrame && m_cachedHasVideo && hasAvailableVideoFrame()) {
550 if (m_readyState < MediaPlayer::ReadyState::HaveCurrentData)
551 newReadyState = MediaPlayer::ReadyState::HaveCurrentData;
552 m_haveReportedFirstVideoFrame = true;
553 m_player->firstVideoFrameAvailable();
554 } else if (!hasAvailableVideoFrame())
555 m_haveReportedFirstVideoFrame = false;
556
557 if (m_networkState != newNetworkState)
558 ALWAYS_LOG(LOGIDENTIFIER, "entered with networkState ", m_networkState, ", exiting with ", newNetworkState);
559 if (m_readyState != newReadyState)
560 ALWAYS_LOG(LOGIDENTIFIER, "entered with readyState ", m_readyState, ", exiting with ", newReadyState);
561
562 setNetworkState(newNetworkState);
563 setReadyState(newReadyState);
564}

626void MediaPlayerPrivateAVFoundation::seekCompleted(bool finished)
627{
628 UNUSED_PARAM(finished);
629 ALWAYS_LOG(LOGIDENTIFIER, "finished = ", finished);
630
631 m_seeking = false;
632
633 WTF::Function<void()> pendingSeek;
634 std::swap(pendingSeek, m_pendingSeek);
635
636 if (pendingSeek) {
637 ALWAYS_LOG(LOGIDENTIFIER, "issuing pending seek");
638 pendingSeek();
639 return;
640 }
641
642 if (currentTextTrack())
643 currentTextTrack()->endSeeking();
644
645 updateStates();
646 m_player->timeChanged();
647}

649void MediaPlayerPrivateAVFoundation::didEnd()
650{
651 // Hang onto the current time and use it as duration from now on since we are definitely at
652 // the end of the movie. Do this because the initial duration is sometimes an estimate.
653 MediaTime now = currentMediaTime();
654 if (now > MediaTime::zeroTime() && !m_seeking)
655 m_cachedDuration = now;
656
657 updateStates();
658 m_player->timeChanged();
659}

661void MediaPlayerPrivateAVFoundation::invalidateCachedDuration()
662{
663 m_cachedDuration = MediaTime::invalidTime();
664
665 // For some media files, reported duration is estimated and updated as media is loaded
666 // so report duration changed when the estimate is upated.
667 MediaTime duration = this->durationMediaTime();
668 if (duration != m_reportedDuration) {
669 INFO_LOG(LOGIDENTIFIER, duration);
670 if (m_reportedDuration.isValid())
671 m_player->durationChanged();
672 m_reportedDuration = duration;
673 }

687void MediaPlayerPrivateAVFoundation::setPreload(MediaPlayer::Preload preload)
688{
689 ALWAYS_LOG(LOGIDENTIFIER, " - ", static_cast<int>(preload));
690 m_preload = preload;
691 if (m_assetURL.isEmpty())
692 return;
693
694 setDelayCallbacks(true);
695
696 if (m_preload >= MediaPlayer::Preload::MetaData && assetStatus() == MediaPlayerAVAssetStatusDoesNotExist)
697 createAVAssetForURL(m_assetURL);
698
699 // Don't force creation of the player and player item unless we already know that the asset is playable. If we aren't
700 // there yet, or if we already know it is not playable, creating them now won't help.
701 if (m_preload == MediaPlayer::Preload::Auto && m_assetIsPlayable) {
702 createAVPlayerItem();
703 createAVPlayer();
704 }
705
706 setDelayCallbacks(false);
707}

742void MediaPlayerPrivateAVFoundation::scheduleMainThreadNotification(Notification&& notification)
743{
744 {
745 Locker locker { m_queuedNotificationsLock };
746
747 // It is important to always process the properties in the order that we are notified,
748 // so always go through the queue because notifications happen on different threads.
749 m_queuedNotifications.append(WTFMove(notification));
750
751#if OS(WINDOWS)
752 bool delayDispatch = true;
753#else
754 bool delayDispatch = m_delayCallbacks || !isMainThread();
755#endif
756 if (delayDispatch && !m_mainThreadCallPending) {
757 m_mainThreadCallPending = true;
758
759 callOnMainThread([weakThis = makeWeakPtr(*this)] {
760 if (!weakThis)
761 return;
762
763 weakThis->mainThreadCallback();
764 });
765 }
766
767 if (delayDispatch)
768 return;
769 }
770
771 dispatchNotification();
772}
773
774void MediaPlayerPrivateAVFoundation::dispatchNotification()
775{
776 ASSERT(isMainThread());
777
778 Notification notification;
779 {
780 Locker locker { m_queuedNotificationsLock };
781
782 if (m_queuedNotifications.isEmpty())
783 return;
784
785 if (!m_delayCallbacks) {
786 // Only dispatch one notification callback per invocation because they can cause recursion.
787 notification = m_queuedNotifications.takeFirst();
788 }
789
790 if (!m_queuedNotifications.isEmpty() && !m_mainThreadCallPending) {
791 callOnMainThread([weakThis = makeWeakPtr(*this)] {
792 if (!weakThis)
793 return;
794
795 weakThis->mainThreadCallback();
796 });
797 }
798
799 if (!notification.isValid())
800 return;
801 }
802
803 switch (notification.type()) {
804 case Notification::ItemDidPlayToEndTime:
805 didEnd();
806 break;
807 case Notification::ItemTracksChanged:
808 tracksChanged();
809 updateStates();
810 break;
811 case Notification::ItemStatusChanged:
812 updateStates();
813 break;

878void MediaPlayerPrivateAVFoundation::configureInbandTracks()
879{
880 RefPtr<InbandTextTrackPrivateAVF> trackToEnable;
881
882#if ENABLE(AVF_CAPTIONS)
883 synchronizeTextTrackState();
884#endif
885
886 // AVFoundation can only emit cues for one track at a time, so enable the first track that is showing, or the first that
887 // is hidden if none are showing. Otherwise disable all tracks.
888 for (unsigned i = 0; i < m_textTracks.size(); ++i) {
889 RefPtr<InbandTextTrackPrivateAVF> track = m_textTracks[i];
890 if (track->mode() == InbandTextTrackPrivate::Mode::Showing) {
891 trackToEnable = track;
892 break;
893 }
894 if (track->mode() == InbandTextTrackPrivate::Mode::Hidden)
895 trackToEnable = track;
896 }
897
898 setCurrentTextTrack(trackToEnable.get());
899}

918void MediaPlayerPrivateAVFoundation::processNewAndRemovedTextTracks(const Vector<RefPtr<InbandTextTrackPrivateAVF>>& removedTextTracks)
919{
920 if (removedTextTracks.size()) {
921 for (unsigned i = 0; i < m_textTracks.size(); ) {
922 if (!removedTextTracks.contains(m_textTracks[i])) {
923 ++i;
924 continue;
925 }
926 player()->removeTextTrack(*m_textTracks[i]);
927 m_textTracks.remove(i);
928 }
929 }
930
931 unsigned trackCount = m_textTracks.size();
932 unsigned inBandCount = 0;
933 for (unsigned i = 0; i < trackCount; ++i) {
934 RefPtr<InbandTextTrackPrivateAVF> track = m_textTracks[i];
935
936#if ENABLE(AVF_CAPTIONS)
937 if (track->textTrackCategory() == InbandTextTrackPrivateAVF::OutOfBand)
938 continue;
939#endif
940
941 track->setTextTrackIndex(inBandCount);
942 ++inBandCount;
943 if (track->hasBeenReported())
944 continue;
945
946 track->setHasBeenReported(true);
947 player()->addTextTrack(*track);
948 }
949
950 if (trackCount != m_textTracks.size())
951 INFO_LOG(LOGIDENTIFIER, "found ", m_textTracks.size(), " text tracks");
952}

963bool MediaPlayerPrivateAVFoundation::extractKeyURIKeyIDAndCertificateFromInitData(Uint8Array* initData, String& keyURI, String& keyID, RefPtr<Uint8Array>& certificate)
964{
965 // initData should have the following layout:
966 // [4 bytes: keyURI length][N bytes: keyURI][4 bytes: contentID length], [N bytes: contentID], [4 bytes: certificate length][N bytes: certificate]
967 if (initData->byteLength() < 4)
968 return false;
969
970 RefPtr<ArrayBuffer> initDataBuffer = initData->unsharedBuffer();
971
972 // Use a DataView to read uint32 values from the buffer, as Uint32Array requires the reads be aligned on 4-byte boundaries.
973 auto initDataView = JSC::DataView::create(initDataBuffer.copyRef(), 0, initDataBuffer->byteLength());
974 uint32_t offset = 0;
975 bool status = true;
976
977 uint32_t keyURILength = initDataView->get<uint32_t>(offset, true, &status);
978 offset += 4;
979 if (!status || offset + keyURILength > initData->length())
980 return false;
981
982 auto keyURIArray = Uint16Array::tryCreate(initDataBuffer.copyRef(), offset, keyURILength);
983 if (!keyURIArray)
984 return false;
985
986 keyURI = String(reinterpret_cast<UChar*>(keyURIArray->data()), keyURILength / sizeof(unsigned short));
987 offset += keyURILength;
988
989 uint32_t keyIDLength = initDataView->get<uint32_t>(offset, true, &status);
990 offset += 4;
991 if (!status || offset + keyIDLength > initData->length())
992 return false;
993
994 auto keyIDArray = Uint8Array::tryCreate(initDataBuffer.copyRef(), offset, keyIDLength);
995 if (!keyIDArray)
996 return false;
997
998 keyID = String(reinterpret_cast<UChar*>(keyIDArray->data()), keyIDLength / sizeof(unsigned short));
999 offset += keyIDLength;
1000
1001 uint32_t certificateLength = initDataView->get<uint32_t>(offset, true, &status);
1002 offset += 4;
1003 if (!status || offset + certificateLength > initData->length())
1004 return false;
1005
1006 certificate = Uint8Array::tryCreate(WTFMove(initDataBuffer), offset, certificateLength);
1007 if (!certificate)
1008 return false;
1009
1010 return true;
1011}

1030bool MediaPlayerPrivateAVFoundation::isUnsupportedMIMEType(const String& type)
1031{
1032 String lowerCaseType = type.convertToASCIILowercase();
1033
1034 // AVFoundation will return non-video MIME types which it claims to support, but which we
1035 // do not support in the <video> element. Reject all non video/, audio/, and application/ types.
1036 if (!lowerCaseType.startsWith("video/") && !lowerCaseType.startsWith("audio/") && !lowerCaseType.startsWith("application/"))
1037 return true;
1038
1039 // Reject types we know AVFoundation does not support that sites commonly ask about.
1040 if (lowerCaseType == "video/webm" || lowerCaseType == "audio/webm" || lowerCaseType == "video/x-webm")
1041 return true;
1042
1043 if (lowerCaseType == "video/x-flv")
1044 return true;
1045
1046 if (lowerCaseType == "audio/ogg" || lowerCaseType == "video/ogg" || lowerCaseType == "application/ogg")
1047 return true;
1048
1049 if (lowerCaseType == "video/h264")
1050 return true;
1051
1052 return false;
1053}
