59void MediaSelectionOptionAVFObjC::setSelected(bool selected)
60{
61 if (!m_group)
62 return;
63
64 if (selected == this->selected())
65 return;
66
67 m_group->setSelectedOption(selected ? this : nullptr);
68}
69
70bool MediaSelectionOptionAVFObjC::selected() const
71{
72 if (!m_group)
73 return false;
74 return this == m_group->selectedOption();
75}
76
77int MediaSelectionOptionAVFObjC::index() const
78{
79 if (!m_group)
80 return 0;
81
82 return [[m_group->avMediaSelectionGroup() options] indexOfObject:m_mediaSelectionOption.get()];
83}

99{
100 for (auto& option : m_options.values())

104void MediaSelectionGroupAVFObjC::updateOptions(const Vector<String>& characteristics)
105{
106 RetainPtr<NSSet> newAVOptions = adoptNS([[NSSet alloc] initWithArray:[PAL::getAVMediaSelectionGroupClass() playableMediaSelectionOptionsFromArray:[m_mediaSelectionGroup options]]]);
107 RetainPtr<NSMutableSet> oldAVOptions = adoptNS([[NSMutableSet alloc] initWithCapacity:m_options.size()]);
108 for (auto& avOption : m_options.keys())
109 [oldAVOptions addObject:(__bridge AVMediaSelectionOption *)avOption];
110
111 RetainPtr<NSMutableSet> addedAVOptions = adoptNS([newAVOptions mutableCopy]);
112 [addedAVOptions minusSet:oldAVOptions.get()];
113
114 RetainPtr<NSMutableSet> removedAVOptions = adoptNS([oldAVOptions mutableCopy]);
115 [removedAVOptions minusSet:newAVOptions.get()];
116
117 for (AVMediaSelectionOption* removedAVOption in removedAVOptions.get()) {
118 if (m_selectedOption && removedAVOption == m_selectedOption->avMediaSelectionOption())
119 m_selectedOption = nullptr;
120
121 m_options.remove((__bridge CFTypeRef)removedAVOption);
122 }
123 ALLOW_DEPRECATED_DECLARATIONS_BEGIN
124 AVMediaSelectionOption* selectedOption = [m_playerItem selectedMediaOptionInMediaSelectionGroup:m_mediaSelectionGroup.get()];
125 ALLOW_DEPRECATED_DECLARATIONS_END
126 for (AVMediaSelectionOption* addedAVOption in addedAVOptions.get()) {
127 auto addedOption = MediaSelectionOptionAVFObjC::create(*this, addedAVOption);
128 if (addedAVOption == selectedOption)
129 m_selectedOption = addedOption.ptr();
130 m_options.set((__bridge CFTypeRef)addedAVOption, WTFMove(addedOption));
131 }
132
133 if (!m_shouldSelectOptionAutomatically)
134 return;
135
136 NSArray* filteredOptions = [PAL::getAVMediaSelectionGroupClass() mediaSelectionOptionsFromArray:[m_mediaSelectionGroup options]
137 filteredAndSortedAccordingToPreferredLanguages:createNSArray(userPreferredLanguages(ShouldMinimizeLanguages::No)).get()];
138
139 if (![filteredOptions count] && characteristics.isEmpty())
140 return;
141
142 // If no options match our language selection, search for matching characteristics across all the group's options
143 if (![filteredOptions count])
144 filteredOptions = [m_mediaSelectionGroup options];
145
146 NSArray* optionsWithCharacteristics = [PAL::getAVMediaSelectionGroupClass() mediaSelectionOptionsFromArray:filteredOptions withMediaCharacteristics:createNSArray(characteristics).get()];
147 if (optionsWithCharacteristics && [optionsWithCharacteristics count])
148 filteredOptions = optionsWithCharacteristics;
149
150 if (![filteredOptions count])
151 return;
152
153 AVMediaSelectionOption* preferredOption = [filteredOptions objectAtIndex:0];
154 if (m_selectedOption && m_selectedOption->avMediaSelectionOption() == preferredOption)
155 return;
156
157 ASSERT(m_options.contains((__bridge CFTypeRef)preferredOption));
158 m_selectedOption = m_options.get((__bridge CFTypeRef)preferredOption);
159 m_selectionTimer.startOneShot(0_s);
160}
161
162void MediaSelectionGroupAVFObjC::setSelectedOption(MediaSelectionOptionAVFObjC* option)
163{
164 if (m_selectedOption == option)
165 return;
166
167 m_shouldSelectOptionAutomatically = false;
168 m_selectedOption = option;
169 if (m_selectionTimer.isActive())
170 m_selectionTimer.stop();
171 m_selectionTimer.startOneShot(0_s);
172}
173
174void MediaSelectionGroupAVFObjC::selectionTimerFired()
175{
176 [m_playerItem selectMediaOption:(m_selectedOption ? m_selectedOption->avMediaSelectionOption() : nil) inMediaSelectionGroup:m_mediaSelectionGroup.get()];
177}
