60AVOutputDeviceMenuControllerTargetPicker::AVOutputDeviceMenuControllerTargetPicker(AVPlaybackTargetPicker::Client& client)
61 : AVPlaybackTargetPicker(client)

63{
64}

66AVOutputDeviceMenuControllerTargetPicker::~AVOutputDeviceMenuControllerTargetPicker()
67{
68 [m_outputDeviceMenuControllerDelegate clearCallback];
69}

71AVOutputDeviceMenuController *AVOutputDeviceMenuControllerTargetPicker::devicePicker()
72{
73 if (!getAVOutputDeviceMenuControllerClass())
74 return nullptr;
75
76 if (!m_outputDeviceMenuController) {
77 RetainPtr<AVOutputContext> context = adoptNS([PAL::allocAVOutputContextInstance() init]);
78 m_outputDeviceMenuController = adoptNS([allocAVOutputDeviceMenuControllerInstance() initWithOutputContext:context.get()]);
79
80 [m_outputDeviceMenuController.get() addObserver:m_outputDeviceMenuControllerDelegate.get() forKeyPath:externalOutputDeviceAvailableKeyName options:NSKeyValueObservingOptionNew context:nullptr];
81 [m_outputDeviceMenuController.get() addObserver:m_outputDeviceMenuControllerDelegate.get() forKeyPath:externalOutputDevicePickedKeyName options:NSKeyValueObservingOptionNew context:nullptr];
82
83 ALLOW_DEPRECATED_DECLARATIONS_BEGIN
84 if (m_outputDeviceMenuController.get().externalOutputDeviceAvailable)
85 availableDevicesDidChange();
86 ALLOW_DEPRECATED_DECLARATIONS_END
87 }
88
89 return m_outputDeviceMenuController.get();
90}

92void AVOutputDeviceMenuControllerTargetPicker::availableDevicesDidChange()

94 if (client())
95 client()->availableDevicesChanged();
96}

98void AVOutputDeviceMenuControllerTargetPicker::currentDeviceDidChange()
99{
100 if (client())
101 client()->currentDeviceChanged();
102}
103
104void AVOutputDeviceMenuControllerTargetPicker::showPlaybackTargetPicker(NSView *, const FloatRect& location, bool hasActiveRoute, bool useDarkAppearance, bool)
105{
106 if (!client() || m_showingMenu)
107 return;
108
109 m_showingMenu = true;
110
111 ALLOW_DEPRECATED_DECLARATIONS_BEGIN
112 bool targetSelected = [devicePicker() showMenuForRect:location appearanceName:(useDarkAppearance ? NSAppearanceNameVibrantDark : NSAppearanceNameVibrantLight) allowReselectionOfSelectedOutputDevice:!hasActiveRoute];
113 ALLOW_DEPRECATED_DECLARATIONS_END
114
115 if (!client())
116 return;
117
118 if (targetSelected != hasActiveRoute)
119 currentDeviceDidChange();
120 else if (!targetSelected && !hasActiveRoute)
121 client()->pickerWasDismissed();
122
123 m_showingMenu = false;
124}

136void AVOutputDeviceMenuControllerTargetPicker::invalidatePlaybackTargets()
137{
138 if (m_outputDeviceMenuController) {
139 [m_outputDeviceMenuController removeObserver:m_outputDeviceMenuControllerDelegate.get() forKeyPath:externalOutputDeviceAvailableKeyName];
140 [m_outputDeviceMenuController removeObserver:m_outputDeviceMenuControllerDelegate.get() forKeyPath:externalOutputDevicePickedKeyName];
141 m_outputDeviceMenuController = nullptr;
142 }
143 currentDeviceDidChange();
144}

160@implementation WebAVOutputDeviceMenuControllerHelper
161- (instancetype)initWithCallback:(WeakPtr<AVOutputDeviceMenuControllerTargetPicker>&&)callback
162{
163 if (!(self = [super init]))
164 return nil;
165
166 m_callback = WTFMove(callback);
167
168 return self;
169}
170
171- (void)clearCallback
172{
173 m_callback = nil;
174}
175
176- (void)observeValueForKeyPath:(id)keyPath ofObject:(id)object change:(NSDictionary *)change context:(void *)context
177{
178 UNUSED_PARAM(object);
179 UNUSED_PARAM(change);
180 UNUSED_PARAM(context);
181
182 if (!m_callback)
183 return;
184
185 if (![keyPath isEqualToString:externalOutputDeviceAvailableKeyName] && ![keyPath isEqualToString:externalOutputDevicePickedKeyName])
186 return;
187
188 callOnMainThread([self, protectedSelf = retainPtr(self), keyPath = retainPtr(keyPath)] {
189 if (!m_callback)
190 return;
191
192 if ([keyPath isEqualToString:externalOutputDeviceAvailableKeyName])
193 m_callback->availableDevicesDidChange();
194 else if ([keyPath isEqualToString:externalOutputDevicePickedKeyName])
195 m_callback->currentDeviceDidChange();
196 });
197}
