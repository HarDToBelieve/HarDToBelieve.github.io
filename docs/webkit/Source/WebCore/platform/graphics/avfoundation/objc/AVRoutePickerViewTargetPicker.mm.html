58bool AVRoutePickerViewTargetPicker::isAvailable()
59{
60 static bool available;
61 static std::once_flag flag;
62 std::call_once(flag, [] () {
63 if (!getAVRoutePickerViewClass())
64 return;
65
66 if (auto picker = adoptNS([allocAVRoutePickerViewInstance() init]))
67 available = [picker respondsToSelector:@selector(showRoutePickingControlsForOutputContext:relativeToRect:ofView:)];
68 });
69
70 return available;
71}

73AVRoutePickerViewTargetPicker::AVRoutePickerViewTargetPicker(AVPlaybackTargetPicker::Client& client)
74 : AVPlaybackTargetPicker(client)
75 , m_routePickerViewDelegate(adoptNS([[WebAVRoutePickerViewHelper alloc] initWithCallback:makeWeakPtr(*this)]))
76{
77 ASSERT(isAvailable());
78}

80AVRoutePickerViewTargetPicker::~AVRoutePickerViewTargetPicker()
81{
82 [m_routePickerViewDelegate clearCallback];

85AVOutputContext * AVRoutePickerViewTargetPicker::outputContextInternal(bool useiTunesAVOutputContext)
86{
87 if (!m_outputContext) {
88 if (useiTunesAVOutputContext)
89 m_outputContext = [PAL::getAVOutputContextClass() iTunesAudioContext];
90 else
91 m_outputContext = [PAL::getAVOutputContextClass() outputContext];
92 ASSERT(m_outputContext);
93 if (m_outputContext)
94 [[NSNotificationCenter defaultCenter] addObserver:m_routePickerViewDelegate.get() selector:@selector(notificationHandler:) name:PAL::get_AVFoundation_AVOutputContextOutputDevicesDidChangeNotification() object:m_outputContext.get()];
95 }
96
97 return m_outputContext.get();
98}

103 m_routePickerView = adoptNS([allocAVRoutePickerViewInstance() init]);

110AVRouteDetector *AVRoutePickerViewTargetPicker::routeDetector()
111{
112 if (!m_routeDetector) {
113 m_routeDetector = adoptNS([PAL::allocAVRouteDetectorInstance() init]);
114 [[NSNotificationCenter defaultCenter] addObserver:m_routePickerViewDelegate.get() selector:@selector(notificationHandler:) name:PAL::get_AVFoundation_AVRouteDetectorMultipleRoutesDetectedDidChangeNotification() object:m_routeDetector.get()];
115 if ([m_routeDetector multipleRoutesDetected])
116 availableDevicesDidChange();
117 }
118
119 return m_routeDetector.get();
120}

122void AVRoutePickerViewTargetPicker::showPlaybackTargetPicker(NSView *view, const FloatRect& rectInScreenCoordinates, bool hasActiveRoute, bool useDarkAppearance, bool useiTunesAVOutputContext)
123{
124 if (!client())
125 return;
126
127 auto *picker = devicePicker();
128 if (useDarkAppearance)
129 picker.routeListAlwaysHasDarkAppearance = YES;
130
131 m_hadActiveRoute = hasActiveRoute;
132
133 auto rectInWindowCoordinates = [view.window convertRectFromScreen:NSMakeRect(rectInScreenCoordinates.x(), rectInScreenCoordinates.y(), 1.0, 1.0)];
134 auto rectInViewCoordinates = [view convertRect:rectInWindowCoordinates fromView:view];
135 [picker showRoutePickingControlsForOutputContext:outputContextInternal(useiTunesAVOutputContext) relativeToRect:rectInViewCoordinates ofView:view];
136}

138void AVRoutePickerViewTargetPicker::startingMonitoringPlaybackTargets()
139{
140 m_ignoreNextMultipleRoutesDetectedDidChangeNotification = false;
141
142 routeDetector().routeDetectionEnabled = YES;
143}
144
145void AVRoutePickerViewTargetPicker::stopMonitoringPlaybackTargets()
146{
147 if (!m_routeDetector)
148 return;
149
150 // `-[AVRouteDetector multipleRoutesDetected]` will always return `NO` if route detection is
151 // disabled and `-[AVRouteDetector setRouteDetectionEnabled:]` will always dispatch a
152 // `AVRouteDetectorMultipleRoutesDetectedDidChange` notification, so ignore the next one in
153 // order to prevent the cached value in the WebProcess from always being `false` when the last
154 // JS `"webkitplaybacktargetavailabilitychanged"` event listener is removed.
155 m_ignoreNextMultipleRoutesDetectedDidChangeNotification = true;
156
157 [m_routeDetector setRouteDetectionEnabled:NO];
158}

170void AVRoutePickerViewTargetPicker::invalidatePlaybackTargets()
171{
172 if (m_routeDetector) {
173 [[NSNotificationCenter defaultCenter] removeObserver:m_routePickerViewDelegate.get() name:PAL::get_AVFoundation_AVRouteDetectorMultipleRoutesDetectedDidChangeNotification() object:m_routeDetector.get()];
174 [m_routeDetector setRouteDetectionEnabled:NO];
175 m_routePickerView = nullptr;
176 }
177
178 if (m_outputContext) {
179 [[NSNotificationCenter defaultCenter] removeObserver:m_routePickerViewDelegate.get() name:PAL::get_AVFoundation_AVOutputContextOutputDevicesDidChangeNotification() object:m_outputContext.get()];
180 m_outputContext = nullptr;
181 }
182
183 if (m_routePickerView) {
184 [m_routePickerView setDelegate:nil];
185 m_routePickerView = nullptr;
186 }
187 currentDeviceDidChange();
188}
189void AVRoutePickerViewTargetPicker::availableDevicesDidChange()
190{
191 if (m_ignoreNextMultipleRoutesDetectedDidChangeNotification) {
192 m_ignoreNextMultipleRoutesDetectedDidChangeNotification = false;
193 return;
194 }
195
196 if (client())
197 client()->availableDevicesChanged();
198}

200bool AVRoutePickerViewTargetPicker::hasActiveRoute() const
201{
202 if (!m_outputContext)
203 return false;
204
205 if ([m_outputContext respondsToSelector:@selector(supportsMultipleOutputDevices)] && [m_outputContext respondsToSelector:@selector(outputDevices)]&& [m_outputContext supportsMultipleOutputDevices]) {
206 for (AVOutputDevice *outputDevice in [m_outputContext outputDevices]) {
207 if (outputDevice.deviceFeatures & (AVOutputDeviceFeatureVideo | AVOutputDeviceFeatureAudio))
208 return true;
209 }
210
211 return false;
212 }
213
214 if ([m_outputContext respondsToSelector:@selector(outputDevice)]) {
215 if (auto *outputDevice = [m_outputContext outputDevice])
216 return outputDevice.deviceFeatures & (AVOutputDeviceFeatureVideo | AVOutputDeviceFeatureAudio);
217 }
218
219 return [m_outputContext deviceName];
220}
221
222void AVRoutePickerViewTargetPicker::currentDeviceDidChange()
223{
224 auto haveActiveRoute = hasActiveRoute();
225 if (!client() || m_hadActiveRoute == haveActiveRoute)
226 return;
227
228 m_hadActiveRoute = haveActiveRoute;
229 client()->currentDeviceChanged();
230}
231
232void AVRoutePickerViewTargetPicker::devicePickerWasDismissed()
233{
234 if (!client())
235 return;
236
237 client()->pickerWasDismissed();
238 currentDeviceDidChange();
239}

243@implementation WebAVRoutePickerViewHelper
244- (instancetype)initWithCallback:(WeakPtr<AVRoutePickerViewTargetPicker>&&)callback
245{
246 if (!(self = [super init]))
247 return nil;
248
249 m_callback = WTFMove(callback);
250
251 return self;
252}
253
254- (void)clearCallback
255{
256 m_callback = nil;
257}
258
259- (void)routePickerViewDidEndPresentingRoutes:(AVRoutePickerView *)routePickerView
260{
261 UNUSED_PARAM(routePickerView);
262
263 if (!m_callback)
264 return;
265
266 callOnMainThread([self, protectedSelf = retainPtr(self)] {
267 if (!m_callback)
268 return;
269
270 m_callback->devicePickerWasDismissed();
271 });
272}
273
274- (void)notificationHandler:(NSNotification *)notification
275{
276 UNUSED_PARAM(notification);
277
278 if (!m_callback)
279 return;
280
281 callOnMainThread([self, protectedSelf = retainPtr(self), notification = retainPtr(notification)] {
282 if (!m_callback)
283 return;
284
285 if ([[notification name] isEqualToString:PAL::get_AVFoundation_AVOutputContextOutputDevicesDidChangeNotification()])
286 m_callback->currentDeviceDidChange();
287 else if ([[notification name] isEqualToString:PAL::get_AVFoundation_AVRouteDetectorMultipleRoutesDetectedDidChangeNotification()])
288 m_callback->availableDevicesDidChange();
289 });
290}
