47AVStreamDataParserMIMETypeCache& AVStreamDataParserMIMETypeCache::singleton()
48{
49 static NeverDestroyed<AVStreamDataParserMIMETypeCache> cache;
50 return cache.get();
51}

53bool AVStreamDataParserMIMETypeCache::isAvailable() const
54{
55#if ENABLE(VIDEO) && USE(AVFOUNDATION)
56 if (!PAL::AVFoundationLibrary())
57 return false;
58
59 return [PAL::getAVStreamDataParserClass() respondsToSelector:@selector(audiovisualMIMETypes)];
60#else
61 return false;
62#endif
63}
64
65MediaPlayerEnums::SupportsType AVStreamDataParserMIMETypeCache::canDecodeType(const String& type)
66{
67 if (isAvailable())
68 return MIMETypeCache::canDecodeType(type);
69
70 auto& assetCache = AVAssetMIMETypeCache::singleton();
71 if (assetCache.isAvailable())
72 return assetCache.canDecodeType(type);
73
74 return MediaPlayerEnums::SupportsType::IsNotSupported;
75}
76
77HashSet<String, ASCIICaseInsensitiveHash>& AVStreamDataParserMIMETypeCache::supportedTypes()
78{
79 if (isAvailable())
80 return MIMETypeCache::supportedTypes();
81
82 auto& assetCache = AVAssetMIMETypeCache::singleton();
83 if (assetCache.isAvailable())
84 return assetCache.supportedTypes();
85
86 return MIMETypeCache::supportedTypes();
87}
88
89bool AVStreamDataParserMIMETypeCache::canDecodeExtendedType(const ContentType& type)
90{
91#if ENABLE(VIDEO) && USE(AVFOUNDATION)
92 ASSERT(isAvailable());
93
94 if ([PAL::getAVStreamDataParserClass() respondsToSelector:@selector(canParseExtendedMIMEType:)])
95 return [PAL::getAVStreamDataParserClass() canParseExtendedMIMEType:type.raw()];
96
97 // FIXME(rdar://50502771) AVStreamDataParser does not have an -canParseExtendedMIMEType: method on this system,
98 // so just replace the container type with a valid one from AVAssetMIMETypeCache and ask that cache if it
99 // can decode this type.
100 auto& assetCache = AVAssetMIMETypeCache::singleton();
101 if (!assetCache.isAvailable() || assetCache.supportedTypes().isEmpty())
102 return false;
103
104 String replacementType { type.raw() };
105 replacementType.replace(type.containerType(), *assetCache.supportedTypes().begin());
106 return assetCache.canDecodeType(replacementType) == MediaPlayerEnums::SupportsType::IsSupported;
107#endif
108
109 return false;
110}
111
112void AVStreamDataParserMIMETypeCache::initializeCache(HashSet<String, ASCIICaseInsensitiveHash>& cache)
113{
114#if ENABLE(VIDEO) && USE(AVFOUNDATION)
115 if (!isAvailable())
116 return;
117
118 for (NSString* type in [PAL::getAVStreamDataParserClass() audiovisualMIMETypes])
119 cache.add(type);
120#endif
121}
