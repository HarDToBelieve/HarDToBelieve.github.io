197static AtomString initTypeForRequest(AVContentKeyRequest* request)
198{
199 if ([request.identifier isKindOfClass:NSString.class] && [request.identifier hasPrefix:@"skd://"])
200 return CDMPrivateFairPlayStreaming::skdName();
201
202 if (![request respondsToSelector:@selector(options)]) {
203 // AVContentKeyRequest.options was added in 10.14.4; if we are running on a previous version
204 // we don't have support for 'cenc' anyway, so just assume 'sinf'.
205 return CDMPrivateFairPlayStreaming::sinfName();
206 }
207
208ALLOW_NEW_API_WITHOUT_GUARDS_BEGIN
209 auto nsInitType = (NSString*)[request.options valueForKey:InitializationDataTypeKey];
210ALLOW_NEW_API_WITHOUT_GUARDS_END
211 if (![nsInitType isKindOfClass:NSString.class]) {
212 // The only way for an initialization data to end up here without an appropriate key in
213 // the options dictionary is for that data to have been generated by the AVStreamDataParser
214 // and currently, the only initialization data supported by the parser is the 'sinf' kind.
215 return CDMPrivateFairPlayStreaming::sinfName();
216 }
217
218 return AtomString(nsInitType);
219}

222{

227 return SharedBuffer::create([request.identifier dataUsingEncoding:NSUTF8StringEncoding allowLossyConversion:YES]);
228
229 return SharedBuffer::create(request.initializationData);
230}

242AVContentKeySession* CDMInstanceFairPlayStreamingAVFObjC::contentKeySession()
243{
244 if (m_session)
245 return m_session.get();
246
247 if (!PAL::canLoad_AVFoundation_AVContentKeySystemFairPlayStreaming())
248 return nullptr;
249
250 if (!PAL::getAVContentKeyReportGroupClass())
251 return nullptr;
252
253 auto storageURL = this->storageURL();
254 if (!persistentStateAllowed() || !storageURL)
255 m_session = [PAL::getAVContentKeySessionClass() contentKeySessionWithKeySystem:AVContentKeySystemFairPlayStreaming];
256 else
257 m_session = [PAL::getAVContentKeySessionClass() contentKeySessionWithKeySystem:AVContentKeySystemFairPlayStreaming storageDirectoryAtURL:storageURL];
258
259 if (!m_session)
260 return nullptr;
261
262 if (!m_delegate)
263 m_delegate = adoptNS([[WebCoreFPSContentKeySessionDelegate alloc] initWithParent:this]);
264
265 [m_session setDelegate:m_delegate.get() queue:dispatch_get_main_queue()];
266 return m_session.get();
267}

269RetainPtr<AVContentKeyRequest> CDMInstanceFairPlayStreamingAVFObjC::takeUnexpectedKeyRequestForInitializationData(const AtomString& initDataType, SharedBuffer& initData)
270{
271 for (auto requestIter = m_unexpectedKeyRequests.begin(); requestIter != m_unexpectedKeyRequests.end(); ++requestIter) {
272 auto& request = *requestIter;
273 auto requestType = initTypeForRequest(request.get());
274 auto requestInitData = initializationDataForRequest(request.get());
275 if (initDataType != requestType || initData != requestInitData)
276 continue;
277
278 return m_unexpectedKeyRequests.take(requestIter);
279 }
280 return nullptr;
281}

320static RefPtr<JSON::Value> parseJSONValue(const SharedBuffer& buffer)
321{
322 // Fail on large buffers whose size doesn't fit into a 32-bit unsigned integer.

359void CDMInstanceFairPlayStreamingAVFObjC::initializeWithConfiguration(const CDMKeySystemConfiguration& configuration, AllowDistinctiveIdentifiers, AllowPersistentState persistentState, SuccessCallback&& callback)
360{
361 // FIXME: verify that FairPlayStreaming does not (and cannot) expose a distinctive identifier to the client
362 auto initialize = [&] () {
363 if (configuration.distinctiveIdentifier == CDMRequirement::Required)
364 return Failed;
365
366 if (configuration.persistentState != CDMRequirement::Required && (configuration.sessionTypes.contains(CDMSessionType::PersistentUsageRecord) || configuration.sessionTypes.contains(CDMSessionType::PersistentLicense)))
367 return Failed;
368
369 if (configuration.persistentState == CDMRequirement::Required && !m_storageURL)
370 return Failed;
371
372 if (configuration.sessionTypes.contains(CDMSessionType::PersistentLicense) && !supportsPersistentKeys())
373 return Failed;
374
375 if (!PAL::canLoad_AVFoundation_AVContentKeySystemFairPlayStreaming())
376 return Failed;
377
378 m_persistentStateAllowed = persistentState == AllowPersistentState::Yes;
379 return Succeeded;
380 };
381 callback(initialize());
382}

391void CDMInstanceFairPlayStreamingAVFObjC::setStorageDirectory(const String& storageDirectory)
392{
393 DEBUG_LOG_IF_POSSIBLE(LOGIDENTIFIER);
394 if (storageDirectory.isEmpty()) {
395 m_storageURL = nil;
396 return;
397 }
398
399 auto storagePath = FileSystem::pathByAppendingComponent(storageDirectory, "SecureStop.plist");
400
401 auto fileType = FileSystem::fileTypeFollowingSymlinks(storageDirectory);
402 if (!fileType) {
403 if (!FileSystem::makeAllDirectories(storageDirectory))
404 return;
405 } else if (*fileType != FileSystem::FileType::Directory) {
406 auto tempDirectory = FileSystem::createTemporaryDirectory(@"MediaKeys");
407 if (!tempDirectory)
408 return;
409
410 auto tempStoragePath = FileSystem::pathByAppendingComponent(tempDirectory, FileSystem::pathFileName(storagePath));
411 if (!FileSystem::moveFile(storageDirectory, tempStoragePath))
412 return;
413
414 if (!FileSystem::moveFile(tempDirectory, storageDirectory))
415 return;
416 }
417
418 m_storageURL = adoptNS([[NSURL alloc] initFileURLWithPath:storagePath isDirectory:NO]);
419}

677void CDMInstanceSessionFairPlayStreamingAVFObjC::requestLicense(LicenseType licenseType, const AtomString& initDataType, Ref<SharedBuffer>&& initData, LicenseCallback&& callback)
678{
679 if (!isLicenseTypeSupported(licenseType)) {
680 DEBUG_LOG_IF_POSSIBLE(LOGIDENTIFIER, " false, licenseType \"", licenseType, "\" not supported");
681 callback(SharedBuffer::create(), emptyString(), false, Failed);
682 return;
683 }
684
685 if (!m_instance->serverCertificate()) {
686 DEBUG_LOG_IF_POSSIBLE(LOGIDENTIFIER, " false, no serverCertificate");
687 callback(SharedBuffer::create(), emptyString(), false, Failed);
688 return;
689 }
690
