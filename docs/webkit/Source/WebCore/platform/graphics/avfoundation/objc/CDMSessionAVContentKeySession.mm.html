131RefPtr<Uint8Array> CDMSessionAVContentKeySession::generateKeyRequest(const String& mimeType, Uint8Array* initData, String& destinationURL, unsigned short& errorCode, uint32_t& systemCode)
132{
133 UNUSED_PARAM(mimeType);
134 UNUSED_PARAM(destinationURL);
135 ASSERT(initData);
136
137 LOG(Media, "CDMSessionAVContentKeySession::generateKeyRequest(%p)", this);
138
139 errorCode = MediaPlayer::NoError;
140 systemCode = 0;
141
142 if (equalLettersIgnoringASCIICase(mimeType, "keyrelease")) {
143 m_mode = KeyRelease;
144 m_certificate = initData;
145 return generateKeyReleaseMessage(errorCode, systemCode);
146 }
147
148 if (m_cdmVersion == 2)
149 m_identifier = initData;
150 else
151 m_initData = initData;
152
153 ASSERT(!m_certificate);
154 String certificateString("certificate"_s);
155 auto array = Uint8Array::create(certificateString.length());
156 for (unsigned i = 0, length = certificateString.length(); i < length; ++i)
157 array->set(i, certificateString[i]);
158 return WTFMove(array);
159}

161void CDMSessionAVContentKeySession::releaseKeys()
162{
163 if (hasContentKeySession()) {
164 m_stopped = true;
165 for (auto& sourceBuffer : m_sourceBuffers)
166 sourceBuffer->flush();
167
168 LOG(Media, "CDMSessionAVContentKeySession::releaseKeys(%p) - expiring stream session", this);
169 [contentKeySession() expire];
170
171 if (!m_certificate)
172 return;
173
174 if (![PAL::getAVContentKeySessionClass() respondsToSelector:@selector(pendingExpiredSessionReportsWithAppIdentifier:storageDirectoryAtURL:)])
175 return;
176
177 auto storagePath = this->storagePath();
178 if (storagePath.isEmpty())
179 return;
180
181 RetainPtr<NSData> certificateData = adoptNS([[NSData alloc] initWithBytes:m_certificate->data() length:m_certificate->length()]);
182 NSArray* expiredSessions = [PAL::getAVContentKeySessionClass() pendingExpiredSessionReportsWithAppIdentifier:certificateData.get() storageDirectoryAtURL:[NSURL fileURLWithPath:storagePath]];
183 for (NSData* expiredSessionData in expiredSessions) {
184 NSDictionary *expiredSession = [NSPropertyListSerialization propertyListWithData:expiredSessionData options:kCFPropertyListImmutable format:nullptr error:nullptr];
185 NSString *playbackSessionIdValue = (NSString *)[expiredSession objectForKey:PlaybackSessionIdKey];
186 if (![playbackSessionIdValue isKindOfClass:[NSString class]])
187 continue;
188
189 if (m_sessionId == String(playbackSessionIdValue)) {
190 LOG(Media, "CDMSessionAVContentKeySession::releaseKeys(%p) - found session, sending expiration message");
191 m_expiredSession = expiredSessionData;
192 m_client->sendMessage(Uint8Array::create(static_cast<const uint8_t*>([m_expiredSession bytes]), [m_expiredSession length]).ptr(), emptyString());
193 break;
194 }
195 }
196 }
197}

199static bool isEqual(Uint8Array* data, const char* literal)
200{
201 ASSERT(data);
202 ASSERT(literal);
203 unsigned length = data->length();
204
205 for (unsigned i = 0; i < length; ++i) {
206 if (!literal[i])
207 return false;
208
209 if (data->item(i) != static_cast<uint8_t>(literal[i]))
210 return false;
211 }
212 return !literal[length];
213}

215bool CDMSessionAVContentKeySession::update(Uint8Array* key, RefPtr<Uint8Array>& nextMessage, unsigned short& errorCode, uint32_t& systemCode)
216{
217 UNUSED_PARAM(nextMessage);
218
219 if (isEqual(key, "acknowledged")) {
220 LOG(Media, "CDMSessionAVContentKeySession::update(%p) - acknowleding secure stop message", this);
221
222 String storagePath = this->storagePath();
223 if (!m_expiredSession || storagePath.isEmpty()) {
224 errorCode = MediaPlayer::InvalidPlayerState;
225 return false;
226 }
227
228 RetainPtr<NSData> certificateData = adoptNS([[NSData alloc] initWithBytes:m_certificate->data() length:m_certificate->length()]);
229
230 if ([PAL::getAVContentKeySessionClass() respondsToSelector:@selector(removePendingExpiredSessionReports:withAppIdentifier:storageDirectoryAtURL:)])
231 [PAL::getAVContentKeySessionClass() removePendingExpiredSessionReports:@[m_expiredSession.get()] withAppIdentifier:certificateData.get() storageDirectoryAtURL:[NSURL fileURLWithPath:storagePath]];
232 m_expiredSession = nullptr;
233 return true;
234 }
235
236 if (m_stopped) {
237 errorCode = MediaPlayer::InvalidPlayerState;
238 return false;
239 }
240
241 bool shouldGenerateKeyRequest = !m_certificate || isEqual(key, "renew");
242 if (!m_certificate) {
243 LOG(Media, "CDMSessionAVContentKeySession::update(%p) - certificate data", this);
244
245 m_certificate = key;
246 }
247
248 if (m_mode == KeyRelease)
249 return false;
250
251 if (m_cdmVersion == 2) {
252 // In the com.apple.fps.2_0 communication protocol, the client must first attach the
253 // session to the protected SourceBuffer in order to get access to the initialization
254 // data.
255 RefPtr<SourceBufferPrivateAVFObjC> protectedSourceBuffer;
256 for (auto& sourceBuffer : m_sourceBuffers) {
257 if (sourceBuffer->protectedTrackID() != notFound) {
258 protectedSourceBuffer = sourceBuffer;
259 break;
260 }
261 }
262
263 if (!protectedSourceBuffer) {
264 errorCode = MediaPlayer::InvalidPlayerState;
265 return false;
266 }
267
268 m_initData = protectedSourceBuffer->initData();
269 }
270
271 if (!m_keyRequest) {
272 NSData* nsInitData = m_initData ? [NSData dataWithBytes:m_initData->data() length:m_initData->length()] : nil;
273 NSData* nsIdentifier = m_identifier ? [NSData dataWithBytes:m_identifier->data() length:m_identifier->length()] : nil;
274 if ([contentKeySession() respondsToSelector:@selector(processContentKeyRequestWithIdentifier:initializationData:options:)])
275 [contentKeySession() processContentKeyRequestWithIdentifier:nsIdentifier initializationData:nsInitData options:nil];
276 else
277 [contentKeySession() processContentKeyRequestInitializationData:nsInitData options:nil];
278 }
279
280 if (shouldGenerateKeyRequest) {
281 ASSERT(m_keyRequest);
282 RetainPtr<NSData> certificateData = adoptNS([[NSData alloc] initWithBytes:m_certificate->data() length:m_certificate->length()]);
283
284 RetainPtr<NSDictionary> options;
285 if (!m_protocolVersions.isEmpty() && PAL::canLoad_AVFoundation_AVContentKeyRequestProtocolVersionsKey()) {
286 options = @{ AVContentKeyRequestProtocolVersionsKey: createNSArray(m_protocolVersions, [] (int version) -> NSNumber * {
287 return version ? @(version) : nil;
288 }).get() };
289 }
290
291 errorCode = MediaPlayer::NoError;
292 systemCode = 0;
293 NSError* error = nil;
294 NSData* nsIdentifier = m_identifier ? [NSData dataWithBytes:m_identifier->data() length:m_identifier->length()] : m_keyRequest.get().identifier;
295
296 NSData* requestData = [m_keyRequest contentKeyRequestDataForApp:certificateData.get() contentIdentifier:nsIdentifier options:options.get() error:&error];
297 if (error) {
298 errorCode = LegacyCDM::DomainError;
299 systemCode = mediaKeyErrorSystemCode(error);
300 return false;
301 }
302
303 nextMessage = Uint8Array::tryCreate(static_cast<const uint8_t*>([requestData bytes]), [requestData length]);
304 return false;
305 }
306
307 LOG(Media, "CDMSessionAVContentKeySession::update(%p) - key data", this);
308 errorCode = MediaPlayer::NoError;
309 systemCode = 0;
310 RetainPtr<NSData> keyData = adoptNS([[NSData alloc] initWithBytes:key->data() length:key->length()]);
311
312 if ([m_keyRequest respondsToSelector:@selector(processContentKeyResponse:)] && [PAL::getAVContentKeyResponseClass() respondsToSelector:@selector(contentKeyResponseWithFairPlayStreamingKeyResponseData:)])
313 [m_keyRequest processContentKeyResponse:[PAL::getAVContentKeyResponseClass() contentKeyResponseWithFairPlayStreamingKeyResponseData:keyData.get()]];
314 else
315 [m_keyRequest processContentKeyResponseData:keyData.get()];
316
317 return true;
318}

336RefPtr<Uint8Array> CDMSessionAVContentKeySession::generateKeyReleaseMessage(unsigned short& errorCode, uint32_t& systemCode)
337{
338 ASSERT(m_mode == KeyRelease);
339 RetainPtr<NSData> certificateData = adoptNS([[NSData alloc] initWithBytes:m_certificate->data() length:m_certificate->length()]);
340
341 String storagePath = this->storagePath();
342 if (storagePath.isEmpty() || ![PAL::getAVContentKeySessionClass() respondsToSelector:@selector(pendingExpiredSessionReportsWithAppIdentifier:storageDirectoryAtURL:)]) {
343 errorCode = MediaPlayer::KeySystemNotSupported;
344 systemCode = '!mor';
345 return nullptr;
346 }
347
348 NSArray* expiredSessions = [PAL::getAVContentKeySessionClass() pendingExpiredSessionReportsWithAppIdentifier:certificateData.get() storageDirectoryAtURL:[NSURL fileURLWithPath:storagePath]];
349 if (![expiredSessions count]) {
350 LOG(Media, "CDMSessionAVContentKeySession::generateKeyReleaseMessage(%p) - no expired sessions found", this);
351
352 errorCode = MediaPlayer::KeySystemNotSupported;
353 systemCode = '!mor';
354 return nullptr;
355 }
356
357 LOG(Media, "CDMSessionAVContentKeySession::generateKeyReleaseMessage(%p) - found %d expired sessions", this, [expiredSessions count]);
358
359 errorCode = 0;
360 systemCode = 0;
361 m_expiredSession = [expiredSessions firstObject];
362 return Uint8Array::tryCreate(static_cast<const uint8_t*>([m_expiredSession bytes]), [m_expiredSession length]);
363}

370AVContentKeySession* CDMSessionAVContentKeySession::contentKeySession()
371{
372 if (m_contentKeySession)
373 return m_contentKeySession.get();
374
375 String storagePath = this->storagePath();
376 if (storagePath.isEmpty()) {
377 if (![PAL::getAVContentKeySessionClass() respondsToSelector:@selector(contentKeySessionWithKeySystem:)] || !PAL::canLoad_AVFoundation_AVContentKeySystemFairPlayStreaming())
378 return nil;
379
380 m_contentKeySession = [PAL::getAVContentKeySessionClass() contentKeySessionWithKeySystem:AVContentKeySystemFairPlayStreaming];
381 } else {
382 String storageDirectory = FileSystem::parentPath(storagePath);
383
384 if (!FileSystem::fileExists(storageDirectory)) {
385 if (!FileSystem::makeAllDirectories(storageDirectory))
386 return nil;
387 }
388
389 auto url = [NSURL fileURLWithPath:storagePath];
390 if ([PAL::getAVContentKeySessionClass() respondsToSelector:@selector(contentKeySessionWithKeySystem:storageDirectoryAtURL:)] && PAL::canLoad_AVFoundation_AVContentKeySystemFairPlayStreaming())
391 m_contentKeySession = [PAL::getAVContentKeySessionClass() contentKeySessionWithKeySystem:AVContentKeySystemFairPlayStreaming storageDirectoryAtURL:url];
392 else
393 m_contentKeySession = adoptNS([PAL::allocAVContentKeySessionInstance() initWithStorageDirectoryAtURL:url]);
394 }
395
396 m_contentKeySession.get().delegate = m_contentKeySessionDelegate.get();
397 return m_contentKeySession.get();
398}
