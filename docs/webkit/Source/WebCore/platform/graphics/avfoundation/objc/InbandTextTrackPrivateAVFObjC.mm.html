59InbandTextTrackPrivate::Kind InbandTextTrackPrivateAVFObjC::kind() const
60{
61 if (!m_mediaSelectionOption)
62 return Kind::None;
63
64 NSString *mediaType = [m_mediaSelectionOption mediaType];
65
66 if ([mediaType isEqualToString:AVMediaTypeClosedCaption])
67 return Kind::Captions;
68 if ([mediaType isEqualToString:AVMediaTypeSubtitle]) {
69
70 if ([m_mediaSelectionOption hasMediaCharacteristic:AVMediaCharacteristicContainsOnlyForcedSubtitles])
71 return Kind::Forced;
72
73 // An "SDH" track is a subtitle track created for the deaf or hard-of-hearing. "captions" in WebVTT are
74 // "labeled as appropriate for the hard-of-hearing", so tag SDH sutitles as "captions".
75 if ([m_mediaSelectionOption hasMediaCharacteristic:AVMediaCharacteristicTranscribesSpokenDialogForAccessibility])
76 return Kind::Captions;
77 if ([m_mediaSelectionOption hasMediaCharacteristic:AVMediaCharacteristicDescribesMusicAndSoundForAccessibility])
78 return Kind::Captions;
79
80 return Kind::Subtitles;
81 }
82
83 return Kind::Captions;
84}

94bool InbandTextTrackPrivateAVFObjC::isSDH() const
95{
96 if (!m_mediaSelectionOption)
97 return false;
98
99 if (![[m_mediaSelectionOption mediaType] isEqualToString:AVMediaTypeSubtitle])
100 return false;
101
102 if ([m_mediaSelectionOption hasMediaCharacteristic:AVMediaCharacteristicTranscribesSpokenDialogForAccessibility] && [m_mediaSelectionOption hasMediaCharacteristic:AVMediaCharacteristicDescribesMusicAndSoundForAccessibility])
103 return true;
104
105 return false;
106}

132AtomString InbandTextTrackPrivateAVFObjC::label() const
133{
134 if (!m_mediaSelectionOption)
135 return emptyAtom();
136
137 NSString *title = 0;
138
139 NSArray *titles = [PAL::getAVMetadataItemClass() metadataItemsFromArray:[m_mediaSelectionOption.get() commonMetadata] withKey:AVMetadataCommonKeyTitle keySpace:AVMetadataKeySpaceCommon];
140 if ([titles count]) {
141 // If possible, return a title in one of the user's preferred languages.
142 NSArray *titlesForPreferredLanguages = [PAL::getAVMetadataItemClass() metadataItemsFromArray:titles filteredAndSortedAccordingToPreferredLanguages:[NSLocale preferredLanguages]];
143 if ([titlesForPreferredLanguages count])
144 title = [[titlesForPreferredLanguages objectAtIndex:0] stringValue];
145
146 if (!title)
147 title = [[titles objectAtIndex:0] stringValue];
148 }
149
150 return title ? AtomString(title) : emptyAtom();
151}
