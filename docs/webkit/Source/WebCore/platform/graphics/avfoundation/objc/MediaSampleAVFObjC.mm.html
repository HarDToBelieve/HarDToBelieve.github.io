48RefPtr<MediaSampleAVFObjC> MediaSampleAVFObjC::createImageSample(PixelBuffer&& pixelBuffer)
49{
50 auto size = pixelBuffer.size();
51 auto width = size.width();
52 auto height = size.height();
53
54 auto data = pixelBuffer.takeData();
55 auto dataBaseAddress = data->data();
56 auto leakedData = &data.leakRef();
57
58 auto derefBuffer = [] (void* context, const void*) {
59 static_cast<JSC::Uint8ClampedArray*>(context)->deref();
60 };
61
62 CVPixelBufferRef cvPixelBufferRaw = nullptr;
63 auto status = CVPixelBufferCreateWithBytes(kCFAllocatorDefault, width, height, kCVPixelFormatType_32BGRA, dataBaseAddress, width * 4, derefBuffer, leakedData, nullptr, &cvPixelBufferRaw);
64
65 auto cvPixelBuffer = adoptCF(cvPixelBufferRaw);
66 if (!cvPixelBuffer) {
67 derefBuffer(leakedData, nullptr);
68 return nullptr;
69 }
70 ASSERT_UNUSED(status, !status);
71
72 CMVideoFormatDescriptionRef formatDescriptionRaw = nullptr;
73 status = PAL::CMVideoFormatDescriptionCreateForImageBuffer(kCFAllocatorDefault, cvPixelBuffer.get(), &formatDescriptionRaw);
74 ASSERT(!status);
75 auto formatDescription = adoptCF(formatDescriptionRaw);
76
77 CMSampleTimingInfo sampleTimingInformation = { PAL::kCMTimeInvalid, PAL::kCMTimeInvalid, PAL::kCMTimeInvalid };
78 CMSampleBufferRef sampleBufferRaw = nullptr;
79 status = PAL::CMSampleBufferCreateReadyWithImageBuffer(kCFAllocatorDefault, cvPixelBuffer.get(), formatDescription.get(), &sampleTimingInformation, &sampleBufferRaw);
80 ASSERT(!status);
81 auto sampleBuffer = adoptCF(sampleBufferRaw);
82
83 CFArrayRef attachmentsArray = PAL::CMSampleBufferGetSampleAttachmentsArray(sampleBuffer.get(), true);
84 for (CFIndex i = 0, count = CFArrayGetCount(attachmentsArray); i < count; ++i) {
85 CFMutableDictionaryRef attachments = checked_cf_cast<CFMutableDictionaryRef>(CFArrayGetValueAtIndex(attachmentsArray, i));
86 CFDictionarySetValue(attachments, PAL::kCMSampleAttachmentKey_DisplayImmediately, kCFBooleanTrue);
87 }
88 return create(sampleBuffer.get());
89}

133{

135}

137static bool doesCMSampleBufferHaveSyncInfo(CMSampleBufferRef sample)
138{
139 return PAL::CMSampleBufferGetSampleAttachmentsArray(sample, false);
140}

142static bool isCMSampleBufferRandomAccess(CMSampleBufferRef sample)
143{
144 CFArrayRef attachments = PAL::CMSampleBufferGetSampleAttachmentsArray(sample, false);
145 if (!attachments)
146 return true;
147
148 for (CFIndex i = 0, count = CFArrayGetCount(attachments); i < count; ++i) {
149 if (!isCMSampleBufferAttachmentRandomAccess(checked_cf_cast<CFDictionaryRef>(CFArrayGetValueAtIndex(attachments, i))))
150 return false;
151 }
152 return true;
153}

163 if (!attachments)

165
166 for (CFIndex i = 0; i < CFArrayGetCount(attachments); ++i) {
167 if (isCMSampleBufferAttachmentNonDisplaying(checked_cf_cast<CFDictionaryRef>(CFArrayGetValueAtIndex(attachments, i))))
168 return true;
169 }
170
171 return false;

174MediaSample::SampleFlags MediaSampleAVFObjC::flags() const
175{
176 int returnValue = MediaSample::None;
177
178 if (doesCMSampleBufferHaveSyncInfo(m_sample.get()))
179 returnValue |= MediaSample::HasSyncInfo;
180
181 if (isCMSampleBufferRandomAccess(m_sample.get()))
182 returnValue |= MediaSample::IsSync;
183
184 if (isCMSampleBufferNonDisplaying(m_sample.get()))
185 returnValue |= MediaSample::IsNonDisplaying;
186
187 return SampleFlags(returnValue);
188}

204void MediaSampleAVFObjC::offsetTimestampsBy(const MediaTime& offset)
205{
206 CMItemCount itemCount = 0;
207 if (noErr != PAL::CMSampleBufferGetSampleTimingInfoArray(m_sample.get(), 0, nullptr, &itemCount))
208 return;
209
210 Vector<CMSampleTimingInfo> timingInfoArray;
211 timingInfoArray.grow(itemCount);
212 if (noErr != PAL::CMSampleBufferGetSampleTimingInfoArray(m_sample.get(), itemCount, timingInfoArray.data(), nullptr))
213 return;
214
215 for (auto& timing : timingInfoArray) {
216 timing.presentationTimeStamp = PAL::toCMTime(PAL::toMediaTime(timing.presentationTimeStamp) + offset);
217 timing.decodeTimeStamp = PAL::toCMTime(PAL::toMediaTime(timing.decodeTimeStamp) + offset);
218 }
219
220 CMSampleBufferRef newSample;
221 if (noErr != PAL::CMSampleBufferCreateCopyWithNewTiming(kCFAllocatorDefault, m_sample.get(), itemCount, timingInfoArray.data(), &newSample))
222 return;
223
224 m_sample = adoptCF(newSample);
225}

250bool MediaSampleAVFObjC::isDivisable() const
251{
252 if (PAL::CMSampleBufferGetNumSamples(m_sample.get()) == 1)
253 return false;
254
255 if (PAL::CMSampleBufferGetSampleSizeArray(m_sample.get(), 0, nullptr, nullptr) == kCMSampleBufferError_BufferHasNoSampleSizes)
256 return false;
257
258 return true;
259}

261std::pair<RefPtr<MediaSample>, RefPtr<MediaSample>> MediaSampleAVFObjC::divide(const MediaTime& presentationTime, UseEndTime useEndTime)
262{
263 if (!isDivisable())
264 return { nullptr, nullptr };
265
266 CFIndex samplesBeforePresentationTime = 0;
267
268 PAL::CMSampleBufferCallBlockForEachSample(m_sample.get(), [&] (CMSampleBufferRef sampleBuffer, CMItemCount) -> OSStatus {
269 auto timeStamp = PAL::CMSampleBufferGetOutputPresentationTimeStamp(sampleBuffer);
270 if (CMTIME_IS_INVALID(timeStamp))
271 timeStamp = PAL::CMSampleBufferGetPresentationTimeStamp(sampleBuffer);
272
273 if (useEndTime == UseEndTime::Use) {
274 auto duration = PAL::CMSampleBufferGetOutputDuration(sampleBuffer);
275 if (CMTIME_IS_INVALID(duration))
276 duration = PAL::CMSampleBufferGetDuration(sampleBuffer);
277
278 if (PAL::toMediaTime(PAL::CMTimeAdd(timeStamp, duration)) > presentationTime)
279 return 1;
280 } else if (PAL::toMediaTime(timeStamp) >= presentationTime)
281 return 1;
282 ++samplesBeforePresentationTime;
283 return noErr;
284 });
285
286 if (!samplesBeforePresentationTime)
287 return { nullptr, this };
288
289 CMItemCount sampleCount = PAL::CMSampleBufferGetNumSamples(m_sample.get());
290 if (samplesBeforePresentationTime >= sampleCount)
291 return { this, nullptr };
292
293 CMSampleBufferRef rawSampleBefore = nullptr;
294 CFRange rangeBefore = CFRangeMake(0, samplesBeforePresentationTime);
295 if (PAL::CMSampleBufferCopySampleBufferForRange(kCFAllocatorDefault, m_sample.get(), rangeBefore, &rawSampleBefore) != noErr)
296 return { nullptr, nullptr };
297 RetainPtr<CMSampleBufferRef> sampleBefore = adoptCF(rawSampleBefore);
298
299 CMSampleBufferRef rawSampleAfter = nullptr;
300 CFRange rangeAfter = CFRangeMake(samplesBeforePresentationTime, sampleCount - samplesBeforePresentationTime);
301 if (PAL::CMSampleBufferCopySampleBufferForRange(kCFAllocatorDefault, m_sample.get(), rangeAfter, &rawSampleAfter) != noErr)
302 return { nullptr, nullptr };
303 RetainPtr<CMSampleBufferRef> sampleAfter = adoptCF(rawSampleAfter);
304
305 return { MediaSampleAVFObjC::create(sampleBefore.get(), m_id), MediaSampleAVFObjC::create(sampleAfter.get(), m_id) };
306}

309{

319
320 return MediaSampleAVFObjC::create(adoptCF(newSampleBuffer).get(), m_id);
321}

323RefPtr<JSC::Uint8ClampedArray> MediaSampleAVFObjC::getRGBAImageData() const
324{
325 const OSType imageFormat = kCVPixelFormatType_32RGBA;
326 RetainPtr<CFNumberRef> imageFormatNumber = adoptCF(CFNumberCreate(nullptr, kCFNumberIntType, &imageFormat));
327
328 RetainPtr<CFMutableDictionaryRef> conformerOptions = adoptCF(CFDictionaryCreateMutable(0, 1, &kCFTypeDictionaryKeyCallBacks, &kCFTypeDictionaryValueCallBacks));
329 CFDictionarySetValue(conformerOptions.get(), kCVPixelBufferPixelFormatTypeKey, imageFormatNumber.get());
330 PixelBufferConformerCV pixelBufferConformer(conformerOptions.get());
331
332 auto pixelBuffer = static_cast<CVPixelBufferRef>(PAL::CMSampleBufferGetImageBuffer(m_sample.get()));
333 auto rgbaPixelBuffer = pixelBufferConformer.convert(pixelBuffer);
334 auto status = CVPixelBufferLockBaseAddress(rgbaPixelBuffer.get(), kCVPixelBufferLock_ReadOnly);
335 ASSERT(status == noErr);
336
337 void* data = CVPixelBufferGetBaseAddressOfPlane(rgbaPixelBuffer.get(), 0);
338 size_t byteLength = CVPixelBufferGetHeight(pixelBuffer) * CVPixelBufferGetWidth(pixelBuffer) * 4;
339 auto result = JSC::Uint8ClampedArray::tryCreate(JSC::ArrayBuffer::create(data, byteLength), 0, byteLength);
340
341 status = CVPixelBufferUnlockBaseAddress(rgbaPixelBuffer.get(), kCVPixelBufferLock_ReadOnly);
342 ASSERT(status == noErr);
343
344 return result;
345}

361bool MediaSampleAVFObjC::isHomogeneous() const
362{
363 CFArrayRef attachmentsArray = PAL::CMSampleBufferGetSampleAttachmentsArray(m_sample.get(), true);
364 if (!attachmentsArray)
365 return true;
366
367 auto count = CFArrayGetCount(attachmentsArray);
368 if (count <= 1)
369 return true;
370
371 CFDictionaryRef firstAttachment = checked_cf_cast<CFDictionaryRef>(CFArrayGetValueAtIndex(attachmentsArray, 0));
372 bool isSync = isCMSampleBufferAttachmentRandomAccess(firstAttachment);
373 bool isNonDisplaying = isCMSampleBufferAttachmentNonDisplaying(firstAttachment);
374
375 for (CFIndex i = 1; i < count; ++i) {
376 auto attachmentDict = checked_cf_cast<CFDictionaryRef>(CFArrayGetValueAtIndex(attachmentsArray, i));
377 if (isSync != isCMSampleBufferAttachmentRandomAccess(attachmentDict))
378 return false;
379
380 if (isNonDisplaying != isCMSampleBufferAttachmentNonDisplaying(attachmentDict))
381 return false;
382 };
383
384 return true;
385}

387Vector<Ref<MediaSampleAVFObjC>> MediaSampleAVFObjC::divideIntoHomogeneousSamples()
388{
389 using SampleVector = Vector<Ref<MediaSampleAVFObjC>>;
390
391 CFArrayRef attachmentsArray = PAL::CMSampleBufferGetSampleAttachmentsArray(m_sample.get(), true);
392 if (!attachmentsArray)
393 return SampleVector::from(makeRef(*this));
394
395 auto count = CFArrayGetCount(attachmentsArray);
396 if (count <= 1)
397 return SampleVector::from(makeRef(*this));
398
399 CFDictionaryRef firstAttachment = checked_cf_cast<CFDictionaryRef>(CFArrayGetValueAtIndex(attachmentsArray, 0));
400 bool isSync = isCMSampleBufferAttachmentRandomAccess(firstAttachment);
401 bool isNonDisplaying = isCMSampleBufferAttachmentNonDisplaying(firstAttachment);
402 Vector<CFRange> ranges;
403 CFIndex currentRangeStart = 0;
404 CFIndex currentRangeLength = 1;
405
406 for (CFIndex i = 1; i < count; ++i, ++currentRangeLength) {
407 CFDictionaryRef attachmentDict = checked_cf_cast<CFDictionaryRef>(CFArrayGetValueAtIndex(attachmentsArray, i));
408 if (isSync == isCMSampleBufferAttachmentRandomAccess(attachmentDict) && isNonDisplaying == isCMSampleBufferAttachmentNonDisplaying(attachmentDict))
409 continue;
410
411 ranges.append(CFRangeMake(currentRangeStart, currentRangeLength));
412 currentRangeStart = i;
413 currentRangeLength = 0;
414 }
415 ranges.append(CFRangeMake(currentRangeStart, currentRangeLength));
416
417 if (ranges.size() == 1)
418 return SampleVector::from(makeRef(*this));
419
420 SampleVector samples;
421 samples.reserveInitialCapacity(ranges.size());
422 for (auto& range : ranges) {
423 CMSampleBufferRef rawSample = nullptr;
424 if (PAL::CMSampleBufferCopySampleBufferForRange(kCFAllocatorDefault, m_sample.get(), range, &rawSample) != noErr || !rawSample)
425 return { };
426 samples.uncheckedAppend(MediaSampleAVFObjC::create(adoptCF(rawSample).get(), m_id));
427 }
428 return samples;
429}

431RetainPtr<CMSampleBufferRef> MediaSampleAVFObjC::cloneSampleBufferAndSetAsDisplayImmediately(CMSampleBufferRef sample)
432{
433 auto pixelBuffer = static_cast<CVImageBufferRef>(PAL::CMSampleBufferGetImageBuffer(sample));
434 if (!pixelBuffer)
435 return nullptr;
436
437 CMVideoFormatDescriptionRef formatDescription = nullptr;
438 auto status = PAL::CMVideoFormatDescriptionCreateForImageBuffer(kCFAllocatorDefault, pixelBuffer, &formatDescription);
439 if (status)
440 return nullptr;
441 auto retainedFormatDescription = adoptCF(formatDescription);
442
443 CMItemCount itemCount = 0;
444 status = PAL::CMSampleBufferGetSampleTimingInfoArray(sample, 0, nullptr, &itemCount);
445 if (status)
446 return nullptr;
447
448 Vector<CMSampleTimingInfo> timingInfoArray;
449 timingInfoArray.grow(itemCount);
450 status = PAL::CMSampleBufferGetSampleTimingInfoArray(sample, itemCount, timingInfoArray.data(), nullptr);
451 if (status)
452 return nullptr;
453
454 CMSampleBufferRef newSampleBuffer;
455 status = PAL::CMSampleBufferCreateReadyWithImageBuffer(kCFAllocatorDefault, pixelBuffer, formatDescription, timingInfoArray.data(), &newSampleBuffer);
456 if (status)
457 return nullptr;
458
459 setSampleBufferAsDisplayImmediately(newSampleBuffer);
460
461 return adoptCF(newSampleBuffer);
462}

482std::optional<MediaSample::ByteRange> MediaSampleAVFObjC::byteRangeForAttachment(CFStringRef key) const
483{
484 auto byteOffsetCF = dynamic_cf_cast<CFNumberRef>(PAL::CMGetAttachment(m_sample.get(), key, nullptr));
485 if (!byteOffsetCF)
486 return std::nullopt;
487
488 int64_t byteOffset = 0;
489 if (!CFNumberGetValue(byteOffsetCF, kCFNumberSInt64Type, &byteOffset))
490 return std::nullopt;
491
492 CMItemCount sizeArrayEntries = 0;
493 PAL::CMSampleBufferGetSampleSizeArray(m_sample.get(), 0, nullptr, &sizeArrayEntries);
494 if (sizeArrayEntries != 1)
495 return std::nullopt;
496
497 size_t singleSizeEntry = 0;
498 PAL::CMSampleBufferGetSampleSizeArray(m_sample.get(), 1, &singleSizeEntry, nullptr);
499 return { { CheckedSize(byteOffset), singleSizeEntry } };
500}
