79MediaSourcePrivate::AddStatus MediaSourcePrivateAVFObjC::addSourceBuffer(const ContentType& contentType, bool webMParserEnabled, RefPtr<SourceBufferPrivate>& outPrivate)
80{
81 DEBUG_LOG(LOGIDENTIFIER, contentType);
82
83 MediaEngineSupportParameters parameters;
84 parameters.isMediaSource = true;
85 parameters.type = contentType;
86 if (MediaPlayerPrivateMediaSourceAVFObjC::supportsTypeAndCodecs(parameters) == MediaPlayer::SupportsType::IsNotSupported)
87 return AddStatus::NotSupported;
88
89 auto parser = SourceBufferParser::create(contentType, webMParserEnabled);
90 if (!parser)
91 return AddStatus::NotSupported;
92
93 auto newSourceBuffer = SourceBufferPrivateAVFObjC::create(this, parser.releaseNonNull());
94#if ENABLE(ENCRYPTED_MEDIA)
95 newSourceBuffer->setCDMInstance(m_cdmInstance.get());
96#endif
97 outPrivate = newSourceBuffer.copyRef();
98 m_sourceBuffers.append(WTFMove(newSourceBuffer));
99
100 return AddStatus::Ok;
101}

103void MediaSourcePrivateAVFObjC::removeSourceBuffer(SourceBufferPrivate* buffer)
104{
105 ASSERT(m_sourceBuffers.contains(buffer));
106
107 if (buffer == m_sourceBufferWithSelectedVideo)
108 m_sourceBufferWithSelectedVideo = nullptr;
109
110 size_t pos = m_activeSourceBuffers.find(buffer);
111 if (pos != notFound) {
112 m_activeSourceBuffers.remove(pos);
113 if (m_player)
114 m_player->notifyActiveSourceBuffersChanged();
115 }
116
117 pos = m_sourceBuffers.find(buffer);
118 m_sourceBuffers[pos]->clearMediaSource();
119 m_sourceBuffers.remove(pos);
120}

179void MediaSourcePrivateAVFObjC::sourceBufferPrivateDidChangeActiveState(SourceBufferPrivateAVFObjC* buffer, bool active)
180{
181 if (active && !m_activeSourceBuffers.contains(buffer)) {
182 m_activeSourceBuffers.append(buffer);
183 if (m_player)
184 m_player->notifyActiveSourceBuffersChanged();
185 }
186
187 if (!active) {
188 size_t position = m_activeSourceBuffers.find(buffer);
189 if (position != notFound) {
190 m_activeSourceBuffers.remove(position);
191 if (m_player)
192 m_player->notifyActiveSourceBuffersChanged();
193 }
194 }
195}

241MediaTime MediaSourcePrivateAVFObjC::fastSeekTimeForMediaTime(const MediaTime& targetTime, const MediaTime& negativeThreshold, const MediaTime& positiveThreshold)
242{
243 MediaTime seekTime = targetTime;
244
245 for (auto it = m_activeSourceBuffers.begin(), end = m_activeSourceBuffers.end(); it != end; ++it) {
246 MediaTime sourceSeekTime = (*it)->fastSeekTimeForMediaTime(targetTime, negativeThreshold, positiveThreshold);
247 if (abs(targetTime - sourceSeekTime) > abs(targetTime - seekTime))
248 seekTime = sourceSeekTime;
249 }
250
251 return seekTime;
252}

264void MediaSourcePrivateAVFObjC::hasSelectedVideoChanged(SourceBufferPrivateAVFObjC& sourceBuffer)
265{
266 bool hasSelectedVideo = sourceBuffer.hasSelectedVideo();
267 if (m_sourceBufferWithSelectedVideo == &sourceBuffer && !hasSelectedVideo)
268 setSourceBufferWithSelectedVideo(nullptr);
269 else if (m_sourceBufferWithSelectedVideo != &sourceBuffer && hasSelectedVideo)
270 setSourceBufferWithSelectedVideo(&sourceBuffer);
271}

294void MediaSourcePrivateAVFObjC::cdmInstanceAttached(CDMInstance& instance)
295{
296 if (m_cdmInstance.get() == &instance)
297 return;

299 ASSERT(!m_cdmInstance);

305void MediaSourcePrivateAVFObjC::cdmInstanceDetached(CDMInstance& instance)
306{
307 ASSERT_UNUSED(instance, m_cdmInstance && m_cdmInstance == &instance);
308 for (auto& sourceBuffer : m_sourceBuffers)
309 sourceBuffer->setCDMInstance(nullptr);
310
311 m_cdmInstance = nullptr;
312}

335void MediaSourcePrivateAVFObjC::setSourceBufferWithSelectedVideo(SourceBufferPrivateAVFObjC* sourceBuffer)
336{
337 if (m_sourceBufferWithSelectedVideo) {
338 m_sourceBufferWithSelectedVideo->setVideoLayer(nullptr);
339 m_sourceBufferWithSelectedVideo->setDecompressionSession(nullptr);
340 }
341
342 m_sourceBufferWithSelectedVideo = sourceBuffer;
343
344 if (m_sourceBufferWithSelectedVideo && m_player) {
345 m_sourceBufferWithSelectedVideo->setVideoLayer(m_player->sampleBufferDisplayLayer());
346 m_sourceBufferWithSelectedVideo->setDecompressionSession(m_player->decompressionSession());
347 }
