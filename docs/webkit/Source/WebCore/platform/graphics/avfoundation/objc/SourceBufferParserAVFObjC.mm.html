58@interface WebAVStreamDataParserListener : NSObject<AVStreamDataParserOutputHandling> {
59 WebCore::SourceBufferParserAVFObjC* _parent;

66@implementation WebAVStreamDataParserListener
67- (id)initWithParser:(AVStreamDataParser*)parser parent:(WebCore::SourceBufferParserAVFObjC*)parent
68{
69 self = [super init];
70 if (!self)
71 return nil;
72
73 ASSERT(parent);
74 _parent = parent;
75 _parser = parser;
76 [_parser setDelegate:self];
77 return self;
78}
79
80@synthesize parent = _parent;
81
82- (void)dealloc
83{
84 [_parser setDelegate:nil];
85 [super dealloc];
86}
87
88- (void)invalidate
89{
90 [_parser setDelegate:nil];
91 _parser = nullptr;
92}

141class MediaDescriptionAVFObjC final : public MediaDescription {
142public:
143 static Ref<MediaDescriptionAVFObjC> create(AVAssetTrack* track) { return adoptRef(*new MediaDescriptionAVFObjC(track)); }
144 virtual ~MediaDescriptionAVFObjC() { }
145
146 bool isVideo() const override { return m_isVideo; }
147 bool isAudio() const override { return m_isAudio; }
148 bool isText() const override { return m_isText; }
149 AtomString codec() const override
150 {
151 // AtomStrings can only be destroyed from the same thread they're
152 // created in. Since this structure is created in a parser thread
153 // only create the AtomString the first time this function is accessed
154 // which is presumably on the main thread.
155 ASSERT(isMainThread());
156 if (!m_codec)
157 m_codec = AtomString(reinterpret_cast<const LChar*>(&m_originalCodec), 4);
158 return *m_codec;
159 }
160
161private:
162 MediaDescriptionAVFObjC(AVAssetTrack* track)
163 : m_isVideo([track hasMediaCharacteristic:AVMediaCharacteristicVisual])
164 , m_isAudio([track hasMediaCharacteristic:AVMediaCharacteristicAudible])
165 , m_isText([track hasMediaCharacteristic:AVMediaCharacteristicLegible])
166 {
167 NSArray* formatDescriptions = [track formatDescriptions];
168 CMFormatDescriptionRef description = [formatDescriptions count] ? (__bridge CMFormatDescriptionRef)[formatDescriptions objectAtIndex:0] : 0;
169 if (description) {
170 m_originalCodec = PAL::softLink_CoreMedia_CMFormatDescriptionGetMediaSubType(description);
171 CFStringRef originalFormatKey = PAL::canLoad_CoreMedia_kCMFormatDescriptionExtension_ProtectedContentOriginalFormat() ? PAL::get_CoreMedia_kCMFormatDescriptionExtension_ProtectedContentOriginalFormat() : CFSTR("CommonEncryptionOriginalFormat");
172 CFTypeRef originalFormat = CMFormatDescriptionGetExtension(description, originalFormatKey);
173 if (originalFormat && CFGetTypeID(originalFormat) == CFNumberGetTypeID())
174 CFNumberGetValue((CFNumberRef)originalFormat, kCFNumberSInt32Type, &m_originalCodec);
175 }
176 }
177
178 FourCharCode m_originalCodec;
179 mutable std::optional<AtomString> m_codec;
180 bool m_isVideo;
181 bool m_isAudio;
182 bool m_isText;
183};

188MediaPlayerEnums::SupportsType SourceBufferParserAVFObjC::isContentTypeSupported(const ContentType& type)
189{
190 String extendedType = type.raw();
191 String outputCodecs = type.parameter(ContentType::codecsParameter());
192 if (!outputCodecs.isEmpty() && [PAL::getAVStreamDataParserClass() respondsToSelector:@selector(outputMIMECodecParameterForInputMIMECodecParameter:)]) {
193 outputCodecs = [PAL::getAVStreamDataParserClass() outputMIMECodecParameterForInputMIMECodecParameter:outputCodecs];
194 extendedType = makeString(type.containerType(), "; codecs=\"", outputCodecs, "\"");
195 }
196
197 return AVStreamDataParserMIMETypeCache::singleton().canDecodeType(extendedType);
198}

212void SourceBufferParserAVFObjC::appendData(Segment&& segment, CompletionHandler<void()>&& completionHandler, AppendFlags flags)
213{
214 auto sharedData = SharedBuffer::create(segment.takeVector());
215 auto nsData = sharedData->createNSData();
216 if (m_parserStateWasReset || flags == AppendFlags::Discontinuity)
217 [m_parser appendStreamData:nsData.get() withFlags:AVStreamDataParserStreamDataDiscontinuity];
218 else
219 [m_parser appendStreamData:nsData.get()];
220 m_parserStateWasReset = false;
221 completionHandler();
222}

261void SourceBufferParserAVFObjC::didParseStreamDataAsAsset(AVAsset* asset)
262{
263 m_callOnClientThreadCallback([this, strongThis = makeRef(*this), asset = retainPtr(asset)] {
264 if (!m_didParseInitializationDataCallback)
265 return;
266
267 InitializationSegment segment;
268
269 if ([asset respondsToSelector:@selector(overallDurationHint)])
270 segment.duration = PAL::toMediaTime([asset overallDurationHint]);
271
272 if (segment.duration.isInvalid() || segment.duration == MediaTime::zeroTime())
273 segment.duration = PAL::toMediaTime([asset duration]);
274
275 for (AVAssetTrack* track in [asset tracks]) {
276 if ([track hasMediaCharacteristic:AVMediaCharacteristicLegible]) {
277 // FIXME(125161): Handle in-band text tracks.
278 continue;
279 }
280
281 if ([track hasMediaCharacteristic:AVMediaCharacteristicVisual]) {
282 SourceBufferPrivateClient::InitializationSegment::VideoTrackInformation info;
283 auto videoTrack = VideoTrackPrivateMediaSourceAVFObjC::create(track);
284 info.track = videoTrack.copyRef();
285 info.description = MediaDescriptionAVFObjC::create(track);
286 segment.videoTracks.append(info);
287 } else if ([track hasMediaCharacteristic:AVMediaCharacteristicAudible]) {
288 SourceBufferPrivateClient::InitializationSegment::AudioTrackInformation info;
289 auto audioTrack = AudioTrackPrivateMediaSourceAVFObjC::create(track);
290 info.track = audioTrack.copyRef();
291 info.description = MediaDescriptionAVFObjC::create(track);
292 segment.audioTracks.append(info);
293 }
294
295 // FIXME(125161) : Add TextTrack support
296 }
297
298 m_didParseInitializationDataCallback(WTFMove(segment));
299 });
300}

302void SourceBufferParserAVFObjC::didFailToParseStreamDataWithError(NSError* error)
303{
304 m_callOnClientThreadCallback([this, strongThis = makeRef(*this), error = retainPtr(error)] {
305 if (m_didEncounterErrorDuringParsingCallback)
306 m_didEncounterErrorDuringParsingCallback(error.get().code);
307 });
308}

310void SourceBufferParserAVFObjC::didProvideMediaDataForTrackID(uint64_t trackID, CMSampleBufferRef sampleBuffer, const String& mediaType, unsigned flags)
311{
312 UNUSED_PARAM(flags);
313 m_callOnClientThreadCallback([this, strongThis = makeRef(*this), sampleBuffer = retainPtr(sampleBuffer), trackID, mediaType = mediaType] {
314 if (!m_didProvideMediaDataCallback)
315 return;
316
317 auto mediaSample = MediaSampleAVFObjC::create(sampleBuffer.get(), trackID);
318
319 if (mediaSample->isHomogeneous()) {
320 m_didProvideMediaDataCallback(WTFMove(mediaSample), trackID, mediaType);
321 return;
322 }
323
324 for (auto& sample : mediaSample->divideIntoHomogeneousSamples())
325 m_didProvideMediaDataCallback(WTFMove(sample), trackID, mediaType);
326 });
327}

334void SourceBufferParserAVFObjC::didProvideContentKeyRequestInitializationDataForTrackID(NSData* nsInitData, uint64_t trackID)
335{
336 auto initData = Uint8Array::create(nsInitData.length);
337 [nsInitData getBytes:initData->data() length:initData->length()];
338
339 m_didProvideContentKeyRequestInitializationDataForTrackIDCallback(WTFMove(initData), trackID);
340}
