104@implementation WebAVSampleBufferErrorListener

106- (id)initWithParent:(WeakPtr<WebCore::SourceBufferPrivateAVFObjC>&&)parent
107{
108 if (!(self = [super init]))
109 return nil;
110
111 _parent = WTFMove(parent);
112 return self;
113}
114
115- (void)dealloc
116{
117 [self invalidate];
118 [super dealloc];
119}
120
121- (void)invalidate
122{
123 if (!_parent && !_layers.size() && !_renderers.size())
124 return;
125
126 for (auto& layer : _layers) {
127 [layer removeObserver:self forKeyPath:@"error"];
128 [layer removeObserver:self forKeyPath:@"outputObscuredDueToInsufficientExternalProtection"];
129 }
130 _layers.clear();
131
132 for (auto& renderer : _renderers)
133 [renderer removeObserver:self forKeyPath:@"error"];
134 _renderers.clear();
135
136 [[NSNotificationCenter defaultCenter] removeObserver:self];
137
138 _parent = nullptr;
139}
140
141- (void)beginObservingLayer:(AVSampleBufferDisplayLayer*)layer
142{
143 ASSERT(_parent);
144 ASSERT(!_layers.contains(layer));
145
146 _layers.append(layer);
147 [layer addObserver:self forKeyPath:@"error" options:NSKeyValueObservingOptionNew context:nullptr];
148 [layer addObserver:self forKeyPath:@"outputObscuredDueToInsufficientExternalProtection" options:NSKeyValueObservingOptionNew context:nullptr];
149 [[NSNotificationCenter defaultCenter] addObserver:self selector:@selector(layerFailedToDecode:) name:AVSampleBufferDisplayLayerFailedToDecodeNotification object:layer];
150}
151
152- (void)stopObservingLayer:(AVSampleBufferDisplayLayer*)layer
153{
154 ASSERT(_parent);
155 ASSERT(_layers.contains(layer));
156
157 [layer removeObserver:self forKeyPath:@"error"];
158 [layer removeObserver:self forKeyPath:@"outputObscuredDueToInsufficientExternalProtection"];
159 _layers.remove(_layers.find(layer));
160
161 [[NSNotificationCenter defaultCenter] removeObserver:self name:AVSampleBufferDisplayLayerFailedToDecodeNotification object:layer];
162}
163
164ALLOW_NEW_API_WITHOUT_GUARDS_BEGIN
165- (void)beginObservingRenderer:(AVSampleBufferAudioRenderer*)renderer
166{
167ALLOW_NEW_API_WITHOUT_GUARDS_END
168 ASSERT(_parent);
169 ASSERT(!_renderers.contains(renderer));
170
171 _renderers.append(renderer);
172 [renderer addObserver:self forKeyPath:@"error" options:NSKeyValueObservingOptionNew context:nullptr];
173 [[NSNotificationCenter defaultCenter] addObserver:self selector:@selector(audioRendererWasAutomaticallyFlushed:) name:AVSampleBufferAudioRendererWasFlushedAutomaticallyNotification object:renderer];
174}
175
176ALLOW_NEW_API_WITHOUT_GUARDS_BEGIN
177- (void)stopObservingRenderer:(AVSampleBufferAudioRenderer*)renderer
178ALLOW_NEW_API_WITHOUT_GUARDS_END
179{
180 ASSERT(_parent);
181 ASSERT(_renderers.contains(renderer));
182
183 [renderer removeObserver:self forKeyPath:@"error"];
184 [[NSNotificationCenter defaultCenter] removeObserver:self name:AVSampleBufferAudioRendererWasFlushedAutomaticallyNotification object:renderer];
185
186 _renderers.remove(_renderers.find(renderer));
187}

189- (void)observeValueForKeyPath:(NSString *)keyPath ofObject:(id)object change:(NSDictionary *)change context:(void *)context
190{
191 UNUSED_PARAM(context);
192 UNUSED_PARAM(keyPath);
193 ASSERT(_parent);
194
195 if ([object isKindOfClass:PAL::getAVSampleBufferDisplayLayerClass()]) {
196 RetainPtr<AVSampleBufferDisplayLayer> layer = (AVSampleBufferDisplayLayer *)object;
197 ASSERT(_layers.contains(layer.get()));
198
199 if ([keyPath isEqualToString:@"error"]) {
200 RetainPtr<NSError> error = [change valueForKey:NSKeyValueChangeNewKey];
201 if ([error isKindOfClass:[NSNull class]])
202 return;
203
204 callOnMainThread([parent = _parent, layer = WTFMove(layer), error = WTFMove(error)] {
205 if (parent)
206 parent->layerDidReceiveError(layer.get(), error.get());
207 });
208 } else if ([keyPath isEqualToString:@"outputObscuredDueToInsufficientExternalProtection"]) {
209 callOnMainThread([parent = _parent, obscured = [[change valueForKey:NSKeyValueChangeNewKey] boolValue]] {
210 if (parent)
211 parent->outputObscuredDueToInsufficientExternalProtectionChanged(obscured);
212 });
213 } else
214 ASSERT_NOT_REACHED();
215
216 } else if ([object isKindOfClass:PAL::getAVSampleBufferAudioRendererClass()]) {
217 ALLOW_NEW_API_WITHOUT_GUARDS_BEGIN
218 RetainPtr<AVSampleBufferAudioRenderer> renderer = (AVSampleBufferAudioRenderer *)object;
219 ALLOW_NEW_API_WITHOUT_GUARDS_END
220 RetainPtr<NSError> error = [change valueForKey:NSKeyValueChangeNewKey];
221 if ([error isKindOfClass:[NSNull class]])
222 return;
223
224 ASSERT(_renderers.contains(renderer.get()));
225 ASSERT([keyPath isEqualToString:@"error"]);
226
227 callOnMainThread([parent = _parent, renderer = WTFMove(renderer), error = WTFMove(error)] {
228 if (parent)
229 parent->rendererDidReceiveError(renderer.get(), error.get());
230 });
231 } else
232 ASSERT_NOT_REACHED();
233}
234
235- (void)layerFailedToDecode:(NSNotification*)note
236{
237 RetainPtr<AVSampleBufferDisplayLayer> layer = (AVSampleBufferDisplayLayer *)[note object];
238 if (!_layers.contains(layer.get()))
239 return;
240
241 callOnMainThread([parent = _parent, layer = WTFMove(layer), error = retainPtr([[note userInfo] valueForKey:AVSampleBufferDisplayLayerFailedToDecodeNotificationErrorKey])] {
242 if (parent)
243 parent->layerDidReceiveError(layer.get(), error.get());
244 });
245}
246
247- (void)audioRendererWasAutomaticallyFlushed:(NSNotification*)note
248{
249 RetainPtr<AVSampleBufferAudioRenderer> renderer = (AVSampleBufferAudioRenderer *)[note object];
250 if (!_renderers.contains(renderer.get()))
251 return;
252
253 callOnMainThread([parent = _parent, renderer = WTFMove(renderer), flushTime = [[[note userInfo] valueForKey:AVSampleBufferAudioRendererFlushTimeKey] CMTimeValue]] {
254 if (parent)
255 parent->rendererWasAutomaticallyFlushed(renderer.get(), flushTime);
256 });
257}

277static void bufferWasConsumedCallback(CMNotificationCenterRef, const void* listener, CFStringRef notificationName, const void*, CFTypeRef)
278{
279 if (!CFEqual(PAL::kCMSampleBufferConsumerNotification_BufferConsumed, notificationName))
280 return;
281
282 if (!isMainThread()) {
283 callOnMainThread([notificationName, listener] {
284 bufferWasConsumedCallback(nullptr, listener, notificationName, nullptr, nullptr);
285 });
286 return;
287 }
288
289 uint64_t mapID = reinterpret_cast<uint64_t>(listener);
290 if (!mapID) {
291 RELEASE_LOG(MediaSource, "bufferWasConsumedCallback - ERROR: didn't find ID %llu in map", mapID);
292 return;
293 }
294
295 if (auto sourceBuffer = sourceBufferMap().get(mapID).get())
296 sourceBuffer->bufferWasConsumed();
297}

304SourceBufferPrivateAVFObjC::SourceBufferPrivateAVFObjC(MediaSourcePrivateAVFObjC* parent, Ref<SourceBufferParser>&& parser)
305 : m_parser(WTFMove(parser))
306 , m_errorListener(adoptNS([[WebAVSampleBufferErrorListener alloc] initWithParent:makeWeakPtr(*this)]))
307 , m_isAppendingGroup(adoptOSObject(dispatch_group_create()))
308 , m_mediaSource(parent)
309 , m_mapID(nextMapID())
310#if !RELEASE_LOG_DISABLED
311 , m_logger(parent->logger())
312 , m_logIdentifier(parent->nextSourceBufferLogIdentifier())
313#endif
314{
315 ALWAYS_LOG(LOGIDENTIFIER);
316
317 if (![PAL::getAVSampleBufferDisplayLayerClass() instancesRespondToSelector:@selector(prerollDecodeWithCompletionHandler:)])
318 PAL::CMNotificationCenterAddListener(PAL::CMNotificationCenterGetDefaultLocalCenter(), reinterpret_cast<void*>(m_mapID), bufferWasConsumedCallback, PAL::kCMSampleBufferConsumerNotification_BufferConsumed, nullptr, 0);
319
320#if !RELEASE_LOG_DISABLED
321 m_parser->setLogger(m_logger.get(), m_logIdentifier);
322#endif
323
324 sourceBufferMap().add(m_mapID, makeWeakPtr(*this));
325
326 m_parser->setDidParseInitializationDataCallback([weakThis = makeWeakPtr(this)] (InitializationSegment&& segment) mutable {
327 ASSERT(isMainThread());
328 if (weakThis)
329 weakThis->didParseInitializationData(WTFMove(segment));
330 });
331
332 m_parser->setDidEncounterErrorDuringParsingCallback([weakThis = makeWeakPtr(this)] (int32_t errorCode) mutable {
333 ASSERT(isMainThread());
334 if (weakThis)
335 weakThis->didEncounterErrorDuringParsing(errorCode);
336 });
337
338 m_parser->setDidProvideMediaDataCallback([weakThis = makeWeakPtr(this)] (Ref<MediaSample>&& sample, uint64_t trackId, const String& mediaType) mutable {
339 ASSERT(isMainThread());
340 if (weakThis)
341 weakThis->didProvideMediaDataForTrackId(WTFMove(sample), trackId, mediaType);
342 });
343}

345SourceBufferPrivateAVFObjC::~SourceBufferPrivateAVFObjC()
346{
347 ALWAYS_LOG(LOGIDENTIFIER);
348
349 ASSERT(!m_client);
350 sourceBufferMap().remove(m_mapID);
351 destroyStreamDataParser();

353 clearTracks();

357

361void SourceBufferPrivateAVFObjC::didParseInitializationData(InitializationSegment&& segment)
362{
363 ALWAYS_LOG(LOGIDENTIFIER);
364
365 if (!m_mediaSource) {
366 return;
367 }
368
369 if (auto player = this->player(); player && player->shouldCheckHardwareSupport()) {
370 for (auto& info : segment.videoTracks) {
371 auto codec = FourCC::fromString(info.description->codec());
372 if (!codec)
373 continue;
374 if (!codecsMeetHardwareDecodeRequirements({{ *codec }}, player->mediaContentTypesRequiringHardwareSupport())) {
375 m_parsingSucceeded = false;
376 return;
377 }
378 }
379 }
380
381 clearTracks();
382
383 for (auto videoTrackInfo : segment.videoTracks) {
384 videoTrackInfo.track->setSelectedChangedCallback([weakThis = makeWeakPtr(this)] (VideoTrackPrivate& track, bool selected) {
385 if (weakThis)
386 weakThis->trackDidChangeSelected(track, selected);
387 });
388 m_videoTracks.append(videoTrackInfo.track);
389 }
390
391 for (auto audioTrackInfo : segment.audioTracks) {
392 audioTrackInfo.track->setEnabledChangedCallback([weakThis = makeWeakPtr(this)] (AudioTrackPrivate& track, bool enabled) {
393 if (weakThis)
394 weakThis->trackDidChangeEnabled(track, enabled);
395 });
396 m_audioTracks.append(audioTrackInfo.track);
397 }
398
399 if (auto player = this->player())
400 player->characteristicsChanged();
401
402 m_initializationSegmentIsHandled = false;
403 didReceiveInitializationSegment(WTFMove(segment), [this, weakThis = makeWeakPtr(*this)]() mutable {
404 ASSERT(isMainThread());
405 if (!weakThis)
406 return;
407
408 if (auto player = this->player())
409 player->characteristicsChanged();
410
411 if (m_mediaSamples.isEmpty()) {
412 m_initializationSegmentIsHandled = true;
413 ALWAYS_LOG(LOGIDENTIFIER, "initialization segment is handled");
414 return;
415 }
416
417 callOnMainThread([this, weakThis = WTFMove(weakThis)] {
418 if (!weakThis)
419 return;
420
421 auto mediaSamples = std::exchange(m_mediaSamples, { });
422 for (auto& trackIdMediaSamplePair : mediaSamples) {
423 auto trackId = trackIdMediaSamplePair.first;
424 auto& mediaSample = trackIdMediaSamplePair.second;
425 if (trackId == m_enabledVideoTrackID || m_audioRenderers.contains(trackId)) {
426 DEBUG_LOG(LOGIDENTIFIER, mediaSample.get());
427 didReceiveSample(WTFMove(mediaSample));
428 }
429 }
430
431 m_initializationSegmentIsHandled = true;
432 ALWAYS_LOG(LOGIDENTIFIER, "initialization segment is handled");
433
434 if (m_hasPendingAppendCompletedCallback) {
435 m_hasPendingAppendCompletedCallback = false;
436 appendCompleted();
437 }
438 });
439 });
440}

452void SourceBufferPrivateAVFObjC::didProvideMediaDataForTrackId(Ref<MediaSample>&& mediaSample, uint64_t trackId, const String& mediaType)
453{
454 UNUSED_PARAM(mediaType);
455 if (!m_initializationSegmentIsHandled) {
456 DEBUG_LOG(LOGIDENTIFIER, mediaSample.get());
457 m_mediaSamples.append(std::make_pair(trackId, WTFMove(mediaSample)));
458 return;
459 }
460
461 if (trackId != m_enabledVideoTrackID && !m_audioRenderers.contains(trackId)) {
462 // FIXME(125161): We don't handle text tracks, and passing this sample up to SourceBuffer
463 // will just confuse its state. Drop this sample until we can handle text tracks properly.
464 return;
465 }
466
467 DEBUG_LOG(LOGIDENTIFIER, mediaSample.get());
468 didReceiveSample(WTFMove(mediaSample));
469}

471void SourceBufferPrivateAVFObjC::willProvideContentKeyRequestInitializationDataForTrackID(uint64_t trackID)
472{
473 if (!m_mediaSource)
474 return;
475
476#if HAVE(AVSTREAMSESSION) && ENABLE(LEGACY_ENCRYPTED_MEDIA)
477 ALWAYS_LOG(LOGIDENTIFIER, "track = ", trackID);
478
479 m_protectedTrackID = trackID;
480
481 auto parser = this->streamDataParser();
482 if (!parser)
483 return;
484
485 auto player = this->player();
486 if (!player)
487 return;
488
489 if (CDMSessionMediaSourceAVFObjC* session = player->cdmSession())
490 session->addParser(parser);
491 else if (!CDMSessionAVContentKeySession::isAvailable()) {
492 BEGIN_BLOCK_OBJC_EXCEPTIONS
493 [player->streamSession() addStreamDataParser:parser];
494 END_BLOCK_OBJC_EXCEPTIONS
495 }
496#else
497 UNUSED_PARAM(trackID);
498#endif
499}

501void SourceBufferPrivateAVFObjC::didProvideContentKeyRequestInitializationDataForTrackID(Ref<Uint8Array>&& initData, uint64_t trackID, Box<BinarySemaphore> hasSessionSemaphore)
502{
503 auto player = this->player();
504 if (!player)
505 return;
506
507#if HAVE(AVCONTENTKEYSESSION) && ENABLE(LEGACY_ENCRYPTED_MEDIA)
508 ALWAYS_LOG(LOGIDENTIFIER, "track = ", trackID);
509
510 m_protectedTrackID = trackID;
511 m_initData = WTFMove(initData);
512 m_mediaSource->sourceBufferKeyNeeded(this, m_initData.get());
513 if (auto session = player->cdmSession()) {
514 if (auto parser = this->streamDataParser())
515 session->addParser(parser);
516 hasSessionSemaphore->signal();
517 return;
518 }
519#endif
520
521 if (m_hasSessionSemaphore)
522 m_hasSessionSemaphore->signal();
523 m_hasSessionSemaphore = hasSessionSemaphore;
524
525#if ENABLE(ENCRYPTED_MEDIA) && HAVE(AVCONTENTKEYSESSION)
526 auto initDataBuffer = SharedBuffer::create(m_initData->data(), m_initData->byteLength());
527 auto keyIDs = CDMPrivateFairPlayStreaming::extractKeyIDsSinf(initDataBuffer);
528 if (!keyIDs)
529 return;
530
531 if (m_cdmInstance) {
532 if (auto instanceSession = m_cdmInstance->sessionForKeyIDs(keyIDs.value())) {
533 if (auto parser = this->streamDataParser())
534 [instanceSession->contentKeySession() addContentKeyRecipient:parser];
535 if (m_hasSessionSemaphore) {
536 m_hasSessionSemaphore->signal();
537 m_hasSessionSemaphore = nullptr;
538 }
539 m_waitingForKey = false;
540 return;
541 }
542 }
543
544 m_keyIDs = WTFMove(keyIDs.value());
545 player->initializationDataEncountered("sinf", initDataBuffer->tryCreateArrayBuffer());
546
547 m_waitingForKey = true;
548 player->waitingForKeyChanged();
549#endif
550
551 UNUSED_PARAM(initData);
552 UNUSED_PARAM(trackID);
553 UNUSED_PARAM(hasSessionSemaphore);
554}

558 static dispatch_queue_t globalQueue;
559 static dispatch_once_t onceToken;
560 dispatch_once(&onceToken, ^{
561 globalQueue = dispatch_queue_create("SourceBufferPrivateAVFObjC data parser queue", DISPATCH_QUEUE_CONCURRENT);

563 return globalQueue;

566void SourceBufferPrivateAVFObjC::append(Vector<unsigned char>&& data)
567{
568 ALWAYS_LOG(LOGIDENTIFIER, "data length = ", data.size());
569
570 // Queue a task to preserve the ordering of operations started by the
571 // abort process and ensure that the new appendBuffer will only
572 // ever deal with a sane parsing context.
573 callOnMainThread([weakThis = makeWeakPtr(*this), data = WTFMove(data), this]() mutable {
574 if (!weakThis)
575 return;
576
577 ASSERT(!m_hasSessionSemaphore);
578 ASSERT(!m_abortSemaphore);
579
580 if (m_client)
581 m_client->sourceBufferPrivateReportExtraMemoryCost(totalTrackBufferSizeInBytes());
582
583 m_abortSemaphore = Box<Semaphore>::create(0);
584 m_parser->setWillProvideContentKeyRequestInitializationDataForTrackIDCallback([weakThis, abortSemaphore = m_abortSemaphore](uint64_t trackID) mutable {
585 // We must call synchronously to the main thread, as the AVStreamSession must be associated
586 // with the streamDataParser before the delegate method returns.
587 Box<BinarySemaphore> respondedSemaphore = Box<BinarySemaphore>::create();
588 callOnMainThread([weakThis = WTFMove(weakThis), trackID, respondedSemaphore]() {
589 if (weakThis)
590 weakThis->willProvideContentKeyRequestInitializationDataForTrackID(trackID);
591 respondedSemaphore->signal();
592 });
593
594 while (true) {
595 if (respondedSemaphore->waitFor(100_ms))
596 return;
597
598 if (abortSemaphore->waitFor(100_ms)) {
599 abortSemaphore->signal();
600 return;
601 }
602 }
603 });
604
605 m_parser->setDidProvideContentKeyRequestInitializationDataForTrackIDCallback([weakThis, abortSemaphore = m_abortSemaphore](Ref<Uint8Array>&& initData, uint64_t trackID) mutable {
606 // Called on the data parser queue.
607 Box<BinarySemaphore> hasSessionSemaphore = Box<BinarySemaphore>::create();
608 callOnMainThread([weakThis = WTFMove(weakThis), initData = WTFMove(initData), trackID, hasSessionSemaphore]() mutable {
609 if (weakThis)
610 weakThis->didProvideContentKeyRequestInitializationDataForTrackID(WTFMove(initData), trackID, hasSessionSemaphore);
611 });
612
613 while (true) {
614 if (hasSessionSemaphore->waitFor(100_ms))
615 return;
616
617 if (abortSemaphore->waitFor(100_ms)) {
618 abortSemaphore->signal();
619 return;
620 }
621 }
622 });
623
624 m_parsingSucceeded = true;
625 dispatch_group_enter(m_isAppendingGroup.get());
626
627 dispatch_async(globalDataParserQueue(), [data = WTFMove(data), weakThis = m_appendWeakFactory.createWeakPtr(*this), parser = m_parser, isAppendingGroup = m_isAppendingGroup]() mutable {
628 parser->appendData(WTFMove(data), [weakThis = WTFMove(weakThis)]() mutable {
629 callOnMainThread([weakThis = WTFMove(weakThis)] {
630 if (!weakThis)
631 return;
632
633 if (!weakThis->m_mediaSamples.isEmpty()) {
634 weakThis->m_hasPendingAppendCompletedCallback = true;
635 return;
636 }
637
638 weakThis->appendCompleted();
639 });
640 });
641 dispatch_group_leave(isAppendingGroup.get());
642 });
643 });
644}

646void SourceBufferPrivateAVFObjC::appendCompleted()
647{
648 ALWAYS_LOG(LOGIDENTIFIER);
649 if (m_abortSemaphore) {
650 m_abortSemaphore->signal();
651 m_abortSemaphore = nil;
652 }
653
654 if (m_hasSessionSemaphore) {
655 m_hasSessionSemaphore->signal();
656 m_hasSessionSemaphore = nil;
657 }
658
659 if (m_abortCalled)
660 return;
661
662 if (auto player = this->player(); player && m_parsingSucceeded)
663 player->setLoadingProgresssed(true);
664
665 SourceBufferPrivate::appendCompleted(m_parsingSucceeded, m_mediaSource ? m_mediaSource->isEnded() : true);
666}

668void SourceBufferPrivateAVFObjC::removeCodedFrames(const MediaTime& start, const MediaTime& end, const MediaTime& currentMediaTime, bool isEnded, CompletionHandler<void()>&& completionHandler)
669{
670 // Queue a task to preserve the ordering of operations started by any
671 // earlier call to abort and that pending frames will be removed if needed.
672 callOnMainThread([weakThis = makeWeakPtr(*this), start, end, currentMediaTime, isEnded, completionHandler = WTFMove(completionHandler)]() mutable {
673 if (!weakThis) {
674 completionHandler();
675 return;
676 }
677 weakThis->SourceBufferPrivate::removeCodedFrames(start, end, currentMediaTime, isEnded, WTFMove(completionHandler));
678 });
679}
680
681// The MSE spec requires that we abort the current buffer append algorithm
682// https://w3c.github.io/media-source/#dfn-buffer-append
683// which is then followed by a call to resetParserState
684// as per https://w3c.github.io/media-source/#dom-sourcebuffer-abort
685// However due to our asynchronous design this causes inherent difficulties.
686// The SourceBuffe's abortIfUpdating method would have already cancelled any
687// pending update not yet despatched by its timer.
688// As the spec behaviour is non deterministic anyway, we instead process all
689// pending frames found in the input buffer.
690void SourceBufferPrivateAVFObjC::abort()
691{
692 ALWAYS_LOG(LOGIDENTIFIER);
693
694 // Queue a task to preserve the ordering of operations started by any
695 // earlier call to abort.
696 callOnMainThread([weakThis = makeWeakPtr(*this), this]() {
697 if (!weakThis)
698 return;
699 ASSERT(!m_abortCalled, "Abort should only be called if we were currently updating, resetParserState must have been called in between");
700 m_abortCalled = true;
701 });
702}
703
704void SourceBufferPrivateAVFObjC::resetParserState()
705{
706 ALWAYS_LOG(LOGIDENTIFIER);
707
708 // The parsing queue may be blocked waiting for the main thread to provide it an AVStreamSession. We
709 // were asked to abort, and that cancels all outstanding append operations. Without cancelling this
710 // semaphore, the m_isAppendingGroup wait operation will deadlock.
711 if (m_hasSessionSemaphore) {
712 m_hasSessionSemaphore->signal();
713 m_hasSessionSemaphore = nullptr;
714 }
715 if (m_abortSemaphore) {
716 m_abortSemaphore->signal();
717 m_abortSemaphore = nullptr;
718 }
719
720 // Wait for any pending parsing to complete.
721 dispatch_group_wait(m_isAppendingGroup.get(), DISPATCH_TIME_FOREVER);
722
723 // Dispatch a task to complete all tasks that may have been queued by the
724 // appending group get to run first.
725 // We keep a strong reference to the parser so that it can't be destructed
726 // and can perform all required cleaning operations at the end of all operations.
727 callOnMainThread([weakThis = makeWeakPtr(*this), parser = m_parser, this]() {
728 parser->resetParserState();
729 if (!weakThis)
730 return;
731 ASSERT(m_mediaSamples.isEmpty(), "All pending frames should have been processed");
732 m_abortCalled = false;
733 });
734}

1136void SourceBufferPrivateAVFObjC::enqueueSample(Ref<MediaSample>&& sample, const AtomString& trackIDString)
1137{
1138 auto trackID = parseIntegerAllowingTrailingJunk<uint64_t>(trackIDString).value_or(0);
1139 if (trackID != m_enabledVideoTrackID && !m_audioRenderers.contains(trackID))
1140 return;
1141
1142 PlatformSample platformSample = sample->platformSample();
1143 if (platformSample.type != PlatformSample::CMSampleBufferType)
1144 return;
1145
1146 auto logSiteIdentifier = LOGIDENTIFIER;
1147 DEBUG_LOG(logSiteIdentifier, "track ID = ", trackID, ", sample = ", sample.get());
1148
1149 CMFormatDescriptionRef formatDescription = PAL::CMSampleBufferGetFormatDescription(platformSample.sample.cmSampleBuffer);
1150 ASSERT(formatDescription);
1151 if (!formatDescription) {
1152 ERROR_LOG(logSiteIdentifier, "Received sample with a null formatDescription. Bailing.");
1153 return;
1154 }
1155 auto mediaType = PAL::CMFormatDescriptionGetMediaType(formatDescription);
1156
1157 if (trackID == m_enabledVideoTrackID) {
1158 // AVSampleBufferDisplayLayer will throw an un-documented exception if passed a sample
1159 // whose media type is not kCMMediaType_Video. This condition is exceptional; we should
1160 // never enqueue a non-video sample in a AVSampleBufferDisplayLayer.
1161 ASSERT(mediaType == kCMMediaType_Video);
1162 if (mediaType != kCMMediaType_Video) {
1163 ERROR_LOG(logSiteIdentifier, "Expected sample of type '", FourCC(kCMMediaType_Video), "', got '", FourCC(mediaType), "'. Bailing.");
1164 return;
1165 }
1166
1167 auto player = this->player();
1168 FloatSize formatSize = FloatSize(PAL::CMVideoFormatDescriptionGetPresentationDimensions(formatDescription, true, true));
1169 if (!m_cachedSize || formatSize != m_cachedSize.value()) {
1170 DEBUG_LOG(logSiteIdentifier, "size changed to ", formatSize);
1171 bool sizeWasNull = !m_cachedSize;
1172 m_cachedSize = formatSize;
1173 if (player) {
1174 if (sizeWasNull)
1175 player->setNaturalSize(formatSize);
1176 else
1177 player->sizeWillChangeAtTime(sample->presentationTime(), formatSize);
1178 }
1179 }
1180
1181 if (m_decompressionSession)
1182 m_decompressionSession->enqueueSample(platformSample.sample.cmSampleBuffer);
1183
1184 if (!m_displayLayer)
1185 return;
1186
1187 if (player && !player->hasAvailableVideoFrame() && !sample->isNonDisplaying()) {
1188 DEBUG_LOG(logSiteIdentifier, "adding buffer attachment");
1189
1190 bool havePrerollDecodeWithCompletionHandler = [PAL::getAVSampleBufferDisplayLayerClass() instancesRespondToSelector:@selector(prerollDecodeWithCompletionHandler:)];
1191
1192 if (!havePrerollDecodeWithCompletionHandler) {
1193 CMSampleBufferRef rawSampleCopy;
1194 PAL::CMSampleBufferCreateCopy(kCFAllocatorDefault, platformSample.sample.cmSampleBuffer, &rawSampleCopy);
1195 auto sampleCopy = adoptCF(rawSampleCopy);
1196 PAL::CMSetAttachment(sampleCopy.get(), PAL::kCMSampleBufferAttachmentKey_PostNotificationWhenConsumed, (__bridge CFDictionaryRef)@{ (__bridge NSString *)PAL::kCMSampleBufferAttachmentKey_PostNotificationWhenConsumed : @YES }, kCMAttachmentMode_ShouldNotPropagate);
1197 [m_displayLayer enqueueSampleBuffer:sampleCopy.get()];
1198#if PLATFORM(IOS_FAMILY)
1199 player->setHasAvailableVideoFrame(true);
1200#endif
1201 } else {
1202
1203 [m_displayLayer enqueueSampleBuffer:platformSample.sample.cmSampleBuffer];
1204 [m_displayLayer prerollDecodeWithCompletionHandler:[this, logSiteIdentifier, weakThis = makeWeakPtr(*this)] (BOOL success) mutable {
1205 if (!weakThis)
1206 return;
1207
1208 if (!success) {
1209 ERROR_LOG(logSiteIdentifier, "prerollDecodeWithCompletionHandler failed");
1210 return;
1211 }
1212
1213 callOnMainThread([weakThis = WTFMove(weakThis)] () mutable {
1214 if (!weakThis)
1215 return;
1216
1217 weakThis->bufferWasConsumed();
1218 });
1219 }];
1220 }
1221 } else
1222 [m_displayLayer enqueueSampleBuffer:platformSample.sample.cmSampleBuffer];
1223
1224 } else {
1225 // AVSampleBufferAudioRenderer will throw an un-documented exception if passed a sample
1226 // whose media type is not kCMMediaType_Audio. This condition is exceptional; we should
1227 // never enqueue a non-video sample in a AVSampleBufferAudioRenderer.
1228 ASSERT(mediaType == kCMMediaType_Audio);
1229 if (mediaType != kCMMediaType_Audio) {
1230 ERROR_LOG(logSiteIdentifier, "Expected sample of type '", FourCC(kCMMediaType_Audio), "', got '", FourCC(mediaType), "'. Bailing.");
1231 return;
1232 }
1233
1234 auto renderer = m_audioRenderers.get(trackID);
1235 [renderer enqueueSampleBuffer:platformSample.sample.cmSampleBuffer];
1236 if (auto player = this->player(); player && !sample->isNonDisplaying())
1237 player->setHasAvailableAudioSample(renderer.get(), true);
1238 }
1239}

1400 ASSERT(!layer || !m_decompressionSession || hasSelectedVideo());
