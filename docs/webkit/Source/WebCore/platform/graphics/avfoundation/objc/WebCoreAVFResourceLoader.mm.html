71std::unique_ptr<CachedResourceMediaLoader> CachedResourceMediaLoader::create(WebCoreAVFResourceLoader& parent, CachedResourceLoader& loader, ResourceRequest&& resourceRequest)
72{
73 // FIXME: Skip Content Security Policy check if the element that inititated this request
74 // is in a user-agent shadow tree. See <https://bugs.webkit.org/show_bug.cgi?id=173498>.
75 ResourceLoaderOptions loaderOptions(
76 SendCallbackPolicy::SendCallbacks,
77 ContentSniffingPolicy::DoNotSniffContent,
78 DataBufferingPolicy::BufferData,
79 StoredCredentialsPolicy::DoNotUse,
80 ClientCredentialPolicy::CannotAskClientForCredentials,
81 FetchOptions::Credentials::Omit,
82 SecurityCheckPolicy::DoSecurityCheck,
83 FetchOptions::Mode::NoCors,
84 CertificateInfoPolicy::DoNotIncludeCertificateInfo,
85 ContentSecurityPolicyImposition::DoPolicyCheck,
86 DefersLoadingPolicy::AllowDefersLoading,
87 CachingPolicy::DisallowCaching
88 );
89 loaderOptions.destination = FetchOptions::Destination::Video;
90 CachedResourceRequest request(WTFMove(resourceRequest), loaderOptions);
91
92 auto resource = loader.requestMedia(WTFMove(request)).value_or(nullptr);
93 if (!resource)
94 return nullptr;
95 return std::unique_ptr<CachedResourceMediaLoader>(new CachedResourceMediaLoader { parent, WTFMove(resource) });
96}

114void CachedResourceMediaLoader::responseReceived(CachedResource& resource, const ResourceResponse& response, CompletionHandler<void()>&& completionHandler)

122void CachedResourceMediaLoader::notifyFinished(CachedResource& resource, const NetworkLoadMetrics&)
123{
124 if (resource.loadFailedOrCanceled()) {
125 m_parent.loadFailed(resource.resourceError());
126 return;
127 }
128 m_parent.loadFinished();
129}

131void CachedResourceMediaLoader::dataReceived(CachedResource& resource, const uint8_t*, int)
132{
133 ASSERT(&resource == m_resource);
134 if (auto* data = resource.resourceBuffer())
135 m_parent.newDataStoredInSharedBuffer(*data);

167WeakPtr<PlatformResourceMediaLoader> PlatformResourceMediaLoader::create(WebCoreAVFResourceLoader& parent, PlatformMediaResourceLoader& loader, ResourceRequest&& request)
168{
169 auto resource = loader.requestResource(WTFMove(request), PlatformMediaResourceLoader::LoadOption::DisallowCaching);
170 if (!resource)
171 return nullptr;
172 auto* resourcePointer = resource.get();
173 auto client = adoptRef(*new PlatformResourceMediaLoader { parent, resource.releaseNonNull() });
174 auto result = makeWeakPtr(client.get());
175
176 resourcePointer->setClient(WTFMove(client));
177 return result;
178}

239void WebCoreAVFResourceLoader::startLoading()
240{
241 if (m_resourceMediaLoader || m_platformMediaLoader || !m_parent)
242 return;
243
244 NSURLRequest *nsRequest = [m_avRequest.get() request];
245
246 ResourceRequest request(nsRequest);
247 request.setPriority(ResourceLoadPriority::Low);
248
249 if (AVAssetResourceLoadingDataRequest *dataRequest = [m_avRequest dataRequest]; dataRequest.requestedLength
250 && !request.hasHTTPHeaderField(HTTPHeaderName::Range)
251 && !request.url().protocolIsBlob()) {
252 String rangeEnd = dataRequest.requestsAllDataToEndOfResource ? "*"_s : makeString(dataRequest.requestedOffset + dataRequest.requestedLength - 1);
253 request.addHTTPHeaderField(HTTPHeaderName::Range, makeString("bytes=", dataRequest.requestedOffset, '-', rangeEnd));
254 }
255
256 if (auto* loader = m_parent->player()->cachedResourceLoader()) {
257 m_resourceMediaLoader = CachedResourceMediaLoader::create(*this, *loader, ResourceRequest(request));
258 if (m_resourceMediaLoader)
259 return;
260 }
261
262 if (auto loader = m_parent->player()->createResourceLoader()) {
263 m_platformMediaLoader = PlatformResourceMediaLoader::create(*this, *loader, WTFMove(request));
264 if (m_platformMediaLoader)
265 return;
266 }
267
268 LOG_ERROR("Failed to start load for media at url %s", [[[nsRequest URL] absoluteString] UTF8String]);
269 [m_avRequest.get() finishLoadingWithError:0];
270}
271
272void WebCoreAVFResourceLoader::stopLoading()
273{
274 m_resourceMediaLoader = nullptr;
275
276 if (m_platformMediaLoader) {
277 m_platformMediaLoader->stop();
278 m_platformMediaLoader = nullptr;
279 }
280 if (m_parent && m_avRequest)
281 m_parent->didStopLoadingRequest(m_avRequest.get());
282}

284void WebCoreAVFResourceLoader::invalidate()
285{
286 if (!m_parent)
287 return;
288
289 m_parent = nullptr;
290
291 callOnMainThread([protectedThis = makeRef(*this)] () mutable {
292 protectedThis->stopLoading();
293 });
294}

296void WebCoreAVFResourceLoader::responseReceived(const ResourceResponse& response)
297{
298 int status = response.httpStatusCode();
299 if (status && (status < 200 || status > 299)) {
300 [m_avRequest.get() finishLoadingWithError:0];
301 return;
302 }
303
304 auto& contentRange = response.contentRange();
305 if (contentRange.isValid())
306 m_responseOffset = static_cast<NSUInteger>(contentRange.firstBytePosition());
307
308 if (AVAssetResourceLoadingContentInformationRequest* contentInfo = [m_avRequest.get() contentInformationRequest]) {
309 String uti = UTIFromMIMEType(response.mimeType());
310
311 [contentInfo setContentType:uti];
312
313 [contentInfo setContentLength:contentRange.isValid() ? contentRange.instanceLength() : response.expectedContentLength()];
314 [contentInfo setByteRangeAccessSupported:YES];
315
316 if ([contentInfo respondsToSelector:@selector(setEntireLengthAvailableOnDemand:)])
317 [contentInfo setEntireLengthAvailableOnDemand:YES];
318
319 if (![m_avRequest dataRequest]) {
320 [m_avRequest.get() finishLoading];
321 stopLoading();
322 }
323 }

326void WebCoreAVFResourceLoader::loadFailed(const ResourceError& error)
327{
328 // <rdar://problem/13987417> Set the contentType of the contentInformationRequest to an empty
329 // string to trigger AVAsset's playable value to complete loading.
330 if ([m_avRequest.get() contentInformationRequest] && ![[m_avRequest.get() contentInformationRequest] contentType])
331 [[m_avRequest.get() contentInformationRequest] setContentType:@""];
332
333 [m_avRequest.get() finishLoadingWithError:error.nsError()];
334 stopLoading();
335}
336
337void WebCoreAVFResourceLoader::loadFinished()
338{
339 [m_avRequest.get() finishLoading];
340 stopLoading();
341}

343void WebCoreAVFResourceLoader::newDataStoredInSharedBuffer(SharedBuffer& data)
344{
345 AVAssetResourceLoadingDataRequest* dataRequest = [m_avRequest dataRequest];
346 if (!dataRequest)
347 return;
348
349 // Check for possible unsigned overflow.
350 ASSERT(dataRequest.currentOffset >= dataRequest.requestedOffset);
351 ASSERT(dataRequest.requestedLength >= (dataRequest.currentOffset - dataRequest.requestedOffset));
352
353 NSUInteger remainingLength = dataRequest.requestedLength - static_cast<NSUInteger>(dataRequest.currentOffset - dataRequest.requestedOffset);
354
355 auto bytesToSkip = dataRequest.currentOffset - m_responseOffset;
356 auto array = data.createNSDataArray();
357 for (NSData *segment in array.get()) {
358 if (bytesToSkip) {
359 if (bytesToSkip > segment.length) {
360 bytesToSkip -= segment.length;
361 continue;
362 }
363 auto bytesToUse = segment.length - bytesToSkip;
364 [dataRequest respondWithData:[segment subdataWithRange:NSMakeRange(static_cast<NSUInteger>(bytesToSkip), static_cast<NSUInteger>(segment.length - bytesToSkip))]];
365 bytesToSkip = 0;
366 remainingLength -= bytesToUse;
367 continue;
368 }
369 if (segment.length <= remainingLength) {
370 [dataRequest respondWithData:segment];
371 remainingLength -= segment.length;
372 continue;
373 }
374 [dataRequest respondWithData:[segment subdataWithRange:NSMakeRange(0, remainingLength)]];
375 remainingLength = 0;
376 break;
377 }
378
379 // There was not enough data in the buffer to satisfy the data request.
380 if (remainingLength)
381 return;
382
383 if (dataRequest.currentOffset + dataRequest.requestedLength >= dataRequest.requestedOffset) {
384 [m_avRequest.get() finishLoading];
385 stopLoading();
386 }
387}
