27#define LayerPool_h
28
29#include "IntSize.h"
30#include "IntSizeHash.h"
31#include "PlatformCALayer.h"
32#include "Timer.h"
33#include <wtf/Deque.h>
34#include <wtf/HashMap.h>
35#include <wtf/RetainPtr.h>
36#include <wtf/Vector.h>
37
38namespace WebCore {
39
40class LayerPool {
41 WTF_MAKE_NONCOPYABLE(LayerPool);
42public:
43 WEBCORE_EXPORT LayerPool();
44 WEBCORE_EXPORT ~LayerPool();
45
46 static HashSet<LayerPool*>& allLayerPools();
47
48 void addLayer(const RefPtr<PlatformCALayer>&);
49 RefPtr<PlatformCALayer> takeLayerWithSize(const IntSize&);
50
51 void drain();
52
53 // The maximum size of all queued layers in bytes.
54 unsigned capacity() const { return m_maxBytesForPool; }
55
56private:
57 typedef Deque<RefPtr<PlatformCALayer>> LayerList;
58
59 unsigned decayedCapacity() const;
60
61 bool canReuseLayerWithSize(const IntSize& size) const { return m_maxBytesForPool && !size.isEmpty(); }
62 void schedulePrune();
63 void pruneTimerFired();
64
65 typedef enum { LeaveUnchanged, MarkAsUsed } AccessType;
66 LayerList& listOfLayersWithSize(const IntSize&, AccessType = LeaveUnchanged);
67
68 static unsigned backingStoreBytesForSize(const IntSize&);
69
70 HashMap<IntSize, LayerList> m_reuseLists;
71 // Ordered by recent use. The last size is the most recently used.
72 Vector<IntSize> m_sizesInPruneOrder;
73 unsigned m_totalBytes;
74 unsigned m_maxBytesForPool;
75
76 Timer m_pruneTimer;
77
78 MonotonicTime m_lastAddTime;
79};

81}
