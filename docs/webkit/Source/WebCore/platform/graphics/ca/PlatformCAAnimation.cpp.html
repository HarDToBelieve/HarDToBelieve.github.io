26#include "config.h"
27#include "PlatformCAAnimation.h"
28
29#include <wtf/text/TextStream.h>
30
31namespace WebCore {
32
33TextStream& operator<<(TextStream& ts, PlatformCAAnimation::AnimationType type)
34{
35 switch (type) {
36 case PlatformCAAnimation::Basic: ts << "basic"; break;
37 case PlatformCAAnimation::Group: ts << "group"; break;
38 case PlatformCAAnimation::Keyframe: ts << "keyframe"; break;
39 case PlatformCAAnimation::Spring: ts << "spring"; break;
40 }
41 return ts;
42}
43
44TextStream& operator<<(TextStream& ts, PlatformCAAnimation::FillModeType fillMode)
45{
46 switch (fillMode) {
47 case PlatformCAAnimation::NoFillMode: ts << "none"; break;
48 case PlatformCAAnimation::Forwards: ts << "forwards"; break;
49 case PlatformCAAnimation::Backwards: ts << "backwards"; break;
50 case PlatformCAAnimation::Both: ts << "both"; break;
51 }
52 return ts;
53}
54
55TextStream& operator<<(TextStream& ts, PlatformCAAnimation::ValueFunctionType valueFunctionType)
56{
57 switch (valueFunctionType) {
58 case PlatformCAAnimation::NoValueFunction: ts << "none"; break;
59 case PlatformCAAnimation::RotateX: ts << "rotateX"; break;
60 case PlatformCAAnimation::RotateY: ts << "rotateY"; break;
61 case PlatformCAAnimation::RotateZ: ts << "rotateX"; break;
62 case PlatformCAAnimation::ScaleX: ts << "scaleX"; break;
63 case PlatformCAAnimation::ScaleY: ts << "scaleY"; break;
64 case PlatformCAAnimation::ScaleZ: ts << "scaleX"; break;
65 case PlatformCAAnimation::Scale: ts << "scale"; break;
66 case PlatformCAAnimation::TranslateX: ts << "translateX"; break;
67 case PlatformCAAnimation::TranslateY: ts << "translateY"; break;
68 case PlatformCAAnimation::TranslateZ: ts << "translateZ"; break;
69 case PlatformCAAnimation::Translate: ts << "translate"; break;
70 }
71 return ts;
72}
73
74bool PlatformCAAnimation::isBasicAnimation() const
75{
76 return animationType() == Basic || animationType() == Spring;
77}
