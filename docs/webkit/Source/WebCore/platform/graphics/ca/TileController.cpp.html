106void TileController::setContentsScale(float scale)
107{
108 ASSERT(owningGraphicsLayer()->isCommittingChanges());
109
110 float deviceScaleFactor = owningGraphicsLayer()->platformCALayerDeviceScaleFactor();
111 // The scale we get is the product of the page scale factor and device scale factor.
112 // Divide by the device scale factor so we'll get the page scale factor.
113 scale /= deviceScaleFactor;
114
115 if (tileGrid().scale() == scale && m_deviceScaleFactor == deviceScaleFactor && !m_hasTilesWithTemporaryScaleFactor)
116 return;
117
118 m_hasTilesWithTemporaryScaleFactor = false;
119 m_deviceScaleFactor = deviceScaleFactor;
120
121 if (m_coverageMap)
122 m_coverageMap->setDeviceScaleFactor(deviceScaleFactor);
123
124 if (m_zoomedOutTileGrid && m_zoomedOutTileGrid->scale() == scale) {
125 m_tileGrid = WTFMove(m_zoomedOutTileGrid);
126 m_tileGrid->setIsZoomedOutTileGrid(false);
127 m_tileGrid->revalidateTiles();
128 tileGridsChanged();
129 return;
130 }
131
132 if (m_zoomedOutContentsScale && m_zoomedOutContentsScale == tileGrid().scale() && tileGrid().scale() != scale && !m_hasTilesWithTemporaryScaleFactor) {
133 m_zoomedOutTileGrid = WTFMove(m_tileGrid);
134 m_zoomedOutTileGrid->setIsZoomedOutTileGrid(true);
135 m_tileGrid = makeUnique<TileGrid>(*this);
136 tileGridsChanged();
137 }
138
139 tileGrid().setScale(scale);
140 tileGrid().setNeedsDisplay();
141}

340IntRect TileController::boundsForSize(const FloatSize& size) const
341{
342 IntPoint boundsOriginIncludingMargin(-leftMarginWidth(), -topMarginHeight());
343 IntSize boundsSizeIncludingMargin = expandedIntSize(size);
344 boundsSizeIncludingMargin.expand(leftMarginWidth() + rightMarginWidth(), topMarginHeight() + bottomMarginHeight());

346 return IntRect(boundsOriginIncludingMargin, boundsSizeIncludingMargin);

366FloatRect TileController::adjustTileCoverageRect(const FloatRect& coverageRect, const FloatRect& previousVisibleRect, const FloatRect& currentVisibleRect, bool sizeChanged)
367{
368 if (sizeChanged || MemoryPressureHandler::singleton().isUnderMemoryPressure())

374#if !PLATFORM(IOS_FAMILY)
375// Coverage expansion for less memory-constrained devices.
376// Kept separate to preserve historical behavior; should be merged with adjustTileCoverageWithScrollingVelocity eventually.
377FloatRect TileController::adjustTileCoverageForDesktopPageScrolling(const FloatRect& coverageRect, const FloatSize& newSize, const FloatRect& previousVisibleRect, const FloatRect& visibleRect) const
378{
379 // FIXME: look at how far the document can scroll in each dimension.
380 FloatSize coverageSize = visibleRect.size();
381
382 bool largeVisibleRectChange = !previousVisibleRect.isEmpty() && !visibleRect.intersects(previousVisibleRect);
383
384 // Inflate the coverage rect so that it covers 2x of the visible width and 3x of the visible height.
385 // These values were chosen because it's more common to have tall pages and to scroll vertically,
386 // so we keep more tiles above and below the current area.
387 float widthScale = 1;
388 float heightScale = 1;
389
390 if (m_tileCoverage & CoverageForHorizontalScrolling && !largeVisibleRectChange)
391 widthScale = 2;
392
393 if (m_tileCoverage & CoverageForVerticalScrolling && !largeVisibleRectChange)
394 heightScale = 3;
395
396 coverageSize.scale(widthScale, heightScale);
397
398 FloatRect coverageBounds = boundsForSize(newSize);
399
400 // Return 'rect' padded evenly on all sides to achieve 'newSize', but make the padding uneven to contain within constrainingRect.
401 auto expandRectWithinRect = [](const FloatRect& rect, const FloatSize& newSize, const FloatRect& constrainingRect) {
402 ASSERT(newSize.width() >= rect.width() && newSize.height() >= rect.height());
403
404 FloatSize extraSize = newSize - rect.size();
405
406 FloatRect expandedRect = rect;
407 expandedRect.inflateX(extraSize.width() / 2);
408 expandedRect.inflateY(extraSize.height() / 2);
409
410 if (expandedRect.x() < constrainingRect.x())
411 expandedRect.setX(constrainingRect.x());
412 else if (expandedRect.maxX() > constrainingRect.maxX())
413 expandedRect.setX(constrainingRect.maxX() - expandedRect.width());
414
415 if (expandedRect.y() < constrainingRect.y())
416 expandedRect.setY(constrainingRect.y());
417 else if (expandedRect.maxY() > constrainingRect.maxY())
418 expandedRect.setY(constrainingRect.maxY() - expandedRect.height());
419
420 return intersection(expandedRect, constrainingRect);
421 };
422
423 FloatRect coverage = expandRectWithinRect(visibleRect, coverageSize, coverageBounds);
424 LOG_WITH_STREAM(Tiling, stream << "TileController::adjustTileCoverageForDesktopPageScrolling newSize=" << newSize << " mode " << m_tileCoverage << " expanded to " << coverageSize << " bounds with margin " << coverageBounds << " coverage " << coverage);
425 return unionRect(coverageRect, coverage);
426}
427#endif

429FloatRect TileController::adjustTileCoverageWithScrollingVelocity(const FloatRect& coverageRect, const FloatSize& newSize, const FloatRect& visibleRect, float contentsScale) const
430{
431 if (m_tileCoverage == CoverageForVisibleArea || MemoryPressureHandler::singleton().isUnderMemoryPressure())
432 return visibleRect;
433
434 double horizontalMargin = kDefaultTileSize / contentsScale;
435 double verticalMargin = kDefaultTileSize / contentsScale;
436
437 MonotonicTime currentTime = MonotonicTime::now();
438 Seconds timeDelta = currentTime - m_velocity.lastUpdateTime;
439
440 FloatRect futureRect = visibleRect;
441 futureRect.setLocation(FloatPoint(
442 futureRect.location().x() + timeDelta.value() * m_velocity.horizontalVelocity,
443 futureRect.location().y() + timeDelta.value() * m_velocity.verticalVelocity));
444
445 if (m_velocity.horizontalVelocity) {
446 futureRect.setWidth(futureRect.width() + horizontalMargin);
447 if (m_velocity.horizontalVelocity < 0)
448 futureRect.setX(futureRect.x() - horizontalMargin);
449 }
450
451 if (m_velocity.verticalVelocity) {
452 futureRect.setHeight(futureRect.height() + verticalMargin);
453 if (m_velocity.verticalVelocity < 0)
454 futureRect.setY(futureRect.y() - verticalMargin);
455 }
456
457 if (!m_velocity.horizontalVelocity && !m_velocity.verticalVelocity) {
458 if (m_velocity.scaleChangeRate > 0) {
459 LOG_WITH_STREAM(Tiling, stream << "TileController " << this << " computeTileCoverageRect - zooming, coverage is visible rect " << coverageRect);
460 return visibleRect;
461 }
462 futureRect.setWidth(futureRect.width() + horizontalMargin);
463 futureRect.setHeight(futureRect.height() + verticalMargin);
464 futureRect.setX(futureRect.x() - horizontalMargin / 2);
465 futureRect.setY(futureRect.y() - verticalMargin / 2);
466 }
467
468 // Can't use m_tileCacheLayer->bounds() here, because the size of the underlying platform layer
469 // hasn't been updated for the current commit.
470 IntSize contentSize = expandedIntSize(newSize);
471 if (futureRect.maxX() > contentSize.width())
472 futureRect.setX(contentSize.width() - futureRect.width());
473 if (futureRect.maxY() > contentSize.height())
474 futureRect.setY(contentSize.height() - futureRect.height());
475 if (futureRect.x() < 0)
476 futureRect.setX(0);
477 if (futureRect.y() < 0)
478 futureRect.setY(0);
479
480 LOG_WITH_STREAM(Tiling, stream << "TileController " << this << " adjustTileCoverageForScrolling - coverage " << coverageRect << " expanded to " << unionRect(coverageRect, futureRect) << " velocity " << m_velocity);
481
482 return unionRect(coverageRect, futureRect);
483}

485FloatRect TileController::adjustTileCoverageRectForScrolling(const FloatRect& coverageRect, const FloatSize& newSize, const FloatRect& previousVisibleRect, const FloatRect& visibleRect, float contentsScale)
486{
487 // If the page is not in a window (for example if it's in a background tab), we limit the tile coverage rect to the visible rect.
488 if (!m_isInWindow)
489 return visibleRect;
490
491#if !PLATFORM(IOS_FAMILY)
492 if (m_tileCacheLayer->isPageTiledBackingLayer())
493 return adjustTileCoverageForDesktopPageScrolling(coverageRect, newSize, previousVisibleRect, visibleRect);
494#else
495 UNUSED_PARAM(previousVisibleRect);
496#endif
497
498 auto computeVelocityIfNecessary = [&](FloatPoint scrollOffset) {
499 if (m_haveExternalVelocityData)
500 return;
501
502 if (!m_historicalVelocityData)
503 m_historicalVelocityData = makeUnique<HistoricalVelocityData>();
504
505 m_velocity = m_historicalVelocityData->velocityForNewData(scrollOffset, contentsScale, MonotonicTime::now());
506 };
507
508 computeVelocityIfNecessary(visibleRect.location());
509
510 return adjustTileCoverageWithScrollingVelocity(coverageRect, newSize, visibleRect, contentsScale);
511}

564IntSize TileController::computeTileSize()
565{
566 if (m_inLiveResize || m_tileSizeLocked)
567 return tileGrid().tileSize();
568
569 const int kLowestCommonDenominatorMaxTileSize = 4 * 1024;
570 IntSize maxTileSize(kLowestCommonDenominatorMaxTileSize, kLowestCommonDenominatorMaxTileSize);
571
572#if HAVE(IOSURFACE)
573 IntSize surfaceSizeLimit = IOSurface::maximumSize();
574 surfaceSizeLimit.scale(1 / m_deviceScaleFactor);
575 maxTileSize = maxTileSize.shrunkTo(surfaceSizeLimit);
576#endif
577
578 if (owningGraphicsLayer()->platformCALayerUseGiantTiles())
579 return maxTileSize;
580
581 IntSize tileSize(kDefaultTileSize, kDefaultTileSize);
582
583 if (m_scrollability == NotScrollable) {
584 IntSize scaledSize = expandedIntSize(boundsWithoutMargin().size() * tileGrid().scale());
585 tileSize = scaledSize.constrainedBetween(IntSize(kDefaultTileSize, kDefaultTileSize), maxTileSize);
586 } else if (m_scrollability == VerticallyScrollable)
587 tileSize.setWidth(std::min(std::max<int>(ceilf(boundsWithoutMargin().width() * tileGrid().scale()), kDefaultTileSize), maxTileSize.width()));
588
589 LOG_WITH_STREAM(Scrolling, stream << "TileController::tileSize newSize=" << tileSize);
590
591 m_tileSizeLocked = true;
592 return tileSize;
593}

607{

635unsigned TileController::blankPixelCountForTiles(const PlatformLayerList& tiles, const FloatRect& visibleRect, const IntPoint& tileTranslation)
636{
637 Region paintedVisibleTiles;
638
639 for (PlatformLayerList::const_iterator it = tiles.begin(), end = tiles.end(); it != end; ++it) {
640 const PlatformLayer* tileLayer = it->get();
641
642 FloatRect visiblePart(CGRectOffset(PlatformCALayer::frameForLayer(tileLayer), tileTranslation.x(), tileTranslation.y()));
643 visiblePart.intersect(visibleRect);
644
645 if (!visiblePart.isEmpty())
646 paintedVisibleTiles.unite(enclosingIntRect(visiblePart));
647 }
648
649 Region uncoveredRegion(enclosingIntRect(visibleRect));
650 uncoveredRegion.subtract(paintedVisibleTiles);
651
652 return static_cast<unsigned>(uncoveredRegion.totalArea());
653}

757Ref<PlatformCALayer> TileController::createTileLayer(const IntRect& tileRect, TileGrid& grid)
758{
759 float temporaryScaleFactor = owningGraphicsLayer()->platformCALayerContentsScaleMultiplierForNewTiles(m_tileCacheLayer);
760 m_hasTilesWithTemporaryScaleFactor |= temporaryScaleFactor != 1;
761
762 auto layer = m_tileCacheLayer->createCompatibleLayerOrTakeFromPool(PlatformCALayer::LayerTypeTiledBackingTileLayer, &grid, tileRect.size());
763 layer->setAnchorPoint(FloatPoint3D());
764 layer->setPosition(tileRect.location());
765 layer->setBorderColor(m_tileDebugBorderColor);
766 layer->setBorderWidth(m_tileDebugBorderWidth);
767 layer->setEdgeAntialiasingMask(0);
768 layer->setOpaque(m_tilesAreOpaque);
769 layer->setName(makeString("tile at ", tileRect.location().x(), ',', tileRect.location().y()));
770 layer->setContentsScale(m_deviceScaleFactor * temporaryScaleFactor);
771 layer->setAcceleratesDrawing(m_acceleratesDrawing);
772 layer->setWantsDeepColorBackingStore(m_wantsDeepColorBackingStore);
773 layer->setSupportsSubpixelAntialiasedText(m_supportsSubpixelAntialiasedText);
774 layer->setNeedsDisplay();
775 return layer;
776}
