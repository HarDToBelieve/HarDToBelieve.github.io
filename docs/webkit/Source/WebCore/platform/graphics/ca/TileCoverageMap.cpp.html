35TileCoverageMap::TileCoverageMap(const TileController& controller)
36 : m_controller(controller)
37 , m_updateTimer(*this, &TileCoverageMap::updateTimerFired)
38 , m_layer(controller.rootLayer().createCompatibleLayer(PlatformCALayer::LayerTypeSimpleLayer, this))
39 , m_visibleViewportIndicatorLayer(controller.rootLayer().createCompatibleLayer(PlatformCALayer::LayerTypeLayer, nullptr))
40 , m_layoutViewportIndicatorLayer(controller.rootLayer().createCompatibleLayer(PlatformCALayer::LayerTypeLayer, nullptr))
41 , m_coverageRectIndicatorLayer(controller.rootLayer().createCompatibleLayer(PlatformCALayer::LayerTypeLayer, nullptr))
42 , m_position(FloatPoint(0, controller.topContentInset()))
43{
44 m_layer.get().setOpacity(0.75);
45 m_layer.get().setAnchorPoint(FloatPoint3D());
46 m_layer.get().setBorderColor(Color::black);
47 m_layer.get().setBorderWidth(1);
48 m_layer.get().setPosition(FloatPoint(2, 2));
49 m_layer.get().setContentsScale(m_controller.deviceScaleFactor());
50
51 m_visibleViewportIndicatorLayer.get().setName(MAKE_STATIC_STRING_IMPL("visible viewport indicator"));
52 m_visibleViewportIndicatorLayer.get().setBorderWidth(2);
53 m_visibleViewportIndicatorLayer.get().setAnchorPoint(FloatPoint3D());
54 m_visibleViewportIndicatorLayer.get().setBorderColor(Color::red.colorWithAlphaByte(200));
55
56 m_layoutViewportIndicatorLayer.get().setName(MAKE_STATIC_STRING_IMPL("layout viewport indicator"));
57 m_layoutViewportIndicatorLayer.get().setBorderWidth(2);
58 m_layoutViewportIndicatorLayer.get().setAnchorPoint(FloatPoint3D());
59 m_layoutViewportIndicatorLayer.get().setBorderColor(SRGBA<uint8_t> { 0, 128, 128, 200 });
60
61 m_coverageRectIndicatorLayer.get().setName(MAKE_STATIC_STRING_IMPL("coverage indicator"));
62 m_coverageRectIndicatorLayer.get().setAnchorPoint(FloatPoint3D());
63 m_coverageRectIndicatorLayer.get().setBackgroundColor(SRGBA<uint8_t> { 64, 64, 64, 50 });
64
65 m_layer.get().appendSublayer(m_coverageRectIndicatorLayer);
66 m_layer.get().appendSublayer(m_visibleViewportIndicatorLayer);
67
68 if (m_controller.layoutViewportRect())
69 m_layer.get().appendSublayer(m_layoutViewportIndicatorLayer);
70
71 update();
72}

90void TileCoverageMap::update()
91{
92 FloatRect containerBounds = m_controller.bounds();
93 FloatRect visibleRect = m_controller.visibleRect();
94 FloatRect coverageRect = m_controller.coverageRect();
95 visibleRect.contract(4, 4); // Layer is positioned 2px from top and left edges.
96
97 float widthScale = 1;
98 float scale = 1;
99 if (!containerBounds.isEmpty()) {
100 widthScale = std::min<float>(visibleRect.width() / containerBounds.width(), 0.1);
101 float visibleHeight = visibleRect.height() - std::min(m_controller.topContentInset(), visibleRect.y());
102 scale = std::min(widthScale, visibleHeight / containerBounds.height());
103 }
104
105 float indicatorScale = scale * m_controller.tileGrid().scale();
106
107 FloatRect mapBounds = containerBounds;
108 mapBounds.scale(indicatorScale);
109
110 m_layer.get().setPosition(m_position + FloatPoint(2, 2));
111 m_layer.get().setBounds(mapBounds);
112 m_layer.get().setNeedsDisplay();
113
114 visibleRect.scale(indicatorScale);
115 visibleRect.expand(2, 2);
116 m_visibleViewportIndicatorLayer->setPosition(visibleRect.location());
117 m_visibleViewportIndicatorLayer->setBounds(FloatRect(FloatPoint(), visibleRect.size()));
118
119 if (auto layoutViewportRect = m_controller.layoutViewportRect()) {
120 FloatRect layoutRect = layoutViewportRect.value();
121 layoutRect.scale(indicatorScale);
122 layoutRect.expand(2, 2);
123 m_layoutViewportIndicatorLayer->setPosition(layoutRect.location());
124 m_layoutViewportIndicatorLayer->setBounds(FloatRect(FloatPoint(), layoutRect.size()));
125
126 if (!m_layoutViewportIndicatorLayer->superlayer())
127 m_layer.get().appendSublayer(m_layoutViewportIndicatorLayer);
128 } else if (m_layoutViewportIndicatorLayer->superlayer())
129 m_layoutViewportIndicatorLayer->removeFromSuperlayer();
130
131 coverageRect.scale(indicatorScale);
132 coverageRect.expand(2, 2);
133 m_coverageRectIndicatorLayer->setPosition(coverageRect.location());
134 m_coverageRectIndicatorLayer->setBounds(FloatRect(FloatPoint(), coverageRect.size()));
135
136 Color visibleRectIndicatorColor;
137 switch (m_controller.indicatorMode()) {
138 case SynchronousScrollingBecauseOfLackOfScrollingCoordinatorIndication:
139 visibleRectIndicatorColor = SRGBA<uint8_t> { 200, 80, 255 };
140 break;
141 case SynchronousScrollingBecauseOfStyleIndication:
142 visibleRectIndicatorColor = Color::red;
143 break;
144 case SynchronousScrollingBecauseOfEventHandlersIndication:
145 visibleRectIndicatorColor = Color::yellow;
146 break;
147 case AsyncScrollingIndication:
148 visibleRectIndicatorColor = SRGBA<uint8_t> { 0, 200, 0 };
149 break;
150 }
151
152 m_visibleViewportIndicatorLayer.get().setBorderColor(visibleRectIndicatorColor);
153}

155void TileCoverageMap::platformCALayerPaintContents(PlatformCALayer* platformCALayer, GraphicsContext& context, const FloatRect&, GraphicsLayerPaintBehavior)
156{
157 ASSERT_UNUSED(platformCALayer, platformCALayer == m_layer.ptr());

159}
