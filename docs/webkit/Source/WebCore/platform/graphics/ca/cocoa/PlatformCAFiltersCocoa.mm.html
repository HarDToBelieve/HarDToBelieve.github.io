53void PlatformCAFilters::setFiltersOnLayer(PlatformLayer* layer, const FilterOperations& filters)
54{
55 if (!filters.size()) {
56 BEGIN_BLOCK_OBJC_EXCEPTIONS
57 [layer setFilters:nil];
58 // FIXME: this adds shadow properties to the layer even when it had none.
59 [layer setShadowOffset:CGSizeZero];
60 [layer setShadowColor:nil];
61 [layer setShadowRadius:0];
62 [layer setShadowOpacity:0];
63 END_BLOCK_OBJC_EXCEPTIONS
64 return;
65 }
66
67 // Assume filtersCanBeComposited was called and it returned true.
68 ASSERT(PlatformCALayerCocoa::filtersCanBeComposited(filters));
69
70 BEGIN_BLOCK_OBJC_EXCEPTIONS
71
72 unsigned i = 0;
73 auto array = createNSArray(filters.operations(), [&] (auto& operationPtr) -> id {
74 auto filterName = makeString("filter_", i++);
75 auto& filterOperation = *operationPtr;
76 switch (filterOperation.type()) {
77 case FilterOperation::DEFAULT:
78 case FilterOperation::REFERENCE:
79 case FilterOperation::NONE:
80 ASSERT_NOT_REACHED();
81 return nil;
82 case FilterOperation::DROP_SHADOW: {
83 // FIXME: For now assume drop shadow is the last filter, put it on the layer.
84 // <rdar://problem/10959969> Handle case where drop-shadow is not the last filter.
85 const auto& dropShadowOperation = downcast<DropShadowFilterOperation>(filterOperation);
86 [layer setShadowOffset:CGSizeMake(dropShadowOperation.x(), dropShadowOperation.y())];
87 [layer setShadowColor:cachedCGColor(dropShadowOperation.color())];
88 [layer setShadowRadius:dropShadowOperation.stdDeviation()];
89 [layer setShadowOpacity:1];
90 return nil;
91 }
92#if USE_CA_FILTERS
93 case FilterOperation::GRAYSCALE: {
94 const auto& colorMatrixOperation = downcast<BasicColorMatrixFilterOperation>(filterOperation);
95 CAFilter *filter = [CAFilter filterWithType:kCAFilterColorMonochrome];
96 [filter setValue:[NSNumber numberWithFloat:colorMatrixOperation.amount()] forKey:@"inputAmount"];
97 [filter setName:filterName];
98 return filter;
99 }
100 case FilterOperation::SEPIA: {
101 RetainPtr<NSValue> colorMatrixValue = PlatformCAFilters::colorMatrixValueForFilter(filterOperation.type(), &filterOperation);
102 CAFilter *filter = [CAFilter filterWithType:kCAFilterColorMatrix];
103 [filter setValue:colorMatrixValue.get() forKey:@"inputColorMatrix"];
104 [filter setName:filterName];
105 return filter;
106 }
107 case FilterOperation::SATURATE: {
108 const auto& colorMatrixOperation = downcast<BasicColorMatrixFilterOperation>(filterOperation);
109 CAFilter *filter = [CAFilter filterWithType:kCAFilterColorSaturate];
110 [filter setValue:[NSNumber numberWithFloat:colorMatrixOperation.amount()] forKey:@"inputAmount"];
111 [filter setName:filterName];
112 return filter;
113 }
114 case FilterOperation::HUE_ROTATE: {
115 const auto& colorMatrixOperation = downcast<BasicColorMatrixFilterOperation>(filterOperation);
116 CAFilter *filter = [CAFilter filterWithType:kCAFilterColorHueRotate];
117 [filter setValue:[NSNumber numberWithFloat:deg2rad(colorMatrixOperation.amount())] forKey:@"inputAngle"];
118 [filter setName:@"hueRotate"];
119 [filter setName:filterName];
120 return filter;
121 }
122 case FilterOperation::INVERT: {
123 RetainPtr<NSValue> colorMatrixValue = PlatformCAFilters::colorMatrixValueForFilter(filterOperation.type(), &filterOperation);
124 CAFilter *filter = [CAFilter filterWithType:kCAFilterColorMatrix];
125 [filter setValue:colorMatrixValue.get() forKey:@"inputColorMatrix"];
126 [filter setName:filterName];
127 return filter;
128 }
129 case FilterOperation::APPLE_INVERT_LIGHTNESS:
130 ASSERT_NOT_REACHED(); // APPLE_INVERT_LIGHTNESS is only used in -apple-color-filter.
131 break;
132 case FilterOperation::OPACITY: {
133 RetainPtr<NSValue> colorMatrixValue = PlatformCAFilters::colorMatrixValueForFilter(filterOperation.type(), &filterOperation);
134 CAFilter *filter = [CAFilter filterWithType:kCAFilterColorMatrix];
135 [filter setValue:colorMatrixValue.get() forKey:@"inputColorMatrix"];
136 [filter setName:filterName];
137 return filter;
138 }
139 case FilterOperation::BRIGHTNESS: {
140 RetainPtr<NSValue> colorMatrixValue = PlatformCAFilters::colorMatrixValueForFilter(filterOperation.type(), &filterOperation);
141 CAFilter *filter = [CAFilter filterWithType:kCAFilterColorMatrix];
142 [filter setValue:colorMatrixValue.get() forKey:@"inputColorMatrix"];
143 [filter setName:filterName];
144 return filter;
145 }
146 case FilterOperation::CONTRAST: {
147 RetainPtr<NSValue> colorMatrixValue = PlatformCAFilters::colorMatrixValueForFilter(filterOperation.type(), &filterOperation);
148 CAFilter *filter = [CAFilter filterWithType:kCAFilterColorMatrix];
149 [filter setValue:colorMatrixValue.get() forKey:@"inputColorMatrix"];
150 [filter setName:filterName];
151 return filter;
152 }
153 case FilterOperation::BLUR: {
154 const auto& blurOperation = downcast<BlurFilterOperation>(filterOperation);
155 CAFilter *filter = [CAFilter filterWithType:kCAFilterGaussianBlur];
156 [filter setValue:[NSNumber numberWithFloat:floatValueForLength(blurOperation.stdDeviation(), 0)] forKey:@"inputRadius"];
157#if ENABLE(FILTERS_LEVEL_2)
158 if ([layer isKindOfClass:[CABackdropLayer class]])
159 [filter setValue:@YES forKey:@"inputNormalizeEdges"];
160#endif
161 [filter setName:filterName];
162 return filter;
163 }
164#else
165 case FilterOperation::GRAYSCALE: {
166 const auto& colorMatrixOperation = downcast<BasicColorMatrixFilterOperation>(filterOperation);
167 CIFilter* filter = [CIFilter filterWithName:@"CIColorMonochrome"];
168 [filter setDefaults];
169 [filter setValue:[NSNumber numberWithFloat:colorMatrixOperation.amount()] forKey:@"inputIntensity"];
170 [filter setValue:[CIColor colorWithRed:0.67 green:0.67 blue:0.67] forKey:@"inputColor"]; // Color derived empirically to match zero saturation levels.
171 [filter setName:filterName];
172 return filter;
173 }
174 case FilterOperation::SEPIA: {
175 const auto& colorMatrixOperation = downcast<BasicColorMatrixFilterOperation>(filterOperation);
176 CIFilter* filter = [CIFilter filterWithName:@"CIColorMatrix"];
177 [filter setDefaults];
178
179 double t = colorMatrixOperation.amount();
180 t = std::min(std::max(0.0, t), 1.0);
181 // FIXME: results don't match the software filter.
182 [filter setValue:[CIVector vectorWithX:WebCore::blend(sepiaNoneConstants[0][0], sepiaFullConstants[0][0], t)
183 Y:WebCore::blend(sepiaNoneConstants[0][1], sepiaFullConstants[0][1], t)
184 Z:WebCore::blend(sepiaNoneConstants[0][2], sepiaFullConstants[0][2], t) W:0] forKey:@"inputRVector"];
185 [filter setValue:[CIVector vectorWithX:WebCore::blend(sepiaNoneConstants[1][0], sepiaFullConstants[1][0], t)
186 Y:WebCore::blend(sepiaNoneConstants[1][1], sepiaFullConstants[1][1], t)
187 Z:WebCore::blend(sepiaNoneConstants[1][2], sepiaFullConstants[1][2], t) W:0] forKey:@"inputGVector"];
188 [filter setValue:[CIVector vectorWithX:WebCore::blend(sepiaNoneConstants[2][0], sepiaFullConstants[2][0], t)
189 Y:WebCore::blend(sepiaNoneConstants[2][1], sepiaFullConstants[2][1], t)
190 Z:WebCore::blend(sepiaNoneConstants[2][2], sepiaFullConstants[2][2], t) W:0] forKey:@"inputBVector"];
191 [filter setName:filterName];
192 return filter;
193 }
194 case FilterOperation::SATURATE: {
195 const auto& colorMatrixOperation = downcast<BasicColorMatrixFilterOperation>(filterOperation);
196 CIFilter* filter = [CIFilter filterWithName:@"CIColorControls"];
197 [filter setDefaults];
198 [filter setValue:[NSNumber numberWithFloat:colorMatrixOperation.amount()] forKey:@"inputSaturation"];
199 [filter setName:filterName];
200 return filter;
201 }
202 case FilterOperation::HUE_ROTATE: {
203 const auto& colorMatrixOperation = downcast<BasicColorMatrixFilterOperation>(filterOperation);
204 CIFilter* filter = [CIFilter filterWithName:@"CIHueAdjust"];
205 [filter setDefaults];
206
207 [filter setValue:[NSNumber numberWithFloat:deg2rad(colorMatrixOperation.amount())] forKey:@"inputAngle"];
208 [filter setName:filterName];
209 return filter;
210 }
211 case FilterOperation::INVERT: {
212 const auto& componentTransferOperation = downcast<BasicComponentTransferFilterOperation>(filterOperation);
213 CIFilter* filter = [CIFilter filterWithName:@"CIColorMatrix"];
214 [filter setDefaults];
215
216 double multiplier = 1 - componentTransferOperation.amount() * 2;
217
218 // FIXME: the results of this filter look wrong.
219 [filter setValue:[CIVector vectorWithX:multiplier Y:0 Z:0 W:0] forKey:@"inputRVector"];
220 [filter setValue:[CIVector vectorWithX:0 Y:multiplier Z:0 W:0] forKey:@"inputGVector"];
221 [filter setValue:[CIVector vectorWithX:0 Y:0 Z:multiplier W:0] forKey:@"inputBVector"];
222 [filter setValue:[CIVector vectorWithX:0 Y:0 Z:0 W:1] forKey:@"inputAVector"];
223 [filter setValue:[CIVector vectorWithX:op->amount() Y:op->amount() Z:op->amount() W:0] forKey:@"inputBiasVector"];
224 [filter setName:filterName];
225 return filter;
226 }
227 case FilterOperation::APPLE_INVERT_LIGHTNESS:
228 ASSERT_NOT_REACHED(); // APPLE_INVERT_LIGHTNESS is only used in -apple-color-filter.
229 break;
230 case FilterOperation::OPACITY: {
231 const auto& componentTransferOperation = downcast<BasicComponentTransferFilterOperation>(filterOperation);
232 CIFilter* filter = [CIFilter filterWithName:@"CIColorMatrix"];
233 [filter setDefaults];
234
235 [filter setValue:[CIVector vectorWithX:1 Y:0 Z:0 W:0] forKey:@"inputRVector"];
236 [filter setValue:[CIVector vectorWithX:0 Y:1 Z:0 W:0] forKey:@"inputGVector"];
237 [filter setValue:[CIVector vectorWithX:0 Y:0 Z:1 W:0] forKey:@"inputBVector"];
238 [filter setValue:[CIVector vectorWithX:0 Y:0 Z:0 W:componentTransferOperation.amount()] forKey:@"inputAVector"];
239 [filter setValue:[CIVector vectorWithX:0 Y:0 Z:0 W:0] forKey:@"inputBiasVector"];
240 [filter setName:filterName];
241 return filter;
242 }
243 case FilterOperation::BRIGHTNESS: {
244 const auto& componentTransferOperation = downcast<BasicComponentTransferFilterOperation>(filterOperation);
245 CIFilter* filter = [CIFilter filterWithName:@"CIColorMatrix"];
246 [filter setDefaults];
247 double amount = componentTransferOperation.amount();
248
249 [filter setValue:[CIVector vectorWithX:amount Y:0 Z:0 W:0] forKey:@"inputRVector"];
250 [filter setValue:[CIVector vectorWithX:0 Y:amount Z:0 W:0] forKey:@"inputGVector"];
251 [filter setValue:[CIVector vectorWithX:0 Y:0 Z:amount W:0] forKey:@"inputBVector"];
252 [filter setName:filterName];
253 return filter;
254 }
255 case FilterOperation::CONTRAST: {
256 const auto& componentTransferOperation = downcast<BasicComponentTransferFilterOperation>(filterOperation);
257 CIFilter* filter = [CIFilter filterWithName:@"CIColorControls"];
258 [filter setDefaults];
259 [filter setValue:[NSNumber numberWithFloat:componentTransferOperation.amount()] forKey:@"inputContrast"];
260 [filter setName:filterName];
261 return filter;
262 }
263 case FilterOperation::BLUR: {
264 // FIXME: For now we ignore stdDeviationY.
265 const auto& blurOperation = downcast<BlurFilterOperation>(filterOperation);
266 CIFilter* filter = [CIFilter filterWithName:@"CIGaussianBlur"];
267 [filter setDefaults];
268 [filter setValue:[NSNumber numberWithFloat:floatValueForLength(blurOperation.stdDeviation(), 0)] forKey:@"inputRadius"];
269 [filter setName:filterName];
270 return filter;
271 }
272#endif
273 case FilterOperation::PASSTHROUGH:
274 return nil;
275 }
276 ASSERT_NOT_REACHED();
277 return nil;
278 });
279
280 if ([array count])
281 [layer setFilters:array.get()];
282
283 END_BLOCK_OBJC_EXCEPTIONS
284}

286RetainPtr<NSValue> PlatformCAFilters::filterValueForOperation(const FilterOperation* operation, int internalFilterPropertyIndex)
287{
288#if USE_CA_FILTERS
289 UNUSED_PARAM(internalFilterPropertyIndex);
290#endif
291 FilterOperation::OperationType type = operation->type();
292 RetainPtr<id> value;
293
294 if (is<DefaultFilterOperation>(*operation)) {
295 type = downcast<DefaultFilterOperation>(*operation).representedType();
296 operation = nullptr;
297 }
298
299 switch (type) {
300 case FilterOperation::DEFAULT:
301 ASSERT_NOT_REACHED();
302 break;
303 case FilterOperation::GRAYSCALE: {
304 // CIFilter: inputIntensity
305 // CAFilter: inputAmount
306 double amount = 0;
307 if (operation)
308 amount = downcast<BasicColorMatrixFilterOperation>(*operation).amount();
309
310 value = @(amount);
311 break;
312 }
313 case FilterOperation::SEPIA: {
314#if USE_CA_FILTERS
315 // CAFilter: inputColorMatrix
316 value = PlatformCAFilters::colorMatrixValueForFilter(type, operation);
317#else
318 // CIFilter: inputRVector, inputGVector, inputBVector
319 double amount = 0;
320 if (operation)
321 amount = downcast<BasicColorMatrixFilterOperation>(*operation).amount();
322
323 RetainPtr<CIVector> rowVector;
324 switch (internalFilterPropertyIndex) {
325 case 0: rowVector = adoptNS([[CIVector alloc] initWithX:WebCore::blend(sepiaNoneConstants[0][0], sepiaFullConstants[0][0], amount)
326 Y:WebCore::blend(sepiaNoneConstants[0][1], sepiaFullConstants[0][1], amount)
327 Z:WebCore::blend(sepiaNoneConstants[0][2], sepiaFullConstants[0][2], amount) W:0]);
328 break; // inputRVector
329 case 1: rowVector = adoptNS([[CIVector alloc] initWithX:WebCore::blend(sepiaNoneConstants[1][0], sepiaFullConstants[1][0], amount)
330 Y:WebCore::blend(sepiaNoneConstants[1][1], sepiaFullConstants[1][1], amount)
331 Z:WebCore::blend(sepiaNoneConstants[1][2], sepiaFullConstants[1][2], amount) W:0]);
332 break; // inputGVector
333 case 2: rowVector = adoptNS([[CIVector alloc] initWithX:WebCore::blend(sepiaNoneConstants[2][0], sepiaFullConstants[2][0], amount)
334 Y:WebCore::blend(sepiaNoneConstants[2][1], sepiaFullConstants[2][1], amount)
335 Z:WebCore::blend(sepiaNoneConstants[2][2], sepiaFullConstants[2][2], amount) W:0]);
336 break; // inputBVector
337 }
338 value = WTFMove(rowVector);
339#endif
340 break;
341 }
342 case FilterOperation::SATURATE: {
343 // CIFilter: inputSaturation
344 // CAFilter: inputAmount
345 double amount = 1;
346 if (operation)
347 amount = downcast<BasicColorMatrixFilterOperation>(*operation).amount();
348
349 value = @(amount);
350 break;
351 }
352 case FilterOperation::HUE_ROTATE: {
353 // Hue rotate CIFilter: inputAngle
354 // Hue rotate CAFilter: inputAngle
355 double amount = 0;
356 if (operation)
357 amount = downcast<BasicColorMatrixFilterOperation>(*operation).amount();
358
359 amount = deg2rad(amount);
360 value = @(amount);
361 break;
362 }
363 case FilterOperation::INVERT: {
364#if USE_CA_FILTERS
365 // CAFilter: inputColorMatrix
366 value = PlatformCAFilters::colorMatrixValueForFilter(type, operation);
367#else
368 // CIFilter: inputRVector, inputGVector, inputBVector, inputBiasVector
369 double amount = 0;
370 if (operation)
371 amount = downcast<BasicComponentTransferFilterOperation>(*operation).amount();
372
373 double multiplier = 1 - amount * 2;
374
375 // The color matrix animation for invert does a scale of each color component by a value that goes from
376 // 1 (when amount is 0) to -1 (when amount is 1). Then the color values are offset by amount. This has the
377 // effect of performing the operation: c' = c * -1 + 1, which inverts the color.
378 RetainPtr<CIVector> rowVector;
379 switch (internalFilterPropertyIndex) {
380 case 0: rowVector = adoptNS([[CIVector alloc] initWithX:multiplier Y:0 Z:0 W:0]); break; // inputRVector
381 case 1: rowVector = adoptNS([[CIVector alloc] initWithX:0 Y:multiplier Z:0 W:0]); break; // inputGVector
382 case 2: rowVector = adoptNS([[CIVector alloc] initWithX:0 Y:0 Z:multiplier W:0]); break; // inputBVector
383 case 3: rowVector = adoptNS([[CIVector alloc] initWithX:amount Y:amount Z:amount W:0]); break; // inputBiasVector
384 }
385 value = WTFMove(rowVector);
386#endif
387 break;
388 }
389 case FilterOperation::APPLE_INVERT_LIGHTNESS:
390 ASSERT_NOT_REACHED(); // APPLE_INVERT_LIGHTNESS is only used in -apple-color-filter.
391 break;
392 case FilterOperation::OPACITY: {
393#if USE_CA_FILTERS
394 // Opacity CAFilter: inputColorMatrix
395 value = PlatformCAFilters::colorMatrixValueForFilter(type, operation);
396#else
397 // Opacity CIFilter: inputAVector
398 double amount = 1;
399 if (operation)
400 amount = downcast<BasicComponentTransferFilterOperation>(*operation).amount();
401
402 value = adoptNS([[CIVector alloc] initWithX:0 Y:0 Z:0 W:amount]);
403#endif
404 break;
405 }
406
407 case FilterOperation::BRIGHTNESS: {
408#if USE_CA_FILTERS
409 // Brightness CAFilter: inputColorMatrix
410 value = PlatformCAFilters::colorMatrixValueForFilter(type, operation);
411#else
412 // Brightness CIFilter: inputColorMatrix
413 double amount = 1;
414 if (operation)
415 amount = downcast<BasicComponentTransferFilterOperation>(*operation).amount();
416
417 RetainPtr<CIVector> rowVector;
418 switch (internalFilterPropertyIndex) {
419 case 0: rowVector = adoptNS([[CIVector alloc] initWithX:amount Y:0 Z:0 W:0]); break; // inputRVector
420 case 1: rowVector = adoptNS([[CIVector alloc] initWithX:0 Y:amount Z:0 W:0]); break; // inputGVector
421 case 2: rowVector = adoptNS([[CIVector alloc] initWithX:0 Y:0 Z:amount W:0]); break; // inputBVector
422 }
423 value = WTFMove(rowVector);
424#endif
425 break;
426 }
427
428 case FilterOperation::CONTRAST: {
429#if USE_CA_FILTERS
430 // Contrast CAFilter: inputColorMatrix
431 value = PlatformCAFilters::colorMatrixValueForFilter(type, operation);
432#else
433 // Contrast CIFilter: inputContrast
434 double amount = 1;
435 if (operation)
436 amount = downcast<BasicComponentTransferFilterOperation>(*operation).amount();
437
438 value = @(amount);
439#endif
440 break;
441 }
442 case FilterOperation::BLUR: {
443 // CIFilter: inputRadius
444 // CAFilter: inputRadius
445 double amount = 0;
446 if (operation)
447 amount = floatValueForLength(downcast<BlurFilterOperation>(*operation).stdDeviation(), 0);
448
449 value = @(amount);
450 break;
451 }
452 default:
453 break;
454 }
455
456 return value;
457}

459#if USE_CA_FILTERS
460RetainPtr<NSValue> PlatformCAFilters::colorMatrixValueForFilter(FilterOperation::OperationType type, const FilterOperation* filterOperation)
461{
462 switch (type) {
463 case FilterOperation::SEPIA: {
464 double t = filterOperation ? downcast<BasicColorMatrixFilterOperation>(*filterOperation).amount() : 0;
465 t = std::min(std::max(0.0, t), 1.0);
466 BlendingContext context { t };
467 CAColorMatrix colorMatrix = {
468 static_cast<float>(WebCore::blend(sepiaNoneConstants[0][0], sepiaFullConstants[0][0], context)),
469 static_cast<float>(WebCore::blend(sepiaNoneConstants[0][1], sepiaFullConstants[0][1], context)),
470 static_cast<float>(WebCore::blend(sepiaNoneConstants[0][2], sepiaFullConstants[0][2], context)), 0, 0,
471
472 static_cast<float>(WebCore::blend(sepiaNoneConstants[1][0], sepiaFullConstants[1][0], context)),
473 static_cast<float>(WebCore::blend(sepiaNoneConstants[1][1], sepiaFullConstants[1][1], context)),
474 static_cast<float>(WebCore::blend(sepiaNoneConstants[1][2], sepiaFullConstants[1][2], context)), 0, 0,
475
476 static_cast<float>(WebCore::blend(sepiaNoneConstants[2][0], sepiaFullConstants[2][0], context)),
477 static_cast<float>(WebCore::blend(sepiaNoneConstants[2][1], sepiaFullConstants[2][1], context)),
478 static_cast<float>(WebCore::blend(sepiaNoneConstants[2][2], sepiaFullConstants[2][2], context)), 0, 0,
479 0, 0, 0, 1, 0
480 };
481 return [NSValue valueWithCAColorMatrix:colorMatrix];
482 }
483 case FilterOperation::INVERT: {
484 float amount = filterOperation ? downcast<BasicComponentTransferFilterOperation>(*filterOperation).amount() : 0;
485 float multiplier = 1 - amount * 2;
486 CAColorMatrix colorMatrix = {
487 multiplier, 0, 0, 0, amount,
488 0, multiplier, 0, 0, amount,
489 0, 0, multiplier, 0, amount,
490 0, 0, 0, 1, 0
491 };
492 return [NSValue valueWithCAColorMatrix:colorMatrix];
493 }
494 case FilterOperation::APPLE_INVERT_LIGHTNESS:
495 ASSERT_NOT_REACHED(); // APPLE_INVERT_LIGHTNESS is only used in -apple-color-filter.
496 return nullptr;
497 case FilterOperation::OPACITY: {
498 float amount = filterOperation ? downcast<BasicComponentTransferFilterOperation>(filterOperation)->amount() : 1;
499 CAColorMatrix colorMatrix = {
500 1, 0, 0, 0, 0,
501 0, 1, 0, 0, 0,
502 0, 0, 1, 0, 0,
503 0, 0, 0, amount, 0
504 };
505 return [NSValue valueWithCAColorMatrix:colorMatrix];
506 }
507 case FilterOperation::CONTRAST: {
508 float amount = filterOperation ? downcast<BasicComponentTransferFilterOperation>(filterOperation)->amount() : 1;
509 float intercept = -0.5 * amount + 0.5;
510 CAColorMatrix colorMatrix = {
511 amount, 0, 0, 0, intercept,
512 0, amount, 0, 0, intercept,
513 0, 0, amount, 0, intercept,
514 0, 0, 0, 1, 0
515 };
516 return [NSValue valueWithCAColorMatrix:colorMatrix];
517 }
518 case FilterOperation::BRIGHTNESS: {
519 float amount = filterOperation ? downcast<BasicComponentTransferFilterOperation>(filterOperation)->amount() : 1;
520 CAColorMatrix colorMatrix = {
521 amount, 0, 0, 0, 0,
522 0, amount, 0, 0, 0,
523 0, 0, amount, 0, 0,
524 0, 0, 0, 1, 0
525 };
526 return [NSValue valueWithCAColorMatrix:colorMatrix];
527 }
528 default:
529 ASSERT_NOT_REACHED();
530 return 0;
531 }
532}

535void PlatformCAFilters::setBlendingFiltersOnLayer(PlatformLayer* layer, const BlendMode blendMode)
536{
537#if USE_CA_FILTERS
538 BEGIN_BLOCK_OBJC_EXCEPTIONS
539
540 CAFilter* filter = nil;
541
542 switch (blendMode) {
543 case BlendMode::Normal:
544 // No need to set an actual filter object in this case.
545 break;
546 case BlendMode::Overlay:
547 filter = [CAFilter filterWithType:kCAFilterOverlayBlendMode];
548 break;
549 case BlendMode::ColorDodge:
550 filter = [CAFilter filterWithType:kCAFilterColorDodgeBlendMode];
551 break;
552 case BlendMode::ColorBurn:
553 filter = [CAFilter filterWithType:kCAFilterColorBurnBlendMode];
554 break;
555 case BlendMode::Darken:
556 filter = [CAFilter filterWithType:kCAFilterDarkenBlendMode];
557 break;
558 case BlendMode::Difference:
559 filter = [CAFilter filterWithType:kCAFilterDifferenceBlendMode];
560 break;
561 case BlendMode::Exclusion:
562 filter = [CAFilter filterWithType:kCAFilterExclusionBlendMode];
563 break;
564 case BlendMode::HardLight:
565 filter = [CAFilter filterWithType:kCAFilterHardLightBlendMode];
566 break;
567 case BlendMode::Multiply:
568 filter = [CAFilter filterWithType:kCAFilterMultiplyBlendMode];
569 break;
570 case BlendMode::Lighten:
571 filter = [CAFilter filterWithType:kCAFilterLightenBlendMode];
572 break;
573 case BlendMode::SoftLight:
574 filter = [CAFilter filterWithType:kCAFilterSoftLightBlendMode];
575 break;
576 case BlendMode::Screen:
577 filter = [CAFilter filterWithType:kCAFilterScreenBlendMode];
578 break;
579 case BlendMode::PlusDarker:
580 filter = [CAFilter filterWithType:kCAFilterPlusD];
581 break;
582 case BlendMode::PlusLighter:
583 filter = [CAFilter filterWithType:kCAFilterPlusL];
584 break;
585 case BlendMode::Hue:
586 case BlendMode::Saturation:
587 case BlendMode::Color:
588 case BlendMode::Luminosity:
589 // FIXME: CA does't support non-separable blend modes on compositing filters.
590 break;
591 default:
592 ASSERT_NOT_REACHED();
593 }
594
595 [layer setCompositingFilter:filter];
596
597 END_BLOCK_OBJC_EXCEPTIONS
598#else
599 UNUSED_PARAM(layer);
600 UNUSED_PARAM(blendMode);
