106static MonotonicTime mediaTimeToCurrentTime(CFTimeInterval t)
107{
108 return MonotonicTime::now() + Seconds(t - CACurrentMediaTime());
109}

113// Delegate for animationDidStart callback
114@interface WebAnimationDelegate : NSObject {
115 WebCore::PlatformCALayer* m_owner;
116}
117
118- (void)animationDidStart:(CAAnimation *)anim;
119- (void)setOwner:(WebCore::PlatformCALayer*)owner;
120
121@end
122
123@implementation WebAnimationDelegate
124
125- (void)animationDidStart:(CAAnimation *)animation
126{
127 using namespace WebCore;
128#if PLATFORM(IOS_FAMILY)
129 WebThreadLock();
130#endif
131 if (!m_owner)
132 return;
133
134 MonotonicTime startTime;
135 if (hasExplicitBeginTime(animation)) {
136 // We don't know what time CA used to commit the animation, so just use the current time
137 // (even though this will be slightly off).
138 startTime = mediaTimeToCurrentTime(CACurrentMediaTime());
139 } else
140 startTime = mediaTimeToCurrentTime([animation beginTime]);
141
142 CALayer *layer = m_owner->platformLayer();
143
144 String animationKey;
145 for (NSString *key in [layer animationKeys]) {
146 if ([layer animationForKey:key] == animation) {
147 animationKey = key;
148 break;
149 }
150 }
151
152 if (!animationKey.isEmpty())
153 m_owner->animationStarted(animationKey, startTime);
154}
155
156- (void)animationDidStop:(CAAnimation *)animation finished:(BOOL)finished
157{
158 using namespace WebCore;
159#if PLATFORM(IOS_FAMILY)
160 WebThreadLock();
161#endif
162 UNUSED_PARAM(finished);
163
164 if (!m_owner)
165 return;
166
167 CALayer *layer = m_owner->platformLayer();
168
169 String animationKey;
170 for (NSString *key in [layer animationKeys]) {
171 if ([layer animationForKey:key] == animation) {
172 animationKey = key;
173 break;
174 }
175 }
176
177 if (!animationKey.isEmpty())
178 m_owner->animationEnded(animationKey);
179}
180
181- (void)setOwner:(WebCore::PlatformCALayer*)owner
182{
183 m_owner = owner;
184}
185
186@end

209PlatformCALayer::LayerType PlatformCALayerCocoa::layerTypeForPlatformLayer(PlatformLayer* layer)
210{
211 if (PAL::isAVFoundationFrameworkAvailable() && [layer isKindOfClass:PAL::getAVPlayerLayerClass()])
212 return LayerTypeAVPlayerLayer;
213
214 if ([layer isKindOfClass:WebVideoContainerLayer.class]
215 && [(WebVideoContainerLayer*)layer playerLayer])
216 return LayerTypeAVPlayerLayer;
217
218 if ([layer isKindOfClass:[WebGLLayer class]])
219 return LayerTypeContentsProvidedLayer;
220
221#if ENABLE(WEBGPU)
222 if ([layer isKindOfClass:[WebGPULayer class]])
223 return LayerTypeContentsProvidedLayer;
224#endif
225
226 return LayerTypeCustom;
227}

229PlatformCALayerCocoa::PlatformCALayerCocoa(LayerType layerType, PlatformCALayerClient* owner)
230 : PlatformCALayer(layerType, owner)
231{
232 Class layerClass = Nil;
233 switch (layerType) {
234 case LayerTypeLayer:
235 case LayerTypeRootLayer:
236 layerClass = [CALayer class];
237 break;
238 case LayerTypeScrollContainerLayer:
239 // Scroll container layers only have special behavior with PlatformCALayerRemote.
240 // fallthrough
241 case LayerTypeWebLayer:
242 layerClass = [WebLayer class];
243 break;
244 case LayerTypeSimpleLayer:
245 case LayerTypeTiledBackingTileLayer:
246 layerClass = [WebSimpleLayer class];
247 break;
248 case LayerTypeTransformLayer:
249 layerClass = [CATransformLayer class];
250 break;
251#if ENABLE(FILTERS_LEVEL_2)
252 case LayerTypeBackdropLayer:
253 layerClass = [CABackdropLayer class];
254 break;
255 case LayerTypeLightSystemBackdropLayer:
256 layerClass = [WebLightSystemBackdropLayer class];
257 break;
258 case LayerTypeDarkSystemBackdropLayer:
259 layerClass = [WebDarkSystemBackdropLayer class];
260 break;
261#else
262 case LayerTypeBackdropLayer:
263 case LayerTypeLightSystemBackdropLayer:
264 case LayerTypeDarkSystemBackdropLayer:
265 ASSERT_NOT_REACHED();
266 layerClass = [CALayer class];
267 break;
268#endif
269 case LayerTypeTiledBackingLayer:
270 case LayerTypePageTiledBackingLayer:
271 layerClass = [WebTiledBackingLayer class];
272 break;
273 case LayerTypeAVPlayerLayer:
274 if (PAL::isAVFoundationFrameworkAvailable())
275 layerClass = PAL::getAVPlayerLayerClass();
276 break;
277 case LayerTypeContentsProvidedLayer:
278 // We don't create PlatformCALayerCocoas wrapped around WebGLLayers or WebGPULayers.
279 ASSERT_NOT_REACHED();
280 break;
281#if ENABLE(MODEL_ELEMENT)
282 case LayerTypeModelLayer:
283 layerClass = [CALayer class];
284 break;
285#endif
286 case LayerTypeShapeLayer:
287 layerClass = [CAShapeLayer class];
288 // fillColor defaults to opaque black.
289 break;
290 case LayerTypeCustom:
291 break;
292 }
293
294 if (layerClass)
295 m_layer = adoptNS([(CALayer *)[layerClass alloc] init]);
296
297#if ENABLE(FILTERS_LEVEL_2) && PLATFORM(MAC)
298 if (layerType == LayerTypeBackdropLayer)
299 [(CABackdropLayer*)m_layer.get() setWindowServerAware:NO];
300#endif
301
302 commonInit();
303}

312void PlatformCALayerCocoa::commonInit()
313{
314 BEGIN_BLOCK_OBJC_EXCEPTIONS
315
316 {
317 Locker locker { layerToPlatformLayerMapLock };
318 layerToPlatformLayerMap().add(m_layer.get(), this);
319 }
320
321 // Clear all the implicit animations on the CALayer
322 if (m_layerType == LayerTypeAVPlayerLayer || m_layerType == LayerTypeContentsProvidedLayer || m_layerType == LayerTypeScrollContainerLayer || m_layerType == LayerTypeCustom)
323 [m_layer web_disableAllActions];
324 else
325 [m_layer setDelegate:[WebActionDisablingCALayerDelegate shared]];
326
327 // So that the scrolling thread's performance logging code can find all the tiles, mark this as being a tile.
328 if (m_layerType == LayerTypeTiledBackingTileLayer)
329 [m_layer setValue:@YES forKey:@"isTile"];
330
331 if (usesTiledBackingLayer()) {
332 WebTiledBackingLayer* tiledBackingLayer = static_cast<WebTiledBackingLayer*>(m_layer.get());
333 TileController* tileController = [tiledBackingLayer createTileController:this];
334
335 m_customSublayers = makeUnique<PlatformCALayerList>(tileController->containerLayers());
336 }
337
338 END_BLOCK_OBJC_EXCEPTIONS
339}

341Ref<PlatformCALayer> PlatformCALayerCocoa::clone(PlatformCALayerClient* owner) const
342{
343 LayerType type;
344 switch (layerType()) {
345 case LayerTypeTransformLayer:
346 type = LayerTypeTransformLayer;
347 break;
348 case LayerTypeAVPlayerLayer:
349 type = LayerTypeAVPlayerLayer;
350 break;
351 case LayerTypeShapeLayer:
352 type = LayerTypeShapeLayer;
353 break;
354 case LayerTypeBackdropLayer:
355 type = LayerTypeBackdropLayer;
356 break;
357 case LayerTypeLayer:
358 default:
359 type = LayerTypeLayer;
360 break;
361 };
362 auto newLayer = PlatformCALayerCocoa::create(type, owner);
363
364 newLayer->setPosition(position());
365 newLayer->setBounds(bounds());
366 newLayer->setAnchorPoint(anchorPoint());
367 newLayer->setTransform(transform());
368 newLayer->setSublayerTransform(sublayerTransform());
369 newLayer->setContents(contents());
370 newLayer->setMasksToBounds(masksToBounds());
371 newLayer->setDoubleSided(isDoubleSided());
372 newLayer->setOpaque(isOpaque());
373 newLayer->setBackgroundColor(backgroundColor());
374 newLayer->setContentsScale(contentsScale());
375 newLayer->setCornerRadius(cornerRadius());
376 newLayer->copyFiltersFrom(*this);
377 newLayer->updateCustomAppearance(customAppearance());
378
379 if (type == LayerTypeAVPlayerLayer) {
380 ASSERT(PAL::isAVFoundationFrameworkAvailable() && [newLayer->platformLayer() isKindOfClass:PAL::getAVPlayerLayerClass()]);
381
382 AVPlayerLayer *destinationPlayerLayer = newLayer->avPlayerLayer();
383 AVPlayerLayer *sourcePlayerLayer = avPlayerLayer();
384 ASSERT(sourcePlayerLayer);
385
386 RunLoop::main().dispatch([destinationPlayerLayer = retainPtr(destinationPlayerLayer), sourcePlayerLayer = retainPtr(sourcePlayerLayer)] {
387 [destinationPlayerLayer setPlayer:[sourcePlayerLayer player]];
388 });
389 }
390
391 if (type == LayerTypeShapeLayer)
392 newLayer->setShapeRoundedRect(shapeRoundedRect());
393
394 return newLayer;
395}

397PlatformCALayerCocoa::~PlatformCALayerCocoa()
398{
399 {
400 Locker locker { layerToPlatformLayerMapLock };
401 layerToPlatformLayerMap().remove(m_layer.get());
402 }
403
404 // Remove the owner pointer from the delegate in case there is a pending animationStarted event.
405 [static_cast<WebAnimationDelegate*>(m_delegate.get()) setOwner:nil];
406
407 if (usesTiledBackingLayer())
408 [static_cast<WebTiledBackingLayer *>(m_layer.get()) invalidate];
409}

466void PlatformCALayerCocoa::setSublayers(const PlatformCALayerList& list)
467{
468 // Short circuiting here avoids the allocation of the array below.
469 if (!list.size()) {
470 removeAllSublayers();
471 return;
472 }
473
474 BEGIN_BLOCK_OBJC_EXCEPTIONS
475 [m_layer setSublayers:createNSArray(list, [] (auto& layer) {
476 return layer->m_layer;
477 }).get()];
478 END_BLOCK_OBJC_EXCEPTIONS
479}

481PlatformCALayerList PlatformCALayerCocoa::sublayersForLogging() const
482{
483 PlatformCALayerList sublayers;
484 BEGIN_BLOCK_OBJC_EXCEPTIONS
485 [[m_layer sublayers] enumerateObjectsUsingBlock:makeBlockPtr([&] (CALayer *layer, NSUInteger, BOOL *) {
486 auto platformCALayer = PlatformCALayer::platformCALayerForLayer(layer);
487 if (!platformCALayer)
488 return;
489 sublayers.append(platformCALayer);
490 }).get()];
491 END_BLOCK_OBJC_EXCEPTIONS
492
493 return sublayers;
494}

534void PlatformCALayerCocoa::addAnimationForKey(const String& key, PlatformCAAnimation& animation)
535{
536 // Add the delegate
537 if (!m_delegate) {
538 auto webAnimationDelegate = adoptNS([[WebAnimationDelegate alloc] init]);
539 [webAnimationDelegate setOwner:this];
540 m_delegate = WTFMove(webAnimationDelegate);
541 }
542
543 CAAnimation *propertyAnimation = static_cast<CAAnimation *>(downcast<PlatformCAAnimationCocoa>(animation).platformAnimation());
544 if (![propertyAnimation delegate])
545 [propertyAnimation setDelegate:static_cast<id>(m_delegate.get())];
546
547 BEGIN_BLOCK_OBJC_EXCEPTIONS
548 [m_layer addAnimation:propertyAnimation forKey:key];
549 END_BLOCK_OBJC_EXCEPTIONS
550}

886bool PlatformCALayerCocoa::filtersCanBeComposited(const FilterOperations& filters)
887{
888 // Return false if there are no filters to avoid needless work
889 if (!filters.size())
890 return false;
891
892 for (unsigned i = 0; i < filters.size(); ++i) {
893 const FilterOperation* filterOperation = filters.at(i);
894 switch (filterOperation->type()) {
895 case FilterOperation::REFERENCE:
896 return false;
897 case FilterOperation::DROP_SHADOW:
898 // FIXME: For now we can only handle drop-shadow is if it's last in the list
899 if (i < (filters.size() - 1))
900 return false;
901 break;
902 default:
903 break;
904 }
905 }
906
907 return true;
908}

1187PlatformCALayer::RepaintRectList PlatformCALayer::collectRectsToPaint(GraphicsContext& context, PlatformCALayer* platformCALayer)
1188{
1189 __block double totalRectArea = 0;
1190 __block unsigned rectCount = 0;
1191 __block RepaintRectList dirtyRects;
1192
1193 platformCALayer->enumerateRectsBeingDrawn(context, ^(FloatRect rect) {
1194 if (++rectCount > webLayerMaxRectsToPaint)
1195 return;
1196
1197 totalRectArea += rect.area();
1198 dirtyRects.append(rect);
1199 });
1200
1201 FloatRect clipBounds = context.clipBounds();
1202 double clipArea = clipBounds.width() * clipBounds.height();
1203
1204 if (rectCount >= webLayerMaxRectsToPaint || totalRectArea >= clipArea * webLayerWastedSpaceThreshold) {
1205 dirtyRects.clear();
1206 dirtyRects.append(clipBounds);
1207 }
1208
1209 return dirtyRects;
1210}

1212void PlatformCALayer::drawLayerContents(GraphicsContext& graphicsContext, WebCore::PlatformCALayer* platformCALayer, RepaintRectList& dirtyRects, GraphicsLayerPaintBehavior layerPaintBehavior)
1213{
1214 WebCore::PlatformCALayerClient* layerContents = platformCALayer->owner();
1215 if (!layerContents)
1216 return;
1217
1218 if (!layerContents->platformCALayerRepaintCount(platformCALayer))
1219 layerPaintBehavior |= GraphicsLayerPaintFirstTilePaint;
1220
1221 {
1222 GraphicsContextStateSaver saver(graphicsContext);
1223 std::optional<LocalCurrentGraphicsContext> platformContextSaver;
1224#if PLATFORM(IOS_FAMILY)
1225 std::optional<FontAntialiasingStateSaver> fontAntialiasingState;
1226#endif
1227
1228 // We never use CompositingCoordinatesOrientation::BottomUp on Mac.
1229 ASSERT(layerContents->platformCALayerContentsOrientation() == GraphicsLayer::CompositingCoordinatesOrientation::TopDown);
1230
1231 if (graphicsContext.hasPlatformContext()) {
1232 platformContextSaver.emplace(graphicsContext);
1233#if PLATFORM(IOS_FAMILY)
1234 CGContextRef context = graphicsContext.platformContext();
1235 WKSetCurrentGraphicsContext(context);
1236 fontAntialiasingState.emplace(context, !![platformCALayer->platformLayer() isOpaque]);
1237 fontAntialiasingState->setup([WAKWindow hasLandscapeOrientation]);
1238#endif
1239 graphicsContext.setIsCALayerContext(true);
1240 graphicsContext.setIsAcceleratedContext(platformCALayer->acceleratesDrawing());
1241 }
1242
1243 {
1244 if (!layerContents->platformCALayerContentsOpaque() && !platformCALayer->supportsSubpixelAntialiasedText() && FontCascade::isSubpixelAntialiasingAvailable()) {
1245 // Turn off font smoothing to improve the appearance of text rendered onto a transparent background.
1246 graphicsContext.setShouldSmoothFonts(false);
1247 }
1248
1249#if PLATFORM(MAC)
1250 // It's important to get the clip from the context, because it may be significantly
1251 // smaller than the layer bounds (e.g. tiled layers)
1252 ThemeMac::setFocusRingClipRect(graphicsContext.clipBounds());
1253#endif
1254
1255 for (const auto& rect : dirtyRects) {
1256 GraphicsContextStateSaver stateSaver(graphicsContext);
1257 graphicsContext.clip(rect);
1258
1259 layerContents->platformCALayerPaintContents(platformCALayer, graphicsContext, rect, layerPaintBehavior);
1260 }
1261
1262#if PLATFORM(MAC)
1263 ThemeMac::setFocusRingClipRect(FloatRect());
1264#endif
1265 }
1266 }
1267
1268 // Re-fetch the layer owner, since <rdar://problem/9125151> indicates that it might have been destroyed during painting.
1269 layerContents = platformCALayer->owner();
1270 ASSERT(layerContents);
1271
1272 if (!layerContents)
1273 return;
1274
1275 // Always update the repaint count so that it's accurate even if the count itself is not shown. This will be useful
1276 // for the Web Inspector feeding this information through the LayerTreeAgent.
1277 int repaintCount = layerContents->platformCALayerIncrementRepaintCount(platformCALayer);
1278
1279 if (!platformCALayer->usesTiledBackingLayer() && layerContents->platformCALayerShowRepaintCounter(platformCALayer))
1280 drawRepaintIndicator(graphicsContext, platformCALayer, repaintCount);
1281}

1293void PlatformCALayerCocoa::enumerateRectsBeingDrawn(GraphicsContext& context, void (^block)(FloatRect))
1294{
1295 CGSRegionObj region = (CGSRegionObj)[m_layer regionBeingDrawn];
1296 if (!region) {
1297 block(context.clipBounds());
1298 return;
1299 }
1300
1301 CGAffineTransform inverseTransform = CGAffineTransformInvert(context.getCTM());
1302 CGSRegionEnumeratorObj enumerator = CGSRegionEnumerator(region);
1303 const CGRect* nextRect;
1304 while ((nextRect = CGSNextRect(enumerator))) {
1305 CGRect rectToDraw = CGRectApplyAffineTransform(*nextRect, inverseTransform);
1306 block(rectToDraw);
1307 }
1308
1309 CGSReleaseRegionEnumerator(enumerator);
1310}
