39@implementation WebLightSystemBackdropLayer
40
41- (id)init
42{
43 self = [super init];
44 if (!self)
45 return nil;
46
47#ifndef NDEBUG
48 [self setName:@"WebLightSystemBackdropLayer"];
49#endif
50
51 CGFloat components[4] = { 0.8, 0.8, 0.8, 0.8 };
52 [super setBackgroundColor:adoptCF(CGColorCreate(WebCore::sRGBColorSpaceRef(), components)).get()];
53
54 return self;
55}
56
57- (void)setBackgroundColor:(CGColorRef)backgroundColor
58{
59 // Empty implementation to stop clients pushing the wrong color.
60 UNUSED_PARAM(backgroundColor);
61}
62
63@end
