43void Gradient::createCGGradient()
44{
45 sortStops();
46
47 auto colorsArray = adoptCF(CFArrayCreateMutable(0, m_stops.size(), &kCFTypeArrayCallBacks));
48 unsigned numStops = m_stops.size();
49
50 const int reservedStops = 3;
51 Vector<CGFloat, reservedStops> locations;
52 locations.reserveInitialCapacity(numStops);
53
54 Vector<CGFloat, 4 * reservedStops> colorComponents;
55 colorComponents.reserveInitialCapacity(numStops * 4);
56
57 // FIXME: Consider making this into two loops to avoid unnecessary allocation of the
58 // CGColorRefs in the common case of all ColorSpace::SRGB.
59
60 bool hasOnlyBoundedSRGBColorStops = true;
61 for (const auto& stop : m_stops) {
62 // If all the stops are bounded sRGB (as represented by the color having the color space
63 // ColorSpace::SRGB, it is faster to create a gradient using components than CGColors.
64 if (stop.color.colorSpace() != ColorSpace::SRGB)
65 hasOnlyBoundedSRGBColorStops = false;
66
67 auto [colorSpace, components] = stop.color.colorSpaceAndComponents();
68 auto [r, g, b, a] = components;
69 colorComponents.uncheckedAppend(r);
70 colorComponents.uncheckedAppend(g);
71 colorComponents.uncheckedAppend(b);
72 colorComponents.uncheckedAppend(a);
73
74 CFArrayAppendValue(colorsArray.get(), cachedCGColor(stop.color));
75 locations.uncheckedAppend(stop.offset);
76 }
77
78#if HAVE(CORE_GRAPHICS_EXTENDED_SRGB_COLOR_SPACE)
79 auto extendedColorsGradientColorSpace = extendedSRGBColorSpaceRef();
80#else
81 auto extendedColorsGradientColorSpace = sRGBColorSpaceRef();
82#endif
83
84 if (hasOnlyBoundedSRGBColorStops)
85 m_gradient = adoptCF(CGGradientCreateWithColorComponents(sRGBColorSpaceRef(), colorComponents.data(), locations.data(), numStops));
86 else
87 m_gradient = adoptCF(CGGradientCreateWithColors(extendedColorsGradientColorSpace, colorsArray.get(), locations.data()));
88}

101void Gradient::paint(CGContextRef platformContext)
102{
103 CGGradientDrawingOptions extendOptions = kCGGradientDrawsBeforeStartLocation | kCGGradientDrawsAfterEndLocation;
104
105 if (!m_gradient)
106 createCGGradient();
107
108 WTF::switchOn(m_data,
109 [&] (const LinearData& data) {
110 switch (m_spreadMethod) {
111 case GradientSpreadMethod::Repeat:
112 case GradientSpreadMethod::Reflect: {
113 CGContextStateSaver saveState(platformContext);
114
115 FloatPoint gradientVectorNorm(data.point1 - data.point0);
116 gradientVectorNorm.normalize();
117 CGFloat angle = acos(gradientVectorNorm.dot({ 1, 0 }));
118 CGContextRotateCTM(platformContext, angle);
119
120 CGAffineTransform transform = CGAffineTransformMakeRotation(-angle);
121 FloatPoint point0 = CGPointApplyAffineTransform(data.point0, transform);
122 FloatPoint point1 = CGPointApplyAffineTransform(data.point1, transform);
123
124 CGRect boundingBox = CGContextGetClipBoundingBox(platformContext);
125 CGFloat width = point1.x() - point0.x();
126 CGFloat pixelSize = CGFAbs(CGContextConvertSizeToUserSpace(platformContext, CGSizeMake(1, 1)).width);
127
128 if (width > 0 && !CGRectIsInfinite(boundingBox) && !CGRectIsEmpty(boundingBox)) {
129 extendOptions = 0;
130 if (width < pixelSize)
131 width = pixelSize;
132
133 CGFloat gradientStart = point0.x();
134 CGFloat gradientEnd = point1.x();
135 bool flip = m_spreadMethod == GradientSpreadMethod::Reflect;
136
137 // Find first gradient position to the left of the bounding box
138 int n = CGFloor((boundingBox.origin.x - gradientStart) / width);
139 gradientStart += n * width;
140 if (!(n % 2))
141 flip = false;
142
143 gradientEnd -= CGFloor((gradientEnd - CGRectGetMaxX(boundingBox)) / width) * width;
144
145 for (CGFloat start = gradientStart; start <= gradientEnd; start += width) {
146 CGPoint left = CGPointMake(flip ? start + width : start, boundingBox.origin.y);
147 CGPoint right = CGPointMake(flip ? start : start + width, boundingBox.origin.y);
148
149 CGContextDrawLinearGradient(platformContext, m_gradient.get(), left, right, extendOptions);
150
151 if (m_spreadMethod == GradientSpreadMethod::Reflect)
152 flip = !flip;
153 }
154
155 break;
156 }
157
158 FALLTHROUGH;
159 }
160 case GradientSpreadMethod::Pad:
161 CGContextDrawLinearGradient(platformContext, m_gradient.get(), data.point0, data.point1, extendOptions);
162 }
163 },
164 [&] (const RadialData& data) {
165 bool needScaling = data.aspectRatio != 1;
166 if (needScaling) {
167 CGContextSaveGState(platformContext);
168 // Scale from the center of the gradient. We only ever scale non-deprecated gradients,
169 // for which point0 == point1.
170 ASSERT(data.point0 == data.point1);
171 CGContextTranslateCTM(platformContext, data.point0.x(), data.point0.y());
172 CGContextScaleCTM(platformContext, 1, 1 / data.aspectRatio);
173 CGContextTranslateCTM(platformContext, -data.point0.x(), -data.point0.y());
174 }
175
176 CGContextDrawRadialGradient(platformContext, m_gradient.get(), data.point0, data.startRadius, data.point1, data.endRadius, extendOptions);
177
178 if (needScaling)
179 CGContextRestoreGState(platformContext);
180 },
181 [&] (const ConicData& data) {
182// FIXME: Seems like this should be HAVE(CG_CONTEXT_DRAW_CONIC_GRADIENT).
183// FIXME: Can we change tvOS to be like the other Cocoa platforms?
184#if PLATFORM(COCOA) && !PLATFORM(APPLETV)
185 CGContextSaveGState(platformContext);
186 CGContextTranslateCTM(platformContext, data.point0.x(), data.point0.y());
187 CGContextRotateCTM(platformContext, (CGFloat)-M_PI_2);
188 CGContextTranslateCTM(platformContext, -data.point0.x(), -data.point0.y());
189 CGContextDrawConicGradient(platformContext, m_gradient.get(), data.point0, data.angleRadians);
190 CGContextRestoreGState(platformContext);
191#else
192 UNUSED_PARAM(data);
193#endif
194 }
195 );
196}
