63static InterpolationQuality coreInterpolationQuality(CGInterpolationQuality quality)
64{
65 switch (quality) {
66 case kCGInterpolationDefault:
67 return InterpolationQuality::Default;
68 case kCGInterpolationNone:
69 return InterpolationQuality::DoNotInterpolate;
70 case kCGInterpolationLow:
71 return InterpolationQuality::Low;
72 case kCGInterpolationMedium:

233void GraphicsContextCG::drawNativeImage(NativeImage& nativeImage, const FloatSize& imageSize, const FloatRect& destRect, const FloatRect& srcRect, const ImagePaintingOptions& options)
234{
235 auto image = nativeImage.platformImage();
236 auto imageRect = FloatRect { { }, imageSize };
237 auto normalizedSrcRect = normalizeRect(srcRect);
238 auto normalizedDestRect = normalizeRect(destRect);
239
240 if (!image || !imageRect.intersects(normalizedSrcRect))
241 return;
242
243#if !LOG_DISABLED
244 MonotonicTime startTime = MonotonicTime::now();
245#endif
246
247 auto shouldUseSubimage = [](CGInterpolationQuality interpolationQuality, const FloatRect& destRect, const FloatRect& srcRect, const AffineTransform& transform) -> bool {
248 if (interpolationQuality == kCGInterpolationNone)
249 return false;
250 if (transform.isRotateOrShear())
251 return true;
252 auto xScale = destRect.width() * transform.xScale() / srcRect.width();
253 auto yScale = destRect.height() * transform.yScale() / srcRect.height();
254 return !WTF::areEssentiallyEqual(xScale, yScale) || xScale > 1;
255 };
256
257 auto getSubimage = [](CGImageRef image, const FloatSize& imageSize, const FloatRect& subimageRect, const ImagePaintingOptions& options) -> RetainPtr<CGImageRef> {
258 auto physicalSubimageRect = subimageRect;
259
260 if (options.orientation() != ImageOrientation::None) {
261 // subimageRect is in logical coordinates. getSubimage() deals with none-oriented
262 // image. We need to convert subimageRect to physical image coordinates.
263 if (auto transform = options.orientation().transformFromDefault(imageSize).inverse())
264 physicalSubimageRect = transform.value().mapRect(physicalSubimageRect);
265 }
266
267#if CACHE_SUBIMAGES
268 return SubimageCacheWithTimer::getSubimage(image, physicalSubimageRect);
269#else
270 return adoptCF(CGImageCreateWithImageInRect(image, physicalSubimageRect));
271#endif
272 };
273
274 auto imageLogicalSize = [](CGImageRef image, const ImagePaintingOptions& options) -> FloatSize {
275 FloatSize size = FloatSize(CGImageGetWidth(image), CGImageGetHeight(image));
276 return options.orientation().usesWidthAsHeight() ? size.transposedSize() : size;
277 };
278
279 auto context = platformContext();
280 CGContextStateSaver stateSaver(context, false);
281 auto transform = CGContextGetCTM(context);
282
283 auto subImage = image;
284 auto currentImageSize = imageLogicalSize(image.get(), options);
285
286 auto adjustedDestRect = normalizedDestRect;
287
288 if (normalizedSrcRect != imageRect) {
289 CGInterpolationQuality interpolationQuality = CGContextGetInterpolationQuality(context);
290 auto scale = normalizedDestRect.size() / normalizedSrcRect.size();
291
292 if (shouldUseSubimage(interpolationQuality, normalizedDestRect, normalizedSrcRect, transform)) {
293 auto subimageRect = enclosingIntRect(normalizedSrcRect);
294
295 // When the image is scaled using high-quality interpolation, we create a temporary CGImage
296 // containing only the portion we want to display. We need to do this because high-quality
297 // interpolation smoothes sharp edges, causing pixels from outside the source rect to bleed
298 // into the destination rect. See <rdar://problem/6112909>.
299 subImage = getSubimage(subImage.get(), imageSize, subimageRect, options);
300 adjustedDestRect = enclosingIntRect(adjustedDestRect);
301
302 // If the image is only partially loaded, then shrink the destination rect that we're drawing
303 // into accordingly.
304 if (currentImageSize.height() < normalizedSrcRect.maxY()) {
305 auto currentSubimageSize = imageLogicalSize(subImage.get(), options);
306 adjustedDestRect.setHeight(currentSubimageSize.height() * scale.height());
307 }
308 } else {
309 // If the source rect is a subportion of the image, then we compute an inflated destination rect
310 // that will hold the entire image and then set a clip to the portion that we want to display.
311 adjustedDestRect = { adjustedDestRect.location() - toFloatSize(normalizedSrcRect.location()) * scale, imageSize * scale };
312 }
313
314 if (!normalizedDestRect.contains(adjustedDestRect)) {
315 stateSaver.save();
316 CGContextClipToRect(context, normalizedDestRect);
317 }
318 }
319
320 // If the image is only partially loaded, then shrink the destination rect that we're drawing into accordingly.
321 if (subImage == image && currentImageSize.height() < imageSize.height())
322 adjustedDestRect.setHeight(adjustedDestRect.height() * currentImageSize.height() / imageSize.height());
323
324#if PLATFORM(IOS_FAMILY)
325 bool wasAntialiased = CGContextGetShouldAntialias(context);
326 // Anti-aliasing is on by default on the iPhone. Need to turn it off when drawing images.
327 CGContextSetShouldAntialias(context, false);
328
329 // Align to pixel boundaries
330 adjustedDestRect = roundToDevicePixels(adjustedDestRect);
331#endif
332
333 auto oldCompositeOperator = compositeOperation();
334 auto oldBlendMode = blendModeOperation();
335 setCompositeOperation(options.compositeOperator(), options.blendMode());
336
337 // Make the origin be at adjustedDestRect.location()
338 CGContextTranslateCTM(context, adjustedDestRect.x(), adjustedDestRect.y());
339 adjustedDestRect.setLocation(FloatPoint::zero());
340
341 if (options.orientation() != ImageOrientation::None) {
342 CGContextConcatCTM(context, options.orientation().transformFromDefault(adjustedDestRect.size()));
343
344 // The destination rect will have its width and height already reversed for the orientation of
345 // the image, as it was needed for page layout, so we need to reverse it back here.
346 if (options.orientation().usesWidthAsHeight())
347 adjustedDestRect = adjustedDestRect.transposedRect();
348 }
349
350 // Flip the coords.
351 CGContextTranslateCTM(context, 0, adjustedDestRect.height());
352 CGContextScaleCTM(context, 1, -1);
353
354 // Draw the image.
355 CGContextDrawImage(context, adjustedDestRect, subImage.get());
356
357 if (!stateSaver.didSave()) {
358 CGContextSetCTM(context, transform);
359#if PLATFORM(IOS_FAMILY)
360 CGContextSetShouldAntialias(context, wasAntialiased);
361#endif
362 setCompositeOperation(oldCompositeOperator, oldBlendMode);
363 }
364
365 LOG_WITH_STREAM(Images, stream << "GraphicsContextCG::drawNativeImage " << image.get() << " size " << imageSize << " into " << destRect << " took " << (MonotonicTime::now() - startTime).milliseconds() << "ms");
366}

368static void drawPatternCallback(void* info, CGContextRef context)
369{
370 CGImageRef image = (CGImageRef)info;
371 CGFloat height = CGImageGetHeight(image);
372#if PLATFORM(IOS_FAMILY)
373 CGContextScaleCTM(context, 1, -1);
374 CGContextTranslateCTM(context, 0, -height);
375#endif
376 CGContextDrawImage(context, GraphicsContextCG(context).roundToDevicePixels(FloatRect(0, 0, CGImageGetWidth(image), height)), image);
377}

384void GraphicsContextCG::drawPattern(NativeImage& nativeImage, const FloatSize& imageSize, const FloatRect& destRect, const FloatRect& tileRect, const AffineTransform& patternTransform, const FloatPoint& phase, const FloatSize& spacing, const ImagePaintingOptions& options)
385{
386 if (!patternTransform.isInvertible())
387 return;
388
389 auto image = nativeImage.platformImage();
390
391 CGContextRef context = platformContext();
392 CGContextStateSaver stateSaver(context);
393 CGContextClipToRect(context, destRect);
394
395 setCompositeOperation(options.compositeOperator(), options.blendMode());
396
397 CGContextTranslateCTM(context, destRect.x(), destRect.y() + destRect.height());
398 CGContextScaleCTM(context, 1, -1);
399
400 // Compute the scaled tile size.
401 float scaledTileHeight = tileRect.height() * narrowPrecisionToFloat(patternTransform.d());
402
403 // We have to adjust the phase to deal with the fact we're in Cartesian space now (with the bottom left corner of destRect being
404 // the origin).
405 float adjustedX = phase.x() - destRect.x() + tileRect.x() * narrowPrecisionToFloat(patternTransform.a()); // We translated the context so that destRect.x() is the origin, so subtract it out.
406 float adjustedY = destRect.height() - (phase.y() - destRect.y() + tileRect.y() * narrowPrecisionToFloat(patternTransform.d()) + scaledTileHeight);
407
408 float h = CGImageGetHeight(image.get());
409
410 RetainPtr<CGImageRef> subImage;
411 if (tileRect.size() == imageSize)
412 subImage = image;
413 else {
414 // Copying a sub-image out of a partially-decoded image stops the decoding of the original image. It should never happen
415 // because sub-images are only used for border-image, which only renders when the image is fully decoded.
416 ASSERT(h == imageSize.height());
417 subImage = adoptCF(CGImageCreateWithImageInRect(image.get(), tileRect));
418 }
419
420 // If we need to paint gaps between tiles because we have a partially loaded image or non-zero spacing,
421 // fall back to the less efficient CGPattern-based mechanism.
422 float scaledTileWidth = tileRect.width() * narrowPrecisionToFloat(patternTransform.a());
423 float w = CGImageGetWidth(image.get());
424 if (w == imageSize.width() && h == imageSize.height() && !spacing.width() && !spacing.height()) {
425 // FIXME: CG seems to snap the images to integral sizes. When we care (e.g. with border-image-repeat: round),
426 // we should tile all but the last, and stretch the last image to fit.
427 CGContextDrawTiledImage(context, FloatRect(adjustedX, adjustedY, scaledTileWidth, scaledTileHeight), subImage.get());
428 } else {
429 static const CGPatternCallbacks patternCallbacks = { 0, drawPatternCallback, patternReleaseCallback };
430 CGAffineTransform matrix = CGAffineTransformMake(narrowPrecisionToCGFloat(patternTransform.a()), 0, 0, narrowPrecisionToCGFloat(patternTransform.d()), adjustedX, adjustedY);
431 matrix = CGAffineTransformConcat(matrix, CGContextGetCTM(context));
432 // The top of a partially-decoded image is drawn at the bottom of the tile. Map it to the top.
433 matrix = CGAffineTransformTranslate(matrix, 0, imageSize.height() - h);
434#if PLATFORM(IOS_FAMILY)
435 matrix = CGAffineTransformScale(matrix, 1, -1);
436 matrix = CGAffineTransformTranslate(matrix, 0, -h);
437#endif
438 CGImageRef platformImage = CGImageRetain(subImage.get());
439 RetainPtr<CGPatternRef> pattern = adoptCF(CGPatternCreate(platformImage, CGRectMake(0, 0, tileRect.width(), tileRect.height()), matrix,
440 tileRect.width() + spacing.width() * (1 / narrowPrecisionToFloat(patternTransform.a())),
441 tileRect.height() + spacing.height() * (1 / narrowPrecisionToFloat(patternTransform.d())),
442 kCGPatternTilingConstantSpacing, true, &patternCallbacks));
443
444 if (!pattern)
445 return;
446
447 RetainPtr<CGColorSpaceRef> patternSpace = adoptCF(CGColorSpaceCreatePattern(nullptr));
448
449 CGFloat alpha = 1;
450 RetainPtr<CGColorRef> color = adoptCF(CGColorCreateWithPattern(patternSpace.get(), pattern.get(), &alpha));
451 CGContextSetFillColorSpace(context, patternSpace.get());
452
453 CGContextSetBaseCTM(context, CGAffineTransformIdentity);
454 CGContextSetPatternPhase(context, CGSizeZero);
455
456 CGContextSetFillColorWithColor(context, color.get());
457 CGContextFillRect(context, CGContextGetClipBoundingBox(context)); // FIXME: we know the clip; we set it above.
458 }
459}

461// Draws a filled rectangle with a stroked border.
462void GraphicsContextCG::drawRect(const FloatRect& rect, float borderThickness)
463{
464 // FIXME: this function does not handle patterns and gradients like drawPath does, it probably should.
465 ASSERT(!rect.isEmpty());
466
467 CGContextRef context = platformContext();
468
469 CGContextFillRect(context, rect);
470
471 if (strokeStyle() != NoStroke) {
472 // We do a fill of four rects to simulate the stroke of a border.
473 Color oldFillColor = fillColor();
474 if (oldFillColor != strokeColor())
475 setCGFillColor(context, strokeColor());
476 CGRect rects[4] = {
477 FloatRect(rect.x(), rect.y(), rect.width(), borderThickness),
478 FloatRect(rect.x(), rect.maxY() - borderThickness, rect.width(), borderThickness),
479 FloatRect(rect.x(), rect.y() + borderThickness, borderThickness, rect.height() - 2 * borderThickness),
480 FloatRect(rect.maxX() - borderThickness, rect.y() + borderThickness, borderThickness, rect.height() - 2 * borderThickness)
481 };
482 CGContextFillRects(context, rects, 4);
483 if (oldFillColor != strokeColor())
484 setCGFillColor(context, oldFillColor);
485 }
486}

488// This is only used to draw borders.
489void GraphicsContextCG::drawLine(const FloatPoint& point1, const FloatPoint& point2)
490{
491 if (strokeStyle() == NoStroke)
492 return;
493
494 float thickness = strokeThickness();
495 bool isVerticalLine = (point1.x() + thickness == point2.x());
496 float strokeWidth = isVerticalLine ? point2.y() - point1.y() : point2.x() - point1.x();
497 if (!thickness || !strokeWidth)
498 return;
499
500 CGContextRef context = platformContext();
501
502 StrokeStyle strokeStyle = this->strokeStyle();
503 float cornerWidth = 0;
504 bool drawsDashedLine = strokeStyle == DottedStroke || strokeStyle == DashedStroke;
505
506 CGContextStateSaver stateSaver(context, drawsDashedLine);
507 if (drawsDashedLine) {
508 // Figure out end points to ensure we always paint corners.
509 cornerWidth = dashedLineCornerWidthForStrokeWidth(strokeWidth);
510 setCGFillColor(context, strokeColor());
511 if (isVerticalLine) {
512 CGContextFillRect(context, FloatRect(point1.x(), point1.y(), thickness, cornerWidth));
513 CGContextFillRect(context, FloatRect(point1.x(), point2.y() - cornerWidth, thickness, cornerWidth));
514 } else {
515 CGContextFillRect(context, FloatRect(point1.x(), point1.y(), cornerWidth, thickness));
516 CGContextFillRect(context, FloatRect(point2.x() - cornerWidth, point1.y(), cornerWidth, thickness));
517 }
518 strokeWidth -= 2 * cornerWidth;
519 float patternWidth = dashedLinePatternWidthForStrokeWidth(strokeWidth);
520 // Check if corner drawing sufficiently covers the line.
521 if (strokeWidth <= patternWidth + 1)
522 return;
523
524 float patternOffset = dashedLinePatternOffsetForPatternAndStrokeWidth(patternWidth, strokeWidth);
525 const CGFloat dashedLine[2] = { static_cast<CGFloat>(patternWidth), static_cast<CGFloat>(patternWidth) };
526 CGContextSetLineDash(context, patternOffset, dashedLine, 2);
527 }
528
529 auto centeredPoints = centerLineAndCutOffCorners(isVerticalLine, cornerWidth, point1, point2);
530 auto p1 = centeredPoints[0];
531 auto p2 = centeredPoints[1];
532
533 if (shouldAntialias()) {
534#if PLATFORM(IOS_FAMILY)
535 // Force antialiasing on for line patterns as they don't look good with it turned off (<rdar://problem/5459772>).
536 CGContextSetShouldAntialias(context, strokeStyle == DottedStroke || strokeStyle == DashedStroke);
537#else
538 CGContextSetShouldAntialias(context, false);
539#endif
540 }
541 CGContextBeginPath(context);
542 CGContextMoveToPoint(context, p1.x(), p1.y());
543 CGContextAddLineToPoint(context, p2.x(), p2.y());
544 CGContextStrokePath(context);
545 if (shouldAntialias())
546 CGContextSetShouldAntialias(context, true);
547}

594static inline bool calculateDrawingMode(const GraphicsContextState& state, CGPathDrawingMode& mode)
595{
596 bool shouldFill = state.fillPattern || state.fillColor.isVisible();
597 bool shouldStroke = state.strokePattern || (state.strokeStyle != NoStroke && state.strokeColor.isVisible());
598 bool useEOFill = state.fillRule == WindRule::EvenOdd;
599
600 if (shouldFill) {
601 if (shouldStroke) {
602 if (useEOFill)
603 mode = kCGPathEOFillStroke;
604 else
605 mode = kCGPathFillStroke;
606 } else { // fill, no stroke
607 if (useEOFill)
608 mode = kCGPathEOFill;
609 else
610 mode = kCGPathFill;
611 }
612 } else {
613 // Setting mode to kCGPathStroke even if shouldStroke is false. In that case, we return false and mode will not be used,
614 // but the compiler will not complain about an uninitialized variable.
615 mode = kCGPathStroke;
616 }
617
618 return shouldFill || shouldStroke;
619}

621void GraphicsContextCG::drawPath(const Path& path)
622{
623 if (path.isEmpty())
624 return;
625
626 CGContextRef context = platformContext();
627 const GraphicsContextState& state = m_state;
628
629 if (state.fillGradient || state.strokeGradient) {
630 // We don't have any optimized way to fill & stroke a path using gradients
631 // FIXME: Be smarter about this.
632 fillPath(path);
633 strokePath(path);
634 return;
635 }
636
637 if (state.fillPattern)
638 applyFillPattern();
639 if (state.strokePattern)
640 applyStrokePattern();
641
642 CGPathDrawingMode drawingMode;
643 if (calculateDrawingMode(state, drawingMode)) {

713void GraphicsContextCG::strokePath(const Path& path)
714{
715 if (path.isEmpty())
716 return;
717
718 CGContextRef context = platformContext();
719
720 if (m_state.strokeGradient) {
721 if (hasShadow()) {
722 FloatRect rect = path.fastBoundingRect();
723 float lineWidth = strokeThickness();
724 float doubleLineWidth = lineWidth * 2;
725 float adjustedWidth = ceilf(rect.width() + doubleLineWidth);
726 float adjustedHeight = ceilf(rect.height() + doubleLineWidth);
727
728 FloatSize layerSize = getCTM().mapSize(FloatSize(adjustedWidth, adjustedHeight));
729
730 auto layer = adoptCF(CGLayerCreateWithContext(context, layerSize, 0));
731 CGContextRef layerContext = CGLayerGetContext(layer.get());
732 CGContextSetLineWidth(layerContext, lineWidth);
733
734 // Compensate for the line width, otherwise the layer's top-left corner would be
735 // aligned with the rect's top-left corner. This would result in leaving pixels out of
736 // the layer on the left and top sides.
737 float translationX = lineWidth - rect.x();
738 float translationY = lineWidth - rect.y();
739 CGContextScaleCTM(layerContext, layerSize.width() / adjustedWidth, layerSize.height() / adjustedHeight);
740 CGContextTranslateCTM(layerContext, translationX, translationY);
741
742 CGContextAddPath(layerContext, path.platformPath());
743 CGContextReplacePathWithStrokedPath(layerContext);
744 CGContextClip(layerContext);
745 CGContextConcatCTM(layerContext, m_state.strokeGradientSpaceTransform);
746 m_state.strokeGradient->paint(layerContext);
747
748 float destinationX = roundf(rect.x() - lineWidth);
749 float destinationY = roundf(rect.y() - lineWidth);
750 CGContextDrawLayerInRect(context, CGRectMake(destinationX, destinationY, adjustedWidth, adjustedHeight), layer.get());
751 } else {
752 CGContextStateSaver stateSaver(context);
753 CGContextBeginPath(context);
754 CGContextAddPath(context, path.platformPath());
755 CGContextReplacePathWithStrokedPath(context);
756 CGContextClip(context);
757 CGContextConcatCTM(context, m_state.strokeGradientSpaceTransform);
758 m_state.strokeGradient->paint(*this);
759 }
760 return;
761 }
762
763 if (m_state.strokePattern)
764 applyStrokePattern();
765
766#if USE(CG_CONTEXT_STROKE_LINE_SEGMENTS_WHEN_STROKING_PATH)
767 if (path.hasInlineData<LineData>()) {
768 auto& lineData = path.inlineData<LineData>();
769 CGPoint points[2] { lineData.start, lineData.end };
770 CGContextStrokeLineSegments(context, points, 2);
771 return;
772 }
773#endif
774
775#if HAVE(CG_CONTEXT_DRAW_PATH_DIRECT)
776 CGContextDrawPathDirect(context, kCGPathStroke, path.platformPath(), nullptr);
777#else
778 CGContextBeginPath(context);
779 CGContextAddPath(context, path.platformPath());
780 CGContextStrokePath(context);
781#endif
782}

855void GraphicsContextCG::fillRoundedRectImpl(const FloatRoundedRect& rect, const Color& color)
856{
857 CGContextRef context = platformContext();
858 Color oldFillColor = fillColor();
859
860 if (oldFillColor != color)
861 setCGFillColor(context, color);
862
863 bool drawOwnShadow = (renderingMode() == RenderingMode::Unaccelerated) && hasBlurredShadow() && !m_state.shadowsIgnoreTransforms;
864 CGContextStateSaver stateSaver(context, drawOwnShadow);
865 if (drawOwnShadow) {
866 // Turn off CG shadows.
867 CGContextSetShadowWithColor(platformContext(), CGSizeZero, 0, 0);
868
869 ShadowBlur contextShadow(m_state);
870 contextShadow.drawRectShadow(*this, rect);
871 }
872
873 const FloatRect& r = rect.rect();
874 const FloatRoundedRect::Radii& radii = rect.radii();
875 bool equalWidths = (radii.topLeft().width() == radii.topRight().width() && radii.topRight().width() == radii.bottomLeft().width() && radii.bottomLeft().width() == radii.bottomRight().width());
876 bool equalHeights = (radii.topLeft().height() == radii.bottomLeft().height() && radii.bottomLeft().height() == radii.topRight().height() && radii.topRight().height() == radii.bottomRight().height());
877 bool hasCustomFill = m_state.fillGradient || m_state.fillPattern;
878 if (!hasCustomFill && equalWidths && equalHeights && radii.topLeft().width() * 2 == r.width() && radii.topLeft().height() * 2 == r.height())
879 CGContextFillEllipseInRect(context, r);
880 else {
881 Path path;
882 path.addRoundedRect(rect);
883 fillPath(path);
884 }
885
886 if (drawOwnShadow)
887 stateSaver.restore();
888
889 if (oldFillColor != color)
890 setCGFillColor(context, oldFillColor);
891}

1005static void applyShadowOffsetWorkaroundIfNeeded(const GraphicsContext& context, CGFloat& xOffset, CGFloat& yOffset)
1006{
1007#if PLATFORM(IOS_FAMILY) || PLATFORM(WIN)
1008 UNUSED_PARAM(context);
1009 UNUSED_PARAM(xOffset);
1010 UNUSED_PARAM(yOffset);
1011#else
1012 if (context.renderingMode() == RenderingMode::Accelerated)
1013 return;
1014
1015 if (CGContextDrawsWithCorrectShadowOffsets(context.platformContext()))
1016 return;
1017
1018 // Work around <rdar://problem/5539388> by ensuring that the offsets will get truncated
1019 // to the desired integer. Also see: <rdar://problem/10056277>
1020 static const CGFloat extraShadowOffset = narrowPrecisionToCGFloat(1.0 / 128);
1021 if (xOffset > 0)
1022 xOffset += extraShadowOffset;
1023 else if (xOffset < 0)
1024 xOffset -= extraShadowOffset;
1025
1026 if (yOffset > 0)
1027 yOffset += extraShadowOffset;
1028 else if (yOffset < 0)
1029 yOffset -= extraShadowOffset;
1030#endif
1031}

1033static void setCGShadow(const GraphicsContext& graphicsContext, const FloatSize& offset, float blur, const Color& color)
1034{
1035 CGContextRef context = graphicsContext.platformContext();
1036
1037 if (offset.isZero() && !blur) {
1038 CGContextSetShadowWithColor(context, CGSizeZero, 0, 0);
1039 return;
1040 }
1041
1042 // FIXME: we could avoid the shadow setup cost when we know we'll render the shadow ourselves.
1043
1044 CGFloat xOffset = offset.width();
1045 CGFloat yOffset = offset.height();
1046 CGFloat blurRadius = blur;
1047
1048 if (!graphicsContext.shadowsIgnoreTransforms()) {
1049 CGAffineTransform userToBaseCTM = getUserToBaseCTM(context);
1050
1051 CGFloat A = userToBaseCTM.a * userToBaseCTM.a + userToBaseCTM.b * userToBaseCTM.b;
1052 CGFloat B = userToBaseCTM.a * userToBaseCTM.c + userToBaseCTM.b * userToBaseCTM.d;
1053 CGFloat C = B;
1054 CGFloat D = userToBaseCTM.c * userToBaseCTM.c + userToBaseCTM.d * userToBaseCTM.d;
1055
1056 CGFloat smallEigenvalue = narrowPrecisionToCGFloat(sqrt(0.5 * ((A + D) - sqrt(4 * B * C + (A - D) * (A - D)))));
1057
1058 blurRadius = blur * smallEigenvalue;
1059
1060 CGSize offsetInBaseSpace = CGSizeApplyAffineTransform(offset, userToBaseCTM);
1061
1062 xOffset = offsetInBaseSpace.width;
1063 yOffset = offsetInBaseSpace.height;
1064 }
1065
1066 // Extreme "blur" values can make text drawing crash or take crazy long times, so clamp
1067 blurRadius = std::min(blurRadius, narrowPrecisionToCGFloat(1000.0));
1068
1069 applyShadowOffsetWorkaroundIfNeeded(graphicsContext, xOffset, yOffset);
1070
1071 // Check for an invalid color, as this means that the color was not set for the shadow
1072 // and we should therefore just use the default shadow color.
1073 if (!color.isValid())
1074 CGContextSetShadow(context, CGSizeMake(xOffset, yOffset), blurRadius);
1075 else
1076 CGContextSetShadowWithColor(context, CGSizeMake(xOffset, yOffset), blurRadius, cachedCGColor(color));
1077}

1272
1273FloatRect GraphicsContextCG::roundToDevicePixels(const FloatRect& rect, RoundingMode roundingMode)
1274{
1275 // It is not enough just to round to pixels in device space. The rotation part of the
1276 // affine transform matrix to device space can mess with this conversion if we have a
1277 // rotating image like the hands of the world clock widget. We just need the scale, so
1278 // we get the affine transform matrix and extract the scale.
1279
1280 if (m_data->m_userToDeviceTransformKnownToBeIdentity)
1281 return roundedIntRect(rect);
1282
1283 CGAffineTransform deviceMatrix = CGContextGetUserSpaceToDeviceSpaceTransform(platformContext());
1284 if (CGAffineTransformIsIdentity(deviceMatrix)) {
1285 m_data->m_userToDeviceTransformKnownToBeIdentity = true;
1286 return roundedIntRect(rect);
1287 }
1288
1289 float deviceScaleX = std::hypot(deviceMatrix.a, deviceMatrix.b);
1290 float deviceScaleY = std::hypot(deviceMatrix.c, deviceMatrix.d);
1291
1292 CGPoint deviceOrigin = CGPointMake(rect.x() * deviceScaleX, rect.y() * deviceScaleY);
1293 CGPoint deviceLowerRight = CGPointMake((rect.x() + rect.width()) * deviceScaleX,
1294 (rect.y() + rect.height()) * deviceScaleY);
1295
1296 deviceOrigin.x = roundf(deviceOrigin.x);
1297 deviceOrigin.y = roundf(deviceOrigin.y);
1298 if (roundingMode == RoundAllSides) {
1299 deviceLowerRight.x = roundf(deviceLowerRight.x);
1300 deviceLowerRight.y = roundf(deviceLowerRight.y);
1301 } else {
1302 deviceLowerRight.x = deviceOrigin.x + roundf(rect.width() * deviceScaleX);
1303 deviceLowerRight.y = deviceOrigin.y + roundf(rect.height() * deviceScaleY);
1304 }
1305
1306 // Don't let the height or width round to 0 unless either was originally 0
1307 if (deviceOrigin.y == deviceLowerRight.y && rect.height())
1308 deviceLowerRight.y += 1;
1309 if (deviceOrigin.x == deviceLowerRight.x && rect.width())
1310 deviceLowerRight.x += 1;
1311
1312 FloatPoint roundedOrigin = FloatPoint(deviceOrigin.x / deviceScaleX, deviceOrigin.y / deviceScaleY);
1313 FloatPoint roundedLowerRight = FloatPoint(deviceLowerRight.x / deviceScaleX, deviceLowerRight.y / deviceScaleY);
1314 return FloatRect(roundedOrigin, roundedLowerRight - roundedOrigin);
1315}

1317void GraphicsContextCG::drawLinesForText(const FloatPoint& point, float thickness, const DashArray& widths, bool printing, bool doubleLines, StrokeStyle strokeStyle)
1318{
1319 if (!widths.size())
1320 return;
1321
1322 Color localStrokeColor(strokeColor());
1323
1324 FloatRect bounds = computeLineBoundsAndAntialiasingModeForText(FloatRect(point, FloatSize(widths.last(), thickness)), printing, localStrokeColor);
1325 if (bounds.isEmpty())
1326 return;
1327
1328 bool fillColorIsNotEqualToStrokeColor = fillColor() != localStrokeColor;
1329
1330 Vector<CGRect, 4> dashBounds;
1331 ASSERT(!(widths.size() % 2));
1332 dashBounds.reserveInitialCapacity(dashBounds.size() / 2);
1333
1334 float dashWidth = 0;
1335 switch (strokeStyle) {
1336 case DottedStroke:
1337 dashWidth = bounds.height();
1338 break;
1339 case DashedStroke:
1340 dashWidth = 2 * bounds.height();
1341 break;
1342 case SolidStroke:
1343 default:
1344 break;
1345 }
1346
1347 for (size_t i = 0; i < widths.size(); i += 2) {
1348 auto left = widths[i];
1349 auto width = widths[i+1] - widths[i];
1350 if (!dashWidth)
1351 dashBounds.append(CGRectMake(bounds.x() + left, bounds.y(), width, bounds.height()));
1352 else {
1353 auto startParticle = static_cast<int>(std::ceil(left / (2 * dashWidth)));
1354 auto endParticle = static_cast<int>((left + width) / (2 * dashWidth));
1355 for (auto j = startParticle; j < endParticle; ++j)
1356 dashBounds.append(CGRectMake(bounds.x() + j * 2 * dashWidth, bounds.y(), dashWidth, bounds.height()));
1357 }
1358 }
1359
1360 if (doubleLines) {
1361 // The space between double underlines is equal to the height of the underline
1362 for (size_t i = 0; i < widths.size(); i += 2)
1363 dashBounds.append(CGRectMake(bounds.x() + widths[i], bounds.y() + 2 * bounds.height(), widths[i+1] - widths[i], bounds.height()));
1364 }
1365
1366 if (fillColorIsNotEqualToStrokeColor)
1367 setCGFillColor(platformContext(), localStrokeColor);
1368
1369 CGContextFillRects(platformContext(), dashBounds.data(), dashBounds.size());
1370
1371 if (fillColorIsNotEqualToStrokeColor)
1372 setCGFillColor(platformContext(), fillColor());
1373}
