53enum SourceDataFormatBase {
54 SourceFormatBaseR = 0,
55 SourceFormatBaseA,
56 SourceFormatBaseRA,
57 SourceFormatBaseAR,
58 SourceFormatBaseRGB,
59 SourceFormatBaseRGBA,
60 SourceFormatBaseARGB,
61 SourceFormatBaseNumFormats
62};

64enum AlphaFormat {
65 AlphaFormatNone = 0,
66 AlphaFormatFirst,
67 AlphaFormatLast,
68 AlphaFormatNumFormats
69};

71// This returns SourceFormatNumFormats if the combination of input parameters is unsupported.
72static GraphicsContextGL::DataFormat getSourceDataFormat(unsigned componentsPerPixel, AlphaFormat alphaFormat, bool is16BitFormat, bool bigEndian)
73{
74 const static SourceDataFormatBase formatTableBase[4][AlphaFormatNumFormats] = { // componentsPerPixel x AlphaFormat
75 // AlphaFormatNone AlphaFormatFirst AlphaFormatLast
76 { SourceFormatBaseR, SourceFormatBaseA, SourceFormatBaseA }, // 1 componentsPerPixel
77 { SourceFormatBaseNumFormats, SourceFormatBaseAR, SourceFormatBaseRA }, // 2 componentsPerPixel
78 { SourceFormatBaseRGB, SourceFormatBaseNumFormats, SourceFormatBaseNumFormats }, // 3 componentsPerPixel
79 { SourceFormatBaseNumFormats, SourceFormatBaseARGB, SourceFormatBaseRGBA } // 4 componentsPerPixel
80 };
81 const static GraphicsContextGL::DataFormat formatTable[SourceFormatBaseNumFormats][4] = { // SourceDataFormat::Base x bitsPerComponent x endian
82 // 8bits, little endian 8bits, big endian 16bits, little endian 16bits, big endian
83 { GraphicsContextGL::DataFormat::R8, GraphicsContextGL::DataFormat::R8, GraphicsContextGL::DataFormat::R16Little, GraphicsContextGL::DataFormat::R16Big },
84 { GraphicsContextGL::DataFormat::A8, GraphicsContextGL::DataFormat::A8, GraphicsContextGL::DataFormat::A16Little, GraphicsContextGL::DataFormat::A16Big },
85 { GraphicsContextGL::DataFormat::AR8, GraphicsContextGL::DataFormat::RA8, GraphicsContextGL::DataFormat::RA16Little, GraphicsContextGL::DataFormat::RA16Big },
86 { GraphicsContextGL::DataFormat::RA8, GraphicsContextGL::DataFormat::AR8, GraphicsContextGL::DataFormat::AR16Little, GraphicsContextGL::DataFormat::AR16Big },
87 { GraphicsContextGL::DataFormat::BGR8, GraphicsContextGL::DataFormat::RGB8, GraphicsContextGL::DataFormat::RGB16Little, GraphicsContextGL::DataFormat::RGB16Big },
88 { GraphicsContextGL::DataFormat::ABGR8, GraphicsContextGL::DataFormat::RGBA8, GraphicsContextGL::DataFormat::RGBA16Little, GraphicsContextGL::DataFormat::RGBA16Big },
89 { GraphicsContextGL::DataFormat::BGRA8, GraphicsContextGL::DataFormat::ARGB8, GraphicsContextGL::DataFormat::ARGB16Little, GraphicsContextGL::DataFormat::ARGB16Big }
90 };
91
92 ASSERT(componentsPerPixel <= 4 && componentsPerPixel > 0);
93 SourceDataFormatBase formatBase = formatTableBase[componentsPerPixel - 1][alphaFormat];
94 if (formatBase == SourceFormatBaseNumFormats)
95 return GraphicsContextGL::DataFormat::NumFormats;
96 return formatTable[formatBase][(is16BitFormat ? 2 : 0) + (bigEndian ? 1 : 0)];
97}

100uint8_t convertColor16LittleTo8(uint16_t value)
101{
102 return value >> 8;
103}
104
105uint8_t convertColor16BigTo8(uint16_t value)
106{
107 return static_cast<uint8_t>(value & 0x00FF);
108}

116template<> ALWAYS_INLINE void convert16BitFormatToRGBA8<GraphicsContextGL::DataFormat::RGBA16Little, uint16_t, uint8_t>(const uint16_t* source, uint8_t* destination, unsigned pixelsPerRow)
117{
118#if HAVE(ARM_NEON_INTRINSICS)
119 SIMD::unpackOneRowOfRGBA16LittleToRGBA8(source, destination, pixelsPerRow);
120#endif
121 for (unsigned i = 0; i < pixelsPerRow; ++i) {
122 destination[0] = convertColor16LittleTo8(source[0]);
123 destination[1] = convertColor16LittleTo8(source[1]);
124 destination[2] = convertColor16LittleTo8(source[2]);
125 destination[3] = convertColor16LittleTo8(source[3]);
126 source += 4;
127 destination += 4;
128 }
129}

293void convert16BitFormatToRGBA8(GraphicsContextGL::DataFormat srcFormat, const uint16_t* source, uint8_t* destination, unsigned pixelsPerRow)
294{
295#define CONVERT16BITFORMATTORGBA8(SrcFormat) \
296 case SrcFormat: \
297 return convert16BitFormatToRGBA8<SrcFormat>(source, destination, pixelsPerRow);
298
299 switch (srcFormat) {
300 CONVERT16BITFORMATTORGBA8(GraphicsContextGL::DataFormat::R16Little)
301 CONVERT16BITFORMATTORGBA8(GraphicsContextGL::DataFormat::R16Big)
302 CONVERT16BITFORMATTORGBA8(GraphicsContextGL::DataFormat::A16Little)
303 CONVERT16BITFORMATTORGBA8(GraphicsContextGL::DataFormat::A16Big)
304 CONVERT16BITFORMATTORGBA8(GraphicsContextGL::DataFormat::RA16Little)
305 CONVERT16BITFORMATTORGBA8(GraphicsContextGL::DataFormat::RA16Big)
306 CONVERT16BITFORMATTORGBA8(GraphicsContextGL::DataFormat::AR16Little)
307 CONVERT16BITFORMATTORGBA8(GraphicsContextGL::DataFormat::AR16Big)
308 CONVERT16BITFORMATTORGBA8(GraphicsContextGL::DataFormat::RGB16Little)
309 CONVERT16BITFORMATTORGBA8(GraphicsContextGL::DataFormat::RGB16Big)
310 CONVERT16BITFORMATTORGBA8(GraphicsContextGL::DataFormat::RGBA16Little)
311 CONVERT16BITFORMATTORGBA8(GraphicsContextGL::DataFormat::RGBA16Big)
312 CONVERT16BITFORMATTORGBA8(GraphicsContextGL::DataFormat::ARGB16Little)
313 CONVERT16BITFORMATTORGBA8(GraphicsContextGL::DataFormat::ARGB16Big)
314 default:
315 ASSERT_NOT_REACHED();
316 }
317#undef CONVERT16BITFORMATTORGBA8
318}

324bool GraphicsContextGLImageExtractor::extractImage(bool premultiplyAlpha, bool ignoreGammaAndColorProfile, bool ignoreNativeImageAlphaPremultiplication)
325{
326 if (!m_image)
327 return false;
328
329 RefPtr<NativeImage> decodedImage;
330 bool hasAlpha = !m_image->currentFrameKnownToBeOpaque();
331
332 if ((ignoreGammaAndColorProfile || (hasAlpha && !premultiplyAlpha)) && m_image->data()) {
333 auto source = ImageSource::create(nullptr, AlphaOption::NotPremultiplied, ignoreGammaAndColorProfile ? GammaAndColorProfileOption::Ignored : GammaAndColorProfileOption::Applied);
334 source->setData(m_image->data(), true);
335 if (!source->frameCount())
336 return false;
337
338 decodedImage = source->createFrameImageAtIndex(0);
339 } else
340 decodedImage = m_image->nativeImageForCurrentFrame();
341
342 if (!decodedImage)
343 return false;
344
345 m_imageWidth = CGImageGetWidth(decodedImage->platformImage().get());
346 m_imageHeight = CGImageGetHeight(decodedImage->platformImage().get());
347 if (!m_imageWidth || !m_imageHeight)
348 return false;
349
350 // See whether the image is using an indexed color space, and if
351 // so, re-render it into an RGB color space. The image re-packing
352 // code requires color data, not color table indices, for the
353 // image data.
354 CGColorSpaceRef colorSpace = CGImageGetColorSpace(decodedImage->platformImage().get());
355 CGColorSpaceModel model = CGColorSpaceGetModel(colorSpace);
356 if (model == kCGColorSpaceModelIndexed) {
357 RetainPtr<CGContextRef> bitmapContext;
358 // FIXME: we should probably manually convert the image by indexing into
359 // the color table, which would allow us to avoid premultiplying the
360 // alpha channel. Creation of a bitmap context with an alpha channel
361 // doesn't seem to work unless it's premultiplied.
362 bitmapContext = adoptCF(CGBitmapContextCreate(0, m_imageWidth, m_imageHeight, 8, m_imageWidth * 4,
363 sRGBColorSpaceRef(), kCGImageAlphaPremultipliedFirst | kCGBitmapByteOrder32Host));
364 if (!bitmapContext)
365 return false;
366
367 CGContextSetBlendMode(bitmapContext.get(), kCGBlendModeCopy);
368 CGContextSetInterpolationQuality(bitmapContext.get(), kCGInterpolationNone);
369 CGContextDrawImage(bitmapContext.get(), CGRectMake(0, 0, m_imageWidth, m_imageHeight), decodedImage->platformImage().get());
370
371 // Now discard the original CG image and replace it with a copy from the bitmap context.
372 decodedImage = NativeImage::create(adoptCF(CGBitmapContextCreateImage(bitmapContext.get())));
373 }
374
375 if (!decodedImage)
376 return false;
377
378 size_t bitsPerComponent = CGImageGetBitsPerComponent(decodedImage->platformImage().get());
379 size_t bitsPerPixel = CGImageGetBitsPerPixel(decodedImage->platformImage().get());
380 if (bitsPerComponent != 8 && bitsPerComponent != 16)
381 return false;
382 if (bitsPerPixel % bitsPerComponent)
383 return false;
384 size_t componentsPerPixel = bitsPerPixel / bitsPerComponent;
385
386 CGBitmapInfo bitInfo = CGImageGetBitmapInfo(decodedImage->platformImage().get());
387 bool bigEndianSource = false;
388 // These could technically be combined into one large switch
389 // statement, but we prefer not to so that we fail fast if we
390 // encounter an unexpected image configuration.
391 if (bitsPerComponent == 16) {
392 switch (bitInfo & kCGBitmapByteOrderMask) {
393 case kCGBitmapByteOrder16Big:
394 bigEndianSource = true;
395 break;
396 case kCGBitmapByteOrder16Little:
397 bigEndianSource = false;
398 break;
399 case kCGBitmapByteOrderDefault:
400 // This is a bug in earlier version of cg where the default endian
401 // is little whereas the decoded 16-bit png image data is actually
402 // Big. Later version (10.6.4) no longer returns ByteOrderDefault.
403 bigEndianSource = true;
404 break;
405 default:
406 return false;
407 }
408 } else {
409 switch (bitInfo & kCGBitmapByteOrderMask) {
410 case kCGBitmapByteOrder32Big:
411 bigEndianSource = true;
412 break;
413 case kCGBitmapByteOrder32Little:
414 bigEndianSource = false;
415 break;
416 case kCGBitmapByteOrderDefault:
417 // It appears that the default byte order is actually big
418 // endian even on little endian architectures.
419 bigEndianSource = true;
420 break;
421 default:
422 return false;
423 }
424 }
425
426 m_alphaOp = AlphaOp::DoNothing;
427 AlphaFormat alphaFormat = AlphaFormatNone;
428 switch (CGImageGetAlphaInfo(decodedImage->platformImage().get())) {
429 case kCGImageAlphaPremultipliedFirst:
430 if (!premultiplyAlpha)
431 m_alphaOp = AlphaOp::DoUnmultiply;
432 else if (ignoreNativeImageAlphaPremultiplication)
433 m_alphaOp = AlphaOp::DoPremultiply;
434 alphaFormat = AlphaFormatFirst;
435 break;
436 case kCGImageAlphaFirst:
437 // This path is only accessible for MacOS earlier than 10.6.4.
438 if (premultiplyAlpha)
439 m_alphaOp = AlphaOp::DoPremultiply;
440 alphaFormat = AlphaFormatFirst;
441 break;
442 case kCGImageAlphaNoneSkipFirst:
443 // This path is only accessible for MacOS earlier than 10.6.4.
444 alphaFormat = AlphaFormatFirst;
445 break;
446 case kCGImageAlphaPremultipliedLast:
447 if (!premultiplyAlpha)
448 m_alphaOp = AlphaOp::DoUnmultiply;
449 else if (ignoreNativeImageAlphaPremultiplication)
450 m_alphaOp = AlphaOp::DoPremultiply;
451 alphaFormat = AlphaFormatLast;
452 break;
453 case kCGImageAlphaLast:
454 if (premultiplyAlpha)
455 m_alphaOp = AlphaOp::DoPremultiply;
456 alphaFormat = AlphaFormatLast;
457 break;
458 case kCGImageAlphaNoneSkipLast:
459 alphaFormat = AlphaFormatLast;
460 break;
461 case kCGImageAlphaNone:
462 alphaFormat = AlphaFormatNone;
463 break;
464 default:
465 return false;
466 }
467
468 m_imageSourceFormat = getSourceDataFormat(componentsPerPixel, alphaFormat, bitsPerComponent == 16, bigEndianSource);
469 if (m_imageSourceFormat == DataFormat::NumFormats)
470 return false;
471
472 m_pixelData = adoptCF(CGDataProviderCopyData(CGImageGetDataProvider(decodedImage->platformImage().get())));
473 if (!m_pixelData)
474 return false;
475
476 m_imagePixelData = reinterpret_cast<const void*>(CFDataGetBytePtr(m_pixelData.get()));
477
478 unsigned srcUnpackAlignment = 0;
479 size_t bytesPerRow = CGImageGetBytesPerRow(decodedImage->platformImage().get());
480 unsigned padding = bytesPerRow - bitsPerPixel / 8 * m_imageWidth;
481 if (padding) {
482 srcUnpackAlignment = padding + 1;
483 while (bytesPerRow % srcUnpackAlignment)
484 ++srcUnpackAlignment;
485 }
486
487 m_imageSourceUnpackAlignment = srcUnpackAlignment;
488 // Using a bitmap context created according to destination format and drawing the CGImage to the bitmap context can also do the format conversion,
489 // but it would premultiply the alpha channel as a side effect.
490 // Prefer to mannually Convert 16bit per-component formats to RGBA8 formats instead.
491 if (bitsPerComponent == 16) {
492 m_formalizedRGBA8Data = makeUniqueArray<uint8_t>(Checked<size_t>(m_imageWidth) * m_imageHeight * 4U);
493 const uint16_t* source = reinterpret_cast<const uint16_t*>(m_imagePixelData);
494 uint8_t* destination = m_formalizedRGBA8Data.get();
495 const ptrdiff_t srcStrideInElements = bytesPerRow / sizeof(uint16_t);
496 const ptrdiff_t dstStrideInElements = 4 * m_imageWidth;
497 for (unsigned i =0; i < m_imageHeight; i++) {
498 convert16BitFormatToRGBA8(m_imageSourceFormat, source, destination, m_imageWidth);
499 source += srcStrideInElements;
500 destination += dstStrideInElements;
501 }
502 m_imagePixelData = reinterpret_cast<const void*>(m_formalizedRGBA8Data.get());
503 m_imageSourceFormat = DataFormat::RGBA8;
504 m_imageSourceUnpackAlignment = 1;
505 }
506 return true;
507}

509void GraphicsContextGLOpenGL::paintToCanvas(const GraphicsContextGLAttributes& sourceContextAttributes, PixelBuffer&& pixelBuffer, const IntSize& canvasSize, GraphicsContext& context)
510{
511 ASSERT(!pixelBuffer.size().isEmpty());
512 if (canvasSize.isEmpty())
513 return;
514 // Input is GL_RGBA == kCGBitmapByteOrder32Big | kCGImageAlpha*Last.
515 // GL_BGRA would be kCGBitmapByteOrder32Little | kCGImageAlpha*First.
516 CGBitmapInfo bitmapInfo = kCGBitmapByteOrder32Big;
517 if (!sourceContextAttributes.alpha)
518 bitmapInfo |= kCGImageAlphaNoneSkipLast;
519 else if (sourceContextAttributes.premultipliedAlpha)
520 bitmapInfo |= kCGImageAlphaPremultipliedLast;
521 else
522 bitmapInfo |= kCGImageAlphaLast;
523
524 auto pixelArray = makeRef(pixelBuffer.data());
525 auto dataSize = pixelArray->byteLength();
526 auto data = pixelArray->data();
527
528 verifyImageBufferIsBigEnough(data, dataSize);
529
530 auto dataProvider = adoptCF(CGDataProviderCreateWithData(&pixelArray.leakRef(), data, dataSize, [] (void* context, const void*, size_t) {
531 static_cast<JSC::Uint8ClampedArray*>(context)->deref();
532 }));
533
534 auto imageSize = pixelBuffer.size();
535 auto image = NativeImage::create(adoptCF(CGImageCreate(imageSize.width(), imageSize.height(), 8, 32, 4 * imageSize.width(), pixelBuffer.format().colorSpace.platformColorSpace(), bitmapInfo, dataProvider.get(), 0, false, kCGRenderingIntentDefault)));
536
537 // CSS styling may cause the canvas's content to be resized on
538 // the page. Go back to the Canvas to figure out the correct
539 // width and height to draw.
540 FloatRect canvasRect(FloatPoint(), canvasSize);
541 // We want to completely overwrite the previous frame's
542 // rendering results.
543
544 GraphicsContextStateSaver stateSaver(context);
545 context.scale(FloatSize(1, -1));
546 context.translate(0, -imageSize.height());
547 context.setImageInterpolationQuality(InterpolationQuality::DoNotInterpolate);
548 context.drawNativeImage(*image, imageSize, canvasRect, FloatRect(FloatPoint(), imageSize), { CompositeOperator::Copy });
549}
