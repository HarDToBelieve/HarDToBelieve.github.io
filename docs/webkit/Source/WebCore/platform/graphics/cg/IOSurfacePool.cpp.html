63static bool surfaceMatchesParameters(IOSurface& surface, IntSize requestedSize, const DestinationColorSpace& colorSpace, IOSurface::Format format)
64{
65 if (format != surface.format())
66 return false;
67 if (colorSpace != surface.colorSpace())
68 return false;
69 if (requestedSize != surface.size())
70 return false;
71 return true;
72}

74void IOSurfacePool::willAddSurface(IOSurface& surface, bool inUse)
75{
76 CachedSurfaceDetails& details = m_surfaceDetails.add(&surface, CachedSurfaceDetails()).iterator->value;
77 details.resetLastUseTime();
78
79 surface.releaseGraphicsContext();
80
81 size_t surfaceBytes = surface.totalBytes();
82
83 evict(surfaceBytes);
84
85 m_bytesCached += surfaceBytes;
86 if (inUse)
87 m_inUseBytesCached += surfaceBytes;
88}
89
90void IOSurfacePool::didRemoveSurface(IOSurface& surface, bool inUse)
91{
92 size_t surfaceBytes = surface.totalBytes();
93 m_bytesCached -= surfaceBytes;
94 if (inUse)
95 m_inUseBytesCached -= surfaceBytes;
96
97 m_surfaceDetails.remove(&surface);
98}

100void IOSurfacePool::didUseSurfaceOfSize(IntSize size)
101{
102 m_sizesInPruneOrder.remove(m_sizesInPruneOrder.reverseFind(size));
103 m_sizesInPruneOrder.append(size);
104}

106std::unique_ptr<IOSurface> IOSurfacePool::takeSurface(IntSize size, const DestinationColorSpace& colorSpace, IOSurface::Format format)
107{
108 Locker locker { m_lock };
109 CachedSurfaceMap::iterator mapIter = m_cachedSurfaces.find(size);
110
111 if (mapIter == m_cachedSurfaces.end()) {
112 DUMP_POOL_STATISTICS("takeSurface - none with matching size");
113 return nullptr;
114 }
115
116 for (auto surfaceIter = mapIter->value.begin(); surfaceIter != mapIter->value.end(); ++surfaceIter) {
117 if (!surfaceMatchesParameters(*surfaceIter->get(), size, colorSpace, format))
118 continue;
119
120 auto surface = WTFMove(*surfaceIter);
121 mapIter->value.remove(surfaceIter);
122
123 didUseSurfaceOfSize(size);
124
125 if (mapIter->value.isEmpty()) {
126 m_cachedSurfaces.remove(mapIter);
127 m_sizesInPruneOrder.removeLast();
128 }
129
130 didRemoveSurface(*surface, false);
131
132 surface->setVolatile(false);
133
134 DUMP_POOL_STATISTICS("takeSurface - taking");
135 return surface;
136 }
137
138 // Some of the in-use surfaces may no longer actually be in-use, but we haven't moved them over yet.
139 for (auto surfaceIter = m_inUseSurfaces.begin(); surfaceIter != m_inUseSurfaces.end(); ++surfaceIter) {
140 if (!surfaceMatchesParameters(*surfaceIter->get(), size, colorSpace, format))
141 continue;
142 if (surfaceIter->get()->isInUse())
143 continue;
144
145 auto surface = WTFMove(*surfaceIter);
146 m_inUseSurfaces.remove(surfaceIter);
147 didRemoveSurface(*surface, true);
148
149 surface->setVolatile(false);
150
151 DUMP_POOL_STATISTICS("takeSurface - taking in-use");
152 return surface;
153 }
154
155 DUMP_POOL_STATISTICS("takeSurface - failing");
156 return nullptr;
157}

159bool IOSurfacePool::shouldCacheSurface(const IOSurface& surface) const
160{
161 if (surface.totalBytes() > m_maximumBytesCached)
162 return false;
163
164 // There's no reason to pool empty surfaces; we should never allocate them in the first place.
165 // This also covers isZero(), which would cause trouble when used as the key in m_cachedSurfaces.
166 if (surface.size().isEmpty())
167 return false;
168
169 return true;
170}

172void IOSurfacePool::addSurface(std::unique_ptr<IOSurface> surface)
173{
174 Locker locker { m_lock };
175 if (!shouldCacheSurface(*surface))
176 return;
177
178 bool surfaceIsInUse = surface->isInUse();
179
180 willAddSurface(*surface, surfaceIsInUse);
181
182 if (surfaceIsInUse) {
183 m_inUseSurfaces.prepend(WTFMove(surface));
184 scheduleCollectionTimer();
185 DUMP_POOL_STATISTICS("addSurface - in-use");
186 return;
187 }
188
189 insertSurfaceIntoPool(WTFMove(surface));
190 DUMP_POOL_STATISTICS("addSurface");
191}
192
193void IOSurfacePool::insertSurfaceIntoPool(std::unique_ptr<IOSurface> surface)
194{
195 IntSize surfaceSize = surface->size();
196 auto insertedTuple = m_cachedSurfaces.add(surfaceSize, CachedSurfaceQueue());
197 insertedTuple.iterator->value.prepend(WTFMove(surface));
198 if (!insertedTuple.isNewEntry)
199 m_sizesInPruneOrder.remove(m_sizesInPruneOrder.reverseFind(surfaceSize));
200 m_sizesInPruneOrder.append(surfaceSize);
201
202 scheduleCollectionTimer();
203}

209 evict(0);

212void IOSurfacePool::tryEvictInUseSurface()
213{
214 if (m_inUseSurfaces.isEmpty())
215 return;
216
217 auto surface = m_inUseSurfaces.takeLast();
218 didRemoveSurface(*surface, true);
219}
220
221void IOSurfacePool::tryEvictOldestCachedSurface()
222{
223 if (m_cachedSurfaces.isEmpty())
224 return;
225
226 if (m_sizesInPruneOrder.isEmpty())
227 return;
228
229 CachedSurfaceMap::iterator surfaceQueueIter = m_cachedSurfaces.find(m_sizesInPruneOrder.first());
230 ASSERT(!surfaceQueueIter->value.isEmpty());
231 auto surface = surfaceQueueIter->value.takeLast();
232 didRemoveSurface(*surface, false);
233
234 if (surfaceQueueIter->value.isEmpty()) {
235 m_cachedSurfaces.remove(surfaceQueueIter);
236 m_sizesInPruneOrder.remove(0);
237 }
238}

240void IOSurfacePool::evict(size_t additionalSize)
241{
242 DUMP_POOL_STATISTICS("before evict");
243
244 if (additionalSize >= m_maximumBytesCached) {
245 discardAllSurfacesInternal();
246 DUMP_POOL_STATISTICS("after evict all");
247 return;
248 }
249
250 // FIXME: Perhaps purgeable surfaces should count less against the cap?
251 // We don't want to end up with a ton of empty (purged) surfaces, though, as that would defeat the purpose of the pool.
252 size_t targetSize = m_maximumBytesCached - additionalSize;
253
254 // Interleave eviction of old cached surfaces and more recent in-use surfaces.
255 // In-use surfaces are more recently used, but less useful in the pool, as they aren't
256 // immediately available when requested.
257 while (m_bytesCached > targetSize) {
258 tryEvictOldestCachedSurface();
259
260 if (m_inUseBytesCached > maximumInUseBytes || m_bytesCached > targetSize)
261 tryEvictInUseSurface();
262 }
263
264 while (m_inUseBytesCached > maximumInUseBytes || m_bytesCached > targetSize)
265 tryEvictInUseSurface();
266
267 DUMP_POOL_STATISTICS("after evict");
268}

270void IOSurfacePool::collectInUseSurfaces()
271{
272 CachedSurfaceQueue newInUseSurfaces;
273 for (CachedSurfaceQueue::iterator surfaceIter = m_inUseSurfaces.begin(); surfaceIter != m_inUseSurfaces.end(); ++surfaceIter) {
274 IOSurface* surface = surfaceIter->get();
275 if (surface->isInUse()) {
276 newInUseSurfaces.append(WTFMove(*surfaceIter));
277 continue;
278 }
279
280 m_inUseBytesCached -= surface->totalBytes();
281 insertSurfaceIntoPool(WTFMove(*surfaceIter));
282 }
283
284 m_inUseSurfaces = WTFMove(newInUseSurfaces);
285}

287bool IOSurfacePool::markOlderSurfacesPurgeable()
288{
289 bool markedAllSurfaces = true;
290 auto markTime = MonotonicTime::now();
291
292 for (auto& surfaceAndDetails : m_surfaceDetails) {
293 if (surfaceAndDetails.value.hasMarkedPurgeable)
294 continue;
295
296 if (markTime - surfaceAndDetails.value.lastUseTime < surfaceAgeBeforeMarkingPurgeable) {
297 markedAllSurfaces = false;
298 continue;
299 }
300
301 surfaceAndDetails.key->setVolatile(true);
302 surfaceAndDetails.value.hasMarkedPurgeable = true;
303 }
304
305 return markedAllSurfaces;
306}

308void IOSurfacePool::collectionTimerFired()
309{
310 Locker locker { m_lock };
311 collectInUseSurfaces();
312 bool markedAllSurfaces = markOlderSurfacesPurgeable();
313
314 if (!m_inUseSurfaces.size() && markedAllSurfaces)
315 m_collectionTimer.stop();
316
317 platformGarbageCollectNow();
318 DUMP_POOL_STATISTICS("collectionTimerFired");
319}

321void IOSurfacePool::scheduleCollectionTimer()
322{
323 if (!m_collectionTimer.isActive())
324 m_collectionTimer.startRepeating(collectionInterval);
325}

327void IOSurfacePool::discardAllSurfaces()
328{
329 Locker locker { m_lock };
330 discardAllSurfacesInternal();
331}

333void IOSurfacePool::discardAllSurfacesInternal()
334{
335 m_bytesCached = 0;
336 m_inUseBytesCached = 0;
337 m_surfaceDetails.clear();
338 m_cachedSurfaces.clear();
339 m_inUseSurfaces.clear();
340 m_sizesInPruneOrder.clear();
341 m_collectionTimer.stop();
342 platformGarbageCollectNow();
343}

345void IOSurfacePool::showPoolStatistics(const char* reason)
346{
347#if ENABLE_IOSURFACE_POOL_STATISTICS
348 WTFLogAlways("IOSurfacePool Statistics: %s\n", reason);
349 unsigned totalSurfaces = 0;
350 size_t totalSize = 0;
351 size_t totalPurgeableSize = 0;
352
353 for (const auto& keyAndSurfaces : m_cachedSurfaces) {
354 ASSERT(!keyAndSurfaces.value.isEmpty());
355 size_t queueSize = 0;
356 size_t queuePurgeableSize = 0;
357 for (const auto& surface : keyAndSurfaces.value) {
358 size_t surfaceBytes = surface->totalBytes();
359
360 totalSurfaces++;
361 queueSize += surfaceBytes;
362
363 if (surface->isVolatile())
364 queuePurgeableSize += surfaceBytes;
365 }
366
367 totalSize += queueSize;
368 totalPurgeableSize += queuePurgeableSize;
369
370 WTFLogAlways(" %d x %d: %zu surfaces for %.2f MB (%.2f MB purgeable)", keyAndSurfaces.key.width(), keyAndSurfaces.key.height(), keyAndSurfaces.value.size(), queueSize / (1024.0 * 1024.0), queuePurgeableSize / (1024.0 * 1024.0));
371 }
372
373 size_t inUseSize = 0;
374 for (const auto& surface : m_inUseSurfaces) {
375 totalSurfaces++;
376 inUseSize += surface->totalBytes();
377 }
378
379 totalSize += inUseSize;
380 WTFLogAlways(" IN USE: %zu surfaces for %.2f MB", m_inUseSurfaces.size(), inUseSize / (1024.0 * 1024.0));
381
382 // FIXME: Should move consistency checks elsewhere, and always perform them in debug builds.
383 ASSERT(m_bytesCached == totalSize);
384 ASSERT(m_bytesCached <= m_maximumBytesCached);
385
386 WTFLogAlways(" TOTAL: %d surfaces for %.2f MB (%.2f MB purgeable)\n", totalSurfaces, totalSize / (1024.0 * 1024.0), totalPurgeableSize / (1024.0 * 1024.0));
387#else
388 UNUSED_PARAM(reason);
389#endif
390}
