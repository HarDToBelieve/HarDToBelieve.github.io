60unsigned ImageBufferCGBackend::calculateBytesPerRow(const IntSize& backendSize)

81void ImageBufferCGBackend::setupContext() const
82{
83 // The initial CTM matches DisplayList::Recorder::clipToDrawingCommands()'s initial CTM.
84 context().scale(FloatSize(1, -1));
85 context().translate(0, -backendSize().height());
86 context().applyDeviceScaleFactor(resolutionScale());
87}

89static RetainPtr<CGImageRef> createCroppedImageIfNecessary(CGImageRef image, const IntSize& backendSize)
90{
91 if (image && (CGImageGetWidth(image) != static_cast<size_t>(backendSize.width()) || CGImageGetHeight(image) != static_cast<size_t>(backendSize.height())))
92 return adoptCF(CGImageCreateWithImageInRect(image, CGRectMake(0, 0, backendSize.width(), backendSize.height())));
93 return image;
94}

96static RefPtr<Image> createBitmapImageAfterScalingIfNeeded(RefPtr<NativeImage>&& image, const IntSize& logicalSize, const IntSize& backendSize, float resolutionScale, PreserveResolution preserveResolution)
97{
98 if (!image)
99 return nullptr;
100
101 if (resolutionScale == 1 || preserveResolution == PreserveResolution::Yes)
102 image = NativeImage::create(createCroppedImageIfNecessary(image->platformImage().get(), backendSize));
103 else {
104 auto context = adoptCF(CGBitmapContextCreate(0, logicalSize.width(), logicalSize.height(), 8, 4 * logicalSize.width(), sRGBColorSpaceRef(), kCGImageAlphaPremultipliedFirst | kCGBitmapByteOrder32Host));
105 CGContextSetBlendMode(context.get(), kCGBlendModeCopy);
106 CGContextClipToRect(context.get(), FloatRect(FloatPoint::zero(), logicalSize));
107 FloatSize imageSizeInUserSpace = logicalSize;
108 CGContextDrawImage(context.get(), FloatRect(FloatPoint::zero(), imageSizeInUserSpace), image->platformImage().get());
109 image = NativeImage::create(adoptCF(CGBitmapContextCreateImage(context.get())));
110 }
111
112 if (!image)
113 return nullptr;
114
115 return BitmapImage::create(WTFMove(image));
116}

118RefPtr<Image> ImageBufferCGBackend::copyImage(BackingStoreCopy copyBehavior, PreserveResolution preserveResolution) const
119{
120 RefPtr<NativeImage> image;
121 if (resolutionScale() == 1 || preserveResolution == PreserveResolution::Yes)
122 image = copyNativeImage(copyBehavior);
123 else
124 image = copyNativeImage(DontCopyBackingStore);
125 return createBitmapImageAfterScalingIfNeeded(WTFMove(image), logicalSize(), backendSize(), resolutionScale(), preserveResolution);
126}
127
128RefPtr<Image> ImageBufferCGBackend::sinkIntoImage(PreserveResolution preserveResolution)
129{
130 // Get the backend size before sinking the it into a NativeImage. sinkIntoNativeImage() sets the IOSurface to null if it's an accelerated backend.
131 auto backendSize = this->backendSize();
132 return createBitmapImageAfterScalingIfNeeded(sinkIntoNativeImage(), logicalSize(), backendSize, resolutionScale(), preserveResolution);
133}

135void ImageBufferCGBackend::draw(GraphicsContext& destContext, const FloatRect& destRect, const FloatRect& srcRect, const ImagePaintingOptions& options)
136{
137 FloatRect srcRectScaled = srcRect;
138 srcRectScaled.scale(resolutionScale());
139
140 if (auto image = copyNativeImage(&destContext == &context() ? CopyBackingStore : DontCopyBackingStore))
141 destContext.drawNativeImage(*image, backendSize(), destRect, srcRectScaled, options);
142}

153void ImageBufferCGBackend::clipToMask(GraphicsContext& destContext, const FloatRect& destRect)
154{
155 auto nativeImage = copyNativeImage(DontCopyBackingStore);
156 if (!nativeImage)
157 return;
158
159 CGContextRef cgContext = destContext.platformContext();
160
161 // FIXME: This image needs to be grayscale to be used as an alpha mask here.
162 CGContextTranslateCTM(cgContext, destRect.x(), destRect.maxY());
163 CGContextScaleCTM(cgContext, 1, -1);
164 CGContextClipToRect(cgContext, { { }, destRect.size() });
165 CGContextClipToMask(cgContext, { { }, destRect.size() }, nativeImage->platformImage().get());
166 CGContextScaleCTM(cgContext, 1, -1);
167 CGContextTranslateCTM(cgContext, -destRect.x(), -destRect.maxY());
168}

170RetainPtr<CGImageRef> ImageBufferCGBackend::copyCGImageForEncoding(CFStringRef destinationUTI, PreserveResolution preserveResolution) const
171{
172 if (CFEqual(destinationUTI, jpegUTI())) {
173 // FIXME: Should this be using the same logic as ImageBufferUtilitiesCG?
174
175 // JPEGs don't have an alpha channel, so we have to manually composite on top of black.
176 PixelBufferFormat format { AlphaPremultiplication::Premultiplied, PixelFormat::RGBA8, DestinationColorSpace::SRGB() };
177 auto pixelBuffer = getPixelBuffer(format, logicalRect());
178 if (!pixelBuffer)
179 return nullptr;
180
181 auto pixelArray = makeRef(pixelBuffer->data());
182 auto dataSize = pixelArray->byteLength();
183 auto data = pixelArray->data();
184
185 verifyImageBufferIsBigEnough(data, dataSize);
186
187 auto dataProvider = adoptCF(CGDataProviderCreateWithData(&pixelArray.leakRef(), data, dataSize, [] (void* context, const void*, size_t) {
188 static_cast<JSC::Uint8ClampedArray*>(context)->deref();
189 }));
190 if (!dataProvider)
191 return nullptr;
192
193 auto imageSize = pixelBuffer->size();
194 return adoptCF(CGImageCreate(imageSize.width(), imageSize.height(), 8, 32, 4 * imageSize.width(), pixelBuffer->format().colorSpace.platformColorSpace(), kCGBitmapByteOrderDefault | kCGImageAlphaNoneSkipLast, dataProvider.get(), 0, false, kCGRenderingIntentDefault));
195 }
196
197 if (resolutionScale() == 1 || preserveResolution == PreserveResolution::Yes) {
198 auto nativeImage = copyNativeImage(CopyBackingStore);
199 if (!nativeImage)
200 return nullptr;
201 return createCroppedImageIfNecessary(nativeImage->platformImage().get(), backendSize());
202 }
203
204 auto nativeImage = copyNativeImage(DontCopyBackingStore);
205 if (!nativeImage)
206 return nullptr;
207 auto image = nativeImage->platformImage();
208 auto context = adoptCF(CGBitmapContextCreate(0, backendSize().width(), backendSize().height(), 8, 4 * backendSize().width(), sRGBColorSpaceRef(), kCGImageAlphaPremultipliedFirst | kCGBitmapByteOrder32Host));
209 CGContextSetBlendMode(context.get(), kCGBlendModeCopy);
210 CGContextClipToRect(context.get(), CGRectMake(0, 0, backendSize().width(), backendSize().height()));
211 CGContextDrawImage(context.get(), CGRectMake(0, 0, backendSize().width(), backendSize().height()), image.get());
212 return adoptCF(CGBitmapContextCreateImage(context.get()));
213}

215Vector<uint8_t> ImageBufferCGBackend::toData(const String& mimeType, std::optional<double> quality) const
216{
217#if ENABLE(GPU_PROCESS)
218 ASSERT_IMPLIES(!isInGPUProcess(), MIMETypeRegistry::isSupportedImageMIMETypeForEncoding(mimeType));
219#else
220 ASSERT(MIMETypeRegistry::isSupportedImageMIMETypeForEncoding(mimeType));
221#endif
222
223 auto destinationUTI = utiFromImageBufferMIMEType(mimeType);
224 auto image = copyCGImageForEncoding(destinationUTI.get(), PreserveResolution::No);
225
226 return WebCore::data(image.get(), destinationUTI.get(), quality);
227}

239
