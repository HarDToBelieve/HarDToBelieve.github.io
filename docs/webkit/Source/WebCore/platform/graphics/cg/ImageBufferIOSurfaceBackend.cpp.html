44IntSize ImageBufferIOSurfaceBackend::calculateSafeBackendSize(const Parameters& parameters)
45{
46 IntSize backendSize = calculateBackendSize(parameters);
47 if (backendSize.isEmpty())
48 return { };
49
50 IntSize maxSize = IOSurface::maximumSize();
51 if (backendSize.width() > maxSize.width() || backendSize.height() > maxSize.height())
52 return { };
53
54 return backendSize;
55}

57unsigned ImageBufferIOSurfaceBackend::calculateBytesPerRow(const IntSize& backendSize)
58{

60 return IOSurfaceAlignProperty(kIOSurfaceBytesPerRow, bytesPerRow);
61}

74RetainPtr<CGColorSpaceRef> ImageBufferIOSurfaceBackend::contextColorSpace(const GraphicsContext& context)
75{
76 CGContextRef cgContext = context.platformContext();
77
78 if (CGContextGetType(cgContext) == kCGContextTypeIOSurface)
79 return CGIOSurfaceContextGetColorSpace(cgContext);
80
81 return ImageBufferCGBackend::contextColorSpace(context);
82}

84std::unique_ptr<ImageBufferIOSurfaceBackend> ImageBufferIOSurfaceBackend::create(const Parameters& parameters, const HostWindow* hostWindow)
85{
86 IntSize backendSize = calculateSafeBackendSize(parameters);
87 if (backendSize.isEmpty())
88 return nullptr;
89
90 auto surface = IOSurface::create(backendSize, backendSize, parameters.colorSpace, IOSurface::formatForPixelFormat(parameters.pixelFormat));
91 if (!surface)
92 return nullptr;
93
94 RetainPtr<CGContextRef> cgContext = surface->ensurePlatformContext(hostWindow);
95 if (!cgContext)
96 return nullptr;
97
98 CGContextClearRect(cgContext.get(), FloatRect(FloatPoint::zero(), backendSize));
99
100 return makeUnique<ImageBufferIOSurfaceBackend>(parameters, WTFMove(surface));
101}
102
103std::unique_ptr<ImageBufferIOSurfaceBackend> ImageBufferIOSurfaceBackend::create(const Parameters& parameters, const GraphicsContext& context)
104{
105 if (auto cgColorSpace = context.hasPlatformContext() ? contextColorSpace(context) : nullptr) {
106 auto overrideParameters = parameters;
107 overrideParameters.colorSpace = DestinationColorSpace { cgColorSpace };
108
109 return ImageBufferIOSurfaceBackend::create(overrideParameters, nullptr);
110 }
111
112 return ImageBufferIOSurfaceBackend::create(parameters, nullptr);
113}

158void ImageBufferIOSurfaceBackend::drawConsuming(GraphicsContext& destContext, const FloatRect& destRect, const FloatRect& srcRect, const ImagePaintingOptions& options)
159{
160 FloatRect adjustedSrcRect = srcRect;
161 adjustedSrcRect.scale(resolutionScale());
162
163 auto backendSize = this->backendSize();
164 if (auto image = sinkIntoNativeImage())
165 destContext.drawNativeImage(*image, backendSize, destRect, adjustedSrcRect, options);
166}

168RetainPtr<CGImageRef> ImageBufferIOSurfaceBackend::copyCGImageForEncoding(CFStringRef destinationUTI, PreserveResolution preserveResolution) const
169{
170 if (m_requiresDrawAfterPutPixelBuffer) {
171 // Force recreating the IOSurface cached image.
172 // See https://bugs.webkit.org/show_bug.cgi?id=157966 for explaining why this is necessary.
173 context().fillRect(FloatRect(1, 1, 0, 0));
174 m_requiresDrawAfterPutPixelBuffer = false;
175 }
176 return ImageBufferCGBackend::copyCGImageForEncoding(destinationUTI, preserveResolution);
177}
