47uint8_t verifyImageBufferIsBigEnough(const void* buffer, size_t bufferSize)
48{
49 RELEASE_ASSERT(bufferSize);
50
51 uintptr_t lastByte;
52 bool isSafe = WTF::safeAdd((uintptr_t)buffer, bufferSize - 1, lastByte);
53 RELEASE_ASSERT(isSafe);
54
55 return *(uint8_t*)lastByte;
56}

58CFStringRef jpegUTI()
59{
60ALLOW_DEPRECATED_DECLARATIONS_BEGIN
61#if PLATFORM(IOS_FAMILY) || PLATFORM(WIN)
62 static const CFStringRef kUTTypeJPEG = CFSTR("public.jpeg");
63#endif
64 return kUTTypeJPEG;
65ALLOW_DEPRECATED_DECLARATIONS_END
66}

68RetainPtr<CFStringRef> utiFromImageBufferMIMEType(const String& mimeType)
69{
70ALLOW_DEPRECATED_DECLARATIONS_BEGIN
71 // FIXME: Why doesn't iOS use the CoreServices version?
72#if PLATFORM(MAC)
73 return UTIFromMIMEType(mimeType).createCFString();
74#else
75 // FIXME: Add Windows support for all the supported UTIs when a way to convert from MIMEType to UTI reliably is found.
76 // For now, only support PNG, JPEG, and GIF. See <rdar://problem/6095286>.
77 static CFStringRef kUTTypePNG;
78 static CFStringRef kUTTypeGIF;
79
80 static std::once_flag onceKey;
81 std::call_once(onceKey, [&] {
82 kUTTypePNG = CFSTR("public.png");
83 kUTTypeGIF = CFSTR("com.compuserve.gif");
84 });
85
86 if (equalLettersIgnoringASCIICase(mimeType, "image/png"))
87 return kUTTypePNG;
88 if (equalLettersIgnoringASCIICase(mimeType, "image/jpeg"))
89 return jpegUTI();
90 if (equalLettersIgnoringASCIICase(mimeType, "image/gif"))
91 return kUTTypeGIF;
92
93 ASSERT_NOT_REACHED();
94 return kUTTypePNG;
95#endif
96ALLOW_DEPRECATED_DECLARATIONS_END
97}

99static bool encode(CGImageRef image, CFStringRef destinationUTI, std::optional<double> quality, const ScopedLambda<PutBytesCallback>& function)
100{
101 if (!image || !destinationUTI)
102 return false;
103
104 CGDataConsumerCallbacks callbacks {
105 [](void* context, const void* buffer, size_t count) -> size_t {
106 auto functor = *static_cast<const ScopedLambda<PutBytesCallback>*>(context);
107 return functor(buffer, count);
108 },
109 nullptr
110 };
111
112 auto consumer = adoptCF(CGDataConsumerCreate(const_cast<ScopedLambda<PutBytesCallback>*>(&function), &callbacks));
113 auto destination = adoptCF(CGImageDestinationCreateWithDataConsumer(consumer.get(), destinationUTI, 1, nullptr));
114
115 auto imageProperties = [&] () -> RetainPtr<CFDictionaryRef> {
116 if (CFEqual(destinationUTI, jpegUTI()) && quality && *quality >= 0.0 && *quality <= 1.0) {
117 // Apply the compression quality to the JPEG image destination.
118 auto compressionQuality = adoptCF(CFNumberCreate(kCFAllocatorDefault, kCFNumberDoubleType, &*quality));
119 const void* key = kCGImageDestinationLossyCompressionQuality;
120 const void* value = compressionQuality.get();
121 return adoptCF(CFDictionaryCreate(0, &key, &value, 1, &kCFTypeDictionaryKeyCallBacks, &kCFTypeDictionaryValueCallBacks));
122 }
123 return nullptr;
124 }();
125
126 // FIXME: Setting kCGImageDestinationBackgroundColor to black for JPEG images in imageProperties would save some math
127 // in the calling functions, but it doesn't seem to work.
128
129 CGImageDestinationAddImage(destination.get(), image, imageProperties.get());
130
131 return CGImageDestinationFinalize(destination.get());
132}

134static bool encode(const PixelBuffer& source, const String& mimeType, std::optional<double> quality, const ScopedLambda<PutBytesCallback>& function)
135{
136 ASSERT(MIMETypeRegistry::isSupportedImageMIMETypeForEncoding(mimeType));
137
138 auto destinationUTI = utiFromImageBufferMIMEType(mimeType);
139
140 CGImageAlphaInfo dataAlphaInfo = kCGImageAlphaLast;
141
142 auto data = source.data().data();
143 auto dataSize = source.data().byteLength();
144
145 Vector<uint8_t> premultipliedData;
146
147 if (CFEqual(destinationUTI.get(), jpegUTI())) {
148 // FIXME: Use PixelBufferConversion for this once it supports RGBX.
149
150 // JPEGs don't have an alpha channel, so we have to manually composite on top of black.
151 if (!premultipliedData.tryReserveCapacity(dataSize))
152 return false;
153
154 premultipliedData.grow(dataSize);
155 unsigned char* buffer = premultipliedData.data();
156 for (size_t i = 0; i < dataSize; i += 4) {
157 unsigned alpha = data[i + 3];
158 if (alpha != 255) {
159 buffer[i + 0] = data[i + 0] * alpha / 255;
160 buffer[i + 1] = data[i + 1] * alpha / 255;
161 buffer[i + 2] = data[i + 2] * alpha / 255;
162 } else {
163 buffer[i + 0] = data[i + 0];
164 buffer[i + 1] = data[i + 1];
165 buffer[i + 2] = data[i + 2];
166 }
167 }
168
169 dataAlphaInfo = kCGImageAlphaNoneSkipLast; // Ignore the alpha channel.
170 data = premultipliedData.data();
171 }
172
173 verifyImageBufferIsBigEnough(data, dataSize);
174
175 auto dataProvider = adoptCF(CGDataProviderCreateWithData(nullptr, data, dataSize, nullptr));
176 if (!dataProvider)
177 return nullptr;
178
179 auto imageSize = source.size();
180 auto image = adoptCF(CGImageCreate(imageSize.width(), imageSize.height(), 8, 32, 4 * imageSize.width(), source.format().colorSpace.platformColorSpace(), kCGBitmapByteOrderDefault | dataAlphaInfo, dataProvider.get(), 0, false, kCGRenderingIntentDefault));
181
182 return encode(image.get(), destinationUTI.get(), quality, function);
183}

185template<typename Source, typename SourceDescription> static Vector<uint8_t> encodeToVector(Source&& source, SourceDescription&& sourceDescription, std::optional<double> quality)
186{
187 Vector<uint8_t> result;
188
189 bool success = encode(std::forward<Source>(source), std::forward<SourceDescription>(sourceDescription), quality, scopedLambdaRef<PutBytesCallback>([&] (const void* data, size_t length) {
190 result.append(static_cast<const uint8_t*>(data), length);
191 return length;
192 }));
193 if (!success)
194 return { };
195
196 return result;
197}

199template<typename Source, typename SourceDescription> static String encodeToDataURL(Source&& source, SourceDescription&& sourceDescription, const String& mimeType, std::optional<double> quality)
200{
201 // FIXME: This could be done more efficiently with a streaming base64 encoder.
202
203 auto encodedData = encodeToVector(std::forward<Source>(source), std::forward<SourceDescription>(sourceDescription), quality);
204 if (encodedData.isEmpty())
205 return "data:,"_s;
206
207 return makeString("data:", mimeType, ";base64,", base64Encoded(encodedData));
208}
