30namespace WebCore {
31
32class ImageDecoderCG final : public ImageDecoder {
33 WTF_MAKE_FAST_ALLOCATED;
34public:
35 ImageDecoderCG(SharedBuffer& data, AlphaOption, GammaAndColorProfileOption);
36
37 static Ref<ImageDecoderCG> create(SharedBuffer& data, AlphaOption alphaOption, GammaAndColorProfileOption gammaAndColorProfileOption)
38 {
39 return adoptRef(*new ImageDecoderCG(data, alphaOption, gammaAndColorProfileOption));
40 }
41
42 static bool supportsMediaType(MediaType type) { return type == MediaType::Image; }
43 static bool canDecodeType(const String&);
44
45 size_t bytesDecodedToDetermineProperties() const final;
46
47 EncodedDataStatus encodedDataStatus() const final;
48 IntSize size() const final { return IntSize(); }
49 size_t frameCount() const final;
50 RepetitionCount repetitionCount() const final;
51 String uti() const final;
52 String filenameExtension() const final;
53 String accessibilityDescription() const final;
54 std::optional<IntPoint> hotSpot() const final;
55
56 IntSize frameSizeAtIndex(size_t, SubsamplingLevel = SubsamplingLevel::Default) const final;
57 bool frameIsCompleteAtIndex(size_t) const final;
58 ImageDecoder::FrameMetadata frameMetadataAtIndex(size_t) const final;
59
60 Seconds frameDurationAtIndex(size_t) const final;
61 bool frameHasAlphaAtIndex(size_t) const final;
62 bool frameAllowSubsamplingAtIndex(size_t) const final;
63 unsigned frameBytesAtIndex(size_t, SubsamplingLevel = SubsamplingLevel::Default) const final;
64
65 PlatformImagePtr createFrameImageAtIndex(size_t, SubsamplingLevel = SubsamplingLevel::Default, const DecodingOptions& = DecodingOptions(DecodingMode::Synchronous)) final;
66
67 void setData(SharedBuffer&, bool allDataReceived) final;
68 bool isAllDataReceived() const final { return m_isAllDataReceived; }
69 void clearFrameBufferCache(size_t) final { }
70
71private:
72 bool m_isAllDataReceived { false };
73 mutable EncodedDataStatus m_encodedDataStatus { EncodedDataStatus::Unknown };
74 RetainPtr<CGImageSourceRef> m_nativeDecoder;
75};
