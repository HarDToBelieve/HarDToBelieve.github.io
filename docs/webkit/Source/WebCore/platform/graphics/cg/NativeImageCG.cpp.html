36IntSize NativeImage::size() const
37{
38 return IntSize(CGImageGetWidth(m_platformImage.get()), CGImageGetHeight(m_platformImage.get()));
39}
40
41bool NativeImage::hasAlpha() const
42{
43 CGImageAlphaInfo info = CGImageGetAlphaInfo(m_platformImage.get());
44 return (info >= kCGImageAlphaPremultipliedLast) && (info <= kCGImageAlphaFirst);
45}
46
47Color NativeImage::singlePixelSolidColor() const
48{
49 if (size() != IntSize(1, 1))
50 return Color();
51
52 unsigned char pixel[4]; // RGBA
53 auto bitmapContext = adoptCF(CGBitmapContextCreate(pixel, 1, 1, 8, sizeof(pixel), sRGBColorSpaceRef(), kCGImageAlphaPremultipliedLast | kCGBitmapByteOrder32Big));
54
55 if (!bitmapContext)
56 return Color();
57
58 CGContextSetBlendMode(bitmapContext.get(), kCGBlendModeCopy);
59 CGContextDrawImage(bitmapContext.get(), CGRectMake(0, 0, 1, 1), m_platformImage.get());
60
61 if (!pixel[3])
62 return Color::transparentBlack;
63
64 return makeFromComponentsClampingExceptAlpha<SRGBA<uint8_t>>(pixel[0] * 255 / pixel[3], pixel[1] * 255 / pixel[3], pixel[2] * 255 / pixel[3], pixel[3]);
65}
