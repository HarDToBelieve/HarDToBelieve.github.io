109bool PDFDocumentImage::cacheParametersMatch(GraphicsContext& context, const FloatRect& dstRect, const FloatRect& srcRect) const
110{
111 // Old and new source rectangles have to match.
112 if (srcRect != m_cachedSourceRect)
113 return false;
114
115 // Old and new scaling factors "dest / src" have to match.
116 if (dstRect.size() != m_cachedDestinationRect.size())
117 return false;
118
119 // m_cachedImageRect can be moved if srcRect and dstRect.size() did not change.
120 FloatRect movedCachedImageRect = m_cachedImageRect;
121 movedCachedImageRect.move(FloatSize(dstRect.location() - m_cachedDestinationRect.location()));
122
123 // movedCachedImageRect has to contain the whole dirty rectangle.
124 FloatRect dirtyRect = intersection(context.clipBounds(), dstRect);
125 if (!movedCachedImageRect.contains(dirtyRect))
126 return false;
127
128 // Old and new context scaling factors have to match as well.
129 AffineTransform::DecomposedType decomposedTransform;
130 context.getCTM(GraphicsContext::DefinitelyIncludeDeviceScale).decompose(decomposedTransform);
131
132 AffineTransform::DecomposedType cachedDecomposedTransform;
133 m_cachedTransform.decompose(cachedDecomposedTransform);
134 if (decomposedTransform.scaleX != cachedDecomposedTransform.scaleX || decomposedTransform.scaleY != cachedDecomposedTransform.scaleY)
135 return false;
136
137 return true;
138}

140static void transformContextForPainting(GraphicsContext& context, const FloatRect& dstRect, const FloatRect& srcRect)
141{
142 float hScale = dstRect.width() / srcRect.width();
143 float vScale = dstRect.height() / srcRect.height();
144
145 if (hScale != vScale) {
146 float minimumScale = std::max((dstRect.width() - 0.5) / srcRect.width(), (dstRect.height() - 0.5) / srcRect.height());
147 float maximumScale = std::min((dstRect.width() + 0.5) / srcRect.width(), (dstRect.height() + 0.5) / srcRect.height());
148
149 // If the difference between the two scales is due to integer rounding of image sizes,
150 // use the smaller of the two original scales to ensure that the image fits inside the
151 // space originally allocated for it.
152 if (minimumScale <= maximumScale) {
153 hScale = std::min(hScale, vScale);
154 vScale = hScale;
155 }
156 }
157
158 // drawPDFPage() relies on drawing the whole PDF into a context starting at (0, 0). We need
159 // to transform the destination context such that srcRect of the source context will be drawn
160 // in dstRect of destination context.
161 context.translate(dstRect.location() - srcRect.location());
162 context.scale(FloatSize(hScale, -vScale));
163 context.translate(0, -srcRect.height());
164}

166// To avoid the jetsam on iOS, we are going to limit the size of all the PDF cachedImages to be 64MB.
167static const size_t s_maxCachedImageSide = 4 * 1024;
168static const size_t s_maxCachedImageArea = s_maxCachedImageSide * s_maxCachedImageSide;
169
170static const size_t s_maxDecodedDataSize = s_maxCachedImageArea * 4;
171static size_t s_allDecodedDataSize = 0;
172
173static FloatRect cachedImageRect(GraphicsContext& context, const FloatRect& dstRect)
174{
175 FloatRect dirtyRect = context.clipBounds();
176
177 // Calculate the maximum rectangle we can cache around the center of the clipping bounds.
178 FloatSize maxSize = s_maxCachedImageSide / context.scaleFactor();
179 FloatPoint minLocation = FloatPoint(dirtyRect.center() - maxSize / 2);
180
181 // Ensure the clipping bounds are all included but within the bounds of the dstRect
182 return intersection(unionRect(dirtyRect, FloatRect(minLocation, maxSize)), dstRect);
183}

185void PDFDocumentImage::decodedSizeChanged(size_t newCachedBytes)
186{
187 if (!m_cachedBytes && !newCachedBytes)
188 return;
189
190 if (imageObserver())
191 imageObserver()->decodedSizeChanged(*this, -static_cast<long long>(m_cachedBytes) + newCachedBytes);
192
193 ASSERT(s_allDecodedDataSize >= m_cachedBytes);
194 // Update with the difference in two steps to avoid unsigned underflow subtraction.
195 s_allDecodedDataSize -= m_cachedBytes;
196 s_allDecodedDataSize += newCachedBytes;
197
198 m_cachedBytes = newCachedBytes;
199}

201void PDFDocumentImage::updateCachedImageIfNeeded(GraphicsContext& context, const FloatRect& dstRect, const FloatRect& srcRect)
202{
203 // Clipped option is for testing only. Force re-caching the PDF with each draw.
204 bool forceUpdateCachedImage = m_pdfImageCachingPolicy == PDFImageCachingPolicy::ClipBoundsOnly || !m_cachedImageBuffer;
205 if (!forceUpdateCachedImage && cacheParametersMatch(context, dstRect, srcRect)) {
206 // Adjust the view-port rectangles if no re-caching will happen.
207 m_cachedImageRect.move(FloatSize(dstRect.location() - m_cachedDestinationRect.location()));
208 m_cachedDestinationRect = dstRect;
209 return;
210 }
211
212 switch (m_pdfImageCachingPolicy) {
213 case PDFImageCachingPolicy::Disabled:
214 return;
215 case PDFImageCachingPolicy::BelowMemoryLimit:
216 // Keep the memory used by the cached image below some threshold, otherwise WebKit process
217 // will jetsam if it exceeds its memory limit. Only a rectangle from the PDF may be cached.
218 m_cachedImageRect = cachedImageRect(context, dstRect);
219 break;
220 case PDFImageCachingPolicy::ClipBoundsOnly:
221 m_cachedImageRect = intersection(context.clipBounds(), dstRect);
222 break;
223 case PDFImageCachingPolicy::Enabled:
224 m_cachedImageRect = dstRect;
225 break;
226 }
227
228 FloatSize cachedImageSize = FloatRect(enclosingIntRect(m_cachedImageRect)).size();
229
230 // Cache the PDF image only if the size of the new image won't exceed the cache threshold.
231 if (m_pdfImageCachingPolicy == PDFImageCachingPolicy::BelowMemoryLimit) {
232 IntSize scaledSize = ImageBuffer::compatibleBufferSize(cachedImageSize, context);
233 if (s_allDecodedDataSize + scaledSize.unclampedArea() * 4 - m_cachedBytes > s_maxDecodedDataSize) {
234 destroyDecodedData();
235 return;
236 }
237 }
238
239 m_cachedImageBuffer = ImageBuffer::createCompatibleBuffer(cachedImageSize, context);
240 if (!m_cachedImageBuffer) {
241 destroyDecodedData();
242 return;
243 }
244
245 auto& bufferContext = m_cachedImageBuffer->context();
246 // We need to transform the coordinate system such that top-left of m_cachedImageRect will be mapped to the
247 // top-left of dstRect. Although only m_cachedImageRect.size() of the image copied, the sizes of srcRect
248 // and dstRect should be passed to this function because they are used to calculate the image scaling.
249 transformContextForPainting(bufferContext, dstRect, FloatRect(m_cachedImageRect.location(), srcRect.size()));
250 drawPDFPage(bufferContext);
251
252 m_cachedTransform = context.getCTM(GraphicsContext::DefinitelyIncludeDeviceScale);
253 m_cachedDestinationRect = dstRect;
254 m_cachedSourceRect = srcRect;
255 ++m_cachingCountForTesting;
256
257 IntSize backendSize = m_cachedImageBuffer->backendSize();
258 decodedSizeChanged(backendSize.unclampedArea() * 4);
259}

261ImageDrawResult PDFDocumentImage::draw(GraphicsContext& context, const FloatRect& dstRect, const FloatRect& srcRect, const ImagePaintingOptions& options)
262{
263 if (!m_document || !m_hasPage)
264 return ImageDrawResult::DidNothing;
265
266 updateCachedImageIfNeeded(context, dstRect, srcRect);
267
268 {
269 GraphicsContextStateSaver stateSaver(context);
270 context.setCompositeOperation(options.compositeOperator());
271
272 if (m_cachedImageBuffer) {
273 // Draw the ImageBuffer 'm_cachedImageBuffer' to the rectangle 'm_cachedImageRect'
274 // on the destination context. Since the pixels of the rectangle 'm_cachedImageRect'
275 // of the source PDF was copied to 'm_cachedImageBuffer', the sizes of the source
276 // and the destination rectangles will be equal and no scaling will be needed here.
277 context.drawImageBuffer(*m_cachedImageBuffer, m_cachedImageRect);
278 }
279 else {
280 transformContextForPainting(context, dstRect, srcRect);
281 drawPDFPage(context);
282 }
283 }
284
285 if (imageObserver())
286 imageObserver()->didDraw(*this);
287
288 return ImageDrawResult::DidDraw;
289}

306void PDFDocumentImage::computeBoundsForCurrentPage()
307{
308 ASSERT(pageCount() > 0);
309 CGPDFPageRef cgPage = CGPDFDocumentGetPage(m_document.get(), 1);
310 CGRect mediaBox = CGPDFPageGetBoxRect(cgPage, kCGPDFMediaBox);
311
312 // Get crop box (not always there). If not, use media box.
313 CGRect r = CGPDFPageGetBoxRect(cgPage, kCGPDFCropBox);
314 if (!CGRectIsEmpty(r))
315 m_cropBox = r;
316 else
317 m_cropBox = mediaBox;
318
319 m_rotationDegrees = CGPDFPageGetRotationAngle(cgPage);
320}

322unsigned PDFDocumentImage::pageCount() const
323{
324 return CGPDFDocumentGetNumberOfPages(m_document.get());
325}

327static void applyRotationForPainting(GraphicsContext& context, FloatSize size, int rotationDegrees)
328{
329 if (rotationDegrees == 90)
330 context.translate(0, size.height());
331 else if (rotationDegrees == 180)
332 context.translate(size);
333 else if (rotationDegrees == 270)
334 context.translate(size.width(), 0);
335
336 context.rotate(-deg2rad(static_cast<float>(rotationDegrees)));
337}
338
339void PDFDocumentImage::drawPDFPage(GraphicsContext& context)
340{
341 applyRotationForPainting(context, size(), m_rotationDegrees);
342
343 context.translate(-m_cropBox.location());
344
345#if USE(DIRECT2D)
346 notImplemented();
347#else
348 // CGPDF pages are indexed from 1.
349#if PLATFORM(COCOA)
350 CGContextDrawPDFPageWithAnnotations(context.platformContext(), CGPDFDocumentGetPage(m_document.get(), 1), nullptr);
351#else
352 CGContextDrawPDFPage(context.platformContext(), CGPDFDocumentGetPage(m_document.get(), 1));
353#endif
354#endif
355}

361RetainPtr<CFMutableDataRef> PDFDocumentImage::convertPostScriptDataToPDF(RetainPtr<CFDataRef>&& postScriptData)
362{
363 // Convert PostScript to PDF using the Quartz 2D API.
364 // http://developer.apple.com/documentation/GraphicsImaging/Conceptual/drawingwithquartz2d/dq_ps_convert/chapter_16_section_1.html
365
366 CGPSConverterCallbacks callbacks = { };
367 auto converter = adoptCF(CGPSConverterCreate(0, &callbacks, 0));
368 auto provider = adoptCF(CGDataProviderCreateWithCFData(postScriptData.get()));
369 auto pdfData = adoptCF(CFDataCreateMutable(kCFAllocatorDefault, 0));
370 auto consumer = adoptCF(CGDataConsumerCreateWithCFData(pdfData.get()));
371
372 CGPSConverterConvert(converter.get(), provider.get(), consumer.get(), 0);
373 return pdfData;
374}
