31namespace WebCore {
32
33Ref<AudioTrackPrivateWebM> AudioTrackPrivateWebM::create(webm::TrackEntry&& trackEntry)
34{
35 return adoptRef(*new AudioTrackPrivateWebM(WTFMove(trackEntry)));
36}
37
38AudioTrackPrivateWebM::AudioTrackPrivateWebM(webm::TrackEntry&& trackEntry)
39 : m_track(WTFMove(trackEntry))
40{
41 if (m_track.is_enabled.is_present())
42 setEnabled(m_track.is_enabled.value());
43}
44
45AtomString AudioTrackPrivateWebM::id() const
46{
47 if (m_trackID.isNull()) {
48 auto uid = trackUID();
49 m_trackID = uid ? AtomString::number(*uid) : emptyAtom();
50 }
51 return m_trackID;
52}
53
54std::optional<uint64_t> AudioTrackPrivateWebM::trackUID() const
55{
56 if (m_track.track_uid.is_present())
57 return m_track.track_uid.value();
58 return std::nullopt;
59}
60
61std::optional<bool> AudioTrackPrivateWebM::defaultEnabled() const
62{
63 if (m_track.is_enabled.is_present())
64 return m_track.is_enabled.value();
65 return std::nullopt;
66}
67
68AtomString AudioTrackPrivateWebM::label() const
69{
70 if (m_label.isNull())
71 m_label = m_track.name.is_present() ? AtomString::fromUTF8(m_track.name.value().data(), m_track.name.value().length()) : emptyAtom();
72 return m_label;
73}
74
75AtomString AudioTrackPrivateWebM::language() const
76{
77 if (m_language.isNull())
78 m_language = m_track.language.is_present() ? AtomString::fromUTF8(m_track.language.value().data(), m_track.language.value().length()) : emptyAtom();
79 return m_language;
80}
81
82int AudioTrackPrivateWebM::trackIndex() const
83{
84 if (m_track.track_number.is_present())
85 return m_track.track_number.value();
86 return 0;
87}
