34template<typename T, typename U, std::size_t size, std::size_t... indices> std::array<T, size> convertArray(U (&array)[size], std::index_sequence<indices...>)
35{
36 return { { array[indices]... } };
37}

39template<typename T, typename U, std::size_t size> inline std::array<T, size> convertArray(U (&array)[size])
40{
41 return convertArray<T>(array, std::make_index_sequence<size> { });
42}

44static inline std::optional<SystemFontKind> matchSystemFontUse(const AtomString& string)
45{
46 if (equalLettersIgnoringASCIICase(string, "-webkit-system-font")
47 || equalLettersIgnoringASCIICase(string, "-apple-system")
48 || equalLettersIgnoringASCIICase(string, "-apple-system-font")
49 || equalLettersIgnoringASCIICase(string, "system-ui")
50 || equalLettersIgnoringASCIICase(string, "ui-sans-serif"))
51 return SystemFontKind::SystemUI;
52
53#if HAVE(DESIGN_SYSTEM_UI_FONTS)
54 if (equalLettersIgnoringASCIICase(string, "ui-serif"))
55 return SystemFontKind::UISerif;
56 if (equalLettersIgnoringASCIICase(string, "ui-monospace"))
57 return SystemFontKind::UIMonospace;
58 if (equalLettersIgnoringASCIICase(string, "ui-rounded"))
59 return SystemFontKind::UIRounded;
60#endif

82 auto compareAsPointer = [](const AtomString& lhs, const AtomString& rhs) {
83 return lhs.impl() < rhs.impl();
84 };
85 static auto strings = makeNeverDestroyed([&compareAsPointer] {
86 auto result = convertArray<AtomString>(styles);
87 std::sort(result.begin(), result.end(), compareAsPointer);
88 return result;
89 }());
90 if (std::binary_search(strings.get().begin(), strings.get().end(), string, compareAsPointer))
91 return SystemFontKind::TextStyle;
92
93 return std::nullopt;
94}

101unsigned FontCascadeDescription::effectiveFamilyCount() const
102{
103 // FIXME: Move all the other system font keywords from fontWithFamilySpecialCase() to here.
104 unsigned result = 0;
105 for (unsigned i = 0; i < familyCount(); ++i) {
106 const auto& cssFamily = familyAt(i);
107 if (auto use = matchSystemFontUse(cssFamily))
108 result += systemFontCascadeList(*this, cssFamily, *use, shouldAllowUserInstalledFonts()).size();
109 else
110 ++result;
111 }
112 return result;
113}

115FontFamilySpecification FontCascadeDescription::effectiveFamilyAt(unsigned index) const
116{
117 // The special cases in this function need to match the behavior in FontCacheCoreText.cpp. This code
118 // is used for regular (element style) lookups, and the code in FontDescriptionCocoa.cpp is used when
119 // src:local(special-cased-name) is specified inside an @font-face block.
120 // FIXME: Currently, an @font-face block corresponds to a single item in the font-family: fallback list, which
121 // means that "src:local(system-ui)" can't follow the Core Text cascade list (the way it does for regular lookups).
122 // These two behaviors should be unified, which would hopefully allow us to delete this duplicate code.
123 for (unsigned i = 0; i < familyCount(); ++i) {
124 const auto& cssFamily = familyAt(i);
125 if (auto use = matchSystemFontUse(cssFamily)) {
126 auto cascadeList = systemFontCascadeList(*this, cssFamily, *use, shouldAllowUserInstalledFonts());
127 if (index < cascadeList.size())
128 return FontFamilySpecification(cascadeList[index].get());
129 index -= cascadeList.size();
130 }
131 else if (!index)
132 return cssFamily;
133 else
134 --index;
135 }
136 ASSERT_NOT_REACHED();
137 return nullAtom();
138}

140AtomString FontDescription::platformResolveGenericFamily(UScriptCode script, const AtomString& locale, const AtomString& familyName)
141{
142 ASSERT((locale.isNull() && script == USCRIPT_COMMON) || !locale.isNull());
143 if (script == USCRIPT_COMMON)
144 return nullAtom();
145
146 // FIXME: Use the system font database to handle standardFamily
147 if (familyName == serifFamily)
148 return SystemFontDatabaseCoreText::singleton().serifFamily(locale.string());
149 if (familyName == sansSerifFamily)
150 return SystemFontDatabaseCoreText::singleton().sansSerifFamily(locale.string());
151 if (familyName == cursiveFamily)
152 return SystemFontDatabaseCoreText::singleton().cursiveFamily(locale.string());
153 if (familyName == fantasyFamily)
154 return SystemFontDatabaseCoreText::singleton().fantasyFamily(locale.string());
155 if (familyName == monospaceFamily)
156 return SystemFontDatabaseCoreText::singleton().monospaceFamily(locale.string());
157
158 return nullAtom();
159}
