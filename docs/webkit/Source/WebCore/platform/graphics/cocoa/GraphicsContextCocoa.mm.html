63static bool drawFocusRingAtTime(CGContextRef context, NSTimeInterval timeOffset, const Color& color)
64{
65#if USE(APPKIT)
66 CGFocusRingStyle focusRingStyle;
67 BOOL needsRepaint = NSInitializeCGFocusRingStyleForTime(NSFocusRingOnly, &focusRingStyle, timeOffset);
68#else
69 BOOL needsRepaint = NO;
70 UNUSED_PARAM(timeOffset);
71
72 CGFocusRingStyle focusRingStyle;
73 focusRingStyle.version = 0;
74 focusRingStyle.tint = kCGFocusRingTintBlue;
75 focusRingStyle.ordering = kCGFocusRingOrderingNone;
76 focusRingStyle.alpha = [PAL::getUIFocusRingStyleClass() maxAlpha];
77 focusRingStyle.radius = [PAL::getUIFocusRingStyleClass() borderThickness];
78 focusRingStyle.threshold = [PAL::getUIFocusRingStyleClass() alphaThreshold];
79 focusRingStyle.bounds = CGRectZero;
80#endif
81
82 // We want to respect the CGContext clipping and also not overpaint any
83 // existing focus ring. The way to do this is set accumulate to
84 // -1. According to CoreGraphics, the reasoning for this behavior has been
85 // lost in time.
86 focusRingStyle.accumulate = -1;
87 auto style = adoptCF(CGStyleCreateFocusRingWithColor(&focusRingStyle, cachedCGColor(color)));
88
89 CGContextStateSaver stateSaver(context);
90
91 CGContextSetStyle(context, style.get());
92 CGContextFillPath(context);
93
94 return needsRepaint;
95}

97inline static void drawFocusRing(CGContextRef context, const Color& color)
98{
99 drawFocusRingAtTime(context, std::numeric_limits<double>::max(), color);
100}
101
102static void drawFocusRingToContext(CGContextRef context, CGPathRef focusRingPath, const Color& color)
103{
104 CGContextBeginPath(context);
105 CGContextAddPath(context, focusRingPath);
106 drawFocusRing(context, color);
107}

109void GraphicsContextCG::drawFocusRing(const Path& path, float, float, const Color& color)
110{
111 if (path.isNull())
112 return;
113
114 drawFocusRingToContext(platformContext(), path.platformPath(), color);
115}

126void GraphicsContextCG::drawFocusRing(const Path& path, double timeOffset, bool& needsRedraw, const Color& color)

128 if (path.isNull())
129 return;
130
131 needsRedraw = drawFocusRingToContextAtTime(platformContext(), path.platformPath(), timeOffset, color);

134void GraphicsContextCG::drawFocusRing(const Vector<FloatRect>& rects, double timeOffset, bool& needsRedraw, const Color& color)
135{
136 RetainPtr<CGMutablePathRef> focusRingPath = adoptCF(CGPathCreateMutable());
137 for (const auto& rect : rects)
138 CGPathAddRect(focusRingPath.get(), 0, CGRect(rect));
139
140 needsRedraw = drawFocusRingToContextAtTime(platformContext(), focusRingPath.get(), timeOffset, color);
141}

143#endif // PLATFORM(MAC)

145void GraphicsContextCG::drawFocusRing(const Vector<FloatRect>& rects, float, float offset, const Color& color)
146{
147 RetainPtr<CGMutablePathRef> focusRingPath = adoptCF(CGPathCreateMutable());
148 for (auto& rect : rects)
149 CGPathAddRect(focusRingPath.get(), 0, CGRectInset(rect, -offset, -offset));
150
151 drawFocusRingToContext(platformContext(), focusRingPath.get(), color);
152}

162static CGColorRef colorForMarkerLineStyle(DocumentMarkerLineStyle::Mode style, bool useDarkMode)
163{
164 switch (style) {
165 // Red
166 case DocumentMarkerLineStyle::Mode::Spelling:
167 return cachedCGColor(useDarkMode ? SRGBA<uint8_t> { 255, 140, 140, 217 } : SRGBA<uint8_t> { 255, 59, 48, 191 });
168 // Blue
169 case DocumentMarkerLineStyle::Mode::DictationAlternatives:
170 case DocumentMarkerLineStyle::Mode::TextCheckingDictationPhraseWithAlternatives:
171 case DocumentMarkerLineStyle::Mode::AutocorrectionReplacement:
172 return cachedCGColor(useDarkMode ? SRGBA<uint8_t> { 40, 145, 255, 217 } : SRGBA<uint8_t> { 0, 122, 255, 191 });
173 // Green
174 case DocumentMarkerLineStyle::Mode::Grammar:
175 return cachedCGColor(useDarkMode ? SRGBA<uint8_t> { 50, 215, 75, 217 } : SRGBA<uint8_t> { 25, 175, 50, 191 });
176 }
177}

179void GraphicsContextCG::drawDotsForDocumentMarker(const FloatRect& rect, DocumentMarkerLineStyle style)
180{
181 // We want to find the number of full dots, so we're solving the equations:
182 // dotDiameter = height
183 // dotDiameter / dotGap = 13.247 / 9.457
184 // numberOfGaps = numberOfDots - 1
185 // dotDiameter * numberOfDots + dotGap * numberOfGaps = width
186
187 auto width = rect.width();
188 auto dotDiameter = rect.height();
189 auto dotGap = dotDiameter * 9.457 / 13.247;
190 auto numberOfDots = (width + dotGap) / (dotDiameter + dotGap);
191 auto numberOfWholeDots = static_cast<unsigned>(numberOfDots);
192 auto numberOfWholeGaps = numberOfWholeDots - 1;
193
194 // Center the dots
195 auto offset = (width - (dotDiameter * numberOfWholeDots + dotGap * numberOfWholeGaps)) / 2;
196
197 auto circleColor = colorForMarkerLineStyle(style.mode, style.shouldUseDarkAppearance);
198
199 CGContextRef platformContext = this->platformContext();
200 CGContextStateSaver stateSaver { platformContext };
201 CGContextSetFillColorWithColor(platformContext, circleColor);
202 for (unsigned i = 0; i < numberOfWholeDots; ++i) {
203 auto location = rect.location();
204 location.move(offset + i * (dotDiameter + dotGap), 0);
205 auto size = FloatSize(dotDiameter, dotDiameter);
206 CGContextAddEllipseInRect(platformContext, FloatRect(location, size));
207 }
208 CGContextSetCompositeOperation(platformContext, kCGCompositeSover);
209 CGContextFillPath(platformContext);
210}
