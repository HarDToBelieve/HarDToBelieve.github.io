53class IOSurface final {
54 WTF_MAKE_FAST_ALLOCATED;
55public:
56 enum class Format {
57 BGRA,
58 YUV422,
59#if HAVE(IOSURFACE_RGB10)
60 RGB10,
61 RGB10A8,
62#endif
63 };
64
65 WEBCORE_EXPORT static IOSurface::Format formatForPixelFormat(WebCore::PixelFormat);
66
67 class Locker {
68 public:
69 enum class AccessMode {
70 ReadOnly,
71 ReadWrite
72 };
73
74 Locker(IOSurface& surface, AccessMode mode = AccessMode::ReadOnly)
75 : m_surface(surface)
76 , m_flags(flagsFromMode(mode))
77 {
78 IOSurfaceLock(m_surface.surface(), m_flags, nullptr);
79 }
80
81 ~Locker()
82 {
83 IOSurfaceUnlock(m_surface.surface(), m_flags, nullptr);
84 }
85
86 void * surfaceBaseAddress() const
87 {
88 return IOSurfaceGetBaseAddress(m_surface.surface());
89 }
90
91 private:
92 static uint32_t flagsFromMode(AccessMode mode)
93 {
94 return mode == AccessMode::ReadOnly ? kIOSurfaceLockReadOnly : 0;
95 }
96 IOSurface& m_surface;
97 uint32_t m_flags;
98 };
