43static CMVideoCodecType videoCodecTypeFromRFC4281Type(StringView type)
44{
45 if (type.startsWith("mp4v"))
46 return kCMVideoCodecType_MPEG4Video;
47 if (type.startsWith("avc1") || type.startsWith("avc3"))
48 return kCMVideoCodecType_H264;
49 if (type.startsWith("hvc1") || type.startsWith("hev1"))
50 return kCMVideoCodecType_HEVC;
51#if ENABLE(VP9)
52 if (type.startsWith("vp09"))
53 return kCMVideoCodecType_VP9;
54#endif
55 return 0;
56}

58static std::optional<MediaCapabilitiesInfo> computeMediaCapabilitiesInfo(const MediaDecodingConfiguration& configuration)
59{
60 MediaCapabilitiesInfo info;
61
62 if (configuration.video) {
63 auto& videoConfiguration = configuration.video.value();
64 MediaEngineSupportParameters parameters { };
65 parameters.type = ContentType(videoConfiguration.contentType);
66 parameters.isMediaSource = configuration.type == MediaDecodingType::MediaSource;
67 if (MediaPlayer::supportsType(parameters) != MediaPlayer::SupportsType::IsSupported)
68 return std::nullopt;
69
70 auto codecs = parameters.type.codecs();
71 if (codecs.size() != 1)
72 return std::nullopt;
73
74 info.supported = true;
75 auto& codec = codecs[0];
76 auto videoCodecType = videoCodecTypeFromRFC4281Type(codec);
77 if (!videoCodecType && !(codec.startsWith("dvh1") || codec.startsWith("dvhe")))
78 return std::nullopt;
79
80 bool hdrSupported = videoConfiguration.colorGamut || videoConfiguration.hdrMetadataType || videoConfiguration.transferFunction;
81 bool alphaChannel = videoConfiguration.alphaChannel && videoConfiguration.alphaChannel.value();
82
83 if (videoCodecType == kCMVideoCodecType_HEVC) {
84 auto parameters = parseHEVCCodecParameters(codec);
85 if (!parameters)
86 return std::nullopt;
87 auto parsedInfo = validateHEVCParameters(*parameters, alphaChannel, hdrSupported);
88 if (!parsedInfo)
89 return std::nullopt;
90 info = *parsedInfo;
91 } else if (codec.startsWith("dvh1") || codec.startsWith("dvhe")) {
92 auto parameters = parseDoViCodecParameters(codec);
93 if (!parameters)
94 return std::nullopt;
95 auto parsedInfo = validateDoViParameters(*parameters, alphaChannel, hdrSupported);
96 if (!parsedInfo)
97 return std::nullopt;
98 info = *parsedInfo;
99#if ENABLE(VP9)
100 } else if (videoCodecType == kCMVideoCodecType_VP9) {
101 if (!configuration.canExposeVP9)
102 return std::nullopt;
103 auto parameters = parseVPCodecParameters(codec);
104 if (!parameters)
105 return std::nullopt;
106 auto parsedInfo = validateVPParameters(*parameters, videoConfiguration);
107 if (!parsedInfo)
108 return std::nullopt;
109 info = *parsedInfo;
110#endif
111 } else {
112 if (alphaChannel || hdrSupported)
113 return std::nullopt;
114
115 if (canLoad_VideoToolbox_VTIsHardwareDecodeSupported()) {
116 info.powerEfficient = VTIsHardwareDecodeSupported(videoCodecType);
117 info.smooth = true;
118 }
119 }
120 }
121
122 if (configuration.audio) {
123 MediaEngineSupportParameters parameters { };
124 parameters.type = ContentType(configuration.audio.value().contentType);
125 parameters.isMediaSource = configuration.type == MediaDecodingType::MediaSource;
126 if (MediaPlayer::supportsType(parameters) != MediaPlayer::SupportsType::IsSupported)
127 return std::nullopt;
128
129 if (configuration.audio->spatialRendering.value_or(false)) {
130 auto context = PAL::OutputContext::sharedAudioPresentationOutputContext();
131 if (!context)
132 return std::nullopt;
133
134 auto devices = context->outputDevices();
135 if (devices.isEmpty() || !WTF::allOf(devices, [](auto& device) {
136 return device.supportsSpatialAudio();
137 }))
138 return std::nullopt;
139
140 // Only multichannel audio can be spatially rendered.
141 if (!configuration.audio->channels.isNull() && configuration.audio->channels.toDouble() <= 2)
142 return std::nullopt;
143 }
144 info.supported = true;
145 }
146
147 return info;
148}
149
150void createMediaPlayerDecodingConfigurationCocoa(MediaDecodingConfiguration&& configuration, WTF::Function<void(MediaCapabilitiesDecodingInfo&&)>&& callback)
151{
152 auto info = computeMediaCapabilitiesInfo(configuration);
153 if (!info)
154 callback({ { }, WTFMove(configuration) });
155 else {
156 MediaCapabilitiesDecodingInfo infoWithConfiguration = { WTFMove(*info), WTFMove(configuration) };
157 callback(WTFMove(infoWithConfiguration));
158 }
159}
