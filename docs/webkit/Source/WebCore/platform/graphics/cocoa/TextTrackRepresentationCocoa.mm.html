43@interface WebCoreTextTrackRepresentationCocoaHelper : NSObject <CALayerDelegate> {
44 WebCore::TextTrackRepresentationCocoa* _parent;
45}
46- (id)initWithParent:(WebCore::TextTrackRepresentationCocoa*)parent;
47@property (assign) WebCore::TextTrackRepresentationCocoa* parent;
48@end
49
50@implementation WebCoreTextTrackRepresentationCocoaHelper
51- (id)initWithParent:(WebCore::TextTrackRepresentationCocoa*)parent
52{
53 if (!(self = [super init]))
54 return nil;
55
56 self.parent = parent;
57
58 return self;
59}
60
61- (void)dealloc
62{
63 self.parent = nullptr;
64 [super dealloc];
65}
66
67- (void)setParent:(WebCore::TextTrackRepresentationCocoa*)parent
68{
69 if (_parent)
70 [_parent->platformLayer() removeObserver:self forKeyPath:@"bounds"];
71
72 _parent = parent;
73
74 if (_parent)
75 [_parent->platformLayer() addObserver:self forKeyPath:@"bounds" options:0 context:0];
76}
77
78- (WebCore::TextTrackRepresentationCocoa*)parent
79{
80 return _parent;
81}
82
83- (void)observeValueForKeyPath:(NSString *)keyPath ofObject:(id)object change:(NSDictionary *)change context:(void *)context
84{
85 UNUSED_PARAM(change);
86 UNUSED_PARAM(context);
87#if PLATFORM(IOS_FAMILY)
88 WebThreadRun(^{
89 if (_parent && [keyPath isEqual:@"bounds"] && object == _parent->platformLayer())
90 _parent->client().textTrackRepresentationBoundsChanged(_parent->bounds());
91 });
92#else
93 if (_parent && [keyPath isEqual:@"bounds"] && object == _parent->platformLayer())
94 _parent->boundsChanged();
95#endif
96}
97
98- (id)actionForLayer:(CALayer *)layer forKey:(NSString *)event
99{
100 UNUSED_PARAM(layer);
101 UNUSED_PARAM(event);
102 // Returning a NSNull from this delegate method disables all implicit CALayer actions.
103 return [NSNull null];
104}
105
106@end

110std::unique_ptr<TextTrackRepresentation> TextTrackRepresentation::create(TextTrackRepresentationClient& client)
111{
112 return makeUnique<TextTrackRepresentationCocoa>(client);
113}
114
115TextTrackRepresentationCocoa::TextTrackRepresentationCocoa(TextTrackRepresentationClient& client)
116 : m_client(client)
117 , m_layer(adoptNS([[CALayer alloc] init]))
118 , m_delegate(adoptNS([[WebCoreTextTrackRepresentationCocoaHelper alloc] initWithParent:this]))
119{
120 [m_layer.get() setDelegate:m_delegate.get()];
121 [m_layer.get() setContentsGravity:kCAGravityBottom];
122}
123
124TextTrackRepresentationCocoa::~TextTrackRepresentationCocoa()
125{
126 [m_layer.get() setDelegate:nil];
127 [m_delegate.get() setParent:nullptr];
128}
129
130void TextTrackRepresentationCocoa::update()
131{
132 if (auto representation = m_client.createTextTrackRepresentationImage())
133 [m_layer.get() setContents:(__bridge id)representation->nativeImage()->platformImage().get()];
134}
135
136void TextTrackRepresentationCocoa::setContentScale(float scale)
137{
138 [m_layer.get() setContentsScale:scale];
139}
140
141void TextTrackRepresentationCocoa::setHidden(bool hidden) const
142{
143 [m_layer.get() setHidden:hidden];
144}
145
146IntRect TextTrackRepresentationCocoa::bounds() const
147{
148 return enclosingIntRect(FloatRect([m_layer.get() bounds]));
149}
150
151void TextTrackRepresentationCocoa::boundsChanged()
152{
153 callOnMainThread([weakThis = makeWeakPtr(*this)] {
154 if (weakThis)
155 weakThis->client().textTrackRepresentationBoundsChanged(weakThis->bounds());
156 });
157}
