59- (void)_web_setLayerBoundsOrigin:(CGPoint)origin

62 bounds.origin = origin;
63 [self setBounds:bounds];
64}

66- (void)_web_setLayerTopLeftPosition:(CGPoint)position
67{
68 CGSize layerSize = [self bounds].size;
69 CGPoint anchorPoint = [self anchorPoint];
70 CGPoint newPosition = CGPointMake(position.x + anchorPoint.x * layerSize.width, position.y + anchorPoint.y * layerSize.height);
71 if (isnan(newPosition.x) || isnan(newPosition.y)) {
72 WTFLogAlways("Attempt to call [CALayer setPosition] with NaN: newPosition=(%f, %f) position=(%f, %f) anchorPoint=(%f, %f)",
73 newPosition.x, newPosition.y, position.x, position.y, anchorPoint.x, anchorPoint.y);
74 ASSERT_NOT_REACHED();
75 return;
76 }
77
78 [self setPosition:newPosition];
79}

91- (BOOL)_web_maskContainsPoint:(CGPoint)point
92{
93 if (!self.mask)
94 return NO;
95
96 CGPoint pointInMask = [self.mask convertPoint:point fromLayer:self];
97 if (auto *shapeMask = dynamic_objc_cast<CAShapeLayer>(self.mask)) {
98 bool isEvenOddFill = [shapeMask.fillRule isEqualToString:kCAFillRuleEvenOdd];
99 return CGPathContainsPoint(shapeMask.path, nullptr, pointInMask, isEvenOddFill);
100 }
101
102 return [self.mask containsPoint:pointInMask];
103}
104
105- (BOOL)_web_maskMayIntersectRect:(CGRect)rect
106{
107 if (!self.mask)
108 return NO;
109
110 CGRect rectInMask = [self.mask convertRect:rect fromLayer:self];
111 if (auto *shapeMask = dynamic_objc_cast<CAShapeLayer>(self.mask)) {
112 CGRect pathBounds = CGPathGetPathBoundingBox(shapeMask.path);
113 return CGRectIntersectsRect(pathBounds, rectInMask);
114 }
115
116 return CGRectIntersectsRect(self.mask.bounds, rectInMask);
117}
