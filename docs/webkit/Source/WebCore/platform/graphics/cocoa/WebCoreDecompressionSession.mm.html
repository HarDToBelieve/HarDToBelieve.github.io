72void WebCoreDecompressionSession::setTimebase(CMTimebaseRef timebase)
73{
74 if (m_timebase == timebase)
75 return;
76
77 if (m_timebase)
78 PAL::CMTimebaseRemoveTimerDispatchSource(m_timebase.get(), m_timerSource.get());
79
80 m_timebase = timebase;
81
82 if (m_timebase) {
83 if (!m_timerSource) {
84 m_timerSource = adoptOSObject(dispatch_source_create(DISPATCH_SOURCE_TYPE_TIMER, 0, 0, dispatch_get_main_queue()));
85 dispatch_source_set_event_handler(m_timerSource.get(), [this] {
86 automaticDequeue();
87 });
88 dispatch_activate(m_timerSource.get());
89 }
90 PAL::CMTimebaseAddTimerDispatchSource(m_timebase.get(), m_timerSource.get());
91 }

107void WebCoreDecompressionSession::enqueueSample(CMSampleBufferRef sampleBuffer, bool displaying)
108{
109 CMItemCount itemCount = 0;
110 if (noErr != PAL::CMSampleBufferGetSampleTimingInfoArray(sampleBuffer, 0, nullptr, &itemCount))
111 return;
112
113 Vector<CMSampleTimingInfo> timingInfoArray;
114 timingInfoArray.grow(itemCount);
115 if (noErr != PAL::CMSampleBufferGetSampleTimingInfoArray(sampleBuffer, itemCount, timingInfoArray.data(), nullptr))
116 return;
117
118 // CMBufferCallbacks contains 64-bit pointers that aren't 8-byte aligned. To suppress the linker
119 // warning about this, we prepend 4 bytes of padding when building.
120 const size_t padSize = 4;
121
122 if (!m_producerQueue) {
123 CMBufferQueueRef outQueue { nullptr };
124#pragma pack(push, 4)
125 struct BufferCallbacks { uint8_t pad[padSize]; CMBufferCallbacks callbacks; } callbacks { { }, {
126 0,
127 nullptr,
128 &getDecodeTime,
129 &getPresentationTime,
130 &getDuration,
131 nullptr,
132 &compareBuffers,
133 nullptr,
134 nullptr,
135 } };
136#pragma pack(pop)
137 static_assert(sizeof(callbacks.callbacks.version) == sizeof(uint32_t), "Version field must be 4 bytes");
138 static_assert(alignof(BufferCallbacks) == 4, "CMBufferCallbacks struct must have 4 byte alignment");
139
140 PAL::CMBufferQueueCreate(kCFAllocatorDefault, kMaximumCapacity, &callbacks.callbacks, &outQueue);
141 m_producerQueue = adoptCF(outQueue);
142 }
143
144 if (!m_consumerQueue) {
145 CMBufferQueueRef outQueue { nullptr };
146#pragma pack(push, 4)
147 struct BufferCallbacks { uint8_t pad[padSize]; CMBufferCallbacks callbacks; } callbacks { { }, {
148 0,
149 nullptr,
150 &getDecodeTime,
151 &getPresentationTime,
152 &getDuration,
153 nullptr,
154 &compareBuffers,
155 nullptr,
156 nullptr,
157 } };
158#pragma pack(pop)
159 static_assert(sizeof(callbacks.callbacks.version) == sizeof(uint32_t), "Version field must be 4 bytes");
160 static_assert(alignof(BufferCallbacks) == 4, "CMBufferCallbacks struct alignment must be 4");
161
162 PAL::CMBufferQueueCreate(kCFAllocatorDefault, kMaximumCapacity, &callbacks.callbacks, &outQueue);
163 m_consumerQueue = adoptCF(outQueue);
164 }
165
166 ++m_framesBeingDecoded;
167
168 LOG(Media, "WebCoreDecompressionSession::enqueueSample(%p) - framesBeingDecoded(%d)", this, m_framesBeingDecoded);
169
170 m_decompressionQueue->dispatch([protectedThis = makeRefPtr(*this), strongBuffer = retainPtr(sampleBuffer), displaying] {
171 protectedThis->decodeSample(strongBuffer.get(), displaying);
172 });
173}

175bool WebCoreDecompressionSession::shouldDecodeSample(CMSampleBufferRef sample, bool displaying)
176{
177 if (!displaying)
178 return true;
179
180 if (!m_timebase)
181 return true;
182
183 auto currentTime = PAL::CMTimebaseGetTime(m_timebase.get());
184 auto presentationStartTime = PAL::CMSampleBufferGetPresentationTimeStamp(sample);
185 auto duration = PAL::CMSampleBufferGetDuration(sample);
186 auto presentationEndTime = PAL::CMTimeAdd(presentationStartTime, duration);
187 if (PAL::CMTimeCompare(presentationEndTime, currentTime) >= 0)
188 return true;
189
190 CFArrayRef attachments = PAL::CMSampleBufferGetSampleAttachmentsArray(sample, false);
191 if (!attachments)
192 return true;
193
194 for (CFIndex index = 0, count = CFArrayGetCount(attachments); index < count; ++index) {
195 CFDictionaryRef attachmentDict = (CFDictionaryRef)CFArrayGetValueAtIndex(attachments, index);
196 CFBooleanRef dependedOn = (CFBooleanRef)CFDictionaryGetValue(attachmentDict, PAL::kCMSampleAttachmentKey_IsDependedOnByOthers);
197 if (dependedOn && !CFBooleanGetValue(dependedOn))
198 return false;
199 }
200
201 return true;
202}

204void WebCoreDecompressionSession::ensureDecompressionSessionForSample(CMSampleBufferRef sample)
205{
206 if (isInvalidated())
207 return;
208
209 CMVideoFormatDescriptionRef videoFormatDescription = PAL::CMSampleBufferGetFormatDescription(sample);
210 if (m_decompressionSession && !VTDecompressionSessionCanAcceptFormatDescription(m_decompressionSession.get(), videoFormatDescription)) {
211 VTDecompressionSessionWaitForAsynchronousFrames(m_decompressionSession.get());
212 m_decompressionSession = nullptr;
213 }
214
215 if (!m_decompressionSession) {
216 CMVideoFormatDescriptionRef videoFormatDescription = PAL::CMSampleBufferGetFormatDescription(sample);
217 auto videoDecoderSpecification = @{ (__bridge NSString *)kVTVideoDecoderSpecification_EnableHardwareAcceleratedVideoDecoder: @( m_hardwareDecoderEnabled ) };
218
219 NSDictionary *attributes;
220 if (m_mode == OpenGL)
221 attributes = @{ (__bridge NSString *)kCVPixelBufferIOSurfacePropertiesKey: @{ /* empty dictionary */ } };
222 else {
223 attributes = @{
224 (__bridge NSString *)kCVPixelBufferPixelFormatTypeKey: @(kCVPixelFormatType_32BGRA),
225 (__bridge NSString *)kCVPixelBufferIOSurfacePropertiesKey: @{ /* empty dictionary */ }
226 };
227 }
228
229 VTDecompressionSessionRef decompressionSessionOut = nullptr;
230 if (noErr == VTDecompressionSessionCreate(kCFAllocatorDefault, videoFormatDescription, (__bridge CFDictionaryRef)videoDecoderSpecification, (__bridge CFDictionaryRef)attributes, nullptr, &decompressionSessionOut)) {
231 m_decompressionSession = adoptCF(decompressionSessionOut);
232 CFArrayRef rawSuggestedQualityOfServiceTiers = nullptr;
233 VTSessionCopyProperty(decompressionSessionOut, kVTDecompressionPropertyKey_SuggestedQualityOfServiceTiers, kCFAllocatorDefault, &rawSuggestedQualityOfServiceTiers);
234 m_qosTiers = adoptCF(rawSuggestedQualityOfServiceTiers);
235 m_currentQosTier = 0;
236 resetQosTier();
237 }
238 }

241void WebCoreDecompressionSession::decodeSample(CMSampleBufferRef sample, bool displaying)
242{
243 if (isInvalidated())
244 return;
245
246 ensureDecompressionSessionForSample(sample);
247
248 VTDecodeInfoFlags flags { kVTDecodeFrame_EnableTemporalProcessing };
249 if (!displaying)
250 flags |= kVTDecodeFrame_DoNotOutputFrame;
251
252 if (!shouldDecodeSample(sample, displaying)) {
253 ++m_totalVideoFrames;
254 ++m_droppedVideoFrames;
255 --m_framesBeingDecoded;
256 maybeBecomeReadyForMoreMediaData();
257 return;
258 }
259
260 MonotonicTime startTime = MonotonicTime::now();
261 VTDecompressionSessionDecodeFrameWithOutputHandler(m_decompressionSession.get(), sample, flags, nullptr, [this, displaying, startTime](OSStatus status, VTDecodeInfoFlags infoFlags, CVImageBufferRef imageBuffer, CMTime presentationTimeStamp, CMTime presentationDuration) {
262 double deltaRatio = (MonotonicTime::now() - startTime).seconds() / PAL::CMTimeGetSeconds(presentationDuration);
263
264 updateQosWithDecodeTimeStatistics(deltaRatio);
265 handleDecompressionOutput(displaying, status, infoFlags, imageBuffer, presentationTimeStamp, presentationDuration);
266 });
267}

269RetainPtr<CVPixelBufferRef> WebCoreDecompressionSession::decodeSampleSync(CMSampleBufferRef sample)
270{
271 if (isInvalidated())
272 return nullptr;
273
274 ensureDecompressionSessionForSample(sample);
275
276 RetainPtr<CVPixelBufferRef> pixelBuffer;
277 VTDecodeInfoFlags flags { 0 };
278 WTF::Semaphore syncDecompressionOutputSemaphore { 0 };
279 VTDecompressionSessionDecodeFrameWithOutputHandler(m_decompressionSession.get(), sample, flags, nullptr, [&] (OSStatus, VTDecodeInfoFlags, CVImageBufferRef imageBuffer, CMTime, CMTime) mutable {
280 if (imageBuffer && CFGetTypeID(imageBuffer) == CVPixelBufferGetTypeID())
281 pixelBuffer = (CVPixelBufferRef)imageBuffer;
282 syncDecompressionOutputSemaphore.signal();
283 });
284 syncDecompressionOutputSemaphore.wait();
285 return pixelBuffer;
286}

288void WebCoreDecompressionSession::handleDecompressionOutput(bool displaying, OSStatus status, VTDecodeInfoFlags infoFlags, CVImageBufferRef rawImageBuffer, CMTime presentationTimeStamp, CMTime presentationDuration)
289{
290 ++m_totalVideoFrames;
291 if (infoFlags & kVTDecodeInfo_FrameDropped)
292 ++m_droppedVideoFrames;
293
294 CMVideoFormatDescriptionRef rawImageBufferDescription = nullptr;
295 if (status != noErr || noErr != PAL::CMVideoFormatDescriptionCreateForImageBuffer(kCFAllocatorDefault, rawImageBuffer, &rawImageBufferDescription)) {
296 ++m_corruptedVideoFrames;
297 --m_framesBeingDecoded;
298 maybeBecomeReadyForMoreMediaData();
299 return;
300 }
301 RetainPtr<CMVideoFormatDescriptionRef> imageBufferDescription = adoptCF(rawImageBufferDescription);
302
303 CMSampleTimingInfo imageBufferTiming {
304 presentationDuration,
305 presentationTimeStamp,
306 presentationTimeStamp,
307 };
308
309 CMSampleBufferRef rawImageSampleBuffer = nullptr;
310 if (noErr != PAL::CMSampleBufferCreateReadyWithImageBuffer(kCFAllocatorDefault, rawImageBuffer, imageBufferDescription.get(), &imageBufferTiming, &rawImageSampleBuffer)) {
311 ++m_corruptedVideoFrames;
312 --m_framesBeingDecoded;
313 maybeBecomeReadyForMoreMediaData();
314 return;
315 }
316
317 m_enqueingQueue->dispatch([protectedThis = makeRefPtr(this), imageSampleBuffer = adoptCF(rawImageSampleBuffer), displaying] {
318 protectedThis->enqueueDecodedSample(imageSampleBuffer.get(), displaying);
319 });
320}

322RetainPtr<CVPixelBufferRef> WebCoreDecompressionSession::getFirstVideoFrame()
323{
324 if (!m_producerQueue || PAL::CMBufferQueueIsEmpty(m_producerQueue.get()))
325 return nullptr;
326
327 RetainPtr<CMSampleBufferRef> currentSample = adoptCF(checked_cf_cast<CMSampleBufferRef>(PAL::CMBufferQueueDequeueAndRetain(m_producerQueue.get())));
328 RetainPtr<CVPixelBufferRef> imageBuffer = (CVPixelBufferRef)PAL::CMSampleBufferGetImageBuffer(currentSample.get());
329 ASSERT(CFGetTypeID(imageBuffer.get()) == CVPixelBufferGetTypeID());
330
331 maybeBecomeReadyForMoreMediaData();
332
333 return imageBuffer;

336void WebCoreDecompressionSession::automaticDequeue()
337{
338 if (!m_timebase)
339 return;
340
341 auto time = PAL::toMediaTime(PAL::CMTimebaseGetTime(m_timebase.get()));
342 LOG(Media, "WebCoreDecompressionSession::automaticDequeue(%p) - purging all samples before time(%s)", this, toString(time).utf8().data());
343
344 MediaTime nextFireTime = MediaTime::positiveInfiniteTime();
345 bool releasedImageBuffers = false;
346
347 while (CMSampleBufferRef firstSample = checked_cf_cast<CMSampleBufferRef>(PAL::CMBufferQueueGetHead(m_producerQueue.get()))) {
348 MediaTime presentationTimestamp = PAL::toMediaTime(PAL::CMSampleBufferGetPresentationTimeStamp(firstSample));
349 MediaTime duration = PAL::toMediaTime(PAL::CMSampleBufferGetDuration(firstSample));
350 MediaTime presentationEndTimestamp = presentationTimestamp + duration;
351 if (time > presentationEndTimestamp) {
352 CFRelease(PAL::CMBufferQueueDequeueAndRetain(m_producerQueue.get()));
353 releasedImageBuffers = true;
354 continue;
355 }
356
357#if !LOG_DISABLED
358 auto begin = PAL::toMediaTime(PAL::CMBufferQueueGetFirstPresentationTimeStamp(m_producerQueue.get()));
359 auto end = PAL::toMediaTime(PAL::CMBufferQueueGetEndPresentationTimeStamp(m_producerQueue.get()));
360 LOG(Media, "WebCoreDecompressionSession::automaticDequeue(%p) - queue(%s -> %s)", this, toString(begin).utf8().data(), toString(end).utf8().data());
361#endif
362
363 nextFireTime = presentationEndTimestamp;
364 break;
365 }
366
367 if (releasedImageBuffers)
368 maybeBecomeReadyForMoreMediaData();
369
370 LOG(Media, "WebCoreDecompressionSession::automaticDequeue(%p) - queue empty", this);
371 PAL::CMTimebaseSetTimerDispatchSourceNextFireTime(m_timebase.get(), m_timerSource.get(), PAL::toCMTime(nextFireTime), 0);
372}

374void WebCoreDecompressionSession::enqueueDecodedSample(CMSampleBufferRef sample, bool displaying)
375{
376 if (isInvalidated())
377 return;
378
379 --m_framesBeingDecoded;
380
381 if (!displaying) {
382 maybeBecomeReadyForMoreMediaData();
383 return;
384 }
385
386 bool shouldNotify = true;
387
388 if (displaying && m_timebase) {
389 auto currentRate = PAL::CMTimebaseGetRate(m_timebase.get());
390 auto currentTime = PAL::toMediaTime(PAL::CMTimebaseGetTime(m_timebase.get()));
391 auto presentationStartTime = PAL::toMediaTime(PAL::CMSampleBufferGetPresentationTimeStamp(sample));
392 auto presentationEndTime = presentationStartTime + PAL::toMediaTime(PAL::CMSampleBufferGetDuration(sample));
393 if (currentTime < presentationStartTime || currentTime >= presentationEndTime)
394 shouldNotify = false;
395
396 if (currentRate > 0 && presentationEndTime < currentTime) {
397#if !LOG_DISABLED
398 auto begin = PAL::toMediaTime(PAL::CMBufferQueueGetFirstPresentationTimeStamp(m_producerQueue.get()));
399 auto end = PAL::toMediaTime(PAL::CMBufferQueueGetEndPresentationTimeStamp(m_producerQueue.get()));
400 LOG(Media, "WebCoreDecompressionSession::enqueueDecodedSample(%p) - dropping frame late by %s, framesBeingDecoded(%d), producerQueue(%s -> %s)", this, toString(presentationEndTime - currentTime).utf8().data(), m_framesBeingDecoded, toString(begin).utf8().data(), toString(end).utf8().data());
401#endif
402 ++m_droppedVideoFrames;
403 return;
404 }
405 }
406
407 PAL::CMBufferQueueEnqueue(m_producerQueue.get(), sample);
408
409#if !LOG_DISABLED
410 auto begin = PAL::toMediaTime(PAL::CMBufferQueueGetFirstPresentationTimeStamp(m_producerQueue.get()));
411 auto end = PAL::toMediaTime(PAL::CMBufferQueueGetEndPresentationTimeStamp(m_producerQueue.get()));
412 auto presentationTime = PAL::toMediaTime(PAL::CMSampleBufferGetPresentationTimeStamp(sample));
413 LOG(Media, "WebCoreDecompressionSession::enqueueDecodedSample(%p) - presentationTime(%s), framesBeingDecoded(%d), producerQueue(%s -> %s)", this, toString(presentationTime).utf8().data(), m_framesBeingDecoded, toString(begin).utf8().data(), toString(end).utf8().data());
414#endif
415
416 if (m_timebase)
417 PAL::CMTimebaseSetTimerDispatchSourceToFireImmediately(m_timebase.get(), m_timerSource.get());
418
419 if (!m_hasAvailableFrameCallback)
420 return;
421
422 if (!shouldNotify)
423 return;
424
425 RunLoop::main().dispatch([protectedThis = makeRefPtr(this), callback = WTFMove(m_hasAvailableFrameCallback)] {
426 callback();
427 });
428}

467RetainPtr<CVPixelBufferRef> WebCoreDecompressionSession::imageForTime(const MediaTime& time, ImageForTimeFlags flags)
468{
469 if (PAL::CMBufferQueueIsEmpty(m_producerQueue.get())) {
470 LOG(Media, "WebCoreDecompressionSession::imageForTime(%p) - time(%s), queue empty", this, toString(time).utf8().data());
471 return nullptr;
472 }
473
474 bool allowEarlier = flags == WebCoreDecompressionSession::AllowEarlier;
475 bool allowLater = flags == WebCoreDecompressionSession::AllowLater;
476
477 MediaTime startTime = PAL::toMediaTime(PAL::CMBufferQueueGetFirstPresentationTimeStamp(m_producerQueue.get()));
478#if !LOG_DISABLED
479 MediaTime endTime = PAL::toMediaTime(PAL::CMBufferQueueGetEndPresentationTimeStamp(m_producerQueue.get()));
480#endif
481 if (!allowLater && time < startTime) {
482 LOG(Media, "WebCoreDecompressionSession::imageForTime(%p) - time(%s) too early for queue(%s -> %s)", this, toString(time).utf8().data(), toString(startTime).utf8().data(), toString(endTime).utf8().data());
483 return nullptr;
484 }
485
486 bool releasedImageBuffers = false;
487
488 while (CMSampleBufferRef firstSample = checked_cf_cast<CMSampleBufferRef>(PAL::CMBufferQueueGetHead(m_producerQueue.get()))) {
489 MediaTime presentationTimestamp = PAL::toMediaTime(PAL::CMSampleBufferGetPresentationTimeStamp(firstSample));
490 MediaTime duration = PAL::toMediaTime(PAL::CMSampleBufferGetDuration(firstSample));
491 MediaTime presentationEndTimestamp = presentationTimestamp + duration;
492 if (!allowLater && presentationTimestamp > time)
493 return nullptr;
494 if (!allowEarlier && presentationEndTimestamp < time) {
495 CFRelease(PAL::CMBufferQueueDequeueAndRetain(m_producerQueue.get()));
496 releasedImageBuffers = true;
497 continue;
498 }
499
500 RetainPtr<CMSampleBufferRef> currentSample = adoptCF(checked_cf_cast<CMSampleBufferRef>(PAL::CMBufferQueueDequeueAndRetain(m_producerQueue.get())));
501 RetainPtr<CVPixelBufferRef> imageBuffer = (CVPixelBufferRef)PAL::CMSampleBufferGetImageBuffer(currentSample.get());
502 ASSERT(CFGetTypeID(imageBuffer.get()) == CVPixelBufferGetTypeID());
503
504 if (m_timebase)
505 PAL::CMTimebaseSetTimerDispatchSourceToFireImmediately(m_timebase.get(), m_timerSource.get());
506
507 maybeBecomeReadyForMoreMediaData();
508
509 LOG(Media, "WebCoreDecompressionSession::imageForTime(%p) - found sample for time(%s) in queue(%s -> %s)", this, toString(time).utf8().data(), toString(startTime).utf8().data(), toString(endTime).utf8().data());
510 return imageBuffer;
511 }
512
513 if (m_timebase)
514 PAL::CMTimebaseSetTimerDispatchSourceToFireImmediately(m_timebase.get(), m_timerSource.get());
515
516 if (releasedImageBuffers)
517 maybeBecomeReadyForMoreMediaData();
518
519 LOG(Media, "WebCoreDecompressionSession::imageForTime(%p) - no matching sample for time(%s) in queue(%s -> %s)", this, toString(time).utf8().data(), toString(startTime).utf8().data(), toString(endTime).utf8().data());
520 return nullptr;
521}

559void WebCoreDecompressionSession::resetQosTier()
560{
561 if (!m_qosTiers || !m_decompressionSession)
562 return;
563
564 if (m_currentQosTier < 0 || m_currentQosTier >= CFArrayGetCount(m_qosTiers.get()))
565 return;
566
567 auto tier = (CFDictionaryRef)CFArrayGetValueAtIndex(m_qosTiers.get(), m_currentQosTier);
568 LOG(Media, "WebCoreDecompressionSession::resetQosTier(%p) - currentQosTier(%ld), tier(%@)", this, m_currentQosTier, [(__bridge NSDictionary *)tier description]);
569
570 VTSessionSetProperties(m_decompressionSession.get(), tier);
571 m_framesSinceLastQosCheck = 0;
572}

583 resetQosTier();
584}

586void WebCoreDecompressionSession::decreaseQosTier()

594 --m_currentQosTier;
595 resetQosTier();
596}

598void WebCoreDecompressionSession::updateQosWithDecodeTimeStatistics(double ratio)
599{
600 static const double kMovingAverageAlphaValue = 0.1;
601 static const unsigned kNumberOfFramesBeforeSwitchingTiers = 60;
602 static const double kHighWaterDecodeRatio = 1.;
603 static const double kLowWaterDecodeRatio = 0.5;
604
605 if (!m_timebase)
606 return;
607
608 double rate = PAL::CMTimebaseGetRate(m_timebase.get());
609 if (!rate)
610 rate = 1;
611
612 m_decodeRatioMovingAverage += kMovingAverageAlphaValue * (ratio - m_decodeRatioMovingAverage) * rate;
613 if (++m_framesSinceLastQosCheck < kNumberOfFramesBeforeSwitchingTiers)
614 return;
615
616 LOG(Media, "WebCoreDecompressionSession::updateQosWithDecodeTimeStatistics(%p) - framesSinceLastQosCheck(%ld), decodeRatioMovingAverage(%g)", this, m_framesSinceLastQosCheck, m_decodeRatioMovingAverage);
617 if (m_decodeRatioMovingAverage > kHighWaterDecodeRatio)
618 increaseQosTier();
619 else if (m_decodeRatioMovingAverage < kLowWaterDecodeRatio)
620 decreaseQosTier();
621 m_framesSinceLastQosCheck = 0;
622}
