45FontCascade::FontCascade(const FontPlatformData& fontData, FontSmoothingMode fontSmoothingMode)
46 : m_fonts(FontCascadeFonts::createForPlatformFont(fontData))
47 , m_enableKerning(computeEnableKerning())
48 , m_requiresShaping(computeRequiresShaping())
49{
50 m_fontDescription.setFontSmoothing(fontSmoothingMode);
51 m_fontDescription.setSpecifiedSize(CTFontGetSize(fontData.ctFont()));

54 m_fontDescription.setWeight((CTFontGetSymbolicTraits(fontData.ctFont()) & kCTFontTraitBold) ? boldWeightValue() : normalWeightValue());
55}

57static const AffineTransform& rotateLeftTransform()
58{
59 static AffineTransform result(0, -1, 1, 0, 0, 0);
60 return result;
61}
62
63AffineTransform computeOverallTextMatrix(const Font& font)
64{
65 auto& platformData = font.platformData();
66 AffineTransform result;
67 if (!platformData.isColorBitmapFont())
68 result = CTFontGetMatrix(platformData.ctFont());
69 result.setB(-result.b());
70 result.setD(-result.d());
71 if (platformData.syntheticOblique()) {
72 float obliqueSkew = tanf(deg2rad(FontCascade::syntheticObliqueAngle()));
73 if (platformData.orientation() == FontOrientation::Vertical) {
74 if (font.isTextOrientationFallback())
75 result = AffineTransform(1, obliqueSkew, 0, 1, 0, 0) * result;
76 else
77 result = AffineTransform(1, -obliqueSkew, 0, 1, 0, 0) * result;
78 } else
79 result = AffineTransform(1, 0, -obliqueSkew, 1, 0, 0) * result;
80 }
81
82 // We're emulating the behavior of CGContextSetTextPosition() by adding constant amounts to each glyph's position
83 // (see fillVectorWithHorizontalGlyphPositions() and fillVectorWithVerticalGlyphPositions()).
84 // CGContextSetTextPosition() has the side effect of clobbering the E and F fields of the text matrix,
85 // so we do that explicitly here.
86 result.setE(0);
87 result.setF(0);
88 return result;
89}

91AffineTransform computeVerticalTextMatrix(const Font& font, const AffineTransform& previousTextMatrix)
92{
93 ASSERT_UNUSED(font, font.platformData().orientation() == FontOrientation::Vertical);
94 // The translation here ("e" and "f" fields) are irrelevant, because
95 // this matrix is inverted in fillVectorWithVerticalGlyphPositions to place the glyphs in the CTM's coordinate system.
96 // All we're trying to do here is rotate the text matrix so glyphs appear visually upright.
97 // We have to include the previous text matrix because it includes things like synthetic oblique.
98 return rotateLeftTransform() * previousTextMatrix;
99}

121void fillVectorWithHorizontalGlyphPositions(Vector<CGPoint, 256>& positions, CGContextRef context, const CGSize* advances, unsigned count, const FloatPoint& point)
122{
123 // The input positions are in the context's coordinate system, without the text matrix.
124 // However, the positions that CT/CG accept are in the text matrix's coordinate system.
125 // CGContextGetTextMatrix() gives us the matrix that maps from text's coordinate system to the context's (non-text) coordinate system.
126 // We need to figure out what to deliver CT, inside the text's coordinate system, such that it ends up coincident with the input in the context's coordinate system.
127 //
128 // CTM * text matrix * positions we need to deliver to CT = CTM * input positions
129 // Solving for the positions we need to deliver to CT, we get
130 // positions we need to deliver to CT = inverse(text matrix) * input positions
131 CGAffineTransform matrix = CGAffineTransformInvert(CGContextGetTextMatrix(context));
132 positions[0] = CGPointApplyAffineTransform(point, matrix);
133 for (unsigned i = 1; i < count; ++i) {
134 CGSize advance = CGSizeApplyAffineTransform(advances[i - 1], matrix);
135 positions[i].x = positions[i - 1].x + advance.width;
136 positions[i].y = positions[i - 1].y + advance.height;
137 }
138}

140static void fillVectorWithVerticalGlyphPositions(Vector<CGPoint, 256>& positions, CGContextRef context, const CGSize* translations, const CGSize* advances, unsigned count, const FloatPoint& point, float ascentDelta)
141{
142 CGAffineTransform transform = CGAffineTransformInvert(CGContextGetTextMatrix(context));
143
144 auto position = CGPointMake(point.x(), point.y() + ascentDelta);
145 for (unsigned i = 0; i < count; ++i) {
146 CGSize translation = CGSizeApplyAffineTransform(translations[i], rotateLeftTransform());
147 positions[i] = CGPointApplyAffineTransform(CGPointMake(position.x - translation.width, position.y + translation.height), transform);
148 position.x += advances[i].width;
149 position.y += advances[i].height;
150 }
151}

153static inline bool shouldUseLetterpressEffect(const GraphicsContext& context)
154{
155#if ENABLE(LETTERPRESS)

159 return false;

163static void showGlyphsWithAdvances(const FloatPoint& point, const Font& font, CGContextRef context, const CGGlyph* glyphs, const CGSize* advances, unsigned count, const AffineTransform& textMatrix)
164{
165 if (!count)
166 return;
167
168 const FontPlatformData& platformData = font.platformData();
169 Vector<CGPoint, 256> positions(count);
170 if (platformData.orientation() == FontOrientation::Vertical) {
171 ScopedTextMatrix savedMatrix(computeVerticalTextMatrix(font, textMatrix), context);
172
173 Vector<CGSize, 256> translations(count);
174 CTFontGetVerticalTranslationsForGlyphs(platformData.ctFont(), glyphs, translations.data(), count);
175
176 auto ascentDelta = font.fontMetrics().floatAscent(IdeographicBaseline) - font.fontMetrics().floatAscent();
177 fillVectorWithVerticalGlyphPositions(positions, context, translations.data(), advances, count, point, ascentDelta);
178 CTFontDrawGlyphs(platformData.ctFont(), glyphs, positions.data(), count, context);
179 } else {
180 fillVectorWithHorizontalGlyphPositions(positions, context, advances, count, point);
181 CTFontDrawGlyphs(platformData.ctFont(), glyphs, positions.data(), count, context);
182 }
183}

185static void setCGFontRenderingMode(GraphicsContext& context)
186{
187 CGContextRef cgContext = context.platformContext();
188 CGContextSetShouldAntialiasFonts(cgContext, true);
189
190 CGAffineTransform contextTransform = CGContextGetCTM(cgContext);
191 bool isTranslationOrIntegralScale = WTF::isIntegral(contextTransform.a) && WTF::isIntegral(contextTransform.d) && contextTransform.b == 0.f && contextTransform.c == 0.f;
192 bool isRotated = ((contextTransform.b || contextTransform.c) && (contextTransform.a || contextTransform.d));
193 bool doSubpixelQuantization = isTranslationOrIntegralScale || (!isRotated && context.shouldSubpixelQuantizeFonts());
194
195 CGContextSetShouldSubpixelPositionFonts(cgContext, true);
196 CGContextSetShouldSubpixelQuantizeFonts(cgContext, doSubpixelQuantization);
197}

199void FontCascade::drawGlyphs(GraphicsContext& context, const Font& font, const GlyphBufferGlyph* glyphs, const GlyphBufferAdvance* advances, unsigned numGlyphs, const FloatPoint& anchorPoint, FontSmoothingMode smoothingMode)
200{
201 const auto& platformData = font.platformData();
202 if (!platformData.size())
203 return;
204
205 if (isInGPUProcess() && font.hasAnyComplexColorFormatGlyphs(glyphs, numGlyphs)) {
206 ASSERT_NOT_REACHED();
207 return;
208 }
209
210 CGContextRef cgContext = context.platformContext();
211
212 bool shouldAntialias = true;
213 bool shouldSmoothFonts = true;
214
215 if (!font.allowsAntialiasing())
216 smoothingMode = FontSmoothingMode::NoSmoothing;
217
218 switch (smoothingMode) {
219 case FontSmoothingMode::Antialiased:
220 shouldSmoothFonts = false;
221 break;
222 case FontSmoothingMode::AutoSmoothing:
223 case FontSmoothingMode::SubpixelAntialiased:
224 shouldAntialias = true;
225 break;
226 case FontSmoothingMode::NoSmoothing:
227 shouldAntialias = false;
228 shouldSmoothFonts = false;
229 break;
230 }
231
232 if (!shouldUseSmoothing())
233 shouldSmoothFonts = false;
234
235#if !PLATFORM(IOS_FAMILY)
236 bool originalShouldUseFontSmoothing = CGContextGetShouldSmoothFonts(cgContext);
237 if (shouldSmoothFonts != originalShouldUseFontSmoothing)
238 CGContextSetShouldSmoothFonts(cgContext, shouldSmoothFonts);
239#endif
240
241 bool originalShouldAntialias = CGContextGetShouldAntialias(cgContext);
242 if (shouldAntialias != originalShouldAntialias)
243 CGContextSetShouldAntialias(cgContext, shouldAntialias);
244
245 bool useLetterpressEffect = shouldUseLetterpressEffect(context);
246 UNUSED_VARIABLE(useLetterpressEffect);
247 FloatPoint point = anchorPoint;
248
249 auto textMatrix = computeOverallTextMatrix(font);
250 ScopedTextMatrix restorer(textMatrix, cgContext);
251
252 setCGFontRenderingMode(context);
253 CGContextSetFontSize(cgContext, platformData.size());
254
255 FloatSize shadowOffset;
256 float shadowBlur;
257 Color shadowColor;
258 context.getShadow(shadowOffset, shadowBlur, shadowColor);
259
260 AffineTransform contextCTM = context.getCTM();
261 float syntheticBoldOffset = font.syntheticBoldOffset();
262 if (syntheticBoldOffset && !contextCTM.isIdentityOrTranslationOrFlipped()) {
263 FloatSize horizontalUnitSizeInDevicePixels = contextCTM.mapSize(FloatSize(1, 0));
264 float horizontalUnitLengthInDevicePixels = sqrtf(horizontalUnitSizeInDevicePixels.width() * horizontalUnitSizeInDevicePixels.width() + horizontalUnitSizeInDevicePixels.height() * horizontalUnitSizeInDevicePixels.height());
265 if (horizontalUnitLengthInDevicePixels) {
266 // Make sure that a scaled down context won't blow up the gap between the glyphs.
267 syntheticBoldOffset = std::min(syntheticBoldOffset, syntheticBoldOffset / horizontalUnitLengthInDevicePixels);
268 }
269 };
270
271 bool hasSimpleShadow = context.textDrawingMode() == TextDrawingMode::Fill && shadowColor.isValid() && !shadowBlur && !platformData.isColorBitmapFont() && (!context.shadowsIgnoreTransforms() || contextCTM.isIdentityOrTranslationOrFlipped()) && !context.isInTransparencyLayer();
272 if (hasSimpleShadow) {
273 // Paint simple shadows ourselves instead of relying on CG shadows, to avoid losing subpixel antialiasing.
274 context.clearShadow();
275 Color fillColor = context.fillColor();
276 Color shadowFillColor = shadowColor.colorWithAlphaMultipliedBy(fillColor.alphaAsFloat());
277 context.setFillColor(shadowFillColor);
278 float shadowTextX = point.x() + shadowOffset.width();
279 // If shadows are ignoring transforms, then we haven't applied the Y coordinate flip yet, so down is negative.
280 float shadowTextY = point.y() + shadowOffset.height() * (context.shadowsIgnoreTransforms() ? -1 : 1);
281 showGlyphsWithAdvances(FloatPoint(shadowTextX, shadowTextY), font, cgContext, glyphs, advances, numGlyphs, textMatrix);
282 if (syntheticBoldOffset)
283 showGlyphsWithAdvances(FloatPoint(shadowTextX + syntheticBoldOffset, shadowTextY), font, cgContext, glyphs, advances, numGlyphs, textMatrix);
284 context.setFillColor(fillColor);
285 }
286
287#if ENABLE(LETTERPRESS)
288 if (useLetterpressEffect)
289 showLetterpressedGlyphsWithAdvances(point, font, context, glyphs, advances, numGlyphs);
290 else
291#endif
292 {
293 showGlyphsWithAdvances(point, font, cgContext, glyphs, advances, numGlyphs, textMatrix);
294 }
295
296 if (syntheticBoldOffset)
297 showGlyphsWithAdvances(FloatPoint(point.x() + syntheticBoldOffset, point.y()), font, cgContext, glyphs, advances, numGlyphs, textMatrix);
298
299 if (hasSimpleShadow)
300 context.setShadow(shadowOffset, shadowBlur, shadowColor);
301
302#if !PLATFORM(IOS_FAMILY)
303 if (shouldSmoothFonts != originalShouldUseFontSmoothing)
304 CGContextSetShouldSmoothFonts(cgContext, originalShouldUseFontSmoothing);
305#endif
306
307 if (shouldAntialias != originalShouldAntialias)
308 CGContextSetShouldAntialias(cgContext, originalShouldAntialias);
309}

318const Font* FontCascade::fontForCombiningCharacterSequence(const UChar* characters, size_t length) const
319{
320 UChar32 baseCharacter;
321 size_t baseCharacterLength = 0;
322 U16_NEXT(characters, baseCharacterLength, length, baseCharacter);
323
324 GlyphData baseCharacterGlyphData = glyphDataForCharacter(baseCharacter, false, NormalVariant);
325
326 if (!baseCharacterGlyphData.glyph)
327 return nullptr;
328
329 if (length == baseCharacterLength)
330 return baseCharacterGlyphData.font;
331
332 bool triedBaseCharacterFont = false;
333
334 for (unsigned i = 0; !fallbackRangesAt(i).isNull(); ++i) {
335 const Font* font = fallbackRangesAt(i).fontForCharacter(baseCharacter);
336 if (!font)
337 continue;
338#if PLATFORM(IOS_FAMILY)
339 if (baseCharacter >= 0x0600 && baseCharacter <= 0x06ff && font->shouldNotBeUsedForArabic())
340 continue;
341#endif
342 if (font->platformData().orientation() == FontOrientation::Vertical) {
343 if (isCJKIdeographOrSymbol(baseCharacter)) {
344 if (!font->hasVerticalGlyphs())
345 font = &font->brokenIdeographFont();
346 } else if (m_fontDescription.nonCJKGlyphOrientation() == NonCJKGlyphOrientation::Mixed) {
347 const Font& verticalRightFont = font->verticalRightOrientationFont();
348 Glyph verticalRightGlyph = verticalRightFont.glyphForCharacter(baseCharacter);
349 if (verticalRightGlyph == baseCharacterGlyphData.glyph)
350 font = &verticalRightFont;
351 } else {
352 const Font& uprightFont = font->uprightOrientationFont();
353 Glyph uprightGlyph = uprightFont.glyphForCharacter(baseCharacter);
354 if (uprightGlyph != baseCharacterGlyphData.glyph)
355 font = &uprightFont;
356 }
357 }
358
359 if (font == baseCharacterGlyphData.font)
360 triedBaseCharacterFont = true;
361
362 if (font->canRenderCombiningCharacterSequence(characters, length))
363 return font;
364 }
365
366 if (!triedBaseCharacterFont && baseCharacterGlyphData.font && baseCharacterGlyphData.font->canRenderCombiningCharacterSequence(characters, length))
367 return baseCharacterGlyphData.font;
368
369 return Font::systemFallback();
370}
