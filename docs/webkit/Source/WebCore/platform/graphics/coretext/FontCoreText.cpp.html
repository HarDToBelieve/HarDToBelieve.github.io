55static inline bool caseInsensitiveCompare(CFStringRef a, CFStringRef b)
56{
57 return a && CFStringCompare(a, b, kCFCompareCaseInsensitive) == kCFCompareEqualTo;
58}

68bool fontFamilyShouldNotBeUsedForArabic(CFStringRef fontFamilyName)
69{
70 if (!fontFamilyName)
71 return false;
72
73 // Times New Roman and Arial are not performant enough to use. <rdar://problem/21333326>
74 // FIXME <rdar://problem/12096835> remove this function once the above bug is fixed.
75 return (CFStringCompare(CFSTR("Times New Roman"), fontFamilyName, 0) == kCFCompareEqualTo)
76 || (CFStringCompare(CFSTR("Arial"), fontFamilyName, 0) == kCFCompareEqualTo);
77}

81static bool shouldUseAdjustment(CTFontRef font)
82{
83 RetainPtr<CFStringRef> familyName = adoptCF(CTFontCopyFamilyName(font));
84
85 if (!familyName || !CFStringGetLength(familyName.get()))
86 return false;
87
88 return caseInsensitiveCompare(familyName.get(), CFSTR("Times"))
89 || caseInsensitiveCompare(familyName.get(), CFSTR("Helvetica"))
90 || caseInsensitiveCompare(familyName.get(), CFSTR(".Helvetica NeueUI"));
91}
92
93#else
94
95static bool needsAscentAdjustment(CFStringRef familyName)
96{
97 return familyName && (caseInsensitiveCompare(familyName, CFSTR("Times"))
98 || caseInsensitiveCompare(familyName, CFSTR("Helvetica"))
99 || caseInsensitiveCompare(familyName, CFSTR("Courier")));
100}

109bool fontHasTable(CTFontRef ctFont, unsigned tableTag)
110{
111 auto tableTags = adoptCF(CTFontCopyAvailableTables(ctFont, kCTFontTableOptionNoOptions));
112 if (!tableTags)
113 return false;
114 CFIndex numTables = CFArrayGetCount(tableTags.get());
115 for (CFIndex index = 0; index < numTables; ++index) {
116 auto tag = static_cast<CTFontTableTag>(reinterpret_cast<uintptr_t>(CFArrayGetValueAtIndex(tableTags.get(), index)));
117 if (tag == tableTag)
118 return true;
119 }
120 return false;
121}
122
123bool fontHasEitherTable(CTFontRef ctFont, unsigned tableTag1, unsigned tableTag2)
124{
125 auto tableTags = adoptCF(CTFontCopyAvailableTables(ctFont, kCTFontTableOptionNoOptions));
126 if (!tableTags)
127 return false;
128 CFIndex numTables = CFArrayGetCount(tableTags.get());
129 for (CFIndex index = 0; index < numTables; ++index) {
130 auto tag = static_cast<CTFontTableTag>(reinterpret_cast<uintptr_t>(CFArrayGetValueAtIndex(tableTags.get(), index)));
131 if (tag == tableTag1 || tag == tableTag2)
132 return true;
133 }
134 return false;
135}

137void Font::platformInit()
138{
139#if PLATFORM(IOS_FAMILY)
140 m_syntheticBoldOffset = m_platformData.syntheticBold() ? ceilf(m_platformData.size() / 24.0f) : 0.f;
141#else
142 m_syntheticBoldOffset = m_platformData.syntheticBold() ? 1.0f : 0.f;
143#endif
144
145#if PLATFORM(WIN)
146 m_scriptCache = 0;
147 m_scriptFontProperties = nullptr;
148 if (m_platformData.useGDI())
149 return initGDIFont();
150#endif
151
152 unsigned unitsPerEm = CTFontGetUnitsPerEm(m_platformData.font());
153 float pointSize = m_platformData.size();
154 CGFloat capHeight = pointSize ? CTFontGetCapHeight(m_platformData.font()) : 0;
155 CGFloat lineGap = pointSize ? CTFontGetLeading(m_platformData.font()) : 0;
156 CGFloat ascent = pointSize ? CTFontGetAscent(m_platformData.font()) : 0;
157 CGFloat descent = pointSize ? CTFontGetDescent(m_platformData.font()) : 0;
158
159 // The Open Font Format describes the OS/2 USE_TYPO_METRICS flag as follows:
160 // "If set, it is strongly recommended to use OS/2.sTypoAscender - OS/2.sTypoDescender+ OS/2.sTypoLineGap as a value for default line spacing for this font."
161 // On OS X, we only apply this rule in the important case of fonts with a MATH table.
162 if (fontHasTable(m_platformData.ctFont(), kCTFontTableMATH)) {
163 short typoAscent, typoDescent, typoLineGap;
164 if (OpenType::tryGetTypoMetrics(m_platformData.font(), typoAscent, typoDescent, typoLineGap)) {
165 ascent = scaleEmToUnits(typoAscent, unitsPerEm) * pointSize;
166 descent = -scaleEmToUnits(typoDescent, unitsPerEm) * pointSize;
167 lineGap = scaleEmToUnits(typoLineGap, unitsPerEm) * pointSize;
168 }
169 }
170
171 auto familyName = adoptCF(CTFontCopyFamilyName(m_platformData.font()));
172
173 // Disable antialiasing when rendering with Ahem because many tests require this.
174 if (isAhemFont(familyName.get()))
175 m_allowsAntialiasing = false;
176
177#if PLATFORM(MAC)
178 // We need to adjust Times, Helvetica, and Courier to closely match the
179 // vertical metrics of their Microsoft counterparts that are the de facto
180 // web standard. The AppKit adjustment of 20% is too big and is
181 // incorrectly added to line spacing, so we use a 15% adjustment instead
182 // and add it to the ascent.
183 if (origin() == Origin::Local && needsAscentAdjustment(familyName.get()))
184 ascent += std::round((ascent + descent) * 0.15f);
185#endif
186
187 // Compute line spacing before the line metrics hacks are applied.
188 float lineSpacing = lroundf(ascent) + lroundf(descent) + lroundf(lineGap);
189
190#if PLATFORM(MAC)
191 // Hack Hiragino line metrics to allow room for marked text underlines.
192 // <rdar://problem/5386183>
193 if (descent < 3 && lineGap >= 3 && familyName && CFStringHasPrefix(familyName.get(), CFSTR("Hiragino"))) {
194 lineGap -= 3 - descent;
195 descent = 3;
196 }
197#endif
198
199 if (platformData().orientation() == FontOrientation::Vertical && !isTextOrientationFallback())
200 m_hasVerticalGlyphs = fontHasVerticalGlyphs(m_platformData.ctFont());
201
202#if PLATFORM(IOS_FAMILY)
203 CGFloat adjustment = shouldUseAdjustment(m_platformData.font()) ? ceil((ascent + descent) * kLineHeightAdjustment) : 0;
204
205 lineGap = ceilf(lineGap);
206 lineSpacing = ceil(ascent) + adjustment + ceil(descent) + lineGap;
207 ascent = ceilf(ascent + adjustment);
208 descent = ceilf(descent);
209
210 m_shouldNotBeUsedForArabic = fontFamilyShouldNotBeUsedForArabic(familyName.get());
211#endif
212
213 CGFloat xHeight = 0;
214 if (m_platformData.size()) {
215 if (platformData().orientation() == FontOrientation::Horizontal) {
216 // Measure the actual character "x", since it's possible for it to extend below the baseline, and we need the
217 // reported x-height to only include the portion of the glyph that is above the baseline.
218 Glyph xGlyph = glyphForCharacter('x');
219 if (xGlyph)
220 xHeight = -CGRectGetMinY(platformBoundsForGlyph(xGlyph));
221 else
222 xHeight = CTFontGetXHeight(m_platformData.font());
223 } else
224 xHeight = verticalRightOrientationFont().fontMetrics().xHeight();
225 }
226
227 m_fontMetrics.setUnitsPerEm(unitsPerEm);
228 m_fontMetrics.setAscent(ascent);
229 m_fontMetrics.setDescent(descent);
230 m_fontMetrics.setCapHeight(capHeight);
231 m_fontMetrics.setLineGap(lineGap);
232 m_fontMetrics.setXHeight(xHeight);
233 m_fontMetrics.setLineSpacing(lineSpacing);
234 m_fontMetrics.setUnderlinePosition(-CTFontGetUnderlinePosition(m_platformData.font()));
235 m_fontMetrics.setUnderlineThickness(CTFontGetUnderlineThickness(m_platformData.font()));
236}

238void Font::platformCharWidthInit()
239{
240 m_avgCharWidth = 0;
241 m_maxCharWidth = 0;
242
243 auto os2Table = adoptCF(CTFontCopyTable(m_platformData.font(), kCTFontTableOS2, kCTFontTableOptionNoOptions));
244 if (os2Table && CFDataGetLength(os2Table.get()) >= 4) {
245 const UInt8* os2 = CFDataGetBytePtr(os2Table.get());
246 SInt16 os2AvgCharWidth = os2[2] * 256 + os2[3];
247 m_avgCharWidth = scaleEmToUnits(os2AvgCharWidth, m_fontMetrics.unitsPerEm()) * m_platformData.size();
248 }
249
250 auto headTable = adoptCF(CTFontCopyTable(m_platformData.font(), kCTFontTableHead, kCTFontTableOptionNoOptions));
251 if (headTable && CFDataGetLength(headTable.get()) >= 42) {
252 const UInt8* head = CFDataGetBytePtr(headTable.get());
253 unsigned uxMin = head[36] * 256 + head[37];
254 unsigned uxMax = head[40] * 256 + head[41];
255 SInt16 xMin = static_cast<SInt16>(uxMin);
256 SInt16 xMax = static_cast<SInt16>(uxMax);
257 float diff = static_cast<float>(xMax - xMin);
258 m_maxCharWidth = scaleEmToUnits(diff, m_fontMetrics.unitsPerEm()) * m_platformData.size();
259 }
260
261 // Fallback to a cross-platform estimate, which will populate these values if they are non-positive.
262 initCharWidths();
263}

265bool Font::variantCapsSupportsCharacterForSynthesis(FontVariantCaps fontVariantCaps, UChar32 character) const
266{
267#if (PLATFORM(IOS_FAMILY) && TARGET_OS_IOS) || PLATFORM(MAC)
268 Glyph glyph = glyphForCharacter(character);
269 if (!glyph)
270 return false;
271
272 switch (fontVariantCaps) {
273 case FontVariantCaps::Small: {
274 auto& supported = glyphsSupportedBySmallCaps();
275 return supported.size() > glyph && supported.get(glyph);
276 }
277 case FontVariantCaps::Petite: {
278 auto& supported = glyphsSupportedByPetiteCaps();
279 return supported.size() > glyph && supported.get(glyph);
280 }
281 case FontVariantCaps::AllSmall: {
282 auto& supported = glyphsSupportedByAllSmallCaps();
283 return supported.size() > glyph && supported.get(glyph);
284 }
285 case FontVariantCaps::AllPetite: {
286 auto& supported = glyphsSupportedByAllPetiteCaps();
287 return supported.size() > glyph && supported.get(glyph);
288 }
289 default:
290 // Synthesis only supports the variant-caps values listed above.
291 return true;
292 }
293#else
294 UNUSED_PARAM(character);
295
296 switch (fontVariantCaps) {
297 case FontVariantCaps::Small:
298 case FontVariantCaps::Petite:
299 case FontVariantCaps::AllSmall:
300 case FontVariantCaps::AllPetite:
301 return false;
302 default:
303 // Synthesis only supports the variant-caps values listed above.
304 return true;
305 }
306#endif
307}

327static void unionBitVectors(BitVector& result, CFBitVectorRef source)
328{
329 CFIndex length = CFBitVectorGetCount(source);
330 result.ensureSize(length);
331 CFIndex min = 0;
332 while (min < length) {
333 CFIndex nextIndex = CFBitVectorGetFirstIndexOfBit(source, CFRangeMake(min, length - min), 1);
334 if (nextIndex == kCFNotFound)
335 break;
336 result.set(nextIndex, true);
337 min = nextIndex + 1;
338 }
339}

398static RefPtr<Font> createDerivativeFont(CTFontRef font, float size, FontOrientation orientation, CTFontSymbolicTraits fontTraits, bool syntheticBold, bool syntheticItalic)
399{
400 if (!font)
401 return nullptr;
402
403 if (syntheticBold)
404 fontTraits |= kCTFontBoldTrait;
405 if (syntheticItalic)
406 fontTraits |= kCTFontItalicTrait;
407
408 CTFontSymbolicTraits scaledFontTraits = CTFontGetSymbolicTraits(font);
409
410 bool usedSyntheticBold = (fontTraits & kCTFontBoldTrait) && !(scaledFontTraits & kCTFontTraitBold);
411 bool usedSyntheticOblique = (fontTraits & kCTFontItalicTrait) && !(scaledFontTraits & kCTFontTraitItalic);
412 FontPlatformData scaledFontData(font, size, usedSyntheticBold, usedSyntheticOblique, orientation);
413
414 return Font::create(scaledFontData);
415}

417static inline bool isOpenTypeFeature(CFDictionaryRef feature)
418{
419 return CFDictionaryContainsKey(feature, kCTFontOpenTypeFeatureTag) && CFDictionaryContainsKey(feature, kCTFontOpenTypeFeatureValue);
420}
421
422static inline bool isTrueTypeFeature(CFDictionaryRef feature)
423{
424 return CFDictionaryContainsKey(feature, kCTFontFeatureTypeIdentifierKey) && CFDictionaryContainsKey(feature, kCTFontFeatureSelectorIdentifierKey);
425}

427static inline std::optional<CFStringRef> openTypeFeature(CFDictionaryRef feature)
428{
429 ASSERT(isOpenTypeFeature(feature));
430 CFStringRef tag = static_cast<CFStringRef>(CFDictionaryGetValue(feature, kCTFontOpenTypeFeatureTag));
431 int rawValue;
432 CFNumberRef value = static_cast<CFNumberRef>(CFDictionaryGetValue(feature, kCTFontOpenTypeFeatureValue));
433 auto success = CFNumberGetValue(value, kCFNumberIntType, &rawValue);
434 ASSERT_UNUSED(success, success);
435 return rawValue ? std::optional<CFStringRef>(tag) : std::nullopt;
436}
437
438static inline std::pair<int, int> trueTypeFeature(CFDictionaryRef feature)
439{
440 ASSERT(isTrueTypeFeature(feature));
441 int rawType;
442 CFNumberRef type = static_cast<CFNumberRef>(CFDictionaryGetValue(feature, kCTFontFeatureTypeIdentifierKey));
443 auto success = CFNumberGetValue(type, kCFNumberIntType, &rawType);
444 ASSERT_UNUSED(success, success);
445 int rawSelector;
446 CFNumberRef selector = static_cast<CFNumberRef>(CFDictionaryGetValue(feature, kCTFontFeatureSelectorIdentifierKey));
447 success = CFNumberGetValue(selector, kCFNumberIntType, &rawSelector);
448 ASSERT_UNUSED(success, success);
449 return std::make_pair(rawType, rawSelector);
450}

452static inline CFNumberRef defaultSelectorForTrueTypeFeature(int key, CTFontRef font)
453{
454 RetainPtr<CFArrayRef> features = adoptCF(CTFontCopyFeatures(font));
455 CFIndex featureCount = CFArrayGetCount(features.get());
456 for (CFIndex i = 0; i < featureCount; ++i) {
457 CFDictionaryRef featureType = static_cast<CFDictionaryRef>(CFArrayGetValueAtIndex(features.get(), i));
458 CFNumberRef featureKey = static_cast<CFNumberRef>(CFDictionaryGetValue(featureType, kCTFontFeatureTypeIdentifierKey));
459 if (!featureKey)
460 continue;
461 int rawFeatureKey;
462 CFNumberGetValue(featureKey, kCFNumberIntType, &rawFeatureKey);
463 if (rawFeatureKey != key)
464 continue;
465
466 CFArrayRef featureSelectors = static_cast<CFArrayRef>(CFDictionaryGetValue(featureType, kCTFontFeatureTypeSelectorsKey));
467 if (!featureSelectors)
468 continue;
469 CFIndex selectorsCount = CFArrayGetCount(featureSelectors);
470 for (CFIndex j = 0; j < selectorsCount; ++j) {
471 CFDictionaryRef featureSelector = static_cast<CFDictionaryRef>(CFArrayGetValueAtIndex(featureSelectors, j));
472 CFNumberRef isDefault = static_cast<CFNumberRef>(CFDictionaryGetValue(featureSelector, kCTFontFeatureSelectorDefaultKey));
473 if (!isDefault)
474 continue;
475 int rawIsDefault;
476 CFNumberGetValue(isDefault, kCFNumberIntType, &rawIsDefault);
477 if (!rawIsDefault)
478 continue;
479 return static_cast<CFNumberRef>(CFDictionaryGetValue(featureSelector, kCTFontFeatureSelectorIdentifierKey));
480 }
481 }
482 return nullptr;
483}

485static inline RetainPtr<CFDictionaryRef> removedFeature(CFDictionaryRef feature, CTFontRef font)
486{
487 bool isOpenType = isOpenTypeFeature(feature);
488 bool isTrueType = isTrueTypeFeature(feature);
489 if (!isOpenType && !isTrueType)
490 return feature; // We don't understand this font format.
491 RetainPtr<CFMutableDictionaryRef> result = adoptCF(CFDictionaryCreateMutable(kCFAllocatorDefault, 0, &kCFTypeDictionaryKeyCallBacks, &kCFTypeDictionaryValueCallBacks));
492 if (isOpenType) {
493 auto featureTag = openTypeFeature(feature);
494 if (featureTag && (CFEqual(featureTag.value(), CFSTR("smcp"))
495 || CFEqual(featureTag.value(), CFSTR("c2sc"))
496 || CFEqual(featureTag.value(), CFSTR("pcap"))
497 || CFEqual(featureTag.value(), CFSTR("c2pc")))) {
498 int rawZero = 0;
499 RetainPtr<CFNumberRef> zero = adoptCF(CFNumberCreate(kCFAllocatorDefault, kCFNumberIntType, &rawZero));
500 CFDictionaryAddValue(result.get(), kCTFontOpenTypeFeatureTag, featureTag.value());
501 CFDictionaryAddValue(result.get(), kCTFontOpenTypeFeatureValue, zero.get());
502 } else {
503 CFDictionaryAddValue(result.get(), kCTFontOpenTypeFeatureTag, CFDictionaryGetValue(feature, kCTFontOpenTypeFeatureTag));
504 CFDictionaryAddValue(result.get(), kCTFontOpenTypeFeatureValue, CFDictionaryGetValue(feature, kCTFontOpenTypeFeatureValue));
505 }
506 }
507 if (isTrueType) {
508 auto trueTypeFeaturePair = trueTypeFeature(feature);
509 if (trueTypeFeaturePair.first == kLowerCaseType && (trueTypeFeaturePair.second == kLowerCaseSmallCapsSelector || trueTypeFeaturePair.second == kLowerCasePetiteCapsSelector)) {
510 CFDictionaryAddValue(result.get(), kCTFontFeatureTypeIdentifierKey, CFDictionaryGetValue(feature, kCTFontFeatureTypeIdentifierKey));
511 if (CFNumberRef defaultSelector = defaultSelectorForTrueTypeFeature(kLowerCaseType, font))
512 CFDictionaryAddValue(result.get(), kCTFontFeatureSelectorIdentifierKey, defaultSelector);
513 else
514 CFDictionaryAddValue(result.get(), kCTFontFeatureSelectorIdentifierKey, CFDictionaryGetValue(feature, kCTFontFeatureSelectorIdentifierKey));
515 } else if (trueTypeFeaturePair.first == kUpperCaseType && (trueTypeFeaturePair.second == kUpperCaseSmallCapsSelector || trueTypeFeaturePair.second == kUpperCasePetiteCapsSelector)) {
516 CFDictionaryAddValue(result.get(), kCTFontFeatureTypeIdentifierKey, CFDictionaryGetValue(feature, kCTFontFeatureTypeIdentifierKey));
517 if (CFNumberRef defaultSelector = defaultSelectorForTrueTypeFeature(kUpperCaseType, font))
518 CFDictionaryAddValue(result.get(), kCTFontFeatureSelectorIdentifierKey, defaultSelector);
519 else
520 CFDictionaryAddValue(result.get(), kCTFontFeatureSelectorIdentifierKey, CFDictionaryGetValue(feature, kCTFontFeatureSelectorIdentifierKey));
521 } else {
522 CFDictionaryAddValue(result.get(), kCTFontFeatureTypeIdentifierKey, CFDictionaryGetValue(feature, kCTFontFeatureTypeIdentifierKey));
523 CFDictionaryAddValue(result.get(), kCTFontFeatureSelectorIdentifierKey, CFDictionaryGetValue(feature, kCTFontFeatureSelectorIdentifierKey));
524 }
525 }
526 return result;
527}

529static RetainPtr<CTFontRef> createCTFontWithoutSynthesizableFeatures(CTFontRef font)
530{
531 RetainPtr<CFArrayRef> features = adoptCF(static_cast<CFArrayRef>(CTFontCopyAttribute(font, kCTFontFeatureSettingsAttribute)));
532 if (!features)
533 return font;
534 CFIndex featureCount = CFArrayGetCount(features.get());
535 RetainPtr<CFMutableArrayRef> newFeatures = adoptCF(CFArrayCreateMutable(kCFAllocatorDefault, featureCount, &kCFTypeArrayCallBacks));
536 for (CFIndex i = 0; i < featureCount; ++i) {
537 CFDictionaryRef feature = static_cast<CFDictionaryRef>(CFArrayGetValueAtIndex(features.get(), i));
538 CFArrayAppendValue(newFeatures.get(), removedFeature(feature, font).get());
539 }
540 CFTypeRef keys[] = { kCTFontFeatureSettingsAttribute };
541 CFTypeRef values[] = { newFeatures.get() };
542 RetainPtr<CFDictionaryRef> attributes = adoptCF(CFDictionaryCreate(kCFAllocatorDefault, keys, values, WTF_ARRAY_LENGTH(keys), &kCFTypeDictionaryKeyCallBacks, &kCFTypeDictionaryValueCallBacks));

587void Font::applyTransforms(GlyphBuffer& glyphBuffer, unsigned beginningGlyphIndex, unsigned beginningStringIndex, bool enableKerning, bool requiresShaping, const AtomString& locale, StringView text, TextDirection textDirection) const
588{
589 UNUSED_PARAM(requiresShaping);
590
591 // FIXME: Implement GlyphBuffer initial advance.
592#if USE(CTFONTSHAPEGLYPHS)
593 auto handler = ^(CFRange range, CGGlyph** newGlyphsPointer, CGSize** newAdvancesPointer, CGPoint** newOffsetsPointer, CFIndex** newIndicesPointer)
594 {
595 range.location = std::min(std::max(range.location, static_cast<CFIndex>(0)), static_cast<CFIndex>(glyphBuffer.size()));
596 if (range.length < 0) {
597 range.length = std::min(range.location, -range.length);
598 range.location = range.location - range.length;
599 glyphBuffer.remove(beginningGlyphIndex + range.location, range.length);
600 } else
601 glyphBuffer.makeHole(beginningGlyphIndex + range.location, range.length, this);
602
603 *newGlyphsPointer = glyphBuffer.glyphs(beginningGlyphIndex);
604 *newAdvancesPointer = glyphBuffer.advances(beginningGlyphIndex);
605 *newOffsetsPointer = glyphBuffer.origins(beginningGlyphIndex);
606 *newIndicesPointer = glyphBuffer.offsetsInString(beginningGlyphIndex);
607 };
608
609 auto substring = text.substring(beginningStringIndex);
610 auto upconvertedCharacters = substring.upconvertedCharacters();
611 auto localeString = LocaleCocoa::canonicalLanguageIdentifierFromString(locale).string().createCFString();
612 CTFontShapeOptions options = kCTFontShapeWithClusterComposition
613 | (enableKerning ? kCTFontShapeWithKerning : 0)
614 | (textDirection == TextDirection::RTL ? kCTFontShapeRightToLeft : 0);
615
616 for (unsigned i = 0; i < glyphBuffer.size() - beginningGlyphIndex; ++i)
617 glyphBuffer.offsetsInString(beginningGlyphIndex)[i] -= beginningStringIndex;
618
619 CTFontShapeGlyphs(
620 m_platformData.ctFont(),
621 glyphBuffer.glyphs(beginningGlyphIndex),
622 reinterpret_cast<CGSize*>(glyphBuffer.advances(beginningGlyphIndex)),
623 reinterpret_cast<CGPoint*>(glyphBuffer.origins(beginningGlyphIndex)),
624 glyphBuffer.offsetsInString(beginningGlyphIndex),
625 reinterpret_cast<const UniChar*>(upconvertedCharacters.get()),
626 glyphBuffer.size() - beginningGlyphIndex,
627 options,
628 localeString.get(),
629 handler);
630
631 for (unsigned i = 0; i < glyphBuffer.size() - beginningGlyphIndex; ++i)
632 glyphBuffer.offsetsInString(beginningGlyphIndex)[i] += beginningStringIndex;
633
634#else
635
636 UNUSED_PARAM(beginningStringIndex);
637 UNUSED_PARAM(locale);
638 UNUSED_PARAM(text);
639
640 // CTFontTransformGlyphs() operates in visual order, but WidthIterator iterates in logical order.
641 // Temporarily put us in visual order just for the call, then put us back into logical order when
642 // the call is done.
643 // We don't have a global view of the entire GlyphBuffer; we're just operating on a single chunk of it.
644 // WidthIterator encounters the chunks out in logical order, so we have to maintain that invariant.
645 // Eventually, FontCascade::layoutSimpleText() will reverse the whole buffer to put the entire thing
646 // in visual order, but that's okay because it has a view of the entire GlyphBuffer.
647 // On the other hand, CTFontShapeGlyphs() accepts the buffer in logical order but returns it in physical
648 // order, which means the second reverse() in this function still needs to execute when
649 // CTFontShapeGlyphs() is being used.
650 if (textDirection == TextDirection::RTL)
651 glyphBuffer.reverse(beginningGlyphIndex, glyphBuffer.size() - beginningGlyphIndex);
652
653 CTFontTransformOptions options = (enableKerning ? kCTFontTransformApplyPositioning : 0) | kCTFontTransformApplyShaping;
654 CTFontTransformGlyphs(m_platformData.ctFont(), glyphBuffer.glyphs(beginningGlyphIndex), reinterpret_cast<CGSize*>(glyphBuffer.advances(beginningGlyphIndex)), glyphBuffer.size() - beginningGlyphIndex, options);
655#endif
656
657 // See the comment above in this function where the other call to reverse() is.
658 if (textDirection == TextDirection::RTL)
659 glyphBuffer.reverse(beginningGlyphIndex, glyphBuffer.size() - beginningGlyphIndex);
660}

670void Font::determinePitch()
671{
672 CTFontRef ctFont = m_platformData.ctFont();
673 ASSERT(ctFont);
674
675 // Special case Osaka-Mono.
676 // According to <rdar://problem/3999467>, we should treat Osaka-Mono as fixed pitch.
677 // Note that the AppKit does not report Osaka-Mono as fixed pitch.
678
679 // Special case MS-PGothic.
680 // According to <rdar://problem/4032938>, we should not treat MS-PGothic as fixed pitch.
681 // Note that AppKit does report MS-PGothic as fixed pitch.
682
683 // Special case MonotypeCorsiva
684 // According to <rdar://problem/5454704>, we should not treat MonotypeCorsiva as fixed pitch.
685 // Note that AppKit does report MonotypeCorsiva as fixed pitch.
686
687 auto fullName = adoptCF(CTFontCopyFullName(ctFont));
688 auto familyName = adoptCF(CTFontCopyFamilyName(ctFont));
689
690 int fixedPitch = extractNumber(adoptCF(static_cast<CFNumberRef>(CTFontCopyAttribute(m_platformData.ctFont(), kCTFontFixedAdvanceAttribute))).get());
691 m_treatAsFixedPitch = (CTFontGetSymbolicTraits(ctFont) & kCTFontMonoSpaceTrait) || fixedPitch || (caseInsensitiveCompare(fullName.get(), CFSTR("Osaka-Mono")) || caseInsensitiveCompare(fullName.get(), CFSTR("MS-PGothic")) || caseInsensitiveCompare(fullName.get(), CFSTR("MonotypeCorsiva")));
692#if PLATFORM(IOS_FAMILY)
693 if (familyName && caseInsensitiveCompare(familyName.get(), CFSTR("Courier New"))) {
694 // Special case Courier New to not be treated as fixed pitch, as this will make use of a hacked space width which is undesireable for iPhone (see rdar://6269783).
695 m_treatAsFixedPitch = false;
696 }
697#endif
698}

738bool Font::isProbablyOnlyUsedToRenderIcons() const
739{
740 auto platformFont = platformData().ctFont();
741 if (!platformFont)
742 return false;
743
744 // Allow most non-icon fonts to bail early here by testing a single character 'a', without iterating over all basic latin characters.
745 UniChar lowercaseACharacter = 'a';
746 CGGlyph lowercaseAGlyph;
747 if (CTFontGetGlyphsForCharacters(platformFont, &lowercaseACharacter, &lowercaseAGlyph, 1)) {
748 CGRect ignoredRect = { };
749 if (!CGRectIsEmpty(CTFontGetBoundingRectsForGlyphs(platformFont, kCTFontOrientationDefault, &lowercaseAGlyph, &ignoredRect, 1)))
750 return false;
751 }
752
753 auto supportedCharacters = adoptCF(CTFontCopyCharacterSet(platformFont));
754 if (CFCharacterSetHasMemberInPlane(supportedCharacters.get(), 1) || CFCharacterSetHasMemberInPlane(supportedCharacters.get(), 2))
755 return false;
756
757 // This encompasses all basic Latin non-control characters.
758 constexpr UniChar firstCharacterToTest = ' ';
759 constexpr UniChar lastCharacterToTest = '~';
760 constexpr auto numberOfCharactersToTest = lastCharacterToTest - firstCharacterToTest + 1;
761
762 Vector<CGGlyph> glyphs;
763 glyphs.fill(0, numberOfCharactersToTest);
764 CTFontGetGlyphsForCharacterRange(platformFont, glyphs.begin(), CFRangeMake(firstCharacterToTest, numberOfCharactersToTest));
765 glyphs.removeAll(0);
766
767 if (glyphs.isEmpty())
768 return false;
769
770 Vector<CGRect> boundingRects;
771 boundingRects.fill(CGRectZero, glyphs.size());
772 CTFontGetBoundingRectsForGlyphs(platformFont, kCTFontOrientationDefault, glyphs.begin(), boundingRects.begin(), glyphs.size());
773 return notFound == boundingRects.findMatching([](auto& rect) {
774 return !CGRectIsEmpty(rect);
775 });
776}
