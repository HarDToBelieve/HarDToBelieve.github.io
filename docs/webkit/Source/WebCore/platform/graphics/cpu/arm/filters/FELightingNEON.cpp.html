157#define GET_LENGTH(source, temp) \
158 "vmul.f32 " temp##_Q ", " source##_Q ", " source##_Q NL \
159 "vadd.f32 " source##_S3 ", " temp##_S0 ", " temp##_S1 NL \
160 "vadd.f32 " source##_S3 ", " source##_S3 ", " temp##_S2 NL \
161 "vsqrt.f32 " source##_S3 ", " source##_S3 NL
162
163// destination##_S3 can contain the multiply of length.
164#define DOT_PRODUCT(destination, source1, source2) \
165 "vmul.f32 " destination##_Q ", " source1##_Q ", " source2##_Q NL \
166 "vadd.f32 " destination##_S0 ", " destination##_S0 ", " destination##_S1 NL \
167 "vadd.f32 " destination##_S0 ", " destination##_S0 ", " destination##_S2 NL

169#define MULTIPLY_BY_DIFFUSE_CONST(normalVectorLength, dotProductLength) \
170 "tst " FLAGS_R ", #" TOSTRING(FLAG_DIFFUSE_CONST_IS_1) NL \
171 "vmuleq.f32 " TMP2_S1 ", " DIFFUSE_CONST_S ", " normalVectorLength NL \
172 "vdiveq.f32 " TMP2_S1 ", " TMP2_S1 ", " dotProductLength NL \
173 "vdivne.f32 " TMP2_S1 ", " normalVectorLength ", " dotProductLength NL

175#define POWF_SQR(value, exponent, current, remaining) \
176 "tst " exponent ", #" ASSTRING(current) NL \
177 "vmulne.f32 " value ", " value ", " POWF_INT_S NL \
178 "tst " exponent ", #" ASSTRING(remaining) NL \
179 "vmulne.f32 " POWF_INT_S ", " POWF_INT_S ", " POWF_INT_S NL

181#define POWF_SQRT(value, exponent, current, remaining) \
182 "tst " exponent ", #" ASSTRING(remaining) NL \
183 "vsqrtne.f32 " POWF_FRAC_S ", " POWF_FRAC_S NL \
184 "tst " exponent ", #" ASSTRING(current) NL \
185 "vmulne.f32 " value ", " value ", " POWF_FRAC_S NL

187// This simplified powf function is sufficiently accurate.
188#define POWF(value, exponent) \
189 "tst " exponent ", #0xfc0" NL \
190 "vmovne.f32 " POWF_INT_S ", " value NL \
191 "tst " exponent ", #0x03f" NL \
192 "vmovne.f32 " POWF_FRAC_S ", " value NL \
193 "vmov.f32 " value ", " CONST_ONE_S NL \
194 \
195 POWF_SQR(value, exponent, 0x040, 0xf80) \
196 POWF_SQR(value, exponent, 0x080, 0xf00) \
197 POWF_SQR(value, exponent, 0x100, 0xe00) \
198 POWF_SQR(value, exponent, 0x200, 0xc00) \
199 POWF_SQR(value, exponent, 0x400, 0x800) \
200 "tst " exponent ", #0x800" NL \
201 "vmulne.f32 " value ", " value ", " POWF_INT_S NL \
202 \
203 POWF_SQRT(value, exponent, 0x20, 0x3f) \
204 POWF_SQRT(value, exponent, 0x10, 0x1f) \
205 POWF_SQRT(value, exponent, 0x08, 0x0f) \
206 POWF_SQRT(value, exponent, 0x04, 0x07) \
207 POWF_SQRT(value, exponent, 0x02, 0x03) \
208 POWF_SQRT(value, exponent, 0x01, 0x01)

210// The following algorithm is an ARM-NEON optimized version of
211// the main loop found in FELighting.cpp. Since the whole code
212// is redesigned to be as effective as possible (ARM specific
213// thinking), it is four times faster than its C++ counterpart.
214
215asm ( // NOLINT
216".globl " TOSTRING(neonDrawLighting) NL
217TOSTRING(neonDrawLighting) ":" NL
218 // Because of the clever register allocation, nothing is stored on the stack
219 // except the saved registers.
220 // Stack must be aligned to 8 bytes.
221 "stmdb sp!, {r4-r8, r10, r11, lr}" NL
222 "vstmdb sp!, {d8-d15}" NL
223 "mov " PAINTING_DATA_R ", r0" NL
224
225 // The following two arguments are loaded to SIMD registers.
226 "ldr r0, [" PAINTING_DATA_R ", #" FLOAT_ARGUMENTS_OFFSET "]" NL
227 "ldr r1, [" PAINTING_DATA_R ", #" PAINTING_CONSTANTS_OFFSET "]" NL
228 "ldr " PIXELS_R ", [" PAINTING_DATA_R ", #" PIXELS_OFFSET "]" NL
229 "vldr.f32 " POSITION_Y_S ", [" PAINTING_DATA_R ", #" YSTART_OFFSET "]" NL
230 "ldr " WIDTH_R ", [" PAINTING_DATA_R ", #" WIDTH_OFFSET "]" NL
231 "ldr " HEIGHT_R ", [" PAINTING_DATA_R ", #" HEIGHT_OFFSET "]" NL
232 "ldr " FLAGS_R ", [" PAINTING_DATA_R ", #" FLAGS_OFFSET "]" NL
233 "ldr " SPECULAR_EXPONENT_R ", [" PAINTING_DATA_R ", #" SPECULAR_EXPONENT_OFFSET "]" NL
234 "ldr " CONE_EXPONENT_R ", [" PAINTING_DATA_R ", #" CONE_EXPONENT_OFFSET "]" NL
235
236 // Load all data to the SIMD registers with the least number of instructions.
237 "vld1.f32 { " READ1_RANGE " }, [r0]!" NL
238 "vld1.f32 { " READ2_RANGE " }, [r0]!" NL
239 "vld1.f32 { " READ3_RANGE " }, [r0]!" NL
240 "vld1.s16 {" READ4_RANGE "}, [r1]!" NL
241 "vld1.s16 {" READ5_RANGE "}, [r1]!" NL
242
243 // Initializing local variables.
244 "mov " SCANLINE_R ", " WIDTH_R ", lsl #2" NL
245 "add " SCANLINE_R ", " SCANLINE_R ", #8" NL
246 "add " PIXELS_R ", " PIXELS_R ", " SCANLINE_R NL
247 "add " PIXELS_R ", " PIXELS_R ", #3" NL
248 "mov r0, #0" NL
249 "vmov.f32 " CONST_ZERO_S ", r0" NL
250 "tst " FLAGS_R ", #" TOSTRING(FLAG_SPOT_LIGHT) NL
251 "vmov.f32 " SPOT_COLOR_Q ", " COLOR_Q NL
252 "mov " RESET_WIDTH_R ", " WIDTH_R NL
253
254".mainLoop:" NL
255 "mov r3, #3" NL
256 "vmov.f32 " POSITION_X_S ", " CONST_ONE_S NL
257
258".scanline:" NL
259 // The ROW registers are storing the alpha channel of the last three pixels.
260 // The alpha channel is stored as signed short (sint16) values. The fourth value
261 // is garbage. The following instructions are shifting out the unnecessary alpha
262 // values and load the next ones.
263 "ldrb r0, [" PIXELS_R ", -" SCANLINE_R "]" NL
264 "ldrb r1, [" PIXELS_R ", +" SCANLINE_R "]" NL
265 "ldrb r2, [" PIXELS_R "], #4" NL
266 "vext.s16 " TOP_ROW_D ", " TOP_ROW_D ", " TOP_ROW_D ", #3" NL
267 "vext.s16 " MIDDLE_ROW_D ", " MIDDLE_ROW_D ", " MIDDLE_ROW_D ", #3" NL
268 "vext.s16 " BOTTOM_ROW_D ", " BOTTOM_ROW_D ", " BOTTOM_ROW_D ", #3" NL
269 "vmov.s16 " TOP_ROW_D "[1], r0" NL
270 "vmov.s16 " MIDDLE_ROW_D "[1], r2" NL
271 "vmov.s16 " BOTTOM_ROW_D "[1], r1" NL
272
273 // The two border pixels (rightmost and leftmost) are skipped when
274 // the next scanline is reached. It also jumps, when the algorithm
275 // is started, and the first free alpha values are loaded to each row.
276 "subs r3, r3, #1" NL
277 "bne .scanline" NL
278
279 // The light vector goes to TMP1_Q. It is constant in case of distant light.
280 // The fourth value contains the length of the light vector.
281 "tst " FLAGS_R ", #" TOSTRING(FLAG_POINT_LIGHT | FLAG_SPOT_LIGHT) NL
282 "beq .distantLight" NL
283
284 "vmov.s16 r3, " MIDDLE_ROW_D "[2]" NL
285 "vmov.f32 " POSITION_Z_S ", r3" NL
286 "vcvt.f32.s32 " POSITION_Z_S ", " POSITION_Z_S NL
287 "vmul.f32 " POSITION_Z_S ", " POSITION_Z_S ", " SCALE_S NL
288
289 "vsub.f32 " TMP1_Q ", " LIGHT_Q ", " POSITION_Q NL
290 GET_LENGTH(TMP1, TMP2)
291
292 "tst " FLAGS_R ", #" TOSTRING(FLAG_SPOT_LIGHT) NL
293 "bne .cosineOfAngle" NL
294".visiblePixel:" NL
295
296 // | -1 0 1 | | -1 -2 -1 |
297 // X = | -2 0 2 | Y = | 0 0 0 |
298 // | -1 0 1 | | 1 2 1 |
299
300 // Multiply the alpha values by the X and Y matrices.
301
302 // Moving the 8 alpha value to TMP3.
303 "vtbl.8 " TMP3_D0 ", " ALL_ROWS_D ", " REMAPX_D NL
304 "vtbl.8 " TMP3_D1 ", " ALL_ROWS_D ", " REMAPY_D NL
305
306 "vmul.s16 " TMP2_Q ", " TMP3_Q ", " ALPHAX_Q NL
307 "vpadd.s16 " TMP2_D0 ", " TMP2_D0 ", " TMP2_D1 NL
308 "vpadd.s16 " TMP2_D0 ", " TMP2_D0 ", " TMP2_D0 NL
309 "vpadd.s16 " TMP2_D0 ", " TMP2_D0 ", " TMP2_D0 NL
310 "vmov.s16 r0, " TMP2_D0 "[0]" NL
311
312 "vmul.s16 " TMP2_Q ", " TMP3_Q ", " ALPHAY_Q NL
313 "vpadd.s16 " TMP2_D0 ", " TMP2_D0 ", " TMP2_D1 NL
314 "vpadd.s16 " TMP2_D0 ", " TMP2_D0 ", " TMP2_D0 NL
315 "vpadd.s16 " TMP2_D0 ", " TMP2_D0 ", " TMP2_D0 NL
316 "vmov.s16 r1, " TMP2_D0 "[0]" NL
317
318 // r0 and r1 contains the X and Y coordinates of the
319 // normal vector, respectively.
320
321 // Calculating the spot light strength.
322 "tst " FLAGS_R ", #" TOSTRING(FLAG_SPOT_LIGHT) NL
323 "beq .endLight" NL
324
325 "vneg.f32 " TMP3_S1 ", " COSINE_OF_ANGLE NL
326 "tst " FLAGS_R ", #" TOSTRING(FLAG_CONE_EXPONENT_IS_1) NL
327 "beq .coneExpPowf" NL
328".coneExpPowfFinished:" NL
329
330 // Smoothing the cone edge if necessary.
331 "vcmp.f32 " COSINE_OF_ANGLE ", " CONE_FULL_LIGHT_S NL
332 "fmstat" NL
333 "bhi .cutOff" NL
334".cutOffFinished:" NL
335
336 "vmin.f32 " TMP3_D0 ", " TMP3_D0 ", " CONST_ONE_HI_D NL
337 "vmul.f32 " COLOR_Q ", " SPOT_COLOR_Q ", " TMP3_D0 "[1]" NL
338
339".endLight:" NL
340 // Summarize:
341 // r0 and r1 contains the normalVector.
342 // TMP1_Q contains the light vector and its length.
343 // COLOR_Q contains the color of the light vector.
344
345 // Test whether both r0 and r1 are zero (Normal vector is (0, 0, 1)).
346 "orrs r2, r0, r1" NL
347 "bne .normalVectorIsNonZero" NL
348
349 "tst " FLAGS_R ", #" TOSTRING(FLAG_SPECULAR_LIGHT) NL
350 "bne .specularLight1" NL
351
352 // Calculate diffuse light strength.
353 MULTIPLY_BY_DIFFUSE_CONST(TMP1_S2, TMP1_S3)
354 "b .lightStrengthCalculated" NL
355
356".specularLight1:" NL
357 // Calculating specular light strength.
358 "vadd.f32 " TMP1_S2 ", " TMP1_S2 ", " TMP1_S3 NL
359 GET_LENGTH(TMP1, TMP2)
360
361 // When the exponent is 1, we don't need to call an expensive powf function.
362 "tst " FLAGS_R ", #" TOSTRING(FLAG_SPECULAR_EXPONENT_IS_1) NL
363 "vdiveq.f32 " TMP2_S1 ", " TMP1_S2 ", " TMP1_S3 NL
364 "beq .specularExpPowf" NL
365
366 MULTIPLY_BY_DIFFUSE_CONST(TMP1_S2, TMP1_S3)
367 "b .lightStrengthCalculated" NL
368
369".normalVectorIsNonZero:" NL
370 // Normal vector goes to TMP2, and its length is calculated as well.
371 "vmov.s32 " TMP2_S0 ", r0" NL
372 "vcvt.f32.s32 " TMP2_S0 ", " TMP2_S0 NL
373 "vmul.f32 " TMP2_S0 ", " TMP2_S0 ", " SCALE_DIV4_S NL
374 "vmov.s32 " TMP2_S1 ", r1" NL
375 "vcvt.f32.s32 " TMP2_S1 ", " TMP2_S1 NL
376 "vmul.f32 " TMP2_S1 ", " TMP2_S1 ", " SCALE_DIV4_S NL
377 "vmov.f32 " TMP2_S2 ", " CONST_ONE_S NL
378 GET_LENGTH(TMP2, TMP3)
379
380 "tst " FLAGS_R ", #" TOSTRING(FLAG_SPECULAR_LIGHT) NL
381 "bne .specularLight2" NL
382
383 // Calculating diffuse light strength.
384 DOT_PRODUCT(TMP3, TMP2, TMP1)
385 MULTIPLY_BY_DIFFUSE_CONST(TMP3_S0, TMP3_S3)
386 "b .lightStrengthCalculated" NL
387
388".specularLight2:" NL
389 // Calculating specular light strength.
390 "vadd.f32 " TMP1_S2 ", " TMP1_S2 ", " TMP1_S3 NL
391 GET_LENGTH(TMP1, TMP3)
392 DOT_PRODUCT(TMP3, TMP2, TMP1)
393
394 // When the exponent is 1, we don't need to call an expensive powf function.
395 "tst " FLAGS_R ", #" TOSTRING(FLAG_SPECULAR_EXPONENT_IS_1) NL
396 "vdiveq.f32 " TMP2_S1 ", " TMP3_S0 ", " TMP3_S3 NL
397 "beq .specularExpPowf" NL
398 MULTIPLY_BY_DIFFUSE_CONST(TMP3_S0, TMP3_S3)
399
400".lightStrengthCalculated:" NL
401 // TMP2_S1 contains the light strength. Clamp it to [0, 1]
402 "vmax.f32 " TMP2_D0 ", " TMP2_D0 ", " CONST_ZERO_HI_D NL
403 "vmin.f32 " TMP2_D0 ", " TMP2_D0 ", " CONST_ONE_HI_D NL
404 "vmul.f32 " TMP3_Q ", " COLOR_Q ", " TMP2_D0 "[1]" NL
405 "vcvt.u32.f32 " TMP3_Q ", " TMP3_Q NL
406 "vmov.u32 r2, r3, " TMP3_S0 ", " TMP3_S1 NL
407 // The color values are stored in-place.
408 "strb r2, [" PIXELS_R ", #-11]" NL
409 "strb r3, [" PIXELS_R ", #-10]" NL
410 "vmov.u32 r2, " TMP3_S2 NL
411 "strb r2, [" PIXELS_R ", #-9]" NL
412
413 // Continue to the next pixel.
414".blackPixel:" NL
415 "vadd.f32 " POSITION_X_S ", " CONST_ONE_S NL
416 "mov r3, #1" NL
417 "subs " WIDTH_R ", " WIDTH_R ", #1" NL
418 "bne .scanline" NL
419
420 // If the end of the scanline is reached, we continue
421 // to the next scanline.
422 "vadd.f32 " POSITION_Y_S ", " CONST_ONE_S NL
423 "mov " WIDTH_R ", " RESET_WIDTH_R NL
424 "subs " HEIGHT_R ", " HEIGHT_R ", #1" NL
425 "bne .mainLoop" NL
426
427 // Return.
428 "vldmia sp!, {d8-d15}" NL
429 "ldmia sp!, {r4-r8, r10, r11, pc}" NL
430
431".distantLight:" NL
432 // In case of distant light, the light vector is constant,
433 // we simply copy it.
434 "vmov.f32 " TMP1_Q ", " LIGHT_Q NL
435 "b .visiblePixel" NL
436
437".cosineOfAngle:" NL
438 // If the pixel is outside of the cone angle, it is simply a black pixel.
439 DOT_PRODUCT(TMP3, TMP1, DIRECTION)
440 "vdiv.f32 " COSINE_OF_ANGLE ", " TMP3_S0 ", " TMP1_S3 NL
441 "vcmp.f32 " COSINE_OF_ANGLE ", " CONE_CUT_OFF_S NL
442 "fmstat" NL
443 "bls .visiblePixel" NL
444 "mov r0, #0" NL
445 "strh r0, [" PIXELS_R ", #-11]" NL
446 "strb r0, [" PIXELS_R ", #-9]" NL
447 "b .blackPixel" NL
448
449".cutOff:" NL
450 // Smoothing the light strength on the cone edge.
451 "vsub.f32 " TMP3_S0 ", " CONE_CUT_OFF_S ", " COSINE_OF_ANGLE NL
452 "vdiv.f32 " TMP3_S0 ", " TMP3_S0 ", " CONE_CUT_OFF_RANGE_S NL
453 "vmul.f32 " TMP3_S1 ", " TMP3_S1 ", " TMP3_S0 NL
454 "b .cutOffFinished" NL
455
456".coneExpPowf:" NL
457 POWF(TMP3_S1, CONE_EXPONENT_R)
458 "b .coneExpPowfFinished" NL
459
460".specularExpPowf:" NL
461 POWF(TMP2_S1, SPECULAR_EXPONENT_R)
462 "tst " FLAGS_R ", #" TOSTRING(FLAG_DIFFUSE_CONST_IS_1) NL
463 "vmuleq.f32 " TMP2_S1 ", " TMP2_S1 ", " DIFFUSE_CONST_S NL
464 "b .lightStrengthCalculated" NL
465); // NOLINT

467int FELighting::getPowerCoefficients(float exponent)
468{
469 // Calling a powf function from the assembly code would require to save
470 // and reload a lot of NEON registers. Since the base is in range [0..1]
471 // and only 8 bit precision is required, we use our own powf function.
472 // This is probably not the best, but it uses only a few registers and
473 // gives us enough precision (modifying the exponent field directly would
474 // also be possible).
475
476 // First, we limit the exponent to maximum of 64, which gives us enough
477 // precision. We split the exponent to an integer and fraction part,
478 // since a^x = (a^y)*(a^z) where x = y+z. The integer exponent of the
479 // power is estimated by square, and the fraction exponent of the power
480 // is estimated by square root assembly instructions.
481 int i, result;
482
483 if (exponent < 0)
484 exponent = 1 / (-exponent);
485
486 if (exponent > 63.99)
487 exponent = 63.99;
488
489 exponent /= 64;
490 result = 0;
491 for (i = 11; i >= 0; --i) {
492 exponent *= 2;
493 if (exponent >= 1) {
494 result |= 1 << i;
495 exponent -= 1;
496 }
497 }
498 return result;
499}
