31#import "GraphicsContextGLOpenGL.h"

33#import <wtf/UnsafePointer.h>
34
35namespace WebCore {
36
37// GraphicsContextGLCV implementation for ANGLE flavour of GraphicsContextGLOpenGL.
38class GraphicsContextGLCVANGLE final : public GraphicsContextGLCV {
39 WTF_MAKE_FAST_ALLOCATED;
40public:
41 GraphicsContextGLCVANGLE(GraphicsContextGLOpenGL&);
42 ~GraphicsContextGLCVANGLE() final;
43
44 bool copyPixelBufferToTexture(CVPixelBufferRef, PlatformGLObject outputTexture, GCGLint level, GCGLenum internalFormat, GCGLenum format, GCGLenum type, FlipY) final;
45
46private:
47 bool initializeUVContextObjects();
48
49 unsigned lastTextureSeed(GCGLuint texture)
50 {
51 return m_lastTextureSeed.get(texture);
52 }
53
54 // Returns a handle which, if non-null, must be released via the
55 // detach call below.
56 void* attachIOSurfaceToTexture(GCGLenum target, GCGLenum internalFormat, GCGLsizei width, GCGLsizei height, GCGLenum type, IOSurfaceRef, GCGLuint plane);
57 void detachIOSurfaceFromTexture(void* handle);
58
59 Ref<GraphicsContextGLOpenGL> m_context;
60
61 PlatformGLObject m_framebuffer { 0 };
62 PlatformGLObject m_yuvProgram { 0 };
63 PlatformGLObject m_yuvVertexBuffer { 0 };
64 GCGLint m_yTextureUniformLocation { -1 };
65 GCGLint m_uvTextureUniformLocation { -1 };
66 GCGLint m_yuvFlipYUniformLocation { -1 };
67 GCGLint m_colorMatrixUniformLocation { -1 };
68 GCGLint m_yuvPositionAttributeLocation { -1 };
69 GCGLint m_yTextureSizeUniformLocation { -1 };
70 GCGLint m_uvTextureSizeUniformLocation { -1 };
71
72 FlipY m_lastFlipY { FlipY::No };
73 UnsafePointer<IOSurfaceRef> m_lastSurface;
74 uint32_t m_lastSurfaceSeed { 0 };
75
76 using TextureSeedMap = HashMap<GCGLuint, unsigned, WTF::IntHash<GCGLuint>, WTF::UnsignedWithZeroKeyHashTraits<GCGLuint>>;
77 TextureSeedMap m_lastTextureSeed;
78};
79
80}
81
82#endif
