36DrawingContext::DrawingContext(const FloatSize& logicalSize, const AffineTransform& initialCTM, Recorder::Delegate* delegate)
37 : m_context(m_displayList, GraphicsContextState(), FloatRect({ }, logicalSize), initialCTM, delegate)
38{
39}
40
41void DrawingContext::setTracksDisplayListReplay(bool tracksDisplayListReplay)
42{
43 m_tracksDisplayListReplay = tracksDisplayListReplay;
44 m_replayedDisplayList.reset();
45}
46
47void DrawingContext::replayDisplayList(GraphicsContext& destContext)
48{
49 if (m_displayList.isEmpty())
50 return;
51
52 Replayer replayer(destContext, m_displayList);
53 if (m_tracksDisplayListReplay)
54 m_replayedDisplayList = replayer.replay({ }, m_tracksDisplayListReplay).trackedDisplayList;
55 else
56 replayer.replay();
57 m_displayList.clear();
58}
