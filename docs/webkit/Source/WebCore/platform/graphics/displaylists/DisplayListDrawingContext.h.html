32namespace WebCore {
33namespace DisplayList {
34
35class InMemoryDisplayList;
36
37class DrawingContext {
38 WTF_MAKE_FAST_ALLOCATED;
39public:
40 WEBCORE_EXPORT DrawingContext(const FloatSize& logicalSize, const AffineTransform& initialCTM = { }, Recorder::Delegate* = nullptr);
41
42 GraphicsContext& context() const { return const_cast<DrawingContext&>(*this).m_context; }
43 Recorder& recorder() { return m_context; };
44 DisplayList takeDisplayList() { return std::exchange(m_displayList, { }); }
45 DisplayList& displayList() { return m_displayList; }
46 const DisplayList& displayList() const { return m_displayList; }
47 const InMemoryDisplayList* replayedDisplayList() const { return m_replayedDisplayList.get(); }
48
49 WEBCORE_EXPORT void setTracksDisplayListReplay(bool);
50 WEBCORE_EXPORT void replayDisplayList(GraphicsContext&);
51
52protected:
53 Recorder m_context;
54 DisplayList m_displayList;
55 std::unique_ptr<InMemoryDisplayList> m_replayedDisplayList;
56 bool m_tracksDisplayListReplay { false };
57};
