471template<typename, typename = void> inline constexpr bool HasIsValid = false;
472template<typename T> inline constexpr bool HasIsValid<T, std::void_t<decltype(std::declval<T>().isValid())>> = true;
473
474template<typename Item>
475static inline typename std::enable_if_t<!HasIsValid<Item>, bool> isValid(const Item&)
476{
477 return true;
478}
479
480template<typename Item>
481static inline typename std::enable_if_t<HasIsValid<Item>, bool> isValid(const Item& item)
482{
483 return item.isValid();
484}

486template<typename Item>
487static inline bool copyInto(uint8_t* destinationWithOffset, const Item& item)
488{
489 auto* newItem = new (destinationWithOffset) Item(item);
490 return isValid(*newItem);
491}

493template<typename Item>
494static inline bool copyInto(uint8_t* destinationWithOffset, const ItemHandle& itemHandle)
495{
496 return copyInto(destinationWithOffset, itemHandle.get<Item>());

651bool safeCopy(ItemHandle destination, const DisplayListItem& source)
652{
653 return WTF::visit([&](const auto& source) {
654 using DisplayListItemType = typename WTF::RemoveCVAndReference<decltype(source)>::type;
655 constexpr auto itemType = DisplayListItemType::itemType;
656 destination.data[0] = static_cast<uint8_t>(itemType);
657 auto itemOffset = destination.data + sizeof(uint64_t);
658 return copyInto<DisplayListItemType>(itemOffset, source);
659 }, source);
660}

734DidChangeItemBuffer ItemBuffer::swapWritableBufferIfNeeded(size_t numberOfBytes)
735{
736 auto sizeForBufferSwitchItem = paddedSizeOfTypeAndItemInBytes(ItemType::MetaCommandChangeItemBuffer);
737 if (m_writtenNumberOfBytes + numberOfBytes + sizeForBufferSwitchItem <= m_writableBuffer.capacity)
738 return DidChangeItemBuffer::No;
739
740 auto nextBuffer = createItemBuffer(numberOfBytes + sizeForBufferSwitchItem);
741 bool hadPreviousBuffer = m_writableBuffer && m_writableBuffer.identifier != nextBuffer.identifier;
742 if (hadPreviousBuffer) {
743 uncheckedAppend<MetaCommandChangeItemBuffer>(DidChangeItemBuffer::No, nextBuffer.identifier);
744 m_writableBuffer.capacity = m_writtenNumberOfBytes;
745 m_readOnlyBuffers.append(m_writableBuffer);
746 }
747 m_writtenNumberOfBytes = 0;
748 m_writableBuffer = WTFMove(nextBuffer);
749 return hadPreviousBuffer ? DidChangeItemBuffer::Yes : DidChangeItemBuffer::No;
750}

752void ItemBuffer::append(const DisplayListItem& temporaryItem)
753{
754 auto requiredSizeForItem = m_writingClient->requiredSizeForItem(temporaryItem);
755 RefPtr<SharedBuffer> outOfLineItem;
756 if (!requiredSizeForItem) {
757 outOfLineItem = m_writingClient->encodeItemOutOfLine(temporaryItem);
758 if (!outOfLineItem)
759 return;
760 }
761
762 auto dataLength = valueOrCompute(requiredSizeForItem, [&] {
763 ASSERT(outOfLineItem);
764 return outOfLineItem->size();
765 });
766 auto additionalCapacityForEncodedItem = 2 * sizeof(uint64_t) + roundUpToMultipleOf(alignof(uint64_t), dataLength);
767
768 auto bufferChanged = swapWritableBufferIfNeeded(additionalCapacityForEncodedItem);
769
770 m_writableBuffer.data[m_writtenNumberOfBytes] = static_cast<uint8_t>(displayListItemType(temporaryItem));
771 reinterpret_cast<uint64_t*>(m_writableBuffer.data + m_writtenNumberOfBytes)[1] = dataLength;
772 auto* location = m_writableBuffer.data + m_writtenNumberOfBytes + 2 * sizeof(uint64_t);
773
774 if (requiredSizeForItem)
775 m_writingClient->encodeItemInline(temporaryItem, location);
776 else
777 outOfLineItem->copyTo(location, dataLength);
778
779 didAppendData(additionalCapacityForEncodedItem, bufferChanged);
780}

782void ItemBuffer::didAppendData(size_t numberOfBytes, DidChangeItemBuffer didChangeItemBuffer)
783{
784 m_writtenNumberOfBytes += numberOfBytes;
785 if (m_writingClient)
786 m_writingClient->didAppendData(m_writableBuffer, numberOfBytes, didChangeItemBuffer);
787}
