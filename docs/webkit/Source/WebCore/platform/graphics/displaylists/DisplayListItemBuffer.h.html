43// An ItemBufferHandle wraps a pointer to a buffer that contains display list item data.
44struct ItemBufferHandle {
45 ItemBufferIdentifier identifier;
46 uint8_t* data { nullptr };
47 size_t capacity { 0 };
48
49 operator bool() const { return !!(*this); }
50 bool operator!() const { return !data; }
51};
52
53using ItemBufferHandles = Vector<ItemBufferHandle, 2>;
54
55// An ItemHandle wraps a pointer to an ItemType followed immediately by an item of that type.
56// Each item handle data pointer is aligned to 8 bytes, and the item itself is also aligned
57// to 8 bytes. To ensure this, the item type consists of 8 bytes (1 byte for the type and 7
58// bytes of padding).
59struct ItemHandle {
60 uint8_t* data { nullptr };
61
62 void apply(GraphicsContext&);
63 WEBCORE_EXPORT void destroy();
64 bool isDrawingItem() const { return WebCore::DisplayList::isDrawingItem(type()); }
65
66 operator bool() const { return !!(*this); }
67 bool operator!() const { return !data; }
68
69#if !defined(NDEBUG) || !LOG_DISABLED
70 CString description() const;
71#endif
72
73 ItemType type() const { return static_cast<ItemType>(data[0]); }
74
75 template<typename T> bool is() const { return type() == T::itemType; }
76 template<typename T> T& get() const
77 {
78 ASSERT(is<T>());
79 return *reinterpret_cast<T*>(data + sizeof(uint64_t));
80 }
81
82 bool safeCopy(ItemType, ItemHandle destination) const;
83};

129// An ItemBuffer contains display list item data, and consists of a readwrite ItemBufferHandle (to which display
130// list items are appended) as well as a number of readonly ItemBufferHandles. Items are appended to the readwrite
131// buffer until all available capacity is exhausted, at which point we will move this writable handle to the list
132// of readonly handles.
133class ItemBuffer {
134 WTF_MAKE_NONCOPYABLE(ItemBuffer); WTF_MAKE_FAST_ALLOCATED;
135public:
136 friend class DisplayList;
137
138 WEBCORE_EXPORT ItemBuffer();
139 WEBCORE_EXPORT ~ItemBuffer();
140 ItemBuffer(ItemBufferHandles&&);
141
142 WEBCORE_EXPORT ItemBuffer(ItemBuffer&&);
143 WEBCORE_EXPORT ItemBuffer& operator=(ItemBuffer&&);
144
145 size_t sizeInBytes() const
146 {
147 size_t result = 0;
148 for (auto buffer : m_readOnlyBuffers)
149 result += buffer.capacity;
150 result += m_writtenNumberOfBytes;
151 return result;
152 }
153
154 void clear();
155 void shrinkToFit();
156
157 bool isEmpty() const
158 {
159 return !m_writtenNumberOfBytes && m_readOnlyBuffers.isEmpty();
160 }
161
162 WEBCORE_EXPORT ItemBufferHandle createItemBuffer(size_t);
163
164 // ItemBuffer::append<T>(...) appends a display list item of type T to the end of the current
165 // writable buffer handle; if remaining buffer capacity is insufficient to store the item, a
166 // new buffer will be allocated (either by the ItemBufferWritingClient, if set, or by the item
167 // buffer itself if there is no client). Items are placed back-to-back in these item buffers,
168 // with padding between each item to ensure that all items are aligned to 8 bytes.
169 //
170 // If a writing client is present and requires custom encoding for the given item type T, the
171 // item buffer will ask the client for an opaque SharedBuffer containing encoded data for the
172 // item. This encoded data is then appended to the item buffer, with padding to ensure that
173 // the start and end of this data are aligned to 8 bytes, if necessary. When consuming encoded
174 // item data, a corresponding ItemBufferReadingClient will be required to convert this encoded
175 // data back into an item of type T.
176 template<typename T, class... Args> void append(Args&&... args)
177 {
178 static_assert(std::is_trivially_destructible<T>::value == T::isInlineItem);
179
180 if constexpr (!T::isInlineItem) {
181 RELEASE_ASSERT(m_writingClient);
182 append(DisplayListItem(T(std::forward<Args>(args)...)));
183 return;
184 }
185
186 auto bufferChanged = swapWritableBufferIfNeeded(paddedSizeOfTypeAndItemInBytes(T::itemType));
187
188 uncheckedAppend<T>(bufferChanged, std::forward<Args>(args)...);
189 }
190
191 void setClient(ItemBufferWritingClient* client) { m_writingClient = client; }
192 void setClient(ItemBufferReadingClient* client) { m_readingClient = client; }
193 void prepareToAppend(ItemBufferHandle&&);
194
195private:
196 const ItemBufferHandles& readOnlyBuffers() const { return m_readOnlyBuffers; }
197 void forEachItemBuffer(Function<void(const ItemBufferHandle&)>&&) const;
198 WEBCORE_EXPORT void didAppendData(size_t numberOfBytes, DidChangeItemBuffer);
199
200 WEBCORE_EXPORT DidChangeItemBuffer swapWritableBufferIfNeeded(size_t numberOfBytes);
201 WEBCORE_EXPORT void append(const DisplayListItem&);
202 template<typename T, class... Args> void uncheckedAppend(DidChangeItemBuffer didChangeItemBuffer, Args&&... args)
203 {
204 auto* startOfItem = &m_writableBuffer.data[m_writtenNumberOfBytes];
205 startOfItem[0] = static_cast<uint8_t>(T::itemType);
206 new (startOfItem + sizeof(uint64_t)) T(std::forward<Args>(args)...);
207 didAppendData(paddedSizeOfTypeAndItemInBytes(T::itemType), didChangeItemBuffer);
208 }
209
210 ItemBufferReadingClient* m_readingClient { nullptr };
211 ItemBufferWritingClient* m_writingClient { nullptr };
212 Vector<uint8_t*> m_allocatedBuffers;
213 ItemBufferHandles m_readOnlyBuffers;
214 ItemBufferHandle m_writableBuffer;
215 size_t m_writtenNumberOfBytes { 0 };
216};
