117SetInlineFillGradient::SetInlineFillGradient(const Gradient& gradient, const AffineTransform& gradientSpaceTransform)
118 : m_data(gradient.data())
119 , m_gradientSpaceTransform(gradientSpaceTransform)
120 , m_spreadMethod(gradient.spreadMethod())
121 , m_colorStopCount(static_cast<uint8_t>(gradient.stops().size()))
122{
123 RELEASE_ASSERT(m_colorStopCount <= maxColorStopCount);
124 for (uint8_t i = 0; i < m_colorStopCount; ++i) {
125 m_offsets[i] = gradient.stops()[i].offset;
126 m_colors[i] = *gradient.stops()[i].color.tryGetAsSRGBABytes();
127 }
128}
129
130SetInlineFillGradient::SetInlineFillGradient(float offsets[maxColorStopCount], SRGBA<uint8_t> colors[maxColorStopCount], const Gradient::Data& data, const AffineTransform& gradientSpaceTransform, GradientSpreadMethod spreadMethod, uint8_t colorStopCount)
131 : m_data(data)
132 , m_gradientSpaceTransform(gradientSpaceTransform)
133 , m_spreadMethod(spreadMethod)
134 , m_colorStopCount(colorStopCount)
135{
136 RELEASE_ASSERT(m_colorStopCount <= maxColorStopCount);
137 for (uint8_t i = 0; i < m_colorStopCount; ++i) {
138 m_offsets[i] = offsets[i];
139 m_colors[i] = colors[i];
140 }
141}
142
143SetInlineFillGradient::SetInlineFillGradient(const SetInlineFillGradient& other)
144{
145 if (WTF::holds_alternative<Gradient::RadialData>(other.m_data) || WTF::holds_alternative<Gradient::LinearData>(other.m_data) || WTF::holds_alternative<Gradient::ConicData>(other.m_data)) {
146 m_data = other.m_data;
147 m_gradientSpaceTransform = other.m_gradientSpaceTransform;
148 m_spreadMethod = other.m_spreadMethod;
149 m_colorStopCount = other.m_colorStopCount;
150 if (m_colorStopCount > maxColorStopCount)
151 m_colorStopCount = 0;
152 for (uint8_t i = 0; i < m_colorStopCount; ++i) {
153 m_offsets[i] = other.m_offsets[i];
154 m_colors[i] = other.m_colors[i];
155 }
156 } else
157 m_isValid = false;
158}
159
160Ref<Gradient> SetInlineFillGradient::gradient() const
161{
162 auto gradient = Gradient::create(Gradient::Data(m_data));
163 for (uint8_t i = 0; i < m_colorStopCount; ++i)
164 gradient->addColorStop({ m_offsets[i], Color(m_colors[i]) });
165 gradient->setSpreadMethod(m_spreadMethod);
166 return gradient;
167}
168
169void SetInlineFillGradient::apply(GraphicsContext& context) const
170{
171 if (m_colorStopCount <= maxColorStopCount)
172 context.setFillGradient(gradient(), m_gradientSpaceTransform);
173}
174
175bool SetInlineFillGradient::isInline(const Gradient& gradient)
176{
177 if (gradient.stops().size() > SetInlineFillGradient::maxColorStopCount)
178 return false;
179
180 for (auto& colorStop : gradient.stops()) {
181 if (!colorStop.color.tryGetAsSRGBABytes())
182 return false;
183 }
184
185 return true;
186}

389DrawGlyphs::DrawGlyphs(const Font& font, const GlyphBufferGlyph* glyphs, const GlyphBufferAdvance* advances, unsigned count, const FloatPoint& localAnchor, FontSmoothingMode smoothingMode)
390 : m_fontIdentifier(font.renderingResourceIdentifier())
391 , m_localAnchor(localAnchor)
392 , m_smoothingMode(smoothingMode)
393{
394 m_glyphs.reserveInitialCapacity(count);
395 m_advances.reserveInitialCapacity(count);
396 for (unsigned i = 0; i < count; ++i) {
397 m_glyphs.uncheckedAppend(glyphs[i]);
398 m_advances.uncheckedAppend(advances[i]);
399 }
400 computeBounds(font);
401}

403void DrawGlyphs::apply(GraphicsContext& context, const Font& font) const
404{
405 context.drawGlyphs(font, m_glyphs.data(), m_advances.data(), m_glyphs.size(), anchorPoint(), m_smoothingMode);
406}
407
408void DrawGlyphs::computeBounds(const Font& font)
409{
410 // FIXME: This code doesn't actually take the extents of the glyphs into consideration. It assumes that
411 // the glyph lies entirely within its [(ascent + descent), advance] rect.
412 float ascent = font.fontMetrics().floatAscent();
413 float descent = font.fontMetrics().floatDescent();
414 FloatPoint current = localAnchor();
415 size_t numGlyphs = m_glyphs.size();
416 for (size_t i = 0; i < numGlyphs; ++i) {
417 GlyphBufferAdvance advance = m_advances[i];
418 FloatRect glyphRect = FloatRect(current.x(), current.y() - ascent, width(advance), ascent + descent);
419 m_bounds.unite(glyphRect);
420
421 current.move(width(advance), height(advance));
422 }
423}

429 ts.dumpProperty("anchor-point", item.anchorPoint());
430 ts.dumpProperty("length", item.glyphs().size());

432 return ts;

551std::optional<FloatRect> DrawLinesForText::localBounds(const GraphicsContext&) const
552{
553 // This function needs to return a value equal to or enclosing what GraphicsContext::computeLineBoundsAndAntialiasingModeForText() returns.
554
555 if (!m_widths.size())
556 return FloatRect();
557
558 FloatRect result(point(), FloatSize(m_widths.last(), m_thickness));
559 result.inflate(1); // Account for pixel snapping. FIXME: This isn't perfect, as it doesn't take the CTM into account.
560 return result;
561}
