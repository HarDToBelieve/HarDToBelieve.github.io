32bool DisplayList::Iterator::atEnd() const
33{
34 if (m_displayList.isEmpty() || !m_isValid)
35 return true;
36
37 auto& items = *itemBuffer();
38 auto endCursor = items.m_writableBuffer.data + items.m_writtenNumberOfBytes;
39 return m_cursor == endCursor;
40}

42auto DisplayList::Iterator::updateCurrentDrawingItemExtent(ItemType itemType) -> ExtentUpdateResult
43{
44 auto& extents = m_displayList.m_drawingItemExtents;
45 if (extents.isEmpty())
46 return ExtentUpdateResult::Success;
47
48 if (!isDrawingItem(itemType)) {
49 m_currentExtent = std::nullopt;
50 return ExtentUpdateResult::Success;
51 }
52
53 if (m_drawingItemIndex >= extents.size())
54 return ExtentUpdateResult::Failure;
55
56 m_currentExtent = extents[m_drawingItemIndex];
57 m_drawingItemIndex++;
58 return ExtentUpdateResult::Success;
59}

61void DisplayList::Iterator::updateCurrentItem()
62{
63 clearCurrentItem();
64
65 if (atEnd())
66 return;
67
68 auto& items = *itemBuffer();
69 auto rawItemTypeValue = m_cursor[0];
70 if (UNLIKELY(!isValidEnum<ItemType>(rawItemTypeValue))) {
71 m_isValid = false;
72 return;
73 }
74
75 auto itemType = static_cast<ItemType>(rawItemTypeValue);
76 if (UNLIKELY(updateCurrentDrawingItemExtent(itemType) == ExtentUpdateResult::Failure)) {
77 m_isValid = false;
78 return;
79 }
80
81 auto remainingCapacityInBuffer = static_cast<uint64_t>(m_currentEndOfBuffer - m_cursor);
82 auto paddedSizeOfTypeAndItem = paddedSizeOfTypeAndItemInBytes(itemType);
83 m_currentBufferForItem = paddedSizeOfTypeAndItem <= sizeOfFixedBufferForCurrentItem ? m_fixedBufferForCurrentItem : static_cast<uint8_t*>(fastMalloc(paddedSizeOfTypeAndItem));
84 if (isInlineItem(itemType)) {
85 if (UNLIKELY(remainingCapacityInBuffer < paddedSizeOfTypeAndItem)) {
86 m_isValid = false;
87 return;
88 }
89
90 if (UNLIKELY(!ItemHandle { m_cursor }.safeCopy(itemType, { m_currentBufferForItem }))) {
91 m_isValid = false;
92 return;
93 }
94
95 m_currentItemSizeInBuffer = paddedSizeOfTypeAndItem;
96 } else {
97 auto* client = items.m_readingClient;
98 RELEASE_ASSERT(client);
99 constexpr auto sizeOfTypeAndDataLength = 2 * sizeof(uint64_t);
100
101 if (UNLIKELY(remainingCapacityInBuffer < sizeOfTypeAndDataLength)) {
102 m_isValid = false;
103 return;
104 }
105
106 auto dataLength = reinterpret_cast<uint64_t*>(m_cursor)[1];
107 if (UNLIKELY(dataLength >= std::numeric_limits<uint32_t>::max() - alignof(uint64_t) - sizeOfTypeAndDataLength)) {
108 m_isValid = false;
109 return;
110 }
111
112 auto itemSizeInBuffer = roundUpToMultipleOf<alignof(uint64_t)>(dataLength) + sizeOfTypeAndDataLength;
113 if (UNLIKELY(remainingCapacityInBuffer < itemSizeInBuffer)) {
114 m_isValid = false;
115 return;
116 }
117
118 auto* startOfData = m_cursor + sizeOfTypeAndDataLength;
119 auto decodedItemHandle = client->decodeItem(startOfData, dataLength, itemType, m_currentBufferForItem);
120 if (UNLIKELY(!decodedItemHandle)) {
121 m_isValid = false;
122 return;
123 }
124
125 m_currentBufferForItem[0] = static_cast<uint8_t>(itemType);
126 m_currentItemSizeInBuffer = itemSizeInBuffer;
127 }
128}

130void DisplayList::Iterator::advance()
131{
132 if (atEnd())
133 return;
134
135 m_cursor += m_currentItemSizeInBuffer;
136
137 if (m_cursor == m_currentEndOfBuffer && m_readOnlyBufferIndex < itemBuffer()->m_readOnlyBuffers.size()) {
138 m_readOnlyBufferIndex++;
139 moveCursorToStartOfCurrentBuffer();
140 }
141
142 updateCurrentItem();
143}
144
145void DisplayList::Iterator::clearCurrentItem()
146{
147 auto items = itemBuffer();
148 if (items && items->m_readingClient && m_currentBufferForItem) {
149 if (LIKELY(m_isValid))
150 ItemHandle { m_currentBufferForItem }.destroy();
151
152 if (UNLIKELY(m_currentBufferForItem != m_fixedBufferForCurrentItem))
153 fastFree(m_currentBufferForItem);
154 }
155
156 m_currentItemSizeInBuffer = 0;
157 m_currentBufferForItem = nullptr;
158}

160void DisplayList::Iterator::moveToEnd()
161{
162 if (auto items = itemBuffer()) {
163 m_cursor = items->m_writableBuffer.data + items->m_writtenNumberOfBytes;
164 m_currentEndOfBuffer = m_cursor;
165 m_readOnlyBufferIndex = items->m_readOnlyBuffers.size();
166 }
167}
168
169void DisplayList::Iterator::moveCursorToStartOfCurrentBuffer()
170{
171 auto items = itemBuffer();
172 if (!items)
173 return;
174
175 auto numberOfReadOnlyBuffers = items->m_readOnlyBuffers.size();
176 if (m_readOnlyBufferIndex < numberOfReadOnlyBuffers) {
177 auto& nextBufferHandle = items->m_readOnlyBuffers[m_readOnlyBufferIndex];
178 m_cursor = nextBufferHandle.data;
179 m_currentEndOfBuffer = m_cursor + nextBufferHandle.capacity;
180 } else if (m_readOnlyBufferIndex == numberOfReadOnlyBuffers) {
181 m_cursor = items->m_writableBuffer.data;
182 m_currentEndOfBuffer = m_cursor + items->m_writtenNumberOfBytes;
183 }
184}
