175 template<typename T, class... Args>
176 void append(Args&&... args)
177 {
178 if (UNLIKELY(!canAppendItemOfType(T::itemType)))
179 return;
180
181 if constexpr (itemNeedsState<T>())
182 appendStateChangeItemIfNecessary();
183
184 m_displayList.append<T>(std::forward<Args>(args)...);
185
186 if constexpr (T::isDrawingItem) {
187 if (LIKELY(!m_displayList.tracksDrawingItemExtents()))
188 return;
189
190 auto item = T(std::forward<Args>(args)...);
191 if (auto rect = item.localBounds(*this))
192 m_displayList.addDrawingItemExtent(extentFromLocalBounds(*rect));
193 else if (auto rect = item.globalBounds())
194 m_displayList.addDrawingItemExtent(*rect);
195 else
196 m_displayList.addDrawingItemExtent(std::nullopt);
197 }
198 }

214 struct ContextState {
215 AffineTransform ctm;
216 FloatRect clipBounds;
217 GraphicsContextStateChange stateChange;
218 GraphicsContextState lastDrawingState;
219
220 ContextState(const GraphicsContextState& state, const AffineTransform& transform, const FloatRect& clip)
221 : ctm(transform)
222 , clipBounds(clip)
223 , lastDrawingState(state)
224 {
225 }
226
227 ContextState cloneForSave() const
228 {
229 ContextState state(lastDrawingState, ctm, clipBounds);
230 state.stateChange = stateChange;
231 return state;
232 }
233
234 ContextState cloneForTransparencyLayer() const
235 {
236 auto state = cloneForSave();
237 state.lastDrawingState.alpha = 1;
238 return state;
239 }

260template<typename T>
261constexpr bool Recorder::itemNeedsState()
262{
263 if (T::isDrawingItem)
264 return true;
265
266#if USE(CG)
267 if (T::itemType == ItemType::ApplyFillPattern || T::itemType == ItemType::ApplyStrokePattern)
268 return true;
269#endif
270
271 return false;
272}
