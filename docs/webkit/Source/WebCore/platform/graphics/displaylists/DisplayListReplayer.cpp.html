39Replayer::Replayer(GraphicsContext& context, const DisplayList& displayList, const ImageBufferHashMap* imageBuffers, const NativeImageHashMap* nativeImages, const FontRenderingResourceMap* fonts, ImageBuffer* maskImageBuffer, Delegate* delegate)
40 : m_context(context)
41 , m_maskImageBuffer(maskImageBuffer)
42 , m_displayList(displayList)
43 , m_imageBuffers(imageBuffers ? *imageBuffers : m_displayList.imageBuffers())
44 , m_nativeImages(nativeImages ? *nativeImages : m_displayList.nativeImages())
45 , m_fonts(fonts ? *fonts : m_displayList.fonts())
46 , m_delegate(delegate)
47{
48}

50Replayer::~Replayer() = default;

52GraphicsContext& Replayer::context() const
53{
54 return m_maskImageBuffer ? m_maskImageBuffer->context() : m_context;
55}

57template<class T>
58inline static std::optional<RenderingResourceIdentifier> applyImageBufferItem(GraphicsContext& context, const ImageBufferHashMap& imageBuffers, ItemHandle item)
59{
60 auto& imageBufferItem = item.get<T>();
61 auto resourceIdentifier = imageBufferItem.imageBufferIdentifier();
62 if (auto* imageBuffer = imageBuffers.get(resourceIdentifier)) {
63 imageBufferItem.apply(context, *imageBuffer);
64 return std::nullopt;
65 }
66 return resourceIdentifier;
67}
68
69template<class T>
70inline static std::optional<RenderingResourceIdentifier> applyNativeImageItem(GraphicsContext& context, const NativeImageHashMap& nativeImages, ItemHandle item)
71{
72 auto& nativeImageItem = item.get<T>();
73 auto resourceIdentifier = nativeImageItem.imageIdentifier();
74 if (auto* image = nativeImages.get(resourceIdentifier)) {
75 nativeImageItem.apply(context, *image);
76 return std::nullopt;
77 }
78 return resourceIdentifier;
79}

81inline static std::optional<RenderingResourceIdentifier> applySetStateItem(GraphicsContext& context, const NativeImageHashMap& nativeImages, ItemHandle item)
82{
83 auto& setStateItem = item.get<SetState>();
84
85 NativeImage* strokePatternImage = nullptr;
86 NativeImage* fillPatternImage = nullptr;
87
88 if (auto resourceIdentifier = setStateItem.strokePatternImageIdentifier()) {
89 strokePatternImage = nativeImages.get(resourceIdentifier);
90 if (!strokePatternImage)
91 return resourceIdentifier;
92 }
93
94 if (auto resourceIdentifier = setStateItem.fillPatternImageIdentifier()) {
95 fillPatternImage = nativeImages.get(resourceIdentifier);
96 if (!fillPatternImage)
97 return resourceIdentifier;
98 }
99
100 setStateItem.apply(context, strokePatternImage, fillPatternImage);
101 return std::nullopt;
102}

104template<class T>
105inline static std::optional<RenderingResourceIdentifier> applyFontItem(GraphicsContext& context, const FontRenderingResourceMap& fonts, ItemHandle item)
106{
107 auto& fontItem = item.get<T>();
108 auto resourceIdentifier = fontItem.fontIdentifier();
109 if (auto* font = fonts.get(resourceIdentifier)) {
110 fontItem.apply(context, *font);
111 return std::nullopt;
112 }
113 return resourceIdentifier;
114}

116std::pair<std::optional<StopReplayReason>, std::optional<RenderingResourceIdentifier>> Replayer::applyItem(ItemHandle item)
117{
118 if (m_delegate && m_delegate->apply(item, context()))
119 return { std::nullopt, std::nullopt };
120
121 if (item.is<DrawImageBuffer>()) {
122 if (auto missingCachedResourceIdentifier = applyImageBufferItem<DrawImageBuffer>(context(), m_imageBuffers, item))
123 return { StopReplayReason::MissingCachedResource, WTFMove(missingCachedResourceIdentifier) };
124 return { std::nullopt, std::nullopt };
125 }
126
127 if (item.is<ClipToImageBuffer>()) {
128 if (auto missingCachedResourceIdentifier = applyImageBufferItem<ClipToImageBuffer>(context(), m_imageBuffers, item))
129 return { StopReplayReason::MissingCachedResource, WTFMove(missingCachedResourceIdentifier) };
130 return { std::nullopt, std::nullopt };
131 }
132
133 if (item.is<DrawNativeImage>()) {
134 if (auto missingCachedResourceIdentifier = applyNativeImageItem<DrawNativeImage>(context(), m_nativeImages, item))
135 return { StopReplayReason::MissingCachedResource, WTFMove(missingCachedResourceIdentifier) };
136 return { std::nullopt, std::nullopt };
137 }
138
139 if (item.is<DrawGlyphs>()) {
140 if (auto missingCachedResourceIdentifier = applyFontItem<DrawGlyphs>(context(), m_fonts, item))
141 return { StopReplayReason::MissingCachedResource, WTFMove(missingCachedResourceIdentifier) };
142 return { std::nullopt, std::nullopt };
143 }
144
145 if (item.is<DrawPattern>()) {
146 if (auto missingCachedResourceIdentifier = applyNativeImageItem<DrawPattern>(context(), m_nativeImages, item))
147 return { StopReplayReason::MissingCachedResource, WTFMove(missingCachedResourceIdentifier) };
148 return { std::nullopt, std::nullopt };
149 }
150
151 if (item.is<SetState>()) {
152 if (auto missingCachedResourceIdentifier = applySetStateItem(context(), m_nativeImages, item))
153 return { StopReplayReason::MissingCachedResource, WTFMove(missingCachedResourceIdentifier) };
154 return { std::nullopt, std::nullopt };
155 }
156
157 if (item.is<BeginClipToDrawingCommands>()) {
158 if (m_maskImageBuffer)
159 return { StopReplayReason::InvalidItemOrExtent, std::nullopt };
160 auto& clipItem = item.get<BeginClipToDrawingCommands>();
161 m_maskImageBuffer = ImageBuffer::createCompatibleBuffer(clipItem.destination().size(), clipItem.colorSpace(), m_context);
162 if (!m_maskImageBuffer)
163 return { StopReplayReason::OutOfMemory, std::nullopt };
164 if (m_delegate)
165 m_delegate->didCreateMaskImageBuffer(*m_maskImageBuffer);
166 return { std::nullopt, std::nullopt };
167 }
168
169 if (item.is<EndClipToDrawingCommands>()) {
170 if (!m_maskImageBuffer)
171 return { StopReplayReason::InvalidItemOrExtent, std::nullopt };
172 auto& clipItem = item.get<EndClipToDrawingCommands>();
173 m_context.clipToImageBuffer(*m_maskImageBuffer, clipItem.destination());
174 m_maskImageBuffer = nullptr;
175 if (m_delegate)
176 m_delegate->didResetMaskImageBuffer();
177 return { std::nullopt, std::nullopt };
178 }
179
180 item.apply(context());
181 return { std::nullopt, std::nullopt };
182}

184ReplayResult Replayer::replay(const FloatRect& initialClip, bool trackReplayList)
185{
186 LOG_WITH_STREAM(DisplayLists, stream << "\nReplaying with clip " << initialClip);
187 UNUSED_PARAM(initialClip);
188
189 std::unique_ptr<InMemoryDisplayList> replayList;
190 if (UNLIKELY(trackReplayList))
191 replayList = makeUnique<InMemoryDisplayList>();
192
193#if !LOG_DISABLED
194 size_t i = 0;
195#endif
196 ReplayResult result;
197 for (auto displayListItem : m_displayList) {
198 if (!displayListItem) {
199 result.reasonForStopping = StopReplayReason::InvalidItemOrExtent;
200 break;
201 }
202
203 auto [item, extent, itemSizeInBuffer] = displayListItem.value();
204
205 if (!initialClip.isZero() && extent && !extent->intersects(initialClip)) {
206 LOG_WITH_STREAM(DisplayLists, stream << "skipping " << i++ << " " << item);
207 result.numberOfBytesRead += itemSizeInBuffer;
208 continue;
209 }
210
211 LOG_WITH_STREAM(DisplayLists, stream << "applying " << i++ << " " << item);
212
213 if (item.is<MetaCommandChangeDestinationImageBuffer>()) {
214 result.numberOfBytesRead += itemSizeInBuffer;
215 result.reasonForStopping = StopReplayReason::ChangeDestinationImageBuffer;
216 result.nextDestinationImageBuffer = item.get<MetaCommandChangeDestinationImageBuffer>().identifier();
217 break;
218 }
219
220 if (auto [reasonForStopping, missingCachedResourceIdentifier] = applyItem(item); reasonForStopping) {
221 result.reasonForStopping = *reasonForStopping;
222 result.missingCachedResourceIdentifier = WTFMove(missingCachedResourceIdentifier);
223 break;
224 }
225
226 result.numberOfBytesRead += itemSizeInBuffer;
227
228 if (UNLIKELY(trackReplayList)) {
229 replayList->append(item);
230 if (item.isDrawingItem())
231 replayList->addDrawingItemExtent(WTFMove(extent));
232 }
233 }
234
235 result.trackedDisplayList = WTFMove(replayList);
236 return result;
237}
