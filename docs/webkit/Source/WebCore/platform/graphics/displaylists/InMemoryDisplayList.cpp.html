34std::optional<std::size_t> InMemoryDisplayList::WritingClient::requiredSizeForItem(const DisplayListItem& displayListItem) const
35{
36 return paddedSizeOfTypeAndItemInBytes(displayListItem);
37}
38
39void InMemoryDisplayList::WritingClient::encodeItemInline(const DisplayListItem& displayListItem, uint8_t* location) const
40{
41 safeCopy({ location }, displayListItem);
42}
43
44std::optional<ItemHandle> WARN_UNUSED_RETURN InMemoryDisplayList::ReadingClient::decodeItem(const uint8_t* data, size_t dataLength, ItemType type, uint8_t* handleLocation)
45{
46 const ItemHandle dataHandle { const_cast<uint8_t*>(data) };
47 ASSERT_UNUSED(dataLength, dataLength >= paddedSizeOfTypeAndItemInBytes(dataHandle.type()));
48 ItemHandle result { handleLocation };
49 if (dataHandle.safeCopy(type, result))
50 return result;
51 return std::nullopt;
52}
53
54InMemoryDisplayList::InMemoryDisplayList()
55 : m_writingClient(WTF::makeUnique<WritingClient>())
56 , m_readingClient(WTF::makeUnique<ReadingClient>())
57{
58 setItemBufferWritingClient(m_writingClient.get());
59 setItemBufferReadingClient(m_readingClient.get());
60}
61
62InMemoryDisplayList::~InMemoryDisplayList()
63{
64 auto end = this->end();
65 for (auto displayListItem : *this) {
66 auto item = displayListItem->item;
67 ASSERT(item);
68 if (!item)
69 break;
70 item.destroy();
71 }
72}
