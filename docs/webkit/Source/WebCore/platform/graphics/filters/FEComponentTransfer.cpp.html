56void FEComponentTransfer::computeTabularTable(LookupTable& values, const ComponentTransferFunction& transferFunction)
57{
58 const Vector<float>& tableValues = transferFunction.tableValues;
59 unsigned n = tableValues.size();
60 if (n < 1)
61 return;
62 for (unsigned i = 0; i < values.size(); ++i) {
63 double c = i / 255.0;
64 unsigned k = static_cast<unsigned>(c * (n - 1));
65 double v1 = tableValues[k];
66 double v2 = tableValues[std::min((k + 1), (n - 1))];
67 double val = 255.0 * (v1 + (c * (n - 1) - k) * (v2 - v1));
68 val = std::max(0.0, std::min(255.0, val));
69 values[i] = static_cast<uint8_t>(val);
70 }
71}
72
73void FEComponentTransfer::computeDiscreteTable(LookupTable& values, const ComponentTransferFunction& transferFunction)
74{
75 const Vector<float>& tableValues = transferFunction.tableValues;
76 unsigned n = tableValues.size();
77 if (n < 1)
78 return;
79 for (unsigned i = 0; i < values.size(); ++i) {
80 unsigned k = static_cast<unsigned>((i * n) / 255.0);
81 k = std::min(k, n - 1);
82 double val = 255 * tableValues[k];
83 val = std::max(0.0, std::min(255.0, val));
84 values[i] = static_cast<uint8_t>(val);
85 }
86}
87
88void FEComponentTransfer::computeLinearTable(LookupTable& values, const ComponentTransferFunction& transferFunction)
89{
90 for (unsigned i = 0; i < values.size(); ++i) {
91 double val = transferFunction.slope * i + 255 * transferFunction.intercept;
92 val = std::max(0.0, std::min(255.0, val));
93 values[i] = static_cast<uint8_t>(val);
94 }
95}
96
97void FEComponentTransfer::computeGammaTable(LookupTable& values, const ComponentTransferFunction& transferFunction)
98{
99 for (unsigned i = 0; i < values.size(); ++i) {
100 double exponent = transferFunction.exponent; // RCVT doesn't like passing a double and a float to pow, so promote this to double
101 double val = 255.0 * (transferFunction.amplitude * pow((i / 255.0), exponent) + transferFunction.offset);
102 val = std::max(0.0, std::min(255.0, val));
103 values[i] = static_cast<uint8_t>(val);
104 }
105}

107void FEComponentTransfer::platformApplySoftware()
108{
109 FilterEffect* in = inputEffect(0);
110
111 auto& destinationPixelBuffer = createUnmultipliedImageResult();
112 if (!destinationPixelBuffer)
113 return;
114
115 auto& destinationPixelArray = destinationPixelBuffer->data();
116
117 LookupTable redTable;
118 LookupTable greenTable;
119 LookupTable blueTable;
120 LookupTable alphaTable;
121 computeLookupTables(redTable, greenTable, blueTable, alphaTable);
122
123 IntRect drawingRect = requestedRegionOfInputPixelBuffer(in->absolutePaintRect());
124 in->copyUnmultipliedResult(destinationPixelArray, drawingRect, operatingColorSpace());
125 unsigned destinationPixelArrayLength = destinationPixelArray.length();
126 uint8_t* data = destinationPixelArray.data();
127 for (unsigned pixelOffset = 0; pixelOffset < destinationPixelArrayLength; pixelOffset += 4) {
128 data[pixelOffset] = redTable[data[pixelOffset]];
129 data[pixelOffset + 1] = greenTable[data[pixelOffset + 1]];
130 data[pixelOffset + 2] = blueTable[data[pixelOffset + 2]];
131 data[pixelOffset + 3] = alphaTable[data[pixelOffset + 3]];
132 }
133}

135void FEComponentTransfer::computeLookupTables(LookupTable& redTable, LookupTable& greenTable, LookupTable& blueTable, LookupTable& alphaTable)
136{
137 for (unsigned i = 0; i < redTable.size(); ++i)
138 redTable[i] = greenTable[i] = blueTable[i] = alphaTable[i] = i;
139
140 using TransferType = void (*)(LookupTable&, const ComponentTransferFunction&);
141 TransferType callEffect[] = {
142 computeIdentityTable, // FECOMPONENTTRANSFER_TYPE_UNKNOWN
143 computeIdentityTable, // FECOMPONENTTRANSFER_TYPE_IDENTITY
144 computeTabularTable, // FECOMPONENTTRANSFER_TYPE_TABLE
145 computeDiscreteTable, // FECOMPONENTTRANSFER_TYPE_DISCRETE
146 computeLinearTable, // FECOMPONENTTRANSFER_TYPE_LINEAR
147 computeGammaTable // FECOMPONENTTRANSFER_TYPE_GAMMA
148 };
149
150 ASSERT_WITH_SECURITY_IMPLICATION(static_cast<size_t>(m_redFunction.type) < WTF_ARRAY_LENGTH(callEffect));
151 ASSERT_WITH_SECURITY_IMPLICATION(static_cast<size_t>(m_greenFunction.type) < WTF_ARRAY_LENGTH(callEffect));
152 ASSERT_WITH_SECURITY_IMPLICATION(static_cast<size_t>(m_blueFunction.type) < WTF_ARRAY_LENGTH(callEffect));
153 ASSERT_WITH_SECURITY_IMPLICATION(static_cast<size_t>(m_alphaFunction.type) < WTF_ARRAY_LENGTH(callEffect));
154
155 callEffect[m_redFunction.type](redTable, m_redFunction);
156 callEffect[m_greenFunction.type](greenTable, m_greenFunction);
157 callEffect[m_blueFunction.type](blueTable, m_blueFunction);
158 callEffect[m_alphaFunction.type](alphaTable, m_alphaFunction);
159}

161static TextStream& operator<<(TextStream& ts, ComponentTransferType type)
162{
163 switch (type) {
164 case FECOMPONENTTRANSFER_TYPE_UNKNOWN:
165 ts << "UNKNOWN";
166 break;
167 case FECOMPONENTTRANSFER_TYPE_IDENTITY:
168 ts << "IDENTITY";
169 break;
170 case FECOMPONENTTRANSFER_TYPE_TABLE:
171 ts << "TABLE";
172 break;
173 case FECOMPONENTTRANSFER_TYPE_DISCRETE:
174 ts << "DISCRETE";
175 break;
176 case FECOMPONENTTRANSFER_TYPE_LINEAR:
177 ts << "LINEAR";
178 break;
179 case FECOMPONENTTRANSFER_TYPE_GAMMA:
180 ts << "GAMMA";
181 break;
182 }
183 return ts;
184}

186static TextStream& operator<<(TextStream& ts, const ComponentTransferFunction& function)
187{
188 ts << "type=\"" << function.type;
189
190 switch (function.type) {
191 case FECOMPONENTTRANSFER_TYPE_UNKNOWN:
192 break;
193 case FECOMPONENTTRANSFER_TYPE_IDENTITY:
194 break;
195 case FECOMPONENTTRANSFER_TYPE_TABLE:
196 ts << " " << function.tableValues;
197 break;
198 case FECOMPONENTTRANSFER_TYPE_DISCRETE:
199 ts << " " << function.tableValues;
200 break;
201 case FECOMPONENTTRANSFER_TYPE_LINEAR:
202 ts << "\" slope=\"" << function.slope << "\" intercept=\"" << function.intercept << "\"";
203 break;
204 case FECOMPONENTTRANSFER_TYPE_GAMMA:
205 ts << "\" amplitude=\"" << function.amplitude << "\" exponent=\"" << function.exponent << "\" offset=\"" << function.offset << "\"";
206 break;
207 }
208
209 return ts;
210}
