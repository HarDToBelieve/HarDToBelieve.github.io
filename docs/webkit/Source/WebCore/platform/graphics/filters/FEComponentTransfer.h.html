29namespace WebCore {
30
31enum ComponentTransferType {
32 FECOMPONENTTRANSFER_TYPE_UNKNOWN = 0,
33 FECOMPONENTTRANSFER_TYPE_IDENTITY = 1,
34 FECOMPONENTTRANSFER_TYPE_TABLE = 2,
35 FECOMPONENTTRANSFER_TYPE_DISCRETE = 3,
36 FECOMPONENTTRANSFER_TYPE_LINEAR = 4,
37 FECOMPONENTTRANSFER_TYPE_GAMMA = 5
38};
39
40struct ComponentTransferFunction {
41 ComponentTransferFunction() = default;
42
43 ComponentTransferType type { FECOMPONENTTRANSFER_TYPE_UNKNOWN };
44
45 float slope { 0 };
46 float intercept { 0 };
47 float amplitude { 0 };
48 float exponent { 0 };
49 float offset { 0 };
50
51 Vector<float> tableValues;
52};
53
54class FEComponentTransfer : public FilterEffect {
55public:
56 static Ref<FEComponentTransfer> create(Filter&, const ComponentTransferFunction& redFunc, const ComponentTransferFunction& greenFunc,
57 const ComponentTransferFunction& blueFunc, const ComponentTransferFunction& alphaFunc);
58
59 ComponentTransferFunction redFunction() const { return m_redFunction; }
60 ComponentTransferFunction greenFunction() const { return m_greenFunction; }
61 ComponentTransferFunction blueFunction() const { return m_blueFunction; }
62 ComponentTransferFunction alphaFunction() const { return m_alphaFunction; }
63
64private:
65 FEComponentTransfer(Filter&, const ComponentTransferFunction& redFunc, const ComponentTransferFunction& greenFunc,
66 const ComponentTransferFunction& blueFunc, const ComponentTransferFunction& alphaFunc);
67
68 const char* filterName() const final { return "FEComponentTransfer"; }
69
70 using LookupTable = std::array<uint8_t, 256>;
71
72 static void computeIdentityTable(LookupTable&, const ComponentTransferFunction&);
73 static void computeTabularTable(LookupTable&, const ComponentTransferFunction&);
74 static void computeDiscreteTable(LookupTable&, const ComponentTransferFunction&);
75 static void computeLinearTable(LookupTable&, const ComponentTransferFunction&);
76 static void computeGammaTable(LookupTable&, const ComponentTransferFunction&);
77
78 void computeLookupTables(LookupTable& redTable, LookupTable& greenTable, LookupTable& blueTable, LookupTable& alphaTable);
79
80 void platformApplySoftware() override;
81
82 WTF::TextStream& externalRepresentation(WTF::TextStream&, RepresentationType) const override;
83
84 ComponentTransferFunction m_redFunction;
85 ComponentTransferFunction m_greenFunction;
86 ComponentTransferFunction m_blueFunction;
87 ComponentTransferFunction m_alphaFunction;
88};
89
90} // namespace WebCore
