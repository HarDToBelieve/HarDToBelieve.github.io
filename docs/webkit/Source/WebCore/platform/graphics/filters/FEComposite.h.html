29namespace WebCore {
30
31enum CompositeOperationType {
32 FECOMPOSITE_OPERATOR_UNKNOWN = 0,
33 FECOMPOSITE_OPERATOR_OVER = 1,
34 FECOMPOSITE_OPERATOR_IN = 2,
35 FECOMPOSITE_OPERATOR_OUT = 3,
36 FECOMPOSITE_OPERATOR_ATOP = 4,
37 FECOMPOSITE_OPERATOR_XOR = 5,
38 FECOMPOSITE_OPERATOR_ARITHMETIC = 6,
39 FECOMPOSITE_OPERATOR_LIGHTER = 7
40};
41
42class FEComposite : public FilterEffect {
43public:
44 static Ref<FEComposite> create(Filter&, const CompositeOperationType&, float, float, float, float);
45
46 CompositeOperationType operation() const { return m_type; }
47 bool setOperation(CompositeOperationType);
48
49 float k1() const { return m_k1; }
50 bool setK1(float);
51
52 float k2() const { return m_k2; }
53 bool setK2(float);
54
55 float k3() const { return m_k3; }
56 bool setK3(float);
57
58 float k4() const { return m_k4; }
59 bool setK4(float);
60
61protected:
62 bool requiresValidPreMultipliedPixels() override { return m_type != FECOMPOSITE_OPERATOR_ARITHMETIC; }
63
64private:
65 FEComposite(Filter&, const CompositeOperationType&, float, float, float, float);
66
67 const char* filterName() const final { return "FEComposite"; }
68
69 void correctFilterResultIfNeeded() override;
70 void determineAbsolutePaintRect() override;
71
72 void platformApplySoftware() override;
73 WTF::TextStream& externalRepresentation(WTF::TextStream&, RepresentationType) const override;
74
75 inline void platformArithmeticSoftware(const Uint8ClampedArray& source, Uint8ClampedArray& destination, float k1, float k2, float k3, float k4);
76
77#if HAVE(ARM_NEON_INTRINSICS)
78 template <int b1, int b4>
79 static inline void computeArithmeticPixelsNeon(const uint8_t* source, uint8_t* destination, unsigned pixelArrayLength, float k1, float k2, float k3, float k4);
80
81 static inline void platformArithmeticNeon(const uint8_t* source, uint8_t* destination, unsigned pixelArrayLength, float k1, float k2, float k3, float k4);
82#endif
83
84 CompositeOperationType m_type;
85 float m_k1;
86 float m_k2;
87 float m_k3;
88 float m_k4;
89};
