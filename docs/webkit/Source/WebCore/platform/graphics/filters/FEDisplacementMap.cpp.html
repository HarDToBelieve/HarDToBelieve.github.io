72void FEDisplacementMap::setResultColorSpace(const DestinationColorSpace&)
73{
74 // Spec: The 'color-interpolation-filters' property only applies to the 'in2' source image
75 // and does not apply to the 'in' source image. The 'in' source image must remain in its
76 // current color space.
77 // The result is in that smae color space because it is a displacement of the 'in' image.
78 FilterEffect::setResultColorSpace(inputEffect(0)->resultColorSpace());
79}

81void FEDisplacementMap::transformResultColorSpace(FilterEffect* in, const int index)
82{
83 // Do not transform the first primitive input, as per the spec.
84 if (index)
85 in->transformResultColorSpace(operatingColorSpace());
86}

88static inline unsigned byteOffsetOfPixel(unsigned x, unsigned y, unsigned rowBytes)
89{
90 const unsigned bytesPerPixel = 4;
91 return x * bytesPerPixel + y * rowBytes;
92}
93
94void FEDisplacementMap::platformApplySoftware()
95{
96 FilterEffect* in = inputEffect(0);
97 FilterEffect* in2 = inputEffect(1);
98
99 ASSERT(m_xChannelSelector != CHANNEL_UNKNOWN);
100 ASSERT(m_yChannelSelector != CHANNEL_UNKNOWN);
101
102 auto& destinationPixelBuffer = createPremultipliedImageResult();
103 if (!destinationPixelBuffer)
104 return;
105
106 auto& destinationPixelArray = destinationPixelBuffer->data();
107
108 IntRect effectADrawingRect = requestedRegionOfInputPixelBuffer(in->absolutePaintRect());
109 auto inputImage = in->premultipliedResult(effectADrawingRect);
110
111 IntRect effectBDrawingRect = requestedRegionOfInputPixelBuffer(in2->absolutePaintRect());
112 // The calculations using the pixel values from ‘in2’ are performed using non-premultiplied color values.
113 auto displacementImage = in2->unmultipliedResult(effectBDrawingRect);
114
115 if (!inputImage || !displacementImage)
116 return;
117
118 ASSERT(inputImage->length() == displacementImage->length());
119
120 Filter& filter = this->filter();
121 IntSize paintSize = absolutePaintRect().size();
122 paintSize.scale(filter.filterScale());
123
124 FloatSize scale = filter.scaledByFilterResolution({ m_scale, m_scale });
125 float scaleForColorX = scale.width() / 255.0;
126 float scaleForColorY = scale.height() / 255.0;
127 float scaledOffsetX = 0.5 - scale.width() * 0.5;
128 float scaledOffsetY = 0.5 - scale.height() * 0.5;
129
130 int displacementChannelX = xChannelIndex();
131 int displacementChannelY = yChannelIndex();
132
133 int rowBytes = paintSize.width() * 4;
134
135 for (int y = 0; y < paintSize.height(); ++y) {
136 int lineStartOffset = y * rowBytes;
137
138 for (int x = 0; x < paintSize.width(); ++x) {
139 int destinationIndex = lineStartOffset + x * 4;
140
141 int srcX = x + static_cast<int>(scaleForColorX * displacementImage->item(destinationIndex + displacementChannelX) + scaledOffsetX);
142 int srcY = y + static_cast<int>(scaleForColorY * displacementImage->item(destinationIndex + displacementChannelY) + scaledOffsetY);
143
144 unsigned* destinationPixelPtr = reinterpret_cast<unsigned*>(destinationPixelArray.data() + destinationIndex);
145 if (srcX < 0 || srcX >= paintSize.width() || srcY < 0 || srcY >= paintSize.height()) {
146 *destinationPixelPtr = 0;
147 continue;
148 }
149
150 *destinationPixelPtr = *reinterpret_cast<unsigned*>(inputImage->data() + byteOffsetOfPixel(srcX, srcY, rowBytes));
151 }
152 }
153}

155static TextStream& operator<<(TextStream& ts, const ChannelSelectorType& type)
156{
157 switch (type) {
158 case CHANNEL_UNKNOWN:
159 ts << "UNKNOWN";
160 break;
161 case CHANNEL_R:
162 ts << "RED";
163 break;
164 case CHANNEL_G:
165 ts << "GREEN";
166 break;
167 case CHANNEL_B:
168 ts << "BLUE";
169 break;
170 case CHANNEL_A:
171 ts << "ALPHA";
172 break;
173 }
174 return ts;
175}

177TextStream& FEDisplacementMap::externalRepresentation(TextStream& ts, RepresentationType representation) const
178{
179 ts << indent << "[feDisplacementMap";
180 FilterEffect::externalRepresentation(ts, representation);
181 ts << " scale=\"" << m_scale << "\" "
182 << "xChannelSelector=\"" << m_xChannelSelector << "\" "
183 << "yChannelSelector=\"" << m_yChannelSelector << "\"]\n";
184
185 TextStream::IndentScope indentScope(ts);
186 inputEffect(0)->externalRepresentation(ts, representation);
187 inputEffect(1)->externalRepresentation(ts, representation);
188 return ts;
189}
