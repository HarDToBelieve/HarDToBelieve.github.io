28namespace WebCore {
29
30enum ChannelSelectorType {
31 CHANNEL_UNKNOWN = 0,
32 CHANNEL_R = 1,
33 CHANNEL_G = 2,
34 CHANNEL_B = 3,
35 CHANNEL_A = 4
36};
37
38class FEDisplacementMap : public FilterEffect {
39public:
40 static Ref<FEDisplacementMap> create(Filter&, ChannelSelectorType xChannelSelector, ChannelSelectorType yChannelSelector, float scale);
41
42 ChannelSelectorType xChannelSelector() const { return m_xChannelSelector; }
43 bool setXChannelSelector(const ChannelSelectorType);
44
45 ChannelSelectorType yChannelSelector() const { return m_yChannelSelector; }
46 bool setYChannelSelector(const ChannelSelectorType);
47
48 float scale() const { return m_scale; }
49 bool setScale(float);
50
51 void setResultColorSpace(const DestinationColorSpace&) override;
52 void transformResultColorSpace(FilterEffect*, const int) override;
53
54private:
55 FEDisplacementMap(Filter&, ChannelSelectorType xChannelSelector, ChannelSelectorType yChannelSelector, float);
56
57 const char* filterName() const final { return "FEDisplacementMap"; }
58
59 void platformApplySoftware() override;
60
61 void determineAbsolutePaintRect() override { setAbsolutePaintRect(enclosingIntRect(maxEffectRect())); }
62
63 int xChannelIndex() const { return m_xChannelSelector - 1; }
64 int yChannelIndex() const { return m_yChannelSelector - 1; }
65
66 WTF::TextStream& externalRepresentation(WTF::TextStream&, RepresentationType) const override;
67
68 ChannelSelectorType m_xChannelSelector;
69 ChannelSelectorType m_yChannelSelector;
70 float m_scale;
71};
