50void FEDropShadow::determineAbsolutePaintRect()
51{
52 Filter& filter = this->filter();
53
54 FloatRect absolutePaintRect = inputEffect(0)->absolutePaintRect();
55 FloatRect absoluteOffsetPaintRect(absolutePaintRect);
56 absoluteOffsetPaintRect.move(filter.scaledByFilterResolution({ m_dx, m_dy }));
57 absolutePaintRect.unite(absoluteOffsetPaintRect);
58
59 IntSize kernelSize = FEGaussianBlur::calculateKernelSize(filter, { m_stdX, m_stdY });
60
61 // We take the half kernel size and multiply it with three, because we run box blur three times.
62 absolutePaintRect.inflateX(3 * kernelSize.width() * 0.5f);
63 absolutePaintRect.inflateY(3 * kernelSize.height() * 0.5f);
64
65 if (clipsToBounds())
66 absolutePaintRect.intersect(maxEffectRect());
67 else
68 absolutePaintRect.unite(maxEffectRect());
69
70 setAbsolutePaintRect(enclosingIntRect(absolutePaintRect));
71}

73void FEDropShadow::platformApplySoftware()
74{
75 FilterEffect* in = inputEffect(0);
76
77 ImageBuffer* resultImage = createImageBufferResult();
78 if (!resultImage)
79 return;
80
81 Filter& filter = this->filter();
82
83 FloatSize blurRadius = 2 * filter.scaledByFilterResolution({ m_stdX, m_stdY });
84 blurRadius.scale(filter.filterScale());
85 FloatSize offset = filter.scaledByFilterResolution({ m_dx, m_dy });
86
87 FloatRect drawingRegion = drawingRegionOfInputImage(in->absolutePaintRect());
88 FloatRect drawingRegionWithOffset(drawingRegion);
89 drawingRegionWithOffset.move(offset);
90
91 ImageBuffer* sourceImage = in->imageBufferResult();
92 if (!sourceImage)
93 return;
94
95 GraphicsContext& resultContext = resultImage->context();
96 resultContext.setAlpha(m_shadowOpacity);
97 resultContext.drawImageBuffer(*sourceImage, drawingRegionWithOffset);
98 resultContext.setAlpha(1);
99
100 ShadowBlur contextShadow(blurRadius, offset, m_shadowColor);
101
102 PixelBufferFormat format { AlphaPremultiplication::Premultiplied, PixelFormat::RGBA8, resultColorSpace() };
103 IntRect shadowArea(IntPoint(), resultImage->logicalSize());
104 auto pixelBuffer = resultImage->getPixelBuffer(format, shadowArea);
105 if (!pixelBuffer)
106 return;
107
108 auto& sourcePixelArray = pixelBuffer->data();
109 contextShadow.blurLayerImage(sourcePixelArray.data(), pixelBuffer->size(), 4 * pixelBuffer->size().width());
110
111 resultImage->putPixelBuffer(*pixelBuffer, shadowArea);
112
113 resultContext.setCompositeOperation(CompositeOperator::SourceIn);
114 resultContext.fillRect(FloatRect(FloatPoint(), absolutePaintRect().size()), m_shadowColor);
115 resultContext.setCompositeOperation(CompositeOperator::DestinationOver);
116
117 resultImage->context().drawImageBuffer(*sourceImage, drawingRegion);
118}

120IntOutsets FEDropShadow::outsets() const
121{
122 IntSize outsetSize = FEGaussianBlur::calculateOutsetSize({ m_stdX, m_stdY });
123 return {
124 std::max<int>(0, outsetSize.height() - m_dy),
125 std::max<int>(0, outsetSize.width() + m_dx),
126 std::max<int>(0, outsetSize.height() + m_dy),
127 std::max<int>(0, outsetSize.width() - m_dx)
128 };
129}

131TextStream& FEDropShadow::externalRepresentation(TextStream& ts, RepresentationType representation) const
132{
133 ts << indent <<"[feDropShadow";
134 FilterEffect::externalRepresentation(ts, representation);
135 ts << " stdDeviation=\"" << m_stdX << ", " << m_stdY << "\" dx=\"" << m_dx << "\" dy=\"" << m_dy << "\" flood-color=\"" << serializationForRenderTreeAsText(m_shadowColor) <<"\" flood-opacity=\"" << m_shadowOpacity << "]\n";
136
137 TextStream::IndentScope indentScope(ts);
138 inputEffect(0)->externalRepresentation(ts, representation);
139 return ts;
140}
