33FEFlood::FEFlood(Filter& filter, const Color& floodColor, float floodOpacity)
34 : FilterEffect(filter, Type::Flood)
35 , m_floodColor(floodColor)
36 , m_floodOpacity(floodOpacity)
37{
38}
39
40Ref<FEFlood> FEFlood::create(Filter& filter, const Color& floodColor, float floodOpacity)
41{
42 return adoptRef(*new FEFlood(filter, floodColor, floodOpacity));
43}
44
45bool FEFlood::setFloodColor(const Color& color)
46{
47 if (m_floodColor == color)
48 return false;
49 m_floodColor = color;
50 return true;
51}
52
53bool FEFlood::setFloodOpacity(float floodOpacity)
54{
55 if (m_floodOpacity == floodOpacity)
56 return false;
57 m_floodOpacity = floodOpacity;
58 return true;
59}
60
61void FEFlood::platformApplySoftware()
62{
63 ImageBuffer* resultImage = createImageBufferResult();
64 if (!resultImage)
65 return;
66
67 auto color = floodColor().colorWithAlphaMultipliedBy(floodOpacity());
68 resultImage->context().fillRect(FloatRect(FloatPoint(), absolutePaintRect().size()), color);
69}
70
71TextStream& FEFlood::externalRepresentation(TextStream& ts, RepresentationType representation) const
72{
73 ts << indent << "[feFlood";
74 FilterEffect::externalRepresentation(ts, representation);
75 ts << " flood-color=\"" << serializationForRenderTreeAsText(floodColor()) << "\" "
76 << "flood-opacity=\"" << floodOpacity() << "\"]\n";
77 return ts;
78}
