88static constexpr int cPixelSize = 4;
89static constexpr int cAlphaChannelOffset = 3;
90static constexpr uint8_t cOpaqueAlpha = static_cast<uint8_t>(0xFF);
91
92// These factors and the normal coefficients come from the table under https://www.w3.org/TR/SVG/filters.html#feDiffuseLightingElement.
93static constexpr float cFactor1div2 = -1 / 2.f;
94static constexpr float cFactor1div3 = -1 / 3.f;
95static constexpr float cFactor1div4 = -1 / 4.f;
96static constexpr float cFactor2div3 = -2 / 3.f;

98inline IntSize FELighting::LightingData::topLeftNormal(int offset) const
99{
100 int center = static_cast<int>(pixels->item(offset + cAlphaChannelOffset));
101 int right = static_cast<int>(pixels->item(offset + cPixelSize + cAlphaChannelOffset));
102 offset += widthMultipliedByPixelSize;
103 int bottom = static_cast<int>(pixels->item(offset + cAlphaChannelOffset));
104 int bottomRight = static_cast<int>(pixels->item(offset + cPixelSize + cAlphaChannelOffset));
105 return {
106 -2 * center + 2 * right - bottom + bottomRight,
107 -2 * center - right + 2 * bottom + bottomRight
108 };
109}

111inline IntSize FELighting::LightingData::topRowNormal(int offset) const
112{
113 int left = static_cast<int>(pixels->item(offset - cPixelSize + cAlphaChannelOffset));
114 int center = static_cast<int>(pixels->item(offset + cAlphaChannelOffset));
115 int right = static_cast<int>(pixels->item(offset + cPixelSize + cAlphaChannelOffset));
116 offset += widthMultipliedByPixelSize;
117 int bottomLeft = static_cast<int>(pixels->item(offset - cPixelSize + cAlphaChannelOffset));
118 int bottom = static_cast<int>(pixels->item(offset + cAlphaChannelOffset));
119 int bottomRight = static_cast<int>(pixels->item(offset + cPixelSize + cAlphaChannelOffset));
120 return {
121 -2 * left + 2 * right - bottomLeft + bottomRight,
122 -left - 2 * center - right + bottomLeft + 2 * bottom + bottomRight
123 };
124}

126inline IntSize FELighting::LightingData::topRightNormal(int offset) const
127{
128 int left = static_cast<int>(pixels->item(offset - cPixelSize + cAlphaChannelOffset));
129 int center = static_cast<int>(pixels->item(offset + cAlphaChannelOffset));
130 offset += widthMultipliedByPixelSize;
131 int bottomLeft = static_cast<int>(pixels->item(offset - cPixelSize + cAlphaChannelOffset));
132 int bottom = static_cast<int>(pixels->item(offset + cAlphaChannelOffset));
133 return {
134 -2 * left + 2 * center - bottomLeft + bottom,
135 -left - 2 * center + bottomLeft + 2 * bottom
136 };
137}

139inline IntSize FELighting::LightingData::leftColumnNormal(int offset) const
140{
141 int center = static_cast<int>(pixels->item(offset + cAlphaChannelOffset));
142 int right = static_cast<int>(pixels->item(offset + cPixelSize + cAlphaChannelOffset));
143 offset -= widthMultipliedByPixelSize;
144 int top = static_cast<int>(pixels->item(offset + cAlphaChannelOffset));
145 int topRight = static_cast<int>(pixels->item(offset + cPixelSize + cAlphaChannelOffset));
146 offset += 2 * widthMultipliedByPixelSize;
147 int bottom = static_cast<int>(pixels->item(offset + cAlphaChannelOffset));
148 int bottomRight = static_cast<int>(pixels->item(offset + cPixelSize + cAlphaChannelOffset));
149 return {
150 -top + topRight - 2 * center + 2 * right - bottom + bottomRight,
151 -2 * top - topRight + 2 * bottom + bottomRight
152 };
153}

155inline IntSize FELighting::LightingData::interiorNormal(int offset, AlphaWindow& alphaWindow) const
156{
157 int rightAlphaOffset = offset + cPixelSize + cAlphaChannelOffset;
158
159 int right = static_cast<int>(pixels->item(rightAlphaOffset));
160 int topRight = static_cast<int>(pixels->item(rightAlphaOffset - widthMultipliedByPixelSize));
161 int bottomRight = static_cast<int>(pixels->item(rightAlphaOffset + widthMultipliedByPixelSize));
162
163 int left = alphaWindow.left();
164 int topLeft = alphaWindow.topLeft();
165 int top = alphaWindow.top();
166
167 int bottomLeft = alphaWindow.bottomLeft();
168 int bottom = alphaWindow.bottom();
169
170 // The alphaWindow has been shifted, and here we fill in the right column.
171 alphaWindow.alpha[0][2] = topRight;
172 alphaWindow.alpha[1][2] = right;
173 alphaWindow.alpha[2][2] = bottomRight;
174
175 // Check that the alphaWindow is working with some spot-checks.
176 ASSERT(alphaWindow.topLeft() == pixels->item(offset - cPixelSize - widthMultipliedByPixelSize + cAlphaChannelOffset)); // topLeft
177 ASSERT(alphaWindow.top() == pixels->item(offset - widthMultipliedByPixelSize + cAlphaChannelOffset)); // top
178
179 return {
180 -topLeft + topRight - 2 * left + 2 * right - bottomLeft + bottomRight,
181 -topLeft - 2 * top - topRight + bottomLeft + 2 * bottom + bottomRight
182 };
183}

185inline IntSize FELighting::LightingData::rightColumnNormal(int offset) const
186{
187 int left = static_cast<int>(pixels->item(offset - cPixelSize + cAlphaChannelOffset));
188 int center = static_cast<int>(pixels->item(offset + cAlphaChannelOffset));
189 offset -= widthMultipliedByPixelSize;
190 int topLeft = static_cast<int>(pixels->item(offset - cPixelSize + cAlphaChannelOffset));
191 int top = static_cast<int>(pixels->item(offset + cAlphaChannelOffset));
192 offset += 2 * widthMultipliedByPixelSize;
193 int bottomLeft = static_cast<int>(pixels->item(offset - cPixelSize + cAlphaChannelOffset));
194 int bottom = static_cast<int>(pixels->item(offset + cAlphaChannelOffset));
195 return {
196 -topLeft + top - 2 * left + 2 * center - bottomLeft + bottom,
197 -topLeft - 2 * top + bottomLeft + 2 * bottom
198 };
199}

201inline IntSize FELighting::LightingData::bottomLeftNormal(int offset) const
202{
203 int center = static_cast<int>(pixels->item(offset + cAlphaChannelOffset));
204 int right = static_cast<int>(pixels->item(offset + cPixelSize + cAlphaChannelOffset));
205 offset -= widthMultipliedByPixelSize;
206 int top = static_cast<int>(pixels->item(offset + cAlphaChannelOffset));
207 int topRight = static_cast<int>(pixels->item(offset + cPixelSize + cAlphaChannelOffset));
208 return {
209 -top + topRight - 2 * center + 2 * right,
210 -2 * top - topRight + 2 * center + right

214inline IntSize FELighting::LightingData::bottomRowNormal(int offset) const
215{
216 int left = static_cast<int>(pixels->item(offset - cPixelSize + cAlphaChannelOffset));
217 int center = static_cast<int>(pixels->item(offset + cAlphaChannelOffset));
218 int right = static_cast<int>(pixels->item(offset + cPixelSize + cAlphaChannelOffset));
219 offset -= widthMultipliedByPixelSize;
220 int topLeft = static_cast<int>(pixels->item(offset - cPixelSize + cAlphaChannelOffset));
221 int top = static_cast<int>(pixels->item(offset + cAlphaChannelOffset));
222 int topRight = static_cast<int>(pixels->item(offset + cPixelSize + cAlphaChannelOffset));
223 return {
224 -topLeft + topRight - 2 * left + 2 * right,
225 -topLeft - 2 * top - topRight + left + 2 * center + right
226 };

229inline IntSize FELighting::LightingData::bottomRightNormal(int offset) const
230{
231 int left = static_cast<int>(pixels->item(offset - cPixelSize + cAlphaChannelOffset));
232 int center = static_cast<int>(pixels->item(offset + cAlphaChannelOffset));
233 offset -= widthMultipliedByPixelSize;
234 int topLeft = static_cast<int>(pixels->item(offset - cPixelSize + cAlphaChannelOffset));
235 int top = static_cast<int>(pixels->item(offset + cAlphaChannelOffset));
236 return {
237 -topLeft + top - 2 * left + 2 * center,
238 -topLeft - 2 * top + left + 2 * center

247void FELighting::setPixelInternal(int offset, const LightingData& data, const LightSource::PaintingData& paintingData, int x, int y, float factorX, float factorY, IntSize normal2DVector, float alpha)
248{
249 float z = alpha * data.surfaceScale;
250 LightSource::ComputedLightingData lightingData = m_lightSource->computePixelLightingData(paintingData, x, y, z);
251
252 float lightStrength;
253 if (normal2DVector.isZero()) {
254 // Normal vector is (0, 0, 1). This is a quite frequent case.
255 if (m_lightingType == FELighting::DiffuseLighting)
256 lightStrength = m_diffuseConstant * lightingData.lightVector.z() / lightingData.lightVectorLength;
257 else {
258 FloatPoint3D halfwayVector = {
259 lightingData.lightVector.x(),
260 lightingData.lightVector.y(),
261 lightingData.lightVector.z() + lightingData.lightVectorLength
262 };
263 float halfwayVectorLength = halfwayVector.length();
264 if (m_specularExponent == 1)
265 lightStrength = m_specularConstant * halfwayVector.z() / halfwayVectorLength;
266 else
267 lightStrength = m_specularConstant * powf(halfwayVector.z() / halfwayVectorLength, m_specularExponent);
268 }
269 } else {
270 FloatPoint3D normalVector = {
271 factorX * normal2DVector.width() * data.surfaceScale,
272 factorY * normal2DVector.height() * data.surfaceScale,
273 1.0f
274 };
275 float normalVectorLength = normalVector.length();
276
277 if (m_lightingType == FELighting::DiffuseLighting)
278 lightStrength = m_diffuseConstant * (normalVector * lightingData.lightVector) / (normalVectorLength * lightingData.lightVectorLength);
279 else {
280 FloatPoint3D halfwayVector = {
281 lightingData.lightVector.x(),
282 lightingData.lightVector.y(),
283 lightingData.lightVector.z() + lightingData.lightVectorLength
284 };
285 float halfwayVectorLength = halfwayVector.length();
286 if (m_specularExponent == 1)
287 lightStrength = m_specularConstant * (normalVector * halfwayVector) / (normalVectorLength * halfwayVectorLength);
288 else
289 lightStrength = m_specularConstant * powf((normalVector * halfwayVector) / (normalVectorLength * halfwayVectorLength), m_specularExponent);
290 }
291 }
292
293 if (lightStrength > 1)
294 lightStrength = 1;
295 if (lightStrength < 0)
296 lightStrength = 0;
297
298 uint8_t pixelValue[3] = {
299 static_cast<uint8_t>(lightStrength * lightingData.colorVector.x() * 255.0f),
300 static_cast<uint8_t>(lightStrength * lightingData.colorVector.y() * 255.0f),
301 static_cast<uint8_t>(lightStrength * lightingData.colorVector.z() * 255.0f)
302 };
303
304 data.pixels->setRange(pixelValue, 3, offset);
305}

307// This appears to read from and write to the same pixel buffer, but it only reads the alpha channel, and writes the non-alpha channels.
308void FELighting::platformApplyGenericPaint(const LightingData& data, const LightSource::PaintingData& paintingData, int startY, int endY)
309{
310 // Make sure startY is > 0 since we read from the previous row in the loop.
311 ASSERT(startY);
312 ASSERT(endY > startY);
313
314 for (int y = startY; y < endY; ++y) {
315 int rowStartOffset = y * data.widthMultipliedByPixelSize;
316 int previousRowStart = rowStartOffset - data.widthMultipliedByPixelSize;
317 int nextRowStart = rowStartOffset + data.widthMultipliedByPixelSize;
318
319 // alphaWindow is a local cache of alpha values.
320 // Fill the two right columns putting the left edge value in the center column.
321 // For each pixel, we shift each row left then fill the right column.
322 AlphaWindow alphaWindow;
323 alphaWindow.setTop(data.pixels->item(previousRowStart + cAlphaChannelOffset));
324 alphaWindow.setTopRight(data.pixels->item(previousRowStart + cPixelSize + cAlphaChannelOffset));
325
326 alphaWindow.setCenter(data.pixels->item(rowStartOffset + cAlphaChannelOffset));
327 alphaWindow.setRight(data.pixels->item(rowStartOffset + cPixelSize + cAlphaChannelOffset));
328
329 alphaWindow.setBottom(data.pixels->item(nextRowStart + cAlphaChannelOffset));
330 alphaWindow.setBottomRight(data.pixels->item(nextRowStart + cPixelSize + cAlphaChannelOffset));
331
332 int offset = rowStartOffset + cPixelSize;
333 for (int x = 1; x < data.widthDecreasedByOne; ++x, offset += cPixelSize) {
334 alphaWindow.shift();
335 setPixelInternal(offset, data, paintingData, x, y, cFactor1div4, cFactor1div4, data.interiorNormal(offset, alphaWindow), alphaWindow.center());
336 }
337 }
338}

345void FELighting::platformApplyGeneric(const LightingData& data, const LightSource::PaintingData& paintingData)
346{
347 unsigned rowsToProcess = data.heightDecreasedByOne - 1;
348 unsigned maxNumThreads = rowsToProcess / 8;
349 unsigned optimalThreadNumber = std::min<unsigned>(((data.widthDecreasedByOne - 1) * rowsToProcess) / s_minimalRectDimension, maxNumThreads);
350 if (optimalThreadNumber > 1) {
351 // Initialize parallel jobs
352 WTF::ParallelJobs<PlatformApplyGenericParameters> parallelJobs(&platformApplyGenericWorker, optimalThreadNumber);
353
354 // Fill the parameter array
355 int job = parallelJobs.numberOfJobs();
356 if (job > 1) {
357 // Split the job into "yStep"-sized jobs but there a few jobs that need to be slightly larger since
358 // yStep * jobs < total size. These extras are handled by the remainder "jobsWithExtra".
359 const int yStep = rowsToProcess / job;
360 const int jobsWithExtra = rowsToProcess % job;
361
362 int yStart = 1;
363 for (--job; job >= 0; --job) {
364 PlatformApplyGenericParameters& params = parallelJobs.parameter(job);
365 params.filter = this;
366 params.data = data;
367 params.paintingData = paintingData;
368 params.yStart = yStart;
369 yStart += job < jobsWithExtra ? yStep + 1 : yStep;
370 params.yEnd = yStart;
371 }
372 parallelJobs.execute();
373 return;
374 }
375 // Fallback to single threaded mode.
376 }
377
378 platformApplyGenericPaint(data, paintingData, 1, data.heightDecreasedByOne);
379}

391bool FELighting::drawLighting(Uint8ClampedArray& pixels, int width, int height)
392{
393 LightSource::PaintingData paintingData;
394 LightingData data;
395
396 // FIXME: do something if width or height (or both) is 1 pixel.
397 // The W3 spec does not define this case. Now the filter just returns.
398 if (width <= 2 || height <= 2)
399 return false;
400
401 data.pixels = &pixels;
402 data.surfaceScale = m_surfaceScale / 255.0f;
403 data.widthMultipliedByPixelSize = width * cPixelSize;
404 data.widthDecreasedByOne = width - 1;
405 data.heightDecreasedByOne = height - 1;
406
407 auto [r, g, b, a] = m_lightingColor.toColorComponentsInColorSpace(operatingColorSpace());
408 paintingData.initialLightingData.colorVector = FloatPoint3D(r, g, b);
409
410 m_lightSource->initPaintingData(*this, paintingData);
411
412 // Top left.
413 int offset = 0;
414 setPixel(offset, data, paintingData, 0, 0, cFactor2div3, cFactor2div3, data.topLeftNormal(offset));
415
416 // Top right.
417 offset = data.widthMultipliedByPixelSize - cPixelSize;
418 setPixel(offset, data, paintingData, data.widthDecreasedByOne, 0, cFactor2div3, cFactor2div3, data.topRightNormal(offset));
419
420 // Bottom left.
421 offset = data.heightDecreasedByOne * data.widthMultipliedByPixelSize;
422 setPixel(offset, data, paintingData, 0, data.heightDecreasedByOne, cFactor2div3, cFactor2div3, data.bottomLeftNormal(offset));
423
424 // Bottom right.
425 offset = height * data.widthMultipliedByPixelSize - cPixelSize;
426 setPixel(offset, data, paintingData, data.widthDecreasedByOne, data.heightDecreasedByOne, cFactor2div3, cFactor2div3, data.bottomRightNormal(offset));
427
428 if (width >= 3) {
429 // Top row.
430 offset = cPixelSize;
431 for (int x = 1; x < data.widthDecreasedByOne; ++x, offset += cPixelSize)
432 setPixel(offset, data, paintingData, x, 0, cFactor1div3, cFactor1div2, data.topRowNormal(offset));
433
434 // Bottom row.
435 offset = data.heightDecreasedByOne * data.widthMultipliedByPixelSize + cPixelSize;
436 for (int x = 1; x < data.widthDecreasedByOne; ++x, offset += cPixelSize)
437 setPixel(offset, data, paintingData, x, data.heightDecreasedByOne, cFactor1div3, cFactor1div2, data.bottomRowNormal(offset));
438 }
439
440 if (height >= 3) {
441 // Left column.
442 offset = data.widthMultipliedByPixelSize;
443 for (int y = 1; y < data.heightDecreasedByOne; ++y, offset += data.widthMultipliedByPixelSize)
444 setPixel(offset, data, paintingData, 0, y, cFactor1div2, cFactor1div3, data.leftColumnNormal(offset));
445
446 // Right column.
447 offset = 2 * data.widthMultipliedByPixelSize - cPixelSize;
448 for (int y = 1; y < data.heightDecreasedByOne; ++y, offset += data.widthMultipliedByPixelSize)
449 setPixel(offset, data, paintingData, data.widthDecreasedByOne, y, cFactor1div2, cFactor1div3, data.rightColumnNormal(offset));
450 }
451
452 if (width >= 3 && height >= 3) {
453 // Interior pixels.
454 platformApply(data, paintingData);
455 }
456
457 int lastPixel = data.widthMultipliedByPixelSize * height;
458 if (m_lightingType == DiffuseLighting) {
459 for (int i = cAlphaChannelOffset; i < lastPixel; i += cPixelSize)
460 data.pixels->set(i, cOpaqueAlpha);
461 } else {
462 for (int i = 0; i < lastPixel; i += cPixelSize) {
463 uint8_t a1 = data.pixels->item(i);
464 uint8_t a2 = data.pixels->item(i + 1);
465 uint8_t a3 = data.pixels->item(i + 2);
466 // alpha set to set to max(a1, a2, a3)
467 data.pixels->set(i + 3, a1 >= a2 ? (a1 >= a3 ? a1 : a3) : (a2 >= a3 ? a2 : a3));
468 }
469 }
470
471 return true;
472}

474void FELighting::platformApplySoftware()
475{
476 FilterEffect* in = inputEffect(0);
477
478 auto& destinationPixelBuffer = createPremultipliedImageResult();
479 if (!destinationPixelBuffer)
480 return;
481
482 auto& destinationPixelArray = destinationPixelBuffer->data();
483
484 setIsAlphaImage(false);
485
486 IntRect effectDrawingRect = requestedRegionOfInputPixelBuffer(in->absolutePaintRect());
487 in->copyPremultipliedResult(destinationPixelArray, effectDrawingRect, operatingColorSpace());
488 // FIXME: support kernelUnitLengths other than (1,1). The issue here is that the W3
489 // standard has no test case for them, and other browsers (like Firefox) has strange
490 // output for various kernelUnitLengths, and I am not sure they are reliable.
491 // Anyway, feConvolveMatrix should also use the implementation
492
493 IntSize absolutePaintSize = absolutePaintRect().size();
494 drawLighting(destinationPixelArray, absolutePaintSize.width(), absolutePaintSize.height());
495}
