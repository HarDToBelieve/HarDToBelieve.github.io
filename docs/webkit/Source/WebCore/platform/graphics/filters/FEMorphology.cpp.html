75void FEMorphology::determineAbsolutePaintRect()
76{
77 FloatRect paintRect = inputEffect(0)->absolutePaintRect();
78 Filter& filter = this->filter();
79 paintRect.inflate(filter.scaledByFilterResolution({ m_radiusX, m_radiusY }));
80 if (clipsToBounds())
81 paintRect.intersect(maxEffectRect());
82 else
83 paintRect.unite(maxEffectRect());
84 setAbsolutePaintRect(enclosingIntRect(paintRect));
85}

87static inline int pixelArrayIndex(int x, int y, int width)
88{
89 return (y * width + x) * 4;
90}

92inline ColorComponents<uint8_t, 4> makeColorComponentsfromPixelValue(PackedColor::RGBA pixel)
93{
94 return asColorComponents(asSRGBA(pixel));
95}

97inline PackedColor::RGBA makePixelValueFromColorComponents(const ColorComponents<uint8_t, 4>& components)
98{
99 return PackedColor::RGBA { makeFromComponents<SRGBA<uint8_t>>(components) };

102template<MorphologyOperatorType type>
103ALWAYS_INLINE ColorComponents<uint8_t, 4> minOrMax(const ColorComponents<uint8_t, 4>& a, const ColorComponents<uint8_t, 4>& b)
104{
105 if (type == FEMORPHOLOGY_OPERATOR_ERODE)
106 return perComponentMin(a, b);
107
108 return perComponentMax(a, b);
109}
110
111template<MorphologyOperatorType type>
112ALWAYS_INLINE ColorComponents<uint8_t, 4> columnExtremum(const Uint8ClampedArray& srcPixelArray, int x, int yStart, int yEnd, int width)
113{
114 auto extremum = makeColorComponentsfromPixelValue(PackedColor::RGBA { *reinterpret_cast<const unsigned*>(srcPixelArray.data() + pixelArrayIndex(x, yStart, width)) });
115
116 for (int y = yStart + 1; y < yEnd; ++y) {
117 auto pixel = makeColorComponentsfromPixelValue(PackedColor::RGBA { *reinterpret_cast<const unsigned*>(srcPixelArray.data() + pixelArrayIndex(x, y, width)) });
118 extremum = minOrMax<type>(extremum, pixel);
119 }
120 return extremum;
121}
122
123ALWAYS_INLINE ColorComponents<uint8_t, 4> columnExtremum(const Uint8ClampedArray& srcPixelArray, int x, int yStart, int yEnd, int width, MorphologyOperatorType type)
124{
125 if (type == FEMORPHOLOGY_OPERATOR_ERODE)
126 return columnExtremum<FEMORPHOLOGY_OPERATOR_ERODE>(srcPixelArray, x, yStart, yEnd, width);
127
128 return columnExtremum<FEMORPHOLOGY_OPERATOR_DILATE>(srcPixelArray, x, yStart, yEnd, width);
129}
130
131using ColumnExtrema = Vector<ColorComponents<uint8_t, 4>, 16>;
132
133template<MorphologyOperatorType type>
134ALWAYS_INLINE ColorComponents<uint8_t, 4> kernelExtremum(const ColumnExtrema& kernel)
135{
136 auto extremum = kernel[0];
137 for (size_t i = 1; i < kernel.size(); ++i)
138 extremum = minOrMax<type>(extremum, kernel[i]);
139
140 return extremum;
141}
142
143ALWAYS_INLINE ColorComponents<uint8_t, 4> kernelExtremum(const ColumnExtrema& kernel, MorphologyOperatorType type)
144{
145 if (type == FEMORPHOLOGY_OPERATOR_ERODE)
146 return kernelExtremum<FEMORPHOLOGY_OPERATOR_ERODE>(kernel);
147
148 return kernelExtremum<FEMORPHOLOGY_OPERATOR_DILATE>(kernel);
149}

151void FEMorphology::platformApplyGeneric(const PaintingData& paintingData, int startY, int endY)
152{
153 ASSERT(endY > startY);
154
155 const auto& srcPixelArray = *paintingData.srcPixelArray;
156 auto& dstPixelArray = *paintingData.dstPixelArray;
157
158 const int radiusX = paintingData.radiusX;
159 const int radiusY = paintingData.radiusY;
160 const int width = paintingData.width;
161 const int height = paintingData.height;
162
163 ASSERT(radiusX <= width || radiusY <= height);
164 ASSERT(startY >= 0 && endY <= height && startY < endY);
165
166 ColumnExtrema extrema;
167 extrema.reserveInitialCapacity(2 * radiusX + 1);
168
169 for (int y = startY; y < endY; ++y) {
170 int yRadiusStart = std::max(0, y - radiusY);
171 int yRadiusEnd = std::min(height, y + radiusY + 1);
172
173 extrema.shrink(0);
174
175 // We start at the left edge, so compute extreme for the radiusX columns.
176 for (int x = 0; x < radiusX; ++x)
177 extrema.append(columnExtremum(srcPixelArray, x, yRadiusStart, yRadiusEnd, width, m_type));
178
179 // Kernel is filled, get extrema of next column
180 for (int x = 0; x < width; ++x) {
181 if (x < width - radiusX)
182 extrema.append(columnExtremum(srcPixelArray, x + radiusX, yRadiusStart, yRadiusEnd, width, m_type));
183
184 if (x > radiusX)
185 extrema.remove(0);
186
187 unsigned* destPixel = reinterpret_cast<unsigned*>(dstPixelArray.data() + pixelArrayIndex(x, y, width));
188 *destPixel = makePixelValueFromColorComponents(kernelExtremum(extrema, m_type)).value;
189 }
190 }
191}

198void FEMorphology::platformApply(const PaintingData& paintingData)
199{
200 // Empirically, runtime is approximately linear over reasonable kernel sizes with a slope of about 0.65.
201 float kernelFactor = sqrt(paintingData.radiusX * paintingData.radiusY) * 0.65;
202
203 static const int minimalArea = (160 * 160); // Empirical data limit for parallel jobs
204
205 unsigned maxNumThreads = paintingData.height / 8;
206 unsigned optimalThreadNumber = std::min<unsigned>((paintingData.width * paintingData.height * kernelFactor) / minimalArea, maxNumThreads);
207 if (optimalThreadNumber > 1) {
208 WTF::ParallelJobs<PlatformApplyParameters> parallelJobs(&WebCore::FEMorphology::platformApplyWorker, optimalThreadNumber);
209 auto numOfThreads = parallelJobs.numberOfJobs();
210 if (numOfThreads > 1) {
211 // Split the job into "jobSize"-sized jobs but there a few jobs that need to be slightly larger since
212 // jobSize * jobs < total size. These extras are handled by the remainder "jobsWithExtra".
213 int jobSize = paintingData.height / numOfThreads;
214 int jobsWithExtra = paintingData.height % numOfThreads;
215 int currentY = 0;
216 for (int job = numOfThreads - 1; job >= 0; --job) {
217 PlatformApplyParameters& param = parallelJobs.parameter(job);
218 param.filter = this;
219 param.startY = currentY;
220 currentY += job < jobsWithExtra ? jobSize + 1 : jobSize;
221 param.endY = currentY;
222 param.paintingData = &paintingData;
223 }
224 parallelJobs.execute();
225 return;
226 }
227 // Fallback to single thread model
228 }
229
230 platformApplyGeneric(paintingData, 0, paintingData.height);
231}

233bool FEMorphology::platformApplyDegenerate(Uint8ClampedArray& dstPixelArray, const IntRect& imageRect, int radiusX, int radiusY)
234{
235 if (radiusX < 0 || radiusY < 0 || (!radiusX && !radiusY)) {
236 FilterEffect* in = inputEffect(0);
237 in->copyPremultipliedResult(dstPixelArray, imageRect, operatingColorSpace());
238 return true;
239 }
240 return false;
241}

243void FEMorphology::platformApplySoftware()
244{
245 FilterEffect* in = inputEffect(0);
246
247 auto& destinationPixelBuffer = createPremultipliedImageResult();
248 if (!destinationPixelBuffer)
249 return;
250
251 auto& destinationPixelArray = destinationPixelBuffer->data();
252
253 setIsAlphaImage(in->isAlphaImage());
254
255 IntRect effectDrawingRect = requestedRegionOfInputPixelBuffer(in->absolutePaintRect());
256
257 IntSize radius = flooredIntSize(FloatSize(m_radiusX, m_radiusY));
258 if (platformApplyDegenerate(destinationPixelArray, effectDrawingRect, radius.width(), radius.height()))
259 return;
260
261 Filter& filter = this->filter();
262 auto sourcePixelArray = in->premultipliedResult(effectDrawingRect, operatingColorSpace());
263 if (!sourcePixelArray)
264 return;
265
266 radius = flooredIntSize(filter.scaledByFilterResolution({ m_radiusX, m_radiusY }));
267 int radiusX = std::min(effectDrawingRect.width() - 1, radius.width());
268 int radiusY = std::min(effectDrawingRect.height() - 1, radius.height());
269
270 if (platformApplyDegenerate(destinationPixelArray, effectDrawingRect, radiusX, radiusY))
271 return;
272
273 PaintingData paintingData;
274 paintingData.srcPixelArray = sourcePixelArray.get();
275 paintingData.dstPixelArray = &destinationPixelArray;
276 paintingData.width = ceilf(effectDrawingRect.width() * filter.filterScale());
277 paintingData.height = ceilf(effectDrawingRect.height() * filter.filterScale());
278 paintingData.radiusX = ceilf(radiusX * filter.filterScale());
279 paintingData.radiusY = ceilf(radiusY * filter.filterScale());
280
281 platformApply(paintingData);
282}

284static TextStream& operator<<(TextStream& ts, const MorphologyOperatorType& type)
285{
286 switch (type) {
287 case FEMORPHOLOGY_OPERATOR_UNKNOWN:
288 ts << "UNKNOWN";
