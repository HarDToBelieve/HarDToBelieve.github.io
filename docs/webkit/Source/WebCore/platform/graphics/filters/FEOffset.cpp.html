52 m_dy = dy;

54
55void FEOffset::determineAbsolutePaintRect()
56{
57 FloatRect paintRect = inputEffect(0)->absolutePaintRect();
58 Filter& filter = this->filter();
59 paintRect.move(filter.scaledByFilterResolution({ m_dx, m_dy }));
60 if (clipsToBounds())
61 paintRect.intersect(maxEffectRect());
62 else
63 paintRect.unite(maxEffectRect());
64 setAbsolutePaintRect(enclosingIntRect(paintRect));
65}
66
67void FEOffset::platformApplySoftware()
68{
69 FilterEffect* in = inputEffect(0);
70
71 ImageBuffer* resultImage = createImageBufferResult();
72 ImageBuffer* inBuffer = in->imageBufferResult();
73 if (!resultImage || !inBuffer)
74 return;
75
76 setIsAlphaImage(in->isAlphaImage());
77
78 FloatRect drawingRegion = drawingRegionOfInputImage(in->absolutePaintRect());
79 Filter& filter = this->filter();
80 drawingRegion.move(filter.scaledByFilterResolution({ m_dx, m_dy }));
81 resultImage->context().drawImageBuffer(*inBuffer, drawingRegion);
82}

84TextStream& FEOffset::externalRepresentation(TextStream& ts, RepresentationType representation) const
85{
86 ts << indent << "[feOffset";
87 FilterEffect::externalRepresentation(ts, representation);
88 ts << " dx=\"" << dx() << "\" dy=\"" << dy() << "\"]\n";
89
90 TextStream::IndentScope indentScope(ts);
91 inputEffect(0)->externalRepresentation(ts, representation);
92 return ts;
93}
