56void FilterEffect::determineAbsolutePaintRect()
57{
58 m_absolutePaintRect = IntRect();
59 for (auto& effect : m_inputEffects)
60 m_absolutePaintRect.unite(effect->absolutePaintRect());
61 clipAbsolutePaintRect();
62}

64void FilterEffect::clipAbsolutePaintRect()
65{
66 // Filters in SVG clip to primitive subregion, while CSS doesn't.
67 if (m_clipsToBounds)
68 m_absolutePaintRect.intersect(enclosingIntRect(m_maxEffectRect));
69 else
70 m_absolutePaintRect.unite(enclosingIntRect(m_maxEffectRect));
71}

99FloatRect FilterEffect::determineFilterPrimitiveSubregion()
100{
101 // FETile, FETurbulence, FEFlood don't have input effects, take the filter region as unite rect.
102 FloatRect subregion;
103 if (unsigned numberOfInputEffects = inputEffects().size()) {
104 subregion = inputEffect(0)->determineFilterPrimitiveSubregion();
105 for (unsigned i = 1; i < numberOfInputEffects; ++i) {
106 auto inputPrimitiveSubregion = inputEffect(i)->determineFilterPrimitiveSubregion();
107 subregion.unite(inputPrimitiveSubregion);
108 }
109 } else
110 subregion = m_filter.filterRegionInUserSpace();
111
112 // After calling determineFilterPrimitiveSubregion on the target effect, reset the subregion again for <feTile>.
113 if (filterEffectType() == FilterEffectTypeTile)
114 subregion = m_filter.filterRegionInUserSpace();
115
116 auto boundaries = effectBoundaries();
117 if (hasX())
118 subregion.setX(boundaries.x());
119 if (hasY())
120 subregion.setY(boundaries.y());
121 if (hasWidth())
122 subregion.setWidth(boundaries.width());
123 if (hasHeight())
124 subregion.setHeight(boundaries.height());
125
126 setFilterPrimitiveSubregion(subregion);
127
128 auto absoluteSubregion = m_filter.absoluteTransform().mapRect(subregion);
129 auto filterResolution = m_filter.filterResolution();
130 absoluteSubregion.scale(filterResolution);
131 // Save this before clipping so we can use it to map lighting points from user space to buffer coordinates.
132 setUnclippedAbsoluteSubregion(absoluteSubregion);
133
134 // Clip every filter effect to the filter region.
135 auto absoluteScaledFilterRegion = m_filter.filterRegion();
136 absoluteScaledFilterRegion.scale(filterResolution);
137 absoluteSubregion.intersect(absoluteScaledFilterRegion);
138
139 setMaxEffectRect(absoluteSubregion);
140 return subregion;
141}

166void FilterEffect::apply()
167{
168 if (hasResult())
169 return;
170 unsigned size = m_inputEffects.size();
171 for (unsigned i = 0; i < size; ++i) {
172 FilterEffect* in = m_inputEffects.at(i).get();
173 in->apply();
174 if (!in->hasResult())
175 return;
176
177 // Convert input results to the current effect's color space.
178 transformResultColorSpace(in, i);
179 }
180
181 determineAbsolutePaintRect();
182 setResultColorSpace(m_operatingColorSpace);
183
184 LOG_WITH_STREAM(Filters, stream << "FilterEffect " << filterName() << " " << this << " apply():\n filterPrimitiveSubregion " << m_filterPrimitiveSubregion << "\n effectBoundaries " << m_effectBoundaries << "\n absoluteUnclippedSubregion " << m_absoluteUnclippedSubregion << "\n absolutePaintRect " << m_absolutePaintRect << "\n maxEffectRect " << m_maxEffectRect << "\n filter scale " << m_filter.filterScale() << "\n filter resolution " << m_filter.filterResolution());
185
186 if (m_absolutePaintRect.isEmpty() || ImageBuffer::sizeNeedsClamping(m_absolutePaintRect.size()))
187 return;
188
189 if (requiresValidPreMultipliedPixels()) {
190 for (unsigned i = 0; i < size; ++i)
191 inputEffect(i)->correctFilterResultIfNeeded();
192 }
193
194 // Add platform specific apply functions here and return earlier.
195 platformApplySoftware();
196}

198void FilterEffect::forceValidPreMultipliedPixels()
199{
200 // Must operate on pre-multiplied results; other formats cannot have invalid pixels.
201 if (!m_premultipliedImageResult)
202 return;
203
204 auto& imageArray = m_premultipliedImageResult->data();
205 uint8_t* pixelData = imageArray.data();
206 int pixelArrayLength = imageArray.length();
207
208 // We must have four bytes per pixel, and complete pixels
209 ASSERT(!(pixelArrayLength % 4));
210
211#if HAVE(ARM_NEON_INTRINSICS)
212 if (pixelArrayLength >= 64) {
213 uint8_t* lastPixel = pixelData + (pixelArrayLength & ~0x3f);
214 do {
215 // Increments pixelData by 64.
216 uint8x16x4_t sixteenPixels = vld4q_u8(pixelData);
217 sixteenPixels.val[0] = vminq_u8(sixteenPixels.val[0], sixteenPixels.val[3]);
218 sixteenPixels.val[1] = vminq_u8(sixteenPixels.val[1], sixteenPixels.val[3]);
219 sixteenPixels.val[2] = vminq_u8(sixteenPixels.val[2], sixteenPixels.val[3]);
220 vst4q_u8(pixelData, sixteenPixels);
221 pixelData += 64;
222 } while (pixelData < lastPixel);
223
224 pixelArrayLength &= 0x3f;
225 if (!pixelArrayLength)
226 return;
227 }
228#endif
229
230 int numPixels = pixelArrayLength / 4;
231
232 // Iterate over each pixel, checking alpha and adjusting color components if necessary
233 while (--numPixels >= 0) {
234 // Alpha is the 4th byte in a pixel
235 uint8_t a = *(pixelData + 3);
236 // Clamp each component to alpha, and increment the pixel location
237 for (int i = 0; i < 3; ++i) {
238 if (*pixelData > a)
239 *pixelData = a;
240 ++pixelData;
241 }
242 // Increment for alpha
243 ++pixelData;
244 }
245}

288RefPtr<Uint8ClampedArray> FilterEffect::unmultipliedResult(const IntRect& rect, std::optional<DestinationColorSpace> colorSpace)
289{
290 IntSize scaledSize(rect.size());
291 ASSERT(!ImageBuffer::sizeNeedsClamping(scaledSize));
292 scaledSize.scale(m_filter.filterScale());
293 auto checkedArea = scaledSize.area<RecordOverflow>() * 4;
294 if (checkedArea.hasOverflowed())
295 return nullptr;
296 auto pixelArray = Uint8ClampedArray::tryCreateUninitialized(checkedArea);
297 if (!pixelArray)
298 return nullptr;
299 copyUnmultipliedResult(*pixelArray, rect, colorSpace);
300 return pixelArray;
301}

318void FilterEffect::copyImageBytes(const Uint8ClampedArray& source, Uint8ClampedArray& destination, const IntRect& rect) const
319{
320 IntRect scaledRect(rect);
321 scaledRect.scale(m_filter.filterScale());
322 IntSize scaledPaintSize(m_absolutePaintRect.size());
323 scaledPaintSize.scale(m_filter.filterScale());
324
325 // Initialize the destination to transparent black, if not entirely covered by the source.
326 if (scaledRect.x() < 0 || scaledRect.y() < 0 || scaledRect.maxX() > scaledPaintSize.width() || scaledRect.maxY() > scaledPaintSize.height())
327 memset(destination.data(), 0, destination.length());
328
329 // Early return if the rect does not intersect with the source.
330 if (scaledRect.maxX() <= 0 || scaledRect.maxY() <= 0 || scaledRect.x() >= scaledPaintSize.width() || scaledRect.y() >= scaledPaintSize.height())
331 return;
332
333 int xOrigin = scaledRect.x();
334 int xDest = 0;
335 if (xOrigin < 0) {
336 xDest = -xOrigin;
337 xOrigin = 0;
338 }
339 int xEnd = scaledRect.maxX();
340 if (xEnd > scaledPaintSize.width())
341 xEnd = scaledPaintSize.width();
342
343 int yOrigin = scaledRect.y();
344 int yDest = 0;
345 if (yOrigin < 0) {
346 yDest = -yOrigin;
347 yOrigin = 0;
348 }
349 int yEnd = scaledRect.maxY();
350 if (yEnd > scaledPaintSize.height())
351 yEnd = scaledPaintSize.height();
352
353 int size = (xEnd - xOrigin) * 4;
354 int destinationScanline = scaledRect.width() * 4;
355 int sourceScanline = scaledPaintSize.width() * 4;
356 uint8_t* destinationPixel = destination.data() + ((yDest * scaledRect.width()) + xDest) * 4;
357 const uint8_t* sourcePixel = source.data() + ((yOrigin * scaledPaintSize.width()) + xOrigin) * 4;
358
359 while (yOrigin < yEnd) {
360 memcpy(destinationPixel, sourcePixel, size);
361 destinationPixel += destinationScanline;
362 sourcePixel += sourceScanline;
363 ++yOrigin;
364 }
365}

367static void copyPremultiplyingAlpha(const Uint8ClampedArray& source, Uint8ClampedArray& destination, const IntSize& inputSize)
368{
369#if USE(ACCELERATE)
370 size_t rowBytes = inputSize.width() * 4;
371
372 vImage_Buffer src;
373 src.width = inputSize.width();
374 src.height = inputSize.height();
375 src.rowBytes = rowBytes;
376 src.data = reinterpret_cast<void*>(source.data());
377
378 vImage_Buffer dest;
379 dest.width = inputSize.width();
380 dest.height = inputSize.height();
381 dest.rowBytes = rowBytes;
382 dest.data = reinterpret_cast<void*>(destination.data());
383
384 vImagePremultiplyData_RGBA8888(&src, &dest, kvImageNoFlags);
385#else
386 const uint8_t* sourceComponent = source.data();
387 const uint8_t* end = sourceComponent + (inputSize.area() * 4).value();
388 uint8_t* destinationComponent = destination.data();
389
390 while (sourceComponent < end) {
391 int alpha = sourceComponent[3];
392 destinationComponent[0] = static_cast<int>(sourceComponent[0]) * alpha / 255;
393 destinationComponent[1] = static_cast<int>(sourceComponent[1]) * alpha / 255;
394 destinationComponent[2] = static_cast<int>(sourceComponent[2]) * alpha / 255;
395 destinationComponent[3] = alpha;
396 sourceComponent += 4;
397 destinationComponent += 4;
398 }
399#endif
400}

402static void copyUnpremultiplyingAlpha(const Uint8ClampedArray& source, Uint8ClampedArray& destination, const IntSize& inputSize)
403{
404#if USE(ACCELERATE)
405 size_t rowBytes = inputSize.width() * 4;
406
407 vImage_Buffer src;
408 src.width = inputSize.width();
409 src.height = inputSize.height();
410 src.rowBytes = rowBytes;
411 src.data = reinterpret_cast<void*>(source.data());
412
413 vImage_Buffer dest;
414 dest.width = inputSize.width();
415 dest.height = inputSize.height();
416 dest.rowBytes = rowBytes;
417 dest.data = reinterpret_cast<void*>(destination.data());
418
419 vImageUnpremultiplyData_RGBA8888(&src, &dest, kvImageNoFlags);
420#else
421 const uint8_t* sourceComponent = source.data();
422 const uint8_t* end = sourceComponent + (inputSize.area() * 4).value();
423 uint8_t* destinationComponent = destination.data();
424 while (sourceComponent < end) {
425 int alpha = sourceComponent[3];
426 if (alpha) {
427 destinationComponent[0] = static_cast<int>(sourceComponent[0]) * 255 / alpha;
428 destinationComponent[1] = static_cast<int>(sourceComponent[1]) * 255 / alpha;
429 destinationComponent[2] = static_cast<int>(sourceComponent[2]) * 255 / alpha;
430 } else {
431 destinationComponent[0] = 0;
432 destinationComponent[1] = 0;
433 destinationComponent[2] = 0;
434 }
435 destinationComponent[3] = alpha;
436 sourceComponent += 4;
437 destinationComponent += 4;
438 }
439#endif
440}

442std::optional<PixelBuffer> FilterEffect::convertPixelBufferToColorSpace(const DestinationColorSpace& targetColorSpace, PixelBuffer& pixelBuffer)
443{
444 // FIXME: Using an ImageBuffer to perform the color space conversion is unnecessary. We can do it directly.
445
446 IntRect destinationRect(IntPoint(), pixelBuffer.size());
447 destinationRect.scale(1 / m_filter.filterScale());
448 FloatSize clampedSize = ImageBuffer::clampedSize(destinationRect.size());
449 // Create an ImageBuffer to store incoming PixelBuffer
450 auto buffer = ImageBuffer::create(clampedSize, m_filter.renderingMode(), m_filter.filterScale(), operatingColorSpace(), PixelFormat::BGRA8);
451 if (!buffer)
452 return std::nullopt;
453 buffer->putPixelBuffer(pixelBuffer, destinationRect);
454 return convertImageBufferToColorSpace(targetColorSpace, *buffer, destinationRect, pixelBuffer.format().alphaFormat);
455}

457std::optional<PixelBuffer> FilterEffect::convertImageBufferToColorSpace(const DestinationColorSpace& targetColorSpace, ImageBuffer& inputBuffer, const IntRect& rect, AlphaPremultiplication outputAlphaFormat)
458{
459 // FIXME: This can be done more directly using PixelBufferConversion.
460
461 FloatSize clampedSize = ImageBuffer::clampedSize(rect.size());
462
463 // Create an ImageBuffer with the correct color space and utilize CG to handle color space conversion
464 auto convertedBuffer = ImageBuffer::create(clampedSize, m_filter.renderingMode(), m_filter.filterScale(), targetColorSpace, PixelFormat::BGRA8);
465 if (!convertedBuffer)
466 return std::nullopt;
467
468 // Color space conversion happens internally when drawing from one image buffer to another
469 convertedBuffer->context().drawImageBuffer(inputBuffer, rect);
470
471 PixelBufferFormat format { outputAlphaFormat, PixelFormat::RGBA8, targetColorSpace };
472 return convertedBuffer->getPixelBuffer(format, rect);
473}

494void FilterEffect::copyUnmultipliedResult(Uint8ClampedArray& destination, const IntRect& rect, std::optional<DestinationColorSpace> colorSpace)
495{
496 ASSERT(hasResult());
497
498 LOG_WITH_STREAM(Filters, stream << "FilterEffect " << filterName() << " " << this << " copyUnmultipliedResult(). Existing image buffer " << m_imageBufferResult.get() << " m_premultipliedImageResult " << m_premultipliedImageResult << " m_unmultipliedImageResult " << m_unmultipliedImageResult);
499
500 if (!m_unmultipliedImageResult) {
501 // We prefer a conversion from the image buffer.
502 if (m_imageBufferResult) {
503 if (requiresPixelBufferColorSpaceConversion(colorSpace)) {
504 copyConvertedImageBufferToDestination(destination, *colorSpace, AlphaPremultiplication::Unpremultiplied, rect);
505 return;
506 }
507
508 ASSERT(m_imageBufferResult->colorSpace() == m_resultColorSpace);
509 PixelBufferFormat format { AlphaPremultiplication::Unpremultiplied, PixelFormat::RGBA8, m_resultColorSpace };
510 m_unmultipliedImageResult = m_imageBufferResult->getPixelBuffer(format, { IntPoint(), m_absolutePaintRect.size() });
511 if (!m_unmultipliedImageResult)
512 return;
513 } else {
514 IntSize inputSize(m_absolutePaintRect.size());
515 ASSERT(!ImageBuffer::sizeNeedsClamping(inputSize));
516 inputSize.scale(m_filter.filterScale());
517
518 ASSERT(m_premultipliedImageResult->format().colorSpace == m_resultColorSpace);
519 PixelBufferFormat format { AlphaPremultiplication::Unpremultiplied, PixelFormat::RGBA8, m_resultColorSpace };
520 m_unmultipliedImageResult = PixelBuffer::tryCreate(format, inputSize);
521 if (!m_unmultipliedImageResult)
522 return;
523 copyUnpremultiplyingAlpha(m_premultipliedImageResult->data(), m_unmultipliedImageResult->data(), inputSize);
524 }
525 }
526 if (requiresPixelBufferColorSpaceConversion(colorSpace)) {
527 copyConvertedPixelBufferToDestination(destination, *m_unmultipliedImageResult, *colorSpace, rect);
528 return;
529 }
530 copyImageBytes(m_unmultipliedImageResult->data(), destination, rect);
531}

625bool FilterEffect::requiresPixelBufferColorSpaceConversion(std::optional<DestinationColorSpace> destinationColorSpace)
626{
627#if USE(CG)
628 // This function determines whether we need the step of an extra color space conversion
629 // We only need extra color conversion when 1) color space is different in the input
630 // AND 2) the filter is manipulating raw pixels
631 return destinationColorSpace && resultColorSpace() != *destinationColorSpace;
632#else
633 // Additional color space conversion is not needed on non-CG
634 UNUSED_PARAM(destinationColorSpace);
635 return false;
636#endif
637}
638
639void FilterEffect::transformResultColorSpace(const DestinationColorSpace& destinationColorSpace)
640{
641#if USE(CG)
642 // CG handles color space adjustments internally.
643 UNUSED_PARAM(destinationColorSpace);
644#else
645 if (!hasResult() || destinationColorSpace == m_resultColorSpace)
646 return;
647
648 // FIXME: We can avoid this potentially unnecessary ImageBuffer conversion by adding
649 // color space transform support for the {pre,un}multiplied arrays.
650 imageBufferResult()->transformToColorSpace(destinationColorSpace);
651
652 m_resultColorSpace = destinationColorSpace;
653
654 m_unmultipliedImageResult = std::nullopt;
655 m_premultipliedImageResult = std::nullopt;
656#endif
657}
