49enum FilterEffectType {
50 FilterEffectTypeUnknown,

52 FilterEffectTypeTile,

56class FilterEffect : public RefCounted<FilterEffect> {
57public:
58 enum class Type : uint8_t {
59 Blend,
60 ColorMatrix,
61 ComponentTransfer,
62 Composite,
63 ConvolveMatrix,
64 DiffuseLighting,
65 DisplacementMap,
66 DropShadow,
67 Flood,
68 GaussianBlur,
69 Image,
70 Lighting,
71 Merge,
72 Morphology,
73 Offset,
74 SpecularLighting,
75 Tile,
76 Turbulence,
77 SourceAlpha,
78 SourceGraphic
79 };
80 virtual ~FilterEffect();
81
82 void clearResult();
83 void clearResultsRecursive();
84
85 ImageBuffer* imageBufferResult();
86 RefPtr<Uint8ClampedArray> unmultipliedResult(const IntRect&, std::optional<DestinationColorSpace> = std::nullopt);
87 RefPtr<Uint8ClampedArray> premultipliedResult(const IntRect&, std::optional<DestinationColorSpace> = std::nullopt);
88 void copyUnmultipliedResult(Uint8ClampedArray& destination, const IntRect&, std::optional<DestinationColorSpace> = std::nullopt);
89 void copyPremultipliedResult(Uint8ClampedArray& destination, const IntRect&, std::optional<DestinationColorSpace> = std::nullopt);
90 FilterEffectVector& inputEffects() { return m_inputEffects; }
91 FilterEffect* inputEffect(unsigned) const;
92 unsigned numberOfEffectInputs() const { return m_inputEffects.size(); }
93 unsigned totalNumberOfEffectInputs() const;
94
95 inline bool hasResult() const
96 {
97 // This function needs platform specific checks, if the memory managment is not done by FilterEffect.
98 return m_imageBufferResult
99 || m_unmultipliedImageResult
100 || m_premultipliedImageResult;
101 }
102
103 FloatRect drawingRegionOfInputImage(const IntRect&) const;
104 IntRect requestedRegionOfInputPixelBuffer(const IntRect&) const;
105
106 // Recurses on inputs.
107 FloatRect determineFilterPrimitiveSubregion();
108
109 // Solid black image with different alpha values.
110 bool isAlphaImage() const { return m_alphaImage; }
111 void setIsAlphaImage(bool alphaImage) { m_alphaImage = alphaImage; }
112
113 IntRect absolutePaintRect() const { return m_absolutePaintRect; }
114 void setAbsolutePaintRect(const IntRect& absolutePaintRect) { m_absolutePaintRect = absolutePaintRect; }
115
116 FloatRect maxEffectRect() const { return m_maxEffectRect; }
117 void setMaxEffectRect(const FloatRect& maxEffectRect) { m_maxEffectRect = maxEffectRect; }
118
119 void apply();
120
121 // Correct any invalid pixels, if necessary, in the result of a filter operation.
122 // This method is used to ensure valid pixel values on filter inputs and the final result.
123 // Only the arithmetic composite filter ever needs to perform correction.
124 virtual void correctFilterResultIfNeeded() { }
125
126 virtual void determineAbsolutePaintRect();
127
128 virtual FilterEffectType filterEffectType() const { return FilterEffectTypeUnknown; }
129
130 virtual IntOutsets outsets() const { return IntOutsets(); }
131
132 enum class RepresentationType { TestOutput, Debugging };
133 virtual WTF::TextStream& externalRepresentation(WTF::TextStream&, RepresentationType = RepresentationType::TestOutput) const;
134
135 // The following functions are SVG specific and will move to RenderSVGResourceFilterPrimitive.

137 bool hasX() const { return m_hasX; }
138 void setHasX(bool value) { m_hasX = value; }
139
140 bool hasY() const { return m_hasY; }
141 void setHasY(bool value) { m_hasY = value; }
142
143 bool hasWidth() const { return m_hasWidth; }
144 void setHasWidth(bool value) { m_hasWidth = value; }
145
146 bool hasHeight() const { return m_hasHeight; }
147 void setHasHeight(bool value) { m_hasHeight = value; }

149 FloatRect filterPrimitiveSubregion() const { return m_filterPrimitiveSubregion; }
150 void setFilterPrimitiveSubregion(const FloatRect& filterPrimitiveSubregion) { m_filterPrimitiveSubregion = filterPrimitiveSubregion; }
151
152 FloatRect effectBoundaries() const { return m_effectBoundaries; }
153 void setEffectBoundaries(const FloatRect& effectBoundaries) { m_effectBoundaries = effectBoundaries; }
154
155 void setUnclippedAbsoluteSubregion(const FloatRect& r) { m_absoluteUnclippedSubregion = r; }
156
157 FloatPoint mapPointFromUserSpaceToBuffer(FloatPoint) const;
158
159 Type filterEffectClassType() const { return m_filterEffectClassType; }

161 Filter& filter() { return m_filter; }
162 const Filter& filter() const { return m_filter; }
163
164 bool clipsToBounds() const { return m_clipsToBounds; }
165 void setClipsToBounds(bool value) { m_clipsToBounds = value; }

167 const DestinationColorSpace& operatingColorSpace() const { return m_operatingColorSpace; }

169 const DestinationColorSpace& resultColorSpace() const { return m_resultColorSpace; }

173 void transformResultColorSpace(const DestinationColorSpace&);

175 static Vector<float> normalizedFloats(const Vector<float>& values)
176 {
177 Vector<float> normalizedValues(values.size());
178 for (size_t i = 0; i < values.size(); ++i)
179 normalizedValues[i] = normalizedFloat(values[i]);
180 return normalizedValues;
181 }

184 FilterEffect(Filter&, Type);

188 ImageBuffer* createImageBufferResult();
189 std::optional<PixelBuffer>& createUnmultipliedImageResult();
190 std::optional<PixelBuffer>& createPremultipliedImageResult();

192 // Return true if the filter will only operate correctly on valid RGBA values, with

212 Filter& m_filter;
213 FilterEffectVector m_inputEffects;

215 RefPtr<ImageBuffer> m_imageBufferResult;
216 std::optional<PixelBuffer> m_unmultipliedImageResult;
217 std::optional<PixelBuffer> m_premultipliedImageResult;

219 IntRect m_absolutePaintRect;
220
221 // The maximum size of a filter primitive. In SVG this is the primitive subregion in absolute coordinate space.
222 // The absolute paint rect should never be bigger than m_maxEffectRect.
223 FloatRect m_maxEffectRect;
224
225 // The subregion of a filter primitive according to the SVG Filter specification in local coordinates.
226 // This is SVG specific and needs to move to RenderSVGResourceFilterPrimitive.
227 FloatRect m_filterPrimitiveSubregion;
228
229 // x, y, width and height of the actual SVGFE*Element. Is needed to determine the subregion of the
230 // filter primitive on a later step.
231 FloatRect m_effectBoundaries;
232
233 // filterPrimitiveSubregion mapped to absolute coordinates before clipping.
234 FloatRect m_absoluteUnclippedSubregion;
235
236 bool m_alphaImage { false };
237 bool m_hasX { false };
238 bool m_hasY { false };
239 bool m_hasWidth { false };
240 bool m_hasHeight { false };
241
242 // Should the effect clip to its primitive region, or expand to use the combined region of its inputs.
243 bool m_clipsToBounds { true };
244
245#if ENABLE(DESTINATION_COLOR_SPACE_LINEAR_SRGB)
246 DestinationColorSpace m_operatingColorSpace { DestinationColorSpace::LinearSRGB() };

248 DestinationColorSpace m_operatingColorSpace { DestinationColorSpace::SRGB() };
249#endif
250 DestinationColorSpace m_resultColorSpace { DestinationColorSpace::SRGB() };

252 const Type m_filterEffectClassType;
253};
