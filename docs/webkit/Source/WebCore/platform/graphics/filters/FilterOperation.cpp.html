90bool BasicColorMatrixFilterOperation::transformColor(SRGBA<float>& color) const
91{
92 switch (m_type) {
93 case GRAYSCALE: {
94 color = makeFromComponentsClamping<SRGBA<float>>(grayscaleColorMatrix(m_amount).transformedColorComponents(asColorComponents(color)));
95 return true;
96 }
97 case SEPIA: {
98 color = makeFromComponentsClamping<SRGBA<float>>(sepiaColorMatrix(m_amount).transformedColorComponents(asColorComponents(color)));
99 return true;
100 }
101 case HUE_ROTATE: {
102 color = makeFromComponentsClamping<SRGBA<float>>(hueRotateColorMatrix(m_amount).transformedColorComponents(asColorComponents(color)));
103 return true;
104 }
105 case SATURATE: {
106 color = makeFromComponentsClamping<SRGBA<float>>(saturationColorMatrix(m_amount).transformedColorComponents(asColorComponents(color)));
107 return true;
108 }
109 default:
110 ASSERT_NOT_REACHED();
111 return false;
112 }
113
114 return false;
115}

153bool BasicComponentTransferFilterOperation::transformColor(SRGBA<float>& color) const
154{
155 switch (m_type) {
156 case OPACITY:
157 color.alpha *= m_amount;
158 return true;
159 case INVERT: {
160 float oneMinusAmount = 1.0f - m_amount;
161 color = colorByModifingEachNonAlphaComponent(color, [&](float component) {
162 return 1.0f - (oneMinusAmount + component * (m_amount - oneMinusAmount));
163 });
164 return true;
165 }
166 case CONTRAST: {
167 float intercept = -(0.5f * m_amount) + 0.5f;
168 color = colorByModifingEachNonAlphaComponent(color, [&](float component) {
169 return std::clamp<float>(intercept + m_amount * component, 0.0f, 1.0f);
170 });
171 return true;
172 }
173 case BRIGHTNESS:
174 color = colorByModifingEachNonAlphaComponent(color, [&](float component) {
175 return std::clamp<float>(m_amount * component, 0.0f, 1.0f);
176 });
177 return true;
178 default:
179 ASSERT_NOT_REACHED();
180 return false;
181 }
182 return false;
183}

227// FIXME: This hueRotate code exists to allow InvertLightnessFilterOperation to perform hue rotation
228// on color values outside of the non-extended SRGB value range (0-1) to maintain the behavior of colors
229// prior to clamping being enforced. It should likely just use the existing hueRotateColorMatrix(amount)
230// in ColorMatrix.h
231static ColorComponents<float, 4> hueRotate(const ColorComponents<float, 4>& color, float amount)
232{
233 auto [r, g, b, alpha] = color;
234
235 auto [min, max] = std::minmax({ r, g, b });
236 float chroma = max - min;
237
238 float lightness = 0.5f * (max + min);
239 float saturation;
240 if (!chroma)
241 saturation = 0;
242 else if (lightness <= 0.5f)
243 saturation = (chroma / (max + min));
244 else
245 saturation = (chroma / (2.0f - (max + min)));
246
247 if (!saturation)
248 return { lightness, lightness, lightness, alpha };
249
250 float hue;
251 if (!chroma)
252 hue = 0;
253 else if (max == r)
254 hue = (60.0f * ((g - b) / chroma)) + 360.0f;
255 else if (max == g)
256 hue = (60.0f * ((b - r) / chroma)) + 120.0f;
257 else
258 hue = (60.0f * ((r - g) / chroma)) + 240.0f;
259
260 if (hue >= 360.0f)
261 hue -= 360.0f;
262
263 hue /= 360.0f;
264
265 // Perform rotation.
266 hue = std::fmod(hue + amount, 1.0f);
267
268 float temp2 = lightness <= 0.5f ? lightness * (1.0f + saturation) : lightness + saturation - lightness * saturation;
269 float temp1 = 2.0f * lightness - temp2;
270
271 hue *= 6.0f; // calcHue() wants hue in the 0-6 range.
272
273 // Hue is in the range 0-6, other args in 0-1.
274 auto calcHue = [](float temp1, float temp2, float hueVal) {
275 if (hueVal < 0.0f)
276 hueVal += 6.0f;
277 else if (hueVal >= 6.0f)
278 hueVal -= 6.0f;
279 if (hueVal < 1.0f)
280 return temp1 + (temp2 - temp1) * hueVal;
281 if (hueVal < 3.0f)
282 return temp2;
283 if (hueVal < 4.0f)
284 return temp1 + (temp2 - temp1) * (4.0f - hueVal);
285 return temp1;
286 };
287
288 return {
289 calcHue(temp1, temp2, hue + 2.0f),
290 calcHue(temp1, temp2, hue),
291 calcHue(temp1, temp2, hue - 2.0f),
292 alpha
293 };
294}

296bool InvertLightnessFilterOperation::transformColor(SRGBA<float>& color) const
297{
298 auto hueRotatedSRGBAComponents = hueRotate(asColorComponents(color), 0.5f);
299
300 // Apply the matrix. See rdar://problem/41146650 for how this matrix was derived.
301 constexpr ColorMatrix<5, 3> toDarkModeMatrix {
302 -0.770f, 0.059f, -0.089f, 0.0f, 1.0f,
303 0.030f, -0.741f, -0.089f, 0.0f, 1.0f,
304 0.030f, 0.059f, -0.890f, 0.0f, 1.0f
305 };
306 color = makeFromComponentsClamping<SRGBA<float>>(toDarkModeMatrix.transformedColorComponents(hueRotatedSRGBAComponents));
307 return true;
308}
309
310bool InvertLightnessFilterOperation::inverseTransformColor(SRGBA<float>& color) const
311{
312 // Apply the matrix.
313 constexpr ColorMatrix<5, 3> toLightModeMatrix {
314 -1.300f, -0.097f, 0.147f, 0.0f, 1.25f,
315 -0.049f, -1.347f, 0.146f, 0.0f, 1.25f,
316 -0.049f, -0.097f, -1.104f, 0.0f, 1.25f
317 };
318 auto convertedToLightModeComponents = toLightModeMatrix.transformedColorComponents(asColorComponents(color));
319
320 auto hueRotatedSRGBAComponents = hueRotate(convertedToLightModeComponents, 0.5f);
321
322 color = makeFromComponentsClamping<SRGBA<float>>(hueRotatedSRGBAComponents);
323 return true;
324}

357RefPtr<FilterOperation> DropShadowFilterOperation::blend(const FilterOperation* from, const BlendingContext& context, bool blendToPassthrough)
358{
359 if (from && !from->isSameType(*this))
360 return this;
361
362 if (blendToPassthrough)
363 return DropShadowFilterOperation::create(
364 WebCore::blend(m_location, IntPoint(), context),
365 WebCore::blend(m_stdDeviation, 0, context),
366 WebCore::blend(m_color, Color::transparentBlack, context));
367
368 const DropShadowFilterOperation* fromOperation = downcast<DropShadowFilterOperation>(from);
369 IntPoint fromLocation = fromOperation ? fromOperation->location() : IntPoint();
370 int fromStdDeviation = fromOperation ? fromOperation->stdDeviation() : 0;
371 Color fromColor = fromOperation ? fromOperation->color() : Color::transparentBlack;
372
373 return DropShadowFilterOperation::create(
374 WebCore::blend(fromLocation, m_location, context),
375 WebCore::blend(fromStdDeviation, m_stdDeviation, context),
376 WebCore::blend(fromColor, m_color, context));
377}

379TextStream& operator<<(TextStream& ts, const FilterOperation& filter)
380{
381 switch (filter.type()) {
382 case FilterOperation::REFERENCE:
383 ts << "reference";
384 break;
385 case FilterOperation::GRAYSCALE: {
386 const auto& colorMatrixFilter = downcast<BasicColorMatrixFilterOperation>(filter);
387 ts << "grayscale(" << colorMatrixFilter.amount() << ")";
388 break;
389 }
390 case FilterOperation::SEPIA: {
391 const auto& colorMatrixFilter = downcast<BasicColorMatrixFilterOperation>(filter);
392 ts << "sepia(" << colorMatrixFilter.amount() << ")";
393 break;
394 }
395 case FilterOperation::SATURATE: {
396 const auto& colorMatrixFilter = downcast<BasicColorMatrixFilterOperation>(filter);
397 ts << "saturate(" << colorMatrixFilter.amount() << ")";
398 break;
399 }
400 case FilterOperation::HUE_ROTATE: {
401 const auto& colorMatrixFilter = downcast<BasicColorMatrixFilterOperation>(filter);
402 ts << "hue-rotate(" << colorMatrixFilter.amount() << ")";
403 break;
404 }
405 case FilterOperation::INVERT: {
406 const auto& componentTransferFilter = downcast<BasicComponentTransferFilterOperation>(filter);
407 ts << "invert(" << componentTransferFilter.amount() << ")";
408 break;
409 }
410 case FilterOperation::APPLE_INVERT_LIGHTNESS: {
411 ts << "apple-invert-lightness()";
412 break;
413 }
414 case FilterOperation::OPACITY: {
415 const auto& componentTransferFilter = downcast<BasicComponentTransferFilterOperation>(filter);
416 ts << "opacity(" << componentTransferFilter.amount() << ")";
417 break;
418 }
419 case FilterOperation::BRIGHTNESS: {
420 const auto& componentTransferFilter = downcast<BasicComponentTransferFilterOperation>(filter);
421 ts << "brightness(" << componentTransferFilter.amount() << ")";
422 break;
423 }
424 case FilterOperation::CONTRAST: {
425 const auto& componentTransferFilter = downcast<BasicComponentTransferFilterOperation>(filter);
426 ts << "contrast(" << componentTransferFilter.amount() << ")";
427 break;
428 }
429 case FilterOperation::BLUR: {
430 const auto& blurFilter = downcast<BlurFilterOperation>(filter);
431 ts << "blur(" << blurFilter.stdDeviation().value() << ")"; // FIXME: should call floatValueForLength() but that's outisde of platform/.
432 break;
433 }
434 case FilterOperation::DROP_SHADOW: {
435 const auto& dropShadowFilter = downcast<DropShadowFilterOperation>(filter);
436 ts << "drop-shadow(" << dropShadowFilter.x() << " " << dropShadowFilter.y() << " " << dropShadowFilter.location() << " ";
437 ts << dropShadowFilter.color() << ")";
438 break;
439 }
440 case FilterOperation::PASSTHROUGH:
441 ts << "passthrough";
442 break;
443 case FilterOperation::DEFAULT: {
444 const auto& defaultFilter = downcast<DefaultFilterOperation>(filter);
445 ts << "default type=" << (int)defaultFilter.representedType();
446 break;
447 }
448 case FilterOperation::NONE:
449 ts << "none";
450 break;
451 }
452 return ts;
453}
