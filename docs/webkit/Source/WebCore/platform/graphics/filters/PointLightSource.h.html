23#pragma once
24
25#include "LightSource.h"
26#include <wtf/Ref.h>
27
28namespace WebCore {
29
30class PointLightSource : public LightSource {
31public:
32 static Ref<PointLightSource> create(const FloatPoint3D& position)
33 {
34 return adoptRef(*new PointLightSource(position));
35 }
36
37 const FloatPoint3D& position() const { return m_userSpacePosition; }
38 bool setX(float) override;
39 bool setY(float) override;
40 bool setZ(float) override;
41
42 void initPaintingData(const FilterEffect&, PaintingData&) override;
43 ComputedLightingData computePixelLightingData(const PaintingData&, int x, int y, float z) const final;
44
45 WTF::TextStream& externalRepresentation(WTF::TextStream&) const override;
46
47private:
48 PointLightSource(const FloatPoint3D& position)
49 : LightSource(LS_POINT)
50 , m_userSpacePosition(position)
51 {
52 }
53
54 FloatPoint3D m_userSpacePosition;
55 FloatPoint3D m_bufferPosition;
56};
57
58} // namespace WebCore
59
60SPECIALIZE_TYPE_TRAITS_LIGHTSOURCE(PointLightSource, LS_POINT)
