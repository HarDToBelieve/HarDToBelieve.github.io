39namespace JSC {
40class ArrayBuffer;
41}
42
43namespace WebCore {
44
45class GPUDevice;
46class GPUErrorScopes;
47
48struct GPUBufferDescriptor;
49
50enum class GPUBufferMappedOption;
51
52class GPUBuffer : public RefCounted<GPUBuffer> {
53public:
54 enum class State {
55 Mapped,
56 Unmapped,
57 Destroyed
58 };
59
60 ~GPUBuffer();
61
62 static RefPtr<GPUBuffer> tryCreate(GPUDevice&, const GPUBufferDescriptor&, GPUBufferMappedOption, GPUErrorScopes&);
63
64 PlatformBuffer* platformBuffer() const { return m_platformBuffer.get(); }
65 size_t byteLength() const { return m_byteLength; }
66 bool isCopySource() const { return m_usage.contains(GPUBufferUsage::Flags::CopySource); }
67 bool isCopyDestination() const { return m_usage.contains(GPUBufferUsage::Flags::CopyDestination); }
68 bool isIndex() const { return m_usage.contains(GPUBufferUsage::Flags::Index); }
69 bool isVertex() const { return m_usage.contains(GPUBufferUsage::Flags::Vertex); }
70 bool isUniform() const { return m_usage.contains(GPUBufferUsage::Flags::Uniform); }
71 bool isStorage() const { return m_usage.contains(GPUBufferUsage::Flags::Storage); }
72 bool isReadOnly() const;
73 bool isMappable() const { return m_usage.containsAny({ GPUBufferUsage::Flags::MapWrite, GPUBufferUsage::Flags::MapRead }); }
74 unsigned platformUsage() const { return m_platformUsage; }
75 State state() const;
76
77 JSC::ArrayBuffer* mapOnCreation();
78
79#if USE(METAL)
80 void commandBufferCommitted(MTLCommandBuffer *);
81 void commandBufferCompleted();
82#endif
83
84 using MappingCallback = WTF::Function<void(JSC::ArrayBuffer*)>;
85 void registerMappingCallback(MappingCallback&&, bool, GPUErrorScopes&);
86 void unmap(GPUErrorScopes*);
87 void destroy(GPUErrorScopes*);
88
89private:
90 struct PendingMappingCallback : public RefCounted<PendingMappingCallback> {
91 static Ref<PendingMappingCallback> create(MappingCallback&& callback)
92 {
93 return adoptRef(*new PendingMappingCallback(WTFMove(callback)));
94 }
95
96 MappingCallback callback;
97
98 private:
99 PendingMappingCallback(MappingCallback&&);
100 };
101
102 GPUBuffer(PlatformBufferSmartPtr&&, GPUDevice&, size_t, OptionSet<GPUBufferUsage::Flags>, GPUBufferMappedOption);
103 static bool validateBufferUsage(const GPUDevice&, OptionSet<GPUBufferUsage::Flags>, GPUErrorScopes&);
104
105 JSC::ArrayBuffer* stagingBufferForRead();
106 JSC::ArrayBuffer* stagingBufferForWrite();
107 void runMappingCallback();
108 void copyStagingBufferToGPU(GPUErrorScopes*);
109
110 bool isMapWrite() const { return m_usage.contains(GPUBufferUsage::Flags::MapWrite); }
111 bool isMapRead() const { return m_usage.contains(GPUBufferUsage::Flags::MapRead); }
112 bool isMapWriteable() const { return isMapWrite() && state() == State::Unmapped; }
113 bool isMapReadable() const { return isMapRead() && state() == State::Unmapped; }
114
115 PlatformBufferSmartPtr m_platformBuffer;
116 Ref<GPUDevice> m_device;
117
118 RefPtr<JSC::ArrayBuffer> m_stagingBuffer;
119 RefPtr<PendingMappingCallback> m_mappingCallback;
120 MainThreadDeferrableTask m_mappingCallbackTask;
121
122 size_t m_byteLength;
123 OptionSet<GPUBufferUsage::Flags> m_usage;
124 unsigned m_platformUsage;
125 unsigned m_numScheduledCommandBuffers { 0 };
126 bool m_isMappedFromCreation { false };
127};

129} // namespace WebCore

131#endif // ENABLE(WEBGPU)
