40namespace WebCore {
41
42class GPUDevice;
43class GPUErrorScopes;
44class GPUPipelineLayout;
45
46struct GPUComputePipelineDescriptor;
47
48class GPUComputePipeline final : public GPUPipeline {
49public:
50 virtual ~GPUComputePipeline();
51
52 static RefPtr<GPUComputePipeline> tryCreate(const GPUDevice&, const GPUComputePipelineDescriptor&, GPUErrorScopes&);
53
54 bool isComputePipeline() const { return true; }
55
56 bool recompile(const GPUDevice&, GPUProgrammableStageDescriptor&& computeStage);
57
58 const PlatformComputePipeline* platformComputePipeline() const { return m_platformComputePipeline.get(); }
59
60#if ENABLE(WHLSL_COMPILER)
61 WHLSL::ComputeDimensions computeDimensions() const { return m_computeDimensions; }
62#endif
63
64private:
65#if ENABLE(WHLSL_COMPILER)
66 GPUComputePipeline(PlatformComputePipelineSmartPtr&&, WHLSL::ComputeDimensions, const RefPtr<GPUPipelineLayout>&);
67#endif
68
69 PlatformComputePipelineSmartPtr m_platformComputePipeline;
70#if ENABLE(WHLSL_COMPILER)
71 WHLSL::ComputeDimensions m_computeDimensions { 0, 0, 0 };
72#endif
73
74 // Preserved for Web Inspector recompilation.
75 RefPtr<GPUPipelineLayout> m_layout;
76};
77
78} // namespace WebCore
