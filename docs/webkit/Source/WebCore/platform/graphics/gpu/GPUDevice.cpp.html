64static unsigned maximumMipLevelCount(GPUTextureDimension dimension, GPUExtent3D size)
65{
66 unsigned width = size.width;
67 unsigned height = 1;
68 unsigned depth = 1;
69 switch (dimension) {
70 case GPUTextureDimension::_1d:
71 break;
72 case GPUTextureDimension::_2d:
73 height = size.height;
74 break;
75 case GPUTextureDimension::_3d:
76 height = size.height;
77 depth = size.depth;
78 break;
79 }
80
81 auto maxDimension = std::max(std::max(width, height), depth);
82 int i = sizeof(GPUExtent3D::InnerType) * 8 - 1;
83 while (!(maxDimension & (1 << i)) && i >= 0)
84 --i;
85 ASSERT(i >= 0);
86 return i + 1;
87}

89RefPtr<GPUTexture> GPUDevice::tryCreateTexture(const GPUTextureDescriptor& descriptor) const
90{
91 if (!descriptor.size.width || !descriptor.size.height || !descriptor.size.depth) {
92 m_errorScopes->generatePrefixedError("Textures can't have 0 dimensions.");
93 return { };
94 }
95
96 if (!descriptor.mipLevelCount) {
97 m_errorScopes->generatePrefixedError("Textures can't have 0 miplevels.");
98 return { };
99 }
100
101 if (!descriptor.sampleCount) {
102 m_errorScopes->generatePrefixedError("Textures can't have 0 samples.");
103 return { };
104 }
105
106 // https://developer.apple.com/metal/Metal-Feature-Set-Tables.pdf
107 unsigned maxRegularSize = 8192;
108 unsigned maxDepthSize = 2048;
109 switch (descriptor.dimension) {
110 case GPUTextureDimension::_1d:
111 if (descriptor.size.width > maxRegularSize) {
112 m_errorScopes->generatePrefixedError(makeString("1D texture is too wide: ", descriptor.size.width, " needs to be <= ", maxRegularSize, "."));
113 return { };
114 }
115 if (descriptor.size.height > maxDepthSize) {
116 m_errorScopes->generatePrefixedError(makeString("1D texture array has too many layers: ", descriptor.size.height, " needs to be <= ", maxDepthSize, "."));
117 return { };
118 }
119 break;
120 case GPUTextureDimension::_2d:
121 if (descriptor.size.width > maxRegularSize || descriptor.size.height > maxRegularSize) {
122 m_errorScopes->generatePrefixedError(makeString("2D texture is too large:", descriptor.size.width, " and ", descriptor.size.height, " need to be <= ", maxRegularSize, "."));
123 return { };
124 }
125 if (descriptor.size.depth > maxDepthSize) {
126 m_errorScopes->generatePrefixedError(makeString("2D texture array has too many layers: ", descriptor.size.depth, " needs to be <= ", maxDepthSize, "."));
127 return { };
128 }
129 break;
130 case GPUTextureDimension::_3d:
131 if (descriptor.size.width > maxDepthSize || descriptor.size.height > maxDepthSize || descriptor.size.depth > maxDepthSize) {
132 m_errorScopes->generatePrefixedError(makeString("3D texture is too large: ", descriptor.size.width, ", ", descriptor.size.height, ", and ", descriptor.size.depth, " need to be <= ", maxDepthSize, "."));
133 return { };
134 }
135 break;
136 }
137
138 if (descriptor.dimension == GPUTextureDimension::_1d && descriptor.mipLevelCount != 1) {
139 m_errorScopes->generatePrefixedError("1D textures can't have mipmaps");
140 return { };
141 }
142
143 auto maxLevels = maximumMipLevelCount(descriptor.dimension, descriptor.size);
144 if (descriptor.mipLevelCount > maxLevels) {
145 m_errorScopes->generatePrefixedError(makeString("Too many miplevels: ", descriptor.mipLevelCount, " needs to be <= ", maxLevels, "."));
146 return { };
147 }
148
149 if (descriptor.sampleCount != 1 && descriptor.sampleCount != 4) {
150 m_errorScopes->generatePrefixedError("Texture sampleCount can only be 1 or 4.");
151 return { };
152 }
153
154 if (descriptor.sampleCount != 1) {
155 if (descriptor.dimension != GPUTextureDimension::_2d) {
156 m_errorScopes->generatePrefixedError("Texture dimension incompatible with multisampling.");
157 return { };
158 }
159 if (descriptor.mipLevelCount != 1) {
160 m_errorScopes->generatePrefixedError("Multisampled textures can't have mipmaps.");
161 return { };
162 }
163 if (descriptor.size.depth != 1) {
164 m_errorScopes->generatePrefixedError("Array textures can't be multisampled.");
165 return { };
166 }
167 if (descriptor.usage & static_cast<GPUTextureUsageFlags>(GPUTextureUsage::Flags::Storage)) {
168 m_errorScopes->generatePrefixedError("Multisampled textures can't have STORAGE usage.");
169 return { };
170 }
171 // FIXME: Only some pixel formats are capable of MSAA. When we add those new pixel formats, we'll have to add filtering code here.
172 }
173
174 // The height component for 1d textures is the array length.
175 // The depth component for 2d textures is the array length.
176 if (descriptor.dimension == GPUTextureDimension::_1d && descriptor.size.depth != 1) {
177 m_errorScopes->generatePrefixedError("1D textures can't have depth != 1.");
178 return { };
179 }
180
181 if (descriptor.format == GPUTextureFormat::Depth32floatStencil8 && descriptor.dimension != GPUTextureDimension::_2d) {
182 m_errorScopes->generatePrefixedError("Depth/stencil textures must be 2d.");
183 return { };
184 }
185
186 if (descriptor.format == GPUTextureFormat::Depth32floatStencil8 && descriptor.sampleCount != 1) {
187 m_errorScopes->generatePrefixedError("Depth/stencil textures can't be multisampled.");
188 return { };
189 }
190
191 if (static_cast<GPUTextureUsageFlags>(descriptor.usage) >= static_cast<GPUTextureUsageFlags>(GPUTextureUsage::Flags::MaximumValue)) {
192 m_errorScopes->generatePrefixedError("Invalid usage flags.");
193 return { };
194 }
195
196 return GPUTexture::tryCreate(*this, descriptor, makeRef(*m_errorScopes));
197}
