69class GPUDevice : public RefCounted<GPUDevice>, public CanMakeWeakPtr<GPUDevice> {
70public:
71 static RefPtr<GPUDevice> tryCreate(const std::optional<GPURequestAdapterOptions>&);
72 static void prewarm();
73
74 RefPtr<GPUBuffer> tryCreateBuffer(const GPUBufferDescriptor&, GPUBufferMappedOption, GPUErrorScopes&);
75 RefPtr<GPUTexture> tryCreateTexture(const GPUTextureDescriptor&) const;
76 RefPtr<GPUSampler> tryCreateSampler(const GPUSamplerDescriptor&) const;
77
78 RefPtr<GPUBindGroupLayout> tryCreateBindGroupLayout(const GPUBindGroupLayoutDescriptor&) const;
79 Ref<GPUPipelineLayout> createPipelineLayout(GPUPipelineLayoutDescriptor&&) const;
80 RefPtr<GPUBindGroup> tryCreateBindGroup(const GPUBindGroupDescriptor&, GPUErrorScopes&) const;
81
82 RefPtr<GPUShaderModule> tryCreateShaderModule(const GPUShaderModuleDescriptor&) const;
83 RefPtr<GPURenderPipeline> tryCreateRenderPipeline(const GPURenderPipelineDescriptor&, GPUErrorScopes&) const;
84 RefPtr<GPUComputePipeline> tryCreateComputePipeline(const GPUComputePipelineDescriptor&, GPUErrorScopes&) const;
85
86 RefPtr<GPUCommandBuffer> tryCreateCommandBuffer() const;
87
88 RefPtr<GPUQueue> tryGetQueue() const;
89
90 PlatformDevice* platformDevice() const { return m_platformDevice.get(); }
91 GPUSwapChain* swapChain() const { return m_swapChain.get(); }
92 void setSwapChain(RefPtr<GPUSwapChain>&&);
93
94 void setErrorScopes(Ref<GPUErrorScopes>&& errorScopes) { m_errorScopes = WTFMove(errorScopes); }
95
96 static constexpr bool useWHLSL = true;
97
98private:
99 explicit GPUDevice(PlatformDeviceSmartPtr&&);
100
101 PlatformDeviceSmartPtr m_platformDevice;
102 mutable RefPtr<GPUQueue> m_queue;
103 RefPtr<GPUSwapChain> m_swapChain;
104 mutable RefPtr<GPUBindGroupAllocator> m_bindGroupAllocator;
105 RefPtr<GPUErrorScopes> m_errorScopes;
106};
