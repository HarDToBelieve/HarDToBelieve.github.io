26#include "config.h"
27#include "GPUError.h"
28
29#if ENABLE(WEBGPU)
30
31#include "GPUErrorFilter.h"
32
33namespace WebCore {
34
35GPUError createError(GPUErrorFilter filter, const String& message)
36{
37 switch (filter) {
38 case GPUErrorFilter::OutOfMemory:
39 return GPUError(RefPtr<GPUOutOfMemoryError>(GPUOutOfMemoryError::create()));
40 case GPUErrorFilter::Validation:
41 return GPUError(RefPtr<GPUValidationError>(GPUValidationError::create(message)));
42 case GPUErrorFilter::None:
43 ASSERT_NOT_REACHED();
44 return GPUError(RefPtr<GPUValidationError>(GPUValidationError::create("Bad error!")));
45 }
46}
47
48} // namespace WebCore
49
50#endif // ENABLE(WEBGPU)
