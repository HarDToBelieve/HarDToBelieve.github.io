33const unsigned maxUncapturedErrorEventsAllowed = 256;
34
35Ref<GPUErrorScopes> GPUErrorScopes::create(UncapturedErrorCallback&& callback)
36{
37 return adoptRef(*new GPUErrorScopes(WTFMove(callback)));
38}
39
40GPUErrorScopes::GPUErrorScopes(UncapturedErrorCallback&& callback)
41 : m_uncapturedErrorCallback(WTFMove(callback))
42 , m_numUncapturedErrorEventsAllowed(maxUncapturedErrorEventsAllowed)
43{
44}
45
46void GPUErrorScopes::pushErrorScope(GPUErrorFilter filter)
47{
48 m_errorScopes.append(ErrorScope { filter, std::nullopt });
49}
50
51std::optional<GPUError> GPUErrorScopes::popErrorScope(String& failMessage)
52{
53 if (m_errorScopes.isEmpty()) {
54 failMessage = "No error scope exists!";
55 return std::nullopt;
56 }
57
58 auto scope = m_errorScopes.takeLast();
59 return scope.filter == GPUErrorFilter::None ? std::nullopt : scope.error;
60}
61
62void GPUErrorScopes::generateError(const String& message, GPUErrorFilter filter)
63{
64 auto iterator = std::find_if(m_errorScopes.rbegin(), m_errorScopes.rend(), [filter](const ErrorScope& scope) {
65 return scope.filter == GPUErrorFilter::None || scope.filter == filter;
66 });
67
68 if (iterator == m_errorScopes.rend()) {
69 if (!m_numUncapturedErrorEventsAllowed)
70 return;
71
72 m_uncapturedErrorCallback(createError(filter, message));
73 if (!(--m_numUncapturedErrorEventsAllowed))
74 m_uncapturedErrorCallback(createError(GPUErrorFilter::Validation, "WebGPU: Too many errors; no more error events will fire on this GPUDevice."));
75
76 return;
77 }
78
79 // If the scope has already captured an error, ignore this new one.
80 if (iterator->error)
81 return;
82
83 iterator->error = createError(filter, message);
84}
85
86void GPUErrorScopes::generatePrefixedError(const String& message)
87{
88 generateError(m_prefix + message, GPUErrorFilter::Validation);
89}
