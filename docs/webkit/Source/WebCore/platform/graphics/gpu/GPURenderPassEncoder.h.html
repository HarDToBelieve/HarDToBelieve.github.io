37namespace WebCore {
38
39class GPUBuffer;
40class GPUCommandBuffer;
41
42struct GPUColor;
43struct GPURenderPassDescriptor;
44
45class GPURenderPassEncoder : public GPUProgrammablePassEncoder {
46public:
47 static RefPtr<GPURenderPassEncoder> tryCreate(Ref<GPUCommandBuffer>&&, GPURenderPassDescriptor&&);
48
49 void setPipeline(Ref<const GPURenderPipeline>&&);
50 void setBlendColor(const GPUColor&);
51 void setViewport(float x, float y, float width, float height, float minDepth, float maxDepth);
52 void setScissorRect(unsigned x, unsigned y, unsigned width, unsigned height);
53 void setIndexBuffer(GPUBuffer&, uint64_t offset);
54 void setVertexBuffers(unsigned index, const Vector<Ref<GPUBuffer>>&, const Vector<uint64_t>& offsets);
55 void draw(unsigned vertexCount, unsigned instanceCount, unsigned firstVertex, unsigned firstInstance);
56 void drawIndexed(unsigned indexCount, unsigned instanceCount, unsigned firstIndex, int baseVertex, unsigned firstInstance);
57
58private:
59 GPURenderPassEncoder(Ref<GPUCommandBuffer>&&, PlatformRenderPassEncoderSmartPtr&&);
60 ~GPURenderPassEncoder() { endPass(); }
61
62 // GPUProgrammablePassEncoder
63 const PlatformProgrammablePassEncoder* platformPassEncoder() const final;
64 void invalidateEncoder() final { m_platformRenderPassEncoder = nullptr; }
65#if USE(METAL)
66 void useResource(const MTLResource *, unsigned usage) final;
67 void setVertexBuffer(const MTLBuffer *, NSUInteger offset, unsigned index) final;
68 void setFragmentBuffer(const MTLBuffer *, NSUInteger offset, unsigned index) final;
69
70 RefPtr<GPUBuffer> m_indexBuffer;
71 uint64_t m_indexBufferOffset;
72#endif // USE(METAL)
73
74 PlatformRenderPassEncoderSmartPtr m_platformRenderPassEncoder;
75 RefPtr<const GPURenderPipeline> m_pipeline;
76};
