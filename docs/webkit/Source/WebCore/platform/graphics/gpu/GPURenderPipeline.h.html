27
28#if ENABLE(WEBGPU)
29
30#include "GPUPipeline.h"
31#include "GPUPlatformTypes.h"
32#include "GPUProgrammableStageDescriptor.h"
33#include "GPURenderPipelineDescriptor.h"
34#include <wtf/RefCounted.h>
35#include <wtf/RefPtr.h>
36
37namespace WebCore {
38
39class GPUDevice;
40class GPUErrorScopes;
41
42class GPURenderPipeline final : public GPUPipeline {
43public:
44 virtual ~GPURenderPipeline();
45
46 static RefPtr<GPURenderPipeline> tryCreate(const GPUDevice&, const GPURenderPipelineDescriptor&, GPUErrorScopes&);
47
48 bool isRenderPipeline() const { return true; }
49
50 bool recompile(const GPUDevice&, GPUProgrammableStageDescriptor&& vertexStage, std::optional<GPUProgrammableStageDescriptor>&& fragmentStage);
51
52#if USE(METAL)
53 MTLDepthStencilState *depthStencilState() const { return m_depthStencilState.get(); }
54#endif
55 PlatformRenderPipeline* platformRenderPipeline() const { return m_platformRenderPipeline.get(); }
56 GPUPrimitiveTopology primitiveTopology() const { return m_primitiveTopology; }
57 std::optional<GPUIndexFormat> indexFormat() const { return m_indexFormat; }
58
59private:
60#if USE(METAL)
61 GPURenderPipeline(RetainPtr<MTLDepthStencilState>&&, PlatformRenderPipelineSmartPtr&&, GPUPrimitiveTopology, std::optional<GPUIndexFormat>, const RefPtr<GPUPipelineLayout>&, const GPURenderPipelineDescriptorBase&);
62
63 RetainPtr<MTLDepthStencilState> m_depthStencilState;
64#endif // USE(METAL)
65 PlatformRenderPipelineSmartPtr m_platformRenderPipeline;
66 GPUPrimitiveTopology m_primitiveTopology;
67 std::optional<GPUIndexFormat> m_indexFormat;
68
69 // Preserved for Web Inspector recompilation.
70 RefPtr<GPUPipelineLayout> m_layout;
71 GPURenderPipelineDescriptorBase m_renderDescriptorBase;
72};
73
74} // namespace WebCore

76SPECIALIZE_TYPE_TRAITS_GPUPIPELINE(WebCore::GPURenderPipeline, isRenderPipeline())

78#endif // ENABLE(WEBGPU)
