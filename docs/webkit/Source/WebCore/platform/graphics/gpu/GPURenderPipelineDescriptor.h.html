37namespace WebCore {
38
39enum class GPUPrimitiveTopology {
40 PointList,
41 LineList,
42 LineStrip,
43 TriangleList,
44 TriangleStrip
45};
46
47struct GPURenderPipelineDescriptorBase {
48 GPUPrimitiveTopology primitiveTopology;
49 Vector<GPUColorStateDescriptor> colorStates;
50 std::optional<GPUDepthStencilStateDescriptor> depthStencilState;
51 GPUVertexInputDescriptor vertexInput;
52};
53
54struct GPURenderPipelineDescriptor : GPUPipelineDescriptorBase, GPURenderPipelineDescriptorBase {
55 GPURenderPipelineDescriptor(RefPtr<GPUPipelineLayout>&& layout, GPUProgrammableStageDescriptor&& vertex, std::optional<GPUProgrammableStageDescriptor>&& fragment, const GPURenderPipelineDescriptorBase& base)
56 : GPUPipelineDescriptorBase { WTFMove(layout) }
57 , GPURenderPipelineDescriptorBase(base)
58 , vertexStage(WTFMove(vertex))
59 , fragmentStage(WTFMove(fragment))
60 {
61 }
62
63 GPUProgrammableStageDescriptor vertexStage;
64 std::optional<GPUProgrammableStageDescriptor> fragmentStage;
65};

67} // namespace WebCore
