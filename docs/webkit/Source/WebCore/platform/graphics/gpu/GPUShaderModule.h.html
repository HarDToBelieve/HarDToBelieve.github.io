37class GPUDevice;

39struct GPUShaderModuleDescriptor;
40
41class GPUShaderModule : public RefCounted<GPUShaderModule> {
42public:
43 static RefPtr<GPUShaderModule> tryCreate(const GPUDevice&, const GPUShaderModuleDescriptor&);
44
45#if ENABLE(WHLSL_COMPILER)
46 PlatformShaderModule* platformShaderModule() const { return m_whlslModule ? nullptr : m_platformShaderModule.get(); }

48#endif
49
50private:
51 GPUShaderModule(PlatformShaderModuleSmartPtr&&);
52#if ENABLE(WHLSL_COMPILER)
53 GPUShaderModule(UniqueRef<WHLSL::ShaderModule>&& whlslModule);
54#endif
55
56 PlatformShaderModuleSmartPtr m_platformShaderModule;
57#if ENABLE(WHLSL_COMPILER)
58 std::unique_ptr<WHLSL::ShaderModule> m_whlslModule;
59#endif
60};

62} // namespace WebCore
