26#pragma once
27
28#if ENABLE(WEBGPU)
29
30#include "GPUPlatformTypes.h"
31#include "GPUTextureUsage.h"
32#include <wtf/OptionSet.h>
33#include <wtf/RefCounted.h>
34#include <wtf/RefPtr.h>
35
36namespace WebCore {
37
38class GPUDevice;
39class GPUErrorScopes;
40
41struct GPUTextureDescriptor;
42
43class GPUTexture : public RefCounted<GPUTexture> {
44public:
45 static RefPtr<GPUTexture> tryCreate(const GPUDevice&, const GPUTextureDescriptor&, GPUErrorScopes&);
46 static Ref<GPUTexture> create(PlatformTextureSmartPtr&&, OptionSet<GPUTextureUsage::Flags>);
47
48 PlatformTexture *platformTexture() const { return m_platformTexture.get(); }
49 bool isCopySource() const { return m_usage.contains(GPUTextureUsage::Flags::CopySource); }
50 bool isCopyDestination() const { return m_usage.contains(GPUTextureUsage::Flags::CopyDestination); }
51 bool isOutputAttachment() const { return m_usage.contains(GPUTextureUsage::Flags::OutputAttachment); }
52 bool isReadOnly() const { return m_usage.containsAny({ GPUTextureUsage::Flags::CopySource, GPUTextureUsage::Flags::Sampled }); }
53 bool isSampled() const { return m_usage.contains(GPUTextureUsage::Flags::Sampled); }
54 bool isStorage() const { return m_usage.contains(GPUTextureUsage::Flags::Storage); }
55 unsigned platformUsage() const { return m_platformUsage; }
56
57 RefPtr<GPUTexture> tryCreateDefaultTextureView();
58 void destroy() { m_platformTexture = nullptr; }
59
60private:
61 explicit GPUTexture(PlatformTextureSmartPtr&&, OptionSet<GPUTextureUsage::Flags>);
62
63 PlatformTextureSmartPtr m_platformTexture;
64
65 OptionSet<GPUTextureUsage::Flags> m_usage;
66 unsigned m_platformUsage;
67};
68
69} // namespace WebCore
70
71#endif // ENABLE(WEBGPU)
