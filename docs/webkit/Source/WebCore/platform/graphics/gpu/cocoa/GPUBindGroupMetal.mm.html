43static std::optional<GPUBufferBinding> tryGetResourceAsBufferBinding(const GPUBindingResource& resource, const char* const functionName)
44{
45#if LOG_DISABLED
46 UNUSED_PARAM(functionName);
47#endif
48 if (!WTF::holds_alternative<GPUBufferBinding>(resource)) {
49 LOG(WebGPU, "%s: Resource is not a buffer type!", functionName);
50 return std::nullopt;
51 }
52 auto& bufferBinding = WTF::get<GPUBufferBinding>(resource);
53 if (!bufferBinding.buffer->platformBuffer()) {
54 LOG(WebGPU, "%s: Invalid MTLBuffer in GPUBufferBinding!", functionName);
55 return std::nullopt;
56 }
57 if (!isInBounds<NSUInteger>(bufferBinding.size) || bufferBinding.size > bufferBinding.buffer->byteLength()) {
58 LOG(WebGPU, "%s: GPUBufferBinding size is too large!", functionName);
59 return std::nullopt;
60 }
61 // MTLBuffer size (NSUInteger) is 32 bits on some platforms.
62 if (!isInBounds<NSUInteger>(bufferBinding.offset)) {
63 LOG(WebGPU, "%s: Buffer offset is too large!", functionName);
64 return std::nullopt;
65 }
66 return GPUBufferBinding { bufferBinding.buffer.copyRef(), bufferBinding.offset, bufferBinding.size };
67}

69static void setBufferOnEncoder(MTLArgumentEncoder *argumentEncoder, const GPUBufferBinding& bufferBinding, unsigned name, unsigned lengthName)
70{
71 ASSERT(argumentEncoder && bufferBinding.buffer->platformBuffer());
72
73 BEGIN_BLOCK_OBJC_EXCEPTIONS
74 // Bounds check when converting GPUBufferBinding ensures that NSUInteger cast of uint64_t offset is safe.
75 [argumentEncoder setBuffer:bufferBinding.buffer->platformBuffer() offset:static_cast<NSUInteger>(bufferBinding.offset) atIndex:name];
76 void* lengthPointer = [argumentEncoder constantDataAtIndex:lengthName];
77 memcpy(lengthPointer, &bufferBinding.size, sizeof(uint64_t));
78 END_BLOCK_OBJC_EXCEPTIONS
79}

81static MTLSamplerState *tryGetResourceAsMtlSampler(const GPUBindingResource& resource, const char* const functionName)
82{
83#if LOG_DISABLED
84 UNUSED_PARAM(functionName);
85#endif
86 if (!WTF::holds_alternative<Ref<const GPUSampler>>(resource)) {
87 LOG(WebGPU, "%s: Resource is not a GPUSampler!", functionName);
88 return nullptr;
89 }
90 auto samplerState = WTF::get<Ref<const GPUSampler>>(resource)->platformSampler();
91 if (!samplerState) {
92 LOG(WebGPU, "%s: Invalid MTLSamplerState in GPUSampler binding!", functionName);
93 return nullptr;
94 }
95 return samplerState;
96}

98static void setSamplerOnEncoder(MTLArgumentEncoder *argumentEncoder, MTLSamplerState *sampler, unsigned index)
99{
100 ASSERT(argumentEncoder && sampler);
101
102 BEGIN_BLOCK_OBJC_EXCEPTIONS
103 [argumentEncoder setSamplerState:sampler atIndex:index];
104 END_BLOCK_OBJC_EXCEPTIONS
105}

107static RefPtr<GPUTexture> tryGetResourceAsTexture(const GPUBindingResource& resource, const char* const functionName)
108 {
109#if LOG_DISABLED
110 UNUSED_PARAM(functionName);
111#endif
112 if (!WTF::holds_alternative<Ref<GPUTexture>>(resource)) {
113 LOG(WebGPU, "%s: Resource is not a GPUTextureView!", functionName);
114 return nullptr;
115 }
116 auto& textureRef = WTF::get<Ref<GPUTexture>>(resource);
117 if (!textureRef->platformTexture()) {
118 LOG(WebGPU, "%s: Invalid MTLTexture in GPUTextureView binding!", functionName);
119 return nullptr;
120 }
121 return textureRef.copyRef();
122}

133RefPtr<GPUBindGroup> GPUBindGroup::tryCreate(const GPUBindGroupDescriptor& descriptor, GPUBindGroupAllocator& allocator)
134{
135 const char* const functionName = "GPUBindGroup::tryCreate()";
136
137 MTLArgumentEncoder *vertexEncoder = descriptor.layout->vertexEncoder();
138 MTLArgumentEncoder *fragmentEncoder = descriptor.layout->fragmentEncoder();
139 MTLArgumentEncoder *computeEncoder = descriptor.layout->computeEncoder();
140
141 auto offsets = allocator.allocateAndSetEncoders(vertexEncoder, fragmentEncoder, computeEncoder);
142 if (!offsets)
143 return nullptr;
144
145 HashSet<Ref<GPUBuffer>> boundBuffers;
146 HashSet<Ref<GPUTexture>> boundTextures;
147
148 // Set each resource on each MTLArgumentEncoder it should be visible on.
149 const auto& layoutBindingsMap = descriptor.layout->bindingsMap();
150 for (const auto& resourceBinding : descriptor.bindings) {
151 auto index = resourceBinding.binding;
152 auto layoutIterator = layoutBindingsMap.find(index);
153 if (layoutIterator == layoutBindingsMap.end()) {
154 LOG(WebGPU, "%s: GPUBindGroupBinding %u not found in GPUBindGroupLayout!", functionName, index);
155 return nullptr;
156 }
157 auto layoutBinding = layoutIterator->value;
158 if (layoutBinding.externalBinding.visibility == GPUShaderStage::Flags::None)
159 continue;
160
161 bool isForVertex = layoutBinding.externalBinding.visibility & GPUShaderStage::Flags::Vertex;
162 bool isForFragment = layoutBinding.externalBinding.visibility & GPUShaderStage::Flags::Fragment;
163 bool isForCompute = layoutBinding.externalBinding.visibility & GPUShaderStage::Flags::Compute;
164
165 if (isForVertex && !vertexEncoder) {
166 LOG(WebGPU, "%s: No vertex argument encoder found for binding %u!", functionName, index);
167 return nullptr;
168 }
169 if (isForFragment && !fragmentEncoder) {
170 LOG(WebGPU, "%s: No fragment argument encoder found for binding %u!", functionName, index);
171 return nullptr;
172 }
173 if (isForCompute && !computeEncoder) {
174 LOG(WebGPU, "%s: No compute argument encoder found for binding %u!", functionName, index);
175 return nullptr;
176 }
177
178 auto handleBuffer = [&](unsigned internalLengthName) -> bool {
179 auto bufferResource = tryGetResourceAsBufferBinding(resourceBinding.resource, functionName);
180 if (!bufferResource)
181 return false;
182 if (isForVertex)
183 setBufferOnEncoder(vertexEncoder, *bufferResource, layoutBinding.internalName, internalLengthName);
184 if (isForFragment)
185 setBufferOnEncoder(fragmentEncoder, *bufferResource, layoutBinding.internalName, internalLengthName);
186 if (isForCompute)
187 setBufferOnEncoder(computeEncoder, *bufferResource, layoutBinding.internalName, internalLengthName);
188 boundBuffers.addVoid(WTFMove(bufferResource->buffer));
189 return true;
190 };
191
192 auto success = WTF::visit(WTF::makeVisitor([&](GPUBindGroupLayout::UniformBuffer& uniformBuffer) -> bool {
193 return handleBuffer(uniformBuffer.internalLengthName);
194 }, [&](GPUBindGroupLayout::DynamicUniformBuffer& dynamicUniformBuffer) -> bool {
195 return handleBuffer(dynamicUniformBuffer.internalLengthName);
196 }, [&](GPUBindGroupLayout::Sampler&) -> bool {
197 auto samplerState = tryGetResourceAsMtlSampler(resourceBinding.resource, functionName);
198 if (!samplerState)
199 return false;
200 if (isForVertex)
201 setSamplerOnEncoder(vertexEncoder, samplerState, layoutBinding.internalName);
202 if (isForFragment)
203 setSamplerOnEncoder(fragmentEncoder, samplerState, layoutBinding.internalName);
204 if (isForCompute)
205 setSamplerOnEncoder(computeEncoder, samplerState, layoutBinding.internalName);
206 return true;
207 }, [&](GPUBindGroupLayout::SampledTexture&) -> bool {
208 auto textureResource = tryGetResourceAsTexture(resourceBinding.resource, functionName);
209 if (!textureResource)
210 return false;
211 if (isForVertex)
212 setTextureOnEncoder(vertexEncoder, textureResource->platformTexture(), layoutBinding.internalName);
213 if (isForFragment)
214 setTextureOnEncoder(fragmentEncoder, textureResource->platformTexture(), layoutBinding.internalName);
215 if (isForCompute)
216 setTextureOnEncoder(computeEncoder, textureResource->platformTexture(), layoutBinding.internalName);
217 boundTextures.addVoid(textureResource.releaseNonNull());
218 return true;
219 }, [&](GPUBindGroupLayout::StorageBuffer& storageBuffer) -> bool {
220 return handleBuffer(storageBuffer.internalLengthName);
221 }, [&](GPUBindGroupLayout::DynamicStorageBuffer& dynamicStorageBuffer) -> bool {
222 return handleBuffer(dynamicStorageBuffer.internalLengthName);
223 }), layoutBinding.internalBindingDetails);
224 if (!success)
225 return nullptr;
226 }
227
228 return adoptRef(new GPUBindGroup(WTFMove(*offsets), allocator, WTFMove(boundBuffers), WTFMove(boundTextures)));
229}
