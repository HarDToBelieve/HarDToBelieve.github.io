43static constexpr auto readOnlyFlags = OptionSet<GPUBufferUsage::Flags> { GPUBufferUsage::Flags::Index, GPUBufferUsage::Flags::Vertex, GPUBufferUsage::Flags::Uniform, GPUBufferUsage::Flags::CopySource };

46bool GPUBuffer::validateBufferUsage(const GPUDevice& device, OptionSet<GPUBufferUsage::Flags> usage, GPUErrorScopes& errorScopes)
47{
48 if (!device.platformDevice()) {
49 errorScopes.generatePrefixedError("Invalid GPUDevice!");
50 return false;
51 }
52
53 if (usage.containsAll({ GPUBufferUsage::Flags::MapWrite, GPUBufferUsage::Flags::MapRead })) {
54 errorScopes.generatePrefixedError("Buffer cannot have both MAP_READ and MAP_WRITE usage!");
55 return false;
56 }
57
58 return true;
59}

61RefPtr<GPUBuffer> GPUBuffer::tryCreate(GPUDevice& device, const GPUBufferDescriptor& descriptor, GPUBufferMappedOption isMapped, GPUErrorScopes& errorScopes)
62{
63 // MTLBuffer size (NSUInteger) is 32 bits on some platforms.
64 NSUInteger size = 0;
65 if (!WTF::convertSafely(descriptor.size, size)) {
66 errorScopes.generateError("", GPUErrorFilter::OutOfMemory);
67 return nullptr;
68 }
69
70 auto usage = OptionSet<GPUBufferUsage::Flags>::fromRaw(descriptor.usage);
71 if (!validateBufferUsage(device, usage, errorScopes))
72 return nullptr;
73
74 // copyBufferToBuffer calls require 4-byte alignment. "Unmapping" a mapped-on-creation GPUBuffer
75 // that is otherwise unmappable requires such a copy to upload data.
76 if (isMapped == GPUBufferMappedOption::IsMapped
77 && !usage.containsAny({ GPUBufferUsage::Flags::MapWrite, GPUBufferUsage::Flags::MapRead })
78 && descriptor.size % 4) {
79 errorScopes.generatePrefixedError("Data must be aligned to a multiple of 4 bytes!");
80 return nullptr;
81 }
82
83 // FIXME: Metal best practices: Read-only one-time-use data less than 4 KB should not allocate a MTLBuffer and be used in [MTLCommandEncoder set*Bytes] calls instead.
84
85 MTLResourceOptions resourceOptions = MTLResourceCPUCacheModeDefaultCache;
86
87 // Mappable buffers use shared storage allocation.
88 resourceOptions |= usage.containsAny({ GPUBufferUsage::Flags::MapWrite, GPUBufferUsage::Flags::MapRead }) ? MTLResourceStorageModeShared : MTLResourceStorageModePrivate;
89
90 RetainPtr<MTLBuffer> mtlBuffer;
91
92 BEGIN_BLOCK_OBJC_EXCEPTIONS
93
94 mtlBuffer = adoptNS([device.platformDevice() newBufferWithLength:static_cast<NSUInteger>(descriptor.size) options:resourceOptions]);
95
96 END_BLOCK_OBJC_EXCEPTIONS
97
98 if (!mtlBuffer) {
99 errorScopes.generateError("", GPUErrorFilter::OutOfMemory);
100 return nullptr;
101 }
102
103 return adoptRef(*new GPUBuffer(WTFMove(mtlBuffer), device, size, usage, isMapped));
104}

106GPUBuffer::GPUBuffer(RetainPtr<MTLBuffer>&& buffer, GPUDevice& device, size_t size, OptionSet<GPUBufferUsage::Flags> usage, GPUBufferMappedOption isMapped)
107 : m_platformBuffer(WTFMove(buffer))
108 , m_device(makeRef(device))
109 , m_byteLength(size)
110 , m_usage(usage)
111 , m_isMappedFromCreation(isMapped == GPUBufferMappedOption::IsMapped)
112{
113 m_platformUsage = MTLResourceUsageRead;
114 if (isStorage())
115 m_platformUsage |= MTLResourceUsageWrite;
116}

118GPUBuffer::~GPUBuffer()

123bool GPUBuffer::isReadOnly() const
124{
125 return m_usage.containsAny(readOnlyFlags);

128GPUBuffer::State GPUBuffer::state() const
129{
130 if (!m_platformBuffer)
131 return State::Destroyed;
132 if (m_isMappedFromCreation || m_mappingCallback)
133 return State::Mapped;
134
135 return State::Unmapped;
136}

144#if USE(METAL)
145void GPUBuffer::commandBufferCommitted(MTLCommandBuffer *commandBuffer)
146{
147 ASSERT(isMainThread());
148 ++m_numScheduledCommandBuffers;
149
150 BEGIN_BLOCK_OBJC_EXCEPTIONS
151 // Make sure |this| only gets ref'd / deref'd on the main thread since it is not ThreadSafeRefCounted.
152 [commandBuffer addCompletedHandler:makeBlockPtr([this, protectedThis = makeRef(*this)](id<MTLCommandBuffer>) mutable {
153 callOnMainThread([this, protectedThis = WTFMove(protectedThis)] {
154 commandBufferCompleted();
155 });
156 }).get()];
157 END_BLOCK_OBJC_EXCEPTIONS
158}
159
160void GPUBuffer::commandBufferCompleted()
161{
162 ASSERT(isMainThread());
163 ASSERT(m_numScheduledCommandBuffers);
164
165 if (m_numScheduledCommandBuffers == 1 && state() == State::Mapped)
166 runMappingCallback();
167
168 --m_numScheduledCommandBuffers;
169}
170#endif // USE(METAL)

172void GPUBuffer::registerMappingCallback(MappingCallback&& callback, bool isRead, GPUErrorScopes& errorScopes)
173{
174 // Reject if request is invalid.
175 if (isRead && !isMapReadable()) {
176 errorScopes.generatePrefixedError("Invalid operation!");
177 callback(nullptr);
178 return;
179 }
180 if (!isRead && !isMapWriteable()) {
181 errorScopes.generatePrefixedError("Invalid operation!");
182 callback(nullptr);
183 return;
184 }
185
186 ASSERT(!m_mappingCallback && !m_mappingCallbackTask.isPending());
187
188 // An existing callback means this buffer is in the mapped state.
189 m_mappingCallback = PendingMappingCallback::create(WTFMove(callback));
190
191 // If GPU is not using this buffer, run the callback ASAP.
192 if (!m_numScheduledCommandBuffers) {
193 m_mappingCallbackTask.scheduleTask([this, protectedThis = makeRef(*this)] () mutable {
194 runMappingCallback();
195 });
196 }
197}
198
199void GPUBuffer::runMappingCallback()
200{
201 if (m_mappingCallback)
202 m_mappingCallback->callback(isMapRead() ? stagingBufferForRead() : stagingBufferForWrite());
203}

205JSC::ArrayBuffer* GPUBuffer::stagingBufferForRead()
206{
207 if (!m_stagingBuffer)
208 m_stagingBuffer = ArrayBuffer::create(m_platformBuffer.get().contents, m_byteLength);
209 else
210 memcpy(m_stagingBuffer->data(), m_platformBuffer.get().contents, m_byteLength);
211
212 return m_stagingBuffer.get();
213}
214
215JSC::ArrayBuffer* GPUBuffer::stagingBufferForWrite()
216{
217 m_stagingBuffer = ArrayBuffer::create(1, m_byteLength);
218 return m_stagingBuffer.get();
219}

221void GPUBuffer::copyStagingBufferToGPU(GPUErrorScopes* errorScopes)
222{
223 MTLCommandQueue *queue;
224 if (!m_device->tryGetQueue() || !(queue = m_device->tryGetQueue()->platformQueue()))
225 return;
226
227 RetainPtr<MTLBuffer> stagingMtlBuffer;
228
229 BEGIN_BLOCK_OBJC_EXCEPTIONS
230 // GPUBuffer creation validation ensures m_byteSize fits in NSUInteger.
231 stagingMtlBuffer = adoptNS([m_device->platformDevice() newBufferWithBytes:m_stagingBuffer->data() length:static_cast<NSUInteger>(m_byteLength) options:MTLResourceCPUCacheModeDefaultCache]);
232 END_BLOCK_OBJC_EXCEPTIONS
233
234 if (!stagingMtlBuffer && errorScopes) {
235 errorScopes->generateError("", GPUErrorFilter::OutOfMemory);
236 return;
237 }
238
239 BEGIN_BLOCK_OBJC_EXCEPTIONS
240
241 auto commandBuffer = retainPtr([queue commandBuffer]);
242 auto blitEncoder = retainPtr([commandBuffer blitCommandEncoder]);
243
244 [blitEncoder copyFromBuffer:stagingMtlBuffer.get() sourceOffset:0 toBuffer:m_platformBuffer.get() destinationOffset:0 size:static_cast<NSUInteger>(m_byteLength)];
245 [blitEncoder endEncoding];
246 [commandBuffer commit];
247
248 END_BLOCK_OBJC_EXCEPTIONS
249}

251void GPUBuffer::unmap(GPUErrorScopes* errorScopes)
252{
253 if (!m_isMappedFromCreation && !isMappable() && errorScopes) {
254 errorScopes->generatePrefixedError("Invalid operation: GPUBuffer is not mappable!");
255 return;
256 }
257
258 if (m_stagingBuffer) {
259 if (isMappable()) {
260 // MAP_WRITE and MAP_READ buffers have shared, CPU-accessible storage.
261 ASSERT(m_platformBuffer && m_platformBuffer.get().contents);
262 memcpy(m_platformBuffer.get().contents, m_stagingBuffer->data(), m_byteLength);
263 } else if (m_isMappedFromCreation)
264 copyStagingBufferToGPU(errorScopes);
265
266 m_isMappedFromCreation = false;
267 m_stagingBuffer = nullptr;
268 }
269
270 if (m_mappingCallback) {
271 m_mappingCallbackTask.cancelTask();
272 m_mappingCallback->callback(nullptr);
273 m_mappingCallback = nullptr;
274 }
275}
276
277void GPUBuffer::destroy(GPUErrorScopes* errorScopes)
278{
279 if (state() == State::Mapped)
280 unmap(errorScopes);
281
282 m_platformBuffer = nullptr;
283}
