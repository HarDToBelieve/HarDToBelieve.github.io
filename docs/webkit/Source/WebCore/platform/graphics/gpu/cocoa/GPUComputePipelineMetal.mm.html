45static bool trySetMetalFunctions(MTLLibrary *computeMetalLibrary, MTLComputePipelineDescriptor *mtlDescriptor, const String& computeEntryPointName, GPUErrorScopes& errorScopes)
46{
47 BEGIN_BLOCK_OBJC_EXCEPTIONS
48
49 if (!computeMetalLibrary) {
50 errorScopes.generatePrefixedError("MTLLibrary for compute stage does not exist!");
51 return false;
52 }
53
54 auto function = adoptNS([computeMetalLibrary newFunctionWithName:computeEntryPointName]);
55 if (!function) {
56 errorScopes.generatePrefixedError(makeString("Cannot create compute MTLFunction '", computeEntryPointName, "'!"));
57 return false;
58 }
59
60 [mtlDescriptor setComputeFunction:function.get()];
61
62 END_BLOCK_OBJC_EXCEPTIONS
63
64 return true;
65}

67static std::optional<WHLSL::ComputeDimensions> trySetFunctions(const GPUProgrammableStageDescriptor& computeStage, const GPUDevice& device, MTLComputePipelineDescriptor* mtlDescriptor, std::optional<WHLSL::ComputePipelineDescriptor>& whlslDescriptor, GPUErrorScopes& errorScopes)
68{
69 RetainPtr<MTLLibrary> computeLibrary;
70 String computeEntryPoint;
71
72 WHLSL::ComputeDimensions computeDimensions { 1, 1, 1 };
73
74 if (whlslDescriptor) {
75 ASSERT(computeStage.module->whlslModule());
76
77 whlslDescriptor->entryPointName = computeStage.entryPoint;
78
79 auto whlslCompileResult = WHLSL::prepare(*computeStage.module->whlslModule(), *whlslDescriptor);
80 if (!whlslCompileResult) {
81 errorScopes.generatePrefixedError(makeString("WHLSL compile error: ", whlslCompileResult.error()));
82 return std::nullopt;
83 }
84
85 computeDimensions = whlslCompileResult->computeDimensions;
86
87 NSError *error = nil;
88
89 BEGIN_BLOCK_OBJC_EXCEPTIONS
90 MonotonicTime startTime;
91 if (WHLSL::dumpMetalCompileTimes)
92 startTime = MonotonicTime::now();
93 // FIXME: https://webkit.org/b/200474 Add direct StringBuilder -> NSString conversion to avoid extra copy into a WTF::String
94 computeLibrary = adoptNS([device.platformDevice() newLibraryWithSource:whlslCompileResult->metalSource.toString() options:nil error:&error]);
95 if (WHLSL::dumpMetalCompileTimes)
96 dataLogLn("Metal compile times: ", (MonotonicTime::now() - startTime).milliseconds(), " ms");
97 END_BLOCK_OBJC_EXCEPTIONS
98#ifndef NDEBUG
99 if (!computeLibrary)
100 NSLog(@"%@", error);
101#endif
102 ASSERT(computeLibrary);
103 // FIXME: https://bugs.webkit.org/show_bug.cgi?id=195771 Once we zero-fill variables, there should be no warnings, so we should be able to ASSERT(!error) here.
104
105 computeEntryPoint = whlslCompileResult->mangledEntryPointName.toString();
106 } else {
107 computeLibrary = computeStage.module->platformShaderModule();
108 computeEntryPoint = computeStage.entryPoint;
109 }
110
111 if (trySetMetalFunctions(computeLibrary.get(), mtlDescriptor, computeEntryPoint, errorScopes))
112 return computeDimensions;
113
114 return std::nullopt;
115}

117struct ConvertResult {
118 RetainPtr<MTLComputePipelineDescriptor> pipelineDescriptor;
119 WHLSL::ComputeDimensions computeDimensions;
120};

122static std::optional<ConvertResult> convertComputePipelineDescriptor(const GPUComputePipelineDescriptor& descriptor, const GPUDevice& device, GPUErrorScopes& errorScopes)
123{
124 RetainPtr<MTLComputePipelineDescriptor> mtlDescriptor;
125
126 BEGIN_BLOCK_OBJC_EXCEPTIONS
127 mtlDescriptor = adoptNS([MTLComputePipelineDescriptor new]);
128 END_BLOCK_OBJC_EXCEPTIONS
129
130 if (!mtlDescriptor) {
131 errorScopes.generatePrefixedError("Error creating MTLComputePipelineDescriptor!");
132 return std::nullopt;
133 }
134
135 const auto& computeStage = descriptor.computeStage;
136
137 bool isWhlsl = computeStage.module->whlslModule();
138
139 std::optional<WHLSL::ComputePipelineDescriptor> whlslDescriptor;
140 if (isWhlsl)
141 whlslDescriptor = WHLSL::ComputePipelineDescriptor();
142
143 if (descriptor.layout && whlslDescriptor) {
144 if (auto layout = convertLayout(*descriptor.layout))
145 whlslDescriptor->layout = WTFMove(*layout);
146 else {
147 errorScopes.generatePrefixedError("Error converting GPUPipelineLayout!");
148 return std::nullopt;
149 }
150 }
151
152 if (auto computeDimensions = trySetFunctions(computeStage, device, mtlDescriptor.get(), whlslDescriptor, errorScopes))
153 return {{ mtlDescriptor, *computeDimensions }};
154
155 return std::nullopt;
156}

158struct CreateResult {
159 RetainPtr<MTLComputePipelineState> pipelineState;
160 WHLSL::ComputeDimensions computeDimensions;
161};

163static std::optional<CreateResult> tryCreateMTLComputePipelineState(const GPUDevice& device, const GPUComputePipelineDescriptor& descriptor, GPUErrorScopes& errorScopes)
164{
165 if (!device.platformDevice()) {
166 errorScopes.generatePrefixedError("Invalid GPUDevice!");
167 return std::nullopt;
168 }
169
170 auto convertResult = convertComputePipelineDescriptor(descriptor, device, errorScopes);
171 if (!convertResult)
172 return std::nullopt;
173 ASSERT(convertResult->pipelineDescriptor);
174 auto mtlDescriptor = convertResult->pipelineDescriptor;
175
176 RetainPtr<MTLComputePipelineState> pipeline;
177
178 BEGIN_BLOCK_OBJC_EXCEPTIONS
179
180 NSError *error = nil;
181 pipeline = adoptNS([device.platformDevice() newComputePipelineStateWithDescriptor:mtlDescriptor.get() options:MTLPipelineOptionNone reflection:nil error:&error]);
182 if (!pipeline) {
183 errorScopes.generatePrefixedError(error ? error.localizedDescription.UTF8String : "Unable to create MTLComputePipelineState!");
184 return std::nullopt;
185 }
186
187 END_BLOCK_OBJC_EXCEPTIONS
188
189 return {{ pipeline, convertResult->computeDimensions }};
190}

192RefPtr<GPUComputePipeline> GPUComputePipeline::tryCreate(const GPUDevice& device, const GPUComputePipelineDescriptor& descriptor, GPUErrorScopes& errorScopes)
193{
194 auto createResult = tryCreateMTLComputePipelineState(device, descriptor, errorScopes);
195 if (!createResult)
196 return nullptr;
197
198 return adoptRef(new GPUComputePipeline(WTFMove(createResult->pipelineState), createResult->computeDimensions, descriptor.layout));
199}

211bool GPUComputePipeline::recompile(const GPUDevice& device, GPUProgrammableStageDescriptor&& computeStage)
212{
213 GPUComputePipelineDescriptor descriptor(makeRefPtr(m_layout.get()), WTFMove(computeStage));
214 auto errorScopes = GPUErrorScopes::create([] (GPUError&&) { });
215 if (auto createResult = tryCreateMTLComputePipelineState(device, descriptor, errorScopes)) {
216 m_platformComputePipeline = WTFMove(createResult->pipelineState);
217 m_computeDimensions = createResult->computeDimensions;
218 return true;
219 }
220
221 return false;
222}
