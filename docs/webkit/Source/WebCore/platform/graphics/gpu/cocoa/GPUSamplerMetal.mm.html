40static MTLSamplerAddressMode mtlAddressModeForAddressMode(GPUAddressMode mode)
41{
42 switch (mode) {
43 case GPUAddressMode::ClampToEdge:
44 return MTLSamplerAddressModeClampToEdge;
45 case GPUAddressMode::Repeat:
46 return MTLSamplerAddressModeRepeat;
47 case GPUAddressMode::MirrorRepeat:
48 return MTLSamplerAddressModeMirrorRepeat;
49 }
50
51 ASSERT_NOT_REACHED();
52}
53
54static MTLSamplerMinMagFilter mtlMinMagFilterForFilterMode(GPUFilterMode mode)
55{
56 switch (mode) {
57 case GPUFilterMode::Nearest:
58 return MTLSamplerMinMagFilterNearest;
59 case GPUFilterMode::Linear:
60 return MTLSamplerMinMagFilterLinear;
61 }
62
63 ASSERT_NOT_REACHED();
64}

66static MTLSamplerMipFilter mtlMipFilterForFilterMode(GPUFilterMode mode)

70 return MTLSamplerMipFilterNearest;

78static RetainPtr<MTLSamplerState> tryCreateMtlSamplerState(const GPUDevice& device, const GPUSamplerDescriptor& descriptor)
79{
80 if (!device.platformDevice()) {
81 LOG(WebGPU, "GPUSampler::tryCreate(): Invalid GPUDevice!");
82 return nullptr;
83 }
84
85 RetainPtr<MTLSamplerDescriptor> mtlDescriptor;
86
87 BEGIN_BLOCK_OBJC_EXCEPTIONS
88 mtlDescriptor = adoptNS([MTLSamplerDescriptor new]);
89 END_BLOCK_OBJC_EXCEPTIONS
90
91 if (!mtlDescriptor) {
92 LOG(WebGPU, "GPUSampler::tryCreate(): Error creating MTLSamplerDescriptor!");
93 return nullptr;
94 }
95
96 RetainPtr<MTLSamplerState> sampler;
97
98 BEGIN_BLOCK_OBJC_EXCEPTIONS
99 [mtlDescriptor setRAddressMode:mtlAddressModeForAddressMode(descriptor.addressModeU)];
100 [mtlDescriptor setSAddressMode:mtlAddressModeForAddressMode(descriptor.addressModeV)];
101 [mtlDescriptor setTAddressMode:mtlAddressModeForAddressMode(descriptor.addressModeW)];
102 [mtlDescriptor setMinFilter:mtlMinMagFilterForFilterMode(descriptor.minFilter)];
103 [mtlDescriptor setMagFilter:mtlMinMagFilterForFilterMode(descriptor.magFilter)];
104 [mtlDescriptor setMipFilter:mtlMipFilterForFilterMode(descriptor.mipmapFilter)];
105 [mtlDescriptor setLodMinClamp:descriptor.lodMinClamp];
106 [mtlDescriptor setLodMaxClamp:descriptor.lodMaxClamp];
107 [mtlDescriptor setMaxAnisotropy:descriptor.maxAnisotropy];
108 [mtlDescriptor setCompareFunction:static_cast<MTLCompareFunction>(platformCompareFunctionForGPUCompareFunction(descriptor.compareFunction))];
109
110 [mtlDescriptor setSupportArgumentBuffers:YES];
111
112 sampler = adoptNS([device.platformDevice() newSamplerStateWithDescriptor:mtlDescriptor.get()]);
113 END_BLOCK_OBJC_EXCEPTIONS
114
115 if (!sampler)
116 LOG(WebGPU, "GPUSampler::tryCreate(): Error creating MTLSamplerState!");
117
118 return sampler;
119}
120
121RefPtr<GPUSampler> GPUSampler::tryCreate(const GPUDevice& device, const GPUSamplerDescriptor& descriptor)
122{
123 auto sampler = tryCreateMtlSamplerState(device, descriptor);
124 if (!sampler)
125 return nullptr;
126
127 return adoptRef(new GPUSampler(WTFMove(sampler)));
128}
