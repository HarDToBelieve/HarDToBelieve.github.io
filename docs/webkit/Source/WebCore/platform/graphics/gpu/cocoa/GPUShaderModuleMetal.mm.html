41RefPtr<GPUShaderModule> GPUShaderModule::tryCreate(const GPUDevice& device, const GPUShaderModuleDescriptor& descriptor)
42{
43 if (!device.platformDevice()) {
44 LOG(WebGPU, "GPUShaderModule::create(): Invalid GPUDevice!");
45 return nullptr;
46 }
47
48 if (GPUDevice::useWHLSL)
49 return adoptRef(new GPUShaderModule(WHLSL::createShaderModule(descriptor.code)));
50
51 PlatformShaderModuleSmartPtr module;
52
53 BEGIN_BLOCK_OBJC_EXCEPTIONS
54
55 NSError *error = [NSError errorWithDomain:@"com.apple.WebKit.GPU" code:1 userInfo:nil];
56 module = adoptNS([device.platformDevice() newLibraryWithSource:descriptor.code options:nil error:&error]);
57 if (!module)
58 LOG(WebGPU, "Shader compilation error: %s", [[error localizedDescription] UTF8String]);
59
60 END_BLOCK_OBJC_EXCEPTIONS
61
62 return module ? adoptRef(new GPUShaderModule(WTFMove(module))) : nullptr;
63}
