35PlatformTextureFormat platformTextureFormatForGPUTextureFormat(GPUTextureFormat format)
36{
37 switch (format) {
38 case GPUTextureFormat::Rgba8unorm:
39 return MTLPixelFormatRGBA8Unorm;
40 case GPUTextureFormat::Rgba8uint:
41 return MTLPixelFormatRGBA8Uint;
42 case GPUTextureFormat::Bgra8unorm:
43 return MTLPixelFormatBGRA8Unorm;
44 case GPUTextureFormat::Depth32floatStencil8:
45 return MTLPixelFormatDepth32Float_Stencil8;
46 case GPUTextureFormat::Bgra8unormSRGB:
47 return MTLPixelFormatBGRA8Unorm_sRGB;
48 case GPUTextureFormat::Rgba16float:
49 return MTLPixelFormatRGBA16Float;
50 }
51
52 ASSERT_NOT_REACHED();
53}
54
55PlatformCompareFunction platformCompareFunctionForGPUCompareFunction(GPUCompareFunction func)
56{
57 switch (func) {
58 case GPUCompareFunction::Never:
59 return MTLCompareFunctionNever;
60 case GPUCompareFunction::Less:
61 return MTLCompareFunctionLess;
62 case GPUCompareFunction::Equal:
63 return MTLCompareFunctionEqual;
64 case GPUCompareFunction::LessEqual:
65 return MTLCompareFunctionLessEqual;
66 case GPUCompareFunction::Greater:
67 return MTLCompareFunctionGreater;
68 case GPUCompareFunction::NotEqual:
69 return MTLCompareFunctionNotEqual;
70 case GPUCompareFunction::GreaterEqual:
71 return MTLCompareFunctionGreaterEqual;
72 case GPUCompareFunction::Always:
73 return MTLCompareFunctionAlways;
74 }
75
76 ASSERT_NOT_REACHED();
77}
