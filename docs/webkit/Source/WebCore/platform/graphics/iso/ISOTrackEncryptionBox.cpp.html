35bool ISOTrackEncryptionBox::parse(DataView& view, unsigned& offset)
36{
37 // ISO/IEC 23001-7-2015 Section 8.2.2
38 if (!ISOFullBox::parse(view, offset))
39 return false;
40
41 // unsigned int(8) reserved = 0;
42 offset += 1;
43
44 if (!m_version) {
45 // unsigned int(8) reserved = 0;
46 offset += 1;
47 } else {
48 int8_t cryptAndSkip = 0;
49 if (!checkedRead<int8_t>(cryptAndSkip, view, offset, BigEndian))
50 return false;
51
52 m_defaultCryptByteBlock = cryptAndSkip >> 4;
53 m_defaultSkipByteBlock = cryptAndSkip & 0xF;
54 }
55
56 if (!checkedRead<int8_t>(m_defaultIsProtected, view, offset, BigEndian))
57 return false;
58
59 if (!checkedRead<int8_t>(m_defaultPerSampleIVSize, view, offset, BigEndian))
60 return false;
61
62 auto buffer = view.possiblySharedBuffer();
63 if (!buffer)
64 return false;
65
66 auto keyIDBuffer = buffer->slice(offset, offset + 16);
67 offset += 16;
68
69 m_defaultKID.resize(16);
70 memcpy(m_defaultKID.data(), keyIDBuffer->data(), 16);
71
72 if (m_defaultIsProtected == 1 && !m_defaultPerSampleIVSize) {
73 int8_t defaultConstantIVSize = 0;
74 if (!checkedRead<int8_t>(defaultConstantIVSize, view, offset, BigEndian))
75 return false;
76
77 Vector<uint8_t> defaultConstantIV;
78 defaultConstantIV.reserveInitialCapacity(defaultConstantIVSize);
79 while (defaultConstantIVSize--) {
80 int8_t character = 0;
81 if (!checkedRead<int8_t>(character, view, offset, BigEndian))
82 return false;
83 defaultConstantIV.uncheckedAppend(character);
84 }
85 m_defaultConstantIV = WTFMove(defaultConstantIV);
86 }
87
88 return true;
89}
