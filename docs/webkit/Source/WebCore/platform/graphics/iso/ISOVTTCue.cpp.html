38class ISOStringBox final : public ISOBox {
39public:
40 const String& contents() { return m_contents; }
41
42private:
43 bool parse(JSC::DataView& view, unsigned& offset) override
44 {
45 unsigned localOffset = offset;
46 if (!ISOBox::parse(view, localOffset))
47 return false;
48
49 auto characterCount = m_size - (localOffset - offset);
50 if (!characterCount) {
51 m_contents = emptyString();
52 return true;
53 }
54
55 Vector<LChar> characters;
56 characters.reserveInitialCapacity(static_cast<size_t>(characterCount));
57 while (characterCount--) {
58 int8_t character = 0;
59 if (!checkedRead<int8_t>(character, view, localOffset, BigEndian))
60 return false;
61 characters.uncheckedAppend(character);
62 }
63
64 m_contents = String::fromUTF8(characters);
65 offset = localOffset;
66 return true;
67 }
68 String m_contents;
69};

94bool ISOWebVTTCue::parse(DataView& view, unsigned& offset)
95{
96 if (!ISOBox::parse(view, offset))
97 return false;
98
99 ISOStringBox stringBox;
100
101 while (stringBox.read(view, offset)) {
102 if (stringBox.boxType() == vttCueSourceIDBoxType())
103 m_sourceID = stringBox.contents();
104 else if (stringBox.boxType() == vttIdBoxType())
105 m_identifier = stringBox.contents();
106 else if (stringBox.boxType() == vttCurrentTimeBoxType())
107 m_originalStartTime = stringBox.contents();
108 else if (stringBox.boxType() == vttSettingsBoxType())
109 m_settings = stringBox.contents();
110 else if (stringBox.boxType() == vttPayloadBoxType())
111 m_cueText = stringBox.contents();
112 else
113 LOG(Media, "ISOWebVTTCue::ISOWebVTTCue - skipping box id = \"%s\", size = %zu", stringBox.boxType().toString().utf8().data(), (size_t)stringBox.size());
114 }
115 return true;
116}

118String ISOWebVTTCue::toJSONString() const
119{
120 auto object = JSON::Object::create();
121
122#if !LOG_DISABLED
123 object->setString("text"_s, m_cueText);
124#endif
125 object->setString("sourceId"_s, encodeWithURLEscapeSequences(m_sourceID));
126 object->setString("id"_s, encodeWithURLEscapeSequences(m_identifier));
127
128 object->setString("originalStartTime"_s, encodeWithURLEscapeSequences(m_originalStartTime));
129 object->setString("settings"_s, encodeWithURLEscapeSequences(m_settings));
130
131 object->setDouble("presentationTime"_s, m_presentationTime.toDouble());
132 object->setDouble("duration"_s, m_duration.toDouble());
133
134 return object->toJSONString();
135}
