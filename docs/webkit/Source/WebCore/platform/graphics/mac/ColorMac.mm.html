67static std::optional<SRGBA<uint8_t>> makeSimpleColorFromNSColor(NSColor *color)
68{
69 // FIXME: ExtendedColor - needs to handle color spaces.
70
71 if (!color)
72 return std::nullopt;
73
74 CGFloat redComponent;
75 CGFloat greenComponent;
76 CGFloat blueComponent;
77 CGFloat alpha;
78
79 BEGIN_BLOCK_OBJC_EXCEPTIONS
80 NSColor *rgbColor = [color colorUsingColorSpace:NSColorSpace.deviceRGBColorSpace];
81 if (!rgbColor) {
82 // The color space conversion above can fail if the NSColor is in the NSPatternColorSpace.
83 // These colors are actually a repeating pattern, not just a solid color. To workaround
84 // this we simply draw a one pixel image of the color and use that pixel's color.
85 // FIXME: It might be better to use an average of the colors in the pattern instead.
86 RetainPtr<NSBitmapImageRep> offscreenRep = adoptNS([[NSBitmapImageRep alloc] initWithBitmapDataPlanes:nil pixelsWide:1 pixelsHigh:1
87 bitsPerSample:8 samplesPerPixel:4 hasAlpha:YES isPlanar:NO colorSpaceName:NSDeviceRGBColorSpace bytesPerRow:4 bitsPerPixel:32]);
88
89 GraphicsContextCG bitmapContext([NSGraphicsContext graphicsContextWithBitmapImageRep:offscreenRep.get()].CGContext);
90 LocalCurrentGraphicsContext localContext(bitmapContext);
91
92 [color drawSwatchInRect:NSMakeRect(0, 0, 1, 1)];
93
94 NSUInteger pixel[4];
95 [offscreenRep getPixel:pixel atX:0 y:0];
96
97 return makeFromComponentsClamping<SRGBA<uint8_t>>(pixel[0], pixel[1], pixel[2], pixel[3]);
98 }
99
100 [rgbColor getRed:&redComponent green:&greenComponent blue:&blueComponent alpha:&alpha];
101 END_BLOCK_OBJC_EXCEPTIONS
102
103 return convertColor<SRGBA<uint8_t>>(SRGBA<float> { static_cast<float>(redComponent), static_cast<float>(greenComponent), static_cast<float>(blueComponent), static_cast<float>(alpha) });
104}

116NSColor *nsColor(const Color& color)
117{
118 if (auto srgb = color.tryGetAsSRGBABytes()) {
119 switch (PackedColor::RGBA { *srgb }.value) {
120 case PackedColor::RGBA { Color::transparentBlack }.value: {
121 static LazyNeverDestroyed<RetainPtr<NSColor>> clearColor;
122 static std::once_flag onceFlag;
123 std::call_once(onceFlag, [] {
124 clearColor.construct([NSColor colorWithSRGBRed:0 green:0 blue:0 alpha:0]);
125 });
126 return clearColor.get().get();
127 }
128 case PackedColor::RGBA { Color::black }.value: {
129 static LazyNeverDestroyed<RetainPtr<NSColor>> blackColor;
130 static std::once_flag onceFlag;
131 std::call_once(onceFlag, [] {
132 blackColor.construct([NSColor colorWithSRGBRed:0 green:0 blue:0 alpha:1]);
133 });
134 return blackColor.get().get();
135 }
136 case PackedColor::RGBA { Color::white }.value: {
137 static LazyNeverDestroyed<RetainPtr<NSColor>> whiteColor;
138 static std::once_flag onceFlag;
139 std::call_once(onceFlag, [] {
140 whiteColor.construct([NSColor colorWithSRGBRed:1 green:1 blue:1 alpha:1]);
141 });
142 return whiteColor.get().get();
143 }
144 }
145 }
146
147 static Lock cachedColorLock;
148 Locker locker { cachedColorLock };
149
150 static NeverDestroyed<TinyLRUCache<Color, RetainPtr<NSColor>, 32>> cache;
151 return cache.get().get(color).get();
152}
