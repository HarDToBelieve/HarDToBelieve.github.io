37ComplexTextController::ComplexTextRun::ComplexTextRun(CTRunRef ctRun, const Font& font, const UChar* characters, unsigned stringLocation, unsigned stringLength, unsigned indexBegin, unsigned indexEnd)
38 : m_initialAdvance(CTRunGetInitialAdvance(ctRun))
39 , m_font(font)
40 , m_characters(characters)
41 , m_stringLength(stringLength)
42 , m_indexBegin(indexBegin)
43 , m_indexEnd(indexEnd)
44 , m_glyphCount(CTRunGetGlyphCount(ctRun))
45 , m_stringLocation(stringLocation)
46 , m_isLTR(!(CTRunGetStatus(ctRun) & kCTRunStatusRightToLeft))
47{
48 const CFIndex* coreTextIndicesPtr = CTRunGetStringIndicesPtr(ctRun);
49 Vector<CFIndex> coreTextIndices;
50 if (!coreTextIndicesPtr) {
51 coreTextIndices.grow(m_glyphCount);
52 CTRunGetStringIndices(ctRun, CFRangeMake(0, 0), coreTextIndices.data());
53 coreTextIndicesPtr = coreTextIndices.data();
54 }
55 m_coreTextIndices.reserveInitialCapacity(m_glyphCount);
56 for (unsigned i = 0; i < m_glyphCount; ++i)
57 m_coreTextIndices.uncheckedAppend(coreTextIndicesPtr[i]);
58
59 const CGGlyph* glyphsPtr = CTRunGetGlyphsPtr(ctRun);
60 Vector<CGGlyph> glyphs;
61 if (!glyphsPtr) {
62 glyphs.grow(m_glyphCount);
63 CTRunGetGlyphs(ctRun, CFRangeMake(0, 0), glyphs.data());
64 glyphsPtr = glyphs.data();
65 }
66 m_glyphs.reserveInitialCapacity(m_glyphCount);
67 for (unsigned i = 0; i < m_glyphCount; ++i)
68 m_glyphs.uncheckedAppend(glyphsPtr[i]);
69
70 if (CTRunGetStatus(ctRun) & kCTRunStatusHasOrigins) {
71 Vector<CGSize> baseAdvances(m_glyphCount);
72 Vector<CGPoint> glyphOrigins(m_glyphCount);
73 CTRunGetBaseAdvancesAndOrigins(ctRun, CFRangeMake(0, 0), baseAdvances.data(), glyphOrigins.data());
74 m_baseAdvances.reserveInitialCapacity(m_glyphCount);
75 m_glyphOrigins.reserveInitialCapacity(m_glyphCount);
76 for (unsigned i = 0; i < m_glyphCount; ++i) {
77 m_baseAdvances.uncheckedAppend(baseAdvances[i]);
78 m_glyphOrigins.uncheckedAppend(glyphOrigins[i]);
79 }
80 } else {
81 const CGSize* baseAdvances = CTRunGetAdvancesPtr(ctRun);
82 Vector<CGSize> baseAdvancesVector;
83 if (!baseAdvances) {
84 baseAdvancesVector.grow(m_glyphCount);
85 CTRunGetAdvances(ctRun, CFRangeMake(0, 0), baseAdvancesVector.data());
86 baseAdvances = baseAdvancesVector.data();
87 }
88 m_baseAdvances.reserveInitialCapacity(m_glyphCount);
89 for (unsigned i = 0; i < m_glyphCount; ++i)
90 m_baseAdvances.uncheckedAppend(baseAdvances[i]);
91 }

94struct ProviderInfo {
95 const UChar* cp;
96 unsigned length;
97 CFDictionaryRef attributes;
98};
99
100static const UniChar* provideStringAndAttributes(CFIndex stringIndex, CFIndex* charCount, CFDictionaryRef* attributes, void* refCon)
101{
102 ProviderInfo* info = static_cast<struct ProviderInfo*>(refCon);
103 if (stringIndex < 0 || static_cast<unsigned>(stringIndex) >= info->length)
104 return 0;
105
106 *charCount = info->length - stringIndex;
107 *attributes = info->attributes;
108 return reinterpret_cast<const UniChar*>(info->cp + stringIndex);
109}

111void ComplexTextController::collectComplexTextRunsForCharacters(const UChar* cp, unsigned length, unsigned stringLocation, const Font* font)
112{
113 if (!font) {
114 // Create a run of missing glyphs from the primary font.
115 m_complexTextRuns.append(ComplexTextRun::create(m_font.primaryFont(), cp, stringLocation, length, 0, length, m_run.ltr()));
116 return;
117 }
118
119 bool isSystemFallback = false;
120
121 UChar32 baseCharacter = 0;
122 RetainPtr<CFDictionaryRef> stringAttributes;
123 if (font == Font::systemFallback()) {
124 // FIXME: This code path does not support small caps.
125 isSystemFallback = true;
126
127 U16_GET(cp, 0, 0, length, baseCharacter);
128 font = m_font.fallbackRangesAt(0).fontForCharacter(baseCharacter);
129 if (!font)
130 font = &m_font.fallbackRangesAt(0).fontForFirstRange();
131 stringAttributes = adoptCF(CFDictionaryCreateMutableCopy(kCFAllocatorDefault, 0, font->getCFStringAttributes(m_font.enableKerning(), font->platformData().orientation(), m_font.fontDescription().computedLocale()).get()));
132 // We don't know which font should be used to render this grapheme cluster, so enable CoreText's fallback mechanism by using the CTFont which doesn't have CoreText's fallback disabled.
133 CFDictionarySetValue(const_cast<CFMutableDictionaryRef>(stringAttributes.get()), kCTFontAttributeName, font->platformData().font());
134 } else
135 stringAttributes = font->getCFStringAttributes(m_font.enableKerning(), font->platformData().orientation(), m_font.fontDescription().computedLocale());
136
137 RetainPtr<CTLineRef> line;
138
139 if (!m_mayUseNaturalWritingDirection || m_run.directionalOverride()) {
140 const short ltrForcedEmbeddingLevelValue = 0;
141 const short rtlForcedEmbeddingLevelValue = 1;
142 static const void* optionKeys[] = { kCTTypesetterOptionForcedEmbeddingLevel };
143 static const void* ltrOptionValues[] = { CFNumberCreate(kCFAllocatorDefault, kCFNumberShortType, &ltrForcedEmbeddingLevelValue) };
144 static const void* rtlOptionValues[] = { CFNumberCreate(kCFAllocatorDefault, kCFNumberShortType, &rtlForcedEmbeddingLevelValue) };
145 static NeverDestroyed<RetainPtr<CFDictionaryRef>> ltrTypesetterOptions = adoptCF(CFDictionaryCreate(kCFAllocatorDefault, optionKeys, ltrOptionValues, WTF_ARRAY_LENGTH(optionKeys), &kCFCopyStringDictionaryKeyCallBacks, &kCFTypeDictionaryValueCallBacks));
146 static NeverDestroyed<RetainPtr<CFDictionaryRef>> rtlTypesetterOptions = adoptCF(CFDictionaryCreate(kCFAllocatorDefault, optionKeys, rtlOptionValues, WTF_ARRAY_LENGTH(optionKeys), &kCFCopyStringDictionaryKeyCallBacks, &kCFTypeDictionaryValueCallBacks));
147
148 ProviderInfo info = { cp, length, stringAttributes.get() };
149 // FIXME: Some SDKs complain that the second parameter below cannot be null.
150 IGNORE_NULL_CHECK_WARNINGS_BEGIN
151 RetainPtr<CTTypesetterRef> typesetter = adoptCF(CTTypesetterCreateWithUniCharProviderAndOptions(&provideStringAndAttributes, 0, &info, m_run.ltr() ? ltrTypesetterOptions.get().get() : rtlTypesetterOptions.get().get()));
152 IGNORE_NULL_CHECK_WARNINGS_END
153
154 if (!typesetter)
155 return;
156
157 line = adoptCF(CTTypesetterCreateLine(typesetter.get(), CFRangeMake(0, 0)));
158 } else {
159 ProviderInfo info = { cp, length, stringAttributes.get() };
160
161 line = adoptCF(CTLineCreateWithUniCharProvider(&provideStringAndAttributes, nullptr, &info));
162 }
163
164 if (!line)
165 return;
166
167 m_coreTextLines.append(line.get());
168
169 CFArrayRef runArray = CTLineGetGlyphRuns(line.get());
170
171 if (!runArray)
172 return;
173
174 CFIndex runCount = CFArrayGetCount(runArray);
175
176 for (CFIndex r = 0; r < runCount; r++) {
177 CTRunRef ctRun = static_cast<CTRunRef>(CFArrayGetValueAtIndex(runArray, m_run.ltr() ? r : runCount - 1 - r));
178 ASSERT(CFGetTypeID(ctRun) == CTRunGetTypeID());
179 CFRange runRange = CTRunGetStringRange(ctRun);
180 const Font* runFont = font;
181 // If isSystemFallback is false, it means we disabled CoreText's font fallback mechanism, which means all the runs must use this exact font.
182 // Therefore, we only need to inspect which font was actually used if isSystemFallback is true.
183 if (isSystemFallback) {
184 CFDictionaryRef runAttributes = CTRunGetAttributes(ctRun);
185 CTFontRef runCTFont = static_cast<CTFontRef>(CFDictionaryGetValue(runAttributes, kCTFontAttributeName));
186 ASSERT(runCTFont && CFGetTypeID(runCTFont) == CTFontGetTypeID());
187 RetainPtr<CFTypeRef> runFontEqualityObject = FontPlatformData::objectForEqualityCheck(runCTFont);
188 if (!safeCFEqual(runFontEqualityObject.get(), font->platformData().objectForEqualityCheck().get())) {
189 // Begin trying to see if runFont matches any of the fonts in the fallback list.
190 for (unsigned i = 0; !m_font.fallbackRangesAt(i).isNull(); ++i) {
191 runFont = m_font.fallbackRangesAt(i).fontForCharacter(baseCharacter);
192 if (!runFont)
193 continue;
194 if (safeCFEqual(runFont->platformData().objectForEqualityCheck().get(), runFontEqualityObject.get()))
195 break;
196 runFont = nullptr;
197 }
198 if (!runFont) {
199 RetainPtr<CFStringRef> fontName = adoptCF(CTFontCopyPostScriptName(runCTFont));
200 if (CFEqual(fontName.get(), CFSTR("LastResort"))) {
201 m_complexTextRuns.append(ComplexTextRun::create(m_font.primaryFont(), cp, stringLocation, length, runRange.location, runRange.location + runRange.length, m_run.ltr()));
202 continue;
203 }
204 FontPlatformData runFontPlatformData(runCTFont, CTFontGetSize(runCTFont));
205 runFont = FontCache::singleton().fontForPlatformData(runFontPlatformData).ptr();
206 }
207 if (m_fallbackFonts && runFont != &m_font.primaryFont())
208 m_fallbackFonts->add(runFont);
209 }
210 }
211 if (m_fallbackFonts && runFont != &m_font.primaryFont())
212 m_fallbackFonts->add(font);
213
214 m_complexTextRuns.append(ComplexTextRun::create(ctRun, *runFont, cp, stringLocation, length, runRange.location, runRange.location + runRange.length));
215 }
