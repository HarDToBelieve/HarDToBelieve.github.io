38FontPlatformData FontCustomPlatformData::fontPlatformData(const FontDescription& fontDescription, bool bold, bool italic, const FontFeatureSettings& fontFaceFeatures, FontSelectionSpecifiedCapabilities fontFaceCapabilities)
39{
40 auto attributes = adoptCF(CFDictionaryCreateMutable(kCFAllocatorDefault, 0, &kCFTypeDictionaryKeyCallBacks, &kCFTypeDictionaryValueCallBacks));
41 addAttributesForWebFonts(attributes.get(), fontDescription.shouldAllowUserInstalledFonts());
42 auto modifiedFontDescriptor = adoptCF(CTFontDescriptorCreateCopyWithAttributes(fontDescriptor.get(), attributes.get()));
43 ASSERT(modifiedFontDescriptor);
44
45 int size = fontDescription.computedPixelSize();
46 FontOrientation orientation = fontDescription.orientation();
47 FontWidthVariant widthVariant = fontDescription.widthVariant();
48 auto font = adoptCF(CTFontCreateWithFontDescriptor(modifiedFontDescriptor.get(), size, nullptr));
49 font = preparePlatformFont(font.get(), fontDescription, &fontFaceFeatures, fontFaceCapabilities);
50 ASSERT(font);
51 return FontPlatformData(font.get(), size, bold, italic, orientation, widthVariant, fontDescription.textRenderingMode(), &creationData);
52}

54std::unique_ptr<FontCustomPlatformData> createFontCustomPlatformData(SharedBuffer& buffer, const String& itemInCollection)
55{
56 RetainPtr<CFDataRef> bufferData = buffer.createCFData();
57
58 RetainPtr<CTFontDescriptorRef> fontDescriptor;
59 auto array = adoptCF(CTFontManagerCreateFontDescriptorsFromData(bufferData.get()));
60 if (!array)
61 return nullptr;
62 auto length = CFArrayGetCount(array.get());
63 if (length <= 0)
64 return nullptr;
65 if (!itemInCollection.isNull()) {
66 if (auto desiredName = itemInCollection.createCFString()) {
67 for (CFIndex i = 0; i < length; ++i) {
68 auto candidate = static_cast<CTFontDescriptorRef>(CFArrayGetValueAtIndex(array.get(), i));
69 auto postScriptName = adoptCF(static_cast<CFStringRef>(CTFontDescriptorCopyAttribute(candidate, kCTFontNameAttribute)));
70 if (CFStringCompare(postScriptName.get(), desiredName.get(), 0) == kCFCompareEqualTo) {
71 fontDescriptor = candidate;
72 break;
73 }
74 }
75 }
76 }
77 if (!fontDescriptor)
78 fontDescriptor = static_cast<CTFontDescriptorRef>(CFArrayGetValueAtIndex(array.get(), 0));
79
80 FontPlatformData::CreationData creationData = { buffer, itemInCollection };
81 return makeUnique<FontCustomPlatformData>(fontDescriptor.get(), WTFMove(creationData));
82}

84bool FontCustomPlatformData::supportsFormat(const String& format)
85{
86 return equalLettersIgnoringASCIICase(format, "truetype")
87 || equalLettersIgnoringASCIICase(format, "opentype")
88 || equalLettersIgnoringASCIICase(format, "woff2")
89 || equalLettersIgnoringASCIICase(format, "woff2-variations")
90 || equalLettersIgnoringASCIICase(format, "woff-variations")
91 || equalLettersIgnoringASCIICase(format, "truetype-variations")
92 || equalLettersIgnoringASCIICase(format, "opentype-variations")
93 || equalLettersIgnoringASCIICase(format, "woff");
94}
