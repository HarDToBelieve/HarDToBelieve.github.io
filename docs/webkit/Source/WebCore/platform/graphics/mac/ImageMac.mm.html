48void BitmapImage::invalidatePlatformData()

59Ref<Image> Image::loadPlatformResource(const char *name)
60{
61 NSBundle *bundle = [NSBundle bundleForClass:[WebCoreBundleFinder class]];
62 NSString *imagePath = [bundle pathForResource:[NSString stringWithUTF8String:name] ofType:@"png"];
63 NSData *namedImageData = [NSData dataWithContentsOfFile:imagePath];
64 if (namedImageData) {
65 auto image = BitmapImage::create();
66 image->setData(SharedBuffer::create(namedImageData), true);
67 return WTFMove(image);
68 }
69
70 // We have reports indicating resource loads are failing, but we don't yet know the root cause(s).
71 // Two theories are bad installs (image files are missing), and too-many-open-files.
72 // See rdar://5607381
73 ASSERT_NOT_REACHED();
74 return Image::nullImage();
75}

77RetainPtr<CFDataRef> BitmapImage::tiffRepresentation(const Vector<Ref<NativeImage>>& nativeImages)
78{
79 // If nativeImages.size() is zero, we know for certain this image doesn't have valid data
80 // Even though the call to CGImageDestinationCreateWithData will fail and we'll handle it gracefully,
81 // in certain circumstances that call will spam the console with an error message
82 if (!nativeImages.size())
83 return nullptr;
84
85 RetainPtr<CFMutableDataRef> data = adoptCF(CFDataCreateMutable(0, 0));
86
87ALLOW_DEPRECATED_DECLARATIONS_BEGIN
88 RetainPtr<CGImageDestinationRef> destination = adoptCF(CGImageDestinationCreateWithData(data.get(), kUTTypeTIFF, nativeImages.size(), 0));
89ALLOW_DEPRECATED_DECLARATIONS_END
90
91 if (!destination)
92 return nullptr;
93
94 for (const auto& nativeImage : nativeImages)
95 CGImageDestinationAddImage(destination.get(), nativeImage->platformImage().get(), 0);
96
97 CGImageDestinationFinalize(destination.get());
98 return data;
99}

101CFDataRef BitmapImage::tiffRepresentation()
102{
103 if (m_tiffRep)
104 return m_tiffRep.get();
105
106 auto data = tiffRepresentation(framesNativeImages());
107 if (!data)
108 return nullptr;
109
110 m_tiffRep = data;
111 return m_tiffRep.get();
112}

114#if USE(APPKIT)
115NSImage* BitmapImage::nsImage()
116{
117 if (m_nsImage)
118 return m_nsImage.get();
119
120 CFDataRef data = tiffRepresentation();
121 if (!data)
122 return nullptr;
123
124 m_nsImage = adoptNS([[NSImage alloc] initWithData:(__bridge NSData *)data]);
125 return m_nsImage.get();
126}

128RetainPtr<NSImage> BitmapImage::snapshotNSImage()
129{
130 auto nativeImage = this->nativeImageForCurrentFrame();
131 if (!nativeImage)
132 return nullptr;
133
134 auto data = tiffRepresentation({ makeRef(*nativeImage) });
135 if (!data)
136 return nullptr;
137
138 return adoptNS([[NSImage alloc] initWithData:(__bridge NSData *)data.get()]);
139}
