31namespace WebCore {
32namespace OpenType {

34#if PLATFORM(WIN)
35static const unsigned long kCTFontTableOS2 = 'OS/2';
36#endif
37
38static inline short readShortFromTable(const UInt8* os2Data, CFIndex offset)
39{
40 return *(reinterpret_cast<const OpenType::Int16*>(os2Data + offset));
41}
42
43bool tryGetTypoMetrics(CTFontRef font, short& ascent, short& descent, short& lineGap)
44{
45 bool result = false;
46 if (auto os2Table = adoptCF(CTFontCopyTable(font, kCTFontTableOS2, kCTFontTableOptionNoOptions))) {
47 // For the structure of the OS/2 table, see
48 // https://developer.apple.com/fonts/TrueType-Reference-Manual/RM06/Chap6OS2.html
49 const CFIndex fsSelectionOffset = 16 * 2 + 10 + 4 * 4 + 4 * 1;
50 const CFIndex sTypoAscenderOffset = fsSelectionOffset + 3 * 2;
51 const CFIndex sTypoDescenderOffset = sTypoAscenderOffset + 2;
52 const CFIndex sTypoLineGapOffset = sTypoDescenderOffset + 2;
53 if (CFDataGetLength(os2Table.get()) >= sTypoLineGapOffset + 2) {
54 const UInt8* os2Data = CFDataGetBytePtr(os2Table.get());
55 // We test the use typo bit on the least significant byte of fsSelection.
56 const UInt8 useTypoMetricsMask = 1 << 7;
57 if (*(os2Data + fsSelectionOffset + 1) & useTypoMetricsMask) {
58 ascent = readShortFromTable(os2Data, sTypoAscenderOffset);
59 descent = readShortFromTable(os2Data, sTypoDescenderOffset);
60 lineGap = readShortFromTable(os2Data, sTypoLineGapOffset);
61 result = true;
62 }
63 }
64 }
65 return result;
66}
