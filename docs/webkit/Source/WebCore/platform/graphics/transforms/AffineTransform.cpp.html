63void AffineTransform::makeIdentity()
64{
65 setMatrix(1, 0, 0, 1, 0, 0);
66}

68void AffineTransform::setMatrix(double a, double b, double c, double d, double e, double f)
69{
70 m_transform[0] = a;
71 m_transform[1] = b;
72 m_transform[2] = c;
73 m_transform[3] = d;
74 m_transform[4] = e;
75 m_transform[5] = f;
76}
77
78bool AffineTransform::isIdentity() const
79{
80 return (m_transform[0] == 1 && m_transform[1] == 0
81 && m_transform[2] == 0 && m_transform[3] == 1
82 && m_transform[4] == 0 && m_transform[5] == 0);
83}
84
85double AffineTransform::xScale() const
86{
87 return std::hypot(m_transform[0], m_transform[1]);
88}
89
90double AffineTransform::yScale() const
91{
92 return std::hypot(m_transform[2], m_transform[3]);
93}

95static double det(const std::array<double, 6>& transform)
96{
97 return transform[0] * transform[3] - transform[1] * transform[2];
98}
99
100bool AffineTransform::isInvertible() const
101{
102 double determinant = det(m_transform);
103
104 return std::isfinite(determinant) && determinant != 0;
105}
106
107std::optional<AffineTransform> AffineTransform::inverse() const
108{
109 double determinant = det(m_transform);
110 if (!std::isfinite(determinant) || determinant == 0)
111 return std::nullopt;
112
113 AffineTransform result;
114 if (isIdentityOrTranslation()) {
115 result.m_transform[4] = -m_transform[4];
116 result.m_transform[5] = -m_transform[5];
117 return result;
118 }
119
120 result.m_transform[0] = m_transform[3] / determinant;
121 result.m_transform[1] = -m_transform[1] / determinant;
122 result.m_transform[2] = -m_transform[2] / determinant;
123 result.m_transform[3] = m_transform[0] / determinant;
124 result.m_transform[4] = (m_transform[2] * m_transform[5]
125 - m_transform[3] * m_transform[4]) / determinant;
126 result.m_transform[5] = (m_transform[1] * m_transform[4]
127 - m_transform[0] * m_transform[5]) / determinant;
128
129 return result;
130}

133// Multiplies this AffineTransform by the provided AffineTransform - i.e.
134// this = this * other;
135AffineTransform& AffineTransform::multiply(const AffineTransform& other)
136{
137 AffineTransform trans;
138
139 trans.m_transform[0] = other.m_transform[0] * m_transform[0] + other.m_transform[1] * m_transform[2];
140 trans.m_transform[1] = other.m_transform[0] * m_transform[1] + other.m_transform[1] * m_transform[3];
141 trans.m_transform[2] = other.m_transform[2] * m_transform[0] + other.m_transform[3] * m_transform[2];
142 trans.m_transform[3] = other.m_transform[2] * m_transform[1] + other.m_transform[3] * m_transform[3];
143 trans.m_transform[4] = other.m_transform[4] * m_transform[0] + other.m_transform[5] * m_transform[2] + m_transform[4];
144 trans.m_transform[5] = other.m_transform[4] * m_transform[1] + other.m_transform[5] * m_transform[3] + m_transform[5];
145
146 *this = trans;
147 return *this;
148}

150AffineTransform& AffineTransform::rotate(double a)
151{
152 // angle is in degree. Switch to radian
153 a = deg2rad(a);
154 double cosAngle = cos(a);
155 double sinAngle = sin(a);
156 AffineTransform rot(cosAngle, sinAngle, -sinAngle, cosAngle, 0, 0);
157
158 multiply(rot);
159 return *this;
160}

186// *this = *this * translation
187AffineTransform& AffineTransform::translate(double tx, double ty)
188{
189 if (isIdentityOrTranslation()) {
190 m_transform[4] += tx;
191 m_transform[5] += ty;
192 return *this;
193 }
194
195 m_transform[4] += tx * m_transform[0] + ty * m_transform[2];
196 m_transform[5] += tx * m_transform[1] + ty * m_transform[3];
197 return *this;
198}

225AffineTransform& AffineTransform::shear(double sx, double sy)
226{
227 double a = m_transform[0];
228 double b = m_transform[1];
229
230 m_transform[0] += sy * m_transform[2];
231 m_transform[1] += sy * m_transform[3];
232 m_transform[2] += sx * a;
233 m_transform[3] += sx * b;
234
235 return *this;
236}

253AffineTransform makeMapBetweenRects(const FloatRect& source, const FloatRect& dest)
254{
255 AffineTransform transform;
256 transform.translate(dest.x() - source.x(), dest.y() - source.y());
257 transform.scale(dest.size() / source.size());
258 return transform;
259}

261void AffineTransform::map(double x, double y, double& x2, double& y2) const
262{
263 x2 = (m_transform[0] * x + m_transform[2] * y + m_transform[4]);
264 y2 = (m_transform[1] * x + m_transform[3] * y + m_transform[5]);
265}

305FloatRect AffineTransform::mapRect(const FloatRect& rect) const
306{
307 if (isIdentityOrTranslation()) {
308 FloatRect mappedRect(rect);
309 mappedRect.move(narrowPrecisionToFloat(m_transform[4]), narrowPrecisionToFloat(m_transform[5]));
310 return mappedRect;
311 }
312
313 FloatQuad result;
314 result.setP1(mapPoint(rect.location()));
315 result.setP2(mapPoint(FloatPoint(rect.maxX(), rect.y())));
316 result.setP3(mapPoint(FloatPoint(rect.maxX(), rect.maxY())));
317 result.setP4(mapPoint(FloatPoint(rect.x(), rect.maxY())));
318 return result.boundingBox();
319}

321FloatQuad AffineTransform::mapQuad(const FloatQuad& q) const
322{
323 if (isIdentityOrTranslation()) {
324 FloatQuad mappedQuad(q);
325 mappedQuad.move(narrowPrecisionToFloat(m_transform[4]), narrowPrecisionToFloat(m_transform[5]));
326 return mappedQuad;
327 }
328
329 FloatQuad result;
330 result.setP1(mapPoint(q.p1()));
331 result.setP2(mapPoint(q.p2()));
332 result.setP3(mapPoint(q.p3()));
333 result.setP4(mapPoint(q.p4()));
334 return result;
335}

337Region AffineTransform::mapRegion(const Region& region) const
338{
339 if (isIdentityOrTranslation()) {
340 Region mappedRegion(region);
341 mappedRegion.translate(roundedIntSize(FloatSize(narrowPrecisionToFloat(m_transform[4]), narrowPrecisionToFloat(m_transform[5]))));
342 return mappedRegion;
343 }
344
345 Region mappedRegion;
346 for (auto& rect : region.rects())
347 mappedRegion.unite(mapRect(rect));
348
349 return mappedRegion;
350}

352void AffineTransform::blend(const AffineTransform& from, double progress)
353{
354 DecomposedType srA, srB;
355
356 from.decompose(srA);
357 this->decompose(srB);
358
359 // If x-axis of one is flipped, and y-axis of the other, convert to an unflipped rotation.
360 if ((srA.scaleX < 0 && srB.scaleY < 0) || (srA.scaleY < 0 && srB.scaleX < 0)) {
361 srA.scaleX = -srA.scaleX;
362 srA.scaleY = -srA.scaleY;
363 srA.angle += srA.angle < 0 ? piDouble : -piDouble;
364 }
365
366 // Don't rotate the long way around.
367 srA.angle = fmod(srA.angle, 2 * piDouble);
368 srB.angle = fmod(srB.angle, 2 * piDouble);
369
370 if (fabs(srA.angle - srB.angle) > piDouble) {
371 if (srA.angle > srB.angle)
372 srA.angle -= piDouble * 2;
373 else
374 srB.angle -= piDouble * 2;
375 }
376
377 srA.scaleX += progress * (srB.scaleX - srA.scaleX);
378 srA.scaleY += progress * (srB.scaleY - srA.scaleY);
379 srA.angle += progress * (srB.angle - srA.angle);
380 srA.remainderA += progress * (srB.remainderA - srA.remainderA);
381 srA.remainderB += progress * (srB.remainderB - srA.remainderB);
382 srA.remainderC += progress * (srB.remainderC - srA.remainderC);
383 srA.remainderD += progress * (srB.remainderD - srA.remainderD);
384 srA.translateX += progress * (srB.translateX - srA.translateX);
385 srA.translateY += progress * (srB.translateY - srA.translateY);
386
387 this->recompose(srA);
388}

396bool AffineTransform::decompose(DecomposedType& decomp) const
397{
398 AffineTransform m(*this);
399
400 // Compute scaling factors
401 double sx = xScale();
402 double sy = yScale();
403
404 // Compute cross product of transformed unit vectors. If negative,
405 // one axis was flipped.
406 if (m.a() * m.d() - m.c() * m.b() < 0) {
407 // Flip axis with minimum unit vector dot product
408 if (m.a() < m.d())
409 sx = -sx;
410 else
411 sy = -sy;
412 }
413
414 // Remove scale from matrix
415 m.scale(1 / sx, 1 / sy);
416
417 // Compute rotation
418 double angle = atan2(m.b(), m.a());
419
420 // Remove rotation from matrix
421 m.rotate(rad2deg(-angle));
422
423 // Return results
424 decomp.scaleX = sx;
425 decomp.scaleY = sy;
426 decomp.angle = angle;
427 decomp.remainderA = m.a();
428 decomp.remainderB = m.b();
429 decomp.remainderC = m.c();
430 decomp.remainderD = m.d();
431 decomp.translateX = m.e();
432 decomp.translateY = m.f();
433
434 return true;
435}
436
437void AffineTransform::recompose(const DecomposedType& decomp)
438{
439 this->setA(decomp.remainderA);
440 this->setB(decomp.remainderB);
441 this->setC(decomp.remainderC);
442 this->setD(decomp.remainderD);
443 this->setE(decomp.translateX);
444 this->setF(decomp.translateY);
445 this->rotate(rad2deg(decomp.angle));
446 this->scale(decomp.scaleX, decomp.scaleY);
447}

449TextStream& operator<<(TextStream& ts, const AffineTransform& transform)
450{
451 if (transform.isIdentity())
452 ts << "identity";
453 else
454 ts << "{m=(("
455 << transform.a() << "," << transform.b()
456 << ")("
457 << transform.c() << "," << transform.d()
458 << ")) t=("
459 << transform.e() << "," << transform.f()
460 << ")}";
461
462 return ts;
463}
