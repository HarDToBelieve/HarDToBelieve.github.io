25#pragma once
26
27#include "TransformOperation.h"
28#include <wtf/Ref.h>
29
30namespace WebCore {
31
32struct BlendingContext;
33
34class IdentityTransformOperation final : public TransformOperation {
35public:
36 static Ref<IdentityTransformOperation> create()
37 {
38 return adoptRef(*new IdentityTransformOperation());
39 }
40
41 Ref<TransformOperation> clone() const override
42 {
43 return create();
44 }
45
46private:
47 bool isIdentity() const override { return true; }
48
49 bool operator==(const TransformOperation& o) const override
50 {
51 return isSameType(o);
52 }
53
54 bool apply(TransformationMatrix&, const FloatSize&) const override
55 {
56 return false;
57 }
58
59 Ref<TransformOperation> blend(const TransformOperation*, const BlendingContext&, bool = false) override
60 {
61 return *this;
62 }
63
64 void dump(WTF::TextStream&) const final;
65
66 IdentityTransformOperation()
67 : TransformOperation(IDENTITY)
68 {
69 }
70};
71
72} // namespace WebCore
73
74SPECIALIZE_TYPE_TRAITS_TRANSFORMOPERATION(WebCore::IdentityTransformOperation, type() == WebCore::TransformOperation::IDENTITY)
