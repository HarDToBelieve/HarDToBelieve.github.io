35bool Matrix3DTransformOperation::operator==(const TransformOperation& other) const
36{
37 return isSameType(other) && m_matrix == downcast<Matrix3DTransformOperation>(other).m_matrix;
38}
39
40static Ref<TransformOperation> createOperation(TransformationMatrix& to, TransformationMatrix& from, const BlendingContext& context)
41{
42 to.blend(from, context.progress);
43 return Matrix3DTransformOperation::create(to);
44}
45
46Ref<TransformOperation> Matrix3DTransformOperation::blend(const TransformOperation* from, const BlendingContext& context, bool blendToIdentity)
47{
48 if (from && !from->isSameType(*this))
49 return *this;
50
51 // Convert the TransformOperations into matrices
52 FloatSize size;
53 TransformationMatrix fromT;
54 TransformationMatrix toT;
55 if (from)
56 from->apply(fromT, size);
57
58 apply(toT, size);
59
60 if (blendToIdentity)
61 return createOperation(fromT, toT, context);
62 return createOperation(toT, fromT, context);
63}
64
65bool Matrix3DTransformOperation::isRepresentableIn2D() const
66{
67 return m_matrix.isAffine();
68}

70void Matrix3DTransformOperation::dump(TextStream& ts) const
71{
72 ts << type() << "(" << m_matrix << ")";
73}
