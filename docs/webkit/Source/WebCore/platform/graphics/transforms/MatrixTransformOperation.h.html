25#pragma once
26
27#include "TransformOperation.h"
28#include "TransformationMatrix.h"
29#include <wtf/Ref.h>
30
31namespace WebCore {
32
33struct BlendingContext;
34
35class MatrixTransformOperation final : public TransformOperation {
36public:
37 static Ref<MatrixTransformOperation> create(double a, double b, double c, double d, double e, double f)
38 {
39 return adoptRef(*new MatrixTransformOperation(a, b, c, d, e, f));
40 }
41
42 static Ref<MatrixTransformOperation> create(const TransformationMatrix& t)
43 {
44 return adoptRef(*new MatrixTransformOperation(t));
45 }
46
47 Ref<TransformOperation> clone() const override
48 {
49 return adoptRef(*new MatrixTransformOperation(matrix()));
50 }
51
52 TransformationMatrix matrix() const { return TransformationMatrix(m_a, m_b, m_c, m_d, m_e, m_f); }
53
54private:
55 bool isIdentity() const override { return m_a == 1 && m_b == 0 && m_c == 0 && m_d == 1 && m_e == 0 && m_f == 0; }
56 bool isAffectedByTransformOrigin() const override { return !isIdentity(); }
57
58 bool operator==(const TransformOperation&) const override;
59
60 bool apply(TransformationMatrix& transform, const FloatSize&) const override
61 {
62 TransformationMatrix matrix(m_a, m_b, m_c, m_d, m_e, m_f);
63 transform.multiply(matrix);
64 return false;
65 }
66
67 Ref<TransformOperation> blend(const TransformOperation* from, const BlendingContext&, bool blendToIdentity = false) override;
68
69 void dump(WTF::TextStream&) const final;
70
71 MatrixTransformOperation(double a, double b, double c, double d, double e, double f)
72 : TransformOperation(MATRIX)
73 , m_a(a)
74 , m_b(b)
75 , m_c(c)
76 , m_d(d)
77 , m_e(e)
78 , m_f(f)
79 {
80 }
81
82 MatrixTransformOperation(const TransformationMatrix& t)
83 : TransformOperation(MATRIX)
84 , m_a(t.a())
85 , m_b(t.b())
86 , m_c(t.c())
87 , m_d(t.d())
88 , m_e(t.e())
89 , m_f(t.f())
90 {
91 }
92
93 double m_a;
94 double m_b;
95 double m_c;
96 double m_d;
97 double m_e;
98 double m_f;
99};
100
101} // namespace WebCore
102
103SPECIALIZE_TYPE_TRAITS_TRANSFORMOPERATION(WebCore::MatrixTransformOperation, type() == WebCore::TransformOperation::MATRIX)
