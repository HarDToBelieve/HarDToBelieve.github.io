26#pragma once
27
28#include "Length.h"
29#include "LengthFunctions.h"
30#include "TransformOperation.h"
31#include <wtf/Ref.h>
32
33namespace WebCore {
34
35struct BlendingContext;
36
37class PerspectiveTransformOperation final : public TransformOperation {
38public:
39 static Ref<PerspectiveTransformOperation> create(const Length& p)
40 {
41 return adoptRef(*new PerspectiveTransformOperation(p));
42 }
43
44 Ref<TransformOperation> clone() const override
45 {
46 return adoptRef(*new PerspectiveTransformOperation(m_p));
47 }
48
49 Length perspective() const { return m_p; }
50
51private:
52 bool isIdentity() const override { return !floatValueForLength(m_p, 1); }
53 bool isAffectedByTransformOrigin() const override { return !isIdentity(); }
54 bool isRepresentableIn2D() const final { return false; }
55
56 bool operator==(const TransformOperation&) const override;
57
58 bool apply(TransformationMatrix& transform, const FloatSize&) const override
59 {
60 transform.applyPerspective(floatValueForLength(m_p, 1));
61 return false;
62 }
63
64 Ref<TransformOperation> blend(const TransformOperation* from, const BlendingContext&, bool blendToIdentity = false) override;
65
66 void dump(WTF::TextStream&) const final;
67
68 PerspectiveTransformOperation(const Length& p)
69 : TransformOperation(PERSPECTIVE)
70 , m_p(p)
71 {
72 ASSERT(p.isFixed());
73 }
74
75 Length m_p;
76};
77
78} // namespace WebCore
79
80SPECIALIZE_TYPE_TRAITS_TRANSFORMOPERATION(WebCore::PerspectiveTransformOperation, type() == WebCore::TransformOperation::PERSPECTIVE)
