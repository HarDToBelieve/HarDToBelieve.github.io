25#pragma once
26
27#include "TransformOperation.h"
28#include <wtf/Ref.h>
29
30namespace WebCore {
31
32struct BlendingContext;
33
34class RotateTransformOperation final : public TransformOperation {
35public:
36 static Ref<RotateTransformOperation> create(double angle, OperationType type)
37 {
38 return adoptRef(*new RotateTransformOperation(0, 0, 1, angle, type));
39 }
40
41 static Ref<RotateTransformOperation> create(double x, double y, double z, double angle, OperationType type)
42 {
43 return adoptRef(*new RotateTransformOperation(x, y, z, angle, type));
44 }
45
46 Ref<TransformOperation> clone() const override
47 {
48 return adoptRef(*new RotateTransformOperation(m_x, m_y, m_z, m_angle, type()));
49 }
50
51 double x() const { return m_x; }
52 double y() const { return m_y; }
53 double z() const { return m_z; }
54 double angle() const { return m_angle; }
55
56 bool operator==(const TransformOperation&) const override;
57
58 Ref<TransformOperation> blend(const TransformOperation* from, const BlendingContext&, bool blendToIdentity = false) final;
59
60 bool isIdentity() const final { return !m_angle; }
61
62 bool isRepresentableIn2D() const final { return (!m_x && !m_y) || !m_angle; }
63
64private:
65 bool isAffectedByTransformOrigin() const override { return !isIdentity(); }
66
67 bool apply(TransformationMatrix& transform, const FloatSize& /*borderBoxSize*/) const override
68 {
69 if (type() == TransformOperation::ROTATE)
70 transform.rotate(m_angle);
71 else
72 transform.rotate3d(m_x, m_y, m_z, m_angle);
73 return false;
74 }
75
76 void dump(WTF::TextStream&) const final;
77
78 RotateTransformOperation(double x, double y, double z, double angle, OperationType type)
79 : TransformOperation(type)
80 , m_x(x)
81 , m_y(y)
82 , m_z(z)
83 , m_angle(angle)
84 {
85 ASSERT(isRotateTransformOperationType());
86 }
87
88 double m_x;
89 double m_y;
90 double m_z;
91 double m_angle;
92};
93
94} // namespace WebCore
95
96SPECIALIZE_TYPE_TRAITS_TRANSFORMOPERATION(WebCore::RotateTransformOperation, isRotateTransformOperationType())
