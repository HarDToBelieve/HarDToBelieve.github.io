30bool ScaleTransformOperation::operator==(const TransformOperation& other) const
31{
32 if (!isSameType(other))
33 return false;
34 const ScaleTransformOperation& s = downcast<ScaleTransformOperation>(other);
35 return m_x == s.m_x && m_y == s.m_y && m_z == s.m_z;
36}
37
38Ref<TransformOperation> ScaleTransformOperation::blend(const TransformOperation* from, const BlendingContext& context, bool blendToIdentity)
39{
40 if (from && !from->isSameType(*this))
41 return *this;
42
43 if (blendToIdentity)
44 return ScaleTransformOperation::create(WebCore::blend(m_x, 1.0, context),
45 WebCore::blend(m_y, 1.0, context),
46 WebCore::blend(m_z, 1.0, context), type());
47
48 const ScaleTransformOperation* fromOp = downcast<ScaleTransformOperation>(from);
49 double fromX = fromOp ? fromOp->m_x : 1.0;
50 double fromY = fromOp ? fromOp->m_y : 1.0;
51 double fromZ = fromOp ? fromOp->m_z : 1.0;
52 return ScaleTransformOperation::create(WebCore::blend(fromX, m_x, context),
53 WebCore::blend(fromY, m_y, context),
54 WebCore::blend(fromZ, m_z, context), type());
55}
56
57void ScaleTransformOperation::dump(TextStream& ts) const
58{
59 ts << type() << "(" << TextStream::FormatNumberRespectingIntegers(m_x) << ", " << TextStream::FormatNumberRespectingIntegers(m_y) << ", " << TextStream::FormatNumberRespectingIntegers(m_z) << ")";
60}
