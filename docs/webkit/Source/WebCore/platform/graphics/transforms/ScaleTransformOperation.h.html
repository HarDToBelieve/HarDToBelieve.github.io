25#pragma once
26
27#include "TransformOperation.h"
28#include <wtf/Ref.h>
29
30namespace WebCore {
31
32struct BlendingContext;
33
34class ScaleTransformOperation final : public TransformOperation {
35public:
36 static Ref<ScaleTransformOperation> create(double sx, double sy, OperationType type)
37 {
38 return adoptRef(*new ScaleTransformOperation(sx, sy, 1, type));
39 }
40
41 static Ref<ScaleTransformOperation> create(double sx, double sy, double sz, OperationType type)
42 {
43 return adoptRef(*new ScaleTransformOperation(sx, sy, sz, type));
44 }
45
46 Ref<TransformOperation> clone() const override
47 {
48 return adoptRef(*new ScaleTransformOperation(m_x, m_y, m_z, type()));
49 }
50
51 double x() const { return m_x; }
52 double y() const { return m_y; }
53 double z() const { return m_z; }
54
55 bool operator==(const TransformOperation&) const final;
56
57 Ref<TransformOperation> blend(const TransformOperation* from, const BlendingContext&, bool blendToIdentity = false) final;
58
59 bool isIdentity() const final { return m_x == 1 && m_y == 1 && m_z == 1; }
60
61 bool isRepresentableIn2D() const final { return m_z == 1; }
62
63private:
64 bool isAffectedByTransformOrigin() const override { return !isIdentity(); }
65
66 bool apply(TransformationMatrix& transform, const FloatSize&) const override
67 {
68 transform.scale3d(m_x, m_y, m_z);
69 return false;
70 }
71
72 void dump(WTF::TextStream&) const final;
73
74 ScaleTransformOperation(double sx, double sy, double sz, OperationType type)
75 : TransformOperation(type)
76 , m_x(sx)
77 , m_y(sy)
78 , m_z(sz)
79 {
80 ASSERT(isScaleTransformOperationType());
81 }
82
83 double m_x;
84 double m_y;
85 double m_z;
86};
87
88} // namespace WebCore
89
90SPECIALIZE_TYPE_TRAITS_TRANSFORMOPERATION(WebCore::ScaleTransformOperation, isScaleTransformOperationType())
