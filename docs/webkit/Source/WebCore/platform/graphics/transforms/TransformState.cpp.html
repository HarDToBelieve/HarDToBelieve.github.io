33TransformState& TransformState::operator=(const TransformState& other)
34{
35 m_accumulatedOffset = other.m_accumulatedOffset;
36 m_mapPoint = other.m_mapPoint;
37 m_mapQuad = other.m_mapQuad;
38 if (m_mapPoint)
39 m_lastPlanarPoint = other.m_lastPlanarPoint;
40 if (m_mapQuad) {
41 m_lastPlanarQuad = other.m_lastPlanarQuad;
42 m_lastPlanarSecondaryQuad = other.m_lastPlanarSecondaryQuad;
43 }
44 m_accumulatingTransform = other.m_accumulatingTransform;
45 m_direction = other.m_direction;
46
47 m_accumulatedTransform = nullptr;
48
49 if (other.m_accumulatedTransform)
50 m_accumulatedTransform = makeUnique<TransformationMatrix>(*other.m_accumulatedTransform);
51
52 return *this;
53}

55void TransformState::translateTransform(const LayoutSize& offset)
56{
57 if (m_direction == ApplyTransformDirection)
58 m_accumulatedTransform->translateRight(offset.width(), offset.height());
59 else
60 m_accumulatedTransform->translate(offset.width(), offset.height());
61}
62
63void TransformState::translateMappedCoordinates(const LayoutSize& offset)
64{
65 LayoutSize adjustedOffset = (m_direction == ApplyTransformDirection) ? offset : -offset;
66 if (m_mapPoint)
67 m_lastPlanarPoint.move(adjustedOffset);
68 if (m_mapQuad) {
69 m_lastPlanarQuad.move(adjustedOffset);
70 if (m_lastPlanarSecondaryQuad)
71 m_lastPlanarSecondaryQuad->move(adjustedOffset);
72 }
73}
74
75void TransformState::move(const LayoutSize& offset, TransformAccumulation accumulate)
76{
77 if (accumulate == FlattenTransform && !m_accumulatedTransform)
78 m_accumulatedOffset += offset;
79 else {
80 applyAccumulatedOffset();
81 if (m_accumulatingTransform && m_accumulatedTransform) {
82 // If we're accumulating into an existing transform, apply the translation.
83 translateTransform(offset);
84
85 // Then flatten if necessary.
86 if (accumulate == FlattenTransform)
87 flatten();
88 } else
89 // Just move the point and/or quad.
90 translateMappedCoordinates(offset);
91 }
92 m_accumulatingTransform = accumulate == AccumulateTransform;
93}
94
95void TransformState::applyAccumulatedOffset()
96{
97 LayoutSize offset = m_accumulatedOffset;
98 m_accumulatedOffset = LayoutSize();
99 if (!offset.isZero()) {
100 if (m_accumulatedTransform) {
101 translateTransform(offset);
102 flatten();
103 } else
104 translateMappedCoordinates(offset);
105 }
106}

108// FIXME: We transform AffineTransform to TransformationMatrix. This is rather inefficient.
109void TransformState::applyTransform(const AffineTransform& transformFromContainer, TransformAccumulation accumulate, bool* wasClamped)
110{
111 applyTransform(transformFromContainer.toTransformationMatrix(), accumulate, wasClamped);
112}
113
114void TransformState::applyTransform(const TransformationMatrix& transformFromContainer, TransformAccumulation accumulate, bool* wasClamped)
115{
116 if (wasClamped)
117 *wasClamped = false;
118
119 if (transformFromContainer.isIntegerTranslation()) {
120 move(LayoutSize(transformFromContainer.e(), transformFromContainer.f()), accumulate);
121 return;
122 }
123
124 applyAccumulatedOffset();
125
126 // If we have an accumulated transform from last time, multiply in this transform
127 if (m_accumulatedTransform) {
128 if (m_direction == ApplyTransformDirection)
129 m_accumulatedTransform = makeUnique<TransformationMatrix>(transformFromContainer * *m_accumulatedTransform);
130 else
131 m_accumulatedTransform->multiply(transformFromContainer);
132 } else if (accumulate == AccumulateTransform) {
133 // Make one if we started to accumulate
134 m_accumulatedTransform = makeUnique<TransformationMatrix>(transformFromContainer);
135 }
136
137 if (accumulate == FlattenTransform) {
138 const TransformationMatrix* finalTransform = m_accumulatedTransform ? m_accumulatedTransform.get() : &transformFromContainer;
139 flattenWithTransform(*finalTransform, wasClamped);
140 }
141 m_accumulatingTransform = accumulate == AccumulateTransform;
142}
143
144void TransformState::flatten(bool* wasClamped)
145{
146 if (wasClamped)
147 *wasClamped = false;
148
149 applyAccumulatedOffset();
150
151 if (!m_accumulatedTransform) {
152 m_accumulatingTransform = false;
153 return;
154 }
155
156 flattenWithTransform(*m_accumulatedTransform, wasClamped);
157}

159FloatPoint TransformState::mappedPoint(bool* wasClamped) const
160{
161 if (wasClamped)
162 *wasClamped = false;
163
164 FloatPoint point = m_lastPlanarPoint;
165 point.move((m_direction == ApplyTransformDirection) ? m_accumulatedOffset : -m_accumulatedOffset);
166 if (!m_accumulatedTransform)
167 return point;
168
169 if (m_direction == ApplyTransformDirection)
170 return m_accumulatedTransform->mapPoint(point);
171
172 return m_accumulatedTransform->inverse().value_or(TransformationMatrix()).projectPoint(point, wasClamped);
173}

175FloatQuad TransformState::mappedQuad(bool* wasClamped) const
176{
177 if (wasClamped)
178 *wasClamped = false;
179
180 FloatQuad quad = m_lastPlanarQuad;
181 mapQuad(quad, m_direction, wasClamped);
182 return quad;
183}

185std::optional<FloatQuad> TransformState::mappedSecondaryQuad(bool* wasClamped) const
186{
187 if (wasClamped)
188 *wasClamped = false;
189
190 if (!m_lastPlanarSecondaryQuad)
191 return std::nullopt;
192
193 FloatQuad quad = *m_lastPlanarSecondaryQuad;
194 mapQuad(quad, m_direction, wasClamped);
195 return quad;
196}
197
198void TransformState::setLastPlanarSecondaryQuad(const std::optional<FloatQuad>& quad)
199{
200 if (!quad) {
201 m_lastPlanarSecondaryQuad = std::nullopt;
202 return;
203 }
204
205 // Map the quad back through any transform or offset back into the last flattening coordinate space.
206 FloatQuad backMappedQuad(*quad);
207 mapQuad(backMappedQuad, inverseDirection());
208 m_lastPlanarSecondaryQuad = backMappedQuad;
209}
210
211void TransformState::mapQuad(FloatQuad& quad, TransformDirection direction, bool* wasClamped) const
212{
213 quad.move((direction == ApplyTransformDirection) ? m_accumulatedOffset : -m_accumulatedOffset);
214 if (!m_accumulatedTransform)
215 return;
216
217 if (direction == ApplyTransformDirection) {
218 quad = m_accumulatedTransform->mapQuad(quad);
219 return;
220 }
221
222 quad = m_accumulatedTransform->inverse().value_or(TransformationMatrix()).projectQuad(quad, wasClamped);
223}
224
225void TransformState::flattenWithTransform(const TransformationMatrix& t, bool* wasClamped)
226{
227 if (m_direction == ApplyTransformDirection) {
228 if (m_mapPoint)
229 m_lastPlanarPoint = t.mapPoint(m_lastPlanarPoint);
230 if (m_mapQuad) {
231 m_lastPlanarQuad = t.mapQuad(m_lastPlanarQuad);
232 if (m_lastPlanarSecondaryQuad)
233 m_lastPlanarSecondaryQuad = t.mapQuad(*m_lastPlanarSecondaryQuad);
234 }
235 } else {
236 TransformationMatrix inverseTransform = t.inverse().value_or(TransformationMatrix());
237 if (m_mapPoint)
238 m_lastPlanarPoint = inverseTransform.projectPoint(m_lastPlanarPoint);
239 if (m_mapQuad) {
240 m_lastPlanarQuad = inverseTransform.projectQuad(m_lastPlanarQuad, wasClamped);
241 if (m_lastPlanarSecondaryQuad)
242 m_lastPlanarSecondaryQuad = inverseTransform.projectQuad(*m_lastPlanarSecondaryQuad, wasClamped);
243 }
244 }
245
246 // We could throw away m_accumulatedTransform if we wanted to here, but that
247 // would cause thrash when traversing hierarchies with alternating
248 // preserve-3d and flat elements.
249 if (m_accumulatedTransform)
250 m_accumulatedTransform->makeIdentity();
251 m_accumulatingTransform = false;
252}

254TextStream& operator<<(TextStream& ts, const TransformState& state)
255{
256 TextStream multilineStream;
257 multilineStream.setIndent(ts.indent() + 2);
258
259 multilineStream.dumpProperty("last planar point", state.lastPlanarPoint());
260 multilineStream.dumpProperty("last planar quad", state.lastPlanarQuad());
261
262 if (state.lastPlanarSecondaryQuad())
263 multilineStream.dumpProperty("last planar secondary quad", *state.lastPlanarSecondaryQuad());
264
265 if (state.accumulatedTransform())
266 multilineStream.dumpProperty("accumulated transform", ValueOrNull(state.accumulatedTransform()));
267
268 {
269 TextStream::GroupScope scope(ts);
270 ts << "TransformState " << multilineStream.release();
271 }
272 return ts;
273}
