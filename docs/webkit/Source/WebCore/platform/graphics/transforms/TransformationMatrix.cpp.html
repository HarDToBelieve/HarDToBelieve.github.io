56// A clarification about the storage of matrix elements
57//
58// This class uses a 2 dimensional array internally to store the elements of the matrix. The first index into
59// the array refers to the column that the element lies in; the second index refers to the row.
60//
61// In other words, this is the layout of the matrix:
62//
63// | m_matrix[0][0] m_matrix[1][0] m_matrix[2][0] m_matrix[3][0] |
64// | m_matrix[0][1] m_matrix[1][1] m_matrix[2][1] m_matrix[3][1] |
65// | m_matrix[0][2] m_matrix[1][2] m_matrix[2][2] m_matrix[3][2] |

83// double = determinant2x2(double a, double b, double c, double d)
84//
85// calculate the determinant of a 2x2 matrix.
86
87static double determinant2x2(double a, double b, double c, double d)
88{
89 return a * d - b * c;
90}

92// double = determinant3x3(a1, a2, a3, b1, b2, b3, c1, c2, c3)
93//
94// Calculate the determinant of a 3x3 matrix
95// in the form
96//
97// | a1, b1, c1 |
98// | a2, b2, c2 |
99// | a3, b3, c3 |
100
101static double determinant3x3(double a1, double a2, double a3, double b1, double b2, double b3, double c1, double c2, double c3)
102{
103 return a1 * determinant2x2(b2, b3, c2, c3)
104 - b1 * determinant2x2(a2, a3, c2, c3)
105 + c1 * determinant2x2(a2, a3, b2, b3);
106}

108// double = determinant4x4(matrix)
109//
110// calculate the determinant of a 4x4 matrix.
111
112static double determinant4x4(const TransformationMatrix::Matrix4& m)
113{
114 // Assign to individual variable names to aid selecting
115 // correct elements
116
117 double a1 = m[0][0];
118 double b1 = m[0][1];
119 double c1 = m[0][2];
120 double d1 = m[0][3];
121
122 double a2 = m[1][0];
123 double b2 = m[1][1];
124 double c2 = m[1][2];
125 double d2 = m[1][3];
126
127 double a3 = m[2][0];
128 double b3 = m[2][1];
129 double c3 = m[2][2];
130 double d3 = m[2][3];
131
132 double a4 = m[3][0];
133 double b4 = m[3][1];
134 double c4 = m[3][2];
135 double d4 = m[3][3];
136
137 return a1 * determinant3x3(b2, b3, b4, c2, c3, c4, d2, d3, d4)
138 - b1 * determinant3x3(a2, a3, a4, c2, c3, c4, d2, d3, d4)
139 + c1 * determinant3x3(a2, a3, a4, b2, b3, b4, d2, d3, d4)
140 - d1 * determinant3x3(a2, a3, a4, b2, b3, b4, c2, c3, c4);
141}

143// adjoint( original_matrix, inverse_matrix )
144//
145// calculate the adjoint of a 4x4 matrix
146//
147// Let a denote the minor determinant of matrix A obtained by
148// ij
149//
150// deleting the ith row and jth column from A.
151//
152// i+j
153// Let b = (-1) a
154// ij ji
155//
156// The matrix B = (b ) is the adjoint of A
157// ij
158
159static void adjoint(const TransformationMatrix::Matrix4& matrix, TransformationMatrix::Matrix4& result)
160{
161 // Assign to individual variable names to aid
162 // selecting correct values
163 double a1 = matrix[0][0];
164 double b1 = matrix[0][1];
165 double c1 = matrix[0][2];
166 double d1 = matrix[0][3];
167
168 double a2 = matrix[1][0];
169 double b2 = matrix[1][1];
170 double c2 = matrix[1][2];
171 double d2 = matrix[1][3];
172
173 double a3 = matrix[2][0];
174 double b3 = matrix[2][1];
175 double c3 = matrix[2][2];
176 double d3 = matrix[2][3];
177
178 double a4 = matrix[3][0];
179 double b4 = matrix[3][1];
180 double c4 = matrix[3][2];
181 double d4 = matrix[3][3];
182
183 // Row column labeling reversed since we transpose rows & columns
184 result[0][0] = determinant3x3(b2, b3, b4, c2, c3, c4, d2, d3, d4);
185 result[1][0] = - determinant3x3(a2, a3, a4, c2, c3, c4, d2, d3, d4);
186 result[2][0] = determinant3x3(a2, a3, a4, b2, b3, b4, d2, d3, d4);
187 result[3][0] = - determinant3x3(a2, a3, a4, b2, b3, b4, c2, c3, c4);
188
189 result[0][1] = - determinant3x3(b1, b3, b4, c1, c3, c4, d1, d3, d4);
190 result[1][1] = determinant3x3(a1, a3, a4, c1, c3, c4, d1, d3, d4);
191 result[2][1] = - determinant3x3(a1, a3, a4, b1, b3, b4, d1, d3, d4);
192 result[3][1] = determinant3x3(a1, a3, a4, b1, b3, b4, c1, c3, c4);
193
194 result[0][2] = determinant3x3(b1, b2, b4, c1, c2, c4, d1, d2, d4);
195 result[1][2] = - determinant3x3(a1, a2, a4, c1, c2, c4, d1, d2, d4);
196 result[2][2] = determinant3x3(a1, a2, a4, b1, b2, b4, d1, d2, d4);
197 result[3][2] = - determinant3x3(a1, a2, a4, b1, b2, b4, c1, c2, c4);
198
199 result[0][3] = - determinant3x3(b1, b2, b3, c1, c2, c3, d1, d2, d3);
200 result[1][3] = determinant3x3(a1, a2, a3, c1, c2, c3, d1, d2, d3);
201 result[2][3] = - determinant3x3(a1, a2, a3, b1, b2, b3, d1, d2, d3);
202 result[3][3] = determinant3x3(a1, a2, a3, b1, b2, b3, c1, c2, c3);
203}

205// Returns false if the matrix is not invertible
206static bool inverse(const TransformationMatrix::Matrix4& matrix, TransformationMatrix::Matrix4& result)
207{
208 // Calculate the adjoint matrix
209 adjoint(matrix, result);
210
211 // Calculate the 4x4 determinant
212 // If the determinant is zero,
213 // then the inverse matrix is not unique.
214 double det = determinant4x4(matrix);
215
216 if (fabs(det) < SMALL_NUMBER)
217 return false;
218
219 // Scale the adjoint matrix to get the inverse
220
221 for (int i = 0; i < 4; i++)
222 for (int j = 0; j < 4; j++)
223 result[i][j] = result[i][j] / det;
224
225 return true;
226}

234static void transposeMatrix4(const TransformationMatrix::Matrix4& a, TransformationMatrix::Matrix4& b)
235{
236 for (int i = 0; i < 4; i++)
237 for (int j = 0; j < 4; j++)
238 b[i][j] = a[j][i];

241// Multiply a homogeneous point by a matrix and return the transformed point
242static void v4MulPointByMatrix(const Vector4 p, const TransformationMatrix::Matrix4& m, Vector4 result)
243{
244 result[0] = (p[0] * m[0][0]) + (p[1] * m[1][0]) +
245 (p[2] * m[2][0]) + (p[3] * m[3][0]);
246 result[1] = (p[0] * m[0][1]) + (p[1] * m[1][1]) +
247 (p[2] * m[2][1]) + (p[3] * m[3][1]);
248 result[2] = (p[0] * m[0][2]) + (p[1] * m[1][2]) +
249 (p[2] * m[2][2]) + (p[3] * m[3][2]);
250 result[3] = (p[0] * m[0][3]) + (p[1] * m[1][3]) +
251 (p[2] * m[2][3]) + (p[3] * m[3][3]);
252}

254static double v3Length(Vector3 a)
255{
256 return std::hypot(a[0], a[1], a[2]);
257}

259static void v3Scale(Vector3 v, double desiredLength)
260{
261 double len = v3Length(v);
262 if (len != 0) {
263 double l = desiredLength / len;
264 v[0] *= l;
265 v[1] *= l;
266 v[2] *= l;
267 }
268}

270static double v3Dot(const Vector3 a, const Vector3 b)
271{
272 return (a[0] * b[0]) + (a[1] * b[1]) + (a[2] * b[2]);
273}

275// Make a linear combination of two vectors and return the result.
276// result = (a * ascl) + (b * bscl)
277static void v3Combine(const Vector3 a, const Vector3 b, Vector3 result, double ascl, double bscl)
278{
279 result[0] = (ascl * a[0]) + (bscl * b[0]);
280 result[1] = (ascl * a[1]) + (bscl * b[1]);
281 result[2] = (ascl * a[2]) + (bscl * b[2]);
282}

284// Return the cross product result = a cross b */
285static void v3Cross(const Vector3 a, const Vector3 b, Vector3 result)
286{
287 result[0] = (a[1] * b[2]) - (a[2] * b[1]);
288 result[1] = (a[2] * b[0]) - (a[0] * b[2]);
289 result[2] = (a[0] * b[1]) - (a[1] * b[0]);
290}

292static bool decompose2(const TransformationMatrix::Matrix4& matrix, TransformationMatrix::Decomposed2Type& result)
293{
294 double row0x = matrix[0][0];
295 double row0y = matrix[0][1];
296 double row1x = matrix[1][0];
297 double row1y = matrix[1][1];
298 result.translateX = matrix[3][0];
299 result.translateY = matrix[3][1];
300
301 // Compute scaling factors.
302 result.scaleX = std::hypot(row0x, row0y);
303 result.scaleY = std::hypot(row1x, row1y);
304
305 // If determinant is negative, one axis was flipped.
306 double determinant = row0x * row1y - row0y * row1x;
307 if (determinant < 0) {
308 // Flip axis with minimum unit vector dot product.
309 if (row0x < row1y)
310 result.scaleX = -result.scaleX;
311 else
312 result.scaleY = -result.scaleY;
313 }
314
315 // Renormalize matrix to remove scale.
316 if (result.scaleX) {
317 row0x *= 1 / result.scaleX;
318 row0y *= 1 / result.scaleX;
319 }
320 if (result.scaleY) {
321 row1x *= 1 / result.scaleY;
322 row1y *= 1 / result.scaleY;
323 }
324
325 // Compute rotation and renormalize matrix.
326 result.angle = atan2(row0y, row0x);
327
328 if (result.angle) {
329 // Rotate(-angle) = [cos(angle), sin(angle), -sin(angle), cos(angle)]
330 // = [row0x, -row0y, row0y, row0x]
331 // Thanks to the normalization above.
332 double sn = -row0y;
333 double cs = row0x;
334 double m11 = row0x, m12 = row0y;
335 double m21 = row1x, m22 = row1y;
336
337 row0x = cs * m11 + sn * m21;
338 row0y = cs * m12 + sn * m22;
339 row1x = -sn * m11 + cs * m21;
340 row1y = -sn * m12 + cs * m22;
341 }
342
343 result.m11 = row0x;
344 result.m12 = row0y;
345 result.m21 = row1x;
346 result.m22 = row1y;
347
348 // Convert into degrees because our rotation functions expect it.
349 result.angle = rad2deg(result.angle);
350
351 return true;
352}

354static bool decompose4(const TransformationMatrix::Matrix4& mat, TransformationMatrix::Decomposed4Type& result)
355{
356 TransformationMatrix::Matrix4 localMatrix;
357 memcpy(localMatrix, mat, sizeof(TransformationMatrix::Matrix4));
358
359 // Normalize the matrix.
360 if (localMatrix[3][3] == 0)
361 return false;
362
363 int i, j;
364 for (i = 0; i < 4; i++)
365 for (j = 0; j < 4; j++)
366 localMatrix[i][j] /= localMatrix[3][3];
367
368 // perspectiveMatrix is used to solve for perspective, but it also provides
369 // an easy way to test for singularity of the upper 3x3 component.
370 TransformationMatrix::Matrix4 perspectiveMatrix;
371 memcpy(perspectiveMatrix, localMatrix, sizeof(TransformationMatrix::Matrix4));
372 for (i = 0; i < 3; i++)
373 perspectiveMatrix[i][3] = 0;
374 perspectiveMatrix[3][3] = 1;
375
376 if (determinant4x4(perspectiveMatrix) == 0)
377 return false;
378
379 // First, isolate perspective. This is the messiest.
380 if (localMatrix[0][3] != 0 || localMatrix[1][3] != 0 || localMatrix[2][3] != 0) {
381 // rightHandSide is the right hand side of the equation.
382 Vector4 rightHandSide;
383 rightHandSide[0] = localMatrix[0][3];
384 rightHandSide[1] = localMatrix[1][3];
385 rightHandSide[2] = localMatrix[2][3];
386 rightHandSide[3] = localMatrix[3][3];
387
388 // Solve the equation by inverting perspectiveMatrix and multiplying
389 // rightHandSide by the inverse. (This is the easiest way, not
390 // necessarily the best.)
391 TransformationMatrix::Matrix4 inversePerspectiveMatrix, transposedInversePerspectiveMatrix;
392 inverse(perspectiveMatrix, inversePerspectiveMatrix);
393 transposeMatrix4(inversePerspectiveMatrix, transposedInversePerspectiveMatrix);
394
395 Vector4 perspectivePoint;
396 v4MulPointByMatrix(rightHandSide, transposedInversePerspectiveMatrix, perspectivePoint);
397
398 result.perspectiveX = perspectivePoint[0];
399 result.perspectiveY = perspectivePoint[1];
400 result.perspectiveZ = perspectivePoint[2];
401 result.perspectiveW = perspectivePoint[3];
402
403 // Clear the perspective partition
404 localMatrix[0][3] = localMatrix[1][3] = localMatrix[2][3] = 0;
405 localMatrix[3][3] = 1;
406 } else {
407 // No perspective.
408 result.perspectiveX = result.perspectiveY = result.perspectiveZ = 0;
409 result.perspectiveW = 1;
410 }
411
412 // Next take care of translation (easy).
413 result.translateX = localMatrix[3][0];
414 localMatrix[3][0] = 0;
415 result.translateY = localMatrix[3][1];
416 localMatrix[3][1] = 0;
417 result.translateZ = localMatrix[3][2];
418 localMatrix[3][2] = 0;
419
420 // Vector4 type and functions need to be added to the common set.
421 Vector3 row[3], pdum3;
422
423 // Now get scale and shear.
424 for (i = 0; i < 3; i++) {
425 row[i][0] = localMatrix[i][0];
426 row[i][1] = localMatrix[i][1];
427 row[i][2] = localMatrix[i][2];
428 }
429
430 // Compute X scale factor and normalize first row.
431 result.scaleX = v3Length(row[0]);
432 v3Scale(row[0], 1.0);
433
434 // Compute XY shear factor and make 2nd row orthogonal to 1st.
435 result.skewXY = v3Dot(row[0], row[1]);
436 v3Combine(row[1], row[0], row[1], 1.0, -result.skewXY);
437
438 // Now, compute Y scale and normalize 2nd row.
439 result.scaleY = v3Length(row[1]);
440 v3Scale(row[1], 1.0);
441 result.skewXY /= result.scaleY;
442
443 // Compute XZ and YZ shears, orthogonalize 3rd row.
444 result.skewXZ = v3Dot(row[0], row[2]);
445 v3Combine(row[2], row[0], row[2], 1.0, -result.skewXZ);
446 result.skewYZ = v3Dot(row[1], row[2]);
447 v3Combine(row[2], row[1], row[2], 1.0, -result.skewYZ);
448
449 // Next, get Z scale and normalize 3rd row.
450 result.scaleZ = v3Length(row[2]);
451 v3Scale(row[2], 1.0);
452 result.skewXZ /= result.scaleZ;
453 result.skewYZ /= result.scaleZ;
454
455 // At this point, the matrix (in rows[]) is orthonormal.
456 // Check for a coordinate system flip. If the determinant
457 // is -1, then negate the matrix and the scaling factors.
458 v3Cross(row[1], row[2], pdum3);
459 if (v3Dot(row[0], pdum3) < 0) {
460
461 result.scaleX *= -1;
462 result.scaleY *= -1;
463 result.scaleZ *= -1;
464
465 for (i = 0; i < 3; i++) {
466 row[i][0] *= -1;
467 row[i][1] *= -1;
468 row[i][2] *= -1;
469 }
470 }
471
472 // Now, get the rotations out, as described in the gem.
473
474 // FIXME - Add the ability to return either quaternions (which are
475 // easier to recompose with) or Euler angles (rx, ry, rz), which
476 // are easier for authors to deal with. The latter will only be useful
477 // when we fix https://bugs.webkit.org/show_bug.cgi?id=23799, so I
478 // will leave the Euler angle code here for now.
479
480 // ret.rotateY = asin(-row[0][2]);
481 // if (cos(ret.rotateY) != 0) {
482 // ret.rotateX = atan2(row[1][2], row[2][2]);
483 // ret.rotateZ = atan2(row[0][1], row[0][0]);
484 // } else {
485 // ret.rotateX = atan2(-row[2][0], row[1][1]);
486 // ret.rotateZ = 0;
487 // }
488
489 double s, t, x, y, z, w;
490
491 t = row[0][0] + row[1][1] + row[2][2] + 1.0;
492
493 if (t > 1e-4) {
494 s = 0.5 / sqrt(t);
495 w = 0.25 / s;
496 x = (row[2][1] - row[1][2]) * s;
497 y = (row[0][2] - row[2][0]) * s;
498 z = (row[1][0] - row[0][1]) * s;
499 } else if (row[0][0] > row[1][1] && row[0][0] > row[2][2]) {
500 s = sqrt(1.0 + row[0][0] - row[1][1] - row[2][2]) * 2.0; // S = 4 * qx.
501 x = 0.25 * s;
502 y = (row[0][1] + row[1][0]) / s;
503 z = (row[0][2] + row[2][0]) / s;
504 w = (row[2][1] - row[1][2]) / s;
505 } else if (row[1][1] > row[2][2]) {
506 s = sqrt(1.0 + row[1][1] - row[0][0] - row[2][2]) * 2.0; // S = 4 * qy.
507 x = (row[0][1] + row[1][0]) / s;
508 y = 0.25 * s;
509 z = (row[1][2] + row[2][1]) / s;
510 w = (row[0][2] - row[2][0]) / s;
511 } else {
512 s = sqrt(1.0 + row[2][2] - row[0][0] - row[1][1]) * 2.0; // S = 4 * qz.
513 x = (row[0][2] + row[2][0]) / s;
514 y = (row[1][2] + row[2][1]) / s;
515 z = 0.25 * s;
516 w = (row[1][0] - row[0][1]) / s;
517 }
518
519 result.quaternionX = x;
520 result.quaternionY = y;
521 result.quaternionZ = z;
522 result.quaternionW = w;
523
524 return true;
525}

527// Perform a spherical linear interpolation between the two
528// passed quaternions with 0 <= t <= 1.
529static void slerp(double qa[4], const double qb[4], double t)
530{
531 double ax, ay, az, aw;
532 double bx, by, bz, bw;
533 double cx, cy, cz, cw;
534 double angle;
535 double th, invth, scale, invscale;
536
537 ax = qa[0]; ay = qa[1]; az = qa[2]; aw = qa[3];
538 bx = qb[0]; by = qb[1]; bz = qb[2]; bw = qb[3];
539
540 angle = ax * bx + ay * by + az * bz + aw * bw;
541
542 if (angle < 0.0) {
543 ax = -ax; ay = -ay;
544 az = -az; aw = -aw;
545 angle = -angle;
546 }
547
548 if (angle + 1.0 > .05) {
549 if (1.0 - angle >= .05) {
550 th = acos (angle);
551 invth = 1.0 / sin (th);
552 scale = sin (th * (1.0 - t)) * invth;
553 invscale = sin (th * t) * invth;
554 } else {
555 scale = 1.0 - t;
556 invscale = t;
557 }
558 } else {
559 bx = -ay;
560 by = ax;
561 bz = -aw;
562 bw = az;
563 scale = sin(piDouble * (.5 - t));
564 invscale = sin (piDouble * t);
565 }
566
567 cx = ax * scale + bx * invscale;
568 cy = ay * scale + by * invscale;
569 cz = az * scale + bz * invscale;
570 cw = aw * scale + bw * invscale;
571
572 qa[0] = cx; qa[1] = cy; qa[2] = cz; qa[3] = cw;
573}

583// FIXME: Once https://bugs.webkit.org/show_bug.cgi?id=220856 is addressed we can reuse this function in TransformationMatrix::recompose4().
584TransformationMatrix TransformationMatrix::fromQuaternion(double qx, double qy, double qz, double qw)
585{
586 double xx = qx * qx;
587 double yy = qy * qy;
588 double zz = qz * qz;
589 double xz = qx * qz;
590 double xy = qx * qy;
591 double yz = qy * qz;
592 double xw = qw * qx;
593 double yw = qw * qy;
594 double zw = qw * qz;
595
596 return TransformationMatrix(1 - 2 * (yy + zz), 2 * (xy + zw), 2 * (xz - yw), 0,
597 2 * (xy - zw), 1 - 2 * (xx + zz), 2 * (yz + xw), 0,
598 2 * (xz + yw), 2 * (yz - xw), 1 - 2 * (xx + yy), 0,
599 0, 0, 0, 1);
600}

603TransformationMatrix TransformationMatrix::fromProjection(double fovUp, double fovDown, double fovLeft, double fovRight, double depthNear, double depthFar)
604{
605 double upTan = tan(fovUp);
606 double downTan = tan(fovDown);
607 double leftTan = tan(fovLeft);
608 double rightTan = tan(fovRight);
609 double xScale = 2.0 / (leftTan + rightTan);
610 double yScale = 2.0 / (upTan + downTan);
611 double invDepth = 1.0 / (depthNear - depthFar);
612
613 return TransformationMatrix(xScale, 0.0f, 0.0f, 0.0f,
614 0.0f, yScale, 0.0f, 0.0f,
615 (leftTan - rightTan) * xScale * -0.5, (upTan - downTan) * yScale * 0.5, (depthNear + depthFar) * invDepth, -1.0f,
616 0.0f, 0.0f, (2.0f * depthFar * depthNear) * invDepth, 0.0f);
617}

650FloatPoint TransformationMatrix::projectPoint(const FloatPoint& p, bool* clamped) const
651{
652 // This is basically raytracing. We have a point in the destination
653 // plane with z=0, and we cast a ray parallel to the z-axis from that
654 // point to find the z-position at which it intersects the z=0 plane
655 // with the transform applied. Once we have that point we apply the
656 // inverse transform to find the corresponding point in the source
657 // space.
658 //
659 // Given a plane with normal Pn, and a ray starting at point R0 and
660 // with direction defined by the vector Rd, we can find the
661 // intersection point as a distance d from R0 in units of Rd by:
662 //
663 // d = -dot (Pn', R0) / dot (Pn', Rd)
664 if (clamped)
665 *clamped = false;
666
667 if (m33() == 0) {
668 // In this case, the projection plane is parallel to the ray we are trying to
669 // trace, and there is no well-defined value for the projection.
670 return FloatPoint();
671 }
672
673 double x = p.x();
674 double y = p.y();
675 double z = -(m13() * x + m23() * y + m43()) / m33();
676
677 // FIXME: use multVecMatrix()
678 double outX = x * m11() + y * m21() + z * m31() + m41();
679 double outY = x * m12() + y * m22() + z * m32() + m42();
680
681 double w = x * m14() + y * m24() + z * m34() + m44();
682 if (w <= 0) {
683 // Using int max causes overflow when other code uses the projected point. To
684 // represent infinity yet reduce the risk of overflow, we use a large but
685 // not-too-large number here when clamping.
686 const int largeNumber = 100000000 / kFixedPointDenominator;
687 outX = copysign(largeNumber, outX);
688 outY = copysign(largeNumber, outY);
689 if (clamped)
690 *clamped = true;
691 } else if (w != 1) {
692 outX /= w;
693 outY /= w;
694 }
695
696 return FloatPoint(static_cast<float>(outX), static_cast<float>(outY));
697}

699FloatQuad TransformationMatrix::projectQuad(const FloatQuad& q, bool* clamped) const
700{
701 FloatQuad projectedQuad;
702
703 bool clamped1 = false;
704 bool clamped2 = false;
705 bool clamped3 = false;
706 bool clamped4 = false;
707
708 projectedQuad.setP1(projectPoint(q.p1(), &clamped1));
709 projectedQuad.setP2(projectPoint(q.p2(), &clamped2));
710 projectedQuad.setP3(projectPoint(q.p3(), &clamped3));
711 projectedQuad.setP4(projectPoint(q.p4(), &clamped4));
712
713 if (clamped)
714 *clamped = clamped1 || clamped2 || clamped3 || clamped4;
715
716 // If all points on the quad had w < 0, then the entire quad would not be visible to the projected surface.
717 bool everythingWasClipped = clamped1 && clamped2 && clamped3 && clamped4;
718 if (everythingWasClipped)
719 return FloatQuad();
720
721 return projectedQuad;
722}

910TransformationMatrix& TransformationMatrix::rotate3d(double x, double y, double z, double angle)
911{
912 // Normalize the axis of rotation
913 double length = std::hypot(x, y, z);
914 if (length == 0) {
915 // A direction vector that cannot be normalized, such as [0, 0, 0], will cause the rotation to not be applied.
916 return *this;
917 } else if (length != 1) {
918 x /= length;
919 y /= length;
920 z /= length;
921 }
922
923 // Angles are in degrees. Switch to radians.
924 angle = deg2rad(angle);
925
926 double sinTheta = sin(angle);
927 double cosTheta = cos(angle);
928
929 TransformationMatrix mat;
930
931 // Optimize cases where the axis is along a major axis
932 if (x == 1.0 && y == 0.0 && z == 0.0) {
933 mat.m_matrix[0][0] = 1.0;
934 mat.m_matrix[0][1] = 0.0;
935 mat.m_matrix[0][2] = 0.0;
936 mat.m_matrix[1][0] = 0.0;
937 mat.m_matrix[1][1] = cosTheta;
938 mat.m_matrix[1][2] = sinTheta;
939 mat.m_matrix[2][0] = 0.0;
940 mat.m_matrix[2][1] = -sinTheta;
941 mat.m_matrix[2][2] = cosTheta;
942 mat.m_matrix[0][3] = mat.m_matrix[1][3] = mat.m_matrix[2][3] = 0.0;
943 mat.m_matrix[3][0] = mat.m_matrix[3][1] = mat.m_matrix[3][2] = 0.0;
944 mat.m_matrix[3][3] = 1.0;
945 } else if (x == 0.0 && y == 1.0 && z == 0.0) {
946 mat.m_matrix[0][0] = cosTheta;
947 mat.m_matrix[0][1] = 0.0;
948 mat.m_matrix[0][2] = -sinTheta;
949 mat.m_matrix[1][0] = 0.0;
950 mat.m_matrix[1][1] = 1.0;
951 mat.m_matrix[1][2] = 0.0;
952 mat.m_matrix[2][0] = sinTheta;
953 mat.m_matrix[2][1] = 0.0;
954 mat.m_matrix[2][2] = cosTheta;
955 mat.m_matrix[0][3] = mat.m_matrix[1][3] = mat.m_matrix[2][3] = 0.0;
956 mat.m_matrix[3][0] = mat.m_matrix[3][1] = mat.m_matrix[3][2] = 0.0;
957 mat.m_matrix[3][3] = 1.0;
958 } else if (x == 0.0 && y == 0.0 && z == 1.0) {
959 mat.m_matrix[0][0] = cosTheta;
960 mat.m_matrix[0][1] = sinTheta;
961 mat.m_matrix[0][2] = 0.0;
962 mat.m_matrix[1][0] = -sinTheta;
963 mat.m_matrix[1][1] = cosTheta;
964 mat.m_matrix[1][2] = 0.0;
965 mat.m_matrix[2][0] = 0.0;
966 mat.m_matrix[2][1] = 0.0;
967 mat.m_matrix[2][2] = 1.0;
968 mat.m_matrix[0][3] = mat.m_matrix[1][3] = mat.m_matrix[2][3] = 0.0;
969 mat.m_matrix[3][0] = mat.m_matrix[3][1] = mat.m_matrix[3][2] = 0.0;
970 mat.m_matrix[3][3] = 1.0;
971 } else {
972 // This case is the rotation about an arbitrary unit vector.
973 //
974 // Formula is adapted from Wikipedia article on Rotation matrix,
975 // http://en.wikipedia.org/wiki/Rotation_matrix#Rotation_matrix_from_axis_and_angle
976 //
977 // An alternate resource with the same matrix: http://www.fastgraph.com/makegames/3drotation/
978 //
979 double oneMinusCosTheta = 1 - cosTheta;
980 mat.m_matrix[0][0] = cosTheta + x * x * oneMinusCosTheta;
981 mat.m_matrix[0][1] = y * x * oneMinusCosTheta + z * sinTheta;
982 mat.m_matrix[0][2] = z * x * oneMinusCosTheta - y * sinTheta;
983 mat.m_matrix[1][0] = x * y * oneMinusCosTheta - z * sinTheta;
984 mat.m_matrix[1][1] = cosTheta + y * y * oneMinusCosTheta;
985 mat.m_matrix[1][2] = z * y * oneMinusCosTheta + x * sinTheta;
986 mat.m_matrix[2][0] = x * z * oneMinusCosTheta + y * sinTheta;
987 mat.m_matrix[2][1] = y * z * oneMinusCosTheta - x * sinTheta;
988 mat.m_matrix[2][2] = cosTheta + z * z * oneMinusCosTheta;
989 mat.m_matrix[0][3] = mat.m_matrix[1][3] = mat.m_matrix[2][3] = 0.0;
990 mat.m_matrix[3][0] = mat.m_matrix[3][1] = mat.m_matrix[3][2] = 0.0;
991 mat.m_matrix[3][3] = 1.0;
992 }
993 multiply(mat);
994 return *this;
995}

1649 if (progress < 0.5)

1652 }
1653
1654 // If x-axis of one is flipped, and y-axis of the other, convert to an unflipped rotation.
1655 if ((fromDecomp.scaleX < 0 && toDecomp.scaleY < 0) || (fromDecomp.scaleY < 0 && toDecomp.scaleX < 0)) {

1669 fromDecomp.angle -= 360;
1670 else
1671 toDecomp.angle -= 360;
1672 }

1675 blendFloat(fromDecomp.m12, toDecomp.m12, progress);

1873bool TransformationMatrix::isBackFaceVisible() const
1874{
1875 // Back-face visibility is determined by transforming the normal vector (0, 0, 1) and
1876 // checking the sign of the resulting z component. However, normals cannot be
1877 // transformed by the original matrix, they require being transformed by the
1878 // inverse-transpose.
1879 //
1880 // Since we know we will be using (0, 0, 1), and we only care about the z-component of
1881 // the transformed normal, then we only need the m33() element of the
1882 // inverse-transpose. Therefore we do not need the transpose.
1883 //
1884 // Additionally, if we only need the m33() element, we do not need to compute a full
1885 // inverse. Instead, knowing the inverse of a matrix is adjoint(matrix) / determinant,
1886 // we can simply compute the m33() of the adjoint (adjugate) matrix, without computing
1887 // the full adjoint.
1888
1889 double determinant = WebCore::determinant4x4(m_matrix);
1890
1891 // If the matrix is not invertible, then we assume its backface is not visible.
1892 if (fabs(determinant) < SMALL_NUMBER)
1893 return false;
1894
1895 double cofactor33 = determinant3x3(m11(), m12(), m14(), m21(), m22(), m24(), m41(), m42(), m44());
1896 double zComponentOfTransformedNormal = cofactor33 / determinant;
1897
1898 return zComponentOfTransformedNormal < 0;
1899}
