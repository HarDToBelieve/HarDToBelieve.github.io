137static const char *safeTypeForDOMToReadAndWriteForPlatformType(const String& platformType, PlatformPasteboard::IncludeImageTypes includeImageTypes)
138{
139 auto cfType = platformType.createCFString();
140ALLOW_DEPRECATED_DECLARATIONS_BEGIN
141 if (UTTypeConformsTo(cfType.get(), kUTTypePlainText))
142 return "text/plain"_s;
143
144 if (UTTypeConformsTo(cfType.get(), kUTTypeHTML) || UTTypeConformsTo(cfType.get(), (CFStringRef)WebArchivePboardType)
145 || UTTypeConformsTo(cfType.get(), kUTTypeRTF) || UTTypeConformsTo(cfType.get(), kUTTypeFlatRTFD))
146 return "text/html"_s;
147
148 if (UTTypeConformsTo(cfType.get(), kUTTypeURL))
149 return "text/uri-list"_s;
150
151 if (includeImageTypes == PlatformPasteboard::IncludeImageTypes::Yes && UTTypeConformsTo(cfType.get(), kUTTypePNG))
152 return "image/png"_s;
153ALLOW_DEPRECATED_DECLARATIONS_END
154
155 return nullptr;
156}

158static Vector<String> webSafeTypes(NSArray<NSString *> *platformTypes, PlatformPasteboard::IncludeImageTypes includeImageTypes, Function<bool()>&& shouldAvoidExposingURLType)
159{
160 ListHashSet<String> domPasteboardTypes;
161 for (NSString *type in platformTypes) {
162 if ([type isEqualToString:@(PasteboardCustomData::cocoaType())])
163 continue;
164
165 if (Pasteboard::isSafeTypeForDOMToReadAndWrite(type)) {
166 domPasteboardTypes.add(type);
167 continue;
168 }
169
170 if (auto* coercedType = safeTypeForDOMToReadAndWriteForPlatformType(type, includeImageTypes)) {
171 auto domTypeAsString = String::fromUTF8(coercedType);
172ALLOW_DEPRECATED_DECLARATIONS_BEGIN
173 if (domTypeAsString == "text/uri-list"_s && ([platformTypes containsObject:(__bridge NSString *)kUTTypeFileURL] || shouldAvoidExposingURLType()))
174 continue;
175ALLOW_DEPRECATED_DECLARATIONS_END
176
177 domPasteboardTypes.add(WTFMove(domTypeAsString));
178 }
179 }
180 return copyToVector(domPasteboardTypes);
181}

202std::optional<PasteboardItemInfo> PlatformPasteboard::informationForItemAtIndex(size_t index, int64_t changeCount)
203{
204 if (index >= static_cast<NSUInteger>([m_pasteboard numberOfItems]))
205 return std::nullopt;
206
207 if (this->changeCount() != changeCount)
208 return std::nullopt;
209
210 PasteboardItemInfo info;
211 NSItemProvider *itemProvider = [[m_pasteboard itemProviders] objectAtIndex:index];
212 if ([m_pasteboard respondsToSelector:@selector(fileUploadURLsAtIndex:fileTypes:)]) {
213 NSArray<NSString *> *fileTypes = nil;
214 NSArray *urls = [m_pasteboard fileUploadURLsAtIndex:index fileTypes:&fileTypes];
215 ASSERT(fileTypes.count == urls.count);
216
217 info.pathsForFileUpload.reserveInitialCapacity(urls.count);
218 for (NSURL *url in urls)
219 info.pathsForFileUpload.uncheckedAppend(url.path);
220
221 info.platformTypesForFileUpload.reserveInitialCapacity(fileTypes.count);
222 for (NSString *fileType in fileTypes)
223 info.platformTypesForFileUpload.uncheckedAppend(fileType);
224 } else {
225 NSArray *fileTypes = itemProvider.web_fileUploadContentTypes;
226 info.platformTypesForFileUpload.reserveInitialCapacity(fileTypes.count);
227 info.pathsForFileUpload.reserveInitialCapacity(fileTypes.count);
228 for (NSString *fileType in fileTypes) {
229 info.platformTypesForFileUpload.uncheckedAppend(fileType);
230 info.pathsForFileUpload.uncheckedAppend({ });
231 }
232 }
233
234#if PASTEBOARD_SUPPORTS_PRESENTATION_STYLE_AND_TEAM_DATA
235 info.preferredPresentationStyle = pasteboardItemPresentationStyle(itemProvider.preferredPresentationStyle);
236#endif
237 if (!CGSizeEqualToSize(itemProvider.preferredPresentationSize, CGSizeZero)) {
238 auto adjustedPreferredPresentationHeight = [](auto height) -> std::optional<double> {
239 if (!IOSApplication::isMobileMail() && !IOSApplication::isMailCompositionService())
240 return { height };
241 // Mail's max-width: 100%; default style is in conflict with the preferred presentation size and can lead to unexpectedly stretched images. Not setting the height forces layout to preserve the aspect ratio.
242 return { };
243 };
244 info.preferredPresentationSize = PresentationSize { itemProvider.preferredPresentationSize.width, adjustedPreferredPresentationHeight(itemProvider.preferredPresentationSize.height) };
245 }
246 info.containsFileURLAndFileUploadContent = itemProvider.web_containsFileURLAndFileUploadContent;
247 info.suggestedFileName = itemProvider.suggestedName;
248 NSArray<NSString *> *registeredTypeIdentifiers = itemProvider.registeredTypeIdentifiers;
249 info.platformTypesByFidelity.reserveInitialCapacity(registeredTypeIdentifiers.count);
250 for (NSString *typeIdentifier in registeredTypeIdentifiers) {
251 info.platformTypesByFidelity.uncheckedAppend(typeIdentifier);
252 CFStringRef cfTypeIdentifier = (CFStringRef)typeIdentifier;
253ALLOW_DEPRECATED_DECLARATIONS_BEGIN
254 if (!UTTypeIsDeclared(cfTypeIdentifier))
255 continue;
256
257 if (UTTypeConformsTo(cfTypeIdentifier, kUTTypeText))
258 continue;
259
260 if (UTTypeConformsTo(cfTypeIdentifier, kUTTypeURL))
261 continue;
262
263 if (UTTypeConformsTo(cfTypeIdentifier, kUTTypeRTFD))
264 continue;
265
266 if (UTTypeConformsTo(cfTypeIdentifier, kUTTypeFlatRTFD))
267 continue;
268ALLOW_DEPRECATED_DECLARATIONS_END
269
270 info.isNonTextType = true;
271 }
272
273 info.webSafeTypesByFidelity = webSafeTypes(registeredTypeIdentifiers, IncludeImageTypes::Yes, [&] {
274ALLOW_DEPRECATED_DECLARATIONS_BEGIN
275 return shouldTreatAtLeastOneTypeAsFile(registeredTypeIdentifiers) && !Pasteboard::canExposeURLToDOMWhenPasteboardContainsFiles(readString(index, kUTTypeURL));
276ALLOW_DEPRECATED_DECLARATIONS_END
277 });
278
279 return info;
280}

384static void registerItemsToPasteboard(NSArray<WebItemProviderRegistrationInfoList *> *itemLists, id <AbstractPasteboard> pasteboard)
385{
386#if PLATFORM(MACCATALYST)
387 // In macCatalyst, -[UIPasteboard setItemProviders:] is not yet supported, so we fall back to setting an item dictionary when
388 // populating the pasteboard upon copy.
389 if ([pasteboard isKindOfClass:PAL::getUIPasteboardClass()]) {
390 auto itemDictionaries = adoptNS([[NSMutableArray alloc] initWithCapacity:itemLists.count]);
391 for (WebItemProviderRegistrationInfoList *representationsToRegister in itemLists) {
392 auto itemDictionary = adoptNS([[NSMutableDictionary alloc] initWithCapacity:representationsToRegister.numberOfItems]);
393 [representationsToRegister enumerateItems:[itemDictionary] (id <WebItemProviderRegistrar> item, NSUInteger) {
394 if ([item respondsToSelector:@selector(typeIdentifierForClient)] && [item respondsToSelector:@selector(dataForClient)])
395 [itemDictionary setObject:item.dataForClient forKey:item.typeIdentifierForClient];
396 }];
397 [itemDictionaries addObject:itemDictionary.get()];
398 }
399 [pasteboard setItems:itemDictionaries.get()];
400 return;
401 }
402#endif // PLATFORM(MACCATALYST)
403
404 auto itemProviders = adoptNS([[NSMutableArray alloc] initWithCapacity:itemLists.count]);
405 for (WebItemProviderRegistrationInfoList *representationsToRegister in itemLists) {
406 if (auto *itemProvider = representationsToRegister.itemProvider)
407 [itemProviders addObject:itemProvider];
408 }
409
410 [pasteboard setItemProviders:itemProviders.get()];
411
412 if ([pasteboard respondsToSelector:@selector(stageRegistrationLists:)])
413 [pasteboard stageRegistrationLists:itemLists];
414}

430static void addRepresentationsForPlainText(WebItemProviderRegistrationInfoList *itemsToRegister, const String& plainText)
431{
432 if (plainText.isEmpty())
433 return;

435 NSURL *platformURL = [NSURL URLWithString:plainText];

437 [itemsToRegister addRepresentingObject:platformURL];
438
439ALLOW_DEPRECATED_DECLARATIONS_BEGIN

441ALLOW_DEPRECATED_DECLARATIONS_END

445{

457void PlatformPasteboard::write(const PasteboardWebContent& content)
458{
459 auto representationsToRegister = adoptNS([[WebItemProviderRegistrationInfoList alloc] init]);
460
461#if !PLATFORM(MACCATALYST)
462 [representationsToRegister addData:[webIOSPastePboardType dataUsingEncoding:NSUTF8StringEncoding] forType:webIOSPastePboardType];
463#endif
464
465 ASSERT(content.clientTypes.size() == content.clientData.size());
466 for (size_t i = 0, size = content.clientTypes.size(); i < size; ++i)
467 [representationsToRegister addData:content.clientData[i]->createNSData().get() forType:content.clientTypes[i]];
468
469 if (content.dataInWebArchiveFormat) {
470 auto webArchiveData = content.dataInWebArchiveFormat->createNSData();
471#if PLATFORM(MACCATALYST)
472ALLOW_DEPRECATED_DECLARATIONS_BEGIN
473 NSString *webArchiveType = (__bridge NSString *)kUTTypeWebArchive;
474ALLOW_DEPRECATED_DECLARATIONS_END
475#else
476 // FIXME: We should additionally register "com.apple.webarchive" once <rdar://problem/46830277> is fixed.
477 NSString *webArchiveType = WebArchivePboardType;
478#endif
479 [representationsToRegister addData:webArchiveData.get() forType:webArchiveType];
480 }
481
482 if (content.dataInAttributedStringFormat) {
483 if (NSAttributedString *attributedString = [NSKeyedUnarchiver unarchivedObjectOfClasses:[NSSet setWithObject:NSAttributedString.class] fromData:content.dataInAttributedStringFormat->createNSData().get() error:nullptr])
484 [representationsToRegister addRepresentingObject:attributedString];
485 }
486
487ALLOW_DEPRECATED_DECLARATIONS_BEGIN
488 if (content.dataInRTFDFormat)
489 [representationsToRegister addData:content.dataInRTFDFormat->createNSData().get() forType:(NSString *)kUTTypeFlatRTFD];
490
491 if (content.dataInRTFFormat)
492 [representationsToRegister addData:content.dataInRTFFormat->createNSData().get() forType:(NSString *)kUTTypeRTF];
493
494 if (!content.dataInHTMLFormat.isEmpty()) {
495 NSData *htmlAsData = [(NSString *)content.dataInHTMLFormat dataUsingEncoding:NSUTF8StringEncoding];
496 [representationsToRegister addData:htmlAsData forType:(NSString *)kUTTypeHTML];
497 }
498ALLOW_DEPRECATED_DECLARATIONS_END
499
500 if (!content.dataInStringFormat.isEmpty())
501 addRepresentationsForPlainText(representationsToRegister.get(), content.dataInStringFormat);
502
503 PasteboardCustomData customData;
504 customData.setOrigin(content.contentOrigin);
505 [representationsToRegister addData:customData.createSharedBuffer()->createNSData().get() forType:@(PasteboardCustomData::cocoaType())];
506
507 registerItemToPasteboard(representationsToRegister.get(), m_pasteboard.get());
508}

510void PlatformPasteboard::write(const PasteboardImage& pasteboardImage)
511{
512 auto representationsToRegister = adoptNS([[WebItemProviderRegistrationInfoList alloc] init]);
513
514 auto& types = pasteboardImage.clientTypes;
515 auto& data = pasteboardImage.clientData;
516 ASSERT(types.size() == data.size());
517 for (size_t i = 0, size = types.size(); i < size; ++i)
518 [representationsToRegister addData:data[i]->createNSData().get() forType:types[i]];
519
520 if (pasteboardImage.resourceData && !pasteboardImage.resourceMIMEType.isEmpty()) {
521 auto utiOrMIMEType = pasteboardImage.resourceMIMEType;
522 if (!isDeclaredUTI(utiOrMIMEType))
523 utiOrMIMEType = UTIFromMIMEType(utiOrMIMEType);
524
525 auto imageData = pasteboardImage.resourceData->createNSData();
526 [representationsToRegister addData:imageData.get() forType:(NSString *)utiOrMIMEType];
527 [representationsToRegister setPreferredPresentationSize:pasteboardImage.imageSize];
528 [representationsToRegister setSuggestedName:pasteboardImage.suggestedName];
529 }
530
531 // FIXME: When writing a PasteboardImage, we currently always place the image data at a higer fidelity than the
532 // associated image URL. However, in the case of an image enclosed by an anchor, we might want to consider the
533 // the URL (i.e. the anchor's href attribute) to be a higher fidelity representation.
534 auto& pasteboardURL = pasteboardImage.url;
535 if (NSURL *nsURL = pasteboardURL.url) {
536#if NSURL_SUPPORTS_TITLE
537 nsURL._title = pasteboardURL.title.isEmpty() ? WTF::userVisibleString(pasteboardURL.url) : (NSString *)pasteboardURL.title;
538#endif
539 [representationsToRegister addRepresentingObject:nsURL];
540 }
541
542 registerItemToPasteboard(representationsToRegister.get(), m_pasteboard.get());
543}

545void PlatformPasteboard::write(const String& pasteboardType, const String& text)
546{
547 auto representationsToRegister = adoptNS([[WebItemProviderRegistrationInfoList alloc] init]);
548 [representationsToRegister setPreferredPresentationStyle:WebPreferredPresentationStyleInline];
549
550 NSString *pasteboardTypeAsNSString = pasteboardType;
551 if (!text.isEmpty() && pasteboardTypeAsNSString.length) {
552 auto pasteboardTypeAsCFString = (CFStringRef)pasteboardTypeAsNSString;
553ALLOW_DEPRECATED_DECLARATIONS_BEGIN
554 if (UTTypeConformsTo(pasteboardTypeAsCFString, kUTTypeURL) || UTTypeConformsTo(pasteboardTypeAsCFString, kUTTypeText))
555 addRepresentationsForPlainText(representationsToRegister.get(), text);
556 else
557 [representationsToRegister addData:[pasteboardTypeAsNSString dataUsingEncoding:NSUTF8StringEncoding] forType:pasteboardType];
558ALLOW_DEPRECATED_DECLARATIONS_END
559 }
560
561 registerItemToPasteboard(representationsToRegister.get(), m_pasteboard.get());
562}

581static const char originKeyForTeamData[] = "com.apple.WebKit.drag-and-drop-team-data.origin";
582static const char customTypesKeyForTeamData[] = "com.apple.WebKit.drag-and-drop-team-data.custom-types";
583
584Vector<String> PlatformPasteboard::typesSafeForDOMToReadAndWrite(const String& origin) const
585{
586 ListHashSet<String> domPasteboardTypes;
587#if PASTEBOARD_SUPPORTS_PRESENTATION_STYLE_AND_TEAM_DATA
588 for (NSItemProvider *provider in [m_pasteboard itemProviders]) {
589 if (!provider.teamData.length)
590 continue;
591
592 NSSet *classes = [NSSet setWithObjects:NSDictionary.class, NSString.class, NSArray.class, nil];
593 id teamDataObject = [NSKeyedUnarchiver unarchivedObjectOfClasses:classes fromData:provider.teamData error:nullptr];
594 if (![teamDataObject isKindOfClass:NSDictionary.class])
595 continue;
596
597 id originInTeamData = [teamDataObject objectForKey:@(originKeyForTeamData)];
598 if (![originInTeamData isKindOfClass:NSString.class])
599 continue;
600 if (String((NSString *)originInTeamData) != origin)
601 continue;
602
603 id customTypes = [teamDataObject objectForKey:@(customTypesKeyForTeamData)];
604 if (![customTypes isKindOfClass:NSArray.class])
605 continue;
606
607 for (NSString *type in customTypes)
608 domPasteboardTypes.add(type);
609 }
610#endif // PASTEBOARD_SUPPORTS_PRESENTATION_STYLE_AND_TEAM_DATA

633static RetainPtr<WebItemProviderRegistrationInfoList> createItemProviderRegistrationList(const PasteboardCustomData& data)
634{
635 auto representationsToRegister = adoptNS([[WebItemProviderRegistrationInfoList alloc] init]);
636 [representationsToRegister setPreferredPresentationStyle:WebPreferredPresentationStyleInline];
637
638 if (data.hasSameOriginCustomData() || !data.origin().isEmpty()) {
639 if (auto serializedSharedBuffer = data.createSharedBuffer()->createNSData()) {
640 // We stash the list of supplied pasteboard types in teamData here for compatibility with drag and drop.
641 // Since the contents of item providers cannot be loaded prior to drop, but the pasteboard types are
642 // contained within the custom data blob and we need to vend them to the page when firing `dragover`
643 // events, we need an additional in-memory representation of the pasteboard types array that contains
644 // all of the custom types. We use the teamData property, available on NSItemProvider on iOS, to store
645 // this information, since the contents of teamData are immediately available prior to the drop.
646 NSDictionary *teamDataDictionary = @{ @(originKeyForTeamData) : data.origin(), @(customTypesKeyForTeamData) : createNSArray(data.orderedTypes()).get() };
647 if (NSData *teamData = [NSKeyedArchiver archivedDataWithRootObject:teamDataDictionary requiringSecureCoding:YES error:nullptr]) {
648 [representationsToRegister setTeamData:teamData];
649 [representationsToRegister addData:serializedSharedBuffer.get() forType:@(PasteboardCustomData::cocoaType())];
650 }
651 }
652 }
653
654 data.forEachPlatformStringOrBuffer([&] (auto& type, auto& value) {
655 auto cocoaType = PlatformPasteboard::platformPasteboardTypeForSafeTypeForDOMToReadAndWrite(type, PlatformPasteboard::IncludeImageTypes::Yes);
656 if (cocoaType.isEmpty())
657 return;
658
659 if (WTF::holds_alternative<String>(value)) {
660 if (WTF::get<String>(value).isNull())
661 return;
662
663 NSString *nsStringValue = WTF::get<String>(value);
664 auto cfType = cocoaType.createCFString();
665ALLOW_DEPRECATED_DECLARATIONS_BEGIN
666 if (UTTypeConformsTo(cfType.get(), kUTTypeURL))
667 [representationsToRegister addRepresentingObject:[NSURL URLWithString:nsStringValue]];
668 else if (UTTypeConformsTo(cfType.get(), kUTTypePlainText))
669 [representationsToRegister addRepresentingObject:nsStringValue];
670 else
671 [representationsToRegister addData:[nsStringValue dataUsingEncoding:NSUTF8StringEncoding] forType:(NSString *)cocoaType];
672ALLOW_DEPRECATED_DECLARATIONS_END
673 return;
674 }
675
676 auto buffer = WTF::get<Ref<SharedBuffer>>(value);
677 [representationsToRegister addData:buffer->createNSData().get() forType:(NSString *)cocoaType];
678 });
679
680 return representationsToRegister;
681}

765{
766ALLOW_DEPRECATED_DECLARATIONS_BEGIN

768 String title;

771
772 if ((NSInteger)index < 0 || (NSInteger)index >= [m_pasteboard numberOfItems])
773 return { };

775 NSIndexSet *indexSet = [NSIndexSet indexSetWithIndex:index];
776 auto value = retainPtr([m_pasteboard valuesForPasteboardType:type inItemSet:indexSet].firstObject ?: [m_pasteboard dataForPasteboardType:type inItemSet:indexSet].firstObject);
777 if (!value)
778 return { };
779
780 if ([value isKindOfClass:[NSData class]])
781 value = adoptNS([[NSString alloc] initWithData:(NSData *)value.get() encoding:NSUTF8StringEncoding]);

783 if (type == String(kUTTypePlainText) || type == String(kUTTypeHTML)) {
784 ASSERT([value isKindOfClass:[NSString class]]);

786 }
787 if (type == String(kUTTypeText)) {
788 ASSERT([value isKindOfClass:[NSString class]] || [value isKindOfClass:[NSAttributedString class]]);
789 if ([value isKindOfClass:[NSString class]])

791 if ([value isKindOfClass:[NSAttributedString class]])
792 return [(NSAttributedString *)value string];
793 }
794ALLOW_DEPRECATED_DECLARATIONS_END

800{

802 return { };
803
804ALLOW_DEPRECATED_DECLARATIONS_BEGIN

840bool PlatformPasteboard::containsURLStringSuitableForLoading()
841{
842 Vector<String> types;
843 getTypes(types);
844ALLOW_DEPRECATED_DECLARATIONS_BEGIN
845 if (!types.contains(String(kUTTypeURL)))
846 return false;
847
848 auto urlString = stringForType(kUTTypeURL);
849ALLOW_DEPRECATED_DECLARATIONS_END
850 if (urlString.isEmpty()) {
851 // On iOS, we don't get access to the contents of NSItemProviders until we perform the drag operation.
852 // Thus, we consider DragData to contain an URL if it contains the `public.url` UTI type. Later down the
853 // road, when we perform the drag operation, we can then check if the URL's protocol is http or https,
854 // and if it isn't, we bail out of page navigation.
855 return true;
856 }
857 return URL { [NSURL URLWithString:urlString] }.protocolIsInHTTPFamily();
858}
