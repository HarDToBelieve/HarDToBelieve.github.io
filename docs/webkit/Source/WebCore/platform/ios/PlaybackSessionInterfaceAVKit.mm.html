51PlaybackSessionInterfaceAVKit::PlaybackSessionInterfaceAVKit(PlaybackSessionModel& model)
52 : m_playerController(adoptNS([[WebAVPlayerController alloc] init]))
53 , m_playbackSessionModel(&model)
54{
55 ASSERT(isUIThread());
56 model.addClient(*this);
57 [m_playerController setPlaybackSessionInterface:this];
58 [m_playerController setDelegate:&model];
59
60 durationChanged(model.duration());
61 currentTimeChanged(model.currentTime(), [[NSProcessInfo processInfo] systemUptime]);
62 bufferedTimeChanged(model.bufferedTime());
63 OptionSet<PlaybackSessionModel::PlaybackState> playbackState;
64 if (model.isPlaying())
65 playbackState.add(PlaybackSessionModel::PlaybackState::Playing);
66 if (model.isStalled())
67 playbackState.add(PlaybackSessionModel::PlaybackState::Stalled);
68 rateChanged(playbackState, model.playbackRate(), model.defaultPlaybackRate());
69 seekableRangesChanged(model.seekableRanges(), model.seekableTimeRangesLastModifiedTime(), model.liveUpdateInterval());
70 canPlayFastReverseChanged(model.canPlayFastReverse());
71 audioMediaSelectionOptionsChanged(model.audioMediaSelectionOptions(), model.audioMediaSelectedIndex());
72 legibleMediaSelectionOptionsChanged(model.legibleMediaSelectionOptions(), model.legibleMediaSelectedIndex());
73 externalPlaybackChanged(model.externalPlaybackEnabled(), model.externalPlaybackTargetType(), model.externalPlaybackLocalizedDeviceName());
74 wirelessVideoPlaybackDisabledChanged(model.wirelessVideoPlaybackDisabled());
75}

77PlaybackSessionInterfaceAVKit::~PlaybackSessionInterfaceAVKit()
78{
79 ASSERT(isUIThread());
80 [m_playerController setPlaybackSessionInterface:nullptr];
81 [m_playerController setExternalPlaybackActive:false];
82
83 invalidate();
84}

91void PlaybackSessionInterfaceAVKit::durationChanged(double duration)
92{
93 WebAVPlayerController* playerController = m_playerController.get();
94
95 playerController.contentDuration = duration;
96 playerController.contentDurationWithinEndTimes = duration;
97
98 // FIXME: we take this as an indication that playback is ready.
99 playerController.canPlay = YES;
100 playerController.canPause = YES;
101 playerController.canTogglePlayback = YES;
102 playerController.hasEnabledAudio = YES;
103 playerController.canSeek = YES;
104 playerController.status = AVPlayerControllerStatusReadyToPlay;
105}

107void PlaybackSessionInterfaceAVKit::currentTimeChanged(double currentTime, double anchorTime)
108{
109 if ([m_playerController isScrubbing])
110 return;
111
112 NSTimeInterval anchorTimeStamp = ![m_playerController rate] ? NAN : anchorTime;
113 AVValueTiming *timing = [getAVValueTimingClass() valueTimingWithAnchorValue:currentTime
114 anchorTimeStamp:anchorTimeStamp rate:0];
115
116 [m_playerController setTiming:timing];
117}
118
119void PlaybackSessionInterfaceAVKit::bufferedTimeChanged(double bufferedTime)
120{
121 WebAVPlayerController* playerController = m_playerController.get();
122 double duration = playerController.contentDuration;
123 double normalizedBufferedTime;
124 if (!duration)
125 normalizedBufferedTime = 0;
126 else
127 normalizedBufferedTime = bufferedTime / duration;
128 playerController.loadedTimeRanges = @[@0, @(normalizedBufferedTime)];
129}

131void PlaybackSessionInterfaceAVKit::rateChanged(OptionSet<PlaybackSessionModel::PlaybackState> playbackState, double playbackRate, double defaultPlaybackRate)
132{
133 [m_playerController setDefaultPlaybackRate:defaultPlaybackRate];
134 if (!playbackState.contains(PlaybackSessionModel::PlaybackState::Stalled))
135 [m_playerController setRate:playbackState.contains(PlaybackSessionModel::PlaybackState::Playing) ? playbackRate : 0.];
136}

138void PlaybackSessionInterfaceAVKit::seekableRangesChanged(const TimeRanges& timeRanges, double lastModifiedTime, double liveUpdateInterval)
139{
140 RetainPtr<NSMutableArray> seekableRanges = adoptNS([[NSMutableArray alloc] init]);
141
142#if !PLATFORM(WATCHOS)
143 for (unsigned i = 0; i < timeRanges.length(); i++) {
144 double start = timeRanges.start(i).releaseReturnValue();
145 double end = timeRanges.end(i).releaseReturnValue();
146
147 CMTimeRange range = PAL::CMTimeRangeMake(PAL::CMTimeMakeWithSeconds(start, 1000), PAL::CMTimeMakeWithSeconds(end-start, 1000));
148 [seekableRanges addObject:[NSValue valueWithCMTimeRange:range]];
149 }
150#else
151 UNUSED_PARAM(timeRanges);
152#endif
153
154 [m_playerController setSeekableTimeRanges:seekableRanges.get()];
155 [m_playerController setSeekableTimeRangesLastModifiedTime: lastModifiedTime];
156 [m_playerController setLiveUpdateInterval:liveUpdateInterval];
157}

164static RetainPtr<NSArray> mediaSelectionOptions(const Vector<MediaSelectionOption>& options)
165{
166 return createNSArray(options, [] (auto& option) {
167 auto webOption = adoptNS([[WebAVMediaSelectionOption alloc] init]);
168 [webOption setLocalizedDisplayName:option.displayName];
169 return webOption;
170 });
171}

173void PlaybackSessionInterfaceAVKit::audioMediaSelectionOptionsChanged(const Vector<MediaSelectionOption>& options, uint64_t selectedIndex)
174{
175 auto webOptions = mediaSelectionOptions(options);
176 [m_playerController setAudioMediaSelectionOptions:webOptions.get()];
177 if (selectedIndex < [webOptions count])
178 [m_playerController setCurrentAudioMediaSelectionOption:[webOptions objectAtIndex:static_cast<NSUInteger>(selectedIndex)]];
179}

189void PlaybackSessionInterfaceAVKit::externalPlaybackChanged(bool enabled, PlaybackSessionModel::ExternalPlaybackTargetType targetType, const String& localizedDeviceName)
190{
191 AVPlayerControllerExternalPlaybackType externalPlaybackType = AVPlayerControllerExternalPlaybackTypeNone;
192 if (enabled && targetType == PlaybackSessionModel::TargetTypeAirPlay)
193 externalPlaybackType = AVPlayerControllerExternalPlaybackTypeAirPlay;
194 else if (enabled && targetType == PlaybackSessionModel::TargetTypeTVOut)
195 externalPlaybackType = AVPlayerControllerExternalPlaybackTypeTVOut;
196
197 WebAVPlayerController* playerController = m_playerController.get();
198 playerController.externalPlaybackAirPlayDeviceLocalizedName = localizedDeviceName;
199 playerController.externalPlaybackType = externalPlaybackType;
200 playerController.externalPlaybackActive = enabled;
201}

218void PlaybackSessionInterfaceAVKit::invalidate()
219{
220 if (!m_playbackSessionModel)
221 return;
222
223 [m_playerController setDelegate:nullptr];
224 m_playbackSessionModel->removeClient(*this);
225 m_playbackSessionModel = nullptr;
226}
227
228void PlaybackSessionInterfaceAVKit::modelDestroyed()
229{
230 ASSERT(isUIThread());
231 invalidate();
232 ASSERT(!m_playbackSessionModel);
233}
