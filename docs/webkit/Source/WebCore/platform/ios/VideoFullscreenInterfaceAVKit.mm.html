270- (FloatRect)calculateTargetVideoFrame
271{
272 FloatRect targetVideoFrame;
273 float videoAspectRatio = self.videoDimensions.width / self.videoDimensions.height;
274
275 if ([AVLayerVideoGravityResize isEqualToString:self.videoGravity])
276 targetVideoFrame = self.bounds;
277 else if ([AVLayerVideoGravityResizeAspect isEqualToString:self.videoGravity])
278 targetVideoFrame = largestRectWithAspectRatioInsideRect(videoAspectRatio, self.bounds);
279 else if ([AVLayerVideoGravityResizeAspectFill isEqualToString:self.videoGravity])
280 targetVideoFrame = smallestRectWithAspectRatioAroundRect(videoAspectRatio, self.bounds);
281 else
282 ASSERT_NOT_REACHED();
283
284 return targetVideoFrame;
285}

287- (void)layoutSublayers
288{
289 if ([_videoSublayer superlayer] != self)
290 return;
291
292 if (![_avPlayerController delegate])
293 return;
294
295 [_videoSublayer setPosition:CGPointMake(CGRectGetMidX(self.bounds), CGRectGetMidY(self.bounds))];
296
297 if (self.videoDimensions.height <= 0 || self.videoDimensions.width <= 0)
298 return;
299
300 FloatRect sourceVideoFrame;
301 FloatRect targetVideoFrame;
302 float videoAspectRatio = self.videoDimensions.width / self.videoDimensions.height;
303
304 if ([AVLayerVideoGravityResize isEqualToString:_previousVideoGravity.get()])
305 sourceVideoFrame = self.modelVideoLayerFrame;
306 else if ([AVLayerVideoGravityResizeAspect isEqualToString:_previousVideoGravity.get()])
307 sourceVideoFrame = largestRectWithAspectRatioInsideRect(videoAspectRatio, self.modelVideoLayerFrame);
308 else if ([AVLayerVideoGravityResizeAspectFill isEqualToString:_previousVideoGravity.get()])
309 sourceVideoFrame = smallestRectWithAspectRatioAroundRect(videoAspectRatio, self.modelVideoLayerFrame);
310 else
311 ASSERT_NOT_REACHED();
312
313 targetVideoFrame = [self calculateTargetVideoFrame];
314
315 UIView *view = (UIView *)[_videoSublayer delegate];
316 CGAffineTransform transform = CGAffineTransformMakeScale(targetVideoFrame.width() / sourceVideoFrame.width(), targetVideoFrame.height() / sourceVideoFrame.height());
317 [view setTransform:transform];
318
319 NSTimeInterval animationDuration = [CATransaction animationDuration];
320 RunLoop::main().dispatch([self, strongSelf = retainPtr(self), targetVideoFrame, animationDuration] {
321 [NSObject cancelPreviousPerformRequestsWithTarget:self selector:@selector(resolveBounds) object:nil];
322
323 _videoSublayerFrame = targetVideoFrame;
324 [self performSelector:@selector(resolveBounds) withObject:nil afterDelay:animationDuration + 0.1];
325 });
326}
327
328- (void)resolveBounds
329{
330 [NSObject cancelPreviousPerformRequestsWithTarget:self selector:@selector(resolveBounds) object:nil];
331 if (![_avPlayerController delegate])
332 return;
333
334 if ([_videoSublayer superlayer] != self)
335 return;
336
337 if (CGRectEqualToRect(self.modelVideoLayerFrame, [self bounds]) && CGAffineTransformIsIdentity([(UIView *)[_videoSublayer delegate] transform]))
338 return;
339
340 [CATransaction begin];
341 [CATransaction setAnimationDuration:0];
342 [CATransaction setDisableActions:YES];
343
344 self.modelVideoLayerFrame = [self bounds];
345 if (auto* model = _fullscreenInterface->videoFullscreenModel())
346 model->setVideoLayerFrame(_videoSublayerFrame);
347
348 _previousVideoGravity = _videoGravity;
349
350 [(UIView *)[_videoSublayer delegate] setTransform:CGAffineTransformIdentity];
351
352 [CATransaction commit];
353}

355- (void)setVideoGravity:(NSString *)videoGravity
356{
357#if PLATFORM(MACCATALYST)
358 // FIXME<rdar://46011230>: remove this #if once this radar lands.
359 if (!videoGravity)
360 videoGravity = AVLayerVideoGravityResizeAspect;
361#endif
362
363 if ([_videoGravity.get() isEqualToString:videoGravity])
364 return;
365
366 _previousVideoGravity = _videoGravity;
367 _videoGravity = videoGravity;
368
369 if (![_avPlayerController delegate])
370 return;
371
372 MediaPlayerEnums::VideoGravity gravity = MediaPlayerEnums::VideoGravity::ResizeAspect;
373 if (videoGravity == AVLayerVideoGravityResize)
374 gravity = MediaPlayerEnums::VideoGravity::Resize;
375 if (videoGravity == AVLayerVideoGravityResizeAspect)
376 gravity = MediaPlayerEnums::VideoGravity::ResizeAspect;
377 else if (videoGravity == AVLayerVideoGravityResizeAspectFill)
378 gravity = MediaPlayerEnums::VideoGravity::ResizeAspectFill;
379 else
380 ASSERT_NOT_REACHED();
381
382 if (auto* model = _fullscreenInterface->videoFullscreenModel())
383 model->setVideoLayerGravity(gravity);
384
385 [self setNeedsLayout];
386}

1251bool VideoFullscreenInterfaceAVKit::shouldExitFullscreenWithReason(VideoFullscreenInterfaceAVKit::ExitFullScreenReason reason)
1252{
1253 // AVKit calls playerViewController:shouldExitFullScreenWithReason in the scenario that the exit fullscreen request
1254 // is from the web process (e.g., through Javascript API videoElement.webkitExitFullscreen()).
1255 // We have to ignore the callback in that case.
1256 if (m_shouldIgnoreAVKitCallbackAboutExitFullscreenReason)
1257 return true;
1258
1259 if (!m_videoFullscreenModel)
1260 return true;
1261
1262 if (reason == ExitFullScreenReason::PictureInPictureStarted)
1263 return false;
1264
1265 if (playbackSessionModel() && (reason == ExitFullScreenReason::DoneButtonTapped || reason == ExitFullScreenReason::RemoteControlStopEventReceived))
1266 playbackSessionModel()->pause();
1267
1268 if (!m_watchdogTimer.isActive() && !ignoreWatchdogForDebugging)
1269 m_watchdogTimer.startOneShot(defaultWatchdogTimerInterval);
1270
1271#if PLATFORM(WATCHOS)
1272 if (m_fullscreenChangeObserver) {
1273 m_waitingForPreparedToExit = true;
1274 m_fullscreenChangeObserver->willExitFullscreen();
1275 return false;
1276 }
1277#endif
1278
1279 BOOL finished = reason == ExitFullScreenReason::DoneButtonTapped || reason == ExitFullScreenReason::PinchGestureHandled;
1280 ASSERT(m_videoFullscreenModel);
1281 if (m_videoFullscreenModel)
1282 m_videoFullscreenModel->requestFullscreenMode(HTMLMediaElementEnums::VideoFullscreenModeNone, finished);
1283
1284 return false;
1285}
