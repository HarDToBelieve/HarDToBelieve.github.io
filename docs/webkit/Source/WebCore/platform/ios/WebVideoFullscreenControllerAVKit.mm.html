82static IntRect elementRectInWindow(HTMLVideoElement* videoElement)
83{
84 if (!videoElement)
85 return { };
86 auto* renderer = videoElement->renderer();
87 auto* view = videoElement->document().view();
88 if (!renderer || !view)
89 return { };
90 return view->convertToContainingWindow(renderer->absoluteBoundingBoxRect());
91}

226VideoFullscreenControllerContext::~VideoFullscreenControllerContext()
227{
228 auto notifyClientsModelWasDestroyed = [this] {
229 while (!m_playbackClients.isEmpty())
230 (*m_playbackClients.begin())->modelDestroyed();
231 while (!m_fullscreenClients.isEmpty())
232 (*m_fullscreenClients.begin())->modelDestroyed();
233 };
234 if (isUIThread()) {
235 WebThreadLock();
236 notifyClientsModelWasDestroyed();
237 m_playbackModel = nullptr;
238 m_fullscreenModel = nullptr;
239 } else
240 WorkQueue::main().dispatchSync(WTFMove(notifyClientsModelWasDestroyed));
241}

347void VideoFullscreenControllerContext::didCleanupFullscreen()
348{
349 ASSERT(isUIThread());
350 m_interface->setVideoFullscreenModel(nullptr);
351 m_interface->setVideoFullscreenChangeObserver(nullptr);
352 m_interface = nullptr;
353 m_videoFullscreenView = nil;
354
355 WebThreadRun([protectedThis = makeRefPtr(this), this] {
356 m_fullscreenModel->setVideoFullscreenLayer(nil);
357 m_fullscreenModel->setVideoElement(nullptr);
358 m_playbackModel->setMediaElement(nullptr);
359 m_playbackModel->removeClient(*this);
360 m_fullscreenModel->removeClient(*this);
361 m_fullscreenModel = nullptr;
362 m_videoElement = nullptr;
363
364 [m_controller didFinishFullscreen:this];
365 });
366}

485static Vector<MediaSelectionOption> isolatedCopy(const Vector<MediaSelectionOption>& options)
486{
487 Vector<MediaSelectionOption> optionsCopy;
488 optionsCopy.reserveInitialCapacity(options.size());
489 for (auto& option : options)
490 optionsCopy.uncheckedAppend({ option.displayName.isolatedCopy(), option.type });
491 return optionsCopy;
492}

1001void VideoFullscreenControllerContext::setUpFullscreen(HTMLVideoElement& videoElement, UIView *view, HTMLMediaElementEnums::VideoFullscreenMode mode)
1002{
1003 ASSERT(isMainThread());
1004 RetainPtr<UIView> viewRef = view;
1005 m_videoElement = &videoElement;
1006 m_playbackModel = PlaybackSessionModelMediaElement::create();
1007 m_playbackModel->addClient(*this);
1008 m_playbackModel->setMediaElement(m_videoElement.get());
1009
1010 m_fullscreenModel = VideoFullscreenModelVideoElement::create();
1011 m_fullscreenModel->addClient(*this);
1012 m_fullscreenModel->setVideoElement(m_videoElement.get());
1013
1014 bool allowsPictureInPicture = m_videoElement->webkitSupportsPresentationMode(HTMLVideoElement::VideoPresentationMode::PictureInPicture);
1015
1016 IntRect videoElementClientRect = elementRectInWindow(m_videoElement.get());
1017 FloatRect videoLayerFrame = FloatRect(FloatPoint(), videoElementClientRect.size());
1018 m_fullscreenModel->setVideoLayerFrame(videoLayerFrame);
1019
1020 FloatSize videoDimensions = { (float)videoElement.videoWidth(), (float)videoElement.videoHeight() };
1021
1022 RunLoop::main().dispatch([protectedThis = makeRefPtr(this), this, videoElementClientRect, videoDimensions, viewRef, mode, allowsPictureInPicture] {
1023 ASSERT(isUIThread());
1024 WebThreadLock();
1025
1026 Ref<PlaybackSessionInterfaceAVKit> sessionInterface = PlaybackSessionInterfaceAVKit::create(*this);
1027 m_interface = VideoFullscreenInterfaceAVKit::create(sessionInterface.get());
1028 m_interface->setVideoFullscreenChangeObserver(this);
1029 m_interface->setVideoFullscreenModel(this);
1030
1031 m_videoFullscreenView = adoptNS([PAL::allocUIViewInstance() init]);
1032
1033 m_interface->setupFullscreen(*m_videoFullscreenView.get(), videoElementClientRect, videoDimensions, viewRef.get(), mode, allowsPictureInPicture, false, false);
1034 });
1035}
