34FloatingPointEnvironment& FloatingPointEnvironment::singleton()
35{
36 static NeverDestroyed<FloatingPointEnvironment> floatingPointEnvironment;
37 return floatingPointEnvironment;
38}
39
40#if PLATFORM(IOS_FAMILY) && (CPU(ARM) || CPU(ARM64))
41
42FloatingPointEnvironment::FloatingPointEnvironment()
43 : m_isInitialized(false)
44{
45}
46
47void FloatingPointEnvironment::enableDenormalSupport()
48{
49 RELEASE_ASSERT(isUIThread());
50#if defined _ARM_ARCH_7
51 fenv_t env;
52 fegetenv(&env);
53 env.__fpscr &= ~0x01000000U;
54 fesetenv(&env);
55#endif
56 // Supporting denormal mode is already the default on x86, x86_64, and ARM64.
57}
58
59void FloatingPointEnvironment::saveMainThreadEnvironment()
60{
61 RELEASE_ASSERT(!m_isInitialized);
62 RELEASE_ASSERT(isUIThread());
63 fegetenv(&m_mainThreadEnvironment);
64 m_isInitialized = true;
65}
66
67void FloatingPointEnvironment::propagateMainThreadEnvironment()
68{
69 RELEASE_ASSERT(m_isInitialized);
70 RELEASE_ASSERT(!isUIThread());
71 fesetenv(&m_mainThreadEnvironment);
72}
