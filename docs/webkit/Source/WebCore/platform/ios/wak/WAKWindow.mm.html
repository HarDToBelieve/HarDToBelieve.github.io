216- (BOOL)resignFirstResponder
217{
218 BOOL shouldResign = [super resignFirstResponder];
219 if (shouldResign && _responderView && WKViewResignFirstResponder([_responderView _viewRef])) {
220 _nextResponder = nil;
221 [_responderView release];
222 _responderView = nil;
223 return YES;
224 }
225 return NO;
226}
227
228- (BOOL)makeFirstResponder:(NSResponder *)responder
229{
230 if (![responder isKindOfClass:[WAKView class]])
231 return NO;
232
233 WAKView *view = static_cast<WAKView*>(responder);
234 BOOL result = YES;
235 if (view != _responderView) {
236 // We need to handle the case of the view not accepting to be a first responder,
237 // where we don't need to resign as first responder.
238 BOOL acceptsFirstResponder = view ? WKViewAcceptsFirstResponder([view _viewRef]) : YES;
239 if (acceptsFirstResponder && _responderView) {
240 _nextResponder = view;
241 if (WKViewResignFirstResponder([_responderView _viewRef])) {
242 _nextResponder = nil;
243 [_responderView release];
244 _responderView = nil;
245 } else {
246 _nextResponder = nil;
247 result = NO;
248 }
249 }
250
251 if (result && view) {
252 if (acceptsFirstResponder && WKViewBecomeFirstResponder([view _viewRef]))
253 _responderView = [view retain];
254 else
255 result = NO;
256 }
257 }
258
259 return result;
260}

339- (void)sendEventSynchronously:(WebEvent *)event
340{
341 ASSERT(event);
342 ASSERT(WebThreadIsLockedOrDisabled());
343 auto lastEvent = std::exchange(currentEvent(), event);
344
345 switch (event.type) {
346 case WebEventMouseMoved:
347 case WebEventScrollWheel:
348 if (WAKView *hitView = [_contentView hitTest:event.locationInWindow])
349 [hitView handleEvent:event];
350 break;
351
352 case WebEventMouseUp:
353 case WebEventKeyDown:
354 case WebEventKeyUp:
355 case WebEventTouchChange:
356 [_responderView handleEvent:event];
357 break;
358
359 case WebEventMouseDown:
360 case WebEventTouchBegin:
361 case WebEventTouchEnd:
362 case WebEventTouchCancel:
363 if (WAKView *hitView = [_contentView hitTest:event.locationInWindow]) {
364 [self makeFirstResponder:hitView];
365 [hitView handleEvent:event];
366 }
367 break;
368 }
369
370 currentEvent() = WTFMove(lastEvent);
371}

467- (CGRect)_visibleRectRespectingMasksToBounds:(BOOL)respectsMasksToBounds
468{
469 if (!CGRectIsNull(_frozenVisibleRect))
470 return _frozenVisibleRect;
471
472 CALayer* layer = _hostLayer;
473 CGRect bounds = [layer bounds];
474 if (_entireWindowVisibleForTesting)
475 return bounds;
476 CGRect rect = bounds;
477 CALayer* superlayer = [layer superlayer];
478
479 static Class windowClass = NSClassFromString(@"UIWindow");
480
481 while (superlayer && layer != _rootLayer && (!layer.delegate || ![layer.delegate isKindOfClass:windowClass])) {
482 CGRect rectInSuper = [superlayer convertRect:rect fromLayer:layer];
483 if ([superlayer masksToBounds] || !respectsMasksToBounds)
484 rect = CGRectIntersection([superlayer bounds], rectInSuper);
485 else
486 rect = rectInSuper;
487 layer = superlayer;
488 superlayer = [layer superlayer];
489 }
490
491 if (layer != _hostLayer) {
492 rect = CGRectIntersection([layer bounds], rect);
493 rect = [_hostLayer convertRect:rect fromLayer:layer];
494 }
495
496 return CGRectIntersection(bounds, rect);
497}

651- (void)dumpTiles
652{
653 CGRect savedFrozenVisibleRect = _frozenVisibleRect;
654 NSLog(@"=================");
655 if (!CGRectIsNull(_frozenVisibleRect)) {
656 NSLog(@"Visibility::Visible RECT IS CACHED: [%6.1f %6.1f %6.1f %6.1f]", _frozenVisibleRect.origin.x, _frozenVisibleRect.origin.y, _frozenVisibleRect.size.width, _frozenVisibleRect.size.height);
657 _frozenVisibleRect = CGRectNull;
658 }
659 CGRect visibleRect = [self visibleRect];
660 NSLog(@"visibleRect = [%6.1f %6.1f %6.1f %6.1f]", visibleRect.origin.x, visibleRect.origin.y, visibleRect.size.width, visibleRect.size.height);
661 _frozenVisibleRect = savedFrozenVisibleRect;
662 _tileCache->dumpTiles();
663}

665- (NSString *)description
666{
667 NSMutableString *description = [NSMutableString stringWithFormat:@"<%@: WAK: %p; ", [self class], self];
668
669 CGRect frame = [self frame];
670 [description appendFormat:@"frame = (%g %g; %g %g); ", frame.origin.x, frame.origin.y, frame.size.width, frame.size.height];
671
672 [description appendFormat:@"first responder = WK %p; ", _responderView];
673 [description appendFormat:@"next responder = WK %p; ", _nextResponder];
674
675 [description appendFormat:@"layer = %@>", [_hostLayer description]];
676
677 return description;
678}
