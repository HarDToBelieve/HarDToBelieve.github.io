41class DataDetectorHighlight;

47class DataDetectorHighlightClient : public CanMakeWeakPtr<DataDetectorHighlightClient> {
48public:
49 virtual ~DataDetectorHighlightClient() = default;
50
51 virtual DataDetectorHighlight* activeHighlight() const = 0;

54class DataDetectorHighlight : public RefCounted<DataDetectorHighlight>, private GraphicsLayerClient, public CanMakeWeakPtr<DataDetectorHighlight> {
55 WTF_MAKE_NONCOPYABLE(DataDetectorHighlight);
56public:
57 static Ref<DataDetectorHighlight> createForSelection(Page&, DataDetectorHighlightClient&, RetainPtr<DDHighlightRef>&&, SimpleRange&&);
58 static Ref<DataDetectorHighlight> createForTelephoneNumber(Page&, DataDetectorHighlightClient&, RetainPtr<DDHighlightRef>&&, SimpleRange&&);
59 static Ref<DataDetectorHighlight> createForImageOverlay(Page&, DataDetectorHighlightClient&, RetainPtr<DDHighlightRef>&&, SimpleRange&&);
60
61 ~DataDetectorHighlight() = default;
62
63 void invalidate();
64
65 DDHighlightRef highlight() const { return m_highlight.get(); }
66 const SimpleRange& range() const { return m_range; }
67 GraphicsLayer& layer() const { return m_graphicsLayer.get(); }
68
69 enum class Type : uint8_t {
70 None = 0,
71 TelephoneNumber = 1 << 0,
72 Selection = 1 << 1,
73 ImageOverlay = 1 << 2,
74 };
75
76 Type type() const { return m_type; }
77
78 void fadeIn();
79 void fadeOut();
80
81 void setHighlight(DDHighlightRef);
82
83private:
84 DataDetectorHighlight(Page&, DataDetectorHighlightClient&, Type, RetainPtr<DDHighlightRef>&&, SimpleRange&&);
85
86 // GraphicsLayerClient
87 void notifyFlushRequired(const GraphicsLayer*) override;
88 void paintContents(const GraphicsLayer*, GraphicsContext&, const FloatRect& inClip, GraphicsLayerPaintBehavior) override;
89 float deviceScaleFactor() const override;
90
91 void didFinishFadeOutAnimation();
92
93 WeakPtr<DataDetectorHighlightClient> m_client;
94 WeakPtr<Page> m_page;
95 RetainPtr<DDHighlightRef> m_highlight;
96 SimpleRange m_range;
97 Ref<GraphicsLayer> m_graphicsLayer;
98 Type m_type { Type::None };
99};
