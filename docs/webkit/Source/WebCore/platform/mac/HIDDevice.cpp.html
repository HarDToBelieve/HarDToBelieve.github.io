42HIDDevice::HIDDevice(IOHIDDeviceRef device)
43 : m_rawDevice(device)
44{
45 CFNumberRef cfVendorID = checked_cf_cast<CFNumberRef>(IOHIDDeviceGetProperty(device, CFSTR(kIOHIDVendorIDKey)));
46 CFNumberRef cfProductID = checked_cf_cast<CFNumberRef>(IOHIDDeviceGetProperty(device, CFSTR(kIOHIDProductIDKey)));
47
48 int vendorID, productID;
49 CFNumberGetValue(cfVendorID, kCFNumberIntType, &vendorID);
50 CFNumberGetValue(cfProductID, kCFNumberIntType, &productID);
51
52 if (vendorID < 0 || vendorID > std::numeric_limits<uint16_t>::max()) {
53 LOG(HID, "Device attached with malformed vendor ID 0x%x. Resetting to 0.", vendorID);
54 vendorID = 0;
55 }
56 if (productID < 0 || productID > std::numeric_limits<uint16_t>::max()) {
57 LOG(HID, "Device attached with malformed product ID 0x%x. Resetting to 0.", productID);
58 productID = 0;
59 }
60
61 m_vendorID = (uint16_t)vendorID;
62 m_productID = (uint16_t)productID;
63
64 CFStringRef cfProductName = checked_cf_cast<CFStringRef>(IOHIDDeviceGetProperty(device, CFSTR(kIOHIDProductKey)));
65 m_productName = cfProductName ? String(cfProductName) : String("Unknown"_s);
66}

68Vector<HIDElement> HIDDevice::uniqueInputElementsInDeviceTreeOrder() const
69{
70 HashSet<IOHIDElementCookie> encounteredCookies;
71 Deque<IOHIDElementRef> elementQueue;
72
73 RetainPtr<CFArrayRef> elements = adoptCF(IOHIDDeviceCopyMatchingElements(m_rawDevice.get(), NULL, kIOHIDOptionsTypeNone));
74 for (CFIndex i = 0; i < CFArrayGetCount(elements.get()); ++i)
75 elementQueue.append(checked_cf_cast<IOHIDElementRef>(CFArrayGetValueAtIndex(elements.get(), i)));
76
77 Vector<HIDElement> result;
78
79 while (!elementQueue.isEmpty()) {
80 auto element = elementQueue.takeFirst();
81 IOHIDElementCookie cookie = IOHIDElementGetCookie(element);
82 if (encounteredCookies.contains(cookie))
83 continue;
84
85 switch (IOHIDElementGetType(element)) {
86 case kIOHIDElementTypeCollection: {
87 auto children = IOHIDElementGetChildren(element);
88 for (CFIndex i = CFArrayGetCount(children) - 1; i >= 0; --i)
89 elementQueue.prepend(checked_cf_cast<IOHIDElementRef>(CFArrayGetValueAtIndex(children, i)));
90 continue;
91 }
92 case kIOHIDElementTypeInput_Misc:
93 case kIOHIDElementTypeInput_Button:
94 case kIOHIDElementTypeInput_Axis:
95 encounteredCookies.add(cookie);
96 result.append(element);
97 continue;
98 default:
99 continue;
100 }
101 }
102
103 return result;
104}
