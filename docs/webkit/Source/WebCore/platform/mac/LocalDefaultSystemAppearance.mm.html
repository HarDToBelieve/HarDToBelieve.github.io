33
34LocalDefaultSystemAppearance::LocalDefaultSystemAppearance(bool useDarkAppearance)
35{
36#if HAVE(OS_DARK_MODE_SUPPORT)
37 ALLOW_DEPRECATED_DECLARATIONS_BEGIN
38 m_savedSystemAppearance = [NSAppearance currentAppearance];
39 ALLOW_DEPRECATED_DECLARATIONS_END
40 m_usingDarkAppearance = useDarkAppearance;
41
42 ALLOW_DEPRECATED_DECLARATIONS_BEGIN
43 [NSAppearance setCurrentAppearance:[NSAppearance appearanceNamed:m_usingDarkAppearance ? NSAppearanceNameDarkAqua : NSAppearanceNameAqua]];
44 ALLOW_DEPRECATED_DECLARATIONS_END
45#else
46 UNUSED_PARAM(useDarkAppearance);
47#endif
48}
49
50LocalDefaultSystemAppearance::~LocalDefaultSystemAppearance()
51{
52#if HAVE(OS_DARK_MODE_SUPPORT)
53 ALLOW_DEPRECATED_DECLARATIONS_BEGIN
54 [NSAppearance setCurrentAppearance:m_savedSystemAppearance.get()];
55 ALLOW_DEPRECATED_DECLARATIONS_END
56#endif
57}
58
59}

61#endif // USE(APPKIT)
