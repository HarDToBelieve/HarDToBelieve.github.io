133void Pasteboard::write(const PasteboardWebContent& content)
134{
135 Vector<String> types;
136
137 if (content.canSmartCopyOrDelete)
138 types.append(WebSmartPastePboardType);
139 if (content.dataInWebArchiveFormat) {
140 types.append(WebArchivePboardType);
141
142ALLOW_DEPRECATED_DECLARATIONS_BEGIN
143 types.append(kUTTypeWebArchive);
144ALLOW_DEPRECATED_DECLARATIONS_END
145 }
146 if (content.dataInRTFDFormat)
147 types.append(String(legacyRTFDPasteboardType()));
148 if (content.dataInRTFFormat)
149 types.append(String(legacyRTFPasteboardType()));
150 if (!content.dataInHTMLFormat.isNull())
151 types.append(String(legacyHTMLPasteboardType()));
152 if (!content.dataInStringFormat.isNull())
153 types.append(String(legacyStringPasteboardType()));
154 types.appendVector(content.clientTypes);
155 types.append(PasteboardCustomData::cocoaType());
156
157 m_changeCount = platformStrategies()->pasteboardStrategy()->setTypes(types, m_pasteboardName, context());
158
159 // FIXME: The following code should be refactored, such that it only requires a single call out to the client layer.
160 // In WebKit2, this currently results in many unnecessary synchronous round-trip IPC messages.
161
162 ASSERT(content.clientTypes.size() == content.clientData.size());
163 for (size_t i = 0, size = content.clientTypes.size(); i < size; ++i)
164 m_changeCount = platformStrategies()->pasteboardStrategy()->setBufferForType(content.clientData[i].get(), content.clientTypes[i], m_pasteboardName, context());
165 if (content.canSmartCopyOrDelete)
166 m_changeCount = platformStrategies()->pasteboardStrategy()->setBufferForType(nullptr, WebSmartPastePboardType, m_pasteboardName, context());
167 if (content.dataInWebArchiveFormat) {
168 m_changeCount = platformStrategies()->pasteboardStrategy()->setBufferForType(content.dataInWebArchiveFormat.get(), WebArchivePboardType, m_pasteboardName, context());
169
170 ALLOW_DEPRECATED_DECLARATIONS_BEGIN
171 m_changeCount = platformStrategies()->pasteboardStrategy()->setBufferForType(content.dataInWebArchiveFormat.get(), kUTTypeWebArchive, m_pasteboardName, context());
172 ALLOW_DEPRECATED_DECLARATIONS_END
173 }
174 if (content.dataInRTFDFormat)
175 m_changeCount = platformStrategies()->pasteboardStrategy()->setBufferForType(content.dataInRTFDFormat.get(), legacyRTFDPasteboardType(), m_pasteboardName, context());
176 if (content.dataInRTFFormat)
177 m_changeCount = platformStrategies()->pasteboardStrategy()->setBufferForType(content.dataInRTFFormat.get(), legacyRTFPasteboardType(), m_pasteboardName, context());
178 if (!content.dataInHTMLFormat.isNull())
179 m_changeCount = platformStrategies()->pasteboardStrategy()->setStringForType(content.dataInHTMLFormat, legacyHTMLPasteboardType(), m_pasteboardName, context());
180 if (!content.dataInStringFormat.isNull())
181 m_changeCount = platformStrategies()->pasteboardStrategy()->setStringForType(content.dataInStringFormat, legacyStringPasteboardType(), m_pasteboardName, context());
182
183 PasteboardCustomData data;
184 data.setOrigin(content.contentOrigin);
185 m_changeCount = platformStrategies()->pasteboardStrategy()->setBufferForType(data.createSharedBuffer().ptr(), PasteboardCustomData::cocoaType(), m_pasteboardName, context());
186
187}

189void Pasteboard::writePlainText(const String& text, SmartReplaceOption smartReplaceOption)
190{
191 Vector<String> types;
192 types.append(legacyStringPasteboardType());
193 if (smartReplaceOption == CanSmartReplace)
194 types.append(WebSmartPastePboardType);
195
196 platformStrategies()->pasteboardStrategy()->setTypes(types, m_pasteboardName, context());
197 m_changeCount = platformStrategies()->pasteboardStrategy()->setStringForType(text, legacyStringPasteboardType(), m_pasteboardName, context());

199 m_changeCount = platformStrategies()->pasteboardStrategy()->setBufferForType(0, WebSmartPastePboardType, m_pasteboardName, context());
200}

202static long writeURLForTypes(const Vector<String>& types, const String& pasteboardName, const PasteboardURL& pasteboardURL, const PasteboardContext* context)
203{
204 auto newChangeCount = platformStrategies()->pasteboardStrategy()->setTypes(types, pasteboardName, context);
205
206 ASSERT(!pasteboardURL.url.isEmpty());
207
208 NSURL *cocoaURL = pasteboardURL.url;
209 NSString *userVisibleString = pasteboardURL.userVisibleForm;
210 NSString *title = (NSString *)pasteboardURL.title;
211 if (![title length]) {
212 title = [[cocoaURL path] lastPathComponent];
213 if (![title length])
214 title = userVisibleString;
215 }
216
217 if (types.contains(WebURLsWithTitlesPboardType)) {
218 PasteboardURL url = { pasteboardURL.url, String(title).stripWhiteSpace(), emptyString() };
219 newChangeCount = platformStrategies()->pasteboardStrategy()->setURL(url, pasteboardName, context);
220 }
221 if (types.contains(String(legacyURLPasteboardType())))
222 newChangeCount = platformStrategies()->pasteboardStrategy()->setStringForType([cocoaURL absoluteString], legacyURLPasteboardType(), pasteboardName, context);
223 if (types.contains(WebURLPboardType))
224 newChangeCount = platformStrategies()->pasteboardStrategy()->setStringForType(userVisibleString, WebURLPboardType, pasteboardName, context);
225 if (types.contains(WebURLNamePboardType))
226 newChangeCount = platformStrategies()->pasteboardStrategy()->setStringForType(title, WebURLNamePboardType, pasteboardName, context);
227 if (types.contains(String(legacyStringPasteboardType())))
228 newChangeCount = platformStrategies()->pasteboardStrategy()->setStringForType(userVisibleString, legacyStringPasteboardType(), pasteboardName, context);
229
230 return newChangeCount;
231}

251static NSFileWrapper* fileWrapper(const PasteboardImage& pasteboardImage)
252{
253 auto wrapper = adoptNS([[NSFileWrapper alloc] initRegularFileWithContents:pasteboardImage.resourceData->createNSData().get()]);
254 [wrapper setPreferredFilename:suggestedFilenameWithMIMEType(pasteboardImage.url.url, pasteboardImage.resourceMIMEType)];
255 return wrapper.autorelease();
256}

258static void writeFileWrapperAsRTFDAttachment(NSFileWrapper *wrapper, const String& pasteboardName, int64_t& newChangeCount, const PasteboardContext* context)
259{
260 NSAttributedString *string = [NSAttributedString attributedStringWithAttachment:adoptNS([[NSTextAttachment alloc] initWithFileWrapper:wrapper]).get()];
261
262 NSData *RTFDData = [string RTFDFromRange:NSMakeRange(0, [string length]) documentAttributes:@{ }];
263 if (!RTFDData)
264 return;
265
266 newChangeCount = platformStrategies()->pasteboardStrategy()->setBufferForType(SharedBuffer::create(RTFDData).ptr(), legacyRTFDPasteboardType(), pasteboardName, context);
267}

269void Pasteboard::write(const PasteboardImage& pasteboardImage)
270{
271 CFDataRef imageData = pasteboardImage.image->tiffRepresentation();
272 if (!imageData)
273 return;
274
275 // FIXME: Why can we assert this? It doesn't seem like it's guaranteed.
276 ASSERT(MIMETypeRegistry::isSupportedImageMIMEType(pasteboardImage.resourceMIMEType));
277
278 auto types = writableTypesForImage();
279 if (pasteboardImage.dataInWebArchiveFormat) {
280 types.append(WebArchivePboardType);
281
282ALLOW_DEPRECATED_DECLARATIONS_BEGIN
283 types.append(kUTTypeWebArchive);
284ALLOW_DEPRECATED_DECLARATIONS_END
285 }
286
287 m_changeCount = writeURLForTypes(types, m_pasteboardName, pasteboardImage.url, context());
288 m_changeCount = platformStrategies()->pasteboardStrategy()->setBufferForType(SharedBuffer::create(imageData).ptr(), legacyTIFFPasteboardType(), m_pasteboardName, context());
289 if (auto archiveData = pasteboardImage.dataInWebArchiveFormat) {
290 m_changeCount = platformStrategies()->pasteboardStrategy()->setBufferForType(archiveData.get(), WebArchivePboardType, m_pasteboardName, context());
291
292 ALLOW_DEPRECATED_DECLARATIONS_BEGIN
293 m_changeCount = platformStrategies()->pasteboardStrategy()->setBufferForType(archiveData.get(), kUTTypeWebArchive, m_pasteboardName, context());
294 ALLOW_DEPRECATED_DECLARATIONS_END
295 }
296 if (!pasteboardImage.dataInHTMLFormat.isEmpty())
297 m_changeCount = platformStrategies()->pasteboardStrategy()->setStringForType(pasteboardImage.dataInHTMLFormat, legacyHTMLPasteboardType(), m_pasteboardName, context());
298 writeFileWrapperAsRTFDAttachment(fileWrapper(pasteboardImage), m_pasteboardName, m_changeCount, context());
299}

312// FIXME: This should be a general utility function for Vectors of Strings (or things that can be
313// converted to Strings). It could also be faster by computing the total length and reserving that
314// capacity in the StringBuilder.
315static String joinPathnames(const Vector<String>& pathnames)
316{
317 StringBuilder builder;
318 for (auto& path : pathnames) {
319 if (!builder.isEmpty())
320 builder.append('\n');
321 builder.append(path);
322 }
323 return builder.toString();
324}

340void Pasteboard::read(PasteboardPlainText& text, PlainTextURLReadingPolicy allowURL, std::optional<size_t> itemIndex)
341{
342 auto& strategy = *platformStrategies()->pasteboardStrategy();
343
344 Vector<String> types;
345 if (itemIndex) {
346 if (auto itemInfo = strategy.informationForItemAtIndex(*itemIndex, m_pasteboardName, m_changeCount, context()))
347 types = itemInfo->platformTypesByFidelity;
348 } else
349 strategy.getTypes(types, m_pasteboardName, context());
350
351 if (types.contains(String(NSPasteboardTypeString))) {
352 text.text = readStringAtPreferredItemIndex(NSPasteboardTypeString, itemIndex, strategy, m_pasteboardName, context());
353 text.isURL = false;
354 return;
355 }
356
357 if (types.contains(String(legacyStringPasteboardType()))) {
358 text.text = readStringAtPreferredItemIndex(legacyStringPasteboardType(), itemIndex, strategy, m_pasteboardName, context());
359 text.isURL = false;
360 return;
361 }
362
363 if (types.contains(String(legacyRTFDPasteboardType()))) {
364 if (auto data = readBufferAtPreferredItemIndex(legacyRTFDPasteboardType(), itemIndex, strategy, m_pasteboardName, context())) {
365 if (auto attributedString = adoptNS([[NSAttributedString alloc] initWithRTFD:data->createNSData().get() documentAttributes:nil])) {
366 text.text = [attributedString string];
367 text.isURL = false;
368 return;
369 }
370 }
371 }
372
373 if (types.contains(String(NSPasteboardTypeRTFD))) {
374 if (auto data = readBufferAtPreferredItemIndex(NSPasteboardTypeRTFD, itemIndex, strategy, m_pasteboardName, context())) {
375 if (auto attributedString = adoptNS([[NSAttributedString alloc] initWithRTFD:data->createNSData().get() documentAttributes:nil])) {
376 text.text = [attributedString string];
377 text.isURL = false;
378 return;
379 }
380 }
381 }
382
383 if (types.contains(String(legacyRTFPasteboardType()))) {
384 if (auto data = readBufferAtPreferredItemIndex(legacyRTFPasteboardType(), itemIndex, strategy, m_pasteboardName, context())) {
385 if (auto attributedString = adoptNS([[NSAttributedString alloc] initWithRTF:data->createNSData().get() documentAttributes:nil])) {
386 text.text = [attributedString string];
387 text.isURL = false;
388 return;
389 }
390 }
391 }
392
393 if (types.contains(String(NSPasteboardTypeRTF))) {
394 if (auto data = readBufferAtPreferredItemIndex(NSPasteboardTypeRTF, itemIndex, strategy, m_pasteboardName, context())) {
395 if (auto attributedString = adoptNS([[NSAttributedString alloc] initWithRTF:data->createNSData().get() documentAttributes:nil])) {
396 text.text = [attributedString string];
397 text.isURL = false;
398 return;
399 }
400 }
401 }
402
403 if (types.contains(String(legacyFilesPromisePasteboardType()))) {
404 text.text = joinPathnames(m_promisedFilePaths);
405 text.isURL = false;
406 return;
407 }
408
409 if (types.contains(String(legacyFilenamesPasteboardType()))) {
410 Vector<String> pathnames;
411 strategy.getPathnamesForType(pathnames, legacyFilenamesPasteboardType(), m_pasteboardName, context());
412 text.text = joinPathnames(pathnames);
413 text.isURL = false;
414 return;
415 }
416
417 // FIXME: The code above looks at the types vector first, but this just gets the string without checking. Why the difference?
418 if (allowURL == PlainTextURLReadingPolicy::AllowURL) {
419 text.text = readStringAtPreferredItemIndex(legacyURLPasteboardType(), itemIndex, strategy, m_pasteboardName, context());
420 text.isURL = !text.text.isNull();
421 }
422}

424void Pasteboard::read(PasteboardWebContentReader& reader, WebContentReadingPolicy policy, std::optional<size_t> itemIndex)
425{
426 auto& strategy = *platformStrategies()->pasteboardStrategy();
427
428 Vector<String> types;
429 if (itemIndex) {
430 if (auto itemInfo = strategy.informationForItemAtIndex(*itemIndex, m_pasteboardName, m_changeCount, context()))
431 types = itemInfo->platformTypesByFidelity;
432 } else
433 strategy.getTypes(types, m_pasteboardName, context());
434
435 reader.contentOrigin = readOrigin();
436
437 if (types.contains(WebArchivePboardType)) {
438 if (auto buffer = readBufferAtPreferredItemIndex(WebArchivePboardType, itemIndex, strategy, m_pasteboardName, context())) {
439 if (m_changeCount != changeCount() || reader.readWebArchive(*buffer))
440 return;
441 }
442 }
443
444ALLOW_DEPRECATED_DECLARATIONS_BEGIN
445 if (types.contains(String(kUTTypeWebArchive))) {
446 if (auto buffer = readBufferAtPreferredItemIndex(kUTTypeWebArchive, itemIndex, strategy, m_pasteboardName, context())) {
447 if (m_changeCount != changeCount() || reader.readWebArchive(*buffer))
448 return;
449 }
450 }
451ALLOW_DEPRECATED_DECLARATIONS_END
452
453 if (policy == WebContentReadingPolicy::AnyType && types.contains(String(legacyFilesPromisePasteboardType()))) {
454 if (m_changeCount != changeCount() || reader.readFilePaths(m_promisedFilePaths))
455 return;
456 }
457
458 if (policy == WebContentReadingPolicy::AnyType && types.contains(String(legacyFilenamesPasteboardType()))) {
459 Vector<String> paths;
460 strategy.getPathnamesForType(paths, legacyFilenamesPasteboardType(), m_pasteboardName, context());
461 if (m_changeCount != changeCount() || reader.readFilePaths(paths))
462 return;
463 }
464
465 if (types.contains(String(legacyHTMLPasteboardType()))) {
466 String string = readStringAtPreferredItemIndex(legacyHTMLPasteboardType(), itemIndex, strategy, m_pasteboardName, context());
467 if (m_changeCount != changeCount() || (!string.isNull() && reader.readHTML(string)))
468 return;
469 }
470
471 if (types.contains(String(NSPasteboardTypeHTML))) {
472 String string = readStringAtPreferredItemIndex(NSPasteboardTypeHTML, itemIndex, strategy, m_pasteboardName, context());
473 if (m_changeCount != changeCount() || (!string.isNull() && reader.readHTML(string)))
474 return;
475 }
476
477 if (types.contains(String(legacyRTFDPasteboardType()))) {
478 if (auto buffer = readBufferAtPreferredItemIndex(legacyRTFDPasteboardType(), itemIndex, strategy, m_pasteboardName, context())) {
479 if (m_changeCount != changeCount() || reader.readRTFD(*buffer))
480 return;
481 }
482 }
483
484 if (types.contains(String(NSPasteboardTypeRTFD))) {
485 if (auto buffer = readBufferAtPreferredItemIndex(NSPasteboardTypeRTFD, itemIndex, strategy, m_pasteboardName, context())) {
486 if (m_changeCount != changeCount() || reader.readRTFD(*buffer))
487 return;
488 }
489 }
490
491 if (types.contains(String(legacyRTFPasteboardType()))) {
492 if (auto buffer = readBufferAtPreferredItemIndex(legacyRTFPasteboardType(), itemIndex, strategy, m_pasteboardName, context())) {
493 if (m_changeCount != changeCount() || reader.readRTF(*buffer))
494 return;
495 }
496 }
497
498 if (types.contains(String(NSPasteboardTypeRTF))) {
499 if (auto buffer = readBufferAtPreferredItemIndex(NSPasteboardTypeRTF, itemIndex, strategy, m_pasteboardName, context())) {
500 if (m_changeCount != changeCount() || reader.readRTF(*buffer))
501 return;
502 }
503 }
504
505 if (policy == WebContentReadingPolicy::OnlyRichTextTypes)
506 return;
507
508 if (types.contains(String(legacyTIFFPasteboardType()))) {
509 if (auto buffer = readBufferAtPreferredItemIndex(legacyTIFFPasteboardType(), itemIndex, strategy, m_pasteboardName, context())) {
510 if (m_changeCount != changeCount() || reader.readImage(buffer.releaseNonNull(), "image/tiff"_s))
511 return;
512 }
513 }
514
515 if (types.contains(String(NSPasteboardTypeTIFF))) {
516 if (auto buffer = readBufferAtPreferredItemIndex(NSPasteboardTypeTIFF, itemIndex, strategy, m_pasteboardName, context())) {
517 if (m_changeCount != changeCount() || reader.readImage(buffer.releaseNonNull(), "image/tiff"_s))
518 return;
519 }
520 }
521
522 if (types.contains(String(legacyPDFPasteboardType()))) {
523 if (auto buffer = readBufferAtPreferredItemIndex(legacyPDFPasteboardType(), itemIndex, strategy, m_pasteboardName, context())) {
524 if (m_changeCount != changeCount() || reader.readImage(buffer.releaseNonNull(), "application/pdf"_s))
525 return;
526 }
527 }
528
529 if (types.contains(String(NSPasteboardTypePDF))) {
530 if (auto buffer = readBufferAtPreferredItemIndex(NSPasteboardTypePDF, itemIndex, strategy, m_pasteboardName, context())) {
531 if (m_changeCount != changeCount() || reader.readImage(buffer.releaseNonNull(), "application/pdf"_s))
532 return;
533 }
534 }
535
536ALLOW_DEPRECATED_DECLARATIONS_BEGIN
537 if (types.contains(String(kUTTypePNG))) {
538 if (auto buffer = readBufferAtPreferredItemIndex(kUTTypePNG, itemIndex, strategy, m_pasteboardName, context())) {
539 if (m_changeCount != changeCount() || reader.readImage(buffer.releaseNonNull(), "image/png"_s))
540 return;
541 }
542 }
543ALLOW_DEPRECATED_DECLARATIONS_END
544
545ALLOW_DEPRECATED_DECLARATIONS_BEGIN
546 if (types.contains(String(kUTTypeJPEG))) {
547 if (auto buffer = readBufferAtPreferredItemIndex(kUTTypeJPEG, itemIndex, strategy, m_pasteboardName, context())) {
548 if (m_changeCount != changeCount() || reader.readImage(buffer.releaseNonNull(), "image/jpeg"_s))
549 return;
550 }
551 }
552ALLOW_DEPRECATED_DECLARATIONS_END
553
554 if (types.contains(String(legacyURLPasteboardType()))) {
555 URL url = strategy.url(m_pasteboardName, context());
556 String title = readStringAtPreferredItemIndex(WebURLNamePboardType, itemIndex, strategy, m_pasteboardName, context());
557 if (m_changeCount != changeCount() || (!url.isNull() && reader.readURL(url, title)))
558 return;
559 }
560
561 if (types.contains(String(legacyStringPasteboardType()))) {
562 String string = readStringAtPreferredItemIndex(legacyStringPasteboardType(), itemIndex, strategy, m_pasteboardName, context());

614Vector<String> Pasteboard::readPlatformValuesAsStrings(const String& domType, int64_t changeCount, const String& pasteboardName)
615{
616 auto& strategy = *platformStrategies()->pasteboardStrategy();
617 auto cocoaType = cocoaTypeFromHTMLClipboardType(domType);
618 if (cocoaType.isEmpty())
619 return { };
620
621 auto values = strategy.allStringsForType(cocoaType, pasteboardName, context());
622 if (cocoaType == String(legacyStringPasteboardType())) {
623 values = values.map([&] (auto& value) -> String {
624 return [value precomposedStringWithCanonicalMapping];
625 });
626 }
627
628 // Enforce changeCount ourselves for security. We check after reading instead of before to be
629 // sure it doesn't change between our testing the change count and accessing the data.
630 if (changeCount != platformStrategies()->pasteboardStrategy()->changeCount(pasteboardName, context()))
631 return { };
632
633 return values;
634}

647void Pasteboard::addHTMLClipboardTypesForCocoaType(ListHashSet<String>& resultTypes, const String& cocoaType)
648{
649 if (cocoaType == "NeXT plain ascii pasteboard type")
650 return; // Skip this ancient type that gets auto-supplied by some system conversion.
651
652 // UTI may not do these right, so make sure we get the right, predictable result
653 if (cocoaType == String(legacyStringPasteboardType()) || cocoaType == String(NSPasteboardTypeString)) {
654 resultTypes.add("text/plain"_s);
655 return;
656 }
657 if (cocoaType == String(legacyURLPasteboardType())) {
658 resultTypes.add("text/uri-list"_s);
659 return;
660 }
661 if (cocoaType == String(legacyFilenamesPasteboardType()) || Pasteboard::shouldTreatCocoaTypeAsFile(cocoaType))
662 return;
663 String utiType = utiTypeFromCocoaType(cocoaType);
664 if (!utiType.isEmpty()) {
665 resultTypes.add(utiType);
666 return;
667 }
668 // No mapping, just pass the whole string through.
669 resultTypes.add(cocoaType);
670}

721static void flipImageSpec(CoreDragImageSpec* imageSpec)
722{
723 unsigned char* tempRow = (unsigned char*)fastMalloc(imageSpec->bytesPerRow);
724 int planes = imageSpec->isPlanar ? imageSpec->samplesPerPixel : 1;
725
726 for (int p = 0; p < planes; ++p) {
727 unsigned char* topRow = const_cast<unsigned char*>(imageSpec->data[p]);
728 unsigned char* botRow = topRow + (imageSpec->pixelsHigh - 1) * imageSpec->bytesPerRow;
729 for (int i = 0; i < imageSpec->pixelsHigh / 2; ++i, topRow += imageSpec->bytesPerRow, botRow -= imageSpec->bytesPerRow) {
730 bcopy(topRow, tempRow, imageSpec->bytesPerRow);
731 bcopy(botRow, topRow, imageSpec->bytesPerRow);
732 bcopy(tempRow, botRow, imageSpec->bytesPerRow);
733 }
734 }
735
736 fastFree(tempRow);
737}

739static void setDragImageImpl(NSImage *image, NSPoint offset)
740{
741 bool flipImage;
742 NSSize imageSize = image.size;
743 CGRect imageRect = CGRectMake(0, 0, imageSize.width, imageSize.height);
744 NSImageRep *imageRep = [image bestRepresentationForRect:NSRectFromCGRect(imageRect) context:nil hints:nil];
745 RetainPtr<NSBitmapImageRep> bitmapImage;
746 if (!imageRep || ![imageRep isKindOfClass:[NSBitmapImageRep class]] || !NSEqualSizes(imageRep.size, imageSize)) {
747 [image lockFocus];
748 ALLOW_DEPRECATED_DECLARATIONS_BEGIN
749 bitmapImage = adoptNS([[NSBitmapImageRep alloc] initWithFocusedViewRect:*(NSRect*)&imageRect]);
750 ALLOW_DEPRECATED_DECLARATIONS_END
751 [image unlockFocus];
752
753 // we may have to flip the bits we just read if the image was flipped since it means the cache was also
754 // and CoreDragSetImage can't take a transform for rendering.
755 ALLOW_DEPRECATED_DECLARATIONS_BEGIN
756 flipImage = image.isFlipped;
757 ALLOW_DEPRECATED_DECLARATIONS_END
758 } else {
759 flipImage = false;
760 bitmapImage = (NSBitmapImageRep *)imageRep;
761 }
762 ASSERT(bitmapImage);
763
764 CoreDragImageSpec imageSpec;
765 imageSpec.version = kCoreDragImageSpecVersionOne;
766 imageSpec.pixelsWide = [bitmapImage pixelsWide];
767 imageSpec.pixelsHigh = [bitmapImage pixelsHigh];
768 imageSpec.bitsPerSample = [bitmapImage bitsPerSample];
769 imageSpec.samplesPerPixel = [bitmapImage samplesPerPixel];
770 imageSpec.bitsPerPixel = [bitmapImage bitsPerPixel];
771 imageSpec.bytesPerRow = [bitmapImage bytesPerRow];
772 imageSpec.isPlanar = [bitmapImage isPlanar];
773 imageSpec.hasAlpha = [bitmapImage hasAlpha];
774 [bitmapImage getBitmapDataPlanes:const_cast<unsigned char**>(imageSpec.data)];
775
776 // if image was flipped, we have an upside down bitmap since the cache is rendered flipped
777 if (flipImage)
778 flipImageSpec(&imageSpec);
779
780 CGSRegionObj imageShape;
781 OSStatus error = CGSNewRegionWithRect(&imageRect, &imageShape);
782 ASSERT(error == kCGErrorSuccess);
783 if (error != kCGErrorSuccess)
784 return;
785
786 // make sure image has integer offset
787 CGPoint imageOffset = { -offset.x, -(imageSize.height - offset.y) };
788 imageOffset.x = floor(imageOffset.x + 0.5);
789 imageOffset.y = floor(imageOffset.y + 0.5);
790
791 error = CoreDragSetImage(CoreDragGetCurrentDrag(), imageOffset, &imageSpec, imageShape, 1.0);
792 CGSReleaseRegion(imageShape);
793 ASSERT(error == kCGErrorSuccess);
794}

796void Pasteboard::setDragImage(DragImage image, const IntPoint& location)
797{
798 // Don't allow setting the drag image if someone kept a pasteboard and is trying to set the image too late.
799 if (m_changeCount != platformStrategies()->pasteboardStrategy()->changeCount(m_pasteboardName, context()))
800 return;
801
802 // Dashboard wants to be able to set the drag image during dragging, but Cocoa does not allow this.
803 // Instead we must drop down to the CoreGraphics API.
804 setDragImageImpl(image.get().get(), location);
805
806 // Hack: We must post an event to wake up the NSDragManager, which is sitting in a nextEvent call
807 // up the stack from us because the CoreFoundation drag manager does not use the run loop by itself.
808 // This is the most innocuous event to use, per Kristin Forster.
809 // This is only relevant in WK1. Do not execute in the WebContent process, since it is now using
810 // NSRunLoop, and not the NSApplication run loop.
811 if ([NSApp isRunning]) {
812 ASSERT(hasProcessPrivilege(ProcessPrivilege::CanCommunicateWithWindowServer));
813 NSEvent* event = [NSEvent mouseEventWithType:NSEventTypeMouseMoved location:NSZeroPoint
814 modifierFlags:0 timestamp:0 windowNumber:0 context:nil eventNumber:0 clickCount:0 pressure:0];
815 [NSApp postEvent:event atStart:YES];
816 }
