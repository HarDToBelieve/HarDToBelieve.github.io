39static RetainPtr<NSString> toUTI(NSString *pasteboardType)
40{
41ALLOW_DEPRECATED_DECLARATIONS_BEGIN
42 return adoptNS((__bridge NSString *)UTTypeCreatePreferredIdentifierForTag(kUTTagClassNSPboardType, (__bridge CFStringRef)pasteboardType, nullptr));
43ALLOW_DEPRECATED_DECLARATIONS_END
44}
45
46static RetainPtr<NSString> toUTIUnlessAlreadyUTI(NSString *type)
47{
48ALLOW_DEPRECATED_DECLARATIONS_BEGIN
49 if (UTTypeIsDeclared((__bridge CFStringRef)type) || UTTypeIsDynamic((__bridge CFStringRef)type)) {
50 // This is already a UTI.
51 return type;
52 }
53ALLOW_DEPRECATED_DECLARATIONS_END
54
55 return toUTI(type);
56}

58RetainPtr<id <NSPasteboardWriting>> createPasteboardWriter(const PasteboardWriterData& data)
59{
60 auto pasteboardItem = adoptNS([[NSPasteboardItem alloc] init]);
61
62 if (auto& plainText = data.plainText()) {
63 [pasteboardItem setString:plainText->text forType:NSPasteboardTypeString];
64 if (plainText->canSmartCopyOrDelete) {
65ALLOW_DEPRECATED_DECLARATIONS_BEGIN
66 auto smartPasteType = adoptNS((__bridge NSString *)UTTypeCreatePreferredIdentifierForTag(kUTTagClassNSPboardType, (__bridge CFStringRef)_NXSmartPaste, nullptr));
67ALLOW_DEPRECATED_DECLARATIONS_END
68 [pasteboardItem setData:[NSData data] forType:smartPasteType.get()];
69 }
70 }
71
72 if (auto& urlData = data.urlData()) {
73 NSURL *cocoaURL = urlData->url;
74 NSString *userVisibleString = urlData->userVisibleForm;
75 NSString *title = (NSString *)urlData->title;
76 if (!title.length) {
77 title = cocoaURL.path.lastPathComponent;
78 if (!title.length)
79 title = userVisibleString;
80 }
81
82 // WebURLsWithTitlesPboardType.
83 auto paths = adoptNS([[NSArray alloc] initWithObjects:@[ @[ cocoaURL.absoluteString ] ], @[ urlData->title.stripWhiteSpace() ], nil]);
84 [pasteboardItem setPropertyList:paths.get() forType:toUTI(@"WebURLsWithTitlesPboardType").get()];
85
86 // NSURLPboardType.
87 if (NSURL *baseCocoaURL = cocoaURL.baseURL)
88 [pasteboardItem setPropertyList:@[ cocoaURL.relativeString, baseCocoaURL.absoluteString ] forType:toUTI(WebCore::legacyURLPasteboardType()).get()];
89 else if (cocoaURL)
90 [pasteboardItem setPropertyList:@[ cocoaURL.absoluteString, @"" ] forType:toUTI(WebCore::legacyURLPasteboardType()).get()];
91 else
92 [pasteboardItem setPropertyList:@[ @"", @"" ] forType:toUTI(WebCore::legacyURLPasteboardType()).get()];
93
94ALLOW_DEPRECATED_DECLARATIONS_BEGIN
95 if (cocoaURL.fileURL)
96 [pasteboardItem setString:cocoaURL.absoluteString forType:(NSString *)kUTTypeFileURL];
97 [pasteboardItem setString:userVisibleString forType:(NSString *)kUTTypeURL];
98ALLOW_DEPRECATED_DECLARATIONS_END
99
100 // WebURLNamePboardType.
101 [pasteboardItem setString:title forType:@"public.url-name"];
102
103 // NSPasteboardTypeString.
104 [pasteboardItem setString:userVisibleString forType:NSPasteboardTypeString];
105 }
106
107 if (auto& webContent = data.webContent()) {
108 if (webContent->canSmartCopyOrDelete) {
109ALLOW_DEPRECATED_DECLARATIONS_BEGIN
110 auto smartPasteType = adoptNS((__bridge NSString *)UTTypeCreatePreferredIdentifierForTag(kUTTagClassNSPboardType, (__bridge CFStringRef)_NXSmartPaste, nullptr));
111ALLOW_DEPRECATED_DECLARATIONS_END
112 [pasteboardItem setData:[NSData data] forType:smartPasteType.get()];
113 }
114 if (webContent->dataInWebArchiveFormat) {
115ALLOW_DEPRECATED_DECLARATIONS_BEGIN
116 auto webArchiveType = adoptNS((__bridge NSString *)UTTypeCreatePreferredIdentifierForTag(kUTTagClassNSPboardType, (__bridge CFStringRef)@"Apple Web Archive pasteboard type", nullptr));
117ALLOW_DEPRECATED_DECLARATIONS_END
118 [pasteboardItem setData:webContent->dataInWebArchiveFormat->createNSData().get() forType:webArchiveType.get()];
119 }
120 if (webContent->dataInRTFDFormat)
121 [pasteboardItem setData:webContent->dataInRTFDFormat->createNSData().get() forType:NSPasteboardTypeRTFD];
122 if (webContent->dataInRTFFormat)
123 [pasteboardItem setData:webContent->dataInRTFFormat->createNSData().get() forType:NSPasteboardTypeRTF];
124 if (!webContent->dataInHTMLFormat.isNull())
125 [pasteboardItem setString:webContent->dataInHTMLFormat forType:NSPasteboardTypeHTML];
126 if (!webContent->dataInStringFormat.isNull())
127 [pasteboardItem setString:webContent->dataInStringFormat forType:NSPasteboardTypeString];
128
129 for (unsigned i = 0; i < webContent->clientTypes.size(); ++i)
130 [pasteboardItem setData:webContent->clientData[i]->createNSData().get() forType:toUTIUnlessAlreadyUTI(webContent->clientTypes[i]).get()];
131
132 PasteboardCustomData customData;
133 customData.setOrigin(webContent->contentOrigin);
134 [pasteboardItem setData:customData.createSharedBuffer()->createNSData().get() forType:toUTIUnlessAlreadyUTI(String(PasteboardCustomData::cocoaType())).get()];
135 }
136
137 return pasteboardItem;
138}
