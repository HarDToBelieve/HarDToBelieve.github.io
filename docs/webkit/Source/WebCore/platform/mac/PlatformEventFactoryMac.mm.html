218String keyForKeyEvent(NSEvent *event)
219{
220 switch ([event keyCode]) {
221 case kVK_RightCommand:
222 case kVK_Command:
223 return "Meta"_s;
224 case kVK_Shift:
225 case kVK_RightShift:
226 return "Shift"_s;
227 case kVK_CapsLock:
228 return "CapsLock"_s;
229 case kVK_Option: // Left Alt.
230 case kVK_RightOption: // Right Alt.
231 return "Alt"_s;
232 case kVK_Control:
233 case kVK_RightControl:
234 return "Control"_s;
235 }
236
237 // If the event is an NSEventTypeFlagsChanged events and we have not returned yet then this means we could not
238 // identify the modifier key. We return now and report the key as "Unidentified".
239 // Note that [event characters] below raises an exception if called on an NSEventTypeFlagsChanged event.
240 if ([event type] == NSEventTypeFlagsChanged)
241 return "Unidentified"_s;
242
243 // If more than one key is being pressed and the key combination includes one or more modifier keys
244 // that result in the key no longer producing a printable character (e.g., Control + a), then the
245 // key value should be the printable key value that would have been produced if the key had been
246 // typed with the default keyboard layout with no modifier keys except for Shift and AltGr applied.
247 // See <https://www.w3.org/TR/2015/WD-uievents-20151215/#keys-guidelines>.
248 bool isControlDown = ([event modifierFlags] & NSEventModifierFlagControl);
249 NSString *s = isControlDown ? [event charactersIgnoringModifiers] : [event characters];
250 auto length = [s length];
251 // characters / charactersIgnoringModifiers return an empty string for dead keys.
252 // https://developer.apple.com/reference/appkit/nsevent/1534183-characters
253 if (!length)
254 return "Dead"_s;
255 // High unicode codepoints are coded with a character sequence in Mac OS X.
256 if (length > 1)
257 return s;
258 return keyForCharCode([s characterAtIndex:0]);
259}

473String keyIdentifierForKeyEvent(NSEvent* event)
474{
475 if ([event type] == NSEventTypeFlagsChanged) {
476 switch ([event keyCode]) {
477 case 54: // Right Command
478 case 55: // Left Command
479 return "Meta"_str;
480
481 case 57: // Capslock
482 return "CapsLock"_str;
483
484 case 56: // Left Shift
485 case 60: // Right Shift
486 return "Shift"_str;
487
488 case 58: // Left Alt
489 case 61: // Right Alt
490 return "Alt"_str;
491
492 case 59: // Left Ctrl
493 case 62: // Right Ctrl
494 return "Control"_str;
495
496 default:
497 ASSERT_NOT_REACHED();
498 return emptyString();
499 }
500 }
501
502 NSString *s = [event charactersIgnoringModifiers];
503 if ([s length] != 1) {
504 LOG(Events, "received an unexpected number of characters in key event: %u", [s length]);
505 return "Unidentified";
506 }
507 return keyIdentifierForCharCode([s characterAtIndex:0]);
508}

510static bool isKeypadEvent(NSEvent* event)
511{
512 // Check that this is the type of event that has a keyCode.
513 switch ([event type]) {
514 case NSEventTypeKeyDown:
515 case NSEventTypeKeyUp:
516 case NSEventTypeFlagsChanged:
517 break;
518 default:
519 return false;
520 }
521
522 if ([event modifierFlags] & NSEventModifierFlagNumericPad)
523 return true;
524
525 switch ([event keyCode]) {
526 case 71: // Clear
527 case 81: // =
528 case 75: // /
529 case 67: // *
530 case 78: // -
531 case 69: // +
532 case 76: // Enter
533 case 65: // .
534 case 82: // 0
535 case 83: // 1
536 case 84: // 2
537 case 85: // 3
538 case 86: // 4
539 case 87: // 5
540 case 88: // 6
541 case 89: // 7
542 case 91: // 8
543 case 92: // 9
544 return true;
545 }
546
547 return false;
548}

550int windowsKeyCodeForKeyEvent(NSEvent* event)
551{
552 int code = 0;
553 // There are several kinds of characters for which we produce key code from char code:
554 // 1. Roman letters. Windows keyboard layouts affect both virtual key codes and character codes for these,
555 // so e.g. 'A' gets the same keyCode on QWERTY, AZERTY or Dvorak layouts.
556 // 2. Keys for which there is no known Mac virtual key codes, like PrintScreen.
557 // 3. Certain punctuation keys. On Windows, these are also remapped depending on current keyboard layout,
558 // but see comment in windowsKeyCodeForCharCode().
559 if (!isKeypadEvent(event) && ([event type] == NSEventTypeKeyDown || [event type] == NSEventTypeKeyUp)) {
560 // Cmd switches Roman letters for Dvorak-QWERTY layout, so try modified characters first.
561 NSString* s = [event characters];
562 code = [s length] > 0 ? windowsKeyCodeForCharCode([s characterAtIndex:0]) : 0;
563 if (code)
564 return code;
565
566 // Ctrl+A on an AZERTY keyboard would get VK_Q keyCode if we relied on -[NSEvent keyCode] below.
567 s = [event charactersIgnoringModifiers];
568 code = [s length] > 0 ? windowsKeyCodeForCharCode([s characterAtIndex:0]) : 0;
569 if (code)
570 return code;
571 }
572
573 // Map Mac virtual key code directly to Windows one for any keys not handled above.
574 // E.g. the key next to Caps Lock has the same Event.keyCode on U.S. keyboard ('A') and on Russian keyboard (CYRILLIC LETTER EF).
575 return windowsKeyCodeForKeyCode([event keyCode]);
576}

580static void updateSystemStartupTimeIntervalSince1970()
581{
582 // CFAbsoluteTimeGetCurrent() provides the absolute time in seconds since 2001.
583 // mach_absolute_time() provides a relative system time since startup minus the time the computer was suspended.
584 mach_timebase_info_data_t timebase_info;
585 mach_timebase_info(&timebase_info);
586 double elapsedTimeSinceStartup = static_cast<double>(mach_absolute_time()) * timebase_info.numer / timebase_info.denom / 1e9;
587 systemStartupTime = kCFAbsoluteTimeIntervalSince1970 + CFAbsoluteTimeGetCurrent() - elapsedTimeSinceStartup;
588}
589
590static CFTimeInterval cachedStartupTimeIntervalSince1970()
591{
592 static dispatch_once_t once;
593 dispatch_once(&once, ^{
594 void (^updateBlock)(NSNotification *) = Block_copy(^(NSNotification *){ updateSystemStartupTimeIntervalSince1970(); });
595 [[[NSWorkspace sharedWorkspace] notificationCenter] addObserverForName:NSWorkspaceDidWakeNotification
596 object:nil
597 queue:nil
598 usingBlock:updateBlock];
599 [[NSNotificationCenter defaultCenter] addObserverForName:NSSystemClockDidChangeNotification
600 object:nil
601 queue:nil
602 usingBlock:updateBlock];
603 Block_release(updateBlock);
604
605 updateSystemStartupTimeIntervalSince1970();
606 });
607 return systemStartupTime;
608}

615static inline bool isKeyUpEvent(NSEvent *event)
616{
617 if ([event type] != NSEventTypeFlagsChanged)
618 return [event type] == NSEventTypeKeyUp;
619 // FIXME: This logic fails if the user presses both Shift keys at once, for example:
620 // we treat releasing one of them as keyDown.
621 switch ([event keyCode]) {
622 case 54: // Right Command
623 case 55: // Left Command
624 return !([event modifierFlags] & NSEventModifierFlagCommand);
625
626 case 57: // Capslock
627 return !([event modifierFlags] & NSEventModifierFlagCapsLock);
628
629 case 56: // Left Shift
630 case 60: // Right Shift
631 return !([event modifierFlags] & NSEventModifierFlagShift);
632
633 case 58: // Left Alt
634 case 61: // Right Alt
635 return !([event modifierFlags] & NSEventModifierFlagOption);
636
637 case 59: // Left Ctrl
638 case 62: // Right Ctrl
639 return !([event modifierFlags] & NSEventModifierFlagControl);
640
641 case 63: // Function
642 return !([event modifierFlags] & NSEventModifierFlagFunction);
643 }
644 return false;
645}

647OptionSet<PlatformEvent::Modifier> modifiersForEvent(NSEvent *event)
648{
649 OptionSet<PlatformEvent::Modifier> modifiers;
650
651 if (event.modifierFlags & NSEventModifierFlagShift)
652 modifiers.add(PlatformEvent::Modifier::ShiftKey);
653 if (event.modifierFlags & NSEventModifierFlagControl)
654 modifiers.add(PlatformEvent::Modifier::ControlKey);
655 if (event.modifierFlags & NSEventModifierFlagOption)
656 modifiers.add(PlatformEvent::Modifier::AltKey);

684UInt8 keyCharForEvent(NSEvent *event)
685{
686 EventRef eventRef = (EventRef)[event _eventRef];
687 if (!eventRef)
688 return 0;
689
690 ByteCount keyCharCount = 0;
691 if (GetEventParameter(eventRef, kEventParamKeyMacCharCodes, typeChar, 0, 0, &keyCharCount, 0) != noErr)
692 return 0;
693 if (keyCharCount != 1)
694 return 0;
695
696 UInt8 keyChar = 0;
697 if (GetEventParameter(eventRef, kEventParamKeyMacCharCodes, typeChar, 0, sizeof(keyChar), &keyCharCount, &keyChar) != noErr)
698 return 0;
699
700 return keyChar;
701}

703class PlatformMouseEventBuilder : public PlatformMouseEvent {
704public:
705 PlatformMouseEventBuilder(NSEvent *event, NSEvent *correspondingPressureEvent, NSView *windowView)
706 {
707 // PlatformEvent
708 m_type = mouseEventTypeForEvent(event);
709
710 BOOL eventIsPressureEvent = [event type] == NSEventTypePressure;
711 if (eventIsPressureEvent) {
712 // Since AppKit doesn't send mouse events for force down or force up, we have to use the current pressure
713 // event and correspondingPressureEvent to detect if this is MouseForceDown, MouseForceUp, or just MouseForceChanged.
714 if (correspondingPressureEvent.stage == 1 && event.stage == 2)
715 m_type = PlatformEvent::MouseForceDown;
716 else if (correspondingPressureEvent.stage == 2 && event.stage == 1)
717 m_type = PlatformEvent::MouseForceUp;
718 else
719 m_type = PlatformEvent::MouseForceChanged;
720 }
721
722 m_modifiers = modifiersForEvent(event);
723 m_timestamp = eventTimeStampSince1970(event);
724
725 // PlatformMouseEvent
726 m_position = pointForEvent(event, windowView);
727 m_globalPosition = IntPoint(globalPointForEvent(event));
728 m_button = mouseButtonForEvent(event);
729 m_buttons = currentlyPressedMouseButtons();
730 m_clickCount = clickCountForEvent(event);
731 m_movementDelta = IntPoint(event.deltaX, event.deltaY);
732
733 m_force = 0;
734 int stage = eventIsPressureEvent ? event.stage : correspondingPressureEvent.stage;
735 double pressure = eventIsPressureEvent ? event.pressure : correspondingPressureEvent.pressure;
736 m_force = pressure + stage;
737
738 // Mac specific
739 m_modifierFlags = [event modifierFlags];
740 m_eventNumber = [event eventNumber];
741 m_menuTypeForEvent = typeForEvent(event);
742 }
