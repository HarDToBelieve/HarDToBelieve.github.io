44static bool canWritePasteboardType(const String& type)
45{
46 auto cfString = type.createCFString();
47ALLOW_DEPRECATED_DECLARATIONS_BEGIN
48 if (UTTypeIsDeclared(cfString.get()) || UTTypeIsDynamic(cfString.get()))
49 return true;
50ALLOW_DEPRECATED_DECLARATIONS_END
51
52 return [(__bridge NSString *)cfString.get() lengthOfBytesUsingEncoding:NSString.defaultCStringEncoding];
53}

79int PlatformPasteboard::numberOfFiles() const
80{
81 Vector<String> files;
82
83 NSArray *pasteboardTypes = [m_pasteboard types];
84 if ([pasteboardTypes containsObject:legacyFilesPromisePasteboardType()]) {
85 // FIXME: legacyFilesPromisePasteboardType() contains file types, not path names, but in
86 // this case we are only concerned with the count of them. The count of types should equal
87 // the count of files, but this isn't guaranteed as some legacy providers might only write
88 // unique file types.
89 getPathnamesForType(files, String(legacyFilesPromisePasteboardType()));
90 return files.size();
91 }
92
93 if ([pasteboardTypes containsObject:legacyFilenamesPasteboardType()]) {
94 getPathnamesForType(files, String(legacyFilenamesPasteboardType()));
95 return files.size();
96 }
97
98 return 0;
99}

111static bool pasteboardMayContainFilePaths(NSPasteboard *pasteboard)
112{
113 for (NSString *type in pasteboard.types) {
114 if ([type isEqualToString:(NSString *)legacyFilenamesPasteboardType()] || [type isEqualToString:(NSString *)legacyFilesPromisePasteboardType()] || Pasteboard::shouldTreatCocoaTypeAsFile(type))
115 return true;
116 }
117 return false;

120String PlatformPasteboard::stringForType(const String& pasteboardType) const
121{
122 if (pasteboardType == String { legacyURLPasteboardType() }) {
123 String urlString = ([NSURL URLFromPasteboard:m_pasteboard.get()] ?: [NSURL URLWithString:[m_pasteboard stringForType:legacyURLPasteboardType()]]).absoluteString;
124 if (pasteboardMayContainFilePaths(m_pasteboard.get()) && !Pasteboard::canExposeURLToDOMWhenPasteboardContainsFiles(urlString))
125 return { };
126 return urlString;
127 }
128
129 return [m_pasteboard stringForType:pasteboardType];

132static Vector<String> urlStringsFromPasteboard(NSPasteboard *pasteboard)
133{
134 NSArray<NSPasteboardItem *> *items = pasteboard.pasteboardItems;
135 Vector<String> urlStrings;
136 urlStrings.reserveInitialCapacity(items.count);
137 if (items.count > 1) {
138 for (NSPasteboardItem *item in items) {
139ALLOW_DEPRECATED_DECLARATIONS_BEGIN
140 if (id propertyList = [item propertyListForType:(__bridge NSString *)kUTTypeURL]) {
141 if (auto urlFromItem = adoptNS([[NSURL alloc] initWithPasteboardPropertyList:propertyList ofType:(__bridge NSString *)kUTTypeURL]))
142 urlStrings.uncheckedAppend([urlFromItem absoluteString]);
143 }
144ALLOW_DEPRECATED_DECLARATIONS_END
145 }
146 } else if (NSURL *urlFromPasteboard = [NSURL URLFromPasteboard:pasteboard])
147 urlStrings.uncheckedAppend(urlFromPasteboard.absoluteString);
148 else if (NSString *urlStringFromPasteboard = [pasteboard stringForType:legacyURLPasteboardType()])
149 urlStrings.uncheckedAppend(urlStringFromPasteboard);
150
151 bool mayContainFiles = pasteboardMayContainFilePaths(pasteboard);
152 urlStrings.removeAllMatching([&] (auto& urlString) {
153 return urlString.isEmpty() || (mayContainFiles && !Pasteboard::canExposeURLToDOMWhenPasteboardContainsFiles(urlString));
154 });
155
156 return urlStrings;
157}

159static String typeIdentifierForPasteboardType(const String& pasteboardType)
160{
161ALLOW_DEPRECATED_DECLARATIONS_BEGIN
162 if (UTTypeIsDeclared(pasteboardType.createCFString().get()))
163 return pasteboardType;
164
165 if (pasteboardType == String(legacyStringPasteboardType()))
166 return kUTTypeUTF8PlainText;
167
168 if (pasteboardType == String(legacyHTMLPasteboardType()))
169 return kUTTypeHTML;
170
171 if (pasteboardType == String(legacyURLPasteboardType()))
172 return kUTTypeURL;
173ALLOW_DEPRECATED_DECLARATIONS_END
174
175 return { };
176}

178Vector<String> PlatformPasteboard::allStringsForType(const String& pasteboardType) const
179{
180 auto typeIdentifier = typeIdentifierForPasteboardType(pasteboardType);
181ALLOW_DEPRECATED_DECLARATIONS_BEGIN
182 if (typeIdentifier == String(kUTTypeURL))
183 return urlStringsFromPasteboard(m_pasteboard.get());
184ALLOW_DEPRECATED_DECLARATIONS_END
185
186 NSArray<NSPasteboardItem *> *items = [m_pasteboard pasteboardItems];
187 Vector<String> strings;
188 strings.reserveInitialCapacity(items.count);
189 if (items.count > 1 && !typeIdentifier.isNull()) {
190 for (NSPasteboardItem *item in items) {
191 if (NSString *stringFromItem = [item stringForType:typeIdentifier])
192 strings.append(stringFromItem);
193 }
194 } else if (NSString *stringFromPasteboard = [m_pasteboard stringForType:pasteboardType])
195 strings.append(stringFromPasteboard);
196
197 return strings;
198}

200static const char* safeTypeForDOMToReadAndWriteForPlatformType(const String& platformType)
201{
202 if (platformType == String(legacyStringPasteboardType()) || platformType == String(NSPasteboardTypeString))
203 return "text/plain"_s;
204
205 if (platformType == String(legacyURLPasteboardType()))
206 return "text/uri-list"_s;
207
208ALLOW_DEPRECATED_DECLARATIONS_BEGIN
209 if (platformType == String(legacyHTMLPasteboardType()) || platformType == String(WebArchivePboardType) || platformType == String(kUTTypeWebArchive)
210 || platformType == String(legacyRTFDPasteboardType()) || platformType == String(legacyRTFPasteboardType()))
211 return "text/html"_s;
212ALLOW_DEPRECATED_DECLARATIONS_END
213
214 return nullptr;
215}

217Vector<String> PlatformPasteboard::typesSafeForDOMToReadAndWrite(const String& origin) const
218{
219 ListHashSet<String> domPasteboardTypes;
220 if (NSData *serializedCustomData = [m_pasteboard dataForType:@(PasteboardCustomData::cocoaType())]) {
221 auto data = PasteboardCustomData::fromSharedBuffer(SharedBuffer::create(serializedCustomData).get());
222 if (data.origin() == origin) {
223 for (auto& type : data.orderedTypes())
224 domPasteboardTypes.add(type);
225 }
226 }
227
228 NSArray<NSString *> *allTypes = [m_pasteboard types];
229 for (NSString *type in allTypes) {
230 if ([type isEqualToString:@(PasteboardCustomData::cocoaType())])
231 continue;
232
233 if (Pasteboard::isSafeTypeForDOMToReadAndWrite(type))
234 domPasteboardTypes.add(type);
235 else if (auto* domType = safeTypeForDOMToReadAndWriteForPlatformType(type)) {
236 auto domTypeAsString = String::fromUTF8(domType);
237 if (domTypeAsString == "text/uri-list" && stringForType(legacyURLPasteboardType()).isEmpty())
238 continue;
239 domPasteboardTypes.add(WTFMove(domTypeAsString));
240 }
241 }
242
243 return copyToVector(domPasteboardTypes);
244}

246int64_t PlatformPasteboard::write(const PasteboardCustomData& data)
247{
248 NSMutableArray *types = [NSMutableArray array];
249 data.forEachType([&] (auto& type) {
250 NSString *platformType = platformPasteboardTypeForSafeTypeForDOMToReadAndWrite(type, IncludeImageTypes::Yes);
251 if (platformType.length)
252 [types addObject:platformType];
253 });
254
255 bool shouldWriteCustomData = data.hasSameOriginCustomData() || !data.origin().isEmpty();
256 if (shouldWriteCustomData)
257 [types addObject:@(PasteboardCustomData::cocoaType())];
258
259 [m_pasteboard declareTypes:types owner:nil];
260
261 data.forEachPlatformStringOrBuffer([&] (auto& type, auto& stringOrBuffer) {
262 auto platformType = platformPasteboardTypeForSafeTypeForDOMToReadAndWrite(type, IncludeImageTypes::Yes);
263 if (platformType.isEmpty())
264 return;
265
266 if (WTF::holds_alternative<Ref<SharedBuffer>>(stringOrBuffer)) {
267 if (auto platformData = WTF::get<Ref<SharedBuffer>>(stringOrBuffer)->createNSData())
268 [m_pasteboard setData:platformData.get() forType:platformType];
269 } else if (WTF::holds_alternative<String>(stringOrBuffer)) {
270 auto string = WTF::get<String>(stringOrBuffer);
271 if (!!string)
272 [m_pasteboard setString:string forType:platformType];
273 }
274 });
275
276 if (shouldWriteCustomData) {
277 if (auto serializedCustomData = data.createSharedBuffer()->createNSData())
278 [m_pasteboard setData:serializedCustomData.get() forType:@(PasteboardCustomData::cocoaType())];
279 }
280
281 return changeCount();
282}

289String PlatformPasteboard::platformPasteboardTypeForSafeTypeForDOMToReadAndWrite(const String& domType, IncludeImageTypes includeImageTypes)
290{
291 if (domType == "text/plain")
292 return legacyStringPasteboardType();
293
294 if (domType == "text/html")
295 return legacyHTMLPasteboardType();
296
297 if (domType == "text/uri-list")
298 return legacyURLPasteboardType();
299

301 return legacyPNGPasteboardType();
302
303 return { };

316int64_t PlatformPasteboard::copy(const String& fromPasteboard)
317{
318 NSPasteboard* pasteboard = [NSPasteboard pasteboardWithName:fromPasteboard];
319 NSArray* types = [pasteboard types];
320
321 [m_pasteboard.get() addTypes:types owner:nil];
322 for (NSUInteger i = 0; i < [types count]; i++) {
323 NSString* type = [types objectAtIndex:i];
324 if (![m_pasteboard.get() setData:[pasteboard dataForType:type] forType:type])
325 return 0;
326 }
327 return changeCount();
328}

377int64_t PlatformPasteboard::setStringForType(const String& string, const String& pasteboardType)
378{
379 if (!canWritePasteboardType(pasteboardType))
380 return 0;
381
382 BOOL didWriteData;
383
384 if (pasteboardType == String(legacyURLPasteboardType())) {
385 // We cannot just use -NSPasteboard writeObjects:], because -declareTypes has been already called, implicitly creating an item.
386 NSURL *url = [NSURL URLWithString:string];
387 if ([[m_pasteboard.get() types] containsObject:legacyURLPasteboardType()]) {
388 NSURL *base = [url baseURL];
389 if (base)
390 didWriteData = [m_pasteboard.get() setPropertyList:@[[url relativeString], [base absoluteString]] forType:legacyURLPasteboardType()];
391 else if (url)
392 didWriteData = [m_pasteboard.get() setPropertyList:@[[url absoluteString], @""] forType:legacyURLPasteboardType()];
393 else
394 didWriteData = [m_pasteboard.get() setPropertyList:@[@"", @""] forType:legacyURLPasteboardType()];
395
396 if (!didWriteData)
397 return 0;
398 }
399
400ALLOW_DEPRECATED_DECLARATIONS_BEGIN
401 if ([[m_pasteboard.get() types] containsObject:(NSString *)kUTTypeURL]) {
402 didWriteData = [m_pasteboard.get() setString:[url absoluteString] forType:(NSString *)kUTTypeURL];
403 if (!didWriteData)
404 return 0;
405 }
406
407 if ([[m_pasteboard.get() types] containsObject:(NSString *)kUTTypeFileURL] && [url isFileURL]) {
408 didWriteData = [m_pasteboard.get() setString:[url absoluteString] forType:(NSString *)kUTTypeFileURL];
409 if (!didWriteData)
410 return 0;
411 }
412ALLOW_DEPRECATED_DECLARATIONS_END
413
414 } else {
415 didWriteData = [m_pasteboard.get() setString:string forType:pasteboardType];
416 if (!didWriteData)
417 return 0;
418 }
419
420 return changeCount();
421}

423static NSPasteboardType modernPasteboardTypeForWebSafeMIMEType(const String& webSafeType)
424{
425 if (webSafeType == "text/plain"_s)
426 return NSPasteboardTypeString;
427 if (webSafeType == "text/html"_s)
428 return NSPasteboardTypeHTML;
429 if (webSafeType == "text/uri-list"_s)
430 return NSPasteboardTypeURL;
431 if (webSafeType == "image/png"_s)
432 return NSPasteboardTypePNG;
433 return nil;
434}

436enum class ContainsFileURL { No, Yes };
437static String webSafeMIMETypeForModernPasteboardType(NSPasteboardType platformType, ContainsFileURL containsFileURL)
438{
439 if ([platformType isEqual:NSPasteboardTypeString] && containsFileURL == ContainsFileURL::No)
440 return "text/plain"_s;
441 if ([platformType isEqual:NSPasteboardTypeHTML] || [platformType isEqual:NSPasteboardTypeRTF] || [platformType isEqual:NSPasteboardTypeRTFD])
442 return "text/html"_s;
443 if ([platformType isEqual:NSPasteboardTypeURL] && containsFileURL == ContainsFileURL::No)
444 return "text/uri-list"_s;
445 if ([platformType isEqual:NSPasteboardTypePNG] || [platformType isEqual:NSPasteboardTypeTIFF])
446 return "image/png"_s;
447 return { };
448}

494static RetainPtr<NSPasteboardItem> createPasteboardItem(const PasteboardCustomData& data)
495{
496 auto item = adoptNS([[NSPasteboardItem alloc] init]);
497
498 if (data.hasSameOriginCustomData() || !data.origin().isEmpty()) {
499 if (auto serializedCustomData = data.createSharedBuffer()->createNSData())
500 [item setData:serializedCustomData.get() forType:@(PasteboardCustomData::cocoaType())];
501 }
502
503 data.forEachPlatformStringOrBuffer([&] (auto& type, auto& stringOrBuffer) {
504 auto platformType = modernPasteboardTypeForWebSafeMIMEType(type);
505 if (!platformType)
506 return;
507
508 if (WTF::holds_alternative<Ref<SharedBuffer>>(stringOrBuffer)) {
509 if (auto platformData = WTF::get<Ref<SharedBuffer>>(stringOrBuffer)->createNSData())
510 [item setData:platformData.get() forType:platformType];
511 } else if (WTF::holds_alternative<String>(stringOrBuffer)) {
512 auto string = WTF::get<String>(stringOrBuffer);
513 if (!!string)
514 [item setString:string forType:platformType];
515 }
516 });
517
518 return item;
519}

521int64_t PlatformPasteboard::write(const Vector<PasteboardCustomData>& itemData)
522{
523 if (itemData.size() == 1)
524 return write(itemData.first());
525
526 [m_pasteboard clearContents];
527 [m_pasteboard writeObjects:createNSArray(itemData, [] (auto& data) {
528 return createPasteboardItem(data);
529 }).get()];
530 return [m_pasteboard changeCount];

533std::optional<PasteboardItemInfo> PlatformPasteboard::informationForItemAtIndex(size_t index, int64_t changeCount)
534{
535 if (changeCount != [m_pasteboard changeCount])
536 return std::nullopt;
537
538 NSPasteboardItem *item = itemAtIndex(index);
539 if (!item)
540 return std::nullopt;
541
542 PasteboardItemInfo info;
543 NSArray<NSPasteboardType> *platformTypes = [item types];
544 auto containsFileURL = [platformTypes containsObject:NSPasteboardTypeFileURL] ? ContainsFileURL::Yes : ContainsFileURL::No;
545 ListHashSet<String> webSafeTypes;
546 info.platformTypesByFidelity.reserveInitialCapacity(platformTypes.count);
547 for (NSPasteboardType type in platformTypes) {
548 info.platformTypesByFidelity.uncheckedAppend(type);
549 auto webSafeType = webSafeMIMETypeForModernPasteboardType(type, containsFileURL);
550 if (webSafeType.isEmpty())
551 continue;
552
553 webSafeTypes.add(WTFMove(webSafeType));
554 }
555 info.containsFileURLAndFileUploadContent = containsFileURL == ContainsFileURL::Yes;
556 info.webSafeTypesByFidelity = copyToVector(webSafeTypes);
557 return info;
558}
