49// These functions scale between screen and page coordinates because JavaScript/DOM operations
50// assume that the screen and the page share the same coordinate system.

52PlatformDisplayID displayID(NSScreen *screen)
53{
54 ASSERT(hasProcessPrivilege(ProcessPrivilege::CanCommunicateWithWindowServer));
55 return [[screen.deviceDescription objectForKey:@"NSScreenNumber"] intValue];
56}
57
58static PlatformDisplayID displayID(Widget* widget)
59{
60 if (!widget)
61 return 0;
62
63 auto* view = widget->root();
64 if (!view)
65 return 0;
66
67 auto* hostWindow = view->hostWindow();
68 if (!hostWindow)
69 return 0;
70
71 return hostWindow->displayID();
72}

74// Screen containing the menubar.
75static NSScreen *firstScreen()
76{
77 ASSERT(hasProcessPrivilege(ProcessPrivilege::CanCommunicateWithWindowServer));
78 NSArray *screens = [NSScreen screens];
79 if (![screens count])
80 return nil;
81 return [screens objectAtIndex:0];
82}

84static NSWindow *window(Widget* widget)
85{
86 ASSERT(hasProcessPrivilege(ProcessPrivilege::CanCommunicateWithWindowServer));
87 if (!widget)
88 return nil;
89 return widget->platformWidget().window;

92static NSScreen *screen(Widget* widget)
93{
94 ASSERT(hasProcessPrivilege(ProcessPrivilege::CanCommunicateWithWindowServer));
95 // If the widget is in a window, use that, otherwise use the display ID from the host window.
96 // First case is for when the NSWindow is in the same process, second case for when it's not.
97 if (auto screenFromWindow = window(widget).screen)
98 return screenFromWindow;
99 return screen(displayID(widget));
100}

103static DynamicRangeMode convertAVVideoRangeToEnum(NSString* range)
104{
105 if (!range)
106 return DynamicRangeMode::None;
107 if (PAL::canLoad_AVFoundation_AVVideoRangeSDR() && [range isEqualTo:PAL::get_AVFoundation_AVVideoRangeSDR()])
108 return DynamicRangeMode::Standard;
109 if (PAL::canLoad_AVFoundation_AVVideoRangeHLG() && [range isEqualTo:PAL::get_AVFoundation_AVVideoRangeHLG()])
110 return DynamicRangeMode::HLG;
111 if (PAL::canLoad_AVFoundation_AVVideoRangeHDR10() && [range isEqualTo:PAL::get_AVFoundation_AVVideoRangeHDR10()])
112 return DynamicRangeMode::HDR10;
113 if (PAL::canLoad_AVFoundation_AVVideoRangeDolbyVisionPQ() && [range isEqualTo:PAL::get_AVFoundation_AVVideoRangeDolbyVisionPQ()])
114 return DynamicRangeMode::DolbyVisionPQ;
115
116 ASSERT_NOT_REACHED();
117 return DynamicRangeMode::None;
118}

121ScreenProperties collectScreenProperties()
122{
123 ASSERT(hasProcessPrivilege(ProcessPrivilege::CanCommunicateWithWindowServer));
124
125 ScreenProperties screenProperties;
126 bool screenHasInvertedColors = [[NSWorkspace sharedWorkspace] accessibilityDisplayShouldInvertColors];
127
128 for (NSScreen *screen in [NSScreen screens]) {
129 auto displayID = WebCore::displayID(screen);
130 FloatRect screenAvailableRect = screen.visibleFrame;
131 screenAvailableRect.setY(NSMaxY(screen.frame) - (screenAvailableRect.y() + screenAvailableRect.height())); // flip
132 FloatRect screenRect = screen.frame;
133 DestinationColorSpace colorSpace { screen.colorSpace.CGColorSpace };
134 int screenDepth = NSBitsPerPixelFromDepth(screen.depth);
135 int screenDepthPerComponent = NSBitsPerSampleFromDepth(screen.depth);
136 bool screenSupportsExtendedColor = [screen canRepresentDisplayGamut:NSDisplayGamutP3];
137 bool screenIsMonochrome = CGDisplayUsesForceToGray();
138 uint32_t displayMask = CGDisplayIDToOpenGLDisplayMask(displayID);
139 IORegistryGPUID gpuID = 0;
140 bool screenSupportsHighDynamicRange = false;
141 float scaleFactor = screen.backingScaleFactor;
142 DynamicRangeMode dynamicRangeMode = DynamicRangeMode::None;
143
144#if HAVE(AVPLAYER_VIDEORANGEOVERRIDE)
145 if (PAL::isAVFoundationFrameworkAvailable() && [PAL::getAVPlayerClass() respondsToSelector:@selector(preferredVideoRangeForDisplays:)]) {
146 dynamicRangeMode = convertAVVideoRangeToEnum([PAL::getAVPlayerClass() preferredVideoRangeForDisplays:@[ @(displayID) ]]);
147 screenSupportsHighDynamicRange = dynamicRangeMode > DynamicRangeMode::Standard;
148 }
149#endif
150#if HAVE(AVPLAYER_VIDEORANGEOVERRIDE) && USE(MEDIATOOLBOX)
151 else
152#endif
153#if USE(MEDIATOOLBOX)
154 if (PAL::isMediaToolboxFrameworkAvailable() && PAL::canLoad_MediaToolbox_MTShouldPlayHDRVideo())
155 screenSupportsHighDynamicRange = PAL::softLink_MediaToolbox_MTShouldPlayHDRVideo((__bridge CFArrayRef)@[ @(displayID) ]);
156#endif
157
158 if (!screenSupportsHighDynamicRange && dynamicRangeMode > DynamicRangeMode::Standard)
159 dynamicRangeMode = DynamicRangeMode::Standard;
160
161 if (displayMask)
162 gpuID = gpuIDForDisplayMask(displayMask);
163
164 screenProperties.screenDataMap.set(displayID, ScreenData { screenAvailableRect, screenRect, WTFMove(colorSpace), screenDepth, screenDepthPerComponent, screenSupportsExtendedColor, screenHasInvertedColors, screenSupportsHighDynamicRange, screenIsMonochrome, displayMask, gpuID, dynamicRangeMode, scaleFactor });
165
166 if (!screenProperties.primaryDisplayID)
167 screenProperties.primaryDisplayID = displayID;
168 }
169
170 return screenProperties;
171}

209IORegistryGPUID gpuIDForDisplayMask(GLuint displayMask)
210{
211 GLint numRenderers = 0;
212 CGLRendererInfoObj rendererInfo = nullptr;
213 CGLError error = CGLQueryRendererInfo(displayMask, &rendererInfo, &numRenderers);
214 if (!numRenderers || !rendererInfo || error != kCGLNoError)
215 return 0;
216
217 // The 0th renderer should not be the software renderer.
218 GLint isAccelerated;
219 error = CGLDescribeRenderer(rendererInfo, 0, kCGLRPAccelerated, &isAccelerated);
220 if (!isAccelerated || error != kCGLNoError) {
221 CGLDestroyRendererInfo(rendererInfo);
222 return 0;
223 }
224
225 GLint gpuIDLow = 0;
226 GLint gpuIDHigh = 0;
227
228 error = CGLDescribeRenderer(rendererInfo, 0, kCGLRPRegistryIDLow, &gpuIDLow);
229 if (error != kCGLNoError) {
230 CGLDestroyRendererInfo(rendererInfo);
231 return 0;
232 }
233
234 error = CGLDescribeRenderer(rendererInfo, 0, kCGLRPRegistryIDHigh, &gpuIDHigh);
235 if (error != kCGLNoError) {
236 CGLDestroyRendererInfo(rendererInfo);
237 return 0;
238 }
239
240 CGLDestroyRendererInfo(rendererInfo);
241 return (IORegistryGPUID) gpuIDHigh << 32 | gpuIDLow;
242}

325NSScreen *screen(NSWindow *window)
326{
327 ASSERT(hasProcessPrivilege(ProcessPrivilege::CanCommunicateWithWindowServer));
328 return [window screen] ?: firstScreen();

331NSScreen *screen(PlatformDisplayID displayID)
332{
333 ASSERT(hasProcessPrivilege(ProcessPrivilege::CanCommunicateWithWindowServer));
334 for (NSScreen *screen in [NSScreen screens]) {
335 if (WebCore::displayID(screen) == displayID)
336 return screen;
337 }
338 return firstScreen();
339}

390FloatRect toUserSpace(const NSRect& rect, NSWindow *destination)
391{
392 FloatRect userRect = rect;
393 userRect.setY(NSMaxY([screen(destination) frame]) - (userRect.y() + userRect.height())); // flip
394 return userRect;
395}
396
397FloatRect toUserSpaceForPrimaryScreen(const NSRect& rect)
398{
399 FloatRect userRect = rect;
400 userRect.setY(NSMaxY(screenRectForDisplay(primaryScreenDisplayID())) - (userRect.y() + userRect.height())); // flip
401 return userRect;
402}
403
404NSRect toDeviceSpace(const FloatRect& rect, NSWindow *source)
405{
406 FloatRect deviceRect = rect;
407 deviceRect.setY(NSMaxY([screen(source) frame]) - (deviceRect.y() + deviceRect.height())); // flip
408 return deviceRect;
409}

411NSPoint flipScreenPoint(const NSPoint& screenPoint, NSScreen *screen)
412{
413 NSPoint flippedPoint = screenPoint;
414 flippedPoint.y = NSMaxY([screen frame]) - flippedPoint.y;
415 return flippedPoint;
416}
