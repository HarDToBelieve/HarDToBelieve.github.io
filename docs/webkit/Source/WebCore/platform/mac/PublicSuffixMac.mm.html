39bool isPublicSuffix(const String& domain)
40{
41 // Explicitly cast the domain to a NSString before calling decodeHostName() so we get a NSString back instead of a String.
42 NSString *host = decodeHostName((NSString *)domain);
43 return host && _CFHostIsDomainTopLevel((__bridge CFStringRef)host);
44}
45
46String topPrivatelyControlledDomain(const String& domain)
47{
48 if (domain.isEmpty())
49 return { };
50 if (!domain.isAllASCII())
51 return domain;
52
53 static Lock cacheLock;
54 Locker locker { cacheLock };
55
56 static NeverDestroyed<HashMap<String, String, ASCIICaseInsensitiveHash>> cache;
57
58 auto isolatedDomain = domain.isolatedCopy();
59
60 constexpr auto maximumSizeToPreventUnlimitedGrowth = 128;
61 if (cache.get().size() == maximumSizeToPreventUnlimitedGrowth)
62 cache.get().remove(cache.get().random());
63
64 return cache.get().ensure(isolatedDomain, [&isolatedDomain] {
65 const auto lowercaseDomain = isolatedDomain.convertToASCIILowercase();
66 if (lowercaseDomain == "localhost")
67 return lowercaseDomain;
68
69 if (URL::hostIsIPAddress(lowercaseDomain))
70 return lowercaseDomain;
71
72 size_t separatorPosition;
73 for (unsigned labelStart = 0; (separatorPosition = lowercaseDomain.find('.', labelStart)) != notFound; labelStart = separatorPosition + 1) {
74 if (isPublicSuffix(lowercaseDomain.substring(separatorPosition + 1)))
75 return lowercaseDomain.substring(labelStart);
76 }
77 return String();
78 }).iterator->value.isolatedCopy();
79}

81String decodeHostName(const String& domain)
82{
83 return WTF::decodeHostName(domain);
84}
