88bool ScrollController::handleWheelEvent(const PlatformWheelEvent& wheelEvent)
89{
90 if (processWheelEventForScrollSnap(wheelEvent))
91 return true;
92
93 if (wheelEvent.phase() == PlatformWheelEventPhase::MayBegin || wheelEvent.phase() == PlatformWheelEventPhase::Cancelled)
94 return false;
95
96 if (wheelEvent.phase() == PlatformWheelEventPhase::Began) {
97 // FIXME: Trying to decide if a gesture is horizontal or vertical at the "began" phase is very error-prone.
98 auto direction = directionFromEvent(wheelEvent, ScrollEventAxis::Horizontal);
99 if (direction && m_client.isPinnedForScrollDelta(FloatSize(-wheelEvent.deltaX(), 0)) && !shouldRubberBandInDirection(direction.value()))
100 return false;
101
102 direction = directionFromEvent(wheelEvent, ScrollEventAxis::Vertical);
103 if (direction && m_client.isPinnedForScrollDelta(FloatSize(0, -wheelEvent.deltaY())) && !shouldRubberBandInDirection(direction.value()))
104 return false;
105
106 m_momentumScrollInProgress = false;
107 m_ignoreMomentumScrolls = false;
108 m_lastMomentumScrollTimestamp = { };
109 m_momentumVelocity = { };
110
111 IntSize stretchAmount = m_client.stretchAmount();
112 m_stretchScrollForce.setWidth(reboundDeltaForElasticDelta(stretchAmount.width()));
113 m_stretchScrollForce.setHeight(reboundDeltaForElasticDelta(stretchAmount.height()));
114 m_overflowScrollDelta = { };
115
116 stopSnapRubberbandAnimation();
117 updateRubberBandingState();
118 return true;
119 }
120
121 if (wheelEvent.phase() == PlatformWheelEventPhase::Ended) {
122 // FIXME: This triggers the rubberband timer even when we don't start rubberbanding.
123 snapRubberBand();
124 updateRubberBandingState();
125 return true;
126 }
127
128 bool isMomentumScrollEvent = (wheelEvent.momentumPhase() != PlatformWheelEventPhase::None);
129 if (m_ignoreMomentumScrolls && (isMomentumScrollEvent || m_isAnimatingRubberBand)) {
130 if (wheelEvent.momentumPhase() == PlatformWheelEventPhase::Ended) {
131 m_ignoreMomentumScrolls = false;
132 return true;
133 }
134
135 if (isMomentumScrollEvent && m_ignoreMomentumScrolls)
136 return true;
137
138 return false;
139 }
140
141 float deltaX = m_overflowScrollDelta.width();
142 float deltaY = m_overflowScrollDelta.height();
143
144 // Reset overflow values because we may decide to remove delta at various points and put it into overflow.
145 m_overflowScrollDelta = { };
146
147 IntSize stretchAmount = m_client.stretchAmount();
148 bool isVerticallyStretched = stretchAmount.height();
149 bool isHorizontallyStretched = stretchAmount.width();
150
151 float eventCoalescedDeltaX;
152 float eventCoalescedDeltaY;
153
154 if (isVerticallyStretched || isHorizontallyStretched) {
155 eventCoalescedDeltaX = -wheelEvent.unacceleratedScrollingDeltaX();
156 eventCoalescedDeltaY = -wheelEvent.unacceleratedScrollingDeltaY();
157 } else {
158 eventCoalescedDeltaX = -wheelEvent.deltaX();
159 eventCoalescedDeltaY = -wheelEvent.deltaY();
160 }
161
162 deltaX += eventCoalescedDeltaX;
163 deltaY += eventCoalescedDeltaY;
164
165 // Slightly prefer scrolling vertically by applying the = case to deltaY
166 // FIXME: Use wheelDeltaBiasingTowardsVertical().
167 if (fabsf(deltaY) >= fabsf(deltaX))
168 deltaX = 0;
169 else
170 deltaY = 0;
171
172 bool shouldStretch = false;
173
174 PlatformWheelEventPhase momentumPhase = wheelEvent.momentumPhase();
175
176 // If we are starting momentum scrolling then do some setup.
177 if (!m_momentumScrollInProgress && (momentumPhase == PlatformWheelEventPhase::Began || momentumPhase == PlatformWheelEventPhase::Changed))
178 m_momentumScrollInProgress = true;
179
180 auto timeDelta = wheelEvent.timestamp() - m_lastMomentumScrollTimestamp;
181 if (m_inScrollGesture || m_momentumScrollInProgress) {
182 if (m_lastMomentumScrollTimestamp && timeDelta > 0_s && timeDelta < scrollVelocityZeroingTimeout) {
183 m_momentumVelocity.setWidth(eventCoalescedDeltaX / timeDelta.seconds());
184 m_momentumVelocity.setHeight(eventCoalescedDeltaY / timeDelta.seconds());
185 m_lastMomentumScrollTimestamp = wheelEvent.timestamp();
186 } else {
187 m_lastMomentumScrollTimestamp = wheelEvent.timestamp();
188 m_momentumVelocity = FloatSize();
189 }
190
191 if (isVerticallyStretched) {
192 if (!isHorizontallyStretched && m_client.isPinnedForScrollDelta(FloatSize(deltaX, 0))) {
193 // Stretching only in the vertical.
194 if (deltaY && (fabsf(deltaX / deltaY) < rubberbandDirectionLockStretchRatio))
195 deltaX = 0;
196 else if (fabsf(deltaX) < rubberbandMinimumRequiredDeltaBeforeStretch) {
197 m_overflowScrollDelta.setWidth(m_overflowScrollDelta.width() + deltaX);
198 deltaX = 0;
199 } else
200 m_overflowScrollDelta.setWidth(m_overflowScrollDelta.width() + deltaX);
201 }
202 } else if (isHorizontallyStretched) {
203 // Stretching only in the horizontal.
204 if (m_client.isPinnedForScrollDelta(FloatSize(0, deltaY))) {
205 if (deltaX && (fabsf(deltaY / deltaX) < rubberbandDirectionLockStretchRatio))
206 deltaY = 0;
207 else if (fabsf(deltaY) < rubberbandMinimumRequiredDeltaBeforeStretch) {
208 m_overflowScrollDelta.setHeight(m_overflowScrollDelta.height() + deltaY);
209 deltaY = 0;
210 } else
211 m_overflowScrollDelta.setHeight(m_overflowScrollDelta.height() + deltaY);
212 }
213 } else {
214 // Not stretching at all yet.
215 if (m_client.isPinnedForScrollDelta(FloatSize(deltaX, deltaY))) {
216 if (fabsf(deltaY) >= fabsf(deltaX)) {
217 if (fabsf(deltaX) < rubberbandMinimumRequiredDeltaBeforeStretch) {
218 m_overflowScrollDelta.setWidth(m_overflowScrollDelta.width() + deltaX);
219 deltaX = 0;
220 } else
221 m_overflowScrollDelta.setWidth(m_overflowScrollDelta.width() + deltaX);
222 }
223
224 if (!m_client.allowsHorizontalStretching(wheelEvent))
225 deltaX = 0;
226
227 if (!m_client.allowsVerticalStretching(wheelEvent))
228 deltaY = 0;
229
230 shouldStretch = deltaX || deltaY;
231 }
232 }
233
234 LOG_WITH_STREAM(Scrolling, stream << "ScrollController::handleWheelEvent() - deltaX " << deltaX << " deltaY " << deltaY << " pinned " << m_client.isPinnedForScrollDelta(FloatSize(deltaX, deltaY)) << " shouldStretch " << shouldStretch);
235 }
236
237 bool handled = true;
238
239 if (deltaX || deltaY) {
240 if (!(shouldStretch || isVerticallyStretched || isHorizontallyStretched)) {
241 if (deltaY) {
242 deltaY *= scrollWheelMultiplier();
243 m_client.immediateScrollBy(FloatSize(0, deltaY));
244 }
245 if (deltaX) {
246 deltaX *= scrollWheelMultiplier();
247 m_client.immediateScrollBy(FloatSize(deltaX, 0));
248 }
249 } else {
250 if (deltaX) {
251 if (!m_client.allowsHorizontalStretching(wheelEvent)) {
252 deltaX = 0;
253 eventCoalescedDeltaX = 0;
254 handled = false;
255 } else if (!isHorizontallyStretched && !m_client.isPinnedForScrollDelta(FloatSize(deltaX, 0))) {
256 deltaX *= scrollWheelMultiplier();
257
258 m_client.immediateScrollByWithoutContentEdgeConstraints(FloatSize(deltaX, 0));
259 deltaX = 0;
260 }
261 }
262
263 if (deltaY) {
264 if (!m_client.allowsVerticalStretching(wheelEvent)) {
265 deltaY = 0;
266 eventCoalescedDeltaY = 0;
267 handled = false;
268 } else if (!isVerticallyStretched && !m_client.isPinnedForScrollDelta(FloatSize(0, deltaY))) {
269 deltaY *= scrollWheelMultiplier();
270
271 m_client.immediateScrollByWithoutContentEdgeConstraints(FloatSize(0, deltaY));
272 deltaY = 0;
273 }
274 }
275
276 IntSize stretchAmount = m_client.stretchAmount();
277
278 if (m_momentumScrollInProgress) {
279 if ((m_client.isPinnedForScrollDelta(FloatSize(eventCoalescedDeltaX, eventCoalescedDeltaY)) || (fabsf(eventCoalescedDeltaX) + fabsf(eventCoalescedDeltaY) <= 0)) && m_lastMomentumScrollTimestamp) {
280 m_ignoreMomentumScrolls = true;
281 m_momentumScrollInProgress = false;
282 snapRubberBand();
283 }
284 }
285
286 m_stretchScrollForce.setWidth(m_stretchScrollForce.width() + deltaX);
287 m_stretchScrollForce.setHeight(m_stretchScrollForce.height() + deltaY);
288
289 FloatSize dampedDelta(ceilf(elasticDeltaForReboundDelta(m_stretchScrollForce.width())), ceilf(elasticDeltaForReboundDelta(m_stretchScrollForce.height())));
290
291 m_client.immediateScrollByWithoutContentEdgeConstraints(dampedDelta - stretchAmount);
292 }
293 }
294
295 if (m_momentumScrollInProgress && momentumPhase == PlatformWheelEventPhase::Ended) {
296 m_momentumScrollInProgress = false;
297 m_ignoreMomentumScrolls = false;
298 m_lastMomentumScrollTimestamp = { };
299 }
300
301 updateRubberBandingState();
302
303 return handled;
304}

306FloatSize ScrollController::wheelDeltaBiasingTowardsVertical(const PlatformWheelEvent& wheelEvent)
307{
308 auto deltaX = wheelEvent.deltaX();
309 auto deltaY = wheelEvent.deltaY();
310
311 if (fabsf(deltaY) >= fabsf(deltaX))
312 deltaX = 0;
313 else
314 deltaY = 0;
315
316 return { deltaX, deltaY };
317}

319std::optional<ScrollDirection> ScrollController::directionFromEvent(const PlatformWheelEvent& wheelEvent, std::optional<ScrollEventAxis> axis, WheelAxisBias bias)
320{
321 // FIXME: It's impossible to infer direction from a single event, since the start of a gesture is either zero or
322 // has small deltas on both axes.
323
324 auto wheelDelta = FloatSize { wheelEvent.deltaX(), wheelEvent.deltaY() };
325 if (bias == WheelAxisBias::Vertical)
326 wheelDelta = wheelDeltaBiasingTowardsVertical(wheelEvent);
327
328 if (axis) {
329 switch (axis.value()) {
330 case ScrollEventAxis::Vertical:
331 if (wheelDelta.height() < 0)
332 return ScrollDown;
333
334 if (wheelDelta.height() > 0)
335 return ScrollUp;
336 break;
337
338 case ScrollEventAxis::Horizontal:
339 if (wheelDelta.width() > 0)
340 return ScrollLeft;
341
342 if (wheelDelta.width() < 0)
343 return ScrollRight;
344 }
345
346 return std::nullopt;
347 }
348
349 // Check Y first because vertical scrolling dominates.
350 if (wheelDelta.height() < 0)
351 return ScrollDown;
352
353 if (wheelDelta.height() > 0)
354 return ScrollUp;
355
356 if (wheelDelta.width() > 0)
357 return ScrollLeft;
358
359 if (wheelDelta.width() < 0)
360 return ScrollRight;
361
362 return std::nullopt;
363}

365#if ENABLE(RUBBER_BANDING)
366static inline float roundTowardZero(float num)
367{
368 return num > 0 ? ceilf(num - 0.5f) : floorf(num + 0.5f);
369}
370
371static inline float roundToDevicePixelTowardZero(float num)
372{
373 float roundedNum = roundf(num);
374 if (fabs(num - roundedNum) < 0.125)
375 num = roundedNum;
376
377 return roundTowardZero(num);
378}

380void ScrollController::updateRubberBandAnimatingState(MonotonicTime currentTime)
381{
382 if (!m_isAnimatingRubberBand)
383 return;
384
385 if (isScrollSnapInProgress())
386 return;
387
388 LOG_WITH_STREAM(Scrolling, stream << "ScrollController::updateRubberBandAnimatingState() - main thread " << isMainThread());
389
390 if (!m_momentumScrollInProgress || m_ignoreMomentumScrolls) {
391 auto timeDelta = currentTime - m_startTime;
392
393 if (m_startStretch.isZero()) {
394 m_startStretch = m_client.stretchAmount();
395 if (m_startStretch == FloatSize()) {
396 stopRubberbanding();
397 return;
398 }
399
400 m_origVelocity = m_momentumVelocity;
401
402 // Just like normal scrolling, prefer vertical rubberbanding
403 if (fabsf(m_origVelocity.height()) >= fabsf(m_origVelocity.width()))
404 m_origVelocity.setWidth(0);
405
406 // Don't rubber-band horizontally if it's not possible to scroll horizontally
407 if (!m_client.allowsHorizontalScrolling())
408 m_origVelocity.setWidth(0);
409
410 // Don't rubber-band vertically if it's not possible to scroll vertically
411 if (!m_client.allowsVerticalScrolling())
412 m_origVelocity.setHeight(0);
413 }
414
415 FloatPoint delta(roundToDevicePixelTowardZero(elasticDeltaForTimeDelta(m_startStretch.width(), -m_origVelocity.width(), timeDelta)),
416 roundToDevicePixelTowardZero(elasticDeltaForTimeDelta(m_startStretch.height(), -m_origVelocity.height(), timeDelta)));
417
418 if (fabs(delta.x()) >= 1 || fabs(delta.y()) >= 1) {
419 m_client.immediateScrollByWithoutContentEdgeConstraints(FloatSize(delta.x(), delta.y()) - m_client.stretchAmount());
420
421 FloatSize newStretch = m_client.stretchAmount();
422
423 m_stretchScrollForce.setWidth(reboundDeltaForElasticDelta(newStretch.width()));
424 m_stretchScrollForce.setHeight(reboundDeltaForElasticDelta(newStretch.height()));
425 } else {
426 m_client.adjustScrollPositionToBoundsIfNecessary();
427 stopRubberbanding();
428 }
429 } else {
430 m_startTime = currentTime;
431 m_startStretch = { };
432 if (!isRubberBandInProgressInternal())
433 stopSnapRubberbandAnimation();
434 }
435
436 updateRubberBandingState();
437}

503void ScrollController::snapRubberBand()
504{
505 auto timeDelta = WallTime::now() - m_lastMomentumScrollTimestamp;
506 if (m_lastMomentumScrollTimestamp && timeDelta >= scrollVelocityZeroingTimeout)
507 m_momentumVelocity = { };
508
509 if (m_isAnimatingRubberBand)
510 return;
511
512 m_startTime = MonotonicTime::now();
513 m_startStretch = { };
514 m_origVelocity = { };
515
516 startRubberbandAnimation();
517}

541void ScrollController::updateRubberBandingState()
542{
543 bool isRubberBanding = isRubberBandInProgressInternal();
544 if (isRubberBanding == m_isRubberBanding)
545 return;
546
547 m_isRubberBanding = isRubberBanding;
548 if (m_isRubberBanding)
549 updateRubberBandingEdges(m_client.stretchAmount());
550 else
551 m_rubberBandingEdges = { };
552
553 m_client.rubberBandingStateChanged(m_isRubberBanding);
554}

567enum class WheelEventStatus {
568 UserScrollBegin,
569 UserScrolling,
570 UserScrollEnd,
571 MomentumScrollBegin,
572 MomentumScrolling,
573 MomentumScrollEnd,
574 StatelessScrollEvent,
575 Unknown
576};
577
578static inline WheelEventStatus toWheelEventStatus(PlatformWheelEventPhase phase, PlatformWheelEventPhase momentumPhase)
579{
580 if (phase == PlatformWheelEventPhase::None) {
581 switch (momentumPhase) {
582 case PlatformWheelEventPhase::Began:
583 return WheelEventStatus::MomentumScrollBegin;
584
585 case PlatformWheelEventPhase::Changed:
586 return WheelEventStatus::MomentumScrolling;
587
588 case PlatformWheelEventPhase::Ended:
589 return WheelEventStatus::MomentumScrollEnd;
590
591 case PlatformWheelEventPhase::None:
592 return WheelEventStatus::StatelessScrollEvent;
593
594 default:
595 return WheelEventStatus::Unknown;
596 }
597 }
598 if (momentumPhase == PlatformWheelEventPhase::None) {
599 switch (phase) {
600 case PlatformWheelEventPhase::Began:
601 case PlatformWheelEventPhase::MayBegin:
602 return WheelEventStatus::UserScrollBegin;
603
604 case PlatformWheelEventPhase::Changed:
605 return WheelEventStatus::UserScrolling;
606
607 case PlatformWheelEventPhase::Ended:
608 case PlatformWheelEventPhase::Cancelled:
609 return WheelEventStatus::UserScrollEnd;
610
611 default:
612 return WheelEventStatus::Unknown;
613 }
614 }
615 return WheelEventStatus::Unknown;
616}

618#if !LOG_DISABLED
619static TextStream& operator<<(TextStream& ts, WheelEventStatus status)
620{
621 switch (status) {
622 case WheelEventStatus::UserScrollBegin: ts << "UserScrollBegin"; break;
623 case WheelEventStatus::UserScrolling: ts << "UserScrolling"; break;
624 case WheelEventStatus::UserScrollEnd: ts << "UserScrollEnd"; break;
625 case WheelEventStatus::MomentumScrollBegin: ts << "MomentumScrollBegin"; break;
626 case WheelEventStatus::MomentumScrolling: ts << "MomentumScrolling"; break;
627 case WheelEventStatus::MomentumScrollEnd: ts << "MomentumScrollEnd"; break;
628 case WheelEventStatus::StatelessScrollEvent: ts << "StatelessScrollEvent"; break;
629 case WheelEventStatus::Unknown: ts << "Unknown"; break;
630 }
631 return ts;
632}

636{
637 if (!usesScrollSnap())
638 return false;
639

641 return scrollSnapState == ScrollSnapState::Gliding || scrollSnapState == ScrollSnapState::DestinationReached;

644void ScrollController::scheduleStatelessScrollSnap()
645{
646 stopScrollSnapAnimation();
647 if (m_statelessSnapTransitionTimer) {
648 m_statelessSnapTransitionTimer->stop();
649 m_statelessSnapTransitionTimer = nullptr;
650 }
651 if (!usesScrollSnap())
652 return;
653
654 static const Seconds statelessScrollSnapDelay = 750_ms;
655 m_statelessSnapTransitionTimer = m_client.createTimer([this] {
656 statelessSnapTransitionTimerFired();
657 });
658 m_statelessSnapTransitionTimer->startOneShot(statelessScrollSnapDelay);
659 startDeferringWheelEventTestCompletionDueToScrollSnapping();
660}
661
662void ScrollController::statelessSnapTransitionTimerFired()
663{
664 m_statelessSnapTransitionTimer = nullptr;
665
666 if (!usesScrollSnap())
667 return;
668
669 m_scrollSnapState->transitionToSnapAnimationState(m_client.scrollExtent(), m_client.viewportSize(), m_client.pageScaleFactor(), m_client.scrollOffset());
670 startScrollSnapAnimation();
671}

683bool ScrollController::processWheelEventForScrollSnap(const PlatformWheelEvent& wheelEvent)
684{
685 if (!usesScrollSnap())
686 return false;
687
688 if (m_scrollSnapState->snapOffsetsForAxis(ScrollEventAxis::Horizontal).isEmpty() && m_scrollSnapState->snapOffsetsForAxis(ScrollEventAxis::Vertical).isEmpty())
689 return false;
690
691 auto status = toWheelEventStatus(wheelEvent.phase(), wheelEvent.momentumPhase());
692
693 LOG_WITH_STREAM(ScrollSnap, stream << "ScrollController " << this << " processWheelEventForScrollSnap: status " << status);
694
695 bool isMomentumScrolling = false;
696 switch (status) {
697 case WheelEventStatus::UserScrollBegin:
698 case WheelEventStatus::UserScrolling:
699 stopScrollSnapAnimation();
700 m_scrollSnapState->transitionToUserInteractionState();
701 m_dragEndedScrollingVelocity = -wheelEvent.scrollingVelocity();
702 break;
703 case WheelEventStatus::UserScrollEnd:
704 m_scrollSnapState->transitionToSnapAnimationState(m_client.scrollExtent(), m_client.viewportSize(), m_client.pageScaleFactor(), m_client.scrollOffset());
705 startScrollSnapAnimation();
706 break;
707 case WheelEventStatus::MomentumScrollBegin:
708 m_scrollSnapState->transitionToGlideAnimationState(m_client.scrollExtent(), m_client.viewportSize(), m_client.pageScaleFactor(), m_client.scrollOffset(), m_dragEndedScrollingVelocity, FloatSize(-wheelEvent.deltaX(), -wheelEvent.deltaY()));
709 m_dragEndedScrollingVelocity = { };
710 isMomentumScrolling = true;
711 break;
712 case WheelEventStatus::MomentumScrolling:
713 case WheelEventStatus::MomentumScrollEnd:
714 isMomentumScrolling = true;
715 break;
716 case WheelEventStatus::StatelessScrollEvent:
717 m_scrollSnapState->transitionToUserInteractionState();
718 scheduleStatelessScrollSnap();
719 break;
720 case WheelEventStatus::Unknown:
721 ASSERT_NOT_REACHED();
722 break;
723 }
724
725 return isMomentumScrolling && shouldOverrideMomentumScrolling();
726}

763void ScrollController::updateScrollSnapAnimatingState(MonotonicTime currentTime)
764{
765 if (!m_isAnimatingScrollSnap)
766 return;
767
768 if (!usesScrollSnap()) {
769 ASSERT_NOT_REACHED();
770 return;
771 }
772
773 bool isAnimationComplete;
774 auto animationOffset = m_scrollSnapState->currentAnimatedScrollOffset(currentTime, isAnimationComplete);
775 auto currentOffset = m_client.scrollOffset();
776
777 LOG_WITH_STREAM(ScrollSnap, stream << "ScrollController " << this << " updateScrollSnapAnimatingState - isAnimationComplete " << isAnimationComplete << " currentOffset " << currentOffset << " (main thread " << isMainThread() << ")");
778
779 m_client.immediateScrollByWithoutContentEdgeConstraints(FloatSize(animationOffset.x() - currentOffset.x(), animationOffset.y() - currentOffset.y()));
780 if (isAnimationComplete) {
781 m_scrollSnapState->transitionToDestinationReachedState();
782 stopScrollSnapAnimation();
783 }
