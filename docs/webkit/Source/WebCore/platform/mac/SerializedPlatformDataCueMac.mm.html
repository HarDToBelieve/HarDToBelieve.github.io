71JSC::JSValue SerializedPlatformDataCueMac::deserialize(JSC::JSGlobalObject* lexicalGlobalObject) const
72{
73#if JSC_OBJC_API_ENABLED
74 if (!m_nativeValue)
75 return JSC::jsNull();
76
77 JSGlobalContextRef jsGlobalContextRef = toGlobalRef(lexicalGlobalObject);
78 JSContext *jsContext = [JSContext contextWithJSGlobalContextRef:jsGlobalContextRef];
79 JSValue *serializedValue = jsValueWithValueInContext(m_nativeValue.get(), jsContext);
80
81 return toJS(lexicalGlobalObject, [serializedValue JSValueRef]);
82#else
83 UNUSED_PARAM(lexicalGlobalObject);
84 return JSC::jsNull();
85#endif
86}

88bool SerializedPlatformDataCueMac::isEqual(const SerializedPlatformDataCue& other) const
89{
90 if (other.platformType() != PlatformType::ObjC)
91 return false;
92
93 const SerializedPlatformDataCueMac* otherObjC = toSerializedPlatformDataCueMac(&other);
94
95 if (!m_nativeValue || !otherObjC->nativeValue())
96 return false;
97
98 return [m_nativeValue.get() isEqual:otherObjC->nativeValue()];
99}

127static JSValue *jsValueWithValueInContext(id value, JSContext *context)
128{
129 if ([value isKindOfClass:[NSString class]] || [value isKindOfClass:[NSNumber class]])
130 return [JSValue valueWithObject:value inContext:context];
131
132 if ([value isKindOfClass:[NSLocale class]])
133 return [JSValue valueWithObject:[value localeIdentifier] inContext:context];
134
135 if ([value isKindOfClass:[NSDictionary class]])
136 return jsValueWithDictionaryInContext(value, context);
137
138 if ([value isKindOfClass:[NSArray class]])
139 return jsValueWithArrayInContext(value, context);
140
141 if ([value isKindOfClass:[NSData class]])
142 return jsValueWithDataInContext(value, context);
143
144 if ([value isKindOfClass:PAL::getAVMetadataItemClass()])
145 return jsValueWithAVMetadataItemInContext(value, context);
146
147 return nil;
148}

150static JSValue *jsValueWithDataInContext(NSData *data, JSContext *context)
151{
152 auto dataArray = ArrayBuffer::tryCreate([data bytes], [data length]);
153
154 auto* lexicalGlobalObject = toJS([context JSGlobalContextRef]);
155 JSC::JSValue array = toJS(lexicalGlobalObject, JSC::jsCast<JSDOMGlobalObject*>(lexicalGlobalObject), dataArray.get());
156
157 return [JSValue valueWithJSValueRef:toRef(lexicalGlobalObject, array) inContext:context];
158}
159
160static JSValue *jsValueWithArrayInContext(NSArray *array, JSContext *context)
161{
162 JSValueRef exception = 0;
163 JSValue *result = [JSValue valueWithNewArrayInContext:context];
164 JSObjectRef resultObject = JSValueToObject([context JSGlobalContextRef], [result JSValueRef], &exception);
165 if (exception)
166 return [JSValue valueWithUndefinedInContext:context];
167
168 NSUInteger count = [array count];
169 for (NSUInteger i = 0; i < count; ++i) {
170 JSValue *value = jsValueWithValueInContext([array objectAtIndex:i], context);
171 if (!value)
172 continue;
173
174 JSObjectSetPropertyAtIndex([context JSGlobalContextRef], resultObject, (unsigned)i, [value JSValueRef], &exception);
175 if (exception)
176 continue;
177 }
178
179 return result;
180}
181
182static JSValue *jsValueWithDictionaryInContext(NSDictionary *dictionary, JSContext *context)
183{
184 JSValueRef exception = 0;
185 JSValue *result = [JSValue valueWithNewObjectInContext:context];
186 JSObjectRef resultObject = JSValueToObject([context JSGlobalContextRef], [result JSValueRef], &exception);
187 if (exception)
188 return [JSValue valueWithUndefinedInContext:context];
189
190 for (id key in [dictionary keyEnumerator]) {
191 if (![key isKindOfClass:[NSString class]])
192 continue;
193
194 JSValue *value = jsValueWithValueInContext([dictionary objectForKey:key], context);
195 if (!value)
196 continue;
197
198 auto name = OpaqueJSString::tryCreate(key);
199 JSObjectSetProperty([context JSGlobalContextRef], resultObject, name.get(), [value JSValueRef], 0, &exception);
200 if (exception)
201 continue;
202 }
203
204 return result;
205}

207static JSValue *jsValueWithAVMetadataItemInContext(AVMetadataItem *item, JSContext *context)
208{
209 return jsValueWithDictionaryInContext(NSDictionaryWithAVMetadataItem(item).get(), context);

212static RetainPtr<NSDictionary> NSDictionaryWithAVMetadataItem(AVMetadataItem *item)
213{
214 auto dictionary = adoptNS([[NSMutableDictionary alloc] init]);
215 NSDictionary *extras = [item extraAttributes];
216
217 for (id key in [extras keyEnumerator]) {
218 if (![key isKindOfClass:[NSString class]])
219 continue;
220 id value = [extras objectForKey:key];
221 NSString *keyString = key;
222
223 if ([key isEqualToString:@"MIMEtype"])
224 keyString = @"type";
225 else if ([key isEqualToString:@"dataTypeNamespace"] || [key isEqualToString:@"pictureType"])
226 continue;
227 else if ([key isEqualToString:@"dataType"]) {
228 id dataTypeNamespace = [extras objectForKey:@"dataTypeNamespace"];
229 if (!dataTypeNamespace || ![dataTypeNamespace isKindOfClass:[NSString class]] || ![dataTypeNamespace isEqualToString:@"org.iana.media-type"])
230 continue;
231 keyString = @"type";
232 } else if ([value isKindOfClass:[NSString class]]) {
233 if (![value length])
234 continue;
235 keyString = [key lowercaseString];
236 }
237
238 [dictionary setObject:value forKey:keyString];
239 }
240
241 if (item.key)
242 [dictionary setObject:item.key forKey:@"key"];
243
244 if (item.locale)
245 [dictionary setObject:item.locale forKey:@"locale"];
246
247 if (item.value)
248 [dictionary setObject:item.value forKey:@"data"];
249
250 return WTFMove(dictionary);
251}
