26#import "config.h"
27#import "UserActivity.h"

29namespace WebCore {
30
31#if HAVE(NS_ACTIVITY)
32
33UserActivity::Impl::Impl(const char* description)
34 : m_description([NSString stringWithUTF8String:description])
35{
36}
37
38void UserActivity::Impl::beginActivity()
39{
40 if (!m_activity) {
41 NSActivityOptions options = (NSActivityUserInitiatedAllowingIdleSystemSleep | NSActivityLatencyCritical) & ~(NSActivitySuddenTerminationDisabled | NSActivityAutomaticTerminationDisabled);
42 m_activity = [[NSProcessInfo processInfo] beginActivityWithOptions:options reason:m_description.get()];
43 }
44}
45
46void UserActivity::Impl::endActivity()
47{
48 [[NSProcessInfo processInfo] endActivity:m_activity.get()];
49 m_activity.clear();
50}
