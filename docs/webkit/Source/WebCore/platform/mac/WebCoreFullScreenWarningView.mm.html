26#import "config.h"
27#import "WebCoreFullScreenWarningView.h"
28
29#if PLATFORM(MAC) && ENABLE(FULLSCREEN_API)
30
31#import "LocalizedStrings.h"
32#import <wtf/text/WTFString.h>
33
34static const CGFloat WarningViewTextWhite = 0.9;
35static const CGFloat WarningViewTextAlpha = 1;
36static const CGFloat WarningViewTextSize = 48;
37static const CGFloat WarningViewPadding = 20;
38static const CGFloat WarningViewCornerRadius = 10;
39static const CGFloat WarningViewBorderWhite = 0.9;
40static const CGFloat WarningViewBorderAlpha = 0.2;
41static const CGFloat WarningViewBackgroundWhite = 0.1;
42static const CGFloat WarningViewBackgroundAlpha = 0.9;
43static const CGFloat WarningViewShadowWhite = 0.1;
44static const CGFloat WarningViewShadowAlpha = 1;
45static const NSSize WarningViewShadowOffset = {0, -2};
46static const CGFloat WarningViewShadowRadius = 5;
47
48@implementation WebCoreFullScreenWarningView
49
50- (id)initWithTitle:(NSString*)title
51{
52 self = [super initWithFrame:NSZeroRect];
53 if (!self)
54 return nil;
55
56 [self setAutoresizingMask:(NSViewMinXMargin | NSViewMaxXMargin | NSViewMinYMargin | NSViewMaxYMargin)];
57 [self setBoxType:NSBoxCustom];
58 [self setTitlePosition:NSNoTitle];
59
60 _textField = adoptNS([[NSTextField alloc] initWithFrame:NSZeroRect]);
61 [_textField.get() setEditable:NO];
62 [_textField.get() setSelectable:NO];
63 [_textField.get() setBordered:NO];
64 [_textField.get() setDrawsBackground:NO];
65
66 NSFont* textFont = [NSFont boldSystemFontOfSize:WarningViewTextSize];
67 NSColor* textColor = [NSColor colorWithCalibratedWhite:WarningViewTextWhite alpha:WarningViewTextAlpha];
68 RetainPtr<NSDictionary> attributes = adoptNS([[NSDictionary alloc] initWithObjectsAndKeys:
69 textFont, NSFontAttributeName,
70 textColor, NSForegroundColorAttributeName,
71 nil]);
72 RetainPtr<NSAttributedString> text = adoptNS([[NSAttributedString alloc] initWithString:title attributes:attributes.get()]);
73 [_textField.get() setAttributedStringValue:text.get()];
74 [_textField.get() sizeToFit];
75 NSRect textFieldFrame = [_textField.get() frame];
76 NSSize frameSize = textFieldFrame.size;
77 frameSize.width += WarningViewPadding * 2;
78 frameSize.height += WarningViewPadding * 2;
79 [self setFrameSize:frameSize];
80
81 textFieldFrame.origin = NSMakePoint(
82 (frameSize.width - textFieldFrame.size.width) / 2,
83 (frameSize.height - textFieldFrame.size.height) / 2);
84
85 // Offset the origin by the font's descender, to center the text field about the baseline:
86 textFieldFrame.origin.y += [[_textField.get() font] descender];
87
88 [_textField.get() setFrame:NSIntegralRect(textFieldFrame)];
89 [self addSubview:_textField.get()];
90
91 NSColor* backgroundColor = [NSColor colorWithCalibratedWhite:WarningViewBackgroundWhite alpha:WarningViewBackgroundAlpha];
92 [self setFillColor:backgroundColor];
93 [self setCornerRadius:WarningViewCornerRadius];
94
95 NSColor* borderColor = [NSColor colorWithCalibratedWhite:WarningViewBorderWhite alpha:WarningViewBorderAlpha];
96 [self setBorderColor:borderColor];
97
98 RetainPtr<NSShadow> shadow = adoptNS([[NSShadow alloc] init]);
99 RetainPtr<NSColor> shadowColor = [NSColor colorWithCalibratedWhite:WarningViewShadowWhite alpha:WarningViewShadowAlpha];
100 [shadow.get() setShadowColor:shadowColor.get()];
101 [shadow.get() setShadowOffset:WarningViewShadowOffset];
102 [shadow.get() setShadowBlurRadius:WarningViewShadowRadius];
103 [self setShadow:shadow.get()];
104
105 return self;
106}
107@end
108
109#endif // PLATFORM(MAC) && ENABLE(FULLSCREEN_API)
