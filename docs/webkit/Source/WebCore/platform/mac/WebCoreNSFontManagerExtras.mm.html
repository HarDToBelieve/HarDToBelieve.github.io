37static NSFont *firstFontConversionSpecimen(NSFontManager *fontManager)
38{
39 const double standardWeight = 5.;
40 return [fontManager fontWithFamily:@"Helvetica" traits:0 weight:standardWeight size:10.];
41}
42
43static NSFont *secondFontConversionSpecimen(NSFontManager *fontManager)
44{
45 const double standardBoldWeight = 9.;
46 return [fontManager fontWithFamily:@"Times" traits:NSFontItalicTrait weight:standardBoldWeight size:12.];
47}
48
49static FontChanges computedFontChanges(NSFontManager *fontManager, NSFont *originalFontA, NSFont *convertedFontA, NSFont *convertedFontB)
50{
51 if (!convertedFontA || !convertedFontB)
52 return { };
53
54 // Since there's still no way to directly ask NSFontManager what style change it's going
55 // to do we instead pass two "specimen" fonts to it and let it change them. We then deduce
56 // what style change it was doing by looking at what happened to each of the two fonts.
57 // So if it was making the text bold, both fonts will be bold after the fact.
58 // This logic was originally from WebHTMLView, and is now in WebCore so that it is shared
59 // between WebKitLegacy and WebKit.
60 const double minimumBoldWeight = 7.;
61
62 NSString *convertedFamilyNameA = convertedFontA.familyName;
63 NSString *convertedFamilyNameB = convertedFontB.familyName;
64
65 auto convertedPointSizeA = convertedFontA.pointSize;
66 auto convertedPointSizeB = convertedFontB.pointSize;
67
68 auto convertedFontWeightA = [fontManager weightOfFont:convertedFontA];
69 auto convertedFontWeightB = [fontManager weightOfFont:convertedFontB];
70
71 bool convertedFontAIsItalic = !!([fontManager traitsOfFont:convertedFontA] & NSItalicFontMask);
72 bool convertedFontBIsItalic = !!([fontManager traitsOfFont:convertedFontB] & NSItalicFontMask);
73
74 bool convertedFontAIsBold = convertedFontWeightA > minimumBoldWeight;
75
76 FontChanges changes;
77 if ([convertedFamilyNameA isEqualToString:convertedFamilyNameB]) {
78 changes.setFontName(convertedFontA.fontName);
79 changes.setFontFamily(convertedFamilyNameA);
80 }
81
82 int originalPointSizeA = originalFontA.pointSize;
83 if (convertedPointSizeA == convertedPointSizeB)
84 changes.setFontSize(convertedPointSizeA);
85 else if (convertedPointSizeA < originalPointSizeA)
86 changes.setFontSizeDelta(-1);
87 else if (convertedPointSizeA > originalPointSizeA)
88 changes.setFontSizeDelta(1);
89
90 if (convertedFontWeightA == convertedFontWeightB)
91 changes.setBold(convertedFontAIsBold);
92
93 if (convertedFontAIsItalic == convertedFontBIsItalic)
94 changes.setItalic(convertedFontAIsItalic);
95
96 return changes;
97}
98
99FontChanges computedFontChanges(NSFontManager *fontManager)
100{
101 NSFont *originalFontA = firstFontConversionSpecimen(fontManager);
102 return computedFontChanges(fontManager, originalFontA, [fontManager convertFont:originalFontA], [fontManager convertFont:secondFontConversionSpecimen(fontManager)]);
103}

105FontAttributeChanges computedFontAttributeChanges(NSFontManager *fontManager, id attributeConverter)
106{
107 FontAttributeChanges changes;
108
109 auto shadow = adoptNS([[NSShadow alloc] init]);
110 [shadow setShadowOffset:NSMakeSize(1, 1)];
111
112 NSFont *originalFontA = firstFontConversionSpecimen(fontManager);
113 NSDictionary *originalAttributesA = @{ NSFontAttributeName : originalFontA };
114 NSDictionary *originalAttributesB = @{
115 NSBackgroundColorAttributeName : NSColor.blackColor,
116 NSFontAttributeName : secondFontConversionSpecimen(fontManager),
117 NSForegroundColorAttributeName : NSColor.whiteColor,
118 NSShadowAttributeName : shadow.get(),
119 NSStrikethroughStyleAttributeName : @(NSUnderlineStyleSingle),
120 NSSuperscriptAttributeName : @1,
121 NSUnderlineStyleAttributeName : @(NSUnderlineStyleSingle)
122 };
123
124 NSDictionary *convertedAttributesA = [attributeConverter convertAttributes:originalAttributesA];
125 NSDictionary *convertedAttributesB = [attributeConverter convertAttributes:originalAttributesB];
126
127 NSColor *convertedBackgroundColorA = [convertedAttributesA objectForKey:NSBackgroundColorAttributeName];
128 if (convertedBackgroundColorA == [convertedAttributesB objectForKey:NSBackgroundColorAttributeName])
129 changes.setBackgroundColor(colorFromNSColor(convertedBackgroundColorA ?: NSColor.clearColor));
130
131 changes.setFontChanges(computedFontChanges(fontManager, originalFontA, [convertedAttributesA objectForKey:NSFontAttributeName], [convertedAttributesB objectForKey:NSFontAttributeName]));
132
133 NSColor *convertedForegroundColorA = [convertedAttributesA objectForKey:NSForegroundColorAttributeName];
134 if (convertedForegroundColorA == [convertedAttributesB objectForKey:NSForegroundColorAttributeName])
135 changes.setForegroundColor(colorFromNSColor(convertedForegroundColorA ?: NSColor.blackColor));
136
137 NSShadow *convertedShadow = [convertedAttributesA objectForKey:NSShadowAttributeName];
138 if (convertedShadow) {
139 FloatSize offset { static_cast<float>(convertedShadow.shadowOffset.width), static_cast<float>(convertedShadow.shadowOffset.height) };
140 changes.setShadow({ colorFromNSColor(convertedShadow.shadowColor ?: NSColor.blackColor), offset, convertedShadow.shadowBlurRadius });
141 } else if (![convertedAttributesB objectForKey:NSShadowAttributeName])
142 changes.setShadow({ });
143
144 int convertedSuperscriptA = [[convertedAttributesA objectForKey:NSSuperscriptAttributeName] intValue];
145 if (convertedSuperscriptA == [[convertedAttributesB objectForKey:NSSuperscriptAttributeName] intValue]) {
146 if (convertedSuperscriptA > 0)
147 changes.setVerticalAlign(VerticalAlignChange::Superscript);
148 else if (convertedSuperscriptA < 0)
149 changes.setVerticalAlign(VerticalAlignChange::Subscript);
150 else
151 changes.setVerticalAlign(VerticalAlignChange::Baseline);
152 }
153
154 int convertedStrikeThroughA = [[convertedAttributesA objectForKey:NSStrikethroughStyleAttributeName] intValue];
155 if (convertedStrikeThroughA == [[convertedAttributesB objectForKey:NSStrikethroughStyleAttributeName] intValue])
156 changes.setStrikeThrough(convertedStrikeThroughA != NSUnderlineStyleNone);
157
158 int convertedUnderlineA = [[convertedAttributesA objectForKey:NSUnderlineStyleAttributeName] intValue];
159 if (convertedUnderlineA == [[convertedAttributesB objectForKey:NSUnderlineStyleAttributeName] intValue])
160 changes.setUnderline(convertedUnderlineA != NSUnderlineStyleNone);
161
162 return changes;
163}
