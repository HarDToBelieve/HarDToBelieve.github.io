40// This is like the isKindOfClass: method, bypassing it to get the correct answer for our purposes even for classes that override it.
41// At the time of this writing, that included WebKit's WKObject class.
42static bool safeIsKindOfClass(id object, Class testClass)
43{
44 if (!object)
45 return false;
46 for (auto ancestorClass = object_getClass(object); ancestorClass; ancestorClass = class_getSuperclass(ancestorClass)) {
47 if (ancestorClass == testClass)
48 return true;
49 }
50 return false;
51}

55bool WebCoreObjCScheduleDeallocateOnMainThread(Class deallocMethodClass, id object)
56{
57 ASSERT(safeIsKindOfClass(object, deallocMethodClass));
58
59 if (isMainThread())
60 return false;
61
62 callOnMainThread([deallocMethodClass, object] {
63 auto deallocSelector = sel_registerName("dealloc");
64 wtfCallIMP<void>(method_getImplementation(class_getInstanceMethod(deallocMethodClass, deallocSelector)), object, deallocSelector);
65 });
66
67 return true;
68}
69
70bool WebCoreObjCScheduleDeallocateOnMainRunLoop(Class deallocMethodClass, id object)
71{
72 ASSERT(safeIsKindOfClass(object, deallocMethodClass));
73
74 if (isMainRunLoop())
75 return false;
76
77 callOnMainRunLoop([deallocMethodClass, object] {
78 auto deallocSelector = sel_registerName("dealloc");
79 wtfCallIMP<void>(method_getImplementation(class_getInstanceMethod(deallocMethodClass, deallocSelector)), object, deallocSelector);
80 });
81
82 return true;
83}
