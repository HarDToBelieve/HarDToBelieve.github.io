63static void safeRemoveFromSuperview(NSView *view)
64{
65 // If the view is the first responder, then set the window's first responder to nil so
66 // we don't leave the window pointing to a view that's no longer in it.
67 NSWindow *window = [view window];
68 NSResponder *firstResponder = [window firstResponder];
69 if ([firstResponder isKindOfClass:[NSView class]] && [(NSView *)firstResponder isDescendantOf:view])
70 [window makeFirstResponder:nil];
71
72 // Suppress the resetting of drag margins since we know we can't affect them.
73 BOOL resetDragMargins = [window _needsToResetDragMargins];
74 [window _setNeedsToResetDragMargins:NO];
75 [view removeFromSuperview];
76 [window _setNeedsToResetDragMargins:resetDragMargins];
77}

89void Widget::setFocus(bool focused)
90{
91 if (!focused)
92 return;
93
94 Frame* frame = Frame::frameForWidget(*this);
95 if (!frame)
96 return;
97
98 BEGIN_BLOCK_OBJC_EXCEPTIONS
99
100 // Call this even when there is no platformWidget(). WK2 will focus on the widget in the UIProcess.
101 NSView *view = [platformWidget() _webcore_effectiveFirstResponder];
102 if (Page* page = frame->page())
103 page->chrome().focusNSView(view);
104
105 END_BLOCK_OBJC_EXCEPTIONS
106}

152void Widget::setFrameRect(const IntRect& rect)
153{
154 m_frame = rect;
155
156 BEGIN_BLOCK_OBJC_EXCEPTIONS
157
158 NSView *outerView = getOuterView();
159 if (!outerView)
160 return;
161
162 // Take a reference to this Widget, because sending messages to outerView can invoke arbitrary
163 // code including recalc style/layout, which can deref it.
164 Ref<Widget> protectedThis(*this);
165
166 NSRect frame = rect;
167 if (!NSEqualRects(frame, outerView.frame)) {
168 outerView.frame = frame;
169 outerView.needsDisplay = NO;
170 }
171
172 END_BLOCK_OBJC_EXCEPTIONS
173}

175NSView *Widget::getOuterView() const
176{
177 NSView *view = platformWidget();
178
179 // If this widget's view is a WebCoreFrameScrollView then we
180 // resize its containing view, a WebFrameView.
181 if ([view conformsToProtocol:@protocol(WebCoreFrameScrollView)]) {
182 view = [view superview];
183 ASSERT(view);
184 }
185
186 return view;
187}

189void Widget::paint(GraphicsContext& p, const IntRect& r, SecurityOriginPaintPolicy, EventRegionContext*)
190{
191 if (p.paintingDisabled())
192 return;
193 NSView *view = getOuterView();
194
195 // We don't want to paint the view at all if it's layer backed, because then we'll end up
196 // with multiple copies of the view contents, one in the view's layer itself and one in the
197 // WebHTMLView's backing store (either a layer or the window backing store).
198 if (view.layer) {
199#if PLATFORM(MAC)
200 // However, Quicken Essentials has a plug-in that depends on drawing to update the layer (see <rdar://problem/15221231>).
201 if (!MacApplication::isQuickenEssentials())
202#endif
203 return;
204 }
205
206 // Take a reference to this Widget, because sending messages to the views can invoke arbitrary
207 // code, which can deref it.
208 Ref<Widget> protectedThis(*this);
209
210 NSGraphicsContext *currentContext = [NSGraphicsContext currentContext];
211 ALLOW_DEPRECATED_DECLARATIONS_BEGIN
212 if (currentContext == [[view window] graphicsContext] || ![currentContext isDrawingToScreen]) {
213 ALLOW_DEPRECATED_DECLARATIONS_END
214 // This is the common case of drawing into a window or an inclusive layer, or printing.
215 BEGIN_BLOCK_OBJC_EXCEPTIONS
216 [view displayRectIgnoringOpacity:[view convertRect:r fromView:[view superview]]];
217 END_BLOCK_OBJC_EXCEPTIONS
218 return;
219 }
220
221 // This is the case of drawing into a bitmap context other than a window backing store. It gets hit beneath
222 // -cacheDisplayInRect:toBitmapImageRep:, and when painting into compositing layers.
223
224 // Transparent subframes are in fact implemented with scroll views that return YES from -drawsBackground (whenever the WebView
225 // itself is in drawsBackground mode). In the normal drawing code path, the scroll views are never asked to draw the background,
226 // so this is not an issue, but in this code path they are, so the following code temporarily turns background drwaing off.
227 NSView *innerView = platformWidget();
228 NSScrollView *scrollView = 0;
229 if ([innerView conformsToProtocol:@protocol(WebCoreFrameScrollView)]) {
230 ASSERT([innerView isKindOfClass:[NSScrollView class]]);
231 NSScrollView *scrollView = static_cast<NSScrollView *>(innerView);
232 // -copiesOnScroll will return NO whenever the content view is not fully opaque.
233 ALLOW_DEPRECATED_DECLARATIONS_BEGIN
234 if ([scrollView drawsBackground] && ![[scrollView contentView] copiesOnScroll])
235 [scrollView setDrawsBackground:NO];
236 else
237 scrollView = 0;
238 ALLOW_DEPRECATED_DECLARATIONS_END
239 }
240
241 CGContextRef cgContext = p.platformContext();
242 ASSERT(cgContext == [currentContext CGContext]);
243 CGContextSaveGState(cgContext);
244
245 NSRect viewFrame = [view frame];
246 NSRect viewBounds = [view bounds];
247
248 // Set up the translation and (flipped) orientation of the graphics context. In normal drawing, AppKit does it as it descends down
249 // the view hierarchy. Since Widget::paint is always called with a context that has a flipped coordinate system, and
250 // -[NSView displayRectIgnoringOpacity:inContext:] expects an unflipped context we always flip here.
251 CGContextTranslateCTM(cgContext, viewFrame.origin.x - viewBounds.origin.x, viewFrame.origin.y + viewFrame.size.height + viewBounds.origin.y);
252 CGContextScaleCTM(cgContext, 1, -1);
253
254 BEGIN_BLOCK_OBJC_EXCEPTIONS
255 {
256 NSGraphicsContext *nsContext = [NSGraphicsContext graphicsContextWithCGContext:cgContext flipped:NO];
257 [view displayRectIgnoringOpacity:[view convertRect:r fromView:[view superview]] inContext:nsContext];
258 }
259 END_BLOCK_OBJC_EXCEPTIONS
260
261 CGContextRestoreGState(cgContext);
262
263 if (scrollView)
264 [scrollView setDrawsBackground:YES];
265}

267void Widget::setIsSelected(bool isSelected)
268{
269 NSView *view = platformWidget();
270
271 BEGIN_BLOCK_OBJC_EXCEPTIONS
272 if ([view respondsToSelector:@selector(webPlugInSetIsSelected:)])
273 [view webPlugInSetIsSelected:isSelected];
274 else if ([view respondsToSelector:@selector(setIsSelected:)])
275 [view setIsSelected:isSelected];
276 END_BLOCK_OBJC_EXCEPTIONS
277}
