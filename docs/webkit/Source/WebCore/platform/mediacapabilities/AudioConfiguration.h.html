26#pragma once
27
28#include <wtf/text/WTFString.h>
29
30namespace WebCore {
31
32struct AudioConfiguration {
33 String contentType;
34 String channels;
35 std::optional<uint64_t> bitrate;
36 std::optional<uint32_t> samplerate;
37 std::optional<bool> spatialRendering;
38
39 template<class Encoder> void encode(Encoder&) const;
40 template<class Decoder> static std::optional<AudioConfiguration> decode(Decoder&);
41};
42
43template<class Encoder>
44void AudioConfiguration::encode(Encoder& encoder) const
45{
46 encoder << contentType;
47 encoder << channels;
48 encoder << bitrate;
49 encoder << samplerate;
50 encoder << spatialRendering;
51}
52
53template<class Decoder>
54std::optional<AudioConfiguration> AudioConfiguration::decode(Decoder& decoder)
55{
56 std::optional<String> contentType;
57 decoder >> contentType;
58 if (!contentType)
59 return std::nullopt;
60
61 std::optional<String> channels;
62 decoder >> channels;
63 if (!channels)
64 return std::nullopt;
65
66 std::optional<std::optional<uint64_t>> bitrate;
67 decoder >> bitrate;
68 if (!bitrate)
69 return std::nullopt;
70
71 std::optional<std::optional<uint32_t>> sampleRate;
72 decoder >> sampleRate;
73 if (!sampleRate)
74 return std::nullopt;
75
76 std::optional<std::optional<bool>> spatialRendering;
77 decoder >> spatialRendering;
78 if (!spatialRendering)
79 return std::nullopt;
80
81 return {{
82 *contentType,
83 *channels,
84 *bitrate,
85 *sampleRate,
86 *spatialRendering
87 }};
88}

90} // namespace WebCore
