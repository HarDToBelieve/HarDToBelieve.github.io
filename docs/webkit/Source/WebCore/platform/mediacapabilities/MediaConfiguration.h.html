26#pragma once
27
28#include "AudioConfiguration.h"
29#include "VideoConfiguration.h"
30
31namespace WebCore {
32
33struct MediaConfiguration {
34 std::optional<VideoConfiguration> video;
35 std::optional<AudioConfiguration> audio;
36
37 template<class Encoder> void encode(Encoder&) const;
38 template<class Decoder> static std::optional<MediaConfiguration> decode(Decoder&);
39};
40
41template<class Encoder>
42void MediaConfiguration::encode(Encoder& encoder) const
43{
44 encoder << video;
45 encoder << audio;
46}
47
48template<class Decoder>
49std::optional<MediaConfiguration> MediaConfiguration::decode(Decoder& decoder)
50{
51 std::optional<std::optional<VideoConfiguration>> video;
52 decoder >> video;
53 if (!video)
54 return std::nullopt;
55
56 std::optional<std::optional<AudioConfiguration>> audio;
57 decoder >> audio;
58 if (!audio)
59 return std::nullopt;
60
61 return {{
62 *video,
63 *audio,
64 }};
65}

67} // namespace WebCore
