26#pragma once
27
28#include "ColorGamut.h"
29#include "HdrMetadataType.h"
30#include "TransferFunction.h"
31#include <wtf/text/WTFString.h>
32
33namespace WebCore {
34
35struct VideoConfiguration {
36 String contentType;
37 uint32_t width;
38 uint32_t height;
39 uint64_t bitrate;
40 double framerate;
41 std::optional<bool> alphaChannel;
42 std::optional<ColorGamut> colorGamut;
43 std::optional<HdrMetadataType> hdrMetadataType;
44 std::optional<TransferFunction> transferFunction;
45
46 template<class Encoder> void encode(Encoder&) const;
47 template<class Decoder> static std::optional<VideoConfiguration> decode(Decoder&);
48};
49
50template<class Encoder>
51void VideoConfiguration::encode(Encoder& encoder) const
52{
53 encoder << contentType;
54 encoder << width;
55 encoder << height;
56 encoder << bitrate;
57 encoder << framerate;
58 encoder << alphaChannel;
59 encoder << colorGamut;
60 encoder << hdrMetadataType;
61 encoder << transferFunction;
62}
63
64template<class Decoder>
65std::optional<VideoConfiguration> VideoConfiguration::decode(Decoder& decoder)
66{
67 std::optional<String> contentType;
68 decoder >> contentType;
69 if (!contentType)
70 return std::nullopt;
71
72 std::optional<uint32_t> width;
73 decoder >> width;
74 if (!width)
75 return std::nullopt;
76
77 std::optional<uint32_t> height;
78 decoder >> height;
79 if (!height)
80 return std::nullopt;
81
82 std::optional<uint64_t> bitrate;
83 decoder >> bitrate;
84 if (!bitrate)
85 return std::nullopt;
86
87 std::optional<double> framerate;
88 decoder >> framerate;
89 if (!framerate)
90 return std::nullopt;
91
92 std::optional<std::optional<bool>> alphaChannel;
93 decoder >> alphaChannel;
94 if (!alphaChannel)
95 return std::nullopt;
96
97 std::optional<std::optional<ColorGamut>> colorGamut;
98 decoder >> colorGamut;
99 if (!colorGamut)
100 return std::nullopt;
101
102 std::optional<std::optional<HdrMetadataType>> hdrMetadataType;
103 decoder >> hdrMetadataType;
104 if (!hdrMetadataType)
105 return std::nullopt;
106
107 std::optional<std::optional<TransferFunction>> transferFunction;
108 decoder >> transferFunction;
109 if (!transferFunction)
110 return std::nullopt;
111
112 return {{
113 *contentType,
114 *width,
115 *height,
116 *bitrate,
117 *framerate,
118 *alphaChannel,
119 *colorGamut,
120 *hdrMetadataType,
121 *transferFunction,
122 }};
123}
