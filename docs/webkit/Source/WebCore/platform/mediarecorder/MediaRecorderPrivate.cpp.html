35constexpr unsigned LargeAudioBitRate = 192000;
36constexpr unsigned LargeVideoBitRate = 10000000;
37
38MediaRecorderPrivate::AudioVideoSelectedTracks MediaRecorderPrivate::selectTracks(MediaStreamPrivate& stream)
39{
40 AudioVideoSelectedTracks selectedTracks;
41 stream.forEachTrack([&](auto& track) {
42 if (track.ended())
43 return;
44 switch (track.type()) {
45 case RealtimeMediaSource::Type::Video: {
46 auto& settings = track.settings();
47 if (!selectedTracks.videoTrack && settings.supportsWidth() && settings.supportsHeight())
48 selectedTracks.videoTrack = &track;
49 break;
50 }
51 case RealtimeMediaSource::Type::Audio:
52 if (!selectedTracks.audioTrack)
53 selectedTracks.audioTrack = &track;
54 break;
55 case RealtimeMediaSource::Type::None:
56 break;
57 }
58 });
59 return selectedTracks;
60}
61
62void MediaRecorderPrivate::checkTrackState(const MediaStreamTrackPrivate& track)
63{
64 if (&track.source() == m_audioSource.get()) {
65 m_shouldMuteAudio = track.muted() || !track.enabled();
66 return;
67 }
68 if (&track.source() == m_videoSource.get())
69 m_shouldMuteVideo = track.muted() || !track.enabled();
70}
71
72void MediaRecorderPrivate::stop(CompletionHandler<void()>&& completionHandler)
73{
74 setAudioSource(nullptr);
75 setVideoSource(nullptr);
76 stopRecording(WTFMove(completionHandler));
77}
78
79void MediaRecorderPrivate::pause(CompletionHandler<void()>&& completionHandler)
80{
81 ASSERT(!m_pausedAudioSource);
82 ASSERT(!m_pausedVideoSource);
83
84 m_pausedAudioSource = m_audioSource;
85 m_pausedVideoSource = m_videoSource;
86
87 setAudioSource(nullptr);
88 setVideoSource(nullptr);
89
90 pauseRecording(WTFMove(completionHandler));
91}
92
93void MediaRecorderPrivate::resume(CompletionHandler<void()>&& completionHandler)
94{
95 ASSERT(m_pausedAudioSource || m_pausedVideoSource);
96
97 setAudioSource(WTFMove(m_pausedAudioSource));
98 setVideoSource(WTFMove(m_pausedVideoSource));
99
100 resumeRecording(WTFMove(completionHandler));
101}

103void MediaRecorderPrivate::updateOptions(MediaRecorderPrivateOptions& options)
104{
105 // FIXME: Add support for options.bitsPerSecond.
106 if (!options.audioBitsPerSecond)
107 options.audioBitsPerSecond = LargeAudioBitRate;
108 if (!options.videoBitsPerSecond)
109 options.videoBitsPerSecond = LargeVideoBitRate;
110}
