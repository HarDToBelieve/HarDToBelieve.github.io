41class AudioStreamDescription;

48struct MediaRecorderPrivateOptions;

50class MediaRecorderPrivate
51 : public RealtimeMediaSource::AudioSampleObserver
52 , public RealtimeMediaSource::VideoSampleObserver {
53public:
54 ~MediaRecorderPrivate();
55
56 struct AudioVideoSelectedTracks {
57 MediaStreamTrackPrivate* audioTrack { nullptr };
58 MediaStreamTrackPrivate* videoTrack { nullptr };
59 };
60 WEBCORE_EXPORT static AudioVideoSelectedTracks selectTracks(MediaStreamPrivate&);
61
62 using FetchDataCallback = CompletionHandler<void(RefPtr<SharedBuffer>&&, const String& mimeType, double)>;
63 virtual void fetchData(FetchDataCallback&&) = 0;
64 virtual const String& mimeType() const = 0;
65
66 void stop(CompletionHandler<void()>&&);
67 void pause(CompletionHandler<void()>&&);
68 void resume(CompletionHandler<void()>&&);
69
70 using StartRecordingCallback = CompletionHandler<void(ExceptionOr<String>&&, unsigned, unsigned)>;
71 virtual void startRecording(StartRecordingCallback&& callback) { callback(String(mimeType()), 0, 0); }
72
73 void trackMutedChanged(MediaStreamTrackPrivate& track) { checkTrackState(track); }
74 void trackEnabledChanged(MediaStreamTrackPrivate& track) { checkTrackState(track); }
75
76 static void updateOptions(MediaRecorderPrivateOptions&);
77
78protected:
79 void setAudioSource(RefPtr<RealtimeMediaSource>&&);
80 void setVideoSource(RefPtr<RealtimeMediaSource>&&);
81
82 void checkTrackState(const MediaStreamTrackPrivate&);
83
84 bool shouldMuteAudio() const { return m_shouldMuteAudio; }
85 bool shouldMuteVideo() const { return m_shouldMuteVideo; }
86
87private:
88 virtual void stopRecording(CompletionHandler<void()>&&) = 0;
89 virtual void pauseRecording(CompletionHandler<void()>&&) = 0;
90 virtual void resumeRecording(CompletionHandler<void()>&&) = 0;
91
92private:
93 bool m_shouldMuteAudio { false };
94 bool m_shouldMuteVideo { false };
95 RefPtr<RealtimeMediaSource> m_audioSource;
96 RefPtr<RealtimeMediaSource> m_videoSource;
97 RefPtr<RealtimeMediaSource> m_pausedAudioSource;
98 RefPtr<RealtimeMediaSource> m_pausedVideoSource;
99};
100
101inline void MediaRecorderPrivate::setAudioSource(RefPtr<RealtimeMediaSource>&& audioSource)
102{
103 if (m_audioSource)
104 m_audioSource->removeAudioSampleObserver(*this);
105
106 m_audioSource = WTFMove(audioSource);
107
108 if (m_audioSource)
109 m_audioSource->addAudioSampleObserver(*this);
110}
111
112inline void MediaRecorderPrivate::setVideoSource(RefPtr<RealtimeMediaSource>&& videoSource)
113{
114 if (m_videoSource)
115 m_videoSource->removeVideoSampleObserver(*this);
116
117 m_videoSource = WTFMove(videoSource);
118
119 if (m_videoSource)
120 m_videoSource->addVideoSampleObserver(*this);
121}
122
123inline MediaRecorderPrivate::~MediaRecorderPrivate()
124{
125 // Subclasses should stop observing sonner than here. Otherwise they might be called from a background thread while half destroyed
126 ASSERT(!m_audioSource);
127 ASSERT(!m_videoSource);
128 if (m_audioSource)
129 m_audioSource->removeAudioSampleObserver(*this);
130 if (m_videoSource)
131 m_videoSource->removeVideoSampleObserver(*this);
132}
