38MediaRecorderPrivateMock::MediaRecorderPrivateMock(MediaStreamPrivate& stream)
39{
40 auto selectedTracks = MediaRecorderPrivate::selectTracks(stream);
41 if (selectedTracks.audioTrack) {
42 m_audioTrackID = selectedTracks.audioTrack->id();
43 setAudioSource(&selectedTracks.audioTrack->source());
44 }
45 if (selectedTracks.videoTrack) {
46 m_videoTrackID = selectedTracks.videoTrack->id();
47 setVideoSource(&selectedTracks.videoTrack->source());
48 }
49}

78void MediaRecorderPrivateMock::audioSamplesAvailable(const WTF::MediaTime&, const PlatformAudioData&, const AudioStreamDescription&, size_t)
79{
80 // Heap allocations are forbidden on the audio thread for performance reasons so we need to
81 // explicitly allow the following allocation(s).
82 DisableMallocRestrictionsForCurrentThreadScope disableMallocRestrictions;
83 Locker locker { m_bufferLock };
84 m_buffer.append("Audio Track ID: ");
85 m_buffer.append(m_audioTrackID);
86 generateMockCounterString();
87}

89void MediaRecorderPrivateMock::generateMockCounterString()
90{
91 m_buffer.append(" Counter: ", ++m_counter, "\r\n---------\r\n");
92}
93
94void MediaRecorderPrivateMock::fetchData(FetchDataCallback&& completionHandler)
95{
96 RefPtr<SharedBuffer> buffer;
97 {
98 Locker locker { m_bufferLock };
99 Vector<uint8_t> value { m_buffer.characters8(), m_buffer.length() };
100 m_buffer.clear();
101 buffer = SharedBuffer::create(WTFMove(value));
102 }
103
104 // Delay calling the completion handler a bit to mimick real writer behavior.
105 Timer::schedule(50_ms, [completionHandler = WTFMove(completionHandler), buffer = WTFMove(buffer), mimeType = mimeType(), timeCode = MonotonicTime::now().secondsSinceEpoch().value()]() mutable {
106 completionHandler(WTFMove(buffer), mimeType, timeCode);
107 });
108}
