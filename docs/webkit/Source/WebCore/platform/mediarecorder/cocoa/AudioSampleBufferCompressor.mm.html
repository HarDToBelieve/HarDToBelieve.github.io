67bool AudioSampleBufferCompressor::initialize(CMBufferQueueTriggerCallback callback, void* callbackObject)
68{
69 CMBufferQueueRef inputBufferQueue;
70 if (auto error = PAL::CMBufferQueueCreate(kCFAllocatorDefault, 0, PAL::CMBufferQueueGetCallbacksForUnsortedSampleBuffers(), &inputBufferQueue)) {
71 RELEASE_LOG_ERROR(MediaStream, "AudioSampleBufferCompressor CMBufferQueueCreate for m_inputBufferQueue failed with %d", error);
72 return false;
73 }
74 m_inputBufferQueue = adoptCF(inputBufferQueue);
75
76 CMBufferQueueRef outputBufferQueue;
77 if (auto error = PAL::CMBufferQueueCreate(kCFAllocatorDefault, 0, PAL::CMBufferQueueGetCallbacksForUnsortedSampleBuffers(), &outputBufferQueue)) {
78 RELEASE_LOG_ERROR(MediaStream, "AudioSampleBufferCompressor CMBufferQueueCreate for m_outputBufferQueue failed with %d", error);
79 return false;
80 }
81 m_outputBufferQueue = adoptCF(outputBufferQueue);
82 PAL::CMBufferQueueInstallTrigger(m_outputBufferQueue.get(), callback, callbackObject, kCMBufferQueueTrigger_WhenDataBecomesReady, PAL::kCMTimeZero, NULL);
83
84 m_isEncoding = true;
85 return true;
86}

88void AudioSampleBufferCompressor::finish()
89{
90 m_serialDispatchQueue->dispatchSync([this] {
91 processSampleBuffersUntilLowWaterTime(PAL::kCMTimeInvalid);
92 auto error = PAL::CMBufferQueueMarkEndOfData(m_outputBufferQueue.get());
93 RELEASE_LOG_ERROR_IF(error, MediaStream, "AudioSampleBufferCompressor CMBufferQueueMarkEndOfData failed %d", error);
94 m_isEncoding = false;
95 });
96}

103UInt32 AudioSampleBufferCompressor::defaultOutputBitRate(const AudioStreamBasicDescription& destinationFormat) const
104{
105 if (destinationFormat.mSampleRate >= 44100)
106 return 192000;
107 if (destinationFormat.mSampleRate < 22000)
108 return 32000;
109 return 64000;

112bool AudioSampleBufferCompressor::initAudioConverterForSourceFormatDescription(CMFormatDescriptionRef formatDescription, AudioFormatID outputFormatID)
113{
114 const auto *audioFormatListItem = PAL::CMAudioFormatDescriptionGetRichestDecodableFormat(formatDescription);
115 m_sourceFormat = audioFormatListItem->mASBD;
116
117 memset(&m_destinationFormat, 0, sizeof(AudioStreamBasicDescription));
118 m_destinationFormat.mFormatID = outputFormatID;
119 m_destinationFormat.mSampleRate = m_sourceFormat.mSampleRate;
120 m_destinationFormat.mChannelsPerFrame = m_sourceFormat.mChannelsPerFrame;
121
122 UInt32 size = sizeof(m_destinationFormat);
123 if (auto error = PAL::AudioFormatGetProperty(kAudioFormatProperty_FormatInfo, 0, NULL, &size, &m_destinationFormat)) {
124 RELEASE_LOG_ERROR(MediaStream, "AudioSampleBufferCompressor AudioFormatGetProperty failed with %d", error);
125 return false;
126 }
127
128 AudioConverterRef converter;
129 if (auto error = PAL::AudioConverterNew(&m_sourceFormat, &m_destinationFormat, &converter)) {
130 RELEASE_LOG_ERROR(MediaStream, "AudioSampleBufferCompressor AudioConverterNew failed with %d", error);
131 return false;
132 }
133 m_converter = converter;
134
135 auto cleanupInCaseOfError = makeScopeExit([&] {
136 PAL::AudioConverterDispose(m_converter);
137 m_converter = nullptr;
138 });
139
140 size_t cookieSize = 0;
141 const void *cookie = PAL::CMAudioFormatDescriptionGetMagicCookie(formatDescription, &cookieSize);
142 if (cookieSize) {
143 if (auto error = PAL::AudioConverterSetProperty(m_converter, kAudioConverterDecompressionMagicCookie, (UInt32)cookieSize, cookie)) {
144 RELEASE_LOG_ERROR(MediaStream, "AudioSampleBufferCompressor setting kAudioConverterDecompressionMagicCookie failed with %d", error);
145 return false;
146 }
147 }
148
149 size = sizeof(m_sourceFormat);
150 if (auto error = PAL::AudioConverterGetProperty(m_converter, kAudioConverterCurrentInputStreamDescription, &size, &m_sourceFormat)) {
151 RELEASE_LOG_ERROR(MediaStream, "AudioSampleBufferCompressor getting kAudioConverterCurrentInputStreamDescription failed with %d", error);
152 return false;
153 }
154
155 if (!m_sourceFormat.mBytesPerPacket) {
156 RELEASE_LOG_ERROR(MediaStream, "mBytesPerPacket should not be zero");
157 return false;
158 }
159
160 size = sizeof(m_destinationFormat);
161 if (auto error = PAL::AudioConverterGetProperty(m_converter, kAudioConverterCurrentOutputStreamDescription, &size, &m_destinationFormat)) {
162 RELEASE_LOG_ERROR(MediaStream, "AudioSampleBufferCompressor getting kAudioConverterCurrentOutputStreamDescription failed with %d", error);
163 return false;
164 }
165
166 if (m_destinationFormat.mFormatID == kAudioFormatMPEG4AAC) {
167 bool shouldSetDefaultOutputBitRate = true;
168 if (m_outputBitRate) {
169 auto error = PAL::AudioConverterSetProperty(m_converter, kAudioConverterEncodeBitRate, sizeof(*m_outputBitRate), &m_outputBitRate.value());
170 RELEASE_LOG_ERROR_IF(error, MediaStream, "AudioSampleBufferCompressor setting kAudioConverterEncodeBitRate failed with %d", error);
171 shouldSetDefaultOutputBitRate = !!error;
172 }
173 if (shouldSetDefaultOutputBitRate) {
174 auto outputBitRate = defaultOutputBitRate(m_destinationFormat);
175 size = sizeof(outputBitRate);
176 if (auto error = PAL::AudioConverterSetProperty(m_converter, kAudioConverterEncodeBitRate, size, &outputBitRate))
177 RELEASE_LOG_ERROR(MediaStream, "AudioSampleBufferCompressor setting default kAudioConverterEncodeBitRate failed with %d", error);
178 }
179 }
180
181 if (!m_destinationFormat.mBytesPerPacket) {
182 // If the destination format is VBR, we need to get max size per packet from the converter.
183 size = sizeof(m_maxOutputPacketSize);
184
185 if (auto error = PAL::AudioConverterGetProperty(m_converter, kAudioConverterPropertyMaximumOutputPacketSize, &size, &m_maxOutputPacketSize)) {
186 RELEASE_LOG_ERROR(MediaStream, "AudioSampleBufferCompressor getting kAudioConverterPropertyMaximumOutputPacketSize failed with %d", error);
187 return false;
188 }
189 }
190
191 cleanupInCaseOfError.release();
192
193 auto destinationBufferSize = computeBufferSizeForAudioFormat(m_destinationFormat, m_maxOutputPacketSize, LOW_WATER_TIME_IN_SECONDS);
194 if (m_destinationBuffer.size() < destinationBufferSize)
195 m_destinationBuffer.resize(destinationBufferSize);
196 if (!m_destinationFormat.mBytesPerPacket)
197 m_destinationPacketDescriptions.resize(m_destinationBuffer.capacity() / m_maxOutputPacketSize);
198
199 return true;
200}

202size_t AudioSampleBufferCompressor::computeBufferSizeForAudioFormat(AudioStreamBasicDescription format, UInt32 maxOutputPacketSize, Float32 duration)
203{
204 UInt32 numPackets = (format.mSampleRate * duration) / format.mFramesPerPacket;
205 UInt32 outputPacketSize = format.mBytesPerPacket ? format.mBytesPerPacket : maxOutputPacketSize;
206 UInt32 bufferSize = numPackets * outputPacketSize;
207
208 return bufferSize;
209}

211void AudioSampleBufferCompressor::attachPrimingTrimsIfNeeded(CMSampleBufferRef buffer)
212{
213 using namespace PAL;
214
215 if (CMTIME_IS_INVALID(m_remainingPrimeDuration)) {
216 AudioConverterPrimeInfo primeInfo { 0, 0 };
217 UInt32 size = sizeof(primeInfo);
218
219 if (auto error = AudioConverterGetProperty(m_converter, kAudioConverterPrimeInfo, &size, &primeInfo)) {
220 RELEASE_LOG_ERROR(MediaStream, "AudioSampleBufferCompressor getting kAudioConverterPrimeInfo failed with %d", error);
221 return;
222 }
223
224 m_remainingPrimeDuration = CMTimeMake(primeInfo.leadingFrames, m_destinationFormat.mSampleRate);
225 }
226
227 if (CMTIME_COMPARE_INLINE(kCMTimeZero, <, m_remainingPrimeDuration)) {
228 CMTime sampleDuration = CMSampleBufferGetDuration(buffer);
229 CMTime trimDuration = CMTimeMinimum(sampleDuration, m_remainingPrimeDuration);
230 auto trimAtStartDict = adoptCF(CMTimeCopyAsDictionary(trimDuration, kCFAllocatorDefault));
231 CMSetAttachment(buffer, kCMSampleBufferAttachmentKey_TrimDurationAtStart, trimAtStartDict.get(), kCMAttachmentMode_ShouldPropagate);
232 m_remainingPrimeDuration = CMTimeSubtract(m_remainingPrimeDuration, trimDuration);
233 }
234}

236RetainPtr<NSNumber> AudioSampleBufferCompressor::gradualDecoderRefreshCount()
237{
238 UInt32 delaySize = sizeof(uint32_t);
239 uint32_t originalDelayMode = 0;
240 if (auto error = PAL::AudioConverterGetProperty(m_converter, kAudioCodecPropertyDelayMode, &delaySize, &originalDelayMode)) {
241 RELEASE_LOG_ERROR(MediaStream, "AudioSampleBufferCompressor getting kAudioCodecPropertyDelayMode failed with %d", error);
242 return nil;
243 }
244
245 uint32_t optimalDelayMode = kAudioCodecDelayMode_Optimal;
246 if (auto error = PAL::AudioConverterSetProperty(m_converter, kAudioCodecPropertyDelayMode, delaySize, &optimalDelayMode)) {
247 RELEASE_LOG_ERROR(MediaStream, "AudioSampleBufferCompressor setting kAudioCodecPropertyDelayMode failed with %d", error);
248 return nil;
249 }
250
251 UInt32 primeSize = sizeof(AudioCodecPrimeInfo);
252 AudioCodecPrimeInfo primeInfo { 0, 0 };
253 if (auto error = PAL::AudioConverterGetProperty(m_converter, kAudioCodecPropertyPrimeInfo, &primeSize, &primeInfo)) {
254 RELEASE_LOG_ERROR(MediaStream, "AudioSampleBufferCompressor getting kAudioCodecPropertyPrimeInfo failed with %d", error);
255 return nil;
256 }
257
258 if (auto error = PAL::AudioConverterSetProperty(m_converter, kAudioCodecPropertyDelayMode, delaySize, &originalDelayMode)) {
259 RELEASE_LOG_ERROR(MediaStream, "AudioSampleBufferCompressor setting kAudioCodecPropertyDelayMode failed with %d", error);
260 return nil;
261 }
262 return retainPtr([NSNumber numberWithInt:(primeInfo.leadingFrames / m_destinationFormat.mFramesPerPacket)]);
263}

265CMSampleBufferRef AudioSampleBufferCompressor::sampleBufferWithNumPackets(UInt32 numPackets, AudioBufferList fillBufferList)
266{
267 Vector<char> cookie;
268 if (!m_destinationFormatDescription) {
269 UInt32 cookieSize = 0;
270
271 auto error = PAL::AudioConverterGetPropertyInfo(m_converter, kAudioConverterCompressionMagicCookie, &cookieSize, NULL);
272 if ((error == noErr) && !!cookieSize) {
273 cookie.resize(cookieSize);
274
275 if (auto error = PAL::AudioConverterGetProperty(m_converter, kAudioConverterCompressionMagicCookie, &cookieSize, cookie.data())) {
276 RELEASE_LOG_ERROR(MediaStream, "AudioSampleBufferCompressor getting kAudioConverterCompressionMagicCookie failed with %d", error);
277 return nil;
278 }
279 }
280
281 CMFormatDescriptionRef destinationFormatDescription;
282 if (auto error = PAL::CMAudioFormatDescriptionCreate(kCFAllocatorDefault, &m_destinationFormat, 0, NULL, cookieSize, cookie.data(), NULL, &destinationFormatDescription)) {
283 RELEASE_LOG_ERROR(MediaStream, "AudioSampleBufferCompressor CMAudioFormatDescriptionCreate failed with %d", error);
284 return nil;
285 }
286 m_destinationFormatDescription = adoptCF(destinationFormatDescription);
287 m_gdrCountNum = gradualDecoderRefreshCount();
288 }
289
290 char *data = static_cast<char*>(fillBufferList.mBuffers[0].mData);
291 size_t dataSize = fillBufferList.mBuffers[0].mDataByteSize;
292
293 CMBlockBufferRef blockBuffer;
294 if (auto error = PAL::CMBlockBufferCreateWithMemoryBlock(kCFAllocatorDefault, NULL, dataSize, kCFAllocatorDefault, NULL, 0, dataSize, kCMBlockBufferAssureMemoryNowFlag, &blockBuffer)) {
295 RELEASE_LOG_ERROR(MediaStream, "AudioSampleBufferCompressor CMBlockBufferCreateWithMemoryBlock failed with %d", error);
296 return nil;
297 }
298 auto buffer = adoptCF(blockBuffer);
299
300 if (auto error = PAL::CMBlockBufferReplaceDataBytes(data, buffer.get(), 0, dataSize)) {
301 RELEASE_LOG_ERROR(MediaStream, "AudioSampleBufferCompressor CMBlockBufferReplaceDataBytes failed with %d", error);
302 return nil;
303 }
304
305 CMSampleBufferRef sampleBuffer;
306 auto error = PAL::CMAudioSampleBufferCreateWithPacketDescriptions(kCFAllocatorDefault, buffer.get(), true, NULL, NULL, m_destinationFormatDescription.get(), numPackets, m_currentNativePresentationTimeStamp, m_destinationPacketDescriptions.data(), &sampleBuffer);
307 if (error) {
308 RELEASE_LOG_ERROR(MediaStream, "AudioSampleBufferCompressor CMAudioSampleBufferCreateWithPacketDescriptions failed with %d", error);
309 return nil;
310 }
311
312 if ([m_gdrCountNum intValue])
313 PAL::CMSetAttachment(sampleBuffer, PAL::kCMSampleBufferAttachmentKey_GradualDecoderRefresh, (__bridge CFTypeRef)m_gdrCountNum.get(), kCMAttachmentMode_ShouldPropagate);
314
315 return sampleBuffer;
316}

324OSStatus AudioSampleBufferCompressor::provideSourceDataNumOutputPackets(UInt32* numOutputPacketsPtr, AudioBufferList* audioBufferList, AudioStreamPacketDescription** packetDescriptionOut)
325{
326 if (packetDescriptionOut)
327 *packetDescriptionOut = NULL;
328
329 const UInt32 numPacketsToCopy = *numOutputPacketsPtr;
330 size_t numBytesToCopy = (numPacketsToCopy * m_sourceFormat.mBytesPerPacket);
331
332 if (audioBufferList->mNumberBuffers == 1) {
333 size_t currentOffsetInSourceBuffer = 0;
334
335 if (m_sourceBuffer.size() < numBytesToCopy)
336 m_sourceBuffer.resize(numBytesToCopy);
337
338 while (numBytesToCopy) {
339 if (m_sampleBlockBufferSize <= m_currentOffsetInSampleBlockBuffer) {
340 if (m_sampleBlockBuffer) {
341 m_sampleBlockBuffer = nullptr;
342 m_sampleBlockBufferSize = 0;
343 }
344
345 if (PAL::CMBufferQueueIsEmpty(m_inputBufferQueue.get()))
346 break;
347
348 auto sampleBuffer = adoptCF((CMSampleBufferRef)(const_cast<void*>(PAL::CMBufferQueueDequeueAndRetain(m_inputBufferQueue.get()))));
349 m_sampleBlockBuffer = PAL::CMSampleBufferGetDataBuffer(sampleBuffer.get());
350 if (!m_sampleBlockBuffer) {
351 RELEASE_LOG_ERROR(MediaStream, "AudioSampleBufferCompressor CMSampleBufferGetDataBuffer failed");
352 m_sampleBlockBufferSize = 0;
353 m_currentOffsetInSampleBlockBuffer = 0;
354 continue;
355 }
356 m_sampleBlockBufferSize = PAL::CMBlockBufferGetDataLength(m_sampleBlockBuffer.get());
357 m_currentOffsetInSampleBlockBuffer = 0;
358 }
359
360 if (m_sampleBlockBuffer) {
361 size_t numBytesToCopyFromSampleBbuf = std::min(numBytesToCopy, (m_sampleBlockBufferSize - m_currentOffsetInSampleBlockBuffer));
362 if (auto error = PAL::CMBlockBufferCopyDataBytes(m_sampleBlockBuffer.get(), m_currentOffsetInSampleBlockBuffer, numBytesToCopyFromSampleBbuf, (m_sourceBuffer.data() + currentOffsetInSourceBuffer))) {
363 RELEASE_LOG_ERROR(MediaStream, "AudioSampleBufferCompressor CMBlockBufferCopyDataBytes failed with %d", error);
364 return error;
365 }
366 numBytesToCopy -= numBytesToCopyFromSampleBbuf;
367 currentOffsetInSourceBuffer += numBytesToCopyFromSampleBbuf;
368 m_currentOffsetInSampleBlockBuffer += numBytesToCopyFromSampleBbuf;
369 }
370 }
371 audioBufferList->mBuffers[0].mData = m_sourceBuffer.data();
372 audioBufferList->mBuffers[0].mDataByteSize = currentOffsetInSourceBuffer;
373 audioBufferList->mBuffers[0].mNumberChannels = m_sourceFormat.mChannelsPerFrame;
374
375 *numOutputPacketsPtr = (audioBufferList->mBuffers[0].mDataByteSize / m_sourceFormat.mBytesPerPacket);
376 return noErr;
377 }
378
379 ASSERT(audioBufferList->mNumberBuffers == 2);
380
381 // FIXME: Support interleaved data by uninterleaving m_sourceBuffer if needed.
382 ASSERT(m_sourceFormat.mFormatFlags & kAudioFormatFlagIsNonInterleaved);
383
384 if (m_sourceBuffer.size() < 2 * numBytesToCopy)
385 m_sourceBuffer.resize(2 * numBytesToCopy);
386 auto* firstChannel = m_sourceBuffer.data();
387 auto* secondChannel = m_sourceBuffer.data() + numBytesToCopy;
388
389 size_t currentOffsetInSourceBuffer = 0;
390 while (numBytesToCopy) {
391 if (m_sampleBlockBufferSize <= m_currentOffsetInSampleBlockBuffer) {
392 if (m_sampleBlockBuffer) {
393 m_sampleBlockBuffer = nullptr;
394 m_sampleBlockBufferSize = 0;
395 }
396
397 if (PAL::CMBufferQueueIsEmpty(m_inputBufferQueue.get()))
398 break;
399
400 auto sampleBuffer = adoptCF((CMSampleBufferRef)(const_cast<void*>(PAL::CMBufferQueueDequeueAndRetain(m_inputBufferQueue.get()))));
401 m_sampleBlockBuffer = adoptCF((CMBlockBufferRef)(const_cast<void*>(CFRetain(PAL::CMSampleBufferGetDataBuffer(sampleBuffer.get())))));
402 if (!m_sampleBlockBuffer) {
403 RELEASE_LOG_ERROR(MediaStream, "AudioSampleBufferCompressor CMSampleBufferGetDataBuffer failed");
404 m_sampleBlockBufferSize = 0;
405 m_currentOffsetInSampleBlockBuffer = 0;
406 continue;
407 }
408 m_sampleBlockBufferSize = PAL::CMBlockBufferGetDataLength(m_sampleBlockBuffer.get()) / 2;
409 m_currentOffsetInSampleBlockBuffer = 0;
410 }
411
412 if (m_sampleBlockBuffer) {
413 size_t numBytesToCopyFromSampleBbuf = std::min(numBytesToCopy, (m_sampleBlockBufferSize - m_currentOffsetInSampleBlockBuffer));
414 if (auto error = PAL::CMBlockBufferCopyDataBytes(m_sampleBlockBuffer.get(), m_currentOffsetInSampleBlockBuffer, numBytesToCopyFromSampleBbuf, (firstChannel + currentOffsetInSourceBuffer))) {
415 RELEASE_LOG_ERROR(MediaStream, "AudioSampleBufferCompressor CMBlockBufferCopyDataBytes first channel failed with %d", error);
416 return error;
417 }
418 if (auto error = PAL::CMBlockBufferCopyDataBytes(m_sampleBlockBuffer.get(), m_currentOffsetInSampleBlockBuffer + m_sampleBlockBufferSize, numBytesToCopyFromSampleBbuf, (secondChannel + currentOffsetInSourceBuffer))) {
419 RELEASE_LOG_ERROR(MediaStream, "AudioSampleBufferCompressor CMBlockBufferCopyDataBytes second channel failed with %d", error);
420 return error;
421 }
422 numBytesToCopy -= numBytesToCopyFromSampleBbuf;
423 currentOffsetInSourceBuffer += numBytesToCopyFromSampleBbuf;
424 m_currentOffsetInSampleBlockBuffer += numBytesToCopyFromSampleBbuf;
425 }
426 }
427
428 audioBufferList->mBuffers[0].mData = firstChannel;
429 audioBufferList->mBuffers[0].mDataByteSize = currentOffsetInSourceBuffer;
430 audioBufferList->mBuffers[0].mNumberChannels = 1;
431
432 audioBufferList->mBuffers[1].mData = secondChannel;
433 audioBufferList->mBuffers[1].mDataByteSize = currentOffsetInSourceBuffer;
434 audioBufferList->mBuffers[1].mNumberChannels = 1;
435
436 *numOutputPacketsPtr = (audioBufferList->mBuffers[0].mDataByteSize / m_sourceFormat.mBytesPerPacket);
437 return noErr;
438}

440void AudioSampleBufferCompressor::processSampleBuffersUntilLowWaterTime(CMTime lowWaterTime)
441{
442 using namespace PAL; // For CMTIME_COMPARE_INLINE
443
444 if (!m_converter) {
445 if (CMBufferQueueIsEmpty(m_inputBufferQueue.get()))
446 return;
447
448 auto buffer = (CMSampleBufferRef)(const_cast<void*>(CMBufferQueueGetHead(m_inputBufferQueue.get())));
449 ASSERT(buffer);
450
451 m_currentNativePresentationTimeStamp = CMSampleBufferGetPresentationTimeStamp(buffer);
452 m_currentOutputPresentationTimeStamp = CMSampleBufferGetOutputPresentationTimeStamp(buffer);
453
454 auto formatDescription = CMSampleBufferGetFormatDescription(buffer);
455 if (!initAudioConverterForSourceFormatDescription(formatDescription, m_outputCodecType)) {
456 // FIXME: Maybe we should error the media recorder if we are not able to get a correct converter.
457 return;
458 }
459 }
460
461 while (CMTIME_IS_INVALID(lowWaterTime) || CMTIME_COMPARE_INLINE(lowWaterTime, <, CMBufferQueueGetDuration(m_inputBufferQueue.get()))) {
462 AudioBufferList fillBufferList;
463
464 fillBufferList.mNumberBuffers = 1;
465 fillBufferList.mBuffers[0].mNumberChannels = m_destinationFormat.mChannelsPerFrame;
466 fillBufferList.mBuffers[0].mDataByteSize = (UInt32)m_destinationBuffer.capacity();
467 fillBufferList.mBuffers[0].mData = m_destinationBuffer.data();
468
469 UInt32 outputPacketSize = m_destinationFormat.mBytesPerPacket ? m_destinationFormat.mBytesPerPacket : m_maxOutputPacketSize;
470 UInt32 numOutputPackets = (UInt32)m_destinationBuffer.capacity() / outputPacketSize;
471
472 auto error = AudioConverterFillComplexBuffer(m_converter, audioConverterComplexInputDataProc, this, &numOutputPackets, &fillBufferList, m_destinationPacketDescriptions.data());
473 if (error) {
474 RELEASE_LOG_ERROR(MediaStream, "AudioSampleBufferCompressor AudioConverterFillComplexBuffer failed with %d", error);
475 return;
476 }
477
478 if (!numOutputPackets)
479 break;
480
481 auto buffer = sampleBufferWithNumPackets(numOutputPackets, fillBufferList);
482
483 attachPrimingTrimsIfNeeded(buffer);
484
485 error = CMSampleBufferSetOutputPresentationTimeStamp(buffer, m_currentOutputPresentationTimeStamp);
486 if (error) {
487 RELEASE_LOG_ERROR(MediaStream, "AudioSampleBufferCompressor CMSampleBufferSetOutputPresentationTimeStamp failed with %d", error);
488 return;
489 }
490
491 CMTime nativeDuration = CMSampleBufferGetDuration(buffer);
492 m_currentNativePresentationTimeStamp = CMTimeAdd(m_currentNativePresentationTimeStamp, nativeDuration);
493
494 CMTime outputDuration = CMSampleBufferGetOutputDuration(buffer);
495 m_currentOutputPresentationTimeStamp = CMTimeAdd(m_currentOutputPresentationTimeStamp, outputDuration);
496
497 error = CMBufferQueueEnqueue(m_outputBufferQueue.get(), buffer);
498 if (error) {
499 RELEASE_LOG_ERROR(MediaStream, "AudioSampleBufferCompressor CMBufferQueueEnqueue failed with %d", error);
500 return;
501 }
502 }
503}

505void AudioSampleBufferCompressor::processSampleBuffer(CMSampleBufferRef buffer)
506{
507 auto error = PAL::CMBufferQueueEnqueue(m_inputBufferQueue.get(), buffer);
508 RELEASE_LOG_ERROR_IF(error, MediaStream, "AudioSampleBufferCompressor CMBufferQueueEnqueue failed with %d", error);
509
510 processSampleBuffersUntilLowWaterTime(m_lowWaterTime);
511}

513void AudioSampleBufferCompressor::addSampleBuffer(CMSampleBufferRef buffer)
514{
515 // Heap allocations are forbidden on the audio thread for performance reasons so we need to
516 // explicitly allow the following allocation(s).
517 DisableMallocRestrictionsForCurrentThreadScope disableMallocRestrictions;
518 m_serialDispatchQueue->dispatchSync([this, buffer] {
519 if (m_isEncoding)
520 processSampleBuffer(buffer);
521 });
522}
