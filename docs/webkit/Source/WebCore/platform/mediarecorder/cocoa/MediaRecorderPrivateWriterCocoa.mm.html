131MediaRecorderPrivateWriter::~MediaRecorderPrivateWriter()
132{
133 m_pendingAudioSampleQueue.clear();
134 m_pendingVideoSampleQueue.clear();
135 if (m_writer) {
136 [m_writer cancelWriting];
137 m_writer.clear();
138 }
139
140 // At this pointer, we should no longer be writing any data, so it should be safe to close and nullify m_data without locking.
141 if (m_writerDelegate)
142 [m_writerDelegate close];
143 m_data = nullptr;
144
145 if (auto completionHandler = WTFMove(m_fetchDataCompletionHandler))
146 completionHandler(nullptr, 0);
147}

149bool MediaRecorderPrivateWriter::initialize(const MediaRecorderPrivateOptions& options)
150{
151 NSError *error = nil;
152 ALLOW_DEPRECATED_DECLARATIONS_BEGIN
153 m_writer = adoptNS([PAL::allocAVAssetWriterInstance() initWithFileType:AVFileTypeMPEG4 error:&error]);
154 ALLOW_DEPRECATED_DECLARATIONS_END
155 if (error) {
156 RELEASE_LOG_ERROR(MediaStream, "create AVAssetWriter instance failed with error code %ld", (long)error.code);
157 return false;
158 }
159
160 m_writerDelegate = adoptNS([[WebAVAssetWriterDelegate alloc] initWithWriter: *this]);
161 [m_writer.get() setDelegate:m_writerDelegate.get()];
162
163 if (m_hasAudio) {
164 m_audioCompressor = AudioSampleBufferCompressor::create(compressedAudioOutputBufferCallback, this);
165 if (!m_audioCompressor)
166 return false;
167 if (options.audioBitsPerSecond)
168 m_audioCompressor->setBitsPerSecond(*options.audioBitsPerSecond);
169 }
170 if (m_hasVideo) {
171 m_videoCompressor = VideoSampleBufferCompressor::create(options.mimeType, compressedVideoOutputBufferCallback, this);
172 if (!m_videoCompressor)
173 return false;
174 if (options.videoBitsPerSecond)
175 m_videoCompressor->setBitsPerSecond(*options.videoBitsPerSecond);
176 }
177
178 return true;
179}

181void MediaRecorderPrivateWriter::processNewCompressedVideoSampleBuffers()
182{
183 ASSERT(m_hasVideo);
184 if (!m_videoFormatDescription) {
185 m_videoFormatDescription = PAL::CMSampleBufferGetFormatDescription(m_videoCompressor->getOutputSampleBuffer());
186
187 if (m_hasAudio && !m_audioFormatDescription)
188 return;
189
190 startAssetWriter();
191 }
192
193 if (!m_hasStartedWriting)
194 return;
195 appendCompressedSampleBuffers();
196}

198void MediaRecorderPrivateWriter::processNewCompressedAudioSampleBuffers()
199{
200 ASSERT(m_hasAudio);
201 if (!m_audioFormatDescription) {
202 m_audioFormatDescription = PAL::CMSampleBufferGetFormatDescription(m_audioCompressor->getOutputSampleBuffer());
203 if (m_hasVideo && !m_videoFormatDescription)
204 return;
205
206 startAssetWriter();
207 }
208
209 if (!m_hasStartedWriting)
210 return;
211 appendCompressedSampleBuffers();
212}

214void MediaRecorderPrivateWriter::startAssetWriter()
215{
216 if (m_hasVideo) {
217 m_videoAssetWriterInput = adoptNS([PAL::allocAVAssetWriterInputInstance() initWithMediaType:AVMediaTypeVideo outputSettings:nil sourceFormatHint:m_videoFormatDescription.get()]);
218 [m_videoAssetWriterInput setExpectsMediaDataInRealTime:true];
219 if (m_videoTransform)
220 m_videoAssetWriterInput.get().transform = *m_videoTransform;
221
222 if (![m_writer.get() canAddInput:m_videoAssetWriterInput.get()]) {
223 RELEASE_LOG_ERROR(MediaStream, "MediaRecorderPrivateWriter::startAssetWriter failed canAddInput for video");
224 return;
225 }
226 [m_writer.get() addInput:m_videoAssetWriterInput.get()];
227 }
228
229 if (m_hasAudio) {
230 m_audioAssetWriterInput = adoptNS([PAL::allocAVAssetWriterInputInstance() initWithMediaType:AVMediaTypeAudio outputSettings:nil sourceFormatHint:m_audioFormatDescription.get()]);
231 [m_audioAssetWriterInput setExpectsMediaDataInRealTime:true];
232 if (![m_writer.get() canAddInput:m_audioAssetWriterInput.get()]) {
233 RELEASE_LOG_ERROR(MediaStream, "MediaRecorderPrivateWriter::startAssetWriter failed canAddInput for audio");
234 return;
235 }
236 [m_writer.get() addInput:m_audioAssetWriterInput.get()];
237 }
238
239 if (![m_writer.get() startWriting]) {
240 RELEASE_LOG_ERROR(MediaStream, "MediaRecorderPrivateWriter::startAssetWriter failed startWriting");
241 return;
242 }
243
244 [m_writer.get() startSessionAtSourceTime:PAL::kCMTimeZero];
245
246 appendCompressedSampleBuffers();
247
248 m_hasStartedWriting = true;
249}

251bool MediaRecorderPrivateWriter::appendCompressedAudioSampleBufferIfPossible()
252{
253 if (!m_audioCompressor)
254 return false;
255
256 auto buffer = m_audioCompressor->takeOutputSampleBuffer();
257 if (!buffer)
258 return false;
259
260 if (m_isFlushingSamples) {
261 m_pendingAudioSampleQueue.append(WTFMove(buffer));
262 return true;
263 }
264
265 while (!m_pendingAudioSampleQueue.isEmpty() && [m_audioAssetWriterInput isReadyForMoreMediaData])
266 [m_audioAssetWriterInput.get() appendSampleBuffer:m_pendingAudioSampleQueue.takeFirst().get()];
267
268 if (![m_audioAssetWriterInput isReadyForMoreMediaData]) {
269 m_pendingAudioSampleQueue.append(WTFMove(buffer));
270 return true;
271 }
272
273 [m_audioAssetWriterInput.get() appendSampleBuffer:buffer.get()];
274 return true;
275}

277bool MediaRecorderPrivateWriter::appendCompressedVideoSampleBufferIfPossible()
278{
279 if (!m_videoCompressor)
280 return false;
281
282 auto buffer = m_videoCompressor->takeOutputSampleBuffer();
283 if (!buffer)
284 return false;
285
286 if (m_isFlushingSamples) {
287 m_pendingVideoSampleQueue.append(WTFMove(buffer));
288 return true;
289 }
290
291 while (!m_pendingVideoSampleQueue.isEmpty() && [m_videoAssetWriterInput isReadyForMoreMediaData])
292 appendCompressedVideoSampleBuffer(m_pendingVideoSampleQueue.takeFirst().get());
293
294 if (![m_videoAssetWriterInput isReadyForMoreMediaData]) {
295 m_pendingVideoSampleQueue.append(WTFMove(buffer));
296 return true;
297 }
298
299 appendCompressedVideoSampleBuffer(buffer.get());
300 return true;
301}

303void MediaRecorderPrivateWriter::appendCompressedVideoSampleBuffer(CMSampleBufferRef buffer)
304{
305 ASSERT([m_videoAssetWriterInput isReadyForMoreMediaData]);
306 m_lastVideoPresentationTime = PAL::CMSampleBufferGetPresentationTimeStamp(buffer);
307 m_lastVideoDecodingTime = PAL::CMSampleBufferGetDecodeTimeStamp(buffer);
308 m_hasEncodedVideoSamples = true;
309
310 [m_videoAssetWriterInput.get() appendSampleBuffer:buffer];
311}

313void MediaRecorderPrivateWriter::appendCompressedSampleBuffers()
314{
315 while (appendCompressedVideoSampleBufferIfPossible() || appendCompressedAudioSampleBufferIfPossible()) { };
316}

318static inline void appendEndsPreviousSampleDurationMarker(AVAssetWriterInput *assetWriterInput, CMTime presentationTimeStamp, CMTime decodingTimeStamp)
319{
320 CMSampleTimingInfo timingInfo = { PAL::kCMTimeInvalid, presentationTimeStamp, decodingTimeStamp};
321
322 CMSampleBufferRef buffer = NULL;
323 auto error = PAL::CMSampleBufferCreate(kCFAllocatorDefault, NULL, true, NULL, NULL, NULL, 0, 1, &timingInfo, 0, NULL, &buffer);
324 if (error) {
325 RELEASE_LOG_ERROR(MediaStream, "MediaRecorderPrivateWriter appendEndsPreviousSampleDurationMarker failed CMSampleBufferCreate with %d", error);
326 return;
327 }
328 auto sampleBuffer = adoptCF(buffer);
329
330 PAL::CMSetAttachment(sampleBuffer.get(), PAL::kCMSampleBufferAttachmentKey_EndsPreviousSampleDuration, kCFBooleanTrue, kCMAttachmentMode_ShouldPropagate);
331 if (![assetWriterInput appendSampleBuffer:sampleBuffer.get()])
332 RELEASE_LOG_ERROR(MediaStream, "MediaRecorderPrivateWriter appendSampleBuffer to writer input failed");
333}

335void MediaRecorderPrivateWriter::flushCompressedSampleBuffers(Function<void()>&& callback)
336{
337 bool hasPendingAudioSamples = !m_pendingAudioSampleQueue.isEmpty();
338 bool hasPendingVideoSamples = !m_pendingVideoSampleQueue.isEmpty();
339
340 if (m_hasEncodedVideoSamples) {
341 hasPendingVideoSamples |= ![m_videoAssetWriterInput isReadyForMoreMediaData];
342 if (!hasPendingVideoSamples)
343 appendEndsPreviousSampleDurationMarker(m_videoAssetWriterInput.get(), m_lastVideoPresentationTime, m_lastVideoDecodingTime);
344 }
345
346 if (!hasPendingAudioSamples && !hasPendingVideoSamples) {
347 callback();
348 return;
349 }
350
351 ASSERT(!m_isFlushingSamples);
352 m_isFlushingSamples = true;
353 auto block = makeBlockPtr([this, weakThis = makeWeakPtr(*this), hasPendingAudioSamples, hasPendingVideoSamples, audioSampleQueue = WTFMove(m_pendingAudioSampleQueue), videoSampleQueue = WTFMove(m_pendingVideoSampleQueue), callback = WTFMove(callback)]() mutable {
354 if (!weakThis) {
355 callback();
356 return;
357 }
358
359 while (!audioSampleQueue.isEmpty() && [m_audioAssetWriterInput isReadyForMoreMediaData])
360 [m_audioAssetWriterInput.get() appendSampleBuffer:audioSampleQueue.takeFirst().get()];
361
362 while (!videoSampleQueue.isEmpty() && [m_videoAssetWriterInput isReadyForMoreMediaData])
363 appendCompressedVideoSampleBuffer(videoSampleQueue.takeFirst().get());
364
365 if (!audioSampleQueue.isEmpty() || !videoSampleQueue.isEmpty() || (hasPendingVideoSamples && ![m_videoAssetWriterInput isReadyForMoreMediaData]))
366 return;
367
368 if (hasPendingAudioSamples)
369 [m_audioAssetWriterInput markAsFinished];
370 if (hasPendingVideoSamples) {
371 appendEndsPreviousSampleDurationMarker(m_videoAssetWriterInput.get(), m_lastVideoPresentationTime, m_lastVideoDecodingTime);
372 [m_videoAssetWriterInput markAsFinished];
373 }
374 m_isFlushingSamples = false;
375 callback();
376 finishedFlushingSamples();
377 });
378
379 if (hasPendingAudioSamples)
380 [m_audioAssetWriterInput requestMediaDataWhenReadyOnQueue:dispatch_get_main_queue() usingBlock:block.get()];
381 if (hasPendingVideoSamples)
382 [m_videoAssetWriterInput requestMediaDataWhenReadyOnQueue:dispatch_get_main_queue() usingBlock:block.get()];
383}

385static inline RetainPtr<CMSampleBufferRef> copySampleBufferWithCurrentTimeStamp(CMSampleBufferRef originalBuffer, CMTime startTime)
386{
387 CMItemCount count = 0;
388 PAL::CMSampleBufferGetSampleTimingInfoArray(originalBuffer, 0, nil, &count);
389
390 Vector<CMSampleTimingInfo> timeInfo(count);
391 PAL::CMSampleBufferGetSampleTimingInfoArray(originalBuffer, count, timeInfo.data(), &count);
392
393 for (auto i = 0; i < count; i++) {
394 timeInfo[i].decodeTimeStamp = PAL::kCMTimeInvalid;
395 timeInfo[i].presentationTimeStamp = startTime;
396 }
397
398 CMSampleBufferRef newBuffer = nullptr;
399 if (auto error = PAL::CMSampleBufferCreateCopyWithNewTiming(kCFAllocatorDefault, originalBuffer, count, timeInfo.data(), &newBuffer)) {
400 RELEASE_LOG_ERROR(MediaStream, "MediaRecorderPrivateWriter CMSampleBufferCreateCopyWithNewTiming failed with %d", error);
401 return nullptr;
402 }
403 return adoptCF(newBuffer);
404}

406void MediaRecorderPrivateWriter::appendVideoSampleBuffer(MediaSample& sample)
407{
408 if (!m_firstVideoFrame) {
409 m_firstVideoFrame = true;
410 m_resumedVideoTime = PAL::CMClockGetTime(PAL::CMClockGetHostTimeClock());
411 if (sample.videoRotation() != MediaSample::VideoRotation::None || sample.videoMirrored()) {
412 m_videoTransform = CGAffineTransformMakeRotation(static_cast<int>(sample.videoRotation()) * M_PI / 180);
413 if (sample.videoMirrored())
414 m_videoTransform = CGAffineTransformScale(*m_videoTransform, -1, 1);
415 }
416 }
417
418 auto sampleTime = PAL::CMTimeSubtract(PAL::CMClockGetTime(PAL::CMClockGetHostTimeClock()), m_resumedVideoTime);
419 sampleTime = PAL::CMTimeAdd(sampleTime, m_currentVideoDuration);
420 if (auto bufferWithCurrentTime = copySampleBufferWithCurrentTimeStamp(sample.platformSample().sample.cmSampleBuffer, sampleTime))
421 m_videoCompressor->addSampleBuffer(bufferWithCurrentTime.get());
422}

424void MediaRecorderPrivateWriter::appendAudioSampleBuffer(const PlatformAudioData& data, const AudioStreamDescription& description, const WTF::MediaTime&, size_t sampleCount)
425{
426 if (auto sampleBuffer = createAudioSampleBuffer(data, description, m_currentAudioSampleTime, sampleCount))
427 m_audioCompressor->addSampleBuffer(sampleBuffer.get());
428 m_currentAudioSampleTime = PAL::CMTimeAdd(m_currentAudioSampleTime, PAL::toCMTime(MediaTime(sampleCount, description.sampleRate())));
429}

437void MediaRecorderPrivateWriter::stopRecording()
438{
439 if (m_isFlushingSamples) {
440 m_shouldStopAfterFlushingSamples = true;
441 return;
442 }
443
444 if (m_isStopped)
445 return;
446
447 m_isStopped = true;
448
449 if (m_videoCompressor)
450 m_videoCompressor->finish();
451 if (m_audioCompressor)
452 m_audioCompressor->finish();
453
454 m_isStopping = true;
455 // We hop to the main thread since finishing the video compressor might trigger starting the writer asynchronously.
456 callOnMainThread([this, weakThis = makeWeakPtr(this)]() mutable {
457 if (!weakThis)
458 return;
459
460 auto whenFinished = [this, weakThis] {
461 if (!weakThis)
462 return;
463
464 m_isStopping = false;
465 m_isStopped = false;
466 m_hasStartedWriting = false;
467
468 if (m_writer) {
469 [m_writer cancelWriting];
470 m_writer.clear();
471 }
472
473 if (m_fetchDataCompletionHandler)
474 m_fetchDataCompletionHandler(takeData(), 0);
475 };
476
477 if (!m_hasStartedWriting) {
478 whenFinished();
479 return;
480 }
481
482 ASSERT([m_writer status] == AVAssetWriterStatusWriting);
483 flushCompressedSampleBuffers([this, weakThis = WTFMove(weakThis), whenFinished = WTFMove(whenFinished)]() mutable {
484 if (!weakThis)
485 return;
486
487 ALLOW_DEPRECATED_DECLARATIONS_BEGIN
488 [m_writer flush];
489 ALLOW_DEPRECATED_DECLARATIONS_END
490
491 [m_writer finishWritingWithCompletionHandler:[whenFinished = WTFMove(whenFinished)]() mutable {
492 callOnMainThread(WTFMove(whenFinished));
493 }];
494 });
495 });
496}

498void MediaRecorderPrivateWriter::fetchData(CompletionHandler<void(RefPtr<SharedBuffer>&&, double)>&& completionHandler)
499{
500 m_fetchDataCompletionHandler = WTFMove(completionHandler);
501
502 if (m_isStopping)
503 return;
504
505 if (!m_hasStartedWriting) {
506 completeFetchData();
507 return;
508 }
509
510 flushCompressedSampleBuffers([weakThis = makeWeakPtr(this)]() mutable {
511 if (!weakThis)
512 return;
513
514 ALLOW_DEPRECATED_DECLARATIONS_BEGIN
515 [weakThis->m_writer flush];
516 ALLOW_DEPRECATED_DECLARATIONS_END
517
518 callOnMainThread([weakThis = WTFMove(weakThis)] {
519 if (weakThis)
520 weakThis->completeFetchData();
521 });
522 });
523}

525void MediaRecorderPrivateWriter::completeFetchData()
526{
527 auto currentTimeCode = m_timeCode;
528 if (m_hasAudio)
529 m_timeCode = PAL::CMTimeGetSeconds(m_currentAudioSampleTime);
530 else {
531 auto sampleTime = PAL::CMTimeSubtract(PAL::CMClockGetTime(PAL::CMClockGetHostTimeClock()), m_resumedVideoTime);
532 m_timeCode = PAL::CMTimeGetSeconds(PAL::CMTimeAdd(sampleTime, m_currentVideoDuration));
533 }
534 m_fetchDataCompletionHandler(takeData(), currentTimeCode);
535}

537void MediaRecorderPrivateWriter::appendData(const uint8_t* data, size_t size)
538{
539 Locker locker { m_dataLock };

541 m_data = SharedBuffer::create(data, size);

544 m_data->append(data, size);
545}

554void MediaRecorderPrivateWriter::pause()
555{
556 auto recordingDuration = PAL::CMTimeSubtract(PAL::CMClockGetTime(PAL::CMClockGetHostTimeClock()), m_resumedVideoTime);
557 m_currentVideoDuration = PAL::CMTimeAdd(recordingDuration, m_currentVideoDuration);
558}
