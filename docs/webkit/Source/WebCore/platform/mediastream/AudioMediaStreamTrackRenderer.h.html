27
28#if ENABLE(MEDIA_STREAM)
29
30#include <wtf/Function.h>
31#include <wtf/LoggerHelper.h>
32
33namespace WTF {
34class MediaTime;
35}
36
37namespace WebCore {
38
39class AudioStreamDescription;
40class PlatformAudioData;
41
42class WEBCORE_EXPORT AudioMediaStreamTrackRenderer : public LoggerHelper {
43 WTF_MAKE_FAST_ALLOCATED;
44public:
45 static std::unique_ptr<AudioMediaStreamTrackRenderer> create();
46 virtual ~AudioMediaStreamTrackRenderer() = default;
47
48 virtual void start(CompletionHandler<void()>&&) = 0;
49 virtual void stop() = 0;
50 virtual void clear() = 0;
51 // May be called on a background thread. It should only be called after start/before stop is called.
52 virtual void pushSamples(const WTF::MediaTime&, const PlatformAudioData&, const AudioStreamDescription&, size_t) = 0;
53
54 virtual void setVolume(float);
55 float volume() const;
56
57 virtual void setAudioOutputDevice(const String&);
58
59#if !RELEASE_LOG_DISABLED
60 void setLogger(const Logger&, const void*);
61#endif
62
63 using RendererCreator = std::unique_ptr<AudioMediaStreamTrackRenderer> (*)();
64 static void setCreator(RendererCreator);
65
66 void setCrashCallback(Function<void()>&& callback) { m_crashCallback = WTFMove(callback); }
67
68protected:
69#if !RELEASE_LOG_DISABLED
70 const Logger& logger() const final;
71 const void* logIdentifier() const final;
72
73 const char* logClassName() const final;
74 WTFLogChannel& logChannel() const final;
75#endif
76
77 void crashed();
78
79private:
80 static RendererCreator m_rendererCreator;
81
82 // Main thread writable members
83 float m_volume { 1 };
84 Function<void()> m_crashCallback;
85
86#if !RELEASE_LOG_DISABLED
87 RefPtr<const Logger> m_logger;
88 const void* m_logIdentifier;
89#endif
90};
91
92inline void AudioMediaStreamTrackRenderer::setVolume(float volume)
93{
94 m_volume = volume;
95}
96
97inline float AudioMediaStreamTrackRenderer::volume() const
98{
99 return m_volume;
100}
101
102inline void AudioMediaStreamTrackRenderer::crashed()
103{
104 if (m_crashCallback)
105 m_crashCallback();
106}
107
108#if !RELEASE_LOG_DISABLED
109inline const Logger& AudioMediaStreamTrackRenderer::logger() const
110{
111 return *m_logger;
112
113}
114
115inline const void* AudioMediaStreamTrackRenderer::logIdentifier() const
116{
117 return m_logIdentifier;
118}
119
120inline const char* AudioMediaStreamTrackRenderer::logClassName() const
121{
122 return "AudioMediaStreamTrackRenderer";
123}
124#endif
125
126inline void AudioMediaStreamTrackRenderer::setAudioOutputDevice(const String&)
127{
128}
129
130}
