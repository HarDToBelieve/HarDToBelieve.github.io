59MediaStreamTrackPrivate::MediaStreamTrackPrivate(Ref<const Logger>&& trackLogger, Ref<RealtimeMediaSource>&& source, String&& id)
60 : m_source(WTFMove(source))
61 , m_id(WTFMove(id))
62 , m_logger(WTFMove(trackLogger))
63#if !RELEASE_LOG_DISABLED
64 , m_logIdentifier(uniqueLogIdentifier())
65#endif
66{
67 ASSERT(isMainThread());
68 UNUSED_PARAM(trackLogger);
69 ALWAYS_LOG(LOGIDENTIFIER);
70#if !RELEASE_LOG_DISABLED
71 m_source->setLogger(m_logger.copyRef(), m_logIdentifier);
72#endif
73 m_source->addObserver(*this);
74}

124void MediaStreamTrackPrivate::setEnabled(bool enabled)
125{
126 if (m_isEnabled == enabled)
127 return;
128
129 ALWAYS_LOG(LOGIDENTIFIER, enabled);
130
131 // Always update the enabled state regardless of the track being ended.
132 m_isEnabled = enabled;
133
134 forEachObserver([this](auto& observer) {
135 observer.trackEnabledChanged(*this);
136 });
137}
138
139void MediaStreamTrackPrivate::endTrack()
140{
141 if (m_isEnded)
142 return;
143
144 ALWAYS_LOG(LOGIDENTIFIER);
145
146 // Set m_isEnded to true before telling the source it can stop, so if this is the
147 // only track using the source and it does stop, we will only call each observer's
148 // trackEnded method once.
149 m_isEnded = true;
150 updateReadyState();
151
152 m_source->requestToEnd(*this);
153
154 forEachObserver([this](auto& observer) {
155 observer.trackEnded(*this);
156 });
157}

159Ref<MediaStreamTrackPrivate> MediaStreamTrackPrivate::clone()
160{
161 auto clonedMediaStreamTrackPrivate = create(m_logger.copyRef(), m_source->clone());
162
163 ALWAYS_LOG(LOGIDENTIFIER, clonedMediaStreamTrackPrivate->logIdentifier());
164
165 clonedMediaStreamTrackPrivate->m_isEnabled = this->m_isEnabled;
166 clonedMediaStreamTrackPrivate->m_isEnded = this->m_isEnded;
167 clonedMediaStreamTrackPrivate->m_contentHint = this->m_contentHint;
168 clonedMediaStreamTrackPrivate->updateReadyState();
169
170 if (isProducingData())
171 clonedMediaStreamTrackPrivate->startProducingData();
172
173 return clonedMediaStreamTrackPrivate;
174}

252{

259void MediaStreamTrackPrivate::hasStartedProducingData()
260{
261 ASSERT(isMainThread());
262 if (m_hasStartedProducingData)
263 return;
264 ALWAYS_LOG(LOGIDENTIFIER);
265 m_hasStartedProducingData = true;
266 updateReadyState();
267}
268
269void MediaStreamTrackPrivate::updateReadyState()
270{
271 ReadyState state = ReadyState::None;
272
273 if (m_isEnded)
274 state = ReadyState::Ended;
275 else if (m_hasStartedProducingData)
276 state = ReadyState::Live;
277
278 if (state == m_readyState)
279 return;
280
281 ALWAYS_LOG(LOGIDENTIFIER, state == ReadyState::Ended ? "Ended" : "Live");
282
283 m_readyState = state;
284 forEachObserver([this](auto& observer) {
285 observer.readyStateChanged(*this);
286 });
287}
