41RealtimeIncomingAudioSource::RealtimeIncomingAudioSource(rtc::scoped_refptr<webrtc::AudioTrackInterface>&& audioTrack, String&& audioTrackId)
42 : RealtimeMediaSource(RealtimeMediaSource::Type::Audio, "remote audio"_s, WTFMove(audioTrackId))
43 , m_audioTrack(WTFMove(audioTrack))
44{
45 ASSERT(m_audioTrack);
46 m_audioTrack->RegisterObserver(this);
47}
48
49RealtimeIncomingAudioSource::~RealtimeIncomingAudioSource()
50{
51 stop();
52 m_audioTrack->UnregisterObserver(this);
53}
54
55void RealtimeIncomingAudioSource::startProducingData()
56{
57 m_audioTrack->AddSink(this);
58}
59
60void RealtimeIncomingAudioSource::stopProducingData()
61{
62 m_audioTrack->RemoveSink(this);
63}
64
65void RealtimeIncomingAudioSource::OnChanged()
66{
67 callOnMainThread([protectedThis = makeRef(*this)] {
68 if (protectedThis->m_audioTrack->state() == webrtc::MediaStreamTrackInterface::kEnded)
69 protectedThis->end();
70 });
71}
72
73const RealtimeMediaSourceCapabilities& RealtimeIncomingAudioSource::capabilities()
74{
75 return RealtimeMediaSourceCapabilities::emptyCapabilities();
76}
77
78const RealtimeMediaSourceSettings& RealtimeIncomingAudioSource::settings()
79{
80 return m_currentSettings;
81}
