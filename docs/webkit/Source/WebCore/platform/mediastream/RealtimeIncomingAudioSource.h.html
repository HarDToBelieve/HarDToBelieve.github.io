40#include <webrtc/api/media_stream_interface.h>

44#include <wtf/RetainPtr.h>
45
46namespace WebCore {
47
48class RealtimeIncomingAudioSource
49 : public RealtimeMediaSource
50 , private webrtc::AudioTrackSinkInterface
51 , private webrtc::ObserverInterface
52{
53public:
54 static Ref<RealtimeIncomingAudioSource> create(rtc::scoped_refptr<webrtc::AudioTrackInterface>&&, String&&);
55
56protected:
57 RealtimeIncomingAudioSource(rtc::scoped_refptr<webrtc::AudioTrackInterface>&&, String&&);
58 ~RealtimeIncomingAudioSource();
59
60#if !RELEASE_LOG_DISABLED
61 const char* logClassName() const final { return "RealtimeIncomingAudioSource"; }
62#endif
63
64 // RealtimeMediaSource API
65 void startProducingData() override;
66 void stopProducingData() override;
67
68private:
69 // webrtc::AudioTrackSinkInterface API
70 void OnData(const void* /* audioData */, int /* bitsPerSample */, int /* sampleRate */, size_t /* numberOfChannels */, size_t /* numberOfFrames */) override { };
71
72 // webrtc::ObserverInterface API
73 void OnChanged() final;
74
75 const RealtimeMediaSourceCapabilities& capabilities() final;
76 const RealtimeMediaSourceSettings& settings() final;
77
78 bool isIncomingAudioSource() const final { return true; }
79
80 RealtimeMediaSourceSettings m_currentSettings;
81 rtc::scoped_refptr<webrtc::AudioTrackInterface> m_audioTrack;
82
83#if !RELEASE_LOG_DISABLED
84 mutable RefPtr<const Logger> m_logger;
85 const void* m_logIdentifier;
86#endif
87};
88
89} // namespace WebCore
90
91SPECIALIZE_TYPE_TRAITS_BEGIN(WebCore::RealtimeIncomingAudioSource)
92 static bool isType(const WebCore::RealtimeMediaSource& source) { return source.isIncomingAudioSource(); }
93SPECIALIZE_TYPE_TRAITS_END()

95#endif // USE(LIBWEBRTC)
