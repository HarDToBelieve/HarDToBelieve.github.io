46namespace WebCore {
47
48class CaptureDevice;
49
50class RealtimeIncomingVideoSource
51 : public RealtimeMediaSource
52 , private rtc::VideoSinkInterface<webrtc::VideoFrame>
53 , private webrtc::ObserverInterface
54{
55public:
56 static Ref<RealtimeIncomingVideoSource> create(rtc::scoped_refptr<webrtc::VideoTrackInterface>&&, String&&);
57 ~RealtimeIncomingVideoSource();
58
59protected:
60 RealtimeIncomingVideoSource(rtc::scoped_refptr<webrtc::VideoTrackInterface>&&, String&&);
61
62#if !RELEASE_LOG_DISABLED
63 const char* logClassName() const final { return "RealtimeIncomingVideoSource"; }
64#endif
65
66private:
67 // RealtimeMediaSource API
68 void startProducingData() final;
69 void stopProducingData() final;
70 void settingsDidChange(OptionSet<RealtimeMediaSourceSettings::Flag>) final;
71
72 const RealtimeMediaSourceCapabilities& capabilities() final;
73 const RealtimeMediaSourceSettings& settings() final;
74
75 bool isIncomingVideoSource() const final { return true; }
76
77 // webrtc::ObserverInterface API
78 void OnChanged() final;
79
80 std::optional<RealtimeMediaSourceSettings> m_currentSettings;
81 rtc::scoped_refptr<webrtc::VideoTrackInterface> m_videoTrack;
82
83#if !RELEASE_LOG_DISABLED
84 mutable RefPtr<const Logger> m_logger;
85 const void* m_logIdentifier;
86#endif
87};
88
89} // namespace WebCore
