105void RealtimeMediaSource::setMuted(bool muted)
106{
107 // Changed m_muted before calling start/stop so muted() will reflect the correct state.
108 bool changed = m_muted != muted;
109
110 if (changed)
111 ALWAYS_LOG_IF(m_logger, LOGIDENTIFIER, muted);
112
113 m_muted = muted;
114 if (muted)
115 stop();
116 else
117 start();
118
119 if (changed)
120 notifyMutedObservers();
121}

174{

179 // explicitly allow the following allocation(s).
180 DisableMallocRestrictionsForCurrentThreadScope disableMallocRestrictions;
181 callOnMainThread([protectedThis = makeRef(*this)] {
182 if (protectedThis->m_hasStartedProducingData)
183 return;
184 protectedThis->m_hasStartedProducingData = true;
185 protectedThis->forEachObserver([&](auto& observer) {
186 observer.hasStartedProducingData();
187 });
188 });

191void RealtimeMediaSource::videoSampleAvailable(MediaSample& mediaSample)
192{
193#if !RELEASE_LOG_DISABLED
194 ++m_frameCount;
195
196 auto timestamp = MonotonicTime::now();
197 auto delta = timestamp - m_lastFrameLogTime;
198 if (!m_lastFrameLogTime || delta >= 1_s) {
199 if (m_lastFrameLogTime) {
200 INFO_LOG_IF(loggerPtr(), LOGIDENTIFIER, m_frameCount, " frames sent in ", delta.value(), " seconds");
201 m_frameCount = 0;
202 }
203 m_lastFrameLogTime = timestamp;
204 }
205#endif
206
207 updateHasStartedProducingData();
208
209 Locker locker { m_videoSampleObserversLock };
210 for (auto* observer : m_videoSampleObservers)
211 observer->videoSampleAvailable(mediaSample);
212}

253{

255 forEachObserver([&](auto& observer) {

301 // The size and frame rate are within the capability limits, so they are supported.
302 return true;

305bool RealtimeMediaSource::supportsSizeAndFrameRate(std::optional<IntConstraint> widthConstraint, std::optional<IntConstraint> heightConstraint, std::optional<DoubleConstraint> frameRateConstraint, String& badConstraint, double& distance)
306{
307 if (!widthConstraint && !heightConstraint && !frameRateConstraint)
308 return true;
309
310 auto& capabilities = this->capabilities();
311
312 distance = std::numeric_limits<double>::infinity();
313
314 std::optional<int> width;
315 if (widthConstraint && capabilities.supportsWidth()) {
316 double constraintDistance = fitnessDistance(*widthConstraint);
317 if (std::isinf(constraintDistance)) {
318 auto range = capabilities.width();
319 WTFLogAlways("RealtimeMediaSource::supportsSizeAndFrameRate failed width constraint, capabilities are [%d, %d]", range.rangeMin().asInt, range.rangeMax().asInt);
320 badConstraint = widthConstraint->name();
321 return false;
322 }
323
324 distance = std::min(distance, constraintDistance);
325 if (widthConstraint->isMandatory()) {
326 auto range = capabilities.width();
327 width = widthConstraint->valueForCapabilityRange(size().width(), range.rangeMin().asInt, range.rangeMax().asInt);
328 }
329 }
330
331 std::optional<int> height;
332 if (heightConstraint && capabilities.supportsHeight()) {
333 double constraintDistance = fitnessDistance(*heightConstraint);
334 if (std::isinf(constraintDistance)) {
335 auto range = capabilities.height();
336 WTFLogAlways("RealtimeMediaSource::supportsSizeAndFrameRate failed height constraint, capabilities are [%d, %d]", range.rangeMin().asInt, range.rangeMax().asInt);
337 badConstraint = heightConstraint->name();
338 return false;
339 }
340
341 distance = std::min(distance, constraintDistance);
342 if (heightConstraint->isMandatory()) {
343 auto range = capabilities.height();
344 height = heightConstraint->valueForCapabilityRange(size().height(), range.rangeMin().asInt, range.rangeMax().asInt);
345 }
346 }
347
348 std::optional<double> frameRate;
349 if (frameRateConstraint && capabilities.supportsFrameRate()) {
350 double constraintDistance = fitnessDistance(*frameRateConstraint);
351 if (std::isinf(constraintDistance)) {
352 auto range = capabilities.frameRate();
353 WTFLogAlways("RealtimeMediaSource::supportsSizeAndFrameRate failed frame rate constraint, capabilities are [%d, %d]", range.rangeMin().asInt, range.rangeMax().asInt);
354 badConstraint = frameRateConstraint->name();
355 return false;
356 }
357
358 distance = std::min(distance, constraintDistance);
359 if (frameRateConstraint->isMandatory()) {
360 auto range = capabilities.frameRate();
361 frameRate = frameRateConstraint->valueForCapabilityRange(this->frameRate(), range.rangeMin().asDouble, range.rangeMax().asDouble);
362 }
363 }
364
365 // Each of the non-null values is supported individually, see if they all can be applied at the same time.
366 if (!supportsSizeAndFrameRate(WTFMove(width), WTFMove(height), WTFMove(frameRate))) {
367 // Let's try without frame rate constraint if not mandatory.
368 if (frameRateConstraint && !frameRateConstraint->isMandatory() && supportsSizeAndFrameRate(WTFMove(width), WTFMove(height), { }))
369 return true;
370
371 if (widthConstraint)
372 badConstraint = widthConstraint->name();
373 else if (heightConstraint)
374 badConstraint = heightConstraint->name();
375 else
376 badConstraint = frameRateConstraint->name();
377 return false;
378 }
379
380 return true;
381}

383double RealtimeMediaSource::fitnessDistance(const MediaConstraint& constraint)
384{
385 auto& capabilities = this->capabilities();
386
387 switch (constraint.constraintType()) {
388 case MediaConstraintType::Width: {
389 ASSERT(constraint.isInt());
390 if (!capabilities.supportsWidth())
391 return 0;
392
393 auto range = capabilities.width();
394 return downcast<IntConstraint>(constraint).fitnessDistance(range.rangeMin().asInt, range.rangeMax().asInt);
395 break;
396 }
397
398 case MediaConstraintType::Height: {
399 ASSERT(constraint.isInt());
400 if (!capabilities.supportsHeight())
401 return 0;
402
403 auto range = capabilities.height();
404 return downcast<IntConstraint>(constraint).fitnessDistance(range.rangeMin().asInt, range.rangeMax().asInt);
405 break;
406 }
407
408 case MediaConstraintType::FrameRate: {
409 ASSERT(constraint.isDouble());
410 if (!capabilities.supportsFrameRate())
411 return 0;
412
413 auto range = capabilities.frameRate();
414 return downcast<DoubleConstraint>(constraint).fitnessDistance(range.rangeMin().asDouble, range.rangeMax().asDouble);
415 break;
416 }
417
418 case MediaConstraintType::AspectRatio: {
419 ASSERT(constraint.isDouble());
420 if (!capabilities.supportsAspectRatio())
421 return 0;
422
423 auto range = capabilities.aspectRatio();
424 return downcast<DoubleConstraint>(constraint).fitnessDistance(range.rangeMin().asDouble, range.rangeMax().asDouble);
425 break;
426 }
427
428 case MediaConstraintType::Volume: {
429 ASSERT(constraint.isDouble());
430 if (!capabilities.supportsVolume())
431 return 0;
432
433 auto range = capabilities.volume();
434 return downcast<DoubleConstraint>(constraint).fitnessDistance(range.rangeMin().asDouble, range.rangeMax().asDouble);
435 break;
436 }
437
438 case MediaConstraintType::SampleRate: {
439 ASSERT(constraint.isInt());
440 if (!capabilities.supportsSampleRate())
441 return 0;
442
443 if (auto discreteRates = discreteSampleRates())
444 return downcast<IntConstraint>(constraint).fitnessDistance(*discreteRates);
445
446 auto range = capabilities.sampleRate();
447 return downcast<IntConstraint>(constraint).fitnessDistance(range.rangeMin().asInt, range.rangeMax().asInt);
448 break;
449 }
450
451 case MediaConstraintType::SampleSize: {
452 ASSERT(constraint.isInt());
453 if (!capabilities.supportsSampleSize())
454 return 0;
455
456 if (auto discreteSizes = discreteSampleSizes())
457 return downcast<IntConstraint>(constraint).fitnessDistance(*discreteSizes);
458
459 auto range = capabilities.sampleSize();
460 return downcast<IntConstraint>(constraint).fitnessDistance(range.rangeMin().asInt, range.rangeMax().asInt);
461 break;
462 }
463
464 case MediaConstraintType::FacingMode: {
465 ASSERT(constraint.isString());
466 if (!capabilities.supportsFacingMode())
467 return 0;
468
469 auto& modes = capabilities.facingMode();
470 Vector<String> supportedModes;
471 supportedModes.reserveInitialCapacity(modes.size());
472 for (auto& mode : modes)
473 supportedModes.uncheckedAppend(RealtimeMediaSourceSettings::facingMode(mode));
474 return downcast<StringConstraint>(constraint).fitnessDistance(supportedModes);
475 break;
476 }
477
478 case MediaConstraintType::EchoCancellation: {
479 ASSERT(constraint.isBoolean());
480 if (!capabilities.supportsEchoCancellation())
481 return 0;
482
483 bool echoCancellationReadWrite = capabilities.echoCancellation() == RealtimeMediaSourceCapabilities::EchoCancellation::ReadWrite;
484 return downcast<BooleanConstraint>(constraint).fitnessDistance(echoCancellationReadWrite);
485 break;
486 }
487
488 case MediaConstraintType::DeviceId:
489 ASSERT(!m_hashedID.isEmpty());
490 return downcast<StringConstraint>(constraint).fitnessDistance(m_hashedID);
491 break;
492
493 case MediaConstraintType::GroupId: {
494 ASSERT(constraint.isString());
495 if (!capabilities.supportsDeviceId())
496 return 0;
497
498 return downcast<StringConstraint>(constraint).fitnessDistance(settings().groupId());
499 break;
500 }
501
502 case MediaConstraintType::DisplaySurface:
503 case MediaConstraintType::LogicalSurface:
504 break;
505
506 case MediaConstraintType::Unknown:
507 // Unknown (or unsupported) constraints should be ignored.
508 break;
509 }
510
511 return 0;
512}

514template <typename ValueType>
515static void applyNumericConstraint(const NumericConstraint<ValueType>& constraint, ValueType current, std::optional<Vector<ValueType>> discreteCapabilityValues, ValueType capabilityMin, ValueType capabilityMax, RealtimeMediaSource& source, void (RealtimeMediaSource::*applier)(ValueType))
516{
517 if (discreteCapabilityValues) {
518 auto value = constraint.valueForDiscreteCapabilityValues(current, *discreteCapabilityValues);
519 if (value && *value != current)
520 (source.*applier)(*value);
521 return;
522 }
523
524 ValueType value = constraint.valueForCapabilityRange(current, capabilityMin, capabilityMax);
525 if (value != current)
526 (source.*applier)(value);
527}

541void RealtimeMediaSource::applyConstraint(const MediaConstraint& constraint)
542{
543 ALWAYS_LOG_IF(m_logger, LOGIDENTIFIER, constraint.name());
544
545 auto& capabilities = this->capabilities();
546 switch (constraint.constraintType()) {
547 case MediaConstraintType::Width:
548 ASSERT_NOT_REACHED();
549 break;
550
551 case MediaConstraintType::Height:
552 ASSERT_NOT_REACHED();
553 break;
554
555 case MediaConstraintType::FrameRate:
556 ASSERT_NOT_REACHED();
557 break;
558
559 case MediaConstraintType::AspectRatio: {
560 ASSERT(constraint.isDouble());
561 if (!capabilities.supportsAspectRatio())
562 return;
563
564 auto range = capabilities.aspectRatio();
565 applyNumericConstraint(downcast<DoubleConstraint>(constraint), aspectRatio(), { }, range.rangeMin().asDouble, range.rangeMax().asDouble, *this, &RealtimeMediaSource::setAspectRatio);
566 break;
567 }
568
569 case MediaConstraintType::Volume: {
570 ASSERT(constraint.isDouble());
571 if (!capabilities.supportsVolume())
572 return;
573
574 auto range = capabilities.volume();
575 applyNumericConstraint(downcast<DoubleConstraint>(constraint), volume(), { }, range.rangeMin().asDouble, range.rangeMax().asDouble, *this, &RealtimeMediaSource::setVolume);
576 break;
577 }
578
579 case MediaConstraintType::SampleRate: {
580 ASSERT(constraint.isInt());
581 if (!capabilities.supportsSampleRate())
582 return;
583
584 auto range = capabilities.sampleRate();
585 applyNumericConstraint(downcast<IntConstraint>(constraint), sampleRate(), discreteSampleRates(), range.rangeMin().asInt, range.rangeMax().asInt, *this, &RealtimeMediaSource::setSampleRate);
586 break;
587 }
588
589 case MediaConstraintType::SampleSize: {
590 ASSERT(constraint.isInt());
591 if (!capabilities.supportsSampleSize())
592 return;
593
594 auto range = capabilities.sampleSize();
595 applyNumericConstraint(downcast<IntConstraint>(constraint), sampleSize(), { }, range.rangeMin().asInt, range.rangeMax().asInt, *this, &RealtimeMediaSource::setSampleSize);
596 break;
597 }
598
599 case MediaConstraintType::EchoCancellation: {
600 ASSERT(constraint.isBoolean());
601 if (!capabilities.supportsEchoCancellation())
602 return;
603
604 bool setting;
605 const BooleanConstraint& boolConstraint = downcast<BooleanConstraint>(constraint);
606 if (boolConstraint.getExact(setting) || boolConstraint.getIdeal(setting))
607 setEchoCancellation(setting);
608 break;
609 }
610
611 case MediaConstraintType::FacingMode: {
612 ASSERT(constraint.isString());
613 if (!capabilities.supportsFacingMode())
614 return;
615
616 auto& supportedModes = capabilities.facingMode();
617 auto filter = [supportedModes](const String& modeString) {
618 auto mode = RealtimeMediaSourceSettings::videoFacingModeEnum(modeString);
619 for (auto& supportedMode : supportedModes) {
620 if (mode == supportedMode)
621 return true;
622 }
623 return false;
624 };
625
626 auto modeString = downcast<StringConstraint>(constraint).find(WTFMove(filter));
627 if (!modeString.isEmpty())
628 setFacingMode(RealtimeMediaSourceSettings::videoFacingModeEnum(modeString));
629 break;
630 }
631
632 case MediaConstraintType::DeviceId:
633 case MediaConstraintType::GroupId:
634 ASSERT(constraint.isString());
635 // There is nothing to do here, neither can be changed.
636 break;
637
638 case MediaConstraintType::DisplaySurface:
639 case MediaConstraintType::LogicalSurface:
640 ASSERT(constraint.isBoolean());
641 break;
642
643 case MediaConstraintType::Unknown:
644 break;
645 }
646}

648bool RealtimeMediaSource::selectSettings(const MediaConstraints& constraints, FlattenedConstraint& candidates, String& failedConstraint)
649{
650 double minimumDistance = std::numeric_limits<double>::infinity();
651
652 // https://w3c.github.io/mediacapture-main/#dfn-selectsettings
653 //
654 // 1. Each constraint specifies one or more values (or a range of values) for its property.
655 // A property may appear more than once in the list of 'advanced' ConstraintSets. If an
656 // empty object or list has been given as the value for a constraint, it must be interpreted
657 // as if the constraint were not specified (in other words, an empty constraint == no constraint).
658 //
659 // Note that unknown properties are discarded by WebIDL, which means that unknown/unsupported required
660 // constraints will silently disappear. To avoid this being a surprise, application authors are
661 // expected to first use the getSupportedConstraints() method as shown in the Examples below.
662
663 // 2. Let object be the ConstrainablePattern object on which this algorithm is applied. Let copy be an
664 // unconstrained copy of object (i.e., copy should behave as if it were object with all ConstraintSets
665 // removed.)
666
667 // 3. For every possible settings dictionary of copy compute its fitness distance, treating bare values of
668 // properties as ideal values. Let candidates be the set of settings dictionaries for which the fitness
669 // distance is finite.
670
671 failedConstraint = emptyString();
672
673 // Check width, height and frame rate jointly, because while they may be supported individually the combination may not be supported.
674 double distance = std::numeric_limits<double>::infinity();
675 if (!supportsSizeAndFrameRate(constraints.mandatoryConstraints.width(), constraints.mandatoryConstraints.height(), constraints.mandatoryConstraints.frameRate(), failedConstraint, minimumDistance))
676 return false;
677
678 constraints.mandatoryConstraints.filter([&](auto& constraint) {
679 if (!supportsConstraint(constraint))
680 return false;
681
682 if (constraint.constraintType() == MediaConstraintType::Width || constraint.constraintType() == MediaConstraintType::Height || constraint.constraintType() == MediaConstraintType::FrameRate) {
683 candidates.set(constraint);
684 return false;
685 }
686
687 double constraintDistance = fitnessDistance(constraint);
688 if (std::isinf(constraintDistance)) {
689 WTFLogAlways("RealtimeMediaSource::selectSettings failed constraint %d", static_cast<int>(constraint.constraintType()));
690 failedConstraint = constraint.name();
691 return true;
692 }
693
694 distance = std::min(distance, constraintDistance);
695 candidates.set(constraint);
696 return false;
697 });
698
699 if (!failedConstraint.isEmpty())
700 return false;
701
702 minimumDistance = distance;
703
704 // 4. If candidates is empty, return undefined as the result of the SelectSettings() algorithm.
705 if (candidates.isEmpty())
706 return true;
707
708 // 5. Iterate over the 'advanced' ConstraintSets in newConstraints in the order in which they were specified.
709 // For each ConstraintSet:
710
711 // 5.1 compute the fitness distance between it and each settings dictionary in candidates, treating bare
712 // values of properties as exact.
713 Vector<std::pair<double, MediaTrackConstraintSetMap>> supportedConstraints;
714
715 for (const auto& advancedConstraint : constraints.advancedConstraints) {
716 double constraintDistance = 0;
717 bool supported = false;
718
719 if (advancedConstraint.width() || advancedConstraint.height() || advancedConstraint.frameRate()) {
720 String dummy;
721 if (!supportsSizeAndFrameRate(advancedConstraint.width(), advancedConstraint.height(), advancedConstraint.frameRate(), dummy, constraintDistance))
722 continue;
723
724 supported = true;
725 }
726
727 advancedConstraint.forEach([&](const MediaConstraint& constraint) {
728
729 if (constraint.constraintType() == MediaConstraintType::Width || constraint.constraintType() == MediaConstraintType::Height || constraint.constraintType() == MediaConstraintType::FrameRate)
730 return;
731
732 distance = fitnessDistance(constraint);
733 constraintDistance += distance;
734 if (!std::isinf(distance))
735 supported = true;
736 });
737
738 minimumDistance = std::min(minimumDistance, constraintDistance);
739
740 // 5.2 If the fitness distance is finite for one or more settings dictionaries in candidates, keep those
741 // settings dictionaries in candidates, discarding others.
742 // If the fitness distance is infinite for all settings dictionaries in candidates, ignore this ConstraintSet.
743 if (supported)
744 supportedConstraints.append({constraintDistance, advancedConstraint});
745 }
746
747 // 6. Select one settings dictionary from candidates, and return it as the result of the SelectSettings() algorithm.
748 // The UA should use the one with the smallest fitness distance, as calculated in step 3.
749 if (!supportedConstraints.isEmpty()) {
750 supportedConstraints.removeAllMatching([&](const std::pair<double, MediaTrackConstraintSetMap>& pair) -> bool {
751 return std::isinf(pair.first) || pair.first > minimumDistance;
752 });
753
754 if (!supportedConstraints.isEmpty()) {
755 auto& advancedConstraint = supportedConstraints[0].second;
756 advancedConstraint.forEach([&](const MediaConstraint& constraint) {
757 candidates.merge(constraint);
758 });
759
760 minimumDistance = std::min(minimumDistance, supportedConstraints[0].first);
761 }
762 }
763
764 return true;

844bool RealtimeMediaSource::supportsConstraints(const MediaConstraints& constraints, String& invalidConstraint)
845{
846 ASSERT(constraints.isValid);
847
848 ALWAYS_LOG_IF(m_logger, LOGIDENTIFIER);
849
850 FlattenedConstraint candidates;
851 if (!selectSettings(constraints, candidates, invalidConstraint))
852 return false;
853
854 m_fitnessScore = 0;
855 for (auto& variant : candidates) {
856 double distance = fitnessDistance(variant);
857 switch (variant.constraintType()) {
858 case MediaConstraintType::DeviceId:
859 case MediaConstraintType::FacingMode:
860 m_fitnessScore += distance ? 1 : 32;
861 break;
862
863 case MediaConstraintType::Width:
864 case MediaConstraintType::Height:
865 case MediaConstraintType::FrameRate:
866 case MediaConstraintType::AspectRatio:
867 case MediaConstraintType::Volume:
868 case MediaConstraintType::SampleRate:
869 case MediaConstraintType::SampleSize:
870 case MediaConstraintType::EchoCancellation:
871 case MediaConstraintType::GroupId:
872 case MediaConstraintType::DisplaySurface:
873 case MediaConstraintType::LogicalSurface:
874 case MediaConstraintType::Unknown:
875 m_fitnessScore += distance ? 1 : 2;
876 break;
877 }
878 }
879
880 ALWAYS_LOG_IF(m_logger, LOGIDENTIFIER, "fitness distance : ", m_fitnessScore);
881
882 return true;

885void RealtimeMediaSource::applyConstraints(const FlattenedConstraint& constraints)
886{
887 if (constraints.isEmpty())
888 return;
889
890 beginConfiguration();
891
892 auto& capabilities = this->capabilities();
893
894 std::optional<int> width;
895 if (const MediaConstraint* constraint = constraints.find(MediaConstraintType::Width)) {
896 ASSERT(constraint->isInt());
897 if (capabilities.supportsWidth()) {
898 auto range = capabilities.width();
899 width = downcast<IntConstraint>(*constraint).valueForCapabilityRange(size().width(), range.rangeMin().asInt, range.rangeMax().asInt);
900 }
901 }
902
903 std::optional<int> height;
904 if (const MediaConstraint* constraint = constraints.find(MediaConstraintType::Height)) {
905 ASSERT(constraint->isInt());
906 if (capabilities.supportsHeight()) {
907 auto range = capabilities.height();
908 height = downcast<IntConstraint>(*constraint).valueForCapabilityRange(size().height(), range.rangeMin().asInt, range.rangeMax().asInt);
909 }
910 }
911
912 std::optional<double> frameRate;
913 if (const MediaConstraint* constraint = constraints.find(MediaConstraintType::FrameRate)) {
914 ASSERT(constraint->isDouble());
915 if (capabilities.supportsFrameRate()) {
916 auto range = capabilities.frameRate();
917 frameRate = downcast<DoubleConstraint>(*constraint).valueForCapabilityRange(this->frameRate(), range.rangeMin().asDouble, range.rangeMax().asDouble);
918 }
919 }
920
921 if (width || height || frameRate)
922 setSizeAndFrameRate(WTFMove(width), WTFMove(height), WTFMove(frameRate));
923
924 for (auto& variant : constraints) {
925 if (variant.constraintType() == MediaConstraintType::Width || variant.constraintType() == MediaConstraintType::Height || variant.constraintType() == MediaConstraintType::FrameRate)
926 continue;
927
928 applyConstraint(variant);
929 }
930
931 commitConfiguration();
932}
