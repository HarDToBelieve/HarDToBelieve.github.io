31namespace WebCore {

33bool RealtimeMediaSourceSupportedConstraints::supportsConstraint(MediaConstraintType constraint) const
34{
35 switch (constraint) {
36 case MediaConstraintType::Unknown:
37 return false;
38 case MediaConstraintType::Width:
39 return supportsWidth();
40 case MediaConstraintType::Height:
41 return supportsHeight();
42 case MediaConstraintType::AspectRatio:
43 return supportsAspectRatio();
44 case MediaConstraintType::FrameRate:
45 return supportsFrameRate();
46 case MediaConstraintType::FacingMode:
47 return supportsFacingMode();
48 case MediaConstraintType::Volume:
49 return supportsVolume();
50 case MediaConstraintType::SampleRate:
51 return supportsSampleRate();
52 case MediaConstraintType::SampleSize:
53 return supportsSampleSize();
54 case MediaConstraintType::EchoCancellation:
55 return supportsEchoCancellation();
56 case MediaConstraintType::DeviceId:
57 return supportsDeviceId();
58 case MediaConstraintType::GroupId:
59 return supportsGroupId();
60 case MediaConstraintType::DisplaySurface:
61 return supportsDisplaySurface();
62 case MediaConstraintType::LogicalSurface:
63 return supportsLogicalSurface();
64 }
65
66 ASSERT_NOT_REACHED();
67 return false;
68}
