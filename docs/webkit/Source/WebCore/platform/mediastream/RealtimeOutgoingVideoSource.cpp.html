47RealtimeOutgoingVideoSource::RealtimeOutgoingVideoSource(Ref<MediaStreamTrackPrivate>&& videoSource)
48 : m_videoSource(WTFMove(videoSource))
49 , m_blackFrameTimer(*this, &RealtimeOutgoingVideoSource::sendOneBlackFrame)
50#if !RELEASE_LOG_DISABLED
51 , m_logger(m_videoSource->logger())
52 , m_logIdentifier(m_videoSource->logIdentifier())
53#endif
54{
55}
56
57RealtimeOutgoingVideoSource::~RealtimeOutgoingVideoSource()
58{
59ASSERT(!m_videoSource->hasObserver(*this));
60#if ASSERT_ENABLED
61 Locker locker { m_sinksLock };
62#endif
63 ASSERT(m_sinks.isEmpty());
64 stop();
65}
66
67void RealtimeOutgoingVideoSource::observeSource()
68{
69 ASSERT(!m_videoSource->hasObserver(*this));
70 m_videoSource->addObserver(*this);
71 initializeFromSource();
72}
73
74void RealtimeOutgoingVideoSource::unobserveSource()
75{
76 m_videoSource->removeObserver(*this);
77 m_videoSource->source().removeVideoSampleObserver(*this);
78}
79
80void RealtimeOutgoingVideoSource::setSource(Ref<MediaStreamTrackPrivate>&& newSource)
81{
82 ASSERT(isMainThread());
83 ASSERT(!m_videoSource->hasObserver(*this));
84 m_videoSource = WTFMove(newSource);
85
86 if (!m_areSinksAskingToApplyRotation)
87 return;
88 if (!m_videoSource->source().setShouldApplyRotation(true))
89 m_shouldApplyRotation = true;
90}
91
92void RealtimeOutgoingVideoSource::applyRotation()
93{
94 ensureOnMainThread([this, protectedThis = makeRef(*this)] {
95 if (m_areSinksAskingToApplyRotation)
96 return;
97
98 m_areSinksAskingToApplyRotation = true;
99 if (!m_videoSource->source().setShouldApplyRotation(true))
100 m_shouldApplyRotation = true;
101 });
102}
103
104void RealtimeOutgoingVideoSource::stop()
105{
106 ASSERT(isMainThread());
107 unobserveSource();
108 m_blackFrameTimer.stop();
109}
110
111void RealtimeOutgoingVideoSource::updateBlackFramesSending()
112{
113 if (!m_muted && m_enabled) {
114 m_videoSource->source().addVideoSampleObserver(*this);
115 if (m_blackFrameTimer.isActive())
116 m_blackFrameTimer.stop();
117 return;
118 }
119
120 m_videoSource->source().removeVideoSampleObserver(*this);
121 sendBlackFramesIfNeeded();
122}
123
124void RealtimeOutgoingVideoSource::sourceMutedChanged()
125{
126 ASSERT(m_muted != m_videoSource->muted());
127
128 m_muted = m_videoSource->muted();
129
130 updateBlackFramesSending();
131}
132
133void RealtimeOutgoingVideoSource::sourceEnabledChanged()
134{
135 ASSERT(m_enabled != m_videoSource->enabled());
136
137 m_enabled = m_videoSource->enabled();
138
139 updateBlackFramesSending();
140}

142void RealtimeOutgoingVideoSource::initializeFromSource()
143{
144 const auto& settings = m_videoSource->source().settings();
145 m_width = settings.width();
146 m_height = settings.height();
147
148 m_muted = m_videoSource->muted();
149 m_enabled = m_videoSource->enabled();
150
151 updateBlackFramesSending();
152}

154void RealtimeOutgoingVideoSource::AddOrUpdateSink(rtc::VideoSinkInterface<webrtc::VideoFrame>* sink, const rtc::VideoSinkWants& sinkWants)
155{
156 ASSERT(!sinkWants.black_frames);
157
158 if (sinkWants.rotation_applied)
159 applyRotation();
160
161 Locker locker { m_sinksLock };
162 m_sinks.add(sink);
163}

165void RealtimeOutgoingVideoSource::RemoveSink(rtc::VideoSinkInterface<webrtc::VideoFrame>* sink)
166{
167 Locker locker { m_sinksLock };
168 m_sinks.remove(sink);
169}

171void RealtimeOutgoingVideoSource::sendBlackFramesIfNeeded()
172{
173 if (m_blackFrameTimer.isActive())
174 return;
175
176 if (!m_muted && m_enabled)
177 return;
178
179 if (!m_width || !m_height)
180 return;
181
182 if (!m_blackFrame) {
183 auto width = m_width;
184 auto height = m_height;
185 if (m_shouldApplyRotation && (m_currentRotation == webrtc::kVideoRotation_270 || m_currentRotation == webrtc::kVideoRotation_90))
186 std::swap(width, height);
187 m_blackFrame = createBlackFrame(width, height);
188 ASSERT(m_blackFrame);
189 if (!m_blackFrame) {
190 ALWAYS_LOG(LOGIDENTIFIER, "Unable to send black frames");
191 return;
192 }
193 }
194 sendOneBlackFrame();
195 m_blackFrameTimer.startRepeating(1_s);
196}
197
198void RealtimeOutgoingVideoSource::sendOneBlackFrame()
199{
200 ALWAYS_LOG(LOGIDENTIFIER);
201 sendFrame(rtc::scoped_refptr<webrtc::VideoFrameBuffer>(m_blackFrame));
202}
203
204void RealtimeOutgoingVideoSource::sendFrame(rtc::scoped_refptr<webrtc::VideoFrameBuffer>&& buffer)
205{
206 MonotonicTime timestamp = MonotonicTime::now();
207 webrtc::VideoFrame frame(buffer, m_shouldApplyRotation ? webrtc::kVideoRotation_0 : m_currentRotation, static_cast<int64_t>(timestamp.secondsSinceEpoch().microseconds()));
208
209#if !RELEASE_LOG_DISABLED
210 ++m_frameCount;
211
212 auto delta = timestamp - m_lastFrameLogTime;
213 if (!m_lastFrameLogTime || delta >= 1_s) {
214 if (m_lastFrameLogTime) {
215 INFO_LOG(LOGIDENTIFIER, m_frameCount, " frames sent in ", delta.value(), " seconds");
216 m_frameCount = 0;
217 }
218 m_lastFrameLogTime = timestamp;
219 }
220#endif
221
222 Locker locker { m_sinksLock };
223 for (auto* sink : m_sinks)
224 sink->OnFrame(frame);
225}
