37RealtimeVideoSource::RealtimeVideoSource(Ref<RealtimeVideoCaptureSource>&& source)
38 : RealtimeMediaSource(Type::Video, String { source->name() }, String { source->persistentID() }, String { source->deviceIDHashSalt() })
39 , m_source(WTFMove(source))
40{
41 m_source->addObserver(*this);
42 m_currentSettings = m_source->settings();
43 setSize(m_source->size());
44 setFrameRate(m_source->frameRate());
45}

47RealtimeVideoSource::~RealtimeVideoSource()
48{
49 m_source->removeVideoSampleObserver(*this);
50 m_source->removeObserver(*this);
51}

53void RealtimeVideoSource::whenReady(CompletionHandler<void(String)>&& callback)
54{
55 m_source->whenReady([this, protectedThis = makeRef(*this), callback = WTFMove(callback)](auto message) mutable {

57 m_currentSettings = m_source->settings();

59 setFrameRate(m_source->frameRate());

61 });
62}

81void RealtimeVideoSource::setSizeAndFrameRate(std::optional<int> width, std::optional<int> height, std::optional<double> frameRate)
82{
83 if (!width && !height) {
84 width = size().width();
85 height = size().height();
86 }
87
88 m_source->clientUpdatedSizeAndFrameRate(width, height, frameRate);
89 auto sourceSize = m_source->size();
90 ASSERT(sourceSize.height());
91 ASSERT(sourceSize.width());
92
93 if (!width)
94 width = sourceSize.width() * height.value() / sourceSize.height();
95 m_currentSettings.setWidth(*width);
96
97 if (!height)
98 height = sourceSize.height() * width.value() / sourceSize.width();
99 m_currentSettings.setHeight(*height);
100
101 if (frameRate)
102 m_currentSettings.setFrameRate(static_cast<float>(*frameRate));
103
104 RealtimeMediaSource::setSizeAndFrameRate(width, height, frameRate);
105}

112void RealtimeVideoSource::sourceSettingsChanged()
113{
114 auto rotation = m_source->sampleRotation();
115 auto size = this->size();
116 if (size.isEmpty())
117 size = m_source->size();
118 if (rotation == MediaSample::VideoRotation::Left || rotation == MediaSample::VideoRotation::Right)
119 size = size.transposedSize();
120
121 m_currentSettings.setWidth(size.width());
122 m_currentSettings.setHeight(size.height());
123
124 forEachObserver([&](auto& observer) {
125 observer.sourceSettingsChanged();
126 });
127}

129bool RealtimeVideoSource::preventSourceFromStopping()
130{
131 if (!isProducingData())
132 return false;
133
134 bool hasObserverPreventingStopping = false;
135 forEachObserver([&](auto& observer) {
136 if (observer.preventSourceFromStopping())
137 hasObserverPreventingStopping = true;
138 });
139 return hasObserverPreventingStopping;
140}

152void RealtimeVideoSource::sourceStopped()
153{
154 if (m_source->captureDidFail()) {
155 captureFailed();
156 return;
157 }
158 stop();
159 forEachObserver([](auto& observer) {
160 observer.sourceStopped();
161 });
162}

165RefPtr<MediaSample> RealtimeVideoSource::adaptVideoSample(MediaSample& sample)
166{
167 if (!m_imageTransferSession || m_imageTransferSession->pixelFormat() != sample.videoPixelFormat())
168 m_imageTransferSession = ImageTransferSessionVT::create(sample.videoPixelFormat());
169
170 ASSERT(m_imageTransferSession);
171 if (!m_imageTransferSession)
172 return nullptr;
173
174 auto mediaSample = m_imageTransferSession->convertMediaSample(sample, size());
175 ASSERT(mediaSample);
176
177 return mediaSample;
178}

181void RealtimeVideoSource::videoSampleAvailable(MediaSample& sample)
182{
183 if (m_frameDecimation > 1 && ++m_frameDecimationCounter % m_frameDecimation)
184 return;
185
186 auto frameRate = this->frameRate();
187 m_frameDecimation = frameRate ? static_cast<size_t>(m_source->observedFrameRate() / frameRate) : 1;
188 if (!m_frameDecimation)
189 m_frameDecimation = 1;
190
191#if PLATFORM(COCOA)
192 auto size = this->size();
193 if (!size.isEmpty() && size != expandedIntSize(sample.presentationSize())) {
194 if (auto mediaSample = adaptVideoSample(sample)) {
195 RealtimeMediaSource::videoSampleAvailable(*mediaSample);
196 return;
197 }
198 }
199#endif
200
201 RealtimeMediaSource::videoSampleAvailable(sample);
202}

204Ref<RealtimeMediaSource> RealtimeVideoSource::clone()
205{
206 auto source = create(m_source.copyRef());
207 source->m_currentSettings = m_currentSettings;
208 source->setSize(size());
209#if !RELEASE_LOG_DISABLED
210 source->setLogger(logger(), childLogIdentifier(logIdentifier(), ++m_cloneCounter));
211#endif
212 return source;
213}
