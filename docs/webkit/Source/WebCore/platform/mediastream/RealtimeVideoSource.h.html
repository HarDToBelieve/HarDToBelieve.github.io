27
28#if ENABLE(MEDIA_STREAM)
29
30#include "RealtimeVideoCaptureSource.h"
31
32namespace WebCore {
33
34class ImageTransferSessionVT;
35
36class RealtimeVideoSource final
37 : public RealtimeMediaSource
38 , public RealtimeMediaSource::Observer
39 , public RealtimeMediaSource::VideoSampleObserver {
40public:
41 static Ref<RealtimeVideoSource> create(Ref<RealtimeVideoCaptureSource>&& source) { return adoptRef(*new RealtimeVideoSource(WTFMove(source))); }
42
43 Vector<VideoPresetData> presetsData() { return m_source->presetsData(); }
44
45private:
46 WEBCORE_EXPORT explicit RealtimeVideoSource(Ref<RealtimeVideoCaptureSource>&&);
47 ~RealtimeVideoSource();
48
49 // RealtimeMediaSource
50 void startProducingData() final;
51 void stopProducingData() final;
52 bool supportsSizeAndFrameRate(std::optional<int> width, std::optional<int> height, std::optional<double> frameRate) final;
53 void setSizeAndFrameRate(std::optional<int> width, std::optional<int> height, std::optional<double> frameRate) final;
54 Ref<RealtimeMediaSource> clone() final;
55 void requestToEnd(RealtimeMediaSource::Observer& callingObserver) final;
56 void stopBeingObserved() final;
57
58 const RealtimeMediaSourceCapabilities& capabilities() final { return m_source->capabilities(); }
59 const RealtimeMediaSourceSettings& settings() final { return m_currentSettings; }
60 bool isCaptureSource() const final { return m_source->isCaptureSource(); }
61 CaptureDevice::DeviceType deviceType() const final { return m_source->deviceType(); }
62 void monitorOrientation(OrientationNotifier& notifier) final { m_source->monitorOrientation(notifier); }
63 bool interrupted() const final { return m_source->interrupted(); }
64 bool isSameAs(RealtimeMediaSource& source) const final { return this == &source || m_source.ptr() == &source; }
65 void whenReady(CompletionHandler<void(String)>&&) final;
66 bool isVideoSource() const final { return true; }
67
68 // RealtimeMediaSource::Observer
69 void sourceMutedChanged() final;
70 void sourceSettingsChanged() final;
71 void sourceStopped() final;
72 bool preventSourceFromStopping() final;
73
74 // RealtimeMediaSource::VideoSampleObserver
75 void videoSampleAvailable(MediaSample&) final;
76
77#if PLATFORM(COCOA)
78 RefPtr<MediaSample> adaptVideoSample(MediaSample&);
79#endif
80
81#if !RELEASE_LOG_DISABLED
82 void setLogger(const Logger&, const void*) final;
83#endif
84
85 Ref<RealtimeVideoCaptureSource> m_source;
86 RealtimeMediaSourceSettings m_currentSettings;
87#if PLATFORM(COCOA)
88 std::unique_ptr<ImageTransferSessionVT> m_imageTransferSession;
89#endif
90 size_t m_frameDecimation { 1 };
91 size_t m_frameDecimationCounter { 0 };
92#if !RELEASE_LOG_DISABLED
93 uint64_t m_cloneCounter { 0 };
94#endif
95};
96
97} // namespace WebCore

99#endif // ENABLE(MEDIA_STREAM)
