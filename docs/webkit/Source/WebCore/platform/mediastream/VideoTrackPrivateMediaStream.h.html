28#if ENABLE(MEDIA_STREAM)
29
30#include "MediaStreamTrackPrivate.h"
31#include "VideoTrackPrivate.h"
32
33namespace WebCore {
34
35class VideoTrackPrivateMediaStream final : public VideoTrackPrivate {
36 WTF_MAKE_NONCOPYABLE(VideoTrackPrivateMediaStream)
37public:
38 static Ref<VideoTrackPrivateMediaStream> create(MediaStreamTrackPrivate& streamTrack)
39 {
40 return adoptRef(*new VideoTrackPrivateMediaStream(streamTrack));
41 }
42
43 void setTrackIndex(int index) { m_index = index; }
44
45 MediaStreamTrackPrivate& streamTrack() { return m_streamTrack.get(); }
46
47private:
48 VideoTrackPrivateMediaStream(MediaStreamTrackPrivate& track)
49 : m_streamTrack(track)
50 , m_id(track.id())
51 , m_label(track.label())
52 {
53 }
54
55 Kind kind() const final { return Kind::Main; }
56 AtomString id() const final { return m_id; }
57 AtomString label() const final { return m_label; }
58 AtomString language() const final { return emptyAtom(); }
59 int trackIndex() const final { return m_index; }
60
61 Ref<MediaStreamTrackPrivate> m_streamTrack;
62 AtomString m_id;
63 AtomString m_label;
64 int m_index { 0 };
65};
66
67}

69#endif // ENABLE(MEDIA_STREAM)
