47static UniqueRef<AudioMediaStreamTrackRendererInternalUnit> createInternalUnit(AudioMediaStreamTrackRendererUnit& unit)
48{
49 AudioMediaStreamTrackRendererInternalUnit::RenderCallback callback = [&unit](auto sampleCount, auto& list, auto sampleTime, auto hostTime, auto& flags) {
50 unit.render(sampleCount, list, sampleTime, hostTime, flags);
51 return 0;
52 };
53
54 auto& function = getCreateInternalUnitFunction();
55 if (function)
56 return function(WTFMove(callback));
57
58 return AudioMediaStreamTrackRendererInternalUnit::createLocalInternalUnit(WTFMove(callback));
59}

82void AudioMediaStreamTrackRendererUnit::addSource(Ref<AudioSampleDataSource>&& source)
83{
84#if !RELEASE_LOG_DISABLED
85 source->logger().logAlways(LogWebRTC, "AudioMediaStreamTrackRendererUnit::addSource ", source->logIdentifier());
86#endif
87 ASSERT(isMainThread());
88
89 ASSERT(!m_sources.contains(source.get()));
90 bool shouldStart = m_sources.isEmpty();
91 m_sources.add(WTFMove(source));
92
93 {
94 Locker locker { m_pendingRenderSourcesLock };
95 m_pendingRenderSources = copyToVector(m_sources);
96 m_hasPendingRenderSources = true;
97 }
98
99 if (shouldStart)
100 start();
101}
102
103void AudioMediaStreamTrackRendererUnit::removeSource(AudioSampleDataSource& source)
104{
105#if !RELEASE_LOG_DISABLED
106 source.logger().logAlways(LogWebRTC, "AudioMediaStreamTrackRendererUnit::removeSource ", source.logIdentifier());
107#endif
108 ASSERT(isMainThread());
109
110 bool shouldStop = !m_sources.isEmpty();
111 m_sources.remove(source);
112 shouldStop &= m_sources.isEmpty();
113
114 {
115 Locker locker { m_pendingRenderSourcesLock };
116 m_pendingRenderSources = copyToVector(m_sources);
117 m_hasPendingRenderSources = true;
118 }
119
120 if (shouldStop)
121 stop();
122}

146void AudioMediaStreamTrackRendererUnit::updateRenderSourcesIfNecessary()
147{
148 if (!m_pendingRenderSourcesLock.tryLock())
149 return;
150
151 Locker locker { AdoptLock, m_pendingRenderSourcesLock };
152 if (!m_hasPendingRenderSources)
153 return;
154
155 m_renderSources = WTFMove(m_pendingRenderSources);
156 m_hasPendingRenderSources = false;
157}
158
159void AudioMediaStreamTrackRendererUnit::render(size_t sampleCount, AudioBufferList& ioData, uint64_t sampleTime, double hostTime, AudioUnitRenderActionFlags& actionFlags)
160{
161 // For performance reasons, we forbid heap allocations while doing rendering on the audio thread.
162 ForbidMallocUseForCurrentThreadScope forbidMallocUse;
163
164 ASSERT(!isMainThread());
165
166 updateRenderSourcesIfNecessary();
167
168 if (m_renderSources.isEmpty()) {
169 actionFlags = kAudioUnitRenderAction_OutputIsSilence;
170 return;
171 }
172
173 // Mix all sources.
174 bool isFirstSource = true;
175 for (auto& source : m_renderSources) {
176 source->pullSamples(ioData, sampleCount, sampleTime, hostTime, isFirstSource ? AudioSampleDataSource::Copy : AudioSampleDataSource::Mix);
177 isFirstSource = false;
178 }
179 return;
180}
