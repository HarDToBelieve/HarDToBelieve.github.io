62void AVCaptureDeviceManager::computeCaptureDevices(CompletionHandler<void()>&& callback)
63{
64 if (!m_isInitialized) {
65 refreshCaptureDevices([this, callback = WTFMove(callback)]() mutable {
66 m_isInitialized = true;
67 callback();
68 });
69 return;
70 }
71 callback();
72}

81inline static bool deviceIsAvailable(AVCaptureDevice *device)
82{
83 if (![device isConnected])
84 return false;
85
86#if !PLATFORM(IOS_FAMILY)
87 if ([device isSuspended])
88 return false;
89#endif
90
91 return true;
92}

94void AVCaptureDeviceManager::updateCachedAVCaptureDevices()
95{
96 ASSERT(!isMainThread());
97 auto* currentDevices = [PAL::getAVCaptureDeviceClass() devices];
98 auto changedDevices = adoptNS([[NSMutableArray alloc] init]);
99 for (AVCaptureDevice *cachedDevice in m_avCaptureDevices.get()) {
100 if (![currentDevices containsObject:cachedDevice])
101 [changedDevices addObject:cachedDevice];
102 }
103
104 if ([changedDevices count]) {
105 for (AVCaptureDevice *device in changedDevices.get())
106 [device removeObserver:m_objcObserver.get() forKeyPath:@"suspended"];
107 [m_avCaptureDevices removeObjectsInArray:changedDevices.get()];
108 }
109
110 for (AVCaptureDevice *device in currentDevices) {
111
112 if (![device hasMediaType:AVMediaTypeVideo] && ![device hasMediaType:AVMediaTypeMuxed])
113 continue;
114
115 if ([m_avCaptureDevices.get() containsObject:device])
116 continue;
117
118 [device addObserver:m_objcObserver.get() forKeyPath:@"suspended" options:NSKeyValueObservingOptionNew context:(void *)nil];
119 [m_avCaptureDevices.get() addObject:device];
120 }
121
122}

124static inline CaptureDevice toCaptureDevice(AVCaptureDevice *device)
125{
126 CaptureDevice captureDevice { device.uniqueID, CaptureDevice::DeviceType::Camera, device.localizedName };
127 captureDevice.setEnabled(deviceIsAvailable(device));
128 return captureDevice;
129}
130
131static inline bool isVideoDevice(AVCaptureDevice *device)
132{
133 return [device hasMediaType:AVMediaTypeVideo] || [device hasMediaType:AVMediaTypeMuxed];
134}

136Vector<CaptureDevice> AVCaptureDeviceManager::retrieveCaptureDevices()
137{
138 ASSERT(!isMainThread());
139 if (!isAvailable())
140 return { };
141
142 if (!m_avCaptureDevices) {
143 m_avCaptureDevices = adoptNS([[NSMutableArray alloc] init]);
144 registerForDeviceNotifications();
145 }
146
147 updateCachedAVCaptureDevices();
148
149 auto* currentDevices = [PAL::getAVCaptureDeviceClass() devices];
150 Vector<CaptureDevice> deviceList;
151
152 auto* defaultVideoDevice = [PAL::getAVCaptureDeviceClass() defaultDeviceWithMediaType: AVMediaTypeVideo];
153#if PLATFORM(IOS)
154 if ([defaultVideoDevice position] != AVCaptureDevicePositionFront) {
155 defaultVideoDevice = nullptr;
156 for (AVCaptureDevice *platformDevice in currentDevices) {
157 if (!isVideoDevice(platformDevice))
158 continue;
159
160 if ([platformDevice position] == AVCaptureDevicePositionFront) {
161 defaultVideoDevice = platformDevice;
162 break;
163 }
164 }
165 }
166#endif
167
168 if (defaultVideoDevice) {
169 auto device = toCaptureDevice(defaultVideoDevice);
170 device.setIsDefault(true);
171 deviceList.append(WTFMove(device));
172 }
173 for (AVCaptureDevice *platformDevice in currentDevices) {
174 if (isVideoDevice(platformDevice) && platformDevice.uniqueID != defaultVideoDevice.uniqueID)
175 deviceList.append(toCaptureDevice(platformDevice));
176 }
177 return deviceList;
178}

180void AVCaptureDeviceManager::refreshCaptureDevices(CompletionHandler<void()>&& callback)
181{
182 m_dispatchQueue->dispatch([this, callback = WTFMove(callback)]() mutable {
183 RunLoop::main().dispatch([this, callback = WTFMove(callback), deviceList = retrieveCaptureDevices().isolatedCopy()]() mutable {
184 bool deviceHasChanged = m_devices.size() != deviceList.size();
185 if (!deviceHasChanged) {
186 for (size_t cptr = 0; cptr < deviceList.size(); ++cptr) {
187 if (m_devices[cptr].persistentId() != deviceList[cptr].persistentId() || m_devices[cptr].enabled() != deviceList[cptr].enabled()) {
188 deviceHasChanged = true;
189 break;
190 }
191 }
192 }
193
194 if (deviceHasChanged) {
195 m_devices = WTFMove(deviceList);
196 if (m_isInitialized)
197 deviceChanged();
198 }
199 callback();
200 });
201 });
202}

215AVCaptureDeviceManager::AVCaptureDeviceManager()
216 : m_objcObserver(adoptNS([[WebCoreAVCaptureDeviceManagerObserver alloc] initWithCallback: this]))
217 , m_dispatchQueue(WorkQueue::create("com.apple.WebKit.AVCaptureDeviceManager"))
218{
219}
220
221AVCaptureDeviceManager::~AVCaptureDeviceManager()
222{
223 [[NSNotificationCenter defaultCenter] removeObserver:m_objcObserver.get()];
224 [m_objcObserver disconnect];
225 for (AVCaptureDevice *device in m_avCaptureDevices.get())
226 [device removeObserver:m_objcObserver.get() forKeyPath:@"suspended"];
227}
228
229void AVCaptureDeviceManager::registerForDeviceNotifications()
230{
231 [[NSNotificationCenter defaultCenter] addObserver:m_objcObserver.get() selector:@selector(deviceConnectedDidChange:) name:AVCaptureDeviceWasConnectedNotification object:nil];
232 [[NSNotificationCenter defaultCenter] addObserver:m_objcObserver.get() selector:@selector(deviceConnectedDidChange:) name:AVCaptureDeviceWasDisconnectedNotification object:nil];
233}

237@implementation WebCoreAVCaptureDeviceManagerObserver
238
239- (id)initWithCallback:(AVCaptureDeviceManager*)callback
240{
241 self = [super init];
242 if (!self)
243 return nil;
244 m_callback = callback;
245 return self;
246}
247
248- (void)disconnect
249{
250 [NSObject cancelPreviousPerformRequestsWithTarget:self];
251 m_callback = nil;
252}
253
254- (void)deviceConnectedDidChange:(NSNotification *)unusedNotification
255{
256 UNUSED_PARAM(unusedNotification);
257 if (!m_callback)
258 return;
259
260 RunLoop::main().dispatch([self, protectedSelf = retainPtr(self)] {
261 if (m_callback)
262 m_callback->refreshCaptureDevices();
263 });
264}
265
266- (void)observeValueForKeyPath:keyPath ofObject:(id)object change:(NSDictionary*)change context:(void*)context
267{
268 UNUSED_PARAM(object);
269 UNUSED_PARAM(context);
270 UNUSED_PARAM(change);
271
272 if (!m_callback)
273 return;
274
275 if ([keyPath isEqualToString:@"suspended"])
276 m_callback->refreshCaptureDevices();
277}
