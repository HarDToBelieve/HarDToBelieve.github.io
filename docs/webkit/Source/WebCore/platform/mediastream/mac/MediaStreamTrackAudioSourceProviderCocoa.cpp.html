35Ref<MediaStreamTrackAudioSourceProviderCocoa> MediaStreamTrackAudioSourceProviderCocoa::create(MediaStreamTrackPrivate& source)
36{
37 return adoptRef(*new MediaStreamTrackAudioSourceProviderCocoa(source));
38}
39
40MediaStreamTrackAudioSourceProviderCocoa::MediaStreamTrackAudioSourceProviderCocoa(MediaStreamTrackPrivate& source)
41 : m_captureSource(makeWeakPtr(source))
42 , m_source(source.source())
43{
44#if USE(LIBWEBRTC)
45 if (m_source->isIncomingAudioSource())
46 setPollSamplesCount(LibWebRTCAudioModule::PollSamplesCount + 1);
47#endif
48}
49
50MediaStreamTrackAudioSourceProviderCocoa::~MediaStreamTrackAudioSourceProviderCocoa()
51{
52 ASSERT(!m_connected);
53 m_source->removeAudioSampleObserver(*this);
54}
55
56void MediaStreamTrackAudioSourceProviderCocoa::hasNewClient(AudioSourceProviderClient* client)
57{
58 bool shouldBeConnected = !!client;
59 if (m_connected == shouldBeConnected)
60 return;
61
62 m_connected = shouldBeConnected;
63 if (!client) {
64 if (m_captureSource)
65 m_captureSource->removeObserver(*this);
66 m_source->removeAudioSampleObserver(*this);
67 return;
68 }
69
70 m_enabled = m_captureSource->enabled();
71 m_captureSource->addObserver(*this);
72 m_source->addAudioSampleObserver(*this);
73}
74
75void MediaStreamTrackAudioSourceProviderCocoa::trackEnabledChanged(MediaStreamTrackPrivate& track)
76{
77 m_enabled = track.enabled();
78}
79
80// May get called on a background thread.
81void MediaStreamTrackAudioSourceProviderCocoa::audioSamplesAvailable(const MediaTime&, const PlatformAudioData& data, const AudioStreamDescription& description, size_t frameCount)
82{
83 if (!m_enabled)
84 return;
85
86 receivedNewAudioSamples(data, description, frameCount);
87}
