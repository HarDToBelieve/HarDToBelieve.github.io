55CaptureSourceOrError MockRealtimeVideoSource::create(String&& deviceID, String&& name, String&& hashSalt, const MediaConstraints* constraints)
56{
57#ifndef NDEBUG
58 auto device = MockRealtimeMediaSourceCenter::mockDeviceWithPersistentID(deviceID);
59 ASSERT(device);
60 if (!device)
61 return { "No mock camera device"_s };
62#endif
63
64 auto source = adoptRef(*new MockRealtimeVideoSourceMac(WTFMove(deviceID), WTFMove(name), WTFMove(hashSalt)));
65 if (constraints) {
66 if (auto error = source->applyConstraints(*constraints))
67 return WTFMove(error->message);
68 }
69
70 return CaptureSourceOrError(RealtimeVideoSource::create(WTFMove(source)));
71}

84void MockRealtimeVideoSourceMac::updateSampleBuffer()
85{
86 auto imageBuffer = this->imageBuffer();
87 if (!imageBuffer)
88 return;
89
90 if (!m_imageTransferSession)
91 m_imageTransferSession = ImageTransferSessionVT::create(preferedPixelBufferFormat());
92
93 PlatformImagePtr platformImage;
94 if (auto nativeImage = imageBuffer->copyImage()->nativeImage())
95 platformImage = nativeImage->platformImage();
96
97 auto sampleTime = MediaTime::createWithDouble((elapsedTime() + 100_ms).seconds());
98 auto sampleBuffer = m_imageTransferSession->createMediaSample(platformImage.get(), sampleTime, size(), sampleRotation());
99 if (!sampleBuffer)
100 return;
101
102 m_workQueue->dispatch([this, protectedThis = makeRef(*this), sampleBuffer = WTFMove(sampleBuffer)]() mutable {
103 dispatchMediaSampleToObservers(*sampleBuffer);
104 });
105}
