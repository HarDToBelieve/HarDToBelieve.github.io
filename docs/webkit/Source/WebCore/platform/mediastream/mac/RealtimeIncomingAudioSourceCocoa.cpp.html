44Ref<RealtimeIncomingAudioSource> RealtimeIncomingAudioSource::create(rtc::scoped_refptr<webrtc::AudioTrackInterface>&& audioTrack, String&& audioTrackId)
45{
46 auto source = RealtimeIncomingAudioSourceCocoa::create(WTFMove(audioTrack), WTFMove(audioTrackId));
47 source->start();

51Ref<RealtimeIncomingAudioSourceCocoa> RealtimeIncomingAudioSourceCocoa::create(rtc::scoped_refptr<webrtc::AudioTrackInterface>&& audioTrack, String&& audioTrackId)
52{
53 return adoptRef(*new RealtimeIncomingAudioSourceCocoa(WTFMove(audioTrack), WTFMove(audioTrackId)));
54}

56static inline AudioStreamBasicDescription streamDescription(size_t sampleRate, size_t channelCount)
57{
58 AudioStreamBasicDescription streamFormat;
59 FillOutASBDForLPCM(streamFormat, sampleRate, channelCount, LibWebRTCAudioFormat::sampleSize, LibWebRTCAudioFormat::sampleSize, LibWebRTCAudioFormat::isFloat, LibWebRTCAudioFormat::isBigEndian, LibWebRTCAudioFormat::isNonInterleaved);
60 return streamFormat;
61}

63RealtimeIncomingAudioSourceCocoa::RealtimeIncomingAudioSourceCocoa(rtc::scoped_refptr<webrtc::AudioTrackInterface>&& audioTrack, String&& audioTrackId)
64 : RealtimeIncomingAudioSource(WTFMove(audioTrack), WTFMove(audioTrackId))
65 , m_sampleRate(LibWebRTCAudioFormat::sampleRate)
66 , m_numberOfChannels(1)
67 , m_streamDescription(streamDescription(m_sampleRate, m_numberOfChannels))
68 , m_audioBufferList(makeUnique<WebAudioBufferList>(m_streamDescription))
69#if !RELEASE_LOG_DISABLED
70 , m_logTimer(*this, &RealtimeIncomingAudioSourceCocoa::logTimerFired)
71#endif
72{
73}

91#if !RELEASE_LOG_DISABLED
92void RealtimeIncomingAudioSourceCocoa::logTimerFired()
93{
94 if (!m_lastChunksReceived || (m_chunksReceived - m_lastChunksReceived) >= ChunksReceivedCountForLogging) {
95 m_lastChunksReceived = m_chunksReceived;
96 ALWAYS_LOG_IF(loggerPtr(), LOGIDENTIFIER, "chunk ", m_chunksReceived);
97 }
98 if (m_audioFormatChanged) {
99 ALWAYS_LOG_IF(loggerPtr(), LOGIDENTIFIER, "new audio buffer list for sampleRate ", m_sampleRate, " and ", m_numberOfChannels, " channel(s)");
100 m_audioFormatChanged = false;
101 }
102}

105void RealtimeIncomingAudioSourceCocoa::OnData(const void* audioData, int bitsPerSample, int sampleRate, size_t numberOfChannels, size_t numberOfFrames)
106{
107 ++m_chunksReceived;
108
109 static constexpr size_t initialSampleRate = 16000;
110 static constexpr size_t initialChunksReceived = 20;
111 // We usually receive some initial callbacks with no data at 16000, then we got real data at the actual sample rate.
112 // To limit reallocations, let's skip these initial calls.
113 if (m_chunksReceived < initialChunksReceived && sampleRate == initialSampleRate)
114 return;
115
116 if (!m_audioBufferList || m_numberOfChannels != numberOfChannels || m_sampleRate != sampleRate) {
117#if !RELEASE_LOG_DISABLED
118 m_audioFormatChanged = true;
119#endif
120
121 m_sampleRate = sampleRate;
122 m_numberOfChannels = numberOfChannels;
123 m_streamDescription = streamDescription(sampleRate, numberOfChannels);
124
125 {
126 DisableMallocRestrictionsForCurrentThreadScope scope;
127 m_audioBufferList = makeUnique<WebAudioBufferList>(m_streamDescription);
128 }
129 if (m_sampleRate && m_numberOfFrames)
130 m_numberOfFrames = m_numberOfFrames * sampleRate / m_sampleRate;
131 else
132 m_numberOfFrames = 0;
133 }
134
135 CMTime startTime = PAL::CMTimeMake(m_numberOfFrames, sampleRate);
136 auto mediaTime = PAL::toMediaTime(startTime);
137 m_numberOfFrames += numberOfFrames;
138
139 auto& bufferList = *m_audioBufferList->buffer(0);
140 bufferList.mDataByteSize = numberOfChannels * numberOfFrames * bitsPerSample / 8;
141 bufferList.mNumberChannels = numberOfChannels;
142 bufferList.mData = const_cast<void*>(audioData);
143
144 audioSamplesAvailable(mediaTime, *m_audioBufferList, m_streamDescription, numberOfFrames);
145}
