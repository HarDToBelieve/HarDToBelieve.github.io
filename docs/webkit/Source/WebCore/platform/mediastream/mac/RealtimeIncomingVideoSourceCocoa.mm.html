63RetainPtr<CVPixelBufferRef> createBlackPixelBuffer(size_t width, size_t height)
64{
65 OSType format = preferedPixelBufferFormat();
66 ASSERT(format == kCVPixelFormatType_420YpCbCr8BiPlanarVideoRange || format == kCVPixelFormatType_420YpCbCr8BiPlanarFullRange);
67
68 CVPixelBufferRef pixelBuffer = nullptr;
69 auto status = CVPixelBufferCreate(kCFAllocatorDefault, width, height, format, nullptr, &pixelBuffer);
70 ASSERT_UNUSED(status, status == noErr);
71
72 status = CVPixelBufferLockBaseAddress(pixelBuffer, 0);
73 ASSERT(status == noErr);
74
75 auto* yPlane = static_cast<uint8_t*>(CVPixelBufferGetBaseAddressOfPlane(pixelBuffer, 0));
76 size_t yStride = CVPixelBufferGetBytesPerRowOfPlane(pixelBuffer, 0);
77 for (unsigned i = 0; i < height; ++i)
78 memset(&yPlane[i * yStride], 0, width);
79
80 auto* uvPlane = static_cast<uint8_t*>(CVPixelBufferGetBaseAddressOfPlane(pixelBuffer, 1));
81 size_t uvStride = CVPixelBufferGetBytesPerRowOfPlane(pixelBuffer, 1);
82 for (unsigned i = 0; i < height / 2; ++i)
83 memset(&uvPlane[i * uvStride], 128, width);
84
85 status = CVPixelBufferUnlockBaseAddress(pixelBuffer, 0);
86 ASSERT(!status);
87 return adoptCF(pixelBuffer);
88}

90CVPixelBufferPoolRef RealtimeIncomingVideoSourceCocoa::pixelBufferPool(size_t width, size_t height, webrtc::BufferType bufferType)
91{
92 if (!m_pixelBufferPool || m_pixelBufferPoolWidth != width || m_pixelBufferPoolHeight != height || m_pixelBufferPoolBufferType != bufferType) {
93 OSType poolBufferType;
94 switch (bufferType) {
95 case webrtc::BufferType::I420:
96 poolBufferType = kCVPixelFormatType_420YpCbCr8BiPlanarVideoRange;
97 break;
98 case webrtc::BufferType::I010:
99 poolBufferType = kCVPixelFormatType_420YpCbCr10BiPlanarVideoRange;
100 }
101 m_pixelBufferPool = createPixelBufferPool(width, height, poolBufferType);
102 m_pixelBufferPoolWidth = width;
103 m_pixelBufferPoolHeight = height;
104 m_pixelBufferPoolBufferType = bufferType;
105 }
106 return m_pixelBufferPool.get();
107}

109RetainPtr<CVPixelBufferRef> RealtimeIncomingVideoSourceCocoa::pixelBufferFromVideoFrame(const webrtc::VideoFrame& frame)
110{
111 if (muted()) {
112 if (!m_blackFrame || m_blackFrameWidth != frame.width() || m_blackFrameHeight != frame.height()) {
113 m_blackFrameWidth = frame.width();
114 m_blackFrameHeight = frame.height();
115 m_blackFrame = createBlackPixelBuffer(m_blackFrameWidth, m_blackFrameHeight);
116 }
117 return m_blackFrame.get();
118 }
119
120 RetainPtr<CVPixelBufferRef> newPixelBuffer;
121 return webrtc::pixelBufferFromFrame(frame, [this, &newPixelBuffer](size_t width, size_t height, webrtc::BufferType bufferType) -> CVPixelBufferRef {
122 auto pixelBufferPool = this->pixelBufferPool(width, height, bufferType);
123 if (!pixelBufferPool)
124 return nullptr;
125
126 CVPixelBufferRef pixelBuffer = nullptr;
127 auto status = CVPixelBufferPoolCreatePixelBuffer(kCFAllocatorDefault, m_pixelBufferPool.get(), &pixelBuffer);
128
129 if (status != kCVReturnSuccess) {
130 ERROR_LOG_IF(loggerPtr(), LOGIDENTIFIER, "Failed creating a pixel buffer with error ", status);
131 return nullptr;
132 }
133 newPixelBuffer = adoptCF(pixelBuffer);
134 return newPixelBuffer.get();
135 });
136}

138void RealtimeIncomingVideoSourceCocoa::OnFrame(const webrtc::VideoFrame& frame)
139{
140 if (!isProducingData())
141 return;
142
143 unsigned width = frame.width();
144 unsigned height = frame.height();
145
146 MediaSample::VideoRotation rotation;
147 switch (frame.rotation()) {
148 case webrtc::kVideoRotation_0:
149 rotation = MediaSample::VideoRotation::None;
150 break;
151 case webrtc::kVideoRotation_180:
152 rotation = MediaSample::VideoRotation::UpsideDown;
153 break;
154 case webrtc::kVideoRotation_90:
155 rotation = MediaSample::VideoRotation::Right;
156 std::swap(width, height);
157 break;
158 case webrtc::kVideoRotation_270:
159 rotation = MediaSample::VideoRotation::Left;
160 std::swap(width, height);
161 break;
162 }
163
164#if !RELEASE_LOG_DISABLED
165 ALWAYS_LOG_IF(loggerPtr() && !(++m_numberOfFrames % 60), LOGIDENTIFIER, "frame ", m_numberOfFrames, ", rotation ", frame.rotation(), " size ", width, "x", height);
166#endif
167
168 auto pixelBuffer = pixelBufferFromVideoFrame(frame);
169 if (!pixelBuffer) {
170 ERROR_LOG_IF(loggerPtr(), LOGIDENTIFIER, "Failed to get a pixel buffer from a frame");
171 return;
172 }
173
174 CMSampleTimingInfo timingInfo;
175 timingInfo.presentationTimeStamp = PAL::CMTimeMake(frame.timestamp_us(), 1000000);
176 timingInfo.decodeTimeStamp = PAL::kCMTimeInvalid;
177 timingInfo.duration = PAL::kCMTimeInvalid;
178
179 CMVideoFormatDescriptionRef formatDescription;
180 OSStatus ostatus = PAL::CMVideoFormatDescriptionCreateForImageBuffer(kCFAllocatorDefault, (CVImageBufferRef)pixelBuffer, &formatDescription);
181 if (ostatus != noErr) {
182 ERROR_LOG_IF(loggerPtr(), LOGIDENTIFIER, "Failed to initialize CMVideoFormatDescription with error ", static_cast<int>(ostatus));
183 return;
184 }
185
186 CMSampleBufferRef sampleBuffer;
187 ostatus = PAL::CMSampleBufferCreateReadyWithImageBuffer(kCFAllocatorDefault, (CVImageBufferRef)pixelBuffer, formatDescription, &timingInfo, &sampleBuffer);
188 CFRelease(formatDescription);
189 if (ostatus != noErr) {
190 ERROR_LOG_IF(loggerPtr(), LOGIDENTIFIER, "Failed to create the sample buffer with error ", static_cast<int>(ostatus));
191 return;
192 }
193
194 auto sample = adoptCF(sampleBuffer);
195
196 setIntrinsicSize(IntSize(width, height));
197 videoSampleAvailable(MediaSampleAVFObjC::create(sample.get(), rotation));
198}
