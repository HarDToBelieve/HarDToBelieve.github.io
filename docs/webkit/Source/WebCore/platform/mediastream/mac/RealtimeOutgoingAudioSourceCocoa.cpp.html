38static inline AudioStreamBasicDescription libwebrtcAudioFormat(Float64 sampleRate, size_t channelCount)
39{
40 // FIXME: Microphones can have more than two channels. In such case, we should do the mix down based on AudioChannelLayoutTag.
41 size_t libWebRTCChannelCount = channelCount >= 2 ? 2 : channelCount;
42 AudioStreamBasicDescription streamFormat;
43 FillOutASBDForLPCM(streamFormat, sampleRate, libWebRTCChannelCount, LibWebRTCAudioFormat::sampleSize, LibWebRTCAudioFormat::sampleSize, LibWebRTCAudioFormat::isFloat, LibWebRTCAudioFormat::isBigEndian, LibWebRTCAudioFormat::isNonInterleaved);
44 return streamFormat;
45}

60bool RealtimeOutgoingAudioSourceCocoa::isReachingBufferedAudioDataHighLimit()
61{
62 auto writtenAudioDuration = m_writeCount / m_inputStreamDescription.sampleRate();
63 auto readAudioDuration = m_readCount / m_outputStreamDescription.sampleRate();
64
65 ASSERT(writtenAudioDuration >= readAudioDuration);
66 return writtenAudioDuration > readAudioDuration + 0.5;
67}
68
69bool RealtimeOutgoingAudioSourceCocoa::isReachingBufferedAudioDataLowLimit()
70{
71 auto writtenAudioDuration = m_writeCount / m_inputStreamDescription.sampleRate();
72 auto readAudioDuration = m_readCount / m_outputStreamDescription.sampleRate();
73
74 ASSERT(writtenAudioDuration >= readAudioDuration);
75 return writtenAudioDuration < readAudioDuration + 0.1;
76}
77
78bool RealtimeOutgoingAudioSourceCocoa::hasBufferedEnoughData()
79{
80 auto writtenAudioDuration = m_writeCount / m_inputStreamDescription.sampleRate();
81 auto readAudioDuration = m_readCount / m_outputStreamDescription.sampleRate();
82
83 ASSERT(writtenAudioDuration >= readAudioDuration);
84 return writtenAudioDuration >= readAudioDuration + 0.01;
85}

87// May get called on a background thread.
88void RealtimeOutgoingAudioSourceCocoa::audioSamplesAvailable(const MediaTime&, const PlatformAudioData& audioData, const AudioStreamDescription& streamDescription, size_t sampleCount)
89{
90 if (m_inputStreamDescription != streamDescription) {
91 if (m_writeCount && m_inputStreamDescription.sampleRate()) {
92 // Update m_writeCount to be valid according the new sampleRate.
93 m_writeCount = (m_writeCount * streamDescription.sampleRate()) / m_inputStreamDescription.sampleRate();
94 }
95
96 m_inputStreamDescription = toCAAudioStreamDescription(streamDescription);
97 auto status = m_sampleConverter->setInputFormat(m_inputStreamDescription);
98 ASSERT_UNUSED(status, !status);
99
100 m_outputStreamDescription = libwebrtcAudioFormat(LibWebRTCAudioFormat::sampleRate, streamDescription.numberOfChannels());
101 status = m_sampleConverter->setOutputFormat(m_outputStreamDescription.streamDescription());
102 ASSERT(!status);
103 }
104
105 // Let's skip pushing samples if we are too slow pulling them.
106 if (m_skippingAudioData) {
107 if (!isReachingBufferedAudioDataLowLimit())
108 return;
109 m_skippingAudioData = false;
110 } else if (isReachingBufferedAudioDataHighLimit()) {
111 m_skippingAudioData = true;
112 return;
113 }
114
115 m_sampleConverter->pushSamples(MediaTime(m_writeCount, static_cast<uint32_t>(m_inputStreamDescription.sampleRate())), audioData, sampleCount);
116 m_writeCount += sampleCount;
117
118 if (!hasBufferedEnoughData())
119 return;
120
121 // Heap allocations are forbidden on the audio thread for performance reasons so we need to
122 // explicitly allow the following allocation(s).
123 DisableMallocRestrictionsForCurrentThreadScope disableMallocRestrictions;
124 LibWebRTCProvider::callOnWebRTCSignalingThread([protectedThis = makeRef(*this)] {
125 protectedThis->pullAudioData();
126 });
127}

129void RealtimeOutgoingAudioSourceCocoa::pullAudioData()
130{
131 // libwebrtc expects 10 ms chunks.
132 size_t chunkSampleCount = m_outputStreamDescription.sampleRate() / 100;
133 size_t bufferSize = chunkSampleCount * LibWebRTCAudioFormat::sampleByteSize * m_outputStreamDescription.numberOfChannels();
134 m_audioBuffer.grow(bufferSize);
135
136 AudioBufferList bufferList;
137 bufferList.mNumberBuffers = 1;
138 bufferList.mBuffers[0].mNumberChannels = m_outputStreamDescription.numberOfChannels();
139 bufferList.mBuffers[0].mDataByteSize = bufferSize;
140 bufferList.mBuffers[0].mData = m_audioBuffer.data();
141
142 if (isSilenced() != m_sampleConverter->muted())
143 m_sampleConverter->setMuted(isSilenced());
144
145 m_sampleConverter->pullAvailableSamplesAsChunks(bufferList, chunkSampleCount, m_readCount, [this, chunkSampleCount] {
146 m_readCount += chunkSampleCount;
147 sendAudioFrames(m_audioBuffer.data(), LibWebRTCAudioFormat::sampleSize, m_outputStreamDescription.sampleRate(), m_outputStreamDescription.numberOfChannels(), chunkSampleCount);
148 });
149}
