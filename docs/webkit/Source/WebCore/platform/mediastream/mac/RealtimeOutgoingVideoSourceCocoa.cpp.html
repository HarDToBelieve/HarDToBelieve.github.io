64void RealtimeOutgoingVideoSourceCocoa::videoSampleAvailable(MediaSample& sample)
65{
66#if !RELEASE_LOG_DISABLED
67 if (!(++m_numberOfFrames % 60))
68 ALWAYS_LOG(LOGIDENTIFIER, "frame ", m_numberOfFrames);
69#endif
70
71 switch (sample.videoRotation()) {
72 case MediaSample::VideoRotation::None:
73 m_currentRotation = webrtc::kVideoRotation_0;
74 break;
75 case MediaSample::VideoRotation::UpsideDown:
76 m_currentRotation = webrtc::kVideoRotation_180;
77 break;
78 case MediaSample::VideoRotation::Right:
79 m_currentRotation = webrtc::kVideoRotation_90;
80 break;
81 case MediaSample::VideoRotation::Left:
82 m_currentRotation = webrtc::kVideoRotation_270;
83 break;
84 }
85
86 ASSERT(sample.platformSample().type == PlatformSample::CMSampleBufferType);
87 auto pixelBuffer = static_cast<CVPixelBufferRef>(PAL::CMSampleBufferGetImageBuffer(sample.platformSample().sample.cmSampleBuffer));
88 auto pixelFormatType = CVPixelBufferGetPixelFormatType(pixelBuffer);
89
90 RetainPtr<CVPixelBufferRef> convertedBuffer = pixelBuffer;
91 if (pixelFormatType != preferedPixelBufferFormat())
92 convertedBuffer = convertToYUV(pixelBuffer);
93
94 if (m_shouldApplyRotation && m_currentRotation != webrtc::kVideoRotation_0)
95 convertedBuffer = rotatePixelBuffer(convertedBuffer.get(), m_currentRotation);
96
97 sendFrame(webrtc::pixelBufferToFrame(convertedBuffer.get()));
98}
