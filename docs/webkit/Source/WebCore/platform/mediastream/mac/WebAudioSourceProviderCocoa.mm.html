66void WebAudioSourceProviderCocoa::provideInput(AudioBus* bus, size_t framesToProcess)
67{
68 if (!m_lock.tryLock()) {
69 bus->zero();
70 return;
71 }
72 Locker locker { AdoptLock, m_lock };
73 if (!m_dataSource || !m_audioBufferList) {
74 bus->zero();
75 return;
76 }
77
78 if (m_writeCount <= m_readCount) {
79 bus->zero();
80 return;
81 }
82
83 if (bus->numberOfChannels() < m_audioBufferList->bufferCount()) {
84 bus->zero();
85 return;
86 }
87
88 for (unsigned i = 0; i < bus->numberOfChannels(); ++i) {
89 auto& channel = *bus->channel(i);
90 if (i >= m_audioBufferList->bufferCount()) {
91 channel.zero();
92 continue;
93 }
94 auto* buffer = m_audioBufferList->buffer(i);
95 buffer->mNumberChannels = 1;
96 buffer->mData = channel.mutableData();
97 buffer->mDataByteSize = channel.length() * sizeof(float);
98 }
99
100 ASSERT(framesToProcess <= bus->length());
101 m_dataSource->pullSamples(*m_audioBufferList->list(), framesToProcess, m_readCount, 0, AudioSampleDataSource::Copy);
102 m_readCount += framesToProcess;
103}

105void WebAudioSourceProviderCocoa::prepare(const AudioStreamBasicDescription& format)
106{
107 DisableMallocRestrictionsForCurrentThreadScope scope;
108
109 Locker locker { m_lock };
110
111 LOG(Media, "WebAudioSourceProviderCocoa::prepare(%p)", this);
112
113 m_inputDescription = CAAudioStreamDescription(format);
114 int numberOfChannels = format.mChannelsPerFrame;
115 double sampleRate = format.mSampleRate;
116 ASSERT(sampleRate >= 0);
117
118 const int bytesPerFloat = sizeof(Float32);
119 const int bitsPerByte = 8;
120 const bool isFloat = true;
121 const bool isBigEndian = false;
122 const bool isNonInterleaved = true;
123 AudioStreamBasicDescription outputDescription { };
124 FillOutASBDForLPCM(outputDescription, sampleRate, numberOfChannels, bitsPerByte * bytesPerFloat, bitsPerByte * bytesPerFloat, isFloat, isBigEndian, isNonInterleaved);
125 m_outputDescription = CAAudioStreamDescription(outputDescription);
126 m_audioBufferList = makeUnique<WebAudioBufferList>(m_outputDescription.value());
127
128 if (!m_dataSource)
129 m_dataSource = AudioSampleDataSource::create(kRingBufferDuration * sampleRate, loggerHelper(), m_pollSamplesCount);
130 m_dataSource->setInputFormat(m_inputDescription.value());
131 m_dataSource->setOutputFormat(m_outputDescription.value());
132
133 callOnMainThread([protectedThis = makeRef(*this), numberOfChannels, sampleRate] {
134 if (protectedThis->m_client)
135 protectedThis->m_client->setFormat(numberOfChannels, sampleRate);
136 });
137}

139// May get called on a background thread.
140void WebAudioSourceProviderCocoa::receivedNewAudioSamples(const PlatformAudioData& data, const AudioStreamDescription& description, size_t frameCount)
141{
142 ASSERT(description.platformDescription().type == PlatformDescription::CAAudioStreamBasicType);
143 auto& basicDescription = *WTF::get<const AudioStreamBasicDescription*>(description.platformDescription().description);
144 if (!m_inputDescription || m_inputDescription->streamDescription() != basicDescription)
145 prepare(basicDescription);
146
147 if (!m_dataSource)
148 return;
149
150 m_dataSource->pushSamples(MediaTime(m_writeCount, m_inputDescription->sampleRate()), data, frameCount);
151
152 m_writeCount += frameCount;
153}
