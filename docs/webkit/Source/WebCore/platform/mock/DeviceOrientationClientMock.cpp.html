26#include "config.h"
27#include "DeviceOrientationClientMock.h"
28
29#include "DeviceOrientationController.h"
30
31namespace WebCore {
32
33DeviceOrientationClientMock::DeviceOrientationClientMock()
34 : m_controller(0)
35 , m_timer(*this, &DeviceOrientationClientMock::timerFired)
36 , m_isUpdating(false)
37{
38}
39
40void DeviceOrientationClientMock::setController(DeviceOrientationController* controller)
41{
42 ASSERT(!m_controller);
43 m_controller = controller;
44 ASSERT(m_controller);
45}
46
47void DeviceOrientationClientMock::startUpdating()
48{
49 m_isUpdating = true;
50}
51
52void DeviceOrientationClientMock::stopUpdating()
53{
54 m_isUpdating = false;
55 m_timer.stop();
56}
57
58void DeviceOrientationClientMock::setOrientation(RefPtr<DeviceOrientationData>&& orientation)
59{
60 m_orientation = WTFMove(orientation);
61 if (m_isUpdating && !m_timer.isActive())
62 m_timer.startOneShot(0_s);
63}
64
65void DeviceOrientationClientMock::timerFired()
66{
67 m_timer.stop();
68 m_controller->didChangeDeviceOrientation(m_orientation.get());
69}
