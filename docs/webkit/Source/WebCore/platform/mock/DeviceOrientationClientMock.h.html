29#include "DeviceOrientationClient.h"

33#include <wtf/RefPtr.h>

35namespace WebCore {
36
37class DeviceOrientationController;
38
39// A mock implementation of DeviceOrientationClient used to test the feature in
40// DumpRenderTree. Embedders should should configure the Page object to use this
41// client when running DumpRenderTree.
42class DeviceOrientationClientMock : public DeviceOrientationClient {
43public:
44 WEBCORE_EXPORT DeviceOrientationClientMock();
45
46 // DeviceOrientationClient
47 void setController(DeviceOrientationController*) override;
48 void startUpdating() override;
49 void stopUpdating() override;
50 DeviceOrientationData* lastOrientation() const override { return m_orientation.get(); }
51 void deviceOrientationControllerDestroyed() override { }
52
53 WEBCORE_EXPORT void setOrientation(RefPtr<DeviceOrientationData>&&);
54
55private:
56 void timerFired();
57
58 RefPtr<DeviceOrientationData> m_orientation;
59 DeviceOrientationController* m_controller;
60 Timer m_timer;
61 bool m_isUpdating;
62};
63
64} // namespace WebCore

66#endif // DeviceOrientationClientMock_h
