39static bool canDecodeMedia(const MediaDecodingConfiguration& configuration)
40{
41 // The mock implementation supports only local file playback.
42 if (configuration.type == MediaDecodingType::MediaSource)
43 return false;
44
45 // Maxing out video decoding support at 720P.
46 auto videoConfig = configuration.video;
47 if (videoConfig && videoConfig->width > 1280 && videoConfig->height > 720)
48 return false;
49
50 // Only the "mock-with-alpha" codec supports alphaChannel
51 if (videoConfig && videoConfig->alphaChannel && videoConfig->alphaChannel.value()) {
52 if (ContentType(videoConfig->contentType).parameter(ContentType::codecsParameter()) != "mock-with-alpha")
53 return false;
54 }
55
56 // Only the "mock-with-hdr" codec supports HDR)
57 if (videoConfig && (videoConfig->colorGamut || videoConfig->hdrMetadataType || videoConfig->transferFunction)) {
58 if (ContentType(videoConfig->contentType).parameter(ContentType::codecsParameter()) != "mock-with-hdr")
59 return false;
60 }
61
62 // Audio decoding support limited to audio/mp4.
63 auto audioConfig = configuration.audio;
64 if (audioConfig) {
65 if (ContentType(audioConfig->contentType).containerType() != "audio/mp4")
66 return false;
67
68 // Can only support spatial rendering of tracks with multichannel audio:
69 if (audioConfig->spatialRendering.value_or(false) && audioConfig->channels.toDouble() <= 2)
70 return false;
71 }
72
73 return true;
74}

76static bool canSmoothlyDecodeMedia(const MediaDecodingConfiguration& configuration)
77{
78 auto videoConfig = configuration.video;
79 if (videoConfig && videoConfig->framerate > 30)
80 return false;
81
82 auto audioConfig = configuration.audio;
83 if (audioConfig && !audioConfig->channels.isNull())
84 return audioConfig->channels == "2";
85
86 return true;
87}

89static bool canPowerEfficientlyDecodeMedia(const MediaDecodingConfiguration& configuration)
90{
91 auto videoConfig = configuration.video;
92 if (videoConfig && ContentType(videoConfig->contentType).containerType() != "video/mp4")
93 return false;
94
95 auto audioConfig = configuration.audio;
96 if (audioConfig && audioConfig->bitrate)
97 return audioConfig->bitrate.value() <= 1000;
98
99 return true;
100}

102static bool canEncodeMedia(const MediaEncodingConfiguration& configuration)
103{
104 // The mock implementation supports only local file playback.
105 if (configuration.type == MediaEncodingType::Record)
106 return false;
107
108 // Maxing out video encoding support at 720P.
109 auto videoConfig = configuration.video;
110 if (videoConfig && videoConfig->width > 1280 && videoConfig->height > 720)
111 return false;
112
113 // Only the "mock-with-alpha" codec supports alphaChannel
114 if (videoConfig && videoConfig->alphaChannel && videoConfig->alphaChannel.value()) {
115 if (ContentType(videoConfig->contentType).parameter(ContentType::codecsParameter()) != "mock-with-alpha")
116 return false;
117 }
118
119 // Audio encoding support limited to audio/mp4.
120 auto audioConfig = configuration.audio;
121 if (audioConfig && ContentType(audioConfig->contentType).containerType() != "audio/mp4")
122 return false;
123
124 return true;
125}

127static bool canSmoothlyEncodeMedia(const MediaEncodingConfiguration& configuration)
128{
129 auto videoConfig = configuration.video;
130 if (videoConfig && videoConfig->framerate > 30)
131 return false;
132
133 auto audioConfig = configuration.audio;
134 if (audioConfig && !audioConfig->channels.isNull() && audioConfig->channels != "2")
135 return false;
136
137 return true;
138}
139
140static bool canPowerEfficientlyEncodeMedia(const MediaEncodingConfiguration& configuration)
141{
142 auto videoConfig = configuration.video;
143 if (videoConfig && ContentType(videoConfig->contentType).containerType() != "video/mp4")
144 return false;
145
146 auto audioConfig = configuration.audio;
147 if (audioConfig && audioConfig->bitrate && audioConfig->bitrate.value() > 1000)
148 return false;
149
150 return true;
151}

153void MediaEngineConfigurationFactoryMock::createDecodingConfiguration(MediaDecodingConfiguration&& configuration, DecodingConfigurationCallback&& callback)
154{
155 if (!canDecodeMedia(configuration)) {
156 MediaCapabilitiesDecodingInfo info { WTFMove(configuration) };
157 callback(WTFMove(info));
158 return;
159 }
160 callback({{ true, canSmoothlyDecodeMedia(configuration), canPowerEfficientlyDecodeMedia(configuration) }, WTFMove(configuration)});
161}
162
163void MediaEngineConfigurationFactoryMock::createEncodingConfiguration(MediaEncodingConfiguration&& configuration, EncodingConfigurationCallback&& callback)
164{
165 if (!canEncodeMedia(configuration)) {
166 callback({{ }, WTFMove(configuration) });
167 return;
168 }
169 callback({{ true, canSmoothlyEncodeMedia(configuration), canPowerEfficientlyEncodeMedia(configuration) }, WTFMove(configuration)});
170}
