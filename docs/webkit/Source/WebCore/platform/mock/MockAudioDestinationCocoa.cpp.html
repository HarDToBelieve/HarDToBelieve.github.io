38MockAudioDestinationCocoa::MockAudioDestinationCocoa(AudioIOCallback& callback, float sampleRate)
39 : AudioDestinationCocoa(callback, 2, sampleRate)
40 , m_workQueue(WorkQueue::create("MockAudioDestinationCocoa Render Queue"))
41 , m_timer(RunLoop::current(), this, &MockAudioDestinationCocoa::tick)
42{
43}
44
45MockAudioDestinationCocoa::~MockAudioDestinationCocoa() = default;
46
47void MockAudioDestinationCocoa::startRendering(CompletionHandler<void(bool)>&& completionHandler)
48{
49 m_timer.startRepeating(Seconds { m_numberOfFramesToProcess / sampleRate() });
50 setIsPlaying(true);
51
52 callOnMainThread([completionHandler = WTFMove(completionHandler)]() mutable {
53 completionHandler(true);
54 });
55}
56
57void MockAudioDestinationCocoa::stopRendering(CompletionHandler<void(bool)>&& completionHandler)
58{
59 m_timer.stop();
60 setIsPlaying(false);
61
62 callOnMainThread([completionHandler = WTFMove(completionHandler)]() mutable {
63 completionHandler(true);
64 });
65}
66
67void MockAudioDestinationCocoa::tick()
68{
69 m_workQueue->dispatch([this, protectedThis = makeRef(*this), sampleRate = sampleRate(), numberOfFramesToProcess = m_numberOfFramesToProcess] {
70 AudioStreamBasicDescription streamFormat;
71 getAudioStreamBasicDescription(streamFormat);
72
73 WebAudioBufferList webAudioBufferList { streamFormat, numberOfFramesToProcess };
74 render(0., 0, numberOfFramesToProcess, webAudioBufferList.list());
75 });
76}
