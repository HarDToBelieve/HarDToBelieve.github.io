48CaptureSourceOrError MockRealtimeAudioSource::create(String&& deviceID, String&& name, String&& hashSalt, const MediaConstraints* constraints)
49{
50#ifndef NDEBUG
51 auto device = MockRealtimeMediaSourceCenter::mockDeviceWithPersistentID(deviceID);
52 ASSERT(device);
53 if (!device)
54 return { "No mock microphone device"_s };
55#endif
56
57 auto source = adoptRef(*new MockRealtimeAudioSource(WTFMove(deviceID), WTFMove(name), WTFMove(hashSalt)));
58 if (constraints && source->applyConstraints(*constraints))
59 return { };
60
61 return CaptureSourceOrError(WTFMove(source));
62}
63#endif

65MockRealtimeAudioSource::MockRealtimeAudioSource(String&& deviceID, String&& name, String&& hashSalt)
66 : RealtimeMediaSource(RealtimeMediaSource::Type::Audio, WTFMove(name), WTFMove(deviceID), WTFMove(hashSalt))
67 , m_workQueue(WorkQueue::create("MockRealtimeAudioSource Render Queue"))
68 , m_timer(RunLoop::current(), this, &MockRealtimeAudioSource::tick)
69{
70 auto device = MockRealtimeMediaSourceCenter::mockDeviceWithPersistentID(persistentID());
71 ASSERT(device);
72 m_device = *device;
73
74 setSampleRate(WTF::get<MockMicrophoneProperties>(m_device.properties).defaultSampleRate);
75 initializeEchoCancellation(true);
76}

78MockRealtimeAudioSource::~MockRealtimeAudioSource()

80#if PLATFORM(IOS_FAMILY)
81 RealtimeMediaSourceCenter::singleton().audioCaptureFactory().unsetActiveSource(*this);
82#endif
83}

85const RealtimeMediaSourceSettings& MockRealtimeAudioSource::settings()
86{
87 if (!m_currentSettings) {
88 RealtimeMediaSourceSettings settings;
89 settings.setDeviceId(hashedId());
90 settings.setVolume(volume());
91 settings.setEchoCancellation(echoCancellation());
92 settings.setSampleRate(sampleRate());
93 settings.setLabel(name());
94
95 RealtimeMediaSourceSupportedConstraints supportedConstraints;
96 supportedConstraints.setSupportsDeviceId(true);
97 supportedConstraints.setSupportsVolume(true);
98 supportedConstraints.setSupportsEchoCancellation(true);
99 supportedConstraints.setSupportsSampleRate(true);
100 settings.setSupportedConstraints(supportedConstraints);
101
102 m_currentSettings = WTFMove(settings);
103 }
104 return m_currentSettings.value();
105}

107void MockRealtimeAudioSource::setChannelCount(unsigned channelCount)
108{
109 if (channelCount > 2)
110 return;
111
112 m_channelCount = channelCount;
113 settingsDidChange(RealtimeMediaSourceSettings::Flag::SampleRate);
114}

116const RealtimeMediaSourceCapabilities& MockRealtimeAudioSource::capabilities()
117{
118 if (!m_capabilities) {
119 RealtimeMediaSourceCapabilities capabilities(settings().supportedConstraints());
120
121 capabilities.setDeviceId(hashedId());
122 capabilities.setVolume(CapabilityValueOrRange(0.0, 1.0));
123 capabilities.setEchoCancellation(RealtimeMediaSourceCapabilities::EchoCancellation::ReadWrite);
124 capabilities.setSampleRate(CapabilityValueOrRange(44100, 48000));
125
126 m_capabilities = WTFMove(capabilities);
127 }
128 return m_capabilities.value();
129}
130
131void MockRealtimeAudioSource::settingsDidChange(OptionSet<RealtimeMediaSourceSettings::Flag>)
132{
133 m_currentSettings = std::nullopt;
134}

136void MockRealtimeAudioSource::startProducingData()
137{
138#if PLATFORM(IOS_FAMILY)
139 RealtimeMediaSourceCenter::singleton().audioCaptureFactory().setActiveSource(*this);
140 PlatformMediaSessionManager::sharedManager().sessionCanProduceAudioChanged();
141 ASSERT(AudioSession::sharedSession().category() == AudioSession::CategoryType::PlayAndRecord);
142#endif
143
144 if (!sampleRate())
145 setSampleRate(WTF::get<MockMicrophoneProperties>(m_device.properties).defaultSampleRate);
146
147 m_startTime = MonotonicTime::now();
148 m_timer.startRepeating(renderInterval());
149}
150
151void MockRealtimeAudioSource::stopProducingData()
152{
153 m_timer.stop();
154 m_startTime = MonotonicTime::nan();
155}
156
157void MockRealtimeAudioSource::tick()
158{
159 if (std::isnan(m_lastRenderTime))
160 m_lastRenderTime = MonotonicTime::now();
161
162 MonotonicTime now = MonotonicTime::now();
163
164 if (m_delayUntil) {
165 if (m_delayUntil < now)
166 return;
167 m_delayUntil = MonotonicTime();
168 }
169
170 Seconds delta = now - m_lastRenderTime;
171 m_lastRenderTime = now;
172
173 m_workQueue->dispatch([this, delta, protectedThis = makeRef(*this)] {
174 render(delta);
175 });
176}
177
178void MockRealtimeAudioSource::delaySamples(Seconds delta)
179{
180 m_delayUntil = MonotonicTime::now() + delta;
181}
