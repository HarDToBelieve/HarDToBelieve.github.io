43MockBox::MockBox(ArrayBuffer* data)
44{
45 m_type = peekType(data);
46 m_length = peekLength(data);
47 ASSERT(m_length >= 8);
48}
49
50String MockBox::peekType(ArrayBuffer* data)
51{
52 StringBuilder builder;
53 auto array = JSC::Int8Array::create(data, 0, 4);
54 for (int i = 0; i < 4; ++i)
55 builder.append(static_cast<char>(array->item(i)));
56 return builder.toString();
57}
58
59size_t MockBox::peekLength(ArrayBuffer* data)
60{
61 auto view = JSC::DataView::create(data, 0, data->byteLength());
62 return view->get<uint32_t>(4, true);
63}

65MockTrackBox::MockTrackBox(ArrayBuffer* data)
66 : MockBox(data)
67{
68 ASSERT(m_length == 17);
69
70 auto view = JSC::DataView::create(data, 0, data->byteLength());
71 m_trackID = view->get<int32_t>(8, true);
72
73 StringBuilder builder;
74 auto array = JSC::Int8Array::create(data, 12, 4);
75 for (int i = 0; i < 4; ++i)
76 builder.append(static_cast<char>(array->item(i)));
77 m_codec = builder.toString();
78
79 m_kind = static_cast<TrackKind>(view->get<uint8_t>(16, true));
80}

88MockInitializationBox::MockInitializationBox(ArrayBuffer* data)
89 : MockBox(data)
90{
91 ASSERT(m_length >= 13);
92
93 auto view = JSC::DataView::create(data, 0, data->byteLength());
94 int32_t timeValue = view->get<int32_t>(8, true);
95 int32_t timeScale = view->get<int32_t>(12, true);
96 m_duration = MediaTime(timeValue, timeScale);
97
98 size_t offset = 16;
99
100 while (offset < m_length) {
101 auto subBuffer = data->slice(offset);
102 if (MockBox::peekType(subBuffer.get()) != MockTrackBox::type())
103 break;
104
105 MockTrackBox trackBox(subBuffer.get());
106 offset += trackBox.length();
107 m_tracks.append(trackBox);
108 }
109}

113 static NeverDestroyed<String> init(MAKE_STATIC_STRING_IMPL("init"));
114 return init;
115}

117MockSampleBox::MockSampleBox(ArrayBuffer* data)
118 : MockBox(data)
119{
120 ASSERT(m_length == 30);
121
122 auto view = JSC::DataView::create(data, 0, data->byteLength());
123 int32_t timeScale = view->get<int32_t>(8, true);
124
125 int32_t timeValue = view->get<int32_t>(12, true);
126 m_presentationTimestamp = MediaTime(timeValue, timeScale);
127
128 timeValue = view->get<int32_t>(16, true);
129 m_decodeTimestamp = MediaTime(timeValue, timeScale);
130
131 timeValue = view->get<int32_t>(20, true);
132 m_duration = MediaTime(timeValue, timeScale);
133
134 m_trackID = view->get<int32_t>(24, true);
135 m_flags = view->get<uint8_t>(28, true);
136 m_generation = view->get<uint8_t>(29, true);
137}
