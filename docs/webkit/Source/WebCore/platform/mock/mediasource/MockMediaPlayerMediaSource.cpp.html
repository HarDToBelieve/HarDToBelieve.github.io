78MediaPlayer::SupportsType MockMediaPlayerMediaSource::supportsType(const MediaEngineSupportParameters& parameters)
79{
80 if (!parameters.isMediaSource)
81 return MediaPlayer::SupportsType::IsNotSupported;
82
83 auto containerType = parameters.type.containerType();
84 if (containerType.isEmpty() || !mimeTypeCache().contains(containerType))
85 return MediaPlayer::SupportsType::IsNotSupported;
86
87 auto codecs = parameters.type.parameter(ContentType::codecsParameter());
88 if (codecs.isEmpty())
89 return MediaPlayer::SupportsType::MayBeSupported;
90
91 if (codecs == "mock" || codecs == "kcom")
92 return MediaPlayer::SupportsType::IsSupported;
93
94 return MediaPlayer::SupportsType::MayBeSupported;
95}

211void MockMediaPlayerMediaSource::seekWithTolerance(const MediaTime& time, const MediaTime& negativeTolerance, const MediaTime& positiveTolerance)
212{
213 if (!negativeTolerance && !positiveTolerance) {
214 m_currentTime = time;
215 m_mediaSourcePrivate->seekToTime(time);
216 } else
217 m_currentTime = m_mediaSourcePrivate->seekToTime(time, negativeTolerance, positiveTolerance);
218
219 if (m_seekCompleted) {
220 m_player->timeChanged();
221
222 if (m_playing)
223 callOnMainThread([this] {
224 advanceCurrentTime();
225 });
226 }
227}
228
229void MockMediaPlayerMediaSource::advanceCurrentTime()
230{
231 if (!m_mediaSourcePrivate)
232 return;
233
234 auto buffered = m_mediaSourcePrivate->buffered();
235 size_t pos = buffered->find(m_currentTime);
236 if (pos == notFound)
237 return;
238
239 bool ignoreError;
240 m_currentTime = std::min(m_duration, buffered->end(pos, ignoreError));
241 m_player->timeChanged();
242}
