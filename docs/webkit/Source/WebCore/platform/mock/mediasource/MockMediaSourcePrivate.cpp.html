59MockMediaSourcePrivate::~MockMediaSourcePrivate()
60{
61 for (auto& buffer : m_sourceBuffers)
62 buffer->clearMediaSource();
63}
64
65MediaSourcePrivate::AddStatus MockMediaSourcePrivate::addSourceBuffer(const ContentType& contentType, bool, RefPtr<SourceBufferPrivate>& outPrivate)
66{
67 MediaEngineSupportParameters parameters;
68 parameters.isMediaSource = true;
69 parameters.type = contentType;
70 if (MockMediaPlayerMediaSource::supportsType(parameters) == MediaPlayer::SupportsType::IsNotSupported)
71 return AddStatus::NotSupported;
72
73 m_sourceBuffers.append(MockSourceBufferPrivate::create(this));
74 outPrivate = m_sourceBuffers.last();
75
76 return AddStatus::Ok;
77}

79void MockMediaSourcePrivate::removeSourceBuffer(SourceBufferPrivate* buffer)
80{
81 ASSERT(m_sourceBuffers.contains(buffer));
82 m_activeSourceBuffers.removeFirst(buffer);
83 m_sourceBuffers.removeFirst(buffer);
84}

133void MockMediaSourcePrivate::sourceBufferPrivateDidChangeActiveState(MockSourceBufferPrivate* buffer, bool active)
134{
135 if (active && !m_activeSourceBuffers.contains(buffer))
136 m_activeSourceBuffers.append(buffer);

138 if (!active)
139 m_activeSourceBuffers.removeFirst(buffer);

167MediaTime MockMediaSourcePrivate::seekToTime(const MediaTime& targetTime, const MediaTime& negativeThreshold, const MediaTime& positiveThreshold)
168{
169 MediaTime seekTime = targetTime;
170 for (auto& buffer : m_activeSourceBuffers) {
171 MediaTime sourceSeekTime = buffer->fastSeekTimeForMediaTime(targetTime, negativeThreshold, positiveThreshold);
172 if (abs(targetTime - sourceSeekTime) > abs(targetTime - seekTime))
173 seekTime = sourceSeekTime;
174 }
175
176 return seekTime;
177}
