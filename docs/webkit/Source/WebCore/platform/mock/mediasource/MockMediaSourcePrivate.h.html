28#if ENABLE(MEDIA_SOURCE)

30#include "MediaSourcePrivate.h"
31#include <wtf/MediaTime.h>
32
33namespace WebCore {
34
35class MockMediaPlayerMediaSource;
36class MockSourceBufferPrivate;
37
38class MockMediaSourcePrivate final
39 : public MediaSourcePrivate
40#if !RELEASE_LOG_DISABLED
41 , private LoggerHelper
42#endif
43{
44public:
45 static Ref<MockMediaSourcePrivate> create(MockMediaPlayerMediaSource&, MediaSourcePrivateClient&);
46 virtual ~MockMediaSourcePrivate();
47
48 const Vector<MockSourceBufferPrivate*>& activeSourceBuffers() const { return m_activeSourceBuffers; }
49
50 bool hasAudio() const;
51 bool hasVideo() const;
52
53 MediaTime duration();
54 std::unique_ptr<PlatformTimeRanges> buffered();
55
56 MockMediaPlayerMediaSource& player() const { return m_player; }
57
58 void seekToTime(const MediaTime&);
59 MediaTime seekToTime(const MediaTime&, const MediaTime& negativeThreshold, const MediaTime& positiveThreshold);
60 MediaTime currentMediaTime() const;
61
62 std::optional<VideoPlaybackQualityMetrics> videoPlaybackQualityMetrics();
63
64 void incrementTotalVideoFrames() { ++m_totalVideoFrames; }
65 void incrementDroppedFrames() { ++m_droppedVideoFrames; }
66 void incrementCorruptedFrames() { ++m_corruptedVideoFrames; }
67 void incrementTotalFrameDelayBy(const MediaTime& delay) { m_totalFrameDelay += delay; }
68
69#if !RELEASE_LOG_DISABLED
70 const Logger& logger() const final { return m_logger.get(); }
71 const char* logClassName() const override { return "MockMediaSourcePrivate"; }
72 const void* logIdentifier() const final { return m_logIdentifier; }
73 WTFLogChannel& logChannel() const final;
74
75 const void* nextSourceBufferLogIdentifier() { return childLogIdentifier(m_logIdentifier, ++m_nextSourceBufferID); }
76#endif
77
78private:
79 MockMediaSourcePrivate(MockMediaPlayerMediaSource&, MediaSourcePrivateClient&);
80
81 // MediaSourcePrivate Overrides
82 AddStatus addSourceBuffer(const ContentType&, bool webMParserEnabled, RefPtr<SourceBufferPrivate>&) override;
83 void durationChanged(const MediaTime&) override;
84 void markEndOfStream(EndOfStreamStatus) override;
85 bool isEnded() const override { return m_isEnded; }
86 void unmarkEndOfStream() override;
87 MediaPlayer::ReadyState readyState() const override;
88 void setReadyState(MediaPlayer::ReadyState) override;
89 void waitForSeekCompleted() override;
90 void seekCompleted() override;
91
92 void sourceBufferPrivateDidChangeActiveState(MockSourceBufferPrivate*, bool active);
93 void removeSourceBuffer(SourceBufferPrivate*);
94
95 friend class MockSourceBufferPrivate;
96
97 MockMediaPlayerMediaSource& m_player;
98 Ref<MediaSourcePrivateClient> m_client;
99 Vector<RefPtr<MockSourceBufferPrivate>> m_sourceBuffers;
100 Vector<MockSourceBufferPrivate*> m_activeSourceBuffers;
101 bool m_isEnded { false };
102
103 unsigned m_totalVideoFrames { 0 };
104 unsigned m_droppedVideoFrames { 0 };
105 unsigned m_corruptedVideoFrames { 0 };
106 MediaTime m_totalFrameDelay;
107#if !RELEASE_LOG_DISABLED
108 Ref<const Logger> m_logger;
109 const void* m_logIdentifier;
110 uint64_t m_nextSourceBufferID { 0 };
111#endif
112};
113
114}

116#endif // ENABLE(MEDIA_SOURCE)
