76 MockSampleBox m_box;

80MediaSample::SampleFlags MockMediaSample::flags() const
81{
82 unsigned flags = None;
83 if (m_box.isSync())
84 flags |= IsSync;
85 if (m_box.isNonDisplaying())
86 flags |= IsNonDisplaying;
87 return SampleFlags(flags);
88}

139void MockSourceBufferPrivate::append(Vector<uint8_t>&& data)
140{
141 m_inputBuffer.appendVector(data);
142 SourceBufferPrivateClient::AppendResult result = SourceBufferPrivateClient::AppendResult::AppendSucceeded;
143
144 while (m_inputBuffer.size() && result == SourceBufferPrivateClient::AppendResult::AppendSucceeded) {
145 auto buffer = ArrayBuffer::create(m_inputBuffer.data(), m_inputBuffer.size());
146 uint64_t boxLength = MockBox::peekLength(buffer.ptr());
147 if (boxLength > buffer->byteLength())
148 break;
149
150 String type = MockBox::peekType(buffer.ptr());
151 if (type == MockInitializationBox::type()) {
152 MockInitializationBox initBox = MockInitializationBox(buffer.ptr());
153 didReceiveInitializationSegment(initBox);
154 } else if (type == MockSampleBox::type()) {
155 MockSampleBox sampleBox = MockSampleBox(buffer.ptr());
156 didReceiveSample(sampleBox);
157 } else
158 result = SourceBufferPrivateClient::AppendResult::ParsingFailed;
159
160 m_inputBuffer.remove(0, boxLength);
161 }
162
163 // Resolve the changes in TrackBuffers' buffered ranges
164 // into the SourceBuffer's buffered ranges
165 updateBufferedFromTrackBuffers(m_mediaSource->isEnded());
166
167 if (m_client)
168 m_client->sourceBufferPrivateAppendComplete(result);
169}
170
171void MockSourceBufferPrivate::didReceiveInitializationSegment(const MockInitializationBox& initBox)
172{
173 if (!m_client)
174 return;
175
176 SourceBufferPrivateClient::InitializationSegment segment;
177 segment.duration = initBox.duration();
178
179 for (auto it = initBox.tracks().begin(); it != initBox.tracks().end(); ++it) {
180 const MockTrackBox& trackBox = *it;
181 if (trackBox.kind() == MockTrackBox::Video) {
182 SourceBufferPrivateClient::InitializationSegment::VideoTrackInformation info;
183 info.track = MockVideoTrackPrivate::create(trackBox);
184 info.description = MockMediaDescription::create(trackBox);
185 segment.videoTracks.append(info);
186 } else if (trackBox.kind() == MockTrackBox::Audio) {
187 SourceBufferPrivateClient::InitializationSegment::AudioTrackInformation info;
188 info.track = MockAudioTrackPrivate::create(trackBox);
189 info.description = MockMediaDescription::create(trackBox);
190 segment.audioTracks.append(info);
191 } else if (trackBox.kind() == MockTrackBox::Text) {
192 SourceBufferPrivateClient::InitializationSegment::TextTrackInformation info;
193 info.track = MockTextTrackPrivate::create(trackBox);
194 info.description = MockMediaDescription::create(trackBox);
195 segment.textTracks.append(info);
196 }
197 }
198
199 SourceBufferPrivate::didReceiveInitializationSegment(WTFMove(segment), []() { });
200}

306void MockSourceBufferPrivate::enqueueSample(Ref<MediaSample>&& sample, const AtomString&)
307{
308 if (!m_mediaSource)
309 return;
310
311 PlatformSample platformSample = sample->platformSample();
312 if (platformSample.type != PlatformSample::MockSampleBoxType)
313 return;
314
315 auto* box = platformSample.sample.mockSampleBox;
316 if (!box)
317 return;
318
319 m_mediaSource->incrementTotalVideoFrames();
320 if (box->isCorrupted())
321 m_mediaSource->incrementCorruptedFrames();
322 if (box->isDropped())
323 m_mediaSource->incrementDroppedFrames();
324 if (box->isDelayed())
325 m_mediaSource->incrementTotalFrameDelayBy(MediaTime(1, 1));
326
327 m_enqueuedSamples.append(toString(sample.get()));
328}
