32namespace WebCore {
33
34class AudioTrackPrivate;
35class InbandTextTrackPrivate;
36class MockInitializationBox;
37class MockMediaSourcePrivate;
38class MockSampleBox;
39class TimeRanges;
40class VideoTrackPrivate;
41
42class MockSourceBufferPrivate final : public SourceBufferPrivate {
43public:
44 static Ref<MockSourceBufferPrivate> create(MockMediaSourcePrivate*);
45 virtual ~MockSourceBufferPrivate();
46
47 void clearMediaSource() { m_mediaSource = nullptr; }
48
49private:
50 explicit MockSourceBufferPrivate(MockMediaSourcePrivate*);
51
52 // SourceBufferPrivate overrides
53 void append(Vector<uint8_t>&&) final;
54 void abort() final;
55 void resetParserState() final;
56 void removedFromMediaSource() final;
57 MediaPlayer::ReadyState readyState() const final;
58 void setReadyState(MediaPlayer::ReadyState) final;
59 bool canSetMinimumUpcomingPresentationTime(const AtomString&) const final;
60 void setMinimumUpcomingPresentationTime(const AtomString&, const MediaTime&) final;
61 void clearMinimumUpcomingPresentationTime(const AtomString&) final;
62 bool canSwitchToType(const ContentType&) final;
63 bool isSeeking() const final;
64 MediaTime currentMediaTime() const final;
65 MediaTime duration() const final;
66
67 void flush(const AtomString&) final { m_enqueuedSamples.clear(); m_minimumUpcomingPresentationTime = MediaTime::invalidTime(); }
68 void enqueueSample(Ref<MediaSample>&&, const AtomString&) final;
69 bool isReadyForMoreSamples(const AtomString&) final { return !m_maxQueueDepth || m_enqueuedSamples.size() < m_maxQueueDepth.value(); }
70 void setActive(bool) final;
71 bool isActive() const final;
72
73 void enqueuedSamplesForTrackID(const AtomString&, CompletionHandler<void(Vector<String>&&)>&&) final;
74 MediaTime minimumUpcomingPresentationTimeForTrackID(const AtomString&) final;
75 void setMaximumQueueDepthForTrackID(const AtomString&, uint64_t) final;
76
77 void didReceiveInitializationSegment(const MockInitializationBox&);
78 void didReceiveSample(const MockSampleBox&);
79
80#if !RELEASE_LOG_DISABLED
81 const Logger& logger() const final { return m_logger.get(); }
82 const char* logClassName() const override { return "MockSourceBufferPrivate"; }
83 const void* logIdentifier() const final { return m_logIdentifier; }
84 WTFLogChannel& logChannel() const final;
85
86 const Logger& sourceBufferLogger() const final { return m_logger.get(); }
87 const void* sourceBufferLogIdentifier() final { return logIdentifier(); }
88#endif
89
90 MockMediaSourcePrivate* m_mediaSource;
91 bool m_isActive { false };
92 MediaTime m_minimumUpcomingPresentationTime;
93 Vector<String> m_enqueuedSamples;
94 std::optional<uint64_t> m_maxQueueDepth;
95 Vector<uint8_t> m_inputBuffer;
96
97#if !RELEASE_LOG_DISABLED
98 Ref<const Logger> m_logger;
99 const void* m_logIdentifier;
100#endif
101};
102
103}
