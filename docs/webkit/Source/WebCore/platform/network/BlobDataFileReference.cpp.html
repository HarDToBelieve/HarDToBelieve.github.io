34BlobDataFileReference::BlobDataFileReference(const String& path, const String& replacementPath)
35 : m_path(path)
36 , m_replacementPath(replacementPath)
37{
38}
39
40BlobDataFileReference::~BlobDataFileReference()
41{
42 if (!m_replacementPath.isNull())
43 FileSystem::deleteFile(m_replacementPath);
44}

46const String& BlobDataFileReference::path()
47{
48#if ENABLE(FILE_REPLACEMENT)
49 if (m_replacementShouldBeGenerated)
50 generateReplacementFile();
51#endif
52 if (!m_replacementPath.isNull())
53 return m_replacementPath;
54
55 return m_path;
56}

58unsigned long long BlobDataFileReference::size()
59{
60#if ENABLE(FILE_REPLACEMENT)
61 if (m_replacementShouldBeGenerated)
62 generateReplacementFile();
63#endif
64
65 return m_size;
66}

68std::optional<WallTime> BlobDataFileReference::expectedModificationTime()
69{
70#if ENABLE(FILE_REPLACEMENT)
71 // We do not currently track modifications for generated files, because we have a snapshot.
72 // Unfortunately, this is inconsistent with regular file handling - File objects should be invalidated when underlying files change.
73 if (m_replacementShouldBeGenerated || !m_replacementPath.isNull())
74 return std::nullopt;
75#endif
76 return m_expectedModificationTime;
77}
78
79void BlobDataFileReference::startTrackingModifications()
80{
81 // This is not done automatically by the constructor, because BlobDataFileReference is
82 // also used to pass paths around before registration. Only registered blobs need to pay
83 // the cost of tracking file modifications.
84
85#if ENABLE(FILE_REPLACEMENT)
86 m_replacementShouldBeGenerated = File::shouldReplaceFile(m_path);
87#endif
88
89 // FIXME: Some platforms provide better ways to listen for file system object changes, consider using these.
90 auto modificationTime = FileSystem::fileModificationTime(m_path);
91 if (!modificationTime)
92 return;
93
94 m_expectedModificationTime = *modificationTime;
95
96#if ENABLE(FILE_REPLACEMENT)
97 if (m_replacementShouldBeGenerated)
98 return;
99#endif
100
101 // This is a registered blob with a replacement file. Get the size of the replacement file.
102 auto fileSize = FileSystem::fileSize(m_replacementPath.isNull() ? m_path : m_replacementPath);
103 if (!fileSize)
104 return;
105
106 m_size = *fileSize;
107}
