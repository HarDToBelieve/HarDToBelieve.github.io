26#ifndef BlobDataFileReference_h
27#define BlobDataFileReference_h
28
29#include <wtf/Markable.h>
30#include <wtf/RefCounted.h>
31#include <wtf/WallTime.h>
32#include <wtf/text/WTFString.h>
33
34namespace WebCore {
35
36class WEBCORE_EXPORT BlobDataFileReference : public RefCounted<BlobDataFileReference> {
37public:
38 static Ref<BlobDataFileReference> create(const String& path, const String& replacementPath = { })
39 {
40 return adoptRef(*new BlobDataFileReference(path, replacementPath));
41 }
42
43 virtual ~BlobDataFileReference();
44
45 void startTrackingModifications();
46
47 const String& path();
48 unsigned long long size();
49 std::optional<WallTime> expectedModificationTime();
50
51 virtual void prepareForFileAccess();
52 virtual void revokeFileAccess();
53
54protected:
55 BlobDataFileReference(const String& path, const String& replacementPath);
56
57private:
58#if ENABLE(FILE_REPLACEMENT)
59 void generateReplacementFile();
60#endif
61
62 String m_path;
63 String m_replacementPath;
64#if ENABLE(FILE_REPLACEMENT)
65 bool m_replacementShouldBeGenerated { false };
66#endif
67 unsigned long long m_size { 0 };
68 Markable<WallTime, WallTime::MarkableTraits> m_expectedModificationTime;
69};
70
71}
72
73#endif // BlobDataFileReference_h
