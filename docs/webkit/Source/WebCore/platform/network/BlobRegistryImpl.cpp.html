82void BlobRegistryImpl::appendStorageItems(BlobData* blobData, const BlobDataItemList& items, long long offset, long long length)
83{
84 ASSERT(length != BlobDataItem::toEndOfFile);
85
86 BlobDataItemList::const_iterator iter = items.begin();
87 if (offset) {
88 for (; iter != items.end(); ++iter) {
89 if (offset >= iter->length())
90 offset -= iter->length();
91 else
92 break;
93 }
94 }
95
96 for (; iter != items.end() && length > 0; ++iter) {
97 long long currentLength = iter->length() - offset;
98 long long newLength = currentLength > length ? length : currentLength;
99 if (iter->type() == BlobDataItem::Type::Data)
100 blobData->appendData(iter->data(), iter->offset() + offset, newLength);
101 else {
102 ASSERT(iter->type() == BlobDataItem::Type::File);
103 blobData->appendFile(iter->file(), iter->offset() + offset, newLength);
104 }
105 length -= newLength;
106 offset = 0;
107 }
108 ASSERT(!length);
109}

121void BlobRegistryImpl::registerBlobURL(const URL& url, Vector<BlobPart>&& blobParts, const String& contentType)
122{
123 ASSERT(isMainThread());
124 registerBlobResourceHandleConstructor();
125
126 auto blobData = BlobData::create(contentType);
127
128 // The blob data is stored in the "canonical" way. That is, it only contains a list of Data and File items.
129 // 1) The Data item is denoted by the raw data and the range.
130 // 2) The File item is denoted by the file path, the range and the expected modification time.
131 // 3) The URL item is denoted by the URL, the range and the expected modification time.
132 // All the Blob items in the passing blob data are resolved and expanded into a set of Data and File items.
133
134 for (BlobPart& part : blobParts) {
135 switch (part.type()) {
136 case BlobPart::Type::Data: {
137 auto movedData = part.moveData();
138 auto data = ThreadSafeDataBuffer::create(WTFMove(movedData));
139 blobData->appendData(data);
140 break;
141 }
142 case BlobPart::Type::Blob: {
143 if (auto blob = m_blobs.get(part.url().string())) {
144 for (const BlobDataItem& item : blob->items())
145 blobData->m_items.append(item);
146 }
147 break;
148 }
149 }
150 }
151
152 m_blobs.set(url.string(), WTFMove(blobData));
153}

160void BlobRegistryImpl::registerBlobURLOptionallyFileBacked(const URL& url, const URL& srcURL, RefPtr<BlobDataFileReference>&& file, const String& contentType)
161{
162 ASSERT(isMainThread());
163 registerBlobResourceHandleConstructor();
164
165 BlobData* src = getBlobDataFromURL(srcURL);
166 if (src) {
167 m_blobs.set(url.string(), src);
168 return;
169 }
170
171 if (!file || file->path().isEmpty())
172 return;
173
174 auto backingFile = BlobData::create(contentType);
175 backingFile->appendFile(file.releaseNonNull());
176
177 m_blobs.set(url.string(), WTFMove(backingFile));
178}

180void BlobRegistryImpl::registerBlobURLForSlice(const URL& url, const URL& srcURL, long long start, long long end, const String& contentType)
181{
182 ASSERT(isMainThread());
183 BlobData* originalData = getBlobDataFromURL(srcURL);
184 if (!originalData)
185 return;
186
187 unsigned long long originalSize = blobSize(srcURL);
188
189 // Convert the negative value that is used to select from the end.
190 if (start < 0)
191 start = start + originalSize;
192 if (end < 0)
193 end = end + originalSize;
194
195 // Clamp the range if it exceeds the size limit.
196 if (start < 0)
197 start = 0;
198 if (end < 0)
199 end = 0;
200 if (static_cast<unsigned long long>(start) >= originalSize) {
201 start = 0;
202 end = 0;
203 } else if (end < start)
204 end = start;
205 else if (static_cast<unsigned long long>(end) > originalSize)
206 end = originalSize;
207
208 unsigned long long newLength = end - start;
209 auto newData = BlobData::create(contentType.isEmpty() ? originalData->contentType() : contentType);
210
211 appendStorageItems(newData.ptr(), originalData->items(), start, newLength);
212
213 m_blobs.set(url.string(), WTFMove(newData));
214}

222BlobData* BlobRegistryImpl::getBlobDataFromURL(const URL& url) const
223{
224 ASSERT(isMainThread());
225 if (url.hasFragmentIdentifier())
226 return m_blobs.get(url.stringWithoutFragmentIdentifier().toStringWithoutCopying());
227 return m_blobs.get(url.string());
228}

230unsigned long long BlobRegistryImpl::blobSize(const URL& url)
231{
232 ASSERT(isMainThread());
233 BlobData* data = getBlobDataFromURL(url);
234 if (!data)
235 return 0;
236
237 unsigned long long result = 0;
238 for (const BlobDataItem& item : data->items())
239 result += item.length();
240
241 return result;
242}

250bool BlobRegistryImpl::populateBlobsForFileWriting(const Vector<String>& blobURLs, Vector<BlobForFileWriting>& blobsForWriting)
251{
252 for (auto& url : blobURLs) {
253 blobsForWriting.append({ });
254 blobsForWriting.last().blobURL = url.isolatedCopy();
255
256 auto* blobData = getBlobDataFromURL({ { }, url });
257 if (!blobData)
258 return false;
259
260 for (auto& item : blobData->items()) {
261 switch (item.type()) {
262 case BlobDataItem::Type::Data:
263 blobsForWriting.last().filePathsOrDataBuffers.append({ { }, item.data() });
264 break;
265 case BlobDataItem::Type::File:
266 blobsForWriting.last().filePathsOrDataBuffers.append({ item.file()->path().isolatedCopy(), { } });
267 break;
268 default:
269 ASSERT_NOT_REACHED();
270 }
271 }
272 }
273 return true;
274}

276static bool writeFilePathsOrDataBuffersToFile(const Vector<std::pair<String, ThreadSafeDataBuffer>>& filePathsOrDataBuffers, FileSystem::PlatformFileHandle file, const String& path)
277{
278 auto fileCloser = WTF::makeScopeExit([file]() mutable {
279 FileSystem::closeFile(file);
280 });
281
282 if (path.isEmpty() || !FileSystem::isHandleValid(file)) {
283 LOG_ERROR("Failed to open temporary file for writing a Blob");
284 return false;
285 }
286
287 for (auto& part : filePathsOrDataBuffers) {
288 if (part.second.data()) {
289 int length = part.second.data()->size();
290 if (FileSystem::writeToFile(file, part.second.data()->data(), length) != length) {
291 LOG_ERROR("Failed writing a Blob to temporary file");
292 return false;
293 }
294 } else {
295 ASSERT(!part.first.isEmpty());
296 if (!FileSystem::appendFileContentsToFileHandle(part.first, file)) {
297 LOG_ERROR("Failed copying File contents to a Blob temporary file (%s to %s)", part.first.utf8().data(), path.utf8().data());
298 return false;
299 }
300 }
301 }
302 return true;
303}

305void BlobRegistryImpl::writeBlobsToTemporaryFiles(const Vector<String>& blobURLs, CompletionHandler<void(Vector<String>&& filePaths)>&& completionHandler)
306{
307 Vector<BlobForFileWriting> blobsForWriting;
308 if (!populateBlobsForFileWriting(blobURLs, blobsForWriting)) {
309 completionHandler({ });
310 return;
311 }
312
313 blobUtilityQueue().dispatch([blobsForWriting = WTFMove(blobsForWriting), completionHandler = WTFMove(completionHandler)]() mutable {
314 Vector<String> filePaths;
315 for (auto& blob : blobsForWriting) {
316 FileSystem::PlatformFileHandle file;
317 String tempFilePath = FileSystem::openTemporaryFile("Blob"_s, file);
318 if (!writeFilePathsOrDataBuffersToFile(blob.filePathsOrDataBuffers, file, tempFilePath)) {
319 filePaths.clear();
320 break;
321 }
322 filePaths.append(tempFilePath.isolatedCopy());
323 }
324
325 callOnMainThread([completionHandler = WTFMove(completionHandler), filePaths = WTFMove(filePaths)] () mutable {
326 completionHandler(WTFMove(filePaths));
327 });
328 });
329}

331void BlobRegistryImpl::writeBlobToFilePath(const URL& blobURL, const String& path, Function<void(bool success)>&& completionHandler)
332{
333 Vector<BlobForFileWriting> blobsForWriting;
334 if (!populateBlobsForFileWriting({ blobURL.string() }, blobsForWriting) || blobsForWriting.size() != 1) {
335 completionHandler(false);
336 return;
337 }
338
339 blobUtilityQueue().dispatch([path, blobsForWriting = WTFMove(blobsForWriting), completionHandler = WTFMove(completionHandler)]() mutable {
340 bool success = writeFilePathsOrDataBuffersToFile(blobsForWriting.first().filePathsOrDataBuffers, FileSystem::openFile(path, FileSystem::FileOpenMode::Write), path);
341 callOnMainThread([success, completionHandler = WTFMove(completionHandler)]() {
342 completionHandler(success);
343 });
344 });
345}

347Vector<RefPtr<BlobDataFileReference>> BlobRegistryImpl::filesInBlob(const URL& url) const
348{
349 auto* blobData = getBlobDataFromURL(url);
350 if (!blobData)
351 return { };
352
353 Vector<RefPtr<BlobDataFileReference>> result;
354 for (const BlobDataItem& item : blobData->items()) {
355 if (item.type() == BlobDataItem::Type::File)
356 result.append(item.file());
357 }
358
359 return result;
360}
