111Ref<FormData> FormData::isolatedCopy() const
112{
113 // FIXME: isolatedCopy() does not copy m_identifier, m_boundary, or m_containsPasswordData.
114 // Is all of that correct and intentional?
115
116 auto formData = create();
117
118 formData->m_alwaysStream = m_alwaysStream;
119
120 formData->m_elements.reserveInitialCapacity(m_elements.size());
121 for (auto& element : m_elements)
122 formData->m_elements.uncheckedAppend(element.isolatedCopy());
123
124 return formData;
125}

127unsigned FormData::imageOrMediaFilesCount() const
128{
129 unsigned imageOrMediaFilesCount = 0;
130 for (auto& element : m_elements) {
131 auto* encodedFileData = WTF::get_if<FormDataElement::EncodedFileData>(element.data);
132 if (!encodedFileData)
133 continue;
134
135 auto mimeType = MIMETypeRegistry::mimeTypeForPath(encodedFileData->filename);
136 if (MIMETypeRegistry::isSupportedImageMIMEType(mimeType) || MIMETypeRegistry::isSupportedMediaMIMEType(mimeType))
137 ++imageOrMediaFilesCount;
138 }
139 return imageOrMediaFilesCount;
140}

142uint64_t FormDataElement::lengthInBytes(const Function<uint64_t(const URL&)>& blobSize) const
143{
144 return switchOn(data,
145 [] (const Vector<uint8_t>& bytes) {
146 return static_cast<uint64_t>(bytes.size());
147 }, [] (const FormDataElement::EncodedFileData& fileData) {
148 if (fileData.fileLength != BlobDataItem::toEndOfFile)
149 return static_cast<uint64_t>(fileData.fileLength);
150 return FileSystem::fileSize(fileData.filename).value_or(0);
151 }, [&blobSize] (const FormDataElement::EncodedBlobData& blobData) {
152 return blobSize(blobData.url);
153 }
154 );
155}

177void FormData::appendData(const void* data, size_t size)
178{
179 m_lengthInBytes = std::nullopt;
180 if (!m_elements.isEmpty()) {
181 if (auto* vector = WTF::get_if<Vector<uint8_t>>(m_elements.last().data)) {
182 vector->append(static_cast<const uint8_t*>(data), size);
183 return;
184 }
185 }
186 m_elements.append(Vector { static_cast<const uint8_t*>(data), size });
187}

207static Vector<uint8_t> normalizeStringData(TextEncoding& encoding, const String& value)
208{
209 return normalizeLineEndingsToCRLF(encoding.encode(value, UnencodableHandling::Entities, NFCNormalize::No));
210}

212void FormData::appendMultiPartFileValue(const File& file, Vector<char>& header, TextEncoding& encoding)
213{
214 auto name = file.name();
215
216 // We have to include the filename=".." part in the header, even if the filename is empty
217 FormDataBuilder::addFilenameToMultiPartHeader(header, encoding, name);
218
219 // Add the content type if available, or "application/octet-stream" otherwise (RFC 1867).
220 auto contentType = file.type();
221 if (contentType.isEmpty())
222 contentType = "application/octet-stream"_s;
223 ASSERT(Blob::isNormalizedContentType(contentType));
224
225 FormDataBuilder::addContentTypeToMultiPartHeader(header, contentType.ascii());
226
227 FormDataBuilder::finishMultiPartHeader(header);
228 appendData(header.data(), header.size());
229
230 if (!file.path().isEmpty())
231 appendFile(file.path());
232 else if (file.size())
233 appendBlob(file.url());
234}
235
236void FormData::appendMultiPartStringValue(const String& string, Vector<char>& header, TextEncoding& encoding)
237{
238 FormDataBuilder::finishMultiPartHeader(header);
239 appendData(header.data(), header.size());
240
241 auto normalizedStringData = normalizeStringData(encoding, string);
242 appendData(normalizedStringData.data(), normalizedStringData.size());
243}
244
245void FormData::appendMultiPartKeyValuePairItems(const DOMFormData& formData)
246{
247 m_boundary = FormDataBuilder::generateUniqueBoundaryString();
248
249 auto encoding = formData.encoding();
250
251 Vector<char> encodedData;
252 for (auto& item : formData.items()) {
253 auto normalizedName = normalizeStringData(encoding, item.name);
254
255 Vector<char> header;
256 FormDataBuilder::beginMultiPartHeader(header, m_boundary.data(), normalizedName);
257
258 if (WTF::holds_alternative<RefPtr<File>>(item.data))
259 appendMultiPartFileValue(*WTF::get<RefPtr<File>>(item.data), header, encoding);
260 else
261 appendMultiPartStringValue(WTF::get<String>(item.data), header, encoding);
262
263 appendData("\r\n", 2);
264 }
265
266 FormDataBuilder::addBoundaryToMultiPartHeader(encodedData, m_boundary.data(), true);
267
268 appendData(encodedData.data(), encodedData.size());
269}
270
271void FormData::appendNonMultiPartKeyValuePairItems(const DOMFormData& formData, EncodingType encodingType)
272{
273 auto encoding = formData.encoding();
274
275 Vector<char> encodedData;
276 for (auto& item : formData.items()) {
277 ASSERT(WTF::holds_alternative<String>(item.data));
278
279 auto normalizedName = normalizeStringData(encoding, item.name);
280 auto normalizedStringData = normalizeStringData(encoding, WTF::get<String>(item.data));
281 FormDataBuilder::addKeyValuePairAsFormData(encodedData, normalizedName, normalizedStringData, encodingType);
282 }
283
284 appendData(encodedData.data(), encodedData.size());
285}

287Vector<uint8_t> FormData::flatten() const
288{
289 // Concatenate all the byte arrays, but omit any files.
290 Vector<uint8_t> data;
291 for (auto& element : m_elements) {
292 if (auto* vector = WTF::get_if<Vector<uint8_t>>(element.data))
293 data.append(vector->data(), vector->size());
294 }
295 return data;
296}
297
298String FormData::flattenToString() const
299{
300 auto bytes = flatten();
301 return Latin1Encoding().decode(bytes.data(), bytes.size());
302}

304static void appendBlobResolved(BlobRegistryImpl* blobRegistry, FormData& formData, const URL& url)
305{
306 if (!blobRegistry) {
307 LOG_ERROR("Tried to resolve a blob without a usable registry");
308 return;
309 }
310
311 auto* blobData = blobRegistry->getBlobDataFromURL(url);
312 if (!blobData) {
313 LOG_ERROR("Could not get blob data from a registry");
314 return;
315 }
316
317 for (const auto& blobItem : blobData->items()) {
318 if (blobItem.type() == BlobDataItem::Type::Data) {
319 ASSERT(blobItem.data().data());
320 formData.appendData(blobItem.data().data()->data() + static_cast<int>(blobItem.offset()), static_cast<int>(blobItem.length()));
321 } else if (blobItem.type() == BlobDataItem::Type::File)
322 formData.appendFileRange(blobItem.file()->path(), blobItem.offset(), blobItem.length(), blobItem.file()->expectedModificationTime());
323 else
324 ASSERT_NOT_REACHED();
325 }
326}

337Ref<FormData> FormData::resolveBlobReferences(BlobRegistryImpl* blobRegistryImpl)
338{
339 // First check if any blobs needs to be resolved, or we can take the fast path.
340 if (!containsBlobElement())
341 return *this;
342
343 // Create a copy to append the result into.
344 auto newFormData = FormData::create();
345 newFormData->setAlwaysStream(alwaysStream());
346 newFormData->setIdentifier(identifier());
347
348 for (auto& element : m_elements) {
349 switchOn(element.data,
350 [&] (const Vector<uint8_t>& bytes) {
351 newFormData->appendData(bytes.data(), bytes.size());
352 }, [&] (const FormDataElement::EncodedFileData& fileData) {
353 newFormData->appendFileRange(fileData.filename, fileData.fileStart, fileData.fileLength, fileData.expectedFileModificationTime);
354 }, [&] (const FormDataElement::EncodedBlobData& blobData) {
355 appendBlobResolved(blobRegistryImpl ? blobRegistryImpl : blobRegistry().blobRegistryImpl(), newFormData.get(), blobData.url);
356 }
357 );
358 }
359 return newFormData;
360}

362FormDataForUpload FormData::prepareForUpload()
363{
364 Vector<String> generatedFiles;
365 for (auto& element : m_elements) {
366 auto* fileData = WTF::get_if<FormDataElement::EncodedFileData>(element.data);
367 if (!fileData)
368 continue;
369 if (FileSystem::fileTypeFollowingSymlinks(fileData->filename) != FileSystem::FileType::Directory)
370 continue;
371 if (fileData->fileStart || fileData->fileLength != BlobDataItem::toEndOfFile)
372 continue;
373 if (!fileData->fileModificationTimeMatchesExpectation())
374 continue;
375
376 auto generatedFilename = FileSystem::createTemporaryZipArchive(fileData->filename);
377 if (!generatedFilename)
378 continue;
379 fileData->filename = generatedFilename;
380 generatedFiles.append(WTFMove(generatedFilename));
381 }
382
383 return { *this, WTFMove(generatedFiles) };
384}

429bool FormDataElement::EncodedFileData::fileModificationTimeMatchesExpectation() const
430{
431 if (!expectedFileModificationTime)
432 return true;
433
434 auto fileModificationTime = FileSystem::fileModificationTime(filename);
435 if (!fileModificationTime)
436 return false;
437
438 if (fileModificationTime->secondsSinceEpoch().secondsAs<time_t>() != expectedFileModificationTime->secondsSinceEpoch().secondsAs<time_t>())
439 return false;
440
441 return true;
442}
