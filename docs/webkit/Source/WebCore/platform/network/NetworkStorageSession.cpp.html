92bool NetworkStorageSession::shouldBlockThirdPartyCookies(const RegistrableDomain& registrableDomain) const
93{
94 if (!m_isResourceLoadStatisticsEnabled || registrableDomain.isEmpty())
95 return false;
96
97 ASSERT(!(m_registrableDomainsToBlockAndDeleteCookiesFor.contains(registrableDomain) && m_registrableDomainsToBlockButKeepCookiesFor.contains(registrableDomain)));
98
99 return m_registrableDomainsToBlockAndDeleteCookiesFor.contains(registrableDomain)
100 || m_registrableDomainsToBlockButKeepCookiesFor.contains(registrableDomain);
101}

134bool NetworkStorageSession::shouldBlockCookies(const URL& firstPartyForCookies, const URL& resource, std::optional<FrameIdentifier> frameID, std::optional<PageIdentifier> pageID, ShouldRelaxThirdPartyCookieBlocking shouldRelaxThirdPartyCookieBlocking) const
135{
136 if (shouldRelaxThirdPartyCookieBlocking == ShouldRelaxThirdPartyCookieBlocking::Yes)
137 return false;
138
139 if (!m_isResourceLoadStatisticsEnabled)
140 return false;
141
142 RegistrableDomain firstPartyDomain { firstPartyForCookies };
143 if (firstPartyDomain.isEmpty())
144 return false;
145
146 RegistrableDomain resourceDomain { resource };
147 if (resourceDomain.isEmpty())
148 return false;
149
150 if (firstPartyDomain == resourceDomain)
151 return false;
152
153 if (pageID && hasStorageAccess(resourceDomain, firstPartyDomain, frameID, pageID.value()))
154 return false;
155
156 switch (m_thirdPartyCookieBlockingMode) {
157 case ThirdPartyCookieBlockingMode::All:
158 return true;
159 case ThirdPartyCookieBlockingMode::AllExceptBetweenAppBoundDomains:
160 return !shouldExemptDomainPairFromThirdPartyCookieBlocking(firstPartyDomain, resourceDomain);
161 case ThirdPartyCookieBlockingMode::AllOnSitesWithoutUserInteraction:
162 if (!hasHadUserInteractionAsFirstParty(firstPartyDomain))
163 return true;
164 FALLTHROUGH;
165 case ThirdPartyCookieBlockingMode::OnlyAccordingToPerDomainPolicy:
166 return shouldBlockThirdPartyCookies(resourceDomain);
167 }
168 ASSERT_NOT_REACHED();
169 return false;
170}

172bool NetworkStorageSession::shouldExemptDomainPairFromThirdPartyCookieBlocking(const RegistrableDomain& topFrameDomain, const RegistrableDomain& resourceDomain) const
173{
174 ASSERT(topFrameDomain != resourceDomain);
175 if (topFrameDomain.isEmpty() || resourceDomain.isEmpty())
176 return false;
177
178 return topFrameDomain == resourceDomain || (m_appBoundDomains.contains(topFrameDomain) && m_appBoundDomains.contains(resourceDomain));
179}

219void NetworkStorageSession::grantCrossPageStorageAccess(const TopFrameDomain& topFrameDomain, const SubResourceDomain& resourceDomain)
220{
221 m_pairsGrantedCrossPageStorageAccess.ensure(topFrameDomain, [] { return HashSet<RegistrableDomain> { };
222 }).iterator->value.add(resourceDomain);
223
224 // Some sites have quirks where multiple login domains require storage access.
225 if (auto additionalLoginDomain = findAdditionalLoginDomain(topFrameDomain, resourceDomain)) {
226 m_pairsGrantedCrossPageStorageAccess.ensure(topFrameDomain, [] { return HashSet<RegistrableDomain> { };
227 }).iterator->value.add(*additionalLoginDomain);
228 }
229}

231bool NetworkStorageSession::hasStorageAccess(const RegistrableDomain& resourceDomain, const RegistrableDomain& firstPartyDomain, std::optional<FrameIdentifier> frameID, PageIdentifier pageID) const
232{
233 if (frameID) {
234 auto framesGrantedIterator = m_framesGrantedStorageAccess.find(pageID);
235 if (framesGrantedIterator != m_framesGrantedStorageAccess.end()) {
236 auto it = framesGrantedIterator->value.find(frameID.value());
237 if (it != framesGrantedIterator->value.end() && it->value == resourceDomain)
238 return true;
239 }
240 }
241
242 if (!firstPartyDomain.isEmpty()) {
243 auto pagesGrantedIterator = m_pagesGrantedStorageAccess.find(pageID);
244 if (pagesGrantedIterator != m_pagesGrantedStorageAccess.end()) {
245 auto it = pagesGrantedIterator->value.find(firstPartyDomain);
246 if (it != pagesGrantedIterator->value.end() && it->value == resourceDomain)
247 return true;
248 }
249
250 auto it = m_pairsGrantedCrossPageStorageAccess.find(firstPartyDomain);
251 if (it != m_pairsGrantedCrossPageStorageAccess.end() && it->value.contains(resourceDomain))
252 return true;
253 }
254
255 return false;
256}

268void NetworkStorageSession::grantStorageAccess(const RegistrableDomain& resourceDomain, const RegistrableDomain& firstPartyDomain, std::optional<FrameIdentifier> frameID, PageIdentifier pageID)
269{
270 if (NetworkStorageSession::loginDomainMatchesRequestingDomain(firstPartyDomain, resourceDomain)) {
271 grantCrossPageStorageAccess(firstPartyDomain, resourceDomain);
272 return;
273 }
274
275 if (!frameID) {
276 if (firstPartyDomain.isEmpty())
277 return;
278 auto pagesGrantedIterator = m_pagesGrantedStorageAccess.find(pageID);
279 if (pagesGrantedIterator == m_pagesGrantedStorageAccess.end()) {
280 HashMap<RegistrableDomain, RegistrableDomain> entry;
281 entry.add(firstPartyDomain, resourceDomain);
282 m_pagesGrantedStorageAccess.add(pageID, entry);
283 } else {
284 auto firstPartyDomainIterator = pagesGrantedIterator->value.find(firstPartyDomain);
285 if (firstPartyDomainIterator == pagesGrantedIterator->value.end())
286 pagesGrantedIterator->value.add(firstPartyDomain, resourceDomain);
287 else
288 firstPartyDomainIterator->value = resourceDomain;
289 }
290 return;
291 }
292
293 auto pagesGrantedIterator = m_framesGrantedStorageAccess.find(pageID);
294 if (pagesGrantedIterator == m_framesGrantedStorageAccess.end()) {
295 HashMap<FrameIdentifier, RegistrableDomain> entry;
296 entry.add(frameID.value(), resourceDomain);
297 m_framesGrantedStorageAccess.add(pageID, entry);
298 } else {
299 auto framesGrantedIterator = pagesGrantedIterator->value.find(frameID.value());
300 if (framesGrantedIterator == pagesGrantedIterator->value.end())
301 pagesGrantedIterator->value.add(frameID.value(), resourceDomain);
302 else
303 framesGrantedIterator->value = resourceDomain;
304 }
305}

369std::optional<Seconds> NetworkStorageSession::clientSideCookieCap(const RegistrableDomain& firstParty, std::optional<PageIdentifier> pageID) const
370{
371 if (!m_ageCapForClientSideCookies || !pageID || m_navigatedToWithLinkDecorationByPrevalentResource.isEmpty())
372 return m_ageCapForClientSideCookies;
373
374 auto domainIterator = m_navigatedToWithLinkDecorationByPrevalentResource.find(*pageID);
375 if (domainIterator == m_navigatedToWithLinkDecorationByPrevalentResource.end())
376 return m_ageCapForClientSideCookies;
377
378 if (domainIterator->value == firstParty)
379 return m_ageCapForClientSideCookiesShort;
380
381 return m_ageCapForClientSideCookies;
382}

384const HashMap<RegistrableDomain, HashSet<RegistrableDomain>>& NetworkStorageSession::storageAccessQuirks()
385{
386 static NeverDestroyed<HashMap<RegistrableDomain, HashSet<RegistrableDomain>>> map = [] {
387 HashMap<RegistrableDomain, HashSet<RegistrableDomain>> map;
388 map.add(RegistrableDomain::uncheckedCreateFromRegistrableDomainString("microsoft.com"),
389 HashSet { RegistrableDomain::uncheckedCreateFromRegistrableDomainString("microsoftonline.com"_s) });
390 map.add(RegistrableDomain::uncheckedCreateFromRegistrableDomainString("live.com"),
391 HashSet { RegistrableDomain::uncheckedCreateFromRegistrableDomainString("skype.com"_s) });
392 map.add(RegistrableDomain::uncheckedCreateFromRegistrableDomainString("playstation.com"), HashSet {
393 RegistrableDomain::uncheckedCreateFromRegistrableDomainString("sonyentertainmentnetwork.com"_s),
394 RegistrableDomain::uncheckedCreateFromRegistrableDomainString("sony.com"_s) });
395 map.add(RegistrableDomain::uncheckedCreateFromRegistrableDomainString("bbc.co.uk"), HashSet {
396 RegistrableDomain::uncheckedCreateFromRegistrableDomainString("radioplayer.co.uk"_s) });
397 return map;
398 }();
399 return map.get();
400}

402bool NetworkStorageSession::loginDomainMatchesRequestingDomain(const TopFrameDomain& topFrameDomain, const SubResourceDomain& resourceDomain)
403{
404 auto loginDomains = WebCore::NetworkStorageSession::subResourceDomainsInNeedOfStorageAccessForFirstParty(topFrameDomain);
405 return loginDomains && loginDomains.value().contains(resourceDomain);
406}

411}

413std::optional<HashSet<RegistrableDomain>> NetworkStorageSession::subResourceDomainsInNeedOfStorageAccessForFirstParty(const RegistrableDomain& topFrameDomain)
414{
415 auto it = storageAccessQuirks().find(topFrameDomain);
416 if (it != storageAccessQuirks().end())
417 return it->value;
418 return std::nullopt;
419}

421std::optional<RegistrableDomain> NetworkStorageSession::findAdditionalLoginDomain(const TopFrameDomain& topDomain, const SubResourceDomain& subDomain)
422{
423 if (subDomain.string() == "sony.com"_s && topDomain.string() == "playstation.com"_s)
424 return RegistrableDomain::uncheckedCreateFromRegistrableDomainString("sonyentertainmentnetwork.com"_s);
425
426 if (subDomain.string() == "sonyentertainmentnetwork.com"_s && topDomain.string() == "playstation.com"_s)
427 return RegistrableDomain::uncheckedCreateFromRegistrableDomainString("sony.com"_s);
428
429 return std::nullopt;
430}
