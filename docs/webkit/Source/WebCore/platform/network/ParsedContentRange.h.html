26#pragma once
27
28#include <wtf/Forward.h>
29#include <wtf/Markable.h>
30
31namespace WebCore {
32
33class ParsedContentRange {
34public:
35 static constexpr int64_t invalidLength = std::numeric_limits<int64_t>::min();
36 static constexpr int64_t unknownLength = std::numeric_limits<int64_t>::max();
37
38 WEBCORE_EXPORT explicit ParsedContentRange(const String&);
39 WEBCORE_EXPORT ParsedContentRange(int64_t firstBytePosition, int64_t lastBytePosition, int64_t instanceLength);
40 ParsedContentRange() = default;
41
42 bool isValid() const { return m_instanceLength != invalidLength; }
43 int64_t firstBytePosition() const { return m_firstBytePosition; }
44 int64_t lastBytePosition() const { return m_lastBytePosition; }
45 int64_t instanceLength() const { return m_instanceLength; }
46
47 static ParsedContentRange invalidValue()
48 {
49 return ParsedContentRange();
50 }
51
52 WEBCORE_EXPORT String headerValue() const;
53
54 struct MarkableTraits {
55 static bool isEmptyValue(const ParsedContentRange& range)
56 {
57 return !range.isValid();
58 }
59
60 static ParsedContentRange emptyValue()
61 {
62 return ParsedContentRange::invalidValue();
63 }
64 };
65
66private:
67 int64_t m_firstBytePosition { 0 };
68 int64_t m_lastBytePosition { 0 };
69 int64_t m_instanceLength { invalidLength };
70};
