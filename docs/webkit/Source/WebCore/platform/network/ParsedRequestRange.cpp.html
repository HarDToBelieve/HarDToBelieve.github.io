33std::optional<ParsedRequestRange> ParsedRequestRange::parse(StringView input)
34{
35 // https://tools.ietf.org/html/rfc7233#section-2.1 but assuming there will always be a begin and an end or parsing will fail
36 if (!input.startsWith(StringView("bytes="_s)))
37 return std::nullopt;
38
39 size_t begin { 0 };
40 size_t end { 0 };
41 size_t rangeBeginPosition = 6;
42 size_t dashPosition = input.find('-', rangeBeginPosition);
43 if (dashPosition == notFound)
44 return std::nullopt;
45
46 auto beginString = input.substring(rangeBeginPosition, dashPosition - rangeBeginPosition);
47 auto optionalBegin = parseInteger<uint64_t>(beginString);
48 if (!optionalBegin)
49 return std::nullopt;
50 begin = *optionalBegin;
51
52 auto endString = input.substring(dashPosition + 1);
53 auto optionalEnd = parseInteger<uint64_t>(endString);
54 if (!optionalEnd)
55 return std::nullopt;
56 end = *optionalEnd;
57
58 if (begin > end)
59 return std::nullopt;
60
61 return {{ begin, end }};
62}
