26#include "config.h"
27#include "ProxyServer.h"
28
29#include <wtf/text/StringBuilder.h>
30
31namespace WebCore {
32
33static void appendProxyServerString(StringBuilder& builder, const ProxyServer& proxyServer)
34{
35 ASSERT(!proxyServer.hostName().isNull());
36 ASSERT(proxyServer.port() != -1);
37
38 switch (proxyServer.type()) {
39 case ProxyServer::Direct:
40 builder.append("DIRECT");
41 break;
42 case ProxyServer::HTTP:
43 case ProxyServer::HTTPS:
44 builder.append("PROXY ", proxyServer.hostName(), ':', proxyServer.port());
45 break;
46 case ProxyServer::SOCKS:
47 builder.append("SOCKS ", proxyServer.hostName(), ':', proxyServer.port());
48 break;
49 }
50}
51
52String toString(const Vector<ProxyServer>& proxyServers)
53{
54 if (proxyServers.isEmpty())
55 return "DIRECT";
56
57 StringBuilder stringBuilder;
58 for (size_t i = 0; i < proxyServers.size(); ++i) {
59 if (i)
60 stringBuilder.append("; ");
61 appendProxyServerString(stringBuilder, proxyServers[i]);
62 }
63 return stringBuilder.toString();
64}
