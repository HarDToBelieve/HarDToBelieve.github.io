60void ResourceRequestBase::setAsIsolatedCopy(const ResourceRequest& other)
61{
62 setURL(other.url().isolatedCopy());
63 setCachePolicy(other.cachePolicy());
64 setTimeoutInterval(other.timeoutInterval());
65 setFirstPartyForCookies(other.firstPartyForCookies().isolatedCopy());
66 setHTTPMethod(other.httpMethod().isolatedCopy());
67 setPriority(other.priority());
68 setRequester(other.requester());
69 setInitiatorIdentifier(other.initiatorIdentifier().isolatedCopy());
70 setCachePartition(other.cachePartition().isolatedCopy());
71
72 if (auto inspectorInitiatorNodeIdentifier = other.inspectorInitiatorNodeIdentifier())
73 setInspectorInitiatorNodeIdentifier(*inspectorInitiatorNodeIdentifier);
74
75 if (!other.isSameSiteUnspecified())
76 setIsSameSite(other.isSameSite());
77 setIsTopSite(other.isTopSite());
78
79 updateResourceRequest();
80 m_httpHeaderFields = other.httpHeaderFields().isolatedCopy();
81
82 size_t encodingCount = other.m_responseContentDispositionEncodingFallbackArray.size();
83 if (encodingCount > 0) {
84 String encoding1 = other.m_responseContentDispositionEncodingFallbackArray[0].isolatedCopy();
85 String encoding2;
86 String encoding3;
87 if (encodingCount > 1) {
88 encoding2 = other.m_responseContentDispositionEncodingFallbackArray[1].isolatedCopy();
89 if (encodingCount > 2)
90 encoding3 = other.m_responseContentDispositionEncodingFallbackArray[2].isolatedCopy();
91 }
92 ASSERT(encodingCount <= 3);
93 setResponseContentDispositionEncodingFallbackArray(encoding1, encoding2, encoding3);
94 }
95 if (other.m_httpBody)
96 setHTTPBody(other.m_httpBody->isolatedCopy());
97 setAllowCookies(other.m_allowCookies);
98 setIsAppBound(other.isAppBound());
99}

131static bool shouldUseGet(const ResourceRequestBase& request, const ResourceResponse& redirectResponse)
132{
133 if (equalLettersIgnoringASCIICase(request.httpMethod(), "get") || equalLettersIgnoringASCIICase(request.httpMethod(), "head"))
134 return false;
135 if (redirectResponse.httpStatusCode() == 301 || redirectResponse.httpStatusCode() == 302)
136 return equalLettersIgnoringASCIICase(request.httpMethod(), "post");
137 return redirectResponse.httpStatusCode() == 303;
138}

140// https://fetch.spec.whatwg.org/#concept-http-redirect-fetch Step 11
141void ResourceRequestBase::redirectAsGETIfNeeded(const ResourceRequestBase &redirectRequest, const ResourceResponse& redirectResponse)
142{
143 if (shouldUseGet(redirectRequest, redirectResponse)) {
144 setHTTPMethod("GET"_s);
145 setHTTPBody(nullptr);
146 m_httpHeaderFields.remove(HTTPHeaderName::ContentLength);
147 m_httpHeaderFields.remove(HTTPHeaderName::ContentLanguage);
148 m_httpHeaderFields.remove(HTTPHeaderName::ContentEncoding);
149 m_httpHeaderFields.remove(HTTPHeaderName::ContentLocation);
150 clearHTTPContentType();
151 }
152}

154ResourceRequest ResourceRequestBase::redirectedRequest(const ResourceResponse& redirectResponse, bool shouldClearReferrerOnHTTPSToHTTPRedirect) const
155{
156 ASSERT(redirectResponse.isRedirection());
157 // This method is based on https://fetch.spec.whatwg.org/#http-redirect-fetch.
158 // It also implements additional processing like done by CFNetwork layer.
159
160 auto request = asResourceRequest();
161 auto location = redirectResponse.httpHeaderField(HTTPHeaderName::Location);
162
163 request.setURL(location.isEmpty() ? URL { } : URL { redirectResponse.url(), location });
164
165 request.redirectAsGETIfNeeded(*this, redirectResponse);
166
167 if (shouldClearReferrerOnHTTPSToHTTPRedirect && !request.url().protocolIs("https") && WTF::protocolIs(request.httpReferrer(), "https"))
168 request.clearHTTPReferrer();
169
170 if (!protocolHostAndPortAreEqual(request.url(), redirectResponse.url()))
171 request.clearHTTPOrigin();
172 request.clearHTTPAuthorization();
173 request.m_httpHeaderFields.remove(HTTPHeaderName::ProxyAuthorization);
174
175 return request;
176}

391void ResourceRequestBase::setHTTPReferrer(const String& httpReferrer)
392{
393 // https://w3c.github.io/webappsec-referrer-policy/#determine-requests-referrer
394 constexpr size_t maxLength = 4096;
395 if (httpReferrer.length() > maxLength) {
396 RELEASE_LOG(Loading, "Truncating HTTP referer");
397 String origin = SecurityOrigin::create(URL(URL(), httpReferrer))->toString();
398 if (origin.length() <= maxLength)
399 setHTTPHeaderField(HTTPHeaderName::Referer, origin);
400 } else
401 setHTTPHeaderField(HTTPHeaderName::Referer, httpReferrer);
402}

478void ResourceRequestBase::setResponseContentDispositionEncodingFallbackArray(const String& encoding1, const String& encoding2, const String& encoding3)
479{
480 updateResourceRequest();
481
482 m_responseContentDispositionEncodingFallbackArray.clear();
483 m_responseContentDispositionEncodingFallbackArray.reserveInitialCapacity(!encoding1.isNull() + !encoding2.isNull() + !encoding3.isNull());
484 if (!encoding1.isNull())
485 m_responseContentDispositionEncodingFallbackArray.uncheckedAppend(encoding1);
486 if (!encoding2.isNull())
487 m_responseContentDispositionEncodingFallbackArray.uncheckedAppend(encoding2);
488 if (!encoding3.isNull())
489 m_responseContentDispositionEncodingFallbackArray.uncheckedAppend(encoding3);
490
491 m_platformRequestUpdated = false;
492}

623bool equalIgnoringHeaderFields(const ResourceRequestBase& a, const ResourceRequestBase& b)
624{
625 if (a.url() != b.url())
626 return false;
627
628 if (a.cachePolicy() != b.cachePolicy())
629 return false;
630
631 if (a.timeoutInterval() != b.timeoutInterval())
632 return false;
633
634 if (a.firstPartyForCookies() != b.firstPartyForCookies())
635 return false;
636
637 if (a.isSameSite() != b.isSameSite())
638 return false;
639
640 if (a.isTopSite() != b.isTopSite())
641 return false;
642
643 if (a.httpMethod() != b.httpMethod())
644 return false;
645
646 if (a.allowCookies() != b.allowCookies())
647 return false;
648
649 if (a.priority() != b.priority())
650 return false;
651
652 if (a.requester() != b.requester())
653 return false;
654
655 return arePointingToEqualData(a.httpBody(), b.httpBody());
656}

658bool ResourceRequestBase::equal(const ResourceRequest& a, const ResourceRequest& b)
659{
660 if (!equalIgnoringHeaderFields(a, b))
661 return false;
662
663 if (a.httpHeaderFields() != b.httpHeaderFields())
664 return false;
665
666 return ResourceRequest::platformCompare(a, b);
667}

707void ResourceRequestBase::updatePlatformRequest(HTTPBodyUpdatePolicy bodyPolicy) const
708{
709 if (!m_platformRequestUpdated) {
710 ASSERT(m_resourceRequestUpdated);
711 const_cast<ResourceRequest&>(asResourceRequest()).doUpdatePlatformRequest();
712 m_platformRequestUpdated = true;
713 }
714
715 if (!m_platformRequestBodyUpdated && bodyPolicy == HTTPBodyUpdatePolicy::UpdateHTTPBody) {
716 ASSERT(m_resourceRequestBodyUpdated);
717 const_cast<ResourceRequest&>(asResourceRequest()).doUpdatePlatformHTTPBody();
718 m_platformRequestBodyUpdated = true;
719 }
720}

722void ResourceRequestBase::updateResourceRequest(HTTPBodyUpdatePolicy bodyPolicy) const
723{
724 if (!m_resourceRequestUpdated) {
725 ASSERT(m_platformRequestUpdated);
726 const_cast<ResourceRequest&>(asResourceRequest()).doUpdateResourceRequest();
727 m_resourceRequestUpdated = true;
728 }
729
730 if (!m_resourceRequestBodyUpdated && bodyPolicy == HTTPBodyUpdatePolicy::UpdateHTTPBody) {
731 ASSERT(m_platformRequestBodyUpdated);
732 const_cast<ResourceRequest&>(asResourceRequest()).doUpdateResourceHTTPBody();
733 m_resourceRequestBodyUpdated = true;
734 }
735}
