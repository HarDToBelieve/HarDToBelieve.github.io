40enum class ResourceRequestCachePolicy : uint8_t {
41 UseProtocolCachePolicy, // normal load, equivalent to fetch "default" cache mode.
42 ReloadIgnoringCacheData, // reload, equivalent to fetch "reload"cache mode.
43 ReturnCacheDataElseLoad, // back/forward or encoding change - allow stale data, equivalent to fetch "force-cache" cache mode.
44 ReturnCacheDataDontLoad, // results of a post - allow stale data and only use cache, equivalent to fetch "only-if-cached" cache mode.
45 DoNotUseAnyCache, // Bypass the cache entirely, equivalent to fetch "no-store" cache mode.
46 RefreshAnyCacheData, // Serve cache data only if revalidated, equivalent to fetch "no-cache" mode.
47};

282template<class Encoder>
283ALWAYS_INLINE void ResourceRequestBase::encodeBase(Encoder& encoder) const
284{
285 encoder << m_url;
286 encoder << m_timeoutInterval;
287 encoder << m_firstPartyForCookies.string();
288 encoder << m_httpMethod;
289 encoder << m_httpHeaderFields;
290 encoder << m_responseContentDispositionEncodingFallbackArray;
291 encoder << m_cachePolicy;
292 encoder << m_allowCookies;
293 encoder << m_sameSiteDisposition;
294 encoder << m_isTopSite;
295 encoder << m_priority;
296 encoder << m_requester;
297 encoder << m_isAppBound;
298}

300template<class Decoder>
301ALWAYS_INLINE bool ResourceRequestBase::decodeBase(Decoder& decoder)
302{
303 std::optional<URL> url;
304 decoder >> url;
305 if (!url)
306 return false;
307 m_url = WTFMove(*url);
308
309 std::optional<double> timeoutInterval;
310 decoder >> timeoutInterval;
311 if (!timeoutInterval)
312 return false;
313 m_timeoutInterval = WTFMove(*timeoutInterval);
314
315 std::optional<String> firstPartyForCookies;
316 decoder >> firstPartyForCookies;
317 if (!firstPartyForCookies)
318 return false;
319 m_firstPartyForCookies = URL({ }, *firstPartyForCookies);
320
321 std::optional<String> httpMethod;
322 decoder >> httpMethod;
323 if (!httpMethod)
324 return false;
325 m_httpMethod = WTFMove(*httpMethod);
326
327 std::optional<HTTPHeaderMap> fields;
328 decoder >> fields;
329 if (!fields)
330 return false;
331 m_httpHeaderFields = WTFMove(*fields);
332
333 std::optional<Vector<String>> array;
334 decoder >> array;
335 if (!array)
336 return false;
337 m_responseContentDispositionEncodingFallbackArray = WTFMove(*array);
338
339 std::optional<ResourceRequestCachePolicy> cachePolicy;
340 decoder >> cachePolicy;
341 if (!cachePolicy)
342 return false;
343 m_cachePolicy = *cachePolicy;
344
345 std::optional<bool> allowCookies;
346 decoder >> allowCookies;
347 if (!allowCookies)
348 return false;
349 m_allowCookies = *allowCookies;
350
351 std::optional<SameSiteDisposition> sameSiteDisposition;
352 decoder >> sameSiteDisposition;
353 if (!sameSiteDisposition)
354 return false;
355 m_sameSiteDisposition = *sameSiteDisposition;
356
357 std::optional<bool> isTopSite;
358 decoder >> isTopSite;
359 if (!isTopSite)
360 return false;
361 m_isTopSite = *isTopSite;
362
363 std::optional<ResourceLoadPriority> priority;
364 decoder >> priority;
365 if (!priority)
366 return false;
367 m_priority = *priority;
368
369 std::optional<Requester> requester;
370 decoder >> requester;
371 if (!requester)
372 return false;
373 m_requester = *requester;
374
375 std::optional<bool> isAppBound;
376 decoder >> isAppBound;
377 if (!isAppBound)
378 return false;
379 m_isAppBound = *isAppBound;
380
381 return true;
382}

384template<class Encoder>
385void ResourceRequestBase::encodeWithoutPlatformData(Encoder& encoder) const
386{
387 ASSERT(!m_httpBody);
388 ASSERT(!m_platformRequestUpdated);
389 encodeBase(encoder);
390}

392template<class Decoder>
393bool ResourceRequestBase::decodeWithoutPlatformData(Decoder& decoder)
394{
395 return decodeBase(decoder);
