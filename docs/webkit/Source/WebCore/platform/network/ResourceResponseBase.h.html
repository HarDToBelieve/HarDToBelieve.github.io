287template<class Encoder>
288void ResourceResponseBase::encode(Encoder& encoder) const
289{
290 encoder << m_isNull;
291 if (m_isNull)
292 return;
293 lazyInit(AllFields);
294
295 encoder << m_url;
296 encoder << m_mimeType;
297 encoder << static_cast<int64_t>(m_expectedContentLength);
298 encoder << m_textEncodingName;
299 encoder << m_httpStatusText;
300 encoder << m_httpVersion;
301 encoder << m_httpHeaderFields;
302
303 // We don't want to put the networkLoadMetrics info
304 // into the disk cache, because we will never use the old info.
305 if constexpr (Encoder::isIPCEncoder)
306 encoder << m_networkLoadMetrics;
307
308 encoder << m_httpStatusCode;
309 encoder << m_certificateInfo;
310 encoder << m_source;
311 encoder << m_type;
312 encoder << m_tainting;
313 encoder << m_isRedirected;
314 UsedLegacyTLS usedLegacyTLS = m_usedLegacyTLS;
315 encoder << usedLegacyTLS;
316 encoder << m_isRangeRequested;
317}

319template<class Decoder>
320bool ResourceResponseBase::decode(Decoder& decoder, ResourceResponseBase& response)
321{
322 ASSERT(response.m_isNull);
323 std::optional<bool> responseIsNull;
324 decoder >> responseIsNull;
325 if (!responseIsNull)
326 return false;
327 if (*responseIsNull)
328 return true;
329
330 response.m_isNull = false;
331
332 std::optional<URL> url;
333 decoder >> url;
334 if (!url)
335 return false;
336 response.m_url = WTFMove(*url);
337
338 std::optional<String> mimeType;
339 decoder >> mimeType;
340 if (!mimeType)
341 return false;
342 response.m_mimeType = WTFMove(*mimeType);
343
344 std::optional<int64_t> expectedContentLength;
345 decoder >> expectedContentLength;
346 if (!expectedContentLength)
347 return false;
348 response.m_expectedContentLength = *expectedContentLength;
349
350 std::optional<AtomString> textEncodingName;
351 decoder >> textEncodingName;
352 if (!textEncodingName)
353 return false;
354 response.m_textEncodingName = WTFMove(*textEncodingName);
355
356 std::optional<AtomString> httpStatusText;
357 decoder >> httpStatusText;
358 if (!httpStatusText)
359 return false;
360 response.m_httpStatusText = WTFMove(*httpStatusText);
361
362 std::optional<AtomString> httpVersion;
363 decoder >> httpVersion;
364 if (!httpVersion)
365 return false;
366 response.m_httpVersion = WTFMove(*httpVersion);
367
368 std::optional<HTTPHeaderMap> httpHeaderFields;
369 decoder >> httpHeaderFields;
370 if (!httpHeaderFields)
371 return false;
372 response.m_httpHeaderFields = WTFMove(*httpHeaderFields);
373
374 // The networkLoadMetrics info is only send over IPC and not stored in disk cache.
375 if constexpr (Decoder::isIPCDecoder) {
376 std::optional<Box<NetworkLoadMetrics>> networkLoadMetrics;
377 decoder >> networkLoadMetrics;
378 if (!networkLoadMetrics)
379 return false;
380 response.m_networkLoadMetrics = WTFMove(*networkLoadMetrics);
381 }
382
383 std::optional<short> httpStatusCode;
384 decoder >> httpStatusCode;
385 if (!httpStatusCode)
386 return false;
387 response.m_httpStatusCode = WTFMove(*httpStatusCode);
388
389 std::optional<std::optional<CertificateInfo>> certificateInfo;
390 decoder >> certificateInfo;
391 if (!certificateInfo)
392 return false;
393 response.m_certificateInfo = WTFMove(*certificateInfo);
394
395 std::optional<Source> source;
396 decoder >> source;
397 if (!source)
398 return false;
399 response.m_source = WTFMove(*source);
400
401 std::optional<Type> type;
402 decoder >> type;
403 if (!type)
404 return false;
405 response.m_type = WTFMove(*type);
406
407 std::optional<Tainting> tainting;
408 decoder >> tainting;
409 if (!tainting)
410 return false;
411 response.m_tainting = WTFMove(*tainting);
412
413 std::optional<bool> isRedirected;
414 decoder >> isRedirected;
415 if (!isRedirected)
416 return false;
417 response.m_isRedirected = WTFMove(*isRedirected);
418
419 std::optional<UsedLegacyTLS> usedLegacyTLS;
420 decoder >> usedLegacyTLS;
421 if (!usedLegacyTLS)
422 return false;
423 response.m_usedLegacyTLS = WTFMove(*usedLegacyTLS);
424
425 std::optional<bool> isRangeRequested;
426 decoder >> isRangeRequested;
427 if (!isRangeRequested)
428 return false;
429 response.m_isRangeRequested = WTFMove(*isRangeRequested);
430
431 return true;
432}
