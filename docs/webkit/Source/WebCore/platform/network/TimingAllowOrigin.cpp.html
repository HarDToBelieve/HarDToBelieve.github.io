33namespace WebCore {

35bool passesTimingAllowOriginCheck(const ResourceResponse& response, const SecurityOrigin& initiatorSecurityOrigin)
36{
37 // https://fetch.spec.whatwg.org/#tao-check
38 auto resourceOrigin = SecurityOrigin::create(response.url());
39 if (resourceOrigin->isSameSchemeHostPort(initiatorSecurityOrigin))
40 return true;
41
42 const auto& timingAllowOriginString = response.httpHeaderField(HTTPHeaderName::TimingAllowOrigin);
43 const auto& securityOrigin = initiatorSecurityOrigin.toString();
44 for (auto originWithSpace : StringView(timingAllowOriginString).split(',')) {
45 auto origin = stripLeadingAndTrailingHTTPSpaces(originWithSpace);
46 if (origin == "*" || origin == securityOrigin)
47 return true;
48 }
49
50 return false;
51}
