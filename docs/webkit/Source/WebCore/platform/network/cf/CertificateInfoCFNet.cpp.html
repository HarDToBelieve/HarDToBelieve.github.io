35#if PLATFORM(COCOA)
36bool certificatesMatch(SecTrustRef trust1, SecTrustRef trust2)
37{
38 if (!trust1 || !trust2)
39 return false;
40
41#if HAVE(SEC_TRUST_COPY_CERTIFICATE_CHAIN)
42 auto chain1 = adoptCF(SecTrustCopyCertificateChain(trust1));
43 auto chain2 = adoptCF(SecTrustCopyCertificateChain(trust2));
44#endif
45
46 CFIndex count1 = SecTrustGetCertificateCount(trust1);
47 CFIndex count2 = SecTrustGetCertificateCount(trust2);
48 if (count1 != count2)
49 return false;
50
51 for (CFIndex i = 0; i < count1; i++) {
52#if HAVE(SEC_TRUST_COPY_CERTIFICATE_CHAIN)
53 auto cert1 = CFArrayGetValueAtIndex(chain1.get(), i);
54 auto cert2 = CFArrayGetValueAtIndex(chain2.get(), i);
55#else
56 auto cert1 = SecTrustGetCertificateAtIndex(trust1, i);
57 auto cert2 = SecTrustGetCertificateAtIndex(trust2, i);
58#endif
59 RELEASE_ASSERT(cert1);
60 RELEASE_ASSERT(cert2);
61 if (!CFEqual(cert1, cert2))
62 return false;
63 }
64
65 return true;
66}

68RetainPtr<CFArrayRef> CertificateInfo::certificateChainFromSecTrust(SecTrustRef trust)
69{
70#if HAVE(SEC_TRUST_COPY_CERTIFICATE_CHAIN)
71 return adoptCF(SecTrustCopyCertificateChain(trust));
72#else
73 auto count = SecTrustGetCertificateCount(trust);
74 auto certificateChain = adoptCF(CFArrayCreateMutable(0, count, &kCFTypeArrayCallBacks));
75 for (CFIndex i = 0; i < count; i++)
76 CFArrayAppendValue(certificateChain.get(), SecTrustGetCertificateAtIndex(trust, i));
77 return certificateChain;
78#endif
79}
80#endif

82CertificateInfo::Type CertificateInfo::type() const
83{
84#if HAVE(SEC_TRUST_SERIALIZATION)
85 if (m_trust)
86 return Type::Trust;
87#endif
88 if (m_certificateChain)
89 return Type::CertificateChain;
90 return Type::None;

93CFArrayRef CertificateInfo::certificateChain() const
94{
95#if HAVE(SEC_TRUST_SERIALIZATION)
96 if (m_certificateChain)
97 return m_certificateChain.get();
98
99 if (m_trust)
100 m_certificateChain = CertificateInfo::certificateChainFromSecTrust(m_trust.get());
101#endif
102
103 return m_certificateChain.get();

106bool CertificateInfo::containsNonRootSHA1SignedCertificate() const
107{
108#if HAVE(SEC_TRUST_SERIALIZATION)
109 if (m_trust) {
110#if HAVE(SEC_TRUST_COPY_CERTIFICATE_CHAIN)
111 auto chain = adoptCF(SecTrustCopyCertificateChain(trust()));
112#endif
113 // Allow only the root certificate (the last in the chain) to be SHA1.
114 for (CFIndex i = 0, size = SecTrustGetCertificateCount(trust()) - 1; i < size; ++i) {
115#if HAVE(SEC_TRUST_COPY_CERTIFICATE_CHAIN)
116 auto certificate = checked_cf_cast<SecCertificateRef>(CFArrayGetValueAtIndex(chain.get(), i));
117#else
118 auto certificate = SecTrustGetCertificateAtIndex(trust(), i);
119#endif
120 if (SecCertificateGetSignatureHashAlgorithm(certificate) == kSecSignatureHashAlgorithmSHA1)
121 return true;
122 }
123
124 return false;
125 }
126#endif // HAVE(SEC_TRUST_SERIALIZATION)
127
128#if PLATFORM(COCOA)
129 if (m_certificateChain) {
130 // Allow only the root certificate (the last in the chain) to be SHA1.
131 for (CFIndex i = 0, size = CFArrayGetCount(m_certificateChain.get()) - 1; i < size; ++i) {
132 auto certificate = checked_cf_cast<SecCertificateRef>(CFArrayGetValueAtIndex(m_certificateChain.get(), i));
133 if (SecCertificateGetSignatureHashAlgorithm(certificate) == kSecSignatureHashAlgorithmSHA1)
134 return true;
135 }
136 return false;
137 }
138#endif
139
140 return false;
141}

143std::optional<CertificateSummary> CertificateInfo::summary() const
144{
145 auto chain = certificateChain();
146 if (!chain)
147 return std::nullopt;
148
149 CertificateSummary summaryInfo;
150
151#if PLATFORM(COCOA) && !PLATFORM(IOS_FAMILY_SIMULATOR) && !PLATFORM(MACCATALYST)
152 auto leafCertificate = checked_cf_cast<SecCertificateRef>(CFArrayGetValueAtIndex(chain, 0));
153
154 auto subjectCF = adoptCF(SecCertificateCopySubjectSummary(leafCertificate));
155 summaryInfo.subject = subjectCF.get();
156#endif
157
158#if PLATFORM(MAC)
159 if (auto certificateDictionary = adoptCF(SecCertificateCopyValues(leafCertificate, nullptr, nullptr))) {
160 // CFAbsoluteTime is relative to 01/01/1970 00:00:00 GMT.
161 const Seconds absoluteReferenceDate(978307200);
162
163 if (auto validNotBefore = checked_cf_cast<CFDictionaryRef>(CFDictionaryGetValue(certificateDictionary.get(), kSecOIDX509V1ValidityNotBefore))) {
164 if (auto number = checked_cf_cast<CFNumberRef>(CFDictionaryGetValue(validNotBefore, CFSTR("value")))) {
165 double numberValue;
166 if (CFNumberGetValue(number, kCFNumberDoubleType, &numberValue))
167 summaryInfo.validFrom = absoluteReferenceDate + Seconds(numberValue);
168 }
169 }
170
171 if (auto validNotAfter = checked_cf_cast<CFDictionaryRef>(CFDictionaryGetValue(certificateDictionary.get(), kSecOIDX509V1ValidityNotAfter))) {
172 if (auto number = checked_cf_cast<CFNumberRef>(CFDictionaryGetValue(validNotAfter, CFSTR("value")))) {
173 double numberValue;
174 if (CFNumberGetValue(number, kCFNumberDoubleType, &numberValue))
175 summaryInfo.validUntil = absoluteReferenceDate + Seconds(numberValue);
176 }
177 }
178
179 if (auto dnsNames = checked_cf_cast<CFDictionaryRef>(CFDictionaryGetValue(certificateDictionary.get(), CFSTR("DNSNAMES")))) {
180 if (auto dnsNamesArray = checked_cf_cast<CFArrayRef>(CFDictionaryGetValue(dnsNames, CFSTR("value")))) {
181 for (CFIndex i = 0, count = CFArrayGetCount(dnsNamesArray); i < count; ++i) {
182 if (auto dnsName = checked_cf_cast<CFStringRef>(CFArrayGetValueAtIndex(dnsNamesArray, i)))
183 summaryInfo.dnsNames.append(dnsName);
184 }
185 }
186 }
187
188 if (auto ipAddresses = checked_cf_cast<CFDictionaryRef>(CFDictionaryGetValue(certificateDictionary.get(), CFSTR("IPADDRESSES")))) {
189 if (auto ipAddressesArray = checked_cf_cast<CFArrayRef>(CFDictionaryGetValue(ipAddresses, CFSTR("value")))) {
190 for (CFIndex i = 0, count = CFArrayGetCount(ipAddressesArray); i < count; ++i) {
191 if (auto ipAddress = checked_cf_cast<CFStringRef>(CFArrayGetValueAtIndex(ipAddressesArray, i)))
192 summaryInfo.ipAddresses.append(ipAddress);
193 }
194 }
195 }
196 }
197#endif
198
199 return summaryInfo;
200}

207static void encodeCFData(Encoder& encoder, CFDataRef data)
208{
209 uint64_t length = CFDataGetLength(data);
210 const uint8_t* bytePtr = CFDataGetBytePtr(data);
211
212 encoder << length;
213 encoder.encodeFixedLengthData(bytePtr, static_cast<size_t>(length));
214}
215
216static std::optional<RetainPtr<CFDataRef>> decodeCFData(Decoder& decoder)
217{
218 std::optional<uint64_t> size;
219 decoder >> size;
220
221 if (UNLIKELY(!isInBounds<size_t>(*size)))
222 return std::nullopt;
223
224 auto pointer = decoder.bufferPointerForDirectRead(static_cast<size_t>(*size));
225 if (!pointer)
226 return std::nullopt;
227
228 return adoptCF(CFDataCreate(nullptr, pointer, *size));
229}

231#if HAVE(SEC_TRUST_SERIALIZATION)
232static void encodeSecTrustRef(Encoder& encoder, SecTrustRef trust)
233{
234 auto data = adoptCF(SecTrustSerialize(trust, nullptr));
235 if (!data) {
236 encoder << false;
237 return;
238 }
239
240 encoder << true;
241 encodeCFData(encoder, data.get());
242}
243
244static std::optional<RetainPtr<SecTrustRef>> decodeSecTrustRef(Decoder& decoder)
245{
246 std::optional<bool> hasTrust;
247 decoder >> hasTrust;
248 if (!hasTrust)
249 return std::nullopt;
250
251 if (!*hasTrust)
252 return { nullptr };
253
254 auto trustData = decodeCFData(decoder);
255 if (!trustData)
256 return std::nullopt;
257
258 auto trust = adoptCF(SecTrustDeserialize(trustData->get(), nullptr));
259 if (!trust)
260 return std::nullopt;
261
262 return trust;
263}
264#endif

266#if PLATFORM(COCOA)
267static void encodeCertificateChain(Encoder& encoder, CFArrayRef certificateChain)
268{
269 CFIndex size = CFArrayGetCount(certificateChain);
270 Vector<CFTypeRef, 32> values(size);
271
272 CFArrayGetValues(certificateChain, CFRangeMake(0, size), values.data());
273
274 encoder << static_cast<uint64_t>(size);
275
276 for (CFIndex i = 0; i < size; ++i) {
277 ASSERT(values[i]);
278 auto data = adoptCF(SecCertificateCopyData(checked_cf_cast<SecCertificateRef>(values[i])));
279 encodeCFData(encoder, data.get());
280 }
281}
282
283static std::optional<RetainPtr<CFArrayRef>> decodeCertificateChain(Decoder& decoder)
284{
285 std::optional<uint64_t> size;
286 decoder >> size;
287 if (!size)
288 return std::nullopt;
289
290 auto array = adoptCF(CFArrayCreateMutable(0, 0, &kCFTypeArrayCallBacks));
291
292 for (size_t i = 0; i < *size; ++i) {
293 auto data = decodeCFData(decoder);
294 if (!data)
295 return std::nullopt;
296
297 auto certificate = adoptCF(SecCertificateCreateWithData(0, data->get()));
298 CFArrayAppendValue(array.get(), certificate.get());
299 }
300
301 return { WTFMove(array) };
302}
303#endif

305void Coder<WebCore::CertificateInfo>::encode(Encoder& encoder, const WebCore::CertificateInfo& certificateInfo)
306{
307 encoder << certificateInfo.type();
308
309 switch (certificateInfo.type()) {
310#if HAVE(SEC_TRUST_SERIALIZATION)
311 case WebCore::CertificateInfo::Type::Trust:
312 encodeSecTrustRef(encoder, certificateInfo.trust());
313 break;
314#endif
315#if PLATFORM(COCOA)
316 case WebCore::CertificateInfo::Type::CertificateChain: {
317 encodeCertificateChain(encoder, certificateInfo.certificateChain());
318 break;
319 }
320#endif
321 case WebCore::CertificateInfo::Type::None:
322 // Do nothing.
323 break;
324 }
325}
326
327std::optional<WebCore::CertificateInfo> Coder<WebCore::CertificateInfo>::decode(Decoder& decoder)
328{
329 std::optional<WebCore::CertificateInfo::Type> certificateInfoType;
330 decoder >> certificateInfoType;
331 if (!certificateInfoType)
332 return std::nullopt;
333
334 switch (*certificateInfoType) {
335#if HAVE(SEC_TRUST_SERIALIZATION)
336 case WebCore::CertificateInfo::Type::Trust: {
337 auto trust = decodeSecTrustRef(decoder);
338 if (!trust)
339 return std::nullopt;
340
341 return WebCore::CertificateInfo(WTFMove(*trust));
342 }
343#endif
344#if PLATFORM(COCOA)
345 case WebCore::CertificateInfo::Type::CertificateChain: {
346 auto certificateChain = decodeCertificateChain(decoder);
347 if (!certificateChain)
348 return std::nullopt;
349
350 return WebCore::CertificateInfo(WTFMove(*certificateChain));
351 }
352#endif
353 case WebCore::CertificateInfo::Type::None:
354 // Do nothing.
355 return WebCore::CertificateInfo();
356 }
357
358 return std::nullopt;
359}
