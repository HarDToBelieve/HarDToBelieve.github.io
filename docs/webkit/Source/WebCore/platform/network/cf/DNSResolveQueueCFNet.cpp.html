50void DNSResolveQueueCFNet::updateIsUsingProxy()
51{
52 RetainPtr<CFDictionaryRef> proxySettings = adoptCF(CFNetworkCopySystemProxySettings());
53 if (!proxySettings) {
54 m_isUsingProxy = false;
55 return;
56 }
57
58 RetainPtr<CFURLRef> httpCFURL = URL({ }, "http://example.com/").createCFURL();
59 RetainPtr<CFURLRef> httpsCFURL = URL({ }, "https://example.com/").createCFURL();
60
61 RetainPtr<CFArrayRef> httpProxyArray = adoptCF(CFNetworkCopyProxiesForURL(httpCFURL.get(), proxySettings.get()));
62 RetainPtr<CFArrayRef> httpsProxyArray = adoptCF(CFNetworkCopyProxiesForURL(httpsCFURL.get(), proxySettings.get()));
63
64 CFIndex httpProxyCount = CFArrayGetCount(httpProxyArray.get());
65 CFIndex httpsProxyCount = CFArrayGetCount(httpsProxyArray.get());
66 if (httpProxyCount == 1 && CFEqual(CFDictionaryGetValue(static_cast<CFDictionaryRef>(CFArrayGetValueAtIndex(httpProxyArray.get(), 0)), kCFProxyTypeKey), kCFProxyTypeNone))
67 httpProxyCount = 0;
68 if (httpsProxyCount == 1 && CFEqual(CFDictionaryGetValue(static_cast<CFDictionaryRef>(CFArrayGetValueAtIndex(httpsProxyArray.get(), 0)), kCFProxyTypeKey), kCFProxyTypeNone))
69 httpsProxyCount = 0;
70
71 m_isUsingProxy = httpProxyCount || httpsProxyCount;
72}

74static void clientCallback(CFHostRef theHost, CFHostInfoType, const CFStreamError*, void*)
75{
76 DNSResolveQueue::singleton().decrementRequestCount(); // It's ok to call singleton() from a secondary thread, the static variable has already been initialized by now.
77 CFRelease(theHost);
78}

80void DNSResolveQueueCFNet::platformResolve(const String& hostname)
81{
82 ASSERT(isMainThread());
83
84 RetainPtr<CFHostRef> host = adoptCF(CFHostCreateWithName(0, hostname.createCFString().get()));
85 if (!host) {
86 decrementRequestCount();
87 return;
88 }
89
90 CFHostClientContext context = { 0, 0, 0, 0, 0 };
91 CFHostRef leakedHost = host.leakRef(); // The host will be released from clientCallback().
92 Boolean result = CFHostSetClient(leakedHost, clientCallback, &context);
93 ASSERT_UNUSED(result, result);
94#if !PLATFORM(WIN)
95 CFHostScheduleWithRunLoop(leakedHost, CFRunLoopGetMain(), kCFRunLoopCommonModes);
96#else
97 // On Windows, we run a separate thread with CFRunLoop, which is where clientCallback will be called.
98 CFHostScheduleWithRunLoop(leakedHost, loaderRunLoop(), kCFRunLoopDefaultMode);
99#endif
100 CFHostStartInfoResolution(leakedHost, kCFHostAddresses, 0);
101}
