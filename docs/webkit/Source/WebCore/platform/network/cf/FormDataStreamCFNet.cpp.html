98struct FormStreamFields {
99 FormStreamFields(FormDataForUpload&& data)
100 : data(WTFMove(data)) { }
101
102 FormDataForUpload data;
103 SchedulePairHashSet scheduledRunLoopPairs;
104 Vector<FormDataElement> remainingElements; // in reverse order
105 RetainPtr<CFReadStreamRef> currentStream;
106 long long currentStreamRangeLength { BlobDataItem::toEndOfFile };
107 MallocPtr<uint8_t, WTF::VectorMalloc> currentData;
108 CFReadStreamRef formStream { nullptr };
109 unsigned long long streamLength { 0 };
110 unsigned long long bytesSent { 0 };
111 Lock streamIsBeingOpenedOrClosedLock;
112};
113
114static void closeCurrentStream(FormStreamFields* form)
115{
116 ASSERT(form->streamIsBeingOpenedOrClosedLock.isHeld());
117
118 if (form->currentStream) {
119 CFReadStreamClose(form->currentStream.get());
120 CFReadStreamSetClient(form->currentStream.get(), kCFStreamEventNone, 0, 0);
121 form->currentStream = nullptr;
122 form->currentStreamRangeLength = BlobDataItem::toEndOfFile;
123 }
124 form->currentData = nullptr;
125}
126
127// Return false if we cannot advance the stream. Currently the only possible failure is that the underlying file has been removed or changed since File.slice.
128static bool advanceCurrentStream(FormStreamFields* form)
129{
130 ASSERT(form->streamIsBeingOpenedOrClosedLock.isHeld());
131
132 closeCurrentStream(form);
133
134 if (form->remainingElements.isEmpty())
135 return true;
136
137 // Create the new stream.
138 FormDataElement& nextInput = form->remainingElements.last();
139
140 bool success = switchOn(nextInput.data,
141 [form] (Vector<uint8_t>& bytes) {
142 size_t size = bytes.size();
143 MallocPtr<uint8_t, WTF::VectorMalloc> data = bytes.releaseBuffer();
144 form->currentStream = adoptCF(CFReadStreamCreateWithBytesNoCopy(0, data.get(), size, kCFAllocatorNull));
145 form->currentData = WTFMove(data);
146 return true;
147 }, [form] (const FormDataElement::EncodedFileData& fileData) {
148 // Check if the file has been changed.
149 if (!fileData.fileModificationTimeMatchesExpectation())
150 return false;
151
152 const String& path = fileData.filename;
153 form->currentStream = adoptCF(CFReadStreamCreateWithFile(0, FileSystem::pathAsURL(path).get()));
154 if (!form->currentStream) {
155 // The file must have been removed or become unreadable.
156 return false;
157 }
158 if (fileData.fileStart > 0) {
159 RetainPtr<CFNumberRef> position = adoptCF(CFNumberCreate(0, kCFNumberLongLongType, &fileData.fileStart));
160 CFReadStreamSetProperty(form->currentStream.get(), kCFStreamPropertyFileCurrentOffset, position.get());
161 }
162 form->currentStreamRangeLength = fileData.fileLength;
163 return true;
164 }, [] (const FormDataElement::EncodedBlobData&) {
165 ASSERT_NOT_REACHED();
166 return false;
167 }
168 );
169
170 if (!success)
171 return false;
172
173 form->remainingElements.removeLast();
174
175 // Set up the callback.
176 CFStreamClientContext context = { 0, form, 0, 0, 0 };
177 CFReadStreamSetClient(form->currentStream.get(), kCFStreamEventHasBytesAvailable | kCFStreamEventErrorOccurred | kCFStreamEventEndEncountered,
178 formEventCallback, &context);
179
180 // Schedule with the current set of run loops.
181 SchedulePairHashSet::iterator end = form->scheduledRunLoopPairs.end();
182 for (SchedulePairHashSet::iterator it = form->scheduledRunLoopPairs.begin(); it != end; ++it)
183 CFReadStreamScheduleWithRunLoop(form->currentStream.get(), (*it)->runLoop(), (*it)->mode());
184
185 return true;
186}

188static bool openNextStream(FormStreamFields* form)
189{
190 // CFReadStreamOpen() can cause this function to be re-entered from another thread before it returns.
191 // One example when this can occur is when the stream being opened has no data. See <rdar://problem/23550269>.
192 Locker locker { form->streamIsBeingOpenedOrClosedLock };
193
194 // Skip over any streams we can't open.
195 if (!advanceCurrentStream(form))
196 return false;
197 while (form->currentStream && !CFReadStreamOpen(form->currentStream.get())) {
198 if (!advanceCurrentStream(form))
199 return false;
200 }
201 return true;
202}

204static void* formCreate(CFReadStreamRef stream, void* context)
205{
206 FormCreationContext* formContext = static_cast<FormCreationContext*>(context);
207
208 FormStreamFields* newInfo = new FormStreamFields(WTFMove(formContext->data));
209 newInfo->formStream = stream; // Don't retain. That would create a reference cycle.
210 newInfo->streamLength = formContext->streamLength;
211
212 callOnMainThread([formContext] {
213 delete formContext;
214 });
215
216 // Append in reverse order since we remove elements from the end.
217 size_t size = newInfo->data.data().elements().size();
218 newInfo->remainingElements.reserveInitialCapacity(size);
219 for (size_t i = 0; i < size; ++i)
220 newInfo->remainingElements.uncheckedAppend(newInfo->data.data().elements()[size - i - 1]);
221
222 return newInfo;
223}

225static void formFinalize(CFReadStreamRef stream, void* context)
226{
227 FormStreamFields* form = static_cast<FormStreamFields*>(context);
228 ASSERT_UNUSED(stream, form->formStream == stream);
229
230 callOnMainThread([form] {
231 {
232 Locker locker { form->streamIsBeingOpenedOrClosedLock };
233 closeCurrentStream(form);
234 }
235 delete form;
236 });
237}

239static Boolean formOpen(CFReadStreamRef, CFStreamError* error, Boolean* openComplete, void* context)
240{
241 FormStreamFields* form = static_cast<FormStreamFields*>(context);
242
243 bool opened = openNextStream(form);
244
245 *openComplete = opened;
246 error->error = opened ? 0 :
247#if PLATFORM(WIN)
248 ENOENT;
249#else
250 fileNotFoundError;
251#endif
252 return opened;
253}

255static CFIndex formRead(CFReadStreamRef, UInt8* buffer, CFIndex bufferLength, CFStreamError* error, Boolean* atEOF, void* context)
256{
257 FormStreamFields* form = static_cast<FormStreamFields*>(context);
258
259 while (form->currentStream) {
260 CFIndex bytesToRead = bufferLength;
261 if (form->currentStreamRangeLength != BlobDataItem::toEndOfFile && form->currentStreamRangeLength < bytesToRead)
262 bytesToRead = static_cast<CFIndex>(form->currentStreamRangeLength);
263 CFIndex bytesRead = CFReadStreamRead(form->currentStream.get(), buffer, bytesToRead);
264 if (bytesRead < 0) {
265 *error = CFReadStreamGetError(form->currentStream.get());
266 return -1;
267 }
268 if (bytesRead > 0) {
269 error->error = 0;
270 *atEOF = FALSE;
271 form->bytesSent += bytesRead;
272 if (form->currentStreamRangeLength != BlobDataItem::toEndOfFile)
273 form->currentStreamRangeLength -= bytesRead;
274
275 return bytesRead;
276 }
277 openNextStream(form);
278 }
279
280 error->error = 0;
281 *atEOF = TRUE;
282 return 0;
283}

285static Boolean formCanRead(CFReadStreamRef stream, void* context)
286{
287 FormStreamFields* form = static_cast<FormStreamFields*>(context);
288
289 while (form->currentStream && CFReadStreamGetStatus(form->currentStream.get()) == kCFStreamStatusAtEnd)
290 openNextStream(form);
291
292 if (!form->currentStream) {
293 CFReadStreamSignalEvent(stream, kCFStreamEventEndEncountered, 0);
294 return FALSE;
295 }
296 return CFReadStreamHasBytesAvailable(form->currentStream.get());
297}

301 FormStreamFields* form = static_cast<FormStreamFields*>(context);
302 Locker locker { form->streamIsBeingOpenedOrClosedLock };
303
304 closeCurrentStream(form);

307static CFTypeRef formCopyProperty(CFReadStreamRef, CFStringRef propertyName, void *context)
308{
309 FormStreamFields* form = static_cast<FormStreamFields*>(context);
310
311 if (kCFCompareEqualTo == CFStringCompare(propertyName, formDataPointerPropertyName, 0)) {
312 long long formDataAsNumber = static_cast<long long>(reinterpret_cast<intptr_t>(&form->data.data()));
313 return CFNumberCreate(0, kCFNumberLongLongType, &formDataAsNumber);
314 }
315
316 if (kCFCompareEqualTo == CFStringCompare(propertyName, formDataStreamLengthPropertyName(), 0))
317 return CFStringCreateWithFormat(0, 0, CFSTR("%llu"), form->streamLength);
318
319 return 0;
320}

322static void formSchedule(CFReadStreamRef, CFRunLoopRef runLoop, CFStringRef runLoopMode, void* context)
323{
324 FormStreamFields* form = static_cast<FormStreamFields*>(context);
325
326 if (form->currentStream)
327 CFReadStreamScheduleWithRunLoop(form->currentStream.get(), runLoop, runLoopMode);
328 form->scheduledRunLoopPairs.add(SchedulePair::create(runLoop, runLoopMode));
329}

331static void formUnschedule(CFReadStreamRef, CFRunLoopRef runLoop, CFStringRef runLoopMode, void* context)
332{
333 FormStreamFields* form = static_cast<FormStreamFields*>(context);
334
335 if (form->currentStream)
336 CFReadStreamUnscheduleFromRunLoop(form->currentStream.get(), runLoop, runLoopMode);
337 form->scheduledRunLoopPairs.remove(SchedulePair::create(runLoop, runLoopMode));
338}

340static void formEventCallback(CFReadStreamRef stream, CFStreamEventType type, void* context)
341{
342 FormStreamFields* form = static_cast<FormStreamFields*>(context);
343
344 switch (type) {
345 case kCFStreamEventHasBytesAvailable:
346 CFReadStreamSignalEvent(form->formStream, kCFStreamEventHasBytesAvailable, 0);
347 break;
348 case kCFStreamEventErrorOccurred: {
349 CFStreamError readStreamError = CFReadStreamGetError(stream);
350 CFReadStreamSignalEvent(form->formStream, kCFStreamEventErrorOccurred, &readStreamError);
351 break;
352 }
353 case kCFStreamEventEndEncountered:
354 openNextStream(form);
355 if (!form->currentStream)
356 CFReadStreamSignalEvent(form->formStream, kCFStreamEventEndEncountered, 0);
357 break;
358 case kCFStreamEventNone:
359 LOG_ERROR("unexpected kCFStreamEventNone");
360 break;
361 case kCFStreamEventOpenCompleted:
362 LOG_ERROR("unexpected kCFStreamEventOpenCompleted");
363 break;
364 case kCFStreamEventCanAcceptBytes:
365 LOG_ERROR("unexpected kCFStreamEventCanAcceptBytes");
366 break;
367 }
368}

370RetainPtr<CFReadStreamRef> createHTTPBodyCFReadStream(FormData& formData)
371{
372 if (!hasPlatformStrategies() && formData.containsBlobElement())
373 return nullptr;
374
375 auto resolvedFormData = formData.resolveBlobReferences();
376 auto dataForUpload = resolvedFormData->prepareForUpload();
377
378 // Precompute the content length so CFNetwork doesn't use chunked mode.
379 unsigned long long length = 0;
380 for (auto& element : dataForUpload.data().elements()) {
381 length += element.lengthInBytes([](auto& url) {
382 return blobRegistry().blobRegistryImpl()->blobSize(url);
383 });
384 }
385 FormCreationContext* formContext = new FormCreationContext { WTFMove(dataForUpload), length };
386 CFReadStreamCallBacksV1 callBacks = { 1, formCreate, formFinalize, nullptr, formOpen, nullptr, formRead, nullptr, formCanRead, formClose, formCopyProperty, nullptr, nullptr, formSchedule, formUnschedule };
387 return adoptCF(CFReadStreamCreate(nullptr, static_cast<const void*>(&callBacks), formContext));
388}

390void setHTTPBody(CFMutableURLRequestRef request, FormData* formData)
391{
392 if (!formData)
393 return;
394
395 // Handle the common special case of one piece of form data, with no files.
396 auto& elements = formData->elements();
397 if (elements.size() == 1 && !formData->alwaysStream()) {
398 if (auto* vector = WTF::get_if<Vector<uint8_t>>(elements[0].data)) {
399 auto data = adoptCF(CFDataCreate(nullptr, vector->data(), vector->size()));
400 CFURLRequestSetHTTPRequestBody(request, data.get());
401 return;
402 }
403 }
404
405 CFURLRequestSetHTTPRequestBodyStream(request, createHTTPBodyCFReadStream(*formData).get());
406}

408FormData* httpBodyFromStream(CFReadStreamRef stream)
409{
410 if (!stream)
411 return nullptr;
412
413 // Passing the pointer as property appears to be the only way to associate a stream with FormData.
414 // A new stream is always created in CFURLRequestCopyHTTPRequestBodyStream (or -[NSURLRequest HTTPBodyStream]),
415 // so a side HashMap wouldn't work.
416 // Even the stream's context pointer is different from the one we returned from formCreate().
417
418 RetainPtr<CFNumberRef> formDataPointerAsCFNumber = adoptCF(static_cast<CFNumberRef>(CFReadStreamCopyProperty(stream, formDataPointerPropertyName)));
419 if (!formDataPointerAsCFNumber)
420 return nullptr;
421
422 long formDataPointerAsNumber;
423 if (!CFNumberGetValue(formDataPointerAsCFNumber.get(), kCFNumberLongType, &formDataPointerAsNumber))
424 return nullptr;
425
426 return reinterpret_cast<FormData*>(static_cast<intptr_t>(formDataPointerAsNumber));
427}
