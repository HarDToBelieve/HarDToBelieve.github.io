27
28#include "ResourceErrorBase.h"
29
30#include <wtf/RetainPtr.h>
31#if USE(CFURLCONNECTION)
32#include <CoreFoundation/CFStream.h>
33#endif
34#if PLATFORM(WIN)
35#include <windows.h>
36#include <wincrypt.h> // windows.h must be included before wincrypt.h.
37#endif

39#if PLATFORM(COCOA)
40OBJC_CLASS NSError;
41#endif
42
43namespace WebCore {
44
45class ResourceError : public ResourceErrorBase {
46public:
47 ResourceError(Type type = Type::Null)
48 : ResourceErrorBase(type)
49 , m_dataIsUpToDate(true)
50 {
51 }
52
53 ResourceError(const String& domain, int errorCode, const URL& failingURL, const String& localizedDescription, Type type = Type::General)
54 : ResourceErrorBase(domain, errorCode, failingURL, localizedDescription, type)
55 , m_dataIsUpToDate(true)
56 {
57#if PLATFORM(COCOA)
58 ASSERT(domain != getNSURLErrorDomain());
59 ASSERT(domain != getCFErrorDomainCFNetwork());
60#endif
61 }
62
63 WEBCORE_EXPORT ResourceError(CFErrorRef error);
64
65 WEBCORE_EXPORT CFErrorRef cfError() const;
66 WEBCORE_EXPORT operator CFErrorRef() const;
67
68#if USE(CFURLCONNECTION)
69 ResourceError(const String& domain, int errorCode, const URL& failingURL, const String& localizedDescription, CFDataRef certificate);
70 PCCERT_CONTEXT certificate() const;
71 void setCertificate(CFDataRef);
72 ResourceError(CFStreamError error);
73 CFStreamError cfStreamError() const;
74 operator CFStreamError() const;
75 static const void* getSSLPeerCertificateDataBytePtr(CFDictionaryRef);
76
77#endif
78
79#if PLATFORM(COCOA)
80 WEBCORE_EXPORT ResourceError(NSError *);
81 WEBCORE_EXPORT NSError *nsError() const;
82 WEBCORE_EXPORT operator NSError *() const;
83#endif
84
85 static bool platformCompare(const ResourceError& a, const ResourceError& b);
86
87private:
88 friend class ResourceErrorBase;
89
90#if PLATFORM(COCOA)
91 WEBCORE_EXPORT const String& getNSURLErrorDomain() const;
92 WEBCORE_EXPORT const String& getCFErrorDomainCFNetwork() const;
93 WEBCORE_EXPORT void mapPlatformError();
94#endif
95 void platformLazyInit();
96
97 void doPlatformIsolatedCopy(const ResourceError&);
98
99 bool m_dataIsUpToDate;
100#if USE(CFURLCONNECTION)
101 mutable RetainPtr<CFErrorRef> m_platformError;
102#if PLATFORM(WIN)
103 RetainPtr<CFDataRef> m_certificate;
104#endif
105#else
106 mutable RetainPtr<NSError> m_platformError;
107#endif
108};
109
110} // namespace WebCore
