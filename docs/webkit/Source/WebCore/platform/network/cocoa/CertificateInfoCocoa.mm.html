27#import "CertificateInfo.h"

29namespace WebCore {
30
31#ifndef NDEBUG
32void CertificateInfo::dump() const
33{
34#if HAVE(SEC_TRUST_SERIALIZATION)
35 if (m_trust) {
36 CFIndex entries = SecTrustGetCertificateCount(trust());
37
38 NSLog(@"CertificateInfo SecTrust\n");
39 NSLog(@" Entries: %ld\n", entries);
40#if HAVE(SEC_TRUST_COPY_CERTIFICATE_CHAIN)
41 auto chain = adoptCF(SecTrustCopyCertificateChain(trust()));
42#endif
43 for (CFIndex i = 0; i < entries; ++i) {
44#if HAVE(SEC_TRUST_COPY_CERTIFICATE_CHAIN)
45 RetainPtr<CFStringRef> summary = adoptCF(SecCertificateCopySubjectSummary(checked_cf_cast<SecCertificateRef>(CFArrayGetValueAtIndex(chain.get(), i))));
46#else
47 RetainPtr<CFStringRef> summary = adoptCF(SecCertificateCopySubjectSummary(SecTrustGetCertificateAtIndex(trust(), i)));
48#endif
49 NSLog(@" %@", (__bridge NSString *)summary.get());
50 }
51
52 return;
53 }
54#endif
55 if (m_certificateChain) {
56 CFIndex entries = CFArrayGetCount(m_certificateChain.get());
57
58 NSLog(@"CertificateInfo (Certificate Chain)\n");
59 NSLog(@" Entries: %ld\n", entries);
60 for (CFIndex i = 0; i < entries; ++i) {
61 RetainPtr<CFStringRef> summary = adoptCF(SecCertificateCopySubjectSummary(checked_cf_cast<SecCertificateRef>(CFArrayGetValueAtIndex(m_certificateChain.get(), i))));
62 NSLog(@" %@", (__bridge NSString *)summary.get());
63 }
64
65 return;
66 }
67
68 NSLog(@"CertificateInfo (Empty)\n");
69}
70#endif
71
72} // namespace WebCore
