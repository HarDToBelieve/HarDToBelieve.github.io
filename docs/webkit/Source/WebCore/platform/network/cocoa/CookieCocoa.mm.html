44static Vector<uint16_t> portVectorFromList(NSArray<NSNumber *> *portList)
45{
46 Vector<uint16_t> ports;
47 ports.reserveInitialCapacity(portList.count);
48
49 for (NSNumber *port : portList)
50 ports.uncheckedAppend(port.unsignedShortValue);
51
52 return ports;
53}
54
55static NSString * _Nullable portStringFromVector(const Vector<uint16_t>& ports)
56{
57 if (ports.isEmpty())
58 return nil;
59
60 auto *string = [NSMutableString stringWithCapacity:ports.size() * 5];
61
62 for (size_t i = 0; i < ports.size() - 1; ++i)
63 [string appendFormat:@"%" PRIu16 ", ", ports[i]];
64
65 [string appendFormat:@"%" PRIu16, ports.last()];
66
67 return string;
68}

70static double cookieCreated(NSHTTPCookie *cookie)
71{
72 id value = cookie.properties[@"Created"];
73
74 auto toCanonicalFormat = [](double referenceFormat) {
75 return 1000.0 * (referenceFormat + NSTimeIntervalSince1970);
76 };
77
78 if ([value isKindOfClass:[NSNumber class]])
79 return toCanonicalFormat(((NSNumber *)value).doubleValue);
80
81 if ([value isKindOfClass:[NSString class]])
82 return toCanonicalFormat(((NSString *)value).doubleValue);
83
84 return 0;
85}

87static std::optional<double> cookieExpiry(NSHTTPCookie *cookie)
88{
89 NSDate *expiryDate = cookie.expiresDate;
90 if (!expiryDate)
91 return std::nullopt;
92 return [expiryDate timeIntervalSince1970] * 1000.0;
93}

95static Cookie::SameSitePolicy coreSameSitePolicy(NSHTTPCookieStringPolicy _Nullable policy)
96{
97 if (!policy)
98 return Cookie::SameSitePolicy::None;
99 ALLOW_NEW_API_WITHOUT_GUARDS_BEGIN
100 if ([policy isEqualToString:NSHTTPCookieSameSiteLax])
101 return Cookie::SameSitePolicy::Lax;
102 if ([policy isEqualToString:NSHTTPCookieSameSiteStrict])
103 return Cookie::SameSitePolicy::Strict;
104 ALLOW_NEW_API_WITHOUT_GUARDS_END
105 ASSERT_NOT_REACHED();
106 return Cookie::SameSitePolicy::None;
107}
108
109static NSHTTPCookieStringPolicy _Nullable nsSameSitePolicy(Cookie::SameSitePolicy policy)
110{
111 switch (policy) {
112 case Cookie::SameSitePolicy::None:
113 return nil;
114 ALLOW_NEW_API_WITHOUT_GUARDS_BEGIN
115 case Cookie::SameSitePolicy::Lax:
116 return NSHTTPCookieSameSiteLax;
117 case Cookie::SameSitePolicy::Strict:
118 return NSHTTPCookieSameSiteStrict;
119 ALLOW_NEW_API_WITHOUT_GUARDS_END
120 }
121}

123Cookie::Cookie(NSHTTPCookie *cookie)
124 : name { cookie.name }
125 , value { cookie.value }
126 , domain { cookie.domain }
127 , path { cookie.path }
128 , created { cookieCreated(cookie) }
129 , expires { cookieExpiry(cookie) }
130 , httpOnly { static_cast<bool>(cookie.HTTPOnly) }
131 , secure { static_cast<bool>(cookie.secure) }
132 , session { static_cast<bool>(cookie.sessionOnly) }
133 , comment { cookie.comment }
134 , commentURL { cookie.commentURL }
135 , ports { portVectorFromList(cookie.portList) }
136{
137 ALLOW_NEW_API_WITHOUT_GUARDS_BEGIN
138 if ([cookie respondsToSelector:@selector(sameSitePolicy)])
139 sameSite = coreSameSitePolicy(cookie.sameSitePolicy);
140 ALLOW_NEW_API_WITHOUT_GUARDS_END
141}

143Cookie::operator NSHTTPCookie * _Nullable () const
144{
145 if (isNull())
146 return nil;
147
148 NSMutableDictionary *properties = [NSMutableDictionary dictionaryWithCapacity:14];
149
150 if (!comment.isNull())
151 [properties setObject:(NSString *)comment forKey:NSHTTPCookieComment];
152
153 if (!commentURL.isNull())
154 [properties setObject:(NSURL *)commentURL forKey:NSHTTPCookieCommentURL];
155
156 if (!domain.isNull())
157 [properties setObject:(NSString *)domain forKey:NSHTTPCookieDomain];
158
159 if (!name.isNull())
160 [properties setObject:(NSString *)name forKey:NSHTTPCookieName];
161
162 if (!path.isNull())
163 [properties setObject:(NSString *)path forKey:NSHTTPCookiePath];
164
165 if (!value.isNull())
166 [properties setObject:(NSString *)value forKey:NSHTTPCookieValue];
167
168 if (expires) {
169 NSDate *expirationDate = [NSDate dateWithTimeIntervalSince1970:*expires / 1000.0];
170 [properties setObject:expirationDate forKey:NSHTTPCookieExpires];
171 }
172
173 [properties setObject:@(created / 1000.0 - NSTimeIntervalSince1970) forKey:@"Created"];
174
175 auto* portString = portStringFromVector(ports);
176 if (portString)
177 [properties setObject:portString forKey:NSHTTPCookiePort];
178
179 if (secure)
180 [properties setObject:@YES forKey:NSHTTPCookieSecure];
181
182 if (session)
183 [properties setObject:@YES forKey:NSHTTPCookieDiscard];
184
185 if (httpOnly)
186 [properties setObject:@YES forKey:@"HttpOnly"];
187
188 if (auto* sameSitePolicy = nsSameSitePolicy(sameSite))
189 [properties setObject:sameSitePolicy forKey:@"SameSite"];
190
191 [properties setObject:@"1" forKey:NSHTTPCookieVersion];
192
193 return [NSHTTPCookie cookieWithProperties:properties];
194}

196bool Cookie::operator==(const Cookie& other) const
197{
198 ASSERT(!name.isHashTableDeletedValue());
199 bool thisNull = isNull();
200 bool otherNull = other.isNull();
201 if (thisNull || otherNull)
202 return thisNull == otherNull;
203 return [static_cast<NSHTTPCookie *>(*this) isEqual:other];
204}
205
206unsigned Cookie::hash() const
207{
208 ASSERT(!name.isHashTableDeletedValue());
209 ASSERT(!isNull());
210 return static_cast<NSHTTPCookie *>(*this).hash;
211}
