31static NSURLCredentialPersistence toNSURLCredentialPersistence(CredentialPersistence persistence)
32{
33 switch (persistence) {
34 case CredentialPersistenceNone:
35 return NSURLCredentialPersistenceNone;
36 case CredentialPersistenceForSession:
37 return NSURLCredentialPersistenceForSession;
38 case CredentialPersistencePermanent:
39 return NSURLCredentialPersistencePermanent;
40 }
41
42 ASSERT_NOT_REACHED();
43 return NSURLCredentialPersistenceNone;
44}
45
46static CredentialPersistence toCredentialPersistence(NSURLCredentialPersistence persistence)
47{
48 switch (persistence) {
49 case NSURLCredentialPersistenceNone:
50 return CredentialPersistenceNone;
51 case NSURLCredentialPersistenceForSession:
52 return CredentialPersistenceForSession;
53 case NSURLCredentialPersistencePermanent:
54 case NSURLCredentialPersistenceSynchronizable:
55 return CredentialPersistencePermanent;
56 }
57
58 ASSERT_NOT_REACHED();
59 return CredentialPersistenceNone;
60}
61
62Credential::Credential(const Credential& original, CredentialPersistence persistence)
63 : CredentialBase(original, persistence)
64{
65 NSURLCredential *originalNSURLCredential = original.m_nsCredential.get();
66 if (!originalNSURLCredential)
67 return;
68
69 if (NSString *user = originalNSURLCredential.user)
70 m_nsCredential = adoptNS([[NSURLCredential alloc] initWithUser:user password:originalNSURLCredential.password persistence:toNSURLCredentialPersistence(persistence)]);
71 else if (SecIdentityRef identity = originalNSURLCredential.identity)
72 m_nsCredential = adoptNS([[NSURLCredential alloc] initWithIdentity:identity certificates:originalNSURLCredential.certificates persistence:toNSURLCredentialPersistence(persistence)]);
73 else {
74 // It is not possible to set the persistence of server trust credentials.
75 ASSERT_NOT_REACHED();
76 m_nsCredential = originalNSURLCredential;
77 }
78}
79
80Credential::Credential(NSURLCredential *credential)
81 : CredentialBase(credential.user, credential.password, toCredentialPersistence(credential.persistence))
82 , m_nsCredential(credential)
83{
84}
85
86NSURLCredential *Credential::nsCredential() const
87{
88 if (m_nsCredential)
89 return m_nsCredential.get();
90
91 if (CredentialBase::isEmpty())
92 return nil;
93
94 m_nsCredential = adoptNS([[NSURLCredential alloc] initWithUser:user() password:password() persistence:toNSURLCredentialPersistence(persistence())]);
95
96 return m_nsCredential.get();
97}
98
99bool Credential::isEmpty() const
100{
101 if (m_nsCredential)
102 return false;
103
104 return CredentialBase::isEmpty();
105}
106
107bool Credential::platformCompare(const Credential& a, const Credential& b)
108{
109 if (!a.m_nsCredential && !b.m_nsCredential)
110 return true;
111
112 return [a.nsCredential() isEqual:b.nsCredential()];
113}
114
115bool Credential::encodingRequiresPlatformData(NSURLCredential *credential)
116{
117 return !credential.user;
118}
