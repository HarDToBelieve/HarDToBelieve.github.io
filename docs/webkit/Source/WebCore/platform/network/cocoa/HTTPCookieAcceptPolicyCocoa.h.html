31namespace WebCore {
32
33inline HTTPCookieAcceptPolicy toHTTPCookieAcceptPolicy(NSHTTPCookieAcceptPolicy policy)
34{
35 switch (static_cast<NSUInteger>(policy)) {
36 case NSHTTPCookieAcceptPolicyAlways:
37 return HTTPCookieAcceptPolicy::AlwaysAccept;
38 case NSHTTPCookieAcceptPolicyNever:
39 return HTTPCookieAcceptPolicy::Never;
40 case NSHTTPCookieAcceptPolicyOnlyFromMainDocumentDomain:
41 return HTTPCookieAcceptPolicy::OnlyFromMainDocumentDomain;
42 case NSHTTPCookieAcceptPolicyExclusivelyFromMainDocumentDomain:
43 return HTTPCookieAcceptPolicy::ExclusivelyFromMainDocumentDomain;
44 }
45
46 ASSERT_NOT_REACHED();
47 return HTTPCookieAcceptPolicy::Never;
48}
49
50inline NSHTTPCookieAcceptPolicy toNSHTTPCookieAcceptPolicy(HTTPCookieAcceptPolicy policy)
51{
52 switch (policy) {
53 case HTTPCookieAcceptPolicy::AlwaysAccept:
54 return NSHTTPCookieAcceptPolicyAlways;
55 case HTTPCookieAcceptPolicy::Never:
56 return NSHTTPCookieAcceptPolicyNever;
57 case HTTPCookieAcceptPolicy::OnlyFromMainDocumentDomain:
58 return NSHTTPCookieAcceptPolicyOnlyFromMainDocumentDomain;
59 case HTTPCookieAcceptPolicy::ExclusivelyFromMainDocumentDomain:
60 return (NSHTTPCookieAcceptPolicy)NSHTTPCookieAcceptPolicyExclusivelyFromMainDocumentDomain;
61 }
62 ASSERT_NOT_REACHED();
63 return NSHTTPCookieAcceptPolicyAlways;
64}
