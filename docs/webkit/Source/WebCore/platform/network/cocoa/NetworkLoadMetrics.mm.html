34static MonotonicTime dateToMonotonicTime(NSDate *date)
35{
36 if (auto interval = date.timeIntervalSince1970)
37 return WallTime::fromRawSeconds(interval).approximateMonotonicTime();
38 return { };
39}

41static Box<NetworkLoadMetrics> packageTimingData(MonotonicTime redirectStart, NSDate *fetchStart, NSDate *domainLookupStart, NSDate *domainLookupEnd, NSDate *connectStart, NSDate *secureConnectionStart, NSDate *connectEnd, NSDate *requestStart, NSDate *responseStart, bool reusedTLSConnection, NSString *protocol, uint16_t redirectCount, bool failsTAOCheck, bool hasCrossOriginRedirect)
42{
43
44 auto timing = Box<NetworkLoadMetrics>::create();
45
46 timing->redirectStart = redirectStart;
47 timing->fetchStart = dateToMonotonicTime(fetchStart);
48 timing->domainLookupStart = dateToMonotonicTime(domainLookupStart);
49 timing->domainLookupEnd = dateToMonotonicTime(domainLookupEnd);
50 timing->connectStart = dateToMonotonicTime(connectStart);
51 if (reusedTLSConnection && [protocol isEqualToString:@"https"])
52 timing->secureConnectionStart = reusedTLSConnectionSentinel;
53 else
54 timing->secureConnectionStart = dateToMonotonicTime(secureConnectionStart);
55 timing->connectEnd = dateToMonotonicTime(connectEnd);
56 timing->requestStart = dateToMonotonicTime(requestStart);
57 timing->responseStart = dateToMonotonicTime(responseStart);
58 timing->redirectCount = redirectCount;
59 timing->failsTAOCheck = failsTAOCheck;
60 timing->hasCrossOriginRedirect = hasCrossOriginRedirect;
61
62 // NOTE: responseEnd is not populated in this code path.
63
64 return timing;
65}

67Box<NetworkLoadMetrics> copyTimingData(NSURLSessionTaskMetrics *incompleteMetrics, const NetworkLoadMetrics& metricsFromTask)
68{
69 NSArray<NSURLSessionTaskTransactionMetrics *> *transactionMetrics = incompleteMetrics.transactionMetrics;
70 NSURLSessionTaskTransactionMetrics *metrics = transactionMetrics.lastObject;
71 return packageTimingData(
72 dateToMonotonicTime(transactionMetrics.firstObject.fetchStartDate),
73 metrics.fetchStartDate,
74 metrics.domainLookupStartDate,
75 metrics.domainLookupEndDate,
76 metrics.connectStartDate,
77 metrics.secureConnectionStartDate,
78 metrics.connectEndDate,
79 metrics.requestStartDate,
80 metrics.responseStartDate,
81 metrics.reusedConnection,
82 metrics.response.URL.scheme,
83 incompleteMetrics.redirectCount,
84 metricsFromTask.failsTAOCheck,
85 metricsFromTask.hasCrossOriginRedirect
86 );
87}

89Box<NetworkLoadMetrics> copyTimingData(NSURLConnection *connection, const ResourceHandle& handle)
90{
91 NSDictionary *timingData = [connection _timingData];
92
93 auto timingValue = [&](NSString *key) -> RetainPtr<NSDate> {
94 if (NSNumber *number = [timingData objectForKey:key]) {
95 if (double doubleValue = number.doubleValue)
96 return adoptNS([[NSDate alloc] initWithTimeIntervalSinceReferenceDate:doubleValue]);
97 }
98 return { };
99 };
100
101 auto data = packageTimingData(
102 handle.startTimeBeforeRedirects(),
103 timingValue(@"_kCFNTimingDataFetchStart").get(),
104 timingValue(@"_kCFNTimingDataDomainLookupStart").get(),
105 timingValue(@"_kCFNTimingDataDomainLookupEnd").get(),
106 timingValue(@"_kCFNTimingDataConnectStart").get(),
107 timingValue(@"_kCFNTimingDataSecureConnectionStart").get(),
108 timingValue(@"_kCFNTimingDataConnectEnd").get(),
109 timingValue(@"_kCFNTimingDataRequestStart").get(),
110 timingValue(@"_kCFNTimingDataResponseStart").get(),
111 timingValue(@"_kCFNTimingDataConnectionReused").get(),
112 connection.currentRequest.URL.scheme,
113 handle.redirectCount(),
114 handle.failsTAOCheck(),
115 handle.hasCrossOriginRedirect()
116 );
117
118 if (!data->fetchStart)
119 data->fetchStart = data->redirectStart;
120
121 return data;
122}
