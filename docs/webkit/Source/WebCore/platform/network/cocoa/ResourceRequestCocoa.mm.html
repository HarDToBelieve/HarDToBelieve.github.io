56static inline ResourceRequestCachePolicy fromPlatformRequestCachePolicy(NSURLRequestCachePolicy policy)
57{
58 switch (policy) {
59 case NSURLRequestUseProtocolCachePolicy:
60 return ResourceRequestCachePolicy::UseProtocolCachePolicy;
61 case NSURLRequestReturnCacheDataElseLoad:
62 return ResourceRequestCachePolicy::ReturnCacheDataElseLoad;
63 case NSURLRequestReturnCacheDataDontLoad:
64 return ResourceRequestCachePolicy::ReturnCacheDataDontLoad;
65 default:
66 return ResourceRequestCachePolicy::ReloadIgnoringCacheData;
67 }
68}
69
70static inline NSURLRequestCachePolicy toPlatformRequestCachePolicy(ResourceRequestCachePolicy policy)
71{
72 switch (policy) {
73 case ResourceRequestCachePolicy::UseProtocolCachePolicy:
74 return NSURLRequestUseProtocolCachePolicy;
75 case ResourceRequestCachePolicy::ReturnCacheDataElseLoad:
76 return NSURLRequestReturnCacheDataElseLoad;
77 case ResourceRequestCachePolicy::ReturnCacheDataDontLoad:
78 return NSURLRequestReturnCacheDataDontLoad;
79 default:
80 return NSURLRequestReloadIgnoringLocalCacheData;
81 }
82}

84void ResourceRequest::doUpdateResourceRequest()
85{
86 m_url = [m_nsRequest.get() URL];
87
88 if (m_cachePolicy == ResourceRequestCachePolicy::UseProtocolCachePolicy)
89 m_cachePolicy = fromPlatformRequestCachePolicy([m_nsRequest.get() cachePolicy]);
90 m_timeoutInterval = [m_nsRequest.get() timeoutInterval];
91 m_firstPartyForCookies = [m_nsRequest.get() mainDocumentURL];
92
93 URL siteForCookies { [m_nsRequest.get() _propertyForKey:@"_kCFHTTPCookiePolicyPropertySiteForCookies"] };
94 m_sameSiteDisposition = siteForCookies.isNull() ? SameSiteDisposition::Unspecified : (areRegistrableDomainsEqual(siteForCookies, m_url) ? SameSiteDisposition::SameSite : SameSiteDisposition::CrossSite);
95
96 m_isTopSite = static_cast<NSNumber*>([m_nsRequest.get() _propertyForKey:@"_kCFHTTPCookiePolicyPropertyIsTopLevelNavigation"]).boolValue;
97
98 if (NSString* method = [m_nsRequest.get() HTTPMethod])
99 m_httpMethod = method;
100 m_allowCookies = [m_nsRequest.get() HTTPShouldHandleCookies];
101
102 if (resourcePrioritiesEnabled())
103 m_priority = toResourceLoadPriority(m_nsRequest ? CFURLRequestGetRequestPriority([m_nsRequest _CFURLRequest]) : 0);
104
105 m_httpHeaderFields.clear();
106 [[m_nsRequest allHTTPHeaderFields] enumerateKeysAndObjectsUsingBlock: ^(NSString *name, NSString *value, BOOL *) {
107 m_httpHeaderFields.set(name, value);
108 }];
109
110 m_responseContentDispositionEncodingFallbackArray.clear();
111 NSArray *encodingFallbacks = [m_nsRequest.get() contentDispositionEncodingFallbackArray];
112 m_responseContentDispositionEncodingFallbackArray.reserveCapacity([encodingFallbacks count]);
113 for (NSNumber *encodingFallback in [m_nsRequest contentDispositionEncodingFallbackArray]) {
114 CFStringEncoding encoding = CFStringConvertNSStringEncodingToEncoding([encodingFallback unsignedLongValue]);
115 if (encoding != kCFStringEncodingInvalidId)
116 m_responseContentDispositionEncodingFallbackArray.uncheckedAppend(CFStringConvertEncodingToIANACharSetName(encoding));
117 }
118
119 if (m_nsRequest) {
120 NSString* cachePartition = [NSURLProtocol propertyForKey:(NSString *)_kCFURLCachePartitionKey inRequest:m_nsRequest.get()];
121 if (cachePartition)
122 m_cachePartition = cachePartition;
123 }
124}

126void ResourceRequest::doUpdateResourceHTTPBody()
127{
128 if (NSData* bodyData = [m_nsRequest.get() HTTPBody])
129 m_httpBody = FormData::create([bodyData bytes], [bodyData length]);
130 else if (NSInputStream* bodyStream = [m_nsRequest.get() HTTPBodyStream]) {
131 FormData* formData = httpBodyFromStream(bodyStream);
132 // There is no FormData object if a client provided a custom data stream.
133 // We shouldn't be looking at http body after client callbacks.
134 ASSERT(formData);
135 if (formData)
136 m_httpBody = formData;
137 }
138}

140static NSURL *siteForCookies(ResourceRequest::SameSiteDisposition disposition, NSURL *url)
141{
142 switch (disposition) {
143 case ResourceRequest::SameSiteDisposition::Unspecified:
144 return { };
145 case ResourceRequest::SameSiteDisposition::SameSite:
146 return url;
147 case ResourceRequest::SameSiteDisposition::CrossSite:
148 static NSURL *emptyURL = [[NSURL alloc] initWithString:@""];
149 return emptyURL;
150 }
151}

161void ResourceRequest::doUpdatePlatformRequest()
162{
163 if (isNull()) {
164 m_nsRequest = nil;
165 return;
166 }
167
168 auto nsRequest = adoptNS([m_nsRequest.get() mutableCopy]);
169
170 if (nsRequest)
171 [nsRequest setURL:url()];
172 else
173 nsRequest = adoptNS([[NSMutableURLRequest alloc] initWithURL:url()]);
174
175 if (ResourceRequest::httpPipeliningEnabled())
176 CFURLRequestSetShouldPipelineHTTP([nsRequest _CFURLRequest], true, true);
177
178 if (ResourceRequest::resourcePrioritiesEnabled()) {
179 CFURLRequestSetRequestPriority([nsRequest _CFURLRequest], toPlatformRequestPriority(priority()));
180
181 // Used by PLT to ignore very low priority beacon and ping loads.
182 if (priority() == ResourceLoadPriority::VeryLow)
183 _CFURLRequestSetProtocolProperty([nsRequest _CFURLRequest], CFSTR("WKVeryLowLoadPriority"), kCFBooleanTrue);
184 }
185
186 [nsRequest setCachePolicy:toPlatformRequestCachePolicy(cachePolicy())];
187 _CFURLRequestSetProtocolProperty([nsRequest _CFURLRequest], kCFURLRequestAllowAllPOSTCaching, kCFBooleanTrue);
188
189 double timeoutInterval = ResourceRequestBase::timeoutInterval();
190 if (timeoutInterval)
191 [nsRequest setTimeoutInterval:timeoutInterval];
192 // Otherwise, respect NSURLRequest default timeout.
193
194 [nsRequest setMainDocumentURL:firstPartyForCookies()];
195 if (!httpMethod().isEmpty())
196 [nsRequest setHTTPMethod:httpMethod()];
197 [nsRequest setHTTPShouldHandleCookies:allowCookies()];
198
199 [nsRequest _setProperty:siteForCookies(m_sameSiteDisposition, [nsRequest URL]) forKey:@"_kCFHTTPCookiePolicyPropertySiteForCookies"];
200 [nsRequest _setProperty:m_isTopSite ? @YES : @NO forKey:@"_kCFHTTPCookiePolicyPropertyIsTopLevelNavigation"];
201
202 // Cannot just use setAllHTTPHeaderFields here, because it does not remove headers.
203 for (NSString *oldHeaderName in [nsRequest allHTTPHeaderFields])
204 [nsRequest setValue:nil forHTTPHeaderField:oldHeaderName];
205 for (const auto& header : httpHeaderFields()) {
206 auto encodedValue = httpHeaderValueUsingSuitableEncoding(header);
207 [nsRequest setValue:(__bridge NSString *)encodedValue.get() forHTTPHeaderField:header.key];
208 }
209
210 [nsRequest setContentDispositionEncodingFallbackArray:createNSArray(m_responseContentDispositionEncodingFallbackArray, [] (auto& name) -> NSNumber * {
211 auto encoding = CFStringConvertEncodingToNSStringEncoding(CFStringConvertIANACharSetNameToEncoding(name.createCFString().get()));
212 if (encoding == kCFStringEncodingInvalidId)
213 return nil;
214 return @(encoding);
215 }).get()];
216
217 String partition = cachePartition();
218 if (!partition.isNull() && !partition.isEmpty()) {
219 NSString *partitionValue = [NSString stringWithUTF8String:partition.utf8().data()];
220 [NSURLProtocol setProperty:partitionValue forKey:(NSString *)_kCFURLCachePartitionKey inRequest:nsRequest.get()];
221 }
222
223#if PLATFORM(MAC)
224 if (m_url.isLocalFile()) {
225 auto fsRepFile = FileSystem::fileSystemRepresentation(m_url.fileSystemPath());
226 if (!fsRepFile.isNull()) {
227 auto fileDevice = FileSystem::getFileDeviceId(fsRepFile);
228 if (fileDevice && fileDevice.value())
229 [nsRequest _setProperty:[NSNumber numberWithInteger:fileDevice.value()] forKey:@"NSURLRequestFileProtocolExpectedDevice"];
230 }
231 }
232#endif
233
234 m_nsRequest = WTFMove(nsRequest);
235}

237void ResourceRequest::doUpdatePlatformHTTPBody()
238{
239 if (isNull()) {
240 ASSERT(!m_nsRequest);
241 return;
242 }
243
244 auto nsRequest = adoptNS([m_nsRequest.get() mutableCopy]);
245
246 if (nsRequest)
247 [nsRequest setURL:url()];
248 else
249 nsRequest = adoptNS([[NSMutableURLRequest alloc] initWithURL:url()]);
250
251 FormData* formData = httpBody();
252 if (formData && !formData->isEmpty())
253 WebCore::setHTTPBody(nsRequest.get(), formData);
254
255 if (NSInputStream *bodyStream = [nsRequest HTTPBodyStream]) {
256 // For streams, provide a Content-Length to avoid using chunked encoding, and to get accurate total length in callbacks.
257 NSString *lengthString = [bodyStream propertyForKey:(__bridge NSString *)formDataStreamLengthPropertyName()];
258 if (lengthString) {
259 [nsRequest setValue:lengthString forHTTPHeaderField:@"Content-Length"];
260 // Since resource request is already marked updated, we need to keep it up to date too.
261 ASSERT(m_resourceRequestUpdated);
262 m_httpHeaderFields.set(HTTPHeaderName::ContentLength, lengthString);
263 }
264 }
265
266 m_nsRequest = WTFMove(nsRequest);
267}

269void ResourceRequest::setStorageSession(CFURLStorageSessionRef storageSession)

271 updatePlatformRequest();

275RetainPtr<NSURLRequest> copyRequestWithStorageSession(CFURLStorageSessionRef storageSession, NSURLRequest *request)
276{
277 if (!storageSession || !request)
278 return adoptNS([request copy]);
279
280 auto cfRequest = adoptCF(CFURLRequestCreateMutableCopy(kCFAllocatorDefault, [request _CFURLRequest]));
281 _CFURLRequestSetStorageSession(cfRequest.get(), storageSession);
282 return adoptNS([[NSURLRequest alloc] _initWithCFURLRequest:cfRequest.get()]);
283}

285NSCachedURLResponse *cachedResponseForRequest(CFURLStorageSessionRef storageSession, NSURLRequest *request)
286{
287 if (!storageSession)
288 return [[NSURLCache sharedURLCache] cachedResponseForRequest:request];
289
290 auto cache = adoptCF(_CFURLStorageSessionCopyCache(kCFAllocatorDefault, storageSession));
291 auto cachedResponse = adoptCF(CFURLCacheCopyResponseForRequest(cache.get(), [request _CFURLRequest]));
292 if (!cachedResponse)
293 return nil;
294
295 return adoptNS([[NSCachedURLResponse alloc] _initWithCFCachedURLResponse:cachedResponse.get()]).autorelease();
296}
