45void ResourceResponse::initNSURLResponse() const
46{
47 if (!m_httpStatusCode || !m_url.protocolIsInHTTPFamily()) {
48 // Work around a mistake in the NSURLResponse class - <rdar://problem/6875219>.
49 // The init function takes an NSInteger, even though the accessor returns a long long.
50 // For values that won't fit in an NSInteger, pass -1 instead.
51 NSInteger expectedContentLength;
52 if (m_expectedContentLength < 0 || m_expectedContentLength > std::numeric_limits<NSInteger>::max())
53 expectedContentLength = -1;
54 else
55 expectedContentLength = static_cast<NSInteger>(m_expectedContentLength);
56
57 NSString* encodingNSString = nsStringNilIfEmpty(m_textEncodingName);
58 m_nsResponse = adoptNS([[NSURLResponse alloc] initWithURL:m_url MIMEType:m_mimeType expectedContentLength:expectedContentLength textEncodingName:encodingNSString]);
59 return;
60 }
61
62 // FIXME: We lose the status text and the HTTP version here.
63 NSMutableDictionary* headerDictionary = [NSMutableDictionary dictionary];
64 for (auto& header : m_httpHeaderFields)
65 [headerDictionary setObject:(NSString *)header.value forKey:(NSString *)header.key];
66
67 m_nsResponse = adoptNS([[NSHTTPURLResponse alloc] initWithURL:m_url statusCode:m_httpStatusCode HTTPVersion:(NSString*)kCFHTTPVersion1_1 headerFields:headerDictionary]);
68
69 // Mime type sniffing doesn't work with a synthesized response.
70 [m_nsResponse.get() _setMIMEType:(NSString *)m_mimeType];
71}

78CertificateInfo ResourceResponse::platformCertificateInfo() const
79{
80 CFURLResponseRef cfResponse = [m_nsResponse _CFURLResponse];
81 if (!cfResponse)
82 return { };
83
84 CFDictionaryRef context = _CFURLResponseGetSSLCertificateContext(cfResponse);
85 if (!context)
86 return { };
87
88 auto trustValue = CFDictionaryGetValue(context, kCFStreamPropertySSLPeerTrust);
89 if (!trustValue)
90 return { };
91 auto trust = checked_cf_cast<SecTrustRef>(trustValue);
92
93 SecTrustResultType trustResultType;
94 OSStatus result = SecTrustGetTrustResult(trust, &trustResultType);
95 if (result != errSecSuccess)
96 return { };
97
98 if (trustResultType == kSecTrustResultInvalid) {
99 if (!SecTrustEvaluateWithError(trust, nullptr))
100 return { };
101 }
102
103#if HAVE(SEC_TRUST_SERIALIZATION)
104 return CertificateInfo(trust);
105#else
106 return CertificateInfo(CertificateInfo::certificateChainFromSecTrust(trust));
107#endif
108}

110NSURLResponse *ResourceResponse::nsURLResponse() const
111{
112 if (!m_nsResponse && !m_isNull)
113 initNSURLResponse();
114 return m_nsResponse.get();

117static void addToHTTPHeaderMap(const void* key, const void* value, void* context)
118{
119 HTTPHeaderMap* httpHeaderMap = (HTTPHeaderMap*)context;
120 httpHeaderMap->set((CFStringRef)key, (CFStringRef)value);
121}
122
123static inline AtomString stripLeadingAndTrailingDoubleQuote(const String& value)
124{
125 unsigned length = value.length();
126 if (length < 2 || value[0u] != '"' || value[length - 1] != '"')
127 return value;
128
129 return StringView(value).substring(1, length - 2).toAtomString();
130}
131
132static inline HTTPHeaderMap initializeHTTPHeaders(CFHTTPMessageRef messageRef)
133{
134 // Avoid calling [NSURLResponse allHeaderFields] to minimize copying (<rdar://problem/26778863>).
135 auto headers = adoptCF(CFHTTPMessageCopyAllHeaderFields(messageRef));
136
137 HTTPHeaderMap headersMap;
138 CFDictionaryApplyFunction(headers.get(), addToHTTPHeaderMap, &headersMap);
139 return headersMap;
140}
141
142static inline AtomString extractHTTPStatusText(CFHTTPMessageRef messageRef)
143{
144 if (auto httpStatusLine = adoptCF(CFHTTPMessageCopyResponseStatusLine(messageRef)))
145 return extractReasonPhraseFromHTTPStatusLine(httpStatusLine.get());
146
147 static MainThreadNeverDestroyed<const AtomString> defaultStatusText("OK", AtomString::ConstructFromLiteral);
148 return defaultStatusText;
149}

151void ResourceResponse::platformLazyInit(InitLevel initLevel)
152{
153 ASSERT(initLevel >= CommonFieldsOnly);
154
155 if (m_initLevel >= initLevel)
156 return;
157
158 if (m_isNull || !m_nsResponse)
159 return;
160
161 @autoreleasepool {
162
163 auto messageRef = [m_nsResponse.get() isKindOfClass:[NSHTTPURLResponse class]] ? CFURLResponseGetHTTPResponse([ (NSHTTPURLResponse *)m_nsResponse.get() _CFURLResponse]) : nullptr;
164
165 if (m_initLevel < CommonFieldsOnly) {
166 m_url = [m_nsResponse.get() URL];
167 m_mimeType = [m_nsResponse.get() MIMEType];
168 m_expectedContentLength = [m_nsResponse.get() expectedContentLength];
169 // Stripping double quotes as a workaround for <rdar://problem/8757088>, can be removed once that is fixed.
170 m_textEncodingName = stripLeadingAndTrailingDoubleQuote([m_nsResponse.get() textEncodingName]);
171 m_httpStatusCode = messageRef ? CFHTTPMessageGetResponseStatusCode(messageRef) : 0;
172 if (messageRef)
173 m_httpHeaderFields = initializeHTTPHeaders(messageRef);
174 }
175 if (messageRef && initLevel == AllFields) {
176 m_httpStatusText = extractHTTPStatusText(messageRef);
177 m_httpVersion = String(adoptCF(CFHTTPMessageCopyVersion(messageRef)).get()).convertToASCIIUppercase();
178 }
179 }
180
181 m_initLevel = initLevel;
182}
