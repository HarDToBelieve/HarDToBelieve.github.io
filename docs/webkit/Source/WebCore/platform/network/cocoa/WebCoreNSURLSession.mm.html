46static NSDate * __nullable networkLoadMetricsDate(MonotonicTime time)
47{
48 if (!time)
49 return nil;
50 NSTimeInterval value = time.approximateWallTime().secondsSinceEpoch().seconds();
51 if (value <= 0)
52 return nil;
53 return [NSDate dateWithTimeIntervalSince1970:value];
54}

295- (void)taskCompleted:(WebCoreNSURLSessionDataTask *)task
296{
297 task.session = nil;
298
299 {
300 Locker<Lock> locker(_dataTasksLock);
301
302 ASSERT(_dataTasks.contains(task));
303 _dataTasks.remove(task);
304 if (!_dataTasks.isEmpty() || !_invalidated)
305 return;
306 }
307
308 RetainPtr<WebCoreNSURLSession> strongSelf { self };
309 [self addDelegateOperation:[strongSelf] {
310 if ([strongSelf.get().delegate respondsToSelector:@selector(URLSession:didBecomeInvalidWithError:)])
311 [strongSelf.get().delegate URLSession:(NSURLSession *)strongSelf.get() didBecomeInvalidWithError:nil];
312 }];
313}
314
315- (void)addDelegateOperation:(Function<void()>&&)function
316{
317 RetainPtr<WebCoreNSURLSession> strongSelf { self };
318 RetainPtr<NSBlockOperation> operation = [NSBlockOperation blockOperationWithBlock:makeBlockPtr(WTFMove(function)).get()];
319 dispatch_async(_internalQueue.get(), [strongSelf, operation] {
320 [strongSelf.get().delegateQueue addOperation:operation.get()];
321 [operation waitUntilFinished];
322 });
323}

400- (void)finishTasksAndInvalidate
401{
402 _invalidated = YES;
403 {
404 Locker<Lock> locker(_dataTasksLock);
405 if (!_dataTasks.isEmpty())
406 return;
407 }
408
409 [self addDelegateOperation:[strongSelf = retainPtr(self)] {
410 auto delegate = strongSelf.get().delegate;
411 if ([delegate respondsToSelector:@selector(URLSession:didBecomeInvalidWithError:)])
412 [delegate URLSession:(NSURLSession *)strongSelf.get() didBecomeInvalidWithError:nil];
413 }];
414}

416- (void)invalidateAndCancel
417{
418 Vector<RetainPtr<WebCoreNSURLSessionDataTask>> tasksCopy;
419 {
420 Locker<Lock> locker(_dataTasksLock);
421 tasksCopy = copyToVector(_dataTasks);
422 }
423
424 for (auto& task : tasksCopy)
425 [task cancel];
426
427 [self finishTasksAndInvalidate];
428}

492- (void)sendH2Ping:(NSURL *)url pongHandler:(void (^)(NSError *error, NSTimeInterval interval))pongHandler
493{
494 callOnMainThread([self, strongSelf = retainPtr(self), url = retainPtr(url), pongHandler = makeBlockPtr(pongHandler)] () mutable {
495
496 if (_invalidated)
497 return pongHandler(adoptNS([[NSError alloc] initWithDomain:NSURLErrorDomain code:NSURLErrorUnknown userInfo:nil]).get(), 0);
498
499 self.loader.sendH2Ping(url.get(), [self, strongSelf = WTFMove(strongSelf), pongHandler = WTFMove(pongHandler)] (Expected<Seconds, ResourceError>&& result) mutable {
500 NSTimeInterval interval = 0;
501 RetainPtr<NSError> error;
502 if (result)
503 interval = result.value().value();
504 else
505 error = result.error();
506 [self addDelegateOperation:[pongHandler = WTFMove(pongHandler), error = WTFMove(error), interval] {
507 pongHandler(error.get(), interval);
508 }];
509 });
510 });
511}

688- (id)initWithSession:(WebCoreNSURLSession *)session identifier:(NSUInteger)identifier request:(NSURLRequest *)request
689{
690 self.taskIdentifier = identifier;
691 self.session = session;
692 self.state = NSURLSessionTaskStateSuspended;
693 self.priority = NSURLSessionTaskPriorityDefault;
694
695 // CoreMedia will explicitly add a user agent header. Remove if present.
696 RetainPtr<NSMutableURLRequest> mutableRequest;
697 if (auto* userAgentValue = [request valueForHTTPHeaderField:@"User-Agent"]) {
698 mutableRequest = adoptNS([request mutableCopyWithZone:nil]);
699 [mutableRequest setValue:nil forHTTPHeaderField:@"User-Agent"];
700 request = mutableRequest.get();
701 }
702
703 self.originalRequest = self.currentRequest = request;
704
705 return self;
706}

714#pragma mark - Internal methods
715
716- (void)_restart
717{
718 ASSERT(isMainThread());
719
720 if (!self.session)
721 return;
722
723 [self _cancel];
724
725 if ([self.session rangeResponseGenerator].willHandleRequest(self, self.originalRequest))
726 return;
727
728 _resource = self.session.loader.requestResource(self.originalRequest, PlatformMediaResourceLoader::LoadOption::DisallowCaching);
729 if (_resource)
730 _resource->setClient(adoptRef(*new WebCoreNSURLSessionDataTaskClient(self)));
731}
732
733- (void)_cancel
734{
735 ASSERT(isMainThread());
736 if (_resource) {
737 _resource->stop();
738 _resource->setClient(nullptr);
739 _resource = nil;
740 }
741 if (auto *session = self.session)
742 [session rangeResponseGenerator].removeTask(self);
743}

841{
842 ASSERT(response.source() == ResourceResponse::Source::Network || response.source() == ResourceResponse::Source::DiskCache || response.source() == ResourceResponse::Source::DiskCacheAfterValidation || response.source() == ResourceResponse::Source::ServiceWorker);
843 ASSERT_UNUSED(resource, !resource || resource == _resource);
844 ASSERT(isMainThread());
845 [self.session task:self didReceiveResponseFromOrigin:SecurityOrigin::create(response.url())];
846 // FIXME: Think about this and make sure it's safe.
847 [self.session task:self didReceiveCORSAccessCheckResult:resource ? resource->didPassAccessControlCheck() : YES];
848 self.countOfBytesExpectedToReceive = response.expectedContentLength();
849 RetainPtr<NSURLResponse> strongResponse = response.nsURLResponse();
850
851 if (resource && self.session && [self.session rangeResponseGenerator].willSynthesizeRangeResponses(self, *resource, response)) {
852 _resource = nullptr;
853 return completionHandler(ShouldContinuePolicyCheck::Yes);
854 }
855
856 RetainPtr<WebCoreNSURLSessionDataTask> strongSelf { self };
857 if (!self.session)
858 return completionHandler(ShouldContinuePolicyCheck::No);
859 [self.session addDelegateOperation:[strongSelf, strongResponse, completionHandler = WTFMove(completionHandler)] () mutable {
860 strongSelf->_response = strongResponse.get();
861
862 id<NSURLSessionDataDelegate> dataDelegate = (id<NSURLSessionDataDelegate>)strongSelf.get().session.delegate;
863 if (![dataDelegate respondsToSelector:@selector(URLSession:dataTask:didReceiveResponse:completionHandler:)]) {
864 callOnMainThread([strongSelf, completionHandler = WTFMove(completionHandler)] () mutable {
865 completionHandler(ShouldContinuePolicyCheck::Yes);
866 });
867 return;
868 }
869
870 [dataDelegate URLSession:(NSURLSession *)strongSelf.get().session dataTask:(NSURLSessionDataTask *)strongSelf.get() didReceiveResponse:strongResponse.get() completionHandler:makeBlockPtr([strongSelf, completionHandler = WTFMove(completionHandler)] (NSURLSessionResponseDisposition disposition) mutable {
871 callOnMainThread([strongSelf, disposition, completionHandler = WTFMove(completionHandler)] () mutable {
872 if (disposition == NSURLSessionResponseCancel)
873 completionHandler(ShouldContinuePolicyCheck::No);
874 else {
875 ASSERT(disposition == NSURLSessionResponseAllow);
876 completionHandler(ShouldContinuePolicyCheck::Yes);
877 }

935{
936 ASSERT_UNUSED(resource, !resource || resource == _resource);
937 if (self.state == NSURLSessionTaskStateCompleted)
938 return;
939 self.state = NSURLSessionTaskStateCompleted;
940
941 RetainPtr<WebCoreNSURLSessionDataTask> strongSelf { self };
942 RetainPtr<WebCoreNSURLSession> strongSession { self.session };
943 RetainPtr<NSError> strongError { error };
944 [self.session addDelegateOperation:[strongSelf, strongSession, strongError, metrics = metrics.isolatedCopy()] {
945 id<NSURLSessionTaskDelegate> delegate = (id<NSURLSessionTaskDelegate>)strongSession.get().delegate;
946
947 if ([delegate respondsToSelector:@selector(URLSession:task:didFinishCollectingMetrics:)])
948 [delegate URLSession:(NSURLSession *)strongSession.get() task:(NSURLSessionDataTask *)strongSelf.get() didFinishCollectingMetrics:(NSURLSessionTaskMetrics *)adoptNS([[WebCoreNSURLSessionTaskMetrics alloc] _initWithMetrics:metrics]).get()];
949
950 if ([delegate respondsToSelector:@selector(URLSession:task:didCompleteWithError:)])
951 [delegate URLSession:(NSURLSession *)strongSession.get() task:(NSURLSessionDataTask *)strongSelf.get() didCompleteWithError:strongError.get()];
952
953 callOnMainThread([strongSelf, strongSession] {
954 [strongSession taskCompleted:strongSelf.get()];
955 });
