41String MIMETypeFromUTI(const String& uti)
42{
43ALLOW_DEPRECATED_DECLARATIONS_BEGIN

48String MIMETypeFromUTITree(const String& uti)
49{
50 auto utiCF = uti.createCFString();
51
52ALLOW_DEPRECATED_DECLARATIONS_BEGIN
53 // Check if this UTI has a MIME type.
54 RetainPtr<CFStringRef> mimeType = adoptCF(UTTypeCopyPreferredTagWithClass(utiCF.get(), kUTTagClassMIMEType));
55 if (mimeType)
56 return mimeType.get();
57
58 // If not, walk the ancestory of this UTI via its "ConformsTo" tags and return the first MIME type we find.
59 RetainPtr<CFDictionaryRef> decl = adoptCF(UTTypeCopyDeclaration(utiCF.get()));
60 if (!decl)
61 return emptyString();
62 CFTypeRef value = CFDictionaryGetValue(decl.get(), kUTTypeConformsToKey);
63ALLOW_DEPRECATED_DECLARATIONS_END
64 if (!value)
65 return emptyString();
66 CFTypeID typeID = CFGetTypeID(value);
67
68 if (typeID == CFStringGetTypeID())
69 return MIMETypeFromUTITree((CFStringRef)value);
70
71 if (typeID == CFArrayGetTypeID()) {
72 CFArrayRef newTypes = (CFArrayRef)value;
73 CFIndex count = CFArrayGetCount(newTypes);
74 for (CFIndex i = 0; i < count; ++i) {
75 CFTypeRef object = CFArrayGetValueAtIndex(newTypes, i);
76 if (CFGetTypeID(object) != CFStringGetTypeID())
77 continue;
78
79 String mimeType = MIMETypeFromUTITree((CFStringRef)object);
80 if (!mimeType.isEmpty())
81 return mimeType;
82 }
83 }
84
85 return emptyString();
86}

88static CFStringRef UTIFromUnknownMIMEType(StringView mimeType)
89{
90 static constexpr std::pair<ComparableLettersLiteral, NSString *> typesArray[] = {
91 { "model/usd", @"com.pixar.universal-scene-description-mobile" },
92 { "model/vnd.pixar.usd", @"com.pixar.universal-scene-description-mobile" },
93 { "model/vnd.reality", @"com.apple.reality" },
94 { "model/vnd.usdz+zip", @"com.pixar.universal-scene-description-mobile" },
95 };
96 static constexpr SortedArrayMap typesMap { typesArray };
97 return (__bridge CFStringRef)typesMap.get(mimeType, @"");
98}

100struct UTIFromMIMETypeCachePolicy : TinyLRUCachePolicy<String, RetainPtr<CFStringRef>> {
101public:
102 static RetainPtr<CFStringRef> createValueForKey(const String& mimeType)
103 {
104ALLOW_DEPRECATED_DECLARATIONS_BEGIN
105 auto type = adoptCF(UTTypeCreatePreferredIdentifierForTag(kUTTagClassMIMEType, mimeType.createCFString().get(), 0));
106ALLOW_DEPRECATED_DECLARATIONS_END
107 if (type)
108 return type;
109 return UTIFromUnknownMIMEType(mimeType);
110 }
111
112 static String createKeyForStorage(const String& key) { return key.isolatedCopy(); }
113};

116static TinyLRUCache<String, RetainPtr<CFStringRef>, 16, UTIFromMIMETypeCachePolicy>& cacheUTIFromMIMEType() WTF_REQUIRES_LOCK(cacheUTIFromMIMETypeLock)
117{
118 static NeverDestroyed<TinyLRUCache<String, RetainPtr<CFStringRef>, 16, UTIFromMIMETypeCachePolicy>> cache;
119 return cache;
120}

122String UTIFromMIMEType(const String& mimeType)
123{
124 Locker locker { cacheUTIFromMIMETypeLock };
125 return cacheUTIFromMIMEType().get(mimeType).get();
126}

136{
