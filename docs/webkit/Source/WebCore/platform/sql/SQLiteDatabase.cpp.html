49static void unauthorizedSQLFunction(sqlite3_context *context, int, sqlite3_value **)
50{
51 const char* functionName = (const char*)sqlite3_user_data(context);
52 sqlite3_result_error(context, makeString("Function ", functionName, " is unauthorized").utf8().data(), -1);
53}

55static void initializeSQLiteIfNecessary()
56{
57 static std::once_flag flag;
58 std::call_once(flag, [] {
59 // It should be safe to call this outside of std::call_once, since it is documented to be
60 // completely threadsafe. But in the past it was not safe, and the SQLite developers still
61 // aren't confident that it really is, and we still support ancient versions of SQLite. So
62 // std::call_once is used to stay on the safe side. See bug #143245.
63 int ret = sqlite3_initialize();
64 if (ret != SQLITE_OK) {
65#if SQLITE_VERSION_NUMBER >= 3007015
66 WTFLogAlways("Failed to initialize SQLite: %s", sqlite3_errstr(ret));
67#else
68 WTFLogAlways("Failed to initialize SQLite");
69#endif
70 CRASH();
71 }
72 });
73}

91bool SQLiteDatabase::open(const String& filename, OpenMode openMode)
92{
93 initializeSQLiteIfNecessary();
94
95 close();
96
97 {
98 Locker locker { isDatabaseOpeningForbiddenLock };
99 if (isDatabaseOpeningForbidden) {
100 m_openErrorMessage = "opening database is forbidden";
101 return false;
102 }
103
104 int flags = SQLITE_OPEN_AUTOPROXY;
105 switch (openMode) {
106 case OpenMode::ReadOnly:
107 flags |= SQLITE_OPEN_READONLY;
108 break;
109 case OpenMode::ReadWrite:
110 flags |= SQLITE_OPEN_READWRITE;
111 break;
112 case OpenMode::ReadWriteCreate:
113 flags |= SQLITE_OPEN_READWRITE | SQLITE_OPEN_CREATE;
114 break;
115 }
116
117 m_openError = sqlite3_open_v2(FileSystem::fileSystemRepresentation(filename).data(), &m_db, flags, nullptr);
118 if (m_openError != SQLITE_OK) {
119 m_openErrorMessage = m_db ? sqlite3_errmsg(m_db) : "sqlite_open returned null";
120 LOG_ERROR("SQLite database failed to load from %s\nCause - %s", filename.ascii().data(),
121 m_openErrorMessage.data());
122 close(ShouldSetErrorState::No);
123 return false;
124 }
125 }
126
127 overrideUnauthorizedFunctions();
128
129 m_openError = sqlite3_extended_result_codes(m_db, 1);
130 if (m_openError != SQLITE_OK) {
131 m_openErrorMessage = sqlite3_errmsg(m_db);
132 LOG_ERROR("SQLite database error when enabling extended errors - %s", m_openErrorMessage.data());
133 close(ShouldSetErrorState::No);
134 return false;
135 }
136
137 if (isOpen())
138 m_openingThread = &Thread::current();
139 else
140 m_openErrorMessage = "sqlite_open returned null";
141
142 {
143 SQLiteTransactionInProgressAutoCounter transactionCounter;
144 if (!executeCommand("PRAGMA temp_store = MEMORY;"_s))
145 LOG_ERROR("SQLite database could not set temp_store to memory");
146 }
147
148 if (openMode != OpenMode::ReadOnly)
149 useWALJournalMode();
150
151 auto shmFileName = makeString(filename, "-shm"_s);
152 if (FileSystem::fileExists(shmFileName)) {
153 if (!FileSystem::isSafeToUseMemoryMapForPath(shmFileName)) {
154 RELEASE_LOG_FAULT(SQLDatabase, "Opened an SQLite database with a Class A -shm file. This may trigger a crash when the user locks the device. (%s)", shmFileName.latin1().data());
155 FileSystem::makeSafeToUseMemoryMapForPath(shmFileName);
156 }
157 }
158
159 return isOpen();
160}

162static int walAutomaticTruncationHook(void* context, sqlite3* db, const char* dbName, int walPageCount)
163{
164 UNUSED_PARAM(context);
165
166 static constexpr int checkpointThreshold = 1000; // matches SQLITE_DEFAULT_WAL_AUTOCHECKPOINT
167
168 if (walPageCount >= checkpointThreshold) {
169 int newWalPageCount = 0;
170 int result = sqlite3_wal_checkpoint_v2(db, dbName, SQLITE_CHECKPOINT_TRUNCATE, &newWalPageCount, nullptr);
171
172#if LOG_DISABLED
173 UNUSED_VARIABLE(result);
174#else
175 if (result != SQLITE_OK || newWalPageCount) {
176 LOG(SQLDatabase, "Can't fully checkpoint SQLite db %p", db);
177
178 sqlite3_stmt* stmt = nullptr;
179 while ((stmt = sqlite3_next_stmt(db, stmt))) {
180 if (sqlite3_stmt_busy(stmt))
181 LOG(SQLDatabase, "SQLite db %p has busy stmt %p blocking checkpoint: %s", db, stmt, sqlite3_sql(stmt));
182 }
183 }
184#endif
185 }
186
187 return SQLITE_OK;
188}

195void SQLiteDatabase::useWALJournalMode()
196{
197 m_useWAL = true;
198 {
199 auto walStatement = prepareStatement("PRAGMA journal_mode=WAL;"_s);
200 if (walStatement && walStatement->step() == SQLITE_ROW) {
201#ifndef NDEBUG
202 String mode = walStatement->columnText(0);
203 if (!equalLettersIgnoringASCIICase(mode, "wal"))
204 LOG_ERROR("journal_mode of database should be 'WAL', but is '%s'", mode.utf8().data());
205#endif
206 } else
207 LOG_ERROR("SQLite database failed to set journal_mode to WAL, error: %s", lastErrorMsg());
208 }
209
210 {
211 SQLiteTransactionInProgressAutoCounter transactionCounter;
212 auto checkpointStatement = prepareStatement("PRAGMA wal_checkpoint(TRUNCATE)"_s);
213 if (checkpointStatement && checkpointStatement->step() == SQLITE_ROW) {
214 if (checkpointStatement->columnInt(0))
215 LOG(SQLDatabase, "SQLite database checkpoint is blocked");
216 } else
217 LOG_ERROR("SQLite database failed to checkpoint: %s", lastErrorMsg());
218 }
219}

221void SQLiteDatabase::close(ShouldSetErrorState shouldSetErrorState)
222{
223 if (m_db) {
224 ASSERT_WITH_MESSAGE(!m_statementCount, "All SQLiteTransaction objects should be destroyed before closing the database");
225
226 // FIXME: This is being called on the main thread during JS GC. <rdar://problem/5739818>
227 // ASSERT(m_openingThread == &Thread::current());
228 sqlite3* db = m_db;
229 {
230 Locker locker { m_databaseClosingMutex };
231 m_db = 0;
232 }
233
234 int closeResult;
235 if (m_useWAL) {
236 // Close in the scope of counter as it may acquire lock of database.
237 SQLiteTransactionInProgressAutoCounter transactionCounter;
238 closeResult = sqlite3_close(db);
239 } else
240 closeResult = sqlite3_close(db);
241
242 if (closeResult != SQLITE_OK)
243 RELEASE_LOG_ERROR(SQLDatabase, "SQLiteDatabase::close: Failed to close database (%d) - %{public}s", closeResult, lastErrorMsg());
244 }
245
246 if (shouldSetErrorState == ShouldSetErrorState::Yes) {
247 m_openingThread = nullptr;
248 m_openError = SQLITE_ERROR;
249 m_openErrorMessage = CString();
250 }
251}

253void SQLiteDatabase::overrideUnauthorizedFunctions()
254{
255 static const std::pair<const char*, int> functionParameters[] = {
256 { "rtreenode", 2 },
257 { "rtreedepth", 1 },
258 { "eval", 1 },
259 { "eval", 2 },
260 { "printf", -1 },
261 { "fts3_tokenizer", 1 },
262 { "fts3_tokenizer", 2 },
263 };
264
265 for (auto& functionParameter : functionParameters)
266 sqlite3_create_function(m_db, functionParameter.first, functionParameter.second, SQLITE_UTF8, const_cast<char*>(functionParameter.first), unauthorizedSQLFunction, 0, 0);

277int64_t SQLiteDatabase::maximumSize()
278{
279 int64_t maxPageCount = 0;
280
281 {
282 Locker locker { m_authorizerLock };
283 enableAuthorizer(false);
284 auto statement = prepareStatement("PRAGMA max_page_count"_s);
285 maxPageCount = statement ? statement->columnInt64(0) : 0;
286 enableAuthorizer(true);
287 }
288
289 return maxPageCount * pageSize();
290}

292void SQLiteDatabase::setMaximumSize(int64_t size)
293{
294 if (size < 0)
295 size = 0;
296
297 int currentPageSize = pageSize();
298
299 ASSERT(currentPageSize || !m_db);
300 int64_t newMaxPageCount = currentPageSize ? size / currentPageSize : 0;
301
302 Locker locker { m_authorizerLock };
303 enableAuthorizer(false);
304
305 auto statement = prepareStatementSlow(makeString("PRAGMA max_page_count = ", newMaxPageCount));
306 if (!statement || statement->step() != SQLITE_ROW)
307 LOG_ERROR("Failed to set maximum size of database to %lli bytes", static_cast<long long>(size));
308
309 enableAuthorizer(true);
310
311}

324 enableAuthorizer(true);

326

423void SQLiteDatabase::clearAllTables()
424{
425 auto statement = prepareStatement("SELECT name FROM sqlite_master WHERE type='table' AND name NOT LIKE 'sqlite_%';"_s);
426 if (!statement) {
427 LOG(SQLDatabase, "Failed to prepare statement to retrieve list of tables from database");
428 return;
429 }
430 Vector<String> tables;
431 while (statement->step() == SQLITE_ROW)
432 tables.append(statement->columnText(0));
433 for (auto& table : tables) {
434 if (!executeCommandSlow("DROP TABLE " + table))
435 LOG(SQLDatabase, "Unable to drop table %s", table.ascii().data());
436 }
437}

500int SQLiteDatabase::authorizerFunction(void* userData, int actionCode, const char* parameter1, const char* parameter2, const char* /*databaseName*/, const char* /*trigger_or_view*/)
501{
502 DatabaseAuthorizer* auth = static_cast<DatabaseAuthorizer*>(userData);
503 ASSERT(auth);
504
505 switch (actionCode) {
506 case SQLITE_CREATE_INDEX:
507 return auth->createIndex(parameter1, parameter2);
508 case SQLITE_CREATE_TABLE:
509 return auth->createTable(parameter1);
510 case SQLITE_CREATE_TEMP_INDEX:
511 return auth->createTempIndex(parameter1, parameter2);
512 case SQLITE_CREATE_TEMP_TABLE:
513 return auth->createTempTable(parameter1);
514 case SQLITE_CREATE_TEMP_TRIGGER:
515 return auth->createTempTrigger(parameter1, parameter2);
516 case SQLITE_CREATE_TEMP_VIEW:
517 return auth->createTempView(parameter1);
518 case SQLITE_CREATE_TRIGGER:
519 return auth->createTrigger(parameter1, parameter2);
520 case SQLITE_CREATE_VIEW:
521 return auth->createView(parameter1);
522 case SQLITE_DELETE:
523 return auth->allowDelete(parameter1);
524 case SQLITE_DROP_INDEX:
525 return auth->dropIndex(parameter1, parameter2);
526 case SQLITE_DROP_TABLE:
527 return auth->dropTable(parameter1);
528 case SQLITE_DROP_TEMP_INDEX:
529 return auth->dropTempIndex(parameter1, parameter2);
530 case SQLITE_DROP_TEMP_TABLE:
531 return auth->dropTempTable(parameter1);
532 case SQLITE_DROP_TEMP_TRIGGER:
533 return auth->dropTempTrigger(parameter1, parameter2);
534 case SQLITE_DROP_TEMP_VIEW:
535 return auth->dropTempView(parameter1);
536 case SQLITE_DROP_TRIGGER:
537 return auth->dropTrigger(parameter1, parameter2);
538 case SQLITE_DROP_VIEW:
539 return auth->dropView(parameter1);
540 case SQLITE_INSERT:
541 return auth->allowInsert(parameter1);
542 case SQLITE_PRAGMA:
543 return auth->allowPragma(parameter1, parameter2);
544 case SQLITE_READ:
545 return auth->allowRead(parameter1, parameter2);
546 case SQLITE_SELECT:
547 return auth->allowSelect();
548 case SQLITE_TRANSACTION:
549 return auth->allowTransaction();
550 case SQLITE_UPDATE:
551 return auth->allowUpdate(parameter1, parameter2);
552 case SQLITE_ATTACH:
553 return auth->allowAttach(parameter1);
554 case SQLITE_DETACH:
555 return auth->allowDetach(parameter1);
556 case SQLITE_ALTER_TABLE:
557 return auth->allowAlterTable(parameter1, parameter2);
558 case SQLITE_REINDEX:
559 return auth->allowReindex(parameter1);
560 case SQLITE_ANALYZE:
561 return auth->allowAnalyze(parameter1);
562 case SQLITE_CREATE_VTABLE:
563 return auth->createVTable(parameter1, parameter2);
564 case SQLITE_DROP_VTABLE:
565 return auth->dropVTable(parameter1, parameter2);
566 case SQLITE_FUNCTION:
567 return auth->allowFunction(parameter2);
568 default:
569 ASSERT_NOT_REACHED();
570 return SQLAuthDeny;
571 }
572}

602bool SQLiteDatabase::turnOnIncrementalAutoVacuum()
603{
604 int autoVacuumMode = 0;
605 if (auto statement = prepareStatement("PRAGMA auto_vacuum"_s))
606 autoVacuumMode = statement->columnInt(0);
607 int error = lastError();
608
609 // Check if we got an error while trying to get the value of the auto_vacuum flag.
610 // If we got a SQLITE_BUSY error, then there's probably another transaction in
611 // progress on this database. In this case, keep the current value of the
612 // auto_vacuum flag and try to set it to INCREMENTAL the next time we open this
613 // database. If the error is not SQLITE_BUSY, then we probably ran into a more
614 // serious problem and should return false (to log an error message).
615 if (error != SQLITE_ROW)
616 return false;
617
618 switch (autoVacuumMode) {
619 case AutoVacuumIncremental:
620 return true;
621 case AutoVacuumFull:
622 return executeCommand("PRAGMA auto_vacuum = 2"_s);
623 case AutoVacuumNone:
624 default:
625 if (!executeCommand("PRAGMA auto_vacuum = 2"_s))
626 return false;
627 runVacuumCommand();
628 error = lastError();
629 return (error == SQLITE_OK);
630 }
631}

664static Expected<sqlite3_stmt*, int> constructAndPrepareStatement(SQLiteDatabase& database, const char* query, size_t queryLength)
665{
666 Locker databaseLock { database.databaseMutex() };
667 LOG(SQLDatabase, "SQL - prepare - %s", query);
668
669 // Pass the length of the string including the null character to sqlite3_prepare_v2;
670 // this lets SQLite avoid an extra string copy.
671 size_t lengthIncludingNullCharacter = queryLength + 1;
672
673 sqlite3_stmt* statement { nullptr };
674 const char* tail = nullptr;
675 int error = sqlite3_prepare_v2(database.sqlite3Handle(), query, lengthIncludingNullCharacter, &statement, &tail);
676 if (error != SQLITE_OK)
677 LOG(SQLDatabase, "sqlite3_prepare16 failed (%i)\n%s\n%s", error, query, sqlite3_errmsg(database.sqlite3Handle()));
678
679 if (tail && *tail)
680 error = SQLITE_ERROR;
681
682 if (error != SQLITE_OK) {
683 sqlite3_finalize(statement);
684 return makeUnexpected(error);
685 }
686
687 // When the query is an empty string, sqlite3_prepare_v2() sets the statement to nullptr and yet doesn't return an error.
688 if (!statement)
689 return makeUnexpected(SQLITE_ERROR);
690
691 return statement;
692}

694Expected<SQLiteStatement, int> SQLiteDatabase::prepareStatementSlow(const String& queryString)
695{
696 CString query = queryString.stripWhiteSpace().utf8();
697 auto sqlStatement = constructAndPrepareStatement(*this, query.data(), query.length());
698 if (!sqlStatement) {
699 RELEASE_LOG_ERROR(SQLDatabase, "SQLiteDatabase::prepareStatement: Failed to prepare statement %{public}s", query.data());
700 return makeUnexpected(sqlStatement.error());
701 }
702 return SQLiteStatement { *this, sqlStatement.value() };
703}

715Expected<UniqueRef<SQLiteStatement>, int> SQLiteDatabase::prepareHeapStatementSlow(const String& queryString)
716{
717 CString query = queryString.stripWhiteSpace().utf8();
718 auto sqlStatement = constructAndPrepareStatement(*this, query.data(), query.length());
