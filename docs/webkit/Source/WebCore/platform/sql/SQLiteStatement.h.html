26#pragma once
27
28#include "SQLValue.h"
29#include "SQLiteDatabase.h"
30#include <wtf/Span.h>
31
32struct sqlite3_stmt;
33
34namespace WebCore {
35
36class SQLiteStatement {
37 WTF_MAKE_NONCOPYABLE(SQLiteStatement); WTF_MAKE_FAST_ALLOCATED;
38public:
39 WEBCORE_EXPORT ~SQLiteStatement();
40 WEBCORE_EXPORT SQLiteStatement(SQLiteStatement&&);
41
42 WEBCORE_EXPORT int bindBlob(int index, Span<const uint8_t>);
43 WEBCORE_EXPORT int bindBlob(int index, const String&);
44 WEBCORE_EXPORT int bindText(int index, StringView);
45 WEBCORE_EXPORT int bindInt(int index, int);
46 WEBCORE_EXPORT int bindInt64(int index, int64_t);
47 WEBCORE_EXPORT int bindDouble(int index, double);
48 WEBCORE_EXPORT int bindNull(int index);
49 WEBCORE_EXPORT int bindValue(int index, const SQLValue&);
50 WEBCORE_EXPORT unsigned bindParameterCount() const;
51
52 WEBCORE_EXPORT int step();
53 WEBCORE_EXPORT int reset();
54
55 // steps and finalizes the query.
56 // returns true if all 3 steps succeed with step() returning SQLITE_DONE
57 // returns false otherwise
58 WEBCORE_EXPORT bool executeCommand();
59
60 // Returns -1 on last-step failing. Otherwise, returns number of rows
61 // returned in the last step()
62 int columnCount();
63
64 bool isReadOnly();
65
66 WEBCORE_EXPORT bool isColumnDeclaredAsBlob(int col);
67 String columnName(int col);
68 SQLValue columnValue(int col);
69 WEBCORE_EXPORT String columnText(int col);
70 WEBCORE_EXPORT double columnDouble(int col);
71 WEBCORE_EXPORT int columnInt(int col);
72 WEBCORE_EXPORT int64_t columnInt64(int col);
73 WEBCORE_EXPORT String columnBlobAsString(int col);
74 WEBCORE_EXPORT Vector<uint8_t> columnBlob(int col);
75
76 // The returned Span stays valid until the next step() / reset() or destruction of the statement.
77 Span<const uint8_t> columnBlobAsSpan(int col);
78
79 SQLiteDatabase& database() { return m_database; }
80
81private:
82 friend class SQLiteDatabase;
83 SQLiteStatement(SQLiteDatabase&, sqlite3_stmt*);
84
85 // Returns true if the prepared statement has been stepped at least once using step() but has neither run to completion (returned SQLITE_DONE from step()) nor been reset().
86 bool hasStartedStepping();
87
88 SQLiteDatabase& m_database;
89 sqlite3_stmt* m_statement;
90};
91
92} // namespace WebCore
