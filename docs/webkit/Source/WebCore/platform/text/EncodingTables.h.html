58struct CompareFirst {
59 template<typename TypeA, typename TypeB> bool operator()(const TypeA& a, const TypeB& b)
60 {
61 return a.first < b.first;
62 }
63};
64
65struct EqualFirst {
66 template<typename TypeA, typename TypeB> bool operator()(const TypeA& a, const TypeB& b)
67 {
68 return a.first == b.first;
69 }
70};
71
72struct CompareSecond {
73 template<typename TypeA, typename TypeB> bool operator()(const TypeA& a, const TypeB& b)
74 {
75 return a.second < b.second;
76 }
77};

79template<typename T> struct FirstAdapter {
80 const T& first;
81};
82template<typename T> FirstAdapter<T> makeFirstAdapter(const T& value)
83{
84 return { value };

87template<typename T> struct SecondAdapter {
88 const T& second;

92 return { value };

95template<typename CollectionType> void sortByFirst(CollectionType& collection)
96{
97 std::sort(std::begin(collection), std::end(collection), CompareFirst { });
98}
99
100template<typename CollectionType> void stableSortByFirst(CollectionType& collection)
101{
102 std::stable_sort(std::begin(collection), std::end(collection), CompareFirst { });
103}
104
105template<typename CollectionType> bool isSortedByFirst(const CollectionType& collection)
106{
107 return std::is_sorted(std::begin(collection), std::end(collection), CompareFirst { });
108}
109
110template<typename CollectionType> bool sortedFirstsAreUnique(const CollectionType& collection)
111{
112 return std::adjacent_find(std::begin(collection), std::end(collection), EqualFirst { }) == std::end(collection);
113}

115template<typename CollectionType, typename KeyType> static auto findFirstInSortedPairs(const CollectionType& collection, const KeyType& key) -> std::optional<decltype(std::begin(collection)->second)>
116{
117 if constexpr (std::is_integral_v<KeyType>) {
118 if (key != decltype(std::begin(collection)->first)(key))
119 return std::nullopt;
120 }
121 auto iterator = std::lower_bound(std::begin(collection), std::end(collection), makeFirstAdapter(key), CompareFirst { });
122 if (iterator == std::end(collection) || key < iterator->first)
123 return std::nullopt;
124 return iterator->second;
125}
126
127template<typename CollectionType, typename KeyType> static auto findInSortedPairs(const CollectionType& collection, const KeyType& key) -> std::pair<decltype(std::begin(collection)), decltype(std::begin(collection))>
128{
129 if constexpr (std::is_integral_v<KeyType>) {
130 if (key != decltype(std::begin(collection)->first)(key))
131 return { std::end(collection), std::end(collection) };
132 }
133 return std::equal_range(std::begin(collection), std::end(collection), makeFirstAdapter(key), CompareFirst { });
134}
