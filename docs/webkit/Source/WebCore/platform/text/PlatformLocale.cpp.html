72bool DateTimeStringBuilder::build(const String& formatString)
73{
74 m_builder.reserveCapacity(formatString.length());
75 return DateTimeFormat::parse(formatString, *this);
76}

78String DateTimeStringBuilder::zeroPadString(const String& string, size_t width)
79{
80 if (string.length() >= width)
81 return string;
82 StringBuilder zeroPaddedStringBuilder;
83 zeroPaddedStringBuilder.reserveCapacity(width);
84 for (size_t i = string.length(); i < width; ++i)
85 zeroPaddedStringBuilder.append('0');
86 zeroPaddedStringBuilder.append(string);
87 return zeroPaddedStringBuilder.toString();
88}

90void DateTimeStringBuilder::appendNumber(int number, size_t width)
91{
92 String zeroPaddedNumberString = zeroPadString(String::number(number), width);
93 m_builder.append(m_localizer.convertToLocalizedNumber(zeroPaddedNumberString));
94}

96void DateTimeStringBuilder::visitField(DateTimeFormat::FieldType fieldType, int numberOfPatternCharacters)
97{
98 switch (fieldType) {
99 case DateTimeFormat::FieldTypeYear:
100 // Always use padding width of 4 so it matches DateTimeEditElement.
101 appendNumber(m_date.fullYear(), 4);
102 return;
103 case DateTimeFormat::FieldTypeMonth:
104 if (numberOfPatternCharacters == 3)
105 m_builder.append(m_localizer.shortMonthLabels()[m_date.month()]);
106 else if (numberOfPatternCharacters == 4)
107 m_builder.append(m_localizer.monthLabels()[m_date.month()]);
108 else {
109 // Always use padding width of 2 so it matches DateTimeEditElement.
110 appendNumber(m_date.month() + 1, 2);
111 }
112 return;
113 case DateTimeFormat::FieldTypeMonthStandAlone:
114 if (numberOfPatternCharacters == 3)
115 m_builder.append(m_localizer.shortStandAloneMonthLabels()[m_date.month()]);
116 else if (numberOfPatternCharacters == 4)
117 m_builder.append(m_localizer.standAloneMonthLabels()[m_date.month()]);
118 else {
119 // Always use padding width of 2 so it matches DateTimeEditElement.
120 appendNumber(m_date.month() + 1, 2);
121 }
122 return;
123 case DateTimeFormat::FieldTypeDayOfMonth:
124 // Always use padding width of 2 so it matches DateTimeEditElement.
125 appendNumber(m_date.monthDay(), 2);
126 return;
127 case DateTimeFormat::FieldTypeWeekOfYear:
128 // Always use padding width of 2 so it matches DateTimeEditElement.
129 appendNumber(m_date.week(), 2);
130 return;
131 case DateTimeFormat::FieldTypePeriod:
132 m_builder.append(m_localizer.timeAMPMLabels()[(m_date.hour() >= 12 ? 1 : 0)]);
133 return;
134 case DateTimeFormat::FieldTypeHour12: {
135 int hour12 = m_date.hour() % 12;
136 if (!hour12)
137 hour12 = 12;
138 appendNumber(hour12, numberOfPatternCharacters);
139 return;
140 }
141 case DateTimeFormat::FieldTypeHour23:
142 appendNumber(m_date.hour(), numberOfPatternCharacters);
143 return;
144 case DateTimeFormat::FieldTypeHour11:
145 appendNumber(m_date.hour() % 12, numberOfPatternCharacters);
146 return;
147 case DateTimeFormat::FieldTypeHour24: {
148 int hour24 = m_date.hour();
149 if (!hour24)
150 hour24 = 24;
151 appendNumber(hour24, numberOfPatternCharacters);
152 return;
153 }
154 case DateTimeFormat::FieldTypeMinute:
155 appendNumber(m_date.minute(), numberOfPatternCharacters);
156 return;
157 case DateTimeFormat::FieldTypeSecond:
158 if (!m_date.millisecond())
159 appendNumber(m_date.second(), numberOfPatternCharacters);
160 else {
161 double second = m_date.second() + m_date.millisecond() / 1000.0;
162 String zeroPaddedSecondString = zeroPadString(String::numberToStringFixedWidth(second, 3), numberOfPatternCharacters + 4);
163 m_builder.append(m_localizer.convertToLocalizedNumber(zeroPaddedSecondString));
164 }
165 return;
166 default:
167 return;
168 }
169}

186void Locale::setLocaleData(const Vector<String, DecimalSymbolsSize>& symbols, const String& positivePrefix, const String& positiveSuffix, const String& negativePrefix, const String& negativeSuffix)
187{
188 for (size_t i = 0; i < symbols.size(); ++i) {
189 ASSERT(!symbols[i].isEmpty());
190 m_decimalSymbols[i] = symbols[i];
191 }
192 m_positivePrefix = positivePrefix;
193 m_positiveSuffix = positiveSuffix;
194 m_negativePrefix = negativePrefix;
195 m_negativeSuffix = negativeSuffix;
196 ASSERT(!m_positivePrefix.isEmpty() || !m_positiveSuffix.isEmpty() || !m_negativePrefix.isEmpty() || !m_negativeSuffix.isEmpty());
197 m_hasLocaleData = true;
198}

200String Locale::convertToLocalizedNumber(const String& input)
201{
202 initializeLocaleData();
203 if (!m_hasLocaleData || input.isEmpty())
204 return input;
205
206 unsigned i = 0;
207 bool isNegative = false;
208 StringBuilder builder;
209 builder.reserveCapacity(input.length());
210
211 if (input[0] == '-') {
212 ++i;
213 isNegative = true;
214 builder.append(m_negativePrefix);
215 } else
216 builder.append(m_positivePrefix);
217
218 for (; i < input.length(); ++i) {
219 switch (input[i]) {
220 case '0':
221 case '1':
222 case '2':
223 case '3':
224 case '4':
225 case '5':
226 case '6':
227 case '7':
228 case '8':
229 case '9':
230 builder.append(m_decimalSymbols[input[i] - '0']);
231 break;
232 case '.':
233 builder.append(m_decimalSymbols[DecimalSeparatorIndex]);
234 break;
235 default:
236 ASSERT_NOT_REACHED();
237 }
238 }
239
240 builder.append(isNegative ? m_negativeSuffix : m_positiveSuffix);
241
242 return builder.toString();
243}

245static bool matches(const String& text, unsigned position, const String& part)
246{
247 if (part.isEmpty())
248 return true;
249 if (position + part.length() > text.length())
250 return false;
251 for (unsigned i = 0; i < part.length(); ++i) {
252 if (text[position + i] != part[i])
253 return false;
254 }
255 return true;
256}

258bool Locale::detectSignAndGetDigitRange(const String& input, bool& isNegative, unsigned& startIndex, unsigned& endIndex)
259{
260 startIndex = 0;
261 endIndex = input.length();
262 if (m_negativePrefix.isEmpty() && m_negativeSuffix.isEmpty()) {
263 if (input.startsWith(m_positivePrefix) && input.endsWith(m_positiveSuffix)) {
264 isNegative = false;
265 startIndex = m_positivePrefix.length();
266 endIndex -= m_positiveSuffix.length();
267 } else
268 isNegative = true;
269 } else {
270 if (input.startsWith(m_negativePrefix) && input.endsWith(m_negativeSuffix)) {
271 isNegative = true;
272 startIndex = m_negativePrefix.length();
273 endIndex -= m_negativeSuffix.length();
274 } else {
275 isNegative = false;
276 if (input.startsWith(m_positivePrefix) && input.endsWith(m_positiveSuffix)) {
277 startIndex = m_positivePrefix.length();
278 endIndex -= m_positiveSuffix.length();
279 } else
280 return false;
281 }
282 }
283 return true;
284}

286unsigned Locale::matchedDecimalSymbolIndex(const String& input, unsigned& position)
287{
288 for (unsigned symbolIndex = 0; symbolIndex < DecimalSymbolsSize; ++symbolIndex) {
289 if (m_decimalSymbols[symbolIndex].length() && matches(input, position, m_decimalSymbols[symbolIndex])) {
290 position += m_decimalSymbols[symbolIndex].length();
291 return symbolIndex;
292 }
293 }
294 return DecimalSymbolsSize;
295}

297String Locale::convertFromLocalizedNumber(const String& localized)
298{
299 initializeLocaleData();
300 String input = localized.stripWhiteSpace();
301 if (!m_hasLocaleData || input.isEmpty())
302 return input;
303
304 bool isNegative;
305 unsigned startIndex;
306 unsigned endIndex;
307 if (!detectSignAndGetDigitRange(input, isNegative, startIndex, endIndex))
308 return input;
309
310 StringBuilder builder;
311 builder.reserveCapacity(input.length());
312 if (isNegative)
313 builder.append('-');
314 for (unsigned i = startIndex; i < endIndex;) {
315 unsigned symbolIndex = matchedDecimalSymbolIndex(input, i);
316 if (symbolIndex >= DecimalSymbolsSize)
317 return input;
318 if (symbolIndex == DecimalSeparatorIndex)
319 builder.append('.');
320 else if (symbolIndex == GroupSeparatorIndex)
321 return input;
322 else
323 builder.append(static_cast<UChar>('0' + symbolIndex));
324 }
325 return builder.toString();
326}

328#if ENABLE(DATE_AND_TIME_INPUT_TYPES)
329String Locale::formatDateTime(const DateComponents& date, FormatType formatType)
330{
331 if (date.type() == DateComponentsType::Invalid)
332 return String();
333
334 DateTimeStringBuilder builder(*this, date);
335 switch (date.type()) {
336 case DateComponentsType::Time:
337 builder.build(formatType == FormatTypeShort ? shortTimeFormat() : timeFormat());
338 break;
339 case DateComponentsType::Date:
340 builder.build(dateFormat());
341 break;
342 case DateComponentsType::Month:
343 builder.build(formatType == FormatTypeShort ? shortMonthFormat() : monthFormat());
344 break;
345 case DateComponentsType::Week:
346 // FIXME: Add support for formatting weeks.
347 break;
348 case DateComponentsType::DateTimeLocal:
349 builder.build(formatType == FormatTypeShort ? dateTimeFormatWithoutSeconds() : dateTimeFormatWithSeconds());
350 break;
351 case DateComponentsType::Invalid:
352 ASSERT_NOT_REACHED();
353 break;
354 }
355 return builder.toString();
356}
