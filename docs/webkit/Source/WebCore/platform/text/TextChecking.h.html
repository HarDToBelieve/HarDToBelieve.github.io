43enum class TextCheckingType : uint8_t {
44 None = 0,
45 Spelling = 1 << 0,
46 Grammar = 1 << 1,
47 Link = 1 << 2,

61 TextCheckingProcessBatch,
62 TextCheckingProcessIncremental
63};

65struct GrammarDetail {
66 CharacterRange range;
67 Vector<String> guesses;
68 String userDescription;
69};

71struct TextCheckingResult {
72 TextCheckingType type;
73 CharacterRange range;
74 Vector<GrammarDetail> details;
75 String replacement;
76};

78struct TextCheckingGuesses {

84enum TextCheckingRequestIdentifierType { };
85using TextCheckingRequestIdentifier = ObjectIdentifier<TextCheckingRequestIdentifierType>;
86
87class TextCheckingRequestData {
88 friend class SpellCheckRequest; // For access to m_identifier.
89public:
90 TextCheckingRequestData() = default;
91 TextCheckingRequestData(std::optional<TextCheckingRequestIdentifier> identifier, const String& text, OptionSet<TextCheckingType> checkingTypes, TextCheckingProcessType processType)
92 : m_text { text }
93 , m_identifier { identifier }
94 , m_processType { processType }
95 , m_checkingTypes { checkingTypes }
96 {
97 }
98
99 std::optional<TextCheckingRequestIdentifier> identifier() const { return m_identifier; }
100 const String& text() const { return m_text; }
101 OptionSet<TextCheckingType> checkingTypes() const { return m_checkingTypes; }
102 TextCheckingProcessType processType() const { return m_processType; }
103
104private:
105 String m_text;
106 std::optional<TextCheckingRequestIdentifier> m_identifier;
107 TextCheckingProcessType m_processType { TextCheckingProcessIncremental };
108 OptionSet<TextCheckingType> m_checkingTypes;
109};
110
111class TextCheckingRequest : public RefCounted<TextCheckingRequest> {
112public:
113 virtual ~TextCheckingRequest() = default;
114
115 virtual const TextCheckingRequestData& data() const = 0;
116 virtual void didSucceed(const Vector<TextCheckingResult>&) = 0;
117 virtual void didCancel() = 0;
118};
