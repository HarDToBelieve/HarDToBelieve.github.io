38class TextEncoding;

40using UnencodableReplacementArray = std::array<char, 32>;
41
42class TextCodec {
43 WTF_MAKE_NONCOPYABLE(TextCodec); WTF_MAKE_FAST_ALLOCATED;
44public:
45 TextCodec() = default;
46 virtual ~TextCodec() = default;
47
48 virtual void stripByteOrderMark() { }
49 virtual String decode(const char*, size_t length, bool flush, bool stopOnError, bool& sawError) = 0;
50
51 virtual Vector<uint8_t> encode(StringView, UnencodableHandling) const = 0;
52
53 // Fills a null-terminated string representation of the given
54 // unencodable character into the given replacement buffer.
55 // The length of the string (not including the null) will be returned.
56 static int getUnencodableReplacement(UChar32, UnencodableHandling, UnencodableReplacementArray&);
57};
