165using JIS0208EncodeIndex = std::array<std::pair<UChar, uint16_t>, sizeof(jis0208()) / sizeof(jis0208()[0])>;
166static const JIS0208EncodeIndex& jis0208EncodeIndex()
167{
168 // Allocate this at runtime because building it at compile time would make the binary much larger and this is often not used.
169 static JIS0208EncodeIndex* table;
170 static std::once_flag once;
171 std::call_once(once, [&] {
172 table = new JIS0208EncodeIndex;
173 auto& index = jis0208();
174 for (size_t i = 0; i < index.size(); i++)
175 (*table)[i] = { index[i].second, index[i].first };
176 stableSortByFirst(*table);
177 });
178 return *table;
179}

181String TextCodecCJK::decodeCommon(const uint8_t* bytes, size_t length, bool flush, bool stopOnError, bool& sawError, const Function<SawError(uint8_t, StringBuilder&)>& byteParser)
182{
183 StringBuilder result;
184 result.reserveCapacity(length);
185
186 if (m_prependedByte && byteParser(*std::exchange(m_prependedByte, std::nullopt), result) == SawError::Yes) {
187 sawError = true;
188 result.append(replacementCharacter);
189 if (stopOnError) {
190 m_lead = 0x00;
191 return result.toString();
192 }
193 }
194 for (size_t i = 0; i < length; i++) {
195 if (byteParser(bytes[i], result) == SawError::Yes) {
196 sawError = true;
197 result.append(replacementCharacter);
198 if (stopOnError) {
199 m_lead = 0x00;
200 return result.toString();
201 }
202 }
203 if (m_prependedByte && byteParser(*std::exchange(m_prependedByte, std::nullopt), result) == SawError::Yes) {
204 sawError = true;
205 result.append(replacementCharacter);
206 if (stopOnError) {
207 m_lead = 0x00;
208 return result.toString();
209 }
210 }
211 }
212
213 if (flush && m_lead) {
214 m_lead = 0x00;
215 sawError = true;
216 result.append(replacementCharacter);
217 }
218
219 return result.toString();
220}

232// https://encoding.spec.whatwg.org/#euc-jp-decoder
233String TextCodecCJK::eucJPDecode(const uint8_t* bytes, size_t length, bool flush, bool stopOnError, bool& sawError)
234{
235 return decodeCommon(bytes, length, flush, stopOnError, sawError, [this] (uint8_t byte, StringBuilder& result) {
236 if (uint8_t lead = std::exchange(m_lead, 0x00)) {
237 if (lead == 0x8E && byte >= 0xA1 && byte <= 0xDF) {
238 result.appendCharacter(0xFF61 - 0xA1 + byte);
239 return SawError::No;
240 }
241 if (lead == 0x8F && byte >= 0xA1 && byte <= 0xFE) {
242 m_jis0212 = true;
243 m_lead = byte;
244 return SawError::No;
245 }
246 if (lead >= 0xA1 && lead <= 0xFE && byte >= 0xA1 && byte <= 0xFE) {
247 uint16_t pointer = (lead - 0xA1) * 94 + byte - 0xA1;
248 if (auto codePoint = std::exchange(m_jis0212, false) ? codePointJIS0212(pointer) : codePointJIS0208(pointer)) {
249 result.append(*codePoint);
250 return SawError::No;
251 }
252 }
253 if (isASCII(byte))
254 m_prependedByte = byte;
255 return SawError::Yes;
256 }
257 if (isASCII(byte)) {
258 result.append(static_cast<char>(byte));
259 return SawError::No;
260 }
261 if (byte == 0x8E || byte == 0x8F || (byte >= 0xA1 && byte <= 0xFE)) {
262 m_lead = byte;
263 return SawError::No;
264 }
265 return SawError::Yes;
266 });
267}

269// https://encoding.spec.whatwg.org/#euc-jp-encoder
270static Vector<uint8_t> eucJPEncode(StringView string, Function<void(UChar32, Vector<uint8_t>&)>&& unencodableHandler)
271{
272 Vector<uint8_t> result;
273 result.reserveInitialCapacity(string.length());
274
275 auto characters = string.upconvertedCharacters();
276 for (WTF::CodePointIterator<UChar> iterator(characters.get(), characters.get() + string.length()); !iterator.atEnd(); ++iterator) {
277 auto codePoint = *iterator;
278 if (isASCII(codePoint)) {
279 result.append(codePoint);
280 continue;
281 }
282 if (codePoint == 0x00A5) {
283 result.append(0x5C);
284 continue;
285 }
286 if (codePoint == 0x203E) {
287 result.append(0x7E);
288 continue;
289 }
290 if (codePoint >= 0xFF61 && codePoint <= 0xFF9F) {
291 result.append(0x8E);
292 result.append(codePoint - 0xFF61 + 0xA1);
293 continue;
294 }
295 if (codePoint == 0x2212)
296 codePoint = 0xFF0D;
297
298 auto pointer = findFirstInSortedPairs(jis0208EncodeIndex(), codePoint);
299 if (!pointer) {
300 unencodableHandler(codePoint, result);
301 continue;
302 }
303 result.append(*pointer / 94 + 0xA1);
304 result.append(*pointer % 94 + 0xA1);
305 }
306 return result;
307}

309// https://encoding.spec.whatwg.org/#iso-2022-jp-decoder
310String TextCodecCJK::iso2022JPDecode(const uint8_t* bytes, size_t length, bool flush, bool stopOnError, bool& sawError)
311{
312 auto byteParser = [&] (uint8_t byte, StringBuilder& result) {
313 switch (m_iso2022JPDecoderState) {
314 case ISO2022JPDecoderState::ASCII:
315 if (byte == 0x1B) {
316 m_iso2022JPDecoderState = ISO2022JPDecoderState::EscapeStart;
317 break;
318 }
319 if (byte <= 0x7F && byte != 0x0E && byte != 0x0F && byte != 0x1B) {
320 m_iso2022JPOutput = false;
321 result.append(byte);
322 break;
323 }
324 m_iso2022JPOutput = false;
325 return SawError::Yes;
326 case ISO2022JPDecoderState::Roman:
327 if (byte == 0x1B) {
328 m_iso2022JPDecoderState = ISO2022JPDecoderState::EscapeStart;
329 break;
330 }
331 if (byte == 0x5C) {
332 m_iso2022JPOutput = false;
333 result.append(static_cast<UChar>(0x00A5));
334 break;
335 }
336 if (byte == 0x7E) {
337 m_iso2022JPOutput = false;
338 result.append(static_cast<UChar>(0x203E));
339 break;
340 }
341 if (byte <= 0x7F && byte != 0x0E && byte != 0x0F && byte != 0x1B && byte != 0x5C && byte != 0x7E) {
342 m_iso2022JPOutput = false;
343 result.append(byte);
344 break;
345 }
346 m_iso2022JPOutput = false;
347 return SawError::Yes;
348 case ISO2022JPDecoderState::Katakana:
349 if (byte == 0x1B) {
350 m_iso2022JPDecoderState = ISO2022JPDecoderState::EscapeStart;
351 break;
352 }
353 if (byte >= 0x21 && byte <= 0x5F) {
354 m_iso2022JPOutput = false;
355 result.append(static_cast<UChar>(0xFF61 - 0x21 + byte));
356 break;
357 }
358 m_iso2022JPOutput = false;
359 return SawError::Yes;
360 case ISO2022JPDecoderState::LeadByte:
361 if (byte == 0x1B) {
362 m_iso2022JPDecoderState = ISO2022JPDecoderState::EscapeStart;
363 break;
364 }
365 if (byte >= 0x21 && byte <= 0x7E) {
366 m_iso2022JPOutput = false;
367 m_lead = byte;
368 m_iso2022JPDecoderState = ISO2022JPDecoderState::TrailByte;
369 break;
370 }
371 m_iso2022JPOutput = false;
372 return SawError::Yes;
373 case ISO2022JPDecoderState::TrailByte:
374 if (byte == 0x1B) {
375 m_iso2022JPDecoderState = ISO2022JPDecoderState::EscapeStart;
376 return SawError::Yes;
377 }
378 m_iso2022JPDecoderState = ISO2022JPDecoderState::LeadByte;
379 if (byte >= 0x21 && byte <= 0x7E) {
380 uint16_t pointer = (m_lead - 0x21) * 94 + byte - 0x21;
381 if (auto codePoint = codePointJIS0208(pointer)) {
382 result.append(*codePoint);
383 break;
384 }
385 return SawError::Yes;
386 }
387 return SawError::Yes;
388 case ISO2022JPDecoderState::EscapeStart:
389 if (byte == 0x24 || byte == 0x28) {
390 m_lead = byte;
391 m_iso2022JPDecoderState = ISO2022JPDecoderState::Escape;
392 break;
393 }
394 m_prependedByte = byte;
395 m_iso2022JPOutput = false;
396 m_iso2022JPDecoderState = m_iso2022JPDecoderOutputState;
397 return SawError::Yes;
398 case ISO2022JPDecoderState::Escape: {
399 uint8_t lead = std::exchange(m_lead, 0x00);
400 std::optional<ISO2022JPDecoderState> state;
401 if (lead == 0x28) {
402 if (byte == 0x42)
403 state = ISO2022JPDecoderState::ASCII;
404 else if (byte == 0x4A)
405 state = ISO2022JPDecoderState::Roman;
406 else if (byte == 0x49)
407 state = ISO2022JPDecoderState::Katakana;
408 } else if (lead == 0x24 && (byte == 0x40 || byte == 0x42))
409 state = ISO2022JPDecoderState::LeadByte;
410 if (state) {
411 m_iso2022JPDecoderState = *state;
412 m_iso2022JPDecoderOutputState = *state;
413 if (std::exchange(m_iso2022JPOutput, true))
414 return SawError::Yes;
415 break;
416 }
417 m_prependedByte = lead;
418 m_iso2022JPSecondPrependedByte = byte;
419 m_iso2022JPOutput = false;
420 m_iso2022JPDecoderState = m_iso2022JPDecoderOutputState;
421 return SawError::Yes;
422 }
423 }
424 return SawError::No;
425 };
426
427 StringBuilder result;
428 result.reserveCapacity(length);
429
430 if (m_prependedByte && byteParser(*std::exchange(m_prependedByte, std::nullopt), result) == SawError::Yes) {
431 sawError = true;
432 result.append(replacementCharacter);
433 if (stopOnError) {
434 m_lead = 0x00;
435 return result.toString();
436 }
437 }
438 if (m_iso2022JPSecondPrependedByte && byteParser(*std::exchange(m_iso2022JPSecondPrependedByte, std::nullopt), result) == SawError::Yes && stopOnError) {
439 sawError = true;
440 result.append(replacementCharacter);
441 if (stopOnError) {
442 m_lead = 0x00;
443 return result.toString();
444 }
445 }
446 for (size_t i = 0; i < length; i++) {
447 if (byteParser(bytes[i], result) == SawError::Yes) {
448 sawError = true;
449 result.append(replacementCharacter);
450 if (stopOnError) {
451 m_lead = 0x00;
452 return result.toString();
453 }
454 }
455 if (m_prependedByte && byteParser(*std::exchange(m_prependedByte, std::nullopt), result) == SawError::Yes) {
456 sawError = true;
457 result.append(replacementCharacter);
458 if (stopOnError) {
459 m_lead = 0x00;
460 return result.toString();
461 }
462 }
463 if (m_iso2022JPSecondPrependedByte && byteParser(*std::exchange(m_iso2022JPSecondPrependedByte, std::nullopt), result) == SawError::Yes && stopOnError) {
464 sawError = true;
465 result.append(replacementCharacter);
466 if (stopOnError) {
467 m_lead = 0x00;
468 return result.toString();
469 }
470 }
471 }
472
473 if (flush) {
474 switch (m_iso2022JPDecoderState) {
475 case ISO2022JPDecoderState::ASCII:
476 case ISO2022JPDecoderState::Roman:
477 case ISO2022JPDecoderState::Katakana:
478 case ISO2022JPDecoderState::LeadByte:
479 break;
480 case ISO2022JPDecoderState::TrailByte:
481 m_iso2022JPDecoderState = ISO2022JPDecoderState::LeadByte;
482 FALLTHROUGH;
483 case ISO2022JPDecoderState::EscapeStart:
484 sawError = true;
485 result.append(replacementCharacter);
486 break;
487 case ISO2022JPDecoderState::Escape:
488 sawError = true;
489 result.append(replacementCharacter);
490 if (m_lead) {
491 ASSERT(isASCII(m_lead));
492 result.append(std::exchange(m_lead, 0x00));
493 }
494 break;
495 }
496 }
497
498 return result.toString();
499}

501// https://encoding.spec.whatwg.org/#iso-2022-jp-encoder
502static Vector<uint8_t> iso2022JPEncode(StringView string, Function<void(UChar32, Vector<uint8_t>&)>&& unencodableHandler)
503{
504 enum class State : uint8_t { ASCII, Roman, Jis0208 };
505 State state { State::ASCII };
506
507 Vector<uint8_t> result;
508 result.reserveInitialCapacity(string.length());
509
510 auto changeStateToASCII = [&] {
511 state = State::ASCII;
512 result.append(0x1B);
513 result.append(0x28);
514 result.append(0x42);
515 };
516
517 auto statefulUnencodableHandler = [&] (UChar32 codePoint, Vector<uint8_t>& result) {
518 if (state == State::Jis0208)
519 changeStateToASCII();
520 unencodableHandler(codePoint, result);
521 };
522
523 Function<void(UChar32)> parseCodePoint;
524 parseCodePoint = [&] (UChar32 codePoint) {
525 if ((state == State::ASCII || state == State::Roman) && (codePoint == 0x000E || codePoint == 0x000F || codePoint == 0x001B)) {
526 statefulUnencodableHandler(replacementCharacter, result);
527 return;
528 }
529 if (state == State::ASCII && isASCII(codePoint)) {
530 result.append(codePoint);
531 return;
532 }
533 if (state == State::Roman) {
534 if (isASCII(codePoint) && codePoint != 0x005C && codePoint !=0x007E) {
535 result.append(codePoint);
536 return;
537 }
538 if (codePoint == 0x00A5) {
539 result.append(0x5C);
540 return;
541 }
542 if (codePoint == 0x203E) {
543 result.append(0x7E);
544 return;
545 }
546 }
547 if (isASCII(codePoint) && state != State::ASCII) {
548 if (state != State::ASCII)
549 changeStateToASCII();
550 parseCodePoint(codePoint);
551 return;
552 }
553 if ((codePoint == 0x00A5 || codePoint == 0x203E) && state != State::Roman) {
554 state = State::Roman;
555 result.append(0x1B);
556 result.append(0x28);
557 result.append(0x4A);
558 parseCodePoint(codePoint);
559 return;
560 }
561 if (codePoint == 0x2212)
562 codePoint = 0xFF0D;
563 if (codePoint >= 0xFF61 && codePoint <= 0xFF9F) {
564 // From https://encoding.spec.whatwg.org/index-iso-2022-jp-katakana.txt
565 static constexpr std::array<UChar32, 63> iso2022JPKatakana {
566 0x3002, 0x300C, 0x300D, 0x3001, 0x30FB, 0x30F2, 0x30A1, 0x30A3, 0x30A5, 0x30A7, 0x30A9, 0x30E3, 0x30E5, 0x30E7, 0x30C3, 0x30FC,
567 0x30A2, 0x30A4, 0x30A6, 0x30A8, 0x30AA, 0x30AB, 0x30AD, 0x30AF, 0x30B1, 0x30B3, 0x30B5, 0x30B7, 0x30B9, 0x30BB, 0x30BD, 0x30BF,
568 0x30C1, 0x30C4, 0x30C6, 0x30C8, 0x30CA, 0x30CB, 0x30CC, 0x30CD, 0x30CE, 0x30CF, 0x30D2, 0x30D5, 0x30D8, 0x30DB, 0x30DE, 0x30DF,
569 0x30E0, 0x30E1, 0x30E2, 0x30E4, 0x30E6, 0x30E8, 0x30E9, 0x30EA, 0x30EB, 0x30EC, 0x30ED, 0x30EF, 0x30F3, 0x309B, 0x309C
570 };
571 static_assert(std::size(iso2022JPKatakana) == 0xFF9F - 0xFF61 + 1);
572 codePoint = iso2022JPKatakana[codePoint - 0xFF61];
573 }
574
575 auto pointer = findFirstInSortedPairs(jis0208EncodeIndex(), codePoint);
576 if (!pointer) {
577 statefulUnencodableHandler(codePoint, result);
578 return;
579 }
580 if (state != State::Jis0208) {
581 state = State::Jis0208;
582 result.append(0x1B);
583 result.append(0x24);
584 result.append(0x42);
585 parseCodePoint(codePoint);
586 return;
587 }
588 result.append(*pointer / 94 + 0x21);
589 result.append(*pointer % 94 + 0x21);
590 };
591
592 auto characters = string.upconvertedCharacters();
593 for (WTF::CodePointIterator<UChar> iterator(characters.get(), characters.get() + string.length()); !iterator.atEnd(); ++iterator)
594 parseCodePoint(*iterator);
595
596 if (state != State::ASCII)
597 changeStateToASCII();
598
599 return result;
600}

602// https://encoding.spec.whatwg.org/#shift_jis-decoder
603String TextCodecCJK::shiftJISDecode(const uint8_t* bytes, size_t length, bool flush, bool stopOnError, bool& sawError)
604{
605 return decodeCommon(bytes, length, flush, stopOnError, sawError, [this] (uint8_t byte, StringBuilder& result) {
606 if (uint8_t lead = std::exchange(m_lead, 0x00)) {
607 uint8_t offset = byte < 0x7F ? 0x40 : 0x41;
608 uint8_t leadOffset = lead < 0xA0 ? 0x81 : 0xC1;
609 if ((byte >= 0x40 && byte <= 0x7E) || (byte >= 0x80 && byte <= 0xFC)) {
610 uint16_t pointer = (lead - leadOffset) * 188 + byte - offset;
611 if (pointer >= 8836 && pointer <= 10715) {
612 result.append(static_cast<UChar>(0xE000 - 8836 + pointer));
613 return SawError::No;
614 }
615 if (auto codePoint = codePointJIS0208(pointer)) {
616 result.append(*codePoint);
617 return SawError::No;
618 }
619 }
620 if (isASCII(byte))
621 m_prependedByte = byte;
622 return SawError::Yes;
623 }
624 if (isASCII(byte) || byte == 0x80) {
625 result.append(byte);
626 return SawError::No;
627 }
628 if (byte >= 0xA1 && byte <= 0xDF) {
629 result.append(static_cast<UChar>(0xFF61 - 0xA1 + byte));
630 return SawError::No;
631 }
632 if ((byte >= 0x81 && byte <= 0x9F) || (byte >= 0xE0 && byte <= 0xFC)) {
633 m_lead = byte;
634 return SawError::No;
635 }
636 return SawError::Yes;
637 });
638}

640// https://encoding.spec.whatwg.org/#shift_jis-encoder
641static Vector<uint8_t> shiftJISEncode(StringView string, Function<void(UChar32, Vector<uint8_t>&)>&& unencodableHandler)
642{
643 Vector<uint8_t> result;
644 result.reserveInitialCapacity(string.length());
645
646 auto characters = string.upconvertedCharacters();
647 for (WTF::CodePointIterator<UChar> iterator(characters.get(), characters.get() + string.length()); !iterator.atEnd(); ++iterator) {
648 auto codePoint = *iterator;
649 if (isASCII(codePoint) || codePoint == 0x0080) {
650 result.append(codePoint);
651 continue;
652 }
653 if (codePoint == 0x00A5) {
654 result.append(0x5C);
655 continue;
656 }
657 if (codePoint == 0x203E) {
658 result.append(0x7E);
659 continue;
660 }
661 if (codePoint >= 0xFF61 && codePoint <= 0xFF9F) {
662 result.append(codePoint - 0xFF61 + 0xA1);
663 continue;
664 }
665 if (codePoint == 0x2212)
666 codePoint = 0xFF0D;
667
668 auto range = findInSortedPairs(jis0208EncodeIndex(), codePoint);
669 if (range.first == range.second) {
670 unencodableHandler(codePoint, result);
671 continue;
672 }
673
674 ASSERT(range.first + 3 >= range.second);
675 for (auto pair = range.first; pair < range.second; pair++) {
676 uint16_t pointer = pair->second;
677 if (pointer >= 8272 && pointer <= 8835)
678 continue;
679 uint8_t lead = pointer / 188;
680 uint8_t leadOffset = lead < 0x1F ? 0x81 : 0xC1;
681 uint8_t trail = pointer % 188;
682 uint8_t offset = trail < 0x3F ? 0x40 : 0x41;
683 result.append(lead + leadOffset);
684 result.append(trail + offset);
685 break;
686 }
687 }
688 return result;
689}

855{

895// https://encoding.spec.whatwg.org/#gb18030-decoder
896String TextCodecCJK::gb18030Decode(const uint8_t* bytes, size_t length, bool flush, bool stopOnError, bool& sawError)
897{
898 Function<SawError(uint8_t, StringBuilder&)> parseByte;
899 parseByte = [&] (uint8_t byte, StringBuilder& result) {
900 if (m_gb18030Third) {
901 if (byte < 0x30 || byte > 0x39) {
902 sawError = true;
903 result.append(replacementCharacter);
904 m_gb18030First = 0x00;
905 uint8_t second = std::exchange(m_gb18030Second, 0x00);
906 uint8_t third = std::exchange(m_gb18030Third, 0x00);
907 if (parseByte(second, result) == SawError::Yes) {
908 sawError = true;
909 result.append(replacementCharacter);
910 }
911 if (parseByte(third, result) == SawError::Yes) {
912 sawError = true;
913 result.append(replacementCharacter);
914 }
915 return parseByte(byte, result);
916 }
917 uint8_t first = std::exchange(m_gb18030First, 0x00);
918 uint8_t second = std::exchange(m_gb18030Second, 0x00);
919 uint8_t third = std::exchange(m_gb18030Third, 0x00);
920 if (auto codePoint = gb18030RangesCodePoint(((first - 0x81) * 10 * 126 * 10) + ((second - 0x30) * 10 * 126) + ((third - 0x81) * 10) + byte - 0x30)) {
921 result.appendCharacter(*codePoint);
922 return SawError::No;
923 }
924 return SawError::Yes;
925 }
926 if (m_gb18030Second) {
927 if (byte >= 0x81 && byte <= 0xFE) {
928 m_gb18030Third = byte;
929 return SawError::No;
930 }
931 sawError = true;
932 result.append(replacementCharacter);
933 m_gb18030First = 0x00;
934 if (parseByte(std::exchange(m_gb18030Second, 0x00), result) == SawError::Yes) {
935 sawError = true;
936 result.append(replacementCharacter);
937 }
938 return parseByte(byte, result);
939 }
940 if (m_gb18030First) {
941 if (byte >= 0x30 && byte <= 0x39) {
942 m_gb18030Second = byte;
943 return SawError::No;
944 }
945 uint8_t lead = std::exchange(m_gb18030First, 0x00);
946 uint8_t offset = byte < 0x7F ? 0x40 : 0x41;
947 if ((byte >= 0x40 && byte <= 0x7E) || (byte >= 0x80 && byte <= 0xFE)) {
948 size_t pointer = (lead - 0x81) * 190 + byte - offset;
949 if (pointer < gb18030().size()) {
950 result.append(gb18030()[pointer]);
951 return SawError::No;
952 }
953 }
954 if (isASCII(byte))
955 m_prependedByte = byte;
956 return SawError::Yes;
957 }
958 if (isASCII(byte)) {
959 result.append(byte);
960 return SawError::No;
961 }
962 if (byte == 0x80) {
963 result.appendCharacter(0x20AC);
964 return SawError::No;
965 }
966 if (byte >= 0x81 && byte <= 0xFE) {
967 m_gb18030First = byte;
968 return SawError::No;
969 }
970 return SawError::Yes;
971 };
972
973 auto result = decodeCommon(bytes, length, flush, stopOnError, sawError, parseByte);
974 if (flush && (m_gb18030First || m_gb18030Second || m_gb18030Third)) {
975 m_gb18030First = 0x00;
976 m_gb18030Second = 0x00;
977 m_gb18030Third = 0x00;
978 sawError = true;
979 result = makeString(result, replacementCharacter);
980 }
981 return result;
982}

985enum class IsGBK : bool { No, Yes };
986static Vector<uint8_t> gbEncodeShared(StringView string, Function<void(UChar32, Vector<uint8_t>&)>&& unencodableHandler, IsGBK isGBK)
987{
988 Vector<uint8_t> result;
989 result.reserveInitialCapacity(string.length());
990
991 auto characters = string.upconvertedCharacters();
992 for (WTF::CodePointIterator<UChar> iterator(characters.get(), characters.get() + string.length()); !iterator.atEnd(); ++iterator) {
993 auto codePoint = *iterator;
994 if (isASCII(codePoint)) {
995 result.append(codePoint);
996 continue;
997 }
998 if (codePoint == 0xE5E5) {
999 unencodableHandler(codePoint, result);
1000 continue;
1001 }
1002 if (isGBK == IsGBK::Yes && codePoint == 0x20AC) {
1003 result.append(0x80);
1004 continue;
1005 }
1006 auto pointerRange = findInSortedPairs(gb18030EncodeIndex(), codePoint);
1007 if (pointerRange.first != pointerRange.second) {
1008 uint16_t pointer = pointerRange.first->second;
1009 uint8_t lead = pointer / 190 + 0x81;
1010 uint8_t trail = pointer % 190;
1011 uint8_t offset = trail < 0x3F ? 0x40 : 0x41;
1012 result.append(lead);
1013 result.append(trail + offset);
1014 continue;
1015 }
1016 if (isGBK == IsGBK::Yes) {
1017 unencodableHandler(codePoint, result);
1018 continue;
1019 }
1020 uint32_t pointer = gb18030RangesPointer(codePoint);
1021 uint8_t byte1 = pointer / (10 * 126 * 10);
1022 pointer = pointer % (10 * 126 * 10);
1023 uint8_t byte2 = pointer / (10 * 126);
1024 pointer = pointer % (10 * 126);
1025 uint8_t byte3 = pointer / 10;
1026 uint8_t byte4 = pointer % 10;
1027 result.append(byte1 + 0x81);
1028 result.append(byte2 + 0x30);
1029 result.append(byte3 + 0x81);
1030 result.append(byte4 + 0x30);
1031 }
1032 return result;
1033}
