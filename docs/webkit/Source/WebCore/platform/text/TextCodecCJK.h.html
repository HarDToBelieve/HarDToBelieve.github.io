26#pragma once
27
28#include "TextCodec.h"
29#include <optional>
30
31namespace WebCore {
32
33class TextCodecCJK final : public TextCodec {
34public:
35 static void registerEncodingNames(EncodingNameRegistrar);
36 static void registerCodecs(TextCodecRegistrar);
37
38 enum class Encoding : uint8_t;
39 explicit TextCodecCJK(Encoding);
40
41private:
42 String decode(const char*, size_t length, bool flush, bool stopOnError, bool& sawError) final;
43 Vector<uint8_t> encode(StringView, UnencodableHandling) const final;
44
45 enum class SawError : bool { No, Yes };
46 String decodeCommon(const uint8_t*, size_t, bool, bool, bool&, const Function<SawError(uint8_t, StringBuilder&)>&);
47
48 String eucJPDecode(const uint8_t*, size_t, bool, bool, bool&);
49 String iso2022JPDecode(const uint8_t*, size_t, bool, bool, bool&);
50 String shiftJISDecode(const uint8_t*, size_t, bool, bool, bool&);
51 String eucKRDecode(const uint8_t*, size_t, bool, bool, bool&);
52 String big5Decode(const uint8_t*, size_t, bool, bool, bool&);
53 String gbkDecode(const uint8_t*, size_t, bool, bool, bool&);
54 String gb18030Decode(const uint8_t*, size_t, bool, bool, bool&);
55
56 const Encoding m_encoding;
57
58 bool m_jis0212 { false };
59
60 enum class ISO2022JPDecoderState : uint8_t { ASCII, Roman, Katakana, LeadByte, TrailByte, EscapeStart, Escape };
61 ISO2022JPDecoderState m_iso2022JPDecoderState { ISO2022JPDecoderState::ASCII };
62 ISO2022JPDecoderState m_iso2022JPDecoderOutputState { ISO2022JPDecoderState::ASCII };
63 bool m_iso2022JPOutput { false };
64 std::optional<uint8_t> m_iso2022JPSecondPrependedByte;
65
66 uint8_t m_gb18030First { 0x00 };
67 uint8_t m_gb18030Second { 0x00 };
68 uint8_t m_gb18030Third { 0x00 };
69
70 uint8_t m_lead { 0x00 };
71 std::optional<uint8_t> m_prependedByte;
72};
73
74} // namespace WebCore
