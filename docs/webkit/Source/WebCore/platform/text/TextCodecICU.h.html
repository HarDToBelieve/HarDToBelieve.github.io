35using ICUConverterPtr = std::unique_ptr<UConverter, ICUDeleter<ucnv_close>>;

37class TextCodecICU final : public TextCodec {
38public:
39 static void registerEncodingNames(EncodingNameRegistrar);
40 static void registerCodecs(TextCodecRegistrar);
41
42 explicit TextCodecICU(const char* encoding, const char* canonicalConverterName);
43 virtual ~TextCodecICU();
44
45private:
46 String decode(const char*, size_t length, bool flush, bool stopOnError, bool& sawError) final;
47 Vector<uint8_t> encode(StringView, UnencodableHandling) const final;
48
49 void createICUConverter() const;
50 void releaseICUConverter() const;
51
52 int decodeToBuffer(UChar* buffer, UChar* bufferLimit, const char*& source, const char* sourceLimit, int32_t* offsets, bool flush, UErrorCode&);
53
54 const char* const m_encodingName;
55 const char* const m_canonicalConverterName;
56 mutable ICUConverterPtr m_converter;
57};
