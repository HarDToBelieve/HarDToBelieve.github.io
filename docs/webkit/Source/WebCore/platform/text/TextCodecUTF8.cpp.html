87static inline int decodeNonASCIISequence(const uint8_t* sequence, int& length)
88{
89 ASSERT(!isASCII(sequence[0]));
90 if (length == 2) {
91 ASSERT(sequence[0] >= 0xC2);
92 ASSERT(sequence[0] <= 0xDF);
93 if (sequence[1] < 0x80 || sequence[1] > 0xBF) {
94 length = 1;
95 return nonCharacter;
96 }
97 return ((sequence[0] << 6) + sequence[1]) - 0x00003080;
98 }
99 if (length == 3) {
100 ASSERT(sequence[0] >= 0xE0);
101 ASSERT(sequence[0] <= 0xEF);
102 switch (sequence[0]) {
103 case 0xE0:
104 if (sequence[1] < 0xA0 || sequence[1] > 0xBF) {
105 length = 1;
106 return nonCharacter;
107 }
108 break;
109 case 0xED:
110 if (sequence[1] < 0x80 || sequence[1] > 0x9F) {
111 length = 1;
112 return nonCharacter;
113 }
114 break;
115 default:
116 if (sequence[1] < 0x80 || sequence[1] > 0xBF) {
117 length = 1;
118 return nonCharacter;
119 }
120 }
121 if (sequence[2] < 0x80 || sequence[2] > 0xBF) {
122 length = 2;
123 return nonCharacter;
124 }
125 return ((sequence[0] << 12) + (sequence[1] << 6) + sequence[2]) - 0x000E2080;
126 }
127 ASSERT(length == 4);
128 ASSERT(sequence[0] >= 0xF0);
129 ASSERT(sequence[0] <= 0xF4);
130 switch (sequence[0]) {
131 case 0xF0:
132 if (sequence[1] < 0x90 || sequence[1] > 0xBF) {
133 length = 1;
134 return nonCharacter;
135 }
136 break;
137 case 0xF4:
138 if (sequence[1] < 0x80 || sequence[1] > 0x8F) {
139 length = 1;
140 return nonCharacter;
141 }
142 break;
143 default:
144 if (sequence[1] < 0x80 || sequence[1] > 0xBF) {
145 length = 1;
146 return nonCharacter;
147 }
148 }
149 if (sequence[2] < 0x80 || sequence[2] > 0xBF) {
150 length = 2;
151 return nonCharacter;
152 }
153 if (sequence[3] < 0x80 || sequence[3] > 0xBF) {
154 length = 3;
155 return nonCharacter;
156 }
157 return ((sequence[0] << 18) + (sequence[1] << 12) + (sequence[2] << 6) + sequence[3]) - 0x03C82080;
158}

160static inline UChar* appendCharacter(UChar* destination, int character)
161{
162 ASSERT(character != nonCharacter);
163 ASSERT(!U_IS_SURROGATE(character));
164 if (U_IS_BMP(character))
165 *destination++ = character;
166 else {
167 *destination++ = U16_LEAD(character);
168 *destination++ = U16_TRAIL(character);
169 }
170 return destination;
171}

173void TextCodecUTF8::consumePartialSequenceByte()
174{
175 --m_partialSequenceSize;
176 memmove(m_partialSequence, m_partialSequence + 1, m_partialSequenceSize);

179bool TextCodecUTF8::handlePartialSequence(LChar*& destination, const uint8_t*& source, const uint8_t* end, bool flush)
180{
181 ASSERT(m_partialSequenceSize);
182 do {
183 if (isASCII(m_partialSequence[0])) {
184 *destination++ = m_partialSequence[0];
185 consumePartialSequenceByte();
186 continue;
187 }
188 int count = nonASCIISequenceLength(m_partialSequence[0]);
189 if (!count)
190 return true;
191
192 if (count > m_partialSequenceSize) {
193 if (count - m_partialSequenceSize > end - source) {
194 if (!flush) {
195 // The new data is not enough to complete the sequence, so
196 // add it to the existing partial sequence.
197 memcpy(m_partialSequence + m_partialSequenceSize, source, end - source);
198 m_partialSequenceSize += end - source;
199 return false;
200 }
201 // An incomplete partial sequence at the end is an error, but it will create
202 // a 16 bit string due to the replacementCharacter. Let the 16 bit path handle
203 // the error.
204 return true;
205 }
206 memcpy(m_partialSequence + m_partialSequenceSize, source, count - m_partialSequenceSize);
207 source += count - m_partialSequenceSize;
208 m_partialSequenceSize = count;
209 }
210 int character = decodeNonASCIISequence(m_partialSequence, count);
211 if (!isLatin1(character))
212 return true;
213
214 m_partialSequenceSize -= count;
215 *destination++ = character;
216 } while (m_partialSequenceSize);
217
218 return false;
219}

221void TextCodecUTF8::handlePartialSequence(UChar*& destination, const uint8_t*& source, const uint8_t* end, bool flush, bool stopOnError, bool& sawError)
222{
223 ASSERT(m_partialSequenceSize);
224 do {
225 if (isASCII(m_partialSequence[0])) {
226 *destination++ = m_partialSequence[0];
227 consumePartialSequenceByte();
228 continue;
229 }
230 int count = nonASCIISequenceLength(m_partialSequence[0]);
231 if (!count) {
232 sawError = true;
233 if (stopOnError)
234 return;
235 *destination++ = replacementCharacter;
236 consumePartialSequenceByte();
237 continue;
238 }
239 if (count > m_partialSequenceSize) {
240 if (count - m_partialSequenceSize > end - source) {
241 if (!flush) {
242 // The new data is not enough to complete the sequence, so
243 // add it to the existing partial sequence.
244 memcpy(m_partialSequence + m_partialSequenceSize, source, end - source);
245 m_partialSequenceSize += end - source;
246 return;
247 }
248 // An incomplete partial sequence at the end is an error.
249 sawError = true;
250 if (stopOnError)
251 return;
252 *destination++ = replacementCharacter;
253 m_partialSequenceSize = 0;
254 source = end;
255 continue;
256 }
257 memcpy(m_partialSequence + m_partialSequenceSize, source, count - m_partialSequenceSize);
258 source += count - m_partialSequenceSize;
259 m_partialSequenceSize = count;
260 }
261 int character = decodeNonASCIISequence(m_partialSequence, count);
262 if (character == nonCharacter) {
263 sawError = true;
264 if (stopOnError)
265 return;
266 *destination++ = replacementCharacter;
267 m_partialSequenceSize -= count;
268 memmove(m_partialSequence, m_partialSequence + count, m_partialSequenceSize);
269 continue;
270 }
271
272 m_partialSequenceSize -= count;
273 if (std::exchange(m_shouldStripByteOrderMark, false) && character == byteOrderMark)
274 continue;
275 destination = appendCharacter(destination, character);
276 } while (m_partialSequenceSize);
277}

279String TextCodecUTF8::decode(const char* bytes, size_t length, bool flush, bool stopOnError, bool& sawError)
280{
281 // Each input byte might turn into a character.
282 // That includes all bytes in the partial-sequence buffer because
283 // each byte in an invalid sequence will turn into a replacement character.
284 StringBuffer<LChar> buffer(m_partialSequenceSize + length);
285
286 const uint8_t* source = reinterpret_cast<const uint8_t*>(bytes);
287 const uint8_t* end = source + length;
288 const uint8_t* alignedEnd = WTF::alignToMachineWord(end);
289 LChar* destination = buffer.characters();
290
291 do {
292 if (m_partialSequenceSize) {
293 // Explicitly copy destination and source pointers to avoid taking pointers to the
294 // local variables, which may harm code generation by disabling some optimizations
295 // in some compilers.
296 LChar* destinationForHandlePartialSequence = destination;
297 const uint8_t* sourceForHandlePartialSequence = source;
298 if (handlePartialSequence(destinationForHandlePartialSequence, sourceForHandlePartialSequence, end, flush)) {
299 source = sourceForHandlePartialSequence;
300 goto upConvertTo16Bit;
301 }
302 destination = destinationForHandlePartialSequence;
303 source = sourceForHandlePartialSequence;
304 if (m_partialSequenceSize)
305 break;
306 }
307
308 while (source < end) {
309 if (isASCII(*source)) {
310 // Fast path for ASCII. Most UTF-8 text will be ASCII.
311 if (WTF::isAlignedToMachineWord(source)) {
312 while (source < alignedEnd) {
313 auto chunk = *reinterpret_cast_ptr<const WTF::MachineWord*>(source);
314 if (!WTF::isAllASCII<LChar>(chunk))
315 break;
316 copyASCIIMachineWord(destination, source);
317 source += sizeof(WTF::MachineWord);
318 destination += sizeof(WTF::MachineWord);
319 }
320 if (source == end)
321 break;
322 if (!isASCII(*source))
323 continue;
324 }
325 *destination++ = *source++;
326 continue;
327 }
328 int count = nonASCIISequenceLength(*source);
329 int character;
330 if (!count)
331 character = nonCharacter;
332 else {
333 if (count > end - source) {
334 ASSERT_WITH_SECURITY_IMPLICATION(end - source < static_cast<ptrdiff_t>(sizeof(m_partialSequence)));
335 ASSERT(!m_partialSequenceSize);
336 m_partialSequenceSize = end - source;
337 memcpy(m_partialSequence, source, m_partialSequenceSize);
338 source = end;
339 break;
340 }
341 character = decodeNonASCIISequence(source, count);
342 }
343 if (character == nonCharacter) {
344 sawError = true;
345 if (stopOnError)
346 break;
347
348 goto upConvertTo16Bit;
349 }
350 if (!isLatin1(character))
351 goto upConvertTo16Bit;
352
353 source += count;
354 *destination++ = character;
355 }
356 } while (flush && m_partialSequenceSize);
357
358 buffer.shrink(destination - buffer.characters());
359 if (flush)
360 m_partialSequenceSize = 0;
361 if (flush || buffer.length())
362 m_shouldStripByteOrderMark = false;
363 return String::adopt(WTFMove(buffer));
364
365upConvertTo16Bit:
366 StringBuffer<UChar> buffer16(m_partialSequenceSize + length);
367
368 UChar* destination16 = buffer16.characters();
369
370 // Copy the already converted characters
371 for (LChar* converted8 = buffer.characters(); converted8 < destination;)
372 *destination16++ = *converted8++;
373
374 do {
375 if (m_partialSequenceSize) {
376 // Explicitly copy destination and source pointers to avoid taking pointers to the
377 // local variables, which may harm code generation by disabling some optimizations
378 // in some compilers.
379 UChar* destinationForHandlePartialSequence = destination16;
380 const uint8_t* sourceForHandlePartialSequence = source;
381 handlePartialSequence(destinationForHandlePartialSequence, sourceForHandlePartialSequence, end, flush, stopOnError, sawError);
382 destination16 = destinationForHandlePartialSequence;
383 source = sourceForHandlePartialSequence;
384 if (m_partialSequenceSize)
385 break;
386 }
387
388 while (source < end) {
389 if (isASCII(*source)) {
390 // Fast path for ASCII. Most UTF-8 text will be ASCII.
391 if (WTF::isAlignedToMachineWord(source)) {
392 while (source < alignedEnd) {
393 auto chunk = *reinterpret_cast_ptr<const WTF::MachineWord*>(source);
394 if (!WTF::isAllASCII<LChar>(chunk))
395 break;
396 copyASCIIMachineWord(destination16, source);
397 source += sizeof(WTF::MachineWord);
398 destination16 += sizeof(WTF::MachineWord);
399 }
400 if (source == end)
401 break;
402 if (!isASCII(*source))
403 continue;
404 }
405 *destination16++ = *source++;
406 continue;
407 }
408 int count = nonASCIISequenceLength(*source);
409 int character;
410 if (!count)
411 character = nonCharacter;
412 else {
413 if (count > end - source) {
414 ASSERT_WITH_SECURITY_IMPLICATION(end - source < static_cast<ptrdiff_t>(sizeof(m_partialSequence)));
415 ASSERT(!m_partialSequenceSize);
416 m_partialSequenceSize = end - source;
417 memcpy(m_partialSequence, source, m_partialSequenceSize);
418 source = end;
419 break;
420 }
421 character = decodeNonASCIISequence(source, count);
422 }
423 if (character == nonCharacter) {
424 sawError = true;
425 if (stopOnError)
426 break;
427 *destination16++ = replacementCharacter;
428 source += count ? count : 1;
429 continue;
430 }
431 source += count;
432 if (character == byteOrderMark && destination16 == buffer16.characters() && std::exchange(m_shouldStripByteOrderMark, false))
433 continue;
434 destination16 = appendCharacter(destination16, character);
435 }
436 } while (flush && m_partialSequenceSize);
437
438 buffer16.shrink(destination16 - buffer16.characters());
439 if (flush)
440 m_partialSequenceSize = 0;
441 if (flush || buffer16.length())
442 m_shouldStripByteOrderMark = false;
443 return String::adopt(WTFMove(buffer16));
444}

446Vector<uint8_t> TextCodecUTF8::encodeUTF8(StringView string, UnencodableHandling)
447{
448 // The maximum number of UTF-8 bytes needed per UTF-16 code unit is 3.
449 // BMP characters take only one UTF-16 code unit and can take up to 3 bytes (3x).
450 // Non-BMP characters take two UTF-16 code units and can take up to 4 bytes (2x).
451 Vector<uint8_t> bytes(WTF::checkedProduct<size_t>(string.length(), 3));
452 size_t bytesWritten = 0;
453 for (auto character : string.codePoints())
454 U8_APPEND_UNSAFE(bytes.data(), bytesWritten, character);
455 bytes.shrink(bytesWritten);
456 return bytes;
457}
458
459Vector<uint8_t> TextCodecUTF8::encode(StringView string, UnencodableHandling unencodableHandling) const
460{
461 return encodeUTF8(string, unencodableHandling);
462}
