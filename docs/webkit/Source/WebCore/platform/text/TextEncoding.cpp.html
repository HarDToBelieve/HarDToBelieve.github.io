58String TextEncoding::decode(const char* data, size_t length, bool stopOnError, bool& sawError) const
59{
60 if (!m_name)
61 return String();
62
63 return newTextCodec(*this)->decode(data, length, true, stopOnError, sawError);
64}
65
66Vector<uint8_t> TextEncoding::encode(StringView string, UnencodableHandling handling, NFCNormalize normalize) const
67{
68 if (!m_name || string.isEmpty())
69 return { };
70
71 // FIXME: What's the right place to do normalization?
72 // It's a little strange to do it inside the encode function.
73 // Perhaps normalization should be an explicit step done before calling encode.
74 if (normalize == NFCNormalize::Yes)
75 return newTextCodec(*this)->encode(normalizedNFC(string).view, handling);
76 return newTextCodec(*this)->encode(string, handling);
77}

79const char* TextEncoding::domName() const
80{
81 if (noExtendedTextEncodingNameUsed())
82 return m_name;
83
84 // We treat EUC-KR as windows-949 (its superset), but need to expose
85 // the name 'EUC-KR' because the name 'windows-949' is not recognized by
86 // most Korean web servers even though they do use the encoding
87 // 'windows-949' with the name 'EUC-KR'.
88 // FIXME: This is not thread-safe. At the moment, this function is
89 // only accessed in a single thread, but eventually has to be made
90 // thread-safe along with usesVisualOrdering().
91 static const char* const a = atomCanonicalTextEncodingName("windows-949");
92 if (m_name == a)
93 return "EUC-KR";
94 return m_name;
95}

97bool TextEncoding::usesVisualOrdering() const
98{
99 if (noExtendedTextEncodingNameUsed())
100 return false;
101
102 static const char* const a = atomCanonicalTextEncodingName("ISO-8859-8");
103 return m_name == a;
104}

116bool TextEncoding::isNonByteBasedEncoding() const
117{
118 return *this == UTF16LittleEndianEncoding() || *this == UTF16BigEndianEncoding();
119}
120
121bool TextEncoding::isUTF7Encoding() const
122{
123 if (noExtendedTextEncodingNameUsed())
124 return false;
125
126 return *this == UTF7Encoding();
127}

129const TextEncoding& TextEncoding::closestByteBasedEquivalent() const
130{
131 if (isNonByteBasedEncoding())
132 return UTF8Encoding();
133 return *this;
134}

136// HTML5 specifies that UTF-8 be used in form submission when a form is
137// is a part of a document in UTF-16 probably because UTF-16 is not a
138// byte-based encoding and can contain 0x00. By extension, the same
139// should be done for UTF-32. In case of UTF-7, it is a byte-based encoding,
140// but it's fraught with problems and we'd rather steer clear of it.
141const TextEncoding& TextEncoding::encodingForFormSubmissionOrURLParsing() const
142{
143 if (isNonByteBasedEncoding() || isUTF7Encoding())
144 return UTF8Encoding();
145 return *this;
146}

186{
