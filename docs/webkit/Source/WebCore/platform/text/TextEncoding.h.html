26#pragma once
27
28#include <pal/text/UnencodableHandling.h>
29#include <wtf/URL.h>
30#include <wtf/text/StringView.h>
31
32namespace WebCore {
33
34enum class NFCNormalize : bool { No, Yes };
35
36class TextEncoding : public WTF::URLTextEncoding {
37public:
38 TextEncoding() = default;
39 WEBCORE_EXPORT TextEncoding(const char* name);
40 WEBCORE_EXPORT TextEncoding(const String& name);
41
42 bool isValid() const { return m_name; }
43 const char* name() const { return m_name; }
44 WEBCORE_EXPORT const char* domName() const; // name exposed via DOM
45 bool usesVisualOrdering() const;
46 bool isJapanese() const;
47
48 const TextEncoding& closestByteBasedEquivalent() const;
49 const TextEncoding& encodingForFormSubmissionOrURLParsing() const;
50
51 WEBCORE_EXPORT String decode(const char*, size_t length, bool stopOnError, bool& sawError) const;
52 String decode(const char*, size_t length) const;
53 String decode(const uint8_t* data, size_t length) const { return decode(reinterpret_cast<const char*>(data), length); }
54 WEBCORE_EXPORT Vector<uint8_t> encode(StringView, UnencodableHandling, NFCNormalize = NFCNormalize::Yes) const;
55 Vector<uint8_t> encodeForURLParsing(StringView string) const final { return encode(string, UnencodableHandling::URLEncodedEntities, NFCNormalize::No); }
56
57 UChar backslashAsCurrencySymbol() const;
58 bool isByteBasedEncoding() const { return !isNonByteBasedEncoding(); }
59
60private:
61 bool isNonByteBasedEncoding() const;
62 bool isUTF7Encoding() const;
63
64 const char* m_name { nullptr };
65 UChar m_backslashAsCurrencySymbol;
66};
67
68inline bool operator==(const TextEncoding& a, const TextEncoding& b) { return a.name() == b.name(); }
69inline bool operator!=(const TextEncoding& a, const TextEncoding& b) { return a.name() != b.name(); }
70
71const TextEncoding& ASCIIEncoding();
72const TextEncoding& Latin1Encoding();
73const TextEncoding& UTF16BigEndianEncoding();
74const TextEncoding& UTF16LittleEndianEncoding();
75WEBCORE_EXPORT const TextEncoding& UTF8Encoding();
76WEBCORE_EXPORT const TextEncoding& WindowsLatin1Encoding();
77
78// Unescapes the given string using URL escaping rules.
79// DANGER: If the URL has "%00" in it,
80// the resulting string will have embedded null characters!
81WEBCORE_EXPORT String decodeURLEscapeSequences(StringView, const TextEncoding& = UTF8Encoding());
82
83inline String TextEncoding::decode(const char* characters, size_t length) const
84{
85 bool ignored;
86 return decode(characters, length, false, ignored);
87}
