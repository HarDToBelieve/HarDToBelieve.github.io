35namespace WTF {
36
37template<>
38class TinyLRUCachePolicy<AtomString, RetainPtr<CFLocaleRef>>
39{
40public:
41 static TinyLRUCache<AtomString, RetainPtr<CFLocaleRef>>& cache()
42 {
43 static NeverDestroyed<TinyLRUCache<AtomString, RetainPtr<CFLocaleRef>>> cache;
44 return cache;
45 }
46
47 static bool isKeyNull(const AtomString& localeIdentifier)
48 {
49 return localeIdentifier.isNull();
50 }
51
52 static RetainPtr<CFLocaleRef> createValueForNullKey()
53 {
54 // CF hyphenation functions use locale (regional formats) language, which doesn't necessarily match primary UI language,
55 // so we can't use default locale here. See <rdar://problem/14897664>.
56 RetainPtr<CFLocaleRef> locale = adoptCF(CFLocaleCreate(kCFAllocatorDefault, defaultLanguage().createCFString().get()));
57 return CFStringIsHyphenationAvailableForLocale(locale.get()) ? locale : nullptr;
58 }
59
60 static RetainPtr<CFLocaleRef> createValueForKey(const AtomString& localeIdentifier)
61 {
62 RetainPtr<CFLocaleRef> locale = adoptCF(CFLocaleCreate(kCFAllocatorDefault, localeIdentifier.string().createCFString().get()));
63
64 return CFStringIsHyphenationAvailableForLocale(locale.get()) ? locale : nullptr;
65 }
66
67 static AtomString createKeyForStorage(const AtomString& key) { return key; }
68};
69}

71namespace WebCore {
72
73bool canHyphenate(const AtomString& localeIdentifier)
74{
75 return TinyLRUCachePolicy<AtomString, RetainPtr<CFLocaleRef>>::cache().get(localeIdentifier);
76}
77
78size_t lastHyphenLocation(StringView text, size_t beforeIndex, const AtomString& localeIdentifier)
79{
80 RetainPtr<CFLocaleRef> locale = TinyLRUCachePolicy<AtomString, RetainPtr<CFLocaleRef>>::cache().get(localeIdentifier);
81
82 CFOptionFlags searchAcrossWordBoundaries = 1;
83 CFIndex result = CFStringGetHyphenationLocationBeforeIndex(text.createCFStringWithoutCopying().get(), beforeIndex, CFRangeMake(0, text.length()), searchAcrossWordBoundaries, locale.get(), nullptr);
84 return result == kCFNotFound ? 0 : result;
85}
