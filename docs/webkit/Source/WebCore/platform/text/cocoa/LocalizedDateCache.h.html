26#pragma once
27
28#include "FontCascade.h"
29#include <wtf/HashMap.h>
30#include <wtf/RetainPtr.h>
31
32namespace WebCore {
33
34enum class DateComponentsType : uint8_t;
35
36class MeasureTextClient {
37public:
38 virtual float measureText(const String&) const = 0;
39 virtual ~MeasureTextClient() = default;
40};
41
42class LocalizedDateCache {
43public:
44 NSDateFormatter *formatterForDateType(DateComponentsType);
45 float maximumWidthForDateType(DateComponentsType, const FontCascade&, const MeasureTextClient&);
46 void localeChanged();
47
48private:
49 LocalizedDateCache();
50 ~LocalizedDateCache();
51
52 RetainPtr<NSDateFormatter> createFormatterForType(DateComponentsType);
53 float calculateMaximumWidth(DateComponentsType, const MeasureTextClient&);
54
55 // Using int instead of DateComponentsType for the key because the enum
56 // does not have a default hash and hash traits. Usage of the maps
57 // casts the DateComponents::Type into an int as the key.
58 typedef HashMap<int, RetainPtr<NSDateFormatter>> DateTypeFormatterMap;
59 typedef HashMap<int, float> DateTypeMaxWidthMap;
60 DateTypeFormatterMap m_formatterMap;
61 DateTypeMaxWidthMap m_maxWidthMap;
62 FontCascade m_font;
63
64 friend LocalizedDateCache& localizedDateCache();
65 friend NeverDestroyed<LocalizedDateCache>;
66};
67
68// Singleton.
69LocalizedDateCache& localizedDateCache();
70
71} // namespace WebCore
