41LocalizedDateCache& localizedDateCache()
42{
43 static NeverDestroyed<LocalizedDateCache> cache;
44 return cache;
45}
46
47static void _localeChanged(CFNotificationCenterRef, void*, CFStringRef, const void*, CFDictionaryRef)
48{
49 localizedDateCache().localeChanged();
50}
51
52LocalizedDateCache::LocalizedDateCache()
53{
54 // Listen to CF Notifications for locale change, and clear the cache when it does.
55 CFNotificationCenterAddObserver(CFNotificationCenterGetLocalCenter(), (void*)this, _localeChanged, kCFLocaleCurrentLocaleDidChangeNotification, NULL, CFNotificationSuspensionBehaviorDeliverImmediately);
56}
57
58LocalizedDateCache::~LocalizedDateCache()
59{
60 // NOTE: Singleton does not expect to be deconstructed.
61 CFNotificationCenterRemoveObserver(CFNotificationCenterGetLocalCenter(), (void*)this, kCFLocaleCurrentLocaleDidChangeNotification, NULL);
62}

64void LocalizedDateCache::localeChanged()
65{
66 m_maxWidthMap.clear();
67 m_formatterMap.clear();
68}

70NSDateFormatter *LocalizedDateCache::formatterForDateType(DateComponentsType type)
71{
72 int key = static_cast<int>(type);
73 if (m_formatterMap.contains(key))
74 return m_formatterMap.get(key).get();
75
76 auto dateFormatter = createFormatterForType(type);
77 m_formatterMap.set(key, dateFormatter.get());
78 return dateFormatter.autorelease();
79}

81float LocalizedDateCache::maximumWidthForDateType(DateComponentsType type, const FontCascade& font, const MeasureTextClient& measurer)
82{
83 int key = static_cast<int>(type);
84 if (m_font == font) {
85 if (m_maxWidthMap.contains(key))
86 return m_maxWidthMap.get(key);
87 } else {
88 m_font = FontCascade(font);
89 m_maxWidthMap.clear();
90 }
91
92 float calculatedMaximum = calculateMaximumWidth(type, measurer);
93 m_maxWidthMap.set(key, calculatedMaximum);
94 return calculatedMaximum;
95}

97RetainPtr<NSDateFormatter> LocalizedDateCache::createFormatterForType(DateComponentsType type)
98{
99 auto dateFormatter = adoptNS([[NSDateFormatter alloc] init]);
100 NSLocale *currentLocale = [NSLocale currentLocale];
101 [dateFormatter setLocale:currentLocale];
102
103 switch (type) {
104 case DateComponentsType::Invalid:
105 ASSERT_NOT_REACHED();
106 break;
107 case DateComponentsType::Date:
108 [dateFormatter setTimeZone:[NSTimeZone timeZoneForSecondsFromGMT:0]];
109 [dateFormatter setTimeStyle:NSDateFormatterNoStyle];
110 [dateFormatter setDateStyle:NSDateFormatterMediumStyle];
111 break;
112 case DateComponentsType::DateTimeLocal:
113 [dateFormatter setTimeZone:[NSTimeZone timeZoneForSecondsFromGMT:0]];
114 [dateFormatter setTimeStyle:NSDateFormatterShortStyle];
115 [dateFormatter setDateStyle:NSDateFormatterMediumStyle];
116 break;
117 case DateComponentsType::Month:
118 [dateFormatter setTimeZone:[NSTimeZone timeZoneForSecondsFromGMT:0]];
119 [dateFormatter setDateFormat:[NSDateFormatter dateFormatFromTemplate:@"MMMMyyyy" options:0 locale:currentLocale]];
120 break;
121 case DateComponentsType::Time:
122 [dateFormatter setTimeZone:[NSTimeZone timeZoneForSecondsFromGMT:0]];
123 [dateFormatter setTimeStyle:NSDateFormatterShortStyle];
124 [dateFormatter setDateStyle:NSDateFormatterNoStyle];
125 break;
126 case DateComponentsType::Week:
127 ASSERT_NOT_REACHED();
128 break;
129 }
130
131 return dateFormatter;
132}

134// NOTE: This does not check for the widest day of the week.
135// We assume no formatter option shows that information.
136float LocalizedDateCache::calculateMaximumWidth(DateComponentsType type, const MeasureTextClient& measurer)
137{
138 float maximumWidth = 0;
139
140 // Get the formatter we would use, copy it because we will force its time zone to be UTC.
141 auto dateFormatter = adoptNS([formatterForDateType(type) copy]);
142 [dateFormatter setTimeZone:[NSTimeZone timeZoneForSecondsFromGMT:0]];
143
144 // Sample date with a 4 digit year and 2 digit day, hour, and minute. Digits are
145 // typically all equally wide. Force UTC timezone for the test date below so the
146 // date doesn't adjust for the current timezone. This is an arbitrary date
147 // (x-27-2007) and time (10:45 PM).
148 RetainPtr<NSCalendar> gregorian = adoptNS([[NSCalendar alloc] initWithCalendarIdentifier:NSCalendarIdentifierGregorian]);
149 [gregorian.get() setTimeZone:[NSTimeZone timeZoneForSecondsFromGMT:0]];
150 RetainPtr<NSDateComponents> components = adoptNS([[NSDateComponents alloc] init]);
151 [components.get() setDay:27];
152 [components.get() setYear:2007];
153 [components.get() setHour:22];
154 [components.get() setMinute:45];
155
156 static const NSUInteger numberOfGregorianMonths = [[dateFormatter monthSymbols] count];
157 ASSERT(numberOfGregorianMonths == 12);
158
159 // For each month (in the Gregorian Calendar), format a date and measure its length.
160 NSUInteger totalMonthsToTest = 1;
161 if (type == DateComponentsType::Date
162 || type == DateComponentsType::DateTimeLocal
163 || type == DateComponentsType::Month)
164 totalMonthsToTest = numberOfGregorianMonths;
165 for (NSUInteger i = 0; i < totalMonthsToTest; ++i) {
166 [components.get() setMonth:(i + 1)];
167 NSDate *date = [gregorian.get() dateFromComponents:components.get()];
168 NSString *formattedDate = [dateFormatter stringFromDate:date];
169 maximumWidth = max(maximumWidth, measurer.measureText(String(formattedDate)));
170 }
171
172 return maximumWidth;
173}
