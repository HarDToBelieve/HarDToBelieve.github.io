33namespace WebCore {
34
35NSTextCheckingTypes nsTextCheckingTypes(OptionSet<TextCheckingType> types)
36{
37 NSTextCheckingTypes mask = 0;
38 if (types.contains(TextCheckingType::Spelling))
39 mask |= NSTextCheckingTypeSpelling;
40 if (types.contains(TextCheckingType::Grammar))
41 mask |= NSTextCheckingTypeGrammar;
42 if (types.contains(TextCheckingType::Link))
43 mask |= NSTextCheckingTypeLink;
44 if (types.contains(TextCheckingType::Quote))
45 mask |= NSTextCheckingTypeQuote;
46 if (types.contains(TextCheckingType::Dash))
47 mask |= NSTextCheckingTypeDash;
48 if (types.contains(TextCheckingType::Replacement))
49 mask |= NSTextCheckingTypeReplacement;
50 if (types.contains(TextCheckingType::Correction))
51 mask |= NSTextCheckingTypeCorrection;
52 return mask;
53}
