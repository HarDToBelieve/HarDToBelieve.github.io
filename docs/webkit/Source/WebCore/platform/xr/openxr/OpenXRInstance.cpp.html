33struct Instance::Impl {
34 WTF_MAKE_STRUCT_FAST_ALLOCATED;
35
36 Impl();
37 ~Impl();
38
39 XrInstance xrInstance() const { return m_instance; }
40 WorkQueue& queue() const { return m_workQueue; }
41 OpenXRExtensions* extensions() { return m_extensions.get(); }
42
43private:
44 XrInstance m_instance { XR_NULL_HANDLE };
45 Ref<WorkQueue> m_workQueue;
46 std::unique_ptr<OpenXRExtensions> m_extensions;
47};
48
49Instance::Impl::Impl()
50 : m_workQueue(WorkQueue::create("OpenXR queue"))
51{
52 m_workQueue->dispatch([this]() {
53 LOG(XR, "OpenXR: initializing\n");
54
55 m_extensions = OpenXRExtensions::create();
56 if (!m_extensions)
57 return;
58
59 static const char* s_applicationName = "WebXR (WebKit)";
60 static const uint32_t s_applicationVersion = 1;
61
62 const char* const enabledExtensions[] = {
63 XR_KHR_OPENGL_ENABLE_EXTENSION_NAME,
64 XR_MNDX_EGL_ENABLE_EXTENSION_NAME
65 };
66
67 auto createInfo = createStructure<XrInstanceCreateInfo, XR_TYPE_INSTANCE_CREATE_INFO>();
68 createInfo.createFlags = 0;
69 std::memcpy(createInfo.applicationInfo.applicationName, s_applicationName, XR_MAX_APPLICATION_NAME_SIZE);
70 createInfo.applicationInfo.apiVersion = XR_CURRENT_API_VERSION;
71 createInfo.applicationInfo.applicationVersion = s_applicationVersion;
72 createInfo.enabledApiLayerCount = 0;
73 createInfo.enabledExtensionCount = WTF_ARRAY_LENGTH(enabledExtensions);
74 createInfo.enabledExtensionNames = enabledExtensions;
75
76 XrResult result = xrCreateInstance(&createInfo, &m_instance);
77 RETURN_IF_FAILED(result, "xrCreateInstance()", m_instance);
78 if (m_instance == XR_NULL_HANDLE)
79 return;
80
81 m_extensions->loadMethods(m_instance);
82
83 LOG(XR, "xrCreateInstance(): using instance %p\n", m_instance);
84 });
85}
86
87Instance::Impl::~Impl()
88{
89 m_workQueue->dispatch([instance = m_instance] {
90 if (instance != XR_NULL_HANDLE)
91 xrDestroyInstance(instance);
92 });
93}

95Instance& Instance::singleton()
96{
97 static LazyNeverDestroyed<Instance> s_instance;
98 static std::once_flag s_onceFlag;
99 std::call_once(s_onceFlag,
100 [&] {
101 s_instance.construct();
102 });
103 return s_instance.get();
104}

106Instance::Instance()
107 : m_impl(makeUniqueRef<Impl>())
108{
109}

111void Instance::enumerateImmersiveXRDevices(CompletionHandler<void(const DeviceList& devices)>&& callback)
112{
113 m_impl->queue().dispatch([this, callback = WTFMove(callback)]() mutable {
114 auto callbackOnExit = makeScopeExit([&]() {
115 callOnMainThread([callback = WTFMove(callback)]() mutable {
116 callback({ });
117 });
118 });
119
120 if (m_impl->xrInstance() == XR_NULL_HANDLE || !m_impl->extensions()) {
121 LOG(XR, "%s Unable to enumerate XR devices. No XrInstance present\n", __FUNCTION__);
122 return;
123 }
124
125 auto systemGetInfo = createStructure<XrSystemGetInfo, XR_TYPE_SYSTEM_GET_INFO>();
126 systemGetInfo.formFactor = XR_FORM_FACTOR_HEAD_MOUNTED_DISPLAY;
127
128 XrSystemId systemId;
129 XrResult result = xrGetSystem(m_impl->xrInstance(), &systemGetInfo, &systemId);
130 RETURN_IF_FAILED(result, "xrGetSystem", m_impl->xrInstance());
131
132 callbackOnExit.release();
133
134 callOnMainThread([this, callback = WTFMove(callback), systemId]() mutable {
135 m_immersiveXRDevices = DeviceList::from(OpenXRDevice::create(m_impl->xrInstance(), systemId, m_impl->queue(), *m_impl->extensions(), [this, callback = WTFMove(callback)]() mutable {
136 ASSERT(isMainThread());
137 callback(m_immersiveXRDevices);
138 }));
139 });
140 });
141}
