20#include "config.h"
21#include "DOMMimeTypeArray.h"
22
23#include "DOMMimeType.h"
24#include "Navigator.h"
25#include <wtf/IsoMallocInlines.h>
26#include <wtf/text/AtomString.h>
27
28namespace WebCore {
29
30WTF_MAKE_ISO_ALLOCATED_IMPL(DOMMimeTypeArray);
31
32Ref<DOMMimeTypeArray> DOMMimeTypeArray::create(Navigator& navigator, Vector<Ref<DOMMimeType>>&& types)
33{
34 return adoptRef(*new DOMMimeTypeArray(navigator, WTFMove(types)));
35}
36
37DOMMimeTypeArray::DOMMimeTypeArray(Navigator& navigator, Vector<Ref<DOMMimeType>>&& types)
38 : m_navigator(makeWeakPtr(navigator))
39 , m_types(WTFMove(types))
40{
41}
42
43DOMMimeTypeArray::~DOMMimeTypeArray() = default;
44
45unsigned DOMMimeTypeArray::length() const
46{
47 return m_types.size();
48}
49
50RefPtr<DOMMimeType> DOMMimeTypeArray::item(unsigned index)
51{
52 if (index >= m_types.size())
53 return nullptr;
54 return m_types[index].ptr();
55}
56
57RefPtr<DOMMimeType> DOMMimeTypeArray::namedItem(const AtomString& propertyName)
58{
59 for (auto& type : m_types) {
60 if (type->type() == propertyName)
61 return type.ptr();
62 }
63 return nullptr;
64}
65
66Vector<AtomString> DOMMimeTypeArray::supportedPropertyNames()
67{
68 Vector<AtomString> result;
69 result.reserveInitialCapacity(m_types.size());
70 for (auto& type : m_types)
71 result.uncheckedAppend(type->type());
72 return result;
73}
