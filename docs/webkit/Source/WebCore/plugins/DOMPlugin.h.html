49 Vector<AtomString> supportedPropertyNames();
50
51 const Vector<Ref<DOMMimeType>>& mimeTypes() const { return m_mimeTypes; }
52
53 Navigator* navigator() { return m_navigator.get(); }
54
55private:
56 DOMPlugin(Navigator&, const PluginInfo&);

59 PluginInfo m_info;
60 Vector<Ref<DOMMimeType>> m_mimeTypes;
