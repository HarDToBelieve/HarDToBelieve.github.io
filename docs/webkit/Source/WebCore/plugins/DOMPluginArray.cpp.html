33Ref<DOMPluginArray> DOMPluginArray::create(Navigator& navigator, Vector<Ref<DOMPlugin>>&& publiclyVisiblePlugins, Vector<Ref<DOMPlugin>>&& additionalWebVisibilePlugins)
34{
35 return adoptRef(*new DOMPluginArray(navigator, WTFMove(publiclyVisiblePlugins), WTFMove(additionalWebVisibilePlugins)));
36}
37
38DOMPluginArray::DOMPluginArray(Navigator& navigator, Vector<Ref<DOMPlugin>>&& publiclyVisiblePlugins, Vector<Ref<DOMPlugin>>&& additionalWebVisibilePlugins)
39 : m_navigator(makeWeakPtr(navigator))
40 , m_publiclyVisiblePlugins(WTFMove(publiclyVisiblePlugins))
41 , m_additionalWebVisibilePlugins(WTFMove(additionalWebVisibilePlugins))
42{
43}
44
45DOMPluginArray::~DOMPluginArray() = default;
46
47unsigned DOMPluginArray::length() const
48{
49 return m_publiclyVisiblePlugins.size();
50}
51
52RefPtr<DOMPlugin> DOMPluginArray::item(unsigned index)
53{
54 if (index >= m_publiclyVisiblePlugins.size())
55 return nullptr;
56 return m_publiclyVisiblePlugins[index].ptr();
57}
58
59RefPtr<DOMPlugin> DOMPluginArray::namedItem(const AtomString& propertyName)
60{
61 for (auto& plugin : m_publiclyVisiblePlugins) {
62 if (plugin->name() == propertyName)
63 return plugin.ptr();
64 }
65
66 for (auto& plugin : m_additionalWebVisibilePlugins) {
67 if (plugin->name() == propertyName)
68 return plugin.ptr();
69 }
70
71 return nullptr;
72}
73
74Vector<AtomString> DOMPluginArray::supportedPropertyNames()
75{
76 Vector<AtomString> result;
77 result.reserveInitialCapacity(m_publiclyVisiblePlugins.size());
78 for (auto& plugin : m_publiclyVisiblePlugins)
79 result.uncheckedAppend(plugin->name());
80 return result;
81}
82
83void DOMPluginArray::refresh(bool reloadPages)
84{
85 if (!m_navigator)
86 return;
87
88 auto* frame = m_navigator->frame();
89 if (!frame)
90 return;
91
92 if (!frame->page())
93 return;
94
95 Page::refreshPlugins(reloadPages);
96}
