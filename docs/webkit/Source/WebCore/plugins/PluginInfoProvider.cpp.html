36PluginInfoProvider::~PluginInfoProvider()
37{
38 ASSERT(m_pages.computesEmpty());
39}
40
41void PluginInfoProvider::clearPagesPluginData()
42{
43 for (auto& page : m_pages)
44 page.clearPluginData();
45}
46
47void PluginInfoProvider::refresh(bool reloadPages)
48{
49 refreshPlugins();
50
51 Vector<Ref<Frame>> framesNeedingReload;
52
53 for (auto& page : m_pages) {
54 page.clearPluginData();
55
56 if (!reloadPages)
57 continue;
58
59 for (auto* frame = &page.mainFrame(); frame; frame = frame->tree().traverseNext()) {
60 if (frame->loader().subframeLoader().containsPlugins())
61 framesNeedingReload.append(page.mainFrame());
62 }
63 }
64
65 for (auto& frame : framesNeedingReload)
66 frame->loader().reload();
67}
68
69void PluginInfoProvider::addPage(Page& page)
70{
71 ASSERT(!m_pages.contains(page));
72
73 m_pages.add(page);
74}
75
76void PluginInfoProvider::removePage(Page& page)
77{
78 ASSERT(m_pages.contains(page));
79
80 m_pages.remove(page);
81}
