45void AutoTableLayout::recalcColumn(unsigned effCol)
46{
47 Layout& columnLayout = m_layoutStruct[effCol];
48
49 RenderTableCell* fixedContributor = nullptr;
50 RenderTableCell* maxContributor = nullptr;
51
52 for (auto& child : childrenOfType<RenderObject>(*m_table)) {
53 if (is<RenderTableCol>(child)) {
54 // RenderTableCols don't have the concept of preferred logical width, but we need to clear their dirty bits
55 // so that if we call setPreferredWidthsDirty(true) on a col or one of its descendants, we'll mark it's
56 // ancestors as dirty.
57 downcast<RenderTableCol>(child).clearPreferredLogicalWidthsDirtyBits();
58 } else if (is<RenderTableSection>(child)) {
59 auto& section = downcast<RenderTableSection>(child);
60 unsigned numRows = section.numRows();
61 for (unsigned i = 0; i < numRows; ++i) {
62 RenderTableSection::CellStruct current = section.cellAt(i, effCol);
63 RenderTableCell* cell = current.primaryCell();
64
65 if (current.inColSpan || !cell)
66 continue;
67
68 bool cellHasContent = cell->firstChild() || cell->style().hasBorder() || cell->style().hasPadding() || cell->style().hasBackground();
69 if (cellHasContent)
70 columnLayout.emptyCellsOnly = false;
71
72 // A cell originates in this column. Ensure we have
73 // a min/max width of at least 1px for this column now.
74 columnLayout.minLogicalWidth = std::max(columnLayout.minLogicalWidth, 0.f);
75 columnLayout.maxLogicalWidth = std::max(columnLayout.maxLogicalWidth, 0.f);
76
77 if (cell->colSpan() == 1) {
78 columnLayout.minLogicalWidth = std::max(cell->minPreferredLogicalWidth().ceilToFloat(), columnLayout.minLogicalWidth);
79 float maxPreferredWidth = cell->maxPreferredLogicalWidth().ceilToFloat();
80 if (maxPreferredWidth > columnLayout.maxLogicalWidth) {
81 columnLayout.maxLogicalWidth = maxPreferredWidth;
82 maxContributor = cell;
83 }
84
85 // All browsers implement a size limit on the cell's max width.
86 // Our limit is based on KHTML's representation that used 16 bits widths.
87 // FIXME: Other browsers have a lower limit for the cell's max width.
88 const float cCellMaxWidth = 32760;
89 Length cellLogicalWidth = cell->styleOrColLogicalWidth();
90 if (cellLogicalWidth.value() > cCellMaxWidth)
91 cellLogicalWidth.setValue(LengthType::Fixed, cCellMaxWidth);
92 if (cellLogicalWidth.isNegative())
93 cellLogicalWidth.setValue(LengthType::Fixed, 0);
94 switch (cellLogicalWidth.type()) {
95 case LengthType::Fixed:
96 // ignore width=0
97 if (cellLogicalWidth.isPositive() && !columnLayout.logicalWidth.isPercentOrCalculated()) {
98 float logicalWidth = cell->adjustBorderBoxLogicalWidthForBoxSizing(cellLogicalWidth);
99 if (columnLayout.logicalWidth.isFixed()) {
100 // Nav/IE weirdness
101 if ((logicalWidth > columnLayout.logicalWidth.value())
102 || ((columnLayout.logicalWidth.value() == logicalWidth) && (maxContributor == cell))) {
103 columnLayout.logicalWidth.setValue(LengthType::Fixed, logicalWidth);
104 fixedContributor = cell;
105 }
106 } else {
107 columnLayout.logicalWidth.setValue(LengthType::Fixed, logicalWidth);
108 fixedContributor = cell;
109 }
110 }
111 break;
112 case LengthType::Percent:
113 m_hasPercent = true;
114 if (cellLogicalWidth.isPositive() && (!columnLayout.logicalWidth.isPercent() || cellLogicalWidth.percent() > columnLayout.logicalWidth.percent()))
115 columnLayout.logicalWidth = cellLogicalWidth;
116 break;
117 case LengthType::Relative:
118 // FIXME: Need to understand this case and whether it makes sense to compare values
119 // which are not necessarily of the same type.
120 if (cellLogicalWidth.value() > columnLayout.logicalWidth.value())
121 columnLayout.logicalWidth = cellLogicalWidth;
122 break;
123 default:
124 break;
125 }
126 } else if (!effCol || section.primaryCellAt(i, effCol - 1) != cell) {
127 // This spanning cell originates in this column. Insert the cell into spanning cells list.
128 insertSpanCell(cell);
129 }
130 }
131 }
132 }
133
134 // Nav/IE weirdness
135 if (columnLayout.logicalWidth.isFixed()) {
136 if (m_table->document().inQuirksMode() && columnLayout.maxLogicalWidth > columnLayout.logicalWidth.value() && fixedContributor != maxContributor) {
137 columnLayout.logicalWidth = Length();
138 fixedContributor = nullptr;
139 }
140 }
141
142 columnLayout.maxLogicalWidth = std::max(columnLayout.maxLogicalWidth, columnLayout.minLogicalWidth);
143}

145void AutoTableLayout::fullRecalc()
146{
147 m_hasPercent = false;
148 m_effectiveLogicalWidthDirty = true;
149
150 unsigned nEffCols = m_table->numEffCols();
151 m_layoutStruct.resizeToFit(nEffCols);
152 m_layoutStruct.fill(Layout());
153 m_spanCells.fill(0);
154
155 Length groupLogicalWidth;
156 unsigned currentColumn = 0;
157 for (RenderTableCol* column = m_table->firstColumn(); column; column = column->nextColumn()) {
158 if (column->isTableColumnGroupWithColumnChildren())
159 groupLogicalWidth = column->style().logicalWidth();
160 else {
161 Length colLogicalWidth = column->style().logicalWidth();
162 if (colLogicalWidth.isAuto())
163 colLogicalWidth = groupLogicalWidth;
164 if ((colLogicalWidth.isFixed() || colLogicalWidth.isPercentOrCalculated()) && colLogicalWidth.isZero())
165 colLogicalWidth = Length();
166 unsigned effCol = m_table->colToEffCol(currentColumn);
167 unsigned span = column->span();
168 if (!colLogicalWidth.isAuto() && span == 1 && effCol < nEffCols && m_table->spanOfEffCol(effCol) == 1) {
169 m_layoutStruct[effCol].logicalWidth = colLogicalWidth;
170 if (colLogicalWidth.isFixed() && m_layoutStruct[effCol].maxLogicalWidth < colLogicalWidth.value())
171 m_layoutStruct[effCol].maxLogicalWidth = colLogicalWidth.value();
172 }
173 currentColumn += span;
174 }
175
176 // For the last column in a column-group, we invalidate our group logical width.
177 if (column->isTableColumn() && !column->nextSibling())
178 groupLogicalWidth = Length();
179 }
180
181 for (unsigned i = 0; i < nEffCols; i++)
182 recalcColumn(i);
183}

185static bool shouldScaleColumnsForParent(const RenderTable& table)
186{
187 RenderBlock* containingBlock = table.containingBlock();
188 while (containingBlock && !is<RenderView>(containingBlock)) {
189 // It doesn't matter if our table is auto or fixed: auto means we don't
190 // scale. Fixed doesn't care if we do or not because it doesn't depend
191 // on the cell contents' preferred widths.
192 if (is<RenderTableCell>(containingBlock))
193 return false;
194 // The max logical width of a table may be "infinity" (or tableMaxWidth, to be more exact) if the sum if the
195 // columns' percentages is 100% or more, AND there is at least one column that has a non-percentage-based positive
196 // logical width. In such situations no table logical width will be large enough to satisfy the constraint
197 // set by the contents. So the idea is to use ~infinity to make sure we use all available size in the containing
198 // block. However, this just doesn't work if this is a flex or grid item, so disallow scaling in that case.
199 if (is<RenderFlexibleBox>(containingBlock) || is<RenderGrid>(containingBlock))
200 return false;
201 containingBlock = containingBlock->containingBlock();
202 }
203 return true;
204}

206// FIXME: This needs to be adapted for vertical writing modes.
207static bool shouldScaleColumnsForSelf(RenderTable* table)
208{
209 // Normally, scale all columns to satisfy this from CSS2.2:
210 // "A percentage value for a column width is relative to the table width.
211 // If the table has 'width: auto', a percentage represents a constraint on the column's width"
212
213 // A special case. If this table is not fixed width and contained inside
214 // a cell, then don't bloat the maxwidth by examining percentage growth.
215 bool scale = true;
216 while (table) {
217 Length tableWidth = table->style().width();
218 if ((tableWidth.isAuto() || tableWidth.isPercentOrCalculated()) && !table->isOutOfFlowPositioned()) {
219 RenderBlock* containingBlock = table->containingBlock();
220 while (containingBlock && !is<RenderView>(*containingBlock) && !is<RenderTableCell>(*containingBlock)
221 && containingBlock->style().width().isAuto() && !containingBlock->isOutOfFlowPositioned())
222 containingBlock = containingBlock->containingBlock();
223
224 table = nullptr;
225 if (is<RenderTableCell>(containingBlock)
226 && (containingBlock->style().width().isAuto() || containingBlock->style().width().isPercentOrCalculated())) {
227 RenderTableCell& cell = downcast<RenderTableCell>(*containingBlock);
228 if (cell.colSpan() > 1 || cell.table()->style().width().isAuto())
229 scale = false;
230 else
231 table = cell.table();
232 }
233 }
234 else
235 table = nullptr;
236 }
237 return scale;
238}

240void AutoTableLayout::computeIntrinsicLogicalWidths(LayoutUnit& minWidth, LayoutUnit& maxWidth, TableIntrinsics intrinsics)
241{
242 fullRecalc();
243
244 float spanMaxLogicalWidth = calcEffectiveLogicalWidth();
245 minWidth = 0;
246 maxWidth = 0;
247 float maxPercent = 0;
248 float maxNonPercent = 0;
249 bool scaleColumnsForSelf = shouldScaleColumnsForSelf(m_table) && intrinsics == TableIntrinsics::ForLayout;
250
251 // We substitute 0 percent by (epsilon / percentScaleFactor) percent in two places below to avoid division by zero.
252 // FIXME: Handle the 0% cases properly.
253 const float epsilon = 1 / 128.0f;
254
255 float remainingPercent = 100;
256 for (size_t i = 0; i < m_layoutStruct.size(); ++i) {
257 minWidth += m_layoutStruct[i].effectiveMinLogicalWidth;
258 maxWidth += m_layoutStruct[i].effectiveMaxLogicalWidth;
259 if (scaleColumnsForSelf) {
260 if (m_layoutStruct[i].effectiveLogicalWidth.isPercent()) {
261 float percent = std::min(m_layoutStruct[i].effectiveLogicalWidth.percent(), remainingPercent);
262 float logicalWidth = m_layoutStruct[i].effectiveMaxLogicalWidth * 100 / std::max(percent, epsilon);
263 maxPercent = std::max(logicalWidth, maxPercent);
264 remainingPercent -= percent;
265 } else
266 maxNonPercent += m_layoutStruct[i].effectiveMaxLogicalWidth;
267 }
268 }
269
270 if (scaleColumnsForSelf) {
271 maxNonPercent = maxNonPercent * 100 / std::max(remainingPercent, epsilon);
272 m_scaledWidthFromPercentColumns = LayoutUnit(std::min<float>(maxNonPercent, tableMaxWidth));
273 m_scaledWidthFromPercentColumns = std::max(m_scaledWidthFromPercentColumns, LayoutUnit(std::min<float>(maxPercent, tableMaxWidth)));
274 if (m_scaledWidthFromPercentColumns > maxWidth && shouldScaleColumnsForParent(*m_table))
275 maxWidth = m_scaledWidthFromPercentColumns;
276 }
277
278 maxWidth = std::max(maxWidth, LayoutUnit(spanMaxLogicalWidth));
279}

281void AutoTableLayout::applyPreferredLogicalWidthQuirks(LayoutUnit& minWidth, LayoutUnit& maxWidth) const
282{
283 if (m_table->hasOverridingLogicalWidth())
284 minWidth = maxWidth = std::max(minWidth, m_table->overridingLogicalWidth());
285 else if (auto tableLogicalWidth = m_table->style().logicalWidth(); tableLogicalWidth.isFixed() && tableLogicalWidth.isPositive())
286 minWidth = maxWidth = std::max(minWidth, LayoutUnit(tableLogicalWidth.value()));
287}

289/*
290 This method takes care of colspans.
291 effWidth is the same as width for cells without colspans. If we have colspans, they get modified.
292 */
293float AutoTableLayout::calcEffectiveLogicalWidth()
294{
295 float maxLogicalWidth = 0;
296
297 size_t nEffCols = m_layoutStruct.size();
298 float spacingInRowDirection = m_table->hBorderSpacing();
299
300 for (size_t i = 0; i < nEffCols; ++i) {
301 m_layoutStruct[i].effectiveLogicalWidth = m_layoutStruct[i].logicalWidth;
302 m_layoutStruct[i].effectiveMinLogicalWidth = m_layoutStruct[i].minLogicalWidth;
303 m_layoutStruct[i].effectiveMaxLogicalWidth = m_layoutStruct[i].maxLogicalWidth;
304 }
305
306 for (size_t i = 0; i < m_spanCells.size(); ++i) {
307 RenderTableCell* cell = m_spanCells[i];
308 if (!cell)
309 break;
310
311 unsigned span = cell->colSpan();
312
313 Length cellLogicalWidth = cell->styleOrColLogicalWidth();
314 if (!cellLogicalWidth.isRelative() && cellLogicalWidth.isZero())
315 cellLogicalWidth = Length(); // make it Auto
316
317 unsigned effCol = m_table->colToEffCol(cell->col());
318 size_t lastCol = effCol;
319 float cellMinLogicalWidth = cell->minPreferredLogicalWidth() + spacingInRowDirection;
320 float cellMaxLogicalWidth = cell->maxPreferredLogicalWidth() + spacingInRowDirection;
321 float totalPercent = 0;
322 float spanMinLogicalWidth = 0;
323 float spanMaxLogicalWidth = 0;
324 bool allColsArePercent = true;
325 bool allColsAreFixed = true;
326 bool haveAuto = false;
327 bool spanHasEmptyCellsOnly = true;
328 float fixedWidth = 0;
329 while (lastCol < nEffCols && span > 0) {
330 Layout& columnLayout = m_layoutStruct[lastCol];
331 switch (columnLayout.logicalWidth.type()) {
332 case LengthType::Percent:
333 totalPercent += columnLayout.logicalWidth.percent();
334 allColsAreFixed = false;
335 break;
336 case LengthType::Fixed:
337 if (columnLayout.logicalWidth.value() > 0) {
338 fixedWidth += columnLayout.logicalWidth.value();
339 allColsArePercent = false;
340 // IE resets effWidth to Auto here, but this breaks the konqueror about page and seems to be some bad
341 // legacy behaviour anyway. mozilla doesn't do this so I decided we don't neither.
342 break;
343 }
344 FALLTHROUGH;
345 case LengthType::Auto:
346 haveAuto = true;
347 FALLTHROUGH;
348 default:
349 // If the column is a percentage width, do not let the spanning cell overwrite the
350 // width value. This caused a mis-rendering on amazon.com.
351 // Sample snippet:
352 // <table border=2 width=100%><
353 // <tr><td>1</td><td colspan=2>2-3</tr>
354 // <tr><td>1</td><td colspan=2 width=100%>2-3</td></tr>
355 // </table>
356 if (!columnLayout.effectiveLogicalWidth.isPercent()) {
357 columnLayout.effectiveLogicalWidth = Length();
358 allColsArePercent = false;
359 } else
360 totalPercent += columnLayout.effectiveLogicalWidth.percent();
361 allColsAreFixed = false;
362 }
363 if (!columnLayout.emptyCellsOnly)
364 spanHasEmptyCellsOnly = false;
365 span -= m_table->spanOfEffCol(lastCol);
366 spanMinLogicalWidth += columnLayout.effectiveMinLogicalWidth;
367 spanMaxLogicalWidth += columnLayout.effectiveMaxLogicalWidth;
368 lastCol++;
369 cellMinLogicalWidth -= spacingInRowDirection;
370 cellMaxLogicalWidth -= spacingInRowDirection;
371 }
372
373 // adjust table max width if needed
374 if (cellLogicalWidth.isPercent()) {
375 if (totalPercent > cellLogicalWidth.percent() || allColsArePercent) {
376 // can't satify this condition, treat as variable
377 cellLogicalWidth = Length();
378 } else {
379 maxLogicalWidth = std::max(maxLogicalWidth, std::max(spanMaxLogicalWidth, cellMaxLogicalWidth) * 100 / cellLogicalWidth.percent());
380
381 // all non percent columns in the span get percent values to sum up correctly.
382 float percentMissing = cellLogicalWidth.percent() - totalPercent;
383 float totalWidth = 0;
384 for (unsigned pos = effCol; pos < lastCol; ++pos) {
385 if (!m_layoutStruct[pos].effectiveLogicalWidth.isPercentOrCalculated())
386 totalWidth += m_layoutStruct[pos].effectiveMaxLogicalWidth;
387 }
388
389 for (unsigned pos = effCol; pos < lastCol; ++pos) {
390 if (!m_layoutStruct[pos].effectiveLogicalWidth.isPercentOrCalculated()) {
391 // Handle the case when there's only one cell with 'width: percent' and it's empty.
392 auto percent = percentMissing * (totalWidth ? m_layoutStruct[pos].effectiveMaxLogicalWidth / totalWidth : 1);
393 totalWidth -= m_layoutStruct[pos].effectiveMaxLogicalWidth;
394 percentMissing -= percent;
395 if (percent > 0)
396 m_layoutStruct[pos].effectiveLogicalWidth.setValue(LengthType::Percent, percent);
397 else
398 m_layoutStruct[pos].effectiveLogicalWidth = Length();
399 }
400 if (totalWidth <= 0)
401 break;
402 }
403 }
404 }
405
406 // make sure minWidth and maxWidth of the spanning cell are honoured
407 if (cellMinLogicalWidth > spanMinLogicalWidth) {
408 if (allColsAreFixed) {
409 for (unsigned pos = effCol; fixedWidth > 0 && pos < lastCol; ++pos) {
410 float cellLogicalWidth = std::max(m_layoutStruct[pos].effectiveMinLogicalWidth, cellMinLogicalWidth * m_layoutStruct[pos].logicalWidth.value() / fixedWidth);
411 fixedWidth -= m_layoutStruct[pos].logicalWidth.value();
412 cellMinLogicalWidth -= cellLogicalWidth;
413 m_layoutStruct[pos].effectiveMinLogicalWidth = cellLogicalWidth;
414 }
415 } else if (allColsArePercent) {
416 // In this case, we just split the colspan's min amd max widths following the percentage.
417 float allocatedMinLogicalWidth = 0;
418 float allocatedMaxLogicalWidth = 0;
419 for (unsigned pos = effCol; pos < lastCol; ++pos) {
420 ASSERT(m_layoutStruct[pos].logicalWidth.isPercent() || m_layoutStruct[pos].effectiveLogicalWidth.isPercent());
421 // |allColsArePercent| means that either the logicalWidth *or* the effectiveLogicalWidth are percents, handle both of them here.
422 float percent = m_layoutStruct[pos].logicalWidth.isPercent() ? m_layoutStruct[pos].logicalWidth.percent() : m_layoutStruct[pos].effectiveLogicalWidth.percent();
423 float columnMinLogicalWidth = percent * cellMinLogicalWidth / totalPercent;
424 float columnMaxLogicalWidth = percent * cellMaxLogicalWidth / totalPercent;
425 m_layoutStruct[pos].effectiveMinLogicalWidth = std::max(m_layoutStruct[pos].effectiveMinLogicalWidth, columnMinLogicalWidth);
426 m_layoutStruct[pos].effectiveMaxLogicalWidth = columnMaxLogicalWidth;
427 allocatedMinLogicalWidth += columnMinLogicalWidth;
428 allocatedMaxLogicalWidth += columnMaxLogicalWidth;
429 }
430 ASSERT(allocatedMinLogicalWidth < cellMinLogicalWidth || WTF::areEssentiallyEqual(allocatedMinLogicalWidth, cellMinLogicalWidth));
431 ASSERT(allocatedMaxLogicalWidth < cellMaxLogicalWidth || WTF::areEssentiallyEqual(allocatedMaxLogicalWidth, cellMaxLogicalWidth));
432 cellMinLogicalWidth -= allocatedMinLogicalWidth;
433 cellMaxLogicalWidth -= allocatedMaxLogicalWidth;
434 } else {
435 float remainingMaxLogicalWidth = spanMaxLogicalWidth;
436 float remainingMinLogicalWidth = spanMinLogicalWidth;
437
438 // Give min to variable first, to fixed second, and to others third.
439 for (unsigned pos = effCol; remainingMaxLogicalWidth >= 0 && pos < lastCol; ++pos) {
440 if (m_layoutStruct[pos].logicalWidth.isFixed() && haveAuto && fixedWidth <= cellMinLogicalWidth) {
441 float colMinLogicalWidth = std::max(m_layoutStruct[pos].effectiveMinLogicalWidth, m_layoutStruct[pos].logicalWidth.value());
442 fixedWidth -= m_layoutStruct[pos].logicalWidth.value();
443 remainingMinLogicalWidth -= m_layoutStruct[pos].effectiveMinLogicalWidth;
444 remainingMaxLogicalWidth -= m_layoutStruct[pos].effectiveMaxLogicalWidth;
445 cellMinLogicalWidth -= colMinLogicalWidth;
446 m_layoutStruct[pos].effectiveMinLogicalWidth = colMinLogicalWidth;
447 }
448 }
449
450 for (unsigned pos = effCol; remainingMaxLogicalWidth >= 0 && pos < lastCol && remainingMinLogicalWidth < cellMinLogicalWidth; ++pos) {
451 if (!(m_layoutStruct[pos].logicalWidth.isFixed() && haveAuto && fixedWidth <= cellMinLogicalWidth)) {
452 float colMinLogicalWidth = std::max(m_layoutStruct[pos].effectiveMinLogicalWidth, remainingMaxLogicalWidth ? cellMinLogicalWidth * m_layoutStruct[pos].effectiveMaxLogicalWidth / remainingMaxLogicalWidth : cellMinLogicalWidth);
453 colMinLogicalWidth = std::min(m_layoutStruct[pos].effectiveMinLogicalWidth + (cellMinLogicalWidth - remainingMinLogicalWidth), colMinLogicalWidth);
454 remainingMaxLogicalWidth -= m_layoutStruct[pos].effectiveMaxLogicalWidth;
455 remainingMinLogicalWidth -= m_layoutStruct[pos].effectiveMinLogicalWidth;
456 cellMinLogicalWidth -= colMinLogicalWidth;
457 m_layoutStruct[pos].effectiveMinLogicalWidth = colMinLogicalWidth;
458 }
459 }
460 }
461 }
462 if (!cellLogicalWidth.isPercentOrCalculated()) {
463 if (cellMaxLogicalWidth > spanMaxLogicalWidth) {
464 for (unsigned pos = effCol; spanMaxLogicalWidth >= 0 && pos < lastCol; ++pos) {
465 float colMaxLogicalWidth = std::max(m_layoutStruct[pos].effectiveMaxLogicalWidth, spanMaxLogicalWidth ? cellMaxLogicalWidth * m_layoutStruct[pos].effectiveMaxLogicalWidth / spanMaxLogicalWidth : cellMaxLogicalWidth);
466 spanMaxLogicalWidth -= m_layoutStruct[pos].effectiveMaxLogicalWidth;
467 cellMaxLogicalWidth -= colMaxLogicalWidth;
468 m_layoutStruct[pos].effectiveMaxLogicalWidth = colMaxLogicalWidth;
469 }
470 }
471 } else {
472 for (unsigned pos = effCol; pos < lastCol; ++pos)
473 m_layoutStruct[pos].maxLogicalWidth = std::max(m_layoutStruct[pos].maxLogicalWidth, m_layoutStruct[pos].minLogicalWidth);
474 }
475 // treat span ranges consisting of empty cells only as if they had content
476 if (spanHasEmptyCellsOnly) {
477 for (unsigned pos = effCol; pos < lastCol; ++pos)
478 m_layoutStruct[pos].emptyCellsOnly = false;
479 }
480 }
481 m_effectiveLogicalWidthDirty = false;
482
483 return std::min<float>(maxLogicalWidth, tableMaxWidth);
484}

486/* gets all cells that originate in a column and have a cellspan > 1
487 Sorts them by increasing cellspan
488*/
489void AutoTableLayout::insertSpanCell(RenderTableCell *cell)
490{
491 ASSERT_ARG(cell, cell && cell->colSpan() != 1);
492 if (!cell || cell->colSpan() == 1)
493 return;
494
495 unsigned size = m_spanCells.size();
496 if (!size || m_spanCells[size-1] != 0) {
497 m_spanCells.grow(size + 10);
498 for (unsigned i = 0; i < 10; i++)
499 m_spanCells[size + i] = 0;
500 size += 10;
501 }
502
503 // add them in sort. This is a slow algorithm, and a binary search or a fast sorting after collection would be better
504 unsigned pos = 0;
505 unsigned span = cell->colSpan();
506 while (pos < m_spanCells.size() && m_spanCells[pos] && span > m_spanCells[pos]->colSpan())
507 pos++;
508 memmove(m_spanCells.data()+pos+1, m_spanCells.data()+pos, (size-pos-1)*sizeof(RenderTableCell *));
509 m_spanCells[pos] = cell;
510}

513void AutoTableLayout::layout()
514{
515 // table layout based on the values collected in the layout structure.
516 float tableLogicalWidth = m_table->logicalWidth() - m_table->bordersPaddingAndSpacingInRowDirection();
517 float available = tableLogicalWidth;
518 size_t nEffCols = m_table->numEffCols();
519
520 // FIXME: It is possible to be called without having properly updated our internal representation.
521 // This means that our preferred logical widths were not recomputed as expected.
522 if (nEffCols != m_layoutStruct.size()) {
523 fullRecalc();
524 // FIXME: Table layout shouldn't modify our table structure (but does due to columns and column-groups).
525 nEffCols = m_table->numEffCols();
526 }
527
528 if (m_effectiveLogicalWidthDirty)
529 calcEffectiveLogicalWidth();
530
531 bool havePercent = false;
532 float totalRelative = 0;
533 int numFixed = 0;
534 size_t numberOfNonEmptyAuto = 0;
535 std::optional<float> totalAuto;
536 float totalFixed = 0;
537 float totalPercent = 0;
538 float allocAuto = 0;
539 unsigned numAutoEmptyCellsOnly = 0;
540
541 // fill up every cell with its minWidth
542 for (size_t i = 0; i < nEffCols; ++i) {
543 float cellLogicalWidth = m_layoutStruct[i].effectiveMinLogicalWidth;
544 m_layoutStruct[i].computedLogicalWidth = cellLogicalWidth;
545 available -= cellLogicalWidth;
546 Length& logicalWidth = m_layoutStruct[i].effectiveLogicalWidth;
547 switch (logicalWidth.type()) {
548 case LengthType::Percent:
549 havePercent = true;
550 totalPercent += logicalWidth.percent();
551 break;
552 case LengthType::Relative:
553 totalRelative += logicalWidth.value();
554 break;
555 case LengthType::Fixed:
556 numFixed++;
557 totalFixed += m_layoutStruct[i].effectiveMaxLogicalWidth;
558 break;
559 case LengthType::Auto:
560 if (m_layoutStruct[i].emptyCellsOnly)
561 numAutoEmptyCellsOnly++;
562 else {
563 ++numberOfNonEmptyAuto;
564 totalAuto = totalAuto.value_or(0.f) + m_layoutStruct[i].effectiveMaxLogicalWidth;
565 allocAuto += cellLogicalWidth;
566 }
567 break;
568 default:
569 break;
570 }
571 }
572
573 // allocate width to percent cols
574 if (available > 0 && havePercent) {
575 for (size_t i = 0; i < nEffCols; ++i) {
576 Length& logicalWidth = m_layoutStruct[i].effectiveLogicalWidth;
577 if (logicalWidth.isPercentOrCalculated()) {
578 float cellLogicalWidth = std::max<float>(m_layoutStruct[i].effectiveMinLogicalWidth, minimumValueForLength(logicalWidth, tableLogicalWidth));
579 available += m_layoutStruct[i].computedLogicalWidth - cellLogicalWidth;
580 m_layoutStruct[i].computedLogicalWidth = cellLogicalWidth;
581 }
582 }
583 if (totalPercent > 100) {
584 // remove overallocated space from the last columns
585 float excess = tableLogicalWidth * (totalPercent - 100) / 100;
586 for (unsigned i = nEffCols; i; ) {
587 --i;
588 if (m_layoutStruct[i].effectiveLogicalWidth.isPercentOrCalculated()) {
589 float cellLogicalWidth = m_layoutStruct[i].computedLogicalWidth;
590 float reduction = std::min(cellLogicalWidth, excess);
591 // the lines below might look inconsistent, but that's the way it's handled in mozilla
592 excess -= reduction;
593 float newLogicalWidth = std::max(m_layoutStruct[i].effectiveMinLogicalWidth, cellLogicalWidth - reduction);
594 available += cellLogicalWidth - newLogicalWidth;
595 m_layoutStruct[i].computedLogicalWidth = newLogicalWidth;
596 }
597 }
598 }
599 }
600
601 // then allocate width to fixed cols
602 if (available > 0) {
603 for (size_t i = 0; i < nEffCols; ++i) {
604 Length& logicalWidth = m_layoutStruct[i].effectiveLogicalWidth;
605 if (logicalWidth.isFixed() && logicalWidth.value() > m_layoutStruct[i].computedLogicalWidth) {
606 available += m_layoutStruct[i].computedLogicalWidth - logicalWidth.value();
607 m_layoutStruct[i].computedLogicalWidth = logicalWidth.value();
608 }
609 }
610 }
611
612 // now satisfy relative
613 if (available > 0) {
614 for (size_t i = 0; i < nEffCols; ++i) {
615 Length& logicalWidth = m_layoutStruct[i].effectiveLogicalWidth;
616 if (logicalWidth.isRelative() && logicalWidth.value() != 0) {
617 // width=0* gets effMinWidth.
618 float cellLogicalWidth = logicalWidth.value() * tableLogicalWidth / totalRelative;
619 available += m_layoutStruct[i].computedLogicalWidth - cellLogicalWidth;
620 m_layoutStruct[i].computedLogicalWidth = cellLogicalWidth;
621 }
622 }
623 }
624
625 // now satisfy variable
626 if (available > 0 && numberOfNonEmptyAuto) {
627 ASSERT(totalAuto);
628 available += allocAuto; // this gets redistributed.
629 auto equalWidthForZeroLengthColumns = std::optional<float> { };
630 if (!*totalAuto) {
631 // All columns in this table are (non-empty)zero length with 'width: auto'.
632 equalWidthForZeroLengthColumns = available / numberOfNonEmptyAuto;
633 }
634 for (size_t i = 0; i < nEffCols; ++i) {
635 auto& column = m_layoutStruct[i];
636 if (!column.effectiveLogicalWidth.isAuto() || column.emptyCellsOnly)
637 continue;
638 auto columnWidthCandidate = equalWidthForZeroLengthColumns ? *equalWidthForZeroLengthColumns : available * column.effectiveMaxLogicalWidth / *totalAuto;
639 column.computedLogicalWidth = std::max(column.computedLogicalWidth, columnWidthCandidate);
640 available -= column.computedLogicalWidth;
641 if (!equalWidthForZeroLengthColumns) {
642 *totalAuto -= column.effectiveMaxLogicalWidth;
643 if (*totalAuto <= 0)
644 break;
645 }
646 }
647 }
648
649 // spread over fixed columns
650 if (available > 0 && numFixed) {
651 for (size_t i = 0; i < nEffCols; ++i) {
652 Length& logicalWidth = m_layoutStruct[i].effectiveLogicalWidth;
653 if (logicalWidth.isFixed()) {
654 float cellLogicalWidth = available * m_layoutStruct[i].effectiveMaxLogicalWidth / totalFixed;
655 available -= cellLogicalWidth;
656 totalFixed -= m_layoutStruct[i].effectiveMaxLogicalWidth;
657 m_layoutStruct[i].computedLogicalWidth += cellLogicalWidth;
658 }
659 }
660 }
661
662 // spread over percent colums
663 if (available > 0 && m_hasPercent && totalPercent < 100) {
664 for (size_t i = 0; i < nEffCols; ++i) {
665 Length& logicalWidth = m_layoutStruct[i].effectiveLogicalWidth;
666 if (logicalWidth.isPercent()) {
667 float cellLogicalWidth = available * logicalWidth.percent() / totalPercent;
668 available -= cellLogicalWidth;
669 totalPercent -= logicalWidth.percent();
670 m_layoutStruct[i].computedLogicalWidth += cellLogicalWidth;
671 if (!available || !totalPercent)
672 break;
673 }
674 }
675 }
676
677 // spread over the rest
678 if (available > 0 && nEffCols > numAutoEmptyCellsOnly) {
679 unsigned total = nEffCols - numAutoEmptyCellsOnly;
680 // still have some width to spread
681 for (unsigned i = nEffCols; i; ) {
682 --i;
683 // variable columns with empty cells only don't get any width
684 if (m_layoutStruct[i].effectiveLogicalWidth.isAuto() && m_layoutStruct[i].emptyCellsOnly)
685 continue;
686 float cellLogicalWidth = available / total;
687 available -= cellLogicalWidth;
688 total--;
689 m_layoutStruct[i].computedLogicalWidth += cellLogicalWidth;
690 }
691 }
692
693 // If we have overallocated, reduce every cell according to the difference between desired width and minwidth
694 // this seems to produce to the pixel exact results with IE. Wonder if some of this also holds for width distributing.
695 if (available < 0) {
696 // Need to reduce cells with the following prioritization:
697 // (1) Auto
698 // (2) Relative
699 // (3) Fixed
700 // (4) Percent
701 // This is basically the reverse of how we grew the cells.
702 if (available < 0) {
703 float logicalWidthBeyondMin = 0;
704 for (unsigned i = nEffCols; i; ) {
705 --i;
706 Length& logicalWidth = m_layoutStruct[i].effectiveLogicalWidth;
707 if (logicalWidth.isAuto())
708 logicalWidthBeyondMin += m_layoutStruct[i].computedLogicalWidth - m_layoutStruct[i].effectiveMinLogicalWidth;
709 }
710
711 for (unsigned i = nEffCols; i && logicalWidthBeyondMin > 0; ) {
712 --i;
713 Length& logicalWidth = m_layoutStruct[i].effectiveLogicalWidth;
714 if (logicalWidth.isAuto()) {
715 float minMaxDiff = m_layoutStruct[i].computedLogicalWidth - m_layoutStruct[i].effectiveMinLogicalWidth;
716 float reduce = available * minMaxDiff / logicalWidthBeyondMin;
717 m_layoutStruct[i].computedLogicalWidth += reduce;
718 available -= reduce;
719 logicalWidthBeyondMin -= minMaxDiff;
720 if (available >= 0)
721 break;
722 }
723 }
724 }
725
726 if (available < 0) {
727 float logicalWidthBeyondMin = 0;
728 for (unsigned i = nEffCols; i; ) {
729 --i;
730 Length& logicalWidth = m_layoutStruct[i].effectiveLogicalWidth;
731 if (logicalWidth.isRelative())
732 logicalWidthBeyondMin += m_layoutStruct[i].computedLogicalWidth - m_layoutStruct[i].effectiveMinLogicalWidth;
733 }
734
735 for (unsigned i = nEffCols; i && logicalWidthBeyondMin > 0; ) {
736 --i;
737 Length& logicalWidth = m_layoutStruct[i].effectiveLogicalWidth;
738 if (logicalWidth.isRelative()) {
739 float minMaxDiff = m_layoutStruct[i].computedLogicalWidth - m_layoutStruct[i].effectiveMinLogicalWidth;
740 float reduce = available * minMaxDiff / logicalWidthBeyondMin;
741 m_layoutStruct[i].computedLogicalWidth += reduce;
742 available -= reduce;
743 logicalWidthBeyondMin -= minMaxDiff;
744 if (available >= 0)
745 break;
746 }
747 }
748 }
749
750 if (available < 0) {
751 float logicalWidthBeyondMin = 0;
752 for (unsigned i = nEffCols; i; ) {
753 --i;
754 Length& logicalWidth = m_layoutStruct[i].effectiveLogicalWidth;
755 if (logicalWidth.isFixed())
756 logicalWidthBeyondMin += m_layoutStruct[i].computedLogicalWidth - m_layoutStruct[i].effectiveMinLogicalWidth;
757 }
758
759 for (unsigned i = nEffCols; i && logicalWidthBeyondMin > 0; ) {
760 --i;
761 Length& logicalWidth = m_layoutStruct[i].effectiveLogicalWidth;
762 if (logicalWidth.isFixed()) {
763 float minMaxDiff = m_layoutStruct[i].computedLogicalWidth - m_layoutStruct[i].effectiveMinLogicalWidth;
764 float reduce = available * minMaxDiff / logicalWidthBeyondMin;
765 m_layoutStruct[i].computedLogicalWidth += reduce;
766 available -= reduce;
767 logicalWidthBeyondMin -= minMaxDiff;
768 if (available >= 0)
769 break;
770 }
771 }
772 }
773
774 if (available < 0) {
775 float logicalWidthBeyondMin = 0;
776 for (unsigned i = nEffCols; i; ) {
777 --i;
778 Length& logicalWidth = m_layoutStruct[i].effectiveLogicalWidth;
779 if (logicalWidth.isPercentOrCalculated())
780 logicalWidthBeyondMin += m_layoutStruct[i].computedLogicalWidth - m_layoutStruct[i].effectiveMinLogicalWidth;
781 }
782
783 for (unsigned i = nEffCols; i && logicalWidthBeyondMin > 0; ) {
784 --i;
785 Length& logicalWidth = m_layoutStruct[i].effectiveLogicalWidth;
786 if (logicalWidth.isPercentOrCalculated()) {
787 float minMaxDiff = m_layoutStruct[i].computedLogicalWidth - m_layoutStruct[i].effectiveMinLogicalWidth;
788 float reduce = available * minMaxDiff / logicalWidthBeyondMin;
789 m_layoutStruct[i].computedLogicalWidth += reduce;
790 available -= reduce;
791 logicalWidthBeyondMin -= minMaxDiff;
792 if (available >= 0)
793 break;
794 }
795 }
796 }
797 }

799 LayoutUnit pos;
800 for (size_t i = 0; i < nEffCols; ++i) {
801 m_table->setColumnPosition(i, pos);
802 pos += LayoutUnit::fromFloatCeil(m_layoutStruct[i].computedLogicalWidth) + m_table->hBorderSpacing();
803 }
