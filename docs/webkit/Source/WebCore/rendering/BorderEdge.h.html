37class RenderStyle;

39class BorderEdge {
40public:
41 BorderEdge() = default;
42 BorderEdge(float edgeWidth, Color edgeColor, BorderStyle edgeStyle, bool edgeIsTransparent, bool edgeIsPresent, float devicePixelRatio);
43
44 BorderStyle style() const { return m_style; }
45 const Color& color() const { return m_color; }
46 bool isTransparent() const { return m_isTransparent; }
47 bool isPresent() const { return m_isPresent; }
48
49 inline bool hasVisibleColorAndStyle() const { return m_style > BorderStyle::Hidden && !m_isTransparent; }
50 inline bool shouldRender() const { return m_isPresent && widthForPainting() && hasVisibleColorAndStyle(); }
51 inline bool presentButInvisible() const { return widthForPainting() && !hasVisibleColorAndStyle(); }
52 inline float widthForPainting() const { return m_isPresent ? m_flooredToDevicePixelWidth : 0; }
53 void getDoubleBorderStripeWidths(LayoutUnit& outerWidth, LayoutUnit& innerWidth) const;
54 bool obscuresBackgroundEdge(float scale) const;
55 bool obscuresBackground() const;
56
57private:
58 inline float borderWidthInDevicePixel(int logicalPixels) const { return LayoutUnit(logicalPixels / m_devicePixelRatio).toFloat(); }
59
60 Color m_color;
61 LayoutUnit m_width;
62 float m_flooredToDevicePixelWidth { 0 };
63 float m_devicePixelRatio { 1 };
64 BorderStyle m_style { BorderStyle::Hidden };
65 bool m_isTransparent { false };
66 bool m_isPresent { false };
67};
68
69using BorderEdges = RectEdges<BorderEdge>;
70BorderEdges borderEdges(const RenderStyle&, float deviceScaleFactor, bool includeLogicalLeftEdge = true, bool includeLogicalRightEdge = true);
71
72inline bool edgesShareColor(const BorderEdge& firstEdge, const BorderEdge& secondEdge) { return firstEdge.color() == secondEdge.color(); }
73inline BoxSideFlag edgeFlagForSide(BoxSide side) { return static_cast<BoxSideFlag>(1 << static_cast<unsigned>(side)); }
74inline bool includesEdge(OptionSet<BoxSideFlag> flags, BoxSide side) { return flags.contains(edgeFlagForSide(side)); }
75
76inline bool includesAdjacentEdges(OptionSet<BoxSideFlag> flags)
77{
78 return flags.containsAll({ BoxSideFlag::Top, BoxSideFlag::Right })
79 || flags.containsAll({ BoxSideFlag::Right, BoxSideFlag::Bottom })
80 || flags.containsAll({ BoxSideFlag::Bottom, BoxSideFlag::Left })
81 || flags.containsAll({ BoxSideFlag::Left, BoxSideFlag::Top });
82}
