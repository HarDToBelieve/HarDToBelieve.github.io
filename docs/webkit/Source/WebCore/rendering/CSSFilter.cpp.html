73RefPtr<FilterEffect> CSSFilter::buildReferenceFilter(RenderElement& renderer, FilterEffect& previousEffect, ReferenceFilterOperation& filterOperation)
74{
75 auto* cachedSVGDocumentReference = filterOperation.cachedSVGDocumentReference();
76 auto* cachedSVGDocument = cachedSVGDocumentReference ? cachedSVGDocumentReference->document() : nullptr;
77
78 // If we have an SVG document, this is an external reference. Otherwise
79 // we look up the referenced node in the current document.
80 Document* document;
81 if (!cachedSVGDocument)
82 document = &renderer.document();
83 else {
84 document = cachedSVGDocument->document();
85 if (!document)
86 return nullptr;
87 }
88
89 auto* filter = document->getElementById(filterOperation.fragment());
90 if (!filter) {
91 // Although we did not find the referenced filter, it might exist later in the document.
92 // FIXME: This skips anonymous RenderObjects. <https://webkit.org/b/131085>
93 if (auto* element = renderer.element())
94 document->accessSVGExtensions().addPendingResource(filterOperation.fragment(), *element);
95 return nullptr;
96 }
97
98 auto builder = makeUnique<SVGFilterBuilder>(&previousEffect);
99 m_sourceAlpha = builder->getEffectById(SourceAlpha::effectName());
100
101 RefPtr<FilterEffect> effect;
102 Vector<Ref<FilterEffect>> referenceEffects;
103
104 for (auto& effectElement : childrenOfType<SVGFilterPrimitiveStandardAttributes>(*filter)) {
105 effect = effectElement.build(builder.get(), *this);
106 if (!effect) {
107 LOG_WITH_STREAM(Filters, stream << "CSSFilter " << this << " buildReferenceFilter: failed to build effect from " << effectElement);
108 return nullptr;
109 }
110
111 effectElement.setStandardAttributes(effect.get());
112 if (effectElement.renderer()) {
113#if ENABLE(DESTINATION_COLOR_SPACE_LINEAR_SRGB)
114 effect->setOperatingColorSpace(effectElement.renderer()->style().svgStyle().colorInterpolationFilters() == ColorInterpolation::LinearRGB ? DestinationColorSpace::LinearSRGB() : DestinationColorSpace::SRGB());
115#else
116 effect->setOperatingColorSpace(DestinationColorSpace::SRGB());
117#endif
118 }
119 builder->add(effectElement.result(), effect);
120 referenceEffects.append(*effect);
121 }
122
123 m_effects.appendVector(WTFMove(referenceEffects));
124 return effect;
125}

127bool CSSFilter::build(RenderElement& renderer, const FilterOperations& operations, FilterConsumer consumer)
128{
129 m_hasFilterThatMovesPixels = operations.hasFilterThatMovesPixels();
130 m_hasFilterThatShouldBeRestrictedBySecurityOrigin = operations.hasFilterThatShouldBeRestrictedBySecurityOrigin();
131
132 m_effects.clear();
133 m_outsets = { };
134
135 RefPtr<FilterEffect> previousEffect = m_sourceGraphic.ptr();
136 for (auto& operation : operations.operations()) {
137 RefPtr<FilterEffect> effect;
138 auto& filterOperation = *operation;
139 switch (filterOperation.type()) {
140 case FilterOperation::REFERENCE: {
141 auto& referenceOperation = downcast<ReferenceFilterOperation>(filterOperation);
142 effect = buildReferenceFilter(renderer, *previousEffect, referenceOperation);
143 break;
144 }
145 case FilterOperation::GRAYSCALE: {
146 auto& colorMatrixOperation = downcast<BasicColorMatrixFilterOperation>(filterOperation);
147 double oneMinusAmount = clampTo(1 - colorMatrixOperation.amount(), 0.0, 1.0);
148
149 // See https://dvcs.w3.org/hg/FXTF/raw-file/tip/filters/index.html#grayscaleEquivalent
150 // for information on parameters.
151
152 Vector<float> inputParameters {
153 narrowPrecisionToFloat(0.2126 + 0.7874 * oneMinusAmount),
154 narrowPrecisionToFloat(0.7152 - 0.7152 * oneMinusAmount),
155 narrowPrecisionToFloat(0.0722 - 0.0722 * oneMinusAmount),
156 0,
157 0,
158
159 narrowPrecisionToFloat(0.2126 - 0.2126 * oneMinusAmount),
160 narrowPrecisionToFloat(0.7152 + 0.2848 * oneMinusAmount),
161 narrowPrecisionToFloat(0.0722 - 0.0722 * oneMinusAmount),
162 0,
163 0,
164
165 narrowPrecisionToFloat(0.2126 - 0.2126 * oneMinusAmount),
166 narrowPrecisionToFloat(0.7152 - 0.7152 * oneMinusAmount),
167 narrowPrecisionToFloat(0.0722 + 0.9278 * oneMinusAmount),
168 0,
169 0,
170
171 0,
172 0,
173 0,
174 1,
175 0,
176 };
177
178 effect = FEColorMatrix::create(*this, FECOLORMATRIX_TYPE_MATRIX, WTFMove(inputParameters));
179 break;
180 }
181 case FilterOperation::SEPIA: {
182 auto& colorMatrixOperation = downcast<BasicColorMatrixFilterOperation>(filterOperation);
183 double oneMinusAmount = clampTo(1 - colorMatrixOperation.amount(), 0.0, 1.0);
184
185 // See https://dvcs.w3.org/hg/FXTF/raw-file/tip/filters/index.html#sepiaEquivalent
186 // for information on parameters.
187
188 Vector<float> inputParameters {
189 narrowPrecisionToFloat(0.393 + 0.607 * oneMinusAmount),
190 narrowPrecisionToFloat(0.769 - 0.769 * oneMinusAmount),
191 narrowPrecisionToFloat(0.189 - 0.189 * oneMinusAmount),
192 0,
193 0,
194
195 narrowPrecisionToFloat(0.349 - 0.349 * oneMinusAmount),
196 narrowPrecisionToFloat(0.686 + 0.314 * oneMinusAmount),
197 narrowPrecisionToFloat(0.168 - 0.168 * oneMinusAmount),
198 0,
199 0,
200
201 narrowPrecisionToFloat(0.272 - 0.272 * oneMinusAmount),
202 narrowPrecisionToFloat(0.534 - 0.534 * oneMinusAmount),
203 narrowPrecisionToFloat(0.131 + 0.869 * oneMinusAmount),
204 0,
205 0,
206
207 0,
208 0,
209 0,
210 1,
211 0,
212 };
213
214 effect = FEColorMatrix::create(*this, FECOLORMATRIX_TYPE_MATRIX, WTFMove(inputParameters));
215 break;
216 }
217 case FilterOperation::SATURATE: {
218 auto& colorMatrixOperation = downcast<BasicColorMatrixFilterOperation>(filterOperation);
219 Vector<float> inputParameters { narrowPrecisionToFloat(colorMatrixOperation.amount()) };
220 effect = FEColorMatrix::create(*this, FECOLORMATRIX_TYPE_SATURATE, WTFMove(inputParameters));
221 break;
222 }
223 case FilterOperation::HUE_ROTATE: {
224 auto& colorMatrixOperation = downcast<BasicColorMatrixFilterOperation>(filterOperation);
225 Vector<float> inputParameters { narrowPrecisionToFloat(colorMatrixOperation.amount()) };
226 effect = FEColorMatrix::create(*this, FECOLORMATRIX_TYPE_HUEROTATE, WTFMove(inputParameters));
227 break;
228 }
229 case FilterOperation::INVERT: {
230 auto& componentTransferOperation = downcast<BasicComponentTransferFilterOperation>(filterOperation);
231 ComponentTransferFunction transferFunction;
232 transferFunction.type = FECOMPONENTTRANSFER_TYPE_LINEAR;
233 float amount = narrowPrecisionToFloat(componentTransferOperation.amount());
234 transferFunction.slope = 1 - 2 * amount;
235 transferFunction.intercept = amount;
236
237 ComponentTransferFunction nullFunction;
238 effect = FEComponentTransfer::create(*this, transferFunction, transferFunction, transferFunction, nullFunction);
239 break;
240 }
241 case FilterOperation::APPLE_INVERT_LIGHTNESS:
242 ASSERT_NOT_REACHED(); // APPLE_INVERT_LIGHTNESS is only used in -apple-color-filter.
243 break;
244 case FilterOperation::OPACITY: {
245 auto& componentTransferOperation = downcast<BasicComponentTransferFilterOperation>(filterOperation);
246 ComponentTransferFunction transferFunction;
247 transferFunction.type = FECOMPONENTTRANSFER_TYPE_LINEAR;
248 float amount = narrowPrecisionToFloat(componentTransferOperation.amount());
249 transferFunction.slope = amount;
250 transferFunction.intercept = 0;
251
252 ComponentTransferFunction nullFunction;
253 effect = FEComponentTransfer::create(*this, nullFunction, nullFunction, nullFunction, transferFunction);
254 break;
255 }
256 case FilterOperation::BRIGHTNESS: {
257 auto& componentTransferOperation = downcast<BasicComponentTransferFilterOperation>(filterOperation);
258 ComponentTransferFunction transferFunction;
259 transferFunction.type = FECOMPONENTTRANSFER_TYPE_LINEAR;
260 transferFunction.slope = narrowPrecisionToFloat(componentTransferOperation.amount());
261 transferFunction.intercept = 0;
262
263 ComponentTransferFunction nullFunction;
264 effect = FEComponentTransfer::create(*this, transferFunction, transferFunction, transferFunction, nullFunction);
265 break;
266 }
267 case FilterOperation::CONTRAST: {
268 auto& componentTransferOperation = downcast<BasicComponentTransferFilterOperation>(filterOperation);
269 ComponentTransferFunction transferFunction;
270 transferFunction.type = FECOMPONENTTRANSFER_TYPE_LINEAR;
271 float amount = narrowPrecisionToFloat(componentTransferOperation.amount());
272 transferFunction.slope = amount;
273 transferFunction.intercept = -0.5 * amount + 0.5;
274
275 ComponentTransferFunction nullFunction;
276 effect = FEComponentTransfer::create(*this, transferFunction, transferFunction, transferFunction, nullFunction);
277 break;
278 }
279 case FilterOperation::BLUR: {
280 auto& blurOperation = downcast<BlurFilterOperation>(filterOperation);
281 float stdDeviation = floatValueForLength(blurOperation.stdDeviation(), 0);
282 effect = FEGaussianBlur::create(*this, stdDeviation, stdDeviation, consumer == FilterConsumer::FilterProperty ? EDGEMODE_NONE : EDGEMODE_DUPLICATE);
283 break;
284 }
285 case FilterOperation::DROP_SHADOW: {
286 auto& dropShadowOperation = downcast<DropShadowFilterOperation>(filterOperation);
287 effect = FEDropShadow::create(*this, dropShadowOperation.stdDeviation(), dropShadowOperation.stdDeviation(),
288 dropShadowOperation.x(), dropShadowOperation.y(), dropShadowOperation.color(), 1);
289 break;
290 }
291 default:
292 break;
293 }
294
295 if (effect) {
296 // Unlike SVG Filters and CSSFilterImages, filter functions on the filter
297 // property applied here should not clip to their primitive subregions.
298 effect->setClipsToBounds(consumer == FilterConsumer::FilterFunction);
299 effect->setOperatingColorSpace(DestinationColorSpace::SRGB());
300
301 if (filterOperation.type() != FilterOperation::REFERENCE) {
302 effect->inputEffects() = { WTFMove(previousEffect) };
303 m_effects.append(*effect);
304 }
305 previousEffect = WTFMove(effect);
306 }
307 }
308
309 // If we didn't make any effects, tell our caller we are not valid.
310 if (m_effects.isEmpty())
311 return false;
312
313 m_effects.shrinkToFit();
314
315 setMaxEffectRects(m_sourceDrawingRegion);
316#if USE(CORE_IMAGE)
317 if (!m_filterRenderer)
318 m_filterRenderer = FilterEffectRenderer::tryCreate(renderer.settings().coreImageAcceleratedFilterRenderEnabled(), m_effects.last().get());
319#endif
320 return true;
321}

323bool CSSFilter::updateBackingStoreRect(const FloatRect& filterRect)
324{
325 if (filterRect.isEmpty() || ImageBuffer::sizeNeedsClamping(filterRect.size()))
326 return false;
327
328 if (filterRect == sourceImageRect())
329 return false;
330
331 setSourceImageRect(filterRect);
332 return true;
333}
334
335void CSSFilter::allocateBackingStoreIfNeeded(const GraphicsContext& targetContext)
336{
337 // At this point the effect chain has been built, and the
338 // source image sizes set. We just need to attach the graphic
339 // buffer if we have not yet done so.
340
341 if (m_graphicsBufferAttached)
342 return;
343
344 IntSize logicalSize { m_sourceDrawingRegion.size() };
345 if (!sourceImage() || sourceImage()->logicalSize() != logicalSize) {
346#if USE(DIRECT2D)
347 setSourceImage(ImageBuffer::create(logicalSize, renderingMode(), &targetContext, filterScale(), DestinationColorSpace::SRGB(), PixelFormat::BGRA8));
348#else
349 UNUSED_PARAM(targetContext);
350 RenderingMode mode = m_filterRenderer ? RenderingMode::Accelerated : renderingMode();
351 setSourceImage(ImageBuffer::create(logicalSize, mode, filterScale(), DestinationColorSpace::SRGB(), PixelFormat::BGRA8));
352#endif
353 }
354 m_graphicsBufferAttached = true;
355}

357void CSSFilter::determineFilterPrimitiveSubregion()
358{
359 auto& lastEffect = m_effects.last().get();
360 lastEffect.determineFilterPrimitiveSubregion();
361 FloatRect subRegion = lastEffect.maxEffectRect();
362 // At least one FilterEffect has a too big image size, recalculate the effect sizes with new scale factors.
363 FloatSize scale;
364 if (ImageBuffer::sizeNeedsClamping(subRegion.size(), scale)) {
365 setFilterResolution(scale);
366 lastEffect.determineFilterPrimitiveSubregion();
367 }
368}

370void CSSFilter::clearIntermediateResults()
371{
372 m_sourceGraphic->clearResult();
373 if (m_sourceAlpha)
374 m_sourceAlpha->clearResult();
375 for (auto& effect : m_effects)
376 effect->clearResult();
377}
378
379void CSSFilter::apply()
380{
381 auto& effect = m_effects.last().get();
382 if (m_filterRenderer) {
383 m_filterRenderer->applyEffects(effect);
384 if (m_filterRenderer->hasResult()) {
385 effect.transformResultColorSpace(DestinationColorSpace::SRGB());
386 return;
387 }
388 }
389 effect.apply();
390 effect.transformResultColorSpace(DestinationColorSpace::SRGB());
391}

393LayoutRect CSSFilter::computeSourceImageRectForDirtyRect(const LayoutRect& filterBoxRect, const LayoutRect& dirtyRect)
394{
395 // The result of this function is the area in the "filterBoxRect" that needs to be repainted, so that we fully cover the "dirtyRect".
396 auto rectForRepaint = dirtyRect;
397 if (hasFilterThatMovesPixels())
398 rectForRepaint += outsets();
399 rectForRepaint.intersect(filterBoxRect);
400 return rectForRepaint;
401}

403ImageBuffer* CSSFilter::output() const
404{
405 if (m_filterRenderer && m_filterRenderer->hasResult())
406 return m_filterRenderer->output();
407
408 return m_effects.last()->imageBufferResult();

411void CSSFilter::setSourceImageRect(const FloatRect& sourceImageRect)
412{
413 m_sourceDrawingRegion = sourceImageRect;
414 setMaxEffectRects(sourceImageRect);
415 setFilterRegion(sourceImageRect);
416 m_graphicsBufferAttached = false;
417}
418
419void CSSFilter::setMaxEffectRects(const FloatRect& effectRect)
420{
421 for (auto& effect : m_effects)
422 effect->setMaxEffectRect(effectRect);
423}
424
425IntRect CSSFilter::outputRect() const
426{
427 auto& lastEffect = m_effects.last().get();
428
429 if (lastEffect.hasResult() || (m_filterRenderer && m_filterRenderer->hasResult()))
430 return lastEffect.requestedRegionOfInputPixelBuffer(IntRect { m_filterRegion });
431
432 return { };
433}
434
435IntOutsets CSSFilter::outsets() const
436{
437 if (!m_hasFilterThatMovesPixels)
438 return { };
439
440 if (!m_outsets.isZero())
441 return m_outsets;
442
443 for (auto& effect : m_effects)
444 m_outsets += effect->outsets();
445 return m_outsets;
446}
