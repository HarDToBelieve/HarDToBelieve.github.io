60bool FlexLayoutAlgorithm::computeNextFlexLine(size_t& nextIndex, Vector<FlexItem>& lineItems, LayoutUnit& sumFlexBaseSize, double& totalFlexGrow, double& totalFlexShrink, double& totalWeightedFlexShrink, LayoutUnit& sumHypotheticalMainSize)
61{
62 lineItems.clear();
63 sumFlexBaseSize = 0_lu;
64 totalFlexGrow = totalFlexShrink = totalWeightedFlexShrink = 0;
65 sumHypotheticalMainSize = 0_lu;
66
67 for (; nextIndex < m_allItems.size(); ++nextIndex) {
68 const auto& flexItem = m_allItems[nextIndex];
69 ASSERT(!flexItem.box.isOutOfFlowPositioned());
70 if (isMultiline() && sumHypotheticalMainSize + flexItem.hypotheticalMainAxisMarginBoxSize() > m_lineBreakLength && !lineItems.isEmpty())
71 break;
72 lineItems.append(flexItem);
73 sumFlexBaseSize += flexItem.flexBaseMarginBoxSize() + m_gapBetweenItems;
74 totalFlexGrow += flexItem.box.style().flexGrow();
75 totalFlexShrink += flexItem.box.style().flexShrink();
76 totalWeightedFlexShrink += flexItem.box.style().flexShrink() * flexItem.flexBaseContentSize;
77 sumHypotheticalMainSize += flexItem.hypotheticalMainAxisMarginBoxSize() + m_gapBetweenItems;
78 }
79
80 if (!lineItems.isEmpty()) {
81 // We added a gap after every item but there shouldn't be one after the last item, so subtract it here. Note that
82 // sums might be negative here due to negative margins in flex items.
83 sumHypotheticalMainSize -= m_gapBetweenItems;
84 sumFlexBaseSize -= m_gapBetweenItems;
85 }
86
87 ASSERT(lineItems.size() > 0 || nextIndex == m_allItems.size());
88 return lineItems.size() > 0;
89}

91LayoutUnit FlexItem::constrainSizeByMinMax(const LayoutUnit size) const
92{
93 return std::max(minMaxSizes.first, std::min(size, minMaxSizes.second));
94}
