39unsigned Grid::numTracks(GridTrackSizingDirection direction) const
40{

46void Grid::ensureGridSize(unsigned maximumRowSize, unsigned maximumColumnSize)
47{
48 ASSERT(static_cast<int>(maximumRowSize) < GridPosition::max() * 2);
49 ASSERT(static_cast<int>(maximumColumnSize) < GridPosition::max() * 2);
50 const size_t oldColumnSize = numTracks(ForColumns);
51 const size_t oldRowSize = numTracks(ForRows);
52 if (maximumRowSize > oldRowSize) {
53 m_grid.grow(maximumRowSize);
54 for (size_t row = oldRowSize; row < maximumRowSize; ++row)
55 m_grid[row].grow(oldColumnSize);
56 }
57
58 if (maximumColumnSize > oldColumnSize) {
59 for (size_t row = 0; row < numTracks(ForRows); ++row)
60 m_grid[row].grow(maximumColumnSize);
61 }
62}

64void Grid::insert(RenderBox& child, const GridArea& area)
65{
66 ASSERT(area.rows.isTranslatedDefinite() && area.columns.isTranslatedDefinite());
67 ensureGridSize(area.rows.endLine(), area.columns.endLine());
68
69 for (auto row : area.rows) {
70 for (auto column : area.columns)
71 m_grid[row][column].append(makeWeakPtr(child));
72 }
73
74 setGridItemArea(child, area);
75}

147void Grid::setNeedsItemsPlacement(bool needsItemsPlacement)
148{
149 m_needsItemsPlacement = needsItemsPlacement;
150
151 if (!needsItemsPlacement) {
152 m_grid.shrinkToFit();
153 return;
154 }
155
156 m_grid.shrink(0);
157 m_gridItemArea.clear();
158 m_explicitRowStart = 0;
159 m_explicitColumnStart = 0;
160 m_autoRepeatEmptyColumns = nullptr;
161 m_autoRepeatEmptyRows = nullptr;
162 m_autoRepeatColumns = 0;
163 m_autoRepeatRows = 0;
164}

166GridIterator::GridIterator(const Grid& grid, GridTrackSizingDirection direction, unsigned fixedTrackIndex, unsigned varyingTrackIndex)
167 : m_grid(grid.m_grid)
168 , m_direction(direction)

179RenderBox* GridIterator::nextGridItem()
180{
181 ASSERT(!m_grid.isEmpty());
182 ASSERT(!m_grid[0].isEmpty());
183
184 unsigned& varyingTrackIndex = (m_direction == ForColumns) ? m_rowIndex : m_columnIndex;
185 const unsigned endOfVaryingTrackIndex = (m_direction == ForColumns) ? m_grid.size() : m_grid[0].size();
186 for (; varyingTrackIndex < endOfVaryingTrackIndex; ++varyingTrackIndex) {
187 const auto& children = m_grid[m_rowIndex][m_columnIndex];
188 if (m_childIndex < children.size())
189 return children[m_childIndex++].get();
190
191 m_childIndex = 0;
192 }
193 return 0;
194}

196bool GridIterator::isEmptyAreaEnough(unsigned rowSpan, unsigned columnSpan) const
197{
198 ASSERT(!m_grid.isEmpty());
199 ASSERT(!m_grid[0].isEmpty());
200
201 // Ignore cells outside current grid as we will grow it later if needed.
202 unsigned maxRows = std::min<unsigned>(m_rowIndex + rowSpan, m_grid.size());
203 unsigned maxColumns = std::min<unsigned>(m_columnIndex + columnSpan, m_grid[0].size());
204
205 // This adds a O(N^2) behavior that shouldn't be a big deal as we expect spanning areas to be small.
206 for (unsigned row = m_rowIndex; row < maxRows; ++row) {
207 for (unsigned column = m_columnIndex; column < maxColumns; ++column) {
208 auto& children = m_grid[row][column];
209 if (!children.isEmpty())
210 return false;
211 }
212 }
213
214 return true;
215}

217std::unique_ptr<GridArea> GridIterator::nextEmptyGridArea(unsigned fixedTrackSpan, unsigned varyingTrackSpan)
218{
219 ASSERT(!m_grid.isEmpty());
220 ASSERT(!m_grid[0].isEmpty());
221 ASSERT(fixedTrackSpan >= 1);
222 ASSERT(varyingTrackSpan >= 1);
223
224 if (m_grid.isEmpty())
225 return nullptr;
226
227 unsigned rowSpan = (m_direction == ForColumns) ? varyingTrackSpan : fixedTrackSpan;
228 unsigned columnSpan = (m_direction == ForColumns) ? fixedTrackSpan : varyingTrackSpan;
229
230 unsigned& varyingTrackIndex = (m_direction == ForColumns) ? m_rowIndex : m_columnIndex;
231 const unsigned endOfVaryingTrackIndex = (m_direction == ForColumns) ? m_grid.size() : m_grid[0].size();
232 for (; varyingTrackIndex < endOfVaryingTrackIndex; ++varyingTrackIndex) {
233 if (isEmptyAreaEnough(rowSpan, columnSpan)) {
234 std::unique_ptr<GridArea> result = makeUnique<GridArea>(GridSpan::translatedDefiniteGridSpan(m_rowIndex, m_rowIndex + rowSpan), GridSpan::translatedDefiniteGridSpan(m_columnIndex, m_columnIndex + columnSpan));
235 // Advance the iterator to avoid an infinite loop where we would return the same grid area over and over.
236 ++varyingTrackIndex;
237 return result;
238 }
239 }
240 return nullptr;
241}
