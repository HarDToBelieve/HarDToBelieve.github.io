160const GridTrackSize& GridTrackSizingAlgorithm::rawGridTrackSize(GridTrackSizingDirection direction, unsigned translatedIndex) const
161{
162 bool isRowAxis = direction == ForColumns;
163 auto& renderStyle = m_renderGrid->style();
164 auto& trackStyles = isRowAxis ? renderStyle.gridColumns() : renderStyle.gridRows();
165 auto& autoRepeatTrackStyles = isRowAxis ? renderStyle.gridAutoRepeatColumns() : renderStyle.gridAutoRepeatRows();
166 auto& autoTrackStyles = isRowAxis ? renderStyle.gridAutoColumns() : renderStyle.gridAutoRows();
167 unsigned insertionPoint = isRowAxis ? renderStyle.gridAutoRepeatColumnsInsertionPoint() : renderStyle.gridAutoRepeatRowsInsertionPoint();
168 unsigned autoRepeatTracksCount = m_grid.autoRepeatTracks(direction);
169
170 // We should not use GridPositionsResolver::explicitGridXXXCount() for this because the
171 // explicit grid might be larger than the number of tracks in grid-template-rows|columns (if
172 // grid-template-areas is specified for example).
173 unsigned explicitTracksCount = trackStyles.size() + autoRepeatTracksCount;
174
175 int untranslatedIndexAsInt = translatedIndex - m_grid.explicitGridStart(direction);
176 unsigned autoTrackStylesSize = autoTrackStyles.size();
177 if (untranslatedIndexAsInt < 0) {
178 int index = untranslatedIndexAsInt % static_cast<int>(autoTrackStylesSize);
179 // We need to traspose the index because the first negative implicit line will get the last defined auto track and so on.
180 index += index ? autoTrackStylesSize : 0;
181 ASSERT(index >= 0);
182 return autoTrackStyles[index];
183 }
184
185 unsigned untranslatedIndex = static_cast<unsigned>(untranslatedIndexAsInt);
186 if (untranslatedIndex >= explicitTracksCount)
187 return autoTrackStyles[(untranslatedIndex - explicitTracksCount) % autoTrackStylesSize];
188
189 if (!autoRepeatTracksCount || untranslatedIndex < insertionPoint)
190 return trackStyles[untranslatedIndex];
191
192 if (untranslatedIndex < (insertionPoint + autoRepeatTracksCount)) {
193 unsigned autoRepeatLocalIndex = untranslatedIndexAsInt - insertionPoint;
194 return autoRepeatTrackStyles[autoRepeatLocalIndex % autoRepeatTrackStyles.size()];
195 }
196
197 return trackStyles[untranslatedIndex - autoRepeatTracksCount];
198}

212LayoutUnit GridTrackSizingAlgorithm::initialBaseSize(const GridTrackSize& trackSize) const
213{
214 const GridLength& gridLength = trackSize.minTrackBreadth();
215 if (gridLength.isFlex())
216 return 0;
217
218 const Length& trackLength = gridLength.length();
219 if (trackLength.isSpecified())
220 return valueForLength(trackLength, std::max<LayoutUnit>(availableSpace().value_or(0), 0));
221
222 ASSERT(trackLength.isMinContent() || trackLength.isAuto() || trackLength.isMaxContent());
223 return 0;
224}

240void GridTrackSizingAlgorithm::sizeTrackToFitNonSpanningItem(const GridSpan& span, RenderBox& gridItem, GridTrack& track)
241{
242 unsigned trackPosition = span.startLine();
243 const auto& trackSize = tracks(m_direction)[trackPosition].cachedTrackSize();
244
245 if (trackSize.hasMinContentMinTrackBreadth()) {
246 track.setBaseSize(std::max(track.baseSize(), m_strategy->minContentForChild(gridItem)));
247 } else if (trackSize.hasMaxContentMinTrackBreadth()) {
248 track.setBaseSize(std::max(track.baseSize(), m_strategy->maxContentForChild(gridItem)));
249 } else if (trackSize.hasAutoMinTrackBreadth()) {
250 track.setBaseSize(std::max(track.baseSize(), m_strategy->minSizeForChild(gridItem)));
251 }
252
253 if (trackSize.hasMinContentMaxTrackBreadth()) {
254 track.setGrowthLimit(std::max(track.growthLimit(), m_strategy->minContentForChild(gridItem)));
255 } else if (trackSize.hasMaxContentOrAutoMaxTrackBreadth()) {
256 LayoutUnit growthLimit = m_strategy->maxContentForChild(gridItem);
257 if (trackSize.isFitContent())
258 growthLimit = std::min(growthLimit, valueForLength(trackSize.fitContentTrackBreadth().length(), availableSpace().value_or(0)));
259 track.setGrowthLimit(std::max(track.growthLimit(), growthLimit));
260 }
261}

303LayoutUnit GridTrackSizingAlgorithm::itemSizeForTrackSizeComputationPhase(TrackSizeComputationPhase phase, RenderBox& gridItem) const
304{
305 switch (phase) {
306 case ResolveIntrinsicMinimums:
307 return m_strategy->minSizeForChild(gridItem);
308 case ResolveContentBasedMinimums:
309 case ResolveIntrinsicMaximums:
310 return m_strategy->minContentForChild(gridItem);
311 case ResolveMaxContentMinimums:
312 case ResolveMaxContentMaximums:
313 return m_strategy->maxContentForChild(gridItem);
314 case MaximizeTracks:
315 ASSERT_NOT_REACHED();
316 return 0;
317 }
318
319 ASSERT_NOT_REACHED();
320 return 0;
321}

323static bool shouldProcessTrackForTrackSizeComputationPhase(TrackSizeComputationPhase phase, const GridTrackSize& trackSize)
324{
325 switch (phase) {
326 case ResolveIntrinsicMinimums:
327 return trackSize.hasIntrinsicMinTrackBreadth();
328 case ResolveContentBasedMinimums:
329 return trackSize.hasMinOrMaxContentMinTrackBreadth();
330 case ResolveMaxContentMinimums:
331 return trackSize.hasMaxContentMinTrackBreadth();
332 case ResolveIntrinsicMaximums:
333 return trackSize.hasIntrinsicMaxTrackBreadth();
334 case ResolveMaxContentMaximums:
335 return trackSize.hasMaxContentOrAutoMaxTrackBreadth();
336 case MaximizeTracks:
337 ASSERT_NOT_REACHED();
338 return false;
339 }
340
341 ASSERT_NOT_REACHED();
342 return false;
343}
344
345static LayoutUnit trackSizeForTrackSizeComputationPhase(TrackSizeComputationPhase phase, GridTrack& track, TrackSizeRestriction restriction)
346{
347 switch (phase) {
348 case ResolveIntrinsicMinimums:
349 case ResolveContentBasedMinimums:
350 case ResolveMaxContentMinimums:
351 case MaximizeTracks:
352 return track.baseSize();
353 case ResolveIntrinsicMaximums:
354 case ResolveMaxContentMaximums:
355 return restriction == AllowInfinity ? track.growthLimit() : track.growthLimitIfNotInfinite();
356 }
357
358 ASSERT_NOT_REACHED();
359 return track.baseSize();
360}

362static void updateTrackSizeForTrackSizeComputationPhase(TrackSizeComputationPhase phase, GridTrack& track)
363{
364 switch (phase) {
365 case ResolveIntrinsicMinimums:
366 case ResolveContentBasedMinimums:
367 case ResolveMaxContentMinimums:
368 track.setBaseSize(track.plannedSize());
369 return;
370 case ResolveIntrinsicMaximums:
371 case ResolveMaxContentMaximums:
372 track.setGrowthLimit(track.plannedSize());
373 return;
374 case MaximizeTracks:
375 ASSERT_NOT_REACHED();
376 return;
377 }
378
379 ASSERT_NOT_REACHED();
380}
381
382static bool trackShouldGrowBeyondGrowthLimitsForTrackSizeComputationPhase(TrackSizeComputationPhase phase, const GridTrackSize& trackSize)
383{
384 switch (phase) {
385 case ResolveIntrinsicMinimums:
386 case ResolveContentBasedMinimums:
387 return trackSize.hasAutoOrMinContentMinTrackBreadthAndIntrinsicMaxTrackBreadth();
388 case ResolveMaxContentMinimums:
389 return trackSize.hasMaxContentMinTrackBreadthAndMaxContentMaxTrackBreadth();
390 case ResolveIntrinsicMaximums:
391 case ResolveMaxContentMaximums:
392 return true;
393 case MaximizeTracks:
394 ASSERT_NOT_REACHED();
395 return false;
396 }
397
398 ASSERT_NOT_REACHED();
399 return false;
400}

402static void markAsInfinitelyGrowableForTrackSizeComputationPhase(TrackSizeComputationPhase phase, GridTrack& track)
403{
404 switch (phase) {
405 case ResolveIntrinsicMinimums:
406 case ResolveContentBasedMinimums:
407 case ResolveMaxContentMinimums:
408 return;
409 case ResolveIntrinsicMaximums:
410 if (trackSizeForTrackSizeComputationPhase(phase, track, AllowInfinity) == infinity && track.plannedSize() != infinity)
411 track.setInfinitelyGrowable(true);
412 return;
413 case ResolveMaxContentMaximums:
414 if (track.infinitelyGrowable())
415 track.setInfinitelyGrowable(false);
416 return;
417 case MaximizeTracks:
418 ASSERT_NOT_REACHED();
419 return;
420 }
421
422 ASSERT_NOT_REACHED();
423}

425template <TrackSizeComputationPhase phase>
426void GridTrackSizingAlgorithm::increaseSizesToAccommodateSpanningItems(const GridItemsSpanGroupRange& gridItemsWithSpan)
427{
428 Vector<GridTrack>& allTracks = tracks(m_direction);
429 for (const auto& trackIndex : m_contentSizedTracksIndex) {
430 GridTrack& track = allTracks[trackIndex];
431 track.setPlannedSize(trackSizeForTrackSizeComputationPhase(phase, track, AllowInfinity));
432 }
433
434 Vector<GridTrack*> growBeyondGrowthLimitsTracks;
435 Vector<GridTrack*> filteredTracks;
436 for (auto it = gridItemsWithSpan.rangeStart; it != gridItemsWithSpan.rangeEnd; ++it) {
437 GridItemWithSpan& gridItemWithSpan = *it;
438 ASSERT(gridItemWithSpan.span().integerSpan() > 1);
439 const GridSpan& itemSpan = gridItemWithSpan.span();
440
441 filteredTracks.shrink(0);
442 growBeyondGrowthLimitsTracks.shrink(0);
443 LayoutUnit spanningTracksSize;
444 for (auto trackPosition : itemSpan) {
445 GridTrack& track = allTracks[trackPosition];
446 const auto& trackSize = track.cachedTrackSize();
447 spanningTracksSize += trackSizeForTrackSizeComputationPhase(phase, track, ForbidInfinity);
448 if (!shouldProcessTrackForTrackSizeComputationPhase(phase, trackSize))
449 continue;
450
451 filteredTracks.append(&track);
452
453 if (trackShouldGrowBeyondGrowthLimitsForTrackSizeComputationPhase(phase, trackSize))
454 growBeyondGrowthLimitsTracks.append(&track);
455 }
456
457 if (filteredTracks.isEmpty())
458 continue;
459
460 spanningTracksSize += m_renderGrid->guttersSize(m_grid, m_direction, itemSpan.startLine(), itemSpan.integerSpan(), availableSpace());
461
462 LayoutUnit extraSpace = itemSizeForTrackSizeComputationPhase(phase, gridItemWithSpan.gridItem()) - spanningTracksSize;
463 extraSpace = std::max<LayoutUnit>(extraSpace, 0);
464 auto& tracksToGrowBeyondGrowthLimits = growBeyondGrowthLimitsTracks.isEmpty() ? filteredTracks : growBeyondGrowthLimitsTracks;
465 distributeSpaceToTracks<phase>(filteredTracks, &tracksToGrowBeyondGrowthLimits, extraSpace);
466 }
467
468 for (const auto& trackIndex : m_contentSizedTracksIndex) {
469 GridTrack& track = allTracks[trackIndex];
470 markAsInfinitelyGrowableForTrackSizeComputationPhase(phase, track);
471 updateTrackSizeForTrackSizeComputationPhase(phase, track);
472 }
473}

475static bool sortByGridTrackGrowthPotential(const GridTrack* track1, const GridTrack* track2)
476{
477 // This check ensures that we respect the irreflexivity property of the strict weak ordering required by std::sort
478 // (forall x: NOT x < x).
479 bool track1HasInfiniteGrowthPotentialWithoutCap = track1->infiniteGrowthPotential() && !track1->growthLimitCap();
480 bool track2HasInfiniteGrowthPotentialWithoutCap = track2->infiniteGrowthPotential() && !track2->growthLimitCap();
481
482 if (track1HasInfiniteGrowthPotentialWithoutCap && track2HasInfiniteGrowthPotentialWithoutCap)
483 return false;
484
485 if (track1HasInfiniteGrowthPotentialWithoutCap || track2HasInfiniteGrowthPotentialWithoutCap)
486 return track2HasInfiniteGrowthPotentialWithoutCap;
487
488 LayoutUnit track1Limit = track1->growthLimitCap().value_or(track1->growthLimit());
489 LayoutUnit track2Limit = track2->growthLimitCap().value_or(track2->growthLimit());
490 return (track1Limit - track1->baseSize()) < (track2Limit - track2->baseSize());
491}

493static void clampGrowthShareIfNeeded(TrackSizeComputationPhase phase, const GridTrack& track, LayoutUnit& growthShare)
494{
495 if (phase != ResolveMaxContentMaximums || !track.growthLimitCap())
496 return;
497
498 LayoutUnit distanceToCap = track.growthLimitCap().value() - track.tempSize();
499 if (distanceToCap <= 0)
500 return;
501
502 growthShare = std::min(growthShare, distanceToCap);
503}

505template <TrackSizeComputationPhase phase>
506void GridTrackSizingAlgorithm::distributeSpaceToTracks(Vector<GridTrack*>& tracks, Vector<GridTrack*>* growBeyondGrowthLimitsTracks, LayoutUnit& freeSpace) const
507{
508 ASSERT(freeSpace >= 0);
509
510 for (auto* track : tracks)
511 track->setTempSize(trackSizeForTrackSizeComputationPhase(phase, *track, ForbidInfinity));
512
513 if (freeSpace > 0) {
514 std::sort(tracks.begin(), tracks.end(), sortByGridTrackGrowthPotential);
515
516 unsigned tracksSize = tracks.size();
517 for (unsigned i = 0; i < tracksSize; ++i) {
518 GridTrack& track = *tracks[i];
519 const LayoutUnit& trackBreadth = trackSizeForTrackSizeComputationPhase(phase, track, ForbidInfinity);
520 bool infiniteGrowthPotential = track.infiniteGrowthPotential();
521 LayoutUnit trackGrowthPotential = infiniteGrowthPotential ? track.growthLimit() : track.growthLimit() - trackBreadth;
522 // Let's avoid computing availableLogicalSpaceShare as much as possible as it's a hot spot in performance tests.
523 if (trackGrowthPotential > 0 || infiniteGrowthPotential) {
524 LayoutUnit availableLogicalSpaceShare = freeSpace / (tracksSize - i);
525 LayoutUnit growthShare = infiniteGrowthPotential ? availableLogicalSpaceShare : std::min(availableLogicalSpaceShare, trackGrowthPotential);
526 clampGrowthShareIfNeeded(phase, track, growthShare);
527 ASSERT_WITH_MESSAGE(growthShare >= 0, "We should never shrink any grid track or else we can't guarantee we abide by our min-sizing function. We can still have 0 as growthShare if the amount of tracks greatly exceeds the freeSpace.");
528 track.growTempSize(growthShare);
529 freeSpace -= growthShare;
530 }
531 }
532 }
533
534 if (freeSpace > 0 && growBeyondGrowthLimitsTracks) {
535 // We need to sort them because there might be tracks with growth limit caps (like the ones
536 // with fit-content()) which cannot indefinitely grow over the limits.
537 if (phase == ResolveMaxContentMaximums)
538 std::sort(growBeyondGrowthLimitsTracks->begin(), growBeyondGrowthLimitsTracks->end(), sortByGridTrackGrowthPotential);
539
540 unsigned tracksGrowingBeyondGrowthLimitsSize = growBeyondGrowthLimitsTracks->size();
541 for (unsigned i = 0; i < tracksGrowingBeyondGrowthLimitsSize; ++i) {
542 GridTrack* track = growBeyondGrowthLimitsTracks->at(i);
543 LayoutUnit growthShare = freeSpace / (tracksGrowingBeyondGrowthLimitsSize - i);
544 clampGrowthShareIfNeeded(phase, *track, growthShare);
545 track->growTempSize(growthShare);
546 freeSpace -= growthShare;
547 }
548 }
549
550 for (auto* track : tracks)
551 track->setPlannedSize(track->plannedSize() == infinity ? track->tempSize() : std::max(track->plannedSize(), track->tempSize()));
552}

554std::optional<LayoutUnit> GridTrackSizingAlgorithm::estimatedGridAreaBreadthForChild(const RenderBox& child, GridTrackSizingDirection direction) const
555{
556 const GridSpan& span = m_grid.gridItemSpan(child, direction);
557 LayoutUnit gridAreaSize;
558 bool gridAreaIsIndefinite = false;
559 std::optional<LayoutUnit> availableSize = availableSpace(direction);
560 for (auto trackPosition : span) {
561 // We may need to estimate the grid area size before running the track sizing algorithm in order to perform the pre-layout of orthogonal items.
562 // We cannot use tracks(direction)[trackPosition].cachedTrackSize() because tracks(direction) is empty, since we are either performing pre-layout
563 // or are running the track sizing algorithm in the opposite direction and haven't run it in the desired direction yet.
564 const auto& trackSize = wasSetup() ? calculateGridTrackSize(direction, trackPosition) : rawGridTrackSize(direction, trackPosition);
565 GridLength maxTrackSize = trackSize.maxTrackBreadth();
566 if (maxTrackSize.isContentSized() || maxTrackSize.isFlex() || isRelativeGridLengthAsAuto(maxTrackSize, direction))
567 gridAreaIsIndefinite = true;
568 else
569 gridAreaSize += valueForLength(maxTrackSize.length(), availableSize.value_or(0_lu));
570 }
571
572 gridAreaSize += m_renderGrid->guttersSize(m_grid, direction, span.startLine(), span.integerSpan(), availableSize);
573
574 GridTrackSizingDirection childInlineDirection = GridLayoutFunctions::flowAwareDirectionForChild(*m_renderGrid, child, ForColumns);
575 if (gridAreaIsIndefinite)
576 return direction == childInlineDirection ? std::make_optional(std::max(child.maxPreferredLogicalWidth(), gridAreaSize)) : std::nullopt;
577 return gridAreaSize;
578}

580std::optional<LayoutUnit> GridTrackSizingAlgorithm::gridAreaBreadthForChild(const RenderBox& child, GridTrackSizingDirection direction) const
581{
582 bool addContentAlignmentOffset =
583 direction == ForColumns && (m_sizingState == RowSizingFirstIteration || m_sizingState == RowSizingExtraIterationForSizeContainment);
584 // To determine the column track's size based on an orthogonal grid item we need it's logical
585 // height, which may depend on the row track's size. It's possible that the row tracks sizing
586 // logic has not been performed yet, so we will need to do an estimation.
587 if (direction == ForRows && (m_sizingState == ColumnSizingFirstIteration || m_sizingState == ColumnSizingSecondIteration)) {
588 ASSERT(GridLayoutFunctions::isOrthogonalChild(*m_renderGrid, child));
589 // FIXME (jfernandez) Content Alignment should account for this heuristic.
590 // https://github.com/w3c/csswg-drafts/issues/2697
591 if (m_sizingState == ColumnSizingFirstIteration)
592 return estimatedGridAreaBreadthForChild(child, ForRows);
593 addContentAlignmentOffset = true;
594 }
595
596 const Vector<GridTrack>& allTracks = tracks(direction);
597 const GridSpan& span = m_grid.gridItemSpan(child, direction);
598 LayoutUnit gridAreaBreadth;
599 for (auto trackPosition : span)
600 gridAreaBreadth += allTracks[trackPosition].baseSize();
601
602 if (addContentAlignmentOffset)
603 gridAreaBreadth += (span.integerSpan() - 1) * m_renderGrid->gridItemOffset(direction);
604
605 gridAreaBreadth += m_renderGrid->guttersSize(m_grid, direction, span.startLine(), span.integerSpan(), availableSpace(direction));
606
607 return gridAreaBreadth;
608}

615bool GridTrackSizingAlgorithm::isIntrinsicSizedGridArea(const RenderBox& child, GridAxis axis) const
616{
617 ASSERT(wasSetup());
618 GridTrackSizingDirection direction = gridDirectionForAxis(axis);
619 const GridSpan& span = m_grid.gridItemSpan(child, direction);
620 for (auto trackPosition : span) {
621 const auto& trackSize = rawGridTrackSize(direction, trackPosition);
622 // We consider fr units as 'auto' for the min sizing function.
623 // FIXME(jfernandez): https://github.com/w3c/csswg-drafts/issues/2611
624 //
625 // The use of AvailableSize function may imply different results
626 // for the same item when assuming indefinite or definite size
627 // constraints depending on the phase we evaluate the item's
628 // baseline participation.
629 // FIXME(jfernandez): https://github.com/w3c/csswg-drafts/issues/3046
630 if (trackSize.isContentSized() || trackSize.isFitContent() || trackSize.minTrackBreadth().isFlex() || (trackSize.maxTrackBreadth().isFlex() && !availableSpace(direction)))
631 return true;
632 }
633 return false;
634}
635
636GridTrackSize GridTrackSizingAlgorithm::calculateGridTrackSize(GridTrackSizingDirection direction, unsigned translatedIndex) const
637{
638 ASSERT(wasSetup());
639 // Collapse empty auto repeat tracks if auto-fit.
640 if (m_grid.hasAutoRepeatEmptyTracks(direction) && m_grid.isEmptyAutoRepeatTrack(direction, translatedIndex))
641 return { Length(LengthType::Fixed), LengthTrackSizing };
642
643 auto& trackSize = rawGridTrackSize(direction, translatedIndex);
644 if (trackSize.isFitContent())
645 return isRelativeGridLengthAsAuto(trackSize.fitContentTrackBreadth(), direction) ? GridTrackSize(Length(LengthType::Auto), Length(LengthType::MaxContent)) : trackSize;
646
647 GridLength minTrackBreadth = trackSize.minTrackBreadth();
648 GridLength maxTrackBreadth = trackSize.maxTrackBreadth();
649 // If the logical width/height of the grid container is indefinite, percentage
650 // values are treated as <auto>.
651 if (isRelativeGridLengthAsAuto(trackSize.minTrackBreadth(), direction))
652 minTrackBreadth = Length(LengthType::Auto);
653 if (isRelativeGridLengthAsAuto(trackSize.maxTrackBreadth(), direction))
654 maxTrackBreadth = Length(LengthType::Auto);
655
656 // Flex sizes are invalid as a min sizing function. However we still can have a flexible |minTrackBreadth|
657 // if the track size is just a flex size (e.g. "1fr"), the spec says that in this case it implies an automatic minimum.
658 if (minTrackBreadth.isFlex())
659 minTrackBreadth = Length(LengthType::Auto);
660
661 return GridTrackSize(minTrackBreadth, maxTrackBreadth);
662}

664double GridTrackSizingAlgorithm::computeFlexFactorUnitSize(const Vector<GridTrack>& tracks, double flexFactorSum, LayoutUnit& leftOverSpace, const Vector<unsigned, 8>& flexibleTracksIndexes, std::unique_ptr<TrackIndexSet> tracksToTreatAsInflexible) const
665{
666 // We want to avoid the effect of flex factors sum below 1 making the factor unit size to grow exponentially.
667 double hypotheticalFactorUnitSize = leftOverSpace / std::max<double>(1, flexFactorSum);
668
669 // product of the hypothetical "flex factor unit" and any flexible track's "flex factor" must be grater than such track's "base size".
670 bool validFlexFactorUnit = true;
671 for (auto index : flexibleTracksIndexes) {
672 if (tracksToTreatAsInflexible && tracksToTreatAsInflexible->contains(index))
673 continue;
674 LayoutUnit baseSize = tracks[index].baseSize();
675 double flexFactor = tracks[index].cachedTrackSize().maxTrackBreadth().flex();
676 // treating all such tracks as inflexible.
677 if (baseSize > hypotheticalFactorUnitSize * flexFactor) {
678 leftOverSpace -= baseSize;
679 flexFactorSum -= flexFactor;
680 if (!tracksToTreatAsInflexible)
681 tracksToTreatAsInflexible = makeUnique<TrackIndexSet>();
682 tracksToTreatAsInflexible->add(index);
683 validFlexFactorUnit = false;
684 }
685 }
686 if (!validFlexFactorUnit)
687 return computeFlexFactorUnitSize(tracks, flexFactorSum, leftOverSpace, flexibleTracksIndexes, WTFMove(tracksToTreatAsInflexible));
688 return hypotheticalFactorUnitSize;
689}
690
691void GridTrackSizingAlgorithm::computeFlexSizedTracksGrowth(double flexFraction, Vector<LayoutUnit>& increments, LayoutUnit& totalGrowth) const
692{
693 size_t numFlexTracks = m_flexibleSizedTracksIndex.size();
694 ASSERT(increments.size() == numFlexTracks);
695 const Vector<GridTrack>& allTracks = tracks(m_direction);
696 for (size_t i = 0; i < numFlexTracks; ++i) {
697 unsigned trackIndex = m_flexibleSizedTracksIndex[i];
698 const auto& trackSize = allTracks[trackIndex].cachedTrackSize();
699 ASSERT(trackSize.maxTrackBreadth().isFlex());
700 LayoutUnit oldBaseSize = allTracks[trackIndex].baseSize();
701 LayoutUnit newBaseSize = std::max(oldBaseSize, LayoutUnit(flexFraction * trackSize.maxTrackBreadth().flex()));
702 increments[i] = newBaseSize - oldBaseSize;
703 totalGrowth += increments[i];
704 }
705}

707double GridTrackSizingAlgorithm::findFrUnitSize(const GridSpan& tracksSpan, LayoutUnit leftOverSpace) const
708{
709 if (leftOverSpace <= 0)
710 return 0;
711
712 const Vector<GridTrack>& allTracks = tracks(m_direction);
713 double flexFactorSum = 0;
714 Vector<unsigned, 8> flexibleTracksIndexes;
715 for (auto trackIndex : tracksSpan) {
716 const auto& trackSize = allTracks[trackIndex].cachedTrackSize();
717 if (!trackSize.maxTrackBreadth().isFlex())
718 leftOverSpace -= allTracks[trackIndex].baseSize();
719 else {
720 double flexFactor = trackSize.maxTrackBreadth().flex();
721 flexibleTracksIndexes.append(trackIndex);
722 flexFactorSum += flexFactor;
723 }
724 }
725 // We don't remove the gutters from left_over_space here, because that was already done before.
726
727 // The function is not called if we don't have <flex> grid tracks.
728 ASSERT(!flexibleTracksIndexes.isEmpty());
729
730 return computeFlexFactorUnitSize(allTracks, flexFactorSum, leftOverSpace, flexibleTracksIndexes);
731}

799LayoutUnit GridTrackSizingAlgorithmStrategy::minSizeForChild(RenderBox& child) const
800{
801 GridTrackSizingDirection childInlineDirection = GridLayoutFunctions::flowAwareDirectionForChild(*renderGrid(), child, ForColumns);
802 bool isRowAxis = direction() == childInlineDirection;
803 const Length& childSize = isRowAxis ? child.style().logicalWidth() : child.style().logicalHeight();
804 if (!childSize.isAuto() && !childSize.isPercentOrCalculated())
805 return minContentForChild(child);
806
807 const Length& childMinSize = isRowAxis ? child.style().logicalMinWidth() : child.style().logicalMinHeight();
808 bool overflowIsVisible = isRowAxis ? child.style().overflowInlineDirection() == Overflow::Visible : child.style().overflowBlockDirection() == Overflow::Visible;
809 LayoutUnit baselineShim = m_algorithm.baselineOffsetForChild(child, gridAxisForDirection(direction()));
810
811 if (childMinSize.isAuto() && overflowIsVisible) {
812 auto minSize = minContentForChild(child);
813 LayoutUnit maxBreadth;
814 auto allTracks = m_algorithm.tracks(direction());
815 for (auto trackPosition : m_algorithm.grid().gridItemSpan(child, direction())) {
816 const auto& trackSize = allTracks[trackPosition].cachedTrackSize();
817 if (!trackSize.hasFixedMaxTrackBreadth())
818 return minSize;
819 maxBreadth += valueForLength(trackSize.maxTrackBreadth().length(), availableSpace().value_or(0_lu));
820 }
821 if (minSize > maxBreadth) {
822 auto marginAndBorderAndPadding = GridLayoutFunctions::marginLogicalSizeForChild(*renderGrid(), direction(), child);
823 marginAndBorderAndPadding += isRowAxis ? child.borderAndPaddingLogicalWidth() : child.borderAndPaddingLogicalHeight();
824 minSize = std::max(maxBreadth, marginAndBorderAndPadding + baselineShim);
825 }
826 return minSize;
827 }
828
829 std::optional<LayoutUnit> gridAreaSize = m_algorithm.gridAreaBreadthForChild(child, childInlineDirection);
830 return minLogicalSizeForChild(child, childMinSize, gridAreaSize) + baselineShim;
831}

954{

959double IndefiniteSizeStrategy::findUsedFlexFraction(Vector<unsigned>& flexibleSizedTracksIndex, GridTrackSizingDirection direction, std::optional<LayoutUnit> freeSpace) const
960{
961 UNUSED_PARAM(freeSpace);
962 auto allTracks = m_algorithm.tracks(direction);
963
964 double flexFraction = 0;
965 for (const auto& trackIndex : flexibleSizedTracksIndex) {
966 // FIXME: we pass TrackSizing to gridTrackSize() because it does not really matter
967 // as we know the track is a flex sized track. It'd be nice not to have to do that.
968 flexFraction = std::max(flexFraction, normalizedFlexFraction(allTracks[trackIndex]));
969 }
970
971 const Grid& grid = m_algorithm.grid();
972 if (!grid.hasGridItems())
973 return flexFraction;
974
975 HashSet<RenderBox*> itemsSet;
976 for (const auto& trackIndex : flexibleSizedTracksIndex) {
977 GridIterator iterator(grid, direction, trackIndex);
978 while (auto* gridItem = iterator.nextGridItem()) {
979 // Do not include already processed items.
980 if (!itemsSet.add(gridItem).isNewEntry)
981 continue;
982
983 const GridSpan& span = grid.gridItemSpan(*gridItem, direction);
984
985 // Removing gutters from the max-content contribution of the item, so they are not taken into account in FindFrUnitSize().
986 LayoutUnit leftOverSpace = maxContentForChild(*gridItem) - renderGrid()->guttersSize(m_algorithm.grid(), direction, span.startLine(), span.integerSpan(), availableSpace());
987 flexFraction = std::max(flexFraction, findFrUnitSize(span, leftOverSpace));
988 }
989 }
990
991 return flexFraction;
992}

994bool IndefiniteSizeStrategy::recomputeUsedFlexFractionIfNeeded(double& flexFraction, LayoutUnit& totalGrowth) const
995{
996 if (direction() == ForColumns)
997 return false;
998
999 const RenderGrid* renderGrid = this->renderGrid();
1000
1001 auto minSize = renderGrid->computeContentLogicalHeight(MinSize, renderGrid->style().logicalMinHeight(), std::nullopt);
1002 auto maxSize = renderGrid->computeContentLogicalHeight(MaxSize, renderGrid->style().logicalMaxHeight(), std::nullopt);
1003
1004 // Redo the flex fraction computation using min|max-height as definite available space in case
1005 // the total height is smaller than min-height or larger than max-height.
1006 LayoutUnit rowsSize = totalGrowth + computeTrackBasedSize();
1007 bool checkMinSize = minSize && rowsSize < minSize.value();
1008 bool checkMaxSize = maxSize && rowsSize > maxSize.value();
1009 if (!checkMinSize && !checkMaxSize)
1010 return false;
1011

1013 const Grid& grid = m_algorithm.grid();
1014 freeSpace = std::max(freeSpace, minSize.value_or(0_lu)) - renderGrid->guttersSize(grid, ForRows, 0, grid.numTracks(ForRows), availableSpace());
1015
1016 size_t numberOfTracks = m_algorithm.tracks(ForRows).size();
1017 flexFraction = findFrUnitSize(GridSpan::translatedDefiniteGridSpan(0, numberOfTracks), freeSpace);
1018 return true;
1019}

1112
1113void GridTrackSizingAlgorithm::initializeTrackSizes()
1114{
1115 ASSERT(m_contentSizedTracksIndex.isEmpty());
1116 ASSERT(m_flexibleSizedTracksIndex.isEmpty());
1117 ASSERT(m_autoSizedTracksForStretchIndex.isEmpty());
1118 ASSERT(!m_hasPercentSizedRowsIndefiniteHeight);
1119
1120 Vector<GridTrack>& allTracks = tracks(m_direction);
1121 const bool indefiniteHeight = m_direction == ForRows && !m_renderGrid->hasDefiniteLogicalHeight();
1122 LayoutUnit maxSize = std::max(0_lu, availableSpace().value_or(0_lu));
1123 // 1. Initialize per Grid track variables.
1124 for (unsigned i = 0; i < allTracks.size(); ++i) {
1125 GridTrack& track = allTracks[i];
1126 const auto& trackSize = calculateGridTrackSize(m_direction, i);
1127 track.setCachedTrackSize(trackSize);
1128 track.setBaseSize(initialBaseSize(trackSize));
1129 track.setGrowthLimit(initialGrowthLimit(trackSize, track.baseSize()));
1130 track.setInfinitelyGrowable(false);
1131
1132 if (trackSize.isFitContent())
1133 track.setGrowthLimitCap(valueForLength(trackSize.fitContentTrackBreadth().length(), maxSize));
1134 if (trackSize.isContentSized())
1135 m_contentSizedTracksIndex.append(i);
1136 if (trackSize.maxTrackBreadth().isFlex())
1137 m_flexibleSizedTracksIndex.append(i);
1138 if (trackSize.hasAutoMaxTrackBreadth() && !trackSize.isFitContent())
1139 m_autoSizedTracksForStretchIndex.append(i);
1140
1141 if (!m_hasPercentSizedRowsIndefiniteHeight && indefiniteHeight) {
1142 auto& rawTrackSize = rawGridTrackSize(m_direction, i);
1143 if (rawTrackSize.minTrackBreadth().isPercentage() || rawTrackSize.maxTrackBreadth().isPercentage())
1144 m_hasPercentSizedRowsIndefiniteHeight = true;
1145 }
1146 }

1149void GridTrackSizingAlgorithm::resolveIntrinsicTrackSizes()
1150{
1151 Vector<GridTrack>& allTracks = tracks(m_direction);
1152 auto handleInfinityGrowthLimit = [&]() {
1153 for (auto trackIndex : m_contentSizedTracksIndex) {
1154 GridTrack& track = allTracks[trackIndex];
1155 if (track.growthLimit() == infinity)
1156 track.setGrowthLimit(track.baseSize());
1157 }
1158 };
1159
1160 if (m_strategy->isComputingSizeContainment()) {
1161 handleInfinityGrowthLimit();
1162 return;
1163 }
1164
1165 Vector<GridItemWithSpan> itemsSortedByIncreasingSpan;
1166 HashSet<RenderBox*> itemsSet;
1167 if (m_grid.hasGridItems()) {
1168 for (auto trackIndex : m_contentSizedTracksIndex) {
1169 GridIterator iterator(m_grid, m_direction, trackIndex);
1170 GridTrack& track = allTracks[trackIndex];
1171
1172 while (auto* gridItem = iterator.nextGridItem()) {
1173 if (itemsSet.add(gridItem).isNewEntry) {
1174 const GridSpan& span = m_grid.gridItemSpan(*gridItem, m_direction);
1175 if (span.integerSpan() == 1)
1176 sizeTrackToFitNonSpanningItem(span, *gridItem, track);
1177 else if (!spanningItemCrossesFlexibleSizedTracks(span))
1178 itemsSortedByIncreasingSpan.append(GridItemWithSpan(*gridItem, span));
1179 }
1180 }
1181 }
1182 std::sort(itemsSortedByIncreasingSpan.begin(), itemsSortedByIncreasingSpan.end());
1183 }
1184
1185 auto it = itemsSortedByIncreasingSpan.begin();
1186 auto end = itemsSortedByIncreasingSpan.end();
1187 while (it != end) {
1188 GridItemsSpanGroupRange spanGroupRange = { it, std::upper_bound(it, end, *it) };
1189 increaseSizesToAccommodateSpanningItems<ResolveIntrinsicMinimums>(spanGroupRange);
1190 increaseSizesToAccommodateSpanningItems<ResolveContentBasedMinimums>(spanGroupRange);
1191 increaseSizesToAccommodateSpanningItems<ResolveMaxContentMinimums>(spanGroupRange);
1192 increaseSizesToAccommodateSpanningItems<ResolveIntrinsicMaximums>(spanGroupRange);
1193 increaseSizesToAccommodateSpanningItems<ResolveMaxContentMaximums>(spanGroupRange);
1194 it = spanGroupRange.rangeEnd;
1195 }
1196 handleInfinityGrowthLimit();
1197}

1199void GridTrackSizingAlgorithm::stretchFlexibleTracks(std::optional<LayoutUnit> freeSpace)
1200{
1201 if (m_flexibleSizedTracksIndex.isEmpty())
1202 return;
1203
1204 double flexFraction = m_strategy->findUsedFlexFraction(m_flexibleSizedTracksIndex, m_direction, freeSpace);
1205
1206 LayoutUnit totalGrowth;
1207 Vector<LayoutUnit> increments;
1208 increments.grow(m_flexibleSizedTracksIndex.size());
1209 computeFlexSizedTracksGrowth(flexFraction, increments, totalGrowth);
1210
1211 if (m_strategy->recomputeUsedFlexFractionIfNeeded(flexFraction, totalGrowth)) {
1212 totalGrowth = 0_lu;
1213 computeFlexSizedTracksGrowth(flexFraction, increments, totalGrowth);
1214 }
1215
1216 size_t i = 0;
1217 Vector<GridTrack>& allTracks = tracks(m_direction);
1218 for (auto trackIndex : m_flexibleSizedTracksIndex) {
1219 auto& track = allTracks[trackIndex];
1220 if (LayoutUnit increment = increments[i++])
1221 track.setBaseSize(track.baseSize() + increment);
1222 }
1223 if (this->freeSpace(m_direction))
1224 setFreeSpace(m_direction, this->freeSpace(m_direction).value() - totalGrowth);
1225 m_maxContentSize += totalGrowth;
1226}

1228void GridTrackSizingAlgorithm::stretchAutoTracks()
1229{
1230 auto currentFreeSpace = m_strategy->freeSpaceForStretchAutoTracksStep();
1231 if (m_autoSizedTracksForStretchIndex.isEmpty() || currentFreeSpace <= 0
1232 || (m_renderGrid->contentAlignment(m_direction).distribution() != ContentDistribution::Stretch))
1233 return;
1234
1235 Vector<GridTrack>& allTracks = tracks(m_direction);
1236 unsigned numberOfAutoSizedTracks = m_autoSizedTracksForStretchIndex.size();
1237 LayoutUnit sizeToIncrease = currentFreeSpace / numberOfAutoSizedTracks;
1238 for (const auto& trackIndex : m_autoSizedTracksForStretchIndex) {
1239 auto& track = allTracks[trackIndex];
1240 track.setBaseSize(track.baseSize() + sizeToIncrease);
1241 }
1242 setFreeSpace(m_direction, 0_lu);
1243}

1337void GridTrackSizingAlgorithm::run()
1338{
1339 ASSERT(wasSetup());
1340 StateMachine stateMachine(*this);
1341
1342 // Step 1.
1343 const std::optional<LayoutUnit> initialFreeSpace = freeSpace(m_direction);
1344 initializeTrackSizes();
1345
1346 // Step 2.
1347 if (!m_contentSizedTracksIndex.isEmpty())
1348 resolveIntrinsicTrackSizes();
1349
1350 // This is not exactly a step of the track sizing algorithm, but we use the track sizes computed
1351 // up to this moment (before maximization) to calculate the grid container intrinsic sizes.
1352 computeGridContainerIntrinsicSizes();
1353
1354 if (freeSpace(m_direction)) {
1355 LayoutUnit updatedFreeSpace = freeSpace(m_direction).value() - m_minContentSize;
1356 setFreeSpace(m_direction, updatedFreeSpace);
1357 if (updatedFreeSpace <= 0)
1358 return;
1359 }
1360
1361 // Step 3.
1362 m_strategy->maximizeTracks(tracks(m_direction), m_direction == ForColumns ? m_freeSpaceColumns : m_freeSpaceRows);
1363 if (m_strategy->isComputingSizeContainment())
1364 return;
1365
1366 // Step 4.
1367 stretchFlexibleTracks(initialFreeSpace);
1368
1369 // Step 5.
1370 stretchAutoTracks();
1371}
