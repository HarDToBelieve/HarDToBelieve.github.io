22#pragma once
23
24#include "RoundedRect.h"
25
26namespace WebCore {
27
28class HitTestLocation {
29public:
30 WEBCORE_EXPORT HitTestLocation();
31 HitTestLocation(const LayoutPoint&);
32 HitTestLocation(const FloatPoint&, const FloatQuad&);
33
34 HitTestLocation(const LayoutRect&);
35
36 // Make a copy the HitTestLocation in a new region by applying given offset to internal point and area.
37 HitTestLocation(const HitTestLocation&, const LayoutSize& offset);
38 WEBCORE_EXPORT HitTestLocation(const HitTestLocation&);
39 WEBCORE_EXPORT ~HitTestLocation();
40 HitTestLocation& operator=(const HitTestLocation&);
41
42 const LayoutPoint& point() const { return m_point; }
43 IntPoint roundedPoint() const { return roundedIntPoint(m_point); }
44
45 // Rect-based hit test related methods.
46 bool isRectBasedTest() const { return m_isRectBased; }
47 bool isRectilinear() const { return m_isRectilinear; }
48 LayoutRect boundingBox() const { return m_boundingBox; }
49
50 int topPadding() const { return roundedPoint().y() - m_boundingBox.y(); }
51 int rightPadding() const { return m_boundingBox.maxX() - roundedPoint().x() - 1; }
52 int bottomPadding() const { return m_boundingBox.maxY() - roundedPoint().y() - 1; }
53 int leftPadding() const { return roundedPoint().x() - m_boundingBox.x(); }
54
55 WEBCORE_EXPORT bool intersects(const LayoutRect&) const;
56 bool intersects(const FloatRect&) const;
57 bool intersects(const RoundedRect&) const;
58
59 const FloatPoint& transformedPoint() const { return m_transformedPoint; }
60 const FloatQuad& transformedRect() const { return m_transformedRect; }
61
62private:
63 template<typename RectType> bool intersectsRect(const RectType&) const;
64
65 void move(const LayoutSize&);
66
67 // These are the cached forms of the more accurate point and rect below.
68 LayoutPoint m_point;
69 LayoutRect m_boundingBox;
70
71 FloatPoint m_transformedPoint;
72 FloatQuad m_transformedRect;
73
74 bool m_isRectBased { false };
75 bool m_isRectilinear { true };
76};
77
78} // namespace WebCore
