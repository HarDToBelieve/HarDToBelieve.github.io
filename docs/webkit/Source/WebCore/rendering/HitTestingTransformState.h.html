35namespace WebCore {
36
37// FIXME: Now that TransformState lazily creates its TransformationMatrix it takes up less space.
38// So there's really no need for a ref counted version. So This class should be removed and replaced
39// with TransformState. There are some minor differences (like the way translate() works slightly
40// differently than move()) so care has to be taken when this is done.
41class HitTestingTransformState : public RefCounted<HitTestingTransformState> {
42public:
43 static Ref<HitTestingTransformState> create(const FloatPoint& p, const FloatQuad& quad, const FloatQuad& area)
44 {
45 return adoptRef(*new HitTestingTransformState(p, quad, area));
46 }
47
48 static Ref<HitTestingTransformState> create(const HitTestingTransformState& other)
49 {
50 return adoptRef(*new HitTestingTransformState(other));
51 }
52
53 enum TransformAccumulation { FlattenTransform, AccumulateTransform };
54 void translate(int x, int y, TransformAccumulation);
55 void applyTransform(const TransformationMatrix& transformFromContainer, TransformAccumulation);
56
57 FloatPoint mappedPoint() const;
58 FloatQuad mappedQuad() const;
59 FloatQuad mappedArea() const;
60 LayoutRect boundsOfMappedArea() const;
61 void flatten();
62
63 FloatPoint m_lastPlanarPoint;
64 FloatQuad m_lastPlanarQuad;
65 FloatQuad m_lastPlanarArea;
66 TransformationMatrix m_accumulatedTransform;
67 bool m_accumulatingTransform;
68
69private:
70 HitTestingTransformState(const FloatPoint& p, const FloatQuad& quad, const FloatQuad& area)
71 : m_lastPlanarPoint(p)
72 , m_lastPlanarQuad(quad)
73 , m_lastPlanarArea(area)
74 , m_accumulatingTransform(false)
75 {
76 }
77
78 HitTestingTransformState(const HitTestingTransformState& other)
79 : RefCounted<HitTestingTransformState>()
80 , m_lastPlanarPoint(other.m_lastPlanarPoint)
81 , m_lastPlanarQuad(other.m_lastPlanarQuad)
82 , m_lastPlanarArea(other.m_lastPlanarArea)
83 , m_accumulatedTransform(other.m_accumulatedTransform)
84 , m_accumulatingTransform(other.m_accumulatingTransform)
85 {
86 }
87
88 void flattenWithTransform(const TransformationMatrix&);
89};
90
91} // namespace WebCore
