31namespace WebCore {
32
33LayoutRepainter::LayoutRepainter(RenderElement& object, bool checkForRepaint)
34 : m_object(object)
35 , m_repaintContainer(0)
36 , m_checkForRepaint(checkForRepaint)
37{
38 if (m_checkForRepaint) {
39 m_repaintContainer = m_object.containerForRepaint();
40 m_oldBounds = m_object.clippedOverflowRectForRepaint(m_repaintContainer);
41 m_oldOutlineBox = m_object.outlineBoundsForRepaint(m_repaintContainer);
42 }
43}
44
45bool LayoutRepainter::repaintAfterLayout()
46{
47 return m_checkForRepaint ? m_object.repaintAfterLayoutIfNeeded(m_repaintContainer, m_oldBounds, m_oldOutlineBox) : false;
48}
