113void LegacyInlineBox::outputLineBox(TextStream& stream, bool mark, int depth) const
114{
115 stream << "-------- " << (isDirty() ? "D" : "-") << "-";
116 int printedCharacters = 0;
117 if (mark) {
118 stream << "*";
119 ++printedCharacters;
120 }
121 while (++printedCharacters <= depth * 2)
122 stream << " ";
123 stream << boxName() << " " << FloatRect(x(), y(), width(), height()) << " (" << this << ") renderer->(" << &renderer() << ")";
124 stream.nextLine();
125}

129float LegacyInlineBox::logicalHeight() const
130{
131 if (hasVirtualLogicalHeight())
132 return virtualLogicalHeight();
133
134 if (is<LegacyRootInlineBox>(*this) && downcast<LegacyRootInlineBox>(*this).isForTrailingFloats())
135 return 0;
136
137 const RenderStyle& lineStyle = this->lineStyle();
138 if (renderer().isTextOrLineBreak())
139 return lineStyle.fontMetrics().height();
140 if (is<RenderBox>(renderer()) && parent())
141 return isHorizontal() ? downcast<RenderBox>(renderer()).height() : downcast<RenderBox>(renderer()).width();
142
143 ASSERT(isInlineFlowBox());
144 RenderBoxModelObject* flowObject = boxModelObject();
145 const FontMetrics& fontMetrics = lineStyle.fontMetrics();
146 float result = fontMetrics.height();
147 if (parent())
148 result += flowObject->borderAndPaddingLogicalHeight();
149 return result;
150}

204bool LegacyInlineBox::nextOnLineExists() const
205{
206 if (!m_bitfields.determinedIfNextOnLineExists()) {
207 m_bitfields.setDeterminedIfNextOnLineExists(true);
208
209 if (!parent())
210 m_bitfields.setNextOnLineExists(false);
211 else if (nextOnLine())
212 m_bitfields.setNextOnLineExists(true);
213 else
214 m_bitfields.setNextOnLineExists(parent()->nextOnLineExists());
215 }
216 return m_bitfields.nextOnLineExists();
217}

228LegacyInlineBox* LegacyInlineBox::nextLeafOnLine() const
229{
230 LegacyInlineBox* leaf = nullptr;
231 for (auto* box = nextOnLine(); box && !leaf; box = box->nextOnLine())
232 leaf = box->isLeaf() ? box : downcast<LegacyInlineFlowBox>(*box).firstLeafDescendant();
233 if (!leaf && parent())
234 leaf = parent()->nextLeafOnLine();
235 return leaf;
236}
237
238LegacyInlineBox* LegacyInlineBox::previousLeafOnLine() const
239{
240 LegacyInlineBox* leaf = nullptr;
241 for (auto* box = previousOnLine(); box && !leaf; box = box->previousOnLine())
242 leaf = box->isLeaf() ? box : downcast<LegacyInlineFlowBox>(*box).lastLeafDescendant();
243 if (!leaf && parent())
244 leaf = parent()->previousLeafOnLine();
245 return leaf;
246}

253bool LegacyInlineBox::canAccommodateEllipsis(bool ltr, int blockEdge, int ellipsisWidth) const
254{
255 // Non-replaced elements can always accommodate an ellipsis.
256 if (!renderer().isReplaced())
257 return true;
258
259 IntRect boxRect(left(), 0, m_logicalWidth, 10);
260 IntRect ellipsisRect(ltr ? blockEdge - ellipsisWidth : blockEdge, 0, ellipsisWidth, 10);
261 return !(boxRect.intersects(ellipsisRect));
262}

264float LegacyInlineBox::placeEllipsisBox(bool, float, float, float, float& truncatedWidth, bool&)
265{
266 // Use -1 to mean "we didn't set the position."
267 truncatedWidth += logicalWidth();
268 return -1;
269}

278FloatPoint LegacyInlineBox::locationIncludingFlipping() const
279{
280 if (!renderer().style().isFlippedBlocksWritingMode())
281 return topLeft();
282 RenderBlockFlow& block = root().blockFlow();
283 if (block.style().isHorizontalWritingMode())
284 return { x(), block.height() - height() - y() };
285 return { block.width() - width() - x(), y() };
286}
