233 bool visibleToHitTesting(std::optional<HitTestRequest> hitTestRequest = std::nullopt) const
234 {
235 if (renderer().style().visibility() != Visibility::Visible)
236 return false;
237
238 if ((!hitTestRequest || !hitTestRequest->ignoreCSSPointerEventsProperty()) && renderer().style().pointerEvents() == PointerEvents::None)
239 return false;
240
241 return true;
242 }

343 ADD_BOOLEAN_BITFIELD(dirty, Dirty);
344 ADD_BOOLEAN_BITFIELD(extracted, Extracted);
