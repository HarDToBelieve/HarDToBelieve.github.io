62 // Internet Explorer and Firefox always create a marker for list items, even when the list-style-type is none. We do not make a marker
63 // in the list-style-type: none case, since it is wasteful to do so. However, in order to match other browsers we have to pretend like
64 // an invisible marker exists. The side effect of having an invisible marker is that the quirks mode behavior of shrinking lines with no
65 // text children must not apply. This change also means that gaps will exist between image bullet list items. Even when the list bullet
66 // is an image, the line is still considered to be immune from the quirk.
67 m_hasTextChildren = renderer.style().display() == DisplayType::ListItem;
68 m_hasTextDescendants = m_hasTextChildren;
69 }

220 // Line visual and layout overflow are in the coordinate space of the block. This means that they aren't purely physical directions.
221 // For horizontal-tb and vertical-lr they will match physical directions, but for horizontal-bt and vertical-rl, the top/bottom and left/right
222 // respectively are flipped when compared to their physical counterparts. For example minX is on the left in vertical-lr, but it is on the right in vertical-rl.
223 LayoutRect layoutOverflowRect(LayoutUnit lineTop, LayoutUnit lineBottom) const
224 {
225 return m_overflow ? m_overflow->layoutOverflowRect() : enclosingLayoutRect(frameRectIncludingLineHeight(lineTop, lineBottom));
226 }
227 LayoutUnit logicalTopLayoutOverflow(LayoutUnit lineTop) const
228 {
229 if (m_overflow)
230 return isHorizontal() ? m_overflow->layoutOverflowRect().y() : m_overflow->layoutOverflowRect().x();
231 return lineTop;
232 }
233 LayoutUnit logicalBottomLayoutOverflow(LayoutUnit lineBottom) const
234 {
235 if (m_overflow)
236 return isHorizontal() ? m_overflow->layoutOverflowRect().maxY() : m_overflow->layoutOverflowRect().maxX();
237 return lineBottom;
238 }
239 LayoutRect logicalLayoutOverflowRect(LayoutUnit lineTop, LayoutUnit lineBottom) const
240 {
241 LayoutRect result = layoutOverflowRect(lineTop, lineBottom);
242 if (!renderer().isHorizontalWritingMode())
243 result = result.transposedRect();
244 return result;
245 }
246
247 LayoutRect visualOverflowRect(LayoutUnit lineTop, LayoutUnit lineBottom) const
248 {
249 return m_overflow ? m_overflow->visualOverflowRect() : enclosingLayoutRect(frameRectIncludingLineHeight(lineTop, lineBottom));
250 }
251 LayoutUnit logicalLeftVisualOverflow() const { return m_overflow ? (isHorizontal() ? m_overflow->visualOverflowRect().x() : m_overflow->visualOverflowRect().y()) : LayoutUnit(logicalLeft()); }
252 LayoutUnit logicalRightVisualOverflow() const { return m_overflow ? (isHorizontal() ? m_overflow->visualOverflowRect().maxX() : m_overflow->visualOverflowRect().maxY()) : LayoutUnit(ceilf(logicalRight())); }
253 LayoutUnit logicalTopVisualOverflow(LayoutUnit lineTop) const
254 {
255 if (m_overflow)
256 return isHorizontal() ? m_overflow->visualOverflowRect().y() : m_overflow->visualOverflowRect().x();
257 return lineTop;
258 }
259 LayoutUnit logicalBottomVisualOverflow(LayoutUnit lineBottom) const
260 {
261 if (m_overflow)
262 return isHorizontal() ? m_overflow->visualOverflowRect().maxY() : m_overflow->visualOverflowRect().maxX();
263 return lineBottom;
264 }
265 LayoutRect logicalVisualOverflowRect(LayoutUnit lineTop, LayoutUnit lineBottom) const
266 {
267 LayoutRect result = visualOverflowRect(lineTop, lineBottom);
268 if (!renderer().isHorizontalWritingMode())
269 result = result.transposedRect();
270 return result;
271 }
