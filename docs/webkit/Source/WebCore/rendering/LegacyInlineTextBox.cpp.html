161RenderObject::HighlightState LegacyInlineTextBox::selectionState()
162{
163 auto state = renderer().view().selection().highlightStateForTextBox(renderer(), selectableRange());
164
165 // FIXME: this code mutates selection state, but it's used at a simple getter elsewhere
166 // in this file. This code should likely live in HighlightData, or somewhere else.
167 // <rdar://problem/58125978>
168 // https://bugs.webkit.org/show_bug.cgi?id=205528
169 // If there are ellipsis following, make sure their selection is updated.
170 if (m_truncation != cNoTruncation && root().ellipsisBox()) {
171 LegacyEllipsisBox* ellipsis = root().ellipsisBox();
172 if (state != RenderObject::HighlightState::None) {
173 auto [selectionStart, selectionEnd] = selectionStartEnd();
174 // The ellipsis should be considered to be selected if the end of
175 // the selection is past the beginning of the truncation and the
176 // beginning of the selection is before or at the beginning of the
177 // truncation.
178 ellipsis->setSelectionState(selectionEnd >= m_truncation && selectionStart <= m_truncation ?
179 RenderObject::HighlightState::Inside : RenderObject::HighlightState::None);
180 } else
181 ellipsis->setSelectionState(RenderObject::HighlightState::None);
182 }
183
184 return state;
185}

192LayoutRect snappedSelectionRect(const LayoutRect& selectionRect, float logicalRight, float selectionTop, float selectionHeight, bool isHorizontal)
193{
194 auto snappedSelectionRect = enclosingIntRect(selectionRect);
195 LayoutUnit logicalWidth = snappedSelectionRect.width();
196 if (snappedSelectionRect.x() > logicalRight)
197 logicalWidth = 0;
198 else if (snappedSelectionRect.maxX() > logicalRight)
199 logicalWidth = logicalRight - snappedSelectionRect.x();
200
201 LayoutPoint topPoint;
202 LayoutUnit width;
203 LayoutUnit height;
204 if (isHorizontal) {
205 topPoint = LayoutPoint { snappedSelectionRect.x(), selectionTop };
206 width = logicalWidth;
207 height = selectionHeight;
208 } else {
209 topPoint = LayoutPoint { selectionTop, snappedSelectionRect.x() };
210 width = selectionHeight;
211 height = logicalWidth;
212 }
213 return LayoutRect { topPoint, LayoutSize { width, height } };
214}

216// FIXME: Share more code with paintMarkedTextBackground().
217LayoutRect LegacyInlineTextBox::localSelectionRect(unsigned startPos, unsigned endPos) const
218{
219 auto [clampedStart, clampedEnd] = selectableRange().clamp(startPos, endPos);
220
221 if (clampedStart >= clampedEnd && !(startPos == endPos && startPos >= start() && startPos <= (start() + len())))
222 return { };
223
224 LayoutUnit selectionTop = this->selectionTop();
225 LayoutUnit selectionHeight = this->selectionHeight();
226
227 TextRun textRun = createTextRun();
228
229 LayoutRect selectionRect { LayoutUnit(logicalLeft()), selectionTop, LayoutUnit(logicalWidth()), selectionHeight };
230 // Avoid measuring the text when the entire line box is selected as an optimization.
231 if (clampedStart || clampedEnd != textRun.length())
232 lineFont().adjustSelectionRectForText(textRun, selectionRect, clampedStart, clampedEnd);
233 // FIXME: The computation of the snapped selection rect differs from the computation of this rect
234 // in paintMarkedTextBackground(). See <https://bugs.webkit.org/show_bug.cgi?id=138913>.
235 return snappedSelectionRect(selectionRect, logicalRight(), selectionTop, selectionHeight, isHorizontal());
236}

260float LegacyInlineTextBox::placeEllipsisBox(bool flowIsLTR, float visibleLeftEdge, float visibleRightEdge, float ellipsisWidth, float &truncatedWidth, bool& foundBox)
261{
262 if (foundBox) {
263 m_truncation = cFullTruncation;
264 return -1;
265 }
266
267 // For LTR this is the left edge of the box, for RTL, the right edge in parent coordinates.
268 float ellipsisX = flowIsLTR ? visibleRightEdge - ellipsisWidth : visibleLeftEdge + ellipsisWidth;
269
270 // Criteria for full truncation:
271 // LTR: the left edge of the ellipsis is to the left of our text run.
272 // RTL: the right edge of the ellipsis is to the right of our text run.
273 bool ltrFullTruncation = flowIsLTR && ellipsisX <= left();
274 bool rtlFullTruncation = !flowIsLTR && ellipsisX >= left() + logicalWidth();
275 if (ltrFullTruncation || rtlFullTruncation) {
276 // Too far. Just set full truncation, but return -1 and let the ellipsis just be placed at the edge of the box.
277 m_truncation = cFullTruncation;
278 foundBox = true;
279 return -1;
280 }
281
282 bool ltrEllipsisWithinBox = flowIsLTR && (ellipsisX < right());
283 bool rtlEllipsisWithinBox = !flowIsLTR && (ellipsisX > left());
284 if (ltrEllipsisWithinBox || rtlEllipsisWithinBox) {
285 foundBox = true;
286
287 // The inline box may have different directionality than it's parent. Since truncation
288 // behavior depends both on both the parent and the inline block's directionality, we
289 // must keep track of these separately.
290 bool ltr = isLeftToRightDirection();
291 if (ltr != flowIsLTR) {
292 // Width in pixels of the visible portion of the box, excluding the ellipsis.
293 int visibleBoxWidth = visibleRightEdge - visibleLeftEdge - ellipsisWidth;
294 ellipsisX = ltr ? left() + visibleBoxWidth : right() - visibleBoxWidth;
295 }
296
297 int offset = offsetForPosition(ellipsisX, false);
298 if (!offset) {
299 // No characters should be rendered. Set ourselves to full truncation and place the ellipsis at the min of our start
300 // and the ellipsis edge.
301 m_truncation = cFullTruncation;
302 truncatedWidth += ellipsisWidth;
303 return flowIsLTR ? std::min(ellipsisX, x()) : std::max(ellipsisX, right() - ellipsisWidth);
304 }
305
306 // Set the truncation index on the text run.
307 m_truncation = offset;
308
309 // If we got here that means that we were only partially truncated and we need to return the pixel offset at which
310 // to place the ellipsis.
311 float widthOfVisibleText = renderer().width(m_start, offset, textPos(), isFirstLine());
312
313 // The ellipsis needs to be placed just after the last visible character.
314 // Where "after" is defined by the flow directionality, not the inline
315 // box directionality.
316 // e.g. In the case of an LTR inline box truncated in an RTL flow then we can
317 // have a situation such as |Hello| -> |...He|
318 truncatedWidth += widthOfVisibleText + ellipsisWidth;
319 if (flowIsLTR)
320 return left() + widthOfVisibleText;
321
322 return right() - widthOfVisibleText - ellipsisWidth;
323 }
324 truncatedWidth += logicalWidth();
325 return -1;
326}

333bool LegacyInlineTextBox::nodeAtPoint(const HitTestRequest& request, HitTestResult& result, const HitTestLocation& locationInContainer, const LayoutPoint& accumulatedOffset, LayoutUnit /* lineTop */, LayoutUnit /*lineBottom*/,
334 HitTestAction /*hitTestAction*/)
335{
336 if (!visibleToHitTesting(request))
337 return false;
338
339 if (isLineBreak())
340 return false;
341
342 if (m_truncation == cFullTruncation)
343 return false;
344
345 FloatRect rect(locationIncludingFlipping(), size());
346 // Make sure truncated text is ignored while hittesting.
347 if (m_truncation != cNoTruncation) {
348 LayoutUnit widthOfVisibleText { renderer().width(m_start, m_truncation, textPos(), isFirstLine()) };
349
350 if (isHorizontal())
351 renderer().style().isLeftToRightDirection() ? rect.setWidth(widthOfVisibleText) : rect.shiftXEdgeTo(right() - widthOfVisibleText);
352 else
353 rect.setHeight(widthOfVisibleText);
354 }
355
356 rect.moveBy(accumulatedOffset);
357
358 if (locationInContainer.intersects(rect)) {
359 renderer().updateHitTestResult(result, flipForWritingMode(locationInContainer.point() - toLayoutSize(accumulatedOffset)));
360 if (result.addNodeToListBasedTestResult(renderer().nodeForHitTest(), request, locationInContainer, rect) == HitTestProgress::Stop)
361 return true;
362 }
363 return false;
364}

366std::optional<bool> LegacyInlineTextBox::emphasisMarkExistsAndIsAbove(const RenderStyle& style) const
367{
368 // This function returns true if there are text emphasis marks and they are suppressed by ruby text.
369 if (style.textEmphasisMark() == TextEmphasisMark::None)
370 return std::nullopt;
371
372 const OptionSet<TextEmphasisPosition> horizontalMask { TextEmphasisPosition::Left, TextEmphasisPosition::Right };
373
374 auto emphasisPosition = style.textEmphasisPosition();
375 auto emphasisPositionHorizontalValue = emphasisPosition & horizontalMask;
376 ASSERT(!((emphasisPosition & TextEmphasisPosition::Over) && (emphasisPosition & TextEmphasisPosition::Under)));
377 ASSERT(emphasisPositionHorizontalValue != horizontalMask);
378
379 bool isAbove = false;
380 if (!emphasisPositionHorizontalValue)
381 isAbove = emphasisPosition.contains(TextEmphasisPosition::Over);
382 else if (style.isHorizontalWritingMode())
383 isAbove = emphasisPosition.contains(TextEmphasisPosition::Over);
384 else
385 isAbove = emphasisPositionHorizontalValue == TextEmphasisPosition::Right;
386
387 if ((style.isHorizontalWritingMode() && (emphasisPosition & TextEmphasisPosition::Under))
388 || (!style.isHorizontalWritingMode() && (emphasisPosition & TextEmphasisPosition::Left)))
389 return isAbove; // Ruby text is always over, so it cannot suppress emphasis marks under.
390
391 RenderBlock* containingBlock = renderer().containingBlock();
392 if (!containingBlock || !containingBlock->isRubyBase())
393 return isAbove; // This text is not inside a ruby base, so it does not have ruby text over it.
394
395 if (!is<RenderRubyRun>(*containingBlock->parent()))
396 return isAbove; // Cannot get the ruby text.
397
398 RenderRubyText* rubyText = downcast<RenderRubyRun>(*containingBlock->parent()).rubyText();
399
400 // The emphasis marks over are suppressed only if there is a ruby text box and it not empty.
401 if (rubyText && rubyText->hasLines())
402 return std::nullopt;
403
404 return isAbove;
405}

415void LegacyInlineTextBox::paint(PaintInfo& paintInfo, const LayoutPoint& paintOffset, LayoutUnit /*lineTop*/, LayoutUnit /*lineBottom*/)
416{
417 if (isLineBreak() || !paintInfo.shouldPaintWithinRoot(renderer()) || renderer().style().visibility() != Visibility::Visible
418 || m_truncation == cFullTruncation || paintInfo.phase == PaintPhase::Outline || !hasTextContent())
419 return;
420
421 ASSERT(paintInfo.phase != PaintPhase::SelfOutline && paintInfo.phase != PaintPhase::ChildOutlines);
422
423 LayoutUnit logicalLeftSide = logicalLeftVisualOverflow();
424 LayoutUnit logicalRightSide = logicalRightVisualOverflow();
425 LayoutUnit logicalStart = logicalLeftSide + (isHorizontal() ? paintOffset.x() : paintOffset.y());
426 LayoutUnit logicalExtent = logicalRightSide - logicalLeftSide;
427
428 LayoutUnit paintEnd = isHorizontal() ? paintInfo.rect.maxX() : paintInfo.rect.maxY();
429 LayoutUnit paintStart = isHorizontal() ? paintInfo.rect.x() : paintInfo.rect.y();
430
431 FloatPoint localPaintOffset(paintOffset);
432
433 if (logicalStart >= paintEnd || logicalStart + logicalExtent <= paintStart)
434 return;
435
436 bool isPrinting = renderer().document().printing();
437
438 // Determine whether or not we're selected.
439 bool haveSelection = !isPrinting && paintInfo.phase != PaintPhase::TextClip && selectionState() != RenderObject::HighlightState::None;
440 if (!haveSelection && paintInfo.phase == PaintPhase::Selection) {
441 // When only painting the selection, don't bother to paint if there is none.
442 return;
443 }
444
445 if (m_truncation != cNoTruncation) {
446 if (renderer().containingBlock()->style().isLeftToRightDirection() != isLeftToRightDirection()) {
447 // Make the visible fragment of text hug the edge closest to the rest of the run by moving the origin
448 // at which we start drawing text.
449 // e.g. In the case of LTR text truncated in an RTL Context, the correct behavior is:
450 // |Hello|CBA| -> |...He|CBA|
451 // In order to draw the fragment "He" aligned to the right edge of it's box, we need to start drawing
452 // farther to the right.
453 // NOTE: WebKit's behavior differs from that of IE which appears to just overlay the ellipsis on top of the
454 // truncated string i.e. |Hello|CBA| -> |...lo|CBA|
455 LayoutUnit widthOfVisibleText { renderer().width(m_start, m_truncation, textPos(), isFirstLine()) };
456 LayoutUnit widthOfHiddenText { logicalWidth() - widthOfVisibleText };
457 LayoutSize truncationOffset(isLeftToRightDirection() ? widthOfHiddenText : -widthOfHiddenText, 0_lu);
458 localPaintOffset.move(isHorizontal() ? truncationOffset : truncationOffset.transposedSize());
459 }
460 }
461
462 GraphicsContext& context = paintInfo.context();
463
464 const RenderStyle& lineStyle = this->lineStyle();
465
466 localPaintOffset.move(0, lineStyle.isHorizontalWritingMode() ? 0 : -logicalHeight());
467
468 FloatPoint boxOrigin = locationIncludingFlipping();
469 boxOrigin.moveBy(localPaintOffset);
470 FloatRect boxRect(boxOrigin, FloatSize(logicalWidth(), logicalHeight()));
471
472 if (paintInfo.phase == PaintPhase::EventRegion) {
473 if (visibleToHitTesting())
474 paintInfo.eventRegionContext->unite(enclosingIntRect(boxRect), renderer().style());
475 return;
476 }
477
478 auto* combinedText = this->combinedText();
479
480 bool shouldRotate = !isHorizontal() && !combinedText;
481 if (shouldRotate)
482 context.concatCTM(rotation(boxRect, Clockwise));
483
484 // Determine whether or not we have composition underlines to draw.
485 bool containsComposition = renderer().textNode() && renderer().frame().editor().compositionNode() == renderer().textNode();
486 bool useCustomUnderlines = containsComposition && renderer().frame().editor().compositionUsesCustomUnderlines();
487
488 // 1. Paint backgrounds behind text if needed. Examples of such backgrounds include selection
489 // and composition underlines.
490 if (paintInfo.phase != PaintPhase::Selection && paintInfo.phase != PaintPhase::TextClip && !isPrinting) {
491 if (containsComposition && !useCustomUnderlines)
492 paintCompositionBackground(paintInfo, boxOrigin);
493
494 auto selectableRange = this->selectableRange();
495
496 Vector<MarkedText> markedTexts;
497 markedTexts.appendVector(MarkedText::collectForDocumentMarkers(renderer(), selectableRange, MarkedText::PaintPhase::Background));
498 markedTexts.appendVector(MarkedText::collectForHighlights(renderer(), parent()->renderer(), selectableRange, MarkedText::PaintPhase::Background));
499
500#if ENABLE(TEXT_SELECTION)
501 if (haveSelection && !useCustomUnderlines && !context.paintingDisabled()) {
502 auto selectionMarkedText = createMarkedTextFromSelectionInBox(*this);
503 if (!selectionMarkedText.isEmpty())
504 markedTexts.append(WTFMove(selectionMarkedText));
505 }
506#endif
507 auto styledMarkedTexts = subdivideAndResolveStyle(markedTexts, renderer(), isFirstLine(), paintInfo);
508
509 // Coalesce styles of adjacent marked texts to minimize the number of drawing commands.
510 auto coalescedStyledMarkedTexts = coalesceAdjacentMarkedTexts(styledMarkedTexts, &MarkedTextStyle::areBackgroundMarkedTextStylesEqual);
511
512 paintMarkedTexts(paintInfo, MarkedText::PaintPhase::Background, boxRect, coalescedStyledMarkedTexts);
513 }
514
515 // FIXME: Right now, LegacyInlineTextBoxes never call addRelevantUnpaintedObject() even though they might
516 // legitimately be unpainted if they are waiting on a slow-loading web font. We should fix that, and
517 // when we do, we will have to account for the fact the LegacyInlineTextBoxes do not always have unique
518 // renderers and Page currently relies on each unpainted object having a unique renderer.
519 if (paintInfo.phase == PaintPhase::Foreground)
520 renderer().page().addRelevantRepaintedObject(&renderer(), IntRect(boxOrigin.x(), boxOrigin.y(), logicalWidth(), logicalHeight()));
521
522 if (paintInfo.phase == PaintPhase::Foreground)
523 paintPlatformDocumentMarkers(context, boxOrigin);
524
525 // 2. Now paint the foreground, including text and decorations like underline/overline (in quirks mode only).
526 bool shouldPaintSelectionForeground = haveSelection && !useCustomUnderlines;
527 Vector<MarkedText> markedTexts;
528 if (paintInfo.phase != PaintPhase::Selection) {
529 auto selectableRange = this->selectableRange();
530
531 // The marked texts for the gaps between document markers and selection are implicitly created by subdividing the entire line.
532 markedTexts.append({ selectableRange.clamp(m_start), selectableRange.clamp(end()), MarkedText::Unmarked });
533
534 if (!isPrinting) {
535 markedTexts.appendVector(MarkedText::collectForDocumentMarkers(renderer(), selectableRange, MarkedText::PaintPhase::Foreground));
536 markedTexts.appendVector(MarkedText::collectForHighlights(renderer(), parent()->renderer(), selectableRange, MarkedText::PaintPhase::Foreground));
537
538 bool shouldPaintDraggedContent = !(paintInfo.paintBehavior.contains(PaintBehavior::ExcludeSelection));
539 if (shouldPaintDraggedContent) {
540 auto markedTextsForDraggedContent = MarkedText::collectForDraggedContent(renderer(), selectableRange);
541 if (!markedTextsForDraggedContent.isEmpty()) {
542 shouldPaintSelectionForeground = false;
543 markedTexts.appendVector(markedTextsForDraggedContent);
544 }
545 }
546 }
547 }
548 // The selection marked text acts as a placeholder when computing the marked texts for the gaps...
549 if (shouldPaintSelectionForeground) {
550 ASSERT(!isPrinting);
551 auto selectionMarkedText = createMarkedTextFromSelectionInBox(*this);
552 if (!selectionMarkedText.isEmpty())
553 markedTexts.append(WTFMove(selectionMarkedText));
554 }
555
556 auto styledMarkedTexts = subdivideAndResolveStyle(markedTexts, renderer(), isFirstLine(), paintInfo);
557
558 // ... now remove the selection marked text if we are excluding selection.
559 if (!isPrinting && paintInfo.paintBehavior.contains(PaintBehavior::ExcludeSelection)) {
560 styledMarkedTexts.removeAllMatching([] (const StyledMarkedText& markedText) {
561 return markedText.type == MarkedText::Selection;
562 });
563 }
564
565 // Coalesce styles of adjacent marked texts to minimize the number of drawing commands.
566 auto coalescedStyledMarkedTexts = coalesceAdjacentMarkedTexts(styledMarkedTexts, &MarkedTextStyle::areForegroundMarkedTextStylesEqual);
567
568 paintMarkedTexts(paintInfo, MarkedText::PaintPhase::Foreground, boxRect, coalescedStyledMarkedTexts);
569
570 // Paint decorations
571 auto textDecorations = lineStyle.textDecorationsInEffect();
572 bool highlightDecorations = !MarkedText::collectForHighlights(renderer(), parent()->renderer(), selectableRange(), MarkedText::PaintPhase::Decoration).isEmpty();
573 bool lineDecorations = !textDecorations.isEmpty();
574 if ((lineDecorations || highlightDecorations) && paintInfo.phase != PaintPhase::Selection) {
575 TextRun textRun = createTextRun();
576 unsigned length = textRun.length();
577 if (m_truncation != cNoTruncation)
578 length = m_truncation;
579 unsigned selectionStart = 0;
580 unsigned selectionEnd = 0;
581 if (haveSelection)
582 std::tie(selectionStart, selectionEnd) = selectionStartEnd();
583
584 FloatRect textDecorationSelectionClipOutRect;
585 if ((paintInfo.paintBehavior.contains(PaintBehavior::ExcludeSelection)) && selectionStart < selectionEnd && selectionEnd <= length) {
586 textDecorationSelectionClipOutRect = logicalOverflowRect();
587 textDecorationSelectionClipOutRect.moveBy(localPaintOffset);
588 float logicalWidthBeforeRange;
589 float logicalWidthAfterRange;
590 float logicalSelectionWidth = lineFont().widthOfTextRange(textRun, selectionStart, selectionEnd, nullptr, &logicalWidthBeforeRange, &logicalWidthAfterRange);
591 // FIXME: Do we need to handle vertical bottom to top text?
592 if (!isHorizontal()) {
593 textDecorationSelectionClipOutRect.move(0, logicalWidthBeforeRange);
594 textDecorationSelectionClipOutRect.setHeight(logicalSelectionWidth);
595 } else if (direction() == TextDirection::RTL) {
596 textDecorationSelectionClipOutRect.move(logicalWidthAfterRange, 0);
597 textDecorationSelectionClipOutRect.setWidth(logicalSelectionWidth);
598 } else {
599 textDecorationSelectionClipOutRect.move(logicalWidthBeforeRange, 0);
600 textDecorationSelectionClipOutRect.setWidth(logicalSelectionWidth);
601 }
602 }
603
604 // Coalesce styles of adjacent marked texts to minimize the number of drawing commands.
605 auto coalescedStyledMarkedTexts = coalesceAdjacentMarkedTexts(styledMarkedTexts, &MarkedTextStyle::areDecorationMarkedTextStylesEqual);
606
607 paintMarkedTexts(paintInfo, MarkedText::PaintPhase::Decoration, boxRect, coalescedStyledMarkedTexts, textDecorationSelectionClipOutRect);
608 }
609
610 // 3. Paint fancy decorations, including composition underlines and platform-specific underlines for spelling errors, grammar errors, et cetera.
611 if (paintInfo.phase == PaintPhase::Foreground && useCustomUnderlines)
612 paintCompositionUnderlines(paintInfo, boxOrigin);
613

618TextBoxSelectableRange LegacyInlineTextBox::selectableRange() const
619{
620 // Fix up the offset if we are combined text or have a hyphen because we manage these embellishments.
621 // That is, they are not reflected in renderer().text(). We treat combined text as a single unit.
622 // We also treat the last codepoint in this box and the hyphen as a single unit.
623 auto additionalLengthAtEnd = [&] {
624 if (auto* combinedText = this->combinedText())
625 return combinedText->combinedStringForRendering().length() - m_len;
626 if (hasHyphen())
627 return lineStyle().hyphenString().length();
628 return 0u;
629 }();
630
631 auto truncation = [&]() -> std::optional<unsigned> {
632 if (m_truncation == cNoTruncation)
633 return { };
634 if (m_truncation == cFullTruncation)
635 return std::numeric_limits<unsigned>::max();
636 return m_truncation;
637 }();
638
639 return {
640 m_start,
641 m_len,
642 additionalLengthAtEnd,
643 isLineBreak(),
644 truncation
645 };
646}

676FloatRect LegacyInlineTextBox::calculateDocumentMarkerBounds(const MarkedText& markedText) const
677{
678 auto& font = lineFont();
679 auto ascent = font.fontMetrics().ascent();
680 auto fontSize = std::min(std::max(font.size(), 10.0f), 40.0f);
681 auto y = ascent + 0.11035 * fontSize;
682 auto height = 0.13247 * fontSize;
683
684 // Avoid measuring the text when the entire line box is selected as an optimization.
685 if (markedText.startOffset || markedText.endOffset != selectableRange().clamp(end())) {
686 TextRun run = createTextRun();

688 lineFont().adjustSelectionRectForText(run, selectionRect, markedText.startOffset, markedText.endOffset);

696{

699 return;

764void LegacyInlineTextBox::paintMarkedTextBackground(PaintInfo& paintInfo, const FloatPoint& boxOrigin, const Color& color, unsigned clampedStartOffset, unsigned clampedEndOffset, MarkedTextBackgroundStyle style)
765{
766 if (clampedStartOffset >= clampedEndOffset)
767 return;
768
769 GraphicsContext& context = paintInfo.context();
770 GraphicsContextStateSaver stateSaver { context };
771 updateGraphicsContext(context, TextPaintStyle { color }); // Don't draw text at all!
772
773 // Note that if the text is truncated, we let the thing being painted in the truncation
774 // draw its own highlight.
775 TextRun textRun = createTextRun();
776
777 const LegacyRootInlineBox& rootBox = root();
778 LayoutUnit selectionBottom = rootBox.selectionBottom();
779 LayoutUnit selectionTop = rootBox.selectionTopAdjustedForPrecedingBlock();
780
781 // Use same y positioning and height as for selection, so that when the selection and this subrange are on
782 // the same word there are no pieces sticking out.
783 LayoutUnit deltaY { renderer().style().isFlippedLinesWritingMode() ? selectionBottom - logicalBottom() : logicalTop() - selectionTop };
784 LayoutUnit selectionHeight = std::max<LayoutUnit>(0, selectionBottom - selectionTop);
785
786 LayoutRect selectionRect { LayoutUnit(boxOrigin.x()), LayoutUnit(boxOrigin.y() - deltaY), LayoutUnit(logicalWidth()), selectionHeight };
787 lineFont().adjustSelectionRectForText(textRun, selectionRect, clampedStartOffset, clampedEndOffset);
788
789 // FIXME: Support painting combined text. See <https://bugs.webkit.org/show_bug.cgi?id=180993>.
790 auto backgroundRect = snapRectToDevicePixelsWithWritingDirection(selectionRect, renderer().document().deviceScaleFactor(), textRun.ltr());
791 if (style == MarkedTextBackgroundStyle::Rounded) {
792 backgroundRect.expand(-1, -1);
793 backgroundRect.move(0.5, 0.5);
794 context.fillRoundedRect(FloatRoundedRect { backgroundRect, FloatRoundedRect::Radii { 2 } }, color);
795 return;
796 }
797
798 context.fillRect(backgroundRect, color);
799}

801void LegacyInlineTextBox::paintMarkedTextForeground(PaintInfo& paintInfo, const FloatRect& boxRect, const StyledMarkedText& markedText)
802{
803 if (markedText.startOffset >= markedText.endOffset)
804 return;
805
806 GraphicsContext& context = paintInfo.context();
807 const FontCascade& font = lineFont();
808 const RenderStyle& lineStyle = this->lineStyle();
809
810 float emphasisMarkOffset = 0;
811 std::optional<bool> markExistsAndIsAbove = emphasisMarkExistsAndIsAbove(lineStyle);
812 const AtomString& emphasisMark = markExistsAndIsAbove ? lineStyle.textEmphasisMarkString() : nullAtom();
813 if (!emphasisMark.isEmpty())
814 emphasisMarkOffset = *markExistsAndIsAbove ? -font.fontMetrics().ascent() - font.emphasisMarkDescent(emphasisMark) : font.fontMetrics().descent() + font.emphasisMarkAscent(emphasisMark);
815
816 TextPainter textPainter { context };
817 textPainter.setFont(font);
818 textPainter.setStyle(markedText.style.textStyles);
819 textPainter.setIsHorizontal(isHorizontal());
820 if (markedText.style.textShadow) {
821 textPainter.setShadow(&markedText.style.textShadow.value());
822 if (lineStyle.hasAppleColorFilter())
823 textPainter.setShadowColorFilter(&lineStyle.appleColorFilter());
824 }
825 textPainter.setEmphasisMark(emphasisMark, emphasisMarkOffset, combinedText());
826 if (auto* debugShadow = debugTextShadow())
827 textPainter.setShadow(debugShadow);
828
829 TextRun textRun = createTextRun();
830 textPainter.setGlyphDisplayListIfNeeded(*this, paintInfo, font, context, textRun);
831
832 GraphicsContextStateSaver stateSaver { context, false };
833 if (markedText.type == MarkedText::DraggedContent) {
834 stateSaver.save();
835 context.setAlpha(markedText.style.alpha);
836 }
837 // TextPainter wants the box rectangle and text origin of the entire line box.
838 textPainter.paintRange(textRun, boxRect, textOriginFromBoxRect(boxRect), markedText.startOffset, markedText.endOffset);
839}

841void LegacyInlineTextBox::paintMarkedTextDecoration(PaintInfo& paintInfo, const FloatRect& boxRect, const FloatRect& clipOutRect, const StyledMarkedText& markedText)
842{
843 if (m_truncation == cFullTruncation)
844 return;
845
846 GraphicsContext& context = paintInfo.context();
847 updateGraphicsContext(context, markedText.style.textStyles);
848
849 bool isCombinedText = combinedText();
850 if (isCombinedText)
851 context.concatCTM(rotation(boxRect, Clockwise));
852
853 // 1. Compute text selection
854 unsigned startOffset = markedText.startOffset;
855 unsigned endOffset = markedText.endOffset;
856 if (startOffset >= endOffset)
857 return;
858
859 // Note that if the text is truncated, we let the thing being painted in the truncation
860 // draw its own decoration.
861 TextRun textRun = createTextRun();
862
863 // Avoid measuring the text when the entire line box is selected as an optimization.
864 FloatRect snappedSelectionRect = boxRect;
865 if (startOffset || endOffset != textRun.length()) {
866 LayoutRect selectionRect = { boxRect.x(), boxRect.y(), boxRect.width(), boxRect.height() };
867 lineFont().adjustSelectionRectForText(textRun, selectionRect, startOffset, endOffset);
868 snappedSelectionRect = snapRectToDevicePixelsWithWritingDirection(selectionRect, renderer().document().deviceScaleFactor(), textRun.ltr());
869 }
870
871 // 2. Paint
872 auto textDecorations = lineStyle().textDecorationsInEffect();
873 textDecorations.add(TextDecorationPainter::textDecorationsInEffectForStyle(markedText.style.textDecorationStyles));
874 TextDecorationPainter decorationPainter { context, textDecorations, renderer(), isFirstLine(), lineFont(), markedText.style.textDecorationStyles };
875 decorationPainter.setInlineTextBox(this);
876 decorationPainter.setWidth(snappedSelectionRect.width());
877 decorationPainter.setIsHorizontal(isHorizontal());
878 if (markedText.style.textShadow) {
879 decorationPainter.setTextShadow(&markedText.style.textShadow.value());
880 if (lineStyle().hasAppleColorFilter())
881 decorationPainter.setShadowColorFilter(&lineStyle().appleColorFilter());
882 }
883
884 {
885 GraphicsContextStateSaver stateSaver { context, false };
886 bool isDraggedContent = markedText.type == MarkedText::DraggedContent;
887 if (isDraggedContent || !clipOutRect.isEmpty()) {
888 stateSaver.save();
889 if (isDraggedContent)
890 context.setAlpha(markedText.style.alpha);
891 if (!clipOutRect.isEmpty())
892 context.clipOut(clipOutRect);
893 }
894 decorationPainter.paintTextDecoration(textRun.subRun(startOffset, endOffset - startOffset), textOriginFromBoxRect(snappedSelectionRect), snappedSelectionRect.location());
895 }
896
897 if (isCombinedText)
898 context.concatCTM(rotation(boxRect, Counterclockwise));
899}

901void LegacyInlineTextBox::paintCompositionBackground(PaintInfo& paintInfo, const FloatPoint& boxOrigin)
902{
903 auto selectableRange = this->selectableRange();
904
905 if (!renderer().frame().editor().compositionUsesCustomHighlights()) {
906 paintMarkedTextBackground(paintInfo, boxOrigin, CompositionHighlight::defaultCompositionFillColor, selectableRange.clamp(renderer().frame().editor().compositionStart()), selectableRange.clamp(renderer().frame().editor().compositionEnd()));
907 return;
908 }
909

911 if (highlight.endOffset <= m_start)
912 continue;
913
914 if (highlight.startOffset >= end())
915 break;
916
917 auto [clampedStart, clampedEnd] = selectableRange.clamp(highlight.startOffset, highlight.endOffset);
918 paintMarkedTextBackground(paintInfo, boxOrigin, highlight.color, clampedStart, clampedEnd, MarkedTextBackgroundStyle::Rounded);
919
920 if (highlight.endOffset > end())
921 break;
922 }

925void LegacyInlineTextBox::paintCompositionUnderlines(PaintInfo& paintInfo, const FloatPoint& boxOrigin) const
926{
927 if (m_truncation == cFullTruncation)
928 return;
929
930 for (auto& underline : renderer().frame().editor().customCompositionUnderlines()) {
931 if (underline.endOffset <= m_start) {
932 // Underline is completely before this run. This might be an underline that sits
933 // before the first run we draw, or underlines that were within runs we skipped
934 // due to truncation.
935 continue;
936 }
937
938 if (underline.startOffset >= end())
939 break; // Underline is completely after this run, bail. A later run will paint it.
940
941 // Underline intersects this run. Paint it.
942 paintCompositionUnderline(paintInfo, boxOrigin, underline);
943
944 if (underline.endOffset > end())
945 break; // Underline also runs into the next run. Bail now, no more marker advancement.
946 }
947}

949static inline void mirrorRTLSegment(float logicalWidth, TextDirection direction, float& start, float width)
950{
951 if (direction == TextDirection::LTR)
952 return;
953 start = logicalWidth - width - start;
954}

956void LegacyInlineTextBox::paintCompositionUnderline(PaintInfo& paintInfo, const FloatPoint& boxOrigin, const CompositionUnderline& underline) const
957{
958 if (m_truncation == cFullTruncation)
959 return;
960
961 float start = 0; // start of line to draw, relative to tx
962 float width = logicalWidth(); // how much line to draw
963 bool useWholeWidth = true;
964 unsigned paintStart = m_start;
965 unsigned paintEnd = end();
966 if (paintStart <= underline.startOffset) {
967 paintStart = underline.startOffset;
968 useWholeWidth = false;
969 start = renderer().width(m_start, paintStart - m_start, textPos(), isFirstLine());
970 }
971 if (paintEnd != underline.endOffset) {
972 paintEnd = std::min(paintEnd, (unsigned)underline.endOffset);
973 useWholeWidth = false;
974 }
975 if (m_truncation != cNoTruncation) {
976 paintEnd = std::min(paintEnd, (unsigned)m_start + m_truncation);
977 useWholeWidth = false;
978 }
979 if (!useWholeWidth) {
980 width = renderer().width(paintStart, paintEnd - paintStart, textPos() + start, isFirstLine());
981 mirrorRTLSegment(logicalWidth(), direction(), start, width);
982 }
983
984 // Thick marked text underlines are 2px thick as long as there is room for the 2px line under the baseline.
985 // All other marked text underlines are 1px thick.
986 // If there's not enough space the underline will touch or overlap characters.
987 int lineThickness = 1;
988 int baseline = lineStyle().fontMetrics().ascent();
989 if (underline.thick && logicalHeight() - baseline >= 2)
990 lineThickness = 2;
991
992 // We need to have some space between underlines of subsequent clauses, because some input methods do not use different underline styles for those.
993 // We make each line shorter, which has a harmless side effect of shortening the first and last clauses, too.
994 start += 1;
995 width -= 2;
996
997 GraphicsContext& context = paintInfo.context();
998 Color underlineColor = underline.compositionUnderlineColor == CompositionUnderlineColor::TextColor ? renderer().style().visitedDependentColorWithColorFilter(CSSPropertyWebkitTextFillColor) : renderer().style().colorByApplyingColorFilter(underline.color);
999 context.setStrokeColor(underlineColor);
1000 context.setStrokeThickness(lineThickness);
1001 context.drawLineForText(FloatRect(boxOrigin.x() + start, boxOrigin.y() + logicalHeight() - lineThickness, width, lineThickness), renderer().document().printing());
1002}
