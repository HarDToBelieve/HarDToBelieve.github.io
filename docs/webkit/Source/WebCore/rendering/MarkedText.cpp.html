45Vector<MarkedText> MarkedText::subdivide(const Vector<MarkedText>& markedTexts, OverlapStrategy overlapStrategy)
46{
47 if (markedTexts.isEmpty())
48 return { };
49
50 struct Offset {
51 enum Kind { Begin, End };
52 Kind kind;
53 unsigned value; // Copy of markedText.startOffset/endOffset to avoid the need to branch based on kind.
54 const MarkedText* markedText;
55 };
56
57 // 1. Build table of all offsets.
58 Vector<Offset> offsets;
59 ASSERT(markedTexts.size() < std::numeric_limits<unsigned>::max() / 2);
60 unsigned numberOfMarkedTexts = markedTexts.size();
61 unsigned numberOfOffsets = 2 * numberOfMarkedTexts;
62 offsets.reserveInitialCapacity(numberOfOffsets);
63 for (auto& markedText : markedTexts) {
64 offsets.uncheckedAppend({ Offset::Begin, markedText.startOffset, &markedText });
65 offsets.uncheckedAppend({ Offset::End, markedText.endOffset, &markedText });
66 }
67
68 // 2. Sort offsets such that begin offsets are in paint order and end offsets are in reverse paint order.
69 std::sort(offsets.begin(), offsets.end(), [] (const Offset& a, const Offset& b) {
70 return a.value < b.value || (a.value == b.value && a.kind == b.kind && a.kind == Offset::Begin && a.markedText->type < b.markedText->type)
71 || (a.value == b.value && a.kind == b.kind && a.kind == Offset::End && a.markedText->type > b.markedText->type);
72 });
73
74 // 3. Compute intersection.
75 Vector<MarkedText> result;
76 result.reserveInitialCapacity(numberOfMarkedTexts);
77 HashSet<const MarkedText*> processedMarkedTexts;
78 unsigned offsetSoFar = offsets[0].value;
79 for (unsigned i = 1; i < numberOfOffsets; ++i) {
80 if (offsets[i].value > offsets[i - 1].value) {
81 if (overlapStrategy == OverlapStrategy::Frontmost) {
82 std::optional<unsigned> frontmost;
83 for (unsigned j = 0; j < i; ++j) {
84 if (!processedMarkedTexts.contains(offsets[j].markedText) && (!frontmost || offsets[j].markedText->type > offsets[*frontmost].markedText->type))
85 frontmost = j;
86 }
87 if (frontmost)
88 result.append({ offsetSoFar, offsets[i].value, offsets[*frontmost].markedText->type, offsets[*frontmost].markedText->marker, offsets[*frontmost].markedText->highlightName });
89 } else {
90 // The appended marked texts may not be in paint order. We will fix this up at the end of this function.
91 for (unsigned j = 0; j < i; ++j) {
92 if (!processedMarkedTexts.contains(offsets[j].markedText))
93 result.append({ offsetSoFar, offsets[i].value, offsets[j].markedText->type, offsets[j].markedText->marker, offsets[j].markedText->highlightName });
94 }
95 }
96 offsetSoFar = offsets[i].value;
97 }
98 if (offsets[i].kind == Offset::End)
99 processedMarkedTexts.add(offsets[i].markedText);
100 }
101 // Fix up; sort the marked texts so that they are in paint order.
102 if (overlapStrategy == OverlapStrategy::None)
103 std::sort(result.begin(), result.end(), [] (const MarkedText& a, const MarkedText& b) { return a.startOffset < b.startOffset || (a.startOffset == b.startOffset && a.type < b.type); });
104 return result;
105}

107Vector<MarkedText> MarkedText::collectForHighlights(RenderText& renderer, RenderBoxModelObject& parentRenderer, const TextBoxSelectableRange& selectableRange, PaintPhase phase)
108{
109 Vector<MarkedText> markedTexts;
110 HighlightData highlightData;
111 if (RuntimeEnabledFeatures::sharedFeatures().highlightAPIEnabled()) {
112 auto& parentStyle = parentRenderer.style();
113 if (auto highlightRegister = renderer.document().highlightRegisterIfExists()) {
114 for (auto& highlight : highlightRegister->map()) {
115 auto renderStyle = parentRenderer.getUncachedPseudoStyle({ PseudoId::Highlight, highlight.key }, &parentStyle);
116 if (!renderStyle)
117 continue;
118 if (renderStyle->textDecorationsInEffect().isEmpty() && phase == PaintPhase::Decoration)
119 continue;
120 for (auto& rangeData : highlight.value->rangesData()) {
121 if (!highlightData.setRenderRange(rangeData))
122 continue;
123
124 auto [highlightStart, highlightEnd] = highlightData.rangeForTextBox(renderer, selectableRange);
125 if (highlightStart < highlightEnd)
126 markedTexts.append({ highlightStart, highlightEnd, MarkedText::Highlight, nullptr, highlight.key });
127 }
128 }
129 }
130 }
131#if ENABLE(APP_HIGHLIGHTS)
132 if (auto appHighlightRegister = renderer.document().appHighlightRegisterIfExists()) {
133 if (appHighlightRegister->highlightsVisibility() == HighlightVisibility::Visible) {
134 for (auto& highlight : appHighlightRegister->map()) {
135 for (auto& rangeData : highlight.value->rangesData()) {
136 if (!highlightData.setRenderRange(rangeData))
137 continue;
138
139 auto [highlightStart, highlightEnd] = highlightData.rangeForTextBox(renderer, selectableRange);
140 if (highlightStart < highlightEnd)
141 markedTexts.append({ highlightStart, highlightEnd, MarkedText::AppHighlight });
142 }
143 }
144 }
145 }
146#endif
147 return markedTexts;
148}

150Vector<MarkedText> MarkedText::collectForDocumentMarkers(RenderText& renderer, const TextBoxSelectableRange& selectableRange, PaintPhase phase)
151{
152 if (!renderer.textNode())
153 return { };
154
155 Vector<RenderedDocumentMarker*> markers = renderer.document().markers().markersFor(*renderer.textNode());
156
157 auto markedTextTypeForMarkerType = [] (DocumentMarker::MarkerType type) {
158 switch (type) {
159 case DocumentMarker::Spelling:
160 return MarkedText::SpellingError;
161 case DocumentMarker::Grammar:
162 return MarkedText::GrammarError;
163 case DocumentMarker::CorrectionIndicator:
164 return MarkedText::Correction;
165 case DocumentMarker::TextMatch:
166 return MarkedText::TextMatch;
167 case DocumentMarker::DictationAlternatives:
168 return MarkedText::DictationAlternatives;
169#if PLATFORM(IOS_FAMILY)
170 case DocumentMarker::DictationPhraseWithAlternatives:
171 return MarkedText::DictationPhraseWithAlternatives;
172#endif
173 default:
174 return MarkedText::Unmarked;
175 }
176 };
177
178 Vector<MarkedText> markedTexts;
179 markedTexts.reserveInitialCapacity(markers.size());
180
181 // Give any document markers that touch this run a chance to draw before the text has been drawn.
182 // Note end() points at the last char, not one past it like endOffset and ranges do.
183 for (auto* marker : markers) {
184 // Collect either the background markers or the foreground markers, but not both
185 switch (marker->type()) {
186 case DocumentMarker::Grammar:
187 case DocumentMarker::Spelling:
188 case DocumentMarker::CorrectionIndicator:
189 case DocumentMarker::Replacement:
190 case DocumentMarker::DictationAlternatives:
191#if PLATFORM(IOS_FAMILY)
192 // FIXME: Remove the PLATFORM(IOS_FAMILY)-guard.
193 case DocumentMarker::DictationPhraseWithAlternatives:
194#endif
195 if (phase != MarkedText::PaintPhase::Decoration)
196 continue;
197 break;
198 case DocumentMarker::TextMatch:
199 if (!renderer.frame().editor().markedTextMatchesAreHighlighted())
200 continue;
201 if (phase == MarkedText::PaintPhase::Decoration)
202 continue;
203 break;
204#if ENABLE(TELEPHONE_NUMBER_DETECTION)
205 case DocumentMarker::TelephoneNumber:
206 if (!renderer.frame().editor().markedTextMatchesAreHighlighted())
207 continue;
208 if (phase != MarkedText::PaintPhase::Background)
209 continue;
210 break;
211#endif
212 default:
213 continue;
214 }
215
216 if (marker->endOffset() <= selectableRange.start) {
217 // Marker is completely before this run. This might be a marker that sits before the
218 // first run we draw, or markers that were within runs we skipped due to truncation.
219 continue;
220 }
221
222 if (marker->startOffset() >= selectableRange.start + selectableRange.length) {
223 // Marker is completely after this run, bail. A later run will paint it.
224 break;
225 }
226
227 // Marker intersects this run. Collect it.
228 switch (marker->type()) {
229 case DocumentMarker::Spelling:
230 case DocumentMarker::CorrectionIndicator:
231 case DocumentMarker::DictationAlternatives:
232 case DocumentMarker::Grammar:
233#if PLATFORM(IOS_FAMILY)
234 // FIXME: See <rdar://problem/8933352>. Also, remove the PLATFORM(IOS_FAMILY)-guard.
235 case DocumentMarker::DictationPhraseWithAlternatives:
236#endif
237 case DocumentMarker::TextMatch: {
238 auto [clampedStart, clampedEnd] = selectableRange.clamp(marker->startOffset(), marker->endOffset());
239 markedTexts.uncheckedAppend({ clampedStart, clampedEnd, markedTextTypeForMarkerType(marker->type()), marker });
240 break;
241 }
242 case DocumentMarker::Replacement:
243 break;
244#if ENABLE(TELEPHONE_NUMBER_DETECTION)
245 case DocumentMarker::TelephoneNumber:
246 break;
247#endif
248 default:
249 ASSERT_NOT_REACHED();
250 }
251 }
252 return markedTexts;
253}

255Vector<MarkedText> MarkedText::collectForDraggedContent(RenderText& renderer, const TextBoxSelectableRange& selectableRange)
256{
257 auto draggedContentRanges = renderer.draggedContentRangesBetweenOffsets(selectableRange.start, selectableRange.start + selectableRange.length);
258
259 return draggedContentRanges.map([&](const auto& range) -> MarkedText {
260 return { selectableRange.clamp(range.first), selectableRange.clamp(range.second), MarkedText::DraggedContent };
261 });
262}
