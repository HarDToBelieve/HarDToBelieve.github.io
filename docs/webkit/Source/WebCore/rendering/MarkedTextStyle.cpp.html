36static StyledMarkedText resolveStyleForMarkedText(const MarkedText& markedText, const MarkedTextStyle& baseStyle, const RenderText& renderer, const RenderStyle& lineStyle, const PaintInfo& paintInfo)
37{
38 MarkedTextStyle style = baseStyle;
39 switch (markedText.type) {
40 case MarkedText::Correction:
41 case MarkedText::DictationAlternatives:
42#if PLATFORM(IOS_FAMILY)
43 // FIXME: See <rdar://problem/8933352>. Also, remove the PLATFORM(IOS_FAMILY)-guard.
44 case MarkedText::DictationPhraseWithAlternatives:
45#endif
46 case MarkedText::GrammarError:
47 case MarkedText::SpellingError:
48 case MarkedText::Unmarked:
49 break;
50 case MarkedText::Highlight:
51 if (auto renderStyle = renderer.parent()->getUncachedPseudoStyle({ PseudoId::Highlight, markedText.highlightName }, &renderer.style())) {
52 style.backgroundColor = renderStyle->backgroundColor();
53 style.textStyles.fillColor = renderStyle->computedStrokeColor();
54 style.textStyles.strokeColor = renderStyle->computedStrokeColor();
55
56 auto color = TextDecorationPainter::decorationColor(*renderStyle.get());
57 auto decorationStyle = renderStyle->textDecorationStyle();
58 auto decorations = renderStyle->textDecorationsInEffect();
59
60 if (decorations.contains(TextDecoration::Underline)) {
61 style.textDecorationStyles.underlineColor = color;
62 style.textDecorationStyles.underlineStyle = decorationStyle;
63 }
64 if (decorations.contains(TextDecoration::Overline)) {
65 style.textDecorationStyles.overlineColor = color;
66 style.textDecorationStyles.overlineStyle = decorationStyle;
67 }
68 if (decorations.contains(TextDecoration::LineThrough)) {
69 style.textDecorationStyles.linethroughColor = color;
70 style.textDecorationStyles.linethroughStyle = decorationStyle;
71 }
72 }
73 break;
74#if ENABLE(APP_HIGHLIGHTS)
75 case MarkedText::AppHighlight: {
76 OptionSet<StyleColor::Options> styleColorOptions = { StyleColor::Options::UseSystemAppearance };
77 style.backgroundColor = renderer.theme().appHighlightColor(styleColorOptions);
78 break;
79 }
80#endif
81 case MarkedText::DraggedContent:
82 style.alpha = 0.25;
83 break;
84 case MarkedText::Selection: {
85 style.textStyles = computeTextSelectionPaintStyle(style.textStyles, renderer, lineStyle, paintInfo, style.textShadow);
86
87 Color selectionBackgroundColor = renderer.selectionBackgroundColor();
88 style.backgroundColor = selectionBackgroundColor;
89 if (selectionBackgroundColor.isValid() && selectionBackgroundColor.isVisible() && style.textStyles.fillColor == selectionBackgroundColor)
90 style.backgroundColor = selectionBackgroundColor.invertedColorWithAlpha(1.0);
91 break;
92 }
93 case MarkedText::TextMatch: {
94 // Text matches always use the light system appearance.
95 OptionSet<StyleColor::Options> styleColorOptions = { StyleColor::Options::UseSystemAppearance };
96#if PLATFORM(MAC)
97 style.textStyles.fillColor = renderer.theme().systemColor(CSSValueAppleSystemLabel, styleColorOptions);
98#endif
99 style.backgroundColor = renderer.theme().textSearchHighlightColor(styleColorOptions);
100 break;
101 }
102 }
103 StyledMarkedText styledMarkedText = markedText;
104 styledMarkedText.style = WTFMove(style);
105 return styledMarkedText;
106}

108static MarkedTextStyle computeStyleForUnmarkedMarkedText(const RenderText& renderer, const RenderStyle& lineStyle, bool isFirstLine, const PaintInfo& paintInfo)
109{
110 MarkedTextStyle style;
111 style.textDecorationStyles = TextDecorationPainter::stylesForRenderer(renderer, lineStyle.textDecorationsInEffect(), isFirstLine);
112 style.textStyles = computeTextPaintStyle(renderer.frame(), lineStyle, paintInfo);
113 style.textShadow = ShadowData::clone(paintInfo.forceTextColor() ? nullptr : lineStyle.textShadow());
114 style.alpha = 1;
115 return style;
116}

118Vector<StyledMarkedText> subdivideAndResolveStyle(const Vector<MarkedText>& textsToSubdivide, const RenderText& renderer, bool isFirstLine, const PaintInfo& paintInfo)
119{
120 if (textsToSubdivide.isEmpty())
121 return { };
122
123 Vector<StyledMarkedText> frontmostMarkedTexts;
124
125 auto& lineStyle = isFirstLine ? renderer.firstLineStyle() : renderer.style();
126 auto baseStyle = computeStyleForUnmarkedMarkedText(renderer, lineStyle, isFirstLine, paintInfo);
127
128 if (textsToSubdivide.size() == 1 && textsToSubdivide[0].type == MarkedText::Unmarked) {
129 StyledMarkedText styledMarkedText = textsToSubdivide[0];
130 styledMarkedText.style = WTFMove(baseStyle);
131 return { styledMarkedText };
132 }
133
134 auto markedTexts = MarkedText::subdivide(textsToSubdivide);
135 ASSERT(!markedTexts.isEmpty());
136 if (UNLIKELY(markedTexts.isEmpty()))
137 return { };
138
139 // Compute frontmost overlapping styled marked texts.
140 frontmostMarkedTexts.reserveInitialCapacity(markedTexts.size());
141 frontmostMarkedTexts.uncheckedAppend(resolveStyleForMarkedText(markedTexts[0], baseStyle, renderer, lineStyle, paintInfo));
142 for (auto it = markedTexts.begin() + 1, end = markedTexts.end(); it != end; ++it) {
143 StyledMarkedText& previousStyledMarkedText = frontmostMarkedTexts.last();
144 if (previousStyledMarkedText.startOffset == it->startOffset && previousStyledMarkedText.endOffset == it->endOffset) {
145 // Marked texts completely cover each other.
146 previousStyledMarkedText = resolveStyleForMarkedText(*it, previousStyledMarkedText.style, renderer, lineStyle, paintInfo);
147 continue;
148 }
149 frontmostMarkedTexts.uncheckedAppend(resolveStyleForMarkedText(*it, baseStyle, renderer, lineStyle, paintInfo));
150 }
151
152 return frontmostMarkedTexts;
153}

155Vector<StyledMarkedText> coalesceAdjacentMarkedTexts(const Vector<StyledMarkedText>& textsToCoalesce, MarkedTextStylesEqualityFunction areMarkedTextStylesEqual)
156{
157 if (textsToCoalesce.size() <= 1)
158 return textsToCoalesce;
159
160 auto areAdjacentMarkedTextsWithSameStyle = [&] (const StyledMarkedText& a, const StyledMarkedText& b) {
161 return a.endOffset == b.startOffset && areMarkedTextStylesEqual(a.style, b.style);
162 };
163
164 Vector<StyledMarkedText> styledMarkedTexts;
165 styledMarkedTexts.reserveInitialCapacity(textsToCoalesce.size());
166 styledMarkedTexts.uncheckedAppend(textsToCoalesce[0]);
167 for (auto it = textsToCoalesce.begin() + 1, end = textsToCoalesce.end(); it != end; ++it) {
168 StyledMarkedText& previousStyledMarkedText = styledMarkedTexts.last();
169 if (areAdjacentMarkedTextsWithSameStyle(previousStyledMarkedText, *it)) {
170 previousStyledMarkedText.endOffset = it->endOffset;
171 continue;
172 }
173 styledMarkedTexts.uncheckedAppend(*it);
174 }
175
176 return styledMarkedTexts;
177}
