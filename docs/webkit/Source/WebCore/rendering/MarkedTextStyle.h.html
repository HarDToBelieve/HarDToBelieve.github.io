38struct MarkedTextStyle {
39 static bool areBackgroundMarkedTextStylesEqual(const MarkedTextStyle& a, const MarkedTextStyle& b)
40 {
41 return a.backgroundColor == b.backgroundColor;
42 }
43 static bool areForegroundMarkedTextStylesEqual(const MarkedTextStyle& a, const MarkedTextStyle& b)
44 {
45 return a.textStyles == b.textStyles && a.textShadow == b.textShadow && a.alpha == b.alpha;
46 }
47 static bool areDecorationMarkedTextStylesEqual(const MarkedTextStyle& a, const MarkedTextStyle& b)
48 {
49 return a.textDecorationStyles == b.textDecorationStyles && a.textShadow == b.textShadow && a.alpha == b.alpha;
50 }
51
52 Color backgroundColor;
53 TextPaintStyle textStyles;
54 TextDecorationPainter::Styles textDecorationStyles;
55 std::optional<ShadowData> textShadow;
56 float alpha;
57};
