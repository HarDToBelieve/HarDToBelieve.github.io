25PointerEventsHitRules::PointerEventsHitRules(EHitTesting hitTesting, const HitTestRequest& request, PointerEvents pointerEvents)
26 : requireVisible(false)
27 , requireFill(false)
28 , requireStroke(false)
29 , canHitStroke(false)
30 , canHitFill(false)
31 , canHitBoundingBox(false)
32{
33 if (request.svgClipContent())
34 pointerEvents = PointerEvents::Fill;
35
36 if (hitTesting == SVG_PATH_HITTESTING) {
37 switch (pointerEvents)
38 {
39 case PointerEvents::VisiblePainted:
40 case PointerEvents::Auto: // "auto" is like "visiblePainted" when in SVG content
41 requireFill = true;
42 requireStroke = true;
43 FALLTHROUGH;
44 case PointerEvents::Visible:
45 requireVisible = true;
46 canHitFill = true;
47 canHitStroke = true;
48 break;
49 case PointerEvents::VisibleFill:
50 requireVisible = true;
51 canHitFill = true;
52 break;
53 case PointerEvents::VisibleStroke:
54 requireVisible = true;
55 canHitStroke = true;
56 break;
57 case PointerEvents::Painted:
58 requireFill = true;
59 requireStroke = true;
60 FALLTHROUGH;
61 case PointerEvents::All:
62 canHitFill = true;
63 canHitStroke = true;
64 break;
65 case PointerEvents::Fill:
66 canHitFill = true;
67 break;
68 case PointerEvents::Stroke:
69 canHitStroke = true;
70 break;
71 case PointerEvents::BoundingBox:
72 canHitFill = true;
73 canHitBoundingBox = true;
74 break;
75 case PointerEvents::None:
76 // nothing to do here, defaults are all false.
77 break;
78 }
79 } else {
80 switch (pointerEvents)
81 {
82 case PointerEvents::VisiblePainted:
83 case PointerEvents::Auto: // "auto" is like "visiblePainted" when in SVG content
84 requireVisible = true;
85 requireFill = true;
86 requireStroke = true;
87 canHitFill = true;
88 canHitStroke = true;
89 break;
90 case PointerEvents::VisibleFill:
91 case PointerEvents::VisibleStroke:
92 case PointerEvents::Visible:
93 requireVisible = true;
94 canHitFill = true;
95 canHitStroke = true;
96 break;
97 case PointerEvents::Painted:
98 requireFill = true;
99 requireStroke = true;
100 canHitFill = true;
101 canHitStroke = true;
102 break;
103 case PointerEvents::Fill:
104 case PointerEvents::Stroke:
105 case PointerEvents::All:
106 canHitFill = true;
107 canHitStroke = true;
108 break;
109 case PointerEvents::BoundingBox:
110 canHitFill = true;
111 canHitBoundingBox = true;
112 break;
113 case PointerEvents::None:
114 // nothing to do here, defaults are all false.
115 break;
116 }
117 }
118}
