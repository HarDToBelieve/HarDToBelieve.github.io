45RenderAttachment::RenderAttachment(HTMLAttachmentElement& element, RenderStyle&& style)
46 : RenderReplaced(element, WTFMove(style), LayoutSize())

48}
49
50HTMLAttachmentElement& RenderAttachment::attachmentElement() const
51{
52 return downcast<HTMLAttachmentElement>(nodeForNonAnonymous());
53}
54
55void RenderAttachment::layout()
56{
57 LayoutSize newIntrinsicSize = theme().attachmentIntrinsicSize(*this);
58
59 if (!theme().attachmentShouldAllowWidthToShrink(*this)) {
60 m_minimumIntrinsicWidth = std::max(m_minimumIntrinsicWidth, newIntrinsicSize.width());
61 newIntrinsicSize.setWidth(m_minimumIntrinsicWidth);
62 }
63
64 setIntrinsicSize(newIntrinsicSize);
65
66 RenderReplaced::layout();
67}
68
69void RenderAttachment::invalidate()
70{
71 setNeedsLayout();
72 repaint();
73}
74
75LayoutUnit RenderAttachment::baselinePosition(FontBaseline, bool, LineDirectionMode, LinePositionMode) const
76{
77 return theme().attachmentBaseline(*this);
78}
79
80bool RenderAttachment::shouldDrawBorder() const
81{
82 if (style().appearance() == BorderlessAttachmentPart)
83 return false;
84 return m_shouldDrawBorder;
85}
86
87void RenderAttachment::paintReplaced(PaintInfo& paintInfo, const LayoutPoint& offset)
88{
89 if (paintInfo.phase != PaintPhase::Selection || !hasVisibleBoxDecorations() || !style().hasAppearance())
90 return;
91
92 auto paintRect = borderBoxRect();
93 paintRect.moveBy(offset);
94
95 ControlStates controlStates;
96 theme().paint(*this, controlStates, paintInfo, paintRect);
97}
