42class FlexBoxIterator {
43public:
44 FlexBoxIterator(RenderDeprecatedFlexibleBox* parent)
45 : m_box(parent)
46 , m_largestOrdinal(1)
47 {
48 if (m_box->style().boxOrient() == BoxOrient::Horizontal && !m_box->style().isLeftToRightDirection())
49 m_forward = m_box->style().boxDirection() != BoxDirection::Normal;
50 else
51 m_forward = m_box->style().boxDirection() == BoxDirection::Normal;
52 if (!m_forward) {
53 // No choice, since we're going backwards, we have to find out the highest ordinal up front.
54 RenderBox* child = m_box->firstChildBox();
55 while (child) {
56 if (child->style().boxOrdinalGroup() > m_largestOrdinal)
57 m_largestOrdinal = child->style().boxOrdinalGroup();
58 child = child->nextSiblingBox();
59 }
60 }
61
62 reset();
63 }
64
65 void reset()
66 {
67 m_currentChild = nullptr;
68 m_ordinalIteration = std::numeric_limits<unsigned>::max();
69 }
70
71 RenderBox* first()
72 {
73 reset();
74 return next();
75 }
76
77 RenderBox* next()
78 {
79 do {
80 if (!m_currentChild) {
81 ++m_ordinalIteration;
82
83 if (!m_ordinalIteration)
84 m_currentOrdinal = m_forward ? 1 : m_largestOrdinal;
85 else {
86 if (m_ordinalIteration > m_ordinalValues.size())
87 return nullptr;
88
89 // Only copy+sort the values once per layout even if the iterator is reset.
90 if (static_cast<size_t>(m_ordinalValues.size()) != m_sortedOrdinalValues.size()) {
91 m_sortedOrdinalValues = copyToVector(m_ordinalValues);
92 std::sort(m_sortedOrdinalValues.begin(), m_sortedOrdinalValues.end());
93 }
94 m_currentOrdinal = m_forward ? m_sortedOrdinalValues[m_ordinalIteration - 1] : m_sortedOrdinalValues[m_sortedOrdinalValues.size() - m_ordinalIteration];
95 }
96
97 m_currentChild = m_forward ? m_box->firstChildBox() : m_box->lastChildBox();
98 } else
99 m_currentChild = m_forward ? m_currentChild->nextSiblingBox() : m_currentChild->previousSiblingBox();
100
101 if (m_currentChild && notFirstOrdinalValue())
102 m_ordinalValues.add(m_currentChild->style().boxOrdinalGroup());
103 } while (!m_currentChild || m_currentChild->isExcludedFromNormalLayout() || (!m_currentChild->isAnonymous()
104 && m_currentChild->style().boxOrdinalGroup() != m_currentOrdinal));
105 return m_currentChild;
106 }
107
108private:
109 bool notFirstOrdinalValue()
110 {
111 unsigned int firstOrdinalValue = m_forward ? 1 : m_largestOrdinal;
112 return m_currentOrdinal == firstOrdinalValue && m_currentChild->style().boxOrdinalGroup() != firstOrdinalValue;
113 }
114
115 RenderDeprecatedFlexibleBox* m_box;
116 RenderBox* m_currentChild;
117 bool m_forward;
118 unsigned m_currentOrdinal;
119 unsigned m_largestOrdinal;
120 HashSet<unsigned> m_ordinalValues;
121 Vector<unsigned> m_sortedOrdinalValues;
122 unsigned m_ordinalIteration;
123};

134static LayoutUnit marginWidthForChild(RenderBox* child)
135{
136 // A margin basically has three types: fixed, percentage, and auto (variable).
137 // Auto and percentage margins simply become 0 when computing min/max width.
138 // Fixed margins can be added in as is.
139 Length marginLeft = child->style().marginLeft();
140 Length marginRight = child->style().marginRight();
141 LayoutUnit margin;
142 if (marginLeft.isFixed())
143 margin += marginLeft.value();
144 if (marginRight.isFixed())
145 margin += marginRight.value();
146 return margin;
147}

149static bool childDoesNotAffectWidthOrFlexing(RenderObject* child)
150{
151 // Positioned children and collapsed children don't affect the min/max width.
152 return child->isOutOfFlowPositioned() || child->style().visibility() == Visibility::Collapse;
153}

155static LayoutUnit widthForChild(RenderBox* child)
156{
157 if (child->hasOverridingLogicalWidth())
158 return child->overridingLogicalWidth();
159 return child->logicalWidth();
160}
161
162static LayoutUnit heightForChild(RenderBox* child)
163{
164 if (child->hasOverridingLogicalHeight())
165 return child->overridingLogicalHeight();
166 return child->logicalHeight();
167}
168
169static LayoutUnit contentWidthForChild(RenderBox* child)
170{
171 return std::max<LayoutUnit>(0, widthForChild(child) - child->borderAndPaddingLogicalWidth());
172}
173
174static LayoutUnit contentHeightForChild(RenderBox* child)
175{
176 return std::max<LayoutUnit>(0, heightForChild(child) - child->borderAndPaddingLogicalHeight());
177}

188void RenderDeprecatedFlexibleBox::computeIntrinsicLogicalWidths(LayoutUnit& minLogicalWidth, LayoutUnit& maxLogicalWidth) const
189{
190 auto addScrollbarWidth = [&]() {
191 LayoutUnit scrollbarWidth = intrinsicScrollbarLogicalWidth();
192 maxLogicalWidth += scrollbarWidth;
193 minLogicalWidth += scrollbarWidth;
194 };
195
196 if (shouldApplySizeContainment(*this)) {
197 addScrollbarWidth();
198 return;
199 }
200
201 if (hasMultipleLines() || isVertical()) {
202 for (RenderBox* child = firstChildBox(); child; child = child->nextSiblingBox()) {
203 if (childDoesNotAffectWidthOrFlexing(child))
204 continue;
205
206 LayoutUnit margin = marginWidthForChild(child);
207 LayoutUnit width = child->minPreferredLogicalWidth() + margin;
208 minLogicalWidth = std::max(width, minLogicalWidth);
209
210 width = child->maxPreferredLogicalWidth() + margin;
211 maxLogicalWidth = std::max(width, maxLogicalWidth);
212 }
213 } else {
214 for (RenderBox* child = firstChildBox(); child; child = child->nextSiblingBox()) {
215 if (childDoesNotAffectWidthOrFlexing(child))
216 continue;
217
218 LayoutUnit margin = marginWidthForChild(child);
219 minLogicalWidth += child->minPreferredLogicalWidth() + margin;
220 maxLogicalWidth += child->maxPreferredLogicalWidth() + margin;
221 }
222 }
223
224 maxLogicalWidth = std::max(minLogicalWidth, maxLogicalWidth);
225 addScrollbarWidth();
226}

228void RenderDeprecatedFlexibleBox::computePreferredLogicalWidths()
229{
230 ASSERT(preferredLogicalWidthsDirty());
231
232 m_minPreferredLogicalWidth = m_maxPreferredLogicalWidth = 0;
233 if (style().width().isFixed() && style().width().value() > 0)
234 m_minPreferredLogicalWidth = m_maxPreferredLogicalWidth = adjustContentBoxLogicalWidthForBoxSizing(style().width());
235 else
236 computeIntrinsicLogicalWidths(m_minPreferredLogicalWidth, m_maxPreferredLogicalWidth);
237
238 RenderBox::computePreferredLogicalWidths(style().minWidth(), style().maxWidth(), borderAndPaddingLogicalWidth());
239
240 setPreferredLogicalWidthsDirty(false);
241}

243// Use an inline capacity of 8, since flexbox containers usually have less than 8 children.
244typedef Vector<LayoutRect, 8> ChildFrameRects;
245typedef Vector<LayoutSize, 8> ChildLayoutDeltas;

247static void appendChildFrameRects(RenderDeprecatedFlexibleBox* box, ChildFrameRects& childFrameRects)
248{
249 FlexBoxIterator iterator(box);
250 for (RenderBox* child = iterator.first(); child; child = iterator.next()) {
251 if (!child->isOutOfFlowPositioned())
252 childFrameRects.append(child->frameRect());
253 }
254}

256static void appendChildLayoutDeltas(RenderDeprecatedFlexibleBox* box, ChildLayoutDeltas& childLayoutDeltas)
257{
258 FlexBoxIterator iterator(box);
259 for (RenderBox* child = iterator.first(); child; child = iterator.next()) {
260 if (!child->isOutOfFlowPositioned())
261 childLayoutDeltas.append(LayoutSize());
262 }
263}

265static void repaintChildrenDuringLayoutIfMoved(RenderDeprecatedFlexibleBox* box, const ChildFrameRects& oldChildRects)
266{
267 size_t childIndex = 0;
268 FlexBoxIterator iterator(box);
269 for (RenderBox* child = iterator.first(); child; child = iterator.next()) {
270 if (child->isOutOfFlowPositioned())
271 continue;
272
273 // If the child moved, we have to repaint it as well as any floating/positioned
274 // descendants. An exception is if we need a layout. In this case, we know we're going to
275 // repaint ourselves (and the child) anyway.
276 if (!box->selfNeedsLayout() && child->checkForRepaintDuringLayout())
277 child->repaintDuringLayoutIfMoved(oldChildRects[childIndex]);
278
279 ++childIndex;
280 }
281 ASSERT(childIndex == oldChildRects.size());
282}

284void RenderDeprecatedFlexibleBox::layoutBlock(bool relayoutChildren, LayoutUnit)
285{
286 ASSERT(needsLayout());
287
288 if (!relayoutChildren && simplifiedLayout())
289 return;
290
291 LayoutRepainter repainter(*this, checkForRepaintDuringLayout());
292 {
293 LayoutStateMaintainer statePusher(*this, locationOffset(), hasTransform() || hasReflection() || style().isFlippedBlocksWritingMode());
294
295 resetLogicalHeightBeforeLayoutIfNeeded();
296 preparePaginationBeforeBlockLayout(relayoutChildren);
297
298 LayoutSize previousSize = size();
299
300 updateLogicalWidth();
301 updateLogicalHeight();
302
303 if (previousSize != size()
304 || (parent()->isDeprecatedFlexibleBox() && parent()->style().boxOrient() == BoxOrient::Horizontal
305 && parent()->style().boxAlign() == BoxAlignment::Stretch))
306 relayoutChildren = true;
307
308 setHeight(0);
309
310 m_stretchingChildren = false;
311
312#if ASSERT_ENABLED
313 LayoutSize oldLayoutDelta = view().frameView().layoutContext().layoutDelta();
314#endif
315
316 // Fieldsets need to find their legend and position it inside the border of the object.
317 // The legend then gets skipped during normal layout. The same is true for ruby text.
318 // It doesn't get included in the normal layout process but is instead skipped.
319 layoutExcludedChildren(relayoutChildren);
320
321 ChildFrameRects oldChildRects;
322 appendChildFrameRects(this, oldChildRects);
323
324 if (isHorizontal())
325 layoutHorizontalBox(relayoutChildren);
326 else
327 layoutVerticalBox(relayoutChildren);
328
329 repaintChildrenDuringLayoutIfMoved(this, oldChildRects);
330 ASSERT(view().frameView().layoutContext().layoutDeltaMatches(oldLayoutDelta));
331
332 LayoutUnit oldClientAfterEdge = clientLogicalBottom();
333 updateLogicalHeight();
334
335 if (previousSize.height() != height())
336 relayoutChildren = true;
337
338 layoutPositionedObjects(relayoutChildren || isDocumentElementRenderer());
339
340 computeOverflow(oldClientAfterEdge);
341 }
342
343 updateLayerTransform();
344
345 auto* layoutState = view().frameView().layoutContext().layoutState();
346 if (layoutState && layoutState->pageLogicalHeight())
347 setPageLogicalOffset(layoutState->pageLogicalOffset(this, logicalTop()));
348
349 // Update our scrollbars if we're overflow:auto/scroll/hidden now that we know if
350 // we overflow or not.
351 updateScrollInfoAfterLayout();
352
353 // Repaint with our new bounds if they are different from our old bounds.
354 repainter.repaintAfterLayout();
355
356 clearNeedsLayout();
357}

359// The first walk over our kids is to find out if we have any flexible children.
360static void gatherFlexChildrenInfo(FlexBoxIterator& iterator, bool relayoutChildren, unsigned int& highestFlexGroup, unsigned int& lowestFlexGroup, bool& haveFlex)
361{
362 for (RenderBox* child = iterator.first(); child; child = iterator.next()) {
363 // Check to see if this child flexes.
364 if (!childDoesNotAffectWidthOrFlexing(child) && child->style().boxFlex() > 0.0f) {
365 // We always have to lay out flexible objects again, since the flex distribution
366 // may have changed, and we need to reallocate space.
367 child->clearOverridingContentSize();
368 if (!relayoutChildren)
369 child->setChildNeedsLayout(MarkOnlyThis);
370 haveFlex = true;
371 unsigned flexGroup = child->style().boxFlexGroup();
372 if (lowestFlexGroup == 0)
373 lowestFlexGroup = flexGroup;
374 if (flexGroup < lowestFlexGroup)
375 lowestFlexGroup = flexGroup;
376 if (flexGroup > highestFlexGroup)
377 highestFlexGroup = flexGroup;
378 }
379 }
380}

382static void layoutChildIfNeededApplyingDelta(RenderBox* child, const LayoutSize& layoutDelta)
383{
384 if (!child->needsLayout())
385 return;
386
387 child->view().frameView().layoutContext().addLayoutDelta(layoutDelta);
388 child->layoutIfNeeded();
389 child->view().frameView().layoutContext().addLayoutDelta(-layoutDelta);
390}

392void RenderDeprecatedFlexibleBox::layoutHorizontalBox(bool relayoutChildren)
393{
394 LayoutUnit toAdd = borderBottom() + paddingBottom() + horizontalScrollbarHeight();
395 LayoutUnit yPos = borderTop() + paddingTop();
396 LayoutUnit xPos = borderLeft() + paddingLeft();
397 bool heightSpecified = false;
398 LayoutUnit oldHeight;
399
400 LayoutUnit remainingSpace;
401
402 FlexBoxIterator iterator(this);
403 unsigned int highestFlexGroup = 0;
404 unsigned int lowestFlexGroup = 0;
405 bool haveFlex = false, flexingChildren = false;
406 gatherFlexChildrenInfo(iterator, relayoutChildren, highestFlexGroup, lowestFlexGroup, haveFlex);
407
408 beginUpdateScrollInfoAfterLayoutTransaction();
409
410 ChildLayoutDeltas childLayoutDeltas;
411 appendChildLayoutDeltas(this, childLayoutDeltas);
412
413 // We do 2 passes. The first pass is simply to lay everyone out at
414 // their preferred widths. The subsequent passes handle flexing the children.
415 // The first pass skips flexible objects completely.
416 do {
417 // Reset our height.
418 setHeight(yPos);
419
420 xPos = borderLeft() + paddingLeft();
421
422 size_t childIndex = 0;
423
424 // Our first pass is done without flexing. We simply lay the children
425 // out within the box. We have to do a layout first in order to determine
426 // our box's intrinsic height.
427 LayoutUnit maxAscent, maxDescent;
428 for (RenderBox* child = iterator.first(); child; child = iterator.next()) {
429 if (relayoutChildren)
430 child->setChildNeedsLayout(MarkOnlyThis);
431
432 if (child->isOutOfFlowPositioned())
433 continue;
434
435 LayoutSize& childLayoutDelta = childLayoutDeltas[childIndex++];
436
437 // Compute the child's vertical margins.
438 child->computeAndSetBlockDirectionMargins(*this);
439
440 child->markForPaginationRelayoutIfNeeded();
441
442 // Apply the child's current layout delta.
443 layoutChildIfNeededApplyingDelta(child, childLayoutDelta);
444
445 // Update our height and overflow height.
446 if (style().boxAlign() == BoxAlignment::Baseline) {
447 LayoutUnit ascent = child->firstLineBaseline().value_or(child->height() + child->marginBottom());
448 ascent += child->marginTop();
449 LayoutUnit descent = (child->height() + child->verticalMarginExtent()) - ascent;
450
451 // Update our maximum ascent.
452 maxAscent = std::max(maxAscent, ascent);
453
454 // Update our maximum descent.
455 maxDescent = std::max(maxDescent, descent);
456
457 // Now update our height.
458 setHeight(std::max(yPos + maxAscent + maxDescent, height()));
459 }
460 else
461 setHeight(std::max(height(), yPos + child->height() + child->verticalMarginExtent()));
462 }
463 ASSERT(childIndex == childLayoutDeltas.size());
464
465 if (!iterator.first() && hasLineIfEmpty())
466 setHeight(height() + lineHeight(true, style().isHorizontalWritingMode() ? HorizontalLine : VerticalLine, PositionOfInteriorLineBoxes));
467
468 setHeight(height() + toAdd);
469
470 oldHeight = height();
471 updateLogicalHeight();
472
473 relayoutChildren = false;
474 if (oldHeight != height())
475 heightSpecified = true;
476
477 // Now that our height is actually known, we can place our boxes.
478 childIndex = 0;
479 m_stretchingChildren = (style().boxAlign() == BoxAlignment::Stretch);
480 for (RenderBox* child = iterator.first(); child; child = iterator.next()) {
481 if (child->isOutOfFlowPositioned()) {
482 child->containingBlock()->insertPositionedObject(*child);
483 RenderLayer* childLayer = child->layer();
484 childLayer->setStaticInlinePosition(xPos); // FIXME: Not right for regions.
485 if (childLayer->staticBlockPosition() != yPos) {
486 childLayer->setStaticBlockPosition(yPos);
487 if (child->style().hasStaticBlockPosition(style().isHorizontalWritingMode()))
488 child->setChildNeedsLayout(MarkOnlyThis);
489 }
490 continue;
491 }
492
493 LayoutSize& childLayoutDelta = childLayoutDeltas[childIndex++];
494
495 if (child->style().visibility() == Visibility::Collapse) {
496 // visibility: collapsed children do not participate in our positioning.
497 // But we need to lay them out.
498 layoutChildIfNeededApplyingDelta(child, childLayoutDelta);
499 continue;
500 }
501
502 // We need to see if this child's height has changed, since we make block elements
503 // fill the height of a containing box by default.
504 // Now do a layout.
505 LayoutUnit oldChildHeight = child->height();
506 child->updateLogicalHeight();
507 if (oldChildHeight != child->height())
508 child->setChildNeedsLayout(MarkOnlyThis);
509
510 child->markForPaginationRelayoutIfNeeded();
511
512 layoutChildIfNeededApplyingDelta(child, childLayoutDelta);
513
514 // We can place the child now, using our value of box-align.
515 xPos += child->marginLeft();
516 LayoutUnit childY = yPos;
517 switch (style().boxAlign()) {
518 case BoxAlignment::Center:
519 childY += child->marginTop() + std::max<LayoutUnit>(0, (contentHeight() - (child->height() + child->verticalMarginExtent())) / 2);
520 break;
521 case BoxAlignment::Baseline: {
522 LayoutUnit ascent = child->firstLineBaseline().value_or(child->height() + child->marginBottom());
523 ascent += child->marginTop();
524 childY += child->marginTop() + (maxAscent - ascent);
525 break;
526 }
527 case BoxAlignment::End:
528 childY += contentHeight() - child->marginBottom() - child->height();
529 break;
530 default: // BoxAlignment::Start
531 childY += child->marginTop();
532 break;
533 }
534
535 placeChild(child, LayoutPoint(xPos, childY), &childLayoutDelta);
536
537 xPos += child->width() + child->marginRight();
538 }
539 ASSERT(childIndex == childLayoutDeltas.size());
540
541 remainingSpace = borderLeft() + paddingLeft() + contentWidth() - xPos;
542
543 m_stretchingChildren = false;
544 if (flexingChildren)
545 haveFlex = false; // We're done.
546 else if (haveFlex) {
547 // We have some flexible objects. See if we need to grow/shrink them at all.
548 if (!remainingSpace)
549 break;
550
551 // Allocate the remaining space among the flexible objects. If we are trying to
552 // grow, then we go from the lowest flex group to the highest flex group. For shrinking,
553 // we go from the highest flex group to the lowest group.
554 bool expanding = remainingSpace > 0;
555 unsigned int start = expanding ? lowestFlexGroup : highestFlexGroup;
556 unsigned int end = expanding? highestFlexGroup : lowestFlexGroup;
557 for (unsigned int i = start; i <= end && remainingSpace; i++) {
558 // Always start off by assuming the group can get all the remaining space.
559 LayoutUnit groupRemainingSpace = remainingSpace;
560 do {
561 // Flexing consists of multiple passes, since we have to change ratios every time an object hits its max/min-width
562 // For a given pass, we always start off by computing the totalFlex of all objects that can grow/shrink at all, and
563 // computing the allowed growth before an object hits its min/max width (and thus
564 // forces a totalFlex recomputation).
565 LayoutUnit groupRemainingSpaceAtBeginning = groupRemainingSpace;
566 float totalFlex = 0.0f;
567 for (RenderBox* child = iterator.first(); child; child = iterator.next()) {
568 if (allowedChildFlex(child, expanding, i))
569 totalFlex += child->style().boxFlex();
570 }
571 LayoutUnit spaceAvailableThisPass = groupRemainingSpace;
572 for (RenderBox* child = iterator.first(); child; child = iterator.next()) {
573 LayoutUnit allowedFlex = allowedChildFlex(child, expanding, i);
574 if (allowedFlex) {
575 LayoutUnit projectedFlex = (allowedFlex == LayoutUnit::max()) ? allowedFlex : LayoutUnit(allowedFlex * (totalFlex / child->style().boxFlex()));
576 spaceAvailableThisPass = expanding ? std::min(spaceAvailableThisPass, projectedFlex) : std::max(spaceAvailableThisPass, projectedFlex);
577 }
578 }
579
580 // The flex groups may not have any flexible objects this time around.
581 if (!spaceAvailableThisPass || totalFlex == 0.0f) {
582 // If we just couldn't grow/shrink any more, then it's time to transition to the next flex group.
583 groupRemainingSpace = 0;
584 continue;
585 }
586
587 // Now distribute the space to objects.
588 for (RenderBox* child = iterator.first(); child && spaceAvailableThisPass && totalFlex; child = iterator.next()) {
589 if (child->style().visibility() == Visibility::Collapse)
590 continue;
591
592 if (allowedChildFlex(child, expanding, i)) {
593 LayoutUnit spaceAdd = LayoutUnit(spaceAvailableThisPass * (child->style().boxFlex() / totalFlex));
594 if (spaceAdd) {
595 child->setOverridingLogicalWidth(widthForChild(child) + spaceAdd);
596 flexingChildren = true;
597 relayoutChildren = true;
598 }
599
600 spaceAvailableThisPass -= spaceAdd;
601 remainingSpace -= spaceAdd;
602 groupRemainingSpace -= spaceAdd;
603
604 totalFlex -= child->style().boxFlex();
605 }
606 }
607 if (groupRemainingSpace == groupRemainingSpaceAtBeginning) {
608 // This is not advancing, avoid getting stuck by distributing the remaining pixels.
609 LayoutUnit spaceAdd = groupRemainingSpace > 0 ? 1 : -1;
610 for (RenderBox* child = iterator.first(); child && groupRemainingSpace; child = iterator.next()) {
611 if (allowedChildFlex(child, expanding, i)) {
612 child->setOverridingLogicalWidth(widthForChild(child) + spaceAdd);
613 flexingChildren = true;
614 relayoutChildren = true;
615 remainingSpace -= spaceAdd;
616 groupRemainingSpace -= spaceAdd;
617 }
618 }
619 }
620 } while (absoluteValue(groupRemainingSpace) >= 1);
621 }
622
623 // We didn't find any children that could grow.
624 if (haveFlex && !flexingChildren)
625 haveFlex = false;
626 }
627 } while (haveFlex);
628
629 endAndCommitUpdateScrollInfoAfterLayoutTransaction();
630
631 if (remainingSpace > 0 && ((style().isLeftToRightDirection() && style().boxPack() != BoxPack::Start)
632 || (!style().isLeftToRightDirection() && style().boxPack() != BoxPack::End))) {
633 // Children must be repositioned.
634 LayoutUnit offset;
635 if (style().boxPack() == BoxPack::Justify) {
636 // Determine the total number of children.
637 int totalChildren = 0;
638 for (RenderBox* child = iterator.first(); child; child = iterator.next()) {
639 if (childDoesNotAffectWidthOrFlexing(child))
640 continue;
641 ++totalChildren;
642 }
643
644 // Iterate over the children and space them out according to the
645 // justification level.
646 if (totalChildren > 1) {
647 --totalChildren;
648 bool firstChild = true;
649 for (RenderBox* child = iterator.first(); child; child = iterator.next()) {
650 if (childDoesNotAffectWidthOrFlexing(child))
651 continue;
652
653 if (firstChild) {
654 firstChild = false;
655 continue;
656 }
657
658 offset += remainingSpace/totalChildren;
659 remainingSpace -= (remainingSpace/totalChildren);
660 --totalChildren;
661
662 placeChild(child, child->location() + LayoutSize(offset, 0_lu));
663 }
664 }
665 } else {
666 if (style().boxPack() == BoxPack::Center)
667 offset += remainingSpace / 2;
668 else // BoxPack::End for LTR, BoxPack::Start for RTL
669 offset += remainingSpace;
670 for (RenderBox* child = iterator.first(); child; child = iterator.next()) {
671 if (childDoesNotAffectWidthOrFlexing(child))
672 continue;
673
674 placeChild(child, child->location() + LayoutSize(offset, 0_lu));
675 }
676 }
677 }
678
679 // So that the computeLogicalHeight in layoutBlock() knows to relayout positioned objects because of
680 // a height change, we revert our height back to the intrinsic height before returning.
681 if (heightSpecified)
682 setHeight(oldHeight);
683}

685void RenderDeprecatedFlexibleBox::layoutVerticalBox(bool relayoutChildren)
686{
687 LayoutUnit yPos = borderTop() + paddingTop();
688 LayoutUnit toAdd = borderBottom() + paddingBottom() + horizontalScrollbarHeight();
689 bool heightSpecified = false;
690 LayoutUnit oldHeight;
691
692 LayoutUnit remainingSpace;
693
694 FlexBoxIterator iterator(this);
695 unsigned int highestFlexGroup = 0;
696 unsigned int lowestFlexGroup = 0;
697 bool haveFlex = false, flexingChildren = false;
698 gatherFlexChildrenInfo(iterator, relayoutChildren, highestFlexGroup, lowestFlexGroup, haveFlex);
699
700 // We confine the line clamp ugliness to vertical flexible boxes (thus keeping it out of
701 // mainstream block layout); this is not really part of the XUL box model.
702 bool haveLineClamp = !style().lineClamp().isNone();
703 if (haveLineClamp)
704 applyLineClamp(iterator, relayoutChildren);
705
706 beginUpdateScrollInfoAfterLayoutTransaction();
707
708 ChildLayoutDeltas childLayoutDeltas;
709 appendChildLayoutDeltas(this, childLayoutDeltas);
710
711 // We do 2 passes. The first pass is simply to lay everyone out at
712 // their preferred widths. The second pass handles flexing the children.
713 // Our first pass is done without flexing. We simply lay the children
714 // out within the box.
715 do {
716 setHeight(borderTop() + paddingTop());
717 LayoutUnit minHeight = height() + toAdd;
718
719 size_t childIndex = 0;
720 for (RenderBox* child = iterator.first(); child; child = iterator.next()) {
721 // Make sure we relayout children if we need it.
722 if (!haveLineClamp && relayoutChildren)
723 child->setChildNeedsLayout(MarkOnlyThis);
724
725 if (child->isOutOfFlowPositioned()) {
726 child->containingBlock()->insertPositionedObject(*child);
727 RenderLayer* childLayer = child->layer();
728 childLayer->setStaticInlinePosition(borderStart() + paddingStart()); // FIXME: Not right for regions.
729 if (childLayer->staticBlockPosition() != height()) {
730 childLayer->setStaticBlockPosition(height());
731 if (child->style().hasStaticBlockPosition(style().isHorizontalWritingMode()))
732 child->setChildNeedsLayout(MarkOnlyThis);
733 }
734 continue;
735 }
736
737 LayoutSize& childLayoutDelta = childLayoutDeltas[childIndex++];
738
739 if (child->style().visibility() == Visibility::Collapse) {
740 // visibility: collapsed children do not participate in our positioning.
741 // But we need to lay them down.
742 layoutChildIfNeededApplyingDelta(child, childLayoutDelta);
743 continue;
744 }
745
746 // Compute the child's vertical margins.
747 child->computeAndSetBlockDirectionMargins(*this);
748
749 // Add in the child's marginTop to our height.
750 setHeight(height() + child->marginTop());
751
752 child->markForPaginationRelayoutIfNeeded();
753
754 // Now do a layout.
755 layoutChildIfNeededApplyingDelta(child, childLayoutDelta);
756
757 // We can place the child now, using our value of box-align.
758 LayoutUnit childX = borderLeft() + paddingLeft();
759 switch (style().boxAlign()) {
760 case BoxAlignment::Center:
761 case BoxAlignment::Baseline: // Baseline just maps to center for vertical boxes
762 childX += child->marginLeft() + std::max<LayoutUnit>(0, (contentWidth() - (child->width() + child->horizontalMarginExtent())) / 2);
763 break;
764 case BoxAlignment::End:
765 if (!style().isLeftToRightDirection())
766 childX += child->marginLeft();
767 else
768 childX += contentWidth() - child->marginRight() - child->width();
769 break;
770 default: // BoxAlignment::Start/BoxAlignment::Stretch
771 if (style().isLeftToRightDirection())
772 childX += child->marginLeft();
773 else
774 childX += contentWidth() - child->marginRight() - child->width();
775 break;
776 }
777
778 // Place the child.
779 placeChild(child, LayoutPoint(childX, height()), &childLayoutDelta);
780 setHeight(height() + child->height() + child->marginBottom());
781 }
782 ASSERT(childIndex == childLayoutDeltas.size());
783
784 yPos = height();
785
786 if (!iterator.first() && hasLineIfEmpty())
787 setHeight(height() + lineHeight(true, style().isHorizontalWritingMode() ? HorizontalLine : VerticalLine, PositionOfInteriorLineBoxes));
788
789 setHeight(height() + toAdd);
790
791 // Negative margins can cause our height to shrink below our minimal height (border/padding).
792 // If this happens, ensure that the computed height is increased to the minimal height.
793 if (height() < minHeight)
794 setHeight(minHeight);
795
796 // Now we have to calc our height, so we know how much space we have remaining.
797 oldHeight = height();
798 updateLogicalHeight();
799 if (oldHeight != height())
800 heightSpecified = true;
801
802 remainingSpace = borderTop() + paddingTop() + contentHeight() - yPos;
803
804 if (flexingChildren)
805 haveFlex = false; // We're done.
806 else if (haveFlex) {
807 // We have some flexible objects. See if we need to grow/shrink them at all.
808 if (!remainingSpace)
809 break;
810
811 // Allocate the remaining space among the flexible objects. If we are trying to
812 // grow, then we go from the lowest flex group to the highest flex group. For shrinking,
813 // we go from the highest flex group to the lowest group.
814 bool expanding = remainingSpace > 0;
815 unsigned int start = expanding ? lowestFlexGroup : highestFlexGroup;
816 unsigned int end = expanding? highestFlexGroup : lowestFlexGroup;
817 for (unsigned int i = start; i <= end && remainingSpace; i++) {
818 // Always start off by assuming the group can get all the remaining space.
819 LayoutUnit groupRemainingSpace = remainingSpace;
820 do {
821 // Flexing consists of multiple passes, since we have to change ratios every time an object hits its max/min-width
822 // For a given pass, we always start off by computing the totalFlex of all objects that can grow/shrink at all, and
823 // computing the allowed growth before an object hits its min/max width (and thus
824 // forces a totalFlex recomputation).
825 LayoutUnit groupRemainingSpaceAtBeginning = groupRemainingSpace;
826 float totalFlex = 0.0f;
827 for (RenderBox* child = iterator.first(); child; child = iterator.next()) {
828 if (allowedChildFlex(child, expanding, i))
829 totalFlex += child->style().boxFlex();
830 }
831 LayoutUnit spaceAvailableThisPass = groupRemainingSpace;
832 for (RenderBox* child = iterator.first(); child; child = iterator.next()) {
833 LayoutUnit allowedFlex = allowedChildFlex(child, expanding, i);
834 if (allowedFlex) {
835 LayoutUnit projectedFlex = (allowedFlex == LayoutUnit::max()) ? allowedFlex : LayoutUnit(allowedFlex * (totalFlex / child->style().boxFlex()));
836 spaceAvailableThisPass = expanding ? std::min(spaceAvailableThisPass, projectedFlex) : std::max(spaceAvailableThisPass, projectedFlex);
837 }
838 }
839
840 // The flex groups may not have any flexible objects this time around.
841 if (!spaceAvailableThisPass || totalFlex == 0.0f) {
842 // If we just couldn't grow/shrink any more, then it's time to transition to the next flex group.
843 groupRemainingSpace = 0;
844 continue;
845 }
846
847 // Now distribute the space to objects.
848 for (RenderBox* child = iterator.first(); child && spaceAvailableThisPass && totalFlex; child = iterator.next()) {
849 if (allowedChildFlex(child, expanding, i)) {
850 LayoutUnit spaceAdd { spaceAvailableThisPass * (child->style().boxFlex() / totalFlex) };
851 if (spaceAdd) {
852 child->setOverridingLogicalHeight(heightForChild(child) + spaceAdd);
853 flexingChildren = true;
854 relayoutChildren = true;
855 }
856
857 spaceAvailableThisPass -= spaceAdd;
858 remainingSpace -= spaceAdd;
859 groupRemainingSpace -= spaceAdd;
860
861 totalFlex -= child->style().boxFlex();
862 }
863 }
864 if (groupRemainingSpace == groupRemainingSpaceAtBeginning) {
865 // This is not advancing, avoid getting stuck by distributing the remaining pixels.
866 LayoutUnit spaceAdd = groupRemainingSpace > 0 ? 1 : -1;
867 for (RenderBox* child = iterator.first(); child && groupRemainingSpace; child = iterator.next()) {
868 if (allowedChildFlex(child, expanding, i)) {
869 child->setOverridingLogicalHeight(heightForChild(child) + spaceAdd);
870 flexingChildren = true;
871 relayoutChildren = true;
872 remainingSpace -= spaceAdd;
873 groupRemainingSpace -= spaceAdd;
874 }
875 }
876 }
877 } while (absoluteValue(groupRemainingSpace) >= 1);
878 }
879
880 // We didn't find any children that could grow.
881 if (haveFlex && !flexingChildren)
882 haveFlex = false;
883 }
884 } while (haveFlex);
885
886 endAndCommitUpdateScrollInfoAfterLayoutTransaction();
887
888 if (style().boxPack() != BoxPack::Start && remainingSpace > 0) {
889 // Children must be repositioned.
890 LayoutUnit offset;
891 if (style().boxPack() == BoxPack::Justify) {
892 // Determine the total number of children.
893 int totalChildren = 0;
894 for (RenderBox* child = iterator.first(); child; child = iterator.next()) {
895 if (childDoesNotAffectWidthOrFlexing(child))
896 continue;
897
898 ++totalChildren;
899 }
900
901 // Iterate over the children and space them out according to the
902 // justification level.
903 if (totalChildren > 1) {
904 --totalChildren;
905 bool firstChild = true;
906 for (RenderBox* child = iterator.first(); child; child = iterator.next()) {
907 if (childDoesNotAffectWidthOrFlexing(child))
908 continue;
909
910 if (firstChild) {
911 firstChild = false;
912 continue;
913 }
914
915 offset += remainingSpace/totalChildren;
916 remainingSpace -= (remainingSpace/totalChildren);
917 --totalChildren;
918 placeChild(child, child->location() + LayoutSize(0_lu, offset));
919 }
920 }
921 } else {
922 if (style().boxPack() == BoxPack::Center)
923 offset += remainingSpace / 2;
924 else // BoxPack::End
925 offset += remainingSpace;
926 for (RenderBox* child = iterator.first(); child; child = iterator.next()) {
927 if (childDoesNotAffectWidthOrFlexing(child))
928 continue;
929 placeChild(child, child->location() + LayoutSize(0_lu, offset));
930 }
931 }
932 }
933
934 // So that the computeLogicalHeight in layoutBlock() knows to relayout positioned objects because of
935 // a height change, we revert our height back to the intrinsic height before returning.
936 if (heightSpecified)
937 setHeight(oldHeight);
938}

940static LegacyRootInlineBox* lineAtIndex(const RenderBlockFlow& flow, int i)
941{
942 ASSERT(i >= 0);
943
944 if (flow.style().visibility() != Visibility::Visible)
945 return nullptr;
946
947 if (flow.childrenInline()) {
948 for (auto* box = flow.firstRootBox(); box; box = box->nextRootBox()) {
949 if (!i--)
950 return box;
951 }
952 return nullptr;
953 }
954
955 for (auto& blockFlow : childrenOfType<RenderBlockFlow>(flow)) {
956 if (!shouldIncludeLinesForParentLineCount(blockFlow))
957 continue;
958 if (LegacyRootInlineBox* box = lineAtIndex(blockFlow, i))
959 return box;
960 }
961
962 return nullptr;
963}

965static int getHeightForLineCount(const RenderBlockFlow& block, int lineCount, bool includeBottom, int& count)
966{
967 if (block.style().visibility() != Visibility::Visible)
968 return -1;
969
970 if (block.childrenInline()) {
971 for (auto* box = block.firstRootBox(); box; box = box->nextRootBox()) {
972 if (++count == lineCount)
973 return box->lineBottom() + (includeBottom ? (block.borderBottom() + block.paddingBottom()) : 0_lu);
974 }
975 } else {
976 RenderBox* normalFlowChildWithoutLines = nullptr;
977 for (auto* obj = block.firstChildBox(); obj; obj = obj->nextSiblingBox()) {
978 if (is<RenderBlockFlow>(*obj) && shouldIncludeLinesForParentLineCount(downcast<RenderBlockFlow>(*obj))) {
979 int result = getHeightForLineCount(downcast<RenderBlockFlow>(*obj), lineCount, false, count);
980 if (result != -1)
981 return result + obj->y() + (includeBottom ? (block.borderBottom() + block.paddingBottom()) : 0_lu);
982 } else if (!obj->isFloatingOrOutOfFlowPositioned())
983 normalFlowChildWithoutLines = obj;
984 }
985 if (normalFlowChildWithoutLines && !lineCount)
986 return normalFlowChildWithoutLines->y() + normalFlowChildWithoutLines->height();
987 }
988
989 return -1;
990}

998void RenderDeprecatedFlexibleBox::applyLineClamp(FlexBoxIterator& iterator, bool relayoutChildren)
999{
1000 int maxLineCount = 0;
1001 for (RenderBox* child = iterator.first(); child; child = iterator.next()) {
1002 if (childDoesNotAffectWidthOrFlexing(child))
1003 continue;
1004
1005 child->clearOverridingContentSize();
1006 if (relayoutChildren || (child->isReplaced() && (child->style().width().isPercentOrCalculated() || child->style().height().isPercentOrCalculated()))
1007 || (child->style().height().isAuto() && is<RenderBlockFlow>(*child))) {
1008 child->setChildNeedsLayout(MarkOnlyThis);
1009
1010 // Dirty all the positioned objects.
1011 if (is<RenderBlockFlow>(*child)) {
1012 downcast<RenderBlockFlow>(*child).markPositionedObjectsForLayout();
1013 downcast<RenderBlockFlow>(*child).clearTruncation();
1014 }
1015 }
1016 child->layoutIfNeeded();
1017 if (child->style().height().isAuto() && is<RenderBlockFlow>(*child))
1018 maxLineCount = std::max(maxLineCount, downcast<RenderBlockFlow>(*child).lineCount());
1019 }
1020
1021 // Get the number of lines and then alter all block flow children with auto height to use the
1022 // specified height. We always try to leave room for at least one line.
1023 LineClampValue lineClamp = style().lineClamp();
1024 int numVisibleLines = lineClamp.isPercentage() ? std::max(1, (maxLineCount + 1) * lineClamp.value() / 100) : lineClamp.value();
1025 if (numVisibleLines >= maxLineCount)
1026 return;
1027
1028 for (RenderBox* child = iterator.first(); child; child = iterator.next()) {
1029 if (childDoesNotAffectWidthOrFlexing(child) || !child->style().height().isAuto() || !is<RenderBlockFlow>(*child))
1030 continue;
1031
1032 RenderBlockFlow& blockChild = downcast<RenderBlockFlow>(*child);
1033 int lineCount = blockChild.lineCount();
1034 if (lineCount <= numVisibleLines)
1035 continue;
1036
1037 LayoutUnit newHeight = heightForLineCount(blockChild, numVisibleLines);
1038 if (newHeight == child->height())
1039 continue;
1040
1041 child->setChildNeedsLayout(MarkOnlyThis);
1042 child->setOverridingLogicalHeight(newHeight);
1043 child->layoutIfNeeded();
1044
1045 // FIXME: For now don't support RTL.
1046 if (style().direction() != TextDirection::LTR)
1047 continue;
1048
1049 // Get the last line
1050 LegacyRootInlineBox* lastLine = lineAtIndex(blockChild, lineCount - 1);
1051 if (!lastLine)
1052 continue;
1053
1054 LegacyRootInlineBox* lastVisibleLine = lineAtIndex(blockChild, numVisibleLines - 1);
1055 if (!lastVisibleLine || !lastVisibleLine->firstChild())
1056 continue;
1057
1058 const UChar ellipsisAndSpace[2] = { horizontalEllipsis, ' ' };
1059 static MainThreadNeverDestroyed<const AtomString> ellipsisAndSpaceStr(ellipsisAndSpace, 2);
1060 static MainThreadNeverDestroyed<const AtomString> ellipsisStr(&horizontalEllipsis, 1);
1061 const RenderStyle& lineStyle = numVisibleLines == 1 ? firstLineStyle() : style();
1062 const FontCascade& font = lineStyle.fontCascade();
1063
1064 // Get ellipsis width, and if the last child is an anchor, it will go after the ellipsis, so add in a space and the anchor width too
1065 LayoutUnit totalWidth;
1066 LegacyInlineBox* anchorBox = lastLine->lastChild();
1067 auto& lastVisibleRenderer = lastVisibleLine->firstChild()->renderer();
1068 if (anchorBox && anchorBox->renderer().style().isLink() && &lastVisibleRenderer != &anchorBox->renderer())
1069 totalWidth = anchorBox->logicalWidth() + font.width(constructTextRun(ellipsisAndSpace, 2, style()));
1070 else {
1071 anchorBox = nullptr;
1072 totalWidth = font.width(constructTextRun(&horizontalEllipsis, 1, style()));
1073 }
1074
1075 // See if this width can be accommodated on the last visible line
1076 RenderBlockFlow& destBlock = lastVisibleLine->blockFlow();
1077 RenderBlockFlow& srcBlock = lastLine->blockFlow();
1078
1079 // FIXME: Directions of src/destBlock could be different from our direction and from one another.
1080 if (!srcBlock.style().isLeftToRightDirection())
1081 continue;
1082
1083 bool leftToRight = destBlock.style().isLeftToRightDirection();
1084 if (!leftToRight)
1085 continue;
1086
1087 LayoutUnit blockRightEdge = destBlock.logicalRightOffsetForLine(LayoutUnit(lastVisibleLine->y()), DoNotIndentText);
1088 if (!lastVisibleLine->lineCanAccommodateEllipsis(leftToRight, blockRightEdge, lastVisibleLine->x() + lastVisibleLine->logicalWidth(), totalWidth))
1089 continue;
1090
1091 // Let the truncation code kick in.
1092 // FIXME: the text alignment should be recomputed after the width changes due to truncation.
1093 LayoutUnit blockLeftEdge = destBlock.logicalLeftOffsetForLine(LayoutUnit(lastVisibleLine->y()), DoNotIndentText);
1094 lastVisibleLine->placeEllipsis(anchorBox ? ellipsisAndSpaceStr : ellipsisStr, leftToRight, blockLeftEdge, blockRightEdge, totalWidth, anchorBox);
1095 destBlock.setHasMarkupTruncation(true);
1096 }
1097}

1127LayoutUnit RenderDeprecatedFlexibleBox::allowedChildFlex(RenderBox* child, bool expanding, unsigned int group)
1128{
1129 if (childDoesNotAffectWidthOrFlexing(child) || child->style().boxFlex() == 0.0f || child->style().boxFlexGroup() != group)
1130 return 0;
1131
1132 if (expanding) {
1133 if (isHorizontal()) {
1134 // FIXME: For now just handle fixed values.
1135 LayoutUnit maxWidth = LayoutUnit::max();
1136 LayoutUnit width = contentWidthForChild(child);
1137 if (!child->style().maxWidth().isUndefined() && child->style().maxWidth().isFixed())
1138 maxWidth = child->style().maxWidth().value();
1139 else if (child->style().maxWidth().type() == LengthType::Intrinsic)
1140 maxWidth = child->maxPreferredLogicalWidth();
1141 else if (child->style().maxWidth().isMinIntrinsic())
1142 maxWidth = child->minPreferredLogicalWidth();
1143 if (maxWidth == LayoutUnit::max())
1144 return maxWidth;
1145 return std::max<LayoutUnit>(0, maxWidth - width);
1146 } else {
1147 // FIXME: For now just handle fixed values.
1148 LayoutUnit maxHeight = LayoutUnit::max();
1149 LayoutUnit height = contentHeightForChild(child);
1150 if (!child->style().maxHeight().isUndefined() && child->style().maxHeight().isFixed())
1151 maxHeight = child->style().maxHeight().value();
1152 if (maxHeight == LayoutUnit::max())
1153 return maxHeight;
1154 return std::max<LayoutUnit>(0, maxHeight - height);
1155 }
1156 }
1157
1158 // FIXME: For now just handle fixed values.
1159 if (isHorizontal()) {
1160 LayoutUnit minWidth = child->minPreferredLogicalWidth();
1161 LayoutUnit width = contentWidthForChild(child);
1162 if (child->style().minWidth().isFixed())
1163 minWidth = child->style().minWidth().value();
1164 else if (child->style().minWidth().type() == LengthType::Intrinsic)
1165 minWidth = child->maxPreferredLogicalWidth();
1166 else if (child->style().minWidth().isMinIntrinsic())
1167 minWidth = child->minPreferredLogicalWidth();
1168 else if (child->style().minWidth().isAuto())
1169 minWidth = 0;
1170
1171 LayoutUnit allowedShrinkage = std::min<LayoutUnit>(0, minWidth - width);
1172 return allowedShrinkage;
1173 } else {
1174 Length minHeight = child->style().minHeight();
1175 if (minHeight.isFixed() || minHeight.isAuto()) {
1176 LayoutUnit minHeight { child->style().minHeight().value() };
1177 LayoutUnit height = contentHeightForChild(child);
1178 LayoutUnit allowedShrinkage = std::min<LayoutUnit>(0, minHeight - height);
1179 return allowedShrinkage;
1180 }
1181 }
1182
1183 return 0;
