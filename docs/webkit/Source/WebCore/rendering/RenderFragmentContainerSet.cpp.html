48void RenderFragmentContainerSet::expandToEncompassFragmentedFlowContentsIfNeeded()
49{
50 // Whenever the last region is a set, it always expands its region rect to consume all
51 // of the flow thread content. This is because it is always capable of generating an
52 // infinite number of boxes in order to hold all of the remaining content.
53 auto rect = fragmentedFlowPortionRect();
54
55 // Get the offset within the flow thread in its block progression direction. Then get the
56 // flow thread's remaining logical height including its overflow and expand our rect
57 // to encompass that remaining height and overflow. The idea is that we will generate
58 // additional columns and pages to hold that overflow, since people do write bad
59 // content like <body style="height:0px"> in multi-column layouts.
60 bool isHorizontal = fragmentedFlow()->isHorizontalWritingMode();
61 auto logicalTopOffset = isHorizontal ? rect.y() : rect.x();
62 auto overflowHeight = isHorizontal ? fragmentedFlow()->layoutOverflowRect().maxY() : fragmentedFlow()->layoutOverflowRect().maxX();
63 auto logicalHeightWithOverflow = logicalTopOffset == RenderFragmentedFlow::maxLogicalHeight() ? overflowHeight : overflowHeight - logicalTopOffset;
64 setFragmentedFlowPortionRect({ rect.x(), rect.y(), isHorizontal ? rect.width() : logicalHeightWithOverflow, isHorizontal ? logicalHeightWithOverflow : rect.height() });
65}
