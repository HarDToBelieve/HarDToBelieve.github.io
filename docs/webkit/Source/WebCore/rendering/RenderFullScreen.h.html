32namespace WebCore {
33
34class RenderFullScreen final : public RenderFlexibleBox {
35 WTF_MAKE_ISO_ALLOCATED(RenderFullScreen);
36public:
37 RenderFullScreen(Document&, RenderStyle&&);
38 virtual ~RenderFullScreen();
39
40 const char* renderName() const override { return "RenderFullScreen"; }
41
42 RenderBlock* placeholder() { return m_placeholder.get(); }
43 void setPlaceholder(RenderBlock& placeholder) { m_placeholder = makeWeakPtr(placeholder); }
44
45 static RenderPtr<RenderFullScreen> wrapNewRenderer(RenderTreeBuilder&, RenderPtr<RenderElement>, RenderElement& parent, Document&);
46 static void wrapExistingRenderer(RenderElement&, Document&);
47 void unwrapRenderer(bool& requiresRenderTreeRebuild);
48
49 ItemPosition selfAlignmentNormalBehavior(const RenderBox* = nullptr) const override { return ItemPosition::Center; }
50
51private:
52 bool isRenderFullScreen() const override { return true; }
53 bool isFlexibleBoxImpl() const override { return true; }
54
55 WeakPtr<RenderBlock> m_placeholder;
56};
57
58class RenderFullScreenPlaceholder final : public RenderBlockFlow {
59 WTF_MAKE_ISO_ALLOCATED(RenderFullScreenPlaceholder);
60public:
61 RenderFullScreenPlaceholder(Document&, RenderStyle&&);
62
63private:
64 bool isRenderFullScreenPlaceholder() const override;
65};

67} // namespace WebCore
