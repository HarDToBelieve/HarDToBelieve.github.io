58bool RenderHTMLCanvas::requiresLayer() const
59{
60 if (RenderReplaced::requiresLayer())
61 return true;
62
63 if (CanvasRenderingContext* context = canvasElement().renderingContext())
64 return context->isAccelerated();
65
66 return false;
67}

69void RenderHTMLCanvas::paintReplaced(PaintInfo& paintInfo, const LayoutPoint& paintOffset)
70{
71 GraphicsContext& context = paintInfo.context();
72
73 LayoutRect contentBoxRect = this->contentBoxRect();
74
75 if (context.detectingContentfulPaint()) {
76 if (!context.contenfulPaintDetected() && canvasElement().renderingContext())
77 context.setContentfulPaintDetected();
78 return;
79 }
80
81 contentBoxRect.moveBy(paintOffset);
82 LayoutRect replacedContentRect = this->replacedContentRect();
83 replacedContentRect.moveBy(paintOffset);
84
85 // Not allowed to overflow the content box.
86 bool clip = !contentBoxRect.contains(replacedContentRect);
87 GraphicsContextStateSaver stateSaver(paintInfo.context(), clip);
88 if (clip)
89 paintInfo.context().clip(snappedIntRect(contentBoxRect));
90
91 if (paintInfo.phase == PaintPhase::Foreground)
92 page().addRelevantRepaintedObject(this, intersection(replacedContentRect, contentBoxRect));
93
94 InterpolationQualityMaintainer interpolationMaintainer(context, ImageQualityController::interpolationQualityFromStyle(style()));
95
96 canvasElement().setIsSnapshotting(paintInfo.paintBehavior.contains(PaintBehavior::Snapshotting));
97 canvasElement().paint(context, replacedContentRect);
98 canvasElement().setIsSnapshotting(false);
99}

101void RenderHTMLCanvas::canvasSizeChanged()
102{
103 IntSize canvasSize = canvasElement().size();
104 LayoutSize zoomedSize(canvasSize.width() * style().effectiveZoom(), canvasSize.height() * style().effectiveZoom());
105
106 if (zoomedSize == intrinsicSize())
107 return;
108
109 setIntrinsicSize(zoomedSize);
110
111 if (!parent())
112 return;
113 setNeedsLayoutIfNeededAfterIntrinsicSizeChange();
114}
