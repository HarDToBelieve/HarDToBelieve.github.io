71void RenderInline::willBeDestroyed()
72{
73#if ASSERT_ENABLED
74 // Make sure we do not retain "this" in the continuation outline table map of our containing blocks.
75 if (parent() && style().visibility() == Visibility::Visible && hasOutline()) {
76 bool containingBlockPaintsContinuationOutline = continuation() || isContinuation();
77 if (containingBlockPaintsContinuationOutline) {
78 if (RenderBlock* cb = containingBlock()) {
79 if (RenderBlock* cbCb = cb->containingBlock())
80 ASSERT(!cbCb->paintsContinuationOutline(this));
81 }
82 }
83 }
84#endif // ASSERT_ENABLED
85
86 if (!renderTreeBeingDestroyed()) {
87 if (firstLineBox()) {
88 // We can't wait for RenderBoxModelObject::destroy to clear the selection,
89 // because by then we will have nuked the line boxes.
90 if (isSelectionBorder())
91 frame().selection().setNeedsSelectionUpdate();
92
93 // If line boxes are contained inside a root, that means we're an inline.
94 // In that case, we need to remove all the line boxes so that the parent
95 // lines aren't pointing to deleted children. If the first line box does
96 // not have a parent that means they are either already disconnected or
97 // root lines that can just be destroyed without disconnecting.
98 if (firstLineBox()->parent()) {
99 for (auto* box = firstLineBox(); box; box = box->nextLineBox())
100 box->removeFromParent();
101 }
102 } else if (parent())
103 parent()->dirtyLinesFromChangedChild(*this);
104 }
105
106 m_lineBoxes.deleteLineBoxes();
107
108 RenderBoxModelObject::willBeDestroyed();
109}

120static RenderElement* inFlowPositionedInlineAncestor(RenderElement* p)
121{
122 while (p && p->isRenderInline()) {
123 if (p->isInFlowPositioned())
124 return p;
125 p = p->parent();
126 }
127 return nullptr;

130static void updateStyleOfAnonymousBlockContinuations(const RenderBlock& block, const RenderStyle* newStyle, const RenderStyle* oldStyle)
131{
132 // If any descendant blocks exist then they will be in the next anonymous block and its siblings.
133 for (RenderBox* box = block.nextSiblingBox(); box && box->isAnonymousBlock(); box = box->nextSiblingBox()) {
134 if (box->style().position() == newStyle->position())
135 continue;
136
137 if (!is<RenderBlock>(*box))
138 continue;
139
140 RenderBlock& block = downcast<RenderBlock>(*box);
141 if (!block.isContinuation())
142 continue;
143
144 // If we are no longer in-flow positioned but our descendant block(s) still have an in-flow positioned ancestor then
145 // their containing anonymous block should keep its in-flow positioning.
146 RenderInline* continuation = block.inlineContinuation();
147 if (oldStyle->hasInFlowPosition() && inFlowPositionedInlineAncestor(continuation))
148 continue;
149 auto blockStyle = RenderStyle::createAnonymousStyleWithDisplay(block.style(), DisplayType::Block);
150 blockStyle.setPosition(newStyle->position());
151 block.setStyle(WTFMove(blockStyle));
152 }
153}

155void RenderInline::styleWillChange(StyleDifference diff, const RenderStyle& newStyle)
156{
157 RenderBoxModelObject::styleWillChange(diff, newStyle);
158 // RenderInlines forward their absolute positioned descendants to their (non-anonymous) containing block.
159 // Check if this non-anonymous containing block can hold the absolute positioned elements when the inline is no longer positioned.
160 if (canContainAbsolutelyPositionedObjects() && newStyle.position() == PositionType::Static) {
161 auto* container = containingBlockForAbsolutePosition();
162 if (container && !container->canContainAbsolutelyPositionedObjects())
163 container->removePositionedObjects(nullptr, NewContainingBlock);
164 }

167void RenderInline::styleDidChange(StyleDifference diff, const RenderStyle* oldStyle)
168{
169 RenderBoxModelObject::styleDidChange(diff, oldStyle);
170
171 // Ensure that all of the split inlines pick up the new style. We
172 // only do this if we're an inline, since we don't want to propagate
173 // a block's style to the other inlines.
174 // e.g., <font>foo <h4>goo</h4> moo</font>. The <font> inlines before
175 // and after the block share the same style, but the block doesn't
176 // need to pass its style on to anyone else.
177 auto& newStyle = style();
178 RenderInline* continuation = inlineContinuation();
179 if (continuation && !isContinuation()) {
180 for (RenderInline* currCont = continuation; currCont; currCont = currCont->inlineContinuation())
181 currCont->setStyle(RenderStyle::clone(newStyle));
182 // If an inline's in-flow positioning has changed and it is part of an active continuation as a descendant of an anonymous containing block,
183 // then any descendant blocks will need to change their in-flow positioning accordingly.
184 // Do this by updating the position of the descendant blocks' containing anonymous blocks - there may be more than one.
185 if (containingBlock()->isAnonymousBlock() && oldStyle && newStyle.position() != oldStyle->position() && (newStyle.hasInFlowPosition() || oldStyle->hasInFlowPosition()))
186 updateStyleOfAnonymousBlockContinuations(*containingBlock(), &newStyle, oldStyle);
187 }
188
189 if (!alwaysCreateLineBoxes()) {
190 bool alwaysCreateLineBoxes = hasSelfPaintingLayer() || hasVisibleBoxDecorations() || newStyle.hasBorder() || newStyle.hasPadding() || newStyle.hasMargin() || hasOutline();
191 if (oldStyle && alwaysCreateLineBoxes) {
192 dirtyLineBoxes(false);
193 setNeedsLayout();
194 }
195 setRenderInlineAlwaysCreatesLineBoxes(alwaysCreateLineBoxes);
196 }
197
198#if ENABLE(LAYOUT_FORMATTING_CONTEXT)
199 if (diff >= StyleDifference::Repaint) {
200 if (auto* lineLayout = LayoutIntegration::LineLayout::containing(*this)) {
201 if (selfNeedsLayout()) {
202 // FIXME: Add support for partial invalidation.
203 if (auto* container = LayoutIntegration::LineLayout::blockContainer(*this))
204 container->invalidateLineLayoutPath();
205 } else
206 lineLayout->updateStyle(*this);
207 }
208 }
209#endif
210}

212bool RenderInline::shouldCreateLineBoxes() const
213{
214 // Test if we can get away with culling.
215 auto* parentStyle = &parent()->style();
216 RenderInline* parentRenderInline = is<RenderInline>(*parent()) ? downcast<RenderInline>(parent()) : nullptr;
217 auto hasHardLineBreakChildOnly = firstChild() && firstChild() == lastChild() && firstChild()->isBR();
218 bool checkFonts = document().inNoQuirksMode();
219 auto needsLineBoxes = (parentRenderInline && parentRenderInline->alwaysCreateLineBoxes())
220 || (parentRenderInline && parentStyle->verticalAlign() != VerticalAlign::Baseline)
221 || style().verticalAlign() != VerticalAlign::Baseline
222 || style().textEmphasisMark() != TextEmphasisMark::None
223 || (checkFonts && (!parentStyle->fontCascade().fontMetrics().hasIdenticalAscentDescentAndLineGap(style().fontCascade().fontMetrics())
224 || parentStyle->lineHeight() != style().lineHeight()))
225 || hasHardLineBreakChildOnly;
226
227 if (!needsLineBoxes && checkFonts && view().usesFirstLineRules()) {
228 // Have to check the first line style as well.
229 parentStyle = &parent()->firstLineStyle();
230 auto& childStyle = firstLineStyle();
231 needsLineBoxes = !parentStyle->fontCascade().fontMetrics().hasIdenticalAscentDescentAndLineGap(childStyle.fontCascade().fontMetrics())
232 || childStyle.verticalAlign() != VerticalAlign::Baseline
233 || parentStyle->lineHeight() != childStyle.lineHeight();
234 }
235 return needsLineBoxes;
236}

238void RenderInline::updateAlwaysCreateLineBoxes(bool fullLayout)
239{
240 // Once we have been tainted once, just assume it will happen again. This way effects like hover highlighting that change the
241 // background color will only cause a layout on the first rollover.
242 if (alwaysCreateLineBoxes() || !shouldCreateLineBoxes())
243 return;
244
245 setAlwaysCreateLineBoxes();
246 if (!fullLayout)
247 dirtyLineBoxes(false);
248}

255template<typename GeneratorContext>
256void RenderInline::generateLineBoxRects(GeneratorContext& context) const
257{
258 if (!alwaysCreateLineBoxes())
259 generateCulledLineBoxRects(context, this);
260 else if (LegacyInlineFlowBox* curr = firstLineBox()) {
261 for (; curr; curr = curr->nextLineBox())
262 context.addRect(FloatRect(curr->topLeft(), curr->size()));
263 } else
264 context.addRect(FloatRect());
265}
266
267template<typename GeneratorContext>
268void RenderInline::generateCulledLineBoxRects(GeneratorContext& context, const RenderInline* container) const
269{
270 if (!culledInlineFirstLineBox()) {
271 context.addRect(FloatRect());
272 return;
273 }
274
275 bool isHorizontal = style().isHorizontalWritingMode();
276
277 for (auto& current : childrenOfType<RenderObject>(*this)) {
278 if (current.isFloatingOrOutOfFlowPositioned())
279 continue;
280
281 // We want to get the margin box in the inline direction, and then use our font ascent/descent in the block
282 // direction (aligned to the root box's baseline).
283 if (is<RenderBox>(current)) {
284 auto& renderBox = downcast<RenderBox>(current);
285 if (renderBox.inlineBoxWrapper()) {
286 const LegacyRootInlineBox& rootBox = renderBox.inlineBoxWrapper()->root();
287 const RenderStyle& containerStyle = rootBox.isFirstLine() ? container->firstLineStyle() : container->style();
288 int logicalTop = rootBox.logicalTop() + (rootBox.lineStyle().fontCascade().fontMetrics().ascent() - containerStyle.fontCascade().fontMetrics().ascent());
289 int logicalHeight = containerStyle.fontCascade().fontMetrics().height();
290 if (isHorizontal)
291 context.addRect(FloatRect(renderBox.inlineBoxWrapper()->x() - renderBox.marginLeft(), logicalTop, renderBox.width() + renderBox.horizontalMarginExtent(), logicalHeight));
292 else
293 context.addRect(FloatRect(logicalTop, renderBox.inlineBoxWrapper()->y() - renderBox.marginTop(), logicalHeight, renderBox.height() + renderBox.verticalMarginExtent()));
294 }
295 } else if (is<RenderInline>(current)) {
296 // If the child doesn't need line boxes either, then we can recur.
297 auto& renderInline = downcast<RenderInline>(current);
298 if (!renderInline.alwaysCreateLineBoxes())
299 renderInline.generateCulledLineBoxRects(context, container);
300 else {
301 for (auto* childLine = renderInline.firstLineBox(); childLine; childLine = childLine->nextLineBox()) {
302 const LegacyRootInlineBox& rootBox = childLine->root();
303 const RenderStyle& containerStyle = rootBox.isFirstLine() ? container->firstLineStyle() : container->style();
304 int logicalTop = rootBox.logicalTop() + (rootBox.lineStyle().fontCascade().fontMetrics().ascent() - containerStyle.fontCascade().fontMetrics().ascent());
305 int logicalHeight = containerStyle.fontMetrics().height();
306 if (isHorizontal) {
307 context.addRect(FloatRect(childLine->x() - childLine->marginLogicalLeft(),
308 logicalTop,
309 childLine->logicalWidth() + childLine->marginLogicalLeft() + childLine->marginLogicalRight(),
310 logicalHeight));
311 } else {
312 context.addRect(FloatRect(logicalTop,
313 childLine->y() - childLine->marginLogicalLeft(),
314 logicalHeight,
315 childLine->logicalWidth() + childLine->marginLogicalLeft() + childLine->marginLogicalRight()));
316 }
317 }
318 }
319 } else if (is<RenderText>(current)) {
320 auto& currText = downcast<RenderText>(current);
321 for (auto* childText = currText.firstTextBox(); childText; childText = childText->nextTextBox()) {
322 const LegacyRootInlineBox& rootBox = childText->root();
323 const RenderStyle& containerStyle = rootBox.isFirstLine() ? container->firstLineStyle() : container->style();
324 int logicalTop = rootBox.logicalTop() + (rootBox.lineStyle().fontCascade().fontMetrics().ascent() - containerStyle.fontCascade().fontMetrics().ascent());
325 int logicalHeight = containerStyle.fontCascade().fontMetrics().height();
326 if (isHorizontal)
327 context.addRect(FloatRect(childText->x(), logicalTop, childText->logicalWidth(), logicalHeight));
328 else
329 context.addRect(FloatRect(logicalTop, childText->y(), logicalHeight, childText->logicalWidth()));
330 }
331 } else if (is<RenderLineBreak>(current)) {
332 if (auto* inlineBox = downcast<RenderLineBreak>(current).inlineBoxWrapper()) {
333 // FIXME: This could use a helper to share these with text path.
334 const LegacyRootInlineBox& rootBox = inlineBox->root();
335 const RenderStyle& containerStyle = rootBox.isFirstLine() ? container->firstLineStyle() : container->style();
336 int logicalTop = rootBox.logicalTop() + (rootBox.lineStyle().fontCascade().fontMetrics().ascent() - containerStyle.fontCascade().fontMetrics().ascent());
337 int logicalHeight = containerStyle.fontMetrics().height();
338 if (isHorizontal)
339 context.addRect(FloatRect(inlineBox->x(), logicalTop, inlineBox->logicalWidth(), logicalHeight));
340 else
341 context.addRect(FloatRect(logicalTop, inlineBox->y(), logicalHeight, inlineBox->logicalWidth()));
342 }
343 }
344 }

368void RenderInline::absoluteRects(Vector<IntRect>& rects, const LayoutPoint& accumulatedOffset) const
369{
370#if ENABLE(LAYOUT_FORMATTING_CONTEXT)
371 if (auto* lineLayout = LayoutIntegration::LineLayout::containing(const_cast<RenderInline&>(*this)))
372 lineLayout->flow().ensureLineBoxes();
373#endif
374 Vector<LayoutRect> lineboxRects;
375 AbsoluteRectsGeneratorContext context(lineboxRects, accumulatedOffset);
376 generateLineBoxRects(context);
377 for (const auto& rect : lineboxRects)
378 rects.append(snappedIntRect(rect));
379
380 if (RenderBoxModelObject* continuation = this->continuation()) {
381 if (is<RenderBox>(*continuation)) {
382 auto& box = downcast<RenderBox>(*continuation);
383 continuation->absoluteRects(rects, toLayoutPoint(accumulatedOffset - containingBlock()->location() + box.locationOffset()));
384 } else
385 continuation->absoluteRects(rects, toLayoutPoint(accumulatedOffset - containingBlock()->location()));
386 }
387}

469static LayoutUnit computeMargin(const RenderInline* renderer, const Length& margin)
470{
471 if (margin.isAuto())
472 return 0;
473 if (margin.isFixed())
474 return LayoutUnit(margin.value());
475 if (margin.isPercentOrCalculated())
476 return minimumValueForLength(margin, std::max<LayoutUnit>(0, renderer->containingBlock()->availableLogicalWidth()));
477 return 0;
478}

570bool RenderInline::hitTestCulledInline(const HitTestRequest& request, HitTestResult& result, const HitTestLocation& locationInContainer, const LayoutPoint& accumulatedOffset)
571{
572 ASSERT(result.isRectBasedTest() && !alwaysCreateLineBoxes());
573 if (!visibleToHitTesting(request))
574 return false;
575
576 HitTestLocation tmpLocation(locationInContainer, -toLayoutSize(accumulatedOffset));
577
578 Region regionResult;
579 HitTestCulledInlinesGeneratorContext context(regionResult, tmpLocation);
580 generateCulledLineBoxRects(context, this);
581
582 if (context.intersected()) {
583 updateHitTestResult(result, tmpLocation.point());
584 // We cannot use addNodeToListBasedTestResult to determine if we fully enclose the hit-test area
585 // because it can only handle rectangular targets.
586 result.addNodeToListBasedTestResult(nodeForHitTest(), request, locationInContainer);
587 return regionResult.contains(enclosingIntRect(tmpLocation.boundingBox()));
588 }
589 return false;
590}

592VisiblePosition RenderInline::positionForPoint(const LayoutPoint& point, const RenderFragmentContainer* fragment)
593{
594 // FIXME: Does not deal with relative or sticky positioned inlines (should it?)
595 RenderBlock& containingBlock = *this->containingBlock();
596 if (firstLineBox()) {
597 // This inline actually has a line box. We must have clicked in the border/padding of one of these boxes. We
598 // should try to find a result by asking our containing block.
599 return containingBlock.positionForPoint(point, fragment);
600 }
601
602 // Translate the coords from the pre-anonymous block to the post-anonymous block.
603 LayoutPoint parentBlockPoint = containingBlock.location() + point;
604 RenderBoxModelObject* continuation = this->continuation();
605 while (continuation) {
606 RenderBlock* currentBlock = continuation->isInline() ? continuation->containingBlock() : downcast<RenderBlock>(continuation);
607 if (continuation->isInline() || continuation->firstChild())
608 return continuation->positionForPoint(parentBlockPoint - currentBlock->locationOffset(), fragment);
609 continuation = continuation->inlineContinuation();
610 }
611
612 return RenderBoxModelObject::positionForPoint(point, fragment);
613}

631IntRect RenderInline::linesBoundingBox() const
632{
633#if ENABLE(LAYOUT_FORMATTING_CONTEXT)
634 if (auto* layout = LayoutIntegration::LineLayout::containing(*this))
635 return enclosingIntRect(layout->enclosingBorderBoxRectFor(*this));
636#endif
637
638 if (!alwaysCreateLineBoxes()) {
639 ASSERT(!firstLineBox());
640 FloatRect floatResult;
641 LinesBoundingBoxGeneratorContext context(floatResult);
642 generateCulledLineBoxRects(context, this);
643 return enclosingIntRect(floatResult);
644 }
645
646 IntRect result;
647
648 // See <rdar://problem/5289721>, for an unknown reason the linked list here is sometimes inconsistent, first is non-zero and last is zero. We have been
649 // unable to reproduce this at all (and consequently unable to figure ot why this is happening). The assert will hopefully catch the problem in debug
650 // builds and help us someday figure out why. We also put in a redundant check of lastLineBox() to avoid the crash for now.
651 ASSERT(!firstLineBox() == !lastLineBox()); // Either both are null or both exist.
652 if (firstLineBox() && lastLineBox()) {
653 // Return the width of the minimal left side and the maximal right side.
654 float logicalLeftSide = 0;
655 float logicalRightSide = 0;
656 for (auto* curr = firstLineBox(); curr; curr = curr->nextLineBox()) {
657 if (curr == firstLineBox() || curr->logicalLeft() < logicalLeftSide)
658 logicalLeftSide = curr->logicalLeft();
659 if (curr == firstLineBox() || curr->logicalRight() > logicalRightSide)
660 logicalRightSide = curr->logicalRight();
661 }
662
663 bool isHorizontal = style().isHorizontalWritingMode();
664
665 float x = isHorizontal ? logicalLeftSide : firstLineBox()->x();
666 float y = isHorizontal ? firstLineBox()->y() : logicalLeftSide;
667 float width = isHorizontal ? logicalRightSide - logicalLeftSide : lastLineBox()->logicalBottom() - x;
668 float height = isHorizontal ? lastLineBox()->logicalBottom() - y : logicalRightSide - logicalLeftSide;
669 result = enclosingIntRect(FloatRect(x, y, width, height));
670 }
671
672 return result;
673}

675LegacyInlineBox* RenderInline::culledInlineFirstLineBox() const
676{
677 for (auto& current : childrenOfType<RenderObject>(*this)) {
678 if (current.isFloatingOrOutOfFlowPositioned())
679 continue;
680
681 // We want to get the margin box in the inline direction, and then use our font ascent/descent in the block
682 // direction (aligned to the root box's baseline).
683 if (is<RenderBox>(current)) {
684 auto& renderBox = downcast<RenderBox>(current);
685 if (renderBox.inlineBoxWrapper())
686 return renderBox.inlineBoxWrapper();
687 } else if (is<RenderLineBreak>(current)) {
688 auto& renderBR = downcast<RenderLineBreak>(current);
689 if (renderBR.inlineBoxWrapper())
690 return renderBR.inlineBoxWrapper();
691 } else if (is<RenderInline>(current)) {
692 auto& renderInline = downcast<RenderInline>(current);
693 if (LegacyInlineBox* result = renderInline.firstLineBoxIncludingCulling())
694 return result;
695 } else if (is<RenderText>(current)) {
696 auto& renderText = downcast<RenderText>(current);
697 if (renderText.firstTextBox())
698 return renderText.firstTextBox();
699 }
700 }
701 return nullptr;
702}

733LayoutRect RenderInline::culledInlineVisualOverflowBoundingBox() const
734{
735 FloatRect floatResult;
736 LinesBoundingBoxGeneratorContext context(floatResult);
737 generateCulledLineBoxRects(context, this);
738 LayoutRect result(enclosingLayoutRect(floatResult));
739 bool isHorizontal = style().isHorizontalWritingMode();
740 for (auto& current : childrenOfType<RenderObject>(*this)) {
741 if (current.isFloatingOrOutOfFlowPositioned())
742 continue;
743
744 // For overflow we just have to propagate by hand and recompute it all.
745 if (is<RenderBox>(current)) {
746 auto& renderBox = downcast<RenderBox>(current);
747 if (!renderBox.hasSelfPaintingLayer() && renderBox.inlineBoxWrapper()) {
748 LayoutRect logicalRect = renderBox.logicalVisualOverflowRectForPropagation(&style());
749 if (isHorizontal) {
750 logicalRect.moveBy(renderBox.location());
751 result.uniteIfNonZero(logicalRect);
752 } else {
753 logicalRect.moveBy(renderBox.location());
754 result.uniteIfNonZero(logicalRect.transposedRect());
755 }
756 }
757 } else if (is<RenderInline>(current)) {
758 // If the child doesn't need line boxes either, then we can recur.
759 auto& renderInline = downcast<RenderInline>(current);
760 if (!renderInline.alwaysCreateLineBoxes())
761 result.uniteIfNonZero(renderInline.culledInlineVisualOverflowBoundingBox());
762 else if (!renderInline.hasSelfPaintingLayer())
763 result.uniteIfNonZero(renderInline.linesVisualOverflowBoundingBox());
764 } else if (is<RenderText>(current)) {
765 // FIXME; Overflow from text boxes is lost. We will need to cache this information in
766 // LegacyInlineTextBoxes.
767 auto& renderText = downcast<RenderText>(current);
768 result.uniteIfNonZero(renderText.linesVisualOverflowBoundingBox());
769 }
770 }
771 return result;
772}

774LayoutRect RenderInline::linesVisualOverflowBoundingBox() const
775{
776#if ENABLE(LAYOUT_FORMATTING_CONTEXT)
777 if (auto* layout = LayoutIntegration::LineLayout::containing(*this))
778 return layout->visualOverflowBoundingBoxRectFor(*this);
779#endif
780
781 if (!alwaysCreateLineBoxes())
782 return culledInlineVisualOverflowBoundingBox();
783
784 if (!firstLineBox() || !lastLineBox())
785 return LayoutRect();
786
787 // Return the width of the minimal left side and the maximal right side.
788 LayoutUnit logicalLeftSide = LayoutUnit::max();
789 LayoutUnit logicalRightSide = LayoutUnit::min();
790 for (auto* curr = firstLineBox(); curr; curr = curr->nextLineBox()) {
791 logicalLeftSide = std::min(logicalLeftSide, curr->logicalLeftVisualOverflow());
792 logicalRightSide = std::max(logicalRightSide, curr->logicalRightVisualOverflow());
793 }
794
795 const LegacyRootInlineBox& firstRootBox = firstLineBox()->root();
796 const LegacyRootInlineBox& lastRootBox = lastLineBox()->root();
797
798 LayoutUnit logicalTop = firstLineBox()->logicalTopVisualOverflow(firstRootBox.lineTop());
799 LayoutUnit logicalWidth = logicalRightSide - logicalLeftSide;
800 LayoutUnit logicalHeight = lastLineBox()->logicalBottomVisualOverflow(lastRootBox.lineBottom()) - logicalTop;
801
802 LayoutRect rect(logicalLeftSide, logicalTop, logicalWidth, logicalHeight);
803 if (!style().isHorizontalWritingMode())
804 rect = rect.transposedRect();
805 return rect;
806}

808LayoutRect RenderInline::linesVisualOverflowBoundingBoxInFragment(const RenderFragmentContainer* fragment) const
809{
810 ASSERT(alwaysCreateLineBoxes());
811 ASSERT(fragment);
812
813 if (!firstLineBox() || !lastLineBox())
814 return LayoutRect();
815
816 // Return the width of the minimal left side and the maximal right side.
817 LayoutUnit logicalLeftSide = LayoutUnit::max();
818 LayoutUnit logicalRightSide = LayoutUnit::min();
819 LayoutUnit logicalTop;
820 LayoutUnit logicalHeight;
821 LegacyInlineFlowBox* lastInlineInFragment = 0;
822 for (auto* curr = firstLineBox(); curr; curr = curr->nextLineBox()) {

828 }

836 logicalRightSide = std::max(logicalRightSide, curr->logicalRightVisualOverflow());
837 }

840 return LayoutRect();
841
842 logicalHeight = lastInlineInFragment->logicalBottomVisualOverflow(lastInlineInFragment->root().lineBottom()) - logicalTop;
843
844 LayoutUnit logicalWidth = logicalRightSide - logicalLeftSide;
845
846 LayoutRect rect(logicalLeftSide, logicalTop, logicalWidth, logicalHeight);
847 if (!style().isHorizontalWritingMode())
848 rect = rect.transposedRect();
849 return rect;

852LayoutRect RenderInline::clippedOverflowRect(const RenderLayerModelObject* repaintContainer, VisibleRectContext context) const
853{
854 // Only first-letter renderers are allowed in here during layout. They mutate the tree triggering repaints.
855 ASSERT(!view().frameView().layoutContext().isPaintOffsetCacheEnabled() || style().styleType() == PseudoId::FirstLetter || hasSelfPaintingLayer());
856
857 auto knownEmpty = [&] {
858 if (firstLineBoxIncludingCulling())
859 return false;
860 if (continuation())
861 return false;
862#if ENABLE(LAYOUT_FORMATTING_CONTEXT)
863 if (LayoutIntegration::LineLayout::containing(*this))
864 return false;
865#endif
866 return true;
867 };
868
869 if (knownEmpty())
870 return LayoutRect();
871
872 LayoutRect repaintRect(linesVisualOverflowBoundingBox());
873 bool hitRepaintContainer = false;
874
875 // We need to add in the in-flow position offsets of any inlines (including us) up to our
876 // containing block.
877 RenderBlock* containingBlock = this->containingBlock();
878 for (const RenderElement* inlineFlow = this; is<RenderInline>(inlineFlow) && inlineFlow != containingBlock;
879 inlineFlow = inlineFlow->parent()) {
880 if (inlineFlow == repaintContainer) {
881 hitRepaintContainer = true;
882 break;
883 }
884 if (inlineFlow->style().hasInFlowPosition() && inlineFlow->hasLayer())
885 repaintRect.move(downcast<RenderInline>(*inlineFlow).layer()->offsetForInFlowPosition());
886 }
887
888 LayoutUnit outlineSize { style().outlineSize() };
889 repaintRect.inflate(outlineSize);
890
891 if (hitRepaintContainer || !containingBlock)
892 return repaintRect;
893
894 if (containingBlock->hasOverflowClip())
895 containingBlock->applyCachedClipAndScrollPosition(repaintRect, repaintContainer, context);
896
897 repaintRect = containingBlock->computeRect(repaintRect, repaintContainer, context);
898
899 if (outlineSize) {
900 for (auto& child : childrenOfType<RenderElement>(*this))
901 repaintRect.unite(child.rectWithOutlineForRepaint(repaintContainer, outlineSize));
902
903 if (RenderBoxModelObject* continuation = this->continuation()) {
904 if (!continuation->isInline() && continuation->parent())
905 repaintRect.unite(continuation->rectWithOutlineForRepaint(repaintContainer, outlineSize));
906 }
907 }
908
909 return repaintRect;
910}

912LayoutRect RenderInline::rectWithOutlineForRepaint(const RenderLayerModelObject* repaintContainer, LayoutUnit outlineWidth) const
913{
914 LayoutRect r(RenderBoxModelObject::rectWithOutlineForRepaint(repaintContainer, outlineWidth));
915 for (auto& child : childrenOfType<RenderElement>(*this))
916 r.unite(child.rectWithOutlineForRepaint(repaintContainer, outlineWidth));
917 return r;

920LayoutRect RenderInline::computeVisibleRectUsingPaintOffset(const LayoutRect& rect) const
921{
922 LayoutRect adjustedRect = rect;
923 auto* layoutState = view().frameView().layoutContext().layoutState();
924 if (style().hasInFlowPosition() && layer())
925 adjustedRect.move(layer()->offsetForInFlowPosition());

932std::optional<LayoutRect> RenderInline::computeVisibleRectInContainer(const LayoutRect& rect, const RenderLayerModelObject* container, VisibleRectContext context) const
933{
934 // Repaint offset cache is only valid for root-relative repainting
935 if (view().frameView().layoutContext().isPaintOffsetCacheEnabled() && !container && !context.options.contains(VisibleRectContextOption::UseEdgeInclusiveIntersection))
936 return computeVisibleRectUsingPaintOffset(rect);
937
938 if (container == this)
939 return rect;
940
941 bool containerSkipped;
942 RenderElement* localContainer = this->container(container, containerSkipped);
943 if (!localContainer)
944 return rect;
945
946 LayoutRect adjustedRect = rect;
947 LayoutPoint topLeft = adjustedRect.location();
948
949 if (style().hasInFlowPosition() && layer()) {
950 // Apply the in-flow position offset when invalidating a rectangle. The layer
951 // is translated, but the render box isn't, so we need to do this to get the
952 // right dirty rect. Since this is called from RenderObject::setStyle, the relative or sticky position
953 // flag on the RenderObject has been cleared, so use the one on the style().
954 topLeft += layer()->offsetForInFlowPosition();
955 }
956
957 // FIXME: We ignore the lightweight clipping rect that controls use, since if |o| is in mid-layout,
958 // its controlClipRect will be wrong. For overflow clip we use the values cached by the layer.
959 adjustedRect.setLocation(topLeft);
960 if (localContainer->hasOverflowClip()) {
961 // FIXME: Respect the value of context.options.
962 SetForScope<OptionSet<VisibleRectContextOption>> change(context.options, context.options | VisibleRectContextOption::ApplyCompositedContainerScrolls);
963 bool isEmpty = !downcast<RenderBox>(*localContainer).applyCachedClipAndScrollPosition(adjustedRect, container, context);
964 if (isEmpty) {
965 if (context.options.contains(VisibleRectContextOption::UseEdgeInclusiveIntersection))
966 return std::nullopt;
967 return adjustedRect;
968 }
969 }
970
971 if (containerSkipped) {
972 // If the repaintContainer is below o, then we need to map the rect into repaintContainer's coordinates.
973 LayoutSize containerOffset = container->offsetFromAncestorContainer(*localContainer);
974 adjustedRect.move(-containerOffset);
975 return adjustedRect;
976 }
977 return localContainer->computeVisibleRectInContainer(adjustedRect, container, context);
978}

980LayoutSize RenderInline::offsetFromContainer(RenderElement& container, const LayoutPoint&, bool* offsetDependsOnPoint) const
981{
982 ASSERT(&container == this->container());
983
984 LayoutSize offset;
985 if (isInFlowPositioned())
986 offset += offsetForInFlowPosition();
987
988 if (is<RenderBox>(container))
989 offset -= toLayoutSize(downcast<RenderBox>(container).scrollPosition());
990
991 if (offsetDependsOnPoint)
992 *offsetDependsOnPoint = (is<RenderBox>(container) && container.style().isFlippedBlocksWritingMode()) || is<RenderFragmentedFlow>(container);

997void RenderInline::mapLocalToContainer(const RenderLayerModelObject* ancestorContainer, TransformState& transformState, MapCoordinatesFlags mode, bool* wasFixed) const
998{
999 if (ancestorContainer == this)
1000 return;
1001
1002 if (view().frameView().layoutContext().isPaintOffsetCacheEnabled() && !ancestorContainer) {
1003 auto* layoutState = view().frameView().layoutContext().layoutState();
1004 LayoutSize offset = layoutState->paintOffset();
1005 if (style().hasInFlowPosition() && layer())
1006 offset += layer()->offsetForInFlowPosition();
1007 transformState.move(offset);
1008 return;
1009 }
1010
1011 bool containerSkipped;
1012 RenderElement* container = this->container(ancestorContainer, containerSkipped);
1013 if (!container)
1014 return;
1015
1016 if (mode & ApplyContainerFlip && is<RenderBox>(*container)) {
1017 if (container->style().isFlippedBlocksWritingMode()) {
1018 LayoutPoint centerPoint(transformState.mappedPoint());
1019 transformState.move(downcast<RenderBox>(*container).flipForWritingMode(centerPoint) - centerPoint);
1020 }
1021 mode &= ~ApplyContainerFlip;
1022 }
1023
1024 LayoutSize containerOffset = offsetFromContainer(*container, LayoutPoint(transformState.mappedPoint()));
1025
1026 bool preserve3D = mode & UseTransforms && (container->style().preserves3D() || style().preserves3D());
1027 if (mode & UseTransforms && shouldUseTransformFromContainer(container)) {
1028 TransformationMatrix t;
1029 getTransformFromContainer(container, containerOffset, t);
1030 transformState.applyTransform(t, preserve3D ? TransformState::AccumulateTransform : TransformState::FlattenTransform);
1031 } else
1032 transformState.move(containerOffset.width(), containerOffset.height(), preserve3D ? TransformState::AccumulateTransform : TransformState::FlattenTransform);
1033
1034 if (containerSkipped) {
1035 // There can't be a transform between ancestorContainer and o, because transforms create containers, so it should be safe
1036 // to just subtract the delta between the ancestorContainer and o.
1037 LayoutSize containerOffset = ancestorContainer->offsetFromAncestorContainer(*container);
1038 transformState.move(-containerOffset.width(), -containerOffset.height(), preserve3D ? TransformState::AccumulateTransform : TransformState::FlattenTransform);
1039 return;
1040 }
1041
1042 container->mapLocalToContainer(ancestorContainer, transformState, mode, wasFixed);
1043}
1044
1045const RenderObject* RenderInline::pushMappingToContainer(const RenderLayerModelObject* ancestorToStopAt, RenderGeometryMap& geometryMap) const
1046{
1047 ASSERT(ancestorToStopAt != this);
1048
1049 bool ancestorSkipped;
1050 RenderElement* container = this->container(ancestorToStopAt, ancestorSkipped);
1051 if (!container)
1052 return nullptr;
1053
1054 LayoutSize adjustmentForSkippedAncestor;
1055 if (ancestorSkipped) {
1056 // There can't be a transform between ancestorToStopAt and container, because transforms create containers, so it should be safe
1057 // to just subtract the delta between the ancestor and container.
1058 adjustmentForSkippedAncestor = -ancestorToStopAt->offsetFromAncestorContainer(*container);
1059 }
1060
1061 bool offsetDependsOnPoint = false;
1062 LayoutSize containerOffset = offsetFromContainer(*container, LayoutPoint(), &offsetDependsOnPoint);
1063
1064 bool preserve3D = container->style().preserves3D() || style().preserves3D();
1065 if (shouldUseTransformFromContainer(container)) {
1066 TransformationMatrix t;
1067 getTransformFromContainer(container, containerOffset, t);
1068 t.translateRight(adjustmentForSkippedAncestor.width(), adjustmentForSkippedAncestor.height()); // FIXME: right?
1069 geometryMap.push(this, t, preserve3D, offsetDependsOnPoint);
1070 } else {
1071 containerOffset += adjustmentForSkippedAncestor;
1072 geometryMap.push(this, containerOffset, preserve3D, offsetDependsOnPoint);
1073 }
1074
1075 return ancestorSkipped ? ancestorToStopAt : container;
1076}

1099void RenderInline::dirtyLineBoxes(bool fullLayout)
1100{
1101 if (fullLayout) {
1102 m_lineBoxes.deleteLineBoxes();
1103 return;
1104 }
1105
1106 if (!alwaysCreateLineBoxes()) {
1107 // We have to grovel into our children in order to dirty the appropriate lines.
1108 for (auto& current : childrenOfType<RenderObject>(*this)) {
1109 if (current.isFloatingOrOutOfFlowPositioned())
1110 continue;
1111 if (is<RenderBox>(current) && !current.needsLayout()) {
1112 auto& renderBox = downcast<RenderBox>(current);
1113 if (renderBox.inlineBoxWrapper())
1114 renderBox.inlineBoxWrapper()->root().markDirty();
1115 } else if (!current.selfNeedsLayout()) {
1116 if (is<RenderInline>(current)) {
1117 auto& renderInline = downcast<RenderInline>(current);
1118 for (auto* childLine = renderInline.firstLineBox(); childLine; childLine = childLine->nextLineBox())
1119 childLine->root().markDirty();
1120 } else if (is<RenderText>(current)) {
1121 auto& renderText = downcast<RenderText>(current);
1122 for (auto* childText = renderText.firstTextBox(); childText; childText = childText->nextTextBox())
1123 childText->root().markDirty();
1124 } else if (is<RenderLineBreak>(current)) {
1125 auto& renderBR = downcast<RenderLineBreak>(current);
1126 if (renderBR.inlineBoxWrapper())
1127 renderBR.inlineBoxWrapper()->root().markDirty();
1128 }
1129 }
1130 }
1131 } else
1132 m_lineBoxes.dirtyLineBoxes();
1133}

1171
1172LayoutSize RenderInline::offsetForInFlowPositionedInline(const RenderBox* child) const
1173{
1174 // FIXME: This function isn't right with mixed writing modes.
1175
1176 ASSERT(isInFlowPositioned());
1177 if (!isInFlowPositioned())
1178 return LayoutSize();
1179
1180 // When we have an enclosing relpositioned inline, we need to add in the offset of the first line
1181 // box from the rest of the content, but only in the cases where we know we're positioned
1182 // relative to the inline itself.
1183
1184 LayoutSize logicalOffset;
1185 LayoutUnit inlinePosition;
1186 LayoutUnit blockPosition;
1187 if (firstLineBox()) {
1188 inlinePosition = LayoutUnit::fromFloatRound(firstLineBox()->logicalLeft());
1189 blockPosition = firstLineBox()->logicalTop();
1190 } else {
1191 inlinePosition = layer()->staticInlinePosition();
1192 blockPosition = layer()->staticBlockPosition();
1193 }
1194
1195 if (!child->style().hasStaticInlinePosition(style().isHorizontalWritingMode()))
1196 logicalOffset.setWidth(inlinePosition);
1197
1198 // This is not terribly intuitive, but we have to match other browsers. Despite being a block display type inside
1199 // an inline, we still keep our x locked to the left of the relative positioned inline. Arguably the correct
1200 // behavior would be to go flush left to the block that contains the inline, but that isn't what other browsers
1201 // do.
1202 else if (!child->style().isOriginalDisplayInlineType())
1203 // Avoid adding in the left border/padding of the containing block twice. Subtract it out.
1204 logicalOffset.setWidth(inlinePosition - child->containingBlock()->borderAndPaddingLogicalLeft());
1205
1206 if (!child->style().hasStaticBlockPosition(style().isHorizontalWritingMode()))
1207 logicalOffset.setHeight(blockPosition);
1208
1209 return style().isHorizontalWritingMode() ? logicalOffset : logicalOffset.transposedSize();
1210}

1250void RenderInline::paintOutline(PaintInfo& paintInfo, const LayoutPoint& paintOffset)
1251{
1252 if (!hasOutline())
1253 return;
1254
1255 auto& styleToUse = style();
1256 // Only paint the focus ring by hand if the theme isn't able to draw it.
1257 if (styleToUse.outlineStyleIsAuto() == OutlineIsAuto::On && !theme().supportsFocusRing(styleToUse)) {
1258 Vector<LayoutRect> focusRingRects;
1259 addFocusRingRects(focusRingRects, paintOffset, paintInfo.paintContainer);
1260 paintFocusRing(paintInfo, styleToUse, focusRingRects);
1261 }
1262
1263 if (hasOutlineAnnotation() && styleToUse.outlineStyleIsAuto() == OutlineIsAuto::Off && !theme().supportsFocusRing(styleToUse))
1264 addPDFURLRect(paintInfo, paintOffset);
1265
1266 GraphicsContext& graphicsContext = paintInfo.context();
1267 if (graphicsContext.paintingDisabled())
1268 return;
1269
1270 if (styleToUse.outlineStyleIsAuto() == OutlineIsAuto::On || !styleToUse.hasOutline())
1271 return;
1272
1273 Vector<LayoutRect> rects;
1274 rects.append(LayoutRect());
1275 for (auto* curr = firstLineBox(); curr; curr = curr->nextLineBox()) {
1276 const LegacyRootInlineBox& rootBox = curr->root();
1277 LayoutUnit top = std::max(rootBox.lineTop(), LayoutUnit(curr->logicalTop()));
1278 LayoutUnit bottom = std::min(rootBox.lineBottom(), LayoutUnit(curr->logicalBottom()));
1279 rects.append({ LayoutUnit(curr->x()), top, LayoutUnit(curr->logicalWidth()), bottom - top });
1280 }
1281 rects.append(LayoutRect());
1282
1283 Color outlineColor = styleToUse.visitedDependentColorWithColorFilter(CSSPropertyOutlineColor);
1284 bool useTransparencyLayer = !outlineColor.isOpaque();
1285 if (useTransparencyLayer) {
1286 graphicsContext.beginTransparencyLayer(outlineColor.alphaAsFloat());
1287 outlineColor = outlineColor.opaqueColor();
1288 }
1289
1290 for (unsigned i = 1; i < rects.size() - 1; i++)
1291 paintOutlineForLine(graphicsContext, paintOffset, rects.at(i - 1), rects.at(i), rects.at(i + 1), outlineColor);
1292
1293 if (useTransparencyLayer)
1294 graphicsContext.endTransparencyLayer();
1295}
