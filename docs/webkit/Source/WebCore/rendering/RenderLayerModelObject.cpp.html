61void RenderLayerModelObject::willBeDestroyed()
62{
63 if (isPositioned()) {
64 if (style().hasViewportConstrainedPosition())
65 view().frameView().removeViewportConstrainedObject(*this);
66 }
67
68 if (hasLayer()) {
69 setHasLayer(false);
70 destroyLayer();
71 }
72
73 RenderElement::willBeDestroyed();
74}

76void RenderLayerModelObject::destroyLayer()
77{
78 ASSERT(!hasLayer());
79 ASSERT(m_layer);
80 m_layer = nullptr;
81}

83void RenderLayerModelObject::createLayer()
84{
85 ASSERT(!m_layer);
86 m_layer = makeUnique<RenderLayer>(*this);
87 setHasLayer(true);
88 m_layer->insertOnlyThisLayer(RenderLayer::LayerChangeTiming::StyleChange);
89}

96void RenderLayerModelObject::styleWillChange(StyleDifference diff, const RenderStyle& newStyle)
97{
98 s_wasFloating = isFloating();
99 s_hadLayer = hasLayer();
100 s_hadTransform = hasTransform();
101 if (s_hadLayer)
102 s_layerWasSelfPainting = layer()->isSelfPaintingLayer();
103
104 auto* oldStyle = hasInitializedStyle() ? &style() : nullptr;
105 if (diff == StyleDifference::RepaintLayer && parent() && oldStyle && oldStyle->clip() != newStyle.clip())
106 layer()->clearClipRectsIncludingDescendants();
107 RenderElement::styleWillChange(diff, newStyle);
108}

110void RenderLayerModelObject::styleDidChange(StyleDifference diff, const RenderStyle* oldStyle)
111{
112 RenderElement::styleDidChange(diff, oldStyle);
113 updateFromStyle();
114
115 if (requiresLayer()) {
116 if (!layer() && layerCreationAllowedForSubtree()) {
117 if (s_wasFloating && isFloating())
118 setChildNeedsLayout();
119 createLayer();
120 if (parent() && !needsLayout() && containingBlock())
121 layer()->setRepaintStatus(NeedsFullRepaint);
122 }
123 } else if (layer() && layer()->parent()) {
124#if ENABLE(CSS_COMPOSITING)
125 if (oldStyle && oldStyle->hasBlendMode())
126 layer()->willRemoveChildWithBlendMode();
127#endif
128 setHasTransformRelatedProperty(false); // All transform-related properties force layers, so we know we don't have one or the object doesn't support them.
129 setHasReflection(false);
130
131 // Repaint the about to be destroyed self-painting layer when style change also triggers repaint.
132 if (layer()->isSelfPaintingLayer() && layer()->repaintStatus() == NeedsFullRepaint && layer()->repaintRects())
133 repaintUsingContainer(containerForRepaint(), layer()->repaintRects()->clippedOverflowRect);
134
135 layer()->removeOnlyThisLayer(RenderLayer::LayerChangeTiming::StyleChange); // calls destroyLayer() which clears m_layer
136 if (s_wasFloating && isFloating())
137 setChildNeedsLayout();
138 if (s_hadTransform)
139 setNeedsLayoutAndPrefWidthsRecalc();
140 }
141
142 if (layer()) {
143 layer()->styleChanged(diff, oldStyle);
144 if (s_hadLayer && layer()->isSelfPaintingLayer() != s_layerWasSelfPainting)
145 setChildNeedsLayout();
146 }
147
148 bool newStyleIsViewportConstrained = style().hasViewportConstrainedPosition();
149 bool oldStyleIsViewportConstrained = oldStyle && oldStyle->hasViewportConstrainedPosition();
150 if (newStyleIsViewportConstrained != oldStyleIsViewportConstrained) {
151 if (newStyleIsViewportConstrained && layer())
152 view().frameView().addViewportConstrainedObject(*this);
153 else
154 view().frameView().removeViewportConstrainedObject(*this);
155 }
156
157 const RenderStyle& newStyle = style();
158 if (oldStyle && oldStyle->scrollPadding() != newStyle.scrollPadding()) {
159 if (isDocumentElementRenderer()) {
160 FrameView& frameView = view().frameView();
161 frameView.updateScrollbarSteps();
162 } else if (RenderLayer* renderLayer = layer())
163 renderLayer->updateScrollbarSteps();
164 }
165
166 bool scrollMarginChanged = oldStyle && oldStyle->scrollMargin() != newStyle.scrollMargin();
167 bool scrollAlignChanged = oldStyle && oldStyle->scrollSnapAlign() != newStyle.scrollSnapAlign();
168 bool scrollSnapStopChanged = oldStyle && oldStyle->scrollSnapStop() != newStyle.scrollSnapStop();
169 if (scrollMarginChanged || scrollAlignChanged || scrollSnapStopChanged) {
170 if (auto* scrollSnapBox = enclosingScrollableContainerForSnapping())
171 scrollSnapBox->setNeedsLayout();
172 }
173}

175bool RenderLayerModelObject::shouldPlaceVerticalScrollbarOnLeft() const
176{
177// RTL Scrollbars require some system support, and this system support does not exist on certain versions of OS X. iOS uses a separate mechanism.
178#if PLATFORM(IOS_FAMILY)
179 return false;
180#else
181 switch (settings().userInterfaceDirectionPolicy()) {
182 case UserInterfaceDirectionPolicy::Content:
183 return style().shouldPlaceVerticalScrollbarOnLeft();
184 case UserInterfaceDirectionPolicy::System:
185 return settings().systemLayoutDirection() == TextDirection::RTL;
186 }
187 ASSERT_NOT_REACHED();
188 return style().shouldPlaceVerticalScrollbarOnLeft();
189#endif
190}
