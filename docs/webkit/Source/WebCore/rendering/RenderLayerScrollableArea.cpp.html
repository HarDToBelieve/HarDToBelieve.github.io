84void RenderLayerScrollableArea::clear()
85{
86 auto& renderer = m_layer.renderer();
87 ASSERT(m_registeredScrollableArea == renderer.view().frameView().containsScrollableArea(this));
88
89 if (m_registeredScrollableArea)
90 renderer.view().frameView().removeScrollableArea(this);
91
92#if ENABLE(IOS_TOUCH_EVENTS)
93 unregisterAsTouchEventListenerForScrolling();
94#endif
95 if (Element* element = renderer.element())
96 element->setSavedLayerScrollPosition(m_scrollPosition);
97
98 destroyScrollbar(HorizontalScrollbar);
99 destroyScrollbar(VerticalScrollbar);
100
101 if (auto* scrollingCoordinator = renderer.page().scrollingCoordinator())
102 scrollingCoordinator->willDestroyScrollableArea(*this);
103
104 clearScrollCorner();
105 clearResizer();
106}

108void RenderLayerScrollableArea::restoreScrollPosition()
109{
110 auto* element = m_layer.renderer().element();
111 if (!element)
112 return;
113
114 if (m_layer.renderBox()) {
115 // We save and restore only the scrollOffset as the other scroll values are recalculated.
116 m_scrollPosition = element->savedLayerScrollPosition();
117 if (!m_scrollPosition.isZero())
118 scrollAnimator().setCurrentPosition(m_scrollPosition);
119 }
120
121 element->setSavedLayerScrollPosition(IntPoint());
122}

257void RenderLayerScrollableArea::scrollToOffset(const ScrollOffset& scrollOffset, const ScrollPositionChangeOptions& options)
258{
259 if (currentScrollBehaviorStatus() == ScrollBehaviorStatus::InNonNativeAnimation)
260 scrollAnimator().cancelAnimations();
261
262 ScrollOffset clampedScrollOffset = options.clamping == ScrollClamping::Clamped ? clampScrollOffset(scrollOffset) : scrollOffset;
263 if (clampedScrollOffset == this->scrollOffset())
264 return;
265
266 auto previousScrollType = currentScrollType();
267 setCurrentScrollType(options.type);
268
269 ScrollOffset snappedOffset = ceiledIntPoint(scrollAnimator().adjustScrollOffsetForSnappingIfNeeded(clampedScrollOffset, options.snapPointSelectionMethod));
270 auto snappedPosition = scrollPositionFromOffset(snappedOffset);
271 if (options.animated == AnimatedScroll::Yes)
272 ScrollableArea::scrollToPositionWithAnimation(snappedPosition);
273 else {
274 if (!requestScrollPositionUpdate(snappedPosition, options.type, options.clamping))
275 scrollToPositionWithoutAnimation(snappedPosition, options.clamping);
276 setScrollBehaviorStatus(ScrollBehaviorStatus::NotInAnimation);
277 }
278
279 setCurrentScrollType(previousScrollType);
280}

282void RenderLayerScrollableArea::scrollTo(const ScrollPosition& position)
283{
284 RenderBox* box = m_layer.renderBox();
285 if (!box)
286 return;
287
288 LOG_WITH_STREAM(Scrolling, stream << "RenderLayerScrollableArea [" << scrollingNodeID() << "] scrollTo " << position << " from " << m_scrollPosition << " (is user scroll " << (currentScrollType() == ScrollType::User) << ")");
289
290 ScrollPosition newPosition = position;
291 if (!box->isHTMLMarquee()) {
292 // Ensure that the dimensions will be computed if they need to be (for overflow:hidden blocks).
293 if (m_scrollDimensionsDirty)
294 computeScrollDimensions();
295#if PLATFORM(IOS_FAMILY)
296 if (adjustForIOSCaretWhenScrolling()) {
297 // FIXME: It's not clear what this code is trying to do. Behavior seems reasonable with it removed.
298 int maxOffset = scrollWidth() - roundToInt(box->clientWidth());
299 ScrollOffset newOffset = scrollOffsetFromPosition(newPosition);
300 int scrollXOffset = newOffset.x();
301 if (scrollXOffset > maxOffset - caretWidth) {
302 scrollXOffset += caretWidth;
303 if (scrollXOffset <= caretWidth)
304 scrollXOffset = 0;
305 } else if (scrollXOffset < m_scrollPosition.x() - caretWidth)
306 scrollXOffset -= caretWidth;
307
308 newOffset.setX(scrollXOffset);
309 newPosition = scrollPositionFromOffset(newOffset);
310 }
311#endif
312 }
313
314 if (m_scrollPosition == newPosition && currentScrollBehaviorStatus() == ScrollBehaviorStatus::NotInAnimation) {
315 // FIXME: Nothing guarantees we get a scrollTo() with an unchanged position at the end of a user gesture.
316 // The ScrollingCoordinator probably needs to message the main thread when a gesture ends.
317 if (requiresScrollPositionReconciliation()) {
318 m_layer.setNeedsCompositingGeometryUpdate();
319 updateCompositingLayersAfterScroll();
320 }
321 return;
322 }
323
324 m_scrollPosition = newPosition;
325
326 auto& renderer = m_layer.renderer();
327 RenderView& view = renderer.view();
328
329 // Update the positions of our child layers (if needed as only fixed layers should be impacted by a scroll).
330 // We don't update compositing layers, because we need to do a deep update from the compositing ancestor.
331 if (!view.frameView().layoutContext().isInRenderTreeLayout()) {
332 // If we're in the middle of layout, we'll just update layers once layout has finished.
333 updateLayerPositionsAfterOverflowScroll();
334
335 view.frameView().scheduleUpdateWidgetPositions();
336
337 if (!m_updatingMarqueePosition) {
338 // Avoid updating compositing layers if, higher on the stack, we're already updating layer
339 // positions. Updating layer positions requires a full walk of up-to-date RenderLayers, and
340 // in this case we're still updating their positions; we'll update compositing layers later
341 // when that completes.
342 if (usesCompositedScrolling()) {
343 m_layer.setNeedsCompositingGeometryUpdate();
344
345 // Scroll position can affect the location of a composited descendant (which may be a sibling in z-order),
346 // so trigger a descendant walk from the stacking context.
347 if (auto* paintParent = m_layer.stackingContext())
348 paintParent->setDescendantsNeedUpdateBackingAndHierarchyTraversal();
349 }
350
351 updateCompositingLayersAfterScroll();
352 }
353
354 // Update regions, scrolling may change the clip of a particular region.
355 renderer.document().invalidateRenderingDependentRegions();
356 DebugPageOverlays::didLayout(renderer.frame());
357 }
358
359 Frame& frame = renderer.frame();
360 RenderLayerModelObject* repaintContainer = renderer.containerForRepaint();
361 // The caret rect needs to be invalidated after scrolling
362 frame.selection().setCaretRectNeedsUpdate();
363
364 LayoutRect rectForRepaint = layer().repaintRects() ? layer().repaintRects()->clippedOverflowRect : renderer.clippedOverflowRectForRepaint(repaintContainer);
365
366 FloatQuad quadForFakeMouseMoveEvent = FloatQuad(rectForRepaint);
367 if (repaintContainer)
368 quadForFakeMouseMoveEvent = repaintContainer->localToAbsoluteQuad(quadForFakeMouseMoveEvent);
369 frame.eventHandler().dispatchFakeMouseMoveEventSoonInQuad(quadForFakeMouseMoveEvent);
370
371 bool requiresRepaint = true;
372 if (usesCompositedScrolling()) {
373 m_layer.setNeedsCompositingGeometryUpdate();
374 m_layer.setDescendantsNeedUpdateBackingAndHierarchyTraversal();
375 requiresRepaint = m_layer.backing()->needsRepaintOnCompositedScroll();
376 }
377
378 // Just schedule a full repaint of our object.
379 if (requiresRepaint)
380 renderer.repaintUsingContainer(repaintContainer, rectForRepaint);
381
382 // Schedule the scroll and scroll-related DOM events.
383 if (Element* element = renderer.element())
384 element->document().addPendingScrollEventTarget(*element);
385
386 if (scrollsOverflow())
387 view.frameView().didChangeScrollOffset();
388
389 view.frameView().viewportContentsChanged();
390 frame.editor().renderLayerDidScroll(m_layer);
391}

393void RenderLayerScrollableArea::updateCompositingLayersAfterScroll()
394{
395 if (m_layer.compositor().hasContentCompositingLayers()) {
396 // Our stacking container is guaranteed to contain all of our descendants that may need
397 // repositioning, so update compositing layers from there.
398 if (RenderLayer* compositingAncestor = m_layer.stackingContext()->enclosingCompositingLayer()) {
399 if (usesCompositedScrolling())
400 m_layer.compositor().updateCompositingLayers(CompositingUpdateType::OnCompositedScroll, compositingAncestor);
401 else {
402 // FIXME: would be nice to only dirty layers whose positions were affected by scrolling.
403 compositingAncestor->setDescendantsNeedUpdateBackingAndHierarchyTraversal();
404 m_layer.compositor().updateCompositingLayers(CompositingUpdateType::OnScroll, compositingAncestor);
405 }
406 }
407 }

512IntSize RenderLayerScrollableArea::overhangAmount() const
513{
514#if ENABLE(RUBBER_BANDING)
515 auto& renderer = m_layer.renderer();
516 if (!renderer.settings().rubberBandingForSubScrollableRegionsEnabled())
517 return IntSize();
518
519 IntSize stretch;
520
521 // FIXME: use maximumScrollOffset(), or just move this to ScrollableArea.
522 ScrollOffset scrollOffset = scrollOffsetFromPosition(scrollPosition());
523 auto reachableSize = reachableTotalContentsSize();
524 if (scrollOffset.y() < 0)
525 stretch.setHeight(scrollOffset.y());
526 else if (reachableSize.height() && scrollOffset.y() > reachableSize.height() - visibleHeight())
527 stretch.setHeight(scrollOffset.y() - (reachableSize.height() - visibleHeight()));
528
529 if (scrollOffset.x() < 0)
530 stretch.setWidth(scrollOffset.x());
531 else if (reachableSize.width() && scrollOffset.x() > reachableSize.width() - visibleWidth())
532 stretch.setWidth(scrollOffset.x() - (reachableSize.width() - visibleWidth()));
533
534 return stretch;
535#else
536 return IntSize();
537#endif
538}

642RenderLayer::OverflowControlRects RenderLayerScrollableArea::overflowControlsRects() const
643{
644 auto& renderer = m_layer.renderer();
645 ASSERT(is<RenderBox>(renderer));
646 auto& renderBox = downcast<RenderBox>(renderer);
647 // Scrollbars sit inside the border box.
648 auto overflowControlsPositioningRect = snappedIntRect(renderBox.paddingBoxRectIncludingScrollbar());
649
650 auto horizontalScrollbarHeight = m_hBar ? m_hBar->height() : 0;
651 auto verticalScrollbarWidth = m_vBar ? m_vBar->width() : 0;
652
653 auto isNonOverlayScrollbar = [](const Scrollbar* scrollbar) {
654 return scrollbar && !scrollbar->isOverlayScrollbar();
655 };
656
657 bool haveNonOverlayHorizontalScrollbar = isNonOverlayScrollbar(m_hBar.get());
658 bool haveNonOverlayVerticalScrollbar = isNonOverlayScrollbar(m_vBar.get());
659 bool placeVerticalScrollbarOnTheLeft = shouldPlaceVerticalScrollbarOnLeft();
660 bool haveResizer = renderer.style().resize() != Resize::None;
661 bool scrollbarsAvoidCorner = (haveNonOverlayHorizontalScrollbar && haveNonOverlayVerticalScrollbar) || (haveResizer && (haveNonOverlayHorizontalScrollbar || haveNonOverlayVerticalScrollbar));
662
663 IntSize cornerSize;
664 if (scrollbarsAvoidCorner) {
665 // If only one scrollbar is present, the corner is square.
666 cornerSize = IntSize {
667 verticalScrollbarWidth ? verticalScrollbarWidth : horizontalScrollbarHeight,
668 horizontalScrollbarHeight ? horizontalScrollbarHeight : verticalScrollbarWidth
669 };
670 }
671
672 RenderLayer::OverflowControlRects result;
673
674 if (m_hBar) {
675 auto barRect = overflowControlsPositioningRect;
676 barRect.shiftYEdgeTo(barRect.maxY() - horizontalScrollbarHeight);
677 if (scrollbarsAvoidCorner) {
678 if (placeVerticalScrollbarOnTheLeft)
679 barRect.shiftXEdgeTo(barRect.x() + cornerSize.width());
680 else
681 barRect.contract(cornerSize.width(), 0);
682 }
683
684 result.horizontalScrollbar = barRect;
685 }
686
687 if (m_vBar) {
688 auto barRect = overflowControlsPositioningRect;
689 if (placeVerticalScrollbarOnTheLeft)
690 barRect.setWidth(verticalScrollbarWidth);
691 else
692 barRect.shiftXEdgeTo(barRect.maxX() - verticalScrollbarWidth);
693
694 if (scrollbarsAvoidCorner)
695 barRect.contract(0, cornerSize.height());
696
697 result.verticalScrollbar = barRect;
698 }
699
700 auto cornerRect = [&](IntSize cornerSize) {
701 if (placeVerticalScrollbarOnTheLeft) {
702 auto bottomLeftCorner = overflowControlsPositioningRect.minXMaxYCorner();
703 return IntRect { { bottomLeftCorner.x(), bottomLeftCorner.y() - cornerSize.height(), }, cornerSize };
704 }
705 return IntRect { overflowControlsPositioningRect.maxXMaxYCorner() - cornerSize, cornerSize };
706 };
707
708 if (scrollbarsAvoidCorner)
709 result.scrollCorner = cornerRect(cornerSize);
710
711 if (haveResizer) {
712 if (scrollbarsAvoidCorner)
713 result.resizer = result.scrollCorner;
714 else {
715 auto scrollbarThickness = ScrollbarTheme::theme().scrollbarThickness();
716 result.resizer = cornerRect({ scrollbarThickness, scrollbarThickness });
717 }
718 }
719
720 return result;
721}

1067void RenderLayerScrollableArea::updateScrollbarPresenceAndState(std::optional<bool> hasHorizontalOverflow, std::optional<bool> hasVerticalOverflow)
1068{
1069 auto* box = m_layer.renderBox();
1070 ASSERT(box);
1071
1072 enum class ScrollbarState {
1073 NoScrollbar,
1074 Enabled,
1075 Disabled
1076 };
1077
1078 auto scrollbarForAxis = [&](ScrollbarOrientation orientation) -> RefPtr<Scrollbar>& {
1079 return orientation == ScrollbarOrientation::HorizontalScrollbar ? m_hBar : m_vBar;
1080 };
1081
1082 auto stateForScrollbar = [&](ScrollbarOrientation orientation, std::optional<bool> hasOverflow, ScrollbarState nonScrollableState) {
1083 if (hasOverflow)
1084 return *hasOverflow ? ScrollbarState::Enabled : nonScrollableState;
1085
1086 // If we don't have information about overflow (because we haven't done layout yet), just return the current state of the scrollbar.
1087 auto existingScrollbar = scrollbarForAxis(orientation);
1088 return (existingScrollbar && existingScrollbar->enabled()) ? ScrollbarState::Enabled : nonScrollableState;
1089 };
1090
1091 auto stateForScrollbarOnAxis = [&](ScrollbarOrientation orientation, std::optional<bool> hasOverflow) {
1092 if (box->hasAlwaysPresentScrollbar(orientation))
1093 return stateForScrollbar(orientation, hasOverflow, ScrollbarState::Disabled);
1094
1095 if (box->hasAutoScrollbar(orientation))
1096 return stateForScrollbar(orientation, hasOverflow, ScrollbarState::NoScrollbar);
1097
1098 return ScrollbarState::NoScrollbar;
1099 };
1100
1101 auto horizontalBarState = stateForScrollbarOnAxis(ScrollbarOrientation::HorizontalScrollbar, hasHorizontalOverflow);
1102 setHasHorizontalScrollbar(horizontalBarState != ScrollbarState::NoScrollbar);
1103 if (horizontalBarState != ScrollbarState::NoScrollbar)
1104 m_hBar->setEnabled(horizontalBarState == ScrollbarState::Enabled);
1105
1106 auto verticalBarState = stateForScrollbarOnAxis(ScrollbarOrientation::VerticalScrollbar, hasVerticalOverflow);
1107 setHasVerticalScrollbar(verticalBarState != ScrollbarState::NoScrollbar);
1108 if (verticalBarState != ScrollbarState::NoScrollbar)
1109 m_vBar->setEnabled(verticalBarState == ScrollbarState::Enabled);
1110}

1117 return;
1118

1120 if (box->style().appearance() == ListboxPart)

1124 updateScrollbarPresenceAndState();
1125 bool hasVerticalScrollbar = this->hasVerticalScrollbar();

1134void RenderLayerScrollableArea::updateScrollbarsAfterLayout()
1135{
1136 RenderBox* box = m_layer.renderBox();
1137 ASSERT(box);
1138
1139 // List box parts handle the scrollbars by themselves so we have nothing to do.
1140 if (box->style().appearance() == ListboxPart)
1141 return;
1142
1143 bool hadHorizontalScrollbar = hasHorizontalScrollbar();
1144 bool hadVerticalScrollbar = hasVerticalScrollbar();
1145
1146 updateScrollbarPresenceAndState(hasHorizontalOverflow(), hasVerticalOverflow());
1147
1148 // Scrollbars with auto behavior may need to lay out again if scrollbars got added or removed.
1149 bool autoHorizontalScrollBarChanged = box->hasAutoScrollbar(ScrollbarOrientation::HorizontalScrollbar) && (hadHorizontalScrollbar != hasHorizontalScrollbar());
1150 bool autoVerticalScrollBarChanged = box->hasAutoScrollbar(ScrollbarOrientation::VerticalScrollbar) && (hadVerticalScrollbar != hasVerticalScrollbar());
1151
1152 if (autoHorizontalScrollBarChanged || autoVerticalScrollBarChanged) {
1153 if (autoVerticalScrollBarChanged && shouldPlaceVerticalScrollbarOnLeft())
1154 computeScrollOrigin();
1155
1156 m_layer.updateSelfPaintingLayer();
1157
1158 auto& renderer = m_layer.renderer();
1159 renderer.repaint();
1160
1161 if (renderer.style().overflowX() == Overflow::Auto || renderer.style().overflowY() == Overflow::Auto) {
1162 if (!m_inOverflowRelayout) {
1163 SetForScope<bool> inOverflowRelayoutScope(m_inOverflowRelayout, true);
1164 renderer.setNeedsLayout(MarkOnlyThis);
1165 if (is<RenderBlock>(renderer)) {
1166 auto& block = downcast<RenderBlock>(renderer);
1167 block.scrollbarsChanged(autoHorizontalScrollBarChanged, autoVerticalScrollBarChanged);
1168 block.layoutBlock(true);
1169 } else
1170 renderer.layout();
1171 }
1172 }
1173
1174 // FIXME: This does not belong here.
1175 RenderObject* parent = renderer.parent();
1176 if (parent && parent->isFlexibleBox() && renderer.isBox())
1177 downcast<RenderFlexibleBox>(parent)->clearCachedMainSizeForChild(*m_layer.renderBox());
1178 }
1179
1180 // Set up the range.
1181 if (m_hBar)
1182 m_hBar->setProportion(roundToInt(box->clientWidth()), m_scrollWidth);
1183 if (m_vBar)
1184 m_vBar->setProportion(roundToInt(box->clientHeight()), m_scrollHeight);
1185
1186 updateScrollbarSteps();
1187
1188 updateScrollableAreaSet(hasScrollableHorizontalOverflow() || hasScrollableVerticalOverflow());
1189}

1291void RenderLayerScrollableArea::paintOverflowControls(GraphicsContext& context, const IntPoint& paintOffset, const IntRect& damageRect, bool paintingOverlayControls)
1292{
1293 // Don't do anything if we have no overflow.
1294 auto& renderer = m_layer.renderer();
1295 if (!renderer.hasOverflowClip())
1296 return;
1297
1298 if (!showsOverflowControls())
1299 return;
1300
1301 // Overlay scrollbars paint in a second pass through the layer tree so that they will paint
1302 // on top of everything else. If this is the normal painting pass, paintingOverlayControls
1303 // will be false, and we should just tell the root layer that there are overlay scrollbars
1304 // that need to be painted. That will cause the second pass through the layer tree to run,
1305 // and we'll paint the scrollbars then. In the meantime, cache tx and ty so that the
1306 // second pass doesn't need to re-enter the RenderTree to get it right.
1307 if (hasOverlayScrollbars() && !paintingOverlayControls) {
1308 m_cachedOverlayScrollbarOffset = paintOffset;
1309
1310 // It's not necessary to do the second pass if the scrollbars paint into layers.
1311 if ((m_hBar && layerForHorizontalScrollbar()) || (m_vBar && layerForVerticalScrollbar()))
1312 return;
1313 IntRect localDamgeRect = damageRect;
1314 localDamgeRect.moveBy(-paintOffset);
1315 if (!overflowControlsIntersectRect(localDamgeRect))
1316 return;
1317
1318 RenderLayer* paintingRoot = m_layer.enclosingCompositingLayer();
1319 if (!paintingRoot)
1320 paintingRoot = renderer.view().layer();
1321
1322 if (auto* scrollableArea = paintingRoot->scrollableArea())
1323 scrollableArea->setContainsDirtyOverlayScrollbars(true);
1324 return;
1325 }
1326
1327 // This check is required to avoid painting custom CSS scrollbars twice.
1328 if (paintingOverlayControls && !hasOverlayScrollbars())
1329 return;
1330
1331 IntPoint adjustedPaintOffset = paintOffset;
1332 if (paintingOverlayControls)
1333 adjustedPaintOffset = m_cachedOverlayScrollbarOffset;
1334
1335 // Move the scrollbar widgets if necessary. We normally move and resize widgets during layout, but sometimes
1336 // widgets can move without layout occurring (most notably when you scroll a document that
1337 // contains fixed positioned elements).
1338 positionOverflowControls(toIntSize(adjustedPaintOffset));
1339
1340 // Now that we're sure the scrollbars are in the right place, paint them.
1341 if (m_hBar && !layerForHorizontalScrollbar())
1342 m_hBar->paint(context, damageRect);
1343 if (m_vBar && !layerForVerticalScrollbar())
1344 m_vBar->paint(context, damageRect);
1345
1346 if (layerForScrollCorner())
1347 return;
1348
1349 // We fill our scroll corner with white if we have a scrollbar that doesn't run all the way up to the
1350 // edge of the box.
