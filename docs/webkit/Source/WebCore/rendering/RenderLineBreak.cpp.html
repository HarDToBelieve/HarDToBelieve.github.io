51RenderLineBreak::RenderLineBreak(HTMLElement& element, RenderStyle&& style)
52 : RenderBoxModelObject(element, WTFMove(style), 0)
53 , m_inlineBoxWrapper(nullptr)
54 , m_cachedLineHeight(invalidLineHeight)
55 , m_isWBR(is<HTMLWBRElement>(element))
56{
57 setIsLineBreak();
58}
59
60RenderLineBreak::~RenderLineBreak()
61{
62 delete m_inlineBoxWrapper;
63}

65LayoutUnit RenderLineBreak::lineHeight(bool firstLine, LineDirectionMode /*direction*/, LinePositionMode /*linePositionMode*/) const
66{
67 if (firstLine && view().usesFirstLineRules()) {
68 const RenderStyle& firstLineStyle = this->firstLineStyle();
69 if (&firstLineStyle != &style())
70 return firstLineStyle.computedLineHeight();
71 }
72
73 if (m_cachedLineHeight == invalidLineHeight)
74 m_cachedLineHeight = style().computedLineHeight();
75
76 return m_cachedLineHeight;
77}
78
79LayoutUnit RenderLineBreak::baselinePosition(FontBaseline baselineType, bool firstLine, LineDirectionMode direction, LinePositionMode linePositionMode) const
80{
81 const RenderStyle& style = firstLine ? firstLineStyle() : this->style();
82 const FontMetrics& fontMetrics = style.fontMetrics();
83 return LayoutUnit { (fontMetrics.ascent(baselineType) + (lineHeight(firstLine, direction, linePositionMode) - fontMetrics.height()) / 2).toInt() };
84}

91void RenderLineBreak::setInlineBoxWrapper(LegacyInlineElementBox* inlineBox)
92{
93 ASSERT(!inlineBox || !m_inlineBoxWrapper);
94 m_inlineBoxWrapper = inlineBox;
95}
96
97void RenderLineBreak::replaceInlineBoxWrapper(LegacyInlineElementBox& inlineBox)
98{
99 deleteInlineBoxWrapper();
100 setInlineBoxWrapper(&inlineBox);
101}
102
103void RenderLineBreak::deleteInlineBoxWrapper()
104{
105 if (!m_inlineBoxWrapper)
106 return;
107 if (!renderTreeBeingDestroyed())
108 m_inlineBoxWrapper->removeFromParent();
109 delete m_inlineBoxWrapper;
110 m_inlineBoxWrapper = nullptr;
111}
112
113void RenderLineBreak::dirtyLineBoxes(bool fullLayout)
114{
115 if (!m_inlineBoxWrapper)
116 return;
117 if (fullLayout) {
118 delete m_inlineBoxWrapper;
119 m_inlineBoxWrapper = nullptr;
120 return;
121 }
122 m_inlineBoxWrapper->dirtyLineBoxes();
123}

154void RenderLineBreak::absoluteRects(Vector<IntRect>& rects, const LayoutPoint& accumulatedOffset) const
155{
156 auto box = LayoutIntegration::runFor(*this);
157 if (!box)
158 return;
159
160 auto rect = box->rect();
161 rects.append(enclosingIntRect(FloatRect(accumulatedOffset + rect.location(), rect.size())));
162}

180#if PLATFORM(IOS_FAMILY)
181void RenderLineBreak::collectSelectionGeometries(Vector<SelectionGeometry>& rects, unsigned, unsigned)
182{
183 auto run = LayoutIntegration::runFor(*this);
184
185 if (!run)
186 return;
187 auto line = run.line();
188
189 auto lineSelectionRect = line->selectionRect();
190 LayoutRect rect = IntRect(run->logicalLeft(), lineSelectionRect.y(), 0, lineSelectionRect.height());
191 if (!line->isHorizontal())
192 rect = rect.transposedRect();
193
194 if (line->legacyRootInlineBox() && line->legacyRootInlineBox()->isFirstAfterPageBreak()) {
195 if (run->isHorizontal())
196 rect.shiftYEdgeTo(line->lineBoxTop());
197 else
198 rect.shiftXEdgeTo(line->lineBoxTop());
199 }
200
201 auto* containingBlock = containingBlockForObjectInFlow();
202 // Map rect, extended left to leftOffset, and right to rightOffset, through transforms to get minX and maxX.
203 LogicalSelectionOffsetCaches cache(*containingBlock);
204 LayoutUnit leftOffset = containingBlock->logicalLeftSelectionOffset(*containingBlock, LayoutUnit(run->logicalTop()), cache);
205 LayoutUnit rightOffset = containingBlock->logicalRightSelectionOffset(*containingBlock, LayoutUnit(run->logicalTop()), cache);
206 LayoutRect extentsRect = rect;
207 if (run->isHorizontal()) {
208 extentsRect.setX(leftOffset);
209 extentsRect.setWidth(rightOffset - leftOffset);
210 } else {
211 extentsRect.setY(leftOffset);
212 extentsRect.setHeight(rightOffset - leftOffset);
213 }
214 extentsRect = localToAbsoluteQuad(FloatRect(extentsRect)).enclosingBoundingBox();
215 if (!run->isHorizontal())
216 extentsRect = extentsRect.transposedRect();
217 bool isFirstOnLine = !run.previousOnLine();
218 bool isLastOnLine = !run.nextOnLine();
219 if (containingBlock->isRubyBase() || containingBlock->isRubyText())
220 isLastOnLine = !containingBlock->containingBlock()->inlineBoxWrapper()->nextOnLineExists();
221
222 bool isFixed = false;
223 auto absoluteQuad = localToAbsoluteQuad(FloatRect(rect), UseTransforms, &isFixed);
224 bool boxIsHorizontal = !is<SVGInlineTextBox>(run->legacyInlineBox()) ? run->isHorizontal() : !style().isVerticalWritingMode();
225 // If the containing block is an inline element, we want to check the inlineBoxWrapper orientation
226 // to determine the orientation of the block. In this case we also use the inlineBoxWrapper to
227 // determine if the element is the last on the line.
228 if (containingBlock->inlineBoxWrapper()) {
229 if (containingBlock->inlineBoxWrapper()->isHorizontal() != boxIsHorizontal) {
230 boxIsHorizontal = containingBlock->inlineBoxWrapper()->isHorizontal();
231 isLastOnLine = !containingBlock->inlineBoxWrapper()->nextOnLineExists();
232 }
233 }
234
235 rects.append(SelectionGeometry(absoluteQuad, HTMLElement::selectionRenderingBehavior(element()), run->direction(), extentsRect.x(), extentsRect.maxX(), extentsRect.maxY(), 0, run->isLineBreak(), isFirstOnLine, isLastOnLine, false, false, boxIsHorizontal, isFixed, containingBlock->isRubyText(), view().pageNumberForBlockProgressionOffset(absoluteQuad.enclosingBoundingBox().x())));
236}
237#endif
