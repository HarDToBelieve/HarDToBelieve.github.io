119static FontCascade bolder(Document& document, const FontCascade& font)
120{
121 auto description = font.fontDescription();
122 description.setWeight(description.bolderWeight());
123 auto result = FontCascade { WTFMove(description), font.letterSpacing(), font.wordSpacing() };
124 result.update(&document.fontSelector());
125 return result;
126}

128void RenderListBox::updateFromElement()
129{
130 if (m_optionsChanged) {
131 float width = 0;
132 auto& normalFont = style().fontCascade();
133 std::optional<FontCascade> boldFont;
134 for (auto* element : selectElement().listItems()) {
135 String text;
136 WTF::Function<const FontCascade&()> selectFont = [&normalFont] () -> const FontCascade& {
137 return normalFont;
138 };
139 if (is<HTMLOptionElement>(*element))
140 text = downcast<HTMLOptionElement>(*element).textIndentedToRespectGroupLabel();
141 else if (is<HTMLOptGroupElement>(*element)) {
142 text = downcast<HTMLOptGroupElement>(*element).groupLabelText();
143 selectFont = [this, &normalFont, &boldFont] () -> const FontCascade& {
144 if (!boldFont)
145 boldFont = bolder(document(), normalFont);
146 return boldFont.value();
147 };
148 }
149 if (text.isEmpty())
150 continue;
151 text = applyTextTransform(style(), text, ' ');
152 auto textRun = constructTextRun(text, style(), AllowRightExpansion);
153 width = std::max(width, selectFont().width(textRun));
154 }
155 // FIXME: Is ceiling right here, or should we be doing some kind of rounding instead?
156 m_optionsWidth = static_cast<int>(std::ceil(width));
157 m_optionsChanged = false;
158
159 setHasVerticalScrollbar(true);
160
161 computeFirstIndexesVisibleInPaddingTopBottomAreas();
162
163 setNeedsLayoutAndPrefWidthsRecalc();
164 }
165}

167void RenderListBox::selectionChanged()
168{
169 repaint();
170 if (!m_inAutoscroll) {
171 if (m_optionsChanged || needsLayout())
172 m_scrollToRevealSelectionAfterLayout = true;
173 else
174 scrollToRevealSelection();
175 }
176
177 if (AXObjectCache* cache = document().existingAXObjectCache())
178 cache->deferSelectedChildrenChangedIfNeeded(selectElement());
179}

181void RenderListBox::layout()
182{
183 StackStats::LayoutCheckPoint layoutCheckPoint;
184 RenderBlockFlow::layout();
185
186 if (m_vBar) {
187 bool enabled = numVisibleItems() < numItems();
188 m_vBar->setEnabled(enabled);
189 m_vBar->setSteps(1, std::max(1, numVisibleItems() - 1), itemHeight());
190 m_vBar->setProportion(numVisibleItems(), numItems());
191 if (!enabled) {
192 scrollToOffsetWithoutAnimation(VerticalScrollbar, 0);
193 m_indexOffset = 0;
194 }
195 }
196
197 if (m_scrollToRevealSelectionAfterLayout) {
198 LayoutStateDisabler layoutStateDisabler(view().frameView().layoutContext());
199 scrollToRevealSelection();
200 }
201}

203void RenderListBox::scrollToRevealSelection()
204{
205 m_scrollToRevealSelectionAfterLayout = false;
206
207 int firstIndex = selectElement().activeSelectionStartListIndex();
208 if (firstIndex >= 0 && !listIndexIsVisible(selectElement().activeSelectionEndListIndex()))
209 scrollToRevealElementAtListIndex(firstIndex);
210}

213{
214 maxLogicalWidth = shouldApplySizeContainment(*this) ? 2 * optionsSpacingHorizontal : m_optionsWidth + 2 * optionsSpacingHorizontal;

221void RenderListBox::computePreferredLogicalWidths()
222{
223 // Nested style recal do not fire post recal callbacks. see webkit.org/b/153767
224 ASSERT(!m_optionsChanged || Style::postResolutionCallbacksAreSuspended());
225
226 m_minPreferredLogicalWidth = 0;
227 m_maxPreferredLogicalWidth = 0;
228
229 if (style().width().isFixed() && style().width().value() > 0)
230 m_minPreferredLogicalWidth = m_maxPreferredLogicalWidth = adjustContentBoxLogicalWidthForBoxSizing(style().width());
231 else
232 computeIntrinsicLogicalWidths(m_minPreferredLogicalWidth, m_maxPreferredLogicalWidth);
233
234 RenderBox::computePreferredLogicalWidths(style().minWidth(), style().maxWidth(), horizontalBorderAndPaddingExtent());
235
236 setPreferredLogicalWidthsDirty(false);
237}

248int RenderListBox::numVisibleItems(ConsiderPadding considerPadding) const
249{
250 // Only count fully visible rows. But don't return 0 even if only part of a row shows.
251 int visibleItemsExcludingPadding = std::max<int>(1, (contentHeight() + rowSpacing) / itemHeight());
252 if (considerPadding == ConsiderPadding::No)
253 return visibleItemsExcludingPadding;
254
255 return numberOfVisibleItemsInPaddingTop() + visibleItemsExcludingPadding + numberOfVisibleItemsInPaddingBottom();
256}

284LayoutRect RenderListBox::itemBoundingBoxRect(const LayoutPoint& additionalOffset, int index)
285{
286 LayoutUnit x = additionalOffset.x() + borderLeft() + paddingLeft();
287 if (shouldPlaceVerticalScrollbarOnLeft() && m_vBar)
288 x += m_vBar->occupiedWidth();
289 LayoutUnit y = additionalOffset.y() + borderTop() + paddingTop() + itemHeight() * (index - m_indexOffset);
290 return LayoutRect(x, y, contentWidth(), itemHeight());
291}

293void RenderListBox::paintItem(PaintInfo& paintInfo, const LayoutPoint& paintOffset, const PaintFunction& paintFunction)
294{
295 int listItemsSize = numItems();
296 int firstVisibleItem = m_indexOfFirstVisibleItemInsidePaddingTopArea.value_or(m_indexOffset);
297 int endIndex = firstVisibleItem + numVisibleItems(ConsiderPadding::Yes);
298 for (int i = firstVisibleItem; i < listItemsSize && i < endIndex; ++i)
299 paintFunction(paintInfo, paintOffset, i);
300}
301
302void RenderListBox::paintObject(PaintInfo& paintInfo, const LayoutPoint& paintOffset)
303{
304 if (style().visibility() != Visibility::Visible)
305 return;
306
307 if (paintInfo.phase == PaintPhase::Foreground) {
308 paintItem(paintInfo, paintOffset, [this](PaintInfo& paintInfo, const LayoutPoint& paintOffset, int listItemIndex) {
309 paintItemForeground(paintInfo, paintOffset, listItemIndex);
310 });
311 }
312
313 // Paint the children.
314 RenderBlockFlow::paintObject(paintInfo, paintOffset);
315
316 switch (paintInfo.phase) {
317 // Depending on whether we have overlay scrollbars they
318 // get rendered in the foreground or background phases
319 case PaintPhase::Foreground:
320 if (m_vBar->isOverlayScrollbar())
321 paintScrollbar(paintInfo, paintOffset);
322 break;
323 case PaintPhase::BlockBackground:
324 if (!m_vBar->isOverlayScrollbar())
325 paintScrollbar(paintInfo, paintOffset);
326 break;
327 case PaintPhase::ChildBlockBackground:
328 case PaintPhase::ChildBlockBackgrounds: {
329 paintItem(paintInfo, paintOffset, [this](PaintInfo& paintInfo, const LayoutPoint& paintOffset, int listItemIndex) {
330 paintItemBackground(paintInfo, paintOffset, listItemIndex);
331 });
332 break;
333 }
334 default:
335 break;
336 }
337}

339void RenderListBox::addFocusRingRects(Vector<LayoutRect>& rects, const LayoutPoint& additionalOffset, const RenderLayerModelObject* paintContainer)
340{
341 if (!selectElement().allowsNonContiguousSelection())
342 return RenderBlockFlow::addFocusRingRects(rects, additionalOffset, paintContainer);
343
344 // Focus the last selected item.
345 int selectedItem = selectElement().activeSelectionEndListIndex();
346 if (selectedItem >= 0) {
347 rects.append(snappedIntRect(itemBoundingBoxRect(additionalOffset, selectedItem)));
348 return;
349 }
350
351 // No selected items, find the first non-disabled item.
352 int size = numItems();
353 const Vector<HTMLElement*>& listItems = selectElement().listItems();
354 for (int i = 0; i < size; ++i) {
355 HTMLElement* element = listItems[i];
356 if (is<HTMLOptionElement>(*element) && !element->isDisabledFormControl()) {
357 selectElement().setActiveSelectionEndIndex(i);
358 rects.append(itemBoundingBoxRect(additionalOffset, i));
359 return;
360 }
361 }
362}

364void RenderListBox::paintScrollbar(PaintInfo& paintInfo, const LayoutPoint& paintOffset)
365{
366 if (!m_vBar)
367 return;
368
369 LayoutUnit left = paintOffset.x() + (shouldPlaceVerticalScrollbarOnLeft() ? borderLeft() : width() - borderRight() - m_vBar->width());
370 LayoutUnit top = paintOffset.y() + borderTop();
371 LayoutUnit width = m_vBar->width();
372 LayoutUnit height = this->height() - (borderTop() + borderBottom());
373 IntRect scrollRect = snappedIntRect(left, top, width, height);
374 m_vBar->setFrameRect(scrollRect);
375 m_vBar->paint(paintInfo.context(), snappedIntRect(paintInfo.rect));
376}

378static LayoutSize itemOffsetForAlignment(TextRun textRun, const RenderStyle* itemStyle, FontCascade itemFont, LayoutRect itemBoudingBox)
379{
380 TextAlignMode actualAlignment = itemStyle->textAlign();
381 // FIXME: Firefox doesn't respect TextAlignMode::Justify. Should we?
382 // FIXME: Handle TextAlignMode::End here
383 if (actualAlignment == TextAlignMode::Start || actualAlignment == TextAlignMode::Justify)
384 actualAlignment = itemStyle->isLeftToRightDirection() ? TextAlignMode::Left : TextAlignMode::Right;
385
386 LayoutSize offset = LayoutSize(0, itemFont.fontMetrics().ascent());
387 if (actualAlignment == TextAlignMode::Right || actualAlignment == TextAlignMode::WebKitRight) {
388 float textWidth = itemFont.width(textRun);
389 offset.setWidth(itemBoudingBox.width() - textWidth - optionsSpacingHorizontal);
390 } else if (actualAlignment == TextAlignMode::Center || actualAlignment == TextAlignMode::WebKitCenter) {
391 float textWidth = itemFont.width(textRun);
392 offset.setWidth((itemBoudingBox.width() - textWidth) / 2);
393 } else
394 offset.setWidth(optionsSpacingHorizontal);
395 return offset;
396}

398void RenderListBox::paintItemForeground(PaintInfo& paintInfo, const LayoutPoint& paintOffset, int listIndex)
399{
400 const Vector<HTMLElement*>& listItems = selectElement().listItems();
401 HTMLElement* listItemElement = listItems[listIndex];
402
403 auto& itemStyle = *listItemElement->computedStyle();
404
405 if (itemStyle.visibility() == Visibility::Hidden)
406 return;
407
408 String itemText;
409 bool isOptionElement = is<HTMLOptionElement>(*listItemElement);
410 if (isOptionElement)
411 itemText = downcast<HTMLOptionElement>(*listItemElement).textIndentedToRespectGroupLabel();
412 else if (is<HTMLOptGroupElement>(*listItemElement))
413 itemText = downcast<HTMLOptGroupElement>(*listItemElement).groupLabelText();
414 itemText = applyTextTransform(style(), itemText, ' ');
415
416 if (itemText.isNull())
417 return;
418
419 Color textColor = itemStyle.visitedDependentColorWithColorFilter(CSSPropertyColor);
420 if (isOptionElement && downcast<HTMLOptionElement>(*listItemElement).selected()) {
421 if (frame().selection().isFocusedAndActive() && document().focusedElement() == &selectElement())
422 textColor = theme().activeListBoxSelectionForegroundColor(styleColorOptions());
423 // Honor the foreground color for disabled items
424 else if (!listItemElement->isDisabledFormControl() && !selectElement().isDisabledFormControl())
425 textColor = theme().inactiveListBoxSelectionForegroundColor(styleColorOptions());
426 }
427
428 paintInfo.context().setFillColor(textColor);
429
430 TextRun textRun(itemText, 0, 0, AllowRightExpansion, itemStyle.direction(), isOverride(itemStyle.unicodeBidi()), true);
431 FontCascade itemFont = style().fontCascade();
432 LayoutRect r = itemBoundingBoxRect(paintOffset, listIndex);
433 r.move(itemOffsetForAlignment(textRun, &itemStyle, itemFont, r));
434
435 if (is<HTMLOptGroupElement>(*listItemElement)) {
436 auto description = itemFont.fontDescription();
437 description.setWeight(description.bolderWeight());
438 itemFont = FontCascade(WTFMove(description), itemFont.letterSpacing(), itemFont.wordSpacing());
439 itemFont.update(&document().fontSelector());
440 }
441
442 // Draw the item text
443 paintInfo.context().drawBidiText(itemFont, textRun, roundedIntPoint(r.location()));
444}

446void RenderListBox::paintItemBackground(PaintInfo& paintInfo, const LayoutPoint& paintOffset, int listIndex)
447{
448 const Vector<HTMLElement*>& listItems = selectElement().listItems();
449 HTMLElement* listItemElement = listItems[listIndex];
450 auto& itemStyle = *listItemElement->computedStyle();
451
452 Color backColor;
453 if (is<HTMLOptionElement>(*listItemElement) && downcast<HTMLOptionElement>(*listItemElement).selected()) {
454 if (frame().selection().isFocusedAndActive() && document().focusedElement() == &selectElement())
455 backColor = theme().activeListBoxSelectionBackgroundColor(styleColorOptions());
456 else
457 backColor = theme().inactiveListBoxSelectionBackgroundColor(styleColorOptions());
458 } else
459 backColor = itemStyle.visitedDependentColorWithColorFilter(CSSPropertyBackgroundColor);
460
461 // Draw the background for this list box item
462 if (itemStyle.visibility() == Visibility::Hidden)
463 return;
464
465 LayoutRect itemRect = itemBoundingBoxRect(paintOffset, listIndex);
466 itemRect.intersect(controlClipRect(paintOffset));
467 paintInfo.context().fillRect(snappedIntRect(itemRect), backColor);
468}

470bool RenderListBox::isPointInOverflowControl(HitTestResult& result, const LayoutPoint& locationInContainer, const LayoutPoint& accumulatedOffset)
471{
472 if (!m_vBar || !m_vBar->shouldParticipateInHitTesting())
473 return false;
474
475 LayoutUnit x = accumulatedOffset.x() + (shouldPlaceVerticalScrollbarOnLeft() ? borderLeft() : width() - borderRight() - m_vBar->width());
476 LayoutUnit y = accumulatedOffset.y() + borderTop();
477 LayoutUnit width = m_vBar->width();
478 LayoutUnit height = this->height() - borderTop() - borderBottom();
479 LayoutRect vertRect(x, y, width, height);
480
481 if (!vertRect.contains(locationInContainer))
482 return false;
483
484 result.setScrollbar(m_vBar.get());
485 return true;
486}

488int RenderListBox::listIndexAtOffset(const LayoutSize& offset)
489{
490 if (!numItems())
491 return -1;
492
493 if (offset.height() < borderTop() || offset.height() > height() - borderBottom())
494 return -1;
495
496 int scrollbarWidth = m_vBar ? m_vBar->width() : 0;
497 if (shouldPlaceVerticalScrollbarOnLeft() && (offset.width() < borderLeft() + paddingLeft() + scrollbarWidth || offset.width() > width() - borderRight() - paddingRight()))
498 return -1;
499 if (!shouldPlaceVerticalScrollbarOnLeft() && (offset.width() < borderLeft() + paddingLeft() || offset.width() > width() - borderRight() - paddingRight() - scrollbarWidth))
500 return -1;
501
502 int newOffset = (offset.height() - borderTop() - paddingTop()) / itemHeight() + m_indexOffset;
503 return newOffset < numItems() ? newOffset : -1;
504}

506void RenderListBox::panScroll(const IntPoint& panStartMousePosition)
507{
508 const int maxSpeed = 20;
509 const int iconRadius = 7;
510 const int speedReducer = 4;
511
512 // FIXME: This doesn't work correctly with transforms.
513 FloatPoint absOffset = localToAbsolute();
514
515 IntPoint lastKnownMousePosition = frame().eventHandler().lastKnownMousePosition();
516 // We need to check if the last known mouse position is out of the window. When the mouse is out of the window, the position is incoherent
517 static IntPoint previousMousePosition;
518 if (lastKnownMousePosition.y() < 0)
519 lastKnownMousePosition = previousMousePosition;
520 else
521 previousMousePosition = lastKnownMousePosition;
522
523 int yDelta = lastKnownMousePosition.y() - panStartMousePosition.y();
524
525 // If the point is too far from the center we limit the speed
526 yDelta = std::max<int>(std::min<int>(yDelta, maxSpeed), -maxSpeed);
527
528 if (abs(yDelta) < iconRadius) // at the center we let the space for the icon
529 return;
530
531 if (yDelta > 0)
532 //offsetY = view()->viewHeight();
533 absOffset.move(0, listHeight());
534 else if (yDelta < 0)
535 yDelta--;
536
537 // Let's attenuate the speed
538 yDelta /= speedReducer;
539
540 IntPoint scrollPoint(0, 0);
541 scrollPoint.setY(absOffset.y() + yDelta);
542 int newOffset = scrollToward(scrollPoint);
543 if (newOffset < 0)
544 return;
545
546 m_inAutoscroll = true;
547 selectElement().updateListBoxSelection(!selectElement().multiple());
548 m_inAutoscroll = false;
549}
550
551int RenderListBox::scrollToward(const IntPoint& destination)
552{
553 // FIXME: This doesn't work correctly with transforms.
554 FloatPoint absPos = localToAbsolute();
555 IntSize positionOffset = roundedIntSize(destination - absPos);
556
557 int rows = numVisibleItems();
558 int offset = m_indexOffset;
559
560 if (positionOffset.height() < borderTop() + paddingTop() && scrollToRevealElementAtListIndex(offset - 1))
561 return offset - 1;
562
563 if (positionOffset.height() > height() - paddingBottom() - borderBottom() && scrollToRevealElementAtListIndex(offset + rows))
564 return offset + rows - 1;
565
566 return listIndexAtOffset(positionOffset);
567}

569void RenderListBox::autoscroll(const IntPoint&)
570{
571 IntPoint pos = frame().view()->windowToContents(frame().eventHandler().lastKnownMousePosition());
572
573 int endIndex = scrollToward(pos);
574 if (selectElement().isDisabledFormControl())
575 return;
576
577 if (endIndex >= 0) {
578 m_inAutoscroll = true;
579
580 if (!selectElement().multiple())
581 selectElement().setActiveSelectionAnchorIndex(endIndex);
582
583 selectElement().setActiveSelectionEndIndex(endIndex);
584 selectElement().updateListBoxSelection(!selectElement().multiple());
585 m_inAutoscroll = false;
586 }
587}

597bool RenderListBox::scrollToRevealElementAtListIndex(int index)
598{
599 if (index < 0 || index >= numItems() || listIndexIsVisible(index))
600 return false;
601
602 int newOffset;
603 if (index < m_indexOffset)
604 newOffset = index;
605 else
606 newOffset = index - numVisibleItems() + 1;
607
608 scrollToOffsetWithoutAnimation(VerticalScrollbar, newOffset);
609
610 return true;
611}

613bool RenderListBox::listIndexIsVisible(int index)
614{
615 int firstIndex = m_indexOfFirstVisibleItemInsidePaddingTopArea.value_or(m_indexOffset);
616 int endIndex = m_indexOfFirstVisibleItemInsidePaddingBottomArea
617 ? m_indexOfFirstVisibleItemInsidePaddingBottomArea.value() + numberOfVisibleItemsInPaddingBottom()
618 : m_indexOffset + numVisibleItems();
619
620 return index >= firstIndex && index < endIndex;
621}

680void RenderListBox::computeFirstIndexesVisibleInPaddingTopBottomAreas()
681{
682 m_indexOfFirstVisibleItemInsidePaddingTopArea = std::nullopt;
683 m_indexOfFirstVisibleItemInsidePaddingBottomArea = std::nullopt;
684
685 int maximumNumberOfItemsThatFitInPaddingTopArea = paddingTop() / itemHeight();
686 if (maximumNumberOfItemsThatFitInPaddingTopArea) {
687 if (m_indexOffset)
688 m_indexOfFirstVisibleItemInsidePaddingTopArea = std::max(0, m_indexOffset - maximumNumberOfItemsThatFitInPaddingTopArea);
689 }
690
691 if (maximumNumberOfItemsThatFitInPaddingBottomArea()) {
692 if (numItems() > (m_indexOffset + numVisibleItems()))
693 m_indexOfFirstVisibleItemInsidePaddingBottomArea = m_indexOffset + numVisibleItems();
694 }
695}

704 computeFirstIndexesVisibleInPaddingTopBottomAreas();

766bool RenderListBox::nodeAtPoint(const HitTestRequest& request, HitTestResult& result, const HitTestLocation& locationInContainer, const LayoutPoint& accumulatedOffset, HitTestAction hitTestAction)
767{
768 if (!RenderBlockFlow::nodeAtPoint(request, result, locationInContainer, accumulatedOffset, hitTestAction))
769 return false;
770 const Vector<HTMLElement*>& listItems = selectElement().listItems();
771 int size = numItems();
772 LayoutPoint adjustedLocation = accumulatedOffset + location();
773
774 for (int i = 0; i < size; ++i) {
775 if (!itemBoundingBoxRect(adjustedLocation, i).contains(locationInContainer.point()))
776 continue;
777 if (Element* node = listItems[i]) {
778 result.setInnerNode(node);
779 if (!result.innerNonSharedNode())
780 result.setInnerNonSharedNode(node);
781 result.setLocalPoint(locationInContainer.point() - toLayoutSize(adjustedLocation));
782 break;
783 }
784 }
785
786 return true;
787}

916Ref<Scrollbar> RenderListBox::createScrollbar()
917{
918 RefPtr<Scrollbar> widget;
919 bool hasCustomScrollbarStyle = style().hasPseudoStyle(PseudoId::Scrollbar);
920 if (hasCustomScrollbarStyle)
921 widget = RenderScrollbar::createCustomScrollbar(*this, VerticalScrollbar, &selectElement());
922 else {
923 widget = Scrollbar::createNativeScrollbar(*this, VerticalScrollbar, theme().scrollbarControlSizeForPart(ListboxPart));
924 didAddScrollbar(widget.get(), VerticalScrollbar);
925 if (page().isMonitoringWheelEvents())
926 scrollAnimator().setWheelEventTestMonitor(page().wheelEventTestMonitor());
927 }
928 view().frameView().addChild(*widget);
929 return widget.releaseNonNull();
