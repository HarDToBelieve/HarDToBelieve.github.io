60RenderStyle RenderListItem::computeMarkerStyle() const
61{
62 if (!is<PseudoElement>(element())) {
63 if (auto markerStyle = getCachedPseudoStyle(PseudoId::Marker, &style()))
64 return RenderStyle::clone(*markerStyle);
65 }
66
67 // The marker always inherits from the list item, regardless of where it might end
68 // up (e.g., in some deeply nested line box). See CSS3 spec.
69 auto markerStyle = RenderStyle::create();
70 markerStyle.inheritFrom(style());
71
72 // In the case of a ::before or ::after pseudo-element, we manually apply the properties
73 // otherwise set in the user-agent stylesheet since we don't support ::before::marker or
74 // ::after::marker. See bugs.webkit.org/b/218897.
75 auto fontDescription = style().fontDescription();
76 fontDescription.setVariantNumericSpacing(FontVariantNumericSpacing::TabularNumbers);
77 markerStyle.setFontDescription(WTFMove(fontDescription));
78 markerStyle.fontCascade().update(&document().fontSelector());
79 markerStyle.setUnicodeBidi(EUnicodeBidi::Isolate);
80 markerStyle.setWhiteSpace(WhiteSpace::Pre);
81 markerStyle.setTextTransform(TextTransform::None);
82 return markerStyle;
83}

106// Returns the enclosing list with respect to the DOM order.
107static Element* enclosingList(const RenderListItem& listItem)
108{
109 auto& element = listItem.element();
110 auto* parent = is<PseudoElement>(element) ? downcast<PseudoElement>(element).hostElement() : element.parentElement();
111 for (auto* ancestor = parent; ancestor; ancestor = ancestor->parentElement()) {
112 if (isHTMLListElement(*ancestor))
113 return ancestor;
114 }
115
116 // If there's no actual list element, then the parent element acts as our
117 // list for purposes of determining what other list items should be numbered as
118 // part of the same list.
119 return parent;
120}
121
122static RenderListItem* nextListItemHelper(const Element& list, const Element& element)
123{
124 auto* current = &element;
125 auto advance = [&] {
126 current = ElementTraversal::nextIncludingPseudo(*current, &list);
127 };
128 advance();
129 while (current) {
130 auto* renderer = current->renderer();
131 if (!is<RenderListItem>(renderer)) {
132 advance();
133 continue;
134 }
135 auto& item = downcast<RenderListItem>(*renderer);
136 auto* otherList = enclosingList(item);
137 if (!otherList) {
138 advance();
139 continue;
140 }
141
142 // This item is part of our current list, so it's what we're looking for.
143 if (&list == otherList)
144 return &item;
145
146 // We found ourself inside another list; skip the rest of its contents.
147 current = ElementTraversal::nextIncludingPseudoSkippingChildren(*current, &list);
148 }
149
150 return nullptr;
151}
152
153static inline RenderListItem* nextListItem(const Element& list, const RenderListItem& item)
154{
155 return nextListItemHelper(list, item.element());
156}
157
158static inline RenderListItem* firstListItem(const Element& list)
159{
160 return nextListItemHelper(list, list);
161}
162
163static RenderListItem* previousListItem(const Element& list, const RenderListItem& item)
164{
165 auto* current = &item.element();
166 auto advance = [&] {
167 current = ElementTraversal::previousIncludingPseudo(*current, &list);
168 };
169 advance();
170 while (current) {
171 auto* renderer = current->renderer();
172 if (!is<RenderListItem>(renderer)) {
173 advance();
174 continue;
175 }
176 auto& item = downcast<RenderListItem>(*renderer);
177 auto* otherList = enclosingList(item);
178 if (!otherList) {
179 advance();
180 continue;
181 }
182
183 // This item is part of our current list, so we found what we're looking for.
184 if (&list == otherList)
185 return &item;
186
187 // We found ourself inside another list; skip the rest of its contents by
188 // advancing to it. However, since the list itself might be a list item,
189 // don't advance past it.
190 current = otherList;
191 }
192 return nullptr;
193}

209void RenderListItem::updateValueNow() const
210{
211 auto* list = enclosingList(*this);
212 auto* orderedList = is<HTMLOListElement>(list) ? downcast<HTMLOListElement>(list) : nullptr;
213
214 // The start item is either the closest item before this one in the list that already has a value,
215 // or the first item in the list if none have before this have values yet.
216 auto* startItem = this;
217 if (list) {
218 auto* item = this;
219 while ((item = previousListItem(*list, *item))) {
220 startItem = item;
221 if (item->m_value)
222 break;
223 }
224 }
225
226 auto& startValue = startItem->m_value;
227 if (!startValue)
228 startValue = orderedList ? orderedList->start() : 1;
229 int value = *startValue;
230 int increment = (orderedList && orderedList->isReversed()) ? -1 : 1;
231
232 for (auto* item = startItem; item != this; ) {
233 item = nextListItem(*list, *item);
234 item->m_value = (value += increment);
235 }
236}

238void RenderListItem::updateValue()
239{

244 }

293void RenderListItem::explicitValueChanged()
294{
295 if (m_marker)
296 m_marker->setNeedsLayoutAndPrefWidthsRecalc();
297
298 updateValue();
299 auto* list = enclosingList(*this);
300 if (!list)
301 return;
302 auto* item = this;
303 while ((item = nextListItem(*list, *item)))
304 item->updateValue();
305}

307void RenderListItem::setExplicitValue(std::optional<int> value)
308{
309 if (!value) {
310 if (!m_valueWasSetExplicitly)
311 return;
312 } else {
313 if (m_valueWasSetExplicitly && m_value == value)
314 return;
315 }
316 m_valueWasSetExplicitly = value.has_value();
317 m_value = value;
318 explicitValueChanged();
319}

321void RenderListItem::updateListMarkerNumbers()
322{
323 auto* list = enclosingList(*this);
324 if (!list)
325 return;
326
327 bool isInReversedOrderedList = false;
328 if (is<HTMLOListElement>(*list)) {
329 auto& orderedList = downcast<HTMLOListElement>(*list);
330 orderedList.itemCountChanged();
331 isInReversedOrderedList = orderedList.isReversed();
332 }
333
334 // If an item has been marked for update before, we know that all following items have, too.
335 // This gives us the opportunity to stop and avoid marking the same nodes again.
336 auto* item = this;
337 auto subsequentListItem = isInReversedOrderedList ? previousListItem : nextListItem;
338 while ((item = subsequentListItem(*list, *item)) && item->m_value)
339 item->updateValue();
340}

342bool RenderListItem::isInReversedOrderedList() const
343{
344 auto* list = enclosingList(*this);
345 return is<HTMLOListElement>(list) && downcast<HTMLOListElement>(*list).isReversed();
346}
