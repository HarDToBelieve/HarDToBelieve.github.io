58template<SequenceType type, typename CharacterType> static String toAlphabeticOrNumeric(int number, const CharacterType* sequence, unsigned sequenceSize)
59{
60 ASSERT(sequenceSize >= 2);
61
62 // Taking sizeof(number) in the expression below doesn't work with some compilers.
63 constexpr unsigned lettersSize = sizeof(int) * 8 + 1; // Binary is the worst case; requires one character per bit plus a minus sign.
64
65 CharacterType letters[lettersSize];
66
67 bool isNegativeNumber = false;
68 unsigned numberShadow = number;
69 if constexpr (type == SequenceType::Alphabetic) {
70 ASSERT(number > 0);
71 --numberShadow;
72 } else if (number < 0) {
73 numberShadow = -number;
74 isNegativeNumber = true;
75 }
76 letters[lettersSize - 1] = sequence[numberShadow % sequenceSize];
77 unsigned length = 1;
78
79 if constexpr (type == SequenceType::Alphabetic) {
80 while ((numberShadow /= sequenceSize) > 0) {
81 --numberShadow;
82 letters[lettersSize - ++length] = sequence[numberShadow % sequenceSize];
83 }
84 } else {
85 while ((numberShadow /= sequenceSize) > 0)
86 letters[lettersSize - ++length] = sequence[numberShadow % sequenceSize];
87 }
88 if (isNegativeNumber)
89 letters[lettersSize - ++length] = hyphenMinus;
90
91 ASSERT(length <= lettersSize);
92 return { &letters[lettersSize - length], length };
93}

95template<typename CharacterType> static String toSymbolic(int number, const CharacterType* symbols, unsigned symbolsSize)
96{
97 ASSERT(number > 0);
98 ASSERT(symbolsSize >= 1);
99
100 // The asterisks list-style-type is the worst case; we show |number| asterisks.
101 auto symbol = symbols[(number - 1) % symbolsSize];
102 unsigned count = (number - 1) / symbolsSize + 1;
103
104 CharacterType* characters;
105 String result = String::createUninitialized(count, characters);
106 for (unsigned i = 0; i < count; ++i)
107 characters[i] = symbol;
108 return result;
109}

136// This table format was derived from an old draft of the CSS specification: 3 group markers, 3 digit markers, 10 digits, negative sign.
137static String toCJKIdeographic(int number, const UChar table[17], Formality formality)
138{
139 enum AbstractCJKCharacter {
140 NoChar,
141 SecondGroupMarker, ThirdGroupMarker, FourthGroupMarker,
142 SecondDigitMarker, ThirdDigitMarker, FourthDigitMarker,
143 Digit0, Digit1, Digit2, Digit3, Digit4,
144 Digit5, Digit6, Digit7, Digit8, Digit9,
145 NegativeSign
146 };
147
148 if (!number)
149 return { &table[Digit0 - 1] , 1 };
150
151 ASSERT(number != std::numeric_limits<int>::min());
152 bool needsNegativeSign = number < 0;
153 if (needsNegativeSign)
154 number = -number;
155
156 constexpr unsigned groupLength = 8; // 4 digits, 3 digit markers, and a group marker
157 constexpr unsigned bufferLength = 4 * groupLength;
158 AbstractCJKCharacter buffer[bufferLength] = { NoChar };
159
160 for (int i = 0; i < 4; ++i) {
161 int groupValue = number % 10000;
162 number /= 10000;
163
164 // Process least-significant group first, but put it in the buffer last.
165 auto group = &buffer[(3 - i) * groupLength];
166
167 if (groupValue && i)
168 group[7] = static_cast<AbstractCJKCharacter>(SecondGroupMarker - 1 + i);
169
170 // Put in the four digits and digit markers for any non-zero digits.
171 group[6] = static_cast<AbstractCJKCharacter>(Digit0 + (groupValue % 10));
172 if (number != 0 || groupValue > 9) {
173 int digitValue = ((groupValue / 10) % 10);
174 group[4] = static_cast<AbstractCJKCharacter>(Digit0 + digitValue);
175 if (digitValue)
176 group[5] = SecondDigitMarker;
177 }
178 if (number != 0 || groupValue > 99) {
179 int digitValue = ((groupValue / 100) % 10);
180 group[2] = static_cast<AbstractCJKCharacter>(Digit0 + digitValue);
181 if (digitValue)
182 group[3] = ThirdDigitMarker;
183 }
184 if (number != 0 || groupValue > 999) {
185 int digitValue = groupValue / 1000;
186 group[0] = static_cast<AbstractCJKCharacter>(Digit0 + digitValue);
187 if (digitValue)
188 group[1] = FourthDigitMarker;
189 }
190
191 if (formality == Formality::Informal && groupValue < 20) {
192 // Remove the tens digit, but leave the marker.
193 ASSERT(group[4] == NoChar || group[4] == Digit0 || group[4] == Digit1);
194 group[4] = NoChar;
195 }
196
197 if (!number)
198 break;
199 }
200
201 // Convert into characters, omitting consecutive runs of digit0 and trailing digit0.
202 unsigned length = 0;
203 UChar characters[1 + bufferLength];
204 auto last = NoChar;
205 if (needsNegativeSign)
206 characters[length++] = table[NegativeSign - 1];
207 for (unsigned i = 0; i < bufferLength; ++i) {
208 auto character = buffer[i];
209 if (character != NoChar) {
210 if (character != Digit0 || last != Digit0)
211 characters[length++] = table[character - 1];
212 last = character;
213 }
214 }
215 if (last == Digit0)
216 --length;
217
218 return { characters, length };
219}

223 std::initializer_list<UChar> characters;

226 std::initializer_list<AdditiveSymbol> symbols;
227 std::initializer_list<UChar> negative;

230static String toPredefinedAdditiveSystem(int value, const AdditiveSystem& system)
231{
232 if (!value) {
233 ASSERT(!system.symbols.end()[-1].value);
234 auto& zeroCharacters = system.symbols.end()[-1].characters;
235 return { zeroCharacters.begin(), static_cast<unsigned>(zeroCharacters.size()) };
236 }
237
238 StringBuilder result;
239 auto append = [&] (std::initializer_list<UChar> characters) {
240 result.append(StringView { characters.begin(), static_cast<unsigned>(characters.size()) });
241 };
242 if (value < 0) {
243 ASSERT(value != std::numeric_limits<int>::min());
244 append(system.negative);
245 value = -value;
246 }
247 for (auto& symbol : system.symbols) {
248 while (value >= symbol.value) {
249 append(symbol.characters);
250 value -= symbol.value;
251 }
252 if (!value)
253 break;
254 }
255 ASSERT(!value);
256 return result.toString();
257}

259static String toEthiopicNumeric(int value)
260{
261 ASSERT(value >= 1);
262
263 if (value == 1) {
264 UChar ethiopicDigitOne = 0x1369;
265 return { &ethiopicDigitOne, 1 };
266 }
267
268 // Split the number into groups of two digits, starting with the least significant decimal digit.
269 uint8_t groups[5];
270 for (auto& group : groups) {
271 group = value % 100;
272 value /= 100;
273 }
274
275 UChar buffer[std::size(groups) * 3];
276 unsigned length = 0;
277 bool isMostSignificantGroup = true;
278 for (int i = std::size(groups) - 1; i >= 0; --i) {
279 auto value = groups[i];
280 bool isOddIndex = i & 1;
281 // If the group has the value zero, or if the group is the most significant one and has the value 1,
282 // or if the group has an odd index (as given in the previous step) and has the value 1,
283 // then remove the digits (but leave the group, so it still has a separator appended below).
284 if (!(value == 1 && (isMostSignificantGroup || isOddIndex))) {
285 if (auto tens = value / 10)
286 buffer[length++] = 0x1371 + tens;
287 if (auto ones = value % 10)
288 buffer[length++] = 0x1368 + ones;
289 }
290 if (value && isOddIndex)
291 buffer[length++] = 0x137B;
292 if ((value || !isMostSignificantGroup) && !isOddIndex && i)
293 buffer[length++] = 0x137C;
294 if (value)
295 isMostSignificantGroup = false;
296 }
297
298 return { buffer, length };
299}

1578void RenderListMarker::paint(PaintInfo& paintInfo, const LayoutPoint& paintOffset)
1579{
1580 if (paintInfo.phase != PaintPhase::Foreground)
1581 return;
1582
1583 if (style().visibility() != Visibility::Visible)
1584 return;
1585
1586 LayoutPoint boxOrigin(paintOffset + location());
1587 LayoutRect overflowRect(visualOverflowRect());
1588 overflowRect.moveBy(boxOrigin);
1589 if (!paintInfo.rect.intersects(overflowRect))
1590 return;
1591
1592 LayoutRect box(boxOrigin, size());
1593
1594 auto markerRect = relativeMarkerRect();
1595 markerRect.moveBy(boxOrigin);
1596 if (markerRect.isEmpty())
1597 return;
1598
1599 GraphicsContext& context = paintInfo.context();
1600
1601 if (isImage()) {
1602 if (RefPtr<Image> markerImage = m_image->image(this, markerRect.size()))
1603 context.drawImage(*markerImage, markerRect);
1604 if (selectionState() != HighlightState::None) {
1605 LayoutRect selRect = localSelectionRect();
1606 selRect.moveBy(boxOrigin);
1607 context.fillRect(snappedIntRect(selRect), m_listItem->selectionBackgroundColor());
1608 }
1609 return;
1610 }
1611
1612 if (selectionState() != HighlightState::None) {
1613 LayoutRect selRect = localSelectionRect();
1614 selRect.moveBy(boxOrigin);
1615 context.fillRect(snappedIntRect(selRect), m_listItem->selectionBackgroundColor());
1616 }
1617
1618 auto color = style().visitedDependentColorWithColorFilter(CSSPropertyColor);
1619 context.setStrokeColor(color);
1620 context.setStrokeStyle(SolidStroke);
1621 context.setStrokeThickness(1.0f);
1622 context.setFillColor(color);
1623
1624 switch (style().listStyleType()) {
1625 case ListStyleType::Disc:
1626 context.drawEllipse(markerRect);
1627 return;
1628 case ListStyleType::Circle:
1629 context.setFillColor(Color::transparentBlack);
1630 context.drawEllipse(markerRect);
1631 return;
1632 case ListStyleType::Square:
1633 context.drawRect(markerRect);
1634 return;
1635 default:
1636 break;
1637 }
1638 if (m_textWithSuffix.isEmpty())
1639 return;
1640
1641 GraphicsContextStateSaver stateSaver(context, false);
1642 if (!style().isHorizontalWritingMode()) {
1643 markerRect.moveBy(-boxOrigin);
1644 markerRect = markerRect.transposedRect();
1645 markerRect.moveBy(FloatPoint(box.x(), box.y() - logicalHeight()));
1646 stateSaver.save();
1647 context.translate(markerRect.x(), markerRect.maxY());
1648 context.rotate(static_cast<float>(deg2rad(90.)));
1649 context.translate(-markerRect.x(), -markerRect.maxY());
1650 }
1651
1652 FloatPoint textOrigin = FloatPoint(markerRect.x(), markerRect.y() + style().fontMetrics().ascent());
1653 textOrigin = roundPointToDevicePixels(LayoutPoint(textOrigin), document().deviceScaleFactor(), style().isLeftToRightDirection());

1670{
1671 ASSERT(m_listItem);
1672 if (!parent() || !parent()->isBox())
1673 return;
1674
1675 if (isInside() || !inlineBoxWrapper())
1676 return;
1677
1678 LayoutUnit markerOldLogicalLeft = logicalLeft();
1679 LayoutUnit blockOffset;
1680 LayoutUnit lineOffset;
1681 for (auto* ancestor = parentBox(*this); ancestor && ancestor != m_listItem.get(); ancestor = parentBox(*ancestor)) {
1682 blockOffset += ancestor->logicalTop();
1683 lineOffset += ancestor->logicalLeft();
1684 }
1685
1686 bool adjustOverflow = false;
1687 LayoutUnit markerLogicalLeft;
1688 bool hitSelfPaintingLayer = false;
1689
1690 const LegacyRootInlineBox& rootBox = inlineBoxWrapper()->root();
1691 LayoutUnit lineTop = rootBox.lineTop();
1692 LayoutUnit lineBottom = rootBox.lineBottom();
1693
1694 // FIXME: Need to account for relative positioning in the layout overflow.
1695 if (m_listItem->style().isLeftToRightDirection()) {
1696 markerLogicalLeft = m_lineOffsetForListItem - lineOffset - m_listItem->paddingStart() - m_listItem->borderStart() + marginStart();
1697 inlineBoxWrapper()->adjustLineDirectionPosition(markerLogicalLeft - markerOldLogicalLeft);
1698 for (auto* box = inlineBoxWrapper()->parent(); box; box = box->parent()) {
1699 auto newLogicalVisualOverflowRect = box->logicalVisualOverflowRect(lineTop, lineBottom);
1700 auto newLogicalLayoutOverflowRect = box->logicalLayoutOverflowRect(lineTop, lineBottom);
1701 if (markerLogicalLeft < newLogicalVisualOverflowRect.x() && !hitSelfPaintingLayer) {
1702 newLogicalVisualOverflowRect.setWidth(newLogicalVisualOverflowRect.maxX() - markerLogicalLeft);
1703 newLogicalVisualOverflowRect.setX(markerLogicalLeft);
1704 if (box == &rootBox)
1705 adjustOverflow = true;
1706 }
1707 if (markerLogicalLeft < newLogicalLayoutOverflowRect.x()) {
1708 newLogicalLayoutOverflowRect.setWidth(newLogicalLayoutOverflowRect.maxX() - markerLogicalLeft);
1709 newLogicalLayoutOverflowRect.setX(markerLogicalLeft);
1710 if (box == &rootBox)
1711 adjustOverflow = true;
1712 }
1713 box->setOverflowFromLogicalRects(newLogicalLayoutOverflowRect, newLogicalVisualOverflowRect, lineTop, lineBottom);
1714 if (box->renderer().hasSelfPaintingLayer())
1715 hitSelfPaintingLayer = true;
1716 }
1717 } else {
1718 markerLogicalLeft = m_lineOffsetForListItem - lineOffset + m_listItem->paddingStart() + m_listItem->borderStart() + marginEnd();
1719 inlineBoxWrapper()->adjustLineDirectionPosition(markerLogicalLeft - markerOldLogicalLeft);
1720 for (auto* box = inlineBoxWrapper()->parent(); box; box = box->parent()) {
1721 auto newLogicalVisualOverflowRect = box->logicalVisualOverflowRect(lineTop, lineBottom);
1722 auto newLogicalLayoutOverflowRect = box->logicalLayoutOverflowRect(lineTop, lineBottom);
1723 if (markerLogicalLeft + logicalWidth() > newLogicalVisualOverflowRect.maxX() && !hitSelfPaintingLayer) {
1724 newLogicalVisualOverflowRect.setWidth(markerLogicalLeft + logicalWidth() - newLogicalVisualOverflowRect.x());
1725 if (box == &rootBox)
1726 adjustOverflow = true;
1727 }
1728 if (markerLogicalLeft + logicalWidth() > newLogicalLayoutOverflowRect.maxX()) {
1729 newLogicalLayoutOverflowRect.setWidth(markerLogicalLeft + logicalWidth() - newLogicalLayoutOverflowRect.x());
1730 if (box == &rootBox)
1731 adjustOverflow = true;
1732 }
1733 box->setOverflowFromLogicalRects(newLogicalLayoutOverflowRect, newLogicalVisualOverflowRect, lineTop, lineBottom);
1734 if (box->renderer().hasSelfPaintingLayer())
1735 hitSelfPaintingLayer = true;
1736 }
1737 }
1738
1739 if (adjustOverflow) {
1740 LayoutRect markerRect(markerLogicalLeft + lineOffset, blockOffset, width(), height());
1741 if (!m_listItem->style().isHorizontalWritingMode())
1742 markerRect = markerRect.transposedRect();
1743 RenderBox* markerAncestor = this;
1744 bool propagateVisualOverflow = true;
1745 bool propagateLayoutOverflow = true;
1746 do {
1747 markerAncestor = parentBox(*markerAncestor);
1748 if (markerAncestor->hasOverflowClip())
1749 propagateVisualOverflow = false;
1750 if (is<RenderBlock>(*markerAncestor)) {
1751 if (propagateVisualOverflow)
1752 downcast<RenderBlock>(*markerAncestor).addVisualOverflow(markerRect);
1753 if (propagateLayoutOverflow)
1754 downcast<RenderBlock>(*markerAncestor).addLayoutOverflow(markerRect);
1755 }
1756 if (markerAncestor->hasOverflowClip())
1757 propagateLayoutOverflow = false;
1758 if (markerAncestor->hasSelfPaintingLayer())
1759 propagateVisualOverflow = false;
1760 markerRect.moveBy(-markerAncestor->location());
1761 } while (markerAncestor != m_listItem.get() && propagateVisualOverflow && propagateLayoutOverflow);
1762 }
