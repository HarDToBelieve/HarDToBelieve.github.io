52class RenderLayer;

54// This class handles the auto-scrolling for <marquee>
55class RenderMarquee final {
56 WTF_MAKE_NONCOPYABLE(RenderMarquee); WTF_MAKE_FAST_ALLOCATED;
57public:
58 explicit RenderMarquee(RenderLayer*);
59 ~RenderMarquee();
60
61 bool isHorizontal() const;
62
63 void start();
64 void suspend();
65 void stop();
66
67 void updateMarqueeStyle();
68 void updateMarqueePosition();
69
70private:
71
72 int speed() const { return m_speed; }
73 int marqueeSpeed() const;
74
75 MarqueeDirection direction() const;
76
77 int computePosition(MarqueeDirection, bool stopAtClientEdge);
78
79 void setEnd(int end) { m_end = end; }
80
81 void timerFired();
82
83 RenderLayer* m_layer;
84 Timer m_timer;
85 int m_currentLoop { 0 };
86 int m_totalLoops { 0 };
87 int m_start { 0 };
88 int m_end { 0 };
89 int m_speed { 0 };
90 Length m_height;
91 MarqueeDirection m_direction { MarqueeDirection::Auto };
92 bool m_reset { false };
93 bool m_suspended { false };
94 bool m_stopped { false };
95};
