38WTF_MAKE_ISO_ALLOCATED_IMPL(RenderMedia);

40RenderMedia::RenderMedia(HTMLMediaElement& element, RenderStyle&& style)
41 : RenderImage(element, WTFMove(style))
42{
43 setHasShadowControls(true);
44}
45
46RenderMedia::RenderMedia(HTMLMediaElement& element, RenderStyle&& style, const IntSize& intrinsicSize)
47 : RenderImage(element, WTFMove(style))
48{
49 setIntrinsicSize(intrinsicSize);
50 setHasShadowControls(true);
51}
52
53RenderMedia::~RenderMedia() = default;
54
55void RenderMedia::paintReplaced(PaintInfo&, const LayoutPoint&)
56{
57}
58
59void RenderMedia::layout()
60{
61 LayoutSize oldSize = size();
62 RenderImage::layout();
63 if (oldSize != size())
64 mediaElement().layoutSizeChanged();
65}
66
67void RenderMedia::styleDidChange(StyleDifference difference, const RenderStyle* oldStyle)
68{
69 RenderImage::styleDidChange(difference, oldStyle);
70 if (!oldStyle || style().visibility() != oldStyle->visibility())
71 mediaElement().visibilityDidChange();
72}
