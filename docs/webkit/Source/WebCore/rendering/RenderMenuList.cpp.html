64static size_t selectedOptionCount(const RenderMenuList& renderMenuList)
65{
66 const Vector<HTMLElement*>& listItems = renderMenuList.selectElement().listItems();
67 size_t numberOfItems = listItems.size();
68
69 size_t count = 0;
70 for (size_t i = 0; i < numberOfItems; ++i) {
71 if (is<HTMLOptionElement>(*listItems[i]) && downcast<HTMLOptionElement>(*listItems[i]).selected())
72 ++count;
73 }
74 return count;
75}

111void RenderMenuList::adjustInnerStyle()
112{
113 auto& innerStyle = m_innerBlock->mutableStyle();
114 innerStyle.setFlexGrow(1);
115 innerStyle.setFlexShrink(1);
116 // min-width: 0; is needed for correct shrinking.
117 innerStyle.setMinWidth(Length(0, LengthType::Fixed));
118 // Use margin:auto instead of align-items:center to get safe centering, i.e.
119 // when the content overflows, treat it the same as align-items: flex-start.
120 // But we only do that for the cases where html.css would otherwise use center.
121 if (style().alignItems().position() == ItemPosition::Center) {
122 innerStyle.setMarginTop(Length());
123 innerStyle.setMarginBottom(Length());
124 innerStyle.setAlignSelfPosition(ItemPosition::FlexStart);
125 }
126
127 innerStyle.setPaddingBox(theme().popupInternalPaddingBox(style(), document().settings()));
128
129 if (document().page()->chrome().selectItemWritingDirectionIsNatural()) {
130 // Items in the popup will not respect the CSS text-align and direction properties,
131 // so we must adjust our own style to match.
132 innerStyle.setTextAlign(TextAlignMode::Left);
133 TextDirection direction = (m_buttonText && m_buttonText->text().defaultWritingDirection() == U_RIGHT_TO_LEFT) ? TextDirection::RTL : TextDirection::LTR;
134 innerStyle.setDirection(direction);
135#if PLATFORM(IOS_FAMILY)
136 } else if (document().page()->chrome().selectItemAlignmentFollowsMenuWritingDirection()) {
137 innerStyle.setTextAlign(style().direction() == TextDirection::LTR ? TextAlignMode::Left : TextAlignMode::Right);
138 TextDirection direction;
139 EUnicodeBidi unicodeBidi;
140 if (multiple() && selectedOptionCount(*this) != 1) {
141 direction = (m_buttonText && m_buttonText->text().defaultWritingDirection() == U_RIGHT_TO_LEFT) ? TextDirection::RTL : TextDirection::LTR;
142 unicodeBidi = UBNormal;
143 } else if (m_optionStyle) {
144 direction = m_optionStyle->direction();
145 unicodeBidi = m_optionStyle->unicodeBidi();
146 } else {
147 direction = style().direction();
148 unicodeBidi = style().unicodeBidi();
149 }
150
151 innerStyle.setDirection(direction);
152 innerStyle.setUnicodeBidi(unicodeBidi);
153 }
154#else
155 } else if (m_optionStyle && document().page()->chrome().selectItemAlignmentFollowsMenuWritingDirection()) {
156 if ((m_optionStyle->direction() != innerStyle.direction() || m_optionStyle->unicodeBidi() != innerStyle.unicodeBidi()))
157 m_innerBlock->setNeedsLayoutAndPrefWidthsRecalc();
158 innerStyle.setTextAlign(style().isLeftToRightDirection() ? TextAlignMode::Left : TextAlignMode::Right);
159 innerStyle.setDirection(m_optionStyle->direction());
160 innerStyle.setUnicodeBidi(m_optionStyle->unicodeBidi());
161 }
162#endif // !PLATFORM(IOS_FAMILY)
163}

190void RenderMenuList::updateOptionsWidth()
191{
192 float maxOptionWidth = 0;
193 const Vector<HTMLElement*>& listItems = selectElement().listItems();
194 int size = listItems.size();
195
196 for (int i = 0; i < size; ++i) {
197 HTMLElement* element = listItems[i];
198 if (!is<HTMLOptionElement>(*element))
199 continue;
200
201 String text = downcast<HTMLOptionElement>(*element).textIndentedToRespectGroupLabel();
202 text = applyTextTransform(style(), text, ' ');
203 if (theme().popupOptionSupportsTextIndent()) {
204 // Add in the option's text indent. We can't calculate percentage values for now.
205 float optionWidth = 0;
206 if (auto* optionStyle = element->computedStyle())
207 optionWidth += minimumValueForLength(optionStyle->textIndent(), 0);
208 if (!text.isEmpty()) {
209 const FontCascade& font = style().fontCascade();
210 TextRun run = RenderBlock::constructTextRun(text, style());
211 optionWidth += font.width(run);
212 }
213 maxOptionWidth = std::max(maxOptionWidth, optionWidth);
214 } else if (!text.isEmpty()) {
215 const FontCascade& font = style().fontCascade();
216 TextRun run = RenderBlock::constructTextRun(text, style());
217 maxOptionWidth = std::max(maxOptionWidth, font.width(run));
218 }
219 }
220
221 int width = static_cast<int>(ceilf(maxOptionWidth));
222 if (m_optionsWidth == width)
223 return;
224
225 m_optionsWidth = width;
226 if (parent())
227 setNeedsLayoutAndPrefWidthsRecalc();
228}

245void RenderMenuList::setTextFromOption(int optionIndex)
246{
247 const Vector<HTMLElement*>& listItems = selectElement().listItems();
248 int size = listItems.size();
249
250 int i = selectElement().optionToListIndex(optionIndex);
251 String text = emptyString();
252 if (i >= 0 && i < size) {
253 Element* element = listItems[i];
254 if (is<HTMLOptionElement>(*element)) {
255 text = downcast<HTMLOptionElement>(*element).textIndentedToRespectGroupLabel();
256 auto* style = element->computedStyle();
257 m_optionStyle = style ? RenderStyle::clonePtr(*style) : nullptr;
258 }
259 }
260
261#if PLATFORM(IOS_FAMILY)
262 if (multiple()) {
263 size_t count = selectedOptionCount(*this);
264 if (count != 1)
265 text = htmlSelectMultipleItems(count);
266 }
267#endif
268
269 setText(text.stripWhiteSpace());
270 didUpdateActiveOption(optionIndex);
271}

273void RenderMenuList::setText(const String& s)
274{
275 String textToUse = s.isEmpty() ? "\n"_str : s;
276
277 if (m_buttonText) {
278 m_buttonText->setText(textToUse.impl(), true);
279 m_buttonText->dirtyLineBoxes(false);
280 } else {
281 auto newButtonText = createRenderer<RenderText>(document(), textToUse);
282 m_buttonText = makeWeakPtr(*newButtonText);
283 // FIXME: This mutation should go through the normal RenderTreeBuilder path.
284 if (RenderTreeBuilder::current())
285 RenderTreeBuilder::current()->attach(*this, WTFMove(newButtonText));
286 else
287 RenderTreeBuilder(*document().renderView()).attach(*this, WTFMove(newButtonText));
288 }
289
290 adjustInnerStyle();
291}

298LayoutRect RenderMenuList::controlClipRect(const LayoutPoint& additionalOffset) const
299{
300 // Clip to the intersection of the content box and the content box for the inner box
301 // This will leave room for the arrows which sit in the inner box padding,
302 // and if the inner box ever spills out of the outer box, that will get clipped too.
303 LayoutRect outerBox(additionalOffset.x() + borderLeft() + paddingLeft(),
304 additionalOffset.y() + borderTop() + paddingTop(),
305 contentWidth(),
306 contentHeight());
307
308 LayoutRect innerBox(additionalOffset.x() + m_innerBlock->x() + m_innerBlock->paddingLeft(),
309 additionalOffset.y() + m_innerBlock->y() + m_innerBlock->paddingTop(),
310 m_innerBlock->contentWidth(),
311 m_innerBlock->contentHeight());
312
313 return intersection(outerBox, innerBox);
314}

316void RenderMenuList::computeIntrinsicLogicalWidths(LayoutUnit& minLogicalWidth, LayoutUnit& maxLogicalWidth) const
317{
318 maxLogicalWidth = shouldApplySizeContainment(*this) ? theme().minimumMenuListSize(style()) : std::max(m_optionsWidth, theme().minimumMenuListSize(style()));
319 maxLogicalWidth += m_innerBlock->paddingLeft() + m_innerBlock->paddingRight();
320 if (!style().width().isPercentOrCalculated())
321 minLogicalWidth = maxLogicalWidth;
322}

324void RenderMenuList::computePreferredLogicalWidths()
325{
326 m_minPreferredLogicalWidth = 0;
327 m_maxPreferredLogicalWidth = 0;
328
329 if (style().width().isFixed() && style().width().value() > 0)
330 m_minPreferredLogicalWidth = m_maxPreferredLogicalWidth = adjustContentBoxLogicalWidthForBoxSizing(style().width());
331 else
332 computeIntrinsicLogicalWidths(m_minPreferredLogicalWidth, m_maxPreferredLogicalWidth);
333
334 RenderBox::computePreferredLogicalWidths(style().minWidth(), style().maxWidth(), horizontalBorderAndPaddingExtent());
335
336 setPreferredLogicalWidthsDirty(false);
337}

398void RenderMenuList::didUpdateActiveOption(int optionIndex)
399{
400 if (!AXObjectCache::accessibilityEnabled())
401 return;
402
403 auto* axCache = document().existingAXObjectCache();
404 if (!axCache)
405 return;
406
407 if (m_lastActiveIndex == optionIndex)
408 return;
409 m_lastActiveIndex = optionIndex;
410
411 int listIndex = selectElement().optionToListIndex(optionIndex);
412 if (listIndex < 0 || listIndex >= static_cast<int>(selectElement().listItems().size()))
413 return;
414
415 auto* axObject = axCache->get(this);
416 if (is<AccessibilityMenuList>(axObject))
417 downcast<AccessibilityMenuList>(*axObject).didUpdateActiveOption(optionIndex);

463bool RenderMenuList::itemIsEnabled(unsigned listIndex) const
464{
465 const Vector<HTMLElement*>& listItems = selectElement().listItems();
466 if (listIndex >= listItems.size())
467 return false;
468 HTMLElement* element = listItems[listIndex];
469 if (!is<HTMLOptionElement>(*element))
470 return false;
471
472 bool groupEnabled = true;
473 if (Element* parentElement = element->parentElement()) {
474 if (is<HTMLOptGroupElement>(*parentElement))
475 groupEnabled = !parentElement->isDisabledFormControl();
476 }
477 if (!groupEnabled)
478 return false;
479
480 return !element->isDisabledFormControl();
481}

483PopupMenuStyle RenderMenuList::itemStyle(unsigned listIndex) const
484{
485 const Vector<HTMLElement*>& listItems = selectElement().listItems();
486 if (listIndex >= listItems.size()) {
487 // If we are making an out of bounds access, then we want to use the style
488 // of a different option element (index 0). However, if there isn't an option element
489 // before at index 0, we fall back to the menu's style.
490 if (!listIndex)
491 return menuStyle();
492
493 // Try to retrieve the style of an option element we know exists (index 0).
494 listIndex = 0;
495 }
496 HTMLElement* element = listItems[listIndex];
497
498 Color itemBackgroundColor;
499 bool itemHasCustomBackgroundColor;
500 getItemBackgroundColor(listIndex, itemBackgroundColor, itemHasCustomBackgroundColor);
501
502 auto& style = *element->computedStyle();
503 return PopupMenuStyle(style.visitedDependentColorWithColorFilter(CSSPropertyColor), itemBackgroundColor, style.fontCascade(), style.visibility() == Visibility::Visible,
504 style.display() == DisplayType::None, true, style.textIndent(), style.direction(), isOverride(style.unicodeBidi()),
505 itemHasCustomBackgroundColor ? PopupMenuStyle::CustomBackgroundColor : PopupMenuStyle::DefaultBackgroundColor);
506}

508void RenderMenuList::getItemBackgroundColor(unsigned listIndex, Color& itemBackgroundColor, bool& itemHasCustomBackgroundColor) const
509{
510 const Vector<HTMLElement*>& listItems = selectElement().listItems();
511 if (listIndex >= listItems.size()) {
512 itemBackgroundColor = style().visitedDependentColorWithColorFilter(CSSPropertyBackgroundColor);
513 itemHasCustomBackgroundColor = false;
514 return;
515 }
516 HTMLElement* element = listItems[listIndex];
517
518 Color backgroundColor = element->computedStyle()->visitedDependentColorWithColorFilter(CSSPropertyBackgroundColor);
519 itemHasCustomBackgroundColor = backgroundColor.isValid() && backgroundColor.isVisible();
520 // If the item has an opaque background color, return that.
521 if (backgroundColor.isOpaque()) {
522 itemBackgroundColor = backgroundColor;
523 return;
524 }
525
526 // Otherwise, the item's background is overlayed on top of the menu background.
527 backgroundColor = blendSourceOver(style().visitedDependentColorWithColorFilter(CSSPropertyBackgroundColor), backgroundColor);
528 if (backgroundColor.isOpaque()) {
529 itemBackgroundColor = backgroundColor;
530 return;
531 }
532
533 // If the menu background is not opaque, then add an opaque white background behind.
534 itemBackgroundColor = blendSourceOver(Color::white, backgroundColor);
535}

571const int endOfLinePadding = 2;
572
573LayoutUnit RenderMenuList::clientPaddingLeft() const
574{
575 if ((style().appearance() == MenulistPart || style().appearance() == MenulistButtonPart) && style().direction() == TextDirection::RTL) {
576 // For these appearance values, the theme applies padding to leave room for the
577 // drop-down button. But leaving room for the button inside the popup menu itself
578 // looks strange, so we return a small default padding to avoid having a large empty
579 // space appear on the side of the popup menu.
580 return endOfLinePadding;
581 }
582 // If the appearance isn't MenulistPart, then the select is styled (non-native), so
583 // we want to return the user specified padding.
584 return paddingLeft() + m_innerBlock->paddingLeft();
585}

587LayoutUnit RenderMenuList::clientPaddingRight() const
588{
589 if ((style().appearance() == MenulistPart || style().appearance() == MenulistButtonPart) && style().direction() == TextDirection::LTR)
590 return endOfLinePadding;
591
592 return paddingRight() + m_innerBlock->paddingRight();
593}

644#if PLATFORM(IOS_FAMILY)
645void RenderMenuList::layout()
646{
647 RenderFlexibleBox::layout();
648
649 // Ideally, we should not be adjusting styles during layout. However, for a
650 // pill-shaped appearance, the horizontal border radius is dependent on the
651 // computed height of the box. This means that the appearance cannot be declared
652 // prior to layout, since CSS only allows the horizontal border radius to be
653 // dependent on the computed width of the box.
654 //
655 // Ignoring the style's border radius and forcing a pill-shaped appearance at
656 // paint time is not an option, since focus rings and tap highlights will not
657 // use the correct border radius. Consequently, we need to adjust the border
658 // radius here.
659 //
660 // Note that similar adjustments are made in RenderSliderThumb, RenderButton
661 // and RenderTextControlSingleLine.
662 RenderThemeIOS::adjustRoundBorderRadius(mutableStyle(), *this);
663}
664#endif
