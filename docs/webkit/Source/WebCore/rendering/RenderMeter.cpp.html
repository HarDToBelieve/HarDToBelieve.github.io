42HTMLMeterElement* RenderMeter::meterElement() const
43{
44 ASSERT(element());
45
46 if (is<HTMLMeterElement>(*element()))
47 return downcast<HTMLMeterElement>(element());
48
49 ASSERT(element()->shadowHost());
50 return downcast<HTMLMeterElement>(element()->shadowHost());
51}

53void RenderMeter::updateLogicalWidth()
54{
55 RenderBox::updateLogicalWidth();
56
57 IntSize frameSize = theme().meterSizeForBounds(*this, snappedIntRect(frameRect()));
58 setLogicalWidth(isHorizontalWritingMode() ? frameSize.width() : frameSize.height());
59}
60
61RenderBox::LogicalExtentComputedValues RenderMeter::computeLogicalHeight(LayoutUnit logicalHeight, LayoutUnit logicalTop) const
62{
63 auto computedValues = RenderBox::computeLogicalHeight(logicalHeight, logicalTop);
64 LayoutRect frame = frameRect();
65 if (isHorizontalWritingMode())
66 frame.setHeight(computedValues.m_extent);
67 else
68 frame.setWidth(computedValues.m_extent);
69 IntSize frameSize = theme().meterSizeForBounds(*this, snappedIntRect(frame));
70 computedValues.m_extent = isHorizontalWritingMode() ? frameSize.height() : frameSize.width();
71 return computedValues;
72}
