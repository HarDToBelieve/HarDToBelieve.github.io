26#pragma once
27
28#include <wtf/TypeCasts.h>
29
30namespace WebCore {
31
32class RenderObject;
33
34struct RenderObjectDeleter
35{
36 void operator() (RenderObject*) const;
37};
38
39template <typename T> using RenderPtr = std::unique_ptr<T, RenderObjectDeleter>;
40
41template<typename T, class... Args> inline RenderPtr<T> createRenderer(Args&&... args)
42{
43 return RenderPtr<T>(new T(std::forward<Args>(args)...));
44}
45
46template<typename T, typename U> inline RenderPtr<T> static_pointer_cast(RenderPtr<U>&& p)
47{
48 return RenderPtr<T>(downcast<T>(p.release()));
49}
50
51} // namespace WebCore
