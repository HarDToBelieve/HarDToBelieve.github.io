75static void checkNumberOfDistinctQuoteCharacters(UChar character)
76{
77 ASSERT(character);
78 static UChar distinctQuoteCharacters[maxDistinctQuoteCharacters];
79 for (unsigned i = 0; i < maxDistinctQuoteCharacters; ++i) {
80 if (distinctQuoteCharacters[i] == character)
81 return;
82 if (!distinctQuoteCharacters[i]) {
83 distinctQuoteCharacters[i] = character;
84 return;
85 }
86 }
87 ASSERT_NOT_REACHED();
88}

92struct SubtagComparison {
93 size_t keyLength;
94 size_t keyContinue;
95 size_t rangeLength;
96 size_t rangeContinue;
97 int comparison;
98};
99
100static SubtagComparison subtagCompare(const char* key, const char* range)
101{
102 SubtagComparison result;
103
104 result.keyLength = strlen(key);
105 result.keyContinue = result.keyLength;
106 if (auto* hyphenPointer = strchr(key, '-')) {
107 result.keyLength = hyphenPointer - key;
108 result.keyContinue = result.keyLength + 1;
109 }
110
111 result.rangeLength = strlen(range);
112 result.rangeContinue = result.rangeLength;
113 if (auto* hyphenPointer = strchr(range, '-')) {
114 result.rangeLength = hyphenPointer - range;
115 result.rangeContinue = result.rangeLength + 1;
116 }
117
118 if (result.keyLength == result.rangeLength)
119 result.comparison = memcmp(key, range, result.keyLength);
120 else
121 result.comparison = strcmp(key, range);
122
123 return result;
124}
125
126// These strings need to be compared according to "Extended Filtering", as in Section 3.3.2 in RFC4647.
127// https://tools.ietf.org/html/rfc4647#page-10
128//
129// The "checkFurther" field is needed in one specific situation.
130// In the quoteTable below, there are lines like:
131// { "de" , 0x201e, 0x201c, 0x201a, 0x2018 },
132// { "de-ch", 0x00ab, 0x00bb, 0x2039, 0x203a },
133// Let's say the binary search arbitrarily decided to test our key against the upper line "de" first.
134// If the key we're testing against is "de-ch", then we should report "greater than",
135// so the binary search will keep searching and eventually find the "de-ch" line.
136// However, if the key we're testing against is "de-de", then we should report "equal to",
137// because these are the quotes we should use for all "de" except for "de-ch".
138struct QuotesForLanguage {
139 const char* language;
140 UChar checkFurther;
141 UChar open1;
142 UChar close1;
143 UChar open2;
144 UChar close2;
145};
146
147static int quoteTableLanguageComparisonFunction(const void* a, const void* b)
148{
149 // These strings need to be compared according to "Extended Filtering", as in Section 3.3.2 in RFC4647.
150 // https://tools.ietf.org/html/rfc4647#page-10
151 //
152 // We can exploit a few things here to improve perf:
153 // 1. The first subtag must be matched exactly
154 // 2. All the ranges have either 1 or 2 subtags
155 // 3. None of the subtags in any of the ranges are wildcards
156 //
157 // Also, see the comment just above the QuotesForLanguage struct.
158
159 auto* key = static_cast<const QuotesForLanguage*>(a);
160 auto* range = static_cast<const QuotesForLanguage*>(b);
161
162 auto firstSubtagComparison = subtagCompare(key->language, range->language);
163
164 if (firstSubtagComparison.keyLength != firstSubtagComparison.rangeLength)
165 return firstSubtagComparison.comparison;
166
167 if (firstSubtagComparison.comparison)
168 return firstSubtagComparison.comparison;
169
170 for (UChar i = 1; i <= range->checkFurther; ++i) {
171 if (!quoteTableLanguageComparisonFunction(key, range + i)) {
172 // Tell the binary search to check later in the array of ranges, to eventually find the match we just found here.
173 return 1;
174 }
175 }
176
177 for (size_t keyOffset = firstSubtagComparison.keyContinue; ;) {
178 auto nextSubtagComparison = subtagCompare(key->language + keyOffset, range->language + firstSubtagComparison.rangeContinue);
179
180 if (!nextSubtagComparison.rangeLength) {
181 // E.g. The key is "zh-Hans" and the range is "zh".
182 return 0;
183 }
184
185 if (!nextSubtagComparison.keyLength) {
186 // E.g. the key is "zh" and the range is "zh-Hant".
187 return nextSubtagComparison.comparison;
188 }
189
190 if (nextSubtagComparison.keyLength == 1) {
191 // E.g. the key is "zh-x-Hant" and the range is "zh-Hant".
192 // We want to try to find the range "zh", so tell the binary search to check earlier in the array of ranges.
193 return -1;
194 }
195
196 if (nextSubtagComparison.keyLength == nextSubtagComparison.rangeLength && !nextSubtagComparison.comparison) {
197 // E.g. the key is "de-Latn-ch" and the range is "de-ch".
198 return 0;
199 }
200
201 keyOffset += nextSubtagComparison.keyContinue;
202 }
203}

387 return nullptr;
388

390 for (unsigned i = 0; i < length; ++i) {
391 UChar character = toASCIILower(language[i]);
392 if (!(isASCIILower(character) || character == '-'))
393 return nullptr;
394 languageKeyBuffer[i] = static_cast<char>(character);
395 }
396 languageKeyBuffer[length] = 0;
397
398 QuotesForLanguage languageKey = { languageKeyBuffer.data(), 0, 0, 0, 0, 0 };
399
400 return static_cast<const QuotesForLanguage*>(bsearch(&languageKey,
401 quoteTable, WTF_ARRAY_LENGTH(quoteTable), sizeof(quoteTable[0]), quoteTableLanguageComparisonFunction));
402}

404static StringImpl* stringForQuoteCharacter(UChar character)
405{
406 // Use linear search because there is a small number of distinct characters, thus binary search is unneeded.
407 ASSERT(character);
408 struct StringForCharacter {
409 UChar character;
410 StringImpl* string;
411 };
412 static StringForCharacter strings[maxDistinctQuoteCharacters];
413 for (unsigned i = 0; i < maxDistinctQuoteCharacters; ++i) {
414 if (strings[i].character == character)
415 return strings[i].string;
416 if (!strings[i].character) {
417 strings[i].character = character;
418 strings[i].string = &StringImpl::create8BitIfPossible(&character, 1).leakRef();
419 return strings[i].string;
420 }
421 }
422 ASSERT_NOT_REACHED();
423 return StringImpl::empty();
424}

438static RenderTextFragment* quoteTextRenderer(RenderObject* lastChild)
439{
440 if (!lastChild)
441 return nullptr;
442
443 if (!is<RenderTextFragment>(lastChild))
444 return nullptr;
445
446 return downcast<RenderTextFragment>(lastChild);
447}

449void RenderQuote::updateTextRenderer(RenderTreeBuilder& builder)
450{
451 ASSERT_WITH_SECURITY_IMPLICATION(document().inRenderTreeUpdate());
452 String text = computeText();
453 if (m_text == text)
454 return;
455 m_text = text;
456 if (auto* renderText = quoteTextRenderer(lastChild())) {
457 renderText->setContentString(m_text);
458 renderText->dirtyLineBoxes(false);
459 return;
460 }
461 builder.attach(*this, createRenderer<RenderTextFragment>(document(), m_text));
462}

464String RenderQuote::computeText() const
465{
466 if (m_depth < 0)
467 return emptyString();
468 bool isOpenQuote = false;
469 switch (m_type) {
470 case QuoteType::NoOpenQuote:
471 case QuoteType::NoCloseQuote:
472 return emptyString();
473 case QuoteType::OpenQuote:
474 isOpenQuote = true;
475 FALLTHROUGH;
476 case QuoteType::CloseQuote:
477 if (const auto* quotes = style().quotes())
478 return isOpenQuote ? quotes->openQuote(m_depth).impl() : quotes->closeQuote(m_depth).impl();
479 if (const auto* quotes = quotesForLanguage(style().computedLocale()))
480 return stringForQuoteCharacter(isOpenQuote ? (m_depth ? quotes->open2 : quotes->open1) : (m_depth ? quotes->close2 : quotes->close1));
481 // FIXME: Should the default be the quotes for "en" rather than straight quotes?
482 // (According to https://html.spec.whatwg.org/multipage/rendering.html#quotes, the answer is "yes".)
483 return m_depth ? apostropheString() : quotationMarkString();
484 }
485 ASSERT_NOT_REACHED();
486 return emptyString();
487}

489bool RenderQuote::isOpen() const
490{
491 switch (m_type) {
492 case QuoteType::OpenQuote:
493 case QuoteType::NoOpenQuote:
494 return true;
495 case QuoteType::CloseQuote:
496 case QuoteType::NoCloseQuote:
497 return false;
498 }
499 ASSERT_NOT_REACHED();
500 return false;
501}

503void RenderQuote::updateRenderer(RenderTreeBuilder& builder, RenderQuote* previousQuote)
504{
505 ASSERT_WITH_SECURITY_IMPLICATION(document().inRenderTreeUpdate());
506 int depth = -1;
507 if (previousQuote) {
508 depth = previousQuote->m_depth;
509 if (previousQuote->isOpen())
510 ++depth;
511 }
512
513 if (!isOpen())
514 --depth;
515 else if (depth < 0)
516 depth = 0;
517
518 if (m_depth == depth && !m_needsTextUpdate)
519 return;
520
521 m_depth = depth;
522 m_needsTextUpdate = false;
523 updateTextRenderer(builder);
524}
