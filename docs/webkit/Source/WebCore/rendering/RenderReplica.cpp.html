40RenderReplica::RenderReplica(Document& document, RenderStyle&& style)
41 : RenderBox(document, WTFMove(style), 0)
42{
43 // This is a hack. Replicas are synthetic, and don't pick up the attributes of the
44 // renderers being replicated, so they always report that they are inline, non-replaced.
45 // However, we need transforms to be applied to replicas for reflections, so have to pass
46 // the if (!isInline() || isReplaced()) check before setHasTransform().
47 setReplaced(true);
48}

67void RenderReplica::paint(PaintInfo& paintInfo, const LayoutPoint& paintOffset)
68{
69 if (paintInfo.phase != PaintPhase::Foreground && paintInfo.phase != PaintPhase::Mask)
70 return;
71
72 LayoutPoint adjustedPaintOffset = paintOffset + location();
73
74 if (paintInfo.phase == PaintPhase::Foreground) {
75 // Turn around and paint the parent layer. Use temporary clipRects, so that the layer doesn't end up caching clip rects
76 // computing using the wrong rootLayer
77 RenderLayer* rootPaintingLayer = layer()->transform() ? layer()->parent() : layer()->enclosingTransformedAncestor();
78 RenderLayer::LayerPaintingInfo paintingInfo(rootPaintingLayer, paintInfo.rect, PaintBehavior::Normal, LayoutSize(), 0);
79 OptionSet<RenderLayer::PaintLayerFlag> flags { RenderLayer::PaintLayerFlag::HaveTransparency, RenderLayer::PaintLayerFlag::AppliedTransform, RenderLayer::PaintLayerFlag::TemporaryClipRects, RenderLayer::PaintLayerFlag::PaintingReflection };
80 layer()->parent()->paintLayer(paintInfo.context(), paintingInfo, flags);
81 } else if (paintInfo.phase == PaintPhase::Mask)
82 paintMask(paintInfo, adjustedPaintOffset);
83}
