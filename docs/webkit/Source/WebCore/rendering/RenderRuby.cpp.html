48//=== ruby as inline object ===
49
50RenderRubyAsInline::RenderRubyAsInline(Element& element, RenderStyle&& style)
51 : RenderInline(element, WTFMove(style))
52{
53}
54
55RenderRubyAsInline::~RenderRubyAsInline() = default;
56
57void RenderRubyAsInline::styleDidChange(StyleDifference diff, const RenderStyle* oldStyle)
58{
59 RenderInline::styleDidChange(diff, oldStyle);
60 propagateStyleToAnonymousChildren(PropagateToAllChildren);
61}

63//=== ruby as block object ===
64
65RenderRubyAsBlock::RenderRubyAsBlock(Element& element, RenderStyle&& style)
66 : RenderBlockFlow(element, WTFMove(style))
67{
68}
69
70RenderRubyAsBlock::~RenderRubyAsBlock() = default;
71
72void RenderRubyAsBlock::styleDidChange(StyleDifference diff, const RenderStyle* oldStyle)
73{
74 RenderBlockFlow::styleDidChange(diff, oldStyle);
75 propagateStyleToAnonymousChildren(PropagateToAllChildren);
76}
