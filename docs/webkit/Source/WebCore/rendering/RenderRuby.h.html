36namespace WebCore {

38// Following the HTML 5 spec, the box object model for a <ruby> element allows several runs of ruby
39// bases with their respective ruby texts looks as follows:
40//
41// 1 RenderRuby object, corresponding to the whole <ruby> HTML element
42// 1+ RenderRubyRun (anonymous)
43// 0 or 1 RenderRubyText - shuffled to the front in order to re-use existing block layouting
44// 0-n inline object(s)
45// 0 or 1 RenderRubyBase - contains the inline objects that make up the ruby base
46// 1-n inline object(s)
47//
48// Note: <rp> elements are defined as having 'display:none' and thus normally are not assigned a renderer.
49//
50// Generated :before/:after content is shunted into anonymous inline blocks
51
52// <ruby> when used as 'display:inline'
53class RenderRubyAsInline final : public RenderInline {
54 WTF_MAKE_ISO_ALLOCATED(RenderRubyAsInline);
55public:
56 RenderRubyAsInline(Element&, RenderStyle&&);
57 virtual ~RenderRubyAsInline();
58
59private:
60 void styleDidChange(StyleDifference, const RenderStyle* oldStyle) override;
61
62 bool isRubyInline() const final { return true; }
63 const char* renderName() const override { return "RenderRuby (inline)"; }
64 bool createsAnonymousWrapper() const override { return true; }
65};
66
67// <ruby> when used as 'display:block' or 'display:inline-block'
68class RenderRubyAsBlock final : public RenderBlockFlow {
69 WTF_MAKE_ISO_ALLOCATED(RenderRubyAsBlock);
70public:
71 RenderRubyAsBlock(Element&, RenderStyle&&);
72 virtual ~RenderRubyAsBlock();
73
74 Element& element() const { return downcast<Element>(nodeForNonAnonymous()); }
75
76private:
77 void styleDidChange(StyleDifference, const RenderStyle* oldStyle) override;
78
79 bool isRubyBlock() const final { return true; }
80 const char* renderName() const override { return "RenderRuby (block)"; }
81 bool createsAnonymousWrapper() const override { return true; }
82};
83
84
85inline bool isRuby(const RenderObject& renderer) { return (is<RenderRubyAsInline>(renderer) || is<RenderRubyAsBlock>(renderer)); }
86inline bool isRuby(const RenderObject* renderer) { return (renderer && isRuby(*renderer)); }
87
88} // namespace WebCore
