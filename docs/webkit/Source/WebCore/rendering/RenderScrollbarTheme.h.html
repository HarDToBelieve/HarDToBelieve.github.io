26#pragma once
27
28#include "ScrollbarThemeComposite.h"
29
30namespace WebCore {
31
32class PlatformMouseEvent;
33class Scrollbar;
34class ScrollView;
35
36class RenderScrollbarTheme final : public ScrollbarThemeComposite {
37public:
38 virtual ~RenderScrollbarTheme() = default;
39
40 int scrollbarThickness(ScrollbarControlSize controlSize, ScrollbarExpansionState expansionState = ScrollbarExpansionState::Expanded) override { return ScrollbarTheme::theme().scrollbarThickness(controlSize, expansionState); }
41
42 ScrollbarButtonsPlacement buttonsPlacement() const override { return ScrollbarTheme::theme().buttonsPlacement(); }
43
44 bool supportsControlTints() const override { return true; }
45
46 void paintScrollCorner(GraphicsContext&, const IntRect& cornerRect) override;
47
48 ScrollbarButtonPressAction handleMousePressEvent(Scrollbar& scrollbar, const PlatformMouseEvent& event, ScrollbarPart pressedPart) override { return ScrollbarTheme::theme().handleMousePressEvent(scrollbar, event, pressedPart); }
49
50 Seconds initialAutoscrollTimerDelay() override { return ScrollbarTheme::theme().initialAutoscrollTimerDelay(); }
51 Seconds autoscrollTimerDelay() override { return ScrollbarTheme::theme().autoscrollTimerDelay(); }
52
53 void registerScrollbar(Scrollbar& scrollbar) override { return ScrollbarTheme::theme().registerScrollbar(scrollbar); }
54 void unregisterScrollbar(Scrollbar& scrollbar) override { return ScrollbarTheme::theme().unregisterScrollbar(scrollbar); }
55
56 int minimumThumbLength(Scrollbar&) override;
57
58 void buttonSizesAlongTrackAxis(Scrollbar&, int& beforeSize, int& afterSize);
59
60 static RenderScrollbarTheme* renderScrollbarTheme();
61
62private:
63 bool hasButtons(Scrollbar&) override;
64 bool hasThumb(Scrollbar&) override;
65
66 IntRect backButtonRect(Scrollbar&, ScrollbarPart, bool painting = false) override;
67 IntRect forwardButtonRect(Scrollbar&, ScrollbarPart, bool painting = false) override;
68 IntRect trackRect(Scrollbar&, bool painting = false) override;
69
70 void willPaintScrollbar(GraphicsContext&, Scrollbar&) override;
71 void didPaintScrollbar(GraphicsContext&, Scrollbar&) override;
72
73 void paintScrollbarBackground(GraphicsContext&, Scrollbar&) override;
74 void paintTrackBackground(GraphicsContext&, Scrollbar&, const IntRect&) override;
75 void paintTrackPiece(GraphicsContext&, Scrollbar&, const IntRect&, ScrollbarPart) override;
76 void paintButton(GraphicsContext&, Scrollbar&, const IntRect&, ScrollbarPart) override;
77 void paintThumb(GraphicsContext&, Scrollbar&, const IntRect&) override;
78 void paintTickmarks(GraphicsContext&, Scrollbar&, const IntRect&) override;
79
80 IntRect constrainTrackRectToTrackPieces(Scrollbar&, const IntRect&) override;
81};
82
83} // namespace WebCore
