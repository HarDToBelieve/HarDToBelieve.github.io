124void RenderTable::styleDidChange(StyleDifference diff, const RenderStyle* oldStyle)
125{
126 RenderBlock::styleDidChange(diff, oldStyle);
127 propagateStyleToAnonymousChildren(PropagateToAllChildren);
128
129 auto oldTableLayout = oldStyle ? oldStyle->tableLayout() : TableLayoutType::Auto;
130
131 // In the collapsed border model, there is no cell spacing.
132 m_hSpacing = collapseBorders() ? 0 : style().horizontalBorderSpacing();
133 m_vSpacing = collapseBorders() ? 0 : style().verticalBorderSpacing();
134 m_columnPos[0] = m_hSpacing;
135
136 if (!m_tableLayout || style().tableLayout() != oldTableLayout) {
137 // According to the CSS2 spec, you only use fixed table layout if an
138 // explicit width is specified on the table. Auto width implies auto table layout.
139 if (style().tableLayout() == TableLayoutType::Fixed && !style().logicalWidth().isAuto())
140 m_tableLayout = makeUnique<FixedTableLayout>(this);
141 else
142 m_tableLayout = makeUnique<AutoTableLayout>(this);
143 }
144
145 // If border was changed, invalidate collapsed borders cache.
146 if (oldStyle && oldStyle->border() != style().border())
147 invalidateCollapsedBorders();
148}

150static inline void resetSectionPointerIfNotBefore(WeakPtr<RenderTableSection>& section, RenderObject* before)
151{
152 if (!before || !section)
153 return;
154 auto* previousSibling = before->previousSibling();
155 while (previousSibling && previousSibling != section)
156 previousSibling = previousSibling->previousSibling();
157 if (!previousSibling)
158 section.clear();
159}

166void RenderTable::willInsertTableSection(RenderTableSection& child, RenderObject* beforeChild)
167{
168 switch (child.style().display()) {
169 case DisplayType::TableHeaderGroup:
170 resetSectionPointerIfNotBefore(m_head, beforeChild);
171 if (!m_head)
172 m_head = makeWeakPtr(child);
173 else {
174 resetSectionPointerIfNotBefore(m_firstBody, beforeChild);
175 if (!m_firstBody)
176 m_firstBody = makeWeakPtr(child);
177 }
178 break;
179 case DisplayType::TableFooterGroup:
180 resetSectionPointerIfNotBefore(m_foot, beforeChild);
181 if (!m_foot) {
182 m_foot = makeWeakPtr(child);
183 break;
184 }
185 FALLTHROUGH;
186 case DisplayType::TableRowGroup:
187 resetSectionPointerIfNotBefore(m_firstBody, beforeChild);
188 if (!m_firstBody)
189 m_firstBody = makeWeakPtr(child);
190 break;
191 default:
192 ASSERT_NOT_REACHED();
193 }
194
195 setNeedsSectionRecalc();
196}

237void RenderTable::updateLogicalWidth()
238{
239 recalcSectionsIfNeeded();
240
241 if (isGridItem()) {
242 // FIXME: Investigate whether the grid layout algorithm provides all the logic
243 // needed and that we're not skipping anything essential due to the early return here.
244 RenderBlock::updateLogicalWidth();
245 return;
246 }
247
248 if (isOutOfFlowPositioned()) {
249 LogicalExtentComputedValues computedValues;
250 computePositionedLogicalWidth(computedValues);
251 setLogicalWidth(computedValues.m_extent);
252 setLogicalLeft(computedValues.m_position);
253 setMarginStart(computedValues.m_margins.m_start);
254 setMarginEnd(computedValues.m_margins.m_end);
255 }
256
257 RenderBlock& cb = *containingBlock();
258
259 LayoutUnit availableLogicalWidth = containingBlockLogicalWidthForContent();
260 bool hasPerpendicularContainingBlock = cb.style().isHorizontalWritingMode() != style().isHorizontalWritingMode();
261 LayoutUnit containerWidthInInlineDirection = hasPerpendicularContainingBlock ? perpendicularContainingBlockLogicalHeight() : availableLogicalWidth;
262
263 Length styleLogicalWidth = style().logicalWidth();
264 if (hasOverridingLogicalWidth())
265 setLogicalWidth(overridingLogicalWidth());
266 else if ((styleLogicalWidth.isSpecified() && styleLogicalWidth.isPositive()) || styleLogicalWidth.isIntrinsic())
267 setLogicalWidth(convertStyleLogicalWidthToComputedWidth(styleLogicalWidth, containerWidthInInlineDirection));
268 else {
269 // Subtract out any fixed margins from our available width for auto width tables.
270 LayoutUnit marginStart = minimumValueForLength(style().marginStart(), availableLogicalWidth);
271 LayoutUnit marginEnd = minimumValueForLength(style().marginEnd(), availableLogicalWidth);
272 LayoutUnit marginTotal = marginStart + marginEnd;
273
274 // Subtract out our margins to get the available content width.
275 LayoutUnit availableContentLogicalWidth = std::max<LayoutUnit>(0, containerWidthInInlineDirection - marginTotal);
276 if (shrinkToAvoidFloats() && cb.containsFloats() && !hasPerpendicularContainingBlock) {
277 // FIXME: Work with regions someday.
278 availableContentLogicalWidth = shrinkLogicalWidthToAvoidFloats(marginStart, marginEnd, cb, 0);
279 }
280
281 // Ensure we aren't bigger than our available width.
282 setLogicalWidth(std::min(availableContentLogicalWidth, maxPreferredLogicalWidth()));
283 LayoutUnit maxWidth = maxPreferredLogicalWidth();
284 // scaledWidthFromPercentColumns depends on m_layoutStruct in TableLayoutAlgorithmAuto, which
285 // maxPreferredLogicalWidth fills in. So scaledWidthFromPercentColumns has to be called after
286 // maxPreferredLogicalWidth.
287 LayoutUnit scaledWidth = m_tableLayout->scaledWidthFromPercentColumns() + bordersPaddingAndSpacingInRowDirection();
288 maxWidth = std::max(scaledWidth, maxWidth);
289 setLogicalWidth(std::min(availableContentLogicalWidth, maxWidth));
290 }
291
292 // Ensure we aren't bigger than our max-width style.
293 Length styleMaxLogicalWidth = style().logicalMaxWidth();
294 if ((styleMaxLogicalWidth.isSpecified() && !styleMaxLogicalWidth.isNegative()) || styleMaxLogicalWidth.isIntrinsic()) {
295 LayoutUnit computedMaxLogicalWidth = convertStyleLogicalWidthToComputedWidth(styleMaxLogicalWidth, availableLogicalWidth);
296 setLogicalWidth(std::min(logicalWidth(), computedMaxLogicalWidth));
297 }
298
299 // Ensure we aren't smaller than our min preferred width.
300 setLogicalWidth(std::max(logicalWidth(), minPreferredLogicalWidth()));
301
302 // Ensure we aren't smaller than our min-width style.
303 Length styleMinLogicalWidth = style().logicalMinWidth();
304 if ((styleMinLogicalWidth.isSpecified() && !styleMinLogicalWidth.isNegative()) || styleMinLogicalWidth.isIntrinsic()) {
305 LayoutUnit computedMinLogicalWidth = convertStyleLogicalWidthToComputedWidth(styleMinLogicalWidth, availableLogicalWidth);
306 setLogicalWidth(std::max(logicalWidth(), computedMinLogicalWidth));
307 }
308
309 // Finally, with our true width determined, compute our margins for real.
310 setMarginStart(0);
311 setMarginEnd(0);
312 if (!hasPerpendicularContainingBlock) {
313 LayoutUnit containerLogicalWidthForAutoMargins = availableLogicalWidth;
314 if (avoidsFloats() && cb.containsFloats())
315 containerLogicalWidthForAutoMargins = containingBlockAvailableLineWidthInFragment(0); // FIXME: Work with regions someday.
316 ComputedMarginValues marginValues;
317 bool hasInvertedDirection = cb.style().isLeftToRightDirection() == style().isLeftToRightDirection();
318 computeInlineDirectionMargins(cb, containerLogicalWidthForAutoMargins, logicalWidth(),
319 hasInvertedDirection ? marginValues.m_start : marginValues.m_end,
320 hasInvertedDirection ? marginValues.m_end : marginValues.m_start);
321 setMarginStart(marginValues.m_start);
322 setMarginEnd(marginValues.m_end);
323 } else {
324 setMarginStart(minimumValueForLength(style().marginStart(), availableLogicalWidth));
325 setMarginEnd(minimumValueForLength(style().marginEnd(), availableLogicalWidth));
326 }
327}

329// This method takes a RenderStyle's logical width, min-width, or max-width length and computes its actual value.
330LayoutUnit RenderTable::convertStyleLogicalWidthToComputedWidth(const Length& styleLogicalWidth, LayoutUnit availableWidth)
331{
332 if (styleLogicalWidth.isIntrinsic())
333 return computeIntrinsicLogicalWidthUsing(styleLogicalWidth, availableWidth, bordersPaddingAndSpacingInRowDirection());
334
335 // HTML tables' width styles already include borders and padding, but CSS tables' width styles do not.
336 LayoutUnit borders;
337 bool isCSSTable = !is<HTMLTableElement>(element());
338 if (isCSSTable && styleLogicalWidth.isSpecified() && styleLogicalWidth.isPositive() && style().boxSizing() == BoxSizing::ContentBox)
339 borders = borderStart() + borderEnd() + (collapseBorders() ? 0_lu : paddingStart() + paddingEnd());
340
341 return minimumValueForLength(styleLogicalWidth, availableWidth) + borders;
342}

344LayoutUnit RenderTable::convertStyleLogicalHeightToComputedHeight(const Length& styleLogicalHeight)
345{
346 LayoutUnit borderAndPaddingBefore = borderBefore() + (collapseBorders() ? 0_lu : paddingBefore());
347 LayoutUnit borderAndPaddingAfter = borderAfter() + (collapseBorders() ? 0_lu : paddingAfter());
348 LayoutUnit borderAndPadding = borderAndPaddingBefore + borderAndPaddingAfter;
349 if (styleLogicalHeight.isFixed()) {
350 // HTML tables size as though CSS height includes border/padding, CSS tables do not.
351 LayoutUnit borders;
352 // FIXME: We cannot apply box-sizing: content-box on <table> which other browsers allow.
353 if (is<HTMLTableElement>(element()) || style().boxSizing() == BoxSizing::BorderBox) {
354 borders = borderAndPadding;
355 }
356 return LayoutUnit(styleLogicalHeight.value() - borders);
357 } else if (styleLogicalHeight.isPercentOrCalculated())
358 return computePercentageLogicalHeight(styleLogicalHeight).value_or(0);
359 else if (styleLogicalHeight.isIntrinsic())
360 return computeIntrinsicLogicalContentHeightUsing(styleLogicalHeight, logicalHeight() - borderAndPadding, borderAndPadding).value_or(0);
361 else
362 ASSERT_NOT_REACHED();
363 return 0_lu;
364}

366void RenderTable::layoutCaption(RenderTableCaption& caption)
367{
368 LayoutRect captionRect(caption.frameRect());
369
370 if (caption.needsLayout()) {
371 // The margins may not be available but ensure the caption is at least located beneath any previous sibling caption
372 // so that it does not mistakenly think any floats in the previous caption intrude into it.
373 caption.setLogicalLocation(LayoutPoint(caption.marginStart(), caption.marginBefore() + logicalHeight()));
374 // If RenderTableCaption ever gets a layout() function, use it here.
375 caption.layoutIfNeeded();
376 }
377 // Apply the margins to the location now that they are definitely available from layout
378 caption.setLogicalLocation(LayoutPoint(caption.marginStart(), caption.marginBefore() + logicalHeight()));
379
380 if (!selfNeedsLayout() && caption.checkForRepaintDuringLayout())
381 caption.repaintDuringLayoutIfMoved(captionRect);
382
383 setLogicalHeight(logicalHeight() + caption.logicalHeight() + caption.marginBefore() + caption.marginAfter());
384}

421void RenderTable::layout()
422{
423 StackStats::LayoutCheckPoint layoutCheckPoint;
424 ASSERT(needsLayout());
425
426 if (simplifiedLayout())
427 return;
428
429 recalcSectionsIfNeeded();
430 // FIXME: We should do this recalc lazily in borderStart/borderEnd so that we don't have to make sure
431 // to call this before we call borderStart/borderEnd to avoid getting a stale value.
432 recalcBordersInRowDirection();
433 bool sectionMoved = false;
434 LayoutUnit movedSectionLogicalTop;
435 unsigned sectionCount = 0;
436 bool shouldCacheIntrinsicContentLogicalHeightForFlexItem = true;
437
438 LayoutRepainter repainter(*this, checkForRepaintDuringLayout());
439 {
440 LayoutStateMaintainer statePusher(*this, locationOffset(), hasTransform() || hasReflection() || style().isFlippedBlocksWritingMode());
441
442 LayoutUnit oldLogicalWidth = logicalWidth();
443 LayoutUnit oldLogicalHeight = logicalHeight();
444 resetLogicalHeightBeforeLayoutIfNeeded();
445 updateLogicalWidth();
446
447 if (logicalWidth() != oldLogicalWidth) {
448 for (unsigned i = 0; i < m_captions.size(); i++)
449 m_captions[i]->setNeedsLayout(MarkOnlyThis);
450 }
451 // FIXME: The optimisation below doesn't work since the internal table
452 // layout could have changed. We need to add a flag to the table
453 // layout that tells us if something has changed in the min max
454 // calculations to do it correctly.
455 // if ( oldWidth != width() || columns.size() + 1 != columnPos.size() )
456 m_tableLayout->layout();
457
458 LayoutUnit totalSectionLogicalHeight;
459 LayoutUnit oldTableLogicalTop;
460 for (unsigned i = 0; i < m_captions.size(); i++) {
461 if (m_captions[i]->style().captionSide() == CaptionSide::Bottom)
462 continue;
463 oldTableLogicalTop += m_captions[i]->logicalHeight() + m_captions[i]->marginBefore() + m_captions[i]->marginAfter();
464 }
465
466 bool collapsing = collapseBorders();
467
468 for (auto& child : childrenOfType<RenderElement>(*this)) {
469 if (is<RenderTableSection>(child)) {
470 RenderTableSection& section = downcast<RenderTableSection>(child);
471 if (m_columnLogicalWidthChanged)
472 section.setChildNeedsLayout(MarkOnlyThis);
473 section.layoutIfNeeded();
474 totalSectionLogicalHeight += section.calcRowLogicalHeight();
475 if (collapsing)
476 section.recalcOuterBorder();
477 ASSERT(!section.needsLayout());
478 } else if (is<RenderTableCol>(child)) {
479 downcast<RenderTableCol>(child).layoutIfNeeded();
480 ASSERT(!child.needsLayout());
481 }
482 }
483
484 // If any table section moved vertically, we will just repaint everything from that
485 // section down (it is quite unlikely that any of the following sections
486 // did not shift).
487 layoutCaptions();
488 if (!m_captions.isEmpty() && logicalHeight() != oldTableLogicalTop) {
489 sectionMoved = true;
490 movedSectionLogicalTop = std::min(logicalHeight(), oldTableLogicalTop);
491 }
492
493 LayoutUnit borderAndPaddingBefore = borderBefore() + (collapsing ? 0_lu : paddingBefore());
494 LayoutUnit borderAndPaddingAfter = borderAfter() + (collapsing ? 0_lu : paddingAfter());
495
496 setLogicalHeight(logicalHeight() + borderAndPaddingBefore);
497
498 if (!isOutOfFlowPositioned())
499 updateLogicalHeight();
500
501 LayoutUnit computedLogicalHeight;
502
503 Length logicalHeightLength = style().logicalHeight();
504 if (logicalHeightLength.isIntrinsic() || (logicalHeightLength.isSpecified() && logicalHeightLength.isPositive()))
505 computedLogicalHeight = convertStyleLogicalHeightToComputedHeight(logicalHeightLength);
506
507 if (hasOverridingLogicalHeight()) {
508 LayoutUnit captionLogicalHeight;
509 for (auto& caption : m_captions)
510 captionLogicalHeight += caption->logicalHeight() + caption->marginBefore() + caption->marginAfter();
511 computedLogicalHeight = std::max(computedLogicalHeight, overridingLogicalHeight() - captionLogicalHeight);
512 }
513
514 Length logicalMaxHeightLength = style().logicalMaxHeight();
515 if (logicalMaxHeightLength.isIntrinsic() || (logicalMaxHeightLength.isSpecified() && !logicalMaxHeightLength.isNegative())) {
516 LayoutUnit computedMaxLogicalHeight = convertStyleLogicalHeightToComputedHeight(logicalMaxHeightLength);
517 computedLogicalHeight = std::min(computedLogicalHeight, computedMaxLogicalHeight);
518 }
519
520 Length logicalMinHeightLength = style().logicalMinHeight();
521 if (logicalMinHeightLength.isIntrinsic() || (logicalMinHeightLength.isSpecified() && !logicalMinHeightLength.isNegative())) {
522 LayoutUnit computedMinLogicalHeight = convertStyleLogicalHeightToComputedHeight(logicalMinHeightLength);
523 computedLogicalHeight = std::max(computedLogicalHeight, computedMinLogicalHeight);
524 }
525
526 distributeExtraLogicalHeight(computedLogicalHeight - totalSectionLogicalHeight);
527
528 for (RenderTableSection* section = topSection(); section; section = sectionBelow(section))
529 section->layoutRows();
530
531 if (!topSection() && computedLogicalHeight > totalSectionLogicalHeight && !document().inQuirksMode()) {
532 // Completely empty tables (with no sections or anything) should at least honor their
533 // overriding or specified height in strict mode, but this value will not be cached.
534 shouldCacheIntrinsicContentLogicalHeightForFlexItem = false;
535 setLogicalHeight(hasOverridingLogicalHeight() ? overridingLogicalHeight() : logicalHeight() + computedLogicalHeight);
536 }
537
538 LayoutUnit sectionLogicalLeft = style().isLeftToRightDirection() ? borderStart() : borderEnd();
539 if (!collapsing)
540 sectionLogicalLeft += style().isLeftToRightDirection() ? paddingStart() : paddingEnd();
541
542 // position the table sections
543 RenderTableSection* section = topSection();
544 while (section) {
545 sectionCount++;
546 if (!sectionMoved && section->logicalTop() != logicalHeight()) {
547 sectionMoved = true;
548 movedSectionLogicalTop = std::min(logicalHeight(), section->logicalTop()) + (style().isHorizontalWritingMode() ? section->visualOverflowRect().y() : section->visualOverflowRect().x());
549 }
550 section->setLogicalLocation(LayoutPoint(sectionLogicalLeft, logicalHeight()));
551
552 setLogicalHeight(logicalHeight() + section->logicalHeight());
553 section = sectionBelow(section);
554 }
555
556 setLogicalHeight(logicalHeight() + borderAndPaddingAfter);
557
558 layoutCaptions(BottomCaptionLayoutPhase::Yes);
559
560 if (isOutOfFlowPositioned())
561 updateLogicalHeight();
562
563 // table can be containing block of positioned elements.
564 bool dimensionChanged = oldLogicalWidth != logicalWidth() || oldLogicalHeight != logicalHeight();
565 layoutPositionedObjects(dimensionChanged);
566
567 updateLayerTransform();
568
569 // Layout was changed, so probably borders too.
570 invalidateCollapsedBorders();
571
572 // The location or height of one or more sections may have changed.
573 invalidateCachedColumnOffsets();
574
575 computeOverflow(clientLogicalBottom());
576 }
577
578 auto* layoutState = view().frameView().layoutContext().layoutState();
579 if (layoutState && layoutState->pageLogicalHeight())
580 setPageLogicalOffset(layoutState->pageLogicalOffset(this, logicalTop()));
581
582 bool didFullRepaint = repainter.repaintAfterLayout();
583 // Repaint with our new bounds if they are different from our old bounds.
584 if (!didFullRepaint && sectionMoved) {
585 if (style().isHorizontalWritingMode())
586 repaintRectangle(LayoutRect(visualOverflowRect().x(), movedSectionLogicalTop, visualOverflowRect().width(), visualOverflowRect().maxY() - movedSectionLogicalTop));
587 else
588 repaintRectangle(LayoutRect(movedSectionLogicalTop, visualOverflowRect().y(), visualOverflowRect().maxX() - movedSectionLogicalTop, visualOverflowRect().height()));
589 }
590
591 bool paginated = layoutState && layoutState->isPaginated();
592 if (sectionMoved && paginated) {
593 // FIXME: Table layout should always stabilize even when section moves (see webkit.org/b/174412).
594 if (m_recursiveSectionMovedWithPaginationLevel < sectionCount) {
595 SetForScope<unsigned> recursiveSectionMovedWithPaginationLevel(m_recursiveSectionMovedWithPaginationLevel, m_recursiveSectionMovedWithPaginationLevel + 1);
596 markForPaginationRelayoutIfNeeded();
597 layoutIfNeeded();
598 } else
599 ASSERT_NOT_REACHED();
600 }
601
602 // FIXME: This value isn't the intrinsic content logical height, but we need
603 // to update the value as its used by flexbox layout. crbug.com/367324
604 if (shouldCacheIntrinsicContentLogicalHeightForFlexItem)
605 cacheIntrinsicContentLogicalHeightForFlexItem(contentLogicalHeight());
606
607 m_columnLogicalWidthChanged = false;
608 clearNeedsLayout();
609}

611void RenderTable::invalidateCollapsedBorders(RenderTableCell* cellWithStyleChange)
612{
613 m_collapsedBordersValid = false;
614 m_collapsedBorders.clear();
615
616 for (auto& section : childrenOfType<RenderTableSection>(*this))
617 section.clearCachedCollapsedBorders();
618
619 if (!m_collapsedEmptyBorderIsPresent)
620 return;
621
622 if (cellWithStyleChange) {
623 // It is enough to invalidate just the surrounding cells when cell border style changes.
624 cellWithStyleChange->invalidateHasEmptyCollapsedBorders();
625 if (auto* below = cellBelow(cellWithStyleChange))
626 below->invalidateHasEmptyCollapsedBorders();
627 if (auto* above = cellAbove(cellWithStyleChange))
628 above->invalidateHasEmptyCollapsedBorders();
629 if (auto* before = cellBefore(cellWithStyleChange))
630 before->invalidateHasEmptyCollapsedBorders();
631 if (auto* after = cellAfter(cellWithStyleChange))
632 after->invalidateHasEmptyCollapsedBorders();

634 }
635
636 for (auto& section : childrenOfType<RenderTableSection>(*this)) {
637 for (auto* row = section.firstRow(); row; row = row->nextRow()) {
638 for (auto* cell = row->firstCell(); cell; cell = cell->nextCell()) {

640 cell->invalidateHasEmptyCollapsedBorders();
641 }
642 }
643 }
644 m_collapsedEmptyBorderIsPresent = false;

665void RenderTable::addOverflowFromChildren()
666{
667 // Add overflow from borders.
668 // Technically it's odd that we are incorporating the borders into layout overflow, which is only supposed to be about overflow from our
669 // descendant objects, but since tables don't support overflow:auto, this works out fine.
670 if (collapseBorders()) {
671 LayoutUnit rightBorderOverflow = width() + outerBorderRight() - borderRight();
672 LayoutUnit leftBorderOverflow = borderLeft() - outerBorderLeft();
673 LayoutUnit bottomBorderOverflow = height() + outerBorderBottom() - borderBottom();
674 LayoutUnit topBorderOverflow = borderTop() - outerBorderTop();
675 LayoutRect borderOverflowRect(leftBorderOverflow, topBorderOverflow, rightBorderOverflow - leftBorderOverflow, bottomBorderOverflow - topBorderOverflow);
676 if (borderOverflowRect != borderBoxRect()) {
677 addLayoutOverflow(borderOverflowRect);
678 addVisualOverflow(borderOverflowRect);
679 }

683 for (unsigned i = 0; i < m_captions.size(); i++)
684 addOverflowFromChild(m_captions[i].get());

686 // Add overflow from our sections.
687 for (RenderTableSection* section = topSection(); section; section = sectionBelow(section))

691void RenderTable::paint(PaintInfo& paintInfo, const LayoutPoint& paintOffset)
692{
693 LayoutPoint adjustedPaintOffset = paintOffset + location();
694
695 PaintPhase paintPhase = paintInfo.phase;
696
697 if (!isDocumentElementRenderer()) {
698 LayoutRect overflowBox = visualOverflowRect();
699 flipForWritingMode(overflowBox);
700 overflowBox.moveBy(adjustedPaintOffset);
701 if (!overflowBox.intersects(paintInfo.rect))
702 return;
703 }
704
705 bool pushedClip = pushContentsClip(paintInfo, adjustedPaintOffset);
706 paintObject(paintInfo, adjustedPaintOffset);
707 if (pushedClip)
708 popContentsClip(paintInfo, paintPhase, adjustedPaintOffset);
709}
710
711void RenderTable::paintObject(PaintInfo& paintInfo, const LayoutPoint& paintOffset)
712{
713 PaintPhase paintPhase = paintInfo.phase;
714 if ((paintPhase == PaintPhase::BlockBackground || paintPhase == PaintPhase::ChildBlockBackground) && hasVisibleBoxDecorations() && style().visibility() == Visibility::Visible)
715 paintBoxDecorations(paintInfo, paintOffset);
716
717 if (paintPhase == PaintPhase::Mask) {
718 paintMask(paintInfo, paintOffset);
719 return;
720 }
721
722 // We're done. We don't bother painting any children.
723 if (paintPhase == PaintPhase::BlockBackground)
724 return;
725
726 // We don't paint our own background, but we do let the kids paint their backgrounds.
727 if (paintPhase == PaintPhase::ChildBlockBackgrounds)
728 paintPhase = PaintPhase::ChildBlockBackground;
729
730 PaintInfo info(paintInfo);
731 info.phase = paintPhase;
732 info.updateSubtreePaintRootForChildren(this);
733
734 for (auto& box : childrenOfType<RenderBox>(*this)) {
735 if (!box.hasSelfPaintingLayer() && (box.isTableSection() || box.isTableCaption())) {
736 LayoutPoint childPoint = flipForWritingModeForChild(&box, paintOffset);
737 box.paint(info, childPoint);
738 }
739 }
740
741 if (collapseBorders() && paintPhase == PaintPhase::ChildBlockBackground && style().visibility() == Visibility::Visible) {
742 recalcCollapsedBorders();
743 // Using our cached sorted styles, we then do individual passes,
744 // painting each style of border from lowest precedence to highest precedence.
745 info.phase = PaintPhase::CollapsedTableBorders;
746 size_t count = m_collapsedBorders.size();
747 for (size_t i = 0; i < count; ++i) {
748 m_currentBorder = &m_collapsedBorders[i];
749 for (RenderTableSection* section = bottomSection(); section; section = sectionAbove(section)) {
750 LayoutPoint childPoint = flipForWritingModeForChild(section, paintOffset);
751 section->paint(info, childPoint);
752 }
753 }
754 m_currentBorder = 0;
755 }
756
757 // Paint outline.
758 if ((paintPhase == PaintPhase::Outline || paintPhase == PaintPhase::SelfOutline) && hasOutline() && style().visibility() == Visibility::Visible)
759 paintOutline(paintInfo, LayoutRect(paintOffset, size()));
760}

987LayoutUnit RenderTable::offsetWidthForColumn(const RenderTableCol& column) const
988{
989 const RenderTableCol* currentColumn = &column;
990 bool hasColumnChildren;
991 if ((hasColumnChildren = currentColumn->isTableColumnGroupWithColumnChildren()))
992 currentColumn = currentColumn->nextColumn(); // First column in column-group
993 unsigned numberOfEffectiveColumns = numEffCols();
994 ASSERT_WITH_SECURITY_IMPLICATION(m_columnPos.size() >= numberOfEffectiveColumns + 1);
995 LayoutUnit width;
996 LayoutUnit spacing = m_hSpacing;
997 while (currentColumn) {
998 unsigned columnIndex = effectiveIndexOfColumn(*currentColumn);
999 unsigned span = currentColumn->span();
1000 while (span && columnIndex < numberOfEffectiveColumns) {
1001 width += m_columnPos[columnIndex + 1] - m_columnPos[columnIndex] - spacing;
1002 span -= m_columns[columnIndex].span;
1003 ++columnIndex;
1004 if (span)
1005 width += spacing;
1006 }
1007 if (!hasColumnChildren)
1008 break;
1009 currentColumn = currentColumn->nextColumn();
1010 if (!currentColumn || currentColumn->isTableColumnGroup())
1011 break;
1012 width += spacing;
1013 }
1014 return width;
1015}

1059void RenderTable::recalcSections() const
1060{
1061 ASSERT(m_needsSectionRecalc);
1062
1063 m_head.clear();
1064 m_foot.clear();
1065 m_firstBody.clear();
1066 m_hasColElements = false;
1067 m_hasCellColspanThatDeterminesTableWidth = hasCellColspanThatDeterminesTableWidth();
1068
1069 // We need to get valid pointers to caption, head, foot and first body again
1070 RenderObject* nextSibling;
1071 for (RenderObject* child = firstChild(); child; child = nextSibling) {
1072 nextSibling = child->nextSibling();
1073 switch (child->style().display()) {
1074 case DisplayType::TableColumn:
1075 case DisplayType::TableColumnGroup:
1076 m_hasColElements = true;
1077 break;
1078 case DisplayType::TableHeaderGroup:
1079 if (is<RenderTableSection>(*child)) {
1080 RenderTableSection& section = downcast<RenderTableSection>(*child);
1081 if (!m_head)
1082 m_head = makeWeakPtr(section);
1083 else if (!m_firstBody)
1084 m_firstBody = makeWeakPtr(section);
1085 section.recalcCellsIfNeeded();
1086 }
1087 break;
1088 case DisplayType::TableFooterGroup:
1089 if (is<RenderTableSection>(*child)) {
1090 RenderTableSection& section = downcast<RenderTableSection>(*child);
1091 if (!m_foot)
1092 m_foot = makeWeakPtr(section);
1093 else if (!m_firstBody)
1094 m_firstBody = makeWeakPtr(section);
1095 section.recalcCellsIfNeeded();
1096 }
1097 break;
1098 case DisplayType::TableRowGroup:
1099 if (is<RenderTableSection>(*child)) {
1100 RenderTableSection& section = downcast<RenderTableSection>(*child);
1101 if (!m_firstBody)
1102 m_firstBody = makeWeakPtr(section);
1103 section.recalcCellsIfNeeded();
1104 }
1105 break;
1106 default:
1107 break;
1108 }
1109 }
1110
1111 // repair column count (addChild can grow it too much, because it always adds elements to the last row of a section)
1112 unsigned maxCols = 0;
1113 for (auto& section : childrenOfType<RenderTableSection>(*this)) {
1114 unsigned sectionCols = section.numColumns();
1115 if (sectionCols > maxCols)
1116 maxCols = sectionCols;
1117 }
1118
1119 m_columns.resize(maxCols);
1120 m_columnPos.resize(maxCols + 1);
1121
1122 // Now that we know the number of maximum number of columns, let's shrink the sections grids if needed.
1123 for (auto& section : childrenOfType<RenderTableSection>(const_cast<RenderTable&>(*this)))
1124 section.removeRedundantColumns();
1125
1126 ASSERT(selfNeedsLayout());
1127
1128 m_needsSectionRecalc = false;
1129}

1529LayoutRect RenderTable::overflowClipRect(const LayoutPoint& location, RenderFragmentContainer* fragment, OverlayScrollbarSizeRelevancy relevancy, PaintPhase phase) const
1530{

1532 // Don't clip out the table's side of the collapsed borders if we're in the paint phase that will ask the sections to paint them.
1533 // Likewise, if we're self-painting we avoid clipping them out as the clip rect that will be passed down to child layers from RenderLayer will do that instead.
1534 if (phase == PaintPhase::ChildBlockBackgrounds || layer()->isSelfPaintingLayer()) {
1535 rect = borderBoxRectInFragment(fragment);
1536 rect.setLocation(location + rect.location());
1537 } else
1538 rect = RenderBox::overflowClipRect(location, fragment, relevancy);
1539
1540 // If we have a caption, expand the clip to include the caption.
1541 // FIXME: Technically this is wrong, but it's virtually impossible to fix this
1542 // for real until captions have been re-written.
1543 // FIXME: This code assumes (like all our other caption code) that only top/bottom are
1544 // supported. When we actually support left/right and stop mapping them to top/bottom,
1545 // we might have to hack this code first (depending on what order we do these bug fixes in).
1546 if (!m_captions.isEmpty()) {
1547 if (style().isHorizontalWritingMode()) {
1548 rect.setHeight(height());
1549 rect.setY(location.y());
1550 } else {
1551 rect.setWidth(width());
1552 rect.setX(location.x());

1559bool RenderTable::nodeAtPoint(const HitTestRequest& request, HitTestResult& result, const HitTestLocation& locationInContainer, const LayoutPoint& accumulatedOffset, HitTestAction action)
1560{
1561 LayoutPoint adjustedLocation = accumulatedOffset + location();
1562
1563 // Check kids first.
1564 if (!hasOverflowClip() || locationInContainer.intersects(overflowClipRect(adjustedLocation, nullptr))) {
1565 for (RenderObject* child = lastChild(); child; child = child->previousSibling()) {
1566 if (is<RenderBox>(*child) && !downcast<RenderBox>(*child).hasSelfPaintingLayer() && (child->isTableSection() || child->isTableCaption())) {
1567 LayoutPoint childPoint = flipForWritingModeForChild(downcast<RenderBox>(child), adjustedLocation);
1568 if (child->nodeAtPoint(request, result, locationInContainer, childPoint, action)) {
1569 updateHitTestResult(result, toLayoutPoint(locationInContainer.point() - childPoint));
1570 return true;
1571 }
1572 }
1573 }
1574 }
1575
1576 // Check our bounds next.

1578 if (visibleToHitTesting(request) && (action == HitTestBlockBackground || action == HitTestChildBlockBackground) && locationInContainer.intersects(boundsRect)) {

1580 if (result.addNodeToListBasedTestResult(nodeForHitTest(), request, locationInContainer, boundsRect) == HitTestProgress::Stop)
