174 unsigned colToEffCol(unsigned column) const
175 {
176 if (!m_hasCellColspanThatDeterminesTableWidth)
177 return column;
178
179 unsigned effColumn = 0;
180 unsigned numColumns = numEffCols();
181 for (unsigned c = 0; effColumn < numColumns && c + m_columns[effColumn].span - 1 < column; ++effColumn)
182 c += m_columns[effColumn].span;
183 return effColumn;
184 }
185
186 unsigned effColToCol(unsigned effCol) const
187 {
188 if (!m_hasCellColspanThatDeterminesTableWidth)
189 return effCol;
190
191 unsigned c = 0;
192 for (unsigned i = 0; i < effCol; i++)
193 c += m_columns[i].span;
194 return c;
195 }
196
197 LayoutUnit borderSpacingInRowDirection() const
198 {
199 if (unsigned effectiveColumnCount = numEffCols())
200 return (effectiveColumnCount + 1) * hBorderSpacing();
201
202 return 0;
203 }

205 LayoutUnit bordersPaddingAndSpacingInRowDirection() const
206 {
207 // 'border-spacing' only applies to separate borders (see 17.6.1 The separated borders model).
208 return borderStart() + borderEnd() + (collapseBorders() ? 0_lu : (paddingStart() + paddingEnd() + borderSpacingInRowDirection()));
209 }
