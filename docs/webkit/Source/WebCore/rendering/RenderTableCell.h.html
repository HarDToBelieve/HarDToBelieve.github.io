276inline Length RenderTableCell::styleOrColLogicalWidth() const
277{
278 Length styleWidth = style().logicalWidth();
279 if (!styleWidth.isAuto())
280 return styleWidth;
281 if (RenderTableCol* firstColumn = table()->colElement(col()))
282 return logicalWidthFromColumns(firstColumn, styleWidth);
283 return styleWidth;
284}

286inline LayoutUnit RenderTableCell::logicalHeightForRowSizing() const
287{
288 // FIXME: This function does too much work, and is very hot during table layout!
289 LayoutUnit adjustedLogicalHeight = logicalHeight() - (intrinsicPaddingBefore() + intrinsicPaddingAfter());
290 if (!style().logicalHeight().isSpecified())
291 return adjustedLogicalHeight;
292 LayoutUnit styleLogicalHeight = valueForLength(style().logicalHeight(), 0);
293 // In strict mode, box-sizing: content-box do the right thing and actually add in the border and padding.
294 // Call computedCSSPadding* directly to avoid including implicitPadding.
295 if (!document().inQuirksMode() && style().boxSizing() != BoxSizing::BorderBox)
296 styleLogicalHeight += computedCSSPaddingBefore() + computedCSSPaddingAfter() + borderBefore() + borderAfter();
297 return std::max(styleLogicalHeight, adjustedLogicalHeight);
298}

300inline bool RenderTableCell::isBaselineAligned() const
301{
302 VerticalAlign va = style().verticalAlign();
303 return va == VerticalAlign::Baseline || va == VerticalAlign::TextBottom || va == VerticalAlign::TextTop || va == VerticalAlign::Super || va == VerticalAlign::Sub || va == VerticalAlign::Length;
304}

306inline const BorderValue& RenderTableCell::borderAdjoiningTableStart() const
307{
308 ASSERT(isFirstOrLastCellInRow());
309 if (isDirectionSame(section(), table()))
310 return style().borderStart();
311
312 return style().borderEnd();
313}
314
315inline const BorderValue& RenderTableCell::borderAdjoiningTableEnd() const
316{
317 ASSERT(isFirstOrLastCellInRow());
318 if (isDirectionSame(section(), table()))
319 return style().borderEnd();
320
321 return style().borderStart();
322}

348inline void RenderTableCell::setHasEmptyCollapsedBorder(CollapsedBorderSide side, bool empty) const
349{
350 switch (side) {
351 case CBSAfter: {
352 m_hasEmptyCollapsedAfterBorder = empty;
353 break;
354 }
355 case CBSBefore: {
356 m_hasEmptyCollapsedBeforeBorder = empty;
357 break;
358 }
359 case CBSStart: {
360 m_hasEmptyCollapsedStartBorder = empty;
361 break;
362 }
363 case CBSEnd: {
364 m_hasEmptyCollapsedEndBorder = empty;
365 break;
366 }
367 }
368 if (empty)
369 table()->collapsedEmptyBorderIsPresent();
370}
