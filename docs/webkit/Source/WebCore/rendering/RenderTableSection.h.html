206 // These two functions take a rectangle as input that has been flipped by logicalRectForWritingModeAndDirection.
207 // The returned span of rows or columns is end-exclusive, and empty if start==end.
208 // The IncludeAllIntersectingCells argument is used to determine which cells to include when
209 // an edge of the flippedRect lies exactly on a cell boundary. Using IncludeAllIntersectingCells
210 // will return both cells, and using DoNotIncludeAllIntersectingCells will return only the cell
211 // that hittesting should return.
212 CellSpan spannedRows(const LayoutRect& flippedRect, ShouldIncludeAllIntersectingCells) const;
213 CellSpan spannedColumns(const LayoutRect& flippedRect, ShouldIncludeAllIntersectingCells) const;

285inline LayoutUnit RenderTableSection::outerBorderLeft(const RenderStyle* styleForCellFlow) const
286{
287 if (styleForCellFlow->isHorizontalWritingMode())
288 return styleForCellFlow->isLeftToRightDirection() ? outerBorderStart() : outerBorderEnd();
289 return styleForCellFlow->isFlippedBlocksWritingMode() ? outerBorderAfter() : outerBorderBefore();
290}
291
292inline LayoutUnit RenderTableSection::outerBorderRight(const RenderStyle* styleForCellFlow) const
293{
294 if (styleForCellFlow->isHorizontalWritingMode())
295 return styleForCellFlow->isLeftToRightDirection() ? outerBorderEnd() : outerBorderStart();
296 return styleForCellFlow->isFlippedBlocksWritingMode() ? outerBorderBefore() : outerBorderAfter();
297}
298
299inline LayoutUnit RenderTableSection::outerBorderTop(const RenderStyle* styleForCellFlow) const
300{
301 if (styleForCellFlow->isHorizontalWritingMode())
302 return styleForCellFlow->isFlippedBlocksWritingMode() ? outerBorderAfter() : outerBorderBefore();
303 return styleForCellFlow->isLeftToRightDirection() ? outerBorderStart() : outerBorderEnd();
304}
305
306inline LayoutUnit RenderTableSection::outerBorderBottom(const RenderStyle* styleForCellFlow) const
307{
308 if (styleForCellFlow->isHorizontalWritingMode())
309 return styleForCellFlow->isFlippedBlocksWritingMode() ? outerBorderBefore() : outerBorderAfter();
310 return styleForCellFlow->isLeftToRightDirection() ? outerBorderEnd() : outerBorderStart();
311}
