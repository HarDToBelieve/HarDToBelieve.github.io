60void RenderTextControl::styleDidChange(StyleDifference diff, const RenderStyle* oldStyle)
61{
62 RenderBlockFlow::styleDidChange(diff, oldStyle);
63 auto innerText = innerTextElement();
64 if (!innerText)
65 return;
66 RenderTextControlInnerBlock* innerTextRenderer = innerText->renderer();
67 if (innerTextRenderer && oldStyle) {
68 // FIXME: The height property of the inner text block style may be mutated by RenderTextControlSingleLine::layout.
69 // See if the original has changed before setting it and triggering a layout.
70 auto newInnerTextStyle = textFormControlElement().createInnerTextStyle(style());
71 auto oldInnerTextStyle = textFormControlElement().createInnerTextStyle(*oldStyle);
72 if (newInnerTextStyle != oldInnerTextStyle)
73 innerTextRenderer->setStyle(WTFMove(newInnerTextStyle));
74 }
75 textFormControlElement().updatePlaceholderVisibility();
76}

83int RenderTextControl::textBlockLogicalWidth() const
84{
85 auto innerText = innerTextElement();
86 ASSERT(innerText);
87
88 LayoutUnit unitWidth = logicalWidth() - borderAndPaddingLogicalWidth();
89 if (innerText->renderer())
90 unitWidth -= innerText->renderBox()->paddingStart() + innerText->renderBox()->paddingEnd();
91
92 return unitWidth;
93}

101RenderBox::LogicalExtentComputedValues RenderTextControl::computeLogicalHeight(LayoutUnit logicalHeight, LayoutUnit logicalTop) const
102{
103 auto innerText = innerTextElement();
104 ASSERT(innerText);
105 if (RenderBox* innerTextBox = innerText->renderBox()) {
106 LayoutUnit nonContentHeight = innerTextBox->verticalBorderAndPaddingExtent() + innerTextBox->verticalMarginExtent();
107 logicalHeight = computeControlLogicalHeight(innerTextBox->lineHeight(true, HorizontalLine, PositionOfInteriorLineBoxes), nonContentHeight);
108
109 // We are able to have a horizontal scrollbar if the overflow style is scroll, or if its auto and there's no word wrap.
110 if ((isHorizontalWritingMode() && (style().overflowX() == Overflow::Scroll || (style().overflowX() == Overflow::Auto && innerText->renderer()->style().overflowWrap() == OverflowWrap::Normal)))
111 || (!isHorizontalWritingMode() && (style().overflowY() == Overflow::Scroll || (style().overflowY() == Overflow::Auto && innerText->renderer()->style().overflowWrap() == OverflowWrap::Normal))))
112 logicalHeight += scrollbarThickness();
113
114 // FIXME: The logical height of the inner text box should have been added
115 // before calling computeLogicalHeight to avoid this hack.
116 cacheIntrinsicContentLogicalHeightForFlexItem(logicalHeight);
117
118 logicalHeight += verticalBorderAndPaddingExtent();
119 }
120
121 return RenderBox::computeLogicalHeight(logicalHeight, logicalTop);
122}

124void RenderTextControl::hitInnerTextElement(HitTestResult& result, const LayoutPoint& pointInContainer, const LayoutPoint& accumulatedOffset)
125{
126 auto innerText = innerTextElement();
127 if (!innerText->renderer())
128 return;
129
130 LayoutPoint adjustedLocation = accumulatedOffset + location();
131 LayoutPoint localPoint = pointInContainer - toLayoutSize(adjustedLocation + innerText->renderBox()->location()) + toLayoutSize(scrollPosition());
132 result.setInnerNode(innerText.get());
133 result.setInnerNonSharedNode(innerText.get());
134 result.setLocalPoint(localPoint);
135}

137float RenderTextControl::getAverageCharWidth()
138{
139 float width;
140 if (style().fontCascade().fastAverageCharWidthIfAvailable(width))
141 return width;
142
143 const UChar ch = '0';
144 const String str = String(&ch, 1);
145 const FontCascade& font = style().fontCascade();
146 TextRun textRun = constructTextRun(str, style(), AllowRightExpansion);
147 return font.width(textRun);
148}

150float RenderTextControl::scaleEmToUnits(int x) const
151{
152 // This matches the unitsPerEm value for MS Shell Dlg and Courier New from the "head" font table.
153 float unitsPerEm = 2048.0f;
154 return roundf(style().fontCascade().size() * x / unitsPerEm);
155}

157void RenderTextControl::computeIntrinsicLogicalWidths(LayoutUnit& minLogicalWidth, LayoutUnit& maxLogicalWidth) const
158{
159 if (shouldApplySizeContainment(*this))
160 return;
161 // Use average character width. Matches IE.
162 maxLogicalWidth = preferredContentLogicalWidth(const_cast<RenderTextControl*>(this)->getAverageCharWidth());
163 if (RenderBox* innerTextRenderBox = innerTextElement()->renderBox())
164 maxLogicalWidth += innerTextRenderBox->paddingStart() + innerTextRenderBox->paddingEnd();
165 if (!style().logicalWidth().isPercentOrCalculated())
166 minLogicalWidth = maxLogicalWidth;
167}

169void RenderTextControl::computePreferredLogicalWidths()
170{
171 ASSERT(preferredLogicalWidthsDirty());
172
173 m_minPreferredLogicalWidth = 0;
174 m_maxPreferredLogicalWidth = 0;
175
176 if (style().logicalWidth().isFixed() && style().logicalWidth().value() >= 0)
177 m_minPreferredLogicalWidth = m_maxPreferredLogicalWidth = adjustContentBoxLogicalWidthForBoxSizing(style().logicalWidth());
178 else
179 computeIntrinsicLogicalWidths(m_minPreferredLogicalWidth, m_maxPreferredLogicalWidth);
180
181 RenderBox::computePreferredLogicalWidths(style().logicalMinWidth(), style().logicalMaxWidth(), borderAndPaddingLogicalWidth());
182
183 setPreferredLogicalWidthsDirty(false);
184}

192void RenderTextControl::layoutExcludedChildren(bool relayoutChildren)
193{
194 RenderBlockFlow::layoutExcludedChildren(relayoutChildren);
195
196 HTMLElement* placeholder = textFormControlElement().placeholderElement();
197 RenderElement* placeholderRenderer = placeholder ? placeholder->renderer() : 0;
198 if (!placeholderRenderer)
199 return;
200 placeholderRenderer->setIsExcludedFromNormalLayout(true);
201
202 if (relayoutChildren) {
203 // The markParents arguments should be false because this function is
204 // called from layout() of the parent and the placeholder layout doesn't
205 // affect the parent layout.
206 placeholderRenderer->setChildNeedsLayout(MarkOnlyThis);
207 }
208}
