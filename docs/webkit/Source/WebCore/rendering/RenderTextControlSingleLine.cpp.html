69void RenderTextControlSingleLine::centerRenderer(RenderBox& renderer) const

75static void resetOverriddenHeight(RenderBox* box, const RenderObject* ancestor)
76{
77 ASSERT(box != ancestor);
78 if (!box || box->style().logicalHeight().isAuto())
79 return; // Null box or its height was not overridden.
80 box->mutableStyle().setLogicalHeight(Length { LengthType::Auto });
81 for (RenderObject* renderer = box; renderer != ancestor; renderer = renderer->parent()) {
82 ASSERT(renderer);
83 renderer->setNeedsLayout(MarkOnlyThis);
84 }
85}

87void RenderTextControlSingleLine::layout()
88{
89 StackStats::LayoutCheckPoint layoutCheckPoint;
90
91 // FIXME: We should remove the height-related hacks in layout() and
92 // styleDidChange(). We need them because we want to:
93 // - Center the inner elements vertically if the input height is taller than
94 // the intrinsic height of the inner elements.
95 // - Shrink the heights of the inner elements if the input height is smaller
96 // than the intrinsic heights of the inner elements.
97 // - Make the height of the container element equal to the intrinsic height of
98 // the inner elements when the field has a strong password button.
99 //
100 // We don't honor padding and borders for textfields without decorations
101 // and type=search if the text height is taller than the contentHeight()
102 // because of compability.
103
104 RenderTextControlInnerBlock* innerTextRenderer = innerTextElement()->renderer();
105 RenderBox* innerBlockRenderer = innerBlockElement() ? innerBlockElement()->renderBox() : nullptr;
106 HTMLElement* container = containerElement();
107 RenderBox* containerRenderer = container ? container->renderBox() : nullptr;
108
109 // To ensure consistency between layouts, we need to reset any conditionally overridden height.
110 resetOverriddenHeight(innerTextRenderer, this);
111 resetOverriddenHeight(innerBlockRenderer, this);
112 resetOverriddenHeight(containerRenderer, this);
113
114 // Save the old size of the inner text (if we have one) as we will need to layout the placeholder
115 // and update selection if it changes. One way the size may change is if text decorations are
116 // toggled. For example, hiding and showing the caps lock indicator will cause a size change.
117 LayoutSize oldInnerTextSize;
118 if (innerTextRenderer)
119 oldInnerTextSize = innerTextRenderer->size();
120
121 RenderBlockFlow::layoutBlock(false);
122
123 // Set the text block height
124 LayoutUnit desiredLogicalHeight = textBlockLogicalHeight();
125 LayoutUnit logicalHeightLimit = logicalHeight();
126 LayoutUnit innerTextLogicalHeight;
127 if (innerTextRenderer)
128 innerTextLogicalHeight = innerTextRenderer->logicalHeight();
129 if (innerTextRenderer && innerTextLogicalHeight > logicalHeightLimit) {
130 if (desiredLogicalHeight != innerTextLogicalHeight)
131 setNeedsLayout(MarkOnlyThis);
132
133 innerTextRenderer->mutableStyle().setLogicalHeight(Length(desiredLogicalHeight, LengthType::Fixed));
134 innerTextRenderer->setNeedsLayout(MarkOnlyThis);
135 if (innerBlockRenderer) {
136 innerBlockRenderer->mutableStyle().setLogicalHeight(Length(desiredLogicalHeight, LengthType::Fixed));
137 innerBlockRenderer->setNeedsLayout(MarkOnlyThis);
138 }
139 innerTextLogicalHeight = desiredLogicalHeight;
140 }
141 // The container might be taller because of decoration elements.
142 LayoutUnit oldContainerLogicalTop;
143 if (containerRenderer) {
144 containerRenderer->layoutIfNeeded();
145 oldContainerLogicalTop = containerRenderer->logicalTop();
146 LayoutUnit containerLogicalHeight = containerRenderer->logicalHeight();
147 if (inputElement().hasAutoFillStrongPasswordButton() && innerTextRenderer && containerLogicalHeight != innerTextLogicalHeight) {
148 containerRenderer->mutableStyle().setLogicalHeight(Length { innerTextLogicalHeight, LengthType::Fixed });
149 setNeedsLayout(MarkOnlyThis);
150 } else if (containerLogicalHeight > logicalHeightLimit) {
151 containerRenderer->mutableStyle().setLogicalHeight(Length(logicalHeightLimit, LengthType::Fixed));
152 setNeedsLayout(MarkOnlyThis);
153 } else if (containerRenderer->logicalHeight() < contentLogicalHeight()) {
154 containerRenderer->mutableStyle().setLogicalHeight(Length(contentLogicalHeight(), LengthType::Fixed));
155 setNeedsLayout(MarkOnlyThis);
156 } else
157 containerRenderer->mutableStyle().setLogicalHeight(Length(containerLogicalHeight, LengthType::Fixed));
158 }
159
160 // If we need another layout pass, we have changed one of children's height so we need to relayout them.
161 if (needsLayout())
162 RenderBlockFlow::layoutBlock(true);
163
164 // Fix up the y-position of the container as it may have been flexed when the strong password or strong
165 // confirmation password button wraps to the next line.
166 if (inputElement().hasAutoFillStrongPasswordButton() && containerRenderer)
167 containerRenderer->setLogicalTop(oldContainerLogicalTop);
168
169 // Center the child block in the block progression direction (vertical centering for horizontal text fields).
170 if (!container && innerTextRenderer && innerTextRenderer->height() != contentLogicalHeight())
171 centerRenderer(*innerTextRenderer);
172 else if (container && containerRenderer && containerRenderer->height() != contentLogicalHeight())
173 centerRenderer(*containerRenderer);
174
175 bool innerTextSizeChanged = innerTextRenderer && innerTextRenderer->size() != oldInnerTextSize;
176
177 HTMLElement* placeholderElement = inputElement().placeholderElement();
178 if (RenderBox* placeholderBox = placeholderElement ? placeholderElement->renderBox() : 0) {
179 auto innerTextWidth = LayoutUnit { };
180 if (innerTextRenderer)
181 innerTextWidth = innerTextRenderer->logicalWidth();
182 placeholderBox->mutableStyle().setWidth(Length(innerTextWidth - placeholderBox->horizontalBorderAndPaddingExtent(), LengthType::Fixed));
183 bool neededLayout = placeholderBox->needsLayout();
184 bool placeholderBoxHadLayout = placeholderBox->everHadLayout();
185 if (innerTextSizeChanged) {
186 // The caps lock indicator was hidden. Layout the placeholder. Its layout does not affect its parent.
187 placeholderBox->setChildNeedsLayout(MarkOnlyThis);
188 }
189 placeholderBox->layoutIfNeeded();
190 auto placeholderTopLeft = containerRenderer ? containerRenderer->location() : LayoutPoint { };
191 auto* innerBlockRenderer = innerBlockElement() ? innerBlockElement()->renderBox() : nullptr;
192 if (innerBlockRenderer)
193 placeholderTopLeft += toLayoutSize(innerBlockRenderer->location());
194 if (innerTextRenderer)
195 placeholderTopLeft += toLayoutSize(innerTextRenderer->location());
196 placeholderBox->setLogicalLeft(placeholderTopLeft.x());
197 // Here the container box indicates the renderer that the placeholder content is aligned with (no parent and/or containing block relationship).
198 auto* containerBox = innerTextRenderer ? innerTextRenderer : innerBlockRenderer ? innerBlockRenderer : containerRenderer;
199 if (containerBox) {
200 // Center vertical align the placeholder content.
201 auto logicalTop = placeholderTopLeft.y() + (containerBox->logicalHeight() / 2 - placeholderBox->logicalHeight() / 2);
202 placeholderBox->setLogicalTop(logicalTop);
203 }
204 if (!placeholderBoxHadLayout && placeholderBox->checkForRepaintDuringLayout()) {
205 // This assumes a shadow tree without floats. If floats are added, the
206 // logic should be shared with RenderBlock::layoutBlockChild.
207 placeholderBox->repaint();
208 }
209 // The placeholder gets layout last, after the parent text control and its other children,
210 // so in order to get the correct overflow from the placeholder we need to recompute it now.
211 if (neededLayout)
212 computeOverflow(clientLogicalBottom());
213 }
214
215#if PLATFORM(IOS_FAMILY)
216 // FIXME: We should not be adjusting styles during layout. <rdar://problem/7675493>
217 if (inputElement().isSearchField())
218 RenderThemeIOS::adjustRoundBorderRadius(mutableStyle(), *this);
219#endif
220 if (innerTextSizeChanged && frame().selection().isFocusedAndActive() && document().focusedElement() == &inputElement()) {
221 // The caps lock indicator was hidden or shown. If it is now visible then it may be occluding
222 // the current selection (say, the caret was after the last character in the text field).
223 // Schedule an update and reveal of the current selection.
224 frame().selection().setNeedsSelectionUpdate(FrameSelection::RevealSelectionAfterUpdate::Forced);
225 }
226}

228bool RenderTextControlSingleLine::nodeAtPoint(const HitTestRequest& request, HitTestResult& result, const HitTestLocation& locationInContainer, const LayoutPoint& accumulatedOffset, HitTestAction hitTestAction)
229{
230 if (!RenderTextControl::nodeAtPoint(request, result, locationInContainer, accumulatedOffset, hitTestAction))
231 return false;
232
233 // Say that we hit the inner text element if
234 // - we hit a node inside the inner text element,
235 // - we hit the <input> element (e.g. we're over the border or padding), or
236 // - we hit regions not in any decoration buttons.
237 HTMLElement* container = containerElement();
238 if (result.innerNode()->isDescendantOf(innerTextElement().get()) || result.innerNode() == &inputElement() || (container && container == result.innerNode())) {
239 LayoutPoint pointInParent = locationInContainer.point();
240 if (container && innerBlockElement()) {
241 if (innerBlockElement()->renderBox())
242 pointInParent -= toLayoutSize(innerBlockElement()->renderBox()->location());
243 if (container->renderBox())
244 pointInParent -= toLayoutSize(container->renderBox()->location());
245 }
246 hitInnerTextElement(result, pointInParent, accumulatedOffset);
247 }
248 return true;
249}

251void RenderTextControlSingleLine::styleDidChange(StyleDifference diff, const RenderStyle* oldStyle)
252{
253 RenderTextControl::styleDidChange(diff, oldStyle);
254
255 // We may have set the width and the height in the old style in layout().
256 // Reset them now to avoid getting a spurious layout hint.
257 HTMLElement* innerBlock = innerBlockElement();
258 if (auto* innerBlockRenderer = innerBlock ? innerBlock->renderer() : nullptr) {
259 innerBlockRenderer->mutableStyle().setHeight(Length());
260 innerBlockRenderer->mutableStyle().setWidth(Length());
261 }
262 HTMLElement* container = containerElement();
263 if (auto* containerRenderer = container ? container->renderer() : nullptr) {
264 containerRenderer->mutableStyle().setHeight(Length());
265 containerRenderer->mutableStyle().setWidth(Length());
266 }
267 if (diff == StyleDifference::Layout) {
268 if (auto innerTextRenderer = innerTextElement()->renderer())
269 innerTextRenderer->setNeedsLayout(MarkContainingBlockChain);
270 if (auto* placeholder = inputElement().placeholderElement()) {
271 if (placeholder->renderer())
272 placeholder->renderer()->setNeedsLayout(MarkContainingBlockChain);
273 }
274 }
275 setHasOverflowClip(false);
276}

284LayoutRect RenderTextControlSingleLine::controlClipRect(const LayoutPoint& additionalOffset) const
285{
286 ASSERT(hasControlClip());
287 LayoutRect clipRect = contentBoxRect();
288 if (containerElement()->renderBox())
289 clipRect = unionRect(clipRect, containerElement()->renderBox()->frameRect());
290 clipRect.moveBy(additionalOffset);
291 return clipRect;
292}

294float RenderTextControlSingleLine::getAverageCharWidth()
295{
296#if !PLATFORM(IOS_FAMILY)
297 // Since Lucida Grande is the default font, we want this to match the width
298 // of MS Shell Dlg, the default font for textareas in Firefox, Safari Win and
299 // IE for some encodings (in IE, the default font is encoding specific).
300 // 901 is the avgCharWidth value in the OS/2 table for MS Shell Dlg.
301 if (style().fontCascade().firstFamily() == "Lucida Grande")
302 return scaleEmToUnits(901);
303#endif
304
305 return RenderTextControl::getAverageCharWidth();
306}

308LayoutUnit RenderTextControlSingleLine::preferredContentLogicalWidth(float charWidth) const
309{
310 int factor = 0;
311 bool includesDecoration = inputElement().sizeShouldIncludeDecoration(factor);
312 if (factor <= 0)
313 factor = 20;
314
315 LayoutUnit result = LayoutUnit::fromFloatCeil(charWidth * factor);
316
317 float maxCharWidth = 0.f;
318
319#if !PLATFORM(IOS_FAMILY)
320 const AtomString& family = style().fontCascade().firstFamily();
321 // Since Lucida Grande is the default font, we want this to match the width
322 // of MS Shell Dlg, the default font for textareas in Firefox, Safari Win and
323 // IE for some encodings (in IE, the default font is encoding specific).
324 // 4027 is the (xMax - xMin) value in the "head" font table for MS Shell Dlg.
325 if (family == "Lucida Grande")
326 maxCharWidth = scaleEmToUnits(4027);
327 else if (style().fontCascade().hasValidAverageCharWidth())
328 maxCharWidth = roundf(style().fontCascade().primaryFont().maxCharWidth());
329#endif
330
331 // For text inputs, IE adds some extra width.
332 if (maxCharWidth > 0.f)
333 result += maxCharWidth - charWidth;
334
335 if (includesDecoration)
336 result += inputElement().decorationWidth();
337
338 if (auto* innerRenderer = innerTextElement()->renderer())
339 result += innerRenderer->endPaddingWidthForCaret();
340
341 return result;
342}

399bool RenderTextControlSingleLine::scroll(ScrollDirection direction, ScrollGranularity granularity, float multiplier, Element** stopElement, RenderBox* startBox, const IntPoint& wheelEventAbsolutePoint)
400{
401 auto* renderer = innerTextElement()->renderer();
402 if (!renderer)
403 return false;
404 auto* scrollableArea = renderer->layer() ? renderer->layer()->scrollableArea() : nullptr;
405 if (scrollableArea && scrollableArea->scroll(direction, granularity, multiplier))
406 return true;
407 return RenderBlockFlow::scroll(direction, granularity, multiplier, stopElement, startBox, wheelEventAbsolutePoint);
408}

410bool RenderTextControlSingleLine::logicalScroll(ScrollLogicalDirection direction, ScrollGranularity granularity, float multiplier, Element** stopElement)
411{
412 auto* layer = innerTextElement()->renderer()->layer();
413 auto* scrollableArea = layer ? layer->scrollableArea() : nullptr;
414 if (scrollableArea && scrollableArea->scroll(logicalToPhysical(direction, style().isHorizontalWritingMode(), style().isFlippedBlocksWritingMode()), granularity, multiplier))
415 return true;
416 return RenderBlockFlow::logicalScroll(direction, granularity, multiplier, stopElement);
417}
