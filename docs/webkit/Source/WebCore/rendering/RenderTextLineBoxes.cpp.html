39RenderTextLineBoxes::RenderTextLineBoxes()
40 : m_first(nullptr)
41 , m_last(nullptr)
42{
43}

45LegacyInlineTextBox* RenderTextLineBoxes::createAndAppendLineBox(RenderText& renderText)
46{
47 auto textBox = renderText.createTextBox();
48 if (!m_first) {
49 m_first = textBox.get();
50 m_last = textBox.get();
51 } else {
52 m_last->setNextTextBox(textBox.get());
53 textBox->setPreviousTextBox(m_last);
54 m_last = textBox.get();
55 }
56 return textBox.release();
57}

59void RenderTextLineBoxes::extract(LegacyInlineTextBox& box)
60{
61 checkConsistency();
62
63 m_last = box.prevTextBox();
64 if (&box == m_first)
65 m_first = nullptr;
66 if (box.prevTextBox())
67 box.prevTextBox()->setNextTextBox(nullptr);
68 box.setPreviousTextBox(nullptr);
69 for (auto* current = &box; current; current = current->nextTextBox())
70 current->setExtracted();
71
72 checkConsistency();
73}
74
75void RenderTextLineBoxes::attach(LegacyInlineTextBox& box)
76{
77 checkConsistency();
78
79 if (m_last) {
80 m_last->setNextTextBox(&box);
81 box.setPreviousTextBox(m_last);
82 } else
83 m_first = &box;
84 LegacyInlineTextBox* last = nullptr;
85 for (auto* current = &box; current; current = current->nextTextBox()) {
86 current->setExtracted(false);
87 last = current;
88 }
89 m_last = last;
90
91 checkConsistency();
92}
93
94void RenderTextLineBoxes::remove(LegacyInlineTextBox& box)
95{
96 checkConsistency();
97
98 if (&box == m_first)
99 m_first = box.nextTextBox();
100 if (&box == m_last)
101 m_last = box.prevTextBox();
102 if (box.nextTextBox())
103 box.nextTextBox()->setPreviousTextBox(box.prevTextBox());
104 if (box.prevTextBox())
105 box.prevTextBox()->setNextTextBox(box.nextTextBox());
106
107 checkConsistency();
108}

110void RenderTextLineBoxes::removeAllFromParent(RenderText& renderer)
111{
112 if (!m_first) {
113 if (renderer.parent())
114 renderer.parent()->dirtyLinesFromChangedChild(renderer);
115 return;
116 }
117 for (auto* box = m_first; box; box = box->nextTextBox())
118 box->removeFromParent();
119}

121void RenderTextLineBoxes::deleteAll()
122{
123 if (!m_first)
124 return;
125 LegacyInlineTextBox* next;
126 for (auto* current = m_first; current; current = next) {
127 next = current->nextTextBox();
128 delete current;
129 }
130 m_first = nullptr;
131 m_last = nullptr;
132}

134LegacyInlineTextBox* RenderTextLineBoxes::findNext(int offset, int& position) const
135{
136 if (!m_first)
137 return nullptr;
138 // FIXME: This looks buggy. The function is only used for debugging purposes.
139 auto current = m_first;
140 int currentOffset = current->len();
141 while (offset > currentOffset && current->nextTextBox()) {
142 current = current->nextTextBox();
143 currentOffset = current->start() + current->len();
144 }
145 // we are now in the correct text run
146 position = (offset > currentOffset ? current->len() : current->len() - (currentOffset - offset));
147 return current;
148}

150LayoutRect RenderTextLineBoxes::visualOverflowBoundingBox(const RenderText& renderer) const
151{
152 if (!m_first)
153 return LayoutRect();
154
155 // Return the width of the minimal left side and the maximal right side.
156 auto logicalLeftSide = LayoutUnit::max();
157 auto logicalRightSide = LayoutUnit::min();
158 for (auto* current = m_first; current; current = current->nextTextBox()) {
159 logicalLeftSide = std::min(logicalLeftSide, current->logicalLeftVisualOverflow());
160 logicalRightSide = std::max(logicalRightSide, current->logicalRightVisualOverflow());
161 }
162
163 auto logicalTop = m_first->logicalTopVisualOverflow();
164 auto logicalWidth = logicalRightSide - logicalLeftSide;
165 auto logicalHeight = m_last->logicalBottomVisualOverflow() - logicalTop;
166
167 LayoutRect rect(logicalLeftSide, logicalTop, logicalWidth, logicalHeight);
168 if (!renderer.style().isHorizontalWritingMode())
169 rect = rect.transposedRect();
170 return rect;
171}

173void RenderTextLineBoxes::dirtyAll()

179bool RenderTextLineBoxes::dirtyRange(RenderText& renderer, unsigned start, unsigned end, int lengthDelta)
180{
181 LegacyRootInlineBox* firstRootBox = nullptr;
182 LegacyRootInlineBox* lastRootBox = nullptr;
183
184 // Dirty all text boxes that include characters in between offset and offset+len.
185 bool dirtiedLines = false;
186 for (auto* current = m_first; current; current = current->nextTextBox()) {
187 // FIXME: This shouldn't rely on the end of a dirty line box. See https://bugs.webkit.org/show_bug.cgi?id=97264
188 // Text run is entirely before the affected range.
189 if (current->end() <= start)
190 continue;
191 // Text run is entirely after the affected range.
192 if (current->start() >= end) {
193 current->offsetRun(lengthDelta);
194 auto& rootBox = current->root();
195 if (!firstRootBox) {
196 firstRootBox = &rootBox;
197 if (!dirtiedLines) {
198 // The affected area was in between two runs. Mark the root box of the run after the affected area as dirty.
199 firstRootBox->markDirty();
200 dirtiedLines = true;
201 }
202 }
203 lastRootBox = &rootBox;
204 continue;
205 }
206 if (current->end() > start && current->end() <= end) {
207 // Text run overlaps with the left end of the affected range.
208 current->dirtyLineBoxes();
209 dirtiedLines = true;
210 continue;
211 }
212 if (current->start() <= start && current->end() >= end) {
213 // Text run subsumes the affected range.
214 current->dirtyLineBoxes();
215 dirtiedLines = true;
216 continue;
217 }
218 if (current->start() < end && current->end() >= end) {
219 // Text run overlaps with right end of the affected range.
220 current->dirtyLineBoxes();
221 dirtiedLines = true;
222 continue;
223 }
224 }
225
226 // Now we have to walk all of the clean lines and adjust their cached line break information
227 // to reflect our updated offsets.
228 if (lastRootBox)
229 lastRootBox = lastRootBox->nextRootBox();
230 if (firstRootBox) {
231 auto previousRootBox = firstRootBox->prevRootBox();
232 if (previousRootBox)
233 firstRootBox = previousRootBox;
234 } else if (m_last) {
235 ASSERT(!lastRootBox);
236 firstRootBox = &m_last->root();
237 firstRootBox->markDirty();
238 dirtiedLines = true;
239 }
240
241 for (auto* current = firstRootBox; current && current != lastRootBox; current = current->nextRootBox()) {
242 auto lineBreakPos = current->lineBreakPos();
243 if (current->lineBreakObj() == &renderer && (lineBreakPos > end || (start != end && lineBreakPos == end)))
244 current->setLineBreakPos(current->lineBreakPos() + lengthDelta);
245 }
246
247 // If the text node is empty, dirty the line where new text will be inserted.
248 if (!m_first && renderer.parent()) {
249 renderer.parent()->dirtyLinesFromChangedChild(renderer);
250 dirtiedLines = true;
251 }
252 return dirtiedLines;
253}

255inline void RenderTextLineBoxes::checkConsistency() const
256{
257#if ASSERT_ENABLED
258#ifdef CHECK_CONSISTENCY
259 const LegacyInlineTextBox* prev = nullptr;
260 for (auto* child = m_first; child; child = child->nextTextBox()) {
261 ASSERT(child->renderer() == this);
262 ASSERT(child->prevTextBox() == prev);
263 prev = child;
264 }
265 ASSERT(prev == m_last);
266#endif
267#endif // ASSERT_ENABLED
268}
