78void RenderTheme::adjustStyle(RenderStyle& style, const Element* element, const RenderStyle* userAgentAppearanceStyle)
79{
80 // Force inline and table display styles to be inline-block (except for table- which is block)
81 ControlPart part = style.appearance();
82 if (style.display() == DisplayType::Inline || style.display() == DisplayType::InlineTable || style.display() == DisplayType::TableRowGroup
83 || style.display() == DisplayType::TableHeaderGroup || style.display() == DisplayType::TableFooterGroup
84 || style.display() == DisplayType::TableRow || style.display() == DisplayType::TableColumnGroup || style.display() == DisplayType::TableColumn
85 || style.display() == DisplayType::TableCell || style.display() == DisplayType::TableCaption)
86 style.setEffectiveDisplay(DisplayType::InlineBlock);
87 else if (style.display() == DisplayType::ListItem || style.display() == DisplayType::Table)
88 style.setEffectiveDisplay(DisplayType::Block);
89
90 if (userAgentAppearanceStyle && isControlStyled(style, *userAgentAppearanceStyle)) {
91 switch (part) {
92 case MenulistPart:
93 style.setAppearance(MenulistButtonPart);
94 part = MenulistButtonPart;
95 break;
96 case TextFieldPart:
97 adjustTextFieldStyle(style, element);
98 FALLTHROUGH;
99 default:
100 style.setAppearance(NoControlPart);
101 break;
102 }
103 }
104
105 if (!style.hasAppearance())
106 return;
107
108 if (!supportsBoxShadow(style))
109 style.setBoxShadow(nullptr);
110
111#if USE(NEW_THEME)
112 switch (part) {
113 case CheckboxPart:
114 case InnerSpinButtonPart:
115 case RadioPart:
116 case PushButtonPart:
117 case SquareButtonPart:
118#if ENABLE(INPUT_TYPE_COLOR)
119 case ColorWellPart:
120#endif
121 case DefaultButtonPart:
122 case ButtonPart: {
123 // Border
124 LengthBox borderBox(style.borderTopWidth(), style.borderRightWidth(), style.borderBottomWidth(), style.borderLeftWidth());
125 borderBox = Theme::singleton().controlBorder(part, style.fontCascade(), borderBox, style.effectiveZoom());
126 if (borderBox.top().value() != static_cast<int>(style.borderTopWidth())) {
127 if (borderBox.top().value())
128 style.setBorderTopWidth(borderBox.top().value());
129 else
130 style.resetBorderTop();
131 }
132 if (borderBox.right().value() != static_cast<int>(style.borderRightWidth())) {
133 if (borderBox.right().value())
134 style.setBorderRightWidth(borderBox.right().value());
135 else
136 style.resetBorderRight();
137 }
138 if (borderBox.bottom().value() != static_cast<int>(style.borderBottomWidth())) {
139 style.setBorderBottomWidth(borderBox.bottom().value());
140 if (borderBox.bottom().value())
141 style.setBorderBottomWidth(borderBox.bottom().value());
142 else
143 style.resetBorderBottom();
144 }
145 if (borderBox.left().value() != static_cast<int>(style.borderLeftWidth())) {
146 style.setBorderLeftWidth(borderBox.left().value());
147 if (borderBox.left().value())
148 style.setBorderLeftWidth(borderBox.left().value());
149 else
150 style.resetBorderLeft();
151 }
152
153 // Padding
154 LengthBox paddingBox = Theme::singleton().controlPadding(part, style.fontCascade(), style.paddingBox(), style.effectiveZoom());
155 if (paddingBox != style.paddingBox())
156 style.setPaddingBox(WTFMove(paddingBox));
157
158 // Whitespace
159 if (Theme::singleton().controlRequiresPreWhiteSpace(part))
160 style.setWhiteSpace(WhiteSpace::Pre);
161
162 // Width / Height
163 // The width and height here are affected by the zoom.
164 // FIXME: Check is flawed, since it doesn't take min-width/max-width into account.
165 LengthSize controlSize = Theme::singleton().controlSize(part, style.fontCascade(), { style.width(), style.height() }, style.effectiveZoom());
166 if (controlSize.width != style.width())
167 style.setWidth(WTFMove(controlSize.width));
168 if (controlSize.height != style.height())
169 style.setHeight(WTFMove(controlSize.height));
170
171 // Min-Width / Min-Height
172 LengthSize minControlSize = Theme::singleton().minimumControlSize(part, style.fontCascade(), { style.minWidth(), style.minHeight() }, { style.width(), style.height() }, style.effectiveZoom());
173 if (minControlSize.width != style.minWidth())
174 style.setMinWidth(WTFMove(minControlSize.width));
175 if (minControlSize.height != style.minHeight())
176 style.setMinHeight(WTFMove(minControlSize.height));
177
178 // Font
179 if (auto themeFont = Theme::singleton().controlFont(part, style.fontCascade(), style.effectiveZoom())) {
180 // If overriding the specified font with the theme font, also override the line height with the standard line height.
181 style.setLineHeight(RenderStyle::initialLineHeight());
182 if (style.setFontDescription(WTFMove(themeFont.value())))
183 style.fontCascade().update(nullptr);
184 }
185
186 // Special style that tells enabled default buttons in active windows to use the ActiveButtonText color.
187 // The active window part of the test has to be done at paint time since it's not triggered by a style change.
188 style.setInsideDefaultButton(part == DefaultButtonPart && element && !element->isDisabledFormControl());
189 break;
190 }
191 default:
192 break;
193 }
194#endif
195
196 // Call the appropriate style adjustment method based off the appearance value.
197 switch (style.appearance()) {
198#if !USE(NEW_THEME)
199 case CheckboxPart:
200 return adjustCheckboxStyle(style, element);
201 case RadioPart:
202 return adjustRadioStyle(style, element);
203#if ENABLE(INPUT_TYPE_COLOR)
204 case ColorWellPart:
205 return adjustColorWellStyle(style, element);
206#endif
207 case PushButtonPart:
208 case SquareButtonPart:
209 case DefaultButtonPart:
210 case ButtonPart:
211 return adjustButtonStyle(style, element);
212 case InnerSpinButtonPart:
213 return adjustInnerSpinButtonStyle(style, element);
214#endif
215 case TextFieldPart:
216 return adjustTextFieldStyle(style, element);
217 case TextAreaPart:
218 return adjustTextAreaStyle(style, element);
219 case MenulistPart:
220 return adjustMenuListStyle(style, element);
221 case MenulistButtonPart:
222 return adjustMenuListButtonStyle(style, element);
223 case MediaPlayButtonPart:
224 case MediaCurrentTimePart:
225 case MediaTimeRemainingPart:
226 case MediaEnterFullscreenButtonPart:
227 case MediaExitFullscreenButtonPart:
228 case MediaMuteButtonPart:
229 case MediaVolumeSliderContainerPart:
230 return adjustMediaControlStyle(style, element);
231 case MediaSliderPart:
232 case MediaVolumeSliderPart:
233 case MediaFullScreenVolumeSliderPart:
234 case SliderHorizontalPart:
235 case SliderVerticalPart:
236 return adjustSliderTrackStyle(style, element);
237 case SliderThumbHorizontalPart:
238 case SliderThumbVerticalPart:
239 return adjustSliderThumbStyle(style, element);
240 case SearchFieldPart:
241 return adjustSearchFieldStyle(style, element);
242 case SearchFieldCancelButtonPart:
243 return adjustSearchFieldCancelButtonStyle(style, element);
244 case SearchFieldDecorationPart:
245 case SearchFieldResultsDecorationPart:
246 case SearchFieldResultsButtonPart:
247 return adjustSearchFieldDecorationStyle(style, element);
248 case ProgressBarPart:
249 return adjustProgressBarStyle(style, element);
250 case MeterPart:
251 case RelevancyLevelIndicatorPart:
252 case ContinuousCapacityLevelIndicatorPart:
253 case DiscreteCapacityLevelIndicatorPart:
254 case RatingLevelIndicatorPart:
255 return adjustMeterStyle(style, element);
256 case CapsLockIndicatorPart:
257 return adjustCapsLockIndicatorStyle(style, element);
258#if ENABLE(APPLE_PAY)
259 case ApplePayButtonPart:
260 return adjustApplePayButtonStyle(style, element);
261#endif
262#if ENABLE(ATTACHMENT_ELEMENT)
263 case AttachmentPart:
264 case BorderlessAttachmentPart:
265 return adjustAttachmentStyle(style, element);
266#endif
267#if ENABLE(DATALIST_ELEMENT)
268 case ListButtonPart:
269 return adjustListButtonStyle(style, element);
270#endif
271 default:
272 break;
273 }

295bool RenderTheme::paint(const RenderBox& box, ControlStates& controlStates, const PaintInfo& paintInfo, const LayoutRect& rect)
296{
297 // If painting is disabled, but we aren't updating control tints, then just bail.
298 // If we are updating control tints, just schedule a repaint if the theme supports tinting
299 // for that control.
300 if (paintInfo.context().invalidatingControlTints()) {
301 if (controlSupportsTints(box))
302 box.repaint();
303 return false;
304 }
305 if (paintInfo.context().paintingDisabled())
306 return false;
307
308 if (UNLIKELY(!canPaint(paintInfo, box.settings())))
309 return false;
310
311 ControlPart part = box.style().appearance();
312 IntRect integralSnappedRect = snappedIntRect(rect);
313 float deviceScaleFactor = box.document().deviceScaleFactor();
314 FloatRect devicePixelSnappedRect = snapRectToDevicePixels(rect, deviceScaleFactor);
315
316#if USE(NEW_THEME)
317 float pageScaleFactor = box.page().pageScaleFactor();
318
319 switch (part) {
320 case CheckboxPart:
321 case RadioPart:
322 case PushButtonPart:
323 case SquareButtonPart:
324#if ENABLE(INPUT_TYPE_COLOR)
325 case ColorWellPart:
326#endif
327 case DefaultButtonPart:
328 case ButtonPart:
329 case InnerSpinButtonPart:
330 updateControlStatesForRenderer(box, controlStates);
331 Theme::singleton().paint(part, controlStates, paintInfo.context(), devicePixelSnappedRect, box.style().effectiveZoom(), &box.view().frameView(), deviceScaleFactor, pageScaleFactor, box.document().useSystemAppearance(), box.useDarkAppearance());
332 return false;
333 default:
334 break;
335 }
336#else
337 UNUSED_PARAM(controlStates);
338#endif
339
340 // Call the appropriate paint method based off the appearance value.
341 switch (part) {
342#if !USE(NEW_THEME)
343 case CheckboxPart:
344 return paintCheckbox(box, paintInfo, devicePixelSnappedRect);
345 case RadioPart:
346 return paintRadio(box, paintInfo, devicePixelSnappedRect);
347#if ENABLE(INPUT_TYPE_COLOR)
348 case ColorWellPart:
349 return paintColorWell(box, paintInfo, integralSnappedRect);
350#endif
351 case PushButtonPart:
352 case SquareButtonPart:
353 case DefaultButtonPart:
354 case ButtonPart:
355 return paintButton(box, paintInfo, integralSnappedRect);
356 case InnerSpinButtonPart:
357 return paintInnerSpinButton(box, paintInfo, integralSnappedRect);
358#endif
359 case MenulistPart:
360 return paintMenuList(box, paintInfo, devicePixelSnappedRect);
361 case MeterPart:
362 case RelevancyLevelIndicatorPart:
363 case ContinuousCapacityLevelIndicatorPart:
364 case DiscreteCapacityLevelIndicatorPart:
365 case RatingLevelIndicatorPart:
366 return paintMeter(box, paintInfo, integralSnappedRect);
367 case ProgressBarPart:
368 return paintProgressBar(box, paintInfo, integralSnappedRect);
369 case SliderHorizontalPart:
370 case SliderVerticalPart:
371 return paintSliderTrack(box, paintInfo, integralSnappedRect);
372 case SliderThumbHorizontalPart:
373 case SliderThumbVerticalPart:
374 return paintSliderThumb(box, paintInfo, integralSnappedRect);
375 case MediaEnterFullscreenButtonPart:
376 case MediaExitFullscreenButtonPart:
377 return paintMediaFullscreenButton(box, paintInfo, integralSnappedRect);
378 case MediaPlayButtonPart:
379 return paintMediaPlayButton(box, paintInfo, integralSnappedRect);
380 case MediaOverlayPlayButtonPart:
381 return paintMediaOverlayPlayButton(box, paintInfo, integralSnappedRect);
382 case MediaMuteButtonPart:
383 return paintMediaMuteButton(box, paintInfo, integralSnappedRect);
384 case MediaSeekBackButtonPart:
385 return paintMediaSeekBackButton(box, paintInfo, integralSnappedRect);
386 case MediaSeekForwardButtonPart:
387 return paintMediaSeekForwardButton(box, paintInfo, integralSnappedRect);
388 case MediaRewindButtonPart:
389 return paintMediaRewindButton(box, paintInfo, integralSnappedRect);
390 case MediaReturnToRealtimeButtonPart:
391 return paintMediaReturnToRealtimeButton(box, paintInfo, integralSnappedRect);
392 case MediaToggleClosedCaptionsButtonPart:
393 return paintMediaToggleClosedCaptionsButton(box, paintInfo, integralSnappedRect);
394 case MediaSliderPart:
395 return paintMediaSliderTrack(box, paintInfo, integralSnappedRect);
396 case MediaSliderThumbPart:
397 return paintMediaSliderThumb(box, paintInfo, integralSnappedRect);
398 case MediaVolumeSliderMuteButtonPart:
399 return paintMediaMuteButton(box, paintInfo, integralSnappedRect);
400 case MediaVolumeSliderContainerPart:
401 return paintMediaVolumeSliderContainer(box, paintInfo, integralSnappedRect);
402 case MediaVolumeSliderPart:
403 return paintMediaVolumeSliderTrack(box, paintInfo, integralSnappedRect);
404 case MediaVolumeSliderThumbPart:
405 return paintMediaVolumeSliderThumb(box, paintInfo, integralSnappedRect);
406 case MediaFullScreenVolumeSliderPart:
407 return paintMediaFullScreenVolumeSliderTrack(box, paintInfo, integralSnappedRect);
408 case MediaFullScreenVolumeSliderThumbPart:
409 return paintMediaFullScreenVolumeSliderThumb(box, paintInfo, integralSnappedRect);
410 case MediaTimeRemainingPart:
411 return paintMediaTimeRemaining(box, paintInfo, integralSnappedRect);
412 case MediaCurrentTimePart:
413 return paintMediaCurrentTime(box, paintInfo, integralSnappedRect);
414 case MediaControlsBackgroundPart:
415 return paintMediaControlsBackground(box, paintInfo, integralSnappedRect);
416 case MenulistButtonPart:
417 case TextFieldPart:
418 case TextAreaPart:
419 case ListboxPart:
420 return true;
421 case SearchFieldPart:
422 return paintSearchField(box, paintInfo, integralSnappedRect);
423 case SearchFieldCancelButtonPart:
424 return paintSearchFieldCancelButton(box, paintInfo, integralSnappedRect);
425 case SearchFieldDecorationPart:
426 return paintSearchFieldDecorationPart(box, paintInfo, integralSnappedRect);
427 case SearchFieldResultsDecorationPart:
428 return paintSearchFieldResultsDecorationPart(box, paintInfo, integralSnappedRect);
429 case SearchFieldResultsButtonPart:
430 return paintSearchFieldResultsButton(box, paintInfo, integralSnappedRect);
431 case CapsLockIndicatorPart:
432 return paintCapsLockIndicator(box, paintInfo, integralSnappedRect);
433#if ENABLE(APPLE_PAY)
434 case ApplePayButtonPart:
435 return paintApplePayButton(box, paintInfo, integralSnappedRect);
436#endif
437#if ENABLE(ATTACHMENT_ELEMENT)
438 case AttachmentPart:
439 case BorderlessAttachmentPart:
440 return paintAttachment(box, paintInfo, integralSnappedRect);
441#endif
442 default:
443 break;
444 }
445
446 return true; // We don't support the appearance, so let the normal background/border paint.
447}

574String RenderTheme::formatMediaControlsTime(float time) const
575{
576 if (!std::isfinite(time))
577 time = 0;
578 // FIXME: Seems like it would be better to use std::lround here.
579 int seconds = static_cast<int>(std::abs(time));
580 int hours = seconds / (60 * 60);
581 int minutes = (seconds / 60) % 60;
582 seconds %= 60;
583 if (hours)
584 return makeString((time < 0 ? "-" : ""), hours, ':', pad('0', 2, minutes), ':', pad('0', 2, seconds));
585 return makeString((time < 0 ? "-" : ""), pad('0', 2, minutes), ':', pad('0', 2, seconds));
586}

598LayoutPoint RenderTheme::volumeSliderOffsetFromMuteButton(const RenderBox& muteButtonBox, const LayoutSize& size) const
599{
600 LayoutUnit y = -size.height();
601 FloatPoint absPoint = muteButtonBox.localToAbsolute(FloatPoint(muteButtonBox.offsetLeft(), y), IsFixed | UseTransforms);
602 if (absPoint.y() < 0)
603 y = muteButtonBox.height();
604 return LayoutPoint(0_lu, y);
605}

785bool RenderTheme::stateChanged(const RenderObject& o, ControlStates::States state) const
786{
787 // Default implementation assumes the controls don't respond to changes in :hover state
788 if (state == ControlStates::States::Hovered && !supportsHover(o.style()))
789 return false;
790
791 // Assume pressed state is only responded to if the control is enabled.
792 if (state == ControlStates::States::Pressed && !isEnabled(o))
793 return false;
794
795 // Repaint the control.
796 o.repaint();
797 return true;
798}

800void RenderTheme::updateControlStatesForRenderer(const RenderBox& box, ControlStates& controlStates) const
801{
802 ControlStates newStates = extractControlStatesForRenderer(box);
803 controlStates.setStates(newStates.states());
804 if (isFocused(box))
805 controlStates.setTimeSinceControlWasFocused(box.page().focusController().timeSinceFocusWasSet());
806}
807
808OptionSet<ControlStates::States> RenderTheme::extractControlStatesForRenderer(const RenderObject& o) const
809{
810 OptionSet<ControlStates::States> states;
811 if (isHovered(o)) {
812 states.add(ControlStates::States::Hovered);
813 if (isSpinUpButtonPartHovered(o))
814 states.add(ControlStates::States::SpinUp);
815 }
816 if (isPressed(o)) {
817 states.add(ControlStates::States::Pressed);
818 if (isSpinUpButtonPartPressed(o))
819 states.add(ControlStates::States::SpinUp);
820 }
821 if (isFocused(o) && o.style().outlineStyleIsAuto() == OutlineIsAuto::On)
822 states.add(ControlStates::States::Focused);
823 if (isEnabled(o))
824 states.add(ControlStates::States::Enabled);
825 if (isChecked(o))
826 states.add(ControlStates::States::Checked);
827 if (isDefault(o))
828 states.add(ControlStates::States::Default);
829 if (!isActive(o))
830 states.add(ControlStates::States::WindowInactive);
831 if (isIndeterminate(o))
832 states.add(ControlStates::States::Indeterminate);
833 if (isPresenting(o))
834 states.add(ControlStates::States::Presenting);
835 return states;
836}

1080void RenderTheme::paintSliderTicks(const RenderObject& o, const PaintInfo& paintInfo, const FloatRect& rect)
1081{
1082 if (!is<HTMLInputElement>(o.node()))
1083 return;
1084
1085 auto& input = downcast<HTMLInputElement>(*o.node());
1086 if (!input.isRangeControl())
1087 return;
1088
1089 auto dataList = input.dataList();
1090 if (!dataList)
1091 return;
1092
1093 double min = input.minimum();
1094 double max = input.maximum();
1095 ControlPart part = o.style().appearance();
1096 // We don't support ticks on alternate sliders like MediaVolumeSliders.
1097 if (part != SliderHorizontalPart && part != SliderVerticalPart)
1098 return;
1099 bool isHorizontal = part == SliderHorizontalPart;
1100
1101 IntSize thumbSize;
1102 const RenderObject* thumbRenderer = input.sliderThumbElement()->renderer();
1103 if (thumbRenderer) {
1104 const RenderStyle& thumbStyle = thumbRenderer->style();
1105 int thumbWidth = thumbStyle.width().intValue();
1106 int thumbHeight = thumbStyle.height().intValue();
1107 thumbSize.setWidth(isHorizontal ? thumbWidth : thumbHeight);
1108 thumbSize.setHeight(isHorizontal ? thumbHeight : thumbWidth);
1109 }
1110
1111 IntSize tickSize = sliderTickSize();
1112 float zoomFactor = o.style().effectiveZoom();
1113 FloatRect tickRect;
1114 int tickRegionSideMargin = 0;
1115 int tickRegionWidth = 0;
1116 IntRect trackBounds;
1117 RenderObject* trackRenderer = input.sliderTrackElement()->renderer();
1118 // We can ignoring transforms because transform is handled by the graphics context.
1119 if (trackRenderer)
1120 trackBounds = trackRenderer->absoluteBoundingBoxRectIgnoringTransforms();
1121 IntRect sliderBounds = o.absoluteBoundingBoxRectIgnoringTransforms();
1122
1123 // Make position relative to the transformed ancestor element.
1124 trackBounds.setX(trackBounds.x() - sliderBounds.x() + rect.x());
1125 trackBounds.setY(trackBounds.y() - sliderBounds.y() + rect.y());
1126
1127 if (isHorizontal) {
1128 tickRect.setWidth(floor(tickSize.width() * zoomFactor));
1129 tickRect.setHeight(floor(tickSize.height() * zoomFactor));
1130 tickRect.setY(floor(rect.y() + rect.height() / 2.0 + sliderTickOffsetFromTrackCenter() * zoomFactor));
1131 tickRegionSideMargin = trackBounds.x() + (thumbSize.width() - tickSize.width() * zoomFactor) / 2.0;
1132 tickRegionWidth = trackBounds.width() - thumbSize.width();
1133 } else {
1134 tickRect.setWidth(floor(tickSize.height() * zoomFactor));
1135 tickRect.setHeight(floor(tickSize.width() * zoomFactor));
1136 tickRect.setX(floor(rect.x() + rect.width() / 2.0 + sliderTickOffsetFromTrackCenter() * zoomFactor));
1137 tickRegionSideMargin = trackBounds.y() + (thumbSize.width() - tickSize.width() * zoomFactor) / 2.0;
1138 tickRegionWidth = trackBounds.height() - thumbSize.width();
1139 }
1140 GraphicsContextStateSaver stateSaver(paintInfo.context());
1141 paintInfo.context().setFillColor(o.style().visitedDependentColorWithColorFilter(CSSPropertyColor));
1142 for (auto& optionElement : dataList->suggestions()) {
1143 if (auto optionValue = input.listOptionValueAsDouble(optionElement)) {
1144 double tickFraction = (*optionValue - min) / (max - min);
1145 double tickRatio = isHorizontal && o.style().isLeftToRightDirection() ? tickFraction : 1.0 - tickFraction;
1146 double tickPosition = round(tickRegionSideMargin + tickRegionWidth * tickRatio);
1147 if (isHorizontal)
1148 tickRect.setX(tickPosition);
1149 else
1150 tickRect.setY(tickPosition);
1151 paintInfo.context().fillRect(tickRect);
1152 }
1153 }
1154}

1425// Value chosen by observation. This can be tweaked.
1426constexpr double minColorContrastValue = 1.195;
1427
1428// For transparent or translucent background color, use lightening.
1429constexpr float minDisabledColorAlphaValue = 0.5f;
1430
1431Color RenderTheme::disabledTextColor(const Color& textColor, const Color& backgroundColor) const
1432{
1433 // The explicit check for black is an optimization for the 99% case (black on white).
1434 // This also means that black on black will turn into grey on black when disabled.
1435 Color disabledColor;
1436 if (equalIgnoringSemanticColor(textColor, Color::black) || backgroundColor.alphaAsFloat() < minDisabledColorAlphaValue || textColor.luminance() < backgroundColor.luminance())
1437 disabledColor = textColor.lightened();
1438 else
1439 disabledColor = textColor.darkened();
1440
1441 // If there's not very much contrast between the disabled color and the background color,
1442 // just leave the text color alone. We don't want to change a good contrast color scheme so that it has really bad contrast.
1443 // If the contrast was already poor, then it doesn't do any good to change it to a different poor contrast color scheme.
1444 if (contrastRatio(disabledColor, backgroundColor) < minColorContrastValue)
1445 return textColor;
1446
1447 return disabledColor;
1448}

1450// Value chosen to return dark gray for both white on black and black on white.
1451constexpr float datePlaceholderColorLightnessAdjustmentFactor = 0.66f;
1452
1453Color RenderTheme::datePlaceholderTextColor(const Color& textColor, const Color& backgroundColor) const
1454{
1455 // FIXME: Consider using LCHA<float> rather than HSLA<float> for better perceptual results and to avoid clamping to sRGB gamut, which is what HSLA does.
1456 auto hsla = textColor.toColorTypeLossy<HSLA<float>>();
1457 if (textColor.luminance() < backgroundColor.luminance())
1458 hsla.lightness += datePlaceholderColorLightnessAdjustmentFactor * (100.0f - hsla.lightness);
1459 else
1460 hsla.lightness *= datePlaceholderColorLightnessAdjustmentFactor;
1461
1462 // FIXME: Consider keeping color in LCHA (if that change is made) or converting back to the initial underlying color type to avoid unnecessarily clamping colors outside of sRGB.
1463 return convertColor<SRGBA<float>>(hsla);
1464}
