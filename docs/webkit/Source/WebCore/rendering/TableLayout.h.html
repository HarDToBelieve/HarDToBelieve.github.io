21#pragma once
22
23#include "LayoutUnit.h"
24#include <wtf/FastMalloc.h>
25#include <wtf/Noncopyable.h>
26
27namespace WebCore {
28
29class RenderTable;
30
31enum class TableIntrinsics : uint8_t;
32
33DECLARE_ALLOCATOR_WITH_HEAP_IDENTIFIER(TableLayout);
34class TableLayout {
35 WTF_MAKE_NONCOPYABLE(TableLayout);
36 WTF_MAKE_FAST_ALLOCATED_WITH_HEAP_IDENTIFIER(TableLayout);
37public:
38 explicit TableLayout(RenderTable* table)
39 : m_table(table)
40 {
41 }
42
43 virtual ~TableLayout() = default;
44
45 virtual void computeIntrinsicLogicalWidths(LayoutUnit& minWidth, LayoutUnit& maxWidth, TableIntrinsics) = 0;
46 virtual LayoutUnit scaledWidthFromPercentColumns() const { return 0_lu; }
47 virtual void applyPreferredLogicalWidthQuirks(LayoutUnit& minWidth, LayoutUnit& maxWidth) const = 0;
48 virtual void layout() = 0;
49
50protected:
51 // FIXME: Once we enable SATURATED_LAYOUT_ARITHMETHIC, this should just be LayoutUnit::nearlyMax().
52 // Until then though, using nearlyMax causes overflow in some tests, so we just pick a large number.
53 static constexpr int tableMaxWidth = 1000000;
54
55 RenderTable* m_table;
56};
57
58} // namespace WebCore
