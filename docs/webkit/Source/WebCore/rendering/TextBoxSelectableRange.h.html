25#pragma once
26
27#include <optional>
28
29namespace WebCore {
30
31struct TextBoxSelectableRange {
32 const unsigned start;
33 const unsigned length;
34 const unsigned additionalLengthAtEnd { 0 };
35 const bool isLineBreak { false };
36 const std::optional<unsigned> truncation { };
37
38 unsigned clamp(unsigned offset) const
39 {
40 auto clampedOffset = std::clamp(offset, start, start + length) - start;
41
42 if (truncation)
43 return std::min<unsigned>(clampedOffset, *truncation);
44
45 if (clampedOffset == length)
46 clampedOffset += additionalLengthAtEnd;
47
48 return clampedOffset;
49 }
50
51 std::pair<unsigned, unsigned> clamp(unsigned startOffset, unsigned endOffset) const
52 {
53 return { clamp(startOffset), clamp(endOffset) };
54 }
55
56 bool intersects(unsigned startOffset, unsigned endOffset) const
57 {
58 return clamp(startOffset) < clamp(endOffset);
59 }
60};
