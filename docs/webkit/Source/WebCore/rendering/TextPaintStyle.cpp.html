43TextPaintStyle::TextPaintStyle(const Color& color)
44 : fillColor(color)
45 , strokeColor(color)
46{
47}

49bool TextPaintStyle::operator==(const TextPaintStyle& other) const
50{
51 return fillColor == other.fillColor && strokeColor == other.strokeColor && emphasisMarkColor == other.emphasisMarkColor
52 && strokeWidth == other.strokeWidth && paintOrder == other.paintOrder && lineJoin == other.lineJoin
53#if ENABLE(LETTERPRESS)
54 && useLetterpressEffect == other.useLetterpressEffect
55#endif
56#if HAVE(OS_DARK_MODE_SUPPORT)
57 && useDarkAppearance == other.useDarkAppearance
58#endif
59 && lineCap == other.lineCap && miterLimit == other.miterLimit;
60}

62bool textColorIsLegibleAgainstBackgroundColor(const Color& textColor, const Color& backgroundColor)
63{
64 // Uses the WCAG 2.0 definition of legibility: a contrast ratio of 4.5:1 or greater.
65 // https://www.w3.org/TR/WCAG20/#visual-audio-contrast-contrast
66 return contrastRatio(textColor, backgroundColor) >= 4.5;
67}
68
69static Color adjustColorForVisibilityOnBackground(const Color& textColor, const Color& backgroundColor)
70{
71 if (textColorIsLegibleAgainstBackgroundColor(textColor, backgroundColor))
72 return textColor;
73
74 if (textColor.luminance() > 0.5)
75 return textColor.darkened();
76 return textColor.lightened();
77}

79TextPaintStyle computeTextPaintStyle(const Frame& frame, const RenderStyle& lineStyle, const PaintInfo& paintInfo)
80{
81 TextPaintStyle paintStyle;
82
83#if ENABLE(LETTERPRESS)
84 paintStyle.useLetterpressEffect = lineStyle.textDecorationsInEffect().contains(TextDecoration::Letterpress);
85#endif
86
87#if HAVE(OS_DARK_MODE_SUPPORT)
88 paintStyle.useDarkAppearance = frame.document() ? frame.document()->useDarkAppearance(&lineStyle) : false;
89#endif
90
91 auto viewportSize = frame.view() ? frame.view()->size() : IntSize();
92 paintStyle.strokeWidth = lineStyle.computedStrokeWidth(viewportSize);
93 paintStyle.paintOrder = lineStyle.paintOrder();
94 paintStyle.lineJoin = lineStyle.joinStyle();
95 paintStyle.lineCap = lineStyle.capStyle();
96 paintStyle.miterLimit = lineStyle.strokeMiterLimit();
97
98 if (paintInfo.forceTextColor()) {
99 paintStyle.fillColor = paintInfo.forcedTextColor();
100 paintStyle.strokeColor = paintInfo.forcedTextColor();
101 paintStyle.emphasisMarkColor = paintInfo.forcedTextColor();
102 return paintStyle;
103 }
104
105 if (lineStyle.insideDefaultButton()) {
106 Page* page = frame.page();
107 if (page && page->focusController().isActive()) {
108 OptionSet<StyleColor::Options> options;
109 if (page->useSystemAppearance())
110 options.add(StyleColor::Options::UseSystemAppearance);
111 paintStyle.fillColor = RenderTheme::singleton().defaultButtonTextColor(options);
112 return paintStyle;
113 }
114 }
115
116 paintStyle.fillColor = lineStyle.visitedDependentColorWithColorFilter(CSSPropertyWebkitTextFillColor);
117
118 bool forceBackgroundToWhite = false;
119 if (frame.document() && frame.document()->printing()) {
120 if (lineStyle.printColorAdjust() == PrintColorAdjust::Economy)
121 forceBackgroundToWhite = true;
122 if (frame.settings().shouldPrintBackgrounds())
123 forceBackgroundToWhite = false;
124 }
125
126 // Make the text fill color legible against a white background
127 if (forceBackgroundToWhite)
128 paintStyle.fillColor = adjustColorForVisibilityOnBackground(paintStyle.fillColor, Color::white);
129
130 paintStyle.strokeColor = lineStyle.colorByApplyingColorFilter(lineStyle.computedStrokeColor());
131
132 // Make the text stroke color legible against a white background
133 if (forceBackgroundToWhite)
134 paintStyle.strokeColor = adjustColorForVisibilityOnBackground(paintStyle.strokeColor, Color::white);
135
136 paintStyle.emphasisMarkColor = lineStyle.visitedDependentColorWithColorFilter(CSSPropertyWebkitTextEmphasisColor);
137
138 // Make the text stroke color legible against a white background
139 if (forceBackgroundToWhite)
140 paintStyle.emphasisMarkColor = adjustColorForVisibilityOnBackground(paintStyle.emphasisMarkColor, Color::white);
141
142 return paintStyle;
143}

145TextPaintStyle computeTextSelectionPaintStyle(const TextPaintStyle& textPaintStyle, const RenderText& renderer, const RenderStyle& lineStyle, const PaintInfo& paintInfo, std::optional<ShadowData>& selectionShadow)
146{
147 TextPaintStyle selectionPaintStyle = textPaintStyle;
148
149#if ENABLE(TEXT_SELECTION)
150 Color foreground = paintInfo.forceTextColor() ? paintInfo.forcedTextColor() : renderer.selectionForegroundColor();
151 if (foreground.isValid() && foreground != selectionPaintStyle.fillColor)
152 selectionPaintStyle.fillColor = foreground;
153
154 Color emphasisMarkForeground = paintInfo.forceTextColor() ? paintInfo.forcedTextColor() : renderer.selectionEmphasisMarkColor();
155 if (emphasisMarkForeground.isValid() && emphasisMarkForeground != selectionPaintStyle.emphasisMarkColor)
156 selectionPaintStyle.emphasisMarkColor = emphasisMarkForeground;
157
158 if (auto pseudoStyle = renderer.selectionPseudoStyle()) {
159 selectionShadow = ShadowData::clone(paintInfo.forceTextColor() ? nullptr : pseudoStyle->textShadow());
160 auto viewportSize = renderer.frame().view() ? renderer.frame().view()->size() : IntSize();
161 float strokeWidth = pseudoStyle->computedStrokeWidth(viewportSize);
162 if (strokeWidth != selectionPaintStyle.strokeWidth)
163 selectionPaintStyle.strokeWidth = strokeWidth;
164
165 Color stroke = paintInfo.forceTextColor() ? paintInfo.forcedTextColor() : pseudoStyle->computedStrokeColor();
166 if (stroke != selectionPaintStyle.strokeColor)
167 selectionPaintStyle.strokeColor = stroke;
168 } else
169 selectionShadow = ShadowData::clone(paintInfo.forceTextColor() ? nullptr : lineStyle.textShadow());
170#else
171 UNUSED_PARAM(renderer);
172 UNUSED_PARAM(lineStyle);
173 UNUSED_PARAM(paintInfo);
174 selectionShadow = ShadowData::clone(paintInfo.forceTextColor() ? nullptr : lineStyle.textShadow());
175#endif
176 return selectionPaintStyle;
177}

179void updateGraphicsContext(GraphicsContext& context, const TextPaintStyle& paintStyle, FillColorType fillColorType)
180{
181 TextDrawingModeFlags mode = context.textDrawingMode();
182 TextDrawingModeFlags newMode = mode;
183#if ENABLE(LETTERPRESS)
184 if (paintStyle.useLetterpressEffect)
185 newMode.add(TextDrawingMode::Letterpress);
186 else
187 newMode.remove(TextDrawingMode::Letterpress);
188#endif
189 if (paintStyle.strokeWidth > 0 && paintStyle.strokeColor.isVisible())
190 newMode.add(TextDrawingMode::Stroke);
191 if (mode != newMode) {
192 context.setTextDrawingMode(newMode);
193 mode = newMode;
194 }
195
196#if HAVE(OS_DARK_MODE_SUPPORT)
197 context.setUseDarkAppearance(paintStyle.useDarkAppearance);
198#endif
199
200 Color fillColor = fillColorType == UseEmphasisMarkColor ? paintStyle.emphasisMarkColor : paintStyle.fillColor;
201 if (mode.contains(TextDrawingMode::Fill) && (fillColor != context.fillColor()))
202 context.setFillColor(fillColor);
203
204 if (mode & TextDrawingMode::Stroke) {
205 if (paintStyle.strokeColor != context.strokeColor())
206 context.setStrokeColor(paintStyle.strokeColor);
207 if (paintStyle.strokeWidth != context.strokeThickness())
208 context.setStrokeThickness(paintStyle.strokeWidth);
209 context.setLineJoin(paintStyle.lineJoin);
210 context.setLineCap(paintStyle.lineCap);
211 if (paintStyle.lineJoin == MiterJoin)
212 context.setMiterLimit(paintStyle.miterLimit);
213 }
214}
