67struct WordTrailingSpace {
68 WordTrailingSpace(const RenderStyle& style, bool measuringWithTrailingWhitespaceEnabled = true)
69 : m_style(style)
70 {
71 if (!measuringWithTrailingWhitespaceEnabled || !m_style.fontCascade().enableKerning())
72 m_state = WordTrailingSpaceState::Initialized;
73 }
74
75 std::optional<float> width(HashSet<const Font*>& fallbackFonts)
76 {
77 if (m_state == WordTrailingSpaceState::Initialized)
78 return m_width;
79
80 auto& font = m_style.fontCascade();
81 m_width = font.width(RenderBlock::constructTextRun(&space, 1, m_style), &fallbackFonts) + font.wordSpacing();
82 m_state = WordTrailingSpaceState::Initialized;
83 return m_width;
84 }
85
86private:
87 enum class WordTrailingSpaceState { Uninitialized, Initialized };
88 const RenderStyle& m_style;
89 WordTrailingSpaceState m_state { WordTrailingSpaceState::Uninitialized };
90 std::optional<float> m_width;
91};

235inline void BreakingContext::initializeForCurrentObject()
236{
237 m_hadUncommittedWidthBeforeCurrent = !!m_width.uncommittedWidth();
238 auto& renderer = *m_current.renderer();
239
240 m_nextObject = bidiNextSkippingEmptyInlines(m_block, &renderer);
241 if (m_nextObject && m_nextObject->parent() && !m_nextObject->parent()->isDescendantOf(renderer.parent()))
242 m_includeEndWidth = true;
243
244 m_currWS = renderer.isReplaced() ? renderer.parent()->style().whiteSpace() : renderer.style().whiteSpace();
245 m_lastWS = m_lastObject->isReplaced() ? m_lastObject->parent()->style().whiteSpace() : m_lastObject->style().whiteSpace();
246
247 m_autoWrap = RenderStyle::autoWrap(m_currWS);
248 m_autoWrapWasEverTrueOnLine = m_autoWrapWasEverTrueOnLine || m_autoWrap;
249
250 m_preservesNewline = renderer.isSVGInlineText() ? false : RenderStyle::preserveNewline(m_currWS);
251
252 m_collapseWhiteSpace = RenderStyle::collapseWhiteSpace(m_currWS);
253}

255inline void BreakingContext::increment()
256{
257 // Clear out our character space bool, since inline <pre>s don't collapse whitespace
258 // with adjacent inline normal/nowrap spans.
259 if (!m_collapseWhiteSpace)
260 m_currentCharacterIsSpace = false;
261
262 if (m_nextObject)
263 m_current.moveToStartOf(*m_nextObject);
264 else
265 m_current.clear();
266 m_atStart = false;
267}

269inline void BreakingContext::handleBR(UsedClear& usedClear)
270{
271 if (fitsOnLineOrHangsAtEnd()) {
272 auto& br = *m_current.renderer();
273 m_lineBreak.moveToStartOf(br);
274 m_lineBreak.increment();
275
276 // A <br> always breaks a line, so don't let the line be collapsed
277 // away. Also, the space at the end of a line with a <br> does not
278 // get collapsed away. It only does this if the previous line broke
279 // cleanly. Otherwise the <br> has no effect on whether the line is
280 // empty or not.
281 if (m_startingNewParagraph)
282 m_lineInfo.setEmpty(false, &m_block, &m_width);
283 m_trailingObjects.clear();
284 m_lineInfo.setPreviousLineBrokeCleanly(true);
285
286 // A <br> with clearance always needs a linebox in case the lines below it get dirtied later and
287 // need to check for floats to clear - so if we're ignoring spaces, stop ignoring them and add a
288 // run for this object.
289 if (m_ignoringSpaces && RenderStyle::usedClear(br) != UsedClear::None)
290 m_lineWhitespaceCollapsingState.ensureLineBoxInsideIgnoredSpaces(br);
291 // If we were preceded by collapsing space and are in a right-aligned container we need to ensure the space gets
292 // collapsed away so that it doesn't push the text out from the container's right-hand edge.
293 // FIXME: Do this regardless of the container's alignment - will require rebaselining a lot of test results.
294 else if (m_ignoringSpaces && (m_blockStyle.textAlign() == TextAlignMode::Right || m_blockStyle.textAlign() == TextAlignMode::WebKitRight))
295 m_lineWhitespaceCollapsingState.stopIgnoringSpaces(InlineIterator(0, m_current.renderer(), m_current.offset()));
296
297 if (!m_lineInfo.isEmpty())
298 usedClear = RenderStyle::usedClear(br);
299 }
300 m_atEnd = true;
301}

313inline LayoutUnit inlineLogicalWidth(const RenderObject& renderer, bool checkStartEdge = true, bool checkEndEdge = true)
314{
315 auto previousInFlowSibling = [] (const auto& renderer) {
316 auto* previousSibling = renderer.previousSibling();
317 for (; previousSibling && previousSibling->isOutOfFlowPositioned(); previousSibling = previousSibling->previousSibling()) { }
318 return previousSibling;
319 };
320
321 auto shouldAddBorderPaddingMargin = [] (const auto& renderer) {
322 // When deciding whether we're at the edge of an inline, adjacent collapsed whitespace is the same as no sibling at all.
323 if (is<RenderText>(renderer) && !downcast<RenderText>(renderer).text().length())
324 return true;
325#if ENABLE(CSS_BOX_DECORATION_BREAK)
326 if (is<RenderLineBreak>(renderer) && renderer.parent()->style().boxDecorationBreak() == BoxDecorationBreak::Clone)
327 return true;
328#endif
329 return false;
330 };
331
332 unsigned lineDepth = 1;
333 auto extraWidth = LayoutUnit { };
334 auto* parent = renderer.parent();
335 auto* child = &renderer;
336 while (is<RenderInline>(*parent) && lineDepth++ < cMaxLineDepth) {
337 const auto& parentAsRenderInline = downcast<RenderInline>(*parent);
338 if (!isEmptyInline(parentAsRenderInline)) {
339 auto* previousSibling = previousInFlowSibling(*child);
340 checkStartEdge = checkStartEdge && (!previousSibling || shouldAddBorderPaddingMargin(*previousSibling));
341 if (checkStartEdge)
342 extraWidth += borderPaddingMarginStart(parentAsRenderInline);
343 auto* nextSibling = child->nextSibling();
344 checkEndEdge = checkEndEdge && (!nextSibling || shouldAddBorderPaddingMargin(*nextSibling));
345 if (checkEndEdge)
346 extraWidth += borderPaddingMarginEnd(parentAsRenderInline);
347 if (!checkStartEdge && !checkEndEdge)
348 return extraWidth;
349 }
350 child = parent;
351 parent = child->parent();
352 ASSERT(parent);
353 }
354 return extraWidth;
355}

357inline void BreakingContext::handleOutOfFlowPositioned(Vector<RenderBox*>& positionedObjects)
358{
359 // If our original display wasn't an inline type, then we can determine our static inline position now.
360 auto& box = downcast<RenderBox>(*m_current.renderer());
361 bool isInlineType = box.style().isOriginalDisplayInlineType();
362 if (!isInlineType)
363 m_block.setStaticInlinePositionForChild(box, m_block.logicalHeight(), m_block.startOffsetForContent(m_block.logicalHeight()));
364 else {
365 // If our original display was an DisplayType::Inline type, then we can determine our static y position now.
366 box.layer()->setStaticBlockPosition(m_block.logicalHeight());
367 }
368
369 // If we're ignoring spaces, we have to stop and include this object and
370 // then start ignoring spaces again.
371 if (isInlineType || box.container()->isRenderInline()) {
372 if (m_ignoringSpaces)
373 m_lineWhitespaceCollapsingState.ensureLineBoxInsideIgnoredSpaces(box);
374 m_trailingObjects.appendBoxIfNeeded(box);
375 } else
376 positionedObjects.append(&box);
377
378 m_width.addUncommittedWidth(inlineLogicalWidth(box));
379 // Reset prior line break context characters.
380 m_renderTextInfo.lineBreakIterator.resetPriorContext();
381}

383inline void BreakingContext::handleFloat()
384{
385 auto& floatBox = downcast<RenderBox>(*m_current.renderer());
386 const auto& floatingObject = *m_lineBreaker.insertFloatingObject(floatBox);
387 // check if it fits in the current line.
388 // If it does, position it now, otherwise, position
389 // it after moving to next line (in clearFloats() func)
390 if (m_floatsFitOnLine && m_width.fitsOnLineExcludingTrailingWhitespace(m_block.logicalWidthForFloat(floatingObject))) {
391 m_lineBreaker.positionNewFloatOnLine(floatingObject, m_lastFloatFromPreviousLine, m_lineInfo, m_width);
392 if (m_lineBreak.renderer() == m_current.renderer()) {
393 ASSERT(!m_lineBreak.offset());
394 m_lineBreak.increment();
395 }
396 } else
397 m_floatsFitOnLine = false;
398 // Update prior line break context characters, using U+FFFD (OBJECT REPLACEMENT CHARACTER) for floating element.
399 m_renderTextInfo.lineBreakIterator.updatePriorContext(replacementCharacter);
400}

402// This is currently just used for list markers and inline flows that have line boxes. Neither should
403// have an effect on whitespace at the start of the line.
404inline bool shouldSkipWhitespaceAfterStartObject(RenderBlockFlow& block, RenderObject* o, LineWhitespaceCollapsingState& lineWhitespaceCollapsingState)
405{
406 RenderObject* next = bidiNextSkippingEmptyInlines(block, o);
407 while (next && next->isFloatingOrOutOfFlowPositioned())
408 next = bidiNextSkippingEmptyInlines(block, next);
409
410 if (is<RenderText>(next) && downcast<RenderText>(*next).text().length() > 0) {
411 RenderText& nextText = downcast<RenderText>(*next);
412 UChar nextChar = nextText.characterAt(0);
413 if (nextText.style().isCollapsibleWhiteSpace(nextChar)) {
414 lineWhitespaceCollapsingState.startIgnoringSpaces(InlineIterator(nullptr, o, 0));
415 return true;
416 }
417 }
418
419 return false;
420}

422inline void BreakingContext::handleEmptyInline()
423{
424 RenderInline& flowBox = downcast<RenderInline>(*m_current.renderer());
425
426 // This should only end up being called on empty inlines
427 ASSERT(isEmptyInline(flowBox));
428
429 // Now that some inline flows have line boxes, if we are already ignoring spaces, we need
430 // to make sure that we stop to include this object and then start ignoring spaces again.
431 // If this object is at the start of the line, we need to behave like list markers and
432 // start ignoring spaces.
433 bool requiresLineBox = alwaysRequiresLineBox(flowBox);
434 if (requiresLineBox || requiresLineBoxForContent(flowBox, m_lineInfo)) {
435 // An empty inline that only has line-height, vertical-align or font-metrics will only get a
436 // line box to affect the height of the line if the rest of the line is not empty.
437 if (requiresLineBox)
438 m_lineInfo.setEmpty(false, &m_block, &m_width);
439 if (m_ignoringSpaces) {
440 m_trailingObjects.clear();
441 m_lineWhitespaceCollapsingState.ensureLineBoxInsideIgnoredSpaces(*m_current.renderer());
442 } else if (m_blockStyle.collapseWhiteSpace() && m_resolver.position().renderer() == m_current.renderer()
443 && shouldSkipWhitespaceAfterStartObject(m_block, m_current.renderer(), m_lineWhitespaceCollapsingState)) {
444 // Like with list markers, we start ignoring spaces to make sure that any
445 // additional spaces we see will be discarded.
446 m_currentCharacterIsSpace = true;
447 m_currentCharacterIsWS = true;
448 m_ignoringSpaces = true;
449 } else
450 m_trailingObjects.appendBoxIfNeeded(flowBox);
451 }
452
453 float inlineWidth = inlineLogicalWidth(*m_current.renderer()) + borderPaddingMarginStart(flowBox) + borderPaddingMarginEnd(flowBox);
454 m_width.addUncommittedWidth(inlineWidth);
455 if (m_hangsAtEnd && inlineWidth)
456 m_hangsAtEnd = false;
457}

459inline void BreakingContext::handleReplaced()
460{
461 auto& replacedBox = downcast<RenderBox>(*m_current.renderer());
462
463 if (m_atStart)
464 m_width.updateAvailableWidth(replacedBox.logicalHeight());
465
466 // Break on replaced elements if either has normal white-space.
467 if ((m_autoWrap || RenderStyle::autoWrap(m_lastWS)) && (!replacedBox.isImage() || m_allowImagesToBreak)
468 && (!is<RenderRubyRun>(replacedBox) || downcast<RenderRubyRun>(replacedBox).canBreakBefore(m_renderTextInfo.lineBreakIterator))) {
469 if (auto* renderer = m_current.renderer())
470 commitLineBreakAtCurrentWidth(*renderer);
471 else
472 commitLineBreakClear();
473 } else
474 m_hangsAtEnd = false;
475
476 if (m_ignoringSpaces)
477 m_lineWhitespaceCollapsingState.stopIgnoringSpaces(InlineIterator(0, &replacedBox, 0));
478
479 m_lineInfo.setEmpty(false, &m_block, &m_width);
480 m_ignoringSpaces = false;
481 m_currentCharacterIsSpace = false;
482 m_currentCharacterIsWS = false;
483 m_trailingObjects.clear();
484
485 // Optimize for a common case. If we can't find whitespace after the list
486 // item, then this is all moot.
487 LayoutUnit replacedLogicalWidth = m_block.logicalWidthForChild(replacedBox) + m_block.marginStartForChild(replacedBox) + m_block.marginEndForChild(replacedBox) + inlineLogicalWidth(replacedBox);
488 if (is<RenderListMarker>(replacedBox)) {
489 if (m_blockStyle.collapseWhiteSpace() && shouldSkipWhitespaceAfterStartObject(m_block, &replacedBox, m_lineWhitespaceCollapsingState)) {
490 // Like with inline flows, we start ignoring spaces to make sure that any
491 // additional spaces we see will be discarded.
492 m_currentCharacterIsSpace = true;
493 m_currentCharacterIsWS = false;
494 m_ignoringSpaces = true;
495 }
496 if (downcast<RenderListMarker>(replacedBox).isInside())
497 m_width.addUncommittedReplacedWidth(replacedLogicalWidth);
498 } else
499 m_width.addUncommittedReplacedWidth(replacedLogicalWidth);
500 if (is<RenderRubyRun>(replacedBox)) {
501 m_width.applyOverhang(downcast<RenderRubyRun>(replacedBox), m_lastObject, m_nextObject);
502 downcast<RenderRubyRun>(replacedBox).updatePriorContextFromCachedBreakIterator(m_renderTextInfo.lineBreakIterator);
503 } else {
504 // Update prior line break context characters, using U+FFFD (OBJECT REPLACEMENT CHARACTER) for replaced element.
505 m_renderTextInfo.lineBreakIterator.updatePriorContext(replacementCharacter);
506 }
507}

544ALWAYS_INLINE float textWidth(RenderText& text, unsigned from, unsigned len, const FontCascade& font, float xPos, bool isFixedPitch, bool collapseWhiteSpace, HashSet<const Font*>& fallbackFonts, TextLayout* layout = nullptr)
545{
546 const RenderStyle& style = text.style();
547
548 GlyphOverflow glyphOverflow;
549 // FIXME: This is not the right level of abstraction for isFixedPitch. Font fallback may make it such that the fixed pitch font is never actually used!
550 if (isFixedPitch || (!from && len == text.text().length()) || style.hasTextCombine())
551 return text.width(from, len, font, xPos, &fallbackFonts, &glyphOverflow);
552
553 if (layout)
554 return FontCascade::width(*layout, from, len, &fallbackFonts);
555
556 TextRun run = RenderBlock::constructTextRun(text, from, len, style);
557 run.setCharacterScanForCodePath(!text.canUseSimpleFontCodePath());
558 run.setTabSize(!collapseWhiteSpace, style.tabSize());
559 run.setXPos(xPos);
560 return font.width(run, &fallbackFonts, &glyphOverflow);
561}

571inline void tryHyphenating(RenderText& text, const FontCascade& font, const AtomString& localeIdentifier, unsigned consecutiveHyphenatedLines, int consecutiveHyphenatedLinesLimit, int minimumPrefixLimit, int minimumSuffixLimit, unsigned lastSpace, unsigned pos, float xPos, float availableWidth, bool isFixedPitch, bool collapseWhiteSpace, int lastSpaceWordSpacing, InlineIterator& lineBreak, std::optional<unsigned> nextBreakable, bool& hyphenated)
572{
573 // Map 'hyphenate-limit-{before,after}: auto;' to 2.
574 unsigned minimumPrefixLength;
575 unsigned minimumSuffixLength;
576
577 if (minimumPrefixLimit < 0)
578 minimumPrefixLength = 2;
579 else
580 minimumPrefixLength = static_cast<unsigned>(minimumPrefixLimit);
581
582 if (minimumSuffixLimit < 0)
583 minimumSuffixLength = 2;
584 else
585 minimumSuffixLength = static_cast<unsigned>(minimumSuffixLimit);
586
587 if (pos - lastSpace <= minimumSuffixLength)
588 return;
589
590 if (consecutiveHyphenatedLinesLimit >= 0 && consecutiveHyphenatedLines >= static_cast<unsigned>(consecutiveHyphenatedLinesLimit))
591 return;
592
593 float hyphenWidth = measureHyphenWidth(text, font);
594
595 float maxPrefixWidth = availableWidth - xPos - hyphenWidth - lastSpaceWordSpacing;
596 if (!enoughWidthForHyphenation(maxPrefixWidth, font.pixelSize()))
597 return;
598
599 const RenderStyle& style = text.style();
600 TextRun run = RenderBlock::constructTextRun(text, lastSpace, pos - lastSpace, style);
601 run.setTabSize(!collapseWhiteSpace, style.tabSize());
602 run.setXPos(xPos + lastSpaceWordSpacing);
603
604 unsigned prefixLength = font.offsetForPosition(run, maxPrefixWidth, false);
605 if (prefixLength < minimumPrefixLength)
606 return;
607
608 prefixLength = lastHyphenLocation(StringView(text.text()).substring(lastSpace, pos - lastSpace), std::min(prefixLength, pos - lastSpace - minimumSuffixLength) + 1, localeIdentifier);
609 if (!prefixLength || prefixLength < minimumPrefixLength)
610 return;
611
612 // When lastSpace is a space, which it always is except sometimes at the beginning of a line or after collapsed
613 // space, it should not count towards hyphenate-limit-before.
614 if (prefixLength == minimumPrefixLength) {
615 UChar characterAtLastSpace = text.characterAt(lastSpace);
616 if (characterAtLastSpace == ' ' || characterAtLastSpace == '\n' || characterAtLastSpace == '\t' || characterAtLastSpace == noBreakSpace)
617 return;
618 }
619
620 ASSERT(pos - lastSpace - prefixLength >= minimumSuffixLength);
621
622#if ASSERT_ENABLED
623 HashSet<const Font*> fallbackFonts;
624 float prefixWidth = hyphenWidth + textWidth(text, lastSpace, prefixLength, font, xPos, isFixedPitch, collapseWhiteSpace, fallbackFonts) + lastSpaceWordSpacing;
625 ASSERT(xPos + prefixWidth <= availableWidth);
626#else
627 UNUSED_PARAM(isFixedPitch);
628#endif
629
630 lineBreak.moveTo(text, lastSpace + prefixLength, nextBreakable);
631 hyphenated = true;
632}
633
634inline float BreakingContext::computeAdditionalBetweenWordsWidth(RenderText& renderText, TextLayout* textLayout, UChar currentCharacter, WordTrailingSpace& wordTrailingSpace, HashSet<const Font*>& fallbackFonts, WordMeasurements& wordMeasurements, const FontCascade& font, bool isFixedPitch, unsigned lastSpace, float lastSpaceWordSpacing, float wordSpacingForWordMeasurement, unsigned offset)
635{
636 wordMeasurements.grow(wordMeasurements.size() + 1);
637 WordMeasurement& wordMeasurement = wordMeasurements.last();
638
639 wordMeasurement.renderer = &renderText;
640 wordMeasurement.endOffset = offset;
641 wordMeasurement.startOffset = lastSpace;
642
643 float additionalTempWidth = 0;
644 std::optional<float> wordTrailingSpaceWidth;
645 if (currentCharacter == ' ')
646 wordTrailingSpaceWidth = wordTrailingSpace.width(fallbackFonts);
647 if (wordTrailingSpaceWidth)
648 additionalTempWidth = textWidth(renderText, lastSpace, offset + 1 - lastSpace, font, m_width.currentWidth(), isFixedPitch, m_collapseWhiteSpace, wordMeasurement.fallbackFonts, textLayout) - wordTrailingSpaceWidth.value();
649 else
650 additionalTempWidth = textWidth(renderText, lastSpace, offset - lastSpace, font, m_width.currentWidth(), isFixedPitch, m_collapseWhiteSpace, wordMeasurement.fallbackFonts, textLayout);
651
652 if (wordMeasurement.fallbackFonts.isEmpty() && !fallbackFonts.isEmpty())
653 wordMeasurement.fallbackFonts.swap(fallbackFonts);
654 fallbackFonts.clear();
655
656 wordMeasurement.width = additionalTempWidth + wordSpacingForWordMeasurement;
657 additionalTempWidth += lastSpaceWordSpacing;
658 return additionalTempWidth;
659}
660
661inline bool BreakingContext::handleText(WordMeasurements& wordMeasurements, bool& hyphenated, unsigned& consecutiveHyphenatedLines)
662{
663 if (!m_current.offset())
664 m_appliedStartWidth = false;
665
666 auto& renderer = downcast<RenderText>(*m_current.renderer());
667 bool isSVGText = renderer.isSVGInlineText();
668
669 // If we have left a no-wrap inline and entered an autowrap inline while ignoring spaces
670 // then we need to mark the start of the autowrap inline as a potential linebreak now.
671 if (m_autoWrap && !RenderStyle::autoWrap(m_lastWS) && m_ignoringSpaces)
672 commitLineBreakAtCurrentWidth(renderer);
673
674 if (renderer.style().hasTextCombine() && is<RenderCombineText>(renderer)) {
675 auto& combineRenderer = downcast<RenderCombineText>(renderer);
676 combineRenderer.combineTextIfNeeded();
677 // The length of the renderer's text may have changed. Increment stale iterator positions
678 if (iteratorIsBeyondEndOfRenderCombineText(m_lineBreak, combineRenderer)) {
679 ASSERT(iteratorIsBeyondEndOfRenderCombineText(m_resolver.position(), combineRenderer));
680 m_lineBreak.increment();
681 m_resolver.increment();
682 }
683 }
684
685 const RenderStyle& style = lineStyle(renderer, m_lineInfo);
686 const FontCascade& font = style.fontCascade();
687 bool isFixedPitch = font.isFixedPitch();
688 bool canHyphenate = style.hyphens() == Hyphens::Auto && WebCore::canHyphenate(style.computedLocale());
689 bool canHangPunctuationAtStart = style.hangingPunctuation().contains(HangingPunctuation::First);
690 bool canHangPunctuationAtEnd = style.hangingPunctuation().contains(HangingPunctuation::Last);
691 bool canHangStopOrCommaAtLineEnd = style.hangingPunctuation().contains(HangingPunctuation::AllowEnd);
692 int endPunctuationIndex = canHangPunctuationAtEnd && m_collapseWhiteSpace ? renderer.lastCharacterIndexStrippingSpaces() : renderer.text().length() - 1;
693 unsigned lastSpace = m_current.offset();
694 float wordSpacing = style.fontCascade().wordSpacing();
695 float lastSpaceWordSpacing = 0;
696 float wordSpacingForWordMeasurement = 0;
697
698 float wrapWidthOffset = m_width.uncommittedWidth() + inlineLogicalWidth(renderer, !m_appliedStartWidth, true);
699 float wrapW = wrapWidthOffset;
700 float charWidth = 0;
701 bool breakNBSP = m_autoWrap && style.nbspMode() == NBSPMode::Space;
702 // Auto-wrapping text should wrap in the middle of a word only if it could not wrap before the word,
703 // which is only possible if the word is the first thing on the line.
704 bool breakWords = style.breakWords() && ((m_autoWrap && (!m_width.committedWidth() && !m_width.hasCommittedReplaced())) || m_currWS == WhiteSpace::Pre);
705 bool midWordBreak = false;
706 bool breakAnywhere = style.lineBreak() == LineBreak::Anywhere && m_autoWrap;
707 bool breakAll = (style.wordBreak() == WordBreak::BreakAll || breakAnywhere) && m_autoWrap;
708 bool keepAllWords = style.wordBreak() == WordBreak::KeepAll;
709 float hyphenWidth = 0;
710 auto iteratorMode = mapLineBreakToIteratorMode(m_blockStyle.lineBreak());
711 bool canUseLineBreakShortcut = iteratorMode == LineBreakIteratorMode::Default;
712 bool isLineEmpty = m_lineInfo.isEmpty();
713
714 if (isSVGText) {
715 breakWords = false;
716 breakAll = false;
717 }
718
719 if (m_renderTextInfo.text != &renderer) {
720 updateCounterIfNeeded(renderer);
721 m_renderTextInfo.text = &renderer;
722 m_renderTextInfo.font = &font;
723 m_renderTextInfo.layout = font.createLayout(renderer, m_width.currentWidth(), m_collapseWhiteSpace);
724 m_renderTextInfo.lineBreakIterator.resetStringAndReleaseIterator(renderer.text(), style.computedLocale(), iteratorMode);
725 } else if (m_renderTextInfo.layout && m_renderTextInfo.font != &font) {
726 m_renderTextInfo.font = &font;
727 m_renderTextInfo.layout = font.createLayout(renderer, m_width.currentWidth(), m_collapseWhiteSpace);
728 }
729
730 HashSet<const Font*> fallbackFonts;
731 m_hasFormerOpportunity = false;
732 bool canBreakMidWord = breakWords || breakAll;
733 UChar lastCharacterFromPreviousRenderText = m_renderTextInfo.lineBreakIterator.lastCharacter();
734 UChar lastCharacter = m_renderTextInfo.lineBreakIterator.lastCharacter();
735 UChar secondToLastCharacter = m_renderTextInfo.lineBreakIterator.secondToLastCharacter();
736 // Non-zero only when kerning is enabled and TextLayout isn't used, in which case we measure
737 // words with their trailing space, then subtract its width.
738 TextLayout* textLayout = m_renderTextInfo.layout.get();
739 WordTrailingSpace wordTrailingSpace(style, !textLayout);
740 for (; m_current.offset() < renderer.text().length(); m_current.fastIncrementInTextNode()) {
741 ASSERT(&renderer == m_current.renderer());
742 bool previousCharacterIsSpace = m_currentCharacterIsSpace;
743 bool previousCharacterIsWS = m_currentCharacterIsWS;
744 UChar c = m_current.current();
745 m_currentCharacterIsSpace = c == ' ' || c == '\t' || (!m_preservesNewline && (c == '\n'));
746
747 // A single preserved leading white-space doesn't fulfill the 'betweenWords' condition, however it's indeed a
748 // soft-breaking opportunty so we may want to avoid breaking in the middle of the word.
749 if (m_atStart && m_currentCharacterIsSpace && !previousCharacterIsSpace) {
750 m_hasFormerOpportunity = !breakAnywhere;
751 breakWords = false;
752 canBreakMidWord = breakAll;
753 }
754
755 if (canHangPunctuationAtStart && m_width.isFirstLine() && !m_width.committedWidth() && !wrapW && !inlineLogicalWidth(renderer, true, false)) {
756 m_width.addUncommittedWidth(-renderer.hangablePunctuationStartWidth(m_current.offset()));
757 canHangPunctuationAtStart = false;
758 }
759
760 if (canHangPunctuationAtEnd && !m_nextObject && (int)m_current.offset() == endPunctuationIndex && !inlineLogicalWidth(renderer, false, true)) {
761 m_width.addUncommittedWidth(-renderer.hangablePunctuationEndWidth(endPunctuationIndex));
762 canHangPunctuationAtEnd = false;
763 }
764
765 if (!m_collapseWhiteSpace || !m_currentCharacterIsSpace)
766 m_lineInfo.setEmpty(false, &m_block, &m_width);
767
768 if (c == softHyphen && m_autoWrap && !hyphenWidth && style.hyphens() != Hyphens::None) {
769 hyphenWidth = measureHyphenWidth(renderer, font, &fallbackFonts);
770 m_width.addUncommittedWidth(hyphenWidth);
771 }
772
773 bool applyWordSpacing = false;
774
775 m_currentCharacterIsWS = m_currentCharacterIsSpace || (breakNBSP && c == noBreakSpace);
776
777 if (canBreakMidWord && !midWordBreak && (!m_currentCharacterIsSpace || m_atStart || style.whiteSpace() != WhiteSpace::PreWrap)) {
778 wrapW += charWidth;
779 bool midWordBreakIsBeforeSurrogatePair = U16_IS_LEAD(c) && U16_IS_TRAIL(renderer.characterAt(m_current.offset() + 1));
780 charWidth = textWidth(renderer, m_current.offset(), midWordBreakIsBeforeSurrogatePair ? 2 : 1, font, m_width.committedWidth() + wrapW, isFixedPitch, m_collapseWhiteSpace, fallbackFonts, textLayout);
781 midWordBreak = m_width.committedWidth() + wrapW + charWidth > m_width.availableWidth();
782 }
783
784 std::optional<unsigned> nextBreakablePosition = m_current.nextBreakablePosition();
785 bool betweenWords = c == '\n' || (m_currWS != WhiteSpace::Pre && !m_atStart && isBreakable(m_renderTextInfo.lineBreakIterator, m_current.offset(), nextBreakablePosition, breakNBSP, canUseLineBreakShortcut, keepAllWords, breakAnywhere)
786 && (style.hyphens() != Hyphens::None || (m_current.previousInSameNode() != softHyphen)));
787 m_current.setNextBreakablePosition(nextBreakablePosition);
788
789 if (canHangStopOrCommaAtLineEnd && renderer.isHangableStopOrComma(c) && m_width.fitsOnLine()) {
790 // We need to see if a measurement that excludes the stop would fit. If so, then we should hang
791 // the stop/comma at the end. First measure including the comma.
792 m_hangsAtEnd = false;
793 float inlineStartWidth = !m_appliedStartWidth ? inlineLogicalWidth(renderer, true, false) : 0_lu;
794 float widthIncludingComma = computeAdditionalBetweenWordsWidth(renderer, textLayout, c, wordTrailingSpace, fallbackFonts, wordMeasurements, font, isFixedPitch, lastSpace, lastSpaceWordSpacing, wordSpacingForWordMeasurement, m_current.offset() + 1) + inlineStartWidth;
795 m_width.addUncommittedWidth(widthIncludingComma);
796 if (!m_width.fitsOnLine()) {
797 // See if we fit without the comma involved. If we do, then this is a potential hang point.
798 float widthWithoutStopOrComma = computeAdditionalBetweenWordsWidth(renderer, textLayout, lastCharacter, wordTrailingSpace, fallbackFonts, wordMeasurements, font, isFixedPitch, lastSpace, lastSpaceWordSpacing, wordSpacingForWordMeasurement, m_current.offset()) + inlineStartWidth;
799 m_width.addUncommittedWidth(widthWithoutStopOrComma - widthIncludingComma);
800 if (m_width.fitsOnLine())
801 m_hangsAtEnd = true;
802 } else
803 m_width.addUncommittedWidth(-widthIncludingComma);
804 }
805
806 if (betweenWords || midWordBreak) {
807 bool stoppedIgnoringSpaces = false;
808 if (m_ignoringSpaces) {
809 lastSpaceWordSpacing = 0;
810 if (!m_currentCharacterIsSpace) {
811 // Stop ignoring spaces and begin at this new point.
812 m_ignoringSpaces = false;
813 wordSpacingForWordMeasurement = 0;
814 lastSpace = m_current.offset(); // e.g., "Foo goo", don't add in any of the ignored spaces.
815 m_lineWhitespaceCollapsingState.stopIgnoringSpaces(InlineIterator(0, &renderer, m_current.offset()));
816 stoppedIgnoringSpaces = true;
817 } else {
818 // Just keep ignoring these spaces.
819 nextCharacter(c, lastCharacter, secondToLastCharacter);
820 continue;
821 }
822 }
823
824 float additionalTempWidth = computeAdditionalBetweenWordsWidth(renderer, textLayout, c, wordTrailingSpace, fallbackFonts, wordMeasurements, font, isFixedPitch, lastSpace, lastSpaceWordSpacing, wordSpacingForWordMeasurement, m_current.offset());
825 m_width.addUncommittedWidth(additionalTempWidth);
826
827 WordMeasurement& wordMeasurement = wordMeasurements.last();
828
829 if (m_collapseWhiteSpace && previousCharacterIsSpace && m_currentCharacterIsSpace && additionalTempWidth)
830 m_width.setTrailingWhitespaceWidth(additionalTempWidth);
831
832 if (!m_appliedStartWidth) {
833 float inlineStartWidth = inlineLogicalWidth(renderer, true, false);
834 m_width.addUncommittedWidth(inlineStartWidth);
835 m_appliedStartWidth = true;
836 if (m_hangsAtEnd && inlineStartWidth)
837 m_hangsAtEnd = false;
838 }
839
840 applyWordSpacing = wordSpacing && m_currentCharacterIsSpace;
841
842 if (!m_width.hasCommitted() && m_autoWrap && !fitsOnLineOrHangsAtEnd())
843 m_width.fitBelowFloats(m_lineInfo.isFirstLine());
844
845 if (m_autoWrap || breakWords) {
846 // If we break only after white-space, consider the current character
847 // as candidate width for this line.
848 bool lineWasTooWide = false;
849 if (fitsOnLineOrHangsAtEnd() && m_currentCharacterIsWS && style.breakOnlyAfterWhiteSpace() && (!midWordBreak || m_currWS == WhiteSpace::BreakSpaces)) {
850 float charWidth = textWidth(renderer, m_current.offset(), 1, font, m_width.currentWidth(), isFixedPitch, m_collapseWhiteSpace, wordMeasurement.fallbackFonts, textLayout) + (applyWordSpacing ? wordSpacing : 0);
851 // Check if line is too big even without the extra space
852 // at the end of the line. If it is not, do nothing.
853 // If the line needs the extra whitespace to be too long,
854 // then move the line break to the space and skip all
855 // additional whitespace.
856 if (!m_width.fitsOnLineIncludingExtraWidth(charWidth)) {
857 lineWasTooWide = true;
858 if (m_currWS == WhiteSpace::BreakSpaces)
859 trailingSpacesHang(m_lineBreak, renderer, canBreakMidWord, previousCharacterIsSpace);
860 else {
861 m_lineBreak.moveTo(renderer, m_current.offset(), m_current.nextBreakablePosition());
862 m_lineBreaker.skipTrailingWhitespace(m_lineBreak, m_lineInfo);
863 }
864 }
865 }
866 if ((lineWasTooWide || !m_width.fitsOnLine()) && !m_hangsAtEnd) {
867 // Don't try to hyphenate at the final break of a block, since this means there is
868 // no more content, and a hyphenated single word would end up on a line by itself. This looks
869 // bad so just don't allow it.
870 if (canHyphenate && !m_width.fitsOnLine() && (m_nextObject || !renderer.containsOnlyHTMLWhitespace(m_current.offset(), renderer.text().length() - m_current.offset()) || isLineEmpty)) {
871 tryHyphenating(renderer, font, style.computedLocale(), consecutiveHyphenatedLines, m_blockStyle.hyphenationLimitLines(), style.hyphenationLimitBefore(), style.hyphenationLimitAfter(), lastSpace, m_current.offset(), m_width.currentWidth() - additionalTempWidth, m_width.availableWidth(), isFixedPitch, m_collapseWhiteSpace, lastSpaceWordSpacing, m_lineBreak, m_current.nextBreakablePosition(), m_lineBreaker.m_hyphenated);
872 if (m_lineBreaker.m_hyphenated) {
873 m_atEnd = true;
874 return false;
875 }
876 }
877 if (m_lineBreak.atTextParagraphSeparator()) {
878 if (!stoppedIgnoringSpaces && m_current.offset() > 0)
879 ensureCharacterGetsLineBox(m_lineWhitespaceCollapsingState, m_current);
880 m_lineBreak.increment();
881 m_lineInfo.setPreviousLineBrokeCleanly(true);
882 wordMeasurement.endOffset = m_lineBreak.offset();
883 }
884 // Check if the last breaking position is a soft-hyphen.
885 if (!hyphenated && style.hyphens() != Hyphens::None) {
886 std::optional<unsigned> lastBreakingPosition;
887 const RenderObject* rendererAtBreakingPosition = nullptr;
888 if (m_lineBreak.offset() || m_lineBreak.nextBreakablePosition()) {
889 lastBreakingPosition = m_lineBreak.offset();
890 rendererAtBreakingPosition = m_lineBreak.renderer();
891 } else if (m_current.nextBreakablePosition() && m_current.nextBreakablePosition().value() <= m_current.offset()) {
892 // We might just be right after the soft-hyphen
893 lastBreakingPosition = m_current.nextBreakablePosition().value();
894 rendererAtBreakingPosition = &renderer;
895 }
896 if (lastBreakingPosition) {
897 std::optional<UChar> characterBeforeBreakingPosition;
898 // When last breaking position points to the start of the current context, we need to look at the last character from
899 // the previous non-empty text renderer.
900 if (!lastBreakingPosition.value())
901 characterBeforeBreakingPosition = lastCharacterFromPreviousRenderText;
902 else if (is<RenderText>(rendererAtBreakingPosition)) {
903 const auto& textRenderer = downcast<RenderText>(*rendererAtBreakingPosition);
904 ASSERT(lastBreakingPosition.value() >= 1 && textRenderer.text().length() > (lastBreakingPosition.value() - 1));
905 characterBeforeBreakingPosition = textRenderer.characterAt(lastBreakingPosition.value() - 1);
906 }
907 if (characterBeforeBreakingPosition)
908 hyphenated = characterBeforeBreakingPosition.value() == softHyphen;
909 }
910 }
911 if (m_lineBreak.offset() && m_lineBreak.offset() != (unsigned)wordMeasurement.endOffset && !wordMeasurement.width) {
912 if (charWidth) {
913 wordMeasurement.endOffset = m_lineBreak.offset();
914 wordMeasurement.width = charWidth;
915 }
916 }
917 // Didn't fit. Jump to the end unless there's still an opportunity to collapse whitespace.
918 if (m_ignoringSpaces || !m_collapseWhiteSpace || !m_currentCharacterIsSpace || !previousCharacterIsSpace) {
919 m_atEnd = true;
920 return false;
921 }
922 } else {
923 if (!betweenWords || (midWordBreak && !m_autoWrap))
924 m_width.addUncommittedWidth(-additionalTempWidth);
925 if (hyphenWidth) {
926 // Subtract the width of the soft hyphen out since we fit on a line.
927 m_width.addUncommittedWidth(-hyphenWidth);
928 hyphenWidth = 0;
929 }
930 }
931 }
932
933 if (c == '\n' && m_preservesNewline) {
934 if (!stoppedIgnoringSpaces && m_current.offset())
935 ensureCharacterGetsLineBox(m_lineWhitespaceCollapsingState, m_current);
936 commitLineBreakAtCurrentWidth(renderer, m_current.offset(), m_current.nextBreakablePosition());
937 m_lineBreak.increment();
938 m_lineInfo.setPreviousLineBrokeCleanly(true);
939 return true;
940 }
941
942 if (m_autoWrap && betweenWords) {
943 commitLineBreakAtCurrentWidth(renderer, m_current.offset(), m_current.nextBreakablePosition());
944 wrapWidthOffset = 0;
945 wrapW = wrapWidthOffset;
946 // Auto-wrapping text should not wrap in the middle of a word once it has had an
947 // opportunity to break after a word.
948 m_hasFormerOpportunity = !breakAnywhere;
949 breakWords = false;
950 canBreakMidWord = breakAll;
951 }
952
953 if (midWordBreak && !U16_IS_TRAIL(c) && !(U_GET_GC_MASK(c) & U_GC_M_MASK)) {
954 // Remember this as a breakable position in case
955 // adding the end width forces a break.
956 m_lineBreak.moveTo(renderer, m_current.offset(), m_current.nextBreakablePosition());
957 midWordBreak &= canBreakMidWord;
958 }
959
960 if (betweenWords) {
961 lastSpaceWordSpacing = applyWordSpacing ? wordSpacing : 0;
962 wordSpacingForWordMeasurement = (applyWordSpacing && wordMeasurement.width) ? wordSpacing : 0;
963 lastSpace = m_current.offset();
964 }
965
966 if (!m_ignoringSpaces && style.collapseWhiteSpace()) {
967 // If we encounter a newline, or if we encounter a second space,
968 // we need to break up this run and enter a mode where we start collapsing spaces.
969 if (m_currentCharacterIsSpace && previousCharacterIsSpace) {
970 m_ignoringSpaces = true;
971
972 // We just entered a mode where we are ignoring
973 // spaces. Create a transition to terminate the run
974 // before the second space.
975 m_lineWhitespaceCollapsingState.startIgnoringSpaces(m_startOfIgnoredSpaces);
976 m_trailingObjects.updateWhitespaceCollapsingTransitionsForTrailingBoxes(m_lineWhitespaceCollapsingState, InlineIterator(), TrailingObjects::CollapseFirstSpace::No);
977 }
978 }
979 // Measuring the width of complex text character-by-character, rather than measuring it all together,
980 // could produce considerably different width values.
981 if (!renderer.canUseSimpleFontCodePath() && midWordBreak && m_width.fitsOnLine()) {
982 midWordBreak = false;
983 wrapW = wrapWidthOffset + additionalTempWidth;
984 }
985 isLineEmpty = m_lineInfo.isEmpty();
986 } else {
987 if (m_ignoringSpaces) {
988 // Stop ignoring spaces and begin at this new point.
989 m_ignoringSpaces = false;
990 lastSpaceWordSpacing = applyWordSpacing ? wordSpacing : 0;
991 wordSpacingForWordMeasurement = (applyWordSpacing && wordMeasurements.last().width) ? wordSpacing : 0;
992 lastSpace = m_current.offset(); // e.g., "Foo goo", don't add in any of the ignored spaces.
993 m_lineWhitespaceCollapsingState.stopIgnoringSpaces(InlineIterator(nullptr, &renderer, m_current.offset()));
994 }
995 if (m_hangsAtEnd && !renderer.isHangableStopOrComma(c))
996 m_hangsAtEnd = false;
997 }
998
999 if (isSVGText && m_current.offset()) {
1000 // Force creation of new InlineBoxes for each absolute positioned character (those that start new text chunks).
1001 if (downcast<RenderSVGInlineText>(renderer).characterStartsNewTextChunk(m_current.offset()))
1002 ensureCharacterGetsLineBox(m_lineWhitespaceCollapsingState, m_current);
1003 }
1004
1005 if (m_currentCharacterIsSpace && !previousCharacterIsSpace) {
1006 m_startOfIgnoredSpaces.setRenderer(&renderer);
1007 m_startOfIgnoredSpaces.setOffset(m_current.offset());
1008 // Spaces after right-aligned text and before a line-break get collapsed away completely so that the trailing
1009 // space doesn't seem to push the text out from the right-hand edge.
1010 // FIXME: Do this regardless of the container's alignment - will require rebaselining a lot of test results.
1011 if (m_nextObject && m_startOfIgnoredSpaces.offset() && m_nextObject->isBR() && (m_blockStyle.textAlign() == TextAlignMode::Right || m_blockStyle.textAlign() == TextAlignMode::WebKitRight)) {
1012 m_startOfIgnoredSpaces.setOffset(m_startOfIgnoredSpaces.offset() - 1);
1013 // If there's just a single trailing space start ignoring it now so it collapses away.
1014 if (m_current.offset() == renderer.text().length() - 1)
1015 m_lineWhitespaceCollapsingState.startIgnoringSpaces(m_startOfIgnoredSpaces);
1016 }
1017 }
1018
1019 if (!m_currentCharacterIsWS && previousCharacterIsWS) {
1020 if (m_autoWrap && style.breakOnlyAfterWhiteSpace())
1021 m_lineBreak.moveTo(renderer, m_current.offset(), m_current.nextBreakablePosition());
1022 }
1023
1024 if (m_collapseWhiteSpace && m_currentCharacterIsSpace && !m_ignoringSpaces)
1025 m_trailingObjects.setTrailingWhitespace(renderer);
1026 else if (!style.collapseWhiteSpace() || !m_currentCharacterIsSpace)
1027 m_trailingObjects.clear();
1028
1029 m_atStart = false;
1030 nextCharacter(c, lastCharacter, secondToLastCharacter);
1031 }
1032
1033 m_renderTextInfo.lineBreakIterator.setPriorContext(lastCharacter, secondToLastCharacter);
1034
1035 wordMeasurements.grow(wordMeasurements.size() + 1);
1036 WordMeasurement& wordMeasurement = wordMeasurements.last();
1037 wordMeasurement.renderer = &renderer;
1038
1039 // IMPORTANT: current.m_pos is > length here!
1040 float additionalTempWidth = m_ignoringSpaces ? 0 : textWidth(renderer, lastSpace, m_current.offset() - lastSpace, font, m_width.currentWidth(), isFixedPitch, m_collapseWhiteSpace, wordMeasurement.fallbackFonts, textLayout);
1041 wordMeasurement.startOffset = lastSpace;
1042 wordMeasurement.endOffset = m_current.offset();
1043 wordMeasurement.width = m_ignoringSpaces ? 0 : additionalTempWidth + wordSpacingForWordMeasurement;
1044 additionalTempWidth += lastSpaceWordSpacing;
1045
1046 float inlineLogicalTempWidth = inlineLogicalWidth(renderer, !m_appliedStartWidth, m_includeEndWidth);
1047 m_width.addUncommittedWidth(additionalTempWidth + inlineLogicalTempWidth);
1048 if (m_hangsAtEnd && inlineLogicalTempWidth)
1049 m_hangsAtEnd = false;
1050
1051 if (wordMeasurement.fallbackFonts.isEmpty() && !fallbackFonts.isEmpty())
1052 wordMeasurement.fallbackFonts.swap(fallbackFonts);
1053 fallbackFonts.clear();
1054
1055 if (m_collapseWhiteSpace && m_currentCharacterIsSpace && additionalTempWidth)
1056 m_width.setTrailingWhitespaceWidth(additionalTempWidth, inlineLogicalTempWidth);
1057
1058 m_includeEndWidth = false;
1059

1097inline bool BreakingContext::canBreakAtThisPosition()
1098{
1099 // If we are no-wrap and have found a line-breaking opportunity already then we should take it.
1100 if (m_width.committedWidth() && !m_width.fitsOnLine(m_currentCharacterIsSpace) && m_currWS == WhiteSpace::NoWrap)
1101 return true;
1102
1103 // Avoid breaking on empty inlines.
1104 if (is<RenderInline>(*m_current.renderer()) && isEmptyInline(downcast<RenderInline>(*m_current.renderer())))
1105 return false;
1106
1107 // Avoid breaking before empty inlines (as long as the current object isn't replaced).
1108 if (!m_current.renderer()->isReplaced() && is<RenderInline>(m_nextObject) && isEmptyInline(downcast<RenderInline>(*m_nextObject)))
1109 return false;
1110
1111 // Return early if we autowrap and the current character is a space as we will always want to break at such a position.
1112 if (m_autoWrap && m_currentCharacterIsSpace)
1113 return true;
1114
1115 if (m_nextObject && m_nextObject->isLineBreakOpportunity())
1116 return m_autoWrap;
1117
1118 bool nextIsAutoWrappingText = is<RenderText>(m_nextObject) && (m_autoWrap || m_nextObject->style().autoWrap());
1119 if (!nextIsAutoWrappingText)
1120 return m_autoWrap;
1121 RenderText& nextRenderText = downcast<RenderText>(*m_nextObject);
1122 bool currentIsTextOrEmptyInline = is<RenderText>(*m_current.renderer()) || (is<RenderInline>(*m_current.renderer()) && isEmptyInline(downcast<RenderInline>(*m_current.renderer())));
1123 if (!currentIsTextOrEmptyInline)
1124 return m_autoWrap && !m_current.renderer()->isRubyRun();
1125
1126 bool canBreakHere = !m_currentCharacterIsSpace && textBeginsWithBreakablePosition(nextRenderText);
1127
1128 // See if attempting to fit below floats creates more available width on the line.
1129 if (!m_width.fitsOnLine() && !m_width.hasCommitted())
1130 m_width.fitBelowFloats(m_lineInfo.isFirstLine());
1131
1132 bool canPlaceOnLine = m_width.fitsOnLine() || !m_autoWrapWasEverTrueOnLine;
1133
1134 if (canPlaceOnLine && canBreakHere)
1135 commitLineBreakAtCurrentWidth(nextRenderText);
1136
1137 return canBreakHere;
1138}

1143
1144 if (checkForBreak && !m_width.fitsOnLine(m_ignoringSpaces) && !m_hangsAtEnd) {
1145 // if we have floats, try to get below them.
1146 if (m_currentCharacterIsSpace && !m_ignoringSpaces && m_collapseWhiteSpace)
1147 m_trailingObjects.clear();
1148
1149 if (m_width.committedWidth()) {
1150 m_atEnd = true;
1151 return;
1152 }
1153
1154 if (!m_hangsAtEnd)
1155 m_width.fitBelowFloats(m_lineInfo.isFirstLine());
1156
1157 // |width| may have been adjusted because we got shoved down past a float (thus
1158 // giving us more room), so we need to retest, and only jump to
1159 // the end label if we still don't fit on the line. -dwh
1160 if (!m_width.fitsOnLine(m_ignoringSpaces)) {
1161 m_atEnd = true;
1162 return;
1163 }

1166 // wraps and moves below any floats.
1167 m_width.fitBelowFloats(m_lineInfo.isFirstLine());
1168 }

1171 m_lastObject = m_current.renderer();

1181inline TrailingObjects::CollapseFirstSpace checkWhitespaceCollapsingTransitions(LineWhitespaceCollapsingState& lineWhitespaceCollapsingState, const InlineIterator& lBreak)
1182{
1183 // Check to see if our last transition is a start point beyond the line break. If so,
1184 // shave it off the list, and shave off a trailing space if the previous end point doesn't
1185 // preserve whitespace.
1186 if (lBreak.renderer() && lineWhitespaceCollapsingState.numTransitions() && !(lineWhitespaceCollapsingState.numTransitions() % 2)) {
1187 const InlineIterator* transitions = lineWhitespaceCollapsingState.transitions().data();
1188 const InlineIterator& endpoint = transitions[lineWhitespaceCollapsingState.numTransitions() - 2];
1189 const InlineIterator& startpoint = transitions[lineWhitespaceCollapsingState.numTransitions() - 1];
1190 InlineIterator currpoint = endpoint;
1191 while (!currpoint.atEnd() && currpoint != startpoint && currpoint != lBreak)
1192 currpoint.increment();
1193 if (currpoint == lBreak) {
1194 // We hit the line break before the start point. Shave off the start point.
1195 lineWhitespaceCollapsingState.decrementNumTransitions();
1196 if (endpoint.renderer()->style().collapseWhiteSpace() && endpoint.renderer()->isText()) {
1197 lineWhitespaceCollapsingState.decrementTransitionAt(lineWhitespaceCollapsingState.numTransitions() - 1);
1198 return TrailingObjects::CollapseFirstSpace::No;
1199 }
1200 }
1201 }
1202 return TrailingObjects::CollapseFirstSpace::Yes;

1205inline InlineIterator BreakingContext::handleEndOfLine()
1206{
1207 if (m_lineBreak == m_resolver.position()) {
1208 if (!m_lineBreak.renderer() || !m_lineBreak.renderer()->isBR()) {
1209 // we just add as much as possible
1210 if (m_blockStyle.whiteSpace() == WhiteSpace::Pre && !m_current.offset()) {
1211 if (m_lastObject)
1212 commitLineBreakAtCurrentWidth(*m_lastObject, m_lastObject->isText() ? m_lastObject->length() : 0);
1213 else
1214 commitLineBreakClear();
1215 } else if (m_lineBreak.renderer()) {
1216 // Don't ever break in the middle of a word if we can help it.
1217 // There's no room at all. We just have to be on this line,
1218 // even though we'll spill out.
1219 commitLineBreakAtCurrentWidth(*m_current.renderer(), m_current.offset());
1220 }
1221 }
1222 // make sure we consume at least one char/object.
1223 if (m_lineBreak == m_resolver.position())
1224 m_lineBreak.increment();
1225 } else if (!m_current.offset() && !m_width.committedWidth() && m_width.uncommittedWidth() && !m_hadUncommittedWidthBeforeCurrent) {
1226 // Do not push the current object to the next line, when this line has some content, but it is still considered empty.
1227 // Empty inline elements like <span></span> can produce such lines and now we just ignore these break opportunities
1228 // at the start of a line, if no width has been committed yet.
1229 // Behave as if it was actually empty and consume at least one object.
1230 auto overflowingBoxIsInlineLevelBox = m_lineBreak.renderer() && !m_lineBreak.renderer()->isFloatingOrOutOfFlowPositioned();
1231 if (overflowingBoxIsInlineLevelBox)
1232 m_lineBreak.increment();
1233 }
1234
1235 // Sanity check our whitespace collapsing transitions.
1236 auto collapsed = checkWhitespaceCollapsingTransitions(m_lineWhitespaceCollapsingState, m_lineBreak);
1237 m_trailingObjects.updateWhitespaceCollapsingTransitionsForTrailingBoxes(m_lineWhitespaceCollapsingState, m_lineBreak, collapsed);
1238
1239 // We might have made lineBreak an iterator that points past the end
1240 // of the object. Do this adjustment to make it point to the start
1241 // of the next object instead to avoid confusing the rest of the
1242 // code.
1243 if (m_lineBreak.offset()) {
1244 m_lineBreak.setOffset(m_lineBreak.offset() - 1);
1245 m_lineBreak.increment();
1246 }
1247
1248 return m_lineBreak;
1249}
