37void LineInfo::setEmpty(bool empty, RenderBlock* block, LineWidth* lineWidth)
38{
39 if (m_isEmpty == empty)
40 return;
41 m_isEmpty = empty;
42 if (!empty && block && floatPaginationStrut()) {
43 block->setLogicalHeight(block->logicalHeight() + floatPaginationStrut());
44 setFloatPaginationStrut(0);
45 lineWidth->updateAvailableWidth();
46 }
47}
