30#pragma once
31
32#include "LineWidth.h"
33
34namespace WebCore {
35
36class RenderBlock;
37
38class LineInfo {
39public:
40 LineInfo()
41 : m_isFirstLine(true)
42 , m_isLastLine(false)
43 , m_isEmpty(true)
44 , m_previousLineBrokeCleanly(true)
45 , m_floatPaginationStrut(0)
46 , m_runsFromLeadingWhitespace(0)
47 { }
48
49 bool isFirstLine() const { return m_isFirstLine; }
50 bool isLastLine() const { return m_isLastLine; }
51 bool isEmpty() const { return m_isEmpty; }
52 bool previousLineBrokeCleanly() const { return m_previousLineBrokeCleanly; }
53 LayoutUnit floatPaginationStrut() const { return m_floatPaginationStrut; }
54 unsigned runsFromLeadingWhitespace() const { return m_runsFromLeadingWhitespace; }
55 void resetRunsFromLeadingWhitespace() { m_runsFromLeadingWhitespace = 0; }
56 void incrementRunsFromLeadingWhitespace() { m_runsFromLeadingWhitespace++; }
57
58 void setFirstLine(bool firstLine) { m_isFirstLine = firstLine; }
59 void setLastLine(bool lastLine) { m_isLastLine = lastLine; }
60 void setEmpty(bool empty, RenderBlock* block = 0, LineWidth* lineWidth = 0);
61 void setPreviousLineBrokeCleanly(bool previousLineBrokeCleanly) { m_previousLineBrokeCleanly = previousLineBrokeCleanly; }
62 void setFloatPaginationStrut(LayoutUnit strut) { m_floatPaginationStrut = strut; }
63
64private:
65 bool m_isFirstLine;
66 bool m_isLastLine;
67 bool m_isEmpty;
68 bool m_previousLineBrokeCleanly;
69 LayoutUnit m_floatPaginationStrut;
70 unsigned m_runsFromLeadingWhitespace;
71};
72
73} // namespace WebCore
