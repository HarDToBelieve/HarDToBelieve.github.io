38LineWidth::LineWidth(RenderBlockFlow& block, bool isFirstLine, IndentTextOrNot shouldIndentText)
39 : m_block(block)
40 , m_isFirstLine(isFirstLine)
41 , m_shouldIndentText(shouldIndentText)
42{
43 updateAvailableWidth();
44}
45
46bool LineWidth::fitsOnLine(bool ignoringTrailingSpace) const
47{
48 return ignoringTrailingSpace ? fitsOnLineExcludingTrailingCollapsedWhitespace() : fitsOnLineIncludingExtraWidth(0);
49}
50
51bool LineWidth::fitsOnLineIncludingExtraWidth(float extra) const
52{
53 auto adjustedCurrentWidth = currentWidth() + extra;
54 return adjustedCurrentWidth < m_availableWidth || WTF::areEssentiallyEqual(adjustedCurrentWidth, m_availableWidth);
55}
56
57bool LineWidth::fitsOnLineExcludingTrailingWhitespace(float extra) const
58{
59 auto adjustedCurrentWidth = currentWidth() - m_trailingWhitespaceWidth + extra;
60 return adjustedCurrentWidth < m_availableWidth || WTF::areEssentiallyEqual(adjustedCurrentWidth, m_availableWidth);
61}
62
63bool LineWidth::fitsOnLineExcludingTrailingCollapsedWhitespace() const
64{
65 auto adjustedCurrentWidth = currentWidth() - m_trailingCollapsedWhitespaceWidth;
66 return adjustedCurrentWidth < m_availableWidth || WTF::areEssentiallyEqual(adjustedCurrentWidth, m_availableWidth);
67}

69void LineWidth::updateAvailableWidth(LayoutUnit replacedHeight)
70{
71 LayoutUnit height = m_block.logicalHeight();
72 LayoutUnit logicalHeight = m_block.minLineHeightForReplacedRenderer(m_isFirstLine, replacedHeight);
73 m_left = m_block.logicalLeftOffsetForLine(height, shouldIndentText(), logicalHeight);
74 m_right = m_block.logicalRightOffsetForLine(height, shouldIndentText(), logicalHeight);
75
76 computeAvailableWidthFromLeftAndRight();
77}

79static bool newFloatShrinksLine(const FloatingObject& newFloat, const RenderBlockFlow& block, bool isFirstLine)
80{
81 LayoutUnit blockOffset = block.logicalHeight();
82 if (blockOffset >= block.logicalTopForFloat(newFloat) && blockOffset < block.logicalBottomForFloat(newFloat))
83 return true;
84
85 // initial-letter float always shrinks the first line.
86 const auto& style = newFloat.renderer().style();
87 if (isFirstLine && style.styleType() == PseudoId::FirstLetter && !style.initialLetter().isEmpty())
88 return true;
89 return false;
90}

92void LineWidth::shrinkAvailableWidthForNewFloatIfNeeded(const FloatingObject& newFloat)
93{
94 if (!newFloatShrinksLine(newFloat, m_block, m_isFirstLine))
95 return;
96 ShapeOutsideDeltas shapeDeltas;
97 if (ShapeOutsideInfo* shapeOutsideInfo = newFloat.renderer().shapeOutsideInfo()) {
98 LayoutUnit lineHeight = m_block.lineHeight(m_isFirstLine, m_block.isHorizontalWritingMode() ? HorizontalLine : VerticalLine, PositionOfInteriorLineBoxes);
99 shapeDeltas = shapeOutsideInfo->computeDeltasForContainingBlockLine(m_block, newFloat, m_block.logicalHeight(), lineHeight);
100 }
101
102 if (newFloat.type() == FloatingObject::FloatLeft) {
103 float newLeft = m_block.logicalRightForFloat(newFloat);
104 if (shouldIndentText() == IndentText && m_block.style().isLeftToRightDirection())
105 newLeft += floorToInt(m_block.textIndentOffset());
106 if (shapeDeltas.isValid()) {
107 if (shapeDeltas.lineOverlapsShape())
108 newLeft += shapeDeltas.rightMarginBoxDelta();
109 else // If the line doesn't overlap the shape, then we need to act as if this float didn't exist.
110 newLeft = m_left;
111 }
112 m_left = std::max<float>(m_left, newLeft);
113 } else {
114 float newRight = m_block.logicalLeftForFloat(newFloat);
115 if (shouldIndentText() == IndentText && !m_block.style().isLeftToRightDirection())
116 newRight -= floorToInt(m_block.textIndentOffset());
117 if (shapeDeltas.isValid()) {
118 if (shapeDeltas.lineOverlapsShape())
119 newRight += shapeDeltas.leftMarginBoxDelta();
120 else // If the line doesn't overlap the shape, then we need to act as if this float didn't exist.
121 newRight = m_right;
122 }
123 m_right = std::min<float>(m_right, newRight);
124 }
125
126 computeAvailableWidthFromLeftAndRight();
127}

129void LineWidth::commit()
130{
131 m_committedWidth += m_uncommittedWidth;
132 m_uncommittedWidth = 0;
133 if (m_hasUncommittedReplaced) {
134 m_hasCommittedReplaced = true;
135 m_hasUncommittedReplaced = false;
136 }
137 m_hasCommitted = true;
138}

140void LineWidth::applyOverhang(const RenderRubyRun& rubyRun, RenderObject* startRenderer, RenderObject* endRenderer)
141{
142 float startOverhang;
143 float endOverhang;
144 rubyRun.getOverhang(m_isFirstLine, startRenderer, endRenderer, startOverhang, endOverhang);
145
146 startOverhang = std::min(startOverhang, m_committedWidth);
147 m_availableWidth += startOverhang;
148
149 endOverhang = std::max(std::min(endOverhang, m_availableWidth - currentWidth()), 0.0f);
150 m_availableWidth += endOverhang;
151 m_overhangWidth += startOverhang + endOverhang;
152}

154inline static float availableWidthAtOffset(const RenderBlockFlow& block, const LayoutUnit& offset, IndentTextOrNot shouldIndentText,
155 float& newLineLeft, float& newLineRight, const LayoutUnit& lineHeight = 0)
156{
157 newLineLeft = block.logicalLeftOffsetForLine(offset, shouldIndentText, lineHeight);
158 newLineRight = block.logicalRightOffsetForLine(offset, shouldIndentText, lineHeight);
159 return std::max(0.0f, newLineRight - newLineLeft);
160}

162void LineWidth::updateLineDimension(LayoutUnit newLineTop, LayoutUnit newLineWidth, float newLineLeft, float newLineRight)
163{
164 if (newLineWidth <= m_availableWidth)
165 return;
166
167 m_block.setLogicalHeight(newLineTop);
168 m_availableWidth = newLineWidth + m_overhangWidth;
169 m_left = newLineLeft;
170 m_right = newLineRight;
171}

173void LineWidth::wrapNextToShapeOutside(bool isFirstLine)
174{
175 LayoutUnit lineHeight = m_block.lineHeight(isFirstLine, m_block.isHorizontalWritingMode() ? HorizontalLine : VerticalLine, PositionOfInteriorLineBoxes);
176 LayoutUnit lineLogicalTop = m_block.logicalHeight();
177 LayoutUnit newLineTop = lineLogicalTop;
178 LayoutUnit floatLogicalBottom = m_block.nextFloatLogicalBottomBelow(lineLogicalTop);
179
180 float newLineWidth;
181 float newLineLeft = m_left;
182 float newLineRight = m_right;
183 while (true) {
184 newLineWidth = availableWidthAtOffset(m_block, newLineTop, shouldIndentText(), newLineLeft, newLineRight, lineHeight);
185 if (newLineWidth >= m_uncommittedWidth)
186 break;
187
188 if (newLineTop >= floatLogicalBottom)
189 break;
190
191 ++newLineTop;
192 }
193 updateLineDimension(newLineTop, LayoutUnit(newLineWidth), LayoutUnit(newLineLeft), LayoutUnit(newLineRight));
194}

196void LineWidth::fitBelowFloats(bool isFirstLine)
197{
198 ASSERT(!m_committedWidth);
199 ASSERT(!fitsOnLine());
200
201 LayoutUnit floatLogicalBottom;
202 LayoutUnit lastFloatLogicalBottom = m_block.logicalHeight();
203 float newLineWidth = m_availableWidth;
204 float newLineLeft = m_left;
205 float newLineRight = m_right;
206
207 FloatingObject* lastFloatFromPreviousLine = (m_block.containsFloats() ? m_block.m_floatingObjects->set().last().get() : 0);
208 if (lastFloatFromPreviousLine && lastFloatFromPreviousLine->renderer().shapeOutsideInfo())
209 return wrapNextToShapeOutside(isFirstLine);
210
211 while (true) {
212 floatLogicalBottom = m_block.nextFloatLogicalBottomBelow(lastFloatLogicalBottom);
213 if (floatLogicalBottom <= lastFloatLogicalBottom)
214 break;
215
216 newLineWidth = availableWidthAtOffset(m_block, floatLogicalBottom, shouldIndentText(), newLineLeft, newLineRight);
217 lastFloatLogicalBottom = floatLogicalBottom;
218
219 if (newLineWidth >= m_uncommittedWidth)
220 break;
221 }
222
223 updateLineDimension(lastFloatLogicalBottom, LayoutUnit(newLineWidth), LayoutUnit(newLineLeft), LayoutUnit(newLineRight));
224}

226void LineWidth::setTrailingWhitespaceWidth(float collapsedWhitespace, float borderPaddingMargin)
227{
228 m_trailingCollapsedWhitespaceWidth = collapsedWhitespace;
229 m_trailingWhitespaceWidth = collapsedWhitespace + borderPaddingMargin;
230}
231
232void LineWidth::computeAvailableWidthFromLeftAndRight()
233{
234 m_availableWidth = std::max<float>(0, m_right - m_left) + m_overhangWidth;
235}

237IndentTextOrNot requiresIndent(bool isFirstLine, bool isAfterHardLineBreak, const RenderStyle& style)
238{
239 IndentTextOrNot shouldIndentText = DoNotIndentText;
240 if (isFirstLine)
241 shouldIndentText = IndentText;
242 else if (isAfterHardLineBreak && style.textIndentLine() == TextIndentLine::EachLine)
243 shouldIndentText = IndentText;
244
245 if (style.textIndentType() == TextIndentType::Hanging)
246 shouldIndentText = shouldIndentText == IndentText ? DoNotIndentText : IndentText;
247 return shouldIndentText;
248}
