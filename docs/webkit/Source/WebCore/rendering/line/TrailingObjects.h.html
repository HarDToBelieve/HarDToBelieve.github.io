25#pragma once
26
27#include <wtf/Vector.h>
28
29namespace WebCore {
30
31class InlineIterator;
32class RenderBoxModelObject;
33class RenderText;
34
35struct BidiRun;
36struct BidiIsolatedRun;
37
38template <class Iterator, class Run> class BidiResolver;
39template <class Iterator, class Run, class IsolateRun> class BidiResolverWithIsolate;
40template <class Iterator> class WhitespaceCollapsingState;
41typedef BidiResolverWithIsolate<InlineIterator, BidiRun, BidiIsolatedRun> InlineBidiResolver;
42typedef WhitespaceCollapsingState<InlineIterator> LineWhitespaceCollapsingState;
43
44class TrailingObjects {
45public:
46 void setTrailingWhitespace(RenderText& whitespace) { m_whitespace = &whitespace; }
47 void clear()
48 {
49 m_whitespace = { };
50 m_boxes.shrink(0); // Use shrink(0) instead of clear() to retain our capacity.
51 }
52
53 void appendBoxIfNeeded(RenderBoxModelObject& box)
54 {
55 if (m_whitespace)
56 m_boxes.append(box);
57 }
58
59 enum class CollapseFirstSpace { No, Yes };
60 void updateWhitespaceCollapsingTransitionsForTrailingBoxes(LineWhitespaceCollapsingState&, const InlineIterator& lBreak, CollapseFirstSpace);
61
62private:
63 RenderText* m_whitespace { nullptr };
64 Vector<std::reference_wrapper<RenderBoxModelObject>, 4> m_boxes;
65};
66
67} // namespace WebCore
