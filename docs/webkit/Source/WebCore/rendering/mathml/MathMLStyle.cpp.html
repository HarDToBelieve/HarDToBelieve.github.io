50const MathMLStyle* MathMLStyle::getMathMLStyle(RenderObject* renderer)
51{
52 // FIXME: Should we make RenderMathMLTable derive from RenderMathMLBlock in order to simplify this?
53 if (is<RenderMathMLTable>(renderer))
54 return &downcast<RenderMathMLTable>(*renderer).mathMLStyle();
55 if (is<RenderMathMLBlock>(renderer))
56 return &downcast<RenderMathMLBlock>(*renderer).mathMLStyle();
57 return nullptr;
58}

60void MathMLStyle::resolveMathMLStyleTree(RenderObject* renderer)
61{
62 for (auto* child = renderer; child; child = child->nextInPreOrder(renderer)) {
63 // FIXME: Should we make RenderMathMLTable derive from RenderMathMLBlock in order to simplify this?
64 if (is<RenderMathMLTable>(child))
65 downcast<RenderMathMLTable>(*child).mathMLStyle().resolveMathMLStyle(child);
66 else if (is<RenderMathMLBlock>(child))
67 downcast<RenderMathMLBlock>(*child).mathMLStyle().resolveMathMLStyle(child);
68 }
69}

71RenderObject* MathMLStyle::getMathMLParentNode(RenderObject* renderer)
72{
73 auto* parentRenderer = renderer->parent();
74
75 while (parentRenderer && !(is<RenderMathMLTable>(parentRenderer) || is<RenderMathMLBlock>(parentRenderer)))
76 parentRenderer = parentRenderer->parent();
77
78 return parentRenderer;
79}

81void MathMLStyle::updateStyleIfNeeded(RenderObject* renderer, MathMLElement::MathVariant oldMathVariant)
82{
83 // RenderMathMLFencedOperator does not support mathvariant transforms.
84 // See https://bugs.webkit.org/show_bug.cgi?id=160509#c1.
85 bool isNonAnonymousTokenElement = is<RenderMathMLToken>(renderer) && !renderer->isAnonymous();
86
87 if (oldMathVariant != m_mathVariant) {
88 if (isNonAnonymousTokenElement)
89 downcast<RenderMathMLToken>(renderer)->updateTokenContent();
90 }
91}
92
93void MathMLStyle::resolveMathMLStyle(RenderObject* renderer)
94{
95 ASSERT(renderer);
96
97 MathMLElement::MathVariant oldMathVariant = m_mathVariant;
98 auto* parentRenderer = getMathMLParentNode(renderer);
99 const MathMLStyle* parentStyle = getMathMLStyle(parentRenderer);
100
101 // By default, we just inherit the style from our parent.
102 m_mathVariant = MathMLElement::MathVariant::None;
103 if (parentStyle) {
104 setMathVariant(parentStyle->mathVariant());
105 }
106
107 // Early return for anonymous renderers.
108 if (renderer->isAnonymous()) {
109 updateStyleIfNeeded(renderer, oldMathVariant);
110 return;
111 }
112
113 // The mathvariant attributes override the default behavior.
114 auto* element = downcast<RenderElement>(renderer)->element();
115 if (is<MathMLElement>(element)) {
116 std::optional<MathMLElement::MathVariant> mathVariant = downcast<MathMLElement>(element)->specifiedMathVariant();
117 if (mathVariant)
118 m_mathVariant = mathVariant.value();
119 }
120 updateStyleIfNeeded(renderer, oldMathVariant);
121}
