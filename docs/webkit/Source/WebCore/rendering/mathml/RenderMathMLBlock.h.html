42class RenderMathMLBlock : public RenderBlock {
43 WTF_MAKE_ISO_ALLOCATED(RenderMathMLBlock);
44public:
45 RenderMathMLBlock(MathMLPresentationElement&, RenderStyle&&);
46 RenderMathMLBlock(Document&, RenderStyle&&);
47 virtual ~RenderMathMLBlock();
48
49 MathMLStyle& mathMLStyle() const { return m_mathMLStyle; }
50
51 bool isChildAllowed(const RenderObject&, const RenderStyle&) const override;
52
53 // MathML defines an "embellished operator" as roughly an <mo> that may have subscripts,
54 // superscripts, underscripts, overscripts, or a denominator (as in d/dx, where "d" is some
55 // differential operator). The padding, precedence, and stretchiness of the base <mo> should
56 // apply to the embellished operator as a whole. unembellishedOperator() checks for being an
57 // embellished operator, and omits any embellishments.
58 // FIXME: We don't yet handle all the cases in the MathML spec. See
59 // https://bugs.webkit.org/show_bug.cgi?id=78617.
60 virtual RenderMathMLOperator* unembellishedOperator() const { return 0; }
61
62 LayoutUnit baselinePosition(FontBaseline, bool firstLine, LineDirectionMode, LinePositionMode = PositionOnContainingLine) const override;
63
64#if ENABLE(DEBUG_MATH_LAYOUT)
65 virtual void paint(PaintInfo&, const LayoutPoint&);
66#endif
67
68protected:
69 void styleDidChange(StyleDifference, const RenderStyle* oldStyle) override;
70
71 LayoutUnit ruleThicknessFallback() const
72 {
73 // This function returns a value for the default rule thickness (TeX's \xi_8) to be used as a fallback when we lack a MATH table.
74 // This arbitrary value of 0.05em was used in early WebKit MathML implementations for the thickness of the fraction bars.
75 // Note that Gecko has a slower but more accurate version that measures the thickness of U+00AF MACRON to be more accurate and otherwise fallback to some arbitrary value.
76 return LayoutUnit(0.05f * style().fontCascade().size());
77 }
78
79 LayoutUnit mathAxisHeight() const;
80 LayoutUnit mirrorIfNeeded(LayoutUnit horizontalOffset, LayoutUnit boxWidth = 0_lu) const;
81 LayoutUnit mirrorIfNeeded(LayoutUnit horizontalOffset, const RenderBox& child) const { return mirrorIfNeeded(horizontalOffset, child.logicalWidth()); }
82
83 static LayoutUnit ascentForChild(const RenderBox& child)
84 {
85 return child.firstLineBaseline().value_or(child.logicalHeight().toInt());
86 }
87
88 void layoutBlock(bool relayoutChildren, LayoutUnit pageLogicalHeight = 0_lu) override;
89 void layoutInvalidMarkup(bool relayoutChildren);
90
91private:
92 bool isRenderMathMLBlock() const final { return true; }
93 const char* renderName() const override { return "RenderMathMLBlock"; }
94 bool avoidsFloats() const final { return true; }
95 bool canDropAnonymousBlockChild() const final { return false; }
96 void layoutItems(bool relayoutChildren);
97
98 Ref<MathMLStyle> m_mathMLStyle;
99};
