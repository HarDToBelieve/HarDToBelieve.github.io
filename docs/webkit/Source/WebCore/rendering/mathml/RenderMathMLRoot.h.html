28
29#if ENABLE(MATHML)
30
31#include "MathOperator.h"
32#include "RenderMathMLRow.h"
33
34namespace WebCore {
35
36class MathMLRootElement;
37
38enum class RootType { SquareRoot, RootWithIndex };
39
40// Render base^(1/index), or sqrt(base) using radical notation.
41class RenderMathMLRoot final : public RenderMathMLRow {
42 WTF_MAKE_ISO_ALLOCATED(RenderMathMLRoot);
43public:
44 RenderMathMLRoot(MathMLRootElement&, RenderStyle&&);
45 void updateStyle();
46
47private:
48 bool isValid() const;
49 RenderBox& getBase() const;
50 RenderBox& getIndex() const;
51 bool isRenderMathMLRoot() const final { return true; }
52 const char* renderName() const final { return "RenderMathMLRoot"; }
53 MathMLRootElement& element() const;
54 RootType rootType() const;
55
56 void styleDidChange(StyleDifference, const RenderStyle* oldStyle) final;
57
58 void computePreferredLogicalWidths() final;
59 void layoutBlock(bool relayoutChildren, LayoutUnit pageLogicalHeight = 0_lu) final;
60 void paint(PaintInfo&, const LayoutPoint&) final;
61
62 struct HorizontalParameters {
63 LayoutUnit kernBeforeDegree;
64 LayoutUnit kernAfterDegree;
65 };
66 HorizontalParameters horizontalParameters();
67 struct VerticalParameters {
68 LayoutUnit verticalGap;
69 LayoutUnit ruleThickness;
70 LayoutUnit extraAscender;
71 float degreeBottomRaisePercent;
72 };
73 VerticalParameters verticalParameters();
74
75 MathOperator m_radicalOperator;
76 LayoutUnit m_radicalOperatorTop;
77 LayoutUnit m_baseWidth;
78
79 bool isRenderMathMLSquareRoot() const final { return rootType() == RootType::SquareRoot; }
80};
81
82} // namespace WebCore
83
84SPECIALIZE_TYPE_TRAITS_RENDER_OBJECT(RenderMathMLRoot, isRenderMathMLRoot())
85
86#endif // ENABLE(MATHML)
