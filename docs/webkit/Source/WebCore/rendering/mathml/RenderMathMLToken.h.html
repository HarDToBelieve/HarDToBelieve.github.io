36class MathMLTokenElement;

38class RenderMathMLToken : public RenderMathMLBlock {
39 WTF_MAKE_ISO_ALLOCATED(RenderMathMLToken);
40public:
41 RenderMathMLToken(MathMLTokenElement&, RenderStyle&&);
42 RenderMathMLToken(Document&, RenderStyle&&);
43
44 MathMLTokenElement& element();
45
46 virtual void updateTokenContent();
47 void updateFromElement() override;
48
49protected:
50 void paint(PaintInfo&, const LayoutPoint&) override;
51 void paintChildren(PaintInfo& forSelf, const LayoutPoint&, PaintInfo& forChild, bool usePrintRect) override;
52 std::optional<LayoutUnit> firstLineBaseline() const override;
53 void layoutBlock(bool relayoutChildren, LayoutUnit pageLogicalHeight = 0_lu) override;
54 void computePreferredLogicalWidths() override;
55
56private:
57 bool isRenderMathMLToken() const final { return true; }
58 const char* renderName() const override { return "RenderMathMLToken"; }
59 bool isChildAllowed(const RenderObject&, const RenderStyle&) const final { return true; };
60 void styleDidChange(StyleDifference, const RenderStyle* oldStyle) override;
61 void updateMathVariantGlyph();
62 void setMathVariantGlyphDirty()
63 {
64 m_mathVariantGlyphDirty = true;
65 setNeedsLayoutAndPrefWidthsRecalc();
66 }
67 std::optional<UChar32> m_mathVariantCodePoint { std::nullopt };
68 bool m_mathVariantIsMirrored { false };
69 bool m_mathVariantGlyphDirty { false };
70};

72} // namespace WebCore
