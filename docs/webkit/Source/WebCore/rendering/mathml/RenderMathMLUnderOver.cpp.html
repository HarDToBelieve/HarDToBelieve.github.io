53static RenderMathMLOperator* horizontalStretchyOperator(const RenderBox& box)
54{
55 if (!is<RenderMathMLBlock>(box))
56 return nullptr;
57
58 auto* renderOperator = downcast<RenderMathMLBlock>(box).unembellishedOperator();
59 if (!renderOperator)
60 return nullptr;
61
62 if (!renderOperator->isStretchy() || renderOperator->isVertical() || renderOperator->isStretchWidthLocked())
63 return nullptr;
64
65 return renderOperator;
66}
67
68static void fixLayoutAfterStretch(RenderBox& ancestor, RenderMathMLOperator& stretchyOperator)
69{
70 stretchyOperator.setStretchWidthLocked(true);
71 stretchyOperator.setNeedsLayout();
72 ancestor.layoutIfNeeded();
73 stretchyOperator.setStretchWidthLocked(false);
74}

76void RenderMathMLUnderOver::stretchHorizontalOperatorsAndLayoutChildren()
77{
78 ASSERT(isValid());
79 ASSERT(needsLayout());
80
81 // We apply horizontal stretchy rules from the MathML spec (sections 3.2.5.8.3 and 3.2.5.8.4), which
82 // can be roughly summarized as "stretching opersators to the maximum widths of all children" and
83 // minor variations of that algorithm do not affect the result. However, the spec is a bit ambiguous
84 // for embellished operators (section 3.2.5.7.3) and different approaches can lead to significant
85 // stretch size differences. We made the following decisions:
86 // - The unstretched size is the embellished operator width with the <mo> at the core unstretched.
87 // - In general, the target size is just the maximum widths of non-stretchy children because the
88 // embellishments could make widths significantly larger.
89 // - In the edge case when all operators of stretchy, we follow the specification and take the
90 // maximum of all unstretched sizes.
91 // - The <mo> at the core is stretched to cover the target size, even if the embellished operator
92 // might become much wider.
93
94 Vector<RenderBox*, 3> embellishedOperators;
95 Vector<RenderMathMLOperator*, 3> stretchyOperators;
96 bool isAllStretchyOperators = true;
97 LayoutUnit stretchWidth;
98
99 for (auto* child = firstChildBox(); child; child = child->nextSiblingBox()) {
100 if (auto* stretchyOperator = horizontalStretchyOperator(*child)) {
101 embellishedOperators.append(child);
102 stretchyOperators.append(stretchyOperator);
103 } else {
104 isAllStretchyOperators = false;
105 child->layoutIfNeeded();
106 stretchWidth = std::max(stretchWidth, child->logicalWidth());
107 }
108 }
109
110 if (isAllStretchyOperators) {
111 for (size_t i = 0; i < embellishedOperators.size(); i++) {
112 stretchyOperators[i]->resetStretchSize();
113 fixLayoutAfterStretch(*embellishedOperators[i], *stretchyOperators[i]);
114 stretchWidth = std::max(stretchWidth, embellishedOperators[i]->logicalWidth());
115 }
116 }
117
118 for (size_t i = 0; i < embellishedOperators.size(); i++) {
119 stretchyOperators[i]->stretchTo(stretchWidth);
120 fixLayoutAfterStretch(*embellishedOperators[i], *stretchyOperators[i]);
121 }
122}

124bool RenderMathMLUnderOver::isValid() const
125{
126 // Verify whether the list of children is valid:
127 // <munder> base under </munder>
128 // <mover> base over </mover>
129 // <munderover> base under over </munderover>
130 auto* child = firstChildBox();
131 if (!child)
132 return false;
133 child = child->nextSiblingBox();
134 if (!child)
135 return false;
136 child = child->nextSiblingBox();
137 switch (scriptType()) {
138 case MathMLScriptsElement::ScriptType::Over:
139 case MathMLScriptsElement::ScriptType::Under:
140 return !child;
141 case MathMLScriptsElement::ScriptType::UnderOver:
142 return child && !child->nextSiblingBox();
143 default:
144 ASSERT_NOT_REACHED();
145 return false;
146 }
147}

149bool RenderMathMLUnderOver::shouldMoveLimits() const
150{
151 if (style().mathStyle() == MathStyle::Normal)
152 return false;
153 if (auto* renderOperator = unembellishedOperator())
154 return renderOperator->shouldMoveLimits();
155 return false;
156}

158RenderBox& RenderMathMLUnderOver::base() const
159{
160 ASSERT(isValid());
161 return *firstChildBox();
162}

164RenderBox& RenderMathMLUnderOver::under() const
165{
166 ASSERT(isValid());
167 ASSERT(scriptType() == MathMLScriptsElement::ScriptType::Under || scriptType() == MathMLScriptsElement::ScriptType::UnderOver);
168 return *firstChildBox()->nextSiblingBox();
169}

171RenderBox& RenderMathMLUnderOver::over() const
172{
173 ASSERT(isValid());
174 ASSERT(scriptType() == MathMLScriptsElement::ScriptType::Over || scriptType() == MathMLScriptsElement::ScriptType::UnderOver);
175 auto* secondChild = firstChildBox()->nextSiblingBox();
176 return scriptType() == MathMLScriptsElement::ScriptType::Over ? *secondChild : *secondChild->nextSiblingBox();
177}

180void RenderMathMLUnderOver::computePreferredLogicalWidths()
181{
182 ASSERT(preferredLogicalWidthsDirty());
183
184 if (!isValid()) {
185 m_minPreferredLogicalWidth = m_maxPreferredLogicalWidth = 0;
186 setPreferredLogicalWidthsDirty(false);
187 return;
188 }
189
190 if (shouldMoveLimits()) {
191 RenderMathMLScripts::computePreferredLogicalWidths();
192 return;
193 }
194
195 LayoutUnit preferredWidth = base().maxPreferredLogicalWidth();
196
197 if (scriptType() == MathMLScriptsElement::ScriptType::Under || scriptType() == MathMLScriptsElement::ScriptType::UnderOver)
198 preferredWidth = std::max(preferredWidth, under().maxPreferredLogicalWidth());
199
200 if (scriptType() == MathMLScriptsElement::ScriptType::Over || scriptType() == MathMLScriptsElement::ScriptType::UnderOver)
201 preferredWidth = std::max(preferredWidth, over().maxPreferredLogicalWidth());
202
203 m_minPreferredLogicalWidth = m_maxPreferredLogicalWidth = preferredWidth;
204
205 setPreferredLogicalWidthsDirty(false);
206}

208LayoutUnit RenderMathMLUnderOver::horizontalOffset(const RenderBox& child) const
209{
210 return (logicalWidth() - child.logicalWidth()) / 2;

213bool RenderMathMLUnderOver::hasAccent(bool accentUnder) const
214{
215 ASSERT(scriptType() == MathMLScriptsElement::ScriptType::UnderOver || (accentUnder && scriptType() == MathMLScriptsElement::ScriptType::Under) || (!accentUnder && scriptType() == MathMLScriptsElement::ScriptType::Over));
216
217 const MathMLElement::BooleanValue& attributeValue = accentUnder ? element().accentUnder() : element().accent();
218 if (attributeValue == MathMLElement::BooleanValue::True)
219 return true;
220 if (attributeValue == MathMLElement::BooleanValue::False)
221 return false;
222 RenderBox& script = accentUnder ? under() : over();
223 if (!is<RenderMathMLBlock>(script))
224 return false;
225 auto* scriptOperator = downcast<RenderMathMLBlock>(script).unembellishedOperator();
226 return scriptOperator && scriptOperator->hasOperatorFlag(MathMLOperatorDictionary::Accent);
227}

229RenderMathMLUnderOver::VerticalParameters RenderMathMLUnderOver::verticalParameters() const
230{
231 VerticalParameters parameters;
232
233 // By default, we set all values to zero.
234 parameters.underGapMin = 0;
235 parameters.overGapMin = 0;
236 parameters.underShiftMin = 0;
237 parameters.overShiftMin = 0;
238 parameters.underExtraDescender = 0;
239 parameters.overExtraAscender = 0;
240 parameters.accentBaseHeight = 0;
241
242 const auto& primaryFont = style().fontCascade().primaryFont();
243 auto* mathData = primaryFont.mathData();
244 if (!mathData) {
245 // The MATH table specification does not really provide any suggestions, except for some underbar/overbar values and AccentBaseHeight.
246 LayoutUnit defaultLineThickness = ruleThicknessFallback();
247 parameters.underGapMin = 3 * defaultLineThickness;
248 parameters.overGapMin = 3 * defaultLineThickness;
249 parameters.underExtraDescender = defaultLineThickness;
250 parameters.overExtraAscender = defaultLineThickness;
251 parameters.accentBaseHeight = style().fontMetrics().xHeight();
252 parameters.useUnderOverBarFallBack = true;
253 return parameters;
254 }
255
256 if (is<RenderMathMLBlock>(base())) {
257 if (auto* baseOperator = downcast<RenderMathMLBlock>(base()).unembellishedOperator()) {
258 if (baseOperator->hasOperatorFlag(MathMLOperatorDictionary::LargeOp)) {
259 // The base is a large operator so we read UpperLimit/LowerLimit constants from the MATH table.
260 parameters.underGapMin = mathData->getMathConstant(primaryFont, OpenTypeMathData::LowerLimitGapMin);
261 parameters.overGapMin = mathData->getMathConstant(primaryFont, OpenTypeMathData::UpperLimitGapMin);
262 parameters.underShiftMin = mathData->getMathConstant(primaryFont, OpenTypeMathData::LowerLimitBaselineDropMin);
263 parameters.overShiftMin = mathData->getMathConstant(primaryFont, OpenTypeMathData::UpperLimitBaselineRiseMin);
264 parameters.useUnderOverBarFallBack = false;
265 return parameters;
266 }
267 if (baseOperator->isStretchy() && !baseOperator->isVertical()) {
268 // The base is a horizontal stretchy operator, so we read StretchStack constants from the MATH table.
269 parameters.underGapMin = mathData->getMathConstant(primaryFont, OpenTypeMathData::StretchStackGapBelowMin);
270 parameters.overGapMin = mathData->getMathConstant(primaryFont, OpenTypeMathData::StretchStackGapAboveMin);
271 parameters.underShiftMin = mathData->getMathConstant(primaryFont, OpenTypeMathData::StretchStackBottomShiftDown);
272 parameters.overShiftMin = mathData->getMathConstant(primaryFont, OpenTypeMathData::StretchStackTopShiftUp);
273 parameters.useUnderOverBarFallBack = false;
274 return parameters;
275 }
276 }
277 }
278
279 // By default, we just use the underbar/overbar constants.
280 parameters.underGapMin = mathData->getMathConstant(primaryFont, OpenTypeMathData::UnderbarVerticalGap);
281 parameters.overGapMin = mathData->getMathConstant(primaryFont, OpenTypeMathData::OverbarVerticalGap);
282 parameters.underExtraDescender = mathData->getMathConstant(primaryFont, OpenTypeMathData::UnderbarExtraDescender);
283 parameters.overExtraAscender = mathData->getMathConstant(primaryFont, OpenTypeMathData::OverbarExtraAscender);
284 parameters.accentBaseHeight = mathData->getMathConstant(primaryFont, OpenTypeMathData::AccentBaseHeight);
285 parameters.useUnderOverBarFallBack = true;
286 return parameters;
287}

289void RenderMathMLUnderOver::layoutBlock(bool relayoutChildren, LayoutUnit pageLogicalHeight)
290{
291 ASSERT(needsLayout());
292
293 if (!relayoutChildren && simplifiedLayout())
294 return;
295
296 if (!isValid()) {
297 layoutInvalidMarkup(relayoutChildren);
298 return;
299 }
300
301 if (shouldMoveLimits()) {
302 RenderMathMLScripts::layoutBlock(relayoutChildren, pageLogicalHeight);
303 return;
304 }
305
306 recomputeLogicalWidth();
307
308 stretchHorizontalOperatorsAndLayoutChildren();
309
310 ASSERT(!base().needsLayout());
311 ASSERT(scriptType() == MathMLScriptsElement::ScriptType::Over || !under().needsLayout());
312 ASSERT(scriptType() == MathMLScriptsElement::ScriptType::Under || !over().needsLayout());
313
314 LayoutUnit logicalWidth = base().logicalWidth();
315 if (scriptType() == MathMLScriptsElement::ScriptType::Under || scriptType() == MathMLScriptsElement::ScriptType::UnderOver)
316 logicalWidth = std::max(logicalWidth, under().logicalWidth());
317 if (scriptType() == MathMLScriptsElement::ScriptType::Over || scriptType() == MathMLScriptsElement::ScriptType::UnderOver)
318 logicalWidth = std::max(logicalWidth, over().logicalWidth());
319 setLogicalWidth(logicalWidth);
320
321 VerticalParameters parameters = verticalParameters();
322 LayoutUnit verticalOffset;
323 if (scriptType() == MathMLScriptsElement::ScriptType::Over || scriptType() == MathMLScriptsElement::ScriptType::UnderOver) {
324 verticalOffset += parameters.overExtraAscender;
325 over().setLocation(LayoutPoint(horizontalOffset(over()), verticalOffset));
326 if (parameters.useUnderOverBarFallBack) {
327 verticalOffset += over().logicalHeight();
328 if (hasAccent()) {
329 LayoutUnit baseAscent = ascentForChild(base());
330 if (baseAscent < parameters.accentBaseHeight)
331 verticalOffset += parameters.accentBaseHeight - baseAscent;
332 } else
333 verticalOffset += parameters.overGapMin;
334 } else {
335 LayoutUnit overAscent = ascentForChild(over());
336 verticalOffset += std::max(over().logicalHeight() + parameters.overGapMin, overAscent + parameters.overShiftMin);
337 }
338 }
339 base().setLocation(LayoutPoint(horizontalOffset(base()), verticalOffset));
340 verticalOffset += base().logicalHeight();
341 if (scriptType() == MathMLScriptsElement::ScriptType::Under || scriptType() == MathMLScriptsElement::ScriptType::UnderOver) {
342 if (parameters.useUnderOverBarFallBack) {
343 if (!hasAccentUnder())
344 verticalOffset += parameters.underGapMin;
345 } else {
346 LayoutUnit underAscent = ascentForChild(under());
347 verticalOffset += std::max(parameters.underGapMin, parameters.underShiftMin - underAscent);
348 }
349 under().setLocation(LayoutPoint(horizontalOffset(under()), verticalOffset));
350 verticalOffset += under().logicalHeight();
351 verticalOffset += parameters.underExtraDescender;
352 }
353
354 setLogicalHeight(verticalOffset);
355
356 layoutPositionedObjects(relayoutChildren);
357
358 updateScrollInfoAfterLayout();
359
360 clearNeedsLayout();
361}
