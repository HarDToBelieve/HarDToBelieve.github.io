38static inline LayoutUnit adjustRadiusForMarginBoxShape(LayoutUnit radius, LayoutUnit margin)
39{
40 // This algorithm is defined in the CSS Shapes specifcation
41 if (!margin)
42 return radius;
43
44 LayoutUnit ratio = radius / margin;
45 if (ratio < 1)
46 return LayoutUnit(radius + (margin * (1 + pow(ratio - 1, 3.0))));
47
48 return radius + margin;
49}
50
51static inline LayoutSize computeMarginBoxShapeRadius(const LayoutSize& radius, const LayoutSize& adjacentMargins)
52{
53 return LayoutSize(adjustRadiusForMarginBoxShape(radius.width(), adjacentMargins.width()),
54 adjustRadiusForMarginBoxShape(radius.height(), adjacentMargins.height()));
55}
56
57static inline RoundedRect::Radii computeMarginBoxShapeRadii(const RoundedRect::Radii& radii, const RenderBox& renderer)
58{
59 return RoundedRect::Radii(computeMarginBoxShapeRadius(radii.topLeft(), LayoutSize(renderer.marginLeft(), renderer.marginTop())),
60 computeMarginBoxShapeRadius(radii.topRight(), LayoutSize(renderer.marginRight(), renderer.marginTop())),
61 computeMarginBoxShapeRadius(radii.bottomLeft(), LayoutSize(renderer.marginLeft(), renderer.marginBottom())),
62 computeMarginBoxShapeRadius(radii.bottomRight(), LayoutSize(renderer.marginRight(), renderer.marginBottom())));
63}

65RoundedRect computeRoundedRectForBoxShape(CSSBoxType box, const RenderBox& renderer)
66{
67 const RenderStyle& style = renderer.style();
68 switch (box) {
69 case CSSBoxType::MarginBox: {
70 if (!style.hasBorderRadius())
71 return RoundedRect(renderer.marginBoxRect(), RoundedRect::Radii());
72
73 LayoutRect marginBox = renderer.marginBoxRect();
74 RoundedRect::Radii radii = computeMarginBoxShapeRadii(style.getRoundedBorderFor(renderer.borderBoxRect()).radii(), renderer);
75 radii.scale(calcBorderRadiiConstraintScaleFor(marginBox, radii));
76 return RoundedRect(marginBox, radii);
77 }
78 case CSSBoxType::PaddingBox:
79 return style.getRoundedInnerBorderFor(renderer.borderBoxRect());
80 // fill-box compute to content-box for HTML elements.
81 case CSSBoxType::FillBox:
82 case CSSBoxType::ContentBox:
83 return style.getRoundedInnerBorderFor(renderer.borderBoxRect(),
84 renderer.paddingTop() + renderer.borderTop(), renderer.paddingBottom() + renderer.borderBottom(),
85 renderer.paddingLeft() + renderer.borderLeft(), renderer.paddingRight() + renderer.borderRight());
86 // stroke-box, view-box compute to border-box for HTML elements.
87 case CSSBoxType::BorderBox:
88 case CSSBoxType::StrokeBox:
89 case CSSBoxType::ViewBox:
90 case CSSBoxType::BoxMissing:
91 return style.getRoundedBorderFor(renderer.borderBoxRect());
92 }
93
94 ASSERT_NOT_REACHED();
95 return style.getRoundedBorderFor(renderer.borderBoxRect());
96}

98LayoutRect BoxShape::shapeMarginLogicalBoundingBox() const
99{
100 FloatRect marginBounds(m_bounds.rect());
101 if (shapeMargin() > 0)
102 marginBounds.inflate(shapeMargin());
103 return static_cast<LayoutRect>(marginBounds);
104}
105
106FloatRoundedRect BoxShape::shapeMarginBounds() const
107{
108 FloatRoundedRect marginBounds(m_bounds);
109 if (shapeMargin() > 0) {
110 marginBounds.inflate(shapeMargin());
111 marginBounds.expandRadii(shapeMargin());
112 }
113 return marginBounds;
114}

116LineSegment BoxShape::getExcludedInterval(LayoutUnit logicalTop, LayoutUnit logicalHeight) const
117{
118 const FloatRoundedRect& marginBounds = shapeMarginBounds();
119 if (marginBounds.isEmpty() || !lineOverlapsShapeMarginBounds(logicalTop, logicalHeight))
120 return LineSegment();
121
122 float y1 = logicalTop;
123 float y2 = logicalTop + logicalHeight;
124 const FloatRect& rect = marginBounds.rect();
125
126 if (!marginBounds.isRounded())
127 return LineSegment(rect.x(), rect.maxX());
128
129 float topCornerMaxY = std::max<float>(marginBounds.topLeftCorner().maxY(), marginBounds.topRightCorner().maxY());
130 float bottomCornerMinY = std::min<float>(marginBounds.bottomLeftCorner().y(), marginBounds.bottomRightCorner().y());
131
132 if (topCornerMaxY <= bottomCornerMinY && y1 <= topCornerMaxY && y2 >= bottomCornerMinY)
133 return LineSegment(rect.x(), rect.maxX());
134
135 float x1 = rect.maxX();
136 float x2 = rect.x();
137 float minXIntercept;
138 float maxXIntercept;
139
140 if (y1 <= marginBounds.topLeftCorner().maxY() && y2 >= marginBounds.bottomLeftCorner().y())
141 x1 = rect.x();
142
143 if (y1 <= marginBounds.topRightCorner().maxY() && y2 >= marginBounds.bottomRightCorner().y())
144 x2 = rect.maxX();
145
146 if (marginBounds.xInterceptsAtY(y1, minXIntercept, maxXIntercept)) {
147 x1 = std::min<float>(x1, minXIntercept);
148 x2 = std::max<float>(x2, maxXIntercept);
149 }
150
151 if (marginBounds.xInterceptsAtY(y2, minXIntercept, maxXIntercept)) {
152 x1 = std::min<float>(x1, minXIntercept);
153 x2 = std::max<float>(x2, maxXIntercept);
154 }
155
156 ASSERT(x2 >= x1);
157 return LineSegment(x1, x2);
158}

160void BoxShape::buildDisplayPaths(DisplayPaths& paths) const
161{
162 paths.shape.addRoundedRect(m_bounds, Path::RoundedRectStrategy::PreferBezier);
163 if (shapeMargin())
164 paths.marginShape.addRoundedRect(shapeMarginBounds(), Path::RoundedRectStrategy::PreferBezier);
165}
