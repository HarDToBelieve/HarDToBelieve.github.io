30#pragma once
31
32#include "FloatRect.h"
33#include "FloatSize.h"
34#include "Shape.h"
35#include <wtf/Assertions.h>
36
37namespace WebCore {
38
39class RectangleShape final : public Shape {
40public:
41 RectangleShape(const FloatRect& bounds, const FloatSize& radii)
42 : m_bounds(bounds)
43 , m_radii(radii)
44 {
45 }
46
47 LayoutRect shapeMarginLogicalBoundingBox() const override { return static_cast<LayoutRect>(shapeMarginBounds()); }
48 bool isEmpty() const override { return m_bounds.isEmpty(); }
49 LineSegment getExcludedInterval(LayoutUnit logicalTop, LayoutUnit logicalHeight) const override;
50
51 void buildDisplayPaths(DisplayPaths&) const override;
52
53private:
54 FloatRect shapeMarginBounds() const;
55
56 float rx() const { return m_radii.width(); }
57 float ry() const { return m_radii.height(); }
58 float x() const { return m_bounds.x(); }
59 float y() const { return m_bounds.y(); }
60 float width() const { return m_bounds.width(); }
61 float height() const { return m_bounds.height(); }
62
63 FloatRect m_bounds;
64 FloatSize m_radii;
65};
66
67} // namespace WebCore
