32namespace WebCore {
33
34TextStream& operator<<(TextStream& ts, const BorderValue& borderValue)
35{
36 ts << borderValue.width() << " " << borderValue.style() << " " << borderValue.color();
37 return ts;
38}
39
40TextStream& operator<<(TextStream& ts, const OutlineValue& outlineValue)
41{
42 ts << static_cast<const BorderValue&>(outlineValue);
43 ts.dumpProperty("outline-offset", outlineValue.offset());
44 return ts;
45}
46
47void BorderData::dump(TextStream& ts, DumpStyleValues behavior) const
48{
49 if (behavior == DumpStyleValues::All || left() != BorderValue())
50 ts.dumpProperty("left", left());
51 if (behavior == DumpStyleValues::All || right() != BorderValue())
52 ts.dumpProperty("right", right());
53 if (behavior == DumpStyleValues::All || top() != BorderValue())
54 ts.dumpProperty("top", top());
55 if (behavior == DumpStyleValues::All || bottom() != BorderValue())
56 ts.dumpProperty("bottom", bottom());
57
58 ts.dumpProperty("image", image());
59
60 if (behavior == DumpStyleValues::All || !topLeftRadius().isZero())
61 ts.dumpProperty("top-left", topLeftRadius());
62 if (behavior == DumpStyleValues::All || !topRightRadius().isZero())
63 ts.dumpProperty("top-right", topRightRadius());
64 if (behavior == DumpStyleValues::All || !bottomLeftRadius().isZero())
65 ts.dumpProperty("bottom-left", bottomLeftRadius());
66 if (behavior == DumpStyleValues::All || !bottomRightRadius().isZero())
67 ts.dumpProperty("bottom-right", bottomRightRadius());
68}
69
70TextStream& operator<<(TextStream& ts, const BorderData& borderData)
71{
72 borderData.dump(ts);
73 return ts;
74}
