25#pragma once
26
27#include "IntPoint.h"
28#include "StyleImage.h"
29
30namespace WebCore {
31
32class CursorData {
33public:
34 CursorData(RefPtr<StyleImage>&& image, const IntPoint& hotSpot)
35 : m_image(WTFMove(image))
36 , m_hotSpot(hotSpot)
37 {
38 }
39
40 bool operator==(const CursorData& o) const
41 {
42 return m_hotSpot == o.m_hotSpot && m_image == o.m_image;
43 }
44
45 bool operator!=(const CursorData& o) const
46 {
47 return !(*this == o);
48 }
49
50 StyleImage* image() const { return m_image.get(); }
51 void setImage(RefPtr<StyleImage>&& image) { m_image = WTFMove(image); }
52
53 // Hot spot in the image in logical pixels.
54 const IntPoint& hotSpot() const { return m_hotSpot; }
55
56private:
57 RefPtr<StyleImage> m_image;
58 IntPoint m_hotSpot; // for CSS3 support
59};
60
61} // namespace WebCore
