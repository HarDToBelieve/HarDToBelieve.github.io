25#pragma once
26
27#include "CursorData.h"
28#include <wtf/RefCounted.h>
29#include <wtf/Vector.h>
30
31namespace WebCore {
32
33class CursorList : public RefCounted<CursorList> {
34public:
35 static Ref<CursorList> create()
36 {
37 return adoptRef(*new CursorList);
38 }
39
40 const CursorData& operator[](int i) const { return m_vector[i]; }
41 CursorData& operator[](int i) { return m_vector[i]; }
42 const CursorData& at(size_t i) const { return m_vector.at(i); }
43 CursorData& at(size_t i) { return m_vector.at(i); }
44
45 bool operator==(const CursorList& o) const { return m_vector == o.m_vector; }
46 bool operator!=(const CursorList& o) const { return m_vector != o.m_vector; }
47
48 size_t size() const { return m_vector.size(); }
49 void append(const CursorData& cursorData) { m_vector.append(cursorData); }
50
51private:
52 CursorList()
53 {
54 }
55
56 Vector<CursorData> m_vector;
57};
58
59} // namespace WebCore
