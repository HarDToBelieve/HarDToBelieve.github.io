170bool FillLayer::operator==(const FillLayer& o) const
171{
172 // We do not check the "isSet" booleans for each property, since those are only used during initial construction
173 // to propagate patterns into layers. All layer comparisons happen after values have all been filled in anyway.
174 return arePointingToEqualData(m_image.get(), o.m_image.get()) && m_xPosition == o.m_xPosition && m_yPosition == o.m_yPosition
175 && m_backgroundXOrigin == o.m_backgroundXOrigin && m_backgroundYOrigin == o.m_backgroundYOrigin
176 && m_attachment == o.m_attachment && m_clip == o.m_clip && m_composite == o.m_composite
177 && m_blendMode == o.m_blendMode && m_origin == o.m_origin && m_repeatX == o.m_repeatX
178 && m_repeatY == o.m_repeatY && m_sizeType == o.m_sizeType && m_maskSourceType == o.m_maskSourceType
179 && m_sizeLength == o.m_sizeLength && m_type == o.m_type
180 && ((m_next && o.m_next) ? *m_next == *o.m_next : m_next == o.m_next);
181}

183void FillLayer::fillUnsetProperties()
184{
185 FillLayer* curr;
186 for (curr = this; curr && curr->isXPositionSet(); curr = curr->next()) { }
187 if (curr && curr != this) {
188 // We need to fill in the remaining values with the pattern specified.
189 for (FillLayer* pattern = this; curr; curr = curr->next()) {
190 curr->m_xPosition = pattern->m_xPosition;
191 if (pattern->isBackgroundXOriginSet())
192 curr->m_backgroundXOrigin = pattern->m_backgroundXOrigin;
193 if (pattern->isBackgroundYOriginSet())
194 curr->m_backgroundYOrigin = pattern->m_backgroundYOrigin;
195 pattern = pattern->next();
196 if (pattern == curr || !pattern)
197 pattern = this;
198 }
199 }
200
201 for (curr = this; curr && curr->isYPositionSet(); curr = curr->next()) { }
202 if (curr && curr != this) {
203 // We need to fill in the remaining values with the pattern specified.
204 for (FillLayer* pattern = this; curr; curr = curr->next()) {
205 curr->m_yPosition = pattern->m_yPosition;
206 if (pattern->isBackgroundXOriginSet())
207 curr->m_backgroundXOrigin = pattern->m_backgroundXOrigin;
208 if (pattern->isBackgroundYOriginSet())
209 curr->m_backgroundYOrigin = pattern->m_backgroundYOrigin;
210 pattern = pattern->next();
211 if (pattern == curr || !pattern)

217 if (curr && curr != this) {
218 // We need to fill in the remaining values with the pattern specified.

306void FillLayer::cullEmptyLayers()
307{
308 for (FillLayer* layer = this; layer; layer = layer->m_next.get()) {
309 if (layer->m_next && !layer->m_next->isImageSet()) {
310 layer->m_next = nullptr;
311 break;
312 }
313 }

316static inline FillBox clipMax(FillBox clipA, FillBox clipB)
317{
318 if (clipA == FillBox::Border || clipB == FillBox::Border)
319 return FillBox::Border;
320 if (clipA == FillBox::Padding || clipB == FillBox::Padding)
321 return FillBox::Padding;
322 if (clipA == FillBox::Content || clipB == FillBox::Content)
323 return FillBox::Content;
324 return FillBox::Text;
325}
326
327void FillLayer::computeClipMax() const
328{
329 Vector<const FillLayer*, 4> layers;
330 for (auto* layer = this; layer; layer = layer->m_next.get())
331 layers.append(layer);
332 FillBox computedClipMax = FillBox::Text;
333 for (unsigned i = layers.size(); i; --i) {
334 auto& layer = *layers[i - 1];
335 computedClipMax = clipMax(computedClipMax, layer.clip());
336 layer.m_clipMax = static_cast<unsigned>(computedClipMax);
337 }
338}

340bool FillLayer::clipOccludesNextLayers(bool firstLayer) const
341{
342 if (firstLayer)
343 computeClipMax();
344 return m_clip == m_clipMax;

365bool FillLayer::hasOpaqueImage(const RenderElement& renderer) const
366{
367 if (!m_image)
368 return false;
369
370 if (static_cast<CompositeOperator>(m_composite) == CompositeOperator::Clear || static_cast<CompositeOperator>(m_composite) == CompositeOperator::Copy)
371 return true;
372
373 return static_cast<BlendMode>(m_blendMode) == BlendMode::Normal && static_cast<CompositeOperator>(m_composite) == CompositeOperator::SourceOver && m_image->knownToBeOpaque(renderer);
374}

403
404TextStream& operator<<(TextStream& ts, const FillLayer& layer)
405{
406 TextStream::GroupScope scope(ts);
407 ts << "fill-layer";
408
409 ts.startGroup();
410 ts << "position " << layer.xPosition() << " " << layer.yPosition();
411 ts.endGroup();
412
413 ts.dumpProperty("size", layer.size());
414
415 ts.startGroup();
416 ts << "background-origin " << layer.backgroundXOrigin() << " " << layer.backgroundYOrigin();
417 ts.endGroup();
418
419 ts.startGroup();
420 ts << "repeat " << layer.repeatX() << " " << layer.repeatY();
421 ts.endGroup();
422
423 ts.dumpProperty("clip", layer.clip());
424 ts.dumpProperty("origin", layer.origin());
425
426 ts.dumpProperty("composite", layer.composite());
427 ts.dumpProperty("blend-mode", layer.blendMode());
428 ts.dumpProperty("mask-type", layer.maskSourceType());
429
430 if (layer.next())
431 ts << *layer.next();
432
433 return ts;
434}
