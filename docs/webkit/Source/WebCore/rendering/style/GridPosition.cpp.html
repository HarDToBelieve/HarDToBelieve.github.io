31#include "config.h"
32#include "GridPosition.h"
33
34#include <wtf/text/TextStream.h>
35
36namespace WebCore {
37
38std::optional<int> GridPosition::gMaxPositionForTesting;
39static const int kGridMaxPosition = 1000000;
40
41void GridPosition::setExplicitPosition(int position, const String& namedGridLine)
42{
43 m_type = ExplicitPosition;
44 setIntegerPosition(position);
45 m_namedGridLine = namedGridLine;
46}
47
48void GridPosition::setAutoPosition()
49{
50 m_type = AutoPosition;
51 m_integerPosition = 0;
52}
53
54// 'span' values cannot be negative, yet we reuse the <integer> position which can
55// be. This means that we have to convert the span position to an integer, losing
56// some precision here. It shouldn't be an issue in practice though.
57void GridPosition::setSpanPosition(int position, const String& namedGridLine)
58{
59 m_type = SpanPosition;
60 setIntegerPosition(position);
61 m_namedGridLine = namedGridLine;
62}
63
64void GridPosition::setNamedGridArea(const String& namedGridArea)
65{
66 m_type = NamedGridAreaPosition;
67 m_namedGridLine = namedGridArea;
68}
69
70int GridPosition::integerPosition() const
71{
72 ASSERT(type() == ExplicitPosition);
73 return m_integerPosition;
74}
75
76String GridPosition::namedGridLine() const
77{
78 ASSERT(type() == ExplicitPosition || type() == SpanPosition || type() == NamedGridAreaPosition);
79 return m_namedGridLine;
80}
81
82int GridPosition::spanPosition() const
83{
84 ASSERT(type() == SpanPosition);
85 return m_integerPosition;
86}
87
88int GridPosition::max()
89{
90 return gMaxPositionForTesting.value_or(kGridMaxPosition);
91}
92
93int GridPosition::min()
94{
95 return -max();
96}
97
98bool GridPosition::operator==(const GridPosition& other) const
99{
100 return m_type == other.m_type && m_integerPosition == other.m_integerPosition && m_namedGridLine == other.m_namedGridLine;
101}
102
103void GridPosition::setMaxPositionForTesting(unsigned maxPosition)
104{
105 gMaxPositionForTesting = static_cast<int>(maxPosition);
106}
107
108TextStream& operator<<(TextStream& ts, const GridPosition& o)
109{
110 switch (o.type()) {
111 case AutoPosition:
112 return ts << "auto";
113 case ExplicitPosition:
114 return ts << o.namedGridLine() << " " << o.integerPosition();
115 case SpanPosition:
116 return ts << "span" << " " << o.namedGridLine() << " " << o.integerPosition();
117 case NamedGridAreaPosition:
118 return ts << o.namedGridLine();
119 }
120 return ts;
121}
