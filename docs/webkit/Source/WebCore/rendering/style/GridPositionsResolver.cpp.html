41{
42 return side == ColumnStartSide || side == ColumnEndSide;

46{
47 return side == ColumnStartSide || side == RowStartSide;

60NamedLineCollection::NamedLineCollection(const RenderStyle& gridContainerStyle, const String& namedLine, GridTrackSizingDirection direction, unsigned lastLine, unsigned autoRepeatTracksCount)
61 : m_lastLine(lastLine)
62 , m_autoRepeatTotalTracks(autoRepeatTracksCount)
63{
64 bool isRowAxis = direction == ForColumns;
65 const NamedGridLinesMap& gridLineNames = isRowAxis ? gridContainerStyle.namedGridColumnLines() : gridContainerStyle.namedGridRowLines();
66 const NamedGridLinesMap& autoRepeatGridLineNames = isRowAxis ? gridContainerStyle.autoRepeatNamedGridColumnLines() : gridContainerStyle.autoRepeatNamedGridRowLines();
67 const NamedGridLinesMap& implicitGridLineNames = isRowAxis ? gridContainerStyle.implicitNamedGridColumnLines() : gridContainerStyle.implicitNamedGridRowLines();
68
69 auto linesIterator = gridLineNames.find(namedLine);
70 m_namedLinesIndexes = linesIterator == gridLineNames.end() ? nullptr : &linesIterator->value;
71
72 auto autoRepeatLinesIterator = autoRepeatGridLineNames.find(namedLine);
73 m_autoRepeatNamedLinesIndexes = autoRepeatLinesIterator == autoRepeatGridLineNames.end() ? nullptr : &autoRepeatLinesIterator->value;
74
75 auto implicitGridLinesIterator = implicitGridLineNames.find(namedLine);
76 m_implicitNamedLinesIndexes = implicitGridLinesIterator == implicitGridLineNames.end() ? nullptr : &implicitGridLinesIterator->value;
77
78 m_insertionPoint = isRowAxis ? gridContainerStyle.gridAutoRepeatColumnsInsertionPoint() : gridContainerStyle.gridAutoRepeatRowsInsertionPoint();
79
80 m_autoRepeatTrackListLength = isRowAxis ? gridContainerStyle.gridAutoRepeatColumns().size() : gridContainerStyle.gridAutoRepeatRows().size();
81}

93bool NamedLineCollection::contains(unsigned line) const
94{
95 ASSERT(hasNamedLines());
96
97 if (line > m_lastLine)
98 return false;
99
100 auto contains = [](const Vector<unsigned>* indexes, unsigned line) {
101 return indexes && indexes->find(line) != notFound;
102 };
103
104 if (contains(m_implicitNamedLinesIndexes, line))
105 return true;
106
107 if (!m_autoRepeatTrackListLength || line < m_insertionPoint)
108 return contains(m_namedLinesIndexes, line);
109
110 ASSERT(m_autoRepeatTotalTracks);
111
112 if (line > m_insertionPoint + m_autoRepeatTotalTracks)
113 return contains(m_namedLinesIndexes, line - (m_autoRepeatTotalTracks - 1));
114
115 if (line == m_insertionPoint)
116 return contains(m_namedLinesIndexes, line) || contains(m_autoRepeatNamedLinesIndexes, 0);
117
118 if (line == m_insertionPoint + m_autoRepeatTotalTracks)
119 return contains(m_autoRepeatNamedLinesIndexes, m_autoRepeatTrackListLength) || contains(m_namedLinesIndexes, m_insertionPoint + 1);
120
121 size_t autoRepeatIndexInFirstRepetition = (line - m_insertionPoint) % m_autoRepeatTrackListLength;
122 if (!autoRepeatIndexInFirstRepetition && contains(m_autoRepeatNamedLinesIndexes, m_autoRepeatTrackListLength))
123 return true;
124 return contains(m_autoRepeatNamedLinesIndexes, autoRepeatIndexInFirstRepetition);
125}

127unsigned NamedLineCollection::firstExplicitPosition() const
128{
129 ASSERT(hasExplicitNamedLines());
130 unsigned firstLine = 0;
131
132 // If there is no auto repeat(), there must be some named line outside, return the 1st one. Also return it if it precedes the auto-repeat().
133 if (!m_autoRepeatTrackListLength || (m_namedLinesIndexes && m_namedLinesIndexes->at(firstLine) <= m_insertionPoint))
134 return m_namedLinesIndexes->at(firstLine);
135
136 // Return the 1st named line inside the auto repeat(), if any.
137 if (m_autoRepeatNamedLinesIndexes)
138 return m_autoRepeatNamedLinesIndexes->at(firstLine) + m_insertionPoint;
139
140 // The 1st named line must be after the auto repeat().
141 return m_namedLinesIndexes->at(firstLine) + m_autoRepeatTotalTracks - 1;
142}

144unsigned NamedLineCollection::firstPosition() const
145{
146 ASSERT(hasNamedLines());
147 unsigned firstLine = 0;
148 if (!m_implicitNamedLinesIndexes)
149 return firstExplicitPosition();
150 if (!hasExplicitNamedLines())
151 return m_implicitNamedLinesIndexes->at(firstLine);
152 return std::min(firstExplicitPosition(), m_implicitNamedLinesIndexes->at(firstLine));
153}

155static void adjustGridPositionsFromStyle(const RenderBox& gridItem, GridTrackSizingDirection direction, GridPosition& initialPosition, GridPosition& finalPosition)
156{
157 bool isForColumns = direction == ForColumns;
158 initialPosition = isForColumns ? gridItem.style().gridItemColumnStart() : gridItem.style().gridItemRowStart();
159 finalPosition = isForColumns ? gridItem.style().gridItemColumnEnd() : gridItem.style().gridItemRowEnd();
160
161 // We must handle the placement error handling code here instead of in the StyleAdjuster because we don't want to
162 // overwrite the specified values.
163 if (initialPosition.isSpan() && finalPosition.isSpan())
164 finalPosition.setAutoPosition();
165
166 // If the grid item has an automatic position and a grid span for a named line in a given dimension, instead treat the grid span as one.
167 if (initialPosition.isAuto() && finalPosition.isSpan() && !finalPosition.namedGridLine().isNull())
168 finalPosition.setSpanPosition(1, String());
169 if (finalPosition.isAuto() && initialPosition.isSpan() && !initialPosition.namedGridLine().isNull())
170 initialPosition.setSpanPosition(1, String());
171}

188static unsigned lookAheadForNamedGridLine(int start, unsigned numberOfLines, unsigned gridLastLine, NamedLineCollection& linesCollection)
189{
190 ASSERT(numberOfLines);
191
192 // Only implicit lines on the search direction are assumed to have the given name, so we can start to look from first line.
193 // See: https://drafts.csswg.org/css-grid/#grid-placement-span-int
194 unsigned end = std::max(start, 0);
195
196 if (!linesCollection.hasNamedLines())
197 return std::max(end, gridLastLine + 1) + numberOfLines - 1;
198
199 for (; numberOfLines; ++end) {
200 if (end > gridLastLine || linesCollection.contains(end))
201 numberOfLines--;
202 }
203
204 ASSERT(end);
205 return end - 1;
206}
207
208static int lookBackForNamedGridLine(int end, unsigned numberOfLines, int gridLastLine, NamedLineCollection& linesCollection)
209{
210 ASSERT(numberOfLines);
211
212 // Only implicit lines on the search direction are assumed to have the given name, so we can start to look from last line.
213 // See: https://drafts.csswg.org/css-grid/#grid-placement-span-int
214 int start = std::min(end, gridLastLine);
215
216 if (!linesCollection.hasNamedLines())
217 return std::min(start, -1) - numberOfLines + 1;
218
219 for (; numberOfLines; --start) {
220 if (start < 0 || linesCollection.contains(start))
221 numberOfLines--;
222 }
223
224 return start + 1;
225}

227static int resolveNamedGridLinePositionFromStyle(const RenderStyle& gridContainerStyle, const GridPosition& position, GridPositionSide side, unsigned autoRepeatTracksCount)
228{
229 ASSERT(!position.namedGridLine().isNull());
230
231 unsigned lastLine = explicitGridSizeForSide(gridContainerStyle, side, autoRepeatTracksCount);
232 NamedLineCollection linesCollection(gridContainerStyle, position.namedGridLine(), directionFromSide(side), lastLine, autoRepeatTracksCount);
233
234 if (position.isPositive())
235 return lookAheadForNamedGridLine(0, std::abs(position.integerPosition()), lastLine, linesCollection);
236 return lookBackForNamedGridLine(lastLine, std::abs(position.integerPosition()), lastLine, linesCollection);
237}

239static GridSpan definiteGridSpanWithNamedLineSpanAgainstOpposite(int oppositeLine, const GridPosition& position, GridPositionSide side, unsigned lastLine, NamedLineCollection& linesCollection)
240{
241 int start, end;
242 if (side == RowStartSide || side == ColumnStartSide) {
243 start = lookBackForNamedGridLine(oppositeLine - 1, position.spanPosition(), lastLine, linesCollection);
244 end = oppositeLine;
245 } else {
246 start = oppositeLine;
247 end = lookAheadForNamedGridLine(oppositeLine + 1, position.spanPosition(), lastLine, linesCollection);
248 }
249
250 return GridSpan::untranslatedDefiniteGridSpan(start, end);
251}

253static GridSpan resolveNamedGridLinePositionAgainstOppositePosition(const RenderStyle& gridContainerStyle, int oppositeLine, const GridPosition& position, GridPositionSide side, unsigned autoRepeatTracksCount)
254{
255 ASSERT(position.isSpan());
256 ASSERT(!position.namedGridLine().isNull());
257 // Negative positions are not allowed per the specification and should have been handled during parsing.
258 ASSERT(position.spanPosition() > 0);
259
260 unsigned lastLine = explicitGridSizeForSide(gridContainerStyle, side, autoRepeatTracksCount);
261 NamedLineCollection linesCollection(gridContainerStyle, position.namedGridLine(), directionFromSide(side), lastLine, autoRepeatTracksCount);
262 return definiteGridSpanWithNamedLineSpanAgainstOpposite(oppositeLine, position, side, lastLine, linesCollection);
263}

265static GridSpan resolveGridPositionAgainstOppositePosition(const RenderStyle& gridContainerStyle, int oppositeLine, const GridPosition& position, GridPositionSide side, unsigned autoRepeatTracksCount)
266{
267 if (position.isAuto()) {
268 if (isStartSide(side))
269 return GridSpan::untranslatedDefiniteGridSpan(oppositeLine - 1, oppositeLine);
270 return GridSpan::untranslatedDefiniteGridSpan(oppositeLine, oppositeLine + 1);
271 }
272
273 ASSERT(position.isSpan());
274 ASSERT(position.spanPosition() > 0);
275
276 if (!position.namedGridLine().isNull()) {
277 // span 2 'c' -> we need to find the appropriate grid line before / after our opposite position.
278 return resolveNamedGridLinePositionAgainstOppositePosition(gridContainerStyle, oppositeLine, position, side, autoRepeatTracksCount);
279 }
280
281 // 'span 1' is contained inside a single grid track regardless of the direction.
282 // That's why the CSS span value is one more than the offset we apply.
283 unsigned positionOffset = position.spanPosition();
284 if (isStartSide(side))
285 return GridSpan::untranslatedDefiniteGridSpan(oppositeLine - positionOffset, oppositeLine);
286
287 return GridSpan::untranslatedDefiniteGridSpan(oppositeLine, oppositeLine + positionOffset);
288}

300unsigned GridPositionsResolver::spanSizeForAutoPlacedItem(const RenderBox& gridItem, GridTrackSizingDirection direction)
301{
302 GridPosition initialPosition, finalPosition;
303 adjustGridPositionsFromStyle(gridItem, direction, initialPosition, finalPosition);
304
305 // This method will only be used when both positions need to be resolved against the opposite one.
306 ASSERT(initialPosition.shouldBeResolvedAgainstOppositePosition() && finalPosition.shouldBeResolvedAgainstOppositePosition());
307
308 if (initialPosition.isAuto() && finalPosition.isAuto())
309 return 1;
310
311 GridPosition position = initialPosition.isSpan() ? initialPosition : finalPosition;
312 ASSERT(position.isSpan());
313
314 ASSERT(position.spanPosition());
315 return position.spanPosition();
316}

318static int resolveGridPositionFromStyle(const RenderStyle& gridContainerStyle, const GridPosition& position, GridPositionSide side, unsigned autoRepeatTracksCount)
319{
320 switch (position.type()) {
321 case ExplicitPosition: {
322 ASSERT(position.integerPosition());
323
324 if (!position.namedGridLine().isNull())
325 return resolveNamedGridLinePositionFromStyle(gridContainerStyle, position, side, autoRepeatTracksCount);
326
327 // Handle <integer> explicit position.
328 if (position.isPositive())
329 return position.integerPosition() - 1;
330
331 unsigned resolvedPosition = std::abs(position.integerPosition()) - 1;
332 const unsigned endOfTrack = explicitGridSizeForSide(gridContainerStyle, side, autoRepeatTracksCount);
333
334 return endOfTrack - resolvedPosition;
335 }
336 case NamedGridAreaPosition:
337 {
338 // First attempt to match the grid area's edge to a named grid area: if there is a named line with the name
339 // ''<custom-ident>-start (for grid-*-start) / <custom-ident>-end'' (for grid-*-end), contributes the first such
340 // line to the grid item's placement.
341 String namedGridLine = position.namedGridLine();
342 ASSERT(!position.namedGridLine().isNull());
343
344 unsigned lastLine = explicitGridSizeForSide(gridContainerStyle, side, autoRepeatTracksCount);
345 NamedLineCollection implicitLines(gridContainerStyle, implicitNamedGridLineForSide(namedGridLine, side), directionFromSide(side), lastLine, autoRepeatTracksCount);
346 if (implicitLines.hasNamedLines())
347 return implicitLines.firstPosition();
348
349 // Otherwise, if there is a named line with the specified name, contributes the first such line to the grid
350 // item's placement.
351 NamedLineCollection explicitLines(gridContainerStyle, namedGridLine, directionFromSide(side), lastLine, autoRepeatTracksCount);
352 if (explicitLines.hasNamedLines())
353 return explicitLines.firstPosition();
354
355 // If none of the above works specs mandate to assume that all the lines in the implicit grid have this name.
356 return lastLine + 1;
357 }
358 case AutoPosition:
359 case SpanPosition:
360 // 'auto' and span depend on the opposite position for resolution (e.g. grid-row: auto / 1 or grid-column: span 3 / "myHeader").
361 ASSERT_NOT_REACHED();
362 return 0;
363 }
364 ASSERT_NOT_REACHED();
365 return 0;
366}

368GridSpan GridPositionsResolver::resolveGridPositionsFromStyle(const RenderStyle& gridContainerStyle, const RenderBox& gridItem, GridTrackSizingDirection direction, unsigned autoRepeatTracksCount)
369{
370 GridPosition initialPosition, finalPosition;
371 adjustGridPositionsFromStyle(gridItem, direction, initialPosition, finalPosition);
372
373 GridPositionSide initialSide = initialPositionSide(direction);
374 GridPositionSide finalSide = finalPositionSide(direction);
375
376 // We can't get our grid positions without running the auto placement algorithm.
377 if (initialPosition.shouldBeResolvedAgainstOppositePosition() && finalPosition.shouldBeResolvedAgainstOppositePosition())
378 return GridSpan::indefiniteGridSpan();
379
380 if (initialPosition.shouldBeResolvedAgainstOppositePosition()) {
381 // Infer the position from the final position ('auto / 1' or 'span 2 / 3' case).
382 auto endLine = resolveGridPositionFromStyle(gridContainerStyle, finalPosition, finalSide, autoRepeatTracksCount);
383 return resolveGridPositionAgainstOppositePosition(gridContainerStyle, endLine, initialPosition, initialSide, autoRepeatTracksCount);
384 }
385
386 if (finalPosition.shouldBeResolvedAgainstOppositePosition()) {
387 // Infer our position from the initial position ('1 / auto' or '3 / span 2' case).
388 auto startLine = resolveGridPositionFromStyle(gridContainerStyle, initialPosition, initialSide, autoRepeatTracksCount);
389 return resolveGridPositionAgainstOppositePosition(gridContainerStyle, startLine, finalPosition, finalSide, autoRepeatTracksCount);
390 }
391
392 int startLine = resolveGridPositionFromStyle(gridContainerStyle, initialPosition, initialSide, autoRepeatTracksCount);
393 int endLine = resolveGridPositionFromStyle(gridContainerStyle, finalPosition, finalSide, autoRepeatTracksCount);
394
395 if (startLine > endLine)
396 std::swap(startLine, endLine);
397 else if (startLine == endLine)
398 endLine = startLine + 1;
399
400 return GridSpan::untranslatedDefiniteGridSpan(startLine, std::max(startLine, endLine));
401}
