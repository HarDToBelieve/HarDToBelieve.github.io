25#pragma once
26
27#include "RenderStyleConstants.h"
28
29namespace WebCore {
30
31class LineClampValue {
32public:
33 LineClampValue()
34 : m_type(LineClamp::LineCount)
35 , m_value(-1)
36 {
37 }
38
39 LineClampValue(int value, LineClamp type)
40 : m_type(type)
41 , m_value(value)
42 {
43 }
44
45 int value() const { return m_value; }
46
47 bool isPercentage() const { return m_type == LineClamp::Percentage; }
48
49 bool isNone() const { return m_value == -1; }
50
51 bool operator==(const LineClampValue& o) const
52 {
53 return value() == o.value() && isPercentage() == o.isPercentage();
54 }
55
56 bool operator!=(const LineClampValue& o) const
57 {
58 return !(*this == o);
59 }
60
61private:
62 LineClamp m_type;
63 int m_value;
64};
65
66} // namespace WebCore
