34Ref<StyleCachedImage> StyleCachedImage::create(CSSImageValue& cssValue, float scaleFactor)
35{
36 return adoptRef(*new StyleCachedImage(cssValue, scaleFactor));
37}
38
39StyleCachedImage::StyleCachedImage(CSSImageValue& cssValue, float scaleFactor)
40 : m_cssValue(cssValue)
41 , m_scaleFactor(scaleFactor)
42{
43 m_isCachedImage = true;
44 m_cachedImage = m_cssValue->cachedImage();
45 if (m_cachedImage)
46 m_isPending = false;
47}
48
49StyleCachedImage::~StyleCachedImage() = default;
50
51bool StyleCachedImage::operator==(const StyleImage& other) const
52{
53 if (!is<StyleCachedImage>(other))
54 return false;
55 auto& otherCached = downcast<StyleCachedImage>(other);
56 if (&otherCached == this)
57 return true;
58 if (m_scaleFactor != otherCached.m_scaleFactor)
59 return false;
60 if (m_cssValue.ptr() == otherCached.m_cssValue.ptr() || m_cssValue->equals(otherCached.m_cssValue.get()))
61 return true;
62 if (m_cachedImage && m_cachedImage == otherCached.m_cachedImage)
63 return true;
64 return false;
65}

72void StyleCachedImage::load(CachedResourceLoader& loader, const ResourceLoaderOptions& options)
73{
74 ASSERT(m_isPending);

76 m_cachedImage = m_cssValue->loadImage(loader, options);
77}

115FloatSize StyleCachedImage::imageSize(const RenderElement* renderer, float multiplier) const
116{
117 if (!m_cachedImage)
118 return { };
119 FloatSize size = m_cachedImage->imageSizeForRenderer(renderer, multiplier);
120 size.scale(1 / m_scaleFactor);
121 return size;
122}
