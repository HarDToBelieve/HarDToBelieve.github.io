30#include "RenderStyleConstants.h"
31#include <wtf/OptionSet.h>
32
33namespace WebCore {
34
35class StyleColorScheme {
36public:
37 StyleColorScheme() { }
38
39 StyleColorScheme(OptionSet<ColorScheme> colorScheme, bool allowsTransformations)
40 : m_colorScheme(colorScheme)
41 , m_allowsTransformations(allowsTransformations)
42 { }
43
44 bool operator==(const StyleColorScheme& other) const
45 {
46 return m_colorScheme == other.m_colorScheme && m_allowsTransformations == other.m_allowsTransformations;
47 }
48
49 bool operator!=(const StyleColorScheme& other) const { return !(*this == other); }
50
51 bool isAuto() const { return m_colorScheme.isEmpty() && m_allowsTransformations; }
52 bool isOnly() const { return m_colorScheme.isEmpty() && !m_allowsTransformations; }
53
54 OptionSet<ColorScheme> colorScheme() const { return m_colorScheme; }
55
56 void add(ColorScheme colorScheme) { m_colorScheme.add(colorScheme); }
57 bool contains(ColorScheme colorScheme) const { return m_colorScheme.contains(colorScheme); }
58
59 void setAllowsTransformations(bool allow) { m_allowsTransformations = allow; }
60 bool allowsTransformations() const { return m_allowsTransformations; }
61
62private:
63 OptionSet<ColorScheme> m_colorScheme;
64 bool m_allowsTransformations { true };
65};
66
67} // namespace WebCore
68
69#endif // ENABLE(DARK_MODE_CSS)
