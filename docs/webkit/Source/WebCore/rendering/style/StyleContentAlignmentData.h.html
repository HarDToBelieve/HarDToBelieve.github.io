34namespace WebCore {
35
36class StyleContentAlignmentData {
37public:
38 // Style data for Content-Distribution properties: align-content, justify-content.
39 // <content-distribution> || [ <overflow-position>? && <content-position> ]
40 StyleContentAlignmentData(ContentPosition position, ContentDistribution distribution, OverflowAlignment overflow = OverflowAlignment::Default)
41 : m_position(static_cast<unsigned>(position))
42 , m_distribution(static_cast<unsigned>(distribution))
43 , m_overflow(static_cast<unsigned>(overflow))
44 {
45 }
46
47 void setPosition(ContentPosition position) { m_position = static_cast<unsigned>(position); }
48 void setDistribution(ContentDistribution distribution) { m_distribution = static_cast<unsigned>(distribution); }
49 void setOverflow(OverflowAlignment overflow) { m_overflow = static_cast<unsigned>(overflow); }
50
51 ContentPosition position() const { return static_cast<ContentPosition>(m_position); }
52 ContentDistribution distribution() const { return static_cast<ContentDistribution>(m_distribution); }
53 OverflowAlignment overflow() const { return static_cast<OverflowAlignment>(m_overflow); }
54
55 bool operator==(const StyleContentAlignmentData& o) const
56 {
57 return m_position == o.m_position && m_distribution == o.m_distribution && m_overflow == o.m_overflow;
58 }
59
60 bool operator!=(const StyleContentAlignmentData& o) const
61 {
62 return !(*this == o);
63 }
64
65private:
66 unsigned m_position : 4; // ContentPosition
67 unsigned m_distribution : 3; // ContentDistribution
68 unsigned m_overflow : 2; // OverflowAlignment
69};
70
71WTF::TextStream& operator<<(WTF::TextStream&, const StyleContentAlignmentData&);
