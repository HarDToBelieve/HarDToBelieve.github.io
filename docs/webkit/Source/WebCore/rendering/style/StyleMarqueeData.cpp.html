27namespace WebCore {

29StyleMarqueeData::StyleMarqueeData()
30 : increment(RenderStyle::initialMarqueeIncrement())
31 , speed(RenderStyle::initialMarqueeSpeed())
32 , loops(RenderStyle::initialMarqueeLoopCount())
33 , behavior(static_cast<unsigned>(RenderStyle::initialMarqueeBehavior()))
34 , direction(static_cast<unsigned>(RenderStyle::initialMarqueeDirection()))
35{
36}
37
38inline StyleMarqueeData::StyleMarqueeData(const StyleMarqueeData& o)
39 : RefCounted<StyleMarqueeData>()
40 , increment(o.increment)
41 , speed(o.speed)
42 , loops(o.loops)
43 , behavior(o.behavior)
44 , direction(o.direction)
45{
46}
47
48Ref<StyleMarqueeData> StyleMarqueeData::copy() const
49{
50 return adoptRef(*new StyleMarqueeData(*this));
51}
52
53bool StyleMarqueeData::operator==(const StyleMarqueeData& o) const
54{
55 return increment == o.increment && speed == o.speed && direction == o.direction &&
56 behavior == o.behavior && loops == o.loops;
57}
