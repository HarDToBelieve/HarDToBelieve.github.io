34namespace WebCore {
35
36struct ScrollSnapType {
37 ScrollSnapAxis axis { ScrollSnapAxis::Both };
38 ScrollSnapStrictness strictness { ScrollSnapStrictness::None };
39};
40
41inline bool operator==(const ScrollSnapType& a, const ScrollSnapType& b)
42{
43 return a.axis == b.axis && a.strictness == b.strictness;
44}
45
46inline bool operator!=(const ScrollSnapType& a, const ScrollSnapType& b) { return !(a == b); }
47
48struct ScrollSnapAlign {
49 ScrollSnapAxisAlignType blockAlign { ScrollSnapAxisAlignType::None };
50 ScrollSnapAxisAlignType inlineAlign { ScrollSnapAxisAlignType::None };
51};
52
53inline bool operator==(const ScrollSnapAlign& a, const ScrollSnapAlign& b)
54{
55 return a.blockAlign == b.blockAlign && a.inlineAlign == b.inlineAlign;
56}
57
58inline bool operator!=(const ScrollSnapAlign& a, const ScrollSnapAlign& b) { return !(a == b); }
