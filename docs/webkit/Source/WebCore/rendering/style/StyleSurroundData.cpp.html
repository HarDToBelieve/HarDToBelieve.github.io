25namespace WebCore {
26
27DEFINE_ALLOCATOR_WITH_HEAP_IDENTIFIER(StyleSurroundData);
28
29StyleSurroundData::StyleSurroundData()
30 : margin(LengthType::Fixed)
31 , padding(LengthType::Fixed)
32{
33}
34
35inline StyleSurroundData::StyleSurroundData(const StyleSurroundData& o)
36 : RefCounted<StyleSurroundData>()
37 , offset(o.offset)
38 , margin(o.margin)
39 , padding(o.padding)
40 , border(o.border)
41{
42}
43
44Ref<StyleSurroundData> StyleSurroundData::copy() const
45{
46 return adoptRef(*new StyleSurroundData(*this));
47}
48
49bool StyleSurroundData::operator==(const StyleSurroundData& o) const
50{
51 return offset == o.offset && margin == o.margin && padding == o.padding && border == o.border;
52}
