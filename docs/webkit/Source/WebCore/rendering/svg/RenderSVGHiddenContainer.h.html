24namespace WebCore {
25
26class SVGElement;
27
28// This class is for containers which are never drawn, but do need to support style
29// <defs>, <linearGradient>, <radialGradient> are all good examples
30class RenderSVGHiddenContainer : public RenderSVGContainer {
31 WTF_MAKE_ISO_ALLOCATED(RenderSVGHiddenContainer);
32public:
33 RenderSVGHiddenContainer(SVGElement&, RenderStyle&&);
34
35protected:
36 void layout() override;
37
38private:
39 bool isSVGHiddenContainer() const final { return true; }
40 const char* renderName() const override { return "RenderSVGHiddenContainer"; }
41
42 void paint(PaintInfo&, const LayoutPoint&) final;
43
44 LayoutRect clippedOverflowRect(const RenderLayerModelObject*, VisibleRectContext) const final { return LayoutRect(); }
45 void absoluteQuads(Vector<FloatQuad>&, bool* wasFixed) const final;
46
47 bool nodeAtFloatPoint(const HitTestRequest&, HitTestResult&, const FloatPoint& pointInParent, HitTestAction) final;
48};
