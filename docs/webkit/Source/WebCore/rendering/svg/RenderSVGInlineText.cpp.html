46static String applySVGWhitespaceRules(const String& string, bool preserveWhiteSpace)
47{
48 String newString = string;
49 if (preserveWhiteSpace) {
50 // Spec: When xml:space="preserve", the SVG user agent will do the following using a
51 // copy of the original character data content. It will convert all newline and tab
52 // characters into space characters. Then, it will draw all space characters, including
53 // leading, trailing and multiple contiguous space characters.
54 newString.replace('\t', ' ');
55 newString.replace('\n', ' ');
56 newString.replace('\r', ' ');
57 return newString;
58 }
59
60 // Spec: When xml:space="default", the SVG user agent will do the following using a
61 // copy of the original character data content. First, it will remove all newline
62 // characters. Then it will convert all tab characters into space characters.
63 // Then, it will strip off all leading and trailing space characters.
64 // Then, all contiguous space characters will be consolidated.
65 newString.replace('\n', emptyString());
66 newString.replace('\r', emptyString());
67 newString.replace('\t', ' ');
68 return newString;
69}

90void RenderSVGInlineText::styleDidChange(StyleDifference diff, const RenderStyle* oldStyle)
91{
92 RenderText::styleDidChange(diff, oldStyle);
93 updateScaledFont();
94
95 bool newPreserves = style().whiteSpace() == WhiteSpace::Pre;
96 bool oldPreserves = oldStyle ? oldStyle->whiteSpace() == WhiteSpace::Pre : false;
97 if (oldPreserves && !newPreserves) {
98 setText(applySVGWhitespaceRules(originalText(), false), true);
99 return;
100 }
101
102 if (!oldPreserves && newPreserves) {
103 setText(applySVGWhitespaceRules(originalText(), true), true);
104 return;
105 }
106
107 if (diff != StyleDifference::Layout)
108 return;
109
110 // The text metrics may be influenced by style changes.
111 if (auto* textAncestor = RenderSVGText::locateRenderSVGTextAncestor(*this))
112 textAncestor->subtreeStyleDidChange(this);
113}

136bool RenderSVGInlineText::characterStartsNewTextChunk(int position) const
137{
138 ASSERT(position >= 0);
139 ASSERT(position < static_cast<int>(text().length()));
140
141 // Each <textPath> element starts a new text chunk, regardless of any x/y values.
142 if (!position && parent()->isSVGTextPath() && !previousSibling())
143 return true;
144
145 const SVGCharacterDataMap::const_iterator it = m_layoutAttributes.characterDataMap().find(static_cast<unsigned>(position + 1));
146 if (it == m_layoutAttributes.characterDataMap().end())
147 return false;
148
149 return it->value.x != SVGTextLayoutAttributes::emptyValue() || it->value.y != SVGTextLayoutAttributes::emptyValue();
150}

152VisiblePosition RenderSVGInlineText::positionForPoint(const LayoutPoint& point, const RenderFragmentContainer*)
153{
154 if (!firstTextBox() || text().isEmpty())
155 return createVisiblePosition(0, Affinity::Downstream);
156
157 float baseline = m_scaledFont.fontMetrics().floatAscent();
158
159 RenderBlock* containingBlock = this->containingBlock();
160 ASSERT(containingBlock);
161
162 // Map local point to absolute point, as the character origins stored in the text fragments use absolute coordinates.
163 FloatPoint absolutePoint(point);
164 absolutePoint.moveBy(containingBlock->location());
165
166 float closestDistance = std::numeric_limits<float>::max();
167 float closestDistancePosition = 0;
168 const SVGTextFragment* closestDistanceFragment = nullptr;
169 SVGInlineTextBox* closestDistanceBox = nullptr;
170
171 AffineTransform fragmentTransform;
172 for (auto* box = firstTextBox(); box; box = box->nextTextBox()) {
173 Vector<SVGTextFragment>& fragments = box->textFragments();
174
175 unsigned textFragmentsSize = fragments.size();
176 for (unsigned i = 0; i < textFragmentsSize; ++i) {
177 const SVGTextFragment& fragment = fragments.at(i);
178 FloatRect fragmentRect(fragment.x, fragment.y - baseline, fragment.width, fragment.height);
179 fragment.buildFragmentTransform(fragmentTransform);
180 if (!fragmentTransform.isIdentity())
181 fragmentRect = fragmentTransform.mapRect(fragmentRect);
182
183 float distance = powf(fragmentRect.x() - absolutePoint.x(), 2) +
184 powf(fragmentRect.y() + fragmentRect.height() / 2 - absolutePoint.y(), 2);
185
186 if (distance < closestDistance) {
187 closestDistance = distance;
188 closestDistanceBox = box;
189 closestDistanceFragment = &fragment;
190 closestDistancePosition = fragmentRect.x();
191 }
192 }
193 }
194
195 if (!closestDistanceFragment)
196 return createVisiblePosition(0, Affinity::Downstream);
197
198 int offset = closestDistanceBox->offsetForPositionInFragment(*closestDistanceFragment, absolutePoint.x() - closestDistancePosition, true);
199 return createVisiblePosition(offset + closestDistanceBox->start(), offset > 0 ? Affinity::Upstream : Affinity::Downstream);
200}

207void RenderSVGInlineText::computeNewScaledFontForStyle(const RenderObject& renderer, const RenderStyle& style, float& scalingFactor, FontCascade& scaledFont)
208{
209 // Alter font-size to the right on-screen value to avoid scaling the glyphs themselves, except when GeometricPrecision is specified
210 scalingFactor = SVGRenderingContext::calculateScreenFontSizeScalingFactor(renderer);
211 if (!scalingFactor || style.fontDescription().textRenderingMode() == TextRenderingMode::GeometricPrecision) {
212 scalingFactor = 1;
213 scaledFont = style.fontCascade();
214 return;
215 }
216
217 auto fontDescription = style.fontDescription();
218
219 // FIXME: We need to better handle the case when we compute very small fonts below (below 1pt).
220 fontDescription.setComputedSize(Style::computedFontSizeFromSpecifiedSizeForSVGInlineText(fontDescription.computedSize(), fontDescription.isAbsoluteSize(), scalingFactor, renderer.document()));
221
222 // SVG controls its own glyph orientation, so don't allow writing-mode
223 // to affect it.
224 if (fontDescription.orientation() != FontOrientation::Horizontal)
225 fontDescription.setOrientation(FontOrientation::Horizontal);
226
227 scaledFont = FontCascade(WTFMove(fontDescription), 0, 0);
228 scaledFont.update(&renderer.document().fontSelector());
229}
