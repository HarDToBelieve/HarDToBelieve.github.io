47void RenderSVGPath::updateShapeFromElement()
48{
49 RenderSVGShape::updateShapeFromElement();
50 updateZeroLengthSubpaths();
51
52 m_strokeBoundingBox = calculateUpdatedStrokeBoundingBox();
53}

55FloatRect RenderSVGPath::calculateUpdatedStrokeBoundingBox() const
56{
57 FloatRect strokeBoundingBox = m_strokeBoundingBox;
58
59 if (style().svgStyle().hasStroke()) {
60 // FIXME: zero-length subpaths do not respect vector-effect = non-scaling-stroke.
61 float strokeWidth = this->strokeWidth();
62 for (size_t i = 0; i < m_zeroLengthLinecapLocations.size(); ++i)
63 strokeBoundingBox.unite(zeroLengthSubpathRect(m_zeroLengthLinecapLocations[i], strokeWidth));
64 }
65
66 return strokeBoundingBox;
67}

69static void useStrokeStyleToFill(GraphicsContext& context)
70{
71 if (auto gradient = context.strokeGradient())
72 context.setFillGradient(*gradient, context.state().strokeGradientSpaceTransform);
73 else if (Pattern* pattern = context.strokePattern())
74 context.setFillPattern(*pattern);
75 else
76 context.setFillColor(context.strokeColor());
77}
78
79void RenderSVGPath::strokeShape(GraphicsContext& context) const
80{
81 if (!style().hasVisibleStroke())
82 return;
83
84 RenderSVGShape::strokeShape(context);
85
86 if (m_zeroLengthLinecapLocations.isEmpty())
87 return;
88
89 Path* usePath;
90 AffineTransform nonScalingTransform;
91
92 if (hasNonScalingStroke())
93 nonScalingTransform = nonScalingStrokeTransform();
94
95 GraphicsContextStateSaver stateSaver(context, true);
96 useStrokeStyleToFill(context);
97 for (size_t i = 0; i < m_zeroLengthLinecapLocations.size(); ++i) {
98 usePath = zeroLengthLinecapPath(m_zeroLengthLinecapLocations[i]);
99 if (hasNonScalingStroke())
100 usePath = nonScalingStrokePath(usePath, nonScalingTransform);
101 context.fillPath(*usePath);
102 }
103}

105bool RenderSVGPath::shapeDependentStrokeContains(const FloatPoint& point, PointCoordinateSpace pointCoordinateSpace)
106{
107 if (RenderSVGShape::shapeDependentStrokeContains(point, pointCoordinateSpace))
108 return true;
109
110 for (size_t i = 0; i < m_zeroLengthLinecapLocations.size(); ++i) {
111 ASSERT(style().svgStyle().hasStroke());
112 float strokeWidth = this->strokeWidth();
113 if (style().capStyle() == SquareCap) {
114 if (zeroLengthSubpathRect(m_zeroLengthLinecapLocations[i], strokeWidth).contains(point))
115 return true;
116 } else {
117 ASSERT(style().capStyle() == RoundCap);
118 FloatPoint radiusVector(point.x() - m_zeroLengthLinecapLocations[i].x(), point.y() - m_zeroLengthLinecapLocations[i].y());
119 if (radiusVector.lengthSquared() < strokeWidth * strokeWidth * .25f)
120 return true;
121 }
122 }
123 return false;
124}

126bool RenderSVGPath::shouldStrokeZeroLengthSubpath() const
127{
128 // Spec(11.4): Any zero length subpath shall not be stroked if the "stroke-linecap" property has a value of butt
129 // but shall be stroked if the "stroke-linecap" property has a value of round or square
130 return style().svgStyle().hasStroke() && style().capStyle() != ButtCap;
131}
132
133Path* RenderSVGPath::zeroLengthLinecapPath(const FloatPoint& linecapPosition) const
134{
135 static NeverDestroyed<Path> tempPath;
136
137 tempPath.get().clear();
138 if (style().capStyle() == SquareCap)
139 tempPath.get().addRect(zeroLengthSubpathRect(linecapPosition, this->strokeWidth()));
140 else
141 tempPath.get().addEllipse(zeroLengthSubpathRect(linecapPosition, this->strokeWidth()));
142
143 return &tempPath.get();
144}
145
146FloatRect RenderSVGPath::zeroLengthSubpathRect(const FloatPoint& linecapPosition, float strokeWidth) const
147{
148 return FloatRect(linecapPosition.x() - strokeWidth / 2, linecapPosition.y() - strokeWidth / 2, strokeWidth, strokeWidth);
149}
150
151void RenderSVGPath::updateZeroLengthSubpaths()
152{
153 m_zeroLengthLinecapLocations.clear();
154
155 if (!strokeWidth() || !shouldStrokeZeroLengthSubpath())
156 return;
157
158 SVGSubpathData subpathData(m_zeroLengthLinecapLocations);
159 path().apply([&subpathData](const PathElement& pathElement) {
160 SVGSubpathData::updateFromPathElement(subpathData, pathElement);
161 });
162 subpathData.pathIsDone();
163}
164
165bool RenderSVGPath::isRenderingDisabled() const
166{
167 // For a polygon, polyline or path, rendering is disabled if there is no path data.
168 // No path data is possible in the case of a missing or empty 'd' or 'points' attribute.
169 return path().isEmpty();
170}
