49void RenderSVGResourceContainer::layout()
50{
51 StackStats::LayoutCheckPoint layoutCheckPoint;
52 // Invalidate all resources if our layout changed.
53 if (selfNeedsClientInvalidation())
54 RenderSVGRoot::addResourceForClientInvalidation(this);
55
56 RenderSVGHiddenContainer::layout();
57}

59void RenderSVGResourceContainer::willBeDestroyed()
60{
61 SVGResourcesCache::resourceDestroyed(*this);
62
63 if (m_registered) {
64 svgExtensionsFromElement(element()).removeResource(m_id);
65 m_registered = false;
66 }
67
68 RenderSVGHiddenContainer::willBeDestroyed();
69}

71void RenderSVGResourceContainer::styleDidChange(StyleDifference diff, const RenderStyle* oldStyle)
72{
73 RenderSVGHiddenContainer::styleDidChange(diff, oldStyle);
74
75 if (!m_registered) {
76 m_registered = true;
77 registerResource();
78 }

81void RenderSVGResourceContainer::idChanged()
82{
83 // Invalidate all our current clients.
84 removeAllClientsFromCache();
85
86 // Remove old id, that is guaranteed to be present in cache.
87 svgExtensionsFromElement(element()).removeResource(m_id);
88 m_id = element().getIdAttribute();
89
90 registerResource();
91}

98void RenderSVGResourceContainer::markAllClientsForInvalidation(InvalidationMode mode)
99{
100 // FIXME: Style invalidation should either be a pre-layout task or this function
101 // should never get called while in layout. See webkit.org/b/208903.
102 if ((m_clients.isEmpty() && m_clientLayers.isEmpty()) || m_isInvalidating)
103 return;
104
105 SetForScope<bool> isInvalidating(m_isInvalidating, true);
106
107 bool needsLayout = mode == LayoutAndBoundariesInvalidation;
108 bool markForInvalidation = mode != ParentOnlyInvalidation;
109 auto* root = SVGRenderSupport::findTreeRootObject(*this);
110
111 for (auto* client : m_clients) {
112 // We should not mark any client outside the current root for invalidation
113 if (root != SVGRenderSupport::findTreeRootObject(*client))
114 continue;
115
116 if (is<RenderSVGResourceContainer>(*client)) {
117 downcast<RenderSVGResourceContainer>(*client).removeAllClientsFromCache(markForInvalidation);
118 continue;
119 }
120
121 if (markForInvalidation)
122 markClientForInvalidation(*client, mode);
123
124 RenderSVGResource::markForLayoutAndParentResourceInvalidation(*client, needsLayout);
125 }
126
127 markAllClientLayersForInvalidation();
128}
129
130void RenderSVGResourceContainer::markAllClientLayersForInvalidation()
131{
132 if (m_clientLayers.isEmpty())
133 return;
134
135 auto& document = (*m_clientLayers.begin())->renderer().document();
136 if (!document.view() || document.renderTreeBeingDestroyed())
137 return;
138
139 auto inLayout = document.view()->layoutContext().isInLayout();
140 for (auto* clientLayer : m_clientLayers) {
141 // FIXME: We should not get here while in layout. See webkit.org/b/208903.
142 // Repaint should also be triggered through some other means.
143 if (inLayout) {
144 clientLayer->renderer().repaint();
145 continue;
146 }
147 if (auto* enclosingElement = clientLayer->enclosingElement())
148 enclosingElement->invalidateStyleAndLayerComposition();
149 clientLayer->renderer().repaint();
150 }
151}

153void RenderSVGResourceContainer::markClientForInvalidation(RenderObject& client, InvalidationMode mode)
154{
155 ASSERT(!m_clients.isEmpty());
156
157 switch (mode) {
158 case LayoutAndBoundariesInvalidation:
159 case BoundariesInvalidation:
160 client.setNeedsBoundariesUpdate();
161 break;
162 case RepaintInvalidation:
163 if (!client.renderTreeBeingDestroyed())
164 client.repaint();
165 break;
166 case ParentOnlyInvalidation:
167 break;
168 }
169}

194void RenderSVGResourceContainer::registerResource()
195{
196 SVGDocumentExtensions& extensions = svgExtensionsFromElement(element());
197 if (!extensions.isIdOfPendingResource(m_id)) {
198 extensions.addResource(m_id, *this);
199 return;
200 }
201
202 auto elements = copyToVectorOf<Ref<Element>>(extensions.removePendingResource(m_id));
203
204 // Cache us with the new id.
205 extensions.addResource(m_id, *this);
206
207 // Update cached resources of pending clients.
208 for (auto& client : elements) {
209 ASSERT(client->hasPendingResources());
210 extensions.clearHasPendingResourcesIfPossible(client);
211 auto* renderer = client->renderer();
212 if (!renderer)
213 continue;
214 SVGResourcesCache::clientStyleChanged(*renderer, StyleDifference::Layout, renderer->style());
215 renderer->setNeedsLayout();
216 }
217}

219bool RenderSVGResourceContainer::shouldTransformOnTextPainting(const RenderElement& renderer, AffineTransform& resourceTransform)
220{
221#if USE(CG)
222 UNUSED_PARAM(renderer);
223 UNUSED_PARAM(resourceTransform);
224 return false;
225#else
226 // This method should only be called for RenderObjects that deal with text rendering. Cmp. RenderObject.h's is*() methods.
227 ASSERT(renderer.isSVGText() || renderer.isSVGTextPath() || renderer.isSVGInline());
228
229 // In text drawing, the scaling part of the graphics context CTM is removed, compare SVGInlineTextBox::paintTextWithShadows.
230 // So, we use that scaling factor here, too, and then push it down to pattern or gradient space
231 // in order to keep the pattern or gradient correctly scaled.
232 float scalingFactor = SVGRenderingContext::calculateScreenFontSizeScalingFactor(renderer);
233 if (scalingFactor == 1)
234 return false;
235 resourceTransform.scale(scalingFactor);
236 return true;
237#endif
238}

248 transform *= resourceTransform;
249 return transform;
