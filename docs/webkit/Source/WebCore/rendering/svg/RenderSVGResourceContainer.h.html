80
81inline RenderSVGResourceContainer* getRenderSVGResourceContainerById(Document& document, const AtomString& id)
82{
83 if (id.isEmpty())
84 return nullptr;
85
86 if (RenderSVGResourceContainer* renderResource = document.accessSVGExtensions().resourceById(id))
87 return renderResource;
88
89 return nullptr;
90}
91
92template<typename Renderer>
93Renderer* getRenderSVGResourceById(Document& document, const AtomString& id)
94{
95 // Using the RenderSVGResource type here avoids ambiguous casts for types that
96 // descend from both RenderObject and RenderSVGResourceContainer.
97 RenderSVGResource* container = getRenderSVGResourceContainerById(document, id);
98 if (is<Renderer>(container))
99 return downcast<Renderer>(container);
100
101 return nullptr;
102}
