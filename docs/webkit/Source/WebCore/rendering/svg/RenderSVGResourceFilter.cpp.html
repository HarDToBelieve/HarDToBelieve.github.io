67void RenderSVGResourceFilter::removeClientFromCache(RenderElement& client, bool markForInvalidation)
68{
69 LOG(Filters, "RenderSVGResourceFilter %p removing client %p", this, &client);
70
71 auto findResult = m_rendererFilterDataMap.find(&client);
72 if (findResult != m_rendererFilterDataMap.end()) {
73 FilterData& filterData = *findResult->value;
74 if (filterData.savedContext)
75 filterData.state = FilterData::MarkedForRemoval;
76 else
77 m_rendererFilterDataMap.remove(findResult);
78 }
79
80 markClientForInvalidation(client, markForInvalidation ? BoundariesInvalidation : ParentOnlyInvalidation);
81}

83std::unique_ptr<SVGFilterBuilder> RenderSVGResourceFilter::buildPrimitives(SVGFilter& filter) const
84{
85 static const unsigned maxCountChildNodes = 200;
86 if (filterElement().countChildNodes() > maxCountChildNodes)
87 return nullptr;
88
89 FloatRect targetBoundingBox = filter.targetBoundingBox();
90
91 // Add effects to the builder
92 auto builder = makeUnique<SVGFilterBuilder>(SourceGraphic::create(filter));
93 builder->setPrimitiveUnits(filterElement().primitiveUnits());
94 builder->setTargetBoundingBox(targetBoundingBox);
95
96 for (auto& element : childrenOfType<SVGFilterPrimitiveStandardAttributes>(filterElement())) {
97 auto effect = element.build(builder.get(), filter);
98 if (!effect) {
99 builder->clearEffects();
100 return nullptr;
101 }
102 builder->appendEffectToEffectReferences(effect.copyRef(), element.renderer());
103 element.setStandardAttributes(effect.get());
104 effect->setEffectBoundaries(SVGLengthContext::resolveRectangle<SVGFilterPrimitiveStandardAttributes>(&element, filterElement().primitiveUnits(), targetBoundingBox));
105 if (element.renderer()) {
106#if ENABLE(DESTINATION_COLOR_SPACE_LINEAR_SRGB)
107 effect->setOperatingColorSpace(element.renderer()->style().svgStyle().colorInterpolationFilters() == ColorInterpolation::LinearRGB ? DestinationColorSpace::LinearSRGB() : DestinationColorSpace::SRGB());
108#else
109 effect->setOperatingColorSpace(DestinationColorSpace::SRGB());
110#endif
111 }
112 builder->add(element.result(), WTFMove(effect));
113 }
114 return builder;
115}

117bool RenderSVGResourceFilter::applyResource(RenderElement& renderer, const RenderStyle&, GraphicsContext*& context, OptionSet<RenderSVGResourceMode> resourceMode)
118{
119 ASSERT(context);
120 ASSERT_UNUSED(resourceMode, !resourceMode);
121
122 LOG(Filters, "RenderSVGResourceFilter %p applyResource renderer %p", this, &renderer);
123
124 if (m_rendererFilterDataMap.contains(&renderer)) {
125 FilterData* filterData = m_rendererFilterDataMap.get(&renderer);
126 if (filterData->state == FilterData::PaintingSource || filterData->state == FilterData::Applying)
127 filterData->state = FilterData::CycleDetected;
128 return false; // Already built, or we're in a cycle, or we're marked for removal. Regardless, just do nothing more now.
129 }
130
131 auto filterData = makeUnique<FilterData>();
132 FloatRect targetBoundingBox = renderer.objectBoundingBox();
133
134 filterData->boundaries = SVGLengthContext::resolveRectangle<SVGFilterElement>(&filterElement(), filterElement().filterUnits(), targetBoundingBox);
135 if (filterData->boundaries.isEmpty())
136 return false;
137
138 // Determine absolute transformation matrix for filter.
139 AffineTransform absoluteTransform = SVGRenderingContext::calculateTransformationToOutermostCoordinateSystem(renderer);
140 if (!absoluteTransform.isInvertible())
141 return false;
142
143 // Eliminate shear of the absolute transformation matrix, to be able to produce unsheared tile images for feTile.
144 filterData->shearFreeAbsoluteTransform = AffineTransform(absoluteTransform.xScale(), 0, 0, absoluteTransform.yScale(), 0, 0);
145
146 // Determine absolute boundaries of the filter and the drawing region.
147 filterData->drawingRegion = renderer.strokeBoundingBox();
148 filterData->drawingRegion.intersect(filterData->boundaries);
149 FloatRect absoluteDrawingRegion = filterData->shearFreeAbsoluteTransform.mapRect(filterData->drawingRegion);
150
151 // Create the SVGFilter object.
152 bool primitiveBoundingBoxMode = filterElement().primitiveUnits() == SVGUnitTypes::SVG_UNIT_TYPE_OBJECTBOUNDINGBOX;
153 filterData->filter = SVGFilter::create(filterData->shearFreeAbsoluteTransform, absoluteDrawingRegion, targetBoundingBox, filterData->boundaries, primitiveBoundingBoxMode);
154
155 // Create all relevant filter primitives.
156 filterData->builder = buildPrimitives(*filterData->filter);
157 if (!filterData->builder)
158 return false;
159
160 // Determine scale factor for filter. The size of intermediate ImageBuffers shouldn't be bigger than kMaxFilterSize.
161 FloatRect tempSourceRect = absoluteDrawingRegion;
162 FloatSize scale(1, 1);
163 ImageBuffer::sizeNeedsClamping(tempSourceRect.size(), scale);
164 tempSourceRect.scale(scale.width(), scale.height());
165
166 // Set the scale level in SVGFilter.
167 filterData->filter->setFilterResolution(scale);
168
169 static const unsigned maxTotalOfEffectInputs = 100;
170 FilterEffect* lastEffect = filterData->builder->lastEffect();
171 if (!lastEffect || lastEffect->totalNumberOfEffectInputs() > maxTotalOfEffectInputs)
172 return false;
173
174 LOG_WITH_STREAM(Filters, stream << "RenderSVGResourceFilter::applyResource\n" << *filterData->builder->lastEffect());
175
176 lastEffect->determineFilterPrimitiveSubregion();
177 FloatRect subRegion = lastEffect->maxEffectRect();
178 // At least one FilterEffect has a too big image size,
179 // recalculate the effect sizes with new scale factors.
180 if (ImageBuffer::sizeNeedsClamping(subRegion.size(), scale)) {
181 filterData->filter->setFilterResolution(scale);
182 lastEffect->determineFilterPrimitiveSubregion();
183 }
184
185 // If the drawingRegion is empty, we have something like <g filter=".."/>.
186 // Even if the target objectBoundingBox() is empty, we still have to draw the last effect result image in postApplyResource.
187 if (filterData->drawingRegion.isEmpty()) {
188 ASSERT(!m_rendererFilterDataMap.contains(&renderer));
189 filterData->savedContext = context;
190 m_rendererFilterDataMap.set(&renderer, WTFMove(filterData));
191 return false;
192 }
193
194 // Change the coordinate transformation applied to the filtered element to reflect the resolution of the filter.
195 AffineTransform effectiveTransform;
196 effectiveTransform.scale(scale.width(), scale.height());
197 effectiveTransform.multiply(filterData->shearFreeAbsoluteTransform);
198
199 auto renderingMode = renderer.settings().acceleratedFiltersEnabled() ? RenderingMode::Accelerated : RenderingMode::Unaccelerated;
200#if ENABLE(DESTINATION_COLOR_SPACE_LINEAR_SRGB)
201 auto colorSpace = DestinationColorSpace::LinearSRGB();
202#else
203 auto colorSpace = DestinationColorSpace::SRGB();
204#endif
205 auto sourceGraphic = SVGRenderingContext::createImageBuffer(filterData->drawingRegion, effectiveTransform, colorSpace, renderingMode, context);
206 if (!sourceGraphic) {
207 ASSERT(!m_rendererFilterDataMap.contains(&renderer));
208 filterData->savedContext = context;
209 m_rendererFilterDataMap.set(&renderer, WTFMove(filterData));
210 return false;
211 }
212
213 // Set the rendering mode from the page's settings.
214 filterData->filter->setRenderingMode(renderingMode);
215
216 GraphicsContext& sourceGraphicContext = sourceGraphic->context();
217
218 filterData->sourceGraphicBuffer = WTFMove(sourceGraphic);
219 filterData->savedContext = context;
220
221 context = &sourceGraphicContext;
222
223 ASSERT(!m_rendererFilterDataMap.contains(&renderer));
224 m_rendererFilterDataMap.set(&renderer, WTFMove(filterData));
225
226 return true;
227}

229void RenderSVGResourceFilter::postApplyResource(RenderElement& renderer, GraphicsContext*& context, OptionSet<RenderSVGResourceMode> resourceMode, const Path*, const RenderSVGShape*)
230{
231 ASSERT(context);
232 ASSERT_UNUSED(resourceMode, !resourceMode);
233
234 auto findResult = m_rendererFilterDataMap.find(&renderer);
235 if (findResult == m_rendererFilterDataMap.end())
236 return;
237
238 FilterData& filterData = *findResult->value;
239
240 LOG_WITH_STREAM(Filters, stream << "\nRenderSVGResourceFilter " << this << " postApplyResource - renderer " << &renderer << " filter state " << filterData.state);
241
242 switch (filterData.state) {
243 case FilterData::MarkedForRemoval:
244 m_rendererFilterDataMap.remove(findResult);
245 return;
246
247 case FilterData::CycleDetected:
248 case FilterData::Applying:
249 // We have a cycle if we are already applying the data.
250 // This can occur due to FeImage referencing a source that makes use of the FEImage itself.
251 // This is the first place we've hit the cycle, so set the state back to PaintingSource so the return stack
252 // will continue correctly.
253 filterData.state = FilterData::PaintingSource;
254 return;
255
256 case FilterData::PaintingSource:
257 if (!filterData.savedContext) {
258 removeClientFromCache(renderer);
259 return;
260 }
261
262 context = filterData.savedContext;
263 filterData.savedContext = nullptr;
264 break;
265
266 case FilterData::Built:
267 break;
268 }
269
270 FilterEffect* lastEffect = filterData.builder->lastEffect();
271
272 if (lastEffect && !filterData.boundaries.isEmpty() && !lastEffect->filterPrimitiveSubregion().isEmpty()) {
273 // This is the real filtering of the object. It just needs to be called on the
274 // initial filtering process. We just take the stored filter result on a
275 // second drawing.
276 if (filterData.state != FilterData::Built)
277 filterData.filter->setSourceImage(WTFMove(filterData.sourceGraphicBuffer));
278
279 // Always true if filterData is just built (filterData->state == FilterData::Built).
280 if (!lastEffect->hasResult()) {
281 filterData.state = FilterData::Applying;
282 lastEffect->apply();
283 lastEffect->correctFilterResultIfNeeded();
284 lastEffect->transformResultColorSpace(DestinationColorSpace::SRGB());
285 }
286 filterData.state = FilterData::Built;
287
288 ImageBuffer* resultImage = lastEffect->imageBufferResult();
289 if (resultImage) {
290 context->concatCTM(filterData.shearFreeAbsoluteTransform.inverse().value_or(AffineTransform()));
291
292 context->scale(FloatSize(1 / filterData.filter->filterResolution().width(), 1 / filterData.filter->filterResolution().height()));
293 context->drawImageBuffer(*resultImage, lastEffect->absolutePaintRect());
294 context->scale(filterData.filter->filterResolution());
295
296 context->concatCTM(filterData.shearFreeAbsoluteTransform);
297 }
298 }
299 filterData.sourceGraphicBuffer = nullptr;
300
301 LOG_WITH_STREAM(Filters, stream << "RenderSVGResourceFilter " << this << " postApplyResource done\n");
302}

309void RenderSVGResourceFilter::primitiveAttributeChanged(RenderObject* object, const QualifiedName& attribute)
310{
311 SVGFilterPrimitiveStandardAttributes* primitve = static_cast<SVGFilterPrimitiveStandardAttributes*>(object->node());
312
313 LOG(Filters, "RenderSVGResourceFilter %p primitiveAttributeChanged renderer %p", this, object);
314
315 for (const auto& objectFilterDataPair : m_rendererFilterDataMap) {
316 const auto& filterData = objectFilterDataPair.value;
317 if (filterData->state != FilterData::Built)
318 continue;
319
320 SVGFilterBuilder* builder = filterData->builder.get();
321 FilterEffect* effect = builder->effectByRenderer(object);
322 if (!effect)
323 continue;
324 // Since all effects shares the same attribute value, all
325 // or none of them will be changed.
326 if (!primitve->setFilterEffectAttribute(effect, attribute))
327 return;
328 builder->clearResultsRecursive(effect);
329
330 // Repaint the image on the screen.
331 markClientForInvalidation(*objectFilterDataPair.key, RepaintInvalidation);
332 }
333 markAllClientLayersForInvalidation();
334}

342TextStream& operator<<(TextStream& ts, FilterData::FilterDataState state)
343{
344 switch (state) {
345 case FilterData::PaintingSource:
346 ts << "painting source";
347 break;
348 case FilterData::Applying:
349 ts << "applying";
350 break;
351 case FilterData::Built:
352 ts << "built";
353 break;
354 case FilterData::CycleDetected:
355 ts << "cycle detected";
356 break;
357 case FilterData::MarkedForRemoval:
358 ts << "marked for removal";
359 break;
360 }
361 return ts;
362}
