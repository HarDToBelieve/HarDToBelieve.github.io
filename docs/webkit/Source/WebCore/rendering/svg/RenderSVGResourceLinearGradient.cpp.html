28WTF_MAKE_ISO_ALLOCATED_IMPL(RenderSVGResourceLinearGradient);

30RenderSVGResourceLinearGradient::RenderSVGResourceLinearGradient(SVGLinearGradientElement& element, RenderStyle&& style)
31 : RenderSVGResourceGradient(element, WTFMove(style))
32{
33}
34
35RenderSVGResourceLinearGradient::~RenderSVGResourceLinearGradient() = default;
36
37bool RenderSVGResourceLinearGradient::collectGradientAttributes()
38{
39 m_attributes = LinearGradientAttributes();
40 return linearGradientElement().collectGradientAttributes(m_attributes);
41}
42
43FloatPoint RenderSVGResourceLinearGradient::startPoint(const LinearGradientAttributes& attributes) const
44{
45 return SVGLengthContext::resolvePoint(&linearGradientElement(), attributes.gradientUnits(), attributes.x1(), attributes.y1());
46}
47
48FloatPoint RenderSVGResourceLinearGradient::endPoint(const LinearGradientAttributes& attributes) const
49{
50 return SVGLengthContext::resolvePoint(&linearGradientElement(), attributes.gradientUnits(), attributes.x2(), attributes.y2());
51}
52
53Ref<Gradient> RenderSVGResourceLinearGradient::buildGradient(const RenderStyle& style) const
54{
55 auto gradient = Gradient::create(Gradient::LinearData { startPoint(m_attributes), endPoint(m_attributes) });
56 gradient->setSpreadMethod(platformSpreadMethodFromSVGType(m_attributes.spreadMethod()));
57 addStops(gradient, m_attributes.stops(), style);
58 return gradient;
59}
