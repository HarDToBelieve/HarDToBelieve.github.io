57bool RenderSVGResourceMasker::applyResource(RenderElement& renderer, const RenderStyle&, GraphicsContext*& context, OptionSet<RenderSVGResourceMode> resourceMode)
58{
59 ASSERT(context);
60 ASSERT_UNUSED(resourceMode, !resourceMode);
61
62 bool missingMaskerData = !m_masker.contains(&renderer);
63 if (missingMaskerData)
64 m_masker.set(&renderer, makeUnique<MaskerData>());
65
66 MaskerData* maskerData = m_masker.get(&renderer);
67 AffineTransform absoluteTransform = SVGRenderingContext::calculateTransformationToOutermostCoordinateSystem(renderer);
68 FloatRect repaintRect = renderer.repaintRectInLocalCoordinates();
69
70 if (!maskerData->maskImage && !repaintRect.isEmpty()) {
71 const SVGRenderStyle& svgStyle = style().svgStyle();
72
73 auto maskColorSpace = DestinationColorSpace::SRGB();
74 auto drawColorSpace = DestinationColorSpace::SRGB();
75
76#if ENABLE(DESTINATION_COLOR_SPACE_LINEAR_SRGB)
77 if (svgStyle.colorInterpolation() == ColorInterpolation::LinearRGB) {
78#if USE(CG)
79 maskColorSpace = DestinationColorSpace::LinearSRGB();
80#endif
81 drawColorSpace = DestinationColorSpace::LinearSRGB();
82 }
83#endif
84
85 // FIXME (149470): This image buffer should not be unconditionally unaccelerated. Making it match the context breaks alpha masking, though.
86 maskerData->maskImage = SVGRenderingContext::createImageBuffer(repaintRect, absoluteTransform, maskColorSpace, RenderingMode::Unaccelerated, context);
87 if (!maskerData->maskImage)
88 return false;
89
90 if (!drawContentIntoMaskImage(maskerData, drawColorSpace, &renderer))
91 maskerData->maskImage = nullptr;
92 }
93
94 if (!maskerData->maskImage)
95 return false;
96
97 SVGRenderingContext::clipToImageBuffer(*context, absoluteTransform, repaintRect, maskerData->maskImage, missingMaskerData);
98 return true;
99}

101bool RenderSVGResourceMasker::drawContentIntoMaskImage(MaskerData* maskerData, const DestinationColorSpace& colorSpace, RenderObject* object)
102{
103 GraphicsContext& maskImageContext = maskerData->maskImage->context();
104
105 // Eventually adjust the mask image context according to the target objectBoundingBox.
106 AffineTransform maskContentTransformation;
107 if (maskElement().maskContentUnits() == SVGUnitTypes::SVG_UNIT_TYPE_OBJECTBOUNDINGBOX) {
108 FloatRect objectBoundingBox = object->objectBoundingBox();
109 maskContentTransformation.translate(objectBoundingBox.location());
110 maskContentTransformation.scale(objectBoundingBox.size());
111 maskImageContext.concatCTM(maskContentTransformation);
112 }
113
114 // Draw the content into the ImageBuffer.
115 for (auto& child : childrenOfType<SVGElement>(maskElement())) {
116 auto renderer = child.renderer();
117 if (!renderer)
118 continue;
119 if (renderer->needsLayout())
120 return false;
121 const RenderStyle& style = renderer->style();
122 if (style.display() == DisplayType::None || style.visibility() != Visibility::Visible)
123 continue;
124 SVGRenderingContext::renderSubtreeToContext(maskImageContext, *renderer, maskContentTransformation);
125 }
126
127#if !USE(CG)
128 maskerData->maskImage->transformToColorSpace(colorSpace);
129#else
130 UNUSED_PARAM(colorSpace);
131#endif
132
133 // Create the luminance mask.
134 if (style().svgStyle().maskType() == MaskType::Luminance)
135 maskerData->maskImage->convertToLuminanceMask();
136
137 return true;
138}

140void RenderSVGResourceMasker::calculateMaskContentRepaintRect()
141{
142 for (Node* childNode = maskElement().firstChild(); childNode; childNode = childNode->nextSibling()) {
143 RenderObject* renderer = childNode->renderer();
144 if (!childNode->isSVGElement() || !renderer)
145 continue;
146 const RenderStyle& style = renderer->style();
147 if (style.display() == DisplayType::None || style.visibility() != Visibility::Visible)
148 continue;
149 m_maskContentBoundaries.unite(renderer->localToParentTransform().mapRect(renderer->repaintRectInLocalCoordinates()));
150 }
151}

153FloatRect RenderSVGResourceMasker::resourceBoundingBox(const RenderObject& object)
154{
155 FloatRect objectBoundingBox = object.objectBoundingBox();
156 FloatRect maskBoundaries = SVGLengthContext::resolveRectangle<SVGMaskElement>(&maskElement(), maskElement().maskUnits(), objectBoundingBox);
157
158 // Resource was not layouted yet. Give back clipping rect of the mask.
159 if (selfNeedsLayout())
160 return maskBoundaries;
161
162 if (m_maskContentBoundaries.isEmpty())
163 calculateMaskContentRepaintRect();
164
165 FloatRect maskRect = m_maskContentBoundaries;
166 if (maskElement().maskContentUnits() == SVGUnitTypes::SVG_UNIT_TYPE_OBJECTBOUNDINGBOX) {
167 AffineTransform transform;
168 transform.translate(objectBoundingBox.location());
169 transform.scale(objectBoundingBox.size());
170 maskRect = transform.mapRect(maskRect);
171 }
172
173 maskRect.intersect(maskBoundaries);
174 return maskRect;
175}
