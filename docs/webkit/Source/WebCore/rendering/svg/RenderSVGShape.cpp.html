53RenderSVGShape::RenderSVGShape(SVGGraphicsElement& element, RenderStyle&& style)
54 : RenderSVGModelObject(element, WTFMove(style))
55 , m_needsBoundariesUpdate(false) // Default is false, the cached rects are empty from the beginning.
56 , m_needsShapeUpdate(true) // Default is true, so we grab a Path object once from SVGGraphicsElement.
57 , m_needsTransformUpdate(true) // Default is true, so we grab a AffineTransform object once from SVGGraphicsElement.
58{
59}

63void RenderSVGShape::updateShapeFromElement()

72bool RenderSVGShape::isEmpty() const
73{
74 // This function should never be called before assigning a new Path to m_path.
75 // But this bug can happen if this renderer was created and its layout was not
76 // done before painting. Assert this did not happen but do not crash.
77 ASSERT(hasPath());
78 return !hasPath() || path().isEmpty();
79}

86void RenderSVGShape::strokeShape(GraphicsContext& context) const
87{
88 ASSERT(m_path);
89 Path* usePath = m_path.get();

91 if (hasNonScalingStroke())
92 usePath = nonScalingStrokePath(usePath, nonScalingStrokeTransform());
93
94 context.strokePath(*usePath);
95}

97bool RenderSVGShape::shapeDependentStrokeContains(const FloatPoint& point, PointCoordinateSpace pointCoordinateSpace)
98{
99 ASSERT(m_path);
100
101 if (hasNonScalingStroke() && pointCoordinateSpace != LocalCoordinateSpace) {
102 AffineTransform nonScalingTransform = nonScalingStrokeTransform();
103 Path* usePath = nonScalingStrokePath(m_path.get(), nonScalingTransform);
104 return usePath->strokeContains(nonScalingTransform.mapPoint(point), [this] (GraphicsContext& context) {
105 SVGRenderSupport::applyStrokeStyleToContext(context, style(), *this);
106 });
107 }
108
109 return m_path->strokeContains(point, [this] (GraphicsContext& context) {
110 SVGRenderSupport::applyStrokeStyleToContext(context, style(), *this);
111 });
112}

119bool RenderSVGShape::fillContains(const FloatPoint& point, bool requiresFill, const WindRule fillRule)
120{
121 if (m_fillBoundingBox.isEmpty() || !m_fillBoundingBox.contains(point))
122 return false;
123
124 Color fallbackColor;
125 if (requiresFill && !RenderSVGResource::fillPaintingResource(*this, style(), fallbackColor))
126 return false;
127
128 return shapeDependentFillContains(point, fillRule);
129}
130
131bool RenderSVGShape::strokeContains(const FloatPoint& point, bool requiresStroke)
132{
133 if (strokeBoundingBox().isEmpty() || !strokeBoundingBox().contains(point))
134 return false;
135
136 Color fallbackColor;
137 if (requiresStroke && !RenderSVGResource::strokePaintingResource(*this, style(), fallbackColor))
138 return false;
139
140 return shapeDependentStrokeContains(point);
141}

143void RenderSVGShape::layout()
144{
145 StackStats::LayoutCheckPoint layoutCheckPoint;
146 LayoutRepainter repainter(*this, SVGRenderSupport::checkForSVGRepaintDuringLayout(*this) && selfNeedsLayout());
147
148 bool updateCachedBoundariesInParents = false;
149
150 if (m_needsShapeUpdate || m_needsBoundariesUpdate) {
151 updateShapeFromElement();
152 m_needsShapeUpdate = false;
153 updateRepaintBoundingBox();
154 m_needsBoundariesUpdate = false;
155 updateCachedBoundariesInParents = true;
156 }
157
158 if (m_needsTransformUpdate) {
159 m_localTransform = graphicsElement().animatedLocalTransform();
160 m_needsTransformUpdate = false;
161 updateCachedBoundariesInParents = true;
162 }
163
164 // Invalidate all resources of this client if our layout changed.
165 if (everHadLayout() && selfNeedsLayout())
166 SVGResourcesCache::clientLayoutChanged(*this);
167
168 // If our bounds changed, notify the parents.
169 if (updateCachedBoundariesInParents)
170 RenderSVGModelObject::setNeedsBoundariesUpdate();
171
172 repainter.repaintAfterLayout();
173 clearNeedsLayout();
174}

176Path* RenderSVGShape::nonScalingStrokePath(const Path* path, const AffineTransform& strokeTransform) const
177{
178 static NeverDestroyed<Path> tempPath;
179
180 tempPath.get() = *path;

186bool RenderSVGShape::setupNonScalingStrokeContext(AffineTransform& strokeTransform, GraphicsContextStateSaver& stateSaver)
187{
188 std::optional<AffineTransform> inverse = strokeTransform.inverse();
189 if (!inverse)
190 return false;
191
192 stateSaver.save();
193 stateSaver.context()->concatCTM(inverse.value());
194 return true;
195}

202bool RenderSVGShape::shouldGenerateMarkerPositions() const
203{
204 if (!style().svgStyle().hasMarkers())
205 return false;
206
207 if (!graphicsElement().supportsMarkers())
208 return false;
209
210 auto* resources = SVGResourcesCache::cachedResourcesForRenderer(*this);
211 if (!resources)
212 return false;
213
214 return resources->markerStart() || resources->markerMid() || resources->markerEnd();
215}

217void RenderSVGShape::fillShape(const RenderStyle& style, GraphicsContext& originalContext)
218{
219 GraphicsContext* context = &originalContext;
220 Color fallbackColor;
221 if (RenderSVGResource* fillPaintingResource = RenderSVGResource::fillPaintingResource(*this, style, fallbackColor)) {
222 if (fillPaintingResource->applyResource(*this, style, context, RenderSVGResourceMode::ApplyToFill))
223 fillPaintingResource->postApplyResource(*this, context, RenderSVGResourceMode::ApplyToFill, nullptr, this);
224 else if (fallbackColor.isValid()) {
225 RenderSVGResourceSolidColor* fallbackResource = RenderSVGResource::sharedSolidPaintingResource();
226 fallbackResource->setColor(fallbackColor);
227 if (fallbackResource->applyResource(*this, style, context, RenderSVGResourceMode::ApplyToFill))
228 fallbackResource->postApplyResource(*this, context, RenderSVGResourceMode::ApplyToFill, nullptr, this);
229 }
230 }
231}

233void RenderSVGShape::strokeShape(const RenderStyle& style, GraphicsContext& originalContext)
234{
235 GraphicsContext* context = &originalContext;
236 Color fallbackColor;
237 if (RenderSVGResource* strokePaintingResource = RenderSVGResource::strokePaintingResource(*this, style, fallbackColor)) {
238 if (strokePaintingResource->applyResource(*this, style, context, RenderSVGResourceMode::ApplyToStroke))
239 strokePaintingResource->postApplyResource(*this, context, RenderSVGResourceMode::ApplyToStroke, nullptr, this);
240 else if (fallbackColor.isValid()) {
241 RenderSVGResourceSolidColor* fallbackResource = RenderSVGResource::sharedSolidPaintingResource();
242 fallbackResource->setColor(fallbackColor);
243 if (fallbackResource->applyResource(*this, style, context, RenderSVGResourceMode::ApplyToStroke))
244 fallbackResource->postApplyResource(*this, context, RenderSVGResourceMode::ApplyToStroke, nullptr, this);
245 }
246 }

249void RenderSVGShape::strokeShape(GraphicsContext& context)
250{
251 if (!style().hasVisibleStroke())
252 return;
253
254 GraphicsContextStateSaver stateSaver(context, false);
255 if (hasNonScalingStroke()) {
256 AffineTransform nonScalingTransform = nonScalingStrokeTransform();
257 if (!setupNonScalingStrokeContext(nonScalingTransform, stateSaver))
258 return;
259 }
260 strokeShape(style(), context);
261}

263void RenderSVGShape::fillStrokeMarkers(PaintInfo& childPaintInfo)
264{
265 auto paintOrder = RenderStyle::paintTypesForPaintOrder(style().paintOrder());
266 for (unsigned i = 0; i < paintOrder.size(); ++i) {
267 switch (paintOrder.at(i)) {
268 case PaintType::Fill:
269 fillShape(style(), childPaintInfo.context());
270 break;
271 case PaintType::Stroke:
272 strokeShape(childPaintInfo.context());
273 break;
274 case PaintType::Markers:
275 if (!m_markerPositions.isEmpty())
276 drawMarkers(childPaintInfo);
277 break;
278 }
279 }
280}

282void RenderSVGShape::paint(PaintInfo& paintInfo, const LayoutPoint&)
283{
284 if (paintInfo.context().paintingDisabled() || paintInfo.phase != PaintPhase::Foreground
285 || style().visibility() == Visibility::Hidden || isEmpty())
286 return;
287 FloatRect boundingBox = repaintRectInLocalCoordinates();
288 if (!SVGRenderSupport::paintInfoIntersectsRepaintRect(boundingBox, m_localTransform, paintInfo))
289 return;
290
291 PaintInfo childPaintInfo(paintInfo);
292 GraphicsContextStateSaver stateSaver(childPaintInfo.context());
293 childPaintInfo.applyTransform(m_localTransform);
294
295 if (childPaintInfo.phase == PaintPhase::Foreground) {
296 SVGRenderingContext renderingContext(*this, childPaintInfo);
297
298 if (renderingContext.isRenderingPrepared()) {
299 const SVGRenderStyle& svgStyle = style().svgStyle();
300 if (svgStyle.shapeRendering() == ShapeRendering::CrispEdges)
301 childPaintInfo.context().setShouldAntialias(false);
302
303 fillStrokeMarkers(childPaintInfo);
304 }
305 }
306
307 if (style().outlineWidth())
308 paintOutline(childPaintInfo, IntRect(boundingBox));
309}

343bool RenderSVGShape::nodeAtFloatPoint(const HitTestRequest& request, HitTestResult& result, const FloatPoint& pointInParent, HitTestAction hitTestAction)
344{
345 // We only draw in the forground phase, so we only hit-test then.
346 if (hitTestAction != HitTestForeground)
347 return false;
348
349 FloatPoint localPoint = m_localTransform.inverse().value_or(AffineTransform()).mapPoint(pointInParent);
350
351 if (!SVGRenderSupport::pointInClippingArea(*this, localPoint))
352 return false;
353
354 SVGHitTestCycleDetectionScope hitTestScope(*this);
355
356 PointerEventsHitRules hitRules(PointerEventsHitRules::SVG_PATH_HITTESTING, request, style().pointerEvents());
357 bool isVisible = (style().visibility() == Visibility::Visible);
358 if (isVisible || !hitRules.requireVisible) {
359 const SVGRenderStyle& svgStyle = style().svgStyle();
360 WindRule fillRule = svgStyle.fillRule();
361 if (request.svgClipContent())
362 fillRule = svgStyle.clipRule();
363 if ((hitRules.canHitStroke && (svgStyle.hasStroke() || !hitRules.requireStroke) && strokeContains(localPoint, hitRules.requireStroke))
364 || (hitRules.canHitFill && (svgStyle.hasFill() || !hitRules.requireFill) && fillContains(localPoint, hitRules.requireFill, fillRule))
365 || (hitRules.canHitBoundingBox && objectBoundingBox().contains(localPoint))) {
366 updateHitTestResult(result, LayoutPoint(localPoint));
367 if (result.addNodeToListBasedTestResult(nodeForHitTest(), request, flooredLayoutPoint(localPoint)) == HitTestProgress::Stop)
368 return true;
369 }
370 }
371 return false;
372}

374static inline RenderSVGResourceMarker* markerForType(SVGMarkerType type, RenderSVGResourceMarker* markerStart, RenderSVGResourceMarker* markerMid, RenderSVGResourceMarker* markerEnd)
375{
376 switch (type) {
377 case StartMarker:
378 return markerStart;
379 case MidMarker:
380 return markerMid;
381 case EndMarker:
382 return markerEnd;
383 }
384
385 ASSERT_NOT_REACHED();
386 return 0;
387}

389FloatRect RenderSVGShape::markerRect(float strokeWidth) const
390{
391 ASSERT(!m_markerPositions.isEmpty());
392
393 auto* resources = SVGResourcesCache::cachedResourcesForRenderer(*this);
394 ASSERT(resources);
395
396 RenderSVGResourceMarker* markerStart = resources->markerStart();
397 RenderSVGResourceMarker* markerMid = resources->markerMid();
398 RenderSVGResourceMarker* markerEnd = resources->markerEnd();
399 ASSERT(markerStart || markerMid || markerEnd);
400
401 FloatRect boundaries;
402 unsigned size = m_markerPositions.size();
403 for (unsigned i = 0; i < size; ++i) {
404 if (RenderSVGResourceMarker* marker = markerForType(m_markerPositions[i].type, markerStart, markerMid, markerEnd))
405 boundaries.unite(marker->markerBoundaries(marker->markerTransformation(m_markerPositions[i].origin, m_markerPositions[i].angle, strokeWidth)));
406 }
407 return boundaries;
408}

414
415FloatRect RenderSVGShape::calculateStrokeBoundingBox() const
416{
417 ASSERT(m_path);
418 FloatRect strokeBoundingBox = m_fillBoundingBox;
419
420 if (style().svgStyle().hasStroke()) {
421 if (hasNonScalingStroke()) {
422 AffineTransform nonScalingTransform = nonScalingStrokeTransform();
423 if (std::optional<AffineTransform> inverse = nonScalingTransform.inverse()) {
424 Path* usePath = nonScalingStrokePath(m_path.get(), nonScalingTransform);
425 FloatRect strokeBoundingRect = usePath->strokeBoundingRect(Function<void(GraphicsContext&)> { [this] (GraphicsContext& context) {
426 SVGRenderSupport::applyStrokeStyleToContext(context, style(), *this);
427 }});
428 strokeBoundingRect = inverse.value().mapRect(strokeBoundingRect);
429 strokeBoundingBox.unite(strokeBoundingRect);
430 }
431 } else {
432 strokeBoundingBox.unite(path().strokeBoundingRect(Function<void(GraphicsContext&)> { [this] (GraphicsContext& context) {
433 SVGRenderSupport::applyStrokeStyleToContext(context, style(), *this);
434 }}));
435 }
436 }
437
438 if (!m_markerPositions.isEmpty())
439 strokeBoundingBox.unite(markerRect(strokeWidth()));
440
441 return strokeBoundingBox;
442}

458bool RenderSVGShape::hasSmoothStroke() const
459{
460 const SVGRenderStyle& svgStyle = style().svgStyle();
461 return svgStyle.strokeDashArray().isEmpty()
462 && style().strokeMiterLimit() == style().initialStrokeMiterLimit()
463 && style().joinStyle() == style().initialJoinStyle()
464 && style().capStyle() == style().initialCapStyle();

467void RenderSVGShape::drawMarkers(PaintInfo& paintInfo)
468{
469 ASSERT(!m_markerPositions.isEmpty());
470
471 auto* resources = SVGResourcesCache::cachedResourcesForRenderer(*this);
472 if (!resources)
473 return;
474
475 RenderSVGResourceMarker* markerStart = resources->markerStart();
476 RenderSVGResourceMarker* markerMid = resources->markerMid();
477 RenderSVGResourceMarker* markerEnd = resources->markerEnd();
478 if (!markerStart && !markerMid && !markerEnd)
479 return;
480
481 float strokeWidth = this->strokeWidth();
482 unsigned size = m_markerPositions.size();
483 for (unsigned i = 0; i < size; ++i) {
484 if (RenderSVGResourceMarker* marker = markerForType(m_markerPositions[i].type, markerStart, markerMid, markerEnd))
485 marker->draw(paintInfo, marker->markerTransformation(m_markerPositions[i].origin, m_markerPositions[i].angle, strokeWidth));
486 }
487}

494void RenderSVGShape::processMarkerPositions()
495{
496 m_markerPositions.clear();
497
498 if (!shouldGenerateMarkerPositions())
499 return;
500
501 ASSERT(m_path);
502
503 SVGMarkerData markerData(m_markerPositions, SVGResourcesCache::cachedResourcesForRenderer(*this)->markerReverseStart());
504 m_path->apply([&markerData](const PathElement& pathElement) {
505 SVGMarkerData::updateFromPathElement(markerData, pathElement);
506 });
507 markerData.pathIsDone();
508}
