88int SVGInlineTextBox::offsetForPositionInFragment(const SVGTextFragment& fragment, float position, bool includePartialGlyphs) const
89{
90 float scalingFactor = renderer().scalingFactor();
91 ASSERT(scalingFactor);
92
93 TextRun textRun = constructTextRun(renderer().style(), fragment);
94
95 // Eventually handle lengthAdjust="spacingAndGlyphs".
96 // FIXME: Handle vertical text.
97 AffineTransform fragmentTransform;
98 fragment.buildFragmentTransform(fragmentTransform);
99 if (!fragmentTransform.isIdentity())
100 textRun.setHorizontalGlyphStretch(narrowPrecisionToFloat(fragmentTransform.xScale()));
101
102 return fragment.characterOffset - start() + renderer().scaledFont().offsetForPosition(textRun, position * scalingFactor, includePartialGlyphs);
103}

112FloatRect SVGInlineTextBox::selectionRectForTextFragment(const SVGTextFragment& fragment, unsigned startPosition, unsigned endPosition, const RenderStyle& style) const
113{
114 ASSERT_WITH_SECURITY_IMPLICATION(startPosition < endPosition);
115
116 float scalingFactor = renderer().scalingFactor();
117 ASSERT(scalingFactor);
118
119 const FontCascade& scaledFont = renderer().scaledFont();
120 const FontMetrics& scaledFontMetrics = scaledFont.fontMetrics();
121 FloatPoint textOrigin(fragment.x, fragment.y);
122 if (scalingFactor != 1)
123 textOrigin.scale(scalingFactor);
124
125 textOrigin.move(0, -scaledFontMetrics.floatAscent());
126
127 LayoutRect selectionRect { textOrigin, LayoutSize(0, LayoutUnit(fragment.height * scalingFactor)) };
128 TextRun run = constructTextRun(style, fragment);
129 scaledFont.adjustSelectionRectForText(run, selectionRect, startPosition, endPosition);
130 FloatRect snappedSelectionRect = snapRectToDevicePixelsWithWritingDirection(selectionRect, renderer().document().deviceScaleFactor(), run.ltr());
131 if (scalingFactor == 1)
132 return snappedSelectionRect;
133
134 snappedSelectionRect.scale(1 / scalingFactor);
135 return snappedSelectionRect;
136}

138LayoutRect SVGInlineTextBox::localSelectionRect(unsigned start, unsigned end) const
139{
140 auto [clampedStart, clampedEnd] = selectableRange().clamp(start, end);
141
142 if (clampedStart >= clampedEnd)
143 return LayoutRect();
144
145 auto& style = renderer().style();
146
147 AffineTransform fragmentTransform;
148 FloatRect selectionRect;
149 unsigned fragmentStartPosition = 0;
150 unsigned fragmentEndPosition = 0;
151
152 unsigned textFragmentsSize = m_textFragments.size();
153 for (unsigned i = 0; i < textFragmentsSize; ++i) {
154 const SVGTextFragment& fragment = m_textFragments.at(i);
155
156 fragmentStartPosition = clampedStart;
157 fragmentEndPosition = clampedEnd;
158 if (!mapStartEndPositionsIntoFragmentCoordinates(fragment, fragmentStartPosition, fragmentEndPosition))
159 continue;
160
161 FloatRect fragmentRect = selectionRectForTextFragment(fragment, fragmentStartPosition, fragmentEndPosition, style);
162 fragment.buildFragmentTransform(fragmentTransform);
163 if (!fragmentTransform.isIdentity())
164 fragmentRect = fragmentTransform.mapRect(fragmentRect);
165
166 selectionRect.unite(fragmentRect);
167 }
168
169 return enclosingIntRect(selectionRect);
170}

172static inline bool textShouldBePainted(const RenderSVGInlineText& textRenderer)
173{
174 // FontCascade::pixelSize(), returns FontDescription::computedPixelSize(), which returns "int(x + 0.5)".
175 // If the absolute font size on screen is below x=0.5, don't render anything.
176 return textRenderer.scaledFont().pixelSize();
177}

179void SVGInlineTextBox::paintSelectionBackground(PaintInfo& paintInfo)
180{
181 ASSERT(paintInfo.shouldPaintWithinRoot(renderer()));
182 ASSERT(paintInfo.phase == PaintPhase::Foreground || paintInfo.phase == PaintPhase::Selection);
183 ASSERT(truncation() == cNoTruncation);
184
185 if (renderer().style().visibility() != Visibility::Visible)
186 return;
187
188 auto& parentRenderer = parent()->renderer();
189 ASSERT(!parentRenderer.document().printing());
190
191 // Determine whether or not we're selected.
192 bool paintSelectedTextOnly = paintInfo.phase == PaintPhase::Selection;
193 bool hasSelection = selectionState() != RenderObject::HighlightState::None;
194 if (!hasSelection || paintSelectedTextOnly)
195 return;
196
197 Color backgroundColor = renderer().selectionBackgroundColor();
198 if (!backgroundColor.isVisible())
199 return;
200
201 if (!textShouldBePainted(renderer()))
202 return;
203
204 auto& style = parentRenderer.style();
205
206 auto [startPosition, endPosition] = selectionStartEnd();
207
208 unsigned fragmentStartPosition = 0;
209 unsigned fragmentEndPosition = 0;
210 AffineTransform fragmentTransform;
211 unsigned textFragmentsSize = m_textFragments.size();
212 for (unsigned i = 0; i < textFragmentsSize; ++i) {
213 SVGTextFragment& fragment = m_textFragments.at(i);
214 ASSERT(!m_paintingResource);
215
216 fragmentStartPosition = startPosition;
217 fragmentEndPosition = endPosition;
218 if (!mapStartEndPositionsIntoFragmentCoordinates(fragment, fragmentStartPosition, fragmentEndPosition))
219 continue;
220
221 GraphicsContextStateSaver stateSaver(paintInfo.context());
222 fragment.buildFragmentTransform(fragmentTransform);
223 if (!fragmentTransform.isIdentity())
224 paintInfo.context().concatCTM(fragmentTransform);
225
226 paintInfo.context().setFillColor(backgroundColor);
227 paintInfo.context().fillRect(selectionRectForTextFragment(fragment, fragmentStartPosition, fragmentEndPosition, style), backgroundColor);
228
229 setPaintingResourceMode({ });
230 }
231
232 ASSERT(!m_paintingResource);
233}

235void SVGInlineTextBox::paint(PaintInfo& paintInfo, const LayoutPoint& paintOffset, LayoutUnit, LayoutUnit)
236{
237 ASSERT(paintInfo.shouldPaintWithinRoot(renderer()));
238 ASSERT(paintInfo.phase == PaintPhase::Foreground || paintInfo.phase == PaintPhase::Selection);
239 ASSERT(truncation() == cNoTruncation);
240
241 if (renderer().style().visibility() != Visibility::Visible)
242 return;
243
244 // Note: We're explicitly not supporting composition & custom underlines and custom highlighters - unlike LegacyInlineTextBox.
245 // If we ever need that for SVG, it's very easy to refactor and reuse the code.
246
247 auto& parentRenderer = parent()->renderer();
248
249 bool paintSelectedTextOnly = paintInfo.phase == PaintPhase::Selection;
250 bool shouldPaintSelectionHighlight = !(paintInfo.paintBehavior.contains(PaintBehavior::SkipSelectionHighlight));
251 bool hasSelection = !parentRenderer.document().printing() && selectionState() != RenderObject::HighlightState::None;
252 if (!hasSelection && paintSelectedTextOnly)
253 return;
254
255 if (!textShouldBePainted(renderer()))
256 return;
257
258 auto& style = parentRenderer.style();
259
260 const SVGRenderStyle& svgStyle = style.svgStyle();
261
262 bool hasFill = svgStyle.hasFill();
263 bool hasVisibleStroke = style.hasVisibleStroke();
264
265 const RenderStyle* selectionStyle = &style;
266 if (hasSelection && shouldPaintSelectionHighlight) {
267 selectionStyle = parentRenderer.getCachedPseudoStyle(PseudoId::Selection);
268 if (selectionStyle) {
269 const SVGRenderStyle& svgSelectionStyle = selectionStyle->svgStyle();
270
271 if (!hasFill)
272 hasFill = svgSelectionStyle.hasFill();
273 if (!hasVisibleStroke)
274 hasVisibleStroke = selectionStyle->hasVisibleStroke();
275 } else
276 selectionStyle = &style;
277 }
278
279 if (renderer().view().frameView().paintBehavior().contains(PaintBehavior::RenderingSVGMask)) {
280 hasFill = true;
281 hasVisibleStroke = false;
282 }
283
284 AffineTransform fragmentTransform;
285 unsigned textFragmentsSize = m_textFragments.size();
286 for (unsigned i = 0; i < textFragmentsSize; ++i) {
287 SVGTextFragment& fragment = m_textFragments.at(i);
288 ASSERT(!m_paintingResource);
289
290 GraphicsContextStateSaver stateSaver(paintInfo.context());
291 fragment.buildFragmentTransform(fragmentTransform);
292 if (!fragmentTransform.isIdentity())
293 paintInfo.context().concatCTM(fragmentTransform);
294
295 // Spec: All text decorations except line-through should be drawn before the text is filled and stroked; thus, the text is rendered on top of these decorations.
296 auto decorations = style.textDecorationsInEffect();
297 if (decorations & TextDecoration::Underline)
298 paintDecoration(paintInfo.context(), TextDecoration::Underline, fragment);
299 if (decorations & TextDecoration::Overline)
300 paintDecoration(paintInfo.context(), TextDecoration::Overline, fragment);
301
302 auto paintOrder = RenderStyle::paintTypesForPaintOrder(style.paintOrder());
303 for (unsigned i = 0; i < paintOrder.size(); ++i) {
304 switch (paintOrder.at(i)) {
305 case PaintType::Fill:
306 if (!hasFill)
307 continue;
308 setPaintingResourceMode({ RenderSVGResourceMode::ApplyToFill, RenderSVGResourceMode::ApplyToText });
309 ASSERT(selectionStyle);
310 paintText(paintInfo.context(), style, *selectionStyle, fragment, hasSelection, paintSelectedTextOnly);
311 break;
312 case PaintType::Stroke:
313 if (!hasVisibleStroke)
314 continue;
315 setPaintingResourceMode({ RenderSVGResourceMode::ApplyToStroke, RenderSVGResourceMode::ApplyToText});
316 ASSERT(selectionStyle);
317 paintText(paintInfo.context(), style, *selectionStyle, fragment, hasSelection, paintSelectedTextOnly);
318 break;
319 case PaintType::Markers:
320 continue;
321 }
322 }
323
324 // Spec: Line-through should be drawn after the text is filled and stroked; thus, the line-through is rendered on top of the text.
325 if (decorations & TextDecoration::LineThrough)
326 paintDecoration(paintInfo.context(), TextDecoration::LineThrough, fragment);
327
328 setPaintingResourceMode({ });
329 }
330
331 // Finally, paint the outline if any.
332 if (renderer().style().hasOutline() && is<RenderInline>(parentRenderer))
333 downcast<RenderInline>(parentRenderer).paintOutline(paintInfo, paintOffset);
334
335 ASSERT(!m_paintingResource);
336}

338bool SVGInlineTextBox::acquirePaintingResource(GraphicsContext*& context, float scalingFactor, RenderBoxModelObject& renderer, const RenderStyle& style)
339{
340 ASSERT(scalingFactor);
341 ASSERT(!paintingResourceMode().isEmpty());
342
343 Color fallbackColor;
344 if (paintingResourceMode().contains(RenderSVGResourceMode::ApplyToFill))
345 m_paintingResource = RenderSVGResource::fillPaintingResource(renderer, style, fallbackColor);
346 else if (paintingResourceMode().contains(RenderSVGResourceMode::ApplyToStroke))
347 m_paintingResource = RenderSVGResource::strokePaintingResource(renderer, style, fallbackColor);
348 else {
349 // We're either called for stroking or filling.
350 ASSERT_NOT_REACHED();
351 }
352
353 if (!m_paintingResource)
354 return false;
355
356 if (!m_paintingResource->applyResource(renderer, style, context, paintingResourceMode())) {
357 if (fallbackColor.isValid()) {
358 RenderSVGResourceSolidColor* fallbackResource = RenderSVGResource::sharedSolidPaintingResource();
359 fallbackResource->setColor(fallbackColor);
360
361 m_paintingResource = fallbackResource;
362 m_paintingResource->applyResource(renderer, style, context, paintingResourceMode());
363 }
364 }
365
366 if (scalingFactor != 1 && paintingResourceMode().contains(RenderSVGResourceMode::ApplyToStroke))
367 context->setStrokeThickness(context->strokeThickness() * scalingFactor);
368
369 return true;
370}

390TextRun SVGInlineTextBox::constructTextRun(const RenderStyle& style, const SVGTextFragment& fragment) const
391{
392 TextRun run(StringView(renderer().text()).substring(fragment.characterOffset, fragment.length)
393 , 0 /* xPos, only relevant with allowTabs=true */
394 , 0 /* padding, only relevant for justified text, not relevant for SVG */
395 , AllowRightExpansion
396 , direction()
397 , dirOverride() || style.rtlOrdering() == Order::Visual /* directionalOverride */);
398
399 // We handle letter & word spacing ourselves.
400 run.disableSpacing();
401 return run;
402}

404bool SVGInlineTextBox::mapStartEndPositionsIntoFragmentCoordinates(const SVGTextFragment& fragment, unsigned& startPosition, unsigned& endPosition) const
405{
406 if (startPosition >= endPosition)
407 return false;
408
409 ASSERT(fragment.characterOffset >= start());
410 unsigned offset = fragment.characterOffset - start();
411 unsigned length = fragment.length;
412
413 if (startPosition >= offset + length || endPosition <= offset)
414 return false;
415
416 if (startPosition < offset)
417 startPosition = 0;
418 else {
419 ASSERT(startPosition >= offset);
420 startPosition -= offset;
421 }
422
423 if (endPosition > offset + length)
424 endPosition = length;
425 else {
426 ASSERT(endPosition >= offset);
427 endPosition -= offset;
428 }
429
430 ASSERT_WITH_SECURITY_IMPLICATION(startPosition < endPosition);
431 return true;
432}

434static inline float positionOffsetForDecoration(OptionSet<TextDecoration> decoration, const FontMetrics& fontMetrics, float thickness)
435{
436 // FIXME: For SVG Fonts we need to use the attributes defined in the <font-face> if specified.
437 // Compatible with Batik/Opera.
438 if (decoration == TextDecoration::Underline)
439 return fontMetrics.floatAscent() + thickness * 1.5f;
440 if (decoration == TextDecoration::Overline)
441 return thickness;
442 if (decoration == TextDecoration::LineThrough)
443 return fontMetrics.floatAscent() * 5 / 8.0f;
444
445 ASSERT_NOT_REACHED();
446 return 0.0f;
447}
448
449static inline float thicknessForDecoration(OptionSet<TextDecoration>, const FontCascade& font)
450{
451 // FIXME: For SVG Fonts we need to use the attributes defined in the <font-face> if specified.
452 // Compatible with Batik/Opera
453 return font.size() / 20.0f;
454}

456static inline RenderBoxModelObject& findRendererDefininingTextDecoration(LegacyInlineFlowBox* parentBox)
457{
458 // Lookup first render object in parent hierarchy which has text-decoration set.
459 RenderBoxModelObject* renderer = nullptr;
460 while (parentBox) {
461 renderer = &parentBox->renderer();
462
463 if (!renderer->style().textDecoration().isEmpty())
464 break;
465
466 parentBox = parentBox->parent();
467 }
468
469 ASSERT(renderer);
470 return *renderer;
471}

473void SVGInlineTextBox::paintDecoration(GraphicsContext& context, OptionSet<TextDecoration> decoration, const SVGTextFragment& fragment)
474{
475 if (renderer().style().textDecorationsInEffect().isEmpty())
476 return;
477
478 // Find out which render style defined the text-decoration, as its fill/stroke properties have to be used for drawing instead of ours.
479 auto& decorationRenderer = findRendererDefininingTextDecoration(parent());
480 const RenderStyle& decorationStyle = decorationRenderer.style();
481
482 if (decorationStyle.visibility() == Visibility::Hidden)
483 return;
484
485 const SVGRenderStyle& svgDecorationStyle = decorationStyle.svgStyle();
486
487 bool hasDecorationFill = svgDecorationStyle.hasFill();
488 bool hasVisibleDecorationStroke = decorationStyle.hasVisibleStroke();
489
490 if (hasDecorationFill) {
491 setPaintingResourceMode(RenderSVGResourceMode::ApplyToFill);
492 paintDecorationWithStyle(context, decoration, fragment, decorationRenderer);
493 }
494
495 if (hasVisibleDecorationStroke) {
496 setPaintingResourceMode(RenderSVGResourceMode::ApplyToStroke);
497 paintDecorationWithStyle(context, decoration, fragment, decorationRenderer);
498 }
499}

501void SVGInlineTextBox::paintDecorationWithStyle(GraphicsContext& context, OptionSet<TextDecoration> decoration, const SVGTextFragment& fragment, RenderBoxModelObject& decorationRenderer)
502{
503 ASSERT(!m_paintingResource);
504 ASSERT(!paintingResourceMode().isEmpty());
505
506 auto& decorationStyle = decorationRenderer.style();
507
508 float scalingFactor = 1;
509 FontCascade scaledFont;
510 RenderSVGInlineText::computeNewScaledFontForStyle(decorationRenderer, decorationStyle, scalingFactor, scaledFont);
511 ASSERT(scalingFactor);
512
513 // The initial y value refers to overline position.
514 float thickness = thicknessForDecoration(decoration, scaledFont);
515
516 if (fragment.width <= 0 && thickness <= 0)
517 return;
518
519 FloatPoint decorationOrigin(fragment.x, fragment.y);
520 float width = fragment.width;
521 const FontMetrics& scaledFontMetrics = scaledFont.fontMetrics();
522
523 GraphicsContextStateSaver stateSaver(context);
524 if (scalingFactor != 1) {
525 width *= scalingFactor;
526 decorationOrigin.scale(scalingFactor);
527 context.scale(1 / scalingFactor);
528 }
529
530 decorationOrigin.move(0, -scaledFontMetrics.floatAscent() + positionOffsetForDecoration(decoration, scaledFontMetrics, thickness));
531
532 Path path;
533 path.addRect(FloatRect(decorationOrigin, FloatSize(width, thickness)));
534
535 GraphicsContext* contextPtr = &context;
536 if (acquirePaintingResource(contextPtr, scalingFactor, decorationRenderer, decorationStyle))
537 releasePaintingResource(contextPtr, &path);
538}

540void SVGInlineTextBox::paintTextWithShadows(GraphicsContext& context, const RenderStyle& style, TextRun& textRun, const SVGTextFragment& fragment, unsigned startPosition, unsigned endPosition)
541{
542 float scalingFactor = renderer().scalingFactor();
543 ASSERT(scalingFactor);
544
545 const FontCascade& scaledFont = renderer().scaledFont();
546 const ShadowData* shadow = style.textShadow();
547
548 FloatPoint textOrigin(fragment.x, fragment.y);
549 FloatSize textSize(fragment.width, fragment.height);
550
551 if (scalingFactor != 1) {
552 textOrigin.scale(scalingFactor);
553 textSize.scale(scalingFactor);
554 }
555
556 FloatRect shadowRect(FloatPoint(textOrigin.x(), textOrigin.y() - scaledFont.fontMetrics().floatAscent()), textSize);
557
558 GraphicsContext* usedContext = &context;
559 do {
560 if (!prepareGraphicsContextForTextPainting(usedContext, scalingFactor, style))
561 break;
562
563 {
564 ShadowApplier shadowApplier(*usedContext, shadow, nullptr, shadowRect);
565
566 if (!shadowApplier.didSaveContext())
567 usedContext->save();
568 usedContext->scale(1 / scalingFactor);
569
570 scaledFont.drawText(*usedContext, textRun, textOrigin + shadowApplier.extraOffset(), startPosition, endPosition);
571
572 if (!shadowApplier.didSaveContext())
573 usedContext->restore();
574 }
575
576 restoreGraphicsContextAfterTextPainting(usedContext);
577
578 if (!shadow)
579 break;
580
581 shadow = shadow->next();
582 } while (shadow);
583}

585void SVGInlineTextBox::paintText(GraphicsContext& context, const RenderStyle& style, const RenderStyle& selectionStyle, const SVGTextFragment& fragment, bool hasSelection, bool paintSelectedTextOnly)
586{
587 unsigned startPosition = 0;
588 unsigned endPosition = 0;
589 if (hasSelection) {
590 std::tie(startPosition, endPosition) = selectionStartEnd();
591 hasSelection = mapStartEndPositionsIntoFragmentCoordinates(fragment, startPosition, endPosition);
592 }
593
594 // Fast path if there is no selection, just draw the whole chunk part using the regular style
595 TextRun textRun = constructTextRun(style, fragment);
596 if (!hasSelection || startPosition >= endPosition) {
597 paintTextWithShadows(context, style, textRun, fragment, 0, fragment.length);
598 return;
599 }
600
601 // Eventually draw text using regular style until the start position of the selection
602 if (startPosition > 0 && !paintSelectedTextOnly)
603 paintTextWithShadows(context, style, textRun, fragment, 0, startPosition);
604
605 // Draw text using selection style from the start to the end position of the selection
606 {
607 SVGResourcesCache::SetStyleForScope temporaryStyleChange(parent()->renderer(), style, selectionStyle);
608 paintTextWithShadows(context, selectionStyle, textRun, fragment, startPosition, endPosition);
609 }
610
611 // Eventually draw text using regular style from the end position of the selection to the end of the current chunk part
612 if (endPosition < fragment.length && !paintSelectedTextOnly)
613 paintTextWithShadows(context, style, textRun, fragment, endPosition, fragment.length);
614}

616FloatRect SVGInlineTextBox::calculateBoundaries() const
617{
618 FloatRect textRect;
619
620 float scalingFactor = renderer().scalingFactor();
621 ASSERT(scalingFactor);
622
623 float baseline = renderer().scaledFont().fontMetrics().floatAscent() / scalingFactor;
624
625 AffineTransform fragmentTransform;
626 unsigned textFragmentsSize = m_textFragments.size();
627 for (unsigned i = 0; i < textFragmentsSize; ++i) {
628 const SVGTextFragment& fragment = m_textFragments.at(i);
629 FloatRect fragmentRect(fragment.x, fragment.y - baseline, fragment.width, fragment.height);
630 fragment.buildFragmentTransform(fragmentTransform);
631 if (!fragmentTransform.isIdentity())
632 fragmentRect = fragmentTransform.mapRect(fragmentRect);
633
634 textRect.unite(fragmentRect);
635 }
636
637 return textRect;
638}

640bool SVGInlineTextBox::nodeAtPoint(const HitTestRequest& request, HitTestResult& result, const HitTestLocation& locationInContainer, const LayoutPoint& accumulatedOffset, LayoutUnit, LayoutUnit, HitTestAction)
641{
642 // FIXME: integrate with LegacyInlineTextBox::nodeAtPoint better.
643 ASSERT(!isLineBreak());
644
645 PointerEventsHitRules hitRules(PointerEventsHitRules::SVG_TEXT_HITTESTING, request, renderer().style().pointerEvents());
646 bool isVisible = renderer().style().visibility() == Visibility::Visible;
647 if (isVisible || !hitRules.requireVisible) {
648 if ((hitRules.canHitStroke && (renderer().style().svgStyle().hasStroke() || !hitRules.requireStroke))
649 || (hitRules.canHitFill && (renderer().style().svgStyle().hasFill() || !hitRules.requireFill))) {
650 FloatPoint boxOrigin(x(), y());
651 boxOrigin.moveBy(accumulatedOffset);
652 FloatRect rect(boxOrigin, size());
653 if (locationInContainer.intersects(rect)) {
654
655 float scalingFactor = renderer().scalingFactor();
656 ASSERT(scalingFactor);
657
658 float baseline = renderer().scaledFont().fontMetrics().floatAscent() / scalingFactor;
659
660 AffineTransform fragmentTransform;
661 for (auto& fragment : m_textFragments) {
662 FloatQuad fragmentQuad(FloatRect(fragment.x, fragment.y - baseline, fragment.width, fragment.height));
663 fragment.buildFragmentTransform(fragmentTransform);
664 if (!fragmentTransform.isIdentity())
665 fragmentQuad = fragmentTransform.mapQuad(fragmentQuad);
666
667 if (fragmentQuad.containsPoint(locationInContainer.point())) {
668 renderer().updateHitTestResult(result, locationInContainer.point() - toLayoutSize(accumulatedOffset));
669 if (result.addNodeToListBasedTestResult(renderer().nodeForHitTest(), request, locationInContainer, rect) == HitTestProgress::Stop)
670 return true;
671 }
672 }
673 }
674 }
675 }
676 return false;
677}
