53LayoutRect SVGRenderSupport::clippedOverflowRectForRepaint(const RenderElement& renderer, const RenderLayerModelObject* repaintContainer)
54{
55 // Return early for any cases where we don't actually paint
56 if (renderer.style().visibility() != Visibility::Visible && !renderer.enclosingLayer()->hasVisibleContent())
57 return LayoutRect();
58
59 // Pass our local paint rect to computeFloatVisibleRectInContainer() which will
60 // map to parent coords and recurse up the parent chain.
61 FloatRect repaintRect = renderer.repaintRectInLocalCoordinates();
62 const SVGRenderStyle& svgStyle = renderer.style().svgStyle();
63 if (const ShadowData* shadow = svgStyle.shadow())
64 shadow->adjustRectForShadow(repaintRect);
65 return enclosingLayoutRect(renderer.computeFloatRectForRepaint(repaintRect, repaintContainer));
66}

68std::optional<FloatRect> SVGRenderSupport::computeFloatVisibleRectInContainer(const RenderElement& renderer, const FloatRect& rect, const RenderLayerModelObject* container, RenderObject::VisibleRectContext context)
69{
70 // Ensure our parent is an SVG object.
71 ASSERT(renderer.parent());
72 auto& parent = *renderer.parent();
73 if (!is<SVGElement>(parent.element()))
74 return FloatRect();
75
76 FloatRect adjustedRect = rect;
77 const SVGRenderStyle& svgStyle = renderer.style().svgStyle();
78 if (const ShadowData* shadow = svgStyle.shadow())
79 shadow->adjustRectForShadow(adjustedRect);
80 adjustedRect.inflate(renderer.style().outlineWidth());
81
82 // Translate to coords in our parent renderer, and then call computeFloatVisibleRectInContainer() on our parent.
83 adjustedRect = renderer.localToParentTransform().mapRect(adjustedRect);
84
85 return parent.computeFloatVisibleRectInContainer(adjustedRect, container, context);
86}

88const RenderElement& SVGRenderSupport::localToParentTransform(const RenderElement& renderer, AffineTransform &transform)
89{
90 ASSERT(renderer.parent());
91 auto& parent = *renderer.parent();
92
93 // At the SVG/HTML boundary (aka RenderSVGRoot), we apply the localToBorderBoxTransform
94 // to map an element from SVG viewport coordinates to CSS box coordinates.
95 if (is<RenderSVGRoot>(parent))
96 transform = downcast<RenderSVGRoot>(parent).localToBorderBoxTransform() * renderer.localToParentTransform();
97 else
98 transform = renderer.localToParentTransform();
99
100 return parent;
101}

103void SVGRenderSupport::mapLocalToContainer(const RenderElement& renderer, const RenderLayerModelObject* ancestorContainer, TransformState& transformState, bool* wasFixed)
104{
105 AffineTransform transform;
106 auto& parent = localToParentTransform(renderer, transform);
107
108 transformState.applyTransform(transform);
109
110 MapCoordinatesFlags mode = UseTransforms;
111 parent.mapLocalToContainer(ancestorContainer, transformState, mode, wasFixed);
112}

114const RenderElement* SVGRenderSupport::pushMappingToContainer(const RenderElement& renderer, const RenderLayerModelObject* ancestorToStopAt, RenderGeometryMap& geometryMap)
115{
116 ASSERT_UNUSED(ancestorToStopAt, ancestorToStopAt != &renderer);

121 geometryMap.push(&renderer, transform);

125bool SVGRenderSupport::checkForSVGRepaintDuringLayout(const RenderElement& renderer)
126{
127 if (!renderer.checkForRepaintDuringLayout())
128 return false;
129 // When a parent container is transformed in SVG, all children will be painted automatically
130 // so we are able to skip redundant repaint checks.
131 auto parent = renderer.parent();
132 return !(is<RenderSVGContainer>(parent) && downcast<RenderSVGContainer>(*parent).didTransformToRootUpdate());
133}

135// Update a bounding box taking into account the validity of the other bounding box.
136static inline void updateObjectBoundingBox(FloatRect& objectBoundingBox, bool& objectBoundingBoxValid, const RenderObject* other, FloatRect otherBoundingBox)
137{
138 bool otherValid = is<RenderSVGContainer>(*other) ? downcast<RenderSVGContainer>(*other).isObjectBoundingBoxValid() : true;
139 if (!otherValid)
140 return;
141
142 if (!objectBoundingBoxValid) {
143 objectBoundingBox = otherBoundingBox;
144 objectBoundingBoxValid = true;
145 return;
146 }
147
148 objectBoundingBox.uniteEvenIfEmpty(otherBoundingBox);
149}

151void SVGRenderSupport::computeContainerBoundingBoxes(const RenderElement& container, FloatRect& objectBoundingBox, bool& objectBoundingBoxValid, FloatRect& strokeBoundingBox, FloatRect& repaintBoundingBox)
152{
153 objectBoundingBox = FloatRect();
154 objectBoundingBoxValid = false;
155 strokeBoundingBox = FloatRect();
156
157 // When computing the strokeBoundingBox, we use the repaintRects of the container's children so that the container's stroke includes
158 // the resources applied to the children (such as clips and filters). This allows filters applied to containers to correctly bound
159 // the children, and also improves inlining of SVG content, as the stroke bound is used in that situation also.
160 for (auto& current : childrenOfType<RenderObject>(container)) {
161 if (current.isSVGHiddenContainer())
162 continue;
163
164 // Don't include elements in the union that do not render.
165 if (is<RenderSVGShape>(current) && downcast<RenderSVGShape>(current).isRenderingDisabled())
166 continue;
167
168 const AffineTransform& transform = current.localToParentTransform();
169 if (transform.isIdentity()) {
170 updateObjectBoundingBox(objectBoundingBox, objectBoundingBoxValid, &current, current.objectBoundingBox());
171 strokeBoundingBox.unite(current.repaintRectInLocalCoordinates());
172 } else {
173 updateObjectBoundingBox(objectBoundingBox, objectBoundingBoxValid, &current, transform.mapRect(current.objectBoundingBox()));
174 strokeBoundingBox.unite(transform.mapRect(current.repaintRectInLocalCoordinates()));
175 }
176 }
177
178 repaintBoundingBox = strokeBoundingBox;
179}

199static inline void invalidateResourcesOfChildren(RenderElement& renderer)
200{
201 ASSERT(!renderer.needsLayout());
202 if (auto* resources = SVGResourcesCache::cachedResourcesForRenderer(renderer))
203 resources->removeClientFromCache(renderer, false);
204
205 for (auto& child : childrenOfType<RenderElement>(renderer))
206 invalidateResourcesOfChildren(child);
207}

209static inline bool layoutSizeOfNearestViewportChanged(const RenderElement& renderer)
210{
211 const RenderElement* start = &renderer;
212 while (start && !is<RenderSVGRoot>(*start) && !is<RenderSVGViewportContainer>(*start))
213 start = start->parent();
214
215 ASSERT(start);
216 if (is<RenderSVGViewportContainer>(*start))
217 return downcast<RenderSVGViewportContainer>(*start).isLayoutSizeChanged();
218
219 return downcast<RenderSVGRoot>(*start).isLayoutSizeChanged();
220}

222bool SVGRenderSupport::transformToRootChanged(RenderElement* ancestor)
223{
224 while (ancestor && !is<RenderSVGRoot>(*ancestor)) {
225 if (is<RenderSVGTransformableContainer>(*ancestor))
226 return downcast<RenderSVGTransformableContainer>(*ancestor).didTransformToRootUpdate();
227 if (is<RenderSVGViewportContainer>(*ancestor))
228 return downcast<RenderSVGViewportContainer>(*ancestor).didTransformToRootUpdate();
229 ancestor = ancestor->parent();
230 }
231
232 return false;
233}

244void SVGRenderSupport::layoutChildren(RenderElement& start, bool selfNeedsLayout)
245{
246 bool layoutSizeChanged = layoutSizeOfNearestViewportChanged(start);
247 bool transformChanged = transformToRootChanged(&start);
248 HashSet<RenderElement*> elementsThatDidNotReceiveLayout;
249
250 for (auto& child : childrenOfType<RenderObject>(start)) {
251 bool needsLayout = selfNeedsLayout;
252 bool childEverHadLayout = child.everHadLayout();
253
254 if (transformChanged) {
255 // If the transform changed we need to update the text metrics (note: this also happens for layoutSizeChanged=true).
256 if (is<RenderSVGText>(child))
257 downcast<RenderSVGText>(child).setNeedsTextMetricsUpdate();
258 needsLayout = true;
259 }
260
261 if (layoutSizeChanged && is<SVGElement>(*child.node())) {
262 // When selfNeedsLayout is false and the layout size changed, we have to check whether this child uses relative lengths
263 auto& element = downcast<SVGElement>(*child.node());
264 if (element.hasRelativeLengths()) {
265 // When the layout size changed and when using relative values tell the RenderSVGShape to update its shape object
266 if (is<RenderSVGShape>(child))
267 downcast<RenderSVGShape>(child).setNeedsShapeUpdate();
268 else if (is<RenderSVGText>(child)) {
269 auto& svgText = downcast<RenderSVGText>(child);
270 svgText.setNeedsTextMetricsUpdate();
271 svgText.setNeedsPositioningValuesUpdate();
272 }
273
274 needsLayout = true;
275 }
276 }
277
278 if (needsLayout)
279 child.setNeedsLayout(MarkOnlyThis);
280
281 if (child.needsLayout()) {
282 layoutDifferentRootIfNeeded(downcast<RenderElement>(child));
283 downcast<RenderElement>(child).layout();
284 // Renderers are responsible for repainting themselves when changing, except
285 // for the initial paint to avoid potential double-painting caused by non-sensical "old" bounds.
286 // We could handle this in the individual objects, but for now it's easier to have
287 // parent containers call repaint(). (RenderBlock::layout* has similar logic.)
288 if (!childEverHadLayout)
289 child.repaint();
290 } else if (layoutSizeChanged && is<RenderElement>(child))
291 elementsThatDidNotReceiveLayout.add(&downcast<RenderElement>(child));
292
293 ASSERT(!child.needsLayout());
294 }
295
296 if (!layoutSizeChanged) {
297 ASSERT(elementsThatDidNotReceiveLayout.isEmpty());
298 return;
299 }
300
301 // If the layout size changed, invalidate all resources of all children that didn't go through the layout() code path.
302 for (auto* element : elementsThatDidNotReceiveLayout)
303 invalidateResourcesOfChildren(*element);
304}

306bool SVGRenderSupport::isOverflowHidden(const RenderElement& renderer)
307{
308 // RenderSVGRoot should never query for overflow state - it should always clip itself to the initial viewport size.
309 ASSERT(!renderer.isDocumentElementRenderer());
310
311 return renderer.style().overflowX() == Overflow::Hidden || renderer.style().overflowX() == Overflow::Scroll;
312}

314void SVGRenderSupport::intersectRepaintRectWithResources(const RenderElement& renderer, FloatRect& repaintRect)
315{
316 auto* resources = SVGResourcesCache::cachedResourcesForRenderer(renderer);
317 if (!resources)
318 return;
319
320 if (RenderSVGResourceFilter* filter = resources->filter())
321 repaintRect = filter->resourceBoundingBox(renderer);
322
323 if (RenderSVGResourceClipper* clipper = resources->clipper())
324 repaintRect.intersect(clipper->resourceBoundingBox(renderer));
325
326 if (RenderSVGResourceMasker* masker = resources->masker())
327 repaintRect.intersect(masker->resourceBoundingBox(renderer));
328}

330bool SVGRenderSupport::filtersForceContainerLayout(const RenderElement& renderer)
331{
332 // If any of this container's children need to be laid out, and a filter is applied
333 // to the container, we need to repaint the entire container.
334 if (!renderer.normalChildNeedsLayout())
335 return false;
336
337 auto* resources = SVGResourcesCache::cachedResourcesForRenderer(renderer);
338 if (!resources || !resources->filter())
339 return false;
340
341 return true;
342}

344inline FloatRect clipPathReferenceBox(const RenderElement& renderer, CSSBoxType boxType)
345{
346 FloatRect referenceBox;
347 switch (boxType) {
348 case CSSBoxType::BorderBox:
349 case CSSBoxType::MarginBox:
350 case CSSBoxType::StrokeBox:
351 // FIXME: strokeBoundingBox() takes dasharray into account but shouldn't.
352 referenceBox = renderer.strokeBoundingBox();
353 break;
354 case CSSBoxType::ViewBox:
355 if (renderer.element()) {
356 FloatSize viewportSize;
357 SVGLengthContext(downcast<SVGElement>(renderer.element())).determineViewport(viewportSize);
358 referenceBox.setSize(viewportSize);
359 break;
360 }
361 FALLTHROUGH;
362 case CSSBoxType::ContentBox:
363 case CSSBoxType::FillBox:
364 case CSSBoxType::PaddingBox:
365 case CSSBoxType::BoxMissing:
366 referenceBox = renderer.objectBoundingBox();
367 break;
368 }
369 return referenceBox;
370}

372inline bool isPointInCSSClippingArea(const RenderElement& renderer, const FloatPoint& point)
373{
374 ClipPathOperation* clipPathOperation = renderer.style().clipPath();
375 if (is<ShapeClipPathOperation>(clipPathOperation)) {
376 auto& clipPath = downcast<ShapeClipPathOperation>(*clipPathOperation);
377 FloatRect referenceBox = clipPathReferenceBox(renderer, clipPath.referenceBox());
378 if (!referenceBox.contains(point))
379 return false;
380 return clipPath.pathForReferenceRect(referenceBox).contains(point, clipPath.windRule());
381 }
382 if (is<BoxClipPathOperation>(clipPathOperation)) {
383 auto& clipPath = downcast<BoxClipPathOperation>(*clipPathOperation);
384 FloatRect referenceBox = clipPathReferenceBox(renderer, clipPath.referenceBox());
385 if (!referenceBox.contains(point))
386 return false;
387 return clipPath.pathForReferenceRect(FloatRoundedRect {referenceBox}).contains(point);
388 }
389
390 return true;
391}

393void SVGRenderSupport::clipContextToCSSClippingArea(GraphicsContext& context, const RenderElement& renderer)
394{
395 ClipPathOperation* clipPathOperation = renderer.style().clipPath();
396 if (is<ShapeClipPathOperation>(clipPathOperation)) {
397 auto& clipPath = downcast<ShapeClipPathOperation>(*clipPathOperation);
398 auto localToParentTransform = renderer.localToParentTransform();
399
400 auto referenceBox = clipPathReferenceBox(renderer, clipPath.referenceBox());
401 referenceBox = localToParentTransform.mapRect(referenceBox);
402
403 auto path = clipPath.pathForReferenceRect(referenceBox);
404 path.transform(localToParentTransform.inverse().value_or(AffineTransform()));
405
406 context.clipPath(path, clipPath.windRule());
407 }
408 if (is<BoxClipPathOperation>(clipPathOperation)) {
409 auto& clipPath = downcast<BoxClipPathOperation>(*clipPathOperation);
410 FloatRect referenceBox = clipPathReferenceBox(renderer, clipPath.referenceBox());
411 context.clipPath(clipPath.pathForReferenceRect(FloatRoundedRect {referenceBox}));
412 }
413}

415bool SVGRenderSupport::pointInClippingArea(const RenderElement& renderer, const FloatPoint& point)
416{
417 if (SVGHitTestCycleDetectionScope::isVisiting(renderer))
418 return false;
419
420 ClipPathOperation* clipPathOperation = renderer.style().clipPath();
421 if (is<ShapeClipPathOperation>(clipPathOperation) || is<BoxClipPathOperation>(clipPathOperation))
422 return isPointInCSSClippingArea(renderer, point);
423
424 // We just take clippers into account to determine if a point is on the node. The Specification may
425 // change later and we also need to check maskers.
426 auto* resources = SVGResourcesCache::cachedResourcesForRenderer(renderer);
427 if (!resources)
428 return true;
429
430 if (RenderSVGResourceClipper* clipper = resources->clipper())
431 return clipper->hitTestClipContent(renderer.objectBoundingBox(), point);
432
433 return true;
434}

436void SVGRenderSupport::applyStrokeStyleToContext(GraphicsContext& context, const RenderStyle& style, const RenderElement& renderer)
437{
438 Element* element = renderer.element();
439 if (!is<SVGElement>(element)) {
440 ASSERT_NOT_REACHED();
441 return;
442 }
443
444 const SVGRenderStyle& svgStyle = style.svgStyle();
445
446 SVGLengthContext lengthContext(downcast<SVGElement>(renderer.element()));
447 context.setStrokeThickness(lengthContext.valueForLength(style.strokeWidth()));
448 context.setLineCap(style.capStyle());
449 context.setLineJoin(style.joinStyle());
450 if (style.joinStyle() == MiterJoin)
451 context.setMiterLimit(style.strokeMiterLimit());
452
453 const Vector<SVGLengthValue>& dashes = svgStyle.strokeDashArray();
454 if (dashes.isEmpty())
455 context.setStrokeStyle(SolidStroke);
456 else {
457 DashArray dashArray;
458 dashArray.reserveInitialCapacity(dashes.size());
459 bool canSetLineDash = false;
460 float scaleFactor = 1;
461
462 if (is<SVGGeometryElement>(element)) {
463 ASSERT(renderer.isSVGShape());
464 // FIXME: A value of zero is valid. Need to differentiate this case from being unspecified.
465 if (float pathLength = downcast<SVGGeometryElement>(element)->pathLength())
466 scaleFactor = downcast<RenderSVGShape>(renderer).getTotalLength() / pathLength;
467 }
468
469 for (auto& dash : dashes) {
470 dashArray.uncheckedAppend(dash.value(lengthContext) * scaleFactor);
471 if (dashArray.last() > 0)
472 canSetLineDash = true;
473 }
474
475 if (canSetLineDash)
476 context.setLineDash(dashArray, lengthContext.valueForLength(svgStyle.strokeDashOffset()) * scaleFactor);
477 else
478 context.setStrokeStyle(SolidStroke);
479 }
480}

500void SVGRenderSupport::updateMaskedAncestorShouldIsolateBlending(const RenderElement& renderer)
501{
502 ASSERT(renderer.element());
503 ASSERT(renderer.element()->isSVGElement());
504
505 for (auto& ancestor : ancestorsOfType<SVGGraphicsElement>(*renderer.element())) {
506 auto* style = ancestor.computedStyle();
507 if (!style || !isolatesBlending(*style))
508 continue;
509 if (style->svgStyle().hasMasker())
510 ancestor.setShouldIsolateBlending(renderer.style().hasBlendMode());
511 return;
512 }
513}
