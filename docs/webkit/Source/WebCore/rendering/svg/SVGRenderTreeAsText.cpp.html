65/** class + iomanip to help streaming list separators, i.e. ", " in string "a, b, c, d"
66 * Can be used in cases where you don't know which item in the list is the first
67 * one to be printed, but still want to avoid strings like ", b, c".
68 */
69class TextStreamSeparator {
70public:
71 TextStreamSeparator(const String& s)
72 : m_separator(s)
73 , m_needToSeparate(false)
74 {
75 }
76
77private:
78 friend TextStream& operator<<(TextStream&, TextStreamSeparator&);
79
80 String m_separator;
81 bool m_needToSeparate;
82};
83
84TextStream& operator<<(TextStream& ts, TextStreamSeparator& sep)
85{
86 if (sep.m_needToSeparate)
87 ts << sep.m_separator;
88 else
89 sep.m_needToSeparate = true;
90 return ts;
91}

93static TextStream& operator<<(TextStream& ts, const DashArray& a)
94{
95 ts << "{";
96 DashArray::const_iterator end = a.end();
97 for (DashArray::const_iterator it = a.begin(); it != end; ++it) {
98 if (it != a.begin())
99 ts << ", ";
100 ts << *it;
101 }
102 ts << "}";
103 return ts;
104}

149static void writeSVGPaintingResource(TextStream& ts, RenderSVGResource* resource)
150{
151 if (resource->resourceType() == SolidColorResourceType) {
152 ts << "[type=SOLID] [color=" << static_cast<RenderSVGResourceSolidColor*>(resource)->color() << "]";
153 return;
154 }
155
156 // All other resources derive from RenderSVGResourceContainer
157 RenderSVGResourceContainer* container = static_cast<RenderSVGResourceContainer*>(resource);
158 SVGElement& element = container->element();
159
160 if (resource->resourceType() == PatternResourceType)
161 ts << "[type=PATTERN]";
162 else if (resource->resourceType() == LinearGradientResourceType)
163 ts << "[type=LINEAR-GRADIENT]";
164 else if (resource->resourceType() == RadialGradientResourceType)
165 ts << "[type=RADIAL-GRADIENT]";
166
167 ts << " [id=\"" << element.getIdAttribute() << "\"]";
168}

170static void writeStyle(TextStream& ts, const RenderElement& renderer)
171{
172 const RenderStyle& style = renderer.style();
173 const SVGRenderStyle& svgStyle = style.svgStyle();
174
175 if (!renderer.localTransform().isIdentity())
176 writeNameValuePair(ts, "transform", renderer.localTransform());
177 writeIfNotDefault(ts, "image rendering", style.imageRendering(), RenderStyle::initialImageRendering());
178 writeIfNotDefault(ts, "opacity", style.opacity(), RenderStyle::initialOpacity());
179 if (is<RenderSVGShape>(renderer)) {
180 const auto& shape = downcast<RenderSVGShape>(renderer);
181
182 Color fallbackColor;
183 if (RenderSVGResource* strokePaintingResource = RenderSVGResource::strokePaintingResource(const_cast<RenderSVGShape&>(shape), shape.style(), fallbackColor)) {
184 TextStreamSeparator s(" ");
185 ts << " [stroke={" << s;
186 writeSVGPaintingResource(ts, strokePaintingResource);
187
188 SVGLengthContext lengthContext(&shape.graphicsElement());
189 double dashOffset = lengthContext.valueForLength(svgStyle.strokeDashOffset());
190 double strokeWidth = lengthContext.valueForLength(style.strokeWidth());
191 const auto& dashes = svgStyle.strokeDashArray();
192
193 DashArray dashArray;
194 for (auto& length : dashes)
195 dashArray.append(length.value(lengthContext));
196
197 writeIfNotDefault(ts, "opacity", svgStyle.strokeOpacity(), 1.0f);
198 writeIfNotDefault(ts, "stroke width", strokeWidth, 1.0);
199 writeIfNotDefault(ts, "miter limit", style.strokeMiterLimit(), 4.0f);
200 writeIfNotDefault(ts, "line cap", style.capStyle(), ButtCap);
201 writeIfNotDefault(ts, "line join", style.joinStyle(), MiterJoin);
202 writeIfNotDefault(ts, "dash offset", dashOffset, 0.0);
203 if (!dashArray.isEmpty())
204 writeNameValuePair(ts, "dash array", dashArray);
205
206 if (is<SVGGeometryElement>(shape.graphicsElement())) {
207 double pathLength = downcast<SVGGeometryElement>(shape.graphicsElement()).pathLength();
208 writeIfNotDefault(ts, "path length", pathLength, 0.0);
209 }
210
211 ts << "}]";
212 }
213
214 if (RenderSVGResource* fillPaintingResource = RenderSVGResource::fillPaintingResource(const_cast<RenderSVGShape&>(shape), shape.style(), fallbackColor)) {
215 TextStreamSeparator s(" ");
216 ts << " [fill={" << s;
217 writeSVGPaintingResource(ts, fillPaintingResource);
218
219 writeIfNotDefault(ts, "opacity", svgStyle.fillOpacity(), 1.0f);
220 writeIfNotDefault(ts, "fill rule", svgStyle.fillRule(), WindRule::NonZero);
221 ts << "}]";
222 }
223 writeIfNotDefault(ts, "clip rule", svgStyle.clipRule(), WindRule::NonZero);
224 }
225
226 writeIfNotEmpty(ts, "start marker", svgStyle.markerStartResource());
227 writeIfNotEmpty(ts, "middle marker", svgStyle.markerMidResource());
228 writeIfNotEmpty(ts, "end marker", svgStyle.markerEndResource());
229}

231static TextStream& writePositionAndStyle(TextStream& ts, const RenderElement& renderer, OptionSet<RenderAsTextFlag> behavior = { })
232{
233 if (behavior.contains(RenderAsTextFlag::ShowSVGGeometry)) {
234 if (is<RenderBox>(renderer)) {
235 LayoutRect r = downcast<RenderBox>(renderer).frameRect();
236 ts << " " << enclosingIntRect(r);
237 }
238
239 ts << " clipped";
240 }
241
242 ts << " " << enclosingIntRect(renderer.absoluteClippedOverflowRectForRepaint());
243
244 writeStyle(ts, renderer);
245 return ts;
246}

248static TextStream& operator<<(TextStream& ts, const RenderSVGShape& shape)
249{
250 writePositionAndStyle(ts, shape);
251
252 SVGGraphicsElement& svgElement = shape.graphicsElement();
253 SVGLengthContext lengthContext(&svgElement);
254
255 if (is<SVGRectElement>(svgElement)) {
256 const SVGRectElement& element = downcast<SVGRectElement>(svgElement);
257 writeNameValuePair(ts, "x", element.x().value(lengthContext));
258 writeNameValuePair(ts, "y", element.y().value(lengthContext));
259 writeNameValuePair(ts, "width", element.width().value(lengthContext));
260 writeNameValuePair(ts, "height", element.height().value(lengthContext));
261 } else if (is<SVGLineElement>(svgElement)) {
262 const SVGLineElement& element = downcast<SVGLineElement>(svgElement);
263 writeNameValuePair(ts, "x1", element.x1().value(lengthContext));
264 writeNameValuePair(ts, "y1", element.y1().value(lengthContext));
265 writeNameValuePair(ts, "x2", element.x2().value(lengthContext));
266 writeNameValuePair(ts, "y2", element.y2().value(lengthContext));
267 } else if (is<SVGEllipseElement>(svgElement)) {
268 const SVGEllipseElement& element = downcast<SVGEllipseElement>(svgElement);
269 writeNameValuePair(ts, "cx", element.cx().value(lengthContext));
270 writeNameValuePair(ts, "cy", element.cy().value(lengthContext));
271 writeNameValuePair(ts, "rx", element.rx().value(lengthContext));
272 writeNameValuePair(ts, "ry", element.ry().value(lengthContext));
273 } else if (is<SVGCircleElement>(svgElement)) {
274 const SVGCircleElement& element = downcast<SVGCircleElement>(svgElement);
275 writeNameValuePair(ts, "cx", element.cx().value(lengthContext));
276 writeNameValuePair(ts, "cy", element.cy().value(lengthContext));
277 writeNameValuePair(ts, "r", element.r().value(lengthContext));
278 } else if (is<SVGPolyElement>(svgElement)) {
279 const SVGPolyElement& element = downcast<SVGPolyElement>(svgElement);
280 writeNameAndQuotedValue(ts, "points", element.points().valueAsString());
281 } else if (is<SVGPathElement>(svgElement)) {
282 const SVGPathElement& element = downcast<SVGPathElement>(svgElement);
283 String pathString;
284 // FIXME: We should switch to UnalteredParsing here - this will affect the path dumping output of dozens of tests.
285 buildStringFromByteStream(element.pathByteStream(), pathString, NormalizedParsing);
286 writeNameAndQuotedValue(ts, "data", pathString);
287 } else
288 ASSERT_NOT_REACHED();
289 return ts;
290}

292static void writeRenderSVGTextBox(TextStream& ts, const RenderSVGText& text)
293{
294 auto* box = downcast<SVGRootInlineBox>(text.firstRootBox());
295 if (!box)
296 return;
297
298 ts << " " << enclosingIntRect(FloatRect(text.location(), FloatSize(box->logicalWidth(), box->logicalHeight())));
299
300 // FIXME: Remove this hack, once the new text layout engine is completly landed. We want to preserve the old layout test results for now.
301 ts << " contains 1 chunk(s)";
302
303 if (text.parent() && (text.parent()->style().visitedDependentColor(CSSPropertyColor) != text.style().visitedDependentColor(CSSPropertyColor)))
304 writeNameValuePair(ts, "color", serializationForRenderTreeAsText(text.style().visitedDependentColor(CSSPropertyColor)));
305}

307static inline void writeSVGInlineTextBox(TextStream& ts, SVGInlineTextBox* textBox)
308{
309 Vector<SVGTextFragment>& fragments = textBox->textFragments();
310 if (fragments.isEmpty())
311 return;
312
313 const SVGRenderStyle& svgStyle = textBox->renderer().style().svgStyle();
314 String text = textBox->renderer().text();
315
316 TextStream::IndentScope indentScope(ts);
317
318 unsigned fragmentsSize = fragments.size();
319 for (unsigned i = 0; i < fragmentsSize; ++i) {
320 SVGTextFragment& fragment = fragments.at(i);
321 ts << indent;
322
323 unsigned startOffset = fragment.characterOffset;
324 unsigned endOffset = fragment.characterOffset + fragment.length;
325
326 // FIXME: Remove this hack, once the new text layout engine is completly landed. We want to preserve the old layout test results for now.
327 ts << "chunk 1 ";
328 TextAnchor anchor = svgStyle.textAnchor();
329 bool isVerticalText = textBox->renderer().style().isVerticalWritingMode();
330 if (anchor == TextAnchor::Middle) {
331 ts << "(middle anchor";
332 if (isVerticalText)
333 ts << ", vertical";
334 ts << ") ";
335 } else if (anchor == TextAnchor::End) {
336 ts << "(end anchor";
337 if (isVerticalText)
338 ts << ", vertical";
339 ts << ") ";
340 } else if (isVerticalText)
341 ts << "(vertical) ";
342 startOffset -= textBox->start();
343 endOffset -= textBox->start();
344 // </hack>
345
346 ts << "text run " << i + 1 << " at (" << fragment.x << "," << fragment.y << ")";
347 ts << " startOffset " << startOffset << " endOffset " << endOffset;
348 if (isVerticalText)
349 ts << " height " << fragment.height;
350 else
351 ts << " width " << fragment.width;
352
353 if (!textBox->isLeftToRightDirection() || textBox->dirOverride()) {
354 ts << (textBox->isLeftToRightDirection() ? " LTR" : " RTL");
355 if (textBox->dirOverride())
356 ts << " override";
357 }
358
359 ts << ": " << quoteAndEscapeNonPrintables(text.substring(fragment.characterOffset, fragment.length)) << "\n";
360 }
361}

363static inline void writeSVGInlineTextBoxes(TextStream& ts, const RenderText& text)
364{
365 for (auto* box = text.firstTextBox(); box; box = box->nextTextBox()) {
366 if (!is<SVGInlineTextBox>(*box))
367 continue;
368
369 writeSVGInlineTextBox(ts, downcast<SVGInlineTextBox>(box));
370 }

378static void writeStandardPrefix(TextStream& ts, const RenderObject& object, OptionSet<RenderAsTextFlag> behavior, WriteIndentOrNot writeIndent = WriteIndentOrNot::Yes)
379{
380 if (writeIndent == WriteIndentOrNot::Yes)
381 ts << indent;
382
383 ts << object.renderName();
384
385 if (behavior.contains(RenderAsTextFlag::ShowAddresses))
386 ts << " " << &object;
387
388 if (object.node())
389 ts << " {" << object.node()->nodeName() << "}";
390
391 writeDebugInfo(ts, object, behavior);
392}

413void writeSVGResourceContainer(TextStream& ts, const RenderSVGResourceContainer& resource, OptionSet<RenderAsTextFlag> behavior)
414{
415 writeStandardPrefix(ts, resource, behavior);
416
417 const AtomString& id = resource.element().getIdAttribute();
418 writeNameAndQuotedValue(ts, "id", id);
419
420 if (resource.resourceType() == MaskerResourceType) {
421 const auto& masker = static_cast<const RenderSVGResourceMasker&>(resource);
422 writeNameValuePair(ts, "maskUnits", masker.maskUnits());
423 writeNameValuePair(ts, "maskContentUnits", masker.maskContentUnits());
424 ts << "\n";
425 } else if (resource.resourceType() == FilterResourceType) {
426 const auto& filter = static_cast<const RenderSVGResourceFilter&>(resource);
427 writeNameValuePair(ts, "filterUnits", filter.filterUnits());
428 writeNameValuePair(ts, "primitiveUnits", filter.primitiveUnits());
429 ts << "\n";
430 // Creating a placeholder filter which is passed to the builder.
431 FloatRect dummyRect;
432 auto dummyFilter = SVGFilter::create(AffineTransform(), dummyRect, dummyRect, dummyRect, true);
433 if (auto builder = filter.buildPrimitives(dummyFilter.get())) {
434 TextStream::IndentScope indentScope(ts);
435
436 if (FilterEffect* lastEffect = builder->lastEffect())
437 lastEffect->externalRepresentation(ts);
438 }
439 } else if (resource.resourceType() == ClipperResourceType) {
440 const auto& clipper = static_cast<const RenderSVGResourceClipper&>(resource);
441 writeNameValuePair(ts, "clipPathUnits", clipper.clipPathUnits());
442 ts << "\n";
443 } else if (resource.resourceType() == MarkerResourceType) {
444 const auto& marker = static_cast<const RenderSVGResourceMarker&>(resource);
445 writeNameValuePair(ts, "markerUnits", marker.markerUnits());
446 ts << " [ref at " << marker.referencePoint() << "]";
447 ts << " [angle=";
448 if (marker.angle() == -1)
449 ts << "auto" << "]\n";
450 else
451 ts << marker.angle() << "]\n";
452 } else if (resource.resourceType() == PatternResourceType) {
453 const auto& pattern = static_cast<const RenderSVGResourcePattern&>(resource);
454
455 // Dump final results that are used for rendering. No use in asking SVGPatternElement for its patternUnits(), as it may
456 // link to other patterns using xlink:href, we need to build the full inheritance chain, aka. collectPatternProperties()
457 PatternAttributes attributes;
458 pattern.collectPatternAttributes(attributes);
459
460 writeNameValuePair(ts, "patternUnits", attributes.patternUnits());
461 writeNameValuePair(ts, "patternContentUnits", attributes.patternContentUnits());
462
463 AffineTransform transform = attributes.patternTransform();
464 if (!transform.isIdentity())
465 ts << " [patternTransform=" << transform << "]";
466 ts << "\n";
467 } else if (resource.resourceType() == LinearGradientResourceType) {
468 const auto& gradient = static_cast<const RenderSVGResourceLinearGradient&>(resource);
469
470 // Dump final results that are used for rendering. No use in asking SVGGradientElement for its gradientUnits(), as it may
471 // link to other gradients using xlink:href, we need to build the full inheritance chain, aka. collectGradientProperties()
472 LinearGradientAttributes attributes;
473 gradient.linearGradientElement().collectGradientAttributes(attributes);
474 writeCommonGradientProperties(ts, attributes.spreadMethod(), attributes.gradientTransform(), attributes.gradientUnits());
475
476 ts << " [start=" << gradient.startPoint(attributes) << "] [end=" << gradient.endPoint(attributes) << "]\n";
477 } else if (resource.resourceType() == RadialGradientResourceType) {
478 const auto& gradient = static_cast<const RenderSVGResourceRadialGradient&>(resource);
479
480 // Dump final results that are used for rendering. No use in asking SVGGradientElement for its gradientUnits(), as it may
481 // link to other gradients using xlink:href, we need to build the full inheritance chain, aka. collectGradientProperties()
482 RadialGradientAttributes attributes;
483 gradient.radialGradientElement().collectGradientAttributes(attributes);
484 writeCommonGradientProperties(ts, attributes.spreadMethod(), attributes.gradientTransform(), attributes.gradientUnits());
485
486 FloatPoint focalPoint = gradient.focalPoint(attributes);
487 FloatPoint centerPoint = gradient.centerPoint(attributes);
488 float radius = gradient.radius(attributes);
489 float focalRadius = gradient.focalRadius(attributes);
490
491 ts << " [center=" << centerPoint << "] [focal=" << focalPoint << "] [radius=" << radius << "] [focalRadius=" << focalRadius << "]\n";
492 } else
493 ts << "\n";
494 writeChildren(ts, resource, behavior);
495}

556void writeResources(TextStream& ts, const RenderObject& renderer, OptionSet<RenderAsTextFlag> behavior)
557{
558 const RenderStyle& style = renderer.style();
559 const SVGRenderStyle& svgStyle = style.svgStyle();
560
561 // FIXME: We want to use SVGResourcesCache to determine which resources are present, instead of quering the resource <-> id cache.
562 // For now leave the DRT output as is, but later on we should change this so cycles are properly ignored in the DRT output.
563 if (!svgStyle.maskerResource().isEmpty()) {
564 if (RenderSVGResourceMasker* masker = getRenderSVGResourceById<RenderSVGResourceMasker>(renderer.document(), svgStyle.maskerResource())) {
565 ts << indent << " ";
566 writeNameAndQuotedValue(ts, "masker", svgStyle.maskerResource());
567 ts << " ";
568 writeStandardPrefix(ts, *masker, behavior, WriteIndentOrNot::No);
569 ts << " " << masker->resourceBoundingBox(renderer) << "\n";
570 }
571 }
572 if (style.clipPath() && is<ReferenceClipPathOperation>(style.clipPath())) {
573 auto resourceClipPath = downcast<ReferenceClipPathOperation>(style.clipPath());
574 AtomString id = resourceClipPath->fragment();
575 if (RenderSVGResourceClipper* clipper = getRenderSVGResourceById<RenderSVGResourceClipper>(renderer.document(), id)) {
576 ts << indent << " ";
577 writeNameAndQuotedValue(ts, "clipPath", resourceClipPath->fragment());
578 ts << " ";
579 writeStandardPrefix(ts, *clipper, behavior, WriteIndentOrNot::No);
580 ts << " " << clipper->resourceBoundingBox(renderer) << "\n";
581 }
582 }
583 if (style.hasFilter()) {
584 const FilterOperations& filterOperations = style.filter();
585 if (filterOperations.size() == 1) {
586 const FilterOperation& filterOperation = *filterOperations.at(0);
587 if (filterOperation.type() == FilterOperation::REFERENCE) {
588 const auto& referenceFilterOperation = downcast<ReferenceFilterOperation>(filterOperation);
589 AtomString id = SVGURIReference::fragmentIdentifierFromIRIString(referenceFilterOperation.url(), renderer.document());
590 if (RenderSVGResourceFilter* filter = getRenderSVGResourceById<RenderSVGResourceFilter>(renderer.document(), id)) {
591 ts << indent << " ";
592 writeNameAndQuotedValue(ts, "filter", id);
593 ts << " ";
594 writeStandardPrefix(ts, *filter, behavior, WriteIndentOrNot::No);
595 ts << " " << filter->resourceBoundingBox(renderer) << "\n";
596 }
597 }
598 }
599 }
