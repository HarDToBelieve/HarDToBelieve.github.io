29class RenderElement;

35class SVGResourcesCache {

37public:
38 SVGResourcesCache();
39 ~SVGResourcesCache();
40
41 static SVGResources* cachedResourcesForRenderer(const RenderElement&);

43 // Called from all SVG renderers addChild() methods.
44 static void clientWasAddedToTree(RenderObject&);
45
46 // Called from all SVG renderers removeChild() methods.
47 static void clientWillBeRemovedFromTree(RenderObject&);
48
49 // Called from all SVG renderers destroy() methods - except for RenderSVGResourceContainer.
50 static void clientDestroyed(RenderElement&);
51
52 // Called from all SVG renderers layout() methods.
53 static void clientLayoutChanged(RenderElement&);
54
55 // Called from all SVG renderers styleDidChange() methods.
56 static void clientStyleChanged(RenderElement&, StyleDifference, const RenderStyle& newStyle);
57
58 // Called from RenderSVGResourceContainer::willBeDestroyed().
59 static void resourceDestroyed(RenderSVGResourceContainer&);

64 SetStyleForScope(RenderElement&, const RenderStyle& scopedStyle, const RenderStyle& newStyle);
65 ~SetStyleForScope();
66 private:
67 void setStyle(const RenderStyle&);
68
69 RenderElement& m_renderer;
70 const RenderStyle& m_scopedStyle;

72 };

74private:
75 void addResourcesFromRenderer(RenderElement&, const RenderStyle&);
