39float SVGTextLayoutAttributes::emptyValue()
40{
41 static float s_emptyValue = std::numeric_limits<float>::max() - 1;
42 return s_emptyValue;
43}
44
45static inline void dumpSVGCharacterDataMapValue(const char* identifier, float value, bool appendSpace = true)
46{
47 if (value == SVGTextLayoutAttributes::emptyValue()) {
48 fprintf(stderr, "%s=x", identifier);
49 if (appendSpace)
50 fprintf(stderr, " ");
51 return;
52 }
53 fprintf(stderr, "%s=%lf", identifier, value);
54 if (appendSpace)
55 fprintf(stderr, " ");
56}
57
58void SVGTextLayoutAttributes::dump() const
59{
60 fprintf(stderr, "context: %p\n", &m_context);
61 const SVGCharacterDataMap::const_iterator end = m_characterDataMap.end();
62 for (SVGCharacterDataMap::const_iterator it = m_characterDataMap.begin(); it != end; ++it) {
63 const SVGCharacterData& data = it->value;
64 fprintf(stderr, " ---> pos=%i, data={", it->key);
65 dumpSVGCharacterDataMapValue("x", data.x);
66 dumpSVGCharacterDataMapValue("y", data.y);
67 dumpSVGCharacterDataMapValue("dx", data.dx);
68 dumpSVGCharacterDataMapValue("dy", data.dy);
69 dumpSVGCharacterDataMapValue("rotate", data.rotate, false);
70 fprintf(stderr, "}\n");
71 }
72}
