85static void getInlineRun(RenderObject* start, RenderObject* boundary, RenderObject*& inlineRunStart, RenderObject*& inlineRunEnd)
86{
87 // Beginning at |start| we find the largest contiguous run of inlines that
88 // we can. We denote the run with start and end points, |inlineRunStart|
89 // and |inlineRunEnd|. Note that these two values may be the same if
90 // we encounter only one inline.
91 //
92 // We skip any non-inlines we encounter as long as we haven't found any
93 // inlines yet.
94 //
95 // |boundary| indicates a non-inclusive boundary point. Regardless of whether |boundary|
96 // is inline or not, we will not include it in a run with inlines before it. It's as though we encountered
97 // a non-inline.
98
99 // Start by skipping as many non-inlines as we can.
100 auto* curr = start;
101 bool sawInline;
102 do {
103 while (curr && !(curr->isInline() || curr->isFloatingOrOutOfFlowPositioned()))
104 curr = curr->nextSibling();
105
106 inlineRunStart = inlineRunEnd = curr;
107
108 if (!curr)
109 return; // No more inline children to be found.
110
111 sawInline = curr->isInline();
112
113 curr = curr->nextSibling();
114 while (curr && (curr->isInline() || curr->isFloatingOrOutOfFlowPositioned()) && (curr != boundary)) {
115 inlineRunEnd = curr;
116 if (curr->isInline())
117 sawInline = true;
118 curr = curr->nextSibling();
119 }
120 } while (!sawInline);
121}

153void RenderTreeBuilder::destroy(RenderObject& renderer, CanCollapseAnonymousBlock canCollapseAnonymousBlock)
154{
155 RELEASE_ASSERT(RenderTreeMutationDisallowedScope::isMutationAllowed());
156 ASSERT(renderer.parent());
157 auto toDestroy = detach(*renderer.parent(), renderer, canCollapseAnonymousBlock);
158
159#if ENABLE(FULLSCREEN_API)
160 if (is<RenderFullScreen>(renderer))
161 fullScreenBuilder().cleanupOnDestroy(downcast<RenderFullScreen>(renderer));
162#endif
163
164 if (is<RenderTextFragment>(renderer))
165 firstLetterBuilder().cleanupOnDestroy(downcast<RenderTextFragment>(renderer));
166
167 if (is<RenderBoxModelObject>(renderer))
168 continuationBuilder().cleanupOnDestroy(downcast<RenderBoxModelObject>(renderer));
169
170 // We need to detach the subtree first so that the descendants don't have
171 // access to previous/next sublings at detach().
172 // FIXME: webkit.org/b/182909.
173 if (!is<RenderElement>(toDestroy.get()))
174 return;
175
176 auto& childToDestroy = downcast<RenderElement>(*toDestroy.get());
177 while (childToDestroy.firstChild()) {
178 auto& firstChild = *childToDestroy.firstChild();
179 if (auto* node = firstChild.node())
180 node->setRenderer(nullptr);
181 destroy(firstChild);
182 }
183}

191void RenderTreeBuilder::attachInternal(RenderElement& parent, RenderPtr<RenderObject> child, RenderObject* beforeChild)
192{
193 auto insertRecursiveIfNeeded = [&](RenderElement& parentCandidate) {
194 if (&parent == &parentCandidate) {
195 // Parents inside multicols can't call internal attach directly.
196 if (is<RenderBlockFlow>(parent) && downcast<RenderBlockFlow>(parent).multiColumnFlow()) {
197 blockFlowBuilder().attach(downcast<RenderBlockFlow>(parent), WTFMove(child), beforeChild);
198 return;
199 }
200 attachToRenderElement(parent, WTFMove(child), beforeChild);
201 return;
202 }
203 attachInternal(parentCandidate, WTFMove(child), beforeChild);
204 };
205
206 ASSERT(&parent.view() == &m_view);
207
208 if (is<RenderText>(beforeChild)) {
209 if (auto* wrapperInline = downcast<RenderText>(*beforeChild).inlineWrapperForDisplayContents())
210 beforeChild = wrapperInline;
211 } else if (is<RenderBox>(beforeChild)) {
212 // Adjust the beforeChild if it happens to be a spanner and the its actual location is inside the fragmented flow.
213 auto& beforeChildBox = downcast<RenderBox>(*beforeChild);
214 if (auto* enclosingFragmentedFlow = parent.enclosingFragmentedFlow()) {
215 auto columnSpannerPlaceholderForBeforeChild = [&]() -> RenderMultiColumnSpannerPlaceholder* {
216 if (!is<RenderMultiColumnFlow>(enclosingFragmentedFlow))
217 return nullptr;
218 auto& multiColumnFlow = downcast<RenderMultiColumnFlow>(*enclosingFragmentedFlow);
219 return multiColumnFlow.findColumnSpannerPlaceholder(&beforeChildBox);
220 };
221
222 if (auto* spannerPlaceholder = columnSpannerPlaceholderForBeforeChild())
223 beforeChild = spannerPlaceholder;
224 }
225 }
226
227 if (is<RenderTableRow>(parent)) {
228 auto& parentCandidate = tableBuilder().findOrCreateParentForChild(downcast<RenderTableRow>(parent), *child, beforeChild);
229 if (&parentCandidate == &parent) {
230 tableBuilder().attach(downcast<RenderTableRow>(parentCandidate), WTFMove(child), beforeChild);
231 return;
232 }
233 insertRecursiveIfNeeded(parentCandidate);
234 return;
235 }
236
237 if (is<RenderTableSection>(parent)) {
238 auto& parentCandidate = tableBuilder().findOrCreateParentForChild(downcast<RenderTableSection>(parent), *child, beforeChild);
239 if (&parent == &parentCandidate) {
240 tableBuilder().attach(downcast<RenderTableSection>(parent), WTFMove(child), beforeChild);
241 return;
242 }
243 insertRecursiveIfNeeded(parentCandidate);
244 return;
245 }
246
247 if (is<RenderTable>(parent)) {
248 auto& parentCandidate = tableBuilder().findOrCreateParentForChild(downcast<RenderTable>(parent), *child, beforeChild);
249 if (&parentCandidate == &parent) {
250 tableBuilder().attach(downcast<RenderTable>(parentCandidate), WTFMove(child), beforeChild);
251 return;
252 }
253 insertRecursiveIfNeeded(parentCandidate);
254 return;
255 }
256
257 if (is<RenderRubyAsBlock>(parent)) {
258 insertRecursiveIfNeeded(rubyBuilder().findOrCreateParentForChild(downcast<RenderRubyAsBlock>(parent), *child, beforeChild));
259 return;
260 }
261
262 if (is<RenderRubyAsInline>(parent)) {
263 insertRecursiveIfNeeded(rubyBuilder().findOrCreateParentForChild(downcast<RenderRubyAsInline>(parent), *child, beforeChild));
264 return;
265 }
266
267 if (is<RenderRubyRun>(parent)) {
268 rubyBuilder().attach(downcast<RenderRubyRun>(parent), WTFMove(child), beforeChild);
269 return;
270 }
271
272 if (is<RenderButton>(parent)) {
273 formControlsBuilder().attach(downcast<RenderButton>(parent), WTFMove(child), beforeChild);
274 return;
275 }
276
277 if (is<RenderMenuList>(parent)) {
278 formControlsBuilder().attach(downcast<RenderMenuList>(parent), WTFMove(child), beforeChild);
279 return;
280 }
281
282 if (is<RenderSVGContainer>(parent)) {
283 svgBuilder().attach(downcast<RenderSVGContainer>(parent), WTFMove(child), beforeChild);
284 return;
285 }
286
287 if (is<RenderSVGInline>(parent)) {
288 svgBuilder().attach(downcast<RenderSVGInline>(parent), WTFMove(child), beforeChild);
289 return;
290 }
291
292 if (is<RenderSVGRoot>(parent)) {
293 svgBuilder().attach(downcast<RenderSVGRoot>(parent), WTFMove(child), beforeChild);
294 return;
295 }
296
297 if (is<RenderSVGText>(parent)) {
298 svgBuilder().attach(downcast<RenderSVGText>(parent), WTFMove(child), beforeChild);
299 return;
300 }
301
302#if ENABLE(MATHML)
303 if (is<RenderMathMLFenced>(parent)) {
304 mathMLBuilder().attach(downcast<RenderMathMLFenced>(parent), WTFMove(child), beforeChild);
305 return;
306 }
307#endif
308
309 if (is<RenderGrid>(parent)) {
310 attachToRenderGrid(downcast<RenderGrid>(parent), WTFMove(child), beforeChild);
311 return;
312 }
313
314 if (is<RenderBlockFlow>(parent)) {
315 blockFlowBuilder().attach(downcast<RenderBlockFlow>(parent), WTFMove(child), beforeChild);
316 return;
317 }
318
319 if (is<RenderBlock>(parent)) {
320 blockBuilder().attach(downcast<RenderBlock>(parent), WTFMove(child), beforeChild);
321 return;
322 }
323
324 if (is<RenderInline>(parent)) {
325 inlineBuilder().attach(downcast<RenderInline>(parent), WTFMove(child), beforeChild);
326 return;
327 }
328
329 attachToRenderElement(parent, WTFMove(child), beforeChild);
330}

395void RenderTreeBuilder::attachToRenderElement(RenderElement& parent, RenderPtr<RenderObject> child, RenderObject* beforeChild)
396{
397 if (tableBuilder().childRequiresTable(parent, *child)) {
398 RenderTable* table;
399 RenderObject* afterChild = beforeChild ? beforeChild->previousSibling() : parent.lastChild();
400 if (afterChild && afterChild->isAnonymous() && is<RenderTable>(*afterChild) && !afterChild->isBeforeContent())
401 table = downcast<RenderTable>(afterChild);
402 else {
403 auto newTable = RenderTable::createAnonymousWithParentRenderer(parent);
404 table = newTable.get();
405 attach(parent, WTFMove(newTable), beforeChild);
406 }
407
408 attach(*table, WTFMove(child));
409 return;
410 }
411 auto& newChild = *child.get();
412 attachToRenderElementInternal(parent, WTFMove(child), beforeChild);
413 parent.didAttachChild(newChild, beforeChild);
414}

416void RenderTreeBuilder::attachToRenderElementInternal(RenderElement& parent, RenderPtr<RenderObject> child, RenderObject* beforeChild, RenderObject::IsInternalMove isInternalMove)
417{
418 RELEASE_ASSERT_WITH_MESSAGE(!parent.view().frameView().layoutContext().layoutState(), "Layout must not mutate render tree");
419 ASSERT(parent.canHaveChildren() || parent.canHaveGeneratedChildren());
420 ASSERT(!child->parent());
421 ASSERT(!parent.isRenderBlockFlow() || (!child->isTableSection() && !child->isTableRow() && !child->isTableCell()));
422
423 while (beforeChild && beforeChild->parent() && beforeChild->parent() != &parent)
424 beforeChild = beforeChild->parent();
425
426 ASSERT(!beforeChild || beforeChild->parent() == &parent);
427 ASSERT(!is<RenderText>(beforeChild) || !downcast<RenderText>(*beforeChild).inlineWrapperForDisplayContents());
428
429 // Take the ownership.
430 auto* newChild = parent.attachRendererInternal(WTFMove(child), beforeChild);
431
432 newChild->initializeFragmentedFlowStateOnInsertion();
433 if (!parent.renderTreeBeingDestroyed()) {
434 newChild->insertedIntoTree(isInternalMove);
435
436 auto needsStateReset = isInternalMove == RenderObject::IsInternalMove::No;
437 if (needsStateReset) {
438 if (auto* fragmentedFlow = newChild->enclosingFragmentedFlow(); is<RenderMultiColumnFlow>(fragmentedFlow))
439 multiColumnBuilder().multiColumnDescendantInserted(downcast<RenderMultiColumnFlow>(*fragmentedFlow), *newChild);
440
441 if (is<RenderElement>(*newChild))
442 RenderCounter::rendererSubtreeAttached(downcast<RenderElement>(*newChild));
443 }
444 }
445
446 newChild->setNeedsLayoutAndPrefWidthsRecalc();
447 parent.setPreferredLogicalWidthsDirty(true);
448 if (!parent.normalChildNeedsLayout())
449 parent.setChildNeedsLayout(); // We may supply the static position for an absolute positioned child.
450
451 if (AXObjectCache* cache = parent.document().axObjectCache())
452 cache->childrenChanged(&parent, newChild);
453
454 if (parent.hasOutlineAutoAncestor() || parent.outlineStyleForRepaint().outlineStyleIsAuto() == OutlineIsAuto::On)
455 newChild->setHasOutlineAutoAncestor();
456}

458void RenderTreeBuilder::move(RenderBoxModelObject& from, RenderBoxModelObject& to, RenderObject& child, RenderObject* beforeChild, NormalizeAfterInsertion normalizeAfterInsertion)
459{
460 // We assume that callers have cleared their positioned objects list for child moves so the
461 // positioned renderer maps don't become stale. It would be too slow to do the map lookup on each call.
462 ASSERT(normalizeAfterInsertion == NormalizeAfterInsertion::No || !is<RenderBlock>(from) || !downcast<RenderBlock>(from).hasPositionedObjects());
463
464 ASSERT(&from == child.parent());
465 ASSERT(!beforeChild || &to == beforeChild->parent());
466 if (normalizeAfterInsertion == NormalizeAfterInsertion::Yes && (to.isRenderBlock() || to.isRenderInline())) {
467 // Takes care of adding the new child correctly if toBlock and fromBlock
468 // have different kind of children (block vs inline).
469 auto childToMove = detachFromRenderElement(from, child);
470 attach(to, WTFMove(childToMove), beforeChild);
471 } else {
472 auto childToMove = detachFromRenderElement(from, child, WillBeDestroyed::No, RenderObject::IsInternalMove::Yes);
473 attachToRenderElementInternal(to, WTFMove(childToMove), beforeChild, RenderObject::IsInternalMove::Yes);
474 }
475
476 auto findBFCRootAndDestroyInlineTree = [&] {
477 auto* containingBlock = &from;
478 while (containingBlock) {
479 containingBlock->setNeedsLayout();
480 if (is<RenderBlockFlow>(*containingBlock)) {
481 downcast<RenderBlockFlow>(*containingBlock).deleteLines();
482 break;
483 }
484 containingBlock = containingBlock->containingBlock();
485 }
486 };
487 // When moving a subtree out of a BFC we need to make sure that the line boxes generated for the inline tree are not accessible anymore from the renderers.
488 // Let's find the BFC root and nuke the inline tree (At some point we are going to destroy the subtree instead of moving these renderers around.)
489 if (is<RenderInline>(child))
490 findBFCRootAndDestroyInlineTree();
491}

513void RenderTreeBuilder::moveChildren(RenderBoxModelObject& from, RenderBoxModelObject& to, RenderObject* startChild, RenderObject* endChild, RenderObject* beforeChild, NormalizeAfterInsertion normalizeAfterInsertion)
514{
515 // This condition is rarely hit since this function is usually called on
516 // anonymous blocks which can no longer carry positioned objects (see r120761)
517 // or when fullRemoveInsert is false.
518 if (normalizeAfterInsertion == NormalizeAfterInsertion::Yes && is<RenderBlock>(from)) {
519 auto& blockFlow = downcast<RenderBlock>(from);
520 blockFlow.removePositionedObjects(nullptr);
521 removeFloatingObjects(blockFlow);
522 }
523
524 ASSERT(!beforeChild || &to == beforeChild->parent());
525 for (RenderObject* child = startChild; child && child != endChild; ) {
526 // Save our next sibling as moveChildTo will clear it.
527 RenderObject* nextSibling = child->nextSibling();
528
529 // FIXME: This logic here fails to detect the first letter in certain cases
530 // and skips a valid sibling renderer (see webkit.org/b/163737).
531 // Check to make sure we're not saving the firstLetter as the nextSibling.
532 // When the |child| object will be moved, its firstLetter will be recreated,
533 // so saving it now in nextSibling would leave us with a stale object.
534 if (is<RenderTextFragment>(*child) && is<RenderText>(nextSibling)) {
535 RenderObject* firstLetterObj = nullptr;
536 if (RenderBlock* block = downcast<RenderTextFragment>(*child).blockForAccompanyingFirstLetter()) {
537 RenderElement* firstLetterContainer = nullptr;
538 block->getFirstLetter(firstLetterObj, firstLetterContainer, child);
539 }
540
541 // This is the first letter, skip it.
542 if (firstLetterObj == nextSibling)
543 nextSibling = nextSibling->nextSibling();
544 }
545
546 move(from, to, *child, beforeChild, normalizeAfterInsertion);
547 child = nextSibling;
548 }
549}

560void RenderTreeBuilder::normalizeTreeAfterStyleChange(RenderElement& renderer, RenderStyle& oldStyle)
561{
562 if (!renderer.parent())
563 return;
564
565 auto& parent = *renderer.parent();
566
567 bool wasFloating = oldStyle.isFloating();
568 bool wasOutOfFlowPositioned = oldStyle.hasOutOfFlowPosition();
569 bool isFloating = renderer.style().isFloating();
570 bool isOutOfFlowPositioned = renderer.style().hasOutOfFlowPosition();
571 bool startsAffectingParent = false;
572 bool noLongerAffectsParent = false;
573
574 if (is<RenderBlock>(parent))
575 noLongerAffectsParent = (!wasFloating && isFloating) || (!wasOutOfFlowPositioned && isOutOfFlowPositioned);
576
577 if (is<RenderBlockFlow>(parent) || is<RenderInline>(parent)) {
578 startsAffectingParent = (wasFloating || wasOutOfFlowPositioned) && !isFloating && !isOutOfFlowPositioned;
579 ASSERT(!startsAffectingParent || !noLongerAffectsParent);
580 }
581
582 if (startsAffectingParent) {
583 // We have gone from not affecting the inline status of the parent flow to suddenly
584 // having an impact. See if there is a mismatch between the parent flow's
585 // childrenInline() state and our state.
586 if (renderer.isInline() != renderer.parent()->childrenInline())
587 childFlowStateChangesAndAffectsParentBlock(renderer);
588 return;
589 }
590
591 if (noLongerAffectsParent) {
592 childFlowStateChangesAndNoLongerAffectsParentBlock(renderer);
593
594 if (is<RenderBlockFlow>(renderer)) {
595 // Fresh floats need to be reparented if they actually belong to the previous anonymous block.
596 // It copies the logic of RenderBlock::addChildIgnoringContinuation
597 if (isFloating && renderer.previousSibling() && renderer.previousSibling()->isAnonymousBlock())
598 move(downcast<RenderBoxModelObject>(parent), downcast<RenderBoxModelObject>(*renderer.previousSibling()), renderer, RenderTreeBuilder::NormalizeAfterInsertion::No);
599 }
600 }
601
602 // Out of flow children of RenderMultiColumnFlow are not really part of the multicolumn flow. We need to ensure that changes in positioning like this
603 // trigger insertions into the multicolumn flow.
604 if (auto* enclosingFragmentedFlow = parent.enclosingFragmentedFlow(); is<RenderMultiColumnFlow>(enclosingFragmentedFlow)) {
605 auto movingIntoMulticolumn = [&] {
606 if (wasOutOfFlowPositioned && !isOutOfFlowPositioned)
607 return true;
608 if (auto* containingBlock = renderer.containingBlock(); containingBlock && isOutOfFlowPositioned) {
609 // Sometimes the flow state could change even when the renderer stays out-of-flow (e.g when going from fixed to absolute and
610 // the containing block is inside a multi-column flow).
611 return containingBlock->fragmentedFlowState() == RenderObject::InsideInFragmentedFlow
612 && renderer.fragmentedFlowState() == RenderObject::NotInsideFragmentedFlow;
613 }
614 return false;
615 }();
616 if (movingIntoMulticolumn) {
617 renderer.initializeFragmentedFlowStateOnInsertion();
618 multiColumnBuilder().multiColumnDescendantInserted(downcast<RenderMultiColumnFlow>(*enclosingFragmentedFlow), renderer);
619 return;
620 }
621 auto movingOutOfMulticolumn = !wasOutOfFlowPositioned && isOutOfFlowPositioned;
622 if (movingOutOfMulticolumn) {
623 multiColumnBuilder().restoreColumnSpannersForContainer(renderer, downcast<RenderMultiColumnFlow>(*enclosingFragmentedFlow));
624 return;
625 }
626 }

629void RenderTreeBuilder::makeChildrenNonInline(RenderBlock& parent, RenderObject* insertionPoint)
630{
631 // makeChildrenNonInline takes a block whose children are *all* inline and it
632 // makes sure that inline children are coalesced under anonymous
633 // blocks. If |insertionPoint| is defined, then it represents the insertion point for
634 // the new block child that is causing us to have to wrap all the inlines. This
635 // means that we cannot coalesce inlines before |insertionPoint| with inlines following
636 // |insertionPoint|, because the new child is going to be inserted in between the inlines,
637 // splitting them.
638 ASSERT(parent.isInlineBlockOrInlineTable() || !parent.isInline());
639 ASSERT(!insertionPoint || insertionPoint->parent() == &parent);
640
641 parent.setChildrenInline(false);
642
643 auto* child = parent.firstChild();
644 if (!child)
645 return;
646
647 parent.deleteLines();
648
649 while (child) {
650 RenderObject* inlineRunStart = nullptr;
651 RenderObject* inlineRunEnd = nullptr;
652 getInlineRun(child, insertionPoint, inlineRunStart, inlineRunEnd);
653
654 if (!inlineRunStart)
655 break;
656
657 child = inlineRunEnd->nextSibling();
658
659 auto newBlock = parent.createAnonymousBlock();
660 auto& block = *newBlock;
661 attachToRenderElementInternal(parent, WTFMove(newBlock), inlineRunStart);
662 moveChildren(parent, block, inlineRunStart, child, RenderTreeBuilder::NormalizeAfterInsertion::No);
663 }
664#ifndef NDEBUG
665 for (RenderObject* c = parent.firstChild(); c; c = c->nextSibling())
666 ASSERT(!c->isInline());
667#endif
668 parent.repaint();
669}

671RenderObject* RenderTreeBuilder::splitAnonymousBoxesAroundChild(RenderBox& parent, RenderObject& originalBeforeChild)
672{
673 // Adjust beforeChild if it is a column spanner and has been moved out of its original position.
674 auto* beforeChild = RenderTreeBuilder::MultiColumn::adjustBeforeChildForMultiColumnSpannerIfNeeded(originalBeforeChild);
675 bool didSplitParentAnonymousBoxes = false;
676
677 while (beforeChild->parent() != &parent) {
678 auto& boxToSplit = downcast<RenderBox>(*beforeChild->parent());
679 if (boxToSplit.firstChild() != beforeChild && boxToSplit.isAnonymous()) {
680 didSplitParentAnonymousBoxes = true;
681
682 // We have to split the parent box into two boxes and move children
683 // from |beforeChild| to end into the new post box.
684 auto newPostBox = boxToSplit.createAnonymousBoxWithSameTypeAs(parent);
685 auto& postBox = *newPostBox;
686 postBox.setChildrenInline(boxToSplit.childrenInline());
687 RenderBox* parentBox = downcast<RenderBox>(boxToSplit.parent());
688 // We need to invalidate the |parentBox| before inserting the new node
689 // so that the table repainting logic knows the structure is dirty.
690 // See for example RenderTableCell:clippedOverflowRectForRepaint.
691 markBoxForRelayoutAfterSplit(*parentBox);
692 attachToRenderElementInternal(*parentBox, WTFMove(newPostBox), boxToSplit.nextSibling());
693 moveChildren(boxToSplit, postBox, beforeChild, nullptr, RenderTreeBuilder::NormalizeAfterInsertion::Yes);
694
695 markBoxForRelayoutAfterSplit(boxToSplit);
696 markBoxForRelayoutAfterSplit(postBox);
697
698 beforeChild = &postBox;
699 } else
700 beforeChild = &boxToSplit;
701 }
702
703 if (didSplitParentAnonymousBoxes)
704 markBoxForRelayoutAfterSplit(parent);
705
706 ASSERT(beforeChild->parent() == &parent);
707 return beforeChild;
708}

710void RenderTreeBuilder::childFlowStateChangesAndAffectsParentBlock(RenderElement& child)
711{
712 if (!child.isInline()) {
713 auto* currentEnclosingFragment = child.enclosingFragmentedFlow();
714 auto parent = makeWeakPtr(child.parent());
715 if (is<RenderBlock>(*parent))
716 blockBuilder().childBecameNonInline(downcast<RenderBlock>(*parent), child);
717 else if (is<RenderInline>(*parent))
718 inlineBuilder().childBecameNonInline(downcast<RenderInline>(*parent), child);
719 // WARNING: original parent might be deleted at this point.
720
721 if (auto* newParent = child.parent()) {
722 // childBecameNonInline might have re-parented us.
723 if (newParent != parent && is<RenderGrid>(*newParent)) {
724 // We need to re-run the grid items placement if it had gained a new item.
725 downcast<RenderGrid>(*newParent).dirtyGrid();
726 }
727
728 auto newMultiColumnForChildRenderer = [&]() -> RenderMultiColumnFlow* {
729 // Update the state when the child has moved from one multi-column flow to another.
730 auto* newEnclosingFragmentedFlow = newParent->enclosingFragmentedFlow();
731 if (newParent->isMultiColumnBlockFlow()) {
732 // This child is a spanner so it is not in the subtree of the multicolumn renderer. It is parented directly under the block flow so
733 // enclosingFragmentedFlow() returns the parent enclosing flow.
734 ASSERT(is<RenderBox>(child) && downcast<RenderBlockFlow>(*newParent).multiColumnFlow()->spannerMap().contains(&downcast<RenderBox>(child)));
735 newEnclosingFragmentedFlow = downcast<RenderBlockFlow>(*newParent).multiColumnFlow();
736 }
737 return newEnclosingFragmentedFlow != currentEnclosingFragment && is<RenderMultiColumnFlow>(newEnclosingFragmentedFlow) ? downcast<RenderMultiColumnFlow>(newEnclosingFragmentedFlow) : nullptr;
738 };
739 if (auto* newEnclosingMultiColumn = newMultiColumnForChildRenderer()) {
740 // Let the fragmented flow know that it has a new in-flow descendant.
741 child.initializeFragmentedFlowStateOnInsertion();
742 multiColumnBuilder().multiColumnDescendantInserted(*newEnclosingMultiColumn, child);
743 }
744 }
745 } else {
746 // An anonymous block must be made to wrap this inline.
747 auto* parent = child.parent();
748 auto newBlock = downcast<RenderBlock>(*parent).createAnonymousBlock();
749 auto& block = *newBlock;
750 attachToRenderElementInternal(*parent, WTFMove(newBlock), &child);
751 auto thisToMove = detachFromRenderElement(*parent, child);
752 attachToRenderElementInternal(block, WTFMove(thisToMove));
753 }
754}

756void RenderTreeBuilder::removeAnonymousWrappersForInlineChildrenIfNeeded(RenderElement& parent)
757{
758 if (!is<RenderBlock>(parent))
759 return;
760 auto& blockParent = downcast<RenderBlock>(parent);
761 if (!blockParent.canDropAnonymousBlockChild())
762 return;
763
764 // We have changed to floated or out-of-flow positioning so maybe all our parent's
765 // children can be inline now. Bail if there are any block children left on the line,
766 // otherwise we can proceed to stripping solitary anonymous wrappers from the inlines.
767 // FIXME: We should also handle split inlines here - we exclude them at the moment by returning
768 // if we find a continuation.
769 std::optional<bool> shouldAllChildrenBeInline;
770 for (auto* current = blockParent.firstChild(); current; current = current->nextSibling()) {
771 if (current->style().isFloating() || current->style().hasOutOfFlowPosition())
772 continue;
773 if (!current->isAnonymousBlock() || downcast<RenderBlock>(*current).isContinuation())
774 return;
775 // Anonymous block not in continuation. Check if it holds a set of inline or block children and try not to mix them.
776 auto* firstChild = current->firstChildSlow();
777 if (!firstChild)
778 continue;
779 auto isInlineLevelBox = firstChild->isInline();
780 if (!shouldAllChildrenBeInline.has_value()) {
781 shouldAllChildrenBeInline = isInlineLevelBox;
782 continue;
783 }
784 // Mixing inline and block level boxes?
785 if (*shouldAllChildrenBeInline != isInlineLevelBox)
786 return;
787 }
788
789 RenderObject* next = nullptr;
790 for (auto* current = blockParent.firstChild(); current; current = next) {
791 next = current->nextSibling();
792 if (current->isAnonymousBlock())
793 blockBuilder().dropAnonymousBoxChild(blockParent, downcast<RenderBlock>(*current));
794 }
795}

803void RenderTreeBuilder::destroyAndCleanUpAnonymousWrappers(RenderObject& rendererToDestroy)
804{
805 // If the tree is destroyed, there is no need for a clean-up phase.
806 if (rendererToDestroy.renderTreeBeingDestroyed()) {
807 destroy(rendererToDestroy);
808 return;
809 }
810
811 auto isAnonymousAndSafeToDelete = [] (const auto& renderer) {
812 return renderer.isAnonymous() && !renderer.isRenderView() && !renderer.isRenderFragmentedFlow();
813 };
814
815 auto destroyRootIncludingAnonymous = [&] () -> RenderObject& {
816 auto* destroyRoot = &rendererToDestroy;
817 while (!is<RenderView>(*destroyRoot)) {
818 auto& destroyRootParent = *destroyRoot->parent();
819 if (!isAnonymousAndSafeToDelete(destroyRootParent))
820 break;
821 bool destroyingOnlyChild = destroyRootParent.firstChild() == destroyRoot && destroyRootParent.lastChild() == destroyRoot;
822 if (!destroyingOnlyChild)
823 break;
824 destroyRoot = &destroyRootParent;
825 }
826 return *destroyRoot;
827 };
828
829 auto& destroyRoot = destroyRootIncludingAnonymous();
830
831 auto clearFloatsAndOutOfFlowPositionedObjects = [&] {
832 // Remove floats and out-of-flow positioned objects from their containing block before detaching
833 // the renderer from the tree. It includes all the anonymous block descendants that we are about
834 // to destroy as well as part of the cleanup process below.
835 if (!is<RenderElement>(destroyRoot))
836 return;
837 for (auto& descendant : descendantsOfType<RenderBox>(downcast<RenderElement>(destroyRoot))) {
838 if (descendant.isFloatingOrOutOfFlowPositioned())
839 descendant.removeFloatingOrPositionedChildFromBlockLists();
840 }
841 if (is<RenderBox>(destroyRoot) && destroyRoot.isFloatingOrOutOfFlowPositioned())
842 downcast<RenderBox>(destroyRoot).removeFloatingOrPositionedChildFromBlockLists();
843 };
844 clearFloatsAndOutOfFlowPositionedObjects();
845
846 auto collapseAndDestroyAnonymousSiblings = [&] {
847 // FIXME: Probably need to handle other table parts here as well.
848 if (is<RenderTableCell>(destroyRoot)) {
849 tableBuilder().collapseAndDestroyAnonymousSiblingCells(downcast<RenderTableCell>(destroyRoot));
850 return;
851 }
852
853 if (is<RenderTableRow>(destroyRoot)) {
854 tableBuilder().collapseAndDestroyAnonymousSiblingRows(downcast<RenderTableRow>(destroyRoot));
855 return;
856 }
857 };
858 collapseAndDestroyAnonymousSiblings();
859
860 // FIXME: Do not try to collapse/cleanup the anonymous wrappers inside destroy (see webkit.org/b/186746).
861 auto destroyRootParent = makeWeakPtr(*destroyRoot.parent());
862 if (&rendererToDestroy != &destroyRoot) {
863 // Destroy the child renderer first, before we start tearing down the anonymous wrapper ancestor chain.
864 destroy(rendererToDestroy);
865 }
866 destroy(destroyRoot);
867 if (!destroyRootParent)
868 return;
869 removeAnonymousWrappersForInlineChildrenIfNeeded(*destroyRootParent);
870
871 // Anonymous parent might have become empty, try to delete it too.
872 if (isAnonymousAndSafeToDelete(*destroyRootParent) && !destroyRootParent->firstChild())
873 destroyAndCleanUpAnonymousWrappers(*destroyRootParent);
874 // WARNING: rendererToDestroy is deleted here.
875}

900RenderPtr<RenderObject> RenderTreeBuilder::detachFromRenderElement(RenderElement& parent, RenderObject& child, WillBeDestroyed willBeDestroyed, RenderObject::IsInternalMove isInternalMove)
901{
902 RELEASE_ASSERT_WITH_MESSAGE(!parent.view().frameView().layoutContext().layoutState(), "Layout must not mutate render tree");
903
904 ASSERT(parent.canHaveChildren() || parent.canHaveGeneratedChildren());
905 ASSERT(child.parent() == &parent);
906
907 if (child.isFloatingOrOutOfFlowPositioned())
908 downcast<RenderBox>(child).removeFloatingOrPositionedChildFromBlockLists();
909
910 // So that we'll get the appropriate dirty bit set (either that a normal flow child got yanked or
911 // that a positioned child got yanked). We also repaint, so that the area exposed when the child
912 // disappears gets repainted properly.
913 if (!parent.renderTreeBeingDestroyed() && child.everHadLayout()) {
914 child.setNeedsLayoutAndPrefWidthsRecalc();
915 // We only repaint |child| if we have a RenderLayer as its visual overflow may not be tracked by its parent.
916 if (child.isBody())
917 parent.view().repaintRootContents();
918 else
919 child.repaint();
920 }
921
922 // If we have a line box wrapper, delete it.
923 if (is<RenderBox>(child))
924 downcast<RenderBox>(child).deleteLineBoxWrapper();
925 else if (is<RenderLineBreak>(child))
926 downcast<RenderLineBreak>(child).deleteInlineBoxWrapper();
927 else if (is<RenderText>(child))
928 downcast<RenderText>(child).removeAndDestroyTextBoxes();
929
930 if (!parent.renderTreeBeingDestroyed() && is<RenderFlexibleBox>(parent) && !child.isFloatingOrOutOfFlowPositioned() && child.isBox())
931 downcast<RenderFlexibleBox>(parent).clearCachedChildIntrinsicContentLogicalHeight(downcast<RenderBox>(child));
932
933 // If child is the start or end of the selection, then clear the selection to
934 // avoid problems of invalid pointers.
935 if (!parent.renderTreeBeingDestroyed() && willBeDestroyed == WillBeDestroyed::Yes && child.isSelectionBorder())
936 parent.frame().selection().setNeedsSelectionUpdate();
937
938 child.resetFragmentedFlowStateOnRemoval();
939
940 if (!parent.renderTreeBeingDestroyed())
941 child.willBeRemovedFromTree(isInternalMove);
942
943 // WARNING: There should be no code running between willBeRemovedFromTree() and the actual removal below.
944 // This is needed to avoid race conditions where willBeRemovedFromTree() would dirty the tree's structure
945 // and the code running here would force an untimely rebuilding, leaving |child| dangling.
946 auto childToTake = parent.detachRendererInternal(child);
947
948 // rendererRemovedFromTree() walks the whole subtree. We can improve performance
949 // by skipping this step when destroying the entire tree.
950 if (!parent.renderTreeBeingDestroyed() && is<RenderElement>(*childToTake))
951 RenderCounter::rendererRemovedFromTree(downcast<RenderElement>(*childToTake));
952
953 if (!parent.renderTreeBeingDestroyed()) {
954 if (AXObjectCache* cache = parent.document().existingAXObjectCache())
955 cache->childrenChanged(&parent);
956 }
957 return childToTake;
958}

975void RenderTreeBuilder::reportVisuallyNonEmptyContent(const RenderElement& parent, const RenderObject& child)
976{
977 if (is<RenderText>(child)) {
978 auto& style = parent.style();
979 // FIXME: Find out how to increment the visually non empty character count when the font becomes available.
980 if (style.visibility() == Visibility::Visible && !style.fontCascade().isLoadingCustomFonts()) {
981 auto& textRenderer = downcast<RenderText>(child);
982 m_view.frameView().incrementVisuallyNonEmptyCharacterCount(textRenderer.text());
983 }
984 return;
985 }
986 if (is<RenderHTMLCanvas>(child) || is<RenderEmbeddedObject>(child)) {
987 // Actual size is not known yet, report the default intrinsic size for replaced elements.
988 auto& replacedRenderer = downcast<RenderReplaced>(child);
989 m_view.frameView().incrementVisuallyNonEmptyPixelCount(roundedIntSize(replacedRenderer.intrinsicSize()));
990 return;
991 }
992 if (is<RenderSVGRoot>(child)) {
993 auto fixedSize = [] (const auto& renderer) -> std::optional<IntSize> {
994 auto& style = renderer.style();
995 if (!style.width().isFixed() || !style.height().isFixed())
996 return { };
997 return std::make_optional(IntSize { style.width().intValue(), style.height().intValue() });
998 };
999 // SVG content tends to have a fixed size construct. However this is known to be inaccurate in certain cases (box-sizing: border-box) or especially when the parent box is oversized.
1000 auto candidateSize = IntSize { };
1001 if (auto size = fixedSize(child))
1002 candidateSize = *size;
1003 else if (auto size = fixedSize(parent))
1004 candidateSize = *size;
1005
1006 if (!candidateSize.isEmpty())
1007 m_view.frameView().incrementVisuallyNonEmptyPixelCount(candidateSize);
1008 return;
1009 }
