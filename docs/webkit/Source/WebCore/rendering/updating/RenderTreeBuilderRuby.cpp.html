40static inline RenderRubyRun& findRubyRunParent(RenderObject& child)
41{
42 return *lineageOfType<RenderRubyRun>(child).first();
43}

45static inline bool isAnonymousRubyInlineBlock(const RenderObject* object)
46{
47 ASSERT(!object
48 || !isRuby(object->parent())
49 || is<RenderRubyRun>(*object)
50 || (object->isInline() && (object->isBeforeContent() || object->isAfterContent()))
51 || (object->isAnonymous() && is<RenderBlock>(*object) && object->style().display() == DisplayType::InlineBlock)
52 || object->isRenderMultiColumnFlow()
53 || object->isRenderMultiColumnSet());
54
55 return object
56 && isRuby(object->parent())
57 && is<RenderBlock>(*object)
58 && !is<RenderRubyRun>(*object);
59}

61static inline bool isRubyBeforeBlock(const RenderObject* object)
62{
63 return isAnonymousRubyInlineBlock(object)
64 && !object->previousSibling()
65 && downcast<RenderBlock>(*object).firstChild()
66 && downcast<RenderBlock>(*object).firstChild()->style().styleType() == PseudoId::Before;
67}
68
69static inline bool isRubyAfterBlock(const RenderObject* object)
70{
71 return isAnonymousRubyInlineBlock(object)
72 && !object->nextSibling()
73 && downcast<RenderBlock>(*object).firstChild()
74 && downcast<RenderBlock>(*object).firstChild()->style().styleType() == PseudoId::After;
75}

77#if ASSERT_ENABLED
78static inline bool isRubyChildForNormalRemoval(const RenderObject& object)
79{
80 return object.isRubyRun()
81 || object.isBeforeContent()
82 || object.isAfterContent()
83 || object.isRenderMultiColumnFlow()
84 || object.isRenderMultiColumnSet()
85 || isAnonymousRubyInlineBlock(&object);
86}
87#endif // ASSERT_ENABLED

89static inline RenderBlock* rubyBeforeBlock(const RenderElement* ruby)
90{
91 RenderObject* child = ruby->firstChild();
92 return isRubyBeforeBlock(child) ? downcast<RenderBlock>(child) : nullptr;
93}
94
95static inline RenderBlock* rubyAfterBlock(const RenderElement* ruby)
96{
97 RenderObject* child = ruby->lastChild();
98 return isRubyAfterBlock(child) ? downcast<RenderBlock>(child) : nullptr;
99}

101static auto createAnonymousRubyInlineBlock(RenderObject& ruby)
102{
103 auto newBlock = createRenderer<RenderBlockFlow>(ruby.document(), RenderStyle::createAnonymousStyleWithDisplay(ruby.style(), DisplayType::InlineBlock));
104 newBlock->initializeStyle();
105 return newBlock;
106}

108static RenderRubyRun* lastRubyRun(const RenderElement* ruby)
109{
110 RenderObject* child = ruby->lastChild();
111 if (child && !is<RenderRubyRun>(*child))
112 child = child->previousSibling();
113 if (!is<RenderRubyRun>(child)) {
114 ASSERT(!child || child->isBeforeContent() || child == rubyBeforeBlock(ruby) || child->isRenderMultiColumnFlow() || child->isRenderMultiColumnSet());
115 return nullptr;
116 }
117 return downcast<RenderRubyRun>(child);
118}

125void RenderTreeBuilder::Ruby::moveInlineChildren(RenderRubyBase& from, RenderRubyBase& to, RenderObject* beforeChild)
126{
127 ASSERT(from.childrenInline());
128
129 if (!from.firstChild())
130 return;
131
132 RenderBlock* toBlock = nullptr;
133 if (to.childrenInline()) {
134 // The standard and easy case: move the children into the target base
135 toBlock = &to;
136 } else {
137 // We need to wrap the inline objects into an anonymous block.
138 // If toBase has a suitable block, we re-use it, otherwise create a new one.
139 auto* lastChild = to.lastChild();
140 if (lastChild && lastChild->isAnonymousBlock() && lastChild->childrenInline())
141 toBlock = downcast<RenderBlock>(lastChild);
142 else {
143 auto newToBlock = to.createAnonymousBlock();
144 toBlock = newToBlock.get();
145 m_builder.attachToRenderElementInternal(to, WTFMove(newToBlock));
146 }
147 }
148 ASSERT(toBlock);
149 // Move our inline children into the target block we determined above.
150 m_builder.moveChildren(from, *toBlock, from.firstChild(), beforeChild, RenderTreeBuilder::NormalizeAfterInsertion::No);
151}

153void RenderTreeBuilder::Ruby::moveBlockChildren(RenderRubyBase& from, RenderRubyBase& to, RenderObject* beforeChild)
154{
155 ASSERT(!from.childrenInline());
156
157 if (!from.firstChild())
158 return;
159
160 if (to.childrenInline())
161 m_builder.makeChildrenNonInline(to);
162
163 // If an anonymous block would be put next to another such block, then merge those.
164 auto* firstChildHere = from.firstChild();
165 auto* lastChildThere = to.lastChild();
166 if (firstChildHere->isAnonymousBlock() && firstChildHere->childrenInline()
167 && lastChildThere && lastChildThere->isAnonymousBlock() && lastChildThere->childrenInline()) {
168 auto* anonBlockHere = downcast<RenderBlock>(firstChildHere);
169 auto* anonBlockThere = downcast<RenderBlock>(lastChildThere);
170 m_builder.moveAllChildren(*anonBlockHere, *anonBlockThere, RenderTreeBuilder::NormalizeAfterInsertion::Yes);
171 anonBlockHere->deleteLines();
172 m_builder.destroy(*anonBlockHere);
173 }
174 // Move all remaining children normally.
175 m_builder.moveChildren(from, to, from.firstChild(), beforeChild, RenderTreeBuilder::NormalizeAfterInsertion::No);
176}

178void RenderTreeBuilder::Ruby::moveChildren(RenderRubyBase& from, RenderRubyBase& to)
179{
180 moveChildrenInternal(from, to);
181 from.addFloatsToNewParent(to);
182}
183
184void RenderTreeBuilder::Ruby::moveChildrenInternal(RenderRubyBase& from, RenderRubyBase& to, RenderObject* beforeChild)
185{
186 // This function removes all children that are before (!) beforeChild
187 // and appends them to toBase.
188 if (beforeChild && beforeChild->parent() != &from)
189 beforeChild = m_builder.splitAnonymousBoxesAroundChild(from, *beforeChild);
190
191 if (from.childrenInline())
192 moveInlineChildren(from, to, beforeChild);
193 else
194 moveBlockChildren(from, to, beforeChild);
195
196 from.setNeedsLayoutAndPrefWidthsRecalc();
197 to.setNeedsLayoutAndPrefWidthsRecalc();
198}

200void RenderTreeBuilder::Ruby::attach(RenderRubyRun& parent, RenderPtr<RenderObject> child, RenderObject* beforeChild)
201{
202 if (child->isRubyText()) {
203 if (!beforeChild) {
204 // RenderRuby has already ascertained that we can add the child here.
205 ASSERT(!parent.hasRubyText());
206 // prepend ruby texts as first child
207 m_builder.blockFlowBuilder().attach(parent, WTFMove(child), parent.firstChild());
208 return;
209 }
210 if (beforeChild->isRubyText()) {
211 // New text is inserted just before another.
212 // In this case the new text takes the place of the old one, and
213 // the old text goes into a new run that is inserted as next sibling.
214 ASSERT(beforeChild->parent() == &parent);
215 RenderElement* ruby = parent.parent();
216 ASSERT(isRuby(ruby));
217 auto newRun = RenderRubyRun::staticCreateRubyRun(ruby);
218 auto& run = *newRun;
219 m_builder.attach(*ruby, WTFMove(newRun), parent.nextSibling());
220 // Add the new ruby text and move the old one to the new run
221 // Note: Doing it in this order and not using RenderRubyRun's methods,
222 // in order to avoid automatic removal of the ruby run in case there is no
223 // other child besides the old ruby text.
224 m_builder.blockFlowBuilder().attach(parent, WTFMove(child), beforeChild);
225 auto takenBeforeChild = m_builder.blockBuilder().detach(parent, *beforeChild);
226
227 m_builder.attach(run, WTFMove(takenBeforeChild));
228 return;
229 }
230 if (parent.hasRubyBase()) {
231 // Insertion before a ruby base object.
232 // In this case we need insert a new run before the current one and split the base.
233 RenderElement* ruby = parent.parent();
234 auto newRun = RenderRubyRun::staticCreateRubyRun(ruby);
235 auto& run = *newRun;
236 m_builder.attach(*ruby, WTFMove(newRun), &parent);
237 m_builder.attach(run, WTFMove(child));
238 moveChildrenInternal(rubyBaseSafe(parent), rubyBaseSafe(run), beforeChild);
239 }
240 return;
241 }
242 // child is not a text -> insert it into the base
243 if (beforeChild && beforeChild->isRubyText()) {
244 // Append it instead if beforeChild is the ruby text.
245 beforeChild = nullptr;
246 }
247 if (!parent.hasRubyBase()) {
248 // Child is going to be attached to the newly constructed ruby base.
249 beforeChild = nullptr;
250 }
251 m_builder.attach(rubyBaseSafe(parent), WTFMove(child), beforeChild);
252}

254RenderElement& RenderTreeBuilder::Ruby::findOrCreateParentForChild(RenderRubyAsBlock& parent, const RenderObject& child, RenderObject*& beforeChild)
255{
256 // Insert :before and :after content before/after the RenderRubyRun(s)
257 if (child.isBeforeContent()) {
258 // Add generated inline content normally
259 if (child.isInline())
260 return parent;
261 // Wrap non-inline content in an anonymous inline-block.
262 auto* beforeBlock = rubyBeforeBlock(&parent);
263 if (!beforeBlock) {
264 auto newBlock = createAnonymousRubyInlineBlock(parent);
265 beforeBlock = newBlock.get();
266 m_builder.blockFlowBuilder().attach(parent, WTFMove(newBlock), parent.firstChild());
267 }
268 beforeChild = nullptr;
269 return *beforeBlock;
270 }
271
272 if (child.isAfterContent()) {
273 // Add generated inline content normally
274 if (child.isInline())
275 return parent;
276 // Wrap non-inline content with an anonymous inline-block.
277 auto* afterBlock = rubyAfterBlock(&parent);
278 if (!afterBlock) {
279 auto newBlock = createAnonymousRubyInlineBlock(parent);
280 afterBlock = newBlock.get();
281 m_builder.blockFlowBuilder().attach(parent, WTFMove(newBlock), nullptr);
282 }
283 beforeChild = nullptr;
284 return *afterBlock;
285 }
286
287 // If the child is a ruby run, just add it normally.
288 if (child.isRubyRun())
289 return parent;
290
291 if (beforeChild && !parent.isAfterContent(beforeChild)) {
292 // insert child into run
293 ASSERT(!beforeChild->isRubyRun());
294 auto* run = beforeChild->parent();
295 while (run && !run->isRubyRun())
296 run = run->parent();
297 if (run)
298 return *run;
299 ASSERT_NOT_REACHED(); // beforeChild should always have a run as parent!
300 // Emergency fallback: fall through and just append.
301 }
302
303 // If the new child would be appended, try to add the child to the previous run
304 // if possible, or create a new run otherwise.
305 // (The RenderRubyRun object will handle the details)
306 auto* lastRun = lastRubyRun(&parent);
307 if (!lastRun || lastRun->hasRubyText()) {
308 auto newRun = RenderRubyRun::staticCreateRubyRun(&parent);
309 lastRun = newRun.get();
310 m_builder.blockFlowBuilder().attach(parent, WTFMove(newRun), beforeChild);
311 }
312 beforeChild = nullptr;
313 return *lastRun;
314}

335 // Add generated inline content normally

378RenderRubyBase& RenderTreeBuilder::Ruby::rubyBaseSafe(RenderRubyRun& rubyRun)
379{
380 auto* base = rubyRun.rubyBase();
381 if (!base) {
382 auto newBase = rubyRun.createRubyBase();
383 base = newBase.get();
384 m_builder.blockFlowBuilder().attach(rubyRun, WTFMove(newBase), nullptr);
385 }
386 return *base;
387}

389RenderPtr<RenderObject> RenderTreeBuilder::Ruby::detach(RenderRubyAsInline& parent, RenderObject& child)
390{
391 // If the child's parent is *this (must be a ruby run or generated content or anonymous block),
392 // just use the normal remove method.
393 if (child.parent() == &parent) {
394 ASSERT(isRubyChildForNormalRemoval(child));
395 return m_builder.detachFromRenderElement(parent, child);
396 }
397 // If the child's parent is an anoymous block (must be generated :before/:after content)
398 // just use the block's remove method.
399 if (isAnonymousRubyInlineBlock(child.parent())) {
400 ASSERT(child.isBeforeContent() || child.isAfterContent());
401 auto& parent = *child.parent();
402 auto takenChild = m_builder.detach(parent, child);
403 m_builder.destroy(parent);
404 return takenChild;
405 }
406
407 // Otherwise find the containing run and remove it from there.
408 return m_builder.detach(findRubyRunParent(child), child);
409}

433RenderPtr<RenderObject> RenderTreeBuilder::Ruby::detach(RenderRubyRun& parent, RenderObject& child)
434{
435 // If the child is a ruby text, then merge the ruby base with the base of
436 // the right sibling run, if possible.
437 if (!parent.beingDestroyed() && !parent.renderTreeBeingDestroyed() && child.isRubyText()) {
438 RenderRubyBase* base = parent.rubyBase();
439 RenderObject* rightNeighbour = parent.nextSibling();
440 if (base && is<RenderRubyRun>(rightNeighbour)) {
441 // Ruby run without a base can happen only at the first run.
442 RenderRubyRun& rightRun = downcast<RenderRubyRun>(*rightNeighbour);
443 if (rightRun.hasRubyBase()) {
444 RenderRubyBase* rightBase = rightRun.rubyBase();
445 // Collect all children in a single base, then swap the bases.
446 moveChildren(*rightBase, *base);
447 m_builder.move(parent, rightRun, *base, RenderTreeBuilder::NormalizeAfterInsertion::No);
448 m_builder.move(rightRun, parent, *rightBase, RenderTreeBuilder::NormalizeAfterInsertion::No);
449 // The now empty ruby base will be removed below.
450 ASSERT(!parent.rubyBase()->firstChild());
451 }
452 }
453 }
454
455 auto takenChild = m_builder.blockBuilder().detach(parent, child);
456
457 if (!parent.beingDestroyed() && !parent.renderTreeBeingDestroyed()) {
458 // Check if our base (if any) is now empty. If so, destroy it.
459 RenderBlock* base = parent.rubyBase();
460 if (base && !base->firstChild()) {
461 auto takenBase = m_builder.blockBuilder().detach(parent, *base);
462 base->deleteLines();
463 }
464 }
465 return takenChild;
466}
