49#if ENABLE(CONTEXT_MENU_EVENT)

116void UserInputBridge::loadRequest(FrameLoadRequest&& request, InputSource)
117{
118#if ENABLE(WEB_AUTHN)
119 m_page.authenticatorCoordinator().resetUserGestureRequirement();
120#endif
121 m_page.mainFrame().loader().load(WTFMove(request));
122}
123
124void UserInputBridge::reloadFrame(Frame& frame, OptionSet<ReloadOption> options, InputSource)
125{
126#if ENABLE(WEB_AUTHN)
127 m_page.authenticatorCoordinator().resetUserGestureRequirement();
128#endif
129 frame.loader().reload(options);
130}
131
132void UserInputBridge::stopLoadingFrame(Frame& frame, InputSource)
133{
134 frame.loader().stopForUserCancel();
135}
136
137bool UserInputBridge::tryClosePage(InputSource)
138{
139 return m_page.mainFrame().loader().shouldClose();
140}
