75void StorageEvent::initStorageEvent(const AtomString& type, bool canBubble, bool cancelable, const String& key, const String& oldValue, const String& newValue, const String& url, Storage* storageArea)
76{
77 if (isBeingDispatched())
78 return;
79
80 initEvent(type, canBubble, cancelable);
81
82 m_key = key;
83 m_oldValue = oldValue;
84 m_newValue = newValue;
85 m_url = url;
86 m_storageArea = storageArea;
87}
