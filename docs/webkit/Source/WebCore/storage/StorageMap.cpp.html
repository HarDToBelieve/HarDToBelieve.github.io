40void StorageMap::invalidateIterator()
41{
42 m_impl->iterator = m_impl->map.end();
43 m_impl->iteratorIndex = UINT_MAX;
44}
45
46void StorageMap::setIteratorToIndex(unsigned index)
47{
48 // FIXME: Once we have bidirectional iterators for HashMap we can be more intelligent about this.
49 // The requested index will be closest to begin(), our current iterator, or end(), and we
50 // can take the shortest route.
51 // Until that mechanism is available, we'll always increment our iterator from begin() or current.
52
53 if (m_impl->iteratorIndex == index)
54 return;
55
56 if (index < m_impl->iteratorIndex) {
57 m_impl->iteratorIndex = 0;
58 m_impl->iterator = m_impl->map.begin();
59 ASSERT(m_impl->iterator != m_impl->map.end());
60 }
61
62 while (m_impl->iteratorIndex < index) {
63 ++m_impl->iteratorIndex;
64 ++m_impl->iterator;
65 ASSERT(m_impl->iterator != m_impl->map.end());
66 }
67}

74String StorageMap::key(unsigned index)
75{
76 if (index >= length())
77 return String();
78
79 setIteratorToIndex(index);
80 return m_impl->iterator->key;

88void StorageMap::setItem(const String& key, const String& value, String& oldValue, bool& quotaException)
89{
90 ASSERT(!value.isNull());
91 quotaException = false;
92
93 // Implement copy-on-write semantics.
94 if (m_impl->refCount() > 1)
95 m_impl = m_impl->copy();
96
97 oldValue = m_impl->map.get(key);
98
99 // Quota tracking. This is done in a couple of steps to keep the overflow tracking simple.
100 CheckedUint32 newSize = m_impl->currentSize;
101 newSize -= oldValue.sizeInBytes();
102 newSize += value.sizeInBytes();
103 if (oldValue.isNull())
104 newSize += key.sizeInBytes();
105 if (m_quotaSize != noQuota && (newSize.hasOverflowed() || newSize > m_quotaSize)) {
106 quotaException = true;
107 return;
108 }
109 m_impl->currentSize = newSize;
110
111 HashMap<String, String>::AddResult addResult = m_impl->map.add(key, value);
112 if (!addResult.isNewEntry)
113 addResult.iterator->value = value;
114
115 invalidateIterator();
116}

118void StorageMap::setItemIgnoringQuota(const String& key, const String& value)
119{
120 SetForScope<unsigned> quotaSizeChange(m_quotaSize, noQuota);
121
122 String oldValue;
123 bool quotaException;
124 setItem(key, value, oldValue, quotaException);
125 ASSERT(!quotaException);
126}

128void StorageMap::removeItem(const String& key, String& oldValue)
129{
130 // Implement copy-on-write semantics.
131 if (m_impl->refCount() > 1)
132 m_impl = m_impl->copy();
133
134 oldValue = m_impl->map.take(key);
135 if (oldValue.isNull())
136 return;
137
138 invalidateIterator();
139 ASSERT(m_impl->currentSize - key.sizeInBytes() <= m_impl->currentSize);
140 m_impl->currentSize -= key.sizeInBytes();
141 ASSERT(m_impl->currentSize - oldValue.sizeInBytes() <= m_impl->currentSize);
142 m_impl->currentSize -= oldValue.sizeInBytes();
143}

145void StorageMap::clear()
146{
147 if (m_impl->refCount() > 1 && length()) {
148 m_impl = Impl::create();
149 return;
150 }
151 m_impl->map.clear();
152 m_impl->currentSize = 0;
153 invalidateIterator();
154}

161void StorageMap::importItems(HashMap<String, String>&& items)
162{
163 if (m_impl->map.isEmpty()) {
164 // Fast path.
165 m_impl->map = WTFMove(items);
166 for (auto& pair : m_impl->map) {
167 ASSERT(m_impl->currentSize + pair.key.sizeInBytes() + pair.value.sizeInBytes() >= m_impl->currentSize);
168 m_impl->currentSize += (pair.key.sizeInBytes() + pair.value.sizeInBytes());
169 }
170 return;
171 }
172
173 for (auto& item : items) {
174 auto& key = item.key;
175 auto& value = item.value;
176
177 ASSERT(m_impl->currentSize + key.sizeInBytes() + value.sizeInBytes() >= m_impl->currentSize);
178 m_impl->currentSize += (key.sizeInBytes() + value.sizeInBytes());
179
180 auto result = m_impl->map.add(WTFMove(key), WTFMove(value));
181 ASSERT_UNUSED(result, result.isNewEntry); // True if the key didn't exist previously.
182 }
183}

185Ref<StorageMap::Impl> StorageMap::Impl::copy() const

187 auto copy = Impl::create();
