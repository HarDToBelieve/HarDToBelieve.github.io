26#pragma once
27
28#include <wtf/HashMap.h>
29#include <wtf/RefCounted.h>
30#include <wtf/text/StringHash.h>
31#include <wtf/text/WTFString.h>
32
33namespace WebCore {
34
35// This class uses copy-on-write semantics.
36class StorageMap {
37 WTF_MAKE_FAST_ALLOCATED;
38public:
39 // Quota size measured in bytes.
40 WEBCORE_EXPORT explicit StorageMap(unsigned quotaSize);
41
42 WEBCORE_EXPORT unsigned length() const;
43 WEBCORE_EXPORT String key(unsigned index);
44 WEBCORE_EXPORT String getItem(const String&) const;
45 WEBCORE_EXPORT void setItem(const String& key, const String& value, String& oldValue, bool& quotaException);
46 WEBCORE_EXPORT void setItemIgnoringQuota(const String& key, const String& value);
47 WEBCORE_EXPORT void removeItem(const String&, String& oldValue);
48 WEBCORE_EXPORT void clear();
49
50 WEBCORE_EXPORT bool contains(const String& key) const;
51
52 WEBCORE_EXPORT void importItems(HashMap<String, String>&&);
53 const HashMap<String, String>& items() const { return m_impl->map; }
54
55 unsigned quota() const { return m_quotaSize; }
56
57 bool isShared() const { return !m_impl->hasOneRef(); }
58
59 static constexpr unsigned noQuota = std::numeric_limits<unsigned>::max();
60
61private:
62 void invalidateIterator();
63 void setIteratorToIndex(unsigned);
64
65 struct Impl : public RefCounted<Impl> {
66 static Ref<Impl> create()
67 {
68 return adoptRef(*new Impl);
69 }
70
71 Ref<Impl> copy() const;
72
73 HashMap<String, String> map;
74 HashMap<String, String>::iterator iterator { map.end() };
75 unsigned iteratorIndex { std::numeric_limits<unsigned>::max() };
76 unsigned currentSize { 0 };
77 };
78
79 Ref<Impl> m_impl;
80 unsigned m_quotaSize { noQuota }; // Measured in bytes.
81};
82
83} // namespace WebCore
