46void InspectorCSSOMWrappers::collectFromStyleSheetIfNeeded(CSSStyleSheet* styleSheet)
47{
48 if (!m_styleRuleToCSSOMWrapperMap.isEmpty())
49 collect(styleSheet);
50}
51
52template <class ListType>
53void InspectorCSSOMWrappers::collect(ListType* listType)
54{
55 if (!listType)
56 return;
57 unsigned size = listType->length();
58 for (unsigned i = 0; i < size; ++i) {
59 CSSRule* cssRule = listType->item(i);
60 switch (cssRule->type()) {
61 case CSSRule::IMPORT_RULE:
62 collect(downcast<CSSImportRule>(*cssRule).styleSheet());
63 break;
64 case CSSRule::MEDIA_RULE:
65 collect(downcast<CSSMediaRule>(cssRule));
66 break;
67 case CSSRule::SUPPORTS_RULE:
68 collect(downcast<CSSSupportsRule>(cssRule));
69 break;
70 case CSSRule::STYLE_RULE:
71 m_styleRuleToCSSOMWrapperMap.add(&downcast<CSSStyleRule>(*cssRule).styleRule(), downcast<CSSStyleRule>(cssRule));
72 break;
73 default:
74 break;
75 }
76 }
77}

79void InspectorCSSOMWrappers::collectFromStyleSheetContents(StyleSheetContents* styleSheet)
80{
81 if (!styleSheet)
82 return;
83 auto styleSheetWrapper = CSSStyleSheet::create(*styleSheet);
84 m_styleSheetCSSOMWrapperSet.add(styleSheetWrapper.copyRef());
85 collect(styleSheetWrapper.ptr());
86}
87
88void InspectorCSSOMWrappers::collectFromStyleSheets(const Vector<RefPtr<CSSStyleSheet>>& sheets)
89{
90 for (auto& sheet : sheets)
91 collect(sheet.get());
92}
93
94void InspectorCSSOMWrappers::maybeCollectFromStyleSheets(const Vector<RefPtr<CSSStyleSheet>>& sheets)
95{
96 for (auto& sheet : sheets) {
97 if (!m_styleSheetCSSOMWrapperSet.contains(sheet.get())) {
98 m_styleSheetCSSOMWrapperSet.add(sheet);
99 collect(sheet.get());
100 }
101 }
102}

104void InspectorCSSOMWrappers::collectDocumentWrappers(ExtensionStyleSheets& extensionStyleSheets)
105{
106 if (m_styleRuleToCSSOMWrapperMap.isEmpty()) {
107 collectFromStyleSheetContents(UserAgentStyle::defaultStyleSheet);
108 collectFromStyleSheetContents(UserAgentStyle::quirksStyleSheet);
109 collectFromStyleSheetContents(UserAgentStyle::dialogStyleSheet);
110 collectFromStyleSheetContents(UserAgentStyle::svgStyleSheet);
111 collectFromStyleSheetContents(UserAgentStyle::mathMLStyleSheet);
112 collectFromStyleSheetContents(UserAgentStyle::mediaControlsStyleSheet);
113 collectFromStyleSheetContents(UserAgentStyle::fullscreenStyleSheet);
114#if ENABLE(DATALIST_ELEMENT)
115 collectFromStyleSheetContents(UserAgentStyle::dataListStyleSheet);
116#endif
117#if ENABLE(INPUT_TYPE_COLOR)
118 collectFromStyleSheetContents(UserAgentStyle::colorInputStyleSheet);
119#endif
120#if ENABLE(IOS_FORM_CONTROL_REFRESH)
121 collectFromStyleSheetContents(UserAgentStyle::legacyFormControlsIOSStyleSheet);
122#endif
123 collectFromStyleSheetContents(UserAgentStyle::plugInsStyleSheet);
124 collectFromStyleSheetContents(UserAgentStyle::mediaQueryStyleSheet);
125
126 collect(extensionStyleSheets.pageUserSheet());
127 collectFromStyleSheets(extensionStyleSheets.injectedUserStyleSheets());
128 collectFromStyleSheets(extensionStyleSheets.documentUserStyleSheets());
129 collectFromStyleSheets(extensionStyleSheets.injectedAuthorStyleSheets());
130 collectFromStyleSheets(extensionStyleSheets.authorStyleSheetsForTesting());
131 }
132}
133
134void InspectorCSSOMWrappers::collectScopeWrappers(Scope& styleScope)
135{
136 maybeCollectFromStyleSheets(styleScope.activeStyleSheets());
137}
138
139CSSStyleRule* InspectorCSSOMWrappers::getWrapperForRuleInSheets(const StyleRule* rule)
140{
141 return m_styleRuleToCSSOMWrapperMap.get(rule);
142}
