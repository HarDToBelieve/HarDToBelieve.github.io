35void PseudoClassChangeInvalidation::computeInvalidation(CSSSelector::PseudoClassType pseudoClass, InvalidationScope invalidationScope)
36{
37 bool shouldInvalidateCurrent = false;
38 bool mayAffectStyleInShadowTree = false;
39
40 traverseRuleFeatures(m_element, [&] (const RuleFeatureSet& features, bool mayAffectShadowTree) {
41 if (mayAffectShadowTree && features.pseudoClassRules.contains(pseudoClass))
42 mayAffectStyleInShadowTree = true;
43 if (m_element.shadowRoot() && features.pseudoClassesAffectingHost.contains(pseudoClass))
44 shouldInvalidateCurrent = true;
45 });
46
47 if (mayAffectStyleInShadowTree) {
48 // FIXME: We should do fine-grained invalidation for shadow tree.
49 m_element.invalidateStyleForSubtree();
50 }
51
52 if (shouldInvalidateCurrent)
53 m_element.invalidateStyle();
54
55 auto& ruleSets = m_element.styleResolver().ruleSets();
56 if (auto* invalidationRuleSets = ruleSets.pseudoClassInvalidationRuleSets(pseudoClass)) {
57 for (auto& invalidationRuleSet : *invalidationRuleSets) {
58 // For focus/hover we flip the whole ancestor chain. We only need to do deep invalidation traversal in the change root.
59 auto shouldInvalidate = [&] {
60 switch (invalidationScope) {
61 case InvalidationScope::All:
62 return true;
63 case InvalidationScope::SelfChildrenAndSiblings:
64 return invalidationRuleSet.matchElement != MatchElement::Ancestor;
65 case InvalidationScope::Descendants:
66 return invalidationRuleSet.matchElement == MatchElement::Ancestor;
67 }
68 ASSERT_NOT_REACHED();
69 return true;
70 }();
71 if (!shouldInvalidate)
72 continue;
73 Invalidator::addToMatchElementRuleSets(m_matchElementRuleSets, invalidationRuleSet);
74 }
75 }
76}
77
78void PseudoClassChangeInvalidation::invalidateStyleWithRuleSets()
79{
80 Invalidator::invalidateWithMatchElementRuleSets(m_element, m_matchElementRuleSets);
81}
