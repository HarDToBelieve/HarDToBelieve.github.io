55void RuleSet::addToRuleSet(const AtomString& key, AtomRuleMap& map, const RuleData& ruleData)
56{
57 if (key.isNull())
58 return;
59 auto& rules = map.add(key, nullptr).iterator->value;
60 if (!rules)
61 rules = makeUnique<RuleDataVector>();
62 rules->append(ruleData);
63}

65static unsigned rulesCountForName(const RuleSet::AtomRuleMap& map, const AtomString& name)
66{
67 if (const auto* rules = map.get(name))
68 return rules->size();
69 return 0;

72static bool isHostSelectorMatchingInShadowTree(const CSSSelector& startSelector)
73{
74 auto* leftmostSelector = &startSelector;
75 bool hasDescendantOrChildRelation = false;
76 while (auto* previous = leftmostSelector->tagHistory()) {
77 hasDescendantOrChildRelation = leftmostSelector->hasDescendantOrChildRelation();
78 leftmostSelector = previous;
79 }
80 if (!hasDescendantOrChildRelation)
81 return false;
82
83 return leftmostSelector->match() == CSSSelector::PseudoClass && leftmostSelector->pseudoClassType() == CSSSelector::PseudoClassHost;
84}

86void RuleSet::addRule(const StyleRule& rule, unsigned selectorIndex, unsigned selectorListIndex, MediaQueryCollector* mediaQueryCollector)
87{
88 RuleData ruleData(rule, selectorIndex, selectorListIndex, m_ruleCount++);
89
90 m_features.collectFeatures(ruleData);
91
92 if (mediaQueryCollector)
93 mediaQueryCollector->addRuleIfNeeded(ruleData);
94
95 unsigned classBucketSize = 0;
96 const CSSSelector* idSelector = nullptr;
97 const CSSSelector* tagSelector = nullptr;
98 const CSSSelector* classSelector = nullptr;
99 const CSSSelector* linkSelector = nullptr;
100 const CSSSelector* focusSelector = nullptr;
101 const CSSSelector* hostPseudoClassSelector = nullptr;
102 const CSSSelector* customPseudoElementSelector = nullptr;
103 const CSSSelector* slottedPseudoElementSelector = nullptr;
104 const CSSSelector* partPseudoElementSelector = nullptr;
105#if ENABLE(VIDEO)
106 const CSSSelector* cuePseudoElementSelector = nullptr;
107#endif
108 const CSSSelector* selector = ruleData.selector();
109 do {
110 switch (selector->match()) {
111 case CSSSelector::Id:
112 idSelector = selector;
113 break;
114 case CSSSelector::Class: {
115 auto& className = selector->value();
116 if (!classSelector) {
117 classSelector = selector;
118 classBucketSize = rulesCountForName(m_classRules, className);
119 } else if (classBucketSize) {
120 unsigned newClassBucketSize = rulesCountForName(m_classRules, className);
121 if (newClassBucketSize < classBucketSize) {
122 classSelector = selector;
123 classBucketSize = newClassBucketSize;
124 }
125 }
126 break;
127 }
128 case CSSSelector::Tag:
129 if (selector->tagQName().localName() != starAtom())
130 tagSelector = selector;
131 break;
132 case CSSSelector::PseudoElement:
133 switch (selector->pseudoElementType()) {
134 case CSSSelector::PseudoElementWebKitCustom:
135 case CSSSelector::PseudoElementWebKitCustomLegacyPrefixed:
136 customPseudoElementSelector = selector;
137 break;
138 case CSSSelector::PseudoElementSlotted:
139 slottedPseudoElementSelector = selector;
140 break;
141 case CSSSelector::PseudoElementPart:
142 partPseudoElementSelector = selector;
143 break;
144#if ENABLE(VIDEO)
145 case CSSSelector::PseudoElementCue:
146 cuePseudoElementSelector = selector;
147 break;
148#endif
149 default:
150 break;
151 }
152 break;
153 case CSSSelector::PseudoClass:
154 switch (selector->pseudoClassType()) {
155 case CSSSelector::PseudoClassLink:
156 case CSSSelector::PseudoClassVisited:
157 case CSSSelector::PseudoClassAnyLink:
158 case CSSSelector::PseudoClassAnyLinkDeprecated:
159 linkSelector = selector;
160 break;
161 case CSSSelector::PseudoClassDirectFocus:
162 case CSSSelector::PseudoClassFocus:
163 focusSelector = selector;
164 break;
165 case CSSSelector::PseudoClassHost:
166 hostPseudoClassSelector = selector;
167 break;
168 default:
169 break;
170 }
171 break;
172 case CSSSelector::Unknown:
173 case CSSSelector::Exact:
174 case CSSSelector::Set:
175 case CSSSelector::List:
176 case CSSSelector::Hyphen:
177 case CSSSelector::Contain:
178 case CSSSelector::Begin:
179 case CSSSelector::End:
180 case CSSSelector::PagePseudoClass:
181 break;
182 }
183 if (selector->relation() != CSSSelector::Subselector)
184 break;
185 selector = selector->tagHistory();
186 } while (selector);
187
188#if ENABLE(VIDEO)
189 if (cuePseudoElementSelector) {
190 m_cuePseudoRules.append(ruleData);
191 return;
192 }
193#endif
194
195 if (slottedPseudoElementSelector) {
196 // ::slotted pseudo elements work accross shadow boundary making filtering difficult.
197 ruleData.disableSelectorFiltering();
198 m_slottedPseudoElementRules.append(ruleData);
199 return;
200 }
201
202 if (partPseudoElementSelector) {
203 // Filtering doesn't work accross shadow boundaries.
204 ruleData.disableSelectorFiltering();
205 m_partPseudoElementRules.append(ruleData);
206 return;
207 }
208
209 if (customPseudoElementSelector) {
210 // FIXME: Custom pseudo elements are handled by the shadow tree's selector filter. It doesn't know about the main DOM.
211 ruleData.disableSelectorFiltering();
212
213 auto* nextSelector = customPseudoElementSelector->tagHistory();
214 if (nextSelector && nextSelector->match() == CSSSelector::PseudoElement && nextSelector->pseudoElementType() == CSSSelector::PseudoElementPart) {
215 // Handle selectors like ::part(foo)::placeholder with the part codepath.
216 m_partPseudoElementRules.append(ruleData);
217 return;
218 }
219
220 addToRuleSet(customPseudoElementSelector->value(), m_shadowPseudoElementRules, ruleData);
221 return;
222 }
223
224 if (!m_hasHostPseudoClassRulesMatchingInShadowTree)
225 m_hasHostPseudoClassRulesMatchingInShadowTree = isHostSelectorMatchingInShadowTree(*ruleData.selector());
226
227 if (hostPseudoClassSelector) {
228 m_hostPseudoClassRules.append(ruleData);
229 return;
230 }
231
232 if (idSelector) {
233 addToRuleSet(idSelector->value(), m_idRules, ruleData);
234 return;
235 }
236
237 if (classSelector) {
238 addToRuleSet(classSelector->value(), m_classRules, ruleData);
239 return;
240 }
241
242 if (linkSelector) {
243 m_linkPseudoClassRules.append(ruleData);
244 return;
245 }
246
247 if (focusSelector) {
248 m_focusPseudoClassRules.append(ruleData);
249 return;
250 }
251
252 if (tagSelector) {
253 addToRuleSet(tagSelector->tagQName().localName(), m_tagLocalNameRules, ruleData);
254 addToRuleSet(tagSelector->tagLowercaseLocalName(), m_tagLowercaseLocalNameRules, ruleData);
255 return;
256 }
257
258 // If we didn't find a specialized map to stick it in, file under universal rules.
259 m_universalRules.append(ruleData);
260}

267void RuleSet::addChildRules(const Vector<RefPtr<StyleRuleBase>>& rules, MediaQueryCollector& mediaQueryCollector, Resolver* resolver, AddRulesMode mode)
268{
269 for (auto& rule : rules) {
270 if (mode == AddRulesMode::ResolverMutationScan && mediaQueryCollector.didMutateResolverWithinDynamicMediaQuery)
271 break;
272
273 if (is<StyleRule>(*rule)) {
274 if (mode == AddRulesMode::Normal)
275 addStyleRule(downcast<StyleRule>(*rule), mediaQueryCollector);
276 continue;
277 }
278 if (is<StyleRulePage>(*rule)) {
279 if (mode == AddRulesMode::Normal)
280 addPageRule(downcast<StyleRulePage>(*rule));
281 continue;
282 }
283 if (is<StyleRuleMedia>(*rule)) {
284 auto& mediaRule = downcast<StyleRuleMedia>(*rule);
285 if (mediaQueryCollector.pushAndEvaluate(&mediaRule.mediaQueries()))
286 addChildRules(mediaRule.childRules(), mediaQueryCollector, resolver, mode);
287 mediaQueryCollector.pop(&mediaRule.mediaQueries());
288 continue;
289 }
290 if (is<StyleRuleFontFace>(*rule)) {
291 // Add this font face to our set.
292 if (resolver) {
293 resolver->document().fontSelector().addFontFaceRule(downcast<StyleRuleFontFace>(*rule.get()), false);
294 resolver->invalidateMatchedDeclarationsCache();
295 }
296 mediaQueryCollector.didMutateResolver();
297 continue;
298 }
299 if (is<StyleRuleKeyframes>(*rule)) {
300 if (resolver)
301 resolver->addKeyframeStyle(downcast<StyleRuleKeyframes>(*rule));
302 mediaQueryCollector.didMutateResolver();
303 continue;
304 }
305 if (is<StyleRuleSupports>(*rule) && downcast<StyleRuleSupports>(*rule).conditionIsSupported()) {
306 addChildRules(downcast<StyleRuleSupports>(*rule).childRules(), mediaQueryCollector, resolver, mode);
307 continue;
308 }
309 }

318void RuleSet::addRulesFromSheet(StyleSheetContents& sheet, MediaQuerySet* sheetQuery, const MediaQueryEvaluator& evaluator, Style::Resolver& resolver)
319{
320 auto canUseDynamicMediaQueryResolution = [&] {
321 auto mediaQueryCollector = MediaQueryCollector { evaluator, true };
322 if (mediaQueryCollector.pushAndEvaluate(sheetQuery))
323 addRulesFromSheet(sheet, mediaQueryCollector, nullptr, AddRulesMode::ResolverMutationScan);
324 mediaQueryCollector.pop(sheetQuery);
325 return !mediaQueryCollector.didMutateResolverWithinDynamicMediaQuery;
326 }();
327
328 auto mediaQueryCollector = MediaQueryCollector { evaluator, canUseDynamicMediaQueryResolution };
329
330 if (mediaQueryCollector.pushAndEvaluate(sheetQuery))
331 addRulesFromSheet(sheet, mediaQueryCollector, &resolver, AddRulesMode::Normal);
332 mediaQueryCollector.pop(sheetQuery);
333
334 m_hasViewportDependentMediaQueries = mediaQueryCollector.hasViewportDependentMediaQueries;
335
336 if (mediaQueryCollector.dynamicMediaQueryRules.isEmpty())
337 return;
338
339 auto firstNewIndex = m_dynamicMediaQueryRules.size();
340 m_dynamicMediaQueryRules.appendVector(WTFMove(mediaQueryCollector.dynamicMediaQueryRules));
341
342 // Set the initial values.
343 evaluateDynamicMediaQueryRules(evaluator, firstNewIndex);
344}

346void RuleSet::addRulesFromSheet(StyleSheetContents& sheet, MediaQueryCollector& mediaQueryCollector, Resolver* resolver, AddRulesMode mode)
347{
348 for (auto& rule : sheet.importRules()) {
349 if (!rule->styleSheet())
350 continue;
351
352 if (mediaQueryCollector.pushAndEvaluate(rule->mediaQueries()))
353 addRulesFromSheet(*rule->styleSheet(), mediaQueryCollector, resolver, mode);
354 mediaQueryCollector.pop(rule->mediaQueries());
355 }
356
357 addChildRules(sheet.childRules(), mediaQueryCollector, resolver, mode);
358
359 if (m_autoShrinkToFitEnabled && mode == AddRulesMode::Normal)
360 shrinkToFit();
361}

363void RuleSet::addStyleRule(const StyleRule& rule, MediaQueryCollector& mediaQueryCollector)
364{
365 auto& selectorList = rule.selectorList();
366 if (selectorList.isEmpty())
367 return;
368 unsigned selectorListIndex = 0;
369 for (size_t selectorIndex = 0; selectorIndex != notFound; selectorIndex = selectorList.indexOfNextSelectorAfter(selectorIndex))
370 addRule(rule, selectorIndex, selectorListIndex++, &mediaQueryCollector);
371}

373template<typename Function>
374void RuleSet::traverseRuleDatas(Function&& function)
375{
376 auto traverseVector = [&](auto& vector) {
377 for (auto& ruleData : vector)
378 function(ruleData);
379 };
380
381 auto traverseMap = [&](auto& map) {
382 for (auto& ruleDatas : map.values())
383 traverseVector(*ruleDatas);
384 };
385
386 traverseMap(m_idRules);
387 traverseMap(m_classRules);
388 traverseMap(m_tagLocalNameRules);
389 traverseMap(m_tagLowercaseLocalNameRules);
390 traverseMap(m_shadowPseudoElementRules);
391 traverseVector(m_linkPseudoClassRules);
392#if ENABLE(VIDEO)
393 traverseVector(m_cuePseudoRules);
394#endif
395 traverseVector(m_hostPseudoClassRules);
396 traverseVector(m_slottedPseudoElementRules);
397 traverseVector(m_partPseudoElementRules);
398 traverseVector(m_focusPseudoClassRules);
399 traverseVector(m_universalRules);
400}

402std::optional<DynamicMediaQueryEvaluationChanges> RuleSet::evaluateDynamicMediaQueryRules(const MediaQueryEvaluator& evaluator)
403{
404 auto collectedChanges = evaluateDynamicMediaQueryRules(evaluator, 0);
405
406 if (collectedChanges.requiredFullReset)
407 return { { DynamicMediaQueryEvaluationChanges::Type::ResetStyle } };
408
409 if (collectedChanges.changedQueryIndexes.isEmpty())
410 return { };
411
412 auto& ruleSet = m_mediaQueryInvalidationRuleSetCache.ensure(collectedChanges.changedQueryIndexes, [&] {
413 auto ruleSet = RuleSet::create();
414 for (auto* featureVector : collectedChanges.ruleFeatures) {
415 for (auto& feature : *featureVector)
416 ruleSet->addRule(*feature.styleRule, feature.selectorIndex, feature.selectorListIndex);
417 }
418 return ruleSet;
419 }).iterator->value;
420
421 return { { DynamicMediaQueryEvaluationChanges::Type::InvalidateStyle, { ruleSet.copyRef() } } };
422}

424RuleSet::CollectedMediaQueryChanges RuleSet::evaluateDynamicMediaQueryRules(const MediaQueryEvaluator& evaluator, size_t startIndex)
425{
426 CollectedMediaQueryChanges collectedChanges;
427
428 HashMap<size_t, bool, DefaultHash<size_t>, WTF::UnsignedWithZeroKeyHashTraits<size_t>> affectedRulePositionsAndResults;
429
430 for (size_t i = startIndex; i < m_dynamicMediaQueryRules.size(); ++i) {
431 auto& dynamicRules = m_dynamicMediaQueryRules[i];
432 bool result = true;
433 for (auto& set : dynamicRules.mediaQuerySets) {
434 if (!evaluator.evaluate(set.get())) {
435 result = false;
436 break;
437 }
438 }
439
440 if (result != dynamicRules.result) {
441 dynamicRules.result = result;
442
443 if (dynamicRules.requiresFullReset) {
444 collectedChanges.requiredFullReset = true;
445 continue;
446 }
447
448 for (auto position : dynamicRules.affectedRulePositions)
449 affectedRulePositionsAndResults.add(position, result);
450
451 collectedChanges.changedQueryIndexes.append(i);
452 collectedChanges.ruleFeatures.append(&dynamicRules.ruleFeatures);
453 }
454 }
455
456 if (affectedRulePositionsAndResults.isEmpty())
457 return collectedChanges;
458
459 traverseRuleDatas([&](RuleData& ruleData) {
460 auto it = affectedRulePositionsAndResults.find(ruleData.position());
461 if (it == affectedRulePositionsAndResults.end())
462 return;
463 ruleData.setEnabled(it->value);
464 });
465
466 return collectedChanges;
467}

475void RuleSet::shrinkToFit()

481 shrinkMapVectorsToFit(m_shadowPseudoElementRules);

488 m_focusPseudoClassRules.shrinkToFit();
489 m_universalRules.shrinkToFit();
490 m_pageRules.shrinkToFit();
491 m_features.shrinkToFit();
492}

496bool RuleSet::MediaQueryCollector::pushAndEvaluate(const MediaQuerySet* set)
497{
498 if (!set)
499 return true;
500
501 // Only evaluate static expressions that require style rebuild.
502 MediaQueryDynamicResults dynamicResults;
503 auto mode = collectDynamic ? MediaQueryEvaluator::Mode::AlwaysMatchDynamic : MediaQueryEvaluator::Mode::Normal;
504
505 bool result = evaluator.evaluate(*set, &dynamicResults, mode);
506
507 if (!dynamicResults.viewport.isEmpty())
508 hasViewportDependentMediaQueries = true;
509
510 if (!dynamicResults.isEmpty())
511 dynamicContextStack.append({ *set });
512
513 return result;
514}
515
516void RuleSet::MediaQueryCollector::pop(const MediaQuerySet* set)
517{
518 if (!set || dynamicContextStack.isEmpty() || set != &dynamicContextStack.last().set.get())
519 return;
520
521 if (!dynamicContextStack.last().affectedRulePositions.isEmpty() || !collectDynamic) {
522 DynamicMediaQueryRules rules;
523 for (auto& context : dynamicContextStack)
524 rules.mediaQuerySets.append(context.set.get());
525
526 if (collectDynamic) {
527 rules.affectedRulePositions.appendVector(dynamicContextStack.last().affectedRulePositions);
528 rules.ruleFeatures = WTFMove(dynamicContextStack.last().ruleFeatures);
529 rules.ruleFeatures.shrinkToFit();
530 } else
531 rules.requiresFullReset = true;
532
533 dynamicMediaQueryRules.append(WTFMove(rules));
534 }
535
536 dynamicContextStack.removeLast();
537}

539void RuleSet::MediaQueryCollector::didMutateResolver()
540{
541 if (dynamicContextStack.isEmpty())
542 return;
543 didMutateResolverWithinDynamicMediaQuery = true;
544}
545
546void RuleSet::MediaQueryCollector::addRuleIfNeeded(const RuleData& ruleData)
547{
548 if (dynamicContextStack.isEmpty())
549 return;
550
551 auto& context = dynamicContextStack.last();
552 context.affectedRulePositions.append(ruleData.position());
553 context.ruleFeatures.append({ ruleData });
554}
