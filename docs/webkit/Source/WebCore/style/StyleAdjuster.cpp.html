76static void addIntrinsicMargins(RenderStyle& style)
77{
78 // Intrinsic margin value.
79 const int intrinsicMargin = clampToInteger(2 * style.effectiveZoom());
80
81 // FIXME: Using width/height alone and not also dealing with min-width/max-width is flawed.
82 // FIXME: Using "hasQuirk" to decide the margin wasn't set is kind of lame.
83 if (style.width().isIntrinsicOrAuto()) {
84 if (style.marginLeft().hasQuirk())
85 style.setMarginLeft(Length(intrinsicMargin, LengthType::Fixed));
86 if (style.marginRight().hasQuirk())
87 style.setMarginRight(Length(intrinsicMargin, LengthType::Fixed));
88 }
89
90 if (style.height().isAuto()) {
91 if (style.marginTop().hasQuirk())
92 style.setMarginTop(Length(intrinsicMargin, LengthType::Fixed));
93 if (style.marginBottom().hasQuirk())
94 style.setMarginBottom(Length(intrinsicMargin, LengthType::Fixed));
95 }
96}

98static DisplayType equivalentBlockDisplay(const RenderStyle& style, const Document& document)
99{
100 switch (auto display = style.display()) {
101 case DisplayType::Block:
102 case DisplayType::Table:
103 case DisplayType::Box:
104 case DisplayType::Flex:
105 case DisplayType::Grid:
106 case DisplayType::FlowRoot:
107 return display;
108
109 case DisplayType::ListItem:
110 // It is a WinIE bug that floated list items lose their bullets, so we'll emulate the quirk, but only in quirks mode.
111 if (document.inQuirksMode() && style.isFloating())
112 return DisplayType::Block;
113 return display;
114 case DisplayType::InlineTable:
115 return DisplayType::Table;
116 case DisplayType::InlineBox:
117 return DisplayType::Box;
118 case DisplayType::InlineFlex:
119 return DisplayType::Flex;
120 case DisplayType::InlineGrid:
121 return DisplayType::Grid;
122
123 case DisplayType::Inline:
124 case DisplayType::InlineBlock:
125 case DisplayType::TableRowGroup:
126 case DisplayType::TableHeaderGroup:
127 case DisplayType::TableFooterGroup:
128 case DisplayType::TableRow:
129 case DisplayType::TableColumnGroup:
130 case DisplayType::TableColumn:
131 case DisplayType::TableCell:
132 case DisplayType::TableCaption:
133 return DisplayType::Block;
134 case DisplayType::Contents:
135 ASSERT_NOT_REACHED();
136 return DisplayType::Contents;
137 case DisplayType::None:
138 ASSERT_NOT_REACHED();
139 return DisplayType::None;
140 }
141 ASSERT_NOT_REACHED();
142 return DisplayType::Block;
143}

145static bool shouldInheritTextDecorationsInEffect(const RenderStyle& style, const Element* element)
146{
147 if (style.isFloating() || style.hasOutOfFlowPosition())
148 return false;
149
150 auto isAtUserAgentShadowBoundary = [&] {
151 if (!element)
152 return false;
153 auto* parentNode = element->parentNode();
154 return parentNode && parentNode->isUserAgentShadowRoot();
155 }();
156
157 // There is no other good way to prevent decorations from affecting user agent shadow trees.
158 if (isAtUserAgentShadowBoundary)
159 return false;
160
161 switch (style.display()) {
162 case DisplayType::Table:
163 case DisplayType::InlineTable:
164 case DisplayType::InlineBlock:
165 case DisplayType::InlineBox:
166 return false;
167 default:
168 break;
169 };
170
171 return true;
172}

179static OptionSet<TouchAction> computeEffectiveTouchActions(const RenderStyle& style, OptionSet<TouchAction> effectiveTouchActions)
180{
181 // https://w3c.github.io/pointerevents/#determining-supported-touch-behavior
182 // "A touch behavior is supported if it conforms to the touch-action property of each element between
183 // the hit tested element and its nearest ancestor with the default touch behavior (including both the
184 // hit tested element and the element with the default touch behavior)."
185
186 bool hasDefaultTouchBehavior = isScrollableOverflow(style.overflowX()) || isScrollableOverflow(style.overflowY());
187 if (hasDefaultTouchBehavior)
188 effectiveTouchActions = RenderStyle::initialTouchActions();
189
190 auto touchActions = style.touchActions();
191 if (touchActions == RenderStyle::initialTouchActions())
192 return effectiveTouchActions;
193
194 if (effectiveTouchActions.contains(TouchAction::None))
195 return { TouchAction::None };
196
197 if (effectiveTouchActions.containsAny({ TouchAction::Auto, TouchAction::Manipulation }))
198 return touchActions;
199
200 if (touchActions.containsAny({ TouchAction::Auto, TouchAction::Manipulation }))
201 return effectiveTouchActions;
202
203 auto sharedTouchActions = effectiveTouchActions & touchActions;
204 if (sharedTouchActions.isEmpty())
205 return { TouchAction::None };
206
207 return sharedTouchActions;
208}

219OptionSet<EventListenerRegionType> Adjuster::computeEventListenerRegionTypes(const EventTarget& eventTarget, OptionSet<EventListenerRegionType> parentTypes)
220{
221#if ENABLE(WHEEL_EVENT_REGIONS)
222 if (!eventTarget.hasEventListeners())
223 return parentTypes;
224
225 auto types = parentTypes;
226
227 auto findListeners = [&](auto& eventName, auto type, auto nonPassiveType) {
228 auto* eventListenerVector = eventTarget.eventTargetData()->eventListenerMap.find(eventName);
229 if (!eventListenerVector)
230 return;
231
232 types.add(type);
233
234 auto isPassiveOnly = [&] {
235 for (auto& listener : *eventListenerVector) {
236 if (!listener->isPassive())
237 return false;
238 }
239 return true;
240 }();
241
242 if (!isPassiveOnly)
243 types.add(nonPassiveType);
244 };
245
246 findListeners(eventNames().wheelEvent, EventListenerRegionType::Wheel, EventListenerRegionType::NonPassiveWheel);
247 findListeners(eventNames().mousewheelEvent, EventListenerRegionType::Wheel, EventListenerRegionType::NonPassiveWheel);
248
249 return types;
250#else
251 UNUSED_PARAM(eventTarget);
252 UNUSED_PARAM(parentTypes);
253 return { };
254#endif
255}

257void Adjuster::adjust(RenderStyle& style, const RenderStyle* userAgentAppearanceStyle) const
258{
259 if (style.display() == DisplayType::Contents)
260 adjustDisplayContentsStyle(style);
261
262 if (style.display() != DisplayType::None && style.display() != DisplayType::Contents) {
263 if (m_element) {
264 // If we have a <td> that specifies a float property, in quirks mode we just drop the float
265 // property.
266 // Sites also commonly use display:inline/block on <td>s and <table>s. In quirks mode we force
267 // these tags to retain their display types.
268 if (m_document.inQuirksMode()) {
269 if (m_element->hasTagName(tdTag)) {
270 style.setEffectiveDisplay(DisplayType::TableCell);
271 style.setFloating(Float::None);
272 } else if (is<HTMLTableElement>(*m_element))
273 style.setEffectiveDisplay(style.isDisplayInlineType() ? DisplayType::InlineTable : DisplayType::Table);
274 }
275
276 if (m_element->hasTagName(tdTag) || m_element->hasTagName(thTag)) {
277 if (style.whiteSpace() == WhiteSpace::KHTMLNoWrap) {
278 // Figure out if we are really nowrapping or if we should just
279 // use normal instead. If the width of the cell is fixed, then
280 // we don't actually use WhiteSpace::NoWrap.
281 if (style.width().isFixed())
282 style.setWhiteSpace(WhiteSpace::Normal);
283 else
284 style.setWhiteSpace(WhiteSpace::NoWrap);
285 }
286 }
287
288 // Tables never support the -webkit-* values for text-align and will reset back to the default.
289 if (is<HTMLTableElement>(*m_element) && (style.textAlign() == TextAlignMode::WebKitLeft || style.textAlign() == TextAlignMode::WebKitCenter || style.textAlign() == TextAlignMode::WebKitRight))
290 style.setTextAlign(TextAlignMode::Start);
291
292 // Frames and framesets never honor position:relative or position:absolute. This is necessary to
293 // fix a crash where a site tries to position these objects. They also never honor display.
294 if (m_element->hasTagName(frameTag) || m_element->hasTagName(framesetTag)) {
295 style.setPosition(PositionType::Static);
296 style.setEffectiveDisplay(DisplayType::Block);
297 }
298
299 // Ruby text does not support float or position. This might change with evolution of the specification.
300 if (m_element->hasTagName(rtTag)) {
301 style.setPosition(PositionType::Static);
302 style.setFloating(Float::None);
303 }
304
305 // User agents are expected to have a rule in their user agent stylesheet that matches th elements that have a parent
306 // node whose computed value for the 'text-align' property is its initial value, whose declaration block consists of
307 // just a single declaration that sets the 'text-align' property to the value 'center'.
308 // https://html.spec.whatwg.org/multipage/rendering.html#rendering
309 if (m_element->hasTagName(thTag) && !style.hasExplicitlySetTextAlign() && m_parentStyle.textAlign() == RenderStyle::initialTextAlign())
310 style.setTextAlign(TextAlignMode::Center);
311
312 if (m_element->hasTagName(legendTag))
313 style.setEffectiveDisplay(DisplayType::Block);
314 }
315
316 // Absolute/fixed positioned elements, floating elements and the document element need block-like outside display.
317 if (style.hasOutOfFlowPosition() || style.isFloating() || (m_element && m_document.documentElement() == m_element))
318 style.setEffectiveDisplay(equivalentBlockDisplay(style, m_document));
319
320 // FIXME: Don't support this mutation for pseudo styles like first-letter or first-line, since it's not completely
321 // clear how that should work.
322 if (style.display() == DisplayType::Inline && style.styleType() == PseudoId::None && style.writingMode() != m_parentStyle.writingMode())
323 style.setEffectiveDisplay(DisplayType::InlineBlock);
324
325 // After performing the display mutation, check table rows. We do not honor position:relative or position:sticky on
326 // table rows or cells. This has been established for position:relative in CSS2.1 (and caused a crash in containingBlock()
327 // on some sites).
328 if ((style.display() == DisplayType::TableHeaderGroup || style.display() == DisplayType::TableRowGroup
329 || style.display() == DisplayType::TableFooterGroup || style.display() == DisplayType::TableRow)
330 && style.position() == PositionType::Relative)
331 style.setPosition(PositionType::Static);
332
333 // writing-mode does not apply to table row groups, table column groups, table rows, and table columns.
334 // FIXME: Table cells should be allowed to be perpendicular or flipped with respect to the table, though.
335 if (style.display() == DisplayType::TableColumn || style.display() == DisplayType::TableColumnGroup || style.display() == DisplayType::TableFooterGroup
336 || style.display() == DisplayType::TableHeaderGroup || style.display() == DisplayType::TableRow || style.display() == DisplayType::TableRowGroup
337 || style.display() == DisplayType::TableCell)
338 style.setWritingMode(m_parentStyle.writingMode());
339
340 // FIXME: Since we don't support block-flow on flexible boxes yet, disallow setting
341 // of block-flow to anything other than WritingMode::TopToBottom.
342 // https://bugs.webkit.org/show_bug.cgi?id=46418 - Flexible box support.
343 if (style.writingMode() != WritingMode::TopToBottom && (style.display() == DisplayType::Box || style.display() == DisplayType::InlineBox))
344 style.setWritingMode(WritingMode::TopToBottom);
345
346 // https://www.w3.org/TR/css-display/#transformations
347 // "A parent with a grid or flex display value blockifies the box’s display type."
348 if (m_parentBoxStyle.isDisplayFlexibleOrGridBox()) {
349 style.setFloating(Float::None);
350 style.setEffectiveDisplay(equivalentBlockDisplay(style, m_document));
351 }
352 }
353
354 // Make sure our z-index value is only applied if the object is positioned.
355 if (style.hasAutoSpecifiedZIndex() || (style.position() == PositionType::Static && !m_parentBoxStyle.isDisplayFlexibleOrGridBox()))
356 style.setHasAutoUsedZIndex();
357 else
358 style.setUsedZIndex(style.specifiedZIndex());
359
360 // Auto z-index becomes 0 for the root element and transparent objects. This prevents
361 // cases where objects that should be blended as a single unit end up with a non-transparent
362 // object wedged in between them. Auto z-index also becomes 0 for objects that specify transforms/masks/reflections.
363 if (style.hasAutoUsedZIndex()) {
364 if ((m_element && m_document.documentElement() == m_element)
365 || style.hasOpacity()
366 || style.hasTransformRelatedProperty()
367 || style.hasMask()
368 || style.clipPath()
369 || style.boxReflect()
370 || style.hasFilter()
371#if ENABLE(FILTERS_LEVEL_2)
372 || style.hasBackdropFilter()
373#endif
374 || style.hasBlendMode()
375 || style.hasIsolation()
376 || style.position() == PositionType::Sticky
377 || style.position() == PositionType::Fixed
378 || style.willChangeCreatesStackingContext())
379 style.setUsedZIndex(0);
380 }
381
382 if (m_element) {
383 // Textarea considers overflow visible as auto.
384 if (is<HTMLTextAreaElement>(*m_element)) {
385 style.setOverflowX(style.overflowX() == Overflow::Visible ? Overflow::Auto : style.overflowX());
386 style.setOverflowY(style.overflowY() == Overflow::Visible ? Overflow::Auto : style.overflowY());
387 }
388
389 // Disallow -webkit-user-modify on :pseudo and ::pseudo elements.
390 if (!m_element->shadowPseudoId().isNull())
391 style.setUserModify(UserModify::ReadOnly);
392
393 if (is<HTMLMarqueeElement>(*m_element)) {
394 // For now, <marquee> requires an overflow clip to work properly.
395 style.setOverflowX(Overflow::Hidden);
396 style.setOverflowY(Overflow::Hidden);
397
398 bool isVertical = style.marqueeDirection() == MarqueeDirection::Up || style.marqueeDirection() == MarqueeDirection::Down;
399 // Make horizontal marquees not wrap.
400 if (!isVertical) {
401 style.setWhiteSpace(WhiteSpace::NoWrap);
402 style.setTextAlign(TextAlignMode::Start);
403 }
404 // Apparently this is the expected legacy behavior.
405 if (isVertical && style.height().isAuto())
406 style.setHeight(Length(200, LengthType::Fixed));
407 }
408 }
409
410 if (shouldInheritTextDecorationsInEffect(style, m_element))
411 style.addToTextDecorationsInEffect(style.textDecoration());
412 else
413 style.setTextDecorationsInEffect(style.textDecoration());
414
415 // If either overflow value is not visible, change to auto.
416 if (style.overflowX() == Overflow::Visible && style.overflowY() != Overflow::Visible) {
417 // FIXME: Once we implement pagination controls, overflow-x should default to hidden
418 // if overflow-y is set to -webkit-paged-x or -webkit-page-y. For now, we'll let it
419 // default to auto so we can at least scroll through the pages.
420 style.setOverflowX(Overflow::Auto);
421 } else if (style.overflowY() == Overflow::Visible && style.overflowX() != Overflow::Visible)
422 style.setOverflowY(Overflow::Auto);
423
424 // Call setStylesForPaginationMode() if a pagination mode is set for any non-root elements. If these
425 // styles are specified on a root element, then they will be incorporated in
426 // Style::createForm_document.
427 if ((style.overflowY() == Overflow::PagedX || style.overflowY() == Overflow::PagedY) && !(m_element && (m_element->hasTagName(htmlTag) || m_element->hasTagName(bodyTag))))
428 style.setColumnStylesFromPaginationMode(WebCore::paginationModeForRenderStyle(style));
429
430 // Table rows, sections and the table itself will support overflow:hidden and will ignore scroll/auto.
431 // FIXME: Eventually table sections will support auto and scroll.
432 if (style.display() == DisplayType::Table || style.display() == DisplayType::InlineTable
433 || style.display() == DisplayType::TableRowGroup || style.display() == DisplayType::TableRow) {
434 if (style.overflowX() != Overflow::Visible && style.overflowX() != Overflow::Hidden)
435 style.setOverflowX(Overflow::Visible);
436 if (style.overflowY() != Overflow::Visible && style.overflowY() != Overflow::Hidden)
437 style.setOverflowY(Overflow::Visible);
438 }
439
440 // Menulists should have visible overflow
441 if (style.appearance() == MenulistPart) {
442 style.setOverflowX(Overflow::Visible);
443 style.setOverflowY(Overflow::Visible);
444 }
445
446#if ENABLE(OVERFLOW_SCROLLING_TOUCH)
447 // Touch overflow scrolling creates a stacking context.
448 if (style.hasAutoUsedZIndex() && style.useTouchOverflowScrolling() && (isScrollableOverflow(style.overflowX()) || isScrollableOverflow(style.overflowY())))
449 style.setUsedZIndex(0);
450#endif
451
452 // contain: layout creates a stacking context.
453 if (style.hasAutoUsedZIndex() && style.containsLayout())
454 style.setUsedZIndex(0);
455
456 // Cull out any useless layers and also repeat patterns into additional layers.
457 style.adjustBackgroundLayers();
458 style.adjustMaskLayers();
459
460 // Do the same for animations and transitions.
461 style.adjustAnimations();
462 style.adjustTransitions();
463
464 // Important: Intrinsic margins get added to controls before the theme has adjusted the style, since the theme will
465 // alter fonts and heights/widths.
466 if (is<HTMLFormControlElement>(m_element) && style.computedFontPixelSize() >= 11) {
467 // Don't apply intrinsic margins to image buttons. The designer knows how big the images are,
468 // so we have to treat all image buttons as though they were explicitly sized.
469 if (!is<HTMLInputElement>(*m_element) || !downcast<HTMLInputElement>(*m_element).isImageButton())
470 addIntrinsicMargins(style);
471 }
472
473 // Let the theme also have a crack at adjusting the style.
474 if (style.hasAppearance())
475 RenderTheme::singleton().adjustStyle(style, m_element, userAgentAppearanceStyle);
476
477 // If we have first-letter pseudo style, do not share this style.
478 if (style.hasPseudoStyle(PseudoId::FirstLetter))
479 style.setUnique();
480
481 // FIXME: when dropping the -webkit prefix on transform-style, we should also have opacity < 1 cause flattening.
482 if (style.preserves3D() && (style.overflowX() != Overflow::Visible
483 || style.overflowY() != Overflow::Visible
484 || style.hasClip()
485 || style.clipPath()
486 || style.hasFilter()
487#if ENABLE(FILTERS_LEVEL_2)
488 || style.hasBackdropFilter()
489#endif
490 || style.hasBlendMode()))
491 style.setTransformStyle3D(TransformStyle3D::Flat);
492
493 if (is<SVGElement>(m_element))
494 adjustSVGElementStyle(style, downcast<SVGElement>(*m_element));
495
496 // If the inherited value of justify-items includes the 'legacy' keyword (plus 'left', 'right' or
497 // 'center'), 'legacy' computes to the the inherited value. Otherwise, 'auto' computes to 'normal'.
498 if (m_parentBoxStyle.justifyItems().positionType() == ItemPositionType::Legacy && style.justifyItems().position() == ItemPosition::Legacy)
499 style.setJustifyItems(m_parentBoxStyle.justifyItems());
500
501 style.setEffectiveTouchActions(computeEffectiveTouchActions(style, m_parentStyle.effectiveTouchActions()));
502
503 if (m_element)
504 style.setEventListenerRegionTypes(computeEventListenerRegionTypes(*m_element, m_parentStyle.eventListenerRegionTypes()));
505
506#if ENABLE(TEXT_AUTOSIZING)
507 if (m_element && m_document.settings().textAutosizingUsesIdempotentMode())
508 adjustForTextAutosizing(style, *m_element);
509#endif
510
511 adjustForSiteSpecificQuirks(style);
512}

514static bool hasEffectiveDisplayNoneForDisplayContents(const Element& element)
515{
516 // https://drafts.csswg.org/css-display-3/#unbox-html
517 static NeverDestroyed<MemoryCompactLookupOnlyRobinHoodHashSet<AtomString>> tagNames = [] {
518 static const HTMLQualifiedName* const tagList[] = {
519 &brTag.get(),
520 &wbrTag.get(),
521 &meterTag.get(),
522 &appletTag.get(),
523 &progressTag.get(),
524 &canvasTag.get(),
525 &embedTag.get(),
526 &objectTag.get(),
527 &audioTag.get(),
528 &iframeTag.get(),
529 &imgTag.get(),
530 &videoTag.get(),
531 &frameTag.get(),
532 &framesetTag.get(),
533 &inputTag.get(),
534 &textareaTag.get(),
535 &selectTag.get(),
536 };
537 MemoryCompactLookupOnlyRobinHoodHashSet<AtomString> set;
538 set.reserveInitialCapacity(sizeof(tagList));
539 for (auto& name : tagList)
540 set.add(name->localName());
541 return set;
542 }();
543
544 // https://drafts.csswg.org/css-display-3/#unbox-svg
545 // FIXME: <g>, <use> and <tspan> have special (?) behavior for display:contents in the current draft spec.
546 if (is<SVGElement>(element))
547 return true;
548#if ENABLE(MATHML)
549 // Not sure MathML code can handle it.
550 if (is<MathMLElement>(element))
551 return true;
552#endif // ENABLE(MATHML)
553 if (!is<HTMLElement>(element))
554 return false;
555 return tagNames.get().contains(element.localName());
556}

558void Adjuster::adjustDisplayContentsStyle(RenderStyle& style) const
559{
560 if (!m_element) {
561 if (style.styleType() != PseudoId::Before && style.styleType() != PseudoId::After)
562 style.setEffectiveDisplay(DisplayType::None);
563 return;
564 }
565
566 if (m_document.documentElement() == m_element) {
567 style.setEffectiveDisplay(DisplayType::Block);
568 return;
569 }
570
571 if (hasEffectiveDisplayNoneForDisplayContents(*m_element))
572 style.setEffectiveDisplay(DisplayType::None);
573}

575void Adjuster::adjustSVGElementStyle(RenderStyle& style, const SVGElement& svgElement)
576{
577 // Only the root <svg> element in an SVG document fragment tree honors css position
578 auto isPositioningAllowed = svgElement.hasTagName(SVGNames::svgTag) && svgElement.parentNode() && !svgElement.parentNode()->isSVGElement() && !svgElement.correspondingElement();
579 if (!isPositioningAllowed)
580 style.setPosition(RenderStyle::initialPosition());
581
582 // RenderSVGRoot handles zooming for the whole SVG subtree, so foreignObject content should
583 // not be scaled again.
584 if (svgElement.hasTagName(SVGNames::foreignObjectTag))
585 style.setEffectiveZoom(RenderStyle::initialZoom());
586
587 // SVG text layout code expects us to be a block-level style element.
588 if ((svgElement.hasTagName(SVGNames::foreignObjectTag) || svgElement.hasTagName(SVGNames::textTag)) && style.isDisplayInlineType())
589 style.setEffectiveDisplay(DisplayType::Block);
590}

592void Adjuster::adjustAnimatedStyle(RenderStyle& style, OptionSet<AnimationImpact> impact) const
593{
594 adjust(style, nullptr);
595
596 // Set an explicit used z-index in two cases:
597 // 1. When the element respects z-index, and the style has an explicit z-index set (for example, the animation
598 // itself may animate z-index).
599 // 2. When we want the stacking context side-effets of explicit z-index, via forceStackingContext.
600 // It's important to not clobber an existing used z-index, since an earlier animation may have set it, but we
601 // may still need to update the used z-index value from the specified value.
602
603 if (style.hasAutoUsedZIndex() && impact.contains(AnimationImpact::ForcesStackingContext))
604 style.setUsedZIndex(0);
605}

607void Adjuster::adjustForSiteSpecificQuirks(RenderStyle& style) const
608{
609 if (!m_element)
610 return;
611
612 if (m_document.quirks().needsGMailOverflowScrollQuirk()) {

665auto Adjuster::adjustmentForTextAutosizing(const RenderStyle& style, const Element& element) -> AdjustmentForTextAutosizing
666{
667 AdjustmentForTextAutosizing adjustmentForTextAutosizing;
668
669 auto& document = element.document();
670 if (!document.settings().textAutosizingEnabled()
671 || !document.settings().textAutosizingUsesIdempotentMode()
672 || document.settings().idempotentModeAutosizingOnlyHonorsPercentages())
673 return adjustmentForTextAutosizing;
674
675 auto newStatus = AutosizeStatus::computeStatus(style);
676 if (newStatus != style.autosizeStatus())
677 adjustmentForTextAutosizing.newStatus = newStatus;
678
679 if (style.textSizeAdjust().isNone())
680 return adjustmentForTextAutosizing;
681
682 float initialScale = document.page() ? document.page()->initialScaleIgnoringContentSize() : 1;
683 auto adjustLineHeightIfNeeded = [&](auto computedFontSize) {
684 auto lineHeight = style.specifiedLineHeight();
685 constexpr static unsigned eligibleFontSize = 12;
686 if (computedFontSize * initialScale >= eligibleFontSize)
687 return;
688
689 constexpr static float boostFactor = 1.25;
690 auto minimumLineHeight = boostFactor * computedFontSize;
691 if (!lineHeight.isFixed() || lineHeight.value() >= minimumLineHeight)
692 return;
693
694 if (AutosizeStatus::probablyContainsASmallFixedNumberOfLines(style))
695 return;
696
697 adjustmentForTextAutosizing.newLineHeight = minimumLineHeight;
698 };
699
700 auto fontDescription = style.fontDescription();
701 auto initialComputedFontSize = fontDescription.computedSize();
702 auto specifiedFontSize = fontDescription.specifiedSize();
703 bool isCandidate = style.isIdempotentTextAutosizingCandidate(newStatus);
704 if (!isCandidate && WTF::areEssentiallyEqual(initialComputedFontSize, specifiedFontSize))
705 return adjustmentForTextAutosizing;
706
707 auto adjustedFontSize = AutosizeStatus::idempotentTextSize(fontDescription.specifiedSize(), initialScale);
708 if (isCandidate && WTF::areEssentiallyEqual(initialComputedFontSize, adjustedFontSize))
709 return adjustmentForTextAutosizing;
710
711 if (!hasTextChild(element))
712 return adjustmentForTextAutosizing;
713
714 adjustmentForTextAutosizing.newFontSize = isCandidate ? adjustedFontSize : specifiedFontSize;
715
716 // FIXME: We should restore computed line height to its original value in the case where the element is not
717 // an idempotent text autosizing candidate; otherwise, if an element that is a text autosizing candidate contains
718 // children which are not autosized, the non-autosized content will end up with a boosted line height.
719 if (isCandidate)
720 adjustLineHeightIfNeeded(adjustedFontSize);
721
722 return adjustmentForTextAutosizing;
723}

725bool Adjuster::adjustForTextAutosizing(RenderStyle& style, const Element& element, AdjustmentForTextAutosizing adjustment)
726{
727 AutosizeStatus::updateStatus(style);
728 if (auto newFontSize = adjustment.newFontSize) {
729 auto fontDescription = style.fontDescription();
730 fontDescription.setComputedSize(*newFontSize);
731 style.setFontDescription(WTFMove(fontDescription));
732 style.fontCascade().update(&element.document().fontSelector());
733 }
734 if (auto newLineHeight = adjustment.newLineHeight)
735 style.setLineHeight({ *newLineHeight, LengthType::Fixed });
736 if (auto newStatus = adjustment.newStatus)
737 style.setAutosizeStatus(*newStatus);
738 return adjustment.newFontSize || adjustment.newLineHeight;
739}

741bool Adjuster::adjustForTextAutosizing(RenderStyle& style, const Element& element)
742{
743 return adjustForTextAutosizing(style, element, adjustmentForTextAutosizing(style, element));
744}
