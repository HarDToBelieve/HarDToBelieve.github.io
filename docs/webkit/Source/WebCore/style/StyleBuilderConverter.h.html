194inline Length BuilderConverter::convertLength(const BuilderState& builderState, const CSSValue& value)
195{
196 auto& primitiveValue = downcast<CSSPrimitiveValue>(value);
197 CSSToLengthConversionData conversionData = builderState.useSVGZoomRulesForLength() ?
198 builderState.cssToLengthConversionData().copyWithAdjustedZoom(1.0f)
199 : builderState.cssToLengthConversionData();
200
201 if (primitiveValue.isLength()) {
202 Length length = primitiveValue.computeLength<Length>(conversionData);
203 length.setHasQuirk(primitiveValue.primitiveType() == CSSUnitType::CSS_QUIRKY_EMS);
204 return length;
205 }
206
207 if (primitiveValue.isPercentage())
208 return Length(primitiveValue.doubleValue(), LengthType::Percent);
209
210 if (primitiveValue.isCalculatedPercentageWithLength())
211 return Length(primitiveValue.cssCalcValue()->createCalculationValue(conversionData));
212
213 ASSERT_NOT_REACHED();
214 return Length(0, LengthType::Fixed);
215}

224inline Length BuilderConverter::convertLengthSizing(const BuilderState& builderState, const CSSValue& value)
225{
226 auto& primitiveValue = downcast<CSSPrimitiveValue>(value);
227 switch (primitiveValue.valueID()) {
228 case CSSValueInvalid:
229 return convertLength(builderState, value);
230 case CSSValueIntrinsic:
231 return Length(LengthType::Intrinsic);
232 case CSSValueMinIntrinsic:
233 return Length(LengthType::MinIntrinsic);
234 case CSSValueMinContent:
235 case CSSValueWebkitMinContent:
236 return Length(LengthType::MinContent);
237 case CSSValueMaxContent:
238 case CSSValueWebkitMaxContent:
239 return Length(LengthType::MaxContent);
240 case CSSValueWebkitFillAvailable:
241 return Length(LengthType::FillAvailable);
242 case CSSValueFitContent:
243 case CSSValueWebkitFitContent:
244 return Length(LengthType::FitContent);
245 case CSSValueAuto:
246 return Length(LengthType::Auto);
247 default:
248 ASSERT_NOT_REACHED();
249 return Length();
250 }
251}

274template<typename T>
275inline T BuilderConverter::convertLineWidth(BuilderState& builderState, const CSSValue& value)
276{
277 auto& primitiveValue = downcast<CSSPrimitiveValue>(value);
278 switch (primitiveValue.valueID()) {
279 case CSSValueThin:
280 return 1;
281 case CSSValueMedium:
282 return 3;
283 case CSSValueThick:
284 return 5;
285 case CSSValueInvalid: {
286 // Any original result that was >= 1 should not be allowed to fall below 1.
287 // This keeps border lines from vanishing.
288 T result = convertComputedLength<T>(builderState, value);
289 if (builderState.style().effectiveZoom() < 1.0f && result < 1.0) {
290 T originalLength = primitiveValue.computeLength<T>(builderState.cssToLengthConversionData().copyWithAdjustedZoom(1.0));
291 if (originalLength >= 1.0)
292 return 1;
293 }
294 float minimumLineWidth = 1 / builderState.document().deviceScaleFactor();
295 if (result > 0 && result < minimumLineWidth)
296 return minimumLineWidth;
297 return floorToDevicePixel(result, builderState.document().deviceScaleFactor());
298 }
299 default:
300 ASSERT_NOT_REACHED();
301 return 0;
302 }
303}

326inline LengthSize BuilderConverter::convertRadius(BuilderState& builderState, const CSSValue& value)
327{
328 auto* pair = downcast<CSSPrimitiveValue>(value).pairValue();
329 if (!pair || !pair->first() || !pair->second())
330 return { { 0, LengthType::Fixed }, { 0, LengthType::Fixed } };
331
332 CSSToLengthConversionData conversionData = builderState.cssToLengthConversionData();
333 LengthSize radius { convertToRadiusLength(conversionData, *pair->first()), convertToRadiusLength(conversionData, *pair->second()) };
334
335 ASSERT(!radius.width.isNegative());
336 ASSERT(!radius.height.isNegative());
337 return radius;
338}

340inline Length BuilderConverter::convertTo100PercentMinusLength(const Length& length)
341{
342 if (length.isPercent())
343 return Length(100 - length.value(), LengthType::Percent);
344
345 // Turn this into a calc expression: calc(100% - length)
346 Vector<std::unique_ptr<CalcExpressionNode>> lengths;
347 lengths.reserveInitialCapacity(2);
348 lengths.uncheckedAppend(makeUnique<CalcExpressionLength>(Length(100, LengthType::Percent)));
349 lengths.uncheckedAppend(makeUnique<CalcExpressionLength>(length));
350 auto op = makeUnique<CalcExpressionOperation>(WTFMove(lengths), CalcOperator::Subtract);
351 return Length(CalculationValue::create(WTFMove(op), ValueRange::All));
352}

364template<CSSValueID cssValueFor0, CSSValueID cssValueFor100>
365inline Length BuilderConverter::convertPositionComponent(BuilderState& builderState, const CSSPrimitiveValue& value)
366{
367 Length length;
368
369 auto* lengthValue = &value;
370 bool relativeToTrailingEdge = false;
371
372 if (value.isPair()) {
373 auto& first = *value.pairValue()->first();
374 if (first.valueID() == CSSValueRight || first.valueID() == CSSValueBottom)
375 relativeToTrailingEdge = true;
376 lengthValue = value.pairValue()->second();
377 }
378
379 if (value.isValueID()) {
380 switch (value.valueID()) {
381 case cssValueFor0:
382 return Length(0, LengthType::Percent);
383 case cssValueFor100:
384 return Length(100, LengthType::Percent);
385 case CSSValueCenter:
386 return Length(50, LengthType::Percent);
387 default:
388 ASSERT_NOT_REACHED();
389 }
390 }
391
392 length = convertLength(builderState, *lengthValue);
393
394 if (relativeToTrailingEdge)
395 length = convertTo100PercentMinusLength(length);
396
397 return length;
398}

400inline LengthPoint BuilderConverter::convertObjectPosition(BuilderState& builderState, const CSSValue& value)
401{
402 auto& primitiveValue = downcast<CSSPrimitiveValue>(value);
403 Pair* pair = primitiveValue.pairValue();
404 if (!pair || !pair->first() || !pair->second())
405 return RenderStyle::initialObjectPosition();
406
407 Length lengthX = convertPositionComponent<CSSValueLeft, CSSValueRight>(builderState, *pair->first());
408 Length lengthY = convertPositionComponent<CSSValueTop, CSSValueBottom>(builderState, *pair->second());
409
410 return LengthPoint(lengthX, lengthY);
411}

498inline void BuilderConverter::updateColorScheme(const CSSPrimitiveValue& primitiveValue, StyleColorScheme& colorScheme)
499{
500 ASSERT(primitiveValue.isValueID());
501

503 case CSSValueAuto:

521inline StyleColorScheme BuilderConverter::convertColorScheme(BuilderState&, const CSSValue& value)
522{
523 StyleColorScheme colorScheme;
524
525 if (is<CSSValueList>(value)) {
526 for (auto& currentValue : downcast<CSSValueList>(value))
527 updateColorScheme(downcast<CSSPrimitiveValue>(currentValue.get()), colorScheme);
528 } else if (is<CSSPrimitiveValue>(value))
529 updateColorScheme(downcast<CSSPrimitiveValue>(value), colorScheme);
530
531 // If the value was just "only", that is synonymous for "only light".
532 if (colorScheme.isOnly())
533 colorScheme.add(ColorScheme::Light);
534
535 return colorScheme;
536}

590inline TextAlignMode BuilderConverter::convertTextAlign(BuilderState& builderState, const CSSValue& value)
591{
592 auto& primitiveValue = downcast<CSSPrimitiveValue>(value);
593 ASSERT(primitiveValue.isValueID());
594
595 if (primitiveValue.valueID() != CSSValueWebkitMatchParent)
596 return primitiveValue;
597
598 auto& parentStyle = builderState.parentStyle();
599 if (parentStyle.textAlign() == TextAlignMode::Start)
600 return parentStyle.isLeftToRightDirection() ? TextAlignMode::Left : TextAlignMode::Right;
601 if (parentStyle.textAlign() == TextAlignMode::End)
602 return parentStyle.isLeftToRightDirection() ? TextAlignMode::Right : TextAlignMode::Left;
603 return parentStyle.textAlign();
604}

606inline RefPtr<ClipPathOperation> BuilderConverter::convertClipPath(BuilderState& builderState, const CSSValue& value)
607{
608 if (is<CSSPrimitiveValue>(value)) {
609 auto& primitiveValue = downcast<CSSPrimitiveValue>(value);
610 if (primitiveValue.primitiveType() == CSSUnitType::CSS_URI) {
611 String cssURLValue = primitiveValue.stringValue();
612 String fragment = SVGURIReference::fragmentIdentifierFromIRIString(cssURLValue, builderState.document());
613 // FIXME: It doesn't work with external SVG references (see https://bugs.webkit.org/show_bug.cgi?id=126133)
614 return ReferenceClipPathOperation::create(cssURLValue, fragment);
615 }
616 ASSERT(primitiveValue.valueID() == CSSValueNone);
617 return nullptr;

992inline bool BuilderConverter::createGridTrackList(const CSSValue& value, TracksData& tracksData, BuilderState& builderState)
993{
994 // Handle 'none'.
995 if (is<CSSPrimitiveValue>(value))
996 return downcast<CSSPrimitiveValue>(value).valueID() == CSSValueNone;
997
998 if (!is<CSSValueList>(value))
999 return false;
1000
1001 unsigned currentNamedGridLine = 0;
1002 auto handleLineNameOrTrackSize = [&](const CSSValue& currentValue) {
1003 if (is<CSSGridLineNamesValue>(currentValue))
1004 createGridLineNamesList(currentValue, currentNamedGridLine, tracksData.m_namedGridLines, tracksData.m_orderedNamedGridLines);
1005 else {
1006 ++currentNamedGridLine;
1007 tracksData.m_trackSizes.append(createGridTrackSize(currentValue, builderState));
1008 }
1009 };
1010
1011 for (auto& currentValue : downcast<CSSValueList>(value)) {
1012 if (is<CSSGridAutoRepeatValue>(currentValue)) {
1013 ASSERT(tracksData.m_autoRepeatTrackSizes.isEmpty());
1014 unsigned autoRepeatIndex = 0;
1015 CSSValueID autoRepeatID = downcast<CSSGridAutoRepeatValue>(currentValue.get()).autoRepeatID();
1016 ASSERT(autoRepeatID == CSSValueAutoFill || autoRepeatID == CSSValueAutoFit);
1017 tracksData.m_autoRepeatType = autoRepeatID == CSSValueAutoFill ? AutoRepeatType::Fill : AutoRepeatType::Fit;
1018 for (auto& autoRepeatValue : downcast<CSSValueList>(currentValue.get())) {
1019 if (is<CSSGridLineNamesValue>(autoRepeatValue)) {
1020 createGridLineNamesList(autoRepeatValue.get(), autoRepeatIndex, tracksData.m_autoRepeatNamedGridLines, tracksData.m_autoRepeatOrderedNamedGridLines);
1021 continue;
1022 }
1023 ++autoRepeatIndex;
1024 tracksData.m_autoRepeatTrackSizes.append(createGridTrackSize(autoRepeatValue.get(), builderState));
1025 }
1026 tracksData.m_autoRepeatInsertionPoint = currentNamedGridLine++;
1027 continue;
1028 }
1029
1030 if (is<CSSGridIntegerRepeatValue>(currentValue)) {
1031 size_t repetitions = downcast<CSSGridIntegerRepeatValue>(currentValue.get()).repetitions();
1032 for (size_t i = 0; i < repetitions; ++i) {
1033 for (auto& integerRepeatValue : downcast<CSSValueList>(currentValue.get()))
1034 handleLineNameOrTrackSize(integerRepeatValue);
1035 }
1036 continue;
1037 }
1038
1039 handleLineNameOrTrackSize(currentValue);
1040 }
1041
1042 // The parser should have rejected any <track-list> without any <track-size> as
1043 // this is not conformant to the syntax.
1044 ASSERT(!tracksData.m_trackSizes.isEmpty() || !tracksData.m_autoRepeatTrackSizes.isEmpty());
1045 return true;
1046}

1048inline bool BuilderConverter::createGridPosition(const CSSValue& value, GridPosition& position)
1049{
1050 // We accept the specification's grammar:
1051 // auto | <custom-ident> | [ <integer> && <custom-ident>? ] | [ span && [ <integer> || <custom-ident> ] ]
1052 if (is<CSSPrimitiveValue>(value)) {
1053 auto& primitiveValue = downcast<CSSPrimitiveValue>(value);
1054 if (primitiveValue.isCustomIdent()) {
1055 position.setNamedGridArea(primitiveValue.stringValue());
1056 return true;
1057 }
1058
1059 ASSERT(primitiveValue.valueID() == CSSValueAuto);
1060 return true;
1061 }
1062
1063 auto& values = downcast<CSSValueList>(value);
1064 ASSERT(values.length());
1065
1066 auto it = values.begin();
1067 const CSSPrimitiveValue* currentValue = &downcast<CSSPrimitiveValue>(it->get());
1068 bool isSpanPosition = false;
1069 if (currentValue->valueID() == CSSValueSpan) {
1070 isSpanPosition = true;
1071 ++it;
1072 currentValue = it != values.end() ? &downcast<CSSPrimitiveValue>(it->get()) : nullptr;
1073 }
1074
1075 int gridLineNumber = 0;
1076 if (currentValue && currentValue->isNumber()) {
1077 gridLineNumber = currentValue->intValue();
1078 ++it;
1079 currentValue = it != values.end() ? &downcast<CSSPrimitiveValue>(it->get()) : nullptr;
1080 }
1081
1082 String gridLineName;
1083 if (currentValue && currentValue->isCustomIdent()) {
1084 gridLineName = currentValue->stringValue();
1085 ++it;
1086 }
1087
1088 ASSERT(it == values.end());
1089 if (isSpanPosition)
1090 position.setSpanPosition(gridLineNumber ? gridLineNumber : 1, gridLineName);
1091 else
1092 position.setExplicitPosition(gridLineNumber, gridLineName);
1093
1094 return true;
1095}

1524inline std::optional<Length> BuilderConverter::convertLineHeight(BuilderState& builderState, const CSSValue& value, float multiplier)
1525{
1526 auto& primitiveValue = downcast<CSSPrimitiveValue>(value);
1527 if (primitiveValue.valueID() == CSSValueNormal)
1528 return RenderStyle::initialLineHeight();
1529
1530 if (primitiveValue.isLength()) {
1531 auto conversionData = builderState.cssToLengthConversionData().copyWithAdjustedZoomAndPropertyToCompute(zoomWithTextZoomFactor(builderState), CSSPropertyLineHeight);
1532 Length length = primitiveValue.computeLength<Length>(conversionData);
1533 if (multiplier != 1.f)
1534 length = Length(length.value() * multiplier, LengthType::Fixed);
1535 return length;
1536 }
1537
1538 // Line-height percentages need to inherit as if they were Fixed pixel values. In the example:
1539 // <div style="font-size: 10px; line-height: 150%;"><div style="font-size: 100px;"></div></div>
1540 // the inner element should have line-height of 15px. However, in this example:
1541 // <div style="font-size: 10px; line-height: 1.5;"><div style="font-size: 100px;"></div></div>
1542 // the inner element should have a line-height of 150px. Therefore, we map percentages to Fixed
1543 // values and raw numbers to percentages.
1544 if (primitiveValue.isPercentage()) {
1545 // FIXME: percentage should not be restricted to an integer here.
1546 return Length((builderState.style().computedFontSize() * primitiveValue.intValue()) / 100, LengthType::Fixed);
1547 }
1548 if (primitiveValue.isNumber())
1549 return Length(primitiveValue.doubleValue() * 100.0, LengthType::Percent);
1550
1551 // FIXME: The parser should only emit the above types, so this should never be reached. We should change the
1552 // type of this function to return just a Length (and not an Optional).
1553 return std::nullopt;
