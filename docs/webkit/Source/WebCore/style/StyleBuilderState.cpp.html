67// SVG handles zooming in a different way compared to CSS. The whole document is scaled instead
68// of each individual length value in the render style / tree. CSSPrimitiveValue::computeLength*()
69// multiplies each resolved length with the zoom multiplier - so for SVG we need to disable that.
70// Though all CSS values that can be applied to outermost <svg> elements (width/height/border/padding...)
71// need to respect the scaling. RenderBox (the parent class of RenderSVGRoot) grabs values like
72// width/height/border/padding/... from the RenderStyle -> for SVG these values would never scale,
73// if we'd pass a 1.0 zoom factor everyhwere. So we only pass a zoom factor of 1.0 for specific
74// properties that are NOT allowed to scale within a zoomed SVG document (letter/word-spacing/font-size).
75bool BuilderState::useSVGZoomRules() const
76{
77 return is<SVGElement>(element());
78}

80bool BuilderState::useSVGZoomRulesForLength() const
81{
82 return is<SVGElement>(element()) && !(is<SVGSVGElement>(*element()) && element()->parentNode());
83}

85Ref<CSSValue> BuilderState::resolveImageStyles(CSSValue& value)
86{
87 if (is<CSSCrossfadeValue>(value))
88 return downcast<CSSCrossfadeValue>(value).valueWithStylesResolved(*this);
89 if (is<CSSCursorImageValue>(value))
90 return downcast<CSSCursorImageValue>(value).valueWithStylesResolved(*this);
91 if (is<CSSFilterImageValue>(value))
92 return downcast<CSSFilterImageValue>(value).valueWithStylesResolved(*this);
93 if (is<CSSGradientValue>(value))
94 return downcast<CSSGradientValue>(value).valueWithStylesResolved(*this);
95 if (is<CSSImageSetValue>(value))
96 return downcast<CSSImageSetValue>(value).valueWithStylesResolved(*this);
97 if (is<CSSImageValue>(value))
98 return downcast<CSSImageValue>(value).valueWithStylesResolved(*this);
99 return makeRef(value);
100}

102RefPtr<StyleImage> BuilderState::createStyleImage(CSSValue& value)
103{
104 if (is<CSSImageValue>(value))
105 return StyleCachedImage::create(downcast<CSSImageValue>(resolveImageStyles(value).get()));
106 if (is<CSSCursorImageValue>(value))
107 return StyleCursorImage::create(downcast<CSSCursorImageValue>(resolveImageStyles(value).get()));
108 if (is<CSSImageGeneratorValue>(value))
109 return StyleGeneratedImage::create(downcast<CSSImageGeneratorValue>(resolveImageStyles(value).get()));
110 if (is<CSSImageSetValue>(value))
111 return StyleImageSet::create(downcast<CSSImageSetValue>(resolveImageStyles(value).get()));
112 return nullptr;
113}

115static FilterOperation::OperationType filterOperationForType(CSSValueID type)
116{
117 switch (type) {
118 case CSSValueUrl:
119 return FilterOperation::REFERENCE;
120 case CSSValueGrayscale:
121 return FilterOperation::GRAYSCALE;
122 case CSSValueSepia:
123 return FilterOperation::SEPIA;
124 case CSSValueSaturate:
125 return FilterOperation::SATURATE;
126 case CSSValueHueRotate:
127 return FilterOperation::HUE_ROTATE;
128 case CSSValueInvert:
129 return FilterOperation::INVERT;
130 case CSSValueAppleInvertLightness:
131 return FilterOperation::APPLE_INVERT_LIGHTNESS;
132 case CSSValueOpacity:
133 return FilterOperation::OPACITY;
134 case CSSValueBrightness:
135 return FilterOperation::BRIGHTNESS;
136 case CSSValueContrast:
137 return FilterOperation::CONTRAST;
138 case CSSValueBlur:
139 return FilterOperation::BLUR;
140 case CSSValueDropShadow:
141 return FilterOperation::DROP_SHADOW;
142 default:
143 break;
144 }
145 ASSERT_NOT_REACHED();
146 return FilterOperation::NONE;
147}

149bool BuilderState::createFilterOperations(const CSSValue& inValue, FilterOperations& outOperations)
150{
151 // FIXME: Move this code somewhere else.
152
153 ASSERT(outOperations.isEmpty());
154
155 if (is<CSSPrimitiveValue>(inValue)) {
156 auto& primitiveValue = downcast<CSSPrimitiveValue>(inValue);
157 if (primitiveValue.valueID() == CSSValueNone)
158 return true;
159 }
160
161 if (!is<CSSValueList>(inValue))
162 return false;
163
164 FilterOperations operations;
165 for (auto& currentValue : downcast<CSSValueList>(inValue)) {
166 if (is<CSSPrimitiveValue>(currentValue)) {
167 auto& primitiveValue = downcast<CSSPrimitiveValue>(currentValue.get());
168 if (!primitiveValue.isURI())
169 continue;
170
171 auto filterURL = primitiveValue.stringValue();
172 auto fragment = document().completeURL(filterURL).fragmentIdentifier().toString();
173 operations.operations().append(ReferenceFilterOperation::create(filterURL, fragment));
174 continue;
175 }
176
177 if (!is<CSSFunctionValue>(currentValue))
178 continue;
179
180 auto& filterValue = downcast<CSSFunctionValue>(currentValue.get());
181 FilterOperation::OperationType operationType = filterOperationForType(filterValue.name());
182
183 // Check that all parameters are primitive values, with the
184 // exception of drop shadow which has a CSSShadowValue parameter.
185 const CSSPrimitiveValue* firstValue = nullptr;
186 if (operationType != FilterOperation::DROP_SHADOW) {
187 bool haveNonPrimitiveValue = false;
188 for (unsigned j = 0; j < filterValue.length(); ++j) {
189 if (!is<CSSPrimitiveValue>(*filterValue.itemWithoutBoundsCheck(j))) {
190 haveNonPrimitiveValue = true;
191 break;
192 }
193 }
194 if (haveNonPrimitiveValue)
195 continue;
196 if (filterValue.length())
197 firstValue = downcast<CSSPrimitiveValue>(filterValue.itemWithoutBoundsCheck(0));
198 }
199
200 switch (operationType) {
201 case FilterOperation::GRAYSCALE:
202 case FilterOperation::SEPIA:
203 case FilterOperation::SATURATE: {
204 double amount = 1;
205 if (filterValue.length() == 1) {
206 amount = firstValue->doubleValue();
207 if (firstValue->isPercentage())
208 amount /= 100;
209 }
210
211 operations.operations().append(BasicColorMatrixFilterOperation::create(amount, operationType));
212 break;
213 }
214 case FilterOperation::HUE_ROTATE: {
215 double angle = 0;
216 if (filterValue.length() == 1)
217 angle = firstValue->computeDegrees();
218
219 operations.operations().append(BasicColorMatrixFilterOperation::create(angle, operationType));
220 break;
221 }
222 case FilterOperation::INVERT:
223 case FilterOperation::BRIGHTNESS:
224 case FilterOperation::CONTRAST:
225 case FilterOperation::OPACITY: {
226 double amount = 1;
227 if (filterValue.length() == 1) {
228 amount = firstValue->doubleValue();
229 if (firstValue->isPercentage())
230 amount /= 100;
231 }
232
233 operations.operations().append(BasicComponentTransferFilterOperation::create(amount, operationType));
234 break;
235 }
236 case FilterOperation::APPLE_INVERT_LIGHTNESS: {
237 operations.operations().append(InvertLightnessFilterOperation::create());
238 break;
239 }
240 case FilterOperation::BLUR: {
241 Length stdDeviation = Length(0, LengthType::Fixed);
242 if (filterValue.length() >= 1)
243 stdDeviation = convertToFloatLength(firstValue, cssToLengthConversionData());
244 if (stdDeviation.isUndefined())
245 return false;
246
247 operations.operations().append(BlurFilterOperation::create(stdDeviation));
248 break;
249 }
250 case FilterOperation::DROP_SHADOW: {
251 if (filterValue.length() != 1)
252 return false;
253
254 const auto* cssValue = filterValue.itemWithoutBoundsCheck(0);
255 if (!is<CSSShadowValue>(cssValue))
256 continue;
257
258 const auto& item = downcast<CSSShadowValue>(*cssValue);
259 int x = item.x->computeLength<int>(cssToLengthConversionData());
260 int y = item.y->computeLength<int>(cssToLengthConversionData());
261 IntPoint location(x, y);
262 int blur = item.blur ? item.blur->computeLength<int>(cssToLengthConversionData()) : 0;
263 Color color;
264 if (item.color)
265 color = colorFromPrimitiveValueWithResolvedCurrentColor(*item.color);
266
267 operations.operations().append(DropShadowFilterOperation::create(location, blur, color.isValid() ? color : Color::transparentBlack));
268 break;
269 }
270 default:
271 ASSERT_NOT_REACHED();
272 break;
273 }
274 }
275
276 outOperations = operations;
277 return true;
278}

293Color BuilderState::colorFromPrimitiveValue(const CSSPrimitiveValue& value, bool forVisitedLink) const
294{
295 if (value.isRGBColor())
296 return value.color();
297
298 auto identifier = value.valueID();
299 switch (identifier) {
300 case CSSValueWebkitText:
301 return document().textColor();
302 case CSSValueWebkitLink:
303 return (element() && element()->isLink() && forVisitedLink) ? document().visitedLinkColor() : document().linkColor();
304 case CSSValueWebkitActivelink:
305 return document().activeLinkColor();
306 case CSSValueWebkitFocusRingColor:
307 return RenderTheme::singleton().focusRingColor(document().styleColorOptions(&m_style));
308 case CSSValueCurrentcolor:
309 return RenderStyle::currentColor();
310 default:
311 return StyleColor::colorFromKeyword(identifier, document().styleColorOptions(&m_style));
312 }
313}

315Color BuilderState::colorFromPrimitiveValueWithResolvedCurrentColor(const CSSPrimitiveValue& value) const
316{
317 // FIXME: 'currentcolor' should be resolved at use time to make it inherit correctly. https://bugs.webkit.org/show_bug.cgi?id=210005
318 if (value.valueID() == CSSValueCurrentcolor) {
319 // Color is an inherited property so depending on it effectively makes the property inherited.
320 m_style.setHasExplicitlyInheritedProperties();
321 return m_style.color();
322 }
323
324 return colorFromPrimitiveValue(value);
325}

333void BuilderState::adjustStyleForInterCharacterRuby()
334{
335 if (m_style.rubyPosition() != RubyPosition::InterCharacter || !element() || !element()->hasTagName(HTMLNames::rtTag))
336 return;

343void BuilderState::updateFont()
344{
345 auto& fontSelector = const_cast<Document&>(document()).fontSelector();
346
347 auto needsUpdate = [&] {
348 if (m_fontDirty)
349 return true;
350 auto* fonts = m_style.fontCascade().fonts();
351 if (!fonts)
352 return true;
353 return false;
354 };
355
356 if (!needsUpdate())
357 return;
358
359#if ENABLE(TEXT_AUTOSIZING)
360 updateFontForTextSizeAdjust();
361#endif
362 updateFontForGenericFamilyChange();
363 updateFontForZoomChange();
364 updateFontForOrientationChange();
365
366 m_style.fontCascade().update(&fontSelector);
367
368 m_fontDirty = false;
369}

371#if ENABLE(TEXT_AUTOSIZING)
372void BuilderState::updateFontForTextSizeAdjust()
373{
374 if (m_style.textSizeAdjust().isAuto()
375 || !document().settings().textAutosizingEnabled()
376 || (document().settings().textAutosizingUsesIdempotentMode()
377 && !m_style.textSizeAdjust().isNone()
378 && !document().settings().idempotentModeAutosizingOnlyHonorsPercentages()))
379 return;
380
381 auto newFontDescription = m_style.fontDescription();
382 if (!m_style.textSizeAdjust().isNone())
383 newFontDescription.setComputedSize(newFontDescription.specifiedSize() * m_style.textSizeAdjust().multiplier());
384 else
385 newFontDescription.setComputedSize(newFontDescription.specifiedSize());
386
387 m_style.setFontDescription(WTFMove(newFontDescription));
388}
389#endif

391void BuilderState::updateFontForZoomChange()
392{
393 if (m_style.effectiveZoom() == parentStyle().effectiveZoom() && m_style.textZoom() == parentStyle().textZoom())
394 return;
395
396 const auto& childFont = m_style.fontDescription();
397 auto newFontDescription = childFont;
398 setFontSize(newFontDescription, childFont.specifiedSize());
399
400 m_style.setFontDescription(WTFMove(newFontDescription));
401}

403void BuilderState::updateFontForGenericFamilyChange()
404{
405 const auto& childFont = m_style.fontDescription();
406
407 if (childFont.isAbsoluteSize())
408 return;
409
410 const auto& parentFont = parentStyle().fontDescription();
411 if (childFont.useFixedDefaultSize() == parentFont.useFixedDefaultSize())
412 return;
413
414 // We know the parent is monospace or the child is monospace, and that font
415 // size was unspecified. We want to scale our font size as appropriate.
416 // If the font uses a keyword size, then we refetch from the table rather than
417 // multiplying by our scale factor.
418 float size = [&] {
419 if (CSSValueID sizeIdentifier = childFont.keywordSizeAsIdentifier())
420 return Style::fontSizeForKeyword(sizeIdentifier, childFont.useFixedDefaultSize(), document());
421
422 auto fixedSize = document().settings().defaultFixedFontSize();
423 auto defaultSize = document().settings().defaultFontSize();
424 float fixedScaleFactor = (fixedSize && defaultSize) ? static_cast<float>(fixedSize) / defaultSize : 1;
425 return parentFont.useFixedDefaultSize() ? childFont.specifiedSize() / fixedScaleFactor : childFont.specifiedSize() * fixedScaleFactor;
426 }();
427
428 auto newFontDescription = childFont;
429 setFontSize(newFontDescription, size);
430 m_style.setFontDescription(WTFMove(newFontDescription));
431}
432
433void BuilderState::updateFontForOrientationChange()
434{
435 auto [fontOrientation, glyphOrientation] = m_style.fontAndGlyphOrientation();
436
437 const auto& fontDescription = m_style.fontDescription();
438 if (fontDescription.orientation() == fontOrientation && fontDescription.nonCJKGlyphOrientation() == glyphOrientation)
439 return;
440
441 auto newFontDescription = fontDescription;
442 newFontDescription.setNonCJKGlyphOrientation(glyphOrientation);
443 newFontDescription.setOrientation(fontOrientation);
444 m_style.setFontDescription(WTFMove(newFontDescription));
445}

447void BuilderState::setFontSize(FontCascadeDescription& fontDescription, float size)
448{
449 fontDescription.setSpecifiedSize(size);
450 fontDescription.setComputedSize(Style::computedFontSizeFromSpecifiedSize(size, fontDescription.isAbsoluteSize(), useSVGZoomRules(), &style(), document()));
451}
