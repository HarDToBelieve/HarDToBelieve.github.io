42class Builder;
43class BuilderState;
44
45void maybeUpdateFontForLetterSpacing(BuilderState&, CSSValue&);
46
47struct BuilderContext {
48 Ref<const Document> document;
49 const RenderStyle& parentStyle;
50 const RenderStyle* rootElementStyle = nullptr;
51 RefPtr<const Element> element = nullptr;
52};
53
54class BuilderState {
55public:
56 BuilderState(Builder&, RenderStyle&, BuilderContext&&);
57
58 Builder& builder() { return m_builder; }
59
60 RenderStyle& style() { return m_style; }
61 const RenderStyle& parentStyle() const { return m_context.parentStyle; }
62 const RenderStyle* rootElementStyle() const { return m_context.rootElementStyle; }
63
64 const Document& document() const { return m_context.document.get(); }
65 const Element* element() const { return m_context.element.get(); }
66
67 void setFontDescription(FontCascadeDescription&& fontDescription) { m_fontDirty |= m_style.setFontDescription(WTFMove(fontDescription)); }
68 void setFontSize(FontCascadeDescription&, float size);
69 void setZoom(float f) { m_fontDirty |= m_style.setZoom(f); }
70 void setEffectiveZoom(float f) { m_fontDirty |= m_style.setEffectiveZoom(f); }
71 void setWritingMode(WritingMode writingMode) { m_fontDirty |= m_style.setWritingMode(writingMode); }
72 void setTextOrientation(TextOrientation textOrientation) { m_fontDirty |= m_style.setTextOrientation(textOrientation); }
73
74 bool fontDirty() const { return m_fontDirty; }
75 void setFontDirty() { m_fontDirty = true; }
76
77 const FontCascadeDescription& fontDescription() { return m_style.fontDescription(); }
78 const FontCascadeDescription& parentFontDescription() { return parentStyle().fontDescription(); }
79
80 // FIXME: These are mutually exclusive, clean up the code to take that into account.
81 bool applyPropertyToRegularStyle() const { return m_linkMatch != SelectorChecker::MatchVisited; }
82 bool applyPropertyToVisitedLinkStyle() const { return m_linkMatch == SelectorChecker::MatchVisited; }
83
84 bool useSVGZoomRules() const;
85 bool useSVGZoomRulesForLength() const;
86 ScopeOrdinal styleScopeOrdinal() const { return m_styleScopeOrdinal; }
87
88 Ref<CSSValue> resolveImageStyles(CSSValue&);
89 RefPtr<StyleImage> createStyleImage(CSSValue&);
90 bool createFilterOperations(const CSSValue&, FilterOperations& outOperations);
91
92 static bool isColorFromPrimitiveValueDerivedFromElement(const CSSPrimitiveValue&);
93 Color colorFromPrimitiveValue(const CSSPrimitiveValue&, bool forVisitedLink = false) const;
94 // FIXME: Remove. 'currentcolor' should be resolved at use time. All call sites are broken with inheritance.
95 Color colorFromPrimitiveValueWithResolvedCurrentColor(const CSSPrimitiveValue&) const;
96
97 const Vector<AtomString>& registeredContentAttributes() const { return m_registeredContentAttributes; }
98 void registerContentAttribute(const AtomString& attributeLocalName);
99
100 const CSSToLengthConversionData& cssToLengthConversionData() const { return m_cssToLengthConversionData; }
101 CSSToStyleMap& styleMap() { return m_styleMap; }
102
103private:
104 // See the comment in maybeUpdateFontForLetterSpacing() about why this needs to be a friend.
105 friend void maybeUpdateFontForLetterSpacing(BuilderState&, CSSValue&);
106 friend class Builder;
107
108 void adjustStyleForInterCharacterRuby();
109
110 void updateFont();
111#if ENABLE(TEXT_AUTOSIZING)
112 void updateFontForTextSizeAdjust();
113#endif
114 void updateFontForZoomChange();
115 void updateFontForGenericFamilyChange();
116 void updateFontForOrientationChange();
117
118 Builder& m_builder;
119
120 CSSToStyleMap m_styleMap;
121
122 RenderStyle& m_style;
123 const BuilderContext m_context;
124
125 const CSSToLengthConversionData m_cssToLengthConversionData;
126
127 Bitmap<numCSSProperties> m_appliedProperties;
128 HashSet<String> m_appliedCustomProperties;
129 Bitmap<numCSSProperties> m_inProgressProperties;
130 HashSet<String> m_inProgressPropertiesCustom;
131
132 CascadeLevel m_cascadeLevel { };
133 ScopeOrdinal m_styleScopeOrdinal { };
134 SelectorChecker::LinkMatchMask m_linkMatch { };
135
136 bool m_fontDirty { false };
137 Vector<AtomString> m_registeredContentAttributes;
138};
