48static bool shouldDirtyAllStyle(const Vector<RefPtr<StyleRuleBase>>& rules)
49{
50 for (auto& rule : rules) {
51 if (is<StyleRuleMedia>(*rule)) {
52 const auto* childRules = downcast<StyleRuleMedia>(*rule).childRulesWithoutDeferredParsing();
53 if (childRules && shouldDirtyAllStyle(*childRules))
54 return true;
55 continue;
56 }
57 // FIXME: At least font faces don't need full recalc in all cases.
58 if (!is<StyleRule>(*rule))
59 return true;
60 }
61 return false;
62}
63
64static bool shouldDirtyAllStyle(const StyleSheetContents& sheet)
65{
66 for (auto& import : sheet.importRules()) {
67 if (!import->styleSheet())
68 continue;
69 if (shouldDirtyAllStyle(*import->styleSheet()))
70 return true;
71 }
72 if (shouldDirtyAllStyle(sheet.childRules()))
73 return true;
74 return false;
75}
76
77static bool shouldDirtyAllStyle(const Vector<StyleSheetContents*>& sheets)
78{
79 for (auto& sheet : sheets) {
80 if (shouldDirtyAllStyle(*sheet))
81 return true;
82 }
83 return false;
84}

86Invalidator::Invalidator(const Vector<StyleSheetContents*>& sheets, const MediaQueryEvaluator& mediaQueryEvaluator)
87 : m_ownedRuleSet(RuleSet::create())
88 , m_ruleSets({ m_ownedRuleSet })
89 , m_dirtiesAllStyle(shouldDirtyAllStyle(sheets))
90{
91 if (m_dirtiesAllStyle)
92 return;
93
94 m_ownedRuleSet->disableAutoShrinkToFit();
95 for (auto& sheet : sheets)
96 m_ownedRuleSet->addRulesFromSheet(*sheet, mediaQueryEvaluator);
97
98 m_ruleInformation = collectRuleInformation();
99}

110Invalidator::RuleInformation Invalidator::collectRuleInformation()
111{
112 RuleInformation information;
113 for (auto& ruleSet : m_ruleSets) {
114 if (!ruleSet->slottedPseudoElementRules().isEmpty())
115 information.hasSlottedPseudoElementRules = true;
116 if (!ruleSet->hostPseudoClassRules().isEmpty())
117 information.hasHostPseudoClassRules = true;
118 if (ruleSet->hasShadowPseudoElementRules())
119 information.hasShadowPseudoElementRules = true;
120#if ENABLE(VIDEO)
121 if (!ruleSet->cuePseudoRules().isEmpty())
122 information.hasCuePseudoElementRules = true;
123#endif
124 if (!ruleSet->partPseudoElementRules().isEmpty())
125 information.hasPartPseudoElementRules = true;
126 }
127 return information;
128}

130Invalidator::CheckDescendants Invalidator::invalidateIfNeeded(Element& element, const SelectorFilter* filter)
131{
132 invalidateInShadowTreeIfNeeded(element);
133
134 bool shouldCheckForSlots = m_ruleInformation.hasSlottedPseudoElementRules && !m_didInvalidateHostChildren;
135 if (shouldCheckForSlots && is<HTMLSlotElement>(element)) {
136 auto* containingShadowRoot = element.containingShadowRoot();
137 if (containingShadowRoot && containingShadowRoot->host()) {
138 for (auto& possiblySlotted : childrenOfType<Element>(*containingShadowRoot->host()))
139 possiblySlotted.invalidateStyleInternal();
140 }
141 // No need to do this again.
142 m_didInvalidateHostChildren = true;
143 }
144
145 switch (element.styleValidity()) {
146 case Style::Validity::Valid: {
147 for (auto& ruleSet : m_ruleSets) {
148 ElementRuleCollector ruleCollector(element, *ruleSet, filter);
149 ruleCollector.setMode(SelectorChecker::Mode::CollectingRulesIgnoringVirtualPseudoElements);
150
151 if (ruleCollector.matchesAnyAuthorRules()) {
152 element.invalidateStyleInternal();
153 break;
154 }
155 }
156
157 return CheckDescendants::Yes;
158 }
159 case Style::Validity::ElementInvalid:
160 return CheckDescendants::Yes;
161 case Style::Validity::SubtreeInvalid:
162 case Style::Validity::SubtreeAndRenderersInvalid:
163 if (shouldCheckForSlots)
164 return CheckDescendants::Yes;
165 return CheckDescendants::No;
166 }
167 ASSERT_NOT_REACHED();
168 return CheckDescendants::Yes;
169}

171void Invalidator::invalidateStyleForTree(Element& root, SelectorFilter* filter)
172{
173 if (invalidateIfNeeded(root, filter) == CheckDescendants::No)
174 return;
175 invalidateStyleForDescendants(root, filter);
176}
177
178void Invalidator::invalidateStyleForDescendants(Element& root, SelectorFilter* filter)
179{
180 Vector<Element*, 20> parentStack;
181 Element* previousElement = &root;
182 for (auto it = descendantsOfType<Element>(root).begin(); it; ) {
183 auto& descendant = *it;
184 auto* parent = descendant.parentElement();
185 if (parentStack.isEmpty() || parentStack.last() != parent) {
186 if (parent == previousElement) {
187 parentStack.append(parent);
188 if (filter)
189 filter->pushParentInitializingIfNeeded(*parent);
190 } else {
191 while (parentStack.last() != parent) {
192 parentStack.removeLast();
193 if (filter)
194 filter->popParent();
195 }
196 }
197 }
198 previousElement = &descendant;
199
200 if (invalidateIfNeeded(descendant, filter) == CheckDescendants::Yes)
201 it.traverseNext();
202 else
203 it.traverseNextSkippingChildren();
204 }
205}

207void Invalidator::invalidateStyle(Document& document)
208{
209 ASSERT(!m_dirtiesAllStyle);
210
211 Element* documentElement = document.documentElement();
212 if (!documentElement)
213 return;
214
215 SelectorFilter filter;
216 invalidateStyleForTree(*documentElement, &filter);
217}

219void Invalidator::invalidateStyle(Scope& scope)
220{
221 if (m_dirtiesAllStyle) {
222 invalidateAllStyle(scope);
223 return;
224 }
225
226 if (auto* shadowRoot = scope.shadowRoot()) {
227 invalidateStyle(*shadowRoot);
228 return;
229 }
230
231 invalidateStyle(scope.document());
232}

234void Invalidator::invalidateStyle(ShadowRoot& shadowRoot)
235{
236 ASSERT(!m_dirtiesAllStyle);
237
238 if (m_ruleInformation.hasHostPseudoClassRules && shadowRoot.host())
239 shadowRoot.host()->invalidateStyleInternal();
240
241 for (auto& child : childrenOfType<Element>(shadowRoot)) {
242 SelectorFilter filter;
243 invalidateStyleForTree(child, &filter);
244 }
245}

247void Invalidator::invalidateStyle(Element& element)
248{
249 ASSERT(!m_dirtiesAllStyle);
250
251 // Don't use SelectorFilter as the rule sets here tend to be small and the filter would have setup cost deep in the tree.
252 invalidateStyleForTree(element, nullptr);
253}

255void Invalidator::invalidateStyleWithMatchElement(Element& element, MatchElement matchElement)
256{
257 switch (matchElement) {
258 case MatchElement::Subject: {
259 invalidateIfNeeded(element, nullptr);
260 break;
261 }
262 case MatchElement::Parent: {
263 auto children = childrenOfType<Element>(element);
264 for (auto& child : children)
265 invalidateIfNeeded(child, nullptr);
266 break;
267 }
268 case MatchElement::Ancestor: {
269 SelectorFilter filter;
270 invalidateStyleForDescendants(element, &filter);
271 break;
272 }
273 case MatchElement::DirectSibling:
274 if (auto* sibling = element.nextElementSibling())
275 invalidateIfNeeded(*sibling, nullptr);
276 break;
277 case MatchElement::IndirectSibling:
278 for (auto* sibling = element.nextElementSibling(); sibling; sibling = sibling->nextElementSibling())
279 invalidateIfNeeded(*sibling, nullptr);
280 break;
281 case MatchElement::AnySibling:
282 for (auto& parentChild : childrenOfType<Element>(*element.parentNode()))
283 invalidateIfNeeded(parentChild, nullptr);
284 break;
285 case MatchElement::ParentSibling:
286 for (auto* sibling = element.nextElementSibling(); sibling; sibling = sibling->nextElementSibling()) {
287 auto siblingChildren = childrenOfType<Element>(*sibling);
288 for (auto& siblingChild : siblingChildren)
289 invalidateIfNeeded(siblingChild, nullptr);
290 }
291 break;
292 case MatchElement::AncestorSibling: {
293 SelectorFilter filter;
294 for (auto* sibling = element.nextElementSibling(); sibling; sibling = sibling->nextElementSibling()) {
295 filter.popParentsUntil(element.parentElement());
296 invalidateStyleForDescendants(*sibling, &filter);
297 }
298 break;
299 }
300 case MatchElement::Host:
301 invalidateInShadowTreeIfNeeded(element);
302 break;
303 }
304}

306void Invalidator::invalidateShadowParts(ShadowRoot& shadowRoot)
307{
308 if (shadowRoot.mode() == ShadowRootMode::UserAgent)
309 return;
310
311 for (auto& descendant : descendantsOfType<Element>(shadowRoot)) {
312 // FIXME: We could only invalidate part names that actually show up in rules.
313 if (!descendant.partNames().isEmpty())
314 descendant.invalidateStyleInternal();
315
316 auto* nestedShadowRoot = descendant.shadowRoot();
317 if (nestedShadowRoot && !nestedShadowRoot->partMappings().isEmpty())
318 invalidateShadowParts(*nestedShadowRoot);
319 }
320}

322void Invalidator::invalidateShadowPseudoElements(ShadowRoot& shadowRoot)
323{
324 if (shadowRoot.mode() != ShadowRootMode::UserAgent)
325 return;
326
327 for (auto& descendant : descendantsOfType<Element>(shadowRoot)) {
328 auto& shadowPseudoId = descendant.shadowPseudoId();
329 if (!shadowPseudoId)
330 continue;
331 for (auto& ruleSet : m_ruleSets) {
332 if (ruleSet->shadowPseudoElementRules(shadowPseudoId))
333 descendant.invalidateStyleInternal();
334 }
335 }
336}

338void Invalidator::invalidateInShadowTreeIfNeeded(Element& element)
339{
340 auto* shadowRoot = element.shadowRoot();
341 if (!shadowRoot)
342 return;
343
344 if (m_ruleInformation.hasShadowPseudoElementRules)
345 invalidateShadowPseudoElements(*shadowRoot);
346
347#if ENABLE(VIDEO)
348 if (m_ruleInformation.hasCuePseudoElementRules && element.isMediaElement())
349 element.invalidateStyleForSubtreeInternal();
350#endif
351
352 // FIXME: More fine-grained invalidation for ::part()
353 if (m_ruleInformation.hasPartPseudoElementRules)
354 invalidateShadowParts(*shadowRoot);
355}

364void Invalidator::invalidateWithMatchElementRuleSets(Element& element, const MatchElementRuleSets& matchElementRuleSets)
365{
366 SetForScope<bool> isInvalidating(element.styleResolver().ruleSets().isInvalidatingStyleWithRuleSets(), true);
367
368 for (auto& matchElementAndRuleSet : matchElementRuleSets) {
369 Invalidator invalidator(matchElementAndRuleSet.value);
370 invalidator.invalidateStyleWithMatchElement(element, matchElementAndRuleSet.key);
371 }
372}

374void Invalidator::invalidateAllStyle(Scope& scope)
375{
376 if (auto* shadowRoot = scope.shadowRoot()) {
377 for (auto& shadowChild : childrenOfType<Element>(*shadowRoot))
378 shadowChild.invalidateStyleForSubtreeInternal();
379 invalidateHostAndSlottedStyleIfNeeded(*shadowRoot);
380 return;
381 }
382
383 scope.document().scheduleFullStyleRebuild();
384}

386void Invalidator::invalidateHostAndSlottedStyleIfNeeded(ShadowRoot& shadowRoot)
387{
388 auto& host = *shadowRoot.host();
389 auto* resolver = shadowRoot.styleScope().resolverIfExists();
390
391 auto shouldInvalidateHost = [&] {
392 if (!resolver)
393 return true;
394 return !resolver->ruleSets().authorStyle().hostPseudoClassRules().isEmpty();
395 }();
396
397 if (shouldInvalidateHost)
398 host.invalidateStyleInternal();
399
400 auto shouldInvalidateHostChildren = [&] {
401 if (!resolver)
402 return true;
403 return !resolver->ruleSets().authorStyle().slottedPseudoElementRules().isEmpty();
404 }();
405
406 if (shouldInvalidateHostChildren) {
407 for (auto& shadowChild : childrenOfType<Element>(host))
408 shadowChild.invalidateStyleInternal();
409 }
410}
