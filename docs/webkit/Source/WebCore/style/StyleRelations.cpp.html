37std::unique_ptr<Relations> commitRelationsToRenderStyle(RenderStyle& style, const Element& element, const Relations& relations)
38{
39 if (!relations.isEmpty())
40 style.setUnique();
41
42 std::unique_ptr<Relations> remainingRelations;
43
44 auto appendStyleRelation = [&remainingRelations] (const Relation& relation) {
45 if (!remainingRelations)
46 remainingRelations = makeUnique<Relations>();
47 remainingRelations->append(relation);
48 };
49
50 for (auto& relation : relations) {
51 if (relation.element != &element) {
52 appendStyleRelation(relation);
53 continue;
54 }
55 switch (relation.type) {
56 case Relation::AffectedByEmpty:
57 style.setEmptyState(relation.value);
58 appendStyleRelation(relation);
59 break;
60 case Relation::FirstChild:
61 style.setFirstChildState();
62 break;
63 case Relation::LastChild:
64 style.setLastChildState();
65 break;
66 case Relation::Unique:
67 break;
68 case Relation::AffectedByPreviousSibling:
69 case Relation::DescendantsAffectedByPreviousSibling:
70 case Relation::AffectsNextSibling:
71 case Relation::ChildrenAffectedByForwardPositionalRules:
72 case Relation::DescendantsAffectedByForwardPositionalRules:
73 case Relation::ChildrenAffectedByBackwardPositionalRules:
74 case Relation::DescendantsAffectedByBackwardPositionalRules:
75 case Relation::ChildrenAffectedByFirstChildRules:
76 case Relation::ChildrenAffectedByPropertyBasedBackwardPositionalRules:
77 case Relation::ChildrenAffectedByLastChildRules:
78 case Relation::NthChildIndex:
79 appendStyleRelation(relation);
80 break;
81 }
82 }
83 return remainingRelations;
84}

86void commitRelations(std::unique_ptr<Relations> relations, Update& update)
87{
88 if (!relations)
89 return;
90 for (auto& relation : *relations) {
91 auto& element = const_cast<Element&>(*relation.element);
92 switch (relation.type) {
93 case Relation::AffectedByEmpty:
94 element.setStyleAffectedByEmpty();
95 break;
96 case Relation::AffectedByPreviousSibling:
97 element.setStyleIsAffectedByPreviousSibling();
98 break;
99 case Relation::DescendantsAffectedByPreviousSibling:
100 element.setDescendantsAffectedByPreviousSibling();
101 break;
102 case Relation::AffectsNextSibling: {
103 auto* sibling = &element;
104 for (unsigned i = 0; i < relation.value && sibling; ++i, sibling = sibling->nextElementSibling())
105 sibling->setAffectsNextSiblingElementStyle();
106 break;
107 }
108 case Relation::ChildrenAffectedByForwardPositionalRules:
109 element.setChildrenAffectedByForwardPositionalRules();
110 break;
111 case Relation::DescendantsAffectedByForwardPositionalRules:
112 element.setDescendantsAffectedByForwardPositionalRules();
113 break;
114 case Relation::ChildrenAffectedByBackwardPositionalRules:
115 element.setChildrenAffectedByBackwardPositionalRules();
116 break;
117 case Relation::DescendantsAffectedByBackwardPositionalRules:
118 element.setDescendantsAffectedByBackwardPositionalRules();
119 break;
120 case Relation::ChildrenAffectedByFirstChildRules:
121 element.setChildrenAffectedByFirstChildRules();
122 break;
123 case Relation::ChildrenAffectedByPropertyBasedBackwardPositionalRules:
124 element.setChildrenAffectedByBackwardPositionalRules();
125 element.setChildrenAffectedByPropertyBasedBackwardPositionalRules();
126 break;
127 case Relation::ChildrenAffectedByLastChildRules:
128 element.setChildrenAffectedByLastChildRules();
129 break;
130 case Relation::FirstChild:
131 if (auto* style = update.elementStyle(element))
132 style->setFirstChildState();
133 break;
134 case Relation::LastChild:
135 if (auto* style = update.elementStyle(element))
136 style->setLastChildState();
137 break;
138 case Relation::NthChildIndex:
139 if (auto* style = update.elementStyle(element))
140 style->setUnique();
141 element.setChildIndex(relation.value);
142 break;
143 case Relation::Unique:
144 if (auto* style = update.elementStyle(element))
145 style->setUnique();
146 break;
147 }
148 }
149}
