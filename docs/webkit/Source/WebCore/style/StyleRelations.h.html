30namespace WebCore {
31
32class Element;
33class RenderStyle;
34
35namespace Style {
36
37class Update;
38
39struct Relation {
40 enum Type {
41 AffectedByEmpty,
42 AffectedByPreviousSibling,
43 DescendantsAffectedByPreviousSibling,
44 // For AffectsNextSibling 'value' tells how many element siblings to mark starting with 'element'.
45 AffectsNextSibling,
46 ChildrenAffectedByForwardPositionalRules,
47 DescendantsAffectedByForwardPositionalRules,
48 ChildrenAffectedByBackwardPositionalRules,
49 DescendantsAffectedByBackwardPositionalRules,
50 ChildrenAffectedByFirstChildRules,
51 ChildrenAffectedByPropertyBasedBackwardPositionalRules,
52 ChildrenAffectedByLastChildRules,
53 FirstChild,
54 LastChild,
55 NthChildIndex,
56 Unique,
57 };
58 const Element* element;
59 Type type;
60 unsigned value;
61
62 Relation(const Element& element, Type type, unsigned value = 1)
63 : element(&element)
64 , type(type)
65 , value(value)
66 { }
67};
68
69using Relations = Vector<Relation, 8>;
70
71std::unique_ptr<Relations> commitRelationsToRenderStyle(RenderStyle&, const Element&, const Relations&);
72void commitRelations(std::unique_ptr<Relations>, Update&);
73
74}
