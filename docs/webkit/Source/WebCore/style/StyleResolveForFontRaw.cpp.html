53static bool useFixedDefaultSize(const FontCascadeDescription& fontDescription)
54{
55 return fontDescription.familyCount() == 1 && fontDescription.firstFamily() == familyNamesData->at(FamilyNamesIndex::MonospaceFamily);
56}
57
58std::optional<FontCascade> resolveForFontRaw(const FontRaw& fontRaw, FontCascadeDescription&& fontDescription, ScriptExecutionContext& context)
59{
60 ASSERT(context.cssFontSelector());
61
62 // Map the font property longhands into the style.
63 float parentSize = fontDescription.specifiedSize();
64
65 // Font family applied in the same way as StyleBuilderCustom::applyValueFontFamily
66 // Before mapping in a new font-family property, we should reset the generic family.
67 bool oldFamilyUsedFixedDefaultSize = useFixedDefaultSize(fontDescription);
68
69 Vector<AtomString> families;
70 families.reserveInitialCapacity(fontRaw.family.size());
71
72 for (auto& item : fontRaw.family) {
73 AtomString family;
74 bool isGenericFamily = false;
75 switchOn(item, [&] (CSSValueID ident) {
76 isGenericFamily = ident != CSSValueWebkitBody;
77 if (isGenericFamily)
78 family = familyNamesData->at(CSSPropertyParserHelpers::genericFontFamilyIndex(ident));
79 else
80 family = AtomString(context.settingsValues().fontGenericFamilies.standardFontFamily());
81 }, [&] (const String& familyString) {
82 family = familyString;
83 });
84
85 if (family.isEmpty())
86 continue;
87 if (families.isEmpty())
88 fontDescription.setIsSpecifiedFont(!isGenericFamily);
89 families.uncheckedAppend(family);
90 }
91
92 if (families.isEmpty())
93 return std::nullopt;
94 fontDescription.setFamilies(families);
95
96 if (useFixedDefaultSize(fontDescription) != oldFamilyUsedFixedDefaultSize) {
97 if (CSSValueID sizeIdentifier = fontDescription.keywordSizeAsIdentifier()) {
98 auto size = Style::fontSizeForKeyword(sizeIdentifier, !oldFamilyUsedFixedDefaultSize, context.settingsValues());
99 fontDescription.setSpecifiedSize(size);
100 fontDescription.setComputedSize(Style::computedFontSizeFromSpecifiedSize(size, fontDescription.isAbsoluteSize(), 1.0, MinimumFontSizeRule::None, context.settingsValues()));
101 }
102 }
103
104 // Font style applied in the same way as BuilderConverter::convertFontStyleFromValue
105 if (fontRaw.style) {
106 switch (fontRaw.style->style) {
107 case CSSValueNormal:
108 break;
109
110 case CSSValueItalic:
111 fontDescription.setItalic(italicValue());
112 break;
113
114 case CSSValueOblique: {
115 float degrees;
116 if (fontRaw.style->angle)
117 degrees = static_cast<float>(CSSPrimitiveValue::computeDegrees(fontRaw.style->angle->type, fontRaw.style->angle->value));
118 else
119 degrees = 0;
120 fontDescription.setItalic(FontSelectionValue(degrees));
121 break;
122 }
123
124 default:
125 ASSERT_NOT_REACHED();
126 break;
127 }
128 }
129 fontDescription.setFontStyleAxis((fontRaw.style && fontRaw.style->style == CSSValueItalic) ? FontStyleAxis::ital : FontStyleAxis::slnt);
130
131 if (fontRaw.variantCaps) {
132 switch (*fontRaw.variantCaps) {
133 case CSSValueNormal:
134 fontDescription.setVariantCaps(FontVariantCaps::Normal);
135 break;
136 case CSSValueSmallCaps:
137 fontDescription.setVariantCaps(FontVariantCaps::Small);
138 break;
139 case CSSValueAllSmallCaps:
140 fontDescription.setVariantCaps(FontVariantCaps::AllSmall);
141 break;
142 case CSSValuePetiteCaps:
143 fontDescription.setVariantCaps(FontVariantCaps::Petite);
144 break;
145 case CSSValueAllPetiteCaps:
146 fontDescription.setVariantCaps(FontVariantCaps::AllPetite);
147 break;
148 case CSSValueUnicase:
149 fontDescription.setVariantCaps(FontVariantCaps::Unicase);
150 break;
151 case CSSValueTitlingCaps:
152 fontDescription.setVariantCaps(FontVariantCaps::Titling);
153 break;
154 default:
155 ASSERT_NOT_REACHED();
156 break;
157 }
158 }
159
160 if (fontRaw.weight) {
161 auto weight = switchOn(*fontRaw.weight, [&] (CSSValueID ident) {
162 switch (ident) {
163 case CSSValueNormal:
164 return normalWeightValue();
165 case CSSValueBold:
166 return boldWeightValue();
167 case CSSValueBolder:
168 return FontCascadeDescription::bolderWeight(fontDescription.weight());
169 case CSSValueLighter:
170 return FontCascadeDescription::lighterWeight(fontDescription.weight());
171 default:
172 ASSERT_NOT_REACHED();
173 return normalWeightValue();
174 }
175 }, [&] (double weight) {
176 return FontSelectionValue::clampFloat(weight);
177 });
178 fontDescription.setWeight(weight);
179 }
180
181 fontDescription.setKeywordSizeFromIdentifier(CSSValueInvalid);
182 float size = switchOn(fontRaw.size, [&] (CSSValueID ident) {
183 switch (ident) {
184 case CSSValueXxSmall:
185 case CSSValueXSmall:
186 case CSSValueSmall:
187 case CSSValueMedium:
188 case CSSValueLarge:
189 case CSSValueXLarge:
190 case CSSValueXxLarge:
191 case CSSValueWebkitXxxLarge:
192 fontDescription.setKeywordSizeFromIdentifier(ident);
193 return Style::fontSizeForKeyword(ident, fontDescription.useFixedDefaultSize(), context.settingsValues());
194 case CSSValueLarger:
195 return parentSize * 1.2f;
196 case CSSValueSmaller:
197 return parentSize / 1.2f;
198 default:
199 return 0.f;
200 }
201 }, [&] (const CSSPropertyParserHelpers::LengthOrPercentRaw& lengthOrPercent) {
202 return switchOn(lengthOrPercent, [&] (const CSSPropertyParserHelpers::LengthRaw& length) {
203 auto fontCascade = FontCascade(FontCascadeDescription(fontDescription));
204 fontCascade.update(context.cssFontSelector());
205 // FIXME: Passing null for the RenderView parameter means that vw and vh units will evaluate to
206 // zero and vmin and vmax units will evaluate as if they were px units.
207 // It's unclear in the specification if they're expected to work on OffscreenCanvas, given
208 // that it's off-screen and therefore doesn't strictly have an associated viewport.
209 // This needs clarification and possibly fixing.
210 return static_cast<float>(CSSPrimitiveValue::computeUnzoomedNonCalcLengthDouble(length.type, length.value, CSSPropertyFontSize, &fontCascade.fontMetrics(), &fontCascade.fontDescription(), &fontCascade.fontDescription(), is<Document>(context) ? downcast<Document>(context).renderView() : nullptr));
211 }, [&] (double percentage) {
212 return static_cast<float>((parentSize * percentage) / 100.0);
213 });
214 });
215
216 if (size > 0) {
217 fontDescription.setSpecifiedSize(size);
218 fontDescription.setComputedSize(size);
219 }
220
221 // As there is no line-height on FontCascade, there's no need to resolve
222 // it, even though there is line-height information on FontRaw.
223
224 auto fontCascade = FontCascade(WTFMove(fontDescription));
225 fontCascade.update(context.cssFontSelector());
226 return fontCascade;
227}

229}
