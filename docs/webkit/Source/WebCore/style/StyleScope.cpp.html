77Resolver& Scope::resolver()
78{
79 if (!m_resolver) {
80 if (m_shadowRoot)
81 createOrFindSharedShadowTreeResolver();
82 else
83 createDocumentResolver();
84 }
85 return *m_resolver;

88void Scope::createDocumentResolver()
89{
90 ASSERT(!m_resolver);
91 ASSERT(!m_shadowRoot);
92
93 SetForScope<bool> isUpdatingStyleResolver { m_isUpdatingStyleResolver, true };
94
95 m_resolver = Resolver::create(m_document);
96
97 m_document.fontSelector().buildStarted();
98
99 m_resolver->ruleSets().initializeUserStyle();
100 m_resolver->addCurrentSVGFontFaceRules();
101 m_resolver->appendAuthorStyleSheets(m_activeStyleSheets);
102
103 m_document.fontSelector().buildCompleted();
104}

106void Scope::createOrFindSharedShadowTreeResolver()
107{
108 ASSERT(!m_resolver);
109 ASSERT(m_shadowRoot);
110
111 auto key = makeResolverSharingKey();
112
113 auto result = documentScope().m_sharedShadowTreeResolvers.ensure(WTFMove(key), [&] {
114 SetForScope<bool> isUpdatingStyleResolver { m_isUpdatingStyleResolver, true };
115
116 m_resolver = Resolver::create(m_document);
117
118 m_resolver->ruleSets().setUsesSharedUserStyle(!isForUserAgentShadowTree());
119 m_resolver->appendAuthorStyleSheets(m_activeStyleSheets);
120
121 return makeRef(*m_resolver);
122 });
123
124 if (!result.isNewEntry) {
125 m_resolver = result.iterator->value.ptr();
126 m_resolver->setSharedBetweenShadowTrees();
127 }
128}

130void Scope::unshareShadowTreeResolverBeforeMutation()
131{
132 ASSERT(m_shadowRoot);
133 ASSERT(!m_resolver->isSharedBetweenShadowTrees());
134
135 documentScope().m_sharedShadowTreeResolvers.remove(makeResolverSharingKey());

138auto Scope::makeResolverSharingKey() -> ResolverSharingKey
139{
140 constexpr bool isNonEmptyHashTableValue = true;
141 return {
142 m_activeStyleSheets.map([&](auto& sheet) { return makeRefPtr(sheet->contents()); }),
143 isForUserAgentShadowTree(),
144 isNonEmptyHashTableValue
145 };

153void Scope::releaseMemory()
154{
155 if (!m_shadowRoot) {
156 for (auto* descendantShadowRoot : m_document.inDocumentShadowRoots())
157 descendantShadowRoot->styleScope().releaseMemory();
158 }

171 m_sharedShadowTreeResolvers.clear();

183Scope* Scope::forOrdinal(Element& element, ScopeOrdinal ordinal)
184{
185 switch (ordinal) {
186 case ScopeOrdinal::Element:
187 return &forNode(element);
188 case ScopeOrdinal::ContainingHost: {
189 auto* containingShadowRoot = element.containingShadowRoot();
190 if (!containingShadowRoot)
191 return nullptr;
192 return &forNode(*containingShadowRoot->host());
193 }
194 case ScopeOrdinal::Shadow: {
195 auto* shadowRoot = element.shadowRoot();
196 if (!shadowRoot)
197 return nullptr;
198 return &shadowRoot->styleScope();
199 }
200 default: {
201 ASSERT(ordinal >= ScopeOrdinal::FirstSlot);
202 auto slotIndex = ScopeOrdinal::FirstSlot;
203 for (auto* slot = element.assignedSlot(); slot; slot = slot->assignedSlot(), ++slotIndex) {
204 if (slotIndex == ordinal)
205 return &forNode(*slot);
206 }
207 return nullptr;
208 }
209 }

284void Scope::addStyleSheetCandidateNode(Node& node, bool createdByParser)
285{
286 if (!node.isConnected())
287 return;
288
289 // Until the <body> exists, we have no choice but to compare document positions,
290 // since styles outside of the body and head continue to be shunted into the head
291 // (and thus can shift to end up before dynamically added DOM content that is also
292 // outside the body).
293 if ((createdByParser && m_document.bodyOrFrameset()) || m_styleSheetCandidateNodes.isEmpty()) {
294 m_styleSheetCandidateNodes.add(&node);
295 return;
296 }
297
298 // Determine an appropriate insertion point.
299 auto begin = m_styleSheetCandidateNodes.begin();
300 auto end = m_styleSheetCandidateNodes.end();
301 auto it = end;
302 Node* followingNode = nullptr;
303 do {
304 --it;
305 Node* n = *it;
306 unsigned short position = n->compareDocumentPosition(node);
307 if (position == Node::DOCUMENT_POSITION_FOLLOWING) {
308 m_styleSheetCandidateNodes.insertBefore(followingNode, &node);
309 return;
310 }
311 followingNode = n;
312 } while (it != begin);
313
314 m_styleSheetCandidateNodes.insertBefore(followingNode, &node);
315}

320 didChangeActiveStyleSheetCandidates();

336void Scope::collectActiveStyleSheets(Vector<RefPtr<StyleSheet>>& sheets)
337{
338 if (!m_document.settings().authorAndUserStylesEnabled())
339 return;
340
341 for (auto& node : m_styleSheetCandidateNodes) {
342 RefPtr<StyleSheet> sheet;
343 if (is<ProcessingInstruction>(*node)) {
344 if (!downcast<ProcessingInstruction>(*node).isCSS())
345 continue;
346 // We don't support linking to embedded CSS stylesheets, see <https://bugs.webkit.org/show_bug.cgi?id=49281> for discussion.
347 sheet = downcast<ProcessingInstruction>(*node).sheet();
348 } else if (is<HTMLLinkElement>(*node) || is<HTMLStyleElement>(*node) || is<SVGStyleElement>(*node)) {
349 Element& element = downcast<Element>(*node);
350 AtomString title = element.isInShadowTree() ? nullAtom() : element.attributeWithoutSynchronization(titleAttr);
351 bool enabledViaScript = false;
352 if (is<HTMLLinkElement>(element)) {
353 // <LINK> element
354 HTMLLinkElement& linkElement = downcast<HTMLLinkElement>(element);
355 if (linkElement.isDisabled())
356 continue;
357 enabledViaScript = linkElement.isEnabledViaScript();
358 if (linkElement.styleSheetIsLoading()) {
359 // it is loading but we should still decide which style sheet set to use
360 if (!enabledViaScript && !title.isEmpty() && m_preferredStylesheetSetName.isEmpty()) {
361 if (!linkElement.attributeWithoutSynchronization(relAttr).contains("alternate"))
362 m_preferredStylesheetSetName = title;
363 }
364 continue;
365 }
366 if (!linkElement.sheet())
367 title = nullAtom();
368 }
369 // Get the current preferred styleset. This is the
370 // set of sheets that will be enabled.
371 if (is<SVGStyleElement>(element))
372 sheet = downcast<SVGStyleElement>(element).sheet();
373 else if (is<HTMLLinkElement>(element))
374 sheet = downcast<HTMLLinkElement>(element).sheet();
375 else
376 sheet = downcast<HTMLStyleElement>(element).sheet();
377 // Check to see if this sheet belongs to a styleset
378 // (thus making it PREFERRED or ALTERNATE rather than
379 // PERSISTENT).
380 auto& rel = element.attributeWithoutSynchronization(relAttr);
381 if (!enabledViaScript && !title.isEmpty()) {
382 // Yes, we have a title.
383 if (m_preferredStylesheetSetName.isEmpty()) {
384 // No preferred set has been established. If
385 // we are NOT an alternate sheet, then establish
386 // us as the preferred set. Otherwise, just ignore
387 // this sheet.
388 if (is<HTMLStyleElement>(element) || !rel.contains("alternate"))
389 m_preferredStylesheetSetName = title;
390 }
391 if (title != m_preferredStylesheetSetName)
392 sheet = nullptr;
393 }
394
395 if (rel.contains("alternate") && title.isEmpty())
396 sheet = nullptr;
397 }
398 if (sheet)
399 sheets.append(WTFMove(sheet));
400 }

403Scope::StyleSheetChange Scope::analyzeStyleSheetChange(const Vector<RefPtr<CSSStyleSheet>>& newStylesheets)
404{
405 unsigned newStylesheetCount = newStylesheets.size();
406
407 auto* resolver = resolverIfExists();
408 if (!resolver)
409 return { ResolverUpdateType::Reconstruct };
410
411 if (resolver->isSharedBetweenShadowTrees())
412 return { ResolverUpdateType::Reconstruct };
413
414 // Find out which stylesheets are new.
415 unsigned oldStylesheetCount = m_activeStyleSheets.size();
416 if (newStylesheetCount < oldStylesheetCount)
417 return { ResolverUpdateType::Reconstruct };
418
419 Vector<StyleSheetContents*> addedSheets;
420 unsigned newIndex = 0;
421 for (unsigned oldIndex = 0; oldIndex < oldStylesheetCount; ++oldIndex) {
422 if (newIndex >= newStylesheetCount)
423 return { ResolverUpdateType::Reconstruct };
424
425 while (m_activeStyleSheets[oldIndex] != newStylesheets[newIndex]) {
426 addedSheets.append(&newStylesheets[newIndex]->contents());
427 ++newIndex;
428 if (newIndex == newStylesheetCount)
429 return { ResolverUpdateType::Reconstruct };
430 }
431 ++newIndex;
432 }
433 bool hasInsertions = !addedSheets.isEmpty();
434 while (newIndex < newStylesheetCount) {
435 addedSheets.append(&newStylesheets[newIndex]->contents());
436 ++newIndex;
437 }
438
439 // If all new sheets were added at the end of the list we can just add them to existing Resolver.
440 // If there were insertions we need to re-add all the stylesheets so rules are ordered correctly.
441 return { hasInsertions ? ResolverUpdateType::Reset : ResolverUpdateType::Additive, WTFMove(addedSheets) };
442}

444static void filterEnabledNonemptyCSSStyleSheets(Vector<RefPtr<CSSStyleSheet>>& result, const Vector<RefPtr<StyleSheet>>& sheets)
445{
446 for (auto& sheet : sheets) {
447 if (!is<CSSStyleSheet>(*sheet))
448 continue;
449 CSSStyleSheet& styleSheet = downcast<CSSStyleSheet>(*sheet);
450 if (styleSheet.isLoading())
451 continue;
452 if (styleSheet.disabled())
453 continue;
454 if (!styleSheet.length())
455 continue;
456 result.append(&styleSheet);
457 }
458}

460void Scope::updateActiveStyleSheets(UpdateType updateType)
461{
462 ASSERT(!m_pendingUpdate);
463
464 if (!m_document.hasLivingRenderTree())
465 return;
466
467 if (m_document.inStyleRecalc() || m_document.inRenderTreeUpdate()) {
468 // Protect against deleting style resolver in the middle of a style resolution.
469 // Crash stacks indicate we can get here when a resource load fails synchronously (for example due to content blocking).
470 // FIXME: These kind of cases should be eliminated and this path replaced by an assert.
471 m_pendingUpdate = UpdateType::ContentsOrInterpretation;
472 m_document.scheduleFullStyleRebuild();
473 return;
474 }
475
476 Vector<RefPtr<StyleSheet>> activeStyleSheets;
477 collectActiveStyleSheets(activeStyleSheets);
478
479 Vector<RefPtr<CSSStyleSheet>> activeCSSStyleSheets;
480
481 if (!isForUserAgentShadowTree()) {
482 activeCSSStyleSheets.appendVector(m_document.extensionStyleSheets().injectedAuthorStyleSheets());
483 activeCSSStyleSheets.appendVector(m_document.extensionStyleSheets().authorStyleSheetsForTesting());
484 }
485
486 filterEnabledNonemptyCSSStyleSheets(activeCSSStyleSheets, activeStyleSheets);
487
488 auto styleSheetChange = StyleSheetChange { ResolverUpdateType::Reconstruct };
489 if (updateType == UpdateType::ActiveSet)
490 styleSheetChange = analyzeStyleSheetChange(activeCSSStyleSheets);
491
492 updateResolver(activeCSSStyleSheets, styleSheetChange.resolverUpdateType);
493
494 m_weakCopyOfActiveStyleSheetListForFastLookup = nullptr;
495 m_activeStyleSheets.swap(activeCSSStyleSheets);
496 m_styleSheetsForStyleSheetList.swap(activeStyleSheets);
497
498 InspectorInstrumentation::activeStyleSheetsUpdated(m_document);
499
500 for (const auto& sheet : m_activeStyleSheets) {
501 if (sheet->contents().usesStyleBasedEditability())
502 m_usesStyleBasedEditability = true;
503 }
504
505 invalidateStyleAfterStyleSheetChange(styleSheetChange);
506}

508void Scope::invalidateStyleAfterStyleSheetChange(const StyleSheetChange& styleSheetChange)
509{
510 // If we are already parsing the body and so may have significant amount of elements, put some effort into trying to avoid style recalcs.
511 bool invalidateAll = !m_document.bodyOrFrameset() || m_document.hasNodesWithNonFinalStyle() || m_document.hasNodesWithMissingStyle();
512
513 if (styleSheetChange.resolverUpdateType == ResolverUpdateType::Reconstruct || invalidateAll) {
514 Invalidator::invalidateAllStyle(*this);
515 return;
516 }
517
518 Invalidator invalidator(styleSheetChange.addedSheets, m_resolver->mediaQueryEvaluator());
519 invalidator.invalidateStyle(*this);
520}

522void Scope::updateResolver(Vector<RefPtr<CSSStyleSheet>>& activeStyleSheets, ResolverUpdateType updateType)
523{
524 if (updateType == ResolverUpdateType::Reconstruct) {
525 clearResolver();
526 return;
527 }
528
529 if (m_shadowRoot)
530 unshareShadowTreeResolverBeforeMutation();
531
532 SetForScope<bool> isUpdatingStyleResolver { m_isUpdatingStyleResolver, true };
533
534 if (updateType == ResolverUpdateType::Reset) {
535 m_resolver->ruleSets().resetAuthorStyle();
536 m_resolver->appendAuthorStyleSheets(activeStyleSheets);
537 return;
538 }
539
540 ASSERT(updateType == ResolverUpdateType::Additive);
541 ASSERT(activeStyleSheets.size() >= m_activeStyleSheets.size());
542
543 unsigned firstNewIndex = m_activeStyleSheets.size();
544 Vector<RefPtr<CSSStyleSheet>> newStyleSheets;
545 newStyleSheets.appendRange(activeStyleSheets.begin() + firstNewIndex, activeStyleSheets.end());
546 m_resolver->appendAuthorStyleSheets(newStyleSheets);
547}

549const Vector<RefPtr<CSSStyleSheet>> Scope::activeStyleSheetsForInspector()
550{
551 Vector<RefPtr<CSSStyleSheet>> result;
552
553 if (auto* pageUserSheet = m_document.extensionStyleSheets().pageUserSheet())
554 result.append(pageUserSheet);
555 result.appendVector(m_document.extensionStyleSheets().documentUserStyleSheets());
556 result.appendVector(m_document.extensionStyleSheets().injectedUserStyleSheets());
557 result.appendVector(m_document.extensionStyleSheets().injectedAuthorStyleSheets());
558 result.appendVector(m_document.extensionStyleSheets().authorStyleSheetsForTesting());
559
560 for (auto& styleSheet : m_styleSheetsForStyleSheetList) {
561 if (!is<CSSStyleSheet>(*styleSheet))
562 continue;
563
564 CSSStyleSheet& sheet = downcast<CSSStyleSheet>(*styleSheet);
565 if (sheet.disabled())
566 continue;
567
568 result.append(&sheet);
569 }
570
571 return result;
572}

603void Scope::scheduleUpdate(UpdateType update)
604{
605 if (update == UpdateType::ContentsOrInterpretation) {
606 // :host and ::slotted rules might go away.
607 if (m_shadowRoot)
608 Invalidator::invalidateHostAndSlottedStyleIfNeeded(*m_shadowRoot);
609 // FIXME: Animation code may trigger resource load in middle of style recalc and that can add a rule to a content extension stylesheet.
610 // Fix and remove isResolvingTreeStyle() test below, see https://bugs.webkit.org/show_bug.cgi?id=194335
611 // FIXME: The m_isUpdatingStyleResolver test is here because extension stylesheets can get us here from Resolver::appendAuthorStyleSheets.
612 if (!m_isUpdatingStyleResolver && !m_document.isResolvingTreeStyle())
613 clearResolver();
614 }
615
616 if (!m_pendingUpdate || *m_pendingUpdate < update) {
617 m_pendingUpdate = update;
618 if (m_shadowRoot)
619 documentScope().m_hasDescendantWithPendingUpdate = true;
620 }
621
622 m_document.scheduleRenderingUpdate({ });
623}

655auto Scope::collectResolverScopes() -> ResolverScopes
656{
657 ASSERT(!m_shadowRoot);
658
659 if (!resolverIfExists())
660 return { };
661
662 ResolverScopes resolverScopes;
663
664 resolverScopes.add(makeRef(*resolverIfExists()), Vector<CheckedPtr<Scope>> { this });
665
666 for (auto* shadowRoot : m_document.inDocumentShadowRoots()) {
667 auto& scope = shadowRoot->styleScope();
668 auto* resolver = scope.resolverIfExists();
669 if (!resolver)
670 continue;
671 resolverScopes.add(makeRef(*resolver), Vector<CheckedPtr<Scope>> { }).iterator->value.append(&scope);
672 }
673 return resolverScopes;
674}

676template <typename TestFunction>
677void Scope::evaluateMediaQueries(TestFunction&& testFunction)
678{
679 bool hadChanges = false;
680
681 auto resolverScopes = collectResolverScopes();
682 for (auto& [resolver, scopes] : resolverScopes) {
683 auto evaluationChanges = testFunction(resolver.get());
684 if (!evaluationChanges)
685 continue;
686 hadChanges = true;
687
688 for (auto& scope : scopes) {
689 switch (evaluationChanges->type) {
690 case DynamicMediaQueryEvaluationChanges::Type::InvalidateStyle: {
691 Invalidator invalidator(evaluationChanges->invalidationRuleSets);
692 invalidator.invalidateStyle(*scope);
693 break;
694 }
695 case DynamicMediaQueryEvaluationChanges::Type::ResetStyle:
696 scope->scheduleUpdate(UpdateType::ContentsOrInterpretation);
697 break;
698 }
699 }
700 }
701
702 if (hadChanges)
703 InspectorInstrumentation::mediaQueryResultChanged(m_document);
704}

716void Scope::didChangeStyleSheetEnvironment()
717{
718 if (!m_shadowRoot) {
719 m_sharedShadowTreeResolvers.clear();
720
721 for (auto* descendantShadowRoot : m_document.inDocumentShadowRoots()) {
722 // Stylesheets is author shadow roots are potentially affected.
723 if (descendantShadowRoot->mode() != ShadowRootMode::UserAgent)
724 descendantShadowRoot->styleScope().scheduleUpdate(UpdateType::ContentsOrInterpretation);
725 }
726 }
727 scheduleUpdate(UpdateType::ContentsOrInterpretation);
728}

730void Scope::invalidateMatchedDeclarationsCache()
731{
732 if (!m_shadowRoot) {
733 for (auto* descendantShadowRoot : m_document.inDocumentShadowRoots())
734 descendantShadowRoot->styleScope().invalidateMatchedDeclarationsCache();
735 }
