64void ScopeRuleSets::updateUserAgentMediaQueryStyleIfNeeded() const
65{
66 if (!UserAgentStyle::mediaQueryStyleSheet)
67 return;
68
69 auto ruleCount = UserAgentStyle::mediaQueryStyleSheet->ruleCount();
70 if (m_userAgentMediaQueryStyle && ruleCount == m_userAgentMediaQueryRuleCountOnUpdate)
71 return;
72 m_userAgentMediaQueryRuleCountOnUpdate = ruleCount;
73
74 // Media queries on user agent sheet need to evaluated in document context. They behave like author sheets in this respect.
75 auto& mediaQueryEvaluator = m_styleResolver.mediaQueryEvaluator();
76 m_userAgentMediaQueryStyle = RuleSet::create();
77
78 m_userAgentMediaQueryStyle->addRulesFromSheet(*UserAgentStyle::mediaQueryStyleSheet, nullptr, mediaQueryEvaluator, m_styleResolver);
79}

88void ScopeRuleSets::initializeUserStyle()
89{
90 auto& extensionStyleSheets = m_styleResolver.document().extensionStyleSheets();
91 auto& mediaQueryEvaluator = m_styleResolver.mediaQueryEvaluator();
92 auto tempUserStyle = RuleSet::create();
93 if (CSSStyleSheet* pageUserSheet = extensionStyleSheets.pageUserSheet())
94 tempUserStyle->addRulesFromSheet(pageUserSheet->contents(), nullptr, mediaQueryEvaluator, m_styleResolver);
95#if ENABLE(APP_BOUND_DOMAINS)
96 auto* page = m_styleResolver.document().page();
97 if (!extensionStyleSheets.injectedUserStyleSheets().isEmpty() && page && page->mainFrame().loader().client().shouldEnableInAppBrowserPrivacyProtections())
98 m_styleResolver.document().addConsoleMessage(MessageSource::Security, MessageLevel::Warning, "Ignoring user style sheet for non-app bound domain."_s);
99 else {
100 collectRulesFromUserStyleSheets(extensionStyleSheets.injectedUserStyleSheets(), tempUserStyle.get(), mediaQueryEvaluator);
101 if (page && !extensionStyleSheets.injectedUserStyleSheets().isEmpty())
102 page->mainFrame().loader().client().notifyPageOfAppBoundBehavior();
103 }
104#else
105 collectRulesFromUserStyleSheets(extensionStyleSheets.injectedUserStyleSheets(), tempUserStyle.get(), mediaQueryEvaluator);
106#endif
107 collectRulesFromUserStyleSheets(extensionStyleSheets.documentUserStyleSheets(), tempUserStyle.get(), mediaQueryEvaluator);
108 if (tempUserStyle->ruleCount() > 0 || tempUserStyle->pageRules().size() > 0)
109 m_userStyle = WTFMove(tempUserStyle);
110}

112void ScopeRuleSets::collectRulesFromUserStyleSheets(const Vector<RefPtr<CSSStyleSheet>>& userSheets, RuleSet& userStyle, const MediaQueryEvaluator& medium)
113{
114 for (unsigned i = 0; i < userSheets.size(); ++i) {
115 ASSERT(userSheets[i]->contents().isUserStyleSheet());
116 userStyle.addRulesFromSheet(userSheets[i]->contents(), nullptr, medium, m_styleResolver);
117 }
118}

120static RefPtr<RuleSet> makeRuleSet(const RuleFeatureVector& rules)
121{
122 size_t size = rules.size();
123 if (!size)
124 return nullptr;
125 auto ruleSet = RuleSet::create();
126 for (size_t i = 0; i < size; ++i)
127 ruleSet->addRule(*rules[i].styleRule, rules[i].selectorIndex, rules[i].selectorListIndex);
128 ruleSet->shrinkToFit();
129 return ruleSet;
130}

156std::optional<DynamicMediaQueryEvaluationChanges> ScopeRuleSets::evaluateDynamicMediaQueryRules(const MediaQueryEvaluator& evaluator)
157{
158 std::optional<DynamicMediaQueryEvaluationChanges> evaluationChanges;
159
160 auto evaluate = [&](auto* ruleSet) {
161 if (!ruleSet)
162 return;
163 if (auto changes = ruleSet->evaluateDynamicMediaQueryRules(evaluator)) {
164 if (evaluationChanges)
165 evaluationChanges->append(WTFMove(*changes));
166 else
167 evaluationChanges = changes;
168 }
169 };
170
171 evaluate(&authorStyle());
172 evaluate(userStyle());
173 evaluate(userAgentMediaQueryStyle());
174
175 return evaluationChanges;
176}

178void ScopeRuleSets::appendAuthorStyleSheets(const Vector<RefPtr<CSSStyleSheet>>& styleSheets, MediaQueryEvaluator* medium, InspectorCSSOMWrappers& inspectorCSSOMWrappers)
179{
180 for (auto& cssSheet : styleSheets) {
181 ASSERT(!cssSheet->disabled());
182 m_authorStyle->addRulesFromSheet(cssSheet->contents(), cssSheet->mediaQueries(), *medium, m_styleResolver);
183 inspectorCSSOMWrappers.collectFromStyleSheetIfNeeded(cssSheet.get());
184 }
185
186 m_authorStyle->shrinkToFit();
187 collectFeatures();
188}

190void ScopeRuleSets::collectFeatures() const
191{
192 RELEASE_ASSERT(!m_isInvalidatingStyleWithRuleSets);
193
194 m_features.clear();
195 // Collect all ids and rules using sibling selectors (:first-child and similar)
196 // in the current set of stylesheets. Style sharing code uses this information to reject
197 // sharing candidates.
198 if (UserAgentStyle::defaultStyle)
199 m_features.add(UserAgentStyle::defaultStyle->features());
200 m_defaultStyleVersionOnFeatureCollection = UserAgentStyle::defaultStyleVersion;
201
202 if (auto* userAgentMediaQueryStyle = this->userAgentMediaQueryStyle())
203 m_features.add(userAgentMediaQueryStyle->features());
204
205 if (m_authorStyle)
206 m_features.add(m_authorStyle->features());
207 if (auto* userStyle = this->userStyle())
208 m_features.add(userStyle->features());
209
210 m_siblingRuleSet = makeRuleSet(m_features.siblingRules);
211 m_uncommonAttributeRuleSet = makeRuleSet(m_features.uncommonAttributeRules);
212
213 m_classInvalidationRuleSets.clear();
214 m_attributeInvalidationRuleSets.clear();
215 m_pseudoClassInvalidationRuleSets.clear();
216
217 m_cachedHasComplexSelectorsForStyleAttribute = std::nullopt;
218
219 m_features.shrinkToFit();
220}

222template<typename KeyType, typename RuleFeatureVectorType, typename Hash, typename HashTraits>
223static Vector<InvalidationRuleSet>* ensureInvalidationRuleSets(const KeyType& key, HashMap<KeyType, std::unique_ptr<Vector<InvalidationRuleSet>>, Hash, HashTraits>& ruleSetMap, const HashMap<KeyType, std::unique_ptr<RuleFeatureVectorType>, Hash, HashTraits>& ruleFeatures)
224{
225 return ruleSetMap.ensure(key, [&] () -> std::unique_ptr<Vector<InvalidationRuleSet>> {
226 auto* features = ruleFeatures.get(key);
227 if (!features)
228 return nullptr;
229
230 std::array<RefPtr<RuleSet>, matchElementCount> matchElementArray;
231 std::array<Vector<const CSSSelector*>, matchElementCount> invalidationSelectorArray;
232 for (auto& feature : *features) {
233 RELEASE_ASSERT(feature.matchElement);
234 auto arrayIndex = static_cast<unsigned>(*feature.matchElement);
235 RELEASE_ASSERT(arrayIndex < matchElementArray.size());
236
237 auto& ruleSet = matchElementArray[arrayIndex];
238 if (!ruleSet)
239 ruleSet = RuleSet::create();
240 ruleSet->addRule(*feature.styleRule, feature.selectorIndex, feature.selectorListIndex);
241 if constexpr (std::is_same<typename RuleFeatureVectorType::ValueType, RuleFeatureWithInvalidationSelector>::value) {
242 if (feature.invalidationSelector)
243 invalidationSelectorArray[arrayIndex].append(feature.invalidationSelector);
244 }
245 }
246
247 unsigned ruleSetCount = 0;
248 for (const auto& item : matchElementArray) {
249 if (item)
250 ++ruleSetCount;
251 }
252
253 auto invalidationRuleSets = makeUnique<Vector<InvalidationRuleSet>>();
254 invalidationRuleSets->reserveInitialCapacity(ruleSetCount);
255
256 for (unsigned i = 0; i < matchElementArray.size(); ++i) {
257 if (matchElementArray[i]) {
258 matchElementArray[i]->shrinkToFit();
259 invalidationRuleSets->uncheckedAppend({ static_cast<MatchElement>(i), matchElementArray[i].releaseNonNull(), WTFMove(invalidationSelectorArray[i]) });
260 }
261 }
262 return invalidationRuleSets;
263 }).iterator->value.get();
264}

281bool ScopeRuleSets::hasComplexSelectorsForStyleAttribute() const
282{
283 auto compute = [&] {
284 auto* ruleSets = attributeInvalidationRuleSets(HTMLNames::styleAttr->localName());
285 if (!ruleSets)
286 return false;
287 for (auto& ruleSet : *ruleSets) {
288 if (ruleSet.matchElement != MatchElement::Subject)
289 return true;
290 }
291 return false;
292 };
293
294 if (!m_cachedHasComplexSelectorsForStyleAttribute)
295 m_cachedHasComplexSelectorsForStyleAttribute = compute();
296
297 return *m_cachedHasComplexSelectorsForStyleAttribute;
298}
