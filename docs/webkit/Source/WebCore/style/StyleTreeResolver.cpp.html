132std::unique_ptr<RenderStyle> TreeResolver::styleForStyleable(const Styleable& styleable, const RenderStyle& inheritedStyle)
133{
134 auto& element = styleable.element;
135
136 if (element.hasCustomStyleResolveCallbacks()) {
137 RenderStyle* shadowHostStyle = scope().shadowRoot ? m_update->elementStyle(*scope().shadowRoot->host()) : nullptr;
138 if (auto customStyle = element.resolveCustomStyle(inheritedStyle, shadowHostStyle)) {
139 if (customStyle->relations)
140 commitRelations(WTFMove(customStyle->relations), *m_update);
141
142 return WTFMove(customStyle->renderStyle);
143 }
144 }
145
146 if (auto style = scope().sharingResolver.resolve(styleable, *m_update))
147 return style;
148
149 auto elementStyle = scope().resolver->styleForElement(element, &inheritedStyle, parentBoxStyle(), RuleMatchingBehavior::MatchAllRules, &scope().selectorFilter);
150
151 if (elementStyle.relations)
152 commitRelations(WTFMove(elementStyle.relations), *m_update);
153
154 return WTFMove(elementStyle.renderStyle);
155}

157static void resetStyleForNonRenderedDescendants(Element& current)
158{
159 for (auto& child : childrenOfType<Element>(current)) {
160 if (child.needsStyleRecalc()) {
161 child.resetComputedStyle();
162 child.resetStyleRelations();
163 child.setHasValidStyle();
164 }
165
166 if (child.childNeedsStyleRecalc())
167 resetStyleForNonRenderedDescendants(child);
168 }
169 current.clearChildNeedsStyleRecalc();
170}

172static bool affectsRenderedSubtree(Element& element, const RenderStyle& newStyle)
173{
174 if (newStyle.display() != DisplayType::None)
175 return true;
176 if (element.renderOrDisplayContentsStyle())
177 return true;
178 if (element.rendererIsNeeded(newStyle))
179 return true;
180 return false;
181}

183static DescendantsToResolve computeDescendantsToResolve(Change change, Validity validity, DescendantsToResolve parentDescendantsToResolve)
184{
185 if (parentDescendantsToResolve == DescendantsToResolve::All)
186 return DescendantsToResolve::All;
187 if (validity >= Validity::SubtreeInvalid)
188 return DescendantsToResolve::All;
189 switch (change) {
190 case Change::None:
191 return DescendantsToResolve::None;
192 case Change::NonInherited:
193 return DescendantsToResolve::ChildrenWithExplicitInherit;
194 case Change::Inherited:
195 return DescendantsToResolve::Children;
196 case Change::Renderer:
197 return DescendantsToResolve::All;
198 };
199 ASSERT_NOT_REACHED();
200 return DescendantsToResolve::None;
201};

203ElementUpdates TreeResolver::resolveElement(Element& element)
204{
205 if (m_didSeePendingStylesheet && !element.renderer() && !m_document.isIgnoringPendingStylesheets()) {
206 m_document.setHasNodesWithMissingStyle();
207 return { };
208 }
209
210 if (!element.rendererIsEverNeeded() && !element.hasDisplayContents())
211 return { };
212
213 Styleable styleable { element, PseudoId::None };
214 auto newStyle = styleForStyleable(styleable, parent().style);
215
216 if (!affectsRenderedSubtree(element, *newStyle))
217 return { };
218
219 auto* existingStyle = element.renderOrDisplayContentsStyle();
220
221 if (m_didSeePendingStylesheet && (!existingStyle || existingStyle->isNotFinal())) {
222 newStyle->setIsNotFinal();
223 m_document.setHasNodesWithNonFinalStyle();
224 }
225
226 auto update = createAnimatedElementUpdate(WTFMove(newStyle), styleable, parent().change, parent().style, parentBoxStyle());
227 auto descendantsToResolve = computeDescendantsToResolve(update.change, element.styleValidity(), parent().descendantsToResolve);
228
229 if (&element == m_document.documentElement()) {
230 m_documentElementStyle = RenderStyle::clonePtr(*update.style);
231 scope().resolver->setOverrideDocumentElementStyle(m_documentElementStyle.get());
232
233 if (!existingStyle || existingStyle->computedFontPixelSize() != update.style->computedFontPixelSize()) {
234 // "rem" units are relative to the document element's font size so we need to recompute everything.
235 scope().resolver->invalidateMatchedDeclarationsCache();
236 descendantsToResolve = DescendantsToResolve::All;
237 }
238 }
239
240 // This is needed for resolving color:-webkit-text for subsequent elements.
241 // FIXME: We shouldn't mutate document when resolving style.
242 if (&element == m_document.body())
243 m_document.setTextColor(update.style->visitedDependentColor(CSSPropertyColor));
244
245 // FIXME: These elements should not change renderer based on appearance property.
246 if (element.hasTagName(HTMLNames::meterTag)
247 || is<HTMLProgressElement>(element)
248 || (is<HTMLInputElement>(element) && downcast<HTMLInputElement>(element).isSearchField())) {
249 if (existingStyle && update.style->appearance() != existingStyle->appearance()) {
250 update.change = Change::Renderer;
251 descendantsToResolve = DescendantsToResolve::All;
252 }
253 }
254
255 PseudoIdToElementUpdateMap pseudoUpdates;
256 if (auto markerElementUpdate = resolvePseudoStyle(element, update, PseudoId::Marker))
257 pseudoUpdates.set(PseudoId::Marker, WTFMove(*markerElementUpdate));
258 if (auto beforeElementUpdate = resolvePseudoStyle(element, update, PseudoId::Before))
259 pseudoUpdates.set(PseudoId::Before, WTFMove(*beforeElementUpdate));
260 if (auto afterElementUpdate = resolvePseudoStyle(element, update, PseudoId::After))
261 pseudoUpdates.set(PseudoId::After, WTFMove(*afterElementUpdate));
262
263#if ENABLE(TOUCH_ACTION_REGIONS)
264 // FIXME: Track this exactly.
265 if (update.style->touchActions() != TouchAction::Auto && !m_document.quirks().shouldDisablePointerEventsQuirk())
266 m_document.setMayHaveElementsWithNonAutoTouchAction();
267#endif
268#if ENABLE(EDITABLE_REGION)
269 if (update.style->userModify() != UserModify::ReadOnly)
270 m_document.setMayHaveEditableElements();
271#endif
272
273 return { WTFMove(update), descendantsToResolve, WTFMove(pseudoUpdates) };
274}

276std::optional<ElementUpdate> TreeResolver::resolvePseudoStyle(Element& element, const ElementUpdate& elementUpdate, PseudoId pseudoId)
277{
278 if (pseudoId == PseudoId::Marker && elementUpdate.style->display() != DisplayType::ListItem)
279 return { };
280 if (elementUpdate.style->display() == DisplayType::None)
281 return { };
282 if (!elementUpdate.style->hasPseudoStyle(pseudoId))
283 return { };
284
285 auto& parentStyle = *elementUpdate.style;
286 auto* parentBoxStyle = parentBoxStyleForPseudo(elementUpdate);
287
288 auto pseudoStyle = scope().resolver->pseudoStyleForElement(element, { pseudoId }, parentStyle, parentBoxStyle, &scope().selectorFilter);
289 if (!pseudoStyle)
290 return { };
291
292 bool hasAnimations = pseudoStyle->hasAnimationsOrTransitions() || element.hasKeyframeEffects(pseudoId);
293 if (!pseudoElementRendererIsNeeded(pseudoStyle.get()) && !hasAnimations)
294 return { };
295
296 return createAnimatedElementUpdate(WTFMove(pseudoStyle), { element, pseudoId }, elementUpdate.change, parentStyle, parentBoxStyle);
297}

299const RenderStyle* TreeResolver::parentBoxStyle() const
300{
301 // 'display: contents' doesn't generate boxes.
302 for (auto i = m_parentStack.size(); i--;) {
303 auto& parent = m_parentStack[i];
304 if (parent.style.display() == DisplayType::None)
305 return nullptr;
306 if (parent.style.display() != DisplayType::Contents)
307 return &parent.style;
308 }
309 ASSERT_NOT_REACHED();
310 return nullptr;
311}

313const RenderStyle* TreeResolver::parentBoxStyleForPseudo(const ElementUpdate& elementUpdate) const
314{
315 switch (elementUpdate.style->display()) {
316 case DisplayType::None:
317 return nullptr;
318 case DisplayType::Contents:
319 return parentBoxStyle();
320 default:
321 return elementUpdate.style.get();
322 }

325ElementUpdate TreeResolver::createAnimatedElementUpdate(std::unique_ptr<RenderStyle> newStyle, const Styleable& styleable, Change parentChange, const RenderStyle& parentStyle, const RenderStyle* parentBoxStyle)
326{
327 auto& element = styleable.element;
328 auto& document = element.document();
329 auto* oldStyle = element.renderOrDisplayContentsStyle(styleable.pseudoId);
330
331 OptionSet<AnimationImpact> animationImpact;
332
333 // First, we need to make sure that any new CSS animation occuring on this element has a matching WebAnimation
334 // on the document timeline. Note that we get timeline() on the Document here because we need a timeline created
335 // in case no Web Animations have been created through the JS API.
336 if (document.backForwardCacheState() == Document::NotInBackForwardCache && !document.renderView()->printing()) {
337 if (oldStyle && (oldStyle->hasTransitions() || newStyle->hasTransitions()))
338 styleable.updateCSSTransitions(*oldStyle, *newStyle);
339
340 // The order in which CSS Transitions and CSS Animations are updated matters since CSS Transitions define the after-change style
341 // to use CSS Animations as defined in the previous style change event. As such, we update CSS Animations after CSS Transitions
342 // such that when CSS Transitions are updated the CSS Animations data is the same as during the previous style change event.
343 if ((oldStyle && oldStyle->hasAnimations()) || newStyle->hasAnimations())
344 styleable.updateCSSAnimations(oldStyle, *newStyle, &parentStyle);
345 }
346
347 // Now we can update all Web animations, which will include CSS Animations as well
348 // as animations created via the JS API.
349 if (styleable.hasKeyframeEffects()) {
350 auto previousLastStyleChangeEventStyle = styleable.lastStyleChangeEventStyle() ? RenderStyle::clonePtr(*styleable.lastStyleChangeEventStyle()) : RenderStyle::createPtr();
351 // Record the style prior to applying animations for this style change event.
352 styleable.setLastStyleChangeEventStyle(RenderStyle::clonePtr(*newStyle));
353 // Apply all keyframe effects to the new style.
354 auto animatedStyle = RenderStyle::clonePtr(*newStyle);
355 animationImpact = styleable.applyKeyframeEffects(*animatedStyle, *previousLastStyleChangeEventStyle, &parentStyle);
356 newStyle = WTFMove(animatedStyle);
357
358 Adjuster adjuster(document, parentStyle, parentBoxStyle, styleable.pseudoId == PseudoId::None ? &element : nullptr);
359 adjuster.adjustAnimatedStyle(*newStyle, animationImpact);
360 } else
361 styleable.setLastStyleChangeEventStyle(nullptr);
362
363 // Deduplication speeds up equality comparisons as the properties inherit to descendants.
364 // FIXME: There should be a more general mechanism for this.
365 if (oldStyle)
366 newStyle->deduplicateInheritedCustomProperties(*oldStyle);
367
368 auto change = oldStyle ? determineChange(*oldStyle, *newStyle) : Change::Renderer;
369
370 auto validity = element.styleValidity();
371 if (validity >= Validity::SubtreeAndRenderersInvalid || parentChange == Change::Renderer)
372 change = Change::Renderer;
373
374 bool shouldRecompositeLayer = animationImpact.contains(AnimationImpact::RequiresRecomposite) || element.styleResolutionShouldRecompositeLayer();
375 return { WTFMove(newStyle), change, shouldRecompositeLayer };
376}

378void TreeResolver::pushParent(Element& element, const RenderStyle& style, Change change, DescendantsToResolve descendantsToResolve)
379{
380 scope().selectorFilter.pushParent(&element);
381
382 Parent parent(element, style, change, descendantsToResolve);
383
384 if (auto* shadowRoot = element.shadowRoot()) {
385 pushScope(*shadowRoot);
386 parent.didPushScope = true;
387 }
388 else if (is<HTMLSlotElement>(element) && downcast<HTMLSlotElement>(element).assignedNodes()) {
389 pushEnclosingScope();
390 parent.didPushScope = true;
391 }
392
393 m_parentStack.append(WTFMove(parent));
394}

396void TreeResolver::popParent()
397{
398 auto& parentElement = *parent().element;
399
400 parentElement.setHasValidStyle();
401 parentElement.clearChildNeedsStyleRecalc();
402
403 if (parent().didPushScope)
404 popScope();
405
406 scope().selectorFilter.popParent();
407
408 m_parentStack.removeLast();
409}

411void TreeResolver::popParentsToDepth(unsigned depth)
412{
413 ASSERT(depth);
414 ASSERT(m_parentStack.size() >= depth);
415
416 while (m_parentStack.size() > depth)
417 popParent();
418}

420static bool shouldResolvePseudoElement(const PseudoElement* pseudoElement)
421{
422 if (!pseudoElement)
423 return false;
424 return pseudoElement->needsStyleRecalc();
425}
426
427static bool shouldResolveElement(const Element& element, DescendantsToResolve parentDescendantsToResolve)
428{
429 if (element.styleValidity() != Validity::Valid)
430 return true;
431 if (shouldResolvePseudoElement(element.beforePseudoElement()))
432 return true;
433 if (shouldResolvePseudoElement(element.afterPseudoElement()))
434 return true;
435
436 switch (parentDescendantsToResolve) {
437 case DescendantsToResolve::None:
438 return false;
439 case DescendantsToResolve::Children:
440 case DescendantsToResolve::All:
441 return true;
442 case DescendantsToResolve::ChildrenWithExplicitInherit:
443 auto* existingStyle = element.renderOrDisplayContentsStyle();
444 return existingStyle && existingStyle->hasExplicitlyInheritedProperties();
445 };
446 ASSERT_NOT_REACHED();
447 return false;
448}

450static void clearNeedsStyleResolution(Element& element)
451{
452 element.setHasValidStyle();
453 if (auto* before = element.beforePseudoElement())
454 before->setHasValidStyle();
455 if (auto* after = element.afterPseudoElement())
456 after->setHasValidStyle();
457}

459static bool hasLoadingStylesheet(const Style::Scope& styleScope, const Element& element, bool checkDescendants)
460{
461 if (!styleScope.hasPendingSheetsInBody())
462 return false;
463 if (styleScope.hasPendingSheetInBody(element))
464 return true;
465 if (!checkDescendants)
466 return false;
467 for (auto& descendant : descendantsOfType<Element>(element)) {
468 if (styleScope.hasPendingSheetInBody(descendant))
469 return true;
470 };
471 return false;
472}

474static std::unique_ptr<RenderStyle> createInheritedDisplayContentsStyleIfNeeded(const RenderStyle& parentElementStyle, const RenderStyle* parentBoxStyle)
475{
476 if (parentElementStyle.display() != DisplayType::Contents)
477 return nullptr;
478 if (parentBoxStyle && parentBoxStyle->inheritedEqual(parentElementStyle))
479 return nullptr;
480 // Compute style for imaginary unstyled <span> around the text node.
481 auto style = RenderStyle::createPtr();
482 style->inheritFrom(parentElementStyle);
483 return style;
484}

486static void resetDescendantStyleRelations(Element& element, DescendantsToResolve descendantsToResolve)
487{
488 switch (descendantsToResolve) {
489 case DescendantsToResolve::None:
490 case DescendantsToResolve::ChildrenWithExplicitInherit:
491 break;
492 case DescendantsToResolve::Children:
493 element.resetChildStyleRelations();
494 break;
495 case DescendantsToResolve::All:
496 element.resetAllDescendantStyleRelations();
497 break;
498 };

501void TreeResolver::resolveComposedTree()
502{
503 ASSERT(m_parentStack.size() == 1);
504 ASSERT(m_scopeStack.size() == 1);
505
506 auto descendants = composedTreeDescendants(m_document);
507 auto it = descendants.begin();
508 auto end = descendants.end();
509
510 while (it != end) {
511 popParentsToDepth(it.depth());
512
513 auto& node = *it;
514 auto& parent = this->parent();
515
516 ASSERT(node.isConnected());
517 ASSERT(node.containingShadowRoot() == scope().shadowRoot);
518 ASSERT(node.parentElement() == parent.element || is<ShadowRoot>(node.parentNode()) || node.parentElement()->shadowRoot());
519
520 if (is<Text>(node)) {
521 auto& text = downcast<Text>(node);
522
523 if ((text.styleValidity() >= Validity::SubtreeAndRenderersInvalid && parent.change != Change::Renderer) || parent.style.display() == DisplayType::Contents) {
524 TextUpdate textUpdate;
525 textUpdate.inheritedDisplayContentsStyle = createInheritedDisplayContentsStyleIfNeeded(parent.style, parentBoxStyle());
526
527 m_update->addText(text, parent.element, WTFMove(textUpdate));
528 }
529
530 text.setHasValidStyle();
531 it.traverseNextSkippingChildren();
532 continue;
533 }
534
535 auto& element = downcast<Element>(node);
536
537 if (it.depth() > Settings::defaultMaximumRenderTreeDepth) {
538 resetStyleForNonRenderedDescendants(element);
539 it.traverseNextSkippingChildren();
540 continue;
541 }
542
543 auto* style = element.renderOrDisplayContentsStyle();
544 auto change = Change::None;
545 auto descendantsToResolve = DescendantsToResolve::None;
546
547 bool shouldResolve = shouldResolveElement(element, parent.descendantsToResolve);
548 if (shouldResolve) {
549 if (!element.hasDisplayContents())
550 element.resetComputedStyle();
551 element.resetStyleRelations();
552
553 if (element.hasCustomStyleResolveCallbacks())
554 element.willRecalcStyle(parent.change);
555
556 auto elementUpdates = resolveElement(element);
557
558 if (element.hasCustomStyleResolveCallbacks())
559 element.didRecalcStyle(elementUpdates.update.change);
560
561 style = elementUpdates.update.style.get();
562 change = elementUpdates.update.change;
563 descendantsToResolve = elementUpdates.descendantsToResolve;
564
565 if (elementUpdates.update.style)
566 m_update->addElement(element, parent.element, WTFMove(elementUpdates));
567
568 clearNeedsStyleResolution(element);
569 }
570
571 if (!style)
572 resetStyleForNonRenderedDescendants(element);
573
574 bool shouldIterateChildren = style && (element.childNeedsStyleRecalc() || descendantsToResolve != DescendantsToResolve::None);
575
576 if (!m_didSeePendingStylesheet)
577 m_didSeePendingStylesheet = hasLoadingStylesheet(m_document.styleScope(), element, !shouldIterateChildren);
578
579 if (!shouldIterateChildren) {
580 it.traverseNextSkippingChildren();
581 continue;
582 }
583
584 resetDescendantStyleRelations(element, descendantsToResolve);
585
586 pushParent(element, *style, change, descendantsToResolve);
587
588 it.traverseNext();
589 }
590
591 popParentsToDepth(1);
592}

594std::unique_ptr<Update> TreeResolver::resolve()
595{
596 auto& renderView = *m_document.renderView();
597
598 Element* documentElement = m_document.documentElement();
599 if (!documentElement) {
600 m_document.styleScope().resolver();
601 return nullptr;
602 }
603 if (!documentElement->childNeedsStyleRecalc() && !documentElement->needsStyleRecalc())
604 return WTFMove(m_update);
605
606 m_didSeePendingStylesheet = m_document.styleScope().hasPendingSheetsBeforeBody();
607
608 if (!m_update)
609 m_update = makeUnique<Update>(m_document);
610 m_scopeStack.append(adoptRef(*new Scope(m_document)));
611 m_parentStack.append(Parent(m_document));
612
613 auto rootResolver = scope().resolver;
614
615 // Pseudo element removal and similar may only work with these flags still set. Reset them after the style recalc.
616 renderView.setUsesFirstLineRules(renderView.usesFirstLineRules() || rootResolver->usesFirstLineRules());
617 renderView.setUsesFirstLetterRules(renderView.usesFirstLetterRules() || rootResolver->usesFirstLetterRules());
618
619 resolveComposedTree();
620
621 renderView.setUsesFirstLineRules(rootResolver->usesFirstLineRules());
622 renderView.setUsesFirstLetterRules(rootResolver->usesFirstLetterRules());
623
624 ASSERT(m_scopeStack.size() == 1);
625 ASSERT(m_parentStack.size() == 1);
626 m_parentStack.clear();
627 popScope();
628
629 if (m_update->roots().isEmpty())
630 return { };
631
632 return WTFMove(m_update);
633}

652static void suspendMemoryCacheClientCalls(Document& document)

663static unsigned resolutionNestingDepth;
664
665PostResolutionCallbackDisabler::PostResolutionCallbackDisabler(Document& document, DrainCallbacks drainCallbacks)
666 : m_drainCallbacks(drainCallbacks)
667{
668 ++resolutionNestingDepth;
669
670 if (resolutionNestingDepth == 1)
671 platformStrategies()->loaderStrategy()->suspendPendingRequests();
672
673 // FIXME: It's strange to build this into the disabler.
674 suspendMemoryCacheClientCalls(document);
675}
676
677PostResolutionCallbackDisabler::~PostResolutionCallbackDisabler()
678{
679 if (resolutionNestingDepth == 1) {
680 if (m_drainCallbacks == DrainCallbacks::Yes) {
681 // Get size each time through the loop because a callback can add more callbacks to the end of the queue.
682 auto& queue = postResolutionCallbackQueue();
683 for (size_t i = 0; i < queue.size(); ++i)
684 queue[i]();
685 queue.clear();
686 }
687
688 auto& queue = memoryCacheClientCallsResumeQueue();
689 for (size_t i = 0; i < queue.size(); ++i) {
690 if (auto* page = queue[i]->page())
691 page->setMemoryCacheClientCallsEnabled(true);
692 }
693 queue.clear();
694
695 platformStrategies()->loaderStrategy()->resumePendingRequests();
696 }
697
698 --resolutionNestingDepth;
699}
700
701bool postResolutionCallbacksAreSuspended()
702{
703 return resolutionNestingDepth;
704}
