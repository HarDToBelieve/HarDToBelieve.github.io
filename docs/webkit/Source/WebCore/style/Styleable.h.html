35namespace WebCore {
36
37class KeyframeEffectStack;
38class RenderStyle;
39class WebAnimation;
40
41struct Styleable {
42 Element& element;
43 PseudoId pseudoId;
44
45 Styleable(Element& element, PseudoId pseudoId)
46 : element(element)
47 , pseudoId(pseudoId)
48 {
49 ASSERT(!is<PseudoElement>(element));
50 }
51
52 static const Styleable fromElement(Element& element)
53 {
54 if (is<PseudoElement>(element))
55 return Styleable(*downcast<PseudoElement>(element).hostElement(), element.pseudoId());
56 return Styleable(element, element.pseudoId());
57 }
58
59 static const std::optional<const Styleable> fromRenderer(const RenderElement& renderer)
60 {
61 if (auto* element = renderer.element())
62 return fromElement(*element);
63 return std::nullopt;
64 }
65
66 bool operator==(const Styleable& other) const
67 {
68 return (&element == &other.element && pseudoId == other.pseudoId);
69 }
70
71 bool operator!=(const Styleable& other) const
72 {
73 return !(*this == other);
74 }
75
76 KeyframeEffectStack* keyframeEffectStack() const
77 {
78 return element.keyframeEffectStack(pseudoId);
79 }
80
81 KeyframeEffectStack& ensureKeyframeEffectStack() const
82 {
83 return element.ensureKeyframeEffectStack(pseudoId);
84 }
85
86 bool hasKeyframeEffects() const
87 {
88 return element.hasKeyframeEffects(pseudoId);
89 }
90
91 OptionSet<AnimationImpact> applyKeyframeEffects(RenderStyle& targetStyle, const RenderStyle& previousLastStyleChangeEventStyle, const RenderStyle* parentElementStyle) const
92 {
93 return element.ensureKeyframeEffectStack(pseudoId).applyKeyframeEffects(targetStyle, previousLastStyleChangeEventStyle, parentElementStyle);
94 }
95
96 const AnimationCollection* animations() const
97 {
98 return element.animations(pseudoId);
99 }
100
101 bool hasCompletedTransitionForProperty(CSSPropertyID property) const
102 {
103 return element.hasCompletedTransitionForProperty(pseudoId, property);
104 }
105
106 bool hasRunningTransitionForProperty(CSSPropertyID property) const
107 {
108 return element.hasRunningTransitionForProperty(pseudoId, property);
109 }
110
111 bool hasRunningTransitions() const
112 {
113 return element.hasRunningTransitions(pseudoId);
114 }
115
116 AnimationCollection& ensureAnimations() const
117 {
118 return element.ensureAnimations(pseudoId);
119 }
120
121 PropertyToTransitionMap& ensureCompletedTransitionsByProperty() const
122 {
123 return element.ensureCompletedTransitionsByProperty(pseudoId);
124 }
125
126 PropertyToTransitionMap& ensureRunningTransitionsByProperty() const
127 {
128 return element.ensureRunningTransitionsByProperty(pseudoId);
129 }
130
131 CSSAnimationCollection& animationsCreatedByMarkup() const
132 {
133 return element.animationsCreatedByMarkup(pseudoId);
134 }
135
136 void setAnimationsCreatedByMarkup(CSSAnimationCollection&& collection) const
137 {
138 element.setAnimationsCreatedByMarkup(pseudoId, WTFMove(collection));
139 }
140
141 const RenderStyle* lastStyleChangeEventStyle() const
142 {
143 return element.lastStyleChangeEventStyle(pseudoId);
144 }
145
146 void setLastStyleChangeEventStyle(std::unique_ptr<const RenderStyle>&& style) const
147 {
148 element.setLastStyleChangeEventStyle(pseudoId, WTFMove(style));
149 }
150
151 void elementWasRemoved() const;
152
153 void willChangeRenderer() const;
154 void cancelDeclarativeAnimations() const;
155
156 void animationWasAdded(WebAnimation&) const;
157 void animationWasRemoved(WebAnimation&) const;
158
159 void removeDeclarativeAnimationFromListsForOwningElement(WebAnimation&) const;
160
161 void updateCSSAnimations(const RenderStyle* currentStyle, const RenderStyle& afterChangeStyle, const RenderStyle* parentElementStyle) const;
162 void updateCSSTransitions(const RenderStyle& currentStyle, const RenderStyle& newStyle) const;
163};
