101static StyleSheetContents* parseUASheet(const String& str)
102{
103 StyleSheetContents& sheet = StyleSheetContents::create(CSSParserContext(UASheetMode)).leakRef(); // leak the sheet on purpose
104 sheet.parseString(str);
105 return &sheet;
106}

108void UserAgentStyle::addToDefaultStyle(StyleSheetContents& sheet)
109{
110 defaultStyle->addRulesFromSheet(sheet, screenEval());
111 defaultPrintStyle->addRulesFromSheet(sheet, printEval());
112
113 // Build a stylesheet consisting of non-trivial media queries seen in default style.
114 // Rulesets for these can't be global and need to be built in document context.
115 for (auto& rule : sheet.childRules()) {
116 if (!is<StyleRuleMedia>(*rule))
117 continue;
118 auto& mediaRule = downcast<StyleRuleMedia>(*rule);
119 auto& mediaQuery = mediaRule.mediaQueries();
120 if (screenEval().evaluate(mediaQuery, nullptr))
121 continue;
122 if (printEval().evaluate(mediaQuery, nullptr))
123 continue;
124 mediaQueryStyleSheet->parserAppendRule(mediaRule.copy());
125 }
126
127 ++defaultStyleVersion;
128}

130void UserAgentStyle::initDefaultStyleSheet()
131{
132 if (defaultStyle)
133 return;
134
135 defaultStyle = &RuleSet::create().leakRef();
136 defaultPrintStyle = &RuleSet::create().leakRef();
137 defaultQuirksStyle = &RuleSet::create().leakRef();
138 mediaQueryStyleSheet = &StyleSheetContents::create(CSSParserContext(UASheetMode)).leakRef();
139
140 // Strict-mode rules.
141 String defaultRules = String(StringImpl::createWithoutCopying(htmlUserAgentStyleSheet, sizeof(htmlUserAgentStyleSheet))) + RenderTheme::singleton().extraDefaultStyleSheet();
142 defaultStyleSheet = parseUASheet(defaultRules);
143 addToDefaultStyle(*defaultStyleSheet);
144
145 // Quirks-mode rules.
146 String quirksRules = String(StringImpl::createWithoutCopying(quirksUserAgentStyleSheet, sizeof(quirksUserAgentStyleSheet))) + RenderTheme::singleton().extraQuirksStyleSheet();
147 quirksStyleSheet = parseUASheet(quirksRules);
148 defaultQuirksStyle->addRulesFromSheet(*quirksStyleSheet, screenEval());
149
150 ++defaultStyleVersion;
151}

153void UserAgentStyle::ensureDefaultStyleSheetsForElement(const Element& element)
154{
155 if (is<HTMLElement>(element)) {
156 if (is<HTMLObjectElement>(element) || is<HTMLEmbedElement>(element)) {
157 if (!plugInsStyleSheet && element.document().page()) {
158 String plugInsRules = RenderTheme::singleton().extraPlugInsStyleSheet() + element.document().page()->chrome().client().plugInExtraStyleSheet();
159 if (plugInsRules.isEmpty())
160 plugInsRules = String(StringImpl::createWithoutCopying(plugInsUserAgentStyleSheet, sizeof(plugInsUserAgentStyleSheet)));
161 plugInsStyleSheet = parseUASheet(plugInsRules);
162 addToDefaultStyle(*plugInsStyleSheet);
163 }
164 }
165 else if (is<HTMLDialogElement>(element) && RuntimeEnabledFeatures::sharedFeatures().dialogElementEnabled()) {
166 if (!dialogStyleSheet) {
167 dialogStyleSheet = parseUASheet(StringImpl::createWithoutCopying(dialogUserAgentStyleSheet, sizeof(dialogUserAgentStyleSheet)));
168 addToDefaultStyle(*dialogStyleSheet);
169 }
170 }
171#if ENABLE(VIDEO) && !ENABLE(MODERN_MEDIA_CONTROLS)
172 else if (is<HTMLMediaElement>(element)) {
173 if (!mediaControlsStyleSheet) {
174 String mediaRules = RenderTheme::singleton().mediaControlsStyleSheet();
175 if (mediaRules.isEmpty())
176 mediaRules = String(StringImpl::createWithoutCopying(mediaControlsUserAgentStyleSheet, sizeof(mediaControlsUserAgentStyleSheet))) + RenderTheme::singleton().extraMediaControlsStyleSheet();
177 mediaControlsStyleSheet = parseUASheet(mediaRules);
178 addToDefaultStyle(*mediaControlsStyleSheet);
179
180 }
181 }
182#endif // ENABLE(VIDEO) && !ENABLE(MODERN_MEDIA_CONTROLS)
183#if ENABLE(DATALIST_ELEMENT)
184 else if (!dataListStyleSheet && is<HTMLDataListElement>(element)) {
185 dataListStyleSheet = parseUASheet(RenderTheme::singleton().dataListStyleSheet());
186 addToDefaultStyle(*dataListStyleSheet);
187 }
188#endif // ENABLE(DATALIST_ELEMENT)
189#if ENABLE(INPUT_TYPE_COLOR)
190 else if (!colorInputStyleSheet && is<HTMLInputElement>(element) && downcast<HTMLInputElement>(element).isColorControl()) {
191 colorInputStyleSheet = parseUASheet(RenderTheme::singleton().colorInputStyleSheet(element.document().settings()));
192 addToDefaultStyle(*colorInputStyleSheet);
193 }
194#endif // ENABLE(INPUT_TYPE_COLOR)
195 } else if (is<SVGElement>(element)) {
196 if (!svgStyleSheet) {
197 // SVG rules.
198 svgStyleSheet = parseUASheet(StringImpl::createWithoutCopying(svgUserAgentStyleSheet, sizeof(svgUserAgentStyleSheet)));
199 addToDefaultStyle(*svgStyleSheet);
200 }
201 }
202#if ENABLE(MATHML)
203 else if (is<MathMLElement>(element)) {
204 if (!mathMLStyleSheet) {
205 // MathML rules.
206 mathMLStyleSheet = parseUASheet(StringImpl::createWithoutCopying(mathmlUserAgentStyleSheet, sizeof(mathmlUserAgentStyleSheet)));
207 addToDefaultStyle(*mathMLStyleSheet);
208 }
209 }
210#endif // ENABLE(MATHML)
211
212#if ENABLE(FULLSCREEN_API)
213 if (!fullscreenStyleSheet && element.document().fullscreenManager().isFullscreen()) {
214 StringBuilder fullscreenRules;
215 fullscreenRules.appendCharacters(fullscreenUserAgentStyleSheet, sizeof(fullscreenUserAgentStyleSheet));
216 fullscreenRules.append(RenderTheme::singleton().extraFullScreenStyleSheet());
217 if (element.document().quirks().needsBlackFullscreenBackgroundQuirk())
218 fullscreenRules.append(":-webkit-full-screen { background-color: black; }"_s);
219 fullscreenStyleSheet = parseUASheet(fullscreenRules.toString());
220 addToDefaultStyle(*fullscreenStyleSheet);
221 }
222#endif // ENABLE(FULLSCREEN_API)
223
224#if ENABLE(IOS_FORM_CONTROL_REFRESH)
225 if (!legacyFormControlsIOSStyleSheet && !element.document().settings().iOSFormControlRefreshEnabled()) {
226 legacyFormControlsIOSStyleSheet = parseUASheet(StringImpl::createWithoutCopying(legacyFormControlsIOSUserAgentStyleSheet, sizeof(legacyFormControlsIOSUserAgentStyleSheet)));
227 addToDefaultStyle(*legacyFormControlsIOSStyleSheet);
228 }
229#endif
230
231 ASSERT(defaultStyle->features().idsInRules.isEmpty());
232 ASSERT(mathMLStyleSheet || defaultStyle->features().siblingRules.isEmpty());
233}
