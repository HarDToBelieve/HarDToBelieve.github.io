34RefPtr<SVGSVGElement> DocumentSVG::rootElement(const Document& document)
35{
36 auto* element = document.documentElement();
37 if (!is<SVGSVGElement>(element))
38 return nullptr;
39 return downcast<SVGSVGElement>(element);
40}
