33class SVGAElement final : public SVGGraphicsElement, public SVGURIReference {
34 WTF_MAKE_ISO_ALLOCATED(SVGAElement);
35public:
36 static Ref<SVGAElement> create(const QualifiedName&, Document&);
37
38 String target() const final { return m_target->currentValue(); }
39 Ref<SVGAnimatedString>& targetAnimated() { return m_target; }
40
41 SharedStringHash visitedLinkHash() const;
42
43 DOMTokenList& relList();
44
45private:
46 SVGAElement(const QualifiedName&, Document&);
47
48 using PropertyRegistry = SVGPropertyOwnerRegistry<SVGAElement, SVGGraphicsElement, SVGURIReference>;
49 const SVGPropertyRegistry& propertyRegistry() const final { return m_propertyRegistry; }
50
51 void parseAttribute(const QualifiedName&, const AtomString&) final;
52 void svgAttributeChanged(const QualifiedName&) final;
53
54 RenderPtr<RenderElement> createElementRenderer(RenderStyle&&, const RenderTreePosition&) final;
55 bool childShouldCreateRenderer(const Node&) const final;
56
57 bool isValid() const final { return SVGTests::isValid(); }
58 String title() const final;
59 void defaultEventHandler(Event&) final;
60
61 bool supportsFocus() const final;
62 bool isMouseFocusable() const final;
63 bool isKeyboardFocusable(KeyboardEvent*) const final;
64 bool isURLAttribute(const Attribute&) const final;
65 bool canStartSelection() const final;
66 int defaultTabIndex() const final;
67
68 bool willRespondToMouseClickEvents() final;
69
70 PropertyRegistry m_propertyRegistry { *this };
71 Ref<SVGAnimatedString> m_target { SVGAnimatedString::create(this) };
72
73 // This is computed only once and must not be affected by subsequent URL changes.
74 mutable std::optional<SharedStringHash> m_storedVisitedLinkHash;
75
76 std::unique_ptr<DOMTokenList> m_relList;
77};
