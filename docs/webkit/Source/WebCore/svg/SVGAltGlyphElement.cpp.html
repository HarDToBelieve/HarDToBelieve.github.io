80bool SVGAltGlyphElement::hasValidGlyphElements(Vector<String>& glyphNames) const
81{
82 // No need to support altGlyph referencing another node inside a shadow tree.
83 auto target = targetElementFromIRIString(getAttribute(SVGNames::hrefAttr, XLinkNames::hrefAttr), document());
84
85 if (is<SVGGlyphElement>(target.element)) {
86 glyphNames.append(target.identifier);
87 return true;
88 }
89
90 if (!is<SVGAltGlyphDefElement>(target.element))
91 return false;
92
93 return downcast<SVGAltGlyphDefElement>(*target.element).hasValidGlyphElements(glyphNames);
94}
