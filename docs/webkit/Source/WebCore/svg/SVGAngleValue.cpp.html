32float SVGAngleValue::value() const
33{
34 switch (m_unitType) {
35 case SVG_ANGLETYPE_GRAD:
36 return grad2deg(m_valueInSpecifiedUnits);
37 case SVG_ANGLETYPE_RAD:
38 return rad2deg(m_valueInSpecifiedUnits);
39 case SVG_ANGLETYPE_UNSPECIFIED:
40 case SVG_ANGLETYPE_UNKNOWN:
41 case SVG_ANGLETYPE_DEG:
42 return m_valueInSpecifiedUnits;
43 }
44 ASSERT_NOT_REACHED();
45 return 0;
46}
47
48void SVGAngleValue::setValue(float value)
49{
50 switch (m_unitType) {
51 case SVG_ANGLETYPE_GRAD:
52 m_valueInSpecifiedUnits = deg2grad(value);
53 return;
54 case SVG_ANGLETYPE_RAD:
55 m_valueInSpecifiedUnits = deg2rad(value);
56 return;
57 case SVG_ANGLETYPE_UNSPECIFIED:
58 case SVG_ANGLETYPE_UNKNOWN:
59 case SVG_ANGLETYPE_DEG:
60 m_valueInSpecifiedUnits = value;
61 return;
62 }
63 ASSERT_NOT_REACHED();
64}

66String SVGAngleValue::valueAsString() const
67{
68 switch (m_unitType) {
69 case SVG_ANGLETYPE_DEG:
70 return makeString(m_valueInSpecifiedUnits, "deg");
71 case SVG_ANGLETYPE_RAD:
72 return makeString(m_valueInSpecifiedUnits, "rad");
73 case SVG_ANGLETYPE_GRAD:
74 return makeString(m_valueInSpecifiedUnits, "grad");
75 case SVG_ANGLETYPE_UNSPECIFIED:
76 case SVG_ANGLETYPE_UNKNOWN:
77 return String::number(m_valueInSpecifiedUnits);
78 }
79
80 ASSERT_NOT_REACHED();
81 return String();
82}

84template<typename CharacterType> static inline SVGAngleValue::Type parseAngleType(StringParsingBuffer<CharacterType> buffer)
85{
86 switch (buffer.lengthRemaining()) {
87 case 0:
88 return SVGAngleValue::SVG_ANGLETYPE_UNSPECIFIED;
89 case 3:
90 if (buffer[0] == 'd' && buffer[1] == 'e' && buffer[2] == 'g')
91 return SVGAngleValue::SVG_ANGLETYPE_DEG;
92 if (buffer[0] == 'r' && buffer[1] == 'a' && buffer[2] == 'd')
93 return SVGAngleValue::SVG_ANGLETYPE_RAD;
94 break;
95 case 4:
96 if (buffer[0] == 'g' && buffer[1] == 'r' && buffer[2] == 'a' && buffer[3] == 'd')
97 return SVGAngleValue::SVG_ANGLETYPE_GRAD;
98 break;
99 }
100 return SVGAngleValue::SVG_ANGLETYPE_UNKNOWN;
101}

103ExceptionOr<void> SVGAngleValue::setValueAsString(const String& value)
104{
105 if (value.isEmpty()) {
106 m_unitType = SVG_ANGLETYPE_UNSPECIFIED;
107 return { };
108 }
109
110 return readCharactersForParsing(value, [&](auto buffer) -> ExceptionOr<void> {
111 auto valueInSpecifiedUnits = parseNumber(buffer, SuffixSkippingPolicy::DontSkip);
112 if (!valueInSpecifiedUnits)
113 return Exception { SyntaxError };
114
115 auto unitType = parseAngleType(buffer);
116 if (unitType == SVGAngleValue::SVG_ANGLETYPE_UNKNOWN)
117 return Exception { SyntaxError };
118
119 m_unitType = unitType;
120 m_valueInSpecifiedUnits = *valueInSpecifiedUnits;
121 return { };
122 });
123}

125ExceptionOr<void> SVGAngleValue::newValueSpecifiedUnits(unsigned short unitType, float valueInSpecifiedUnits)
126{
127 if (unitType == SVG_ANGLETYPE_UNKNOWN || unitType > SVG_ANGLETYPE_GRAD)
128 return Exception { NotSupportedError };
129
130 m_unitType = static_cast<Type>(unitType);
131 m_valueInSpecifiedUnits = valueInSpecifiedUnits;
132 return { };
133}

135ExceptionOr<void> SVGAngleValue::convertToSpecifiedUnits(unsigned short unitType)
136{
137 if (unitType == SVG_ANGLETYPE_UNKNOWN || m_unitType == SVG_ANGLETYPE_UNKNOWN || unitType > SVG_ANGLETYPE_GRAD)
138 return Exception { NotSupportedError };
139
140 if (unitType == m_unitType)
141 return { };
142
143 switch (m_unitType) {
144 case SVG_ANGLETYPE_RAD:
145 switch (unitType) {
146 case SVG_ANGLETYPE_GRAD:
147 m_valueInSpecifiedUnits = rad2grad(m_valueInSpecifiedUnits);
148 break;
149 case SVG_ANGLETYPE_UNSPECIFIED:
150 case SVG_ANGLETYPE_DEG:
151 m_valueInSpecifiedUnits = rad2deg(m_valueInSpecifiedUnits);
152 break;
153 case SVG_ANGLETYPE_RAD:
154 case SVG_ANGLETYPE_UNKNOWN:
155 ASSERT_NOT_REACHED();
156 break;
157 }
158 break;
159 case SVG_ANGLETYPE_GRAD:
160 switch (unitType) {
161 case SVG_ANGLETYPE_RAD:
162 m_valueInSpecifiedUnits = grad2rad(m_valueInSpecifiedUnits);
163 break;
164 case SVG_ANGLETYPE_UNSPECIFIED:
165 case SVG_ANGLETYPE_DEG:
166 m_valueInSpecifiedUnits = grad2deg(m_valueInSpecifiedUnits);
167 break;
168 case SVG_ANGLETYPE_GRAD:
169 case SVG_ANGLETYPE_UNKNOWN:
170 ASSERT_NOT_REACHED();
171 break;
172 }
173 break;
174 case SVG_ANGLETYPE_UNSPECIFIED:
175 // Spec: For angles, a unitless value is treated the same as if degrees were specified.
176 case SVG_ANGLETYPE_DEG:
177 switch (unitType) {
178 case SVG_ANGLETYPE_RAD:
179 m_valueInSpecifiedUnits = deg2rad(m_valueInSpecifiedUnits);
180 break;
181 case SVG_ANGLETYPE_GRAD:
182 m_valueInSpecifiedUnits = deg2grad(m_valueInSpecifiedUnits);
183 break;
184 case SVG_ANGLETYPE_UNSPECIFIED:
185 case SVG_ANGLETYPE_DEG:
186 break;
187 case SVG_ANGLETYPE_UNKNOWN:
188 ASSERT_NOT_REACHED();
189 break;
190 }
191 break;
192 case SVG_ANGLETYPE_UNKNOWN:
193 ASSERT_NOT_REACHED();
194 break;
195 }
196
197 m_unitType = static_cast<Type>(unitType);
198
199 return { };
200}
