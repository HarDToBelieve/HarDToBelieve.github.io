22#pragma once
23
24#include "ExceptionOr.h"
25#include "SVGPropertyTraits.h"

27namespace WebCore {
28
29class SVGAngleValue {
30 WTF_MAKE_FAST_ALLOCATED;
31public:
32 enum Type {
33 SVG_ANGLETYPE_UNKNOWN = 0,
34 SVG_ANGLETYPE_UNSPECIFIED = 1,
35 SVG_ANGLETYPE_DEG = 2,
36 SVG_ANGLETYPE_RAD = 3,
37 SVG_ANGLETYPE_GRAD = 4
38 };
39
40 Type unitType() const { return m_unitType; }
41
42 void setValue(float);
43 float value() const;
44
45 void setValueInSpecifiedUnits(float valueInSpecifiedUnits) { m_valueInSpecifiedUnits = valueInSpecifiedUnits; }
46 float valueInSpecifiedUnits() const { return m_valueInSpecifiedUnits; }
47
48 ExceptionOr<void> setValueAsString(const String&);
49 String valueAsString() const;
50
51 ExceptionOr<void> newValueSpecifiedUnits(unsigned short unitType, float valueInSpecifiedUnits);
52 ExceptionOr<void> convertToSpecifiedUnits(unsigned short unitType);
53
54private:
55 Type m_unitType { SVG_ANGLETYPE_UNSPECIFIED };
56 float m_valueInSpecifiedUnits { 0 };
57};

59}
