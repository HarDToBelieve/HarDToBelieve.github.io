34inline SVGAnimateTransformElement::SVGAnimateTransformElement(const QualifiedName& tagName, Document& document)
35 : SVGAnimateElementBase(tagName, document)
36 , m_type(SVGTransformValue::SVG_TRANSFORM_UNKNOWN)
37{
38 ASSERT(hasTagName(SVGNames::animateTransformTag));
39}
40
41Ref<SVGAnimateTransformElement> SVGAnimateTransformElement::create(const QualifiedName& tagName, Document& document)
42{
43 return adoptRef(*new SVGAnimateTransformElement(tagName, document));
44}
45
46bool SVGAnimateTransformElement::hasValidAttributeType() const
47{
48 if (!this->targetElement())
49 return false;
50
51 if (attributeType() == AttributeType::CSS)
52 return false;
53
54 return SVGAnimateElementBase::hasValidAttributeType();
55}
56
57void SVGAnimateTransformElement::parseAttribute(const QualifiedName& name, const AtomString& value)
58{
59 if (name == SVGNames::typeAttr) {
60 m_type = SVGTransformable::parseTransformType(value).value_or(SVGTransformValue::SVG_TRANSFORM_UNKNOWN);
61 if (m_type == SVGTransformValue::SVG_TRANSFORM_MATRIX)
62 m_type = SVGTransformValue::SVG_TRANSFORM_UNKNOWN;
63 return;
64 }
65
66 SVGAnimateElementBase::parseAttribute(name, value);
67}
68
69String SVGAnimateTransformElement::animateRangeString(const String& string) const
70{
71 return SVGTransformValue::prefixForTransfromType(m_type) + string + ')';
72}
