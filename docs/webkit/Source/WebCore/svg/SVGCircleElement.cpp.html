33inline SVGCircleElement::SVGCircleElement(const QualifiedName& tagName, Document& document)
34 : SVGGeometryElement(tagName, document)
35{
36 ASSERT(hasTagName(SVGNames::circleTag));
37
38 static std::once_flag onceFlag;
39 std::call_once(onceFlag, [] {
40 PropertyRegistry::registerProperty<SVGNames::cxAttr, &SVGCircleElement::m_cx>();
41 PropertyRegistry::registerProperty<SVGNames::cyAttr, &SVGCircleElement::m_cy>();
42 PropertyRegistry::registerProperty<SVGNames::rAttr, &SVGCircleElement::m_r>();
43 });
44}
45
46Ref<SVGCircleElement> SVGCircleElement::create(const QualifiedName& tagName, Document& document)
47{
48 return adoptRef(*new SVGCircleElement(tagName, document));
49}
50
51void SVGCircleElement::parseAttribute(const QualifiedName& name, const AtomString& value)
52{
53 SVGParsingError parseError = NoError;
54
55 if (name == SVGNames::cxAttr)
56 m_cx->setBaseValInternal(SVGLengthValue::construct(SVGLengthMode::Width, value, parseError));
57 else if (name == SVGNames::cyAttr)
58 m_cy->setBaseValInternal(SVGLengthValue::construct(SVGLengthMode::Height, value, parseError));
59 else if (name == SVGNames::rAttr)
60 m_r->setBaseValInternal(SVGLengthValue::construct(SVGLengthMode::Other, value, parseError, SVGLengthNegativeValuesMode::Forbid));
61
62 reportAttributeParsingError(parseError, name, value);
63
64 SVGGeometryElement::parseAttribute(name, value);
65}
66
67void SVGCircleElement::svgAttributeChanged(const QualifiedName& attrName)
68{
69 if (PropertyRegistry::isKnownAttribute(attrName)) {
70 InstanceInvalidationGuard guard(*this);
71 invalidateSVGPresentationalHintStyle();
72 return;
73 }
74
75 SVGGeometryElement::svgAttributeChanged(attrName);
76}

78RenderPtr<RenderElement> SVGCircleElement::createElementRenderer(RenderStyle&& style, const RenderTreePosition&)
79{
80 return createRenderer<RenderSVGEllipse>(*this, WTFMove(style));
81}
