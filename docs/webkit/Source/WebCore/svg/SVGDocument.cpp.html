34SVGDocument::SVGDocument(Frame* frame, const Settings& settings, const URL& url)
35 : XMLDocument(frame, settings, url, SVGDocumentClass)
36{
37}
38
39bool SVGDocument::zoomAndPanEnabled() const
40{
41 auto element = DocumentSVG::rootElement(*this);
42 if (!element)
43 return false;
44 return (element->useCurrentView() ? element->currentView().zoomAndPan() : element->zoomAndPan()) == SVGZoomAndPanMagnify;
45}
46
47void SVGDocument::startPan(const FloatPoint& start)
48{
49 auto element = DocumentSVG::rootElement(*this);
50 if (!element)
51 return;
52 m_panningOffset = start - element->currentTranslateValue();
53}
54
55void SVGDocument::updatePan(const FloatPoint& position) const
56{
57 auto element = DocumentSVG::rootElement(*this);
58 if (!element)
59 return;
60 element->setCurrentTranslate(position - m_panningOffset);
61}
62
63Ref<Document> SVGDocument::cloneDocumentWithoutChildren() const
64{
65 return create(nullptr, settings(), url());
66}
