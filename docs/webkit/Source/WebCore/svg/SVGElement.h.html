220struct SVGAttributeHashTranslator {
221 static unsigned hash(const QualifiedName& key)
222 {
223 if (key.hasPrefix()) {
224 QualifiedNameComponents components = { nullAtom().impl(), key.localName().impl(), key.namespaceURI().impl() };
225 return hashComponents(components);
226 }
227 return DefaultHash<QualifiedName>::hash(key);
228 }
229 static bool equal(const QualifiedName& a, const QualifiedName& b) { return a.matches(b); }
230};
