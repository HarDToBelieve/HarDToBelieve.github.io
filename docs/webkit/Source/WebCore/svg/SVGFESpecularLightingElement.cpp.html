38inline SVGFESpecularLightingElement::SVGFESpecularLightingElement(const QualifiedName& tagName, Document& document)
39 : SVGFilterPrimitiveStandardAttributes(tagName, document)
40{
41 ASSERT(hasTagName(SVGNames::feSpecularLightingTag));
42
43 static std::once_flag onceFlag;
44 std::call_once(onceFlag, [] {
45 PropertyRegistry::registerProperty<SVGNames::inAttr, &SVGFESpecularLightingElement::m_in1>();
46 PropertyRegistry::registerProperty<SVGNames::specularConstantAttr, &SVGFESpecularLightingElement::m_specularConstant>();
47 PropertyRegistry::registerProperty<SVGNames::specularExponentAttr, &SVGFESpecularLightingElement::m_specularExponent>();
48 PropertyRegistry::registerProperty<SVGNames::surfaceScaleAttr, &SVGFESpecularLightingElement::m_surfaceScale>();
49 PropertyRegistry::registerProperty<SVGNames::kernelUnitLengthAttr, &SVGFESpecularLightingElement::m_kernelUnitLengthX, &SVGFESpecularLightingElement::m_kernelUnitLengthY>();
50 });
51}

58void SVGFESpecularLightingElement::parseAttribute(const QualifiedName& name, const AtomString& value)
59{
60 if (name == SVGNames::inAttr) {
61 m_in1->setBaseValInternal(value);
62 return;
63 }
64
65 if (name == SVGNames::surfaceScaleAttr) {
66 m_surfaceScale->setBaseValInternal(value.toFloat());
67 return;
68 }
69
70 if (name == SVGNames::specularConstantAttr) {
71 m_specularConstant->setBaseValInternal(value.toFloat());
72 return;
73 }
74
75 if (name == SVGNames::specularExponentAttr) {
76 m_specularExponent->setBaseValInternal(value.toFloat());
77 return;
78 }
79
80 if (name == SVGNames::kernelUnitLengthAttr) {
81 if (auto result = parseNumberOptionalNumber(value)) {
82 m_kernelUnitLengthX->setBaseValInternal(result->first);
83 m_kernelUnitLengthY->setBaseValInternal(result->second);
84 }
85 return;
86 }
87
88 SVGFilterPrimitiveStandardAttributes::parseAttribute(name, value);
89}

91bool SVGFESpecularLightingElement::setFilterEffectAttribute(FilterEffect* effect, const QualifiedName& attrName)
92{
93 FESpecularLighting* specularLighting = static_cast<FESpecularLighting*>(effect);
94
95 if (attrName == SVGNames::lighting_colorAttr) {
96 RenderObject* renderer = this->renderer();
97 ASSERT(renderer);
98 Color color = renderer->style().colorByApplyingColorFilter(renderer->style().svgStyle().lightingColor());
99 return specularLighting->setLightingColor(color);
100 }
101 if (attrName == SVGNames::surfaceScaleAttr)
102 return specularLighting->setSurfaceScale(surfaceScale());
103 if (attrName == SVGNames::specularConstantAttr)
104 return specularLighting->setSpecularConstant(specularConstant());
105 if (attrName == SVGNames::specularExponentAttr)
106 return specularLighting->setSpecularExponent(specularExponent());
107
108 auto& lightSource = const_cast<LightSource&>(specularLighting->lightSource());
109 const SVGFELightElement* lightElement = SVGFELightElement::findLightElement(this);
110 ASSERT(lightElement);
111
112 if (attrName == SVGNames::azimuthAttr)
113 return lightSource.setAzimuth(lightElement->azimuth());
114 if (attrName == SVGNames::elevationAttr)
115 return lightSource.setElevation(lightElement->elevation());
116 if (attrName == SVGNames::xAttr)
117 return lightSource.setX(lightElement->x());
118 if (attrName == SVGNames::yAttr)
119 return lightSource.setY(lightElement->y());
120 if (attrName == SVGNames::zAttr)
121 return lightSource.setZ(lightElement->z());
122 if (attrName == SVGNames::pointsAtXAttr)
123 return lightSource.setPointsAtX(lightElement->pointsAtX());
124 if (attrName == SVGNames::pointsAtYAttr)
125 return lightSource.setPointsAtY(lightElement->pointsAtY());
126 if (attrName == SVGNames::pointsAtZAttr)
127 return lightSource.setPointsAtZ(lightElement->pointsAtZ());
128 if (attrName == SVGNames::specularExponentAttr)
129 return lightSource.setSpecularExponent(lightElement->specularExponent());
130 if (attrName == SVGNames::limitingConeAngleAttr)
131 return lightSource.setLimitingConeAngle(lightElement->limitingConeAngle());
132
133 ASSERT_NOT_REACHED();
134 return false;
135}

137void SVGFESpecularLightingElement::svgAttributeChanged(const QualifiedName& attrName)
138{
139 if (attrName == SVGNames::surfaceScaleAttr || attrName == SVGNames::specularConstantAttr || attrName == SVGNames::specularExponentAttr || attrName == SVGNames::kernelUnitLengthAttr) {
140 InstanceInvalidationGuard guard(*this);
141 primitiveAttributeChanged(attrName);
142 return;
143 }
144
145 if (attrName == SVGNames::inAttr) {
146 InstanceInvalidationGuard guard(*this);
147 invalidate();
148 return;
149 }
150
151 SVGFilterPrimitiveStandardAttributes::svgAttributeChanged(attrName);
152}

154void SVGFESpecularLightingElement::lightElementAttributeChanged(const SVGFELightElement* lightElement, const QualifiedName& attrName)
155{
156 if (SVGFELightElement::findLightElement(this) != lightElement)
157 return;
158
159 // The light element has different attribute names so attrName can identify the requested attribute.
160 primitiveAttributeChanged(attrName);
161}

163RefPtr<FilterEffect> SVGFESpecularLightingElement::build(SVGFilterBuilder* filterBuilder, Filter& filter) const
164{
165 auto input1 = filterBuilder->getEffectById(in1());
166
167 if (!input1)
168 return nullptr;
169
170 auto lightElement = makeRefPtr(SVGFELightElement::findLightElement(this));
171 if (!lightElement)
172 return nullptr;
173
174 auto lightSource = lightElement->lightSource(*filterBuilder);
175
176 RenderObject* renderer = this->renderer();
177 if (!renderer)
178 return nullptr;
179
180 Color color = renderer->style().colorByApplyingColorFilter(renderer->style().svgStyle().lightingColor());
181
182 auto effect = FESpecularLighting::create(filter, color, surfaceScale(), specularConstant(), specularExponent(), kernelUnitLengthX(), kernelUnitLengthY(), WTFMove(lightSource));
183 effect->inputEffects() = { input1 };
184 return effect;
185}
