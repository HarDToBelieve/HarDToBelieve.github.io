31class SVGFESpecularLightingElement final : public SVGFilterPrimitiveStandardAttributes {
32 WTF_MAKE_ISO_ALLOCATED(SVGFESpecularLightingElement);
33public:
34 static Ref<SVGFESpecularLightingElement> create(const QualifiedName&, Document&);
35 void lightElementAttributeChanged(const SVGFELightElement*, const QualifiedName&);
36
37 String in1() const { return m_in1->currentValue(); }
38 float specularConstant() const { return m_specularConstant->currentValue(); }
39 float specularExponent() const { return m_specularExponent->currentValue(); }
40 float surfaceScale() const { return m_surfaceScale->currentValue(); }
41 float kernelUnitLengthX() const { return m_kernelUnitLengthX->currentValue(); }
42 float kernelUnitLengthY() const { return m_kernelUnitLengthY->currentValue(); }
43
44 SVGAnimatedString& in1Animated() { return m_in1; }
45 SVGAnimatedNumber& specularConstantAnimated() { return m_specularConstant; }
46 SVGAnimatedNumber& specularExponentAnimated() { return m_specularExponent; }
47 SVGAnimatedNumber& surfaceScaleAnimated() { return m_surfaceScale; }
48 SVGAnimatedNumber& kernelUnitLengthXAnimated() { return m_kernelUnitLengthX; }
49 SVGAnimatedNumber& kernelUnitLengthYAnimated() { return m_kernelUnitLengthY; }
50
51private:
52 SVGFESpecularLightingElement(const QualifiedName&, Document&);
53
54 using PropertyRegistry = SVGPropertyOwnerRegistry<SVGFESpecularLightingElement, SVGFilterPrimitiveStandardAttributes>;
55 const SVGPropertyRegistry& propertyRegistry() const final { return m_propertyRegistry; }
56
57 void parseAttribute(const QualifiedName&, const AtomString&) override;
58 void svgAttributeChanged(const QualifiedName&) override;
59
60 bool setFilterEffectAttribute(FilterEffect*, const QualifiedName&) override;
61 RefPtr<FilterEffect> build(SVGFilterBuilder*, Filter&) const override;
62
63 PropertyRegistry m_propertyRegistry { *this };
64 Ref<SVGAnimatedString> m_in1 { SVGAnimatedString::create(this) };
65 Ref<SVGAnimatedNumber> m_specularConstant { SVGAnimatedNumber::create(this, 1) };
66 Ref<SVGAnimatedNumber> m_specularExponent { SVGAnimatedNumber::create(this, 1) };
67 Ref<SVGAnimatedNumber> m_surfaceScale { SVGAnimatedNumber::create(this, 1) };
68 Ref<SVGAnimatedNumber> m_kernelUnitLengthX { SVGAnimatedNumber::create(this) };
69 Ref<SVGAnimatedNumber> m_kernelUnitLengthY { SVGAnimatedNumber::create(this) };
70};
