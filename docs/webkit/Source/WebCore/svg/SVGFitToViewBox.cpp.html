38SVGFitToViewBox::SVGFitToViewBox(SVGElement* contextElement, SVGPropertyAccess access)
39 : m_viewBox(SVGAnimatedRect::create(contextElement, access))
40 , m_preserveAspectRatio(SVGAnimatedPreserveAspectRatio::create(contextElement, access))
41{
42 static std::once_flag onceFlag;
43 std::call_once(onceFlag, [] {
44 PropertyRegistry::registerProperty<SVGNames::viewBoxAttr, &SVGFitToViewBox::m_viewBox>();
45 PropertyRegistry::registerProperty<SVGNames::preserveAspectRatioAttr, &SVGFitToViewBox::m_preserveAspectRatio>();
46 });
47}

67bool SVGFitToViewBox::parseAttribute(const QualifiedName& name, const AtomString& value)
68{
69 if (name == SVGNames::viewBoxAttr) {
70 if (!value.isNull()) {
71 if (auto result = parseViewBox(value)) {
72 setViewBox(WTFMove(*result));
73 return true;
74 }
75 }
76 resetViewBox();
77 return true;
78 }
79
80 if (name == SVGNames::preserveAspectRatioAttr) {
81 setPreserveAspectRatio(SVGPreserveAspectRatioValue { value });
82 return true;
83 }
84
85 return false;
86}

105template<typename CharacterType> std::optional<FloatRect> SVGFitToViewBox::parseViewBoxGeneric(StringParsingBuffer<CharacterType>& buffer, bool validate)
106{
107 StringView stringToParse = buffer.stringViewOfCharactersRemaining();
108
109 skipOptionalSVGSpaces(buffer);
110
111 auto x = parseNumber(buffer);
112 auto y = parseNumber(buffer);
113 auto width = parseNumber(buffer);
114 auto height = parseNumber(buffer, SuffixSkippingPolicy::DontSkip);
115
116 if (validate) {
117 Document& document = m_viewBox->contextElement()->document();
118
119 if (!x || !y || !width || !height) {
120 document.accessSVGExtensions().reportWarning(makeString("Problem parsing viewBox=\"", stringToParse, "\""));
121 return std::nullopt;
122 }
123
124 // Check that width is positive.
125 if (*width < 0.0) {
126 document.accessSVGExtensions().reportError("A negative value for ViewBox width is not allowed");
127 return std::nullopt;
128 }
129
130 // Check that height is positive.
131 if (*height < 0.0) {
132 document.accessSVGExtensions().reportError("A negative value for ViewBox height is not allowed");
133 return std::nullopt;
134 }
135
136 // Nothing should come after the last, fourth number.
137 skipOptionalSVGSpaces(buffer);
138 if (buffer.hasCharactersRemaining()) {
139 document.accessSVGExtensions().reportWarning(makeString("Problem parsing viewBox=\"", stringToParse, "\""));
140 return std::nullopt;
141 }
142 }
143
144 return FloatRect { x.value_or(0), y.value_or(0), width.value_or(0), height.value_or(0) };
145}

147AffineTransform SVGFitToViewBox::viewBoxToViewTransform(const FloatRect& viewBoxRect, const SVGPreserveAspectRatioValue& preserveAspectRatio, float viewWidth, float viewHeight)
148{
149 if (!viewBoxRect.width() || !viewBoxRect.height() || !viewWidth || !viewHeight)
150 return AffineTransform();
151
152 return preserveAspectRatio.getCTM(viewBoxRect.x(), viewBoxRect.y(), viewBoxRect.width(), viewBoxRect.height(), viewWidth, viewHeight);
153}
