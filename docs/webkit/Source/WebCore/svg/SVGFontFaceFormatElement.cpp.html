33inline SVGFontFaceFormatElement::SVGFontFaceFormatElement(const QualifiedName& tagName, Document& document)

40{
41 return adoptRef(*new SVGFontFaceFormatElement(tagName, document));
42}
43
44void SVGFontFaceFormatElement::childrenChanged(const ChildChange& change)
45{
46 SVGElement::childrenChanged(change);
47
48 if (!parentNode() || !parentNode()->hasTagName(font_face_uriTag))
49 return;
50
51 auto ancestor = makeRefPtr(parentNode()->parentNode());
52 if (!ancestor || !ancestor->hasTagName(font_face_srcTag))
53 return;
54
55 ancestor = ancestor->parentNode();
56 if (ancestor && ancestor->hasTagName(font_faceTag))
57 downcast<SVGFontFaceElement>(*ancestor).rebuildFontFace();
58}
