49Ref<CSSValueList> SVGFontFaceSrcElement::srcValue() const
50{
51 Ref<CSSValueList> list = CSSValueList::createCommaSeparated();
52 for (auto& child : childrenOfType<SVGElement>(*this)) {
53 RefPtr<CSSFontFaceSrcValue> srcValue;
54 if (is<SVGFontFaceUriElement>(child))
55 srcValue = downcast<SVGFontFaceUriElement>(child).srcValue();
56 else if (is<SVGFontFaceNameElement>(child))
57 srcValue = downcast<SVGFontFaceNameElement>(child).srcValue();
58 if (srcValue && srcValue->resource().length())
59 list->append(srcValue.releaseNonNull());
60 }
61 return list;
62}

64void SVGFontFaceSrcElement::childrenChanged(const ChildChange& change)
65{
66 SVGElement::childrenChanged(change);
67 if (is<SVGFontFaceElement>(parentNode()))
68 downcast<SVGFontFaceElement>(*parentNode()).rebuildFontFace();
69}
