73AffineTransform SVGGraphicsElement::animatedLocalTransform() const
74{
75 AffineTransform matrix;
76
77 auto* style = renderer() ? &renderer()->style() : nullptr;
78 bool hasSpecifiedTransform = style && style->hasTransform();
79
80 // Honor any of the transform-related CSS properties if set.
81 if (hasSpecifiedTransform || (style && (style->translate() || style->scale() || style->rotate()))) {
82
83 FloatRect boundingBox;
84 switch (style->transformBox()) {
85 case TransformBox::BorderBox:
86 // For SVG elements without an associated CSS layout box, the used value for border-box is stroke-box.
87 case TransformBox::StrokeBox:
88 boundingBox = renderer()->strokeBoundingBox();
89 break;
90 case TransformBox::ContentBox:
91 // For SVG elements without an associated CSS layout box, the used value for content-box is fill-box.
92 case TransformBox::FillBox:
93 boundingBox = renderer()->objectBoundingBox();
94 break;
95 case TransformBox::ViewBox: {
96 FloatSize viewportSize;
97 SVGLengthContext(this).determineViewport(viewportSize);
98 boundingBox.setSize(viewportSize);
99 break;
100 }
101 }
102
103 // Note: objectBoundingBox is an emptyRect for elements like pattern or clipPath.
104 // See the "Object bounding box units" section of http://dev.w3.org/csswg/css3-transforms/
105 TransformationMatrix transform;
106 style->applyTransform(transform, boundingBox);
107
108 // Flatten any 3D transform.
109 matrix = transform.toAffineTransform();
110 // CSS bakes the zoom factor into lengths, including translation components.
111 // In order to align CSS & SVG transforms, we need to invert this operation.
112 float zoom = style->effectiveZoom();
113 if (zoom != 1) {
114 matrix.setE(matrix.e() / zoom);
115 matrix.setF(matrix.f() / zoom);
116 }
117
118 }
119
120 // If we didn't have the CSS "transform" property set, we must account for the "transform" attribute.
121 if (!hasSpecifiedTransform)
122 matrix *= transform().concatenate();
123
124 if (m_supplementalTransform)
125 return *m_supplementalTransform * matrix;
126 return matrix;
127}

136void SVGGraphicsElement::parseAttribute(const QualifiedName& name, const AtomString& value)
137{
138 if (name == SVGNames::transformAttr) {
139 m_transform->baseVal()->parse(value);
140 return;
141 }
142
143 SVGElement::parseAttribute(name, value);
144 SVGTests::parseAttribute(name, value);
145}
146
147void SVGGraphicsElement::svgAttributeChanged(const QualifiedName& attrName)
148{
149 if (attrName == SVGNames::transformAttr) {
150 InstanceInvalidationGuard guard(*this);
151
152 if (auto renderer = this->renderer()) {
153 renderer->setNeedsTransformUpdate();
154 RenderSVGResource::markForLayoutAndParentResourceInvalidation(*renderer);
155 }
156
157 return;
158 }
159
160 SVGElement::svgAttributeChanged(attrName);
161 SVGTests::svgAttributeChanged(attrName);
162}

190{
