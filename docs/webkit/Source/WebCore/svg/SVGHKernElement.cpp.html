45std::optional<SVGKerningPair> SVGHKernElement::buildHorizontalKerningPair() const
46{
47 // FIXME: Can this be shared with SVGVKernElement::buildVerticalKerningPair?
48
49 auto& u1 = attributeWithoutSynchronization(SVGNames::u1Attr);
50 auto& g1 = attributeWithoutSynchronization(SVGNames::g1Attr);
51 if (u1.isEmpty() && g1.isEmpty())
52 return std::nullopt;
53
54 auto& u2 = attributeWithoutSynchronization(SVGNames::u2Attr);
55 auto& g2 = attributeWithoutSynchronization(SVGNames::g2Attr);
56 if (u2.isEmpty() && g2.isEmpty())
57 return std::nullopt;
58
59 auto glyphName1 = parseGlyphName(g1);
60 if (!glyphName1)
61 return std::nullopt;
62 auto glyphName2 = parseGlyphName(g2);
63 if (!glyphName2)
64 return std::nullopt;
65 auto unicodeString1 = parseKerningUnicodeString(u1);
66 if (!unicodeString1)
67 return std::nullopt;
68 auto unicodeString2 = parseKerningUnicodeString(u2);
69 if (!unicodeString2)
70 return std::nullopt;
71
72 bool ok = false;
73 auto kerning = attributeWithoutSynchronization(SVGNames::kAttr).string().toFloat(&ok);
74 if (!ok)
75 return std::nullopt;
76
77 return SVGKerningPair {
78 WTFMove(unicodeString1->first),
79 WTFMove(unicodeString1->second),
80 WTFMove(*glyphName1),
81 WTFMove(unicodeString2->first),
82 WTFMove(unicodeString2->second),
83 WTFMove(*glyphName2),
84 kerning
85 };
86}
