40inline SVGImageElement::SVGImageElement(const QualifiedName& tagName, Document& document)
41 : SVGGraphicsElement(tagName, document)
42 , SVGURIReference(this)
43 , m_imageLoader(*this)
44{
45 static std::once_flag onceFlag;
46 std::call_once(onceFlag, [] {
47 PropertyRegistry::registerProperty<SVGNames::xAttr, &SVGImageElement::m_x>();
48 PropertyRegistry::registerProperty<SVGNames::yAttr, &SVGImageElement::m_y>();
49 PropertyRegistry::registerProperty<SVGNames::widthAttr, &SVGImageElement::m_width>();
50 PropertyRegistry::registerProperty<SVGNames::heightAttr, &SVGImageElement::m_height>();
51 PropertyRegistry::registerProperty<SVGNames::preserveAspectRatioAttr, &SVGImageElement::m_preserveAspectRatio>();
52 });
53}

60bool SVGImageElement::hasSingleSecurityOrigin() const
61{
62 auto* renderer = downcast<RenderSVGImage>(this->renderer());
63 if (!renderer || !renderer->imageResource().cachedImage())
64 return true;
65 auto* image = renderer->imageResource().cachedImage()->image();
66 return !image || image->hasSingleSecurityOrigin();
67}

69void SVGImageElement::parseAttribute(const QualifiedName& name, const AtomString& value)
70{
71 if (name == SVGNames::preserveAspectRatioAttr) {
72 m_preserveAspectRatio->setBaseValInternal(SVGPreserveAspectRatioValue { value });
73 return;
74 }
75
76 SVGParsingError parseError = NoError;
77
78 if (name == SVGNames::xAttr)
79 m_x->setBaseValInternal(SVGLengthValue::construct(SVGLengthMode::Width, value, parseError));
80 else if (name == SVGNames::yAttr)
81 m_y->setBaseValInternal(SVGLengthValue::construct(SVGLengthMode::Height, value, parseError));
82 else if (name == SVGNames::widthAttr)
83 m_width->setBaseValInternal(SVGLengthValue::construct(SVGLengthMode::Width, value, parseError, SVGLengthNegativeValuesMode::Forbid));
84 else if (name == SVGNames::heightAttr)
85 m_height->setBaseValInternal(SVGLengthValue::construct(SVGLengthMode::Height, value, parseError, SVGLengthNegativeValuesMode::Forbid));
86
87 reportAttributeParsingError(parseError, name, value);
88
89 SVGGraphicsElement::parseAttribute(name, value);
90 SVGURIReference::parseAttribute(name, value);
91}

93void SVGImageElement::svgAttributeChanged(const QualifiedName& attrName)
94{
95 if (attrName == SVGNames::xAttr || attrName == SVGNames::yAttr) {
96 InstanceInvalidationGuard guard(*this);
97 updateRelativeLengthsInformation();
98
99 if (auto* renderer = this->renderer()) {
100 if (downcast<RenderSVGImage>(*renderer).updateImageViewport())
101 RenderSVGResource::markForLayoutAndParentResourceInvalidation(*renderer);
102 }
103 return;
104 }
105
106 if (attrName == SVGNames::widthAttr || attrName == SVGNames::heightAttr) {
107 InstanceInvalidationGuard guard(*this);
108 invalidateSVGPresentationalHintStyle();
109 return;
110 }
111
112 if (attrName == SVGNames::preserveAspectRatioAttr) {
113 InstanceInvalidationGuard guard(*this);
114 if (auto* renderer = this->renderer())
115 RenderSVGResource::markForLayoutAndParentResourceInvalidation(*renderer);
116 return;
117 }
118
119 if (SVGURIReference::isKnownAttribute(attrName)) {
120 m_imageLoader.updateFromElementIgnoringPreviousError();
121 return;
122 }
123
124 SVGGraphicsElement::svgAttributeChanged(attrName);
125}

137void SVGImageElement::didAttachRenderers()
138{
139 if (auto* imageObj = downcast<RenderSVGImage>(renderer())) {
140 if (imageObj->imageResource().cachedImage())
141 return;
142
143 imageObj->imageResource().setCachedImage(m_imageLoader.image());
144 }
145}

147Node::InsertedIntoAncestorResult SVGImageElement::insertedIntoAncestor(InsertionType insertionType, ContainerNode& parentOfInsertedTree)
148{
149 SVGGraphicsElement::insertedIntoAncestor(insertionType, parentOfInsertedTree);
150 if (!insertionType.connectedToDocument)
151 return InsertedIntoAncestorResult::Done;
152 // Update image loader, as soon as we're living in the tree.
153 // We can only resolve base URIs properly, after that!
154 m_imageLoader.updateFromElement();
155 return InsertedIntoAncestorResult::Done;
156}
