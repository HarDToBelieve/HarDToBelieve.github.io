23#include "ExceptionOr.h"

28namespace WebCore {
29
30class SVGElement;
31
32struct Length;
33
34class SVGLengthContext {
35public:
36 explicit SVGLengthContext(const SVGElement*);
37
38 template<typename T>
39 static FloatRect resolveRectangle(const T* context, SVGUnitTypes::SVGUnitType type, const FloatRect& viewport)
40 {
41 return SVGLengthContext::resolveRectangle(context, type, viewport, context->x(), context->y(), context->width(), context->height());
42 }
43
44 static FloatRect resolveRectangle(const SVGElement*, SVGUnitTypes::SVGUnitType, const FloatRect& viewport, const SVGLengthValue& x, const SVGLengthValue& y, const SVGLengthValue& width, const SVGLengthValue& height);
45 static FloatPoint resolvePoint(const SVGElement*, SVGUnitTypes::SVGUnitType, const SVGLengthValue& x, const SVGLengthValue& y);
46 static float resolveLength(const SVGElement*, SVGUnitTypes::SVGUnitType, const SVGLengthValue&);
47
48 float valueForLength(const Length&, SVGLengthMode = SVGLengthMode::Other);
49 ExceptionOr<float> convertValueToUserUnits(float, SVGLengthType, SVGLengthMode) const;
50 ExceptionOr<float> convertValueFromUserUnits(float, SVGLengthType, SVGLengthMode) const;
51
52 bool determineViewport(FloatSize&) const;
53
54private:
55 SVGLengthContext(const SVGElement*, const FloatRect& viewport);
56
57 ExceptionOr<float> convertValueFromUserUnitsToPercentage(float value, SVGLengthMode) const;
58 ExceptionOr<float> convertValueFromPercentageToUserUnits(float value, SVGLengthMode) const;
59
60 ExceptionOr<float> convertValueFromUserUnitsToEMS(float value) const;
61 ExceptionOr<float> convertValueFromEMSToUserUnits(float value) const;
62
63 ExceptionOr<float> convertValueFromUserUnitsToEXS(float value) const;
64 ExceptionOr<float> convertValueFromEXSToUserUnits(float value) const;
65
66 const SVGElement* m_context;
67 FloatRect m_overriddenViewport;
68};
