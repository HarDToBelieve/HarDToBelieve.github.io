26#pragma once
27
28#include "SVGLength.h"
29#include "SVGValuePropertyList.h"
30
31namespace WebCore {
32
33class SVGLengthList final : public SVGValuePropertyList<SVGLength> {
34 using Base = SVGValuePropertyList<SVGLength>;
35 using Base::Base;
36
37public:
38 static Ref<SVGLengthList> create(SVGLengthMode lengthMode = SVGLengthMode::Other)
39 {
40 return adoptRef(*new SVGLengthList(lengthMode));
41 }
42
43 static Ref<SVGLengthList> create(SVGPropertyOwner* owner, SVGPropertyAccess access, SVGLengthMode lengthMode)
44 {
45 return adoptRef(*new SVGLengthList(owner, access, lengthMode));
46 }
47
48 static Ref<SVGLengthList> create(const SVGLengthList& other, SVGPropertyAccess access)
49 {
50 return adoptRef(*new SVGLengthList(other, access));
51 }
52
53 SVGLengthMode lengthMode() const { return m_lengthMode; }
54
55 bool parse(StringView);
56
57 String valueAsString() const override;
58
59private:
60 SVGLengthList(SVGLengthMode lengthMode)
61 : m_lengthMode(lengthMode)
62 {
63 }
64
65 SVGLengthList(SVGPropertyOwner* owner, SVGPropertyAccess access, SVGLengthMode lengthMode)
66 : Base(owner, access)
67 , m_lengthMode(lengthMode)
68 {
69 }
70
71 SVGLengthList(const SVGLengthList& other, SVGPropertyAccess access)
72 : Base(other, access)
73 , m_lengthMode(other.lengthMode())
74 {
75 }
76
77 SVGLengthMode m_lengthMode { SVGLengthMode::Other };
78};
79
80} // namespace WebCore
