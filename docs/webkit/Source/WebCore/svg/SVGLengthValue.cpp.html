35static inline const char* lengthTypeToString(SVGLengthType lengthType)
36{
37 switch (lengthType) {
38 case SVGLengthType::Unknown:
39 case SVGLengthType::Number:
40 return "";
41 case SVGLengthType::Percentage:
42 return "%";
43 case SVGLengthType::Ems:
44 return "em";
45 case SVGLengthType::Exs:
46 return "ex";
47 case SVGLengthType::Pixels:
48 return "px";
49 case SVGLengthType::Centimeters:
50 return "cm";
51 case SVGLengthType::Millimeters:
52 return "mm";
53 case SVGLengthType::Inches:
54 return "in";
55 case SVGLengthType::Points:
56 return "pt";
57 case SVGLengthType::Picas:
58 return "pc";
59 }
60
61 ASSERT_NOT_REACHED();
62 return "";
63}

65template<typename CharacterType> static inline SVGLengthType parseLengthType(StringParsingBuffer<CharacterType>& buffer)
66{
67 if (buffer.atEnd())
68 return SVGLengthType::Number;
69
70 auto firstChar = *buffer++;
71
72 if (buffer.atEnd())
73 return firstChar == '%' ? SVGLengthType::Percentage : SVGLengthType::Unknown;
74
75 auto secondChar = *buffer++;
76
77 if (!buffer.atEnd())
78 return SVGLengthType::Unknown;
79
80 if (firstChar == 'e' && secondChar == 'm')
81 return SVGLengthType::Ems;
82 if (firstChar == 'e' && secondChar == 'x')
83 return SVGLengthType::Exs;
84 if (firstChar == 'p' && secondChar == 'x')
85 return SVGLengthType::Pixels;
86 if (firstChar == 'c' && secondChar == 'm')
87 return SVGLengthType::Centimeters;
88 if (firstChar == 'm' && secondChar == 'm')
89 return SVGLengthType::Millimeters;
90 if (firstChar == 'i' && secondChar == 'n')
91 return SVGLengthType::Inches;
92 if (firstChar == 'p' && secondChar == 't')
93 return SVGLengthType::Points;
94 if (firstChar == 'p' && secondChar == 'c')
95 return SVGLengthType::Picas;
96
97 return SVGLengthType::Unknown;
98}

100static inline SVGLengthType primitiveTypeToLengthType(CSSUnitType primitiveType)
101{
102 switch (primitiveType) {
103 case CSSUnitType::CSS_UNKNOWN:
104 return SVGLengthType::Unknown;
105 case CSSUnitType::CSS_NUMBER:
106 return SVGLengthType::Number;
107 case CSSUnitType::CSS_PERCENTAGE:
108 return SVGLengthType::Percentage;
109 case CSSUnitType::CSS_EMS:
110 return SVGLengthType::Ems;
111 case CSSUnitType::CSS_EXS:
112 return SVGLengthType::Exs;
113 case CSSUnitType::CSS_PX:
114 return SVGLengthType::Pixels;
115 case CSSUnitType::CSS_CM:
116 return SVGLengthType::Centimeters;
117 case CSSUnitType::CSS_MM:
118 return SVGLengthType::Millimeters;
119 case CSSUnitType::CSS_IN:
120 return SVGLengthType::Inches;
121 case CSSUnitType::CSS_PT:
122 return SVGLengthType::Points;
123 case CSSUnitType::CSS_PC:
124 return SVGLengthType::Picas;
125 default:
126 return SVGLengthType::Unknown;
127 }
128
129 return SVGLengthType::Unknown;
130}

193{
194 SVGLengthValue length(lengthMode);
195

204SVGLengthValue SVGLengthValue::blend(const SVGLengthValue& from, const SVGLengthValue& to, float progress)
205{
206 if ((from.isZero() && to.isZero())
207 || from.lengthType() == SVGLengthType::Unknown
208 || to.lengthType() == SVGLengthType::Unknown
209 || (!from.isZero() && from.lengthType() != SVGLengthType::Percentage && to.lengthType() == SVGLengthType::Percentage)
210 || (!to.isZero() && from.lengthType() == SVGLengthType::Percentage && to.lengthType() != SVGLengthType::Percentage)
211 || (!from.isZero() && !to.isZero() && (from.lengthType() == SVGLengthType::Ems || from.lengthType() == SVGLengthType::Exs) && from.lengthType() != to.lengthType()))
212 return to;
213
214 if (from.lengthType() == SVGLengthType::Percentage || to.lengthType() == SVGLengthType::Percentage) {
215 auto fromPercent = from.valueAsPercentage() * 100;
216 auto toPercent = to.valueAsPercentage() * 100;
217 return { WebCore::blend(fromPercent, toPercent, { progress }), SVGLengthType::Percentage };
218 }
219
220 if (from.lengthType() == to.lengthType() || from.isZero() || to.isZero() || from.isRelative()) {
221 auto fromValue = from.valueInSpecifiedUnits();
222 auto toValue = to.valueInSpecifiedUnits();
223 return { WebCore::blend(fromValue, toValue, { progress }), to.isZero() ? from.lengthType() : to.lengthType() };
224 }
225
226 SVGLengthContext nonRelativeLengthContext(nullptr);
227 auto fromValueInUserUnits = nonRelativeLengthContext.convertValueToUserUnits(from.valueInSpecifiedUnits(), from.lengthType(), from.lengthMode());
228 if (fromValueInUserUnits.hasException())
229 return { };
230
231 auto fromValue = nonRelativeLengthContext.convertValueFromUserUnits(fromValueInUserUnits.releaseReturnValue(), to.lengthType(), to.lengthMode());
232 if (fromValue.hasException())
233 return { };
234
235 float toValue = to.valueInSpecifiedUnits();
236 return { WebCore::blend(fromValue.releaseReturnValue(), toValue, { progress }), to.lengthType() };
237}

239SVGLengthValue SVGLengthValue::fromCSSPrimitiveValue(const CSSPrimitiveValue& value)
240{
241 // FIXME: This needs to call value.computeLength() so it can correctly resolve non-absolute units (webkit.org/b/204826).
242 SVGLengthType lengthType = primitiveTypeToLengthType(value.primitiveType());
243 return lengthType == SVGLengthType::Unknown ? SVGLengthValue() : SVGLengthValue(value.floatValue(), lengthType);
244}

277ExceptionOr<void> SVGLengthValue::setValue(const SVGLengthContext& context, float value)
278{
279 // 100% = 100.0 instead of 1.0 for historical reasons, this could eventually be changed
280 if (m_lengthType == SVGLengthType::Percentage)
281 value = value / 100;
282
283 auto convertedValue = context.convertValueFromUserUnits(value, m_lengthType, m_lengthMode);
284 if (convertedValue.hasException())
285 return convertedValue.releaseException();
286 m_valueInSpecifiedUnits = convertedValue.releaseReturnValue();
287 return { };
288}

294 m_lengthType = lengthType;
295 return setValue(context, value);

298ExceptionOr<void> SVGLengthValue::setValueAsString(StringView string)
299{
300 if (string.isEmpty())
301 return { };
302
303 return readCharactersForParsing(string, [&](auto buffer) -> ExceptionOr<void> {
304 auto convertedNumber = parseNumber(buffer, SuffixSkippingPolicy::DontSkip);
305 if (!convertedNumber)
306 return Exception { SyntaxError };
307
308 auto lengthType = parseLengthType(buffer);
309 if (lengthType == SVGLengthType::Unknown)
310 return Exception { SyntaxError };
311
312 m_lengthType = lengthType;
313 m_valueInSpecifiedUnits = *convertedNumber;
314 return { };
315 });
316}

318ExceptionOr<void> SVGLengthValue::convertToSpecifiedUnits(const SVGLengthContext& context, SVGLengthType lengthType)
319{
320 auto valueInUserUnits = valueForBindings(context);
321 if (valueInUserUnits.hasException())
322 return valueInUserUnits.releaseException();
323
324 auto originalLengthType = m_lengthType;
325 m_lengthType = lengthType;
326 auto result = setValue(context, valueInUserUnits.releaseReturnValue());
327 if (!result.hasException())
328 return { };
329
330 m_lengthType = originalLengthType;
331 return result.releaseException();
332}
