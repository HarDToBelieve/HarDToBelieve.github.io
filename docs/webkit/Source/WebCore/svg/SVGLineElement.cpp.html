33inline SVGLineElement::SVGLineElement(const QualifiedName& tagName, Document& document)
34 : SVGGeometryElement(tagName, document)
35{
36 ASSERT(hasTagName(SVGNames::lineTag));
37
38 static std::once_flag onceFlag;
39 std::call_once(onceFlag, [] {
40 PropertyRegistry::registerProperty<SVGNames::x1Attr, &SVGLineElement::m_x1>();
41 PropertyRegistry::registerProperty<SVGNames::y1Attr, &SVGLineElement::m_y1>();
42 PropertyRegistry::registerProperty<SVGNames::x2Attr, &SVGLineElement::m_x2>();
43 PropertyRegistry::registerProperty<SVGNames::y2Attr, &SVGLineElement::m_y2>();
44 });
45}

47Ref<SVGLineElement> SVGLineElement::create(const QualifiedName& tagName, Document& document)
48{
49 return adoptRef(*new SVGLineElement(tagName, document));
50}
51
52void SVGLineElement::parseAttribute(const QualifiedName& name, const AtomString& value)
53{
54 SVGParsingError parseError = NoError;
55
56 if (name == SVGNames::x1Attr)
57 m_x1->setBaseValInternal(SVGLengthValue::construct(SVGLengthMode::Width, value, parseError));
58 else if (name == SVGNames::y1Attr)
59 m_y1->setBaseValInternal(SVGLengthValue::construct(SVGLengthMode::Height, value, parseError));
60 else if (name == SVGNames::x2Attr)
61 m_x2->setBaseValInternal(SVGLengthValue::construct(SVGLengthMode::Width, value, parseError));
62 else if (name == SVGNames::y2Attr)
63 m_y2->setBaseValInternal(SVGLengthValue::construct(SVGLengthMode::Height, value, parseError));
64
65 reportAttributeParsingError(parseError, name, value);
66
67 SVGGeometryElement::parseAttribute(name, value);
68}
69
70void SVGLineElement::svgAttributeChanged(const QualifiedName& attrName)
71{
72 if (PropertyRegistry::isKnownAttribute(attrName)) {
73 InstanceInvalidationGuard guard(*this);
74 updateRelativeLengthsInformation();
75
76 if (auto* renderer = downcast<RenderSVGShape>(this->renderer())) {
77 renderer->setNeedsShapeUpdate();
78 RenderSVGResource::markForLayoutAndParentResourceInvalidation(*renderer);
79 }
80 return;
81 }
82
83 SVGGeometryElement::svgAttributeChanged(attrName);
84}
85
86bool SVGLineElement::selfHasRelativeLengths() const
87{
88 return x1().isRelative()
89 || y1().isRelative()
90 || x2().isRelative()
91 || y2().isRelative();
92}
