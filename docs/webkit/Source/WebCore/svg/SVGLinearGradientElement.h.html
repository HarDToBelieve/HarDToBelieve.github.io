23
24#include "SVGGradientElement.h"
25#include "SVGNames.h"
26
27namespace WebCore {
28
29struct LinearGradientAttributes;
30
31class SVGLinearGradientElement final : public SVGGradientElement {
32 WTF_MAKE_ISO_ALLOCATED(SVGLinearGradientElement);
33public:
34 static Ref<SVGLinearGradientElement> create(const QualifiedName&, Document&);
35
36 bool collectGradientAttributes(LinearGradientAttributes&);
37
38 const SVGLengthValue& x1() const { return m_x1->currentValue(); }
39 const SVGLengthValue& y1() const { return m_y1->currentValue(); }
40 const SVGLengthValue& x2() const { return m_x2->currentValue(); }
41 const SVGLengthValue& y2() const { return m_y2->currentValue(); }
42
43 SVGAnimatedLength& x1Animated() { return m_x1; }
44 SVGAnimatedLength& y1Animated() { return m_y1; }
45 SVGAnimatedLength& x2Animated() { return m_x2; }
46 SVGAnimatedLength& y2Animated() { return m_y2; }
47
48private:
49 SVGLinearGradientElement(const QualifiedName&, Document&);
50
51 using PropertyRegistry = SVGPropertyOwnerRegistry<SVGLinearGradientElement, SVGGradientElement>;
52 const SVGPropertyRegistry& propertyRegistry() const final { return m_propertyRegistry; }
53
54 void parseAttribute(const QualifiedName&, const AtomString&) override;
55 void svgAttributeChanged(const QualifiedName&) override;
56
57 RenderPtr<RenderElement> createElementRenderer(RenderStyle&&, const RenderTreePosition&) override;
58
59 bool selfHasRelativeLengths() const override;
60
61 PropertyRegistry m_propertyRegistry { *this };
62 Ref<SVGAnimatedLength> m_x1 { SVGAnimatedLength::create(this, SVGLengthMode::Width) };
63 Ref<SVGAnimatedLength> m_y1 { SVGAnimatedLength::create(this, SVGLengthMode::Height) };
64 Ref<SVGAnimatedLength> m_x2 { SVGAnimatedLength::create(this, SVGLengthMode::Width, "100%") };
65 Ref<SVGAnimatedLength> m_y2 { SVGAnimatedLength::create(this, SVGLengthMode::Height) };
66};

68} // namespace WebCore
