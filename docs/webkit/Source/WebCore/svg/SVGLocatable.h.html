22#pragma once
23
24#include "AffineTransform.h"
25#include "ExceptionOr.h"
26
27namespace WebCore {
28
29class FloatRect;
30class SVGElement;
31class SVGMatrix;
32
33class SVGLocatable {
34public:
35 virtual ~SVGLocatable() = default;
36
37 // 'SVGLocatable' functions
38 virtual SVGElement* nearestViewportElement() const = 0;
39 virtual SVGElement* farthestViewportElement() const = 0;
40
41 enum StyleUpdateStrategy { AllowStyleUpdate, DisallowStyleUpdate };
42
43 virtual FloatRect getBBox(StyleUpdateStrategy) = 0;
44 virtual AffineTransform getCTM(StyleUpdateStrategy) = 0;
45 virtual AffineTransform getScreenCTM(StyleUpdateStrategy) = 0;
46
47 ExceptionOr<Ref<SVGMatrix>> getTransformToElement(SVGElement*, StyleUpdateStrategy = AllowStyleUpdate);
48
49 static SVGElement* nearestViewportElement(const SVGElement*);
50 static SVGElement* farthestViewportElement(const SVGElement*);
51
52 enum CTMScope {
53 NearestViewportScope, // Used for getCTM()
54 ScreenScope // Used for getScreenCTM()
55 };
56
57protected:
58 virtual AffineTransform localCoordinateSpaceTransform(SVGLocatable::CTMScope) const { return AffineTransform(); }
59
60 static FloatRect getBBox(SVGElement*, StyleUpdateStrategy);
61 static AffineTransform computeCTM(SVGElement*, CTMScope, StyleUpdateStrategy);
62};
63
64} // namespace WebCore
