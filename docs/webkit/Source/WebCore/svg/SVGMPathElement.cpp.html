35inline SVGMPathElement::SVGMPathElement(const QualifiedName& tagName, Document& document)
36 : SVGElement(tagName, document)
37 , SVGURIReference(this)
38{
39 ASSERT(hasTagName(SVGNames::mpathTag));

42Ref<SVGMPathElement> SVGMPathElement::create(const QualifiedName& tagName, Document& document)
43{
44 return adoptRef(*new SVGMPathElement(tagName, document));
45}

47SVGMPathElement::~SVGMPathElement()
48{
49 clearResourceReferences();
50}

52void SVGMPathElement::buildPendingResource()
53{
54 clearResourceReferences();
55 if (!isConnected())
56 return;
57
58 auto target = SVGURIReference::targetElementFromIRIString(href(), treeScope());
59 if (!target.element) {
60 // Do not register as pending if we are already pending this resource.
61 if (document().accessSVGExtensions().isPendingResource(*this, target.identifier))
62 return;
63
64 if (!target.identifier.isEmpty()) {
65 document().accessSVGExtensions().addPendingResource(target.identifier, *this);
66 ASSERT(hasPendingResources());
67 }
68 } else if (is<SVGElement>(*target.element))
69 downcast<SVGElement>(*target.element).addReferencingElement(*this);
70
71 targetPathChanged();
72}
73
74void SVGMPathElement::clearResourceReferences()
75{
76 removeElementReference();
77}
78
79Node::InsertedIntoAncestorResult SVGMPathElement::insertedIntoAncestor(InsertionType insertionType, ContainerNode& parentOfInsertedTree)
80{
81 SVGElement::insertedIntoAncestor(insertionType, parentOfInsertedTree);
82 if (insertionType.connectedToDocument)
83 return InsertedIntoAncestorResult::NeedsPostInsertionCallback;
84 return InsertedIntoAncestorResult::Done;
85}
86
87void SVGMPathElement::didFinishInsertingNode()
88{
89 buildPendingResource();
90}
91
92void SVGMPathElement::removedFromAncestor(RemovalType removalType, ContainerNode& oldParentOfRemovedTree)
93{
94 SVGElement::removedFromAncestor(removalType, oldParentOfRemovedTree);
95 notifyParentOfPathChange(&oldParentOfRemovedTree);
96 if (removalType.disconnectedFromDocument)
97 clearResourceReferences();
98}

100void SVGMPathElement::parseAttribute(const QualifiedName& name, const AtomString& value)
101{
102 SVGElement::parseAttribute(name, value);
103 SVGURIReference::parseAttribute(name, value);
104}
105
106void SVGMPathElement::svgAttributeChanged(const QualifiedName& attrName)
107{
108 if (SVGURIReference::isKnownAttribute(attrName)) {
109 InstanceInvalidationGuard guard(*this);
110 buildPendingResource();
111 return;
112 }
113
114 SVGElement::svgAttributeChanged(attrName);
115}
116
117RefPtr<SVGPathElement> SVGMPathElement::pathElement()
118{
119 auto target = targetElementFromIRIString(href(), treeScope());
120 if (is<SVGPathElement>(target.element))
121 return downcast<SVGPathElement>(target.element.get());
122 return nullptr;
123}
124
125void SVGMPathElement::targetPathChanged()
126{
127 notifyParentOfPathChange(parentNode());
128}
129
130void SVGMPathElement::notifyParentOfPathChange(ContainerNode* parent)
131{
132 if (is<SVGAnimateMotionElement>(parent))
133 downcast<SVGAnimateMotionElement>(*parent).updateAnimationPath();
134}
