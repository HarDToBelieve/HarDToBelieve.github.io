22#pragma once
23
24#include "SVGElement.h"
25#include "SVGFitToViewBox.h"
26#include "SVGMarkerTypes.h"
27
28namespace WebCore {
29
30class SVGMarkerElement final : public SVGElement, public SVGFitToViewBox {
31 WTF_MAKE_ISO_ALLOCATED(SVGMarkerElement);
32public:
33 // Forward declare enumerations in the W3C naming scheme, for IDL generation.
34 enum {
35 SVG_MARKERUNITS_UNKNOWN = SVGMarkerUnitsUnknown,
36 SVG_MARKERUNITS_USERSPACEONUSE = SVGMarkerUnitsUserSpaceOnUse,
37 SVG_MARKERUNITS_STROKEWIDTH = SVGMarkerUnitsStrokeWidth
38 };
39
40 enum {
41 SVG_MARKER_ORIENT_UNKNOWN = SVGMarkerOrientUnknown,
42 SVG_MARKER_ORIENT_AUTO = SVGMarkerOrientAuto,
43 SVG_MARKER_ORIENT_ANGLE = SVGMarkerOrientAngle
44 };
45
46 static Ref<SVGMarkerElement> create(const QualifiedName&, Document&);
47
48 AffineTransform viewBoxToViewTransform(float viewWidth, float viewHeight) const;
49
50 const SVGLengthValue& refX() const { return m_refX->currentValue(); }
51 const SVGLengthValue& refY() const { return m_refY->currentValue(); }
52 const SVGLengthValue& markerWidth() const { return m_markerWidth->currentValue(); }
53 const SVGLengthValue& markerHeight() const { return m_markerHeight->currentValue(); }
54 SVGMarkerUnitsType markerUnits() const { return m_markerUnits->currentValue<SVGMarkerUnitsType>(); }
55 const SVGAngleValue& orientAngle() const { return m_orientAngle->currentValue(); }
56 SVGMarkerOrientType orientType() const { return m_orientType->currentValue<SVGMarkerOrientType>(); }
57
58 SVGAnimatedLength& refXAnimated() { return m_refX; }
59 SVGAnimatedLength& refYAnimated() { return m_refY; }
60 SVGAnimatedLength& markerWidthAnimated() { return m_markerWidth; }
61 SVGAnimatedLength& markerHeightAnimated() { return m_markerHeight; }
62 SVGAnimatedEnumeration& markerUnitsAnimated() { return m_markerUnits; }
63 SVGAnimatedAngle& orientAngleAnimated() { return m_orientAngle; }
64 Ref<SVGAnimatedEnumeration> orientTypeAnimated() { return m_orientType.copyRef(); }
65
66 String orient() const;
67 void setOrient(const String&);
68
69 void setOrientToAuto();
70 void setOrientToAngle(const SVGAngle&);
71
72private:
73 SVGMarkerElement(const QualifiedName&, Document&);
74
75 using PropertyRegistry = SVGPropertyOwnerRegistry<SVGMarkerElement, SVGElement, SVGFitToViewBox>;
76 const SVGPropertyRegistry& propertyRegistry() const final { return m_propertyRegistry; }
77
78 void parseAttribute(const QualifiedName&, const AtomString&) override;
79 void svgAttributeChanged(const QualifiedName&) override;
80 void childrenChanged(const ChildChange&) override;
81
82 RenderPtr<RenderElement> createElementRenderer(RenderStyle&&, const RenderTreePosition&) override;
83 bool rendererIsNeeded(const RenderStyle&) override { return true; }
84
85 bool needsPendingResourceHandling() const override { return false; }
86
87 bool selfHasRelativeLengths() const override;
88
89 PropertyRegistry m_propertyRegistry { *this };
90 Ref<SVGAnimatedLength> m_refX { SVGAnimatedLength::create(this, SVGLengthMode::Width) };
91 Ref<SVGAnimatedLength> m_refY { SVGAnimatedLength::create(this, SVGLengthMode::Height) };
92 Ref<SVGAnimatedLength> m_markerWidth { SVGAnimatedLength::create(this, SVGLengthMode::Width, "3") };
93 Ref<SVGAnimatedLength> m_markerHeight { SVGAnimatedLength::create(this, SVGLengthMode::Height, "3") };
94 Ref<SVGAnimatedEnumeration> m_markerUnits { SVGAnimatedEnumeration::create(this, SVGMarkerUnitsStrokeWidth) };
95 Ref<SVGAnimatedAngle> m_orientAngle { SVGAnimatedAngle::create(this) };
96 Ref<SVGAnimatedOrientType> m_orientType { SVGAnimatedOrientType::create(this, SVGMarkerOrientAngle) };
97};

99} // namespace WebCore
