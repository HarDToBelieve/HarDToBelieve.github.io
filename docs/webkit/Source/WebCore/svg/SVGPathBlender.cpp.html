58static inline FloatPoint blendFloatPoint(const FloatPoint& a, const FloatPoint& b, float progress)
59{
60 BlendingContext context { progress };
61 return FloatPoint(blend(a.x(), b.x(), context), blend(a.y(), b.y(), context));

64float SVGPathBlender::blendAnimatedDimensonalFloat(float from, float to, FloatBlendMode blendMode, float progress)
65{
66 if (m_addTypesCount) {
67 ASSERT(m_fromMode == m_toMode);
68 return from + to * m_addTypesCount;
69 }
70
71 if (m_fromMode == m_toMode)
72 return blend(from, to, { progress });
73
74 float fromValue = blendMode == BlendHorizontal ? m_fromCurrentPoint.x() : m_fromCurrentPoint.y();
75 float toValue = blendMode == BlendHorizontal ? m_toCurrentPoint.x() : m_toCurrentPoint.y();
76
77 // Transform toY to the coordinate mode of fromY
78 float animValue = blend(from, m_fromMode == AbsoluteCoordinates ? to + toValue : to - toValue, { progress });
79
80 if (m_isInFirstHalfOfAnimation)
81 return animValue;
82
83 // Transform the animated point to the coordinate mode, needed for the current progress.
84 float currentValue = blend(fromValue, toValue, { progress });
85 return m_toMode == AbsoluteCoordinates ? animValue + currentValue : animValue - currentValue;
86}

88FloatPoint SVGPathBlender::blendAnimatedFloatPoint(const FloatPoint& fromPoint, const FloatPoint& toPoint, float progress)
89{
90 if (m_addTypesCount) {
91 ASSERT(m_fromMode == m_toMode);
92 FloatPoint repeatedToPoint = toPoint;
93 repeatedToPoint.scale(m_addTypesCount);
94 return fromPoint + repeatedToPoint;
95 }
96
97 if (m_fromMode == m_toMode)
98 return blendFloatPoint(fromPoint, toPoint, progress);
99
100 // Transform toPoint to the coordinate mode of fromPoint
101 FloatPoint animatedPoint = toPoint;
102 if (m_fromMode == AbsoluteCoordinates)
103 animatedPoint += m_toCurrentPoint;
104 else
105 animatedPoint.move(-m_toCurrentPoint.x(), -m_toCurrentPoint.y());
106
107 animatedPoint = blendFloatPoint(fromPoint, animatedPoint, progress);
108
109 if (m_isInFirstHalfOfAnimation)
110 return animatedPoint;
111
112 // Transform the animated point to the coordinate mode, needed for the current progress.
113 FloatPoint currentPoint = blendFloatPoint(m_fromCurrentPoint, m_toCurrentPoint, progress);
114 if (m_toMode == AbsoluteCoordinates)
115 return animatedPoint + currentPoint;
116
117 animatedPoint.move(-currentPoint.x(), -currentPoint.y());
118 return animatedPoint;
119}

121template<typename Function> using InvokeResult = typename std::invoke_result_t<Function, SVGPathSource>::value_type;
122template<typename Function> using ResultPair = std::pair<InvokeResult<Function>, InvokeResult<Function>>;
123template<typename Function> static std::optional<ResultPair<Function>> pullFromSources(SVGPathSource& fromSource, SVGPathSource& toSource, Function&& function)
124{
125 InvokeResult<Function> fromResult;
126 if (fromSource.hasMoreData()) {
127 auto parsedFrom = std::invoke(function, fromSource);
128 if (!parsedFrom)
129 return std::nullopt;
130 fromResult = WTFMove(*parsedFrom);
131 }
132
133 auto parsedTo = std::invoke(std::forward<Function>(function), toSource);
134 if (!parsedTo)
135 return std::nullopt;
136
137 return ResultPair<Function> { WTFMove(fromResult), WTFMove(*parsedTo) };
138}

140bool SVGPathBlender::blendMoveToSegment(float progress)
141{
142 auto result = pullFromSources(m_fromSource, m_toSource, &SVGPathSource::parseMoveToSegment);
143 if (!result)
144 return false;
145
146 if (!m_consumer)
147 return true;
148
149 auto [from, to] = *result;
150
151 m_consumer->moveTo(blendAnimatedFloatPoint(from.targetPoint, to.targetPoint, progress), false, m_isInFirstHalfOfAnimation ? m_fromMode : m_toMode);
152 m_fromCurrentPoint = m_fromMode == AbsoluteCoordinates ? from.targetPoint : m_fromCurrentPoint + from.targetPoint;
153 m_toCurrentPoint = m_toMode == AbsoluteCoordinates ? to.targetPoint : m_toCurrentPoint + to.targetPoint;
154 return true;
155}

283bool SVGPathBlender::blendArcToSegment(float progress)
284{
285 auto result = pullFromSources(m_fromSource, m_toSource, &SVGPathSource::parseArcToSegment);
286 if (!result)
287 return false;
288
289 if (!m_consumer)
290 return true;
291
292 auto [from, to] = *result;
293
294 if (m_addTypesCount) {
295 ASSERT(m_fromMode == m_toMode);
296 auto scaledToTargetPoint = to.targetPoint;
297 scaledToTargetPoint.scale(m_addTypesCount);
298 m_consumer->arcTo(from.rx + to.rx * m_addTypesCount,
299 from.ry + to.ry * m_addTypesCount,
300 from.angle + to.angle * m_addTypesCount,
301 from.largeArc || to.largeArc,
302 from.sweep || to.sweep,
303 from.targetPoint + scaledToTargetPoint,
304 m_fromMode);
305 } else {
306 BlendingContext context { progress };
307 m_consumer->arcTo(blend(from.rx, to.rx, context),
308 blend(from.ry, to.ry, context),
309 blend(from.angle, to.angle, context),
310 m_isInFirstHalfOfAnimation ? from.largeArc : to.largeArc,
311 m_isInFirstHalfOfAnimation ? from.sweep : to.sweep,
312 blendAnimatedFloatPoint(from.targetPoint, to.targetPoint, progress),
313 m_isInFirstHalfOfAnimation ? m_fromMode : m_toMode);
314 }
315 m_fromCurrentPoint = m_fromMode == AbsoluteCoordinates ? from.targetPoint : m_fromCurrentPoint + from.targetPoint;
316 m_toCurrentPoint = m_toMode == AbsoluteCoordinates ? to.targetPoint : m_toCurrentPoint + to.targetPoint;
317 return true;
318}

320static inline PathCoordinateMode coordinateModeOfCommand(const SVGPathSegType& type)
321{
322 if (type < PathSegMoveToAbs)
323 return AbsoluteCoordinates;
324
325 // Odd number = relative command
326 if (type % 2)
327 return RelativeCoordinates;
328
329 return AbsoluteCoordinates;
330}

332static inline bool isSegmentEqual(const SVGPathSegType& fromType, const SVGPathSegType& toType, const PathCoordinateMode& fromMode, const PathCoordinateMode& toMode)
333{
334 if (fromType == toType && (fromType == PathSegUnknown || fromType == PathSegClosePath))
335 return true;
336
337 unsigned short from = fromType;
338 unsigned short to = toType;
339 if (fromMode == toMode)
340 return from == to;
341 if (fromMode == AbsoluteCoordinates)
342 return from == to - 1;
343 return to == from - 1;
344}

352bool SVGPathBlender::canBlendPaths()
353{
354 float progress = 0.5;
355 bool fromSourceHadData = m_fromSource.hasMoreData();
356 while (m_toSource.hasMoreData()) {
357 SVGPathSegType fromCommand;
358 if (fromSourceHadData) {
359 auto parsedFromCommand = m_fromSource.parseSVGSegmentType();
360 if (!parsedFromCommand)
361 return false;
362 fromCommand = *parsedFromCommand;
363 }
364 auto parsedtoCommand = m_toSource.parseSVGSegmentType();
365 if (!parsedtoCommand)
366 return false;
367 SVGPathSegType toCommand = *parsedtoCommand;
368
369 m_toMode = coordinateModeOfCommand(toCommand);
370 m_fromMode = fromSourceHadData ? coordinateModeOfCommand(fromCommand) : m_toMode;
371 if (m_fromMode != m_toMode && m_addTypesCount)
372 return false;
373
374 if (fromSourceHadData && !isSegmentEqual(fromCommand, toCommand, m_fromMode, m_toMode))
375 return false;
376
377 switch (toCommand) {
378 case PathSegMoveToRel:
379 case PathSegMoveToAbs:
380 if (!blendMoveToSegment(progress))
381 return false;
382 break;
383 case PathSegLineToRel:
384 case PathSegLineToAbs:
385 if (!blendLineToSegment(progress))
386 return false;
387 break;
388 case PathSegLineToHorizontalRel:
389 case PathSegLineToHorizontalAbs:
390 if (!blendLineToHorizontalSegment(progress))
391 return false;
392 break;
393 case PathSegLineToVerticalRel:
394 case PathSegLineToVerticalAbs:
395 if (!blendLineToVerticalSegment(progress))
396 return false;
397 break;
398 case PathSegClosePath:
399 break;
400 case PathSegCurveToCubicRel:
401 case PathSegCurveToCubicAbs:
402 if (!blendCurveToCubicSegment(progress))
403 return false;
404 break;
405 case PathSegCurveToCubicSmoothRel:
406 case PathSegCurveToCubicSmoothAbs:
407 if (!blendCurveToCubicSmoothSegment(progress))
408 return false;
409 break;
410 case PathSegCurveToQuadraticRel:
411 case PathSegCurveToQuadraticAbs:
412 if (!blendCurveToQuadraticSegment(progress))
413 return false;
414 break;
415 case PathSegCurveToQuadraticSmoothRel:
416 case PathSegCurveToQuadraticSmoothAbs:
417 if (!blendCurveToQuadraticSmoothSegment(progress))
418 return false;
419 break;
420 case PathSegArcRel:
421 case PathSegArcAbs:
422 if (!blendArcToSegment(progress))
423 return false;
424 break;
425 case PathSegUnknown:
426 return false;
427 }
428
429 if (!fromSourceHadData)
430 continue;
431 if (m_fromSource.hasMoreData() != m_toSource.hasMoreData())
432 return false;
433 if (!m_fromSource.hasMoreData() || !m_toSource.hasMoreData())
434 return true;
435 }
436
437 return true;
438}

440bool SVGPathBlender::blendAnimatedPath(float progress)
441{
442 m_isInFirstHalfOfAnimation = progress < 0.5f;
443
444 bool fromSourceHadData = m_fromSource.hasMoreData();
445 while (m_toSource.hasMoreData()) {
446 SVGPathSegType fromCommand;
447 if (fromSourceHadData) {
448 auto parsedFromCommand = m_fromSource.parseSVGSegmentType();
449 if (!parsedFromCommand)
450 return false;
451 fromCommand = *parsedFromCommand;
452 }
453 auto parsedToCommand = m_toSource.parseSVGSegmentType();
454 if (!parsedToCommand)
455 return false;
456 SVGPathSegType toCommand = *parsedToCommand;
457
458 m_toMode = coordinateModeOfCommand(toCommand);
459 m_fromMode = fromSourceHadData ? coordinateModeOfCommand(fromCommand) : m_toMode;
460 if (m_fromMode != m_toMode && m_addTypesCount)
461 return false;
462
463 if (fromSourceHadData && !isSegmentEqual(fromCommand, toCommand, m_fromMode, m_toMode))
464 return false;
465
466 switch (toCommand) {
467 case PathSegMoveToRel:
468 case PathSegMoveToAbs:
469 if (!blendMoveToSegment(progress))
470 return false;
471 break;
472 case PathSegLineToRel:
473 case PathSegLineToAbs:
474 if (!blendLineToSegment(progress))
475 return false;
476 break;
477 case PathSegLineToHorizontalRel:
478 case PathSegLineToHorizontalAbs:
479 if (!blendLineToHorizontalSegment(progress))
480 return false;
481 break;
482 case PathSegLineToVerticalRel:
483 case PathSegLineToVerticalAbs:
484 if (!blendLineToVerticalSegment(progress))
485 return false;
486 break;
487 case PathSegClosePath:
488 m_consumer->closePath();
489 break;
490 case PathSegCurveToCubicRel:
491 case PathSegCurveToCubicAbs:
492 if (!blendCurveToCubicSegment(progress))
493 return false;
494 break;
495 case PathSegCurveToCubicSmoothRel:
496 case PathSegCurveToCubicSmoothAbs:
497 if (!blendCurveToCubicSmoothSegment(progress))
498 return false;
499 break;
500 case PathSegCurveToQuadraticRel:
501 case PathSegCurveToQuadraticAbs:
502 if (!blendCurveToQuadraticSegment(progress))
503 return false;
504 break;
505 case PathSegCurveToQuadraticSmoothRel:
506 case PathSegCurveToQuadraticSmoothAbs:
507 if (!blendCurveToQuadraticSmoothSegment(progress))
508 return false;
509 break;
510 case PathSegArcRel:
511 case PathSegArcAbs:
512 if (!blendArcToSegment(progress))
513 return false;
514 break;
515 case PathSegUnknown:
516 return false;
517 }
518
519 if (!fromSourceHadData)
520 continue;
521 if (m_fromSource.hasMoreData() != m_toSource.hasMoreData())
522 return false;
523 if (!m_fromSource.hasMoreData() || !m_toSource.hasMoreData())
524 return true;
525 }
526
527 return true;
528}
