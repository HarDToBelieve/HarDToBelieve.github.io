29namespace WebCore {
30
31class Path;
32
33class SVGPathBuilder final : public SVGPathConsumer {
34public:
35 SVGPathBuilder(Path&);
36
37private:
38 void incrementPathSegmentCount() final { }
39 bool continueConsuming() final { return true; }
40
41 // Used in UnalteredParsing/NormalizedParsing modes.
42 void moveTo(const FloatPoint&, bool closed, PathCoordinateMode) final;
43 void lineTo(const FloatPoint&, PathCoordinateMode) final;
44 void curveToCubic(const FloatPoint&, const FloatPoint&, const FloatPoint&, PathCoordinateMode) final;
45 void closePath() final;
46
47 // Only used in UnalteredParsing mode.
48 void lineToHorizontal(float, PathCoordinateMode) final { ASSERT_NOT_REACHED(); }
49 void lineToVertical(float, PathCoordinateMode) final { ASSERT_NOT_REACHED(); }
50 void curveToCubicSmooth(const FloatPoint&, const FloatPoint&, PathCoordinateMode) final { ASSERT_NOT_REACHED(); }
51 void curveToQuadratic(const FloatPoint&, const FloatPoint&, PathCoordinateMode) final { ASSERT_NOT_REACHED(); }
52 void curveToQuadraticSmooth(const FloatPoint&, PathCoordinateMode) final { ASSERT_NOT_REACHED(); }
53 void arcTo(float, float, float, bool, bool, const FloatPoint&, PathCoordinateMode) final { ASSERT_NOT_REACHED(); }
54
55 Path& m_path;
56 FloatPoint m_current;
57};
