35void SVGPathByteStreamBuilder::moveTo(const FloatPoint& targetPoint, bool, PathCoordinateMode mode)
36{
37 writeSegmentType(mode == RelativeCoordinates ? PathSegMoveToRel : PathSegMoveToAbs);
38 writeFloatPoint(targetPoint);
39}

41void SVGPathByteStreamBuilder::lineTo(const FloatPoint& targetPoint, PathCoordinateMode mode)
42{
43 writeSegmentType(mode == RelativeCoordinates ? PathSegLineToRel : PathSegLineToAbs);
44 writeFloatPoint(targetPoint);
45}

59void SVGPathByteStreamBuilder::curveToCubic(const FloatPoint& point1, const FloatPoint& point2, const FloatPoint& targetPoint, PathCoordinateMode mode)

61 writeSegmentType(mode == RelativeCoordinates ? PathSegCurveToCubicRel : PathSegCurveToCubicAbs);

64 writeFloatPoint(targetPoint);
65}

87void SVGPathByteStreamBuilder::arcTo(float r1, float r2, float angle, bool largeArcFlag, bool sweepFlag, const FloatPoint& targetPoint, PathCoordinateMode mode)
88{
89 writeSegmentType(mode == RelativeCoordinates ? PathSegArcRel : PathSegArcAbs);
90 writeFloat(r1);
91 writeFloat(r2);
92 writeFloat(angle);
93 writeFlag(largeArcFlag);
94 writeFlag(sweepFlag);
95 writeFloatPoint(targetPoint);
96}
