24#pragma once
25
26#include "FloatPoint.h"
27#include <wtf/FastMalloc.h>
28#include <wtf/Noncopyable.h>
29
30namespace WebCore {
31
32enum PathCoordinateMode {
33 AbsoluteCoordinates,
34 RelativeCoordinates
35};
36
37enum PathParsingMode {
38 NormalizedParsing,
39 UnalteredParsing
40};
41
42class SVGPathConsumer {
43 WTF_MAKE_NONCOPYABLE(SVGPathConsumer); WTF_MAKE_FAST_ALLOCATED;
44public:
45 SVGPathConsumer() = default;
46 virtual void incrementPathSegmentCount() = 0;
47 virtual bool continueConsuming() = 0;
48
49 // Used in UnalteredParsing/NormalizedParsing modes.
50 virtual void moveTo(const FloatPoint&, bool closed, PathCoordinateMode) = 0;
51 virtual void lineTo(const FloatPoint&, PathCoordinateMode) = 0;
52 virtual void curveToCubic(const FloatPoint&, const FloatPoint&, const FloatPoint&, PathCoordinateMode) = 0;
53 virtual void closePath() = 0;
54
55 // Only used in UnalteredParsing mode.
56 virtual void lineToHorizontal(float, PathCoordinateMode) = 0;
57 virtual void lineToVertical(float, PathCoordinateMode) = 0;
58 virtual void curveToCubicSmooth(const FloatPoint&, const FloatPoint&, PathCoordinateMode) = 0;
59 virtual void curveToQuadratic(const FloatPoint&, const FloatPoint&, PathCoordinateMode) = 0;
60 virtual void curveToQuadraticSmooth(const FloatPoint&, PathCoordinateMode) = 0;
61 virtual void arcTo(float, float, float, bool largeArcFlag, bool sweepFlag, const FloatPoint&, PathCoordinateMode) = 0;
62
63protected:
64 virtual ~SVGPathConsumer() = default;
65};
66
67} // namespace WebCore
