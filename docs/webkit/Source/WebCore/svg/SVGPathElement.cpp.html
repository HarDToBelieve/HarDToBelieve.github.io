38inline SVGPathElement::SVGPathElement(const QualifiedName& tagName, Document& document)
39 : SVGGeometryElement(tagName, document)
40{
41 ASSERT(hasTagName(SVGNames::pathTag));
42
43 static std::once_flag onceFlag;
44 std::call_once(onceFlag, [] {
45 PropertyRegistry::registerProperty<SVGNames::dAttr, &SVGPathElement::m_pathSegList>();
46 });
47}

49Ref<SVGPathElement> SVGPathElement::create(const QualifiedName& tagName, Document& document)
50{
51 return adoptRef(*new SVGPathElement(tagName, document));
52}
53
54void SVGPathElement::parseAttribute(const QualifiedName& name, const AtomString& value)
55{
56 if (name == SVGNames::dAttr) {
57 if (!m_pathSegList->baseVal()->parse(value))
58 document().accessSVGExtensions().reportError("Problem parsing d=\"" + value + "\"");
59 return;
60 }
61
62 SVGGeometryElement::parseAttribute(name, value);
63}
64
65void SVGPathElement::svgAttributeChanged(const QualifiedName& attrName)
66{
67 if (attrName == SVGNames::dAttr) {
68 InstanceInvalidationGuard guard(*this);
69 invalidateMPathDependencies();
70
71 if (auto* renderer = downcast<RenderSVGPath>(this->renderer())) {
72 renderer->setNeedsShapeUpdate();
73 RenderSVGResource::markForLayoutAndParentResourceInvalidation(*renderer);
74 }
75
76 return;
77 }
78
79 SVGGeometryElement::svgAttributeChanged(attrName);
80}
81
82void SVGPathElement::invalidateMPathDependencies()
83{
84 // <mpath> can only reference <path> but this dependency is not handled in
85 // markForLayoutAndParentResourceInvalidation so we update any mpath dependencies manually.
86 for (auto& element : referencingElements()) {
87 if (is<SVGMPathElement>(element))
88 downcast<SVGMPathElement>(element.get()).targetPathChanged();
89 }
90}

110ExceptionOr<Ref<SVGPoint>> SVGPathElement::getPointAtLength(float distance) const
111{
112 // Spec: Clamp distance to [0, length].
113 distance = clampTo<float>(distance, 0, getTotalLength());
114
115 // Spec: Return a newly created, detached SVGPoint object.
116 return SVGPoint::create(getPointAtLengthOfSVGPathByteStream(pathByteStream(), distance));
117}

119unsigned SVGPathElement::getPathSegAtLength(float length) const
120{
121 return getSVGPathSegAtLengthFromSVGPathByteStream(pathByteStream(), length);
122}

124FloatRect SVGPathElement::getBBox(StyleUpdateStrategy styleUpdateStrategy)
125{
126 if (styleUpdateStrategy == AllowStyleUpdate)
127 document().updateLayoutIgnorePendingStylesheets();
128
129 RenderSVGPath* renderer = downcast<RenderSVGPath>(this->renderer());
130
131 // FIXME: Eventually we should support getBBox for detached elements.
132 // FIXME: If the path is null it means we're calling getBBox() before laying out this element,
133 // which is an error.
134 if (!renderer || !renderer->hasPath())
135 return { };
136
137 return renderer->path().boundingRect();
138}
