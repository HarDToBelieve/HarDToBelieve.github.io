42Path buildPathFromString(const String& d)
43{
44 if (d.isEmpty())
45 return { };
46
47 Path path;
48 SVGPathBuilder builder(path);
49 SVGPathStringSource source(d);
50 SVGPathParser::parse(source, builder);
51 return path;
52}

54String buildStringFromPath(const Path& path)
55{
56 StringBuilder builder;
57
58 if (!path.isNull() && !path.isEmpty()) {
59 path.apply([&builder] (const PathElement& element) {
60 switch (element.type) {
61 case PathElement::Type::MoveToPoint:
62 builder.append('M', element.points[0].x(), ' ', element.points[0].y());
63 break;
64 case PathElement::Type::AddLineToPoint:
65 builder.append('L', element.points[0].x(), ' ', element.points[0].y());
66 break;
67 case PathElement::Type::AddQuadCurveToPoint:
68 builder.append('Q', element.points[0].x(), ' ', element.points[0].y(), ',', element.points[1].x(), ' ', element.points[1].y());
69 break;
70 case PathElement::Type::AddCurveToPoint:
71 builder.append('C', element.points[0].x(), ' ', element.points[0].y(), ',', element.points[1].x(), ' ', element.points[1].y(), ',', element.points[2].x(), ' ', element.points[2].y());
72 break;
73 case PathElement::Type::CloseSubpath:
74 builder.append('Z');
75 break;
76 }
77 });
78 }
79
80 return builder.toString();
81}

141bool buildAnimatedSVGPathByteStream(const SVGPathByteStream& fromStream, const SVGPathByteStream& toStream, SVGPathByteStream& result, float progress)
142{
143 ASSERT(&toStream != &result);
144 result.clear();
145 if (toStream.isEmpty())
146 return true;
147
148 SVGPathByteStreamBuilder builder(result);
149
150 SVGPathByteStreamSource fromSource(fromStream);
151 SVGPathByteStreamSource toSource(toStream);
152 return SVGPathBlender::blendAnimatedPath(fromSource, toSource, builder, progress);
153}

155bool addToSVGPathByteStream(SVGPathByteStream& streamToAppendTo, const SVGPathByteStream& byStream, unsigned repeatCount)
156{
157 // The byStream will be blended with streamToAppendTo. So streamToAppendTo has to have elements.
158 if (streamToAppendTo.isEmpty() || byStream.isEmpty())
159 return true;
160
161 // builder is the destination of blending fromSource and bySource. The stream of builder
162 // (i.e. streamToAppendTo) has to be cleared before calling addAnimatedPath.
163 SVGPathByteStreamBuilder builder(streamToAppendTo);
164
165 SVGPathByteStream fromStreamCopy = WTFMove(streamToAppendTo);
166
167 SVGPathByteStreamSource fromSource(fromStreamCopy);
168 SVGPathByteStreamSource bySource(byStream);
169 return SVGPathBlender::addAnimatedPath(fromSource, bySource, builder, repeatCount);
170}

172unsigned getSVGPathSegAtLengthFromSVGPathByteStream(const SVGPathByteStream& stream, float length)
173{
174 if (stream.isEmpty())
175 return 0;
176
177 PathTraversalState traversalState(PathTraversalState::Action::SegmentAtLength);
178 SVGPathTraversalStateBuilder builder(traversalState, length);
179
180 SVGPathByteStreamSource source(stream);
181 SVGPathParser::parse(source, builder);
182 return builder.pathSegmentIndex();
183}
