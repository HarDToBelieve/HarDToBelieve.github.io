22#pragma once
23
24#include "SVGGeometryElement.h"
25#include "SVGNames.h"
26
27namespace WebCore {
28
29class SVGPolyElement : public SVGGeometryElement {
30 WTF_MAKE_ISO_ALLOCATED(SVGPolyElement);
31public:
32 const SVGPointList& points() const { return m_points->currentValue(); }
33
34 SVGPointList& points() { return m_points->baseVal(); }
35 SVGPointList& animatedPoints() { return *m_points->animVal(); }
36
37 size_t approximateMemoryCost() const override;
38
39protected:
40 SVGPolyElement(const QualifiedName&, Document&);
41
42private:
43 using PropertyRegistry = SVGPropertyOwnerRegistry<SVGPolyElement, SVGGeometryElement>;
44 const SVGPropertyRegistry& propertyRegistry() const final { return m_propertyRegistry; }
45
46 void parseAttribute(const QualifiedName&, const AtomString&) override;
47 void svgAttributeChanged(const QualifiedName&) override;
48
49 bool isValid() const override { return SVGTests::isValid(); }
50 bool supportsMarkers() const override { return true; }
51
52 PropertyRegistry m_propertyRegistry { *this };
53 Ref<SVGAnimatedPointList> m_points { SVGAnimatedPointList::create(this) };
54};
55
56} // namespace WebCore
57
58SPECIALIZE_TYPE_TRAITS_BEGIN(WebCore::SVGPolyElement)
59 static bool isType(const WebCore::SVGElement& element) { return element.hasTagName(WebCore::SVGNames::polygonTag) || element.hasTagName(WebCore::SVGNames::polylineTag); }
60 static bool isType(const WebCore::Node& node) { return is<WebCore::SVGElement>(node) && isType(downcast<WebCore::SVGElement>(node)); }
61SPECIALIZE_TYPE_TRAITS_END()
