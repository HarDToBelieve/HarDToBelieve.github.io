45ExceptionOr<void> SVGPreserveAspectRatioValue::setAlign(unsigned short align)
46{
47 if (align == SVG_PRESERVEASPECTRATIO_UNKNOWN || align > SVG_PRESERVEASPECTRATIO_XMAXYMAX)
48 return Exception { NotSupportedError };
49
50 m_align = static_cast<SVGPreserveAspectRatioType>(align);
51 return { };
52}

54ExceptionOr<void> SVGPreserveAspectRatioValue::setMeetOrSlice(unsigned short meetOrSlice)
55{
56 if (meetOrSlice == SVG_MEETORSLICE_UNKNOWN || meetOrSlice > SVG_MEETORSLICE_SLICE)
57 return Exception { NotSupportedError };
58
59 m_meetOrSlice = static_cast<SVGMeetOrSliceType>(meetOrSlice);
60 return { };
61}

80template<typename CharacterType> static constexpr CharacterType deferDesc[] = {'d', 'e', 'f', 'e', 'r'};
81template<typename CharacterType> static constexpr CharacterType noneDesc[] = {'n', 'o', 'n', 'e'};
82template<typename CharacterType> static constexpr CharacterType meetDesc[] = {'m', 'e', 'e', 't'};
83template<typename CharacterType> static constexpr CharacterType sliceDesc[] = {'s', 'l', 'i', 'c', 'e'};
84
85template<typename CharacterType> bool SVGPreserveAspectRatioValue::parseInternal(StringParsingBuffer<CharacterType>& buffer, bool validate)
86{
87 SVGPreserveAspectRatioType align = SVG_PRESERVEASPECTRATIO_XMIDYMID;
88 SVGMeetOrSliceType meetOrSlice = SVG_MEETORSLICE_MEET;
89
90 m_align = align;
91 m_meetOrSlice = meetOrSlice;
92
93 if (!skipOptionalSVGSpaces(buffer))
94 return false;
95
96 if (*buffer == 'd') {
97 if (!skipCharactersExactly(buffer, deferDesc<CharacterType>)) {
98 LOG_ERROR("Skipped to parse except for *defer* value.");
99 return false;
100 }
101
102 // FIXME: We just ignore the "defer" here.
103 if (buffer.atEnd())
104 return true;
105
106 if (!skipOptionalSVGSpaces(buffer))
107 return false;
108 }
109
110 if (*buffer == 'n') {
111 if (!skipCharactersExactly(buffer, noneDesc<CharacterType>)) {
112 LOG_ERROR("Skipped to parse except for *none* value.");
113 return false;
114 }
115 align = SVG_PRESERVEASPECTRATIO_NONE;
116 skipOptionalSVGSpaces(buffer);
117 } else if (*buffer == 'x') {
118 if (buffer.lengthRemaining() < 8)
119 return false;
120 if (buffer[1] != 'M' || buffer[4] != 'Y' || buffer[5] != 'M')
121 return false;
122 if (buffer[2] == 'i') {
123 if (buffer[3] == 'n') {
124 if (buffer[6] == 'i') {
125 if (buffer[7] == 'n')
126 align = SVG_PRESERVEASPECTRATIO_XMINYMIN;
127 else if (buffer[7] == 'd')
128 align = SVG_PRESERVEASPECTRATIO_XMINYMID;
129 else
130 return false;
131 } else if (buffer[6] == 'a' && buffer[7] == 'x')
132 align = SVG_PRESERVEASPECTRATIO_XMINYMAX;
133 else
134 return false;
135 } else if (buffer[3] == 'd') {
136 if (buffer[6] == 'i') {
137 if (buffer[7] == 'n')
138 align = SVG_PRESERVEASPECTRATIO_XMIDYMIN;
139 else if (buffer[7] == 'd')
140 align = SVG_PRESERVEASPECTRATIO_XMIDYMID;
141 else
142 return false;
143 } else if (buffer[6] == 'a' && buffer[7] == 'x')
144 align = SVG_PRESERVEASPECTRATIO_XMIDYMAX;
145 else
146 return false;
147 } else
148 return false;
149 } else if (buffer[2] == 'a' && buffer[3] == 'x') {
150 if (buffer[6] == 'i') {
151 if (buffer[7] == 'n')
152 align = SVG_PRESERVEASPECTRATIO_XMAXYMIN;
153 else if (buffer[7] == 'd')
154 align = SVG_PRESERVEASPECTRATIO_XMAXYMID;
155 else
156 return false;
157 } else if (buffer[6] == 'a' && buffer[7] == 'x')
158 align = SVG_PRESERVEASPECTRATIO_XMAXYMAX;
159 else
160 return false;
161 } else
162 return false;
163 buffer += 8;
164 skipOptionalSVGSpaces(buffer);
165 } else
166 return false;
167
168 if (buffer.hasCharactersRemaining()) {
169 if (*buffer == 'm') {
170 if (!skipCharactersExactly(buffer, meetDesc<CharacterType>)) {
171 LOG_ERROR("Skipped to parse except for *meet* or *slice* value.");
172 return false;
173 }
174 skipOptionalSVGSpaces(buffer);
175 } else if (*buffer == 's') {
176 if (!skipCharactersExactly(buffer, sliceDesc<CharacterType>)) {
177 LOG_ERROR("Skipped to parse except for *meet* or *slice* value.");
178 return false;
179 }
180 skipOptionalSVGSpaces(buffer);
181 if (align != SVG_PRESERVEASPECTRATIO_NONE)
182 meetOrSlice = SVG_MEETORSLICE_SLICE;
183 }
184 }
185
186 if (!buffer.atEnd() && validate)
187 return false;
188
189 m_align = align;
190 m_meetOrSlice = meetOrSlice;
191
192 return true;
193}

195void SVGPreserveAspectRatioValue::transformRect(FloatRect& destRect, FloatRect& srcRect) const
196{
197 if (m_align == SVG_PRESERVEASPECTRATIO_NONE)
198 return;
199
200 FloatSize imageSize = srcRect.size();
201 float origDestWidth = destRect.width();
202 float origDestHeight = destRect.height();
203 switch (m_meetOrSlice) {
204 case SVGPreserveAspectRatioValue::SVG_MEETORSLICE_UNKNOWN:
205 break;
206 case SVGPreserveAspectRatioValue::SVG_MEETORSLICE_MEET: {
207 float widthToHeightMultiplier = srcRect.height() / srcRect.width();
208 if (origDestHeight > origDestWidth * widthToHeightMultiplier) {
209 destRect.setHeight(origDestWidth * widthToHeightMultiplier);
210 switch (m_align) {
211 case SVGPreserveAspectRatioValue::SVG_PRESERVEASPECTRATIO_XMINYMID:
212 case SVGPreserveAspectRatioValue::SVG_PRESERVEASPECTRATIO_XMIDYMID:
213 case SVGPreserveAspectRatioValue::SVG_PRESERVEASPECTRATIO_XMAXYMID:
214 destRect.setY(destRect.y() + origDestHeight / 2 - destRect.height() / 2);
215 break;
216 case SVGPreserveAspectRatioValue::SVG_PRESERVEASPECTRATIO_XMINYMAX:
217 case SVGPreserveAspectRatioValue::SVG_PRESERVEASPECTRATIO_XMIDYMAX:
218 case SVGPreserveAspectRatioValue::SVG_PRESERVEASPECTRATIO_XMAXYMAX:
219 destRect.setY(destRect.y() + origDestHeight - destRect.height());
220 break;
221 default:
222 break;
223 }
224 }
225 if (origDestWidth > origDestHeight / widthToHeightMultiplier) {
226 destRect.setWidth(origDestHeight / widthToHeightMultiplier);
227 switch (m_align) {
228 case SVGPreserveAspectRatioValue::SVG_PRESERVEASPECTRATIO_XMIDYMIN:
229 case SVGPreserveAspectRatioValue::SVG_PRESERVEASPECTRATIO_XMIDYMID:
230 case SVGPreserveAspectRatioValue::SVG_PRESERVEASPECTRATIO_XMIDYMAX:
231 destRect.setX(destRect.x() + origDestWidth / 2 - destRect.width() / 2);
232 break;
233 case SVGPreserveAspectRatioValue::SVG_PRESERVEASPECTRATIO_XMAXYMIN:
234 case SVGPreserveAspectRatioValue::SVG_PRESERVEASPECTRATIO_XMAXYMID:
235 case SVGPreserveAspectRatioValue::SVG_PRESERVEASPECTRATIO_XMAXYMAX:
236 destRect.setX(destRect.x() + origDestWidth - destRect.width());
237 break;
238 default:
239 break;
240 }
241 }
242 break;
243 }
244 case SVGPreserveAspectRatioValue::SVG_MEETORSLICE_SLICE: {
245 float widthToHeightMultiplier = srcRect.height() / srcRect.width();
246 // if the destination height is less than the height of the image we'll be drawing
247 if (origDestHeight < origDestWidth * widthToHeightMultiplier) {
248 float destToSrcMultiplier = srcRect.width() / destRect.width();
249 srcRect.setHeight(destRect.height() * destToSrcMultiplier);
250 switch (m_align) {
251 case SVGPreserveAspectRatioValue::SVG_PRESERVEASPECTRATIO_XMINYMID:
252 case SVGPreserveAspectRatioValue::SVG_PRESERVEASPECTRATIO_XMIDYMID:
253 case SVGPreserveAspectRatioValue::SVG_PRESERVEASPECTRATIO_XMAXYMID:
254 srcRect.setY(srcRect.y() + imageSize.height() / 2 - srcRect.height() / 2);
255 break;
256 case SVGPreserveAspectRatioValue::SVG_PRESERVEASPECTRATIO_XMINYMAX:
257 case SVGPreserveAspectRatioValue::SVG_PRESERVEASPECTRATIO_XMIDYMAX:
258 case SVGPreserveAspectRatioValue::SVG_PRESERVEASPECTRATIO_XMAXYMAX:
259 srcRect.setY(srcRect.y() + imageSize.height() - srcRect.height());
260 break;
261 default:
262 break;
263 }
264 }
265 // if the destination width is less than the width of the image we'll be drawing
266 if (origDestWidth < origDestHeight / widthToHeightMultiplier) {
267 float destToSrcMultiplier = srcRect.height() / destRect.height();
268 srcRect.setWidth(destRect.width() * destToSrcMultiplier);
269 switch (m_align) {
270 case SVGPreserveAspectRatioValue::SVG_PRESERVEASPECTRATIO_XMIDYMIN:
271 case SVGPreserveAspectRatioValue::SVG_PRESERVEASPECTRATIO_XMIDYMID:
272 case SVGPreserveAspectRatioValue::SVG_PRESERVEASPECTRATIO_XMIDYMAX:
273 srcRect.setX(srcRect.x() + imageSize.width() / 2 - srcRect.width() / 2);
274 break;
275 case SVGPreserveAspectRatioValue::SVG_PRESERVEASPECTRATIO_XMAXYMIN:
276 case SVGPreserveAspectRatioValue::SVG_PRESERVEASPECTRATIO_XMAXYMID:
277 case SVGPreserveAspectRatioValue::SVG_PRESERVEASPECTRATIO_XMAXYMAX:
278 srcRect.setX(srcRect.x() + imageSize.width() - srcRect.width());
279 break;
280 default:
281 break;
282 }
283 }
284 break;
285 }
286 }

289AffineTransform SVGPreserveAspectRatioValue::getCTM(float logicalX, float logicalY, float logicalWidth, float logicalHeight, float physicalWidth, float physicalHeight) const
290{
291 AffineTransform transform;
292 if (!logicalWidth || !logicalHeight || !physicalWidth || !physicalHeight) {
293 ASSERT_NOT_REACHED();
294 return transform;
295 }
296
297 if (m_align == SVG_PRESERVEASPECTRATIO_UNKNOWN)
298 return transform;
299
300 double extendedLogicalX = logicalX;
301 double extendedLogicalY = logicalY;
302 double extendedLogicalWidth = logicalWidth;
303 double extendedLogicalHeight = logicalHeight;
304 double extendedPhysicalWidth = physicalWidth;
305 double extendedPhysicalHeight = physicalHeight;
306 double logicalRatio = extendedLogicalWidth / extendedLogicalHeight;
307 double physicalRatio = extendedPhysicalWidth / extendedPhysicalHeight;
308
309 if (m_align == SVG_PRESERVEASPECTRATIO_NONE) {
310 transform.scaleNonUniform(extendedPhysicalWidth / extendedLogicalWidth, extendedPhysicalHeight / extendedLogicalHeight);
311 transform.translate(-extendedLogicalX, -extendedLogicalY);
312 return transform;
313 }
314
315 if ((logicalRatio < physicalRatio && (m_meetOrSlice == SVG_MEETORSLICE_MEET)) || (logicalRatio >= physicalRatio && (m_meetOrSlice == SVG_MEETORSLICE_SLICE))) {
316 transform.scaleNonUniform(extendedPhysicalHeight / extendedLogicalHeight, extendedPhysicalHeight / extendedLogicalHeight);
317
318 if (m_align == SVG_PRESERVEASPECTRATIO_XMINYMIN || m_align == SVG_PRESERVEASPECTRATIO_XMINYMID || m_align == SVG_PRESERVEASPECTRATIO_XMINYMAX)
319 transform.translate(-extendedLogicalX, -extendedLogicalY);
320 else if (m_align == SVG_PRESERVEASPECTRATIO_XMIDYMIN || m_align == SVG_PRESERVEASPECTRATIO_XMIDYMID || m_align == SVG_PRESERVEASPECTRATIO_XMIDYMAX)
321 transform.translate(-extendedLogicalX - (extendedLogicalWidth - extendedPhysicalWidth * extendedLogicalHeight / extendedPhysicalHeight) / 2, -extendedLogicalY);
322 else
323 transform.translate(-extendedLogicalX - (extendedLogicalWidth - extendedPhysicalWidth * extendedLogicalHeight / extendedPhysicalHeight), -extendedLogicalY);
324
325 return transform;
326 }
327
328 transform.scaleNonUniform(extendedPhysicalWidth / extendedLogicalWidth, extendedPhysicalWidth / extendedLogicalWidth);
329
330 if (m_align == SVG_PRESERVEASPECTRATIO_XMINYMIN || m_align == SVG_PRESERVEASPECTRATIO_XMIDYMIN || m_align == SVG_PRESERVEASPECTRATIO_XMAXYMIN)
331 transform.translate(-extendedLogicalX, -extendedLogicalY);
332 else if (m_align == SVG_PRESERVEASPECTRATIO_XMINYMID || m_align == SVG_PRESERVEASPECTRATIO_XMIDYMID || m_align == SVG_PRESERVEASPECTRATIO_XMAXYMID)
333 transform.translate(-extendedLogicalX, -extendedLogicalY - (extendedLogicalHeight - extendedPhysicalHeight * extendedLogicalWidth / extendedPhysicalWidth) / 2);
334 else
335 transform.translate(-extendedLogicalX, -extendedLogicalY - (extendedLogicalHeight - extendedPhysicalHeight * extendedLogicalWidth / extendedPhysicalWidth));
336
337 return transform;
338}

340String SVGPreserveAspectRatioValue::valueAsString() const
341{
342 auto alignType = [&]() {
343 switch (m_align) {
344 case SVG_PRESERVEASPECTRATIO_NONE:
345 return "none"_s;
346 case SVG_PRESERVEASPECTRATIO_XMINYMIN:
347 return "xMinYMin"_s;
348 case SVG_PRESERVEASPECTRATIO_XMIDYMIN:
349 return "xMidYMin"_s;
350 case SVG_PRESERVEASPECTRATIO_XMAXYMIN:
351 return "xMaxYMin"_s;
352 case SVG_PRESERVEASPECTRATIO_XMINYMID:
353 return "xMinYMid"_s;
354 case SVG_PRESERVEASPECTRATIO_XMIDYMID:
355 return "xMidYMid"_s;
356 case SVG_PRESERVEASPECTRATIO_XMAXYMID:
357 return "xMaxYMid"_s;
358 case SVG_PRESERVEASPECTRATIO_XMINYMAX:
359 return "xMinYMax"_s;
360 case SVG_PRESERVEASPECTRATIO_XMIDYMAX:
361 return "xMidYMax"_s;
362 case SVG_PRESERVEASPECTRATIO_XMAXYMAX:
363 return "xMaxYMax"_s;
364 case SVG_PRESERVEASPECTRATIO_UNKNOWN:
365 return "unknown"_s;
366 };
367
368 ASSERT_NOT_REACHED();
369 return "unknown"_s;
370 };
371
372 switch (m_meetOrSlice) {
373 default:
374 case SVG_MEETORSLICE_UNKNOWN:
375 return alignType();
376 case SVG_MEETORSLICE_MEET:
377 return makeString(alignType(), " meet");
378 case SVG_MEETORSLICE_SLICE:
379 return makeString(alignType(), " slice");
380 }
381}
