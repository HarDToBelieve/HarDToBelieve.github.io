29namespace WebCore {
30
31class SVGScriptElement final : public SVGElement, public SVGURIReference, public ScriptElement {
32 WTF_MAKE_ISO_ALLOCATED(SVGScriptElement);
33public:
34 static Ref<SVGScriptElement> create(const QualifiedName&, Document&, bool wasInsertedByParser);
35
36 using SVGElement::ref;
37 using SVGElement::deref;
38
39private:
40 SVGScriptElement(const QualifiedName&, Document&, bool wasInsertedByParser, bool alreadyStarted);
41
42 using PropertyRegistry = SVGPropertyOwnerRegistry<SVGScriptElement, SVGElement, SVGURIReference>;
43 const SVGPropertyRegistry& propertyRegistry() const final { return m_propertyRegistry; }
44
45 void parseAttribute(const QualifiedName&, const AtomString&) final;
46 void svgAttributeChanged(const QualifiedName&) final;
47
48 InsertedIntoAncestorResult insertedIntoAncestor(InsertionType, ContainerNode&) final;
49 void didFinishInsertingNode() final;
50 void childrenChanged(const ChildChange&) final;
51
52 bool isURLAttribute(const Attribute& attribute) const final { return attribute.name() == sourceAttributeValue(); }
53 void addSubresourceAttributeURLs(ListHashSet<URL>&) const final;
54
55 Ref<Element> cloneElementWithoutAttributesAndChildren(Document&) final;
56 bool rendererIsNeeded(const RenderStyle&) final { return false; }
57
58 // ScriptElement
59 String sourceAttributeValue() const final { return href(); }
60 String charsetAttributeValue() const final { return String(); }
61 String typeAttributeValue() const final { return getAttribute(SVGNames::typeAttr).string(); }
62 String languageAttributeValue() const final { return String(); }
63 String forAttributeValue() const final { return String(); }
64 String eventAttributeValue() const final { return String(); }
65 bool hasAsyncAttribute() const final { return false; }
66 bool hasDeferAttribute() const final { return false; }
67 bool hasNoModuleAttribute() const final { return false; }
68 ReferrerPolicy referrerPolicy() const final { return ReferrerPolicy::EmptyString; }
69 bool hasSourceAttribute() const final { return hasAttribute(SVGNames::hrefAttr) || hasAttribute(XLinkNames::hrefAttr); }
70 void dispatchLoadEvent() final { SVGURIReference::dispatchLoadEvent(); }
71 void dispatchErrorEvent() final;
72
73 // SVGElement
74 bool haveLoadedRequiredResources() final { return SVGURIReference::haveLoadedRequiredResources(); }
75 Timer* loadEventTimer() final { return &m_loadEventTimer; }
76
77 // SVGURIReference
78 bool haveFiredLoadEvent() const final { return ScriptElement::haveFiredLoadEvent(); }
79 void setHaveFiredLoadEvent(bool haveFiredLoadEvent) final { ScriptElement::setHaveFiredLoadEvent(haveFiredLoadEvent); }
80 bool errorOccurred() const final { return ScriptElement::errorOccurred(); }
81 void setErrorOccurred(bool errorOccurred) final { ScriptElement::setErrorOccurred(errorOccurred); }
82
83#ifndef NDEBUG
84 bool filterOutAnimatableAttribute(const QualifiedName& name) const final { return name == SVGNames::typeAttr; }
85#endif
86
87 PropertyRegistry m_propertyRegistry { *this };
88 Timer m_loadEventTimer;
89};
