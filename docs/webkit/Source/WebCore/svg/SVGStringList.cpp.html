34bool SVGStringList::parse(StringView data, UChar delimiter)
35{
36 clearItems();
37
38 auto isSVGSpaceOrDelimiter = [delimiter](auto c) {
39 return isSVGSpace(c) || c == delimiter;
40 };
41
42 return readCharactersForParsing(data, [&](auto buffer) {
43 skipOptionalSVGSpaces(buffer);
44
45 while (buffer.hasCharactersRemaining()) {
46 auto start = buffer.position();
47
48 // FIXME: It would be a nice improvement to add a variant of skipUntil which worked
49 // with lambda predicates.
50 while (buffer.hasCharactersRemaining() && !isSVGSpaceOrDelimiter(*buffer))
51 ++buffer;
52
53 if (buffer.position() == start)
54 break;
55
56 m_items.append(String(start, buffer.position() - start));
57 skipOptionalSVGSpacesOrDelimiter(buffer, delimiter);
58 }
59
60 // FIXME: Should this clearItems() on failure like SVGTransformList does?
61
62 return buffer.atEnd();
63 });
64}
65
66String SVGStringList::valueAsString() const
67{
68 StringBuilder builder;
69
70 for (const auto& string : m_items) {
71 if (builder.length())
72 builder.append(' ');
73
74 builder.append(string);
75 }
76
77 return builder.toString();
78}
