30namespace WebCore {
31
32WTF_MAKE_ISO_ALLOCATED_IMPL(SVGSymbolElement);
33
34inline SVGSymbolElement::SVGSymbolElement(const QualifiedName& tagName, Document& document)
35 : SVGGraphicsElement(tagName, document)
36 , SVGFitToViewBox(this)
37{
38 ASSERT(hasTagName(SVGNames::symbolTag));
39}
40
41Ref<SVGSymbolElement> SVGSymbolElement::create(const QualifiedName& tagName, Document& document)
42{
43 return adoptRef(*new SVGSymbolElement(tagName, document));
44}
45
46void SVGSymbolElement::parseAttribute(const QualifiedName& name, const AtomString& value)
47{
48 SVGGraphicsElement::parseAttribute(name, value);
49 SVGFitToViewBox::parseAttribute(name, value);
50}
51
52bool SVGSymbolElement::selfHasRelativeLengths() const
53{
54 return hasAttribute(SVGNames::viewBoxAttr);
55}
56
57RenderPtr<RenderElement> SVGSymbolElement::createElementRenderer(RenderStyle&& style, const RenderTreePosition&)
58{
59 return createRenderer<RenderSVGHiddenContainer>(*this, WTFMove(style));
60}
