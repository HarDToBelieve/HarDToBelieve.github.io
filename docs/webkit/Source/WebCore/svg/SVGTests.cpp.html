89constexpr SortedArraySet supportedSVGFeatureSet { supportedSVGFeatureArray };

91SVGTests::SVGTests(SVGElement* contextElement)
92 : m_contextElement(*contextElement)
93 , m_requiredFeatures(SVGStringList::create(contextElement))
94 , m_requiredExtensions(SVGStringList::create(contextElement))
95 , m_systemLanguage(SVGStringList::create(contextElement))
96{
97 static std::once_flag onceFlag;
98 std::call_once(onceFlag, [] {
99 PropertyRegistry::registerProperty<SVGNames::requiredFeaturesAttr, &SVGTests::m_requiredFeatures>();
100 PropertyRegistry::registerProperty<SVGNames::requiredExtensionsAttr, &SVGTests::m_requiredExtensions>();
101 PropertyRegistry::registerProperty<SVGNames::systemLanguageAttr, &SVGTests::m_systemLanguage>();
102 });
103}

105bool SVGTests::hasExtension(const String& extension)
106{
107 // We recognize XHTML and MathML, as implemented in Gecko and suggested in the SVG Tiny recommendation (http://www.w3.org/TR/SVG11/struct.html#RequiredExtensionsAttribute).
108#if ENABLE(MATHML)
109 if (extension == MathMLNames::mathmlNamespaceURI)
110 return true;
111#endif
112 return extension == HTMLNames::xhtmlNamespaceURI;
113}
114
115bool SVGTests::isValid() const
116{
117 for (auto& feature : m_requiredFeatures->items()) {
118 if (feature.isEmpty() || !supportedSVGFeatureSet.contains(feature))
119 return false;
120 }
121 for (auto& language : m_systemLanguage->items()) {
122 if (language != defaultLanguage().substring(0, 2))
123 return false;
124 }
125 for (auto& extension : m_requiredExtensions->items()) {
126 if (!hasExtension(extension))
127 return false;
128 }
129 return true;
130}

132void SVGTests::parseAttribute(const QualifiedName& attributeName, const AtomString& value)
133{
134 if (attributeName == SVGNames::requiredFeaturesAttr)
135 m_requiredFeatures->reset(value);
136 if (attributeName == SVGNames::requiredExtensionsAttr)
137 m_requiredExtensions->reset(value);
138 if (attributeName == SVGNames::systemLanguageAttr)
139 m_systemLanguage->reset(value);
140}
141
142void SVGTests::svgAttributeChanged(const QualifiedName& attrName)
143{
144 if (!PropertyRegistry::isKnownAttribute(attrName))
145 return;
146
147 if (!m_contextElement.isConnected())
148 return;
149 m_contextElement.invalidateStyleAndRenderersForSubtree();
150}

152void SVGTests::addSupportedAttributes(MemoryCompactLookupOnlyRobinHoodHashSet<QualifiedName>& supportedAttributes)
153{
154 supportedAttributes.add(SVGNames::requiredFeaturesAttr);
155 supportedAttributes.add(SVGNames::requiredExtensionsAttr);
156 supportedAttributes.add(SVGNames::systemLanguageAttr);
157}

159bool SVGTests::hasFeatureForLegacyBindings(const String& feature, const String& version)
160{
161 // FIXME: This function is here only to be exposed in the Objective-C and GObject bindings for both Node and DOMImplementation.
162 // It's likely that we can just remove this and instead have the bindings return true unconditionally.
163 // This is what the DOMImplementation function now does in JavaScript as is now suggested in the DOM specification.
164 // The behavior implemented below is quirky, but preserves what WebKit has done for at least the last few years.
165
166 bool hasSVG10FeaturePrefix = startsWithLettersIgnoringASCIICase(feature, "org.w3c.dom.svg") || startsWithLettersIgnoringASCIICase(feature, "org.w3c.svg");
167 bool hasSVG11FeaturePrefix = startsWithLettersIgnoringASCIICase(feature, "http://www.w3.org/tr/svg");
168
169 // We don't even try to handle feature names that don't look like the SVG ones, so just return true for all of those.
170 if (!(hasSVG10FeaturePrefix || hasSVG11FeaturePrefix))
171 return true;
172
173 // If the version number matches the style of the feature name, then use the set to see if the feature is supported.
174 if (version.isEmpty() || (hasSVG10FeaturePrefix && version == "1.0") || (hasSVG11FeaturePrefix && version == "1.1"))
175 return supportedSVGFeatureSet.contains(feature);
176
177 return false;
178}
