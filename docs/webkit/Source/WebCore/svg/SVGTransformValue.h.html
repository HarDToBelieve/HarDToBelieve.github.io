31namespace WebCore {

33class FloatSize;
34
35class SVGTransformValue {
36public:
37 enum SVGTransformType {
38 SVG_TRANSFORM_UNKNOWN = 0,
39 SVG_TRANSFORM_MATRIX = 1,
40 SVG_TRANSFORM_TRANSLATE = 2,
41 SVG_TRANSFORM_SCALE = 3,
42 SVG_TRANSFORM_ROTATE = 4,
43 SVG_TRANSFORM_SKEWX = 5,
44 SVG_TRANSFORM_SKEWY = 6
45 };
46
47 enum ConstructionMode {
48 ConstructIdentityTransform,
49 ConstructZeroTransform
50 };
51
52 SVGTransformValue(SVGTransformType type = SVG_TRANSFORM_MATRIX, const AffineTransform& transform = { })
53 : m_type(type)
54 , m_matrix(SVGMatrix::create(transform))
55 {
56 }
57
58 SVGTransformValue(const SVGTransformValue& other)
59 : m_type(other.m_type)
60 , m_matrix(SVGMatrix::create(other.matrix()->value()))
61 , m_angle(other.m_angle)
62 , m_rotationCenter(other.m_rotationCenter)
63 {
64 }
65
66 SVGTransformValue(SVGTransformType type, Ref<SVGMatrix>&& matrix, float angle, const FloatPoint& rotationCenter)
67 : m_type(type)
68 , m_matrix(WTFMove(matrix))
69 , m_angle(angle)
70 , m_rotationCenter(rotationCenter)
71 {
72 }
73
74 SVGTransformValue(SVGTransformValue&& other)
75 : m_type(other.m_type)
76 , m_matrix(other.m_matrix.copyRef())
77 , m_angle(other.m_angle)
78 , m_rotationCenter(other.m_rotationCenter)
79 {
80 }
81
82 SVGTransformValue& operator=(const SVGTransformValue& other)
83 {
84 m_type = other.m_type;
85 m_matrix->setValue(other.m_matrix->value());
86 m_angle = other.m_angle;
87 m_rotationCenter = other.m_rotationCenter;
88 return *this;
89 }
90
91 SVGTransformType type() const { return m_type; }
92 const Ref<SVGMatrix>& matrix() const { return m_matrix; }
93 float angle() const { return m_angle; }
94 FloatPoint rotationCenter() const { return m_rotationCenter; }
95
96 bool isValid() const { return m_type != SVG_TRANSFORM_UNKNOWN; }
97
98 void setMatrix(const AffineTransform& matrix)
99 {
100 m_type = SVG_TRANSFORM_MATRIX;
101 m_angle = 0;
102 m_rotationCenter = FloatPoint();
103 m_matrix->setValue(matrix);
104 }
105
106 void matrixDidChange()
107 {
108 // The underlying matrix has been changed, alter the transformation type.
109 // Spec: In case the matrix object is changed directly (i.e., without using the methods on the SVGTransform interface itself)
110 // then the type of the SVGTransform changes to SVG_TRANSFORM_MATRIX.
111 m_type = SVG_TRANSFORM_MATRIX;
112 m_angle = 0;
113 m_rotationCenter = FloatPoint();
114 }
115
116 FloatPoint translate() const
117 {
118 return FloatPoint::narrowPrecision(m_matrix->e(), m_matrix->f());
119 }
120
121 void setTranslate(float tx, float ty)
122 {
123 m_type = SVG_TRANSFORM_TRANSLATE;
124 m_angle = 0;
125 m_rotationCenter = FloatPoint();
126
127 m_matrix->value().makeIdentity();
128 m_matrix->value().translate(tx, ty);
129 }
130
131 FloatSize scale() const
132 {
133 return FloatSize::narrowPrecision(m_matrix->a(), m_matrix->d());
134 }
135
136 void setScale(float sx, float sy)
137 {
138 m_type = SVG_TRANSFORM_SCALE;
139 m_angle = 0;
140 m_rotationCenter = FloatPoint();
141
142 m_matrix->value().makeIdentity();
143 m_matrix->value().scaleNonUniform(sx, sy);
144 }
145
146 void setRotate(float angle, float cx, float cy)
147 {
148 m_type = SVG_TRANSFORM_ROTATE;
149 m_angle = angle;
150 m_rotationCenter = FloatPoint(cx, cy);
151
152 // TODO: toString() implementation, which can show cx, cy (need to be stored?)
153 m_matrix->value().makeIdentity();
154 m_matrix->value().translate(cx, cy);
155 m_matrix->value().rotate(angle);
156 m_matrix->value().translate(-cx, -cy);
157 }
158
159 void setSkewX(float angle)
160 {
161 m_type = SVG_TRANSFORM_SKEWX;
162 m_angle = angle;
163 m_rotationCenter = FloatPoint();
164
165 m_matrix->value().makeIdentity();
166 m_matrix->value().skewX(angle);
167 }
168
169 void setSkewY(float angle)
170 {
171 m_type = SVG_TRANSFORM_SKEWY;
172 m_angle = angle;
173 m_rotationCenter = FloatPoint();
174
175 m_matrix->value().makeIdentity();
176 m_matrix->value().skewY(angle);
177 }
178
179 String valueAsString() const
180 {
181 StringBuilder builder;
182 builder.append(prefixForTransfromType(m_type));
183 switch (m_type) {
184 case SVG_TRANSFORM_UNKNOWN:
185 break;
186 case SVG_TRANSFORM_MATRIX:
187 appendMatrix(builder);
188 break;
189 case SVG_TRANSFORM_TRANSLATE:
190 appendTranslate(builder);
191 break;
192 case SVG_TRANSFORM_SCALE:
193 appendScale(builder);
194 break;
195 case SVG_TRANSFORM_ROTATE:
196 appendRotate(builder);
197 break;
198 case SVG_TRANSFORM_SKEWX:
199 appendSkewX(builder);
200 break;
201 case SVG_TRANSFORM_SKEWY:
202 appendSkewY(builder);
203 break;
204 }
205 return builder.toString();
206 }
207
208 static const char* prefixForTransfromType(SVGTransformType type)
209 {
210 switch (type) {
211 case SVG_TRANSFORM_UNKNOWN:
212 return "";
213 case SVG_TRANSFORM_MATRIX:
214 return "matrix(";
215 case SVG_TRANSFORM_TRANSLATE:
216 return "translate(";
217 case SVG_TRANSFORM_SCALE:
218 return "scale(";
219 case SVG_TRANSFORM_ROTATE:
220 return "rotate(";
221 case SVG_TRANSFORM_SKEWX:
222 return "skewX(";
223 case SVG_TRANSFORM_SKEWY:
224 return "skewY(";
225 }
226 ASSERT_NOT_REACHED();
227 return "";
228 }
229
230private:
231 static void appendFixedPrecisionNumbers(StringBuilder& builder)
232 {
233 builder.append(')');
234 }
235
236 template<typename Number, typename... Numbers>
237 static void appendFixedPrecisionNumbers(StringBuilder& builder, Number number, Numbers... numbers)
238 {
239 if (builder.length() && builder[builder.length() - 1] != '(')
240 builder.append(' ');
241 // FIXME: Shortest form would be better, but fixed precision is required for now to smooth over precision errors caused by converting float to double and back since we use AffineTransform to store transforms.
242 builder.append(FormattedNumber::fixedPrecision(number));
243 appendFixedPrecisionNumbers(builder, numbers...);
244 }
245
246 void appendMatrix(StringBuilder& builder) const
247 {
248 appendFixedPrecisionNumbers(builder, m_matrix->a(), m_matrix->b(), m_matrix->c(), m_matrix->d(), m_matrix->e(), m_matrix->f());
249 }
250
251 void appendTranslate(StringBuilder& builder) const
252 {
253 appendFixedPrecisionNumbers(builder, m_matrix->e(), m_matrix->f());
254 }
255
256 void appendScale(StringBuilder& builder) const
257 {
258 appendFixedPrecisionNumbers(builder, m_matrix->value().xScale(), m_matrix->value().yScale());
259 }
260
261 void appendRotate(StringBuilder& builder) const
262 {
263 double angleInRad = deg2rad(m_angle);
264 double cosAngle = std::cos(angleInRad);
265 double sinAngle = std::sin(angleInRad);
266
267 float cx = narrowPrecisionToFloat(cosAngle != 1 ? (m_matrix->e() * (1 - cosAngle) - m_matrix->f() * sinAngle) / (1 - cosAngle) / 2 : 0);
268 float cy = narrowPrecisionToFloat(cosAngle != 1 ? (m_matrix->e() * sinAngle / (1 - cosAngle) + m_matrix->f()) / 2 : 0);
269
270 if (cx || cy)
271 appendFixedPrecisionNumbers(builder, m_angle, cx, cy);
272 else
273 appendFixedPrecisionNumbers(builder, m_angle);
274 }
275
276 void appendSkewX(StringBuilder& builder) const
277 {
278 appendFixedPrecisionNumbers(builder, m_angle);
279 }
280
281 void appendSkewY(StringBuilder& builder) const
282 {
283 appendFixedPrecisionNumbers(builder, m_angle);
284 }
285
286 SVGTransformType m_type { SVG_TRANSFORM_UNKNOWN };
287 Ref<SVGMatrix> m_matrix;
288 float m_angle { 0 };
289 FloatPoint m_rotationCenter;
290};
