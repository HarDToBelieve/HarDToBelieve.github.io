37template<typename CharacterType> static int parseTransformParamList(StringParsingBuffer<CharacterType>& buffer, float* values, int required, int optional)
38{
39 int optionalParams = 0, requiredParams = 0;
40
41 if (!skipOptionalSVGSpaces(buffer) || *buffer != '(')
42 return -1;
43
44 ++buffer;
45
46 skipOptionalSVGSpaces(buffer);
47
48 while (requiredParams < required) {
49 if (buffer.atEnd())
50 return -1;
51 auto parsedNumber = parseNumber(buffer, SuffixSkippingPolicy::DontSkip);
52 if (!parsedNumber)
53 return -1;
54 values[requiredParams] = *parsedNumber;
55 requiredParams++;
56 if (requiredParams < required)
57 skipOptionalSVGSpacesOrDelimiter(buffer);
58 }
59 if (!skipOptionalSVGSpaces(buffer))
60 return -1;
61
62 bool delimParsed = skipOptionalSVGSpacesOrDelimiter(buffer);
63
64 if (buffer.atEnd())
65 return -1;
66
67 if (*buffer == ')') {
68 // skip optionals
69 ++buffer;
70 if (delimParsed)
71 return -1;
72 } else {
73 while (optionalParams < optional) {
74 if (buffer.atEnd())
75 return -1;
76 auto parsedNumber = parseNumber(buffer, SuffixSkippingPolicy::DontSkip);
77 if (!parsedNumber)
78 return -1;
79 values[requiredParams + optionalParams] = *parsedNumber;
80 optionalParams++;
81 if (optionalParams < optional)
82 skipOptionalSVGSpacesOrDelimiter(buffer);
83 }
84
85 if (!skipOptionalSVGSpaces(buffer))
86 return -1;
87
88 delimParsed = skipOptionalSVGSpacesOrDelimiter(buffer);
89
90 if (buffer.atEnd() || *buffer != ')' || delimParsed)
91 return -1;
92 ++buffer;
93 }
94
95 return requiredParams + optionalParams;
96}

98// These should be kept in sync with enum SVGTransformType
99static constexpr int requiredValuesForType[] = { 0, 6, 1, 1, 1, 1, 1 };
100static constexpr int optionalValuesForType[] = { 0, 0, 1, 1, 2, 0, 0 };

102template<typename CharacterType> static std::optional<SVGTransformValue> parseTransformValueGeneric(SVGTransformValue::SVGTransformType type, StringParsingBuffer<CharacterType>& buffer)
103{
104 if (type == SVGTransformValue::SVG_TRANSFORM_UNKNOWN)
105 return std::nullopt;
106
107 int valueCount = 0;
108 float values[] = {0, 0, 0, 0, 0, 0};
109 if ((valueCount = parseTransformParamList(buffer, values, requiredValuesForType[type], optionalValuesForType[type])) < 0)
110 return std::nullopt;
111
112 SVGTransformValue transform;
113 switch (type) {
114 case SVGTransformValue::SVG_TRANSFORM_UNKNOWN:
115 ASSERT_NOT_REACHED();
116 break;
117 case SVGTransformValue::SVG_TRANSFORM_SKEWX:
118 transform.setSkewX(values[0]);
119 break;
120 case SVGTransformValue::SVG_TRANSFORM_SKEWY:
121 transform.setSkewY(values[0]);
122 break;
123 case SVGTransformValue::SVG_TRANSFORM_SCALE:
124 if (valueCount == 1) // Spec: if only one param given, assume uniform scaling
125 transform.setScale(values[0], values[0]);
126 else
127 transform.setScale(values[0], values[1]);
128 break;
129 case SVGTransformValue::SVG_TRANSFORM_TRANSLATE:
130 if (valueCount == 1) // Spec: if only one param given, assume 2nd param to be 0
131 transform.setTranslate(values[0], 0);
132 else
133 transform.setTranslate(values[0], values[1]);
134 break;
135 case SVGTransformValue::SVG_TRANSFORM_ROTATE:
136 if (valueCount == 1)
137 transform.setRotate(values[0], 0, 0);
138 else
139 transform.setRotate(values[0], values[1], values[2]);
140 break;
141 case SVGTransformValue::SVG_TRANSFORM_MATRIX:
142 transform.setMatrix(AffineTransform(values[0], values[1], values[2], values[3], values[4], values[5]));
143 break;
144 }
145
146 return transform;
147}

159template<typename CharacterType> static constexpr CharacterType skewXDesc[] = {'s', 'k', 'e', 'w', 'X'};
160template<typename CharacterType> static constexpr CharacterType skewYDesc[] = {'s', 'k', 'e', 'w', 'Y'};
161template<typename CharacterType> static constexpr CharacterType scaleDesc[] = {'s', 'c', 'a', 'l', 'e'};
162template<typename CharacterType> static constexpr CharacterType translateDesc[] = {'t', 'r', 'a', 'n', 's', 'l', 'a', 't', 'e'};
163template<typename CharacterType> static constexpr CharacterType rotateDesc[] = {'r', 'o', 't', 'a', 't', 'e'};
164template<typename CharacterType> static constexpr CharacterType matrixDesc[] = {'m', 'a', 't', 'r', 'i', 'x'};
165
166template<typename CharacterType> static std::optional<SVGTransformValue::SVGTransformType> parseTransformTypeGeneric(StringParsingBuffer<CharacterType>& buffer)
167{
168 if (buffer.atEnd())
169 return std::nullopt;
170
171 if (*buffer == 's') {
172 if (skipCharactersExactly(buffer, skewXDesc<CharacterType>))
173 return SVGTransformValue::SVG_TRANSFORM_SKEWX;
174 if (skipCharactersExactly(buffer, skewYDesc<CharacterType>))
175 return SVGTransformValue::SVG_TRANSFORM_SKEWY;
176 if (skipCharactersExactly(buffer, scaleDesc<CharacterType>))
177 return SVGTransformValue::SVG_TRANSFORM_SCALE;
178 return std::nullopt;
179 }
180
181 if (skipCharactersExactly(buffer, translateDesc<CharacterType>))
182 return SVGTransformValue::SVG_TRANSFORM_TRANSLATE;
183 if (skipCharactersExactly(buffer, rotateDesc<CharacterType>))
184 return SVGTransformValue::SVG_TRANSFORM_ROTATE;
185 if (skipCharactersExactly(buffer, matrixDesc<CharacterType>))
186 return SVGTransformValue::SVG_TRANSFORM_MATRIX;
187
188 return std::nullopt;
189}

191std::optional<SVGTransformValue::SVGTransformType> SVGTransformable::parseTransformType(StringView string)
192{
193 return readCharactersForParsing(string, [](auto buffer) {
194 return parseTransformType(buffer);
195 });
196}
