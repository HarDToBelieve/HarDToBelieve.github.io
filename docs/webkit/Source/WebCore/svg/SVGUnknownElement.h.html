26#pragma once
27
28#include "SVGElement.h"
29
30namespace WebCore {
31
32// This type is used for 2 kinds of elements:
33// - Unknown Elements in SVG namespace
34// - Registered custom tag elements in SVG namespace (http://www.w3.org/TR/2013/WD-custom-elements-20130514/#registering-custom-elements)
35//
36// The main purpose of this class at the moment is to override rendererIsNeeded() to return
37// false to make sure we don't attempt to render such elements.
38class SVGUnknownElement final : public SVGElement {
39 WTF_MAKE_ISO_ALLOCATED(SVGUnknownElement);
40public:
41 static Ref<SVGUnknownElement> create(const QualifiedName& tagName, Document& document)
42 {
43 return adoptRef(*new SVGUnknownElement(tagName, document));
44 }
45
46private:
47 SVGUnknownElement(const QualifiedName& tagName, Document& document)
48 : SVGElement(tagName, document, CreateSVGElement | NodeFlag::IsUnknownElement)
49 {
50 }
51
52 bool rendererIsNeeded(const RenderStyle&) final { return false; }
53};
54
55} // namespace WebCore
