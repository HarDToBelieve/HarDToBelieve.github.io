30template<typename CharacterType> static constexpr CharacterType disable[] = { 'd', 'i', 's', 'a', 'b', 'l', 'e' };
31template<typename CharacterType> static constexpr CharacterType magnify[] = { 'm', 'a', 'g', 'n', 'i', 'f', 'y' };
32
33template<typename CharacterType> static std::optional<SVGZoomAndPanType> parseZoomAndPanGeneric(StringParsingBuffer<CharacterType>& buffer)
34{
35 if (skipCharactersExactly(buffer, disable<CharacterType>))
36 return SVGZoomAndPanDisable;
37
38 if (skipCharactersExactly(buffer, magnify<CharacterType>))
39 return SVGZoomAndPanMagnify;
40
41 return std::nullopt;
42}
43
44std::optional<SVGZoomAndPanType> SVGZoomAndPan::parseZoomAndPan(StringParsingBuffer<LChar>& buffer)
45{
46 return parseZoomAndPanGeneric(buffer);
47}
48
49std::optional<SVGZoomAndPanType> SVGZoomAndPan::parseZoomAndPan(StringParsingBuffer<UChar>& buffer)
50{
51 return parseZoomAndPanGeneric(buffer);
52}
53
54void SVGZoomAndPan::parseAttribute(const QualifiedName& attributeName, const AtomString& value)
55{
56 if (attributeName != SVGNames::zoomAndPanAttr)
57 return;
58 m_zoomAndPan = SVGPropertyTraits<SVGZoomAndPanType>::fromString(value);
59}
