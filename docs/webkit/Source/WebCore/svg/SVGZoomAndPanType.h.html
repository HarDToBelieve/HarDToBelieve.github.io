26#pragma once
27
28#include "SVGPropertyTraits.h"
29
30namespace WebCore {
31
32enum SVGZoomAndPanType {
33 SVGZoomAndPanUnknown,
34 SVGZoomAndPanDisable,
35 SVGZoomAndPanMagnify
36};
37
38template<>
39struct SVGPropertyTraits<SVGZoomAndPanType> {
40 static SVGZoomAndPanType initialValue() { return SVGZoomAndPanMagnify; }
41 static String toString(SVGZoomAndPanType) { return emptyString(); }
42 static SVGZoomAndPanType fromString(const String& value)
43 {
44 if (value == "disable")
45 return SVGZoomAndPanDisable;
46 if (value == "magnify")
47 return SVGZoomAndPanMagnify;
48 return SVGZoomAndPanUnknown;
49 }
50};
51
52}
