26#include "config.h"
27#include "SMILTime.h"
28
29#include <float.h>
30
31namespace WebCore {
32
33const double SMILTime::unresolvedValue = DBL_MAX;
34// Just a big value smaller than DBL_MAX. Our times are relative to 0, we don't really need the full range.
35const double SMILTime::indefiniteValue = FLT_MAX;
36
37SMILTime operator+(const SMILTime& a, const SMILTime& b)
38{
39 if (a.isUnresolved() || b.isUnresolved())
40 return SMILTime::unresolved();
41 if (a.isIndefinite() || b.isIndefinite())
42 return SMILTime::indefinite();
43 return a.value() + b.value();
44}
45
46SMILTime operator-(const SMILTime& a, const SMILTime& b)
47{
48 if (a.isUnresolved() || b.isUnresolved())
49 return SMILTime::unresolved();
50 if (a.isIndefinite() || b.isIndefinite())
51 return SMILTime::indefinite();
52 return a.value() - b.value();
53}
54
55SMILTime operator*(const SMILTime& a, const SMILTime& b)
56{
57 if (a.isUnresolved() || b.isUnresolved())
58 return SMILTime::unresolved();
59 if (!a.value() || !b.value())
60 return SMILTime(0);
61 if (a.isIndefinite() || b.isIndefinite())
62 return SMILTime::indefinite();
63 return a.value() * b.value();
64}
