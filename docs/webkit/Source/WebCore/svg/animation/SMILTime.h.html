32namespace WebCore {
33
34class SMILTime {
35public:
36 SMILTime() : m_time(0) { }
37 SMILTime(double time) : m_time(time) { ASSERT(!std::isnan(time)); }
38 SMILTime(Seconds time) : m_time(time.value()) { ASSERT(!std::isnan(time.value())); }
39 SMILTime(const SMILTime& o) : m_time(o.m_time) { }
40
41 static SMILTime unresolved() { return unresolvedValue; }
42 static SMILTime indefinite() { return indefiniteValue; }
43
44 SMILTime& operator=(const SMILTime& o) { m_time = o.m_time; return *this; }
45 double value() const { return m_time; }
46
47 bool isFinite() const { return m_time < indefiniteValue; }
48 bool isIndefinite() const { return m_time == indefiniteValue; }
49 bool isUnresolved() const { return m_time == unresolvedValue; }
50
51private:
52 static const double unresolvedValue;
53 static const double indefiniteValue;
54
55 double m_time;
56};
57
58class SMILTimeWithOrigin {
59public:
60 enum Origin {
61 ParserOrigin,
62 ScriptOrigin
63 };
64
65 SMILTimeWithOrigin()
66 : m_origin(ParserOrigin)
67 {
68 }
69
70 SMILTimeWithOrigin(const SMILTime& time, Origin origin)
71 : m_time(time)
72 , m_origin(origin)
73 {
74 }
75
76 const SMILTime& time() const { return m_time; }
77 bool originIsScript() const { return m_origin == ScriptOrigin; }
78
79private:
80 SMILTime m_time;
81 Origin m_origin;
82};
83
84inline bool operator==(const SMILTime& a, const SMILTime& b) { return a.isFinite() && a.value() == b.value(); }
85inline bool operator!(const SMILTime& a) { return !a.isFinite() || !a.value(); }
86inline bool operator!=(const SMILTime& a, const SMILTime& b) { return !operator==(a, b); }
87inline bool operator>(const SMILTime& a, const SMILTime& b) { return a.value() > b.value(); }
88inline bool operator<(const SMILTime& a, const SMILTime& b) { return a.value() < b.value(); }
89inline bool operator>=(const SMILTime& a, const SMILTime& b) { return a.value() > b.value() || operator==(a, b); }
90inline bool operator<=(const SMILTime& a, const SMILTime& b) { return a.value() < b.value() || operator==(a, b); }
91inline bool operator<(const SMILTimeWithOrigin& a, const SMILTimeWithOrigin& b) { return a.time() < b.time(); }
92
93SMILTime operator+(const SMILTime&, const SMILTime&);
94SMILTime operator-(const SMILTime&, const SMILTime&);
95// So multiplying times does not make too much sense but SMIL defines it for duration * repeatCount
96SMILTime operator*(const SMILTime&, const SMILTime&);
