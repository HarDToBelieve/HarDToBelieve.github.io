104bool SVGImage::hasSingleSecurityOrigin() const
105{
106 auto rootElement = this->rootElement();
107 if (!rootElement)
108 return true;
109
110 // FIXME: Once foreignObject elements within SVG images are updated to not leak cross-origin data
111 // (e.g., visited links, spellcheck) we can remove the SVGForeignObjectElement check here and
112 // research if we can remove the Image::hasSingleSecurityOrigin mechanism entirely.
113 for (auto& element : descendantsOfType<SVGElement>(*rootElement)) {
114 if (is<SVGForeignObjectElement>(element))
115 return false;
116 if (is<SVGImageElement>(element)) {
117 if (!downcast<SVGImageElement>(element).hasSingleSecurityOrigin())
118 return false;
119 } else if (is<SVGFEImageElement>(element)) {
120 if (!downcast<SVGFEImageElement>(element).hasSingleSecurityOrigin())
121 return false;
122 }
123 }
124
125 // Because SVG image rendering disallows external resources and links,
126 // these images effectively are restricted to a single security origin.
127 return true;
128}

130void SVGImage::setContainerSize(const FloatSize& size)
131{
132 if (!usesContainerSize())
133 return;
134
135 auto rootElement = this->rootElement();
136 if (!rootElement)
137 return;
138 auto* renderer = downcast<RenderSVGRoot>(rootElement->renderer());
139 if (!renderer)
140 return;
141
142 auto view = makeRefPtr(frameView());
143 view->resize(this->containerSize());
144
145 renderer->setContainerSize(IntSize(size));
146}
147
148IntSize SVGImage::containerSize() const
149{
150 auto rootElement = this->rootElement();
151 if (!rootElement)
152 return IntSize();
153
154 auto* renderer = downcast<RenderSVGRoot>(rootElement->renderer());
155 if (!renderer)
156 return IntSize();
157
158 // If a container size is available it has precedence.
159 IntSize containerSize = renderer->containerSize();
160 if (!containerSize.isEmpty())
161 return containerSize;
162
163 // Assure that a container size is always given for a non-identity zoom level.
164 ASSERT(renderer->style().effectiveZoom() == 1);
165
166 FloatSize currentSize;
167 if (rootElement->hasIntrinsicWidth() && rootElement->hasIntrinsicHeight())
168 currentSize = rootElement->currentViewportSize();
169 else
170 currentSize = rootElement->currentViewBoxRect().size();
171
172 // Use the default CSS intrinsic size if the above failed.
173 if (currentSize.isEmpty())
174 return IntSize(300, 150);
175
176 return IntSize(currentSize);
177}

179ImageDrawResult SVGImage::drawForContainer(GraphicsContext& context, const FloatSize containerSize, float containerZoom, const URL& initialFragmentURL, const FloatRect& dstRect,
180 const FloatRect& srcRect, const ImagePaintingOptions& options)
181{
182 if (!m_page)
183 return ImageDrawResult::DidNothing;
184
185 ImageObserver* observer = imageObserver();
186 ASSERT(observer);
187
188 // Temporarily reset image observer, we don't want to receive any changeInRect() calls due to this relayout.
189 setImageObserver(nullptr);
190
191 IntSize roundedContainerSize = roundedIntSize(containerSize);
192 setContainerSize(roundedContainerSize);
193
194 FloatRect scaledSrc = srcRect;
195 scaledSrc.scale(1 / containerZoom);
196
197 // Compensate for the container size rounding by adjusting the source rect.
198 FloatSize adjustedSrcSize = scaledSrc.size();
199 adjustedSrcSize.scale(roundedContainerSize.width() / containerSize.width(), roundedContainerSize.height() / containerSize.height());
200 scaledSrc.setSize(adjustedSrcSize);
201
202 frameView()->scrollToFragment(initialFragmentURL);
203
204 ImageDrawResult result = draw(context, dstRect, scaledSrc, options);
205
206 setImageObserver(observer);
207 return result;
208}

215RefPtr<NativeImage> SVGImage::nativeImage(const GraphicsContext*)
216{
217 if (!m_page)
218 return nullptr;
219
220 auto imageBuffer = ImageBuffer::create(size(), RenderingMode::Unaccelerated, 1, DestinationColorSpace::SRGB(), PixelFormat::BGRA8);
221 if (!imageBuffer)
222 return nullptr;
223
224 ImageObserver* observer = imageObserver();
225
226 setImageObserver(nullptr);
227 setContainerSize(size());
228
229 imageBuffer->context().drawImage(*this, FloatPoint(0, 0));
230
231 setImageObserver(observer);
232 return ImageBuffer::sinkIntoNativeImage(WTFMove(imageBuffer));
233}

235void SVGImage::drawPatternForContainer(GraphicsContext& context, const FloatSize& containerSize, float containerZoom, const URL& initialFragmentURL, const FloatRect& srcRect,
236 const AffineTransform& patternTransform, const FloatPoint& phase, const FloatSize& spacing, const FloatRect& dstRect, const ImagePaintingOptions& options)
237{
238 FloatRect zoomedContainerRect = FloatRect(FloatPoint(), containerSize);
239 zoomedContainerRect.scale(containerZoom);
240
241 // The ImageBuffer size needs to be scaled to match the final resolution.
242 AffineTransform transform = context.getCTM();
243 FloatSize imageBufferScale = FloatSize(transform.xScale(), transform.yScale());
244 ASSERT(imageBufferScale.width());
245 ASSERT(imageBufferScale.height());
246
247 FloatRect imageBufferSize = zoomedContainerRect;
248 imageBufferSize.scale(imageBufferScale.width(), imageBufferScale.height());
249
250 auto buffer = ImageBuffer::createCompatibleBuffer(expandedIntSize(imageBufferSize.size()), 1, DestinationColorSpace::SRGB(), context);
251 if (!buffer) // Failed to allocate buffer.
252 return;
253 drawForContainer(buffer->context(), containerSize, containerZoom, initialFragmentURL, imageBufferSize, zoomedContainerRect);
254 if (context.drawLuminanceMask())
255 buffer->convertToLuminanceMask();
256
257 RefPtr<Image> image = ImageBuffer::sinkIntoImage(WTFMove(buffer), PreserveResolution::Yes);
258 if (!image)
259 return;
260
261 // Adjust the source rect and transform due to the image buffer's scaling.
262 FloatRect scaledSrcRect = srcRect;
263 scaledSrcRect.scale(imageBufferScale.width(), imageBufferScale.height());
264 AffineTransform unscaledPatternTransform(patternTransform);
265 unscaledPatternTransform.scale(1 / imageBufferScale.width(), 1 / imageBufferScale.height());
266
267 context.setDrawLuminanceMask(false);
268 image->drawPattern(context, dstRect, scaledSrcRect, unscaledPatternTransform, phase, spacing, options);
269}

271ImageDrawResult SVGImage::draw(GraphicsContext& context, const FloatRect& dstRect, const FloatRect& srcRect, const ImagePaintingOptions& options)
272{
273 if (!m_page)
274 return ImageDrawResult::DidNothing;
275
276 auto view = makeRefPtr(frameView());
277 ASSERT(view);
278
279 GraphicsContextStateSaver stateSaver(context);
280 context.setCompositeOperation(options.compositeOperator(), options.blendMode());
281 context.clip(enclosingIntRect(dstRect));
282
283 float alpha = context.alpha();
284 bool compositingRequiresTransparencyLayer = options.compositeOperator() != CompositeOperator::SourceOver || options.blendMode() != BlendMode::Normal || alpha < 1;
285 if (compositingRequiresTransparencyLayer) {
286 context.beginTransparencyLayer(alpha);
287 context.setCompositeOperation(CompositeOperator::SourceOver, BlendMode::Normal);
288 }
289
290 FloatSize scale(dstRect.size() / srcRect.size());
291
292 // We can only draw the entire frame, clipped to the rect we want. So compute where the top left
293 // of the image would be if we were drawing without clipping, and translate accordingly.
294 FloatSize topLeftOffset(srcRect.location().x() * scale.width(), srcRect.location().y() * scale.height());
295 FloatPoint destOffset = dstRect.location() - topLeftOffset;
296
297 context.translate(destOffset);
298 context.scale(scale);
299
300 view->resize(containerSize());
301
302 {
303 ScriptDisallowedScope::DisableAssertionsInScope disabledScope;
304 if (view->needsLayout())
305 view->layoutContext().layout();
306 }
307
308#if PLATFORM(MAC)
309 LocalDefaultSystemAppearance localAppearance(view->useDarkAppearance());
310#endif
311
312 view->paint(context, intersection(context.clipBounds(), enclosingIntRect(srcRect)));
313
314 if (compositingRequiresTransparencyLayer)
315 context.endTransparencyLayer();
316
317 stateSaver.restore();
318
319 if (imageObserver())
320 imageObserver()->didDraw(*this);
321
322 return ImageDrawResult::DidDraw;
323}

356void SVGImage::computeIntrinsicDimensions(Length& intrinsicWidth, Length& intrinsicHeight, FloatSize& intrinsicRatio)
357{
358 auto rootElement = this->rootElement();
359 if (!rootElement)
360 return;
361
362 intrinsicWidth = rootElement->intrinsicWidth();
363 intrinsicHeight = rootElement->intrinsicHeight();
364 if (rootElement->preserveAspectRatio().align() == SVGPreserveAspectRatioValue::SVG_PRESERVEASPECTRATIO_NONE)
365 return;
366
367 intrinsicRatio = rootElement->viewBox().size();
368 if (intrinsicRatio.isEmpty() && intrinsicWidth.isFixed() && intrinsicHeight.isFixed())
369 intrinsicRatio = FloatSize(floatValueForLength(intrinsicWidth, 0), floatValueForLength(intrinsicHeight, 0));
370}

386{
387 auto rootElement = this->rootElement();

389 return;
390 rootElement->unpauseAnimations();
391 rootElement->setCurrentTime(0);
392}

416void SVGImage::reportApproximateMemoryCost() const
417{
418 auto document = makeRefPtr(m_page->mainFrame().document());
419 size_t decodedImageMemoryCost = 0;
420
421 for (RefPtr<Node> node = document; node; node = NodeTraversal::next(*node))
422 decodedImageMemoryCost += node->approximateMemoryCost();
423
424 JSC::VM& vm = commonVM();
425 JSC::JSLockHolder lock(vm);
426 // FIXME: Adopt reportExtraMemoryVisited, and switch to reportExtraMemoryAllocated.
427 // https://bugs.webkit.org/show_bug.cgi?id=142595
428 vm.heap.deprecatedReportExtraMemory(decodedImageMemoryCost + data()->size());
429}

431EncodedDataStatus SVGImage::dataChanged(bool allDataReceived)
432{
433 // Don't do anything; it is an empty image.
434 if (!data()->size())
435 return EncodedDataStatus::Complete;
436
437 if (allDataReceived) {
438 auto pageConfiguration = pageConfigurationWithEmptyClients(PAL::SessionID::defaultSessionID());
439 m_chromeClient = makeUnique<SVGImageChromeClient>(this);
440 pageConfiguration.chromeClient = m_chromeClient.get();
441
442 // FIXME: If this SVG ends up loading itself, we might leak the world.
443 // The Cache code does not know about CachedImages holding Frames and
444 // won't know to break the cycle.
445 // This will become an issue when SVGImage will be able to load other
446 // SVGImage objects, but we're safe now, because SVGImage can only be
447 // loaded by a top-level document.
448 m_page = makeUnique<Page>(WTFMove(pageConfiguration));
449#if ENABLE(VIDEO)
450 m_page->settings().setMediaEnabled(false);
451#endif
452 m_page->settings().setScriptEnabled(false);
453 m_page->settings().setPluginsEnabled(false);
454 m_page->settings().setAcceleratedCompositingEnabled(false);
455 m_page->settings().setShouldAllowUserInstalledFonts(false);
456
457 Frame& frame = m_page->mainFrame();
458 frame.setView(FrameView::create(frame));
459 frame.init();
460 FrameLoader& loader = frame.loader();
461 loader.forceSandboxFlags(SandboxAll);
462
463 frame.view()->setCanHaveScrollbars(false); // SVG Images will always synthesize a viewBox, if it's not available, and thus never see scrollbars.
464 frame.view()->setTransparent(true); // SVG Images are transparent.
465
466 ASSERT(loader.activeDocumentLoader()); // DocumentLoader should have been created by frame->init().
467 loader.activeDocumentLoader()->writer().setMIMEType("image/svg+xml");
468 loader.activeDocumentLoader()->writer().begin(URL()); // create the empty document
469 data()->forEachSegment([&](auto& segment) {
470 loader.activeDocumentLoader()->writer().addData(segment.data(), segment.size());
471 });
472 loader.activeDocumentLoader()->writer().end();
473
474 frame.document()->updateLayoutIgnorePendingStylesheets();
475
476 // Set the intrinsic size before a container size is available.
477 m_intrinsicSize = containerSize();
478 reportApproximateMemoryCost();
479 }
480
481 return m_page ? EncodedDataStatus::Complete : EncodedDataStatus::Unknown;
482}
