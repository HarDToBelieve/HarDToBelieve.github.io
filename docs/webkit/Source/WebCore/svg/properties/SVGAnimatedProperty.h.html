26#pragma once
27
28#include "SVGAttributeAnimator.h"
29#include "SVGPropertyOwner.h"
30#include <wtf/RefCounted.h>
31#include <wtf/WeakHashSet.h>
32#include <wtf/text/WTFString.h>
33
34namespace WebCore {
35
36class SVGElement;
37
38class SVGAnimatedProperty : public RefCounted<SVGAnimatedProperty>, public SVGPropertyOwner {
39public:
40 virtual ~SVGAnimatedProperty() = default;
41
42 // Manage the relationship with the owner.
43 bool isAttached() const { return m_contextElement; }
44 void detach() { m_contextElement = nullptr; }
45 SVGElement* contextElement() const { return m_contextElement; }
46
47 virtual String baseValAsString() const { return emptyString(); }
48 virtual String animValAsString() const { return emptyString(); }
49
50 // Control the synchronization between the attribute and its reflection in baseVal.
51 virtual bool isDirty() const { return false; }
52 virtual void setDirty() { }
53 virtual std::optional<String> synchronize() { return std::nullopt; }
54
55 // Control the animation life cycle.
56 bool isAnimating() const { return m_animators.computeSize(); }
57 virtual void startAnimation(SVGAttributeAnimator& animator) { m_animators.add(animator); }
58 virtual void stopAnimation(SVGAttributeAnimator& animator) { m_animators.remove(animator); }
59
60 // Attach/Detach the animVal of the traget element's property by the instance element's property.
61 virtual void instanceStartAnimation(SVGAttributeAnimator& animator, SVGAnimatedProperty&) { startAnimation(animator); }
62 virtual void instanceStopAnimation(SVGAttributeAnimator& animator) { stopAnimation(animator); }
63
64protected:
65 SVGAnimatedProperty(SVGElement* contextElement)
66 : m_contextElement(contextElement)
67 {
68 }
69
70 SVGPropertyOwner* owner() const override;
71 void commitPropertyChange(SVGProperty*) override;
72
73 SVGElement* m_contextElement { nullptr };
74 WeakHashSet<SVGAttributeAnimator> m_animators;
75};
76
77} // namespace WebCore
