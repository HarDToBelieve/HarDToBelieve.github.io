63class SVGAnimatedOrientType : public SVGAnimatedEnumeration {
64public:
65 using SVGAnimatedEnumeration::SVGAnimatedEnumeration;
66
67 static Ref<SVGAnimatedOrientType> create(SVGElement* contextElement, SVGMarkerOrientType baseValue)
68 {
69 return SVGAnimatedEnumeration::create<SVGMarkerOrientType, SVGAnimatedOrientType>(contextElement, baseValue);
70 }
71};
72
73class SVGAnimatedPathSegList : public SVGAnimatedPropertyList<SVGPathSegList> {
74 using Base = SVGAnimatedPropertyList<SVGPathSegList>;
75 using Base::Base;
76
77public:
78 static Ref<SVGAnimatedPathSegList> create(SVGElement* contextElement)
79 {
80 return adoptRef(*new SVGAnimatedPathSegList(contextElement));
81 }
82
83 SVGPathByteStream& currentPathByteStream()
84 {
85 return isAnimating() ? animVal()->pathByteStream() : baseVal()->pathByteStream();
86 }
87
88 Path currentPath()
89 {
90 return isAnimating() ? animVal()->path() : baseVal()->path();
91 }
92
93 size_t approximateMemoryCost() const
94 {
95 if (isAnimating())
96 return baseVal()->approximateMemoryCost() + animVal()->approximateMemoryCost();
97 return baseVal()->approximateMemoryCost();
98 }
99};
