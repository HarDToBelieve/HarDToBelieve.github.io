35Color SVGAnimationColorFunction::colorFromString(SVGElement* targetElement, const String& string)
36{
37 static MainThreadNeverDestroyed<const AtomString> currentColor("currentColor", AtomString::ConstructFromLiteral);
38
39 if (string != currentColor.get())
40 return SVGPropertyTraits<Color>::fromString(string);
41
42 if (auto* renderer = targetElement->renderer())
43 return renderer->style().visitedDependentColor(CSSPropertyColor);
44
45 return { };
46}

48std::optional<float> SVGAnimationIntegerFunction::calculateDistance(SVGElement*, const String& from, const String& to) const
49{
50 return std::abs(parseInteger<int>(to).value_or(0) - parseInteger<int>(from).value_or(0));
51}
