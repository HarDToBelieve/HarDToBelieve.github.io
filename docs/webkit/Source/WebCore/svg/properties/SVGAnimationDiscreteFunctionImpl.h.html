31namespace WebCore {
32
33class SVGAnimationBooleanFunction : public SVGAnimationDiscreteFunction<bool> {
34public:
35 using Base = SVGAnimationDiscreteFunction<bool>;
36 using Base::Base;
37
38 void setFromAndToValues(SVGElement*, const String& from, const String& to) override
39 {
40 m_from = SVGPropertyTraits<bool>::fromString(from);
41 m_to = SVGPropertyTraits<bool>::fromString(to);
42 }
43};
44
45template<typename EnumType>
46class SVGAnimationEnumerationFunction : public SVGAnimationDiscreteFunction<EnumType> {
47 using Base = SVGAnimationDiscreteFunction<EnumType>;
48 using Base::m_from;
49 using Base::m_to;
50
51public:
52 using Base::Base;
53
54 void setFromAndToValues(SVGElement*, const String& from, const String& to) override
55 {
56 m_from = SVGPropertyTraits<EnumType>::fromString(from);
57 m_to = SVGPropertyTraits<EnumType>::fromString(to);
58 }
59};
60
61class SVGAnimationOrientTypeFunction : public SVGAnimationDiscreteFunction<SVGMarkerOrientType> {
62public:
63 using Base = SVGAnimationDiscreteFunction<SVGMarkerOrientType>;
64 using Base::Base;
65
66 void setFromAndToValues(SVGElement*, const String&, const String&) override
67 {
68 // Values will be set by SVGAnimatedAngleOrientAnimator.
69 ASSERT_NOT_REACHED();
70 }
71
72private:
73 friend class SVGAnimatedAngleOrientAnimator;
74};

76class SVGAnimationPreserveAspectRatioFunction : public SVGAnimationDiscreteFunction<SVGPreserveAspectRatioValue> {
77public:
78 using Base = SVGAnimationDiscreteFunction<SVGPreserveAspectRatioValue>;

80

85 }
86};

88class SVGAnimationStringFunction : public SVGAnimationDiscreteFunction<String> {

90 using Base = SVGAnimationDiscreteFunction<String>;
91 using Base::Base;
92
93 void setFromAndToValues(SVGElement*, const String& from, const String& to) override

95 m_from = from;
96 m_to = to;
97 }
98};
