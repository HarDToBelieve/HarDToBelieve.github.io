26#pragma once
27
28#include "SVGProperty.h"
29
30namespace WebCore {
31
32template<typename PropertyType>
33class SVGSharedPrimitiveProperty : public SVGProperty {
34public:
35 using ValueType = PropertyType;
36
37 static Ref<SVGSharedPrimitiveProperty> create(const PropertyType& value = SVGPropertyTraits<PropertyType>::initialValue())
38 {
39 return adoptRef(*new SVGSharedPrimitiveProperty(value));
40 }
41
42 // Getter/Setter for the value.
43 const PropertyType& value() const { return m_value; }
44 PropertyType& value() { return m_value; }
45 void setValue(const PropertyType& value) { m_value = value; }
46
47 String valueAsString() const override
48 {
49 return SVGPropertyTraits<PropertyType>::toString(m_value);
50 }
51
52private:
53 SVGSharedPrimitiveProperty(const PropertyType& value)
54 : m_value(value)
55 {
56 }
57
58 PropertyType m_value;
59};
