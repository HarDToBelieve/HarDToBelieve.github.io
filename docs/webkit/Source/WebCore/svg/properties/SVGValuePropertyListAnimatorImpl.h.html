28#include "SVGAnimationFunction.h"
29#include "SVGValuePropertyListAnimator.h"

31namespace WebCore {
32
33class SVGLengthListAnimator final : public SVGValuePropertyListAnimator<SVGLengthList, SVGAnimationLengthListFunction> {
34 using Base = SVGValuePropertyListAnimator<SVGLengthList, SVGAnimationLengthListFunction>;
35 using Base::Base;
36 using Base::m_attributeName;
37 using Base::m_list;
38
39public:
40 static auto create(const QualifiedName& attributeName, Ref<SVGProperty>&& property, AnimationMode animationMode, CalcMode calcMode, bool isAccumulated, bool isAdditive)
41 {
42 return adoptRef(*new SVGLengthListAnimator(attributeName, WTFMove(property), animationMode, calcMode, isAccumulated, isAdditive, SVGLengthMode::Other));
43 }
44
45 void start(SVGElement* targetElement) override
46 {
47 String baseValue = computeCSSPropertyValue(targetElement, cssPropertyID(m_attributeName.localName()));
48 if (!m_list->parse(baseValue))
49 m_list->clearItems();
50 }
51};
