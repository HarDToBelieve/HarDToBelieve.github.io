27
28#include <JavaScriptCore/JSCJSValueInlines.h>
29#include <JavaScriptCore/JSObject.h>
30#include <JavaScriptCore/Weak.h>
31#include <wtf/RefCounted.h>
32#include <wtf/RefPtr.h>
33
34namespace WebCore {
35
36class GCObservation final : public RefCounted<GCObservation> {
37public:
38 template<typename... Args> static Ref<GCObservation> create(Args&&... args)
39 {
40 return adoptRef(*new GCObservation(std::forward<Args>(args)...));
41 }
42
43 bool wasCollected() const { return !m_observedValue; }
44
45private:
46 explicit GCObservation(JSC::JSObject*);
47
48 mutable JSC::Weak<JSC::JSObject> m_observedValue;
49};
50
51} // namespace WebCore
