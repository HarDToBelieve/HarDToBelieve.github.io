76ExceptionOr<RefPtr<Range>> Internals::rangeForDictionaryLookupAtLocation(int x, int y)
77{
78 auto* document = contextDocument();
79 if (!document || !document->frame())
80 return Exception { InvalidAccessError };
81
82 document->updateLayoutIgnorePendingStylesheets();
83
84 constexpr OptionSet<HitTestRequest::Type> hitType { HitTestRequest::Type::ReadOnly, HitTestRequest::Type::Active, HitTestRequest::Type::DisallowUserAgentShadowContent, HitTestRequest::Type::AllowChildFrameContent };
85 auto result = document->frame()->mainFrame().eventHandler().hitTestResultAtPoint(IntPoint(x, y), hitType);
86 auto range = DictionaryLookup::rangeAtHitTestResult(result);
87 if (!range)
88 return nullptr;
89
90 return RefPtr<Range> { createLiveRange(std::get<SimpleRange>(*range)) };
91}

95#if ENABLE(VIDEO)
96double Internals::privatePlayerVolume(const HTMLMediaElement& element)
97{
98 auto corePlayer = element.player();
99 if (!corePlayer)
100 return 0;
101 auto player = corePlayer->objCAVFoundationAVPlayer();
102 if (!player)
103 return 0;
104 return [player volume];
105}
106
107bool Internals::privatePlayerMuted(const HTMLMediaElement& element)
108{
109 auto corePlayer = element.player();
110 if (!corePlayer)
111 return false;
112 auto player = corePlayer->objCAVFoundationAVPlayer();
113 if (!player)
114 return false;
115 return [player isMuted];
116}

119String Internals::encodedPreferenceValue(const String& domain, const String& key)
120{
121 auto userDefaults = adoptNS([[NSUserDefaults alloc] initWithSuiteName:domain]);
122 id value = [userDefaults objectForKey:key];
123 auto data = retainPtr([NSKeyedArchiver archivedDataWithRootObject:value requiringSecureCoding:YES error:nullptr]);
124 return [data base64EncodedStringWithOptions:0];
125}

141bool Internals::hasSandboxIOKitOpenAccessToClass(const String& process, const String& ioKitClass)
142{
143 UNUSED_PARAM(process); // TODO: add support for getting PID of other WebKit processes.
144 pid_t pid = getpid();
145
146 return !sandbox_check(pid, "iokit-open", static_cast<enum sandbox_filter_type>(SANDBOX_FILTER_IOKIT_CONNECTION | SANDBOX_CHECK_NO_REPORT), ioKitClass.utf8().data());
147}
