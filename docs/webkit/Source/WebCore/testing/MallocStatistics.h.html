26#pragma once
27
28#include <wtf/FastMalloc.h>
29#include <wtf/RefCounted.h>
30
31namespace WebCore {
32
33class MallocStatistics : public RefCounted<MallocStatistics> {
34public:
35 static Ref<MallocStatistics> create() { return adoptRef(*new MallocStatistics); }
36
37 size_t reservedVMBytes() const { return m_stats.reservedVMBytes; }
38 size_t committedVMBytes() const { return m_stats.committedVMBytes; }
39 size_t freeListBytes() const { return m_stats.freeListBytes; }
40
41private:
42 MallocStatistics()
43 {
44 m_stats = WTF::fastMallocStatistics();
45 }
46
47 WTF::FastMallocStatistics m_stats;
48};
49
50} // namespace WebCore
