66void MockCDMFactory::addKeysToSessionWithID(const String& id, Vector<Ref<SharedBuffer>>&& keys)
67{
68 auto addResult = m_sessions.add(id, WTFMove(keys));
69 if (addResult.isNewEntry)
70 return;
71
72 auto& value = addResult.iterator->value;
73 for (auto& key : keys)
74 value.append(WTFMove(key));
75}
76
77Vector<Ref<SharedBuffer>> MockCDMFactory::removeKeysFromSessionWithID(const String& id)
78{
79 auto it = m_sessions.find(id);
80 if (it == m_sessions.end())
81 return { };
82
83 return WTFMove(it->value);
84}
85
86const Vector<Ref<SharedBuffer>>* MockCDMFactory::keysForSessionWithID(const String& id) const
87{
88 auto it = m_sessions.find(id);
89 if (it == m_sessions.end())
90 return nullptr;
91 return &it->value;
92}

130bool MockCDM::supportsConfiguration(const MediaKeySystemConfiguration& configuration) const
131{
132 auto capabilityHasSupportedEncryptionScheme = [&] (auto& capability) {
133 if (capability.encryptionScheme)
134 return m_factory->supportedEncryptionSchemes().contains(capability.encryptionScheme.value());
135 return true;
136 };
137
138 if (!configuration.audioCapabilities.isEmpty() && !anyOf(configuration.audioCapabilities, capabilityHasSupportedEncryptionScheme))
139 return false;
140
141 if (!configuration.videoCapabilities.isEmpty() && !anyOf(configuration.videoCapabilities, capabilityHasSupportedEncryptionScheme))
142 return false;
143
144 return true;
145
146}

214RefPtr<SharedBuffer> MockCDM::sanitizeResponse(const SharedBuffer& response) const
215{
216 if (!charactersAreAllASCII(response.data(), response.size()))
217 return nullptr;
218
219 Vector<String> responseArray = String(response.data(), response.size()).split(' ');
220
221 if (!responseArray.contains(String("valid-response"_s)))
222 return nullptr;
223
224 return response.copy();

239void MockCDMInstance::initializeWithConfiguration(const MediaKeySystemConfiguration& configuration, AllowDistinctiveIdentifiers distinctiveIdentifiers, AllowPersistentState persistentState, SuccessCallback&& callback)
240{
241 auto initialize = [&] {
242 if (!m_cdm || !m_cdm->supportsConfiguration(configuration))
243 return Failed;
244
245 MockCDMFactory* factory = m_cdm ? m_cdm->factory() : nullptr;
246 if (!factory)
247 return Failed;
248
249 bool distinctiveIdentifiersAllowed = (distinctiveIdentifiers == AllowDistinctiveIdentifiers::Yes);
250
251 if (m_distinctiveIdentifiersAllowed != distinctiveIdentifiersAllowed) {
252 if (!distinctiveIdentifiersAllowed && factory->distinctiveIdentifiersRequirement() == MediaKeysRequirement::Required)
253 return Failed;
254
255 m_distinctiveIdentifiersAllowed = distinctiveIdentifiersAllowed;
256 }
257
258 bool persistentStateAllowed = (persistentState == AllowPersistentState::Yes);
259
260 if (m_persistentStateAllowed != persistentStateAllowed) {
261 if (!persistentStateAllowed && factory->persistentStateRequirement() == MediaKeysRequirement::Required)
262 return Failed;
263
264 m_persistentStateAllowed = persistentStateAllowed;
265 }
266 return Succeeded;
267 };
268
269 callback(initialize());
270}

300void MockCDMInstanceSession::requestLicense(LicenseType licenseType, const AtomString& initDataType, Ref<SharedBuffer>&& initData, LicenseCallback&& callback)
301{
302 MockCDMFactory* factory = m_instance ? m_instance->factory() : nullptr;
303 if (!factory) {
304 callback(SharedBuffer::create(), emptyAtom(), false, SuccessValue::Failed);
305 return;
306 }
307
308 if (!factory->supportedSessionTypes().contains(licenseType) || !factory->supportedDataTypes().contains(initDataType)) {
309 callback(SharedBuffer::create(), emptyString(), false, SuccessValue::Failed);
310 return;
311 }
312
313 auto keyIDs = InitDataRegistry::shared().extractKeyIDs(initDataType, initData);
314 if (!keyIDs || keyIDs.value().isEmpty()) {
315 callback(SharedBuffer::create(), emptyString(), false, SuccessValue::Failed);
316 return;
317 }
318
319 String sessionID = createCanonicalUUIDString();
320 factory->addKeysToSessionWithID(sessionID, WTFMove(keyIDs.value()));
321
322 CString license { "license" };
323 callback(SharedBuffer::create(license.data(), license.length()), sessionID, false, SuccessValue::Succeeded);
324}

326void MockCDMInstanceSession::updateLicense(const String& sessionID, LicenseType, Ref<SharedBuffer>&& response, LicenseUpdateCallback&& callback)
327{
328 MockCDMFactory* factory = m_instance ? m_instance->factory() : nullptr;
329 if (!factory) {
330 callback(false, std::nullopt, std::nullopt, std::nullopt, SuccessValue::Failed);
331 return;
332 }
333
334 Vector<String> responseVector = String(response->data(), response->size()).split(' ');
335
336 if (responseVector.contains(String("invalid-format"_s))) {
337 callback(false, std::nullopt, std::nullopt, std::nullopt, SuccessValue::Failed);
338 return;
339 }
340
341 std::optional<KeyStatusVector> changedKeys;
342 if (responseVector.contains(String("keys-changed"_s))) {
343 const auto* keys = factory->keysForSessionWithID(sessionID);
344 if (keys) {
345 KeyStatusVector keyStatusVector;
346 keyStatusVector.reserveInitialCapacity(keys->size());
347 for (auto& key : *keys)
348 keyStatusVector.uncheckedAppend({ key.copyRef(), KeyStatus::Usable });
349
350 changedKeys = WTFMove(keyStatusVector);
351 }
352 }
353
354 // FIXME: Session closure, expiration and message handling should be implemented
355 // once the relevant algorithms are supported.
356
357 callback(false, WTFMove(changedKeys), std::nullopt, std::nullopt, SuccessValue::Succeeded);
358}

360void MockCDMInstanceSession::loadSession(LicenseType, const String&, const String&, LoadSessionCallback&& callback)
361{
362 MockCDMFactory* factory = m_instance ? m_instance->factory() : nullptr;
363 if (!factory) {
364 callback(std::nullopt, std::nullopt, std::nullopt, SuccessValue::Failed, SessionLoadFailure::Other);
365 return;
366 }
367
368 // FIXME: Key status and expiration handling should be implemented once the relevant algorithms are supported.
369
370 CString messageData { "session loaded" };
371 Message message { MessageType::LicenseRenewal, SharedBuffer::create(messageData.data(), messageData.length()) };
372
373 callback(std::nullopt, std::nullopt, WTFMove(message), SuccessValue::Succeeded, SessionLoadFailure::None);
374}

376void MockCDMInstanceSession::closeSession(const String& sessionID, CloseSessionCallback&& callback)
377{
378 MockCDMFactory* factory = m_instance ? m_instance->factory() : nullptr;
379 if (!factory) {
380 callback();
381 return;
382 }
383
384 factory->removeSessionWithID(sessionID);
385 callback();
386}

388void MockCDMInstanceSession::removeSessionData(const String& id, LicenseType, RemoveSessionDataCallback&& callback)
389{
390 MockCDMFactory* factory = m_instance ? m_instance->factory() : nullptr;
391 if (!factory) {
392 callback({ }, std::nullopt, SuccessValue::Failed);
393 return;
394 }
395
396 auto keys = factory->removeKeysFromSessionWithID(id);
397 KeyStatusVector keyStatusVector;
398 keyStatusVector.reserveInitialCapacity(keys.size());
399 for (auto& key : keys)
400 keyStatusVector.uncheckedAppend({ WTFMove(key), KeyStatus::Released });
401
402 CString message { "remove-message" };
403 callback(WTFMove(keyStatusVector), SharedBuffer::create(message.data(), message.length()), SuccessValue::Succeeded);
404}
