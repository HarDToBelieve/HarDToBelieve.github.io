31namespace WebCore {
32
33MockGamepad::MockGamepad(unsigned index, const String& gamepadID, const String& mapping, unsigned axisCount, unsigned buttonCount)
34 : PlatformGamepad(index)
35{
36 m_connectTime = m_lastUpdateTime = MonotonicTime::now();
37 updateDetails(gamepadID, mapping, axisCount, buttonCount);
38}
39
40void MockGamepad::updateDetails(const String& gamepadID, const String& mapping, unsigned axisCount, unsigned buttonCount)
41{
42 m_id = gamepadID;
43 m_mapping = mapping;
44 m_axisValues.clear();
45 for (size_t i = 0; i < axisCount; ++i)
46 m_axisValues.append({ });
47 m_buttonValues.clear();
48 for (size_t i = 0; i < buttonCount; ++i)
49 m_buttonValues.append({ });
50 m_lastUpdateTime = MonotonicTime::now();
51}
52
53bool MockGamepad::setAxisValue(unsigned index, double value)
54{
55 if (index >= m_axisValues.size()) {
56 LOG_ERROR("MockGamepad (%u): Attempt to set value on axis %u which doesn't exist", m_index, index);
57 return false;
58 }
59
60 m_axisValues[index].setValue(value);
61 m_lastUpdateTime = MonotonicTime::now();
62 return true;
63}
64
65bool MockGamepad::setButtonValue(unsigned index, double value)
66{
67 if (index >= m_buttonValues.size()) {
68 LOG_ERROR("MockGamepad (%u): Attempt to set value on button %u which doesn't exist", m_index, index);
69 return false;
70 }
71
72 m_buttonValues[index].setValue(value);
73 m_lastUpdateTime = MonotonicTime::now();
74 return true;
75}
