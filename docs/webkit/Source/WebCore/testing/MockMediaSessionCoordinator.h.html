34namespace WebCore {
35
36class ScriptExecutionContext;
37class StringCallback;
38
39class MockMediaSessionCoordinator : public MediaSessionCoordinatorPrivate, public CanMakeWeakPtr<MockMediaSessionCoordinator> {
40 WTF_MAKE_FAST_ALLOCATED;
41public:
42 static Ref<MockMediaSessionCoordinator> create(ScriptExecutionContext&, RefPtr<StringCallback>&&);
43
44 void setCommandsShouldFail(bool shouldFail) { m_failCommands = shouldFail; }
45
46private:
47 MockMediaSessionCoordinator(ScriptExecutionContext&, RefPtr<StringCallback>&&);
48
49 String identifier() const final { return "Mock Coordinator"; }
50
51 void join(CompletionHandler<void(std::optional<Exception>&&)>&&) final;
52 void leave() final;
53
54 void seekTo(double, CompletionHandler<void(std::optional<Exception>&&)>&&) final;
55 void play(CompletionHandler<void(std::optional<Exception>&&)>&&) final;
56 void pause(CompletionHandler<void(std::optional<Exception>&&)>&&) final;
57 void setTrack(const String&, CompletionHandler<void(std::optional<Exception>&&)>&&) final;
58
59 void positionStateChanged(const std::optional<MediaPositionState>&) final;
60 void readyStateChanged(MediaSessionReadyState) final;
61 void playbackStateChanged(MediaSessionPlaybackState) final;
62 void trackIdentifierChanged(const String&) final;
63
64 const char* logClassName() const { return "MockMediaSessionCoordinator"; }
65 WTFLogChannel& logChannel() const;
66
67 std::optional<Exception> result() const;
68
69 Ref<ScriptExecutionContext> m_context;
70 RefPtr<StringCallback> m_stateChangeListener;
71 bool m_failCommands { false };
72};

74}
