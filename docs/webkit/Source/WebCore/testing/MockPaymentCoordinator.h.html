51class MockPaymentCoordinator final : public PaymentCoordinatorClient {
52public:
53 explicit MockPaymentCoordinator(Page&);
54
55 void setCanMakePayments(bool canMakePayments) { m_canMakePayments = canMakePayments; }
56 void setCanMakePaymentsWithActiveCard(bool canMakePaymentsWithActiveCard) { m_canMakePaymentsWithActiveCard = canMakePaymentsWithActiveCard; }
57 void setShippingAddress(MockPaymentAddress&& shippingAddress) { m_shippingAddress = WTFMove(shippingAddress); }
58 void changeShippingOption(String&& shippingOption);
59 void changePaymentMethod(ApplePayPaymentMethod&&);
60#if ENABLE(APPLE_PAY_COUPON_CODE)
61 void changeCouponCode(String&& couponCode);
62#endif
63 void acceptPayment();
64 void cancelPayment();
65
66 void addSetupFeature(ApplePaySetupFeatureState, ApplePaySetupFeatureType, bool supportsInstallments);
67 const ApplePaySetupConfiguration& setupConfiguration() const { return m_setupConfiguration; }
68
69 const ApplePayLineItem& total() const { return m_total; }
70 const Vector<ApplePayLineItem>& lineItems() const { return m_lineItems; }
71 const Vector<MockPaymentError>& errors() const { return m_errors; }
72 const Vector<ApplePayShippingMethod>& shippingMethods() const { return m_shippingMethods; }
73 const MockPaymentContactFields& requiredBillingContactFields() const { return m_requiredBillingContactFields; }
74 const MockPaymentContactFields& requiredShippingContactFields() const { return m_requiredShippingContactFields; }
75
76 bool supportsUnrestrictedApplePay() const final { return m_supportsUnrestrictedApplePay; }
77 void setSupportsUnrestrictedApplePay(bool supports) { m_supportsUnrestrictedApplePay = supports; }
78
79#if ENABLE(APPLE_PAY_INSTALLMENTS)
80 ApplePayInstallmentConfiguration installmentConfiguration() const { return m_installmentConfiguration; }
81#endif
82
83#if ENABLE(APPLE_PAY_COUPON_CODE)
84 std::optional<bool> supportsCouponCode() const { return m_supportsCouponCode; }
85 const String& couponCode() const { return m_couponCode; }
86#endif
87
88#if ENABLE(APPLE_PAY_SHIPPING_CONTACT_EDITING_MODE)
89 std::optional<ApplePayShippingContactEditingMode> shippingContactEditingMode() const { return m_shippingContactEditingMode; }
90#endif
91
92 void ref() const { }
93 void deref() const { }
94
95private:
96 std::optional<String> validatedPaymentNetwork(const String&) final;
97 bool canMakePayments() final;
98 void canMakePaymentsWithActiveCard(const String&, const String&, CompletionHandler<void(bool)>&&) final;
99 void openPaymentSetup(const String&, const String&, CompletionHandler<void(bool)>&&) final;
100 bool showPaymentUI(const URL&, const Vector<URL>&, const ApplePaySessionPaymentRequest&) final;
101 void completeMerchantValidation(const PaymentMerchantSession&) final;
102 void completeShippingMethodSelection(std::optional<ApplePayShippingMethodUpdate>&&) final;
103 void completeShippingContactSelection(std::optional<ApplePayShippingContactUpdate>&&) final;
104 void completePaymentMethodSelection(std::optional<ApplePayPaymentMethodUpdate>&&) final;
105#if ENABLE(APPLE_PAY_COUPON_CODE)
106 void completeCouponCodeChange(std::optional<ApplePayCouponCodeUpdate>&&) final;
107#endif
108 void completePaymentSession(std::optional<PaymentAuthorizationResult>&&) final;
109 void abortPaymentSession() final;
110 void cancelPaymentSession() final;
111 void paymentCoordinatorDestroyed() final;
112
113 bool isMockPaymentCoordinator() const final { return true; }
114
115 void getSetupFeatures(const ApplePaySetupConfiguration&, const URL&, CompletionHandler<void(Vector<Ref<ApplePaySetupFeature>>&&)>&&) final;
116 void beginApplePaySetup(const ApplePaySetupConfiguration&, const URL&, Vector<RefPtr<ApplePaySetupFeature>>&&, CompletionHandler<void(bool)>&&) final;
117
118 void dispatchIfShowing(Function<void()>&&);
119

121 bool m_canMakePayments { true };
122 bool m_canMakePaymentsWithActiveCard { true };
123 ApplePayPaymentContact m_shippingAddress;
124 ApplePayLineItem m_total;

126 Vector<MockPaymentError> m_errors;
127 Vector<ApplePayShippingMethod> m_shippingMethods;

129 MockPaymentContactFields m_requiredBillingContactFields;

131 bool m_supportsUnrestrictedApplePay { false };
132#if ENABLE(APPLE_PAY_INSTALLMENTS)
133 ApplePayInstallmentConfiguration m_installmentConfiguration;
134#endif
135#if ENABLE(APPLE_PAY_COUPON_CODE)
136 std::optional<bool> m_supportsCouponCode;
137 String m_couponCode;
138#endif
139#if ENABLE(APPLE_PAY_SHIPPING_CONTACT_EDITING_MODE)
140 std::optional<ApplePayShippingContactEditingMode> m_shippingContactEditingMode;
141#endif
142 ApplePaySetupConfiguration m_setupConfiguration;
143 Vector<Ref<ApplePaySetupFeature>> m_setupFeatures;

146 MockPaymentCoordinatorAdditions_members
