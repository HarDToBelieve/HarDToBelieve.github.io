45void ServiceWorkerInternals::setOnline(bool isOnline)
46{
47 callOnMainThread([identifier = m_identifier, isOnline] () {
48 if (auto* proxy = SWContextManager::singleton().workerByID(identifier))
49 proxy->notifyNetworkStateChange(isOnline);
50 });
51}
52
53void ServiceWorkerInternals::terminate()
54{
55 callOnMainThread([identifier = m_identifier] () {
56 SWContextManager::singleton().terminateWorker(identifier, Seconds::infinity(), [] { });
57 });
58}

60void ServiceWorkerInternals::waitForFetchEventToFinish(FetchEvent& event, DOMPromiseDeferred<IDLInterface<FetchResponse>>&& promise)
61{
62 event.onResponse([promise = WTFMove(promise), event = makeRef(event)] (auto&& result) mutable {
63 if (!result.has_value()) {
64 String description;
65 if (auto& error = result.error())
66 description = error->localizedDescription();
67 promise.reject(TypeError, description);
68 return;
69 }
70 promise.resolve(WTFMove(result.value()));
71 });
72}

81Ref<FetchResponse> ServiceWorkerInternals::createOpaqueWithBlobBodyResponse(ScriptExecutionContext& context)
82{
83 auto blob = Blob::create(&context);
84 auto formData = FormData::create();
85 formData->appendBlob(blob->url());
86
87 ResourceResponse response;
88 response.setType(ResourceResponse::Type::Cors);
89 response.setTainting(ResourceResponse::Tainting::Opaque);
90 auto fetchResponse = FetchResponse::create(context, FetchBody::fromFormData(context, formData), FetchHeaders::Guard::Response, WTFMove(response));
91 fetchResponse->initializeOpaqueLoadIdentifierForTesting();
92 return fetchResponse;
93}

95Vector<String> ServiceWorkerInternals::fetchResponseHeaderList(FetchResponse& response)
96{
97 Vector<String> headerNames;
98 headerNames.reserveInitialCapacity(response.internalResponseHeaders().size());
99 for (auto keyValue : response.internalResponseHeaders())
100 headerNames.uncheckedAppend(keyValue.key);
101 return headerNames;
102}

122void ServiceWorkerInternals::lastNavigationWasAppBound(Ref<DeferredPromise>&& promise)
123{
124 ASSERT(!m_lastNavigationWasAppBoundPromise);
125 m_lastNavigationWasAppBoundPromise = WTFMove(promise);
126 callOnMainThread([identifier = m_identifier, weakThis = makeWeakPtr(this)]() mutable {
127 if (auto* proxy = SWContextManager::singleton().workerByID(identifier)) {
128 proxy->thread().runLoop().postTaskForMode([weakThis = WTFMove(weakThis), appBound = proxy->lastNavigationWasAppBound()](auto&) {
129 if (!weakThis || !weakThis->m_lastNavigationWasAppBoundPromise)
130 return;
131
132 weakThis->m_lastNavigationWasAppBoundPromise->resolve<IDLBoolean>(appBound);
133 weakThis->m_lastNavigationWasAppBoundPromise = nullptr;
134 }, WorkerRunLoop::defaultMode());
135 }
136 });
137}
