27
28#if ENABLE(WEBXR)
29
30#include "FakeXRButtonStateInit.h"
31#include "FakeXRInputSourceInit.h"
32#include "FakeXRRigidTransformInit.h"
33#include "PlatformXR.h"
34#include "XRHandedness.h"
35#include "XRTargetRayMode.h"
36#include <wtf/HashMap.h>
37#include <wtf/RefCounted.h>
38#include <wtf/Vector.h>
39
40namespace WebCore {
41
42class WebFakeXRInputController final : public RefCounted<WebFakeXRInputController> {
43public:
44 static Ref<WebFakeXRInputController> create(PlatformXR::InputSourceHandle, const FakeXRInputSourceInit&);
45
46 void setHandedness(XRHandedness handeness) { m_handeness = handeness; }
47 void setTargetRayMode(XRTargetRayMode mode) { m_targetRayMode = mode; }
48 void setProfiles(Vector<String>&& profiles) { m_profiles = WTFMove(profiles); }
49 void setGripOrigin(FakeXRRigidTransformInit gripOrigin, bool emulatedPosition = false);
50 void clearGripOrigin() { m_gripOrigin = std::nullopt; }
51 void setPointerOrigin(FakeXRRigidTransformInit pointerOrigin, bool emulatedPosition = false);
52 void disconnect();
53 void reconnect();
54 void startSelection() { m_primarySelected = true; }
55 void endSelection() { m_primarySelected = false; }
56 void simulateSelect() { m_simulateSelect = true; }
57 void setSupportedButtons(const Vector<FakeXRButtonStateInit>&);
58 void updateButtonState(const FakeXRButtonStateInit&);
59 bool isConnected() const { return m_connected; }
60
61 PlatformXR::Device::FrameData::InputSource getFrameData();
62
63private:
64 WebFakeXRInputController(PlatformXR::InputSourceHandle, const FakeXRInputSourceInit&);
65
66 struct ButtonOrPlaceholder {
67 std::optional<PlatformXR::Device::FrameData::InputSourceButton> button;
68 std::optional<Vector<float>> axes;
69 };
70 ButtonOrPlaceholder getButtonOrPlaceholder(FakeXRButtonStateInit::Type) const;
71
72 PlatformXR::InputSourceHandle m_handle { 0 };
73 XRHandedness m_handeness { XRHandedness::None };
74 XRTargetRayMode m_targetRayMode { XRTargetRayMode::Gaze };
75 Vector<String> m_profiles;
76 PlatformXR::Device::FrameData::InputSourcePose m_pointerOrigin;
77 std::optional<PlatformXR::Device::FrameData::InputSourcePose> m_gripOrigin;
78 HashMap<FakeXRButtonStateInit::Type, FakeXRButtonStateInit, WTF::IntHash<FakeXRButtonStateInit::Type>, WTF::StrongEnumHashTraits<FakeXRButtonStateInit::Type>> m_buttons;
79 bool m_connected { true };
80 bool m_primarySelected { false };
81 bool m_simulateSelect { false };
82};
83
84} // namespace WebCore
85
86#endif // ENABLE(WEBXR)
