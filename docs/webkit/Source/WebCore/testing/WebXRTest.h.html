26#pragma once
27
28#if ENABLE(WEBXR)
29
30#include "JSDOMPromiseDeferred.h"
31#include "WebFakeXRDevice.h"
32#include "XRSessionMode.h"
33#include "XRSimulateUserActivationFunction.h"
34#include <JavaScriptCore/JSCJSValue.h>
35#include <wtf/RefCounted.h>
36
37namespace WebCore {
38
39class WebXRSystem;
40
41class WebXRTest final : public RefCounted<WebXRTest> {
42public:
43 struct FakeXRDeviceInit {
44 bool supportsImmersive { false };
45 std::optional<Vector<XRSessionMode>> supportedModes;
46 Vector<FakeXRViewInit> views;
47
48 std::optional<Vector<JSC::JSValue>> supportedFeatures;
49
50 std::optional<Vector<FakeXRBoundsPoint>> boundsCoordinates;
51
52 std::optional<FakeXRRigidTransformInit> floorOrigin;
53 std::optional<FakeXRRigidTransformInit> viewerOrigin;
54 };
55
56 static Ref<WebXRTest> create(WeakPtr<WebXRSystem>&& system) { return adoptRef(*new WebXRTest(WTFMove(system))); }
57 virtual ~WebXRTest();
58
59 using WebFakeXRDevicePromise = DOMPromiseDeferred<IDLInterface<WebFakeXRDevice>>;
60 void simulateDeviceConnection(ScriptExecutionContext& state, const FakeXRDeviceInit&, WebFakeXRDevicePromise&&);
61
62 // Simulates a user activation (aka user gesture) for the current scope.
63 // The activation is only guaranteed to be valid in the provided function and only applies to WebXR
64 // Device API methods.
65 void simulateUserActivation(Document&, XRSimulateUserActivationFunction&);
66
67 // Disconnect all fake devices
68 void disconnectAllDevices(DOMPromiseDeferred<void>&&);
69
70private:
71 WebXRTest(WeakPtr<WebXRSystem>&& system)
72 : m_context(WTFMove(system)) { }
73
74 WeakPtr<WebXRSystem> m_context;
75 Vector<Ref<WebFakeXRDevice>> m_devices;
76};
77
78} // namespace WebCore
79
80#endif // ENABLE(WEBXR)
