42static RetainPtr<CFURLResponseRef> createCFURLResponseFromResponseData(CFDataRef responseData)
43{
44 NSURLResponse *response;
45 auto unarchiver = adoptNS([[NSKeyedUnarchiver alloc] initForReadingFromData:(__bridge NSData *)responseData error:nullptr]);
46 unarchiver.get().decodingFailurePolicy = NSDecodingFailurePolicyRaiseException;
47 @try {
48 response = [unarchiver decodeObjectOfClass:[NSURLResponse class] forKey:@"WebResourceResponse"]; // WebResourceResponseKey in WebResource.m
49 [unarchiver finishDecoding];
50 } @catch (NSException *exception) {
51 LOG_ERROR("Failed to decode NS(HTTP)URLResponse: %@", exception);
52 response = nil;
53 }
54
55 if (![response isKindOfClass:[NSHTTPURLResponse class]])
56 return adoptCF(CFURLResponseCreate(kCFAllocatorDefault, (__bridge CFURLRef)response.URL, (__bridge CFStringRef)response.MIMEType, response.expectedContentLength, (__bridge CFStringRef)response.textEncodingName, kCFURLCacheStorageAllowed));
57
58 NSHTTPURLResponse *httpResponse = (NSHTTPURLResponse *)response;
59
60 // NSURLResponse is not toll-free bridged to CFURLResponse.
61 RetainPtr<CFHTTPMessageRef> httpMessage = adoptCF(CFHTTPMessageCreateResponse(kCFAllocatorDefault, httpResponse.statusCode, nullptr, kCFHTTPVersion1_1));
62
63 NSDictionary *headerFields = httpResponse.allHeaderFields;
64 for (NSString *headerField in [headerFields keyEnumerator])
65 CFHTTPMessageSetHeaderFieldValue(httpMessage.get(), (__bridge CFStringRef)headerField, (__bridge CFStringRef)[headerFields objectForKey:headerField]);
66
67 return adoptCF(CFURLResponseCreateWithHTTPResponse(kCFAllocatorDefault, (__bridge CFURLRef)response.URL, httpMessage.get(), kCFURLCacheStorageAllowed));
68}

70static void convertMIMEType(CFMutableStringRef mimeType)
71{
72 // Workaround for <rdar://problem/6234318> with Dashcode 2.0
73 if (CFStringCompare(mimeType, CFSTR("application/x-javascript"), kCFCompareAnchored | kCFCompareCaseInsensitive) == kCFCompareEqualTo)
74 CFStringReplaceAll(mimeType, CFSTR("text/javascript"));
75}

77static void convertWebResourceDataToString(CFMutableDictionaryRef resource)
78{
79 CFMutableStringRef mimeType = checked_cf_cast<CFMutableStringRef>(CFDictionaryGetValue(resource, CFSTR("WebResourceMIMEType")));
80 CFStringLowercase(mimeType, CFLocaleGetSystem());
81 convertMIMEType(mimeType);
82
83 if (CFStringHasPrefix(mimeType, CFSTR("text/")) || MIMETypeRegistry::isSupportedNonImageMIMEType(mimeType)) {
84 CFStringRef textEncodingName = static_cast<CFStringRef>(CFDictionaryGetValue(resource, CFSTR("WebResourceTextEncodingName")));
85 CFStringEncoding stringEncoding;
86 if (textEncodingName && CFStringGetLength(textEncodingName))
87 stringEncoding = CFStringConvertIANACharSetNameToEncoding(textEncodingName);
88 else
89 stringEncoding = kCFStringEncodingUTF8;
90
91 CFDataRef data = static_cast<CFDataRef>(CFDictionaryGetValue(resource, CFSTR("WebResourceData")));
92 RetainPtr<CFStringRef> dataAsString = adoptCF(CFStringCreateFromExternalRepresentation(kCFAllocatorDefault, data, stringEncoding));
93 if (dataAsString)
94 CFDictionarySetValue(resource, CFSTR("WebResourceData"), dataAsString.get());
95 }
96}

98static void normalizeHTTPResponseHeaderFields(CFMutableDictionaryRef fields)
99{
100 // Normalize headers
101 if (CFDictionaryContainsKey(fields, CFSTR("Date")))
102 CFDictionarySetValue(fields, CFSTR("Date"), CFSTR("Sun, 16 Nov 2008 17:00:00 GMT"));
103 if (CFDictionaryContainsKey(fields, CFSTR("Last-Modified")))
104 CFDictionarySetValue(fields, CFSTR("Last-Modified"), CFSTR("Sun, 16 Nov 2008 16:55:00 GMT"));
105 if (CFDictionaryContainsKey(fields, CFSTR("Etag")))
106 CFDictionarySetValue(fields, CFSTR("Etag"), CFSTR("\"301925-21-45c7d72d3e780\""));
107 if (CFDictionaryContainsKey(fields, CFSTR("Server")))
108 CFDictionarySetValue(fields, CFSTR("Server"), CFSTR("Apache/2.2.9 (Unix) mod_ssl/2.2.9 OpenSSL/0.9.7l PHP/5.2.6"));
109
110 // Remove headers
111 CFDictionaryRemoveValue(fields, CFSTR("Connection"));
112 CFDictionaryRemoveValue(fields, CFSTR("Keep-Alive"));
113}

124static void normalizeWebResourceURL(CFMutableStringRef webResourceURL)
125{
126 static CFStringRef fileScheme = CFSTR("file://");
127 static CFIndex fileSchemeLength = CFStringGetLength(fileScheme);
128 if (CFStringFind(webResourceURL, fileScheme, kCFCompareAnchored).location != kCFNotFound) {
129 CFRange layoutTestsWebArchivePathRange = CFStringFind(webResourceURL, CFSTR("/LayoutTests/"), kCFCompareBackwards);
130 if (layoutTestsWebArchivePathRange.location == kCFNotFound)
131 return;
132 CFRange currentWorkingDirectoryRange = CFRangeMake(fileSchemeLength, layoutTestsWebArchivePathRange.location - fileSchemeLength);
133 CFStringReplace(webResourceURL, currentWorkingDirectoryRange, CFSTR(""));
134 return;
135 }
136
137#if USE(QUICK_LOOK)
138 static CFStringRef quickLookScheme = CFSTR("x-apple-ql-id://");
139 static CFIndex quickLookSchemeLength = CFStringGetLength(quickLookScheme);
140 if (CFStringFind(webResourceURL, quickLookScheme, kCFCompareAnchored).location == kCFNotFound)
141 return;
142
143 CFIndex replacementEnd = CFStringFind(webResourceURL, CFSTR("."), kCFCompareBackwards).location;
144 if (replacementEnd == kCFNotFound)
145 replacementEnd = CFStringGetLength(webResourceURL);
146
147 auto oldResourceURL = adoptCF(CFStringCreateCopy(kCFAllocatorDefault, webResourceURL));
148 CFStringReplace(webResourceURL, CFRangeMake(quickLookSchemeLength, replacementEnd - quickLookSchemeLength), CFSTR("resource"));
149 quickLookURLReplacements().append(std::make_pair(WTFMove(oldResourceURL), webResourceURL));
150#endif
151}

153static void convertWebResourceResponseToDictionary(CFMutableDictionaryRef propertyList)
154{
155 CFDataRef responseData = static_cast<CFDataRef>(CFDictionaryGetValue(propertyList, CFSTR("WebResourceResponse"))); // WebResourceResponseKey in WebResource.m
156 if (CFGetTypeID(responseData) != CFDataGetTypeID())
157 return;
158
159 auto response = createCFURLResponseFromResponseData(responseData);
160 if (!response)
161 return;
162
163 RetainPtr<CFMutableDictionaryRef> responseDictionary = adoptCF(CFDictionaryCreateMutable(kCFAllocatorDefault, 0, &kCFTypeDictionaryKeyCallBacks, &kCFTypeDictionaryValueCallBacks));
164
165 RetainPtr<CFMutableStringRef> urlString = adoptCF(CFStringCreateMutableCopy(kCFAllocatorDefault, 0, CFURLGetString(CFURLResponseGetURL(response.get()))));
166 normalizeWebResourceURL(urlString.get());
167 CFDictionarySetValue(responseDictionary.get(), CFSTR("URL"), urlString.get());
168
169 RetainPtr<CFMutableStringRef> mimeTypeString = adoptCF(CFStringCreateMutableCopy(kCFAllocatorDefault, 0, CFURLResponseGetMIMEType(response.get())));
170 convertMIMEType(mimeTypeString.get());
171 CFDictionarySetValue(responseDictionary.get(), CFSTR("MIMEType"), mimeTypeString.get());
172
173 CFStringRef textEncodingName = CFURLResponseGetTextEncodingName(response.get());
174 if (textEncodingName)
175 CFDictionarySetValue(responseDictionary.get(), CFSTR("textEncodingName"), textEncodingName);
176
177 SInt64 expectedContentLength = CFURLResponseGetExpectedContentLength(response.get());
178 RetainPtr<CFNumberRef> expectedContentLengthNumber = adoptCF(CFNumberCreate(kCFAllocatorDefault, kCFNumberSInt64Type, &expectedContentLength));
179 CFDictionarySetValue(responseDictionary.get(), CFSTR("expectedContentLength"), expectedContentLengthNumber.get());
180
181 if (CFHTTPMessageRef httpMessage = CFURLResponseGetHTTPResponse(response.get())) {
182 RetainPtr<CFDictionaryRef> allHeaders = adoptCF(CFHTTPMessageCopyAllHeaderFields(httpMessage));
183 RetainPtr<CFMutableDictionaryRef> allHeaderFields = adoptCF(CFDictionaryCreateMutableCopy(kCFAllocatorDefault, 0, allHeaders.get()));
184 normalizeHTTPResponseHeaderFields(allHeaderFields.get());
185 CFDictionarySetValue(responseDictionary.get(), CFSTR("allHeaderFields"), allHeaderFields.get());
186
187 CFIndex statusCode = CFHTTPMessageGetResponseStatusCode(httpMessage);
188 RetainPtr<CFNumberRef> statusCodeNumber = adoptCF(CFNumberCreate(kCFAllocatorDefault, kCFNumberCFIndexType, &statusCode));
189 CFDictionarySetValue(responseDictionary.get(), CFSTR("statusCode"), statusCodeNumber.get());
190 }
191
192 CFDictionarySetValue(propertyList, CFSTR("WebResourceResponse"), responseDictionary.get());
193}

195static CFComparisonResult compareResourceURLs(const void *val1, const void *val2, void *)
196{
197 CFStringRef url1 = static_cast<CFStringRef>(CFDictionaryGetValue(static_cast<CFDictionaryRef>(val1), CFSTR("WebResourceURL")));
198 CFStringRef url2 = static_cast<CFStringRef>(CFDictionaryGetValue(static_cast<CFDictionaryRef>(val2), CFSTR("WebResourceURL")));
199
200 return CFStringCompare(url1, url2, kCFCompareAnchored);
201}

203RetainPtr<CFStringRef> createXMLStringFromWebArchiveData(CFDataRef webArchiveData)
204{
205 CFErrorRef error = 0;
206 CFPropertyListFormat format = kCFPropertyListBinaryFormat_v1_0;
207 RetainPtr<CFMutableDictionaryRef> propertyList = adoptCF(checked_cf_cast<CFMutableDictionaryRef>(CFPropertyListCreateWithData(kCFAllocatorDefault, webArchiveData, kCFPropertyListMutableContainersAndLeaves, &format, &error)));
208
209 if (!propertyList) {
210 if (error)
211 return adoptCF(CFErrorCopyDescription(error));
212 return CFSTR("An unknown error occurred converting data to property list.");
213 }
214
215 RetainPtr<CFMutableArrayRef> resources = adoptCF(CFArrayCreateMutable(kCFAllocatorDefault, 0, &kCFTypeArrayCallBacks));
216 CFArrayAppendValue(resources.get(), propertyList.get());
217
218 while (CFArrayGetCount(resources.get())) {
219 RetainPtr<CFMutableDictionaryRef> resourcePropertyList = checked_cf_cast<CFMutableDictionaryRef>(CFArrayGetValueAtIndex(resources.get(), 0));
220 CFArrayRemoveValueAtIndex(resources.get(), 0);
221
222 CFMutableDictionaryRef mainResource = checked_cf_cast<CFMutableDictionaryRef>(CFDictionaryGetValue(resourcePropertyList.get(), CFSTR("WebMainResource")));
223 normalizeWebResourceURL(checked_cf_cast<CFMutableStringRef>(CFDictionaryGetValue(mainResource, CFSTR("WebResourceURL"))));
224 convertWebResourceDataToString(mainResource);
225
226 // Add subframeArchives to list for processing
227 CFMutableArrayRef subframeArchives = checked_cf_cast<CFMutableArrayRef>(CFDictionaryGetValue(resourcePropertyList.get(), CFSTR("WebSubframeArchives"))); // WebSubframeArchivesKey in WebArchive.m
228 if (subframeArchives)
229 CFArrayAppendArray(resources.get(), subframeArchives, CFRangeMake(0, CFArrayGetCount(subframeArchives)));
230
231 CFMutableArrayRef subresources = checked_cf_cast<CFMutableArrayRef>(CFDictionaryGetValue(resourcePropertyList.get(), CFSTR("WebSubresources"))); // WebSubresourcesKey in WebArchive.m
232 if (!subresources)
233 continue;
234
235 CFIndex subresourcesCount = CFArrayGetCount(subresources);
236 for (CFIndex i = 0; i < subresourcesCount; ++i) {
237 CFMutableDictionaryRef subresourcePropertyList = checked_cf_cast<CFMutableDictionaryRef>(CFArrayGetValueAtIndex(subresources, i));
238 normalizeWebResourceURL(checked_cf_cast<CFMutableStringRef>(CFDictionaryGetValue(subresourcePropertyList, CFSTR("WebResourceURL"))));
239 convertWebResourceResponseToDictionary(subresourcePropertyList);
240 convertWebResourceDataToString(subresourcePropertyList);
241 }
242
243 // Sort the subresources so they're always in a predictable order for the dump
244 CFArraySortValues(subresources, CFRangeMake(0, CFArrayGetCount(subresources)), compareResourceURLs, 0);
245 }
246
247 error = 0;
248
249 RetainPtr<CFDataRef> xmlData = adoptCF(CFPropertyListCreateData(kCFAllocatorDefault, propertyList.get(), kCFPropertyListXMLFormat_v1_0, 0, &error));
250
251 if (!xmlData) {
252 if (error)
253 return adoptCF(CFErrorCopyDescription(error));
254 return CFSTR("An unknown error occurred converting property list to data.");
255 }
256
257 RetainPtr<CFStringRef> xmlString = adoptCF(CFStringCreateFromExternalRepresentation(kCFAllocatorDefault, xmlData.get(), kCFStringEncodingUTF8));
258 RetainPtr<CFMutableStringRef> string = adoptCF(CFStringCreateMutableCopy(kCFAllocatorDefault, 0, xmlString.get()));
259
260 // Replace "Apple Computer" with "Apple" in the DTD declaration.
261 CFStringFindAndReplace(string.get(), CFSTR("-//Apple Computer//"), CFSTR("-//Apple//"), CFRangeMake(0, CFStringGetLength(string.get())), 0);
262
263#if USE(QUICK_LOOK)
264 for (auto& replacement : quickLookURLReplacements())
265 CFStringFindAndReplace(string.get(), replacement.first.get(), replacement.second.get(), CFRangeMake(0, CFStringGetLength(string.get())), 0);
266 quickLookURLReplacements().clear();
267#endif
268
269 return string;
270}
