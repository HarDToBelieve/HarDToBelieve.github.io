54+ (BOOL)shouldIdentifyWebViews
55{
56 static std::optional<BOOL> shouldIdentifyWebViews;
57 if (!shouldIdentifyWebViews)
58 shouldIdentifyWebViews = [[NSUserDefaults standardUserDefaults] boolForKey:@"WebKitDebugIdentifyWebViews"];
59 return *shouldIdentifyWebViews;
60}
61
62+ (void)installForWebViewIfNeeded:(PlatformView *)view kind:(NSString *)kind deprecated:(BOOL)isDeprecated
63{
64 if (![self shouldIdentifyWebViews])
65 return;
66 auto overlay = adoptNS([[WebViewVisualIdentificationOverlay alloc] initWithWebView:view kind:kind deprecated:isDeprecated]);
67 objc_setAssociatedObject(self, webViewVisualIdentificationOverlayKey, overlay.get(), OBJC_ASSOCIATION_RETAIN_NONATOMIC);
68}

70- (instancetype)initWithWebView:(PlatformView *)webView kind:(NSString *)kind deprecated:(BOOL)isDeprecated
71{
72 self = [super init];
73 if (!self)
74 return nil;
75
76 _kind = kind;
77
78#if USE(APPKIT)
79 _view = adoptNS([[NSView alloc] initWithFrame:webView.bounds]);
80 [_view setWantsLayer:YES];
81 [_view setAutoresizingMask:NSViewWidthSizable | NSViewHeightSizable];
82#else
83 _view = adoptNS([PAL::allocUIViewInstance() initWithFrame:webView.bounds]);
84 [_view setUserInteractionEnabled:NO];
85 [_view setAutoresizingMask:UIViewAutoresizingFlexibleWidth | UIViewAutoresizingFlexibleHeight];
86#endif
87 [webView addSubview:_view.get()];
88
89#if PLATFORM(MACCATALYST)
90 _kind = [_kind stringByAppendingString:@" (macCatalyst)"];
91#endif
92
93 _layer = adoptNS([[CATiledLayer alloc] init]);
94 [_layer setName:@"WebViewVisualIdentificationOverlay"];
95 [_layer setFrame:CGRectMake(0, 0, [_view bounds].size.width, [_view bounds].size.height)];
96 auto viewColor = isDeprecated ? WebCore::Color::red.colorWithAlphaByte(50) : WebCore::Color::blue.colorWithAlphaByte(32);
97 [_layer setBackgroundColor:cachedCGColor(viewColor)];
98 [_layer setZPosition:999];
99 [_layer setDelegate:self];
100 [_layer web_disableAllActions];
101 [[_view layer] addSublayer:_layer.get()];
102
103 [[_view layer] addObserver:self forKeyPath:@"bounds" options:0 context:boundsObservationContext];
104
105 return self;
106}
107
108- (void)dealloc
109{
110 [[_view layer] removeObserver:self forKeyPath:@"bounds" context:boundsObservationContext];
111
112 [super dealloc];
113}

115- (void)observeValueForKeyPath:(NSString *)keyPath ofObject:(id)object change:(NSDictionary<NSKeyValueChangeKey, id> *)change context:(void *)context
116{
117 UNUSED_PARAM(keyPath);
118 UNUSED_PARAM(object);
119 UNUSED_PARAM(change);
120 if (context == boundsObservationContext) {
121 [_layer setFrame:CGRectMake(0, 0, [_view bounds].size.width, [_view bounds].size.height)];
122 [_layer setNeedsDisplay];
123 }
124}

126static RetainPtr<CTFontRef> createIdentificationFont()
127{
128 auto matrix = CGAffineTransformIdentity;
129 return adoptCF(CTFontCreateWithName(CFSTR("Helvetica"), 20, &matrix));
130}

132constexpr CGFloat horizontalMargin = 15;
133constexpr CGFloat verticalMargin = 5;
134
135static void drawPattern(void *overlayPtr, CGContextRef ctx)
136{
137 WebViewVisualIdentificationOverlay *overlay = (WebViewVisualIdentificationOverlay *)overlayPtr;
138
139 auto attributes = @{
140 (id)kCTFontAttributeName : (id)createIdentificationFont().get(),
141 (id)kCTForegroundColorFromContextAttributeName : @YES
142 };
143 auto attributedString = adoptCF(CFAttributedStringCreate(kCFAllocatorDefault, (__bridge CFStringRef)overlay->_kind.get(), (__bridge CFDictionaryRef)attributes));
144 auto line = adoptCF(CTLineCreateWithAttributedString(attributedString.get()));
145
146 CGSize textSize = [overlay->_kind sizeWithAttributes:attributes];
147
148#if PLATFORM(IOS_FAMILY)
149 CGContextScaleCTM(ctx, 1, -1);
150 CGContextTranslateCTM(ctx, 0, -(textSize.height + verticalMargin) * 2);
151#endif
152
153 CGContextSetTextDrawingMode(ctx, kCGTextFill);
154
155 CGContextSetTextPosition(ctx, 0, 0);
156 CGContextSetFillColorWithColor(ctx, cachedCGColor(WebCore::Color::black));
157 CTLineDraw(line.get(), ctx);
158
159 CGContextSetTextPosition(ctx, 0, textSize.height + 5);
160 CGContextSetFillColorWithColor(ctx, cachedCGColor(WebCore::Color::white));
161 CTLineDraw(line.get(), ctx);
162}
163
164- (void)drawLayer:(CALayer *)layer inContext:(CGContextRef)ctx
165{
166 CGPatternCallbacks callbacks = { 0, &drawPattern, nullptr };
167 auto patternSpace = adoptCF(CGColorSpaceCreatePattern(nullptr));
168 CGContextSetFillColorSpace(ctx, patternSpace.get());
169
170 CGSize textSize = [_kind sizeWithAttributes:@{ (id)kCTFontAttributeName : (id)createIdentificationFont().get() }];
171 CGSize patternSize = CGSizeMake(textSize.width + horizontalMargin, (textSize.height + verticalMargin) * 2);
172 auto pattern = adoptCF(CGPatternCreate(self, layer.bounds, CGAffineTransformMakeRotation(M_PI_4), patternSize.width, patternSize.height, kCGPatternTilingNoDistortion, true, &callbacks));
173 CGFloat alpha = 0.5;
174 CGContextSetFillPattern(ctx, pattern.get(), &alpha);
175
176 CGContextFillRect(ctx, layer.bounds);
177}
