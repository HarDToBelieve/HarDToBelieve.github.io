61void injectInternalsObject(JSContextRef context)
62{
63 JSGlobalObject* lexicalGlobalObject = toJS(context);
64 VM& vm = lexicalGlobalObject->vm();
65 auto scope = DECLARE_CATCH_SCOPE(vm);
66 JSLockHolder lock(vm);
67 JSDOMGlobalObject* globalObject = jsCast<JSDOMGlobalObject*>(lexicalGlobalObject);
68 ScriptExecutionContext* scriptContext = globalObject->scriptExecutionContext();
69 if (is<Document>(*scriptContext)) {
70 globalObject->putDirect(vm, Identifier::fromString(vm, Internals::internalsId), toJS(lexicalGlobalObject, globalObject, Internals::create(downcast<Document>(*scriptContext))));
71 Options::useDollarVM() = true;
72 globalObject->exposeDollarVM(vm);
73 }
74 EXCEPTION_ASSERT_UNUSED(scope, !scope.exception());
75}
76
77void resetInternalsObject(JSContextRef context)
78{
79 JSGlobalObject* lexicalGlobalObject = toJS(context);
80 JSLockHolder lock(lexicalGlobalObject);
81 JSDOMGlobalObject* globalObject = jsCast<JSDOMGlobalObject*>(lexicalGlobalObject);
82 ScriptExecutionContext* scriptContext = globalObject->scriptExecutionContext();
83 Page* page = downcast<Document>(scriptContext)->frame()->page();
84 Internals::resetToConsistentState(*page);
85 InternalSettings::from(page)->resetToConsistentState();
86}

97void setWheelEventMonitorTestCallbackAndStartMonitoring(bool expectWheelEndOrCancel, bool expectMomentumEnd, WebCore::Frame& frame, JSContextRef context, JSObjectRef jsCallbackFunction)
98{
99 Page* page = frame.page();
100 if (!page || !page->isMonitoringWheelEvents())
101 return;
102
103 JSValueProtect(context, jsCallbackFunction);
104
105 if (auto wheelEventTestMonitor = page->wheelEventTestMonitor()) {
106 wheelEventTestMonitor->setTestCallbackAndStartMonitoring(expectWheelEndOrCancel, expectMomentumEnd, [=](void) {
107 JSObjectCallAsFunction(context, jsCallbackFunction, nullptr, 0, nullptr, nullptr);
108 JSValueUnprotect(context, jsCallbackFunction);
109 });
110 }
111}

217void setupNewlyCreatedServiceWorker(uint64_t serviceWorkerIdentifier)
218{
219#if ENABLE(SERVICE_WORKER)
220 auto identifier = makeObjectIdentifier<ServiceWorkerIdentifierType>(serviceWorkerIdentifier);
221 SWContextManager::singleton().postTaskToServiceWorker(identifier, [identifier] (ServiceWorkerGlobalScope& globalScope) {
222 auto* script = globalScope.script();
223 if (!script)
224 return;
225
226 auto& globalObject = *globalScope.globalObject();
227 auto& vm = globalObject.vm();
228 JSLockHolder locker(vm);
229 auto* contextWrapper = script->globalScopeWrapper();
230 contextWrapper->putDirect(vm, Identifier::fromString(vm, Internals::internalsId), toJS(&globalObject, contextWrapper, ServiceWorkerInternals::create(identifier)));
231 });
232#else
233 UNUSED_PARAM(serviceWorkerIdentifier);
234#endif
235}
