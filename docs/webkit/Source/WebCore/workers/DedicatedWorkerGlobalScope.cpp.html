55Ref<DedicatedWorkerGlobalScope> DedicatedWorkerGlobalScope::create(const WorkerParameters& params, Ref<SecurityOrigin>&& origin, DedicatedWorkerThread& thread, Ref<SecurityOrigin>&& topOrigin, IDBClient::IDBConnectionProxy* connectionProxy, SocketProvider* socketProvider)
56{
57 auto context = adoptRef(*new DedicatedWorkerGlobalScope(params, WTFMove(origin), thread, WTFMove(topOrigin), connectionProxy, socketProvider));
58 if (!params.shouldBypassMainWorldContentSecurityPolicy)
59 context->applyContentSecurityPolicyResponseHeaders(params.contentSecurityPolicyResponseHeaders);
60 return context;
61}

81ExceptionOr<void> DedicatedWorkerGlobalScope::postMessage(JSC::JSGlobalObject& state, JSC::JSValue messageValue, PostMessageOptions&& options)
82{
83 Vector<RefPtr<MessagePort>> ports;
84 auto message = SerializedScriptValue::create(state, messageValue, WTFMove(options.transfer), ports, SerializationContext::WorkerPostMessage);
85 if (message.hasException())
86 return message.releaseException();
87
88 // Disentangle the port in preparation for sending it to the remote context.
89 auto channels = MessagePort::disentanglePorts(WTFMove(ports));
90 if (channels.hasException())
91 return channels.releaseException();
92
93 thread().workerObjectProxy().postMessageToWorkerObject({ message.releaseReturnValue(), channels.releaseReturnValue() });
94 return { };
95}

124#if ENABLE(WEB_RTC)
125RefPtr<RTCRtpScriptTransformer> DedicatedWorkerGlobalScope::createRTCRtpScriptTransformer(MessageWithMessagePorts&& options)
126{
127 auto transformerOrException = RTCRtpScriptTransformer::create(*this, WTFMove(options));
128 if (transformerOrException.hasException())
129 return nullptr;
130 auto transformer = transformerOrException.releaseReturnValue();
131 dispatchEvent(RTCTransformEvent::create(eventNames().rtctransformEvent, transformer.copyRef(), Event::IsTrusted::Yes));
132 return transformer;
133}
