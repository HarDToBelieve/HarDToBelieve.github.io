43WorkerFontLoadRequest::WorkerFontLoadRequest(URL&& url, LoadedFromOpaqueSource loadedFromOpaqueSource)
44 : m_url(WTFMove(url))
45 , m_loadedFromOpaqueSource(loadedFromOpaqueSource)
46{
47}

49void WorkerFontLoadRequest::load(WorkerGlobalScope& workerGlobalScope)
50{
51 m_context = makeWeakPtr(static_cast<ScriptExecutionContext*>(&workerGlobalScope));
52
53 ResourceRequest request { m_url };
54 ASSERT(request.httpMethod() == "GET");
55
56 FetchOptions fetchOptions;
57 fetchOptions.mode = FetchOptions::Mode::SameOrigin;
58 fetchOptions.credentials = workerGlobalScope.credentials();
59 fetchOptions.cache = FetchOptions::Cache::Default;
60 fetchOptions.redirect = FetchOptions::Redirect::Follow;
61 fetchOptions.destination = FetchOptions::Destination::Worker;
62
63 ThreadableLoaderOptions options { WTFMove(fetchOptions) };
64 options.sendLoadCallbacks = SendCallbackPolicy::SendCallbacks;
65 options.contentSecurityPolicyEnforcement = m_context->shouldBypassMainWorldContentSecurityPolicy() ? ContentSecurityPolicyEnforcement::DoNotEnforce : ContentSecurityPolicyEnforcement::EnforceChildSrcDirective;
66 options.loadedFromOpaqueSource = m_loadedFromOpaqueSource;
67
68 options.serviceWorkersMode = ServiceWorkersMode::All;
69#if ENABLE(SERVICE_WORKER)
70 if (auto* activeServiceWorker = workerGlobalScope.activeServiceWorker())
71 options.serviceWorkerRegistrationIdentifier = activeServiceWorker->registrationIdentifier();
72#endif
73
74 WorkerThreadableLoader::loadResourceSynchronously(workerGlobalScope, WTFMove(request), *this, options);
75}

77bool WorkerFontLoadRequest::ensureCustomFontData(const AtomString&)
78{
79 if (!m_fontCustomPlatformData && !m_errorOccurred && !m_isLoading) {
80 convertWOFFToSfntIfNecessary(m_data);
81 if (m_data) {
82 m_fontCustomPlatformData = createFontCustomPlatformData(*m_data, m_url.fragmentIdentifier().toString());
83 if (!m_fontCustomPlatformData)
84 m_errorOccurred = true;
85 }
86 }
87
88 return m_fontCustomPlatformData.get();
89}

91RefPtr<Font> WorkerFontLoadRequest::createFont(const FontDescription& fontDescription, const AtomString&, bool syntheticBold, bool syntheticItalic, const FontFeatureSettings& fontFaceFeatures, FontSelectionSpecifiedCapabilities fontFaceCapabilities)
92{
93 ASSERT(m_fontCustomPlatformData);
94 ASSERT(m_context);
95 return Font::create(m_fontCustomPlatformData->fontPlatformData(fontDescription, syntheticBold, syntheticItalic, fontFaceFeatures, fontFaceCapabilities), Font::Origin::Remote, &m_context->fontCache());
96}

98void WorkerFontLoadRequest::setClient(FontLoadRequestClient* client)
99{
100 m_fontLoadRequestClient = client;
101
102 if (m_notifyOnClientSet) {
103 m_notifyOnClientSet = false;
104 m_fontLoadRequestClient->fontLoaded(*this);
105 }
106}
107
108void WorkerFontLoadRequest::didReceiveResponse(unsigned long, const ResourceResponse& response)
109{
110 if (response.httpStatusCode() / 100 != 2 && response.httpStatusCode())
111 m_errorOccurred = true;
112}
113
114void WorkerFontLoadRequest::didReceiveData(const uint8_t* data, int dataLength)
115{
116 if (m_errorOccurred)
117 return;
118
119 if (!m_data)
120 m_data = SharedBuffer::create();
121
122 m_data->append(data, dataLength);
123}
124
125void WorkerFontLoadRequest::didFinishLoading(unsigned long)
126{
127 m_isLoading = false;
128
129 if (!m_errorOccurred) {
130 if (m_fontLoadRequestClient)
131 m_fontLoadRequestClient->fontLoaded(*this);
132 else
133 m_notifyOnClientSet = true;
134 }
135}
136
137void WorkerFontLoadRequest::didFail(const ResourceError&)
138{
139 m_errorOccurred = true;
140}
