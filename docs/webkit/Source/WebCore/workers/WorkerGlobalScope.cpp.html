77WorkerGlobalScope::WorkerGlobalScope(WorkerThreadType type, const WorkerParameters& params, Ref<SecurityOrigin>&& origin, WorkerThread& thread, Ref<SecurityOrigin>&& topOrigin, IDBClient::IDBConnectionProxy* connectionProxy, SocketProvider* socketProvider)
78 : WorkerOrWorkletGlobalScope(type, JSC::VM::create(), &thread)
79 , m_url(params.scriptURL)
80 , m_identifier(params.identifier)
81 , m_userAgent(params.userAgent)
82 , m_isOnline(params.isOnline)
83 , m_shouldBypassMainWorldContentSecurityPolicy(params.shouldBypassMainWorldContentSecurityPolicy)
84 , m_topOrigin(WTFMove(topOrigin))
85 , m_connectionProxy(connectionProxy)
86 , m_socketProvider(socketProvider)
87 , m_performance(Performance::create(this, params.timeOrigin))
88 , m_referrerPolicy(params.referrerPolicy)
89 , m_settingsValues(params.settingsValues)
90 , m_workerType(params.workerType)
91 , m_credentials(params.credentials)
92{
93 {
94 Locker locker { allWorkerGlobalScopeIdentifiersLock };
95 allWorkerGlobalScopeIdentifiers().add(contextIdentifier());
96 }
97
98 if (m_topOrigin->hasUniversalAccess())
99 origin->grantUniversalAccess();
100 if (m_topOrigin->needsStorageAccessFromFileURLsQuirk())
101 origin->grantStorageAccessFromFileURLsQuirk();
102
103 setSecurityOriginPolicy(SecurityOriginPolicy::create(WTFMove(origin)));
104 setContentSecurityPolicy(makeUnique<ContentSecurityPolicy>(URL { m_url }, *this));
105}

107WorkerGlobalScope::~WorkerGlobalScope()
108{
109 ASSERT(thread().thread() == &Thread::current());
110 // We need to remove from the contexts map very early in the destructor so that calling postTask() on this WorkerGlobalScope from another thread is safe.
111 removeFromContextsMap();
112
113 {
114 Locker locker { allWorkerGlobalScopeIdentifiersLock };
115 allWorkerGlobalScopeIdentifiers().remove(contextIdentifier());
116 }
117
118 m_performance = nullptr;
119 m_crypto = nullptr;
120
121 // Notify proxy that we are going away. This can free the WorkerThread object, so do not access it after this.
122 thread().workerReportingProxy().workerGlobalScopeDestroyed();
123}

222void WorkerGlobalScope::close()
223{
224 if (isClosing())
225 return;
226
227 // Let current script run to completion but prevent future script evaluations.
228 // After m_closing is set, all the tasks in the queue continue to be fetched but only
229 // tasks with isCleanupTask()==true will be executed.
230 markAsClosing();
231 postTask({ ScriptExecutionContext::Task::CleanupTask, [] (ScriptExecutionContext& context) {
232 ASSERT_WITH_SECURITY_IMPLICATION(is<WorkerGlobalScope>(context));
233 WorkerGlobalScope& workerGlobalScope = downcast<WorkerGlobalScope>(context);
234 // Notify parent that this context is closed. Parent is responsible for calling WorkerThread::stop().
235 workerGlobalScope.thread().workerReportingProxy().workerGlobalScopeClosed();
236 } });
237}

289ExceptionOr<void> WorkerGlobalScope::importScripts(const Vector<String>& urls)
290{
291 ASSERT(contentSecurityPolicy());
292
293 // https://html.spec.whatwg.org/multipage/workers.html#importing-scripts-and-libraries
294 // 1. If worker global scope's type is "module", throw a TypeError exception.
295 if (m_workerType == WorkerType::Module)
296 return Exception { TypeError, "importScripts cannot be used if worker type is \"module\""_s };
297
298 Vector<URL> completedURLs;
299 completedURLs.reserveInitialCapacity(urls.size());
300 for (auto& entry : urls) {
301 URL url = completeURL(entry);
302 if (!url.isValid())
303 return Exception { SyntaxError };
304 completedURLs.uncheckedAppend(WTFMove(url));
305 }
306
307 FetchOptions::Cache cachePolicy = FetchOptions::Cache::Default;
308
309#if ENABLE(SERVICE_WORKER)
310 bool isServiceWorkerGlobalScope = is<ServiceWorkerGlobalScope>(*this);
311 if (isServiceWorkerGlobalScope) {
312 // FIXME: We need to add support for the 'imported scripts updated' flag as per:
313 // https://w3c.github.io/ServiceWorker/#importscripts
314 auto& serviceWorkerGlobalScope = downcast<ServiceWorkerGlobalScope>(*this);
315 auto& registration = serviceWorkerGlobalScope.registration();
316 if (registration.updateViaCache() == ServiceWorkerUpdateViaCache::None || registration.needsUpdate())
317 cachePolicy = FetchOptions::Cache::NoCache;
318 }
319#endif
320
321 for (auto& url : completedURLs) {
322 // FIXME: Convert this to check the isolated world's Content Security Policy once webkit.org/b/104520 is solved.
323 bool shouldBypassMainWorldContentSecurityPolicy = this->shouldBypassMainWorldContentSecurityPolicy();
324 if (!shouldBypassMainWorldContentSecurityPolicy && !contentSecurityPolicy()->allowScriptFromSource(url))
325 return Exception { NetworkError };
326
327 auto scriptLoader = WorkerScriptLoader::create();
328 auto cspEnforcement = shouldBypassMainWorldContentSecurityPolicy ? ContentSecurityPolicyEnforcement::DoNotEnforce : ContentSecurityPolicyEnforcement::EnforceScriptSrcDirective;
329 if (auto exception = scriptLoader->loadSynchronously(this, url, FetchOptions::Mode::NoCors, cachePolicy, cspEnforcement, resourceRequestIdentifier()))
330 return WTFMove(*exception);
331
332 InspectorInstrumentation::scriptImported(*this, scriptLoader->identifier(), scriptLoader->script().toString());
333
334 WeakPtr<ScriptBufferSourceProvider> sourceProvider;
335 {
336 NakedPtr<JSC::Exception> exception;
337 ScriptSourceCode sourceCode(scriptLoader->script(), URL(scriptLoader->responseURL()));
338 sourceProvider = makeWeakPtr(static_cast<ScriptBufferSourceProvider&>(sourceCode.provider()));
339 script()->evaluate(sourceCode, exception);
340 if (exception) {
341 script()->setException(exception);
342 return { };
343 }
344 }
345 if (sourceProvider)
346 addImportedScriptSourceProvider(url, *sourceProvider);
347 }
348
349 return { };
350}

377void WorkerGlobalScope::addMessage(MessageSource source, MessageLevel level, const String& messageText, const String& sourceURL, unsigned lineNumber, unsigned columnNumber, RefPtr<ScriptCallStack>&& callStack, JSC::JSGlobalObject* state, unsigned long requestIdentifier)
378{
379 if (!isContextThread()) {
380 postTask(AddConsoleMessageTask(source, level, messageText));
381 return;
382 }
383
384 std::unique_ptr<Inspector::ConsoleMessage> message;
385 if (callStack)
386 message = makeUnique<Inspector::ConsoleMessage>(source, MessageType::Log, level, messageText, callStack.releaseNonNull(), requestIdentifier);
387 else
388 message = makeUnique<Inspector::ConsoleMessage>(source, MessageType::Log, level, messageText, sourceURL, lineNumber, columnNumber, state, requestIdentifier);
389 InspectorInstrumentation::addMessageToConsole(*this, WTFMove(message));
390}

394bool WorkerGlobalScope::wrapCryptoKey(const Vector<uint8_t>& key, Vector<uint8_t>& wrappedKey)
395{
396 Ref protectedThis { *this };
397 bool success = false;
398 BinarySemaphore semaphore;
399 thread().workerLoaderProxy().postTaskToLoader([&semaphore, &success, &key, &wrappedKey](auto& context) {
400 success = context.wrapCryptoKey(key, wrappedKey);
401 semaphore.signal();
402 });
403 semaphore.wait();
404 return success;
405}

522void WorkerGlobalScope::deleteJSCodeAndGC(Synchronous synchronous)
523{
524 ASSERT(isContextThread());
525
526 JSC::JSLockHolder lock(vm());
527 vm().deleteAllCode(JSC::DeleteAllCodeIfNotCollecting);
528
529 if (synchronous == Synchronous::Yes) {
530 if (!vm().heap.isCurrentThreadBusy()) {
531 vm().heap.collectNow(JSC::Sync, JSC::CollectionScope::Full);
532 WTF::releaseFastMallocFreeMemory();
533 return;
534 }
535 }
536#if PLATFORM(IOS_FAMILY)
537 if (!vm().heap.isCurrentThreadBusy()) {
538 vm().heap.collectNowFullIfNotDoneRecently(JSC::Async);
539 return;
540 }
541#endif
542#if USE(CF) || USE(GLIB)
543 vm().heap.reportAbandonedObjectGraph();
544#else
545 vm().heap.collectNow(JSC::Async, JSC::CollectionScope::Full);
546#endif
547}

549void WorkerGlobalScope::releaseMemoryInWorkers(Synchronous synchronous)
550{
551 Locker locker { allWorkerGlobalScopeIdentifiersLock };
552 for (auto& globalScopeIdentifier : allWorkerGlobalScopeIdentifiers()) {
553 postTaskTo(globalScopeIdentifier, [synchronous](auto& context) {
554 downcast<WorkerGlobalScope>(context).releaseMemory(synchronous);
555 });
556 }

585void WorkerGlobalScope::updateSourceProviderBuffers(const ScriptBuffer& mainScript, const HashMap<URL, ScriptBuffer>& importedScripts)
586{
587 ASSERT(isContextThread());
588
589 if (mainScript && m_mainScriptSourceProvider)
590 m_mainScriptSourceProvider->tryReplaceScriptBuffer(mainScript);
591
592 for (auto& pair : importedScripts) {
593 auto it = m_importedScriptsSourceProviders.find(pair.key);
594 if (it == m_importedScriptsSourceProviders.end())
595 continue;
596 for (auto& sourceProvider : it->value)
597 sourceProvider.tryReplaceScriptBuffer(pair.value);
598 }
599}
