42HashSet<WorkerInspectorProxy*>& WorkerInspectorProxy::allWorkerInspectorProxies()
43{
44 static NeverDestroyed<HashSet<WorkerInspectorProxy*>> proxies;
45 return proxies;
46}

48WorkerInspectorProxy::WorkerInspectorProxy(const String& identifier)
49 : m_identifier(identifier)
50{
51}
52
53WorkerInspectorProxy::~WorkerInspectorProxy()
54{
55 ASSERT(!m_workerThread);
56 ASSERT(!m_pageChannel);
57}

59WorkerThreadStartMode WorkerInspectorProxy::workerStartMode(ScriptExecutionContext& scriptExecutionContext)
60{
61 bool pauseOnStart = InspectorInstrumentation::shouldWaitForDebuggerOnStart(scriptExecutionContext);
62 return pauseOnStart ? WorkerThreadStartMode::WaitForInspector : WorkerThreadStartMode::Normal;
63}
64
65void WorkerInspectorProxy::workerStarted(ScriptExecutionContext* scriptExecutionContext, WorkerThread* thread, const URL& url, const String& name)
66{
67 ASSERT(!m_workerThread);
68
69 m_scriptExecutionContext = scriptExecutionContext;
70 m_workerThread = thread;
71 m_url = url;
72 m_name = name;
73
74 allWorkerInspectorProxies().add(this);
75
76 InspectorInstrumentation::workerStarted(*this);
77}
78
79void WorkerInspectorProxy::workerTerminated()
80{
81 if (!m_workerThread)
82 return;
83
84 InspectorInstrumentation::workerTerminated(*this);
85
86 allWorkerInspectorProxies().remove(this);
87
88 m_scriptExecutionContext = nullptr;
89 m_workerThread = nullptr;
90 m_pageChannel = nullptr;
91}
92
93void WorkerInspectorProxy::resumeWorkerIfPaused()
94{
95 m_workerThread->runLoop().postDebuggerTask([] (ScriptExecutionContext& context) {
96 downcast<WorkerGlobalScope>(context).thread().stopRunningDebuggerTasks();
97 });
98}
99
100void WorkerInspectorProxy::connectToWorkerInspectorController(PageChannel& channel)
101{
102 ASSERT(m_workerThread);
103 if (!m_workerThread)
104 return;
105
106 m_pageChannel = &channel;
107
108 m_workerThread->runLoop().postDebuggerTask([] (ScriptExecutionContext& context) {
109 downcast<WorkerGlobalScope>(context).inspectorController().connectFrontend();
110 });
111}
112
113void WorkerInspectorProxy::disconnectFromWorkerInspectorController()
114{
115 ASSERT(m_workerThread);
116 if (!m_workerThread)
117 return;
118
119 m_pageChannel = nullptr;
120
121 m_workerThread->runLoop().postDebuggerTask([] (ScriptExecutionContext& context) {
122 downcast<WorkerGlobalScope>(context).inspectorController().disconnectFrontend(DisconnectReason::InspectorDestroyed);
123
124 // In case the worker is paused running debugger tasks, ensure we break out of
125 // the pause since this will be the last debugger task we send to the worker.
126 downcast<WorkerGlobalScope>(context).thread().stopRunningDebuggerTasks();
127 });
128}
129
130void WorkerInspectorProxy::sendMessageToWorkerInspectorController(const String& message)
131{
132 ASSERT(m_workerThread);
133 if (!m_workerThread)
134 return;
135
136 m_workerThread->runLoop().postDebuggerTask([message = message.isolatedCopy()] (ScriptExecutionContext& context) {
137 downcast<WorkerGlobalScope>(context).inspectorController().dispatchMessageFromFrontend(message);
138 });
139}
140
141void WorkerInspectorProxy::sendMessageFromWorkerToFrontend(const String& message)
142{
143 if (!m_pageChannel)
144 return;
145
146 m_pageChannel->sendMessageFromWorkerToFrontend(*this, message);
147}
