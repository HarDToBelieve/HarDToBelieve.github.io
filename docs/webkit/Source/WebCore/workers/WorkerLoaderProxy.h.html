31#pragma once
32
33#include "ScriptExecutionContext.h"
34
35namespace WebCore {
36
37class CacheStorageConnection;
38
39// A proxy to talk to the loader context. Normally, the document on the main thread
40// provides loading services for the subordinate workers. This interface provides 2-way
41// communications to the Document context and back to the worker.
42// Note that in multi-process browsers, the Worker object context and the Document
43// context can be distinct.
44class WorkerLoaderProxy {
45public:
46 virtual ~WorkerLoaderProxy() = default;
47
48 virtual bool isWorkerMessagingProxy() const { return false; }
49
50 // Creates a cache storage connection to be used on the main thread. Method must be called on the main thread.
51 virtual RefPtr<CacheStorageConnection> createCacheStorageConnection() = 0;
52
53 virtual RefPtr<RTCDataChannelRemoteHandlerConnection> createRTCDataChannelRemoteHandlerConnection() = 0;
54
55 // Posts a task to the thread which runs the loading code (normally, the main thread).
56 virtual void postTaskToLoader(ScriptExecutionContext::Task&&) = 0;
57
58 // Posts callbacks from loading code to the WorkerGlobalScope. The 'mode' is used to differentiate
59 // specific synchronous loading requests so they can be 'nested', per spec.
60 // Returns true if the task was posted successfully.
61 virtual bool postTaskForModeToWorkerOrWorkletGlobalScope(ScriptExecutionContext::Task&&, const String& mode) = 0;
62};
63
64} // namespace WebCore
