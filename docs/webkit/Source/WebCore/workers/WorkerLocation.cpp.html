26#include "config.h"
27#include "WorkerLocation.h"
28
29#include "SecurityOrigin.h"
30#include <wtf/text/WTFString.h>
31
32namespace WebCore {
33
34String WorkerLocation::href() const
35{
36 return m_url.string();
37}
38
39String WorkerLocation::protocol() const
40{
41 return makeString(m_url.protocol(), ":");
42}
43
44String WorkerLocation::host() const
45{
46 return m_url.hostAndPort();
47}
48
49String WorkerLocation::hostname() const
50{
51 return m_url.host().toString();
52}
53
54String WorkerLocation::port() const
55{
56 auto port = m_url.port();
57 return port ? String::number(*port) : emptyString();
58}
59
60String WorkerLocation::pathname() const
61{
62 auto path = m_url.path();
63 return path.isEmpty() ? "/"_s : path.toString();
64}
65
66String WorkerLocation::search() const
67{
68 return m_url.query().isEmpty() ? emptyString() : m_url.queryWithLeadingQuestionMark().toString();
69}
70
71String WorkerLocation::hash() const
72{
73 return m_url.fragmentIdentifier().isEmpty() ? emptyString() : m_url.fragmentIdentifierWithLeadingNumberSign().toString();
74}
75
76String WorkerLocation::origin() const
77{
78 return m_origin;
79}
80
81} // namespace WebCore
