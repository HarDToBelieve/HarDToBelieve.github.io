26#pragma once
27
28#include "WorkerGlobalScopeProxy.h"
29#include "WorkerDebuggerProxy.h"
30#include "WorkerLoaderProxy.h"
31#include "WorkerObjectProxy.h"
32#include <wtf/MonotonicTime.h>
33#include <wtf/ThreadSafeRefCounted.h>
34
35namespace WebCore {
36
37class DedicatedWorkerThread;
38class WorkerInspectorProxy;
39
40class WorkerMessagingProxy final : public ThreadSafeRefCounted<WorkerMessagingProxy>, public WorkerGlobalScopeProxy, public WorkerObjectProxy, public WorkerLoaderProxy, public WorkerDebuggerProxy {
41 WTF_MAKE_FAST_ALLOCATED;
42public:
43 explicit WorkerMessagingProxy(Worker&);
44 virtual ~WorkerMessagingProxy();
45
46 // Implementation of WorkerLoaderProxy.
47 // This method is used in the main thread to post task back to the worker thread.
48 bool postTaskForModeToWorkerOrWorkletGlobalScope(ScriptExecutionContext::Task&&, const String& mode) final;
49
50private:
51 // Implementations of WorkerGlobalScopeProxy.
52 // (Only use these functions in the worker object thread.)
53 void startWorkerGlobalScope(const URL& scriptURL, const String& name, const String& userAgent, bool isOnline, const ScriptBuffer& sourceCode, const ContentSecurityPolicyResponseHeaders&, bool shouldBypassMainWorldContentSecurityPolicy, MonotonicTime timeOrigin, ReferrerPolicy, WorkerType, FetchRequestCredentials, JSC::RuntimeFlags) final;
54 void terminateWorkerGlobalScope() final;
55 void postMessageToWorkerGlobalScope(MessageWithMessagePorts&&) final;
56 void postTaskToWorkerGlobalScope(Function<void(ScriptExecutionContext&)>&&) final;
57 bool hasPendingActivity() const final;
58 void workerObjectDestroyed() final;
59 void notifyNetworkStateChange(bool isOnline) final;
60 void suspendForBackForwardCache() final;
61 void resumeForBackForwardCache() final;
62
63 // Implementation of WorkerObjectProxy.
64 // (Only use these functions in the worker context thread.)
65 void postMessageToWorkerObject(MessageWithMessagePorts&&) final;
66 void postTaskToWorkerObject(Function<void(Worker&)>&&) final;
67 void postExceptionToWorkerObject(const String& errorMessage, int lineNumber, int columnNumber, const String& sourceURL) final;
68 void confirmMessageFromWorkerObject(bool hasPendingActivity) final;
69 void reportPendingActivity(bool hasPendingActivity) final;
70 void workerGlobalScopeClosed() final;
71 void workerGlobalScopeDestroyed() final;
72
73 // Implementation of WorkerDebuggerProxy.
74 // (Only use these functions in the worker context thread.)
75 void postMessageToDebugger(const String&) final;
76 void setResourceCachingDisabledByWebInspector(bool) final;
77
78 // Implementation of WorkerLoaderProxy.
79 // These functions are called on different threads to schedule loading
80 // requests and to send callbacks back to WorkerGlobalScope.
81 bool isWorkerMessagingProxy() const final { return true; }
82 void postTaskToLoader(ScriptExecutionContext::Task&&) final;
83 RefPtr<CacheStorageConnection> createCacheStorageConnection() final;
84 RefPtr<RTCDataChannelRemoteHandlerConnection> createRTCDataChannelRemoteHandlerConnection() final;
85
86 void workerThreadCreated(DedicatedWorkerThread&);
87
88 // Only use this method on the worker object thread.
89 bool askedToTerminate() const { return m_askedToTerminate; }
90
91 void workerGlobalScopeDestroyedInternal();
92 void reportPendingActivityInternal(bool confirmingMessage, bool hasPendingActivity);
93 Worker* workerObject() const { return m_workerObject; }
94
95 RefPtr<ScriptExecutionContext> m_scriptExecutionContext;
96 std::unique_ptr<WorkerInspectorProxy> m_inspectorProxy;
97 Worker* m_workerObject;
98 bool m_mayBeDestroyed { false };
99 RefPtr<DedicatedWorkerThread> m_workerThread;
100
101 unsigned m_unconfirmedMessageCount { 0 }; // Unconfirmed messages from worker object to worker thread.
102 bool m_workerThreadHadPendingActivity { false }; // The latest confirmation from worker thread reported that it was still active.
103
104 bool m_askedToSuspend { false };
105 bool m_askedToTerminate { false };
106
107 Vector<std::unique_ptr<ScriptExecutionContext::Task>> m_queuedEarlyTasks; // Tasks are queued here until there's a thread object created.
108};
109
110} // namespace WebCore
111
112SPECIALIZE_TYPE_TRAITS_BEGIN(WebCore::WorkerMessagingProxy)
113 static bool isType(const WebCore::WorkerLoaderProxy& proxy) { return proxy.isWorkerMessagingProxy(); }
114SPECIALIZE_TYPE_TRAITS_END()
