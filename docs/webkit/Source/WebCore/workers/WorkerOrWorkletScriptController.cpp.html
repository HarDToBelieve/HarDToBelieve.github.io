66WorkerOrWorkletScriptController::WorkerOrWorkletScriptController(WorkerThreadType type, Ref<VM>&& vm, WorkerOrWorkletGlobalScope* globalScope)
67 : m_vm(WTFMove(vm))
68 , m_globalScope(globalScope)
69 , m_globalScopeWrapper(*m_vm)
70{
71 m_vm->heap.acquireAccess(); // It's not clear that we have good discipline for heap access, so turn it on permanently.
72 {
73 JSLockHolder lock(m_vm.get());
74 m_vm->ensureTerminationException();
75 }
76
77 JSVMClientData::initNormalWorld(m_vm.get(), type);
78}

120void WorkerOrWorkletScriptController::scheduleExecutionTermination()
121{
122 {
123 // The mutex provides a memory barrier to ensure that once
124 // termination is scheduled, isTerminatingExecution() will
125 // accurately reflect that lexicalGlobalObject when called from another thread.
126 Locker locker { m_scheduledTerminationLock };
127 if (m_isTerminatingExecution)
128 return;
129
130 m_isTerminatingExecution = true;
131 }
132 m_vm->notifyNeedTermination();
133}

202void WorkerOrWorkletScriptController::evaluate(const ScriptSourceCode& sourceCode, String* returnedExceptionMessage)
203{
204 if (isExecutionForbidden())
205 return;
206
207 VM& vm = this->vm();
208 NakedPtr<JSC::Exception> uncaughtException;
209 evaluate(sourceCode, uncaughtException, returnedExceptionMessage);
210 if ((uncaughtException && vm.isTerminationException(uncaughtException)) || isTerminatingExecution()) {
211 forbidExecution();
212 return;
213 }
214 if (uncaughtException) {
215 JSLockHolder lock(vm);
216 reportException(m_globalScopeWrapper.get(), uncaughtException);
217 }
218}

220void WorkerOrWorkletScriptController::evaluate(const ScriptSourceCode& sourceCode, NakedPtr<JSC::Exception>& returnedException, String* returnedExceptionMessage)
221{
222 if (isExecutionForbidden())
223 return;
224
225 initScriptIfNeeded();
226
227 auto& globalObject = *m_globalScopeWrapper.get();
228 VM& vm = globalObject.vm();
229 JSLockHolder lock { vm };
230
231 JSExecState::profiledEvaluate(&globalObject, JSC::ProfilingReason::Other, sourceCode.jsSourceCode(), m_globalScopeWrapper->globalThis(), returnedException);
232
233 if ((returnedException && vm.isTerminationException(returnedException)) || isTerminatingExecution()) {
234 forbidExecution();
235 return;
236 }
237
238 if (returnedException) {
239 if (m_globalScope->canIncludeErrorDetails(sourceCode.cachedScript(), sourceCode.url().string())) {
240 // FIXME: It's not great that this can run arbitrary code to string-ify the value of the exception.
241 // Do we need to do anything to handle that properly, if it, say, raises another exception?
242 if (returnedExceptionMessage)
243 *returnedExceptionMessage = returnedException->value().toWTFString(&globalObject);
244 } else {
245 // Overwrite the detailed error with a generic error.
246 String genericErrorMessage { "Script error."_s };
247 if (returnedExceptionMessage)
248 *returnedExceptionMessage = genericErrorMessage;
249 returnedException = JSC::Exception::create(vm, createError(&globalObject, genericErrorMessage));
250 }
251 }

270MessageQueueWaitResult WorkerOrWorkletScriptController::loadModuleSynchronously(WorkerScriptFetcher& scriptFetcher, const ScriptSourceCode& sourceCode)
271{
272 if (isExecutionForbidden())
273 return MessageQueueTerminated;
274
275 initScriptIfNeeded();
276
277 auto& globalObject = *m_globalScopeWrapper.get();
278 VM& vm = globalObject.vm();
279 JSLockHolder lock { vm };
280
281 auto protector = makeRef(scriptFetcher);
282 {
283 auto& promise = JSExecState::loadModule(globalObject, sourceCode.jsSourceCode(), JSC::JSScriptFetcher::create(vm, { &scriptFetcher }));
284
285 auto& fulfillHandler = *JSNativeStdFunction::create(vm, &globalObject, 1, String(), [protector](JSGlobalObject* globalObject, CallFrame* callFrame) -> JSC::EncodedJSValue {
286 VM& vm = globalObject->vm();
287 JSLockHolder lock { vm };
288 auto scope = DECLARE_THROW_SCOPE(vm);
289 Identifier moduleKey = jsValueToModuleKey(globalObject, callFrame->argument(0));
290 RETURN_IF_EXCEPTION(scope, { });
291 protector->notifyLoadCompleted(*moduleKey.impl());
292 return JSValue::encode(jsUndefined());
293 });
294
295 auto& rejectHandler = *JSNativeStdFunction::create(vm, &globalObject, 1, String(), [protector](JSGlobalObject* globalObject, CallFrame* callFrame) {
296 VM& vm = globalObject->vm();
297 JSLockHolder lock { vm };
298 JSValue errorValue = callFrame->argument(0);
299 if (errorValue.isObject()) {
300 auto* object = JSC::asObject(errorValue);
301 if (JSValue failureKindValue = object->getDirect(vm, static_cast<JSVMClientData&>(*vm.clientData).builtinNames().failureKindPrivateName())) {
302 // This is host propagated error in the module loader pipeline.
303 switch (static_cast<ModuleFetchFailureKind>(failureKindValue.asInt32())) {
304 case ModuleFetchFailureKind::WasErrored:
305 protector->notifyLoadFailed(LoadableScript::Error {
306 LoadableScript::ErrorType::CachedScript,
307 std::nullopt
308 });
309 break;
310 case ModuleFetchFailureKind::WasCanceled:
311 protector->notifyLoadWasCanceled();
312 break;
313 }
314 return JSValue::encode(jsUndefined());
315 }
316 }
317
318 auto scope = DECLARE_CATCH_SCOPE(vm);
319 protector->notifyLoadFailed(LoadableScript::Error {
320 LoadableScript::ErrorType::CachedScript,
321 LoadableScript::ConsoleMessage {
322 MessageSource::JS,
323 MessageLevel::Error,
324 retrieveErrorMessage(*globalObject, vm, errorValue, scope),
325 }
326 });
327 return JSValue::encode(jsUndefined());
328 });
329
330 promise.then(&globalObject, &fulfillHandler, &rejectHandler);
331 }
332 m_globalScope->eventLoop().performMicrotaskCheckpoint();
333
334 // Drive RunLoop until we get either of "Worker is terminated", "Loading is done", or "Loading is failed".
335 WorkerRunLoop& runLoop = m_globalScope->workerOrWorkletThread()->runLoop();
336
337 // We do not want to receive messages that are not related to asynchronous resource loading.
338 // Otherwise, a worker discards some messages from the main thread here in a racy way.
339 // For example, the main thread can postMessage just after creating a Worker. In that case, postMessage's
340 // task is queued in WorkerRunLoop before start running module scripts. This task should not be discarded
341 // in the following driving of the RunLoop which mainly attempt to collect initial load of module scripts.
342 String taskMode = WorkerModuleScriptLoader::taskMode();
343 MessageQueueWaitResult result = MessageQueueMessageReceived;
344 while ((!protector->isLoaded() && !protector->wasCanceled()) && result != MessageQueueTerminated) {
345 result = runLoop.runInMode(m_globalScope, taskMode);
346 if (result != MessageQueueTerminated)
347 m_globalScope->eventLoop().performMicrotaskCheckpoint();
348 }
349
350 return result;
351}

353void WorkerOrWorkletScriptController::linkAndEvaluateModule(WorkerScriptFetcher& scriptFetcher, const ScriptSourceCode& sourceCode, String* returnedExceptionMessage)
354{
355 if (isExecutionForbidden())
356 return;
357
358 initScriptIfNeeded();
359
360 auto& globalObject = *m_globalScopeWrapper.get();
361 VM& vm = globalObject.vm();
362 JSLockHolder lock { vm };
363
364 NakedPtr<JSC::Exception> returnedException;
365 JSExecState::linkAndEvaluateModule(globalObject, Identifier::fromUid(vm, scriptFetcher.moduleKey()), jsUndefined(), returnedException);
366 if ((returnedException && vm.isTerminationException(returnedException)) || isTerminatingExecution()) {
367 forbidExecution();
368 return;
369 }
370
371 if (returnedException) {
372 if (m_globalScope->canIncludeErrorDetails(sourceCode.cachedScript(), sourceCode.url().string())) {
373 // FIXME: It's not great that this can run arbitrary code to string-ify the value of the exception.
374 // Do we need to do anything to handle that properly, if it, say, raises another exception?
375 if (returnedExceptionMessage)
376 *returnedExceptionMessage = returnedException->value().toWTFString(&globalObject);
377 } else {
378 String genericErrorMessage { "Script error."_s };
379 if (returnedExceptionMessage)
380 *returnedExceptionMessage = genericErrorMessage;
381 }
382 }

385void WorkerOrWorkletScriptController::loadAndEvaluateModule(const URL& moduleURL, FetchOptions::Credentials credentials, CompletionHandler<void(std::optional<Exception>&&)>&& completionHandler)
386{
387 if (isExecutionForbidden()) {
388 completionHandler(Exception { NotAllowedError });
389 return;
390 }
391
392 initScriptIfNeeded();
393
394 auto& globalObject = *m_globalScopeWrapper.get();
395 VM& vm = globalObject.vm();
396 JSLockHolder lock { vm };
397
398 auto scriptFetcher = WorkerScriptFetcher::create(credentials, globalScope()->destination(), globalScope()->referrerPolicy());
399 {
400 auto& promise = JSExecState::loadModule(globalObject, moduleURL.string(), JSC::JSScriptFetchParameters::create(vm, makeRef(scriptFetcher->parameters())), JSC::JSScriptFetcher::create(vm, { scriptFetcher.ptr() }));
401
402 auto task = createSharedTask<void(std::optional<Exception>&&)>([completionHandler = WTFMove(completionHandler)](std::optional<Exception>&& exception) mutable {
403 completionHandler(WTFMove(exception));
404 });
405
406 auto& fulfillHandler = *JSNativeStdFunction::create(vm, &globalObject, 1, String(), [task, scriptFetcher](JSGlobalObject* globalObject, CallFrame* callFrame) -> JSC::EncodedJSValue {
407 // task->run(std::nullopt);
408 VM& vm = globalObject->vm();
409 JSLockHolder lock { vm };
410 auto scope = DECLARE_THROW_SCOPE(vm);
411
412 Identifier moduleKey = jsValueToModuleKey(globalObject, callFrame->argument(0));
413 RETURN_IF_EXCEPTION(scope, { });
414 scriptFetcher->notifyLoadCompleted(*moduleKey.impl());
415
416 auto* context = downcast<WorkerOrWorkletGlobalScope>(jsCast<JSDOMGlobalObject*>(globalObject)->scriptExecutionContext());
417 if (!context || !context->script()) {
418 task->run(std::nullopt);
419 return JSValue::encode(jsUndefined());
420 }
421
422 NakedPtr<JSC::Exception> returnedException;
423 JSExecState::linkAndEvaluateModule(*globalObject, moduleKey, jsUndefined(), returnedException);
424 if ((returnedException && vm.isTerminationException(returnedException)) || context->script()->isTerminatingExecution()) {
425 if (context->script())
426 context->script()->forbidExecution();
427 task->run(std::nullopt);
428 return JSValue::encode(jsUndefined());
429 }
430
431 if (returnedException) {
432 String message;
433 if (context->canIncludeErrorDetails(nullptr, moduleKey.string())) {
434 // FIXME: It's not great that this can run arbitrary code to string-ify the value of the exception.
435 // Do we need to do anything to handle that properly, if it, say, raises another exception?
436 message = returnedException->value().toWTFString(globalObject);
437 } else
438 message = "Script error."_s;
439 context->reportException(message, { }, { }, { }, { }, { });
440 }
441
442 task->run(std::nullopt);
443 return JSValue::encode(jsUndefined());
444 });
445
446 auto& rejectHandler = *JSNativeStdFunction::create(vm, &globalObject, 1, String(), [task](JSGlobalObject* globalObject, CallFrame* callFrame) {
447 VM& vm = globalObject->vm();
448 JSLockHolder lock { vm };
449 JSValue errorValue = callFrame->argument(0);
450 if (errorValue.isObject()) {
451 auto* object = JSC::asObject(errorValue);
452 if (object->getDirect(vm, static_cast<JSVMClientData&>(*vm.clientData).builtinNames().failureKindPrivateName())) {
453 auto catchScope = DECLARE_CATCH_SCOPE(vm);
454 String message = retrieveErrorMessageWithoutName(*globalObject, vm, object, catchScope);
455 task->run(Exception { AbortError, message });
456 return JSValue::encode(jsUndefined());
457 }
458 if (object->inherits<ErrorInstance>(vm)) {
459 auto* error = jsCast<ErrorInstance*>(object);
460 switch (error->errorType()) {
461 case ErrorType::TypeError: {
462 auto catchScope = DECLARE_CATCH_SCOPE(vm);
463 String message = retrieveErrorMessageWithoutName(*globalObject, vm, error, catchScope);
464 task->run(Exception { TypeError, message });
465 return JSValue::encode(jsUndefined());
466 }
467 case ErrorType::SyntaxError: {
468 auto catchScope = DECLARE_CATCH_SCOPE(vm);
469 String message = retrieveErrorMessageWithoutName(*globalObject, vm, error, catchScope);
470 task->run(Exception { JSSyntaxError, message });
471 return JSValue::encode(jsUndefined());
472 }
473 default:
474 break;
475 }
476 }
477 }
478
479 auto catchScope = DECLARE_CATCH_SCOPE(vm);
480 String message = retrieveErrorMessageWithoutName(*globalObject, vm, errorValue, catchScope);
481 task->run(Exception { AbortError, message });
482 return JSValue::encode(jsUndefined());
483 });
484
485 promise.then(&globalObject, &fulfillHandler, &rejectHandler);
486 }
487 m_globalScope->eventLoop().performMicrotaskCheckpoint();
488}

490template<typename JSGlobalScopePrototype, typename JSGlobalScope, typename GlobalScope>
491void WorkerOrWorkletScriptController::initScriptWithSubclass()
492{
493 ASSERT(!m_globalScopeWrapper);
494
495 JSLockHolder lock { vm() };
496
497 // Explicitly protect the global object's prototype so it isn't collected
498 // when we allocate the global object. (Once the global object is fully
499 // constructed, it can mark its own prototype.)
500 Structure* contextPrototypeStructure = JSGlobalScopePrototype::createStructure(*m_vm, nullptr, jsNull());
501 auto* contextPrototype = JSGlobalScopePrototype::create(*m_vm, nullptr, contextPrototypeStructure);
502 Structure* structure = JSGlobalScope::createStructure(*m_vm, nullptr, contextPrototype);
503 auto* proxyStructure = JSProxy::createStructure(*m_vm, nullptr, jsNull());
504 auto* proxy = JSProxy::create(*m_vm, proxyStructure);
505
506 m_globalScopeWrapper.set(*m_vm, JSGlobalScope::create(*m_vm, structure, static_cast<GlobalScope&>(*m_globalScope), proxy));
507 contextPrototypeStructure->setGlobalObject(*m_vm, m_globalScopeWrapper.get());
508 ASSERT(structure->globalObject() == m_globalScopeWrapper);
509 ASSERT(m_globalScopeWrapper->structure(*m_vm)->globalObject() == m_globalScopeWrapper);
510 contextPrototype->structure(*m_vm)->setGlobalObject(*m_vm, m_globalScopeWrapper.get());
511 auto* globalScopePrototype = JSGlobalScope::prototype(*m_vm, *m_globalScopeWrapper.get());
512 globalScopePrototype->didBecomePrototype();
513 contextPrototype->structure(*m_vm)->setPrototypeWithoutTransition(*m_vm, globalScopePrototype);
514
515 proxy->setTarget(*m_vm, m_globalScopeWrapper.get());
516 proxy->structure(*m_vm)->setGlobalObject(*m_vm, m_globalScopeWrapper.get());
517
518 ASSERT(m_globalScopeWrapper->globalObject() == m_globalScopeWrapper);
519 ASSERT(asObject(m_globalScopeWrapper->getPrototypeDirect(*m_vm))->globalObject() == m_globalScopeWrapper);
520
521 m_consoleClient = makeUnique<WorkerConsoleClient>(*m_globalScope);
522 m_globalScopeWrapper->setConsoleClient(makeWeakPtr(*m_consoleClient));
523}

525void WorkerOrWorkletScriptController::initScript()
526{
527 ASSERT(m_vm.get());
528 JSC::DeferTermination deferTermination(*m_vm.get());
529
530 if (is<DedicatedWorkerGlobalScope>(m_globalScope)) {
531 initScriptWithSubclass<JSDedicatedWorkerGlobalScopePrototype, JSDedicatedWorkerGlobalScope, DedicatedWorkerGlobalScope>();
532 return;
533 }
534
535#if ENABLE(SERVICE_WORKER)
536 if (is<ServiceWorkerGlobalScope>(m_globalScope)) {
537 initScriptWithSubclass<JSServiceWorkerGlobalScopePrototype, JSServiceWorkerGlobalScope, ServiceWorkerGlobalScope>();
538 return;
539 }
540#endif
541
542#if ENABLE(CSS_PAINTING_API)
543 if (is<PaintWorkletGlobalScope>(m_globalScope)) {
544 initScriptWithSubclass<JSPaintWorkletGlobalScopePrototype, JSPaintWorkletGlobalScope, PaintWorkletGlobalScope>();
545 return;
546 }
547#endif
548
549#if ENABLE(WEB_AUDIO)
550 if (is<AudioWorkletGlobalScope>(m_globalScope)) {
551 initScriptWithSubclass<JSAudioWorkletGlobalScopePrototype, JSAudioWorkletGlobalScope, AudioWorkletGlobalScope>();
552 return;
553 }
554#endif
555
556 ASSERT_NOT_REACHED();
557}
