71void WorkerOrWorkletThread::startRunningDebuggerTasks()
72{
73 ASSERT(!m_pausedForDebugger);
74 m_pausedForDebugger = true;
75
76 MessageQueueWaitResult result;
77 do {
78 result = m_runLoop.runInDebuggerMode(*m_globalScope);
79 } while (result != MessageQueueTerminated && m_pausedForDebugger);
80}
81
82void WorkerOrWorkletThread::stopRunningDebuggerTasks()
83{
84 m_pausedForDebugger = false;
85}

87void WorkerOrWorkletThread::runEventLoop()
88{
89 // Does not return until terminated.
90 m_runLoop.run(m_globalScope.get());
91}

93void WorkerOrWorkletThread::workerOrWorkletThread()
94{
95 auto protectedThis = makeRef(*this);
96
97 // Propagate the mainThread's fenv to workers.
98#if PLATFORM(IOS_FAMILY)
99 FloatingPointEnvironment::singleton().propagateMainThreadEnvironment();
100#endif
101
102#if USE(GLIB)
103 GRefPtr<GMainContext> mainContext = adoptGRef(g_main_context_new());
104 g_main_context_push_thread_default(mainContext.get());
105#endif
106
107 WorkerOrWorkletScriptController* scriptController;
108 {
109 // Mutex protection is necessary to ensure that we don't change m_globalScope
110 // while WorkerThread::stop() is accessing it. Note that WorkerThread::stop() can
111 // be called before we've finished creating the WorkerGlobalScope.
112 Locker locker { m_threadCreationAndGlobalScopeLock };
113 m_globalScope = createGlobalScope();
114
115 // When running out of memory, createGlobalScope() may return null because we could not allocate a JSC::VM.
116 if (!m_globalScope) {
117 WTFLogAlways("Error: Failed to create a WorkerOrWorkerGlobalScope.");
118 return;
119 }
120
121 scriptController = m_globalScope->script();
122
123 if (m_runLoop.terminated()) {
124 // The worker was terminated before the thread had a chance to run. Since the context didn't exist yet,
125 // forbidExecution() couldn't be called from stop().
126 scriptController->scheduleExecutionTermination();
127 scriptController->forbidExecution();
128 }
129 }
130
131 if (shouldWaitForWebInspectorOnStartup()) {
132 startRunningDebuggerTasks();
133
134 // If the worker was somehow terminated while processing debugger commands.
135 if (m_runLoop.terminated())
136 scriptController->forbidExecution();
137 }
138
139 String exceptionMessage;
140 evaluateScriptIfNecessary(exceptionMessage);
141
142 callOnMainThread([evaluateCallback = WTFMove(m_evaluateCallback), message = exceptionMessage.isolatedCopy()] {
143 if (evaluateCallback)
144 evaluateCallback(message);
145 });
146
147 runEventLoop();
148
149#if USE(GLIB)
150 g_main_context_pop_thread_default(mainContext.get());
151#endif
152
153 RefPtr<Thread> protector = m_thread;
154
155 ASSERT(m_globalScope->hasOneRef());
156
157 RefPtr<WorkerOrWorkletGlobalScope> workerGlobalScopeToDelete;
158 {
159 // Mutex protection is necessary to ensure that we don't change m_globalScope
160 // while WorkerThread::stop is accessing it.
161 Locker locker { m_threadCreationAndGlobalScopeLock };
162
163 // Delay the destruction of the WorkerGlobalScope context until after we've unlocked the
164 // m_threadCreationAndWorkerGlobalScopeMutex. This is needed because destructing the
165 // context will trigger the main thread to race against us to delete the WorkerThread
166 // object, and the WorkerThread object owns the mutex we need to unlock after this.
167 workerGlobalScopeToDelete = std::exchange(m_globalScope, nullptr);
168
169 if (m_stoppedCallback)
170 callOnMainThread(WTFMove(m_stoppedCallback));
171 }
172
173 // The below assignment will destroy the context, which will in turn notify messaging proxy.
174 // We cannot let any objects survive past thread exit, because no other thread will run GC or otherwise destroy them.
175 workerGlobalScopeToDelete = nullptr;
176
177 // Clean up WebCore::ThreadGlobalData before WTF::Thread goes away!
178 threadGlobalData().destroy();
179
180 // Send the last WorkerThread Ref to be Deref'ed on the main thread.
181 callOnMainThread([protectedThis = WTFMove(protectedThis)] { });
182
183 // The thread object may be already destroyed from notification now, don't try to access "this".
184 protector->detach();
185}

187void WorkerOrWorkletThread::start(WTF::Function<void(const String&)>&& evaluateCallback)
188{
189 // Mutex protection is necessary to ensure that m_thread is initialized when the thread starts.
190 Locker locker { m_threadCreationAndGlobalScopeLock };
191
192 if (m_thread)
193 return;
194
195 m_evaluateCallback = WTFMove(evaluateCallback);
196
197 auto thread = createThread();
198
199 // Force the Thread object to be initialized fully before storing it to m_thread (and becoming visible to other threads).
200 WTF::storeStoreFence();
201
202 m_thread = WTFMove(thread);
203}

205void WorkerOrWorkletThread::stop(Function<void()>&& stoppedCallback)
206{
207 // Mutex protection is necessary to ensure that m_workerGlobalScope isn't changed by
208 // WorkerThread::workerThread() while we're accessing it. Note also that stop() can
209 // be called before m_workerGlobalScope is fully created.
210 if (!m_threadCreationAndGlobalScopeLock.tryLock()) {
211 // The thread is still starting, spin the runloop and try again to avoid deadlocks if the worker thread
212 // needs to interact with the main thread during startup.
213 callOnMainThread([this, stoppedCallback = WTFMove(stoppedCallback)]() mutable {
214 stop(WTFMove(stoppedCallback));
215 });
216 return;
217 }
218 Locker locker { AdoptLock, m_threadCreationAndGlobalScopeLock };
219
220 // If the thread is suspended, resume it now so that we can dispatch the cleanup tasks below.
221 if (m_isSuspended)
222 resume();
223
224 ASSERT(!m_stoppedCallback);
225 m_stoppedCallback = WTFMove(stoppedCallback);
226
227 // Ensure that tasks are being handled by thread event loop. If script execution weren't forbidden, a while(1) loop in JS could keep the thread alive forever.
228 if (globalScope()) {
229 globalScope()->script()->scheduleExecutionTermination();
230
231 m_runLoop.postTaskAndTerminate({ ScriptExecutionContext::Task::CleanupTask, [] (ScriptExecutionContext& context ) {
232 auto& globalScope = downcast<WorkerOrWorkletGlobalScope>(context);
233
234 globalScope.prepareForDestruction();
235
236 // Stick a shutdown command at the end of the queue, so that we deal
237 // with all the cleanup tasks the databases post first.
238 globalScope.postTask({ ScriptExecutionContext::Task::CleanupTask, [] (ScriptExecutionContext& context) {
239 auto& globalScope = downcast<WorkerOrWorkletGlobalScope>(context);
240 // It's not safe to call clearScript until all the cleanup tasks posted by functions initiated by WorkerThreadShutdownStartTask have completed.
241 globalScope.clearScript();
242 } });
243
244 } });
245 return;
246 }
247 m_runLoop.terminate();
248}

250void WorkerOrWorkletThread::suspend()
251{
252 m_isSuspended = true;
253 m_runLoop.postTask([&](ScriptExecutionContext&) {
254 if (globalScope())
255 globalScope()->suspend();
256
257 m_suspensionSemaphore.wait();
258
259 if (globalScope())
260 globalScope()->resume();
261 });
262}
263
264void WorkerOrWorkletThread::resume()
265{
266 ASSERT(m_isSuspended);
267 m_isSuspended = false;
268 m_suspensionSemaphore.signal();
269}

271void WorkerOrWorkletThread::releaseFastMallocFreeMemoryInAllThreads()
272{
273 Locker locker { workerOrWorkletThreadsLock() };
274 for (auto* workerOrWorkletThread : workerOrWorkletThreads()) {
275 workerOrWorkletThread->runLoop().postTask([] (ScriptExecutionContext&) {
276 WTF::releaseFastMallocFreeMemory();
277 });
278 }
