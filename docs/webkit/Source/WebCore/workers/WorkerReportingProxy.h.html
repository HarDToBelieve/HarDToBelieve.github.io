38
39 virtual void postExceptionToWorkerObject(const String& errorMessage, int lineNumber, int columnNumber, const String& sourceURL) = 0;
40
41 // Invoked when close() is invoked on the worker context.
42 virtual void workerGlobalScopeClosed() = 0;
43
44 // Invoked when the thread has stopped.
45 virtual void workerGlobalScopeDestroyed() = 0;
46};
