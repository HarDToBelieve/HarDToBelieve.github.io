54std::optional<Exception> WorkerScriptLoader::loadSynchronously(ScriptExecutionContext* scriptExecutionContext, const URL& url, FetchOptions::Mode mode, FetchOptions::Cache cachePolicy, ContentSecurityPolicyEnforcement contentSecurityPolicyEnforcement, const String& initiatorIdentifier)
55{
56 ASSERT(scriptExecutionContext);
57 auto& workerGlobalScope = downcast<WorkerGlobalScope>(*scriptExecutionContext);
58
59 m_url = url;
60 m_destination = FetchOptions::Destination::Script;
61
62#if ENABLE(SERVICE_WORKER)
63 bool isServiceWorkerGlobalScope = is<ServiceWorkerGlobalScope>(workerGlobalScope);
64
65 if (isServiceWorkerGlobalScope) {
66 if (auto* scriptResource = downcast<ServiceWorkerGlobalScope>(workerGlobalScope).scriptResource(url)) {
67 m_script = scriptResource->script;
68 m_responseURL = scriptResource->responseURL;
69 m_responseMIMEType = scriptResource->mimeType;
70 return std::nullopt;
71 }
72 }
73#endif
74
75 std::unique_ptr<ResourceRequest> request(createResourceRequest(initiatorIdentifier));
76 if (!request)
77 return std::nullopt;
78
79 ASSERT_WITH_SECURITY_IMPLICATION(is<WorkerGlobalScope>(scriptExecutionContext));
80
81 // Only used for importScripts that prescribes NoCors mode.
82 ASSERT(mode == FetchOptions::Mode::NoCors);
83 request->setRequester(ResourceRequest::Requester::ImportScripts);
84
85 ThreadableLoaderOptions options;
86 options.credentials = FetchOptions::Credentials::Include;
87 options.mode = mode;
88 options.cache = cachePolicy;
89 options.sendLoadCallbacks = SendCallbackPolicy::SendCallbacks;
90 options.contentSecurityPolicyEnforcement = contentSecurityPolicyEnforcement;
91 options.destination = m_destination;
92#if ENABLE(SERVICE_WORKER)
93 options.serviceWorkersMode = isServiceWorkerGlobalScope ? ServiceWorkersMode::None : ServiceWorkersMode::All;
94 if (auto* activeServiceWorker = workerGlobalScope.activeServiceWorker())
95 options.serviceWorkerRegistrationIdentifier = activeServiceWorker->registrationIdentifier();
96#endif
97 WorkerThreadableLoader::loadResourceSynchronously(workerGlobalScope, WTFMove(*request), *this, options);
98
99 // If the fetching attempt failed, throw a NetworkError exception and abort all these steps.
100 if (failed())
101 return Exception { NetworkError, error().localizedDescription() };
102
103#if ENABLE(SERVICE_WORKER)
104 if (isServiceWorkerGlobalScope) {
105 if (!MIMETypeRegistry::isSupportedJavaScriptMIMEType(responseMIMEType()))
106 return Exception { NetworkError, "mime type is not a supported JavaScript mime type"_s };
107
108 downcast<ServiceWorkerGlobalScope>(workerGlobalScope).setScriptResource(url, ServiceWorkerContextData::ImportedScript { script(), m_responseURL, m_responseMIMEType });
109 }
110#endif
111 return std::nullopt;
112}

114void WorkerScriptLoader::loadAsynchronously(ScriptExecutionContext& scriptExecutionContext, ResourceRequest&& scriptRequest, FetchOptions&& fetchOptions, ContentSecurityPolicyEnforcement contentSecurityPolicyEnforcement, ServiceWorkersMode serviceWorkerMode, WorkerScriptLoaderClient& client, String&& taskMode)
115{
116 m_client = &client;
117 m_url = scriptRequest.url();
118 m_destination = fetchOptions.destination;
119
120 ASSERT(scriptRequest.httpMethod() == "GET");
121
122 auto request = makeUnique<ResourceRequest>(WTFMove(scriptRequest));
123 if (!request)
124 return;
125
126 ThreadableLoaderOptions options { WTFMove(fetchOptions) };
127 options.sendLoadCallbacks = SendCallbackPolicy::SendCallbacks;
128 options.contentSecurityPolicyEnforcement = contentSecurityPolicyEnforcement;
129 if (fetchOptions.destination == FetchOptions::Destination::Serviceworker)
130 options.certificateInfoPolicy = CertificateInfoPolicy::IncludeCertificateInfo;
131
132 // FIXME: We should drop the sameOriginDataURLFlag flag and implement the latest Fetch specification.
133 if (fetchOptions.destination != FetchOptions::Destination::Worker)
134 options.sameOriginDataURLFlag = SameOriginDataURLFlag::Set;
135
136 // A service worker job can be executed from a worker context or a document context.
137 options.serviceWorkersMode = serviceWorkerMode;
138#if ENABLE(SERVICE_WORKER)
139 if (auto* activeServiceWorker = scriptExecutionContext.activeServiceWorker())
140 options.serviceWorkerRegistrationIdentifier = activeServiceWorker->registrationIdentifier();
141#endif
142
143 // During create, callbacks may happen which remove the last reference to this object.
144 Ref<WorkerScriptLoader> protectedThis(*this);
145 m_threadableLoader = ThreadableLoader::create(scriptExecutionContext, *this, WTFMove(*request), options, { }, WTFMove(taskMode));
146}

162ResourceError WorkerScriptLoader::validateWorkerResponse(const ResourceResponse& response, FetchOptions::Destination destination)
163{
164 if (response.httpStatusCode() / 100 != 2 && response.httpStatusCode())
165 return { errorDomainWebKitInternal, 0, response.url(), "Response is not 2xx"_s, ResourceError::Type::General };
166
167 if (!isScriptAllowedByNosniff(response)) {
168 auto message = makeString("Refused to execute ", response.url().stringCenterEllipsizedToLength(), " as script because \"X-Content-Type-Options: nosniff\" was given and its Content-Type is not a script MIME type.");
169 return { errorDomainWebKitInternal, 0, response.url(), WTFMove(message), ResourceError::Type::General };
170 }
171
172 if (shouldBlockResponseDueToMIMEType(response, destination)) {
173 auto message = makeString("Refused to execute ", response.url().stringCenterEllipsizedToLength(), " as script because ", response.mimeType(), " is not a script MIME type.");
174 return { errorDomainWebKitInternal, 0, response.url(), WTFMove(message), ResourceError::Type::General };
175 }
176
177 return { };
178}

180void WorkerScriptLoader::didReceiveResponse(unsigned long identifier, const ResourceResponse& response)
181{
182 m_error = validateWorkerResponse(response, m_destination);
183 if (!m_error.isNull()) {
184 m_failed = true;
185 return;
186 }
187
188 m_responseURL = response.url();
189 m_certificateInfo = response.certificateInfo() ? *response.certificateInfo() : CertificateInfo();
190 m_responseMIMEType = response.mimeType();
191 m_responseEncoding = response.textEncodingName();
192 m_responseSource = response.source();
193 m_isRedirected = response.isRedirected();
194 m_contentSecurityPolicy = ContentSecurityPolicyResponseHeaders { response };
195 m_referrerPolicy = response.httpHeaderField(HTTPHeaderName::ReferrerPolicy);
196 if (m_client)
197 m_client->didReceiveResponse(identifier, response);
198}

200void WorkerScriptLoader::didReceiveData(const uint8_t* data, int len)
201{
202 if (m_failed)
203 return;
204
205 if (!m_decoder) {
206 if (!m_responseEncoding.isEmpty())
207 m_decoder = TextResourceDecoder::create("text/javascript"_s, m_responseEncoding);
208 else
209 m_decoder = TextResourceDecoder::create("text/javascript"_s, "UTF-8");
210 }
211
212 if (!len)
213 return;
214
215 if (len == -1)
216 len = strlen(reinterpret_cast<const char*>(data));
217
218 m_script.append(m_decoder->decode(data, len));
219}
220
221void WorkerScriptLoader::didFinishLoading(unsigned long identifier)
222{
223 if (m_failed) {
224 notifyError();
225 return;
226 }
227
228 if (m_decoder)
229 m_script.append(m_decoder->flush());
230
231 m_identifier = identifier;
232 notifyFinished();
233}
234
235void WorkerScriptLoader::didFail(const ResourceError& error)
236{
237 m_error = error;
238 notifyError();
239}
240
241void WorkerScriptLoader::notifyError()
242{
243 m_failed = true;
244 if (m_error.isNull())
245 m_error = { errorDomainWebKitInternal, 0, url(), "Failed to load script", ResourceError::Type::General };
246 notifyFinished();
247}
248
249void WorkerScriptLoader::notifyFinished()
250{
251 m_threadableLoader = nullptr;
252 if (!m_client || m_finishing)
253 return;
254
255 m_finishing = true;
256 m_client->notifyFinished();
257}
258
259void WorkerScriptLoader::cancel()
260{
261 if (!m_threadableLoader)
262 return;
263
264 m_client = nullptr;
265 m_threadableLoader->cancel();
266 m_threadableLoader = nullptr;
267}
