87WorkerThread::WorkerThread(const WorkerParameters& params, const ScriptBuffer& sourceCode, WorkerLoaderProxy& workerLoaderProxy, WorkerDebuggerProxy& workerDebuggerProxy, WorkerReportingProxy& workerReportingProxy, WorkerThreadStartMode startMode, const SecurityOrigin& topOrigin, IDBClient::IDBConnectionProxy* connectionProxy, SocketProvider* socketProvider, JSC::RuntimeFlags runtimeFlags)
88 : WorkerOrWorkletThread(params.identifier.isolatedCopy())
89 , m_workerLoaderProxy(workerLoaderProxy)
90 , m_workerDebuggerProxy(workerDebuggerProxy)
91 , m_workerReportingProxy(workerReportingProxy)
92 , m_runtimeFlags(runtimeFlags)
93 , m_startupData(makeUnique<WorkerThreadStartupData>(params, sourceCode, startMode, topOrigin))
94 , m_idbConnectionProxy(connectionProxy)
95 , m_socketProvider(socketProvider)
96{
97 ++workerThreadCounter;
98}

100WorkerThread::~WorkerThread()
101{

123void WorkerThread::evaluateScriptIfNecessary(String& exceptionMessage)
124{
125 // We are currently holding only the initial script code. If the WorkerType is Module, we should fetch the entire graph before executing the rest of this.
126 // We invoke module loader as if we are executing inline module script tag in Document.
127
128 WeakPtr<ScriptBufferSourceProvider> sourceProvider;
129 if (m_startupData->params.workerType == WorkerType::Classic) {
130 ScriptSourceCode sourceCode(m_startupData->sourceCode, URL(m_startupData->params.scriptURL));
131 sourceProvider = makeWeakPtr(static_cast<ScriptBufferSourceProvider&>(sourceCode.provider()));
132 globalScope()->script()->evaluate(sourceCode, &exceptionMessage);
133 finishedEvaluatingScript();
134 } else {
135 auto scriptFetcher = WorkerScriptFetcher::create(globalScope()->credentials(), globalScope()->destination(), globalScope()->referrerPolicy());
136 ScriptSourceCode sourceCode(m_startupData->sourceCode, URL(m_startupData->params.scriptURL), { }, JSC::SourceProviderSourceType::Module, scriptFetcher.copyRef());
137 sourceProvider = makeWeakPtr(static_cast<ScriptBufferSourceProvider&>(sourceCode.provider()));
138 MessageQueueWaitResult result = globalScope()->script()->loadModuleSynchronously(scriptFetcher.get(), sourceCode);
139 if (result != MessageQueueTerminated) {
140 if (std::optional<LoadableScript::Error> error = scriptFetcher->error()) {
141 if (std::optional<LoadableScript::ConsoleMessage> message = error->consoleMessage)
142 exceptionMessage = message->message;
143 else
144 exceptionMessage = "Importing a module script failed."_s;
145 globalScope()->reportException(exceptionMessage, { }, { }, { }, { }, { });
146 } else if (!scriptFetcher->wasCanceled()) {
147 globalScope()->script()->linkAndEvaluateModule(scriptFetcher.get(), sourceCode, &exceptionMessage);
148 finishedEvaluatingScript();
149 }
150 }
151 }
152 if (sourceProvider)
153 globalScope()->setMainScriptSourceProvider(*sourceProvider);
154
155 // Free the startup data to cause its member variable deref's happen on the worker's thread (since
156 // all ref/derefs of these objects are happening on the thread at this point). Note that
157 // WorkerThread::~WorkerThread happens on a different thread where it was created.
158 m_startupData = nullptr;
159}
