56FetchEvent::~FetchEvent()
57{
58 if (auto callback = WTFMove(m_onResponse)) {
59 RELEASE_LOG_ERROR_IF(m_respondWithEntered, ServiceWorker, "Fetch event is destroyed without a response, respondWithEntered=%d, waitToRespond=%d, respondWithError=%d, respondPromise=%d", m_respondWithEntered, m_waitToRespond, m_respondWithError, !!m_respondPromise);
60 callback(makeUnexpected(std::optional<ResourceError> { }));
61 }
62}

64ResourceError FetchEvent::createResponseError(const URL& url, const String& errorMessage)
65{
66 return ResourceError { errorDomainWebKitServiceWorker, 0, url, makeString("FetchEvent.respondWith received an error: ", errorMessage), ResourceError::Type::General };

68}

70ExceptionOr<void> FetchEvent::respondWith(Ref<DOMPromise>&& promise)
71{
72 if (!isBeingDispatched())
73 return Exception { InvalidStateError, "Event is not being dispatched"_s };
74
75 if (m_respondWithEntered)
76 return Exception { InvalidStateError, "Event respondWith flag is set"_s };
77
78 m_respondPromise = WTFMove(promise);
79 addExtendLifetimePromise(*m_respondPromise);
80
81 auto isRegistered = m_respondPromise->whenSettled([this, protectedThis = makeRef(*this)] {
82 promiseIsSettled();
83 });
84
85 stopPropagation();
86 stopImmediatePropagation();
87
88 m_respondWithEntered = true;
89 m_waitToRespond = true;
90
91 if (isRegistered == DOMPromise::IsCallbackRegistered::No)
92 respondWithError(createResponseError(m_request->url(), "FetchEvent unable to handle respondWith promise."_s));
93
94 return { };
95}

97void FetchEvent::onResponse(ResponseCallback&& callback)
98{
99 ASSERT(!m_onResponse);
100 m_onResponse = WTFMove(callback);
101}
102
103void FetchEvent::respondWithError(ResourceError&& error)
104{
105 m_respondWithError = true;
106 processResponse(makeUnexpected(WTFMove(error)));
107}
108
109void FetchEvent::processResponse(Expected<Ref<FetchResponse>, std::optional<ResourceError>>&& result)
110{
111 m_respondPromise = nullptr;
112 m_waitToRespond = false;
113 if (auto callback = WTFMove(m_onResponse))
114 callback(WTFMove(result));
115}
116
117void FetchEvent::promiseIsSettled()
118{
119 if (m_respondPromise->status() == DOMPromise::Status::Rejected) {
120 auto reason = m_respondPromise->result().toWTFString(m_respondPromise->globalObject());
121 respondWithError(createResponseError(m_request->url(), reason));
122 return;
123 }
124
125 ASSERT(m_respondPromise->status() == DOMPromise::Status::Fulfilled);
126 auto response = JSFetchResponse::toWrapped(m_respondPromise->globalObject()->vm(), m_respondPromise->result());
127 if (!response) {
128 respondWithError(createResponseError(m_request->url(), "Returned response is null."_s));
129 return;
130 }
131
132 if (response->isDisturbedOrLocked()) {
133 respondWithError(createResponseError(m_request->url(), "Response is disturbed or locked."_s));
134 return;
135 }
136
137 processResponse(makeRef(*response));
138}
