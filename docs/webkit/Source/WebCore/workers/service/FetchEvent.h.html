26#pragma once
27
28#if ENABLE(SERVICE_WORKER)
29
30#include "ExtendableEvent.h"
31#include "FetchRequest.h"
32#include <wtf/CompletionHandler.h>
33#include <wtf/Expected.h>
34
35namespace WebCore {
36
37class FetchResponse;
38class ResourceError;
39
40class FetchEvent final : public ExtendableEvent {
41 WTF_MAKE_ISO_ALLOCATED(FetchEvent);
42public:
43 struct Init : ExtendableEventInit {
44 RefPtr<FetchRequest> request;
45 String clientId;
46 String reservedClientId;
47 String targetClientId;
48 };
49
50 WEBCORE_EXPORT static Ref<FetchEvent> createForTesting(ScriptExecutionContext&);
51
52 static Ref<FetchEvent> create(const AtomString& type, Init&& initializer, IsTrusted isTrusted = IsTrusted::No)
53 {
54 return adoptRef(*new FetchEvent(type, WTFMove(initializer), isTrusted));
55 }
56 ~FetchEvent();
57
58 EventInterface eventInterface() const final { return FetchEventInterfaceType; }
59
60 ExceptionOr<void> respondWith(Ref<DOMPromise>&&);
61
62 using ResponseCallback = CompletionHandler<void(Expected<Ref<FetchResponse>, std::optional<ResourceError>>&&)>;
63 WEBCORE_EXPORT void onResponse(ResponseCallback&&);
64
65 FetchRequest& request() { return m_request.get(); }
66 const String& clientId() const { return m_clientId; }
67 const String& reservedClientId() const { return m_reservedClientId; }
68 const String& targetClientId() const { return m_targetClientId; }
69
70 bool respondWithEntered() const { return m_respondWithEntered; }
71
72 static ResourceError createResponseError(const URL&, const String&);
73
74private:
75 WEBCORE_EXPORT FetchEvent(const AtomString&, Init&&, IsTrusted);
76
77 void promiseIsSettled();
78 void processResponse(Expected<Ref<FetchResponse>, std::optional<ResourceError>>&&);
79 void respondWithError(ResourceError&&);
80
81 Ref<FetchRequest> m_request;
82 String m_clientId;
83 String m_reservedClientId;
84 String m_targetClientId;
85
86 bool m_respondWithEntered { false };
87 bool m_waitToRespond { false };
88 bool m_respondWithError { false };
89 RefPtr<DOMPromise> m_respondPromise;
90
91 ResponseCallback m_onResponse;
92};
93
94} // namespace WebCore
95
96#endif // ENABLE(SERVICE_WORKER)
