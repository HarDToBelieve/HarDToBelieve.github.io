39static ServiceWorkerClientFrameType toServiceWorkerClientFrameType(ScriptExecutionContext& context)
40{
41 if (!is<Document>(context))
42 return ServiceWorkerClientFrameType::None;
43
44 auto& document = downcast<Document>(context);
45 auto* frame = document.frame();
46 if (!frame)
47 return ServiceWorkerClientFrameType::None;
48
49 if (frame->isMainFrame()) {
50 if (auto* window = document.domWindow()) {
51 if (window->opener())
52 return ServiceWorkerClientFrameType::Auxiliary;
53 }
54 return ServiceWorkerClientFrameType::TopLevel;
55 }
56 return ServiceWorkerClientFrameType::Nested;
57}
58
59ServiceWorkerClientData ServiceWorkerClientData::isolatedCopy() const
60{
61 return { identifier, type, frameType, url.isolatedCopy(), lastNavigationWasAppBound };
62}
63
64ServiceWorkerClientData ServiceWorkerClientData::from(ScriptExecutionContext& context, SWClientConnection& connection)
65{
66 bool isDocument = is<Document>(context);
67 RELEASE_ASSERT(isDocument); // We do not support dedicated workers as clients yet.
68
69 auto& document = downcast<Document>(context);
70 auto lastNavigationWasAppBound = document.loader() && document.loader()->lastNavigationWasAppBound() ? LastNavigationWasAppBound::Yes : LastNavigationWasAppBound::No;
71
72 return {
73 { connection.serverConnectionIdentifier(), document.identifier() },
74 isDocument ? ServiceWorkerClientType::Window : ServiceWorkerClientType::Worker,
75 toServiceWorkerClientFrameType(context),
76 context.url(), lastNavigationWasAppBound
77 };
78}
