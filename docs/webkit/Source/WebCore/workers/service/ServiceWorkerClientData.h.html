30#include "ServiceWorkerClientIdentifier.h"

32#include "ServiceWorkerTypes.h"
33#include <wtf/URL.h>

35namespace WebCore {
36
37class SWClientConnection;
38class ScriptExecutionContext;
39
40enum class LastNavigationWasAppBound : bool { No, Yes };
41
42struct ServiceWorkerClientData {
43 ServiceWorkerClientIdentifier identifier;
44 ServiceWorkerClientType type;
45 ServiceWorkerClientFrameType frameType;
46 URL url;
47 LastNavigationWasAppBound lastNavigationWasAppBound;
48
49 ServiceWorkerClientData isolatedCopy() const;
50
51 static ServiceWorkerClientData from(ScriptExecutionContext&, SWClientConnection&);
52
53 template<class Encoder> void encode(Encoder&) const;
54 template<class Decoder> static std::optional<ServiceWorkerClientData> decode(Decoder&);
55};
56
57template<class Encoder>
58void ServiceWorkerClientData::encode(Encoder& encoder) const
59{
60 encoder << identifier << type << frameType << url << lastNavigationWasAppBound;
61}
62
63template<class Decoder>
64std::optional<ServiceWorkerClientData> ServiceWorkerClientData::decode(Decoder& decoder)
65{
66 std::optional<ServiceWorkerClientIdentifier> identifier;
67 decoder >> identifier;
68 if (!identifier)
69 return std::nullopt;
70
71 std::optional<ServiceWorkerClientType> type;
72 decoder >> type;
73 if (!type)
74 return std::nullopt;
75
76 std::optional<ServiceWorkerClientFrameType> frameType;
77 decoder >> frameType;
78 if (!frameType)
79 return std::nullopt;
80
81 std::optional<URL> url;
82 decoder >> url;
83 if (!url)
84 return std::nullopt;
85
86 std::optional<LastNavigationWasAppBound> lastNavigationWasAppBound;
87 decoder >> lastNavigationWasAppBound;
88 if (!lastNavigationWasAppBound)
89 return std::nullopt;
90
91 return { { WTFMove(*identifier), WTFMove(*type), WTFMove(*frameType), WTFMove(*url), WTFMove(*lastNavigationWasAppBound) } };
92}
93
94using ServiceWorkerClientsMatchAllCallback = WTF::CompletionHandler<void(Vector<ServiceWorkerClientData>&&)>;

96} // namespace WebCore
