26#pragma once
27
28#if ENABLE(SERVICE_WORKER)
29
30#include "ServiceWorkerClientType.h"
31#include <optional>
32
33namespace WebCore {
34
35struct ServiceWorkerClientQueryOptions {
36 bool includeUncontrolled { false };
37 ServiceWorkerClientType type { ServiceWorkerClientType::Window };
38
39 template<class Encoder> void encode(Encoder&) const;
40 template<class Decoder> static std::optional<ServiceWorkerClientQueryOptions> decode(Decoder&);
41};
42
43template<class Encoder>
44void ServiceWorkerClientQueryOptions::encode(Encoder& encoder) const
45{
46 encoder << includeUncontrolled << type;
47}
48
49template<class Decoder>
50std::optional<ServiceWorkerClientQueryOptions> ServiceWorkerClientQueryOptions::decode(Decoder& decoder)
51{
52 std::optional<bool> includeUncontrolled;
53 decoder >> includeUncontrolled;
54 if (!includeUncontrolled)
55 return std::nullopt;
56
57 std::optional<ServiceWorkerClientType> type;
58 decoder >> type;
59 if (!type)
60 return std::nullopt;
61
62 return { { *includeUncontrolled, *type } };
63}
64
65} // namespace WebCore

67#endif // ENABLE(SERVICE_WORKER)
