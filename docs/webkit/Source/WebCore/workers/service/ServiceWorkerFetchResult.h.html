36namespace WebCore {
37
38struct ServiceWorkerFetchResult {
39 ServiceWorkerJobDataIdentifier jobDataIdentifier;
40 ServiceWorkerRegistrationKey registrationKey;
41 ScriptBuffer script;
42 CertificateInfo certificateInfo;
43 ContentSecurityPolicyResponseHeaders contentSecurityPolicy;
44 String referrerPolicy;
45 ResourceError scriptError;
46
47 ServiceWorkerFetchResult isolatedCopy() const { return { jobDataIdentifier, registrationKey.isolatedCopy(), script.isolatedCopy(), certificateInfo.isolatedCopy(), contentSecurityPolicy.isolatedCopy(), referrerPolicy.isolatedCopy(), scriptError.isolatedCopy() }; }
48
49 template<class Encoder> void encode(Encoder&) const;
50 template<class Decoder> static WARN_UNUSED_RETURN bool decode(Decoder&, ServiceWorkerFetchResult&);
51};
52
53inline ServiceWorkerFetchResult serviceWorkerFetchError(ServiceWorkerJobDataIdentifier jobDataIdentifier, ServiceWorkerRegistrationKey&& registrationKey, ResourceError&& error)
54{
55 return { jobDataIdentifier, WTFMove(registrationKey), { }, { }, { }, { }, WTFMove(error) };
56}
57
58template<class Encoder>
59void ServiceWorkerFetchResult::encode(Encoder& encoder) const
60{
61 encoder << jobDataIdentifier << registrationKey << script << contentSecurityPolicy << referrerPolicy << scriptError;
62 encoder << certificateInfo;
63}
64
65template<class Decoder>
66bool ServiceWorkerFetchResult::decode(Decoder& decoder, ServiceWorkerFetchResult& result)
67{
68 std::optional<ServiceWorkerJobDataIdentifier> jobDataIdentifier;
69 decoder >> jobDataIdentifier;
70 if (!jobDataIdentifier)
71 return false;
72 result.jobDataIdentifier = WTFMove(*jobDataIdentifier);
73
74 auto registrationKey = ServiceWorkerRegistrationKey::decode(decoder);
75 if (!registrationKey)
76 return false;
77 std::swap(*registrationKey, result.registrationKey);
78
79 if (!decoder.decode(result.script))
80 return false;
81 if (!decoder.decode(result.contentSecurityPolicy))
82 return false;
83 if (!decoder.decode(result.referrerPolicy))
84 return false;
85 if (!decoder.decode(result.scriptError))
86 return false;
87
88 std::optional<CertificateInfo> certificateInfo;
89 decoder >> certificateInfo;
90 if (!certificateInfo)
91 return false;
92 result.certificateInfo = WTFMove(*certificateInfo);
93
94 return true;
95}
