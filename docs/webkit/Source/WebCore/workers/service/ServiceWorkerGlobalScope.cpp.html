64void ServiceWorkerGlobalScope::skipWaiting(Ref<DeferredPromise>&& promise)
65{
66 uint64_t requestIdentifier = ++m_lastRequestIdentifier;
67 m_pendingSkipWaitingPromises.add(requestIdentifier, WTFMove(promise));
68
69 callOnMainThread([workerThread = makeRef(thread()), requestIdentifier]() mutable {
70 if (auto* connection = SWContextManager::singleton().connection()) {
71 auto identifier = workerThread->identifier();
72 connection->skipWaiting(identifier, [workerThread = WTFMove(workerThread), requestIdentifier] {
73 workerThread->runLoop().postTask([requestIdentifier](auto& context) {
74 auto& scope = downcast<ServiceWorkerGlobalScope>(context);
75 scope.eventLoop().queueTask(TaskSource::DOMManipulation, [scope = makeRef(scope), requestIdentifier]() mutable {
76 if (auto promise = scope->m_pendingSkipWaitingPromises.take(requestIdentifier))
77 promise->resolve();
78 });
79 });
80 });
81 }
82 });
83}

112// https://w3c.github.io/ServiceWorker/#update-service-worker-extended-events-set-algorithm
113void ServiceWorkerGlobalScope::updateExtendedEventsSet(ExtendableEvent* newEvent)
114{
115 ASSERT(!isMainThread());
116 ASSERT(!newEvent || !newEvent->isBeingDispatched());
117 bool hadPendingEvents = hasPendingEvents();
118 m_extendedEvents.removeAllMatching([](auto& event) {
119 return !event->pendingPromiseCount();
120 });
121
122 if (newEvent && newEvent->pendingPromiseCount()) {
123 m_extendedEvents.append(*newEvent);
124 newEvent->whenAllExtendLifetimePromisesAreSettled([this](auto&&) {
125 this->updateExtendedEventsSet();
126 });
127 // Clear out the event's target as it is the WorkerGlobalScope and we do not want to keep it
128 // alive unnecessarily.
129 newEvent->setTarget(nullptr);
130 }
131
132 bool hasPendingEvents = this->hasPendingEvents();
133 if (hasPendingEvents == hadPendingEvents)
134 return;
135
136 callOnMainThread([threadIdentifier = thread().identifier(), hasPendingEvents] {
137 if (auto* connection = SWContextManager::singleton().connection())
138 connection->setServiceWorkerHasPendingEvents(threadIdentifier, hasPendingEvents);
139 });
140}

142const ServiceWorkerContextData::ImportedScript* ServiceWorkerGlobalScope::scriptResource(const URL& url) const
143{
144 auto iterator = m_contextData.scriptResourceMap.find(url);
145 return iterator == m_contextData.scriptResourceMap.end() ? nullptr : &iterator->value;
146}
147
148void ServiceWorkerGlobalScope::setScriptResource(const URL& url, ServiceWorkerContextData::ImportedScript&& script)
149{
150 callOnMainThread([threadIdentifier = thread().identifier(), url = url.isolatedCopy(), script = script.isolatedCopy()] {
151 if (auto* connection = SWContextManager::singleton().connection())
152 connection->setScriptResource(threadIdentifier, url, script);
153 });
154
155 m_contextData.scriptResourceMap.set(url, WTFMove(script));
156}

158void ServiceWorkerGlobalScope::didSaveScriptsToDisk(ScriptBuffer&& script, HashMap<URL, ScriptBuffer>&& importedScripts)
159{
160 // These scripts should be identical to the ones we have. However, these are mmap'd so using them helps reduce dirty memory usage.
161 updateSourceProviderBuffers(script, importedScripts);
162
163 if (script) {
164 ASSERT(m_contextData.script == script);
165 m_contextData.script = WTFMove(script);
166 }
167 for (auto& pair : importedScripts) {
168 auto it = m_contextData.scriptResourceMap.find(pair.key);
169 if (it == m_contextData.scriptResourceMap.end())
170 continue;
171 ASSERT(it->value.script == pair.value); // Do a memcmp to make sure the scripts are identical.
172 it->value.script = WTFMove(pair.value);
173 }
174}
