26#pragma once
27
28#if ENABLE(SERVICE_WORKER)
29
30#include "ServiceWorkerClientIdentifier.h"
31#include "ServiceWorkerContextData.h"
32#include "ServiceWorkerRegistration.h"
33#include "WorkerGlobalScope.h"
34#include <wtf/URLHash.h>
35
36namespace WebCore {
37
38class DeferredPromise;
39class ExtendableEvent;
40struct ServiceWorkerClientData;
41class ServiceWorkerClient;
42class ServiceWorkerClients;
43class ServiceWorkerThread;
44
45class ServiceWorkerGlobalScope final : public WorkerGlobalScope {
46 WTF_MAKE_ISO_ALLOCATED(ServiceWorkerGlobalScope);
47public:
48 static Ref<ServiceWorkerGlobalScope> create(ServiceWorkerContextData&&, const WorkerParameters&, Ref<SecurityOrigin>&&, ServiceWorkerThread&, Ref<SecurityOrigin>&& topOrigin, IDBClient::IDBConnectionProxy*, SocketProvider*);
49
50 ~ServiceWorkerGlobalScope();
51
52 bool isServiceWorkerGlobalScope() const final { return true; }
53
54 ServiceWorkerClients& clients() { return m_clients.get(); }
55 ServiceWorkerRegistration& registration() { return m_registration.get(); }
56
57 void skipWaiting(Ref<DeferredPromise>&&);
58
59 EventTargetInterface eventTargetInterface() const final;
60
61 ServiceWorkerThread& thread();
62
63 ServiceWorkerClient* serviceWorkerClient(ServiceWorkerClientIdentifier);
64 void addServiceWorkerClient(ServiceWorkerClient&);
65 void removeServiceWorkerClient(ServiceWorkerClient&);
66
67 void updateExtendedEventsSet(ExtendableEvent* newEvent = nullptr);
68
69 const ServiceWorkerContextData::ImportedScript* scriptResource(const URL&) const;
70 void setScriptResource(const URL&, ServiceWorkerContextData::ImportedScript&&);
71
72 void didSaveScriptsToDisk(ScriptBuffer&&, HashMap<URL, ScriptBuffer>&& importedScripts);
73
74 const ServiceWorkerContextData& contextData() const { return m_contextData; }
75 const CertificateInfo& certificateInfo() const { return m_contextData.certificateInfo; }
76
77 FetchOptions::Destination destination() const final { return FetchOptions::Destination::Serviceworker; }
78
79private:
80 ServiceWorkerGlobalScope(ServiceWorkerContextData&&, const WorkerParameters&, Ref<SecurityOrigin>&&, ServiceWorkerThread&, Ref<SecurityOrigin>&& topOrigin, IDBClient::IDBConnectionProxy*, SocketProvider*);
81
82 bool hasPendingEvents() const { return !m_extendedEvents.isEmpty(); }
83
84 ServiceWorkerContextData m_contextData;
85 Ref<ServiceWorkerRegistration> m_registration;
86 Ref<ServiceWorkerClients> m_clients;
87 HashMap<ServiceWorkerClientIdentifier, ServiceWorkerClient*> m_clientMap;
88 Vector<Ref<ExtendableEvent>> m_extendedEvents;
89
90 uint64_t m_lastRequestIdentifier { 0 };
91 HashMap<uint64_t, RefPtr<DeferredPromise>> m_pendingSkipWaitingPromises;
92};
93
94} // namespace WebCore
95
96SPECIALIZE_TYPE_TRAITS_BEGIN(WebCore::ServiceWorkerGlobalScope)
97 static bool isType(const WebCore::ScriptExecutionContext& context) { return is<WebCore::WorkerGlobalScope>(context) && downcast<WebCore::WorkerGlobalScope>(context).isServiceWorkerGlobalScope(); }
98 static bool isType(const WebCore::WorkerGlobalScope& context) { return context.isServiceWorkerGlobalScope(); }
99SPECIALIZE_TYPE_TRAITS_END()

101#endif // ENABLE(SERVICE_WORKER)
