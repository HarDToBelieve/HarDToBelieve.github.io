36class Exception;

41
42class ServiceWorkerJobClient {
43public:
44 virtual ~ServiceWorkerJobClient() = default;
45
46 virtual DocumentOrWorkerIdentifier contextIdentifier() = 0;
47
48 virtual void jobFailedWithException(ServiceWorkerJob&, const Exception&) = 0;

50 virtual void jobResolvedWithUnregistrationResult(ServiceWorkerJob&, bool unregistrationResult) = 0;
51 virtual void startScriptFetchForJob(ServiceWorkerJob&, FetchOptions::Cache) = 0;
52 virtual void jobFinishedLoadingScript(ServiceWorkerJob&, const ScriptBuffer&, const CertificateInfo&, const ContentSecurityPolicyResponseHeaders&, const String& referrerPolicy) = 0;

54};
