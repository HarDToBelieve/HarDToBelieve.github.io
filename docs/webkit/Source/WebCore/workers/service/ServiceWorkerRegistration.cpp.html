49Ref<ServiceWorkerRegistration> ServiceWorkerRegistration::getOrCreate(ScriptExecutionContext& context, Ref<ServiceWorkerContainer>&& container, ServiceWorkerRegistrationData&& data)
50{
51 if (auto* registration = container->registration(data.identifier)) {
52 ASSERT(!registration->isContextStopped());
53 return *registration;
54 }
55
56 return adoptRef(*new ServiceWorkerRegistration(context, WTFMove(container), WTFMove(data)));
57}
58
59ServiceWorkerRegistration::ServiceWorkerRegistration(ScriptExecutionContext& context, Ref<ServiceWorkerContainer>&& container, ServiceWorkerRegistrationData&& registrationData)
60 : ActiveDOMObject(&context)
61 , m_registrationData(WTFMove(registrationData))
62 , m_container(WTFMove(container))
63{
64 LOG(ServiceWorker, "Creating registration %p for registration key %s", this, m_registrationData.key.loggingString().utf8().data());
65 suspendIfNeeded();
66
67 if (m_registrationData.installingWorker)
68 m_installingWorker = ServiceWorker::getOrCreate(context, WTFMove(*m_registrationData.installingWorker));
69 if (m_registrationData.waitingWorker)
70 m_waitingWorker = ServiceWorker::getOrCreate(context, WTFMove(*m_registrationData.waitingWorker));
71 if (m_registrationData.activeWorker)
72 m_activeWorker = ServiceWorker::getOrCreate(context, WTFMove(*m_registrationData.activeWorker));
73
74 REGISTRATION_RELEASE_LOG("ServiceWorkerRegistration: ID %llu, installing=%llu, waiting=%llu, active=%llu", identifier().toUInt64(), m_installingWorker ? m_installingWorker->identifier().toUInt64() : 0, m_waitingWorker ? m_waitingWorker->identifier().toUInt64() : 0, m_activeWorker ? m_activeWorker->identifier().toUInt64() : 0);
75
76 m_container->addRegistration(*this);
77
78 relaxAdoptionRequirement();
79}

81ServiceWorkerRegistration::~ServiceWorkerRegistration()
82{
83 LOG(ServiceWorker, "Deleting registration %p for registration key %s", this, m_registrationData.key.loggingString().utf8().data());
84
85 m_container->removeRegistration(*this);
86}

138void ServiceWorkerRegistration::update(Ref<DeferredPromise>&& promise)
139{
140 if (isContextStopped()) {
141 promise->reject(Exception(InvalidStateError));
142 return;
143 }
144
145 auto* newestWorker = getNewestWorker();
146 if (!newestWorker) {
147 promise->reject(Exception(InvalidStateError, "newestWorker is null"_s));
148 return;
149 }
150
151 m_container->updateRegistration(m_registrationData.scopeURL, newestWorker->scriptURL(), newestWorker->workerType(), WTFMove(promise));
152}
153
154void ServiceWorkerRegistration::unregister(Ref<DeferredPromise>&& promise)
155{
156 if (isContextStopped()) {
157 promise->reject(Exception(InvalidStateError));
158 return;
159 }
160
161 m_container->unregisterRegistration(identifier(), WTFMove(promise));
162}

164void ServiceWorkerRegistration::updateStateFromServer(ServiceWorkerRegistrationState state, RefPtr<ServiceWorker>&& serviceWorker)
165{
166 switch (state) {
167 case ServiceWorkerRegistrationState::Installing:
168 REGISTRATION_RELEASE_LOG("updateStateFromServer: Setting registration %llu installing worker to %llu", identifier().toUInt64(), serviceWorker ? serviceWorker->identifier().toUInt64() : 0);
169 m_installingWorker = WTFMove(serviceWorker);
170 break;
171 case ServiceWorkerRegistrationState::Waiting:
172 REGISTRATION_RELEASE_LOG("updateStateFromServer: Setting registration %llu waiting worker to %llu", identifier().toUInt64(), serviceWorker ? serviceWorker->identifier().toUInt64() : 0);
173 m_waitingWorker = WTFMove(serviceWorker);
174 break;
175 case ServiceWorkerRegistrationState::Active:
176 REGISTRATION_RELEASE_LOG("updateStateFromServer: Setting registration %llu active worker to %llu", identifier().toUInt64(), serviceWorker ? serviceWorker->identifier().toUInt64() : 0);
177 m_activeWorker = WTFMove(serviceWorker);
178 break;
179 }
180}

182void ServiceWorkerRegistration::queueTaskToFireUpdateFoundEvent()
183{
184 if (isContextStopped())
185 return;
186
187 REGISTRATION_RELEASE_LOG("fireUpdateFoundEvent: Firing updatefound event for registration %llu", identifier().toUInt64());
188
189 queueTaskToDispatchEvent(*this, TaskSource::DOMManipulation, Event::create(eventNames().updatefoundEvent, Event::CanBubble::No, Event::IsCancelable::No));
190}
