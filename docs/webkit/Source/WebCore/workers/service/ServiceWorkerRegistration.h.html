36namespace WebCore {
37
38class DeferredPromise;
39class ScriptExecutionContext;
40class ServiceWorker;
41class ServiceWorkerContainer;
42
43class ServiceWorkerRegistration final : public RefCounted<ServiceWorkerRegistration>, public EventTargetWithInlineData, public ActiveDOMObject {
44 WTF_MAKE_ISO_ALLOCATED(ServiceWorkerRegistration);
45public:
46 static Ref<ServiceWorkerRegistration> getOrCreate(ScriptExecutionContext&, Ref<ServiceWorkerContainer>&&, ServiceWorkerRegistrationData&&);
47
48 ~ServiceWorkerRegistration();
49
50 ServiceWorkerRegistrationIdentifier identifier() const { return m_registrationData.identifier; }
51
52 ServiceWorker* installing();
53 ServiceWorker* waiting();
54 ServiceWorker* active();
55
56 ServiceWorker* getNewestWorker() const;
57
58 const String& scope() const;
59
60 ServiceWorkerUpdateViaCache updateViaCache() const;
61 void setUpdateViaCache(ServiceWorkerUpdateViaCache);
62
63 WallTime lastUpdateTime() const;
64 void setLastUpdateTime(WallTime);
65
66 bool needsUpdate() const { return lastUpdateTime() && (WallTime::now() - lastUpdateTime()) > 86400_s; }
67
68 void update(Ref<DeferredPromise>&&);
69 void unregister(Ref<DeferredPromise>&&);
70
71 using RefCounted::ref;
72 using RefCounted::deref;
73
74 const ServiceWorkerRegistrationData& data() const { return m_registrationData; }
75
76 void updateStateFromServer(ServiceWorkerRegistrationState, RefPtr<ServiceWorker>&&);
77 void queueTaskToFireUpdateFoundEvent();
78
79private:
80 ServiceWorkerRegistration(ScriptExecutionContext&, Ref<ServiceWorkerContainer>&&, ServiceWorkerRegistrationData&&);
81
82 EventTargetInterface eventTargetInterface() const final;
83 ScriptExecutionContext* scriptExecutionContext() const final;
84 void refEventTarget() final { ref(); }
85 void derefEventTarget() final { deref(); }
86
87 // ActiveDOMObject.
88 const char* activeDOMObjectName() const final;
89 void stop() final;
90 bool virtualHasPendingActivity() const final;
91
92 ServiceWorkerRegistrationData m_registrationData;
93 Ref<ServiceWorkerContainer> m_container;
94
95 RefPtr<ServiceWorker> m_installingWorker;
96 RefPtr<ServiceWorker> m_waitingWorker;
97 RefPtr<ServiceWorker> m_activeWorker;
98};

100} // namespace WebCore
