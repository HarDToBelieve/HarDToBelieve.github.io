37ServiceWorkerRegistrationKey::ServiceWorkerRegistrationKey(SecurityOriginData&& topOrigin, URL&& scope)
38 : m_topOrigin(WTFMove(topOrigin))
39 , m_scope(WTFMove(scope))
40{
41 ASSERT(!m_scope.hasFragmentIdentifier());
42}
43
44ServiceWorkerRegistrationKey ServiceWorkerRegistrationKey::emptyKey()
45{
46 return { };
47}

49unsigned ServiceWorkerRegistrationKey::hash() const
50{
51 unsigned hashes[2];
52 hashes[0] = SecurityOriginDataHash::hash(m_topOrigin);
53 hashes[1] = StringHash::hash(m_scope.string());
54
55 return StringHasher::hashMemory(hashes, sizeof(hashes));
56}
57
58bool ServiceWorkerRegistrationKey::operator==(const ServiceWorkerRegistrationKey& other) const
59{
60 return m_topOrigin == other.m_topOrigin && m_scope == other.m_scope;
61}
62
63ServiceWorkerRegistrationKey ServiceWorkerRegistrationKey::isolatedCopy() const
64{
65 return { m_topOrigin.isolatedCopy(), m_scope.isolatedCopy() };
66}

68bool ServiceWorkerRegistrationKey::isMatching(const SecurityOriginData& topOrigin, const URL& clientURL) const
69{
70 return originIsMatching(topOrigin, clientURL) && clientURL.string().startsWith(m_scope.string());
71}
72
73bool ServiceWorkerRegistrationKey::originIsMatching(const SecurityOriginData& topOrigin, const URL& clientURL) const
74{
75 if (topOrigin != m_topOrigin)
76 return false;
77
78 return protocolHostAndPortAreEqual(clientURL, m_scope);
79}
80
81bool ServiceWorkerRegistrationKey::relatesToOrigin(const SecurityOriginData& securityOrigin) const
82{
83 if (m_topOrigin == securityOrigin)
84 return true;
85
86 return SecurityOriginData::fromURL(m_scope) == securityOrigin;
87}

89static const char separatorCharacter = '_';
90
91String ServiceWorkerRegistrationKey::toDatabaseKey() const
92{
93 if (m_topOrigin.port)
94 return makeString(m_topOrigin.protocol, separatorCharacter, m_topOrigin.host, separatorCharacter, String::number(m_topOrigin.port.value()), separatorCharacter, m_scope.string());
95 return makeString(m_topOrigin.protocol, separatorCharacter, m_topOrigin.host, separatorCharacter, separatorCharacter, m_scope.string());
96}
97
98std::optional<ServiceWorkerRegistrationKey> ServiceWorkerRegistrationKey::fromDatabaseKey(const String& key)
99{
100 auto first = key.find(separatorCharacter, 0);
101 auto second = key.find(separatorCharacter, first + 1);
102 auto third = key.find(separatorCharacter, second + 1);
103
104 if (first == second || second == third)
105 return std::nullopt;
106
107 std::optional<uint16_t> shortPort;
108
109 // If there's a gap between third and second, we expect to have a port to decode
110 if (third - second > 1) {
111 shortPort = parseInteger<uint16_t>(StringView { key }.substring(second + 1, third - second - 1));
112 if (!shortPort)
113 return std::nullopt;
114 }
115
116 auto scope = URL { URL(), key.substring(third + 1) };
117 if (!scope.isValid())
118 return std::nullopt;
119
120 return ServiceWorkerRegistrationKey { { key.substring(0, first), key.substring(first + 1, second - first - 1), shortPort }, WTFMove(scope) };
121}
