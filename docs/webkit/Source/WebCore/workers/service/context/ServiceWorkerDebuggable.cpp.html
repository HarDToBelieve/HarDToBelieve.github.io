35namespace WebCore {
36
37using namespace Inspector;
38
39ServiceWorkerDebuggable::ServiceWorkerDebuggable(ServiceWorkerThreadProxy& serviceWorkerThreadProxy, const ServiceWorkerContextData& data)
40 : m_serviceWorkerThreadProxy(serviceWorkerThreadProxy)
41 , m_scopeURL(data.registration.scopeURL.string())
42{
43}
44
45void ServiceWorkerDebuggable::connect(FrontendChannel& channel, bool, bool)
46{
47 m_serviceWorkerThreadProxy.inspectorProxy().connectToWorker(channel);
48}
49
50void ServiceWorkerDebuggable::disconnect(FrontendChannel& channel)
51{
52 m_serviceWorkerThreadProxy.inspectorProxy().disconnectFromWorker(channel);
53}
54
55void ServiceWorkerDebuggable::dispatchMessageFromRemote(const String& message)
56{
57 m_serviceWorkerThreadProxy.inspectorProxy().sendMessageToWorker(message);
58}
