26#pragma once
27
28#if ENABLE(REMOTE_INSPECTOR)
29#if ENABLE(SERVICE_WORKER)
30
31#include "ServiceWorkerContextData.h"
32#include <JavaScriptCore/RemoteInspectionTarget.h>
33#include <wtf/Noncopyable.h>
34
35namespace WebCore {
36
37class ServiceWorkerThreadProxy;
38
39class ServiceWorkerDebuggable final : public Inspector::RemoteInspectionTarget {
40 WTF_MAKE_FAST_ALLOCATED;
41 WTF_MAKE_NONCOPYABLE(ServiceWorkerDebuggable);
42public:
43 ServiceWorkerDebuggable(ServiceWorkerThreadProxy&, const ServiceWorkerContextData&);
44 ~ServiceWorkerDebuggable() = default;
45
46 Inspector::RemoteControllableTarget::Type type() const final { return Inspector::RemoteControllableTarget::Type::ServiceWorker; }
47
48 String name() const final { return "ServiceWorker"_s; }
49 String url() const final { return m_scopeURL; }
50 bool hasLocalDebugger() const final { return false; }
51
52 void connect(Inspector::FrontendChannel&, bool isAutomaticConnection = false, bool immediatelyPause = false) final;
53 void disconnect(Inspector::FrontendChannel&) final;
54 void dispatchMessageFromRemote(const String& message) final;
55
56private:
57 ServiceWorkerThreadProxy& m_serviceWorkerThreadProxy;
58 String m_scopeURL;
59};
60
61} // namespace WebCore
62
63SPECIALIZE_TYPE_TRAITS_CONTROLLABLE_TARGET(WebCore::ServiceWorkerDebuggable, ServiceWorker);

65#endif // ENABLE(SERVICE_WORKER)
66#endif // ENABLE(REMOTE_INSPECTOR)
