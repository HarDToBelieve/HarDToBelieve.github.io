103void ServiceWorkerThread::queueTaskToFireFetchEvent(Ref<ServiceWorkerFetch::Client>&& client, std::optional<ServiceWorkerClientIdentifier>&& clientId, ResourceRequest&& request, String&& referrer, FetchOptions&& options)
104{
105 auto serviceWorkerGlobalScope = makeRef(downcast<ServiceWorkerGlobalScope>(*globalScope()));
106 serviceWorkerGlobalScope->eventLoop().queueTask(TaskSource::DOMManipulation, [serviceWorkerGlobalScope, client = WTFMove(client), clientId, request = WTFMove(request), referrer = WTFMove(referrer), options = WTFMove(options)]() mutable {
107 ServiceWorkerFetch::dispatchFetchEvent(WTFMove(client), serviceWorkerGlobalScope, clientId, WTFMove(request), WTFMove(referrer), WTFMove(options));
108 });

111static void fireMessageEvent(ServiceWorkerGlobalScope& scope, MessageWithMessagePorts&& message, ExtendableMessageEventSource&& source, const URL& sourceURL)
112{
113 auto ports = MessagePort::entanglePorts(scope, WTFMove(message.transferredPorts));
114 auto messageEvent = ExtendableMessageEvent::create(WTFMove(ports), WTFMove(message.message), SecurityOriginData::fromURL(sourceURL).toString(), { }, source);
115 scope.dispatchEvent(messageEvent);
116 scope.thread().workerObjectProxy().confirmMessageFromWorkerObject(scope.hasPendingActivity());
117 scope.updateExtendedEventsSet(messageEvent.ptr());
118}

120void ServiceWorkerThread::queueTaskToPostMessage(MessageWithMessagePorts&& message, ServiceWorkerOrClientData&& sourceData)
121{
122 auto serviceWorkerGlobalScope = makeRef(downcast<ServiceWorkerGlobalScope>(*globalScope()));
123 serviceWorkerGlobalScope->eventLoop().queueTask(TaskSource::DOMManipulation, [weakThis = makeWeakPtr(this), serviceWorkerGlobalScope, message = WTFMove(message), sourceData = WTFMove(sourceData)]() mutable {
124 URL sourceURL;
125 ExtendableMessageEventSource source;
126 if (WTF::holds_alternative<ServiceWorkerClientData>(sourceData)) {
127 RefPtr<ServiceWorkerClient> sourceClient = ServiceWorkerClient::getOrCreate(serviceWorkerGlobalScope, WTFMove(WTF::get<ServiceWorkerClientData>(sourceData)));
128
129 RELEASE_ASSERT(!sourceClient->url().protocolIsInHTTPFamily() || !serviceWorkerGlobalScope->url().protocolIsInHTTPFamily() || protocolHostAndPortAreEqual(serviceWorkerGlobalScope->url(), sourceClient->url()));
130
131 sourceURL = sourceClient->url();
132 source = WTFMove(sourceClient);
133 } else {
134 RefPtr<ServiceWorker> sourceWorker = ServiceWorker::getOrCreate(serviceWorkerGlobalScope, WTFMove(WTF::get<ServiceWorkerData>(sourceData)));
135
136 RELEASE_ASSERT(!sourceWorker->scriptURL().protocolIsInHTTPFamily() || !serviceWorkerGlobalScope->url().protocolIsInHTTPFamily() || protocolHostAndPortAreEqual(serviceWorkerGlobalScope->url(), sourceWorker->scriptURL()));
137
138 sourceURL = sourceWorker->scriptURL();
139 source = WTFMove(sourceWorker);
140 }
141 fireMessageEvent(serviceWorkerGlobalScope, WTFMove(message), ExtendableMessageEventSource { source }, sourceURL);
142 callOnMainThread([weakThis = WTFMove(weakThis)] {
143 if (weakThis)
144 weakThis->finishedFiringMessageEvent();
145 });
146 });
147}

149void ServiceWorkerThread::queueTaskToFireInstallEvent()
150{
151 auto serviceWorkerGlobalScope = makeRef(downcast<ServiceWorkerGlobalScope>(*globalScope()));
152 serviceWorkerGlobalScope->eventLoop().queueTask(TaskSource::DOMManipulation, [weakThis = makeWeakPtr(this), serviceWorkerGlobalScope]() mutable {
153 RELEASE_LOG(ServiceWorker, "ServiceWorkerThread::queueTaskToFireInstallEvent firing event for worker %llu", serviceWorkerGlobalScope->thread().identifier().toUInt64());
154
155 auto installEvent = ExtendableEvent::create(eventNames().installEvent, { }, ExtendableEvent::IsTrusted::Yes);
156 serviceWorkerGlobalScope->dispatchEvent(installEvent);
157
158 installEvent->whenAllExtendLifetimePromisesAreSettled([weakThis = WTFMove(weakThis)](HashSet<Ref<DOMPromise>>&& extendLifetimePromises) mutable {
159 bool hasRejectedAnyPromise = false;
160 for (auto& promise : extendLifetimePromises) {
161 if (promise->status() == DOMPromise::Status::Rejected) {
162 hasRejectedAnyPromise = true;
163 break;
164 }
165 }
166 callOnMainThread([weakThis = WTFMove(weakThis), hasRejectedAnyPromise] {
167 RELEASE_LOG(ServiceWorker, "ServiceWorkerThread::queueTaskToFireInstallEvent finishing for worker %llu", weakThis ? weakThis->identifier().toUInt64() : 0);
168 if (weakThis)
169 weakThis->finishedFiringInstallEvent(hasRejectedAnyPromise);
170 });
171 });
172 });
173}

200void ServiceWorkerThread::start(Function<void(const String&, bool)>&& callback)
201{
202 m_state = State::Starting;
203 startHeartBeatTimer();
204
205 WorkerThread::start([callback = WTFMove(callback), weakThis = makeWeakPtr(this)](auto& errorMessage) mutable {
206 bool doesHandleFetch = true;
207 if (weakThis) {
208 weakThis->finishedStarting();
209 doesHandleFetch = weakThis->doesHandleFetch();
210 }
211 callback(errorMessage, doesHandleFetch);
212 });
213}

226void ServiceWorkerThread::startHeartBeatTimer()
227{
228 if (m_heartBeatTimer.isActive())
229 return;
230
231 m_ongoingHeartBeatCheck = true;
232 runLoop().postTask([this, protectedThis = makeRef(*this)](auto&) mutable {
233 callOnMainThread([this, protectedThis = WTFMove(protectedThis)]() {
234 m_ongoingHeartBeatCheck = false;
235 });
236 });
237
238 m_heartBeatTimer.startOneShot(m_heartBeatTimeout);
239}
240
241void ServiceWorkerThread::heartBeatTimerFired()
242{
243 if (!m_ongoingHeartBeatCheck) {
244 if (m_state == State::Installing || m_state == State::Activating || m_isHandlingFetchEvent || m_messageEventCount)
245 startHeartBeatTimer();
246 return;
247 }
248
249 auto* serviceWorkerThreadProxy = SWContextManager::singleton().serviceWorkerThreadProxy(identifier());
250 if (!serviceWorkerThreadProxy || serviceWorkerThreadProxy->isTerminatingOrTerminated())
251 return;
252
253 auto* connection = SWContextManager::singleton().connection();
254 if (!connection)
255 return;
256
257 switch (m_state) {
258 case State::Idle:
259 case State::Activating:
260 connection->didFailHeartBeatCheck(identifier());
261 break;
262 case State::Starting:
263 connection->serviceWorkerFailedToStart(m_jobDataIdentifier, identifier(), "Service Worker script execution timed out"_s);
264 break;
265 case State::Installing:
266 connection->didFinishInstall(m_jobDataIdentifier, identifier(), false);
267 break;
268 }
269}

277void ServiceWorkerThread::finishedFiringInstallEvent(bool hasRejectedAnyPromise)
278{
279 m_state = State::Idle;

282 connection->didFinishInstall(m_jobDataIdentifier, identifier(), !hasRejectedAnyPromise);
283}
