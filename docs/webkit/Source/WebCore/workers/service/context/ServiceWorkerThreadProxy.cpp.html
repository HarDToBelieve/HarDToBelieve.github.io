54{
55 return URL { URL { }, origin.toRawString() };

58static inline UniqueRef<Page> createPageForServiceWorker(PageConfiguration&& configuration, const ServiceWorkerContextData& data, StorageBlockingPolicy storageBlockingPolicy)
59{
60 auto page = makeUniqueRef<Page>(WTFMove(configuration));
61
62 page->settings().setStorageBlockingPolicy(storageBlockingPolicy);
63
64 auto& mainFrame = page->mainFrame();
65 mainFrame.loader().initForSynthesizedDocument({ });
66 auto document = Document::createNonRenderedPlaceholder(mainFrame, data.scriptURL);
67 document->createDOMWindow();
68
69 document->storageBlockingStateDidChange();
70
71 auto origin = data.registration.key.topOrigin().securityOrigin();
72 origin->setStorageBlockingPolicy(storageBlockingPolicy);
73
74 document->setSiteForCookies(topOriginURL(origin));
75 document->setFirstPartyForCookies(topOriginURL(origin));
76 document->setDomainForCachePartition(origin->domainForCachePartition());
77
78 if (auto policy = parseReferrerPolicy(data.referrerPolicy, ReferrerPolicySource::HTTPHeader))
79 document->setReferrerPolicy(*policy);
80
81 mainFrame.setDocument(WTFMove(document));
82 return page;
83}

96ServiceWorkerThreadProxy::ServiceWorkerThreadProxy(PageConfiguration&& pageConfiguration, ServiceWorkerContextData&& data, String&& userAgent, CacheStorageProvider& cacheStorageProvider, StorageBlockingPolicy storageBlockingPolicy)
97 : m_page(createPageForServiceWorker(WTFMove(pageConfiguration), data, storageBlockingPolicy))
98 , m_document(*m_page->mainFrame().document())
99#if ENABLE(REMOTE_INSPECTOR)
100 , m_remoteDebuggable(makeUnique<ServiceWorkerDebuggable>(*this, data))
101#endif
102 , m_serviceWorkerThread(ServiceWorkerThread::create(WTFMove(data), WTFMove(userAgent), m_document->settingsValues(), *this, *this, idbConnectionProxy(m_document), m_document->socketProvider()))
103 , m_cacheStorageProvider(cacheStorageProvider)
104 , m_inspectorProxy(*this)
105{
106 static bool addedListener;
107 if (!addedListener) {
108 platformStrategies()->loaderStrategy()->addOnlineStateChangeListener(&networkStateChanged);
109 addedListener = true;
110 }
111
112 ASSERT(!allServiceWorkerThreadProxies().contains(this));
113 allServiceWorkerThreadProxies().add(this);
114
115#if ENABLE(REMOTE_INSPECTOR)
116 m_remoteDebuggable->setRemoteDebuggingAllowed(true);
117 m_remoteDebuggable->init();
118#endif
119
120 if (data.lastNavigationWasAppBound)
121 setLastNavigationWasAppBound(data.lastNavigationWasAppBound == LastNavigationWasAppBound::Yes);
122}

202void ServiceWorkerThreadProxy::notifyNetworkStateChange(bool isOnline)
203{
204 if (m_isTerminatingOrTerminated)
205 return;
206
207 postTaskForModeToWorkerOrWorkletGlobalScope([isOnline] (ScriptExecutionContext& context) {
208 auto& globalScope = downcast<WorkerGlobalScope>(context);
209 globalScope.setIsOnline(isOnline);
210 globalScope.eventLoop().queueTask(TaskSource::DOMManipulation, [globalScope = makeRef(globalScope), isOnline] {
211 globalScope->dispatchEvent(Event::create(isOnline ? eventNames().onlineEvent : eventNames().offlineEvent, Event::CanBubble::No, Event::IsCancelable::No));
212 });
213 }, WorkerRunLoop::defaultMode());
214}

216void ServiceWorkerThreadProxy::startFetch(SWServerConnectionIdentifier connectionIdentifier, FetchIdentifier fetchIdentifier, Ref<ServiceWorkerFetch::Client>&& client, std::optional<ServiceWorkerClientIdentifier>&& clientId, ResourceRequest&& request, String&& referrer, FetchOptions&& options)
217{
218 RELEASE_LOG(ServiceWorker, "ServiceWorkerThreadProxy::startFetch %llu", fetchIdentifier.toUInt64());
219
220 auto key = std::make_pair(connectionIdentifier, fetchIdentifier);
221
222 if (m_ongoingFetchTasks.isEmpty())
223 thread().startFetchEventMonitoring();
224
225 ASSERT(!m_ongoingFetchTasks.contains(key));
226 m_ongoingFetchTasks.add(key, client.copyRef());
227 postTaskForModeToWorkerOrWorkletGlobalScope([this, protectedThis = makeRef(*this), client = WTFMove(client), clientId, request = request.isolatedCopy(), referrer = referrer.isolatedCopy(), options = options.isolatedCopy()](auto&) mutable {
228 thread().queueTaskToFireFetchEvent(WTFMove(client), WTFMove(clientId), WTFMove(request), WTFMove(referrer), WTFMove(options));
229 }, WorkerRunLoop::defaultMode());
230}

232void ServiceWorkerThreadProxy::cancelFetch(SWServerConnectionIdentifier connectionIdentifier, FetchIdentifier fetchIdentifier)
233{
234 RELEASE_LOG(ServiceWorker, "ServiceWorkerThreadProxy::cancelFetch %llu", fetchIdentifier.toUInt64());
235
236 auto client = m_ongoingFetchTasks.take(std::make_pair(connectionIdentifier, fetchIdentifier));
237 if (!client)
238 return;
239
240 if (m_ongoingFetchTasks.isEmpty())
241 thread().stopFetchEventMonitoring();
242
243 postTaskForModeToWorkerOrWorkletGlobalScope([client = client.releaseNonNull()] (ScriptExecutionContext&) {
244 client->cancel();
245 }, WorkerRunLoop::defaultMode());
246}

248void ServiceWorkerThreadProxy::continueDidReceiveFetchResponse(SWServerConnectionIdentifier connectionIdentifier, FetchIdentifier fetchIdentifier)
249{
250 auto client = m_ongoingFetchTasks.get(std::make_pair(connectionIdentifier, fetchIdentifier));
251 if (!client)
252 return;
253
254 postTaskForModeToWorkerOrWorkletGlobalScope([client = makeRef(*client)] (ScriptExecutionContext&) {
255 client->continueDidReceiveResponse();
256 }, WorkerRunLoop::defaultMode());
257}

259void ServiceWorkerThreadProxy::removeFetch(SWServerConnectionIdentifier connectionIdentifier, FetchIdentifier fetchIdentifier)
260{
261 RELEASE_LOG(ServiceWorker, "ServiceWorkerThreadProxy::removeFetch %llu", fetchIdentifier.toUInt64());
262
263 m_ongoingFetchTasks.remove(std::make_pair(connectionIdentifier, fetchIdentifier));
264
265 if (m_ongoingFetchTasks.isEmpty())
266 thread().stopFetchEventMonitoring();
267}
