104struct ImportedScriptAttributes {
105 URL responseURL;
106 String mimeType;
107
108 template<class Encoder> void encode(Encoder& encoder) const
109 {
110 encoder << responseURL << mimeType;
111 }
112
113 template<class Decoder> static std::optional<ImportedScriptAttributes> decode(Decoder& decoder)
114 {
115 std::optional<URL> responseURL;
116 decoder >> responseURL;
117 if (!responseURL)
118 return std::nullopt;
119
120 std::optional<String> mimeType;
121 decoder >> mimeType;
122 if (!mimeType)
123 return std::nullopt;
124
125 return {{
126 WTFMove(*responseURL),
127 WTFMove(*mimeType)
128 }};
129 }
130};

132static HashMap<URL, ImportedScriptAttributes> stripScriptSources(const HashMap<URL, ServiceWorkerContextData::ImportedScript>& map)
133{
134 HashMap<URL, ImportedScriptAttributes> mapWithoutScripts;
135 for (auto& pair : map)
136 mapWithoutScripts.add(pair.key, ImportedScriptAttributes { pair.value.responseURL, pair.value.mimeType });
137 return mapWithoutScripts;
138}

140static HashMap<URL, ServiceWorkerContextData::ImportedScript> populateScriptSourcesFromDisk(SWScriptStorage& scriptStorage, const ServiceWorkerRegistrationKey& registrationKey, HashMap<URL, ImportedScriptAttributes>&& map)
141{
142 HashMap<URL, ServiceWorkerContextData::ImportedScript> importedScripts;
143 for (auto& pair : map) {
144 auto importedScript = scriptStorage.retrieve(registrationKey, pair.key);
145 if (!importedScript) {
146 RELEASE_LOG_ERROR(ServiceWorker, "RegistrationDatabase::populateScriptSourcesFromDisk: Failed to retrieve imported script for %s from disk", pair.key.string().utf8().data());
147 continue;
148 }
149 importedScripts.add(pair.key, ServiceWorkerContextData::ImportedScript { WTFMove(importedScript), WTFMove(pair.value.responseURL), WTFMove(pair.value.mimeType) });
150 }
151 return importedScripts;
152}

196bool RegistrationDatabase::openSQLiteDatabase(const String& fullFilename)
197{
198 ASSERT(!isMainThread());
199 ASSERT(!m_database);
200
201 auto databaseDirectory = this->databaseDirectoryIsolatedCopy();
202 cleanOldDatabases(databaseDirectory);
203
204 LOG(ServiceWorker, "ServiceWorker RegistrationDatabase opening file %s", fullFilename.utf8().data());
205
206 SQLiteFileSystem::ensureDatabaseDirectoryExists(databaseDirectory);
207
208 m_database = makeUnique<SQLiteDatabase>();
209 if (!m_database->open(fullFilename)) {
210 RELEASE_LOG_ERROR(ServiceWorker, "Failed to open Service Worker registration database");
211 m_database = nullptr;
212 return false;
213 }
214
215 // Disable threading checks. We always access the database from our serial WorkQueue. Such accesses
216 // are safe since work queue tasks are guaranteed to run one after another. However, tasks will not
217 // necessary run on the same thread every time (as per GCD documentation).
218 m_database->disableThreadingChecks();
219
220 auto doRecoveryAttempt = [&] {
221 // Delete the database and files. We will try recreating it when flushing changes.
222 m_database = nullptr;
223 SQLiteFileSystem::deleteDatabaseFile(fullFilename);
224 };
225
226 String errorMessage = ensureValidRecordsTable();
227 if (!errorMessage.isNull()) {
228 RELEASE_LOG_ERROR(ServiceWorker, "ensureValidRecordsTable failed, reason: %" PUBLIC_LOG_STRING, errorMessage.utf8().data());
229 doRecoveryAttempt();
230 return false;
231 }
232
233 errorMessage = importRecords();
234 if (!errorMessage.isNull()) {
235 RELEASE_LOG_ERROR(ServiceWorker, "importRecords failed, reason: %" PUBLIC_LOG_STRING, errorMessage.utf8().data());
236 doRecoveryAttempt();
237 return false;
238 }
239 return true;
240}

273String RegistrationDatabase::ensureValidRecordsTable()
274{
275 ASSERT(!isMainThread());
276 ASSERT(m_database);
277 ASSERT(m_database->isOpen());
278
279 String currentSchema;
280 {
281 // Fetch the schema for an existing records table.
282 auto statement = m_database->prepareStatement("SELECT type, sql FROM sqlite_master WHERE tbl_name='Records'"_s);
283 if (!statement)
284 return "Unable to prepare statement to fetch schema for the Records table."_s;
285
286 int sqliteResult = statement->step();
287
288 // If there is no Records table at all, create it and then bail.
289 if (sqliteResult == SQLITE_DONE) {
290 if (!m_database->executeCommand(recordsTableSchema()))
291 return makeString("Could not create Records table in database (", m_database->lastError(), ") - ", m_database->lastErrorMsg());
292 return { };
293 }
294
295 if (sqliteResult != SQLITE_ROW)
296 return "Error executing statement to fetch schema for the Records table.";
297
298 currentSchema = statement->columnText(1);
299 }
300
301 ASSERT(!currentSchema.isEmpty());
302
303 if (currentSchema == recordsTableSchema() || currentSchema == recordsTableSchemaAlternate())
304 return { };
305
306 return makeString("Unexpected schema: ", currentSchema);
307}

357void RegistrationDatabase::pushChanges(const HashMap<ServiceWorkerRegistrationKey, std::optional<ServiceWorkerContextData>>& changedRegistrations, CompletionHandler<void()>&& completionHandler)
358{
359 Vector<ServiceWorkerContextData> updatedRegistrations;
360 Vector<ServiceWorkerRegistrationKey> removedRegistrations;
361 for (auto& keyValue : changedRegistrations) {
362 if (keyValue.value)
363 updatedRegistrations.append(keyValue.value->isolatedCopy());
364 else
365 removedRegistrations.append(keyValue.key.isolatedCopy());
366 }
367 schedulePushChanges(WTFMove(updatedRegistrations), WTFMove(removedRegistrations), ShouldRetry::Yes, WTFMove(completionHandler));
368}

370void RegistrationDatabase::schedulePushChanges(Vector<ServiceWorkerContextData>&& updatedRegistrations, Vector<ServiceWorkerRegistrationKey>&& removedRegistrations, ShouldRetry shouldRetry, CompletionHandler<void()>&& completionHandler)
371{
372 auto pushCounter = shouldRetry == ShouldRetry::Yes ? m_pushCounter : 0;
373 postTaskToWorkQueue([this, protectedThis = makeRef(*this), pushCounter, updatedRegistrations = WTFMove(updatedRegistrations), removedRegistrations = WTFMove(removedRegistrations), completionHandler = WTFMove(completionHandler)]() mutable {
374 bool success = doPushChanges(updatedRegistrations, removedRegistrations);
375 if (success) {
376 updatedRegistrations.clear();
377 removedRegistrations.clear();
378 }
379 callOnMainThread([this, protectedThis = WTFMove(protectedThis), success, pushCounter, updatedRegistrations = WTFMove(updatedRegistrations).isolatedCopy(), removedRegistrations = WTFMove(removedRegistrations).isolatedCopy(), completionHandler = WTFMove(completionHandler)]() mutable {
380 if (!success && (pushCounter + 1) == m_pushCounter) {
381 // We retry writing once if no other change was pushed.
382 schedulePushChanges(WTFMove(updatedRegistrations), WTFMove(removedRegistrations), ShouldRetry::No, WTFMove(completionHandler));
383 return;
384 }
385 if (completionHandler)
386 completionHandler();
387 });
388 });
389}

413bool RegistrationDatabase::doPushChanges(const Vector<ServiceWorkerContextData>& updatedRegistrations, const Vector<ServiceWorkerRegistrationKey>& removedRegistrations)
414{
415 if (!m_database) {
416 openSQLiteDatabase(m_databaseFilePath);
417 if (!m_database)
418 return false;
419 }
420
421 SQLiteTransaction transaction(*m_database);
422 transaction.begin();
423
424 auto insertStatement = m_database->prepareStatement("INSERT INTO Records VALUES (?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?)"_s);
425 if (!insertStatement) {
426 RELEASE_LOG_ERROR(ServiceWorker, "Failed to prepare statement to store registration data into records table (%i) - %s", m_database->lastError(), m_database->lastErrorMsg());
427 return false;
428 }
429
430 auto& scriptStorage = this->scriptStorage();
431 for (auto& registration : removedRegistrations) {
432 auto deleteStatement = m_database->prepareStatement("DELETE FROM Records WHERE key = ?"_s);
433 if (!deleteStatement
434 || deleteStatement->bindText(1, registration.toDatabaseKey()) != SQLITE_OK
435 || deleteStatement->step() != SQLITE_DONE) {
436 RELEASE_LOG_ERROR(ServiceWorker, "Failed to remove registration data from records table (%i) - %s", m_database->lastError(), m_database->lastErrorMsg());
437 return false;
438 }
439 scriptStorage.clear(registration);
440 }
441
442 for (auto& data : updatedRegistrations) {
443 WTF::Persistence::Encoder cspEncoder;
444 data.contentSecurityPolicy.encode(cspEncoder);
445
446 // We don't actually encode the script sources to the database. They will be stored separately in the ScriptStorage.
447 // As a result, we need to strip the script sources here before encoding the scriptResourceMap.
448 WTF::Persistence::Encoder scriptResourceMapEncoder;
449 scriptResourceMapEncoder << stripScriptSources(data.scriptResourceMap);
450
451 WTF::Persistence::Encoder certificateInfoEncoder;
452 certificateInfoEncoder << data.certificateInfo;
453
454 if (insertStatement->bindText(1, data.registration.key.toDatabaseKey()) != SQLITE_OK
455 || insertStatement->bindText(2, data.registration.scopeURL.protocolHostAndPort()) != SQLITE_OK
456 || insertStatement->bindText(3, data.registration.scopeURL.path().toString()) != SQLITE_OK
457 || insertStatement->bindText(4, data.registration.key.topOrigin().databaseIdentifier()) != SQLITE_OK
458 || insertStatement->bindDouble(5, data.registration.lastUpdateTime.secondsSinceEpoch().value()) != SQLITE_OK
459 || insertStatement->bindText(6, updateViaCacheToString(data.registration.updateViaCache)) != SQLITE_OK
460 || insertStatement->bindText(7, data.scriptURL.string()) != SQLITE_OK
461 || insertStatement->bindText(8, workerTypeToString(data.workerType)) != SQLITE_OK
462 || insertStatement->bindBlob(9, Span { cspEncoder.buffer(), cspEncoder.bufferSize() }) != SQLITE_OK
463 || insertStatement->bindText(10, data.referrerPolicy) != SQLITE_OK
464 || insertStatement->bindBlob(11, Span { scriptResourceMapEncoder.buffer(), scriptResourceMapEncoder.bufferSize() }) != SQLITE_OK
465 || insertStatement->bindBlob(12, Span { certificateInfoEncoder.buffer(), certificateInfoEncoder.bufferSize() }) != SQLITE_OK
466 || insertStatement->step() != SQLITE_DONE) {
467 RELEASE_LOG_ERROR(ServiceWorker, "Failed to store registration data into records table (%i) - %s", m_database->lastError(), m_database->lastErrorMsg());
468 return false;
469 }
470
471 // Save scripts to disk.
472 auto mainScript = scriptStorage.store(data.registration.key, data.scriptURL, data.script);
473 ASSERT(mainScript);
474 if (!mainScript)
475 return false;
476 HashMap<URL, ScriptBuffer> importedScripts;
477 for (auto& pair : data.scriptResourceMap) {
478 auto importedScript = scriptStorage.store(data.registration.key, pair.key, pair.value.script);
479 ASSERT(importedScript);
480 if (importedScript)
481 importedScripts.add(crossThreadCopy(pair.key), crossThreadCopy(importedScript));
482 }
483 callOnMainThread([this, protectedThis = makeRef(*this), serviceWorkerIdentifier = data.serviceWorkerIdentifier, mainScript = crossThreadCopy(mainScript), importedScripts = WTFMove(importedScripts)]() mutable {
484 if (m_store)
485 m_store->didSaveWorkerScriptsToDisk(serviceWorkerIdentifier, WTFMove(mainScript), WTFMove(importedScripts));
486 });
487 }
488
489 transaction.commit();
490 RELEASE_LOG(ServiceWorker, "Updated ServiceWorker registration database (%zu added/updated registrations and %zu removed registrations", updatedRegistrations.size(), removedRegistrations.size());
491 return true;
492}

494String RegistrationDatabase::importRecords()
495{
496 ASSERT(!isMainThread());
497
498 RELEASE_LOG(ServiceWorker, "RegistrationDatabase::importRecords:");
499 auto sql = m_database->prepareStatement("SELECT * FROM Records;"_s);
500 if (!sql)
501 return makeString("Failed to prepare statement to retrieve registrations from records table (", m_database->lastError(), ") - ", m_database->lastErrorMsg());
502
503 int result = sql->step();
504
505 for (; result == SQLITE_ROW; result = sql->step()) {
506 RELEASE_LOG(ServiceWorker, "RegistrationDatabase::importRecords: Importing a registration from the database");
507 auto key = ServiceWorkerRegistrationKey::fromDatabaseKey(sql->columnText(0));
508 auto originURL = URL { URL(), sql->columnText(1) };
509 auto scopePath = sql->columnText(2);
510 auto scopeURL = URL { originURL, scopePath };
511 auto topOrigin = SecurityOriginData::fromDatabaseIdentifier(sql->columnText(3));
512 auto lastUpdateCheckTime = WallTime::fromRawSeconds(sql->columnDouble(4));
513 auto updateViaCache = stringToUpdateViaCache(sql->columnText(5));
514 auto scriptURL = URL { URL(), sql->columnText(6) };
515 auto workerType = stringToWorkerType(sql->columnText(7));
516
517 std::optional<ContentSecurityPolicyResponseHeaders> contentSecurityPolicy;
518 auto contentSecurityPolicyDataSpan = sql->columnBlobAsSpan(8);
519 if (contentSecurityPolicyDataSpan.size()) {
520 WTF::Persistence::Decoder cspDecoder(contentSecurityPolicyDataSpan.data(), contentSecurityPolicyDataSpan.size());
521 cspDecoder >> contentSecurityPolicy;
522 if (!contentSecurityPolicy) {
523 RELEASE_LOG_ERROR(ServiceWorker, "RegistrationDatabase::importRecords: Failed to decode contentSecurityPolicy");
524 continue;
525 }
526 }
527
528 auto referrerPolicy = sql->columnText(9);
529
530 HashMap<URL, ServiceWorkerContextData::ImportedScript> scriptResourceMap;
531 auto scriptResourceMapDataSpan = sql->columnBlobAsSpan(10);
532 if (scriptResourceMapDataSpan.size()) {
533 WTF::Persistence::Decoder scriptResourceMapDecoder(scriptResourceMapDataSpan.data(), scriptResourceMapDataSpan.size());
534 std::optional<HashMap<URL, ImportedScriptAttributes>> scriptResourceMapWithoutScripts;
535 scriptResourceMapDecoder >> scriptResourceMapWithoutScripts;
536 if (!scriptResourceMapWithoutScripts) {
537 RELEASE_LOG_ERROR(ServiceWorker, "RegistrationDatabase::importRecords: Failed to decode scriptResourceMapWithoutScripts");
538 continue;
539 }
540 scriptResourceMap = populateScriptSourcesFromDisk(scriptStorage(), *key, WTFMove(*scriptResourceMapWithoutScripts));
541 }
542
543 auto certificateInfoDataSpan = sql->columnBlobAsSpan(11);
544 std::optional<CertificateInfo> certificateInfo;
545
546 WTF::Persistence::Decoder certificateInfoDecoder(certificateInfoDataSpan.data(), certificateInfoDataSpan.size());
547 certificateInfoDecoder >> certificateInfo;
548 if (!certificateInfo) {
549 RELEASE_LOG_ERROR(ServiceWorker, "RegistrationDatabase::importRecords: Failed to decode certificateInfo");
550 continue;
551 }
552
553 // Validate the input for this registration.
554 // If any part of this input is invalid, let's skip this registration.
555 // FIXME: Should we return an error skipping *all* registrations?
556 if (!key || !originURL.isValid() || !topOrigin || !updateViaCache || !scriptURL.isValid() || !workerType || !scopeURL.isValid()) {
557 RELEASE_LOG_ERROR(ServiceWorker, "RegistrationDatabase::importRecords: Failed to decode part of the registration");
558 continue;
559 }
560
561 auto script = scriptStorage().retrieve(*key, scriptURL);
562 ASSERT(script);
563 if (!script) {
564 RELEASE_LOG_ERROR(ServiceWorker, "RegistrationDatabase::importRecords: Failed to retrieve main script for %s from disk", scriptURL.string().utf8().data());
565 continue;
566 }
567
568 auto workerIdentifier = ServiceWorkerIdentifier::generate();
569 auto registrationIdentifier = ServiceWorkerRegistrationIdentifier::generate();
570 auto serviceWorkerData = ServiceWorkerData { workerIdentifier, scriptURL, ServiceWorkerState::Activated, *workerType, registrationIdentifier };
571 auto registration = ServiceWorkerRegistrationData { WTFMove(*key), registrationIdentifier, WTFMove(scopeURL), *updateViaCache, lastUpdateCheckTime, std::nullopt, std::nullopt, WTFMove(serviceWorkerData) };
572 auto contextData = ServiceWorkerContextData { std::nullopt, WTFMove(registration), workerIdentifier, WTFMove(script), WTFMove(*certificateInfo), WTFMove(*contentSecurityPolicy), WTFMove(referrerPolicy), WTFMove(scriptURL), *workerType, true, LastNavigationWasAppBound::No, WTFMove(scriptResourceMap) };
573
574 callOnMainThread([protectedThis = makeRef(*this), contextData = contextData.isolatedCopy()]() mutable {
575 protectedThis->addRegistrationToStore(WTFMove(contextData));
576 });
577 }
578
579 if (result != SQLITE_DONE)
580 return makeString("Failed to import at least one registration from records table (", m_database->lastError(), ") - ", m_database->lastErrorMsg());
581
582 return { };
583}
