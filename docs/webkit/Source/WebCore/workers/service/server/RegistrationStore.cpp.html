35RegistrationStore::RegistrationStore(SWServer& server, String&& databaseDirectory)
36 : m_server(server)
37 , m_database(RegistrationDatabase::create(*this, WTFMove(databaseDirectory)))
38 , m_databasePushTimer(*this, &RegistrationStore::pushChangesToDatabase)
39{
40}
41
42RegistrationStore::~RegistrationStore()
43{
44}
45
46void RegistrationStore::scheduleDatabasePushIfNecessary()
47{
48 if (m_databasePushTimer.isActive())
49 return;
50 m_databasePushTimer.startOneShot(0_s);
51}
52
53void RegistrationStore::pushChangesToDatabase(CompletionHandler<void()>&& completionHandler)
54{
55 if (m_isSuspended) {
56 m_needsPushingChanges = true;
57 return;
58 }
59
60 m_database->pushChanges(m_updatedRegistrations, WTFMove(completionHandler));
61 m_updatedRegistrations.clear();
62}
63
64void RegistrationStore::clearAll(CompletionHandler<void()>&& completionHandler)
65{
66 m_needsPushingChanges = false;
67 m_updatedRegistrations.clear();
68 m_databasePushTimer.stop();
69 m_database->clearAll(WTFMove(completionHandler));
70}
71
72void RegistrationStore::flushChanges(CompletionHandler<void()>&& completionHandler)
73{
74 if (m_databasePushTimer.isActive()) {
75 pushChangesToDatabase(WTFMove(completionHandler));
76 m_databasePushTimer.stop();
77 return;
78 }
79 completionHandler();
80}
81
82void RegistrationStore::startSuspension(CompletionHandler<void()>&& completionHandler)
83{
84 m_isSuspended = true;
85 closeDatabase(WTFMove(completionHandler));
86}
87
88void RegistrationStore::closeDatabase(CompletionHandler<void()>&& completionHandler)
89{
90 m_database->close(WTFMove(completionHandler));
91}
92
93void RegistrationStore::endSuspension()
94{
95 m_isSuspended = false;
96 if (m_needsPushingChanges)
97 scheduleDatabasePushIfNecessary();
98}
99
100void RegistrationStore::updateRegistration(const ServiceWorkerContextData& data)
101{
102 ASSERT(isMainThread());
103 ASSERT(!data.registration.key.isEmpty());
104 if (data.registration.key.isEmpty())
105 return;
106
107 m_updatedRegistrations.set(data.registration.key, data);
108 scheduleDatabasePushIfNecessary();
109}
110
111void RegistrationStore::removeRegistration(const ServiceWorkerRegistrationKey& key)
112{
113 ASSERT(isMainThread());
114 ASSERT(!key.isEmpty());
115 if (key.isEmpty())
116 return;
117
118 m_updatedRegistrations.set(key, std::nullopt);
119 scheduleDatabasePushIfNecessary();
120}
121
122void RegistrationStore::addRegistrationFromDatabase(ServiceWorkerContextData&& data)
123{
124 ASSERT(!data.registration.key.isEmpty());
125 if (data.registration.key.isEmpty())
126 return;
127
128 m_server.addRegistrationFromStore(WTFMove(data));
129}

131void RegistrationStore::didSaveWorkerScriptsToDisk(ServiceWorkerIdentifier serviceWorkerIdentifier, ScriptBuffer&& mainScript, HashMap<URL, ScriptBuffer>&& importedScripts)
132{
133 m_server.didSaveWorkerScriptsToDisk(serviceWorkerIdentifier, WTFMove(mainScript), WTFMove(importedScripts));
134}

136void RegistrationStore::databaseFailedToOpen()

139}

141void RegistrationStore::databaseOpenedAndRecordsImported()

143 m_server.registrationStoreImportComplete();
