66SWServer::~SWServer()
67{
68 // Destroy the remaining connections before the SWServer gets destroyed since they have a raw pointer
69 // to the server and since they try to unregister clients from the server in their destructor.
70 auto connections = WTFMove(m_connections);
71 connections.clear();
72
73 for (auto& callback : std::exchange(m_importCompletedCallbacks, { }))
74 callback();
75
76 Vector<SWServerWorker*> runningWorkers;
77 for (auto& worker : m_runningOrTerminatingWorkers.values()) {
78 if (worker->isRunning())
79 runningWorkers.append(worker.ptr());
80 }
81 for (auto& runningWorker : runningWorkers)
82 runningWorker->terminate();
83
84 allServers().remove(this);
85}

157void SWServer::addRegistrationFromStore(ServiceWorkerContextData&& data)
158{
159 // Pages should not have been able to make a new registration to this key while the import was still taking place.
160 ASSERT(!m_scopeToRegistrationMap.contains(data.registration.key));
161
162 auto registrableDomain = WebCore::RegistrableDomain(data.scriptURL);
163 validateRegistrationDomain(registrableDomain, [this, weakThis = makeWeakPtr(this), data = WTFMove(data)] (bool isValid) mutable {
164 if (!weakThis)
165 return;
166 if (m_hasServiceWorkerEntitlement || isValid) {
167 auto registration = makeUnique<SWServerRegistration>(*this, data.registration.key, data.registration.updateViaCache, data.registration.scopeURL, data.scriptURL);
168 registration->setLastUpdateTime(data.registration.lastUpdateTime);
169 auto registrationPtr = registration.get();
170 addRegistration(WTFMove(registration));
171
172 auto worker = SWServerWorker::create(*this, *registrationPtr, data.scriptURL, data.script, data.certificateInfo, data.contentSecurityPolicy, WTFMove(data.referrerPolicy), data.workerType, data.serviceWorkerIdentifier, WTFMove(data.scriptResourceMap));
173 registrationPtr->updateRegistrationState(ServiceWorkerRegistrationState::Active, worker.ptr());
174 worker->setState(ServiceWorkerState::Activated);
175 }

209Vector<ServiceWorkerRegistrationData> SWServer::getRegistrations(const SecurityOriginData& topOrigin, const URL& clientURL)
210{
211 Vector<SWServerRegistration*> matchingRegistrations;
212 for (auto& item : m_scopeToRegistrationMap) {
213 if (item.key.originIsMatching(topOrigin, clientURL)) {
214 auto* registration = item.value.get();
215 ASSERT(registration);
216 if (registration)
217 matchingRegistrations.append(registration);
218 }
219 }
220 // The specification mandates that registrations are returned in the insertion order.
221 std::sort(matchingRegistrations.begin(), matchingRegistrations.end(), [](auto& a, auto& b) {
222 return a->creationTime() < b->creationTime();
223 });
224 Vector<ServiceWorkerRegistrationData> matchingRegistrationDatas;
225 matchingRegistrationDatas.reserveInitialCapacity(matchingRegistrations.size());
226 for (auto* registration : matchingRegistrations)
227 matchingRegistrationDatas.uncheckedAppend(registration->data());
228 return matchingRegistrationDatas;
229}

265void SWServer::clear(const SecurityOriginData& securityOrigin, CompletionHandler<void()>&& completionHandler)
266{
267 if (!m_importCompleted) {
268 m_clearCompletionCallbacks.append([this, securityOrigin, completionHandler = WTFMove(completionHandler)] () mutable {
269 ASSERT(m_importCompleted);
270 clear(securityOrigin, WTFMove(completionHandler));
271 });
272 return;
273 }
274
275 m_jobQueues.removeIf([&](auto& keyAndValue) {
276 return keyAndValue.key.relatesToOrigin(securityOrigin);
277 });
278
279 Vector<SWServerRegistration*> registrationsToRemove;
280 for (auto& registration : m_registrations.values()) {
281 if (registration->key().relatesToOrigin(securityOrigin))
282 registrationsToRemove.append(registration.get());
283 }
284
285 for (auto& contextDatas : m_pendingContextDatas.values()) {
286 contextDatas.removeAllMatching([&](auto& contextData) {
287 return contextData.registration.key.relatesToOrigin(securityOrigin);
288 });
289 }
290
291 if (registrationsToRemove.isEmpty()) {
292 completionHandler();
293 return;
294 }
295
296 // Calling SWServerRegistration::clear() takes care of updating m_registrations, m_originStore and m_registrationStore.
297 for (auto* registration : registrationsToRemove)
298 registration->clear();
299
300 if (m_registrationStore)
301 m_registrationStore->flushChanges(WTFMove(completionHandler));
302}

359// https://w3c.github.io/ServiceWorker/#schedule-job-algorithm
360void SWServer::scheduleJob(ServiceWorkerJobData&& jobData)
361{
362 ASSERT(m_connections.contains(jobData.connectionIdentifier()) || jobData.connectionIdentifier() == Process::identifier());
363
364 validateRegistrationDomain(WebCore::RegistrableDomain(jobData.scriptURL), [this, weakThis = makeWeakPtr(this), jobData = WTFMove(jobData)] (bool isValid) mutable {
365 if (!weakThis)
366 return;
367 if (m_hasServiceWorkerEntitlement || isValid) {
368 auto& jobQueue = *m_jobQueues.ensure(jobData.registrationKey(), [this, &jobData] {
369 return makeUnique<SWServerJobQueue>(*this, jobData.registrationKey());
370 }).iterator->value;
371
372 if (!jobQueue.size()) {
373 jobQueue.enqueueJob(WTFMove(jobData));
374 jobQueue.runNextJob();
375 return;
376 }
377 auto& lastJob = jobQueue.lastJob();
378 if (jobData.isEquivalent(lastJob)) {
379 // FIXME: Per the spec, check if this job is equivalent to the last job on the queue.
380 // If it is, stack it along with that job. For now, we just make sure to not call soft-update too often.
381 if (jobData.type == ServiceWorkerJobType::Update && jobData.connectionIdentifier() == Process::identifier())
382 return;
383 }
384 jobQueue.enqueueJob(WTFMove(jobData));
385 if (jobQueue.size() == 1)
386 jobQueue.runNextJob();
387 } else
388 rejectJob(jobData, { TypeError, "Job rejected for non app-bound domain" });
389 });
390}

444void SWServer::startScriptFetch(const ServiceWorkerJobData& jobData, SWServerRegistration& registration)
445{
446 LOG(ServiceWorker, "Server issuing startScriptFetch for current job %s in client", jobData.identifier().loggingString().utf8().data());
447
448 // Set request's cache mode to "no-cache" if any of the following are true:
449 // - registration's update via cache mode is not "all".
450 // - job's force bypass cache flag is set.
451 // - newestWorker is not null, and registration's last update check time is not null and the time difference in seconds calculated by the
452 // current time minus registration's last update check time is greater than 86400.
453 bool shouldRefreshCache = registration.updateViaCache() != ServiceWorkerUpdateViaCache::All || (registration.getNewestWorker() && registration.isStale());
454
455 auto* connection = m_connections.get(jobData.connectionIdentifier());
456 if (connection) {
457 connection->startScriptFetchInClient(jobData.identifier().jobIdentifier, jobData.registrationKey(), shouldRefreshCache ? FetchOptions::Cache::NoCache : FetchOptions::Cache::Default);
458 return;
459 }
460 if (jobData.connectionIdentifier() == Process::identifier()) {
461 ASSERT(jobData.type == ServiceWorkerJobType::Update);
462 // This is a soft-update job, create directly a network load to fetch the script.
463 ResourceRequest request { jobData.scriptURL };
464
465 auto topOrigin = jobData.topOrigin.securityOrigin();
466 auto origin = SecurityOrigin::create(jobData.scriptURL);
467
468 request.setDomainForCachePartition(topOrigin->domainForCachePartition());
469 request.setAllowCookies(true);
470 request.setFirstPartyForCookies(originURL(topOrigin));
471
472 request.setHTTPHeaderField(HTTPHeaderName::Origin, origin->toString());
473 request.setHTTPHeaderField(HTTPHeaderName::ServiceWorker, "script"_s);
474 request.setHTTPReferrer(originURL(origin).string());
475 request.setHTTPUserAgent(serviceWorkerClientUserAgent(ClientOrigin { jobData.topOrigin, SecurityOrigin::create(jobData.scriptURL)->data() }));
476 request.setPriority(ResourceLoadPriority::Low);
477 request.setIsAppBound(registration.isAppBound());
478
479 m_softUpdateCallback(ServiceWorkerJobData { jobData }, shouldRefreshCache, WTFMove(request), [this, weakThis = makeWeakPtr(this)](auto& result) {
480 if (!weakThis)
481 return;
482 scriptFetchFinished(result);
483 });
484 return;
485 }
486 ASSERT_WITH_MESSAGE(connection, "If the connection was lost, this job should have been cancelled");
487}

564// https://w3c.github.io/ServiceWorker/#clients-getall
565void SWServer::matchAll(SWServerWorker& worker, const ServiceWorkerClientQueryOptions& options, ServiceWorkerClientsMatchAllCallback&& callback)
566{
567 // FIXME: Support reserved client filtering.
568 // FIXME: Support WindowClient additional properties.
569
570 Vector<ServiceWorkerClientData> matchingClients;
571 forEachClientForOrigin(worker.origin(), [&](auto& clientData) {
572 if (!options.includeUncontrolled) {
573 auto registrationIdentifier = m_clientToControllingRegistration.get(clientData.identifier);
574 if (worker.data().registrationIdentifier != registrationIdentifier)
575 return;
576 if (&worker != this->activeWorkerFromRegistrationID(registrationIdentifier))
577 return;
578 }
579 if (options.type != ServiceWorkerClientType::All && options.type != clientData.type)
580 return;
581 matchingClients.append(clientData);
582 });
583 callback(WTFMove(matchingClients));
584}

586void SWServer::forEachClientForOrigin(const ClientOrigin& origin, const WTF::Function<void(ServiceWorkerClientData&)>& apply)
587{

590 return;
591
592 for (auto& clientIdentifier : iterator->value.identifiers) {
593 auto clientIterator = m_clientsById.find(clientIdentifier);
594 ASSERT(clientIterator != m_clientsById.end());
595 apply(clientIterator->value);
596 }

599std::optional<ExceptionData> SWServer::claim(SWServerWorker& worker)
600{
601 auto* registration = worker.registration();
602 if (!registration || &worker != registration->activeWorker())
603 return ExceptionData { InvalidStateError, "Service worker is not active"_s };
604
605 auto& origin = worker.origin();
606 forEachClientForOrigin(origin, [&](auto& clientData) {
607 if (doRegistrationMatching(origin.topOrigin, clientData.url) != registration)
608 return;
609
610 auto result = m_clientToControllingRegistration.add(clientData.identifier, registration->identifier());
611 if (!result.isNewEntry) {
612 auto previousIdentifier = result.iterator->value;
613 if (previousIdentifier == registration->identifier())
614 return;
615 result.iterator->value = registration->identifier();
616 if (auto* controllingRegistration = m_registrations.get(previousIdentifier))
617 controllingRegistration->removeClientUsingRegistration(clientData.identifier);
618 }
619 registration->controlClient(clientData.identifier);
620 });
621 return { };
622}

704void SWServer::installContextData(const ServiceWorkerContextData& data)
705{
706 ASSERT_WITH_MESSAGE(!data.loadedFromDisk, "Workers we just read from disk should only be launched as needed");
707
708 if (data.jobDataIdentifier) {
709 // Abort if the job that scheduled this has been cancelled.
710 auto* jobQueue = m_jobQueues.get(data.registration.key);
711 if (!jobQueue || !jobQueue->isCurrentlyProcessingJob(*data.jobDataIdentifier))
712 return;
713 }
714
715 auto* registration = m_scopeToRegistrationMap.get(data.registration.key).get();
716 auto worker = SWServerWorker::create(*this, *registration, data.scriptURL, data.script, data.certificateInfo, data.contentSecurityPolicy, String { data.referrerPolicy }, data.workerType, data.serviceWorkerIdentifier, HashMap<URL, ServiceWorkerContextData::ImportedScript> { data.scriptResourceMap });
717
718 auto* connection = worker->contextConnection();
719 ASSERT(connection);
720
721 registration->setPreInstallationWorker(worker.ptr());
722 worker->setState(SWServerWorker::State::Running);
723 auto userAgent = worker->userAgent();
724 auto result = m_runningOrTerminatingWorkers.add(data.serviceWorkerIdentifier, WTFMove(worker));
725 ASSERT_UNUSED(result, result.isNewEntry);
726
727 connection->installServiceWorkerContext(data, userAgent);
728}

730void SWServer::runServiceWorkerIfNecessary(ServiceWorkerIdentifier identifier, RunServiceWorkerCallback&& callback)
731{
732 auto* worker = workerByID(identifier);
733 if (!worker) {
734 callback(nullptr);
735 return;
736 }
737
738 auto* contextConnection = worker->contextConnection();
739 if (worker->isRunning()) {
740 ASSERT(contextConnection);
741 callback(contextConnection);
742 return;
743 }
744
745 if (worker->state() == ServiceWorkerState::Redundant) {
746 callback(nullptr);
747 return;
748 }
749
750 if (worker->isTerminating()) {
751 worker->whenTerminated([this, weakThis = makeWeakPtr(this), identifier, callback = WTFMove(callback)]() mutable {
752 if (!weakThis)
753 return callback(nullptr);
754 runServiceWorkerIfNecessary(identifier, WTFMove(callback));
755 });
756 return;
757 }
758
759 if (!contextConnection) {
760 auto& serviceWorkerRunRequestsForOrigin = m_serviceWorkerRunRequests.ensure(worker->registrableDomain(), [] {
761 return HashMap<ServiceWorkerIdentifier, Vector<RunServiceWorkerCallback>> { };
762 }).iterator->value;
763 serviceWorkerRunRequestsForOrigin.ensure(identifier, [&] {
764 return Vector<RunServiceWorkerCallback> { };
765 }).iterator->value.append(WTFMove(callback));
766
767 createContextConnection(worker->registrableDomain());
768 return;
769 }
770
771 bool success = runServiceWorker(identifier);
772 callback(success ? contextConnection : nullptr);
773}

775bool SWServer::runServiceWorker(ServiceWorkerIdentifier identifier)
776{
777 auto* worker = workerByID(identifier);
778 if (!worker)
779 return false;
780
781 // If the registration for a worker has been removed then the request to run
782 // the worker is moot.
783 if (!worker->registration())
784 return false;
785
786 ASSERT(!worker->isTerminating());
787 ASSERT(!m_runningOrTerminatingWorkers.contains(identifier));
788 m_runningOrTerminatingWorkers.add(identifier, *worker);
789
790 worker->setState(SWServerWorker::State::Running);
791
792 auto* contextConnection = worker->contextConnection();
793 ASSERT(contextConnection);
794
795 contextConnection->installServiceWorkerContext(worker->contextData(), worker->userAgent());
796
797 return true;
798}

897void SWServer::registerServiceWorkerClient(ClientOrigin&& clientOrigin, ServiceWorkerClientData&& data, const std::optional<ServiceWorkerRegistrationIdentifier>& controllingServiceWorkerRegistrationIdentifier, String&& userAgent)
898{
899 auto clientIdentifier = data.identifier;
900
901 // Update the app-bound value if the new client is app-bound and the current clients for the origin are not marked app-bound.
902 if (data.lastNavigationWasAppBound == LastNavigationWasAppBound::Yes) {
903 if (clientIsAppBoundForRegistrableDomain(clientOrigin.clientRegistrableDomain()) == LastNavigationWasAppBound::No)
904 updateAppBoundValueForWorkers(clientOrigin, data.lastNavigationWasAppBound);
905 }
906
907 ASSERT(!m_clientsById.contains(clientIdentifier));
908 m_clientsById.add(clientIdentifier, WTFMove(data));
909
910 auto& clientIdentifiersForOrigin = m_clientIdentifiersPerOrigin.ensure(clientOrigin, [] {
911 return Clients { };
912 }).iterator->value;
913
914 ASSERT(!clientIdentifiersForOrigin.identifiers.contains(clientIdentifier));
915 clientIdentifiersForOrigin.identifiers.append(clientIdentifier);
916
917 if (!clientIdentifiersForOrigin.userAgent.isNull() && clientIdentifiersForOrigin.userAgent != userAgent)
918 RELEASE_LOG_ERROR(ServiceWorker, "%p - SWServer::registerServiceWorkerClient: Service worker has clients using different user agents", this);
919 clientIdentifiersForOrigin.userAgent = WTFMove(userAgent);
920
921 clientIdentifiersForOrigin.terminateServiceWorkersTimer = nullptr;
922
923 m_clientsByRegistrableDomain.ensure(clientOrigin.clientRegistrableDomain(), [] {
924 return HashSet<ServiceWorkerClientIdentifier> { };
925 }).iterator->value.add(clientIdentifier);
926
927 if (!controllingServiceWorkerRegistrationIdentifier)
928 return;
929
930 auto* controllingRegistration = m_registrations.get(*controllingServiceWorkerRegistrationIdentifier);
931 if (!controllingRegistration || !controllingRegistration->activeWorker())
932 return;
933
934 controllingRegistration->addClientUsingRegistration(clientIdentifier);
935 ASSERT(!m_clientToControllingRegistration.contains(clientIdentifier));
936 m_clientToControllingRegistration.add(clientIdentifier, *controllingServiceWorkerRegistrationIdentifier);
937}

939void SWServer::unregisterServiceWorkerClient(const ClientOrigin& clientOrigin, ServiceWorkerClientIdentifier clientIdentifier)
940{
941 auto clientRegistrableDomain = clientOrigin.clientRegistrableDomain();
942 auto appBoundValueBefore = clientIsAppBoundForRegistrableDomain(clientOrigin.clientRegistrableDomain());
943
944 bool wasRemoved = m_clientsById.remove(clientIdentifier);
945 ASSERT_UNUSED(wasRemoved, wasRemoved);
946
947 auto iterator = m_clientIdentifiersPerOrigin.find(clientOrigin);
948 ASSERT(iterator != m_clientIdentifiersPerOrigin.end());
949
950 auto& clientIdentifiers = iterator->value.identifiers;
951 clientIdentifiers.removeFirstMatching([&] (const auto& identifier) {
952 return clientIdentifier == identifier;
953 });
954
955 if (clientIdentifiers.isEmpty()) {
956 ASSERT(!iterator->value.terminateServiceWorkersTimer);
957 iterator->value.terminateServiceWorkersTimer = makeUnique<Timer>([clientOrigin, clientRegistrableDomain, this] {
958 Vector<SWServerWorker*> workersToTerminate;
959 for (auto& worker : m_runningOrTerminatingWorkers.values()) {
960 if (worker->isRunning() && worker->origin() == clientOrigin)
961 workersToTerminate.append(worker.ptr());
962 }
963 for (auto* worker : workersToTerminate)
964 worker->terminate();
965
966 if (!m_clientsByRegistrableDomain.contains(clientRegistrableDomain)) {
967 if (auto* connection = contextConnectionForRegistrableDomain(clientRegistrableDomain)) {
968 removeContextConnection(*connection);
969 connection->connectionIsNoLongerNeeded();
970 }
971 }
972
973 m_clientIdentifiersPerOrigin.remove(clientOrigin);
974 });
975 iterator->value.terminateServiceWorkersTimer->startOneShot(m_isProcessTerminationDelayEnabled && !MemoryPressureHandler::singleton().isUnderMemoryPressure() ? defaultTerminationDelay : 0_s);
976 }
977
978 auto clientsByRegistrableDomainIterator = m_clientsByRegistrableDomain.find(clientRegistrableDomain);
979 ASSERT(clientsByRegistrableDomainIterator != m_clientsByRegistrableDomain.end());
980 auto& clientsForRegistrableDomain = clientsByRegistrableDomainIterator->value;
981 clientsForRegistrableDomain.remove(clientIdentifier);
982 if (clientsForRegistrableDomain.isEmpty())
983 m_clientsByRegistrableDomain.remove(clientsByRegistrableDomainIterator);
984
985 // If the app-bound value changed after this client was removed, we know it was the only app-bound
986 // client for its origin, and we should update all workers to reflect this.
987 auto appBoundValueAfter = clientIsAppBoundForRegistrableDomain(clientOrigin.clientRegistrableDomain());
988 if (appBoundValueBefore != appBoundValueAfter)
989 updateAppBoundValueForWorkers(clientOrigin, appBoundValueAfter);
990
991 auto registrationIterator = m_clientToControllingRegistration.find(clientIdentifier);
992 if (registrationIterator == m_clientToControllingRegistration.end())
993 return;
994
995 if (auto* registration = m_registrations.get(registrationIterator->value))
996 registration->removeClientUsingRegistration(clientIdentifier);
997
998 m_clientToControllingRegistration.remove(registrationIterator);
999}

1148// https://w3c.github.io/ServiceWorker/#soft-update
1149void SWServer::softUpdate(SWServerRegistration& registration)
1150{
1151 // Let newestWorker be the result of running Get Newest Worker algorithm passing registration as its argument.
1152 // If newestWorker is null, abort these steps.
1153 auto* newestWorker = registration.getNewestWorker();
1154 if (!newestWorker)
1155 return;
1156
1157 ServiceWorkerJobData jobData(Process::identifier(), ServiceWorkerIdentifier::generate());
1158 jobData.scriptURL = registration.scriptURL();
1159 jobData.topOrigin = registration.key().topOrigin();
1160 jobData.scopeURL = registration.scopeURLWithoutFragment();
1161 jobData.workerType = newestWorker->type();
1162 jobData.type = ServiceWorkerJobType::Update;
1163 scheduleJob(WTFMove(jobData));
1164}
