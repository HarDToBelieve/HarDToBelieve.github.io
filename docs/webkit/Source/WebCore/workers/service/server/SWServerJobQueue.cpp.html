55bool SWServerJobQueue::isCurrentlyProcessingJob(const ServiceWorkerJobDataIdentifier& jobDataIdentifier) const
56{
57 return !m_jobQueue.isEmpty() && firstJob().identifier() == jobDataIdentifier;

60static bool doCertificatesMatch(const CertificateInfo& first, const CertificateInfo& second)
61{
62#if PLATFORM(COCOA) && HAVE(SEC_TRUST_SERIALIZATION)
63 return first.trust() == second.trust() || certificatesMatch(first.trust(), second.trust());
64#else
65 // FIXME: Add support for certificate matching in CertificateInfo.
66 UNUSED_PARAM(first);
67 UNUSED_PARAM(second);
68 return true;
69#endif

72void SWServerJobQueue::scriptFetchFinished(const ServiceWorkerFetchResult& result)
73{
74 if (!isCurrentlyProcessingJob(result.jobDataIdentifier))
75 return;
76
77 auto& job = firstJob();
78
79 auto* registration = m_server.getRegistration(m_registrationKey);
80 if (!registration)
81 return;
82
83 auto* newestWorker = registration->getNewestWorker();
84
85 if (!result.scriptError.isNull()) {
86 // Invoke Reject Job Promise with job and TypeError.
87 m_server.rejectJob(job, ExceptionData { TypeError, makeString("Script URL ", job.scriptURL.string(), " fetch resulted in error: ", result.scriptError.localizedDescription()) });
88
89 // If newestWorker is null, invoke Clear Registration algorithm passing registration as its argument.
90 if (!newestWorker)
91 registration->clear();
92
93 // Invoke Finish Job with job and abort these steps.
94 finishCurrentJob();
95 return;
96 }
97
98 registration->setLastUpdateTime(WallTime::now());
99
100 // If newestWorker is not null, newestWorker's script url equals job's script url with the exclude fragments
101 // flag set, and script's source text is a byte-for-byte match with newestWorker's script resource's source
102 // text, then:
103 if (newestWorker && equalIgnoringFragmentIdentifier(newestWorker->scriptURL(), job.scriptURL) && newestWorker->type() == job.workerType && result.script == newestWorker->script() && doCertificatesMatch(result.certificateInfo, newestWorker->certificateInfo())) {
104 RELEASE_LOG(ServiceWorker, "%p - SWServerJobQueue::scriptFetchFinished, script and certificate are matching for registrationID=%llu", this, registration->identifier().toUInt64());
105 // FIXME: for non classic scripts, check the script’s module record's [[ECMAScriptCode]].
106
107 // Invoke Resolve Job Promise with job and registration.
108 m_server.resolveRegistrationJob(job, registration->data(), ShouldNotifyWhenResolved::No);
109
110 // Invoke Finish Job with job and abort these steps.
111 finishCurrentJob();
112 return;
113 }
114
115 // FIXME: Update all the imported scripts as per spec. For now, we just do as if there is none.
116
117 m_server.updateWorker(job.identifier(), *registration, job.scriptURL, result.script, result.certificateInfo, result.contentSecurityPolicy, result.referrerPolicy, job.workerType, { });
118}

120// https://w3c.github.io/ServiceWorker/#update-algorithm
121void SWServerJobQueue::scriptContextFailedToStart(const ServiceWorkerJobDataIdentifier& jobDataIdentifier, ServiceWorkerIdentifier, const String& message)
122{
123 if (!isCurrentlyProcessingJob(jobDataIdentifier))
124 return;
125
126 // If an uncaught runtime script error occurs during the above step, then:
127 auto* registration = m_server.getRegistration(m_registrationKey);
128 ASSERT(registration);
129
130 ASSERT(registration->preInstallationWorker());
131 registration->preInstallationWorker()->terminate();
132 registration->setPreInstallationWorker(nullptr);
133
134 // Invoke Reject Job Promise with job and TypeError.
135 m_server.rejectJob(firstJob(), { TypeError, message });
136
137 // If newestWorker is null, invoke Clear Registration algorithm passing registration as its argument.
138 if (!registration->getNewestWorker())
139 registration->clear();
140
141 // Invoke Finish Job with job and abort these steps.
142 finishCurrentJob();
143}

145void SWServerJobQueue::scriptContextStarted(const ServiceWorkerJobDataIdentifier& jobDataIdentifier, ServiceWorkerIdentifier identifier)
146{
147 if (!isCurrentlyProcessingJob(jobDataIdentifier))
148 return;
149
150 auto* registration = m_server.getRegistration(m_registrationKey);
151 ASSERT(registration);
152
153 install(*registration, identifier);
154}

156// https://w3c.github.io/ServiceWorker/#install
157void SWServerJobQueue::install(SWServerRegistration& registration, ServiceWorkerIdentifier installingWorker)
158{
159 // The Install algorithm should never be invoked with a null worker.
160 auto* worker = m_server.workerByID(installingWorker);
161 RELEASE_ASSERT(worker);
162
163 ASSERT(registration.preInstallationWorker() == worker);
164 registration.setPreInstallationWorker(nullptr);
165
166 registration.updateRegistrationState(ServiceWorkerRegistrationState::Installing, worker);
167 registration.updateWorkerState(*worker, ServiceWorkerState::Installing);
168
169 // Invoke Resolve Job Promise with job and registration.
170 m_server.resolveRegistrationJob(firstJob(), registration.data(), ShouldNotifyWhenResolved::Yes);
171
172 // FIXME: https://bugs.webkit.org/show_bug.cgi?id=215122. We do not need to wait for the registration promise to resolve to continue the install steps.
173}

175// https://w3c.github.io/ServiceWorker/#install (after resolving promise).
176void SWServerJobQueue::didResolveRegistrationPromise()
177{
178 auto* registration = m_server.getRegistration(m_registrationKey);
179 ASSERT(registration);
180 ASSERT(registration->installingWorker());
181
182 if (!registration || !registration->installingWorker()) {
183 RELEASE_LOG_ERROR(ServiceWorker, "%p - SWServerJobQueue::didResolveRegistrationPromise with null registration (%d) or null worker", this, !!registration);
184 return;
185 }
186
187 RELEASE_LOG(ServiceWorker, "%p - SWServerJobQueue::didResolveRegistrationPromise: RegistrationID=%llu. Now proceeding with install", this, registration->identifier().toUInt64());
188
189 // Queue a task to fire an event named updatefound at all the ServiceWorkerRegistration objects
190 // for all the service worker clients whose creation URL matches registration's scope url and
191 // all the service workers whose containing service worker registration is registration.
192 registration->fireUpdateFoundEvent();
193
194 // Queue a task to fire the InstallEvent.
195 ASSERT(registration->installingWorker());
196 m_server.fireInstallEvent(*registration->installingWorker());
197}

199// https://w3c.github.io/ServiceWorker/#install
200void SWServerJobQueue::didFinishInstall(const ServiceWorkerJobDataIdentifier& jobDataIdentifier, SWServerWorker& worker, bool wasSuccessful)
201{
202 if (!isCurrentlyProcessingJob(jobDataIdentifier))
203 return;
204
205 auto* registration = worker.registration();
206 ASSERT(registration);
207 ASSERT(registration->installingWorker() == &worker);
208
209 if (!wasSuccessful) {
210 worker.terminate();
211 // Run the Update Registration State algorithm passing registration, "installing" and null as the arguments.
212 registration->updateRegistrationState(ServiceWorkerRegistrationState::Installing, nullptr);
213 // Run the Update Worker State algorithm passing registration's installing worker and redundant as the arguments.
214 registration->updateWorkerState(worker, ServiceWorkerState::Redundant);
215
216 // If newestWorker is null, invoke Clear Registration algorithm passing registration as its argument.
217 if (!registration->getNewestWorker())
218 registration->clear();
219
220 // Invoke Finish Job with job and abort these steps.
221 finishCurrentJob();
222 return;
223 }
224
225 if (auto* waitingWorker = registration->waitingWorker()) {
226 waitingWorker->terminate();
227 registration->updateWorkerState(*waitingWorker, ServiceWorkerState::Redundant);
228 }
229
230 registration->updateRegistrationState(ServiceWorkerRegistrationState::Waiting, &worker);
231 registration->updateRegistrationState(ServiceWorkerRegistrationState::Installing, nullptr);
232 registration->updateWorkerState(worker, ServiceWorkerState::Installed);
233
234 finishCurrentJob();
235
236 // FIXME: Wait for all the tasks queued by Update Worker State invoked in this algorithm have executed.
237 registration->tryActivate();
238}

244 ASSERT(!m_jobTimer.isActive());
245 m_jobTimer.startOneShot(0_s);

248void SWServerJobQueue::runNextJobSynchronously()
249{
250 ASSERT(!m_jobQueue.isEmpty());
251 if (m_jobQueue.isEmpty())
252 return;
253
254 auto& job = firstJob();
255 switch (job.type) {
256 case ServiceWorkerJobType::Register:
257 runRegisterJob(job);
258 return;
259 case ServiceWorkerJobType::Unregister:
260 runUnregisterJob(job);
261 return;
262 case ServiceWorkerJobType::Update:
263 runUpdateJob(job);
264 return;
265 }
266
267 ASSERT_NOT_REACHED();
268}

270// https://w3c.github.io/ServiceWorker/#register-algorithm
271void SWServerJobQueue::runRegisterJob(const ServiceWorkerJobData& job)
272{
273 ASSERT(job.type == ServiceWorkerJobType::Register);
274
275 if (!shouldTreatAsPotentiallyTrustworthy(job.scriptURL) && !m_server.canHandleScheme(job.scriptURL.protocol()))
276 return rejectCurrentJob(ExceptionData { SecurityError, "Script URL is not potentially trustworthy"_s });
277
278 // If the origin of job's script url is not job's referrer's origin, then:
279 if (!protocolHostAndPortAreEqual(job.scriptURL, job.clientCreationURL))
280 return rejectCurrentJob(ExceptionData { SecurityError, "Script origin does not match the registering client's origin"_s });
281
282 // If the origin of job's scope url is not job's referrer's origin, then:
283 if (!protocolHostAndPortAreEqual(job.scopeURL, job.clientCreationURL))
284 return rejectCurrentJob(ExceptionData { SecurityError, "Scope origin does not match the registering client's origin"_s });
285
286 // If registration is not null (in our parlance "empty"), then:
287 if (auto* registration = m_server.getRegistration(m_registrationKey)) {
288 auto* newestWorker = registration->getNewestWorker();
289 if (newestWorker && equalIgnoringFragmentIdentifier(job.scriptURL, newestWorker->scriptURL()) && job.workerType == newestWorker->type() && job.registrationOptions.updateViaCache == registration->updateViaCache()) {
290 RELEASE_LOG(ServiceWorker, "%p - SWServerJobQueue::runRegisterJob: Found directly reusable registration %llu for job %s (DONE)", this, registration->identifier().toUInt64(), job.identifier().loggingString().utf8().data());
291 m_server.resolveRegistrationJob(job, registration->data(), ShouldNotifyWhenResolved::No);
292 finishCurrentJob();
293 return;
294 }
295 // This is not specified yet (https://github.com/w3c/ServiceWorker/issues/1189).
296 if (registration->updateViaCache() != job.registrationOptions.updateViaCache)
297 registration->setUpdateViaCache(job.registrationOptions.updateViaCache);
298 RELEASE_LOG(ServiceWorker, "%p - SWServerJobQueue::runRegisterJob: Found registration %llu for job %s but it needs updating", this, registration->identifier().toUInt64(), job.identifier().loggingString().utf8().data());
299 } else {
300 auto newRegistration = makeUnique<SWServerRegistration>(m_server, m_registrationKey, job.registrationOptions.updateViaCache, job.scopeURL, job.scriptURL);
301 m_server.addRegistration(WTFMove(newRegistration));
302
303 RELEASE_LOG(ServiceWorker, "%p - SWServerJobQueue::runRegisterJob: No existing registration for job %s, constructing a new one.", this, job.identifier().loggingString().utf8().data());
304 }
305
306 runUpdateJob(job);
307}

309// https://w3c.github.io/ServiceWorker/#unregister-algorithm
310void SWServerJobQueue::runUnregisterJob(const ServiceWorkerJobData& job)
311{
312 // If the origin of job's scope url is not job's client's origin, then:
313 if (!protocolHostAndPortAreEqual(job.scopeURL, job.clientCreationURL))
314 return rejectCurrentJob(ExceptionData { SecurityError, "Origin of scope URL does not match the client's origin"_s });
315
316 // Let registration be the result of running "Get Registration" algorithm passing job's scope url as the argument.
317 auto* registration = m_server.getRegistration(m_registrationKey);
318
319 // If registration is null, then:
320 if (!registration) {
321 // Invoke Resolve Job Promise with job and false.
322 m_server.resolveUnregistrationJob(job, m_registrationKey, false);
323 finishCurrentJob();
324 return;
325 }
326
327 // Remove scope to registration map[job’s scope url].
328 m_server.removeFromScopeToRegistrationMap(m_registrationKey);
329
330 // Invoke Resolve Job Promise with job and true.
331 m_server.resolveUnregistrationJob(job, m_registrationKey, true);
332
333 // Invoke Try Clear Registration with registration.
334 registration->tryClear();
335 finishCurrentJob();
336}

338// https://w3c.github.io/ServiceWorker/#update-algorithm
339void SWServerJobQueue::runUpdateJob(const ServiceWorkerJobData& job)
340{
341 // Let registration be the result of running the Get Registration algorithm passing job's scope url as the argument.
342 auto* registration = m_server.getRegistration(m_registrationKey);
343
344 // If registration is null (in our parlance "empty") or registration's uninstalling flag is set, then:
345 if (!registration)
346 return rejectCurrentJob(ExceptionData { TypeError, "Cannot update a null/nonexistent service worker registration"_s });
347
348 // Let newestWorker be the result of running Get Newest Worker algorithm passing registration as the argument.
349 auto* newestWorker = registration->getNewestWorker();
350
351 // If job’s type is update, and newestWorker is not null and its script url does not equal job’s script url, then:
352 if (job.type == ServiceWorkerJobType::Update && newestWorker && !equalIgnoringFragmentIdentifier(job.scriptURL, newestWorker->scriptURL()))
353 return rejectCurrentJob(ExceptionData { TypeError, "Cannot update a service worker with a requested script URL whose newest worker has a different script URL"_s });
354
355 m_server.startScriptFetch(job, *registration);
356}

358void SWServerJobQueue::rejectCurrentJob(const ExceptionData& exceptionData)
359{
360 m_server.rejectJob(firstJob(), exceptionData);
361
362 finishCurrentJob();
363}

365// https://w3c.github.io/ServiceWorker/#finish-job
366void SWServerJobQueue::finishCurrentJob()
367{
368 ASSERT(!m_jobTimer.isActive());
369
370 m_jobQueue.removeFirst();
371 if (!m_jobQueue.isEmpty())
372 runNextJob();
373}

375void SWServerJobQueue::removeAllJobsMatching(const WTF::Function<bool(ServiceWorkerJobData&)>& matches)
376{
377 bool isFirst = true;
378 bool didRemoveFirstJob = false;
379 m_jobQueue.removeAllMatching([&](auto& job) {
380 bool shouldRemove = matches(job);
381 if (isFirst) {
382 isFirst = false;
383 if (shouldRemove)
384 didRemoveFirstJob = true;
385 }
386 return shouldRemove;
387 });
388
389 if (m_jobTimer.isActive()) {
390 if (m_jobQueue.isEmpty())
391 m_jobTimer.stop();
392 } else if (didRemoveFirstJob && !m_jobQueue.isEmpty())
393 runNextJob();
394}

403void SWServerJobQueue::cancelJobsFromServiceWorker(ServiceWorkerIdentifier serviceWorkerIdentifier)
