95void SWServerToContextConnection::matchAll(uint64_t requestIdentifier, ServiceWorkerIdentifier serviceWorkerIdentifier, const ServiceWorkerClientQueryOptions& options)
96{
97 if (auto* worker = SWServerWorker::existingWorkerForIdentifier(serviceWorkerIdentifier)) {
98 worker->matchAll(options, [&] (auto&& data) {
99 worker->contextConnection()->matchAllCompleted(requestIdentifier, data);
100 });
101 }
102}

104void SWServerToContextConnection::claim(ServiceWorkerIdentifier serviceWorkerIdentifier, CompletionHandler<void(std::optional<ExceptionData>&&)>&& callback)
105{
106 auto* worker = SWServerWorker::existingWorkerForIdentifier(serviceWorkerIdentifier);
107 auto* server = worker ? worker->server() : nullptr;
108 callback(server ? server->claim(*worker) : std::nullopt);
109}

111void SWServerToContextConnection::skipWaiting(ServiceWorkerIdentifier serviceWorkerIdentifier, CompletionHandler<void()>&& completionHandler)
112{
113 if (auto* worker = SWServerWorker::existingWorkerForIdentifier(serviceWorkerIdentifier))
114 worker->skipWaiting();
115
116 completionHandler();
117}
