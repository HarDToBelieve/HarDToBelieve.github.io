56class SWServerWorker : public RefCounted<SWServerWorker> {
57public:
58 template <typename... Args> static Ref<SWServerWorker> create(Args&&... args)
59 {
60 return adoptRef(*new SWServerWorker(std::forward<Args>(args)...));
61 }
62
63 SWServerWorker(const SWServerWorker&) = delete;
64 WEBCORE_EXPORT ~SWServerWorker();
65
66 WEBCORE_EXPORT void terminate(CompletionHandler<void()>&& = [] { });
67 WEBCORE_EXPORT void whenTerminated(CompletionHandler<void()>&&);
68
69 WEBCORE_EXPORT void whenActivated(CompletionHandler<void(bool)>&&);
70
71 enum class State {
72 Running,
73 Terminating,
74 NotRunning,
75 };
76 bool isRunning() const { return m_state == State::Running; }
77 bool isTerminating() const { return m_state == State::Terminating; }
78 bool isNotRunning() const { return m_state == State::NotRunning; }
79 void setState(State);
80
81 SWServer* server() { return m_server.get(); }
82 const ServiceWorkerRegistrationKey& registrationKey() const { return m_registrationKey; }
83 const URL& scriptURL() const { return m_data.scriptURL; }
84 const ScriptBuffer& script() const { return m_script; }
85 const CertificateInfo& certificateInfo() const { return m_certificateInfo; }
86 WorkerType type() const { return m_data.type; }
87
88 ServiceWorkerIdentifier identifier() const { return m_data.identifier; }
89
90 ServiceWorkerState state() const { return m_data.state; }
91 bool isActive() const { return m_data.state == ServiceWorkerState::Activated || m_data.state == ServiceWorkerState::Activating; }
92
93 void setState(ServiceWorkerState);
94
95 bool hasPendingEvents() const { return m_hasPendingEvents; }
96 void setHasPendingEvents(bool);
97
98 void scriptContextFailedToStart(const std::optional<ServiceWorkerJobDataIdentifier>&, const String& message);
99 void scriptContextStarted(const std::optional<ServiceWorkerJobDataIdentifier>&, bool doesHandleFetch);
100 void didFinishInstall(const std::optional<ServiceWorkerJobDataIdentifier>&, bool wasSuccessful);
101 void didFinishActivation();
102 void contextTerminated();
103 WEBCORE_EXPORT std::optional<ServiceWorkerClientData> findClientByIdentifier(const ServiceWorkerClientIdentifier&) const;
104 void matchAll(const ServiceWorkerClientQueryOptions&, ServiceWorkerClientsMatchAllCallback&&);
105 void setScriptResource(URL&&, ServiceWorkerContextData::ImportedScript&&);
106 void didSaveScriptsToDisk(ScriptBuffer&& mainScript, HashMap<URL, ScriptBuffer>&& importedScripts);
107
108 void skipWaiting();
109 bool isSkipWaitingFlagSet() const { return m_isSkipWaitingFlagSet; }
110
111 WEBCORE_EXPORT static SWServerWorker* existingWorkerForIdentifier(ServiceWorkerIdentifier);
112 static HashMap<ServiceWorkerIdentifier, SWServerWorker*>& allWorkers();
113
114 const ServiceWorkerData& data() const { return m_data; }
115 ServiceWorkerContextData contextData() const;
116
117 const ClientOrigin& origin() const;
118 const RegistrableDomain& registrableDomain() const { return m_registrableDomain; }
119
120 WEBCORE_EXPORT SWServerToContextConnection* contextConnection();
121 String userAgent() const;
122
123 bool shouldSkipFetchEvent() const { return m_shouldSkipHandleFetch; }
124
125 SWServerRegistration* registration() const;
126
127 void setHasTimedOutAnyFetchTasks() { m_hasTimedOutAnyFetchTasks = true; }
128 bool hasTimedOutAnyFetchTasks() const { return m_hasTimedOutAnyFetchTasks; }
129 void didFailHeartBeatCheck();
130 void updateAppBoundValue(LastNavigationWasAppBound);
131
132private:
133 SWServerWorker(SWServer&, SWServerRegistration&, const URL&, const ScriptBuffer&, const CertificateInfo&, const ContentSecurityPolicyResponseHeaders&, String&& referrerPolicy, WorkerType, ServiceWorkerIdentifier, HashMap<URL, ServiceWorkerContextData::ImportedScript>&&);
134
135 void callWhenActivatedHandler(bool success);
136
137 void startTermination(CompletionHandler<void()>&&);
138 void terminationCompleted();
139 void terminationTimerFired();
140 void callTerminationCallbacks();
141
142 WeakPtr<SWServer> m_server;
143 ServiceWorkerRegistrationKey m_registrationKey;
144 WeakPtr<SWServerRegistration> m_registration;
145 ServiceWorkerData m_data;
146 ScriptBuffer m_script;
147 CertificateInfo m_certificateInfo;
148 ContentSecurityPolicyResponseHeaders m_contentSecurityPolicy;
149 String m_referrerPolicy;
150 bool m_hasPendingEvents { false };
151 State m_state { State::NotRunning };
152 mutable std::optional<ClientOrigin> m_origin;
153 RegistrableDomain m_registrableDomain;
154 bool m_isSkipWaitingFlagSet { false };
155 Vector<CompletionHandler<void(bool)>> m_whenActivatedHandlers;
156 HashMap<URL, ServiceWorkerContextData::ImportedScript> m_scriptResourceMap;
157 bool m_shouldSkipHandleFetch { false };
158 bool m_hasTimedOutAnyFetchTasks { false };
159 Vector<CompletionHandler<void()>> m_terminationCallbacks;
160 Timer m_terminationTimer;
161 LastNavigationWasAppBound m_lastNavigationWasAppBound;
162};
