73// https://drafts.css-houdini.org/css-paint-api/#registering-custom-paint
74ExceptionOr<void> PaintWorkletGlobalScope::registerPaint(JSC::JSGlobalObject& globalObject, const String& name, Strong<JSObject> paintConstructor)
75{
76 auto& vm = paintConstructor->vm();
77 JSC::JSLockHolder lock(vm);
78 auto scope = DECLARE_THROW_SCOPE(vm);
79
80 // Validate that paintConstructor is a VoidFunction
81 if (!paintConstructor->isCallable(vm))
82 return Exception { TypeError, "paintConstructor must be callable" };
83
84 if (name.isEmpty())
85 return Exception { TypeError, "The first argument must not be the empty string" };
86
87 {
88 Locker locker { paintDefinitionLock() };
89
90 if (paintDefinitionMap().contains(name))
91 return Exception { InvalidModificationError, "This name has already been registered" };
92
93 Vector<String> inputProperties;
94
95 JSValue inputPropertiesIterableValue = paintConstructor->get(&globalObject, Identifier::fromString(vm, "inputProperties"));
96 RETURN_IF_EXCEPTION(scope, Exception { ExistingExceptionError });
97
98 if (!inputPropertiesIterableValue.isUndefined())
99 inputProperties = convert<IDLSequence<IDLDOMString>>(globalObject, inputPropertiesIterableValue);
100 RETURN_IF_EXCEPTION(scope, Exception { ExistingExceptionError });
101
102 // FIXME: Validate input properties here (step 7).
103
104 Vector<String> inputArguments;
105
106 JSValue inputArgumentsIterableValue = paintConstructor->get(&globalObject, Identifier::fromString(vm, "inputArguments"));
107 RETURN_IF_EXCEPTION(scope, Exception { ExistingExceptionError });
108
109 if (!inputArgumentsIterableValue.isUndefined())
110 inputArguments = convert<IDLSequence<IDLDOMString>>(globalObject, inputArgumentsIterableValue);
111 RETURN_IF_EXCEPTION(scope, Exception { ExistingExceptionError });
112
113 // FIXME: Parse syntax for inputArguments here (steps 11 and 12).
114
115 JSValue contextOptionsValue = paintConstructor->get(&globalObject, Identifier::fromString(vm, "contextOptions"));
116 RETURN_IF_EXCEPTION(scope, Exception { ExistingExceptionError });
117 UNUSED_PARAM(contextOptionsValue);
118
119 // FIXME: Convert to PaintRenderingContext2DSettings here (step 14).
120
121 if (!paintConstructor->isConstructor(vm))
122 return Exception { TypeError, "The second argument must be a constructor" };
123
124 JSValue prototypeValue = paintConstructor->get(&globalObject, vm.propertyNames->prototype);
125 RETURN_IF_EXCEPTION(scope, Exception { ExistingExceptionError });
126
127 if (!prototypeValue.isObject())
128 return Exception { TypeError, "The second argument must have a prototype that is an object" };
129
130 JSValue paintValue = prototypeValue.get(&globalObject, Identifier::fromString(vm, "paint"));
131 RETURN_IF_EXCEPTION(scope, Exception { ExistingExceptionError });
132
133 if (paintValue.isUndefined())
134 return Exception { TypeError, "The class must have a paint method" };
135
136 RefPtr<JSCSSPaintCallback> paint = convert<IDLCallbackFunction<JSCSSPaintCallback>>(globalObject, paintValue, *jsCast<JSDOMGlobalObject*>(&globalObject));
137 RETURN_IF_EXCEPTION(scope, Exception { ExistingExceptionError });
138
139 auto paintDefinition = makeUnique<PaintDefinition>(name, paintConstructor.get(), paint.releaseNonNull(), WTFMove(inputProperties), WTFMove(inputArguments));
140 paintDefinitionMap().add(name, WTFMove(paintDefinition));
141 }
142
143 // This is for the case when we have already visited the paint definition map, and the GC is currently running in the background.
144 vm.heap.writeBarrier(&globalObject);
145
146 // FIXME: construct documentDefinition (step 22).
147
148 // FIXME: we should only repaint affected custom paint images <https://bugs.webkit.org/show_bug.cgi?id=192322>.
149 if (responsibleDocument() && responsibleDocument()->renderView())
150 responsibleDocument()->renderView()->repaintRootContents();
151
152 return { };
153}
