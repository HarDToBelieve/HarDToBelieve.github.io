50static std::atomic<unsigned> gNumberOfWorkletGlobalScopes { 0 };
51
52WorkletGlobalScope::WorkletGlobalScope(WorkerOrWorkletThread& thread, Ref<JSC::VM>&& vm, const WorkletParameters& parameters)
53 : WorkerOrWorkletGlobalScope(WorkerThreadType::Worklet, WTFMove(vm), &thread)
54 , m_topOrigin(SecurityOrigin::createUnique())
55 , m_url(parameters.windowURL)
56 , m_jsRuntimeFlags(parameters.jsRuntimeFlags)
57 , m_settingsValues(parameters.settingsValues)
58{
59 ++gNumberOfWorkletGlobalScopes;
60
61 setSecurityOriginPolicy(SecurityOriginPolicy::create(SecurityOrigin::create(this->url())));
62 setContentSecurityPolicy(makeUnique<ContentSecurityPolicy>(URL { this->url() }, *this));
63}
64
65WorkletGlobalScope::WorkletGlobalScope(Document& document, Ref<JSC::VM>&& vm, ScriptSourceCode&& code)
66 : WorkerOrWorkletGlobalScope(WorkerThreadType::Worklet, WTFMove(vm), nullptr)
67 , m_document(makeWeakPtr(document))
68 , m_topOrigin(SecurityOrigin::createUnique())
69 , m_url(code.url())
70 , m_jsRuntimeFlags(document.settings().javaScriptRuntimeFlags())
71 , m_code(WTFMove(code))
72 , m_settingsValues(document.settingsValues().isolatedCopy())
73{
74 ++gNumberOfWorkletGlobalScopes;
75
76 ASSERT(document.page());
77
78 setSecurityOriginPolicy(SecurityOriginPolicy::create(SecurityOrigin::create(this->url())));
79 setContentSecurityPolicy(makeUnique<ContentSecurityPolicy>(URL { this->url() }, *this));
80}
81
82WorkletGlobalScope::~WorkletGlobalScope()
83{
84 ASSERT(!script());
85 removeFromContextsMap();
86 ASSERT(gNumberOfWorkletGlobalScopes);
87 --gNumberOfWorkletGlobalScopes;
88}
89
90unsigned WorkletGlobalScope::numberOfWorkletGlobalScopes()
91{
92 return gNumberOfWorkletGlobalScopes;
93}
94
95void WorkletGlobalScope::prepareForDestruction()
96{
97 WorkerOrWorkletGlobalScope::prepareForDestruction();
98
99 if (script()) {
100 script()->vm().notifyNeedTermination();
101 clearScript();
102 }
103}

105String WorkletGlobalScope::userAgent(const URL& url) const
106{
107 if (!m_document)
108 return "";
109 return m_document->userAgent(url);
110}

112void WorkletGlobalScope::evaluate()
113{
114 if (m_code)
115 script()->evaluate(*m_code);
116}

118URL WorkletGlobalScope::completeURL(const String& url, ForceUTF8) const
119{
120 if (url.isNull())
121 return URL();

123}

125void WorkletGlobalScope::logExceptionToConsole(const String& errorMessage, const String& sourceURL, int lineNumber, int columnNumber, RefPtr<ScriptCallStack>&& stack)
126{
127 if (!m_document || isJSExecutionForbidden())
128 return;
129 m_document->logExceptionToConsole(errorMessage, sourceURL, lineNumber, columnNumber, WTFMove(stack));
130}

132void WorkletGlobalScope::addConsoleMessage(std::unique_ptr<Inspector::ConsoleMessage>&& message)
133{
134 if (!m_document || isJSExecutionForbidden() || !message)
135 return;
136 m_document->addConsoleMessage(makeUnique<Inspector::ConsoleMessage>(message->source(), message->type(), message->level(), message->message(), 0));
137}

158void WorkletGlobalScope::fetchAndInvokeScript(const URL& moduleURL, FetchRequestCredentials credentials, CompletionHandler<void(std::optional<Exception>&&)>&& completionHandler)
