28inline DOMParser::DOMParser(Document& contextDocument)
29 : m_contextDocument(makeWeakPtr(contextDocument))
30 , m_settings(contextDocument.settings())
31{
32}
33
34DOMParser::~DOMParser() = default;
35
36Ref<DOMParser> DOMParser::create(Document& contextDocument)
37{
38 return adoptRef(*new DOMParser(contextDocument));
39}
40
41ExceptionOr<Ref<Document>> DOMParser::parseFromString(const String& string, const String& contentType)
42{
43 if (contentType != "text/html" && contentType != "text/xml" && contentType != "application/xml" && contentType != "application/xhtml+xml" && contentType != "image/svg+xml")
44 return Exception { TypeError };
45 auto document = DOMImplementation::createDocument(contentType, nullptr, m_settings, URL { });
46 if (m_contextDocument)
47 document->setContextDocument(*m_contextDocument.get());
48 document->setContent(string);
49 if (m_contextDocument) {
50 document->setURL(m_contextDocument->url());
51 document->setSecurityOriginPolicy(m_contextDocument->securityOriginPolicy());
52 }
53 return document;
54}
