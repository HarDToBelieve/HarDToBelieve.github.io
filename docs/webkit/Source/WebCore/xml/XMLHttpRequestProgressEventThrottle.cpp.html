37const Seconds XMLHttpRequestProgressEventThrottle::minimumProgressEventDispatchingInterval { 50_ms }; // 50 ms per specification.
38
39XMLHttpRequestProgressEventThrottle::XMLHttpRequestProgressEventThrottle(XMLHttpRequest& target)
40 : m_target(target)
41 , m_dispatchThrottledProgressEventTimer(target.scriptExecutionContext(), *this, &XMLHttpRequestProgressEventThrottle::dispatchThrottledProgressEventTimerFired)
42{
43 m_dispatchThrottledProgressEventTimer.suspendIfNeeded();
44}
45
46XMLHttpRequestProgressEventThrottle::~XMLHttpRequestProgressEventThrottle() = default;
47
48void XMLHttpRequestProgressEventThrottle::dispatchThrottledProgressEvent(bool lengthComputable, unsigned long long loaded, unsigned long long total)
49{
50 m_lengthComputable = lengthComputable;
51 m_loaded = loaded;
52 m_total = total;
53
54 if (!m_target.hasEventListeners(eventNames().progressEvent))
55 return;
56
57 if (!m_shouldDeferEventsDueToSuspension && !m_dispatchThrottledProgressEventTimer.isActive()) {
58 // The timer is not active so the least frequent event for now is every byte. Just dispatch the event.
59
60 // We should not have any throttled progress event.
61 ASSERT(!m_hasPendingThrottledProgressEvent);
62
63 dispatchEventWhenPossible(XMLHttpRequestProgressEvent::create(eventNames().progressEvent, lengthComputable, loaded, total));
64 m_dispatchThrottledProgressEventTimer.startRepeating(minimumProgressEventDispatchingInterval);
65 m_hasPendingThrottledProgressEvent = false;
66 return;
67 }
68
69 // The timer is already active so minimumProgressEventDispatchingInterval is the least frequent event.
70 m_hasPendingThrottledProgressEvent = true;
71}

73void XMLHttpRequestProgressEventThrottle::dispatchReadyStateChangeEvent(Event& event, ProgressEventAction progressEventAction)
74{
75 if (progressEventAction == FlushProgressEvent)
76 flushProgressEvent();
77
78 dispatchEventWhenPossible(event);
79}
80
81void XMLHttpRequestProgressEventThrottle::dispatchEventWhenPossible(Event& event)
82{
83 if (m_shouldDeferEventsDueToSuspension)
84 m_target.queueTaskToDispatchEvent(m_target, TaskSource::Networking, makeRef(event));
85 else
86 m_target.dispatchEvent(event);
87}
88
89void XMLHttpRequestProgressEventThrottle::dispatchProgressEvent(const AtomString& type)
90{
91 ASSERT(type == eventNames().loadstartEvent || type == eventNames().progressEvent || type == eventNames().loadEvent || type == eventNames().loadendEvent || type == eventNames().abortEvent || type == eventNames().errorEvent || type == eventNames().timeoutEvent);
92
93 if (type == eventNames().loadstartEvent) {
94 m_lengthComputable = false;
95 m_loaded = 0;
96 m_total = 0;
97 }
98
99 if (m_target.hasEventListeners(type))
100 dispatchEventWhenPossible(XMLHttpRequestProgressEvent::create(type, m_lengthComputable, m_loaded, m_total));
101}
102
103void XMLHttpRequestProgressEventThrottle::flushProgressEvent()
104{
105 if (!m_hasPendingThrottledProgressEvent)
106 return;
107
108 m_hasPendingThrottledProgressEvent = false;
109 // We stop the timer as this is called when no more events are supposed to occur.
110 m_dispatchThrottledProgressEventTimer.cancel();
111
112 dispatchEventWhenPossible(XMLHttpRequestProgressEvent::create(eventNames().progressEvent, m_lengthComputable, m_loaded, m_total));
113}
114
115void XMLHttpRequestProgressEventThrottle::dispatchThrottledProgressEventTimerFired()
116{
117 ASSERT(m_dispatchThrottledProgressEventTimer.isActive());
118 if (!m_hasPendingThrottledProgressEvent) {
119 // No progress event was queued since the previous dispatch, we can safely stop the timer.
120 m_dispatchThrottledProgressEventTimer.cancel();
121 return;
122 }
123
124 dispatchEventWhenPossible(XMLHttpRequestProgressEvent::create(eventNames().progressEvent, m_lengthComputable, m_loaded, m_total));
125 m_hasPendingThrottledProgressEvent = false;
126}
127
128void XMLHttpRequestProgressEventThrottle::suspend()
129{
130 m_shouldDeferEventsDueToSuspension = true;
131
132 if (m_hasPendingThrottledProgressEvent) {
133 m_target.queueTaskKeepingObjectAlive(m_target, TaskSource::Networking, [this] {
134 flushProgressEvent();
135 });
136 }
137}
138
139void XMLHttpRequestProgressEventThrottle::resume()
140{
141 m_target.queueTaskKeepingObjectAlive(m_target, TaskSource::Networking, [this] {
142 m_shouldDeferEventsDueToSuspension = false;
143 });
144}
