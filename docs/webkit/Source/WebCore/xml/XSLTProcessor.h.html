25#if ENABLE(XSLT)

27#include "Node.h"

32#include <wtf/text/StringHash.h>

34namespace WebCore {
35
36class Frame;
37class Document;
38class DocumentFragment;
39
40class XSLTProcessor : public RefCounted<XSLTProcessor> {
41public:
42 static Ref<XSLTProcessor> create() { return adoptRef(*new XSLTProcessor); }
43 ~XSLTProcessor();
44
45 void setXSLStyleSheet(RefPtr<XSLStyleSheet>&& styleSheet) { m_stylesheet = WTFMove(styleSheet); }
46 bool transformToString(Node& source, String& resultMIMEType, String& resultString, String& resultEncoding);
47 Ref<Document> createDocumentFromSource(const String& source, const String& sourceEncoding, const String& sourceMIMEType, Node* sourceNode, Frame* frame);
48
49 // DOM methods
50 void importStylesheet(RefPtr<Node>&& style)
51 {
52 if (style)
53 m_stylesheetRootNode = WTFMove(style);
54 }
55 RefPtr<DocumentFragment> transformToFragment(Node* source, Document* ouputDoc);
56 RefPtr<Document> transformToDocument(Node* source);
57
58 void setParameter(const String& namespaceURI, const String& localName, const String& value);
59 String getParameter(const String& namespaceURI, const String& localName) const;
60 void removeParameter(const String& namespaceURI, const String& localName);
61 void clearParameters() { m_parameters.clear(); }
62
63 void reset();
64
65 static void parseErrorFunc(void* userData, xmlError*);
66 static void genericErrorFunc(void* userData, const char* msg, ...);
67
68 // Only for libXSLT callbacks
69 XSLStyleSheet* xslStylesheet() const { return m_stylesheet.get(); }
70
71 typedef HashMap<String, String> ParameterMap;
72
73private:
74 XSLTProcessor() = default;
75
76 RefPtr<XSLStyleSheet> m_stylesheet;
77 RefPtr<Node> m_stylesheetRootNode;
78 ParameterMap m_parameters;
79};
80
81} // namespace WebCore
82
83#endif // ENABLE(XSLT)
