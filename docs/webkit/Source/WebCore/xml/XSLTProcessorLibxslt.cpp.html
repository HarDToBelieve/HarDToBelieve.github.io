64void XSLTProcessor::parseErrorFunc(void* userData, xmlError* error)
65{
66 PageConsoleClient* console = static_cast<PageConsoleClient*>(userData);
67 if (!console)
68 return;
69
70 MessageLevel level;
71 switch (error->level) {
72 case XML_ERR_NONE:
73 level = MessageLevel::Debug;
74 break;
75 case XML_ERR_WARNING:
76 level = MessageLevel::Warning;
77 break;
78 case XML_ERR_ERROR:
79 case XML_ERR_FATAL:
80 default:
81 level = MessageLevel::Error;
82 break;
83 }
84
85 // xmlError->int2 is the column number of the error or 0 if N/A.
86 console->addMessage(MessageSource::XML, level, error->message, error->file, error->line, error->int2);
87}

89// FIXME: There seems to be no way to control the ctxt pointer for loading here, thus we have globals.
90static XSLTProcessor* globalProcessor = nullptr;
91static CachedResourceLoader* globalCachedResourceLoader = nullptr;
92static xmlDocPtr docLoaderFunc(const xmlChar* uri,
93 xmlDictPtr,
94 int options,
95 void* ctxt,
96 xsltLoadType type)
97{
98 if (!globalProcessor)
99 return 0;
100
101 switch (type) {
102 case XSLT_LOAD_DOCUMENT: {
103 xsltTransformContextPtr context = (xsltTransformContextPtr)ctxt;
104 xmlChar* base = xmlNodeGetBase(context->document->doc, context->node);
105 URL url(URL({ }, reinterpret_cast<const char*>(base)), reinterpret_cast<const char*>(uri));
106 xmlFree(base);
107 ResourceError error;
108 ResourceResponse response;
109
110 RefPtr<SharedBuffer> data;
111
112 bool requestAllowed = globalCachedResourceLoader->frame() && globalCachedResourceLoader->document()->securityOrigin().canRequest(url);
113 if (requestAllowed) {
114 FetchOptions options;
115 options.mode = FetchOptions::Mode::SameOrigin;
116 options.credentials = FetchOptions::Credentials::Include;
117 globalCachedResourceLoader->frame()->loader().loadResourceSynchronously(url, ClientCredentialPolicy::MayAskClientForCredentials, options, { }, error, response, data);
118 if (error.isNull())
119 requestAllowed = globalCachedResourceLoader->document()->securityOrigin().canRequest(response.url());
120 else if (data)
121 data = nullptr;
122 }
123 if (!requestAllowed) {
124 if (data)
125 data = nullptr;
126 globalCachedResourceLoader->printAccessDeniedMessage(url);
127 }
128
129 PageConsoleClient* console = nullptr;
130 Frame* frame = globalProcessor->xslStylesheet()->ownerDocument()->frame();
131 if (frame && frame->page())
132 console = &frame->page()->console();
133 xmlSetStructuredErrorFunc(console, XSLTProcessor::parseErrorFunc);
134 xmlSetGenericErrorFunc(console, XSLTProcessor::genericErrorFunc);
135
136 // We don't specify an encoding here. Neither Gecko nor WinIE respects
137 // the encoding specified in the HTTP headers.
138 xmlDocPtr doc = xmlReadMemory(data ? data->dataAsCharPtr() : nullptr, data ? data->size() : 0, (const char*)uri, 0, options);
139
140 xmlSetStructuredErrorFunc(0, 0);
141 xmlSetGenericErrorFunc(0, 0);
142
143 return doc;
144 }
145 case XSLT_LOAD_STYLESHEET:
146 return globalProcessor->xslStylesheet()->locateStylesheetSubResource(((xsltStylesheetPtr)ctxt)->doc, uri);
147 default:
148 break;
149 }
150
151 return 0;
152}

154static inline void setXSLTLoadCallBack(xsltDocLoaderFunc func, XSLTProcessor* processor, CachedResourceLoader* cachedResourceLoader)
155{
156 xsltSetLoaderFunc(func);
157 globalProcessor = processor;
158 globalCachedResourceLoader = cachedResourceLoader;
159}

161static int writeToStringBuilder(void* context, const char* buffer, int length)
162{
163 StringBuilder& resultOutput = *static_cast<StringBuilder*>(context);
164
165 // FIXME: Consider ways to make this more efficient by moving it into a
166 // StringBuilder::appendUTF8 function, and then optimizing to not need a
167 // Vector<UChar> and possibly optimize cases that can produce 8-bit Latin-1
168 // strings, but that would need to be sophisticated about not processing
169 // trailing incomplete sequences and communicating that to the caller.
170
171 Vector<UChar> outputBuffer(length);
172
173 UBool error = false;
174 int inputOffset = 0;
175 int outputOffset = 0;
176 while (inputOffset < length) {
177 UChar32 character;
178 int nextInputOffset = inputOffset;
179 U8_NEXT(reinterpret_cast<const uint8_t*>(buffer), nextInputOffset, length, character);
180 if (character < 0) {
181 if (nextInputOffset == length)
182 break;
183 ASSERT_NOT_REACHED();
184 return -1;
185 }
186 inputOffset = nextInputOffset;
187 U16_APPEND(outputBuffer.data(), outputOffset, length, character, error);
188 if (error) {
189 ASSERT_NOT_REACHED();
190 return -1;
191 }
192 }
193
194 resultOutput.appendCharacters(outputBuffer.data(), outputOffset);
195 return inputOffset;
196}

198static bool saveResultToString(xmlDocPtr resultDoc, xsltStylesheetPtr sheet, String& resultString)
199{
200 xmlOutputBufferPtr outputBuf = xmlAllocOutputBuffer(nullptr);
201 if (!outputBuf)
202 return false;
203
204 StringBuilder resultBuilder;
205 outputBuf->context = &resultBuilder;
206 outputBuf->writecallback = writeToStringBuilder;
207
208 int retval = xsltSaveResultTo(outputBuf, resultDoc, sheet);
209 xmlOutputBufferClose(outputBuf);
210 if (retval < 0)
211 return false;
212
213 // Workaround for <http://bugzilla.gnome.org/show_bug.cgi?id=495668>: libxslt appends an extra line feed to the result.
214 if (resultBuilder.length() > 0 && resultBuilder[resultBuilder.length() - 1] == '\n')
215 resultBuilder.shrink(resultBuilder.length() - 1);
216
217 resultString = resultBuilder.toString();
218
219 return true;
220}

222static const char** xsltParamArrayFromParameterMap(XSLTProcessor::ParameterMap& parameters)
223{
224 if (parameters.isEmpty())
225 return 0;
226
227 auto size = (((Checked<size_t>(parameters.size()) * 2U) + 1U) * sizeof(char*)).value();
228 auto** parameterArray = static_cast<const char**>(fastMalloc(size));
229
230 size_t index = 0;
231 for (auto& parameter : parameters) {
232 parameterArray[index++] = fastStrDup(parameter.key.utf8().data());
233 parameterArray[index++] = fastStrDup(parameter.value.utf8().data());
234 }
235 parameterArray[index] = nullptr;
236
237#if !PLATFORM(WIN) && !HAVE(LIBXSLT_FIX_FOR_RADAR_71864140)
238 RELEASE_ASSERT(index <= std::numeric_limits<int>::max());
239#endif
240
241 return parameterArray;
242}

244static void freeXsltParamArray(const char** params)
245{
246 const char** temp = params;
247 if (!params)
248 return;
249
250 while (*temp) {
251 fastFree((void*)*(temp++));
252 fastFree((void*)*(temp++));
253 }
254 fastFree(params);
255}

257static xsltStylesheetPtr xsltStylesheetPointer(RefPtr<XSLStyleSheet>& cachedStylesheet, Node* stylesheetRootNode)
258{
259 if (!cachedStylesheet && stylesheetRootNode) {
260 cachedStylesheet = XSLStyleSheet::createForXSLTProcessor(stylesheetRootNode->parentNode() ? stylesheetRootNode->parentNode() : stylesheetRootNode,
261 stylesheetRootNode->document().url().string(),
262 stylesheetRootNode->document().url()); // FIXME: Should we use baseURL here?
263
264 // According to Mozilla documentation, the node must be a Document node, an xsl:stylesheet or xsl:transform element.
265 // But we just use text content regardless of node type.
266 cachedStylesheet->parseString(serializeFragment(*stylesheetRootNode, SerializedNodes::SubtreeIncludingNode));
267 }
268
269 if (!cachedStylesheet || !cachedStylesheet->document())
270 return 0;
271
272 return cachedStylesheet->compileStyleSheet();
273}

275static inline xmlDocPtr xmlDocPtrFromNode(Node& sourceNode, bool& shouldDelete)
276{
277 Ref<Document> ownerDocument(sourceNode.document());
278 bool sourceIsDocument = (&sourceNode == &ownerDocument.get());
279
280 xmlDocPtr sourceDoc = nullptr;
281 if (sourceIsDocument && ownerDocument->transformSource())
282 sourceDoc = ownerDocument->transformSource()->platformSource();
283 if (!sourceDoc) {
284 sourceDoc = xmlDocPtrForString(ownerDocument->cachedResourceLoader(), serializeFragment(sourceNode, SerializedNodes::SubtreeIncludingNode),
285 sourceIsDocument ? ownerDocument->url().string() : String());
286 shouldDelete = sourceDoc;
287 }
288 return sourceDoc;
289}

291static inline String resultMIMEType(xmlDocPtr resultDoc, xsltStylesheetPtr sheet)
292{
293 // There are three types of output we need to be able to deal with:
294 // HTML (create an HTML document), XML (create an XML document),
295 // and text (wrap in a <pre> and create an XML document).
296
297 const xmlChar* resultType = nullptr;
298 XSLT_GET_IMPORT_PTR(resultType, sheet, method);
299 if (!resultType && resultDoc->type == XML_HTML_DOCUMENT_NODE)
300 resultType = (const xmlChar*)"html";
301
302 if (xmlStrEqual(resultType, (const xmlChar*)"html"))
303 return "text/html";
304 if (xmlStrEqual(resultType, (const xmlChar*)"text"))
305 return "text/plain";
306
307 return "application/xml";
308}

310bool XSLTProcessor::transformToString(Node& sourceNode, String& mimeType, String& resultString, String& resultEncoding)
311{
312 Ref<Document> ownerDocument(sourceNode.document());
313
314 setXSLTLoadCallBack(docLoaderFunc, this, &ownerDocument->cachedResourceLoader());
315 xsltStylesheetPtr sheet = xsltStylesheetPointer(m_stylesheet, m_stylesheetRootNode.get());
316 if (!sheet) {
317 setXSLTLoadCallBack(nullptr, nullptr, nullptr);
318 m_stylesheet = nullptr;
319 return false;
320 }
321 m_stylesheet->clearDocuments();
322
323#if OS(DARWIN) && !PLATFORM(GTK)
324 int origXsltMaxDepth = *xsltMaxDepth;
325 *xsltMaxDepth = 1000;
326#else
327 int origXsltMaxDepth = xsltMaxDepth;
328 xsltMaxDepth = 1000;
329#endif
330
331 xmlChar* origMethod = sheet->method;
332 if (!origMethod && mimeType == "text/html")
333 sheet->method = reinterpret_cast<xmlChar*>(const_cast<char*>("html"));
334
335 bool success = false;
336 bool shouldFreeSourceDoc = false;
337 if (xmlDocPtr sourceDoc = xmlDocPtrFromNode(sourceNode, shouldFreeSourceDoc)) {
338 // The XML declaration would prevent parsing the result as a fragment, and it's not needed even for documents,
339 // as the result of this function is always immediately parsed.
340 sheet->omitXmlDeclaration = true;
341
342 xsltTransformContextPtr transformContext = xsltNewTransformContext(sheet, sourceDoc);
343 registerXSLTExtensions(transformContext);
344
345 xsltSecurityPrefsPtr securityPrefs = xsltNewSecurityPrefs();
346 // Read permissions are checked by docLoaderFunc.
347 if (0 != xsltSetSecurityPrefs(securityPrefs, XSLT_SECPREF_WRITE_FILE, xsltSecurityForbid))
348 CRASH();
349 if (0 != xsltSetSecurityPrefs(securityPrefs, XSLT_SECPREF_CREATE_DIRECTORY, xsltSecurityForbid))
350 CRASH();
351 if (0 != xsltSetSecurityPrefs(securityPrefs, XSLT_SECPREF_WRITE_NETWORK, xsltSecurityForbid))
352 CRASH();
353 if (0 != xsltSetCtxtSecurityPrefs(securityPrefs, transformContext))
354 CRASH();
355
356 // <http://bugs.webkit.org/show_bug.cgi?id=16077>: XSLT processor <xsl:sort> algorithm only compares by code point.
357 xsltSetCtxtSortFunc(transformContext, xsltUnicodeSortFunction);
358
359 // This is a workaround for a bug in libxslt.
360 // The bug has been fixed in version 1.1.13, so once we ship that this can be removed.
361 if (!transformContext->globalVars)
362 transformContext->globalVars = xmlHashCreate(20);
363
364 const char** params = xsltParamArrayFromParameterMap(m_parameters);
365 xsltQuoteUserParams(transformContext, params);
366 xmlDocPtr resultDoc = xsltApplyStylesheetUser(sheet, sourceDoc, 0, 0, 0, transformContext);
367
368 xsltFreeTransformContext(transformContext);
369 xsltFreeSecurityPrefs(securityPrefs);
370 freeXsltParamArray(params);
371
372 if (shouldFreeSourceDoc)
373 xmlFreeDoc(sourceDoc);
374
375 if ((success = saveResultToString(resultDoc, sheet, resultString))) {
376 mimeType = resultMIMEType(resultDoc, sheet);
377 resultEncoding = reinterpret_cast<const char*>(resultDoc->encoding);
378 }
379 xmlFreeDoc(resultDoc);
380 }
381
382 sheet->method = origMethod;
383#if OS(DARWIN) && !PLATFORM(GTK)
384 *xsltMaxDepth = origXsltMaxDepth;
385#else
386 xsltMaxDepth = origXsltMaxDepth;
387#endif
388 setXSLTLoadCallBack(0, 0, 0);
389 xsltFreeStylesheet(sheet);
390 m_stylesheet = nullptr;
391
392 return success;
393}
