145#if PLATFORM(COCOA) && ENABLE(MEDIA_STREAM)
146class GPUProxyForCapture final : public UserMediaCaptureManagerProxy::ConnectionProxy {
147 WTF_MAKE_FAST_ALLOCATED;
148public:
149 explicit GPUProxyForCapture(GPUConnectionToWebProcess& process)
150 : m_process(process)
151 {
152 }
153
154private:
155 Logger& logger() final { return m_process.logger(); }
156 void addMessageReceiver(IPC::ReceiverName, IPC::MessageReceiver&) final { }
157 void removeMessageReceiver(IPC::ReceiverName messageReceiverName) final { }
158 IPC::Connection& connection() final { return m_process.connection(); }
159 bool willStartCapture(CaptureDevice::DeviceType type) const final
160 {
161 switch (type) {
162 case CaptureDevice::DeviceType::Unknown:
163 case CaptureDevice::DeviceType::Speaker:
164 return false;
165 case CaptureDevice::DeviceType::Microphone:
166 return m_process.allowsAudioCapture();
167 case CaptureDevice::DeviceType::Camera:
168 if (!m_process.allowsVideoCapture())
169 return false;
170#if PLATFORM(IOS)
171 MediaSessionManageriOS::providePresentingApplicationPID();
172#endif
173 return true;
174 break;
175 case CaptureDevice::DeviceType::Screen:
176 return m_process.allowsDisplayCapture();
177 case CaptureDevice::DeviceType::Window:
178 return m_process.allowsDisplayCapture();
179 }
180 }
181
182 GPUConnectionToWebProcess& m_process;
183};
184#endif

191GPUConnectionToWebProcess::GPUConnectionToWebProcess(GPUProcess& gpuProcess, WebCore::ProcessIdentifier webProcessIdentifier, IPC::Connection::Identifier connectionIdentifier, PAL::SessionID sessionID, GPUProcessConnectionParameters&& parameters)
192 : m_connection(IPC::Connection::createServerConnection(connectionIdentifier, *this))
193 , m_gpuProcess(gpuProcess)
194 , m_webProcessIdentifier(webProcessIdentifier)
195#if HAVE(TASK_IDENTITY_TOKEN)
196 , m_webProcessIdentityToken(WTFMove(parameters.webProcessIdentityToken))
197#endif
198 , m_remoteMediaPlayerManagerProxy(makeUniqueRef<RemoteMediaPlayerManagerProxy>(*this))
199 , m_sessionID(sessionID)
200#if PLATFORM(COCOA) && USE(LIBWEBRTC)
201 , m_libWebRTCCodecsProxy(LibWebRTCCodecsProxy::create(*this))
202#endif
203#if PLATFORM(COCOA) && ENABLE(MEDIA_STREAM)
204 , m_sampleBufferDisplayLayerManager(RemoteSampleBufferDisplayLayerManager::create(*this))
205#endif
206#if ENABLE(ROUTING_ARBITRATION) && HAVE(AVAUDIO_ROUTING_ARBITER)
207 , m_routingArbitrator(LocalAudioSessionRoutingArbitrator::create(*this))
208#endif
209{
210 RELEASE_ASSERT(RunLoop::isMain());
211
212#if ENABLE(ROUTING_ARBITRATION) && HAVE(AVAUDIO_ROUTING_ARBITER)
213 gpuProcess.audioSessionManager().session().setRoutingArbitrationClient(makeWeakPtr(m_routingArbitrator.get()));
214#endif
215
216 if (!parameters.overrideLanguages.isEmpty())
217 overrideUserPreferredLanguages(parameters.overrideLanguages);
218
219 // Use this flag to force synchronous messages to be treated as asynchronous messages in the WebProcess.
220 // Otherwise, the WebProcess would process incoming synchronous IPC while waiting for a synchronous IPC
221 // reply from the GPU process, which would be unsafe.
222 m_connection->setOnlySendMessagesAsDispatchWhenWaitingForSyncReplyWhenProcessingSuchAMessage(true);
223 m_connection->open();
224}
225
226GPUConnectionToWebProcess::~GPUConnectionToWebProcess()
227{
228 RELEASE_ASSERT(RunLoop::isMain());
229
230 m_connection->invalidate();
231
232#if PLATFORM(COCOA) && ENABLE(MEDIA_STREAM)
233 m_sampleBufferDisplayLayerManager->close();
234#endif
235#if PLATFORM(COCOA) && USE(LIBWEBRTC)
236 m_libWebRTCCodecsProxy->close();
237#endif
238}
239
240void GPUConnectionToWebProcess::didClose(IPC::Connection& connection)
241{
242#if ENABLE(ROUTING_ARBITRATION) && HAVE(AVAUDIO_ROUTING_ARBITER)
243 m_routingArbitrator->processDidTerminate();
244#endif
245
246#if USE(AUDIO_SESSION)
247 if (m_audioSessionProxy) {
248 gpuProcess().audioSessionManager().removeProxy(*m_audioSessionProxy);
249 m_audioSessionProxy = nullptr;
250 }
251#endif
252
253 // RemoteRenderingBackend objects ref their GPUConnectionToWebProcess so we need to make sure
254 // to break the reference cycle by destroying them.
255 m_remoteRenderingBackendMap.clear();
256
257#if ENABLE(WEBGL)
258 // RemoteGraphicsContextsGL objects are unneeded after connection closes.
259 m_remoteGraphicsContextGLMap.clear();
260#endif
261
262 gpuProcess().connectionToWebProcessClosed(connection);
263 gpuProcess().removeGPUConnectionToWebProcess(*this); // May destroy |this|.
264}

267void GPUConnectionToWebProcess::createVisibilityPropagationContextForPage(WebPageProxyIdentifier pageProxyID, WebCore::PageIdentifier pageID, bool canShowWhileLocked)
268{
269 auto contextForVisibilityPropagation = LayerHostingContext::createForExternalHostingProcess({ canShowWhileLocked });
270 RELEASE_LOG(Process, "GPUConnectionToWebProcess::createVisibilityPropagationContextForPage: pageProxyID=%" PRIu64 ", webPageID=%" PRIu64 ", contextID=%u", pageProxyID.toUInt64(), pageID.toUInt64(), contextForVisibilityPropagation->contextID());

306bool GPUConnectionToWebProcess::allowsExitUnderMemoryPressure() const
307{
308 for (auto& remoteRenderingBackend : m_remoteRenderingBackendMap.values()) {
309 if (!remoteRenderingBackend->allowsExitUnderMemoryPressure())
310 return false;
311 }
312#if ENABLE(WEBGL)
313 if (!m_remoteGraphicsContextGLMap.isEmpty())
314 return false;
315#endif
316 if (!m_remoteMediaPlayerManagerProxy->allowsExitUnderMemoryPressure())
317 return false;
318#if ENABLE(WEB_AUDIO)
319 if (m_remoteAudioDestinationManager && !m_remoteAudioDestinationManager->allowsExitUnderMemoryPressure())
320 return false;
321#endif
322#if PLATFORM(COCOA) && ENABLE(MEDIA_STREAM)
323 if (m_userMediaCaptureManagerProxy && m_userMediaCaptureManagerProxy->hasSourceProxies())
324 return false;
325 if (m_audioMediaStreamTrackRendererInternalUnitManager && m_audioMediaStreamTrackRendererInternalUnitManager->hasUnits())
326 return false;
327 if (!m_sampleBufferDisplayLayerManager->allowsExitUnderMemoryPressure())
328 return false;
329#endif
330#if PLATFORM(COCOA) && ENABLE(MEDIA_STREAM) && HAVE(AVASSETWRITERDELEGATE)
331 if (m_remoteMediaRecorderManager && !m_remoteMediaRecorderManager->allowsExitUnderMemoryPressure())
332 return false;
333#endif
334#if HAVE(AVASSETREADER)
335 if (m_imageDecoderAVFProxy && !m_imageDecoderAVFProxy->allowsExitUnderMemoryPressure())
336 return false;
337#endif
338#if ENABLE(ENCRYPTED_MEDIA)
339 if (m_cdmFactoryProxy && !m_cdmFactoryProxy->allowsExitUnderMemoryPressure())
340 return false;
341#endif
342#if ENABLE(LEGACY_ENCRYPTED_MEDIA)
343 if (m_legacyCdmFactoryProxy && !m_legacyCdmFactoryProxy->allowsExitUnderMemoryPressure())
344 return false;
345#endif
346#if PLATFORM(COCOA) && USE(LIBWEBRTC)
347 if (!m_libWebRTCCodecsProxy->allowsExitUnderMemoryPressure())
348 return false;
349#endif
350 return true;
351}

364{

608bool GPUConnectionToWebProcess::dispatchMessage(IPC::Connection& connection, IPC::Decoder& decoder)
609{
610#if ENABLE(WEB_AUDIO)
611 if (decoder.messageReceiverName() == Messages::RemoteAudioDestinationManager::messageReceiverName()) {
612 remoteAudioDestinationManager().didReceiveMessageFromWebProcess(connection, decoder);
613 return true;
614 }
615#endif
616 if (decoder.messageReceiverName() == Messages::RemoteMediaPlayerManagerProxy::messageReceiverName()) {
617 remoteMediaPlayerManagerProxy().didReceiveMessageFromWebProcess(connection, decoder);
618 return true;
619 }
620 if (decoder.messageReceiverName() == Messages::RemoteMediaPlayerProxy::messageReceiverName()) {
621 remoteMediaPlayerManagerProxy().didReceivePlayerMessage(connection, decoder);
622 return true;
623 }
624 if (decoder.messageReceiverName() == Messages::RemoteMediaResourceManager::messageReceiverName()) {
625 remoteMediaResourceManager().didReceiveMessage(connection, decoder);
626 return true;
627 }
628#if PLATFORM(COCOA) && ENABLE(MEDIA_STREAM)
629 if (decoder.messageReceiverName() == Messages::UserMediaCaptureManagerProxy::messageReceiverName()) {
630 userMediaCaptureManagerProxy().didReceiveMessageFromGPUProcess(connection, decoder);
631 return true;
632 }
633 if (decoder.messageReceiverName() == Messages::RemoteAudioMediaStreamTrackRendererInternalUnitManager::messageReceiverName()) {
634 audioMediaStreamTrackRendererInternalUnitManager().didReceiveMessage(connection, decoder);
635 return true;
636 }
637#endif
638#if PLATFORM(COCOA) && ENABLE(MEDIA_STREAM) && HAVE(AVASSETWRITERDELEGATE)
639 if (decoder.messageReceiverName() == Messages::RemoteMediaRecorderManager::messageReceiverName()) {
640 mediaRecorderManager().didReceiveMessageFromWebProcess(connection, decoder);
641 return true;
642 }
