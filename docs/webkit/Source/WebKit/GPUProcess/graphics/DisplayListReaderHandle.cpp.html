34std::optional<size_t> DisplayListReaderHandle::advance(size_t amount)
35{
36 auto previousUnreadBytes = header().unreadBytes.exchangeSub(amount);
37 if (UNLIKELY(previousUnreadBytes < amount))
38 return std::nullopt;
39 return previousUnreadBytes - amount;
40}
41
42std::unique_ptr<DisplayList::DisplayList> DisplayListReaderHandle::displayListForReading(size_t offset, size_t capacity, DisplayList::ItemBufferReadingClient& client) const
43{
44 auto checkedExtent = checkedSum<size_t>(offset, capacity);
45 if (UNLIKELY(checkedExtent.hasOverflowed() || checkedExtent > sharedMemory().size()))
46 return nullptr;
47
48 auto displayList = makeUnique<DisplayList::DisplayList>(DisplayList::ItemBufferHandles {{ identifier(), data() + offset, capacity }});
49 displayList->setItemBufferReadingClient(&client);
50 return displayList;
51}
