83void RemoteGraphicsContextGL::initialize(GraphicsContextGLAttributes&& attributes)
84{
85 assertIsMainRunLoop();
86 remoteGraphicsContextGLStreamWorkQueue().dispatch([attributes = WTFMove(attributes), protectedThis = makeRef(*this)]() mutable {
87 protectedThis->workQueueInitialize(WTFMove(attributes));
88 });
89 m_streamConnection->startReceivingMessages(*this, Messages::RemoteGraphicsContextGL::messageReceiverName(), m_graphicsContextGLIdentifier.toUInt64());
90}
91
92void RemoteGraphicsContextGL::stopListeningForIPC(Ref<RemoteGraphicsContextGL>&& refFromConnection)
93{
94 assertIsMainRunLoop();
95 m_streamConnection->stopReceivingMessages(Messages::RemoteGraphicsContextGL::messageReceiverName(), m_graphicsContextGLIdentifier.toUInt64());
96 remoteGraphicsContextGLStreamWorkQueue().dispatch([protectedThis = WTFMove(refFromConnection)]() {
97 protectedThis->workQueueUninitialize();
98 protectedThis->m_renderingResourcesRequest = { };
99 });
100}

113void RemoteGraphicsContextGL::workQueueInitialize(WebCore::GraphicsContextGLAttributes&& attributes)
114{
115 m_streamThread.reset();
116 assertIsCurrent(m_streamThread);
117 platformWorkQueueInitialize(WTFMove(attributes));
118 if (m_context) {
119 m_context->addClient(*this);
120 String extensions = m_context->getString(GraphicsContextGL::EXTENSIONS);
121 String requestableExtensions = m_context->getString(ExtensionsGL::REQUESTABLE_EXTENSIONS_ANGLE);
122 send(Messages::RemoteGraphicsContextGLProxy::WasCreated(true, remoteGraphicsContextGLStreamWorkQueue().wakeUpSemaphore(), extensions, requestableExtensions));
123 } else
124 send(Messages::RemoteGraphicsContextGLProxy::WasCreated(false, IPC::Semaphore { }, "", ""));
125}

210void RemoteGraphicsContextGL::paintPixelBufferToImageBuffer(std::optional<WebCore::PixelBuffer>&& pixelBuffer, WebCore::RenderingResourceIdentifier target, CompletionHandler<void()>&& completionHandler)
211{
212 assertIsCurrent(m_streamThread);
213 // FIXME: We do not have functioning read/write fences in RemoteRenderingBackend. Thus this is synchronous,
214 // as are the messages that call these.
215 Lock lock;
216 Condition conditionVariable;
217 bool isFinished = false;
218 m_renderingBackend->dispatch([&, contextAttributes = m_context->contextAttributes()]() mutable {
219 if (auto imageBuffer = m_renderingBackend->remoteResourceCache().cachedImageBuffer(target)) {
220 // Here we do not try to play back pending commands for imageBuffer. Currently this call is only made for empty
221 // image buffers and there's no good way to add display lists.
222 if (pixelBuffer)
223 GraphicsContextGLOpenGL::paintToCanvas(contextAttributes, WTFMove(*pixelBuffer), imageBuffer->backendSize(), imageBuffer->context());
224 else
225 imageBuffer->context().clearRect({ IntPoint(), imageBuffer->backendSize() });
226 // Unfortunately "flush" implementation in RemoteRenderingBackend overloads ordering and effects.
227 imageBuffer->flushContext();
228 }
229 Locker locker { lock };
230 isFinished = true;
231 conditionVariable.notifyOne();
232 });
233 Locker locker { lock };
234 conditionVariable.wait(lock, [&] {
235 return isFinished;
236 });
237 completionHandler();
238}

240void RemoteGraphicsContextGL::copyTextureFromMedia(WebCore::MediaPlayerIdentifier mediaPlayerIdentifier, uint32_t texture, uint32_t target, int32_t level, uint32_t internalFormat, uint32_t format, uint32_t type, bool premultiplyAlpha, bool flipY, CompletionHandler<void(bool)>&& completionHandler)
241{
242 assertIsCurrent(m_streamThread);
243#if USE(AVFOUNDATION)
244 UNUSED_VARIABLE(premultiplyAlpha);
245 ASSERT_UNUSED(target, target == GraphicsContextGL::TEXTURE_2D);
246
247 RetainPtr<CVPixelBufferRef> pixelBuffer;
248 auto getPixelBuffer = [&] {
249 if (!m_gpuConnectionToWebProcess)
250 return;
251
252 if (auto mediaPlayer = m_gpuConnectionToWebProcess->remoteMediaPlayerManagerProxy().mediaPlayer(mediaPlayerIdentifier))
253 pixelBuffer = mediaPlayer->pixelBufferForCurrentTime();
254 };
255
256 callOnMainRunLoopAndWait(WTFMove(getPixelBuffer));
257
258 if (!pixelBuffer) {
259 completionHandler(false);
260 return;
261 }
262
263 auto contextCV = m_context->asCV();
264 if (!contextCV) {
265 completionHandler(false);
266 return;
267 }
268
269 completionHandler(contextCV->copyPixelBufferToTexture(pixelBuffer.get(), texture, level, internalFormat, format, type, GraphicsContextGL::FlipY(flipY)));
270#else
271 UNUSED_VARIABLE(mediaPlayerIdentifier);
272 UNUSED_VARIABLE(texture);
273 UNUSED_VARIABLE(target);
274 UNUSED_VARIABLE(level);
275 UNUSED_VARIABLE(internalFormat);
276 UNUSED_VARIABLE(format);
277 UNUSED_VARIABLE(type);
278 UNUSED_VARIABLE(premultiplyAlpha);
279 UNUSED_VARIABLE(flipY);
280
281 notImplemented();
282 completionHandler(false);
283#endif
284}

286void RemoteGraphicsContextGL::simulateEventForTesting(WebCore::GraphicsContextGLOpenGL::SimulatedEventForTesting event)
287{
288 assertIsCurrent(m_streamThread);
289 // FIXME: only run this in testing mode. https://bugs.webkit.org/show_bug.cgi?id=222544
290 if (event == WebCore::GraphicsContextGLOpenGL::SimulatedEventForTesting::Timeout) {
291 // Simulate the timeout by just discarding the context. The subsequent messages act like
292 // unauthorized or old messages from Web process, they are skipped.
293 callOnMainRunLoop([gpuConnectionToWebProcess = m_gpuConnectionToWebProcess, identifier = m_graphicsContextGLIdentifier]() {
294 if (auto connectionToWeb = gpuConnectionToWebProcess.get())
295 connectionToWeb->releaseGraphicsContextGLForTesting(identifier);
296 });
297 return;
298 }
299 if (event == WebCore::GraphicsContextGLOpenGL::SimulatedEventForTesting::ContextChange) {
300#if PLATFORM(MAC)
301 callOnMainRunLoop([weakConnection = m_gpuConnectionToWebProcess]() {
302 if (auto connection = weakConnection.get())
303 connection->dispatchDisplayWasReconfiguredForTesting();
304 });
305#endif
306 return;
307 }
308 m_context->simulateEventForTesting(event);
309}
