58// GPU process side implementation of that receives messages about GraphicsContextGL calls
59// and issues real GraphicsContextGL calls based on the received messages.
60// The implementation is largely generated by running Tools/Scripts/generate-gpup-webgl.
61class RemoteGraphicsContextGL : public ThreadSafeRefCounted<RemoteGraphicsContextGL>, private WebCore::GraphicsContextGL::Client {
62 WTF_MAKE_FAST_ALLOCATED;
63public:
64 static Ref<RemoteGraphicsContextGL> create(GPUConnectionToWebProcess&, WebCore::GraphicsContextGLAttributes&&, GraphicsContextGLIdentifier, RemoteRenderingBackend&, IPC::StreamConnectionBuffer&&);
65 ~RemoteGraphicsContextGL() override;
66 void stopListeningForIPC(Ref<RemoteGraphicsContextGL>&& refFromConnection);
67
68 // IPC::StreamServerConnection<RemoteGraphicsContextGL> template contract implementation.
69 void didReceiveStreamMessage(IPC::StreamServerConnectionBase&, IPC::Decoder&);
70#if PLATFORM(MAC)
71 void displayWasReconfigured();
72#endif
73
74protected:
75 RemoteGraphicsContextGL(GPUConnectionToWebProcess&, GraphicsContextGLIdentifier, RemoteRenderingBackend&, IPC::StreamConnectionBuffer&&);
76 void initialize(WebCore::GraphicsContextGLAttributes&&);
77
78 void workQueueInitialize(WebCore::GraphicsContextGLAttributes&&);
79 virtual void platformWorkQueueInitialize(WebCore::GraphicsContextGLAttributes&&) { };
80 void workQueueUninitialize();
81 template<typename T>
82 bool send(T&& message) const { return m_streamConnection->connection().send(WTFMove(message), m_graphicsContextGLIdentifier); }
83
84 // GraphicsContextGL::Client overrides.
85 void didComposite() final;
86 void forceContextLost() final;
87 void recycleContext() final;
88 void dispatchContextChangedNotification() final;
89
90 // Messages to be received.
91 void reshape(int32_t width, int32_t height);
92 void ensureExtensionEnabled(String&&);
93 void notifyMarkContextChanged();
94#if PLATFORM(COCOA)
95 virtual void prepareForDisplay(CompletionHandler<void(WTF::MachSendRight&&)>&&) = 0;
96#else
97 void prepareForDisplay(CompletionHandler<void()>&&);
98#endif
99 void getError(CompletionHandler<void(uint32_t)>&&);
100 void synthesizeGLError(uint32_t error);
101 void paintRenderingResultsToCanvas(WebCore::RenderingResourceIdentifier, CompletionHandler<void()>&&);
102 void paintCompositedResultsToCanvas(WebCore::RenderingResourceIdentifier, CompletionHandler<void()>&&);
103 void copyTextureFromMedia(WebCore::MediaPlayerIdentifier, uint32_t texture, uint32_t target, int32_t level, uint32_t internalFormat, uint32_t format, uint32_t type, bool premultiplyAlpha, bool flipY, CompletionHandler<void(bool)>&&);
104 void simulateEventForTesting(WebCore::GraphicsContextGLOpenGL::SimulatedEventForTesting);
105
106#include "RemoteGraphicsContextGLFunctionsGenerated.h" // NOLINT
107
108 void paintPixelBufferToImageBuffer(std::optional<WebCore::PixelBuffer>&&, WebCore::RenderingResourceIdentifier, CompletionHandler<void()>&&);
109
110 WeakPtr<GPUConnectionToWebProcess> m_gpuConnectionToWebProcess;
111 RefPtr<IPC::StreamServerConnection<RemoteGraphicsContextGL>> m_streamConnection;
112 RefPtr<WebCore::GraphicsContextGLOpenGL> m_context WTF_GUARDED_BY_LOCK(m_streamThread);
113 GraphicsContextGLIdentifier m_graphicsContextGLIdentifier;
114 Ref<RemoteRenderingBackend> m_renderingBackend;
115 ScopedWebGLRenderingResourcesRequest m_renderingResourcesRequest;
116 NO_UNIQUE_ADDRESS ThreadAssertion m_streamThread;
117};
