40class RemoteGraphicsContextGLCocoa final : public RemoteGraphicsContextGL {

42 RemoteGraphicsContextGLCocoa(GPUConnectionToWebProcess&, GraphicsContextGLIdentifier, RemoteRenderingBackend&, IPC::StreamConnectionBuffer&&);
43 ~RemoteGraphicsContextGLCocoa() final = default;

46 void platformWorkQueueInitialize(WebCore::GraphicsContextGLAttributes&&) final;

57Ref<RemoteGraphicsContextGL> RemoteGraphicsContextGL::create(GPUConnectionToWebProcess& gpuConnectionToWebProcess, GraphicsContextGLAttributes&& attributes, GraphicsContextGLIdentifier graphicsContextGLIdentifier, RemoteRenderingBackend& renderingBackend, IPC::StreamConnectionBuffer&& stream)
58{
59 auto instance = adoptRef(*new RemoteGraphicsContextGLCocoa(gpuConnectionToWebProcess, graphicsContextGLIdentifier, renderingBackend, WTFMove(stream)));
60 instance->initialize(WTFMove(attributes));
61 return instance;
62}

64RemoteGraphicsContextGLCocoa::RemoteGraphicsContextGLCocoa(GPUConnectionToWebProcess& gpuConnectionToWebProcess, GraphicsContextGLIdentifier graphicsContextGLIdentifier, RemoteRenderingBackend& renderingBackend, IPC::StreamConnectionBuffer&& stream)

71}

73void RemoteGraphicsContextGLCocoa::platformWorkQueueInitialize(WebCore::GraphicsContextGLAttributes&& attributes)
74{
75 assertIsCurrent(m_streamThread);
76 m_context = GraphicsContextGLOpenGL::createForGPUProcess(WTFMove(attributes), &m_swapChain);
77}
78
79void RemoteGraphicsContextGLCocoa::prepareForDisplay(CompletionHandler<void(WTF::MachSendRight&&)>&& completionHandler)
80{
81 assertIsCurrent(m_streamThread);
82 m_context->prepareForDisplay();
83 MachSendRight sendRight;
84 if (auto* surface = m_swapChain.displayBuffer().surface.get()) {
85#if HAVE(IOSURFACE_SET_OWNERSHIP_IDENTITY)
86 // Mark the IOSurface as being owned by the WebProcess even though it was constructed by the GPUProcess so that Jetsam knows which process to kill.
87 surface->setOwnershipIdentity(m_webProcessIdentityToken);
88#endif
89 sendRight = surface->createSendRight();
90 }
91 completionHandler(WTFMove(sendRight));
92}
