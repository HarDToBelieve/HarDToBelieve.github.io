38namespace WebKit {
39
40template<typename BackendType>
41class RemoteImageBuffer : public WebCore::ConcreteImageBuffer<BackendType>, public WebCore::DisplayList::Replayer::Delegate {
42 using BaseConcreteImageBuffer = WebCore::ConcreteImageBuffer<BackendType>;
43 using BaseConcreteImageBuffer::context;
44 using BaseConcreteImageBuffer::m_backend;
45 using BaseConcreteImageBuffer::putPixelBuffer;
46
47public:
48 static auto create(const WebCore::FloatSize& size, float resolutionScale, const WebCore::DestinationColorSpace& colorSpace, WebCore::PixelFormat pixelFormat, RemoteRenderingBackend& remoteRenderingBackend, WebCore::RenderingResourceIdentifier renderingResourceIdentifier)
49 {
50 return BaseConcreteImageBuffer::template create<RemoteImageBuffer>(size, resolutionScale, colorSpace, pixelFormat, nullptr, remoteRenderingBackend, renderingResourceIdentifier);
51 }
52
53 RemoteImageBuffer(const WebCore::ImageBufferBackend::Parameters& parameters, std::unique_ptr<BackendType>&& backend, RemoteRenderingBackend& remoteRenderingBackend, WebCore::RenderingResourceIdentifier renderingResourceIdentifier)
54 : BaseConcreteImageBuffer(parameters, WTFMove(backend), renderingResourceIdentifier)
55 , m_remoteRenderingBackend(remoteRenderingBackend)
56 , m_renderingResourceIdentifier(renderingResourceIdentifier)
57 {
58 m_remoteRenderingBackend.didCreateImageBufferBackend(m_backend->createImageBufferBackendHandle(), renderingResourceIdentifier);
59 }
60
61 ~RemoteImageBuffer()
62 {
63 // Unwind the context's state stack before destruction, since calls to restore may not have
64 // been flushed yet, or the web process may have terminated.
65 while (context().stackSize())
66 context().restore();
67 }
68
69#if HAVE(IOSURFACE_SET_OWNERSHIP_IDENTITY)
70 void setProcessOwnership(task_id_token_t newOwner)
71 {
72 if (m_backend)
73 m_backend->setProcessOwnership(newOwner);
74 }
75#endif
76
77private:
78 bool apply(WebCore::DisplayList::ItemHandle item, WebCore::GraphicsContext& context) override
79 {
80 if (item.is<WebCore::DisplayList::GetPixelBuffer>()) {
81 auto& getPixelBufferItem = item.get<WebCore::DisplayList::GetPixelBuffer>();
82 auto pixelBuffer = BaseConcreteImageBuffer::getPixelBuffer(getPixelBufferItem.outputFormat(), getPixelBufferItem.srcRect());
83 m_remoteRenderingBackend.populateGetPixelBufferSharedMemory(WTFMove(pixelBuffer));
84 return true;
85 }
86
87 if (item.is<WebCore::DisplayList::PutPixelBuffer>()) {
88 auto& putPixelBufferItem = item.get<WebCore::DisplayList::PutPixelBuffer>();
89 putPixelBuffer(putPixelBufferItem.pixelBuffer(), putPixelBufferItem.srcRect(), putPixelBufferItem.destPoint(), putPixelBufferItem.destFormat());
90 return true;
91 }
92
93 if (item.is<WebCore::DisplayList::FlushContext>()) {
94 BaseConcreteImageBuffer::flushContext();
95 auto identifier = item.get<WebCore::DisplayList::FlushContext>().identifier();
96 LOG_WITH_STREAM(SharedDisplayLists, stream << "Acknowledging Flush{" << identifier << "} in Image(" << m_renderingResourceIdentifier << ")");
97 m_remoteRenderingBackend.didFlush(identifier, m_renderingResourceIdentifier);
98 return true;
99 }
100
101 if (item.is<WebCore::DisplayList::MetaCommandChangeItemBuffer>()) {
102 auto nextBufferIdentifier = item.get<WebCore::DisplayList::MetaCommandChangeItemBuffer>().identifier();
103 LOG_WITH_STREAM(SharedDisplayLists, stream << "Switching to Items[" << nextBufferIdentifier << "]");
104 m_remoteRenderingBackend.setNextItemBufferToRead(nextBufferIdentifier, m_renderingResourceIdentifier);
105 return true;
106 }
107
108 return m_remoteRenderingBackend.applyMediaItem(item, context);
109 }
110
111 void didCreateMaskImageBuffer(WebCore::ImageBuffer& imageBuffer) final
112 {
113 m_remoteRenderingBackend.didCreateMaskImageBuffer(imageBuffer);
114 }
115
116 void didResetMaskImageBuffer() final
117 {
118 m_remoteRenderingBackend.didResetMaskImageBuffer();
119 }
120
121 RemoteRenderingBackend& m_remoteRenderingBackend;
122 WebCore::RenderingResourceIdentifier m_renderingResourceIdentifier;
123};
