103void RemoteRenderingBackend::stopListeningForIPC()
104{

127bool RemoteRenderingBackend::applyMediaItem(DisplayList::ItemHandle item, GraphicsContext& context)
128{
129 ASSERT(!RunLoop::isMain());
130
131 if (!item.is<DisplayList::PaintFrameForMedia>())
132 return false;
133
134 auto& mediaItem = item.get<DisplayList::PaintFrameForMedia>();
135 callOnMainRunLoopAndWait([&, gpuConnectionToWebProcess = m_gpuConnectionToWebProcess, mediaPlayerIdentifier = mediaItem.identifier()] {
136 auto player = gpuConnectionToWebProcess->remoteMediaPlayerManagerProxy().mediaPlayer(mediaPlayerIdentifier);
137 if (!player)
138 return;
139 // It is currently not safe to call paintFrameForMedia() off the main thread.
140 context.paintFrameForMedia(*player, mediaItem.destination());
141 });
142 return true;
143}

155void RemoteRenderingBackend::createImageBuffer(const FloatSize& logicalSize, RenderingMode renderingMode, float resolutionScale, const DestinationColorSpace& colorSpace, PixelFormat pixelFormat, RenderingResourceIdentifier renderingResourceIdentifier)
156{
157 ASSERT(!RunLoop::isMain());
158 ASSERT(renderingMode == RenderingMode::Accelerated || renderingMode == RenderingMode::Unaccelerated);
159
160 RefPtr<ImageBuffer> imageBuffer;
161
162 if (renderingMode == RenderingMode::Accelerated) {
163 if (auto acceleratedImageBuffer = AcceleratedRemoteImageBuffer::create(logicalSize, resolutionScale, colorSpace, pixelFormat, *this, renderingResourceIdentifier)) {
164#if HAVE(IOSURFACE_SET_OWNERSHIP_IDENTITY)
165 // Mark the IOSurface as being owned by the WebProcess even though it was constructed by the GPUProcess so that Jetsam knows which process to kill.
166 acceleratedImageBuffer->setProcessOwnership(m_gpuConnectionToWebProcess->webProcessIdentityToken());
167#endif
168 imageBuffer = WTFMove(acceleratedImageBuffer);
169 }
170 }
171
172 if (!imageBuffer)
173 imageBuffer = UnacceleratedRemoteImageBuffer::create(logicalSize, resolutionScale, colorSpace, pixelFormat, *this, renderingResourceIdentifier);
174
175 if (!imageBuffer) {
176 ASSERT_NOT_REACHED();
177 return;
178 }
179
180 m_remoteResourceCache.cacheImageBuffer(makeRef(*imageBuffer));
181 updateRenderingResourceRequest();
182
183 if (m_pendingWakeupInfo && m_pendingWakeupInfo->shouldPerformWakeup(renderingResourceIdentifier))
184 wakeUpAndApplyDisplayList(std::exchange(m_pendingWakeupInfo, std::nullopt)->arguments);
185}

187DisplayList::ReplayResult RemoteRenderingBackend::submit(const DisplayList::DisplayList& displayList, ImageBuffer& destination)
188{
189 if (displayList.isEmpty())
190 return { };
191
192 DisplayList::Replayer::Delegate* replayerDelegate = nullptr;
193 if (destination.renderingMode() == RenderingMode::Accelerated)
194 replayerDelegate = static_cast<AcceleratedRemoteImageBuffer*>(&destination);
195 else
196 replayerDelegate = static_cast<UnacceleratedRemoteImageBuffer*>(&destination);
197
198 return WebCore::DisplayList::Replayer {
199 destination.context(),
200 displayList,
201 &remoteResourceCache().imageBuffers(),
202 &remoteResourceCache().nativeImages(),
203 &remoteResourceCache().fonts(),
204 m_currentMaskImageBuffer.get(),
205 replayerDelegate
206 }.replay();
207}

209RefPtr<ImageBuffer> RemoteRenderingBackend::nextDestinationImageBufferAfterApplyingDisplayLists(ImageBuffer& initialDestination, size_t initialOffset, DisplayListReaderHandle& handle, GPUProcessWakeupReason reason)
210{
211 auto destination = makeRefPtr(initialDestination);
212 auto handleProtector = makeRef(handle);
213
214 auto offset = initialOffset;
215 size_t sizeToRead = 0;
216 do {
217 sizeToRead = handle.unreadBytes();
218 } while (!sizeToRead);
219
220 while (destination) {
221 auto displayList = handle.displayListForReading(offset, sizeToRead, *this);
222 MESSAGE_CHECK_WITH_RETURN_VALUE(displayList, nullptr, "Failed to map display list from shared memory");
223
224#if !LOG_DISABLED
225 auto startTime = MonotonicTime::now();
226#endif
227 auto result = submit(*displayList, *destination);
228 LOG_WITH_STREAM(SharedDisplayLists, stream << "Read [" << offset << ", " << offset + result.numberOfBytesRead << "]; Items[" << handle.identifier() << "] => Image(" << destination->renderingResourceIdentifier() << ") in " << MonotonicTime::now() - startTime);
229 MESSAGE_CHECK_WITH_RETURN_VALUE(result.reasonForStopping != DisplayList::StopReplayReason::InvalidItemOrExtent, nullptr, "Detected invalid display list item or extent");
230 MESSAGE_CHECK_WITH_RETURN_VALUE(result.reasonForStopping != DisplayList::StopReplayReason::OutOfMemory, nullptr, "Cound not allocate memory");
231
232 auto advanceResult = handle.advance(result.numberOfBytesRead);
233 MESSAGE_CHECK_WITH_RETURN_VALUE(advanceResult, nullptr, "Failed to advance display list reader handle");
234 sizeToRead = *advanceResult;
235
236 CheckedSize checkedOffset = offset;
237 checkedOffset += result.numberOfBytesRead;
238 MESSAGE_CHECK_WITH_RETURN_VALUE(!checkedOffset.hasOverflowed(), nullptr, "Overflowed when advancing shared display list handle offset");
239
240 offset = checkedOffset;
241 MESSAGE_CHECK_WITH_RETURN_VALUE(offset <= handle.sharedMemory().size(), nullptr, "Out-of-bounds offset into shared display list handle");
242
243 if (result.reasonForStopping == DisplayList::StopReplayReason::ChangeDestinationImageBuffer) {
244 destination = makeRefPtr(m_remoteResourceCache.cachedImageBuffer(*result.nextDestinationImageBuffer));
245 if (!destination) {
246 ASSERT(!m_pendingWakeupInfo);
247 m_pendingWakeupInfo = {{{ handle.identifier(), offset, *result.nextDestinationImageBuffer, reason }, std::nullopt }};
248 }
249 }
250
251 if (result.reasonForStopping == DisplayList::StopReplayReason::MissingCachedResource) {
252 m_pendingWakeupInfo = {{
253 { handle.identifier(), offset, destination->renderingResourceIdentifier(), reason },
254 result.missingCachedResourceIdentifier
255 }};
256 }
257
258 if (m_pendingWakeupInfo)
259 break;
260
261 if (!sizeToRead) {
262 if (reason != GPUProcessWakeupReason::ItemCountHysteresisExceeded)
263 break;
264
265 handle.startWaiting();
266 m_resumeDisplayListSemaphore.waitFor(30_us);
267
268 auto stopWaitingResult = handle.stopWaiting();
269 MESSAGE_CHECK_WITH_RETURN_VALUE(stopWaitingResult, nullptr, "Invalid waiting status detected when resuming display list processing");
270
271 auto resumeReadingInfo = stopWaitingResult.value();
272 if (!resumeReadingInfo)
273 break;
274
275 sizeToRead = handle.unreadBytes();
276 MESSAGE_CHECK_WITH_RETURN_VALUE(sizeToRead, nullptr, "No unread bytes when resuming display list processing");
277
278 auto newDestinationIdentifier = makeObjectIdentifier<RenderingResourceIdentifierType>(resumeReadingInfo->destination);
279 MESSAGE_CHECK_WITH_RETURN_VALUE(newDestinationIdentifier.isValid(), nullptr, "Invalid image buffer destination when resuming display list processing");
280
281 destination = makeRefPtr(m_remoteResourceCache.cachedImageBuffer(newDestinationIdentifier));
282 MESSAGE_CHECK_WITH_RETURN_VALUE(destination, nullptr, "Missing image buffer destination when resuming display list processing");
283
284 offset = resumeReadingInfo->offset;
285 }
286 }
287
288 return destination;
289}

291void RemoteRenderingBackend::wakeUpAndApplyDisplayList(const GPUProcessWakeupMessageArguments& arguments)
292{
293 ASSERT(!RunLoop::isMain());
294
295 TraceScope tracingScope(WakeUpAndApplyDisplayListStart, WakeUpAndApplyDisplayListEnd);
296 auto destinationImageBuffer = makeRefPtr(m_remoteResourceCache.cachedImageBuffer(arguments.destinationImageBufferIdentifier));
297 MESSAGE_CHECK(destinationImageBuffer, "Missing destination image buffer");
298
299 auto initialHandle = m_sharedDisplayListHandles.get(arguments.itemBufferIdentifier);
300 MESSAGE_CHECK(initialHandle, "Missing initial shared display list handle");
301
302 LOG_WITH_STREAM(SharedDisplayLists, stream << "Waking up to Items[" << arguments.itemBufferIdentifier << "] => Image(" << arguments.destinationImageBufferIdentifier << ") at " << arguments.offset);
303 destinationImageBuffer = nextDestinationImageBufferAfterApplyingDisplayLists(*destinationImageBuffer, arguments.offset, *initialHandle, arguments.reason);
304
305 while (destinationImageBuffer && m_pendingWakeupInfo) {
306 if (m_pendingWakeupInfo->missingCachedResourceIdentifier)
307 break;
308
309 auto nextHandle = m_sharedDisplayListHandles.get(m_pendingWakeupInfo->arguments.itemBufferIdentifier);
310 if (!nextHandle) {
311 // If the handle identifier is currently unknown, wait until the GPU process receives an
312 // IPC message with a shared memory handle to the next item buffer.
313 break;
314 }
315
316 // Otherwise, continue reading the next display list item buffer from the start.
317 auto arguments = std::exchange(m_pendingWakeupInfo, std::nullopt)->arguments;
318 destinationImageBuffer = nextDestinationImageBufferAfterApplyingDisplayLists(*destinationImageBuffer, arguments.offset, *nextHandle, arguments.reason);
319 }
320 LOG_WITH_STREAM(SharedDisplayLists, stream << "Going back to sleep.");
321}

328std::optional<SharedMemory::IPCHandle> RemoteRenderingBackend::updateSharedMemoryForGetPixelBufferHelper(size_t byteCount)
329{
330 MESSAGE_CHECK_WITH_RETURN_VALUE(!m_getPixelBufferSharedMemory || byteCount > m_getPixelBufferSharedMemory->size(), std::nullopt, "The existing Shmem for getPixelBuffer() is already big enough to handle the request");
331
332 if (byteCount > 64 * MB) {
333 // Just a sanity check. A 4K image is 36MB.
334 return std::nullopt;
335 }
336
337 destroyGetPixelBufferSharedMemory();
338 m_getPixelBufferSharedMemory = SharedMemory::allocate(byteCount);
339 SharedMemory::Handle handle;
340 if (m_getPixelBufferSharedMemory)
341 m_getPixelBufferSharedMemory->createHandle(handle, SharedMemory::Protection::ReadOnly);
342 return SharedMemory::IPCHandle { WTFMove(handle), m_getPixelBufferSharedMemory ? m_getPixelBufferSharedMemory->size() : 0 };
343}

409void RemoteRenderingBackend::getShareableBitmapForImageBuffer(WebCore::RenderingResourceIdentifier identifier, WebCore::PreserveResolution preserveResolution, CompletionHandler<void(ShareableBitmap::Handle&&)>&& completionHandler)
410{
411 ASSERT(!RunLoop::isMain());
412
413 ShareableBitmap::Handle handle;
414 [&]() {
415 auto imageBuffer = m_remoteResourceCache.cachedImageBuffer(identifier);
416 if (!imageBuffer)
417 return;
418 auto image = imageBuffer->copyNativeImage(WebCore::BackingStoreCopy::DontCopyBackingStore);
419 if (!image)
420 return;
421 auto backendSize = imageBuffer->backendSize();
422 auto resultSize = preserveResolution == WebCore::PreserveResolution::Yes ? backendSize : imageBuffer->logicalSize();
423 auto bitmap = ShareableBitmap::createShareable(resultSize, { });
424 if (!bitmap)
425 return;
426 auto context = bitmap->createGraphicsContext();
427 if (!context)
428 return;
429 context->drawNativeImage(*image, resultSize, FloatRect { { }, resultSize }, FloatRect { { }, backendSize }, { WebCore::CompositeOperator::Copy });
430 bitmap->createHandle(handle);
431 }();
432 completionHandler(WTFMove(handle));
433}

614{

616 bool hasActiveRequest = m_renderingResourcesRequest.isRequested();
617 if (hasActiveDrawables && !hasActiveRequest)
618 m_renderingResourcesRequest = ScopedRenderingResourcesRequest::acquire();
619 else if (!hasActiveDrawables && hasActiveRequest)
620 m_renderingResourcesRequest = { };
