103 template<typename T>
104 std::optional<WebCore::DisplayList::ItemHandle> WARN_UNUSED_RETURN decodeAndCreate(const uint8_t* data, size_t length, uint8_t* handleLocation)
105 {
106 if (auto item = IPC::Decoder::decodeSingleObject<T>(data, length)) {
107 // FIXME: WebKit2 should not need to know that the first 8 bytes at the handle are reserved for the type.
108 // Need to figure out a way to keep this knowledge within display list code in WebCore.
109 new (handleLocation + sizeof(uint64_t)) T(WTFMove(*item));
110 return {{ handleLocation }};
111 }
112 return std::nullopt;
113 }

145 struct PendingWakeupInformation {
146 GPUProcessWakeupMessageArguments arguments;
147 std::optional<WebCore::RenderingResourceIdentifier> missingCachedResourceIdentifier;
148
149 bool shouldPerformWakeup(WebCore::RenderingResourceIdentifier identifier) const
150 {
151 return arguments.destinationImageBufferIdentifier == identifier
152 || missingCachedResourceIdentifier == identifier;
153 }
154
155 bool shouldPerformWakeup(WebCore::DisplayList::ItemBufferIdentifier identifier) const
156 {
157 return arguments.itemBufferIdentifier == identifier;
158 }
