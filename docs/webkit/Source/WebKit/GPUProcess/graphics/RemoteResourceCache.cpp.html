27#include "RemoteResourceCache.h"
28
29#if ENABLE(GPU_PROCESS)
30
31namespace WebKit {
32using namespace WebCore;
33
34void RemoteResourceCache::cacheImageBuffer(Ref<ImageBuffer>&& imageBuffer)
35{
36 auto addResult = m_imageBuffers.add(imageBuffer->renderingResourceIdentifier(), WTFMove(imageBuffer));
37 ASSERT_UNUSED(addResult, addResult.isNewEntry);
38}
39
40ImageBuffer* RemoteResourceCache::cachedImageBuffer(RenderingResourceIdentifier renderingResourceIdentifier)
41{
42 return m_imageBuffers.get(renderingResourceIdentifier);
43}
44
45void RemoteResourceCache::cacheNativeImage(Ref<NativeImage>&& image)
46{
47 auto addResult = m_nativeImages.add(image->renderingResourceIdentifier(), WTFMove(image));
48 ASSERT_UNUSED(addResult, addResult.isNewEntry);
49}
50
51void RemoteResourceCache::cacheFont(Ref<Font>&& font)
52{
53 auto addResult = m_fonts.add(font->renderingResourceIdentifier(), WTFMove(font));
54 ASSERT_UNUSED(addResult, addResult.isNewEntry);
55}
56
57void RemoteResourceCache::deleteAllFonts()
58{
59 m_fonts.clear();
60}
61
62void RemoteResourceCache::releaseRemoteResource(RenderingResourceIdentifier renderingResourceIdentifier)
63{
64 if (m_imageBuffers.remove(renderingResourceIdentifier))
65 return;
66 if (m_nativeImages.remove(renderingResourceIdentifier))
67 return;
68 if (m_fonts.remove(renderingResourceIdentifier))
69 return;
70 // Caching the remote resource should have happened before releasing it.
71 ASSERT_NOT_REACHED();
72}
