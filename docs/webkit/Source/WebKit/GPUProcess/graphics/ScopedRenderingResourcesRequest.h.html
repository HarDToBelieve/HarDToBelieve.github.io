26#pragma once
27
28#include <atomic>
29#include <utility>
30
31namespace WebKit {
32
33class ScopedRenderingResourcesRequest {
34public:
35 ScopedRenderingResourcesRequest() = default;
36 ScopedRenderingResourcesRequest(ScopedRenderingResourcesRequest&& other)
37 : m_requested(std::exchange(other.m_requested, false))
38 {
39 }
40 ~ScopedRenderingResourcesRequest()
41 {
42 reset();
43 }
44 ScopedRenderingResourcesRequest& operator=(ScopedRenderingResourcesRequest&& other)
45 {
46 if (this != &other) {
47 reset();
48 m_requested = std::exchange(other.m_requested, false);
49 }
50 return *this;
51 }
52 bool isRequested() const { return m_requested; }
53 void reset()
54 {
55 if (!m_requested)
56 return;
57 ASSERT(s_requests);
58 --s_requests;
59 if (!s_requests)
60 scheduleFreeRenderingResources();
61 }
62 static ScopedRenderingResourcesRequest acquire()
63 {
64 return { DidRequest };
65 }
66private:
67 static void scheduleFreeRenderingResources();
68 static void freeRenderingResources();
69 enum RequestState { DidRequest };
70 ScopedRenderingResourcesRequest(RequestState)
71 : m_requested(true)
72 {
73 ++s_requests;
74 }
75 static std::atomic<unsigned> s_requests;
76 bool m_requested { false };
77};
78
79}
