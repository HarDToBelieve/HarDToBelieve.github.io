44void GPUProcess::initializeProcess(const AuxiliaryProcessInitializationParameters&)
45{
46#if PLATFORM(MAC) && !PLATFORM(MACCATALYST)
47 // Having a window server connection in this process would result in spin logs (<rdar://problem/13239119>).
48 OSStatus error = SetApplicationIsDaemon(true);
49 ASSERT_UNUSED(error, error == noErr);
50#endif
51
52#if ENABLE(LOWER_FORMATREADERBUNDLE_CODESIGNING_REQUIREMENTS)
53 // For testing in engineering builds, allow CoreMedia to load the MediaFormatReader bundle no matter its code signature.
54 auto userDefaults = adoptNS([[NSUserDefaults alloc] initWithSuiteName:@"com.apple.coremedia"]);
55 [userDefaults registerDefaults:@{ @"pluginformatreader_unsigned": @YES }];
56#endif
57
58 launchServicesCheckIn();
59}
60
61void GPUProcess::initializeProcessName(const AuxiliaryProcessInitializationParameters& parameters)
62{
63#if !PLATFORM(MACCATALYST)
64 NSString *applicationName = [NSString stringWithFormat:WEB_UI_STRING("%@ Graphics and Media", "visible name of the GPU process. The argument is the application name."), (NSString *)parameters.uiProcessName];
65 _LSSetApplicationInformationItem(kLSDefaultSessionID, _LSGetCurrentApplicationASN(), _kLSDisplayNameKey, (CFStringRef)applicationName, nullptr);
66#endif
67}
68
69void GPUProcess::initializeSandbox(const AuxiliaryProcessInitializationParameters& parameters, SandboxInitializationParameters& sandboxParameters)
70{
71 // Need to overide the default, because service has a different bundle ID.
72 NSBundle *webKit2Bundle = [NSBundle bundleForClass:NSClassFromString(@"WKWebView")];
73
74 sandboxParameters.setOverrideSandboxProfilePath([webKit2Bundle pathForResource:@"com.apple.WebKit.GPUProcess" ofType:@"sb"]);
75
76 AuxiliaryProcess::initializeSandbox(parameters, sandboxParameters);
77}
