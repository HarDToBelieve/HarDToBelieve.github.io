38UniqueRef<LocalAudioSessionRoutingArbitrator> LocalAudioSessionRoutingArbitrator::create(GPUConnectionToWebProcess& gpuConnectionToWebProcess)
39{
40 return makeUniqueRef<LocalAudioSessionRoutingArbitrator>(gpuConnectionToWebProcess);
41}
42
43LocalAudioSessionRoutingArbitrator::LocalAudioSessionRoutingArbitrator(GPUConnectionToWebProcess& gpuConnectionToWebProcess)
44 : m_connectionToWebProcess(gpuConnectionToWebProcess)
45{
46}
47
48LocalAudioSessionRoutingArbitrator::~LocalAudioSessionRoutingArbitrator() = default;
49
50void LocalAudioSessionRoutingArbitrator::processDidTerminate()
51{
52 leaveRoutingAbritration();
53}
54
55void LocalAudioSessionRoutingArbitrator::beginRoutingArbitrationWithCategory(AudioSession::CategoryType category, CompletionHandler<void(RoutingArbitrationError, DefaultRouteChanged)>&& callback)
56{
57 m_connectionToWebProcess.connection().sendWithAsyncReply(Messages::GPUProcessConnection::BeginRoutingArbitrationWithCategory(category), WTFMove(callback), 0);
58}
59
60void LocalAudioSessionRoutingArbitrator::leaveRoutingAbritration()
61{
62 m_connectionToWebProcess.connection().send(Messages::GPUProcessConnection::EndRoutingArbitration(), 0);
63}
