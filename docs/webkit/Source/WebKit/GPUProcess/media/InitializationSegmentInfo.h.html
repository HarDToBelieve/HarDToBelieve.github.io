26#pragma once
27
28#if ENABLE(GPU_PROCESS) && ENABLE(MEDIA_SOURCE)
29
30#include "Decoder.h"
31#include "Encoder.h"
32#include "RemoteMediaDescription.h"
33#include "TrackPrivateRemoteIdentifier.h"
34#include <wtf/MediaTime.h>
35#include <wtf/Vector.h>
36
37namespace WebKit {
38
39struct InitializationSegmentInfo {
40 MediaTime duration;
41
42 struct TrackInformation {
43 MediaDescriptionInfo description;
44 TrackPrivateRemoteIdentifier identifier;
45
46 template<class Encoder>
47 void encode(Encoder& encoder) const
48 {
49 encoder << description;
50 encoder << identifier;
51 }
52
53 template <class Decoder>
54 static std::optional<TrackInformation> decode(Decoder& decoder)
55 {
56 std::optional<MediaDescriptionInfo> mediaDescription;
57 decoder >> mediaDescription;
58 if (!mediaDescription)
59 return std::nullopt;
60
61 std::optional<TrackPrivateRemoteIdentifier> identifier;
62 decoder >> identifier;
63 if (!identifier)
64 return std::nullopt;
65
66 return {{
67 WTFMove(*mediaDescription),
68 WTFMove(*identifier)
69 }};
70 }
71 };
72
73 Vector<TrackInformation> audioTracks;
74 Vector<TrackInformation> videoTracks;
75 Vector<TrackInformation> textTracks;
76
77 template<class Encoder>
78 void encode(Encoder& encoder) const
79 {
80 encoder << duration;
81 encoder << audioTracks;
82 encoder << videoTracks;
83 encoder << textTracks;
84 }
85
86 template <class Decoder>
87 static std::optional<InitializationSegmentInfo> decode(Decoder& decoder)
88 {
89 std::optional<MediaTime> duration;
90 decoder >> duration;
91 if (!duration)
92 return std::nullopt;
93
94 std::optional<Vector<TrackInformation>> audioTracks;
95 decoder >> audioTracks;
96 if (!audioTracks)
97 return std::nullopt;
98
99 std::optional<Vector<TrackInformation>> videoTracks;
100 decoder >> videoTracks;
101 if (!videoTracks)
102 return std::nullopt;
103
104 std::optional<Vector<TrackInformation>> textTracks;
105 decoder >> textTracks;
106 if (!textTracks)
107 return std::nullopt;
108
109 return {{
110 WTFMove(*duration),
111 WTFMove(*audioTracks),
112 WTFMove(*videoTracks),
113 WTFMove(*textTracks)
114 }};
115 }
116};
117
118} // namespace WebKit
119
120#endif // ENABLE(GPU_PROCESS) && ENABLE(MEDIA_SOURCE)
