34namespace WebKit {
35
36RemoteAudioHardwareListenerProxy::RemoteAudioHardwareListenerProxy(GPUConnectionToWebProcess& gpuConnection, RemoteAudioHardwareListenerIdentifier&& identifier)
37 : m_gpuConnection(makeWeakPtr(gpuConnection))
38 , m_identifier(WTFMove(identifier))
39 , m_listener(WebCore::AudioHardwareListener::create(*this))
40{
41 audioOutputDeviceChanged();
42}
43
44RemoteAudioHardwareListenerProxy::~RemoteAudioHardwareListenerProxy() = default;
45
46void RemoteAudioHardwareListenerProxy::audioHardwareDidBecomeActive()
47{
48 if (!m_gpuConnection)
49 return;
50
51 m_gpuConnection->connection().send(Messages::RemoteAudioHardwareListener::AudioHardwareDidBecomeActive(), m_identifier);
52}
53
54void RemoteAudioHardwareListenerProxy::audioHardwareDidBecomeInactive()
55{
56 if (!m_gpuConnection)
57 return;
58
59 m_gpuConnection->connection().send(Messages::RemoteAudioHardwareListener::AudioHardwareDidBecomeInactive(), m_identifier);
60}
61
62void RemoteAudioHardwareListenerProxy::audioOutputDeviceChanged()
63{
64 if (!m_gpuConnection)
65 return;
66
67 auto supportedBufferSizes = m_listener->supportedBufferSizes();
68 m_gpuConnection->connection().send(Messages::RemoteAudioHardwareListener::AudioOutputDeviceChanged(supportedBufferSizes.minimum, supportedBufferSizes.maximum), m_identifier);
69}
