41static bool categoryCanMixWithOthers(AudioSession::CategoryType category)
42{
43 return category == AudioSession::CategoryType::AmbientSound;
44}

46RemoteAudioSessionProxyManager::RemoteAudioSessionProxyManager()
47 : m_session(AudioSession::create())
48{
49 m_session->addInterruptionObserver(*this);
50}
51
52RemoteAudioSessionProxyManager::~RemoteAudioSessionProxyManager()
53{
54 m_session->removeInterruptionObserver(*this);
55}

57void RemoteAudioSessionProxyManager::addProxy(RemoteAudioSessionProxy& proxy)
58{
59 ASSERT(!m_proxies.contains(proxy));
60 m_proxies.add(proxy);
61 updateCategory();
62}

64void RemoteAudioSessionProxyManager::removeProxy(RemoteAudioSessionProxy& proxy)
65{
66 ASSERT(m_proxies.contains(proxy));
67 m_proxies.remove(proxy);
68 updateCategory();
69}

71void RemoteAudioSessionProxyManager::updateCategory()
72{
73 AudioSession::CategoryType category = AudioSession::CategoryType::None;
74 RouteSharingPolicy policy = RouteSharingPolicy::Default;
75
76 HashCountedSet<AudioSession::CategoryType, WTF::IntHash<AudioSession::CategoryType>, WTF::StrongEnumHashTraits<AudioSession::CategoryType>> categoryCounts;
77 HashCountedSet<RouteSharingPolicy, WTF::IntHash<RouteSharingPolicy>, WTF::StrongEnumHashTraits<RouteSharingPolicy>> policyCounts;
78 for (auto& otherProxy : m_proxies) {
79 categoryCounts.add(otherProxy.category());
80 policyCounts.add(otherProxy.routeSharingPolicy());
81 }
82
83 if (categoryCounts.contains(AudioSession::CategoryType::PlayAndRecord))
84 category = AudioSession::CategoryType::PlayAndRecord;
85 else if (categoryCounts.contains(AudioSession::CategoryType::RecordAudio))
86 category = AudioSession::CategoryType::RecordAudio;
87 else if (categoryCounts.contains(AudioSession::CategoryType::MediaPlayback))
88 category = AudioSession::CategoryType::MediaPlayback;
89 else if (categoryCounts.contains(AudioSession::CategoryType::SoloAmbientSound))
90 category = AudioSession::CategoryType::SoloAmbientSound;
91 else if (categoryCounts.contains(AudioSession::CategoryType::AmbientSound))
92 category = AudioSession::CategoryType::AmbientSound;
93 else if (categoryCounts.contains(AudioSession::CategoryType::AudioProcessing))
94 category = AudioSession::CategoryType::AudioProcessing;
95 else
96 category = AudioSession::CategoryType::None;
97
98 policy = RouteSharingPolicy::Default;
99 if (policyCounts.contains(RouteSharingPolicy::LongFormVideo))
100 policy = RouteSharingPolicy::LongFormVideo;
101 else if (policyCounts.contains(RouteSharingPolicy::LongFormAudio))
102 policy = RouteSharingPolicy::LongFormAudio;
103 else if (policyCounts.contains(RouteSharingPolicy::Independent))
104 ASSERT_NOT_REACHED();
105
106 m_session->setCategory(category, policy);
107}

109void RemoteAudioSessionProxyManager::setPreferredBufferSizeForProcess(RemoteAudioSessionProxy& proxy, size_t preferredBufferSize)
110{
111 for (auto& otherProxy : m_proxies) {
112 if (otherProxy.preferredBufferSize() && otherProxy.preferredBufferSize() < preferredBufferSize)
113 preferredBufferSize = otherProxy.preferredBufferSize();
114 }
115
116 m_session->setPreferredBufferSize(preferredBufferSize);
117}

119bool RemoteAudioSessionProxyManager::tryToSetActiveForProcess(RemoteAudioSessionProxy& proxy, bool active)
120{
121 ASSERT(m_proxies.contains(proxy));
122
123 size_t activeProxyCount { 0 };
124 for (auto& otherProxy : m_proxies) {
125 if (otherProxy.isActive())
126 ++activeProxyCount;
127 }
128
129 if (!active && activeProxyCount > 1) {
130 // This proxy wants to de-activate, but other proxies are still
131 // active. No-op, and return deactivation was sucessful.
132 return true;
133 }
134
135 if (!active) {
136 // This proxy wants to de-activate, and is the last remaining active
137 // proxy. Deactivate the session, and return whether that deactivation
138 // was sucessful;
139 return m_session->tryToSetActive(false);
140 }
141
142 if (active && !activeProxyCount) {
143 // This proxy and only this proxy wants to become active. Activate
144 // the session, and return whether that activation was successful.
145 return m_session->tryToSetActive(active);
146 }
147
148 // If this proxy is Ambient, and the session is already active, this
149 // proxy will mix with the active proxies. No-op, and return activation
150 // was sucessful.
151 if (categoryCanMixWithOthers(proxy.category()))
152 return true;
153
154#if PLATFORM(IOS_FAMILY)
155 // Otherwise, this proxy wants to become active, but there are other
156 // proxies who are already active. Walk over the proxies, and interrupt
157 // those proxies whose categories indicate they cannot mix with others.
158 for (auto& otherProxy : m_proxies) {
159 if (otherProxy.processIdentifier() == proxy.processIdentifier())
160 continue;
161
162 if (!otherProxy.isActive())
163 continue;
164
165 if (categoryCanMixWithOthers(otherProxy.category()))
166 continue;
167
168 otherProxy.beginInterruption();
169 }
170#endif
171 return true;
172}

174void RemoteAudioSessionProxyManager::beginAudioSessionInterruption()
175{
176 for (auto& proxy : m_proxies) {
177 if (proxy.isActive())
178 proxy.beginInterruption();
179 }
180}

182void RemoteAudioSessionProxyManager::endAudioSessionInterruption(AudioSession::MayResume mayResume)
183{
184 for (auto& proxy : m_proxies) {
185 if (proxy.isActive())
186 proxy.endInterruption(mayResume);
187 }
188}
