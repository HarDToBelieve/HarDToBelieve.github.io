50static CDMFactory* factoryForKeySystem(const String& keySystem)
51{
52 auto& factories = CDMFactory::registeredFactories();
53 auto foundIndex = factories.findMatching([&] (auto& factory) { return factory->supportsKeySystem(keySystem); });
54 if (foundIndex == notFound)
55 return nullptr;
56 return factories[foundIndex];
57}
58
59void RemoteCDMFactoryProxy::createCDM(const String& keySystem, CompletionHandler<void(RemoteCDMIdentifier&&, RemoteCDMConfiguration&&)>&& completion)
60{
61 auto factory = factoryForKeySystem(keySystem);
62 if (!factory) {
63 completion({ }, { });
64 return;
65 }
66
67 auto privateCDM = factory->createCDM(keySystem);
68 if (!privateCDM) {
69 completion({ }, { });
70 return;
71 }
72
73 auto proxy = RemoteCDMProxy::create(makeWeakPtr(this), WTFMove(privateCDM));
74 auto identifier = RemoteCDMIdentifier::generate();
75 RemoteCDMConfiguration configuration = proxy->configuration();
76 addProxy(identifier, WTFMove(proxy));
77 completion(WTFMove(identifier), WTFMove(configuration));
78}
79
80void RemoteCDMFactoryProxy::supportsKeySystem(const String& keySystem, CompletionHandler<void(bool)>&& completion)
81{
82 completion(factoryForKeySystem(keySystem));
83}

124void RemoteCDMFactoryProxy::addProxy(const RemoteCDMIdentifier& identifier, std::unique_ptr<RemoteCDMProxy>&& proxy)
125{
126 ASSERT(!m_proxies.contains(identifier));
127 m_proxies.set(identifier, WTFMove(proxy));
128}
129
130void RemoteCDMFactoryProxy::removeProxy(const RemoteCDMIdentifier& identifier)
131{
132 ASSERT(m_proxies.contains(identifier));
133 m_proxies.remove(identifier);
134}

136void RemoteCDMFactoryProxy::addInstance(const RemoteCDMInstanceIdentifier& identifier, std::unique_ptr<RemoteCDMInstanceProxy>&& instance)
137{
138 ASSERT(!m_instances.contains(identifier));
139 m_instances.set(identifier, WTFMove(instance));
140}
141
142void RemoteCDMFactoryProxy::removeInstance(const RemoteCDMInstanceIdentifier& identifier)
143{
144 ASSERT(m_instances.contains(identifier));
145 m_instances.remove(identifier);
146 if (m_gpuConnectionToWebProcess && allowsExitUnderMemoryPressure())
147 m_gpuConnectionToWebProcess->gpuProcess().tryExitIfUnusedAndUnderMemoryPressure();
148}
149
150RemoteCDMInstanceProxy* RemoteCDMFactoryProxy::getInstance(const RemoteCDMInstanceIdentifier& identifier)
151{
152 return m_instances.get(identifier);
153}

157 ASSERT(!m_sessions.contains(identifier));

159}

161void RemoteCDMFactoryProxy::removeSession(const RemoteCDMInstanceSessionIdentifier& identifier)

167bool RemoteCDMFactoryProxy::allowsExitUnderMemoryPressure() const
