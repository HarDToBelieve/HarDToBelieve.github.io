43RemoteImageDecoderAVFProxy::RemoteImageDecoderAVFProxy(GPUConnectionToWebProcess& connectionToWebProcess)
44 : m_connectionToWebProcess(makeWeakPtr(connectionToWebProcess))
45{
46}

48void RemoteImageDecoderAVFProxy::createDecoder(const IPC::DataReference& data, const String& mimeType, CompletionHandler<void(std::optional<ImageDecoderIdentifier>&&)>&& completionHandler)
49{
50 auto imageDecoder = ImageDecoderAVFObjC::create(SharedBuffer::create(data.data(), data.size()), mimeType, AlphaOption::Premultiplied, GammaAndColorProfileOption::Ignored);
51
52 std::optional<ImageDecoderIdentifier> imageDecoderIdentifier;
53 if (!imageDecoder)
54 return completionHandler(WTFMove(imageDecoderIdentifier));
55
56 auto identifier = ImageDecoderIdentifier::generate();
57 m_imageDecoders.add(identifier, imageDecoder.copyRef());
58
59 imageDecoder->setEncodedDataStatusChangeCallback([this, identifier](auto) mutable {
60 encodedDataStatusChanged(identifier);
61 });
62
63 imageDecoderIdentifier = identifier;
64 completionHandler(WTFMove(imageDecoderIdentifier));
65}
66
67void RemoteImageDecoderAVFProxy::deleteDecoder(ImageDecoderIdentifier identifier)
68{
69 ASSERT(m_imageDecoders.contains(identifier));
70 if (!m_imageDecoders.contains(identifier))
71 return;
72
73 m_imageDecoders.take(identifier);
74 if (m_connectionToWebProcess && allowsExitUnderMemoryPressure())
75 m_connectionToWebProcess->gpuProcess().tryExitIfUnusedAndUnderMemoryPressure();
76}
77
78void RemoteImageDecoderAVFProxy::encodedDataStatusChanged(ImageDecoderIdentifier identifier)
79{
80 if (!m_connectionToWebProcess || !m_imageDecoders.contains(identifier))
81 return;
82
83 auto imageDecoder = m_imageDecoders.get(identifier);
84 m_connectionToWebProcess->connection().send(Messages::RemoteImageDecoderAVFManager::EncodedDataStatusChanged(identifier, imageDecoder->frameCount(), imageDecoder->size(), imageDecoder->hasTrack()), 0);
85}
86
87void RemoteImageDecoderAVFProxy::setExpectedContentSize(ImageDecoderIdentifier identifier, long long expectedContentSize)
88{
89 ASSERT(m_imageDecoders.contains(identifier));
90 if (!m_imageDecoders.contains(identifier))
91 return;
92
93 m_imageDecoders.get(identifier)->setExpectedContentSize(expectedContentSize);
94}

96void RemoteImageDecoderAVFProxy::setData(ImageDecoderIdentifier identifier, const IPC::DataReference& data, bool allDataReceived, CompletionHandler<void(size_t frameCount, const IntSize& size, bool hasTrack, std::optional<Vector<ImageDecoder::FrameInfo>>&&)>&& completionHandler)
97{
98 ASSERT(m_imageDecoders.contains(identifier));
99 if (!m_imageDecoders.contains(identifier)) {
100 completionHandler(0, { }, false, std::nullopt);
101 return;
102 }
103
104 auto imageDecoder = m_imageDecoders.get(identifier);
105 imageDecoder->setData(SharedBuffer::create(data.data(), data.size()), allDataReceived);
106
107 auto frameCount = imageDecoder->frameCount();
108
109 std::optional<Vector<ImageDecoder::FrameInfo>> frameInfos;
110 if (frameCount)
111 frameInfos = imageDecoder->frameInfos();
112
113 completionHandler(frameCount, imageDecoder->size(), imageDecoder->hasTrack(), WTFMove(frameInfos));
114}

116void RemoteImageDecoderAVFProxy::createFrameImageAtIndex(ImageDecoderIdentifier identifier, size_t index, CompletionHandler<void(std::optional<WTF::MachSendRight>&&, std::optional<WebCore::DestinationColorSpace>&&)>&& completionHandler)
117{
118 ASSERT(m_imageDecoders.contains(identifier));
119 std::optional<WTF::MachSendRight> sendRight;
120 std::optional<WebCore::DestinationColorSpace> colorSpace;
121 if (!m_imageDecoders.contains(identifier)) {
122 completionHandler(WTFMove(sendRight), WTFMove(colorSpace));
123 return;
124 }
125
126 auto frameImage = m_imageDecoders.get(identifier)->createFrameImageAtIndex(index);
127 if (!frameImage) {
128 completionHandler(WTFMove(sendRight), WTFMove(colorSpace));
129 return;
130 }
131
132 if (auto surface = IOSurface::createFromImage(frameImage.get())) {
133 sendRight = surface->createSendRight();
134 colorSpace = surface->colorSpace();
135 }
136
137 completionHandler(WTFMove(sendRight), WTFMove(colorSpace));
138}
139
140void RemoteImageDecoderAVFProxy::clearFrameBufferCache(ImageDecoderIdentifier identifier, size_t index)
141{
142 ASSERT(m_imageDecoders.contains(identifier));
143 if (auto* imageDecoder = m_imageDecoders.get(identifier))
144 imageDecoder->clearFrameBufferCache(std::min(index, imageDecoder->frameCount() - 1));
145}
146
147bool RemoteImageDecoderAVFProxy::allowsExitUnderMemoryPressure() const
148{
149 return m_imageDecoders.isEmpty();
150}
