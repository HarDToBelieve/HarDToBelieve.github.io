44RemoteLegacyCDMFactoryProxy::RemoteLegacyCDMFactoryProxy(GPUConnectionToWebProcess& connection)
45 : m_gpuConnectionToWebProcess(makeWeakPtr(connection))
46{
47}
48
49RemoteLegacyCDMFactoryProxy::~RemoteLegacyCDMFactoryProxy()
50{
51 if (!m_gpuConnectionToWebProcess)
52 return;
53
54 for (auto const& session : m_sessions)
55 m_gpuConnectionToWebProcess->messageReceiverMap().removeMessageReceiver(Messages::RemoteLegacyCDMSessionProxy::messageReceiverName(), session.key.toUInt64());
56
57 for (auto const& proxy : m_proxies)
58 m_gpuConnectionToWebProcess->messageReceiverMap().removeMessageReceiver(Messages::RemoteLegacyCDMProxy::messageReceiverName(), proxy.key.toUInt64());
59}

61void RemoteLegacyCDMFactoryProxy::createCDM(const String& keySystem, std::optional<MediaPlayerIdentifier>&& optionalPlayerId, CompletionHandler<void(RemoteLegacyCDMIdentifier&&)>&& completion)
62{
63 auto privateCDM = LegacyCDM::create(keySystem);
64 if (!privateCDM) {
65 completion({ });
66 return;
67 }
68
69 MediaPlayerIdentifier playerId;
70 if (optionalPlayerId)
71 playerId = WTFMove(optionalPlayerId.value());
72
73 auto proxy = RemoteLegacyCDMProxy::create(makeWeakPtr(this), WTFMove(playerId), WTFMove(privateCDM));
74 auto identifier = RemoteLegacyCDMIdentifier::generate();
75 addProxy(identifier, WTFMove(proxy));
76 completion(WTFMove(identifier));
77}
78
79void RemoteLegacyCDMFactoryProxy::supportsKeySystem(const String& keySystem, std::optional<String> mimeType, CompletionHandler<void(bool)>&& completion)
80{
81 if (mimeType)
82 completion(LegacyCDM::keySystemSupportsMimeType(keySystem, *mimeType));
83 else
84 completion(LegacyCDM::supportsKeySystem(keySystem));
85}

89 if (auto* proxy = m_proxies.get(makeObjectIdentifier<RemoteLegacyCDMIdentifierType>(decoder.destinationID())))

113void RemoteLegacyCDMFactoryProxy::addProxy(RemoteLegacyCDMIdentifier identifier, std::unique_ptr<RemoteLegacyCDMProxy>&& proxy)
114{
115 if (!m_gpuConnectionToWebProcess)
116 return;
117
118 m_gpuConnectionToWebProcess->messageReceiverMap().addMessageReceiver(Messages::RemoteLegacyCDMProxy::messageReceiverName(), identifier.toUInt64(), *proxy);
119
120 ASSERT(!m_proxies.contains(identifier));
121 m_proxies.set(identifier, WTFMove(proxy));
122}
123
124void RemoteLegacyCDMFactoryProxy::removeProxy(RemoteLegacyCDMIdentifier identifier)
125{
126 if (!m_gpuConnectionToWebProcess)
127 return;
128
129 m_gpuConnectionToWebProcess->messageReceiverMap().removeMessageReceiver(Messages::RemoteLegacyCDMProxy::messageReceiverName(), identifier.toUInt64());
130
131 ASSERT(m_proxies.contains(identifier));
132 m_proxies.remove(identifier);
133}
134
135void RemoteLegacyCDMFactoryProxy::addSession(RemoteLegacyCDMSessionIdentifier identifier, std::unique_ptr<RemoteLegacyCDMSessionProxy>&& session)
136{
137 if (!m_gpuConnectionToWebProcess)
138 return;
139
140 m_gpuConnectionToWebProcess->messageReceiverMap().addMessageReceiver(Messages::RemoteLegacyCDMSessionProxy::messageReceiverName(), identifier.toUInt64(), *session);
141
142 ASSERT(!m_sessions.contains(identifier));
143 m_sessions.set(identifier, WTFMove(session));
144}
145
146void RemoteLegacyCDMFactoryProxy::removeSession(RemoteLegacyCDMSessionIdentifier identifier)
147{
148 if (!m_gpuConnectionToWebProcess)
149 return;
150
151 m_gpuConnectionToWebProcess->messageReceiverMap().removeMessageReceiver(Messages::RemoteLegacyCDMSessionProxy::messageReceiverName(), identifier.toUInt64());
152
153 ASSERT(m_sessions.contains(identifier));
154 m_sessions.remove(identifier);
155
156 if (m_gpuConnectionToWebProcess && allowsExitUnderMemoryPressure())
157 m_gpuConnectionToWebProcess->gpuProcess().tryExitIfUnusedAndUnderMemoryPressure();
158}
159
160RemoteLegacyCDMSessionProxy* RemoteLegacyCDMFactoryProxy::getSession(const RemoteLegacyCDMSessionIdentifier& identifier) const
161{
162 auto results = m_sessions.find(identifier);
163 if (results != m_sessions.end())
164 return results->value.get();
165 return nullptr;
166}
167
168bool RemoteLegacyCDMFactoryProxy::allowsExitUnderMemoryPressure() const
169{
170 return m_sessions.isEmpty();
171}
