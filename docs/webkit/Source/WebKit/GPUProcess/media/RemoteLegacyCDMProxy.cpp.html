39std::unique_ptr<RemoteLegacyCDMProxy> RemoteLegacyCDMProxy::create(WeakPtr<RemoteLegacyCDMFactoryProxy> factory, MediaPlayerIdentifier&& playerId, std::unique_ptr<WebCore::LegacyCDM>&& cdm)
40{
41 return std::unique_ptr<RemoteLegacyCDMProxy>(new RemoteLegacyCDMProxy(WTFMove(factory), WTFMove(playerId), WTFMove(cdm)));
42}

44RemoteLegacyCDMProxy::RemoteLegacyCDMProxy(WeakPtr<RemoteLegacyCDMFactoryProxy>&& factory, MediaPlayerIdentifier&& playerId, std::unique_ptr<WebCore::LegacyCDM>&& cdm)
45 : m_factory(WTFMove(factory))
46 , m_playerId(WTFMove(playerId))
47 , m_cdm(WTFMove(cdm))
48{
49 m_cdm->setClient(this);
50}
51
52RemoteLegacyCDMProxy::~RemoteLegacyCDMProxy() = default;
53
54void RemoteLegacyCDMProxy::supportsMIMEType(const String& mimeType, SupportsMIMETypeCallback&& callback)
55{
56 if (!m_cdm) {
57 callback(false);
58 return;
59 }
60
61 callback(m_cdm->supportsMIMEType(mimeType));
62}
63
64void RemoteLegacyCDMProxy::createSession(const String& keySystem, CreateSessionCallback&& callback)
65{
66 if (!m_cdm || !m_factory) {
67 callback({ });
68 return;
69 }
70
71 auto identifier = RemoteLegacyCDMSessionIdentifier::generate();
72 auto session = RemoteLegacyCDMSessionProxy::create(makeWeakPtr(m_factory.get()), identifier, *m_cdm);
73 m_factory->addSession(identifier, WTFMove(session));
74 callback(WTFMove(identifier));
75}
76
77void RemoteLegacyCDMProxy::setPlayerId(std::optional<MediaPlayerIdentifier>&& playerId)
78{
79 if (!playerId)
80 m_playerId = { };
81 m_playerId = WTFMove(*playerId);
82}
83
84RefPtr<MediaPlayer> RemoteLegacyCDMProxy::cdmMediaPlayer(const LegacyCDM*) const
85{
86 if (!m_playerId || !m_factory)
87 return nullptr;
88
89 auto* gpuConnectionToWebProcess = m_factory->gpuConnectionToWebProcess();
90 if (!gpuConnectionToWebProcess)
91 return nullptr;
92
93 return gpuConnectionToWebProcess->remoteMediaPlayerManagerProxy().mediaPlayer(m_playerId);
94}
