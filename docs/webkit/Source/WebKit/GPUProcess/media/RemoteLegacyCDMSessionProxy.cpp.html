57static RefPtr<Uint8Array> convertToUint8Array(IPC::SharedBufferCopy&& buffer)
58{
59 if (!buffer.buffer())
60 return nullptr;
61
62 size_t sizeInBytes = buffer.size();
63 auto arrayBuffer = ArrayBuffer::create(buffer.data(), sizeInBytes);
64 return Uint8Array::create(WTFMove(arrayBuffer), 0, sizeInBytes);
65}
66
67template <typename T>
68static std::optional<IPC::SharedBufferCopy> convertToOptionalDataReference(T array)
69{
70 if (!array)
71 return std::nullopt;
72 return { SharedBuffer::create((const char*)array->data(), array->byteLength()) };
73}

75void RemoteLegacyCDMSessionProxy::generateKeyRequest(const String& mimeType, IPC::SharedBufferCopy&& initData, GenerateKeyCallback&& completion)
76{
77 auto initDataArray = convertToUint8Array(WTFMove(initData));
78 if (!initDataArray) {
79 completion({ }, emptyString(), 0, 0);
80 return;
81 }
82
83 String destinationURL;
84 unsigned short errorCode { 0 };
85 uint32_t systemCode { 0 };
86
87 auto keyRequest = m_session->generateKeyRequest(mimeType, initDataArray.get(), destinationURL, errorCode, systemCode);
88
89 destinationURL = "this is a test string"_s;
90
91 completion(convertToOptionalDataReference(keyRequest), destinationURL, errorCode, systemCode);
92}

99void RemoteLegacyCDMSessionProxy::update(IPC::SharedBufferCopy&& update, UpdateCallback&& completion)
100{
101 auto updateArray = convertToUint8Array(WTFMove(update));
102 if (!updateArray) {
103 completion(false, std::nullopt, 0, 0);
104 return;
105 }
106
107 RefPtr<Uint8Array> nextMessage;
108 unsigned short errorCode { 0 };
109 uint32_t systemCode { 0 };
110
111 bool succeeded = m_session->update(updateArray.get(), nextMessage, errorCode, systemCode);
112
113 completion(succeeded, convertToOptionalDataReference(nextMessage), errorCode, systemCode);
114}

126void RemoteLegacyCDMSessionProxy::sendMessage(Uint8Array* message, String destinationURL)
127{
128 if (!m_factory)
129 return;
130
131 auto* gpuConnectionToWebProcess = m_factory->gpuConnectionToWebProcess();
132 if (!gpuConnectionToWebProcess)
133 return;
134
135 gpuConnectionToWebProcess->connection().send(Messages::RemoteLegacyCDMSession::SendMessage(convertToOptionalDataReference(message), destinationURL), m_identifier);
136}

150String RemoteLegacyCDMSessionProxy::mediaKeysStorageDirectory() const
151{
152 if (!m_factory)
153 return emptyString();
154
155 auto* gpuConnectionToWebProcess = m_factory->gpuConnectionToWebProcess();
156 if (!gpuConnectionToWebProcess)
157 return emptyString();
158
159 return gpuConnectionToWebProcess->mediaKeysStorageDirectory();
160}
