52class RemoteLegacyCDMSessionProxy
53 : public IPC::MessageReceiver
54 , public WebCore::LegacyCDMSessionClient {
55public:

57 ~RemoteLegacyCDMSessionProxy();
58
59 RemoteLegacyCDMFactoryProxy* factory() const { return m_factory.get(); }
60 WebCore::LegacyCDMSession* session() const { return m_session.get(); }
61
62 void setPlayer(WeakPtr<RemoteMediaPlayerProxy> player) { m_player = WTFMove(player); }
63
64 RefPtr<ArrayBuffer> getCachedKeyForKeyId(const String&);
65
66private:
67 friend class RemoteLegacyCDMFactoryProxy;
68 RemoteLegacyCDMSessionProxy(WeakPtr<RemoteLegacyCDMFactoryProxy>&&, RemoteLegacyCDMSessionIdentifier, WebCore::LegacyCDM&);
69
70 // IPC::MessageReceiver
71 void didReceiveMessage(IPC::Connection&, IPC::Decoder&) final;
72 bool didReceiveSyncMessage(IPC::Connection&, IPC::Decoder&, UniqueRef<IPC::Encoder>&) final;
73
74 // LegacyCDMSessionClient
75 void sendMessage(Uint8Array*, String destinationURL) final;
76 void sendError(MediaKeyErrorCode, uint32_t systemCode) final;
77 String mediaKeysStorageDirectory() const final;
78
79 // Messages
80 using GenerateKeyCallback = CompletionHandler<void(std::optional<IPC::SharedBufferCopy>&&, const String&, unsigned short, uint32_t)>;
81 void generateKeyRequest(const String& mimeType, IPC::SharedBufferCopy&& initData, GenerateKeyCallback&&);
82 void releaseKeys();
83 using UpdateCallback = CompletionHandler<void(bool, std::optional<IPC::SharedBufferCopy>&&, unsigned short, uint32_t)>;
84 void update(IPC::SharedBufferCopy&& update, UpdateCallback&&);
85 using CachedKeyForKeyIDCallback = CompletionHandler<void(std::optional<IPC::SharedBufferCopy>&&)>;
86 void cachedKeyForKeyID(String keyId, CachedKeyForKeyIDCallback&&);
87
88 WeakPtr<RemoteLegacyCDMFactoryProxy> m_factory;
89 RemoteLegacyCDMSessionIdentifier m_identifier;
90 std::unique_ptr<WebCore::LegacyCDMSession> m_session;
91 WeakPtr<RemoteMediaPlayerProxy> m_player;
92};
