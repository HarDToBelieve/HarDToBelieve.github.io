36namespace WebKit {
37
38struct MediaDescriptionInfo {
39 MediaDescriptionInfo(const AtomString& codec, bool isVideo, bool isAudio, bool isText)
40 : m_codec(codec)
41 , m_isVideo(isVideo)
42 , m_isAudio(isAudio)
43 , m_isText(isText)
44 {
45 }
46
47 MediaDescriptionInfo(const WebCore::MediaDescription& description)
48 : m_codec(description.codec())
49 , m_isVideo(description.isVideo())
50 , m_isAudio(description.isAudio())
51 , m_isText(description.isText())
52 {
53 }
54
55 AtomString m_codec;
56 bool m_isVideo { false };
57 bool m_isAudio { false };
58 bool m_isText { false };
59
60 template<class Encoder>
61 void encode(Encoder& encoder) const
62 {
63 encoder << m_codec;
64 encoder << m_isVideo;
65 encoder << m_isAudio;
66 encoder << m_isText;
67 }
68
69 template <class Decoder>
70 static std::optional<MediaDescriptionInfo> decode(Decoder& decoder)
71 {
72 std::optional<String> codec;
73 decoder >> codec;
74 if (!codec)
75 return std::nullopt;
76
77 std::optional<bool> isVideo;
78 decoder >> isVideo;
79 if (!isVideo)
80 return std::nullopt;
81
82 std::optional<bool> isAudio;
83 decoder >> isAudio;
84 if (!isAudio)
85 return std::nullopt;
86
87 std::optional<bool> isText;
88 decoder >> isText;
89 if (!isText)
90 return std::nullopt;
91
92 return {{
93 WTFMove(*codec),
94 *isVideo,
95 *isAudio,
96 *isText
97 }};
98 }
99};
100
101class RemoteMediaDescription : public WebCore::MediaDescription {
102public:
103 static Ref<RemoteMediaDescription> create(const MediaDescriptionInfo& descriptionInfo)
104 {
105 return adoptRef(*new RemoteMediaDescription(descriptionInfo));
106 }
107
108 virtual ~RemoteMediaDescription() = default;
109
110 AtomString codec() const final { return m_codec; }
111 bool isVideo() const final { return m_isVideo; }
112 bool isAudio() const final { return m_isAudio; }
113 bool isText() const final { return m_isText;}
114
115private:
116 RemoteMediaDescription(const MediaDescriptionInfo& descriptionInfo)
117 : m_codec(descriptionInfo.m_codec)
118 , m_isVideo(descriptionInfo.m_isVideo)
119 , m_isAudio(descriptionInfo.m_isAudio)
120 , m_isText(descriptionInfo.m_isText)
121 {
122 }
123
124 AtomString m_codec;
125 bool m_isVideo { false };
126 bool m_isAudio { false };
127 bool m_isText { false };
128};
