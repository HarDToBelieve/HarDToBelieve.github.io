60void RemoteMediaPlayerManagerProxy::createMediaPlayer(MediaPlayerIdentifier identifier, MediaPlayerEnums::MediaEngineIdentifier engineIdentifier, RemoteMediaPlayerProxyConfiguration&& proxyConfiguration)
61{
62 ASSERT(RunLoop::isMain());
63 ASSERT(m_gpuConnectionToWebProcess);
64 ASSERT(!m_proxies.contains(identifier));
65
66 auto proxy = makeUnique<RemoteMediaPlayerProxy>(*this, identifier, m_gpuConnectionToWebProcess->connection(), engineIdentifier, WTFMove(proxyConfiguration));
67 m_proxies.add(identifier, WTFMove(proxy));
68}
69
70void RemoteMediaPlayerManagerProxy::deleteMediaPlayer(MediaPlayerIdentifier identifier)
71{
72 ASSERT(RunLoop::isMain());
73
74 if (auto proxy = m_proxies.take(identifier))
75 proxy->invalidate();
76
77 if (m_gpuConnectionToWebProcess && allowsExitUnderMemoryPressure())
78 m_gpuConnectionToWebProcess->gpuProcess().tryExitIfUnusedAndUnderMemoryPressure();
79}

81void RemoteMediaPlayerManagerProxy::getSupportedTypes(MediaPlayerEnums::MediaEngineIdentifier engineIdentifier, CompletionHandler<void(Vector<String>&&)>&& completionHandler)
82{
83 auto engine = MediaPlayer::mediaEngine(engineIdentifier);
84 if (!engine) {
85 WTFLogAlways("Failed to find media engine.");
86 completionHandler({ });
87 return;
88 }
89
90 HashSet<String, ASCIICaseInsensitiveHash> engineTypes;
91 engine->getSupportedTypes(engineTypes);
92
93 auto result = WTF::map(engineTypes, [] (auto& type) {
94 return type;
95 });
96
97 completionHandler(WTFMove(result));
98}

158void RemoteMediaPlayerManagerProxy::didReceivePlayerMessage(IPC::Connection& connection, IPC::Decoder& decoder)
159{
160 ASSERT(RunLoop::isMain());
161 if (auto* player = m_proxies.get(makeObjectIdentifier<MediaPlayerIdentifierType>(decoder.destinationID())))
162 player->didReceiveMessage(connection, decoder);
163}
164
165bool RemoteMediaPlayerManagerProxy::didReceiveSyncPlayerMessage(IPC::Connection& connection, IPC::Decoder& decoder, UniqueRef<IPC::Encoder>& encoder)
166{
167 ASSERT(RunLoop::isMain());
168 if (auto* player = m_proxies.get(makeObjectIdentifier<MediaPlayerIdentifierType>(decoder.destinationID())))
169 return player->didReceiveSyncMessage(connection, decoder, encoder);
170 return false;
171}
172
173RefPtr<MediaPlayer> RemoteMediaPlayerManagerProxy::mediaPlayer(const MediaPlayerIdentifier& identifier)
174{
175 ASSERT(RunLoop::isMain());
176 auto results = m_proxies.find(identifier);
177 if (results != m_proxies.end())
178 return results->value->mediaPlayer();
179 return nullptr;
180}
181
182bool RemoteMediaPlayerManagerProxy::allowsExitUnderMemoryPressure() const
183{
184 return m_proxies.isEmpty();
185}
