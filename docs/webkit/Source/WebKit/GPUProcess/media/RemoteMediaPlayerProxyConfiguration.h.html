26#pragma once
27
28#if ENABLE(GPU_PROCESS)
29
30#include <WebCore/ContentType.h>
31#include <WebCore/PlatformTextTrack.h>
32#include <WebCore/SecurityOriginData.h>
33#include <wtf/text/WTFString.h>
34
35namespace WebKit {
36
37struct RemoteMediaPlayerProxyConfiguration {
38 String referrer;
39 String userAgent;
40 String sourceApplicationIdentifier;
41 String networkInterfaceName;
42 Vector<WebCore::ContentType> mediaContentTypesRequiringHardwareSupport;
43 Vector<String> preferredAudioCharacteristics;
44#if ENABLE(AVF_CAPTIONS)
45 Vector<WebCore::PlatformTextTrackData> outOfBandTrackData;
46#endif
47 WebCore::SecurityOriginData documentSecurityOrigin;
48 uint64_t logIdentifier { 0 };
49 bool shouldUsePersistentCache { false };
50 bool isVideo { 0 };
51
52 template<class Encoder>
53 void encode(Encoder& encoder) const
54 {
55 encoder << referrer;
56 encoder << userAgent;
57 encoder << sourceApplicationIdentifier;
58 encoder << networkInterfaceName;
59 encoder << mediaContentTypesRequiringHardwareSupport;
60 encoder << preferredAudioCharacteristics;
61#if ENABLE(AVF_CAPTIONS)
62 encoder << outOfBandTrackData;
63#endif
64 encoder << documentSecurityOrigin;
65 encoder << logIdentifier;
66 encoder << shouldUsePersistentCache;
67 encoder << isVideo;
68 }
69
70 template <class Decoder>
71 static std::optional<RemoteMediaPlayerProxyConfiguration> decode(Decoder& decoder)
72 {
73 std::optional<String> referrer;
74 decoder >> referrer;
75 if (!referrer)
76 return std::nullopt;
77
78 std::optional<String> userAgent;
79 decoder >> userAgent;
80 if (!userAgent)
81 return std::nullopt;
82
83 std::optional<String> sourceApplicationIdentifier;
84 decoder >> sourceApplicationIdentifier;
85 if (!sourceApplicationIdentifier)
86 return std::nullopt;
87
88 std::optional<String> networkInterfaceName;
89 decoder >> networkInterfaceName;
90 if (!networkInterfaceName)
91 return std::nullopt;
92
93 std::optional<Vector<WebCore::ContentType>> mediaContentTypesRequiringHardwareSupport;
94 decoder >> mediaContentTypesRequiringHardwareSupport;
95 if (!mediaContentTypesRequiringHardwareSupport)
96 return std::nullopt;
97
98 std::optional<Vector<String>> preferredAudioCharacteristics;
99 decoder >> preferredAudioCharacteristics;
100 if (!preferredAudioCharacteristics)
101 return std::nullopt;
102
103#if ENABLE(AVF_CAPTIONS)
104 std::optional<Vector<WebCore::PlatformTextTrackData>> outOfBandTrackData;
105 decoder >> outOfBandTrackData;
106 if (!outOfBandTrackData)
107 return std::nullopt;
108#endif
109
110 std::optional<WebCore::SecurityOriginData> documentSecurityOrigin;
111 decoder >> documentSecurityOrigin;
112 if (!documentSecurityOrigin)
113 return std::nullopt;
114
115 std::optional<uint64_t> logIdentifier;
116 decoder >> logIdentifier;
117 if (!logIdentifier)
118 return std::nullopt;
119
120 std::optional<bool> shouldUsePersistentCache;
121 decoder >> shouldUsePersistentCache;
122 if (!shouldUsePersistentCache)
123 return std::nullopt;
124
125 std::optional<bool> isVideo;
126 decoder >> isVideo;
127 if (!isVideo)
128 return std::nullopt;
129
130 return {{
131 WTFMove(*referrer),
132 WTFMove(*userAgent),
133 WTFMove(*sourceApplicationIdentifier),
134 WTFMove(*networkInterfaceName),
135 WTFMove(*mediaContentTypesRequiringHardwareSupport),
136 WTFMove(*preferredAudioCharacteristics),
137#if ENABLE(AVF_CAPTIONS)
138 WTFMove(*outOfBandTrackData),
139#endif
140 WTFMove(*documentSecurityOrigin),
141 *logIdentifier,
142 *shouldUsePersistentCache,
143 *isVideo,
144 }};
145 }
146};
