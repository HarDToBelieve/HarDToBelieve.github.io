39Ref<RemoteMediaResource> RemoteMediaResource::create(RemoteMediaResourceManager& remoteMediaResourceManager, RemoteMediaPlayerProxy& remoteMediaPlayerProxy, RemoteMediaResourceIdentifier identifier)
40{
41 return adoptRef(*new RemoteMediaResource(remoteMediaResourceManager, remoteMediaPlayerProxy, identifier));
42}

44RemoteMediaResource::RemoteMediaResource(RemoteMediaResourceManager& remoteMediaResourceManager, RemoteMediaPlayerProxy& remoteMediaPlayerProxy, RemoteMediaResourceIdentifier identifier)
45 : m_remoteMediaResourceManager(makeWeakPtr(remoteMediaResourceManager))
46 , m_remoteMediaPlayerProxy(makeWeakPtr(remoteMediaPlayerProxy))
47 , m_id(identifier)
48{
49}
50
51RemoteMediaResource::~RemoteMediaResource()
52{
53 ASSERT(isMainRunLoop());
54 stop();
55 if (!m_remoteMediaResourceManager)
56 return;
57
58 m_remoteMediaResourceManager->removeMediaResource(m_id);
59}
60
61void RemoteMediaResource::stop()
62{
63 if (m_remoteMediaPlayerProxy)
64 m_remoteMediaPlayerProxy->removeResource(m_id);
65}
66
67bool RemoteMediaResource::didPassAccessControlCheck() const
68{
69 return m_didPassAccessControlCheck;
70}
71
72void RemoteMediaResource::responseReceived(const ResourceResponse& response, bool didPassAccessControlCheck, CompletionHandler<void(ShouldContinuePolicyCheck)>&& completionHandler)
73{
74 if (!m_client)
75 return;
76
77 m_didPassAccessControlCheck = didPassAccessControlCheck;
78 m_client->responseReceived(*this, response, [protectedThis = makeRef(*this), completionHandler = WTFMove(completionHandler)](auto shouldContinue) mutable {
79 ASSERT(isMainRunLoop());
80 if (shouldContinue == ShouldContinuePolicyCheck::No)
81 protectedThis->stop();
82
83 completionHandler(shouldContinue);
84 });
85}

87void RemoteMediaResource::redirectReceived(ResourceRequest&& request, const ResourceResponse& response, CompletionHandler<void(ResourceRequest&&)>&& completionHandler)
88{
89 if (m_client)
90 m_client->redirectReceived(*this, WTFMove(request), response, WTFMove(completionHandler));
91}
