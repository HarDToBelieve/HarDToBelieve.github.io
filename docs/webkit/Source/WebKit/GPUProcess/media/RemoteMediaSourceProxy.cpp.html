44RemoteMediaSourceProxy::RemoteMediaSourceProxy(GPUConnectionToWebProcess& connectionToWebProcess, RemoteMediaSourceIdentifier identifier, bool webMParserEnabled, RemoteMediaPlayerProxy& remoteMediaPlayerProxy)
45 : m_connectionToWebProcess(makeWeakPtr(connectionToWebProcess))
46 , m_identifier(identifier)
47 , m_webMParserEnabled(webMParserEnabled)
48 , m_remoteMediaPlayerProxy(makeWeakPtr(remoteMediaPlayerProxy))
49{
50 m_connectionToWebProcess->messageReceiverMap().addMessageReceiver(Messages::RemoteMediaSourceProxy::messageReceiverName(), m_identifier.toUInt64(), *this);
51}
52
53RemoteMediaSourceProxy::~RemoteMediaSourceProxy()
54{
55 ASSERT(m_connectionToWebProcess);
56
57 m_connectionToWebProcess->messageReceiverMap().removeMessageReceiver(Messages::RemoteMediaSourceProxy::messageReceiverName(), m_identifier.toUInt64());
58}

103void RemoteMediaSourceProxy::addSourceBuffer(const WebCore::ContentType& contentType, AddSourceBufferCallback&& callback)
104{
105 if (!m_remoteMediaPlayerProxy || !m_connectionToWebProcess)
106 return;
107
108 RefPtr<SourceBufferPrivate> sourceBufferPrivate;
109 MediaSourcePrivate::AddStatus status = m_private->addSourceBuffer(contentType, m_webMParserEnabled, sourceBufferPrivate);
110
111 std::optional<RemoteSourceBufferIdentifier> remoteSourceIdentifier;
112 if (status == MediaSourcePrivate::AddStatus::Ok) {
113 auto identifier = RemoteSourceBufferIdentifier::generate();
114 auto remoteSourceBufferProxy = RemoteSourceBufferProxy::create(*m_connectionToWebProcess, identifier, sourceBufferPrivate.releaseNonNull(), *m_remoteMediaPlayerProxy);
115 m_sourceBuffers.append(WTFMove(remoteSourceBufferProxy));
116 remoteSourceIdentifier = identifier;
117 }
118
119 callback(status, remoteSourceIdentifier);
120}
