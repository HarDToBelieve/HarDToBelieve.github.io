46RemoteTextTrackProxy::RemoteTextTrackProxy(GPUConnectionToWebProcess& connectionToWebProcess, TrackPrivateRemoteIdentifier identifier, InbandTextTrackPrivate& trackPrivate, MediaPlayerIdentifier mediaPlayerIdentifier)
47 : m_connectionToWebProcess(makeWeakPtr(connectionToWebProcess))
48 , m_identifier(identifier)
49 , m_trackPrivate(trackPrivate)
50 , m_mediaPlayerIdentifier(mediaPlayerIdentifier)
51{
52 m_trackPrivate->setClient(this);
53 m_connectionToWebProcess->connection().send(Messages::MediaPlayerPrivateRemote::AddRemoteTextTrack(m_identifier, configuration()), m_mediaPlayerIdentifier);
54}

57{
58}

60TextTrackPrivateRemoteConfiguration& RemoteTextTrackProxy::configuration()
61{
62 static NeverDestroyed<TextTrackPrivateRemoteConfiguration> configuration;
63
64 configuration->trackId = m_trackPrivate->id();
65 configuration->label = m_trackPrivate->label();
66 configuration->language = m_trackPrivate->language();
67 configuration->trackIndex = m_trackPrivate->trackIndex();
68 configuration->inBandMetadataTrackDispatchType = m_trackPrivate->inBandMetadataTrackDispatchType();
69 configuration->startTimeVariance = m_trackPrivate->startTimeVariance();
70
71 configuration->cueFormat = m_trackPrivate->cueFormat();
72 configuration->isClosedCaptions = m_trackPrivate->isClosedCaptions();
73 configuration->isSDH = m_trackPrivate->isSDH();
74 configuration->containsOnlyForcedSubtitles = m_trackPrivate->containsOnlyForcedSubtitles();
75 configuration->isMainProgramContent = m_trackPrivate->isMainProgramContent();
76 configuration->isEasyToRead = m_trackPrivate->isEasyToRead();
77 configuration->isDefault = m_trackPrivate->isDefault();
78 configuration->kind = m_trackPrivate->kind();
79
80 return configuration.get();
81}

83void RemoteTextTrackProxy::configurationChanged()
84{
85 if (!m_connectionToWebProcess)
86 return;
87
88 m_connectionToWebProcess->connection().send(Messages::MediaPlayerPrivateRemote::RemoteTextTrackConfigurationChanged(m_identifier, configuration()), m_mediaPlayerIdentifier);
89}
