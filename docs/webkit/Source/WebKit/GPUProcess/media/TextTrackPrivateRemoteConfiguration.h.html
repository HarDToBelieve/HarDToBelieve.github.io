34namespace WebKit {
35
36struct TextTrackPrivateRemoteConfiguration {
37 AtomString trackId;
38 AtomString label;
39 AtomString language;
40 AtomString inBandMetadataTrackDispatchType;
41 MediaTime startTimeVariance { MediaTime::zeroTime() };
42 int trackIndex;
43
44 WebCore::InbandTextTrackPrivate::CueFormat cueFormat { WebCore::InbandTextTrackPrivate::CueFormat::Generic };
45 WebCore::InbandTextTrackPrivate::Kind kind { WebCore::InbandTextTrackPrivate::Kind::None };
46
47 bool isClosedCaptions { false };
48 bool isSDH { false };
49 bool containsOnlyForcedSubtitles { false };
50 bool isMainProgramContent { true };
51 bool isEasyToRead { false };
52 bool isDefault { false };
53
54 template<class Encoder>
55 void encode(Encoder& encoder) const
56 {
57 encoder << trackId;
58 encoder << label;
59 encoder << language;
60 encoder << inBandMetadataTrackDispatchType;
61 encoder << startTimeVariance;
62 encoder << trackIndex;
63 encoder << cueFormat;
64 encoder << kind;
65 encoder << isClosedCaptions;
66 encoder << isSDH;
67 encoder << containsOnlyForcedSubtitles;
68 encoder << isMainProgramContent;
69 encoder << isEasyToRead;
70 encoder << isDefault;
71 }
72
73 template <class Decoder>
74 static std::optional<TextTrackPrivateRemoteConfiguration> decode(Decoder& decoder)
75 {
76 std::optional<AtomString> trackId;
77 decoder >> trackId;
78 if (!trackId)
79 return std::nullopt;
80
81 std::optional<AtomString> label;
82 decoder >> label;
83 if (!label)
84 return std::nullopt;
85
86 std::optional<AtomString> language;
87 decoder >> language;
88 if (!language)
89 return std::nullopt;
90
91 std::optional<AtomString> inBandMetadataTrackDispatchType;
92 decoder >> inBandMetadataTrackDispatchType;
93 if (!inBandMetadataTrackDispatchType)
94 return std::nullopt;
95
96 std::optional<MediaTime> startTimeVariance;
97 decoder >> startTimeVariance;
98 if (!startTimeVariance)
99 return std::nullopt;
100
101 std::optional<int> trackIndex;
102 decoder >> trackIndex;
103 if (!trackIndex)
104 return std::nullopt;
105
106 std::optional<WebCore::InbandTextTrackPrivate::CueFormat> cueFormat;
107 decoder >> cueFormat;
108 if (!cueFormat)
109 return std::nullopt;
110
111 std::optional<WebCore::InbandTextTrackPrivate::Kind> kind;
112 decoder >> kind;
113 if (!kind)
114 return std::nullopt;
115
116 std::optional<bool> isClosedCaptions;
117 decoder >> isClosedCaptions;
118 if (!isClosedCaptions)
119 return std::nullopt;
120
121 std::optional<bool> isSDH;
122 decoder >> isSDH;
123 if (!isSDH)
124 return std::nullopt;
125
126 std::optional<bool> containsOnlyForcedSubtitles;
127 decoder >> containsOnlyForcedSubtitles;
128 if (!containsOnlyForcedSubtitles)
129 return std::nullopt;
130
131 std::optional<bool> isMainProgramContent;
132 decoder >> isMainProgramContent;
133 if (!isMainProgramContent)
134 return std::nullopt;
135
136 std::optional<bool> isEasyToRead;
137 decoder >> isEasyToRead;
138 if (!isEasyToRead)
139 return std::nullopt;
140
141 std::optional<bool> isDefault;
142 decoder >> isDefault;
143 if (!isDefault)
144 return std::nullopt;
145
146 return {{
147 WTFMove(*trackId),
148 WTFMove(*label),
149 WTFMove(*language),
150 WTFMove(*inBandMetadataTrackDispatchType),
151 WTFMove(*startTimeVariance),
152 *trackIndex,
153 *cueFormat,
154 *kind,
155 *isClosedCaptions,
156 *isSDH,
157 *containsOnlyForcedSubtitles,
158 *isMainProgramContent,
159 *isEasyToRead,
160 *isDefault,
161 }};
162 }
163};
