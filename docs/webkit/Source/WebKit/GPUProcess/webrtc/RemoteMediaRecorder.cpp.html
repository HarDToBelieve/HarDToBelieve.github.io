45std::unique_ptr<RemoteMediaRecorder> RemoteMediaRecorder::create(GPUConnectionToWebProcess& gpuConnectionToWebProcess, MediaRecorderIdentifier identifier, bool recordAudio, bool recordVideo, const MediaRecorderPrivateOptions& options)
46{
47 auto writer = MediaRecorderPrivateWriter::create(recordAudio, recordVideo, options);
48 if (!writer)
49 return nullptr;
50 return std::unique_ptr<RemoteMediaRecorder>(new RemoteMediaRecorder { gpuConnectionToWebProcess, identifier, writer.releaseNonNull(), recordAudio });
51}

53RemoteMediaRecorder::RemoteMediaRecorder(GPUConnectionToWebProcess& gpuConnectionToWebProcess, MediaRecorderIdentifier identifier, Ref<MediaRecorderPrivateWriter>&& writer, bool recordAudio)
54 : m_gpuConnectionToWebProcess(gpuConnectionToWebProcess)
55 , m_identifier(identifier)
56 , m_writer(WTFMove(writer))
57{
58 if (recordAudio)
59 m_ringBuffer = makeUnique<CARingBuffer>();
60}

66void RemoteMediaRecorder::audioSamplesStorageChanged(const SharedMemory::IPCHandle& ipcHandle, const WebCore::CAAudioStreamDescription& description, uint64_t numberOfFrames)
67{
68 MESSAGE_CHECK(m_ringBuffer);
69
70 m_description = description;
71
72 m_ringBuffer = CARingBuffer::adoptStorage(makeUniqueRef<ReadOnlySharedRingBufferStorage>(ipcHandle.handle), description, numberOfFrames).moveToUniquePtr();
73 m_audioBufferList = makeUnique<WebAudioBufferList>(m_description);
74}
75
76void RemoteMediaRecorder::audioSamplesAvailable(MediaTime time, uint64_t numberOfFrames)
77{
78 MESSAGE_CHECK(m_ringBuffer);
79 MESSAGE_CHECK(m_audioBufferList);
80 MESSAGE_CHECK(WebAudioBufferList::isSupportedDescription(m_description, numberOfFrames));
81
82 m_audioBufferList->setSampleCount(numberOfFrames);
83 m_ringBuffer->fetch(m_audioBufferList->list(), numberOfFrames, time.timeValue());
84
85 m_writer->appendAudioSampleBuffer(*m_audioBufferList, m_description, time, numberOfFrames);
86}
87
88void RemoteMediaRecorder::videoSampleAvailable(WebCore::RemoteVideoSample&& remoteSample)
89{
90 if (!m_imageTransferSession || m_imageTransferSession->pixelFormat() != remoteSample.videoFormat())
91 m_imageTransferSession = ImageTransferSessionVT::create(remoteSample.videoFormat());
92
93 if (!m_imageTransferSession) {
94 ASSERT_NOT_REACHED();
95 return;
96 }
97
98 auto sampleBuffer = m_imageTransferSession->createMediaSample(remoteSample);
99 if (!sampleBuffer) {
100 ASSERT_NOT_REACHED();
101 return;
102 }
103
104 m_writer->appendVideoSampleBuffer(*sampleBuffer);
105}

107void RemoteMediaRecorder::fetchData(CompletionHandler<void(IPC::DataReference&&, double)>&& completionHandler)
108{
109 m_writer->fetchData([completionHandler = WTFMove(completionHandler)](auto&& data, auto timeCode) mutable {
110 auto* pointer = data ? data->data() : nullptr;
111 completionHandler(IPC::DataReference { pointer, data ? data->size() : 0 }, timeCode);
112 });
113}
