40std::unique_ptr<RemoteSampleBufferDisplayLayer> RemoteSampleBufferDisplayLayer::create(SampleBufferDisplayLayerIdentifier identifier, Ref<IPC::Connection>&& connection)
41{
42 auto layer = std::unique_ptr<RemoteSampleBufferDisplayLayer>(new RemoteSampleBufferDisplayLayer(identifier, WTFMove(connection)));
43 return layer->m_sampleBufferDisplayLayer ? WTFMove(layer) : nullptr;
44}

46RemoteSampleBufferDisplayLayer::RemoteSampleBufferDisplayLayer(SampleBufferDisplayLayerIdentifier identifier, Ref<IPC::Connection>&& connection)
47 : m_identifier(identifier)
48 , m_connection(WTFMove(connection))
49 , m_sampleBufferDisplayLayer(LocalSampleBufferDisplayLayer::create(*this))

52}

54void RemoteSampleBufferDisplayLayer::initialize(bool hideRootLayer, IntSize size, LayerInitializationCallback&& callback)
55{
56 m_sampleBufferDisplayLayer->initialize(hideRootLayer, size, [this, weakThis = makeWeakPtr(this), callback = WTFMove(callback)](bool didSucceed) mutable {
57 if (!weakThis || !didSucceed)
58 return callback({ });
59 m_layerHostingContext = LayerHostingContext::createForExternalHostingProcess();
60 m_layerHostingContext->setRootLayer(m_sampleBufferDisplayLayer->rootLayer());
61 callback(m_layerHostingContext->contextID());
62 });
63}

109void RemoteSampleBufferDisplayLayer::enqueueSample(WebCore::RemoteVideoSample&& remoteSample)
110{
111 if (!m_imageTransferSession || m_imageTransferSession->pixelFormat() != remoteSample.videoFormat())
112 m_imageTransferSession = ImageTransferSessionVT::create(remoteSample.videoFormat());
113
114 ASSERT(m_imageTransferSession);
115 if (!m_imageTransferSession)
116 return;
117
118 auto sample = m_imageTransferSession->createMediaSample(remoteSample);
119
120 ASSERT(sample);
121 if (!sample)
122 return;
123
124 MediaSampleAVFObjC::setAsDisplayImmediately(*sample);
125 m_sampleBufferDisplayLayer->enqueueSample(*sample);
126}
