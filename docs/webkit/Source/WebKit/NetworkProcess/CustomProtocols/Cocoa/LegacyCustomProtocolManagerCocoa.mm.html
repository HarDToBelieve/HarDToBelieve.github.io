41static RefPtr<NetworkProcess>& firstNetworkProcess()
42{
43 static NeverDestroyed<RefPtr<NetworkProcess>> networkProcess;
44 return networkProcess.get();
45}
46
47void LegacyCustomProtocolManager::networkProcessCreated(NetworkProcess& networkProcess)
48{
49 auto hasRegisteredSchemes = [] (auto* legacyCustomProtocolManager) {
50 if (!legacyCustomProtocolManager)
51 return false;
52 Locker locker { legacyCustomProtocolManager->m_registeredSchemesLock };
53 return !legacyCustomProtocolManager->m_registeredSchemes.isEmpty();
54 };
55
56 RELEASE_ASSERT(!firstNetworkProcess() || !hasRegisteredSchemes(firstNetworkProcess()->supplement<LegacyCustomProtocolManager>()));
57 firstNetworkProcess() = &networkProcess;
58}

60@interface WKCustomProtocol : NSURLProtocol {
61@private
62 LegacyCustomProtocolID _customProtocolID;
63 RetainPtr<CFRunLoopRef> _initializationRunLoop;
64}
65@property (nonatomic, readonly) LegacyCustomProtocolID customProtocolID;
66@property (nonatomic, readonly) CFRunLoopRef initializationRunLoop;
67@end
68
69@implementation WKCustomProtocol
70
71@synthesize customProtocolID = _customProtocolID;
72
73+ (BOOL)canInitWithRequest:(NSURLRequest *)request
74{
75 if (auto* customProtocolManager = firstNetworkProcess()->supplement<LegacyCustomProtocolManager>())
76 return customProtocolManager->supportsScheme([[[request URL] scheme] lowercaseString]);
77 return NO;
78}
79
80+ (NSURLRequest *)canonicalRequestForRequest:(NSURLRequest *)request
81{
82 return request;
83}
84
85+ (BOOL)requestIsCacheEquivalent:(NSURLRequest *)a toRequest:(NSURLRequest *)b
86{
87 return NO;
88}
89
90- (id)initWithRequest:(NSURLRequest *)request cachedResponse:(NSCachedURLResponse *)cachedResponse client:(id<NSURLProtocolClient>)client
91{
92 self = [super initWithRequest:request cachedResponse:cachedResponse client:client];
93 if (!self)
94 return nil;
95
96 if (auto* customProtocolManager = firstNetworkProcess()->supplement<LegacyCustomProtocolManager>())
97 _customProtocolID = customProtocolManager->addCustomProtocol(self);
98 _initializationRunLoop = CFRunLoopGetCurrent();
99
100 return self;
101}
102
103- (CFRunLoopRef)initializationRunLoop
104{
105 return _initializationRunLoop.get();
106}
107
108- (void)startLoading
109{
110 if (auto* customProtocolManager = firstNetworkProcess()->supplement<LegacyCustomProtocolManager>())
111 customProtocolManager->startLoading(self.customProtocolID, [self request]);
112}
113
114- (void)stopLoading
115{
116 if (auto* customProtocolManager = firstNetworkProcess()->supplement<LegacyCustomProtocolManager>()) {
117 customProtocolManager->stopLoading(self.customProtocolID);
118 customProtocolManager->removeCustomProtocol(self.customProtocolID);
119 }
120}
121
122@end

135void LegacyCustomProtocolManager::registerScheme(const String& scheme)
136{
137 ASSERT(!scheme.isNull());
138 Locker locker { m_registeredSchemesLock };
139 m_registeredSchemes.add(scheme);
140}
141
142void LegacyCustomProtocolManager::unregisterScheme(const String& scheme)
143{
144 ASSERT(!scheme.isNull());
145 Locker locker { m_registeredSchemesLock };
146 m_registeredSchemes.remove(scheme);
147}
148
149bool LegacyCustomProtocolManager::supportsScheme(const String& scheme)
150{
151 if (scheme.isNull())
152 return false;
153
154 Locker locker { m_registeredSchemesLock };
155 return m_registeredSchemes.contains(scheme);
156}

158static inline void dispatchOnInitializationRunLoop(WKCustomProtocol* protocol, void (^block)())
159{
160 CFRunLoopRef runloop = protocol.initializationRunLoop;
161 CFRunLoopPerformBlock(runloop, kCFRunLoopDefaultMode, block);
162 CFRunLoopWakeUp(runloop);
163}

165void LegacyCustomProtocolManager::didFailWithError(LegacyCustomProtocolID customProtocolID, const WebCore::ResourceError& error)
166{
167 RetainPtr<WKCustomProtocol> protocol = protocolForID(customProtocolID);
168 if (!protocol)
169 return;
170
171 RetainPtr<NSError> nsError = error.nsError();
172
173 dispatchOnInitializationRunLoop(protocol.get(), ^ {
174 [[protocol client] URLProtocol:protocol.get() didFailWithError:nsError.get()];
175 });
176
177 removeCustomProtocol(customProtocolID);
178}

180void LegacyCustomProtocolManager::didLoadData(LegacyCustomProtocolID customProtocolID, const IPC::DataReference& data)
181{
182 RetainPtr<WKCustomProtocol> protocol = protocolForID(customProtocolID);
183 if (!protocol)
184 return;
185
186 RetainPtr<NSData> nsData = adoptNS([[NSData alloc] initWithBytes:data.data() length:data.size()]);
187
188 dispatchOnInitializationRunLoop(protocol.get(), ^ {
189 [[protocol client] URLProtocol:protocol.get() didLoadData:nsData.get()];
190 });
191}

193void LegacyCustomProtocolManager::didReceiveResponse(LegacyCustomProtocolID customProtocolID, const WebCore::ResourceResponse& response, uint32_t cacheStoragePolicy)
194{
195 RetainPtr<WKCustomProtocol> protocol = protocolForID(customProtocolID);
196 if (!protocol)
197 return;
198
199 RetainPtr<NSURLResponse> nsResponse = response.nsURLResponse();
200
201 dispatchOnInitializationRunLoop(protocol.get(), ^ {
202 [[protocol client] URLProtocol:protocol.get() didReceiveResponse:nsResponse.get() cacheStoragePolicy:static_cast<NSURLCacheStoragePolicy>(cacheStoragePolicy)];
203 });
204}

206void LegacyCustomProtocolManager::didFinishLoading(LegacyCustomProtocolID customProtocolID)
207{
208 RetainPtr<WKCustomProtocol> protocol = protocolForID(customProtocolID);
209 if (!protocol)
210 return;
211
212 dispatchOnInitializationRunLoop(protocol.get(), ^ {
213 [[protocol client] URLProtocolDidFinishLoading:protocol.get()];
214 });
215
216 removeCustomProtocol(customProtocolID);
217}

219void LegacyCustomProtocolManager::wasRedirectedToRequest(LegacyCustomProtocolID customProtocolID, const WebCore::ResourceRequest& request, const WebCore::ResourceResponse& redirectResponse)
220{
221 RetainPtr<WKCustomProtocol> protocol = protocolForID(customProtocolID);
222 if (!protocol)
223 return;
224
225 RetainPtr<NSURLRequest> nsRequest = request.nsURLRequest(WebCore::HTTPBodyUpdatePolicy::DoNotUpdateHTTPBody);
226 RetainPtr<NSURLResponse> nsRedirectResponse = redirectResponse.nsURLResponse();
227
228 dispatchOnInitializationRunLoop(protocol.get(), [protocol, nsRequest, nsRedirectResponse]() {
229 [[protocol client] URLProtocol:protocol.get() wasRedirectedToRequest:nsRequest.get() redirectResponse:nsRedirectResponse.get()];
230 });
231}

233RetainPtr<WKCustomProtocol> LegacyCustomProtocolManager::protocolForID(LegacyCustomProtocolID customProtocolID)
234{
235 Locker locker { m_customProtocolMapLock };
236
237 CustomProtocolMap::const_iterator it = m_customProtocolMap.find(customProtocolID);
238 if (it == m_customProtocolMap.end())
239 return nil;
240
241 ASSERT(it->value);
242 return it->value;
243}
