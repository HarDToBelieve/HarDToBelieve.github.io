44namespace WebCore {
45class ResourceError;
46class ResourceRequest;
47class ResourceResponse;

50namespace WebKit {
51
52class NetworkProcess;
53struct NetworkProcessCreationParameters;
54
55class LegacyCustomProtocolManager : public NetworkProcessSupplement, public IPC::MessageReceiver {
56 WTF_MAKE_FAST_ALLOCATED;
57 WTF_MAKE_NONCOPYABLE(LegacyCustomProtocolManager);
58public:
59 explicit LegacyCustomProtocolManager(NetworkProcess&);
60
61 static const char* supplementName();
62
63 void registerScheme(const String&);
64 void unregisterScheme(const String&);
65 bool supportsScheme(const String&);
66
67#if PLATFORM(COCOA)
68 typedef RetainPtr<WKCustomProtocol> CustomProtocol;
69#endif
70
71 LegacyCustomProtocolID addCustomProtocol(CustomProtocol&&);
72 void removeCustomProtocol(LegacyCustomProtocolID);
73 void startLoading(LegacyCustomProtocolID, const WebCore::ResourceRequest&);
74 void stopLoading(LegacyCustomProtocolID);
75
76#if PLATFORM(COCOA)
77 void registerProtocolClass(NSURLSessionConfiguration*);
78 static void networkProcessCreated(NetworkProcess&);
79#endif
80
81private:
82 // NetworkProcessSupplement
83 void initialize(const NetworkProcessCreationParameters&) override;
84
85 // IPC::MessageReceiver
86 void didReceiveMessage(IPC::Connection&, IPC::Decoder&) override;
87
88 void didFailWithError(LegacyCustomProtocolID, const WebCore::ResourceError&);
89 void didLoadData(LegacyCustomProtocolID, const IPC::DataReference&);
90 void didReceiveResponse(LegacyCustomProtocolID, const WebCore::ResourceResponse&, uint32_t cacheStoragePolicy);
91 void didFinishLoading(LegacyCustomProtocolID);
92 void wasRedirectedToRequest(LegacyCustomProtocolID, const WebCore::ResourceRequest&, const WebCore::ResourceResponse& redirectResponse);
93
94 void registerProtocolClass();
95
96 NetworkProcess& m_networkProcess;
97
98 typedef HashMap<LegacyCustomProtocolID, CustomProtocol> CustomProtocolMap;
99 CustomProtocolMap m_customProtocolMap WTF_GUARDED_BY_LOCK(m_customProtocolMapLock);
100 Lock m_customProtocolMapLock;
101
102#if PLATFORM(COCOA)
103 HashSet<String, ASCIICaseInsensitiveHash> m_registeredSchemes WTF_GUARDED_BY_LOCK(m_registeredSchemesLock);
104 Lock m_registeredSchemesLock;
105
106 // WKCustomProtocol objects can be removed from the m_customProtocolMap from multiple threads.
107 // We return a RetainPtr here because it is unsafe to return a raw pointer since the object might immediately be destroyed from a different thread.
108 RetainPtr<WKCustomProtocol> protocolForID(LegacyCustomProtocolID);
109#endif
110};
