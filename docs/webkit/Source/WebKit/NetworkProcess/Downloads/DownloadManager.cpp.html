46void DownloadManager::startDownload(PAL::SessionID sessionID, DownloadID downloadID, const ResourceRequest& request, std::optional<NavigatingToAppBoundDomain> isNavigatingToAppBoundDomain, const String& suggestedName)
47{
48 auto* networkSession = client().networkSession(sessionID);
49 if (!networkSession)
50 return;
51
52 NetworkLoadParameters parameters;
53 parameters.request = request;
54 parameters.clientCredentialPolicy = ClientCredentialPolicy::MayAskClientForCredentials;
55 parameters.isNavigatingToAppBoundDomain = isNavigatingToAppBoundDomain;
56 if (request.url().protocolIsBlob())
57 parameters.blobFileReferences = client().networkSession(sessionID)->blobRegistry().filesInBlob(request.url());
58 parameters.storedCredentialsPolicy = sessionID.isEphemeral() ? StoredCredentialsPolicy::DoNotUse : StoredCredentialsPolicy::Use;
59
60 m_pendingDownloads.add(downloadID, makeUnique<PendingDownload>(m_client.parentProcessConnectionForDownloads(), WTFMove(parameters), downloadID, *networkSession, &client().networkSession(sessionID)->blobRegistry(), suggestedName));
61}

63void DownloadManager::dataTaskBecameDownloadTask(DownloadID downloadID, std::unique_ptr<Download>&& download)
64{
65 ASSERT(m_pendingDownloads.contains(downloadID));
66 if (auto pendingDownload = m_pendingDownloads.take(downloadID)) {
67#if PLATFORM(COCOA)
68 pendingDownload->didBecomeDownload(download);
69#endif
70 }
71 ASSERT(!m_downloads.contains(downloadID));
72 m_downloadsAfterDestinationDecided.remove(downloadID);
73 m_downloads.add(downloadID, WTFMove(download));
74}

76void DownloadManager::continueWillSendRequest(DownloadID downloadID, WebCore::ResourceRequest&& request)
77{
78 auto* pendingDownload = m_pendingDownloads.get(downloadID);
79 ASSERT(pendingDownload);
80 if (pendingDownload)
81 pendingDownload->continueWillSendRequest(WTFMove(request));
82}

84void DownloadManager::convertNetworkLoadToDownload(DownloadID downloadID, std::unique_ptr<NetworkLoad>&& networkLoad, ResponseCompletionHandler&& completionHandler, Vector<RefPtr<WebCore::BlobDataFileReference>>&& blobFileReferences, const ResourceRequest& request, const ResourceResponse& response)

96void DownloadManager::resumeDownload(PAL::SessionID sessionID, DownloadID downloadID, const IPC::DataReference& resumeData, const String& path, SandboxExtension::Handle&& sandboxExtensionHandle, CallDownloadDidStart callDownloadDidStart)
97{
98#if !PLATFORM(COCOA)
99 notImplemented();
100#else
101 auto* networkSession = m_client.networkSession(sessionID);
102 if (!networkSession)
103 return;
104 auto download = makeUnique<Download>(*this, downloadID, nullptr, *networkSession);
105
106 download->resume(resumeData, path, WTFMove(sandboxExtensionHandle));
107
108 // For compatibility with the legacy download API, only send DidStart if we're using the new API.
109 if (callDownloadDidStart == CallDownloadDidStart::Yes)
110 download->send(Messages::DownloadProxy::DidStart({ }, { }));
111
112 ASSERT(!m_downloads.contains(downloadID));
113 m_downloads.add(downloadID, WTFMove(download));
114#endif
115}

117void DownloadManager::cancelDownload(DownloadID downloadID, CompletionHandler<void(const IPC::DataReference&)>&& completionHandler)
118{
119 if (auto* download = m_downloads.get(downloadID)) {
120 ASSERT(!m_pendingDownloads.contains(downloadID));
121 download->cancel(WTFMove(completionHandler), Download::IgnoreDidFailCallback::Yes);
122 return;
123 }
124 if (auto pendingDownload = m_pendingDownloads.take(downloadID)) {
125 pendingDownload->cancel(WTFMove(completionHandler));
126 return;
127 }
128 ASSERT_NOT_REACHED();
129 completionHandler({ });
130}

133void DownloadManager::publishDownloadProgress(DownloadID downloadID, const URL& url, SandboxExtension::Handle&& sandboxExtensionHandle)
134{
135 if (auto* download = m_downloads.get(downloadID))
136 download->publishProgress(url, WTFMove(sandboxExtensionHandle));
137 else if (auto* pendingDownload = m_pendingDownloads.get(downloadID))
138 pendingDownload->publishProgress(url, WTFMove(sandboxExtensionHandle));
139}
