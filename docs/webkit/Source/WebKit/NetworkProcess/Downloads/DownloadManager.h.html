65class DownloadManager {
66 WTF_MAKE_NONCOPYABLE(DownloadManager);
67
68public:
69 class Client {
70 public:
71 virtual ~Client() { }
72
73 virtual void didCreateDownload() = 0;
74 virtual void didDestroyDownload() = 0;
75 virtual IPC::Connection* downloadProxyConnection() = 0;
76 virtual IPC::Connection* parentProcessConnectionForDownloads() = 0;
77 virtual AuthenticationManager& downloadsAuthenticationManager() = 0;
78 virtual NetworkSession* networkSession(PAL::SessionID) const = 0;
79 virtual void ref() const = 0;
80 virtual void deref() const = 0;
81 };
82
83 explicit DownloadManager(Client&);
84
85 void startDownload(PAL::SessionID, DownloadID, const WebCore::ResourceRequest&, std::optional<NavigatingToAppBoundDomain>, const String& suggestedName = { });
86 void dataTaskBecameDownloadTask(DownloadID, std::unique_ptr<Download>&&);
87 void continueWillSendRequest(DownloadID, WebCore::ResourceRequest&&);
88 void convertNetworkLoadToDownload(DownloadID, std::unique_ptr<NetworkLoad>&&, ResponseCompletionHandler&&, Vector<RefPtr<WebCore::BlobDataFileReference>>&&, const WebCore::ResourceRequest&, const WebCore::ResourceResponse&);
89 void downloadDestinationDecided(DownloadID, Ref<NetworkDataTask>&&);
90
91 void resumeDownload(PAL::SessionID, DownloadID, const IPC::DataReference& resumeData, const String& path, SandboxExtension::Handle&&, CallDownloadDidStart);
92
93 void cancelDownload(DownloadID, CompletionHandler<void(const IPC::DataReference&)>&&);
94#if PLATFORM(COCOA)
95 void publishDownloadProgress(DownloadID, const URL&, SandboxExtension::Handle&&);
96#endif
97
98 Download* download(DownloadID downloadID) { return m_downloads.get(downloadID); }
99
100 void downloadFinished(Download&);
101 bool isDownloading() const { return !m_downloads.isEmpty(); }
102
103 void applicationDidEnterBackground();
104 void applicationWillEnterForeground();
105
106 void didCreateDownload();
107 void didDestroyDownload();
108
109 IPC::Connection* downloadProxyConnection();
110 AuthenticationManager& downloadsAuthenticationManager();
111
112 Client& client() { return m_client; }
113
114private:
115 Client& m_client;
116 HashMap<DownloadID, std::unique_ptr<PendingDownload>> m_pendingDownloads;
117 HashMap<DownloadID, RefPtr<NetworkDataTask>> m_downloadsAfterDestinationDecided;
118 DownloadMap m_downloads;
119};
