36constexpr uint64_t operator"" _kbps(unsigned long long kilobytesPerSecond)
37{
38 return kilobytesPerSecond * 1024;
39}
40
41struct ThroughputInterval {
42 Seconds time;
43 uint64_t bytesPerSecond;
44};
45
46static const ThroughputInterval throughputIntervals[] = {
47 { 1_min, 1_kbps },
48 { 5_min, 2_kbps },
49 { 10_min, 4_kbps },
50 { 15_min, 8_kbps },
51 { 20_min, 16_kbps },
52 { 25_min, 32_kbps },
53 { 30_min, 64_kbps },
54 { 45_min, 96_kbps },
55 { 60_min, 128_kbps }
56};
57
58static Seconds timeUntilNextInterval(size_t currentInterval)
59{
60 RELEASE_ASSERT(currentInterval + 1 < WTF_ARRAY_LENGTH(throughputIntervals));
61 return throughputIntervals[currentInterval + 1].time - throughputIntervals[currentInterval].time;
62}

64DownloadMonitor::DownloadMonitor(Download& download)
65 : m_download(download)
66{
67}
68
69double DownloadMonitor::measuredThroughputRate() const
70{
71 uint64_t bytes { 0 };
72 for (const auto& timestamp : m_timestamps)
73 bytes += timestamp.bytesReceived;
74 if (!bytes)
75 return 0;
76 ASSERT(!m_timestamps.isEmpty());
77 Seconds timeDifference = m_timestamps.last().time.secondsSinceEpoch() - m_timestamps.first().time.secondsSinceEpoch();
78 double seconds = timeDifference.seconds();
79 if (!seconds)
80 return 0;
81 return bytes / seconds;
82}
83
84void DownloadMonitor::downloadReceivedBytes(uint64_t bytesReceived)
85{
86 if (m_timestamps.size() > timestampCapacity - 1) {
87 ASSERT(m_timestamps.size() == timestampCapacity);
88 m_timestamps.removeFirst();
89 }
90 m_timestamps.append({ MonotonicTime::now(), bytesReceived });
91}
92
93void DownloadMonitor::applicationWillEnterForeground()
94{
95 DOWNLOAD_MONITOR_RELEASE_LOG("applicationWillEnterForeground (id = %" PRIu64 ")", m_download.downloadID().toUInt64());
96 m_timer.stop();
97 m_interval = 0;
98}
99
100void DownloadMonitor::applicationDidEnterBackground()
101{
102 DOWNLOAD_MONITOR_RELEASE_LOG("applicationDidEnterBackground (id = %" PRIu64 ")", m_download.downloadID().toUInt64());
103 ASSERT(!m_timer.isActive());
104 ASSERT(!m_interval);
105 m_timer.startOneShot(throughputIntervals[0].time / testSpeedMultiplier());
106}
107
108uint32_t DownloadMonitor::testSpeedMultiplier() const
109{
110 return m_download.testSpeedMultiplier();
111}
112
113void DownloadMonitor::timerFired()
114{
115 downloadReceivedBytes(0);
116
117 RELEASE_ASSERT(m_interval < WTF_ARRAY_LENGTH(throughputIntervals));
118 if (measuredThroughputRate() < throughputIntervals[m_interval].bytesPerSecond) {
119 DOWNLOAD_MONITOR_RELEASE_LOG("timerFired: cancelling download (id = %" PRIu64 ")", m_download.downloadID().toUInt64());
120 m_download.cancel([](auto&) { }, Download::IgnoreDidFailCallback::No);
121 } else if (m_interval + 1 < WTF_ARRAY_LENGTH(throughputIntervals)) {
122 DOWNLOAD_MONITOR_RELEASE_LOG("timerFired: sufficient throughput rate (id = %" PRIu64 ")", m_download.downloadID().toUInt64());
123 m_timer.startOneShot(timeUntilNextInterval(m_interval++) / testSpeedMultiplier());
124 } else
125 DOWNLOAD_MONITOR_RELEASE_LOG("timerFired: Download reached threshold to not be terminated (id = %" PRIu64 ")", m_download.downloadID().toUInt64());
126}
