39PendingDownload::PendingDownload(IPC::Connection* parentProcessConnection, NetworkLoadParameters&& parameters, DownloadID downloadID, NetworkSession& networkSession, WebCore::BlobRegistryImpl* blobRegistry, const String& suggestedName)
40 : m_networkLoad(makeUnique<NetworkLoad>(*this, blobRegistry, WTFMove(parameters), networkSession))
41 , m_parentProcessConnection(parentProcessConnection)
42{
43 m_networkLoad->start();
44 m_isAllowedToAskUserForCredentials = parameters.clientCredentialPolicy == ClientCredentialPolicy::MayAskClientForCredentials;
45
46 m_networkLoad->setPendingDownloadID(downloadID);
47 m_networkLoad->setPendingDownload(*this);
48 m_networkLoad->setSuggestedFilename(suggestedName);
49
50 send(Messages::DownloadProxy::DidStart(m_networkLoad->currentRequest(), suggestedName));
51}
52
53PendingDownload::PendingDownload(IPC::Connection* parentProcessConnection, std::unique_ptr<NetworkLoad>&& networkLoad, ResponseCompletionHandler&& completionHandler, DownloadID downloadID, const ResourceRequest& request, const ResourceResponse& response)
54 : m_networkLoad(WTFMove(networkLoad))
55 , m_parentProcessConnection(parentProcessConnection)
56{
57 m_isAllowedToAskUserForCredentials = m_networkLoad->isAllowedToAskUserForCredentials();
58
59 m_networkLoad->setPendingDownloadID(downloadID);
60 send(Messages::DownloadProxy::DidStart(request, String()));
61
62 m_networkLoad->convertTaskToDownload(*this, request, response, WTFMove(completionHandler));
63}
64
65void PendingDownload::willSendRedirectedRequest(WebCore::ResourceRequest&&, WebCore::ResourceRequest&& redirectRequest, WebCore::ResourceResponse&& redirectResponse)
66{
67 send(Messages::DownloadProxy::WillSendRequest(WTFMove(redirectRequest), WTFMove(redirectResponse)));
68};
69
70void PendingDownload::continueWillSendRequest(WebCore::ResourceRequest&& newRequest)
71{
72 m_networkLoad->continueWillSendRequest(WTFMove(newRequest));
73}
74
75void PendingDownload::cancel(CompletionHandler<void(const IPC::DataReference&)>&& completionHandler)
76{
77 ASSERT(m_networkLoad);
78 m_networkLoad->cancel();
79 completionHandler({ });
80}

82#if PLATFORM(COCOA)
83void PendingDownload::publishProgress(const URL& url, SandboxExtension::Handle&& sandboxExtension)
84{
85 ASSERT(!m_progressURL.isValid());
86 m_progressURL = url;
87 m_progressSandboxExtension = WTFMove(sandboxExtension);
88}
89
90void PendingDownload::didBecomeDownload(const std::unique_ptr<Download>& download)
91{
92 if (m_progressURL.isValid())
93 download->publishProgress(m_progressURL, WTFMove(m_progressSandboxExtension));
94}
95#endif // PLATFORM(COCOA)
96
97void PendingDownload::didFailLoading(const WebCore::ResourceError& error)
98{
99 send(Messages::DownloadProxy::DidFail(error, { }));
100}
101
102IPC::Connection* PendingDownload::messageSenderConnection() const
103{
104 return m_parentProcessConnection.get();
105}
106
107void PendingDownload::didReceiveResponse(WebCore::ResourceResponse&& response, ResponseCompletionHandler&& completionHandler)
108{
109 completionHandler(WebCore::PolicyAction::Download);
110}
111
112uint64_t PendingDownload::messageSenderDestinationID() const
113{
