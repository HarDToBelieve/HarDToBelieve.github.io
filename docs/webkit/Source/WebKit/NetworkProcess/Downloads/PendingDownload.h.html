52public:

64private:

66 void didSendData(unsigned long long bytesSent, unsigned long long totalBytesToBeSent) override { }

74
75 // MessageSender.
76 IPC::Connection* messageSenderConnection() const override;
77 uint64_t messageSenderDestinationID() const override;
78
79private:
