26#import "config.h"
27#import "WKDownloadProgress.h"
28
29#import "Download.h"
30#import <pal/spi/cocoa/NSProgressSPI.h>
31#import <wtf/BlockPtr.h>
32#import <wtf/WeakObjCPtr.h>
33
34static void* WKDownloadProgressBytesExpectedToReceiveCountContext = &WKDownloadProgressBytesExpectedToReceiveCountContext;
35static void* WKDownloadProgressBytesReceivedContext = &WKDownloadProgressBytesReceivedContext;
36
37static NSString * const countOfBytesExpectedToReceiveKeyPath = @"countOfBytesExpectedToReceive";
38static NSString * const countOfBytesReceivedKeyPath = @"countOfBytesReceived";
39
40@implementation WKDownloadProgress {
41 RetainPtr<NSURLSessionDownloadTask> m_task;
42 WeakPtr<WebKit::Download> m_download;
43 RefPtr<WebKit::SandboxExtension> m_sandboxExtension;
44}
45
46- (void)performCancel
47{
48 if (m_download)
49 m_download->cancel([](auto&) { }, WebKit::Download::IgnoreDidFailCallback::No);
50 m_download = nullptr;
51}
52
53- (instancetype)initWithDownloadTask:(NSURLSessionDownloadTask *)task download:(WebKit::Download&)download URL:(NSURL *)fileURL sandboxExtension:(RefPtr<WebKit::SandboxExtension>)sandboxExtension
54{
55 if (!(self = [self initWithParent:nil userInfo:nil]))
56 return nil;
57
58 m_task = task;
59 m_download = makeWeakPtr(download);
60
61 [task addObserver:self forKeyPath:countOfBytesExpectedToReceiveKeyPath options:NSKeyValueObservingOptionNew | NSKeyValueObservingOptionInitial context:WKDownloadProgressBytesExpectedToReceiveCountContext];
62 [task addObserver:self forKeyPath:countOfBytesReceivedKeyPath options:NSKeyValueObservingOptionNew | NSKeyValueObservingOptionInitial context:WKDownloadProgressBytesReceivedContext];
63
64 self.kind = NSProgressKindFile;
65 self.fileOperationKind = NSProgressFileOperationKindDownloading;
66 self.fileURL = fileURL;
67 m_sandboxExtension = sandboxExtension;
68
69 self.cancellable = YES;
70 self.cancellationHandler = makeBlockPtr([weakSelf = WeakObjCPtr<WKDownloadProgress> { self }] () mutable {
71 ensureOnMainRunLoop([weakSelf = WTFMove(weakSelf)] {
72 [weakSelf performCancel];
73 });
74 }).get();
75
76 return self;
77}
78
79#if HAVE(NSPROGRESS_PUBLISHING_SPI)
80- (void)_publish
81#else
82- (void)publish
83#endif
84{
85 BOOL consumedExtension = m_sandboxExtension->consume();
86 ASSERT_UNUSED(consumedExtension, consumedExtension);
87
88#if HAVE(NSPROGRESS_PUBLISHING_SPI)
89 [super _publish];
90#else
91 [super publish];
92#endif
93}
94
95#if HAVE(NSPROGRESS_PUBLISHING_SPI)
96- (void)_unpublish
97#else
98- (void)unpublish
99#endif
100{
101#if HAVE(NSPROGRESS_PUBLISHING_SPI)
102 [super _unpublish];
103#else
104 [super unpublish];
105#endif
106
107 m_sandboxExtension->revoke();
108 m_sandboxExtension = nullptr;
109}
110
111- (void)dealloc
112{
113 [m_task.get() removeObserver:self forKeyPath:countOfBytesExpectedToReceiveKeyPath];
114 [m_task.get() removeObserver:self forKeyPath:countOfBytesReceivedKeyPath];
115
116 ASSERT(!m_sandboxExtension);
117
118 [super dealloc];
119}
120
121- (void)observeValueForKeyPath:(NSString *)keyPath ofObject:(id)object change:(NSDictionary<NSKeyValueChangeKey, id> *)change context:(void *)context
122{
123 if (context == WKDownloadProgressBytesExpectedToReceiveCountContext) {
124 RetainPtr<NSNumber> value = static_cast<NSNumber *>(change[NSKeyValueChangeNewKey]);
125 ASSERT([value isKindOfClass:[NSNumber class]]);
126 int64_t expectedByteCount = value.get().longLongValue;
127 self.totalUnitCount = (expectedByteCount <= 0) ? -1 : expectedByteCount;
128 } else if (context == WKDownloadProgressBytesReceivedContext) {
129 RetainPtr<NSNumber> value = static_cast<NSNumber *>(change[NSKeyValueChangeNewKey]);
130 ASSERT([value isKindOfClass:[NSNumber class]]);
131 self.completedUnitCount = value.get().longLongValue;
132 } else
133 [super observeValueForKeyPath:keyPath ofObject:object change:change context:context];
134}
