47class NetworkCORSPreflightChecker final : private NetworkDataTaskClient {
48 WTF_MAKE_FAST_ALLOCATED;
49public:

54 String referrer;
55 String userAgent;
56 PAL::SessionID sessionID;
57 WebPageProxyIdentifier webPageProxyID;

59 };

62 NetworkCORSPreflightChecker(NetworkProcess&, NetworkResourceLoader*, Parameters&&, bool shouldCaptureExtraNetworkLoadMetrics, CompletionCallback&&);

70private:

73 void didReceiveResponse(WebCore::ResourceResponse&&, NegotiatedLegacyTLS, ResponseCompletionHandler&&) final;
74 void didReceiveData(Ref<WebCore::SharedBuffer>&&) final;

76 void didSendData(uint64_t totalBytesSent, uint64_t totalBytesExpectedToSend) final;
77 void wasBlocked() final;
78 void cannotShowURL() final;
79 void wasBlockedByRestrictions() final;
80
81 Parameters m_parameters;
