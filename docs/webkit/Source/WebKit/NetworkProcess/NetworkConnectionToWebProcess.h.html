147 void addNetworkLoadInformation(ResourceLoadIdentifier identifier, WebCore::NetworkLoadInformation&& information)
148 {
149 ASSERT(!m_networkLoadInformationByID.contains(identifier));

153 void addNetworkLoadInformationMetrics(ResourceLoadIdentifier identifier, const WebCore::NetworkLoadMetrics& metrics)
154 {
155 ASSERT(m_networkLoadInformationByID.contains(identifier));
156 m_networkLoadInformationByID.ensure(identifier, [] {
157 return makeUnique<WebCore::NetworkLoadInformation>();
158 }).iterator->value->metrics = metrics;
159 }
