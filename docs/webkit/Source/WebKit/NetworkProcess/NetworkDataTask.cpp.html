52Ref<NetworkDataTask> NetworkDataTask::create(NetworkSession& session, NetworkDataTaskClient& client, const NetworkLoadParameters& parameters)
53{
54 ASSERT(!parameters.request.url().protocolIsBlob());
55#if PLATFORM(COCOA)
56 return NetworkDataTaskCocoa::create(session, client, parameters);
57#endif
58#if USE(SOUP)
59 return NetworkDataTaskSoup::create(session, client, parameters);
60#endif
61#if USE(CURL)
62 return NetworkDataTaskCurl::create(session, client, parameters.request, parameters.webFrameID, parameters.webPageID, parameters.storedCredentialsPolicy, parameters.contentSniffingPolicy, parameters.contentEncodingSniffingPolicy, parameters.shouldClearReferrerOnHTTPSToHTTPRedirect, parameters.isMainFrameNavigation, parameters.shouldRelaxThirdPartyCookieBlocking);
63#endif
64}

66NetworkDataTask::NetworkDataTask(NetworkSession& session, NetworkDataTaskClient& client, const ResourceRequest& requestWithCredentials, StoredCredentialsPolicy storedCredentialsPolicy, bool shouldClearReferrerOnHTTPSToHTTPRedirect, bool dataTaskIsForMainFrameNavigation)
67 : m_session(makeWeakPtr(session))
68 , m_client(&client)
69 , m_partition(requestWithCredentials.cachePartition())
70 , m_storedCredentialsPolicy(storedCredentialsPolicy)
71 , m_lastHTTPMethod(requestWithCredentials.httpMethod())
72 , m_firstRequest(requestWithCredentials)
73 , m_shouldClearReferrerOnHTTPSToHTTPRedirect(shouldClearReferrerOnHTTPSToHTTPRedirect)
74 , m_dataTaskIsForMainFrameNavigation(dataTaskIsForMainFrameNavigation)
75{
76 ASSERT(RunLoop::isMain());
77
78 if (!requestWithCredentials.url().isValid()) {
79 scheduleFailure(InvalidURLFailure);
80 return;
81 }
82
83 if (!portAllowed(requestWithCredentials.url())) {
84 scheduleFailure(BlockedFailure);
85 return;
86 }
87}

95void NetworkDataTask::scheduleFailure(FailureType type)
96{
97 RunLoop::main().dispatch([this, weakThis = makeWeakPtr(*this), type] {
98 if (!weakThis || !m_client)
99 return;
100
101 switch (type) {
102 case BlockedFailure:
103 m_client->wasBlocked();
104 return;
105 case InvalidURLFailure:
106 m_client->cannotShowURL();
107 return;
108 case RestrictedURLFailure:
109 m_client->wasBlockedByRestrictions();
110 return;
111 }
112 });
113}

115void NetworkDataTask::didReceiveResponse(ResourceResponse&& response, NegotiatedLegacyTLS negotiatedLegacyTLS, ResponseCompletionHandler&& completionHandler)
116{
117 if (response.isHTTP09()) {
118 auto url = response.url();
119 std::optional<uint16_t> port = url.port();
120 if (port && !WTF::isDefaultPortForProtocol(port.value(), url.protocol())) {
121 completionHandler(PolicyAction::Ignore);
122 cancel();
123 if (m_client)
124 m_client->didCompleteWithError({ String(), 0, url, "Cancelled load from '" + url.stringCenterEllipsizedToLength() + "' because it is using HTTP/0.9." });
125 return;
126 }
127 }
128
129 response.setSource(ResourceResponse::Source::Network);
130 if (negotiatedLegacyTLS == NegotiatedLegacyTLS::Yes)
131 response.setUsedLegacyTLS(UsedLegacyTLS::Yes);
132
133 if (m_client)
134 m_client->didReceiveResponse(WTFMove(response), negotiatedLegacyTLS, WTFMove(completionHandler));
135 else
136 completionHandler(PolicyAction::Ignore);
137}

166{

171{
172#if ENABLE(RESOURCE_LOAD_STATISTICS)
173 if ((m_session->sessionID().isEphemeral() || m_session->isResourceLoadStatisticsEnabled()) && m_session->shouldDowngradeReferrer() && isThirdPartyRequest(request))
174 request.setExistingHTTPReferrerToOriginString();
