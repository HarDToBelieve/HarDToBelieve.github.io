56using RedirectCompletionHandler = CompletionHandler<void(WebCore::ResourceRequest&&)>;
57using ChallengeCompletionHandler = CompletionHandler<void(AuthenticationChallengeDisposition, const WebCore::Credential&)>;
58using ResponseCompletionHandler = CompletionHandler<void(WebCore::PolicyAction)>;
59
60class NetworkDataTaskClient {
61public:
62 virtual void willPerformHTTPRedirection(WebCore::ResourceResponse&&, WebCore::ResourceRequest&&, RedirectCompletionHandler&&) = 0;
63 virtual void didReceiveChallenge(WebCore::AuthenticationChallenge&&, NegotiatedLegacyTLS, ChallengeCompletionHandler&&) = 0;
64 virtual void didReceiveResponse(WebCore::ResourceResponse&&, NegotiatedLegacyTLS, ResponseCompletionHandler&&) = 0;
65 virtual void didReceiveData(Ref<WebCore::SharedBuffer>&&) = 0;
66 virtual void didCompleteWithError(const WebCore::ResourceError&, const WebCore::NetworkLoadMetrics&) = 0;
67 virtual void didSendData(uint64_t totalBytesSent, uint64_t totalBytesExpectedToSend) = 0;
68 virtual void wasBlocked() = 0;
69 virtual void cannotShowURL() = 0;
70 virtual void wasBlockedByRestrictions() = 0;
71
72 virtual bool shouldCaptureExtraNetworkLoadMetrics() const { return false; }
73
74 virtual void didNegotiateModernTLS(const URL&) { }
75
76 void didCompleteWithError(const WebCore::ResourceError& error)
77 {
78 WebCore::NetworkLoadMetrics emptyMetrics;
79 didCompleteWithError(error, emptyMetrics);
80 }
81
82 virtual ~NetworkDataTaskClient() { }
83};
84
85class NetworkDataTask : public ThreadSafeRefCounted<NetworkDataTask, WTF::DestructionThread::Main>, public CanMakeWeakPtr<NetworkDataTask> {
86public:
87 static Ref<NetworkDataTask> create(NetworkSession&, NetworkDataTaskClient&, const NetworkLoadParameters&);
88
89 virtual ~NetworkDataTask();
90
91 virtual void cancel() = 0;
92 virtual void resume() = 0;
93 virtual void invalidateAndCancel() = 0;
94
95 void didReceiveResponse(WebCore::ResourceResponse&&, NegotiatedLegacyTLS, ResponseCompletionHandler&&);
96 bool shouldCaptureExtraNetworkLoadMetrics() const;
97
98 enum class State {
99 Running,
100 Suspended,
101 Canceling,
102 Completed
103 };
104 virtual State state() const = 0;
105
106 NetworkDataTaskClient* client() const { return m_client; }
107 void clearClient() { m_client = nullptr; }
108
109 DownloadID pendingDownloadID() const { return m_pendingDownloadID; }
110 PendingDownload* pendingDownload() const { return m_pendingDownload; }
111 void setPendingDownloadID(DownloadID downloadID)
112 {
113 ASSERT(!m_pendingDownloadID);
114 ASSERT(downloadID);
115 m_pendingDownloadID = downloadID;
116 }
117 void setPendingDownload(PendingDownload& pendingDownload)
118 {
119 ASSERT(!m_pendingDownload);
120 m_pendingDownload = &pendingDownload;
121 }
122
123 virtual void setPendingDownloadLocation(const String& filename, SandboxExtension::Handle&&, bool /*allowOverwrite*/) { m_pendingDownloadLocation = filename; }
124 const String& pendingDownloadLocation() const { return m_pendingDownloadLocation; }
125 bool isDownload() const { return !!m_pendingDownloadID; }
126
127 const WebCore::ResourceRequest& firstRequest() const { return m_firstRequest; }
128 virtual String suggestedFilename() const { return String(); }
129 void setSuggestedFilename(const String& suggestedName) { m_suggestedFilename = suggestedName; }
130 const String& partition() { return m_partition; }
131
132 bool isTopLevelNavigation() const { return m_dataTaskIsForMainFrameNavigation; }
133
134 virtual String description() const;
135 virtual void setH2PingCallback(const URL&, CompletionHandler<void(Expected<WTF::Seconds, WebCore::ResourceError>&&)>&&);
136
137 virtual void setPriority(WebCore::ResourceLoadPriority) { }
138
139 PAL::SessionID sessionID() const;
140
141 NetworkSession* networkSession();
142
143protected:
144 NetworkDataTask(NetworkSession&, NetworkDataTaskClient&, const WebCore::ResourceRequest&, WebCore::StoredCredentialsPolicy, bool shouldClearReferrerOnHTTPSToHTTPRedirect, bool dataTaskIsForMainFrameNavigation);
145
146 enum FailureType {
147 BlockedFailure,
148 InvalidURLFailure,
149 RestrictedURLFailure
150 };
151 void scheduleFailure(FailureType);
152
153 bool isThirdPartyRequest(const WebCore::ResourceRequest&) const;
154 void restrictRequestReferrerToOriginIfNeeded(WebCore::ResourceRequest&);
155
156 WeakPtr<NetworkSession> m_session;
157 NetworkDataTaskClient* m_client { nullptr };
158 PendingDownload* m_pendingDownload { nullptr };
159 DownloadID m_pendingDownloadID;
160 String m_user;
161 String m_password;
162 String m_partition;
163#if USE(CREDENTIAL_STORAGE_WITH_NETWORK_SESSION)
164 WebCore::Credential m_initialCredential;
165#endif
166 WebCore::StoredCredentialsPolicy m_storedCredentialsPolicy { WebCore::StoredCredentialsPolicy::DoNotUse };
167 String m_lastHTTPMethod;
168 String m_pendingDownloadLocation;
169 WebCore::ResourceRequest m_firstRequest;
170 bool m_shouldClearReferrerOnHTTPSToHTTPRedirect { true };
171 String m_suggestedFilename;
172 bool m_dataTaskIsForMainFrameNavigation { false };
173};
