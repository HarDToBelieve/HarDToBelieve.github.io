69NetworkDataTaskBlob::NetworkDataTaskBlob(NetworkSession& session, BlobRegistryImpl& blobRegistry, NetworkDataTaskClient& client, const ResourceRequest& request, ContentSniffingPolicy shouldContentSniff, const Vector<RefPtr<WebCore::BlobDataFileReference>>& fileReferences)
70 : NetworkDataTask(session, client, request, StoredCredentialsPolicy::DoNotUse, false, false)
71 , m_stream(makeUnique<AsyncFileStream>(*this))
72 , m_fileReferences(fileReferences)
73 , m_networkProcess(session.networkProcess())
74{
75 for (auto& fileReference : m_fileReferences)
76 fileReference->prepareForFileAccess();
77
78 m_blobData = blobRegistry.getBlobDataFromURL(request.url());
79
80 m_session->registerNetworkDataTask(*this);
81 LOG(NetworkSession, "%p - Created NetworkDataTaskBlob for %s", this, request.url().string().utf8().data());
82}

84NetworkDataTaskBlob::~NetworkDataTaskBlob()
85{
86 for (auto& fileReference : m_fileReferences)
87 fileReference->revokeFileAccess();
88
89 clearStream();
90 if (m_session)
91 m_session->unregisterNetworkDataTask(*this);
92}

94void NetworkDataTaskBlob::clearStream()
95{
96 if (m_state == State::Completed)
97 return;
98
99 m_state = State::Completed;
100
101 if (m_fileOpened) {
102 m_fileOpened = false;
103 m_stream->close();
104 }
105 m_stream = nullptr;
106}

108void NetworkDataTaskBlob::resume()
109{
110 ASSERT(m_state != State::Running);
111 if (m_state == State::Canceling || m_state == State::Completed)
112 return;
113
114 m_state = State::Running;
115
116 RunLoop::main().dispatch([this, protectedThis = makeRef(*this)] {
117 if (m_state == State::Canceling || m_state == State::Completed || !m_client) {
118 clearStream();
119 return;
120 }
121
122 if (!equalLettersIgnoringASCIICase(m_firstRequest.httpMethod(), "get")) {
123 didFail(Error::MethodNotAllowed);
124 return;
125 }
126
127 // If the blob data is not found, fail now.
128 if (!m_blobData) {
129 didFail(Error::NotFoundError);
130 return;
131 }
132
133 // Parse the "Range" header we care about.
134 String range = m_firstRequest.httpHeaderField(HTTPHeaderName::Range);
135 if (!range.isEmpty() && !parseRange(range, m_rangeOffset, m_rangeEnd, m_rangeSuffixLength)) {
136 dispatchDidReceiveResponse(Error::RangeError);
137 return;
138 }
139
140 getSizeForNext();
141 });
142}

144void NetworkDataTaskBlob::cancel()
145{
146 if (m_state == State::Canceling || m_state == State::Completed)
147 return;
148
149 m_state = State::Canceling;
150
151 if (m_fileOpened) {
152 m_fileOpened = false;
153 m_stream->close();
154 }
155
156 if (isDownload())
157 cleanDownloadFiles();
158}

166void NetworkDataTaskBlob::getSizeForNext()
167{
168 ASSERT(RunLoop::isMain());
169
170 // Do we finish validating and counting size for all items?
171 if (m_sizeItemCount >= m_blobData->items().size()) {
172 seek();
173 dispatchDidReceiveResponse();
174 return;
175 }
176
177 const BlobDataItem& item = m_blobData->items().at(m_sizeItemCount);
178 switch (item.type()) {
179 case BlobDataItem::Type::Data:
180 didGetSize(item.length());
181 break;
182 case BlobDataItem::Type::File:
183 // Files know their sizes, but asking the stream to verify that the file wasn't modified.
184 m_stream->getSize(item.file()->path(), item.file()->expectedModificationTime());
185 break;
186 default:
187 ASSERT_NOT_REACHED();
188 }
189}

191void NetworkDataTaskBlob::didGetSize(long long size)
192{
193 ASSERT(RunLoop::isMain());
194
195 if (m_state == State::Canceling || m_state == State::Completed || (!m_client && !isDownload())) {
196 clearStream();
197 return;
198 }
199
200 // If the size is -1, it means the file has been moved or changed. Fail now.
201 if (size == -1) {
202 didFail(Error::NotFoundError);
203 return;
204 }
205
206 // The size passed back is the size of the whole file. If the underlying item is a sliced file, we need to use the slice length.
207 const BlobDataItem& item = m_blobData->items().at(m_sizeItemCount);
208 size = item.length();
209
210 // Cache the size.
211 m_itemLengthList.append(size);
212
213 // Count the size.
214 m_totalSize += size;
215 m_totalRemainingSize += size;
216 m_sizeItemCount++;
217
218 // Continue with the next item.
219 getSizeForNext();
220}

222void NetworkDataTaskBlob::seek()
223{
224 ASSERT(RunLoop::isMain());
225
226 // Convert from the suffix length to the range.
227 if (m_rangeSuffixLength != kPositionNotSpecified) {
228 m_rangeOffset = m_totalRemainingSize - m_rangeSuffixLength;
229 m_rangeEnd = m_rangeOffset + m_rangeSuffixLength - 1;
230 }
231
232 // Bail out if the range is not provided.
233 if (m_rangeOffset == kPositionNotSpecified)
234 return;
235
236 // Skip the initial items that are not in the range.
237 long long offset = m_rangeOffset;
238 for (m_readItemCount = 0; m_readItemCount < m_blobData->items().size() && offset >= m_itemLengthList[m_readItemCount]; ++m_readItemCount)
239 offset -= m_itemLengthList[m_readItemCount];
240
241 // Set the offset that need to jump to for the first item in the range.
242 m_currentItemReadSize = offset;
243
244 // Adjust the total remaining size in order not to go beyond the range.
245 if (m_rangeEnd != kPositionNotSpecified) {
246 long long rangeSize = m_rangeEnd - m_rangeOffset + 1;
247 if (m_totalRemainingSize > rangeSize)
248 m_totalRemainingSize = rangeSize;
249 } else
250 m_totalRemainingSize -= m_rangeOffset;
251}

253void NetworkDataTaskBlob::dispatchDidReceiveResponse(Error errorCode)
254{
255 LOG(NetworkSession, "%p - NetworkDataTaskBlob::dispatchDidReceiveResponse(%u)", this, static_cast<unsigned>(errorCode));
256
257 Ref<NetworkDataTaskBlob> protectedThis(*this);
258 ResourceResponse response(m_firstRequest.url(), errorCode != Error::NoError ? "text/plain" : extractMIMETypeFromMediaType(m_blobData->contentType()), errorCode != Error::NoError ? 0 : m_totalRemainingSize, String());
259 switch (errorCode) {
260 case Error::NoError: {
261 bool isRangeRequest = m_rangeOffset != kPositionNotSpecified;
262 response.setHTTPStatusCode(isRangeRequest ? httpPartialContent : httpOK);
263 response.setHTTPStatusText(isRangeRequest ? httpPartialContentText : httpOKText);
264
265 response.setHTTPHeaderField(HTTPHeaderName::ContentType, m_blobData->contentType());
266 response.setHTTPHeaderField(HTTPHeaderName::ContentLength, String::number(m_totalRemainingSize));
267
268 if (isRangeRequest)
269 response.setHTTPHeaderField(HTTPHeaderName::ContentRange, ParsedContentRange(m_rangeOffset, m_rangeEnd, m_totalSize).headerValue());
270 // FIXME: If a resource identified with a blob: URL is a File object, user agents must use that file's name attribute,
271 // as if the response had a Content-Disposition header with the filename parameter set to the File's name attribute.
272 // Notably, this will affect a name suggested in "File Save As".
273 break;
274 }
275 case Error::RangeError:
276 response.setHTTPStatusCode(httpRequestedRangeNotSatisfiable);
277 response.setHTTPStatusText(httpRequestedRangeNotSatisfiableText);
278 break;
279 case Error::SecurityError:
280 response.setHTTPStatusCode(httpNotAllowed);
281 response.setHTTPStatusText(httpNotAllowedText);
282 break;
283 default:
284 response.setHTTPStatusCode(httpInternalError);
285 response.setHTTPStatusText(httpInternalErrorText);
286 break;
287 }
288
289 didReceiveResponse(WTFMove(response), NegotiatedLegacyTLS::No, [this, protectedThis = WTFMove(protectedThis), errorCode](PolicyAction policyAction) {
290 LOG(NetworkSession, "%p - NetworkDataTaskBlob::didReceiveResponse completionHandler (%u)", this, static_cast<unsigned>(policyAction));
291
292 if (m_state == State::Canceling || m_state == State::Completed) {
293 clearStream();
294 return;
295 }
296
297 if (errorCode != Error::NoError) {
298 didFinish();
299 return;
300 }
301
302 switch (policyAction) {
303 case PolicyAction::Use:
304 m_buffer.resize(bufferSize);
305 read();
306 break;
307 case PolicyAction::StopAllLoads:
308 ASSERT_NOT_REACHED();
309 break;
310 case PolicyAction::Ignore:
311 break;
312 case PolicyAction::Download:
313 download();
314 break;
315 }
316 });
317}

319void NetworkDataTaskBlob::read()
320{
321 ASSERT(RunLoop::isMain());
322
323 // If there is no more remaining data to read, we are done.
324 if (!m_totalRemainingSize || m_readItemCount >= m_blobData->items().size()) {
325 didFinish();
326 return;
327 }
328
329 const BlobDataItem& item = m_blobData->items().at(m_readItemCount);
330 if (item.type() == BlobDataItem::Type::Data)
331 readData(item);
332 else if (item.type() == BlobDataItem::Type::File)
333 readFile(item);
334 else
335 ASSERT_NOT_REACHED();
336}

338void NetworkDataTaskBlob::readData(const BlobDataItem& item)
339{
340 ASSERT(item.data().data());
341
342 long long bytesToRead = item.length() - m_currentItemReadSize;
343 ASSERT(bytesToRead >= 0);
344 if (bytesToRead > m_totalRemainingSize)
345 bytesToRead = m_totalRemainingSize;
346
347 auto* data = item.data().data()->data() + item.offset() + m_currentItemReadSize;
348 m_currentItemReadSize = 0;
349
350 consumeData(data, static_cast<int>(bytesToRead));
351}
352
353void NetworkDataTaskBlob::readFile(const BlobDataItem& item)
354{
355 ASSERT(m_stream);
356
357 if (m_fileOpened) {
358 m_stream->read(m_buffer.data(), m_buffer.size());
359 return;
360 }
361
362 long long bytesToRead = m_itemLengthList[m_readItemCount] - m_currentItemReadSize;
363 if (bytesToRead > m_totalRemainingSize)
364 bytesToRead = static_cast<int>(m_totalRemainingSize);
365 m_stream->openForRead(item.file()->path(), item.offset() + m_currentItemReadSize, bytesToRead);
366 m_fileOpened = true;
367 m_currentItemReadSize = 0;
368}

386void NetworkDataTaskBlob::didRead(int bytesRead)
387{
388 if (m_state == State::Canceling || m_state == State::Completed || (!m_client && !isDownload())) {
389 clearStream();
390 return;
391 }
392
393 if (bytesRead < 0) {
394 didFail(Error::NotReadableError);
395 return;

402void NetworkDataTaskBlob::consumeData(const uint8_t* data, int bytesRead)
403{
404 m_totalRemainingSize -= bytesRead;
405
406 if (bytesRead) {
407 if (m_downloadFile != FileSystem::invalidPlatformFileHandle) {
408 if (!writeDownload(data, bytesRead))
409 return;
410 } else {
411 ASSERT(m_client);
412 m_client->didReceiveData(SharedBuffer::create(data, bytesRead));
413 }
414 }
415
416 if (m_fileOpened) {
417 // When the current item is a file item, the reading is completed only if bytesRead is 0.
418 if (!bytesRead) {
419 // Close the file.
420 m_fileOpened = false;
421 m_stream->close();
422
423 // Move to the next item.
424 m_readItemCount++;
425 }
426 } else {
427 // Otherwise, we read the current text item as a whole and move to the next item.
428 m_readItemCount++;
429 }
430
431 read();
432}

452void NetworkDataTaskBlob::download()
453{
454 ASSERT(isDownload());
455 ASSERT(m_pendingDownloadLocation);
456 ASSERT(m_session);
457
458 LOG(NetworkSession, "%p - NetworkDataTaskBlob::download to %s", this, m_pendingDownloadLocation.utf8().data());
459
460 m_downloadFile = FileSystem::openFile(m_pendingDownloadLocation, FileSystem::FileOpenMode::Write);
461 if (m_downloadFile == FileSystem::invalidPlatformFileHandle) {
462 didFailDownload(cancelledError(m_firstRequest));
463 return;
464 }
465
466 auto& downloadManager = m_networkProcess->downloadManager();
467 auto download = makeUnique<Download>(downloadManager, m_pendingDownloadID, *this, *m_session, suggestedFilename());
468 auto* downloadPtr = download.get();
469 downloadManager.dataTaskBecameDownloadTask(m_pendingDownloadID, WTFMove(download));
470 downloadPtr->didCreateDestination(m_pendingDownloadLocation);
471
472 ASSERT(!m_client);
473
474 m_buffer.resize(bufferSize);
475 read();
476}

478bool NetworkDataTaskBlob::writeDownload(const uint8_t* data, int bytesRead)
479{
480 ASSERT(isDownload());
481 int bytesWritten = FileSystem::writeToFile(m_downloadFile, data, bytesRead);
482 if (bytesWritten != bytesRead) {
483 didFailDownload(cancelledError(m_firstRequest));
484 return false;
485 }
486
487 m_downloadBytesWritten += bytesWritten;
488 auto* download = m_networkProcess->downloadManager().download(m_pendingDownloadID);
489 ASSERT(download);
490 download->didReceiveData(bytesWritten, m_downloadBytesWritten, m_totalSize);
491 return true;
492}

494void NetworkDataTaskBlob::cleanDownloadFiles()
495{
496 if (m_downloadFile != FileSystem::invalidPlatformFileHandle) {
497 FileSystem::closeFile(m_downloadFile);
498 m_downloadFile = FileSystem::invalidPlatformFileHandle;
499 }
500 FileSystem::deleteFile(m_pendingDownloadLocation);
501}

503void NetworkDataTaskBlob::didFailDownload(const ResourceError& error)
504{
505 LOG(NetworkSession, "%p - NetworkDataTaskBlob::didFailDownload", this);
506
507 clearStream();
508 cleanDownloadFiles();
509
510 if (m_sandboxExtension) {
511 m_sandboxExtension->revoke();
512 m_sandboxExtension = nullptr;
513 }
514
515 if (m_client)
516 m_client->didCompleteWithError(error);
517 else {
518 auto* download = m_networkProcess->downloadManager().download(m_pendingDownloadID);
519 ASSERT(download);
520 download->didFail(error, IPC::DataReference());
521 }
522}

524void NetworkDataTaskBlob::didFinishDownload()
525{
526 LOG(NetworkSession, "%p - NetworkDataTaskBlob::didFinishDownload", this);
527
528 ASSERT(isDownload());
529 FileSystem::closeFile(m_downloadFile);
530 m_downloadFile = FileSystem::invalidPlatformFileHandle;
531
532 if (m_sandboxExtension) {
533 m_sandboxExtension->revoke();
534 m_sandboxExtension = nullptr;
535 }
536
537 clearStream();
538 auto* download = m_networkProcess->downloadManager().download(m_pendingDownloadID);
539 ASSERT(download);
540 download->didFinish();
541}

543void NetworkDataTaskBlob::didFail(Error errorCode)
544{
545 ASSERT(!m_sandboxExtension);
546
547 Ref<NetworkDataTaskBlob> protectedThis(*this);
548 if (isDownload()) {
549 didFailDownload(ResourceError(webKitBlobResourceDomain, static_cast<int>(errorCode), m_firstRequest.url(), String()));
550 return;
551 }
552
553 LOG(NetworkSession, "%p - NetworkDataTaskBlob::didFail", this);
554
555 clearStream();
556 ASSERT(m_client);
557 m_client->didCompleteWithError(ResourceError(webKitBlobResourceDomain, static_cast<int>(errorCode), m_firstRequest.url(), String()));
558}
559
560void NetworkDataTaskBlob::didFinish()
561{
562 if (m_downloadFile != FileSystem::invalidPlatformFileHandle) {
563 didFinishDownload();
564 return;
565 }
566
567 ASSERT(!m_sandboxExtension);
568
569 LOG(NetworkSession, "%p - NetworkDataTaskBlob::didFinish", this);
570
571 clearStream();
572 ASSERT(m_client);
573 m_client->didCompleteWithError({ });
574}
