47NetworkLoad::NetworkLoad(NetworkLoadClient& client, BlobRegistryImpl* blobRegistry, NetworkLoadParameters&& parameters, NetworkSession& networkSession)
48 : m_client(client)
49 , m_networkProcess(networkSession.networkProcess())
50 , m_parameters(WTFMove(parameters))
51 , m_currentRequest(m_parameters.request)
52{
53 if (blobRegistry && m_parameters.request.url().protocolIsBlob())
54 m_task = NetworkDataTaskBlob::create(networkSession, *blobRegistry, *this, m_parameters.request, m_parameters.contentSniffingPolicy, m_parameters.blobFileReferences);
55 else
56 m_task = NetworkDataTask::create(networkSession, *this, m_parameters);
57}

65void NetworkLoad::startWithScheduling()
66{
67 if (!m_task || !m_task->networkSession())
68 return;
69 m_scheduler = makeWeakPtr(m_task->networkSession()->networkLoadScheduler());
70 m_scheduler->schedule(*this);
71}
72
73NetworkLoad::~NetworkLoad()
74{
75 ASSERT(RunLoop::isMain());
76 if (m_scheduler)
77 m_scheduler->unschedule(*this);
78 if (m_redirectCompletionHandler)
79 m_redirectCompletionHandler({ });
80 if (m_task)
81 m_task->clearClient();
82}

84void NetworkLoad::cancel()
85{
86 if (m_task)
87 m_task->cancel();
88}

90static inline void updateRequest(ResourceRequest& currentRequest, const ResourceRequest& newRequest)
91{
92#if PLATFORM(COCOA)
93 currentRequest.updateFromDelegatePreservingOldProperties(newRequest.nsURLRequest(HTTPBodyUpdatePolicy::DoNotUpdateHTTPBody));

95 // FIXME: Implement ResourceRequest::updateFromDelegatePreservingOldProperties. See https://bugs.webkit.org/show_bug.cgi?id=126127.

114void NetworkLoad::continueWillSendRequest(WebCore::ResourceRequest&& newRequest)
115{
116 updateRequest(m_currentRequest, newRequest);
117
118 auto redirectCompletionHandler = std::exchange(m_redirectCompletionHandler, nullptr);
119 ASSERT(redirectCompletionHandler);
120 if (m_currentRequest.isNull()) {
121 NetworkLoadMetrics emptyMetrics;
122 didCompleteWithError(cancelledError(m_currentRequest), emptyMetrics);
123 if (redirectCompletionHandler)
124 redirectCompletionHandler({ });
125 return;
126 }
127
128 if (redirectCompletionHandler)
129 redirectCompletionHandler(ResourceRequest(m_currentRequest));
130}

142void NetworkLoad::convertTaskToDownload(PendingDownload& pendingDownload, const ResourceRequest& updatedRequest, const ResourceResponse& response, ResponseCompletionHandler&& completionHandler)
143{
144 if (!m_task)
145 return completionHandler(PolicyAction::Ignore);
146
147 m_client = pendingDownload;
148 m_currentRequest = updatedRequest;
149 m_task->setPendingDownload(pendingDownload);
150
151 m_networkProcess->findPendingDownloadLocation(*m_task.get(), WTFMove(completionHandler), response);
152}

178void NetworkLoad::willPerformHTTPRedirection(ResourceResponse&& redirectResponse, ResourceRequest&& request, RedirectCompletionHandler&& completionHandler)
179{
180 ASSERT(!redirectResponse.isNull());
181 ASSERT(RunLoop::isMain());
182 ASSERT(!m_redirectCompletionHandler);
183
184 redirectResponse.setSource(ResourceResponse::Source::Network);
185 m_redirectCompletionHandler = WTFMove(completionHandler);
186
187 auto oldRequest = WTFMove(m_currentRequest);
188 request.setRequester(oldRequest.requester());
189
190 m_currentRequest = request;
191 m_client.get().willSendRedirectedRequest(WTFMove(oldRequest), WTFMove(request), WTFMove(redirectResponse));
192}

194void NetworkLoad::didReceiveChallenge(AuthenticationChallenge&& challenge, NegotiatedLegacyTLS negotiatedLegacyTLS, ChallengeCompletionHandler&& completionHandler)
195{
196 m_client.get().didReceiveChallenge(challenge);
197
198 auto scheme = challenge.protectionSpace().authenticationScheme();
199 bool isTLSHandshake = scheme == ProtectionSpaceAuthenticationSchemeServerTrustEvaluationRequested
200 || scheme == ProtectionSpaceAuthenticationSchemeClientCertificateRequested;
201 if (!isAllowedToAskUserForCredentials() && !isTLSHandshake && !challenge.protectionSpace().isProxy()) {
202 m_client.get().didBlockAuthenticationChallenge();
203 completionHandler(AuthenticationChallengeDisposition::UseCredential, { });
204 return;
205 }
206
207 if (auto* pendingDownload = m_task->pendingDownload())
208 m_networkProcess->authenticationManager().didReceiveAuthenticationChallenge(*pendingDownload, challenge, WTFMove(completionHandler));
209 else
210 m_networkProcess->authenticationManager().didReceiveAuthenticationChallenge(m_task->sessionID(), m_parameters.webPageProxyID, m_parameters.topOrigin ? &m_parameters.topOrigin->data() : nullptr, challenge, negotiatedLegacyTLS, WTFMove(completionHandler));
211}

213void NetworkLoad::didReceiveResponse(ResourceResponse&& response, NegotiatedLegacyTLS negotiatedLegacyTLS, ResponseCompletionHandler&& completionHandler)
214{
215 ASSERT(RunLoop::isMain());
216
217 if (m_task && m_task->isDownload()) {
218 m_networkProcess->findPendingDownloadLocation(*m_task.get(), WTFMove(completionHandler), response);
219 return;
220 }
221
222 if (negotiatedLegacyTLS == NegotiatedLegacyTLS::Yes)
223 m_networkProcess->authenticationManager().negotiatedLegacyTLS(m_parameters.webPageProxyID);
224
225 notifyDidReceiveResponse(WTFMove(response), negotiatedLegacyTLS, WTFMove(completionHandler));
226}
227
228void NetworkLoad::notifyDidReceiveResponse(ResourceResponse&& response, NegotiatedLegacyTLS negotiatedLegacyTLS, ResponseCompletionHandler&& completionHandler)
229{
230 ASSERT(RunLoop::isMain());
231
232 if (m_parameters.needsCertificateInfo)
233 response.includeCertificateInfo();
234
235 m_client.get().didReceiveResponse(WTFMove(response), WTFMove(completionHandler));
236}

245void NetworkLoad::didCompleteWithError(const ResourceError& error, const WebCore::NetworkLoadMetrics& networkLoadMetrics)
246{
247 if (m_scheduler) {
248 m_scheduler->unschedule(*this);
249 m_scheduler = nullptr;
250 }
251
252 if (error.isNull())
253 m_client.get().didFinishLoading(networkLoadMetrics);
254 else
255 m_client.get().didFailLoading(error);
256}
