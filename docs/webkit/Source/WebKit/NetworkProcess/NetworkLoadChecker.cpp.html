48static inline bool isSameOrigin(const URL& url, const SecurityOrigin* origin)
49{
50 return url.protocolIsData() || url.protocolIsBlob() || !origin || origin->canRequest(url);
51}

53NetworkLoadChecker::NetworkLoadChecker(NetworkProcess& networkProcess, NetworkResourceLoader* networkResourceLoader, NetworkSchemeRegistry* schemeRegistry, FetchOptions&& options, PAL::SessionID sessionID, WebPageProxyIdentifier webPageProxyID, HTTPHeaderMap&& originalRequestHeaders, URL&& url, DocumentURL&& documentURL, RefPtr<SecurityOrigin>&& sourceOrigin, RefPtr<SecurityOrigin>&& topOrigin, PreflightPolicy preflightPolicy, String&& referrer, bool shouldCaptureExtraNetworkLoadMetrics, LoadType requestLoadType)
54 : m_options(WTFMove(options))
55 , m_sessionID(sessionID)
56 , m_networkProcess(networkProcess)
57 , m_webPageProxyID(webPageProxyID)
58 , m_originalRequestHeaders(WTFMove(originalRequestHeaders))
59 , m_url(WTFMove(url))
60 , m_documentURL(WTFMove(documentURL))
61 , m_origin(WTFMove(sourceOrigin))
62 , m_topOrigin(WTFMove(topOrigin))
63 , m_preflightPolicy(preflightPolicy)
64 , m_referrer(WTFMove(referrer))
65 , m_shouldCaptureExtraNetworkLoadMetrics(shouldCaptureExtraNetworkLoadMetrics)
66 , m_requestLoadType(requestLoadType)
67 , m_schemeRegistry(schemeRegistry)
68 , m_networkResourceLoader(makeWeakPtr(networkResourceLoader))
69{
70 m_isSameOriginRequest = isSameOrigin(m_url, m_origin.get());
71 switch (options.credentials) {
72 case FetchOptions::Credentials::Include:
73 m_storedCredentialsPolicy = StoredCredentialsPolicy::Use;
74 break;
75 case FetchOptions::Credentials::SameOrigin:
76 m_storedCredentialsPolicy = m_isSameOriginRequest ? StoredCredentialsPolicy::Use : StoredCredentialsPolicy::DoNotUse;
77 break;
78 case FetchOptions::Credentials::Omit:
79 m_storedCredentialsPolicy = StoredCredentialsPolicy::DoNotUse;
80 break;
81 }
82}

86void NetworkLoadChecker::check(ResourceRequest&& request, ContentSecurityPolicyClient* client, ValidationHandler&& handler)
87{
88 ASSERT(!isChecking());
89
90 if (m_shouldCaptureExtraNetworkLoadMetrics)
91 m_loadInformation.request = request;
92
93 m_firstRequestHeaders = request.httpHeaderFields();
94 checkRequest(WTFMove(request), client, WTFMove(handler));
95}

102void NetworkLoadChecker::checkRedirection(ResourceRequest&& request, ResourceRequest&& redirectRequest, ResourceResponse&& redirectResponse, ContentSecurityPolicyClient* client, RedirectionValidationHandler&& handler)
103{
104 ASSERT(!isChecking());
105
106 auto error = validateResponse(request, redirectResponse);
107 if (!error.isNull()) {
108 handler(redirectionError(redirectResponse, makeString("Cross-origin redirection to ", redirectRequest.url().string(), " denied by Cross-Origin Resource Sharing policy: ", error.localizedDescription())));
109 return;
110 }
111
112 if (m_options.redirect == FetchOptions::Redirect::Error) {
113 handler(redirectionError(redirectResponse, makeString("Not allowed to follow a redirection while loading ", redirectResponse.url().string())));
114 return;
115 }
116 if (m_options.redirect == FetchOptions::Redirect::Manual) {
117 handler(RedirectionTriplet { WTFMove(request), WTFMove(redirectRequest), WTFMove(redirectResponse) });
118 return;
119 }
120
121 // FIXME: We should check that redirections are only HTTP(s) as per fetch spec.
122 // See https://github.com/whatwg/fetch/issues/393
123
124 if (m_options.mode == FetchOptions::Mode::Cors && (!m_isSameOriginRequest || !isSameOrigin(request.url(), m_origin.get()))) {
125 auto location = URL(redirectResponse.url(), redirectResponse.httpHeaderField(HTTPHeaderName::Location));
126 if (m_schemeRegistry && !m_schemeRegistry->shouldTreatURLSchemeAsCORSEnabled(location.protocol())) {
127 handler(redirectionError(redirectResponse, makeString("Cross-origin redirection to ", redirectRequest.url().string(), " denied by Cross-Origin Resource Sharing policy: not allowed to follow a cross-origin CORS redirection with non CORS scheme")));
128 return;
129 }
130 if (location.hasCredentials()) {
131 handler(redirectionError(redirectResponse, makeString("Cross-origin redirection to ", redirectRequest.url().string(), " denied by Cross-Origin Resource Sharing policy: redirection URL ", location.string(), " has credentials")));
132 return;
133 }
134 }
135
136 if (++m_redirectCount > 20) {
137 handler(redirectionError(redirectResponse, "Load cannot follow more than 20 redirections"_s));
138 return;
139 }
140
141 m_previousURL = WTFMove(m_url);
142 m_url = redirectRequest.url();
143
144 checkRequest(WTFMove(redirectRequest), client, [handler = WTFMove(handler), request = WTFMove(request), redirectResponse = WTFMove(redirectResponse)](auto&& result) mutable {
145 WTF::switchOn(result,
146 [&handler] (ResourceError& error) mutable {
147 handler(makeUnexpected(WTFMove(error)));
148 },
149 [&handler, &request, &redirectResponse] (RedirectionTriplet& triplet) mutable {
150 // FIXME: if checkRequest returns a RedirectionTriplet, it means the requested URL has changed and we should update the redirectResponse to match.
151 handler(RedirectionTriplet { WTFMove(request), WTFMove(triplet.redirectRequest), WTFMove(redirectResponse) });
152 },
153 [&handler, &request, &redirectResponse] (ResourceRequest& redirectRequest) mutable {
154 handler(RedirectionTriplet { WTFMove(request), WTFMove(redirectRequest), WTFMove(redirectResponse) });
155 }
156 );
157 });
158}

160ResourceError NetworkLoadChecker::validateResponse(const ResourceRequest& request, ResourceResponse& response)
161{
162 if (m_redirectCount)
163 response.setRedirected(true);
164
165 if (response.type() == ResourceResponse::Type::Opaqueredirect) {
166 response.setTainting(ResourceResponse::Tainting::Opaqueredirect);
167 return { };
168 }
169
170 if (m_options.mode == FetchOptions::Mode::Navigate || m_isSameOriginRequest) {
171 response.setTainting(ResourceResponse::Tainting::Basic);
172 return { };
173 }
174
175 if (request.hasHTTPHeaderField(HTTPHeaderName::Range))
176 response.setAsRangeRequested();
177
178 if (m_options.mode == FetchOptions::Mode::NoCors) {
179 if (auto error = validateCrossOriginResourcePolicy(*m_origin, m_url, response))
180 return WTFMove(*error);
181
182 response.setTainting(ResourceResponse::Tainting::Opaque);
183 return { };
184 }
185
186 ASSERT(m_options.mode == FetchOptions::Mode::Cors);
187
188 // If we have a 304, the cached response is in WebProcess so we let WebProcess do the CORS check on the cached response.
189 if (response.httpStatusCode() == 304)
190 return { };
191
192 auto result = passesAccessControlCheck(response, m_storedCredentialsPolicy, *m_origin, m_networkResourceLoader.get());
193 if (!result)
194 return ResourceError { String { }, 0, m_url, WTFMove(result.error()), ResourceError::Type::AccessControl };
195
196 response.setTainting(ResourceResponse::Tainting::Cors);
197 return { };
198}

202 return ResourceError { String { }, 0, m_url, WTFMove(message), ResourceError::Type::AccessControl };

205void NetworkLoadChecker::checkRequest(ResourceRequest&& request, ContentSecurityPolicyClient* client, ValidationHandler&& handler)
206{
207 ResourceRequest originalRequest = request;
208
209 if (auto* contentSecurityPolicy = this->contentSecurityPolicy()) {
210 if (this->isRedirected()) {
211 auto type = m_options.mode == FetchOptions::Mode::Navigate ? ContentSecurityPolicy::InsecureRequestType::Navigation : ContentSecurityPolicy::InsecureRequestType::Load;
212 contentSecurityPolicy->upgradeInsecureRequestIfNeeded(request, type);
213 }
214 if (!this->isAllowedByContentSecurityPolicy(request, client)) {
215 handler(this->accessControlErrorForValidationHandler("Blocked by Content Security Policy."_s));
216 return;
217 }
218 }
219
220#if ENABLE(CONTENT_EXTENSIONS)
221 this->processContentRuleListsForLoad(WTFMove(request), [weakThis = makeWeakPtr(*this), handler = WTFMove(handler), originalRequest = WTFMove(originalRequest)](auto&& result) mutable {
222 if (!result.has_value()) {
223 ASSERT(result.error().isCancellation());
224 handler(WTFMove(result.error()));
225 return;
226 }
227 if (!weakThis)
228 return handler({ ResourceError { ResourceError::Type::Cancellation }});
229 if (result.value().results.summary.blockedLoad) {
230 handler(weakThis->accessControlErrorForValidationHandler("Blocked by content extension"_s));
231 return;
232 }
233 weakThis->continueCheckingRequestOrDoSyntheticRedirect(WTFMove(originalRequest), WTFMove(result.value().request), WTFMove(handler));
234 });
235#else
236 this->continueCheckingRequestOrDoSyntheticRedirect(WTFMove(originalRequest), WTFMove(request), WTFMove(handler));
237#endif
238}

240void NetworkLoadChecker::continueCheckingRequestOrDoSyntheticRedirect(ResourceRequest&& originalRequest, ResourceRequest&& currentRequest, ValidationHandler&& handler)
241{
242 // If main frame load and request has been modified, trigger a synthetic redirect.
243 if (m_requestLoadType == LoadType::MainFrame && currentRequest.url() != originalRequest.url()) {
244 ResourceResponse redirectResponse = ResourceResponse::syntheticRedirectResponse(originalRequest.url(), currentRequest.url());
245 handler(RedirectionTriplet { WTFMove(originalRequest), WTFMove(currentRequest), WTFMove(redirectResponse) });
246 return;
247 }
248 this->continueCheckingRequest(WTFMove(currentRequest), WTFMove(handler));
249}

251bool NetworkLoadChecker::isAllowedByContentSecurityPolicy(const ResourceRequest& request, WebCore::ContentSecurityPolicyClient* client)
252{
253 auto* contentSecurityPolicy = this->contentSecurityPolicy();
254 contentSecurityPolicy->setClient(client);
255 auto clearContentSecurityPolicyClient = makeScopeExit([&] {
256 contentSecurityPolicy->setClient(nullptr);
257 });
258
259 auto redirectResponseReceived = isRedirected() ? ContentSecurityPolicy::RedirectResponseReceived::Yes : ContentSecurityPolicy::RedirectResponseReceived::No;
260 switch (m_options.destination) {
261 case FetchOptions::Destination::Audioworklet:
262 case FetchOptions::Destination::Paintworklet:
263 case FetchOptions::Destination::Worker:
264 case FetchOptions::Destination::Serviceworker:
265 case FetchOptions::Destination::Sharedworker:
266 return contentSecurityPolicy->allowChildContextFromSource(request.url(), redirectResponseReceived);
267 case FetchOptions::Destination::Script:
268 if (request.requester() == ResourceRequest::Requester::ImportScripts && !contentSecurityPolicy->allowScriptFromSource(request.url(), redirectResponseReceived))
269 return false;
270 // FIXME: Check CSP for non-importScripts() initiated loads.
271 return true;
272 case FetchOptions::Destination::EmptyString:
273 return contentSecurityPolicy->allowConnectToSource(request.url(), redirectResponseReceived);
274 case FetchOptions::Destination::Audio:
275 case FetchOptions::Destination::Document:
276 case FetchOptions::Destination::Embed:
277 case FetchOptions::Destination::Font:
278 case FetchOptions::Destination::Image:
279 case FetchOptions::Destination::Manifest:
280 case FetchOptions::Destination::Model:
281 case FetchOptions::Destination::Object:
282 case FetchOptions::Destination::Report:
283 case FetchOptions::Destination::Style:
284 case FetchOptions::Destination::Track:
285 case FetchOptions::Destination::Video:
286 case FetchOptions::Destination::Xslt:
287 // FIXME: Check CSP for these destinations.
288 return true;
289 }
290 ASSERT_NOT_REACHED();
291 return true;
292}

294void NetworkLoadChecker::continueCheckingRequest(ResourceRequest&& request, ValidationHandler&& handler)
295{
296 if (m_options.credentials == FetchOptions::Credentials::SameOrigin)
297 m_storedCredentialsPolicy = m_isSameOriginRequest && m_origin->canRequest(request.url()) ? StoredCredentialsPolicy::Use : StoredCredentialsPolicy::DoNotUse;
298
299 m_isSameOriginRequest = m_isSameOriginRequest && isSameOrigin(request.url(), m_origin.get());
300
301 if (doesNotNeedCORSCheck(request.url())) {
302 handler(WTFMove(request));
303 return;
304 }
305
306 if (m_options.mode == FetchOptions::Mode::SameOrigin) {
307 handler(accessControlErrorForValidationHandler(makeString("Unsafe attempt to load URL ", request.url().stringCenterEllipsizedToLength(), " from origin ", m_origin->toString(), ". Domains, protocols and ports must match.\n")));
308 return;
309 }
310
311 if (isRedirected()) {
312 LOAD_CHECKER_RELEASE_LOG("checkRequest - Redirect requires CORS checks");
313 checkCORSRedirectedRequest(WTFMove(request), WTFMove(handler));
314 return;
315 }
316
317 checkCORSRequest(WTFMove(request), WTFMove(handler));
318}

320void NetworkLoadChecker::checkCORSRequest(ResourceRequest&& request, ValidationHandler&& handler)
321{
322 ASSERT(m_options.mode == FetchOptions::Mode::Cors);
323
324 // Except in case where preflight is needed, loading should be able to continue on its own.
325 switch (m_preflightPolicy) {
326 case PreflightPolicy::Force:
327 checkCORSRequestWithPreflight(WTFMove(request), WTFMove(handler));
328 break;
329 case PreflightPolicy::Consider:
330 if (!m_isSimpleRequest || !isSimpleCrossOriginAccessRequest(request.httpMethod(), m_originalRequestHeaders)) {
331 checkCORSRequestWithPreflight(WTFMove(request), WTFMove(handler));
332 return;
333 }
334 FALLTHROUGH;
335 case PreflightPolicy::Prevent:
336 updateRequestForAccessControl(request, *m_origin, m_storedCredentialsPolicy);
337 handler(WTFMove(request));
338 break;
339 }
340}

342void NetworkLoadChecker::checkCORSRedirectedRequest(ResourceRequest&& request, ValidationHandler&& handler)
343{
344 ASSERT(m_options.mode == FetchOptions::Mode::Cors);
345 ASSERT(isRedirected());
346
347 // Force any subsequent request to use these checks.
348 m_isSameOriginRequest = false;
349
350 if (!m_origin->canRequest(m_previousURL) && !protocolHostAndPortAreEqual(m_previousURL, request.url())) {
351 // Use a unique origin for subsequent loads if needed.
352 // https://fetch.spec.whatwg.org/#concept-http-redirect-fetch (Step 10).
353 if (!m_origin || !m_origin->isUnique())
354 m_origin = SecurityOrigin::createUnique();
355 }
356
357 // FIXME: We should set the request referrer according the referrer policy.
358
359 // Let's fetch the request with the original headers (equivalent to request cloning specified by fetch algorithm).
360 if (!request.httpHeaderFields().contains(HTTPHeaderName::Authorization))
361 m_firstRequestHeaders.remove(HTTPHeaderName::Authorization);
362 request.setHTTPHeaderFields(m_firstRequestHeaders);
363
364 checkCORSRequest(WTFMove(request), WTFMove(handler));
365}

367void NetworkLoadChecker::checkCORSRequestWithPreflight(ResourceRequest&& request, ValidationHandler&& handler)
368{
369 ASSERT(m_options.mode == FetchOptions::Mode::Cors);
370
371 m_isSimpleRequest = false;
372 if (CrossOriginPreflightResultCache::singleton().canSkipPreflight(m_sessionID, m_origin->toString(), request.url(), m_storedCredentialsPolicy, request.httpMethod(), m_originalRequestHeaders)) {
373 LOAD_CHECKER_RELEASE_LOG("checkCORSRequestWithPreflight - preflight can be skipped thanks to cached result");
374 updateRequestForAccessControl(request, *m_origin, m_storedCredentialsPolicy);
375 handler(WTFMove(request));
376 return;
377 }
378
379 auto requestForPreflight = request;
380 // We need to set header fields to m_originalRequestHeaders to correctly compute Access-Control-Request-Headers header value.
381 requestForPreflight.setHTTPHeaderFields(m_originalRequestHeaders);
382 NetworkCORSPreflightChecker::Parameters parameters = {
383 WTFMove(requestForPreflight),
384 *m_origin,
385 m_topOrigin,
386 request.httpReferrer(),
387 request.httpUserAgent(),
388 m_sessionID,
389 m_webPageProxyID,
390 m_storedCredentialsPolicy
391 };
392 m_corsPreflightChecker = makeUnique<NetworkCORSPreflightChecker>(m_networkProcess.get(), m_networkResourceLoader.get(), WTFMove(parameters), m_shouldCaptureExtraNetworkLoadMetrics, [this, request = WTFMove(request), handler = WTFMove(handler), isRedirected = isRedirected()](auto&& error) mutable {
393 LOAD_CHECKER_RELEASE_LOG("checkCORSRequestWithPreflight - makeCrossOriginAccessRequestWithPreflight preflight complete, success=%d forRedirect=%d", error.isNull(), isRedirected);
394
395 if (!error.isNull()) {
396 handler(WTFMove(error));
397 return;
398 }
399
400 if (m_shouldCaptureExtraNetworkLoadMetrics)
401 m_loadInformation.transactions.append(m_corsPreflightChecker->takeInformation());
402
403 auto corsPreflightChecker = WTFMove(m_corsPreflightChecker);
404 updateRequestForAccessControl(request, *m_origin, m_storedCredentialsPolicy);
405 handler(WTFMove(request));
406 });
407 m_corsPreflightChecker->startPreflight();
408}

410bool NetworkLoadChecker::doesNotNeedCORSCheck(const URL& url) const
411{
412 if (m_options.mode == FetchOptions::Mode::NoCors || m_options.mode == FetchOptions::Mode::Navigate)
413 return true;
414
415 if (m_schemeRegistry && !m_schemeRegistry->shouldTreatURLSchemeAsCORSEnabled(url.protocol()))
416 return true;
417
418 return m_isSameOriginRequest;
419}

421ContentSecurityPolicy* NetworkLoadChecker::contentSecurityPolicy()
422{
423 if (!m_contentSecurityPolicy && m_cspResponseHeaders) {
424 // FIXME: Pass the URL of the protected resource instead of its origin.
425 m_contentSecurityPolicy = makeUnique<ContentSecurityPolicy>(URL { URL { }, m_origin->toRawString() });
426 m_contentSecurityPolicy->didReceiveHeaders(*m_cspResponseHeaders, String { m_referrer }, ContentSecurityPolicy::ReportParsingErrors::No);
427 if (!m_documentURL.isEmpty())
428 m_contentSecurityPolicy->setDocumentURL(m_documentURL);
429 }
430 return m_contentSecurityPolicy.get();
431}

433#if ENABLE(CONTENT_EXTENSIONS)
434void NetworkLoadChecker::processContentRuleListsForLoad(ResourceRequest&& request, ContentExtensionCallback&& callback)
435{
436 // FIXME: Enable content blockers for navigation loads.
437 if (!m_checkContentExtensions || !m_userContentControllerIdentifier || m_options.mode == FetchOptions::Mode::Navigate) {
438 ContentRuleListResults results;
439 callback(ContentExtensionResult { WTFMove(request), results });
440 return;
441 }
442
443 m_networkProcess->networkContentRuleListManager().contentExtensionsBackend(*m_userContentControllerIdentifier, [this, weakThis = makeWeakPtr(this), request = WTFMove(request), callback = WTFMove(callback)](auto& backend) mutable {
444 if (!weakThis) {
445 callback(makeUnexpected(ResourceError { ResourceError::Type::Cancellation }));
446 return;
447 }
448
449 auto results = backend.processContentRuleListsForPingLoad(request.url(), m_mainDocumentURL);
450 WebCore::ContentExtensions::applyResultsToRequest(ContentRuleListResults { results }, nullptr, request);
451 callback(ContentExtensionResult { WTFMove(request), results });
452 });
453}
