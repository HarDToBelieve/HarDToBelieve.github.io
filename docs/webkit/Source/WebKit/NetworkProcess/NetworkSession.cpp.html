80NetworkSession::NetworkSession(NetworkProcess& networkProcess, const NetworkSessionCreationParameters& parameters)
81 : m_sessionID(parameters.sessionID)
82 , m_networkProcess(networkProcess)
83#if ENABLE(RESOURCE_LOAD_STATISTICS)
84 , m_resourceLoadStatisticsDirectory(parameters.resourceLoadStatisticsParameters.directory)
85 , m_shouldIncludeLocalhostInResourceLoadStatistics(parameters.resourceLoadStatisticsParameters.shouldIncludeLocalhost ? ShouldIncludeLocalhost::Yes : ShouldIncludeLocalhost::No)
86 , m_enableResourceLoadStatisticsDebugMode(parameters.resourceLoadStatisticsParameters.enableDebugMode ? EnableResourceLoadStatisticsDebugMode::Yes : EnableResourceLoadStatisticsDebugMode::No)
87 , m_resourceLoadStatisticsManualPrevalentResource(parameters.resourceLoadStatisticsParameters.manualPrevalentResource)
88 , m_enableResourceLoadStatisticsLogTestingEvent(parameters.resourceLoadStatisticsParameters.enableLogTestingEvent)
89 , m_thirdPartyCookieBlockingMode(parameters.resourceLoadStatisticsParameters.thirdPartyCookieBlockingMode)
90 , m_sameSiteStrictEnforcementEnabled(parameters.resourceLoadStatisticsParameters.sameSiteStrictEnforcementEnabled)
91 , m_firstPartyWebsiteDataRemovalMode(parameters.resourceLoadStatisticsParameters.firstPartyWebsiteDataRemovalMode)
92 , m_standaloneApplicationDomain(parameters.resourceLoadStatisticsParameters.standaloneApplicationDomain)
93#endif
94 , m_testSpeedMultiplier(parameters.testSpeedMultiplier)
95 , m_allowsServerPreconnect(parameters.allowsServerPreconnect)
96{
97 if (!m_sessionID.isEphemeral()) {
98 String networkCacheDirectory = parameters.networkCacheDirectory;
99 if (!networkCacheDirectory.isNull()) {
100 SandboxExtension::consumePermanently(parameters.networkCacheDirectoryExtensionHandle);
101
102 auto cacheOptions = networkProcess.cacheOptions();
103#if ENABLE(NETWORK_CACHE_SPECULATIVE_REVALIDATION)
104 if (parameters.networkCacheSpeculativeValidationEnabled)
105 cacheOptions.add(NetworkCache::CacheOption::SpeculativeRevalidation);
106#endif
107 if (parameters.shouldUseTestingNetworkSession)
108 cacheOptions.add(NetworkCache::CacheOption::TestingMode);
109
110 m_cache = NetworkCache::Cache::open(networkProcess, networkCacheDirectory, cacheOptions, m_sessionID);
111
112 if (!m_cache)
113 RELEASE_LOG_ERROR(NetworkCache, "Failed to initialize the WebKit network disk cache");
114 }
115
116 if (!parameters.resourceLoadStatisticsParameters.directory.isEmpty())
117 SandboxExtension::consumePermanently(parameters.resourceLoadStatisticsParameters.directoryExtensionHandle);
118 }
119
120 m_isStaleWhileRevalidateEnabled = parameters.staleWhileRevalidateEnabled;
121
122#if ENABLE(RESOURCE_LOAD_STATISTICS)
123 setResourceLoadStatisticsEnabled(parameters.resourceLoadStatisticsParameters.enabled);
124 m_privateClickMeasurement = makeUnique<PrivateClickMeasurementManager>(*this, networkProcess, parameters.sessionID, [weakThis = makeWeakPtr(this)] (auto&& loadParameters, auto&& completionHandler) {
125 if (!weakThis)
126 return;
127
128 PrivateClickMeasurementNetworkLoader::start(*weakThis, WTFMove(loadParameters), WTFMove(completionHandler));
129 });
130#endif
131}

164#if ENABLE(RESOURCE_LOAD_STATISTICS)
165void NetworkSession::setResourceLoadStatisticsEnabled(bool enable)
166{
167 ASSERT(!m_isInvalidated);
168 if (auto* storageSession = networkStorageSession())
169 storageSession->setResourceLoadStatisticsEnabled(enable);
170 if (!enable) {
171 destroyResourceLoadStatistics([] { });
172 return;
173 }
174
175 if (m_resourceLoadStatistics)
176 return;
177
178 m_resourceLoadStatistics = WebResourceLoadStatisticsStore::create(*this, m_resourceLoadStatisticsDirectory, m_shouldIncludeLocalhostInResourceLoadStatistics, (m_sessionID.isEphemeral() ? ResourceLoadStatistics::IsEphemeral::Yes : ResourceLoadStatistics::IsEphemeral::No));
179 if (!m_sessionID.isEphemeral())
180 m_resourceLoadStatistics->populateMemoryStoreFromDisk([] { });
181
182 if (m_enableResourceLoadStatisticsDebugMode == EnableResourceLoadStatisticsDebugMode::Yes)
183 m_resourceLoadStatistics->setResourceLoadStatisticsDebugMode(true, [] { });
184 // This should always be forwarded since debug mode may be enabled at runtime.
185 if (!m_resourceLoadStatisticsManualPrevalentResource.isEmpty())
186 m_resourceLoadStatistics->setPrevalentResourceForDebugMode(m_resourceLoadStatisticsManualPrevalentResource, [] { });
187 forwardResourceLoadStatisticsSettings();
188}
189
190void NetworkSession::recreateResourceLoadStatisticStore(CompletionHandler<void()>&& completionHandler)
191{
192 destroyResourceLoadStatistics([this, weakThis = makeWeakPtr(*this), completionHandler = WTFMove(completionHandler)] () mutable {
193 if (!weakThis)
194 return completionHandler();
195 m_resourceLoadStatistics = WebResourceLoadStatisticsStore::create(*this, m_resourceLoadStatisticsDirectory, m_shouldIncludeLocalhostInResourceLoadStatistics, (m_sessionID.isEphemeral() ? ResourceLoadStatistics::IsEphemeral::Yes : ResourceLoadStatistics::IsEphemeral::No));
196 forwardResourceLoadStatisticsSettings();
197 if (!m_sessionID.isEphemeral())
198 m_resourceLoadStatistics->populateMemoryStoreFromDisk(WTFMove(completionHandler));
199 else
200 completionHandler();
201 });
202}

269void NetworkSession::setFirstPartyHostCNAMEDomain(String&& firstPartyHost, WebCore::RegistrableDomain&& cnameDomain)
270{
271#if HAVE(CFNETWORK_CNAME_AND_COOKIE_TRANSFORM_SPI)
272 ASSERT(!firstPartyHost.isEmpty() && !cnameDomain.isEmpty() && firstPartyHost != cnameDomain.string());
273 if (firstPartyHost.isEmpty() || cnameDomain.isEmpty() || firstPartyHost == cnameDomain.string())
274 return;
275 m_firstPartyHostCNAMEDomains.add(WTFMove(firstPartyHost), WTFMove(cnameDomain));
276#else
277 UNUSED_PARAM(firstPartyHost);
278 UNUSED_PARAM(cnameDomain);
279#endif
280}
281
282std::optional<WebCore::RegistrableDomain> NetworkSession::firstPartyHostCNAMEDomain(const String& firstPartyHost)
283{
284#if HAVE(CFNETWORK_CNAME_AND_COOKIE_TRANSFORM_SPI)
285 if (!decltype(m_firstPartyHostCNAMEDomains)::isValidKey(firstPartyHost))
286 return std::nullopt;
287
288 auto iterator = m_firstPartyHostCNAMEDomains.find(firstPartyHost);
289 if (iterator == m_firstPartyHostCNAMEDomains.end())
290 return std::nullopt;
291 return iterator->value;
292#else
293 UNUSED_PARAM(firstPartyHost);
294 return std::nullopt;
295#endif
296}
