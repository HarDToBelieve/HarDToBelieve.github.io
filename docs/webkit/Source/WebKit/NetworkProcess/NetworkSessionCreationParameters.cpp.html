41void NetworkSessionCreationParameters::encode(IPC::Encoder& encoder) const
42{
43 encoder << sessionID;
44 encoder << boundInterfaceIdentifier;
45 encoder << allowsCellularAccess;
46#if PLATFORM(COCOA)
47 encoder << proxyConfiguration;
48 encoder << sourceApplicationBundleIdentifier;
49 encoder << sourceApplicationSecondaryIdentifier;
50 encoder << shouldLogCookieInformation;
51 encoder << httpProxy;
52 encoder << httpsProxy;
53#endif
54#if HAVE(CFNETWORK_ALTERNATIVE_SERVICE)
55 encoder << alternativeServiceDirectory;
56 encoder << alternativeServiceDirectoryExtensionHandle;
57 encoder << http3Enabled;
58#endif
59 encoder << hstsStorageDirectory;
60 encoder << hstsStorageDirectoryExtensionHandle;
61#if USE(SOUP)
62 encoder << cookiePersistentStoragePath;
63 encoder << cookiePersistentStorageType;
64 encoder << persistentCredentialStorageEnabled;
65 encoder << ignoreTLSErrors;
66 encoder << proxySettings;
67#endif
68#if USE(CURL)
69 encoder << cookiePersistentStorageFile;
70 encoder << proxySettings;
71#endif
72 encoder << networkCacheDirectory << networkCacheDirectoryExtensionHandle;
73
74 encoder << deviceManagementRestrictionsEnabled;
75 encoder << allLoadsBlockedByDeviceManagementRestrictionsForTesting;
76 encoder << dataConnectionServiceType;
77 encoder << fastServerTrustEvaluationEnabled;
78 encoder << networkCacheSpeculativeValidationEnabled;
79 encoder << shouldUseTestingNetworkSession;
80 encoder << staleWhileRevalidateEnabled;
81 encoder << testSpeedMultiplier;
82 encoder << suppressesConnectionTerminationOnSystemChange;
83 encoder << allowsServerPreconnect;
84 encoder << requiresSecureHTTPSProxyConnection;
85 encoder << preventsSystemHTTPProxyAuthentication;
86 encoder << appHasRequestedCrossWebsiteTrackingPermission;
87 encoder << useNetworkLoader;
88 encoder << resourceLoadStatisticsParameters;
89}

91std::optional<NetworkSessionCreationParameters> NetworkSessionCreationParameters::decode(IPC::Decoder& decoder)
92{
93 std::optional<PAL::SessionID> sessionID;
94 decoder >> sessionID;

100 if (!boundInterfaceIdentifier)

104 decoder >> allowsCellularAccess;
105 if (!allowsCellularAccess)
106 return std::nullopt;
107
108#if PLATFORM(COCOA)

110 if (!decoder.decode(proxyConfiguration))
111 return std::nullopt;

113 std::optional<String> sourceApplicationBundleIdentifier;

117

121 return std::nullopt;

131 return std::nullopt;
132
133 std::optional<URL> httpsProxy;
134 decoder >> httpsProxy;
135 if (!httpsProxy)
136 return std::nullopt;
137#endif
138
139#if HAVE(CFNETWORK_ALTERNATIVE_SERVICE)

141 decoder >> alternativeServiceDirectory;

143 return std::nullopt;

145 std::optional<SandboxExtension::Handle> alternativeServiceDirectoryExtensionHandle;
