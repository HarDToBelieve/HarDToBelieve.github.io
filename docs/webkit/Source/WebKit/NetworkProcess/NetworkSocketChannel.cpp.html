39std::unique_ptr<NetworkSocketChannel> NetworkSocketChannel::create(NetworkConnectionToWebProcess& connection, PAL::SessionID sessionID, const ResourceRequest& request, const String& protocol, WebSocketIdentifier identifier, WebPageProxyIdentifier webPageProxyID)
40{
41 auto result = makeUnique<NetworkSocketChannel>(connection, connection.networkProcess().networkSession(sessionID), request, protocol, identifier, webPageProxyID);
42 if (!result->m_socket) {
43 result->didClose(0, "Cannot create a web socket task"_s);
44 return nullptr;
45 }
46 return result;
47}
48
49NetworkSocketChannel::NetworkSocketChannel(NetworkConnectionToWebProcess& connection, NetworkSession* session, const ResourceRequest& request, const String& protocol, WebSocketIdentifier identifier, WebPageProxyIdentifier webPageProxyID)
50 : m_connectionToWebProcess(connection)
51 , m_identifier(identifier)
52 , m_session(makeWeakPtr(session))
53 , m_errorTimer(*this, &NetworkSocketChannel::sendDelayedError)
54 , m_webPageProxyID(webPageProxyID)
55{
56 if (!m_session)
57 return;
58
59 m_socket = m_session->createWebSocketTask(webPageProxyID, *this, request, protocol);
60 if (m_socket) {
61 m_session->addWebSocketTask(webPageProxyID, *m_socket);
62 m_socket->resume();
63 }
64}
65
66NetworkSocketChannel::~NetworkSocketChannel()
67{
68 if (m_socket) {
69 if (m_session && m_socket->sessionSet())
70 m_session->removeWebSocketTask(*m_socket->sessionSet(), *m_socket);
71 m_socket->cancel();
72 }
73}

85void NetworkSocketChannel::finishClosingIfPossible()
86{
87 if (m_state == State::Open) {
88 m_state = State::Closing;
89 return;
90 }
91 ASSERT(m_state == State::Closing);
92 m_state = State::Closed;
93 m_connectionToWebProcess.removeSocketChannel(m_identifier);
94}
95
96void NetworkSocketChannel::close(int32_t code, const String& reason)
97{
98 m_socket->close(code, reason);
99 finishClosingIfPossible();
100}

117void NetworkSocketChannel::didClose(unsigned short code, const String& reason)
118{
119 if (m_errorTimer.isActive()) {
120 m_closeInfo = std::make_pair(code, reason);
121 return;
122 }
123 send(Messages::WebSocketChannel::DidClose { code, reason });
124 finishClosingIfPossible();
125}
126
127void NetworkSocketChannel::didReceiveMessageError(const String& errorMessage)
128{
129 m_errorMessage = errorMessage;
130 m_errorTimer.startOneShot(NetworkProcess::randomClosedPortDelay());
131}
132
133void NetworkSocketChannel::sendDelayedError()
134{
135 send(Messages::WebSocketChannel::DidReceiveMessageError { m_errorMessage });
136 if (m_closeInfo) {
137 send(Messages::WebSocketChannel::DidClose { m_closeInfo->first, m_closeInfo->second });
138 finishClosingIfPossible();
139 }
140}
