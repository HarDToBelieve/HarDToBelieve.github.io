53PingLoad::PingLoad(NetworkConnectionToWebProcess& connection, NetworkResourceLoadParameters&& parameters, CompletionHandler<void(const ResourceError&, const ResourceResponse&)>&& completionHandler)
54 : m_sessionID(connection.sessionID())
55 , m_parameters(WTFMove(parameters))
56 , m_completionHandler(WTFMove(completionHandler))
57 , m_timeoutTimer(*this, &PingLoad::timeoutTimerFired)
58 , m_networkLoadChecker(makeUniqueRef<NetworkLoadChecker>(connection.networkProcess(), nullptr, &connection.schemeRegistry(), FetchOptions { m_parameters.options}, m_sessionID, m_parameters.webPageProxyID, WTFMove(m_parameters.originalRequestHeaders), URL { m_parameters.request.url() }, URL { m_parameters.documentURL }, m_parameters.sourceOrigin.copyRef(), m_parameters.topOrigin.copyRef(), m_parameters.preflightPolicy, m_parameters.request.httpReferrer()))
59 , m_blobFiles(connection.resolveBlobReferences(m_parameters))
60{
61 for (auto& file : m_blobFiles) {
62 if (file)
63 file->prepareForFileAccess();
64 }
65
66 initialize(connection.networkProcess());
67}

69void PingLoad::initialize(NetworkProcess& networkProcess)
70{
71 m_networkLoadChecker->enableContentExtensionsCheck();
72 if (m_parameters.cspResponseHeaders)
73 m_networkLoadChecker->setCSPResponseHeaders(WTFMove(m_parameters.cspResponseHeaders.value()));
74#if ENABLE(CONTENT_EXTENSIONS)
75 m_networkLoadChecker->setContentExtensionController(WTFMove(m_parameters.mainDocumentURL), m_parameters.userContentControllerIdentifier);
76#endif
77
78 // If the server never responds, this object will hang around forever.
79 // Set a very generous timeout, just in case.
80 m_timeoutTimer.startOneShot(60000_s);
81
82 m_networkLoadChecker->check(ResourceRequest { m_parameters.request }, nullptr, [this, weakThis = makeWeakPtr(*this), networkProcess = makeRef(networkProcess)] (auto&& result) {
83 if (!weakThis)
84 return;
85 WTF::switchOn(result,
86 [this] (ResourceError& error) {
87 this->didFinish(error);
88 },
89 [] (NetworkLoadChecker::RedirectionTriplet& triplet) {
90 // We should never send a synthetic redirect for PingLoads.
91 ASSERT_NOT_REACHED();
92 },
93 [&] (ResourceRequest& request) {
94 this->loadRequest(networkProcess, WTFMove(request));
95 }
96 );
97 });
98}

100PingLoad::~PingLoad()
101{
102 if (m_task) {
103 ASSERT(m_task->client() == this);
104 m_task->clearClient();
105 m_task->cancel();
106 }
107 for (auto& file : m_blobFiles) {
108 if (file)
109 file->revokeFileAccess();
110 }
111}

113void PingLoad::didFinish(const ResourceError& error, const ResourceResponse& response)
114{
115 m_completionHandler(error, response);
116 delete this;
117}

119void PingLoad::loadRequest(NetworkProcess& networkProcess, ResourceRequest&& request)
120{
121 PING_RELEASE_LOG("startNetworkLoad");
122 if (auto* networkSession = networkProcess.networkSession(m_sessionID)) {
123 auto loadParameters = m_parameters;
124 loadParameters.request = WTFMove(request);
125 m_task = NetworkDataTask::create(*networkSession, *this, WTFMove(loadParameters));
126 m_task->resume();
127 } else
128 ASSERT_NOT_REACHED();
129}

131void PingLoad::willPerformHTTPRedirection(ResourceResponse&& redirectResponse, ResourceRequest&& request, RedirectCompletionHandler&& completionHandler)
132{
133 m_networkLoadChecker->checkRedirection(ResourceRequest { }, WTFMove(request), WTFMove(redirectResponse), nullptr, [this, completionHandler = WTFMove(completionHandler)] (auto&& result) mutable {
134 if (!result.has_value()) {
135 completionHandler({ });
136 this->didFinish(result.error());
137 return;
138 }
139 auto request = WTFMove(result->redirectRequest);
140 if (!request.url().protocolIsInHTTPFamily()) {
141 this->didFinish(ResourceError { String { }, 0, request.url(), "Redirection to URL with a scheme that is not HTTP(S)"_s, ResourceError::Type::AccessControl });
142 return;
143 }
144
145 completionHandler(WTFMove(request));
146 });
147}

149void PingLoad::didReceiveChallenge(AuthenticationChallenge&& challenge, NegotiatedLegacyTLS negotiatedLegacyTLS, ChallengeCompletionHandler&& completionHandler)
150{
151 PING_RELEASE_LOG("didReceiveChallenge");
152 if (challenge.protectionSpace().authenticationScheme() == ProtectionSpaceAuthenticationSchemeServerTrustEvaluationRequested) {
153 m_networkLoadChecker->networkProcess().authenticationManager().didReceiveAuthenticationChallenge(m_sessionID, m_parameters.webPageProxyID, m_parameters.topOrigin ? &m_parameters.topOrigin->data() : nullptr, challenge, negotiatedLegacyTLS, WTFMove(completionHandler));
154 return;
155 }
156 auto weakThis = makeWeakPtr(*this);
157 completionHandler(AuthenticationChallengeDisposition::Cancel, { });
158 if (!weakThis)
159 return;
160 didFinish(ResourceError { String(), 0, currentURL(), "Failed HTTP authentication"_s, ResourceError::Type::AccessControl });
161}

163void PingLoad::didReceiveResponse(ResourceResponse&& response, NegotiatedLegacyTLS, ResponseCompletionHandler&& completionHandler)
164{
165 PING_RELEASE_LOG("didReceiveResponse - httpStatusCode=%d", response.httpStatusCode());
166 auto weakThis = makeWeakPtr(*this);
167 completionHandler(PolicyAction::Ignore);
168 if (!weakThis)
169 return;
170 didFinish({ }, response);
171}
