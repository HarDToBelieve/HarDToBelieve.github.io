40void PrivateClickMeasurementNetworkLoader::start(NetworkSession& session, NetworkLoadParameters&& parameters, Callback&& completionHandler)
41{
42 auto loader = std::unique_ptr<PrivateClickMeasurementNetworkLoader>(new PrivateClickMeasurementNetworkLoader(session, WTFMove(parameters), WTFMove(completionHandler)));
43 session.addPrivateClickMeasurementNetworkLoader(WTFMove(loader));
44}
45
46PrivateClickMeasurementNetworkLoader::PrivateClickMeasurementNetworkLoader(NetworkSession& session, NetworkLoadParameters&& parameters, Callback&& completionHandler)
47 : m_session(makeWeakPtr(session))
48 , m_completionHandler(WTFMove(completionHandler))
49{
50 m_networkLoad = makeUnique<NetworkLoad>(*this, nullptr, WTFMove(parameters), *m_session);
51 m_networkLoad->start();
52}
53
54void PrivateClickMeasurementNetworkLoader::fail(ResourceError&& error)
55{
56 if (!m_completionHandler)
57 return;
58
59 m_completionHandler(WTFMove(error), { }, nullptr);
60 didComplete();
61}
62
63void PrivateClickMeasurementNetworkLoader::willSendRedirectedRequest(ResourceRequest&&, ResourceRequest&&, ResourceResponse&&)
64{
65 fail(ResourceError { ResourceError::Type::Cancellation });
66}
67
68void PrivateClickMeasurementNetworkLoader::didReceiveResponse(ResourceResponse&& response, ResponseCompletionHandler&& completionHandler)
69{
70 if (!MIMETypeRegistry::isSupportedJSONMIMEType(response.mimeType())) {
71 fail({ errorDomainWebKitInternal, 0, response.url(), "MIME Type is not a JSON MIME type"_s });
72 completionHandler(PolicyAction::Ignore);
73 return;
74 }
75
76 m_response = WTFMove(response);
77 completionHandler(PolicyAction::Use);
78}
79
80void PrivateClickMeasurementNetworkLoader::didReceiveBuffer(Ref<SharedBuffer>&& buffer, int reportedEncodedDataLength)
81{
82 if (!m_decoder)
83 m_decoder = TextResourceDecoder::create("application/json"_s, m_response.textEncodingName().isEmpty() ? WebCore::TextEncoding("UTF-8") : WebCore::TextEncoding(m_response.textEncodingName()));
84
85 if (auto size = buffer->size())
86 m_jsonString.append(m_decoder->decode(buffer->data(), size));
87}
88
89void PrivateClickMeasurementNetworkLoader::didFinishLoading(const WebCore::NetworkLoadMetrics&)
90{
91 if (m_decoder)
92 m_jsonString.append(m_decoder->flush());
93
94 if (auto jsonValue = JSON::Value::parseJSON(m_jsonString.toString()))
95 m_completionHandler({ }, m_response, jsonValue->asObject());
96 else
97 m_completionHandler({ }, m_response, nullptr);
98
99 didComplete();
100}
101
102void PrivateClickMeasurementNetworkLoader::didFailLoading(const ResourceError& error)
103{
104 fail(ResourceError(error));
105}
106
107void PrivateClickMeasurementNetworkLoader::didComplete()
108{
109 m_networkLoad = nullptr;
110 if (m_session)
111 m_session->removePrivateClickMeasurementNetworkLoader(this);
112}
