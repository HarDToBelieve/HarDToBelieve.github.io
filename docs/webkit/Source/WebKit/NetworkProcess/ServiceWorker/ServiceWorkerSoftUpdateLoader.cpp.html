44void ServiceWorkerSoftUpdateLoader::start(NetworkSession* session, ServiceWorkerJobData&& jobData, bool shouldRefreshCache, ResourceRequest&& request, Handler&& completionHandler)
45{
46 if (!session)
47 return completionHandler(serviceWorkerFetchError(jobData.identifier(), ServiceWorkerRegistrationKey { jobData.registrationKey() }, ResourceError { ResourceError::Type::Cancellation }));
48 auto loader = std::unique_ptr<ServiceWorkerSoftUpdateLoader>(new ServiceWorkerSoftUpdateLoader(*session, WTFMove(jobData), shouldRefreshCache, WTFMove(request), WTFMove(completionHandler)));
49 session->addSoftUpdateLoader(WTFMove(loader));
50}

52ServiceWorkerSoftUpdateLoader::ServiceWorkerSoftUpdateLoader(NetworkSession& session, ServiceWorkerJobData&& jobData, bool shouldRefreshCache, ResourceRequest&& request, Handler&& completionHandler)
53 : m_completionHandler(WTFMove(completionHandler))
54 , m_jobData(WTFMove(jobData))
55 , m_session(makeWeakPtr(session))
56{
57 ASSERT(!request.isConditional());
58
59 if (session.cache()) {
60 // We set cache policy to disable speculative loading/async revalidation from the cache.
61 request.setCachePolicy(ResourceRequestCachePolicy::ReturnCacheDataDontLoad);
62
63 session.cache()->retrieve(request, NetworkCache::GlobalFrameID { }, NavigatingToAppBoundDomain::No, [this, weakThis = makeWeakPtr(*this), request, shouldRefreshCache](auto&& entry, auto&&) mutable {
64 if (!weakThis)
65 return;
66 if (!m_session) {
67 fail(ResourceError { ResourceError::Type::Cancellation });
68 return;
69 }
70 if (!shouldRefreshCache && entry && !entry->needsValidation()) {
71 loadWithCacheEntry(*entry);
72 return;
73 }
74
75 request.setCachePolicy(ResourceRequestCachePolicy::RefreshAnyCacheData);
76 if (entry) {
77 m_cacheEntry = WTFMove(entry);
78
79 String eTag = m_cacheEntry->response().httpHeaderField(HTTPHeaderName::ETag);
80 if (!eTag.isEmpty())
81 request.setHTTPHeaderField(HTTPHeaderName::IfNoneMatch, eTag);
82
83 String lastModified = m_cacheEntry->response().httpHeaderField(HTTPHeaderName::LastModified);
84 if (!lastModified.isEmpty())
85 request.setHTTPHeaderField(HTTPHeaderName::IfModifiedSince, lastModified);
86 }
87 loadFromNetwork(*m_session, WTFMove(request));
88 });
89 return;
90 }
91 loadFromNetwork(session, WTFMove(request));
92}

94ServiceWorkerSoftUpdateLoader::~ServiceWorkerSoftUpdateLoader()
95{
96 if (m_completionHandler)
97 m_completionHandler(serviceWorkerFetchError(m_jobData.identifier(), ServiceWorkerRegistrationKey { m_jobData.registrationKey() }, ResourceError { ResourceError::Type::Cancellation }));
98}

100void ServiceWorkerSoftUpdateLoader::fail(ResourceError&& error)
101{
102 if (!m_completionHandler)
103 return;
104
105 m_completionHandler(serviceWorkerFetchError(m_jobData.identifier(), ServiceWorkerRegistrationKey { m_jobData.registrationKey() }, WTFMove(error)));
106 didComplete();
107}
108
109void ServiceWorkerSoftUpdateLoader::loadWithCacheEntry(NetworkCache::Entry& entry)
110{
111 auto error = processResponse(entry.response());
112 if (!error.isNull()) {
113 fail(WTFMove(error));
114 return;
115 }
116
117 if (entry.buffer())
118 didReceiveBuffer(makeRef(*entry.buffer()), 0);
119 didFinishLoading({ });
120}

122void ServiceWorkerSoftUpdateLoader::loadFromNetwork(NetworkSession& session, ResourceRequest&& request)
123{
124 NetworkLoadParameters parameters;
125 parameters.storedCredentialsPolicy = StoredCredentialsPolicy::Use;
126 parameters.contentSniffingPolicy = ContentSniffingPolicy::DoNotSniffContent;
127 parameters.contentEncodingSniffingPolicy = ContentEncodingSniffingPolicy::Sniff;
128 parameters.needsCertificateInfo = true;
129 parameters.request = WTFMove(request);
130 m_networkLoad = makeUnique<NetworkLoad>(*this, nullptr, WTFMove(parameters), session);
131 m_networkLoad->start();
132
133#if PLATFORM(COCOA)
134 session.appBoundNavigationTestingData().setDidPerformSoftUpdate();
135#endif
136}

143void ServiceWorkerSoftUpdateLoader::didReceiveResponse(ResourceResponse&& response, ResponseCompletionHandler&& completionHandler)
144{
145 m_certificateInfo = *response.certificateInfo();
146 if (response.httpStatusCode() == 304 && m_cacheEntry) {
147 loadWithCacheEntry(*m_cacheEntry);
148 completionHandler(PolicyAction::Ignore);
149 return;
150 }
151 auto error = processResponse(response);
152 if (!error.isNull()) {
153 fail(WTFMove(error));
154 completionHandler(PolicyAction::Ignore);
155 return;
156 }
157 completionHandler(PolicyAction::Use);
158}

160// https://w3c.github.io/ServiceWorker/#update-algorithm, steps 9.7 to 9.17
161ResourceError ServiceWorkerSoftUpdateLoader::processResponse(const ResourceResponse& response)
162{
163 auto error = WorkerScriptLoader::validateWorkerResponse(response, FetchOptions::Destination::Serviceworker);
164 if (!error.isNull())
165 return error;
166
167 error = ServiceWorkerJob::validateServiceWorkerResponse(m_jobData, response);
168 if (!error.isNull())
169 return error;
170
171 m_contentSecurityPolicy = ContentSecurityPolicyResponseHeaders { response };
172 m_referrerPolicy = response.httpHeaderField(HTTPHeaderName::ReferrerPolicy);
173 m_responseEncoding = response.textEncodingName();
174
175 return { };
176}
177
178void ServiceWorkerSoftUpdateLoader::didReceiveBuffer(Ref<SharedBuffer>&& buffer, int reportedEncodedDataLength)
179{
180 if (!m_decoder) {
181 if (!m_responseEncoding.isEmpty())
182 m_decoder = TextResourceDecoder::create("text/javascript"_s, m_responseEncoding);
183 else
184 m_decoder = TextResourceDecoder::create("text/javascript"_s, "UTF-8");
185 }
186
187 if (auto size = buffer->size())
188 m_script.append(m_decoder->decode(buffer->data(), size));
189}
190
191void ServiceWorkerSoftUpdateLoader::didFinishLoading(const WebCore::NetworkLoadMetrics&)
192{
193 if (m_decoder)
194 m_script.append(m_decoder->flush());
195 m_completionHandler({ m_jobData.identifier(), m_jobData.registrationKey(), ScriptBuffer { m_script.toString() }, m_certificateInfo, m_contentSecurityPolicy, m_referrerPolicy, { } });
196 didComplete();
197}
198
199void ServiceWorkerSoftUpdateLoader::didFailLoading(const ResourceError& error)
200{
201 fail(ResourceError(error));
202}
203
204void ServiceWorkerSoftUpdateLoader::didComplete()
205{
206 m_networkLoad = nullptr;
207 if (m_session)
208 m_session->removeSoftUpdateLoader(this);
209}
