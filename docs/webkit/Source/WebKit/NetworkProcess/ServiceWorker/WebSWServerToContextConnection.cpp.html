47WebSWServerToContextConnection::WebSWServerToContextConnection(NetworkConnectionToWebProcess& connection, WebPageProxyIdentifier webPageProxyID, RegistrableDomain&& registrableDomain, SWServer& server)
48 : SWServerToContextConnection(WTFMove(registrableDomain))
49 , m_connection(connection)
50 , m_server(makeWeakPtr(server))
51 , m_webPageProxyID(webPageProxyID)
52{
53 server.addContextConnection(*this);
54}
55
56WebSWServerToContextConnection::~WebSWServerToContextConnection()
57{
58 auto fetches = WTFMove(m_ongoingFetches);
59 for (auto& fetch : fetches.values())
60 fetch->contextClosed();
61
62 if (m_server && m_server->contextConnectionForRegistrableDomain(registrableDomain()) == this)
63 m_server->removeContextConnection(*this);
64}

81void WebSWServerToContextConnection::postMessageToServiceWorkerClient(const ServiceWorkerClientIdentifier& destinationIdentifier, const MessageWithMessagePorts& message, ServiceWorkerIdentifier sourceIdentifier, const String& sourceOrigin)
82{
83 if (!m_server)
84 return;
85
86 if (auto* connection = m_server->connection(destinationIdentifier.serverConnectionIdentifier))
87 connection->postMessageToServiceWorkerClient(destinationIdentifier.contextIdentifier, message, sourceIdentifier, sourceOrigin);
88}

115void WebSWServerToContextConnection::didSaveScriptsToDisk(ServiceWorkerIdentifier serviceWorkerIdentifier, const ScriptBuffer& script, const HashMap<URL, ScriptBuffer>& importedScripts)
116{
117#if ENABLE(SHAREABLE_RESOURCE) && PLATFORM(COCOA)
118 // Send file-mapped ScriptBuffers over to the ServiceWorker process so that it can replace its heap-allocated copies and save on dirty memory.
119 auto scriptToSend = script.containsSingleFileMappedSegment() ? script : ScriptBuffer();
120 HashMap<URL, ScriptBuffer> importedScriptsToSend;
121 for (auto& pair : importedScripts) {
122 if (pair.value.containsSingleFileMappedSegment())
123 importedScriptsToSend.add(pair.key, pair.value);
124 }
125 if (scriptToSend || !importedScriptsToSend.isEmpty())
126 send(Messages::WebSWContextManagerConnection::DidSaveScriptsToDisk { serviceWorkerIdentifier, scriptToSend, importedScriptsToSend });
127#else
128 UNUSED_PARAM(script);
129 UNUSED_PARAM(importedScripts);
130#endif
131}

164void WebSWServerToContextConnection::didReceiveFetchTaskMessage(IPC::Connection& connection, IPC::Decoder& decoder)
165{
166 auto iterator = m_ongoingFetches.find(makeObjectIdentifier<FetchIdentifierType>(decoder.destinationID()));
167 if (iterator == m_ongoingFetches.end())
168 return;
169
170 iterator->value->didReceiveMessage(connection, decoder);
171}
172
173void WebSWServerToContextConnection::registerFetch(ServiceWorkerFetchTask& task)
174{
175 ASSERT(!m_ongoingFetches.contains(task.fetchIdentifier()));
176 m_ongoingFetches.add(task.fetchIdentifier(), makeWeakPtr(task));
177}
178
179void WebSWServerToContextConnection::unregisterFetch(ServiceWorkerFetchTask& task)
180{
181 ASSERT(m_ongoingFetches.contains(task.fetchIdentifier()));
182 m_ongoingFetches.remove(task.fetchIdentifier());
183}
