42class LocalStorageDatabase : public RefCounted<LocalStorageDatabase>, public CanMakeWeakPtr<LocalStorageDatabase> {
43public:
44 static Ref<LocalStorageDatabase> create(Ref<WorkQueue>&&, String&& databasePath, unsigned quotaInBytes);
45 ~LocalStorageDatabase();
46
47 HashMap<String, String> items() const;
48 void openIfExisting();
49 void removeItem(const String& key, String& oldValue);
50 bool clear();
51 String item(const String& key) const;
52 void setItem(const String& key, const String& value, String& oldValue, bool& quotaException);
53
54 // Will block until all pending changes have been written to disk.
55 void close();
56
57 void flushToDisk();
58 void handleLowMemoryWarning();
59
60private:
61 LocalStorageDatabase(Ref<WorkQueue>&&, String&& databasePath, unsigned quotaInBytes);
62
63 enum class ShouldCreateDatabase : bool { No, Yes };
64 bool openDatabase(ShouldCreateDatabase);
65
66 void startTransactionIfNecessary();
67 bool migrateItemTableIfNeeded();
68 bool databaseIsEmpty() const;
69
70 String itemBypassingCache(const String& key) const;
71
72 WebCore::SQLiteStatementAutoResetScope scopedStatement(std::unique_ptr<WebCore::SQLiteStatement>&, ASCIILiteral query) const;
73
74 Ref<WorkQueue> m_workQueue;
75 String m_databasePath;
76 mutable WebCore::SQLiteDatabase m_database;
77 std::unique_ptr<WebCore::SQLiteTransaction> m_transaction;
78 const unsigned m_quotaInBytes { 0 };
79 bool m_isClosed { false };
80
81 // Cached version of the items in memory.
82 // If the value is too large to keep in memory, we store a null String.
83 mutable std::optional<HashMap<String, String>> m_items;
84
85 mutable std::unique_ptr<WebCore::SQLiteStatement> m_clearStatement;
86 mutable std::unique_ptr<WebCore::SQLiteStatement> m_insertStatement;
87 mutable std::unique_ptr<WebCore::SQLiteStatement> m_getItemStatement;
88 mutable std::unique_ptr<WebCore::SQLiteStatement> m_getItemsStatement;
89 mutable std::unique_ptr<WebCore::SQLiteStatement> m_deleteItemStatement;
90};
