85Vector<SecurityOriginData> LocalStorageDatabaseTracker::databasesModifiedSince(WallTime time)
86{
87 Vector<SecurityOriginData> databaseOriginsModified;
88 auto databaseOrigins = origins();
89
90 for (auto origin : databaseOrigins) {
91 auto path = databasePath(origin);
92
93 auto modificationTime = SQLiteFileSystem::databaseModificationTime(path);
94 if (!modificationTime)
95 continue;
96
97 if (modificationTime.value() >= time)
98 databaseOriginsModified.append(origin);
99 }
100
101 return databaseOriginsModified;
102}

104Vector<SecurityOriginData> LocalStorageDatabaseTracker::origins() const
105{
106 Vector<SecurityOriginData> databaseOrigins;
107 for (auto& path : FileSystem::listDirectory(localStorageDirectory())) {
108 auto filename = FileSystem::pathFileName(path);
109 auto originIdentifier = filename.substring(0, filename.length() - strlen(".localstorage"));
110 auto origin = SecurityOriginData::fromDatabaseIdentifier(originIdentifier);
111 if (origin)
112 databaseOrigins.append(origin.value());
113 else
114 RELEASE_LOG_ERROR(LocalStorageDatabaseTracker, "Unable to extract origin from path %s", path.utf8().data());
115 }
116
117 return databaseOrigins;
118}

120Vector<LocalStorageDatabaseTracker::OriginDetails> LocalStorageDatabaseTracker::originDetailsCrossThreadCopy()
121{
122 Vector<OriginDetails> result;
123 auto databaseOrigins = origins();
124 result.reserveInitialCapacity(databaseOrigins.size());
125
126 for (const auto& origin : databaseOrigins) {
127 String path = databasePath(origin);
128
129 OriginDetails details;
130 details.originIdentifier = crossThreadCopy(origin.databaseIdentifier());
131 details.creationTime = SQLiteFileSystem::databaseCreationTime(path);
132 details.modificationTime = SQLiteFileSystem::databaseModificationTime(path);
133 result.uncheckedAppend(WTFMove(details));
134 }
135
136 return result;
137}

139String LocalStorageDatabaseTracker::databasePath(const String& filename) const
140{
141 auto localStorageDirectory = this->localStorageDirectory();
142 if (!SQLiteFileSystem::ensureDatabaseDirectoryExists(localStorageDirectory)) {
143 if (!m_localStorageDirectory.isNull())
144 LOG_ERROR("Unable to create LocalStorage database path %s", m_localStorageDirectory.utf8().data());
145 return String();
146 }
147
148#if PLATFORM(IOS_FAMILY)
149 platformMaybeExcludeFromBackup();
150#endif
151
152 return SQLiteFileSystem::appendDatabaseFileNameToPath(localStorageDirectory, filename);
153}
