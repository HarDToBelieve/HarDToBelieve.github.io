37// We should investigate a way to share it with WebCore.
38LocalStorageNamespace::LocalStorageNamespace(StorageManager& storageManager, StorageNamespaceIdentifier storageNamespaceID)
39 : m_storageManager(storageManager)
40 , m_quotaInBytes(StorageManager::localStorageDatabaseQuotaInBytes)
41{
42 ASSERT(!RunLoop::isMain());
43}
44
45LocalStorageNamespace::~LocalStorageNamespace()
46{
47 ASSERT(!RunLoop::isMain());
48}
49
50StorageArea& LocalStorageNamespace::getOrCreateStorageArea(SecurityOriginData&& securityOrigin, IsEphemeral isEphemeral, Ref<WorkQueue>&& workQueue)
51{
52 ASSERT(!RunLoop::isMain());
53 return *m_storageAreaMap.ensure(securityOrigin, [&]() mutable {
54 return makeUnique<StorageArea>(isEphemeral == IsEphemeral::Yes ? nullptr : this, WTFMove(securityOrigin), m_quotaInBytes, WTFMove(workQueue));
55 }).iterator->value.get();
56}
57
58void LocalStorageNamespace::clearStorageAreasMatchingOrigin(const SecurityOriginData& securityOrigin)
59{
60 ASSERT(!RunLoop::isMain());
61 auto originAndStorageArea = m_storageAreaMap.find(securityOrigin);
62 if (originAndStorageArea != m_storageAreaMap.end())
63 originAndStorageArea->value->clear();
64}
65
66void LocalStorageNamespace::clearAllStorageAreas()
67{
68 ASSERT(!RunLoop::isMain());
69 for (auto& storageArea : m_storageAreaMap.values())
70 storageArea->clear();
71}
72
73void LocalStorageNamespace::flushAndClose(const SecurityOriginData& origin)
74{
75 ASSERT(!RunLoop::isMain());
76 if (auto* storageArea = m_storageAreaMap.get(origin))
77 storageArea->close();
78}
79
80
81void LocalStorageNamespace::removeStorageArea(const SecurityOriginData& origin)
82{
83 ASSERT(!RunLoop::isMain());
84 m_storageAreaMap.remove(origin);
85}
86
87Vector<SecurityOriginData> LocalStorageNamespace::ephemeralOrigins() const
88{
89 ASSERT(!RunLoop::isMain());
90 Vector<SecurityOriginData> origins;
91 for (const auto& storageArea : m_storageAreaMap.values()) {
92 if (!storageArea->items().isEmpty())
93 origins.append(storageArea->securityOrigin());
94 }
95 return origins;
96}
97
98Vector<StorageAreaIdentifier> LocalStorageNamespace::storageAreaIdentifiers() const
99{
100 Vector<StorageAreaIdentifier> identifiers;
101 for (auto& storageArea : m_storageAreaMap.values())
102 identifiers.append(storageArea->identifier());
103 return identifiers;
104}
