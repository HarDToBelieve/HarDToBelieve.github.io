59void StorageManagerSet::add(PAL::SessionID sessionID, const String& localStorageDirectory, SandboxExtension::Handle& localStorageDirectoryHandle)
60{
61 ASSERT(RunLoop::isMain());
62
63 if (m_storageManagerPaths.add(sessionID, localStorageDirectory).isNewEntry) {
64 if (!sessionID.isEphemeral())
65 SandboxExtension::consumePermanently(localStorageDirectoryHandle);
66
67 m_queue->dispatch([this, protectedThis = makeRef(*this), sessionID, localStorageDirectory = localStorageDirectory.isolatedCopy()]() mutable {
68 m_storageManagers.ensure(sessionID, [&]() mutable {
69 return makeUnique<StorageManager>(WTFMove(localStorageDirectory));
70 });
71 });
72 }
73}

75void StorageManagerSet::remove(PAL::SessionID sessionID)
76{
77 ASSERT(RunLoop::isMain());
78
79 if (m_storageManagerPaths.remove(sessionID)) {
80 m_queue->dispatch([this, protectedThis = makeRef(*this), sessionID]() {
81 if (auto storageManager = m_storageManagers.get(sessionID)) {
82 for (auto storageAreaID : storageManager->allStorageAreaIdentifiers())
83 m_storageAreas.remove(storageAreaID);
84
85 m_storageManagers.remove(sessionID);
86 }
87 });
88 }
89}

98void StorageManagerSet::addConnection(IPC::Connection& connection)
99{
100 ASSERT(RunLoop::isMain());
101
102 auto connectionID = connection.uniqueID();
103 ASSERT(!m_connections.contains(connectionID));
104 if (m_connections.add(connectionID).isNewEntry)
105 connection.addWorkQueueMessageReceiver(Messages::StorageManagerSet::messageReceiverName(), m_queue.get(), this);
106}

108void StorageManagerSet::removeConnection(IPC::Connection& connection)
109{
110 ASSERT(RunLoop::isMain());
111
112 auto connectionID = connection.uniqueID();
113 ASSERT(m_connections.contains(connectionID));
114 m_connections.remove(connectionID);
115 connection.removeWorkQueueMessageReceiver(Messages::StorageManagerSet::messageReceiverName());
116
117 m_queue->dispatch([this, protectedThis = makeRef(*this), connectionID]() {
118 Vector<StorageAreaIdentifier> identifiersToRemove;
119 for (auto& [identifier, storageArea] : m_storageAreas) {
120 if (storageArea)
121 storageArea->removeListener(connectionID);
122
123 if (!storageArea)
124 identifiersToRemove.append(identifier);
125 }
126
127 for (auto identifier : identifiersToRemove)
128 m_storageAreas.remove(identifier);
129 });
130}

141void StorageManagerSet::waitUntilTasksFinished()
142{
143 ASSERT(RunLoop::isMain());
144
145 m_queue->dispatchSync([this] {
146 for (auto& storageManager : m_storageManagers.values())
147 storageManager->clearStorageNamespaces();
148
149 m_storageManagers.clear();
150 m_storageAreas.clear();
151 });
152}

154void StorageManagerSet::waitUntilSyncingLocalStorageFinished()
155{
156 ASSERT(RunLoop::isMain());
157
158 BinarySemaphore semaphore;
159 m_queue->dispatch([this, &semaphore] {
160 flushLocalStorage();
161 semaphore.signal();
162 });
163 semaphore.wait();
164}

166void StorageManagerSet::flushLocalStorage()
167{
168 ASSERT(!RunLoop::isMain());
169 for (const auto& storageArea : m_storageAreas.values()) {
170 ASSERT(storageArea);
171 if (storageArea)
172 storageArea->syncToDatabase();
173 }
174}

176void StorageManagerSet::suspend(CompletionHandler<void()>&& completionHandler)
177{
178 ASSERT(RunLoop::isMain());
179
180 CompletionHandlerCallingScope completionHandlerCaller(WTFMove(completionHandler));
181 Locker stateLocker { m_stateLock };
182 if (m_state != State::Running)
183 return;
184 m_state = State::WillSuspend;
185
186 m_queue->dispatch([this, protectedThis = makeRef(*this), completionHandler = completionHandlerCaller.release()] () mutable {
187 Locker stateLocker { m_stateLock };
188 ASSERT(m_state != State::Suspended);
189
190 if (m_state != State::WillSuspend) {
191 RunLoop::main().dispatch(WTFMove(completionHandler));
192 return;
193 }
194
195 // Make sure we flush local storage to disk before we suspend the thread as we want to make sure any pending
196 // SQL transaction has been committed.
197 flushLocalStorage();
198
199 m_state = State::Suspended;
200 RunLoop::main().dispatch(WTFMove(completionHandler));
201
202 while (m_state == State::Suspended)
203 m_stateChangeCondition.wait(m_stateLock);
204 ASSERT(m_state == State::Running);
205 });
206}

208void StorageManagerSet::resume()
209{
210 ASSERT(RunLoop::isMain());
211
212 Locker stateLocker { m_stateLock };
213 auto previousState = m_state;
214 m_state = State::Running;
215 if (previousState == State::Suspended)
216 m_stateChangeCondition.notifyOne();
217}

328void StorageManagerSet::connectToLocalStorageArea(IPC::Connection& connection, PAL::SessionID sessionID, StorageNamespaceIdentifier storageNamespaceID, SecurityOriginData&& originData, ConnectToStorageAreaCallback&& completionHandler)
329{
330 ASSERT(!RunLoop::isMain());
331
332 auto* storageManager = m_storageManagers.get(sessionID);
333 if (!storageManager) {
334 completionHandler(std::nullopt);
335 return;
336 }
337
338 auto* storageArea = storageManager->createLocalStorageArea(storageNamespaceID, WTFMove(originData), m_queue.copyRef());
339 if (!storageArea) {
340 completionHandler(std::nullopt);
341 return;
342 }
343
344 auto storageAreaID = storageArea->identifier();
345 auto iter = m_storageAreas.add(storageAreaID, makeWeakPtr(storageArea)).iterator;
346 ASSERT_UNUSED(iter, storageArea == iter->value.get());
347
348 completionHandler(storageAreaID);
349
350 storageArea->addListener(connection.uniqueID());
351}

403void StorageManagerSet::disconnectFromStorageArea(IPC::Connection& connection, StorageAreaIdentifier storageAreaID)
404{
405 ASSERT(!RunLoop::isMain());
406
407 const auto& storageArea = m_storageAreas.get(storageAreaID);
408 ASSERT(storageArea);
409 ASSERT(storageArea->hasListener(connection.uniqueID()));
410
411 if (!storageArea)
412 return;
413
414 storageArea->removeListener(connection.uniqueID());
415 if (!storageArea)
416 m_storageAreas.remove(storageAreaID);
417}

429void StorageManagerSet::setItem(IPC::Connection& connection, StorageAreaIdentifier storageAreaID, StorageAreaImplIdentifier storageAreaImplID, uint64_t storageMapSeed, const String& key, const String& value, const String& urlString)
430{
431 ASSERT(!RunLoop::isMain());
432
433 const auto& storageArea = m_storageAreas.get(storageAreaID);
434 ASSERT(storageArea);
435
436 bool quotaError = false;
437 if (storageArea)
438 storageArea->setItem(connection.uniqueID(), storageAreaImplID, key, value, urlString, quotaError);

440 connection.send(Messages::StorageAreaMap::DidSetItem(storageMapSeed, key, quotaError), storageAreaID);
