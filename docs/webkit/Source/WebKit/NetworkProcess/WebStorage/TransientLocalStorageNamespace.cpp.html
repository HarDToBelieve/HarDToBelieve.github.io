36TransientLocalStorageNamespace::TransientLocalStorageNamespace()
37 : m_quotaInBytes(StorageManager::localStorageDatabaseQuotaInBytes)
38{
39 ASSERT(!RunLoop::isMain());
40}
41
42TransientLocalStorageNamespace::~TransientLocalStorageNamespace()
43{
44 ASSERT(!RunLoop::isMain());
45}
46
47StorageArea& TransientLocalStorageNamespace::getOrCreateStorageArea(SecurityOriginData&& securityOrigin, Ref<WorkQueue>&& workQueue)
48{
49 ASSERT(!RunLoop::isMain());
50 return *m_storageAreaMap.ensure(securityOrigin, [&]() mutable {
51 return makeUnique<StorageArea>(nullptr, WTFMove(securityOrigin), m_quotaInBytes, WTFMove(workQueue));
52 }).iterator->value.get();
53}
54
55Vector<SecurityOriginData> TransientLocalStorageNamespace::origins() const
56{
57 ASSERT(!RunLoop::isMain());
58 Vector<SecurityOriginData> origins;
59
60 for (const auto& storageArea : m_storageAreaMap.values()) {
61 if (!storageArea->items().isEmpty())
62 origins.append(storageArea->securityOrigin());
63 }
64
65 return origins;
66}
67
68void TransientLocalStorageNamespace::clearStorageAreasMatchingOrigin(const SecurityOriginData& securityOrigin)
69{
70 ASSERT(!RunLoop::isMain());
71 auto originAndStorageArea = m_storageAreaMap.find(securityOrigin);
72 if (originAndStorageArea != m_storageAreaMap.end())
73 originAndStorageArea->value->clear();
74}
75
76void TransientLocalStorageNamespace::clearAllStorageAreas()
77{
78 ASSERT(!RunLoop::isMain());
79 for (auto& storageArea : m_storageAreaMap.values())
80 storageArea->clear();
81}
82
83Vector<StorageAreaIdentifier> TransientLocalStorageNamespace::storageAreaIdentifiers() const
84{
85 Vector<StorageAreaIdentifier> identifiers;
86 for (auto& storageArea : m_storageAreaMap.values())
87 identifiers.append(storageArea->identifier());
88 return identifiers;
89}
