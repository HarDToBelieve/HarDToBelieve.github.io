72static float toNSURLSessionTaskPriority(WebCore::ResourceLoadPriority priority)
73{
74 switch (priority) {
75 case WebCore::ResourceLoadPriority::VeryLow:
76 return 0;
77 case WebCore::ResourceLoadPriority::Low:
78 return 0.25;
79 case WebCore::ResourceLoadPriority::Medium:
80 return 0.5;
81 case WebCore::ResourceLoadPriority::High:
82 return 0.75;
83 case WebCore::ResourceLoadPriority::VeryHigh:
84 return 1;
85 }
86
87 ASSERT_NOT_REACHED();
88 return NSURLSessionTaskPriorityDefault;
89}

91void NetworkDataTaskCocoa::applySniffingPoliciesAndBindRequestToInferfaceIfNeeded(RetainPtr<NSURLRequest>& nsRequest, bool shouldContentSniff, bool shouldContentEncodingSniff)
92{
93#if !USE(CFNETWORK_CONTENT_ENCODING_SNIFFING_OVERRIDE)
94 UNUSED_PARAM(shouldContentEncodingSniff);
95#endif
96
97 auto& cocoaSession = static_cast<NetworkSessionCocoa&>(*m_session);
98 auto& boundInterfaceIdentifier = cocoaSession.boundInterfaceIdentifier();
99 if (shouldContentSniff
100#if USE(CFNETWORK_CONTENT_ENCODING_SNIFFING_OVERRIDE)
101 && shouldContentEncodingSniff
102#endif
103 && boundInterfaceIdentifier.isNull())
104 return;
105
106 auto mutableRequest = adoptNS([nsRequest mutableCopy]);
107
108#if USE(CFNETWORK_CONTENT_ENCODING_SNIFFING_OVERRIDE)
109 if (!shouldContentEncodingSniff)
110 [mutableRequest _setProperty:@YES forKey:(NSString *)kCFURLRequestContentDecoderSkipURLCheck];
111#endif
112
113 if (!shouldContentSniff)
114 [mutableRequest _setProperty:@NO forKey:(NSString *)_kCFURLConnectionPropertyShouldSniff];
115
116 if (!boundInterfaceIdentifier.isNull())
117 [mutableRequest setBoundInterfaceIdentifier:boundInterfaceIdentifier];
118
119 nsRequest = mutableRequest.autorelease();
120}

149static WebCore::RegistrableDomain lastCNAMEDomain(NSArray<NSString *> *cnames)
150{
151 if (auto* lastResolvedCNAMEInChain = [cnames lastObject]) {
152 auto cname = String(lastResolvedCNAMEInChain);
153 if (cname.endsWith('.'))
154 cname.remove(cname.length() - 1);
155 return WebCore::RegistrableDomain::uncheckedCreateFromHost(cname);
156 }
157
158 return { };
159}

161void NetworkDataTaskCocoa::updateFirstPartyInfoForSession(const URL& requestURL)
162{
163 if (!hasCNAMEAndCookieTransformSPI(m_task.get()) || !networkSession() || !networkSession()->networkStorageSession() || !networkSession()->networkStorageSession()->resourceLoadStatisticsEnabled() || requestURL.host().isEmpty())
164 return;
165
166 auto cnameDomain = lastCNAMEDomain([m_task _resolvedCNAMEChain]);
167 if (!cnameDomain.isEmpty())
168 networkSession()->setFirstPartyHostCNAMEDomain(requestURL.host().toString(), WTFMove(cnameDomain));
169}

171void NetworkDataTaskCocoa::applyCookiePolicyForThirdPartyCNAMECloaking(const WebCore::ResourceRequest& request)
172{
173 if (!hasCNAMEAndCookieTransformSPI(m_task.get()) || isTopLevelNavigation() || !networkSession() || !networkSession()->networkStorageSession() || !networkSession()->networkStorageSession()->resourceLoadStatisticsEnabled())
174 return;
175
176 if (isThirdPartyRequest(request)) {
177 m_task.get()._cookieTransformCallback = nil;
178 return;
179 }
180
181 // Cap expiry of incoming cookies in response if it is a same-site
182 // subresource but it resolves to a different CNAME than the top
183 // site request, a.k.a. third-party CNAME cloaking.
184 auto firstPartyURL = request.firstPartyForCookies();
185 auto firstPartyHostCNAME = networkSession()->firstPartyHostCNAMEDomain(firstPartyURL.host().toString());
186
187 m_task.get()._cookieTransformCallback = makeBlockPtr([requestURL = crossThreadCopy(request.url()), firstPartyURL = crossThreadCopy(firstPartyURL), firstPartyHostCNAME = crossThreadCopy(firstPartyHostCNAME), thirdPartyCNAMEDomainForTesting = crossThreadCopy(networkSession()->thirdPartyCNAMEDomainForTesting()), ageCapForCNAMECloakedCookies = crossThreadCopy(m_ageCapForCNAMECloakedCookies), weakTask = WeakObjCPtr<NSURLSessionDataTask>(m_task.get()), debugLoggingEnabled = networkSession()->networkStorageSession()->resourceLoadStatisticsDebugLoggingEnabled()] (NSArray<NSHTTPCookie*> *cookiesSetInResponse) -> NSArray<NSHTTPCookie*> * {
188 auto task = weakTask.get();
189 if (!task || ![cookiesSetInResponse count])
190 return cookiesSetInResponse;
191
192 auto cnameDomain = lastCNAMEDomain([task _resolvedCNAMEChain]);
193 if (cnameDomain.isEmpty()) {
194 if (!thirdPartyCNAMEDomainForTesting)
195 return cookiesSetInResponse;
196 cnameDomain = *thirdPartyCNAMEDomainForTesting;
197 }
198
199 // CNAME cloaking is a first-party sub resource that resolves
200 // through a CNAME that differs from the first-party domain and
201 // also differs from the top frame host's CNAME, if one exists.
202 if (!cnameDomain.matches(firstPartyURL) && (!firstPartyHostCNAME || cnameDomain != *firstPartyHostCNAME)) {
203
204 NSUInteger count = [cookiesSetInResponse count];
205 // Don't use RetainPtr here. This array has to be retained and
206 // auto released to not be released before returned to the code
207 // executing the block.
208 auto* cappedCookies = [NSMutableArray arrayWithCapacity:count];
209 for (NSUInteger i = 0; i < count; ++i) {
210 NSHTTPCookie *cookie = (NSHTTPCookie *)[cookiesSetInResponse objectAtIndex:i];
211 cookie = WebCore::NetworkStorageSession::capExpiryOfPersistentCookie(cookie, ageCapForCNAMECloakedCookies);
212 [cappedCookies addObject:cookie];
213 if (debugLoggingEnabled)
214 RELEASE_LOG_INFO(ITPDebug, "Capped the expiry of third-party CNAME cloaked cookie named %{public}s.", [[cookie name] UTF8String]);
215 }
216 return cappedCookies;
217 }
218
219 return cookiesSetInResponse;
220 }).get();
221}

224void NetworkDataTaskCocoa::blockCookies()
225{
226 ASSERT(hasProcessPrivilege(ProcessPrivilege::CanAccessRawCookies));
227
228 if (m_hasBeenSetToUseStatelessCookieStorage)
229 return;
230
231 [m_task _setExplicitCookieStorage:statelessCookieStorage()._cookieStorage];
232 m_hasBeenSetToUseStatelessCookieStorage = true;
233}
234
235void NetworkDataTaskCocoa::unblockCookies()
236{
237 ASSERT(hasProcessPrivilege(ProcessPrivilege::CanAccessRawCookies));
238
239 if (!m_hasBeenSetToUseStatelessCookieStorage)
240 return;
241
242 if (auto* storageSession = m_session->networkStorageSession()) {
243 [m_task _setExplicitCookieStorage:storageSession->nsCookieStorage()._cookieStorage];
244 m_hasBeenSetToUseStatelessCookieStorage = false;
245 }
246}

248// FIXME: Temporary fix for <rdar://problem/60089022> until content can be updated.
249bool NetworkDataTaskCocoa::needsFirstPartyCookieBlockingLatchModeQuirk(const URL& firstPartyURL, const URL& requestURL, const URL& redirectingURL) const
250{
251 using RegistrableDomain = WebCore::RegistrableDomain;
252 static NeverDestroyed<HashMap<RegistrableDomain, RegistrableDomain>> quirkPairs = [] {
253 HashMap<RegistrableDomain, RegistrableDomain> map;
254 map.add(RegistrableDomain::uncheckedCreateFromRegistrableDomainString("ymail.com"_s), RegistrableDomain::uncheckedCreateFromRegistrableDomainString("yahoo.com"_s));
255 map.add(RegistrableDomain::uncheckedCreateFromRegistrableDomainString("aolmail.com"_s), RegistrableDomain::uncheckedCreateFromRegistrableDomainString("aol.com"_s));
256 return map;
257 }();
258
259 RegistrableDomain firstPartyDomain { firstPartyURL };
260 RegistrableDomain requestDomain { requestURL };
261 if (firstPartyDomain != requestDomain)
262 return false;
263
264 RegistrableDomain redirectingDomain { redirectingURL };
265 auto quirk = quirkPairs.get().find(redirectingDomain);
266 if (quirk == quirkPairs.get().end())
267 return false;
268
269 return quirk->value == requestDomain;
270}

294NetworkDataTaskCocoa::NetworkDataTaskCocoa(NetworkSession& session, NetworkDataTaskClient& client, const NetworkLoadParameters& parameters)
295 : NetworkDataTask(session, client, parameters.request, parameters.storedCredentialsPolicy, parameters.shouldClearReferrerOnHTTPSToHTTPRedirect, parameters.isMainFrameNavigation)
296 , m_sessionWrapper(makeWeakPtr(static_cast<NetworkSessionCocoa&>(session).sessionWrapperForTask(parameters.webPageProxyID, parameters.request, parameters.storedCredentialsPolicy, parameters.isNavigatingToAppBoundDomain)))
297 , m_frameID(parameters.webFrameID)
298 , m_pageID(parameters.webPageID)
299 , m_webPageProxyID(parameters.webPageProxyID)
300 , m_isForMainResourceNavigationForAnyFrame(parameters.isMainResourceNavigationForAnyFrame)
301 , m_isAlwaysOnLoggingAllowed(computeIsAlwaysOnLoggingAllowed(session))
302 , m_shouldRelaxThirdPartyCookieBlocking(parameters.shouldRelaxThirdPartyCookieBlocking)
303 , m_sourceOrigin(parameters.sourceOrigin)
304{
305 auto request = parameters.request;
306 auto url = request.url();
307 if (m_storedCredentialsPolicy == WebCore::StoredCredentialsPolicy::Use && url.protocolIsInHTTPFamily()) {
308 m_user = url.user();
309 m_password = url.password();
310 request.removeCredentials();
311 url = request.url();
312
313#if USE(CREDENTIAL_STORAGE_WITH_NETWORK_SESSION)
314 if (auto* storageSession = m_session->networkStorageSession()) {
315 if (m_user.isEmpty() && m_password.isEmpty())
316 m_initialCredential = storageSession->credentialStorage().get(m_partition, url);
317 else
318 storageSession->credentialStorage().set(m_partition, WebCore::Credential(m_user, m_password, WebCore::CredentialPersistenceNone), url);
319 }
320#endif
321 }
322
323#if USE(CREDENTIAL_STORAGE_WITH_NETWORK_SESSION)
324 if (!m_initialCredential.isEmpty()) {
325 // FIXME: Support Digest authentication, and Proxy-Authorization.
326 applyBasicAuthorizationHeader(request, m_initialCredential);
327 }
328#endif
329
330 bool shouldBlockCookies = false;
331#if ENABLE(RESOURCE_LOAD_STATISTICS)
332 shouldBlockCookies = m_storedCredentialsPolicy == WebCore::StoredCredentialsPolicy::EphemeralStateless;
333 if (auto* networkStorageSession = session.networkStorageSession()) {
334 if (!shouldBlockCookies)
335 shouldBlockCookies = networkStorageSession->shouldBlockCookies(request, frameID(), pageID(), m_shouldRelaxThirdPartyCookieBlocking);
336 }
337#endif
338 restrictRequestReferrerToOriginIfNeeded(request);
339
340 RetainPtr<NSURLRequest> nsRequest = request.nsURLRequest(WebCore::HTTPBodyUpdatePolicy::UpdateHTTPBody);
341 RetainPtr<NSMutableURLRequest> mutableRequest = adoptNS([nsRequest.get() mutableCopy]);
342
343#if ENABLE(APP_PRIVACY_REPORT)
344 ALLOW_DEPRECATED_DECLARATIONS_BEGIN
345 mutableRequest.get()._isNonAppInitiated = !request.isAppBound();
346 ALLOW_DEPRECATED_DECLARATIONS_END
347#endif
348
349 if (parameters.pcmDataCarried)
350 processPCMRequest(*parameters.pcmDataCarried, mutableRequest.get());
351
352 nsRequest = mutableRequest;
353
354 m_session->appBoundNavigationTestingData().updateAppBoundNavigationTestingData(request, request.firstPartyForCookies().isEmpty() ? WebCore::RegistrableDomain(request.url()) : WebCore::RegistrableDomain(request.firstPartyForCookies()));
355
356 applySniffingPoliciesAndBindRequestToInferfaceIfNeeded(nsRequest, parameters.contentSniffingPolicy == WebCore::ContentSniffingPolicy::SniffContent && !url.isLocalFile(), parameters.contentEncodingSniffingPolicy == WebCore::ContentEncodingSniffingPolicy::Sniff);
357
358 m_task = [m_sessionWrapper->session dataTaskWithRequest:nsRequest.get()];
359
360 WTFBeginSignpost(m_task.get(), "DataTask", "%{public}s pri: %.2f preconnect: %d", url.string().ascii().data(), toNSURLSessionTaskPriority(request.priority()), parameters.shouldPreconnectOnly == PreconnectOnly::Yes);
361
362 RELEASE_ASSERT(!m_sessionWrapper->dataTaskMap.contains([m_task taskIdentifier]));
363 m_sessionWrapper->dataTaskMap.add([m_task taskIdentifier], this);
364 LOG(NetworkSession, "%lu Creating NetworkDataTask with URL %s", (unsigned long)[m_task taskIdentifier], [nsRequest URL].absoluteString.UTF8String);
365
366 if (parameters.shouldPreconnectOnly == PreconnectOnly::Yes) {
367#if ENABLE(SERVER_PRECONNECT)
368 m_task.get()._preconnect = true;
369#else
370 ASSERT_NOT_REACHED();
371#endif
372 }
373
374#if ENABLE(RESOURCE_LOAD_STATISTICS)
375#if HAVE(CFNETWORK_CNAME_AND_COOKIE_TRANSFORM_SPI)
376 applyCookiePolicyForThirdPartyCNAMECloaking(request);
377#endif
378 if (shouldBlockCookies) {
379#if !RELEASE_LOG_DISABLED
380 if (m_session->shouldLogCookieInformation())
381 RELEASE_LOG_IF(isAlwaysOnLoggingAllowed(), Network, "%p - NetworkDataTaskCocoa::logCookieInformation: pageID=%" PRIu64 ", frameID=%" PRIu64 ", taskID=%lu: Blocking cookies for URL %s", this, pageID().toUInt64(), frameID().toUInt64(), (unsigned long)[m_task taskIdentifier], [nsRequest URL].absoluteString.UTF8String);
382#else
383 LOG(NetworkSession, "%lu Blocking cookies for URL %s", (unsigned long)[m_task taskIdentifier], [nsRequest URL].absoluteString.UTF8String);
384#endif
385 blockCookies();
386 }
387#endif
388
389 if (WebCore::ResourceRequest::resourcePrioritiesEnabled())
390 m_task.get().priority = toNSURLSessionTaskPriority(request.priority());
391
392 updateTaskWithFirstPartyForSameSiteCookies(m_task.get(), request);
393
394#if HAVE(NW_ACTIVITY)
395 if (parameters.networkActivityTracker)
396 m_task.get()._nw_activity = parameters.networkActivityTracker->getPlatformObject();
397#endif
398}

467void NetworkDataTaskCocoa::willPerformHTTPRedirection(WebCore::ResourceResponse&& redirectResponse, WebCore::ResourceRequest&& request, RedirectCompletionHandler&& completionHandler)
468{
469 WTFEmitSignpost(m_task.get(), "DataTask", "redirect");
470
471 networkLoadMetrics().hasCrossOriginRedirect = networkLoadMetrics().hasCrossOriginRedirect || !WebCore::SecurityOrigin::create(request.url())->canRequest(redirectResponse.url());
472
473 if (redirectResponse.httpStatusCode() == 307 || redirectResponse.httpStatusCode() == 308) {
474 ASSERT(m_lastHTTPMethod == request.httpMethod());
475 WebCore::FormData* body = m_firstRequest.httpBody();
476 if (body && !body->isEmpty() && !equalLettersIgnoringASCIICase(m_lastHTTPMethod, "get"))
477 request.setHTTPBody(body);
478
479 String originalContentType = m_firstRequest.httpContentType();
480 if (!originalContentType.isEmpty())
481 request.setHTTPHeaderField(WebCore::HTTPHeaderName::ContentType, originalContentType);
482 } else if (redirectResponse.httpStatusCode() == 303 && equalLettersIgnoringASCIICase(m_firstRequest.httpMethod(), "head")) // FIXME: (rdar://problem/13706454).
483 request.setHTTPMethod("HEAD"_s);
484
485 // Should not set Referer after a redirect from a secure resource to non-secure one.
486 if (m_shouldClearReferrerOnHTTPSToHTTPRedirect && !request.url().protocolIs("https") && WTF::protocolIs(request.httpReferrer(), "https"))
487 request.clearHTTPReferrer();
488
489 const auto& url = request.url();
490 m_user = url.user();
491 m_password = url.password();
492 m_lastHTTPMethod = request.httpMethod();
493 request.removeCredentials();
494
495 if (!protocolHostAndPortAreEqual(request.url(), redirectResponse.url())) {
496 // The network layer might carry over some headers from the original request that
497 // we want to strip here because the redirect is cross-origin.
498 request.clearHTTPAuthorization();
499 request.clearHTTPOrigin();
500#if USE(CREDENTIAL_STORAGE_WITH_NETWORK_SESSION)
501 } else {
502 // Only consider applying authentication credentials if this is actually a redirect and the redirect
503 // URL didn't include credentials of its own.
504 if (m_user.isEmpty() && m_password.isEmpty() && !redirectResponse.isNull()) {
505 auto credential = m_session->networkStorageSession() ? m_session->networkStorageSession()->credentialStorage().get(m_partition, request.url()) : WebCore::Credential();
506 if (!credential.isEmpty()) {
507 m_initialCredential = credential;
508
509 // FIXME: Support Digest authentication, and Proxy-Authorization.
510 applyBasicAuthorizationHeader(request, m_initialCredential);
511 }
512 }
513#endif
514 }
515
516 if (isTopLevelNavigation())
517 request.setFirstPartyForCookies(request.url());
518
519#if ENABLE(RESOURCE_LOAD_STATISTICS)
520#if HAVE(CFNETWORK_CNAME_AND_COOKIE_TRANSFORM_SPI)
521 applyCookiePolicyForThirdPartyCNAMECloaking(request);
522#endif
523 if (!m_hasBeenSetToUseStatelessCookieStorage) {
524 if (m_storedCredentialsPolicy == WebCore::StoredCredentialsPolicy::EphemeralStateless
525 || (m_session->networkStorageSession() && m_session->networkStorageSession()->shouldBlockCookies(request, m_frameID, m_pageID, m_shouldRelaxThirdPartyCookieBlocking)))
526 blockCookies();
527 } else if (m_storedCredentialsPolicy != WebCore::StoredCredentialsPolicy::EphemeralStateless && needsFirstPartyCookieBlockingLatchModeQuirk(request.firstPartyForCookies(), request.url(), redirectResponse.url()))
528 unblockCookies();
529#if !RELEASE_LOG_DISABLED
530 if (m_session->shouldLogCookieInformation())
531 RELEASE_LOG_IF(isAlwaysOnLoggingAllowed(), Network, "%p - NetworkDataTaskCocoa::willPerformHTTPRedirection::logCookieInformation: pageID=%" PRIu64 ", frameID=%" PRIu64 ", taskID=%lu: %s cookies for redirect URL %s", this, m_pageID.toUInt64(), m_frameID.toUInt64(), (unsigned long)[m_task taskIdentifier], (m_hasBeenSetToUseStatelessCookieStorage ? "Blocking" : "Not blocking"), request.url().string().utf8().data());
532#else
533 LOG(NetworkSession, "%lu %s cookies for redirect URL %s", (unsigned long)[m_task taskIdentifier], (m_hasBeenSetToUseStatelessCookieStorage ? "Blocking" : "Not blocking"), request.url().string().utf8().data());
534#endif
535#endif
536
537 updateTaskWithFirstPartyForSameSiteCookies(m_task.get(), request);
538
539 if (m_client)
540 m_client->willPerformHTTPRedirection(WTFMove(redirectResponse), WTFMove(request), [completionHandler = WTFMove(completionHandler), this, weakThis = makeWeakPtr(*this)] (auto&& request) mutable {
541 if (!weakThis || !m_session)
542 return completionHandler({ });
543 if (!request.isNull())
544 restrictRequestReferrerToOriginIfNeeded(request);
545 completionHandler(WTFMove(request));
546 });

568bool NetworkDataTaskCocoa::tryPasswordBasedAuthentication(const WebCore::AuthenticationChallenge& challenge, ChallengeCompletionHandler& completionHandler)
569{
570 if (!challenge.protectionSpace().isPasswordBased())
571 return false;
572
573 if (!m_user.isNull() && !m_password.isNull()) {
574 auto persistence = m_storedCredentialsPolicy == WebCore::StoredCredentialsPolicy::Use ? WebCore::CredentialPersistenceForSession : WebCore::CredentialPersistenceNone;
575 completionHandler(AuthenticationChallengeDisposition::UseCredential, WebCore::Credential(m_user, m_password, persistence));
576 m_user = String();
577 m_password = String();
578 return true;
579 }
580
581#if USE(CREDENTIAL_STORAGE_WITH_NETWORK_SESSION)
582 if (m_storedCredentialsPolicy == WebCore::StoredCredentialsPolicy::Use) {
583 if (!m_initialCredential.isEmpty() || challenge.previousFailureCount()) {
584 // The stored credential wasn't accepted, stop using it.
585 // There is a race condition here, since a different credential might have already been stored by another ResourceHandle,
586 // but the observable effect should be very minor, if any.
587 if (auto* storageSession = m_session->networkStorageSession())
588 storageSession->credentialStorage().remove(m_partition, challenge.protectionSpace());
589 }
590
591 if (!challenge.previousFailureCount()) {
592 auto credential = m_session->networkStorageSession() ? m_session->networkStorageSession()->credentialStorage().get(m_partition, challenge.protectionSpace()) : WebCore::Credential();
593 if (!credential.isEmpty() && credential != m_initialCredential) {
594 ASSERT(credential.persistence() == WebCore::CredentialPersistenceNone);
595 if (challenge.failureResponse().httpStatusCode() == 401) {
596 // Store the credential back, possibly adding it as a default for this directory.
597 if (auto* storageSession = m_session->networkStorageSession())
598 storageSession->credentialStorage().set(m_partition, credential, challenge.protectionSpace(), challenge.failureResponse().url());
599 }
600 completionHandler(AuthenticationChallengeDisposition::UseCredential, credential);
601 return true;
602 }
603 }
604 }
605#endif
606
607 if (!challenge.proposedCredential().isEmpty() && !challenge.previousFailureCount()) {
608 completionHandler(AuthenticationChallengeDisposition::UseCredential, challenge.proposedCredential());
609 return true;
610 }
611
612 return false;
613}

633void NetworkDataTaskCocoa::resume()
634{
635 WTFEmitSignpost(m_task.get(), "DataTask", "resume");
636
637 auto& cocoaSession = static_cast<NetworkSessionCocoa&>(*m_session);
638 if (cocoaSession.deviceManagementRestrictionsEnabled() && m_isForMainResourceNavigationForAnyFrame) {
639 auto didDetermineDeviceRestrictionPolicyForURL = makeBlockPtr([this, protectedThis = makeRef(*this)](BOOL isBlocked) mutable {
640 callOnMainRunLoop([this, protectedThis = WTFMove(protectedThis), isBlocked] {
641 if (isBlocked) {
642 scheduleFailure(RestrictedURLFailure);
643 return;
644 }
645
646 [m_task resume];
647 });
648 });
649
650#if HAVE(DEVICE_MANAGEMENT)
651 if (cocoaSession.allLoadsBlockedByDeviceManagementRestrictionsForTesting())
652 didDetermineDeviceRestrictionPolicyForURL(true);
653 else {
654 RetainPtr<NSURL> urlToCheck = [m_task currentRequest].URL;
655 [cocoaSession.deviceManagementPolicyMonitor() requestPoliciesForWebsites:@[ urlToCheck.get() ] completionHandler:makeBlockPtr([didDetermineDeviceRestrictionPolicyForURL, urlToCheck] (NSDictionary<NSURL *, NSNumber *> *policies, NSError *error) {
656 bool isBlocked = error || policies[urlToCheck.get()].integerValue != DMFPolicyOK;
657 didDetermineDeviceRestrictionPolicyForURL(isBlocked);
658 }).get()];
659 }
660#else
661 didDetermineDeviceRestrictionPolicyForURL(cocoaSession.allLoadsBlockedByDeviceManagementRestrictionsForTesting());
662#endif
663 return;
664 }
665
666 [m_task resume];
667}
