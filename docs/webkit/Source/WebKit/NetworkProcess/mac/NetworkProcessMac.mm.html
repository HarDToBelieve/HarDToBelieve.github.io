52void NetworkProcess::initializeProcess(const AuxiliaryProcessInitializationParameters&)
53{
54#if PLATFORM(MAC) && !PLATFORM(MACCATALYST)
55 // Having a window server connection in this process would result in spin logs (<rdar://problem/13239119>).
56 OSStatus error = SetApplicationIsDaemon(true);
57 ASSERT_UNUSED(error, error == noErr);
58#endif
59
60 launchServicesCheckIn();
61}
62
63void NetworkProcess::initializeProcessName(const AuxiliaryProcessInitializationParameters& parameters)
64{
65#if !PLATFORM(MACCATALYST)
66 NSString *applicationName = [NSString stringWithFormat:WEB_UI_STRING("%@ Networking", "visible name of the network process. The argument is the application name."), (NSString *)parameters.uiProcessName];
67 _LSSetApplicationInformationItem(kLSDefaultSessionID, _LSGetCurrentApplicationASN(), _kLSDisplayNameKey, (CFStringRef)applicationName, nullptr);
68#endif
69}
70
71void NetworkProcess::platformInitializeNetworkProcess(const NetworkProcessCreationParameters& parameters)
72{
73 platformInitializeNetworkProcessCocoa(parameters);
74
75#if ENABLE(SEC_ITEM_SHIM)
76 // SecItemShim is needed for CFNetwork APIs that query Keychains beneath us.
77 initializeSecItemShim(*this);
78#endif
79}
80
81void NetworkProcess::allowSpecificHTTPSCertificateForHost(const WebCore::CertificateInfo& certificateInfo, const String& host)
82{
83 [NSURLRequest setAllowsSpecificHTTPSCertificate:(__bridge NSArray *)certificateInfo.certificateChain() forHost:(NSString *)host];
84}
85
86void NetworkProcess::initializeSandbox(const AuxiliaryProcessInitializationParameters& parameters, SandboxInitializationParameters& sandboxParameters)
87{
88 // Need to overide the default, because service has a different bundle ID.
89 auto webKitBundle = [NSBundle bundleWithIdentifier:@"com.apple.WebKit"];
90
91 sandboxParameters.setOverrideSandboxProfilePath(makeString(String([webKitBundle resourcePath]), "/com.apple.WebKit.NetworkProcess.sb"));
92
93 AuxiliaryProcess::initializeSandbox(parameters, sandboxParameters);
94}
95
96void NetworkProcess::platformTerminate()
97{
98 if (m_clearCacheDispatchGroup) {
99 dispatch_group_wait(m_clearCacheDispatchGroup.get(), DISPATCH_TIME_FOREVER);
100 m_clearCacheDispatchGroup = nullptr;
101 }
102}
