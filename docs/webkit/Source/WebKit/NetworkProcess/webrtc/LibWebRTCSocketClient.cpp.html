41LibWebRTCSocketClient::LibWebRTCSocketClient(WebCore::LibWebRTCSocketIdentifier identifier, NetworkRTCProvider& rtcProvider, std::unique_ptr<rtc::AsyncPacketSocket>&& socket, Type type, Ref<IPC::Connection>&& connection)
42 : m_identifier(identifier)
43 , m_type(type)
44 , m_rtcProvider(rtcProvider)
45 , m_socket(WTFMove(socket))
46 , m_connection(WTFMove(connection))
47{
48 ASSERT(m_socket);
49
50 m_socket->SignalReadPacket.connect(this, &LibWebRTCSocketClient::signalReadPacket);
51 m_socket->SignalSentPacket.connect(this, &LibWebRTCSocketClient::signalSentPacket);
52 m_socket->SignalClose.connect(this, &LibWebRTCSocketClient::signalClose);
53
54 switch (type) {
55 case Type::ServerConnectionTCP:
56 return;
57 case Type::ClientTCP:
58 m_socket->SignalConnect.connect(this, &LibWebRTCSocketClient::signalConnect);
59 m_socket->SignalAddressReady.connect(this, &LibWebRTCSocketClient::signalAddressReady);
60 return;
61 case Type::ServerTCP:
62 m_socket->SignalConnect.connect(this, &LibWebRTCSocketClient::signalConnect);
63 m_socket->SignalNewConnection.connect(this, &LibWebRTCSocketClient::signalNewConnection);
64 signalAddressReady();
65 return;
66 case Type::UDP:
67 m_socket->SignalConnect.connect(this, &LibWebRTCSocketClient::signalConnect);
68 signalAddressReady();
69 return;
70 }
71}

73void LibWebRTCSocketClient::sendTo(const uint8_t* data, size_t size, const rtc::SocketAddress& socketAddress, const rtc::PacketOptions& options)
74{
75 m_socket->SendTo(data, size, socketAddress, options);
76 auto error = m_socket->GetError();
77 RELEASE_LOG_ERROR_IF(error && m_sendError != error, Network, "LibWebRTCSocketClient::sendTo (ID=%" PRIu64 ") failed with error %d", m_identifier.toUInt64(), error);
78 m_sendError = error;
79}
80
81void LibWebRTCSocketClient::close()
82{
83 ASSERT(m_socket);
84 auto result = m_socket->Close();
85 UNUSED_PARAM(result);
86 RELEASE_LOG_ERROR_IF(result, Network, "LibWebRTCSocketClient::close (ID=%" PRIu64 ") failed with error %d", m_identifier.toUInt64(), m_socket->GetError());
87
88 m_rtcProvider.takeSocket(m_identifier);
89}

91void LibWebRTCSocketClient::setOption(int option, int value)
92{
93 ASSERT(m_socket);
94 auto result = m_socket->SetOption(static_cast<rtc::Socket::Option>(option), value);
95 UNUSED_PARAM(result);
96 RELEASE_LOG_ERROR_IF(result, Network, "LibWebRTCSocketClient::setOption(%d, %d) (ID=%" PRIu64 ") failed with error %d", option, value, m_identifier.toUInt64(), m_socket->GetError());
97}

99void LibWebRTCSocketClient::signalReadPacket(rtc::AsyncPacketSocket* socket, const char* value, size_t length, const rtc::SocketAddress& address, const rtc::PacketTime& packetTime)
100{
101 ASSERT_UNUSED(socket, m_socket.get() == socket);
102 IPC::DataReference data(reinterpret_cast<const uint8_t*>(value), length);
103 m_connection->send(Messages::LibWebRTCNetwork::SignalReadPacket(m_identifier, data, RTCNetwork::IPAddress(address.ipaddr()), address.port(), packetTime), 0);
104}
105
106void LibWebRTCSocketClient::signalSentPacket(rtc::AsyncPacketSocket* socket, const rtc::SentPacket& sentPacket)
107{
108 ASSERT_UNUSED(socket, m_socket.get() == socket);
109 m_connection->send(Messages::LibWebRTCNetwork::SignalSentPacket(m_identifier, sentPacket.packet_id, sentPacket.send_time_ms), 0);
110}

112void LibWebRTCSocketClient::signalNewConnection(rtc::AsyncPacketSocket* socket, rtc::AsyncPacketSocket* newSocket)
113{
114 ASSERT_UNUSED(socket, m_socket.get() == socket);
115 m_rtcProvider.newConnection(*this, std::unique_ptr<rtc::AsyncPacketSocket>(newSocket));
116}
117
118void LibWebRTCSocketClient::signalAddressReady(rtc::AsyncPacketSocket* socket, const rtc::SocketAddress& address)
119{
120 ASSERT_UNUSED(socket, m_socket.get() == socket);
121 m_connection->send(Messages::LibWebRTCNetwork::SignalAddressReady(m_identifier, RTCNetwork::SocketAddress(address)), 0);
122}
123
124void LibWebRTCSocketClient::signalAddressReady()
125{
126 signalAddressReady(m_socket.get(), m_socket->GetLocalAddress());
127}
128
129void LibWebRTCSocketClient::signalConnect(rtc::AsyncPacketSocket* socket)
130{
131 ASSERT_UNUSED(socket, m_socket.get() == socket);
132 m_connection->send(Messages::LibWebRTCNetwork::SignalConnect(m_identifier), 0);
133}
134
135void LibWebRTCSocketClient::signalClose(rtc::AsyncPacketSocket* socket, int error)
136{
137 ASSERT_UNUSED(socket, m_socket.get() == socket);
138 m_connection->send(Messages::LibWebRTCNetwork::SignalClose(m_identifier, error), 0);
139
140 // We want to remove 'this' from the socket map now but we will destroy it asynchronously
141 // so that the socket parameter of signalClose remains alive as the caller of signalClose may actually being using it afterwards.
142 m_rtcProvider.callOnRTCNetworkThread([socket = m_rtcProvider.takeSocket(m_identifier)] { });
143}
