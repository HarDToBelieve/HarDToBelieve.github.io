66struct PendingRegistrationRequest {
67 WTF_MAKE_STRUCT_FAST_ALLOCATED;
68 PendingRegistrationRequest(Ref<IPC::Connection>&& connection, MDNSRegisterIdentifier requestIdentifier, String&& name, PAL::SessionID sessionID)
69 : connection(WTFMove(connection))
70 , requestIdentifier(requestIdentifier)
71 , name(WTFMove(name))
72 , sessionID(sessionID)
73 {
74 }
75
76 Ref<IPC::Connection> connection;
77 MDNSRegisterIdentifier requestIdentifier;
78 String name;
79 PAL::SessionID sessionID;
80 DNSRecordRef record;
81};

84static uintptr_t pendingRegistrationRequestCount = 1;
85static HashMap<uintptr_t, std::unique_ptr<PendingRegistrationRequest>>& pendingRegistrationRequests()
86{
87 ASSERT(RunLoop::isMain());
88 static NeverDestroyed<HashMap<uintptr_t, std::unique_ptr<PendingRegistrationRequest>>> map;
89 return map;
90}
91
92static void registerMDNSNameCallback(DNSServiceRef, DNSRecordRef record, DNSServiceFlags, DNSServiceErrorType errorCode, void *context)
93{
94 auto request = pendingRegistrationRequests().take(reinterpret_cast<uintptr_t>(context));
95 if (!request)
96 return;
97
98 MDNS_RELEASE_LOG_IN_CALLBACK(request->sessionID, "registerMDNSNameCallback with error %d", errorCode);
99
100 if (errorCode) {
101 request->connection->send(Messages::WebMDNSRegister::FinishedRegisteringMDNSName { request->requestIdentifier, makeUnexpected(WebCore::MDNSRegisterError::DNSSD) }, 0);
102 return;
103 }
104 request->connection->send(Messages::WebMDNSRegister::FinishedRegisteringMDNSName { request->requestIdentifier, request->name }, 0);
105}

107void NetworkMDNSRegister::registerMDNSName(MDNSRegisterIdentifier requestIdentifier, WebCore::DocumentIdentifier documentIdentifier, const String& ipAddress)
108{
109 DNSServiceRef service;
110 auto iterator = m_services.find(documentIdentifier);
111 if (iterator == m_services.end()) {
112 auto error = DNSServiceCreateConnection(&service);
113 if (error) {
114 MDNS_RELEASE_LOG("registerMDNSName DNSServiceCreateConnection error %d", error);
115 m_connection.connection().send(Messages::WebMDNSRegister::FinishedRegisteringMDNSName { requestIdentifier, makeUnexpected(WebCore::MDNSRegisterError::DNSSD) }, 0);
116 return;
117 }
118 error = DNSServiceSetDispatchQueue(service, dispatch_get_main_queue());
119 if (error) {
120 MDNS_RELEASE_LOG("registerMDNSName DNSServiceCreateConnection error %d", error);
121 m_connection.connection().send(Messages::WebMDNSRegister::FinishedRegisteringMDNSName { requestIdentifier, makeUnexpected(WebCore::MDNSRegisterError::DNSSD) }, 0);
122 return;
123 }
124 ASSERT(service);
125 m_services.add(documentIdentifier, service);
126 } else
127 service = iterator->value;
128
129 auto name = makeString(createCanonicalUUIDString(), ".local");
130
131 auto ip = inet_addr(ipAddress.utf8().data());
132
133 if (ip == ( in_addr_t)(-1)) {
134 MDNS_RELEASE_LOG("registerMDNSName inet_addr error");
135 m_connection.connection().send(Messages::WebMDNSRegister::FinishedRegisteringMDNSName { requestIdentifier, makeUnexpected(WebCore::MDNSRegisterError::BadParameter) }, 0);
136 return;
137 }
138
139 auto pendingRequest = makeUnique<PendingRegistrationRequest>(makeRef(m_connection.connection()), requestIdentifier, WTFMove(name), sessionID());
140 auto* record = &pendingRequest->record;
141 auto error = DNSServiceRegisterRecord(service,
142 record,
143#if HAVE(MDNS_FAST_REGISTRATION)
144 kDNSServiceFlagsKnownUnique,
145#else
146 kDNSServiceFlagsUnique,
147#endif
148 0,
149 pendingRequest->name.utf8().data(),
150 kDNSServiceType_A,
151 kDNSServiceClass_IN,
152 4,
153 &ip,
154 0,
155 registerMDNSNameCallback,
156 reinterpret_cast<void*>(pendingRegistrationRequestCount));
157 if (error) {
158 MDNS_RELEASE_LOG("registerMDNSName DNSServiceRegisterRecord error %d", error);
159 m_connection.connection().send(Messages::WebMDNSRegister::FinishedRegisteringMDNSName { requestIdentifier, makeUnexpected(WebCore::MDNSRegisterError::DNSSD) }, 0);
160 return;
161 }
162 pendingRegistrationRequests().add(pendingRegistrationRequestCount++, WTFMove(pendingRequest));
163}
