41#define RTC_RELEASE_LOG(fmt, ...) RELEASE_LOG(Network, "%p - NetworkRTCMonitor::" fmt, this, ##__VA_ARGS__)
42
43class NetworkManagerWrapper final : public sigslot::has_slots<> {
44public:
45 void addListener(NetworkRTCMonitor&);
46 void removeListener(NetworkRTCMonitor&);
47
48private:
49 void onNetworksChanged();
50
51 std::unique_ptr<rtc::BasicNetworkManager> m_manager;
52 WeakHashSet<NetworkRTCMonitor> m_observers;
53
54 bool m_didReceiveResults { false };
55 Vector<RTCNetwork> m_networkList;
56 Vector<RTCNetwork> m_filteredNetworkList;
57 RTCNetwork::IPAddress m_ipv4;
58 RTCNetwork::IPAddress m_ipv6;
59};
60
61static NetworkManagerWrapper& networkManager()
62{
63 static NeverDestroyed<NetworkManagerWrapper> networkManager;
64 return networkManager.get();
65}
66
67void NetworkManagerWrapper::addListener(NetworkRTCMonitor& monitor)
68{
69 bool shouldStart = m_observers.computesEmpty();
70 m_observers.add(monitor);
71 if (!shouldStart) {
72 if (m_didReceiveResults)
73 monitor.onNetworksChanged(m_networkList, m_filteredNetworkList, m_ipv4, m_ipv6);
74 return;
75 }
76
77 monitor.rtcProvider().callOnRTCNetworkThread([this]() {
78 if (m_manager)
79 return;
80
81 RELEASE_LOG(WebRTC, "NetworkManagerWrapper startUpdating");
82 m_manager = makeUniqueWithoutFastMallocCheck<rtc::BasicNetworkManager>();
83 m_manager->SignalNetworksChanged.connect(this, &NetworkManagerWrapper::onNetworksChanged);
84 m_manager->StartUpdating();
85 });
86}
87
88void NetworkManagerWrapper::removeListener(NetworkRTCMonitor& monitor)
89{
90 m_observers.remove(monitor);
91 if (!m_observers.computesEmpty())
92 return;
93
94 monitor.rtcProvider().callOnRTCNetworkThread([this]() {
95 if (!m_manager)
96 return;
97
98 RELEASE_LOG(WebRTC, "NetworkManagerWrapper stopUpdating");
99 m_manager->StopUpdating();
100 m_manager = nullptr;
101 });
102}
103
104void NetworkManagerWrapper::onNetworksChanged()
105{
106 RELEASE_LOG(WebRTC, "NetworkManagerWrapper::onNetworksChanged");
107
108 rtc::BasicNetworkManager::NetworkList networks;
109
110 RTCNetwork::IPAddress ipv4;
111 m_manager->GetDefaultLocalAddress(AF_INET, &ipv4.value);
112 RTCNetwork::IPAddress ipv6;
113 m_manager->GetDefaultLocalAddress(AF_INET6, &ipv6.value);
114
115 m_manager->GetNetworks(&networks);
116
117 auto networkList = WTF::map(networks, [](auto& network) { return RTCNetwork { *network }; });
118 Vector<RTCNetwork> filteredNetworkList;
119 for (auto& network : networks) {
120 const auto& ips = network->GetIPs();
121 if (WTF::anyOf(ips, [&](const auto& ip) { return ipv4.value == ip || ipv6.value == ip; }))
122 filteredNetworkList.append(RTCNetwork { *network });
123 }
124
125 callOnMainRunLoop([this, networkList = WTFMove(networkList), filteredNetworkList = WTFMove(filteredNetworkList), ipv4 = WTFMove(ipv4), ipv6 = WTFMove(ipv6)]() mutable {
126 m_didReceiveResults = true;
127
128 m_networkList = WTFMove(networkList);
129 m_filteredNetworkList = WTFMove(filteredNetworkList);
130 m_ipv4 = WTFMove(ipv4);
131 m_ipv6 = WTFMove(ipv6);
132
133 m_observers.forEach([this](auto& observer) {
134 observer.onNetworksChanged(m_networkList, m_filteredNetworkList, m_ipv4, m_ipv6);
135 });
136 });
137}
138
139NetworkRTCMonitor::~NetworkRTCMonitor()
140{
141 ASSERT(!m_manager);
142}
143
144void NetworkRTCMonitor::startUpdatingIfNeeded(bool enableEnumeratingAllNetworkInterfaces)
145{
146 RTC_RELEASE_LOG("startUpdatingIfNeeded m_isStarted=%d", m_isStarted);
147 m_enableEnumeratingAllNetworkInterfaces = enableEnumeratingAllNetworkInterfaces;
148 networkManager().addListener(*this);
149}
150
151void NetworkRTCMonitor::stopUpdating()
152{
153 RTC_RELEASE_LOG("stopUpdating");
154 networkManager().removeListener(*this);
155}
156
157void NetworkRTCMonitor::onNetworksChanged(const Vector<RTCNetwork>& networkList, const Vector<RTCNetwork>& filteredNetworkList, const RTCNetwork::IPAddress& ipv4, const RTCNetwork::IPAddress& ipv6)
158{
159 RTC_RELEASE_LOG("onNetworksChanged sent");
160 m_rtcProvider.connection().send(Messages::WebRTCMonitor::NetworksChanged(m_enableEnumeratingAllNetworkInterfaces ? networkList : filteredNetworkList, ipv4, ipv6), 0);
161}
