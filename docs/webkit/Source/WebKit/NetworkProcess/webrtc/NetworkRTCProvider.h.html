65class NetworkRTCProvider : public rtc::MessageHandler, public IPC::Connection::ThreadMessageReceiverRefCounted {
66public:
67 static Ref<NetworkRTCProvider> create(NetworkConnectionToWebProcess& connection)
68 {
69 auto instance = adoptRef(*new NetworkRTCProvider(connection));
70 instance->startListeningForIPC();
71 return instance;
72 }
73 ~NetworkRTCProvider();
74
75 void didReceiveMessage(IPC::Connection&, IPC::Decoder&);
76 void didReceiveNetworkRTCMonitorMessage(IPC::Connection& connection, IPC::Decoder& decoder) { m_rtcMonitor.didReceiveMessage(connection, decoder); }

78 class Socket {
79 public:
80 virtual ~Socket() = default;
81
82 enum class Type : uint8_t { UDP, ServerTCP, ClientTCP, ServerConnectionTCP };
83 virtual Type type() const = 0;
84 virtual WebCore::LibWebRTCSocketIdentifier identifier() const = 0;
85
86 virtual void close() = 0;
87 virtual void setOption(int option, int value) = 0;
88 virtual void sendTo(const uint8_t*, size_t, const rtc::SocketAddress&, const rtc::PacketOptions&) = 0;
89 };

91 std::unique_ptr<Socket> takeSocket(WebCore::LibWebRTCSocketIdentifier);
92 void resolverDone(uint64_t);
93
94 void close();
95
96 void callOnRTCNetworkThread(Function<void()>&&);
97
98 void newConnection(Socket&, std::unique_ptr<rtc::AsyncPacketSocket>&&);
99
100 void closeListeningSockets(Function<void()>&&);
101 void authorizeListeningSockets() { m_isListeningSocketAuthorized = true; }
102
103 IPC::Connection& connection() { return m_ipcConnection.get(); }
104
105 void closeSocket(WebCore::LibWebRTCSocketIdentifier);

116 void setSocketOption(WebCore::LibWebRTCSocketIdentifier, int option, int value);
