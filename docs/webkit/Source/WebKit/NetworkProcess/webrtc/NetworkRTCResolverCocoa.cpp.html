35static void resolvedName(CFHostRef hostRef, CFHostInfoType typeInfo, const CFStreamError *error, void *info)
36{
37 ASSERT_UNUSED(typeInfo, !typeInfo);
38
39 ASSERT(info);
40 auto* resolver = static_cast<NetworkRTCResolverCocoa*>(info);
41
42 if (error->domain) {
43 // FIXME: Need to handle failure, but info is not provided in the callback.
44 resolver->completed(makeUnexpected(WebCore::DNSError::Unknown));
45 return;
46 }
47
48 Boolean result;
49 CFArrayRef resolvedAddresses = (CFArrayRef)CFHostGetAddressing(hostRef, &result);
50 ASSERT_UNUSED(result, result);
51
52 size_t count = CFArrayGetCount(resolvedAddresses);
53 Vector<WebCore::IPAddress> addresses;
54 addresses.reserveInitialCapacity(count);
55
56 for (size_t index = 0; index < count; ++index) {
57 CFDataRef data = (CFDataRef)CFArrayGetValueAtIndex(resolvedAddresses, index);
58 if (auto address = IPAddress::fromSockAddrIn6(*reinterpret_cast<const struct sockaddr_in6*>(CFDataGetBytePtr(data))))
59 addresses.uncheckedAppend(*address);
60 }
61 if (addresses.isEmpty()) {
62 resolver->completed(makeUnexpected(WebCore::DNSError::CannotResolve));
63 return;
64 }
65 resolver->completed(WTFMove(addresses));
66}

68std::unique_ptr<NetworkRTCResolver> NetworkRTCResolver::create(LibWebRTCResolverIdentifier identifier, WebCore::DNSCompletionHandler&& completionHandler)
69{
70 return std::unique_ptr<NetworkRTCResolver>(new NetworkRTCResolverCocoa(identifier, WTFMove(completionHandler)));
71}

73NetworkRTCResolverCocoa::NetworkRTCResolverCocoa(LibWebRTCResolverIdentifier identifier, WebCore::DNSCompletionHandler&& completionHandler)
74 : NetworkRTCResolver(identifier, WTFMove(completionHandler))
75{
76}
77
78NetworkRTCResolverCocoa::~NetworkRTCResolverCocoa()
79{
80 CFHostUnscheduleFromRunLoop(m_host.get(), CFRunLoopGetCurrent(), kCFRunLoopDefaultMode);
81 CFHostSetClient(m_host.get(), nullptr, nullptr);
82}
83
84void NetworkRTCResolverCocoa::start(const String& address)
85{
86 m_host = adoptCF(CFHostCreateWithName(kCFAllocatorDefault, address.createCFString().get()));
87 CFHostClientContext context = { 0, this, nullptr, nullptr, nullptr };
88 CFHostSetClient(m_host.get(), resolvedName, &context);
89 CFHostScheduleWithRunLoop(m_host.get(), CFRunLoopGetCurrent(), kCFRunLoopDefaultMode);
90 Boolean result = CFHostStartInfoResolution(m_host.get(), kCFHostAddresses, nullptr);
91 ASSERT_UNUSED(result, result);
92}
93
94void NetworkRTCResolverCocoa::stop()
95{
96 CFHostCancelInfoResolution(m_host.get(), CFHostInfoType::kCFHostAddresses);
97 if (auto completionHandler = WTFMove(m_completionHandler))
98 completionHandler(makeUnexpected(WebCore::DNSError::Cancelled));
99}
100
101void NetworkRTCResolverCocoa::completed(WebCore::DNSAddressesOrError&& addressesOrError)
102{
103 if (auto completionHandler = WTFMove(m_completionHandler))
104 completionHandler(WTFMove(addressesOrError));
105}
