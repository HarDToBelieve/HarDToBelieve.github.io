42template<> struct DefaultHash<rtc::SocketAddress> {
43 static unsigned hash(const rtc::SocketAddress& address) { return address.Hash(); }
44 static bool equal(const rtc::SocketAddress& a, const rtc::SocketAddress& b) { return a == b || (a.IsNil() && b.IsNil() && a.scope_id() == b.scope_id()); }
45 static constexpr bool safeToCompareToEmptyOrDeleted = true;
46};
47
48template<> struct HashTraits<rtc::SocketAddress> : GenericHashTraits<rtc::SocketAddress> {
49 static rtc::SocketAddress emptyValue() { return rtc::SocketAddress { }; }
50 static void constructDeletedValue(rtc::SocketAddress& address) { address.SetScopeID(std::numeric_limits<int>::min()); }
51 static bool isDeletedValue(const rtc::SocketAddress& address) { return address.scope_id() == std::numeric_limits<int>::min(); }
52};
