36RTCDataChannelRemoteManagerProxy::RTCDataChannelRemoteManagerProxy()
37 : m_queue(WorkQueue::create("RTCDataChannelRemoteManagerProxy"))
38{
39}
40
41void RTCDataChannelRemoteManagerProxy::registerConnectionToWebProcess(NetworkConnectionToWebProcess& connectionToWebProcess)
42{
43 m_queue->dispatch([this, protectedThis = makeRef(*this), identifier = connectionToWebProcess.webProcessIdentifier(), connectionID = connectionToWebProcess.connection().uniqueID()]() mutable {
44 ASSERT(!m_webProcessConnections.contains(identifier));
45 m_webProcessConnections.add(identifier, connectionID);
46 });
47 connectionToWebProcess.connection().addWorkQueueMessageReceiver(Messages::RTCDataChannelRemoteManagerProxy::messageReceiverName(), m_queue, this);
48}
49
50void RTCDataChannelRemoteManagerProxy::unregisterConnectionToWebProcess(NetworkConnectionToWebProcess& connectionToWebProcess)
51{
52 m_queue->dispatch([this, protectedThis = makeRef(*this), identifier = connectionToWebProcess.webProcessIdentifier()] {
53 ASSERT(m_webProcessConnections.contains(identifier));
54 m_webProcessConnections.remove(identifier);
55 });
56 connectionToWebProcess.connection().removeWorkQueueMessageReceiver(Messages::RTCDataChannelRemoteManagerProxy::messageReceiverName());
57}

59void RTCDataChannelRemoteManagerProxy::sendData(WebCore::RTCDataChannelIdentifier identifier, bool isRaw, const IPC::DataReference& data)
60{
61 if (auto connectionID = m_webProcessConnections.get(identifier.processIdentifier))
62 IPC::Connection::send(connectionID, Messages::RTCDataChannelRemoteManager::SendData { identifier, isRaw, data }, 0);
63}
