36void ArgumentCoder<WallTime>::encode(Encoder& encoder, const WallTime& time)
37{
38 encoder << time.secondsSinceEpoch().value();
39}
40
41WARN_UNUSED_RETURN bool ArgumentCoder<WallTime>::decode(Decoder& decoder, WallTime& time)
42{
43 double value;
44 if (!decoder.decode(value))
45 return false;
46
47 time = WallTime::fromRawSeconds(value);
48 return true;
49}
50
51WARN_UNUSED_RETURN std::optional<WallTime> ArgumentCoder<WallTime>::decode(Decoder& decoder)
52{
53 std::optional<double> time;
54 decoder >> time;
55 if (!time)
56 return std::nullopt;
57 return WallTime::fromRawSeconds(*time);
58}

75void ArgumentCoder<CString>::encode(Encoder& encoder, const CString& string)
76{
77 // Special case the null string.
78 if (string.isNull()) {
79 encoder << std::numeric_limits<uint32_t>::max();
80 return;
81 }
82
83 uint32_t length = string.length();
84 encoder << length;
85 encoder.encodeFixedLengthData(string.dataAsUInt8Ptr(), length, 1);
86}
87
88WARN_UNUSED_RETURN bool ArgumentCoder<CString>::decode(Decoder& decoder, CString& result)
89{
90 uint32_t length;
91 if (!decoder.decode(length))
92 return false;
93
94 if (length == std::numeric_limits<uint32_t>::max()) {
95 // This is the null string.
96 result = CString();
97 return true;
98 }
99
100 // Before allocating the string, make sure that the decoder buffer is big enough.
101 if (!decoder.bufferIsLargeEnoughToContain<char>(length))
102 return false;
103
104 char* buffer;
105 CString string = CString::newUninitialized(length, buffer);
106 if (!decoder.decodeFixedLengthData(reinterpret_cast<uint8_t*>(buffer), length, 1))
107 return false;
108
109 result = string;
110 return true;
111}

113template<typename Encoder>
114void ArgumentCoder<String>::encode(Encoder& encoder, const String& string)
115{
116 // Special case the null string.
117 if (string.isNull()) {
118 encoder << std::numeric_limits<uint32_t>::max();
119 return;
120 }
121
122 uint32_t length = string.length();
123 bool is8Bit = string.is8Bit();
124
125 encoder << length << is8Bit;
126
127 if (is8Bit)
128 encoder.encodeFixedLengthData(string.characters8(), length * sizeof(LChar), alignof(LChar));
129 else
130 encoder.encodeFixedLengthData(reinterpret_cast<const uint8_t*>(string.characters16()), length * sizeof(UChar), alignof(UChar));
131}

135void ArgumentCoder<String>::encode<StreamConnectionEncoder>(StreamConnectionEncoder&, const String&);

137template <typename CharacterType>
138static inline std::optional<String> decodeStringText(Decoder& decoder, uint32_t length)
139{
140 // Before allocating the string, make sure that the decoder buffer is big enough.
141 if (!decoder.bufferIsLargeEnoughToContain<CharacterType>(length))
142 return std::nullopt;
143
144 CharacterType* buffer;
145 String string = String::createUninitialized(length, buffer);
146 if (!decoder.decodeFixedLengthData(reinterpret_cast<uint8_t*>(buffer), length * sizeof(CharacterType), alignof(CharacterType)))
147 return std::nullopt;
148
149 return string;
150}
151
152WARN_UNUSED_RETURN std::optional<String> ArgumentCoder<String>::decode(Decoder& decoder)
153{
154 uint32_t length;
155 if (!decoder.decode(length))
156 return std::nullopt;
157
158 if (length == std::numeric_limits<uint32_t>::max()) {
159 // This is the null string.
160 return String();
161 }
162
163 bool is8Bit;
164 if (!decoder.decode(is8Bit))
165 return std::nullopt;
166
167 if (is8Bit)
168 return decodeStringText<LChar>(decoder, length);
169 return decodeStringText<UChar>(decoder, length);
170}
171
172WARN_UNUSED_RETURN bool ArgumentCoder<String>::decode(Decoder& decoder, String& result)
173{
174 std::optional<String> string;
175 decoder >> string;
176 if (!string)
177 return false;
178 result = WTFMove(*string);
179 return true;
180}

182void ArgumentCoder<SHA1::Digest>::encode(Encoder& encoder, const SHA1::Digest& digest)
183{
184 encoder.encodeFixedLengthData(digest.data(), sizeof(digest), 1);
185}
186
187WARN_UNUSED_RETURN bool ArgumentCoder<SHA1::Digest>::decode(Decoder& decoder, SHA1::Digest& digest)
188{
189 return decoder.decodeFixedLengthData(digest.data(), sizeof(digest), 1);
190}

192#if HAVE(AUDIT_TOKEN)
193void ArgumentCoder<audit_token_t>::encode(Encoder& encoder, const audit_token_t& auditToken)
194{
195 for (unsigned i = 0; i < WTF_ARRAY_LENGTH(auditToken.val); i++)
196 encoder << auditToken.val[i];
197}
198
199WARN_UNUSED_RETURN bool ArgumentCoder<audit_token_t>::decode(Decoder& decoder, audit_token_t& auditToken)
200{
201 for (unsigned i = 0; i < WTF_ARRAY_LENGTH(auditToken.val); i++) {
202 if (!decoder.decode(auditToken.val[i]))
203 return false;
204 }
205 return true;
206}
