61enum class SendOption {
62 // Whether this message should be dispatched when waiting for a sync reply.
63 // This is the default for synchronous messages.
64 DispatchMessageEvenWhenWaitingForSyncReply = 1 << 0,
65 DispatchMessageEvenWhenWaitingForUnboundedSyncReply = 1 << 1,
66 IgnoreFullySynchronousMode = 1 << 2,
67};
68
69enum class SendSyncOption {
70 // Use this to inform that this sync call will suspend this process until the user responds with input.
71 InformPlatformProcessWillSuspend = 1 << 0,
72 UseFullySynchronousModeForTesting = 1 << 1,
73 ForceDispatchWhenDestinationIsWaitingForUnboundedSyncReply = 1 << 2,
74 MaintainOrderingWithAsyncMessages = 1 << 3,
75};
76
77enum class WaitForOption {
78 // Use this to make waitForMessage be interrupted immediately by any incoming sync messages.
79 InterruptWaitingIfSyncMessageArrives = 1 << 0,
80 DispatchIncomingSyncMessagesWhileWaiting = 1 << 1,
81};

89 return; \
90 } \
91} while (0)

97 } \
98} while (0)

537template<typename T, typename C>
538uint64_t Connection::sendWithAsyncReply(T&& message, C&& completionHandler, uint64_t destinationID, OptionSet<SendOption> sendOptions)
539{
540 COMPILE_ASSERT(!T::isSync, AsyncMessageExpected);
541
542 auto encoder = makeUniqueRef<Encoder>(T::name(), destinationID);
543 uint64_t listenerID = nextAsyncReplyHandlerID();
544 addAsyncReplyHandler(*this, listenerID, CompletionHandler<void(Decoder*)>([completionHandler = WTFMove(completionHandler)] (Decoder* decoder) mutable {
545 if (decoder && decoder->isValid())
546 T::callReply(*decoder, WTFMove(completionHandler));
547 else
548 T::cancelReply(WTFMove(completionHandler));
549 }, CompletionHandlerCallThread::MainThread));
550 encoder.get() << listenerID;
551 encoder.get() << message.arguments();
552 sendMessage(WTFMove(encoder), sendOptions);
553 return listenerID;
554}

575template<typename T> Connection::SendSyncResult Connection::sendSync(T&& message, typename T::Reply&& reply, uint64_t destinationID, Timeout timeout, OptionSet<SendSyncOption> sendSyncOptions)
576{
577 COMPILE_ASSERT(T::isSync, SyncMessageExpected);
578 RELEASE_ASSERT(RunLoop::isMain());
579
580 SyncRequestID syncRequestID;
581 auto encoder = createSyncMessageEncoder(T::name(), destinationID, syncRequestID);
582
583 if (sendSyncOptions.contains(SendSyncOption::UseFullySynchronousModeForTesting)) {
584 encoder->setFullySynchronousModeForTesting();
585 m_fullySynchronousModeIsAllowedForTesting = true;
586 }
587
588 // Encode the rest of the input arguments.
589 encoder.get() << message.arguments();
590
591 // Now send the message and wait for a reply.
592 std::unique_ptr<Decoder> replyDecoder = sendSyncMessage(syncRequestID, WTFMove(encoder), timeout, sendSyncOptions);
593 if (!replyDecoder)
594 return { };
595
596 // Decode the reply.
597 std::optional<typename T::ReplyArguments> replyArguments;
598 *replyDecoder >> replyArguments;
599 if (!replyArguments)
600 return { };
601 moveTuple(WTFMove(*replyArguments), reply);
602 return replyDecoder;
603}

605template<typename T> bool Connection::waitForAndDispatchImmediately(uint64_t destinationID, Timeout timeout, OptionSet<WaitForOption> waitForOptions)
606{
607 RELEASE_ASSERT(RunLoop::isMain());
608 std::unique_ptr<Decoder> decoder = waitForMessage(T::name(), destinationID, timeout, waitForOptions);
609 if (!decoder)
610 return false;
611
612 ASSERT(decoder->destinationID() == destinationID);
613 m_client.didReceiveMessage(*this, *decoder);
614 return true;
615}
616
617template<typename T> bool Connection::waitForAsyncCallbackAndDispatchImmediately(uint64_t destinationID, Timeout timeout)
618{
619 RELEASE_ASSERT(RunLoop::isMain());
620 std::unique_ptr<Decoder> decoder = waitForMessage(T::asyncMessageReplyName(), destinationID, timeout, { });
621 if (!decoder)
622 return false;
623
624 ASSERT(decoder->messageReceiverName() == ReceiverName::AsyncReply);
625 ASSERT(decoder->destinationID() == destinationID);
626 auto handler = takeAsyncReplyHandler(*this, decoder->destinationID());
627 if (!handler) {
628 ASSERT_NOT_REACHED();
629 return false;
630 }
631 handler(decoder.get());
632 return true;
633}

636public:

641 }

643 ~UnboundedSynchronousIPCScope()

645 ASSERT(RunLoop::isMain());
646 ASSERT(unboundedSynchronousIPCCount);

653 }
