42static const uint8_t* copyBuffer(const uint8_t* buffer, size_t bufferSize)
43{
44 uint8_t* bufferCopy;
45 if (!tryFastMalloc(bufferSize).getValue(bufferCopy)) {
46 RELEASE_LOG_FAULT(IPC, "Decoder::copyBuffer: tryFastMalloc(%lu) failed", bufferSize);
47 return nullptr;
48 }
49
50 memcpy(bufferCopy, buffer, bufferSize);
51 return bufferCopy;
52}
53
54std::unique_ptr<Decoder> Decoder::create(const uint8_t* buffer, size_t bufferSize, void (*bufferDeallocator)(const uint8_t*, size_t), Vector<Attachment>&& attachments)
55{
56 ASSERT(buffer);
57 if (UNLIKELY(!buffer)) {
58 RELEASE_LOG_FAULT(IPC, "Decoder::create() called with a null buffer (bufferSize: %lu)", bufferSize);
59 return nullptr;
60 }
61
62 const uint8_t* bufferCopy;
63 if (!bufferDeallocator) {
64 bufferCopy = copyBuffer(buffer, bufferSize);
65 ASSERT(bufferCopy);
66 if (UNLIKELY(!bufferCopy))
67 return nullptr;
68 } else
69 bufferCopy = buffer;
70
71 auto decoder = std::unique_ptr<Decoder>(new Decoder(bufferCopy, bufferSize, bufferDeallocator, WTFMove(attachments)));
72 return decoder->isValid() ? WTFMove(decoder) : nullptr;
73}

75Decoder::Decoder(const uint8_t* buffer, size_t bufferSize, void (*bufferDeallocator)(const uint8_t*, size_t), Vector<Attachment>&& attachments)
76 : m_buffer { buffer }
77 , m_bufferPos { m_buffer }
78 , m_bufferEnd { m_buffer + bufferSize }
79 , m_bufferDeallocator { bufferDeallocator }
80 , m_attachments { WTFMove(attachments) }
81{
82 if (UNLIKELY(reinterpret_cast<uintptr_t>(m_buffer) % alignof(uint64_t))) {
83 markInvalid();
84 return;
85 }
86
87 if (UNLIKELY(!decode(m_messageFlags)))
88 return;
89
90 if (UNLIKELY(!decode(m_messageName)))
91 return;
92
93 if (UNLIKELY(!decode(m_destinationID)))
94 return;
95}

118Decoder::~Decoder()
119{
120 ASSERT(m_buffer);
121
122 if (m_bufferDeallocator)
123 m_bufferDeallocator(m_buffer, m_bufferEnd - m_buffer);
124 else
125 fastFree(const_cast<uint8_t*>(m_buffer));
126
127 // FIXME: We need to dispose of the mach ports in cases of failure.
128}

156std::unique_ptr<Decoder> Decoder::unwrapForTesting(Decoder& decoder)
157{
158 ASSERT(decoder.isSyncMessage());
159
160 Vector<Attachment> attachments;
161 Attachment attachment;
162 while (decoder.removeAttachment(attachment))
163 attachments.append(WTFMove(attachment));
164 attachments.reverse();
165
166 DataReference wrappedMessage;
167 if (!decoder.decode(wrappedMessage))
168 return nullptr;
169
170 return Decoder::create(wrappedMessage.data(), wrappedMessage.size(), nullptr, WTFMove(attachments));
171}

173static inline const uint8_t* roundUpToAlignment(const uint8_t* ptr, size_t alignment)
174{
175 // Assert that the alignment is a power of 2.
176 ASSERT(alignment && !(alignment & (alignment - 1)));
177
178 uintptr_t alignmentMask = alignment - 1;
179 return reinterpret_cast<uint8_t*>((reinterpret_cast<uintptr_t>(ptr) + alignmentMask) & ~alignmentMask);
180}

182static inline bool alignedBufferIsLargeEnoughToContain(const uint8_t* alignedPosition, const uint8_t* bufferStart, const uint8_t* bufferEnd, size_t size)
183{
184 // When size == 0 for the last argument and it's a variable length byte array,
185 // bufferStart == alignedPosition == bufferEnd, so checking (bufferEnd >= alignedPosition)
186 // is not an off-by-one error since (static_cast<size_t>(bufferEnd - alignedPosition) >= size)
187 // will catch issues when size != 0.
188 return bufferEnd >= alignedPosition && bufferStart <= alignedPosition && static_cast<size_t>(bufferEnd - alignedPosition) >= size;
189}
190
191bool Decoder::alignBufferPosition(size_t alignment, size_t size)
192{
193 const uint8_t* alignedPosition = roundUpToAlignment(m_bufferPos, alignment);
194 if (UNLIKELY(!alignedBufferIsLargeEnoughToContain(alignedPosition, m_buffer, m_bufferEnd, size))) {
195 // We've walked off the end of this buffer.
196 markInvalid();
197 return false;
198 }
199
200 m_bufferPos = alignedPosition;
201 return true;
202}
203
204bool Decoder::bufferIsLargeEnoughToContain(size_t alignment, size_t size) const
205{
206 return alignedBufferIsLargeEnoughToContain(roundUpToAlignment(m_bufferPos, alignment), m_buffer, m_bufferEnd, size);
207}
208
209bool Decoder::decodeFixedLengthData(uint8_t* data, size_t size, size_t alignment)
210{
211 if (!alignBufferPosition(alignment, size))
212 return false;
213
214 memcpy(data, m_bufferPos, size);
215 m_bufferPos += size;
216
217 return true;
218}
219
220const uint8_t* Decoder::decodeFixedLengthReference(size_t size, size_t alignment)
221{
222 if (!alignBufferPosition(alignment, size))
223 return nullptr;
224
225 const uint8_t* data = m_bufferPos;
226 m_bufferPos += size;
227
228 return data;
229}

231bool Decoder::removeAttachment(Attachment& attachment)
232{
233 if (m_attachments.isEmpty())
234 return false;
235
236 attachment = m_attachments.takeLast();
237 return true;
238}
