42static const uint8_t defaultMessageFlags = 0;
43
44template <typename T>
45static inline bool allocBuffer(T*& buffer, size_t size)
46{
47#if OS(DARWIN)
48 buffer = static_cast<T*>(mmap(0, size, PROT_READ | PROT_WRITE, MAP_ANON | MAP_PRIVATE, -1, 0));
49 return buffer != MAP_FAILED;
50#else
51 buffer = static_cast<T*>(fastMalloc(size));
52 return !!buffer;
53#endif
54}
55
56static inline void freeBuffer(void* addr, size_t size)
57{
58#if OS(DARWIN)
59 munmap(addr, size);
60#else
61 UNUSED_PARAM(size);
62 fastFree(addr);
63#endif
64}

94ShouldDispatchWhenWaitingForSyncReply Encoder::shouldDispatchMessageWhenWaitingForSyncReply() const
95{
96 if (messageFlags().contains(MessageFlags::DispatchMessageWhenWaitingForSyncReply))
97 return ShouldDispatchWhenWaitingForSyncReply::Yes;
98 if (messageFlags().contains(MessageFlags::DispatchMessageWhenWaitingForUnboundedSyncReply))
99 return ShouldDispatchWhenWaitingForSyncReply::YesDuringUnboundedIPC;
100 return ShouldDispatchWhenWaitingForSyncReply::No;
101}
102
103void Encoder::setShouldDispatchMessageWhenWaitingForSyncReply(ShouldDispatchWhenWaitingForSyncReply shouldDispatchWhenWaitingForSyncReply)
104{
105 switch (shouldDispatchWhenWaitingForSyncReply) {
106 case ShouldDispatchWhenWaitingForSyncReply::No:
107 messageFlags().remove(MessageFlags::DispatchMessageWhenWaitingForSyncReply);
108 messageFlags().remove(MessageFlags::DispatchMessageWhenWaitingForUnboundedSyncReply);
109 break;
110 case ShouldDispatchWhenWaitingForSyncReply::Yes:
111 messageFlags().add(MessageFlags::DispatchMessageWhenWaitingForSyncReply);
112 messageFlags().remove(MessageFlags::DispatchMessageWhenWaitingForUnboundedSyncReply);
113 break;
114 case ShouldDispatchWhenWaitingForSyncReply::YesDuringUnboundedIPC:
115 messageFlags().remove(MessageFlags::DispatchMessageWhenWaitingForSyncReply);
116 messageFlags().add(MessageFlags::DispatchMessageWhenWaitingForUnboundedSyncReply);
117 break;
118 }
119}

131void Encoder::wrapForTesting(UniqueRef<Encoder>&& original)
132{
133 ASSERT(isSyncMessage());
134 ASSERT(!original->isSyncMessage());
135
136 original->setShouldDispatchMessageWhenWaitingForSyncReply(ShouldDispatchWhenWaitingForSyncReply::Yes);
137
138 *this << DataReference(original->buffer(), original->bufferSize());
139
140 Vector<Attachment> attachments = original->releaseAttachments();
141 reserve(attachments.size());
142 for (Attachment& attachment : attachments)
143 addAttachment(WTFMove(attachment));
144}

146static inline size_t roundUpToAlignment(size_t value, size_t alignment)
147{
148 return ((value + alignment - 1) / alignment) * alignment;
149}
150
151void Encoder::reserve(size_t size)
152{
153 if (size <= m_bufferCapacity)
154 return;
155
156 size_t newCapacity = roundUpToAlignment(m_bufferCapacity * 2, 4096);
157 while (newCapacity < size)
158 newCapacity *= 2;
159
160 uint8_t* newBuffer;
161 if (!allocBuffer(newBuffer, newCapacity))
162 CRASH();
163
164 memcpy(newBuffer, m_buffer, m_bufferSize);
165
166 if (m_buffer != m_inlineBuffer)
167 freeBuffer(m_buffer, m_bufferCapacity);
168
169 m_buffer = newBuffer;
170 m_bufferCapacity = newCapacity;
171}

173void Encoder::encodeHeader()
174{
175 *this << defaultMessageFlags;
176 *this << m_messageName;
177 *this << m_destinationID;
178}

180OptionSet<MessageFlags>& Encoder::messageFlags()
181{
182 // FIXME: We should probably pass an OptionSet<MessageFlags> into the Encoder constructor instead of encoding defaultMessageFlags then using this to change it later.
183 static_assert(sizeof(OptionSet<MessageFlags>::StorageType) == 1, "Encoder uses the first byte of the buffer for message flags.");
184 return *reinterpret_cast<OptionSet<MessageFlags>*>(buffer());
185}
186
187const OptionSet<MessageFlags>& Encoder::messageFlags() const
188{
189 return *reinterpret_cast<OptionSet<MessageFlags>*>(buffer());
190}

192uint8_t* Encoder::grow(size_t alignment, size_t size)
193{
194 size_t alignedSize = roundUpToAlignment(m_bufferSize, alignment);
195 reserve(alignedSize + size);
196
197 std::memset(m_buffer + m_bufferSize, 0, alignedSize - m_bufferSize);
198
199 m_bufferSize = alignedSize + size;
200 m_bufferPointer = m_buffer + alignedSize + size;
201
202 return m_buffer + alignedSize;
203}
204
205void Encoder::encodeFixedLengthData(const uint8_t* data, size_t size, size_t alignment)
206{
207 ASSERT(!(reinterpret_cast<uintptr_t>(data) % alignment));
208
209 uint8_t* buffer = grow(alignment, size);
210 memcpy(buffer, data, size);
211}

218Vector<Attachment> Encoder::releaseAttachments()
219{

223bool Encoder::hasAttachments() const
