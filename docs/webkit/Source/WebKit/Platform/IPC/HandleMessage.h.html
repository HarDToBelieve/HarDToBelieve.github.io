38// Dispatch functions with no reply arguments.
39
40template <typename C, typename MF, typename ArgsTuple, size_t... ArgsIndex>
41void callMemberFunctionImpl(C* object, MF function, ArgsTuple&& args, std::index_sequence<ArgsIndex...>)
42{
43 (object->*function)(std::get<ArgsIndex>(std::forward<ArgsTuple>(args))...);
44}
45
46template<typename C, typename MF, typename ArgsTuple, typename ArgsIndicies = std::make_index_sequence<std::tuple_size<ArgsTuple>::value>>
47void callMemberFunction(ArgsTuple&& args, C* object, MF function)
48{
49 callMemberFunctionImpl(object, function, std::forward<ArgsTuple>(args), ArgsIndicies());
50}

56{
57 (object->*function)(std::get<ArgsIndex>(std::forward<ArgsTuple>(args))..., WTFMove(completionHandler));

96template<typename T>
97struct CodingType {
98 typedef std::remove_const_t<std::remove_reference_t<T>> Type;
99};
100
101class SharedBufferDataReference;
102template<> struct CodingType<const SharedBufferDataReference&> {
103 using Type = DataReference;
104};
105
106template<typename... Ts>
107struct CodingType<std::tuple<Ts...>> {
108 typedef std::tuple<typename CodingType<Ts>::Type...> Type;
109};
110
111template<typename T, typename C, typename MF>
112void handleMessage(Decoder& decoder, C* object, MF function)
113{
114 std::optional<typename CodingType<typename T::Arguments>::Type> arguments;
115 decoder >> arguments;
116 if (UNLIKELY(!arguments))
117 return;
118
119 callMemberFunction(WTFMove(*arguments), object, function);
120}
121
122template<typename T, typename C, typename MF>
123void handleMessage(Connection& connection, Decoder& decoder, C* object, MF function)
124{
125 std::optional<typename CodingType<typename T::Arguments>::Type> arguments;
126 decoder >> arguments;
127 if (UNLIKELY(!arguments))
128 return;
129 callMemberFunction(connection, WTFMove(*arguments), object, function);
130}

132template<typename T, typename C, typename MF>
133bool handleMessageSynchronous(Connection& connection, Decoder& decoder, UniqueRef<Encoder>& replyEncoder, C* object, MF function)
134{
135 std::optional<typename CodingType<typename T::Arguments>::Type> arguments;
136 decoder >> arguments;
137 if (UNLIKELY(!arguments))
138 return false;
139
140 typename T::DelayedReply completionHandler = [replyEncoder = WTFMove(replyEncoder), connection = makeRef(connection)] (auto&&... args) mutable {
141 T::send(WTFMove(replyEncoder), WTFMove(connection), args...);
142 };
143 callMemberFunction(WTFMove(*arguments), WTFMove(completionHandler), object, function);
144 return true;
145}

162template<typename T, typename C, typename MF>
163void handleMessageSynchronous(StreamServerConnectionBase& connection, Decoder& decoder, C* object, MF function)
164{
165 Connection::SyncRequestID syncRequestID;
166 if (UNLIKELY(!decoder.decode(syncRequestID)))
167 return;
168
169 std::optional<typename CodingType<typename T::Arguments>::Type> arguments;
170 decoder >> arguments;
171 if (UNLIKELY(!arguments))
172 return;
173
174 typename T::DelayedReply completionHandler = [syncRequestID, connection = makeRef(connection)] (auto&&... args) mutable {
175 connection->sendSyncReply<T>(syncRequestID, args...);
176 };
177 callMemberFunction(WTFMove(*arguments), WTFMove(completionHandler), object, function);
178}
179
180template<typename T, typename C, typename MF>
181void handleMessageAsync(Connection& connection, Decoder& decoder, C* object, MF function)
182{
183 std::optional<uint64_t> listenerID;
184 decoder >> listenerID;
185 if (!listenerID)
186 return;
187
188 std::optional<typename CodingType<typename T::Arguments>::Type> arguments;
189 decoder >> arguments;
190 if (UNLIKELY(!arguments))
191 return;
192
193 typename T::AsyncReply completionHandler = { [listenerID = *listenerID, connection = makeRef(connection)] (auto&&... args) mutable {
194 auto encoder = makeUniqueRef<Encoder>(T::asyncMessageReplyName(), listenerID);
195 T::send(WTFMove(encoder), WTFMove(connection), args...);
196 }, T::callbackThread };
197 callMemberFunction(WTFMove(*arguments), WTFMove(completionHandler), object, function);
198}
199
200template<typename T, typename C, typename MF>
201void handleMessageAsyncWantsConnection(Connection& connection, Decoder& decoder, C* object, MF function)
202{
203 std::optional<typename CodingType<typename T::Arguments>::Type> arguments;
204 decoder >> arguments;
205 if (UNLIKELY(!arguments))
206 return;
207
208 typename T::AsyncReply completionHandler = [listenerID = decoder.destinationID(), connection = makeRef(connection)] (auto&&... args) mutable {
209 auto encoder = makeUniqueRef<Encoder>(T::asyncMessageReplyName(), listenerID);
210 T::send(WTFMove(encoder), WTFMove(connection), args...);
211 };
212 callMemberFunction(connection, WTFMove(*arguments), WTFMove(completionHandler), object, function);
213}
