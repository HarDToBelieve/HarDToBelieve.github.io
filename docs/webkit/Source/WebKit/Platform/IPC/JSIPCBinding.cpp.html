40static JSC::JSValue jsValueForDecodedStringArgumentValue(JSC::JSGlobalObject* globalObject, const String& value, ASCIILiteral type)
41{
42 auto& vm = globalObject->vm();
43 auto scope = DECLARE_THROW_SCOPE(vm);
44 auto* object = JSC::constructEmptyObject(globalObject, globalObject->objectPrototype());
45 RETURN_IF_EXCEPTION(scope, JSC::JSValue());
46 object->putDirect(vm, JSC::Identifier::fromString(vm, "type"_s), JSC::jsNontrivialString(vm, type));
47 RETURN_IF_EXCEPTION(scope, JSC::JSValue());
48 object->putDirect(vm, JSC::Identifier::fromString(vm, "value"_s), value.isNull() ? JSC::jsNull() : JSC::jsString(vm, value));
49 RETURN_IF_EXCEPTION(scope, JSC::JSValue());
50 return object;
51}

167{

197bool putJSValueForDecodedArgumentAtIndexOrArrayBufferIfUndefined(JSC::JSGlobalObject* globalObject, JSC::JSArray* array, unsigned index, JSC::JSValue value, const uint8_t* buffer, size_t length)
198{
199 auto scope = DECLARE_THROW_SCOPE(globalObject->vm());
200
201 if (value.isUndefined()) {
202 auto arrayBuffer = JSC::ArrayBuffer::create(buffer, length);
203 if (auto* structure = globalObject->arrayBufferStructure(arrayBuffer->sharingMode()))
204 value = JSC::JSArrayBuffer::create(globalObject->vm(), structure, WTFMove(arrayBuffer));
205 }
206
207 array->putDirectIndex(globalObject, index, value);
208 RETURN_IF_EXCEPTION(scope, false);
209
210 return true;
211}
