26#pragma once
27
28#include "Decoder.h"
29#include "MessageReceiveQueue.h"
30#include <wtf/HashMap.h>
31#include <wtf/Variant.h>
32
33namespace IPC {
34
35enum class ReceiverName : uint8_t;
36
37class MessageReceiveQueueMap {
38public:
39 MessageReceiveQueueMap() = default;
40 ~MessageReceiveQueueMap() = default;
41
42 static bool isValidMessage(const Decoder& message)
43 {
44 return QueueMap::isValidKey(std::make_pair(static_cast<uint8_t>(message.messageReceiverName()), message.destinationID()));
45 }
46
47 // Adds a wildcard filter if destinationID == 0.
48 void add(MessageReceiveQueue& queue, ReceiverName receiverName, uint64_t destinationID) { addImpl(StoreType(&queue), receiverName, destinationID); }
49 void add(std::unique_ptr<MessageReceiveQueue>&& queue, ReceiverName receiverName, uint64_t destinationID) { addImpl(StoreType(WTFMove(queue)), receiverName, destinationID); }
50 void remove(ReceiverName, uint64_t destinationID);
51
52 MessageReceiveQueue* get(const Decoder&) const;
53private:
54 using StoreType = Variant<MessageReceiveQueue*, std::unique_ptr<MessageReceiveQueue>>;
55 void addImpl(StoreType&&, ReceiverName, uint64_t destinationID);
56 using QueueMap = HashMap<std::pair<uint8_t, uint64_t>, StoreType>;
57 QueueMap m_queues;
58};
59
60} // namespace IPC
