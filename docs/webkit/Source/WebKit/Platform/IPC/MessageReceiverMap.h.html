28#include "StringReference.h"
29#include <wtf/HashMap.h>

33namespace IPC {
34
35class Connection;
36class Encoder;
37class Decoder;
38class MessageReceiver;
39enum class ReceiverName : uint8_t;
40
41class MessageReceiverMap {
42public:
43 MessageReceiverMap();
44 ~MessageReceiverMap();
45
46 void addMessageReceiver(ReceiverName, MessageReceiver&);
47 void addMessageReceiver(ReceiverName, uint64_t destinationID, MessageReceiver&);
48
49 void removeMessageReceiver(ReceiverName);
50 void removeMessageReceiver(ReceiverName, uint64_t destinationID);
51 void removeMessageReceiver(MessageReceiver&);
52
53 void invalidate();
54
55 bool dispatchMessage(Connection&, Decoder&);
56 bool dispatchSyncMessage(Connection&, Decoder&, UniqueRef<Encoder>&);
57
58private:
59 // Message receivers that don't require a destination ID.
60 HashMap<ReceiverName, WeakPtr<MessageReceiver>, WTF::IntHash<ReceiverName>, WTF::StrongEnumHashTraits<ReceiverName>> m_globalMessageReceivers;
61
62 HashMap<std::pair<ReceiverName, uint64_t>, WeakPtr<MessageReceiver>, DefaultHash<std::pair<ReceiverName, uint64_t>>, PairHashTraits<WTF::StrongEnumHashTraits<ReceiverName>, HashTraits<uint64_t>>> m_messageReceivers;
63};
64
65};

70template<> struct DefaultHash<IPC::ReceiverName> : IntHash<IPC::ReceiverName> { };

72}
