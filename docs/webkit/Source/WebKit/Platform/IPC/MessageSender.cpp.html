29namespace IPC {
30
31MessageSender::~MessageSender() = default;
32
33bool MessageSender::sendMessage(UniqueRef<Encoder>&& encoder, OptionSet<SendOption> sendOptions, std::optional<std::pair<CompletionHandler<void(IPC::Decoder*)>, uint64_t>>&& asyncReplyInfo)
34{
35 auto* connection = messageSenderConnection();
36 ASSERT(connection);
37
38 if (asyncReplyInfo)
39 IPC::addAsyncReplyHandler(*connection, asyncReplyInfo->second, WTFMove(asyncReplyInfo->first));
40
41 return connection->sendMessage(WTFMove(encoder), sendOptions);
42}
