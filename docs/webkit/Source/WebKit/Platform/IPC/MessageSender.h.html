26#pragma once
27
28#include <wtf/Assertions.h>
29#include "Connection.h"
30#include <wtf/UniqueRef.h>
31
32namespace IPC {
33
34class MessageSender {
35public:
36 virtual ~MessageSender();
37
38 template<typename U> bool send(const U& message, OptionSet<SendOption> sendOptions = { })
39 {
40 return send(message, messageSenderDestinationID(), sendOptions);
41 }
42
43 template<typename U> bool send(const U& message, uint64_t destinationID, OptionSet<SendOption> sendOptions = { })
44 {
45 static_assert(!U::isSync, "Message is sync!");
46
47 auto encoder = makeUniqueRef<Encoder>(U::name(), destinationID);
48 encoder.get() << message.arguments();
49
50 return sendMessage(WTFMove(encoder), sendOptions);
51 }
52
53 template<typename U, typename T>
54 bool send(const U& message, ObjectIdentifier<T> destinationID, OptionSet<SendOption> sendOptions = { })
55 {
56 return send<U>(message, destinationID.toUInt64(), sendOptions);
57 }
58 using SendSyncResult = Connection::SendSyncResult;
59
60 template<typename T>
61 SendSyncResult sendSync(T&& message, typename T::Reply&& reply, Timeout timeout = Seconds::infinity(), OptionSet<SendSyncOption> sendSyncOptions = { })
62 {
63 static_assert(T::isSync, "Message is not sync!");
64
65 return sendSync(std::forward<T>(message), WTFMove(reply), messageSenderDestinationID(), timeout, sendSyncOptions);
66 }
67
68 template<typename T>
69 SendSyncResult sendSync(T&& message, typename T::Reply&& reply, uint64_t destinationID, Timeout timeout = Timeout::infinity(), OptionSet<SendSyncOption> sendSyncOptions = { })
70 {
71 if (auto* connection = messageSenderConnection())
72 return connection->sendSync(WTFMove(message), WTFMove(reply), destinationID, timeout, sendSyncOptions);
73
74 return { };
75 }
76
77 template<typename U, typename T>
78 SendSyncResult sendSync(U&& message, typename U::Reply&& reply, ObjectIdentifier<T> destinationID, Timeout timeout = Timeout::infinity(), OptionSet<SendSyncOption> sendSyncOptions = { })
79 {
80 return sendSync<U>(std::forward<U>(message), WTFMove(reply), destinationID.toUInt64(), timeout, sendSyncOptions);
81 }
82
83 template<typename T, typename C>
84 uint64_t sendWithAsyncReply(T&& message, C&& completionHandler, OptionSet<SendOption> sendOptions = { })
85 {
86 return sendWithAsyncReply(WTFMove(message), WTFMove(completionHandler), messageSenderDestinationID(), sendOptions);
87 }
88
89 template<typename T, typename C>
90 uint64_t sendWithAsyncReply(T&& message, C&& completionHandler, uint64_t destinationID, OptionSet<SendOption> sendOptions = { })
91 {
92 COMPILE_ASSERT(!T::isSync, AsyncMessageExpected);
93
94 auto encoder = makeUniqueRef<IPC::Encoder>(T::name(), destinationID);
95 uint64_t listenerID = IPC::nextAsyncReplyHandlerID();
96 encoder.get() << listenerID;
97 encoder.get() << message.arguments();
98 sendMessage(WTFMove(encoder), sendOptions, {{ [completionHandler = WTFMove(completionHandler)] (IPC::Decoder* decoder) mutable {
99 if (decoder && decoder->isValid())
100 T::callReply(*decoder, WTFMove(completionHandler));
101 else
102 T::cancelReply(WTFMove(completionHandler));
103 }, listenerID }});
104 return listenerID;
105 }
106
107 virtual bool sendMessage(UniqueRef<Encoder>&&, OptionSet<SendOption>, std::optional<std::pair<CompletionHandler<void(IPC::Decoder*)>, uint64_t>>&& = std::nullopt);
108
109private:
110 virtual Connection* messageSenderConnection() const = 0;
111 virtual uint64_t messageSenderDestinationID() const = 0;
112};
