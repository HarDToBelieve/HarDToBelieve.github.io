38void SharedBufferCopy::encode(Encoder& encoder) const
39{
40 uint64_t bufferSize = m_buffer ? m_buffer->size() : 0;
41 encoder.reserve(sizeof(bufferSize) + bufferSize);
42 encoder << bufferSize;
43 if (!bufferSize)
44 return;
45 for (auto& segment : *m_buffer)
46 encoder.encodeFixedLengthData(segment.segment->data(), segment.segment->size(), 1);
47}
48
49std::optional<SharedBufferCopy> SharedBufferCopy::decode(Decoder& decoder)
50{
51 IPC::DataReference data;
52 if (!decoder.decode(data))
53 return std::nullopt;
54 RefPtr<SharedBuffer> buffer;
55 if (data.size())
56 buffer = SharedBuffer::create(data.data(), data.size());
57 return { WTFMove(buffer) };
58}
