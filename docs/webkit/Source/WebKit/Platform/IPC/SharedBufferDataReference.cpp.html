32namespace IPC {

34void SharedBufferDataReference::encode(Encoder& encoder) const
35{
36 if (m_data.isEmpty()) {
37 SharedBufferCopy::encode(encoder);
38 return;
39 }
40
41 ASSERT(!buffer());
42 encoder << m_data;
43}
