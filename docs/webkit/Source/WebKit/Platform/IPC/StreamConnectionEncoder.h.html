28#include "ArgumentCoders.h"
29#include "MessageNames.h"
30#include <wtf/StdLibExtras.h>

32namespace IPC {
33
34template<typename, typename> struct ArgumentCoder;
35
36// IPC encoder which:
37// - Encodes to a caller buffer with fixed size, does not resize, stops when size runs out
38// - Does not initialize alignment gaps
39//
40class StreamConnectionEncoder final {
41public:
42 // Stream message needs to be at least size of StreamSetDestinationID message.
43 static constexpr size_t minimumMessageSize = sizeof(MessageName) + sizeof(uint64_t);
44 static constexpr size_t messageAlignment = alignof(MessageName);
45 static constexpr bool isIPCEncoder = true;
46
47 StreamConnectionEncoder(MessageName messageName, uint8_t* stream, size_t streamCapacity)
48 : m_buffer(stream)
49 , m_bufferCapacity(streamCapacity)
50 {
51 *this << messageName;
52 }
53
54 ~StreamConnectionEncoder() = default;
55
56 bool encodeFixedLengthData(const uint8_t* data, size_t size, size_t alignment)
57 {
58 size_t bufferPointer = static_cast<size_t>(reinterpret_cast<intptr_t>(m_buffer + m_encodedSize));
59 size_t newBufferPointer = roundUpToMultipleOf(alignment, bufferPointer);
60 if (newBufferPointer < bufferPointer)
61 return false;
62 intptr_t alignedSize = m_encodedSize + (newBufferPointer - bufferPointer);
63 if (!reserve(alignedSize, size))
64 return false;
65 uint8_t* buffer = m_buffer + alignedSize;
66 memcpy(buffer, data, size);
67 m_encodedSize = alignedSize + size;
68 return true;
69 }
70
71 template<typename T>
72 StreamConnectionEncoder& operator<<(T&& t)
73 {
74 ArgumentCoder<std::remove_const_t<std::remove_reference_t<T>>, void>::encode(*this, std::forward<T>(t));
75 return *this;
76 }
77
78 size_t size() const { ASSERT(isValid()); return m_encodedSize; }
79 bool isValid() const { return m_bufferCapacity; }
80 operator bool() const { return isValid(); }
81private:
82 bool reserve(size_t alignedSize, size_t additionalSize)
83 {
84 size_t size = alignedSize + additionalSize;
85 if (size < alignedSize || size > m_bufferCapacity) {
86 m_bufferCapacity = 0;
87 return false;
88 }
89 return true;
90 }
91 uint8_t* m_buffer;
92 size_t m_bufferCapacity;
93 size_t m_encodedSize { 0 };
94};
95
96} // namespace IPC
