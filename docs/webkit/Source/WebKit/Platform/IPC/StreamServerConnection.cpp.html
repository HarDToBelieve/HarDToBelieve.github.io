41void StreamServerConnectionBase::startReceivingMessagesImpl(ReceiverName receiverName, uint64_t destinationID)
42{
43 m_connection->addMessageReceiveQueue(*this, receiverName, destinationID);
44 m_workQueue.addStreamConnection(*this);
45}

47void StreamServerConnectionBase::stopReceivingMessagesImpl(ReceiverName receiverName, uint64_t destinationID)
48{
49 m_connection->removeMessageReceiveQueue(receiverName, destinationID);
50 m_workQueue.removeStreamConnection(*this);
51}

53void StreamServerConnectionBase::enqueueMessage(Connection&, std::unique_ptr<Decoder>&& message)
54{
55 {
56 Locker locker { m_outOfStreamMessagesLock };
57 m_outOfStreamMessages.append(WTFMove(message));
58 }
59 m_workQueue.wakeUp();
60}

62std::optional<StreamServerConnectionBase::Span> StreamServerConnectionBase::tryAcquire()
63{
64 ServerLimit serverLimit = sharedServerLimit().load(std::memory_order_acquire);
65 if (serverLimit == ServerLimit::serverIsSleepingTag)
66 return std::nullopt;
67
68 auto result = alignedSpan(m_serverOffset, clampedLimit(serverLimit));
69 if (result.size < minimumMessageSize) {
70 serverLimit = sharedServerLimit().compareExchangeStrong(serverLimit, ServerLimit::serverIsSleepingTag, std::memory_order_acq_rel, std::memory_order_acq_rel);
71 result = alignedSpan(m_serverOffset, clampedLimit(serverLimit));
72 }
73
74 if (result.size < minimumMessageSize)
75 return std::nullopt;
76
77 return result;
78}

80StreamServerConnectionBase::Span StreamServerConnectionBase::acquireAll()
81{
82 return alignedSpan(0, dataSize() - 1);
83}

85void StreamServerConnectionBase::release(size_t readSize)
86{
87 ASSERT(readSize);
88 readSize = std::max(readSize, minimumMessageSize);
89 ServerOffset serverOffset = static_cast<ServerOffset>(wrapOffset(alignOffset(m_serverOffset) + readSize));
90
91 ServerOffset oldServerOffset = sharedServerOffset().exchange(serverOffset, std::memory_order_acq_rel);
92 // If the client wrote over serverOffset, it means the client is waiting.
93 if (oldServerOffset == ServerOffset::clientIsWaitingTag)
94 m_buffer.clientWaitSemaphore().signal();
95 else
96 ASSERT(!(oldServerOffset & ServerOffset::clientIsWaitingTag));
97
98 m_serverOffset = serverOffset;
99}
100
101void StreamServerConnectionBase::releaseAll()
102{
103 sharedServerLimit().store(static_cast<ServerLimit>(0), std::memory_order_release);
104 ServerOffset oldServerOffset = sharedServerOffset().exchange(static_cast<ServerOffset>(0), std::memory_order_acq_rel);
105 // If the client wrote over serverOffset, it means the client is waiting.
106 if (oldServerOffset == ServerOffset::clientIsWaitingTag)
107 m_buffer.clientWaitSemaphore().signal();
108 else
109 ASSERT(!(oldServerOffset & ServerOffset::clientIsWaitingTag));
110 m_serverOffset = 0;
111}

113StreamServerConnectionBase::Span StreamServerConnectionBase::alignedSpan(size_t offset, size_t limit)
114{
115 ASSERT(offset < dataSize());
116 ASSERT(limit < dataSize());
117 size_t aligned = alignOffset(offset);
118 size_t resultSize = 0;
119 if (offset < limit) {
120 if (offset <= aligned && aligned < limit)
121 resultSize = size(aligned, limit);
122 } else if (offset > limit) {
123 if (aligned >= offset || aligned < limit)
124 resultSize = size(aligned, limit);
125 }
126 return { data() + aligned, resultSize };
127}
128
129size_t StreamServerConnectionBase::size(size_t offset, size_t limit)
130{
131 if (offset <= limit)
132 return limit - offset;
133 return dataSize() - offset;
134}
135
136size_t StreamServerConnectionBase::clampedLimit(ServerLimit serverLimit) const
137{
138 ASSERT(!(serverLimit & ServerLimit::serverIsSleepingTag));
139 size_t limit = static_cast<size_t>(serverLimit);
140 ASSERT(limit <= dataSize() - 1);
141 return std::min(limit, dataSize() - 1);
142}
