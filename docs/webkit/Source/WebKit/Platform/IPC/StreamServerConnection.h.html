104template<typename T, typename... Arguments>
105void StreamServerConnectionBase::sendSyncReply(Connection::SyncRequestID syncRequestID, Arguments&&... arguments)
106{
107 if constexpr(T::isReplyStreamEncodable) {
108 if (m_isDispatchingStreamMessage) {
109 auto span = acquireAll();
110 {
111 StreamConnectionEncoder messageEncoder { MessageName::SyncMessageReply, span.data, span.size };
112 if ((messageEncoder << ... << arguments))
113 return;
114 }
115 StreamConnectionEncoder outOfStreamEncoder { MessageName::ProcessOutOfStreamMessage, span.data, span.size };
116 }
117 }
118 auto encoder = makeUniqueRef<Encoder>(MessageName::SyncMessageReply, syncRequestID.toUInt64());
119
120 (encoder.get() << ... << arguments);
121 m_connection->sendSyncReply(WTFMove(encoder));
122}

124// StreamServerConnection represents the connection between stream client and server, as used by the server.
125//
126// StreamServerConnection:
127// * Holds the messages towards the server.
128// * Sends the replies back to the client via the stream or normal Connection fallback.
129//
130// Receiver template contract:
131// void didReceiveStreamMessage(StreamServerConnectionBase&, Decoder&);
132//
133// The StreamServerConnection does not trust the StreamClientConnection.
134template<typename Receiver>
135class StreamServerConnection final : public StreamServerConnectionBase {
136public:
137 static Ref<StreamServerConnection> create(Connection& connection, StreamConnectionBuffer&& streamBuffer, StreamConnectionWorkQueue& workQueue)
138 {
139 return adoptRef(*new StreamServerConnection(connection, WTFMove(streamBuffer), workQueue));
140 }
141 ~StreamServerConnection() final = default;
142
143 void startReceivingMessages(Receiver&, ReceiverName, uint64_t destinationID);
144 // Stops the message receipt. Note: already received messages might still be delivered.
145 void stopReceivingMessages(ReceiverName, uint64_t destinationID);
146
147 // StreamServerConnectionBase overrides.
148 DispatchResult dispatchStreamMessages(size_t messageLimit) final;
149
150private:
151 StreamServerConnection(Connection& connection, StreamConnectionBuffer&& streamBuffer, StreamConnectionWorkQueue& workQueue)
152 : StreamServerConnectionBase(connection, WTFMove(streamBuffer), workQueue)
153 {
154 }
155 bool processSetStreamDestinationID(Decoder&&, RefPtr<Receiver>& currentReceiver);
156 bool dispatchStreamMessage(Decoder&&, Receiver&);
157 bool dispatchOutOfStreamMessage(Decoder&&);
158 Lock m_receiversLock;
159 using ReceiversMap = HashMap<std::pair<uint8_t, uint64_t>, Ref<Receiver>>;
160 ReceiversMap m_receivers WTF_GUARDED_BY_LOCK(m_receiversLock);
161 uint64_t m_currentDestinationID { 0 };
162};

164template<typename Receiver>
165void StreamServerConnection<Receiver>::startReceivingMessages(Receiver& receiver, ReceiverName receiverName, uint64_t destinationID)
166{
167 {
168 auto key = std::make_pair(static_cast<uint8_t>(receiverName), destinationID);
169 Locker locker { m_receiversLock };
170 ASSERT(!m_receivers.contains(key));
171 m_receivers.add(key, makeRef(receiver));
172 }
173 StreamServerConnectionBase::startReceivingMessagesImpl(receiverName, destinationID);
174}
175
176template<typename Receiver>
177void StreamServerConnection<Receiver>::stopReceivingMessages(ReceiverName receiverName, uint64_t destinationID)
178{
179 StreamServerConnectionBase::stopReceivingMessagesImpl(receiverName, destinationID);
180 auto key = std::make_pair(static_cast<uint8_t>(receiverName), destinationID);
181 Locker locker { m_receiversLock };
182 ASSERT(m_receivers.contains(key));
183 m_receivers.remove(key);
184}

186template<typename Receiver>
187StreamServerConnectionBase::DispatchResult StreamServerConnection<Receiver>::dispatchStreamMessages(size_t messageLimit)
188{
189 RefPtr<Receiver> currentReceiver;
190 // FIXME: Implement WTF::isValid(ReceiverName).
191 uint8_t currentReceiverName = static_cast<uint8_t>(ReceiverName::Invalid);
192
193 for (size_t i = 0; i < messageLimit; ++i) {
194 auto span = tryAcquire();
195 if (!span)
196 return DispatchResult::HasNoMessages;
197 IPC::Decoder decoder { span->data, span->size, m_currentDestinationID };
198 if (!decoder.isValid()) {
199 m_connection->dispatchDidReceiveInvalidMessage(decoder.messageName());
200 return DispatchResult::HasNoMessages;
201 }
202 if (decoder.messageName() == MessageName::SetStreamDestinationID) {
203 if (!processSetStreamDestinationID(WTFMove(decoder), currentReceiver))
204 return DispatchResult::HasNoMessages;
205 continue;
206 }
207 if (decoder.messageName() == MessageName::ProcessOutOfStreamMessage) {
208 if (!dispatchOutOfStreamMessage(WTFMove(decoder)))
209 return DispatchResult::HasNoMessages;
210 continue;
211 }
212 if (currentReceiverName != static_cast<uint8_t>(decoder.messageReceiverName())) {
213 currentReceiverName = static_cast<uint8_t>(decoder.messageReceiverName());
214 currentReceiver = nullptr;
215 }
216 if (!currentReceiver) {
217 auto key = std::make_pair(static_cast<uint8_t>(currentReceiverName), m_currentDestinationID);
218 if (!ReceiversMap::isValidKey(key)) {
219 m_connection->dispatchDidReceiveInvalidMessage(decoder.messageName());
220 return DispatchResult::HasNoMessages;
221 }
222 Locker locker { m_receiversLock };
223 currentReceiver = m_receivers.get(key);
224 }
225 if (!currentReceiver) {
226 // Valid scenario is when receiver has been removed, but there are messages for it in the buffer.
227 // FIXME: Since we do not have a receiver, we don't know how to decode the message.
228 // This means we must timeout every receiver in the stream connection.
229 // Currently we assert that the receivers are empty, as we only have up to one receiver in
230 // a stream connection until possibility of skipping is implemented properly.
231 Locker locker { m_receiversLock };
232 ASSERT(m_receivers.isEmpty());
233 return DispatchResult::HasNoMessages;
234 }
235 if (!dispatchStreamMessage(WTFMove(decoder), *currentReceiver))
236 return DispatchResult::HasNoMessages;
237 }
238 return DispatchResult::HasMoreMessages;
239}

241template<typename Receiver>
242bool StreamServerConnection<Receiver>::processSetStreamDestinationID(Decoder&& decoder, RefPtr<Receiver>& currentReceiver)
243{
244 uint64_t destinationID = 0;
245 if (!decoder.decode(destinationID)) {
246 m_connection->dispatchDidReceiveInvalidMessage(decoder.messageName());
247 return false;
248 }
249 if (m_currentDestinationID != destinationID) {
250 m_currentDestinationID = destinationID;
251 currentReceiver = nullptr;
252 }
253 release(decoder.currentBufferPosition());
254 return true;
255}

257template<typename Receiver>
258bool StreamServerConnection<Receiver>::dispatchStreamMessage(Decoder&& decoder, Receiver& receiver)
259{
260 ASSERT(!m_isDispatchingStreamMessage);
261 m_isDispatchingStreamMessage = true;
262 receiver.didReceiveStreamMessage(*this, decoder);
263 m_isDispatchingStreamMessage = false;
264 if (!decoder.isValid()) {
265 m_connection->dispatchDidReceiveInvalidMessage(decoder.messageName());
266 return false;
267 }
268 if (decoder.isSyncMessage())
269 releaseAll();
270 else
271 release(decoder.currentBufferPosition());
272 return true;
273}

275template<typename Receiver>
276bool StreamServerConnection<Receiver>::dispatchOutOfStreamMessage(Decoder&& decoder)
277{
278 std::unique_ptr<Decoder> message;
279 {
280 Locker locker { m_outOfStreamMessagesLock };
281 if (m_outOfStreamMessages.isEmpty())
282 return false;
283 message = m_outOfStreamMessages.takeFirst();
284 }
285 if (!message)
286 return false;
287
288 RefPtr<Receiver> receiver;
289 {
290 auto key = std::make_pair(static_cast<uint8_t>(message->messageReceiverName()), message->destinationID());
291 Locker locker { m_receiversLock };
292 receiver = m_receivers.get(key);
293 }
294 if (receiver) {
295 receiver->didReceiveStreamMessage(*this, *message);
296 if (!message->isValid()) {
297 m_connection->dispatchDidReceiveInvalidMessage(message->messageName());
298 return false;
299 }
300 }
301 // If receiver does not exist if it has been removed but messages are still pending to be
302 // processed. It's ok to skip such messages.
303 // FIXME: Note, corresponding skip is not possible at the moment for stream messages.
304 release(decoder.currentBufferPosition());
305 return true;
306}
