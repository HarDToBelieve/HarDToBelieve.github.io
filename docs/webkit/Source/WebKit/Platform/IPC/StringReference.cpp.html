36namespace IPC {
37
38CString StringReference::toString() const
39{
40 return WTF::CString(m_data, m_size);
41}
42
43void StringReference::encode(Encoder& encoder) const
44{
45 encoder << DataReference(reinterpret_cast<const uint8_t*>(m_data), m_size);
46}
47
48bool StringReference::decode(Decoder& decoder, StringReference& result)
49{
50 DataReference dataReference;
51 if (!decoder.decode(dataReference))
52 return false;
53
54 result.m_data = reinterpret_cast<const char*>(dataReference.data());
55 result.m_size = dataReference.size();
56
57 return true;
58}
59
60unsigned StringReference::Hash::hash(const StringReference& a)
61{
62 return StringHasher::computeHash(reinterpret_cast<const unsigned char*>(a.data()), a.size());
63}
