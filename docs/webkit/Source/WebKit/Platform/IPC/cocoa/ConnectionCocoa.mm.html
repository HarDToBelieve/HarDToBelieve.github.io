76// ConnectionTerminationWatchdog does two things:
77// 1) It sets a watchdog timer to kill the peered process.
78// 2) On iOS, make the process runnable for the duration of the watchdog
79// to ensure it has a chance to terminate cleanly.
80class ConnectionTerminationWatchdog {
81public:
82 static void createConnectionTerminationWatchdog(OSObjectPtr<xpc_connection_t>& xpcConnection, Seconds interval)
83 {
84 new ConnectionTerminationWatchdog(xpcConnection, interval);
85 }
86
87private:
88 ConnectionTerminationWatchdog(OSObjectPtr<xpc_connection_t>& xpcConnection, Seconds interval)
89 : m_xpcConnection(xpcConnection)
90 , m_watchdogTimer(RunLoop::main(), this, &ConnectionTerminationWatchdog::watchdogTimerFired)
91#if PLATFORM(IOS_FAMILY)
92 , m_assertion(makeUnique<WebKit::ProcessAndUIAssertion>(xpc_connection_get_pid(m_xpcConnection.get()), "ConnectionTerminationWatchdog"_s, WebKit::ProcessAssertionType::Background))
93#endif
94 {
95 m_watchdogTimer.startOneShot(interval);
96 }
97
98 void watchdogTimerFired()
99 {
100 terminateWithReason(m_xpcConnection.get(), WebKit::ReasonCode::WatchdogTimerFired, "ConnectionTerminationWatchdog::watchdogTimerFired");
101 delete this;
102 }
103
104 OSObjectPtr<xpc_connection_t> m_xpcConnection;
105 RunLoop::Timer<ConnectionTerminationWatchdog> m_watchdogTimer;
106#if PLATFORM(IOS_FAMILY)
107 std::unique_ptr<WebKit::ProcessAndUIAssertion> m_assertion;
108#endif
109};

111void Connection::platformInvalidate()
112{
113 if (!m_isConnected) {
114 if (m_sendPort) {
115 ASSERT(!m_isServer);
116 deallocateSendRightSafely(m_sendPort);
117 m_sendPort = MACH_PORT_NULL;
118 }
119
120 if (m_receiveSource) {
121 // For a short period of time, when m_isServer is true and open() has been called, m_receiveSource has been initialized
122 // but m_isConnected has not been set to true yet. In this case, we need to cancel m_receiveSource instead of destroying
123 // m_receivePort ourselves.
124 ASSERT(m_isServer);
125 cancelReceiveSource();
126 }
127
128 if (m_receivePort) {
129 ASSERT(m_isServer);
130#if !PLATFORM(WATCHOS)
131 mach_port_unguard(mach_task_self(), m_receivePort, reinterpret_cast<mach_port_context_t>(this));
132#endif
133 mach_port_mod_refs(mach_task_self(), m_receivePort, MACH_PORT_RIGHT_RECEIVE, -1);
134 m_receivePort = MACH_PORT_NULL;
135 }
136
137 return;
138 }
139
140 m_pendingOutgoingMachMessage = nullptr;
141 m_isInitializingSendSource = false;
142 m_isConnected = false;
143
144 ASSERT(m_sendPort);
145 ASSERT(m_receivePort);
146
147 // Unregister our ports.
148 dispatch_source_cancel(m_sendSource.get());
149 m_sendSource = nullptr;
150 m_sendPort = MACH_PORT_NULL;
151
152 cancelReceiveSource();
153}

168void Connection::platformInitialize(Identifier identifier)
169{
170 if (!MACH_PORT_VALID(identifier.port))
171 return;
172
173 if (m_isServer) {
174 m_receivePort = identifier.port;
175 m_sendPort = MACH_PORT_NULL;
176
177#if !PLATFORM(WATCHOS)
178 mach_port_guard(mach_task_self(), m_receivePort, reinterpret_cast<mach_port_context_t>(this), true);
179#endif
180 } else {
181 m_receivePort = MACH_PORT_NULL;
182 m_sendPort = identifier.port;
183 }
184
185 m_sendSource = nullptr;
186 m_receiveSource = nullptr;
187
188 m_xpcConnection = identifier.xpcConnection;
189}

191bool Connection::open()
192{
193 if (m_isServer) {
194 ASSERT(m_receivePort);
195 ASSERT(!m_sendPort);
196 ASSERT(MACH_PORT_VALID(m_receivePort));
197 } else {
198 ASSERT(!m_receivePort);
199 ASSERT(m_sendPort);
200 ASSERT(MACH_PORT_VALID(m_sendPort));
201
202 auto kr = mach_port_allocate(mach_task_self(), MACH_PORT_RIGHT_RECEIVE, &m_receivePort);
203 if (kr != KERN_SUCCESS) {
204 LOG_ERROR("Could not allocate mach port, error %x: %s", kr, mach_error_string(kr));
205 CRASH();
206 }
207#if !PLATFORM(WATCHOS)
208 mach_port_guard(mach_task_self(), m_receivePort, reinterpret_cast<mach_port_context_t>(this), true);
209#endif
210
211#if PLATFORM(MAC)
212 mach_port_set_attributes(mach_task_self(), m_receivePort, MACH_PORT_DENAP_RECEIVER, (mach_port_info_t)0, 0);
213#endif
214
215 m_isConnected = true;
216
217 // Send the initialize message, which contains a send right for the server to use.
218 auto encoder = makeUniqueRef<Encoder>(MessageName::InitializeConnection, 0);
219 encoder.get() << MachPort(m_receivePort, MACH_MSG_TYPE_MAKE_SEND);
220
221 initializeSendSource();
222
223 sendMessage(WTFMove(encoder), { });
224 }
225
226 // Change the message queue length for the receive port.
227 setMachPortQueueLength(m_receivePort, MACH_PORT_QLIMIT_LARGE);
228
229 m_receiveSource = adoptOSObject(dispatch_source_create(DISPATCH_SOURCE_TYPE_MACH_RECV, m_receivePort, 0, m_connectionQueue->dispatchQueue()));
230 dispatch_source_set_event_handler(m_receiveSource.get(), [this, protectedThis = makeRefPtr(this)] {
231 receiveSourceEventHandler();
232 });
233 dispatch_source_set_cancel_handler(m_receiveSource.get(), [protectedThis = makeRefPtr(this), receivePort = m_receivePort] {
234#if !PLATFORM(WATCHOS)
235 mach_port_unguard(mach_task_self(), receivePort, reinterpret_cast<mach_port_context_t>(protectedThis.get()));
236#endif
237 mach_port_mod_refs(mach_task_self(), receivePort, MACH_PORT_RIGHT_RECEIVE, -1);
238 });
239
240 ref();
241 dispatch_async(m_connectionQueue->dispatchQueue(), ^{
242 dispatch_resume(m_receiveSource.get());
243
244 if (m_sendSource)
245 dispatch_resume(m_sendSource.get());
246
247 deref();
248 });
249
250 return true;
251}

253bool Connection::sendMessage(std::unique_ptr<MachMessage> message)
254{
255 ASSERT(message);
256 ASSERT(!m_pendingOutgoingMachMessage);
257 ASSERT(!m_isInitializingSendSource);
258
259 // Send the message.
260 kern_return_t kr = mach_msg(message->header(), MACH_SEND_MSG | MACH_SEND_TIMEOUT | MACH_SEND_NOTIFY, message->size(), 0, MACH_PORT_NULL, MACH_MSG_TIMEOUT_NONE, MACH_PORT_NULL);
261 switch (kr) {
262 case MACH_MSG_SUCCESS:
263 // The kernel has already adopted the descriptors.
264 message->leakDescriptors();
265 return true;
266
267 case MACH_SEND_TIMED_OUT:
268 // We timed out, stash away the message for later.
269 m_pendingOutgoingMachMessage = WTFMove(message);
270 return false;
271
272 case MACH_SEND_INVALID_DEST:
273 // The other end has disappeared, we'll get a dead name notification which will cause us to be invalidated.
274 return false;
275
276 default:
277 auto messageName = message->messageName();
278 auto errorMessage = makeString("Unhandled error code 0x", hex(kr), ", message '", description(messageName), "' (", messageName, ')');
279 WebKit::logAndSetCrashLogMessage(errorMessage.utf8().data());
280 CRASH_WITH_INFO(kr, WTF::enumToUnderlyingType(messageName));
281 }
282}

289bool Connection::sendOutgoingMessage(UniqueRef<Encoder>&& encoder)
290{
291 ASSERT(!m_pendingOutgoingMachMessage && !m_isInitializingSendSource);
292
293 auto attachments = encoder->releaseAttachments();
294
295 auto numberOfPortDescriptors = std::count_if(attachments.begin(), attachments.end(), [](auto& attachment)
296 {
297 return attachment.type() == Attachment::MachPortType;
298 });
299
300 bool messageBodyIsOOL = false;
301 auto messageSize = MachMessage::messageSize(encoder->bufferSize(), numberOfPortDescriptors, messageBodyIsOOL);
302 if (UNLIKELY(messageSize.hasOverflowed()))
303 return false;
304
305 if (messageSize > inlineMessageMaxSize) {
306 messageBodyIsOOL = true;
307 messageSize = MachMessage::messageSize(0, numberOfPortDescriptors, messageBodyIsOOL);
308 if (UNLIKELY(messageSize.hasOverflowed()))
309 return false;
310 }
311
312 size_t safeMessageSize = messageSize;
313 auto message = MachMessage::create(encoder->messageName(), safeMessageSize);
314 if (!message)
315 return false;
316
317 auto* header = message->header();
318 header->msgh_bits = MACH_MSGH_BITS(MACH_MSG_TYPE_COPY_SEND, 0);
319 header->msgh_size = safeMessageSize;
320 header->msgh_remote_port = m_sendPort;
321 header->msgh_local_port = MACH_PORT_NULL;
322 header->msgh_id = messageBodyIsOOL ? outOfLineBodyMessageID : inlineBodyMessageID;
323
324 auto* messageData = reinterpret_cast<uint8_t*>(header + 1);
325
326 bool isComplex = numberOfPortDescriptors || messageBodyIsOOL;
327 if (isComplex) {
328 header->msgh_bits |= MACH_MSGH_BITS_COMPLEX;
329
330 auto* body = reinterpret_cast<mach_msg_body_t*>(messageData);
331 body->msgh_descriptor_count = numberOfPortDescriptors + messageBodyIsOOL;
332 messageData = reinterpret_cast<uint8_t*>(body + 1);
333
334 auto getDescriptorAndAdvance = [](uint8_t*& data, std::size_t toAdvance) {
335 return reinterpret_cast<mach_msg_descriptor_t*>(std::exchange(data, data + toAdvance));
336 };
337
338 for (auto& attachment : attachments) {
339 ASSERT(attachment.type() == Attachment::MachPortType);
340 if (attachment.type() == Attachment::MachPortType) {
341 auto* descriptor = getDescriptorAndAdvance(messageData, sizeof(mach_msg_port_descriptor_t));
342 descriptor->port.name = attachment.port();
343 descriptor->port.disposition = attachment.disposition();
344 descriptor->port.type = MACH_MSG_PORT_DESCRIPTOR;
345 }
346 }
347
348 if (messageBodyIsOOL) {
349 auto* descriptor = getDescriptorAndAdvance(messageData, sizeof(mach_msg_ool_descriptor_t));
350 descriptor->out_of_line.address = encoder->buffer();
351 descriptor->out_of_line.size = encoder->bufferSize();
352 descriptor->out_of_line.copy = MACH_MSG_VIRTUAL_COPY;
353 descriptor->out_of_line.deallocate = false;
354 descriptor->out_of_line.type = MACH_MSG_OOL_DESCRIPTOR;
355 }
356 }
357
358 // Copy the data if it is not being sent out-of-line.
359 if (!messageBodyIsOOL)
360 memcpy(messageData, encoder->buffer(), encoder->bufferSize());
361
362 ASSERT(m_sendPort);
363 ASSERT(MACH_PORT_VALID(m_sendPort));
364
365 return sendMessage(WTFMove(message));
366}

368void Connection::initializeSendSource()
369{
370 m_sendSource = adoptOSObject(dispatch_source_create(DISPATCH_SOURCE_TYPE_MACH_SEND, m_sendPort, DISPATCH_MACH_SEND_DEAD | DISPATCH_MACH_SEND_POSSIBLE, m_connectionQueue->dispatchQueue()));
371 m_isInitializingSendSource = true;
372
373 dispatch_source_set_registration_handler(m_sendSource.get(), [this, protectedThis = makeRefPtr(this)] {
374 if (!m_sendSource)
375 return;
376 m_isInitializingSendSource = false;
377 resumeSendSource();
378 });
379 dispatch_source_set_event_handler(m_sendSource.get(), [this, protectedThis = makeRefPtr(this)] {
380 if (!m_sendSource)
381 return;
382
383 unsigned long data = dispatch_source_get_data(m_sendSource.get());
384
385 if (data & DISPATCH_MACH_SEND_DEAD) {
386 connectionDidClose();
387 return;
388 }
389
390 if (data & DISPATCH_MACH_SEND_POSSIBLE) {
391 // FIXME: Figure out why we get spurious DISPATCH_MACH_SEND_POSSIBLE events.
392 resumeSendSource();
393 return;
394 }
395 });
396
397 if (MACH_PORT_VALID(m_sendPort)) {
398 mach_port_t sendPort = m_sendPort;
399 dispatch_source_set_cancel_handler(m_sendSource.get(), ^{
400 // Release our send right.
401 deallocateSendRightSafely(sendPort);
402 });
403 }
404}

406void Connection::resumeSendSource()
407{
408 ASSERT(!m_isInitializingSendSource);
409 if (m_pendingOutgoingMachMessage)
410 sendMessage(WTFMove(m_pendingOutgoingMachMessage));
411 sendOutgoingMessages();
412}

414static std::unique_ptr<Decoder> createMessageDecoder(mach_msg_header_t* header)
415{
416 if (!(header->msgh_bits & MACH_MSGH_BITS_COMPLEX)) {
417 // We have a simple message.
418 uint8_t* body = reinterpret_cast<uint8_t*>(header + 1);
419 auto bodySize = CheckedSize { header->msgh_size } - sizeof(mach_msg_header_t);
420 if (UNLIKELY(bodySize.hasOverflowed())) {
421 RELEASE_LOG_FAULT(IPC, "createMessageDecoder: Overflow when computing bodySize (header->msgh_size: %lu, sizeof(mach_msg_header_t): %lu)", static_cast<unsigned long>(header->msgh_size), sizeof(mach_msg_header_t));
422 ASSERT_NOT_REACHED();
423 return nullptr;
424 }
425
426 return Decoder::create(body, bodySize, nullptr, Vector<Attachment> { });
427 }
428
429 mach_msg_body_t* body = reinterpret_cast<mach_msg_body_t*>(header + 1);
430 mach_msg_size_t numberOfPortDescriptors = body->msgh_descriptor_count;
431 ASSERT(numberOfPortDescriptors);
432 if (!numberOfPortDescriptors)
433 return nullptr;
434
435 uint8_t* descriptorData = reinterpret_cast<uint8_t*>(body + 1);
436
437 // If the message body was sent out-of-line, don't treat the last descriptor
438 // as an attachment, since it is really the message body.
439 bool messageBodyIsOOL = header->msgh_id == outOfLineBodyMessageID;
440 mach_msg_size_t numberOfAttachments = messageBodyIsOOL ? numberOfPortDescriptors - 1 : numberOfPortDescriptors;
441
442 // Build attachment list
443 Vector<Attachment> attachments(numberOfAttachments);
444
445 for (mach_msg_size_t i = 0; i < numberOfAttachments; ++i) {
446 mach_msg_descriptor_t* descriptor = reinterpret_cast<mach_msg_descriptor_t*>(descriptorData);
447 ASSERT(descriptor->type.type == MACH_MSG_PORT_DESCRIPTOR);
448 if (descriptor->type.type != MACH_MSG_PORT_DESCRIPTOR)
449 return nullptr;
450
451 attachments[numberOfAttachments - i - 1] = Attachment(descriptor->port.name, descriptor->port.disposition);
452 descriptorData += sizeof(mach_msg_port_descriptor_t);
453 }
454
455 if (messageBodyIsOOL) {
456 mach_msg_descriptor_t* descriptor = reinterpret_cast<mach_msg_descriptor_t*>(descriptorData);
457 ASSERT(descriptor->type.type == MACH_MSG_OOL_DESCRIPTOR);
458 if (descriptor->type.type != MACH_MSG_OOL_DESCRIPTOR)
459 return nullptr;
460
461 uint8_t* messageBody = static_cast<uint8_t*>(descriptor->out_of_line.address);
462 size_t messageBodySize = descriptor->out_of_line.size;
463
464 return Decoder::create(messageBody, messageBodySize, [](const uint8_t* buffer, size_t length) {
465 // FIXME: <rdar://problem/62086358> bufferDeallocator block ignores mach_msg_ool_descriptor_t->deallocate
466 vm_deallocate(mach_task_self(), reinterpret_cast<vm_address_t>(buffer), length);
467 }, WTFMove(attachments));
468 }
469
470 uint8_t* messageBody = descriptorData;
471 ASSERT(descriptorData >= reinterpret_cast<uint8_t*>(header));
472 auto messageBodySize = CheckedSize { header->msgh_size } - static_cast<size_t>(descriptorData - reinterpret_cast<uint8_t*>(header));
473 if (UNLIKELY(messageBodySize.hasOverflowed())) {
474 RELEASE_LOG_FAULT(IPC, "createMessageDecoder: Overflow when computing bodySize (header->msgh_size: %lu, (descriptorData - reinterpret_cast<uint8_t*>(header)): %lu)", static_cast<unsigned long>(header->msgh_size), static_cast<unsigned long>(descriptorData - reinterpret_cast<uint8_t*>(header)));
475 ASSERT_NOT_REACHED();
476 return nullptr;
477 }
478
479 return Decoder::create(messageBody, messageBodySize, nullptr, WTFMove(attachments));
480}

482// The receive buffer size should always include the maximum trailer size.
483static const size_t receiveBufferSize = inlineMessageMaxSize + MAX_TRAILER_SIZE;
484typedef Vector<char, receiveBufferSize> ReceiveBuffer;
485
486static mach_msg_header_t* readFromMachPort(mach_port_t machPort, ReceiveBuffer& buffer)
487{
488 ASSERT(MACH_PORT_VALID(machPort));
489
490 buffer.resize(receiveBufferSize);
491
492 mach_msg_header_t* header = reinterpret_cast<mach_msg_header_t*>(buffer.data());
493 kern_return_t kr = mach_msg(header, MACH_RCV_MSG | MACH_RCV_LARGE | MACH_RCV_TIMEOUT | MACH_RCV_VOUCHER, 0, buffer.size(), machPort, 0, MACH_PORT_NULL);
494 if (kr == MACH_RCV_TIMED_OUT)
495 return nullptr;
496
497 if (kr == MACH_RCV_TOO_LARGE) {
498 // The message was too large, resize the buffer and try again.
499 buffer.resize(header->msgh_size + MAX_TRAILER_SIZE);
500 header = reinterpret_cast<mach_msg_header_t*>(buffer.data());
501
502 kr = mach_msg(header, MACH_RCV_MSG | MACH_RCV_LARGE | MACH_RCV_TIMEOUT | MACH_RCV_VOUCHER, 0, buffer.size(), machPort, 0, MACH_PORT_NULL);
503 ASSERT(kr != MACH_RCV_TOO_LARGE);
504 }
505
506 if (kr != MACH_MSG_SUCCESS) {
507#if ASSERT_ENABLED
508 auto errorMessage = makeString("Unhandled error code 0x", hex(kr), " from mach_msg, receive port is 0x", hex(machPort));
509 WebKit::logAndSetCrashLogMessage(errorMessage.utf8().data());
510#endif
511 ASSERT_NOT_REACHED();
512 return nullptr;
513 }
514
515 return header;
516}

518void Connection::receiveSourceEventHandler()
519{
520 ReceiveBuffer buffer;
521
522 ASSERT(MACH_PORT_VALID(m_receivePort));
523 mach_msg_header_t* header = readFromMachPort(m_receivePort, buffer);
524 if (!header)
525 return;
526
527 switch (header->msgh_id) {
528 case MACH_NOTIFY_NO_SENDERS:
529 ASSERT(m_isServer);
530 if (!m_sendPort)
531 connectionDidClose();
532 return;
533
534 case inlineBodyMessageID:
535 case outOfLineBodyMessageID:
536 break;
537
538 case MACH_NOTIFY_SEND_ONCE:
539 default:
540 return;
541 }
542
543 std::unique_ptr<Decoder> decoder = createMessageDecoder(header);
544 if (!decoder)
545 return;
546
547#if PLATFORM(MAC)
548 decoder->setImportanceAssertion(makeUnique<ImportanceAssertion>(header));
549#endif
550
551 if (decoder->messageName() == MessageName::InitializeConnection) {
552 ASSERT(m_isServer);
553 ASSERT(!m_isConnected);
554 ASSERT(!m_sendPort);
555
556 MachPort port;
557 if (!decoder->decode(port)) {
558 // FIXME: Disconnect.
559 return;
560 }
561
562 m_sendPort = port.port();
563
564 if (m_sendPort) {
565 ASSERT(MACH_PORT_VALID(m_receivePort));
566 mach_port_t previousNotificationPort = MACH_PORT_NULL;
567 auto kr = mach_port_request_notification(mach_task_self(), m_receivePort, MACH_NOTIFY_NO_SENDERS, 0, MACH_PORT_NULL, MACH_MSG_TYPE_MOVE_SEND_ONCE, &previousNotificationPort);
568 ASSERT(kr == KERN_SUCCESS);
569 if (kr != KERN_SUCCESS) {
570 // If mach_port_request_notification fails, 'previousNotificationPort' will be uninitialized.
571 LOG_ERROR("mach_port_request_notification failed: (%x) %s", kr, mach_error_string(kr));
572 previousNotificationPort = MACH_PORT_NULL;
573 }
574
575 if (previousNotificationPort != MACH_PORT_NULL)
576 deallocateSendRightSafely(previousNotificationPort);
577
578 initializeSendSource();
579 dispatch_resume(m_sendSource.get());
580 }
581
582 m_isConnected = true;
583
584 // Send any pending outgoing messages.
585 sendOutgoingMessages();
586
587 return;
588 }
589
590 processIncomingMessage(WTFMove(decoder));
591}

636void Connection::didReceiveSyncReply(OptionSet<SendSyncOption> sendSyncOptions)
