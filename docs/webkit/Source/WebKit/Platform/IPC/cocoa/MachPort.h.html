26#pragma once
27
28#include "ArgumentCoder.h"
29#include "Attachment.h"
30
31namespace IPC {
32
33class MachPort {
34public:
35 MachPort()
36 : m_port(MACH_PORT_NULL)
37 , m_disposition(0)
38 {
39 }
40
41 MachPort(mach_port_name_t port, mach_msg_type_name_t disposition)
42 : m_port(port)
43 , m_disposition(disposition)
44 {
45 }
46
47 void encode(Encoder& encoder) const
48 {
49 encoder << Attachment(m_port, m_disposition);
50 }
51
52 static WARN_UNUSED_RETURN bool decode(Decoder& decoder, MachPort& p)
53 {
54 Attachment attachment;
55 if (!decoder.decode(attachment))
56 return false;
57
58 p.m_port = attachment.port();
59 p.m_disposition = attachment.disposition();
60 return true;
61 }
62
63 mach_port_name_t port() const { return m_port; }
64 mach_msg_type_name_t disposition() const { return m_disposition; }
65
66private:
67 mach_port_name_t m_port;
68 mach_msg_type_name_t m_disposition;
69};
70
71} // namespace IPC
