31bool Module::load()
32{
33 RetainPtr<CFURLRef> bundleURL = adoptCF(CFURLCreateWithFileSystemPath(kCFAllocatorDefault, m_path.createCFString().get(), kCFURLPOSIXPathStyle, FALSE));
34 if (!bundleURL)
35 return false;
36
37 RetainPtr<CFBundleRef> bundle = adoptCF(CFBundleCreate(kCFAllocatorDefault, bundleURL.get()));
38 if (!bundle)
39 return false;
40
41 if (!CFBundleLoadExecutable(bundle.get()))
42 return false;
43
44 m_bundle = WTFMove(bundle);
45 return true;
46}
47
48void Module::unload()
49{
50 if (!m_bundle)
51 return;
52
53 // See the comment in Module.h for why we leak the bundle here.
54 CFBundleRef unused = m_bundle.leakRef();
55 (void)unused;
56}
57
58void* Module::platformFunctionPointer(const char* functionName) const
59{
60 if (!m_bundle)
61 return 0;
62 RetainPtr<CFStringRef> functionNameString = adoptCF(CFStringCreateWithCStringNoCopy(kCFAllocatorDefault, functionName, kCFStringEncodingASCII, kCFAllocatorNull));
63 return CFBundleGetFunctionPointerForName(m_bundle.get(), functionNameString.get());
64}
65
66String Module::bundleIdentifier() const
67{
68 return CFBundleGetIdentifier(m_bundle.get());
69}
