26#pragma once
27
28#include <wtf/Forward.h>
29#include <wtf/Noncopyable.h>
30#include <wtf/RetainPtr.h>
31
32OBJC_CLASS CALayer;
33OBJC_CLASS CAContext;
34
35namespace WTF {
36class MachSendRight;
37}
38
39namespace WebKit {
40
41using LayerHostingContextID = uint32_t;
42enum class LayerHostingMode : uint8_t;
43
44struct LayerHostingContextOptions {
45#if PLATFORM(IOS_FAMILY)
46 bool canShowWhileLocked { false };
47#endif
48};
49
50class LayerHostingContext {
51 WTF_MAKE_NONCOPYABLE(LayerHostingContext); WTF_MAKE_FAST_ALLOCATED;
52public:
53 static std::unique_ptr<LayerHostingContext> createForPort(const WTF::MachSendRight& serverPort);
54
55#if HAVE(OUT_OF_PROCESS_LAYER_HOSTING)
56 static std::unique_ptr<LayerHostingContext> createForExternalHostingProcess(const LayerHostingContextOptions& = { });
57
58#if PLATFORM(MAC)
59 static std::unique_ptr<LayerHostingContext> createForExternalPluginHostingProcess();
60#endif
61
62 static RetainPtr<CALayer> createPlatformLayerForHostingContext(LayerHostingContextID);
63
64#endif // HAVE(OUT_OF_PROCESS_LAYER_HOSTING)
65
66 LayerHostingContext();
67 ~LayerHostingContext();
68
69 void setRootLayer(CALayer *);
70 CALayer *rootLayer() const;
71
72 LayerHostingContextID contextID() const;
73 void invalidate();
74
75 LayerHostingMode layerHostingMode() { return m_layerHostingMode; }
76
77 void setColorSpace(CGColorSpaceRef);
78 CGColorSpaceRef colorSpace() const;
79
80#if PLATFORM(MAC)
81 void setColorMatchUntaggedContent(bool);
82 bool colorMatchUntaggedContent() const;
83#endif
84
85 // Fences only work on iOS and OS 10.10+.
86 void setFencePort(mach_port_t);
87
88 // createFencePort does not install the fence port on the LayerHostingContext's
89 // CAContext; call setFencePort() with the newly created port if synchronization
90 // with this context is desired.
91 WTF::MachSendRight createFencePort();
92
93private:
94 LayerHostingMode m_layerHostingMode;
95 RetainPtr<CAContext> m_context;
96};
97
98} // namespace WebKit
