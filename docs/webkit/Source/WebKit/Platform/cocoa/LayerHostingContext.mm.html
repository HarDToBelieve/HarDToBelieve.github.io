37std::unique_ptr<LayerHostingContext> LayerHostingContext::createForPort(const MachSendRight& serverPort)
38{
39 auto layerHostingContext = makeUnique<LayerHostingContext>();
40
41 NSDictionary *options = @{
42 kCAContextPortNumber : @(serverPort.sendRight()),
43#if PLATFORM(MAC)
44 kCAContextCIFilterBehavior : @"ignore",
45#endif
46 };
47
48 layerHostingContext->m_layerHostingMode = LayerHostingMode::InProcess;
49 layerHostingContext->m_context = [CAContext remoteContextWithOptions:options];
50
51 return layerHostingContext;
52}
53
54#if HAVE(OUT_OF_PROCESS_LAYER_HOSTING)
55std::unique_ptr<LayerHostingContext> LayerHostingContext::createForExternalHostingProcess(const LayerHostingContextOptions& options)
56{
57 auto layerHostingContext = makeUnique<LayerHostingContext>();
58 layerHostingContext->m_layerHostingMode = LayerHostingMode::OutOfProcess;
59
60#if PLATFORM(IOS_FAMILY) && !PLATFORM(MACCATALYST)
61 // Use a very large display ID to ensure that the context is never put on-screen
62 // without being explicitly parented. See <rdar://problem/16089267> for details.
63 layerHostingContext->m_context = [CAContext remoteContextWithOptions:@{
64 kCAContextSecure: @(options.canShowWhileLocked),
65#if HAVE(CORE_ANIMATION_RENDER_SERVER)
66 kCAContextIgnoresHitTest : @YES,
67 kCAContextDisplayId : @10000
68#endif
69 }];
70#elif !PLATFORM(MACCATALYST)
71 [CAContext setAllowsCGSConnections:NO];
72 layerHostingContext->m_context = [CAContext remoteContextWithOptions:@{
73 kCAContextCIFilterBehavior : @"ignore",
74 }];
75#else
76 layerHostingContext->m_context = [CAContext contextWithCGSConnection:CGSMainConnectionID() options:@{
77 kCAContextCIFilterBehavior : @"ignore",
78 }];
79#endif
80
81 return layerHostingContext;
82}

84#if PLATFORM(MAC)
85std::unique_ptr<LayerHostingContext> LayerHostingContext::createForExternalPluginHostingProcess()
86{
87 auto layerHostingContext = makeUnique<LayerHostingContext>();
88 layerHostingContext->m_layerHostingMode = LayerHostingMode::OutOfProcess;
89 layerHostingContext->m_context = [CAContext contextWithCGSConnection:CGSMainConnectionID() options:@{ kCAContextCIFilterBehavior : @"ignore" }];
90 return layerHostingContext;
91}
