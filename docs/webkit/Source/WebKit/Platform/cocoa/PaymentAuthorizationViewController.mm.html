107namespace WebKit {

109static RetainPtr<PKPaymentAuthorizationViewController> platformViewController(PKPaymentRequest *request, PKPaymentAuthorizationViewController *viewController)
110{
111#if PLATFORM(IOS_FAMILY)
112 ASSERT(!viewController);
113 return adoptNS([PAL::allocPKPaymentAuthorizationViewControllerInstance() initWithPaymentRequest:request]);
114#else
115 return viewController;
116#endif
117}
118
119PaymentAuthorizationViewController::PaymentAuthorizationViewController(PaymentAuthorizationPresenter::Client& client, PKPaymentRequest *request, PKPaymentAuthorizationViewController *viewController)
120 : PaymentAuthorizationPresenter(client)
121 , m_viewController(platformViewController(request, viewController))
122 , m_delegate(adoptNS([[WKPaymentAuthorizationViewControllerDelegate alloc] initWithRequest:request presenter:*this]))
123{
124 [m_viewController setDelegate:m_delegate.get()];
125 [m_viewController setPrivateDelegate:m_delegate.get()];
126}
127
128WKPaymentAuthorizationDelegate *PaymentAuthorizationViewController::platformDelegate()
129{
130 return m_delegate.get();
131}
132
133void PaymentAuthorizationViewController::dismiss()
134{
135#if PLATFORM(IOS_FAMILY)
136 [[m_viewController presentingViewController] dismissViewControllerAnimated:YES completion:nullptr];
137#endif
138 [m_viewController setDelegate:nil];
139 [m_viewController setPrivateDelegate:nil];
140 m_viewController = nil;
141 [m_delegate invalidate];
142 m_delegate = nil;
143}
144
145#if PLATFORM(IOS_FAMILY)
146
147void PaymentAuthorizationViewController::present(UIViewController *presentingViewController, CompletionHandler<void(bool)>&& completionHandler)
148{
149 if (!presentingViewController || !m_viewController)
150 return completionHandler(false);
151
152 [presentingViewController presentViewController:m_viewController.get() animated:YES completion:nullptr];
153 completionHandler(true);
154}

156#endif
