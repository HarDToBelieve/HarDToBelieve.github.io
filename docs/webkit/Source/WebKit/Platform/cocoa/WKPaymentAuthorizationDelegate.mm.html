51- (NSArray<PKPaymentSummaryItem *> *)summaryItems
52{
53 return _summaryItems.get();
54}
55
56- (NSArray<PKShippingMethod *> *)shippingMethods
57{
58 return _shippingMethods.get();
59}

61- (void)completeMerchantValidation:(PKPaymentMerchantSession *)session error:(NSError *)error
62{
63 std::exchange(_didRequestMerchantSessionCompletion, nil)(session, error);
64}
65
66- (void)completePaymentMethodSelection:(PKPaymentRequestPaymentMethodUpdate *)paymentMethodUpdate
67{
68 auto update = paymentMethodUpdate ? retainPtr(paymentMethodUpdate) : adoptNS([PAL::allocPKPaymentRequestPaymentMethodUpdateInstance() initWithPaymentSummaryItems:_summaryItems.get()]);
69 _summaryItems = adoptNS([[update paymentSummaryItems] copy]);
70 std::exchange(_didSelectPaymentMethodCompletion, nil)(update.get());
71}
72
73- (void)completePaymentSession:(PKPaymentAuthorizationStatus)status errors:(NSArray<NSError *> *)errors
74{
75 auto result = adoptNS([PAL::allocPKPaymentAuthorizationResultInstance() initWithStatus:status errors:errors]);
76 std::exchange(_didAuthorizePaymentCompletion, nil)(result.get());
77}
78- (void)completeShippingContactSelection:(PKPaymentRequestShippingContactUpdate *)shippingContactUpdate
79{
80 auto update = shippingContactUpdate ? retainPtr(shippingContactUpdate) : adoptNS([PAL::allocPKPaymentRequestShippingContactUpdateInstance() initWithErrors:@[] paymentSummaryItems:_summaryItems.get() shippingMethods:_shippingMethods.get()]);
81 _summaryItems = adoptNS([[update paymentSummaryItems] copy]);
82 _shippingMethods = adoptNS([[update shippingMethods] copy]);
83 std::exchange(_didSelectShippingContactCompletion, nil)(update.get());
84}
85
86- (void)completeShippingMethodSelection:(PKPaymentRequestShippingMethodUpdate *)shippingMethodUpdate
87{
88 auto update = shippingMethodUpdate ? retainPtr(shippingMethodUpdate) : adoptNS([PAL::allocPKPaymentRequestShippingMethodUpdateInstance() initWithPaymentSummaryItems:_summaryItems.get()]);
89 _summaryItems = adoptNS([[update paymentSummaryItems] copy]);
90 std::exchange(_didSelectShippingMethodCompletion, nil)(update.get());
91}

95- (void)completeCouponCodeChange:(PKPaymentRequestCouponCodeUpdate *)couponCodeUpdate
96{
97 PKPaymentRequestCouponCodeUpdate *update = couponCodeUpdate ?: adoptNS([PAL::allocPKPaymentRequestCouponCodeUpdateInstance() initWithErrors:@[] paymentSummaryItems:_summaryItems.get() shippingMethods:_shippingMethods.get()]).autorelease();

99 _shippingMethods = adoptNS([update.shippingMethods copy]);
100 std::exchange(_didChangeCouponCodeCompletion, nil)(update);
101}

115- (instancetype)_initWithRequest:(PKPaymentRequest *)request presenter:(WebKit::PaymentAuthorizationPresenter&)presenter
116{
117 if (!(self = [super init]))
118 return nil;
119
120 _presenter = makeWeakPtr(presenter);
121 _request = request;
122 _shippingMethods = request.shippingMethods;
123 _summaryItems = request.paymentSummaryItems;
124 return self;
125}
126
127- (void)_didAuthorizePayment:(PKPayment *)payment completion:(WebKit::DidAuthorizePaymentCompletion::BlockType)completion
128{
129 ASSERT(!_didAuthorizePaymentCompletion);
130 _didAuthorizePaymentCompletion = completion;
131
132 auto presenter = _presenter.get();
133 if (!presenter)
134 return [self completePaymentSession:PKPaymentAuthorizationStatusFailure errors:@[ ]];
135
136 presenter->client().presenterDidAuthorizePayment(*presenter, WebCore::Payment(payment));
137}
138
139- (void)_didFinish
140{
141 if (auto presenter = _presenter.get())
142 presenter->client().presenterDidFinish(*presenter, { std::exchange(_sessionError, nil) });
143}
144
145- (void)_didRequestMerchantSession:(WebKit::DidRequestMerchantSessionCompletion::BlockType)completion
146{
147 ASSERT(!_didRequestMerchantSessionCompletion);
148 _didRequestMerchantSessionCompletion = completion;
149
150 [self _getPaymentServicesMerchantURL:^(NSURL *merchantURL, NSError *error) {
151 if (error)
152 LOG_ERROR("PKCanMakePaymentsWithMerchantIdentifierAndDomain error %@", error);
153
154 RunLoop::main().dispatch([self, protectedSelf = retainPtr(self), merchantURL = retainPtr(merchantURL)] {
155 ASSERT(_didRequestMerchantSessionCompletion);
156
157 auto presenter = _presenter.get();
158 if (!presenter) {
159 _didRequestMerchantSessionCompletion(nil, nil);
160 return;
161 }
162
163 presenter->client().presenterWillValidateMerchant(*presenter, merchantURL.get());
164 });
165 }];
166}

169{

194static WebCore::ApplePayDateComponents toDateComponents(NSDateComponents *dateComponents)
195{
196 ASSERT(dateComponents);
197
198 WebCore::ApplePayDateComponents result;
199 result.years = dateComponents.year;
200 result.months = dateComponents.month;
201 result.days = dateComponents.day;
202 result.hours = dateComponents.hour;
203 return result;
204}
205
206static WebCore::ApplePayDateComponentsRange toDateComponentsRange(PKDateComponentsRange *dateComponentsRange)
207{
208 ASSERT(dateComponentsRange);
209
210 WebCore::ApplePayDateComponentsRange result;
211 result.startDateComponents = toDateComponents(dateComponentsRange.startDateComponents);
212 result.endDateComponents = toDateComponents(dateComponentsRange.endDateComponents);
213 return result;
214}
215
216#endif // HAVE(PASSKIT_SHIPPING_METHOD_DATE_COMPONENTS_RANGE)
217
218static WebCore::ApplePayShippingMethod toShippingMethod(PKShippingMethod *shippingMethod)
219{
220 ASSERT(shippingMethod);
221
222 WebCore::ApplePayShippingMethod result;
223 result.amount = shippingMethod.amount.stringValue;
224 result.detail = shippingMethod.detail;
225 result.identifier = shippingMethod.identifier;
226 result.label = shippingMethod.label;
227#if HAVE(PASSKIT_SHIPPING_METHOD_DATE_COMPONENTS_RANGE)
228 if (shippingMethod.dateComponentsRange)
229 result.dateComponentsRange = toDateComponentsRange(shippingMethod.dateComponentsRange);
230#endif
231 return result;
232}
