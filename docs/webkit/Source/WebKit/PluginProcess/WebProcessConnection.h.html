42// A connection from a plug-in process to a web process.
43
44class WebProcessConnection : public RefCounted<WebProcessConnection>, IPC::Connection::Client {
45public:
46 static Ref<WebProcessConnection> create(IPC::Connection::Identifier);
47 virtual ~WebProcessConnection();
48
49 IPC::Connection* connection() const { return m_connection.get(); }
50 NPRemoteObjectMap* npRemoteObjectMap() const { return m_npRemoteObjectMap.get(); }
51
52 void removePluginControllerProxy(PluginControllerProxy*, Plugin*);
53
54 static void setGlobalException(const String&);
55
56private:
57 WebProcessConnection(IPC::Connection::Identifier);
58
59 void addPluginControllerProxy(std::unique_ptr<PluginControllerProxy>);
60
61 void destroyPluginControllerProxy(PluginControllerProxy*);
62
63 // IPC::Connection::Client
64 void didReceiveMessage(IPC::Connection&, IPC::Decoder&) override;
65 bool didReceiveSyncMessage(IPC::Connection&, IPC::Decoder&, UniqueRef<IPC::Encoder>&) override;
66 void didClose(IPC::Connection&) override;
67 void didReceiveInvalidMessage(IPC::Connection&, IPC::MessageName) override;
68
69 // Message handlers.
70 void didReceiveWebProcessConnectionMessage(IPC::Connection&, IPC::Decoder&);
71 bool didReceiveSyncWebProcessConnectionMessage(IPC::Connection&, IPC::Decoder&, UniqueRef<IPC::Encoder>&);
72 void createPlugin(const PluginCreationParameters&, Messages::WebProcessConnection::CreatePluginDelayedReply&&);
73 void createPluginAsynchronously(const PluginCreationParameters&);
74 void destroyPlugin(uint64_t pluginInstanceID, bool asynchronousCreationIncomplete, Messages::WebProcessConnection::DestroyPluginDelayedReply&&);
75
76 void createPluginInternal(const PluginCreationParameters&, bool& result, bool& wantsWheelEvents, uint32_t& remoteLayerClientID);
77
78 RefPtr<IPC::Connection> m_connection;
79
80 HashMap<uint64_t, std::unique_ptr<PluginControllerProxy>> m_pluginControllers;
81 RefPtr<NPRemoteObjectMap> m_npRemoteObjectMap;
82 HashSet<uint64_t> m_asynchronousInstanceIDsToIgnore;
83};
