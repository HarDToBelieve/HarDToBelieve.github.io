58void PluginControllerProxy::platformInitialize(const PluginCreationParameters& creationParameters)
59{
60 ASSERT(!m_layerHostingContext);
61 updateLayerHostingContext(creationParameters.parameters.layerHostingMode);
62}
63
64void PluginControllerProxy::platformDestroy()
65{
66 if (!m_layerHostingContext)
67 return;
68
69 m_layerHostingContext->invalidate();
70 m_layerHostingContext = nullptr;
71}

81void PluginControllerProxy::platformGeometryDidChange()
82{
83 CALayer *pluginLayer = m_plugin->pluginLayer();
84
85 // We don't want to animate to the new size so we disable actions for this transaction.
86 [CATransaction begin];
87 [CATransaction setValue:@YES forKey:kCATransactionDisableActions];
88 [pluginLayer setFrame:CGRectMake(0, 0, m_pluginSize.width(), m_pluginSize.height())];
89 [CATransaction commit];
90}

102void PluginControllerProxy::setLayerHostingMode(uint32_t opaqueLayerHostingMode)
103{
104 LayerHostingMode layerHostingMode = static_cast<LayerHostingMode>(opaqueLayerHostingMode);
105
106 m_plugin->setLayerHostingMode(layerHostingMode);
107 updateLayerHostingContext(layerHostingMode);
108
109 if (m_layerHostingContext)
110 m_connection->connection()->send(Messages::PluginProxy::SetLayerHostingContextID(m_layerHostingContext->contextID()), m_pluginInstanceID);
111}
112
113void PluginControllerProxy::updateLayerHostingContext(LayerHostingMode layerHostingMode)
114{
115 CALayer *platformLayer = m_plugin->pluginLayer();
116 if (!platformLayer)
117 return;
118
119 if (m_layerHostingContext) {
120 if (m_layerHostingContext->layerHostingMode() == layerHostingMode)
121 return;
122
123 m_layerHostingContext->invalidate();
124 }
125
126 switch (layerHostingMode) {
127 case LayerHostingMode::InProcess:
128 m_layerHostingContext = LayerHostingContext::createForPort(PluginProcess::singleton().compositingRenderServerPort());
129 break;
130#if HAVE(OUT_OF_PROCESS_LAYER_HOSTING)
131 case LayerHostingMode::OutOfProcess:
132 m_layerHostingContext = LayerHostingContext::createForExternalPluginHostingProcess();
133 break;
134#endif
135 }
136
137 m_layerHostingContext->setColorSpace(WebCore::sRGBColorSpaceRef());
138 m_layerHostingContext->setColorMatchUntaggedContent(true);
139
140 m_layerHostingContext->setRootLayer(platformLayer);
141}
