33namespace API {
34
35Ref<FrameHandle> FrameHandle::create(WebCore::FrameIdentifier frameID)
36{
37 return adoptRef(*new FrameHandle(frameID, false));
38}
39
40Ref<FrameHandle> FrameHandle::createAutoconverting(WebCore::FrameIdentifier frameID)
41{
42 return adoptRef(*new FrameHandle(frameID, true));
43}
44
45FrameHandle::FrameHandle(WebCore::FrameIdentifier frameID, bool isAutoconverting)
46 : m_frameID(frameID)
47 , m_isAutoconverting(isAutoconverting)
48{
49}
50
51void FrameHandle::encode(IPC::Encoder& encoder) const
52{
53 encoder << m_frameID;
54 encoder << m_isAutoconverting;
55}
56
57bool FrameHandle::decode(IPC::Decoder& decoder, RefPtr<Object>& result)
58{
59 WebCore::FrameIdentifier frameID;
60 if (!decoder.decode(frameID))
61 return false;
62
63 bool isAutoconverting;
64 if (!decoder.decode(isAutoconverting))
65 return false;
66
67 result = isAutoconverting ? createAutoconverting(frameID) : create(frameID);
68 return true;
69}
