26#pragma once
27
28#include "APIObject.h"
29#include <wtf/text/StringView.h>
30
31namespace API {
32
33class String final : public ObjectImpl<Object::Type::String> {
34public:
35 static Ref<String> createNull()
36 {
37 return adoptRef(*new String);
38 }
39
40 static Ref<String> create(WTF::String&& string)
41 {
42 return adoptRef(*new String(string.isNull() ? WTF::String(StringImpl::empty()) : string.isolatedCopy()));
43 }
44
45 static Ref<String> create(const WTF::String& string)
46 {
47 return create(string.isolatedCopy());
48 }
49
50 virtual ~String()
51 {
52 }
53
54 WTF::StringView stringView() const { return m_string; }
55 WTF::String string() const { return m_string.isolatedCopy(); }
56
57private:
58 String()
59 : m_string()
60 {
61 }
62
63 String(WTF::String&& string)
64 : m_string(WTFMove(string))
65 {
66 ASSERT(!m_string.isNull());
67 ASSERT(m_string.isSafeToSendToAnotherThread());
68 }
69
70 const WTF::String m_string;
71};
72
73} // namespace WebKit
