28
29#include "APIDictionary.h"
30#include <wtf/text/WTFString.h>
31
32namespace IPC {
33class Encoder;
34class Decoder;
35}
36
37namespace WebKit {
38
39class RemoteObjectInvocation {
40public:
41 struct ReplyInfo {
42 WTF_MAKE_FAST_ALLOCATED;
43 public:
44 ReplyInfo(uint64_t replyID, String&& blockSignature)
45 : replyID(replyID)
46 , blockSignature(WTFMove(blockSignature))
47 {
48 }
49
50 const uint64_t replyID;
51 const String blockSignature;
52 };
53 RemoteObjectInvocation();
54 RemoteObjectInvocation(const String& interfaceIdentifier, RefPtr<API::Dictionary>&& encodedInvocation, std::unique_ptr<ReplyInfo>&&);
55
56 const String& interfaceIdentifier() const { return m_interfaceIdentifier; }
57 const API::Dictionary* encodedInvocation() const { return m_encodedInvocation.get(); }
58 const ReplyInfo* replyInfo() const { return m_replyInfo.get(); }
59
60 void encode(IPC::Encoder&) const;
61 static WARN_UNUSED_RETURN bool decode(IPC::Decoder&, RemoteObjectInvocation&);
62
63private:
64 String m_interfaceIdentifier;
65 RefPtr<API::Dictionary> m_encodedInvocation;
66 std::unique_ptr<ReplyInfo> m_replyInfo;
67};

69}
