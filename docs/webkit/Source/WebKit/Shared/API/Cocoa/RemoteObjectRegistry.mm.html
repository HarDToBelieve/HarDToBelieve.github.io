37RemoteObjectRegistry::RemoteObjectRegistry(_WKRemoteObjectRegistry *remoteObjectRegistry)
38 : m_remoteObjectRegistry(remoteObjectRegistry)
39{
40}
41
42RemoteObjectRegistry::~RemoteObjectRegistry()
43{
44}
45
46void RemoteObjectRegistry::sendInvocation(const RemoteObjectInvocation& invocation)
47{
48
49 if (auto* replyInfo = invocation.replyInfo()) {
50 ASSERT(!m_pendingReplies.contains(replyInfo->replyID));
51 m_pendingReplies.add(replyInfo->replyID, backgroundActivity("RemoteObjectRegistry invocation"_s));
52 }
53
54 messageSender().send(Messages::RemoteObjectRegistry::InvokeMethod(invocation), messageDestinationID());
55}
56
57void RemoteObjectRegistry::sendReplyBlock(uint64_t replyID, const UserData& blockInvocation)
58{
59 messageSender().send(Messages::RemoteObjectRegistry::CallReplyBlock(replyID, blockInvocation), messageDestinationID());
60}
61
62void RemoteObjectRegistry::sendUnusedReply(uint64_t replyID)
63{
64 messageSender().send(Messages::RemoteObjectRegistry::ReleaseUnusedReplyBlock(replyID), messageDestinationID());
65}
66
67void RemoteObjectRegistry::invokeMethod(const RemoteObjectInvocation& invocation)
68{
69 [m_remoteObjectRegistry _invokeMethod:invocation];
70}
71
72void RemoteObjectRegistry::callReplyBlock(uint64_t replyID, const UserData& blockInvocation)
73{
74 bool wasRemoved = m_pendingReplies.remove(replyID);
75 ASSERT_UNUSED(wasRemoved, wasRemoved);
76
77 [m_remoteObjectRegistry _callReplyWithID:replyID blockInvocation:blockInvocation];
78}
79
80void RemoteObjectRegistry::releaseUnusedReplyBlock(uint64_t replyID)
81{
82 bool wasRemoved = m_pendingReplies.remove(replyID);
83 ASSERT_UNUSED(wasRemoved, wasRemoved);
84
85 [m_remoteObjectRegistry _releaseReplyWithID:replyID];
86}
