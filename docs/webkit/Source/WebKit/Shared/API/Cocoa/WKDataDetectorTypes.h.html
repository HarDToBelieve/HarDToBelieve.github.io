34/*! @enum WKDataDetectorTypes
35 @abstract The type of data detected.
36 @constant WKDataDetectorTypeNone No detection is performed.
37 @constant WKDataDetectorTypePhoneNumber Phone numbers are detected and turned into links.
38 @constant WKDataDetectorTypeLink URLs in text are detected and turned into links.
39 @constant WKDataDetectorTypeAddress Addresses are detected and turned into links.
40 @constant WKDataDetectorTypeCalendarEvent Dates and times that are in the future are detected and turned into links.
41 @constant WKDataDetectorTypeAll All of the above data types are turned into links when detected. Choosing this value will
42 automatically include any new detection type that is added.
43 */
44typedef NS_OPTIONS(NSUInteger, WKDataDetectorTypes) {
45 WKDataDetectorTypeNone = 0,
46 WKDataDetectorTypePhoneNumber = 1 << 0,
47 WKDataDetectorTypeLink = 1 << 1,
48 WKDataDetectorTypeAddress = 1 << 2,
49 WKDataDetectorTypeCalendarEvent = 1 << 3,
50 WKDataDetectorTypeTrackingNumber = 1 << 4,
51 WKDataDetectorTypeFlightNumber = 1 << 5,
52 WKDataDetectorTypeLookupSuggestion = 1 << 6,
53 WKDataDetectorTypeAll = NSUIntegerMax,
54
55 WKDataDetectorTypeSpotlightSuggestion WK_API_DEPRECATED_WITH_REPLACEMENT("WKDataDetectorTypeLookupSuggestion", ios(10.0, 10.0)) = WKDataDetectorTypeLookupSuggestion,
56} WK_API_AVAILABLE(ios(10.0));
