95static void ensureObjectStream(WKRemoteObjectEncoder *encoder)
96{
97 if (encoder->_objectStream)
98 return;
99
100 Ref<API::Array> objectStream = API::Array::create();
101 encoder->_objectStream = objectStream.ptr();
102
103 encoder->_rootDictionary->set(objectStreamKey, WTFMove(objectStream));
104}

106static void encodeToObjectStream(WKRemoteObjectEncoder *encoder, id value)
107{
108 ensureObjectStream(encoder);
109
110 size_t position = encoder->_objectStream->size();
111 encoder->_objectStream->elements().append(nullptr);
112
113 auto encodedObject = createEncodedObject(encoder, value);
114 ASSERT(!encoder->_objectStream->elements()[position]);
115 encoder->_objectStream->elements()[position] = WTFMove(encodedObject);
116}

118static void encodeInvocationArguments(WKRemoteObjectEncoder *encoder, NSInvocation *invocation, NSUInteger firstArgument)
119{
120 NSMethodSignature *methodSignature = invocation.methodSignature;
121 NSUInteger argumentCount = methodSignature.numberOfArguments;
122
123 ASSERT(firstArgument <= argumentCount);
124
125 for (NSUInteger i = firstArgument; i < argumentCount; ++i) {
126 const char* type = [methodSignature getArgumentTypeAtIndex:i];
127
128 switch (*type) {
129 // double
130 case 'd': {
131 double value;
132 [invocation getArgument:&value atIndex:i];
133
134 encodeToObjectStream(encoder, @(value));
135 break;
136 }
137
138 // float
139 case 'f': {
140 float value;
141 [invocation getArgument:&value atIndex:i];
142
143 encodeToObjectStream(encoder, @(value));
144 break;
145 }
146
147 // short
148 case 's': {
149 short value;
150 [invocation getArgument:&value atIndex:i];
151
152 encodeToObjectStream(encoder, @(value));
153 break;
154 }
155
156 // unsigned short
157 case 'S': {
158 unsigned short value;
159 [invocation getArgument:&value atIndex:i];
160
161 encodeToObjectStream(encoder, @(value));
162 break;
163 }
164
165 // int
166 case 'i': {
167 int value;
168 [invocation getArgument:&value atIndex:i];
169
170 encodeToObjectStream(encoder, @(value));
171 break;
172 }
173
174 // unsigned
175 case 'I': {
176 unsigned value;
177 [invocation getArgument:&value atIndex:i];
178
179 encodeToObjectStream(encoder, @(value));
180 break;
181 }
182
183 // char
184 case 'c': {
185 char value;
186 [invocation getArgument:&value atIndex:i];
187
188 encodeToObjectStream(encoder, @(value));
189 break;
190 }
191
192 // unsigned char
193 case 'C': {
194 unsigned char value;
195 [invocation getArgument:&value atIndex:i];
196
197 encodeToObjectStream(encoder, @(value));
198 break;
199 }
200
201 // bool
202 case 'B': {
203 BOOL value;
204 [invocation getArgument:&value atIndex:i];
205
206 encodeToObjectStream(encoder, @(value));
207 break;
208 }
209
210 // long
211 case 'l': {
212 long value;
213 [invocation getArgument:&value atIndex:i];
214
215 encodeToObjectStream(encoder, @(value));
216 break;
217 }
218
219 // unsigned long
220 case 'L': {
221 unsigned long value;
222 [invocation getArgument:&value atIndex:i];
223
224 encodeToObjectStream(encoder, @(value));
225 break;
226 }
227
228 // long long
229 case 'q': {
230 long long value;
231 [invocation getArgument:&value atIndex:i];
232
233 encodeToObjectStream(encoder, @(value));
234 break;
235 }
236
237 // unsigned long long
238 case 'Q': {
239 unsigned long long value;
240 [invocation getArgument:&value atIndex:i];
241
242 encodeToObjectStream(encoder, @(value));
243 break;
244 }
245
246 // Objective-C object
247 case '@': {
248 id value;
249 [invocation getArgument:&value atIndex:i];
250
251 @try {
252 encodeToObjectStream(encoder, value);
253 } @catch (NSException *e) {
254 RELEASE_LOG_ERROR(IPC, "WKRemoteObjectCode::encodeInvocationArguments: Exception caught when trying to encode an argument of type ObjC Object");
255 }
256
257 break;
258 }
259
260 // struct
261 case '{':
262 if (!strcmp(type, @encode(NSRange))) {
263 NSRange value;
264 [invocation getArgument:&value atIndex:i];
265
266 encodeToObjectStream(encoder, [NSValue valueWithRange:value]);
267 break;
268 } else if (!strcmp(type, @encode(CGSize))) {
269 CGSize value;
270 [invocation getArgument:&value atIndex:i];
271
272 encodeToObjectStream(encoder, @(value.width));
273 encodeToObjectStream(encoder, @(value.height));
274 break;
275 }
276 FALLTHROUGH;
277
278 default:
279 [NSException raise:NSInvalidArgumentException format:@"Unsupported invocation argument type '%s'", type];
280 }
281 }
282}

284static void encodeInvocation(WKRemoteObjectEncoder *encoder, NSInvocation *invocation)
285{
286 NSMethodSignature *methodSignature = invocation.methodSignature;
287 [encoder encodeObject:methodSignature._typeString forKey:typeStringKey];
288
289 if ([invocation isKindOfClass:[NSBlockInvocation class]]) {
290 [encoder encodeBool:YES forKey:isReplyBlockKey];
291 encodeInvocationArguments(encoder, invocation, 1);
292 } else {
293 [encoder encodeObject:NSStringFromSelector(invocation.selector) forKey:selectorKey];
294 encodeInvocationArguments(encoder, invocation, 2);
295 }
296}

303static RetainPtr<id> decodeObjCObject(WKRemoteObjectDecoder *decoder, Class objectClass)
304{
305 id allocation = [objectClass allocWithZone:decoder.zone];
306 if (!allocation)
307 [NSException raise:NSInvalidUnarchiveOperationException format:@"Class \"%@\" returned nil from +alloc while being decoded", NSStringFromClass(objectClass)];
308
309 RetainPtr<id> result = adoptNS([allocation initWithCoder:decoder]);
310 if (!result)
311 [NSException raise:NSInvalidUnarchiveOperationException format:@"Object of class \"%@\" returned nil from -initWithCoder: while being decoded", NSStringFromClass(objectClass)];
312
313 result = adoptNS([result.leakRef() awakeAfterUsingCoder:decoder]);
314 if (!result)
315 [NSException raise:NSInvalidUnarchiveOperationException format:@"Object of class \"%@\" returned nil from -awakeAfterUsingCoder: while being decoded", NSStringFromClass(objectClass)];
316
317 return result;
318}

320static constexpr NSString *peerCertificateKey = @"NSErrorPeerCertificateChainKey";
321static constexpr NSString *peerTrustKey = @"NSURLErrorFailingURLPeerTrustErrorKey";
322static constexpr NSString *clientCertificateKey = @"NSErrorClientCertificateChainKey";
323
324static RetainPtr<NSArray<NSData *>> transformCertificatesToData(NSArray *input)
325{
326 auto dataArray = adoptNS([[NSMutableArray alloc] initWithCapacity:input.count]);
327 for (id certificate in input) {
328 if (CFGetTypeID(certificate) != SecCertificateGetTypeID())
329 [NSException raise:NSInvalidArgumentException format:@"Error encoding invalid certificate in chain"];
330 [dataArray addObject:(NSData *)adoptCF(SecCertificateCopyData((SecCertificateRef)certificate)).get()];
331 }
332 return dataArray;
333}
334
335static RetainPtr<NSData> transformTrustToData(SecTrustRef trust)
336{
337 if (CFGetTypeID(trust) != SecTrustGetTypeID())
338 [NSException raise:NSInvalidArgumentException format:@"Error encoding invalid SecTrustRef"];
339 CFErrorRef error = nullptr;
340 auto data = adoptCF(SecTrustSerialize(trust, &error));
341 if (error)
342 [NSException raise:NSInvalidArgumentException format:@"Error serializing SecTrustRef: %@", error];
343 return data;
344}

346static void encodeError(WKRemoteObjectEncoder *encoder, NSError *error)
347{
348 RetainPtr<NSMutableDictionary> copy;
349 if (error.userInfo[_WKRecoveryAttempterErrorKey]) {
350 copy = adoptNS([error.userInfo mutableCopy]);
351 [copy removeObjectForKey:_WKRecoveryAttempterErrorKey];
352 }
353 if (error.userInfo[clientCertificateKey]) {
354 if (!copy)
355 copy = adoptNS([error.userInfo mutableCopy]);
356 [copy removeObjectForKey:clientCertificateKey];
357 }
358 if (NSArray *certificateChain = error.userInfo[peerCertificateKey]) {
359 if (!copy)
360 copy = adoptNS([error.userInfo mutableCopy]);
361 copy.get()[peerCertificateKey] = transformCertificatesToData(certificateChain).get();
362 }
363 if (id trust = error.userInfo[peerTrustKey]) {
364 if (!copy)
365 copy = adoptNS([error.userInfo mutableCopy]);
366 copy.get()[peerTrustKey] = transformTrustToData((SecTrustRef)trust).get();
367 }
368 if (!copy)
369 [error encodeWithCoder:encoder];
370 else
371 [[NSError errorWithDomain:error.domain code:error.code userInfo:copy.get()] encodeWithCoder:encoder];
372}

374static RetainPtr<NSArray> transformDataToCertificates(NSArray *input)
375{
376 auto array = adoptNS([[NSMutableArray alloc] initWithCapacity:input.count]);
377 for (NSData *data in input) {
378 if (CFGetTypeID(data) != CFDataGetTypeID())
379 [NSException raise:NSInvalidUnarchiveOperationException format:@"Error decoding certificate from object that is not data %@", NSStringFromClass([data class])];
380 auto certificate = adoptCF(SecCertificateCreateWithData(nullptr, (CFDataRef)data));
381 if (!certificate)
382 [NSException raise:NSInvalidUnarchiveOperationException format:@"Error decoding nvalid certificate in chain"];
383 [array addObject:(id)certificate.get()];
384 }
385 return array;
386}

388static RetainPtr<id> transformDataToTrust(NSData *data)
389{
390 if (CFGetTypeID(data) != CFDataGetTypeID())
391 [NSException raise:NSInvalidUnarchiveOperationException format:@"Invalid SecTrustRef data %@", NSStringFromClass([data class])];
392 CFErrorRef error = nullptr;
393 auto trust = adoptCF(SecTrustDeserialize((CFDataRef)data, &error));
394 if (error || !trust)
395 [NSException raise:NSInvalidUnarchiveOperationException format:@"Invalid SecTrustRef %@", error];
396 return trust;
397}

399static RetainPtr<NSError> decodeError(WKRemoteObjectDecoder *decoder)
400{
401 RetainPtr<NSError> error = decodeObjCObject(decoder, [NSError class]);
402 RetainPtr<NSMutableDictionary> copy;
403 if (NSArray *certificateChain = error.get().userInfo[peerCertificateKey]) {
404 copy = adoptNS([error.get().userInfo mutableCopy]);
405 copy.get()[peerCertificateKey] = transformDataToCertificates(certificateChain).get();
406 }
407 if (NSData *trust = error.get().userInfo[peerTrustKey]) {
408 if (!copy)
409 copy = adoptNS([error.get().userInfo mutableCopy]);
410 copy.get()[peerTrustKey] = transformDataToTrust(trust).get();
411 }
412 if (!copy)
413 return error;
414 return [NSError errorWithDomain:error.get().domain code:error.get().code userInfo:copy.get()];
415}

417static void encodeObject(WKRemoteObjectEncoder *encoder, id object)
418{
419 ASSERT(object);
420
421 if (![object conformsToProtocol:@protocol(NSSecureCoding)] && ![object isKindOfClass:[NSInvocation class]])
422 [NSException raise:NSInvalidArgumentException format:@"%@ does not conform to NSSecureCoding", object];
423
424 if (class_isMetaClass(object_getClass(object)))
425 [NSException raise:NSInvalidArgumentException format:@"Class objects may not be encoded"];
426
427 Class objectClass = [object classForCoder];
428 if (!objectClass)
429 [NSException raise:NSInvalidArgumentException format:@"-classForCoder returned nil for %@", object];
430
431 if (encoder->_objectsBeingEncoded.contains(object)) {
432 RELEASE_LOG_FAULT(IPC, "WKRemoteObjectCode::encodeObject: Object of type '%{private}s' contains a cycle", class_getName(object_getClass(object)));
433 [NSException raise:NSInvalidArgumentException format:@"Object of type '%s' contains a cycle", class_getName(object_getClass(object))];
434 return;
435 }
436
437 encoder->_objectsBeingEncoded.add(object);
438 auto exitScope = makeScopeExit([encoder, object] {
439 encoder->_objectsBeingEncoded.remove(object);
440 });
441
442 encoder->_currentDictionary->set(classNameKey, API::String::create(class_getName(objectClass)));
443
444 if ([object isKindOfClass:[NSInvocation class]]) {
445 // We have to special case NSInvocation since we don't want to encode the target.
446 encodeInvocation(encoder, object);
447 return;
448 }
449
450 if (objectClass == [NSString class] || objectClass == [NSMutableString class]) {
451 encodeString(encoder, object);
452 return;
453 }
454
455 if (objectClass == [NSError class])
456 return encodeError(encoder, object);
457
458 [object encodeWithCoder:encoder];
459}

461static RefPtr<API::Dictionary> createEncodedObject(WKRemoteObjectEncoder *encoder, id object)
462{
463 if (!object)
464 return nil;
465
466 Ref<API::Dictionary> dictionary = API::Dictionary::create();
467 SetForScope<API::Dictionary*> dictionaryChange(encoder->_currentDictionary, dictionary.ptr());
468
469 encodeObject(encoder, object);
470
471 return WTFMove(dictionary);
472}

738static id decodeObjectFromObjectStream(WKRemoteObjectDecoder *decoder, const HashSet<CFTypeRef>& allowedClasses)
739{
740 if (!decoder->_objectStream)
741 return nil;
742
743 if (decoder->_objectStreamPosition == decoder->_objectStream->size())
744 return nil;
745
746 const API::Dictionary* dictionary = decoder->_objectStream->at<API::Dictionary>(decoder->_objectStreamPosition++);
747
748 return decodeObject(decoder, dictionary, allowedClasses);
749}

751static void checkIfClassIsAllowed(WKRemoteObjectDecoder *decoder, Class objectClass)
752{
753 auto* allowedClasses = decoder->_allowedClasses;
754 if (!allowedClasses)
755 return;
756
757 if (allowedClasses->contains((__bridge CFTypeRef)objectClass))
758 return;
759
760 for (Class superclass = class_getSuperclass(objectClass); superclass; superclass = class_getSuperclass(superclass)) {
761 if (allowedClasses->contains((__bridge CFTypeRef)superclass))
762 return;
763 }
764
765 [NSException raise:NSInvalidUnarchiveOperationException format:@"Object of class \"%@\" is not allowed. Allowed classes are \"%@\"", objectClass, decoder.allowedClasses];
766}

768static void validateClass(WKRemoteObjectDecoder *decoder, Class objectClass)
769{
770 ASSERT(objectClass);
771
772 checkIfClassIsAllowed(decoder, objectClass);
773
774 // NSInvocation and NSBlockInvocation don't support NSSecureCoding, but we allow them anyway.
775 if (objectClass == [NSInvocation class] || objectClass == [NSBlockInvocation class])
776 return;
777
778 if (![decoder validateClassSupportsSecureCoding:objectClass])
779 [NSException raise:NSInvalidUnarchiveOperationException format:@"Object of class \"%@\" does not support NSSecureCoding.", objectClass];
780}

782static void decodeInvocationArguments(WKRemoteObjectDecoder *decoder, NSInvocation *invocation, const Vector<HashSet<CFTypeRef>>& allowedArgumentClasses, NSUInteger firstArgument)

916static NSInvocation *decodeInvocation(WKRemoteObjectDecoder *decoder)
917{
918 SEL selector = nullptr;
919 NSInvocation* invocation = nil;
920 BOOL isReplyBlock = [decoder decodeBoolForKey:isReplyBlockKey];
921 if (isReplyBlock) {
922 if (!decoder->_replyToSelector)
923 [NSException raise:NSInvalidUnarchiveOperationException format:@"%@: Received unknown reply block", decoder];
924
925 invocation = [decoder->_interface _invocationForReplyBlockOfSelector:decoder->_replyToSelector];
926 if (!invocation)
927 [NSException raise:NSInvalidUnarchiveOperationException format:@"Reply block for selector \"%s\" is not defined in the local interface", sel_getName(decoder->_replyToSelector)];
928 } else {
929 NSString *selectorString = [decoder decodeObjectOfClass:[NSString class] forKey:selectorKey];
930 if (!selectorString)
931 [NSException raise:NSInvalidUnarchiveOperationException format:@"Invocation had no selector"];
932
933 selector = NSSelectorFromString(selectorString);
934 ASSERT(selector);
935
936 invocation = [decoder->_interface _invocationForSelector:selector];
937 if (!invocation)
938 [NSException raise:NSInvalidUnarchiveOperationException format:@"Selector \"%@\" is not defined in the local interface", selectorString];
939 }
940
941 NSString *typeSignature = [decoder decodeObjectOfClass:[NSString class] forKey:typeStringKey];
942 if (!typeSignature)
943 [NSException raise:NSInvalidUnarchiveOperationException format:@"Invocation had no type signature"];
944
945 NSMethodSignature *remoteMethodSignature = [NSMethodSignature signatureWithObjCTypes:typeSignature.UTF8String];
946 if (![[invocation methodSignature] isEqual:remoteMethodSignature])
947 [NSException raise:NSInvalidUnarchiveOperationException format:@"Local and remote method signatures are not equal for method \"%s\"", selector ? sel_getName(selector) : "(no selector)"];
948
949 if (isReplyBlock) {
950 const auto& allowedClasses = [decoder->_interface _allowedArgumentClassesForReplyBlockOfSelector:decoder->_replyToSelector];
951 decodeInvocationArguments(decoder, invocation, allowedClasses, 1);
952 } else {
953 const auto& allowedClasses = [decoder->_interface _allowedArgumentClassesForSelector:selector];
954 decodeInvocationArguments(decoder, invocation, allowedClasses, 2);
955
956 [invocation setArgument:&selector atIndex:1];
957 }
958
959 return invocation;
960}

971static id decodeObject(WKRemoteObjectDecoder *decoder)
972{
973 API::String* classNameString = decoder->_currentDictionary->get<API::String>(classNameKey);
974 if (!classNameString)
975 [NSException raise:NSInvalidUnarchiveOperationException format:@"Class name missing"];
976
977 CString className = classNameString->string().utf8();
978
979 Class objectClass = objc_lookUpClass(className.data());
980 if (!objectClass)
981 [NSException raise:NSInvalidUnarchiveOperationException format:@"Class \"%s\" does not exist", className.data()];
982
983 validateClass(decoder, objectClass);
984
985 if (objectClass == [NSInvocation class] || objectClass == [NSBlockInvocation class])
986 return decodeInvocation(decoder);
987
988 if (objectClass == [NSString class])
989 return decodeString(decoder);
990
991 if (objectClass == [NSError class])
992 return decodeError(decoder).autorelease();
993
994 if (objectClass == [NSMutableString class])
995 return [NSMutableString stringWithString:decodeString(decoder)];
996
997 return decodeObjCObject(decoder, objectClass).autorelease();
998}
