59static bool isContainerClass(Class objectClass)
60{
61 // FIXME: Add more classes here if needed.
62 static Class arrayClass = [NSArray class];
63 static Class dictionaryClass = [NSDictionary class];
64 return objectClass == arrayClass || objectClass == dictionaryClass;
65}

67static HashSet<CFTypeRef>& propertyListClasses()
68{
69 static LazyNeverDestroyed<HashSet<CFTypeRef>> propertyListClasses;
70 static dispatch_once_t onceToken;
71 dispatch_once(&onceToken, ^{
72 propertyListClasses.construct(std::initializer_list<CFTypeRef> {
73 (__bridge CFTypeRef)[NSArray class], (__bridge CFTypeRef)[NSDictionary class],
74 (__bridge CFTypeRef)[NSNumber class], (__bridge CFTypeRef)[NSString class]
75 });
76 });
77
78 return propertyListClasses;
79}

81static const char* methodArgumentTypeEncodingForSelector(Protocol *protocol, SEL selector)
82{
83 // First look at required methods.
84 struct objc_method_description method = protocol_getMethodDescription(protocol, selector, YES, YES);
85 if (method.name)
86 return method.types;
87
88 // Then look at optional methods.
89 method = protocol_getMethodDescription(protocol, selector, NO, YES);
90 if (method.name)
91 return method.types;
92
93 return nullptr;
94}

96static void initializeMethod(MethodInfo& methodInfo, Protocol *protocol, SEL selector, NSMethodSignature *methodSignature, bool forReplyBlock)
97{
98 if (forReplyBlock)
99 methodInfo.replyInvocation = [NSInvocation invocationWithMethodSignature:methodSignature];
100 else {
101 const char* types = methodArgumentTypeEncodingForSelector(protocol, selector);
102 NSMethodSignature *signature = [NSMethodSignature signatureWithObjCTypes:types];
103 methodInfo.invocation = [NSInvocation invocationWithMethodSignature:signature];
104 }
105
106 auto& allowedClasses = forReplyBlock ? methodInfo.allowedReplyClasses : methodInfo.allowedArgumentClasses;
107 NSUInteger firstArgument = forReplyBlock ? 1 : 2;
108 NSUInteger argumentCount = methodSignature.numberOfArguments;
109
110 bool foundBlock = false;
111 for (NSUInteger i = firstArgument; i < argumentCount; ++i) {
112 const char* argumentType = [methodSignature getArgumentTypeAtIndex:i];
113
114 if (*argumentType != '@') {
115 // This is a non-object type; we won't allow any classes to be decoded for it.
116 allowedClasses.append({ });
117 continue;
118 }
119
120 if (*(argumentType + 1) == '?') {
121 if (forReplyBlock)
122 [NSException raise:NSInvalidArgumentException format:@"Blocks as arguments to the reply block of method (%s / %s) are not allowed", protocol_getName(protocol), sel_getName(selector)];
123
124 if (foundBlock)
125 [NSException raise:NSInvalidArgumentException format:@"Only one reply block is allowed per method (%s / %s)", protocol_getName(protocol), sel_getName(selector)];
126 foundBlock = true;
127 NSMethodSignature *blockSignature = [methodSignature _signatureForBlockAtArgumentIndex:i];
128 ASSERT(blockSignature._typeString);
129
130 initializeMethod(methodInfo, protocol, selector, blockSignature, true);
131 }
132
133 Class objectClass = [methodSignature _classForObjectAtArgumentIndex:i];
134 if (!objectClass) {
135 allowedClasses.append({ });
136 continue;
137 }
138
139 if (isContainerClass(objectClass)) {
140 // For container classes, we allow all simple property list classes.
141 allowedClasses.append(propertyListClasses());
142 continue;
143 }
144
145 allowedClasses.append({ (__bridge CFTypeRef)objectClass });
146 }
147}

149static void initializeMethods(_WKRemoteObjectInterface *interface, Protocol *protocol, bool requiredMethods)
150{
151 unsigned methodCount;
152 struct objc_method_description *methods = protocol_copyMethodDescriptionList(protocol, requiredMethods, true, &methodCount);
153
154 for (unsigned i = 0; i < methodCount; ++i) {
155 SEL selector = methods[i].name;
156
157 ASSERT(!interface->_methods.contains(selector));
158 MethodInfo& methodInfo = interface->_methods.add(selector, MethodInfo()).iterator->value;
159
160 const char* methodTypeEncoding = _protocol_getMethodTypeEncoding(protocol, selector, requiredMethods, true);
161 if (!methodTypeEncoding)
162 [NSException raise:NSInvalidArgumentException format:@"Could not find method type encoding for method \"%s\"", sel_getName(selector)];
163
164 NSMethodSignature *methodSignature = [NSMethodSignature signatureWithObjCTypes:methodTypeEncoding];
165
166 initializeMethod(methodInfo, protocol, selector, methodSignature, false);
167 }
168
169 free(methods);
170}

172static void initializeMethods(_WKRemoteObjectInterface *interface, Protocol *protocol)
173{
174 unsigned conformingProtocolCount;
175 auto conformingProtocols = protocol_copyProtocolList(protocol, &conformingProtocolCount);
176
177 for (unsigned i = 0; i < conformingProtocolCount; ++i) {
178 auto conformingProtocol = conformingProtocols[i];
179 if (conformingProtocol == @protocol(NSObject))
180 continue;
181 initializeMethods(interface, conformingProtocol);
182 }
183
184 free(conformingProtocols);
185
186 initializeMethods(interface, protocol, true);
187 initializeMethods(interface, protocol, false);
188}

190- (id)initWithProtocol:(Protocol *)protocol identifier:(NSString *)identifier
191{
192 if (!(self = [super init]))
193 return nil;
194
195 _protocol = protocol;
196 _identifier = adoptNS([identifier copy]);
197
198 initializeMethods(self, _protocol);
199
200 return self;
201}

213- (NSString *)debugDescription
214{
215 auto result = adoptNS([[NSMutableString alloc] initWithFormat:@"<%@: %p; protocol = \"%@\"; identifier = \"%@\"\n", NSStringFromClass(self.class), self, _identifier.get(), NSStringFromProtocol(_protocol)]);
216
217 auto descriptionForClasses = [](auto& allowedClasses) {
218 auto result = adoptNS([[NSMutableString alloc] initWithString:@"["]);
219 bool needsComma = false;
220
221 auto descriptionForArgument = [](auto& allowedArgumentClasses) {
222 auto result = adoptNS([[NSMutableString alloc] initWithString:@"{"]);
223
224 auto orderedArgumentClasses = copyToVector(allowedArgumentClasses);
225 std::sort(orderedArgumentClasses.begin(), orderedArgumentClasses.end(), [](CFTypeRef a, CFTypeRef b) {
226 return CString(class_getName((__bridge Class)a)) < CString(class_getName((__bridge Class)b));
227 });
228
229 bool needsComma = false;
230 for (auto& argumentClass : orderedArgumentClasses) {
231 if (needsComma)
232 [result appendString:@", "];
233
234 [result appendFormat:@"%s", class_getName((__bridge Class)argumentClass)];
235 needsComma = true;
236 }
237
238 [result appendString:@"}"];
239 return result.autorelease();
240 };
241
242 for (auto& argumentClasses : allowedClasses) {
243 if (needsComma)
244 [result appendString:@", "];
245
246 [result appendString:descriptionForArgument(argumentClasses)];
247 needsComma = true;
248 }
249
250 [result appendString:@"]"];
251 return result.autorelease();
252 };
253
254 for (auto& selectorAndMethod : _methods) {
255 [result appendFormat:@" selector = %s\n argument classes = %@\n", sel_getName(selectorAndMethod.key), descriptionForClasses(selectorAndMethod.value.allowedArgumentClasses)];
256
257 if (auto& replyInvocation = selectorAndMethod.value.replyInvocation)
258 [result appendFormat:@" reply block = (argument '%@') %@\n", [replyInvocation methodSignature]._typeString, descriptionForClasses(selectorAndMethod.value.allowedReplyClasses)];
259 }
260
261 [result appendString:@">\n"];
262 return result.autorelease();
263}

265static HashSet<CFTypeRef>& classesForSelectorArgument(_WKRemoteObjectInterface *interface, SEL selector, NSUInteger argumentIndex, bool replyBlock)
266{
267 auto it = interface->_methods.find(selector);
268 if (it == interface->_methods.end())
269 [NSException raise:NSInvalidArgumentException format:@"Interface does not contain selector \"%s\"", sel_getName(selector)];
270
271 MethodInfo& methodInfo = it->value;
272 if (replyBlock) {
273 if (!methodInfo.replyInvocation)
274 [NSException raise:NSInvalidArgumentException format:@"Selector \"%s\" does not have a reply block", sel_getName(selector)];
275
276 if (argumentIndex >= methodInfo.allowedReplyClasses.size())
277 [NSException raise:NSInvalidArgumentException format:@"Argument index %ld is out of range for reply block of selector \"%s\"", (unsigned long)argumentIndex, sel_getName(selector)];
278
279 return methodInfo.allowedReplyClasses[argumentIndex];
280 }
281
282 if (argumentIndex >= methodInfo.allowedArgumentClasses.size())
283 [NSException raise:NSInvalidArgumentException format:@"Argument index %ld is out of range for selector \"%s\"", (unsigned long)argumentIndex, sel_getName(selector)];
284
285 return methodInfo.allowedArgumentClasses[argumentIndex];
286}

288- (NSSet *)classesForSelector:(SEL)selector argumentIndex:(NSUInteger)argumentIndex ofReply:(BOOL)ofReply
289{
290 auto result = adoptNS([[NSMutableSet alloc] init]);
291
292 for (auto allowedClass : classesForSelectorArgument(self, selector, argumentIndex, ofReply))
293 [result addObject:(__bridge Class)allowedClass];
294
295 return result.autorelease();
296}

298- (void)setClasses:(NSSet *)classes forSelector:(SEL)selector argumentIndex:(NSUInteger)argumentIndex ofReply:(BOOL)ofReply
299{
300 HashSet<CFTypeRef> allowedClasses;
301 for (Class allowedClass in classes)
302 allowedClasses.add((__bridge CFTypeRef)allowedClass);
303
304 classesForSelectorArgument(self, selector, argumentIndex, ofReply) = WTFMove(allowedClasses);
305}
