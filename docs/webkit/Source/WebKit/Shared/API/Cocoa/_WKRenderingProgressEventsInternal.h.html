26#import <WebKit/_WKRenderingProgressEvents.h>
27
28#import <WebCore/LayoutMilestone.h>
29
30static inline _WKRenderingProgressEvents renderingProgressEvents(OptionSet<WebCore::LayoutMilestone> milestones)
31{
32 _WKRenderingProgressEvents events = 0;
33
34 if (milestones & WebCore::DidFirstLayout)
35 events |= _WKRenderingProgressEventFirstLayout;
36
37 if (milestones & WebCore::DidFirstVisuallyNonEmptyLayout)
38 events |= _WKRenderingProgressEventFirstVisuallyNonEmptyLayout;
39
40 if (milestones & WebCore::DidHitRelevantRepaintedObjectsAreaThreshold)
41 events |= _WKRenderingProgressEventFirstPaintWithSignificantArea;
42
43 if (milestones & WebCore::ReachedSessionRestorationRenderTreeSizeThreshold)
44 events |= _WKRenderingProgressEventReachedSessionRestorationRenderTreeSizeThreshold;
45
46 if (milestones & WebCore::DidFirstLayoutAfterSuppressedIncrementalRendering)
47 events |= _WKRenderingProgressEventFirstLayoutAfterSuppressedIncrementalRendering;
48
49 if (milestones & WebCore::DidFirstPaintAfterSuppressedIncrementalRendering)
50 events |= _WKRenderingProgressEventFirstPaintAfterSuppressedIncrementalRendering;
51
52 if (milestones & WebCore::DidRenderSignificantAmountOfText)
53 events |= _WKRenderingProgressEventDidRenderSignificantAmountOfText;
54
55 if (milestones & WebCore::DidFirstMeaningfulPaint)
56 events |= _WKRenderingProgressEventFirstMeaningfulPaint;
57
58 return events;
59}
