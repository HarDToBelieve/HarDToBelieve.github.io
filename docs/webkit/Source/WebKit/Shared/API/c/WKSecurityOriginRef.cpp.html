33WKTypeID WKSecurityOriginGetTypeID()
34{
35 return WebKit::toAPI(API::SecurityOrigin::APIType);
36}
37
38WKSecurityOriginRef WKSecurityOriginCreateFromString(WKStringRef string)
39{
40 return WebKit::toAPI(&API::SecurityOrigin::create(WebCore::SecurityOrigin::createFromString(WebKit::toImpl(string)->string())).leakRef());
41}
42
43WKSecurityOriginRef WKSecurityOriginCreateFromDatabaseIdentifier(WKStringRef identifier)
44{
45 auto origin = WebCore::SecurityOriginData::fromDatabaseIdentifier(WebKit::toImpl(identifier)->string());
46 if (!origin)
47 return nullptr;
48 return WebKit::toAPI(&API::SecurityOrigin::create(origin.value().securityOrigin()).leakRef());
49}
50
51WKSecurityOriginRef WKSecurityOriginCreate(WKStringRef protocol, WKStringRef host, int port)
52{
53 std::optional<uint16_t> validPort;
54 if (port && port <= std::numeric_limits<uint16_t>::max())
55 validPort = port;
56 auto securityOrigin = API::SecurityOrigin::create(WebKit::toImpl(protocol)->string(), WebKit::toImpl(host)->string(), validPort);
57 return WebKit::toAPI(&securityOrigin.leakRef());
58}

80unsigned short WKSecurityOriginGetPort(WKSecurityOriginRef securityOrigin)
