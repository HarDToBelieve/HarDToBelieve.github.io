56size_t WKStringGetCharacters(WKStringRef stringRef, WKChar* buffer, size_t bufferLength)
57{
58 static_assert(sizeof(WKChar) == sizeof(UChar), "Size of WKChar must match size of UChar");
59
60 unsigned unsignedBufferLength = std::min<size_t>(bufferLength, std::numeric_limits<unsigned>::max());
61 auto substring = WebKit::toImpl(stringRef)->stringView().substring(0, unsignedBufferLength);
62
63 substring.getCharactersWithUpconvert(reinterpret_cast<UChar*>(buffer));
64 return substring.length();
65}

72enum StrictType { NonStrict = false, Strict = true };
73
74template <StrictType strict>
75size_t WKStringGetUTF8CStringImpl(WKStringRef stringRef, char* buffer, size_t bufferSize)
76{
77 if (!bufferSize)
78 return 0;
79
80 auto stringView = WebKit::toImpl(stringRef)->stringView();
81
82 char* p = buffer;
83
84 if (stringView.is8Bit()) {
85 const LChar* characters = stringView.characters8();
86 if (!WTF::Unicode::convertLatin1ToUTF8(&characters, characters + stringView.length(), &p, p + bufferSize - 1))
87 return 0;
88 } else {
89 const UChar* characters = stringView.characters16();
90 auto result = WTF::Unicode::convertUTF16ToUTF8(&characters, characters + stringView.length(), &p, p + bufferSize - 1, strict);
91 if (result != WTF::Unicode::ConversionOK && result != WTF::Unicode::TargetExhausted)
92 return 0;
93 }
94
95 *p++ = '\0';
96 return p - buffer;
97}

99size_t WKStringGetUTF8CString(WKStringRef stringRef, char* buffer, size_t bufferSize)
100{
101 return WKStringGetUTF8CStringImpl<StrictType::Strict>(stringRef, buffer, bufferSize);
102}

104size_t WKStringGetUTF8CStringNonStrict(WKStringRef stringRef, char* buffer, size_t bufferSize)
105{
106 return WKStringGetUTF8CStringImpl<StrictType::NonStrict>(stringRef, buffer, bufferSize);
107}

109bool WKStringIsEqual(WKStringRef aRef, WKStringRef bRef)

114bool WKStringIsEqualToUTF8CString(WKStringRef aRef, const char* b)
115{
116 // FIXME: Should we add a fast path that avoids memory allocation when the string is all ASCII?
117 // FIXME: We can do even the general case more efficiently if we write a function in StringView that understands UTF-8 C strings.
118 return WebKit::toImpl(aRef)->stringView() == WTF::String::fromUTF8(b);
119}

121bool WKStringIsEqualToUTF8CStringIgnoringCase(WKStringRef aRef, const char* b)
122{
123 // FIXME: Should we add a fast path that avoids memory allocation when the string is all ASCII?
124 // FIXME: We can do even the general case more efficiently if we write a function in StringView that understands UTF-8 C strings.
125 return equalIgnoringASCIICase(WebKit::toImpl(aRef)->stringView(), WTF::String::fromUTF8(b));
126}
