35static inline Class wkNSURLClass()
36{
37 static dispatch_once_t once;
38 static Class wkNSURLClass;
39 dispatch_once(&once, ^{
40 wkNSURLClass = [WKNSURL class];
41 });
42 return wkNSURLClass;
43}

45WKURLRef WKURLCreateWithCFURL(CFURLRef cfURL)
46{
47 if (!cfURL)
48 return 0;
49
50 // Since WKNSURL is an internal class with no subclasses, we can do a simple equality check.
51 if (object_getClass((__bridge NSURL *)cfURL) == wkNSURLClass())
52 return WebKit::toAPI(static_cast<API::URL*>(&[(WKNSURL *)(__bridge NSURL *)CFRetain(cfURL) _apiObject]));
53
54 CString urlBytes;
55 WTF::getURLBytes(cfURL, urlBytes);
56
57 return WebKit::toCopiedURLAPI(urlBytes.data());
58}
59
60CFURLRef WKURLCopyCFURL(CFAllocatorRef allocatorRef, WKURLRef URLRef)
61{
62 auto& string = WebKit::toImpl(URLRef)->string();
63 if (string.isNull())
64 return nullptr;
65
66 // We first create a CString and then create the CFURL from it. This will ensure that the CFURL is stored in
67 // UTF-8 which uses less memory and is what WebKit clients might expect.
68
69 CString buffer = string.utf8();
70 return CFURLCreateAbsoluteURLWithBytes(nullptr, buffer.dataAsUInt8Ptr(), buffer.length(), kCFStringEncodingUTF8, nullptr, true);
71}
