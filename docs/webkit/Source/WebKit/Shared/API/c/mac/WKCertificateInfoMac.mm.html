32WKCertificateInfoRef WKCertificateInfoCreateWithServerTrust(SecTrustRef serverTrust)
33{
34#if HAVE(SEC_TRUST_SERIALIZATION)
35 RefPtr<WebKit::WebCertificateInfo> certificateInfo = WebKit::WebCertificateInfo::create(WebCore::CertificateInfo(serverTrust));
36 return WebKit::toAPI(certificateInfo.leakRef());
37#else
38 return nullptr;
39#endif
40}
41
42WKCertificateInfoRef WKCertificateInfoCreateWithCertficateChain(CFArrayRef certificateChain)
43{
44 RefPtr<WebKit::WebCertificateInfo> certificateInfo = WebKit::WebCertificateInfo::create(WebCore::CertificateInfo(certificateChain));
45 return WebKit::toAPI(certificateInfo.leakRef());
46}
47
48CFArrayRef WKCertificateInfoGetCertificateChain(WKCertificateInfoRef certificateInfoRef)
49{
50 return WebKit::toImpl(certificateInfoRef)->certificateInfo().certificateChain();
51}
52
53SecTrustRef WKCertificateInfoGetServerTrust(WKCertificateInfoRef certificateInfoRef)
54{
55#if HAVE(SEC_TRUST_SERIALIZATION)
56 return WebKit::toImpl(certificateInfoRef)->certificateInfo().trust();
57#else
58 return nullptr;
59#endif
60}
