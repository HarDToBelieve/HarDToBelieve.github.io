33OSStatus WKMediaFormatReaderCreate(CFAllocatorRef allocator, MTPluginFormatReaderRef* formatReaderRef)
34{
35#if ENABLE(WEBM_FORMAT_READER)
36 auto formatReader = WebKit::MediaFormatReader::create(allocator);
37 if (!formatReader)
38 return kMTPluginFormatReaderError_AllocationFailure;
39 *formatReaderRef = formatReader.leakRef()->wrapper();
40 return noErr;
41#else
42 UNUSED_PARAM(allocator);
43 *formatReaderRef = nullptr;
44 return -1;
45#endif
46}
47
48OSStatus WKMediaFormatReaderStartOnMainThread(MTPluginFormatReaderRef formatReaderRef, MTPluginByteSourceRef byteSource)
49{
50#if ENABLE(WEBM_FORMAT_READER)
51 auto formatReader = WebKit::MediaFormatReader::unwrap(formatReaderRef);
52 if (!formatReader)
53 return paramErr;
54 formatReader->startOnMainThread(byteSource);
55 return noErr;
56#else
57 UNUSED_PARAM(formatReaderRef);
58 UNUSED_PARAM(byteSource);
59 return -1;
60#endif
61}
