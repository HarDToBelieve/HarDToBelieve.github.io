32WKTypeID WKObjCTypeWrapperGetTypeID()
33{
34 return WebKit::toAPI(WebKit::ObjCObjectGraph::APIType);
35}
36
37WKObjCTypeWrapperRef WKObjCTypeWrapperCreate(id object)
38{
39 auto objectWrapper = WebKit::ObjCObjectGraph::create(object);
40 return WebKit::toAPI(&objectWrapper.leakRef());
41}
42
43id WKObjCTypeWrapperGetObject(WKObjCTypeWrapperRef wrapperRef)
44{
45 return WebKit::toImpl(wrapperRef)->rootObject();
46}
