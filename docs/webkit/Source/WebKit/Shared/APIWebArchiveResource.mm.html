63static void releaseWebArchiveResourceData(unsigned char*, const void* data)
64{
65 // Balanced by CFRetain in WebArchiveResource::data().
66 CFRelease(data);
67}
68
69Ref<API::Data> WebArchiveResource::data()
70{
71 RetainPtr<CFDataRef> cfData = m_archiveResource->data().createCFData();
72
73 // Balanced by CFRelease in releaseWebArchiveResourceData.
74 CFRetain(cfData.get());
75
76 return API::Data::createWithoutCopying(CFDataGetBytePtr(cfData.get()), CFDataGetLength(cfData.get()), releaseWebArchiveResourceData, cfData.get());
77}
