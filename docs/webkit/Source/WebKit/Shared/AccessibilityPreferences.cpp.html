26#include "config.h"
27#include "AccessibilityPreferences.h"

29namespace IPC {
30
31void ArgumentCoder<WebKit::AccessibilityPreferences>::encode(Encoder& encoder, const WebKit::AccessibilityPreferences& preferences)
32{
33#if HAVE(PER_APP_ACCESSIBILITY_PREFERENCES)
34 encoder << preferences.reduceMotionEnabled;
35 encoder << preferences.increaseButtonLegibility;
36 encoder << preferences.enhanceTextLegibility;
37 encoder << preferences.darkenSystemColors;
38 encoder << preferences.invertColorsEnabled;
39#endif
40#if HAVE(MEDIA_ACCESSIBILITY_FRAMEWORK)
41 encoder << preferences.captionDisplayMode;
42 encoder << preferences.preferredLanguages;
43#endif
44}
45
46std::optional<WebKit::AccessibilityPreferences> ArgumentCoder<WebKit::AccessibilityPreferences>::decode(Decoder& decoder)
47{
48 WebKit::AccessibilityPreferences preferences;
49#if HAVE(PER_APP_ACCESSIBILITY_PREFERENCES)
50 if (!decoder.decode(preferences.reduceMotionEnabled))
51 return std::nullopt;
52 if (!decoder.decode(preferences.increaseButtonLegibility))
53 return std::nullopt;
54 if (!decoder.decode(preferences.enhanceTextLegibility))
55 return std::nullopt;
56 if (!decoder.decode(preferences.darkenSystemColors))
57 return std::nullopt;
58 if (!decoder.decode(preferences.invertColorsEnabled))
59 return std::nullopt;
60#endif
61#if HAVE(MEDIA_ACCESSIBILITY_FRAMEWORK)
62 if (!decoder.decode(preferences.captionDisplayMode))
63 return std::nullopt;
64 if (!decoder.decode(preferences.preferredLanguages))
65 return std::nullopt;
66#endif
67 return preferences;
68}
