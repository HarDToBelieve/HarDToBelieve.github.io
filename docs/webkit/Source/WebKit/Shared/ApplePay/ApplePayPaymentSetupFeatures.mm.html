40static NSArray<PKPaymentSetupFeature *> *toPlatformFeatures(Vector<RefPtr<WebCore::ApplePaySetupFeature>>&& features)
41{
42 NSMutableArray *platformFeatures = [NSMutableArray arrayWithCapacity:features.size()];
43 for (auto& feature : features) {
44 if (feature)
45 [platformFeatures addObject:feature->platformFeature()];
46 }
47 return platformFeatures;
48}

50PaymentSetupFeatures::PaymentSetupFeatures(Vector<RefPtr<WebCore::ApplePaySetupFeature>>&& features)
51 : m_platformFeatures { toPlatformFeatures(WTFMove(features)) }
52{
53}

55PaymentSetupFeatures::PaymentSetupFeatures(RetainPtr<NSArray>&& platformFeatures)
56 : m_platformFeatures { WTFMove(platformFeatures) }

58}

60void PaymentSetupFeatures::encode(IPC::Encoder& encoder) const
61{
62 encoder << m_platformFeatures;
63}

65std::optional<PaymentSetupFeatures> PaymentSetupFeatures::decode(IPC::Decoder& decoder)
66{
67 static NeverDestroyed<RetainPtr<NSArray>> allowedClasses;
68 static std::once_flag onceFlag;
69 std::call_once(onceFlag, [] {
70 auto allowed = adoptNS([[NSMutableArray alloc] initWithCapacity:2]);
71 [allowed addObject:[NSArray class]];
72 if (auto pkPaymentSetupFeatureClass = PAL::getPKPaymentSetupFeatureClass())
73 [allowed addObject:pkPaymentSetupFeatureClass];
74 allowedClasses.get() = adoptNS([allowed copy]);
75 });
76
77 auto platformFeatures = IPC::decode<NSArray<PKPaymentSetupFeature *>>(decoder, allowedClasses.get().get());
78 if (!platformFeatures)
79 return std::nullopt;
80
81 return PaymentSetupFeatures { WTFMove(*platformFeatures) };
82}
83
84PaymentSetupFeatures::operator Vector<Ref<WebCore::ApplePaySetupFeature>>() const
85{
86 Vector<Ref<WebCore::ApplePaySetupFeature>> features;
87 features.reserveInitialCapacity([m_platformFeatures count]);
88 for (PKPaymentSetupFeature *platformFeature in m_platformFeatures.get())
89 features.uncheckedAppend(WebCore::ApplePaySetupFeature::create(platformFeature));
90 return features;
91}
