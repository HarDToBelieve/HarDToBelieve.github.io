45static RetainPtr<PKPaymentSetupConfiguration> toPlatformConfiguration(const WebCore::ApplePaySetupConfiguration& coreConfiguration, const URL& url)
46{
47#if PLATFORM(MAC)
48 if (!PAL::getPKPaymentSetupConfigurationClass())
49 return nil;
50#endif
51
52 auto configuration = adoptNS([PAL::allocPKPaymentSetupConfigurationInstance() init]);
53 [configuration setMerchantIdentifier:coreConfiguration.merchantIdentifier];
54 [configuration setOriginatingURL:url];
55 [configuration setReferrerIdentifier:coreConfiguration.referrerIdentifier];
56ALLOW_NEW_API_WITHOUT_GUARDS_BEGIN
57 [configuration setSignature:coreConfiguration.signature];
58ALLOW_NEW_API_WITHOUT_GUARDS_END
59
60 if ([configuration respondsToSelector:@selector(setSignedFields:)]) {
61 NSMutableArray *signedFields = [NSMutableArray arrayWithCapacity:coreConfiguration.signedFields.size()];
62 for (auto& signedField : coreConfiguration.signedFields)
63 [signedFields addObject:signedField];
64 [configuration setSignedFields:signedFields];
65 }
66
67 return configuration;
68}

80void PaymentSetupConfiguration::encode(IPC::Encoder& encoder) const

82 encoder << m_configuration;
83}

85std::optional<PaymentSetupConfiguration> PaymentSetupConfiguration::decode(IPC::Decoder& decoder)
86{
87 static NeverDestroyed<RetainPtr<NSArray>> allowedClasses;
88 static std::once_flag onceFlag;
89 std::call_once(onceFlag, [] {
90 auto allowed = adoptNS([[NSMutableArray alloc] initWithCapacity:1]);
91 if (auto pkPaymentSetupConfigurationClass = PAL::getPKPaymentSetupConfigurationClass())
92 [allowed addObject:pkPaymentSetupConfigurationClass];
93 allowedClasses.get() = adoptNS([allowed copy]);
94 });
95
96 auto configuration = IPC::decode<PKPaymentSetupConfiguration>(decoder, allowedClasses.get().get());
97 if (!configuration)
98 return std::nullopt;
99
100 return PaymentSetupConfiguration { WTFMove(*configuration) };
101}
