81void WebPaymentCoordinatorProxy::showPaymentUI(WebCore::PageIdentifier destinationID, const String& originatingURLString, const Vector<String>& linkIconURLStrings, const WebCore::ApplePaySessionPaymentRequest& paymentRequest, CompletionHandler<void(bool)>&& completionHandler)
82{
83 if (auto& coordinator = activePaymentCoordinatorProxy())
84 coordinator->didReachFinalState();
85 activePaymentCoordinatorProxy() = makeWeakPtr(this);
86
87 MESSAGE_CHECK(canBegin());
88 MESSAGE_CHECK(!m_destinationID);
89 MESSAGE_CHECK(!m_authorizationPresenter);
90
91 m_destinationID = destinationID;
92 m_state = State::Activating;
93
94 URL originatingURL(URL(), originatingURLString);
95
96 Vector<URL> linkIconURLs;
97 for (const auto& linkIconURLString : linkIconURLStrings)
98 linkIconURLs.append(URL(URL(), linkIconURLString));
99
100 platformShowPaymentUI(originatingURL, linkIconURLs, paymentRequest, [this, weakThis = makeWeakPtr(*this)](bool result) {
101 if (!weakThis)
102 return;
103
104 if (m_state == State::Idle) {
105 ASSERT(!activePaymentCoordinatorProxy());
106 ASSERT(!m_destinationID);
107 ASSERT(m_merchantValidationState == MerchantValidationState::Idle);
108 return;
109 }
110
111 ASSERT(m_state == State::Activating);
112 if (!result) {
113 didReachFinalState();
114 return;
115 }
116
117 m_state = State::Active;
118 });
119
120 completionHandler(true);
121}

123void WebPaymentCoordinatorProxy::completeMerchantValidation(const WebCore::PaymentMerchantSession& paymentMerchantSession)
124{
125 // It's possible that the payment has been canceled already.
126 if (m_state == State::Idle)
127 return;

129 MESSAGE_CHECK(m_merchantValidationState == MerchantValidationState::Validating);
130
131 platformCompleteMerchantValidation(paymentMerchantSession);
132 m_merchantValidationState = MerchantValidationState::ValidationComplete;

187void WebPaymentCoordinatorProxy::completePaymentSession(const std::optional<WebCore::PaymentAuthorizationResult>& result)
188{
189 // It's possible that the payment has been canceled already.
190 if (!canCompletePayment())
191 return;
192
193 bool isFinalStateResult = WebCore::isFinalStateResult(result);
194
195 platformCompletePaymentSession(result);
196
197 if (!isFinalStateResult) {
198 m_state = State::Active;
199 return;
200 }
201
202 m_state = State::Completing;
203}

211 didReachFinalState();
212}

277bool WebPaymentCoordinatorProxy::canBegin() const
278{
279 switch (m_state) {
280 case State::Idle:
281 return true;
282
283 case State::Activating:
284 case State::Active:
285 case State::Authorized:
286 case State::Completing:
287 case State::ShippingMethodSelected:
288 case State::ShippingContactSelected:
289 case State::PaymentMethodSelected:
290#if ENABLE(APPLE_PAY_COUPON_CODE)
291 case State::CouponCodeChanged:
292#endif
293 return false;
294 }
295}

298{

356
357void WebPaymentCoordinatorProxy::didReachFinalState(WebCore::PaymentSessionError&& error)
358{
359 ASSERT(!canBegin());
360
361 if (canCancel())
362 send(Messages::WebPaymentCoordinator::DidCancelPaymentSession(error));
363
364 platformHidePaymentUI();
365
366 m_authorizationPresenter = nullptr;
367 m_destinationID = std::nullopt;
368 m_merchantValidationState = MerchantValidationState::Idle;
369 m_state = State::Idle;
370
371 ASSERT(activePaymentCoordinatorProxy() == this);
372 activePaymentCoordinatorProxy() = nullptr;
373}
