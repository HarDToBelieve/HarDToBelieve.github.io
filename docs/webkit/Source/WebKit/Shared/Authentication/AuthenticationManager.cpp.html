45static uint64_t generateAuthenticationChallengeID()
46{
47 ASSERT(RunLoop::isMain());
48
49 static int64_t uniqueAuthenticationChallengeID;
50 return ++uniqueAuthenticationChallengeID;
51}
52
53static bool canCoalesceChallenge(const WebCore::AuthenticationChallenge& challenge)
54{
55 // Do not coalesce server trust evaluation requests because ProtectionSpace comparison does not evaluate server trust (e.g. certificate).
56 return challenge.protectionSpace().authenticationScheme() != ProtectionSpaceAuthenticationSchemeServerTrustEvaluationRequested;
57}

70uint64_t AuthenticationManager::addChallengeToChallengeMap(UniqueRef<Challenge>&& challenge)
71{
72 ASSERT(RunLoop::isMain());
73
74 uint64_t challengeID = generateAuthenticationChallengeID();
75 m_challenges.set(challengeID, WTFMove(challenge));
76 return challengeID;
77}

79bool AuthenticationManager::shouldCoalesceChallenge(WebPageProxyIdentifier pageID, uint64_t challengeID, const AuthenticationChallenge& challenge) const
80{
81 if (!canCoalesceChallenge(challenge))
82 return false;
83
84 for (auto& item : m_challenges) {
85 if (item.key != challengeID && item.value->pageID == pageID && ProtectionSpace::compare(challenge.protectionSpace(), item.value->challenge.protectionSpace()))
86 return true;
87 }
88 return false;
89}
90
91Vector<uint64_t> AuthenticationManager::coalesceChallengesMatching(uint64_t challengeID) const
92{
93 auto* challenge = m_challenges.get(challengeID);
94 if (!challenge) {
95 ASSERT_NOT_REACHED();
96 return { };
97 }
98
99 Vector<uint64_t> challengesToCoalesce;
100 challengesToCoalesce.append(challengeID);
101
102 if (!canCoalesceChallenge(challenge->challenge))
103 return challengesToCoalesce;
104
105 for (auto& item : m_challenges) {
106 if (item.key != challengeID && item.value->pageID == challenge->pageID && ProtectionSpace::compare(challenge->challenge.protectionSpace(), item.value->challenge.protectionSpace()))
107 challengesToCoalesce.append(item.key);
108 }
109
110 return challengesToCoalesce;
111}

113void AuthenticationManager::didReceiveAuthenticationChallenge(PAL::SessionID sessionID, WebPageProxyIdentifier pageID, const SecurityOriginData* topOrigin, const AuthenticationChallenge& authenticationChallenge, NegotiatedLegacyTLS negotiatedLegacyTLS, ChallengeCompletionHandler&& completionHandler)
114{
115 if (!pageID)
116 return completionHandler(AuthenticationChallengeDisposition::PerformDefaultHandling, { });
117
118 uint64_t challengeID = addChallengeToChallengeMap(makeUniqueRef<Challenge>(pageID, authenticationChallenge, WTFMove(completionHandler)));
119
120 // Coalesce challenges in the same protection space and in the same page.
121 if (shouldCoalesceChallenge(pageID, challengeID, authenticationChallenge))
122 return;
123
124 std::optional<SecurityOriginData> topOriginData;
125 if (topOrigin)
126 topOriginData = *topOrigin;
127 m_process.send(Messages::NetworkProcessProxy::DidReceiveAuthenticationChallenge(sessionID, pageID, topOriginData, authenticationChallenge, negotiatedLegacyTLS == NegotiatedLegacyTLS::Yes, challengeID));
128}

130void AuthenticationManager::didReceiveAuthenticationChallenge(IPC::MessageSender& download, const WebCore::AuthenticationChallenge& authenticationChallenge, ChallengeCompletionHandler&& completionHandler)
131{
132 WebPageProxyIdentifier dummyPageID;
133 uint64_t challengeID = addChallengeToChallengeMap(makeUniqueRef<Challenge>(dummyPageID, authenticationChallenge, WTFMove(completionHandler)));
134
135 // Coalesce challenges in the same protection space and in the same page.
136 if (shouldCoalesceChallenge(dummyPageID, challengeID, authenticationChallenge))
137 return;
138
139 download.send(Messages::DownloadProxy::DidReceiveAuthenticationChallenge(authenticationChallenge, challengeID));
140}

142void AuthenticationManager::completeAuthenticationChallenge(uint64_t challengeID, AuthenticationChallengeDisposition disposition, WebCore::Credential&& credential)
143{
144 ASSERT(RunLoop::isMain());
145
146 for (auto& coalescedChallengeID : coalesceChallengesMatching(challengeID)) {
147 auto challenge = m_challenges.take(coalescedChallengeID);
148 ASSERT(!challenge->challenge.isNull());
149 challenge->completionHandler(disposition, credential);
150 }
151}
