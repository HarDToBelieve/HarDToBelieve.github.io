39namespace WebKit {
40
41class SandboxInitializationParameters;
42struct AuxiliaryProcessInitializationParameters;
43
44class AuxiliaryProcess : public IPC::Connection::Client, public IPC::MessageSender {
45 WTF_MAKE_NONCOPYABLE(AuxiliaryProcess);
46
47public:
48 void initialize(const AuxiliaryProcessInitializationParameters&);
49
50 // disable and enable termination of the process. when disableTermination is called, the
51 // process won't terminate unless a corresponding disableTermination call is made.
52 void disableTermination();
53 void enableTermination();
54
55 void addMessageReceiver(IPC::ReceiverName, IPC::MessageReceiver&);
56 void addMessageReceiver(IPC::ReceiverName, uint64_t destinationID, IPC::MessageReceiver&);
57 void removeMessageReceiver(IPC::ReceiverName, uint64_t destinationID);
58 void removeMessageReceiver(IPC::ReceiverName);
59 void removeMessageReceiver(IPC::MessageReceiver&);
60
61 template <typename T>
62 void addMessageReceiver(IPC::ReceiverName messageReceiverName, ObjectIdentifier<T> destinationID, IPC::MessageReceiver& receiver)
63 {
64 addMessageReceiver(messageReceiverName, destinationID.toUInt64(), receiver);
65 }
66
67 template <typename T>
68 void removeMessageReceiver(IPC::ReceiverName messageReceiverName, ObjectIdentifier<T> destinationID)
69 {
70 removeMessageReceiver(messageReceiverName, destinationID.toUInt64());
71 }
72
73 void mainThreadPing(CompletionHandler<void()>&&);
74 void setProcessSuppressionEnabled(bool);
75
76#if PLATFORM(COCOA)
77 void setApplicationIsDaemon();
78 void launchServicesCheckIn();
79 void setQOS(int latencyQOS, int throughputQOS);
80#endif
81
82 IPC::Connection* parentProcessConnection() const { return m_connection.get(); }
83
84 IPC::MessageReceiverMap& messageReceiverMap() { return m_messageReceiverMap; }
85
86#if PLATFORM(MAC)
87 static bool isSystemWebKit();
88#endif
89
90#if PLATFORM(COCOA)
91 bool parentProcessHasEntitlement(const char* entitlement);
92#endif
93
94protected:
95 explicit AuxiliaryProcess();
96 virtual ~AuxiliaryProcess();
97
98 void setTerminationTimeout(Seconds seconds) { m_terminationTimeout = seconds; }
99
100 virtual void initializeProcess(const AuxiliaryProcessInitializationParameters&);
101 virtual void initializeProcessName(const AuxiliaryProcessInitializationParameters&);
102 virtual void initializeSandbox(const AuxiliaryProcessInitializationParameters&, SandboxInitializationParameters&);
103 virtual void initializeConnection(IPC::Connection*);
104
105 virtual bool shouldTerminate() = 0;
106 virtual void terminate();
107
108 virtual void stopRunLoop();
109
110#if USE(APPKIT)
111 static void stopNSAppRunLoop();
112#endif
113
114#if PLATFORM(MAC) && ENABLE(WEBPROCESS_NSRUNLOOP)
115 static void stopNSRunLoop();
116#endif
117
118 void didReceiveMessage(IPC::Connection&, IPC::Decoder&) override;
119
120#if OS(LINUX)
121 void didReceiveMemoryPressureEvent(bool isCritical);
122#endif
123
124 static std::optional<std::pair<IPC::Connection::Identifier, IPC::Attachment>> createIPCConnectionPair();
125
126private:
127 virtual bool shouldOverrideQuarantine() { return true; }
128
129 // IPC::MessageSender
130 IPC::Connection* messageSenderConnection() const override;
131 uint64_t messageSenderDestinationID() const override;
132
133 // IPC::Connection::Client.
134 void didReceiveInvalidMessage(IPC::Connection&, IPC::MessageName) final;
135 void didClose(IPC::Connection&) override;
136
137 void shutDown();
138
139 void terminationTimerFired();
140
141 void platformInitialize(const AuxiliaryProcessInitializationParameters&);
142 void platformStopRunLoop();
143
144 // The timeout, in seconds, before this process will be terminated if termination
145 // has been enabled. If the timeout is 0 seconds, the process will be terminated immediately.
146 Seconds m_terminationTimeout;
147
148 // A termination counter; when the counter reaches zero, the process will be terminated
149 // after a given period of time.
150 unsigned m_terminationCounter;
151
152 RunLoop::Timer<AuxiliaryProcess> m_terminationTimer;
153
154 RefPtr<IPC::Connection> m_connection;
155 IPC::MessageReceiverMap m_messageReceiverMap;
156
157 UserActivity m_processSuppressionDisabled;
158
159#if PLATFORM(COCOA)
160 OSObjectPtr<xpc_object_t> m_priorityBoostMessage;
161#endif
162};
163
164struct AuxiliaryProcessInitializationParameters {
165 String uiProcessName;
166 String clientIdentifier;
167 String clientBundleIdentifier;
168 uint32_t clientSDKVersion;
169 std::optional<WebCore::ProcessIdentifier> processIdentifier;
170 IPC::Connection::Identifier connectionIdentifier;
171 HashMap<String, String> extraInitializationData;
172 WebCore::AuxiliaryProcessType processType;
173#if PLATFORM(COCOA)
174 OSObjectPtr<xpc_object_t> priorityBoostMessage;
175#endif
176};
