33BlobDataFileReferenceWithSandboxExtension::BlobDataFileReferenceWithSandboxExtension(const String& path, const String& replacementPath, RefPtr<SandboxExtension>&& sandboxExtension)
34 : BlobDataFileReference(path, replacementPath)
35 , m_sandboxExtension(WTFMove(sandboxExtension))
36{
37}
38
39BlobDataFileReferenceWithSandboxExtension::~BlobDataFileReferenceWithSandboxExtension()
40{
41}
42
43void BlobDataFileReferenceWithSandboxExtension::prepareForFileAccess()
44{
45 if (m_sandboxExtension)
46 m_sandboxExtension->consume();
47}
48
49void BlobDataFileReferenceWithSandboxExtension::revokeFileAccess()
50{
51 if (m_sandboxExtension)
52 m_sandboxExtension->revoke();
53}
