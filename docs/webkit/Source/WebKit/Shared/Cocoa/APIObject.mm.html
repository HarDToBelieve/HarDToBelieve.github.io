106static const size_t minimumObjectAlignment = alignof(std::aligned_storage<std::numeric_limits<size_t>::max()>::type);
107static_assert(minimumObjectAlignment >= alignof(void*), "Objects should always be at least pointer-aligned.");
108static const size_t maximumExtraSpaceForAlignment = minimumObjectAlignment - alignof(void*);

112void Object::ref() const
113{

115}

117void Object::deref() const
118{
119 CFRelease((__bridge CFTypeRef)wrapper());

122static id <WKObject> allocateWKObject(Class cls, size_t size)
123{
124 return class_createInstance(cls, size + maximumExtraSpaceForAlignment);
125}
126
127API::Object& Object::fromWKObjectExtraSpace(id <WKObject> obj)
128{
129 size_t size = sizeof(API::Object);
130 size_t spaceAvailable = size + maximumExtraSpaceForAlignment;
131 void* indexedIvars = object_getIndexedIvars(obj);
132 return *static_cast<API::Object*>(std::align(minimumObjectAlignment, size, indexedIvars, spaceAvailable));
133}

135void* Object::newObject(size_t size, Type type)
136{
137 id <WKObject> wrapper;
138
139 // Wrappers that inherit from WKObject store the API::Object in their extra bytes, so they are
140 // allocated using NSAllocatedObject. The other wrapper classes contain inline storage for the
141 // API::Object, so they are allocated using +alloc.
142
143 switch (type) {
144#if ENABLE(APPLICATION_MANIFEST)
145 case Type::ApplicationManifest:
146 wrapper = [_WKApplicationManifest alloc];
147 break;
148#endif
149
150 case Type::Array:
151 wrapper = [WKNSArray alloc];
152 break;
153
154#if ENABLE(ATTACHMENT_ELEMENT)
155 case Type::Attachment:
156 wrapper = [_WKAttachment alloc];
157 break;
158#endif
159
160 case Type::AuthenticationChallenge:
161 ALLOW_DEPRECATED_DECLARATIONS_BEGIN
162 wrapper = allocateWKObject([WKNSURLAuthenticationChallenge class], size);
163 ALLOW_DEPRECATED_DECLARATIONS_END
164 break;
165
166 case Type::AutomationSession:
167 wrapper = [_WKAutomationSession alloc];
168 break;
169
170 case Type::BackForwardList:
171 wrapper = [WKBackForwardList alloc];
172 break;
173
174 case Type::BackForwardListItem:
175 wrapper = [WKBackForwardListItem alloc];
176 break;
177
178 case Type::Boolean:
179 case Type::Double:
180 case Type::UInt64:
181 case Type::Int64:
182 wrapper = [WKNSNumber alloc];
183 ((WKNSNumber *)wrapper)->_type = type;
184 break;
185
186 case Type::Bundle:
187 wrapper = [WKWebProcessPlugInController alloc];
188 break;
189
190 case Type::BundlePage:
191 wrapper = [WKWebProcessPlugInBrowserContextController alloc];
192 break;
193
194 case Type::Connection:
195 // While not actually a WKObject instance, WKConnection uses allocateWKObject to allocate extra space
196 // instead of using ObjectStorage because the wrapped C++ object is a subclass of WebConnection.
197 ALLOW_DEPRECATED_DECLARATIONS_BEGIN
198 wrapper = allocateWKObject([WKConnection class], size);
199 ALLOW_DEPRECATED_DECLARATIONS_END
200 break;

202 case Type::DebuggableInfo:
203 wrapper = [_WKInspectorDebuggableInfo alloc];
204 break;

426 case Type::BundleNodeHandle:

432 wrapper = [WKWebProcessPlugInPageGroup alloc];
433 ALLOW_DEPRECATED_DECLARATIONS_END;
434 break;

436 case Type::BundleRangeHandle:

448
449 Object& object = wrapper._apiObject;
450 object.m_wrapper = wrapper;
451
452 return &object;
453}

455void* Object::wrap(API::Object* object)
456{
457 if (!object)
458 return nullptr;
459
460 return (__bridge void*)object->wrapper();
461}
462
463API::Object* Object::unwrap(void* object)
464{
465 if (!object)
466 return nullptr;
467
468 return &((__bridge id <WKObject>)object)._apiObject;
469}
