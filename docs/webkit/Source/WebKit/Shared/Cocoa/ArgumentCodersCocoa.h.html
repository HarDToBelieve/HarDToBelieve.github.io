32namespace IPC {
33
34void encodeObject(Encoder&, id);
35std::optional<RetainPtr<id>> decodeObject(Decoder&, NSArray<Class> *allowedClasses);
36
37template<typename T> using IsObjCObject = std::enable_if_t<std::is_convertible<T *, id>::value, T *>;
38
39template<typename T, typename = IsObjCObject<T>> void encode(Encoder&, T *);
40template<typename T, typename = IsObjCObject<T>> WARN_UNUSED_RETURN bool decode(Decoder&, RetainPtr<T>&, NSArray<Class> *allowedClasses = @[ [T class] ]);
41template<typename T, typename = IsObjCObject<T>> std::optional<RetainPtr<T>> decode(Decoder&, NSArray<Class> *allowedClasses = @[ [T class] ]);
42template<typename T, typename = IsObjCObject<T>> std::optional<RetainPtr<T>> decode(Decoder&, Class allowedClass);
43
44#if ASSERT_ENABLED
45
46static inline bool isObjectClassAllowed(id object, NSArray<Class> *allowedClasses)
47{
48 for (Class allowedClass in allowedClasses) {
49 if ([object isKindOfClass:allowedClass])
50 return true;
51 }
52 return false;
53}
54
55#endif // ASSERT_ENABLED
56
57template<typename T, typename>
58void encode(Encoder& encoder, T *object)
59{
60 encodeObject(encoder, object);
61}
62
63template<typename T, typename>
64bool decode(Decoder& decoder, RetainPtr<T>& result, NSArray<Class> *allowedClasses)
65{
66 auto object = decodeObject(decoder, allowedClasses);
67 if (!object)
68 return false;
69 result = *object;
70 ASSERT(!*object || isObjectClassAllowed((*object).get(), allowedClasses));
71 return true;
72}
73
74template<typename T, typename>
75std::optional<RetainPtr<T>> decode(Decoder& decoder, NSArray<Class> *allowedClasses)
76{
77 auto result = decodeObject(decoder, allowedClasses);
78 if (!result)
79 return std::nullopt;
80 ASSERT(!*result || isObjectClassAllowed((*result).get(), allowedClasses));
81 return { *result };
82}
83
84template<typename T, typename>
85std::optional<RetainPtr<T>> decode(Decoder& decoder, Class allowedClass)
86{
87 return decode<T>(decoder, allowedClass ? @[ allowedClass ] : @[ ]);
88}
89
90template<typename T> struct ArgumentCoder<T *> {
91 template<typename U = T, typename = IsObjCObject<U>>
92 static void encode(Encoder& encoder, U *object)
93 {
94 encodeObject(encoder, object);
95 }
96};
97
98template<typename T> struct ArgumentCoder<RetainPtr<T>> {
99 template<typename U = T, typename = IsObjCObject<U>>
100 static void encode(Encoder& encoder, const RetainPtr<U>& object)
101 {
102 ArgumentCoder<U *>::encode(encoder, object.get());
103 }
104
105 template<typename U = T, typename = IsObjCObject<U>>
106 static std::optional<RetainPtr<U>> decode(Decoder& decoder)
107 {
108 return IPC::decode<U>(decoder);
109 }
110};
