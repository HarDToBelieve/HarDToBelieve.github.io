66- (nullable id)archiver:(NSKeyedArchiver *)archiver willEncodeObject:(id)object
67{
68#if HAVE(NSFONT_WITH_OPTICAL_SIZING_BUG)
69 if (auto font = dynamic_objc_cast<NSFont>(object)) {
70 // Recreate any serialized fonts after normalizing the
71 // font attributes to work around <rdar://problem/51657880>.
72 return WebKit::fontWithAttributes(font.fontDescriptor.fontAttributes, 0);
73 }
74#endif
75
76 if (auto unwrappedURL = dynamic_objc_cast<NSURL>(object))
77 return adoptNS([[WKSecureCodingURLWrapper alloc] initWithURL:unwrappedURL]).autorelease();
78
79 return object;
80}

82- (id _Nullable)unarchiver:(NSKeyedUnarchiver *)unarchiver didDecodeObject:(id _Nullable) NS_RELEASES_ARGUMENT object NS_RETURNS_RETAINED
83{
84 auto adoptedObject = adoptNS(object);
85 if (auto wrapper = dynamic_objc_cast<WKSecureCodingURLWrapper>(adoptedObject.get()))
86 return retainPtr(wrapper.wrappedURL).leakRef();
87
88 return adoptedObject.leakRef();
89}

107static constexpr NSString *baseURLKey = @"WK.baseURL";
108
109- (void)encodeWithCoder:(NSCoder *)coder
110{
111 RELEASE_ASSERT(m_wrappedURL);
112 auto baseURL = m_wrappedURL.get().baseURL;
113 BOOL hasBaseURL = !!baseURL;
114
115 [coder encodeValueOfObjCType:"c" at:&hasBaseURL];
116 if (hasBaseURL)
117 [coder encodeObject:baseURL forKey:baseURLKey];
118
119 WTF::URLCharBuffer urlBytes;
120 WTF::getURLBytes((__bridge CFURLRef)m_wrappedURL.get(), urlBytes);
121 [coder encodeBytes:urlBytes.data() length:urlBytes.size()];
122}
123
124- (_Nullable instancetype)initWithCoder:(NSCoder *)coder
125{
126 auto selfPtr = adoptNS([super initWithString:@""]);
127 if (!selfPtr)
128 return nil;
129
130 BOOL hasBaseURL;
131 [coder decodeValueOfObjCType:"c" at:&hasBaseURL size:sizeof(hasBaseURL)];
132
133 RetainPtr<NSURL> baseURL;
134 if (hasBaseURL)
135 baseURL = (NSURL *)[coder decodeObjectOfClass:NSURL.class forKey:baseURLKey];
136
137 NSUInteger length;
138 if (auto bytes = (UInt8 *)[coder decodeBytesWithReturnedLength:&length]) {
139 m_wrappedURL = adoptNS((__bridge NSURL*)CFURLCreateAbsoluteURLWithBytes(nullptr, bytes, length, kCFStringEncodingUTF8, (__bridge CFURLRef)baseURL.get(), true));
140 if (!m_wrappedURL)
141 LOG_ERROR("Failed to decode NSURL due to invalid encoding of length %d. Substituting a blank URL", length);
142 }
143
144 if (!m_wrappedURL)
145 m_wrappedURL = [NSURL URLWithString:@""];
146
147 return selfPtr.leakRef();
148}
149
150- (_Nullable instancetype)initWithURL:(NSURL *)url
151{
152 if (self = [super initWithString:@""])
153 m_wrappedURL = url;
154
155 return self;
156}

181static NSType typeFromObject(id object)
182{
183 ASSERT(object);
184
185 // Specific classes handled.
186 if ([object isKindOfClass:[NSArray class]])
187 return NSType::Array;
188 if ([object isKindOfClass:[CocoaColor class]])
189 return NSType::Color;
190 if ([object isKindOfClass:[NSData class]])
191 return NSType::Data;
192 if ([object isKindOfClass:[NSDate class]])
193 return NSType::Date;
194 if ([object isKindOfClass:[NSDictionary class]])
195 return NSType::Dictionary;
196 if ([object isKindOfClass:[CocoaFont class]])
197 return NSType::Font;
198 if ([object isKindOfClass:[NSNumber class]])
199 return NSType::Number;
200 if ([object isKindOfClass:[NSString class]])
201 return NSType::String;
202 if ([object isKindOfClass:[NSURL class]])
203 return NSType::URL;
204
205 // Check NSSecureCoding after the specific cases since
206 // most of the classes above conform to NSSecureCoding,
207 // and we want our special case coders for them instead.
208 if ([object conformsToProtocol:@protocol(NSSecureCoding)])
209 return NSType::SecureCoding;
210
211 ASSERT_NOT_REACHED();
212 return NSType::Unknown;
213}

215static inline bool isSerializableFont(CTFontRef font)
216{
217 return adoptCF(CTFontCopyAttribute(font, kCTFontURLAttribute));
218}

220static inline bool isSerializableValue(id value)
221{
222 if ([value isKindOfClass:[CocoaFont class]])
223 return isSerializableFont((__bridge CTFontRef)value);
224 return typeFromObject(value) != NSType::Unknown;
225}

227#pragma mark - NSArray
228
229static void encodeArrayInternal(Encoder& encoder, NSArray *array)
230{
231 // Even though NSArray is toll free bridged with CFArrayRef, values may not be,
232 // so we should stay within this file's code.
233
234 if (!array.count) {
235 encoder << static_cast<uint64_t>(0);
236 return;
237 }
238
239 HashSet<NSUInteger> invalidIndicies;
240 for (NSUInteger i = 0; i < array.count; ++i) {
241 id value = array[i];
242
243 // Ignore values we don't support.
244 ASSERT(isSerializableValue(value));
245 if (!isSerializableValue(value))
246 invalidIndicies.add(i);
247 }
248
249 encoder << static_cast<uint64_t>(array.count - invalidIndicies.size());
250
251 for (NSUInteger i = 0; i < array.count; ++i) {
252 if (invalidIndicies.contains(i))
253 continue;
254 encodeObject(encoder, array[i]);
255 }
256}
257
258static std::optional<RetainPtr<id>> decodeArrayInternal(Decoder& decoder, NSArray<Class> *allowedClasses)
259{
260 uint64_t size;
261 if (!decoder.decode(size))
262 return std::nullopt;
263
264 auto array = adoptNS([[NSMutableArray alloc] init]);
265 for (uint64_t i = 0; i < size; ++i) {
266 auto value = decodeObject(decoder, allowedClasses);
267 if (!value || !value.value())
268 return std::nullopt;
269 [array addObject:value.value().get()];
270 }
271
272 return { array };
273}

335#pragma mark - NSDictionary
336
337static void encodeDictionaryInternal(Encoder& encoder, NSDictionary *dictionary)
338{
339 // Even though NSDictionary is toll free bridged with CFDictionaryRef, keys/values may not be,
340 // so we should stay within this file's code.
341
342 if (!dictionary.count) {
343 encoder << static_cast<uint64_t>(0);
344 return;
345 }
346
347 HashSet<id> invalidKeys;
348 for (id key in dictionary) {
349 id value = dictionary[key];
350 ASSERT(value);
351
352 // Ignore values we don't support.
353 ASSERT(isSerializableValue(key));
354 ASSERT(isSerializableValue(value));
355 if (!isSerializableValue(key) || !isSerializableValue(value))
356 invalidKeys.add(key);
357 }
358
359 encoder << static_cast<uint64_t>(dictionary.count - invalidKeys.size());
360
361 for (id key in dictionary) {
362 if (invalidKeys.contains(key))
363 continue;
364 encodeObject(encoder, key);
365 encodeObject(encoder, dictionary[key]);
366 }
367}
368
369static std::optional<RetainPtr<id>> decodeDictionaryInternal(Decoder& decoder, NSArray<Class> *allowedClasses)
370{
371 uint64_t size;
372 if (!decoder.decode(size))
373 return std::nullopt;
374
375 RetainPtr<NSMutableDictionary> dictionary = adoptNS([[NSMutableDictionary alloc] initWithCapacity:size]);
376 for (uint64_t i = 0; i < size; ++i) {
377 auto key = decodeObject(decoder, allowedClasses);
378 if (!key)
379 return std::nullopt;
380
381 auto value = decodeObject(decoder, allowedClasses);
382 if (!value)
383 return std::nullopt;
384
385 [dictionary setObject:value.value().get() forKey:key.value().get()];
386 }
387
388 return { dictionary };
389}

428#pragma mark - id <NSSecureCoding>
429
430static void encodeSecureCodingInternal(Encoder& encoder, id <NSObject, NSSecureCoding> object)
431{
432 auto archiver = adoptNS([[NSKeyedArchiver alloc] initRequiringSecureCoding:YES]);
433
434 auto delegate = adoptNS([[WKSecureCodingArchivingDelegate alloc] init]);
435 [archiver setDelegate:delegate.get()];
436
437 [archiver encodeObject:object forKey:NSKeyedArchiveRootObjectKey];
438 [archiver finishEncoding];
439 [archiver setDelegate:nil];
440
441 encoder << (__bridge CFDataRef)[archiver encodedData];
442}
443
444static std::optional<RetainPtr<id>> decodeSecureCodingInternal(Decoder& decoder, NSArray<Class> *allowedClasses)
445{
446 ASSERT(allowedClasses && allowedClasses.count);
447 if (!allowedClasses || !allowedClasses.count)
448 return std::nullopt;
449
450 RetainPtr<CFDataRef> data;
451 if (!decoder.decode(data))
452 return std::nullopt;
453
454 auto unarchiver = adoptNS([[NSKeyedUnarchiver alloc] initForReadingFromData:(__bridge NSData *)data.get() error:nullptr]);
455 unarchiver.get().decodingFailurePolicy = NSDecodingFailurePolicyRaiseException;
456
457 auto delegate = adoptNS([[WKSecureCodingArchivingDelegate alloc] init]);
458 unarchiver.get().delegate = delegate.get();
459
460 auto allowedClassSet = adoptNS([[NSMutableSet alloc] initWithArray:allowedClasses]);
461 [allowedClassSet addObject:WKSecureCodingURLWrapper.class];
462
463 @try {
464 id result = [unarchiver decodeObjectOfClasses:allowedClassSet.get() forKey:NSKeyedArchiveRootObjectKey];
465 ASSERT(!result || [result conformsToProtocol:@protocol(NSSecureCoding)]);
466 return { result };
467 } @catch (NSException *exception) {
468 LOG_ERROR("Failed to decode object of classes %@: %@", allowedClasses, exception);
469 return std::nullopt;
470 } @finally {
471 [unarchiver finishDecoding];
472 unarchiver.get().delegate = nil;
473 }
474}

506#pragma mark - Entry Point Encoder / Decoder
507
508void encodeObject(Encoder& encoder, id object)
509{
510 encoder << static_cast<bool>(!object);
511 if (!object)
512 return;
513
514 NSType type = typeFromObject(object);
515 encoder << type;
516
517 switch (type) {
518 case NSType::Array:
519 encodeArrayInternal(encoder, static_cast<NSArray *>(object));
520 return;
521 case NSType::Color:
522 encodeColorInternal(encoder, static_cast<CocoaColor *>(object));
523 return;
524 case NSType::Dictionary:
525 encodeDictionaryInternal(encoder, static_cast<NSDictionary *>(object));
526 return;
527 case NSType::Font:
528 encodeFontInternal(encoder, static_cast<CocoaFont *>(object));
529 return;
530 case NSType::Number:
531 encodeNumberInternal(encoder, static_cast<NSNumber *>(object));
532 return;
533 case NSType::SecureCoding:
534 encodeSecureCodingInternal(encoder, static_cast<id <NSObject, NSSecureCoding>>(object));
535 return;
536 case NSType::String:
537 encodeStringInternal(encoder, static_cast<NSString *>(object));
538 return;
539 case NSType::Date:
540 encodeDateInternal(encoder, static_cast<NSDate *>(object));
541 return;
542 case NSType::Data:
543 encodeDataInternal(encoder, static_cast<NSData *>(object));
544 return;
545 case NSType::URL:
546 encodeURLInternal(encoder, static_cast<NSURL *>(object));
547 return;
548 case NSType::Unknown:
549 break;
550 }
551
552 ASSERT_NOT_REACHED();
553}
554
555std::optional<RetainPtr<id>> decodeObject(Decoder& decoder, NSArray<Class> *allowedClasses)
556{
557 bool isNull;
558 if (!decoder.decode(isNull))
559 return std::nullopt;
560 if (isNull)
561 return { nullptr };
562
563 NSType type;
564 if (!decoder.decode(type))
565 return std::nullopt;
566
567 switch (type) {
568 case NSType::Array:
569 return decodeArrayInternal(decoder, allowedClasses);
570 case NSType::Color:
571 return decodeColorInternal(decoder);
572 case NSType::Dictionary:
573 return decodeDictionaryInternal(decoder, allowedClasses);
574 case NSType::Font:
575 return decodeFontInternal(decoder);
576 case NSType::Number:
577 return decodeNumberInternal(decoder);
578 case NSType::SecureCoding:
579 return decodeSecureCodingInternal(decoder, allowedClasses);
580 case NSType::String:
581 return decodeStringInternal(decoder);
582 case NSType::Date:
583 return decodeDateInternal(decoder);
584 case NSType::Data:
585 return decodeDataInternal(decoder);
586 case NSType::URL:
587 return decodeURLInternal(decoder);
588 case NSType::Unknown:
589 break;
590 }
591
592 ASSERT_NOT_REACHED();
593 return std::nullopt;
594}
