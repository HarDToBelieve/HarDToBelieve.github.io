39#if PLATFORM(MAC) || PLATFORM(MACCATALYST)
40static void initializeTimerCoalescingPolicy()
41{
42 // Set task_latency and task_throughput QOS tiers as appropriate for a visible application.
43 struct task_qos_policy qosinfo = { LATENCY_QOS_TIER_0, THROUGHPUT_QOS_TIER_0 };
44 kern_return_t kr = task_policy_set(mach_task_self(), TASK_BASE_QOS_POLICY, (task_policy_t)&qosinfo, TASK_QOS_POLICY_COUNT);
45 ASSERT_UNUSED(kr, kr == KERN_SUCCESS);
46}
47#endif
48
49void AuxiliaryProcess::platformInitialize(const AuxiliaryProcessInitializationParameters& parameters)
50{
51#if PLATFORM(MAC) || PLATFORM(MACCATALYST)
52 initializeTimerCoalescingPolicy();
53#endif
54
55 FloatingPointEnvironment& floatingPointEnvironment = FloatingPointEnvironment::singleton();
56#if PLATFORM(IOS_FAMILY) && !PLATFORM(MACCATALYST)
57 floatingPointEnvironment.enableDenormalSupport();
58#endif
59 floatingPointEnvironment.saveMainThreadEnvironment();
60
61 [[NSFileManager defaultManager] changeCurrentDirectoryPath:[[NSBundle mainBundle] bundlePath]];
62
63 WebCore::setApplicationBundleIdentifier(parameters.clientBundleIdentifier);
64 setApplicationSDKVersion(parameters.clientSDKVersion);
65}
66
67void AuxiliaryProcess::didReceiveInvalidMessage(IPC::Connection&, IPC::MessageName messageName)
68{
69 auto errorMessage = makeString("Received invalid message: '", description(messageName), "' (", messageName, ')');
70 logAndSetCrashLogMessage(errorMessage.utf8().data());
71 CRASH_WITH_INFO(WTF::enumToUnderlyingType(messageName));
72}

74bool AuxiliaryProcess::parentProcessHasEntitlement(const char* entitlement)
75{
76 return WTF::hasEntitlement(m_connection->xpcConnection(), entitlement);
77}

79void AuxiliaryProcess::platformStopRunLoop()
80{
81 XPCServiceExit(WTFMove(m_priorityBoostMessage));
82}
