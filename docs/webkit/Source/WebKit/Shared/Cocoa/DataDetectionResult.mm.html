35#if ENABLE(DATA_DETECTION)
36
37void DataDetectionResult::encode(IPC::Encoder& encoder) const
38{
39 encoder << results;
40}
41
42std::optional<DataDetectionResult> DataDetectionResult::decode(IPC::Decoder& decoder)
43{
44 auto results = IPC::decode<NSArray>(decoder, @[ NSArray.class, PAL::getDDScannerResultClass() ]);
45 if (!results)
46 return std::nullopt;
47
48 DataDetectionResult result;
49 result.results = WTFMove(*results);
50 return { WTFMove(result) };
51}
