26#include "config.h"
27#include "InsertTextOptions.h"

29namespace IPC {
30
31void ArgumentCoder<WebKit::InsertTextOptions>::encode(Encoder& encoder, const WebKit::InsertTextOptions& options)
32{
33 encoder << options.registerUndoGroup;
34 encoder << options.suppressSelectionUpdate;
35 encoder << options.processingUserGesture;
36 encoder << options.shouldSimulateKeyboardInput;
37 encoder << options.editingRangeIsRelativeTo;
38}
39
40std::optional<WebKit::InsertTextOptions> ArgumentCoder<WebKit::InsertTextOptions>::decode(Decoder& decoder)
41{
42 WebKit::InsertTextOptions options;
43 if (!decoder.decode(options.registerUndoGroup))
44 return std::nullopt;
45 if (!decoder.decode(options.suppressSelectionUpdate))
46 return std::nullopt;
47 if (!decoder.decode(options.processingUserGesture))
48 return std::nullopt;
49 if (!decoder.decode(options.shouldSimulateKeyboardInput))
50 return std::nullopt;
51 if (!decoder.decode(options.editingRangeIsRelativeTo))
52 return std::nullopt;
53 return options;
54}
