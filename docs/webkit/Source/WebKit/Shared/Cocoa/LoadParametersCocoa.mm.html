36void LoadParameters::platformEncode(IPC::Encoder& encoder) const
37{
38 IPC::encode(encoder, dataDetectionContext.get());
39
40 encoder << networkExtensionSandboxExtensionHandles;
41#if PLATFORM(IOS)
42 encoder << contentFilterExtensionHandle;
43 encoder << frontboardServiceExtensionHandle;
44#endif
45}
46
47bool LoadParameters::platformDecode(IPC::Decoder& decoder, LoadParameters& parameters)
48{
49 if (!IPC::decode(decoder, parameters.dataDetectionContext))
50 return false;
51
52 std::optional<SandboxExtension::HandleArray> networkExtensionSandboxExtensionHandles;
53 decoder >> networkExtensionSandboxExtensionHandles;
54 if (!networkExtensionSandboxExtensionHandles)
55 return false;
56 parameters.networkExtensionSandboxExtensionHandles = WTFMove(*networkExtensionSandboxExtensionHandles);
57
58#if PLATFORM(IOS)
59 std::optional<std::optional<SandboxExtension::Handle>> contentFilterExtensionHandle;
60 decoder >> contentFilterExtensionHandle;
61 if (!contentFilterExtensionHandle)
62 return false;
63 parameters.contentFilterExtensionHandle = WTFMove(*contentFilterExtensionHandle);
64
65 std::optional<std::optional<SandboxExtension::Handle>> frontboardServiceExtensionHandle;
66 decoder >> frontboardServiceExtensionHandle;
67 if (!frontboardServiceExtensionHandle)
68 return false;
69 parameters.frontboardServiceExtensionHandle = WTFMove(*frontboardServiceExtensionHandle);
70#endif
71
72 return true;
73}
