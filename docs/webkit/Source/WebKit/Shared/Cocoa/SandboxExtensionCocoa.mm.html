41class SandboxExtensionImpl {
42 WTF_MAKE_FAST_ALLOCATED;
43public:
44 static std::unique_ptr<SandboxExtensionImpl> create(const char* path, SandboxExtension::Type type, std::optional<audit_token_t> auditToken = std::nullopt, OptionSet<SandboxExtension::Flags> flags = SandboxExtension::Flags::Default)
45 {
46 std::unique_ptr<SandboxExtensionImpl> impl { new SandboxExtensionImpl(path, type, auditToken, flags) };
47 if (!impl->m_token)
48 return nullptr;
49 return impl;
50 }
51
52 SandboxExtensionImpl(const char* serializedFormat, size_t length)
53 : m_token { strndup(serializedFormat, length) }
54 {
55 }
56
57 ~SandboxExtensionImpl()
58 {
59 if (!m_token)
60 return;
61 auto length = strlen(m_token);
62 memset_s(m_token, length, 0, length);
63 free(m_token);
64 }
65
66 bool WARN_UNUSED_RETURN consume()
67 {
68 m_handle = sandbox_extension_consume(m_token);
69#if PLATFORM(IOS_FAMILY_SIMULATOR)
70 return !sandbox_check(getpid(), 0, SANDBOX_FILTER_NONE);
71#else
72 if (m_handle == -1) {
73 LOG_ERROR("Could not create a sandbox extension for '%s', errno = %d", m_token, errno);
74 return false;
75 }
76 return true;
77#endif
78 }
79
80 bool invalidate()
81 {
82 return !sandbox_extension_release(std::exchange(m_handle, 0));
83 }
84
85 const char* WARN_UNUSED_RETURN getSerializedFormat(size_t& length)
86 {
87 length = strlen(m_token);
88 return m_token;
89 }
90
91private:
92 char* sandboxExtensionForType(const char* path, SandboxExtension::Type type, std::optional<audit_token_t> auditToken, OptionSet<SandboxExtension::Flags> flags)
93 {
94 uint32_t extensionFlags = 0;
95 if (flags & SandboxExtension::Flags::NoReport)
96 extensionFlags |= SANDBOX_EXTENSION_NO_REPORT;
97 if (flags & SandboxExtension::Flags::DoNotCanonicalize)
98 extensionFlags |= SANDBOX_EXTENSION_CANONICAL;
99
100 switch (type) {
101 case SandboxExtension::Type::ReadOnly:
102 return sandbox_extension_issue_file(APP_SANDBOX_READ, path, extensionFlags);
103 case SandboxExtension::Type::ReadWrite:
104 return sandbox_extension_issue_file(APP_SANDBOX_READ_WRITE, path, extensionFlags);
105 case SandboxExtension::Type::Mach:
106 if (!auditToken)
107 return sandbox_extension_issue_mach("com.apple.webkit.extension.mach", path, extensionFlags);
108 return sandbox_extension_issue_mach_to_process("com.apple.webkit.extension.mach", path, extensionFlags, *auditToken);
109 case SandboxExtension::Type::IOKit:
110 if (!auditToken)
111 return sandbox_extension_issue_iokit_registry_entry_class("com.apple.webkit.extension.iokit", path, extensionFlags);
112 return sandbox_extension_issue_iokit_registry_entry_class_to_process("com.apple.webkit.extension.iokit", path, extensionFlags, *auditToken);
113 case SandboxExtension::Type::Generic:
114 return sandbox_extension_issue_generic(path, extensionFlags);
115 case SandboxExtension::Type::ReadByProcess:
116 if (!auditToken)
117 return nullptr;
118 return sandbox_extension_issue_file_to_process(APP_SANDBOX_READ, path, extensionFlags, *auditToken);
119 }
120 }
121
122 SandboxExtensionImpl(const char* path, SandboxExtension::Type type, std::optional<audit_token_t> auditToken, OptionSet<SandboxExtension::Flags> flags)
123 : m_token { sandboxExtensionForType(path, type, auditToken, flags) }
124 {
125 }
126
127 char* m_token;
128 int64_t m_handle { 0 };
129};

144void SandboxExtension::Handle::encode(IPC::Encoder& encoder) const
145{
146 if (!m_sandboxExtension) {
147 encoder << IPC::DataReference();
148 return;
149 }
150
151 size_t length = 0;
152 const char* serializedFormat = m_sandboxExtension->getSerializedFormat(length);
153 ASSERT(serializedFormat);
154
155 encoder << IPC::DataReference(reinterpret_cast<const uint8_t*>(serializedFormat), length);
156
157 // Encoding will destroy the sandbox extension locally.
158 m_sandboxExtension = 0;
159}
160
161auto SandboxExtension::Handle::decode(IPC::Decoder& decoder) -> std::optional<Handle>
162{
163 IPC::DataReference dataReference;
164 if (!decoder.decode(dataReference))
165 return std::nullopt;
166
167 if (dataReference.isEmpty())
168 return {{ }};
169
170 Handle handle;
171 handle.m_sandboxExtension = makeUnique<SandboxExtensionImpl>(reinterpret_cast<const char*>(dataReference.data()), dataReference.size());
172 return WTFMove(handle);
173}

222std::optional<SandboxExtension::HandleArray> SandboxExtension::HandleArray::decode(IPC::Decoder& decoder)
223{
224 std::optional<uint64_t> size;
225 decoder >> size;
226 if (!size)
227 return std::nullopt;
228
229 SandboxExtension::HandleArray handles;
230 for (size_t i = 0; i < *size; ++i) {
231 std::optional<SandboxExtension::Handle> handle;
232 decoder >> handle;
233 if (!handle)
234 return std::nullopt;
235 handles.append(WTFMove(*handle));
236 }
237 return WTFMove(handles);
238}

248String stringByResolvingSymlinksInPath(const String& path)
249{
250 char resolvedPath[PATH_MAX] = { 0 };
251 realpath(path.utf8().data(), resolvedPath);
252 return String::fromUTF8(resolvedPath);
253}
254
255String resolveAndCreateReadWriteDirectoryForSandboxExtension(const String& path)
256{
257 NSError *error = nil;
258 NSString *nsPath = path;
259
260 if (![[NSFileManager defaultManager] createDirectoryAtPath:nsPath withIntermediateDirectories:YES attributes:nil error:&error]) {
261 NSLog(@"could not create directory \"%@\" for future sandbox extension, error %@", nsPath, error);
262 return { };
263 }
264
265 return resolvePathForSandboxExtension(path);
266}
267
268String resolvePathForSandboxExtension(const String& path)
269{
270 String resolvedPath = stringByResolvingSymlinksInPath(path);
271 if (resolvedPath.isNull()) {
272 LOG_ERROR("Could not create a valid file system representation for the string '%s' of length %lu", resolvedPath.utf8().data(), resolvedPath.length());
273 return { };
274 }
275
276 return resolvedPath;
277}

279bool SandboxExtension::createHandleWithoutResolvingPath(const String& path, Type type, Handle& handle)
280{
281 ASSERT(!handle.m_sandboxExtension);
282
283 handle.m_sandboxExtension = SandboxExtensionImpl::create(path.utf8().data(), type, std::nullopt, SandboxExtension::Flags::DoNotCanonicalize);
284 if (!handle.m_sandboxExtension) {
285 LOG_ERROR("Could not create a sandbox extension for '%s'", path.utf8().data());
286 return false;
287 }
288 return true;
289}
290
291bool SandboxExtension::createHandle(const String& path, Type type, Handle& handle)
292{
293 ASSERT(!handle.m_sandboxExtension);
294
295 return createHandleWithoutResolvingPath(resolvePathForSandboxExtension(path), type, handle);

298template <typename T>
299static SandboxExtension::HandleArray createHandlesForResources(const Vector<T>& resources, Function<bool(const T&, SandboxExtension::Handle& handle)>&& createFunction)
300{
301 SandboxExtension::HandleArray handleArray;
302
303 if (resources.size() > 0)
304 handleArray.allocate(resources.size());
305
306 size_t currentHandle = 0;
307 for (const auto& resource : resources) {
308 if (!createFunction(resource, handleArray[currentHandle]))
309 continue;
310 ++currentHandle;
311 }
312
313 return handleArray;
314}

316SandboxExtension::HandleArray SandboxExtension::createReadOnlyHandlesForFiles(ASCIILiteral logLabel, const Vector<String>& paths)
317{
318 return createHandlesForResources(paths, Function<bool(const String&, Handle&)>([&logLabel] (const String& path, Handle& handle) {
319 if (!SandboxExtension::createHandle(path, SandboxExtension::Type::ReadOnly, handle)) {
320 // This can legitimately fail if a directory containing the file is deleted after the file was chosen.
321 // We also have reports of cases where this likely fails for some unknown reason, <rdar://problem/10156710>.
322 WTFLogAlways("%s: could not create a sandbox extension for '%s'\n", logLabel.characters(), path.utf8().data());
323 ASSERT_NOT_REACHED();
324 return false;
325 }
326 return true;
327 }));
328}

339String SandboxExtension::createHandleForTemporaryFile(const String& prefix, Type type, Handle& handle)
340{
341 ASSERT(!handle.m_sandboxExtension);
342
343 Vector<char> path(PATH_MAX);
344 if (!confstr(_CS_DARWIN_USER_TEMP_DIR, path.data(), path.size()))
345 return String();
346
347 // Shrink the vector.
348 path.shrink(strlen(path.data()));
349
350 // FIXME: Change to a runtime assertion that the path ends with a slash once <rdar://problem/23579077> is
351 // fixed in all iOS Simulator versions that we use.
352 if (path.last() != '/')
353 path.append('/');
354
355 // Append the file name.
356 path.append(prefix.utf8().data(), prefix.length());
357 path.append('\0');
358
359 handle.m_sandboxExtension = SandboxExtensionImpl::create(FileSystem::fileSystemRepresentation(path.data()).data(), type);
360
361 if (!handle.m_sandboxExtension) {
362 WTFLogAlways("Could not create a sandbox extension for temporary file '%s'", path.data());
363 return String();
364 }
365 return String(path.data());
366}

455bool SandboxExtension::revoke()
456{
457 ASSERT(m_sandboxExtension);
458 ASSERT(m_useCount);
459
460 if (--m_useCount)
461 return true;
462
463 return m_sandboxExtension->invalidate();
464}

466bool SandboxExtension::consume()
467{
468 ASSERT(m_sandboxExtension);
469
470 if (m_useCount++)
471 return true;
472
473 return m_sandboxExtension->consume();
474}

476bool SandboxExtension::consumePermanently()
477{
478 ASSERT(m_sandboxExtension);
479
480 bool result = m_sandboxExtension->consume();
481
482 // Destroy the extension without invalidating it.
483 m_sandboxExtension = nullptr;
484
485 return result;
486}
487
488bool SandboxExtension::consumePermanently(const Handle& handle)
489{
490 if (!handle.m_sandboxExtension)
491 return false;
492
493 bool result = handle.m_sandboxExtension->consume();
494
495 // Destroy the extension without invalidating it.
496 handle.m_sandboxExtension = nullptr;
497
498 return result;
499}
500
501bool SandboxExtension::consumePermanently(const HandleArray& handleArray)
502{
503 bool allSucceeded = true;
504 for (auto& handle : handleArray) {
505 if (!handle.m_sandboxExtension)
506 continue;
507
508 bool ok = SandboxExtension::consumePermanently(handle);
509 ASSERT(ok);
510 allSucceeded &= ok;
511 }
512
513 return allSucceeded;
514}
