26#import "config.h"
27#import "WKNSArray.h"
28
29#import <WebCore/WebCoreObjCExtras.h>
30
31@implementation WKNSArray {
32 API::ObjectStorage<API::Array> _array;
33}
34
35- (void)dealloc
36{
37 if (WebCoreObjCScheduleDeallocateOnMainRunLoop(WKNSArray.class, self))
38 return;
39
40 _array->~Array();
41
42 [super dealloc];
43}
44
45#pragma mark NSArray primitive methods
46
47- (NSUInteger)count
48{
49 return _array->size();
50}
51
52- (id)objectAtIndex:(NSUInteger)i
53{
54 API::Object* object = _array->at(i);
55 return object ? object->wrapper() : [NSNull null];
56}
57
58#pragma mark NSCopying protocol implementation
59
60- (id)copyWithZone:(NSZone *)zone
61{
62 return [self retain];
63}
64
65#pragma mark WKObject protocol implementation
66
67- (API::Object&)_apiObject
68{
69 return *_array;
70}
71
72@end
