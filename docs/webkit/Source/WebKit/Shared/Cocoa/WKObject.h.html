35template<typename ObjectClass> struct ObjectStorage {
36 ObjectClass* get() { return reinterpret_cast<ObjectClass*>(&data); }
37 ObjectClass& operator*() { return *get(); }
38 ObjectClass* operator->() { return get(); }
39
40 typename std::aligned_storage<sizeof(ObjectClass), std::alignment_of<ObjectClass>::value>::type data;
41};

44void* wrap(API::Object*);

48namespace WebKit {
49
50template<typename WrappedObjectClass> struct WrapperTraits;
51
52template<typename DestinationClass, typename SourceClass> inline DestinationClass *checkedObjCCast(SourceClass *source)
53{
54 ASSERT([source isKindOfClass:[DestinationClass class]]);
55 return (DestinationClass *)source;
56}
57
58template<typename ObjectClass> inline typename WrapperTraits<ObjectClass>::WrapperClass *wrapper(ObjectClass& object)
59{
60 return checkedObjCCast<typename WrapperTraits<ObjectClass>::WrapperClass>(object.wrapper());
61}
62
63template<typename ObjectClass> inline typename WrapperTraits<ObjectClass>::WrapperClass *wrapper(ObjectClass* object)
64{
65 return object ? wrapper(*object) : nil;
66}
67
68template<typename ObjectClass> inline typename WrapperTraits<ObjectClass>::WrapperClass *wrapper(const Ref<ObjectClass>& object)
69{
70 return wrapper(object.get());
71}
72
73template<typename ObjectClass> inline typename WrapperTraits<ObjectClass>::WrapperClass *wrapper(const RefPtr<ObjectClass>& object)
74{
75 return wrapper(object.get());
76}
77
78template<typename ObjectClass> inline typename WrapperTraits<ObjectClass>::WrapperClass *wrapper(Ref<ObjectClass>&& object)
79{
80 return adoptNS(wrapper(object.leakRef())).autorelease();
81}
82
83template<typename ObjectClass> inline typename WrapperTraits<ObjectClass>::WrapperClass *wrapper(RefPtr<ObjectClass>&& object)
84{
85 return object ? wrapper(object.releaseNonNull()) : nil;
86}
