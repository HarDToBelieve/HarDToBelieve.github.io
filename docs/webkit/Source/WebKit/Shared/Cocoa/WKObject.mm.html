48@implementation WKObject {
49 BOOL _hasInitializedTarget;
50 RetainPtr<NSObject> _target;
51}
52
53- (void)dealloc
54{
55 if (WebCoreObjCScheduleDeallocateOnMainRunLoop(WKObject.class, self))
56 return;
57
58 API::Object::fromWKObjectExtraSpace(self).~Object();
59
60 [super dealloc];
61}

63static inline void initializeTargetIfNeeded(WKObject *self)
64{
65 if (self->_hasInitializedTarget)
66 return;
67
68 self->_hasInitializedTarget = YES;
69 self->_target = adoptNS([self _web_createTarget]);
70}
71
72- (BOOL)isEqual:(id)object
73{
74 if (object == self)
75 return YES;
76
77 if (!object)
78 return NO;
79
80 initializeTargetIfNeeded(self);
81
82 return [_target isEqual:object];
83}
