40// Because of <rdar://problem/60608008>, WebKit has to parse the feature flags plist file
41bool isFeatureFlagEnabled(const String& featureName)
42{
43 BOOL isWebKitBundleFromStagedFramework = [[[NSBundle mainBundle] bundlePath] hasPrefix:@"/Library/Apple/System/Library/StagedFrameworks/WebKit"];
44
45 if (!isWebKitBundleFromStagedFramework)
46 return _os_feature_enabled_impl("WebKit", (const char*)featureName.utf8().data());
47
48 static NeverDestroyed<RetainPtr<NSDictionary>> dictionary = [NSDictionary dictionaryWithContentsOfFile:@"/Library/Apple/System/Library/FeatureFlags/Domain/WebKit.plist"];
49
50 if (![[dictionary.get() objectForKey:featureName] objectForKey:@"Enabled"])
51 return _os_feature_enabled_impl("WebKit", (const char*)featureName.characters8());
52
53 return [[[dictionary.get() objectForKey:featureName] objectForKey:@"Enabled"] isKindOfClass:[NSNumber class]] && [[[dictionary.get() objectForKey:featureName] objectForKey:@"Enabled"] boolValue];
54}
