41XPCEndpoint::XPCEndpoint()
42{
43 m_connection = adoptOSObject(xpc_connection_create(nullptr, nullptr));
44 m_endpoint = adoptOSObject(xpc_endpoint_create(m_connection.get()));
45
46 xpc_connection_set_target_queue(m_connection.get(), dispatch_get_main_queue());
47 xpc_connection_set_event_handler(m_connection.get(), ^(xpc_object_t message) {
48 xpc_type_t type = xpc_get_type(message);
49
50 if (type == XPC_TYPE_CONNECTION) {
51 OSObjectPtr<xpc_connection_t> connection = message;
52#if USE(APPLE_INTERNAL_SDK)
53 auto pid = xpc_connection_get_pid(connection.get());
54
55 if (pid != getpid() && !WTF::hasEntitlement(connection.get(), "com.apple.private.webkit.use-xpc-endpoint")) {
56 WTFLogAlways("Audit token does not have required entitlement com.apple.private.webkit.use-xpc-endpoint");
57#if PLATFORM(MAC)
58 audit_token_t auditToken;
59 xpc_connection_get_audit_token(connection.get(), &auditToken);
60 bool isPlatformBinary = SecTaskGetCodeSignStatus(adoptCF(SecTaskCreateWithAuditToken(kCFAllocatorDefault, auditToken)).get()) & CS_PLATFORM_BINARY;
61 auto signingIdentifier = codeSigningIdentifier(connection.get());
62
63 if (!isPlatformBinary || !signingIdentifier.startsWith("com.apple.WebKit.WebContent")) {
64 WTFLogAlways("XPC endpoint denied to connect with unknown client");
65 return;
66 }
67#else
68 return;
69#endif
70 }
71#endif // USE(APPLE_INTERNAL_SDK)
72 xpc_connection_set_target_queue(connection.get(), dispatch_get_main_queue());
73 xpc_connection_set_event_handler(connection.get(), ^(xpc_object_t event) {
74 handleEvent(connection.get(), event);
75 });
76 xpc_connection_resume(connection.get());
77 }
78 });
79 xpc_connection_resume(m_connection.get());
80}
81
82void XPCEndpoint::sendEndpointToConnection(xpc_connection_t connection)
83{
84 if (!connection)
85 return;
86
87 auto message = adoptOSObject(xpc_dictionary_create(nullptr, nullptr, 0));
88 xpc_dictionary_set_string(message.get(), xpcEndpointMessageNameKey(), xpcEndpointMessageName());
89 xpc_dictionary_set_value(message.get(), xpcEndpointNameKey(), m_endpoint.get());
90
91 xpc_connection_send_message(connection, message.get());
92}
93
94OSObjectPtr<xpc_endpoint_t> XPCEndpoint::endpoint() const
95{
96 return m_endpoint;
97}
