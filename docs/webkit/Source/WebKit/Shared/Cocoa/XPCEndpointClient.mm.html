33void XPCEndpointClient::setEndpoint(xpc_endpoint_t endpoint)
34{
35 {
36 Locker locker { m_connectionLock };
37
38 if (m_connection)
39 return;
40
41 m_connection = adoptOSObject(xpc_connection_create_from_endpoint(endpoint));
42
43 xpc_connection_set_target_queue(m_connection.get(), dispatch_get_global_queue(DISPATCH_QUEUE_PRIORITY_DEFAULT, 0));
44 xpc_connection_set_event_handler(m_connection.get(), ^(xpc_object_t message) {
45 xpc_type_t type = xpc_get_type(message);
46 if (type == XPC_TYPE_ERROR) {
47 if (message == XPC_ERROR_CONNECTION_INVALID || message == XPC_ERROR_TERMINATION_IMMINENT) {
48 Locker locker { m_connectionLock };
49 m_connection = nullptr;
50 }
51 return;
52 }
53 if (type != XPC_TYPE_DICTIONARY)
54 return;
55
56 auto connection = xpc_dictionary_get_remote_connection(message);
57 if (!connection)
58 return;
59#if USE(APPLE_INTERNAL_SDK)
60 auto pid = xpc_connection_get_pid(connection);
61 if (pid != getpid() && !WTF::hasEntitlement(connection, "com.apple.private.webkit.use-xpc-endpoint")) {
62 WTFLogAlways("Audit token does not have required entitlement com.apple.private.webkit.use-xpc-endpoint");
63 return;
64 }
65#endif
66 handleEvent(message);
67 });
68
69 xpc_connection_resume(m_connection.get());
70 }
71
72 didConnect();
73}
74
75OSObjectPtr<xpc_connection_t> XPCEndpointClient::connection()
76{
77 Locker locker { m_connectionLock };
78 return m_connection;
79}
