36namespace WebKit {
37
38Ref<XRDeviceProxy> XRDeviceProxy::create(XRDeviceInfo&& deviceInfo, PlatformXRSystemProxy& xrSystem)
39{
40 return adoptRef(*new XRDeviceProxy(WTFMove(deviceInfo), xrSystem));
41}
42
43XRDeviceProxy::XRDeviceProxy(XRDeviceInfo&& deviceInfo, PlatformXRSystemProxy& xrSystem)
44 : m_xrSystem(makeWeakPtr(xrSystem))
45{
46 m_identifier = deviceInfo.identifier;
47 m_supportsStereoRendering = deviceInfo.supportsStereoRendering;
48 m_supportsOrientationTracking = deviceInfo.supportsOrientationTracking;
49 m_recommendedResolution = deviceInfo.recommendedResolution;
50 if (!deviceInfo.features.isEmpty())
51 setEnabledFeatures(SessionMode::ImmersiveVr, deviceInfo.features);
52}
53
54void XRDeviceProxy::sessionDidEnd()
55{
56 if (trackingAndRenderingClient())
57 trackingAndRenderingClient()->sessionDidEnd();
58}
59
60void XRDeviceProxy::initializeTrackingAndRendering(PlatformXR::SessionMode sessionMode)
61{
62 if (sessionMode != PlatformXR::SessionMode::ImmersiveVr)
63 return;
64
65 if (m_xrSystem)
66 m_xrSystem->initializeTrackingAndRendering();
67}
68
69void XRDeviceProxy::shutDownTrackingAndRendering()
70{
71 if (m_xrSystem)
72 m_xrSystem->shutDownTrackingAndRendering();
73}
74
75Vector<PlatformXR::Device::ViewData> XRDeviceProxy::views(SessionMode mode) const
76{
77 Vector<Device::ViewData> views;
78 if (m_supportsStereoRendering && mode == SessionMode::ImmersiveVr) {
79 views.append({ .active = true, Eye::Left });
80 views.append({ .active = true, Eye::Right });
81 } else
82 views.append({ .active = true, Eye::None });
83 return views;
84}
85
86void XRDeviceProxy::requestFrame(PlatformXR::Device::RequestFrameCallback&& callback)
87{
88 if (m_xrSystem)
89 m_xrSystem->requestFrame(WTFMove(callback));
90}
91
92std::optional<PlatformXR::LayerHandle> XRDeviceProxy::createLayerProjection(uint32_t width, uint32_t height, bool alpha)
93{
94 return m_xrSystem ? m_xrSystem->createLayerProjection(width, height, alpha) : std::nullopt;
95}
96
97void XRDeviceProxy::submitFrame(Vector<PlatformXR::Device::Layer>&&)
98{
99 if (m_xrSystem)
100 m_xrSystem->submitFrame();
101}
