41namespace WebKit {
42
43class ContextMenuContextData {
44public:
45 using Type = WebCore::ContextMenuContext::Type;
46
47 ContextMenuContextData();
48 ContextMenuContextData(const WebCore::IntPoint& menuLocation, const Vector<WebKit::WebContextMenuItemData>& menuItems, const WebCore::ContextMenuContext&);
49
50 Type type() const { return m_type; }
51 const WebCore::IntPoint& menuLocation() const { return m_menuLocation; }
52 const Vector<WebKit::WebContextMenuItemData>& menuItems() const { return m_menuItems; }
53
54 WebHitTestResultData& webHitTestResultData() { return m_webHitTestResultData; }
55 const WebHitTestResultData& webHitTestResultData() const { return m_webHitTestResultData; }
56 const String& selectedText() const { return m_selectedText; }
57
58#if ENABLE(SERVICE_CONTROLS)
59 ContextMenuContextData(const WebCore::IntPoint& menuLocation, const Vector<uint8_t>& selectionData, const Vector<String>& selectedTelephoneNumbers, bool isEditable)
60 : m_type(Type::ServicesMenu)
61 , m_menuLocation(menuLocation)
62 , m_controlledSelectionData(selectionData)
63 , m_selectedTelephoneNumbers(selectedTelephoneNumbers)
64 , m_selectionIsEditable(isEditable)
65 {
66 }
67
68 ShareableBitmap* controlledImage() const { return m_controlledImage.get(); }
69 const Vector<uint8_t>& controlledSelectionData() const { return m_controlledSelectionData; }
70 const Vector<String>& selectedTelephoneNumbers() const { return m_selectedTelephoneNumbers; }
71
72 bool isServicesMenu() const { return m_type == ContextMenuContextData::Type::ServicesMenu; }
73 bool controlledDataIsEditable() const;
74#endif
75
76 void encode(IPC::Encoder&) const;
77 static WARN_UNUSED_RETURN bool decode(IPC::Decoder&, ContextMenuContextData&);
78
79private:
80 Type m_type;
81
82 WebCore::IntPoint m_menuLocation;
83 Vector<WebKit::WebContextMenuItemData> m_menuItems;
84
85 WebHitTestResultData m_webHitTestResultData;
86 String m_selectedText;
87
88#if ENABLE(SERVICE_CONTROLS)
89 RefPtr<ShareableBitmap> m_controlledImage;
90 Vector<uint8_t> m_controlledSelectionData;
91 Vector<String> m_selectedTelephoneNumbers;
92 bool m_selectionIsEditable;
93#endif
94};

96} // namespace WebKit
