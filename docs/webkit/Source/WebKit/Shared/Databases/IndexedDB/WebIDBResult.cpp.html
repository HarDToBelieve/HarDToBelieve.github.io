31namespace WebKit {
32
33void WebIDBResult::encode(IPC::Encoder& encoder) const
34{
35 m_resultData.encode(encoder);
36 m_handles.encode(encoder);
37}
38
39bool WebIDBResult::decode(IPC::Decoder& decoder, WebIDBResult& result)
40{
41 std::optional<WebCore::IDBResultData> resultData;
42 decoder >> resultData;
43 if (!resultData)
44 return false;
45 result.m_resultData = WTFMove(*resultData);
46
47 std::optional<SandboxExtension::HandleArray> handles;
48 decoder >> handles;
49 if (!handles)
50 return false;
51 result.m_handles = WTFMove(*handles);
52
53 return true;
54}
