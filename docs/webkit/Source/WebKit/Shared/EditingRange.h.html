39enum class EditingRangeIsRelativeTo : uint8_t {
40 EditableRoot,
41 Paragraph,
42};

44// FIXME: Replace this with std::optional<CharacterRange>.
45struct EditingRange {
46 EditingRange()
47 : location(notFound)
48 , length(0)
49 {
50 }
51
52 EditingRange(uint64_t location, uint64_t length)
53 : location(location)
54 , length(length)
55 {
56 }
57
58 // (notFound, 0) is notably valid.
59 bool isValid() const { return location + length >= location; }
60
61 static std::optional<WebCore::SimpleRange> toRange(WebCore::Frame&, const EditingRange&, EditingRangeIsRelativeTo = EditingRangeIsRelativeTo::EditableRoot);
62 static EditingRange fromRange(WebCore::Frame&, const std::optional<WebCore::SimpleRange>&, EditingRangeIsRelativeTo = EditingRangeIsRelativeTo::EditableRoot);
63
64#if defined(__OBJC__)
65 EditingRange(NSRange range)
66 {
67 if (range.location != NSNotFound) {
68 location = range.location;
69 length = range.length;
70 } else {
71 location = notFound;
72 length = 0;
73 }
74 }
75
76 operator NSRange() const
77 {
78 if (location == notFound)
79 return NSMakeRange(NSNotFound, 0);
80 return NSMakeRange(location, length);
81 }
82#endif
83
84 uint64_t location;
85 uint64_t length;
86};
87
