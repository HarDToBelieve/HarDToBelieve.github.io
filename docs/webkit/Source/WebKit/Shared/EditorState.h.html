72struct EditorState {
73 TransactionID transactionID;
74 String originIdentifierForPasteboard;
75 bool shouldIgnoreSelectionChanges { false };
76 bool selectionIsNone { true }; // This will be false when there is a caret selection.
77 bool selectionIsRange { false };
78 bool selectionIsRangeInsideImageOverlay { false };
79 bool isContentEditable { false };
80 bool isContentRichlyEditable { false };
81 bool isInPasswordField { false };
82 bool isInPlugin { false };
83 bool hasComposition { false };
84 bool isMissingPostLayoutData { true };
85
86 struct PostLayoutData {
87 uint32_t typingAttributes { AttributeNone };
88#if PLATFORM(IOS_FAMILY) || PLATFORM(GTK) || PLATFORM(WPE)
89 WebCore::IntRect caretRectAtStart;
90#endif
91#if PLATFORM(COCOA)
92 uint64_t selectedTextLength { 0 };
93 uint32_t textAlignment { NoAlignment };
94 WebCore::Color textColor { WebCore::Color::black };
95 uint32_t enclosingListType { NoList };
96 WebCore::WritingDirection baseWritingDirection { WebCore::WritingDirection::Natural };
97#endif
98#if PLATFORM(IOS_FAMILY)
99 WebCore::IntRect selectionClipRect;
100 WebCore::IntRect caretRectAtEnd;
101 Vector<WebCore::SelectionGeometry> selectionGeometries;
102 Vector<WebCore::SelectionGeometry> markedTextRects;
103 String markedText;
104 WebCore::IntRect markedTextCaretRectAtStart;
105 WebCore::IntRect markedTextCaretRectAtEnd;
106 String wordAtSelection;
107 UChar32 characterAfterSelection { 0 };
108 UChar32 characterBeforeSelection { 0 };
109 UChar32 twoCharacterBeforeSelection { 0 };
110#if USE(DICTATION_ALTERNATIVES)
111 Vector<WebCore::DictationContext> dictationContextsForSelection;
112#endif
113 bool isReplaceAllowed { false };
114 bool hasContent { false };
115 bool isStableStateUpdate { false };
116 bool insideFixedPosition { false };
117 bool hasPlainText { false };
118 bool editableRootIsTransparentOrFullyClipped { false };
119 WebCore::Color caretColor;
120 bool atStartOfSentence { false };
121 bool selectionStartIsAtParagraphBoundary { false };
122 bool selectionEndIsAtParagraphBoundary { false };
123#endif
124#if PLATFORM(MAC)
125 WebCore::IntRect selectionBoundingRect;
126 uint64_t candidateRequestStartPosition { 0 };
127 String paragraphContextForCandidateRequest;
128 String stringForCandidateRequest;
129 bool canEnableAutomaticSpellingCorrection { true };
130#endif
131#if PLATFORM(GTK) || PLATFORM(WPE)
132 String surroundingContext;
133 uint64_t surroundingContextCursorPosition { 0 };
134 uint64_t surroundingContextSelectionPosition { 0 };
135#endif
136
137 std::optional<WebCore::FontAttributes> fontAttributes;
138
139 bool canCut { false };
140 bool canCopy { false };
141 bool canPaste { false };
142
143 void encode(IPC::Encoder&) const;
144 static WARN_UNUSED_RETURN bool decode(IPC::Decoder&, PostLayoutData&);
145 };
146
147 const PostLayoutData& postLayoutData() const;
148 PostLayoutData& postLayoutData();
149
150 void encode(IPC::Encoder&) const;
151 static WARN_UNUSED_RETURN bool decode(IPC::Decoder&, EditorState&);
152
153private:
154 PostLayoutData m_postLayoutData;
155};
156
157inline auto EditorState::postLayoutData() -> PostLayoutData&
158{
159 ASSERT_WITH_MESSAGE(!isMissingPostLayoutData, "Attempt to access post layout data before receiving it");
160 return m_postLayoutData;
161}
162
163inline auto EditorState::postLayoutData() const -> const PostLayoutData&
164{
165 ASSERT_WITH_MESSAGE(!isMissingPostLayoutData, "Attempt to access post layout data before receiving it");
166 return m_postLayoutData;
167}
