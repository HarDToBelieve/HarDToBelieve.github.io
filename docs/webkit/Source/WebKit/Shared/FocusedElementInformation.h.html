68struct OptionItem {
69 OptionItem() = default;
70
71 OptionItem(const String& text, bool isGroup, int parentID, bool selected, bool disabled)
72 : text(text)
73 , isGroup(isGroup)
74 , isSelected(selected)
75 , disabled(disabled)
76 , parentGroupID(parentID)
77 {
78 }
79
80 String text;
81 bool isGroup { false };
82 bool isSelected { false };
83 bool disabled { false };
84 int parentGroupID { 0 };
85
86 void encode(IPC::Encoder&) const;
87 static std::optional<OptionItem> decode(IPC::Decoder&);
88};

99 double nodeFontSize { 0 };
100 bool hasNextNode { false };

102 bool hasPreviousNode { false };
103 WebCore::IntRect previousNodeRect;
104 bool isAutocorrect { false };
105 bool isRTL { false };
106 bool isMultiSelect { false };
107 bool isReadOnly {false };
108 bool allowsUserScaling { false };
109 bool allowsUserScalingIgnoringAlwaysScalable { false };
110 bool insideFixedPosition { false };
111 WebCore::AutocapitalizeType autocapitalizeType { WebCore::AutocapitalizeType::Default };
112 InputType elementType { InputType::None };
113 WebCore::InputMode inputMode { WebCore::InputMode::Unspecified };
114 WebCore::EnterKeyHint enterKeyHint { WebCore::EnterKeyHint::Unspecified };
115 String formAction;
116 Vector<OptionItem> selectOptions;
117 int selectedIndex { -1 };
118 String value;
119 double valueAsNumber { 0 };
120 String title;
121 bool acceptsAutofilledLoginCredentials { false };

123 URL representingPageURL;
124 WebCore::AutofillFieldName autofillFieldName { WebCore::AutofillFieldName::None };
125 String placeholder;
126 String label;
127 String ariaLabel;
128#if ENABLE(DATALIST_ELEMENT)
129 bool hasSuggestions { false };

136 bool shouldSynthesizeKeyEventsForEditing { false };

143 FocusedElementIdentifier focusedElementIdentifier { 0 };
144 WebCore::ScrollingNodeID containerScrollingNodeID { 0 };

146 void encode(IPC::Encoder&) const;
147 static WARN_UNUSED_RETURN bool decode(IPC::Decoder&, FocusedElementInformation&);
148};
