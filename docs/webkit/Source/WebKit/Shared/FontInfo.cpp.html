35namespace WebKit {
36
37void FontInfo::encode(IPC::Encoder& encoder) const
38{
39#if PLATFORM(COCOA)
40 encoder << static_cast<bool>(fontAttributeDictionary);
41 if (fontAttributeDictionary)
42 encoder << fontAttributeDictionary;
43#else
44 UNUSED_PARAM(encoder);
45#endif
46}
47
48bool FontInfo::decode(IPC::Decoder& decoder, FontInfo& fontInfo)
49{
50#if PLATFORM(COCOA)
51 bool hasFontAttributeDictionary;
52 if (!decoder.decode(hasFontAttributeDictionary))
53 return false;
54
55 if (!hasFontAttributeDictionary)
56 return true;
57
58 if (!decoder.decode(fontInfo.fontAttributeDictionary))
59 return false;
60#else
61 UNUSED_PARAM(decoder);
62 UNUSED_PARAM(fontInfo);
63#endif
64
65 return true;
66}
