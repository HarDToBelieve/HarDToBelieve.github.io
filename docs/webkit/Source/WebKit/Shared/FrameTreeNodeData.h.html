26#pragma once
27
28#include "ArgumentCoders.h"
29#include "FrameInfoData.h"
30
31namespace WebKit {
32
33struct FrameTreeNodeData {
34
35 FrameInfoData info;
36 Vector<FrameTreeNodeData> children;
37
38 void encode(IPC::Encoder& encoder) const
39 {
40 encoder << info;
41 encoder << children;
42 }
43
44 static std::optional<FrameTreeNodeData> decode(IPC::Decoder& decoder)
45 {
46 std::optional<FrameInfoData> info;
47 decoder >> info;
48 if (!info)
49 return std::nullopt;
50
51 std::optional<Vector<FrameTreeNodeData>> children;
52 decoder >> children;
53 if (!children)
54 return std::nullopt;
55
56 return {{
57 WTFMove(*info),
58 WTFMove(*children)
59 }};
60 }
61};
62
63}
