26#pragma once
27
28#if ENABLE(GPU_PROCESS)
29
30#include <wtf/MachSendRight.h>
31
32namespace WebKit {
33
34struct GPUProcessConnectionParameters {
35#if HAVE(TASK_IDENTITY_TOKEN)
36 MachSendRight webProcessIdentityToken;
37#endif
38 Vector<String> overrideLanguages;
39#if ENABLE(IPC_TESTING_API)
40 bool ignoreInvalidMessageForTesting { false };
41#endif
42
43 void encode(IPC::Encoder& encoder) const
44 {
45#if HAVE(TASK_IDENTITY_TOKEN)
46 encoder << webProcessIdentityToken;
47#endif
48 encoder << overrideLanguages;
49#if ENABLE(IPC_TESTING_API)
50 encoder << ignoreInvalidMessageForTesting;
51#endif
52 }
53
54 static std::optional<GPUProcessConnectionParameters> decode(IPC::Decoder& decoder)
55 {
56#if HAVE(TASK_IDENTITY_TOKEN)
57 std::optional<MachSendRight> webProcessIdentityToken;
58 decoder >> webProcessIdentityToken;
59 if (!webProcessIdentityToken)
60 return std::nullopt;
61#endif
62
63 std::optional<Vector<String>> overrideLanguages;
64 decoder >> overrideLanguages;
65 if (!overrideLanguages)
66 return std::nullopt;
67
68#if ENABLE(IPC_TESTING_API)
69 std::optional<bool> ignoreInvalidMessageForTesting;
70 decoder >> ignoreInvalidMessageForTesting;
71 if (!ignoreInvalidMessageForTesting)
72 return std::nullopt;
73#endif
74
75 return GPUProcessConnectionParameters {
76#if HAVE(TASK_IDENTITY_TOKEN)
77 WTFMove(*webProcessIdentityToken),
78#endif
79 WTFMove(*overrideLanguages),
80#if ENABLE(IPC_TESTING_API)
81 *ignoreInvalidMessageForTesting,
82#endif
83 };
84 }
85};
86
87} // namespace WebKit
88
89#endif // ENABLE(GPU_PROCESS)
