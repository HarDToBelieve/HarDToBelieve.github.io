30#include <WebCore/DisplayList.h>
31#include <WebCore/RenderingResourceIdentifier.h>
32
33namespace WebKit {
34
35enum class GPUProcessWakeupReason : uint8_t {
36 Unspecified,
37 ItemCountHysteresisExceeded
38};
39
40struct GPUProcessWakeupMessageArguments {
41 WebCore::DisplayList::ItemBufferIdentifier itemBufferIdentifier;
42 uint64_t offset { 0 };
43 WebCore::RenderingResourceIdentifier destinationImageBufferIdentifier;
44 GPUProcessWakeupReason reason { GPUProcessWakeupReason::Unspecified };
45
46 template<class Encoder> void encode(Encoder&) const;
47 template<class Decoder> static std::optional<GPUProcessWakeupMessageArguments> decode(Decoder&);
48};
49
50template<class Encoder>
51void GPUProcessWakeupMessageArguments::encode(Encoder& encoder) const
52{
53 encoder << itemBufferIdentifier;
54 encoder << offset;
55 encoder << destinationImageBufferIdentifier;
56 encoder << reason;
57}
58
59template<class Decoder>
60std::optional<GPUProcessWakeupMessageArguments> GPUProcessWakeupMessageArguments::decode(Decoder& decoder)
61{
62 std::optional<WebCore::DisplayList::ItemBufferIdentifier> itemBufferIdentifier;
63 decoder >> itemBufferIdentifier;
64 if (!itemBufferIdentifier)
65 return std::nullopt;
66
67 std::optional<uint64_t> offset;
68 decoder >> offset;
69 if (!offset)
70 return std::nullopt;
71
72 std::optional<WebCore::RenderingResourceIdentifier> destinationImageBufferIdentifier;
73 decoder >> destinationImageBufferIdentifier;
74 if (!destinationImageBufferIdentifier)
75 return std::nullopt;
76
77 std::optional<GPUProcessWakeupReason> reason;
78 decoder >> reason;
79 if (!reason)
80 return std::nullopt;
81
82 return {{ *itemBufferIdentifier, *offset, *destinationImageBufferIdentifier, *reason }};
83}
