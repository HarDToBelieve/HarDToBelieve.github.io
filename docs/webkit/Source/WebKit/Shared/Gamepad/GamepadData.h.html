41
42class GamepadData {
43public:
44 GamepadData()
45 : m_isNull(true)
46 {
47 }
48
49 GamepadData(unsigned index, const String& id, const String& mapping, const Vector<WebCore::SharedGamepadValue>& axisValues, const Vector<WebCore::SharedGamepadValue>& buttonValues, MonotonicTime lastUpdateTime);
50
51 void encode(IPC::Encoder&) const;
52 static std::optional<GamepadData> decode(IPC::Decoder&);
53
54 bool isNull() const { return m_isNull; }
55
56 MonotonicTime lastUpdateTime() const { return m_lastUpdateTime; }
57 unsigned index() const { return m_index; }
58 const String& id() const { return m_id; }
59 const String& mapping() const { return m_mapping; }
60 const Vector<double>& axisValues() const { return m_axisValues; }
61 const Vector<double>& buttonValues() const { return m_buttonValues; }
62
63private:
64 unsigned m_index;
65 String m_id;
66 String m_mapping;
67 Vector<double> m_axisValues;
68 Vector<double> m_buttonValues;
69 MonotonicTime m_lastUpdateTime;
70
71 bool m_isNull { false };
72
73#if !LOG_DISABLED
74 String loggingString() const;
75#endif
76};
