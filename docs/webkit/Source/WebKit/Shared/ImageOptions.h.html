26#ifndef ImageOptions_h
27#define ImageOptions_h
28
29namespace WebKit {
30
31enum ImageOptions {
32 ImageOptionsShareable = 1 << 0,
33};
34
35enum {
36 SnapshotOptionsShareable = 1 << 0,
37 SnapshotOptionsExcludeSelectionHighlighting = 1 << 1,
38 SnapshotOptionsInViewCoordinates = 1 << 2,
39 SnapshotOptionsPaintSelectionRectangle = 1 << 3,
40 SnapshotOptionsExcludeDeviceScaleFactor = 1 << 5,
41 SnapshotOptionsForceBlackText = 1 << 6,
42 SnapshotOptionsForceWhiteText = 1 << 7,
43 SnapshotOptionsPrinting = 1 << 8,
44 SnapshotOptionsUseScreenColorSpace = 1 << 9,
45};
46typedef uint32_t SnapshotOptions;
47
48inline ImageOptions snapshotOptionsToImageOptions(SnapshotOptions snapshotOptions)
49{
50 unsigned imageOptions = 0;
51
52 if (snapshotOptions & SnapshotOptionsShareable)
53 imageOptions |= ImageOptionsShareable;
54
55 return static_cast<ImageOptions>(imageOptions);
56}
57
58} // namespace WebKit
59
60#endif // ImageOptions_h
