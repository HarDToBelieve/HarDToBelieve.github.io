26#include "config.h"

29#include "ArgumentCoders.h"

33namespace WebKit {
34
35LayerTreeContext::LayerTreeContext()
36 : contextID(0)
37{
38}
39
40LayerTreeContext::~LayerTreeContext()
41{
42}
43
44void LayerTreeContext::encode(IPC::Encoder& encoder) const
45{
46 encoder << contextID;
47}
48
49bool LayerTreeContext::decode(IPC::Decoder& decoder, LayerTreeContext& result)
50{
51 if (!decoder.decode(result.contextID))
52 return false;
53
54 return true;
55}
56
57bool LayerTreeContext::isEmpty() const
58{
59 return !contextID;
60}
61
62bool operator==(const LayerTreeContext& a, const LayerTreeContext& b)
63{
64 return a.contextID == b.contextID;
65}
