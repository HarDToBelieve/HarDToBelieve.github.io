35namespace WebKit {
36
37enum class NavigatingToAppBoundDomain : bool { No, Yes };
38enum class LastNavigationWasAppBound : bool { No, Yes };
39
40using ContextDomain = WebCore::RegistrableDomain;
41using RequestDomain = WebCore::RegistrableDomain;
42
43#if PLATFORM(COCOA)
44struct AppBoundNavigationTestingData {
45
46 void setDidPerformSoftUpdate()
47 {
48 didPerformSoftUpdate = true;
49 }
50
51 void clearAppBoundNavigationDataTesting()
52 {
53 hasLoadedAppBoundRequestTesting = false;
54 hasLoadedNonAppBoundRequestTesting = false;
55 didPerformSoftUpdate = false;
56 contextData.clear();
57 }
58
59 void updateAppBoundNavigationTestingData(const WebCore::ResourceRequest& request, WebCore::RegistrableDomain&& contextDomain)
60 {
61 request.isAppBound() ? hasLoadedAppBoundRequestTesting = true : hasLoadedNonAppBoundRequestTesting = true;
62 contextData.add(WebCore::RegistrableDomain(request.url()), contextDomain);
63 }
64
65 void encode(IPC::Encoder& encoder) const
66 {
67 encoder << hasLoadedAppBoundRequestTesting;
68 encoder << hasLoadedNonAppBoundRequestTesting;
69 encoder << didPerformSoftUpdate;
70 encoder << contextData;
71 }
72
73 static std::optional<AppBoundNavigationTestingData> decode(IPC::Decoder& decoder)
74 {
75 std::optional<bool> hasLoadedAppBoundRequestTesting;
76 decoder >> hasLoadedAppBoundRequestTesting;
77 if (!hasLoadedAppBoundRequestTesting)
78 return std::nullopt;
79
80 std::optional<bool> hasLoadedNonAppBoundRequestTesting;
81 decoder >> hasLoadedNonAppBoundRequestTesting;
82 if (!hasLoadedNonAppBoundRequestTesting)
83 return std::nullopt;
84
85 std::optional<bool> didPerformSoftUpdate;
86 decoder >> didPerformSoftUpdate;
87 if (!didPerformSoftUpdate)
88 return std::nullopt;
89
90 std::optional<HashMap<RequestDomain, ContextDomain>> contextData;
91 decoder >> contextData;
92 if (!contextData)
93 return std::nullopt;
94
95 return {{ *hasLoadedAppBoundRequestTesting, *hasLoadedNonAppBoundRequestTesting, *didPerformSoftUpdate, WTFMove(*contextData) }};
96 }
97
98 bool hasLoadedAppBoundRequestTesting { false };
99 bool hasLoadedNonAppBoundRequestTesting { false };
100 bool didPerformSoftUpdate { false };
101 HashMap<RequestDomain, ContextDomain> contextData;
102};
103#endif
104
105}
