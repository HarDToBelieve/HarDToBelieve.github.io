26#pragma once
27
28#include "CallbackID.h"
29#include <wtf/HashTraits.h>
30#include <wtf/RunLoop.h>
31
32namespace WebKit {
33
34class OptionalCallbackID {
35public:
36 explicit OptionalCallbackID() { }
37
38 ALWAYS_INLINE explicit OptionalCallbackID(const CallbackID& otherID)
39 : m_id(otherID.m_id)
40 {
41 }
42
43 ALWAYS_INLINE OptionalCallbackID(const OptionalCallbackID& otherID)
44 : m_id(otherID.m_id)
45 {
46 ASSERT(!HashTraits<uint64_t>::isDeletedValue(m_id));
47 }
48
49 ALWAYS_INLINE OptionalCallbackID& operator=(const OptionalCallbackID& otherID)
50 {
51 m_id = otherID.m_id;
52 return *this;
53 }
54
55 uint64_t toInteger() { return m_id; }
56 CallbackID callbackID()
57 {
58 RELEASE_ASSERT(CallbackID::isValidCallbackID(m_id));
59 return CallbackID(m_id);
60 }
61
62 operator bool() { return m_id; }
63
64 ALWAYS_INLINE bool isValid() const { return isValidCallbackID(m_id); }
65 static ALWAYS_INLINE bool isValidCallbackID(uint64_t rawId)
66 {
67 return !HashTraits<uint64_t>::isDeletedValue(rawId);
68 }
69
70 template<class Encoder> void encode(Encoder& encoder) const
71 {
72 RELEASE_ASSERT(isValid());
73 encoder << m_id;
74 }
75
76 template<class Decoder> static WARN_UNUSED_RETURN bool decode(Decoder& decoder, OptionalCallbackID& callbackID)
77 {
78 auto result = decoder.decode(callbackID.m_id);
79 RELEASE_ASSERT(callbackID.isValid());
80 return result;
81 }
82
83private:
84 uint64_t m_id { HashTraits<uint64_t>::emptyValue() };
85};
