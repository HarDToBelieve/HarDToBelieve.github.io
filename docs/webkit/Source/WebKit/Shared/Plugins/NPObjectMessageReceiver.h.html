34namespace WebKit {
35
36class NPIdentifierData;
37class NPRemoteObjectMap;
38class NPVariantData;
39class Plugin;
40
41class NPObjectMessageReceiver {
42 WTF_MAKE_FAST_ALLOCATED;
43 WTF_MAKE_NONCOPYABLE(NPObjectMessageReceiver);
44
45public:
46 NPObjectMessageReceiver(NPRemoteObjectMap*, Plugin*, uint64_t npObjectID, NPObject*);
47 ~NPObjectMessageReceiver();
48
49 bool didReceiveSyncNPObjectMessageReceiverMessage(IPC::Connection&, IPC::Decoder&, UniqueRef<IPC::Encoder>&);
50
51 Plugin* plugin() const { return m_plugin; }
52 NPObject* npObject() const { return m_npObject; }
53
54private:
55 // Message handlers.
56 void deallocate(CompletionHandler<void()>&&);
57 void hasMethod(const NPIdentifierData&, CompletionHandler<void(bool)>&&);
58 void invoke(const NPIdentifierData&, const Vector<NPVariantData>& argumentsData, CompletionHandler<void(bool, NPVariantData&&)>&&);
59 void invokeDefault(const Vector<NPVariantData>& argumentsData, CompletionHandler<void(bool, NPVariantData&&)>&&);
60 void hasProperty(const NPIdentifierData&, CompletionHandler<void(bool)>&&);
61 void getProperty(const NPIdentifierData&, CompletionHandler<void(bool, NPVariantData&&)>&&);
62 void setProperty(const NPIdentifierData&, const NPVariantData& propertyValueData, CompletionHandler<void(bool)>&&);
63 void removeProperty(const NPIdentifierData&, CompletionHandler<void(bool)>&&);
64 void enumerate(CompletionHandler<void(bool, Vector<NPIdentifierData>&&)>&&);
65 void construct(const Vector<NPVariantData>& argumentsData, CompletionHandler<void(bool, NPVariantData&&)>&&);
66
67 NPRemoteObjectMap* m_npRemoteObjectMap;
68 Plugin* m_plugin;
69 uint64_t m_npObjectID;
70 NPObject* m_npObject;
71};
