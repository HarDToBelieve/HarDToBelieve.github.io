26#pragma once
27
28#if ENABLE(NETSCAPE_PLUGIN_API)
29
30#include <WebCore/npruntime_internal.h>
31#include <wtf/Noncopyable.h>
32
33namespace WebKit {
34
35class NPRemoteObjectMap;
36class Plugin;
37
38class NPObjectProxy : public NPObject {
39 WTF_MAKE_NONCOPYABLE(NPObjectProxy);
40
41public:
42 static NPObjectProxy* create(NPRemoteObjectMap*, Plugin*, uint64_t npObjectID);
43
44 static bool isNPObjectProxy(NPObject*);
45
46 static NPObjectProxy* toNPObjectProxy(NPObject* npObject)
47 {
48 RELEASE_ASSERT_WITH_SECURITY_IMPLICATION(isNPObjectProxy(npObject));
49 return static_cast<NPObjectProxy*>(npObject);
50 }
51
52 Plugin* plugin() const { return m_plugin; }
53 uint64_t npObjectID() const { return m_npObjectID; }
54
55 void invalidate();
56
57private:
58 NPObjectProxy();
59 ~NPObjectProxy();
60
61 void initialize(NPRemoteObjectMap*, Plugin*, uint64_t npObjectID);
62
63 bool hasMethod(NPIdentifier methodName);
64 bool invoke(NPIdentifier methodName, const NPVariant* arguments, uint32_t argumentCount, NPVariant* result);
65 bool invokeDefault(const NPVariant* arguments, uint32_t argumentCount, NPVariant* result);
66 bool hasProperty(NPIdentifier propertyName);
67 bool getProperty(NPIdentifier propertyName, NPVariant* result);
68 bool setProperty(NPIdentifier propertyName, const NPVariant* value);
69 bool removeProperty(NPIdentifier propertyName);
70 bool enumerate(NPIdentifier** identifiers, uint32_t* identifierCount);
71 bool construct(const NPVariant* arguments, uint32_t argumentCount, NPVariant* result);
72
73 static NPClass* npClass();
74 static NPObject* NP_Allocate(NPP, NPClass*);
75 static void NP_Deallocate(NPObject*);
76 static bool NP_HasMethod(NPObject*, NPIdentifier methodName);
77 static bool NP_Invoke(NPObject*, NPIdentifier methodName, const NPVariant* arguments, uint32_t argumentCount, NPVariant* result);
78 static bool NP_InvokeDefault(NPObject*, const NPVariant* arguments, uint32_t argumentCount, NPVariant* result);
79 static bool NP_HasProperty(NPObject*, NPIdentifier propertyName);
80 static bool NP_GetProperty(NPObject*, NPIdentifier propertyName, NPVariant* result);
81 static bool NP_SetProperty(NPObject*, NPIdentifier propertyName, const NPVariant* value);
82 static bool NP_RemoveProperty(NPObject*, NPIdentifier propertyName);
83 static bool NP_Enumerate(NPObject*, NPIdentifier** identifiers, uint32_t* identifierCount);
84 static bool NP_Construct(NPObject*, const NPVariant* arguments, uint32_t argumentCount, NPVariant* result);
85
86 NPRemoteObjectMap* m_npRemoteObjectMap;
87 Plugin* m_plugin;
88 uint64_t m_npObjectID;
89};
90
91} // namespace WebKit
92
93#endif // ENABLE(NETSCAPE_PLUGIN_API)
