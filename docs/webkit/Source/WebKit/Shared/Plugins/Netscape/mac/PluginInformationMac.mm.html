37namespace WebKit {
38using namespace WebCore;
39
40void getPlatformPluginModuleInformation(const PluginModuleInfo& plugin, API::Dictionary::MapType& map)
41{
42 map.set(pluginInformationBundleIdentifierKey(), API::String::create(plugin.bundleIdentifier));
43 map.set(pluginInformationBundleVersionKey(), API::String::create(plugin.versionString));
44 map.set(pluginInformationBundleShortVersionKey(), API::String::create(plugin.shortVersionString));
45 map.set(pluginInformationUpdatePastLastBlockedVersionIsKnownAvailableKey(), API::Boolean::create(WebCore::PluginBlocklist::isPluginUpdateAvailable(nsStringFromWebCoreString(plugin.bundleIdentifier))));
46 map.set(pluginInformationHasSandboxProfileKey(), API::Boolean::create(plugin.hasSandboxProfile));
47}
