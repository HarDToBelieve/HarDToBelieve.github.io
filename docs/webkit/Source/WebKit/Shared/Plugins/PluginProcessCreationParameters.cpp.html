35namespace WebKit {
36
37PluginProcessCreationParameters::PluginProcessCreationParameters()
38 : supportsAsynchronousPluginInitialization(false)
39{
40}
41
42void PluginProcessCreationParameters::encode(IPC::Encoder& encoder) const
43{
44 encoder << supportsAsynchronousPluginInitialization;
45 encoder << minimumLifetime;
46 encoder << terminationTimeout;
47#if PLATFORM(COCOA)
48 encoder << acceleratedCompositingPort;
49 encoder << networkATSContext;
50#endif
51}
52
53bool PluginProcessCreationParameters::decode(IPC::Decoder& decoder, PluginProcessCreationParameters& result)
54{
55 if (!decoder.decode(result.supportsAsynchronousPluginInitialization))
56 return false;
57 if (!decoder.decode(result.minimumLifetime))
58 return false;
59 if (!decoder.decode(result.terminationTimeout))
60 return false;
61#if PLATFORM(COCOA)
62 if (!decoder.decode(result.acceleratedCompositingPort))
63 return false;
64 if (!decoder.decode(result.networkATSContext))
65 return false;
66#endif
67
68 return true;
69}
