26#pragma once
27
28namespace WebKit {
29
30class PluginQuirks {
31public:
32 enum PluginQuirk {
33 // Mac specific quirks:
34#if PLUGIN_ARCHITECTURE(MAC)
35 // The plug-in wants the call to getprogname() to return "WebKitPluginHost".
36 // Adobe Flash Will not handle key down events otherwise.
37 PrognameShouldBeWebKitPluginHost,
38
39 // Supports receiving a paint event, even when using CoreAnimation rendering.
40 SupportsSnapshotting,
41
42 // Make the plug-in opaque unless it has a "background" attribute set to a transparent color
43 // according to http://msdn.microsoft.com/en-us/library/cc838148(VS.95).aspx
44 // A non-existent "background" attribute is interpreted as the named color White which is opaque.
45 // Microsoft Silverlight doesn't opt into transparency using NPN_SetValue and
46 // NPPVpluginTransparentBool, so we'll always force it unless the plug-in has a "background"
47 // attribute that specifies a opaque color.
48 MakeOpaqueUnlessTransparentSilverlightBackgroundAttributeExists,
49
50 // Whether calling NPP_GetValue with NPPVpluginCoreAnimationLayer returns a retained Core Animation
51 // layer or not. According to the NPAPI specifications, plug-in shouldn't return a retained layer but
52 // WebKit1 expects a retained plug-in layer. We use this for Flash to avoid leaking OpenGL layers.
53 ReturnsRetainedCoreAnimationLayer,
54
55 // Whether NPP_GetValue with NPPVpluginScriptableNPObject returns a non-retained NPObject or not.
56 // Versions of Silverlight prior to 4 never retained the returned NPObject.
57 ReturnsNonRetainedScriptableNPObject,
58
59 // Whether the plug-in wants parameter names to be lowercase.
60 // <rdar://problem/8440903>: AppleConnect has a bug where it does not
61 // understand the parameter names specified in the <object> element that
62 // embeds its plug-in.
63 WantsLowercaseParameterNames,
64
65 // Whether to append Version/3.2.1 to the user-agent passed to the plugin
66 // This is necessary to disable Silverlight's workaround for a Safari 2 leak
67 // which is enabled if it doesn't find Version/3 in the user-agent.
68 AppendVersion3UserAgent,
69
70 // Whether all thrown NSExceptions should be leaked.
71 // <rdar://problem/13003470> Adobe Flash has a bug where exceptions are released too early.
72 LeakAllThrownNSExceptions,
73
74 // X11 specific quirks:
75#elif PLATFORM(X11)
76 // Flash and npwrapper ask the browser about which GTK version does it use
77 // and refuse to load and work if it is not GTK 2 so we need to fake it in
78 // NPN_GetValue even when it is a lie.
79 RequiresGTKToolKit,
80
81 // Some version 10 releases of Flash run under nspluginwrapper will completely
82 // freeze when sending right click events to them in windowed mode.
83 IgnoreRightClickInWindowlessMode,
84
85 // Some ports don't support windowed plugins.
86 ForceFlashWindowlessMode,
87
88 // Flash crashes when NPP_GetValue is called for NPPVpluginCancelSrcStream in windowed mode.
89 DoNotCancelSrcStreamInWindowedMode,
90#endif
91
92 // This isn't really a quirk as much as the opposite of a quirk. By default, we don't send wheel events
93 // to plug-ins unless we know that they handle them correctly. Adobe Reader on Mac handles wheel events correctly.
94 WantsWheelEvents,
95
96 NumPluginQuirks
97 };
98
99 PluginQuirks()
100 : m_quirks(0)
101 {
102 COMPILE_ASSERT(sizeof(m_quirks) * 8 >= NumPluginQuirks, not_enough_room_for_quirks);
103 }
104
105 void add(PluginQuirk quirk)
106 {
107 ASSERT(quirk >= 0);
108 RELEASE_ASSERT_WITH_SECURITY_IMPLICATION(quirk < NumPluginQuirks);
109
110 m_quirks |= (1 << quirk);
111 }
112
113 bool contains(PluginQuirk quirk) const
114 {
115 return m_quirks & (1 << quirk);
116 }
117
118private:
119 uint32_t m_quirks;
120};
121
122} // namespace WebKit
