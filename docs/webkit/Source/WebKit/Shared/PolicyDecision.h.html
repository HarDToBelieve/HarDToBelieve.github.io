27
28#include "DownloadID.h"

32#include <wtf/Forward.h>

34namespace WebKit {
35
36struct PolicyDecision {
37 WebCore::PolicyCheckIdentifier identifier { };
38 std::optional<NavigatingToAppBoundDomain> isNavigatingToAppBoundDomain { std::nullopt };
39 WebCore::PolicyAction policyAction { WebCore::PolicyAction::Ignore };
40 uint64_t navigationID { 0 };
41 std::optional<DownloadID> downloadID { std::nullopt };
42 std::optional<WebsitePoliciesData> websitePoliciesData { std::nullopt };
43 std::optional<SandboxExtension::Handle> sandboxExtensionHandle { std::nullopt };
44
45 template<class Encoder>
46 void encode(Encoder& encoder) const
47 {
48 encoder << identifier;
49 encoder << isNavigatingToAppBoundDomain;
50 encoder << policyAction;
51 encoder << navigationID;
52 encoder << downloadID;
53 encoder << websitePoliciesData;
54 encoder << sandboxExtensionHandle;
55 }
56
57 template<class Decoder>
58 static std::optional<PolicyDecision> decode(Decoder& decoder)
59 {
60 std::optional<WebCore::PolicyCheckIdentifier> decodedIdentifier;
61 decoder >> decodedIdentifier;
62 if (!decodedIdentifier)
63 return std::nullopt;
64
65 std::optional<std::optional<NavigatingToAppBoundDomain>> decodedIsNavigatingToAppBoundDomain;
66 decoder >> decodedIsNavigatingToAppBoundDomain;
67 if (!decodedIsNavigatingToAppBoundDomain)
68 return std::nullopt;
69
70 std::optional<WebCore::PolicyAction> decodedPolicyAction;
71 decoder >> decodedPolicyAction;
72 if (!decodedPolicyAction)
73 return std::nullopt;
74
75 std::optional<uint64_t> decodedNavigationID;
76 decoder >> decodedNavigationID;
77 if (!decodedNavigationID)
78 return std::nullopt;
79
80 std::optional<std::optional<DownloadID>> decodedDownloadID;
81 decoder >> decodedDownloadID;
82 if (!decodedDownloadID)
83 return std::nullopt;
84
85 std::optional<std::optional<WebsitePoliciesData>> decodedWebsitePoliciesData;
86 decoder >> decodedWebsitePoliciesData;
87 if (!decodedWebsitePoliciesData)
88 return std::nullopt;
89
90 std::optional<std::optional<SandboxExtension::Handle>> sandboxExtensionHandle;
91 decoder >> sandboxExtensionHandle;
92 if (!sandboxExtensionHandle)
93 return std::nullopt;
94
95 return {{ WTFMove(*decodedIdentifier), WTFMove(*decodedIsNavigatingToAppBoundDomain), WTFMove(*decodedPolicyAction), WTFMove(*decodedNavigationID), WTFMove(*decodedDownloadID), WTFMove(*decodedWebsitePoliciesData), WTFMove(*sandboxExtensionHandle)}};
96 }
97};
