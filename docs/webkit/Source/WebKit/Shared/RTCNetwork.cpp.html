36RTCNetwork::RTCNetwork(const rtc::Network& network)
37 : name(network.name())
38 , description(network.description())
39 , prefix { network.prefix() }

51rtc::Network RTCNetwork::value() const
52{
53 rtc::Network network(name.data(), description.data(), prefix.value, prefixLength, rtc::AdapterType(type));
54 network.set_id(id);
55 network.set_preference(preference);
56 network.set_active(active);
57 network.set_ignored(ignored);
58 network.set_scope_id(scopeID);
59 network.SetIPs(ips, true);
60 return network;
61}

63auto RTCNetwork::IPAddress::decode(IPC::Decoder& decoder) -> std::optional<IPAddress>
64{
65 IPAddress result;
66 int family;
67 if (!decoder.decode(family))
68 return std::nullopt;
69
70 ASSERT(family == AF_INET || family == AF_INET6 || family == AF_UNSPEC);
71
72 if (family == AF_UNSPEC)
73 return result;
74
75 IPC::DataReference data;
76 if (!decoder.decode(data))
77 return std::nullopt;
78
79 if (family == AF_INET) {
80 if (data.size() != sizeof(in_addr))
81 return std::nullopt;
82 result.value = rtc::IPAddress(*reinterpret_cast<const in_addr*>(data.data()));
83 return result;
84 }
85
86 if (data.size() != sizeof(in6_addr))
87 return std::nullopt;
88 result.value = rtc::IPAddress(*reinterpret_cast<const in6_addr*>(data.data()));
89 return result;
90}

92void RTCNetwork::IPAddress::encode(IPC::Encoder& encoder) const
93{
94 auto family = value.family();
95 ASSERT(family == AF_INET || family == AF_INET6 || family == AF_UNSPEC);
96 encoder << family;
97
98 if (family == AF_UNSPEC)
99 return;
100
101 if (family == AF_INET) {
102 auto address = value.ipv4_address();
103 encoder << IPC::DataReference(reinterpret_cast<const uint8_t*>(&address), sizeof(address));
104 return;
105 }
106
107 auto address = value.ipv6_address();
108 encoder << IPC::DataReference(reinterpret_cast<const uint8_t*>(&address), sizeof(address));
109}

111rtc::SocketAddress RTCNetwork::isolatedCopy(const rtc::SocketAddress& value)
112{
113 rtc::SocketAddress copy;
114 copy.SetPort(value.port());
115 copy.SetScopeID(value.scope_id());
116 copy.SetIP(std::string(value.hostname().data(), value.hostname().size()));
117 if (!value.IsUnresolvedIP())
118 copy.SetResolvedIP(value.ipaddr());
119 return rtc::SocketAddress(copy);
120}

122auto RTCNetwork::SocketAddress::decode(IPC::Decoder& decoder) -> std::optional<SocketAddress>
123{
124 SocketAddress result;
125 uint16_t port;
126 if (!decoder.decode(port))
127 return std::nullopt;
128 int scopeId;
129 if (!decoder.decode(scopeId))
130 return std::nullopt;
131 result.value.SetPort(port);
132 result.value.SetScopeID(scopeId);
133
134 IPC::DataReference hostname;
135 if (!decoder.decode(hostname))
136 return std::nullopt;
137 result.value.SetIP(std::string(reinterpret_cast<const char*>(hostname.data()), hostname.size()));
138
139 bool isUnresolved;
140 if (!decoder.decode(isUnresolved))
141 return std::nullopt;
142 if (isUnresolved)
143 return result;
144
145 std::optional<IPAddress> ipAddress;
146 decoder >> ipAddress;
147 if (!ipAddress)
148 return std::nullopt;
149 result.value.SetResolvedIP(ipAddress->value);
150 return result;
151}

153void RTCNetwork::SocketAddress::encode(IPC::Encoder& encoder) const
154{
155 encoder << value.port();
156 encoder << value.scope_id();
157
158 auto hostname = value.hostname();
159 encoder << IPC::DataReference(reinterpret_cast<const uint8_t*>(hostname.data()), hostname.length());
160
161 if (value.IsUnresolvedIP()) {
162 encoder << true;
163 return;
164 }
165 encoder << false;
166 encoder << RTCNetwork::IPAddress(value.ipaddr());
167}

169std::optional<RTCNetwork> RTCNetwork::decode(IPC::Decoder& decoder)
170{
171 RTCNetwork result;
172 IPC::DataReference name, description;
173 if (!decoder.decode(name))
174 return std::nullopt;
175 result.name = std::string(reinterpret_cast<const char*>(name.data()), name.size());
176 if (!decoder.decode(description))
177 return std::nullopt;
178 result.description = std::string(reinterpret_cast<const char*>(description.data()), description.size());
179 std::optional<IPAddress> prefix;
180 decoder >> prefix;
181 if (!prefix)
182 return std::nullopt;
183 result.prefix = WTFMove(*prefix);
184 if (!decoder.decode(result.prefixLength))
185 return std::nullopt;
186 if (!decoder.decode(result.type))
187 return std::nullopt;
188 if (!decoder.decode(result.id))
189 return std::nullopt;
190 if (!decoder.decode(result.preference))
191 return std::nullopt;
192 if (!decoder.decode(result.active))
193 return std::nullopt;
194 if (!decoder.decode(result.ignored))
195 return std::nullopt;
196 if (!decoder.decode(result.scopeID))
197 return std::nullopt;
198
199 uint64_t length;
200 if (!decoder.decode(length))
201 return std::nullopt;
202 result.ips.reserve(length);
203 for (size_t index = 0; index < length; ++index) {
204 std::optional<IPAddress> address;
205 decoder >> address;
206 if (!address)
207 return std::nullopt;
208 int flags;
209 if (!decoder.decode(flags))
210 return std::nullopt;
211 result.ips.push_back({ address->value, flags });
212 }
213 return result;
214}

216void RTCNetwork::encode(IPC::Encoder& encoder) const
217{
218 encoder << IPC::DataReference(reinterpret_cast<const uint8_t*>(name.data()), name.length());
219 encoder << IPC::DataReference(reinterpret_cast<const uint8_t*>(description.data()), description.length());
220 encoder << prefix;
221 encoder << prefixLength;
222 encoder << type;
223
224 encoder << id;
225 encoder << preference;
226 encoder << active;
227 encoder << ignored;
228 encoder << scopeID;
229
230 encoder << (uint64_t)ips.size();
231 for (auto& ip : ips) {
232 encoder << IPAddress { ip };
233 encoder << ip.ipv6_flags();
234 }
235}
