56RemoteLayerBackingStore::RemoteLayerBackingStore(PlatformCALayerRemote* layer)
57 : m_layer(layer)
58 , m_isOpaque(false)
59 , m_lastDisplayTime(-MonotonicTime::infinity())
60{
61 if (!m_layer)
62 return;
63 if (RemoteLayerTreeContext* context = m_layer->context())
64 context->backingStoreWasCreated(*this);
65}
66
67RemoteLayerBackingStore::~RemoteLayerBackingStore()
68{
69 clearBackingStore();
70
71 if (!m_layer)
72 return;
73
74 if (RemoteLayerTreeContext* context = m_layer->context())
75 context->backingStoreWillBeDestroyed(*this);
76}

78void RemoteLayerBackingStore::ensureBackingStore(Type type, WebCore::FloatSize size, float scale, bool deepColor, bool isOpaque)
79{
80 if (m_type == type && m_size == size && m_scale == scale && m_deepColor == deepColor && m_isOpaque == isOpaque)
81 return;
82
83 m_type = type;
84 m_size = size;
85 m_scale = scale;
86 m_deepColor = deepColor;
87 m_isOpaque = isOpaque;
88
89 if (m_frontBuffer) {
90 // If we have a valid backing store, we need to ensure that it gets completely
91 // repainted the next time display() is called.
92 setNeedsDisplay();
93 }
94
95 clearBackingStore();
96}
97
98void RemoteLayerBackingStore::clearBackingStore()
99{
100 m_frontBuffer.discard();
101 m_backBuffer.discard();
102 m_secondaryBackBuffer.discard();
103}

105void RemoteLayerBackingStore::encode(IPC::Encoder& encoder) const
106{
107 encoder << m_type;
108 encoder << m_size;
109 encoder << m_scale;
110 encoder << m_isOpaque;
111
112 std::optional<ImageBufferBackendHandle> handle;
113 if (m_frontBuffer.imageBuffer) {
114 switch (m_type) {
115 case Type::IOSurface:
116 if (m_frontBuffer.imageBuffer->canMapBackingStore())
117 handle = static_cast<AcceleratedImageBufferShareableMappedBackend&>(*m_frontBuffer.imageBuffer->ensureBackendCreated()).createImageBufferBackendHandle();
118 else
119 handle = static_cast<AcceleratedImageBufferShareableBackend&>(*m_frontBuffer.imageBuffer->ensureBackendCreated()).createImageBufferBackendHandle();
120 break;
121 case Type::Bitmap:
122 handle = static_cast<UnacceleratedImageBufferShareableBackend&>(*m_frontBuffer.imageBuffer->ensureBackendCreated()).createImageBufferBackendHandle();
123 break;
124#if ENABLE(CG_DISPLAY_LIST_BACKED_IMAGE_BUFFER)
125 case Type::CGDisplayList:
126 handle = static_cast<CGDisplayListImageBufferBackend&>(*m_frontBuffer.imageBuffer->ensureBackendCreated()).createImageBufferBackendHandle();
127 break;
128#endif
129 }
130 }
131
132 encoder << handle;
133}

135bool RemoteLayerBackingStore::decode(IPC::Decoder& decoder, RemoteLayerBackingStore& result)
136{
137 if (!decoder.decode(result.m_type))
138 return false;
139
140 if (!decoder.decode(result.m_size))
141 return false;
142
143 if (!decoder.decode(result.m_scale))
144 return false;
145
146 if (!decoder.decode(result.m_isOpaque))
147 return false;
148
149 if (!decoder.decode(result.m_bufferHandle))
150 return false;
151
152 return true;
153}

186void RemoteLayerBackingStore::swapToValidFrontBuffer()
187{
188 // Sometimes, we can get two swaps ahead of the render server.
189 // If we're using shared IOSurfaces, we must wait to modify
190 // a surface until it no longer has outstanding clients.
191 if (m_type == Type::IOSurface) {
192 if (!m_backBuffer.imageBuffer || m_backBuffer.imageBuffer->isInUse()) {
193 std::swap(m_backBuffer, m_secondaryBackBuffer);
194
195 // When pulling the secondary back buffer out of hibernation (to become
196 // the new front buffer), if it is somehow still in use (e.g. we got
197 // three swaps ahead of the render server), just give up and discard it.
198 if (m_backBuffer.imageBuffer && m_backBuffer.imageBuffer->isInUse())
199 m_backBuffer.discard();
200 }
201 }
202
203 std::swap(m_frontBuffer, m_backBuffer);
204
205 if (m_frontBuffer.imageBuffer) {
206 setBufferVolatility(BufferType::Front, false);
207 return;
208 }
209
210 bool shouldUseRemoteRendering = WebProcess::singleton().shouldUseRemoteRenderingFor(WebCore::RenderingPurpose::DOM);
211
212 switch (m_type) {
213 case Type::IOSurface:
214 if (shouldUseRemoteRendering)
215 m_frontBuffer.imageBuffer = m_layer->context()->ensureRemoteRenderingBackendProxy().createImageBuffer(m_size, WebCore::RenderingMode::Accelerated, m_scale, WebCore::DestinationColorSpace::SRGB(), pixelFormat());
216 else
217 m_frontBuffer.imageBuffer = WebCore::ConcreteImageBuffer<AcceleratedImageBufferShareableMappedBackend>::create(m_size, m_scale, WebCore::DestinationColorSpace::SRGB(), pixelFormat(), nullptr);
218 break;
219 case Type::Bitmap:
220 if (shouldUseRemoteRendering)
221 m_frontBuffer.imageBuffer = m_layer->context()->ensureRemoteRenderingBackendProxy().createImageBuffer(m_size, WebCore::RenderingMode::Unaccelerated, m_scale, WebCore::DestinationColorSpace::SRGB(), pixelFormat());
222 else
223 m_frontBuffer.imageBuffer = WebCore::ConcreteImageBuffer<UnacceleratedImageBufferShareableBackend>::create(m_size, m_scale, WebCore::DestinationColorSpace::SRGB(), pixelFormat(), nullptr);
224 break;
225#if ENABLE(CG_DISPLAY_LIST_BACKED_IMAGE_BUFFER)
226 case Type::CGDisplayList:
227 m_frontBuffer.imageBuffer = WebCore::ConcreteImageBuffer<CGDisplayListImageBufferBackend>::create(m_size, m_scale, WebCore::DestinationColorSpace::SRGB(), pixelFormat(), nullptr);
228 break;
229#endif
230 }
231}

233bool RemoteLayerBackingStore::supportsPartialRepaint()
234{
235 switch (m_type) {
236 case Type::IOSurface:
237 case Type::Bitmap:

239#if ENABLE(CG_DISPLAY_LIST_BACKED_IMAGE_BUFFER)

242#endif

244
245 return true;

248bool RemoteLayerBackingStore::display()
249{
250 ASSERT(!m_frontBufferFlusher);
251
252 m_lastDisplayTime = MonotonicTime::now();
253
254 bool needToEncodeBackingStore = false;
255 if (RemoteLayerTreeContext* context = m_layer->context())
256 needToEncodeBackingStore = context->backingStoreWillBeDisplayed(*this);
257
258 // Make the previous front buffer non-volatile early, so that we can dirty the whole layer if it comes back empty.
259 setBufferVolatility(BufferType::Front, false);
260
261 if (m_dirtyRegion.isEmpty() || m_size.isEmpty())
262 return needToEncodeBackingStore;
263
264 WebCore::IntRect layerBounds(WebCore::IntPoint(), WebCore::expandedIntSize(m_size));
265 if (!hasFrontBuffer() || !supportsPartialRepaint())
266 m_dirtyRegion.unite(layerBounds);
267
268 if (m_layer->owner()->platformCALayerShowRepaintCounter(m_layer)) {
269 WebCore::IntRect indicatorRect(0, 0, 52, 27);
270 m_dirtyRegion.unite(indicatorRect);
271 }
272
273 swapToValidFrontBuffer();
274 if (!m_frontBuffer.imageBuffer)
275 return true;
276
277 WebCore::GraphicsContext& context = m_frontBuffer.imageBuffer->context();
278 WebCore::GraphicsContextStateSaver stateSaver(context);
279
280 // If we have less than webLayerMaxRectsToPaint rects to paint and they cover less
281 // than webLayerWastedSpaceThreshold of the total dirty area, we'll repaint each rect separately.
282 // Otherwise, repaint the entire bounding box of the dirty region.
283 WebCore::IntRect dirtyBounds = m_dirtyRegion.bounds();
284
285 auto dirtyRects = m_dirtyRegion.rects();
286 if (dirtyRects.size() > WebCore::PlatformCALayer::webLayerMaxRectsToPaint || m_dirtyRegion.totalArea() > WebCore::PlatformCALayer::webLayerWastedSpaceThreshold * dirtyBounds.width() * dirtyBounds.height()) {
287 dirtyRects.clear();
288 dirtyRects.append(dirtyBounds);
289 }
290
291 // FIXME: find a consistent way to scale and snap dirty and CG clip rects.
292 for (const auto& rect : dirtyRects) {
293 WebCore::FloatRect scaledRect(rect);
294 scaledRect.scale(m_scale);
295 scaledRect = WebCore::enclosingIntRect(scaledRect);
296 scaledRect.scale(1 / m_scale);
297 m_paintingRects.append(scaledRect);
298 }
299
300 if (!m_dirtyRegion.contains(layerBounds)) {
301 ASSERT(m_backBuffer.imageBuffer);
302 context.drawImageBuffer(*m_backBuffer.imageBuffer, { {0, 0}, m_size }, { {0, 0}, m_size }, { WebCore::CompositeOperator::Copy });
303 }
304
305 if (m_paintingRects.size() == 1)
306 context.clip(m_paintingRects[0]);
307 else {
308 WebCore::Path clipPath;
309 for (auto rect : m_paintingRects)
310 clipPath.addRect(rect);
311 context.clipPath(clipPath);
312 }
313
314 if (!m_isOpaque)
315 context.clearRect(layerBounds);
316
317#ifndef NDEBUG
318 if (m_isOpaque)
319 context.fillRect(layerBounds, WebCore::SRGBA<uint8_t> { 255, 47, 146 });
320#endif
321
322 // FIXME: Clarify that WebCore::GraphicsLayerPaintSnapshotting is just about image decoding.
323 auto flags = m_layer->context() && m_layer->context()->nextRenderingUpdateRequiresSynchronousImageDecoding() ? WebCore::GraphicsLayerPaintSnapshotting : WebCore::GraphicsLayerPaintNormal;
324
325 // FIXME: This should be moved to PlatformCALayerRemote for better layering.
326 switch (m_layer->layerType()) {
327 case WebCore::PlatformCALayer::LayerTypeSimpleLayer:
328 case WebCore::PlatformCALayer::LayerTypeTiledBackingTileLayer:
329 m_layer->owner()->platformCALayerPaintContents(m_layer, context, dirtyBounds, flags);
330 break;
331 case WebCore::PlatformCALayer::LayerTypeWebLayer:
332 case WebCore::PlatformCALayer::LayerTypeBackdropLayer:
333 WebCore::PlatformCALayer::drawLayerContents(context, m_layer, m_paintingRects, flags);
334 break;
335 case WebCore::PlatformCALayer::LayerTypeDarkSystemBackdropLayer:
336 case WebCore::PlatformCALayer::LayerTypeLightSystemBackdropLayer:
337 // FIXME: These have a more complicated layer hierarchy. We need to paint into
338 // a child layer in order to see the rendered results.
339 WebCore::PlatformCALayer::drawLayerContents(context, m_layer, m_paintingRects, flags);
340 break;
341 case WebCore::PlatformCALayer::LayerTypeLayer:
342 case WebCore::PlatformCALayer::LayerTypeTransformLayer:
343 case WebCore::PlatformCALayer::LayerTypeTiledBackingLayer:
344 case WebCore::PlatformCALayer::LayerTypePageTiledBackingLayer:
345 case WebCore::PlatformCALayer::LayerTypeRootLayer:
346 case WebCore::PlatformCALayer::LayerTypeAVPlayerLayer:
347 case WebCore::PlatformCALayer::LayerTypeContentsProvidedLayer:
348 case WebCore::PlatformCALayer::LayerTypeShapeLayer:
349 case WebCore::PlatformCALayer::LayerTypeScrollContainerLayer:
350#if ENABLE(MODEL_ELEMENT)
351 case WebCore::PlatformCALayer::LayerTypeModelLayer:
352#endif
353 case WebCore::PlatformCALayer::LayerTypeCustom:
354 ASSERT_NOT_REACHED();
355 break;
356 };
357
358 m_dirtyRegion = WebCore::Region();
359 m_paintingRects.clear();
360
361 m_layer->owner()->platformCALayerLayerDidDisplay(m_layer);
362
363 m_frontBuffer.imageBuffer->flushDrawingContextAsync();
364
365 m_frontBufferFlusher = m_frontBuffer.imageBuffer->createFlusher();
366
367 return true;
368}

370void RemoteLayerBackingStore::enumerateRectsBeingDrawn(WebCore::GraphicsContext& context, void (^block)(WebCore::FloatRect))
371{
372 CGAffineTransform inverseTransform = CGAffineTransformInvert(context.getCTM());
373
374 // We don't want to un-apply the flipping or contentsScale,
375 // because they're not applied to repaint rects.
376 inverseTransform = CGAffineTransformScale(inverseTransform, m_scale, -m_scale);
377 inverseTransform = CGAffineTransformTranslate(inverseTransform, 0, -m_size.height());
378
379 for (const auto& rect : m_paintingRects) {
380 CGRect rectToDraw = CGRectApplyAffineTransform(rect, inverseTransform);
381 block(rectToDraw);
382 }
383}

385void RemoteLayerBackingStore::applyBackingStoreToLayer(CALayer *layer, LayerContentsType contentsType)
386{
387 ASSERT(m_bufferHandle);
388 layer.contentsOpaque = m_isOpaque;
389
390 WTF::switchOn(*m_bufferHandle,
391 [&] (ShareableBitmap::Handle& handle) {
392 ASSERT(m_type == Type::Bitmap);
393 auto bitmap = ShareableBitmap::create(handle);
394 layer.contents = (__bridge id)bitmap->makeCGImageCopy().get();
395 },
396 [&] (MachSendRight& machSendRight) {
397 ASSERT(m_type == Type::IOSurface);
398 switch (contentsType) {
399 case LayerContentsType::IOSurface: {
400 auto surface = WebCore::IOSurface::createFromSendRight(WTFMove(machSendRight), WebCore::DestinationColorSpace::SRGB());
401 layer.contents = surface ? surface->asLayerContents() : nil;
402 break;
403 }
404 case LayerContentsType::CAMachPort:
405 layer.contents = (__bridge id)adoptCF(CAMachPortCreate(machSendRight.leakSendRight())).get();
406 break;
407 }
408 }
409#if ENABLE(CG_DISPLAY_LIST_BACKED_IMAGE_BUFFER)
410 , [&] (IPC::SharedBufferCopy& buffer) {
411 ASSERT(m_type == Type::CGDisplayList);
412 ASSERT([layer isKindOfClass:[WKCompositingLayer class]]);
413 if (![layer isKindOfClass:[WKCompositingLayer class]])
414 return;
415 [layer setValue:@1 forKeyPath:WKCGDisplayListEnabledKey];
416 auto data = buffer.buffer()->createCFData();
417 [(WKCompositingLayer *)layer _setWKContentsDisplayList:data.get()];
418 }
419#endif
420 );
421}

428bool RemoteLayerBackingStore::setBufferVolatility(BufferType bufferType, bool isVolatile)
429{
430 if (m_type != Type::IOSurface)
431 return true;
432
433 // Return value is true if we succeeded in making volatile.
434 auto makeVolatile = [] (Buffer& buffer) -> bool {
435 if (!buffer.imageBuffer || buffer.isVolatile)
436 return true;
437
438 buffer.imageBuffer->releaseGraphicsContext();
439
440 if (!buffer.imageBuffer->isInUse()) {
441 buffer.imageBuffer->setVolatile(true);
442 buffer.isVolatile = true;
443 return true;
444 }
445
446 return false;
447 };
448
449 // Return value is true if we need to repaint.
450 auto makeNonVolatile = [] (Buffer& buffer) -> bool {
451 if (!buffer.imageBuffer || !buffer.isVolatile)
452 return false;
453
454 auto previousState = buffer.imageBuffer->setVolatile(false);
455 buffer.isVolatile = false;
456
457 return previousState == WebCore::VolatilityState::Empty;
458 };
459
460 switch (bufferType) {
461 case BufferType::Front:
462 if (isVolatile)
463 return makeVolatile(m_frontBuffer);
464
465 // Becoming non-volatile and the front buffer was purged, so we need to repaint.
466 if (makeNonVolatile(m_frontBuffer))
467 setNeedsDisplay();
468 break;
469 case BufferType::Back:
470 if (isVolatile)
471 return makeVolatile(m_backBuffer);
472
473 makeNonVolatile(m_backBuffer);
474 break;
475 case BufferType::SecondaryBack:
476 if (isVolatile)
477 return makeVolatile(m_secondaryBackBuffer);
478
479 makeNonVolatile(m_secondaryBackBuffer);
480 break;
481 }
482 return true;
483}

485void RemoteLayerBackingStore::Buffer::discard()
486{
487 isVolatile = false;
488 if (imageBuffer)
489 imageBuffer->releaseBufferToPool();
490 imageBuffer = nullptr;
491}
