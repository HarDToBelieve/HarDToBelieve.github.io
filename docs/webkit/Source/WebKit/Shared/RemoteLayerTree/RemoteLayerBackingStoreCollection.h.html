32namespace WebKit {
33
34class RemoteLayerBackingStore;
35class RemoteLayerTreeContext;
36class RemoteLayerTreeTransaction;
37
38class RemoteLayerBackingStoreCollection {
39 WTF_MAKE_NONCOPYABLE(RemoteLayerBackingStoreCollection);
40 WTF_MAKE_FAST_ALLOCATED;
41public:
42 RemoteLayerBackingStoreCollection();
43
44 void backingStoreWasCreated(RemoteLayerBackingStore&);
45 void backingStoreWillBeDestroyed(RemoteLayerBackingStore&);
46
47 // Return value indicates whether the backing store needs to be included in the transaction.
48 bool backingStoreWillBeDisplayed(RemoteLayerBackingStore&);
49 void backingStoreBecameUnreachable(RemoteLayerBackingStore&);
50
51 void willFlushLayers();
52 void willCommitLayerTree(RemoteLayerTreeTransaction&);
53 void didFlushLayers();
54
55 void volatilityTimerFired();
56 bool markAllBackingStoreVolatileImmediatelyIfPossible();
57
58 void scheduleVolatilityTimer();
59
60private:
61 enum VolatilityMarkingFlag {
62 MarkBuffersIgnoringReachability = 1 << 0
63 };
64 typedef unsigned VolatilityMarkingFlags;
65 bool markBackingStoreVolatileImmediately(RemoteLayerBackingStore&, VolatilityMarkingFlags = 0);
66 bool markBackingStoreVolatile(RemoteLayerBackingStore&, MonotonicTime now);
67
68 HashSet<RemoteLayerBackingStore*> m_liveBackingStore;
69 HashSet<RemoteLayerBackingStore*> m_unparentedBackingStore;
70
71 // Only used during a single flush.
72 HashSet<RemoteLayerBackingStore*> m_reachableBackingStoreInLatestFlush;
73
74 WebCore::Timer m_volatilityTimer;
75
76 bool m_inLayerFlush { false };
77};
