54@interface UIView (WKUIViewUtilities)
55- (void)_web_setSubviews:(NSArray *)subviews;
56@end
57
58@implementation UIView (WKUIViewUtilities)
59
60- (void)_web_setSubviews:(NSArray *)newSubviews
61{
62 NSUInteger numOldSubviews = self.subviews.count;
63 NSUInteger numNewSubviews = newSubviews.count;
64
65 NSUInteger currIndex = 0;
66 for (currIndex = 0; currIndex < numNewSubviews; ++currIndex) {
67 UIView *currNewSubview = [newSubviews objectAtIndex:currIndex];
68
69 if (currIndex < numOldSubviews) {
70 UIView *existingSubview = [self.subviews objectAtIndex:currIndex];
71 if (existingSubview == currNewSubview)
72 continue;
73 }
74
75 // New or moved subview.
76 [self insertSubview:currNewSubview atIndex:currIndex];
77 }
78
79 // Remove views at the end.
80 RetainPtr<NSMutableArray> viewsToRemove;
81 auto appendViewToRemove = [&viewsToRemove](UIView *view) {
82 if (!viewsToRemove)
83 viewsToRemove = adoptNS([[NSMutableArray alloc] init]);
84
85 [viewsToRemove addObject:view];
86 };
87
88 NSUInteger remainingSubviews = self.subviews.count;
89 for (NSUInteger i = currIndex; i < remainingSubviews; ++i) {
90 UIView *subview = [self.subviews objectAtIndex:i];
91 if ([subview conformsToProtocol:@protocol(WKContentControlled)])
92 appendViewToRemove(subview);
93 }
94
95 if (viewsToRemove)
96 [viewsToRemove makeObjectsPerformSelector:@selector(removeFromSuperview)];
97}
98
99@end
100#endif

105static CGColorRef cgColorFromColor(const Color& color)
106{
107 if (!color.isValid())
108 return nil;
109
110 return cachedCGColor(color);
111}

113static NSString *toCAFilterType(PlatformCALayer::FilterType type)
114{
115 switch (type) {
116 case PlatformCALayer::Linear:
117 return kCAFilterLinear;
118 case PlatformCALayer::Nearest:
119 return kCAFilterNearest;
120 case PlatformCALayer::Trilinear:
121 return kCAFilterTrilinear;
122 };
123
124 ASSERT_NOT_REACHED();
125 return 0;
126}

128static void updateCustomAppearance(CALayer *layer, GraphicsLayer::CustomAppearance customAppearance)
129{
130#if ENABLE(RUBBER_BANDING)
131 switch (customAppearance) {
132 case GraphicsLayer::CustomAppearance::None:
133 case GraphicsLayer::CustomAppearance::DarkBackdrop:
134 case GraphicsLayer::CustomAppearance::LightBackdrop:
135 ScrollbarThemeMac::removeOverhangAreaBackground(layer);
136 ScrollbarThemeMac::removeOverhangAreaShadow(layer);
137 break;
138 case GraphicsLayer::CustomAppearance::ScrollingOverhang:
139 ScrollbarThemeMac::setUpOverhangAreaBackground(layer);
140 break;
141 case GraphicsLayer::CustomAppearance::ScrollingShadow:
142 ScrollbarThemeMac::setUpOverhangAreaShadow(layer);
143 break;
144 }
145#else
146 UNUSED_PARAM(customAppearance);
147#endif
148}

150void RemoteLayerTreePropertyApplier::applyPropertiesToLayer(CALayer *layer, RemoteLayerTreeHost* layerTreeHost, const RemoteLayerTreeTransaction::LayerProperties& properties, RemoteLayerBackingStore::LayerContentsType layerContentsType)
151{
152 if (properties.changedProperties & RemoteLayerTreeTransaction::NameChanged)
153 layer.name = properties.name;
154
155 if (properties.changedProperties & RemoteLayerTreeTransaction::PositionChanged) {
156 layer.position = CGPointMake(properties.position.x(), properties.position.y());
157 layer.zPosition = properties.position.z();
158 }
159
160 if (properties.changedProperties & RemoteLayerTreeTransaction::AnchorPointChanged) {
161 layer.anchorPoint = CGPointMake(properties.anchorPoint.x(), properties.anchorPoint.y());
162 layer.anchorPointZ = properties.anchorPoint.z();
163 }
164
165 if (properties.changedProperties & RemoteLayerTreeTransaction::BoundsChanged)
166 layer.bounds = properties.bounds;
167
168 if (properties.changedProperties & RemoteLayerTreeTransaction::BackgroundColorChanged)
169 layer.backgroundColor = cgColorFromColor(properties.backgroundColor);
170
171 if (properties.changedProperties & RemoteLayerTreeTransaction::BorderColorChanged)
172 layer.borderColor = cgColorFromColor(properties.borderColor);
173
174 if (properties.changedProperties & RemoteLayerTreeTransaction::BorderWidthChanged)
175 layer.borderWidth = properties.borderWidth;
176
177 if (properties.changedProperties & RemoteLayerTreeTransaction::OpacityChanged)
178 layer.opacity = properties.opacity;
179
180 if (properties.changedProperties & RemoteLayerTreeTransaction::TransformChanged)
181 layer.transform = properties.transform ? (CATransform3D)*properties.transform.get() : CATransform3DIdentity;
182
183 if (properties.changedProperties & RemoteLayerTreeTransaction::SublayerTransformChanged)
184 layer.sublayerTransform = properties.sublayerTransform ? (CATransform3D)*properties.sublayerTransform.get() : CATransform3DIdentity;
185
186 if (properties.changedProperties & RemoteLayerTreeTransaction::HiddenChanged)
187 layer.hidden = properties.hidden;
188
189 if (properties.changedProperties & RemoteLayerTreeTransaction::GeometryFlippedChanged)
190 layer.geometryFlipped = properties.geometryFlipped;
191
192 if (properties.changedProperties & RemoteLayerTreeTransaction::DoubleSidedChanged)
193 layer.doubleSided = properties.doubleSided;
194
195 if (properties.changedProperties & RemoteLayerTreeTransaction::MasksToBoundsChanged)
196 layer.masksToBounds = properties.masksToBounds;
197
198 if (properties.changedProperties & RemoteLayerTreeTransaction::OpaqueChanged)
199 layer.opaque = properties.opaque;
200
201 if (properties.changedProperties & RemoteLayerTreeTransaction::ContentsRectChanged)
202 layer.contentsRect = properties.contentsRect;
203
204 if (properties.changedProperties & RemoteLayerTreeTransaction::ContentsScaleChanged) {
205 layer.contentsScale = properties.contentsScale;
206 layer.rasterizationScale = properties.contentsScale;
207 }
208
209 if (properties.changedProperties & RemoteLayerTreeTransaction::CornerRadiusChanged)
210 layer.cornerRadius = properties.cornerRadius;
211
212 if (properties.changedProperties & RemoteLayerTreeTransaction::ShapeRoundedRectChanged) {
213 Path path;
214 if (properties.shapeRoundedRect)
215 path.addRoundedRect(*properties.shapeRoundedRect);
216 ASSERT([layer isKindOfClass:[CAShapeLayer class]]);
217 [(CAShapeLayer *)layer setPath:path.platformPath()];
218 }
219
220 if (properties.changedProperties & RemoteLayerTreeTransaction::ShapePathChanged) {
221 ASSERT([layer isKindOfClass:[CAShapeLayer class]]);
222 [(CAShapeLayer *)layer setPath:properties.shapePath.platformPath()];
223 }
224
225 if (properties.changedProperties & RemoteLayerTreeTransaction::MinificationFilterChanged)
226 layer.minificationFilter = toCAFilterType(properties.minificationFilter);
227
228 if (properties.changedProperties & RemoteLayerTreeTransaction::MagnificationFilterChanged)
229 layer.magnificationFilter = toCAFilterType(properties.magnificationFilter);
230
231 if (properties.changedProperties & RemoteLayerTreeTransaction::BlendModeChanged)
232 PlatformCAFilters::setBlendingFiltersOnLayer(layer, properties.blendMode);
233
234 if (properties.changedProperties & RemoteLayerTreeTransaction::WindRuleChanged) {
235 ASSERT([layer isKindOfClass:[CAShapeLayer class]]);
236 CAShapeLayer *shapeLayer = (CAShapeLayer *)layer;
237 switch (properties.windRule) {
238 case WindRule::NonZero:
239 shapeLayer.fillRule = kCAFillRuleNonZero;
240 break;
241 case WindRule::EvenOdd:
242 shapeLayer.fillRule = kCAFillRuleEvenOdd;
243 break;
244 }
245 }
246
247 if (properties.changedProperties & RemoteLayerTreeTransaction::SpeedChanged)
248 layer.speed = properties.speed;
249
250 if (properties.changedProperties & RemoteLayerTreeTransaction::TimeOffsetChanged)
251 layer.timeOffset = properties.timeOffset;
252
253 if (properties.changedProperties & RemoteLayerTreeTransaction::BackingStoreChanged
254 || properties.changedProperties & RemoteLayerTreeTransaction::BackingStoreAttachmentChanged)
255 {
256 RemoteLayerBackingStore* backingStore = properties.backingStore.get();
257 if (backingStore && properties.backingStoreAttached)
258 backingStore->applyBackingStoreToLayer(layer, layerContentsType);
259 else {
260 layer.contents = nil;
261 layer.contentsOpaque = NO;
262 }
263 }
264
265 if (properties.changedProperties & RemoteLayerTreeTransaction::FiltersChanged)
266 PlatformCAFilters::setFiltersOnLayer(layer, properties.filters ? *properties.filters : FilterOperations());
267
268 if (properties.changedProperties & RemoteLayerTreeTransaction::AnimationsChanged)
269 PlatformCAAnimationRemote::updateLayerAnimations(layer, layerTreeHost, properties.addedAnimations, properties.keyPathsOfAnimationsToRemove);
270
271 if (properties.changedProperties & RemoteLayerTreeTransaction::EdgeAntialiasingMaskChanged)
272 layer.edgeAntialiasingMask = properties.edgeAntialiasingMask;
273
274 if (properties.changedProperties & RemoteLayerTreeTransaction::CustomAppearanceChanged)
275 updateCustomAppearance(layer, properties.customAppearance);
276
277#if HAVE(CORE_ANIMATION_SEPARATED_LAYERS)
278 if (properties.changedProperties & RemoteLayerTreeTransaction::SeparatedChanged) {
279 layer.separated = properties.isSeparated;
280 if (properties.isSeparated)
281 configureSeparatedLayer(layer);
282 }
283
284#if HAVE(CORE_ANIMATION_SEPARATED_PORTALS)
285 if (properties.changedProperties & RemoteLayerTreeTransaction::SeparatedPortalChanged) {
286 // FIXME: Implement SeparatedPortalChanged.
287 }
288
289 if (properties.changedProperties & RemoteLayerTreeTransaction::DescendentOfSeparatedPortalChanged) {
290 // FIXME: Implement DescendentOfSeparatedPortalChanged.
291 }
292#endif
293#endif
294}

296void RemoteLayerTreePropertyApplier::applyProperties(RemoteLayerTreeNode& node, RemoteLayerTreeHost* layerTreeHost, const RemoteLayerTreeTransaction::LayerProperties& properties, const RelatedLayerMap& relatedLayers, RemoteLayerBackingStore::LayerContentsType layerContentsType)
297{
298 BEGIN_BLOCK_OBJC_EXCEPTIONS
299
300 applyPropertiesToLayer(node.layer(), layerTreeHost, properties, layerContentsType);
301 updateMask(node, properties, relatedLayers);
302
303 if (properties.changedProperties & RemoteLayerTreeTransaction::EventRegionChanged)
304 node.setEventRegion(properties.eventRegion);
305
306#if PLATFORM(IOS_FAMILY)
307 applyPropertiesToUIView(node.uiView(), properties, relatedLayers);
308#endif
309
310 END_BLOCK_OBJC_EXCEPTIONS
311}

313void RemoteLayerTreePropertyApplier::applyHierarchyUpdates(RemoteLayerTreeNode& node, const RemoteLayerTreeTransaction::LayerProperties& properties, const RelatedLayerMap& relatedLayers)
314{
315 BEGIN_BLOCK_OBJC_EXCEPTIONS
316
317 if (!properties.changedProperties.contains(RemoteLayerTreeTransaction::ChildrenChanged))
318 return;
319
320#if PLATFORM(IOS_FAMILY)
321 auto hasViewChildren = [&] {
322 if (node.uiView() && [[node.uiView() subviews] count])
323 return true;
324 if (properties.children.isEmpty())
325 return false;
326 auto* childNode = relatedLayers.get(properties.children.first());
327 ASSERT(childNode);
328 return childNode && childNode->uiView();
329 };
330
331 auto contentView = [&] {
332 if (properties.customAppearance == GraphicsLayer::CustomAppearance::LightBackdrop || properties.customAppearance == GraphicsLayer::CustomAppearance::DarkBackdrop) {
333 // This is a UIBackdropView, which should have children attached to
334 // its content view, not directly on its layers.
335 return [(_UIBackdropView *)node.uiView() contentView];
336 }
337 return node.uiView();
338 };
339
340 if (hasViewChildren()) {
341 ASSERT(node.uiView());
342 [contentView() _web_setSubviews:createNSArray(properties.children, [&] (auto& child) -> UIView * {
343 auto* childNode = relatedLayers.get(child);
344 ASSERT(childNode);
345 if (!childNode)
346 return nil;
347 ASSERT(childNode->uiView());
348 return childNode->uiView();
349 }).get()];
350 return;
351 }
352#endif
353
354 node.layer().sublayers = createNSArray(properties.children, [&] (auto& child) -> CALayer * {
355 auto* childNode = relatedLayers.get(child);
356 ASSERT(childNode);
357 if (!childNode)
358 return nil;
359#if PLATFORM(IOS_FAMILY)
360 ASSERT(!childNode->uiView());
361#endif
362 return childNode->layer();
363 }).get();
364
365 END_BLOCK_OBJC_EXCEPTIONS
366}

368void RemoteLayerTreePropertyApplier::updateMask(RemoteLayerTreeNode& node, const RemoteLayerTreeTransaction::LayerProperties& properties, const RelatedLayerMap& relatedLayers)
369{
370 if (!properties.changedProperties.contains(RemoteLayerTreeTransaction::MaskLayerChanged))
371 return;
372
373 auto maskOwnerLayer = [&] {
374 CALayer *layer = node.layer();
375#if PLATFORM(IOS_FAMILY)
376 if (properties.customAppearance == GraphicsLayer::CustomAppearance::LightBackdrop || properties.customAppearance == GraphicsLayer::CustomAppearance::DarkBackdrop) {
377 // This is a UIBackdropView, which means any mask must be applied to the CABackdropLayer rather
378 // that the view's layer. The backdrop is the first layer child.
379 if (layer.sublayers.count && [layer.sublayers[0] isKindOfClass:[CABackdropLayer class]])
380 layer = layer.sublayers[0];
381 }
382#endif
383 return layer;
384 };
385
386 if (!properties.maskLayerID) {
387 maskOwnerLayer().mask = nullptr;
388 return;
389 }
390
391 auto* maskNode = relatedLayers.get(properties.maskLayerID);
392 ASSERT(maskNode);
393 if (!maskNode)
394 return;
395 CALayer *maskLayer = maskNode->layer();
396 ASSERT(!maskLayer.superlayer);
397 if (maskLayer.superlayer)
398 return;
399 maskOwnerLayer().mask = maskLayer;
400}
