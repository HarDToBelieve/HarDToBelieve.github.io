26#pragma once
27
28#include "ArgumentCoders.h"
29#include "NetworkResourceLoadIdentifier.h"
30#include <WebCore/FrameIdentifier.h>
31#include <wtf/URL.h>
32#include <wtf/WallTime.h>
33
34namespace WebKit {
35
36struct ResourceLoadInfo {
37
38 enum class Type : uint8_t {
39 ApplicationManifest,
40 Beacon,
41 CSPReport,
42 Document,
43 Fetch,
44 Font,
45 Image,
46 Media,
47 Object,
48 Other,
49 Ping,
50 Script,
51 Stylesheet,
52 XMLHTTPRequest,
53 XSLT
54 };
55
56 NetworkResourceLoadIdentifier resourceLoadID;
57 std::optional<WebCore::FrameIdentifier> frameID;
58 std::optional<WebCore::FrameIdentifier> parentFrameID;
59 URL originalURL;
60 String originalHTTPMethod;
61 WallTime eventTimestamp;
62 bool loadedFromCache { false };
63 Type type { Type::Other };
64
65 void encode(IPC::Encoder& encoder) const
66 {
67 encoder << resourceLoadID;
68 encoder << frameID;
69 encoder << parentFrameID;
70 encoder << originalURL;
71 encoder << originalHTTPMethod;
72 encoder << eventTimestamp;
73 encoder << loadedFromCache;
74 encoder << type;
75 }
76
77 static std::optional<ResourceLoadInfo> decode(IPC::Decoder& decoder)
78 {
79 std::optional<NetworkResourceLoadIdentifier> resourceLoadID;
80 decoder >> resourceLoadID;
81 if (!resourceLoadID)
82 return std::nullopt;
83
84 std::optional<std::optional<WebCore::FrameIdentifier>> frameID;
85 decoder >> frameID;
86 if (!frameID)
87 return std::nullopt;
88
89 std::optional<std::optional<WebCore::FrameIdentifier>> parentFrameID;
90 decoder >> parentFrameID;
91 if (!parentFrameID)
92 return std::nullopt;
93
94 std::optional<URL> originalURL;
95 decoder >> originalURL;
96 if (!originalURL)
97 return std::nullopt;
98
99 std::optional<String> originalHTTPMethod;
100 decoder >> originalHTTPMethod;
101 if (!originalHTTPMethod)
102 return std::nullopt;
103
104 std::optional<WallTime> eventTimestamp;
105 decoder >> eventTimestamp;
106 if (!eventTimestamp)
107 return std::nullopt;
108
109 std::optional<bool> loadedFromCache;
110 decoder >> loadedFromCache;
111 if (!loadedFromCache)
112 return std::nullopt;
113
114 std::optional<Type> type;
115 decoder >> type;
116 if (!type)
117 return std::nullopt;
118
119 return {{
120 WTFMove(*resourceLoadID),
121 WTFMove(*frameID),
122 WTFMove(*parentFrameID),
123 WTFMove(*originalURL),
124 WTFMove(*originalHTTPMethod),
125 WTFMove(*eventTimestamp),
126 WTFMove(*loadedFromCache),
127 WTFMove(*type),
128 }};
129 }
130};
131
132} // namespace WebKit
