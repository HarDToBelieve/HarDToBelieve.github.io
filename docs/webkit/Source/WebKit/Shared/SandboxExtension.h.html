42namespace WebKit {
43
44class SandboxExtensionImpl;
45
46class SandboxExtension : public RefCounted<SandboxExtension> {
47public:
48 enum class Type {
49 ReadOnly,
50 ReadWrite,
51 Mach,
52 IOKit,
53 Generic,
54 ReadByProcess
55 };
56
57 enum class Flags : uint8_t {
58 Default,
59 NoReport,
60 DoNotCanonicalize,
61 };
62
63 class Handle {
64 WTF_MAKE_NONCOPYABLE(Handle);
65 public:
66 Handle();
67#if ENABLE(SANDBOX_EXTENSIONS)
68 Handle(Handle&&);
69 Handle& operator=(Handle&&);
70#else
71 Handle(Handle&&) = default;
72 Handle& operator=(Handle&&) = default;
73#endif
74 ~Handle();
75
76 void encode(IPC::Encoder&) const;
77 static std::optional<Handle> decode(IPC::Decoder&);
78
79 private:
80 friend class SandboxExtension;
81#if ENABLE(SANDBOX_EXTENSIONS)
82 mutable std::unique_ptr<SandboxExtensionImpl> m_sandboxExtension;
83#endif
84 };
85
86 class HandleArray {
87 WTF_MAKE_NONCOPYABLE(HandleArray);
88 public:
89 HandleArray();
90 HandleArray(HandleArray&&) = default;
91 HandleArray& operator=(HandleArray&&) = default;
92 ~HandleArray();
93 void allocate(size_t);
94 void append(Handle&&);
95 Handle& operator[](size_t i);
96 Handle& at(size_t i) { return operator[](i); }
97 const Handle& operator[](size_t i) const;
98 Handle* begin();
99 Handle* end();
100 const Handle* begin() const;
101 const Handle* end() const;
102 size_t size() const;
103 void encode(IPC::Encoder&) const;
104 static std::optional<HandleArray> decode(IPC::Decoder&);
105
106 private:
107#if ENABLE(SANDBOX_EXTENSIONS)
108 Vector<Handle> m_data;
109#else
110 Handle m_emptyHandle;
111#endif
112 };
113
114 static RefPtr<SandboxExtension> create(Handle&&);
115 static bool createHandle(const String& path, Type, Handle&);
116 static SandboxExtension::HandleArray createReadOnlyHandlesForFiles(ASCIILiteral logLabel, const Vector<String>& paths);
117 static bool createHandleWithoutResolvingPath(const String& path, Type, Handle&);
118 static bool createHandleForReadWriteDirectory(const String& path, Handle&); // Will attempt to create the directory.
119 static String createHandleForTemporaryFile(const String& prefix, Type, Handle&);
120 static bool createHandleForGenericExtension(ASCIILiteral extensionClass, Handle&);
121#if HAVE(AUDIT_TOKEN)
122 static bool createHandleForMachLookup(ASCIILiteral service, std::optional<audit_token_t>, Handle&, OptionSet<Flags> = Flags::Default);
123 static HandleArray createHandlesForMachLookup(const Vector<ASCIILiteral>& services, std::optional<audit_token_t>, OptionSet<Flags> = Flags::Default);
124 static bool createHandleForReadByAuditToken(const String& path, audit_token_t, Handle&);
125 static bool createHandleForIOKitClassExtension(ASCIILiteral iokitClass, std::optional<audit_token_t>, Handle&, OptionSet<Flags> = Flags::Default);
126 static HandleArray createHandlesForIOKitClassExtensions(const Vector<ASCIILiteral>& iokitClasses, std::optional<audit_token_t>, OptionSet<Flags> = Flags::Default);
127#endif
128 ~SandboxExtension();
129
130 bool consume();
131 bool revoke();
132
133 bool consumePermanently();
134 static bool consumePermanently(const Handle&);
135 static bool consumePermanently(const HandleArray&);
136
137private:
138 explicit SandboxExtension(const Handle&);
139
140#if ENABLE(SANDBOX_EXTENSIONS)
141 mutable std::unique_ptr<SandboxExtensionImpl> m_sandboxExtension;
142 size_t m_useCount { 0 };
143#endif
144};
