44void ShareableBitmap::Handle::encode(IPC::Encoder& encoder) const
45{
46 SharedMemory::IPCHandle ipcHandle(WTFMove(m_handle), numBytesForSize(m_size, m_configuration));
47 encoder << ipcHandle;
48 encoder << m_size;
49 encoder << m_configuration;
50}
51
52bool ShareableBitmap::Handle::decode(IPC::Decoder& decoder, Handle& handle)
53{
54 SharedMemory::IPCHandle ipcHandle;
55 if (!decoder.decode(ipcHandle))
56 return false;
57 if (!decoder.decode(handle.m_size))
58 return false;
59 if (handle.m_size.width() < 0 || handle.m_size.height() < 0)
60 return false;
61 if (!decoder.decode(handle.m_configuration))
62 return false;
63
64 handle.m_handle = WTFMove(ipcHandle.handle);
65 return true;
66}

68void ShareableBitmap::Handle::clear()
69{
70 m_handle.clear();
71 m_size = IntSize();
72 m_configuration = { };
73}
74
75void ShareableBitmap::Configuration::encode(IPC::Encoder& encoder) const
76{
77 encoder << colorSpace << isOpaque;
78#if USE(DIRECT2D)
79 SharedMemory::Handle::encodeHandle(encoder, sharedResourceHandle);
80
81 // Hand off ownership of our HANDLE to the receiving process. It will close it for us.
82 // FIXME: If the receiving process crashes before it receives the memory, the memory will be
83 // leaked. See <http://webkit.org/b/47502>.
84 sharedResourceHandle = nullptr;
85#endif
86}
87
88bool ShareableBitmap::Configuration::decode(IPC::Decoder& decoder, Configuration& result)
89{
90 std::optional<std::optional<WebCore::DestinationColorSpace>> colorSpace;
91 decoder >> colorSpace;
92 if (!colorSpace)
93 return false;
94
95 std::optional<bool> isOpaque;
96 decoder >> isOpaque;
97 if (!isOpaque)
98 return false;
99
100 result = Configuration { WTFMove(*colorSpace), *isOpaque };
101
102#if USE(DIRECT2D)
103 auto processSpecificHandle = SharedMemory::Handle::decodeHandle(decoder);
104 if (!processSpecificHandle)
105 return false;
106
107 result.sharedResourceHandle = processSpecificHandle.value();
108#endif
109
110 return true;
111}

113RefPtr<ShareableBitmap> ShareableBitmap::create(const IntSize& size, Configuration configuration)
114{
115 auto numBytes = numBytesForSize(size, configuration);
116 if (numBytes.hasOverflowed())
117 return nullptr;
118
119 void* data = 0;
120 data = ShareableBitmapMalloc::tryMalloc(numBytes);
121 if (!data)
122 return nullptr;
123 return adoptRef(new ShareableBitmap(size, configuration, data));
124}
125
126RefPtr<ShareableBitmap> ShareableBitmap::createShareable(const IntSize& size, Configuration configuration)
127{
128 auto numBytes = numBytesForSize(size, configuration);
129 if (numBytes.hasOverflowed())
130 return nullptr;
131
132 RefPtr<SharedMemory> sharedMemory = SharedMemory::allocate(numBytes);
133 if (!sharedMemory)
134 return nullptr;
135
136 return adoptRef(new ShareableBitmap(size, configuration, sharedMemory));
137}
138
139RefPtr<ShareableBitmap> ShareableBitmap::create(const IntSize& size, Configuration configuration, RefPtr<SharedMemory> sharedMemory)
140{
141 ASSERT(sharedMemory);
142
143 auto numBytes = numBytesForSize(size, configuration);
144 if (numBytes.hasOverflowed())
145 return nullptr;
146 if (sharedMemory->size() < numBytes) {
147 ASSERT_NOT_REACHED();
148 return nullptr;
149 }
150
151 return adoptRef(new ShareableBitmap(size, configuration, sharedMemory));
152}
153
154RefPtr<ShareableBitmap> ShareableBitmap::create(const Handle& handle, SharedMemory::Protection protection)
155{
156 auto sharedMemory = SharedMemory::map(handle.m_handle, protection);
157 if (!sharedMemory)
158 return nullptr;
159
160 return create(handle.m_size, handle.m_configuration, WTFMove(sharedMemory));
161}
162
163bool ShareableBitmap::createHandle(Handle& handle, SharedMemory::Protection protection) const
164{
165 ASSERT(isBackedBySharedMemory());
166
167 if (!m_sharedMemory->createHandle(handle.m_handle, protection))
168 return false;
169 handle.m_size = m_size;
170 handle.m_configuration = m_configuration;
171 return true;
172}

181ShareableBitmap::ShareableBitmap(const IntSize& size, Configuration configuration, RefPtr<SharedMemory> sharedMemory)

191
192ShareableBitmap::~ShareableBitmap()
193{
194 if (!isBackedBySharedMemory())
195 ShareableBitmapMalloc::free(m_data);
196#if USE(DIRECT2D)
197 disposeSharedResource();
198#endif
199}
200
201void* ShareableBitmap::data() const
202{
203 if (isBackedBySharedMemory())
204 return m_sharedMemory->data();
205
206 ASSERT(m_data);
207 return m_data;
208}

210CheckedUint32 ShareableBitmap::numBytesForSize(WebCore::IntSize size, const ShareableBitmap::Configuration& configuration)
211{
212#if USE(DIRECT2D)
213 // We pass references to GPU textures, so no need to allocate frame buffers here. Just send a small bit of data.
214 return sizeof(void*);
215#else
216 return calculateBytesPerRow(size, configuration) * size.height();
217#endif
218}
