26#ifndef ShareableResource_h
27#define ShareableResource_h
28
29#if ENABLE(SHAREABLE_RESOURCE)
30
31#include "SharedMemory.h"
32#include <wtf/RefCounted.h>
33#include <wtf/RefPtr.h>
34
35namespace WebCore {
36class SharedBuffer;
37}
38
39namespace WebKit {
40
41class ShareableResource : public RefCounted<ShareableResource> {
42public:
43
44 class Handle {
45 WTF_MAKE_NONCOPYABLE(Handle);
46 public:
47 Handle();
48 Handle(Handle&&) = default;
49 Handle& operator=(Handle&&) = default;
50
51 bool isNull() const { return m_handle.isNull(); }
52 unsigned size() const { return m_size; }
53
54 void encode(IPC::Encoder&) const;
55 static WARN_UNUSED_RETURN bool decode(IPC::Decoder&, Handle&);
56
57 RefPtr<WebCore::SharedBuffer> tryWrapInSharedBuffer() const;
58
59 private:
60 friend class ShareableResource;
61
62 mutable SharedMemory::Handle m_handle;
63 unsigned m_offset { 0 };
64 unsigned m_size { 0 };
65 };
66
67 // Create a shareable resource that uses malloced memory.
68 static RefPtr<ShareableResource> create(Ref<SharedMemory>&&, unsigned offset, unsigned size);
69
70 // Create a shareable resource from a handle.
71 static RefPtr<ShareableResource> map(const Handle&);
72
73 // Create a handle.
74 bool createHandle(Handle&);
75
76 ~ShareableResource();
77
78 const uint8_t* data() const;
79 unsigned size() const;
80
81private:
82 ShareableResource(Ref<SharedMemory>&&, unsigned offset, unsigned size);
83 RefPtr<WebCore::SharedBuffer> wrapInSharedBuffer();
84
85 Ref<SharedMemory> m_sharedMemory;
86
87 unsigned m_offset;
88 unsigned m_size;
89};
90
91} // namespace WebKit
92
93#endif // ENABLE(SHAREABLE_RESOURCE)
94
95#endif // ShareableResource_h
