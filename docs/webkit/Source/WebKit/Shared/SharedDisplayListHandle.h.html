35namespace WebKit {
36
37// This is a skeleton only used by the DisplayListWriterHandle & DisplayListReaderHandle subclasses.
38class SharedDisplayListHandle : public RefCounted<SharedDisplayListHandle> {
39public:
40 virtual ~SharedDisplayListHandle() = default;
41
42 static constexpr size_t headerSize()
43 {
44 return roundUpToMultipleOf<alignof(std::max_align_t)>(sizeof(DisplayListSharedMemoryHeader));
45 }
46
47 SharedMemory& sharedMemory() { return m_sharedMemory.get(); }
48 const SharedMemory& sharedMemory() const { return m_sharedMemory.get(); }
49
50 WebCore::DisplayList::ItemBufferIdentifier identifier() const { return m_identifier; }
51 uint8_t* data() const { return static_cast<uint8_t*>(sharedMemory().data()); }
52
53 uint64_t unreadBytes()
54 {
55 return header().unreadBytes.load();
56 }
57
58 enum class WaitingStatus : uint8_t {
59 NotWaiting,
60 Waiting,
61 Resuming
62 };
63 using WaitingStatusStorageType = std::underlying_type_t<WaitingStatus>;
64
65 struct ResumeReadingInformation {
66 uint64_t offset;
67 uint64_t destination;
68 };
69
70protected:
71 SharedDisplayListHandle(WebCore::DisplayList::ItemBufferIdentifier identifier, Ref<SharedMemory>&& sharedMemory)
72 : m_identifier(identifier)
73 , m_sharedMemory(WTFMove(sharedMemory))
74 {
75 }
76
77 struct DisplayListSharedMemoryHeader {
78 DisplayListSharedMemoryHeader() = delete;
79 ~DisplayListSharedMemoryHeader() = delete;
80
81 Atomic<uint64_t> unreadBytes;
82 ResumeReadingInformation resumeReadingInfo;
83 Atomic<WaitingStatusStorageType> waitingStatus;
84 };
85
86 const DisplayListSharedMemoryHeader& header() const { return *reinterpret_cast<const DisplayListSharedMemoryHeader*>(data()); }
87 DisplayListSharedMemoryHeader& header() { return *reinterpret_cast<DisplayListSharedMemoryHeader*>(data()); }
88
89private:
90 WebCore::DisplayList::ItemBufferIdentifier m_identifier;
91 Ref<SharedMemory> m_sharedMemory;
92};
93
94} // namespace WebKit
