38namespace WebKit {
39
40static ItemType itemType(const String&)
41{
42 return ItemType::Button;
43}
44
45TouchBarMenuItemData::TouchBarMenuItemData(const WebCore::HTMLMenuItemElement& element)
46{
47 type = itemType(element.attributeWithoutSynchronization(WebCore::HTMLNames::typeAttr));
48 identifier = element.attributeWithoutSynchronization(WebCore::HTMLNames::idAttr);
49 priority = element.attributeWithoutSynchronization(WebCore::HTMLNames::valueAttr).toFloat();
50}
51
52void TouchBarMenuItemData::encode(IPC::Encoder& encoder) const
53{
54 encoder << type;
55
56 encoder << identifier;
57 encoder << priority;
58}
59
60std::optional<TouchBarMenuItemData> TouchBarMenuItemData::decode(IPC::Decoder& decoder)
61{
62 TouchBarMenuItemData result;
63 if (!decoder.decode(result.type))
64 return std::nullopt;
65
66 if (!decoder.decode(result.identifier))
67 return std::nullopt;
68
69 if (!decoder.decode(result.priority))
70 return std::nullopt;
71
72 return std::make_optional(WTFMove(result));
73}
