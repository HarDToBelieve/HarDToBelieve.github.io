35void URLSchemeTaskParameters::encode(IPC::Encoder& encoder) const
36{
37 encoder << handlerIdentifier;
38 encoder << taskIdentifier;
39 encoder << request;
40 if (request.httpBody()) {
41 encoder << true;
42 request.httpBody()->encode(encoder);
43 } else
44 encoder << false;
45 encoder << frameInfo;
46}
47
48std::optional<URLSchemeTaskParameters> URLSchemeTaskParameters::decode(IPC::Decoder& decoder)
49{
50 std::optional<uint64_t> handlerIdentifier;
51 decoder >> handlerIdentifier;
52 if (!handlerIdentifier)
53 return std::nullopt;
54
55 std::optional<uint64_t> taskIdentifier;
56 decoder >> taskIdentifier;
57 if (!taskIdentifier)
58 return std::nullopt;
59
60 WebCore::ResourceRequest request;
61 if (!decoder.decode(request))
62 return std::nullopt;
63
64 std::optional<bool> hasHTTPBody;
65 decoder >> hasHTTPBody;
66 if (!hasHTTPBody)
67 return std::nullopt;
68 if (*hasHTTPBody) {
69 RefPtr<WebCore::FormData> formData = WebCore::FormData::decode(decoder);
70 if (!formData)
71 return std::nullopt;
72 request.setHTTPBody(WTFMove(formData));
73 }
74
75 std::optional<FrameInfoData> frameInfo;
76 decoder >> frameInfo;
77 if (!frameInfo)
78 return std::nullopt;
79
80 return {{
81 WTFMove(*handlerIdentifier),
82 WTFMove(*taskIdentifier),
83 WTFMove(request),
84 WTFMove(*frameInfo),
85 }};
86}
