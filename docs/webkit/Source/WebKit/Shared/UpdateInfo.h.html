40namespace WebKit {
41
42class UpdateInfo {
43 WTF_MAKE_NONCOPYABLE(UpdateInfo);
44
45public:
46 UpdateInfo() { }
47 UpdateInfo(UpdateInfo&&) = default;
48 UpdateInfo& operator=(UpdateInfo&&) = default;
49
50 void encode(IPC::Encoder&) const;
51 static WARN_UNUSED_RETURN bool decode(IPC::Decoder&, UpdateInfo&);
52
53 // The size of the web view.
54 WebCore::IntSize viewSize;
55 float deviceScaleFactor;
56
57 // The rect and delta to be scrolled.
58 WebCore::IntRect scrollRect;
59 WebCore::IntSize scrollOffset;
60
61 // The bounds of the update rects.
62 WebCore::IntRect updateRectBounds;
63
64 // All the update rects, in view coordinates.
65 Vector<WebCore::IntRect> updateRects;
66
67 // The page scale factor used to render this update.
68 float updateScaleFactor;
69
70 // The handle of the shareable bitmap containing the updates. Will be null if there are no updates.
71 ShareableBitmap::Handle bitmapHandle;
72
73 // The offset in the bitmap where the rendered contents are.
74 WebCore::IntPoint bitmapOffset;
75};
