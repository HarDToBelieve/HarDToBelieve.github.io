33namespace WebKit {
34
35void UserContentControllerParameters::encode(IPC::Encoder& encoder) const
36{
37 encoder << identifier;
38 encoder << userContentWorlds;
39 encoder << userScripts;
40 encoder << userStyleSheets;
41 encoder << messageHandlers;
42#if ENABLE(CONTENT_EXTENSIONS)
43 encoder << contentRuleLists;
44#endif
45}
46
47std::optional<UserContentControllerParameters> UserContentControllerParameters::decode(IPC::Decoder& decoder)
48{
49 std::optional<UserContentControllerIdentifier> identifier;
50 decoder >> identifier;
51 if (!identifier)
52 return std::nullopt;
53
54 std::optional<Vector<std::pair<ContentWorldIdentifier, String>>> userContentWorlds;
55 decoder >> userContentWorlds;
56 if (!userContentWorlds)
57 return std::nullopt;
58
59 std::optional<Vector<WebUserScriptData>> userScripts;
60 decoder >> userScripts;
61 if (!userScripts)
62 return std::nullopt;
63
64 std::optional<Vector<WebUserStyleSheetData>> userStyleSheets;
65 decoder >> userStyleSheets;
66 if (!userStyleSheets)
67 return std::nullopt;
68
69 std::optional<Vector<WebScriptMessageHandlerData>> messageHandlers;
70 decoder >> messageHandlers;
71 if (!messageHandlers)
72 return std::nullopt;
73
74#if ENABLE(CONTENT_EXTENSIONS)
75 std::optional<Vector<std::pair<String, WebCompiledContentRuleListData>>> contentRuleLists;
76 decoder >> contentRuleLists;
77 if (!contentRuleLists)
78 return std::nullopt;
79#endif
80
81 return {{
82 WTFMove(*identifier),
83 WTFMove(*userContentWorlds),
84 WTFMove(*userScripts),
85 WTFMove(*userStyleSheets),
86 WTFMove(*messageHandlers),
87#if ENABLE(CONTENT_EXTENSIONS)
88 WTFMove(*contentRuleLists),
89#endif
90 }};
91}
