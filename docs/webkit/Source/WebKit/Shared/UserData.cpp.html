71static bool shouldTransform(const API::Object& object, const UserData::Transformer& transformer)
72{
73 if (object.type() == API::Object::Type::Array) {
74 const auto& array = static_cast<const API::Array&>(object);
75
76 for (const auto& element : array.elements()) {
77 if (!element)
78 continue;
79
80 if (shouldTransform(*element, transformer))
81 return true;
82 }
83 }
84
85 if (object.type() == API::Object::Type::Dictionary) {
86 const auto& dictionary = static_cast<const API::Dictionary&>(object);
87
88 for (const auto& keyValuePair : dictionary.map()) {
89 if (!keyValuePair.value)
90 continue;
91
92 if (shouldTransform(*keyValuePair.value, transformer))
93 return true;
94 }
95 }
96
97 return transformer.shouldTransformObject(object);
98}
99
100static RefPtr<API::Object> transformGraph(API::Object& object, const UserData::Transformer& transformer)
101{
102 if (object.type() == API::Object::Type::Array) {
103 auto& array = static_cast<API::Array&>(object);
104
105 Vector<RefPtr<API::Object>> elements;
106 elements.reserveInitialCapacity(array.elements().size());
107 for (const auto& element : array.elements()) {
108 if (!element)
109 elements.uncheckedAppend(nullptr);
110 else
111 elements.uncheckedAppend(transformGraph(*element, transformer));
112 }
113
114 return API::Array::create(WTFMove(elements));
115 }
116
117 if (object.type() == API::Object::Type::Dictionary) {
118 auto& dictionary = static_cast<API::Dictionary&>(object);
119
120 API::Dictionary::MapType map;
121 for (const auto& keyValuePair : dictionary.map()) {
122 if (!keyValuePair.value)
123 map.add(keyValuePair.key, nullptr);
124 else
125 map.add(keyValuePair.key, transformGraph(*keyValuePair.value, transformer));
126 }
127 return API::Dictionary::create(WTFMove(map));
128 }
129
130 return transformer.transformObject(object);
131}
132
133RefPtr<API::Object> UserData::transform(API::Object* object, const Transformer& transformer)
134{
135 if (!object)
136 return nullptr;
137
138 if (!shouldTransform(*object, transformer))
139 return object;
140
141 return transformGraph(*object, transformer);
142}

154void UserData::encode(IPC::Encoder& encoder, const API::Object* object)

156 if (!object) {

158 return;
159 }

161 encode(encoder, *object);

164void UserData::encode(IPC::Encoder& encoder, const API::Object& object)
165{
166 API::Object::Type type = object.type();
167 encoder << type;
168
169 switch (object.type()) {
170 case API::Object::Type::Array: {
171 auto& array = static_cast<const API::Array&>(object);
172 encoder << static_cast<uint64_t>(array.size());
173 for (size_t i = 0; i < array.size(); ++i)
174 encode(encoder, array.at(i));
175 break;
176 }
177
178 case API::Object::Type::Boolean:
179 static_cast<const API::Boolean&>(object).encode(encoder);
180 break;
181
182 case API::Object::Type::CertificateInfo: {
183 const auto& certificateInfo = static_cast<const WebCertificateInfo&>(object);
184 encoder << certificateInfo.certificateInfo();
185 break;
186 }
187
188 case API::Object::Type::Data:
189 static_cast<const API::Data&>(object).encode(encoder);
190 break;
191
192 case API::Object::Type::Dictionary: {
193 auto& dictionary = static_cast<const API::Dictionary&>(object);
194 auto& map = dictionary.map();
195
196 encoder << static_cast<uint64_t>(map.size());
197 for (const auto& keyValuePair : map) {
198 encoder << keyValuePair.key;
199 encode(encoder, keyValuePair.value.get());
200 }
201 break;
202 }
203
204 case API::Object::Type::Double:
205 static_cast<const API::Double&>(object).encode(encoder);
206 break;
207
208 case API::Object::Type::Error:
209 static_cast<const API::Error&>(object).encode(encoder);
210 break;
211
212 case API::Object::Type::FrameHandle:
213 static_cast<const API::FrameHandle&>(object).encode(encoder);
214 break;
215
216 case API::Object::Type::Image: {
217 auto& image = static_cast<const WebImage&>(object);
218
219 ShareableBitmap::Handle handle;
220 ASSERT(image.bitmap().isBackedBySharedMemory());
221 if (!image.bitmap().isBackedBySharedMemory() || !image.bitmap().createHandle(handle)) {
222 // Initial false indicates no allocated bitmap or is not shareable.
223 encoder << false;
224 break;
225 }
226
227 // Initial true indicates a bitmap was allocated and is shareable.
228 encoder << true;
229 encoder << handle;
230 break;
231 }
232
233 case API::Object::Type::PageGroupHandle:
234 static_cast<const API::PageGroupHandle&>(object).encode(encoder);
235 break;
236
237 case API::Object::Type::PageHandle:
238 static_cast<const API::PageHandle&>(object).encode(encoder);
239 break;
240
241 case API::Object::Type::Point:
242 static_cast<const API::Point&>(object).encode(encoder);
243 break;
244
245 case API::Object::Type::Rect:
246 static_cast<const API::Rect&>(object).encode(encoder);
247 break;
248
249 case API::Object::Type::SerializedScriptValue: {
250 auto& serializedScriptValue = static_cast<const API::SerializedScriptValue&>(object);
251 encoder << serializedScriptValue.dataReference();
252 break;
253 }
254
255 case API::Object::Type::Size:
256 static_cast<const API::Size&>(object).encode(encoder);
257 break;
258
259 case API::Object::Type::String: {
260 auto& string = static_cast<const API::String&>(object);
261 encoder << string.string();
262 break;
263 }
264
265 case API::Object::Type::URL:
266 static_cast<const API::URL&>(object).encode(encoder);
267 break;
268
269 case API::Object::Type::URLRequest:
270 static_cast<const API::URLRequest&>(object).encode(encoder);
271 break;
272
273 case API::Object::Type::URLResponse:
274 static_cast<const API::URLResponse&>(object).encode(encoder);
275 break;
276
277 case API::Object::Type::UInt64:
278 static_cast<const API::UInt64&>(object).encode(encoder);
279 break;
280
281 case API::Object::Type::Int64:
282 static_cast<const API::Int64&>(object).encode(encoder);
283 break;
284
285 case API::Object::Type::UserContentURLPattern: {
286 auto& urlPattern = static_cast<const API::UserContentURLPattern&>(object);
287 encoder << urlPattern.patternString();
288 break;
289 }
290
291#if PLATFORM(COCOA)
292 case API::Object::Type::ObjCObjectGraph:
293 static_cast<const ObjCObjectGraph&>(object).encode(encoder);
294 break;
295#endif
296
297 default:
298 ASSERT_NOT_REACHED();
299 }

302bool UserData::decode(IPC::Decoder& decoder, RefPtr<API::Object>& result)
303{
304 API::Object::Type type;
305 if (!decoder.decode(type))
306 return false;
307
308 switch (type) {
309 case API::Object::Type::Array: {
310 uint64_t decodedSize;
311 if (!decoder.decode(decodedSize))
312 return false;
313
314 if (!isInBounds<size_t>(decodedSize))
315 return false;
316
317 auto size = static_cast<size_t>(decodedSize);
318
319 Vector<RefPtr<API::Object>> elements;
320 for (size_t i = 0; i < size; ++i) {
321 RefPtr<API::Object> element;
322 if (!decode(decoder, element))
323 return false;
324
325 elements.append(WTFMove(element));
326 }
327
328 result = API::Array::create(WTFMove(elements));
329 break;
330 }
331
332 case API::Object::Type::Boolean:
333 if (!API::Boolean::decode(decoder, result))
334 return false;
335 break;
336
337 case API::Object::Type::CertificateInfo: {
338 std::optional<WebCore::CertificateInfo> certificateInfo;
339 decoder >> certificateInfo;
340 if (!certificateInfo)
341 return false;
342 result = WebCertificateInfo::create(*certificateInfo);
343 break;
344 }
345
346 case API::Object::Type::Data:
347 if (!API::Data::decode(decoder, result))
348 return false;
349 break;
350
351 case API::Object::Type::Dictionary: {
352 uint64_t decodedSize;
353 if (!decoder.decode(decodedSize))
354 return false;
355
356 if (!isInBounds<size_t>(decodedSize))
357 return false;
358
359 auto size = static_cast<size_t>(decodedSize);
360
361 API::Dictionary::MapType map;
362 for (size_t i = 0; i < size; ++i) {
363 String key;
364 if (!decoder.decode(key))
365 return false;
366
367 RefPtr<API::Object> value;
368 if (!decode(decoder, value))
369 return false;
370
371 if (!map.add(WTFMove(key), WTFMove(value)).isNewEntry)
372 return false;
373 }
374
375 result = API::Dictionary::create(WTFMove(map));
376 break;
377 }
378
379 case API::Object::Type::Double:
380 if (!API::Double::decode(decoder, result))
381 return false;
382 break;
383
384 case API::Object::Type::Error:
385 if (!API::Error::decode(decoder, result))
386 return false;
387 break;
388
389 case API::Object::Type::FrameHandle:
390 if (!API::FrameHandle::decode(decoder, result))
391 return false;
392 break;
393
394 case API::Object::Type::Image: {
395 bool didEncode = false;
396 if (!decoder.decode(didEncode))
397 return false;
398
399 if (!didEncode)
400 break;
401
402 ShareableBitmap::Handle handle;
403 if (!decoder.decode(handle))
404 return false;
405
406 auto bitmap = ShareableBitmap::create(handle);
407 if (!bitmap)
408 return false;
409
410 result = WebImage::create(bitmap.releaseNonNull());
411 break;
412 }
413
414 case API::Object::Type::Null:
415 result = nullptr;
416 break;
417
418 case API::Object::Type::PageGroupHandle:
419 if (!API::PageGroupHandle::decode(decoder, result))
420 return false;
421 break;
422
423 case API::Object::Type::PageHandle:
424 if (!API::PageHandle::decode(decoder, result))
425 return false;
426 break;
427
428 case API::Object::Type::Point:
429 if (!API::Point::decode(decoder, result))
430 return false;
431 break;
432
433 case API::Object::Type::Rect:
434 if (!API::Rect::decode(decoder, result))
435 return false;
436 break;
437
438 case API::Object::Type::SerializedScriptValue: {
439 IPC::DataReference dataReference;
440 if (!decoder.decode(dataReference))
441 return false;
442
443 result = API::SerializedScriptValue::adopt(dataReference.vector());
444 break;
445 }
446
447 case API::Object::Type::Size:
448 if (!API::Size::decode(decoder, result))
449 return false;
450 break;
451
452 case API::Object::Type::String: {
453 String string;
454 if (!decoder.decode(string))
455 return false;
456
457 result = API::String::create(string);
458 break;
459 }
460
461 case API::Object::Type::URL:
462 if (!API::URL::decode(decoder, result))
463 return false;
464 break;
465
466 case API::Object::Type::URLRequest:
467 if (!API::URLRequest::decode(decoder, result))
468 return false;
469 break;
470
471 case API::Object::Type::URLResponse:
472 if (!API::URLResponse::decode(decoder, result))
473 return false;
474 break;
475
476 case API::Object::Type::UInt64:
477 if (!API::UInt64::decode(decoder, result))
478 return false;
479 break;
480
481 case API::Object::Type::Int64:
482 if (!API::Int64::decode(decoder, result))
483 return false;
484 break;
485
486 case API::Object::Type::UserContentURLPattern: {
487 String string;
488 if (!decoder.decode(string))
489 return false;
490 result = API::UserContentURLPattern::create(string);
491 break;
492 }
493
494#if PLATFORM(COCOA)
495 case API::Object::Type::ObjCObjectGraph:
496 if (!ObjCObjectGraph::decode(decoder, result))
497 return false;
498 break;
499#endif
500
501 default:
502 ASSERT_NOT_REACHED();
503 return false;
504 }
505
506 return true;
507}
