40Ref<WebBackForwardListItem> WebBackForwardListItem::create(BackForwardListItemState&& backForwardListItemState, WebPageProxyIdentifier pageID)
41{
42 RELEASE_ASSERT(RunLoop::isMain());
43 return adoptRef(*new WebBackForwardListItem(WTFMove(backForwardListItemState), pageID));
44}
45
46WebBackForwardListItem::WebBackForwardListItem(BackForwardListItemState&& backForwardListItemState, WebPageProxyIdentifier pageID)
47 : m_itemState(WTFMove(backForwardListItemState))
48 , m_pageID(pageID)
49 , m_lastProcessIdentifier(m_itemState.identifier.processIdentifier)
50{
51 auto result = allItems().add(m_itemState.identifier, this);
52 ASSERT_UNUSED(result, result.isNewEntry);
53}
54
55WebBackForwardListItem::~WebBackForwardListItem()
56{
57 RELEASE_ASSERT(RunLoop::isMain());
58 ASSERT(allItems().get(m_itemState.identifier) == this);
59 allItems().remove(m_itemState.identifier);
60 removeFromBackForwardCache();
61}

75static const FrameState* childItemWithDocumentSequenceNumber(const FrameState& frameState, int64_t number)
76{
77 for (const auto& child : frameState.children) {
78 if (child.documentSequenceNumber == number)
79 return &child;
80 }
81
82 return nullptr;
83}
84
85static const FrameState* childItemWithTarget(const FrameState& frameState, const String& target)
86{
87 for (const auto& child : frameState.children) {
88 if (child.target == target)
89 return &child;
90 }
91
92 return nullptr;
93}
94
95static bool documentTreesAreEqual(const FrameState& a, const FrameState& b)
96{
97 if (a.documentSequenceNumber != b.documentSequenceNumber)
98 return false;
99
100 if (a.children.size() != b.children.size())
101 return false;
102
103 for (const auto& child : a.children) {
104 const FrameState* otherChild = childItemWithDocumentSequenceNumber(b, child.documentSequenceNumber);
105 if (!otherChild || !documentTreesAreEqual(child, *otherChild))
106 return false;
107 }
108
109 return true;
110}

112bool WebBackForwardListItem::itemIsInSameDocument(const WebBackForwardListItem& other) const
113{
114 if (m_pageID != other.m_pageID)
115 return false;
116
117 // The following logic must be kept in sync with WebCore::HistoryItem::shouldDoSameDocumentNavigationTo().
118
119 const FrameState& mainFrameState = m_itemState.pageState.mainFrameState;
120 const FrameState& otherMainFrameState = other.m_itemState.pageState.mainFrameState;
121
122 if (mainFrameState.stateObjectData || otherMainFrameState.stateObjectData)
123 return mainFrameState.documentSequenceNumber == otherMainFrameState.documentSequenceNumber;
124
125 URL url = URL({ }, mainFrameState.urlString);
126 URL otherURL = URL({ }, otherMainFrameState.urlString);
127
128 if ((url.hasFragmentIdentifier() || otherURL.hasFragmentIdentifier()) && equalIgnoringFragmentIdentifier(url, otherURL))
129 return mainFrameState.documentSequenceNumber == otherMainFrameState.documentSequenceNumber;
130
131 return documentTreesAreEqual(mainFrameState, otherMainFrameState);
132}
133
134static bool hasSameFrames(const FrameState& a, const FrameState& b)
135{
136 if (a.target != b.target)
137 return false;
138
139 if (a.children.size() != b.children.size())
140 return false;
141
142 for (const auto& child : a.children) {
143 if (!childItemWithTarget(b, child.target))
144 return false;
145 }
146
147 return true;
148}

150bool WebBackForwardListItem::itemIsClone(const WebBackForwardListItem& other)
151{
152 // The following logic must be kept in sync with WebCore::HistoryItem::itemsAreClones().
153
154 if (this == &other)
155 return false;
156
157 const FrameState& mainFrameState = m_itemState.pageState.mainFrameState;
158 const FrameState& otherMainFrameState = other.m_itemState.pageState.mainFrameState;
159
160 if (mainFrameState.itemSequenceNumber != otherMainFrameState.itemSequenceNumber)
161 return false;
162
163 return hasSameFrames(mainFrameState, otherMainFrameState);
164}

166void WebBackForwardListItem::wasRemovedFromBackForwardList()
167{
168 removeFromBackForwardCache();
169}

171void WebBackForwardListItem::removeFromBackForwardCache()
172{
173 if (m_backForwardCacheEntry)
174 m_backForwardCacheEntry->backForwardCache().removeEntry(*this);
175 ASSERT(!m_backForwardCacheEntry);
176}
