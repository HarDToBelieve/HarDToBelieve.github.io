58 const WebCore::BackForwardItemIdentifier& itemID() const { return m_itemState.identifier; }
59 const BackForwardListItemState& itemState() { return m_itemState; }
60 WebPageProxyIdentifier pageID() const { return m_pageID; }

65 void setPageState(PageState&& pageState) { m_itemState.pageState = WTFMove(pageState); }

75 bool itemIsInSameDocument(const WebBackForwardListItem&) const;

78#if PLATFORM(COCOA) || PLATFORM(GTK)
