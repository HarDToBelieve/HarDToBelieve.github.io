37void WebCompiledContentRuleListData::encode(IPC::Encoder& encoder) const
38{
39 SharedMemory::Handle handle;
40 data->createHandle(handle, SharedMemory::Protection::ReadOnly);
41
42#if OS(DARWIN) || OS(WINDOWS)
43 // Exact data size is the last bytecode offset plus its size.
44 uint64_t dataSize = topURLFiltersBytecodeOffset + topURLFiltersBytecodeSize;
45#else
46 uint64_t dataSize = 0;
47#endif
48 encoder << SharedMemory::IPCHandle { WTFMove(handle), dataSize };
49
50 encoder << conditionsApplyOnlyToDomainOffset;
51 encoder << actionsOffset;
52 encoder << actionsSize;
53 encoder << filtersWithoutConditionsBytecodeOffset;
54 encoder << filtersWithoutConditionsBytecodeSize;
55 encoder << filtersWithConditionsBytecodeOffset;
56 encoder << filtersWithConditionsBytecodeSize;
57 encoder << topURLFiltersBytecodeOffset;
58 encoder << topURLFiltersBytecodeSize;
59}

61std::optional<WebCompiledContentRuleListData> WebCompiledContentRuleListData::decode(IPC::Decoder& decoder)
62{
63 SharedMemory::IPCHandle ipcHandle;
64 if (!decoder.decode(ipcHandle))
65 return std::nullopt;
66 RefPtr<SharedMemory> data = SharedMemory::map(ipcHandle.handle, SharedMemory::Protection::ReadOnly);
67
68 std::optional<unsigned> conditionsApplyOnlyToDomainOffset;
69 decoder >> conditionsApplyOnlyToDomainOffset;
70 if (!conditionsApplyOnlyToDomainOffset)
71 return std::nullopt;
72
73 std::optional<unsigned> actionsOffset;
74 decoder >> actionsOffset;
75 if (!actionsOffset)
76 return std::nullopt;
77
78 std::optional<unsigned> actionsSize;
79 decoder >> actionsSize;
80 if (!actionsSize)
81 return std::nullopt;
82
83 std::optional<unsigned> filtersWithoutConditionsBytecodeOffset;
84 decoder >> filtersWithoutConditionsBytecodeOffset;
85 if (!filtersWithoutConditionsBytecodeOffset)
86 return std::nullopt;
87
88 std::optional<unsigned> filtersWithoutConditionsBytecodeSize;
89 decoder >> filtersWithoutConditionsBytecodeSize;
90 if (!filtersWithoutConditionsBytecodeSize)
91 return std::nullopt;
92
93 std::optional<unsigned> filtersWithConditionsBytecodeOffset;
94 decoder >> filtersWithConditionsBytecodeOffset;
95 if (!filtersWithConditionsBytecodeOffset)
96 return std::nullopt;
97
98 std::optional<unsigned> filtersWithConditionsBytecodeSize;
99 decoder >> filtersWithConditionsBytecodeSize;
100 if (!filtersWithConditionsBytecodeSize)
101 return std::nullopt;
102
103 std::optional<unsigned> topURLFiltersBytecodeOffset;
104 decoder >> topURLFiltersBytecodeOffset;
105 if (!topURLFiltersBytecodeOffset)
106 return std::nullopt;
107
108 std::optional<unsigned> topURLFiltersBytecodeSize;
109 decoder >> topURLFiltersBytecodeSize;
110 if (!topURLFiltersBytecodeSize)
111 return std::nullopt;
112
113 return {{
114 WTFMove(data),
115 WTFMove(*conditionsApplyOnlyToDomainOffset),
116 WTFMove(*actionsOffset),
117 WTFMove(*actionsSize),
118 WTFMove(*filtersWithoutConditionsBytecodeOffset),
119 WTFMove(*filtersWithoutConditionsBytecodeSize),
120 WTFMove(*filtersWithConditionsBytecodeOffset),
121 WTFMove(*filtersWithConditionsBytecodeSize),
122 WTFMove(*topURLFiltersBytecodeOffset),
123 WTFMove(*topURLFiltersBytecodeSize)
124 }};
125}
