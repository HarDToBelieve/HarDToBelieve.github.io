42namespace WebKit {
43
44class WebEvent {
45 WTF_MAKE_FAST_ALLOCATED;
46public:
47 enum Type {
48 NoType = -1,
49
50 // WebMouseEvent
51 MouseDown,
52 MouseUp,
53 MouseMove,
54 MouseForceChanged,
55 MouseForceDown,
56 MouseForceUp,
57
58 // WebWheelEvent
59 Wheel,
60
61 // WebKeyboardEvent
62 KeyDown,
63 KeyUp,
64 RawKeyDown,
65 Char,
66
67#if ENABLE(TOUCH_EVENTS)
68 // WebTouchEvent
69 TouchStart,
70 TouchMove,
71 TouchEnd,
72 TouchCancel,
73#endif
74
75#if ENABLE(MAC_GESTURE_EVENTS)
76 GestureStart,
77 GestureChange,
78 GestureEnd,
79#endif
80 };
81
82 enum class Modifier : uint8_t {
83 ShiftKey = 1 << 0,
84 ControlKey = 1 << 1,
85 AltKey = 1 << 2,
86 MetaKey = 1 << 3,
87 CapsLockKey = 1 << 4,
88 };
89
90 Type type() const { return static_cast<Type>(m_type); }
91
92 bool shiftKey() const { return m_modifiers.contains(Modifier::ShiftKey); }
93 bool controlKey() const { return m_modifiers.contains(Modifier::ControlKey); }
94 bool altKey() const { return m_modifiers.contains(Modifier::AltKey); }
95 bool metaKey() const { return m_modifiers.contains(Modifier::MetaKey); }
96 bool capsLockKey() const { return m_modifiers.contains(Modifier::CapsLockKey); }
97
98 OptionSet<Modifier> modifiers() const { return m_modifiers; }
99
100 WallTime timestamp() const { return m_timestamp; }
101
102protected:
103 WebEvent();
104
105 WebEvent(Type, OptionSet<Modifier>, WallTime timestamp);
106
107 void encode(IPC::Encoder&) const;
108 static WARN_UNUSED_RETURN bool decode(IPC::Decoder&, WebEvent&);
109
110private:
111 uint32_t m_type;
112 OptionSet<Modifier> m_modifiers;
113 WallTime m_timestamp;
114};
115
116} // namespace WebKit
