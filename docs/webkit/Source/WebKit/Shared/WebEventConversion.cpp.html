40class WebKit2PlatformMouseEvent : public WebCore::PlatformMouseEvent {
41public:
42 WebKit2PlatformMouseEvent(const WebMouseEvent& webEvent)
43 {
44 // PlatformEvent
45 switch (webEvent.type()) {
46 case WebEvent::MouseDown:
47 m_type = WebCore::PlatformEvent::MousePressed;
48 m_force = WebCore::ForceAtClick;
49 break;
50 case WebEvent::MouseUp:
51 m_type = WebCore::PlatformEvent::MouseReleased;
52 m_force = WebCore::ForceAtClick;
53 break;
54 case WebEvent::MouseMove:
55 m_type = WebCore::PlatformEvent::MouseMoved;
56 m_force = webEvent.force();
57 break;
58 case WebEvent::MouseForceChanged:
59 m_type = WebCore::PlatformEvent::MouseForceChanged;
60 m_force = webEvent.force();
61 break;
62 case WebEvent::MouseForceDown:
63 m_type = WebCore::PlatformEvent::MouseForceDown;
64 m_force = WebCore::ForceAtForceClick;
65 break;
66 case WebEvent::MouseForceUp:
67 m_type = WebCore::PlatformEvent::MouseForceUp;
68 m_force = WebCore::ForceAtForceClick;
69 break;
70 default:
71 ASSERT_NOT_REACHED();
72 }
73
74 if (webEvent.shiftKey())
75 m_modifiers.add(Modifier::ShiftKey);
76 if (webEvent.controlKey())
77 m_modifiers.add(Modifier::ControlKey);
78 if (webEvent.altKey())
79 m_modifiers.add(Modifier::AltKey);
80 if (webEvent.metaKey())
81 m_modifiers.add(Modifier::MetaKey);
82 if (webEvent.capsLockKey())
83 m_modifiers.add(Modifier::CapsLockKey);
84
85 m_timestamp = webEvent.timestamp();
86
87 // PlatformMouseEvent
88 switch (webEvent.button()) {
89 case WebMouseEvent::NoButton:
90 m_button = WebCore::NoButton;
91 break;
92 case WebMouseEvent::LeftButton:
93 m_button = WebCore::LeftButton;
94 break;
95 case WebMouseEvent::MiddleButton:
96 m_button = WebCore::MiddleButton;
97 break;
98 case WebMouseEvent::RightButton:
99 m_button = WebCore::RightButton;
100 break;
101 default:
102 ASSERT_NOT_REACHED();
103 }
104
105 m_buttons = webEvent.buttons();
106
107 m_position = webEvent.position();
108 m_movementDelta = WebCore::IntPoint(webEvent.deltaX(), webEvent.deltaY());
109 m_globalPosition = webEvent.globalPosition();
110 m_clickCount = webEvent.clickCount();
111#if PLATFORM(MAC)
112 m_eventNumber = webEvent.eventNumber();
113 m_menuTypeForEvent = webEvent.menuTypeForEvent();
114#endif
115 m_modifierFlags = 0;
116 if (webEvent.shiftKey())
117 m_modifierFlags |= static_cast<unsigned>(WebEvent::Modifier::ShiftKey);
118 if (webEvent.controlKey())
119 m_modifierFlags |= static_cast<unsigned>(WebEvent::Modifier::ControlKey);
120 if (webEvent.altKey())
121 m_modifierFlags |= static_cast<unsigned>(WebEvent::Modifier::AltKey);
122 if (webEvent.metaKey())
123 m_modifierFlags |= static_cast<unsigned>(WebEvent::Modifier::MetaKey);
124
125 m_pointerId = webEvent.pointerId();
126 m_pointerType = webEvent.pointerType();
127 }
128};
129
130WebCore::PlatformMouseEvent platform(const WebMouseEvent& webEvent)
131{
132 return WebKit2PlatformMouseEvent(webEvent);
133}

135class WebKit2PlatformWheelEvent : public WebCore::PlatformWheelEvent {
136public:
137 WebKit2PlatformWheelEvent(const WebWheelEvent& webEvent)
138 {
139 // PlatformEvent
140 m_type = PlatformEvent::Wheel;
141
142 if (webEvent.shiftKey())
143 m_modifiers.add(Modifier::ShiftKey);
144 if (webEvent.controlKey())
145 m_modifiers.add(Modifier::ControlKey);
146 if (webEvent.altKey())
147 m_modifiers.add(Modifier::AltKey);
148 if (webEvent.metaKey())
149 m_modifiers.add(Modifier::MetaKey);
150 if (webEvent.capsLockKey())
151 m_modifiers.add(Modifier::CapsLockKey);
152
153 m_timestamp = webEvent.timestamp();
154
155 // PlatformWheelEvent
156 m_position = webEvent.position();
157 m_globalPosition = webEvent.globalPosition();
158 m_deltaX = webEvent.delta().width();
159 m_deltaY = webEvent.delta().height();
160 m_wheelTicksX = webEvent.wheelTicks().width();
161 m_wheelTicksY = webEvent.wheelTicks().height();
162 m_granularity = (webEvent.granularity() == WebWheelEvent::ScrollByPageWheelEvent) ? WebCore::ScrollByPageWheelEvent : WebCore::ScrollByPixelWheelEvent;
163 m_directionInvertedFromDevice = webEvent.directionInvertedFromDevice();
164#if ENABLE(KINETIC_SCROLLING)
165 m_phase = static_cast<WebCore::PlatformWheelEventPhase>(webEvent.phase());
166 m_momentumPhase = static_cast<WebCore::PlatformWheelEventPhase>(webEvent.momentumPhase());
167#endif
168#if PLATFORM(COCOA) || PLATFORM(GTK) || USE(LIBWPE)
169 m_hasPreciseScrollingDeltas = webEvent.hasPreciseScrollingDeltas();
170#endif
171#if PLATFORM(COCOA)
172 m_scrollCount = webEvent.scrollCount();
173 m_unacceleratedScrollingDeltaX = webEvent.unacceleratedScrollingDelta().width();
174 m_unacceleratedScrollingDeltaY = webEvent.unacceleratedScrollingDelta().height();
175#endif
176 }
177};

184class WebKit2PlatformKeyboardEvent : public WebCore::PlatformKeyboardEvent {
185public:
186 WebKit2PlatformKeyboardEvent(const WebKeyboardEvent& webEvent)
187 {
188 // PlatformEvent
189 switch (webEvent.type()) {
190 case WebEvent::KeyDown:
191 m_type = WebCore::PlatformEvent::KeyDown;
192 break;
193 case WebEvent::KeyUp:
194 m_type = WebCore::PlatformEvent::KeyUp;
195 break;
196 case WebEvent::RawKeyDown:
197 m_type = WebCore::PlatformEvent::RawKeyDown;
198 break;
199 case WebEvent::Char:
200 m_type = WebCore::PlatformEvent::Char;
201 break;
202 default:
203 ASSERT_NOT_REACHED();
204 }
205
206 if (webEvent.shiftKey())
207 m_modifiers.add(Modifier::ShiftKey);
208 if (webEvent.controlKey())
209 m_modifiers.add(Modifier::ControlKey);
210 if (webEvent.altKey())
211 m_modifiers.add(Modifier::AltKey);
212 if (webEvent.metaKey())
213 m_modifiers.add(Modifier::MetaKey);
214 if (webEvent.capsLockKey())
215 m_modifiers.add(Modifier::CapsLockKey);
216
217 m_timestamp = webEvent.timestamp();
218
219 // PlatformKeyboardEvent
220 m_text = webEvent.text();
221 m_unmodifiedText = webEvent.unmodifiedText();
222 m_key = webEvent.key();
223 m_code = webEvent.code();
224 m_keyIdentifier = webEvent.keyIdentifier();
225 m_windowsVirtualKeyCode = webEvent.windowsVirtualKeyCode();
226#if USE(APPKIT) || USE(UIKIT_KEYBOARD_ADDITIONS) || PLATFORM(GTK) || USE(LIBWPE)
227 m_handledByInputMethod = webEvent.handledByInputMethod();
228#endif
229#if PLATFORM(GTK) || USE(LIBWPE)
230 m_preeditUnderlines = webEvent.preeditUnderlines();
231 if (auto preeditSelectionRange = webEvent.preeditSelectionRange()) {
232 m_preeditSelectionRangeStart = preeditSelectionRange->location;
233 m_preeditSelectionRangeLength = preeditSelectionRange->length;
234 }
235#endif
236#if USE(APPKIT) || PLATFORM(GTK)
237 m_commands = webEvent.commands();
238#endif
239 m_autoRepeat = webEvent.isAutoRepeat();
240 m_isKeypad = webEvent.isKeypad();
241 m_isSystemKey = webEvent.isSystemKey();
242 }
243};

310 case WebPlatformTouchPoint::TouchStationary:

313 case WebPlatformTouchPoint::TouchCancelled:
314 m_state = PlatformTouchPoint::TouchCancelled;

330class WebKit2PlatformTouchEvent : public WebCore::PlatformTouchEvent {
331public:
332 WebKit2PlatformTouchEvent(const WebTouchEvent& webEvent)
333 {
334 // PlatformEvent
335 switch (webEvent.type()) {
336 case WebEvent::TouchStart:
337 m_type = WebCore::PlatformEvent::TouchStart;
338 break;
339 case WebEvent::TouchMove:
340 m_type = WebCore::PlatformEvent::TouchMove;
341 break;
342 case WebEvent::TouchEnd:
343 m_type = WebCore::PlatformEvent::TouchEnd;
344 break;
345 case WebEvent::TouchCancel:
346 m_type = WebCore::PlatformEvent::TouchCancel;
347 break;
348 default:
349 ASSERT_NOT_REACHED();
350 }
351
352 if (webEvent.shiftKey())
353 m_modifiers.add(Modifier::ShiftKey);
354 if (webEvent.controlKey())
355 m_modifiers.add(Modifier::ControlKey);
356 if (webEvent.altKey())
357 m_modifiers.add(Modifier::AltKey);
358 if (webEvent.metaKey())
359 m_modifiers.add(Modifier::MetaKey);
360 if (webEvent.capsLockKey())
361 m_modifiers.add(Modifier::CapsLockKey);
362
363 m_timestamp = webEvent.timestamp();
364
365#if PLATFORM(IOS_FAMILY)
366 unsigned touchCount = webEvent.touchPoints().size();
367 m_touchPoints.reserveInitialCapacity(touchCount);
368 for (unsigned i = 0; i < touchCount; ++i)
369 m_touchPoints.uncheckedAppend(WebKit2PlatformTouchPoint(webEvent.touchPoints().at(i)));
370
371 m_gestureScale = webEvent.gestureScale();
372 m_gestureRotation = webEvent.gestureRotation();
373 m_canPreventNativeGestures = webEvent.canPreventNativeGestures();
374 m_isGesture = webEvent.isGesture();
375 m_isPotentialTap = webEvent.isPotentialTap();
376 m_position = webEvent.position();
377 m_globalPosition = webEvent.position();
378#else
379 // PlatformTouchEvent
380 for (size_t i = 0; i < webEvent.touchPoints().size(); ++i)
381 m_touchPoints.append(WebKit2PlatformTouchPoint(webEvent.touchPoints().at(i)));
382#endif //PLATFORM(IOS_FAMILY)
383 }
384};

386WebCore::PlatformTouchEvent platform(const WebTouchEvent& webEvent)
387{
388 return WebKit2PlatformTouchEvent(webEvent);
389}
