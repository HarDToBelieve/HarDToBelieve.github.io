27#pragma once
28
29#include "EditingRange.h"
30#include "WebEvent.h"
31#include <WebCore/CompositionUnderline.h>
32
33#if USE(APPKIT)
34#include <WebCore/KeypressCommand.h>
35#endif
36
37namespace WebKit {
38
39class WebKeyboardEvent : public WebEvent {
40public:
41 WebKeyboardEvent();
42 ~WebKeyboardEvent();
43
44#if USE(APPKIT)
45 WebKeyboardEvent(Type, const String& text, const String& unmodifiedText, const String& key, const String& code, const String& keyIdentifier, int windowsVirtualKeyCode, int nativeVirtualKeyCode, int macCharCode, bool handledByInputMethod, const Vector<WebCore::KeypressCommand>&, bool isAutoRepeat, bool isKeypad, bool isSystemKey, OptionSet<Modifier>, WallTime timestamp);
46#elif PLATFORM(GTK)
47 WebKeyboardEvent(Type, const String& text, const String& key, const String& code, const String& keyIdentifier, int windowsVirtualKeyCode, int nativeVirtualKeyCode, bool handledByInputMethod, std::optional<Vector<WebCore::CompositionUnderline>>&&, std::optional<EditingRange>&&, Vector<String>&& commands, bool isKeypad, OptionSet<Modifier>, WallTime timestamp);
48#elif PLATFORM(IOS_FAMILY)
49 WebKeyboardEvent(Type, const String& text, const String& unmodifiedText, const String& key, const String& code, const String& keyIdentifier, int windowsVirtualKeyCode, int nativeVirtualKeyCode, int macCharCode, bool handledByInputMethod, bool isAutoRepeat, bool isKeypad, bool isSystemKey, OptionSet<Modifier>, WallTime timestamp);
50#elif USE(LIBWPE)
51 WebKeyboardEvent(Type, const String& text, const String& key, const String& code, const String& keyIdentifier, int windowsVirtualKeyCode, int nativeVirtualKeyCode, bool handledByInputMethod, std::optional<Vector<WebCore::CompositionUnderline>>&&, std::optional<EditingRange>&&, bool isKeypad, OptionSet<Modifier>, WallTime timestamp);
52#else
53 WebKeyboardEvent(Type, const String& text, const String& unmodifiedText, const String& key, const String& code, const String& keyIdentifier, int windowsVirtualKeyCode, int nativeVirtualKeyCode, int macCharCode, bool isAutoRepeat, bool isKeypad, bool isSystemKey, OptionSet<Modifier>, WallTime timestamp);
54#endif
55
56 const String& text() const { return m_text; }
57 const String& unmodifiedText() const { return m_unmodifiedText; }
58 const String& key() const { return m_key; }
59 const String& code() const { return m_code; }
60 const String& keyIdentifier() const { return m_keyIdentifier; }
61 int32_t windowsVirtualKeyCode() const { return m_windowsVirtualKeyCode; }
62 int32_t nativeVirtualKeyCode() const { return m_nativeVirtualKeyCode; }
63 int32_t macCharCode() const { return m_macCharCode; }
64#if USE(APPKIT) || USE(UIKIT_KEYBOARD_ADDITIONS) || PLATFORM(GTK) || USE(LIBWPE)
65 bool handledByInputMethod() const { return m_handledByInputMethod; }
66#endif
67#if PLATFORM(GTK) || USE(LIBWPE)
68 const std::optional<Vector<WebCore::CompositionUnderline>>& preeditUnderlines() const { return m_preeditUnderlines; }
69 const std::optional<EditingRange>& preeditSelectionRange() const { return m_preeditSelectionRange; }
70#endif
71#if USE(APPKIT)
72 const Vector<WebCore::KeypressCommand>& commands() const { return m_commands; }
73#elif PLATFORM(GTK)
74 const Vector<String>& commands() const { return m_commands; }
75#endif
76 bool isAutoRepeat() const { return m_isAutoRepeat; }
77 bool isKeypad() const { return m_isKeypad; }
78 bool isSystemKey() const { return m_isSystemKey; }
79
80 void encode(IPC::Encoder&) const;
81 static WARN_UNUSED_RETURN bool decode(IPC::Decoder&, WebKeyboardEvent&);
82
83 static bool isKeyboardEventType(Type);
84
85private:
86 String m_text;
87 String m_unmodifiedText;
88 String m_key;
89 String m_code;
90 String m_keyIdentifier;
91 int32_t m_windowsVirtualKeyCode { 0 };
92 int32_t m_nativeVirtualKeyCode { 0 };
93 int32_t m_macCharCode { 0 };
94#if USE(APPKIT) || USE(UIKIT_KEYBOARD_ADDITIONS) || PLATFORM(GTK) || USE(LIBWPE)
95 bool m_handledByInputMethod { false };
96#endif
97#if PLATFORM(GTK) || USE(LIBWPE)
98 std::optional<Vector<WebCore::CompositionUnderline>> m_preeditUnderlines;
99 std::optional<EditingRange> m_preeditSelectionRange;
100#endif
101#if USE(APPKIT)
102 Vector<WebCore::KeypressCommand> m_commands;
103#elif PLATFORM(GTK)
104 Vector<String> m_commands;
105#endif
106 bool m_isAutoRepeat { false };
107 bool m_isKeypad { false };
108 bool m_isSystemKey { false };
109};
110
111} // namespace WebKit
