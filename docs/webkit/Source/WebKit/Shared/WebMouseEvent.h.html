27#pragma once
28
29// FIXME: We should probably move to making the WebCore/PlatformFooEvents trivial classes so that
30// we can use them as the event type.
31
32#include "WebEvent.h"
33#include <WebCore/IntPoint.h>
34#include <WebCore/PointerEventTypeNames.h>
35#include <WebCore/PointerID.h>
36
37namespace WebKit {
38
39enum class GestureWasCancelled : bool { No, Yes };
40
41class WebMouseEvent : public WebEvent {
42public:
43 enum Button {
44 LeftButton = 0,
45 MiddleButton,
46 RightButton,
47 NoButton = -2
48 };
49
50 enum SyntheticClickType { NoTap, OneFingerTap, TwoFingerTap };
51
52 WebMouseEvent();
53
54#if PLATFORM(MAC)
55 WebMouseEvent(Type, Button, unsigned short buttons, const WebCore::IntPoint& positionInView, const WebCore::IntPoint& globalPosition, float deltaX, float deltaY, float deltaZ, int clickCount, OptionSet<Modifier>, WallTime timestamp, double force, SyntheticClickType = NoTap, int eventNumber = -1, int menuType = 0, GestureWasCancelled = GestureWasCancelled::No);
56#else
57 WebMouseEvent(Type, Button, unsigned short buttons, const WebCore::IntPoint& positionInView, const WebCore::IntPoint& globalPosition, float deltaX, float deltaY, float deltaZ, int clickCount, OptionSet<Modifier>, WallTime timestamp, double force = 0, SyntheticClickType = NoTap, WebCore::PointerID = WebCore::mousePointerID, const String& pointerType = WebCore::mousePointerEventType(), GestureWasCancelled = GestureWasCancelled::No);
58#endif
59
60 Button button() const { return static_cast<Button>(m_button); }
61 unsigned short buttons() const { return m_buttons; }
62 const WebCore::IntPoint& position() const { return m_position; } // Relative to the view.
63 const WebCore::IntPoint& globalPosition() const { return m_globalPosition; }
64 float deltaX() const { return m_deltaX; }
65 float deltaY() const { return m_deltaY; }
66 float deltaZ() const { return m_deltaZ; }
67 int32_t clickCount() const { return m_clickCount; }
68#if PLATFORM(MAC)
69 int32_t eventNumber() const { return m_eventNumber; }
70 int32_t menuTypeForEvent() const { return m_menuTypeForEvent; }
71#endif
72 double force() const { return m_force; }
73 SyntheticClickType syntheticClickType() const { return static_cast<SyntheticClickType>(m_syntheticClickType); }
74 WebCore::PointerID pointerId() const { return m_pointerId; }
75 const String& pointerType() const { return m_pointerType; }
76 GestureWasCancelled gestureWasCancelled() const { return m_gestureWasCancelled; }
77
78 void encode(IPC::Encoder&) const;
79 static WARN_UNUSED_RETURN bool decode(IPC::Decoder&, WebMouseEvent&);
80
81private:
82 static bool isMouseEventType(Type);
83
84 uint32_t m_button { static_cast<uint32_t>(NoButton) };
85 unsigned short m_buttons { 0 };
86 WebCore::IntPoint m_position; // Relative to the view.
87 WebCore::IntPoint m_globalPosition;
88 float m_deltaX { 0 };
89 float m_deltaY { 0 };
90 float m_deltaZ { 0 };
91 int32_t m_clickCount { 0 };
92#if PLATFORM(MAC)
93 int32_t m_eventNumber { -1 };
94 int32_t m_menuTypeForEvent { 0 };
95#endif
96 double m_force { 0 };
97 uint32_t m_syntheticClickType { NoTap };
98 WebCore::PointerID m_pointerId { WebCore::mousePointerID };
99 String m_pointerType { WebCore::mousePointerEventType() };
100 GestureWasCancelled m_gestureWasCancelled { GestureWasCancelled::No };
101};
102
103} // namespace WebKit
