31namespace WebKit {
32
33void WebPageGroupData::encode(IPC::Encoder& encoder) const
34{
35 encoder << identifier;
36 encoder << pageGroupID;
37}
38
39std::optional<WebPageGroupData> WebPageGroupData::decode(IPC::Decoder& decoder)
40{
41 std::optional<String> identifier;
42 decoder >> identifier;
43 if (!identifier)
44 return std::nullopt;
45
46 std::optional<uint64_t> pageGroupID;
47 decoder >> pageGroupID;
48 if (!pageGroupID)
49 return std::nullopt;
50
51 return {{ WTFMove(*identifier), *pageGroupID }};
52}
