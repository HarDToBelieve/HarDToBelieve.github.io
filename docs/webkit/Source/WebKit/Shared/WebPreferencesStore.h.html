26#pragma once
27
28#include "Decoder.h"
29#include "Encoder.h"
30#include <wtf/RobinHoodHashMap.h>
31#include <wtf/text/StringHash.h>
32#include <wtf/text/WTFString.h>
33
34namespace WebKit {
35
36struct WebPreferencesStore {
37 WebPreferencesStore();
38
39 void encode(IPC::Encoder&) const;
40 static WARN_UNUSED_RETURN bool decode(IPC::Decoder&, WebPreferencesStore&);
41
42 // NOTE: The getters in this class have non-standard names to aid in the use of the preference macros.
43
44 bool setStringValueForKey(const String& key, const String& value);
45 String getStringValueForKey(const String& key) const;
46
47 bool setBoolValueForKey(const String& key, bool value);
48 bool getBoolValueForKey(const String& key) const;
49
50 bool setUInt32ValueForKey(const String& key, uint32_t value);
51 uint32_t getUInt32ValueForKey(const String& key) const;
52
53 bool setDoubleValueForKey(const String& key, double value);
54 double getDoubleValueForKey(const String& key) const;
55
56 void setOverrideDefaultsStringValueForKey(const String& key, String value);
57 void setOverrideDefaultsBoolValueForKey(const String& key, bool value);
58 void setOverrideDefaultsUInt32ValueForKey(const String& key, uint32_t value);
59 void setOverrideDefaultsDoubleValueForKey(const String& key, double value);
60
61 void deleteKey(const String& key);
62
63 // For WebKitTestRunner usage.
64 static void overrideBoolValueForKey(const String& key, bool value);
65 static void removeTestRunnerOverrides();
66
67 using Value = Variant<String, bool, uint32_t, double>;
68
69 using ValueMap = MemoryCompactRobinHoodHashMap<String, Value>;
70 ValueMap m_values;
71 ValueMap m_overriddenDefaults;
72
73 static ValueMap& defaults();
74};
75
76} // namespace WebKit
