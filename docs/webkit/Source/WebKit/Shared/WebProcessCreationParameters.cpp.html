213bool WebProcessCreationParameters::decode(IPC::Decoder& decoder, WebProcessCreationParameters& parameters)
214{
215 if (!decoder.decode(parameters.injectedBundlePath))
216 return false;
217
218 std::optional<SandboxExtension::Handle> injectedBundlePathExtensionHandle;
219 decoder >> injectedBundlePathExtensionHandle;
220 if (!injectedBundlePathExtensionHandle)
221 return false;
222 parameters.injectedBundlePathExtensionHandle = WTFMove(*injectedBundlePathExtensionHandle);
223
224 std::optional<SandboxExtension::HandleArray> additionalSandboxExtensionHandles;
225 decoder >> additionalSandboxExtensionHandles;
226 if (!additionalSandboxExtensionHandles)
227 return false;
228 parameters.additionalSandboxExtensionHandles = WTFMove(*additionalSandboxExtensionHandles);
229 if (!decoder.decode(parameters.initializationUserData))
230 return false;
231
232#if PLATFORM(IOS_FAMILY)
233
234 std::optional<SandboxExtension::Handle> cookieStorageDirectoryExtensionHandle;
235 decoder >> cookieStorageDirectoryExtensionHandle;
236 if (!cookieStorageDirectoryExtensionHandle)
237 return false;
238 parameters.cookieStorageDirectoryExtensionHandle = WTFMove(*cookieStorageDirectoryExtensionHandle);
239
240 std::optional<SandboxExtension::Handle> containerCachesDirectoryExtensionHandle;
241 decoder >> containerCachesDirectoryExtensionHandle;
242 if (!containerCachesDirectoryExtensionHandle)
243 return false;
244 parameters.containerCachesDirectoryExtensionHandle = WTFMove(*containerCachesDirectoryExtensionHandle);
245
246 std::optional<SandboxExtension::Handle> containerTemporaryDirectoryExtensionHandle;
247 decoder >> containerTemporaryDirectoryExtensionHandle;
248 if (!containerTemporaryDirectoryExtensionHandle)
249 return false;
250 parameters.containerTemporaryDirectoryExtensionHandle = WTFMove(*containerTemporaryDirectoryExtensionHandle);
251
252#endif
253#if PLATFORM(COCOA) && ENABLE(REMOTE_INSPECTOR)
254 std::optional<SandboxExtension::Handle> enableRemoteWebInspectorExtensionHandle;
255 decoder >> enableRemoteWebInspectorExtensionHandle;
256 if (!enableRemoteWebInspectorExtensionHandle)
257 return false;
258 parameters.enableRemoteWebInspectorExtensionHandle = WTFMove(*enableRemoteWebInspectorExtensionHandle);
259#endif
260
261 if (!decoder.decode(parameters.webCoreLoggingChannels))
262 return false;
263 if (!decoder.decode(parameters.webKitLoggingChannels))
264 return false;
265
266#if ENABLE(MEDIA_STREAM)
267
268 std::optional<SandboxExtension::Handle> audioCaptureExtensionHandle;
269 decoder >> audioCaptureExtensionHandle;
270 if (!audioCaptureExtensionHandle)
271 return false;
272 parameters.audioCaptureExtensionHandle = WTFMove(*audioCaptureExtensionHandle);
273#endif
274 if (!decoder.decode(parameters.urlSchemesRegisteredAsEmptyDocument))
275 return false;
276 if (!decoder.decode(parameters.urlSchemesRegisteredAsSecure))
277 return false;
278 if (!decoder.decode(parameters.urlSchemesRegisteredAsBypassingContentSecurityPolicy))
279 return false;
280 if (!decoder.decode(parameters.urlSchemesForWhichDomainRelaxationIsForbidden))
281 return false;
282 if (!decoder.decode(parameters.urlSchemesRegisteredAsLocal))
283 return false;
284 if (!decoder.decode(parameters.urlSchemesRegisteredAsNoAccess))
285 return false;
