31namespace WebKit {
32using namespace WebCore;
33
34WebSQLiteDatabaseTracker::WebSQLiteDatabaseTracker(IsHoldingLockedFilesHandler&& isHoldingLockedFilesHandler)
35 : m_isHoldingLockedFilesHandler(WTFMove(isHoldingLockedFilesHandler))
36{
37 ASSERT(RunLoop::isMain());
38 SQLiteDatabaseTracker::setClient(this);
39}
40
41WebSQLiteDatabaseTracker::~WebSQLiteDatabaseTracker()
42{
43 ASSERT(RunLoop::isMain());
44 SQLiteDatabaseTracker::setClient(nullptr);
45
46 if (m_currentHystererisID)
47 setIsHoldingLockedFiles(false);
48}
49
50void WebSQLiteDatabaseTracker::setIsSuspended(bool isSuspended)
51{
52 ASSERT(RunLoop::isMain());
53
54 Locker locker { m_lock };
55 m_isSuspended = isSuspended;
56}
57
58void WebSQLiteDatabaseTracker::willBeginFirstTransaction()
59{
60 Locker locker { m_lock };
61 if (m_currentHystererisID) {
62 // Cancel previous hysteresis task.
63 ++m_currentHystererisID;
64 return;
65 }
66
67 setIsHoldingLockedFiles(true);
68}
69
70void WebSQLiteDatabaseTracker::didFinishLastTransaction()
71{
72 Locker locker { m_lock };
73 RunLoop::main().dispatchAfter(1_s, [this, weakThis = makeWeakPtr(*this), hystererisID = ++m_currentHystererisID] {
74 if (!weakThis)
75 return;
76
77 Locker locker { m_lock };
78 if (m_currentHystererisID != hystererisID)
79 return; // Cancelled.
80
81 m_currentHystererisID = 0;
82 setIsHoldingLockedFiles(false);
83 });
84}
85
86void WebSQLiteDatabaseTracker::setIsHoldingLockedFiles(bool isHoldingLockedFiles)
87{
88 if (m_isSuspended)
89 return;
90
91 m_isHoldingLockedFilesHandler(isHoldingLockedFiles);
92}
