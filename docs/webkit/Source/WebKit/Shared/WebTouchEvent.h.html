37// FIXME: Having "Platform" in the name makes it sound like this event is platform-specific or
38// low-level in some way. That doesn't seem to be the case.
39class WebPlatformTouchPoint {
40public:
41 enum TouchPointState {
42 TouchReleased,
43 TouchPressed,
44 TouchMoved,
45 TouchStationary,
46 TouchCancelled
47 };
48
49 enum class TouchType {
50 Direct,
51 Stylus
52 };
53
54 WebPlatformTouchPoint() = default;
55 WebPlatformTouchPoint(unsigned identifier, WebCore::IntPoint location, TouchPointState phase)
56 : m_identifier(identifier)
57 , m_location(location)
58 , m_phase(phase)
59 {
60 }
61
62 unsigned identifier() const { return m_identifier; }
63 WebCore::IntPoint location() const { return m_location; }
64 TouchPointState phase() const { return static_cast<TouchPointState>(m_phase); }
65 TouchPointState state() const { return phase(); }
66
67#if ENABLE(IOS_TOUCH_EVENTS)
68 void setRadiusX(double radiusX) { m_radiusX = radiusX; }
69 double radiusX() const { return m_radiusX; }
70 void setRadiusY(double radiusY) { m_radiusY = radiusY; }
71 double radiusY() const { return m_radiusY; }
72 void setRotationAngle(double rotationAngle) { m_rotationAngle = rotationAngle; }
73 double rotationAngle() const { return m_rotationAngle; }
74 void setForce(double force) { m_force = force; }
75 double force() const { return m_force; }
76 void setAltitudeAngle(double altitudeAngle) { m_altitudeAngle = altitudeAngle; }
77 double altitudeAngle() const { return m_altitudeAngle; }
78 void setAzimuthAngle(double azimuthAngle) { m_azimuthAngle = azimuthAngle; }
79 double azimuthAngle() const { return m_azimuthAngle; }
80 void setTouchType(TouchType touchType) { m_touchType = static_cast<uint32_t>(touchType); }
81 TouchType touchType() const { return static_cast<TouchType>(m_touchType); }
82#endif
83
84 void encode(IPC::Encoder&) const;
85 static std::optional<WebPlatformTouchPoint> decode(IPC::Decoder&);
86
87private:
88 unsigned m_identifier { 0 };
89 WebCore::IntPoint m_location;
90 uint32_t m_phase { TouchReleased };
91#if ENABLE(IOS_TOUCH_EVENTS)
92 double m_radiusX { 0 };
93 double m_radiusY { 0 };
94 double m_rotationAngle { 0 };
95 double m_force { 0 };
96 double m_altitudeAngle { 0 };
97 double m_azimuthAngle { 0 };
98 uint32_t m_touchType { static_cast<uint32_t>(TouchType::Direct) };
99#endif
100};
101
102class WebTouchEvent : public WebEvent {
103public:
104 WebTouchEvent() = default;
105 WebTouchEvent(WebEvent::Type type, OptionSet<Modifier> modifiers, WallTime timestamp, const Vector<WebPlatformTouchPoint>& touchPoints, WebCore::IntPoint position, bool isPotentialTap, bool isGesture, float gestureScale, float gestureRotation)
106 : WebEvent(type, modifiers, timestamp)
107 , m_touchPoints(touchPoints)
108 , m_position(position)
109 , m_canPreventNativeGestures(true)
110 , m_isPotentialTap(isPotentialTap)
111 , m_isGesture(isGesture)
112 , m_gestureScale(gestureScale)
113 , m_gestureRotation(gestureRotation)
114 {
115 ASSERT(type == TouchStart || type == TouchMove || type == TouchEnd || type == TouchCancel);
116 }
117
118 const Vector<WebPlatformTouchPoint>& touchPoints() const { return m_touchPoints; }
119
120 WebCore::IntPoint position() const { return m_position; }
121
122 bool isPotentialTap() const { return m_isPotentialTap; }
123
124 bool isGesture() const { return m_isGesture; }
125 float gestureScale() const { return m_gestureScale; }
126 float gestureRotation() const { return m_gestureRotation; }
127
128 bool canPreventNativeGestures() const { return m_canPreventNativeGestures; }
129 void setCanPreventNativeGestures(bool canPreventNativeGestures) { m_canPreventNativeGestures = canPreventNativeGestures; }
130
131 bool allTouchPointsAreReleased() const;
132
133 void encode(IPC::Encoder&) const;
134 static WARN_UNUSED_RETURN bool decode(IPC::Decoder&, WebTouchEvent&);
135
136private:
137 Vector<WebPlatformTouchPoint> m_touchPoints;
138
139 WebCore::IntPoint m_position;
140 bool m_canPreventNativeGestures { false };
141 bool m_isPotentialTap { false };
142 bool m_isGesture { false };
143 float m_gestureScale { 0 };
144 float m_gestureRotation { 0 };
145};

149class WebPlatformTouchPoint {
150public:
151 enum TouchPointState {
152 TouchReleased,
153 TouchPressed,
154 TouchMoved,
155 TouchStationary,
156 TouchCancelled
157 };
158
159 WebPlatformTouchPoint()
160 : m_rotationAngle(0.0), m_force(0.0) { }
161
162 WebPlatformTouchPoint(uint32_t id, TouchPointState, const WebCore::IntPoint& screenPosition, const WebCore::IntPoint& position);
163
164 WebPlatformTouchPoint(uint32_t id, TouchPointState, const WebCore::IntPoint& screenPosition, const WebCore::IntPoint& position, const WebCore::IntSize& radius, float rotationAngle = 0.0, float force = 0.0);
165
166 uint32_t id() const { return m_id; }
167 TouchPointState state() const { return static_cast<TouchPointState>(m_state); }
168
169 const WebCore::IntPoint& screenPosition() const { return m_screenPosition; }
170 const WebCore::IntPoint& position() const { return m_position; }
171 const WebCore::IntSize& radius() const { return m_radius; }
172 float rotationAngle() const { return m_rotationAngle; }
173 float force() const { return m_force; }
174
175 void setState(TouchPointState state) { m_state = state; }
176
177 void encode(IPC::Encoder&) const;
178 static std::optional<WebPlatformTouchPoint> decode(IPC::Decoder&);
179
180private:
181 uint32_t m_id;
182 uint32_t m_state;
183 WebCore::IntPoint m_screenPosition;
184 WebCore::IntPoint m_position;
185 WebCore::IntSize m_radius;
186 float m_rotationAngle;
187 float m_force;
188};

190class WebTouchEvent : public WebEvent {
191public:
192 WebTouchEvent() { }
193 WebTouchEvent(Type, Vector<WebPlatformTouchPoint>&&, OptionSet<Modifier>, WallTime timestamp);
194
195 const Vector<WebPlatformTouchPoint>& touchPoints() const { return m_touchPoints; }
196
197 bool allTouchPointsAreReleased() const;
198
199 void encode(IPC::Encoder&) const;
200 static WARN_UNUSED_RETURN bool decode(IPC::Decoder&, WebTouchEvent&);
201
202private:
203 static bool isTouchEventType(Type);
204
205 Vector<WebPlatformTouchPoint> m_touchPoints;
206};
