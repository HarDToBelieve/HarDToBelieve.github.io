79void WebWheelEvent::encode(IPC::Encoder& encoder) const
80{
81 WebEvent::encode(encoder);
82
83 encoder << m_position;
84 encoder << m_globalPosition;
85 encoder << m_delta;
86 encoder << m_wheelTicks;
87 encoder << m_granularity;
88 encoder << m_directionInvertedFromDevice;
89#if PLATFORM(COCOA) || PLATFORM(GTK) || USE(LIBWPE)
90 encoder << m_phase;
91 encoder << m_momentumPhase;
92 encoder << m_hasPreciseScrollingDeltas;
93#endif
94#if PLATFORM(COCOA)
95 encoder << m_scrollCount;
96 encoder << m_unacceleratedScrollingDelta;
97#endif
98}

100bool WebWheelEvent::decode(IPC::Decoder& decoder, WebWheelEvent& t)
101{
102 if (!WebEvent::decode(decoder, t))
103 return false;
104 if (!decoder.decode(t.m_position))
105 return false;
106 if (!decoder.decode(t.m_globalPosition))
107 return false;
108 if (!decoder.decode(t.m_delta))
109 return false;
110 if (!decoder.decode(t.m_wheelTicks))
111 return false;
112 if (!decoder.decode(t.m_granularity))
113 return false;
114 if (!decoder.decode(t.m_directionInvertedFromDevice))
115 return false;
116#if PLATFORM(COCOA) || PLATFORM(GTK) || USE(LIBWPE)
117 if (!decoder.decode(t.m_phase))
118 return false;
119 if (!decoder.decode(t.m_momentumPhase))
120 return false;
121 if (!decoder.decode(t.m_hasPreciseScrollingDeltas))
122 return false;
123#endif
124#if PLATFORM(COCOA)
125 if (!decoder.decode(t.m_scrollCount))
126 return false;
127 if (!decoder.decode(t.m_unacceleratedScrollingDelta))
128 return false;
129#endif
130 return true;
131}

133bool WebWheelEvent::isWheelEventType(Type type)
134{
135 return type == Wheel;
136}
