27#pragma once
28
29#include "WebEvent.h"
30#include <WebCore/FloatSize.h>
31#include <WebCore/IntPoint.h>
32
33namespace WebKit {
34
35class WebWheelEvent : public WebEvent {
36public:
37 enum Granularity {
38 ScrollByPageWheelEvent,
39 ScrollByPixelWheelEvent
40 };
41
42 enum Phase {
43 PhaseNone = 0,
44 PhaseBegan = 1 << 0,
45 PhaseStationary = 1 << 1,
46 PhaseChanged = 1 << 2,
47 PhaseEnded = 1 << 3,
48 PhaseCancelled = 1 << 4,
49 PhaseMayBegin = 1 << 5,
50 };
51
52 WebWheelEvent() = default;
53
54 WebWheelEvent(Type, const WebCore::IntPoint& position, const WebCore::IntPoint& globalPosition, const WebCore::FloatSize& delta, const WebCore::FloatSize& wheelTicks, Granularity, OptionSet<Modifier>, WallTime timestamp);
55#if PLATFORM(COCOA)
56 WebWheelEvent(Type, const WebCore::IntPoint& position, const WebCore::IntPoint& globalPosition, const WebCore::FloatSize& delta, const WebCore::FloatSize& wheelTicks, Granularity, bool directionInvertedFromDevice, Phase, Phase momentumPhase, bool hasPreciseScrollingDeltas, uint32_t scrollCount, const WebCore::FloatSize& unacceleratedScrollingDelta, OptionSet<Modifier>, WallTime timestamp);
57#elif PLATFORM(GTK) || USE(LIBWPE)
58 WebWheelEvent(Type, const WebCore::IntPoint& position, const WebCore::IntPoint& globalPosition, const WebCore::FloatSize& delta, const WebCore::FloatSize& wheelTicks, Phase, Phase momentumPhase, Granularity, bool hasPreciseScrollingDeltas, OptionSet<Modifier>, WallTime timestamp);
59#endif
60
61 const WebCore::IntPoint position() const { return m_position; }
62 const WebCore::IntPoint globalPosition() const { return m_globalPosition; }
63 const WebCore::FloatSize delta() const { return m_delta; }
64 const WebCore::FloatSize wheelTicks() const { return m_wheelTicks; }
65 Granularity granularity() const { return static_cast<Granularity>(m_granularity); }
66 bool directionInvertedFromDevice() const { return m_directionInvertedFromDevice; }
67 Phase phase() const { return static_cast<Phase>(m_phase); }
68 Phase momentumPhase() const { return static_cast<Phase>(m_momentumPhase); }
69#if PLATFORM(COCOA) || PLATFORM(GTK) || USE(LIBWPE)
70 bool hasPreciseScrollingDeltas() const { return m_hasPreciseScrollingDeltas; }
71#endif
72#if PLATFORM(COCOA)
73 uint32_t scrollCount() const { return m_scrollCount; }
74 const WebCore::FloatSize& unacceleratedScrollingDelta() const { return m_unacceleratedScrollingDelta; }
75#endif
76
77 void encode(IPC::Encoder&) const;
78 static WARN_UNUSED_RETURN bool decode(IPC::Decoder&, WebWheelEvent&);
79
80private:
81 static bool isWheelEventType(Type);
82
83 WebCore::IntPoint m_position;
84 WebCore::IntPoint m_globalPosition;
85 WebCore::FloatSize m_delta;
86 WebCore::FloatSize m_wheelTicks;
87 uint32_t m_granularity { ScrollByPageWheelEvent };
88 uint32_t m_phase { Phase::PhaseNone };
89 uint32_t m_momentumPhase { Phase::PhaseNone };
90 bool m_directionInvertedFromDevice { false };
91#if PLATFORM(COCOA) || PLATFORM(GTK) || USE(LIBWPE)
92 bool m_hasPreciseScrollingDeltas { false };
93#endif
94#if PLATFORM(COCOA)
95 uint32_t m_scrollCount { 0 };
96 WebCore::FloatSize m_unacceleratedScrollingDelta;
97#endif
98};
99
100} // namespace WebKit
