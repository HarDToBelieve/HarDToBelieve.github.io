26#pragma once
27
28#include "NativeWebWheelEvent.h"
29#include <wtf/Deque.h>
30#include <wtf/FastMalloc.h>
31#include <wtf/WallTime.h>
32
33namespace WebKit {
34
35class WebWheelEventCoalescer {
36 WTF_MAKE_FAST_ALLOCATED;
37public:
38 // If this returns true, use nextEventToDispatch() to get the event to dispatch.
39 bool shouldDispatchEvent(const NativeWebWheelEvent&);
40 std::optional<WebWheelEvent> nextEventToDispatch();
41
42 NativeWebWheelEvent takeOldestEventBeingProcessed();
43
44 bool hasEventsBeingProcessed() const { return !m_eventsBeingProcessed.isEmpty(); }
45
46 bool shouldCoalesceEventsDuringDeceleration() const { return m_shouldCoalesceEventsDuringDeceleration; }
47 void setShouldCoalesceEventsDuringDeceleration(bool shouldCoalsce) { m_shouldCoalesceEventsDuringDeceleration = shouldCoalsce; }
48
49 void clear();
50
51private:
52 using CoalescedEventSequence = Vector<NativeWebWheelEvent>;
53
54 static bool canCoalesce(const WebWheelEvent&, const WebWheelEvent&);
55 static WebWheelEvent coalesce(const WebWheelEvent&, const WebWheelEvent&);
56
57 static bool isMomentumPhaseEvent(const WebWheelEvent&);
58
59 bool shouldDispatchEventNow(const WebWheelEvent&) const;
60
61 Deque<NativeWebWheelEvent, 2> m_wheelEventQueue;
62 Deque<std::unique_ptr<CoalescedEventSequence>> m_eventsBeingProcessed;
63
64 WallTime m_lastEventTime;
65 WallTime m_lastDispatchedEventTime;
66 bool m_shouldCoalesceEventsDuringDeceleration { false };
67};
68
69} // namespace WebKit
