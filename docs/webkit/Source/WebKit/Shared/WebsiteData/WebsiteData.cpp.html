39{
40 encoder << origin;
41 encoder << type;
42 encoder << size;
43}
44
45auto WebsiteData::Entry::decode(IPC::Decoder& decoder) -> std::optional<Entry>
46{
47 Entry result;
48
49 std::optional<WebCore::SecurityOriginData> securityOriginData;
50 decoder >> securityOriginData;
51 if (!securityOriginData)
52 return std::nullopt;
53 result.origin = WTFMove(*securityOriginData);
54
55 if (!decoder.decode(result.type))
56 return std::nullopt;
57
58 if (!decoder.decode(result.size))
59 return std::nullopt;
60
61 return result;
62}

64void WebsiteData::encode(IPC::Encoder& encoder) const
65{
66 encoder << entries;
67 encoder << hostNamesWithCookies;
68#if ENABLE(NETSCAPE_PLUGIN_API)
69 encoder << hostNamesWithPluginData;
70#endif
71 encoder << hostNamesWithHSTSCache;
72#if ENABLE(RESOURCE_LOAD_STATISTICS)
73 encoder << registrableDomainsWithResourceLoadStatistics;
74#endif
75}
76
77bool WebsiteData::decode(IPC::Decoder& decoder, WebsiteData& result)
78{
79 if (!decoder.decode(result.entries))
80 return false;
81 if (!decoder.decode(result.hostNamesWithCookies))
82 return false;
83#if ENABLE(NETSCAPE_PLUGIN_API)
84 if (!decoder.decode(result.hostNamesWithPluginData))
85 return false;
86#endif
87 if (!decoder.decode(result.hostNamesWithHSTSCache))
88 return false;
89#if ENABLE(RESOURCE_LOAD_STATISTICS)
90 if (!decoder.decode(result.registrableDomainsWithResourceLoadStatistics))
91 return false;
92#endif
93 return true;
94}

96WebsiteDataProcessType WebsiteData::ownerProcess(WebsiteDataType dataType)
97{
98 switch (dataType) {
99 case WebsiteDataType::Cookies:
100 return WebsiteDataProcessType::Network;
101 case WebsiteDataType::DiskCache:
102 return WebsiteDataProcessType::Network;
103 case WebsiteDataType::MemoryCache:
104 return WebsiteDataProcessType::Web;
105 case WebsiteDataType::OfflineWebApplicationCache:
106 return WebsiteDataProcessType::UI;
107 case WebsiteDataType::SessionStorage:
108 return WebsiteDataProcessType::Network;
109 case WebsiteDataType::LocalStorage:
110 return WebsiteDataProcessType::Network;
111 case WebsiteDataType::WebSQLDatabases:
112 return WebsiteDataProcessType::UI;
113 case WebsiteDataType::IndexedDBDatabases:
114 return WebsiteDataProcessType::Network;
115 case WebsiteDataType::MediaKeys:
116 return WebsiteDataProcessType::UI;
117 case WebsiteDataType::HSTSCache:
118 return WebsiteDataProcessType::Network;
119 case WebsiteDataType::SearchFieldRecentSearches:
120 return WebsiteDataProcessType::UI;
121#if ENABLE(NETSCAPE_PLUGIN_API)
122 case WebsiteDataType::PlugInData:
123 return WebsiteDataProcessType::UI;
124#endif
125 case WebsiteDataType::ResourceLoadStatistics:
126 return WebsiteDataProcessType::Network;
127 case WebsiteDataType::Credentials:
128 return WebsiteDataProcessType::Network;
129#if ENABLE(SERVICE_WORKER)
130 case WebsiteDataType::ServiceWorkerRegistrations:
131 return WebsiteDataProcessType::Network;
132#endif
133 case WebsiteDataType::DOMCache:
134 return WebsiteDataProcessType::Network;
135 case WebsiteDataType::DeviceIdHashSalt:
136 return WebsiteDataProcessType::UI;
137 case WebsiteDataType::PrivateClickMeasurements:
138 return WebsiteDataProcessType::Network;
139#if HAVE(CFNETWORK_ALTERNATIVE_SERVICE)
140 case WebsiteDataType::AlternativeServices:
141 return WebsiteDataProcessType::Network;
142#endif
143 }
144
145 RELEASE_ASSERT_NOT_REACHED();
146}

148OptionSet<WebsiteDataType> WebsiteData::filter(OptionSet<WebsiteDataType> unfilteredWebsiteDataTypes, WebsiteDataProcessType WebsiteDataProcessType)
149{
150 OptionSet<WebsiteDataType> filtered;
151 for (auto dataType : unfilteredWebsiteDataTypes) {
152 if (ownerProcess(dataType) == WebsiteDataProcessType)
153 filtered.add(dataType);
154 }
155
156 return filtered;
157}

160{
161 return WebsiteData {
162 crossThreadCopy(entries),
163 crossThreadCopy(hostNamesWithCookies),
164#if ENABLE(NETSCAPE_PLUGIN_API)
165 crossThreadCopy(hostNamesWithPluginData),
166#endif
167 crossThreadCopy(hostNamesWithHSTSCache),
168#if ENABLE(RESOURCE_LOAD_STATISTICS)
169 crossThreadCopy(registrableDomainsWithResourceLoadStatistics),
170#endif
171 };
172}
