36void WebsitePoliciesData::encode(IPC::Encoder& encoder) const
37{
38 encoder << contentBlockersEnabled;
39 encoder << autoplayPolicy;
40#if ENABLE(DEVICE_ORIENTATION)
41 encoder << deviceOrientationAndMotionAccessState;
42#endif
43 encoder << allowedAutoplayQuirks;
44 encoder << customHeaderFields;
45 encoder << popUpPolicy;
46 encoder << customUserAgent;
47 encoder << customUserAgentAsSiteSpecificQuirks;
48 encoder << customNavigatorPlatform;
49 encoder << metaViewportPolicy;
50 encoder << mediaSourcePolicy;
51 encoder << simulatedMouseEventsDispatchPolicy;
52 encoder << legacyOverflowScrollingTouchPolicy;
53 encoder << allowContentChangeObserverQuirk;
54 encoder << allowsContentJavaScript;
55 encoder << mouseEventPolicy;
56 encoder << idempotentModeAutosizingOnlyHonorsPercentages;
57}

59std::optional<WebsitePoliciesData> WebsitePoliciesData::decode(IPC::Decoder& decoder)
60{
61 std::optional<bool> contentBlockersEnabled;
62 decoder >> contentBlockersEnabled;
63 if (!contentBlockersEnabled)
64 return std::nullopt;
65
66 std::optional<WebsiteAutoplayPolicy> autoplayPolicy;
67 decoder >> autoplayPolicy;
68 if (!autoplayPolicy)
69 return std::nullopt;
70
71#if ENABLE(DEVICE_ORIENTATION)
72 std::optional<WebCore::DeviceOrientationOrMotionPermissionState> deviceOrientationAndMotionAccessState;
73 decoder >> deviceOrientationAndMotionAccessState;
74 if (!deviceOrientationAndMotionAccessState)
75 return std::nullopt;
76#endif
77
78 std::optional<OptionSet<WebsiteAutoplayQuirk>> allowedAutoplayQuirks;
79 decoder >> allowedAutoplayQuirks;
80 if (!allowedAutoplayQuirks)
81 return std::nullopt;
82
83 std::optional<Vector<WebCore::CustomHeaderFields>> customHeaderFields;
84 decoder >> customHeaderFields;
85 if (!customHeaderFields)
86 return std::nullopt;
87
88 std::optional<WebsitePopUpPolicy> popUpPolicy;
89 decoder >> popUpPolicy;
90 if (!popUpPolicy)
91 return std::nullopt;
92
93 std::optional<String> customUserAgent;
94 decoder >> customUserAgent;
95 if (!customUserAgent)
96 return std::nullopt;
97
98 std::optional<String> customUserAgentAsSiteSpecificQuirks;
99 decoder >> customUserAgentAsSiteSpecificQuirks;
100 if (!customUserAgentAsSiteSpecificQuirks)
101 return std::nullopt;
102
103 std::optional<String> customNavigatorPlatform;
104 decoder >> customNavigatorPlatform;
105 if (!customNavigatorPlatform)
106 return std::nullopt;
107
108 std::optional<WebsiteMetaViewportPolicy> metaViewportPolicy;
109 decoder >> metaViewportPolicy;
110 if (!metaViewportPolicy)
111 return std::nullopt;
112
113 std::optional<WebsiteMediaSourcePolicy> mediaSourcePolicy;
114 decoder >> mediaSourcePolicy;
115 if (!mediaSourcePolicy)
116 return std::nullopt;
117
118 std::optional<WebsiteSimulatedMouseEventsDispatchPolicy> simulatedMouseEventsDispatchPolicy;
119 decoder >> simulatedMouseEventsDispatchPolicy;
120 if (!simulatedMouseEventsDispatchPolicy)
121 return std::nullopt;
122
123 std::optional<WebsiteLegacyOverflowScrollingTouchPolicy> legacyOverflowScrollingTouchPolicy;
124 decoder >> legacyOverflowScrollingTouchPolicy;
125 if (!legacyOverflowScrollingTouchPolicy)
126 return std::nullopt;
127
128 std::optional<bool> allowContentChangeObserverQuirk;
129 decoder >> allowContentChangeObserverQuirk;
130 if (!allowContentChangeObserverQuirk)
131 return std::nullopt;
132
133 std::optional<WebCore::AllowsContentJavaScript> allowsContentJavaScript;
134 decoder >> allowsContentJavaScript;
135 if (!allowsContentJavaScript)
136 return std::nullopt;
137
138 std::optional<WebCore::MouseEventPolicy> mouseEventPolicy;
139 decoder >> mouseEventPolicy;
140 if (!mouseEventPolicy)
141 return std::nullopt;
142
143 std::optional<bool> idempotentModeAutosizingOnlyHonorsPercentages;
144 decoder >> idempotentModeAutosizingOnlyHonorsPercentages;
145 if (!idempotentModeAutosizingOnlyHonorsPercentages)
146 return std::nullopt;
147
148 return { {
149 WTFMove(*contentBlockersEnabled),
150 WTFMove(*allowedAutoplayQuirks),
151 WTFMove(*autoplayPolicy),
152#if ENABLE(DEVICE_ORIENTATION)
153 WTFMove(*deviceOrientationAndMotionAccessState),
154#endif
155 WTFMove(*customHeaderFields),
156 WTFMove(*popUpPolicy),
157 WTFMove(*customUserAgent),
158 WTFMove(*customUserAgentAsSiteSpecificQuirks),
159 WTFMove(*customNavigatorPlatform),
160 WTFMove(*metaViewportPolicy),
161 WTFMove(*mediaSourcePolicy),
162 WTFMove(*simulatedMouseEventsDispatchPolicy),
163 WTFMove(*legacyOverflowScrollingTouchPolicy),
164 WTFMove(*allowContentChangeObserverQuirk),
165 WTFMove(*allowsContentJavaScript),
166 WTFMove(*mouseEventPolicy),
167 WTFMove(*idempotentModeAutosizingOnlyHonorsPercentages),
168 } };
169}

171void WebsitePoliciesData::applyToDocumentLoader(WebsitePoliciesData&& websitePolicies, WebCore::DocumentLoader& documentLoader)
172{
173 documentLoader.setCustomHeaderFields(WTFMove(websitePolicies.customHeaderFields));
174 documentLoader.setCustomUserAgent(websitePolicies.customUserAgent);
175 documentLoader.setCustomUserAgentAsSiteSpecificQuirks(websitePolicies.customUserAgentAsSiteSpecificQuirks);
176 documentLoader.setCustomNavigatorPlatform(websitePolicies.customNavigatorPlatform);
177
178#if ENABLE(DEVICE_ORIENTATION)
179 documentLoader.setDeviceOrientationAndMotionAccessState(websitePolicies.deviceOrientationAndMotionAccessState);
180#endif
181
182 // Only setUserContentExtensionsEnabled if it hasn't already been disabled by reloading without content blockers.
183 if (documentLoader.userContentExtensionsEnabled())
184 documentLoader.setUserContentExtensionsEnabled(websitePolicies.contentBlockersEnabled);
185
186 OptionSet<WebCore::AutoplayQuirk> quirks;
187 const auto& allowedQuirks = websitePolicies.allowedAutoplayQuirks;
188
189 if (allowedQuirks.contains(WebsiteAutoplayQuirk::InheritedUserGestures))
190 quirks.add(WebCore::AutoplayQuirk::InheritedUserGestures);
191
192 if (allowedQuirks.contains(WebsiteAutoplayQuirk::SynthesizedPauseEvents))
193 quirks.add(WebCore::AutoplayQuirk::SynthesizedPauseEvents);
194
195 if (allowedQuirks.contains(WebsiteAutoplayQuirk::ArbitraryUserGestures))
196 quirks.add(WebCore::AutoplayQuirk::ArbitraryUserGestures);
197
198 if (allowedQuirks.contains(WebsiteAutoplayQuirk::PerDocumentAutoplayBehavior))
199 quirks.add(WebCore::AutoplayQuirk::PerDocumentAutoplayBehavior);
200
201 documentLoader.setAllowedAutoplayQuirks(quirks);
202
203 switch (websitePolicies.autoplayPolicy) {
204 case WebsiteAutoplayPolicy::Default:
205 documentLoader.setAutoplayPolicy(WebCore::AutoplayPolicy::Default);
206 break;
207 case WebsiteAutoplayPolicy::Allow:
208 documentLoader.setAutoplayPolicy(WebCore::AutoplayPolicy::Allow);
209 break;
210 case WebsiteAutoplayPolicy::AllowWithoutSound:
211 documentLoader.setAutoplayPolicy(WebCore::AutoplayPolicy::AllowWithoutSound);
212 break;
213 case WebsiteAutoplayPolicy::Deny:
214 documentLoader.setAutoplayPolicy(WebCore::AutoplayPolicy::Deny);
215 break;
216 }
217
218 switch (websitePolicies.popUpPolicy) {
219 case WebsitePopUpPolicy::Default:
220 documentLoader.setPopUpPolicy(WebCore::PopUpPolicy::Default);
221 break;
222 case WebsitePopUpPolicy::Allow:
223 documentLoader.setPopUpPolicy(WebCore::PopUpPolicy::Allow);
224 break;
225 case WebsitePopUpPolicy::Block:
226 documentLoader.setPopUpPolicy(WebCore::PopUpPolicy::Block);
227 break;
228 }
229
230 switch (websitePolicies.metaViewportPolicy) {
231 case WebsiteMetaViewportPolicy::Default:
232 documentLoader.setMetaViewportPolicy(WebCore::MetaViewportPolicy::Default);
233 break;
234 case WebsiteMetaViewportPolicy::Respect:
235 documentLoader.setMetaViewportPolicy(WebCore::MetaViewportPolicy::Respect);
236 break;
237 case WebsiteMetaViewportPolicy::Ignore:
238 documentLoader.setMetaViewportPolicy(WebCore::MetaViewportPolicy::Ignore);
239 break;
240 }
241
242 switch (websitePolicies.mediaSourcePolicy) {
243 case WebsiteMediaSourcePolicy::Default:
244 documentLoader.setMediaSourcePolicy(WebCore::MediaSourcePolicy::Default);
245 break;
246 case WebsiteMediaSourcePolicy::Disable:
247 documentLoader.setMediaSourcePolicy(WebCore::MediaSourcePolicy::Disable);
248 break;
249 case WebsiteMediaSourcePolicy::Enable:
250 documentLoader.setMediaSourcePolicy(WebCore::MediaSourcePolicy::Enable);
251 break;
252 }
253
254 switch (websitePolicies.simulatedMouseEventsDispatchPolicy) {
255 case WebsiteSimulatedMouseEventsDispatchPolicy::Default:
256 documentLoader.setSimulatedMouseEventsDispatchPolicy(WebCore::SimulatedMouseEventsDispatchPolicy::Default);
257 break;
258 case WebsiteSimulatedMouseEventsDispatchPolicy::Allow:
259 documentLoader.setSimulatedMouseEventsDispatchPolicy(WebCore::SimulatedMouseEventsDispatchPolicy::Allow);
260 break;
261 case WebsiteSimulatedMouseEventsDispatchPolicy::Deny:
262 documentLoader.setSimulatedMouseEventsDispatchPolicy(WebCore::SimulatedMouseEventsDispatchPolicy::Deny);
263 break;
264 }
265
266 switch (websitePolicies.legacyOverflowScrollingTouchPolicy) {
267 case WebsiteLegacyOverflowScrollingTouchPolicy::Default:
268 documentLoader.setLegacyOverflowScrollingTouchPolicy(WebCore::LegacyOverflowScrollingTouchPolicy::Default);
269 break;
270 case WebsiteLegacyOverflowScrollingTouchPolicy::Disable:
271 documentLoader.setLegacyOverflowScrollingTouchPolicy(WebCore::LegacyOverflowScrollingTouchPolicy::Disable);
272 break;
273 case WebsiteLegacyOverflowScrollingTouchPolicy::Enable:
274 documentLoader.setLegacyOverflowScrollingTouchPolicy(WebCore::LegacyOverflowScrollingTouchPolicy::Enable);
275 break;
276 }
277
278 switch (websitePolicies.mouseEventPolicy) {
279 case WebCore::MouseEventPolicy::Default:
280 documentLoader.setMouseEventPolicy(WebCore::MouseEventPolicy::Default);
281 break;
282#if ENABLE(IOS_TOUCH_EVENTS)
283 case WebCore::MouseEventPolicy::SynthesizeTouchEvents:
284 documentLoader.setMouseEventPolicy(WebCore::MouseEventPolicy::SynthesizeTouchEvents);
285 break;
286#endif
287 }
288
289 documentLoader.setAllowContentChangeObserverQuirk(websitePolicies.allowContentChangeObserverQuirk);
290 documentLoader.setIdempotentModeAutosizingOnlyHonorsPercentages(websitePolicies.idempotentModeAutosizingOnlyHonorsPercentages);
291
292 auto* frame = documentLoader.frame();
293 if (!frame)
294 return;
295
296 documentLoader.applyPoliciesToSettings();
297}
