82static CFType typeFromCFTypeRef(CFTypeRef type)
83{
84 ASSERT(type);
85
86 if (type == tokenNullptrTypeRef())
87 return CFType::Nullptr;
88
89 CFTypeID typeID = CFGetTypeID(type);
90 if (typeID == CFArrayGetTypeID())
91 return CFType::CFArray;
92 if (typeID == CFBooleanGetTypeID())
93 return CFType::CFBoolean;
94 if (typeID == CFDataGetTypeID())
95 return CFType::CFData;
96 if (typeID == CFDateGetTypeID())
97 return CFType::CFDate;
98 if (typeID == CFDictionaryGetTypeID())
99 return CFType::CFDictionary;
100 if (typeID == CFNullGetTypeID())
101 return CFType::CFNull;
102 if (typeID == CFNumberGetTypeID())
103 return CFType::CFNumber;
104 if (typeID == CFStringGetTypeID())
105 return CFType::CFString;
106 if (typeID == CFURLGetTypeID())
107 return CFType::CFURL;
108 if (typeID == CGColorSpaceGetTypeID())
109 return CFType::CGColorSpace;
110 if (typeID == SecCertificateGetTypeID())
111 return CFType::SecCertificate;
112#if HAVE(SEC_KEYCHAIN)
113 if (typeID == SecKeychainItemGetTypeID())
114 return CFType::SecKeychainItem;
115#endif
116#if HAVE(SEC_ACCESS_CONTROL)
117 if (typeID == SecAccessControlGetTypeID())
118 return CFType::SecAccessControl;
119#endif
120#if HAVE(SEC_TRUST_SERIALIZATION)
121 if (typeID == SecTrustGetTypeID())
122 return CFType::SecTrust;
123#endif
124
125 ASSERT_NOT_REACHED();
126 return CFType::Unknown;
127}

130template<typename Encoder>
131void ArgumentCoder<CFTypeRef>::encode(Encoder& encoder, CFTypeRef typeRef)
132{
133 auto type = typeFromCFTypeRef(typeRef);
134 encoder << type;
135
136 switch (type) {
137 case CFType::CFArray:
138 encoder << static_cast<CFArrayRef>(typeRef);
139 return;
140 case CFType::CFBoolean:
141 encoder << static_cast<CFBooleanRef>(typeRef);
142 return;
143 case CFType::CFData:
144 encoder << static_cast<CFDataRef>(typeRef);
145 return;
146 case CFType::CFDate:
147 encoder << static_cast<CFDateRef>(typeRef);
148 return;
149 case CFType::CFDictionary:
150 encoder << static_cast<CFDictionaryRef>(typeRef);
151 return;
152 case CFType::CFNull:
153 return;
154 case CFType::CFNumber:
155 encoder << static_cast<CFNumberRef>(typeRef);
156 return;
157 case CFType::CFString:
158 encoder << static_cast<CFStringRef>(typeRef);
159 return;
160 case CFType::CFURL:
161 encoder << static_cast<CFURLRef>(typeRef);
162 return;
163 case CFType::CGColorSpace:
164 encoder << static_cast<CGColorSpaceRef>(const_cast<void*>(typeRef));
165 return;
166 case CFType::SecCertificate:
167 encoder << static_cast<SecCertificateRef>(const_cast<void*>(typeRef));
168 return;
169#if HAVE(SEC_KEYCHAIN)
170 case CFType::SecKeychainItem:
171 encoder << static_cast<SecKeychainItemRef>(const_cast<void*>(typeRef));
172 return;
173#endif
174#if HAVE(SEC_ACCESS_CONTROL)
175 case CFType::SecAccessControl:
176 encoder << static_cast<SecAccessControlRef>(const_cast<void*>(typeRef));
177 return;
178#endif
179#if HAVE(SEC_TRUST_SERIALIZATION)
180 case CFType::SecTrust:
181 encoder << static_cast<SecTrustRef>(const_cast<void*>(typeRef));
182 return;
183#endif
184 case CFType::Nullptr:
185 return;
186 case CFType::Unknown:
187 break;
188 }
189
190 ASSERT_NOT_REACHED();
191}

196bool ArgumentCoder<RetainPtr<CFTypeRef>>::decode(Decoder& decoder, RetainPtr<CFTypeRef>& result)
197{
198 std::optional<CFType> type;
199 decoder >> type;
200 if (!type)
201 return false;
202
203 switch (*type) {
204 case CFType::CFArray: {
205 RetainPtr<CFArrayRef> array;
206 if (!decoder.decode(array))
207 return false;
208 result = WTFMove(array);
209 return true;
210 }
211 case CFType::CFBoolean: {
212 RetainPtr<CFBooleanRef> boolean;
213 if (!decoder.decode(boolean))
214 return false;
215 result = WTFMove(boolean);
216 return true;
217 }
218 case CFType::CFData: {
219 RetainPtr<CFDataRef> data;
220 if (!decoder.decode(data))
221 return false;
222 result = WTFMove(data);
223 return true;
224 }
225 case CFType::CFDate: {
226 RetainPtr<CFDateRef> date;
227 if (!decoder.decode(date))
228 return false;
229 result = WTFMove(date);
230 return true;
231 }
232 case CFType::CFDictionary: {
233 RetainPtr<CFDictionaryRef> dictionary;
234 if (!decoder.decode(dictionary))
235 return false;
236 result = WTFMove(dictionary);
237 return true;
238 }
239 case CFType::CFNull:
240 result = adoptCF(kCFNull);
241 return true;
242 case CFType::CFNumber: {
243 RetainPtr<CFNumberRef> number;
244 if (!decoder.decode(number))
245 return false;
246 result = WTFMove(number);
247 return true;
248 }
249 case CFType::CFString: {
250 RetainPtr<CFStringRef> string;
251 if (!decoder.decode(string))
252 return false;
253 result = WTFMove(string);
254 return true;
255 }
256 case CFType::CFURL: {
257 RetainPtr<CFURLRef> url;
258 if (!decoder.decode(url))
259 return false;
260 result = WTFMove(url);
261 return true;
262 }
263 case CFType::CGColorSpace: {
264 RetainPtr<CGColorSpaceRef> colorSpace;
265 if (!decoder.decode(colorSpace))
266 return false;
267 result = WTFMove(colorSpace);
268 return true;
269 }
270 case CFType::SecCertificate: {
271 RetainPtr<SecCertificateRef> certificate;
272 if (!decoder.decode(certificate))
273 return false;
274 result = WTFMove(certificate);
275 return true;
276 }
277#if HAVE(SEC_KEYCHAIN)
278 case CFType::SecKeychainItem: {
279 RetainPtr<SecKeychainItemRef> keychainItem;
280 if (!decoder.decode(keychainItem))
281 return false;
282 result = WTFMove(keychainItem);
283 return true;
284 }
285#endif
286#if HAVE(SEC_ACCESS_CONTROL)
287 case CFType::SecAccessControl: {
288 RetainPtr<SecAccessControlRef> accessControl;
289 if (!decoder.decode(accessControl))
290 return false;
291 result = WTFMove(accessControl);
292 return true;
293 }
294#endif
295#if HAVE(SEC_TRUST_SERIALIZATION)
296 case CFType::SecTrust: {
297 RetainPtr<SecTrustRef> trust;
298 if (!decoder.decode(trust))
299 return false;
300 result = WTFMove(trust);
301 return true;
302 }
303#endif
304 case CFType::Nullptr:
305 result = tokenNullptrTypeRef();
306 return true;
307 case CFType::Unknown:
308 break;
309 }
310
311 ASSERT_NOT_REACHED();
312 return false;
313}

315template<typename Encoder>
316void ArgumentCoder<CFArrayRef>::encode(Encoder& encoder, CFArrayRef array)
317{
318 if (!array) {
319 encoder << true;
320 return;
321 }
322
323 encoder << false;
324
325 CFIndex size = CFArrayGetCount(array);
326 Vector<CFTypeRef, 32> values(size);
327
328 CFArrayGetValues(array, CFRangeMake(0, size), values.data());
329
330 HashSet<CFIndex> invalidIndicies;
331 for (CFIndex i = 0; i < size; ++i) {
332 // Ignore values we don't support.
333 ASSERT(typeFromCFTypeRef(values[i]) != CFType::Unknown);
334 if (typeFromCFTypeRef(values[i]) == CFType::Unknown)
335 invalidIndicies.add(i);
336 }
337
338 encoder << static_cast<uint64_t>(size - invalidIndicies.size());
339
340 for (CFIndex i = 0; i < size; ++i) {
341 if (invalidIndicies.contains(i))
342 continue;
343 encoder << values[i];
344 }
345}

350bool ArgumentCoder<RetainPtr<CFArrayRef>>::decode(Decoder& decoder, RetainPtr<CFArrayRef>& result)
351{
352 bool isNull = false;
353 if (!decoder.decode(isNull))
354 return false;
355
356 if (isNull) {
357 result = nullptr;
358 return true;
359 }
360
361 uint64_t size;
362 if (!decoder.decode(size))
363 return false;
364
365 auto array = adoptCF(CFArrayCreateMutable(0, 0, &kCFTypeArrayCallBacks));
366
367 for (size_t i = 0; i < size; ++i) {
368 RetainPtr<CFTypeRef> element;
369 if (!decoder.decode(element))
370 return false;
371
372 if (!element)
373 continue;
374
375 CFArrayAppendValue(array.get(), element.get());
376 }
377
378 result = adoptCF(array.leakRef());
379 return true;
380}

439template<typename Encoder>
440void ArgumentCoder<CFDictionaryRef>::encode(Encoder& encoder, CFDictionaryRef dictionary)
441{
442 if (!dictionary) {
443 encoder << true;
444 return;
445 }
446
447 encoder << false;
448
449 CFIndex size = CFDictionaryGetCount(dictionary);
450 Vector<CFTypeRef, 32> keys(size);
451 Vector<CFTypeRef, 32> values(size);
452
453 CFDictionaryGetKeysAndValues(dictionary, keys.data(), values.data());
454
455 HashSet<CFTypeRef> invalidKeys;
456 for (CFIndex i = 0; i < size; ++i) {
457 ASSERT(keys[i]);
458 ASSERT(values[i]);
459
460 // Ignore keys/values we don't support.
461 ASSERT(typeFromCFTypeRef(keys[i]) != CFType::Unknown);
462 ASSERT(typeFromCFTypeRef(values[i]) != CFType::Unknown);
463 if (typeFromCFTypeRef(keys[i]) == CFType::Unknown || typeFromCFTypeRef(values[i]) == CFType::Unknown)
464 invalidKeys.add(keys[i]);
465 }
466
467 encoder << static_cast<uint64_t>(size - invalidKeys.size());
468
469 for (CFIndex i = 0; i < size; ++i) {
470 if (invalidKeys.contains(keys[i]))
471 continue;
472
473 encoder << keys[i] << values[i];
474 }
475}

480bool ArgumentCoder<RetainPtr<CFDictionaryRef>>::decode(Decoder& decoder, RetainPtr<CFDictionaryRef>& result)
481{
482 bool isNull = false;
483 if (!decoder.decode(isNull))
484 return false;
485
486 if (isNull) {
487 result = nullptr;
488 return true;
489 }
490
491 uint64_t size;
492 if (!decoder.decode(size))
493 return false;
494
495 RetainPtr<CFMutableDictionaryRef> dictionary = adoptCF(CFDictionaryCreateMutable(0, 0, &kCFTypeDictionaryKeyCallBacks, &kCFTypeDictionaryValueCallBacks));
496 for (uint64_t i = 0; i < size; ++i) {
497 RetainPtr<CFTypeRef> key;
498 if (!decoder.decode(key) || !key)
499 return false;
500
501 RetainPtr<CFTypeRef> value;
502 if (!decoder.decode(value) || !value)
503 return false;
504
505 CFDictionarySetValue(dictionary.get(), key.get(), value.get());
506 }
507
508 result = WTFMove(dictionary);
509 return true;
510}

527static size_t sizeForNumberType(CFNumberType numberType)
528{
529 switch (numberType) {
530 case kCFNumberSInt8Type:
531 return sizeof(SInt8);
532 case kCFNumberSInt16Type:
533 return sizeof(SInt16);
534 case kCFNumberSInt32Type:
535 return sizeof(SInt32);
536 case kCFNumberSInt64Type:
537 return sizeof(SInt64);
538 case kCFNumberFloat32Type:
539 return sizeof(Float32);
540 case kCFNumberFloat64Type:
541 return sizeof(Float64);
542 case kCFNumberCharType:
543 return sizeof(char);
544 case kCFNumberShortType:
545 return sizeof(short);
546 case kCFNumberIntType:
547 return sizeof(int);
548 case kCFNumberLongType:
549 return sizeof(long);
550 case kCFNumberLongLongType:
551 return sizeof(long long);
552 case kCFNumberFloatType:
553 return sizeof(float);
554 case kCFNumberDoubleType:
555 return sizeof(double);
556 case kCFNumberCFIndexType:
557 return sizeof(CFIndex);
558 case kCFNumberNSIntegerType:
559 return sizeof(long);
560 case kCFNumberCGFloatType:
561 return sizeof(double);
562 }
563
564 return 0;
565}

567bool ArgumentCoder<RetainPtr<CFNumberRef>>::decode(Decoder& decoder, RetainPtr<CFNumberRef>& result)
568{
569 std::optional<uint8_t> numberTypeFromIPC;
570 decoder >> numberTypeFromIPC;
571 if (!numberTypeFromIPC || *numberTypeFromIPC > kCFNumberMaxType)
572 return false;
573 auto numberType = static_cast<CFNumberType>(*numberTypeFromIPC);
574
575 IPC::DataReference dataReference;
576 if (!decoder.decode(dataReference))
577 return false;
578
579 size_t neededBufferSize = sizeForNumberType(numberType);
580 if (!neededBufferSize || dataReference.size() != neededBufferSize)
581 return false;
582
583 ASSERT(dataReference.data());
584 result = adoptCF(CFNumberCreate(0, numberType, dataReference.data()));
585
586 return true;
587}

589template<typename Encoder>
590void ArgumentCoder<CFStringRef>::encode(Encoder& encoder, CFStringRef string)
591{
592 CFIndex length = CFStringGetLength(string);
593 CFStringEncoding encoding = CFStringGetFastestEncoding(string);
594
595 CFRange range = CFRangeMake(0, length);
596 CFIndex bufferLength = 0;
597
598 CFIndex numConvertedBytes = CFStringGetBytes(string, range, encoding, 0, false, 0, 0, &bufferLength);
599 ASSERT(numConvertedBytes == length);
600
601 Vector<UInt8, 128> buffer(bufferLength);
602 numConvertedBytes = CFStringGetBytes(string, range, encoding, 0, false, buffer.data(), buffer.size(), &bufferLength);
603 ASSERT(numConvertedBytes == length);
604
605 encoder << static_cast<uint32_t>(encoding) << IPC::DataReference(buffer);
606}

611bool ArgumentCoder<RetainPtr<CFStringRef>>::decode(Decoder& decoder, RetainPtr<CFStringRef>& result)
612{
613 std::optional<uint32_t> encodingFromIPC;
614 decoder >> encodingFromIPC;
615 if (!encodingFromIPC)
616 return false;
617 auto encoding = static_cast<CFStringEncoding>(*encodingFromIPC);
618
619 if (!CFStringIsEncodingAvailable(encoding))
620 return false;
621
622 IPC::DataReference dataReference;
623 if (!decoder.decode(dataReference))
624 return false;
625
626 auto string = adoptCF(CFStringCreateWithBytes(0, dataReference.data(), dataReference.size(), encoding, false));
627 if (!string)
628 return false;
629
630 result = WTFMove(string);
631 return true;
632}

634template<typename Encoder>
635void ArgumentCoder<CFURLRef>::encode(Encoder& encoder, CFURLRef url)
636{
637 CFURLRef baseURL = CFURLGetBaseURL(url);
638 encoder << static_cast<bool>(baseURL);
639 if (baseURL)
640 encoder << baseURL;
641
642 WTF::URLCharBuffer urlBytes;
643 WTF::getURLBytes(url, urlBytes);
644 encoder << IPC::DataReference(reinterpret_cast<const uint8_t*>(urlBytes.data()), urlBytes.size());
645}

647template void ArgumentCoder<CFURLRef>::encode<Encoder>(Encoder&, CFURLRef);
648template void ArgumentCoder<CFURLRef>::encode<StreamConnectionEncoder>(StreamConnectionEncoder&, CFURLRef);

650bool ArgumentCoder<RetainPtr<CFURLRef>>::decode(Decoder& decoder, RetainPtr<CFURLRef>& result)
651{
652 RetainPtr<CFURLRef> baseURL;
653 bool hasBaseURL;
654 if (!decoder.decode(hasBaseURL))
655 return false;
656 if (hasBaseURL) {
657 if (!decoder.decode(baseURL))
658 return false;
659 }
660
661 IPC::DataReference urlBytes;
662 if (!decoder.decode(urlBytes))
663 return false;
664
665#if USE(FOUNDATION)
666 // FIXME: Move this to ArgumentCodersCFMac.mm and change this file back to be C++
667 // instead of Objective-C++.
668 if (urlBytes.isEmpty()) {
669 // CFURL can't hold an empty URL, unlike NSURL.
670 // FIXME: This discards base URL, which seems incorrect.
671 result = (__bridge CFURLRef)[NSURL URLWithString:@""];
672 return true;
673 }
674#endif
675
676 result = adoptCF(CFURLCreateAbsoluteURLWithBytes(nullptr, reinterpret_cast<const UInt8*>(urlBytes.data()), urlBytes.size(), kCFStringEncodingUTF8, baseURL.get(), true));
677 return result;
678}

683void ArgumentCoder<CGColorSpaceRef>::encode(Encoder& encoder, CGColorSpaceRef colorSpace)
684{
685 if (auto name = CGColorSpaceGetName(colorSpace)) {

690 encoder << CGColorSpaceEncodingScheme::PropertyList << adoptCF(CGColorSpaceCopyPropertyList(colorSpace));

693template void ArgumentCoder<CGColorSpaceRef>::encode<Encoder>(Encoder&, CGColorSpaceRef);
694template void ArgumentCoder<CGColorSpaceRef>::encode<StreamConnectionEncoder>(StreamConnectionEncoder&, CGColorSpaceRef);

696bool ArgumentCoder<RetainPtr<CGColorSpaceRef>>::decode(Decoder& decoder, RetainPtr<CGColorSpaceRef>& result)
697{
698 CGColorSpaceEncodingScheme encodingScheme;
699 if (!decoder.decode(encodingScheme))
700 return false;
701
702 switch (encodingScheme) {
703 case CGColorSpaceEncodingScheme::Name: {
704 RetainPtr<CFStringRef> name;
705 if (!decoder.decode(name))
706 return false;
707
708 auto colorSpace = adoptCF(CGColorSpaceCreateWithName(name.get()));
709 if (!colorSpace)
710 return false;
711
712 result = WTFMove(colorSpace);
713 return true;
714 }
715 case CGColorSpaceEncodingScheme::PropertyList: {
716 RetainPtr<CFTypeRef> propertyList;
717 if (!decoder.decode(propertyList))
718 return false;
719
720 auto colorSpace = adoptCF(CGColorSpaceCreateWithPropertyList(propertyList.get()));
721 if (!colorSpace)
722 return false;
723
724 result = WTFMove(colorSpace);
725 return true;
726 }
727 }
728
729 ASSERT_NOT_REACHED();
730 return false;
731}

772
773 RetainPtr<CFDataRef> data;
774 if (!decoder.decode(data))
775 return false;
776
777 CFDataRef dref = data.get();
778 // SecKeychainItemCopyFromPersistentReference() cannot handle 0-length CFDataRefs.
779 if (!CFDataGetLength(dref))
780 return false;
781
782 SecKeychainItemRef item;
783 if (SecKeychainItemCopyFromPersistentReference(dref, &item) != errSecSuccess || !item)
784 return false;
785
786 result = adoptCF(item);
787 return true;
788}
789#endif

794{
795 auto data = adoptCF(SecAccessControlCopyData(accessControl));
796 if (data)
797 encoder << data;
798}

804{
805 RetainPtr<CFDataRef> data;
806 if (!decoder.decode(data))
807 return false;
808
809 result = adoptCF(SecAccessControlCreateFromData(kCFAllocatorDefault, data.get(), nullptr));
810 if (!result)
811 return false;
812
813 return true;
814}

822 if (!data) {
823 encoder << false;
824 return;
825 }
826
827 encoder << true << data;
828}

830template void ArgumentCoder<SecTrustRef>::encode<Encoder>(Encoder&, SecTrustRef);
831template void ArgumentCoder<SecTrustRef>::encode<StreamConnectionEncoder>(StreamConnectionEncoder&, SecTrustRef);
832
833bool ArgumentCoder<RetainPtr<SecTrustRef>>::decode(Decoder& decoder, RetainPtr<SecTrustRef>& result)
834{
835 bool hasTrust;
836 if (!decoder.decode(hasTrust))
837 return false;
838
839 if (!hasTrust)
840 return true;
841
842 RetainPtr<CFDataRef> trustData;
843 if (!decoder.decode(trustData))
844 return false;
845
846 auto trust = adoptCF(SecTrustDeserialize(trustData.get(), nullptr));
847 if (!trust)
848 return false;
849
850 result = WTFMove(trust);
851 return true;
852}
