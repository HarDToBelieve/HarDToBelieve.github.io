41static CGColorSpaceRef colorSpace(const ShareableBitmap::Configuration& configuration)
42{
43 return configuration.colorSpace ? configuration.colorSpace->platformColorSpace() : WebCore::sRGBColorSpaceRef();
44}
45
46static bool wantsExtendedRange(const ShareableBitmap::Configuration& configuration)
47{
48 return CGColorSpaceUsesExtendedRange(colorSpace(configuration));
49}
50
51static CGBitmapInfo bitmapInfo(const ShareableBitmap::Configuration& configuration)
52{
53 CGBitmapInfo info = 0;
54 if (wantsExtendedRange(configuration)) {
55 info |= kCGBitmapFloatComponents | kCGBitmapByteOrder16Host;
56
57 if (configuration.isOpaque)
58 info |= kCGImageAlphaNoneSkipLast;
59 else
60 info |= kCGImageAlphaPremultipliedLast;
61 } else {
62 info |= kCGBitmapByteOrder32Host;
63
64 if (configuration.isOpaque)
65 info |= kCGImageAlphaNoneSkipFirst;
66 else
67 info |= kCGImageAlphaPremultipliedFirst;
68 }
69
70 return info;
71}

73CheckedUint32 ShareableBitmap::calculateBytesPerRow(WebCore::IntSize size, const Configuration& configuration)
74{
75 CheckedUint32 bytesPerRow = calculateBytesPerPixel(configuration) * size.width();
76#if HAVE(IOSURFACE)
77 if (bytesPerRow.hasOverflowed())
78 return bytesPerRow;
79 return IOSurfaceAlignProperty(kIOSurfaceBytesPerRow, bytesPerRow);
80#else
81 return bytesPerRow;
82#endif
83}
84
85CheckedUint32 ShareableBitmap::calculateBytesPerPixel(const Configuration& configuration)
86{
87 return wantsExtendedRange(configuration) ? 8 : 4;
88}

90std::unique_ptr<GraphicsContext> ShareableBitmap::createGraphicsContext()
91{
92 auto bitsPerComponent = calculateBytesPerPixel(m_configuration) * 8 / 4;
93 if (bitsPerComponent.hasOverflowed())
94 return nullptr;
95
96 auto bytesPerRow = calculateBytesPerRow(m_size, m_configuration);
97 if (bytesPerRow.hasOverflowed())
98 return nullptr;
99
100 RetainPtr<CGContextRef> bitmapContext = adoptCF(CGBitmapContextCreateWithData(data(), m_size.width(), m_size.height(), bitsPerComponent, bytesPerRow, colorSpace(m_configuration), bitmapInfo(m_configuration), releaseBitmapContextData, this));
101 if (!bitmapContext)
102 return nullptr;
103
104 ref(); // Balanced by deref in releaseBitmapContextData.
105
106 // We want the origin to be in the top left corner so we flip the backing store context.
107 CGContextTranslateCTM(bitmapContext.get(), 0, m_size.height());
108 CGContextScaleCTM(bitmapContext.get(), 1, -1);
109
110 return makeUnique<GraphicsContextCG>(bitmapContext.get());
111}

113void ShareableBitmap::paint(WebCore::GraphicsContext& context, const IntPoint& destination, const IntRect& source)
114{
115 paint(context, 1, destination, source);
116}

118void ShareableBitmap::paint(WebCore::GraphicsContext& context, float scaleFactor, const IntPoint& destination, const IntRect& source)
119{
120 CGContextRef cgContext = context.platformContext();
121 CGContextSaveGState(cgContext);
122
123 CGContextClipToRect(cgContext, CGRectMake(destination.x(), destination.y(), source.width(), source.height()));
124 CGContextScaleCTM(cgContext, 1, -1);
125
126 auto image = makeCGImageCopy();
127 CGFloat imageHeight = CGImageGetHeight(image.get()) / scaleFactor;
128 CGFloat imageWidth = CGImageGetWidth(image.get()) / scaleFactor;
129
130 CGFloat destX = destination.x() - source.x();
131 CGFloat destY = -imageHeight - destination.y() + source.y();
132
133 CGContextDrawImage(cgContext, CGRectMake(destX, destY, imageWidth, imageHeight), image.get());
134
135 CGContextRestoreGState(cgContext);
136}

138RetainPtr<CGImageRef> ShareableBitmap::makeCGImageCopy()
139{
140 auto graphicsContext = createGraphicsContext();
141 if (!graphicsContext)
142 return nullptr;
143
144 return adoptCF(CGBitmapContextCreateImage(graphicsContext->platformContext()));
145}

147RetainPtr<CGImageRef> ShareableBitmap::makeCGImage()
148{
149 ref(); // Balanced by deref in releaseDataProviderData.
150 verifyImageBufferIsBigEnough(data(), sizeInBytes());
151 RetainPtr<CGDataProvider> dataProvider = adoptCF(CGDataProviderCreateWithData(this, data(), sizeInBytes(), releaseDataProviderData));
152 return createCGImage(dataProvider.get());
153}
154
155RetainPtr<CGImageRef> ShareableBitmap::createCGImage(CGDataProviderRef dataProvider) const
156{
157 ASSERT_ARG(dataProvider, dataProvider);
158 auto bitsPerPixel = calculateBytesPerPixel(m_configuration) * 8;
159 if (bitsPerPixel.hasOverflowed())
160 return nullptr;
161
162 auto bytesPerRow = calculateBytesPerRow(m_size, m_configuration);
163 if (bytesPerRow.hasOverflowed())
164 return nullptr;
165
166 RetainPtr<CGImageRef> image = adoptCF(CGImageCreate(m_size.width(), m_size.height(), bitsPerPixel / 4, bitsPerPixel, bytesPerRow, colorSpace(m_configuration), bitmapInfo(m_configuration), dataProvider, 0, false, kCGRenderingIntentDefault));
167 return image;
168}

170void ShareableBitmap::releaseBitmapContextData(void* typelessBitmap, void* typelessData)
171{
172 ShareableBitmap* bitmap = static_cast<ShareableBitmap*>(typelessBitmap);
173 ASSERT_UNUSED(typelessData, bitmap->data() == typelessData);
174 bitmap->deref(); // Balanced by ref in createGraphicsContext.
175}
176
177void ShareableBitmap::releaseDataProviderData(void* typelessBitmap, const void* typelessData, size_t)
178{
179 ShareableBitmap* bitmap = static_cast<ShareableBitmap*>(typelessBitmap);
180 ASSERT_UNUSED(typelessData, bitmap->data() == typelessData);
181 bitmap->deref(); // Balanced by ref in createCGImage.
182}

184RefPtr<Image> ShareableBitmap::createImage()
185{
186 if (auto platformImage = makeCGImage())
187 return BitmapImage::create(WTFMove(platformImage));
188 return nullptr;
189}
