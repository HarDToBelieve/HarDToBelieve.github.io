37void InteractionInformationAtPosition::encode(IPC::Encoder& encoder) const
38{
39 encoder << request;
40
41 encoder << canBeValid;
42 encoder << nodeAtPositionHasDoubleClickHandler;
43 encoder << isSelectable;
44 encoder << isSelected;
45 encoder << prefersDraggingOverTextSelection;
46 encoder << isNearMarkedText;
47 encoder << touchCalloutEnabled;
48 encoder << isLink;
49 encoder << isImage;
50 encoder << isAttachment;
51 encoder << isAnimatedImage;
52 encoder << isElement;
53 encoder << isContentEditable;
54 encoder << containerScrollingNodeID;
55 encoder << adjustedPointForNodeRespondingToClickEvents;
56 encoder << url;
57 encoder << imageURL;
58 encoder << title;
59 encoder << idAttribute;
60 encoder << bounds;
61#if PLATFORM(MACCATALYST)
62 encoder << caretRect;
63#endif
64 encoder << textBefore;
65 encoder << textAfter;
66 encoder << caretLength;
67 encoder << lineCaretExtent;
68 encoder << cursor;
69 encoder << linkIndicator;
70
71 ShareableBitmap::Handle handle;
72 if (image)
73 image->createHandle(handle, SharedMemory::Protection::ReadOnly);
74 encoder << handle;
75#if ENABLE(DATA_DETECTION)
76 encoder << isDataDetectorLink;
77 encoder << dataDetectorIdentifier;
78 encoder << dataDetectorResults;
79 encoder << dataDetectorBounds;
80#endif
81#if ENABLE(DATALIST_ELEMENT)
82 encoder << preventTextInteraction;
83#endif
84 encoder << shouldNotUseIBeamInEditableContent;
85 encoder << isImageOverlayText;
86 encoder << isHorizontalWritingMode;
87 encoder << elementContext;
88 encoder << imageElementContext;
89}

91bool InteractionInformationAtPosition::decode(IPC::Decoder& decoder, InteractionInformationAtPosition& result)
92{
93 if (!decoder.decode(result.request))
94 return false;
95
96 if (!decoder.decode(result.canBeValid))
97 return false;
98
99 if (!decoder.decode(result.nodeAtPositionHasDoubleClickHandler))
100 return false;
101
102 if (!decoder.decode(result.isSelectable))
103 return false;
104
105 if (!decoder.decode(result.isSelected))
106 return false;
107
108 if (!decoder.decode(result.prefersDraggingOverTextSelection))
109 return false;
110
111 if (!decoder.decode(result.isNearMarkedText))
112 return false;
113
114 if (!decoder.decode(result.touchCalloutEnabled))
115 return false;
116
117 if (!decoder.decode(result.isLink))
118 return false;
119
120 if (!decoder.decode(result.isImage))
121 return false;
122
123 if (!decoder.decode(result.isAttachment))
124 return false;
125
126 if (!decoder.decode(result.isAnimatedImage))
127 return false;
128
129 if (!decoder.decode(result.isElement))
130 return false;
131
132 if (!decoder.decode(result.isContentEditable))
133 return false;
134
135 if (!decoder.decode(result.containerScrollingNodeID))
136 return false;
137
138 if (!decoder.decode(result.adjustedPointForNodeRespondingToClickEvents))
139 return false;
140
141 if (!decoder.decode(result.url))
142 return false;
143
144 if (!decoder.decode(result.imageURL))
145 return false;
146
147 if (!decoder.decode(result.title))
148 return false;
149
150 if (!decoder.decode(result.idAttribute))
151 return false;
152
153 if (!decoder.decode(result.bounds))
154 return false;
155
156#if PLATFORM(MACCATALYST)
157 if (!decoder.decode(result.caretRect))
158 return false;
159#endif
160
161 if (!decoder.decode(result.textBefore))
162 return false;
163
164 if (!decoder.decode(result.textAfter))
165 return false;
166
167 if (!decoder.decode(result.caretLength))
168 return false;
169
170 if (!decoder.decode(result.lineCaretExtent))
171 return false;
172
173 if (!decoder.decode(result.cursor))
174 return false;
175
176 std::optional<WebCore::TextIndicatorData> linkIndicator;
177 decoder >> linkIndicator;
178 if (!linkIndicator)
179 return false;
180 result.linkIndicator = WTFMove(*linkIndicator);
181
182 ShareableBitmap::Handle handle;
183 if (!decoder.decode(handle))
184 return false;
185
186 if (!handle.isNull())
187 result.image = ShareableBitmap::create(handle, SharedMemory::Protection::ReadOnly);
188
189#if ENABLE(DATA_DETECTION)
190 if (!decoder.decode(result.isDataDetectorLink))
191 return false;
192
193 if (!decoder.decode(result.dataDetectorIdentifier))
194 return false;
195
196 auto dataDetectorResults = IPC::decode<NSArray>(decoder, @[ NSArray.class, PAL::getDDScannerResultClass() ]);
197 if (!dataDetectorResults)
198 return false;
199
200 result.dataDetectorResults = WTFMove(*dataDetectorResults);
201
202 if (!decoder.decode(result.dataDetectorBounds))
203 return false;
204#endif
205
206#if ENABLE(DATALIST_ELEMENT)
207 if (!decoder.decode(result.preventTextInteraction))
208 return false;
209#endif
210
211 if (!decoder.decode(result.shouldNotUseIBeamInEditableContent))
212 return false;
213
214 if (!decoder.decode(result.isImageOverlayText))
215 return false;
216
217 if (!decoder.decode(result.isHorizontalWritingMode))
218 return false;
219
220 if (!decoder.decode(result.elementContext))
221 return false;
222
223 if (!decoder.decode(result.imageElementContext))
224 return false;
225
226 return true;
227}

229void InteractionInformationAtPosition::mergeCompatibleOptionalInformation(const InteractionInformationAtPosition& oldInformation)
230{
231 if (oldInformation.request.point != request.point)
232 return;
233
234 if (oldInformation.request.includeSnapshot && !request.includeSnapshot)
235 image = oldInformation.image;
236
237 if (oldInformation.request.includeLinkIndicator && !request.includeLinkIndicator)
238 linkIndicator = oldInformation.linkIndicator;
239}
