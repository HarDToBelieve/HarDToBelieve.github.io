38template<typename> struct CoreMediaTraits;

40#define DECLARE_CORE_MEDIA_TRAITS(ClassName) \
41namespace WebKit { \
42class Media##ClassName; \
43template<> struct CoreMediaTraits<Media##ClassName> { \
44 using Class = MTPlugin##ClassName##Class; \
45 using Ref = MTPlugin##ClassName##Ref; \
46 using VTable = MTPlugin##ClassName##VTable; \
47}; \
48}

50template<typename Wrapped>
51class CoreMediaWrapped {
52 WTF_FORBID_HEAP_ALLOCATION;
53 WTF_MAKE_NONCOPYABLE(CoreMediaWrapped);
54public:
55 using WrapperRef = typename CoreMediaTraits<Wrapped>::Ref;
56
57 void ref() { CFRetain(m_leakedWrapper); }
58 void deref() { CFRelease(m_leakedWrapper); }
59 WrapperRef wrapper() const { return m_leakedWrapper; }
60 virtual ~CoreMediaWrapped() { m_leakedWrapper = nullptr; }
61
62protected:
63 using WrapperClass = typename CoreMediaTraits<Wrapped>::Class;
64
65 static Wrapped* unwrap(WrapperRef);
66
67 class Allocator {
68 public:
69 Allocator(CFAllocatorRef allocator)
70 : m_allocator(allocator) { }
71
72 void* allocate(size_t);
73 WrapperRef leakWrapper() { return m_wrapper.leakRef(); }
74
75 private:
76 RetainPtr<CFAllocatorRef> m_allocator;
77 RetainPtr<WrapperRef> m_wrapper;
78 };
79
80 explicit CoreMediaWrapped(Allocator&& allocator)
81 : m_leakedWrapper(allocator.leakWrapper()) { }
82
83 void* operator new(size_t size, Allocator& allocator) { return allocator.allocate(size); }
84 CFAllocatorRef allocator() const { return CFGetAllocator(m_leakedWrapper); }
85 virtual void finalize() { this->~CoreMediaWrapped<Wrapped>(); }
86
87 // CMBaseClass
88 virtual String debugDescription() const = 0;
89 virtual OSStatus copyProperty(CFStringRef, CFAllocatorRef, void* copiedValue) = 0;
90
91private:
92 using WrapperVTable = typename CoreMediaTraits<Wrapped>::VTable;
93
94 template<size_t> static constexpr CMBaseClass wrapperClass();
95 static const WrapperVTable& vTable();
96
97 WrapperRef m_leakedWrapper { nullptr };
98};
99
100bool createWrapper(CFAllocatorRef, const CMBaseVTable*, CMBaseClassID, CMBaseObjectRef*);
101void* wrapperStorage(CMBaseObjectRef);
102const CMBaseVTable* wrapperVTable(CMBaseObjectRef);
103
104template<typename Wrapped>
105Wrapped* CoreMediaWrapped<Wrapped>::unwrap(WrapperRef wrapper)
106{
107 auto baseObject = reinterpret_cast<CMBaseObjectRef>(wrapper);
108 if (&vTable().base != wrapperVTable(baseObject))
109 return nullptr;
110 return static_cast<Wrapped*>(wrapperStorage(baseObject));
111}
112
113template<typename Wrapped>
114void* CoreMediaWrapped<Wrapped>::Allocator::allocate(size_t size)
115{
116 ASSERT(!m_wrapper);
117 ASSERT_UNUSED(size, size == sizeof(Wrapped));
118 CMBaseObjectRef object = nullptr;
119 if (!createWrapper(m_allocator.get(), &vTable().base, Wrapped::wrapperClassID(), &object))
120 return nullptr;
121 m_wrapper = adoptCF(reinterpret_cast<WrapperRef>(object));
122 return unwrap(m_wrapper.get());
123}
124
125template<typename Wrapped>
126template<size_t size>
127constexpr CMBaseClass CoreMediaWrapped<Wrapped>::wrapperClass()
128{
129 return {
130 .version = kCMBaseObject_ClassVersion_1,
131 .derivedStorageSize = size,
132 .finalize = [](CMBaseObjectRef object) {
133 Wrapped::unwrap(object)->finalize();
134 },
135 .copyDebugDescription = [](CMBaseObjectRef object) {
136 return Wrapped::unwrap(object)->debugDescription().createCFString().leakRef();
137 },
138 .copyProperty = [](CMBaseObjectRef object, CFStringRef key, CFAllocatorRef allocator, void* copiedValue) {
139 return Wrapped::unwrap(object)->copyProperty(key, allocator, copiedValue);
140 },
141 };
142};
143
144template<typename Wrapped>
145const typename CoreMediaWrapped<Wrapped>::WrapperVTable& CoreMediaWrapped<Wrapped>::vTable()
146{
147 // CMBaseClass contains 64-bit pointers that aren't 8-byte aligned. To suppress the linker
148 // warning about this, we prepend 4 bytes of padding when building.
149#if CPU(X86_64)
150 constexpr size_t padSize = 4;
151#else
152 constexpr size_t padSize = 0;
153#endif
154
155#pragma pack(push, 4)
156 static constexpr struct { uint8_t pad[padSize]; CMBaseClass baseClass; } baseClass { { }, wrapperClass<sizeof(Wrapped)>() };
157#pragma pack(pop)
158 static constexpr WrapperClass derivedClass = Wrapped::wrapperClass();
159
160#if CPU(X86_64)
161 static_assert(sizeof(CMBaseClass::version) != sizeof(void*), "Fig struct fixup is required for CMBaseClass on X86_64");
162 static_assert(sizeof(WrapperClass::version) == sizeof(void*), "Fig struct fixup is not required for WrapperClass on X86_64");
163#else
164 static_assert(sizeof(CMBaseClass::version) == sizeof(void*), "Fig struct fixup only required for CMBaseClass on X86_64");
165 static_assert(sizeof(WrapperClass::version) == sizeof(void*), "Fig struct fixup only required for WrapperClass on X86_64");
166#endif
167 static_assert(alignof(CMBaseClass) == 4, "CMBaseClass must have 4 byte alignment");
168 static_assert(alignof(WrapperClass) == sizeof(void*), "WrapperClass must be natually aligned");
169
170IGNORE_WARNINGS_BEGIN("missing-field-initializers")
171 static constexpr WrapperVTable vTable {
172 { nullptr, &baseClass.baseClass },
173 &derivedClass
174 };
175IGNORE_WARNINGS_END
176 return vTable;
177}
