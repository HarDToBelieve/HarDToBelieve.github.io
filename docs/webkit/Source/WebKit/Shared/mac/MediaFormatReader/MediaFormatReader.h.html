44class MediaTrackReader;

46class MediaFormatReader final : public CoreMediaWrapped<MediaFormatReader> {
47public:
48 using CoreMediaWrapped<MediaFormatReader>::unwrap;
49
50 static constexpr WrapperClass wrapperClass();
51 static CMBaseClassID wrapperClassID();
52 static CoreMediaWrapped<MediaFormatReader>* unwrap(CMBaseObjectRef);
53
54 static RefPtr<MediaFormatReader> create(Allocator&&);
55
56 void startOnMainThread(MTPluginByteSourceRef);
57 const MediaTime& duration() const { return m_duration; }
58
59 const Logger& logger() const { ASSERT(m_logger); return *m_logger.get(); }
60 const Logger* loggerPtr() const { return m_logger.get(); }
61 const void* nextTrackReaderLogIdentifier(uint64_t) const;
62
63private:
64 explicit MediaFormatReader(Allocator&&);
65
66 void parseByteSource(RetainPtr<MTPluginByteSourceRef>&&);
67 void didParseTracks(WebCore::SourceBufferPrivateClient::InitializationSegment&&, uint64_t errorCode);
68 void didSelectVideoTrack(WebCore::VideoTrackPrivate&, bool) { }
69 void didEnableAudioTrack(WebCore::AudioTrackPrivate&, bool) { }
70 void didProvideMediaData(Ref<WebCore::MediaSample>&&, uint64_t, const String&);
71 void finishParsing(Ref<WebCore::SourceBufferParser>&&);
72
73 // CMBaseClass
74 String debugDescription() const final { return "WebKit::MediaFormatReader"_s; }
75 OSStatus copyProperty(CFStringRef, CFAllocatorRef, void* copiedValue) final;
76
77 // WrapperClass
78 OSStatus copyTrackArray(CFArrayRef*);
79
80 const void* logIdentifier() const { return m_logIdentifier; }
81
82 RetainPtr<MTPluginByteSourceRef> m_byteSource WTF_GUARDED_BY_LOCK(m_parseTracksLock);
83 Condition m_parseTracksCondition;
84 Lock m_parseTracksLock;
85 MediaTime m_duration WTF_GUARDED_BY_LOCK(m_parseTracksLock);
86 std::optional<OSStatus> m_parseTracksStatus WTF_GUARDED_BY_LOCK(m_parseTracksLock);
87 Vector<Ref<MediaTrackReader>> m_trackReaders WTF_GUARDED_BY_LOCK(m_parseTracksLock);
88 RefPtr<const Logger> m_logger;
89 const void* m_logIdentifier;
90};

92constexpr MediaFormatReader::WrapperClass MediaFormatReader::wrapperClass()
93{
94 return {
95 .version = kMTPluginFormatReader_ClassVersion_1,
96 .copyTrackArray = [](WrapperRef reader, CFArrayRef* trackArrayCopy) {
97 return unwrap(reader)->copyTrackArray(trackArrayCopy);
98 },
99 };
100}
