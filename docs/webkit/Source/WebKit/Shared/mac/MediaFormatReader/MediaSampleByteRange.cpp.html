36MediaSampleByteRange::MediaSampleByteRange(MediaSample& sample, MTPluginByteSourceRef byteSource, uint64_t trackID)
37 : MediaSampleAVFObjC(nullptr)
38 , m_presentationTime(sample.presentationTime())
39 , m_decodeTime(sample.decodeTime())
40 , m_duration(sample.duration())
41 , m_byteRange(*sample.byteRange())
42 , m_trackID(trackID)
43 , m_sizeInBytes(sample.sizeInBytes())
44 , m_presentationSize(sample.presentationSize())
45 , m_byteSource(byteSource)
46 , m_flags(sample.flags())
47{
48 ASSERT(!isMainRunLoop());
49 ASSERT(m_decodeTime == m_presentationTime || m_decodeTime == MediaTime::invalidTime());
50 auto platformSample = sample.platformSample();
51 switch (platformSample.type) {
52 case PlatformSample::CMSampleBufferType:
53 m_formatDescription = PAL::CMSampleBufferGetFormatDescription(platformSample.sample.cmSampleBuffer);
54 break;
55 case PlatformSample::ByteRangeSampleType:
56 m_formatDescription = platformSample.sample.byteRangeSample.second;
57 break;
58 default:
59 ASSERT_NOT_REACHED();
60 break;
61 }
62}
63
64AtomString MediaSampleByteRange::trackID() const
65{
66 return AtomString::number(m_trackID);
67}
68
69PlatformSample MediaSampleByteRange::platformSample()
70{
71 return {
72 PlatformSample::ByteRangeSampleType,
73 { .byteRangeSample = std::make_pair(m_byteSource.get(), m_formatDescription.get()) },
74 };
75}
76
77void MediaSampleByteRange::offsetTimestampsBy(const MediaTime& offset)
78{
79 setTimestamps(presentationTime() + offset, decodeTime() + offset);
80}
81
82void MediaSampleByteRange::setTimestamps(const MediaTime& presentationTime, const MediaTime& decodeTime)
83{
84 m_presentationTime = presentationTime;
85 m_decodeTime = decodeTime;
86}
