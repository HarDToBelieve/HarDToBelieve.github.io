73MediaTrackReader::MediaTrackReader(Allocator&& allocator, const MediaFormatReader& formatReader, CMMediaType mediaType, uint64_t trackID, std::optional<bool> enabled)
74 : CoreMediaWrapped(WTFMove(allocator))
75 , m_trackID(trackID)
76 , m_mediaType(mediaType)
77 , m_duration(formatReader.duration())
78 , m_logger(formatReader.logger())
79 , m_logIdentifier(formatReader.nextTrackReaderLogIdentifier(trackID))
80{
81 ASSERT(!isMainRunLoop());
82
83 ALWAYS_LOG(LOGIDENTIFIER, mediaTypeString(), " ", trackID);
84 if (enabled)
85 m_isEnabled = enabled.value() ? Enabled::True : Enabled::False;
86}

88MediaTime MediaTrackReader::greatestPresentationTime() const
89{
90 Locker locker { m_sampleStorageLock };
91 auto& sampleMap = m_sampleStorage->sampleMap;
92 if (sampleMap.empty())
93 return MediaTime::invalidTime();
94
95 auto& lastSample = *sampleMap.decodeOrder().rbegin()->second;
96 return lastSample.presentationTime() + lastSample.duration();
97}

99void MediaTrackReader::addSample(Ref<MediaSample>&& sample, MTPluginByteSourceRef byteSource)
100{
101 ASSERT(!isMainRunLoop());
102 Locker locker { m_sampleStorageLock };
103 if (!m_sampleStorage)
104 m_sampleStorage = makeUnique<SampleStorage>();
105
106 ASSERT(!sample->isDivisable() && sample->byteRange());
107 auto sampleToAdd = MediaSampleByteRange::create(sample.get(), byteSource, m_trackID);
108
109 // FIXME: Even though WebM muxer guidelines say this must not happen, some video tracks have two
110 // consecutive frames with the same presentation time. SampleMap will not store the second frame
111 // in these cases, corrupting all subsequent non-key frames. Find a way to store frames with
112 // duplicate presentation times.
113
114 m_sampleStorage->sampleMap.addSample(sampleToAdd.get());
115 m_sampleStorageCondition.notifyAll();
116}

118void MediaTrackReader::waitForSample(Function<bool(SampleMap&, bool)>&& predicate) const
119{
120 Locker locker { m_sampleStorageLock };
121 if (!m_sampleStorage)
122 m_sampleStorage = makeUnique<SampleStorage>();
123 m_sampleStorageCondition.wait(m_sampleStorageLock, [predicate = WTFMove(predicate), this] {
124 assertIsHeld(m_sampleStorageLock);
125 return predicate(m_sampleStorage->sampleMap, m_sampleStorage->hasAllSamples);
126 });
127}
128
129void MediaTrackReader::finishParsing()
130{
131 ASSERT(!isMainRunLoop());
132
133 ALWAYS_LOG(LOGIDENTIFIER);
134 Locker locker { m_sampleStorageLock };
135 if (!m_sampleStorage)
136 m_sampleStorage = makeUnique<SampleStorage>();
137 m_sampleStorage->hasAllSamples = true;
138 if (m_isEnabled == Enabled::Unknown) {
139 m_isEnabled = m_sampleStorage->sampleMap.empty() ? Enabled::False : Enabled::True;
140 if (m_isEnabled == Enabled::False)
141 ERROR_LOG(LOGIDENTIFIER, "ignoring empty ", mediaTypeString(), " track");
142 }
143 m_sampleStorageCondition.notifyAll();
144}

146const char* MediaTrackReader::mediaTypeString() const
147{
148 switch (m_mediaType) {
149 case kCMMediaType_Video:
150 return "video";
151 case kCMMediaType_Audio:
152 return "audio";
153 case kCMMediaType_Text:
154 return "text";
155 }
156 ASSERT_NOT_REACHED();
157 return "unknown";
158}

160OSStatus MediaTrackReader::copyProperty(CFStringRef key, CFAllocatorRef allocator, void* copiedValue)
161{
162 // Don't block waiting for media to answer requests for the enabled state.
163 if (CFEqual(key, PAL::get_MediaToolbox_kMTPluginTrackReaderProperty_Enabled())) {
164 // Assume that a track without an explicit enabled state is enabled by default,
165 // to avoid blocking waiting for media data to be appended.
166 *reinterpret_cast<CFBooleanRef*>(copiedValue) = m_isEnabled == Enabled::False ? kCFBooleanFalse : kCFBooleanTrue;
167 return noErr;
168 }
169
170 Locker locker { m_sampleStorageLock };
171 m_sampleStorageCondition.wait(m_sampleStorageLock, [&] {
172 assertIsHeld(m_sampleStorageLock);
173 return !!m_sampleStorage;
174 });
175
176 auto& sampleMap = m_sampleStorage->sampleMap;
177
178 if (sampleMap.empty()) {
179 ERROR_LOG(LOGIDENTIFIER, "sample table empty when asked for ", String(key));
180 return kCMBaseObjectError_ValueNotAvailable;
181 }
182
183 auto& lastSample = *sampleMap.decodeOrder().rbegin()->second;
184
185 if (CFEqual(key, PAL::get_MediaToolbox_kMTPluginTrackReaderProperty_FormatDescriptionArray())) {
186 RELEASE_ASSERT_WITH_SECURITY_IMPLICATION(lastSample.platformSample().type == PlatformSample::ByteRangeSampleType);
187 const void* descriptions[1] = { lastSample.platformSample().sample.byteRangeSample.second };
188 *reinterpret_cast<CFArrayRef*>(copiedValue) = adoptCF(CFArrayCreate(allocator, descriptions, 1, &kCFTypeArrayCallBacks)).leakRef();
189 return noErr;
190 }
191
192 if (CFEqual(key, PAL::get_MediaToolbox_kMTPluginTrackReaderProperty_NominalFrameRate())) {
193 float frameRate = 1 / lastSample.duration().toFloat();
194 *reinterpret_cast<CFNumberRef*>(copiedValue) = adoptCF(CFNumberCreate(allocator, kCFNumberFloat32Type, &frameRate)).leakRef();
195 return noErr;
196 }
197
198 ERROR_LOG(LOGIDENTIFIER, "asked for unsupported property ", String(key));
199 return kCMBaseObjectError_ValueNotAvailable;
200}

239const void* MediaTrackReader::nextSampleCursorLogIdentifier(uint64_t cursorID) const
240{
241 uint64_t trackID = reinterpret_cast<uint64_t>(m_logIdentifier) & 0xffffull;
242 uint64_t trackAndCursorID = trackID << 8 | (cursorID & 0xffull);
243 return LoggerHelper::childLogIdentifier(m_logIdentifier, trackAndCursorID);
244}
