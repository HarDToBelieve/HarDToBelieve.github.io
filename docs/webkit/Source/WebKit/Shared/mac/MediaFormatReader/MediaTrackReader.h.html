53class MediaTrackReader final : public CoreMediaWrapped<MediaTrackReader> {
54public:
55 static constexpr WrapperClass wrapperClass();
56 static CMBaseClassID wrapperClassID();
57 static CoreMediaWrapped<MediaTrackReader>* unwrap(CMBaseObjectRef);
58 static WTF::WorkQueue& storageQueue();
59
60 static RefPtr<MediaTrackReader> create(Allocator&&, const MediaFormatReader&, CMMediaType, uint64_t, std::optional<bool>);
61
62 uint64_t trackID() const { return m_trackID; }
63 CMMediaType mediaType() const { return m_mediaType; }
64 const MediaTime& duration() const { return m_duration; }
65 MediaTime greatestPresentationTime() const;
66
67 void setEnabled(bool enabled) { m_isEnabled = enabled ? Enabled::True : Enabled::False; }
68 void addSample(Ref<WebCore::MediaSample>&&, MTPluginByteSourceRef);
69 void waitForSample(Function<bool(WebCore::SampleMap&, bool)>&&) const;
70 void finishParsing();
71
72 const WTF::Logger& logger() const { return m_logger; }
73 const void* nextSampleCursorLogIdentifier(uint64_t cursorID) const;
74
75private:
76 using CoreMediaWrapped<MediaTrackReader>::unwrap;
77
78 MediaTrackReader(Allocator&&, const MediaFormatReader&, CMMediaType, uint64_t, std::optional<bool>);
79
80 // CMBaseClass
81 String debugDescription() const final { return "WebKit::MediaTrackReader"_s; }
82 OSStatus copyProperty(CFStringRef, CFAllocatorRef, void* copiedValue) final;
83 void finalize() final;
84
85 // WrapperClass
86 OSStatus getTrackInfo(MTPersistentTrackID*, CMMediaType*);
87 OSStatus createCursorAtPresentationTimeStamp(CMTime, MTPluginSampleCursorRef*);
88 OSStatus createCursorAtFirstSampleInDecodeOrder(MTPluginSampleCursorRef*);
89 OSStatus createCursorAtLastSampleInDecodeOrder(MTPluginSampleCursorRef*);
90
91 struct SampleStorage {
92 WTF_MAKE_STRUCT_FAST_ALLOCATED;
93 ~SampleStorage() { ASSERT(!isMainRunLoop()); }
94 WebCore::SampleMap sampleMap;
95 bool hasAllSamples { false };
96 };
97
98 const char* mediaTypeString() const;
99 const char* logClassName() const { return "MediaTrackReader"; }
100 const void* logIdentifier() const { return m_logIdentifier; }
101 WTFLogChannel& logChannel() const;
102
103 enum Enabled : uint8_t { Unknown, False, True };
104
105 const uint64_t m_trackID;
106 const CMMediaType m_mediaType;
107 const MediaTime m_duration;
108 std::atomic<Enabled> m_isEnabled { Enabled::Unknown };
109 mutable Condition m_sampleStorageCondition;
110 mutable Lock m_sampleStorageLock;
111 mutable std::unique_ptr<SampleStorage> m_sampleStorage WTF_GUARDED_BY_LOCK(m_sampleStorageLock);
112 Ref<const Logger> m_logger;
113 const void* m_logIdentifier;
114};

116constexpr MediaTrackReader::WrapperClass MediaTrackReader::wrapperClass()
117{
118 return {
119 .version = kMTPluginTrackReader_ClassVersion_1,
120 .getTrackInfo = [](MTPluginTrackReaderRef trackReader, MTPersistentTrackID* trackID, CMMediaType* mediaType) {
121 return unwrap(trackReader)->getTrackInfo(trackID, mediaType);
122 },
123 .createCursorAtPresentationTimeStamp = [](MTPluginTrackReaderRef trackReader, CMTime timestamp, MTPluginSampleCursorRef* cursor) {
124 return unwrap(trackReader)->createCursorAtPresentationTimeStamp(timestamp, cursor);
125 },
126 .createCursorAtFirstSampleInDecodeOrder = [](MTPluginTrackReaderRef trackReader, MTPluginSampleCursorRef* cursor) {
127 return unwrap(trackReader)->createCursorAtFirstSampleInDecodeOrder(cursor);
128 },
129 .createCursorAtLastSampleInDecodeOrder = [](MTPluginTrackReaderRef trackReader, MTPluginSampleCursorRef* cursor) {
130 return unwrap(trackReader)->createCursorAtLastSampleInDecodeOrder(cursor);
131 },
132 };
133}
