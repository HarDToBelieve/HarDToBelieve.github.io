28#include "APIObject.h"
29#include <wtf/RetainPtr.h>

31namespace IPC {
32class Decoder;
33class Encoder;
34}
35
36typedef struct objc_object* id;
37
38namespace WebKit {
39
40class ObjCObjectGraph : public API::ObjectImpl<API::Object::Type::ObjCObjectGraph> {
41public:
42 static Ref<ObjCObjectGraph> create(id rootObject)
43 {
44 return adoptRef(*new ObjCObjectGraph(rootObject));
45 }
46
47 id rootObject() const { return m_rootObject.get(); }
48
49 struct Transformer {
50 virtual ~Transformer() { }
51 virtual bool shouldTransformObject(id) const = 0;
52 virtual RetainPtr<id> transformObject(id) const = 0;
53 };
54 static RetainPtr<id> transform(id, const Transformer&);
55
56 void encode(IPC::Encoder&) const;
57 static WARN_UNUSED_RETURN bool decode(IPC::Decoder&, RefPtr<API::Object>&);
58
59private:
60 static void encode(IPC::Encoder&, id);
61 static WARN_UNUSED_RETURN bool decode(IPC::Decoder&, RetainPtr<id>&);
62
63 explicit ObjCObjectGraph(id rootObject)
64 : m_rootObject(rootObject)
65 {
66 }
67
68 RetainPtr<id> m_rootObject;
69};

71} // namespace WebKit
