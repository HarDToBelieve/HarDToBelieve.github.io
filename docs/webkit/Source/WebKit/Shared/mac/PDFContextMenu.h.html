31struct PDFContextMenuItem {
32 String title;
33 bool enabled;
34 bool separator;
35 int state;
36 bool hasAction;
37 int tag;
38
39 template<class Encoder> void encode(Encoder& encoder) const
40 {
41 encoder << title << enabled << separator << state << hasAction << tag;
42 }
43
44 template<class Decoder> static std::optional<PDFContextMenuItem> decode(Decoder& decoder)
45 {
46 std::optional<String> title;
47 decoder >> title;
48 if (!title)
49 return std::nullopt;
50
51 std::optional<bool> enabled;
52 decoder >> enabled;
53 if (!enabled)
54 return std::nullopt;
55
56 std::optional<bool> separator;
57 decoder >> separator;
58 if (!separator)
59 return std::nullopt;
60
61 std::optional<int> state;
62 decoder >> state;
63 if (!state)
64 return std::nullopt;
65
66 std::optional<bool> hasAction;
67 decoder >> hasAction;
68 if (!hasAction)
69 return std::nullopt;
70
71 std::optional<int> tag;
72 decoder >> tag;
73 if (!tag)
74 return std::nullopt;
75
76 return { { WTFMove(*title), WTFMove(*enabled), WTFMove(*separator), WTFMove(*state), WTFMove(*hasAction), WTFMove(*tag) } };
77 }
78};
79
80struct PDFContextMenu {
81 WebCore::IntPoint point;
82 Vector<PDFContextMenuItem> items;
83 std::optional<int> openInPreviewIndex;
84
85 template<class Encoder> void encode(Encoder& encoder) const
86 {
87 encoder << point << items << openInPreviewIndex;
88 }
89
90 template<class Decoder> static std::optional<PDFContextMenu> decode(Decoder& decoder)
91 {
92 std::optional<WebCore::IntPoint> point;
93 decoder >> point;
94 if (!point)
95 return std::nullopt;
96
97 std::optional<Vector<PDFContextMenuItem>> items;
98 decoder >> items;
99 if (!items)
100 return std::nullopt;
101
102 std::optional<std::optional<int>> openInPreviewIndex;
103 decoder >> openInPreviewIndex;
104 if (!openInPreviewIndex)
105 return std::nullopt;
106
107 return {{
108 WTFMove(*point),
109 WTFMove(*items),
110 WTFMove(*openInPreviewIndex)
111 }};
112 }
113
114};
115
116};
