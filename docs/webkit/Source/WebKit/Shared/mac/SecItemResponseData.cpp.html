32namespace WebKit {
33
34SecItemResponseData::SecItemResponseData(OSStatus resultCode, RetainPtr<CFTypeRef>&& resultObject)
35 : m_resultObject(WTFMove(resultObject))
36 , m_resultCode(resultCode)
37{
38}
39
40void SecItemResponseData::encode(IPC::Encoder& encoder) const
41{
42 encoder << static_cast<int64_t>(m_resultCode);
43 encoder << static_cast<bool>(m_resultObject);
44 if (m_resultObject)
45 encoder << m_resultObject;
46}
47
48std::optional<SecItemResponseData> SecItemResponseData::decode(IPC::Decoder& decoder)
49{
50 int64_t resultCode;
51 if (!decoder.decode(resultCode))
52 return std::nullopt;
53
54 bool expectResultObject;
55 if (!decoder.decode(expectResultObject))
56 return std::nullopt;
57
58 RetainPtr<CFTypeRef> result;
59 if (expectResultObject && !decoder.decode(result))
60 return std::nullopt;
61
62 return {{ static_cast<OSStatus>(resultCode), WTFMove(result) }};
63}
