60static WeakPtr<NetworkProcess>& globalNetworkProcess()
61{
62 static NeverDestroyed<WeakPtr<NetworkProcess>> networkProcess;
63 return networkProcess.get();
64}

66static std::optional<SecItemResponseData> sendSecItemRequest(SecItemRequestData::Type requestType, CFDictionaryRef query, CFDictionaryRef attributesToMatch = 0)
67{
68 std::optional<SecItemResponseData> response;
69
70 if (RunLoop::isMain()) {
71 if (!globalNetworkProcess()->parentProcessConnection()->sendSync(Messages::SecItemShimProxy::SecItemRequestSync(SecItemRequestData(requestType, query, attributesToMatch)), Messages::SecItemShimProxy::SecItemRequestSync::Reply(response), 0))
72 return std::nullopt;
73 return response;
74 }
75
76 BinarySemaphore semaphore;
77
78 RunLoop::main().dispatch([&] {
79 if (!globalNetworkProcess()) {
80 semaphore.signal();
81 return;
82 }
83
84 globalNetworkProcess()->parentProcessConnection()->sendWithAsyncReply(Messages::SecItemShimProxy::SecItemRequest(SecItemRequestData(requestType, query, attributesToMatch)), [&](auto reply) {
85 if (reply)
86 response = WTFMove(*reply);
87
88 semaphore.signal();
89 });
90 });
91
92 semaphore.wait();
93
94 return response;
95}

97static OSStatus webSecItemCopyMatching(CFDictionaryRef query, CFTypeRef* result)
98{
99 auto response = sendSecItemRequest(SecItemRequestData::CopyMatching, query);
100 if (!response)
101 return errSecInteractionNotAllowed;
102
103 *result = response->resultObject().leakRef();
104 return response->resultCode();
105}
106
107static OSStatus webSecItemAdd(CFDictionaryRef query, CFTypeRef* unusedResult)
108{
109 // Return value of SecItemAdd should be ignored for WebKit use cases. WebKit can't serialize SecKeychainItemRef, so we do not use it.
110 // If someone passes a result value to be populated, the API contract is being violated so we should assert.
111 if (unusedResult) {
112 ASSERT_NOT_REACHED();
113 return errSecParam;
114 }
115
116 auto response = sendSecItemRequest(SecItemRequestData::Add, query);
117 if (!response)
118 return errSecInteractionNotAllowed;
119
120 return response->resultCode();
121}
122
123static OSStatus webSecItemUpdate(CFDictionaryRef query, CFDictionaryRef attributesToUpdate)
124{
125 auto response = sendSecItemRequest(SecItemRequestData::Update, query, attributesToUpdate);
126 if (!response)
127 return errSecInteractionNotAllowed;
128
129 return response->resultCode();
130}

132static OSStatus webSecItemDelete(CFDictionaryRef query)
133{
134 auto response = sendSecItemRequest(SecItemRequestData::Delete, query);
135 if (!response)
136 return errSecInteractionNotAllowed;
137
138 return response->resultCode();
139}

141void initializeSecItemShim(NetworkProcess& process)
142{
143 globalNetworkProcess() = makeWeakPtr(process);
144
145 struct _CFNFrameworksStubs stubs = {
146 .version = 0,
147 .SecItem_stub_CopyMatching = webSecItemCopyMatching,
148 .SecItem_stub_Add = webSecItemAdd,
149 .SecItem_stub_Update = webSecItemUpdate,
150 .SecItem_stub_Delete = webSecItemDelete,
151 };
152
153 _CFURLConnectionSetFrameworkStubs(&stubs);
154}
