34#include <wtf/WeakPtr.h>

37OBJC_CLASS NSFileWrapper;
38OBJC_CLASS NSString;
39
40namespace WebCore {
41class SharedBuffer;
42}
43
44namespace WebKit {
45class WebPageProxy;
46}
47
48namespace API {
49
50class Attachment final : public ObjectImpl<Object::Type::Attachment> {
51public:
52 static Ref<Attachment> create(const WTF::String& identifier, WebKit::WebPageProxy&);
53 virtual ~Attachment();
54
55 enum class InsertionState : uint8_t { NotInserted, Inserted };
56
57 const WTF::String& identifier() const { return m_identifier; }
58 void updateAttributes(CompletionHandler<void()>&&);
59
60 void invalidate();
61 bool isValid() const { return !!m_webPage; }
62
63#if PLATFORM(COCOA)
64 NSFileWrapper *fileWrapper() const;
65 void setFileWrapper(NSFileWrapper *fileWrapper) { m_fileWrapper = fileWrapper; }
66 void setFileWrapperAndUpdateContentType(NSFileWrapper *, NSString *contentType);
67 void setFileWrapperGenerator(Function<RetainPtr<NSFileWrapper>(void)>&&);
68 void invalidateGeneratedFileWrapper();
69 WTF::String utiType() const;
70#endif
71 WTF::String mimeType() const;
72
73 const WTF::String& filePath() const { return m_filePath; }
74 void setFilePath(const WTF::String& filePath) { m_filePath = filePath; }
75 WTF::String fileName() const;
76
77 const WTF::String& contentType() const { return m_contentType; }
78 void setContentType(const WTF::String& contentType) { m_contentType = contentType; }
79
80 InsertionState insertionState() const { return m_insertionState; }
81 void setInsertionState(InsertionState state) { m_insertionState = state; }
82
83 bool isEmpty() const;
84
85 RefPtr<WebCore::SharedBuffer> enclosingImageData() const;
86 std::optional<uint64_t> fileSizeForDisplay() const;
87
88 void setHasEnclosingImage(bool hasEnclosingImage) { m_hasEnclosingImage = hasEnclosingImage; }
89 bool hasEnclosingImage() const { return m_hasEnclosingImage; }
90
91 RefPtr<WebCore::SharedBuffer> createSerializedRepresentation() const;
92 void updateFromSerializedRepresentation(Ref<WebCore::SharedBuffer>&&, const WTF::String& contentType);
93
94private:
95 explicit Attachment(const WTF::String& identifier, WebKit::WebPageProxy&);
96
97#if PLATFORM(COCOA)
98 mutable RetainPtr<NSFileWrapper> m_fileWrapper;
99 Function<RetainPtr<NSFileWrapper>(void)> m_fileWrapperGenerator;
100#endif
101 WTF::String m_identifier;
102 WTF::String m_filePath;
103 WTF::String m_contentType;
104 WeakPtr<WebKit::WebPageProxy> m_webPage;
105 InsertionState m_insertionState { InsertionState::NotInserted };
106 bool m_hasEnclosingImage { false };
107};
108
109} // namespace API

111#endif // ENABLE(ATTACHMENT_ELEMENT)
