38class ContentRuleList final : public ObjectImpl<Object::Type::ContentRuleList> {
39public:
40#if ENABLE(CONTENT_EXTENSIONS)
41 static Ref<ContentRuleList> create(const WTF::String& name, Ref<WebKit::WebCompiledContentRuleList>&& contentRuleList, WebKit::NetworkCache::Data&& mappedFile)
42 {
43 return adoptRef(*new ContentRuleList(name, WTFMove(contentRuleList), WTFMove(mappedFile)));
44 }
45
46 ContentRuleList(const WTF::String& name, Ref<WebKit::WebCompiledContentRuleList>&&, WebKit::NetworkCache::Data&&);
47 virtual ~ContentRuleList();
48
49 const WTF::String& name() const { return m_name; }
50 const WebKit::WebCompiledContentRuleList& compiledRuleList() const { return m_compiledRuleList.get(); }
51
52 static bool supportsRegularExpression(const WTF::String&);
53
54private:
55 WTF::String m_name;
56 Ref<WebKit::WebCompiledContentRuleList> m_compiledRuleList;
57 WebKit::NetworkCache::Data m_mappedFile;
58#endif // ENABLE(CONTENT_EXTENSIONS)
59};
