97// The size and offset of the densely packed bytes in the file, not sizeof and offsetof, which would
98// represent the size and offset of the structure in memory, possibly with compiler-added padding.
99const size_t ContentRuleListFileHeaderSize = 2 * sizeof(uint32_t) + 5 * sizeof(uint64_t);
100const size_t ConditionsApplyOnlyToDomainOffset = sizeof(uint32_t) + 5 * sizeof(uint64_t);
101
102struct ContentRuleListMetaData {
103 uint32_t version { ContentRuleListStore::CurrentContentRuleListFileVersion };
104 uint64_t sourceSize { 0 };
105 uint64_t actionsSize { 0 };
106 uint64_t filtersWithoutConditionsBytecodeSize { 0 };
107 uint64_t filtersWithConditionsBytecodeSize { 0 };
108 uint64_t conditionedFiltersBytecodeSize { 0 };
109 uint32_t conditionsApplyOnlyToDomain { false };
110
111 size_t fileSize() const
112 {
113 return ContentRuleListFileHeaderSize
114 + sourceSize
115 + actionsSize
116 + filtersWithoutConditionsBytecodeSize
117 + filtersWithConditionsBytecodeSize
118 + conditionedFiltersBytecodeSize;
119 }
120};
121
122static WebKit::NetworkCache::Data encodeContentRuleListMetaData(const ContentRuleListMetaData& metaData)
123{
124 WTF::Persistence::Encoder encoder;
125
126 encoder << metaData.version;
127 encoder << metaData.sourceSize;
128 encoder << metaData.actionsSize;
129 encoder << metaData.filtersWithoutConditionsBytecodeSize;
130 encoder << metaData.filtersWithConditionsBytecodeSize;
131 encoder << metaData.conditionedFiltersBytecodeSize;
132 encoder << metaData.conditionsApplyOnlyToDomain;
133
134 ASSERT(encoder.bufferSize() == ContentRuleListFileHeaderSize);
135 return WebKit::NetworkCache::Data(encoder.buffer(), encoder.bufferSize());
136}

148template<typename T>
149static std::optional<ContentRuleListMetaData> decodeContentRuleListMetaData(const T& fileData)
150{
151 bool success = false;
152 ContentRuleListMetaData metaData;
153 getData(fileData, [&metaData, &success, &fileData](const uint8_t* data, size_t size) {
154 // The file data should be mapped into one continuous memory segment so the size
155 // passed to the applier should always equal the data size.
156 if (size != fileData.size())
157 return false;
158
159 WTF::Persistence::Decoder decoder(data, size);
160
161 std::optional<uint32_t> version;
162 decoder >> version;
163 if (!version)
164 return false;
165 metaData.version = WTFMove(*version);
166
167 std::optional<uint64_t> sourceSize;
168 decoder >> sourceSize;
169 if (!sourceSize)
170 return false;
171 metaData.sourceSize = WTFMove(*sourceSize);
172
173 std::optional<uint64_t> actionsSize;
174 decoder >> actionsSize;
175 if (!actionsSize)
176 return false;
177 metaData.actionsSize = WTFMove(*actionsSize);
178
179 std::optional<uint64_t> filtersWithoutConditionsBytecodeSize;
180 decoder >> filtersWithoutConditionsBytecodeSize;
181 if (!filtersWithoutConditionsBytecodeSize)
182 return false;
183 metaData.filtersWithoutConditionsBytecodeSize = WTFMove(*filtersWithoutConditionsBytecodeSize);
184
185 std::optional<uint64_t> filtersWithConditionsBytecodeSize;
186 decoder >> filtersWithConditionsBytecodeSize;
187 if (!filtersWithConditionsBytecodeSize)
188 return false;
189 metaData.filtersWithConditionsBytecodeSize = WTFMove(*filtersWithConditionsBytecodeSize);
190
191 std::optional<uint64_t> conditionedFiltersBytecodeSize;
192 decoder >> conditionedFiltersBytecodeSize;
193 if (!conditionedFiltersBytecodeSize)
194 return false;
195 metaData.conditionedFiltersBytecodeSize = WTFMove(*conditionedFiltersBytecodeSize);
196
197 std::optional<uint32_t> conditionsApplyOnlyToDomain;
198 decoder >> conditionsApplyOnlyToDomain;
199 if (!conditionsApplyOnlyToDomain)
200 return false;
201 metaData.conditionsApplyOnlyToDomain = WTFMove(*conditionsApplyOnlyToDomain);
202
203 success = true;
204 return false;
205 });
206 if (!success)
207 return std::nullopt;
208 return metaData;
209}

216static std::optional<MappedData> openAndMapContentRuleList(const WTF::String& path)
217{
218 FileSystem::makeSafeToUseMemoryMapForPath(path);
219 WebKit::NetworkCache::Data fileData = mapFile(fileSystemRepresentation(path).data());
220 if (fileData.isNull())
221 return std::nullopt;
222 auto metaData = decodeContentRuleListMetaData(fileData);
223 if (!metaData)
224 return std::nullopt;
225 return {{ WTFMove(*metaData), { WTFMove(fileData) }}};
226}

228static bool writeDataToFile(const WebKit::NetworkCache::Data& fileData, PlatformFileHandle fd)
229{
230 bool success = true;
231 fileData.apply([fd, &success](const uint8_t* data, size_t size) {
232 if (writeToFile(fd, data, size) == -1) {
233 success = false;
234 return false;
235 }
236 return true;
237 });
238
239 return success;
240}

242static Expected<MappedData, std::error_code> compiledToFile(WTF::String&& json, Vector<WebCore::ContentExtensions::ContentExtensionRule>&& parsedRules, const WTF::String& finalFilePath)
243{
244 using namespace WebCore::ContentExtensions;
245
246 class CompilationClient final : public ContentExtensionCompilationClient {
247 public:
248 CompilationClient(PlatformFileHandle fileHandle, ContentRuleListMetaData& metaData)
249 : m_fileHandle(fileHandle)
250 , m_metaData(metaData)
251 {
252 ASSERT(!metaData.sourceSize);
253 ASSERT(!metaData.actionsSize);
254 ASSERT(!metaData.filtersWithoutConditionsBytecodeSize);
255 ASSERT(!metaData.filtersWithConditionsBytecodeSize);
256 ASSERT(!metaData.conditionedFiltersBytecodeSize);
257 ASSERT(!metaData.conditionsApplyOnlyToDomain);
258 }
259
260 void writeSource(WTF::String&& sourceJSON) final
261 {
262 ASSERT(!m_filtersWithoutConditionsBytecodeWritten);
263 ASSERT(!m_filtersWithConditionBytecodeWritten);
264 ASSERT(!m_conditionFiltersBytecodeWritten);
265 ASSERT(!m_actionsWritten);
266 ASSERT(!m_sourceWritten);
267 writeToFile(sourceJSON.is8Bit());
268 m_sourceWritten += sizeof(bool);
269 if (sourceJSON.is8Bit()) {
270 size_t serializedLength = sourceJSON.length() * sizeof(LChar);
271 writeToFile(WebKit::NetworkCache::Data(sourceJSON.characters8(), serializedLength));
272 m_sourceWritten += serializedLength;
273 } else {
274 size_t serializedLength = sourceJSON.length() * sizeof(UChar);
275 writeToFile(WebKit::NetworkCache::Data(reinterpret_cast<const uint8_t*>(sourceJSON.characters16()), serializedLength));
276 m_sourceWritten += serializedLength;
277 }
278 }
279
280 void writeFiltersWithoutConditionsBytecode(Vector<DFABytecode>&& bytecode) final
281 {
282 ASSERT(!m_filtersWithConditionBytecodeWritten);
283 ASSERT(!m_conditionFiltersBytecodeWritten);
284 m_filtersWithoutConditionsBytecodeWritten += bytecode.size();
285 writeToFile(WebKit::NetworkCache::Data(bytecode.data(), bytecode.size()));
286 }
287
288 void writeFiltersWithConditionsBytecode(Vector<DFABytecode>&& bytecode) final
289 {
290 ASSERT(!m_conditionFiltersBytecodeWritten);
291 m_filtersWithConditionBytecodeWritten += bytecode.size();
292 writeToFile(WebKit::NetworkCache::Data(bytecode.data(), bytecode.size()));
293 }
294
295 void writeTopURLFiltersBytecode(Vector<DFABytecode>&& bytecode) final
296 {
297 m_conditionFiltersBytecodeWritten += bytecode.size();
298 writeToFile(WebKit::NetworkCache::Data(bytecode.data(), bytecode.size()));
299 }
300
301 void writeActions(Vector<SerializedActionByte>&& actions, bool conditionsApplyOnlyToDomain) final
302 {
303 ASSERT(!m_filtersWithoutConditionsBytecodeWritten);
304 ASSERT(!m_filtersWithConditionBytecodeWritten);
305 ASSERT(!m_conditionFiltersBytecodeWritten);
306 ASSERT(!m_actionsWritten);
307 m_actionsWritten += actions.size();
308 m_conditionsApplyOnlyToDomain = conditionsApplyOnlyToDomain;
309 writeToFile(WebKit::NetworkCache::Data(actions.data(), actions.size()));
310 }
311
312 void finalize() final
313 {
314 m_metaData.sourceSize = m_sourceWritten;
315 m_metaData.actionsSize = m_actionsWritten;
316 m_metaData.filtersWithoutConditionsBytecodeSize = m_filtersWithoutConditionsBytecodeWritten;
317 m_metaData.filtersWithConditionsBytecodeSize = m_filtersWithConditionBytecodeWritten;
318 m_metaData.conditionedFiltersBytecodeSize = m_conditionFiltersBytecodeWritten;
319 m_metaData.conditionsApplyOnlyToDomain = m_conditionsApplyOnlyToDomain;
320
321 WebKit::NetworkCache::Data header = encodeContentRuleListMetaData(m_metaData);
322 if (!m_fileError && seekFile(m_fileHandle, 0ll, FileSeekOrigin::Beginning) == -1) {
323 closeFile(m_fileHandle);
324 m_fileError = true;
325 }
326 writeToFile(header);
327 }
328
329 bool hadErrorWhileWritingToFile() { return m_fileError; }
330
331 private:
332 void writeToFile(bool value)
333 {
334 writeToFile(WebKit::NetworkCache::Data(reinterpret_cast<const uint8_t*>(&value), sizeof(value)));
335 }
336 void writeToFile(const WebKit::NetworkCache::Data& data)
337 {
338 if (!m_fileError && !writeDataToFile(data, m_fileHandle)) {
339 closeFile(m_fileHandle);
340 m_fileError = true;
341 }
342 }
343
344 PlatformFileHandle m_fileHandle;
345 ContentRuleListMetaData& m_metaData;
346 size_t m_filtersWithoutConditionsBytecodeWritten { 0 };
347 size_t m_filtersWithConditionBytecodeWritten { 0 };
348 size_t m_conditionFiltersBytecodeWritten { 0 };
349 size_t m_actionsWritten { 0 };
350 size_t m_sourceWritten { 0 };
351 bool m_conditionsApplyOnlyToDomain { false };
352 bool m_fileError { false };
353 };
354
355 auto temporaryFileHandle = invalidPlatformFileHandle;
356 WTF::String temporaryFilePath = openTemporaryFile("ContentRuleList", temporaryFileHandle);
357 if (temporaryFileHandle == invalidPlatformFileHandle) {
358 WTFLogAlways("Content Rule List compiling failed: Opening temporary file failed.");
359 return makeUnexpected(ContentRuleListStore::Error::CompileFailed);
360 }
361
362 char invalidHeader[ContentRuleListFileHeaderSize];
363 memset(invalidHeader, 0xFF, sizeof(invalidHeader));
364 // This header will be rewritten in CompilationClient::finalize.
365 if (writeToFile(temporaryFileHandle, invalidHeader, sizeof(invalidHeader)) == -1) {
366 WTFLogAlways("Content Rule List compiling failed: Writing header to file failed.");
367 closeFile(temporaryFileHandle);
368 return makeUnexpected(ContentRuleListStore::Error::CompileFailed);
369 }
370
371 ContentRuleListMetaData metaData;
372 CompilationClient compilationClient(temporaryFileHandle, metaData);
373
374 if (auto compilerError = compileRuleList(compilationClient, WTFMove(json), WTFMove(parsedRules))) {
375 WTFLogAlways("Content Rule List compiling failed: Compiling failed.");
376 closeFile(temporaryFileHandle);
377 return makeUnexpected(compilerError);
378 }
379 if (compilationClient.hadErrorWhileWritingToFile()) {
380 WTFLogAlways("Content Rule List compiling failed: Writing to file failed.");
381 closeFile(temporaryFileHandle);
382 return makeUnexpected(ContentRuleListStore::Error::CompileFailed);
383 }
384
385 auto mappedData = adoptAndMapFile(temporaryFileHandle, 0, metaData.fileSize());
386 if (mappedData.isNull()) {
387 WTFLogAlways("Content Rule List compiling failed: Mapping file failed.");
388 return makeUnexpected(ContentRuleListStore::Error::CompileFailed);
389 }
390
391 // Try and delete any files at the destination instead of overwriting them
392 // in case there is already a file there and it is mmapped.
393 deleteFile(finalFilePath);
394
395 if (!moveFile(temporaryFilePath, finalFilePath)) {
396 WTFLogAlways("Content Rule List compiling failed: Moving file failed.");
397 return makeUnexpected(ContentRuleListStore::Error::CompileFailed);
398 }
399
400 FileSystem::makeSafeToUseMemoryMapForPath(finalFilePath);
401
402 return {{ WTFMove(metaData), WTFMove(mappedData) }};
403}

405static Ref<API::ContentRuleList> createExtension(const WTF::String& identifier, MappedData&& data)
406{
407 auto sharedMemory = data.data.tryCreateSharedMemory();
408
409 // Content extensions are always compiled to files, and at this point the file
410 // has been already mapped, therefore tryCreateSharedMemory() cannot fail.
411 ASSERT(sharedMemory);
412
413 const size_t headerAndSourceSize = ContentRuleListFileHeaderSize + data.metaData.sourceSize;
414 auto compiledContentRuleListData = WebKit::WebCompiledContentRuleListData(
415 WTFMove(sharedMemory),
416 ConditionsApplyOnlyToDomainOffset,
417 headerAndSourceSize,
418 data.metaData.actionsSize,
419 headerAndSourceSize
420 + data.metaData.actionsSize,
421 data.metaData.filtersWithoutConditionsBytecodeSize,
422 headerAndSourceSize
423 + data.metaData.actionsSize
424 + data.metaData.filtersWithoutConditionsBytecodeSize,
425 data.metaData.filtersWithConditionsBytecodeSize,
426 headerAndSourceSize
427 + data.metaData.actionsSize
428 + data.metaData.filtersWithoutConditionsBytecodeSize
429 + data.metaData.filtersWithConditionsBytecodeSize,
430 data.metaData.conditionedFiltersBytecodeSize
431 );
432 auto compiledContentRuleList = WebKit::WebCompiledContentRuleList::create(WTFMove(compiledContentRuleListData));
433 return API::ContentRuleList::create(identifier, WTFMove(compiledContentRuleList), WTFMove(data.data));
434}

436static WTF::String getContentRuleListSourceFromMappedFile(const MappedData& mappedData)
437{
438 ASSERT(!RunLoop::isMain());
439
440 switch (mappedData.metaData.version) {
441 case 9:
442 case 10:
443 case 11:
444 if (!mappedData.metaData.sourceSize)
445 return { };
446 bool is8Bit = mappedData.data.data()[ContentRuleListFileHeaderSize];
447 size_t start = ContentRuleListFileHeaderSize + sizeof(bool);
448 size_t length = mappedData.metaData.sourceSize - sizeof(bool);
449 if (is8Bit)
450 return WTF::String(mappedData.data.data() + start, length);
451 if (length % sizeof(UChar)) {
452 ASSERT_NOT_REACHED();
453 return { };
454 }
455 return WTF::String(reinterpret_cast<const UChar*>(mappedData.data.data() + start), length / sizeof(UChar));
456 }
457
458 // Older versions cannot recover the original JSON source from disk.
459 return { };
460}

462void ContentRuleListStore::lookupContentRuleList(const WTF::String& identifier, CompletionHandler<void(RefPtr<API::ContentRuleList>, std::error_code)> completionHandler)
463{
464 ASSERT(RunLoop::isMain());
465 m_readQueue->dispatch([protectedThis = makeRef(*this), identifier = identifier.isolatedCopy(), storePath = m_storePath.isolatedCopy(), completionHandler = WTFMove(completionHandler)]() mutable {
466 auto path = constructedPath(storePath, identifier, false);
467 auto legacyPath = constructedPath(storePath, identifier, true);
468 if (fileExists(legacyPath)) {
469 // Try and delete any files at the destination instead of overwriting them
470 // in case there is already a file there and it is mmapped.
471 deleteFile(path);
472 if (!moveFile(legacyPath, path)) {
473 WTFLogAlways("Content Rule List lookup failed: Moving a legacy file failed.");
474 if (!deleteFile(legacyPath))
475 WTFLogAlways("Content Rule List lookup failed: Deleting a legacy file failed.");
476 RunLoop::main().dispatch([protectedThis = WTFMove(protectedThis), completionHandler = WTFMove(completionHandler)] () mutable {
477 completionHandler(nullptr, Error::LookupFailed);
478 });
479 return;
480 }
481 }
482
483 auto contentRuleList = openAndMapContentRuleList(path);
484 if (!contentRuleList) {
485 RunLoop::main().dispatch([protectedThis = WTFMove(protectedThis), completionHandler = WTFMove(completionHandler)] () mutable {
486 completionHandler(nullptr, Error::LookupFailed);
487 });
488 return;
489 }
490
491 if (contentRuleList->metaData.version != ContentRuleListStore::CurrentContentRuleListFileVersion) {
492 if (auto sourceFromOldVersion = getContentRuleListSourceFromMappedFile(*contentRuleList); !sourceFromOldVersion.isEmpty()) {
493 RunLoop::main().dispatch([protectedThis = WTFMove(protectedThis), sourceFromOldVersion = sourceFromOldVersion.isolatedCopy(), identifier = identifier.isolatedCopy(), completionHandler = WTFMove(completionHandler)] () mutable {
494 protectedThis->compileContentRuleList(identifier, WTFMove(sourceFromOldVersion), WTFMove(completionHandler));
495 });
496 return;
497 }
498 RunLoop::main().dispatch([protectedThis = WTFMove(protectedThis), completionHandler = WTFMove(completionHandler)] () mutable {
499 completionHandler(nullptr, Error::VersionMismatch);
500 });
501 return;
502 }
503
504 RunLoop::main().dispatch([protectedThis = WTFMove(protectedThis), identifier = identifier.isolatedCopy(), contentRuleList = WTFMove(*contentRuleList), completionHandler = WTFMove(completionHandler)] () mutable {
505 completionHandler(createExtension(identifier, WTFMove(contentRuleList)), { });
506 });
507 });
508}

510void ContentRuleListStore::getAvailableContentRuleListIdentifiers(CompletionHandler<void(WTF::Vector<WTF::String>)> completionHandler)
511{
512 ASSERT(RunLoop::isMain());
513 m_readQueue->dispatch([protectedThis = makeRef(*this), storePath = m_storePath.isolatedCopy(), completionHandler = WTFMove(completionHandler)]() mutable {
514 auto prefix = constructedPathPrefix(false /*legacy*/);
515 auto prefixLength = prefix.length();
516 auto legacyPrefix = constructedPathPrefix(true /*legacy*/);
517 auto legacyPrefixLength = legacyPrefix.length();
518
519 Vector<WTF::String> identifiers;
520 for (auto& fileName : listDirectory(storePath)) {
521 if (fileName.startsWith(prefix))
522 identifiers.append(decodeFromFilename(fileName.substring(prefixLength)));
523 else if (fileName.startsWith(legacyPrefix))
524 identifiers.append(decodeFromFilename(fileName.substring(legacyPrefixLength)));
525 }
526
527 RunLoop::main().dispatch([protectedThis = WTFMove(protectedThis), completionHandler = WTFMove(completionHandler), identifiers = WTFMove(identifiers)]() mutable {
528 completionHandler(WTFMove(identifiers));
529 });
530 });
531}

533void ContentRuleListStore::compileContentRuleList(const WTF::String& identifier, WTF::String&& json, CompletionHandler<void(RefPtr<API::ContentRuleList>, std::error_code)> completionHandler)
534{
535 ASSERT(RunLoop::isMain());
536 AtomString::init();
537 WebCore::QualifiedName::init();
538
539 auto parsedRules = WebCore::ContentExtensions::parseRuleList(json);
540 if (!parsedRules.has_value())
541 return completionHandler(nullptr, parsedRules.error());
542
543 m_compileQueue->dispatch([protectedThis = makeRef(*this), identifier = identifier.isolatedCopy(), json = json.isolatedCopy(), parsedRules = parsedRules.value().isolatedCopy(), storePath = m_storePath.isolatedCopy(), completionHandler = WTFMove(completionHandler)] () mutable {
544 auto path = constructedPath(storePath, identifier, false);
545
546 auto result = compiledToFile(WTFMove(json), WTFMove(parsedRules), path);
547 if (!result.has_value()) {
548 RunLoop::main().dispatch([protectedThis = WTFMove(protectedThis), error = WTFMove(result.error()), completionHandler = WTFMove(completionHandler)] () mutable {
549 completionHandler(nullptr, error);
550 });
551 return;
552 }
553
554 RunLoop::main().dispatch([protectedThis = WTFMove(protectedThis), identifier = WTFMove(identifier), data = WTFMove(result.value()), completionHandler = WTFMove(completionHandler)] () mutable {
555 auto contentRuleList = createExtension(identifier, WTFMove(data));
556 completionHandler(contentRuleList.ptr(), { });
557 });
558 });
559}

561void ContentRuleListStore::removeContentRuleList(const WTF::String& identifier, CompletionHandler<void(std::error_code)> completionHandler)
562{
563 ASSERT(RunLoop::isMain());
564 m_removeQueue->dispatch([protectedThis = makeRef(*this), identifier = identifier.isolatedCopy(), storePath = m_storePath.isolatedCopy(), completionHandler = WTFMove(completionHandler)]() mutable {

566 RunLoop::main().dispatch([protectedThis = WTFMove(protectedThis), completionHandler = WTFMove(completionHandler), error = WTFMove(error)] () mutable {

574 }
575 if (auto path = constructedPath(storePath, identifier, false); !deleteFile(path))
576 return complete(Error::RemoveFailed);
577 complete({ });
578 });

587void ContentRuleListStore::invalidateContentRuleListVersion(const WTF::String& identifier)
588{
589 auto file = openFile(constructedPath(m_storePath, identifier, false), FileOpenMode::Write);
590 if (file == invalidPlatformFileHandle)

598void ContentRuleListStore::getContentRuleListSource(const WTF::String& identifier, CompletionHandler<void(WTF::String)> completionHandler)
599{
600 ASSERT(RunLoop::isMain());
601 m_readQueue->dispatch([protectedThis = makeRef(*this), identifier = identifier.isolatedCopy(), storePath = m_storePath.isolatedCopy(), completionHandler = WTFMove(completionHandler)]() mutable {
602 auto complete = [protectedThis = WTFMove(protectedThis), completionHandler = WTFMove(completionHandler)](WTF::String source) mutable {
603 RunLoop::main().dispatch([protectedThis = WTFMove(protectedThis), completionHandler = WTFMove(completionHandler), source = source.isolatedCopy()] () mutable {
604 completionHandler(source);
605 });
606 };
607
608 auto contentRuleList = openAndMapContentRuleList(constructedPath(storePath, identifier, false));
609 if (!contentRuleList) {
610 auto legacyContentRuleList = openAndMapContentRuleList(constructedPath(storePath, identifier, true));
611 if (!legacyContentRuleList)
612 return complete({ });
613 return complete(getContentRuleListSourceFromMappedFile(*legacyContentRuleList));
614 }
615 complete(getContentRuleListSourceFromMappedFile(*contentRuleList));
616 });
617}

619const std::error_category& contentRuleListStoreErrorCategory()
620{
621 class ContentRuleListStoreErrorCategory : public std::error_category {
622 const char* name() const noexcept final
623 {
624 return "content extension store";
625 }
626
627 std::string message(int errorCode) const final
