26#pragma once
27
28#include "APIObject.h"
29#include <WebCore/CustomHeaderFields.h>
30
31namespace API {
32
33class CustomHeaderFields final : public ObjectImpl<Object::Type::CustomHeaderFields> {
34public:
35 template<typename... Args> static Ref<CustomHeaderFields> create(Args&&... args)
36 {
37 return adoptRef(*new CustomHeaderFields(std::forward<Args>(args)...));
38 }
39
40 CustomHeaderFields() = default;
41
42 const Vector<WebCore::HTTPHeaderField>& fields() const { return m_fields.fields; }
43 void setFields(Vector<WebCore::HTTPHeaderField>&& fields) { m_fields.fields = WTFMove(fields); }
44
45 const Vector<WTF::String> thirdPartyDomains() const { return m_fields.thirdPartyDomains; }
46 void setThirdPartyDomains(Vector<WTF::String>&& domains) { m_fields.thirdPartyDomains = WTFMove(domains); }
47
48 const WebCore::CustomHeaderFields& coreFields() const { return m_fields; }
49
50private:
51 CustomHeaderFields(const WebCore::CustomHeaderFields& fields)
52 : m_fields(fields) { }
53
54 WebCore::CustomHeaderFields m_fields;
55};
56
57} // namespace API
