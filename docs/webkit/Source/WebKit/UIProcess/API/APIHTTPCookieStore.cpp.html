58void HTTPCookieStore::filterAppBoundCookies(Vector<WebCore::Cookie>&& cookies, CompletionHandler<void(Vector<WebCore::Cookie>&&)>&& completionHandler)
59{
60#if ENABLE(APP_BOUND_DOMAINS)
61 if (!m_owningDataStore)
62 return completionHandler({ });
63 m_owningDataStore->getAppBoundDomains([cookies = WTFMove(cookies), completionHandler = WTFMove(completionHandler)] (auto& domains) mutable {
64 Vector<WebCore::Cookie> appBoundCookies;
65 if (!domains.isEmpty() && !isFullWebBrowser()) {
66 for (auto& cookie : WTFMove(cookies)) {
67 if (domains.contains(WebCore::RegistrableDomain::uncheckedCreateFromHost(cookie.domain)))
68 appBoundCookies.append(WTFMove(cookie));
69 }
70 } else
71 appBoundCookies = WTFMove(cookies);
72 completionHandler(WTFMove(appBoundCookies));
73 });
74#else
75 completionHandler(WTFMove(cookies));
76#endif
77}
78
79void HTTPCookieStore::cookies(CompletionHandler<void(const Vector<WebCore::Cookie>&)>&& completionHandler)
80{
81 if (!m_owningDataStore)
82 return completionHandler({ });
83 auto& cookieManager = m_owningDataStore->networkProcess().cookieManager();
84 cookieManager.getAllCookies(m_owningDataStore->sessionID(), [this, protectedThis = makeRef(*this), completionHandler = WTFMove(completionHandler)] (Vector<WebCore::Cookie>&& cookies) mutable {
85 filterAppBoundCookies(WTFMove(cookies), WTFMove(completionHandler));
86 });
87}

89void HTTPCookieStore::cookiesForURL(WTF::URL&& url, CompletionHandler<void(Vector<WebCore::Cookie>&&)>&& completionHandler)
90{
91 if (!m_owningDataStore)
92 return completionHandler({ });
93 auto& cookieManager = m_owningDataStore->networkProcess().cookieManager();
94 cookieManager.getCookies(m_owningDataStore->sessionID(), url, [this, protectedThis = makeRef(*this), completionHandler = WTFMove(completionHandler)] (Vector<WebCore::Cookie>&& cookies) mutable {
95 filterAppBoundCookies(WTFMove(cookies), WTFMove(completionHandler));
96 });
97}

99void HTTPCookieStore::setCookies(Vector<WebCore::Cookie>&& cookies, CompletionHandler<void()>&& completionHandler)
100{
101 filterAppBoundCookies(WTFMove(cookies), [this, protectedThis = makeRef(*this), completionHandler = WTFMove(completionHandler)] (auto&& appBoundCookies) mutable {
102 if (!m_owningDataStore)
103 return;
104 auto& cookieManager = m_owningDataStore->networkProcess().cookieManager();
105 cookieManager.setCookies(m_owningDataStore->sessionID(), appBoundCookies, WTFMove(completionHandler));
106 });
107}

109void HTTPCookieStore::deleteCookie(const WebCore::Cookie& cookie, CompletionHandler<void()>&& completionHandler)
110{
111 if (!m_owningDataStore)
112 return completionHandler();
113 auto& cookieManager = m_owningDataStore->networkProcess().cookieManager();
114 cookieManager.deleteCookie(m_owningDataStore->sessionID(), cookie, WTFMove(completionHandler));
115}

117void HTTPCookieStore::deleteAllCookies(CompletionHandler<void()>&& completionHandler)
118{
119 if (!m_owningDataStore)
120 return completionHandler();
121 auto& cookieManager = m_owningDataStore->networkProcess().cookieManager();
122 cookieManager.deleteAllCookies(m_owningDataStore->sessionID());

142class APIWebCookieManagerProxyObserver : public WebCookieManagerProxy::Observer {
143 WTF_MAKE_FAST_ALLOCATED;
144public:
145 explicit APIWebCookieManagerProxyObserver(API::HTTPCookieStore& cookieStore)
146 : m_cookieStore(cookieStore)
147 {
148 }
149
150private:
151 void cookiesDidChange() final
152 {
153 m_cookieStore.cookiesDidChange();
154 }
155
156 API::HTTPCookieStore& m_cookieStore;
157};

159void HTTPCookieStore::registerObserver(Observer& observer)
160{
161 m_observers.add(&observer);
162
163 if (m_cookieManagerProxyObserver || !m_owningDataStore)
164 return;
165
166 ASSERT(!m_observedCookieManagerProxy);
167
168 m_cookieManagerProxyObserver = makeUnique<APIWebCookieManagerProxyObserver>(*this);
169
170 m_observedCookieManagerProxy = makeWeakPtr(m_owningDataStore->networkProcess().cookieManager());
171 m_observedCookieManagerProxy->registerObserver(m_owningDataStore->sessionID(), *m_cookieManagerProxyObserver);
172}
173
174void HTTPCookieStore::unregisterObserver(Observer& observer)
175{
176 m_observers.remove(observer);
177
178 if (!m_observers.computesEmpty())
179 return;
180
181 if (m_observedCookieManagerProxy && m_owningDataStore)
182 m_observedCookieManagerProxy->unregisterObserver(m_owningDataStore->sessionID(), *m_cookieManagerProxyObserver);
183
184 m_observedCookieManagerProxy = nullptr;
185 m_cookieManagerProxyObserver = nullptr;
186}
187
188void HTTPCookieStore::cookiesDidChange()
189{
190 for (auto& observer : m_observers)
191 observer.cookiesDidChange(*this);
192}
