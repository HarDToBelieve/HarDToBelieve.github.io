52struct SubstituteData {
53 WTF_MAKE_FAST_ALLOCATED;
54public:
55 SubstituteData(Vector<uint8_t>&& content, const WTF::String& MIMEType, const WTF::String& encoding, const WTF::String& baseURL, API::Object* userData)
56 : content(WTFMove(content))
57 , MIMEType(MIMEType)
58 , encoding(encoding)
59 , baseURL(baseURL)
60 , userData(userData)
61 { }
62
63 SubstituteData(Vector<uint8_t>&& content, const WebCore::ResourceResponse&, API::Object* userData);
64
65 Vector<uint8_t> content;
66 WTF::String MIMEType;
67 WTF::String encoding;
68 WTF::String baseURL;
69 RefPtr<API::Object> userData;
70};

72class Navigation : public ObjectImpl<Object::Type::Navigation> {
73 WTF_MAKE_NONCOPYABLE(Navigation);
74public:
75 static Ref<Navigation> create(WebKit::WebNavigationState& state, WebKit::WebBackForwardListItem* currentAndTargetItem)
76 {
77 return adoptRef(*new Navigation(state, currentAndTargetItem));
78 }
79
80 static Ref<Navigation> create(WebKit::WebNavigationState& state, WebKit::WebBackForwardListItem& targetItem, WebKit::WebBackForwardListItem* fromItem, WebCore::FrameLoadType backForwardFrameLoadType)
81 {
82 return adoptRef(*new Navigation(state, targetItem, fromItem, backForwardFrameLoadType));
83 }
84
85 static Ref<Navigation> create(WebKit::WebNavigationState& state, WebCore::ResourceRequest&& request, WebKit::WebBackForwardListItem* fromItem)
86 {
87 return adoptRef(*new Navigation(state, WTFMove(request), fromItem));
88 }
89
90 static Ref<Navigation> create(WebKit::WebNavigationState& state, std::unique_ptr<SubstituteData>&& substituteData)
91 {
92 return adoptRef(*new Navigation(state, WTFMove(substituteData)));
93 }
94
95 static Ref<Navigation> create(WebKit::WebNavigationState& state, WebCore::ResourceRequest&& simulatedRequest, std::unique_ptr<SubstituteData>&& substituteData, WebKit::WebBackForwardListItem* fromItem)
96 {
97 return adoptRef(*new Navigation(state, WTFMove(simulatedRequest), WTFMove(substituteData), fromItem));
98 }
99
100 virtual ~Navigation();
101
102 uint64_t navigationID() const { return m_navigationID; }
103
104 const WebCore::ResourceRequest& originalRequest() const { return m_originalRequest; }
105 void setCurrentRequest(WebCore::ResourceRequest&&, WebCore::ProcessIdentifier);
106 const WebCore::ResourceRequest& currentRequest() const { return m_currentRequest; }
107 std::optional<WebCore::ProcessIdentifier> currentRequestProcessIdentifier() const { return m_currentRequestProcessIdentifier; }
108
109 bool currentRequestIsRedirect() const { return m_lastNavigationAction.isRedirect; }
110
111 WebKit::WebBackForwardListItem* targetItem() const { return m_targetItem.get(); }
112 WebKit::WebBackForwardListItem* fromItem() const { return m_fromItem.get(); }
113 std::optional<WebCore::FrameLoadType> backForwardFrameLoadType() const { return m_backForwardFrameLoadType; }
114 WebKit::WebBackForwardListItem* reloadItem() const { return m_reloadItem.get(); }
115
116 void appendRedirectionURL(const WTF::URL&);
117 Vector<WTF::URL> takeRedirectChain() { return WTFMove(m_redirectChain); }
118
119 bool wasUserInitiated() const { return !!m_lastNavigationAction.userGestureTokenIdentifier; }
120
121 bool shouldPerformDownload() const { return !m_lastNavigationAction.downloadAttribute.isNull(); }
122
123 bool isSystemPreview() const
124 {
125#if USE(SYSTEM_PREVIEW)
126 return currentRequest().isSystemPreview();
127#else
128 return false;
129#endif
130 }
131
132 bool treatAsSameOriginNavigation() const { return m_lastNavigationAction.treatAsSameOriginNavigation; }
133 bool hasOpenedFrames() const { return m_lastNavigationAction.hasOpenedFrames; }
134 bool openedByDOMWithOpener() const { return m_lastNavigationAction.openedByDOMWithOpener; }
135 const WebCore::SecurityOriginData& requesterOrigin() const { return m_lastNavigationAction.requesterOrigin; }
136
137 void setUserContentExtensionsEnabled(bool enabled) { m_userContentExtensionsEnabled = enabled; }
138 bool userContentExtensionsEnabled() const { return m_userContentExtensionsEnabled; }
139
140 WebCore::LockHistory lockHistory() const { return m_lastNavigationAction.lockHistory; }
141 WebCore::LockBackForwardList lockBackForwardList() const { return m_lastNavigationAction.lockBackForwardList; }
142
143 WTF::String clientRedirectSourceForHistory() const { return m_lastNavigationAction.clientRedirectSourceForHistory; }
144
145 void setLastNavigationAction(const WebKit::NavigationActionData& navigationAction) { m_lastNavigationAction = navigationAction; }
146 const WebKit::NavigationActionData& lastNavigationAction() const { return m_lastNavigationAction; }
147
148 void setOriginatingFrameInfo(const WebKit::FrameInfoData& frameInfo) { m_originatingFrameInfo = frameInfo; }
149 const WebKit::FrameInfoData& originatingFrameInfo() const { return m_originatingFrameInfo; }
150
151 void setDestinationFrameSecurityOrigin(const WebCore::SecurityOriginData& origin) { m_destinationFrameSecurityOrigin = origin; }
152 const WebCore::SecurityOriginData& destinationFrameSecurityOrigin() const { return m_destinationFrameSecurityOrigin; }
153
154 void setEffectiveContentMode(WebKit::WebContentMode mode) { m_effectiveContentMode = mode; }
155 WebKit::WebContentMode effectiveContentMode() const { return m_effectiveContentMode; }
156
157#if !LOG_DISABLED
158 const char* loggingString() const;
159#endif
160
161 const std::unique_ptr<SubstituteData>& substituteData() const { return m_substituteData; }
162
163 const std::optional<WebCore::PrivateClickMeasurement>& privateClickMeasurement() const { return m_lastNavigationAction.privateClickMeasurement; }
164
165 void setClientNavigationActivity(WebKit::ProcessThrottler::ActivityVariant&& activity) { m_clientNavigationActivity = WTFMove(activity); }
166
167 void setIsLoadedWithNavigationShared(bool value) { m_isLoadedWithNavigationShared = value; }
168 bool isLoadedWithNavigationShared() const { return m_isLoadedWithNavigationShared; }
169
170private:
171 explicit Navigation(WebKit::WebNavigationState&);
172 Navigation(WebKit::WebNavigationState&, WebKit::WebBackForwardListItem*);
173 Navigation(WebKit::WebNavigationState&, WebCore::ResourceRequest&&, WebKit::WebBackForwardListItem* fromItem);
174 Navigation(WebKit::WebNavigationState&, WebKit::WebBackForwardListItem& targetItem, WebKit::WebBackForwardListItem* fromItem, WebCore::FrameLoadType);
175 Navigation(WebKit::WebNavigationState&, std::unique_ptr<SubstituteData>&&);
176 Navigation(WebKit::WebNavigationState&, WebCore::ResourceRequest&&, std::unique_ptr<SubstituteData>&&, WebKit::WebBackForwardListItem* fromItem);
177
178 uint64_t m_navigationID;
179 WebCore::ResourceRequest m_originalRequest;
180 WebCore::ResourceRequest m_currentRequest;
181 std::optional<WebCore::ProcessIdentifier> m_currentRequestProcessIdentifier;
182 Vector<WTF::URL> m_redirectChain;
183
184 RefPtr<WebKit::WebBackForwardListItem> m_targetItem;
185 RefPtr<WebKit::WebBackForwardListItem> m_fromItem;
186 RefPtr<WebKit::WebBackForwardListItem> m_reloadItem;
187 std::optional<WebCore::FrameLoadType> m_backForwardFrameLoadType;
188 std::unique_ptr<SubstituteData> m_substituteData;
189 WebKit::NavigationActionData m_lastNavigationAction;
190 WebKit::FrameInfoData m_originatingFrameInfo;
191 WebCore::SecurityOriginData m_destinationFrameSecurityOrigin;
192 bool m_userContentExtensionsEnabled { true };
193 WebKit::WebContentMode m_effectiveContentMode { WebKit::WebContentMode::Recommended };
194 WebKit::ProcessThrottler::TimedActivity m_clientNavigationActivity;
195 bool m_isLoadedWithNavigationShared { false };
196};
