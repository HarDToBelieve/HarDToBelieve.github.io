26#pragma once
27
28#include "APIFrameInfo.h"
29#include "APIObject.h"
30#include <WebCore/ResourceRequest.h>
31#include <WebCore/ResourceResponse.h>
32
33namespace API {
34
35class FrameInfo;
36
37class NavigationResponse final : public ObjectImpl<Object::Type::NavigationResponse> {
38public:
39 template<typename... Args> static Ref<NavigationResponse> create(Args&&... args)
40 {
41 return adoptRef(*new NavigationResponse(std::forward<Args>(args)...));
42 }
43
44 FrameInfo& frame() { return m_frame.get(); }
45
46 const WebCore::ResourceRequest& request() const { return m_request; }
47 const WebCore::ResourceResponse& response() const { return m_response; }
48
49 bool canShowMIMEType() const { return m_canShowMIMEType; }
50 const WTF::String& downloadAttribute() const { return m_downloadAttribute; }
51
52private:
53 NavigationResponse(API::FrameInfo& frame, const WebCore::ResourceRequest& request, const WebCore::ResourceResponse& response, bool canShowMIMEType, const WTF::String& downloadAttribute)
54 : m_frame(frame)
55 , m_request(request)
56 , m_response(response)
57 , m_canShowMIMEType(canShowMIMEType)
58 , m_downloadAttribute(downloadAttribute) { }
59
60 Ref<FrameInfo> m_frame;
61
62 WebCore::ResourceRequest m_request;
63 WebCore::ResourceResponse m_response;
64 bool m_canShowMIMEType;
65 WTF::String m_downloadAttribute;
66};
67
68} // namespace API
