40class NotificationProvider {
41 WTF_MAKE_FAST_ALLOCATED;

43 virtual ~NotificationProvider() = default;
44
45 virtual void show(WebKit::WebPageProxy&, WebKit::WebNotification&) { }
46 virtual void cancel(WebKit::WebNotification&) { }

48 virtual void clearNotifications(const Vector<uint64_t>& /*notificationIDs*/) { }
49
50 virtual void addNotificationManager(WebKit::WebNotificationManagerProxy&) { }
51 virtual void removeNotificationManager(WebKit::WebNotificationManagerProxy&) { }
52
53 virtual HashMap<WTF::String, bool> notificationPermissions() { return HashMap<WTF::String, bool>(); };
54};
