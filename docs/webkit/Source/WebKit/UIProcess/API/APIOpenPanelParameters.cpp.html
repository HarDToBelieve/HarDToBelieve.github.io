66
67Ref<API::Array> OpenPanelParameters::allowedFileExtensions() const
68{
69#if PLATFORM(MAC)
70 auto acceptMIMETypes = m_settings.acceptMIMETypes;
71
72 // On macOS allow selecting HEIF/HEIC images if acceptMIMETypes or acceptFileExtensions include at least
73 // one MIME type which CG supports encoding to.
74 if (MIMETypeRegistry::containsImageMIMETypeForEncoding(acceptMIMETypes, m_settings.acceptFileExtensions)) {
75 acceptMIMETypes.append("image/heif"_s);
76 acceptMIMETypes.append("image/heic"_s);
77 }
78
79 return API::Array::createStringArray(WebCore::MIMETypeRegistry::allowedFileExtensions(acceptMIMETypes, m_settings.acceptFileExtensions));
80#else
81 return API::Array::createStringArray(WebCore::MIMETypeRegistry::allowedFileExtensions(m_settings.acceptMIMETypes, m_settings.acceptFileExtensions));
82#endif
83}
