56Ref<PageConfiguration> PageConfiguration::copy() const
57{
58 auto copy = create();
59
60 copy->m_processPool = this->m_processPool;
61 copy->m_userContentController = this->m_userContentController;
62 copy->m_pageGroup = this->m_pageGroup;
63 copy->m_preferences = this->m_preferences;
64 copy->m_relatedPage = this->m_relatedPage;
65 copy->m_visitedLinkStore = this->m_visitedLinkStore;
66 copy->m_websiteDataStore = this->m_websiteDataStore;
67#if PLATFORM(IOS_FAMILY)
68 copy->m_clientNavigationsRunAtForegroundPriority = this->m_clientNavigationsRunAtForegroundPriority;
69 copy->m_canShowWhileLocked = this->m_canShowWhileLocked;
70 copy->m_clickInteractionDriverForTesting = this->m_clickInteractionDriverForTesting;
71#endif
72 copy->m_initialCapitalizationEnabled = this->m_initialCapitalizationEnabled;
73 copy->m_waitsForPaintAfterViewDidMoveToWindow = this->m_waitsForPaintAfterViewDidMoveToWindow;
74 copy->m_drawsBackground = this->m_drawsBackground;
75 copy->m_controlledByAutomation = this->m_controlledByAutomation;
76 copy->m_cpuLimit = this->m_cpuLimit;
77 copy->m_overrideContentSecurityPolicy = this->m_overrideContentSecurityPolicy;
78#if ENABLE(APPLICATION_MANIFEST)
79 copy->m_applicationManifest = this->m_applicationManifest;
80#endif
81 copy->m_shouldRelaxThirdPartyCookieBlocking = this->m_shouldRelaxThirdPartyCookieBlocking;
82 copy->m_attributedBundleIdentifier = this->m_attributedBundleIdentifier;
83 for (auto& pair : this->m_urlSchemeHandlers)
84 copy->m_urlSchemeHandlers.set(pair.key, pair.value.copyRef());
85 copy->m_corsDisablingPatterns = this->m_corsDisablingPatterns;
86 copy->m_crossOriginAccessControlCheckEnabled = this->m_crossOriginAccessControlCheckEnabled;
87 copy->m_userScriptsShouldWaitUntilNotification = this->m_userScriptsShouldWaitUntilNotification;
88
89 copy->m_processDisplayName = this->m_processDisplayName;
90 copy->m_loadsSubresources = this->m_loadsSubresources;
91 copy->m_allowedNetworkHosts = this->m_allowedNetworkHosts;
92#if ENABLE(APP_BOUND_DOMAINS)
93 copy->m_ignoresAppBoundDomains = this->m_ignoresAppBoundDomains;
94 copy->m_limitsNavigationsToAppBoundDomains = this->m_limitsNavigationsToAppBoundDomains;
95#endif
96
97 copy->m_mediaCaptureEnabled = this->m_mediaCaptureEnabled;
98 copy->m_httpsUpgradeEnabled = this->m_httpsUpgradeEnabled;
99
100 return copy;
101}
