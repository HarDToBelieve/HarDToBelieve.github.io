37Ref<WebsitePolicies> WebsitePolicies::copy() const
38{
39 auto policies = WebsitePolicies::create();
40 policies->setContentBlockersEnabled(m_contentBlockersEnabled);
41 policies->setAllowedAutoplayQuirks(m_allowedAutoplayQuirks);
42 policies->setAutoplayPolicy(m_autoplayPolicy);
43#if ENABLE(DEVICE_ORIENTATION)
44 policies->setDeviceOrientationAndMotionAccessState(m_deviceOrientationAndMotionAccessState);
45#endif
46 policies->setPopUpPolicy(m_popUpPolicy);
47 policies->setWebsiteDataStore(m_websiteDataStore.get());
48 policies->setCustomUserAgent(m_customUserAgent);
49 policies->setCustomUserAgentAsSiteSpecificQuirks(m_customUserAgentAsSiteSpecificQuirks);
50 policies->setCustomNavigatorPlatform(m_customNavigatorPlatform);
51 policies->setPreferredContentMode(m_preferredContentMode);
52 policies->setMetaViewportPolicy(m_metaViewportPolicy);
53 policies->setMediaSourcePolicy(m_mediaSourcePolicy);
54 policies->setSimulatedMouseEventsDispatchPolicy(m_simulatedMouseEventsDispatchPolicy);
55 policies->setLegacyOverflowScrollingTouchPolicy(m_legacyOverflowScrollingTouchPolicy);
56 policies->setAllowContentChangeObserverQuirk(m_allowContentChangeObserverQuirk);
57 policies->setWebsiteDataStore(m_websiteDataStore.get());
58 policies->setUserContentController(m_userContentController.get());
59 policies->setIdempotentModeAutosizingOnlyHonorsPercentages(m_idempotentModeAutosizingOnlyHonorsPercentages);
60
61 Vector<WebCore::HTTPHeaderField> legacyCustomHeaderFields;
62 legacyCustomHeaderFields.reserveInitialCapacity(m_legacyCustomHeaderFields.size());
63 for (auto& field : m_legacyCustomHeaderFields)
64 legacyCustomHeaderFields.uncheckedAppend(field);
65 policies->setLegacyCustomHeaderFields(WTFMove(legacyCustomHeaderFields));
66
67 Vector<WebCore::CustomHeaderFields> customHeaderFields;
68 customHeaderFields.reserveInitialCapacity(m_customHeaderFields.size());
69 for (auto& field : m_customHeaderFields)
70 customHeaderFields.uncheckedAppend(field);
71 policies->setCustomHeaderFields(WTFMove(customHeaderFields));
72 policies->setAllowSiteSpecificQuirksToOverrideContentMode(m_allowSiteSpecificQuirksToOverrideContentMode);
73 policies->setApplicationNameForDesktopUserAgent(m_applicationNameForDesktopUserAgent);
74 policies->setAllowsContentJavaScript(m_allowsContentJavaScript);
75 policies->setMouseEventPolicy(m_mouseEventPolicy);
76 return policies;
77}

93WebKit::WebsitePoliciesData WebsitePolicies::data()
94{
95 bool hasLegacyCustomHeaderFields = legacyCustomHeaderFields().size();
96 Vector<WebCore::CustomHeaderFields> customHeaderFields;
97 customHeaderFields.reserveInitialCapacity(this->customHeaderFields().size() + hasLegacyCustomHeaderFields);
98 for (auto& field : this->customHeaderFields())
99 customHeaderFields.uncheckedAppend(field);
100 if (hasLegacyCustomHeaderFields)
101 customHeaderFields.uncheckedAppend({ legacyCustomHeaderFields(), { }});
102
103 return {
104 contentBlockersEnabled(),
105 allowedAutoplayQuirks(),
106 autoplayPolicy(),
107#if ENABLE(DEVICE_ORIENTATION)
108 deviceOrientationAndMotionAccessState(),
109#endif
110 WTFMove(customHeaderFields),
111 popUpPolicy(),
112 m_customUserAgent,
113 m_customUserAgentAsSiteSpecificQuirks,
114 m_customNavigatorPlatform,
115 m_metaViewportPolicy,
116 m_mediaSourcePolicy,
117 m_simulatedMouseEventsDispatchPolicy,
118 m_legacyOverflowScrollingTouchPolicy,
119 m_allowContentChangeObserverQuirk,
120 m_allowsContentJavaScript,
121 m_mouseEventPolicy,
122 m_idempotentModeAutosizingOnlyHonorsPercentages
123 };
124}
