35
36WKTypeID WKFrameInfoGetTypeID()
37{
38 return toAPI(API::FrameInfo::APIType);
39}
40
41WKFrameHandleRef WKFrameInfoCreateFrameHandleRef(WKFrameInfoRef frameInfoRef)
42{
43 return toAPI(&toImpl(frameInfoRef)->handle().leakRef());
44}
45
46WKSecurityOriginRef WKFrameInfoCopySecurityOrigin(WKFrameInfoRef frameInfoRef)
47{
48 auto origin = toImpl(frameInfoRef)->securityOrigin();
49 return toAPI(&API::SecurityOrigin::create(origin.protocol, origin.host, origin.port).leakRef());
50}
