34using namespace WebKit;
35
36WKTypeID WKGeolocationManagerGetTypeID()
37{
38 return toAPI(WebGeolocationManagerProxy::APIType);
39}
40
41void WKGeolocationManagerSetProvider(WKGeolocationManagerRef geolocationManagerRef, const WKGeolocationProviderBase* wkProvider)
42{
43 toImpl(geolocationManagerRef)->setProvider(makeUnique<WebGeolocationProvider>(wkProvider));
44}
45
46void WKGeolocationManagerProviderDidChangePosition(WKGeolocationManagerRef geolocationManagerRef, WKGeolocationPositionRef positionRef)
47{
48 toImpl(geolocationManagerRef)->providerDidChangePosition(toImpl(positionRef));
49}
50
51void WKGeolocationManagerProviderDidFailToDeterminePosition(WKGeolocationManagerRef geolocationManagerRef)
52{
53 toImpl(geolocationManagerRef)->providerDidFailToDeterminePosition();
54}
55
56void WKGeolocationManagerProviderDidFailToDeterminePositionWithErrorMessage(WKGeolocationManagerRef geolocationManagerRef, WKStringRef errorMessage)
57{
58 toImpl(geolocationManagerRef)->providerDidFailToDeterminePosition(toWTFString(errorMessage));
59}
