32WKTypeID WKHTTPCookieStoreGetTypeID()
33{
34 return WebKit::toAPI(API::HTTPCookieStore::APIType);
35}

37void WKHTTPCookieStoreDeleteAllCookies(WKHTTPCookieStoreRef cookieStore, void* context, WKHTTPCookieStoreDeleteAllCookiesFunction callback)
38{
39 WebKit::toImpl(cookieStore)->deleteAllCookies([context, callback] {
40 if (callback)
41 callback(context);
42 });
43}

46{
47 WebKit::toImpl(cookieStore)->setHTTPCookieAcceptPolicy(WebKit::toHTTPCookieAcceptPolicy(policy), [context, callback] {
48 if (callback)
49 callback(context);
50 });
51}
