37void WKAddMockMediaDevice(WKContextRef context, WKStringRef persistentId, WKStringRef label, WKStringRef type)
38{
39#if ENABLE(MEDIA_STREAM)
40 String typeString = WebKit::toImpl(type)->string();
41 Variant<WebCore::MockMicrophoneProperties, WebCore::MockSpeakerProperties, WebCore::MockCameraProperties, WebCore::MockDisplayProperties> properties;
42 if (typeString == "camera")
43 properties = WebCore::MockCameraProperties { };
44 else if (typeString == "screen")
45 properties = WebCore::MockDisplayProperties { };
46 else if (typeString == "speaker")
47 properties = WebCore::MockSpeakerProperties { };
48 else if (typeString != "microphone")
49 return;
50
51 toImpl(context)->addMockMediaDevice({ WebKit::toImpl(persistentId)->string(), WebKit::toImpl(label)->string(), WTFMove(properties) });
52#endif
53}
54
55void WKClearMockMediaDevices(WKContextRef context)
56{
57 toImpl(context)->clearMockMediaDevices();
58}
59
60void WKRemoveMockMediaDevice(WKContextRef context, WKStringRef persistentId)
61{
62 toImpl(context)->removeMockMediaDevice(WebKit::toImpl(persistentId)->string());
63}
64
65void WKResetMockMediaDevices(WKContextRef context)
66{
67 toImpl(context)->resetMockMediaDevices();
68}
