37WKTypeID WKNotificationManagerGetTypeID()
38{
39 return toAPI(WebNotificationManagerProxy::APIType);
40}

42void WKNotificationManagerSetProvider(WKNotificationManagerRef managerRef, const WKNotificationProviderBase* wkProvider)
43{
44 toImpl(managerRef)->setProvider(makeUnique<WebNotificationProvider>(wkProvider));
45}
46
47void WKNotificationManagerProviderDidShowNotification(WKNotificationManagerRef managerRef, uint64_t notificationID)
48{
49 toImpl(managerRef)->providerDidShowNotification(notificationID);
50}
51
52void WKNotificationManagerProviderDidClickNotification(WKNotificationManagerRef managerRef, uint64_t notificationID)
53{
54 toImpl(managerRef)->providerDidClickNotification(notificationID);
55}
56
57void WKNotificationManagerProviderDidCloseNotifications(WKNotificationManagerRef managerRef, WKArrayRef notificationIDs)
58{
59 toImpl(managerRef)->providerDidCloseNotifications(toImpl(notificationIDs));
60}

62void WKNotificationManagerProviderDidUpdateNotificationPolicy(WKNotificationManagerRef managerRef, WKSecurityOriginRef origin, bool allowed)

67void WKNotificationManagerProviderDidRemoveNotificationPolicies(WKNotificationManagerRef managerRef, WKArrayRef origins)

72uint64_t WKNotificationManagerGetLocalIDForTesting(WKNotificationManagerRef manager, WKNotificationRef notification)
