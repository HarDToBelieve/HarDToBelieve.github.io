26#pragma once
27
28#include "WKPageRenderingProgressEvents.h"
29
30#include <WebCore/LayoutMilestone.h>
31
32static inline WKPageRenderingProgressEvents pageRenderingProgressEvents(OptionSet<WebCore::LayoutMilestone> milestones)
33{
34 WKPageRenderingProgressEvents events = 0;
35
36 if (milestones & WebCore::DidFirstLayout)
37 events |= WKPageRenderingProgressEventFirstLayout;
38
39 if (milestones & WebCore::DidFirstVisuallyNonEmptyLayout)
40 events |= WKPageRenderingProgressEventFirstVisuallyNonEmptyLayout;
41
42 if (milestones & WebCore::DidHitRelevantRepaintedObjectsAreaThreshold)
43 events |= WKPageRenderingProgressEventFirstPaintWithSignificantArea;
44
45 if (milestones & WebCore::ReachedSessionRestorationRenderTreeSizeThreshold)
46 events |= WKPageRenderingProgressEventReachedSessionRestorationRenderTreeSizeThreshold;
47
48 if (milestones & WebCore::DidFirstLayoutAfterSuppressedIncrementalRendering)
49 events |= WKPageRenderingProgressEventFirstLayoutAfterSuppressedIncrementalRendering;
50
51 if (milestones & WebCore::DidFirstPaintAfterSuppressedIncrementalRendering)
52 events |= WKPageRenderingProgressEventFirstPaintAfterSuppressedIncrementalRendering;
53
54 if (milestones & WebCore::DidFirstMeaningfulPaint)
55 events |= WKPageRenderingProgressEventFirstMeaningfulPaint;
56
57 return events;
58}
