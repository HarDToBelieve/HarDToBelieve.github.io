26#include "config.h"
27#include "WKUserContentControllerRef.h"
28
29#include "APIArray.h"
30#include "APIContentRuleList.h"
31#include "APIUserScript.h"
32#include "InjectUserScriptImmediately.h"
33#include "WKAPICast.h"
34#include "WebUserContentControllerProxy.h"
35
36using namespace WebKit;
37
38WKTypeID WKUserContentControllerGetTypeID()
39{
40 return toAPI(WebUserContentControllerProxy::APIType);
41}
42
43WKUserContentControllerRef WKUserContentControllerCreate()
44{
45 return toAPI(&WebUserContentControllerProxy::create().leakRef());
46}
47
48WKArrayRef WKUserContentControllerCopyUserScripts(WKUserContentControllerRef userContentControllerRef)
49{
50 Ref<API::Array> userScripts = toImpl(userContentControllerRef)->userScripts().copy();
51 return toAPI(&userScripts.leakRef());
52}
53
54void WKUserContentControllerAddUserScript(WKUserContentControllerRef userContentControllerRef, WKUserScriptRef userScriptRef)
55{
56 toImpl(userContentControllerRef)->addUserScript(*toImpl(userScriptRef), InjectUserScriptImmediately::No);
57}
58
59void WKUserContentControllerRemoveAllUserScripts(WKUserContentControllerRef userContentControllerRef)
60{
61 toImpl(userContentControllerRef)->removeAllUserScripts();
62}
63
64void WKUserContentControllerAddUserContentFilter(WKUserContentControllerRef userContentControllerRef, WKUserContentFilterRef userContentFilterRef)
65{
66#if ENABLE(CONTENT_EXTENSIONS)
67 toImpl(userContentControllerRef)->addContentRuleList(*toImpl(userContentFilterRef));
68#endif
69}
70
71void WKUserContentControllerRemoveAllUserContentFilters(WKUserContentControllerRef userContentControllerRef)
72{
73#if ENABLE(CONTENT_EXTENSIONS)
74 toImpl(userContentControllerRef)->removeAllContentRuleLists();
75#endif
76}
