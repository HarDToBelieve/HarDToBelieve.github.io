41static WTF::String mimeTypeInferredFromFileExtension(const API::Attachment& attachment)
42{
43 if (NSString *fileExtension = [(NSString *)attachment.fileName() pathExtension])
44 return WebCore::MIMETypeRegistry::mimeTypeForExtension(fileExtension);
45
46 return { };
47}
48
49static BOOL isDeclaredOrDynamicTypeIdentifier(NSString *type)
50{
51ALLOW_DEPRECATED_DECLARATIONS_BEGIN
52 return UTTypeIsDeclared((__bridge CFStringRef)type) || UTTypeIsDynamic((__bridge CFStringRef)type);
53ALLOW_DEPRECATED_DECLARATIONS_END
54}

56NSFileWrapper *Attachment::fileWrapper() const
57{
58 if (m_fileWrapperGenerator && !m_fileWrapper)
59 m_fileWrapper = m_fileWrapperGenerator();
60 return m_fileWrapper.get();
61}
62
63void Attachment::invalidateGeneratedFileWrapper()
64{
65 ASSERT(m_fileWrapperGenerator);
66 m_fileWrapper = nil;
67 m_webPage->didInvalidateDataForAttachment(*this);
68}

70WTF::String Attachment::mimeType() const
71{
72 NSString *contentType = m_contentType.isEmpty() ? mimeTypeInferredFromFileExtension(*this) : m_contentType;
73 if (!contentType.length)
74 return nullString();
75 if (!isDeclaredOrDynamicTypeIdentifier(contentType))
76 return contentType;
77
78ALLOW_DEPRECATED_DECLARATIONS_BEGIN
79 return adoptCF(UTTypeCopyPreferredTagWithClass((__bridge CFStringRef)contentType, kUTTagClassMIMEType)).get();
80ALLOW_DEPRECATED_DECLARATIONS_END
81}
82
83WTF::String Attachment::utiType() const
84{
85 NSString *contentType = m_contentType.isEmpty() ? mimeTypeInferredFromFileExtension(*this) : m_contentType;
86 if (!contentType.length)
87 return nullString();
88 if (isDeclaredOrDynamicTypeIdentifier(contentType))
89 return contentType;
90
91ALLOW_DEPRECATED_DECLARATIONS_BEGIN
92 return adoptCF(UTTypeCreatePreferredIdentifierForTag(kUTTagClassMIMEType, (__bridge CFStringRef)contentType, nullptr)).get();
93ALLOW_DEPRECATED_DECLARATIONS_END
94}

96WTF::String Attachment::fileName() const
97{
98 auto fileWrapper = this->fileWrapper();
99
100 if ([fileWrapper filename].length)
101 return [fileWrapper filename];
102
103 return [fileWrapper preferredFilename];
104}

106void Attachment::setFileWrapperAndUpdateContentType(NSFileWrapper *fileWrapper, NSString *contentType)
107{
108 if (!contentType.length) {
109ALLOW_DEPRECATED_DECLARATIONS_BEGIN
110 if (fileWrapper.directory)
111 contentType = (NSString *)kUTTypeDirectory;
112 else if (fileWrapper.regularFile) {
113 if (NSString *pathExtension = (fileWrapper.filename.length ? fileWrapper.filename : fileWrapper.preferredFilename).pathExtension)
114 contentType = WebCore::MIMETypeRegistry::mimeTypeForExtension(pathExtension);
115 if (!contentType.length)
116 contentType = (NSString *)kUTTypeData;
117 }
118ALLOW_DEPRECATED_DECLARATIONS_END
119 }
120
121 setContentType(contentType);
122 setFileWrapper(fileWrapper);
123}

125std::optional<uint64_t> Attachment::fileSizeForDisplay() const
126{
127 auto fileWrapper = this->fileWrapper();
128
129 if (![fileWrapper isRegularFile]) {
130 // FIXME: We should display a size estimate for directory-type file wrappers.
131 return std::nullopt;
132 }
133
134 if (auto fileSize = [[fileWrapper fileAttributes][NSFileSize] unsignedLongLongValue])
135 return fileSize;
136
137 return [fileWrapper regularFileContents].length;
138}

140RefPtr<WebCore::SharedBuffer> Attachment::enclosingImageData() const
141{
142 if (!m_hasEnclosingImage)
143 return nullptr;
144
145 auto fileWrapper = this->fileWrapper();
146
147 if (![fileWrapper isRegularFile])
148 return nullptr;
149
150 NSData *data = [fileWrapper regularFileContents];
151 if (!data)
152 return nullptr;
153
154 return WebCore::SharedBuffer::create(data);
155}

157bool Attachment::isEmpty() const
158{
159 return !m_fileWrapper && !m_fileWrapperGenerator;
160}

162RefPtr<WebCore::SharedBuffer> Attachment::createSerializedRepresentation() const
163{
164 auto fileWrapper = this->fileWrapper();
165
166 if (!fileWrapper || !m_webPage)
167 return nullptr;
168
169 NSData *serializedData = [NSKeyedArchiver archivedDataWithRootObject:fileWrapper requiringSecureCoding:YES error:nullptr];
170 if (!serializedData)
171 return nullptr;
172
173 return WebCore::SharedBuffer::create(serializedData);
174}
175
176void Attachment::updateFromSerializedRepresentation(Ref<WebCore::SharedBuffer>&& serializedRepresentation, const WTF::String& contentType)
177{
178 if (!m_webPage)
179 return;
180
181 auto serializedData = serializedRepresentation->createNSData();
182 if (!serializedData)
183 return;
184
185 NSFileWrapper *fileWrapper = [NSKeyedUnarchiver unarchivedObjectOfClasses:m_webPage->pageClient().serializableFileWrapperClasses() fromData:serializedData.get() error:nullptr];
186 if (![fileWrapper isKindOfClass:NSFileWrapper.class])
187 return;
188
189 setFileWrapperAndUpdateContentType(fileWrapper, contentType);
190 m_webPage->updateAttachmentAttributes(*this, [] { });
191}

193void Attachment::setFileWrapperGenerator(Function<RetainPtr<NSFileWrapper>(void)>&& fileWrapperGenerator)
194{
195 m_fileWrapperGenerator = WTFMove(fileWrapperGenerator);
196 m_fileWrapper = nil;
197 m_webPage->didInvalidateDataForAttachment(*this);
198}
