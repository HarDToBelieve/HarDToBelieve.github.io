36WTF::String ContentRuleListStore::defaultStorePath()
37{
38 static dispatch_once_t onceToken;
39 static NeverDestroyed<RetainPtr<NSURL>> contentRuleListStoreURL;
40
41 dispatch_once(&onceToken, ^{
42 NSURL *url = [[NSFileManager defaultManager] URLForDirectory:NSLibraryDirectory inDomain:NSUserDomainMask appropriateForURL:nullptr create:NO error:nullptr];
43 if (!url)
44 RELEASE_ASSERT_NOT_REACHED();
45
46 url = [url URLByAppendingPathComponent:@"WebKit" isDirectory:YES];
47
48 if (!WebKit::processHasContainer()) {
49 NSString *bundleIdentifier = [NSBundle mainBundle].bundleIdentifier;
50 if (!bundleIdentifier)
51 bundleIdentifier = [NSProcessInfo processInfo].processName;
52 url = [url URLByAppendingPathComponent:bundleIdentifier isDirectory:YES];
53 }
54
55 contentRuleListStoreURL.get() = [url URLByAppendingPathComponent:@"ContentRuleLists" isDirectory:YES];
56 });
57
58 if (![[NSFileManager defaultManager] createDirectoryAtURL:contentRuleListStoreURL.get().get() withIntermediateDirectories:YES attributes:nil error:nullptr])
59 LOG_ERROR("Failed to create directory %@", contentRuleListStoreURL.get().get());
60
61 return [contentRuleListStoreURL.get() absoluteURL].path.fileSystemRepresentation;
62}
