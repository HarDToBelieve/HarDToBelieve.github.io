37namespace API {
38
39class SharedJSContext {
40public:
41 SharedJSContext()
42 : m_timer(RunLoop::main(), this, &SharedJSContext::releaseContext)
43 {
44 }
45
46 JSContext* ensureContext()
47 {
48 if (!m_context) {
49 bool previous = JSRemoteInspectorGetInspectionEnabledByDefault();
50 JSRemoteInspectorSetInspectionEnabledByDefault(false);
51 m_context = adoptNS([[JSContext alloc] init]);
52 JSRemoteInspectorSetInspectionEnabledByDefault(previous);
53
54 m_timer.startOneShot(1_s);
55 }
56 return m_context.get();
57 }
58
59 void releaseContext()
60 {
61 m_context.clear();
62 }
63
64private:
65 RetainPtr<JSContext> m_context;
66 RunLoop::Timer<SharedJSContext> m_timer;
67};
68
69static SharedJSContext& sharedContext()
70{
71 static NeverDestroyed<SharedJSContext> sharedContext;
72 return sharedContext.get();
73}

75id SerializedScriptValue::deserialize(WebCore::SerializedScriptValue& serializedScriptValue, JSValueRef* exception)
76{
77 ASSERT(RunLoop::isMain());
78 JSContext* context = sharedContext().ensureContext();
79
80 JSValueRef valueRef = serializedScriptValue.deserialize([context JSGlobalContextRef], exception);
81 if (!valueRef)
82 return nil;
83
84 JSValue *value = [JSValue valueWithJSValueRef:valueRef inContext:context];
85 return value.toObject;
86}

88static bool validateObject(id argument)
89{
90 if ([argument isKindOfClass:[NSString class]] || [argument isKindOfClass:[NSNumber class]] || [argument isKindOfClass:[NSDate class]] || [argument isKindOfClass:[NSNull class]])
91 return true;
92
93 if ([argument isKindOfClass:[NSArray class]]) {
94 __block BOOL valid = true;
95
96 [argument enumerateObjectsUsingBlock:^(id object, NSUInteger, BOOL *stop) {
97 if (!validateObject(object)) {
98 valid = false;
99 *stop = YES;
100 }
101 }];
102
103 return valid;
104 }
105
106 if ([argument isKindOfClass:[NSDictionary class]]) {
107 __block bool valid = true;
108
109 [argument enumerateKeysAndObjectsUsingBlock:^(id key, id value, BOOL *stop) {
110 if (!validateObject(key) || !validateObject(value)) {
111 valid = false;
112 *stop = YES;
113 }
114 }];
115
116 return valid;
117 }
118
119 return false;
120}
121
122static RefPtr<WebCore::SerializedScriptValue> coreValueFromNSObject(id object)
123{
124 if (object && !validateObject(object))
125 return nullptr;
126
127 ASSERT(RunLoop::isMain());
128 JSContext* context = sharedContext().ensureContext();
129 JSValue *value = [JSValue valueWithObject:object inContext:context];
130 if (!value)
131 return nullptr;
132
133 auto globalObject = toJS([context JSGlobalContextRef]);
134 ASSERT(globalObject);
135 JSC::JSLockHolder lock(globalObject);
136
137 return WebCore::SerializedScriptValue::create(*globalObject, toJS(globalObject, [value JSValueRef]));
138}
139
140RefPtr<SerializedScriptValue> SerializedScriptValue::createFromNSObject(id object)
141{
142 auto coreValue = coreValueFromNSObject(object);
143 if (!coreValue)
144 return nullptr;
145 return create(coreValue.releaseNonNull());
146}
