136+ (WKWebViewConfiguration *)configuration
137{
138 auto& configuration = globalConfiguration();
139 if (!configuration) {
140 configuration = adoptNS([[WKWebViewConfiguration alloc] init]);
141 [configuration setProcessPool:adoptNS([[WKProcessPool alloc] init]).get()];
142 [configuration setWebsiteDataStore:[WKWebsiteDataStore nonPersistentDataStore]];
143 [configuration setMediaTypesRequiringUserActionForPlayback:WKAudiovisualMediaTypeAll];
144 [configuration _setAllowsJavaScriptMarkup:NO];
145 [configuration _setAllowsMetaRefresh:NO];
146 [configuration _setAttachmentElementEnabled:YES];
147 [configuration _setInvisibleAutoplayNotPermitted:YES];
148 [configuration _setMediaDataLoadsAutomatically:NO];
149 [configuration _setNeedsStorageAccessFromFileURLsQuirk:NO];
150#if PLATFORM(IOS_FAMILY)

164+ (RetainPtr<WKWebView>)retrieveOrCreateWebView
165{
166 [self resetPurgeDelay];
167
168 static dispatch_once_t onceToken;
169 dispatch_once(&onceToken, ^{
170 auto& memoryPressureHandler = MemoryPressureHandler::singleton();
171 memoryPressureHandler.setLowMemoryHandler([self] (Critical, Synchronous) {
172 [self purgeAllWebViews];
173 });
174 });
175
176 auto* cache = self.cache;
177 if (cache.count) {
178 RetainPtr<WKWebView> webView = cache.lastObject;
179 [cache removeLastObject];
180 return webView;
181 }
182
183 return adoptNS([[WKWebView alloc] initWithFrame:webViewRect configuration:self.configuration]);
184}
185
186+ (void)cacheWebView:(WKWebView *)webView
187{
188 auto* cache = self.cache;
189 if (cache.count >= maximumWebViewCacheSize)
190 return;
191
192 [cache addObject:webView];
193}
194
195+ (void)resetPurgeDelay
196{
197 static const auto purgeSelector = @selector(purgeSingleWebView);
198 [NSObject cancelPreviousPerformRequestsWithTarget:self selector:purgeSelector object:nil];
199 [self performSelector:purgeSelector withObject:nil afterDelay:purgeWebViewCacheDelay];
200}
201
202+ (void)purgeSingleWebView
203{
204 [NSObject cancelPreviousPerformRequestsWithTarget:self selector:_cmd object:nil];
205
206 auto* cache = self.cache;
207 if (!cache.count)
208 return;
209
210 [cache.lastObject _close];
211 [cache removeLastObject];
212
213 if (!cache.count) {
214 [self clearConfiguration];
215 return;
216 }
217
218 // Keep going until every view is removed, or the delay is reset.
219 [self performSelector:_cmd withObject:nil afterDelay:purgeWebViewCacheDelay];
220}
221
222+ (void)purgeAllWebViews
223{
224 auto* cache = self.cache;
225 if (!cache.count)
226 return;
227
228 [cache makeObjectsPerformSelector:@selector(_close)];
229 [cache removeAllObjects];
230
231 [self clearConfiguration];
232}

238+ (void)_loadFromHTMLWithOptions:(NSDictionary<NSAttributedStringDocumentReadingOptionKey, id> *)options contentLoader:(WKNavigation *(^)(WKWebView *))loadWebContent completionHandler:(NSAttributedStringCompletionHandler)completionHandler
239{
240 if (options[NSWebPreferencesDocumentOption])
241 [NSException raise:NSInvalidArgumentException format:@"NSWebPreferencesDocumentOption option is not supported"];
242 if (options[NSWebResourceLoadDelegateDocumentOption])
243 [NSException raise:NSInvalidArgumentException format:@"NSWebResourceLoadDelegateDocumentOption option is not supported"];
244 if (options[@"WebPolicyDelegate"])
245 [NSException raise:NSInvalidArgumentException format:@"WebPolicyDelegate option is not supported"];
246
247 auto runConversion = ^{
248 __block auto finished = NO;
249 __block auto webView = [_WKAttributedStringWebViewCache retrieveOrCreateWebView];
250 __block auto navigationDelegate = adoptNS([[_WKAttributedStringNavigationDelegate alloc] init]);
251
252 __block RetainPtr<WKNavigation> contentNavigation;
253
254 webView.get().navigationDelegate = navigationDelegate.get();
255
256 if (auto* textZoomFactor = dynamic_objc_cast<NSNumber>(options[NSTextSizeMultiplierDocumentOption]))
257 webView.get()._textZoomFactor = textZoomFactor.doubleValue;
258 else
259 webView.get()._textZoomFactor = 1;
260
261 auto finish = ^(NSAttributedString *attributedString, NSDictionary<NSAttributedStringDocumentAttributeKey, id> *attributes, NSError *error) {
262 if (finished)
263 return;
264
265 finished = YES;
266
267 webView.get().navigationDelegate = nil;
268 navigationDelegate = nil;
269 contentNavigation = nil;
270
271 if (!error)
272 [_WKAttributedStringWebViewCache cacheWebView:webView.get()];
273 webView = nil;
274
275 // Make the string be an instance of the receiver class.
276 RetainPtr<NSAttributedString> newAttributedString;
277 if (attributedString && self != attributedString.class) {
278 newAttributedString = adoptNS([[self alloc] initWithAttributedString:attributedString]);
279 attributedString = newAttributedString.get();
280 }
281
282 // Make the document attributes immutable.
283 RetainPtr<NSDictionary<NSAttributedStringDocumentAttributeKey, id>> newAttributes;
284 if ([attributes isKindOfClass:NSMutableDictionary.class]) {
285 newAttributes = adoptNS([[NSDictionary alloc] initWithDictionary:attributes]);
286 attributes = newAttributes.get();
287 }
288
289 completionHandler(attributedString, attributes, error);
290 };
291
292 auto cancel = ^(WKErrorCode errorCode, NSError* underlyingError) {
293 finish(nil, nil, createNSError(errorCode, underlyingError).get());
294 };
295
296 navigationDelegate.get().decidePolicyForNavigationAction = ^(WKNavigationAction *action, void (^decisionHandler)(WKNavigationActionPolicy)) {
297 if ([action._mainFrameNavigation isEqual:contentNavigation.get()])
298 return decisionHandler(WKNavigationActionPolicyAllow);
299 decisionHandler(WKNavigationActionPolicyCancel);
300 };
301
302 navigationDelegate.get().webContentProcessDidTerminate = ^(WKWebView *) {
303 cancel(WKErrorWebContentProcessTerminated, nil);
304 };
305
306 navigationDelegate.get().didFailProvisionalNavigation = ^(WKWebView *, WKNavigation *, NSError *error) {
307 cancel(WKErrorAttributedStringContentFailedToLoad, error);
308 };
309
310 navigationDelegate.get().didFailNavigation = ^(WKWebView *, WKNavigation *, NSError *error) {
311 cancel(WKErrorAttributedStringContentFailedToLoad, error);
312 };
313
314 navigationDelegate.get().didFinishNavigation = ^(WKWebView *, WKNavigation *) {
315 if (finished)
316 return;
317
318 navigationDelegate = nil;
319
320 [webView _getContentsAsAttributedStringWithCompletionHandler:^(NSAttributedString *attributedString, NSDictionary<NSAttributedStringDocumentAttributeKey, id> *documentAttributes, NSError *error) {
321 if (error)
322 return cancel(WKErrorUnknown, error);
323 finish(attributedString, documentAttributes, nil);
324 }];
325 };
326
327 auto timeoutInterval = defaultTimeoutInterval;
328 if (auto* timeoutOption = dynamic_objc_cast<NSNumber>(options[NSTimeoutDocumentOption])) {
329 if (timeoutOption.doubleValue >= 0)
330 timeoutInterval = timeoutOption.doubleValue;
331 }
332
333 dispatch_after(dispatch_time(DISPATCH_TIME_NOW, timeoutInterval * NSEC_PER_SEC), dispatch_get_main_queue(), ^{
334 if (finished)
335 return;
336 cancel(WKErrorAttributedStringContentLoadTimedOut, nil);
337 });
338
339 contentNavigation = loadWebContent(webView.get());
340
341 ASSERT(contentNavigation);
342 ASSERT(webView.get().loading);
343 };
344
345 if ([NSThread isMainThread])
346 runConversion();
347 else
348 dispatch_async(dispatch_get_main_queue(), runConversion);
349}

378+ (void)loadFromHTMLWithData:(NSData *)data options:(NSDictionary<NSAttributedStringDocumentReadingOptionKey, id> *)options completionHandler:(NSAttributedStringCompletionHandler)completionHandler
379{
380 [self _loadFromHTMLWithOptions:options contentLoader:^WKNavigation *(WKWebView *webView) {
381 auto* textEncodingName = dynamic_objc_cast<NSString>(options[NSTextEncodingNameDocumentOption]);
382 auto characterEncoding = static_cast<NSStringEncoding>(dynamic_objc_cast<NSNumber>(options[NSCharacterEncodingDocumentOption]).unsignedIntegerValue);
383 auto* baseURL = dynamic_objc_cast<NSURL>(options[NSBaseURLDocumentOption]);
384
385 if (characterEncoding && !textEncodingName) {
386 auto stringEncoding = CFStringConvertNSStringEncodingToEncoding(characterEncoding);
387 if (stringEncoding != kCFStringEncodingInvalidId)
388 textEncodingName = (__bridge NSString *)CFStringConvertEncodingToIANACharSetName(stringEncoding);
389 }
390
391 return [webView loadData:data MIMEType:@"text/html" characterEncodingName:textEncodingName baseURL:baseURL];
392 } completionHandler:completionHandler];
393}
