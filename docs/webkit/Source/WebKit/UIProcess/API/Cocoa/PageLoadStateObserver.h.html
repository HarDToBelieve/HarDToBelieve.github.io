28namespace WebKit {
29
30class PageLoadStateObserver : public PageLoadState::Observer {
31 WTF_MAKE_FAST_ALLOCATED;
32public:
33 PageLoadStateObserver(id object, NSString *activeURLKey = @"activeURL")
34 : m_object(object)
35 , m_activeURLKey(adoptNS([activeURLKey copy]))
36 {
37 }
38
39 void clearObject()
40 {
41 m_object = nil;
42 }
43
44private:
45 void willChangeIsLoading() override
46 {
47 [m_object willChangeValueForKey:@"loading"];
48 }
49
50 void didChangeIsLoading() override
51 {
52 [m_object didChangeValueForKey:@"loading"];
53 }
54
55 void willChangeTitle() override
56 {
57 [m_object willChangeValueForKey:@"title"];
58 }
59
60 void didChangeTitle() override
61 {
62 [m_object didChangeValueForKey:@"title"];
63 }
64
65 void willChangeActiveURL() override
66 {
67 [m_object willChangeValueForKey:m_activeURLKey.get()];
68 }
69
70 void didChangeActiveURL() override
71 {
72 [m_object didChangeValueForKey:m_activeURLKey.get()];
73 }
74
75 void willChangeHasOnlySecureContent() override
76 {
77 [m_object willChangeValueForKey:@"hasOnlySecureContent"];
78 }
79
80 void didChangeHasOnlySecureContent() override
81 {
82 [m_object didChangeValueForKey:@"hasOnlySecureContent"];
83 }
84
85 void willChangeEstimatedProgress() override
86 {
87 [m_object willChangeValueForKey:@"estimatedProgress"];
88 }
89
90 void didChangeEstimatedProgress() override
91 {
92 [m_object didChangeValueForKey:@"estimatedProgress"];
93 }
94
95 void willChangeCanGoBack() override { }
96 void didChangeCanGoBack() override { }
97 void willChangeCanGoForward() override { }
98 void didChangeCanGoForward() override { }
99 void willChangeNetworkRequestsInProgress() override { }
100 void didChangeNetworkRequestsInProgress() override { }
101 void willChangeCertificateInfo() override { }
102 void didChangeCertificateInfo() override { }
103 void didSwapWebProcesses() override { }
104
105 void willChangeWebProcessIsResponsive() override
106 {
107 [m_object willChangeValueForKey:@"_webProcessIsResponsive"];
108 }
109
110 void didChangeWebProcessIsResponsive() override
111 {
112 [m_object didChangeValueForKey:@"_webProcessIsResponsive"];
113 }
114
115 id m_object;
116 RetainPtr<NSString> m_activeURLKey;
117};
118
119}
