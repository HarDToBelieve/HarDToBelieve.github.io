93- (void)dealloc
94{
95 if (WebCoreObjCScheduleDeallocateOnMainRunLoop(WKBrowsingContextController.class, self))
96 return;
97
98 ASSERT(browsingContextControllerMap().get(_page.get()) == self);
99 browsingContextControllerMap().remove(_page.get());
100
101 _page->pageLoadState().removeObserver(*_pageLoadStateObserver);
102
103 [super dealloc];
104}

109{
110 if ([NSThread isMainThread])
111 WebKit::WebProcessPool::registerGlobalURLSchemeAsHavingCustomProtocolHandlers(scheme);
112 else {
113 // This cannot be RunLoop::main().dispatch because it is called before the main runloop is initialized. See rdar://problem/73615999

152{

154 [NSException raise:NSInvalidArgumentException format:@"Attempted to load a non-file URL"];

465static BlockPtr<void(WKPolicyDecision)> makePolicyDecisionBlock(WKFramePolicyListenerRef listener)
466{
467 return makeBlockPtr([listener = retainWK(listener)](WKPolicyDecision decision) {
468 switch (decision) {
469 case WKPolicyDecisionCancel:
470 WKFramePolicyListenerIgnore(listener.get());
471 break;
472 case WKPolicyDecisionAllow:
473 WKFramePolicyListenerUse(listener.get());
474 break;
475 case WKPolicyDecisionBecomeDownload:
476 WKFramePolicyListenerDownload(listener.get());
477 break;
478 };
479 });
480}

483static void setUpPagePolicyClient(WKBrowsingContextController *browsingContext, WebKit::WebPageProxy& page)
484ALLOW_DEPRECATED_DECLARATIONS_END
485{
486 WKPagePolicyClientInternal policyClient;
487 memset(&policyClient, 0, sizeof(policyClient));
488
489 policyClient.base.version = 2;
490 policyClient.base.clientInfo = (__bridge void*)browsingContext;
491
492 policyClient.decidePolicyForNavigationAction = [](WKPageRef page, WKFrameRef frame, WKFrameNavigationType navigationType, WKEventModifiers modifiers, WKEventMouseButton mouseButton, WKFrameRef originatingFrame, WKURLRequestRef originalRequest, WKURLRequestRef request, WKFramePolicyListenerRef listener, WKTypeRef userData, const void* clientInfo)
493 {
494 ALLOW_DEPRECATED_DECLARATIONS_BEGIN
495 auto browsingContext = (__bridge WKBrowsingContextController *)clientInfo;
496 ALLOW_DEPRECATED_DECLARATIONS_END
497 auto policyDelegate = browsingContext->_policyDelegate.get();
498
499 if ([policyDelegate respondsToSelector:@selector(browsingContextController:decidePolicyForNavigationAction:decisionHandler:)]) {
500 auto actionDictionary = retainPtr(@{
501 WKActionIsMainFrameKey: @(WKFrameIsMainFrame(frame)),
502 WKActionNavigationTypeKey: @(navigationType),
503 WKActionModifierFlagsKey: @(modifiers),
504 WKActionMouseButtonKey: @(mouseButton),
505 WKActionOriginalURLRequestKey: adoptNS(WKURLRequestCopyNSURLRequest(originalRequest)).get(),
506 WKActionURLRequestKey: adoptNS(WKURLRequestCopyNSURLRequest(request)).get()
507 });
508
509 if (originatingFrame) {
510 actionDictionary = adoptNS([actionDictionary mutableCopy]);
511 [(NSMutableDictionary *)actionDictionary.get() setObject:[NSURL _web_URLWithWTFString:WebKit::toImpl(originatingFrame)->url().string()] forKey:WKActionOriginatingFrameURLKey];
512 }
513
514 [policyDelegate browsingContextController:browsingContext decidePolicyForNavigationAction:actionDictionary.get() decisionHandler:makePolicyDecisionBlock(listener).get()];
515 } else
516 WKFramePolicyListenerUse(listener);
517 };
518
519 policyClient.decidePolicyForNewWindowAction = [](WKPageRef page, WKFrameRef frame, WKFrameNavigationType navigationType, WKEventModifiers modifiers, WKEventMouseButton mouseButton, WKURLRequestRef request, WKStringRef frameName, WKFramePolicyListenerRef listener, WKTypeRef userData, const void* clientInfo)
520 {
521 ALLOW_DEPRECATED_DECLARATIONS_BEGIN
522 auto browsingContext = (__bridge WKBrowsingContextController *)clientInfo;
523 ALLOW_DEPRECATED_DECLARATIONS_END
524 auto policyDelegate = browsingContext->_policyDelegate.get();
525
526 if ([policyDelegate respondsToSelector:@selector(browsingContextController:decidePolicyForNewWindowAction:decisionHandler:)]) {
527 NSDictionary *actionDictionary = @{
528 WKActionIsMainFrameKey: @(WKFrameIsMainFrame(frame)),
529 WKActionNavigationTypeKey: @(navigationType),
530 WKActionModifierFlagsKey: @(modifiers),
531 WKActionMouseButtonKey: @(mouseButton),
532 WKActionURLRequestKey: adoptNS(WKURLRequestCopyNSURLRequest(request)).get(),
533 WKActionFrameNameKey: WebKit::toImpl(frameName)->wrapper()
534 };
535
536 [policyDelegate browsingContextController:browsingContext decidePolicyForNewWindowAction:actionDictionary decisionHandler:makePolicyDecisionBlock(listener).get()];
537 } else
538 WKFramePolicyListenerUse(listener);
539 };

615- (instancetype)_initWithPageRef:(WKPageRef)pageRef
616{
617 if (!(self = [super init]))
618 return nil;
619
620 _page = WebKit::toImpl(pageRef);
621
622 _pageLoadStateObserver = makeUnique<WebKit::PageLoadStateObserver>(self);
623 _page->pageLoadState().addObserver(*_pageLoadStateObserver);
624
625 ASSERT(!browsingContextControllerMap().contains(_page.get()));
626 browsingContextControllerMap().set(_page.get(), self);
627
628 return self;
629}
