50- (void)dealloc
51{
52 if (WebCoreObjCScheduleDeallocateOnMainRunLoop(WKBrowsingContextGroup.class, self))
53 return;
54
55 _pageGroup->~WebPageGroup();
56
57 [super dealloc];
58}

60- (id)initWithIdentifier:(NSString *)identifier
61{
62 self = [super init];
63 if (!self)
64 return nil;
65
66 API::Object::constructInWrapper<WebKit::WebPageGroup>(self, identifier);
67
68 // Give the WKBrowsingContextGroup a identifier-less preferences, so that they
69 // don't get automatically written to the disk. The automatic writing has proven
70 // confusing to users of the API.
71 WKRetainPtr<WKPreferencesRef> preferences = adoptWK(WKPreferencesCreate());
72 WKPageGroupSetPreferences(WebKit::toAPI(_pageGroup.get()), preferences.get());
73
74 return self;
75}

117static WKRetainPtr<WKArrayRef> createWKArray(NSArray *array)
118{
119 NSUInteger count = [array count];
120
121 if (!count)
122 return nullptr;
123
124 Vector<RefPtr<API::Object>> strings;
125 strings.reserveInitialCapacity(count);
126
127 for (id entry in array) {
128 if ([entry isKindOfClass:[NSString class]])
129 strings.uncheckedAppend(adoptRef(WebKit::toImpl(WKStringCreateWithCFString((__bridge CFStringRef)entry))));
130 }
131
132 return WebKit::toAPI(&API::Array::create(WTFMove(strings)).leakRef());
133}

135- (void)addUserStyleSheet:(NSString *)source baseURL:(NSURL *)baseURL includeMatchPatternStrings:(NSArray<NSString *> *)includeMatchPatternStrings excludeMatchPatternStrings:(NSArray<NSString *> *)excludeMatchPatternStrings mainFrameOnly:(BOOL)mainFrameOnly
136{
137 if (!source)
138 CRASH();
139
140 WKRetainPtr<WKStringRef> wkSource = adoptWK(WKStringCreateWithCFString((__bridge CFStringRef)source));
141 WKRetainPtr<WKURLRef> wkBaseURL = adoptWK(WKURLCreateWithCFURL((__bridge CFURLRef)baseURL));
142 auto wkIncludeMatchPatternStrings = createWKArray(includeMatchPatternStrings);
143 auto wkExcludeMatchPatternStrings = createWKArray(excludeMatchPatternStrings);
144 WKUserContentInjectedFrames injectedFrames = mainFrameOnly ? kWKInjectInTopFrameOnly : kWKInjectInAllFrames;
145
146 WKPageGroupAddUserStyleSheet(WebKit::toAPI(_pageGroup.get()), wkSource.get(), wkBaseURL.get(), wkIncludeMatchPatternStrings.get(), wkExcludeMatchPatternStrings.get(), injectedFrames);
147}

161 auto wkIncludeMatchPatternStrings = createWKArray(includeMatchPatternStrings);
