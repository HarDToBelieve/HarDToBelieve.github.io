38static WKErrorCode toWKErrorCode(const std::error_code& error)
39{
40 ASSERT(error.category() == API::contentRuleListStoreErrorCategory());
41 switch (static_cast<API::ContentRuleListStore::Error>(error.value())) {
42 case API::ContentRuleListStore::Error::LookupFailed:
43 return WKErrorContentRuleListStoreLookUpFailed;
44 case API::ContentRuleListStore::Error::VersionMismatch:
45 return WKErrorContentRuleListStoreVersionMismatch;
46 case API::ContentRuleListStore::Error::CompileFailed:
47 return WKErrorContentRuleListStoreCompileFailed;
48 case API::ContentRuleListStore::Error::RemoveFailed:
49 return WKErrorContentRuleListStoreRemoveFailed;
50 }
51 ASSERT_NOT_REACHED();
52 return WKErrorUnknown;
53}

82- (void)lookUpContentRuleListForIdentifier:(NSString *)identifier completionHandler:(void (^)(WKContentRuleList *, NSError *))completionHandler
83{
84 _contentRuleListStore->lookupContentRuleList(identifier, [completionHandler = makeBlockPtr(completionHandler)](RefPtr<API::ContentRuleList> contentRuleList, std::error_code error) {
85 if (error) {
86 auto userInfo = @{NSHelpAnchorErrorKey: [NSString stringWithFormat:@"Rule list lookup failed: %s", error.message().c_str()]};
87 auto wkError = toWKErrorCode(error);
88 ASSERT(wkError == WKErrorContentRuleListStoreLookUpFailed || wkError == WKErrorContentRuleListStoreVersionMismatch);
89 return completionHandler(nil, [NSError errorWithDomain:WKErrorDomain code:wkError userInfo:userInfo]);
90 }
91
92 completionHandler(wrapper(*contentRuleList), nil);
93 });
94}

99 completionHandler(createNSArray(identifiers).get());

103- (void)removeContentRuleListForIdentifier:(NSString *)identifier completionHandler:(void (^)(NSError *))completionHandler
104{
105 _contentRuleListStore->removeContentRuleList(identifier, [completionHandler = makeBlockPtr(completionHandler)](std::error_code error) {
106 if (error) {
107 auto userInfo = @{NSHelpAnchorErrorKey: [NSString stringWithFormat:@"Rule list removal failed: %s", error.message().c_str()]};
108 ASSERT(toWKErrorCode(error) == WKErrorContentRuleListStoreRemoveFailed);
109 return completionHandler([NSError errorWithDomain:WKErrorDomain code:WKErrorContentRuleListStoreRemoveFailed userInfo:userInfo]);
110 }
111
112 completionHandler(nil);
113 });
114}

151- (void)_compileContentRuleListForIdentifier:(NSString *)identifier encodedContentRuleList:(NSString *) encodedContentRuleList completionHandler:(void (^)(WKContentRuleList *, NSError *))completionHandler
152{
153 _contentRuleListStore->compileContentRuleList(identifier, encodedContentRuleList, [completionHandler = makeBlockPtr(completionHandler)](RefPtr<API::ContentRuleList> contentRuleList, std::error_code error) {
154 if (error) {
155 auto userInfo = @{NSHelpAnchorErrorKey: [NSString stringWithFormat:@"Rule list compilation failed: %s", error.message().c_str()]};
156
157 // error.value() could have a specific compiler error that is not equal to WKErrorContentRuleListStoreCompileFailed.
158 // We want to use error.message, but here we want to only pass on CompileFailed with userInfo from the std::error_code.
159 return completionHandler(nil, [NSError errorWithDomain:WKErrorDomain code:WKErrorContentRuleListStoreCompileFailed userInfo:userInfo]);
160 }
161 completionHandler(wrapper(*contentRuleList), nil);
162 });
163}
