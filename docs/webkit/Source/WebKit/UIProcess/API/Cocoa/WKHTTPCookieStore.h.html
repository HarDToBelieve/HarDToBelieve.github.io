28#import <Foundation/Foundation.h>

30NS_ASSUME_NONNULL_BEGIN
31
32@class WKHTTPCookieStore;
33
34WK_API_AVAILABLE(macos(10.13), ios(11.0))
35@protocol WKHTTPCookieStoreObserver <NSObject>
36@optional
37- (void)cookiesDidChangeInCookieStore:(WKHTTPCookieStore *)cookieStore;
38@end
39
40/*!
41 A WKHTTPCookieStore object allows managing the HTTP cookies associated with a particular WKWebsiteDataStore.
42 */
43WK_CLASS_AVAILABLE(macos(10.13), ios(11.0))
44@interface WKHTTPCookieStore : NSObject
45
46- (instancetype)init NS_UNAVAILABLE;
47
48/*! @abstract Fetches all stored cookies.
49 @param completionHandler A block to invoke with the fetched cookies.
50 */
51- (void)getAllCookies:(void (^)(NSArray<NSHTTPCookie *> *))completionHandler;
52
53/*! @abstract Set a cookie.
54 @param cookie The cookie to set.
55 @param completionHandler A block to invoke once the cookie has been stored.
56 */
57- (void)setCookie:(NSHTTPCookie *)cookie completionHandler:(nullable void (^)(void))completionHandler;
58
59/*! @abstract Delete the specified cookie.
60 @param completionHandler A block to invoke once the cookie has been deleted.
61 */
62- (void)deleteCookie:(NSHTTPCookie *)cookie completionHandler:(nullable void (^)(void))completionHandler WK_SWIFT_ASYNC_NAME(deleteCookie(_:));
63
64/*! @abstract Adds a WKHTTPCookieStoreObserver object with the cookie store.
65 @param observer The observer object to add.
66 @discussion The observer is not retained by the receiver. It is your responsibility
67 to unregister the observer before it becomes invalid.
68 */
69- (void)addObserver:(id<WKHTTPCookieStoreObserver>)observer;
70
71/*! @abstract Removes a WKHTTPCookieStoreObserver object from the cookie store.
72 @param observer The observer to remove.
73 */
74- (void)removeObserver:(id<WKHTTPCookieStoreObserver>)observer;
75
76@end
77
78NS_ASSUME_NONNULL_END
