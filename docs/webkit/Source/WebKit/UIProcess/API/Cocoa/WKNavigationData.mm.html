26#import "config.h"
27#import "WKNavigationDataInternal.h"
28
29#import "WKNSURLExtras.h"
30#import <WebCore/ResourceRequest.h>
31#import <WebCore/ResourceResponse.h>
32#import <WebCore/WebCoreObjCExtras.h>
33
34@implementation WKNavigationData {
35 API::ObjectStorage<API::NavigationData> _data;
36}
37
38- (void)dealloc
39{
40 if (WebCoreObjCScheduleDeallocateOnMainRunLoop(WKNavigationData.class, self))
41 return;
42
43 _data->~NavigationData();
44
45 [super dealloc];
46}
47
48- (NSString *)title
49{
50 return _data->title();
51}
52
53- (NSURLRequest *)originalRequest
54{
55 return _data->originalRequest().nsURLRequest(WebCore::HTTPBodyUpdatePolicy::DoNotUpdateHTTPBody);
56}
57
58- (NSURL *)destinationURL
59{
60 return [NSURL _web_URLWithWTFString:_data->url()];
61}
62
63- (NSURLResponse *)response
64{
65 return _data->response().nsURLResponse();
66}
67
68#pragma mark WKObject protocol implementation
69
70- (API::Object&)_apiObject
71{
72 return *_data;
73}
74
75@end
