69- (void)encodeWithCoder:(NSCoder *)coder
70{
71 [coder encodeDouble:self.minimumFontSize forKey:@"minimumFontSize"];
72 [coder encodeBool:self.javaScriptCanOpenWindowsAutomatically forKey:@"javaScriptCanOpenWindowsAutomatically"];
73
74ALLOW_DEPRECATED_DECLARATIONS_BEGIN
75 [coder encodeBool:self.javaScriptEnabled forKey:@"javaScriptEnabled"];
76ALLOW_DEPRECATED_DECLARATIONS_END
77
78#if PLATFORM(MAC)
79ALLOW_DEPRECATED_DECLARATIONS_BEGIN
80 [coder encodeBool:self.javaEnabled forKey:@"javaEnabled"];
81ALLOW_DEPRECATED_DECLARATIONS_END
82 [coder encodeBool:self.tabFocusesLinks forKey:@"tabFocusesLinks"];
83#endif
84 [coder encodeBool:self.textInteractionEnabled forKey:@"textInteractionEnabled"];
85}
86
87- (instancetype)initWithCoder:(NSCoder *)coder
88{
89 if (!(self = [self init]))
90 return nil;
91
92 self.minimumFontSize = [coder decodeDoubleForKey:@"minimumFontSize"];
93 self.javaScriptCanOpenWindowsAutomatically = [coder decodeBoolForKey:@"javaScriptCanOpenWindowsAutomatically"];
94
95ALLOW_DEPRECATED_DECLARATIONS_BEGIN
96 self.javaScriptEnabled = [coder decodeBoolForKey:@"javaScriptEnabled"];
97ALLOW_DEPRECATED_DECLARATIONS_END
98
99#if PLATFORM(MAC)
100ALLOW_DEPRECATED_DECLARATIONS_BEGIN
101 self.javaEnabled = [coder decodeBoolForKey:@"javaEnabled"];
102ALLOW_DEPRECATED_DECLARATIONS_END
103 self.tabFocusesLinks = [coder decodeBoolForKey:@"tabFocusesLinks"];
104#endif
105 if ([coder containsValueForKey:@"textInteractionEnabled"])
106 self.textInteractionEnabled = [coder decodeBoolForKey:@"textInteractionEnabled"];
107
108 return self;
109}

193static WebCore::StorageBlockingPolicy toStorageBlockingPolicy(_WKStorageBlockingPolicy policy)
194{
195 switch (policy) {
196 case _WKStorageBlockingPolicyAllowAll:
197 return WebCore::StorageBlockingPolicy::AllowAll;
198 case _WKStorageBlockingPolicyBlockThirdParty:
199 return WebCore::StorageBlockingPolicy::BlockThirdParty;
200 case _WKStorageBlockingPolicyBlockAll:
201 return WebCore::StorageBlockingPolicy::BlockAll;
202 }
203
204 ASSERT_NOT_REACHED();
205 return WebCore::StorageBlockingPolicy::AllowAll;
206}

209{

214 return _WKStorageBlockingPolicyBlockThirdParty;
215 case WebCore::StorageBlockingPolicy::BlockAll:
216 return _WKStorageBlockingPolicyBlockAll;
217 }
218
219 ASSERT_NOT_REACHED();
220 return _WKStorageBlockingPolicyAllowAll;
221}

748static _WKEditableLinkBehavior toAPI(WebCore::EditableLinkBehavior behavior)
749{
750 switch (behavior) {
751 case WebCore::EditableLinkBehavior::Default:
752 return _WKEditableLinkBehaviorDefault;
753 case WebCore::EditableLinkBehavior::AlwaysLive:
754 return _WKEditableLinkBehaviorAlwaysLive;
755 case WebCore::EditableLinkBehavior::OnlyLiveWithShiftKey:
756 return _WKEditableLinkBehaviorOnlyLiveWithShiftKey;
757 case WebCore::EditableLinkBehavior::LiveWhenNotFocused:
758 return _WKEditableLinkBehaviorLiveWhenNotFocused;
759 case WebCore::EditableLinkBehavior::NeverLive:
760 return _WKEditableLinkBehaviorNeverLive;
761 }
762
763 ASSERT_NOT_REACHED();
764 return _WKEditableLinkBehaviorNeverLive;
765}

778 case _WKEditableLinkBehaviorNeverLive:
