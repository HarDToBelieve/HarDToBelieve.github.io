124- (instancetype)initWithCoder:(NSCoder *)coder
125{
126 if (!(self = [self init]))
127 return nil;
128
129 if ([coder decodeBoolForKey:@"isSharedProcessPool"]) {
130 [self release];
131
132 return [[WKProcessPool _sharedProcessPool] retain];
133 }
134
135 return self;
136}

236- (void)_setObject:(id <NSCopying, NSSecureCoding>)object forBundleParameter:(NSString *)parameter
237{
238 auto copy = adoptNS([(NSObject *)object copy]);
239 auto keyedArchiver = adoptNS([[NSKeyedArchiver alloc] initRequiringSecureCoding:YES]);
240
241 @try {
242 [keyedArchiver encodeObject:copy.get() forKey:@"parameter"];
243 [keyedArchiver finishEncoding];
244 } @catch (NSException *exception) {
245 LOG_ERROR("Failed to encode bundle parameter: %@", exception);
246 }
247
248 if (copy)
249 [_processPool->ensureBundleParameters() setObject:copy.get() forKey:parameter];
250 else
251 [_processPool->ensureBundleParameters() removeObjectForKey:parameter];
252
253 auto data = keyedArchiver.get().encodedData;
254 _processPool->sendToAllProcesses(Messages::WebProcess::SetInjectedBundleParameter(parameter, IPC::DataReference(static_cast<const uint8_t*>([data bytes]), [data length])));
255}
256
257- (void)_setObjectsForBundleParametersWithDictionary:(NSDictionary *)dictionary
258{
259 auto copy = adoptNS([[NSDictionary alloc] initWithDictionary:dictionary copyItems:YES]);
260 auto keyedArchiver = adoptNS([[NSKeyedArchiver alloc] initRequiringSecureCoding:YES]);
261
262 @try {
263 [keyedArchiver encodeObject:copy.get() forKey:@"parameters"];
264 [keyedArchiver finishEncoding];
265 } @catch (NSException *exception) {
266 LOG_ERROR("Failed to encode bundle parameters: %@", exception);
267 }
268
269 [_processPool->ensureBundleParameters() setValuesForKeysWithDictionary:copy.get()];
270
271 auto data = keyedArchiver.get().encodedData;
272 _processPool->sendToAllProcesses(Messages::WebProcess::SetInjectedBundleParameters(IPC::DataReference(static_cast<const uint8_t*>([data bytes]), [data length])));
273}

394- (BOOL)_requestWebProcessTermination:(pid_t)pid
395{
396 for (auto& process : _processPool->processes()) {
397 if (process->processIdentifier() == pid)

399 return YES;
400 }
401 return NO;
