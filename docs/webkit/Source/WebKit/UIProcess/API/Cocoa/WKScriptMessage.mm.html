26#import "config.h"
27#import "WKScriptMessageInternal.h"
28
29#import <WebKit/WKFrameInfo.h>
30#import <wtf/RetainPtr.h>
31#import <wtf/WeakObjCPtr.h>
32
33@implementation WKScriptMessage {
34 RetainPtr<id> _body;
35 WeakObjCPtr<WKWebView> _webView;
36 RetainPtr<WKFrameInfo> _frameInfo;
37 RetainPtr<NSString> _name;
38 RetainPtr<WKContentWorld> _world;
39}
40
41- (instancetype)_initWithBody:(id)body webView:(WKWebView *)webView frameInfo:(WKFrameInfo *)frameInfo name:(NSString *)name world:(WKContentWorld *)world
42{
43 if (!(self = [super init]))
44 return nil;
45
46 _body = adoptNS([body copy]);
47 _webView = webView;
48 _frameInfo = frameInfo;
49 _name = adoptNS([name copy]);
50 _world = world;
51
52 return self;
53}
54
55- (id)body
56{
57 return _body.get();
58}
59
60- (WKWebView *)webView
61{
62 return _webView.getAutoreleased();
63}
64
65- (WKFrameInfo *)frameInfo
66{
67 return _frameInfo.get();
68}
69
70- (NSString *)name
71{
72 return _name.get();
73}
74
75- (WKContentWorld *)world
76{
77 return _world.get();
78}
79
80@end
