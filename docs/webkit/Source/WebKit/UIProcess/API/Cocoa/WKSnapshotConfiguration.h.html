26#import <WebKit/WKFoundation.h>
27
28#import <CoreGraphics/CGGeometry.h>
29#import <Foundation/Foundation.h>
30
31NS_ASSUME_NONNULL_BEGIN
32
33WK_CLASS_AVAILABLE(macos(10.13), ios(11.0))
34@interface WKSnapshotConfiguration : NSObject <NSCopying>
35
36/*! @abstract The rect to snapshot in view coordinates.
37 @discussion This rect should be contained within WKWebView's bounds. If the rect is set to the
38 null rect, the view's bounds will be used. The initial value is the null rect.
39 */
40@property (nonatomic) CGRect rect;
41
42/*! @abstract Specify a custom width to control the size of image you get back. The height will be
43 computed to maintain the aspect ratio established by rect.
44 @discussion snapshotWidth represents the width in points. If the snapshotWidth is nil, rect's
45 width will be used.
46 */
47@property (nullable, nonatomic, copy) NSNumber *snapshotWidth;
48
49/*! @abstract A Boolean value that specifies whether the snapshot should be taken after recent
50 changes have been incorporated. The value NO will capture the screen in its current state,
51 which might not include recent changes.
52 @discussion The default value is YES.
53 */
54@property (nonatomic) BOOL afterScreenUpdates WK_API_AVAILABLE(macos(10.15), ios(13.0));
55
56@end
57
58NS_ASSUME_NONNULL_END
