60/*! A class conforming to the WKUIDelegate protocol provides methods for
61 presenting native UI on behalf of a webpage.
62 */
63@protocol WKUIDelegate <NSObject>
64
65@optional
66
67/*! @abstract Creates a new web view.
68 @param webView The web view invoking the delegate method.
69 @param configuration The configuration to use when creating the new web
70 view. This configuration is a copy of webView.configuration.
71 @param navigationAction The navigation action causing the new web view to
72 be created.
73 @param windowFeatures Window features requested by the webpage.
74 @result A new web view or nil.
75 @discussion The web view returned must be created with the specified configuration. WebKit will load the request in the returned web view.
76
77 If you do not implement this method, the web view will cancel the navigation.
78 */
79- (nullable WKWebView *)webView:(WKWebView *)webView createWebViewWithConfiguration:(WKWebViewConfiguration *)configuration forNavigationAction:(WKNavigationAction *)navigationAction windowFeatures:(WKWindowFeatures *)windowFeatures;
80
81/*! @abstract Notifies your app that the DOM window object's close() method completed successfully.
82 @param webView The web view invoking the delegate method.
83 @discussion Your app should remove the web view from the view hierarchy and update
84 the UI as needed, such as by closing the containing browser tab or window.
85 */
86- (void)webViewDidClose:(WKWebView *)webView WK_API_AVAILABLE(macos(10.11), ios(9.0));
87
88/*! @abstract Displays a JavaScript alert panel.
89 @param webView The web view invoking the delegate method.
90 @param message The message to display.
91 @param frame Information about the frame whose JavaScript initiated this
92 call.
93 @param completionHandler The completion handler to call after the alert
94 panel has been dismissed.
95 @discussion For user security, your app should call attention to the fact
96 that a specific website controls the content in this panel. A simple forumla
97 for identifying the controlling website is frame.request.URL.host.
98 The panel should have a single OK button.
99
100 If you do not implement this method, the web view will behave as if the user selected the OK button.
101 */
102- (void)webView:(WKWebView *)webView runJavaScriptAlertPanelWithMessage:(NSString *)message initiatedByFrame:(WKFrameInfo *)frame completionHandler:(void (^)(void))completionHandler;
103
104/*! @abstract Displays a JavaScript confirm panel.
105 @param webView The web view invoking the delegate method.
106 @param message The message to display.
107 @param frame Information about the frame whose JavaScript initiated this call.
108 @param completionHandler The completion handler to call after the confirm
109 panel has been dismissed. Pass YES if the user chose OK, NO if the user
110 chose Cancel.
111 @discussion For user security, your app should call attention to the fact
112 that a specific website controls the content in this panel. A simple forumla
113 for identifying the controlling website is frame.request.URL.host.
114 The panel should have two buttons, such as OK and Cancel.
115
116 If you do not implement this method, the web view will behave as if the user selected the Cancel button.
117 */
118- (void)webView:(WKWebView *)webView runJavaScriptConfirmPanelWithMessage:(NSString *)message initiatedByFrame:(WKFrameInfo *)frame completionHandler:(void (^)(BOOL result))completionHandler;
119
120/*! @abstract Displays a JavaScript text input panel.
121 @param webView The web view invoking the delegate method.
122 @param prompt The prompt to display.
123 @param defaultText The initial text to display in the text entry field.
124 @param frame Information about the frame whose JavaScript initiated this call.
125 @param completionHandler The completion handler to call after the text
126 input panel has been dismissed. Pass the entered text if the user chose
127 OK, otherwise nil.
128 @discussion For user security, your app should call attention to the fact
129 that a specific website controls the content in this panel. A simple forumla
130 for identifying the controlling website is frame.request.URL.host.
131 The panel should have two buttons, such as OK and Cancel, and a field in
132 which to enter text.
133
134 If you do not implement this method, the web view will behave as if the user selected the Cancel button.
135 */
136- (void)webView:(WKWebView *)webView runJavaScriptTextInputPanelWithPrompt:(NSString *)prompt defaultText:(nullable NSString *)defaultText initiatedByFrame:(WKFrameInfo *)frame completionHandler:(void (^)(NSString * _Nullable result))completionHandler;

139/*! @abstract A delegate to request permission for microphone audio and camera video access.
140 @param webView The web view invoking the delegate method.
141 @param origin The origin of the page.
142 @param frame Information about the frame whose JavaScript initiated this call.
143 @param type The type of capture (camera, microphone).
144 @param decisionHandler The completion handler to call once the decision is made
145 @discussion If not implemented, the result is the same as calling the decisionHandler with WKPermissionDecisionPrompt.
146 */
147- (void)webView:(WKWebView *)webView requestMediaCapturePermissionForOrigin:(WKSecurityOrigin *)origin initiatedByFrame:(WKFrameInfo *)frame type:(WKMediaCaptureType)type decisionHandler:(void (^)(WKPermissionDecision decision))decisionHandler WK_SWIFT_ASYNC_NAME(webView(_:decideMediaCapturePermissionsFor:initiatedBy:type:)) WK_SWIFT_ASYNC(5) WK_API_AVAILABLE(macos(12.0), ios(15.0));

150 @param securityOrigin The security origin which requested access to the device's orientation and motion.
151 @param frame The frame that initiated the request.
152 @param decisionHandler The decision handler to call once the app has made its decision.
153 */
154- (void)webView:(WKWebView *)webView requestDeviceOrientationAndMotionPermissionForOrigin:(WKSecurityOrigin *)origin initiatedByFrame:(WKFrameInfo *)frame decisionHandler:(void (^)(WKPermissionDecision decision))decisionHandler WK_API_AVAILABLE(ios(15.0)) WK_API_UNAVAILABLE(macos);

158/*! @abstract Allows your app to determine whether or not the given element should show a preview.
159 @param webView The web view invoking the delegate method.
160 @param elementInfo The elementInfo for the element the user has started touching.
161 @discussion To disable previews entirely for the given element, return NO. Returning NO will prevent
162 webView:previewingViewControllerForElement:defaultActions: and webView:commitPreviewingViewController:
163 from being invoked.
164
165 This method will only be invoked for elements that have default preview in WebKit, which is
166 limited to links. In the future, it could be invoked for additional elements.
167 */
168- (BOOL)webView:(WKWebView *)webView shouldPreviewElement:(WKPreviewElementInfo *)elementInfo WK_API_DEPRECATED_WITH_REPLACEMENT("webView:contextMenuConfigurationForElement:completionHandler:", ios(10.0, 13.0));
169
170/*! @abstract Allows your app to provide a custom view controller to show when the given element is peeked.
171 @param webView The web view invoking the delegate method.
172 @param elementInfo The elementInfo for the element the user is peeking.
173 @param defaultActions An array of the actions that WebKit would use as previewActionItems for this element by
174 default. These actions would be used if allowsLinkPreview is YES but these delegate methods have not been
175 implemented, or if this delegate method returns nil.
176 @discussion Returning a view controller will result in that view controller being displayed as a peek preview.
177 To use the defaultActions, your app is responsible for returning whichever of those actions it wants in your
178 view controller's implementation of -previewActionItems.
179
180 Returning nil will result in WebKit's default preview behavior. webView:commitPreviewingViewController: will only be invoked
181 if a non-nil view controller was returned.
182 */
183- (nullable UIViewController *)webView:(WKWebView *)webView previewingViewControllerForElement:(WKPreviewElementInfo *)elementInfo defaultActions:(NSArray<id <WKPreviewActionItem>> *)previewActions WK_API_DEPRECATED_WITH_REPLACEMENT("webView:contextMenuConfigurationForElement:completionHandler:", ios(10.0, 13.0));
