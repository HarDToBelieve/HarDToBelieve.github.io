140- (void)_setAllowedAutoplayQuirks:(_WKWebsiteAutoplayQuirk)allowedQuirks
141{
142 OptionSet<WebKit::WebsiteAutoplayQuirk> quirks;
143
144 if (allowedQuirks & _WKWebsiteAutoplayQuirkInheritedUserGestures)
145 quirks.add(WebKit::WebsiteAutoplayQuirk::InheritedUserGestures);
146
147 if (allowedQuirks & _WKWebsiteAutoplayQuirkSynthesizedPauseEvents)
148 quirks.add(WebKit::WebsiteAutoplayQuirk::SynthesizedPauseEvents);
149
150 if (allowedQuirks & _WKWebsiteAutoplayQuirkArbitraryUserGestures)
151 quirks.add(WebKit::WebsiteAutoplayQuirk::ArbitraryUserGestures);
152
153 if (allowedQuirks & _WKWebsiteAutoplayQuirkPerDocumentAutoplayBehavior)
154 quirks.add(WebKit::WebsiteAutoplayQuirk::PerDocumentAutoplayBehavior);
155
156 _websitePolicies->setAllowedAutoplayQuirks(quirks);
157}
158
159- (_WKWebsiteAutoplayQuirk)_allowedAutoplayQuirks
160{
161 _WKWebsiteAutoplayQuirk quirks = 0;
162 auto allowedQuirks = _websitePolicies->allowedAutoplayQuirks();
163
164 if (allowedQuirks.contains(WebKit::WebsiteAutoplayQuirk::InheritedUserGestures))
165 quirks |= _WKWebsiteAutoplayQuirkInheritedUserGestures;
166
167 if (allowedQuirks.contains(WebKit::WebsiteAutoplayQuirk::SynthesizedPauseEvents))
168 quirks |= _WKWebsiteAutoplayQuirkSynthesizedPauseEvents;
169
170 if (allowedQuirks.contains(WebKit::WebsiteAutoplayQuirk::ArbitraryUserGestures))
171 quirks |= _WKWebsiteAutoplayQuirkArbitraryUserGestures;
172
173 if (allowedQuirks.contains(WebKit::WebsiteAutoplayQuirk::PerDocumentAutoplayBehavior))
174 quirks |= _WKWebsiteAutoplayQuirkPerDocumentAutoplayBehavior;
175
176 return quirks;
177}

190 break;
191 case _WKWebsiteAutoplayPolicyDeny:

193 break;
194 }
195}

291- (void)_setCustomHeaderFields:(NSArray<_WKCustomHeaderFields *> *)fields
292{
293 Vector<WebCore::CustomHeaderFields> vector;
294 vector.reserveInitialCapacity(fields.count);
295 for (_WKCustomHeaderFields *element in fields)
