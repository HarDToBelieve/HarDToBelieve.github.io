38static inline std::optional<WebsiteDataType> toWebsiteDataType(NSString *websiteDataType)
39{
40 if ([websiteDataType isEqualToString:WKWebsiteDataTypeCookies])
41 return WebsiteDataType::Cookies;
42 if ([websiteDataType isEqualToString:WKWebsiteDataTypeFetchCache])
43 return WebsiteDataType::DOMCache;
44 if ([websiteDataType isEqualToString:WKWebsiteDataTypeDiskCache])
45 return WebsiteDataType::DiskCache;
46 if ([websiteDataType isEqualToString:WKWebsiteDataTypeMemoryCache])
47 return WebsiteDataType::MemoryCache;
48 if ([websiteDataType isEqualToString:WKWebsiteDataTypeOfflineWebApplicationCache])
49 return WebsiteDataType::OfflineWebApplicationCache;
50 if ([websiteDataType isEqualToString:WKWebsiteDataTypeSessionStorage])
51 return WebsiteDataType::SessionStorage;
52 if ([websiteDataType isEqualToString:WKWebsiteDataTypeLocalStorage])
53 return WebsiteDataType::LocalStorage;
54 if ([websiteDataType isEqualToString:WKWebsiteDataTypeWebSQLDatabases])
55 return WebsiteDataType::WebSQLDatabases;
56 if ([websiteDataType isEqualToString:WKWebsiteDataTypeIndexedDBDatabases])
57 return WebsiteDataType::IndexedDBDatabases;
58#if ENABLE(SERVICE_WORKER)
59 if ([websiteDataType isEqualToString:WKWebsiteDataTypeServiceWorkerRegistrations])
60 return WebsiteDataType::ServiceWorkerRegistrations;
61#endif
62 if ([websiteDataType isEqualToString:_WKWebsiteDataTypeHSTSCache])
63 return WebsiteDataType::HSTSCache;
64 if ([websiteDataType isEqualToString:_WKWebsiteDataTypeMediaKeys])
65 return WebsiteDataType::MediaKeys;
66 if ([websiteDataType isEqualToString:_WKWebsiteDataTypeSearchFieldRecentSearches])
67 return WebsiteDataType::SearchFieldRecentSearches;
68#if ENABLE(NETSCAPE_PLUGIN_API)
69 if ([websiteDataType isEqualToString:_WKWebsiteDataTypePlugInData])
70 return WebsiteDataType::PlugInData;
71#endif
72 if ([websiteDataType isEqualToString:_WKWebsiteDataTypeResourceLoadStatistics])
73 return WebsiteDataType::ResourceLoadStatistics;
74 if ([websiteDataType isEqualToString:_WKWebsiteDataTypeCredentials])
75 return WebsiteDataType::Credentials;
76 if ([websiteDataType isEqualToString:_WKWebsiteDataTypeAdClickAttributions])
77 return WebsiteDataType::PrivateClickMeasurements;
78 if ([websiteDataType isEqualToString:_WKWebsiteDataTypePrivateClickMeasurements])
79 return WebsiteDataType::PrivateClickMeasurements;
80#if HAVE(CFNETWORK_ALTERNATIVE_SERVICE)
81 if ([websiteDataType isEqualToString:_WKWebsiteDataTypeAlternativeServices])
82 return WebsiteDataType::AlternativeServices;
83#endif
84 return std::nullopt;
85}

87static inline OptionSet<WebKit::WebsiteDataType> toWebsiteDataTypes(NSSet *websiteDataTypes)
88{
89 OptionSet<WebKit::WebsiteDataType> result;
90
91 for (NSString *websiteDataType in websiteDataTypes) {
92 if (auto dataType = toWebsiteDataType(websiteDataType))
93 result.add(*dataType);
94 }
95
96 return result;
97}

99static inline RetainPtr<NSSet> toWKWebsiteDataTypes(OptionSet<WebKit::WebsiteDataType> websiteDataTypes)
100{
101 auto wkWebsiteDataTypes = adoptNS([[NSMutableSet alloc] init]);
102
103 if (websiteDataTypes.contains(WebsiteDataType::Cookies))
104 [wkWebsiteDataTypes addObject:WKWebsiteDataTypeCookies];
105 if (websiteDataTypes.contains(WebsiteDataType::DiskCache))
106 [wkWebsiteDataTypes addObject:WKWebsiteDataTypeDiskCache];
107 if (websiteDataTypes.contains(WebsiteDataType::DOMCache))
108 [wkWebsiteDataTypes addObject:WKWebsiteDataTypeFetchCache];
109 if (websiteDataTypes.contains(WebsiteDataType::MemoryCache))
110 [wkWebsiteDataTypes addObject:WKWebsiteDataTypeMemoryCache];
111 if (websiteDataTypes.contains(WebsiteDataType::OfflineWebApplicationCache))
112 [wkWebsiteDataTypes addObject:WKWebsiteDataTypeOfflineWebApplicationCache];
113 if (websiteDataTypes.contains(WebsiteDataType::SessionStorage))
114 [wkWebsiteDataTypes addObject:WKWebsiteDataTypeSessionStorage];
115 if (websiteDataTypes.contains(WebsiteDataType::LocalStorage))
116 [wkWebsiteDataTypes addObject:WKWebsiteDataTypeLocalStorage];
117 if (websiteDataTypes.contains(WebsiteDataType::WebSQLDatabases))
118 [wkWebsiteDataTypes addObject:WKWebsiteDataTypeWebSQLDatabases];
119 if (websiteDataTypes.contains(WebsiteDataType::IndexedDBDatabases))
120 [wkWebsiteDataTypes addObject:WKWebsiteDataTypeIndexedDBDatabases];
121#if ENABLE(SERVICE_WORKER)
122 if (websiteDataTypes.contains(WebsiteDataType::ServiceWorkerRegistrations))
123 [wkWebsiteDataTypes addObject:WKWebsiteDataTypeServiceWorkerRegistrations];
124#endif
125 if (websiteDataTypes.contains(WebsiteDataType::HSTSCache))
126 [wkWebsiteDataTypes addObject:_WKWebsiteDataTypeHSTSCache];
127 if (websiteDataTypes.contains(WebsiteDataType::MediaKeys))
128 [wkWebsiteDataTypes addObject:_WKWebsiteDataTypeMediaKeys];
129 if (websiteDataTypes.contains(WebsiteDataType::SearchFieldRecentSearches))
130 [wkWebsiteDataTypes addObject:_WKWebsiteDataTypeSearchFieldRecentSearches];
131#if ENABLE(NETSCAPE_PLUGIN_API)
132 if (websiteDataTypes.contains(WebsiteDataType::PlugInData))
133 [wkWebsiteDataTypes addObject:_WKWebsiteDataTypePlugInData];
134#endif
135 if (websiteDataTypes.contains(WebsiteDataType::ResourceLoadStatistics))
136 [wkWebsiteDataTypes addObject:_WKWebsiteDataTypeResourceLoadStatistics];
137 if (websiteDataTypes.contains(WebsiteDataType::Credentials))
138 [wkWebsiteDataTypes addObject:_WKWebsiteDataTypeCredentials];
139 if (websiteDataTypes.contains(WebsiteDataType::PrivateClickMeasurements))
140 [wkWebsiteDataTypes addObject:_WKWebsiteDataTypePrivateClickMeasurements];
141#if HAVE(CFNETWORK_ALTERNATIVE_SERVICE)
142 if (websiteDataTypes.contains(WebsiteDataType::AlternativeServices))
143 [wkWebsiteDataTypes addObject:_WKWebsiteDataTypeAlternativeServices];
144#endif
145
146 return wkWebsiteDataTypes;
147}
