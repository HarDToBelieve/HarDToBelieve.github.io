35- (void)dealloc
36{
37 if (WebCoreObjCScheduleDeallocateOnMainRunLoop(WKWindowFeatures.class, self))

40 _windowFeatures->API::WindowFeatures::~WindowFeatures();

42 [super dealloc];
43}
