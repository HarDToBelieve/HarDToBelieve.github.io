42static const NSInteger UnspecifiedAttachmentErrorCode = 1;
43static const NSInteger InvalidAttachmentErrorCode = 2;

46@end

48@implementation _WKAttachmentInfo {
49 RetainPtr<NSFileWrapper> _fileWrapper;
50 RetainPtr<NSString> _mimeType;
51 RetainPtr<NSString> _utiType;
52 RetainPtr<NSString> _filePath;
53}
54
55- (instancetype)initWithFileWrapper:(NSFileWrapper *)fileWrapper filePath:(NSString *)filePath mimeType:(NSString *)mimeType utiType:(NSString *)utiType
56{
57 if (!(self = [super init]))
58 return nil;
59
60 _fileWrapper = fileWrapper;
61 _filePath = filePath;
62 _mimeType = mimeType;
63 _utiType = utiType;
64 return self;
65}
66
67- (NSData *)data
68{
69 if (![_fileWrapper isRegularFile]) {
70 // FIXME: Handle attachments backed by NSFileWrappers that represent directories.
71 return nil;
72 }
73
74 return [_fileWrapper regularFileContents];
75}
76
77- (NSString *)name
78{
79 if ([_fileWrapper filename].length)
80 return [_fileWrapper filename];
81
82 return [_fileWrapper preferredFilename];
83}
84
85- (NSString *)filePath
86{
87 return _filePath.get();
88}
89
90- (NSFileWrapper *)fileWrapper
91{
92 return _fileWrapper.get();
93}
94
95- (NSString *)contentType
96{
97 if ([_mimeType length])
98 return _mimeType.get();
99
100 return _utiType.get();
101}
102
103@end

105@implementation _WKAttachment

107- (void)dealloc
108{
109 if (WebCoreObjCScheduleDeallocateOnMainRunLoop(_WKAttachment.class, self))
110 return;
111
112 _attachment->~Attachment();
113
114 [super dealloc];
115}
116
117- (API::Object&)_apiObject
118{
119 return *_attachment;
120}
121
122- (_WKAttachmentInfo *)info
123{
124 if (!_attachment->isValid())
125 return nil;
126
127 return adoptNS([[_WKAttachmentInfo alloc] initWithFileWrapper:_attachment->fileWrapper() filePath:_attachment->filePath() mimeType:_attachment->mimeType() utiType:_attachment->utiType()]).autorelease();
128}
129
130- (void)requestInfo:(void(^)(_WKAttachmentInfo *, NSError *))completionHandler
131{
132 completionHandler(self.info, nil);
133}
134
135- (void)setFileWrapper:(NSFileWrapper *)fileWrapper contentType:(NSString *)contentType completion:(void (^)(NSError *))completionHandler
136{
137 if (!_attachment->isValid()) {
138 if (completionHandler)
139 completionHandler([NSError errorWithDomain:WKErrorDomain code:InvalidAttachmentErrorCode userInfo:nil]);
140 return;
141 }
142
143 // This file path member is only populated when the attachment is generated upon dropping files. When data is specified via NSFileWrapper
144 // from the SPI client, the corresponding file path of the data is unknown, if it even exists at all.
145 _attachment->setFilePath({ });
146 _attachment->setFileWrapperAndUpdateContentType(fileWrapper, contentType);
147 _attachment->updateAttributes([capturedBlock = makeBlockPtr(completionHandler)] {
148 if (capturedBlock)
149 capturedBlock(nil);
150 });
151}
152
153- (void)setData:(NSData *)data newContentType:(NSString *)newContentType newFilename:(NSString *)newFilename completion:(void(^)(NSError *))completionHandler
154{
155 auto fileWrapper = adoptNS([[NSFileWrapper alloc] initRegularFileWithContents:data]);
156 if (newFilename)
157 [fileWrapper setPreferredFilename:newFilename];
158 [self setFileWrapper:fileWrapper.get() contentType:newContentType completion:completionHandler];
159}
160
161- (NSString *)uniqueIdentifier
162{
163 return _attachment->identifier();
164}
165
166- (NSString *)description
167{
168 return [NSString stringWithFormat:@"<%@ %p id='%@'>", [self class], self, self.uniqueIdentifier];
169}
170
171- (BOOL)isConnected
172{
173 return _attachment->insertionState() == API::Attachment::InsertionState::Inserted;
174}
