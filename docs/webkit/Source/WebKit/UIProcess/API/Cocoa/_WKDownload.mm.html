37ALLOW_DEPRECATED_DECLARATIONS_BEGIN
38static NSMapTable<WKDownload *, _WKDownload *> *downloadWrapperMap()
39{
40 static NeverDestroyed<RetainPtr<NSMapTable>> table;
41 if (!table.get())
42 table.get() = [NSMapTable weakToWeakObjectsMapTable];
43 return table.get().get();
44}
45ALLOW_DEPRECATED_DECLARATIONS_END

51- (instancetype)initWithDownload2:(WKDownload *)download
52{
53 if (!(self = [super init]))
54 return nil;
55 _download = download;
56 return self;
57}
58
59+ (instancetype)downloadWithDownload:(WKDownload *)download
60{
61 if (_WKDownload *wrapper = [downloadWrapperMap() objectForKey:download])
62 return wrapper;
63 auto wrapper = adoptNS([[_WKDownload alloc] initWithDownload2:download]);
64 [downloadWrapperMap() setObject:wrapper.get() forKey:download];
65 return wrapper.autorelease();
66}

68- (void)cancel
69{
70 _download->_download->cancel([download = makeRef(*_download->_download)] (auto*) {
71 download->client().legacyDidCancel(download.get());
72 });
73}
