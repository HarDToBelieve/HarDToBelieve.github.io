26#import "config.h"
27#import "_WKInspectorConfigurationInternal.h"
28
29#import "WKProcessPoolInternal.h"
30#import <WebKit/WKWebViewConfigurationPrivate.h>
31#import "WebURLSchemeHandlerCocoa.h"
32#import <WebCore/WebCoreObjCExtras.h>
33
34@implementation _WKInspectorConfiguration
35
36- (instancetype)init
37{
38 if (!(self = [super init]))
39 return nil;
40
41 API::Object::constructInWrapper<API::InspectorConfiguration>(self);
42
43 return self;
44}
45
46- (void)dealloc
47{
48 if (WebCoreObjCScheduleDeallocateOnMainRunLoop(_WKInspectorConfiguration.class, self))
49 return;
50 _configuration->API::InspectorConfiguration::~InspectorConfiguration();
51 [super dealloc];
52}
53
54- (API::Object&)_apiObject
55{
56 return *_configuration;
57}
58
59- (void)setURLSchemeHandler:(id <WKURLSchemeHandler>)urlSchemeHandler forURLScheme:(NSString *)urlScheme
60{
61 _configuration->addURLSchemeHandler(WebKit::WebURLSchemeHandlerCocoa::create(urlSchemeHandler), urlScheme);
62}
63
64- (void)setProcessPool:(WKProcessPool *)processPool
65{
66 _configuration->setProcessPool(processPool ? processPool->_processPool.get() : nullptr);
67}
68
69- (WKProcessPool *)processPool
70{
71 return wrapper(_configuration->processPool());
72}
73
74- (void)applyToWebViewConfiguration:(WKWebViewConfiguration *)configuration
75{
76 for (auto pair : _configuration->urlSchemeHandlers()) {
77 auto& handler = static_cast<WebKit::WebURLSchemeHandlerCocoa&>(pair.first.get());
78 [configuration setURLSchemeHandler:handler.apiHandler() forURLScheme:pair.second];
79 }
80
81 if (auto* processPool = self.processPool)
82 [configuration setProcessPool:processPool];
83
84 if (auto* groupIdentifier = self.groupIdentifier)
85 [configuration _setGroupIdentifier:groupIdentifier];
86}
87
88- (id)copyWithZone:(NSZone *)zone
89{
90 _WKInspectorConfiguration *configuration = [(_WKInspectorConfiguration *)[[self class] allocWithZone:zone] init];
91
92 for (auto pair : _configuration->urlSchemeHandlers()) {
93 auto& handler = static_cast<WebKit::WebURLSchemeHandlerCocoa&>(pair.first.get());
94 [configuration setURLSchemeHandler:handler.apiHandler() forURLScheme:pair.second];
95 }
96
97 if (auto* processPool = self.processPool)
98 [configuration setProcessPool:processPool];
99
100 if (auto* groupIdentifier = self.groupIdentifier)
101 [configuration setGroupIdentifier:groupIdentifier];
102
103 return configuration;
104}
105
106@end
