45inline Inspector::DebuggableType fromWKInspectorDebuggableType(_WKInspectorDebuggableType debuggableType)
46{
47 switch (debuggableType) {
48 case _WKInspectorDebuggableTypeITML:
49 return Inspector::DebuggableType::ITML;
50 case _WKInspectorDebuggableTypeJavaScript:
51 return Inspector::DebuggableType::JavaScript;
52 case _WKInspectorDebuggableTypePage:
53 return Inspector::DebuggableType::Page;
54 case _WKInspectorDebuggableTypeServiceWorker:
55 return Inspector::DebuggableType::ServiceWorker;
56 case _WKInspectorDebuggableTypeWebPage:
57 return Inspector::DebuggableType::WebPage;
58 }
59
60 ASSERT_NOT_REACHED();
61 return Inspector::DebuggableType::JavaScript;
62}
63
64inline _WKInspectorDebuggableType toWKInspectorDebuggableType(Inspector::DebuggableType debuggableType)
65{
66 switch (debuggableType) {
67 case Inspector::DebuggableType::ITML:
68 return _WKInspectorDebuggableTypeITML;
69 case Inspector::DebuggableType::JavaScript:
70 return _WKInspectorDebuggableTypeJavaScript;
71 case Inspector::DebuggableType::Page:
72 return _WKInspectorDebuggableTypePage;
73 case Inspector::DebuggableType::ServiceWorker:
74 return _WKInspectorDebuggableTypeServiceWorker;
75 case Inspector::DebuggableType::WebPage:
76 return _WKInspectorDebuggableTypeWebPage;
77 }
78
79 ASSERT_NOT_REACHED();
80 return _WKInspectorDebuggableTypeJavaScript;
