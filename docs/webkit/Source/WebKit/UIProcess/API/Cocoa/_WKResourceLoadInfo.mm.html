36static _WKResourceLoadInfoResourceType toWKResourceLoadInfoResourceType(WebKit::ResourceLoadInfo::Type type)
37{
38 using namespace WebKit;
39 switch (type) {
40 case ResourceLoadInfo::Type::ApplicationManifest:
41 return _WKResourceLoadInfoResourceTypeApplicationManifest;
42 case ResourceLoadInfo::Type::Beacon:
43 return _WKResourceLoadInfoResourceTypeBeacon;
44 case ResourceLoadInfo::Type::CSPReport:
45 return _WKResourceLoadInfoResourceTypeCSPReport;
46 case ResourceLoadInfo::Type::Document:
47 return _WKResourceLoadInfoResourceTypeDocument;
48 case ResourceLoadInfo::Type::Fetch:
49 return _WKResourceLoadInfoResourceTypeFetch;
50 case ResourceLoadInfo::Type::Font:
51 return _WKResourceLoadInfoResourceTypeFont;
52 case ResourceLoadInfo::Type::Image:
53 return _WKResourceLoadInfoResourceTypeImage;
54 case ResourceLoadInfo::Type::Media:
55 return _WKResourceLoadInfoResourceTypeMedia;
56 case ResourceLoadInfo::Type::Object:
57 return _WKResourceLoadInfoResourceTypeObject;
58 case ResourceLoadInfo::Type::Other:
59 return _WKResourceLoadInfoResourceTypeOther;
60 case ResourceLoadInfo::Type::Ping:
61 return _WKResourceLoadInfoResourceTypePing;
62 case ResourceLoadInfo::Type::Script:
63 return _WKResourceLoadInfoResourceTypeScript;
64 case ResourceLoadInfo::Type::Stylesheet:
65 return _WKResourceLoadInfoResourceTypeStylesheet;
66 case ResourceLoadInfo::Type::XMLHTTPRequest:
67 return _WKResourceLoadInfoResourceTypeXMLHTTPRequest;
68 case ResourceLoadInfo::Type::XSLT:
69 return _WKResourceLoadInfoResourceTypeXSLT;
70 }
71
72 ASSERT_NOT_REACHED();
73 return _WKResourceLoadInfoResourceTypeOther;
74}

141- (instancetype)initWithCoder:(NSCoder *)coder
142{
143 if (!(self = [super init]))
144 return nil;
145
146 NSNumber *resourceLoadID = [coder decodeObjectOfClass:[NSNumber class] forKey:@"resourceLoadID"];
147 if (!resourceLoadID) {
148 [self release];
149 return nil;
150 }
151
152 NSNumber *frame = [coder decodeObjectOfClass:[NSNumber class] forKey:@"frame"];
153 if (!frame) {
154 [self release];
155 return nil;
156 }
157
158 NSNumber *parentFrame = [coder decodeObjectOfClass:[NSNumber class] forKey:@"parentFrame"];
159 if (!parentFrame) {
160 [self release];
161 return nil;
162 }
163
164 NSURL *originalURL = [coder decodeObjectOfClass:[NSURL class] forKey:@"originalURL"];
165 if (!originalURL) {
166 [self release];
167 return nil;
168 }
169
170 NSString *originalHTTPMethod = [coder decodeObjectOfClass:[NSString class] forKey:@"originalHTTPMethod"];
171 if (!originalHTTPMethod) {
172 [self release];
173 return nil;
174 }
175
176 NSDate *eventTimestamp = [coder decodeObjectOfClass:[NSDate class] forKey:@"eventTimestamp"];
177 if (!eventTimestamp) {
178 [self release];
179 return nil;
180 }
181
182 NSNumber *loadedFromCache = [coder decodeObjectOfClass:[NSNumber class] forKey:@"loadedFromCache"];
183 if (!loadedFromCache) {
184 [self release];
185 return nil;
186 }
187
188 NSNumber *type = [coder decodeObjectOfClass:[NSNumber class] forKey:@"type"];
189 if (!type) {
190 [self release];
191 return nil;
192 }
193
194 WebKit::ResourceLoadInfo info {
195 makeObjectIdentifier<WebKit::NetworkResourceLoadIdentifierType>(resourceLoadID.unsignedLongLongValue),
196 makeObjectIdentifier<WebCore::FrameIdentifierType>(frame.unsignedLongLongValue),
197 makeObjectIdentifier<WebCore::FrameIdentifierType>(parentFrame.unsignedLongLongValue),
198 originalURL,
199 originalHTTPMethod,
200 WallTime::fromRawSeconds(eventTimestamp.timeIntervalSince1970),
201 static_cast<bool>(loadedFromCache.boolValue),
202 static_cast<WebKit::ResourceLoadInfo::Type>(type.unsignedCharValue),
203 };
204
205 API::Object::constructInWrapper<API::ResourceLoadInfo>(self, WTFMove(info));
206
207 return self;
208}

215 [coder encodeObject:self.originalURL forKey:@"originalURL"];
216 [coder encodeObject:self.originalHTTPMethod forKey:@"originalHTTPMethod"];
217 [coder encodeObject:self.eventTimestamp forKey:@"eventTimestamp"];
218 [coder encodeObject:@(self.loadedFromCache) forKey:@"loadedFromCache"];
219 [coder encodeObject:@(static_cast<unsigned char>(_info->resourceLoadType())) forKey:@"type"];
220}
