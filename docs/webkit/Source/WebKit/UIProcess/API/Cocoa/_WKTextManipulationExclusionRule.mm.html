26#import "config.h"
27#import "_WKTextManipulationExclusionRule.h"
28
29#import <wtf/RetainPtr.h>
30
31@implementation _WKTextManipulationExclusionRule {
32 BOOL _isExclusion;
33 RetainPtr<NSString> _elementName;
34 RetainPtr<NSString> _attributeName;
35 RetainPtr<NSString> _attributeValue;
36 RetainPtr<NSString> _className;
37}
38
39- (instancetype)initExclusion:(BOOL)exclusion forElement:(NSString *)localName
40{
41 if (!(self = [super init]))
42 return nil;
43
44 _isExclusion = exclusion;
45 _elementName = localName;
46
47 return self;
48}
49
50- (instancetype)initExclusion:(BOOL)exclusion forAttribute:(NSString *)name value:(NSString *)value
51{
52 if (!(self = [super init]))
53 return nil;
54
55 _isExclusion = exclusion;
56 _attributeName = name;
57 _attributeValue = value;
58
59 return self;
60}
61
62- (instancetype)initExclusion:(BOOL)exclusion forClass:(NSString *)className
63{
64 if (!(self = [super init]))
65 return nil;
66
67 _isExclusion = exclusion;
68 _className = className;
69
70 return self;
71}
72
73- (NSString *)elementName
74{
75 return _elementName.get();
76}
77
78- (NSString *)attributeName
79{
80 return _attributeName.get();
81}
82
83- (NSString *)attributeValue
84{
85 return _attributeValue.get();
86}
87
88- (NSString *)className
89{
90 return _className.get();
91}
92
93@end
