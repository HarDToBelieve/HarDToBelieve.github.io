50- (void)setUserInfo:(NSDictionary<NSString *, id> *)userInfo
51{
52 if (userInfo == _userInfo || [_userInfo isEqual:userInfo])
53 return;
54
55 _userInfo = adoptNS(userInfo.copy);
56}

63static BOOL isEqualOrBothNil(id a, id b)
64{
65 if (a == b)
66 return YES;
67
68 return [a isEqual:b];
69}
70
71- (BOOL)isEqual:(id)object
72{
73 if (self == object)
74 return YES;
75
76 if (![object isKindOfClass:[self class]])
77 return NO;
78
79 return [self isEqualToTextManipulationToken:object includingContentEquality:YES];
80}
81
82- (BOOL)isEqualToTextManipulationToken:(_WKTextManipulationToken *)otherToken includingContentEquality:(BOOL)includingContentEquality
83{
84 if (!otherToken)
85 return NO;
86
87 BOOL equalIdentifiers = isEqualOrBothNil(self.identifier, otherToken.identifier);
88 BOOL equalExclusion = self.isExcluded == otherToken.isExcluded;
89 BOOL equalContent = !includingContentEquality || isEqualOrBothNil(self.content, otherToken.content);
90 BOOL equalUserInfo = isEqualOrBothNil(self.userInfo, otherToken.userInfo);
91
92 return equalIdentifiers && equalExclusion && equalContent && equalUserInfo;
93}

95- (NSString *)description

100- (NSString *)debugDescription

102 return [self _descriptionPreservingPrivacy:NO];

105- (NSString *)_descriptionPreservingPrivacy:(BOOL)preservePrivacy
106{
107 NSMutableString *description = [NSMutableString stringWithFormat:@"<%@: %p; identifier = %@; isExcluded = %i", self.class, self, self.identifier, self.isExcluded];
108 if (preservePrivacy)
109 [description appendFormat:@"; content length = %lu", (unsigned long)self.content.length];
110 else
111 [description appendFormat:@"; content = %@; user info = %@", self.content, self.userInfo];
112
113 [description appendString:@">"];
114
115 return adoptNS([description copy]).autorelease();
116}
