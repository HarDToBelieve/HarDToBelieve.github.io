40@implementation _WKUserStyleSheet
41
42- (instancetype)initWithSource:(NSString *)source forMainFrameOnly:(BOOL)forMainFrameOnly
43{
44 if (!(self = [super init]))
45 return nil;
46
47 // FIXME: In the API test, we can use generateUniqueURL below before the API::Object constructor has done this... where should this really be?
48 WebKit::InitializeWebKit2();
49
50 API::Object::constructInWrapper<API::UserStyleSheet>(self, WebCore::UserStyleSheet { source, API::UserStyleSheet::generateUniqueURL(), { }, { }, forMainFrameOnly ? WebCore::UserContentInjectedFrames::InjectInTopFrameOnly : WebCore::UserContentInjectedFrames::InjectInAllFrames, WebCore::UserStyleUserLevel }, API::ContentWorld::pageContentWorld());
51
52 return self;
53}
54
55- (instancetype)initWithSource:(NSString *)source forWKWebView:(WKWebView *)webView forMainFrameOnly:(BOOL)forMainFrameOnly includeMatchPatternStrings:(NSArray<NSString *> *)includeMatchPatternStrings excludeMatchPatternStrings:(NSArray<NSString *> *)excludeMatchPatternStrings baseURL:(NSURL *)baseURL level:(_WKUserStyleLevel)level contentWorld:(WKContentWorld *)contentWorld
56{
57
58 WebKit::InitializeWebKit2();
59
60 API::Object::constructInWrapper<API::UserStyleSheet>(self, WebCore::UserStyleSheet { source, baseURL, makeVector<String>(includeMatchPatternStrings), makeVector<String>(excludeMatchPatternStrings), forMainFrameOnly ? WebCore::UserContentInjectedFrames::InjectInTopFrameOnly : WebCore::UserContentInjectedFrames::InjectInAllFrames, API::toWebCoreUserStyleLevel(level), webView ? std::optional<WebCore::PageIdentifier>([webView _page]->webPageID()) : std::nullopt }, contentWorld ? *contentWorld->_contentWorld : API::ContentWorld::pageContentWorld());
61
62 return self;
63}
64
65- (void)dealloc
66{
67 if (WebCoreObjCScheduleDeallocateOnMainRunLoop(_WKUserStyleSheet.class, self))
68 return;
69
70 _userStyleSheet->~UserStyleSheet();
71
72 [super dealloc];
73}
74
75- (NSString *)source
76{
77 return _userStyleSheet->userStyleSheet().source();
78}
79
80- (NSURL *)baseURL
81{
82 return _userStyleSheet->userStyleSheet().url();
83}
84
85- (BOOL)isForMainFrameOnly
86{
87 return _userStyleSheet->userStyleSheet().injectedFrames() == WebCore::UserContentInjectedFrames::InjectInTopFrameOnly;
88}
89
90- (id)copyWithZone:(NSZone *)zone
91{
92 return [self retain];
93}
