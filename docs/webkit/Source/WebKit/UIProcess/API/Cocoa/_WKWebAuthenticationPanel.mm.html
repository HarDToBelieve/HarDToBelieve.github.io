62static RetainPtr<NSData> produceClientDataJson(_WKWebAuthenticationType type, NSData *challenge, NSString *origin)
63{
64 auto dictionary = adoptNS([[NSMutableDictionary alloc] init]);
65 switch (type) {
66 case _WKWebAuthenticationTypeCreate:
67 [dictionary setObject:@"webauthn.create" forKey:@"type"];
68 break;
69 case _WKWebAuthenticationTypeGet:
70 [dictionary setObject:@"webauthn.get" forKey:@"type"];
71 break;
72 }
73 [dictionary setObject:base64URLEncodeToString(challenge.bytes, challenge.length) forKey:@"challenge"];
74 [dictionary setObject:origin forKey:@"origin"];
75
76 return [NSJSONSerialization dataWithJSONObject:dictionary.get() options:NSJSONWritingSortedKeys error:nil];
77}

79static Vector<uint8_t> produceClientDataJsonHash(NSData *clientDataJson)
80{
81 auto crypto = PAL::CryptoDigest::create(PAL::CryptoDigest::Algorithm::SHA_256);
82 crypto->addBytes(clientDataJson.bytes, clientDataJson.length);
83 return crypto->computeHash();
84}

202static RetainPtr<NSArray> getAllLocalAuthenticatorCredentialsImpl(NSString *accessGroup)
203{
204 NSDictionary *query = @{
205 (__bridge id)kSecClass: (__bridge id)kSecClassKey,
206 (__bridge id)kSecAttrKeyClass: (__bridge id)kSecAttrKeyClassPrivate,
207 (__bridge id)kSecAttrAccessGroup: accessGroup,
208 (__bridge id)kSecReturnAttributes: @YES,
209 (__bridge id)kSecMatchLimit: (__bridge id)kSecMatchLimitAll,
210#if HAVE(DATA_PROTECTION_KEYCHAIN)
211 (__bridge id)kSecUseDataProtectionKeychain: @YES
212#else
213 (__bridge id)kSecAttrNoLegacy: @YES
214#endif
215 };
216 CFTypeRef attributesArrayRef = nullptr;
217 OSStatus status = SecItemCopyMatching((__bridge CFDictionaryRef)query, &attributesArrayRef);
218 if (status && status != errSecItemNotFound)
219 return nullptr;
220 auto retainAttributesArray = adoptCF(attributesArrayRef);
221
222 auto result = adoptNS([[NSMutableArray alloc] init]);
223 for (NSDictionary *attributes in (NSArray *)attributesArrayRef) {
224 auto decodedResponse = cbor::CBORReader::read(vectorFromNSData(attributes[(__bridge id)kSecAttrApplicationTag]));
225 if (!decodedResponse || !decodedResponse->isMap()) {
226 ASSERT_NOT_REACHED();
227 return nullptr;
228 }
229 auto& responseMap = decodedResponse->getMap();
230
231 auto it = responseMap.find(cbor::CBORValue(fido::kEntityNameMapKey));
232 if (it == responseMap.end() || !it->second.isString()) {
233 ASSERT_NOT_REACHED();
234 return nullptr;
235 }
236 auto& username = it->second.getString();
237
238 [result addObject:@{
239 _WKLocalAuthenticatorCredentialNameKey: username,
240 _WKLocalAuthenticatorCredentialIDKey: attributes[(__bridge id)kSecAttrApplicationLabel],
241 _WKLocalAuthenticatorCredentialRelyingPartyIDKey: attributes[(__bridge id)kSecAttrLabel]
242 }];
243 }
244
245 return result;
246}
247#endif

434{

453 return result;

473- (void)makeCredentialWithChallenge:(NSData *)challenge origin:(NSString *)origin options:(_WKPublicKeyCredentialCreationOptions *)options completionHandler:(void (^)(_WKAuthenticatorAttestationResponse *, NSError *))handler
474{
475#if ENABLE(WEB_AUTHN)

478 auto callback = [handler = makeBlockPtr(handler), clientDataJSON = WTFMove(clientDataJSON)] (Variant<Ref<WebCore::AuthenticatorResponse>, WebCore::ExceptionData>&& result) mutable {
479 WTF::switchOn(result, [&](const Ref<WebCore::AuthenticatorResponse>& response) {
480 handler(wkAuthenticatorAttestationResponse(response->data(), clientDataJSON.get(), response->attachment()).get(), nil);
481 }, [&](const WebCore::ExceptionData& exception) {
482 handler(nil, [NSError errorWithDomain:WKErrorDomain code:WKErrorUnknown userInfo:nil]);
483 });
484 };
485 _panel->handleRequest({ WTFMove(hash), [_WKWebAuthenticationPanel convertToCoreCreationOptionsWithOptions:options], nullptr, WebKit::WebAuthenticationPanelResult::Unavailable, nullptr, std::nullopt, { }, true, String(), nullptr }, WTFMove(callback));
