34template<typename T> class WKRetainPtr {
35public:
36 typedef T PtrType;
37
38 WKRetainPtr()
39 : m_ptr(0)
40 {
41 }
42
43 WKRetainPtr(PtrType ptr)
44 : m_ptr(ptr)
45 {
46 if (ptr)
47 WKRetain(ptr);
48 }
49
50 template<typename U> WKRetainPtr(const WKRetainPtr<U>& o)
51 : m_ptr(o.get())
52 {
53 if (PtrType ptr = m_ptr)
54 WKRetain(ptr);
55 }
56
57 WKRetainPtr(const WKRetainPtr& o)
58 : m_ptr(o.m_ptr)
59 {
60 if (PtrType ptr = m_ptr)
61 WKRetain(ptr);
62 }
63
64 template<typename U> WKRetainPtr(WKRetainPtr<U>&& o)
65 : m_ptr(o.leakRef())
66 {
67 }
68
69 WKRetainPtr(WKRetainPtr&& o)
70 : m_ptr(o.leakRef())
71 {
72 }
73
74 ~WKRetainPtr()
75 {
76 if (PtrType ptr = m_ptr)
77 WKRelease(ptr);
78 }
79
80 // Hash table deleted values, which are only constructed and never copied or destroyed.
81 WKRetainPtr(WTF::HashTableDeletedValueType)
82 : m_ptr(hashTableDeletedValue())
83 {
84 }
85
86 bool isHashTableDeletedValue() const { return m_ptr == hashTableDeletedValue(); }
87 constexpr static T hashTableDeletedValue() { return reinterpret_cast<T>(-1); }
88
89 PtrType get() const { return m_ptr; }
90
91 void clear()
92 {
93 PtrType ptr = m_ptr;
94 m_ptr = 0;
95 if (ptr)
96 WKRelease(ptr);
97 }
98
99 PtrType leakRef()
100 {
101 PtrType ptr = m_ptr;
102 m_ptr = 0;
103 return ptr;
104 }
105
106 PtrType operator->() const { return m_ptr; }
107 bool operator!() const { return !m_ptr; }
108
109 // This conversion operator allows implicit conversion to bool but not to other integer types.
110 typedef PtrType WKRetainPtr::*UnspecifiedBoolType;
111 operator UnspecifiedBoolType() const { return m_ptr ? &WKRetainPtr::m_ptr : 0; }
112
113 WKRetainPtr& operator=(const WKRetainPtr&);
114 template<typename U> WKRetainPtr& operator=(const WKRetainPtr<U>&);
115 WKRetainPtr& operator=(PtrType);
116 template<typename U> WKRetainPtr& operator=(U*);
117
118 WKRetainPtr& operator=(WKRetainPtr&&);
119 template<typename U> WKRetainPtr& operator=(WKRetainPtr<U>&&);
120
121 void adopt(PtrType);
122 void swap(WKRetainPtr&);
123
124private:
125 template<typename U> friend WKRetainPtr<U> adoptWK(U);
126 enum WKAdoptTag { AdoptWK };
127 WKRetainPtr(WKAdoptTag, PtrType ptr)
128 : m_ptr(ptr) { }
129
130 PtrType m_ptr;
131};
132
133template<typename T> inline WKRetainPtr<T>& WKRetainPtr<T>::operator=(const WKRetainPtr<T>& o)
134{
135 PtrType optr = o.get();
136 if (optr)
137 WKRetain(optr);
138 PtrType ptr = m_ptr;
139 m_ptr = optr;
140 if (ptr)
141 WKRelease(ptr);
142 return *this;
143}

179template<typename T> inline WKRetainPtr<T>& WKRetainPtr<T>::operator=(WKRetainPtr<T>&& o)

182 return *this;
183}

191template<typename T> inline void WKRetainPtr<T>::adopt(PtrType optr)
192{
193 PtrType ptr = m_ptr;
194 m_ptr = optr;
195 if (ptr)
196 WKRelease(ptr);
197}

199template<typename T> inline void WKRetainPtr<T>::swap(WKRetainPtr<T>& o)
200{
201 std::swap(m_ptr, o.m_ptr);
202}
203
204template<typename T> inline void swap(WKRetainPtr<T>& a, WKRetainPtr<T>& b)
205{
206 a.swap(b);
207}

209template<typename T, typename U> inline bool operator==(const WKRetainPtr<T>& a, const WKRetainPtr<U>& b)
210{
211 return a.get() == b.get();
212}
213
214template<typename T, typename U> inline bool operator==(const WKRetainPtr<T>& a, U* b)
215{
216 return a.get() == b;
217}
218
219template<typename T, typename U> inline bool operator==(T* a, const WKRetainPtr<U>& b)
220{
221 return a == b.get();
222}
223
224template<typename T, typename U> inline bool operator!=(const WKRetainPtr<T>& a, const WKRetainPtr<U>& b)
225{
226 return a.get() != b.get();
227}
228
229template<typename T, typename U> inline bool operator!=(const WKRetainPtr<T>& a, U* b)
230{
231 return a.get() != b;
232}
233
234template<typename T, typename U> inline bool operator!=(T* a, const WKRetainPtr<U>& b)
235{
236 return a != b.get();
237}

244template<typename T> inline WKRetainPtr<T> adoptWK(T o)
245{
246 return WKRetainPtr<T>(WKRetainPtr<T>::AdoptWK, o);
247}
248
249template<typename T> inline WKRetainPtr<T> retainWK(T ptr)
250{
251 return ptr;
252}
