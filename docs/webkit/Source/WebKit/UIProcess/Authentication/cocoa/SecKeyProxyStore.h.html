28#if HAVE(SEC_KEY_PROXY)
29
30#include <wtf/RefCounted.h>
31#include <wtf/RetainPtr.h>
32#include <wtf/WeakPtr.h>
33
34OBJC_CLASS SecKeyProxy;
35
36namespace WebCore {
37class Credential;
38}
39
40namespace WebKit {
41
42class SecKeyProxyStore : public RefCounted<SecKeyProxyStore>, public CanMakeWeakPtr<SecKeyProxyStore> {
43public:
44 static Ref<SecKeyProxyStore> create() { return adoptRef(* new SecKeyProxyStore()); }
45
46 bool initialize(const WebCore::Credential&);
47 bool isInitialized() const { return !!m_secKeyProxy; }
48
49 auto* get() const { return m_secKeyProxy.get(); }
50
51private:
52 SecKeyProxyStore() = default;
53
54 RetainPtr<SecKeyProxy> m_secKeyProxy;
55};
56
57} // namespace WebKit
58
59#endif // HAVE(SEC_KEY_PROXY)
