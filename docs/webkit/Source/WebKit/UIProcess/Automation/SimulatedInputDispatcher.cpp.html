40SimulatedInputSourceState SimulatedInputSourceState::emptyStateForSourceType(SimulatedInputSourceType type)
41{
42 SimulatedInputSourceState result { };
43 switch (type) {
44 case SimulatedInputSourceType::Null:
45 case SimulatedInputSourceType::Keyboard:
46 break;
47 case SimulatedInputSourceType::Wheel:
48 result.scrollDelta = WebCore::IntSize();
49 FALLTHROUGH;
50 case SimulatedInputSourceType::Mouse:
51 case SimulatedInputSourceType::Touch:
52 case SimulatedInputSourceType::Pen:
53 result.location = WebCore::IntPoint();
54 }
55
56 return result;
57}

65Seconds SimulatedInputKeyFrame::maximumDuration() const
66{
67 // The "compute the tick duration" algorithm (ยง17.4 Dispatching Actions).
68 Seconds result;
69 for (auto& entry : states)
70 result = std::max(result, entry.second.duration.value_or(Seconds(0)));
71
72 return result;
73}

75SimulatedInputKeyFrame SimulatedInputKeyFrame::keyFrameFromStateOfInputSources(const HashMap<String, Ref<SimulatedInputSource>>& inputSources)
76{
77 // The client of this class is required to intern SimulatedInputSource instances if the last state
78 // from the previous command should be used as the inital state for the next command. This is the
79 // case for Perform Actions and Release Actions, but not Element Click or Element Send Keys.
80 Vector<SimulatedInputKeyFrame::StateEntry> entries;
81 entries.reserveCapacity(inputSources.size());
82
83 for (const auto& inputSource : inputSources.values())
84 entries.uncheckedAppend(std::pair<SimulatedInputSource&, SimulatedInputSourceState> { inputSource.get(), inputSource->state });
85
86 return SimulatedInputKeyFrame(WTFMove(entries));
87}
88
89SimulatedInputKeyFrame SimulatedInputKeyFrame::keyFrameToResetInputSources(const HashMap<String, Ref<SimulatedInputSource>>& inputSources)
90{
91 Vector<SimulatedInputKeyFrame::StateEntry> entries;
92 entries.reserveCapacity(inputSources.size());
93
94 for (const auto& inputSource : inputSources.values())
95 entries.uncheckedAppend(std::pair<SimulatedInputSource&, SimulatedInputSourceState> { inputSource.get(), SimulatedInputSourceState::emptyStateForSourceType(inputSource->type) });
96
97 return SimulatedInputKeyFrame(WTFMove(entries));
98}

118void SimulatedInputDispatcher::keyFrameTransitionDurationTimerFired()
119{
120 ASSERT(m_keyFrameTransitionCompletionHandler);
121
122 m_keyFrameTransitionDurationTimer.stop();
123
124 LOG(Automation, "SimulatedInputDispatcher[%p]: timer finished for transition between keyframes: %d --> %d", this, m_keyframeIndex - 1, m_keyframeIndex);
125
126 if (isKeyFrameTransitionComplete()) {
127 auto finish = std::exchange(m_keyFrameTransitionCompletionHandler, nullptr);
128 finish(std::nullopt);
129 }

132bool SimulatedInputDispatcher::isKeyFrameTransitionComplete() const
133{
134 ASSERT(m_keyframeIndex < m_keyframes.size());
135
136 if (m_inputSourceStateIndex < m_keyframes[m_keyframeIndex].states.size())
137 return false;
138
139 if (m_keyFrameTransitionDurationTimer.isActive())
140 return false;
141
142 return true;
143}

145void SimulatedInputDispatcher::transitionToNextKeyFrame()
146{
147 ++m_keyframeIndex;
148 if (m_keyframeIndex == m_keyframes.size()) {
149 finishDispatching(std::nullopt);
150 return;
151 }
152
153 transitionBetweenKeyFrames(m_keyframes[m_keyframeIndex - 1], m_keyframes[m_keyframeIndex], [this, protectedThis = makeRef(*this)](std::optional<AutomationCommandError> error) {
154 if (error) {
155 finishDispatching(error);
156 return;
157 }
158
159 transitionToNextKeyFrame();
160 });
161}

163void SimulatedInputDispatcher::transitionToNextInputSourceState()
164{
165 if (isKeyFrameTransitionComplete()) {
166 auto finish = std::exchange(m_keyFrameTransitionCompletionHandler, nullptr);
167 finish(std::nullopt);
168 return;
169 }
170
171 // In this case, transitions are done but we need to wait for the tick timer.
172 if (m_inputSourceStateIndex == m_keyframes[m_keyframeIndex].states.size())
173 return;
174
175 auto& nextKeyFrame = m_keyframes[m_keyframeIndex];
176 auto& postStateEntry = nextKeyFrame.states[m_inputSourceStateIndex];
177 SimulatedInputSource& inputSource = postStateEntry.first;
178
179 transitionInputSourceToState(inputSource, postStateEntry.second, [this, protectedThis = makeRef(*this)](std::optional<AutomationCommandError> error) {
180 if (error) {
181 auto finish = std::exchange(m_keyFrameTransitionCompletionHandler, nullptr);
182 finish(error);
183 return;
184 }
185
186 // Perform state transitions in the order specified by the currentKeyFrame.
187 ++m_inputSourceStateIndex;
188
189 transitionToNextInputSourceState();
190 });
191}
192
193void SimulatedInputDispatcher::transitionBetweenKeyFrames(const SimulatedInputKeyFrame& a, const SimulatedInputKeyFrame& b, AutomationCompletionHandler&& completionHandler)
194{
195 m_inputSourceStateIndex = 0;
196
197 // The "dispatch tick actions" algorithm (ยง17.4 Dispatching Actions).
198 m_keyFrameTransitionCompletionHandler = WTFMove(completionHandler);
199 m_keyFrameTransitionDurationTimer.startOneShot(b.maximumDuration());
200
201 LOG(Automation, "SimulatedInputDispatcher[%p]: started transition between keyframes: %d --> %d", this, m_keyframeIndex - 1, m_keyframeIndex);
202 LOG(Automation, "SimulatedInputDispatcher[%p]: timer started to ensure minimum duration of %.2f seconds for transition %d --> %d", this, b.maximumDuration().value(), m_keyframeIndex - 1, m_keyframeIndex);
203
204 transitionToNextInputSourceState();
205}

207void SimulatedInputDispatcher::resolveLocation(const WebCore::IntPoint& currentLocation, std::optional<WebCore::IntPoint> location, MouseMoveOrigin origin, std::optional<String> nodeHandle, Function<void (std::optional<WebCore::IntPoint>, std::optional<AutomationCommandError>)>&& completionHandler)
208{
209 if (!location) {
210 completionHandler(currentLocation, std::nullopt);
211 return;
212 }
213
214 switch (origin) {
215 case MouseMoveOrigin::Viewport:
216 completionHandler(location.value(), std::nullopt);
217 break;
218 case MouseMoveOrigin::Pointer: {
219 WebCore::IntPoint destination(currentLocation);
220 destination.moveBy(location.value());
221 completionHandler(destination, std::nullopt);
222 break;
223 }
224 case MouseMoveOrigin::Element: {
225 m_client.viewportInViewCenterPointOfElement(m_page, m_frameID, nodeHandle.value(), [destination = location.value(), completionHandler = WTFMove(completionHandler)](std::optional<WebCore::IntPoint> inViewCenterPoint, std::optional<AutomationCommandError> error) mutable {
226 if (error) {
227 completionHandler(std::nullopt, error);
228 return;
229 }
230
231 if (!inViewCenterPoint) {
232 completionHandler(std::nullopt, AUTOMATION_COMMAND_ERROR_WITH_NAME(ElementNotInteractable));
233 return;
234 }
235
236 destination.moveBy(inViewCenterPoint.value());
237 completionHandler(destination, std::nullopt);
238 });
239 break;
240 }
241 }

244void SimulatedInputDispatcher::transitionInputSourceToState(SimulatedInputSource& inputSource, SimulatedInputSourceState& newState, AutomationCompletionHandler&& completionHandler)
245{
246 // Make cases and conditionals more readable by aliasing pre/post states as 'a' and 'b'.
247 SimulatedInputSourceState& a = inputSource.state;
248 SimulatedInputSourceState& b = newState;
249
250 LOG(Automation, "SimulatedInputDispatcher[%p]: transition started between input source states: [%d.%d] --> %d.%d", this, m_keyframeIndex - 1, m_inputSourceStateIndex, m_keyframeIndex, m_inputSourceStateIndex);
251
252 AutomationCompletionHandler eventDispatchFinished = [this, &inputSource, &newState, completionHandler = WTFMove(completionHandler)](std::optional<AutomationCommandError> error) mutable {
253 if (error) {
254 completionHandler(error);
255 return;
256 }
257
258#if !LOG_DISABLED
259 LOG(Automation, "SimulatedInputDispatcher[%p]: transition finished between input source states: %d.%d --> [%d.%d]", this, m_keyframeIndex - 1, m_inputSourceStateIndex, m_keyframeIndex, m_inputSourceStateIndex);
260#else
261 UNUSED_PARAM(this);
262#endif
263
264 inputSource.state = newState;
265 completionHandler(std::nullopt);
266 };
267
268 switch (inputSource.type) {
269 case SimulatedInputSourceType::Null:
270 // The maximum duration is handled at the keyframe level by m_keyFrameTransitionDurationTimer.
271 eventDispatchFinished(std::nullopt);
272 break;
273 case SimulatedInputSourceType::Mouse:
274 case SimulatedInputSourceType::Pen: {
275#if !ENABLE(WEBDRIVER_MOUSE_INTERACTIONS)
276 RELEASE_ASSERT_NOT_REACHED();
277#else
278 resolveLocation(a.location.value_or(WebCore::IntPoint()), b.location, b.origin.value_or(MouseMoveOrigin::Viewport), b.nodeHandle, [this, &a, &b, inputSource = inputSource.type, eventDispatchFinished = WTFMove(eventDispatchFinished)](std::optional<WebCore::IntPoint> location, std::optional<AutomationCommandError> error) mutable {
279 if (error) {
280 eventDispatchFinished(error);
281 return;
282 }
283
284 if (!location) {
285 eventDispatchFinished(AUTOMATION_COMMAND_ERROR_WITH_NAME(ElementNotInteractable));
286 return;
287 }
288
289 const String& pointerType = inputSource == SimulatedInputSourceType::Mouse ? WebCore::mousePointerEventType() : WebCore::penPointerEventType();
290
291 b.location = location;
292 // The "dispatch a pointer{Down,Up,Move} action" algorithms (ยง17.4 Dispatching Actions).
293 if (!a.pressedMouseButton && b.pressedMouseButton) {
294#if !LOG_DISABLED
295 String mouseButtonName = Inspector::Protocol::AutomationHelpers::getEnumConstantValue(b.pressedMouseButton.value());
296 LOG(Automation, "SimulatedInputDispatcher[%p]: simulating MouseDown[button=%s] @ (%d, %d) for transition to %d.%d", this, mouseButtonName.utf8().data(), b.location.value().x(), b.location.value().y(), m_keyframeIndex, m_inputSourceStateIndex);
297#endif
298 m_client.simulateMouseInteraction(m_page, MouseInteraction::Down, b.pressedMouseButton.value(), b.location.value(), pointerType, WTFMove(eventDispatchFinished));
299 } else if (a.pressedMouseButton && !b.pressedMouseButton) {
300#if !LOG_DISABLED
301 String mouseButtonName = Inspector::Protocol::AutomationHelpers::getEnumConstantValue(a.pressedMouseButton.value());
302 LOG(Automation, "SimulatedInputDispatcher[%p]: simulating MouseUp[button=%s] @ (%d, %d) for transition to %d.%d", this, mouseButtonName.utf8().data(), b.location.value().x(), b.location.value().y(), m_keyframeIndex, m_inputSourceStateIndex);
303#endif
304 m_client.simulateMouseInteraction(m_page, MouseInteraction::Up, a.pressedMouseButton.value(), b.location.value(), pointerType, WTFMove(eventDispatchFinished));
305 } else if (a.location != b.location) {
306 LOG(Automation, "SimulatedInputDispatcher[%p]: simulating MouseMove from (%d, %d) to (%d, %d) for transition to %d.%d", this, a.location.value().x(), a.location.value().y(), b.location.value().x(), b.location.value().y(), m_keyframeIndex, m_inputSourceStateIndex);
307 // FIXME: This does not interpolate mousemoves per the "perform a pointer move" algorithm (ยง17.4 Dispatching Actions).
308 m_client.simulateMouseInteraction(m_page, MouseInteraction::Move, b.pressedMouseButton.value_or(MouseButton::None), b.location.value(), pointerType, WTFMove(eventDispatchFinished));
309 } else
310 eventDispatchFinished(std::nullopt);
311 });
312#endif // ENABLE(WEBDRIVER_MOUSE_INTERACTIONS)
313 break;
314 }
315 case SimulatedInputSourceType::Touch: {
316#if !ENABLE(WEBDRIVER_TOUCH_INTERACTIONS)
317 RELEASE_ASSERT_NOT_REACHED();
318#else
319 resolveLocation(a.location.value_or(WebCore::IntPoint()), b.location, b.origin.value_or(MouseMoveOrigin::Viewport), b.nodeHandle, [this, &a, &b, eventDispatchFinished = WTFMove(eventDispatchFinished)](std::optional<WebCore::IntPoint> location, std::optional<AutomationCommandError> error) mutable {
320 if (error) {
321 eventDispatchFinished(error);
322 return;
323 }
324
325 if (!location) {
326 eventDispatchFinished(AUTOMATION_COMMAND_ERROR_WITH_NAME(ElementNotInteractable));
327 return;
328 }
329
330 b.location = location;
331 // The "dispatch a pointer{Down,Up,Move} action" algorithms (ยง17.4 Dispatching Actions).
332 if (!a.pressedMouseButton && b.pressedMouseButton) {
333 LOG(Automation, "SimulatedInputDispatcher[%p]: simulating TouchDown @ (%d, %d) for transition to %d.%d", this, b.location.value().x(), b.location.value().y(), m_keyframeIndex, m_inputSourceStateIndex);
334 m_client.simulateTouchInteraction(m_page, TouchInteraction::TouchDown, b.location.value(), std::nullopt, WTFMove(eventDispatchFinished));
335 } else if (a.pressedMouseButton && !b.pressedMouseButton) {
336 LOG(Automation, "SimulatedInputDispatcher[%p]: simulating LiftUp @ (%d, %d) for transition to %d.%d", this, b.location.value().x(), b.location.value().y(), m_keyframeIndex, m_inputSourceStateIndex);
337 m_client.simulateTouchInteraction(m_page, TouchInteraction::LiftUp, b.location.value(), std::nullopt, WTFMove(eventDispatchFinished));
338 } else if (a.location != b.location) {
339 LOG(Automation, "SimulatedInputDispatcher[%p]: simulating MoveTo from (%d, %d) to (%d, %d) for transition to %d.%d", this, a.location.value().x(), a.location.value().y(), b.location.value().x(), b.location.value().y(), m_keyframeIndex, m_inputSourceStateIndex);
340 m_client.simulateTouchInteraction(m_page, TouchInteraction::MoveTo, b.location.value(), a.duration.value_or(0_s), WTFMove(eventDispatchFinished));
341 } else
342 eventDispatchFinished(std::nullopt);
343 });
344#endif // !ENABLE(WEBDRIVER_TOUCH_INTERACTIONS)
345 break;
346 }
347 case SimulatedInputSourceType::Keyboard: {
348#if !ENABLE(WEBDRIVER_KEYBOARD_INTERACTIONS)
349 RELEASE_ASSERT_NOT_REACHED();
350#else
351 auto comparePressedCharKeys = [](const auto& a, const auto& b) {
352 if (a.size() != b.size())
353 return false;
354 for (const auto& charKey : a) {
355 if (!b.contains(charKey))
356 return false;
357 }
358 return true;
359 };
360
361 // The "dispatch a key{Down,Up} action" algorithms (ยง17.4 Dispatching Actions).
362 if (!comparePressedCharKeys(a.pressedCharKeys, b.pressedCharKeys)) {
363 bool simulatedAnInteraction = false;
364 for (auto charKey : b.pressedCharKeys) {
365 if (!a.pressedCharKeys.contains(charKey)) {
366 ASSERT_WITH_MESSAGE(!simulatedAnInteraction, "Only one CharKey may differ at a time between two input source states.");
367 if (simulatedAnInteraction)
368 continue;
369 simulatedAnInteraction = true;
370
371 LOG(Automation, "SimulatedInputDispatcher[%p]: simulating KeyPress[key=%c] for transition to %d.%d", this, charKey, m_keyframeIndex, m_inputSourceStateIndex);
372 m_client.simulateKeyboardInteraction(m_page, KeyboardInteraction::KeyPress, charKey, WTFMove(eventDispatchFinished));
373 }
374 }
375
376 for (auto charKey : a.pressedCharKeys) {
377 if (!b.pressedCharKeys.contains(charKey)) {
378 ASSERT_WITH_MESSAGE(!simulatedAnInteraction, "Only one CharKey may differ at a time between two input source states.");
379 if (simulatedAnInteraction)
380 continue;
381 simulatedAnInteraction = true;
382
383 LOG(Automation, "SimulatedInputDispatcher[%p]: simulating KeyRelease[key=%c] for transition to %d.%d", this, charKey, m_keyframeIndex, m_inputSourceStateIndex);
384 m_client.simulateKeyboardInteraction(m_page, KeyboardInteraction::KeyRelease, charKey, WTFMove(eventDispatchFinished));
385 }
386 }
387 } else if (a.pressedVirtualKeys != b.pressedVirtualKeys) {
388 bool simulatedAnInteraction = false;
389 for (const auto& iter : b.pressedVirtualKeys) {
390 if (!a.pressedVirtualKeys.contains(iter.key)) {
391 ASSERT_WITH_MESSAGE(!simulatedAnInteraction, "Only one VirtualKey may differ at a time between two input source states.");
392 if (simulatedAnInteraction)
393 continue;
394 simulatedAnInteraction = true;
395#if !LOG_DISABLED
396 String virtualKeyName = Inspector::Protocol::AutomationHelpers::getEnumConstantValue(iter.value);
397 LOG(Automation, "SimulatedInputDispatcher[%p]: simulating KeyPress[key=%s] for transition to %d.%d", this, virtualKeyName.utf8().data(), m_keyframeIndex, m_inputSourceStateIndex);
398#endif
399 m_client.simulateKeyboardInteraction(m_page, KeyboardInteraction::KeyPress, iter.value, WTFMove(eventDispatchFinished));
400 }
401 }
402
403 for (const auto& iter : a.pressedVirtualKeys) {
404 if (!b.pressedVirtualKeys.contains(iter.key)) {
405 ASSERT_WITH_MESSAGE(!simulatedAnInteraction, "Only one VirtualKey may differ at a time between two input source states.");
406 if (simulatedAnInteraction)
407 continue;
408 simulatedAnInteraction = true;
409#if !LOG_DISABLED
410 String virtualKeyName = Inspector::Protocol::AutomationHelpers::getEnumConstantValue(iter.value);
411 LOG(Automation, "SimulatedInputDispatcher[%p]: simulating KeyRelease[key=%s] for transition to %d.%d", this, virtualKeyName.utf8().data(), m_keyframeIndex, m_inputSourceStateIndex);
412#endif
413 m_client.simulateKeyboardInteraction(m_page, KeyboardInteraction::KeyRelease, iter.value, WTFMove(eventDispatchFinished));
414 }
415 }
416 } else
417 eventDispatchFinished(std::nullopt);
418#endif // !ENABLE(WEBDRIVER_KEYBOARD_INTERACTIONS)
419 break;
420 }
421 case SimulatedInputSourceType::Wheel:
422#if !ENABLE(WEBDRIVER_WHEEL_INTERACTIONS)
423 RELEASE_ASSERT_NOT_REACHED();
424#else
425 resolveLocation(a.location.value_or(WebCore::IntPoint()), b.location, b.origin.value_or(MouseMoveOrigin::Viewport), b.nodeHandle, [this, &a, &b, eventDispatchFinished = WTFMove(eventDispatchFinished)](std::optional<WebCore::IntPoint> location, std::optional<AutomationCommandError> error) mutable {
426 if (error) {
427 eventDispatchFinished(error);
428 return;
429 }
430
431 if (!location) {
432 eventDispatchFinished(AUTOMATION_COMMAND_ERROR_WITH_NAME(ElementNotInteractable));
433 return;
434 }
435
436 b.location = location;
437
438 if (!a.scrollDelta->isZero())
439 b.scrollDelta->contract(a.scrollDelta->width(), a.scrollDelta->height());
440
441 if (!b.scrollDelta->isZero()) {
442 LOG(Automation, "SimulatedInputDispatcher[%p]: simulating Wheel from (%d, %d) to (%d, %d) for transition to %d.%d", this, a.scrollDelta->width(), a.scrollDelta->height(), b.scrollDelta->width(), b.scrollDelta->height(), m_keyframeIndex, m_inputSourceStateIndex);
443 // FIXME: This does not interpolate mouse scrolls per the "perform a scroll" algorithm (ยง15.4.4 Wheel actions).
444 m_client.simulateWheelInteraction(m_page, b.location.value(), b.scrollDelta.value(), WTFMove(eventDispatchFinished));
445 } else
446 eventDispatchFinished(std::nullopt);
447 });
448#endif // !ENABLE(WEBDRIVER_WHEEL_INTERACTIONS)
449 break;
450 }

453void SimulatedInputDispatcher::run(std::optional<WebCore::FrameIdentifier> frameID, Vector<SimulatedInputKeyFrame>&& keyFrames, const HashMap<String, Ref<SimulatedInputSource>>& inputSources, AutomationCompletionHandler&& completionHandler)
454{
455 ASSERT(!isActive());
456 if (isActive()) {
457 completionHandler(AUTOMATION_COMMAND_ERROR_WITH_NAME(InternalError));
458 return;
459 }
460
461 m_frameID = frameID;
462 m_runCompletionHandler = WTFMove(completionHandler);
463
464 // The "dispatch actions" algorithm (ยง17.4 Dispatching Actions).
465 m_keyframes.reserveCapacity(keyFrames.size() + 1);
466 m_keyframes.append(SimulatedInputKeyFrame::keyFrameFromStateOfInputSources(inputSources));
467 m_keyframes.appendVector(WTFMove(keyFrames));
468
469 LOG(Automation, "SimulatedInputDispatcher[%p]: starting input simulation using %zu keyframes", this, m_keyframes.size());
470
471 transitionToNextKeyFrame();
472}

474void SimulatedInputDispatcher::cancel()
475{
476 // If we were waiting for m_client to finish an interaction and the interaction had an error,
477 // then the rest of the async chain will have been torn down. If we are just waiting on a
478 // dispatch timer, then this will cancel the timer and clear
479
480 if (isActive())
481 finishDispatching(AUTOMATION_COMMAND_ERROR_WITH_NAME(InternalError));
482}
483
484void SimulatedInputDispatcher::finishDispatching(std::optional<AutomationCommandError> error)
485{
486 m_keyFrameTransitionDurationTimer.stop();
487
488 LOG(Automation, "SimulatedInputDispatcher[%p]: finished all input simulation at [%u.%u]", this, m_keyframeIndex, m_inputSourceStateIndex);
489
490 auto finish = std::exchange(m_runCompletionHandler, nullptr);
491 m_frameID = std::nullopt;
492 m_keyframes.clear();
493 m_keyframeIndex = 0;
494 m_inputSourceStateIndex = 0;
495
496 finish(error);
497}
