51class AutomationCommandError;

54class WebPageProxy;

56using KeyboardInteraction = Inspector::Protocol::Automation::KeyboardInteractionType;
57using VirtualKey = Inspector::Protocol::Automation::VirtualKey;
58using VirtualKeyMap = HashMap<VirtualKey, VirtualKey, WTF::IntHash<VirtualKey>, WTF::StrongEnumHashTraits<VirtualKey>>;
59using CharKey = UChar32;
60using CharKeySet = ListHashSet<CharKey>;

62using MouseInteraction = Inspector::Protocol::Automation::MouseInteraction;
63using MouseMoveOrigin = Inspector::Protocol::Automation::MouseMoveOrigin;

65enum class SimulatedInputSourceType {
66 Null, // Used to induce a minimum duration.
67 Keyboard,
68 Mouse,
69 Touch,
70 Wheel,
71 Pen,
72};
73
74enum class TouchInteraction {
75 TouchDown,
76 MoveTo,
77 LiftUp,
78};

80struct SimulatedInputSourceState {
81 CharKeySet pressedCharKeys;
82 VirtualKeyMap pressedVirtualKeys;
83 std::optional<MouseButton> pressedMouseButton;
84 std::optional<MouseMoveOrigin> origin;
85 std::optional<String> nodeHandle;
86 std::optional<WebCore::IntPoint> location;
87 std::optional<WebCore::IntSize> scrollDelta;
88 std::optional<Seconds> duration;
89
90 static SimulatedInputSourceState emptyStateForSourceType(SimulatedInputSourceType);
91};
92
93struct SimulatedInputSource : public RefCounted<SimulatedInputSource> {
94public:
95 SimulatedInputSourceType type;
96
97 // The last state associated with this input source.
98 SimulatedInputSourceState state;
99
100 static Ref<SimulatedInputSource> create(SimulatedInputSourceType type)
101 {
102 return adoptRef(*new SimulatedInputSource(type));
103 }
104
105private:
106 SimulatedInputSource(SimulatedInputSourceType type)
107 : type(type)
108 , state(SimulatedInputSourceState::emptyStateForSourceType(type))
109 { }
110};
111
112struct SimulatedInputKeyFrame {
113public:
114 using StateEntry = std::pair<SimulatedInputSource&, SimulatedInputSourceState>;
115
116 explicit SimulatedInputKeyFrame(Vector<StateEntry>&&);
117 Seconds maximumDuration() const;
118
119 static SimulatedInputKeyFrame keyFrameFromStateOfInputSources(const HashMap<String, Ref<SimulatedInputSource>>&);
120 static SimulatedInputKeyFrame keyFrameToResetInputSources(const HashMap<String, Ref<SimulatedInputSource>>&);
121
122 Vector<StateEntry> states;
123};

125class SimulatedInputDispatcher : public RefCounted<SimulatedInputDispatcher> {
126 WTF_MAKE_NONCOPYABLE(SimulatedInputDispatcher);
127public:
128 class Client {
129 public:
130 virtual ~Client() { }
131#if ENABLE(WEBDRIVER_MOUSE_INTERACTIONS)
132 virtual void simulateMouseInteraction(WebPageProxy&, MouseInteraction, MouseButton, const WebCore::IntPoint& locationInView, const String& pointerType, AutomationCompletionHandler&&) = 0;
133#endif
134#if ENABLE(WEBDRIVER_TOUCH_INTERACTIONS)
135 virtual void simulateTouchInteraction(WebPageProxy&, TouchInteraction, const WebCore::IntPoint& locationInView, std::optional<Seconds> duration, AutomationCompletionHandler&&) = 0;
136#endif
137#if ENABLE(WEBDRIVER_KEYBOARD_INTERACTIONS)
138 virtual void simulateKeyboardInteraction(WebPageProxy&, KeyboardInteraction, WTF::Variant<VirtualKey, CharKey>&&, AutomationCompletionHandler&&) = 0;
139#endif
140#if ENABLE(WEBDRIVER_WHEEL_INTERACTIONS)
141 virtual void simulateWheelInteraction(WebPageProxy&, const WebCore::IntPoint& locationInView, const WebCore::IntSize& delta, AutomationCompletionHandler&&) = 0;
142#endif
143 virtual void viewportInViewCenterPointOfElement(WebPageProxy&, std::optional<WebCore::FrameIdentifier>, const String& nodeHandle, Function<void (std::optional<WebCore::IntPoint>, std::optional<AutomationCommandError>)>&&) = 0;
144 };
