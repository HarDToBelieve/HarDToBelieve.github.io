85class AutomationCommandError {
86public:
87 Inspector::Protocol::Automation::ErrorMessage type;

89
90 AutomationCommandError(Inspector::Protocol::Automation::ErrorMessage type)
91 : type(type) { }
92
93 AutomationCommandError(Inspector::Protocol::Automation::ErrorMessage type, const String& message)
94 : type(type)
