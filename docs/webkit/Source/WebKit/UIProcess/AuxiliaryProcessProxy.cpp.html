49AuxiliaryProcessProxy::~AuxiliaryProcessProxy()
50{
51 if (m_connection)
52 m_connection->invalidate();

57 }
58
59 replyToPendingMessages();

62void AuxiliaryProcessProxy::getLaunchOptions(ProcessLauncher::LaunchOptions& launchOptions)
63{
64 launchOptions.processIdentifier = m_processIdentifier;
65
66 if (const char* userDirectorySuffix = getenv("DIRHELPER_USER_DIR_SUFFIX"))
67 launchOptions.extraInitializationData.add("user-directory-suffix"_s, userDirectorySuffix);
68
69 if (m_alwaysRunsAtBackgroundPriority)
70 launchOptions.extraInitializationData.add("always-runs-at-background-priority"_s, "true");
71
72#if ENABLE(DEVELOPER_MODE) && (PLATFORM(GTK) || PLATFORM(WPE))
73 const char* varname;
74 switch (launchOptions.processType) {
75 case ProcessLauncher::ProcessType::Web:
76 varname = "WEB_PROCESS_CMD_PREFIX";
77 break;
78#if ENABLE(NETSCAPE_PLUGIN_API)
79 case ProcessLauncher::ProcessType::Plugin:
80 varname = "PLUGIN_PROCESS_CMD_PREFIX";
81 break;
82#endif
83 case ProcessLauncher::ProcessType::Network:
84 varname = "NETWORK_PROCESS_CMD_PREFIX";
85 break;
86#if ENABLE(GPU_PROCESS)
87 case ProcessLauncher::ProcessType::GPU:
88 varname = "GPU_PROCESS_CMD_PREFIX";
89 break;
90#endif
91#if ENABLE(WEB_AUTHN)
92 case ProcessLauncher::ProcessType::WebAuthn:
93 varname = "WEBAUTHN_PROCESS_CMD_PREFIX";
94 break;
95#endif
96#if ENABLE(BUBBLEWRAP_SANDBOX)
97 case ProcessLauncher::ProcessType::DBusProxy:
98 ASSERT_NOT_REACHED();
99 break;
100#endif
101 }
102 const char* processCmdPrefix = getenv(varname);
103 if (processCmdPrefix && *processCmdPrefix)
104 launchOptions.processCmdPrefix = String::fromUTF8(processCmdPrefix);
105#endif // ENABLE(DEVELOPER_MODE) && (PLATFORM(GTK) || PLATFORM(WPE))
106
107 platformGetLaunchOptions(launchOptions);
108}

110void AuxiliaryProcessProxy::connect()
111{
112 ASSERT(!m_processLauncher);
113 ProcessLauncher::LaunchOptions launchOptions;
114 getLaunchOptions(launchOptions);
115 m_processLauncher = ProcessLauncher::create(this, WTFMove(launchOptions));
116}

118void AuxiliaryProcessProxy::terminate()
119{
120 RELEASE_LOG(Process, "AuxiliaryProcessProxy::terminate: PID=%d", processIdentifier());
121
122#if PLATFORM(COCOA)
123 if (m_connection && m_connection->kill())
124 return;
125#endif
126
127 // FIXME: We should really merge process launching into IPC connection creation and get rid of the process launcher.
128 if (m_processLauncher)
129 m_processLauncher->terminateProcess();
130}

132AuxiliaryProcessProxy::State AuxiliaryProcessProxy::state() const
133{
134 if (m_processLauncher && m_processLauncher->isLaunching())
135 return AuxiliaryProcessProxy::State::Launching;
136
137 if (!m_connection)
138 return AuxiliaryProcessProxy::State::Terminated;
139
140 return AuxiliaryProcessProxy::State::Running;
141}

143bool AuxiliaryProcessProxy::wasTerminated() const
144{
145 switch (state()) {
146 case AuxiliaryProcessProxy::State::Launching:
147 return false;
148 case AuxiliaryProcessProxy::State::Terminated:
149 return true;
150 case AuxiliaryProcessProxy::State::Running:
151 break;
152 }
153
154 auto pid = processIdentifier();
155 if (!pid)
156 return true;
157
158#if PLATFORM(COCOA)
159 // Use kill() with a signal of 0 to make sure there is indeed still a process with the given PID.
160 // This is needed because it sometimes takes a little bit of time for us to get notified that a process
161 // was terminated.
162 return kill(pid, 0) && errno == ESRCH;
163#else
164 return false;
165#endif
166}

168bool AuxiliaryProcessProxy::sendMessage(UniqueRef<IPC::Encoder>&& encoder, OptionSet<IPC::SendOption> sendOptions, std::optional<std::pair<CompletionHandler<void(IPC::Decoder*)>, uint64_t>>&& asyncReplyInfo, ShouldStartProcessThrottlerActivity shouldStartProcessThrottlerActivity)
169{
170 // FIXME: We should turn this into a RELEASE_ASSERT().
171 ASSERT(isMainRunLoop());
172 if (!isMainRunLoop()) {
173 callOnMainRunLoop([protectedThis = makeRef(*this), encoder = WTFMove(encoder), sendOptions, asyncReplyInfo = WTFMove(asyncReplyInfo), shouldStartProcessThrottlerActivity]() mutable {
174 protectedThis->sendMessage(WTFMove(encoder), sendOptions, WTFMove(asyncReplyInfo), shouldStartProcessThrottlerActivity);
175 });
176 return true;
177 }
178
179 if (asyncReplyInfo && canSendMessage() && shouldStartProcessThrottlerActivity == ShouldStartProcessThrottlerActivity::Yes) {
180 auto completionHandler = std::exchange(asyncReplyInfo->first, nullptr);
181 asyncReplyInfo->first = [activity = throttler().backgroundActivity(ASCIILiteral::null()), completionHandler = WTFMove(completionHandler)](IPC::Decoder* decoder) mutable {
182 completionHandler(decoder);
183 };
184 }
185
186 switch (state()) {
187 case State::Launching:
188 // If we're waiting for the child process to launch, we need to stash away the messages so we can send them once we have a connection.
189 m_pendingMessages.append({ WTFMove(encoder), sendOptions, WTFMove(asyncReplyInfo) });
190 return true;
191
192 case State::Running:
193 if (asyncReplyInfo)
194 IPC::addAsyncReplyHandler(*connection(), asyncReplyInfo->second, std::exchange(asyncReplyInfo->first, nullptr));
195 if (connection()->sendMessage(WTFMove(encoder), sendOptions))
196 return true;
197 break;
198
199 case State::Terminated:
200 break;
201 }
202
203 if (asyncReplyInfo && asyncReplyInfo->first) {
204 RunLoop::current().dispatch([completionHandler = WTFMove(asyncReplyInfo->first)]() mutable {
205 completionHandler(nullptr);
206 });
207 }
208
209 return false;
210}

242void AuxiliaryProcessProxy::didFinishLaunching(ProcessLauncher*, IPC::Connection::Identifier connectionIdentifier)
243{
244 ASSERT(!m_connection);
245 ASSERT(isMainRunLoop());
246
247 if (!IPC::Connection::identifierIsValid(connectionIdentifier))
248 return;
249
250 m_connection = IPC::Connection::createServerConnection(connectionIdentifier, *this);
251
252 connectionWillOpen(*m_connection);
253 m_connection->open();
254
255 for (auto&& pendingMessage : std::exchange(m_pendingMessages, { })) {
256 if (!shouldSendPendingMessage(pendingMessage))
257 continue;
258 if (pendingMessage.asyncReplyInfo)
259 IPC::addAsyncReplyHandler(*connection(), pendingMessage.asyncReplyInfo->second, WTFMove(pendingMessage.asyncReplyInfo->first));
260 m_connection->sendMessage(WTFMove(pendingMessage.encoder), pendingMessage.sendOptions);
261 }
262
263 if (m_shouldStartResponsivenessTimerWhenLaunched) {
264 auto useLazyStop = *std::exchange(m_shouldStartResponsivenessTimerWhenLaunched, std::nullopt);
265 startResponsivenessTimer(useLazyStop);
266 }
267}
268
269void AuxiliaryProcessProxy::replyToPendingMessages()
270{
271 ASSERT(isMainRunLoop());
272 for (auto& pendingMessage : std::exchange(m_pendingMessages, { })) {
273 if (pendingMessage.asyncReplyInfo)
274 pendingMessage.asyncReplyInfo->first(nullptr);
275 }
276}

278void AuxiliaryProcessProxy::shutDownProcess()
279{
280 switch (state()) {
281 case State::Launching:
282 m_processLauncher->invalidate();
283 m_processLauncher = nullptr;
284 break;
285 case State::Running:
286#if PLATFORM(IOS_FAMILY)
287 // On iOS deploy a watchdog in the UI process, since the child process may be suspended.
288 // If 30s is insufficient for any outstanding activity to complete cleanly, then it will be killed.
289 ASSERT(m_connection);
290 m_connection->terminateSoon(30_s);
291#endif
292 break;
293 case State::Terminated:
294 return;
295 }
296
297 if (!m_connection)
298 return;
299
300 processWillShutDown(*m_connection);
301
302 if (canSendMessage())
303 send(Messages::AuxiliaryProcess::ShutDown(), 0);
304
305 m_connection->invalidate();
306 m_connection = nullptr;
307 m_responsivenessTimer.invalidate();
308}

331bool AuxiliaryProcessProxy::platformIsBeingDebugged() const
332{
333#if PLATFORM(COCOA)
334 // If the UI process is sandboxed and lacks 'process-info-pidinfo', it cannot find out whether other processes are being debugged.
335 if (currentProcessIsSandboxed() && !!sandbox_check(getpid(), "process-info-pidinfo", SANDBOX_CHECK_NO_REPORT))
336 return false;
337
338 struct kinfo_proc info;
339 int mib[] = { CTL_KERN, KERN_PROC, KERN_PROC_PID, processIdentifier() };
340 size_t size = sizeof(info);
341 if (sysctl(mib, WTF_ARRAY_LENGTH(mib), &info, &size, nullptr, 0) == -1)
342 return false;
343
344 return info.kp_proc.p_flag & P_TRACED;
345#else
346 return false;
347#endif
348}

355void AuxiliaryProcessProxy::startResponsivenessTimer(UseLazyStop useLazyStop)
356{
357 if (isLaunching()) {
358 m_shouldStartResponsivenessTimerWhenLaunched = useLazyStop;
359 return;
360 }
361
362 if (useLazyStop == UseLazyStop::Yes)
363 responsivenessTimer().startWithLazyStop();
364 else
365 responsivenessTimer().start();
366}

378void AuxiliaryProcessProxy::checkForResponsiveness(CompletionHandler<void()>&& responsivenessHandler, UseLazyStop useLazyStop)
379{
380 startResponsivenessTimer(useLazyStop);
381 sendWithAsyncReply(Messages::AuxiliaryProcess::MainThreadPing(), [weakThis = makeWeakPtr(*this), responsivenessHandler = WTFMove(responsivenessHandler)]() mutable {
382 // Schedule an asynchronous task because our completion handler may have been called as a result of the AuxiliaryProcessProxy
383 // being in the middle of destruction.
384 RunLoop::main().dispatch([weakThis = WTFMove(weakThis), responsivenessHandler = WTFMove(responsivenessHandler)]() mutable {
385 if (weakThis)
386 weakThis->stopResponsivenessTimer();
387
388 if (responsivenessHandler)
389 responsivenessHandler();
390 });
391 });
392}
