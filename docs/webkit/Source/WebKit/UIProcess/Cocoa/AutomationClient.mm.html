43AutomationClient::AutomationClient(WKProcessPool *processPool, id <_WKAutomationDelegate> delegate)
44 : m_processPool(processPool)
45 , m_delegate(delegate)
46{
47 m_delegateMethods.allowsRemoteAutomation = [delegate respondsToSelector:@selector(_processPoolAllowsRemoteAutomation:)];
48 m_delegateMethods.requestAutomationSession = [delegate respondsToSelector:@selector(_processPool:didRequestAutomationSessionWithIdentifier:configuration:)];
49 m_delegateMethods.requestedDebuggablesToWakeUp = [delegate respondsToSelector:@selector(_processPoolDidRequestInspectorDebuggablesToWakeUp:)];
50 m_delegateMethods.browserNameForAutomation = [delegate respondsToSelector:@selector(_processPoolBrowserNameForAutomation:)];
51 m_delegateMethods.browserVersionForAutomation = [delegate respondsToSelector:@selector(_processPoolBrowserVersionForAutomation:)];
52
53 RemoteInspector::singleton().setClient(this);
54}
55
56AutomationClient::~AutomationClient()
57{
58 RemoteInspector::singleton().setClient(nullptr);
59}

70bool AutomationClient::remoteAutomationAllowed() const
71{
72 if (m_delegateMethods.allowsRemoteAutomation)
73 return [m_delegate.get() _processPoolAllowsRemoteAutomation:m_processPool];
74
75 return false;
76}

78void AutomationClient::requestAutomationSession(const String& sessionIdentifier, const RemoteInspector::Client::SessionCapabilities& sessionCapabilities)
79{
80 auto configuration = adoptNS([[_WKAutomationSessionConfiguration alloc] init]);
81 if (sessionCapabilities.allowInsecureMediaCapture)
82 [configuration setAllowsInsecureMediaCapture:sessionCapabilities.allowInsecureMediaCapture.value()];
83 if (sessionCapabilities.suppressICECandidateFiltering)
84 [configuration setSuppressesICECandidateFiltering:sessionCapabilities.suppressICECandidateFiltering.value()];
85
86 // Force clients to create and register a session asynchronously. Otherwise,
87 // RemoteInspector will try to acquire its lock to register the new session and
88 // deadlock because it's already taken while handling XPC messages.
89 NSString *requestedSessionIdentifier = sessionIdentifier;
90 RunLoop::main().dispatch([this, requestedSessionIdentifier = retainPtr(requestedSessionIdentifier), configuration = WTFMove(configuration)] {
91 if (m_delegateMethods.requestAutomationSession)
92 [m_delegate.get() _processPool:m_processPool didRequestAutomationSessionWithIdentifier:requestedSessionIdentifier.get() configuration:configuration.get()];
93 });
94}

106String AutomationClient::browserName() const
107{
108 if (m_delegateMethods.browserNameForAutomation)
109 return [m_delegate _processPoolBrowserNameForAutomation:m_processPool];
110
111 // Fall back to using the unlocalized app name (i.e., 'Safari').
112 NSBundle *appBundle = [NSBundle mainBundle];
113 NSString *displayName = appBundle.infoDictionary[(__bridge NSString *)_kCFBundleDisplayNameKey];
114 NSString *readableName = appBundle.infoDictionary[(__bridge NSString *)kCFBundleNameKey];
115 return displayName ?: readableName;
116}
117
118String AutomationClient::browserVersion() const
119{
120 if (m_delegateMethods.browserVersionForAutomation)
121 return [m_delegate _processPoolBrowserVersionForAutomation:m_processPool];
122
123 // Fall back to using the app short version (i.e., '11.1.1').
124 NSBundle *appBundle = [NSBundle mainBundle];
125 return appBundle.infoDictionary[(__bridge NSString *)_kCFBundleShortVersionStringKey];
126}
