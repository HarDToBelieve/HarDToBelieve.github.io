27#import "FindClient.h"

29#import "_WKFindDelegate.h"

31namespace WebKit {
32
33FindClient::FindClient(WKWebView *webView)
34 : m_webView(webView)
35{
36}
37
38RetainPtr<id <_WKFindDelegate>> FindClient::delegate()
39{
40 return m_delegate.get();
41}
42
43void FindClient::setDelegate(id <_WKFindDelegate> delegate)
44{
45 m_delegate = delegate;
46
47 m_delegateMethods.webviewDidCountStringMatches = [delegate respondsToSelector:@selector(_webView:didCountMatches:forString:)];
48 m_delegateMethods.webviewDidFindString = [delegate respondsToSelector:@selector(_webView:didFindMatches:forString:withMatchIndex:)];
49 m_delegateMethods.webviewDidFailToFindString = [delegate respondsToSelector:@selector(_webView:didFailToFindString:)];
50}
51
52void FindClient::didCountStringMatches(WebPageProxy*, const String& string, uint32_t matchCount)
53{
54 if (m_delegateMethods.webviewDidCountStringMatches)
55 [m_delegate.get() _webView:m_webView didCountMatches:matchCount forString:string];
56}
57
58void FindClient::didFindString(WebPageProxy*, const String& string, const Vector<WebCore::IntRect>&, uint32_t matchCount, int32_t matchIndex, bool)
59{
60 if (m_delegateMethods.webviewDidFindString)
61 [m_delegate.get() _webView:m_webView didFindMatches:matchCount forString:string withMatchIndex:matchIndex];
62}
63
64void FindClient::didFailToFindString(WebPageProxy*, const String& string)
65{
66 if (m_delegateMethods.webviewDidFailToFindString)
67 [m_delegate.get() _webView:m_webView didFailToFindString:string];
68}
