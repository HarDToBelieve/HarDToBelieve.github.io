35Ref<GroupActivitiesSession> GroupActivitiesSession::create(RetainPtr<WKGroupSession>&& session)
36{
37 return adoptRef(*new GroupActivitiesSession(WTFMove(session)));
38}
39
40GroupActivitiesSession::GroupActivitiesSession(RetainPtr<WKGroupSession>&& session)
41 : m_groupSession(WTFMove(session))
42{
43 m_groupSession.get().newActivityCallback = [this] (WKURLActivity *activity) {
44 m_fallbackURLObservers.forEach([this, activity] (auto& observer) {
45 observer(*this, activity.fallbackURL);
46 });
47 };
48
49 m_groupSession.get().stateChangedCallback = [this] (WKGroupSessionState state) {
50 static_assert(static_cast<size_t>(State::Waiting) == static_cast<size_t>(WKGroupSessionStateWaiting), "MediaSessionCoordinatorState::Waiting != WKGroupSessionStateWaiting");
51 static_assert(static_cast<size_t>(State::Joined) == static_cast<size_t>(WKGroupSessionStateJoined), "MediaSessionCoordinatorState::Joined != WKGroupSessionStateJoined");
52 static_assert(static_cast<size_t>(State::Invalidated) == static_cast<size_t>(WKGroupSessionStateInvalidated), "MediaSessionCoordinatorState::Closed != WKGroupSessionStateInvalidated");
53 m_stateChangeObservers.forEach([this, state] (auto& observer) {
54 observer(*this, static_cast<State>(state));
55 });
56 };
57}
58
59GroupActivitiesSession::~GroupActivitiesSession()
60{
61 m_groupSession.get().newActivityCallback = nil;
62 m_groupSession.get().stateChangedCallback = nil;
63}
64
65void GroupActivitiesSession::join()
66{
67 [m_groupSession join];
68}
69
70void GroupActivitiesSession::leave()
71{
72 [m_groupSession leave];
73}
74
75auto GroupActivitiesSession::state() const -> State
76{
77 static_assert(static_cast<size_t>(State::Waiting) == static_cast<size_t>(WKGroupSessionStateWaiting), "State::Waiting != WKGroupSessionStateWaiting");
78 static_assert(static_cast<size_t>(State::Joined) == static_cast<size_t>(WKGroupSessionStateJoined), "State::Joined != WKGroupSessionStateJoined");
79 static_assert(static_cast<size_t>(State::Invalidated) == static_cast<size_t>(WKGroupSessionStateInvalidated), "State::Invalidated != WKGroupSessionStateInvalidated");
80 return static_cast<State>([m_groupSession state]);
81}
82
83String GroupActivitiesSession::uuid() const
84{
85 return [m_groupSession uuid].UUIDString;
86}
87
88URL GroupActivitiesSession::fallbackURL() const
89{
90 return [m_groupSession activity].fallbackURL;
91}

98void GroupActivitiesSession::addFallbackURLObserver(const FallbackURLObserver& observer)

100 m_fallbackURLObservers.add(observer);
