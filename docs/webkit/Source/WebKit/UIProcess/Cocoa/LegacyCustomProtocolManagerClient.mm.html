26#import "config.h"
27#import "LegacyCustomProtocolManagerClient.h"

29#import "DataReference.h"
30#import "LegacyCustomProtocolManagerProxy.h"
31#import <WebCore/ResourceError.h>
32#import <WebCore/ResourceRequest.h>
33#import <WebCore/ResourceResponse.h>
34
35@interface WKCustomProtocolLoader : NSObject <NSURLConnectionDelegate> {
36@private
37 WeakPtr<WebKit::LegacyCustomProtocolManagerProxy> _customProtocolManagerProxy;
38 WebKit::LegacyCustomProtocolID _customProtocolID;
39 NSURLCacheStoragePolicy _storagePolicy;
40 RetainPtr<NSURLConnection> _urlConnection;
41}
42- (id)initWithLegacyCustomProtocolManagerProxy:(WebKit::LegacyCustomProtocolManagerProxy&)customProtocolManagerProxy customProtocolID:(WebKit::LegacyCustomProtocolID)customProtocolID request:(NSURLRequest *)request;
43- (void)cancel;
44@end
45
46@implementation WKCustomProtocolLoader
47
48- (id)initWithLegacyCustomProtocolManagerProxy:(WebKit::LegacyCustomProtocolManagerProxy&)customProtocolManagerProxy customProtocolID:(WebKit::LegacyCustomProtocolID)customProtocolID request:(NSURLRequest *)request
49{
50 self = [super init];
51 if (!self)
52 return nil;
53
54 ASSERT(request);
55 _customProtocolManagerProxy = makeWeakPtr(customProtocolManagerProxy);
56 _customProtocolID = customProtocolID;
57 _storagePolicy = NSURLCacheStorageNotAllowed;
58 ALLOW_DEPRECATED_DECLARATIONS_BEGIN
59 _urlConnection = adoptNS([[NSURLConnection alloc] initWithRequest:request delegate:self startImmediately:NO]);
60 [_urlConnection scheduleInRunLoop:[NSRunLoop mainRunLoop] forMode:NSRunLoopCommonModes];
61 [_urlConnection start];
62 ALLOW_DEPRECATED_DECLARATIONS_END
63
64 return self;
65}
66
67- (void)dealloc
68{
69 [_urlConnection cancel];
70 [super dealloc];
71}
72
73- (void)cancel
74{
75 ASSERT(_customProtocolManagerProxy);
76 _customProtocolManagerProxy = nullptr;
77 [_urlConnection cancel];
78}
79
80- (void)connection:(NSURLConnection *)connection didFailWithError:(NSError *)error
81{
82 if (!_customProtocolManagerProxy)
83 return;
84
85 WebCore::ResourceError coreError(error);
86 _customProtocolManagerProxy->didFailWithError(_customProtocolID, coreError);
87 _customProtocolManagerProxy->stopLoading(_customProtocolID);
88}
89
90- (NSCachedURLResponse *)connection:(NSURLConnection *)connection willCacheResponse:(NSCachedURLResponse *)cachedResponse
91{
92 ASSERT(_storagePolicy == NSURLCacheStorageNotAllowed);
93 _storagePolicy = [cachedResponse storagePolicy];
94 return cachedResponse;
95}
96
97- (void)connection:(NSURLConnection *)connection didReceiveResponse:(NSURLResponse *)response
98{
99 if (!_customProtocolManagerProxy)
100 return;
101
102 WebCore::ResourceResponse coreResponse(response);
103 _customProtocolManagerProxy->didReceiveResponse(_customProtocolID, coreResponse, _storagePolicy);
104}
105
106- (void)connection:(NSURLConnection *)connection didReceiveData:(NSData *)data
107{
108 if (!_customProtocolManagerProxy)
109 return;
110
111 IPC::DataReference coreData(static_cast<const uint8_t*>([data bytes]), [data length]);
112 _customProtocolManagerProxy->didLoadData(_customProtocolID, coreData);
113}
114
115- (NSURLRequest *)connection:(NSURLConnection *)connection willSendRequest:(NSURLRequest *)request redirectResponse:(NSURLResponse *)redirectResponse
116{
117 if (!_customProtocolManagerProxy)
118 return nil;
119
120 if (redirectResponse) {
121 _customProtocolManagerProxy->wasRedirectedToRequest(_customProtocolID, request, redirectResponse);
122 return nil;
123 }
124 return request;
125}
126
127- (void)connectionDidFinishLoading:(NSURLConnection *)connection
128{
129 if (!_customProtocolManagerProxy)
130 return;
131
132 _customProtocolManagerProxy->didFinishLoading(_customProtocolID);
133 _customProtocolManagerProxy->stopLoading(_customProtocolID);
134}
135
136@end

138namespace WebKit {
139using namespace WebCore;
140
141void LegacyCustomProtocolManagerClient::startLoading(LegacyCustomProtocolManagerProxy& manager, WebKit::LegacyCustomProtocolID customProtocolID, const ResourceRequest& coreRequest)
142{
143 NSURLRequest *request = coreRequest.nsURLRequest(HTTPBodyUpdatePolicy::DoNotUpdateHTTPBody);
144 if (!request)
145 return;
146
147 auto loader = adoptNS([[WKCustomProtocolLoader alloc] initWithLegacyCustomProtocolManagerProxy:manager customProtocolID:customProtocolID request:request]);
148 ASSERT(loader);
149 ASSERT(!m_loaderMap.contains(customProtocolID));
150 m_loaderMap.add(customProtocolID, WTFMove(loader));
151}
152
153void LegacyCustomProtocolManagerClient::stopLoading(LegacyCustomProtocolManagerProxy&, WebKit::LegacyCustomProtocolID customProtocolID)
154{
155 m_loaderMap.remove(customProtocolID);
156}
157
158void LegacyCustomProtocolManagerClient::invalidate(LegacyCustomProtocolManagerProxy&)
159{
160 while (!m_loaderMap.isEmpty()) {
161 auto loader = m_loaderMap.take(m_loaderMap.begin()->key);
162 ASSERT(loader);
163 [loader cancel];
164 }
165}
