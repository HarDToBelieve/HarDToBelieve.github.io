53LegacyDownloadClient::LegacyDownloadClient(id <_WKDownloadDelegate> delegate)
54 : m_delegate(delegate)
55{
56 m_delegateMethods.downloadDidStart = [delegate respondsToSelector:@selector(_downloadDidStart:)];
57 m_delegateMethods.downloadDidReceiveResponse = [delegate respondsToSelector:@selector(_download:didReceiveResponse:)];
58 m_delegateMethods.downloadDidReceiveData = [delegate respondsToSelector:@selector(_download:didReceiveData:)];
59 m_delegateMethods.downloadDidWriteDataTotalBytesWrittenTotalBytesExpectedToWrite = [delegate respondsToSelector:@selector(_download:didWriteData:totalBytesWritten:totalBytesExpectedToWrite:)];
60 m_delegateMethods.downloadDecideDestinationWithSuggestedFilenameAllowOverwrite = [delegate respondsToSelector:@selector(_download:decideDestinationWithSuggestedFilename:allowOverwrite:)];
61 m_delegateMethods.downloadDecideDestinationWithSuggestedFilenameCompletionHandler = [delegate respondsToSelector:@selector(_download:decideDestinationWithSuggestedFilename:completionHandler:)];
62 m_delegateMethods.downloadDidFinish = [delegate respondsToSelector:@selector(_downloadDidFinish:)];
63 m_delegateMethods.downloadDidFail = [delegate respondsToSelector:@selector(_download:didFailWithError:)];
64 m_delegateMethods.downloadDidCancel = [delegate respondsToSelector:@selector(_downloadDidCancel:)];
65 m_delegateMethods.downloadDidReceiveServerRedirectToURL = [delegate respondsToSelector:@selector(_download:didReceiveServerRedirectToURL:)];
66 m_delegateMethods.downloadDidReceiveAuthenticationChallengeCompletionHandler = [delegate respondsToSelector:@selector(_download:didReceiveAuthenticationChallenge:completionHandler:)];
67 m_delegateMethods.downloadShouldDecodeSourceDataOfMIMEType = [delegate respondsToSelector:@selector(_download:shouldDecodeSourceDataOfMIMEType:)];
68 m_delegateMethods.downloadDidCreateDestination = [delegate respondsToSelector:@selector(_download:didCreateDestination:)];
69 m_delegateMethods.downloadProcessDidCrash = [delegate respondsToSelector:@selector(_downloadProcessDidCrash:)];
70}

72#if USE(SYSTEM_PREVIEW)
73static SystemPreviewController* systemPreviewController(DownloadProxy& downloadProxy)
74{
75 auto* page = downloadProxy.originatingPage();
76 if (!page)
77 return nullptr;
78 return page->systemPreviewController();
79}

82void LegacyDownloadClient::legacyDidStart(DownloadProxy& downloadProxy)
83{
84#if USE(SYSTEM_PREVIEW)
85 if (downloadProxy.isSystemPreviewDownload()) {
86 if (auto* webPage = downloadProxy.originatingPage()) {
87 // FIXME: Update the MIME-type once it is known in the ResourceResponse.
88 webPage->systemPreviewController()->start(URL(URL(), webPage->currentURL()), "application/octet-stream"_s, downloadProxy.systemPreviewDownloadInfo());
89 }
90 takeActivityToken(downloadProxy);
91 return;
92 }
93#endif
94
95 if (m_delegateMethods.downloadDidStart)
96 [m_delegate _downloadDidStart:[_WKDownload downloadWithDownload:wrapper(downloadProxy)]];
97}

99void LegacyDownloadClient::didReceiveResponse(DownloadProxy& downloadProxy, const WebCore::ResourceResponse& response)
100{
101#if USE(SYSTEM_PREVIEW)
102 if (downloadProxy.isSystemPreviewDownload() && response.isSuccessful()) {
103 if (auto* controller = systemPreviewController(downloadProxy))
104 controller->updateProgress(0);
105 return;
106 }
107#endif
108
109 if (m_delegateMethods.downloadDidReceiveResponse)
110 [m_delegate _download:[_WKDownload downloadWithDownload:wrapper(downloadProxy)] didReceiveResponse:response.nsURLResponse()];
111}

113void LegacyDownloadClient::didReceiveData(DownloadProxy& downloadProxy, uint64_t bytesWritten, uint64_t totalBytesWritten, uint64_t totalBytesExpectedToWrite)
114{
115#if USE(SYSTEM_PREVIEW)
116 if (downloadProxy.isSystemPreviewDownload()) {
117 if (auto* controller = systemPreviewController(downloadProxy))
118 controller->updateProgress(static_cast<float>(totalBytesWritten) / totalBytesExpectedToWrite);
119 return;
120 }
121#endif
122
123 if (m_delegateMethods.downloadDidWriteDataTotalBytesWrittenTotalBytesExpectedToWrite)
124 [m_delegate _download:[_WKDownload downloadWithDownload:wrapper(downloadProxy)] didWriteData:bytesWritten totalBytesWritten:totalBytesWritten totalBytesExpectedToWrite:totalBytesExpectedToWrite];
125 else if (m_delegateMethods.downloadDidReceiveData)
126 [m_delegate _download:[_WKDownload downloadWithDownload:wrapper(downloadProxy)] didReceiveData:bytesWritten];
127}

129void LegacyDownloadClient::didReceiveAuthenticationChallenge(DownloadProxy& downloadProxy, AuthenticationChallengeProxy& authenticationChallenge)
130{
131 // FIXME: System Preview needs code here.
132 if (!m_delegateMethods.downloadDidReceiveAuthenticationChallengeCompletionHandler) {
133 authenticationChallenge.listener().completeChallenge(WebKit::AuthenticationChallengeDisposition::PerformDefaultHandling);
134 return;
135 }
136
137 [m_delegate _download:[_WKDownload downloadWithDownload:wrapper(downloadProxy)] didReceiveAuthenticationChallenge:wrapper(authenticationChallenge) completionHandler:makeBlockPtr([authenticationChallenge = makeRef(authenticationChallenge), checker = CompletionHandlerCallChecker::create(m_delegate.get().get(), @selector(_download:didReceiveAuthenticationChallenge:completionHandler:))] (NSURLSessionAuthChallengeDisposition disposition, NSURLCredential *credential) {
138 if (checker->completionHandlerHasBeenCalled())
139 return;
140 checker->didCallCompletionHandler();
141 switch (disposition) {
142 case NSURLSessionAuthChallengeUseCredential:
143 authenticationChallenge->listener().completeChallenge(AuthenticationChallengeDisposition::UseCredential, credential ? WebCore::Credential(credential) : WebCore::Credential());
144 break;
145 case NSURLSessionAuthChallengePerformDefaultHandling:
146 authenticationChallenge->listener().completeChallenge(AuthenticationChallengeDisposition::PerformDefaultHandling);
147 break;
148
149 case NSURLSessionAuthChallengeCancelAuthenticationChallenge:
150 authenticationChallenge->listener().completeChallenge(AuthenticationChallengeDisposition::Cancel);
151 break;
152
153 case NSURLSessionAuthChallengeRejectProtectionSpace:
154 authenticationChallenge->listener().completeChallenge(AuthenticationChallengeDisposition::RejectProtectionSpaceAndContinue);
155 break;
156
157 default:
158 [NSException raise:NSInvalidArgumentException format:@"Invalid NSURLSessionAuthChallengeDisposition (%ld)", (long)disposition];
159 }
160 }).get()];
161}

176void LegacyDownloadClient::processDidCrash(DownloadProxy& downloadProxy)
177{
178#if USE(SYSTEM_PREVIEW)

180 if (auto* controller = systemPreviewController(downloadProxy))

191void LegacyDownloadClient::decideDestinationWithSuggestedFilename(DownloadProxy& downloadProxy, const WebCore::ResourceResponse& response, const String& filename, CompletionHandler<void(AllowOverwrite, String)>&& completionHandler)
192{
193 didReceiveResponse(downloadProxy, response);
194
195#if USE(SYSTEM_PREVIEW)
196 if (downloadProxy.isSystemPreviewDownload()) {
197 NSString *temporaryDirectory = FileSystem::createTemporaryDirectory(@"SystemPreviews");
198 NSString *destination = [temporaryDirectory stringByAppendingPathComponent:filename];
199 completionHandler(AllowOverwrite::Yes, destination);
200 return;
201 }
202#endif
203
204 if (!m_delegateMethods.downloadDecideDestinationWithSuggestedFilenameAllowOverwrite && !m_delegateMethods.downloadDecideDestinationWithSuggestedFilenameCompletionHandler)
205 return completionHandler(AllowOverwrite::No, { });
206
207 if (m_delegateMethods.downloadDecideDestinationWithSuggestedFilenameAllowOverwrite) {
208 BOOL allowOverwrite = NO;
209 ALLOW_DEPRECATED_DECLARATIONS_BEGIN
210 NSString *destination = [m_delegate _download:[_WKDownload downloadWithDownload:wrapper(downloadProxy)] decideDestinationWithSuggestedFilename:filename allowOverwrite:&allowOverwrite];
211 ALLOW_DEPRECATED_DECLARATIONS_END
212 completionHandler(allowOverwrite ? AllowOverwrite::Yes : AllowOverwrite::No, destination);
213 } else {
214 [m_delegate _download:[_WKDownload downloadWithDownload:wrapper(downloadProxy)] decideDestinationWithSuggestedFilename:filename completionHandler:makeBlockPtr([checker = CompletionHandlerCallChecker::create(m_delegate.get().get(), @selector(_download:decideDestinationWithSuggestedFilename:completionHandler:)), completionHandler = WTFMove(completionHandler)] (BOOL allowOverwrite, NSString *destination) mutable {
215 if (checker->completionHandlerHasBeenCalled())
216 return;
217 checker->didCallCompletionHandler();
218 completionHandler(allowOverwrite ? AllowOverwrite::Yes : AllowOverwrite::No, destination);
219 }).get()];
220 }
221}

223void LegacyDownloadClient::didFinish(DownloadProxy& downloadProxy)
224{
225#if USE(SYSTEM_PREVIEW)
226 if (downloadProxy.isSystemPreviewDownload()) {
227 if (auto* controller = systemPreviewController(downloadProxy)) {
228 auto destinationURL = WTF::URL::fileURLWithFileSystemPath(downloadProxy.destinationFilename());
229 if (!destinationURL.hasFragmentIdentifier())
230 destinationURL.setFragmentIdentifier(downloadProxy.request().url().fragmentIdentifier());
231 controller->finish(destinationURL);
232 }
233 releaseActivityTokenIfNecessary(downloadProxy);
234 return;
235 }
236#endif
237
238 if (m_delegateMethods.downloadDidFinish)
239 [m_delegate _downloadDidFinish:[_WKDownload downloadWithDownload:wrapper(downloadProxy)]];
240}

251#endif
252
253 if (m_delegateMethods.downloadDidFail)

255}

272void LegacyDownloadClient::willSendRequest(DownloadProxy& downloadProxy, WebCore::ResourceRequest&& request, const WebCore::ResourceResponse&, CompletionHandler<void(WebCore::ResourceRequest&&)>&& completionHandler)
273{
274 if (m_delegateMethods.downloadDidReceiveServerRedirectToURL)
275 [m_delegate _download:[_WKDownload downloadWithDownload:wrapper(downloadProxy)] didReceiveServerRedirectToURL:[NSURL _web_URLWithWTFString:request.url().string()]];

280#if USE(SYSTEM_PREVIEW)
281void LegacyDownloadClient::takeActivityToken(DownloadProxy& downloadProxy)
282{
283#if PLATFORM(IOS_FAMILY)
284 if (auto* webPage = downloadProxy.originatingPage()) {
285 RELEASE_LOG(ProcessSuspension, "%p - UIProcess is taking a background assertion because it is downloading a system preview", this);
286 ASSERT(!m_activity);
287 m_activity = webPage->process().throttler().backgroundActivity("System preview download"_s).moveToUniquePtr();
288 }
289#else
290 UNUSED_PARAM(downloadProxy);
291#endif
292}
293
294void LegacyDownloadClient::releaseActivityTokenIfNecessary(DownloadProxy& downloadProxy)
295{
296#if PLATFORM(IOS_FAMILY)
297 if (m_activity) {
298 RELEASE_LOG(ProcessSuspension, "%p UIProcess is releasing a background assertion because a system preview download completed", this);
299 m_activity = nullptr;
300 }
301#else
302 UNUSED_PARAM(downloadProxy);
303#endif
304}
305#endif
