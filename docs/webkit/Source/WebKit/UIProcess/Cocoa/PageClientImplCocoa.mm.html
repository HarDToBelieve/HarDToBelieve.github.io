127NSFileWrapper *PageClientImplCocoa::allocFileWrapperInstance() const
128{
129 Class cls = [m_webView configuration]._attachmentFileWrapperClass ?: [NSFileWrapper class];
130 return [cls alloc];
131}
132
133NSSet *PageClientImplCocoa::serializableFileWrapperClasses() const
134{
135 Class defaultFileWrapperClass = NSFileWrapper.class;
136 Class configuredFileWrapperClass = [m_webView configuration]._attachmentFileWrapperClass;
137 if (configuredFileWrapperClass && configuredFileWrapperClass != defaultFileWrapperClass)
138 return [NSSet setWithObjects:configuredFileWrapperClass, defaultFileWrapperClass, nil];
139 return [NSSet setWithObjects:defaultFileWrapperClass, nil];
140}
