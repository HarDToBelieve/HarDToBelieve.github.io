319PlaybackSessionManagerProxy::PlaybackSessionManagerProxy(WebPageProxy& page)
320 : m_page(&page)
321{
322 m_page->process().addMessageReceiver(Messages::PlaybackSessionManagerProxy::messageReceiverName(), m_page->webPageID(), *this);
323}
324
325PlaybackSessionManagerProxy::~PlaybackSessionManagerProxy()
326{
327 if (!m_page)
328 return;
329 invalidate();
330}
331
332void PlaybackSessionManagerProxy::invalidate()
333{
334 m_page->process().removeMessageReceiver(Messages::PlaybackSessionManagerProxy::messageReceiverName(), m_page->webPageID());
335 m_page = nullptr;
336
337 auto contextMap = WTFMove(m_contextMap);
338 m_clientCounts.clear();
339
340 for (auto& [model, interface] : contextMap.values())
341 interface->invalidate();
342}
343
344PlaybackSessionManagerProxy::ModelInterfaceTuple PlaybackSessionManagerProxy::createModelAndInterface(PlaybackSessionContextIdentifier contextId)
345{
346 Ref<PlaybackSessionModelContext> model = PlaybackSessionModelContext::create(*this, contextId);
347 Ref<PlatformPlaybackSessionInterface> interface = PlatformPlaybackSessionInterface::create(model);
348
349 return std::make_tuple(WTFMove(model), WTFMove(interface));
350}
351
352PlaybackSessionManagerProxy::ModelInterfaceTuple& PlaybackSessionManagerProxy::ensureModelAndInterface(PlaybackSessionContextIdentifier contextId)
353{
354 auto addResult = m_contextMap.add(contextId, ModelInterfaceTuple());
355 if (addResult.isNewEntry)
356 addResult.iterator->value = createModelAndInterface(contextId);
357 return addResult.iterator->value;
358}
359
360PlaybackSessionModelContext& PlaybackSessionManagerProxy::ensureModel(PlaybackSessionContextIdentifier contextId)
361{
362 return *std::get<0>(ensureModelAndInterface(contextId));
363}
364
365PlatformPlaybackSessionInterface& PlaybackSessionManagerProxy::ensureInterface(PlaybackSessionContextIdentifier contextId)
366{
367 return *std::get<1>(ensureModelAndInterface(contextId));
368}

372 m_clientCounts.add(contextId);
373}

375void PlaybackSessionManagerProxy::removeClientForContext(PlaybackSessionContextIdentifier contextId)
376{
377 if (!m_clientCounts.remove(contextId))
378 return;
379
380 ensureInterface(contextId).invalidate();
381 m_contextMap.remove(contextId);
382}

386void PlaybackSessionManagerProxy::setUpPlaybackControlsManagerWithID(PlaybackSessionContextIdentifier contextId)
387{
388 if (m_controlsManagerContextId == contextId)
389 return;
390
391 if (m_controlsManagerContextId)
392 removeClientForContext(m_controlsManagerContextId);
393
394 m_controlsManagerContextId = contextId;
395 ensureInterface(m_controlsManagerContextId).ensureControlsManager();
396 addClientForContext(m_controlsManagerContextId);
397
398 m_page->videoControlsManagerDidChange();
399}
400
401void PlaybackSessionManagerProxy::clearPlaybackControlsManager()
402{
403 if (!m_controlsManagerContextId)
404 return;
405
406 removeClientForContext(m_controlsManagerContextId);
407 m_controlsManagerContextId = { };
408 m_page->videoControlsManagerDidChange();
409}

421void PlaybackSessionManagerProxy::seekableRangesVectorChanged(PlaybackSessionContextIdentifier contextId, Vector<std::pair<double, double>> ranges, double lastModifiedTime, double liveUpdateInterval)
422{
423 Ref<TimeRanges> timeRanges = TimeRanges::create();
424 for (const auto& range : ranges) {
425 ASSERT(isfinite(range.first));
426 ASSERT(isfinite(range.second));
427 ASSERT(range.second >= range.first);
428 timeRanges->add(range.first, range.second);
429 }
430
431 ensureModel(contextId).seekableRangesChanged(timeRanges, lastModifiedTime, liveUpdateInterval);
432}
