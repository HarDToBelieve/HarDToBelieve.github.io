77void ResourceLoadDelegate::ResourceLoadClient::didSendRequest(WebKit::ResourceLoadInfo&& loadInfo, WebCore::ResourceRequest&& request) const
78{
79 if (!m_resourceLoadDelegate.m_delegateMethods.didSendRequest)
80 return;
81
82 auto delegate = m_resourceLoadDelegate.m_delegate.get();
83 if (!delegate)
84 return;
85
86 [delegate webView:m_resourceLoadDelegate.m_webView.get().get() resourceLoad:wrapper(API::ResourceLoadInfo::create(WTFMove(loadInfo)).get()) didSendRequest:request.nsURLRequest(WebCore::HTTPBodyUpdatePolicy::UpdateHTTPBody)];
87}

99}
