53static const char* Redirect = "Redirect";
54static const char* PopUp = "PopUp";
55static const char* SubFrame = "SubFrame";

57static const char* toString(const SOAuthorizationSession::InitiatingAction& action)
58{
59 switch (action) {
60 case SOAuthorizationSession::InitiatingAction::Redirect:
61 return Redirect;
62 case SOAuthorizationSession::InitiatingAction::PopUp:
63 return PopUp;
64 case SOAuthorizationSession::InitiatingAction::SubFrame:
65 return SubFrame;
66 }
67
68 ASSERT_NOT_REACHED();
69 return nullptr;
70}
71
72static Vector<WebCore::Cookie> toCookieVector(NSArray<NSHTTPCookie *> *cookies)
73{
74 Vector<WebCore::Cookie> result;
75 result.reserveInitialCapacity(cookies.count);
76 for (id cookie in cookies)
77 result.uncheckedAppend(cookie);
78 return result;
79}

81static bool isSameOrigin(const WebCore::ResourceRequest& request, const WebCore::ResourceResponse& response)
82{
83 auto requestOrigin = WebCore::SecurityOrigin::create(request.url());
84 return requestOrigin->isSameOriginAs(WebCore::SecurityOrigin::create(response.url()).get());
85}

97SOAuthorizationSession::~SOAuthorizationSession()
98{
99 AUTHORIZATIONSESSION_RELEASE_LOG("~SOAuthorizationSession: m_viewController=%p", m_viewController.get());
100#if PLATFORM(MAC)
101 AUTHORIZATIONSESSION_RELEASE_LOG("~SOAuthorizationSession: m_sheetWindow=%p", m_sheetWindow.get());
102#endif
103
104 if (m_state == State::Active && !!m_soAuthorization)
105 [m_soAuthorization cancelAuthorization];
106 if (m_state != State::Idle && m_state != State::Completed)
107 becomeCompleted();
108 else
109 dismissViewController();
110}

117const char* SOAuthorizationSession::stateString() const
118{
119 static const char* Idle = "Idle";
120 static const char* Active = "Active";
121 static const char* Waiting = "Waiting";
122 static const char* Completed = "Completed";
123
124 switch (m_state) {
125 case State::Idle:
126 return Idle;
127 case State::Active:
128 return Active;
129 case State::Waiting:
130 return Waiting;
131 case State::Completed:
132 return Completed;
133 }
134
135 ASSERT_NOT_REACHED();
136 return nullptr;
137}

167void SOAuthorizationSession::start()
168{
169 AUTHORIZATIONSESSION_RELEASE_LOG("start: navigationAction=%p", navigationAction());
170
171 ASSERT((m_state == State::Idle || m_state == State::Waiting) && m_navigationAction);
172 m_state = State::Active;
173 AUTHORIZATIONSESSION_RELEASE_LOG("start: Moving m_state to Active.");
174 [m_soAuthorization getAuthorizationHintsWithURL:m_navigationAction->request().url() responseCode:0 completion:makeBlockPtr([this, weakThis = makeWeakPtr(*this)] (SOAuthorizationHints *authorizationHints, NSError *error) {
175 AUTHORIZATIONSESSION_RELEASE_LOG("start: Receive SOAuthorizationHints (error=%ld)", error ? error.code : 0);
176
177 if (!weakThis) {
178 RELEASE_LOG_ERROR(AppSSO, "SOAuthorizationSession::start (getAuthorizationHintsWithURL completion handler): Returning early because weakThis is now null.");
179 return;
180 }
181
182 if (error || !authorizationHints) {
183 AUTHORIZATIONSESSION_RELEASE_LOG("start (getAuthorizationHintsWithURL completion handler): Returning early due to error or lack of hints.");
184 return;
185 }
186
187 AUTHORIZATIONSESSION_RELEASE_LOG("start (getAuthorizationHintsWithURL completion handler): Receive SOAuthorizationHints.");
188 continueStartAfterGetAuthorizationHints(authorizationHints.localizedExtensionBundleDisplayName);
189 }).get()];
190}

192void SOAuthorizationSession::continueStartAfterGetAuthorizationHints(const String& hints)
193{
194 AUTHORIZATIONSESSION_RELEASE_LOG("continueStartAfterGetAuthorizationHints: (hints=%s)", hints.utf8().data());
195
196 ASSERT(m_state == State::Active);
197 if (!m_page) {
198 AUTHORIZATIONSESSION_RELEASE_LOG("continueStartAfterGetAuthorizationHints: Early return due to null m_page");
199 return;
200 }
201
202 AUTHORIZATIONSESSION_RELEASE_LOG("continueStartAfterGetAuthorizationHints: Checking page for policy choice.");
203 m_page->decidePolicyForSOAuthorizationLoad(hints, [this, weakThis = makeWeakPtr(*this)] (SOAuthorizationLoadPolicy policy) {
204 if (!weakThis)
205 return;
206 continueStartAfterDecidePolicy(policy);
207 });
208}

210void SOAuthorizationSession::continueStartAfterDecidePolicy(const SOAuthorizationLoadPolicy& policy)
211{
212 if (policy == SOAuthorizationLoadPolicy::Ignore) {
213 AUTHORIZATIONSESSION_RELEASE_LOG("continueStartAfterDecidePolicy: Receive SOAuthorizationLoadPolicy::Ignore. Falling back to web path.");
214 fallBackToWebPath();
215 return;
216 }
217
218 AUTHORIZATIONSESSION_RELEASE_LOG("continueStartAfterDecidePolicy: Receive SOAuthorizationLoadPolicy::Allow");
219
220 if (!m_soAuthorization || !m_page || !m_navigationAction) {
221 AUTHORIZATIONSESSION_RELEASE_LOG("continueStartAfterGetAuthorizationHints: Early return m_soAuthorization=%d, m_page=%p, navigationAction=%p.", !!m_soAuthorization, page(), navigationAction());
222 return;
223 }
224
225 auto initiatorOrigin = emptyString();
226 if (m_navigationAction->sourceFrame())
227 initiatorOrigin = m_navigationAction->sourceFrame()->securityOrigin().securityOrigin()->toString();
228 if (m_action == InitiatingAction::SubFrame && m_page->mainFrame())
229 initiatorOrigin = WebCore::SecurityOrigin::create(m_page->mainFrame()->url())->toString();
230 NSDictionary *authorizationOptions = @{
231 SOAuthorizationOptionUserActionInitiated: @(m_navigationAction->isProcessingUserGesture()),
232 SOAuthorizationOptionInitiatorOrigin: (NSString *)initiatorOrigin,
233 SOAuthorizationOptionInitiatingAction: @(static_cast<NSInteger>(m_action))
234 };
235 [m_soAuthorization setAuthorizationOptions:authorizationOptions];
236
237#if PLATFORM(IOS)
238 if (![[m_page->cocoaView() UIDelegate] respondsToSelector:@selector(_presentingViewControllerForWebView:)])
239 [m_soAuthorization setEnableEmbeddedAuthorizationViewController:NO];
240#endif
241
242 auto *nsRequest = m_navigationAction->request().nsURLRequest(WebCore::HTTPBodyUpdatePolicy::UpdateHTTPBody);
243 AUTHORIZATIONSESSION_RELEASE_LOG("continueStartAfterGetAuthorizationHints: Beginning authorization with AppSSO.");
244 [m_soAuthorization beginAuthorizationWithURL:nsRequest.URL httpHeaders:nsRequest.allHTTPHeaderFields httpBody:nsRequest.HTTPBody];
245}

278void SOAuthorizationSession::complete(NSHTTPURLResponse *httpResponse, NSData *data)
279{
280 AUTHORIZATIONSESSION_RELEASE_LOG("complete: httpState=%d, m_viewController=%p", static_cast<int>(httpResponse.statusCode), m_viewController.get());
281#if PLATFORM(MAC)
282 AUTHORIZATIONSESSION_RELEASE_LOG("complete: m_sheetWindow=%p", m_sheetWindow.get());
283#endif
284
285 if (m_state != State::Active) {
286 AUTHORIZATIONSESSION_RELEASE_LOG("complete: Returning early since not active.");
287 dismissViewController();
288 return;
289 }
290
291 ASSERT(m_navigationAction);
292 becomeCompleted();
293
294 auto response = WebCore::ResourceResponse(httpResponse);
295 if (!isSameOrigin(m_navigationAction->request(), response)) {
296 AUTHORIZATIONSESSION_RELEASE_LOG("complete: Origins don't match. Falling back to web path.");
297 fallBackToWebPathInternal();
298 return;
299 }
300
301 // Set cookies.
302 auto cookies = toCookieVector([NSHTTPCookie cookiesWithResponseHeaderFields:httpResponse.allHeaderFields forURL:response.url()]);
303
304 AUTHORIZATIONSESSION_RELEASE_LOG("complete: (httpStatusCode=%d, hasCookies=%d, hasData=%d)", response.httpStatusCode(), !cookies.isEmpty(), !!data.length);
305
306 if (cookies.isEmpty()) {
307 AUTHORIZATIONSESSION_RELEASE_LOG("complete: No cookies to set. Completing (internal).");
308 completeInternal(response, data);
309 return;
310 }
311
312 if (!m_page) {
313 AUTHORIZATIONSESSION_RELEASE_LOG("complete: Returning early because m_page is null.");
314 return;
315 }
316
317 m_page->websiteDataStore().cookieStore().setCookies(WTFMove(cookies), [this, weakThis = makeWeakPtr(*this), response = WTFMove(response), data = adoptNS([[NSData alloc] initWithData:data])] () mutable {
318 if (!weakThis)
319 return;
320
321 AUTHORIZATIONSESSION_RELEASE_LOG("complete: Cookies are set.");
322
323 completeInternal(response, data.get());
324 });
325}

327void SOAuthorizationSession::presentViewController(SOAuthorizationViewController viewController, UICallback uiCallback)
328{
329 AUTHORIZATIONSESSION_RELEASE_LOG("presentViewController: m_viewController=%p", m_viewController.get());
330#if PLATFORM(MAC)
331 AUTHORIZATIONSESSION_RELEASE_LOG("presentViewController: m_sheetWindow=%p", m_sheetWindow.get());
332#endif
333
334 ASSERT(m_state == State::Active);
335 // Only expect at most one UI session for the whole authorization session.
336 if (!m_page || m_page->isClosed() || m_viewController) {
337 AUTHORIZATIONSESSION_RELEASE_LOG("presentViewController: m_page=%p, m_page->isClosed=%d, m_viewController=%p", m_page.get(), m_page ? m_page->isClosed() : 0, m_viewController.get());
338 uiCallback(NO, adoptNS([[NSError alloc] initWithDomain:SOErrorDomain code:kSOErrorAuthorizationPresentationFailed userInfo:nil]).get());
339 return;
340 }
341
342 m_viewController = viewController;
343#if PLATFORM(MAC)
344 ASSERT(!m_sheetWindow);
345 ASSERT(!m_sheetWindowWillCloseObserver);
346
347 dismissModalSheetIfNecessary();
348
349 m_sheetWindow = [NSWindow windowWithContentViewController:m_viewController.get()];
350
351 m_sheetWindowWillCloseObserver = [[NSNotificationCenter defaultCenter] addObserverForName:NSWindowWillCloseNotification object:m_sheetWindow.get() queue:nil usingBlock:[weakThis = makeWeakPtr(*this)] (NSNotification *) {
352 if (!weakThis)
353 return;
354 RELEASE_LOG(AppSSO, "presentViewController: Received NSWindowWillCloseNotification. Dismissing the view controller.");
355 weakThis->dismissViewController();
356 }];
357 AUTHORIZATIONSESSION_RELEASE_LOG("presentViewController: Added m_sheetWindowWillCloseObserver (%p)", m_sheetWindowWillCloseObserver.get());
358
359 NSWindow *presentingWindow = m_page->platformWindow();
360 if (!presentingWindow) {
361 AUTHORIZATIONSESSION_RELEASE_LOG("presentViewController: No presenting window. Returning early.");
362 uiCallback(NO, adoptNS([[NSError alloc] initWithDomain:SOErrorDomain code:kSOErrorAuthorizationPresentationFailed userInfo:nil]).get());
363 return;
364 }
365
366 AUTHORIZATIONSESSION_RELEASE_LOG("presentViewController: Calling beginSheet on %p for sheet %p.", presentingWindow, m_sheetWindow.get());
367 [presentingWindow beginSheet:m_sheetWindow.get() completionHandler:nil];
368#elif PLATFORM(IOS)
369 UIViewController *presentingViewController = m_page->uiClient().presentingViewController();
370 if (!presentingViewController) {
371 uiCallback(NO, adoptNS([[NSError alloc] initWithDomain:SOErrorDomain code:kSOErrorAuthorizationPresentationFailed userInfo:nil]).get());
372 return;
373 }
374
375 [presentingViewController presentViewController:m_viewController.get() animated:YES completion:nil];
376#endif
377
378 uiCallback(YES, nil);
379}

381#if PLATFORM(MAC)
382void SOAuthorizationSession::dismissModalSheetIfNecessary()
383{
384 if (auto *presentingWindow = m_sheetWindow.get().sheetParent) {
385 AUTHORIZATIONSESSION_RELEASE_LOG("dismissModalSheetIfNecessary: Calling endSheet on %p for sheet %p.", presentingWindow, m_sheetWindow.get());
386 [presentingWindow endSheet:m_sheetWindow.get()];
387 }
388 m_sheetWindow = nullptr;
389
390 if (m_sheetWindowWillCloseObserver.get()) {
391 AUTHORIZATIONSESSION_RELEASE_LOG("dismissModalSheetIfNecessary: Removing m_sheetWindowWillCloseObserver (%p)", m_sheetWindowWillCloseObserver.get());
392 [[NSNotificationCenter defaultCenter] removeObserver:m_sheetWindowWillCloseObserver.get()];
393 }
394 m_sheetWindowWillCloseObserver = nullptr;
395}
396#endif

398void SOAuthorizationSession::dismissViewController()
399{
400 AUTHORIZATIONSESSION_RELEASE_LOG("dismissViewController: m_viewController=%p", m_viewController.get());
401 if (!m_viewController) {
402 AUTHORIZATIONSESSION_RELEASE_LOG("dismissViewController: No view controller, so returning early.");
403 return;
404 }
405
406#if PLATFORM(MAC)
407 if (!m_sheetWindow) {
408 ASSERT(!m_sheetWindowWillCloseObserver);
409 AUTHORIZATIONSESSION_RELEASE_LOG("dismissViewController: No view controller or sheet window, so returning early.");
410 return;
411 }
412
413 ASSERT(m_sheetWindowWillCloseObserver);
414
415 // This is a workaround for an AppKit issue: <rdar://problem/59125329>.
416 // [m_sheetWindow sheetParent] is null if the parent is minimized or the host app is hidden.
417 if (m_page && m_page->platformWindow()) {
418 auto *presentingWindow = m_page->platformWindow();
419 if (presentingWindow.miniaturized) {
420 AUTHORIZATIONSESSION_RELEASE_LOG("dismissViewController: Page's window is miniaturized. Waiting to dismiss until active.");
421 if (m_presentingWindowDidDeminiaturizeObserver) {
422 AUTHORIZATIONSESSION_RELEASE_LOG("dismissViewController: [Miniaturized] Already has a deminiaturized observer (%p). Hidden observer is %p", m_presentingWindowDidDeminiaturizeObserver.get(), m_applicationDidUnhideObserver.get());
423 return;
424 }
425 m_presentingWindowDidDeminiaturizeObserver = [[NSNotificationCenter defaultCenter] addObserverForName:NSWindowDidDeminiaturizeNotification object:presentingWindow queue:nil usingBlock:[protectedThis = makeRefPtr(this), this] (NSNotification *) {
426 AUTHORIZATIONSESSION_RELEASE_LOG("dismissViewController: Window has deminiaturized. Completing the dismissal.");
427 dismissViewController();
428 [[NSNotificationCenter defaultCenter] removeObserver:m_presentingWindowDidDeminiaturizeObserver.get()];
429 m_presentingWindowDidDeminiaturizeObserver = nullptr;
430 }];
431 return;
432 }
433 }
434
435 if (NSApp.hidden) {
436 AUTHORIZATIONSESSION_RELEASE_LOG("dismissViewController: Application is hidden. Waiting to dismiss until active.");
437 if (m_applicationDidUnhideObserver) {
438 AUTHORIZATIONSESSION_RELEASE_LOG("dismissViewController: [Hidden] Already has an Unhide observer (%p). Deminiaturized observer is %p", m_presentingWindowDidDeminiaturizeObserver.get(), m_applicationDidUnhideObserver.get());
439 return;
440 }
441 m_applicationDidUnhideObserver = [[NSNotificationCenter defaultCenter] addObserverForName:NSApplicationDidUnhideNotification object:NSApp queue:nil usingBlock:[protectedThis = makeRefPtr(this), this] (NSNotification *) {
442 AUTHORIZATIONSESSION_RELEASE_LOG("dismissViewController: Application is no longer hidden. Completing the dismissal.");
443 dismissViewController();
444 [[NSNotificationCenter defaultCenter] removeObserver:m_applicationDidUnhideObserver.get()];
445 m_applicationDidUnhideObserver = nullptr;
446 }];
447 return;
448 }
449
450 dismissModalSheetIfNecessary();
451 AUTHORIZATIONSESSION_RELEASE_LOG("dismissViewController: Finished call with deminiaturized observer (%p) and Hidden observer (%p)", m_presentingWindowDidDeminiaturizeObserver.get(), m_applicationDidUnhideObserver.get());
452#elif PLATFORM(IOS)
453 [[m_viewController presentingViewController] dismissViewControllerAnimated:YES completion:nil];
454#endif
455
456 m_viewController = nullptr;
457}
