35#define WKSOAUTHORIZATIONDELEGATE_RELEASE_LOG(fmt, ...) RELEASE_LOG(AppSSO, "%p - WKSOAuthorizationDelegate::" fmt, &self, ##__VA_ARGS__)

37@implementation WKSOAuthorizationDelegate
38
39- (void)authorization:(SOAuthorization *)authorization presentViewController:(SOAuthorizationViewController)viewController withCompletion:(void (^)(BOOL success, NSError *error))completion
40{
41 ASSERT(RunLoop::isMain() && completion);
42 WKSOAUTHORIZATIONDELEGATE_RELEASE_LOG("authorization (authorization = %p, _session = %p)", authorization, _session.get());
43 if (!_session) {
44 WKSOAUTHORIZATIONDELEGATE_RELEASE_LOG("authorization: No session, so completing with NO as success state.");
45 ASSERT_NOT_REACHED();
46 completion(NO, nil);
47 return;
48 }
49 WKSOAUTHORIZATIONDELEGATE_RELEASE_LOG("authorization: presentingViewController %p", viewController);
50 _session->presentViewController(viewController, completion);
51}
52
53- (void)authorizationDidNotHandle:(SOAuthorization *)authorization
54{
55 ASSERT(RunLoop::isMain());
56 WKSOAUTHORIZATIONDELEGATE_RELEASE_LOG("authorizationDidNotHandle: (authorization = %p, _session = %p)", authorization, _session.get());
57 LOG_ERROR("Could not handle AppSSO.");
58 if (!_session) {
59 WKSOAUTHORIZATIONDELEGATE_RELEASE_LOG("authorizationDidNotHandle: No session, so returning early.");
60 ASSERT_NOT_REACHED();
61 return;
62 }
63 WKSOAUTHORIZATIONDELEGATE_RELEASE_LOG("authorizationDidNotHandle: Falling back to web path.");
64 _session->fallBackToWebPath();
65}

135- (void)setSession:(RefPtr<WebKit::SOAuthorizationSession>&&)session
136{
137 RELEASE_ASSERT(RunLoop::isMain());
138 WKSOAUTHORIZATIONDELEGATE_RELEASE_LOG("setSession: (existing session = %p, new session = %p)", _session.get(), session.get());
139 _session = WTFMove(session);
140 if (_session)
141 _session->shouldStart();
142}
