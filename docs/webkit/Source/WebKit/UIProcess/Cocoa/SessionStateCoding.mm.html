34RetainPtr<NSData> encodeSessionState(const SessionState& sessionState)
35{
36 return wrapper(WebKit::encodeLegacySessionState(sessionState));
37}

39bool decodeSessionState(NSData *data, SessionState& state)
40{
41 return decodeLegacySessionState(static_cast<const uint8_t*>(data.bytes), data.length, state);
42}
