26#import "config.h"
27#import "WKEditCommand.h"
28
29#import "WebEditCommandProxy.h"
30
31@implementation WKEditCommand
32
33- (instancetype)initWithWebEditCommandProxy:(Ref<WebKit::WebEditCommandProxy>&&)command
34{
35 self = [super init];
36 if (!self)
37 return nil;
38
39 _command = WTFMove(command);
40 return self;
41}
42
43- (WebKit::WebEditCommandProxy&)command
44{
45 return *_command;
46}
47
48@end
49
50@implementation WKEditorUndoTarget
51
52- (void)undoEditing:(id)sender
53{
54 ASSERT([sender isKindOfClass:WKEditCommand.class]);
55 [sender command].unapply();
56}
57
58- (void)redoEditing:(id)sender
59{
60 ASSERT([sender isKindOfClass:WKEditCommand.class]);
61 [sender command].reapply();
62}
63
64@end
