81static CocoaFont *fontOfSize(WarningTextSize size)
82{
83#if PLATFORM(MAC)
84 switch (size) {
85 case WarningTextSize::Title:
86 return [NSFont boldSystemFontOfSize:26];
87 case WarningTextSize::Body:
88 return [NSFont systemFontOfSize:14];
89 }
90#elif HAVE(SAFE_BROWSING)
91 switch (size) {
92 case WarningTextSize::Title:
93#if PLATFORM(WATCHOS)
94 return [UIFont preferredFontForTextStyle:UIFontTextStyleSubheadline];
95#else
96 return [UIFont preferredFontForTextStyle:UIFontTextStyleLargeTitle];
97#endif
98 case WarningTextSize::Body:
99 return [UIFont preferredFontForTextStyle:UIFontTextStyleBody];
100 }
101#else
102 return nil;
103#endif
104}

106static CocoaColor *colorForItem(WarningItem item, ViewType *warning)
107{
108 ASSERT([warning isKindOfClass:[WKSafeBrowsingWarning class]]);
109#if PLATFORM(MAC)
110
111 auto colorNamed = [] (NSString *name) -> CocoaColor * {
112#if HAVE(SAFE_BROWSING)
113 return [NSColor colorNamed:name bundle:[NSBundle bundleWithIdentifier:@"com.apple.WebKit"]];
114#else
115 ASSERT_NOT_REACHED();
116 return nil;
117#endif
118 };
119
120 switch (item) {
121 case WarningItem::Background:
122 return colorNamed(@"WKSafeBrowsingWarningBackground");
123 case WarningItem::BoxBackground:
124 return [NSColor windowBackgroundColor];
125 case WarningItem::TitleText:
126 case WarningItem::ExclamationPoint:
127 return colorNamed(@"WKSafeBrowsingWarningTitle");
128 case WarningItem::MessageText:
129 return colorNamed(@"WKSafeBrowsingWarningText");
130 case WarningItem::ShowDetailsButton:
131 case WarningItem::GoBackButton:
132 ASSERT_NOT_REACHED();
133 return nil;
134 }
135#else
136 UIColor *red = [UIColor colorWithRed:0.998 green:0.239 blue:0.233 alpha:1.0];
137 UIColor *white = [UIColor whiteColor];
138 bool narrow = warning.traitCollection.horizontalSizeClass == UIUserInterfaceSizeClassCompact;
139
140 switch (item) {
141 case WarningItem::Background:
142 return red;
143 case WarningItem::BoxBackground:
144 return narrow ? red : white;
145 case WarningItem::TitleText:
146 case WarningItem::ExclamationPoint:
147 return narrow ? white : red;
148 case WarningItem::MessageText:
149 case WarningItem::ShowDetailsButton:
150 return narrow ? white : [UIColor darkTextColor];
151 case WarningItem::GoBackButton:
152 return narrow ? white : warning.tintColor;
153 }
154#endif
155 ASSERT_NOT_REACHED();
156 return nil;
157}

164- (void)drawRect:(RectType)rect
165{
166 constexpr CGFloat centerX = exclamationPointSize / 2;
167 constexpr CGFloat pointCenterY = exclamationPointSize * 7 / 30;
168 constexpr CGFloat pointRadius = 2.25 * exclamationPointSize / 30;
169 constexpr CGFloat lineBottomCenterY = exclamationPointSize * 13 / 30;
170 constexpr CGFloat lineTopCenterY = exclamationPointSize * 23 / 30;
171 constexpr CGFloat lineRadius = 1.75 * exclamationPointSize / 30;
172 ViewType *warning = self.superview.superview;
173#if PLATFORM(MAC)
174 [colorForItem(WarningItem::ExclamationPoint, warning) set];
175 NSBezierPath *exclamationPoint = [NSBezierPath bezierPathWithOvalInRect:NSMakeRect(0, 0, exclamationPointSize, exclamationPointSize)];
176 [exclamationPoint appendBezierPathWithArcWithCenter: { centerX, lineBottomCenterY } radius:lineRadius startAngle:0 endAngle:180 clockwise:YES];
177 [exclamationPoint appendBezierPathWithArcWithCenter: { centerX, lineTopCenterY } radius:lineRadius startAngle:180 endAngle:360 clockwise:YES];
178 [exclamationPoint lineToPoint: { centerX + lineRadius, lineBottomCenterY }];
179 [exclamationPoint appendBezierPathWithArcWithCenter: { centerX, pointCenterY } radius:pointRadius startAngle:0 endAngle:180 clockwise:YES];
180 [exclamationPoint appendBezierPathWithArcWithCenter: { centerX, pointCenterY } radius:pointRadius startAngle:180 endAngle:360 clockwise:YES];
181#else
182 auto flip = [] (auto y) {
183 return exclamationPointSize - y;
184 };
185 [colorForItem(WarningItem::BoxBackground, warning) set];
186 auto square = CGRectMake(0, 0, exclamationPointSize, exclamationPointSize);
187 [[UIBezierPath bezierPathWithRect:square] fill];
188
189 [colorForItem(WarningItem::ExclamationPoint, warning) set];
190 UIBezierPath *exclamationPoint = [UIBezierPath bezierPathWithOvalInRect:square];
191 [exclamationPoint addArcWithCenter: { centerX, flip(lineTopCenterY) } radius:lineRadius startAngle:2 * piDouble endAngle:piDouble clockwise:NO];
192 [exclamationPoint addArcWithCenter: { centerX, flip(lineBottomCenterY) } radius:lineRadius startAngle:piDouble endAngle:0 clockwise:NO];
193 [exclamationPoint addArcWithCenter: { centerX, flip(pointCenterY) } radius:pointRadius startAngle:0 endAngle:piDouble clockwise:NO];
194 [exclamationPoint addArcWithCenter: { centerX, flip(pointCenterY) } radius:pointRadius startAngle:piDouble endAngle:piDouble * 2 clockwise:NO];
195 [exclamationPoint addLineToPoint: { centerX + lineRadius, flip(lineBottomCenterY) }];
196 [exclamationPoint addLineToPoint: { centerX + lineRadius, flip(lineTopCenterY) }];
197#endif
198 [exclamationPoint fill];
199}

215static ButtonType *makeButton(WarningItem item, WKSafeBrowsingWarning *warning, SEL action)
216{
217 NSString *title = nil;
218 if (item == WarningItem::ShowDetailsButton)
219 title = WEB_UI_NSSTRING(@"Show Details", "Action from safe browsing warning");
220 else
221 title = WEB_UI_NSSTRING(@"Go Back", "Action from safe browsing warning");
222#if PLATFORM(MAC)
223 return [NSButton buttonWithTitle:title target:warning action:action];
224#else
225 UIButton *button = [UIButton buttonWithType:UIButtonTypeSystem];
226 auto attributedTitle = adoptNS([[NSAttributedString alloc] initWithString:title attributes:@{
227 NSUnderlineStyleAttributeName:@(NSUnderlineStyleSingle),
228 NSUnderlineColorAttributeName:[UIColor whiteColor],
229 NSForegroundColorAttributeName:colorForItem(item, warning),
230 NSFontAttributeName:fontOfSize(WarningTextSize::Body)
231 }]);
232 [button setAttributedTitle:attributedTitle.get() forState:UIControlStateNormal];
233 [button addTarget:warning action:action forControlEvents:UIControlEventTouchUpInside];
234 return button;
235#endif
236}

293- (instancetype)initWithFrame:(RectType)frame safeBrowsingWarning:(const WebKit::SafeBrowsingWarning&)warning completionHandler:(CompletionHandler<void(Variant<WebKit::ContinueUnsafeLoad, URL>&&)>&&)completionHandler
294{
295 if (!(self = [super initWithFrame:frame])) {
296 completionHandler(WebKit::ContinueUnsafeLoad::Yes);
297 return nil;
298 }
299 _completionHandler = [weakSelf = WeakObjCPtr<WKSafeBrowsingWarning>(self), completionHandler = WTFMove(completionHandler)] (Variant<WebKit::ContinueUnsafeLoad, URL>&& result) mutable {
300#if PLATFORM(WATCHOS)
301 if (auto strongSelf = weakSelf.get())
302 [strongSelf.get()->_previousFirstResponder becomeFirstResponder];
303#endif
304 completionHandler(WTFMove(result));
305 };
306 _warning = makeRef(warning);
307#if PLATFORM(MAC)
308 [self setSafeBrowsingBackgroundColor:colorForItem(WarningItem::Background, self)];
309 [self addContent];
310#else
311 [self setBackgroundColor:colorForItem(WarningItem::Background, self)];
312#endif
313
314#if PLATFORM(WATCHOS)
315 self.crownInputScrollDirection = PUICCrownInputScrollDirectionVertical;
316#endif
317 return self;
318}

320- (void)addContent
321{
322 auto exclamationPoint = adoptNS([WKSafeBrowsingExclamationPoint new]);
323 auto title = makeLabel(adoptNS([[NSAttributedString alloc] initWithString:_warning->title() attributes:@{
324 NSFontAttributeName:fontOfSize(WarningTextSize::Title),
325 NSForegroundColorAttributeName:colorForItem(WarningItem::TitleText, self)
326#if PLATFORM(WATCHOS)
327 , NSHyphenationFactorDocumentAttribute:@1
328#endif
329 }]).get());
330 auto warning = makeLabel(adoptNS([[NSAttributedString alloc] initWithString:_warning->warning() attributes:@{
331 NSFontAttributeName:fontOfSize(WarningTextSize::Body),
332 NSForegroundColorAttributeName:colorForItem(WarningItem::MessageText, self)
333#if PLATFORM(WATCHOS)
334 , NSHyphenationFactorDocumentAttribute:@1
335#endif
336 }]).get());
337 auto showDetails = makeButton(WarningItem::ShowDetailsButton, self, @selector(showDetailsClicked));
338 auto goBack = makeButton(WarningItem::GoBackButton, self, @selector(goBackClicked));
339 auto box = adoptNS([WKSafeBrowsingBox new]);
340 _box = box.get();
341 [box setSafeBrowsingBackgroundColor:colorForItem(WarningItem::BoxBackground, self)];
342 [box layer].cornerRadius = boxCornerRadius;
343
344 for (ViewType *view in @[exclamationPoint.get(), title.get(), warning.get(), goBack, showDetails]) {
345 view.translatesAutoresizingMaskIntoConstraints = NO;
346 [box addSubview:view];
347 }
348 [box setTranslatesAutoresizingMaskIntoConstraints:NO];
349 [self addSubview:box.get()];
350
351#if PLATFORM(WATCHOS)
352 [NSLayoutConstraint activateConstraints:@[
353 [[[box leadingAnchor] anchorWithOffsetToAnchor:[exclamationPoint leadingAnchor]] constraintEqualToAnchor:[[exclamationPoint trailingAnchor] anchorWithOffsetToAnchor:[box trailingAnchor]]],
354 [[[box leadingAnchor] anchorWithOffsetToAnchor:[title leadingAnchor]] constraintEqualToConstant:marginSize],
355 [[[title bottomAnchor] anchorWithOffsetToAnchor:[warning topAnchor]] constraintEqualToConstant:marginSize],
356 [[[exclamationPoint bottomAnchor] anchorWithOffsetToAnchor:[title topAnchor]] constraintEqualToConstant:marginSize],
357 [[[box topAnchor] anchorWithOffsetToAnchor:[exclamationPoint topAnchor]] constraintEqualToConstant:marginSize + self.frame.size.height / 2],
358 [[self.topAnchor anchorWithOffsetToAnchor:[box topAnchor]] constraintEqualToAnchor:[[box bottomAnchor] anchorWithOffsetToAnchor:self.bottomAnchor] multiplier:0.2],
359 ]];
360#elif HAVE(SAFE_BROWSING)
361 [NSLayoutConstraint activateConstraints:@[
362 [[[box leadingAnchor] anchorWithOffsetToAnchor:[exclamationPoint leadingAnchor]] constraintEqualToConstant:marginSize],
363 [[[box leadingAnchor] anchorWithOffsetToAnchor:[title leadingAnchor]] constraintEqualToConstant:marginSize * 1.5 + exclamationPointSize],
364 [[[title topAnchor] anchorWithOffsetToAnchor:[exclamationPoint topAnchor]] constraintEqualToAnchor:[[exclamationPoint bottomAnchor] anchorWithOffsetToAnchor:[title bottomAnchor]]],
365 [[[title bottomAnchor] anchorWithOffsetToAnchor:[warning topAnchor]] constraintEqualToConstant:marginSize],
366 [[[box topAnchor] anchorWithOffsetToAnchor:[title topAnchor]] constraintEqualToConstant:marginSize],
367 [[self.topAnchor anchorWithOffsetToAnchor:[box topAnchor]] constraintEqualToAnchor:[[box bottomAnchor] anchorWithOffsetToAnchor:self.bottomAnchor] multiplier:0.5],
368 ]];
369#endif
370
371#if HAVE(SAFE_BROWSING)
372 [NSLayoutConstraint activateConstraints:@[
373 [[self.leftAnchor anchorWithOffsetToAnchor:[box leftAnchor]] constraintEqualToAnchor:[[box rightAnchor] anchorWithOffsetToAnchor:self.rightAnchor]],
374
375 [[box widthAnchor] constraintLessThanOrEqualToConstant:maxWidth],
376 [[box widthAnchor] constraintLessThanOrEqualToAnchor:self.widthAnchor],
377
378 [[[box leadingAnchor] anchorWithOffsetToAnchor:[warning leadingAnchor]] constraintEqualToConstant:marginSize],
379
380 [[[title trailingAnchor] anchorWithOffsetToAnchor:[box trailingAnchor]] constraintGreaterThanOrEqualToConstant:marginSize],
381 [[[warning trailingAnchor] anchorWithOffsetToAnchor:[box trailingAnchor]] constraintGreaterThanOrEqualToConstant:marginSize],
382 [[goBack.trailingAnchor anchorWithOffsetToAnchor:[box trailingAnchor]] constraintEqualToConstant:marginSize],
383
384 [[[warning bottomAnchor] anchorWithOffsetToAnchor:goBack.topAnchor] constraintEqualToConstant:marginSize],
385 ]];
386
387 bool needsVerticalButtonLayout = buttonSize(showDetails).width + buttonSize(goBack).width + 3 * marginSize > self.frame.size.width;
388 if (needsVerticalButtonLayout) {
389 [NSLayoutConstraint activateConstraints:@[
390 [[showDetails.trailingAnchor anchorWithOffsetToAnchor:[box trailingAnchor]] constraintEqualToConstant:marginSize],
391 [[goBack.bottomAnchor anchorWithOffsetToAnchor:showDetails.topAnchor] constraintEqualToConstant:marginSize],
392 [[goBack.bottomAnchor anchorWithOffsetToAnchor:[box bottomAnchor]] constraintEqualToConstant:marginSize * 2 + buttonSize(showDetails).height],
393 ]];
394 } else {
395 [NSLayoutConstraint activateConstraints:@[
396 [[showDetails.trailingAnchor anchorWithOffsetToAnchor:goBack.leadingAnchor] constraintEqualToConstant:marginSize],
397 [goBack.topAnchor constraintEqualToAnchor:showDetails.topAnchor],
398 [[goBack.bottomAnchor anchorWithOffsetToAnchor:[box bottomAnchor]] constraintEqualToConstant:marginSize],
399 ]];
400 }
401#if !PLATFORM(MAC)
402 [self updateContentSize];
403#endif
404#endif
405
406#if PLATFORM(WATCHOS)
407 self->_previousFirstResponder = [self firstResponder];
408 [self becomeFirstResponder];
409#endif
410}

412- (void)showDetailsClicked
413{
414 WKSafeBrowsingBox *box = _box.get().get();
415 ButtonType *showDetails = box.subviews.lastObject;
416 [showDetails removeFromSuperview];
417
418 auto text = adoptNS([_warning->details() mutableCopy]);
419 [text addAttributes:@{ NSFontAttributeName:fontOfSize(WarningTextSize::Body) } range:NSMakeRange(0, [text length])];
420 auto details = adoptNS([[WKSafeBrowsingTextView alloc] initWithAttributedString:text.get() forWarning:self]);
421 _details = details.get();
422 auto bottom = adoptNS([WKSafeBrowsingBox new]);
423 [bottom setSafeBrowsingBackgroundColor:colorForItem(WarningItem::BoxBackground, self)];
424 [bottom layer].cornerRadius = boxCornerRadius;
425
426#if HAVE(SAFE_BROWSING)
427 constexpr auto maxY = kCALayerMinXMaxYCorner | kCALayerMaxXMaxYCorner;
428 constexpr auto minY = kCALayerMinXMinYCorner | kCALayerMaxXMinYCorner;
429#if PLATFORM(MAC)
430 box.layer.maskedCorners = maxY;
431 [bottom layer].maskedCorners = minY;
432#else
433 box.layer.maskedCorners = minY;
434 [bottom layer].maskedCorners = maxY;
435#endif
436#endif
437
438 auto line = adoptNS([WKSafeBrowsingBox new]);
439 [line setSafeBrowsingBackgroundColor:[CocoaColor lightGrayColor]];
440 for (ViewType *view in @[details.get(), bottom.get(), line.get()])
441 view.translatesAutoresizingMaskIntoConstraints = NO;
442
443 [self addSubview:bottom.get()];
444 [bottom addSubview:line.get()];
445 [bottom addSubview:details.get()];
446#if HAVE(SAFE_BROWSING)
447 [NSLayoutConstraint activateConstraints:@[
448 [box.widthAnchor constraintEqualToAnchor:[bottom widthAnchor]],
449 [box.bottomAnchor constraintEqualToAnchor:[bottom topAnchor]],
450 [box.leadingAnchor constraintEqualToAnchor:[bottom leadingAnchor]],
451 [[line widthAnchor] constraintEqualToAnchor:[bottom widthAnchor]],
452 [[line leadingAnchor] constraintEqualToAnchor:[bottom leadingAnchor]],
453 [[line topAnchor] constraintEqualToAnchor:[bottom topAnchor]],
454 [[line heightAnchor] constraintEqualToConstant:1],
455 [[[bottom topAnchor] anchorWithOffsetToAnchor:[details topAnchor]] constraintEqualToConstant:marginSize],
456 [[[details bottomAnchor] anchorWithOffsetToAnchor:[bottom bottomAnchor]] constraintEqualToConstant:marginSize],
457 [[[bottom leadingAnchor] anchorWithOffsetToAnchor:[details leadingAnchor]] constraintEqualToConstant:marginSize],
458 [[[details trailingAnchor] anchorWithOffsetToAnchor:[bottom trailingAnchor]] constraintEqualToConstant:marginSize],
459 ]];
460#endif
461 [self layoutText];
462#if !PLATFORM(MAC)
463 [self updateContentSize];
464#endif
465}

529- (void)clickedOnLink:(NSURL *)link
530{
531 if (!_completionHandler)
532 return;
533
534 if ([link isEqual:WebKit::SafeBrowsingWarning::visitUnsafeWebsiteSentinel()])
535 return _completionHandler(WebKit::ContinueUnsafeLoad::Yes);
536
537 if ([link isEqual:WebKit::SafeBrowsingWarning::confirmMalwareSentinel()]) {
538#if PLATFORM(MAC)
539 auto alert = adoptNS([NSAlert new]);
540 [alert setMessageText:WEB_UI_NSSTRING(@"Are you sure you wish to go to this site?", "Malware confirmation dialog title")];
541 [alert setInformativeText:WEB_UI_NSSTRING(@"Merely visiting a site is sufficient for malware to install itself and harm your computer.", "Malware confirmation dialog")];
542 [alert addButtonWithTitle:WEB_UI_NSSTRING(@"Cancel", "Cancel")];
543 [alert addButtonWithTitle:WEB_UI_NSSTRING(@"Continue", "Continue")];
544 [alert beginSheetModalForWindow:self.window completionHandler:makeBlockPtr([weakSelf = WeakObjCPtr<WKSafeBrowsingWarning>(self), alert](NSModalResponse returnCode) {
545 if (auto strongSelf = weakSelf.get()) {
546 if (returnCode == NSAlertSecondButtonReturn && strongSelf->_completionHandler)
547 strongSelf->_completionHandler(WebKit::ContinueUnsafeLoad::Yes);
548 }
549 }).get()];
550#else
551 _completionHandler(WebKit::ContinueUnsafeLoad::Yes);
552#endif
553 return;
554 }
555
556 ASSERT([link isKindOfClass:[NSURL class]]);
557 _completionHandler((NSURL *)link);
558}

569- (instancetype)initWithAttributedString:(NSAttributedString *)attributedString forWarning:(WKSafeBrowsingWarning *)warning
570{
571 if (!(self = [super init]))
572 return nil;
573 self->_warning = warning;
574 self.delegate = warning;
575
576 CocoaColor *foregroundColor = colorForItem(WarningItem::MessageText, warning);
577 auto string = adoptNS([attributedString mutableCopy]);
578 [string addAttributes:@{ NSForegroundColorAttributeName : foregroundColor } range:NSMakeRange(0, [string length])];
579 [self setBackgroundColor:colorForItem(WarningItem::BoxBackground, warning)];
580 [self setLinkTextAttributes:@{ NSForegroundColorAttributeName : foregroundColor }];
581 [self.textStorage appendAttributedString:string.get()];
582 self.editable = NO;
583#if !PLATFORM(MAC)
584 self.scrollEnabled = NO;
585#endif
586
587 return self;
588}

590- (SizeType)intrinsicContentSize
591{
592#if PLATFORM(MAC)
593 [self.layoutManager ensureLayoutForTextContainer:self.textContainer];
594 return { NSViewNoIntrinsicMetric, [self.layoutManager usedRectForTextContainer:self.textContainer].size.height };
595#elif HAVE(SAFE_BROWSING)
596 auto width = std::min<CGFloat>(maxWidth, [_warning frame].size.width) - 2 * marginSize;
597 constexpr auto noHeightConstraint = CGFLOAT_MAX;
598 return { width, [self sizeThatFits: { width, noHeightConstraint }].height };
599#else
600 return { 0, 0 };
601#endif
602}
