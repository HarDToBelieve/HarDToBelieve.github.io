26#import "config.h"
27#import "WebInspectorPreferenceObserver.h"
28
29#import "WebProcessPool.h"
30
31#import <wtf/RetainPtr.h>
32
33@interface WKWebInspectorPreferenceObserver ()
34{
35@private
36 RetainPtr<NSUserDefaults> m_userDefaults;
37}
38@end
39
40@implementation WKWebInspectorPreferenceObserver
41
42+ (id)sharedInstance
43{
44 static NeverDestroyed<RetainPtr<WKWebInspectorPreferenceObserver>> instance = adoptNS([[[self class] alloc] init]);
45 return instance.get().get();
46}
47
48- (instancetype)init
49{
50 if (!(self = [super init]))
51 return nil;
52
53 m_userDefaults = adoptNS([[NSUserDefaults alloc] initWithSuiteName:@"com.apple.Safari.SandboxBroker"]);
54 if (!m_userDefaults) {
55 WTFLogAlways("Could not init user defaults instance for domain com.apple.Safari.SandboxBroker.");
56 return self;
57 }
58 [m_userDefaults.get() addObserver:self forKeyPath:@"ShowDevelopMenu" options:NSKeyValueObservingOptionNew context:nil];
59
60
61 return self;
62}
63
64- (void)observeValueForKeyPath:(NSString *)keyPath ofObject:(id)object change:(NSDictionary<NSKeyValueChangeKey, id> *)change context:(void *)context
65{
66 RunLoop::main().dispatch([] {
67 for (auto& pool : WebKit::WebProcessPool::allProcessPools()) {
68 for (size_t i = 0; i < pool->processes().size(); ++i) {
69 auto process = pool->processes()[i];
70 process->enableRemoteInspectorIfNeeded();
71 }
72 }
73 });
74}
75
76@end
