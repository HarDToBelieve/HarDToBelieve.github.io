29namespace WebKit {

31extern void* WebKitSwiftLibrary(bool isOptional = false);
32void* WebKitSwiftLibrary(bool isOptional)
33{
34 static void* library;
35 static dispatch_once_t once;
36 dispatch_once(&once, ^{
37 // Start by searching for the library in DYLD_LIBRARY_PATH:
38 if ((library = dlopen("libWebKitSwift.dylib", RTLD_NOW)))
39 return;
40
41 // Then search in the Frameworks/ directory of the currently loaded version of WebKit.framework:
42 Dl_info info { };
43 if (dladdr((const void*)&WebKitSwiftLibrary, &info) && strlen(info.dli_fname)) {
44 auto webkitFrameworkDirectory = WTF::FileSystemImpl::parentPath(info.dli_fname);
45 auto dylibPath = WTF::FileSystemImpl::pathByAppendingComponent(webkitFrameworkDirectory, "Frameworks/libWebKitSwift.dylib");
46 if ((library = dlopen(dylibPath.utf8().data(), RTLD_NOW)))
47 return;
48 }
49
50 if (!isOptional)
51 RELEASE_ASSERT_WITH_MESSAGE(library, "%s", dlerror());
52 });
53 return library;
54}
55
56}
