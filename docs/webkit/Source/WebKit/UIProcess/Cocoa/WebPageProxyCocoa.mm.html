131void WebPageProxy::beginSafeBrowsingCheck(const URL& url, bool forMainFrameNavigation, WebFramePolicyListenerProxy& listener)
132{
133#if HAVE(SAFE_BROWSING)
134 SSBLookupContext *context = [SSBLookupContext sharedLookupContext];
135 if (!context)
136 return listener.didReceiveSafeBrowsingResults({ });
137 [context lookUpURL:url completionHandler:makeBlockPtr([listener = makeRef(listener), forMainFrameNavigation, url = url] (SSBLookupResult *result, NSError *error) mutable {
138 RunLoop::main().dispatch([listener = WTFMove(listener), result = retainPtr(result), error = retainPtr(error), forMainFrameNavigation, url = WTFMove(url)] {
139 if (error) {
140 listener->didReceiveSafeBrowsingResults({ });
141 return;
142 }
143
144 for (SSBServiceLookupResult *lookupResult in [result serviceLookupResults]) {
145 if (lookupResult.isPhishing || lookupResult.isMalware || lookupResult.isUnwantedSoftware) {
146 listener->didReceiveSafeBrowsingResults(SafeBrowsingWarning::create(url, forMainFrameNavigation, lookupResult));
147 return;
148 }
149 }
150 listener->didReceiveSafeBrowsingResults({ });
151 });
152 }).get()];
153#else
154 listener.didReceiveSafeBrowsingResults({ });
155#endif
156}

192void WebPageProxy::createSandboxExtensionsIfNeeded(const Vector<String>& files, SandboxExtension::Handle& fileReadHandle, SandboxExtension::HandleArray& fileUploadHandles)
193{
194 if (!files.size())
195 return;
196
197 if (files.size() == 1) {
198 BOOL isDirectory;
199 if ([[NSFileManager defaultManager] fileExistsAtPath:files[0] isDirectory:&isDirectory] && !isDirectory) {
200 ASSERT(process().connection() && process().connection()->getAuditToken());
201 if (process().connection() && process().connection()->getAuditToken())
202 SandboxExtension::createHandleForReadByAuditToken("/", *(process().connection()->getAuditToken()), fileReadHandle);
203 else
204 SandboxExtension::createHandle("/", SandboxExtension::Type::ReadOnly, fileReadHandle);
205 willAcquireUniversalFileReadSandboxExtension(m_process);
206 }
207 }
208
209 fileUploadHandles.allocate(files.size());
210 for (size_t i = 0; i< files.size(); i++) {
211 NSString *file = files[i];
212 if (![[NSFileManager defaultManager] fileExistsAtPath:file])
213 continue;
214 SandboxExtension::createHandle(file, SandboxExtension::Type::ReadOnly, fileUploadHandles[i]);
215 }
216}

296void WebPageProxy::insertDictatedTextAsync(const String& text, const EditingRange& replacementRange, const Vector<TextAlternativeWithRange>& dictationAlternativesWithRange, InsertTextOptions&& options)
297{
298 if (!hasRunningProcess())
299 return;
300
301 Vector<DictationAlternative> dictationAlternatives;
302 for (const auto& alternativeWithRange : dictationAlternativesWithRange) {
303 if (auto context = pageClient().addDictationAlternatives(alternativeWithRange.alternatives.get()))
304 dictationAlternatives.append({ alternativeWithRange.range, context });
305 }
306
307 if (dictationAlternatives.isEmpty()) {
308 insertTextAsync(text, replacementRange, WTFMove(options));
309 return;
310 }
311
312 send(Messages::WebPage::InsertDictatedTextAsync { text, replacementRange, dictationAlternatives, WTFMove(options) });
313}

455{
456 FloatSize originalThumbnailSize([image size]);
457 auto resultRect = roundedIntRect(largestRectWithAspectRatioInsideRect(originalThumbnailSize.aspectRatio(), { { }, fittingSize }));
458 resultRect.setLocation({ });
459
460 WebKit::ShareableBitmap::Configuration bitmapConfiguration;
461 auto bitmap = WebKit::ShareableBitmap::createShareable(resultRect.size(), bitmapConfiguration);
462 if (!bitmap)
463 return nullptr;
464
465 auto graphicsContext = bitmap->createGraphicsContext();
466 if (!graphicsContext)

469 LocalCurrentGraphicsContext savedContext(*graphicsContext);
470#if PLATFORM(IOS_FAMILY)

472#elif USE(APPKIT)
473 [image drawInRect:resultRect fromRect:NSZeroRect operation:NSCompositingOperationSourceOver fraction:1 respectFlipped:YES hints:nil];
474#endif
475
476 return bitmap;
477}

510void WebPageProxy::scheduleActivityStateUpdate()
511{
512 bool hasScheduledObserver = m_activityStateChangeDispatcher->isScheduled();
513 bool hasActiveCATransaction = [CATransaction currentState];
514
515 if (hasScheduledObserver && hasActiveCATransaction) {
516 ASSERT(m_hasScheduledActivityStateUpdate);
517 m_hasScheduledActivityStateUpdate = false;
518 m_activityStateChangeDispatcher->invalidate();
519 }
520
521 if (m_hasScheduledActivityStateUpdate)
522 return;
523 m_hasScheduledActivityStateUpdate = true;
524
525 // If there is an active transaction, we need to dispatch the update after the transaction is committed,
526 // to avoid flash caused by web process setting root layer too early.
527 // If there is no active transaction, likely there is no root layer change or change is committed,
528 // then schedule dispatch on runloop observer to collect changes in the same runloop cycle before dispatching.
529 if (hasActiveCATransaction) {
530 [CATransaction addCommitHandler:[weakThis = makeWeakPtr(*this)] {
531 // We can't call dispatchActivityStateChange directly underneath this commit handler, because it has side-effects
532 // that may result in other frameworks trying to install commit handlers for the same phase, which is not allowed.
533 // So, dispatch_async here; we only care that the activity state change doesn't apply until after the active commit is complete.
534 WorkQueue::main().dispatch([weakThis] {
535 auto protectedThis = makeRefPtr(weakThis.get());
536 if (!protectedThis)
537 return;
538
539 protectedThis->dispatchActivityStateChange();
540 });
541 } forPhase:kCATransactionPhasePostCommit];
542 return;
543 }
544
545 m_activityStateChangeDispatcher->schedule();
546}
547
548void WebPageProxy::addActivityStateUpdateCompletionHandler(CompletionHandler<void()>&& completionHandler)
549{
550 if (!m_hasScheduledActivityStateUpdate) {
551 completionHandler();
552 return;
553 }
554
555 m_activityStateUpdateCallbacks.append(WTFMove(completionHandler));
556}

606{

608 return;

612 ensureOnMainRunLoop([weakThis, isVisible] {
