61void WebPasteboardProxy::grantAccess(WebProcessProxy& process, const String& pasteboardName, PasteboardAccessType type)
62{
63 if (!m_webProcessProxySet.contains(process))
64 return;
65
66 if (pasteboardName.isEmpty()) {
67 ASSERT_NOT_REACHED();
68 return;
69 }
70
71 auto changeCount = PlatformPasteboard(pasteboardName).changeCount();
72 auto changeCountsAndProcesses = m_pasteboardNameToAccessInformationMap.find(pasteboardName);
73 if (changeCountsAndProcesses != m_pasteboardNameToAccessInformationMap.end() && changeCountsAndProcesses->value.changeCount == changeCount) {
74 changeCountsAndProcesses->value.grantAccess(process, type);
75 return;
76 }
77
78 m_pasteboardNameToAccessInformationMap.set(pasteboardName, PasteboardAccessInformation { changeCount, {{ makeWeakPtr(process), type }} });
79}

81void WebPasteboardProxy::revokeAccess(WebProcessProxy& process)
82{
83 for (auto& changeCountAndProcesses : m_pasteboardNameToAccessInformationMap.values())
84 changeCountAndProcesses.revokeAccess(process);
85}

98std::optional<WebPasteboardProxy::PasteboardAccessType> WebPasteboardProxy::accessType(IPC::Connection& connection, const String& pasteboardName) const
99{
100 MESSAGE_CHECK_WITH_RETURN_VALUE(!pasteboardName.isEmpty(), std::nullopt);
101
102 auto* process = webProcessProxyForConnection(connection);
103 MESSAGE_CHECK_WITH_RETURN_VALUE(process, std::nullopt);
104
105 for (auto* page : process->pages()) {
106 auto& preferences = page->preferences();
107 if (!preferences.domPasteAllowed() || !preferences.javaScriptCanAccessClipboard())
108 continue;
109
110 // If a web page already allows JavaScript to programmatically read pasteboard data,
111 // then it is possible for any other web page residing in the same web process to send
112 // IPC messages that can read pasteboard data by pretending to be the page that has
113 // allowed unmitigated pasteboard access from script. As such, there is no security
114 // benefit in limiting the scope of pasteboard data access to only the web page that
115 // enables programmatic pasteboard access.
116 return PasteboardAccessType::TypesAndData;
117 }
118
119 auto changeCountAndProcesses = m_pasteboardNameToAccessInformationMap.find(pasteboardName);
120 if (changeCountAndProcesses == m_pasteboardNameToAccessInformationMap.end())
121 return std::nullopt;
122
123 auto& information = changeCountAndProcesses->value;
124 if (information.changeCount != PlatformPasteboard(pasteboardName).changeCount())
125 return std::nullopt;
126
127 return information.accessType(*process);
128}

130void WebPasteboardProxy::didModifyContentsOfPasteboard(IPC::Connection& connection, const String& pasteboardName, int64_t previousChangeCount, int64_t newChangeCount)
131{
132 auto* process = webProcessProxyForConnection(connection);
133 MESSAGE_CHECK(process);
134
135 auto changeCountAndProcesses = m_pasteboardNameToAccessInformationMap.find(pasteboardName);
136 if (changeCountAndProcesses != m_pasteboardNameToAccessInformationMap.end() && previousChangeCount == changeCountAndProcesses->value.changeCount) {
137 if (auto accessType = changeCountAndProcesses->value.accessType(*process))
138 changeCountAndProcesses->value = PasteboardAccessInformation { newChangeCount, {{ makeWeakPtr(*process), *accessType }} };
139 }
140}

157void WebPasteboardProxy::getPasteboardPathnamesForType(IPC::Connection& connection, const String& pasteboardName, const String& pasteboardType, std::optional<PageIdentifier> pageID,
158 CompletionHandler<void(Vector<String>&& pathnames, SandboxExtension::HandleArray&& sandboxExtensions)>&& completionHandler)
159{
160 MESSAGE_CHECK_COMPLETION(!pasteboardType.isEmpty(), completionHandler({ }, { }));
161
162 // FIXME: This should consult canAccessPasteboardData() instead, and avoid responding with file paths if it returns false.
163 if (!canAccessPasteboardTypes(connection, pasteboardName))
164 return completionHandler({ }, { });
165
166 auto dataOwner = determineDataOwner(connection, pasteboardName, pageID, PasteboardAccessIntent::Read);
167 MESSAGE_CHECK_COMPLETION(dataOwner, completionHandler({ }, { }));
168
169 PlatformPasteboard::performAsDataOwner(*dataOwner, [&] {
170 Vector<String> pathnames;
171 SandboxExtension::HandleArray sandboxExtensions;
172 if (webProcessProxyForConnection(connection)) {
173 PlatformPasteboard(pasteboardName).getPathnamesForType(pathnames, pasteboardType);
174#if PLATFORM(MAC)
175 // On iOS, files are copied into app's container upon paste.
176 sandboxExtensions.allocate(pathnames.size());
177 for (size_t i = 0; i < pathnames.size(); i++) {
178 auto& filename = pathnames[i];
179 if (![[NSFileManager defaultManager] fileExistsAtPath:filename])
180 continue;
181 SandboxExtension::createHandle(filename, SandboxExtension::Type::ReadOnly, sandboxExtensions[i]);
182 }
183#endif
184 }
185 completionHandler(WTFMove(pathnames), WTFMove(sandboxExtensions));
186 });
187}

219void WebPasteboardProxy::getPasteboardBufferForType(IPC::Connection& connection, const String& pasteboardName, const String& pasteboardType, std::optional<PageIdentifier> pageID, CompletionHandler<void(SharedMemory::IPCHandle&&)>&& completionHandler)
220{
221 MESSAGE_CHECK_COMPLETION(!pasteboardType.isEmpty(), completionHandler({ }));
222
223 if (!canAccessPasteboardData(connection, pasteboardName))
224 return completionHandler({ });
225
226 auto dataOwner = determineDataOwner(connection, pasteboardName, pageID, PasteboardAccessIntent::Read);
227 MESSAGE_CHECK_COMPLETION(dataOwner, completionHandler({ }));
228
229 PlatformPasteboard::performAsDataOwner(*dataOwner, [&] {
230 auto buffer = PlatformPasteboard(pasteboardName).bufferForType(pasteboardType);
231 if (!buffer)
232 return completionHandler({ });
233 uint64_t size = buffer->size();
234 if (!size)
235 return completionHandler({ });
236 RefPtr<SharedMemory> sharedMemoryBuffer = SharedMemory::allocate(size);
237 if (!sharedMemoryBuffer)
238 return completionHandler({ });
239 memcpy(sharedMemoryBuffer->data(), buffer->data(), size);
240 SharedMemory::Handle handle;
241 if (!sharedMemoryBuffer->createHandle(handle, SharedMemory::Protection::ReadOnly))
242 return completionHandler({ });
243 completionHandler(SharedMemory::IPCHandle { WTFMove(handle), size });
244 });
245}

404void WebPasteboardProxy::setPasteboardBufferForType(IPC::Connection& connection, const String& pasteboardName, const String& pasteboardType, const SharedMemory::IPCHandle& ipcHandle, std::optional<PageIdentifier> pageID, CompletionHandler<void(int64_t)>&& completionHandler)
405{
406 MESSAGE_CHECK_COMPLETION(!pasteboardName.isEmpty(), completionHandler(0));
407 MESSAGE_CHECK_COMPLETION(!pasteboardType.isEmpty(), completionHandler(0));
408
409 auto dataOwner = determineDataOwner(connection, pasteboardName, pageID, PasteboardAccessIntent::Write);
410 MESSAGE_CHECK_COMPLETION(dataOwner, completionHandler(0));
411
412 PlatformPasteboard::performAsDataOwner(*dataOwner, [&] {
413 auto previousChangeCount = PlatformPasteboard(pasteboardName).changeCount();
414 if (ipcHandle.handle.isNull()) {
415 auto newChangeCount = PlatformPasteboard(pasteboardName).setBufferForType(nullptr, pasteboardType);
416 didModifyContentsOfPasteboard(connection, pasteboardName, previousChangeCount, newChangeCount);
417 return completionHandler(newChangeCount);
418 }
419
420 auto sharedMemoryBuffer = SharedMemory::map(ipcHandle.handle, SharedMemory::Protection::ReadOnly);
421 if (!sharedMemoryBuffer)
422 return completionHandler(0);
423 auto buffer = SharedBuffer::create(static_cast<unsigned char *>(sharedMemoryBuffer->data()), static_cast<size_t>(ipcHandle.dataSize));
424 auto newChangeCount = PlatformPasteboard(pasteboardName).setBufferForType(buffer.ptr(), pasteboardType);
425 didModifyContentsOfPasteboard(connection, pasteboardName, previousChangeCount, newChangeCount);
426 completionHandler(newChangeCount);
427 });
428}

542void WebPasteboardProxy::readBufferFromPasteboard(IPC::Connection& connection, size_t index, const String& pasteboardType, const String& pasteboardName, std::optional<PageIdentifier> pageID, CompletionHandler<void(SharedMemory::IPCHandle&&)>&& completionHandler)
543{

545

547 return completionHandler({ });
548
549 auto dataOwner = determineDataOwner(connection, pasteboardName, pageID, PasteboardAccessIntent::Read);
550 MESSAGE_CHECK_COMPLETION(dataOwner, completionHandler({ }));
551
552 PlatformPasteboard::performAsDataOwner(*dataOwner, [&] {
553 auto buffer = PlatformPasteboard(pasteboardName).readBuffer(index, pasteboardType);
554 if (!buffer)
555 return completionHandler({ });
556 uint64_t size = buffer->size();
557 if (!size)
558 return completionHandler({ });
559 RefPtr<SharedMemory> sharedMemoryBuffer = SharedMemory::allocate(size);
560 if (!sharedMemoryBuffer)
561 return completionHandler({ });
562 memcpy(sharedMemoryBuffer->data(), buffer->data(), size);
563 SharedMemory::Handle handle;
564 if (!sharedMemoryBuffer->createHandle(handle, SharedMemory::Protection::ReadOnly))
565 return completionHandler({ });
566 completionHandler(SharedMemory::IPCHandle { WTFMove(handle), size });
567 });
568}

657std::optional<DataOwnerType> WebPasteboardProxy::determineDataOwner(IPC::Connection& connection, const String& pasteboardName, std::optional<PageIdentifier> pageID, PasteboardAccessIntent intent) const
658{
659 MESSAGE_CHECK_WITH_RETURN_VALUE(!pasteboardName.isEmpty(), std::nullopt);
660
661 auto* process = webProcessProxyForConnection(connection);
662 MESSAGE_CHECK_WITH_RETURN_VALUE(process, std::nullopt);
663
664 if (!pageID)
665 return DataOwnerType::Undefined;
666
667#if HAVE(PASTEBOARD_DATA_OWNER)
668 std::optional<DataOwnerType> result;
669 for (auto* page : process->pages()) {
670 if (page->webPageID() == *pageID) {
671 result = page->dataOwnerForPasteboard(intent);
672 break;
673 }
674 }
675 // If this message check is hit, then the incoming web page ID doesn't correspond to any page
676 // currently known to the UI process.
677 MESSAGE_CHECK_WITH_RETURN_VALUE(result.has_value(), std::nullopt);
678 return result;
679#else
680 UNUSED_PARAM(intent);
681 return DataOwnerType::Undefined;
682#endif
683}

685void WebPasteboardProxy::PasteboardAccessInformation::grantAccess(WebProcessProxy& process, PasteboardAccessType type)
686{
687 auto matchIndex = processes.findMatching([&](auto& processAndType) {
688 return processAndType.first == &process;
689 });
690
691 if (matchIndex == notFound) {
692 processes.append({ makeWeakPtr(process), type });
693 return;
694 }
695
696 if (type == PasteboardAccessType::TypesAndData)
697 processes[matchIndex].second = type;
698
699 processes.removeAllMatching([](auto& processAndType) {
700 return !processAndType.first;
701 });
702}
703
704void WebPasteboardProxy::PasteboardAccessInformation::revokeAccess(WebProcessProxy& process)
705{
706 processes.removeFirstMatching([&](auto& processAndType) {
707 return processAndType.first == &process;
708 });
709}

711std::optional<WebPasteboardProxy::PasteboardAccessType> WebPasteboardProxy::PasteboardAccessInformation::accessType(WebProcessProxy& process) const
712{
713 auto matchIndex = processes.findMatching([&](auto& processAndType) {
714 return processAndType.first == &process;
715 });
716
717 if (matchIndex == notFound)
718 return std::nullopt;
719
720 return processes[matchIndex].second;
