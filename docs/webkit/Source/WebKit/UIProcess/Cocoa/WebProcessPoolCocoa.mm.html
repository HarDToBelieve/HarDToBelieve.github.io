133static void registerUserDefaultsIfNeeded()
134{
135 static bool didRegister;
136 if (didRegister)
137 return;
138
139 didRegister = true;
140 NSMutableDictionary *registrationDictionary = [NSMutableDictionary dictionary];
141
142 [registrationDictionary setObject:@YES forKey:WebKitJSCJITEnabledDefaultsKey];
143 [registrationDictionary setObject:@YES forKey:WebKitJSCFTLJITEnabledDefaultsKey];
144
145 [[NSUserDefaults standardUserDefaults] registerDefaults:registrationDictionary];
146}

148void WebProcessPool::updateProcessSuppressionState()
149{
150 WebsiteDataStore::forEachWebsiteDataStore([enabled = processSuppressionEnabled()] (WebsiteDataStore& dataStore) {
151 if (auto* networkProcess = dataStore.networkProcessIfExists())
152 networkProcess->setProcessSuppressionEnabled(enabled);
153 });
154
155#if ENABLE(NETSCAPE_PLUGIN_API)
156 if (!m_processSuppressionDisabledForPageCounter.value())
157 m_pluginProcessManagerProcessSuppressionDisabledToken = nullptr;
158 else if (!m_pluginProcessManagerProcessSuppressionDisabledToken)
159 m_pluginProcessManagerProcessSuppressionDisabledToken = PluginProcessManager::singleton().processSuppressionDisabledToken();
160#endif
161}

171void WebProcessPool::platformInitialize()
172{
173 registerUserDefaultsIfNeeded();
174 registerNotificationObservers();
175 initializeClassesForParameterCoding();
176
177 // FIXME: This should be able to share code with WebCore's MemoryPressureHandler (and be platform independent).
178 // Right now it cannot because WebKit1 and WebKit2 need to be able to coexist in the UI process,
179 // and you can only have one WebCore::MemoryPressureHandler.
180
181 if (![[NSUserDefaults standardUserDefaults] boolForKey:@"WebKitSuppressMemoryPressureHandler"])
182 installMemoryPressureHandler();
183
184#if PLATFORM(IOS_FAMILY) && !PLATFORM(MACCATALYST)
185 if (!_MGCacheValid()) {
186 dispatch_async(dispatch_get_global_queue(DISPATCH_QUEUE_PRIORITY_DEFAULT, 0), ^{
187 [adoptNS([[objc_getClass("MobileGestaltHelperProxy") alloc] init]) proxyRebuildCache];
188 });
189 }
190#endif
191
192#if PLATFORM(MAC)
193 [WKWebInspectorPreferenceObserver sharedInstance];
194#endif
195}

295void WebProcessPool::platformInitializeWebProcess(const WebProcessProxy& process, WebProcessCreationParameters& parameters)
296{
297 parameters.mediaMIMETypes = process.mediaMIMETypes();
298
299#if PLATFORM(MAC)
300 ALLOW_DEPRECATED_DECLARATIONS_BEGIN
301 ASSERT(hasProcessPrivilege(ProcessPrivilege::CanCommunicateWithWindowServer));
302 parameters.accessibilityEnhancedUserInterfaceEnabled = [[NSApp accessibilityAttributeValue:@"AXEnhancedUserInterface"] boolValue];
303 ALLOW_DEPRECATED_DECLARATIONS_END
304#else
305 parameters.accessibilityEnhancedUserInterfaceEnabled = false;
306#endif
307
308 NSUserDefaults *defaults = [NSUserDefaults standardUserDefaults];
309
310 parameters.shouldEnableJIT = [defaults boolForKey:WebKitJSCJITEnabledDefaultsKey];
311 parameters.shouldEnableFTLJIT = [defaults boolForKey:WebKitJSCFTLJITEnabledDefaultsKey];
312 parameters.shouldEnableMemoryPressureReliefLogging = [defaults boolForKey:@"LogMemoryJetsamDetails"];
313 parameters.shouldSuppressMemoryPressureHandler = [defaults boolForKey:WebKitSuppressMemoryPressureHandlerDefaultsKey];
314
315#if HAVE(HOSTED_CORE_ANIMATION)
316 parameters.acceleratedCompositingPort = MachSendRight::create([CARemoteLayerServer sharedServer].serverPort);
317#endif
318
319 // FIXME: This should really be configurable; we shouldn't just blindly allow read access to the UI process bundle.
320 parameters.uiProcessBundleResourcePath = m_resolvedPaths.uiProcessBundleResourcePath;
321 SandboxExtension::createHandleWithoutResolvingPath(parameters.uiProcessBundleResourcePath, SandboxExtension::Type::ReadOnly, parameters.uiProcessBundleResourcePathExtensionHandle);
322
323 parameters.uiProcessBundleIdentifier = applicationBundleIdentifier();
324
325 parameters.latencyQOS = webProcessLatencyQOS();
326 parameters.throughputQOS = webProcessThroughputQOS();
327
328#if PLATFORM(IOS_FAMILY)
329 if (!m_resolvedPaths.cookieStorageDirectory.isEmpty())
330 SandboxExtension::createHandleWithoutResolvingPath(m_resolvedPaths.cookieStorageDirectory, SandboxExtension::Type::ReadWrite, parameters.cookieStorageDirectoryExtensionHandle);
331
332 if (!m_resolvedPaths.containerCachesDirectory.isEmpty())
333 SandboxExtension::createHandleWithoutResolvingPath(m_resolvedPaths.containerCachesDirectory, SandboxExtension::Type::ReadWrite, parameters.containerCachesDirectoryExtensionHandle);
334
335 if (!m_resolvedPaths.containerTemporaryDirectory.isEmpty())
336 SandboxExtension::createHandleWithoutResolvingPath(m_resolvedPaths.containerTemporaryDirectory, SandboxExtension::Type::ReadWrite, parameters.containerTemporaryDirectoryExtensionHandle);
337#endif
338#if PLATFORM(COCOA) && ENABLE(REMOTE_INSPECTOR)
339 if (WebProcessProxy::shouldEnableRemoteInspector()) {
340 SandboxExtension::Handle enableRemoteWebInspectorExtensionHandle;
341 if (SandboxExtension::createHandleForMachLookup("com.apple.webinspector"_s, std::nullopt, enableRemoteWebInspectorExtensionHandle))
342 parameters.enableRemoteWebInspectorExtensionHandle = WTFMove(enableRemoteWebInspectorExtensionHandle);
343 }
344#endif
345
346 parameters.fontAllowList = m_fontAllowList;
347
348 if (m_bundleParameters) {
349 auto keyedArchiver = adoptNS([[NSKeyedArchiver alloc] initRequiringSecureCoding:YES]);
350
351 @try {
352 [keyedArchiver encodeObject:m_bundleParameters.get() forKey:@"parameters"];
353 [keyedArchiver finishEncoding];
354 } @catch (NSException *exception) {
355 LOG_ERROR("Failed to encode bundle parameters: %@", exception);
356 }
357
358 auto data = keyedArchiver.get().encodedData;
359
360 parameters.bundleParameterData = API::Data::createWithoutCopying(WTFMove(data));
361 }
362 parameters.networkATSContext = adoptCF(_CFNetworkCopyATSContext());
363
364#if ENABLE(MEDIA_STREAM)
365 // Allow microphone access if either preference is set because WebRTC requires microphone access.
366 bool mediaDevicesEnabled = m_defaultPageGroup->preferences().mediaDevicesEnabled();
367 bool webRTCEnabled = m_defaultPageGroup->preferences().peerConnectionEnabled();
368 if ([defaults objectForKey:@"ExperimentalPeerConnectionEnabled"])
369 webRTCEnabled = [defaults boolForKey:@"ExperimentalPeerConnectionEnabled"];
370
371 bool isSafari = false;
372#if PLATFORM(IOS_FAMILY)
373 if (WebCore::IOSApplication::isMobileSafari())
374 isSafari = true;
375#elif PLATFORM(MAC)
376 if (WebCore::MacApplication::isSafari())
377 isSafari = true;
378#endif
379
380#if !LOG_DISABLED || !RELEASE_LOG_DISABLED
381 parameters.webCoreLoggingChannels = [[NSUserDefaults standardUserDefaults] stringForKey:@"WebCoreLogging"];
382 parameters.webKitLoggingChannels = [[NSUserDefaults standardUserDefaults] stringForKey:@"WebKit2Logging"];
383#endif
384
385#if PLATFORM(MAC) || PLATFORM(MACCATALYST)
386 // FIXME: Remove this and related parameter when <rdar://problem/29448368> is fixed.
387 if (isSafari && mediaDevicesEnabled && !m_defaultPageGroup->preferences().captureAudioInUIProcessEnabled() && !m_defaultPageGroup->preferences().captureAudioInGPUProcessEnabled())
388 SandboxExtension::createHandleForGenericExtension("com.apple.webkit.microphone"_s, parameters.audioCaptureExtensionHandle);
389#else
390 UNUSED_PARAM(mediaDevicesEnabled);
391#endif
392#endif
393
394#if ENABLE(RESOURCE_LOAD_STATISTICS) && !RELEASE_LOG_DISABLED
395 parameters.shouldLogUserInteraction = [defaults boolForKey:WebKitLogCookieInformationDefaultsKey];
396#endif
397
398 auto screenProperties = WebCore::collectScreenProperties();
399 parameters.screenProperties = WTFMove(screenProperties);
400#if PLATFORM(MAC)
401 parameters.useOverlayScrollbars = ([NSScroller preferredScrollerStyle] == NSScrollerStyleOverlay);
402#endif
403
404#if PLATFORM(IOS)
405 if (WebCore::deviceHasAGXCompilerService())
406 parameters.compilerServiceExtensionHandles = SandboxExtension::createHandlesForMachLookup(WebCore::agxCompilerServices(), std::nullopt);
407#endif
408
409#if PLATFORM(IOS_FAMILY)
410 if (!WebCore::IOSApplication::isMobileSafari())
411 parameters.dynamicMachExtensionHandles = SandboxExtension::createHandlesForMachLookup(nonBrowserServices(), std::nullopt);
412
413 if (WebCore::deviceHasAGXCompilerService())
414 parameters.dynamicIOKitExtensionHandles = SandboxExtension::createHandlesForIOKitClassExtensions(WebCore::agxCompilerClasses(), std::nullopt);
415#endif
416
417 if (isInternalInstall())
418 parameters.diagnosticsExtensionHandles = SandboxExtension::createHandlesForMachLookup(diagnosticServices(), std::nullopt, SandboxExtension::Flags::NoReport);
419
420 parameters.systemHasBattery = systemHasBattery();
421 parameters.systemHasAC = cachedSystemHasAC().value_or(true);
422
423 if (requiresContainerManagerAccess()) {
424 SandboxExtension::Handle handle;
425 SandboxExtension::createHandleForMachLookup("com.apple.containermanagerd"_s, std::nullopt, handle);
426 parameters.containerManagerExtensionHandle = WTFMove(handle);
427 }
428
429#if PLATFORM(IOS_FAMILY)
430 parameters.currentUserInterfaceIdiomIsPad = currentUserInterfaceIdiomIsPadOrMac();
431 parameters.supportsPictureInPicture = supportsPictureInPicture();
432 parameters.cssValueToSystemColorMap = RenderThemeIOS::cssValueToSystemColorMap();
433 parameters.focusRingColor = RenderThemeIOS::systemFocusRingColor();
434 parameters.localizedDeviceModel = localizedDeviceModel();
435 parameters.contentSizeCategory = RenderThemeCocoa::singleton().contentSizeCategory();
436#endif
437
438#if ENABLE(CFPREFS_DIRECT_MODE) && PLATFORM(IOS_FAMILY)
439 if (_AXSApplicationAccessibilityEnabled())
440 parameters.preferencesExtensionHandles = SandboxExtension::createHandlesForMachLookup({ "com.apple.cfprefsd.agent"_s, "com.apple.cfprefsd.daemon"_s }, std::nullopt);
441#endif
442
443#if PLATFORM(IOS_FAMILY) && !PLATFORM(MACCATALYST)
444 if (!_MGCacheValid()) {
445 SandboxExtension::Handle handle;
446 SandboxExtension::createHandleForMachLookup("com.apple.mobilegestalt.xpc"_s, std::nullopt, handle);
447 parameters.mobileGestaltExtensionHandle = WTFMove(handle);
448 }
449#endif
450
451#if PLATFORM(MAC)
452 SandboxExtension::Handle launchServicesExtensionHandle;
453 SandboxExtension::createHandleForMachLookup("com.apple.coreservices.launchservicesd"_s, std::nullopt, launchServicesExtensionHandle);
454 parameters.launchServicesExtensionHandle = WTFMove(launchServicesExtensionHandle);
455#endif
456
457#if PLATFORM(MAC) && HAVE(VIDEO_RESTRICTED_DECODING)
458 SandboxExtension::Handle trustdAgentExtensionHandle;
459 SandboxExtension::createHandleForMachLookup("com.apple.trustd.agent"_s, std::nullopt, trustdAgentExtensionHandle);
460 parameters.trustdAgentExtensionHandle = WTFMove(trustdAgentExtensionHandle);
461#endif
462
463#if PLATFORM(IOS_FAMILY) && ENABLE(CFPREFS_DIRECT_MODE)
464 if ([UIApplication sharedApplication]) {
465 auto state = [[UIApplication sharedApplication] applicationState];
466 if (state == UIApplicationStateActive)
467 startObservingPreferenceChanges();
468 }
469#endif
470
471#if HAVE(CATALYST_USER_INTERFACE_IDIOM_AND_SCALE_FACTOR)
472 parameters.overrideUserInterfaceIdiomAndScale = { _UIApplicationCatalystUserInterfaceIdiom(), _UIApplicationCatalystScaleFactor() };
473#endif
474
475#if HAVE(MOUSE_DEVICE_OBSERVATION)
476 parameters.hasMouseDevice = [[WKMouseDeviceObserver sharedInstance] hasMouseDevice];
477#endif
478
479#if HAVE(STYLUS_DEVICE_OBSERVATION)
480 parameters.hasStylusDevice = [[WKStylusDeviceObserver sharedInstance] hasStylusDevice];
481#endif
482
483 // If we're using the GPU process for DOM rendering, we can't query the maximum IOSurface size in the Web Content process.
484 // However, querying this is a launch time regression, so limit this to only the necessary case.
485 if (m_defaultPageGroup->preferences().useGPUProcessForDOMRenderingEnabled())
486 parameters.maximumIOSurfaceSize = WebCore::IOSurface::maximumSize();
487
488 parameters.accessibilityPreferences = accessibilityPreferences();
489}

514}

527String WebProcessPool::networkingCachesDirectory()
528{
529 String path = cacheDirectoryInContainerOrHomeDirectory("/Library/Caches/com.apple.WebKit.Networking/"_s);
530
531 NSError *error = nil;
532 NSString* nsPath = path;
533 if (![[NSFileManager defaultManager] createDirectoryAtPath:nsPath withIntermediateDirectories:YES attributes:nil error:&error]) {
534 NSLog(@"could not create networking caches directory \"%@\", error %@", nsPath, error);
535 return String();
536 }
537
538 return path;
539}

921void WebProcessPool::initializeClassesForParameterCoding()
922{
923 const auto& customClasses = m_configuration->customClassesForParameterCoder();
924 if (customClasses.isEmpty())
925 return;
926
927 auto standardClasses = [NSSet setWithObjects:[NSArray class], [NSData class], [NSDate class], [NSDictionary class], [NSNull class],
928 [NSNumber class], [NSSet class], [NSString class], [NSTimeZone class], [NSURL class], [NSUUID class], nil];
929
930 auto mutableSet = adoptNS([standardClasses mutableCopy]);
931
932 for (const auto& customClass : customClasses) {
933 auto className = customClass.utf8();
934 Class objectClass = objc_lookUpClass(className.data());
935 if (!objectClass) {
936 WTFLogAlways("InjectedBundle::extendClassesForParameterCoder - Class %s is not a valid Objective C class.\n", className.data());
937 break;
938 }
939
940 [mutableSet.get() addObject:objectClass];
941 }
942
943 m_classesForParameterCoder = mutableSet;
944}
