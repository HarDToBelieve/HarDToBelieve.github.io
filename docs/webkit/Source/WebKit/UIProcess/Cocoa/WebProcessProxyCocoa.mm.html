74RefPtr<ObjCObjectGraph> WebProcessProxy::transformHandlesToObjects(ObjCObjectGraph& objectGraph)
75{
76 struct Transformer final : ObjCObjectGraph::Transformer {
77 Transformer(WebProcessProxy& webProcessProxy)
78 : m_webProcessProxy(webProcessProxy)
79 {
80 }
81
82 bool shouldTransformObject(id object) const override
83 {
84 if (dynamic_objc_cast<WKBrowsingContextHandle>(object))
85 return true;
86
87 ALLOW_DEPRECATED_DECLARATIONS_BEGIN
88 if (dynamic_objc_cast<WKTypeRefWrapper>(object))
89 return true;
90 ALLOW_DEPRECATED_DECLARATIONS_END
91 return false;
92 }
93
94 RetainPtr<id> transformObject(id object) const override
95 {
96 if (auto* handle = dynamic_objc_cast<WKBrowsingContextHandle>(object)) {
97 if (auto* webPageProxy = m_webProcessProxy.webPage(handle.pageProxyID)) {
98 ALLOW_DEPRECATED_DECLARATIONS_BEGIN
99 return [WKBrowsingContextController _browsingContextControllerForPageRef:toAPI(webPageProxy)];
100 ALLOW_DEPRECATED_DECLARATIONS_END
101 }
102
103 return [NSNull null];
104 }
105
106 ALLOW_DEPRECATED_DECLARATIONS_BEGIN
107 if (auto* wrapper = dynamic_objc_cast<WKTypeRefWrapper>(object))
108 return adoptNS([[WKTypeRefWrapper alloc] initWithObject:toAPI(m_webProcessProxy.transformHandlesToObjects(toImpl(wrapper.object)).get())]);
109 ALLOW_DEPRECATED_DECLARATIONS_END
110 return object;
111 }
112
113 WebProcessProxy& m_webProcessProxy;
114 };
115
116 return ObjCObjectGraph::create(ObjCObjectGraph::transform(objectGraph.rootObject(), Transformer(*this)).get());
117}

119RefPtr<ObjCObjectGraph> WebProcessProxy::transformObjectsToHandles(ObjCObjectGraph& objectGraph)
120{
121 struct Transformer final : ObjCObjectGraph::Transformer {
122 bool shouldTransformObject(id object) const override
123 {
124 ALLOW_DEPRECATED_DECLARATIONS_BEGIN
125 if (dynamic_objc_cast<WKBrowsingContextController>(object))
126 return true;
127 if (dynamic_objc_cast<WKTypeRefWrapper>(object))
128 return true;
129 ALLOW_DEPRECATED_DECLARATIONS_END
130 return false;
131 }
132
133 RetainPtr<id> transformObject(id object) const override
134 {
135 ALLOW_DEPRECATED_DECLARATIONS_BEGIN
136 if (auto* controller = dynamic_objc_cast<WKBrowsingContextController>(object))
137 return controller.handle;
138 if (auto* wrapper = dynamic_objc_cast<WKTypeRefWrapper>(object))
139 return adoptNS([[WKTypeRefWrapper alloc] initWithObject:toAPI(transformObjectsToHandles(toImpl(wrapper.object)).get())]);
140 ALLOW_DEPRECATED_DECLARATIONS_END
141 return object;
142 }
143 };
144
145 return ObjCObjectGraph::create(ObjCObjectGraph::transform(objectGraph.rootObject(), Transformer()).get());
146}

148static Vector<String>& mediaTypeCache()
149{
150 ASSERT(RunLoop::isMain());
151 static NeverDestroyed<Vector<String>> typeCache;
152 return typeCache;
153}
154
155void WebProcessProxy::cacheMediaMIMETypes(const Vector<String>& types)
156{
157 if (!mediaTypeCache().isEmpty())
158 return;
159
160 mediaTypeCache() = types;
161 for (auto& process : processPool().processes()) {
162 if (process != *this)
163 cacheMediaMIMETypesInternal(types);
164 }
165}
166
167void WebProcessProxy::cacheMediaMIMETypesInternal(const Vector<String>& types)
168{
169 if (!mediaTypeCache().isEmpty())
170 return;
171
172 mediaTypeCache() = types;
173 send(Messages::WebProcess::SetMediaMIMETypes(types), 0);
174}
175
176Vector<String> WebProcessProxy::mediaMIMETypes() const
177{
178 return mediaTypeCache();
179}

213void WebProcessProxy::unblockAccessibilityServerIfNeeded()
214{
215 if (m_hasSentMessageToUnblockAccessibilityServer)
216 return;
217#if PLATFORM(IOS_FAMILY)
218 if (!_AXSApplicationAccessibilityEnabled())
219 return;
220#endif
221 if (!processIdentifier())
222 return;
223 if (!canSendMessage())
224 return;
225
226 SandboxExtension::HandleArray handleArray;
227#if PLATFORM(IOS_FAMILY)
228 handleArray = SandboxExtension::createHandlesForMachLookup({ "com.apple.iphone.axserver-systemwide"_s, "com.apple.frontboard.systemappservices"_s }, connection() ? connection()->getAuditToken() : std::nullopt);
229 ASSERT(handleArray.size() == 2);
230#endif
231
232 send(Messages::WebProcess::UnblockServicesRequiredByAccessibility(handleArray), 0);
233 m_hasSentMessageToUnblockAccessibilityServer = true;
234}

236#if ENABLE(CFPREFS_DIRECT_MODE)
237void WebProcessProxy::unblockPreferenceServiceIfNeeded()
238{
239 if (m_hasSentMessageToUnblockPreferenceService)
240 return;
241 if (!processIdentifier())
242 return;
243 if (!canSendMessage())
244 return;
245
246 auto handleArray = SandboxExtension::createHandlesForMachLookup({ "com.apple.cfprefsd.agent"_s, "com.apple.cfprefsd.daemon"_s }, connection() ? connection()->getAuditToken() : std::nullopt);
247 ASSERT(handleArray.size() == 2);
248
249 send(Messages::WebProcess::UnblockPreferenceService(WTFMove(handleArray)), 0);
250 m_hasSentMessageToUnblockPreferenceService = true;
251}
252#endif

268void WebProcessProxy::sendAudioComponentRegistrations()
269{
270 using namespace PAL;
271
272 if (!PAL::isAudioToolboxCoreFrameworkAvailable() || !PAL::canLoad_AudioToolboxCore_AudioComponentFetchServerRegistrations())
273 return;
274
275 dispatch_async(dispatch_get_global_queue(DISPATCH_QUEUE_PRIORITY_DEFAULT, 0), [protectedThis = makeRef(*this)] () mutable {
276 CFDataRef registrations { nullptr };
277 if (noErr != AudioComponentFetchServerRegistrations(&registrations) || !registrations)
278 return;
279
280 RunLoop::main().dispatch([protectedThis = WTFMove(protectedThis), registrations = adoptCF(registrations)] () mutable {
281 auto registrationData = WebCore::SharedBuffer::create(registrations.get());
282 protectedThis->send(Messages::WebProcess::ConsumeAudioComponentRegistrations({ registrationData }), 0);
283 });
284 });
285}

287bool WebProcessProxy::hasCorrectPACEntitlement()
288{
289 if (!hasConnection()) {
290 ASSERT_NOT_REACHED();
291 return false;
292 }
293
294#if HAVE(PAC_SHARED_REGION_ID)
295 auto auditToken = connection()->getAuditToken();
296 if (!auditToken) {
297 ASSERT_NOT_REACHED();
298 RELEASE_LOG_ERROR(Process, "Unable to get parent web process audit token");
299 return false;
300 }
301
302#if USE(APPLE_INTERNAL_SDK)
303 // Confirm that the connection is from a WebContent process:
304 if (!WTF::hasEntitlementValue(auditToken.value(), "com.apple.pac.shared_region_id", "WebContent")) {
305 RELEASE_LOG_ERROR(Process, "Process is not an entitled WebContent process. Process shared_region_id is incorrect.");
306 return false;
307 }
308#endif
309#endif
310
311 return true;
312}
