37namespace WebKit {
38
39class DeviceIdHashSaltStorage : public ThreadSafeRefCounted<DeviceIdHashSaltStorage, WTF::DestructionThread::MainRunLoop> {
40public:
41 static Ref<DeviceIdHashSaltStorage> create(const String& deviceIdHashSaltStorageDirectory);
42 ~DeviceIdHashSaltStorage();
43
44 void deviceIdHashSaltForOrigin(const WebCore::SecurityOrigin& documentOrigin, const WebCore::SecurityOrigin& parentOrigin, CompletionHandler<void(String&&)>&&);
45
46 void getDeviceIdHashSaltOrigins(CompletionHandler<void(HashSet<WebCore::SecurityOriginData>&&)>&&);
47 void deleteDeviceIdHashSaltForOrigins(const Vector<WebCore::SecurityOriginData>&, CompletionHandler<void()>&&);
48 void deleteDeviceIdHashSaltOriginsModifiedSince(WallTime, CompletionHandler<void()>&&);
49
50private:
51 struct HashSaltForOrigin {
52 WTF_MAKE_STRUCT_FAST_ALLOCATED;
53
54 HashSaltForOrigin(WebCore::SecurityOriginData&& documentOrigin, WebCore::SecurityOriginData&& parentOrigin, String&& deviceIdHashSalt)
55 : documentOrigin(WTFMove(documentOrigin))
56 , parentOrigin(WTFMove(parentOrigin))
57 , deviceIdHashSalt(WTFMove(deviceIdHashSalt))
58 , lastTimeUsed(WallTime::now())
59 { };
60
61 HashSaltForOrigin isolatedCopy() const
62 {
63 auto isolatedCopy = HashSaltForOrigin(documentOrigin.isolatedCopy(), parentOrigin.isolatedCopy(), deviceIdHashSalt.isolatedCopy());
64 isolatedCopy.lastTimeUsed = lastTimeUsed;
65 return isolatedCopy;
66 };
67
68 WebCore::SecurityOriginData documentOrigin;
69 WebCore::SecurityOriginData parentOrigin;
70 String deviceIdHashSalt;
71 WallTime lastTimeUsed;
72 };
73
74 DeviceIdHashSaltStorage(const String& deviceIdHashSaltStorageDirectory);
75 void loadStorageFromDisk(CompletionHandler<void(HashMap<String, std::unique_ptr<HashSaltForOrigin>>&&)>&&);
76 void storeHashSaltToDisk(const HashSaltForOrigin&);
77 void deleteHashSaltFromDisk(const HashSaltForOrigin&);
78 std::unique_ptr<WebCore::KeyedEncoder> createEncoderFromData(const HashSaltForOrigin&) const;
79 std::unique_ptr<HashSaltForOrigin> getDataFromDecoder(WebCore::KeyedDecoder*, String&& deviceIdHashSalt) const;
80 void completePendingHandler(CompletionHandler<void(HashSet<WebCore::SecurityOriginData>&&)>&&);
81 void completeDeviceIdHashSaltForOriginCall(WebCore::SecurityOriginData&& documentOrigin, WebCore::SecurityOriginData&& parentOrigin, CompletionHandler<void(String&&)>&&);
82
83 Ref<WorkQueue> m_queue;
84 HashMap<String, std::unique_ptr<HashSaltForOrigin>> m_deviceIdHashSaltForOrigins;
85 bool m_isLoaded { false };
86 Vector<CompletionHandler<void()>> m_pendingCompletionHandlers;
87 const String m_deviceIdHashSaltStorageDirectory;
88};
