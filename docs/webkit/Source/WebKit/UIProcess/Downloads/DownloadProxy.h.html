64class DownloadProxy : public API::ObjectImpl<API::Object::Type::Download>, public IPC::MessageReceiver {
65public:
66
67 template<typename... Args> static Ref<DownloadProxy> create(Args&&... args)
68 {
69 return adoptRef(*new DownloadProxy(std::forward<Args>(args)...));
70 }
71 ~DownloadProxy();
72
73 DownloadID downloadID() const { return m_downloadID; }
74 const WebCore::ResourceRequest& request() const { return m_request; }
75 API::Data* legacyResumeData() const { return m_legacyResumeData.get(); }
76
77 void cancel(CompletionHandler<void(API::Data*)>&&);
78
79 void invalidate();
80 void processDidClose();
81
82 void didReceiveDownloadProxyMessage(IPC::Connection&, IPC::Decoder&);
83 bool didReceiveSyncDownloadProxyMessage(IPC::Connection&, IPC::Decoder&, UniqueRef<IPC::Encoder>&);
84
85 WebPageProxy* originatingPage() const;
86
87 void setRedirectChain(Vector<URL>&& redirectChain) { m_redirectChain = WTFMove(redirectChain); }
88 const Vector<URL>& redirectChain() const { return m_redirectChain; }
89
90 void setWasUserInitiated(bool value) { m_wasUserInitiated = value; }
91 bool wasUserInitiated() const { return m_wasUserInitiated; }
92
93 const String& destinationFilename() const { return m_destinationFilename; }
94 void setDestinationFilename(const String& d) { m_destinationFilename = d; }
95
96#if USE(SYSTEM_PREVIEW)
97 bool isSystemPreviewDownload() const { return request().isSystemPreview(); }
98 WebCore::SystemPreviewInfo systemPreviewDownloadInfo() const { return request().systemPreviewInfo(); }
99#endif
100
101#if PLATFORM(COCOA)
102 void publishProgress(const URL&);
103 void setProgress(NSProgress *progress) { m_progress = progress; }
104 NSProgress *progress() const { return m_progress.get(); }
105#endif
106
107 API::FrameInfo& frameInfo() { return m_frameInfo.get(); }
108
109 API::DownloadClient& client() { return m_client.get(); }
110 void setClient(Ref<API::DownloadClient>&&);
111 void setDidStartCallback(CompletionHandler<void(DownloadProxy*)>&& callback) { m_didStartCallback = WTFMove(callback); }
112 void setSuggestedFilename(const String& suggestedFilename) { m_suggestedFilename = suggestedFilename; }
113
114 // Message handlers.
115 void didStart(const WebCore::ResourceRequest&, const String& suggestedFilename);
116 void didReceiveAuthenticationChallenge(WebCore::AuthenticationChallenge&&, uint64_t challengeID);
117 void didReceiveData(uint64_t bytesWritten, uint64_t totalBytesWritten, uint64_t totalBytesExpectedToWrite);
118 void shouldDecodeSourceDataOfMIMEType(const String& mimeType, bool& result);
119 void didCreateDestination(const String& path);
120 void didFinish();
121 void didFail(const WebCore::ResourceError&, const IPC::DataReference& resumeData);
122 void willSendRequest(WebCore::ResourceRequest&& redirectRequest, const WebCore::ResourceResponse& redirectResponse);
123 void decideDestinationWithSuggestedFilename(const WebCore::ResourceResponse&, String&& suggestedFilename, CompletionHandler<void(String, SandboxExtension::Handle, AllowOverwrite)>&&);
124
125private:
126 explicit DownloadProxy(DownloadProxyMap&, WebsiteDataStore&, API::DownloadClient&, const WebCore::ResourceRequest&, const FrameInfoData&, WebPageProxy*);
127
128 // IPC::MessageReceiver
129 void didReceiveMessage(IPC::Connection&, IPC::Decoder&) override;
130
131 DownloadProxyMap& m_downloadProxyMap;
132 RefPtr<WebsiteDataStore> m_dataStore;
133 Ref<API::DownloadClient> m_client;
134 DownloadID m_downloadID;
135
136 RefPtr<API::Data> m_legacyResumeData;
137 WebCore::ResourceRequest m_request;
138 String m_suggestedFilename;
139 String m_destinationFilename;
140
141 WeakPtr<WebPageProxy> m_originatingPage;
142 Vector<URL> m_redirectChain;
143 bool m_wasUserInitiated { true };
144 Ref<API::FrameInfo> m_frameInfo;
145 CompletionHandler<void(DownloadProxy*)> m_didStartCallback;
146#if PLATFORM(COCOA)
147 RetainPtr<NSProgress> m_progress;
148#endif
149};
