41DrawingAreaProxy::DrawingAreaProxy(DrawingAreaType type, WebPageProxy& webPageProxy, WebProcessProxy& process)
42 : m_type(type)
43 , m_identifier(DrawingAreaIdentifier::generate())
44 , m_webPageProxy(webPageProxy)
45 , m_process(makeRef(process))
46 , m_size(webPageProxy.viewSize())
47#if PLATFORM(MAC)
48 , m_viewExposedRectChangedTimer(RunLoop::main(), this, &DrawingAreaProxy::viewExposedRectChangedTimerFired)
49#endif
50{
51}
52
53DrawingAreaProxy::~DrawingAreaProxy()
54{
55 if (m_startedReceivingMessages)
56 process().removeMessageReceiver(Messages::DrawingAreaProxy::messageReceiverName(), m_identifier);
57}
58
59void DrawingAreaProxy::startReceivingMessages()
60{
61 m_startedReceivingMessages = true;
62 process().addMessageReceiver(Messages::DrawingAreaProxy::messageReceiverName(), m_identifier, *this);
63}
64
65bool DrawingAreaProxy::setSize(const IntSize& size, const IntSize& scrollDelta)
66{
67 if (m_size == size && scrollDelta.isZero())
68 return false;
69
70 m_size = size;
71 m_scrollOffset += scrollDelta;
72 sizeDidChange();
73 return true;
74}

83IPC::Connection* DrawingAreaProxy::messageSenderConnection() const

88bool DrawingAreaProxy::sendMessage(UniqueRef<IPC::Encoder>&& encoder, OptionSet<IPC::SendOption> sendOptions, std::optional<std::pair<CompletionHandler<void(IPC::Decoder*)>, uint64_t>>&& asyncReplyInfo)
89{
90 return process().sendMessage(WTFMove(encoder), sendOptions, WTFMove(asyncReplyInfo));
91}

93#if PLATFORM(MAC)
94void DrawingAreaProxy::didChangeViewExposedRect()
95{
96 if (!m_webPageProxy.hasRunningProcess())
97 return;
98
99 if (!m_viewExposedRectChangedTimer.isActive())
100 m_viewExposedRectChangedTimer.startOneShot(0_s);
101}
102
103void DrawingAreaProxy::viewExposedRectChangedTimerFired()
104{
105 if (!m_webPageProxy.hasRunningProcess())
106 return;
107
108 auto viewExposedRect = m_webPageProxy.viewExposedRect();
109 if (viewExposedRect == m_lastSentViewExposedRect)
110 return;
111
112 send(Messages::DrawingArea::SetViewExposedRect(viewExposedRect));
113 m_lastSentViewExposedRect = viewExposedRect;
114}
115#endif // PLATFORM(MAC)
