101Ref<GPUProcessProxy> GPUProcessProxy::getOrCreate()
102{
103 ASSERT(RunLoop::isMain());
104 if (auto& existingGPUProcess = singleton()) {
105 ASSERT(existingGPUProcess->state() != State::Terminated);
106 return *existingGPUProcess;
107 }
108 auto gpuProcess = adoptRef(*new GPUProcessProxy);
109 gpuProcess->updatePreferences();
110 singleton() = makeWeakPtr(gpuProcess.get());
111 return gpuProcess;
112}

130GPUProcessProxy::GPUProcessProxy()
131 : AuxiliaryProcessProxy()
132 , m_throttler(*this, false)
133#if ENABLE(MEDIA_STREAM)
134 , m_useMockCaptureDevices(MockRealtimeMediaSourceCenter::mockRealtimeMediaSourceCenterEnabled())
135#endif
136{
137 connect();
138
139 GPUProcessCreationParameters parameters;
140#if ENABLE(MEDIA_STREAM)
141 parameters.useMockCaptureDevices = m_useMockCaptureDevices;
142#if PLATFORM(MAC)
143 // FIXME: Remove this and related parameter when <rdar://problem/29448368> is fixed.
144 if (MacApplication::isSafari()) {
145 SandboxExtension::createHandleForGenericExtension("com.apple.webkit.microphone"_s, parameters.microphoneSandboxExtensionHandle);
146 m_hasSentMicrophoneSandboxExtension = true;
147 }
148#endif
149#endif // ENABLE(MEDIA_STREAM)
150
151 parameters.parentPID = getCurrentProcessID();
152
153#if USE(SANDBOX_EXTENSIONS_FOR_CACHE_AND_TEMP_DIRECTORY_ACCESS)
154 auto containerCachesDirectory = resolveAndCreateReadWriteDirectoryForSandboxExtension(gpuProcessCachesDirectory());
155 auto containerTemporaryDirectory = resolveAndCreateReadWriteDirectoryForSandboxExtension(WebProcessPool::containerTemporaryDirectory());
156
157 if (!containerCachesDirectory.isEmpty())
158 SandboxExtension::createHandleWithoutResolvingPath(containerCachesDirectory, SandboxExtension::Type::ReadWrite, parameters.containerCachesDirectoryExtensionHandle);
159
160 if (!containerTemporaryDirectory.isEmpty())
161 SandboxExtension::createHandleWithoutResolvingPath(containerTemporaryDirectory, SandboxExtension::Type::ReadWrite, parameters.containerTemporaryDirectoryExtensionHandle);
162#endif
163#if PLATFORM(IOS_FAMILY)
164 if (WebCore::deviceHasAGXCompilerService()) {
165 parameters.compilerServiceExtensionHandles = SandboxExtension::createHandlesForMachLookup(WebCore::agxCompilerServices(), std::nullopt);
166 parameters.dynamicIOKitExtensionHandles = SandboxExtension::createHandlesForIOKitClassExtensions(WebCore::agxCompilerClasses(), std::nullopt);
167 }
168
169 if (!WebCore::IOSApplication::isMobileSafari())
170 parameters.dynamicMachExtensionHandles = SandboxExtension::createHandlesForMachLookup(nonBrowserServices(), std::nullopt);
171#endif
172
173#if !LOG_DISABLED || !RELEASE_LOG_DISABLED
174 parameters.webCoreLoggingChannels = WebCore::logLevelString();
175 parameters.webKitLoggingChannels = WebKit::logLevelString();
176#endif
177
178 // Initialize the GPU process.
179 send(Messages::GPUProcess::InitializeGPUProcess(parameters), 0);
180 updateProcessAssertion();
181}

202static inline bool addCameraSandboxExtensions(Vector<SandboxExtension::Handle>& extensions)
203{
204 SandboxExtension::Handle sandboxExtensionHandle;
205 if (!SandboxExtension::createHandleForGenericExtension("com.apple.webkit.camera"_s, sandboxExtensionHandle)) {
206 RELEASE_LOG_ERROR(WebRTC, "Unable to create com.apple.webkit.camera sandbox extension");
207 return false;
208 }
209#if HAVE(AUDIT_TOKEN)
210 if (shouldCreateAppleCameraServiceSandboxExtension()) {
211 SandboxExtension::Handle appleCameraServicePathSandboxExtensionHandle;
212 if (!SandboxExtension::createHandleForMachLookup("com.apple.applecamerad"_s, std::nullopt, appleCameraServicePathSandboxExtensionHandle)) {
213 RELEASE_LOG_ERROR(WebRTC, "Unable to create com.apple.applecamerad sandbox extension");
214 return false;
215 }
216#if HAVE(ADDITIONAL_APPLE_CAMERA_SERVICE)
217 SandboxExtension::Handle additionalAppleCameraServicePathSandboxExtensionHandle;
218 if (!SandboxExtension::createHandleForMachLookup("com.apple.appleh13camerad"_s, std::nullopt, additionalAppleCameraServicePathSandboxExtensionHandle)) {
219 RELEASE_LOG_ERROR(WebRTC, "Unable to create com.apple.appleh13camerad sandbox extension");
220 return false;
221 }
222 extensions.append(WTFMove(additionalAppleCameraServicePathSandboxExtensionHandle));
223#endif
224 extensions.append(WTFMove(appleCameraServicePathSandboxExtensionHandle));
225 }
226#endif // HAVE(AUDIT_TOKEN)
227
228 extensions.append(WTFMove(sandboxExtensionHandle));
229 return true;
230}

256void GPUProcessProxy::updateSandboxAccess(bool allowAudioCapture, bool allowVideoCapture)
257{
258 if (m_useMockCaptureDevices)
259 return;
260
261#if PLATFORM(COCOA)
262 Vector<SandboxExtension::Handle> extensions;
263
264 if (allowVideoCapture && !m_hasSentCameraSandboxExtension && addCameraSandboxExtensions(extensions))
265 m_hasSentCameraSandboxExtension = true;
266
267 if (allowAudioCapture && !m_hasSentMicrophoneSandboxExtension && addMicrophoneSandboxExtension(extensions))
268 m_hasSentMicrophoneSandboxExtension = true;
269
270#if PLATFORM(IOS)
271 if ((allowAudioCapture || allowVideoCapture) && !m_hasSentTCCDSandboxExtension && addTCCDSandboxExtension(extensions))
272 m_hasSentTCCDSandboxExtension = true;
273#endif // PLATFORM(IOS)
274
275 if (!extensions.isEmpty())
276 send(Messages::GPUProcess::UpdateSandboxAccess { extensions }, 0);
277#endif // PLATFORM(COCOA)
278}

325void GPUProcessProxy::getGPUProcessConnection(WebProcessProxy& webProcessProxy, const GPUProcessConnectionParameters& parameters, Messages::WebProcessProxy::GetGPUProcessConnection::DelayedReply&& reply)
326{
327 addSession(webProcessProxy.websiteDataStore());
328
329 RELEASE_LOG(ProcessSuspension, "%p - GPUProcessProxy is taking a background assertion because a web process is requesting a connection", this);
330 startResponsivenessTimer(UseLazyStop::No);
331 sendWithAsyncReply(Messages::GPUProcess::CreateGPUConnectionToWebProcess { webProcessProxy.coreProcessIdentifier(), webProcessProxy.sessionID(), parameters }, [this, weakThis = makeWeakPtr(*this), reply = WTFMove(reply)](auto&& identifier) mutable {
332 if (!weakThis) {
333 RELEASE_LOG_ERROR(Process, "GPUProcessProxy::getGPUProcessConnection: GPUProcessProxy deallocated during connection establishment");
334 return reply({ });
335 }
336
337 stopResponsivenessTimer();
338 if (!identifier) {
339 RELEASE_LOG_ERROR(Process, "GPUProcessProxy::getGPUProcessConnection: connection identifier is empty");
340 return reply({ });
341 }
342
343#if USE(UNIX_DOMAIN_SOCKETS) || OS(WINDOWS)
344 reply(GPUProcessConnectionInfo { WTFMove(*identifier) });
345 UNUSED_VARIABLE(this);
346#elif OS(DARWIN)
347 MESSAGE_CHECK(MACH_PORT_VALID(identifier->port()));
348 reply(GPUProcessConnectionInfo { IPC::Attachment { identifier->port(), MACH_MSG_TYPE_MOVE_SEND }, this->connection()->getAuditToken() });
349#else
350 notImplemented();
351#endif
352 }, 0, IPC::SendOption::DispatchMessageEvenWhenWaitingForSyncReply);
353}

355void GPUProcessProxy::gpuProcessExited(GPUProcessTerminationReason reason)
356{
357 auto protectedThis = makeRef(*this);
358
359 switch (reason) {
360 case GPUProcessTerminationReason::Crash:
361 RELEASE_LOG_ERROR(Process, "%p - GPUProcessProxy::gpuProcessExited: reason=crash", this);
362 break;
363 case GPUProcessTerminationReason::IdleExit:
364 RELEASE_LOG(Process, "%p - GPUProcessProxy::gpuProcessExited: reason=idle-exit", this);
365 break;
366 case GPUProcessTerminationReason::Unresponsive:
367 RELEASE_LOG(Process, "%p - GPUProcessProxy::gpuProcessExited: reason=unresponsive", this);
368 break;
369 }
370
371 if (singleton() == this)
372 singleton() = nullptr;
373
374 for (auto& processPool : WebProcessPool::allProcessPools())
375 processPool->gpuProcessExited(processIdentifier(), reason);
376}

420void GPUProcessProxy::updateProcessAssertion()
421{
422 bool hasAnyForegroundWebProcesses = false;
423 bool hasAnyBackgroundWebProcesses = false;
424
425 for (auto& processPool : WebProcessPool::allProcessPools()) {
426 hasAnyForegroundWebProcesses |= processPool->hasForegroundWebProcesses();
427 hasAnyBackgroundWebProcesses |= processPool->hasBackgroundWebProcesses();
428 }
429
430 if (hasAnyForegroundWebProcesses) {
431 if (!ProcessThrottler::isValidForegroundActivity(m_activityFromWebProcesses)) {
432 m_activityFromWebProcesses = throttler().foregroundActivity("GPU for foreground view(s)"_s);
433 send(Messages::GPUProcess::ProcessDidTransitionToForeground(), 0);
434 }
435 return;
436 }
437 if (hasAnyBackgroundWebProcesses) {
438 if (!ProcessThrottler::isValidBackgroundActivity(m_activityFromWebProcesses)) {
439 m_activityFromWebProcesses = throttler().backgroundActivity("GPU for background view(s)"_s);
440 send(Messages::GPUProcess::ProcessDidTransitionToBackground(), 0);
441 }
442 return;
443 }
444
445 // Use std::exchange() instead of a simple nullptr assignment to avoid re-entering this
446 // function during the destructor of the ProcessThrottler activity, before setting
447 // m_activityFromWebProcesses.
448 std::exchange(m_activityFromWebProcesses, nullptr);
449}

451static inline GPUProcessSessionParameters gpuProcessSessionParameters(const WebsiteDataStore& store)
452{
453 GPUProcessSessionParameters parameters;
454
455 parameters.mediaCacheDirectory = store.resolvedMediaCacheDirectory();
456 SandboxExtension::Handle mediaCacheDirectoryExtensionHandle;
457 if (!parameters.mediaCacheDirectory.isEmpty())
458 SandboxExtension::createHandleWithoutResolvingPath(parameters.mediaCacheDirectory, SandboxExtension::Type::ReadWrite, parameters.mediaCacheDirectorySandboxExtensionHandle);
459
460#if ENABLE(LEGACY_ENCRYPTED_MEDIA)
461 parameters.mediaKeysStorageDirectory = store.resolvedMediaKeysDirectory();
462 SandboxExtension::Handle mediaKeysStorageDirectorySandboxExtensionHandle;
463 if (!parameters.mediaKeysStorageDirectory.isEmpty())
464 SandboxExtension::createHandleWithoutResolvingPath(parameters.mediaKeysStorageDirectory, SandboxExtension::Type::ReadWrite, parameters.mediaKeysStorageDirectorySandboxExtensionHandle);
465#endif
466
467 return parameters;
468}

537void GPUProcessProxy::updatePreferences()
538{
539 if (!canSendMessage())
540 return;
541
542#if ENABLE(MEDIA_SOURCE) && ENABLE(VP9)
543 bool hasEnabledWebMParser = false;
544#endif
545
546#if ENABLE(WEBM_FORMAT_READER)
547 bool hasEnabledWebMFormatReader = false;
548#endif
549
550#if ENABLE(OPUS)
551 bool hasEnabledOpus = false;
552#endif
553
554#if ENABLE(VORBIS)
555 bool hasEnabledVorbis = false;
556#endif
557
558 WebPageGroup::forEach([&] (auto& group) mutable {
559 if (!group.preferences().useGPUProcessForMediaEnabled())
560 return;
561
562#if ENABLE(OPUS)
563 if (group.preferences().opusDecoderEnabled())
564 hasEnabledOpus = true;
565#endif
566
567#if ENABLE(VORBIS)
568 if (group.preferences().vorbisDecoderEnabled())
569 hasEnabledVorbis = true;
570#endif
571
572#if ENABLE(WEBM_FORMAT_READER)
573 if (group.preferences().webMFormatReaderEnabled())
574 hasEnabledWebMFormatReader = true;
575#endif
576
577#if ENABLE(MEDIA_SOURCE) && ENABLE(VP9)
578 if (group.preferences().webMParserEnabled())
579 hasEnabledWebMParser = true;
580#endif
581 });
582
583#if ENABLE(MEDIA_SOURCE) && ENABLE(VP9)
584 send(Messages::GPUProcess::SetWebMParserEnabled(hasEnabledWebMParser), 0);
585#endif
586
587#if ENABLE(WEBM_FORMAT_READER)
588 send(Messages::GPUProcess::SetWebMFormatReaderEnabled(hasEnabledWebMFormatReader), 0);
589#endif
590
591#if ENABLE(OPUS)
592 send(Messages::GPUProcess::SetOpusDecoderEnabled(hasEnabledOpus), 0);
593#endif
594
595#if ENABLE(VORBIS)
596 send(Messages::GPUProcess::SetVorbisDecoderEnabled(hasEnabledVorbis), 0);
597#endif
598}

601{
