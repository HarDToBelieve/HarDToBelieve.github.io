34namespace WebKit {
35using namespace WebCore;
36
37UIGamepad::UIGamepad(WebCore::PlatformGamepad& platformGamepad)
38 : m_index(platformGamepad.index())
39 , m_id(platformGamepad.id())
40 , m_mapping(platformGamepad.mapping())
41 , m_lastUpdateTime(platformGamepad.lastUpdateTime())
42{
43 m_axisValues.resize(platformGamepad.axisValues().size());
44 m_buttonValues.resize(platformGamepad.buttonValues().size());
45
46 updateFromPlatformGamepad(platformGamepad);
47}
48
49void UIGamepad::updateFromPlatformGamepad(WebCore::PlatformGamepad& platformGamepad)
50{
51 ASSERT(m_index == platformGamepad.index());
52 ASSERT(m_axisValues.size() == platformGamepad.axisValues().size());
53 ASSERT(m_buttonValues.size() == platformGamepad.buttonValues().size());
54
55 m_axisValues = platformGamepad.axisValues();
56 m_buttonValues = platformGamepad.buttonValues();
57 m_lastUpdateTime = platformGamepad.lastUpdateTime();
58}
59
60GamepadData UIGamepad::gamepadData() const
61{
62 return { m_index, m_id, m_mapping, m_axisValues, m_buttonValues, m_lastUpdateTime };
63}
