50UIGamepadProvider::UIGamepadProvider()
51 : m_gamepadSyncTimer(RunLoop::main(), this, &UIGamepadProvider::gamepadSyncTimerFired)
52{
53 platformSetDefaultGamepadProvider();
54}

56UIGamepadProvider::~UIGamepadProvider()
57{
58 if (!m_processPoolsUsingGamepads.isEmpty())
59 GamepadProvider::singleton().stopMonitoringGamepads(*this);
60}

62void UIGamepadProvider::gamepadSyncTimerFired()
63{
64 auto webPageProxy = platformWebPageProxyForGamepadInput();
65 if (!webPageProxy || !m_processPoolsUsingGamepads.contains(&webPageProxy->process().processPool()))
66 return;
67
68 webPageProxy->gamepadActivity(snapshotGamepads(), m_shouldMakeGamepadsVisibleOnSync ? EventMakesGamepadsVisible::Yes : EventMakesGamepadsVisible::No);
69 m_shouldMakeGamepadsVisibleOnSync = false;
70}
71
72void UIGamepadProvider::scheduleGamepadStateSync()
73{
74 if (!m_isMonitoringGamepads || m_gamepadSyncTimer.isActive())
75 return;
76
77 if (m_gamepads.isEmpty() || m_processPoolsUsingGamepads.isEmpty()) {
78 m_gamepadSyncTimer.stop();
79 return;
80 }
81
82 m_gamepadSyncTimer.startOneShot(maximumGamepadUpdateInterval);
83}
84
85void UIGamepadProvider::platformGamepadConnected(PlatformGamepad& gamepad, EventMakesGamepadsVisible eventVisibility)
86{
87 LOG(Gamepad, "UIGamepadProvider::platformGamepadConnected - Index %i attached (visibility: %i)\n", gamepad.index(), (int)eventVisibility);
88
89 if (m_gamepads.size() <= gamepad.index())
90 m_gamepads.grow(gamepad.index() + 1);
91
92 ASSERT(!m_gamepads[gamepad.index()]);
93 m_gamepads[gamepad.index()] = makeUnique<UIGamepad>(gamepad);
94
95 scheduleGamepadStateSync();
96
97 for (auto& pool : m_processPoolsUsingGamepads)
98 pool->gamepadConnected(*m_gamepads[gamepad.index()], eventVisibility);
99}
100
101void UIGamepadProvider::platformGamepadDisconnected(PlatformGamepad& gamepad)
102{
103 ASSERT(gamepad.index() < m_gamepads.size());
104 ASSERT(m_gamepads[gamepad.index()]);
105
106 std::unique_ptr<UIGamepad> disconnectedGamepad = WTFMove(m_gamepads[gamepad.index()]);
107
108 scheduleGamepadStateSync();
109
110 for (auto& pool : m_processPoolsUsingGamepads)
111 pool->gamepadDisconnected(*disconnectedGamepad);
112}

114void UIGamepadProvider::platformGamepadInputActivity(EventMakesGamepadsVisible eventVisibility)
115{
116 auto platformGamepads = GamepadProvider::singleton().platformGamepads();
117 ASSERT(platformGamepads.size() == m_gamepads.size());
118
119 for (size_t i = 0; i < platformGamepads.size(); ++i) {
120 if (!platformGamepads[i]) {
121 ASSERT(!m_gamepads[i]);
122 continue;
123 }
124
125 ASSERT(m_gamepads[i]);
126 m_gamepads[i]->updateFromPlatformGamepad(*platformGamepads[i]);
127 }
128
129 if (eventVisibility == EventMakesGamepadsVisible::Yes)
130 m_shouldMakeGamepadsVisibleOnSync = true;
131
132 scheduleGamepadStateSync();
133}

135void UIGamepadProvider::processPoolStartedUsingGamepads(WebProcessPool& pool)
136{
137 ASSERT(!m_processPoolsUsingGamepads.contains(&pool));
138 m_processPoolsUsingGamepads.add(&pool);
139
140 if (!m_isMonitoringGamepads && platformWebPageProxyForGamepadInput())
141 startMonitoringGamepads();
142}
143
144void UIGamepadProvider::processPoolStoppedUsingGamepads(WebProcessPool& pool)
145{
146 ASSERT(m_processPoolsUsingGamepads.contains(&pool));
147 m_processPoolsUsingGamepads.remove(&pool);
148
149 if (m_isMonitoringGamepads && !platformWebPageProxyForGamepadInput())
150 platformStopMonitoringInput();
151}
152
153void UIGamepadProvider::viewBecameActive(WebPageProxy& page)
154{
155 if (!m_processPoolsUsingGamepads.contains(&page.process().processPool()))
156 return;
157
158 if (!m_isMonitoringGamepads)
159 startMonitoringGamepads();
160
161 if (platformWebPageProxyForGamepadInput())
162 platformStartMonitoringInput();
163}
164
165void UIGamepadProvider::viewBecameInactive(WebPageProxy& page)
166{
167 auto pageForGamepadInput = platformWebPageProxyForGamepadInput();
168 if (pageForGamepadInput == &page)
169 platformStopMonitoringInput();
170}
171
172void UIGamepadProvider::startMonitoringGamepads()
173{
174 if (m_isMonitoringGamepads)
175 return;
176
177 m_isMonitoringGamepads = true;
178 ASSERT(!m_processPoolsUsingGamepads.isEmpty());
179 GamepadProvider::singleton().startMonitoringGamepads(*this);
180}
181
182void UIGamepadProvider::stopMonitoringGamepads()
183{
184 if (!m_isMonitoringGamepads)
185 return;
186
187 m_isMonitoringGamepads = false;
188
189 ASSERT(m_processPoolsUsingGamepads.isEmpty());
190 GamepadProvider::singleton().stopMonitoringGamepads(*this);
191
192 m_gamepads.clear();
193}

195Vector<GamepadData> UIGamepadProvider::snapshotGamepads()
196{
197 Vector<GamepadData> gamepadDatas;
198 gamepadDatas.reserveInitialCapacity(m_gamepads.size());
199
200 for (auto& gamepad : m_gamepads) {
201 if (gamepad)
202 gamepadDatas.uncheckedAppend(gamepad->gamepadData());
203 else
204 gamepadDatas.uncheckedAppend({ });
205 }
206
207 return gamepadDatas;
208}
