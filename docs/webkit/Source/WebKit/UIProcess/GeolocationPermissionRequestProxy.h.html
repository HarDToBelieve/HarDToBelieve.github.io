32namespace WebKit {
33
34class GeolocationPermissionRequestManagerProxy;
35
36class GeolocationPermissionRequestProxy : public RefCounted<GeolocationPermissionRequestProxy> {
37public:
38 static Ref<GeolocationPermissionRequestProxy> create(GeolocationPermissionRequestManagerProxy* manager, GeolocationIdentifier geolocationID)
39 {
40 return adoptRef(*new GeolocationPermissionRequestProxy(manager, geolocationID));
41 }
42
43 void allow();
44 void deny();
45
46 void invalidate();
47
48private:
49 GeolocationPermissionRequestProxy(GeolocationPermissionRequestManagerProxy*, GeolocationIdentifier);
50
51 GeolocationPermissionRequestManagerProxy* m_manager;
52 GeolocationIdentifier m_geolocationID;
53};
54
55class GeolocationPermissionRequest : public API::ObjectImpl<API::Object::Type::GeolocationPermissionRequest> {
56public:
57 static Ref<GeolocationPermissionRequest> create(Function<void(bool)>&& completionHandler)
58 {
59 return adoptRef(*new GeolocationPermissionRequest(WTFMove(completionHandler)));
60 }
61
62 void allow() { m_completionHandler(true); }
63 void deny() { m_completionHandler(false); }
64
65private:
66 GeolocationPermissionRequest(Function<void(bool)>&& completionHandler)
67 : m_completionHandler(WTFMove(completionHandler))
68 { }
69
70 Function<void(bool)> m_completionHandler;
71};
