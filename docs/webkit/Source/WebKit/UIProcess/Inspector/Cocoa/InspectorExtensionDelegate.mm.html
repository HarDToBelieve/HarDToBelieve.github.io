38InspectorExtensionDelegate::InspectorExtensionDelegate(_WKInspectorExtension *inspectorExtension, id <_WKInspectorExtensionDelegate> delegate)
39 : m_inspectorExtension(inspectorExtension)
40 , m_delegate(delegate)
41{
42 m_delegateMethods.inspectorExtensionDidShowTabWithIdentifier = [delegate respondsToSelector:@selector(inspectorExtension:didShowTabWithIdentifier:)];
43 m_delegateMethods.inspectorExtensionDidHideTabWithIdentifier = [delegate respondsToSelector:@selector(inspectorExtension:didHideTabWithIdentifier:)];
44
45 inspectorExtension->_extension->setClient(makeUniqueRef<InspectorExtensionClient>(*this));
46}
47
48InspectorExtensionDelegate::~InspectorExtensionDelegate() = default;
49
50RetainPtr<id <_WKInspectorExtensionDelegate>> InspectorExtensionDelegate::delegate()
51{
52 return m_delegate.get();
53}

55InspectorExtensionDelegate::InspectorExtensionClient::InspectorExtensionClient(InspectorExtensionDelegate& delegate)
56 : m_inspectorExtensionDelegate(delegate)
57{
58}
59
60InspectorExtensionDelegate::InspectorExtensionClient::~InspectorExtensionClient()
61{
62}
63
64void InspectorExtensionDelegate::InspectorExtensionClient::didShowExtensionTab(const Inspector::ExtensionTabID& extensionTabID)
65{
66 if (!m_inspectorExtensionDelegate.m_delegateMethods.inspectorExtensionDidShowTabWithIdentifier)
67 return;
68
69 auto& delegate = m_inspectorExtensionDelegate.m_delegate;
70 if (!delegate)
71 return;
72
73 [delegate inspectorExtension:m_inspectorExtensionDelegate.m_inspectorExtension.get().get() didShowTabWithIdentifier:extensionTabID];
74}
75
76void InspectorExtensionDelegate::InspectorExtensionClient::didHideExtensionTab(const Inspector::ExtensionTabID& extensionTabID)
77{
78 if (!m_inspectorExtensionDelegate.m_delegateMethods.inspectorExtensionDidHideTabWithIdentifier)
79 return;
80
81 auto& delegate = m_inspectorExtensionDelegate.m_delegate;
82 if (!delegate)
83 return;
84
85 [delegate inspectorExtension:m_inspectorExtensionDelegate.m_inspectorExtension.get().get() didHideTabWithIdentifier:extensionTabID];
86}
