41typedef HashMap<WebPageProxy*, unsigned> PageLevelMap;
42
43static PageLevelMap& pageLevelMap()
44{
45 static NeverDestroyed<PageLevelMap> map;
46 return map;
47}
48
49unsigned inspectorLevelForPage(WebPageProxy* page)
50{
51 if (page) {
52 auto findResult = pageLevelMap().find(page);
53 if (findResult != pageLevelMap().end())
54 return findResult->value + 1;
55 }
56
57 return 1;
58}
59
60String defaultInspectorPageGroupIdentifierForPage(WebPageProxy* page)
61{
62 return makeString("__WebInspectorPageGroupLevel", inspectorLevelForPage(page), "__");
63}
64
65void trackInspectorPage(WebPageProxy* inspectorPage, WebPageProxy* inspectedPage)
66{
67 pageLevelMap().set(inspectorPage, inspectorLevelForPage(inspectedPage));
68}
69
70void untrackInspectorPage(WebPageProxy* inspectorPage)
71{
72 pageLevelMap().remove(inspectorPage);
73}
74
75static WebProcessPool* s_mainInspectorProcessPool;
76static WebProcessPool* s_nestedInspectorProcessPool;
77
78static WeakHashSet<WebProcessPool>& allInspectorProcessPools()
79{
80 static NeverDestroyed<WeakHashSet<WebProcessPool>> allInspectorProcessPools;
81 return allInspectorProcessPools.get();
82}
83
84WebProcessPool& defaultInspectorProcessPool(unsigned inspectionLevel)
85{
86 // Having our own process pool removes us from the main process pool and
87 // guarantees no process sharing for our user interface.
88 WebProcessPool*& pool = (inspectionLevel == 1) ? s_mainInspectorProcessPool : s_nestedInspectorProcessPool;
89 if (!pool) {
90 auto configuration = API::ProcessPoolConfiguration::create();
91 pool = &WebProcessPool::create(configuration.get()).leakRef();
92 prepareProcessPoolForInspector(*pool);
93 }
94 return *pool;
95}
96
97void prepareProcessPoolForInspector(WebProcessPool& processPool)
98{
99 // Do not delay process launch for inspector pages as inspector pages do not know how to transition from a terminated process.
100 processPool.disableDelayedWebProcessLaunch();
101
102 allInspectorProcessPools().add(processPool);
103}
104
105bool isInspectorProcessPool(WebProcessPool& processPool)
106{
107 return allInspectorProcessPools().contains(processPool);
108}
109
110bool isInspectorPage(WebPageProxy& webPage)
111{
112 return pageLevelMap().contains(&webPage);
113}
