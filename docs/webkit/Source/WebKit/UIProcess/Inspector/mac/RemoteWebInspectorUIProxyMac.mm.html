103WebPageProxy* RemoteWebInspectorUIProxy::platformCreateFrontendPageAndWindow()
104{
105 m_objCAdapter = adoptNS([[WKRemoteWebInspectorUIProxyObjCAdapter alloc] initWithRemoteWebInspectorUIProxy:this]);
106
107 Ref<API::InspectorConfiguration> configuration = m_client->configurationForRemoteInspector(*this);
108 m_inspectorView = adoptNS([[WKInspectorViewController alloc] initWithConfiguration: WebKit::wrapper(configuration) inspectedPage:nullptr]);
109 [m_inspectorView.get() setDelegate:m_objCAdapter.get()];
110
111 m_window = WebInspectorUIProxy::createFrontendWindow(NSZeroRect, WebInspectorUIProxy::InspectionTargetType::Remote);
112 [m_window setDelegate:m_objCAdapter.get()];
113 [m_window setFrameAutosaveName:@"WKRemoteWebInspectorWindowFrame"];
114
115 NSView *contentView = m_window.get().contentView;
116 [webView() setFrame:contentView.bounds];
117 [contentView addSubview:webView()];
118
119 return webView()->_page.get();
120}

150void RemoteWebInspectorUIProxy::platformSave(const String& suggestedURL, const String& content, bool base64Encoded, bool forceSaveDialog)
151{
152 // FIXME: Share with WebInspectorUIProxyMac.
153
154 ASSERT(!suggestedURL.isEmpty());
155
156 NSURL *platformURL = m_suggestedToActualURLMap.get(suggestedURL).get();
157 if (!platformURL) {
158 platformURL = [NSURL URLWithString:suggestedURL];
159 // The user must confirm new filenames before we can save to them.
160 forceSaveDialog = true;
161 }
162
163 ASSERT(platformURL);
164 if (!platformURL)
165 return;
166
167 // Necessary for the block below.
168 String suggestedURLCopy = suggestedURL;
169 String contentCopy = content;
170
171 auto saveToURL = ^(NSURL *actualURL) {
172 ASSERT(actualURL);
173
174 m_suggestedToActualURLMap.set(suggestedURLCopy, actualURL);
175
176 if (base64Encoded) {
177 auto decodedData = base64Decode(contentCopy, Base64DecodeOptions::ValidatePadding);
178 if (!decodedData)
179 return;
180 auto dataContent = adoptNS([[NSData alloc] initWithBytes:decodedData->data() length:decodedData->size()]);
181 [dataContent writeToURL:actualURL atomically:YES];
182 } else
183 [contentCopy writeToURL:actualURL atomically:YES encoding:NSUTF8StringEncoding error:NULL];
184
185 m_inspectorPage->send(Messages::RemoteWebInspectorUI::DidSave([actualURL absoluteString]));
186 };
187
188 if (!forceSaveDialog) {
189 saveToURL(platformURL);
190 return;
191 }
192
193 NSSavePanel *panel = [NSSavePanel savePanel];
194 panel.nameFieldStringValue = platformURL.lastPathComponent;
195
196 // If we have a file URL we've already saved this file to a path and
197 // can provide a good directory to show. Otherwise, use the system's
198 // default behavior for the initial directory to show in the dialog.
199 if (platformURL.isFileURL)
200 panel.directoryURL = [platformURL URLByDeletingLastPathComponent];
201
202 auto completionHandler = ^(NSInteger result) {
203 if (result == NSModalResponseCancel)
204 return;
205 ASSERT(result == NSModalResponseOK);
206 saveToURL(panel.URL);
207 };
208
209 NSWindow *window = m_window ? m_window.get() : [NSApp keyWindow];
210 if (window)
211 [panel beginSheetModalForWindow:window completionHandler:completionHandler];
212 else
213 completionHandler([panel runModal]);
214}

216void RemoteWebInspectorUIProxy::platformAppend(const String& suggestedURL, const String& content)
217{
218 // FIXME: Share with WebInspectorUIProxyMac.
219
220 ASSERT(!suggestedURL.isEmpty());
221
222 RetainPtr<NSURL> actualURL = m_suggestedToActualURLMap.get(suggestedURL);
223 // Do not append unless the user has already confirmed this filename in save().
224 if (!actualURL)
225 return;
226
227 NSFileHandle *handle = [NSFileHandle fileHandleForWritingToURL:actualURL.get() error:NULL];
228 [handle seekToEndOfFile];
229 [handle writeData:[content dataUsingEncoding:NSUTF8StringEncoding]];
230 [handle closeFile];
231
232 WebPageProxy* inspectorPage = webView()->_page.get();
233 inspectorPage->send(Messages::RemoteWebInspectorUI::DidAppend([actualURL absoluteString]));
234}

241void RemoteWebInspectorUIProxy::platformSetForcedAppearance(InspectorFrontendClient::Appearance appearance)
242{
243 NSAppearance *platformAppearance;
244 switch (appearance) {
245 case InspectorFrontendClient::Appearance::System:
246 platformAppearance = nil;
247 break;
248
249 case InspectorFrontendClient::Appearance::Light:
250 platformAppearance = [NSAppearance appearanceNamed:NSAppearanceNameAqua];
251 break;
252
253 case InspectorFrontendClient::Appearance::Dark:
254 platformAppearance = [NSAppearance appearanceNamed:NSAppearanceNameDarkAqua];
255 break;
256 }
257
258 webView().appearance = platformAppearance;
259
260 NSWindow *window = m_window.get();
261 ASSERT(window);
262 window.appearance = platformAppearance;
263}

275void RemoteWebInspectorUIProxy::platformShowCertificate(const CertificateInfo& certificateInfo)
276{
277 ASSERT(!certificateInfo.isEmpty());
278
279 RetainPtr<SFCertificatePanel> certificatePanel = adoptNS([[SFCertificatePanel alloc] init]);
280
281 ASSERT(m_window);
282#if HAVE(SEC_TRUST_SERIALIZATION)
283 [certificatePanel beginSheetForWindow:m_window.get() modalDelegate:nil didEndSelector:NULL contextInfo:nullptr trust:certificateInfo.trust() showGroup:YES];
284#else
285 [certificatePanel beginSheetForWindow:m_window.get() modalDelegate:nil didEndSelector:NULL contextInfo:nullptr certificates:(NSArray *)certificateInfo.certificateChain() showGroup:YES];
286#endif
287
288 // This must be called after the trust panel has been displayed, because the certificateView doesn't exist beforehand.
289 SFCertificateView *certificateView = [certificatePanel certificateView];
290 [certificateView setDisplayTrust:YES];
291 [certificateView setEditableTrust:NO];
292 [certificateView setDisplayDetails:YES];
293 [certificateView setDetailsDisclosed:YES];
294}
