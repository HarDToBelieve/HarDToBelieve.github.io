26#pragma once
27
28#include "CacheModel.h"
29#include <wtf/Forward.h>
30#include <wtf/HashSet.h>
31#include <wtf/text/WTFString.h>
32
33namespace WebKit {
34
35class LegacyGlobalSettings {
36 WTF_MAKE_FAST_ALLOCATED;
37public:
38 static LegacyGlobalSettings& singleton();
39
40 void setCacheModel(CacheModel);
41 CacheModel cacheModel() const { return m_cacheModel; }
42
43 const HashSet<String>& schemesToRegisterAsSecure() { return m_schemesToRegisterAsSecure; }
44 void registerURLSchemeAsSecure(const String& scheme) { m_schemesToRegisterAsSecure.add(scheme); }
45
46 const HashSet<String>& schemesToRegisterAsBypassingContentSecurityPolicy() { return m_schemesToRegisterAsBypassingContentSecurityPolicy; }
47 void registerURLSchemeAsBypassingContentSecurityPolicy(const String& scheme) { m_schemesToRegisterAsBypassingContentSecurityPolicy.add(scheme); }
48
49 const HashSet<String>& schemesToRegisterAsLocal() { return m_schemesToRegisterAsLocal; }
50 void registerURLSchemeAsLocal(const String& scheme) { m_schemesToRegisterAsLocal.add(scheme); }
51
52 const HashSet<String>& schemesToRegisterAsNoAccess() { return m_schemesToRegisterAsNoAccess; }
53 void registerURLSchemeAsNoAccess(const String& scheme) { m_schemesToRegisterAsNoAccess.add(scheme); }
54
55private:
56 friend class NeverDestroyed<LegacyGlobalSettings>;
57 LegacyGlobalSettings();
58
59 CacheModel m_cacheModel { CacheModel::PrimaryWebBrowser };
60 HashSet<String> m_schemesToRegisterAsSecure;
61 HashSet<String> m_schemesToRegisterAsBypassingContentSecurityPolicy;
62 HashSet<String> m_schemesToRegisterAsLocal;
63 HashSet<String> m_schemesToRegisterAsNoAccess;
64};
65
66} // namespace WebKit
