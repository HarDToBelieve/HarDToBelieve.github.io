41AudioSessionRoutingArbitratorProxy::AudioSessionRoutingArbitratorProxy(WebProcessProxy& proxy)
42 : m_process(proxy)
43 , m_token(SharedRoutingArbitrator::Token::create())
44{
45 m_process.addMessageReceiver(Messages::AudioSessionRoutingArbitratorProxy::messageReceiverName(), destinationId(), *this);
46}
47
48AudioSessionRoutingArbitratorProxy::~AudioSessionRoutingArbitratorProxy()
49{
50 m_process.removeMessageReceiver(Messages::AudioSessionRoutingArbitratorProxy::messageReceiverName(), destinationId());
51}
52
53void AudioSessionRoutingArbitratorProxy::processDidTerminate()
54{
55 if (SharedRoutingArbitrator::sharedInstance().isInRoutingArbitrationForToken(m_token))
56 endRoutingArbitration();
57}
58
59void AudioSessionRoutingArbitratorProxy::beginRoutingArbitrationWithCategory(WebCore::AudioSession::CategoryType category, ArbitrationCallback&& callback)
60{
61 m_category = category;
62 m_arbitrationStatus = ArbitrationStatus::Pending;
63 m_arbitrationUpdateTime = WallTime::now();
64 SharedRoutingArbitrator::sharedInstance().beginRoutingArbitrationForToken(m_token, category, [weakThis = makeWeakPtr(*this), callback = WTFMove(callback)] (RoutingArbitrationError error, DefaultRouteChanged routeChanged) mutable {
65 if (weakThis)
66 weakThis->m_arbitrationStatus = error == RoutingArbitrationError::None ? ArbitrationStatus::Active : ArbitrationStatus::None;
67 callback(error, routeChanged);
68 });
69}
70
71void AudioSessionRoutingArbitratorProxy::endRoutingArbitration()
72{
73 SharedRoutingArbitrator::sharedInstance().endRoutingArbitrationForToken(m_token);
74 m_arbitrationStatus = ArbitrationStatus::None;
75}
