47static bool usageTrackingAvailable()
48{
49 static bool available;
50
51 static std::once_flag onceFlag;
52 std::call_once(onceFlag, [&] () {
53 available = PAL::isUsageTrackingFrameworkAvailable()
54 && PAL::getUSVideoUsageClass()
55 && PAL::canLoad_UsageTracking_USVideoMetadataKeyCanShowControlsManager()
56 && PAL::canLoad_UsageTracking_USVideoMetadataKeyCanShowNowPlayingControls()
57 && PAL::canLoad_UsageTracking_USVideoMetadataKeyIsSuspended()
58 && PAL::canLoad_UsageTracking_USVideoMetadataKeyIsInActiveDocument()
59 && PAL::canLoad_UsageTracking_USVideoMetadataKeyIsFullscreen()
60 && PAL::canLoad_UsageTracking_USVideoMetadataKeyIsMuted()
61 && PAL::canLoad_UsageTracking_USVideoMetadataKeyIsMediaDocumentInMainFrame()
62 && PAL::canLoad_UsageTracking_USVideoMetadataKeyIsAudio()
63 && PAL::canLoad_UsageTracking_USVideoMetadataKeyAudioElementWithUserGesture()
64 && PAL::canLoad_UsageTracking_USVideoMetadataKeyUserHasPlayedAudioBefore()
65 && PAL::canLoad_UsageTracking_USVideoMetadataKeyIsElementRectMostlyInMainFrame()
66 && PAL::canLoad_UsageTracking_USVideoMetadataKeyNoAudio()
67 && PAL::canLoad_UsageTracking_USVideoMetadataKeyPlaybackPermitted()
68 && PAL::canLoad_UsageTracking_USVideoMetadataKeyPageMediaPlaybackSuspended()
69 && PAL::canLoad_UsageTracking_USVideoMetadataKeyIsMediaDocumentAndNotOwnerElement()
70 && PAL::canLoad_UsageTracking_USVideoMetadataKeyPageExplicitlyAllowsElementToAutoplayInline()
71 && PAL::canLoad_UsageTracking_USVideoMetadataKeyRequiresFullscreenForVideoPlaybackAndFullscreenNotPermitted()
72 && PAL::canLoad_UsageTracking_USVideoMetadataKeyHasHadUserInteractionAndQuirksContainsShouldAutoplayForArbitraryUserGesture()
73 && PAL::canLoad_UsageTracking_USVideoMetadataKeyIsVideoAndRequiresUserGestureForVideoRateChange()
74 && PAL::canLoad_UsageTracking_USVideoMetadataKeyIsAudioAndRequiresUserGestureForAudioRateChange()
75 && PAL::canLoad_UsageTracking_USVideoMetadataKeyIsVideoAndRequiresUserGestureForVideoDueToLowPowerMode()
76 && PAL::canLoad_UsageTracking_USVideoMetadataKeyNoUserGestureRequired()
77 && PAL::canLoad_UsageTracking_USVideoMetadataKeyRequiresPlaybackAndIsNotPlaying()
78 && PAL::canLoad_UsageTracking_USVideoMetadataKeyHasEverNotifiedAboutPlaying()
79 && PAL::canLoad_UsageTracking_USVideoMetadataKeyOutsideOfFullscreen()
80 && PAL::canLoad_UsageTracking_USVideoMetadataKeyIsVideo()
81 && PAL::canLoad_UsageTracking_USVideoMetadataKeyRenderer()
82 && PAL::canLoad_UsageTracking_USVideoMetadataKeyNoVideo()
83 && PAL::canLoad_UsageTracking_USVideoMetadataKeyIsLargeEnoughForMainContent();
84 });
85
86 return available;
87}

94MediaUsageManagerCocoa::~MediaUsageManagerCocoa()
95{
96 reset();
97}

99void MediaUsageManagerCocoa::reset()
100{
101 for (auto& session : m_mediaSessions.values()) {
102 if (session->usageTracker && session->mediaUsageInfo && session->mediaUsageInfo->isPlaying)
103 [session->usageTracker stop];
104 }
105 m_mediaSessions.clear();
106}

108void MediaUsageManagerCocoa::addMediaSession(WebCore::MediaSessionIdentifier identifier, const String& bundleIdentifier, const URL& pageURL)
109{
110 auto addResult = m_mediaSessions.ensure(identifier, [&] {
111 return makeUnique<MediaUsageManagerCocoa::SessionMediaUsage>(identifier, bundleIdentifier, pageURL);
112 });
113 ASSERT_UNUSED(addResult, addResult.isNewEntry);
114}

116void MediaUsageManagerCocoa::removeMediaSession(WebCore::MediaSessionIdentifier identifier)
117{
118 ASSERT(m_mediaSessions.contains(identifier));
119 m_mediaSessions.remove(identifier);
120}

122void MediaUsageManagerCocoa::updateMediaUsage(WebCore::MediaSessionIdentifier identifier, const WebCore::MediaUsageInfo& mediaUsageInfo)
123{
124 ASSERT(m_mediaSessions.contains(identifier));
125 auto session = m_mediaSessions.get(identifier);
126 if (!session)
127 return;
128
129 if (!usageTrackingAvailable())
130 return;
131
132 @try {
133
134 if (session->mediaUsageInfo) {
135 if (session->mediaUsageInfo == mediaUsageInfo)
136 return;
137
138 if (session->usageTracker && session->mediaUsageInfo->mediaURL != mediaUsageInfo.mediaURL) {
139 [session->usageTracker stop];
140 session->usageTracker = nullptr;
141 }
142 }
143
144 NSDictionary<NSString *, id> *metadata = @{
145 USVideoMetadataKeyCanShowControlsManager: @(mediaUsageInfo.canShowControlsManager),
146 USVideoMetadataKeyCanShowNowPlayingControls: @(mediaUsageInfo.canShowNowPlayingControls),
147 USVideoMetadataKeyIsSuspended: @(mediaUsageInfo.isSuspended),
148 USVideoMetadataKeyIsInActiveDocument: @(mediaUsageInfo.isInActiveDocument),
149 USVideoMetadataKeyIsFullscreen: @(mediaUsageInfo.isFullscreen),
150 USVideoMetadataKeyIsMuted: @(mediaUsageInfo.isMuted),
151 USVideoMetadataKeyIsMediaDocumentInMainFrame: @(mediaUsageInfo.isMediaDocumentInMainFrame),
152 USVideoMetadataKeyIsVideo: @(mediaUsageInfo.isVideo),
153 USVideoMetadataKeyIsAudio: @(mediaUsageInfo.isAudio),
154 USVideoMetadataKeyNoVideo: @(!mediaUsageInfo.hasVideo),
155 USVideoMetadataKeyNoAudio: @(!mediaUsageInfo.hasAudio),
156 USVideoMetadataKeyRenderer: @(mediaUsageInfo.hasRenderer),
157 USVideoMetadataKeyAudioElementWithUserGesture: @(mediaUsageInfo.audioElementWithUserGesture),
158 USVideoMetadataKeyUserHasPlayedAudioBefore: @(mediaUsageInfo.userHasPlayedAudioBefore),
159 USVideoMetadataKeyIsElementRectMostlyInMainFrame: @(mediaUsageInfo.isElementRectMostlyInMainFrame),
160 USVideoMetadataKeyPlaybackPermitted: @(mediaUsageInfo.playbackPermitted),
161 USVideoMetadataKeyPageMediaPlaybackSuspended: @(mediaUsageInfo.pageMediaPlaybackSuspended),
162 USVideoMetadataKeyIsMediaDocumentAndNotOwnerElement: @(mediaUsageInfo.isMediaDocumentAndNotOwnerElement),
163 USVideoMetadataKeyPageExplicitlyAllowsElementToAutoplayInline: @(mediaUsageInfo.pageExplicitlyAllowsElementToAutoplayInline),
164 USVideoMetadataKeyRequiresFullscreenForVideoPlaybackAndFullscreenNotPermitted: @(mediaUsageInfo.requiresFullscreenForVideoPlaybackAndFullscreenNotPermitted),
165 USVideoMetadataKeyHasHadUserInteractionAndQuirksContainsShouldAutoplayForArbitraryUserGesture: @(mediaUsageInfo.hasHadUserInteractionAndQuirksContainsShouldAutoplayForArbitraryUserGesture),
166 USVideoMetadataKeyIsVideoAndRequiresUserGestureForVideoRateChange: @(mediaUsageInfo.isVideoAndRequiresUserGestureForVideoRateChange),
167 USVideoMetadataKeyIsAudioAndRequiresUserGestureForAudioRateChange: @(mediaUsageInfo.isAudioAndRequiresUserGestureForAudioRateChange),
168 USVideoMetadataKeyIsVideoAndRequiresUserGestureForVideoDueToLowPowerMode: @(mediaUsageInfo.isVideoAndRequiresUserGestureForVideoDueToLowPowerMode),
169 USVideoMetadataKeyNoUserGestureRequired: @(mediaUsageInfo.noUserGestureRequired),
170 USVideoMetadataKeyRequiresPlaybackAndIsNotPlaying: @(mediaUsageInfo.requiresPlaybackAndIsNotPlaying),
171 USVideoMetadataKeyHasEverNotifiedAboutPlaying: @(mediaUsageInfo.hasEverNotifiedAboutPlaying),
172 USVideoMetadataKeyOutsideOfFullscreen: @(mediaUsageInfo.outsideOfFullscreen),
173 USVideoMetadataKeyIsLargeEnoughForMainContent: @(mediaUsageInfo.isLargeEnoughForMainContent),
174 };
175
176 if (!session->usageTracker) {
177 if (!mediaUsageInfo.isPlaying)
178 return;
179
180 session->usageTracker = adoptNS([PAL::allocUSVideoUsageInstance() initWithBundleIdentifier:session->bundleIdentifier URL:(NSURL *)session->pageURL
181 mediaURL:(NSURL *)mediaUsageInfo.mediaURL videoMetadata:metadata]);
182 ASSERT(session->usageTracker);
183 if (!session->usageTracker)
184 return;
185 } else
186 [session->usageTracker updateVideoMetadata:metadata];
187
188 if (session->mediaUsageInfo && session->mediaUsageInfo->isPlaying != mediaUsageInfo.isPlaying) {
189 if (mediaUsageInfo.isPlaying)
190 [session->usageTracker restart];
191 else
192 [session->usageTracker stop];
193 }
194
195 session->mediaUsageInfo = mediaUsageInfo;
196
197 } @catch(NSException *exception) {
198 WTFLogAlways("MediaUsageManagerCocoa::updateMediaUsage caught exception: %@", [[exception reason] UTF8String]);
199 }
200}
