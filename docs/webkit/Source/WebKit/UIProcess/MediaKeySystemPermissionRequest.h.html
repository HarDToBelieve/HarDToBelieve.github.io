34class MediaKeySystemPermissionRequest : public RefCounted<MediaKeySystemPermissionRequest> {
35public:
36 static Ref<MediaKeySystemPermissionRequest> create(const String& keySystem, CompletionHandler<void(bool)>&& completionHandler)
37 {
38 return adoptRef(*new MediaKeySystemPermissionRequest(keySystem, WTFMove(completionHandler)));
39 }
40
41 void complete(bool success)
42 {
43 auto completionHandler = std::exchange(m_completionHandler, { });
44 completionHandler(success);
45 }
46
47 const String& keySystem() const { return m_keySystem; }
48
49private:
50 MediaKeySystemPermissionRequest(const String& keySystem, CompletionHandler<void(bool)>&& completionHandler)
51 : m_keySystem(keySystem)
52 , m_completionHandler(WTFMove(completionHandler))
53 { }
54
55 String m_keySystem;
56 CompletionHandler<void(bool)> m_completionHandler;
57};
58
59class MediaKeySystemPermissionCallback : public API::ObjectImpl<API::Object::Type::MediaKeySystemPermissionCallback> {
60public:
61 static Ref<MediaKeySystemPermissionCallback> create(CompletionHandler<void(bool)>&& completionHandler)
62 {
63 return adoptRef(*new MediaKeySystemPermissionCallback(WTFMove(completionHandler)));
64 }
65
66 void complete(bool granted) { m_completionHandler(granted); }
67
68private:
69 MediaKeySystemPermissionCallback(CompletionHandler<void(bool)>&& completionHandler)
70 : m_completionHandler(WTFMove(completionHandler))
71 { }
72
73 CompletionHandler<void(bool)> m_completionHandler;
74};
