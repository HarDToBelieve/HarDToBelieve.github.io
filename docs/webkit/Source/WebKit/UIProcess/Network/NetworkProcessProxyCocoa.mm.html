42bool NetworkProcessProxy::XPCEventHandler::handleXPCEvent(xpc_object_t event) const
43{
44 if (!m_networkProcess)
45 return false;
46
47 if (!event || xpc_get_type(event) == XPC_TYPE_ERROR)
48 return false;
49
50 String messageName = xpc_dictionary_get_string(event, XPCEndpoint::xpcMessageNameKey);
51 if (messageName.isEmpty())
52 return false;
53
54 if (messageName == LaunchServicesDatabaseXPCConstants::xpcLaunchServicesDatabaseXPCEndpointMessageName) {
55 m_networkProcess->m_endpointMessage = event;
56 for (auto& dataStore : copyToVectorOf<Ref<WebsiteDataStore>>(m_networkProcess->m_websiteDataStores))
57 dataStore->sendNetworkProcessXPCEndpointToAllWebProcesses();
58 }
59
60 return true;
61}
