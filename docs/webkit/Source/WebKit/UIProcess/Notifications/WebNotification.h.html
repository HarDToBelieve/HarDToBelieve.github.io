27
28#include "APIObject.h"
29#include "APISecurityOrigin.h"
30#include <wtf/RefPtr.h>
31#include <wtf/text/WTFString.h>
32
33namespace WebCore {
34enum class NotificationDirection : uint8_t;
35}
36
37namespace WebKit {
38
39class WebNotification : public API::ObjectImpl<API::Object::Type::Notification> {
40public:
41 static Ref<WebNotification> create(const String& title, const String& body, const String& iconURL, const String& tag, const String& lang, WebCore::NotificationDirection dir, const String& originString, uint64_t notificationID)
42 {
43 return adoptRef(*new WebNotification(title, body, iconURL, tag, lang, dir, originString, notificationID));
44 }
45
46 const String& title() const { return m_title; }
47 const String& body() const { return m_body; }
48 const String& iconURL() const { return m_iconURL; }
49 const String& tag() const { return m_tag; }
50 const String& lang() const { return m_lang; }
51 WebCore::NotificationDirection dir() const { return m_dir; }
52 API::SecurityOrigin* origin() const { return m_origin.get(); }
53
54 uint64_t notificationID() const { return m_notificationID; }
55
56private:
57 WebNotification(const String& title, const String& body, const String& iconURL, const String& tag, const String& lang, WebCore::NotificationDirection, const String& originString, uint64_t notificationID);
58
59 String m_title;
60 String m_body;
61 String m_iconURL;
62 String m_tag;
63 String m_lang;
64 WebCore::NotificationDirection m_dir;
65 RefPtr<API::SecurityOrigin> m_origin;
66 uint64_t m_notificationID;
67};
68
69inline bool isNotificationIDValid(uint64_t id)
70{
71 // This check makes sure that the ID is not equal to values needed by
72 // HashMap for bucketing.
73 return id && id != static_cast<uint64_t>(-1);
74}
