38// The HUD items should have the following spacing:
39// -------------------------------------------------
40// | 12 12 10 12 12 |
41// | ---- ---- | ---- ---- |
42// | 10 |icon| 10 |icon| 10 | 10 |icon| 10 |icon| 10 |
43// | ---- ---- | ---- ---- |
44// | 12 12 10 12 12 |
45// -------------------------------------------------
46// where the 12 point vertical spacing is anchored to the smallest icon image,
47// and all subsequent icons with be centered vertically with the smallest icon.
48
49static const CGFloat layerVerticalOffset = 40;
50static const CGFloat layerCornerRadius = 12;
51static const CGFloat layerGrayComponent = 0;
52static const CGFloat layerAlpha = 0.75;
53static const CGFloat layerImageScale = 1.5;
54static const CGFloat layerSeperatorControllerSize = 1.5;
55static const CGFloat layerControllerHorizontalMargin = 10.0;
56static const CGFloat layerImageVerticalMargin = 12.0;
57static const CGFloat layerSeperatorVerticalMargin = 10.0;
58static const CGFloat controlLayerNormalAlpha = 0.75;
59static const CGFloat controlLayerDownAlpha = 0.45;
60
61static NSString * const PDFHUDZoomInControl = @"plus.magnifyingglass";
62static NSString * const PDFHUDZoomOutControl = @"minus.magnifyingglass";
63static NSString * const PDFHUDLaunchPreviewControl = @"preview";
64static NSString * const PDFHUDSavePDFControl = @"arrow.down.circle";
65static NSString * const PDFHUDSeperatorControl = @"PDFHUDSeperatorControl";
66
67static const CGFloat layerFadeInTimeInterval = 0.25;
68static const CGFloat layerFadeOutTimeInterval = 0.5;
69static const CGFloat initialHideTimeInterval = 3.0;
70
71static NSArray<NSString *> *controlArray()
72{
73 return @[
74 PDFHUDZoomOutControl,
75 PDFHUDZoomInControl,
76 PDFHUDSeperatorControl,
77 PDFHUDLaunchPreviewControl,
78 PDFHUDSavePDFControl
79 ];
80}

82@implementation WKPDFHUDView {
83@private
84 WeakPtr<WebKit::WebPageProxy> _page;
85 RetainPtr<NSString> _activeControl;
86 WebKit::PDFPluginIdentifier _pluginIdentifier;
87 CGFloat _deviceScaleFactor;
88 RetainPtr<CALayer> _layer;
89 RetainPtr<CALayer> _activeLayer;
90 CGSize _frameSize;
91 RetainPtr<NSMutableDictionary<NSString *, NSImage *>> _cachedIcons;
92 BOOL _visible;
93 BOOL _mouseMovedToHUD;
94 BOOL _initialHideTimerFired;
95}
96
97- (instancetype)initWithFrame:(NSRect)frame pluginIdentifier:(WebKit::PDFPluginIdentifier)pluginIdentifier page:(WebKit::WebPageProxy&)page
98{
99 if (!(self = [super initWithFrame:frame]))
100 return nil;
101
102 self.wantsLayer = YES;
103 _cachedIcons = adoptNS([[NSMutableDictionary alloc] init]);
104 _pluginIdentifier = pluginIdentifier;
105 _page = makeWeakPtr(page);
106 _deviceScaleFactor = page.deviceScaleFactor();
107 _visible = YES;
108 [self _setupLayer:self.layer];
109 [self setFrame:frame];
110
111 WeakObjCPtr<WKPDFHUDView> weakSelf = self;
112 WorkQueue::main().dispatchAfter(Seconds { initialHideTimeInterval }, [weakSelf] {
113 [weakSelf _hideTimerFired];
114 });
115 return self;
116}
117
118- (void)dealloc
119{
120 [_layer removeFromSuperlayer];
121 [super dealloc];
122}
123
124- (void)setFrame:(NSRect)rect
125{
126 [super setFrame:rect];
127 _frameSize = rect.size;
128
129 [CATransaction begin];
130 [CATransaction setDisableActions:YES];
131 CGRect layerBounds = [_layer bounds];
132 [_layer setFrame:CGRectMake(rect.size.width / 2.0 - layerBounds.size.width / 2.0, layerVerticalOffset, layerBounds.size.width, layerBounds.size.height)];
133 [CATransaction commit];
134}
135
136- (void)setDeviceScaleFactor:(CGFloat)deviceScaleFactor
137{
138 if (_deviceScaleFactor == deviceScaleFactor)
139 return;
140
141 _deviceScaleFactor = deviceScaleFactor;
142
143 [self _redrawLayer];
144}
145
146- (void)_hideTimerFired
147{
148 _initialHideTimerFired = YES;
149 if (!_mouseMovedToHUD)
150 [self _setVisible:false];
151}
152
153- (void)_setVisible:(bool)isVisible
154{
155 if (_visible == isVisible)
156 return;
157 _visible = isVisible;
158 [CATransaction begin];
159 [CATransaction setAnimationDuration:isVisible ? layerFadeInTimeInterval : layerFadeOutTimeInterval];
160 [self _setLayerOpacity:isVisible ? layerAlpha : 0.0];
161 [CATransaction commit];
162}
163
164- (NSView *)hitTest:(NSPoint)point
165{
166 ASSERT(_page);
167 return _page ? _page->cocoaView().autorelease() : self;
168}
169
170- (void)mouseMoved:(NSEvent *)event
171{
172 if (CGRectContainsPoint([self convertRect:CGRectInset([_layer frame], -16.0, -16.0) toView:nil], NSPointToCGPoint(event.locationInWindow))) {
173 [self _setVisible:true];
174 _mouseMovedToHUD = YES;
175 } else if (_initialHideTimerFired)
176 [self _setVisible:false];
177}
178
179- (void)mouseDown:(NSEvent *)event
180{
181 _activeControl = [self _controlForEvent:event];
182 if ([_activeControl isEqualToString:PDFHUDSeperatorControl])
183 _activeControl = nil;
184 if (_activeControl) {
185 // Update rendering to highlight it..
186 _activeLayer = [self _layerForEvent:event];
187
188 // Update layer image; do not animate
189 [CATransaction begin];
190 [CATransaction setDisableActions:YES];
191
192 [_activeLayer setOpacity:controlLayerDownAlpha];
193
194 [CATransaction commit];
195 }
196}
197
198- (void)mouseUp:(NSEvent *)event
199{
200 if (!_activeControl)
201 return;
202
203 NSString* mouseUpControl = [self _controlForEvent:event];
204 if ([_activeControl isEqualToString:mouseUpControl])
205 [self _performActionForControl:_activeControl.get()];
206
207 [CATransaction begin];
208 [CATransaction setDisableActions:YES];
209 [_activeLayer setOpacity:controlLayerNormalAlpha];
210 [CATransaction commit];
211
212 _activeLayer = nil;
213 _activeControl = nil;
214}
215
216- (std::optional<NSUInteger>)_controlIndexForEvent:(NSEvent *)event
217{
218 CGPoint initialPoint = NSPointToCGPoint(event.locationInWindow);
219 initialPoint.x -= [_layer frame].origin.x;
220 initialPoint.y -= [_layer frame].origin.y;
221 for (NSUInteger index = 0; index < [_layer sublayers].count; index++) {
222 CALayer *subLayer = [_layer sublayers][index];
223 NSRect windowSpaceRect = [self convertRect:subLayer.frame toView:nil];
224 if (CGRectContainsPoint(windowSpaceRect, initialPoint))
225 return index;
226 }
227 return std::nullopt;
228}
229
230- (NSString *)_controlForEvent:(NSEvent *)event
231{
232 if (auto index = [self _controlIndexForEvent:event])
233 return controlArray()[*index];
234 return nil;
235}
236
237- (CALayer *)_layerForEvent:(NSEvent *)event
238{
239 if (auto index = [self _controlIndexForEvent:event])
240 return [_layer sublayers][*index];
241 return nil;
242}
243
244- (void)_performActionForControl:(NSString *)control
245{
246 if (!_visible)
247 return;
248 auto* page = _page.get();
249 if (!page)
250 return;
251 if ([control isEqualToString:PDFHUDZoomInControl])
252 page->pdfZoomIn(_pluginIdentifier);
253 else if ([control isEqualToString:PDFHUDZoomOutControl])
254 page->pdfZoomOut(_pluginIdentifier);
255 else if ([control isEqualToString:PDFHUDSavePDFControl])
256 page->pdfSaveToPDF(_pluginIdentifier);
257 else if ([control isEqualToString:PDFHUDLaunchPreviewControl])
258 page->pdfOpenWithPreview(_pluginIdentifier);
259}
260
261- (void)_loadIconImages
262{
263 for (NSString *controlName in controlArray())
264 [self _getImageForControlName:controlName];
265}
266
267- (void)_setupLayer:(CALayer *)parentLayer
268{
269 _layer = adoptNS([[CALayer alloc] init]);
270 [_layer setCornerRadius:layerCornerRadius];
271
272 [_layer setBackgroundColor:WebCore::cachedCGColor({ WebCore::SRGBA<float>(layerGrayComponent, layerGrayComponent, layerGrayComponent) })];
273 [self _setLayerOpacity:layerAlpha];
274
275 [self _loadIconImages];
276 CGFloat minIconImageHeight = std::numeric_limits<CGFloat>::max();
277 for (NSImage *image in [_cachedIcons allValues])
278 minIconImageHeight = std::min(minIconImageHeight, (image.size.height / _deviceScaleFactor));
279
280 CGFloat dx = layerControllerHorizontalMargin;
281 for (NSString *controlName in controlArray()) {
282 auto controlLayer = adoptNS([[CALayer alloc] init]);
283 CGFloat dy = 0.0;
284 CGFloat controllerWidth = 0.0;
285 CGFloat controllerHeight = 0.0;
286
287 if ([controlName isEqualToString:PDFHUDSeperatorControl]) {
288 dy = layerSeperatorVerticalMargin;
289 controllerWidth = layerSeperatorControllerSize;
290 controllerHeight = minIconImageHeight + (2.0 * layerImageVerticalMargin) - (2.0 * layerSeperatorVerticalMargin);
291
292 [controlLayer setBackgroundColor:[[NSColor lightGrayColor] CGColor]];
293 } else {
294 NSImage *controlImage = [self _getImageForControlName:controlName];
295 [controlLayer setContents:controlImage];
296 [controlLayer setOpacity:controlLayerNormalAlpha];
297
298 dy = layerImageVerticalMargin;
299 controllerWidth = [controlImage size].width / _deviceScaleFactor;
300 controllerHeight = [controlImage size].height / _deviceScaleFactor;
301
302 dy -= (controllerHeight - minIconImageHeight) / 2.0;
303
304 [controlLayer setFilters:@[[CAFilter filterWithType:kCAFilterColorInvert]]];
305 }
306
307 [controlLayer setFrame:CGRectMake(dx, dy, controllerWidth, controllerHeight)];
308 [_layer addSublayer:controlLayer.get()];
309
310 dx += controllerWidth + layerControllerHorizontalMargin;
311 }
312
313 [_layer setFrame:CGRectMake(0, layerVerticalOffset, dx, minIconImageHeight + 2.0 * layerImageVerticalMargin)];
314 [parentLayer addSublayer:_layer.get()];
315}
316
317- (void)_redrawLayer
318{
319 [_cachedIcons removeAllObjects];
320 CALayer *parentLayer = [_layer superlayer];
321 [_layer removeFromSuperlayer];
322 [self _setupLayer:parentLayer];
323 [self setFrameSize:_frameSize];
324}
325
326- (NSImage *)_getImageForControlName:(NSString *)control
327{
328 NSImage *iconImage = _cachedIcons.get()[control];
329 if (iconImage)
330 return iconImage;
331
332 ALLOW_DEPRECATED_DECLARATIONS_BEGIN
333 iconImage = [NSImage _imageWithSystemSymbolName:control];
334 ALLOW_DEPRECATED_DECLARATIONS_END
335 if (!iconImage)
336 return nil;
337
338 iconImage = [iconImage imageWithSymbolConfiguration:[NSImageSymbolConfiguration configurationWithTextStyle:NSFontTextStyleTitle2 scale:NSImageSymbolScaleLarge]];
339
340 NSRect iconImageRect = NSMakeRect(0, 0, [iconImage size].width * layerImageScale * _deviceScaleFactor, [iconImage size].height * layerImageScale * _deviceScaleFactor);
341 NSImageRep *iconImageRep = [iconImage bestRepresentationForRect:iconImageRect context:nil hints:nil];
342 iconImage = [NSImage imageWithImageRep:iconImageRep];
343
344 _cachedIcons.get()[control] = iconImage;
345 return iconImage;
346}
347
348- (void)_setLayerOpacity:(CGFloat)alpha
349{
350 [_layer setOpacity:alpha];
351 for (CALayer *subLayer in [_layer sublayers])
352 [subLayer setOpacity:alpha];
353}
