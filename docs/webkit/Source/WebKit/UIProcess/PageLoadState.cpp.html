81void PageLoadState::endTransaction()
82{
83 ASSERT(m_outstandingTransactionCount > 0);
84
85 if (!--m_outstandingTransactionCount)
86 commitChanges();
87}

89void PageLoadState::commitChanges()
90{
91 if (!m_mayHaveUncommittedChanges)
92 return;
93
94 m_mayHaveUncommittedChanges = false;
95
96 bool canGoBackChanged = m_committedState.canGoBack != m_uncommittedState.canGoBack;
97 bool canGoForwardChanged = m_committedState.canGoForward != m_uncommittedState.canGoForward;
98 bool titleChanged = m_committedState.title != m_uncommittedState.title
99 || m_committedState.titleFromSafeBrowsingWarning != m_uncommittedState.titleFromSafeBrowsingWarning;
100 bool isLoadingChanged = isLoading(m_committedState) != isLoading(m_uncommittedState);
101 bool activeURLChanged = activeURL(m_committedState) != activeURL(m_uncommittedState);
102 bool hasOnlySecureContentChanged = hasOnlySecureContent(m_committedState) != hasOnlySecureContent(m_uncommittedState);
103 bool negotiatedLegacyTLSChanged = m_committedState.negotiatedLegacyTLS != m_uncommittedState.negotiatedLegacyTLS;
104 bool estimatedProgressChanged = estimatedProgress(m_committedState) != estimatedProgress(m_uncommittedState);
105 bool networkRequestsInProgressChanged = m_committedState.networkRequestsInProgress != m_uncommittedState.networkRequestsInProgress;
106 bool certificateInfoChanged = m_committedState.certificateInfo != m_uncommittedState.certificateInfo;
107
108 if (canGoBackChanged)
109 callObserverCallback(&Observer::willChangeCanGoBack);
110 if (canGoForwardChanged)
111 callObserverCallback(&Observer::willChangeCanGoForward);
112 if (titleChanged)
113 callObserverCallback(&Observer::willChangeTitle);
114 if (isLoadingChanged)
115 callObserverCallback(&Observer::willChangeIsLoading);
116 if (activeURLChanged)
117 callObserverCallback(&Observer::willChangeActiveURL);
118 if (hasOnlySecureContentChanged)
119 callObserverCallback(&Observer::willChangeHasOnlySecureContent);
120 if (negotiatedLegacyTLSChanged)
121 callObserverCallback(&Observer::willChangeNegotiatedLegacyTLS);
122 if (estimatedProgressChanged)
123 callObserverCallback(&Observer::willChangeEstimatedProgress);
124 if (networkRequestsInProgressChanged)
125 callObserverCallback(&Observer::willChangeNetworkRequestsInProgress);
126 if (certificateInfoChanged)
127 callObserverCallback(&Observer::willChangeCertificateInfo);
128
129 m_committedState = m_uncommittedState;
130
131 m_webPageProxy.isLoadingChanged();
132
133 // The "did" ordering is the reverse of the "will". This is a requirement of Cocoa Key-Value Observing.
134 if (certificateInfoChanged)
135 callObserverCallback(&Observer::didChangeCertificateInfo);
136 if (networkRequestsInProgressChanged)
137 callObserverCallback(&Observer::didChangeNetworkRequestsInProgress);
138 if (estimatedProgressChanged)
139 callObserverCallback(&Observer::didChangeEstimatedProgress);
140 if (hasOnlySecureContentChanged)
141 callObserverCallback(&Observer::didChangeHasOnlySecureContent);
142 if (negotiatedLegacyTLSChanged)
143 callObserverCallback(&Observer::didChangeNegotiatedLegacyTLS);
144 if (activeURLChanged)
145 callObserverCallback(&Observer::didChangeActiveURL);
146 if (isLoadingChanged)
147 callObserverCallback(&Observer::didChangeIsLoading);
148 if (titleChanged)
149 callObserverCallback(&Observer::didChangeTitle);
150 if (canGoForwardChanged)
151 callObserverCallback(&Observer::didChangeCanGoForward);
152 if (canGoBackChanged)
153 callObserverCallback(&Observer::didChangeCanGoBack);
154}

156void PageLoadState::reset(const Transaction::Token& token)
157{
158 ASSERT_UNUSED(token, &token.m_pageLoadState == this);
159
160 m_uncommittedState.state = State::Finished;
161 m_uncommittedState.hasInsecureContent = false;
162
163 m_uncommittedState.pendingAPIRequest = { };
164 m_uncommittedState.provisionalURL = String();
165 m_uncommittedState.url = String();
166
167 m_uncommittedState.unreachableURL = String();
168 m_lastUnreachableURL = String();
169
170 m_uncommittedState.title = String();
171 m_uncommittedState.titleFromSafeBrowsingWarning = { };
172
173 m_uncommittedState.estimatedProgress = 0;
174 m_uncommittedState.networkRequestsInProgress = false;
175}

187String PageLoadState::activeURL(const Data& data)
188{
189 // If there is a currently pending URL, it is the active URL,
190 // even when there's no main frame yet, as it might be the
191 // first API request.
192 if (!data.pendingAPIRequest.url.isNull())
193 return data.pendingAPIRequest.url;
194
195 if (!data.unreachableURL.isEmpty())
196 return data.unreachableURL;
197
198 switch (data.state) {
199 case State::Provisional:
200 return data.provisionalURL;
201 case State::Committed:
202 case State::Finished:
203 return data.url;
204 }
205
206 ASSERT_NOT_REACHED();
207 return String();
208}

215bool PageLoadState::hasOnlySecureContent(const Data& data)
216{
217 if (data.hasInsecureContent)
218 return false;
219
220 if (data.state == State::Provisional)
221 return WTF::protocolIs(data.provisionalURL, "https");
222
223 return WTF::protocolIs(data.url, "https");
224}

322void PageLoadState::didCommitLoad(const Transaction::Token& token, WebCertificateInfo& certificateInfo, bool hasInsecureContent, bool usedLegacyTLS)
323{
324 ASSERT_UNUSED(token, &token.m_pageLoadState == this);
325 ASSERT(m_uncommittedState.state == State::Provisional);
326
327 m_uncommittedState.state = State::Committed;
328 m_uncommittedState.hasInsecureContent = hasInsecureContent;
329 m_uncommittedState.certificateInfo = &certificateInfo;
330
331 m_uncommittedState.url = m_uncommittedState.provisionalURL;
332 m_uncommittedState.provisionalURL = String();
333 m_uncommittedState.negotiatedLegacyTLS = usedLegacyTLS;
334
335 m_uncommittedState.title = String();
336 m_uncommittedState.titleFromSafeBrowsingWarning = { };
337}

445bool PageLoadState::isLoading(const Data& data)
446{
447 if (!data.pendingAPIRequest.url.isNull())
448 return true;
449
450 switch (data.state) {
451 case State::Provisional:
452 case State::Committed:
453 return true;
454
455 case State::Finished:
456 return false;
457 }
458
459 ASSERT_NOT_REACHED();
460 return false;
461}

478void PageLoadState::callObserverCallback(void (Observer::*callback)())
479{
480 auto protectedPage = makeRef(m_webPageProxy);
481
482 auto observerCopy = m_observers;
483 for (auto* observer : observerCopy) {
484 // This appears potentially inefficient on the surface (searching in a Vector)
485 // but in practice - using only API - there will only ever be (1) observer.
486 if (!m_observers.contains(observer))
487 continue;
488
489 (observer->*callback)();
490 }
491}
