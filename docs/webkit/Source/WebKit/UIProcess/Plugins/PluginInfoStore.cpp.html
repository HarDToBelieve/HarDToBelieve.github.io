58template <typename T, typename U>
59static void addFromVector(T& hashSet, const U& vector)
60{
61 for (size_t i = 0; i < vector.size(); ++i)
62 hashSet.add(vector[i]);
63}

65void PluginInfoStore::loadPluginsIfNecessary()
66{
67 if (m_pluginListIsUpToDate)
68 return;
69
70 ListHashSet<String> uniquePluginPaths;
71
72 // First, load plug-ins from the additional plug-ins directories specified.
73 for (size_t i = 0; i < m_additionalPluginsDirectories.size(); ++i)
74 addFromVector(uniquePluginPaths, pluginPathsInDirectory(m_additionalPluginsDirectories[i]));
75
76 // Then load plug-ins from the standard plug-ins directories.
77 Vector<String> directories = pluginsDirectories();
78 for (size_t i = 0; i < directories.size(); ++i)
79 addFromVector(uniquePluginPaths, pluginPathsInDirectory(directories[i]));
80
81 // Then load plug-ins that are not in the standard plug-ins directories.
82 addFromVector(uniquePluginPaths, individualPluginPaths());
83
84 m_plugins.clear();
85
86 for (const auto& pluginPath : uniquePluginPaths)
87 loadPlugin(m_plugins, pluginPath);
88
89 m_pluginListIsUpToDate = true;
90}

92void PluginInfoStore::loadPlugin(Vector<PluginModuleInfo>& plugins, const String& pluginPath)
93{
94 PluginModuleInfo plugin;
95
96 if (!getPluginInfo(pluginPath, plugin))
97 return;
98
99 if (!shouldUsePlugin(plugins, plugin))
100 return;
101
102 plugins.append(plugin);
103}

111PluginModuleInfo PluginInfoStore::findPluginForMIMEType(const String& mimeType, PluginData::AllowedPluginTypes allowedPluginTypes) const
112{
113 ASSERT(!mimeType.isNull());
114
115 for (const auto& plugin : m_plugins) {
116 if (allowedPluginTypes == PluginData::OnlyApplicationPlugins && !plugin.info.isApplicationPlugin)
117 continue;
118
119 for (const auto& mimeClassInfo : plugin.info.mimes) {
120 if (mimeClassInfo.type == mimeType)
121 return plugin;
122 }
123 }
124
125 return PluginModuleInfo();
126}

128PluginModuleInfo PluginInfoStore::findPluginForExtension(const String& extension, String& mimeType, PluginData::AllowedPluginTypes allowedPluginTypes) const
129{
130 ASSERT(!extension.isNull());
131
132 for (const auto& plugin : m_plugins) {
133 if (allowedPluginTypes == PluginData::OnlyApplicationPlugins && !plugin.info.isApplicationPlugin)
134 continue;
135
136 for (const auto& mimeClassInfo : plugin.info.mimes) {
137 if (mimeClassInfo.extensions.contains(extension)) {
138 // We found a supported extension, set the correct MIME type.
139 mimeType = mimeClassInfo.type;
140 return plugin;
141 }
142 }
143 }
144
145 return PluginModuleInfo();
146}

148static String pathExtension(const URL& url)
149{
150 auto filename = url.lastPathComponent();
151 if (filename.endsWith('/'))
152 return { };
153 size_t lastDotPosition = filename.reverseFind('.');
154 if (lastDotPosition == notFound)
155 return { };
156 return filename.substring(lastDotPosition + 1).convertToASCIILowercase();
157}

174PluginModuleInfo PluginInfoStore::findPlugin(String& mimeType, const URL& url, PluginData::AllowedPluginTypes allowedPluginTypes)
175{
176 loadPluginsIfNecessary();
177
178 // First, check if we can get the plug-in based on its MIME type.
179 if (!mimeType.isNull()) {
180 PluginModuleInfo plugin = findPluginForMIMEType(mimeType, allowedPluginTypes);
181 if (!plugin.path.isNull())
182 return plugin;
183 }
184
185 // Next, check if any plug-ins claim to support the URL extension.
186 String extension = pathExtension(url);
187 if (!extension.isNull() && mimeType.isEmpty()) {
188 PluginModuleInfo plugin = findPluginForExtension(extension, mimeType, allowedPluginTypes);
189 if (!plugin.path.isNull())
190 return plugin;
191
192 // Finally, try to get the MIME type from the extension in a platform specific manner and use that.
193 String extensionMimeType = MIMETypeRegistry::mimeTypeForExtension(extension);
194 if (!extensionMimeType.isNull()) {
195 PluginModuleInfo plugin = findPluginForMIMEType(extensionMimeType, allowedPluginTypes);
196 if (!plugin.path.isNull()) {
197 mimeType = extensionMimeType;
198 return plugin;
199 }
200 }
201 }
202
203 return PluginModuleInfo();
204}

206bool PluginInfoStore::isSupportedPlugin(const PluginInfoStore::SupportedPlugin& plugin, const String& mimeType, const URL& pluginURL)
207{
208 if (!mimeType.isEmpty() && plugin.mimeTypes.contains(mimeType))
209 return true;
210 auto extension = pathExtension(pluginURL);
211 return extension.isEmpty() ? false : plugin.extensions.contains(extension);
212}

214bool PluginInfoStore::isSupportedPlugin(const String& mimeType, const URL& pluginURL, const String&, const URL& pageURL)
215{
216 // We check only pageURL for consistency with WebProcess visible plugins.
217 if (!m_supportedPlugins)
218 return true;
219
220 return m_supportedPlugins->findMatching([&] (auto&& plugin) {
221 return pageURL.isMatchingDomain(plugin.matchingDomain) && isSupportedPlugin(plugin, mimeType, pluginURL);
222 }) != notFound;
223}

225std::optional<Vector<SupportedPluginIdentifier>> PluginInfoStore::supportedPluginIdentifiers()
226{
227 if (!m_supportedPlugins)
228 return std::nullopt;
229
230 return WTF::map(*m_supportedPlugins, [] (auto&& item) {
231 return SupportedPluginIdentifier { item.matchingDomain, item.identifier };
232 });
233}

235void PluginInfoStore::addSupportedPlugin(String&& domainName, String&& identifier, HashSet<String>&& mimeTypes, HashSet<String> extensions)
236{
237 if (!m_supportedPlugins)
238 m_supportedPlugins = Vector<SupportedPlugin> { };
239
240 m_supportedPlugins->append(SupportedPlugin { WTFMove(domainName), WTFMove(identifier), WTFMove(mimeTypes), WTFMove(extensions) });
241}

243PluginModuleInfo PluginInfoStore::infoForPluginWithPath(const String& pluginPath) const
244{
245 for (const auto& plugin : m_plugins) {
246 if (plugin.path == pluginPath)
247 return plugin;
248 }

250 ASSERT_NOT_REACHED();
251 return PluginModuleInfo();
