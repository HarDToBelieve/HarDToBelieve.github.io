26#pragma once
27
28#if ENABLE(NETSCAPE_PLUGIN_API)
29
30#include "PluginModuleInfo.h"
31
32#include <WebCore/PluginData.h>
33#include <wtf/HashSet.h>
34
35namespace WebKit {
36
37class PluginInfoStore;
38
39class PluginInfoStore {
40 WTF_MAKE_NONCOPYABLE(PluginInfoStore);
41
42public:
43 PluginInfoStore();
44
45 void setAdditionalPluginsDirectories(const Vector<String>&);
46
47 void refresh();
48 Vector<PluginModuleInfo> plugins();
49
50 // Returns the info for a plug-in that can handle the given MIME type.
51 // If the MIME type is null, the file extension of the given url will be used to infer the
52 // plug-in type. In that case, mimeType will be filled in with the right MIME type.
53 PluginModuleInfo findPlugin(String& mimeType, const URL&, WebCore::PluginData::AllowedPluginTypes = WebCore::PluginData::AllPlugins);
54
55 // Returns the info for the plug-in with the given bundle identifier.
56 PluginModuleInfo findPluginWithBundleIdentifier(const String& bundleIdentifier);
57
58 // Returns the info for the plug-in with the given path.
59 PluginModuleInfo infoForPluginWithPath(const String& pluginPath) const;
60
61 static PluginModuleLoadPolicy defaultLoadPolicyForPlugin(const PluginModuleInfo&);
62
63 bool isSupportedPlugin(const String& mimeType, const URL& pluginURL, const String& frameURLString, const URL& pageURL);
64 std::optional<Vector<WebCore::SupportedPluginIdentifier>> supportedPluginIdentifiers();
65 void addSupportedPlugin(String&& matchingDomain, String&& identifier, HashSet<String>&& mimeTypes, HashSet<String> extensions);
66 void clearSupportedPlugins() { m_supportedPlugins = std::nullopt; }
67
68private:
69 PluginModuleInfo findPluginForMIMEType(const String& mimeType, WebCore::PluginData::AllowedPluginTypes) const;
70 PluginModuleInfo findPluginForExtension(const String& extension, String& mimeType, WebCore::PluginData::AllowedPluginTypes) const;
71
72 void loadPluginsIfNecessary();
73 static void loadPlugin(Vector<PluginModuleInfo>& plugins, const String& pluginPath);
74
75 // Platform-specific member functions:
76
77 // Returns paths to directories that should be searched for plug-ins (via pluginPathsInDirectory).
78 static Vector<String> pluginsDirectories();
79
80 // Returns paths to all plug-ins in the specified directory.
81 static Vector<String> pluginPathsInDirectory(const String& directory);
82
83 // Returns paths to individual plug-ins that won't be found via pluginsDirectories/pluginPathsInDirectory.
84 static Vector<String> individualPluginPaths();
85
86 // Load plug-in info for the plug-in with the specified path.
87 static bool getPluginInfo(const String& pluginPath, PluginModuleInfo&);
88
89 // Return whether this plug-in should be used (added to the list of plug-ins) or not.
90 static bool shouldUsePlugin(Vector<PluginModuleInfo>& alreadyLoadedPlugins, const PluginModuleInfo&);
91
92 Vector<String> m_additionalPluginsDirectories;
93 Vector<PluginModuleInfo> m_plugins;
94 bool m_pluginListIsUpToDate;
95
96 struct SupportedPlugin {
97 String matchingDomain;
98 String identifier;
99 HashSet<String> mimeTypes;
100 HashSet<String> extensions;
101 };
102 static bool isSupportedPlugin(const SupportedPlugin&, const String& mimeType, const URL& pluginURL);
103
104 std::optional<Vector<SupportedPlugin>> m_supportedPlugins;
105};
106
107} // namespace WebKit
108
109#endif // ENABLE(NETSCAPE_PLUGIN_API)
