101// Asks the plug-in process to create a new connection to a web process. The connection identifier will be
102// encoded in the given argument encoder and sent back to the connection of the given web process.
103void PluginProcessProxy::getPluginProcessConnection(Messages::WebProcessProxy::GetPluginProcessConnection::DelayedReply&& reply)
104{
105 m_pendingConnectionReplies.append(WTFMove(reply));
106
107 if (state() == State::Launching) {
108 m_numPendingConnectionRequests++;
109 return;
110 }
111
112 // Ask the plug-in process to create a connection. Since the plug-in can be waiting for a synchronous reply
113 // we need to make sure that this message is always processed, even when the plug-in is waiting for a synchronus reply.
114 m_connection->send(Messages::PluginProcess::CreateWebProcessConnection(), 0, IPC::SendOption::DispatchMessageEvenWhenWaitingForSyncReply);
115}

117void PluginProcessProxy::fetchWebsiteData(CompletionHandler<void (Vector<String>)>&& completionHandler)
118{
119 uint64_t callbackID = generatePluginProcessCallbackID();
120 m_pendingFetchWebsiteDataCallbacks.set(callbackID, WTFMove(completionHandler));
121
122 if (state() == State::Launching) {
123 m_pendingFetchWebsiteDataRequests.append(callbackID);
124 return;
125 }
126
127 m_connection->send(Messages::PluginProcess::GetSitesWithData(callbackID), 0);
128}

130void PluginProcessProxy::deleteWebsiteData(WallTime modifiedSince, CompletionHandler<void ()>&& completionHandler)

166void PluginProcessProxy::pluginProcessCrashedOrFailedToLaunch()
167{
168 // The plug-in process must have crashed or exited, send any pending sync replies we might have.
169 while (!m_pendingConnectionReplies.isEmpty()) {
170 auto reply = m_pendingConnectionReplies.takeFirst();
171
172#if USE(UNIX_DOMAIN_SOCKETS)
173 reply(IPC::Attachment(), false);
174#elif OS(DARWIN)
175 reply(IPC::Attachment(0, MACH_MSG_TYPE_MOVE_SEND), false);
176#else
177 notImplemented();
178#endif
179 }
180
181 m_pendingFetchWebsiteDataRequests.clear();
182 for (auto&& callback : m_pendingFetchWebsiteDataCallbacks.values())
183 callback({ });
184 m_pendingFetchWebsiteDataCallbacks.clear();
185
186 m_pendingDeleteWebsiteDataRequests.clear();
187 for (auto&& callback : m_pendingDeleteWebsiteDataCallbacks.values())
188 callback();
189 m_pendingDeleteWebsiteDataRequests.clear();
190
191 m_pendingDeleteWebsiteDataForHostNamesRequests.clear();
192 for (auto&& callback : m_pendingDeleteWebsiteDataForHostNamesCallbacks.values())
193 callback();
194 m_pendingDeleteWebsiteDataForHostNamesCallbacks.clear();
195
196 // Tell the plug-in process manager to forget about this plug-in process proxy. This may cause us to be deleted.
197 m_pluginProcessManager->removePluginProcessProxy(this);
198}

200void PluginProcessProxy::didClose(IPC::Connection&)
201{
202#if PLATFORM(COCOA)
203 if (m_modalWindowIsShowing)
204 endModal();
205
206 if (m_fullscreenWindowIsShowing)
207 exitFullscreen();
208#endif
209
210 for (auto& processPool : WebProcessPool::allProcessPools())
211 processPool->sendToAllProcesses(Messages::PluginProcessConnectionManager::PluginProcessCrashed(m_pluginProcessToken));
212
213 // This will cause us to be deleted.
214 pluginProcessCrashedOrFailedToLaunch();
215}

223void PluginProcessProxy::didFinishLaunching(ProcessLauncher*, IPC::Connection::Identifier connectionIdentifier)
224{
225 ASSERT(!m_connection);
226
227 if (!IPC::Connection::identifierIsValid(connectionIdentifier)) {
228 pluginProcessCrashedOrFailedToLaunch();
229 return;
230 }
231
232 m_connection = IPC::Connection::createServerConnection(connectionIdentifier, *this);
233
234 m_connection->open();
235
236 PluginProcessCreationParameters parameters;
237 parameters.minimumLifetime = minimumLifetime;
238 parameters.terminationTimeout = shutdownTimeout;
239
240 platformInitializePluginProcess(parameters);
241
242 // Initialize the plug-in host process.
243 m_connection->send(Messages::PluginProcess::InitializePluginProcess(parameters), 0);
244
245#if PLATFORM(COCOA)
246 m_connection->send(Messages::PluginProcess::SetQOS(pluginProcessLatencyQOS(), pluginProcessThroughputQOS()), 0);
247#endif
248
249 for (auto callbackID : m_pendingFetchWebsiteDataRequests)
250 m_connection->send(Messages::PluginProcess::GetSitesWithData(callbackID), 0);
251 m_pendingFetchWebsiteDataRequests.clear();
252
253 for (auto& request : m_pendingDeleteWebsiteDataRequests)
254 m_connection->send(Messages::PluginProcess::DeleteWebsiteData(request.modifiedSince, request.callbackID), 0);
255 m_pendingDeleteWebsiteDataRequests.clear();
256
257 for (auto& request : m_pendingDeleteWebsiteDataForHostNamesRequests)
258 m_connection->send(Messages::PluginProcess::DeleteWebsiteDataForHostNames(request.hostNames, request.callbackID), 0);
259 m_pendingDeleteWebsiteDataForHostNamesRequests.clear();
260
261 for (unsigned i = 0; i < m_numPendingConnectionRequests; ++i)
262 m_connection->send(Messages::PluginProcess::CreateWebProcessConnection(), 0);
263
264 m_numPendingConnectionRequests = 0;
265
266#if PLATFORM(COCOA)
267 if (!PluginProcessManager::singleton().processSuppressionDisabled())
268 setProcessSuppressionEnabled(true);
269#endif
270}

272void PluginProcessProxy::didCreateWebProcessConnection(const IPC::Attachment& connectionIdentifier, bool supportsAsynchronousPluginInitialization)
273{
274 ASSERT(!m_pendingConnectionReplies.isEmpty());
275
276 // Grab the first pending connection reply.
277 auto reply = m_pendingConnectionReplies.takeFirst();
278
279#if USE(UNIX_DOMAIN_SOCKETS)
280 reply(connectionIdentifier, supportsAsynchronousPluginInitialization);
281#elif OS(DARWIN)
282 reply(IPC::Attachment(connectionIdentifier.port(), MACH_MSG_TYPE_MOVE_SEND), supportsAsynchronousPluginInitialization);
283#else
284 notImplemented();
285#endif
286}
