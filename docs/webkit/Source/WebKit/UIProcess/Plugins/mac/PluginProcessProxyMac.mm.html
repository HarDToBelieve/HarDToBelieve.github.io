62void PluginProcessProxy::platformGetLaunchOptionsWithAttributes(ProcessLauncher::LaunchOptions& launchOptions, const PluginProcessAttributes& pluginProcessAttributes)
63{
64 ASSERT(pluginProcessAttributes.moduleInfo.pluginArchitecture == CPU_TYPE_X86_64);
65 launchOptions.processType = ProcessLauncher::ProcessType::Plugin;
66
67 launchOptions.extraInitializationData.add("plugin-path", pluginProcessAttributes.moduleInfo.path);
68
69 if (PluginProcessManager::singleton().experimentalPlugInSandboxProfilesEnabled())
70 launchOptions.extraInitializationData.add("experimental-sandbox-plugin", "1");
71
72 if (pluginProcessAttributes.sandboxPolicy == PluginProcessSandboxPolicy::Unsandboxed) {
73 if (!currentProcessIsSandboxed())
74 launchOptions.extraInitializationData.add("disable-sandbox", "1");
75 else
76 WTFLogAlways("Main process is sandboxed, ignoring plug-in sandbox policy");
77 }
78}

80void PluginProcessProxy::platformInitializePluginProcess(PluginProcessCreationParameters& parameters)
81{
82 // For now only Flash is known to behave with asynchronous plug-in initialization.
83 parameters.supportsAsynchronousPluginInitialization = m_pluginProcessAttributes.moduleInfo.bundleIdentifier == "com.macromedia.Flash Player.plugin";

86 parameters.networkATSContext = adoptCF(_CFNetworkCopyATSContext());

89bool PluginProcessProxy::getPluginProcessSerialNumber(ProcessSerialNumber& pluginProcessSerialNumber)
90{

97void PluginProcessProxy::makePluginProcessTheFrontProcess()
98{
99 ProcessSerialNumber pluginProcessSerialNumber;
100 if (!getPluginProcessSerialNumber(pluginProcessSerialNumber))
101 return;
102
103 ALLOW_DEPRECATED_DECLARATIONS_BEGIN
104 SetFrontProcess(&pluginProcessSerialNumber);
105 ALLOW_DEPRECATED_DECLARATIONS_END

117void PluginProcessProxy::setFullscreenWindowIsShowing(bool fullscreenWindowIsShowing)
118{
119 if (m_fullscreenWindowIsShowing == fullscreenWindowIsShowing)
120 return;
121
122 m_fullscreenWindowIsShowing = fullscreenWindowIsShowing;
123 if (m_fullscreenWindowIsShowing)
124 enterFullscreen();
125 else
126 exitFullscreen();
127}
128
129void PluginProcessProxy::enterFullscreen()
130{
131 ASSERT(hasProcessPrivilege(ProcessPrivilege::CanCommunicateWithWindowServer));
132 // Get the current presentation options.
133 m_preFullscreenAppPresentationOptions = [NSApp presentationOptions];
134
135 // Figure out which presentation options to use.
136 unsigned presentationOptions = m_preFullscreenAppPresentationOptions & ~(NSApplicationPresentationAutoHideDock | NSApplicationPresentationAutoHideMenuBar);
137 presentationOptions |= NSApplicationPresentationHideDock | NSApplicationPresentationHideMenuBar;
138
139 [NSApp setPresentationOptions:presentationOptions];
140 makePluginProcessTheFrontProcess();
141}
142
143void PluginProcessProxy::exitFullscreen()
144{
145 // If the plug-in host is the current application then we should bring ourselves to the front when it exits full-screen mode.
146 ProcessSerialNumber frontProcessSerialNumber;
147 ALLOW_DEPRECATED_DECLARATIONS_BEGIN
148 GetFrontProcess(&frontProcessSerialNumber);
149 ALLOW_DEPRECATED_DECLARATIONS_END
150
151 // The UI process must be the front process in order to change the presentation mode.
152 makeUIProcessTheFrontProcess();
153 [NSApp setPresentationOptions:m_preFullscreenAppPresentationOptions];
154
155 ProcessSerialNumber pluginProcessSerialNumber;
156 if (!getPluginProcessSerialNumber(pluginProcessSerialNumber))
157 return;
158
159 // If the plug-in process was not the front process, switch back to the previous front process.
160 // (Otherwise we'll keep the UI process as the front process).
161 Boolean isPluginProcessFrontProcess;
162 ALLOW_DEPRECATED_DECLARATIONS_BEGIN
163 SameProcess(&frontProcessSerialNumber, &pluginProcessSerialNumber, &isPluginProcessFrontProcess);
164 ALLOW_DEPRECATED_DECLARATIONS_END
165 if (!isPluginProcessFrontProcess) {
166 ALLOW_DEPRECATED_DECLARATIONS_BEGIN
167 SetFrontProcess(&frontProcessSerialNumber);
168 ALLOW_DEPRECATED_DECLARATIONS_END
169 }
170}

172void PluginProcessProxy::setModalWindowIsShowing(bool modalWindowIsShowing)
173{
174 if (modalWindowIsShowing == m_modalWindowIsShowing)

185void PluginProcessProxy::beginModal()
186{
187 ASSERT(!m_placeholderWindow);
188 ASSERT(!m_activationObserver);
189 ASSERT(hasProcessPrivilege(ProcessPrivilege::CanCommunicateWithWindowServer));
190
191 m_placeholderWindow = adoptNS([[WKPlaceholderModalWindow alloc] initWithContentRect:NSMakeRect(0, 0, 1, 1) styleMask:NSWindowStyleMaskBorderless backing:NSBackingStoreBuffered defer:YES]);
192 [m_placeholderWindow setReleasedWhenClosed:NO];
193
194 m_activationObserver = [[NSNotificationCenter defaultCenter] addObserverForName:NSApplicationWillBecomeActiveNotification object:NSApp queue:nil
195 usingBlock:^(NSNotification *){ applicationDidBecomeActive(); }];
196
197 // The call to -[NSApp runModalForWindow:] below will run a nested run loop, and if the plug-in process
198 // crashes the PluginProcessProxy object can be destroyed. Protect against this here.
199 Ref<PluginProcessProxy> protect(*this);
200
201 [NSApp runModalForWindow:m_placeholderWindow.get()];
202
203 [m_placeholderWindow orderOut:nil];
204 m_placeholderWindow = nullptr;
205}
206
207void PluginProcessProxy::endModal()
208{
209 ASSERT(m_placeholderWindow);
210 ASSERT(m_activationObserver);
211 ASSERT(hasProcessPrivilege(ProcessPrivilege::CanCommunicateWithWindowServer));
212
213 [[NSNotificationCenter defaultCenter] removeObserver:m_activationObserver.get()];
214 m_activationObserver = nullptr;
215
216 [NSApp stopModal];
217
218 makeUIProcessTheFrontProcess();
219}

221void PluginProcessProxy::applicationDidBecomeActive()
222{
223 makePluginProcessTheFrontProcess();
224}

226static bool isFlashUpdater(const String& launchPath, const Vector<String>& arguments)
227{
228 if (launchPath != "/Applications/Utilities/Adobe Flash Player Install Manager.app/Contents/MacOS/Adobe Flash Player Install Manager")
229 return false;
230
231 if (arguments.size() != 1)
232 return false;
233
234 if (arguments[0] != "-update")
235 return false;
236
237 return true;
238}

240static bool shouldLaunchProcess(const PluginProcessAttributes& pluginProcessAttributes, const String& launchPath, const Vector<String>& arguments)
241{
242 if (pluginProcessAttributes.moduleInfo.bundleIdentifier == "com.macromedia.Flash Player.plugin")
243 return isFlashUpdater(launchPath, arguments);
244
245 return false;

248void PluginProcessProxy::launchProcess(const String& launchPath, const Vector<String>& arguments, CompletionHandler<void(bool)>&& completionHandler)
249{
250 if (!shouldLaunchProcess(m_pluginProcessAttributes, launchPath, arguments))
251 return completionHandler(false);
252
253 [NSTask launchedTaskWithLaunchPath:launchPath arguments:createNSArray(arguments).get()];
254 completionHandler(true);
255}

257static bool isJavaUpdaterURL(const PluginProcessAttributes& pluginProcessAttributes, const String& urlString)
258{
259 NSURL *url = [NSURL URLWithString:urlString];
260 if (![url isFileURL])
261 return false;
262
263 NSArray *javaUpdaterAppNames = @[@"Java Updater.app", @"JavaUpdater.app"];
264
265 for (NSString *javaUpdaterAppName in javaUpdaterAppNames) {
266 NSString *javaUpdaterPath = [NSString pathWithComponents:@[(NSString *)pluginProcessAttributes.moduleInfo.path, @"Contents/Resources", javaUpdaterAppName]];
267 if ([url.path isEqualToString:javaUpdaterPath])
268 return YES;
269 }
270
271 return NO;
272}

282void PluginProcessProxy::launchApplicationAtURL(const String& urlString, const Vector<String>& arguments, CompletionHandler<void(bool)>&& completionHandler)

291 completionHandler(true);

309void PluginProcessProxy::openURL(const String& urlString, CompletionHandler<void(bool result, int32_t status, String launchedURLString)>&& completionHandler)
310{
311 if (!shouldOpenURL(m_pluginProcessAttributes, urlString))
312 return completionHandler(false, 0, { });
313
314 CFURLRef launchedURL;
315 uint32_t status = LSOpenCFURLRef(URL({ }, urlString).createCFURL().get(), &launchedURL);
316
317 String launchedURLString;
318 if (launchedURL) {
319 launchedURLString = URL(launchedURL).string();
320 CFRelease(launchedURL);
321 }
322 completionHandler(true, status, launchedURLString);
323}

325static bool shouldOpenFile(const PluginProcessAttributes& pluginProcessAttributes, const String& fullPath)
326{
327 if (pluginProcessAttributes.moduleInfo.bundleIdentifier == "com.macromedia.Flash Player.plugin") {

329 return true;
330 }
331
332 return false;
333}

343 completionHandler(true);
