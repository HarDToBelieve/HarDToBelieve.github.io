61ProvisionalPageProxy::ProvisionalPageProxy(WebPageProxy& page, Ref<WebProcessProxy>&& process, std::unique_ptr<SuspendedPageProxy> suspendedPage, uint64_t navigationID, bool isServerRedirect, const WebCore::ResourceRequest& request, ProcessSwapRequestedByClient processSwapRequestedByClient, API::WebsitePolicies* websitePolicies)
62 : m_page(page)
63 , m_webPageID(suspendedPage ? suspendedPage->webPageID() : PageIdentifier::generate())
64 , m_process(WTFMove(process))
65 , m_navigationID(navigationID)
66 , m_isServerRedirect(isServerRedirect)
67 , m_request(request)
68 , m_processSwapRequestedByClient(processSwapRequestedByClient)
69#if PLATFORM(IOS_FAMILY)
70 , m_provisionalLoadActivity(m_process->throttler().foregroundActivity("Provisional Load"_s))
71#endif
72#if HAVE(VISIBILITY_PROPAGATION_VIEW)
73 , m_contextIDForVisibilityPropagationInWebProcess(suspendedPage ? suspendedPage->contextIDForVisibilityPropagationInWebProcess() : 0)
74#if ENABLE(GPU_PROCESS)
75 , m_contextIDForVisibilityPropagationInGPUProcess(suspendedPage ? suspendedPage->contextIDForVisibilityPropagationInGPUProcess() : 0)
76#endif
77#endif
78{
79 PROVISIONALPAGEPROXY_RELEASE_LOG(ProcessSwapping, "ProvisionalPageProxy: suspendedPage=%p", suspendedPage.get());
80
81 m_process->addMessageReceiver(Messages::WebPageProxy::messageReceiverName(), m_webPageID, *this);
82 m_process->addProvisionalPageProxy(*this);
83
84 if (&m_process->websiteDataStore() != &m_page.websiteDataStore())
85 m_process->processPool().pageBeginUsingWebsiteDataStore(m_page.identifier(), m_process->websiteDataStore());
86
87 // If we are reattaching to a SuspendedPage, then the WebProcess' WebPage already exists and
88 // WebPageProxy::didCreateMainFrame() will not be called to initialize m_mainFrame. In such
89 // case, we need to initialize m_mainFrame to reflect the fact the the WebProcess' WebPage
90 // already exists and already has a main frame.
91 if (suspendedPage) {
92 ASSERT(&suspendedPage->process() == m_process.ptr());
93 suspendedPage->unsuspend();
94 m_mainFrame = WebFrameProxy::create(m_page, suspendedPage->mainFrameID());
95 m_process->frameCreated(suspendedPage->mainFrameID(), *m_mainFrame);
96 }
97
98 initializeWebPage(websitePolicies);
99
100 m_page.inspectorController().didCreateProvisionalPage(*this);
101}

103ProvisionalPageProxy::~ProvisionalPageProxy()
104{
105 if (!m_wasCommitted) {
106 m_page.inspectorController().willDestroyProvisionalPage(*this);
107
108 if (&m_process->websiteDataStore() != &m_page.websiteDataStore())
109 m_process->processPool().pageEndUsingWebsiteDataStore(m_page.identifier(), m_process->websiteDataStore());
110
111 m_process->removeMessageReceiver(Messages::WebPageProxy::messageReceiverName(), m_webPageID);
112 send(Messages::WebPage::Close());
113 m_process->removeVisitedLinkStoreUser(m_page.visitedLinkStore(), m_page.identifier());
114 }
115
116 m_process->removeProvisionalPageProxy(*this);
117}

130void ProvisionalPageProxy::cancel()
131{
132 // If the provisional load started, then indicate that it failed due to cancellation by calling didFailProvisionalLoadForFrame().
133 if (m_provisionalLoadURL.isEmpty())
134 return;
135
136 ASSERT(m_process->state() == WebProcessProxy::State::Running);
137
138 PROVISIONALPAGEPROXY_RELEASE_LOG(ProcessSwapping, "cancel: Simulating a didFailProvisionalLoadForFrame");
139 ASSERT(m_mainFrame);
140 auto error = WebKit::cancelledError(m_request);
141 error.setType(WebCore::ResourceError::Type::Cancellation);
142 FrameInfoData frameInfo {
143 true, // isMainFrame
144 m_request,
145 SecurityOriginData::fromURL(m_request.url()),
146 m_mainFrame->frameID(),
147 std::nullopt,
148 };
149 didFailProvisionalLoadForFrame(m_mainFrame->frameID(), WTFMove(frameInfo), ResourceRequest { m_request }, m_navigationID, m_provisionalLoadURL.string(), error, WebCore::WillContinueLoading::No, UserData { }); // Will delete |this|.
150}

152void ProvisionalPageProxy::initializeWebPage(RefPtr<API::WebsitePolicies>&& websitePolicies)
153{
154 m_drawingArea = m_page.pageClient().createDrawingAreaProxy(m_process);
155
156 auto parameters = m_page.creationParameters(m_process, *m_drawingArea, WTFMove(websitePolicies));
157 parameters.isProcessSwap = true;
158 m_process->send(Messages::WebProcess::CreateWebPage(m_webPageID, parameters), 0);
159 m_process->addVisitedLinkStoreUser(m_page.visitedLinkStore(), m_page.identifier());
160
161 if (m_page.isLayerTreeFrozenDueToSwipeAnimation())
162 send(Messages::WebPage::FreezeLayerTreeDueToSwipeAnimation());
163}

172void ProvisionalPageProxy::loadRequest(API::Navigation& navigation, WebCore::ResourceRequest&& request, API::Object* userData, std::optional<NavigatingToAppBoundDomain> isNavigatingToAppBoundDomain, std::optional<WebsitePoliciesData>&& websitePolicies)
173{
174 PROVISIONALPAGEPROXY_RELEASE_LOG(ProcessSwapping, "loadRequest:");
175
176 // If this is a client-side redirect continuing in a new process, then the new process will overwrite the fromItem's URL. In this case,
177 // we need to make sure we update fromItem's processIdentifier as we want future navigations to this BackForward item to happen in the
178 // new process.
179 if (navigation.fromItem() && navigation.lockBackForwardList() == WebCore::LockBackForwardList::Yes)
180 navigation.fromItem()->setLastProcessIdentifier(m_process->coreProcessIdentifier());
181
182 m_page.loadRequestWithNavigationShared(m_process.copyRef(), m_webPageID, navigation, WTFMove(request), navigation.lastNavigationAction().shouldOpenExternalURLsPolicy, userData, WebCore::ShouldTreatAsContinuingLoad::Yes, isNavigatingToAppBoundDomain, WTFMove(websitePolicies));
183}

185void ProvisionalPageProxy::goToBackForwardItem(API::Navigation& navigation, WebBackForwardListItem& item, RefPtr<API::WebsitePolicies>&& websitePolicies)
186{
187 PROVISIONALPAGEPROXY_RELEASE_LOG(ProcessSwapping, "goToBackForwardItem:");
188
189 auto itemStates = m_page.backForwardList().filteredItemStates([this, targetItem = &item](auto& item) {
190 if (auto* backForwardCacheEntry = item.backForwardCacheEntry()) {
191 if (backForwardCacheEntry->processIdentifier() == m_process->coreProcessIdentifier())
192 return false;
193 }
194 return &item != targetItem;
195 });
196
197 std::optional<WebsitePoliciesData> websitePoliciesData;
198 if (websitePolicies)
199 websitePoliciesData = websitePolicies->data();
200
201 send(Messages::WebPage::UpdateBackForwardListForReattach(WTFMove(itemStates)));
202 send(Messages::WebPage::GoToBackForwardItem(navigation.navigationID(), item.itemID(), *navigation.backForwardFrameLoadType(), WebCore::ShouldTreatAsContinuingLoad::Yes, WTFMove(websitePoliciesData)));
203 m_process->startResponsivenessTimer();
204}

206inline bool ProvisionalPageProxy::validateInput(FrameIdentifier frameID, const std::optional<uint64_t>& navigationID)
207{
208 // If the previous provisional load used an existing process, we may receive leftover IPC for a previous navigation, which we need to ignore.
209 if (!m_mainFrame || m_mainFrame->frameID() != frameID)
210 return false;
211
212 return !navigationID || !*navigationID || *navigationID == m_navigationID;
213}

215void ProvisionalPageProxy::didCreateMainFrame(FrameIdentifier frameID)
216{
217 PROVISIONALPAGEPROXY_RELEASE_LOG(ProcessSwapping, "didCreateMainFrame: frameID=%" PRIu64, frameID.toUInt64());
218 ASSERT(!m_mainFrame);
219
220 m_mainFrame = WebFrameProxy::create(m_page, frameID);
221
222 // Add the frame to the process wide map.
223 m_process->frameCreated(frameID, *m_mainFrame);
224
225 // This navigation was destroyed so no need to notify of redirect.
226 if (!m_page.navigationState().hasNavigation(m_navigationID))
227 return;
228
229 // Restore the main frame's committed URL as some clients may rely on it until the next load is committed.
230 if (auto* mainFrame = m_page.mainFrame())
231 m_mainFrame->frameLoadState().setURL(mainFrame->frameLoadState().url());
232
233 // Normally, notification of a server redirect comes from the WebContent process.
234 // If we are process swapping in response to a server redirect then that notification will not come from the new WebContent process.
235 // In this case we have the UIProcess synthesize the redirect notification at the appropriate time.
236 if (m_isServerRedirect) {
237 m_mainFrame->frameLoadState().didStartProvisionalLoad(m_request.url());
238 m_page.didReceiveServerRedirectForProvisionalLoadForFrameShared(m_process.copyRef(), m_mainFrame->frameID(), m_navigationID, WTFMove(m_request), { });
239 }
240}

250void ProvisionalPageProxy::didStartProvisionalLoadForFrame(FrameIdentifier frameID, FrameInfoData&& frameInfo, ResourceRequest&& request, uint64_t navigationID, URL&& url, URL&& unreachableURL, const UserData& userData)
251{
252 if (!validateInput(frameID, navigationID))
253 return;
254
255 PROVISIONALPAGEPROXY_RELEASE_LOG(ProcessSwapping, "didStartProvisionalLoadForFrame: frameID=%" PRIu64, frameID.toUInt64());
256 ASSERT(m_provisionalLoadURL.isNull());
257 m_provisionalLoadURL = url;
258
259 // Merely following a server side redirect so there is no need to send a didStartProvisionalLoad again.
260 if (m_isServerRedirect)
261 return;
262
263 // Clients expect the Page's main frame's expectedURL to be the provisional one when a provisional load is started.
264 if (auto* pageMainFrame = m_page.mainFrame())
265 pageMainFrame->didStartProvisionalLoad(url);
266
267 m_page.didStartProvisionalLoadForFrameShared(m_process.copyRef(), frameID, WTFMove(frameInfo), WTFMove(request), navigationID, WTFMove(url), WTFMove(unreachableURL), userData);
268}

270void ProvisionalPageProxy::didFailProvisionalLoadForFrame(FrameIdentifier frameID, FrameInfoData&& frameInfo, ResourceRequest&& request, uint64_t navigationID, const String& provisionalURL, const WebCore::ResourceError& error, WebCore::WillContinueLoading willContinueLoading, const UserData& userData)
271{
272 if (!validateInput(frameID, navigationID))
273 return;
274
275 PROVISIONALPAGEPROXY_RELEASE_LOG_ERROR(ProcessSwapping, "didFailProvisionalLoadForFrame: frameID=%" PRIu64, frameID.toUInt64());
276 ASSERT(!m_provisionalLoadURL.isNull());
277 m_provisionalLoadURL = { };
278
279 // Make sure the Page's main frame's expectedURL gets cleared since we updated it in didStartProvisionalLoad.
280 if (auto* pageMainFrame = m_page.mainFrame())
281 pageMainFrame->didFailProvisionalLoad();
282
283 m_page.didFailProvisionalLoadForFrameShared(m_process.copyRef(), frameID, WTFMove(frameInfo), WTFMove(request), navigationID, provisionalURL, error, willContinueLoading, userData); // May delete |this|.
284}

360void ProvisionalPageProxy::decidePolicyForNavigationActionSync(FrameIdentifier frameID, bool isMainFrame, FrameInfoData&& frameInfoData, WebCore::PolicyCheckIdentifier identifier,
361 uint64_t navigationID, NavigationActionData&& navigationActionData, FrameInfoData&& originatingFrameInfo, std::optional<WebPageProxyIdentifier> originatingPageID,
362 const WebCore::ResourceRequest& originalRequest, WebCore::ResourceRequest&& request, IPC::FormDataReference&& requestBody, WebCore::ResourceResponse&& redirectResponse,
363 const UserData& userData, Messages::WebPageProxy::DecidePolicyForNavigationActionSync::DelayedReply&& reply)
364{
365 if (!isMainFrame || (m_mainFrame && m_mainFrame->frameID() != frameID) || navigationID != m_navigationID) {
366 reply(PolicyDecision { identifier, std::nullopt, WebCore::PolicyAction::Ignore, navigationID, std::nullopt, std::nullopt });
367 return;
368 }
369
370 if (!m_mainFrame) {
371 // This synchronous IPC message was processed before the asynchronous DidCreateMainFrame one so we do not know about this frameID yet.
372 didCreateMainFrame(frameID);
373 }
374 ASSERT(m_mainFrame);
375
376 m_page.decidePolicyForNavigationActionSyncShared(m_process.copyRef(), frameID, isMainFrame, WTFMove(frameInfoData), identifier, navigationID, WTFMove(navigationActionData), WTFMove(originatingFrameInfo), originatingPageID, originalRequest, WTFMove(request), WTFMove(requestBody), WTFMove(redirectResponse), userData, WTFMove(reply));
377}

442void ProvisionalPageProxy::didReceiveMessage(IPC::Connection& connection, IPC::Decoder& decoder)
443{
444 ASSERT(decoder.messageReceiverName() == Messages::WebPageProxy::messageReceiverName());
445
446 if (decoder.messageName() == Messages::WebPageProxy::DidStartProgress::name()
447 || decoder.messageName() == Messages::WebPageProxy::DidChangeProgress::name()
448 || decoder.messageName() == Messages::WebPageProxy::DidDestroyNavigation::name()
449 || decoder.messageName() == Messages::WebPageProxy::DidFinishProgress::name()
450 || decoder.messageName() == Messages::WebPageProxy::BackForwardAddItem::name()
451 || decoder.messageName() == Messages::WebPageProxy::SetNetworkRequestsInProgress::name()
452 || decoder.messageName() == Messages::WebPageProxy::WillGoToBackForwardListItem::name()
453#if USE(QUICK_LOOK)
454 || decoder.messageName() == Messages::WebPageProxy::DidStartLoadForQuickLookDocumentInMainFrame::name()
455 || decoder.messageName() == Messages::WebPageProxy::DidFinishLoadForQuickLookDocumentInMainFrame::name()
456#endif
457 || decoder.messageName() == Messages::WebPageProxy::CreateInspectorTarget::name()
458 || decoder.messageName() == Messages::WebPageProxy::DestroyInspectorTarget::name()
459 || decoder.messageName() == Messages::WebPageProxy::SendMessageToInspectorFrontend::name()
460 )
461 {
462 m_page.didReceiveMessage(connection, decoder);
463 return;
464 }
465
466#if PLATFORM(COCOA)
467 if (decoder.messageName() == Messages::WebPageProxy::RegisterWebProcessAccessibilityToken::name()) {
468 IPC::handleMessage<Messages::WebPageProxy::RegisterWebProcessAccessibilityToken>(decoder, this, &ProvisionalPageProxy::registerWebProcessAccessibilityToken);
469 return;
470 }
471#endif
472
473 if (decoder.messageName() == Messages::WebPageProxy::LogDiagnosticMessageFromWebProcess::name()) {
474 IPC::handleMessage<Messages::WebPageProxy::LogDiagnosticMessageFromWebProcess>(decoder, this, &ProvisionalPageProxy::logDiagnosticMessageFromWebProcess);
475 return;
476 }
477
478 if (decoder.messageName() == Messages::WebPageProxy::LogDiagnosticMessageWithEnhancedPrivacyFromWebProcess::name()) {
479 IPC::handleMessage<Messages::WebPageProxy::LogDiagnosticMessageWithEnhancedPrivacyFromWebProcess>(decoder, this, &ProvisionalPageProxy::logDiagnosticMessageWithEnhancedPrivacyFromWebProcess);
480 return;
481 }
482
483 if (decoder.messageName() == Messages::WebPageProxy::LogDiagnosticMessageWithValueDictionaryFromWebProcess::name()) {
484 IPC::handleMessage<Messages::WebPageProxy::LogDiagnosticMessageWithValueDictionaryFromWebProcess>(decoder, this, &ProvisionalPageProxy::logDiagnosticMessageWithValueDictionaryFromWebProcess);
485 return;
