61- (id)initWithDrawingAreaProxy:(WebKit::RemoteLayerTreeDrawingAreaProxy*)drawingAreaProxy
62{
63 if (self = [super init]) {
64 _drawingAreaProxy = drawingAreaProxy;
65 // Note that CADisplayLink retains its target (self), so a call to -invalidate is needed on teardown.
66 _displayLink = [CADisplayLink displayLinkWithTarget:self selector:@selector(displayLinkFired:)];
67 [_displayLink addToRunLoop:[NSRunLoop mainRunLoop] forMode:NSRunLoopCommonModes];
68 _displayLink.paused = YES;
69 _displayLink.preferredFramesPerSecond = 60;
70 }
71 return self;
72}

83}

87 ASSERT(isUIThread());

93 [_displayLink invalidate];

96

98{
99 _displayLink.paused = NO;

104 _displayLink.paused = YES;

114RemoteLayerTreeDrawingAreaProxy::RemoteLayerTreeDrawingAreaProxy(WebPageProxy& webPageProxy, WebProcessProxy& process)
115 : DrawingAreaProxy(DrawingAreaTypeRemoteLayerTree, webPageProxy, process)
116 , m_remoteLayerTreeHost(makeUnique<RemoteLayerTreeHost>(*this))
117{
118 // We don't want to pool surfaces in the UI process.
119 // FIXME: We should do this somewhere else.
120 IOSurfacePool::sharedPool().setPoolSize(0);
121
122 process.addMessageReceiver(Messages::RemoteLayerTreeDrawingAreaProxy::messageReceiverName(), m_identifier, *this);
123
124 if (m_webPageProxy.preferences().tiledScrollingIndicatorVisible())
125 initializeDebugIndicator();
126}
127
128RemoteLayerTreeDrawingAreaProxy::~RemoteLayerTreeDrawingAreaProxy()
129{
130 m_callbacks.invalidate(CallbackBase::Error::OwnerWasInvalidated);
131 process().removeMessageReceiver(Messages::RemoteLayerTreeDrawingAreaProxy::messageReceiverName(), m_identifier);
132
133#if PLATFORM(IOS_FAMILY)
134 [m_displayLinkHandler invalidate];
135#endif
136}

171void RemoteLayerTreeDrawingAreaProxy::didUpdateGeometry()
172{
173 ASSERT(m_isWaitingForDidUpdateGeometry);
174
175 m_isWaitingForDidUpdateGeometry = false;
176
177 // If the WKView was resized while we were waiting for a DidUpdateGeometry reply from the web process,
178 // we need to resend the new size here.
179 if (m_lastSentSize != m_size)
180 sendUpdateGeometry();
181}
182
183void RemoteLayerTreeDrawingAreaProxy::sendUpdateGeometry()
184{
185 m_lastSentSize = m_size;
186 send(Messages::DrawingArea::UpdateGeometry(m_size, false /* flushSynchronously */, MachSendRight()));
187 m_isWaitingForDidUpdateGeometry = true;
188}

204void RemoteLayerTreeDrawingAreaProxy::commitLayerTree(const RemoteLayerTreeTransaction& layerTreeTransaction, const RemoteScrollingCoordinatorTransaction& scrollingTreeTransaction)
205{
206 TraceScope tracingScope(CommitLayerTreeStart, CommitLayerTreeEnd);
207
208 LOG_WITH_STREAM(RemoteLayerTree, stream << "RemoteLayerTreeDrawingAreaProxy::commitLayerTree transaction:" << layerTreeTransaction.description());
209 LOG_WITH_STREAM(RemoteLayerTree, stream << "RemoteLayerTreeDrawingAreaProxy::commitLayerTree scrolling tree:" << scrollingTreeTransaction.description());
210
211 ASSERT(layerTreeTransaction.transactionID() == m_lastVisibleTransactionID.next());
212 m_transactionIDForPendingCACommit = layerTreeTransaction.transactionID();
213 m_activityStateChangeID = layerTreeTransaction.activityStateChangeID();
214
215 bool didUpdateEditorState = layerTreeTransaction.hasEditorState() && m_webPageProxy.updateEditorState(layerTreeTransaction.editorState());
216
217 if (m_remoteLayerTreeHost->updateLayerTree(layerTreeTransaction)) {
218 if (layerTreeTransaction.transactionID() >= m_transactionIDForUnhidingContent)
219 m_webPageProxy.setRemoteLayerTreeRootNode(m_remoteLayerTreeHost->rootNode());
220 else
221 m_remoteLayerTreeHost->detachRootLayer();
222 }
223
224#if ENABLE(ASYNC_SCROLLING)
225 RemoteScrollingCoordinatorProxy::RequestedScrollInfo requestedScrollInfo;
226 m_webPageProxy.scrollingCoordinatorProxy()->commitScrollingTreeState(scrollingTreeTransaction, requestedScrollInfo);
227#endif
228
229 m_webPageProxy.didCommitLayerTree(layerTreeTransaction);
230
231#if ENABLE(ASYNC_SCROLLING)
232 m_webPageProxy.scrollingCoordinatorProxy()->applyScrollingTreeLayerPositionsAfterCommit();
233#if PLATFORM(IOS_FAMILY)
234 m_webPageProxy.adjustLayersForLayoutViewport(m_webPageProxy.unconstrainedLayoutViewportRect());
235#endif
236
237 // Handle requested scroll position updates from the scrolling tree transaction after didCommitLayerTree()
238 // has updated the view size based on the content size.
239 if (requestedScrollInfo.requestsScrollPositionUpdate)
240 m_webPageProxy.requestScroll(requestedScrollInfo.requestedScrollPosition, layerTreeTransaction.scrollOrigin());
241#endif // ENABLE(ASYNC_SCROLLING)
242
243 if (m_debugIndicatorLayerTreeHost) {
244 float scale = indicatorScale(layerTreeTransaction.contentsSize());
245 bool rootLayerChanged = m_debugIndicatorLayerTreeHost->updateLayerTree(layerTreeTransaction, scale);
246 IntPoint scrollPosition;
247#if PLATFORM(MAC)
248 scrollPosition = layerTreeTransaction.scrollPosition();
249#endif
250 updateDebugIndicator(layerTreeTransaction.contentsSize(), rootLayerChanged, scale, scrollPosition);
251 m_debugIndicatorLayerTreeHost->rootLayer().name = @"Indicator host root";
252 }
253
254 m_webPageProxy.layerTreeCommitComplete();
255
256#if PLATFORM(IOS_FAMILY)
257 if (std::exchange(m_didUpdateMessageState, NeedsDidUpdate) == MissedCommit)
258 didRefreshDisplay();
259 [displayLinkHandler() schedule];
260#else
261 m_didUpdateMessageState = NeedsDidUpdate;
262 didRefreshDisplay();
263#endif
264
265 if (didUpdateEditorState)
266 m_webPageProxy.dispatchDidUpdateEditorState();
267
268 if (auto milestones = layerTreeTransaction.newlyReachedPaintingMilestones())
269 m_webPageProxy.didReachLayoutMilestone(milestones);
270
271 for (auto& callbackID : layerTreeTransaction.callbackIDs()) {
272 if (auto callback = m_callbacks.take<VoidCallback>(callbackID))
273 callback->performCallback();
274 }
275}

297FloatPoint RemoteLayerTreeDrawingAreaProxy::indicatorLocation() const
298{
299 if (m_webPageProxy.delegatesScrolling()) {
300#if PLATFORM(IOS_FAMILY)
301 FloatPoint tiledMapLocation = m_webPageProxy.unobscuredContentRect().location().expandedTo(FloatPoint());
302 tiledMapLocation = tiledMapLocation.expandedTo(m_webPageProxy.exposedContentRect().location());
303
304 float absoluteInset = indicatorInset / m_webPageProxy.displayedContentScale();
305 tiledMapLocation += FloatSize(absoluteInset, absoluteInset);
306#else
307 FloatPoint tiledMapLocation;
308 if (auto viewExposedRect = m_webPageProxy.viewExposedRect())
309 tiledMapLocation = viewExposedRect->location();
310
311 tiledMapLocation += FloatSize(indicatorInset, indicatorInset);
312 float scale = 1 / m_webPageProxy.pageScaleFactor();
313 tiledMapLocation.scale(scale);
314#endif
315 return tiledMapLocation;
316 }
317
318 return FloatPoint(indicatorInset, indicatorInset + m_webPageProxy.topContentInset());
319}

329float RemoteLayerTreeDrawingAreaProxy::indicatorScale(IntSize contentsSize) const
330{
331 // Pick a good scale.
332 IntSize viewSize = m_webPageProxy.viewSize();
333
334 float scale = 1;
335 if (!contentsSize.isEmpty()) {
336 float widthScale = std::min<float>((viewSize.width() - 2 * indicatorInset) / contentsSize.width(), 0.05);
337 scale = std::min(widthScale, static_cast<float>(viewSize.height() - 2 * indicatorInset) / contentsSize.height());
338 }
339
340 return scale;
341}

349void RemoteLayerTreeDrawingAreaProxy::updateDebugIndicator(IntSize contentsSize, bool rootLayerChanged, float scale, const IntPoint& scrollPosition)
350{
351 // Make sure we're the last sublayer.
352 CALayer *rootLayer = m_remoteLayerTreeHost->rootLayer();
353 [m_tileMapHostLayer removeFromSuperlayer];
354 [rootLayer addSublayer:m_tileMapHostLayer.get()];
355
356 // Pick a good scale.
357 IntSize viewSize = m_webPageProxy.viewSize();
358
359 [m_tileMapHostLayer setBounds:FloatRect(FloatPoint(), contentsSize)];
360 [m_tileMapHostLayer setPosition:indicatorLocation()];
361 [m_tileMapHostLayer setTransform:CATransform3DMakeScale(scale, scale, 1)];
362
363 if (rootLayerChanged) {
364 [m_tileMapHostLayer setSublayers:@[]];
365 [m_tileMapHostLayer addSublayer:m_debugIndicatorLayerTreeHost->rootLayer()];
366 [m_tileMapHostLayer addSublayer:m_exposedRectIndicatorLayer.get()];
367 }
368
369 const float indicatorBorderWidth = 1;
370 float counterScaledBorder = indicatorBorderWidth / scale;
371
372 [m_exposedRectIndicatorLayer setBorderWidth:counterScaledBorder];
373
374 if (m_webPageProxy.delegatesScrolling()) {
375 FloatRect scaledExposedRect;
376#if PLATFORM(IOS_FAMILY)
377 scaledExposedRect = m_webPageProxy.exposedContentRect();
378#else
379 if (auto viewExposedRect = m_webPageProxy.viewExposedRect())
380 scaledExposedRect = *viewExposedRect;
381 float scale = 1 / m_webPageProxy.pageScaleFactor();
382 scaledExposedRect.scale(scale);
383#endif
384 [m_exposedRectIndicatorLayer setPosition:scaledExposedRect.location()];
385 [m_exposedRectIndicatorLayer setBounds:FloatRect(FloatPoint(), scaledExposedRect.size())];
386 } else {
387 [m_exposedRectIndicatorLayer setPosition:scrollPosition];
388 [m_exposedRectIndicatorLayer setBounds:FloatRect(FloatPoint(), viewSize)];
389 }
390}

392void RemoteLayerTreeDrawingAreaProxy::initializeDebugIndicator()
393{
394 m_debugIndicatorLayerTreeHost = makeUnique<RemoteLayerTreeHost>(*this);
395 m_debugIndicatorLayerTreeHost->setIsDebugLayerTreeHost(true);
396
397 m_tileMapHostLayer = adoptNS([[CALayer alloc] init]);
398 [m_tileMapHostLayer setName:@"Tile map host"];
399 [m_tileMapHostLayer setDelegate:[WebActionDisablingCALayerDelegate shared]];
400 [m_tileMapHostLayer setAnchorPoint:CGPointZero];
401 [m_tileMapHostLayer setOpacity:0.8];
402 [m_tileMapHostLayer setMasksToBounds:YES];
403 [m_tileMapHostLayer setBorderWidth:2];
404
405 CGColorSpaceRef colorSpace = sRGBColorSpaceRef();
406 {
407 const CGFloat components[] = { 1, 1, 1, 0.6 };
408 RetainPtr<CGColorRef> color = adoptCF(CGColorCreate(colorSpace, components));
409 [m_tileMapHostLayer setBackgroundColor:color.get()];
410
411 const CGFloat borderComponents[] = { 0, 0, 0, 1 };
412 RetainPtr<CGColorRef> borderColor = adoptCF(CGColorCreate(colorSpace, borderComponents));
413 [m_tileMapHostLayer setBorderColor:borderColor.get()];
414 }
415
416 m_exposedRectIndicatorLayer = adoptNS([[CALayer alloc] init]);
417 [m_exposedRectIndicatorLayer setDelegate:[WebActionDisablingCALayerDelegate shared]];
418 [m_exposedRectIndicatorLayer setAnchorPoint:CGPointZero];
419
420 {
421 const CGFloat components[] = { 0, 1, 0, 1 };
422 RetainPtr<CGColorRef> color = adoptCF(CGColorCreate(colorSpace, components));
423 [m_exposedRectIndicatorLayer setBorderColor:color.get()];
424 }
425}

427void RemoteLayerTreeDrawingAreaProxy::didRefreshDisplay()
428{
429 if (!m_webPageProxy.hasRunningProcess())
430 return;
431
432 if (m_didUpdateMessageState != NeedsDidUpdate) {
433 m_didUpdateMessageState = MissedCommit;
434#if PLATFORM(IOS_FAMILY)
435 [displayLinkHandler() pause];
436#endif
437 return;
438 }
439
440 m_didUpdateMessageState = DoesNotNeedDidUpdate;
441
442 // Waiting for CA to commit is insufficient, because the render server can still be
443 // using our backing store. We can improve this by waiting for the render server to commit
444 // if we find API to do so, but for now we will make extra buffers if need be.
445 send(Messages::DrawingArea::DidUpdate());
446
447 m_lastVisibleTransactionID = m_transactionIDForPendingCACommit;
448
449 m_webPageProxy.didUpdateActivityState();
450}

452void RemoteLayerTreeDrawingAreaProxy::waitForDidUpdateActivityState(ActivityStateChangeID activityStateChangeID)
453{
454 ASSERT(activityStateChangeID != ActivityStateChangeAsynchronous);
455
456 // We must send the didUpdate message before blocking on the next commit, otherwise
457 // we can be guaranteed that the next commit won't come until after the waitForAndDispatchImmediately times out.
458 if (m_didUpdateMessageState != DoesNotNeedDidUpdate)
459 didRefreshDisplay();
460
461 static Seconds activityStateUpdateTimeout = [] {
462 if (id value = [[NSUserDefaults standardUserDefaults] objectForKey:@"WebKitOverrideActivityStateUpdateTimeout"])
463 return Seconds([value doubleValue]);
464 return Seconds::fromMilliseconds(250);
465 }();
466
467 auto startTime = MonotonicTime::now();
468 while (process().connection()->waitForAndDispatchImmediately<Messages::RemoteLayerTreeDrawingAreaProxy::CommitLayerTree>(m_identifier, activityStateUpdateTimeout - (MonotonicTime::now() - startTime), IPC::WaitForOption::InterruptWaitingIfSyncMessageArrives)) {
469 if (activityStateChangeID == ActivityStateChangeAsynchronous || activityStateChangeID <= m_activityStateChangeID)
470 return;
471 }
472}
