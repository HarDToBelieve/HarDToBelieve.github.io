42class RemoteLayerTreeHost {
43 WTF_MAKE_FAST_ALLOCATED;
44public:
45 explicit RemoteLayerTreeHost(RemoteLayerTreeDrawingAreaProxy&);
46 ~RemoteLayerTreeHost();
47
48 RemoteLayerTreeNode* nodeForID(WebCore::GraphicsLayer::PlatformLayerID) const;
49 RemoteLayerTreeNode* rootNode() const { return m_rootNode; }
50
51 CALayer *layerForID(WebCore::GraphicsLayer::PlatformLayerID) const;
52 CALayer *rootLayer() const;
53
54 // Returns true if the root layer changed.
55 bool updateLayerTree(const RemoteLayerTreeTransaction&, float indicatorScaleFactor = 1);
56
57 void setIsDebugLayerTreeHost(bool flag) { m_isDebugLayerTreeHost = flag; }
58 bool isDebugLayerTreeHost() const { return m_isDebugLayerTreeHost; }
59
60 typedef HashMap<WebCore::GraphicsLayer::PlatformLayerID, RetainPtr<WKAnimationDelegate>> LayerAnimationDelegateMap;
61 LayerAnimationDelegateMap& animationDelegates() { return m_animationDelegates; }
62
63 void animationDidStart(WebCore::GraphicsLayer::PlatformLayerID, CAAnimation *, MonotonicTime startTime);
64 void animationDidEnd(WebCore::GraphicsLayer::PlatformLayerID, CAAnimation *);
65
66 void detachFromDrawingArea();
67 void clearLayers();
68
69 // Detach the root layer; it will be reattached upon the next incoming commit.
70 void detachRootLayer();
71
72 // Turn all CAMachPort objects in layer contents into actual IOSurfaces.
73 // This avoids keeping an outstanding InUse reference when suspended.
74 void mapAllIOSurfaceBackingStore();
75
76 CALayer *layerWithIDForTesting(uint64_t) const;
77
78private:
79 void createLayer(const RemoteLayerTreeTransaction::LayerCreationProperties&);
80 std::unique_ptr<RemoteLayerTreeNode> makeNode(const RemoteLayerTreeTransaction::LayerCreationProperties&);
81
82 void layerWillBeRemoved(WebCore::GraphicsLayer::PlatformLayerID);
83
84 RemoteLayerTreeDrawingAreaProxy* m_drawingArea { nullptr };
85 RemoteLayerTreeNode* m_rootNode { nullptr };
86 HashMap<WebCore::GraphicsLayer::PlatformLayerID, std::unique_ptr<RemoteLayerTreeNode>> m_nodes;
87 HashMap<WebCore::GraphicsLayer::PlatformLayerID, RetainPtr<WKAnimationDelegate>> m_animationDelegates;
88 bool m_isDebugLayerTreeHost { false };
89};
