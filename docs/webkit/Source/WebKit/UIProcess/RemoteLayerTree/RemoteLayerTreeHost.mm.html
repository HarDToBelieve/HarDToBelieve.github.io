67bool RemoteLayerTreeHost::updateLayerTree(const RemoteLayerTreeTransaction& transaction, float indicatorScaleFactor)
68{
69 if (!m_drawingArea)
70 return false;
71
72 for (const auto& createdLayer : transaction.createdLayers())
73 createLayer(createdLayer);
74
75 bool rootLayerChanged = false;
76 auto* rootNode = nodeForID(transaction.rootLayerID());
77
78 if (!rootNode)
79 REMOTE_LAYER_TREE_HOST_RELEASE_LOG("%p RemoteLayerTreeHost::updateLayerTree - failed to find root layer with ID %llu", this, transaction.rootLayerID());
80
81 if (m_rootNode != rootNode) {
82 m_rootNode = rootNode;
83 rootLayerChanged = true;
84 }
85
86 struct LayerAndClone {
87 GraphicsLayer::PlatformLayerID layerID;
88 GraphicsLayer::PlatformLayerID cloneLayerID;
89 };
90 Vector<LayerAndClone> clonesToUpdate;
91
92#if PLATFORM(MAC) || PLATFORM(MACCATALYST)
93 // Can't use the iOS code on macOS yet: rdar://problem/31247730
94 auto layerContentsType = RemoteLayerBackingStore::LayerContentsType::IOSurface;
95#else
96 auto layerContentsType = m_drawingArea->hasDebugIndicator() ? RemoteLayerBackingStore::LayerContentsType::IOSurface : RemoteLayerBackingStore::LayerContentsType::CAMachPort;
97#endif
98
99 for (auto& [layerID, propertiesPointer] : transaction.changedLayerProperties()) {
100 const RemoteLayerTreeTransaction::LayerProperties& properties = *propertiesPointer;
101
102 auto* node = nodeForID(layerID);
103 ASSERT(node);
104
105 if (!node) {
106 // We have evidence that this can still happen, but don't know how (see r241899 for one already-fixed cause).
107 REMOTE_LAYER_TREE_HOST_RELEASE_LOG("%p RemoteLayerTreeHost::updateLayerTree - failed to find layer with ID %llu", this, layerID);
108 continue;
109 }
110
111 RemoteLayerTreePropertyApplier::applyHierarchyUpdates(*node, properties, m_nodes);
112 }
113
114 for (auto& changedLayer : transaction.changedLayerProperties()) {
115 auto layerID = changedLayer.key;
116 const RemoteLayerTreeTransaction::LayerProperties& properties = *changedLayer.value;
117
118 auto* node = nodeForID(layerID);
119 ASSERT(node);
120
121 if (!node) {
122 // We have evidence that this can still happen, but don't know how (see r241899 for one already-fixed cause).
123 REMOTE_LAYER_TREE_HOST_RELEASE_LOG("%p RemoteLayerTreeHost::updateLayerTree - failed to find layer with ID %llu", this, layerID);
124 continue;
125 }
126
127 if (properties.changedProperties.contains(RemoteLayerTreeTransaction::ClonedContentsChanged) && properties.clonedLayerID)
128 clonesToUpdate.append({ layerID, properties.clonedLayerID });
129
130 RemoteLayerTreePropertyApplier::applyProperties(*node, this, properties, m_nodes, layerContentsType);
131
132 if (m_isDebugLayerTreeHost) {
133 if (properties.changedProperties.contains(RemoteLayerTreeTransaction::BorderWidthChanged))
134 node->layer().borderWidth = properties.borderWidth / indicatorScaleFactor;
135 node->layer().masksToBounds = false;
136 }
137 }
138
139 for (const auto& layerAndClone : clonesToUpdate)
140 layerForID(layerAndClone.layerID).contents = layerForID(layerAndClone.cloneLayerID).contents;
141
142 for (auto& destroyedLayer : transaction.destroyedLayers())
143 layerWillBeRemoved(destroyedLayer);
144
145 // Drop the contents of any layers which were unparented; the Web process will re-send
146 // the backing store in the commit that reparents them.
147 for (auto& newlyUnreachableLayerID : transaction.layerIDsWithNewlyUnreachableBackingStore())
148 layerForID(newlyUnreachableLayerID).contents = nullptr;
149
150 return rootLayerChanged;
151}

153RemoteLayerTreeNode* RemoteLayerTreeHost::nodeForID(GraphicsLayer::PlatformLayerID layerID) const
154{
155 if (!layerID)
156 return nullptr;
157
158 return m_nodes.get(layerID);
159}
160
161void RemoteLayerTreeHost::layerWillBeRemoved(WebCore::GraphicsLayer::PlatformLayerID layerID)
162{
163 auto animationDelegateIter = m_animationDelegates.find(layerID);
164 if (animationDelegateIter != m_animationDelegates.end()) {
165 [animationDelegateIter->value invalidate];
166 m_animationDelegates.remove(animationDelegateIter);
167 }
168
169 m_nodes.remove(layerID);
170}

172void RemoteLayerTreeHost::animationDidStart(WebCore::GraphicsLayer::PlatformLayerID layerID, CAAnimation *animation, MonotonicTime startTime)
173{
174 if (!m_drawingArea)
175 return;
176
177 CALayer *layer = layerForID(layerID);
178 if (!layer)
179 return;
180
181 String animationKey;
182 for (NSString *key in [layer animationKeys]) {
183 if ([layer animationForKey:key] == animation) {
184 animationKey = key;
185 break;
186 }
187 }
188
189 if (!animationKey.isEmpty())
190 m_drawingArea->acceleratedAnimationDidStart(layerID, animationKey, startTime);
191}
192
193void RemoteLayerTreeHost::animationDidEnd(WebCore::GraphicsLayer::PlatformLayerID layerID, CAAnimation *animation)
194{
195 if (!m_drawingArea)
196 return;
197
198 CALayer *layer = layerForID(layerID);
199 if (!layer)
200 return;
201
202 String animationKey;
203 for (NSString *key in [layer animationKeys]) {
204 if ([layer animationForKey:key] == animation) {
205 animationKey = key;
206 break;
207 }
208 }
209
210 if (!animationKey.isEmpty())
211 m_drawingArea->acceleratedAnimationDidEnd(layerID, animationKey);
212}

219void RemoteLayerTreeHost::clearLayers()
220{
221 for (auto& keyAndNode : m_nodes) {
222 m_animationDelegates.remove(keyAndNode.key);
223 keyAndNode.value->detachFromParent();
224 }
225
226 m_nodes.clear();
227 m_rootNode = nullptr;
228}

250void RemoteLayerTreeHost::createLayer(const RemoteLayerTreeTransaction::LayerCreationProperties& properties)
251{
252 ASSERT(!m_nodes.contains(properties.layerID));
253
254 auto node = makeNode(properties);
255
256 m_nodes.add(properties.layerID, WTFMove(node));
257}

259#if !PLATFORM(IOS_FAMILY)
260std::unique_ptr<RemoteLayerTreeNode> RemoteLayerTreeHost::makeNode(const RemoteLayerTreeTransaction::LayerCreationProperties& properties)
261{
262 auto makeWithLayer = [&] (RetainPtr<CALayer>&& layer) {
263 return makeUnique<RemoteLayerTreeNode>(properties.layerID, WTFMove(layer));
264 };
265
266 switch (properties.type) {
267 case PlatformCALayer::LayerTypeLayer:
268 case PlatformCALayer::LayerTypeWebLayer:
269 case PlatformCALayer::LayerTypeRootLayer:
270 case PlatformCALayer::LayerTypeSimpleLayer:
271 case PlatformCALayer::LayerTypeTiledBackingLayer:
272 case PlatformCALayer::LayerTypePageTiledBackingLayer:
273 case PlatformCALayer::LayerTypeTiledBackingTileLayer:
274 case PlatformCALayer::LayerTypeScrollContainerLayer:
275#if ENABLE(MODEL_ELEMENT)
276 case PlatformCALayer::LayerTypeModelLayer:
277#endif
278 return RemoteLayerTreeNode::createWithPlainLayer(properties.layerID);
279
280 case PlatformCALayer::LayerTypeTransformLayer:
281 return makeWithLayer(adoptNS([[CATransformLayer alloc] init]));
282
283 case PlatformCALayer::LayerTypeBackdropLayer:
284 case PlatformCALayer::LayerTypeLightSystemBackdropLayer:
285 case PlatformCALayer::LayerTypeDarkSystemBackdropLayer:
286#if ENABLE(FILTERS_LEVEL_2)
287 return makeWithLayer(adoptNS([[CABackdropLayer alloc] init]));
288#else
289 ASSERT_NOT_REACHED();
290 return RemoteLayerTreeNode::createWithPlainLayer(properties.layerID);
291#endif
292 case PlatformCALayer::LayerTypeCustom:
293 case PlatformCALayer::LayerTypeAVPlayerLayer:
294 case PlatformCALayer::LayerTypeContentsProvidedLayer:
295 if (m_isDebugLayerTreeHost)
296 return RemoteLayerTreeNode::createWithPlainLayer(properties.layerID);
297 return makeWithLayer([CALayer _web_renderLayerWithContextID:properties.hostingContextID]);
298
299 case PlatformCALayer::LayerTypeShapeLayer:
300 return makeWithLayer(adoptNS([[CAShapeLayer alloc] init]));
301
302 default:
303 ASSERT_NOT_REACHED();
304 return nullptr;
305 }
306}
307#endif

309void RemoteLayerTreeHost::detachRootLayer()
310{
311 if (!m_rootNode)
312 return;
313 m_rootNode->detachFromParent();
314 m_rootNode = nullptr;
315}

317static void recursivelyMapIOSurfaceBackingStore(CALayer *layer)
318{
319 if (layer.contents && CFGetTypeID((__bridge CFTypeRef)layer.contents) == CAMachPortGetTypeID()) {
320 MachSendRight port = MachSendRight::create(CAMachPortGetPort((__bridge CAMachPortRef)layer.contents));
321 auto surface = WebCore::IOSurface::createFromSendRight(WTFMove(port), WebCore::DestinationColorSpace::SRGB());
322 layer.contents = surface ? surface->asLayerContents() : nil;
323 }
324
325 for (CALayer *sublayer in layer.sublayers)
326 recursivelyMapIOSurfaceBackingStore(sublayer);
327}
328
329void RemoteLayerTreeHost::mapAllIOSurfaceBackingStore()
330{
331 recursivelyMapIOSurfaceBackingStore(rootLayer());
332}
