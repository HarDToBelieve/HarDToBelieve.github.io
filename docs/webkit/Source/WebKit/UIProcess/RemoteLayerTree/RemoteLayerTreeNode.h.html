28#include <WebCore/GraphicsLayer.h>

30#include <wtf/Vector.h>

32OBJC_CLASS CALayer;
33OBJC_CLASS NSString;
34#if PLATFORM(IOS_FAMILY)
35OBJC_CLASS UIView;
36#endif
37
38namespace WebKit {
39
40class RemoteLayerTreeScrollbars;
41
42class RemoteLayerTreeNode {
43 WTF_MAKE_FAST_ALLOCATED;
44public:
45 RemoteLayerTreeNode(WebCore::GraphicsLayer::PlatformLayerID, RetainPtr<CALayer>);
46#if PLATFORM(IOS_FAMILY)
47 RemoteLayerTreeNode(WebCore::GraphicsLayer::PlatformLayerID, RetainPtr<UIView>);
48#endif
49 ~RemoteLayerTreeNode();
50
51 static std::unique_ptr<RemoteLayerTreeNode> createWithPlainLayer(WebCore::GraphicsLayer::PlatformLayerID);
52
53 CALayer *layer() const { return m_layer.get(); }
54#if PLATFORM(IOS_FAMILY)
55 UIView *uiView() const { return m_uiView.get(); }
56#endif
57
58 WebCore::GraphicsLayer::PlatformLayerID layerID() const { return m_layerID; }
59
60 const WebCore::EventRegion& eventRegion() const { return m_eventRegion; }
61 void setEventRegion(const WebCore::EventRegion&);
62
63 // Non-ancestor scroller that controls positioning of the layer.
64 WebCore::GraphicsLayer::PlatformLayerID actingScrollContainerID() const { return m_actingScrollContainerID; }
65 // Ancestor scrollers that don't affect positioning of the layer.
66 const Vector<WebCore::GraphicsLayer::PlatformLayerID>& stationaryScrollContainerIDs() const { return m_stationaryScrollContainerIDs; }
67
68 void setActingScrollContainerID(WebCore::GraphicsLayer::PlatformLayerID value) { m_actingScrollContainerID = value; }
69 void setStationaryScrollContainerIDs(Vector<WebCore::GraphicsLayer::PlatformLayerID>&& value) { m_stationaryScrollContainerIDs = WTFMove(value); }
70
71 void detachFromParent();
72
73 static WebCore::GraphicsLayer::PlatformLayerID layerID(CALayer *);
74 static RemoteLayerTreeNode* forCALayer(CALayer *);
75
76 static NSString *appendLayerDescription(NSString *description, CALayer *);
77
78private:
79 void initializeLayer();
80
81 WebCore::GraphicsLayer::PlatformLayerID m_layerID;
82
83 RetainPtr<CALayer> m_layer;
84#if PLATFORM(IOS_FAMILY)
85 RetainPtr<UIView> m_uiView;
86#endif
87
88 WebCore::EventRegion m_eventRegion;
89
90 WebCore::GraphicsLayer::PlatformLayerID m_actingScrollContainerID { 0 };
91 Vector<WebCore::GraphicsLayer::PlatformLayerID> m_stationaryScrollContainerIDs;
92};
