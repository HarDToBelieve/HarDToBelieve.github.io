39static NSString *const WKRemoteLayerTreeNodePropertyKey = @"WKRemoteLayerTreeNode";
40
41RemoteLayerTreeNode::RemoteLayerTreeNode(WebCore::GraphicsLayer::PlatformLayerID layerID, RetainPtr<CALayer> layer)
42 : m_layerID(layerID)
43 , m_layer(WTFMove(layer))
44{
45 initializeLayer();
46 [m_layer setDelegate:[WebActionDisablingCALayerDelegate shared]];
47}
48
49#if PLATFORM(IOS_FAMILY)
50RemoteLayerTreeNode::RemoteLayerTreeNode(WebCore::GraphicsLayer::PlatformLayerID layerID, RetainPtr<UIView> uiView)
51 : m_layerID(layerID)
52 , m_layer([uiView.get() layer])
53 , m_uiView(WTFMove(uiView))
54{
55 initializeLayer();
56}
57#endif
58
59RemoteLayerTreeNode::~RemoteLayerTreeNode()
60{
61 [layer() setValue:nil forKey:WKRemoteLayerTreeNodePropertyKey];
62}
63
64std::unique_ptr<RemoteLayerTreeNode> RemoteLayerTreeNode::createWithPlainLayer(WebCore::GraphicsLayer::PlatformLayerID layerID)
65{
66 RetainPtr<CALayer> layer = adoptNS([[WKCompositingLayer alloc] init]);
67 return makeUnique<RemoteLayerTreeNode>(layerID, WTFMove(layer));
68}
69
70void RemoteLayerTreeNode::detachFromParent()
71{
72#if PLATFORM(IOS_FAMILY)
73 if (auto view = uiView()) {
74 [view removeFromSuperview];
75 return;
76 }
77#endif
78 [layer() removeFromSuperlayer];
79}
80
81void RemoteLayerTreeNode::setEventRegion(const WebCore::EventRegion& eventRegion)
82{
83 m_eventRegion = eventRegion;
84}
85
86void RemoteLayerTreeNode::initializeLayer()
87{
88 [layer() setValue:[NSValue valueWithPointer:this] forKey:WKRemoteLayerTreeNodePropertyKey];
89}
90
91WebCore::GraphicsLayer::PlatformLayerID RemoteLayerTreeNode::layerID(CALayer *layer)
92{
93 auto* node = forCALayer(layer);
94 return node ? node->layerID() : 0;
95}
96
97RemoteLayerTreeNode* RemoteLayerTreeNode::forCALayer(CALayer *layer)
98{
99 return static_cast<RemoteLayerTreeNode*>([[layer valueForKey:WKRemoteLayerTreeNodePropertyKey] pointerValue]);
100}
101
102NSString *RemoteLayerTreeNode::appendLayerDescription(NSString *description, CALayer *layer)
103{
104 NSString *layerDescription = [NSString stringWithFormat:@" layerID = %llu \"%@\"", WebKit::RemoteLayerTreeNode::layerID(layer), layer.name ? layer.name : @""];
105 return [description stringByAppendingString:layerDescription];
106}
