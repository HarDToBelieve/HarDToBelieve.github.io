58bool RemoteLayerTreeScrollingPerformanceData::BlankPixelCount::canCoalesce(BlankPixelCount::EventType type, unsigned pixelCount) const
59{
60 return eventType == type && blankPixelCount == pixelCount;
61}
62
63void RemoteLayerTreeScrollingPerformanceData::appendBlankPixelCount(BlankPixelCount::EventType eventType, unsigned blankPixelCount)
64{
65 double now = CFAbsoluteTimeGetCurrent();
66 if (!m_blankPixelCounts.isEmpty() && m_blankPixelCounts.last().canCoalesce(eventType, blankPixelCount)) {
67 m_blankPixelCounts.last().endTime = now;
68 return;
69 }
70
71 m_blankPixelCounts.append(BlankPixelCount(now, now, eventType, blankPixelCount));
72}

74NSArray *RemoteLayerTreeScrollingPerformanceData::data()
75{
76 return createNSArray(m_blankPixelCounts, [] (auto& pixelData) {
77 return @[
78 @(pixelData.startTime),
79 (pixelData.eventType == BlankPixelCount::Filled) ? @"filled" : @"exposed",
80 @(pixelData.blankPixelCount)
81 ];
82 }).autorelease();
83}

85static CALayer *findTileGridContainerLayer(CALayer *layer)
86{
87 for (CALayer *currLayer : [layer sublayers]) {
88 String layerName = [currLayer name];
89 if (layerName == TileController::tileGridContainerLayerName())
90 return currLayer;
91
92 if (CALayer *foundLayer = findTileGridContainerLayer(currLayer))
93 return foundLayer;
94 }
95
96 return nil;
97}

99unsigned RemoteLayerTreeScrollingPerformanceData::blankPixelCount(const FloatRect& visibleRect) const
100{
101 CALayer *rootLayer = m_drawingArea.remoteLayerTreeHost().rootLayer();
102
103 CALayer *tileGridContainer = findTileGridContainerLayer(rootLayer);
104 if (!tileGridContainer) {
105 NSLog(@"Failed to find TileGrid Container Layer");
106 return UINT_MAX;
107 }
108
109 FloatRect visibleRectExcludingToolbar = visibleRect;
110 if (visibleRectExcludingToolbar.y() < 0)
111 visibleRectExcludingToolbar.setY(0);
112
113 Region paintedVisibleTileRegion;
114
115 for (CALayer *tileLayer : [tileGridContainer sublayers]) {
116 FloatRect tileRect = [tileLayer convertRect:[tileLayer bounds] toLayer:tileGridContainer];
117
118 tileRect.intersect(visibleRectExcludingToolbar);
119
120 if (!tileRect.isEmpty())
121 paintedVisibleTileRegion.unite(enclosingIntRect(tileRect));
122 }
123
124 Region uncoveredRegion(enclosingIntRect(visibleRectExcludingToolbar));
125 uncoveredRegion.subtract(paintedVisibleTileRegion);
126
127 return uncoveredRegion.totalArea();
128}
