67const RemoteLayerTreeHost* RemoteScrollingCoordinatorProxy::layerTreeHost() const
68{
69 DrawingAreaProxy* drawingArea = m_webPageProxy.drawingArea();
70 if (!is<RemoteLayerTreeDrawingAreaProxy>(drawingArea)) {
71 ASSERT_NOT_REACHED();
72 return nullptr;
73 }
74
75 RemoteLayerTreeDrawingAreaProxy& remoteDrawingArea = downcast<RemoteLayerTreeDrawingAreaProxy>(*drawingArea);
76 return &remoteDrawingArea.remoteLayerTreeHost();
77}

79void RemoteScrollingCoordinatorProxy::commitScrollingTreeState(const RemoteScrollingCoordinatorTransaction& transaction, RequestedScrollInfo& requestedScrollInfo)
80{
81 m_requestedScrollInfo = &requestedScrollInfo;
82
83 auto stateTree = WTFMove(const_cast<RemoteScrollingCoordinatorTransaction&>(transaction).scrollingStateTree());
84
85 auto* layerTreeHost = this->layerTreeHost();
86 if (!layerTreeHost) {
87 ASSERT_NOT_REACHED();
88 return;
89 }
90
91 connectStateNodeLayers(*stateTree, *layerTreeHost);
92 m_scrollingTree->commitTreeState(WTFMove(stateTree));
93
94 establishLayerTreeScrollingRelations(*layerTreeHost);
95
96 m_requestedScrollInfo = nullptr;
97}

101void RemoteScrollingCoordinatorProxy::connectStateNodeLayers(ScrollingStateTree& stateTree, const RemoteLayerTreeHost& layerTreeHost)
102{
103 using PlatformLayerID = GraphicsLayer::PlatformLayerID;
104
105 for (auto& currNode : stateTree.nodeMap().values()) {
106 if (currNode->hasChangedProperty(ScrollingStateNode::Property::Layer))
107 currNode->setLayer(layerTreeHost.layerForID(PlatformLayerID { currNode->layer() }));
108
109 switch (currNode->nodeType()) {
110 case ScrollingNodeType::MainFrame:
111 case ScrollingNodeType::Subframe: {
112 ScrollingStateFrameScrollingNode& scrollingStateNode = downcast<ScrollingStateFrameScrollingNode>(*currNode);
113
114 if (scrollingStateNode.hasChangedProperty(ScrollingStateNode::Property::ScrollContainerLayer))
115 scrollingStateNode.setScrollContainerLayer(layerTreeHost.layerForID(PlatformLayerID { scrollingStateNode.scrollContainerLayer() }));
116
117 if (scrollingStateNode.hasChangedProperty(ScrollingStateNode::Property::ScrolledContentsLayer))
118 scrollingStateNode.setScrolledContentsLayer(layerTreeHost.layerForID(PlatformLayerID { scrollingStateNode.scrolledContentsLayer() }));
119
120 if (scrollingStateNode.hasChangedProperty(ScrollingStateNode::Property::CounterScrollingLayer))
121 scrollingStateNode.setCounterScrollingLayer(layerTreeHost.layerForID(PlatformLayerID { scrollingStateNode.counterScrollingLayer() }));
122
123 if (scrollingStateNode.hasChangedProperty(ScrollingStateNode::Property::InsetClipLayer))
124 scrollingStateNode.setInsetClipLayer(layerTreeHost.layerForID(PlatformLayerID { scrollingStateNode.insetClipLayer() }));
125
126 if (scrollingStateNode.hasChangedProperty(ScrollingStateNode::Property::ContentShadowLayer))
127 scrollingStateNode.setContentShadowLayer(layerTreeHost.layerForID(PlatformLayerID { scrollingStateNode.contentShadowLayer() }));
128
129 // FIXME: we should never have header and footer layers coming from the WebProcess.
130 if (scrollingStateNode.hasChangedProperty(ScrollingStateNode::Property::HeaderLayer))
131 scrollingStateNode.setHeaderLayer(layerTreeHost.layerForID(PlatformLayerID { scrollingStateNode.headerLayer() }));
132
133 if (scrollingStateNode.hasChangedProperty(ScrollingStateNode::Property::FooterLayer))
134 scrollingStateNode.setFooterLayer(layerTreeHost.layerForID(PlatformLayerID { scrollingStateNode.footerLayer() }));
135
136 if (scrollingStateNode.hasChangedProperty(ScrollingStateNode::Property::VerticalScrollbarLayer))
137 scrollingStateNode.setVerticalScrollbarLayer(layerTreeHost.layerForID(PlatformLayerID { scrollingStateNode.verticalScrollbarLayer() }));
138
139 if (scrollingStateNode.hasChangedProperty(ScrollingStateNode::Property::HorizontalScrollbarLayer))
140 scrollingStateNode.setHorizontalScrollbarLayer(layerTreeHost.layerForID(PlatformLayerID { scrollingStateNode.horizontalScrollbarLayer() }));
141
142 if (scrollingStateNode.hasChangedProperty(ScrollingStateNode::Property::RootContentsLayer))
143 scrollingStateNode.setRootContentsLayer(layerTreeHost.layerForID(PlatformLayerID { scrollingStateNode.rootContentsLayer() }));
144 break;
145 }
146 case ScrollingNodeType::Overflow: {
147 ScrollingStateOverflowScrollingNode& scrollingStateNode = downcast<ScrollingStateOverflowScrollingNode>(*currNode);
148 if (scrollingStateNode.hasChangedProperty(ScrollingStateNode::Property::ScrollContainerLayer))
149 scrollingStateNode.setScrollContainerLayer(layerTreeHost.layerForID(PlatformLayerID { scrollingStateNode.scrollContainerLayer() }));
150
151 if (scrollingStateNode.hasChangedProperty(ScrollingStateNode::Property::ScrolledContentsLayer))
152 scrollingStateNode.setScrolledContentsLayer(layerTreeHost.layerForID(PlatformLayerID { scrollingStateNode.scrolledContentsLayer() }));
153
154 if (scrollingStateNode.hasChangedProperty(ScrollingStateNode::Property::VerticalScrollbarLayer))
155 scrollingStateNode.setVerticalScrollbarLayer(layerTreeHost.layerForID(PlatformLayerID { scrollingStateNode.verticalScrollbarLayer() }));
156
157 if (scrollingStateNode.hasChangedProperty(ScrollingStateNode::Property::HorizontalScrollbarLayer))
158 scrollingStateNode.setHorizontalScrollbarLayer(layerTreeHost.layerForID(PlatformLayerID { scrollingStateNode.horizontalScrollbarLayer() }));
159 break;
160 }
161 case ScrollingNodeType::OverflowProxy:
162 case ScrollingNodeType::FrameHosting:
163 case ScrollingNodeType::Fixed:
164 case ScrollingNodeType::Sticky:
165 case ScrollingNodeType::Positioned:
166 break;
167 }
168 }

177bool RemoteScrollingCoordinatorProxy::handleWheelEvent(const PlatformWheelEvent& wheelEvent)
178{
179 if (!m_scrollingTree)
180 return false;
181
182 auto processingSteps = m_scrollingTree->determineWheelEventProcessing(wheelEvent);
183 if (processingSteps.containsAny({ WheelEventProcessingSteps::MainThreadForScrolling, WheelEventProcessingSteps::MainThreadForNonBlockingDOMEventDispatch, WheelEventProcessingSteps::MainThreadForBlockingDOMEventDispatch }))
184 return false;
185
186 if (m_scrollingTree->willWheelEventStartSwipeGesture(wheelEvent))
187 return false;
188
189 auto result = m_scrollingTree->handleWheelEvent(wheelEvent);
190 return result.wasHandled;
191}

218// This comes from the scrolling tree.
219void RemoteScrollingCoordinatorProxy::scrollingTreeNodeDidScroll(ScrollingNodeID scrolledNodeID, const FloatPoint& newScrollPosition, const std::optional<FloatPoint>& layoutViewportOrigin, ScrollingLayerPositionAction scrollingLayerPositionAction)
220{
221 // Scroll updates for the main frame are sent via WebPageProxy::updateVisibleContentRects()
222 // so don't send them here.
223 if (!m_propagatesMainFrameScrolls && scrolledNodeID == rootScrollingNodeID())
224 return;
225
226 if (m_webPageProxy.scrollingUpdatesDisabledForTesting())
227 return;
228
229#if PLATFORM(IOS_FAMILY)
230 m_webPageProxy.scrollingNodeScrollViewDidScroll();
231#endif
232
233 if (m_scrollingTree->isHandlingProgrammaticScroll())
234 return;
235
236 m_webPageProxy.send(Messages::RemoteScrollingCoordinator::ScrollPositionChangedForNode(scrolledNodeID, newScrollPosition, scrollingLayerPositionAction == ScrollingLayerPositionAction::Sync));
237}

239void RemoteScrollingCoordinatorProxy::scrollingTreeNodeRequestsScroll(ScrollingNodeID scrolledNodeID, const FloatPoint& scrollPosition, ScrollType scrollType, ScrollClamping)
240{
241 if (scrolledNodeID == rootScrollingNodeID() && m_requestedScrollInfo) {
242 m_requestedScrollInfo->requestsScrollPositionUpdate = true;
243 m_requestedScrollInfo->requestIsProgrammaticScroll = scrollType == ScrollType::Programmatic;
244 m_requestedScrollInfo->requestedScrollPosition = scrollPosition;
245 }

256bool RemoteScrollingCoordinatorProxy::hasScrollableMainFrame() const
257{
258 auto* rootNode = m_scrollingTree->rootNode();
259 return rootNode && rootNode->canHaveScrollbars();
260}
261
262bool RemoteScrollingCoordinatorProxy::hasScrollableOrZoomedMainFrame() const
263{
264 auto* rootNode = m_scrollingTree->rootNode();
265 if (!rootNode)
266 return false;
267
268#if PLATFORM(IOS_FAMILY)
269 if (WebCore::IOSApplication::isEventbrite() && !linkedOnOrAfter(WebCore::SDKVersion::FirstThatSupportsOverflowHiddenOnMainFrame))
270 return true;
271#endif
272
273 return rootNode->canHaveScrollbars() || rootNode->visualViewportIsSmallerThanLayoutViewport();
274}
