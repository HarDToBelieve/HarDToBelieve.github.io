39Ref<ScrollingTreeOverflowScrollingNodeIOS> ScrollingTreeOverflowScrollingNodeIOS::create(WebCore::ScrollingTree& scrollingTree, WebCore::ScrollingNodeID nodeID)

54UIScrollView* ScrollingTreeOverflowScrollingNodeIOS::scrollView() const
55{
56 return m_scrollingNodeDelegate->scrollView();
57}
58
59void ScrollingTreeOverflowScrollingNodeIOS::commitStateBeforeChildren(const WebCore::ScrollingStateNode& stateNode)
60{
61 if (stateNode.hasChangedProperty(ScrollingStateNode::Property::ScrollContainerLayer))
62 m_scrollingNodeDelegate->resetScrollViewDelegate();
63
64 ScrollingTreeOverflowScrollingNode::commitStateBeforeChildren(stateNode);
65 m_scrollingNodeDelegate->commitStateBeforeChildren(downcast<ScrollingStateScrollingNode>(stateNode));
66}
67
68void ScrollingTreeOverflowScrollingNodeIOS::commitStateAfterChildren(const ScrollingStateNode& stateNode)
69{
70 ScrollingTreeOverflowScrollingNode::commitStateAfterChildren(stateNode);
71
72 const auto& scrollingStateNode = downcast<ScrollingStateScrollingNode>(stateNode);
73 m_scrollingNodeDelegate->commitStateAfterChildren(scrollingStateNode);
74}
75
76void ScrollingTreeOverflowScrollingNodeIOS::repositionScrollingLayers()
77{
78 m_scrollingNodeDelegate->repositionScrollingLayers();
79}
