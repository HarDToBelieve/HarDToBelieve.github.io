40enum class FeatureToAnimate {
41 KnobAlpha,
42 TrackAlpha,
43 UIStateTransition,
44 ExpansionTransition
45};

47@interface WKScrollbarPartAnimation : NSAnimation {

56@implementation WKScrollbarPartAnimation
57
58- (id)initWithScroller:(WebKit::ScrollerMac*)scroller featureToAnimate:(FeatureToAnimate)featureToAnimate animateFrom:(CGFloat)startValue animateTo:(CGFloat)endValue duration:(NSTimeInterval)duration
59{
60 self = [super initWithDuration:duration animationCurve:NSAnimationEaseInOut];
61 if (!self)
62 return nil;
63
64 _scroller = scroller;
65 _featureToAnimate = featureToAnimate;
66 _startValue = startValue;
67 _endValue = endValue;
68
69 [self setAnimationBlockingMode:NSAnimationNonblocking];
70
71 return self;
72}
73
74- (void)startAnimation
75{
76 ASSERT(_scroller);
77
78 [super startAnimation];
79}
80
81- (void)setStartValue:(CGFloat)startValue
82{
83 _startValue = startValue;
84}
85
86- (void)setEndValue:(CGFloat)endValue
87{
88 _endValue = endValue;
89}
90
91- (void)setCurrentProgress:(NSAnimationProgress)progress
92{
93 [super setCurrentProgress:progress];
94
95 CGFloat currentValue;
96 if (_startValue > _endValue)
97 currentValue = 1 - progress;
98 else
99 currentValue = progress;
100
101 switch (_featureToAnimate) {
102 case FeatureToAnimate::KnobAlpha:
103 [_scroller->scrollerImp() setKnobAlpha:currentValue];
104 break;
105 case FeatureToAnimate::TrackAlpha:
106 [_scroller->scrollerImp() setTrackAlpha:currentValue];
107 break;
108 case FeatureToAnimate::UIStateTransition:
109 [_scroller->scrollerImp() setUiStateTransitionProgress:currentValue];
110 break;
111 case FeatureToAnimate::ExpansionTransition:
112 [_scroller->scrollerImp() setExpansionTransitionProgress:currentValue];
113 break;
114 }
115}
116
117- (void)invalidate
118{
119 BEGIN_BLOCK_OBJC_EXCEPTIONS
120 [self stopAnimation];
121 END_BLOCK_OBJC_EXCEPTIONS
122 _scroller = nullptr;
123}

217- (void)setUpAlphaAnimation:(RetainPtr<WKScrollbarPartAnimation>&)scrollbarPartAnimation featureToAnimate:(FeatureToAnimate)featureToAnimate animateAlphaTo:(CGFloat)newAlpha duration:(NSTimeInterval)duration
218{
219 // If we are currently animating,  stop
220 if (scrollbarPartAnimation) {
221 [scrollbarPartAnimation stopAnimation];
222 scrollbarPartAnimation = nil;
223 }
224
225 scrollbarPartAnimation = adoptNS([[WKScrollbarPartAnimation alloc] initWithScroller:_scroller
226 featureToAnimate:featureToAnimate
227 animateFrom:featureToAnimate == FeatureToAnimate::KnobAlpha ? [_scroller->scrollerImp() knobAlpha] : [_scroller->scrollerImp() trackAlpha]
228 animateTo:newAlpha
229 duration:duration]);
230 [scrollbarPartAnimation startAnimation];
231}

233- (void)scrollerImp:(NSScrollerImp *)scrollerImp animateKnobAlphaTo:(CGFloat)newKnobAlpha duration:(NSTimeInterval)duration
234{
235 if (!_scroller)
236 return;

238 ASSERT_UNUSED(scrollerImp, scrollerImp == _scroller->scrollerImp());
239 [self setUpAlphaAnimation:_knobAlphaAnimation featureToAnimate:FeatureToAnimate::KnobAlpha animateAlphaTo:newKnobAlpha duration:duration];
240}

243{

248 [self setUpAlphaAnimation:_trackAlphaAnimation featureToAnimate:FeatureToAnimate::TrackAlpha animateAlphaTo:newTrackAlpha duration:duration];
249}

251- (void)scrollerImp:(NSScrollerImp *)scrollerImp animateUIStateTransitionWithDuration:(NSTimeInterval)duration
252{
253 if (!_scroller)
254 return;
255
256 ASSERT(scrollerImp == _scroller->scrollerImp());
257
258 // UIStateTransition always animates to 1. In case an animation is in progress this avoids a hard transition.
259 [scrollerImp setUiStateTransitionProgress:1 - [scrollerImp uiStateTransitionProgress]];
260
261 if (!_uiStateTransitionAnimation) {
262 _uiStateTransitionAnimation = adoptNS([[WKScrollbarPartAnimation alloc] initWithScroller:_scroller
263 featureToAnimate:FeatureToAnimate::UIStateTransition
264 animateFrom:[scrollerImp uiStateTransitionProgress]
265 animateTo:1.0
266 duration:duration]);
267 } else {
268 // If we don't need to initialize the animation, just reset the values in case they have changed.
269 [_uiStateTransitionAnimation setStartValue:[scrollerImp uiStateTransitionProgress]];
270 [_uiStateTransitionAnimation setEndValue:1.0];
271 [_uiStateTransitionAnimation setDuration:duration];
272 }
273 [_uiStateTransitionAnimation startAnimation];
274}

276- (void)scrollerImp:(NSScrollerImp *)scrollerImp animateExpansionTransitionWithDuration:(NSTimeInterval)duration
277{
278 if (!_scroller)
279 return;
280
281 ASSERT(scrollerImp == _scroller->scrollerImp());
282
283 // ExpansionTransition always animates to 1. In case an animation is in progress this avoids a hard transition.
284 [scrollerImp setExpansionTransitionProgress:1 - [scrollerImp expansionTransitionProgress]];
285
286 if (!_expansionTransitionAnimation) {
287 _expansionTransitionAnimation = adoptNS([[WKScrollbarPartAnimation alloc] initWithScroller:_scroller
288 featureToAnimate:FeatureToAnimate::ExpansionTransition
289 animateFrom:[scrollerImp expansionTransitionProgress]
290 animateTo:1.0
291 duration:duration]);
292 } else {
293 // If we don't need to initialize the animation, just reset the values in case they have changed.
294 [_expansionTransitionAnimation setStartValue:[scrollerImp uiStateTransitionProgress]];
295 [_expansionTransitionAnimation setEndValue:1.0];
296 [_expansionTransitionAnimation setDuration:duration];
297 }
298 [_expansionTransitionAnimation startAnimation];
299}

334void ScrollerMac::attach()
335{
336 m_scrollerImpDelegate = adoptNS([[WKScrollerImpDelegate alloc] initWithScroller:this]);
337
338 NSScrollerStyle newStyle = [m_pair.scrollerImpPair() scrollerStyle];
339 m_scrollerImp = [NSScrollerImp scrollerImpWithStyle:newStyle controlSize:NSControlSizeRegular horizontal:m_orientation == Orientation::Horizontal replacingScrollerImp:nil];
340 [m_scrollerImp setDelegate:m_scrollerImpDelegate.get()];
341}

343void ScrollerMac::setHostLayer(CALayer *layer)
344{
345 if (m_hostLayer == layer)
346 return;
347
348 m_hostLayer = layer;
349
350 [m_scrollerImp setLayer:layer];
351
352 if (m_orientation == Orientation::Vertical)
353 [m_pair.scrollerImpPair() setVerticalScrollerImp:layer ? m_scrollerImp.get() : nil];
354 else
355 [m_pair.scrollerImpPair() setHorizontalScrollerImp:layer ? m_scrollerImp.get() : nil];
356}
357
358void ScrollerMac::updateValues()
359{
360 auto values = m_pair.valuesForOrientation(m_orientation);
361
362 BEGIN_BLOCK_OBJC_EXCEPTIONS
363
364 [m_scrollerImp setEnabled:!!m_hostLayer];
365 [m_scrollerImp setBoundsSize:NSSizeFromCGSize([m_hostLayer bounds].size)];
366 [m_scrollerImp setDoubleValue:values.value];
367 [m_scrollerImp setPresentationValue:values.value];
368 [m_scrollerImp setKnobProportion:values.proportion];
369
370 END_BLOCK_OBJC_EXCEPTIONS
371}

373WebCore::FloatPoint ScrollerMac::convertFromContent(const WebCore::FloatPoint& point) const
374{
375 return WebCore::FloatPoint { [m_hostLayer convertPoint:point fromLayer:[m_hostLayer superlayer]] };
376}
