41void SpeechRecognitionRemoteRealtimeMediaSourceManager::addSource(SpeechRecognitionRemoteRealtimeMediaSource& source, const WebCore::CaptureDevice& captureDevice)
42{
43 auto identifier = source.identifier();
44 ASSERT(!m_sources.contains(identifier));
45 m_sources.add(identifier, makeWeakPtr(source));
46
47#if ENABLE(SANDBOX_EXTENSIONS)
48 if (!captureDevice.isMockDevice()) {
49 m_sourcesNeedingSandboxExtension.add(identifier);
50 if (m_sourcesNeedingSandboxExtension.size() == 1) {
51 SandboxExtension::Handle handleForTCCD;
52 SandboxExtension::createHandleForMachLookup("com.apple.tccd"_s, m_connection->getAuditToken(), handleForTCCD);
53 SandboxExtension::Handle handleForMicrophone;
54 SandboxExtension::createHandleForGenericExtension("com.apple.webkit.microphone"_s, handleForMicrophone);
55 send(Messages::SpeechRecognitionRealtimeMediaSourceManager::GrantSandboxExtensions(handleForTCCD, handleForMicrophone));
56 }
57 }
58#endif
59
60 send(Messages::SpeechRecognitionRealtimeMediaSourceManager::CreateSource(identifier, captureDevice));
61}
62
63void SpeechRecognitionRemoteRealtimeMediaSourceManager::removeSource(SpeechRecognitionRemoteRealtimeMediaSource& source)
64{
65 auto identifier = source.identifier();
66 ASSERT(m_sources.get(identifier) == &source);
67 m_sources.remove(identifier);
68
69#if ENABLE(SANDBOX_EXTENSIONS)
70 if (m_sourcesNeedingSandboxExtension.remove(identifier)) {
71 if (m_sourcesNeedingSandboxExtension.isEmpty())
72 send(Messages::SpeechRecognitionRealtimeMediaSourceManager::RevokeSandboxExtensions());
73 }
74#endif
75
76 send(Messages::SpeechRecognitionRealtimeMediaSourceManager::DeleteSource(identifier));
77}
