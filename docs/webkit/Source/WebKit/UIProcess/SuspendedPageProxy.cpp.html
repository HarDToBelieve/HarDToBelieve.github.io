47static HashSet<SuspendedPageProxy*>& allSuspendedPages()
48{
49 static NeverDestroyed<HashSet<SuspendedPageProxy*>> map;
50 return map;

53RefPtr<WebProcessProxy> SuspendedPageProxy::findReusableSuspendedPageProcess(WebProcessPool& processPool, const RegistrableDomain& registrableDomain, WebsiteDataStore& dataStore)
54{
55 for (auto* suspendedPage : allSuspendedPages()) {
56 auto& process = suspendedPage->process();
57 if (&process.processPool() == &processPool && process.registrableDomain() == registrableDomain && &process.websiteDataStore() == &dataStore)
58 return &process;
59 }
60 return nullptr;
61}

97SuspendedPageProxy::SuspendedPageProxy(WebPageProxy& page, Ref<WebProcessProxy>&& process, FrameIdentifier mainFrameID, ShouldDelayClosingUntilFirstLayerFlush shouldDelayClosingUntilFirstLayerFlush)
98 : m_page(page)
99 , m_webPageID(page.webPageID())
100 , m_process(WTFMove(process))
101 , m_mainFrameID(mainFrameID)
102 , m_shouldDelayClosingUntilFirstLayerFlush(shouldDelayClosingUntilFirstLayerFlush)
103 , m_suspensionTimeoutTimer(RunLoop::main(), this, &SuspendedPageProxy::suspensionTimedOut)
104#if PLATFORM(IOS_FAMILY)
105 , m_suspensionActivity(m_process->throttler().backgroundActivity("Page suspension for back/forward cache"_s).moveToUniquePtr())
106#endif
107#if HAVE(VISIBILITY_PROPAGATION_VIEW)
108 , m_contextIDForVisibilityPropagationInWebProcess(page.contextIDForVisibilityPropagationInWebProcess())
109#if ENABLE(GPU_PROCESS)
110 , m_contextIDForVisibilityPropagationInGPUProcess(page.contextIDForVisibilityPropagationInGPUProcess())
111#endif
112#endif
113{
114 allSuspendedPages().add(this);
115 m_process->incrementSuspendedPageCount();
116 m_process->addMessageReceiver(Messages::WebPageProxy::messageReceiverName(), m_webPageID, *this);
117
118 m_suspensionTimeoutTimer.startOneShot(suspensionTimeout);
119 send(Messages::WebPage::SetIsSuspended(true));
120}
121
122SuspendedPageProxy::~SuspendedPageProxy()
123{
124 allSuspendedPages().remove(this);
125
126 if (m_readyToUnsuspendHandler) {
127 RunLoop::main().dispatch([readyToUnsuspendHandler = WTFMove(m_readyToUnsuspendHandler)]() mutable {
128 readyToUnsuspendHandler(nullptr);
129 });
130 }
131
132 if (m_suspensionState != SuspensionState::Resumed) {
133 // If the suspended page was not consumed before getting destroyed, then close the corresponding page
134 // on the WebProcess side.
135 close();
136
137 if (m_suspensionState == SuspensionState::Suspending)
138 m_process->removeMessageReceiver(Messages::WebPageProxy::messageReceiverName(), m_webPageID);
139 }
140
141 m_process->decrementSuspendedPageCount();
142}

149void SuspendedPageProxy::waitUntilReadyToUnsuspend(CompletionHandler<void(SuspendedPageProxy*)>&& completionHandler)
150{
151 if (m_readyToUnsuspendHandler)
152 m_readyToUnsuspendHandler(nullptr);
153
154 switch (m_suspensionState) {
155 case SuspensionState::Suspending:
156 m_readyToUnsuspendHandler = WTFMove(completionHandler);
157 break;
158 case SuspensionState::FailedToSuspend:
159 case SuspensionState::Suspended:
160 completionHandler(this);
161 break;
162 case SuspensionState::Resumed:
163 ASSERT_NOT_REACHED();
164 completionHandler(nullptr);
165 break;
166 }
167}
168
169void SuspendedPageProxy::unsuspend()
170{
171 ASSERT(m_suspensionState == SuspensionState::Suspended);
172
173 m_suspensionState = SuspensionState::Resumed;
174 send(Messages::WebPage::SetIsSuspended(false));
175}
176
177void SuspendedPageProxy::close()
178{
179 ASSERT(m_suspensionState != SuspensionState::Resumed);
180
181 if (m_isClosed)
182 return;
183
184 RELEASE_LOG(ProcessSwapping, "%p - SuspendedPageProxy::close()", this);
185 m_isClosed = true;
186 send(Messages::WebPage::Close());
187}

189void SuspendedPageProxy::pageDidFirstLayerFlush()
190{
191 m_shouldDelayClosingUntilFirstLayerFlush = ShouldDelayClosingUntilFirstLayerFlush::No;
192
193 if (m_shouldCloseWhenEnteringAcceleratedCompositingMode) {
194 // We needed the suspended page to stay alive to avoid flashing. Now we can get rid of it.
195 close();
196 }
197}
198
199bool SuspendedPageProxy::pageIsClosedOrClosing() const
200{
201 return m_isClosed || m_shouldCloseWhenEnteringAcceleratedCompositingMode;
202}

204void SuspendedPageProxy::closeWithoutFlashing()
205{
206 RELEASE_LOG(ProcessSwapping, "%p - SuspendedPageProxy::closeWithoutFlashing() shouldDelayClosingUntilFirstLayerFlush? %d", this, m_shouldDelayClosingUntilFirstLayerFlush == ShouldDelayClosingUntilFirstLayerFlush::Yes);
207 if (m_shouldDelayClosingUntilFirstLayerFlush == ShouldDelayClosingUntilFirstLayerFlush::Yes) {
208 m_shouldCloseWhenEnteringAcceleratedCompositingMode = true;
209 return;
210 }
211 close();
212}

214void SuspendedPageProxy::didProcessRequestToSuspend(SuspensionState newSuspensionState)
215{
216 LOG(ProcessSwapping, "SuspendedPageProxy %s from process %i finished transition to suspended", loggingString(), m_process->processIdentifier());
217 RELEASE_LOG(ProcessSwapping, "%p - SuspendedPageProxy::didProcessRequestToSuspend() success? %d", this, newSuspensionState == SuspensionState::Suspended);
218
219 ASSERT(m_suspensionState == SuspensionState::Suspending);
220 ASSERT(newSuspensionState == SuspensionState::Suspended || newSuspensionState == SuspensionState::FailedToSuspend);
221
222 m_suspensionState = newSuspensionState;
223
224 m_suspensionTimeoutTimer.stop();
225
226#if PLATFORM(IOS_FAMILY)
227 m_suspensionActivity = nullptr;
228#endif
229
230 m_process->removeMessageReceiver(Messages::WebPageProxy::messageReceiverName(), m_webPageID);
231
232 if (m_suspensionState == SuspensionState::FailedToSuspend)
233 closeWithoutFlashing();
234
235 if (m_readyToUnsuspendHandler)
236 m_readyToUnsuspendHandler(this);
237}

239void SuspendedPageProxy::suspensionTimedOut()
240{
241 RELEASE_LOG_ERROR(ProcessSwapping, "%p - SuspendedPageProxy::suspensionTimedOut() destroying the suspended page because it failed to suspend in time", this);
242 backForwardCache().removeEntry(*this); // Will destroy |this|.
243}

245void SuspendedPageProxy::didReceiveMessage(IPC::Connection&, IPC::Decoder& decoder)
246{
247 ASSERT(decoder.messageReceiverName() == Messages::WebPageProxy::messageReceiverName());
248
249 if (decoder.messageName() == Messages::WebPageProxy::DidSuspendAfterProcessSwap::name()) {
250 didProcessRequestToSuspend(SuspensionState::Suspended);
251 return;
252 }
253
254 if (decoder.messageName() == Messages::WebPageProxy::DidFailToSuspendAfterProcessSwap::name()) {
255 didProcessRequestToSuspend(SuspensionState::FailedToSuspend);
256 return;
257 }
258
259#if !LOG_DISABLED
260 if (!messageNamesToIgnoreWhileSuspended().contains(decoder.messageName()))
261 LOG(ProcessSwapping, "SuspendedPageProxy received unexpected WebPageProxy message '%s'", description(decoder.messageName()));
262#endif
263}

280bool SuspendedPageProxy::sendMessage(UniqueRef<IPC::Encoder>&& encoder, OptionSet<IPC::SendOption> sendOptions, std::optional<std::pair<CompletionHandler<void(IPC::Decoder*)>, uint64_t>>&& asyncReplyInfo)
281{
282 // Send messages via the WebProcessProxy instead of the IPC::Connection since AuxiliaryProcessProxy implements queueing of messages
283 // while the process is still launching.
284 return m_process->sendMessage(WTFMove(encoder), sendOptions, WTFMove(asyncReplyInfo));
