74WebUserContentControllerProxy::~WebUserContentControllerProxy()
75{
76 for (const auto& identifier : m_associatedContentWorlds) {
77 auto* world = API::ContentWorld::worldForIdentifier(identifier);
78 RELEASE_ASSERT(world);
79 world->userContentControllerProxyDestroyed(*this);
80 }
81
82 webUserContentControllerProxies().remove(m_identifier);
83 for (auto& process : m_processes) {
84 process.removeMessageReceiver(Messages::WebUserContentControllerProxy::messageReceiverName(), identifier());
85 process.didDestroyWebUserContentControllerProxy(*this);
86 }
87#if ENABLE(CONTENT_EXTENSIONS)
88 for (auto& process : m_networkProcesses)
89 process.didDestroyWebUserContentControllerProxy(*this);
90#endif
91}

105void WebUserContentControllerProxy::addProcess(WebProcessProxy& webProcessProxy)
106{
107 ASSERT(!m_processes.hasNullReferences());
108
109 if (m_processes.add(webProcessProxy).isNewEntry)
110 webProcessProxy.addMessageReceiver(Messages::WebUserContentControllerProxy::messageReceiverName(), identifier(), *this);

113UserContentControllerParameters WebUserContentControllerProxy::parameters() const
114{
115 UserContentControllerParameters parameters;
116
117 parameters.identifier = identifier();
118
119 ASSERT(parameters.userContentWorlds.isEmpty());
120 for (const auto& identifier : m_associatedContentWorlds) {
121 auto* world = API::ContentWorld::worldForIdentifier(identifier);
122 RELEASE_ASSERT(world);
123 parameters.userContentWorlds.append(world->worldData());
124 }
125
126 for (auto userScript : m_userScripts->elementsOfType<API::UserScript>())
127 parameters.userScripts.append({ userScript->identifier(), userScript->contentWorld().identifier(), userScript->userScript() });
128
129 for (auto userStyleSheet : m_userStyleSheets->elementsOfType<API::UserStyleSheet>())
130 parameters.userStyleSheets.append({ userStyleSheet->identifier(), userStyleSheet->contentWorld().identifier(), userStyleSheet->userStyleSheet() });
131
132 for (auto& handler : m_scriptMessageHandlers.values())
133 parameters.messageHandlers.append({ handler->identifier(), handler->world().identifier(), handler->name() });
134
135#if ENABLE(CONTENT_EXTENSIONS)
136 parameters.contentRuleLists = contentRuleListData();
137#endif
138
139 return parameters;
140}

162void WebUserContentControllerProxy::addContentWorld(API::ContentWorld& world)
163{
164 if (world.identifier() == pageContentWorldIdentifier())
165 return;
166
167 auto addResult = m_associatedContentWorlds.add(world.identifier());
168 if (!addResult.isNewEntry)
169 return;
170
171 world.addAssociatedUserContentControllerProxy(*this);
172 for (auto& process : m_processes)
173 process.send(Messages::WebUserContentController::AddContentWorlds({ world.worldData() }), identifier());
174}

176void WebUserContentControllerProxy::contentWorldDestroyed(API::ContentWorld& world)
177{
178 bool result = m_associatedContentWorlds.remove(world.identifier());
179 ASSERT_UNUSED(result, result);
180
181 for (auto& process : m_processes)
182 process.send(Messages::WebUserContentController::RemoveContentWorlds({ world.identifier() }), identifier());
183}

217void WebUserContentControllerProxy::removeAllUserScripts()
218{
219 HashCountedSet<RefPtr<API::ContentWorld>> worlds;
220 for (auto userScript : m_userScripts->elementsOfType<API::UserScript>())
221 worlds.add(const_cast<API::ContentWorld*>(&userScript->contentWorld()));
222
223 Vector<ContentWorldIdentifier> worldIdentifiers;
224 worldIdentifiers.reserveInitialCapacity(worlds.size());
225 for (const auto& worldCountPair : worlds)
226 worldIdentifiers.uncheckedAppend(worldCountPair.key->identifier());
227
228 for (auto& process : m_processes)
229 process.send(Messages::WebUserContentController::RemoveAllUserScripts(worldIdentifiers), identifier());
230
231 m_userScripts->elements().clear();
232}

283bool WebUserContentControllerProxy::addUserScriptMessageHandler(WebScriptMessageHandler& handler)
284{
285 auto& world = handler.world();
286
287 for (auto& existingHandler : m_scriptMessageHandlers.values()) {
288 if (existingHandler->name() == handler.name() && existingHandler->world().identifier() == world.identifier())
289 return false;
290 }
291
292 addContentWorld(world);
293
294 m_scriptMessageHandlers.add(handler.identifier(), &handler);
295
296 for (auto& process : m_processes)
297 process.send(Messages::WebUserContentController::AddUserScriptMessageHandlers({ { handler.identifier(), world.identifier(), handler.name() } }), identifier());
298
299 return true;
300}

302void WebUserContentControllerProxy::removeUserMessageHandlerForName(const String& name, API::ContentWorld& world)
303{
304 for (auto it = m_scriptMessageHandlers.begin(), end = m_scriptMessageHandlers.end(); it != end; ++it) {
305 if (it->value->name() == name && it->value->world().identifier() == world.identifier()) {
306 for (auto& process : m_processes)
307 process.send(Messages::WebUserContentController::RemoveUserScriptMessageHandler(world.identifier(), it->value->identifier()), identifier());
308
309 m_scriptMessageHandlers.remove(it);
310
311 return;
312 }
313 }
314}

316void WebUserContentControllerProxy::removeAllUserMessageHandlers(API::ContentWorld& world)
317{
318 for (auto& process : m_processes)
319 process.send(Messages::WebUserContentController::RemoveAllUserScriptMessageHandlersForWorlds({ world.identifier() }), identifier());
320
321 unsigned numberRemoved = 0;
322 m_scriptMessageHandlers.removeIf([&](auto& entry) {
323 if (entry.value->world().identifier() == world.identifier()) {
324 ++numberRemoved;
325 return true;
326 }
327 return false;
328 });
329}

337void WebUserContentControllerProxy::didPostMessage(WebPageProxyIdentifier pageProxyID, FrameInfoData&& frameInfoData, uint64_t messageHandlerID, const IPC::DataReference& dataReference, Messages::WebUserContentControllerProxy::DidPostMessage::AsyncReply&& reply)
338{
339 WebPageProxy* page = WebProcessProxy::webPage(pageProxyID);
340 if (!page)
341 return;
342
343 if (!HashMap<uint64_t, RefPtr<WebScriptMessageHandler>>::isValidKey(messageHandlerID))
344 return;
345
346 RefPtr<WebScriptMessageHandler> handler = m_scriptMessageHandlers.get(messageHandlerID);
347 if (!handler)
348 return;
349
350 if (!handler->client().supportsAsyncReply()) {
351 handler->client().didPostMessage(*page, WTFMove(frameInfoData), handler->world(), WebCore::SerializedScriptValue::adopt(dataReference.vector()));
352 reply({ }, { });
353 return;
354 }
355
356 handler->client().didPostMessageWithAsyncReply(*page, WTFMove(frameInfoData), handler->world(), WebCore::SerializedScriptValue::adopt(dataReference.vector()), [reply = WTFMove(reply)](API::SerializedScriptValue* value, const String& errorMessage) mutable {
357 if (errorMessage.isNull()) {
358 ASSERT(value);
359 reply({ value->internalRepresentation().toWireBytes() }, { });
360 return;
361 }
362
363 reply({ }, errorMessage);
364 });
365}
