33namespace WebKit {
34using namespace WebCore;
35
36UserMediaPermissionCheckProxy::UserMediaPermissionCheckProxy(FrameIdentifier frameID, CompletionHandler&& handler, Ref<WebCore::SecurityOrigin>&& userMediaDocumentOrigin, Ref<WebCore::SecurityOrigin>&& topLevelDocumentOrigin)
37 : m_frameID(frameID)
38 , m_completionHandler(WTFMove(handler))
39 , m_userMediaDocumentSecurityOrigin(WTFMove(userMediaDocumentOrigin))
40 , m_topLevelDocumentSecurityOrigin(WTFMove(topLevelDocumentOrigin))
41{
42}
43
44UserMediaPermissionCheckProxy::~UserMediaPermissionCheckProxy()
45{
46 invalidate();
47}
48
49void UserMediaPermissionCheckProxy::setUserMediaAccessInfo(bool allowed)
50{
51 ASSERT(m_completionHandler);
52 complete(allowed ? PermissionInfo::Granted : PermissionInfo::Unknown);
53}
54
55void UserMediaPermissionCheckProxy::complete(PermissionInfo info)
56{
57 if (!m_completionHandler)
58 return;
59
60 auto completionHandler = WTFMove(m_completionHandler);
61 completionHandler(info);
62}
