34namespace WebKit {
35
36class UserMediaPermissionRequestManagerProxy;
37
38class UserMediaPermissionRequestProxy : public API::ObjectImpl<API::Object::Type::UserMediaPermissionRequest> {
39public:
40 static Ref<UserMediaPermissionRequestProxy> create(UserMediaPermissionRequestManagerProxy& manager, uint64_t userMediaID, WebCore::FrameIdentifier mainFrameID, WebCore::FrameIdentifier frameID, Ref<WebCore::SecurityOrigin>&& userMediaDocumentOrigin, Ref<WebCore::SecurityOrigin>&& topLevelDocumentOrigin, Vector<WebCore::CaptureDevice>&& audioDevices, Vector<WebCore::CaptureDevice>&& videoDevices, WebCore::MediaStreamRequest&& request, CompletionHandler<void(bool)>&& decisionCompletionHandler = { })
41 {
42 return adoptRef(*new UserMediaPermissionRequestProxy(manager, userMediaID, mainFrameID, frameID, WTFMove(userMediaDocumentOrigin), WTFMove(topLevelDocumentOrigin), WTFMove(audioDevices), WTFMove(videoDevices), WTFMove(request), WTFMove(decisionCompletionHandler)));
43 }
44
45 void allow(const String& audioDeviceUID, const String& videoDeviceUID);
46 void allow();
47 void prompt();
48 void doDefaultAction() { prompt(); }
49 enum class UserMediaAccessDenialReason { NoConstraints, UserMediaDisabled, NoCaptureDevices, InvalidConstraint, HardwareError, PermissionDenied, OtherFailure };
50 void deny(UserMediaAccessDenialReason = UserMediaAccessDenialReason::UserMediaDisabled);
51
52 void invalidate();
53 bool isPending() const { return m_manager; }
54
55 bool requiresAudioCapture() const { return m_eligibleAudioDevices.size(); }
56 bool requiresVideoCapture() const { return !requiresDisplayCapture() && m_eligibleVideoDevices.size(); }
57 bool requiresDisplayCapture() const { return m_request.type == WebCore::MediaStreamRequest::Type::DisplayMedia && m_eligibleVideoDevices.size(); }
58
59 void setEligibleVideoDeviceUIDs(Vector<WebCore::CaptureDevice>&& devices) { m_eligibleVideoDevices = WTFMove(devices); }
60 void setEligibleAudioDeviceUIDs(Vector<WebCore::CaptureDevice>&& devices) { m_eligibleAudioDevices = WTFMove(devices); }
61
62 Vector<String> videoDeviceUIDs() const;
63 Vector<String> audioDeviceUIDs() const;
64 bool hasAudioDevice() const { return !m_eligibleAudioDevices.isEmpty(); }
65 bool hasVideoDevice() const { return !m_eligibleVideoDevices.isEmpty(); }
66
67 bool hasPersistentAccess() const { return m_hasPersistentAccess; }
68 void setHasPersistentAccess() { m_hasPersistentAccess = true; }
69
70 uint64_t userMediaID() const { return m_userMediaID; }
71 WebCore::FrameIdentifier mainFrameID() const { return m_mainFrameID; }
72 WebCore::FrameIdentifier frameID() const { return m_frameID; }
73
74 WebCore::SecurityOrigin& topLevelDocumentSecurityOrigin() { return m_topLevelDocumentSecurityOrigin.get(); }
75 WebCore::SecurityOrigin& userMediaDocumentSecurityOrigin() { return m_userMediaDocumentSecurityOrigin.get(); }
76 const WebCore::SecurityOrigin& topLevelDocumentSecurityOrigin() const { return m_topLevelDocumentSecurityOrigin.get(); }
77 const WebCore::SecurityOrigin& userMediaDocumentSecurityOrigin() const { return m_userMediaDocumentSecurityOrigin.get(); }
78
79 const WebCore::MediaStreamRequest& userRequest() const { return m_request; }
80
81 WebCore::MediaStreamRequest::Type requestType() const { return m_request.type; }
82
83 void setDeviceIdentifierHashSalt(String&& salt) { m_deviceIdentifierHashSalt = WTFMove(salt); }
84 const String& deviceIdentifierHashSalt() const { return m_deviceIdentifierHashSalt; }
85
86 WebCore::CaptureDevice audioDevice() const { return m_eligibleAudioDevices.isEmpty() ? WebCore::CaptureDevice { } : m_eligibleAudioDevices[0]; }
87 WebCore::CaptureDevice videoDevice() const { return m_eligibleVideoDevices.isEmpty() ? WebCore::CaptureDevice { } : m_eligibleVideoDevices[0]; }
88
89#if ENABLE(MEDIA_STREAM)
90 bool isUserGesturePriviledged() const { return m_request.isUserGesturePriviledged; }
91#endif
92
93 CompletionHandler<void(bool)> decisionCompletionHandler() { return std::exchange(m_decisionCompletionHandler, { }); }
94
95private:
96 UserMediaPermissionRequestProxy(UserMediaPermissionRequestManagerProxy&, uint64_t userMediaID, WebCore::FrameIdentifier mainFrameID, WebCore::FrameIdentifier, Ref<WebCore::SecurityOrigin>&& userMediaDocumentOrigin, Ref<WebCore::SecurityOrigin>&& topLevelDocumentOrigin, Vector<WebCore::CaptureDevice>&& audioDevices, Vector<WebCore::CaptureDevice>&& videoDevices, WebCore::MediaStreamRequest&&, CompletionHandler<void(bool)>&&);
97
98 UserMediaPermissionRequestManagerProxy* m_manager;
99 uint64_t m_userMediaID;
100 WebCore::FrameIdentifier m_mainFrameID;
101 WebCore::FrameIdentifier m_frameID;
102 Ref<WebCore::SecurityOrigin> m_userMediaDocumentSecurityOrigin;
103 Ref<WebCore::SecurityOrigin> m_topLevelDocumentSecurityOrigin;
104 Vector<WebCore::CaptureDevice> m_eligibleVideoDevices;
105 Vector<WebCore::CaptureDevice> m_eligibleAudioDevices;
106 WebCore::MediaStreamRequest m_request;
107 bool m_hasPersistentAccess { false };
108 String m_deviceIdentifierHashSalt;
109 CompletionHandler<void(bool)> m_decisionCompletionHandler;
110};
