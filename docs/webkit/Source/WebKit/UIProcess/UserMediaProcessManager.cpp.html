66bool UserMediaProcessManager::willCreateMediaStream(UserMediaPermissionRequestManagerProxy& proxy, bool withAudio, bool withVideo)
67{
68 ASSERT(withAudio || withVideo);
69
70 if (m_denyNextRequest) {
71 m_denyNextRequest = false;
72 return false;
73 }
74
75#if ENABLE(SANDBOX_EXTENSIONS) && USE(APPLE_INTERNAL_SDK)
76 auto& process = proxy.page().process();
77 size_t extensionCount = 0;
78
79 bool needsAudioSandboxExtension = withAudio && !process.hasAudioCaptureExtension() && !proxy.page().preferences().captureAudioInUIProcessEnabled() && !proxy.page().preferences().captureAudioInGPUProcessEnabled();
80 if (needsAudioSandboxExtension)
81 extensionCount++;
82
83 bool needsVideoSandboxExtension = withVideo && !process.hasVideoCaptureExtension() && !proxy.page().preferences().captureVideoInUIProcessEnabled() && !proxy.page().preferences().captureVideoInGPUProcessEnabled();
84 if (needsVideoSandboxExtension)
85 extensionCount++;
86
87 bool needsAppleCameraSandboxExtension = needsVideoSandboxExtension && needsAppleCameraService();
88 if (needsAppleCameraSandboxExtension) {
89 extensionCount++;
90#if HAVE(ADDITIONAL_APPLE_CAMERA_SERVICE)
91 extensionCount++;
92#endif
93 }
94
95 if (extensionCount) {
96 SandboxExtension::HandleArray handles;
97 Vector<String> ids;
98
99 if (!proxy.page().preferences().mockCaptureDevicesEnabled()) {
100 handles.allocate(extensionCount);
101 ids.reserveInitialCapacity(extensionCount);
102
103 if (needsAudioSandboxExtension && SandboxExtension::createHandleForGenericExtension(audioExtensionPath, handles[--extensionCount]))
104 ids.uncheckedAppend(audioExtensionPath);
105
106 if (needsVideoSandboxExtension && SandboxExtension::createHandleForGenericExtension(videoExtensionPath, handles[--extensionCount]))
107 ids.uncheckedAppend(videoExtensionPath);
108
109 if (needsAppleCameraSandboxExtension) {
110 if (SandboxExtension::createHandleForMachLookup(appleCameraServicePath, std::nullopt, handles[--extensionCount]))
111 ids.uncheckedAppend(appleCameraServicePath);
112#if HAVE(ADDITIONAL_APPLE_CAMERA_SERVICE)
113 if (SandboxExtension::createHandleForMachLookup(additionalAppleCameraServicePath, std::nullopt, handles[--extensionCount]))
114 ids.uncheckedAppend(additionalAppleCameraServicePath);
115#endif
116 }
117
118 if (ids.size() != handles.size()) {
119 WTFLogAlways("Could not create a required sandbox extension, capture will fail!");
120 return false;
121 }
122 }
123
124 for (const auto& id : ids)
125 RELEASE_LOG(WebRTC, "UserMediaProcessManager::willCreateMediaStream - granting extension %s", id.utf8().data());
126
127 if (needsAudioSandboxExtension)
128 process.grantAudioCaptureExtension();
129 if (needsVideoSandboxExtension)
130 process.grantVideoCaptureExtension();
131 process.send(Messages::WebProcess::GrantUserMediaDeviceSandboxExtensions(MediaDeviceSandboxExtensions(ids, WTFMove(handles))), 0);
132 }
133#else
134 UNUSED_PARAM(proxy);
135 UNUSED_PARAM(withAudio);
136 UNUSED_PARAM(withVideo);
137#endif
138
139 proxy.page().activateMediaStreamCaptureInPage();
140
141 return true;
142}

144void UserMediaProcessManager::revokeSandboxExtensionsIfNeeded(WebProcessProxy& process)
145{
146#if ENABLE(SANDBOX_EXTENSIONS)
147 bool hasAudioCapture = false;
148 bool hasVideoCapture = false;
149 bool hasPendingCapture = false;
150
151 UserMediaPermissionRequestManagerProxy::forEach([&hasAudioCapture, &hasVideoCapture, &hasPendingCapture, &process](auto& managerProxy) {
152 if (&process != &managerProxy.page().process())
153 return;
154 hasAudioCapture |= managerProxy.page().isCapturingAudio();
155 hasVideoCapture |= managerProxy.page().isCapturingVideo();
156 hasPendingCapture |= managerProxy.hasPendingCapture();
157 });
158
159 if (hasPendingCapture)
160 return;
161
162 if (hasAudioCapture && hasVideoCapture)
163 return;
164
165 Vector<String> params;
166 if (!hasAudioCapture && process.hasAudioCaptureExtension()) {
167 params.append(audioExtensionPath);
168 process.revokeAudioCaptureExtension();
169 }
170 if (!hasVideoCapture && process.hasVideoCaptureExtension()) {
171 params.append(videoExtensionPath);
172 if (needsAppleCameraService()) {
173 params.append(appleCameraServicePath);
174#if USE(APPLE_INTERNAL_SDK) && HAVE(ADDITIONAL_APPLE_CAMERA_SERVICE)
175 params.append(additionalAppleCameraServicePath);
176#endif
177 }
178 process.revokeVideoCaptureExtension();
179 }
180
181 if (params.isEmpty())
182 return;
183
184 for (const auto& id : params)
185 RELEASE_LOG(WebRTC, "UserMediaProcessManager::endedCaptureSession - revoking extension %s", id.utf8().data());
186
187 process.send(Messages::WebProcess::RevokeUserMediaDeviceSandboxExtensions(params), 0);
188#endif
189}

191void UserMediaProcessManager::setCaptureEnabled(bool enabled)
192{
193 if (enabled == m_captureEnabled)
194 return;
195
196 m_captureEnabled = enabled;
197
198 if (enabled)
199 return;
200
201 UserMediaPermissionRequestManagerProxy::forEach([](auto& proxy) {
202 proxy.stopCapture();
203 });
204}

206void UserMediaProcessManager::captureDevicesChanged()

209 proxy.captureDevicesChanged();

213void UserMediaProcessManager::updateCaptureDevices(ShouldNotify shouldNotify)
214{
215 WebCore::RealtimeMediaSourceCenter::singleton().getMediaStreamDevices([weakThis = makeWeakPtr(*this), this, shouldNotify](auto&& newDevices) mutable {
216 if (!weakThis)
217 return;
218
219 if (!haveDevicesChanged(m_captureDevices, newDevices))
220 return;
221
222 m_captureDevices = WTFMove(newDevices);
223 if (shouldNotify == ShouldNotify::Yes)
224 captureDevicesChanged();
225 });
226}
227
228void UserMediaProcessManager::devicesChanged()
229{
230 updateCaptureDevices(ShouldNotify::Yes);
231}
232
233void UserMediaProcessManager::beginMonitoringCaptureDevices()
234{
235 static std::once_flag onceFlag;
236
237 std::call_once(onceFlag, [this] {
238 updateCaptureDevices(ShouldNotify::No);
239 WebCore::RealtimeMediaSourceCenter::singleton().addDevicesChangedObserver(*this);
240 });
241}
