30#include "WebAuthenticationFlags.h"
31#include "WebAuthenticationRequestData.h"
32#include <WebCore/AuthenticatorResponse.h>
33#include <WebCore/ExceptionData.h>
34#include <wtf/Forward.h>
35#include <wtf/RefCounted.h>
36#include <wtf/WeakPtr.h>
37#include <wtf/spi/cocoa/SecuritySPI.h>
38
39OBJC_CLASS LAContext;
40
41namespace WebCore {
42class AuthenticatorAssertionResponse;
43}
44
45namespace WebKit {
46
47class Authenticator : public RefCounted<Authenticator>, public CanMakeWeakPtr<Authenticator> {
48public:
49 using Respond = Variant<Ref<WebCore::AuthenticatorResponse>, WebCore::ExceptionData>;
50
51 class Observer : public CanMakeWeakPtr<Observer> {
52 public:
53 virtual ~Observer() = default;
54 virtual void respondReceived(Respond&&) = 0;
55 virtual void downgrade(Authenticator* id, Ref<Authenticator>&& downgradedAuthenticator) = 0;
56 virtual void authenticatorStatusUpdated(WebAuthenticationStatus) = 0;
57 virtual void requestPin(uint64_t retries, CompletionHandler<void(const WTF::String&)>&&) = 0;
58 virtual void selectAssertionResponse(Vector<Ref<WebCore::AuthenticatorAssertionResponse>>&&, WebAuthenticationSource, CompletionHandler<void(WebCore::AuthenticatorAssertionResponse*)>&&) = 0;
59 virtual void decidePolicyForLocalAuthenticator(CompletionHandler<void(LocalAuthenticatorPolicy)>&&) = 0;
60 virtual void requestLAContextForUserVerification(CompletionHandler<void(LAContext *)>&&) = 0;
61 virtual void cancelRequest() = 0;
62 };
63
64 virtual ~Authenticator() = default;
65
66 void setObserver(Observer& observer) { m_observer = makeWeakPtr(observer); }
67
68 // This operation is guaranteed to execute asynchronously.
69 void handleRequest(const WebAuthenticationRequestData&);
70
71 void setWebAuthenticationModernEnabled(bool webAuthenticationModernEnabled) { m_webAuthenticationModernEnabled = webAuthenticationModernEnabled; }
72
73protected:
74 Authenticator() = default;
75
76 Observer* observer() const { return m_observer.get(); }
77 const WebAuthenticationRequestData& requestData() const { return m_pendingRequestData; }
78 bool webAuthenticationModernEnabled() const { return m_webAuthenticationModernEnabled; }
79
80 void receiveRespond(Respond&&) const;
81
82private:
83 virtual void makeCredential() = 0;
84 virtual void getAssertion() = 0;
85
86 WeakPtr<Observer> m_observer;
87 WebAuthenticationRequestData m_pendingRequestData;
88 bool m_webAuthenticationModernEnabled { false };
89};
90
91} // namespace WebKit
92
93#endif // ENABLE(WEB_AUTHN)
