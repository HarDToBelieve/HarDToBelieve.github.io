55// FIXME(188625): Support BLE authenticators.
56static AuthenticatorManager::TransportSet collectTransports(const std::optional<PublicKeyCredentialCreationOptions::AuthenticatorSelectionCriteria>& authenticatorSelection)
57{
58 AuthenticatorManager::TransportSet result;
59 if (!authenticatorSelection || !authenticatorSelection->authenticatorAttachment) {
60 auto addResult = result.add(AuthenticatorTransport::Internal);
61 ASSERT_UNUSED(addResult, addResult.isNewEntry);
62 addResult = result.add(AuthenticatorTransport::Usb);
63 ASSERT_UNUSED(addResult, addResult.isNewEntry);
64 addResult = result.add(AuthenticatorTransport::Nfc);
65 ASSERT_UNUSED(addResult, addResult.isNewEntry);
66 return result;
67 }
68
69 if (authenticatorSelection->authenticatorAttachment == AuthenticatorAttachment::Platform) {
70 auto addResult = result.add(AuthenticatorTransport::Internal);
71 ASSERT_UNUSED(addResult, addResult.isNewEntry);
72 return result;
73 }
74 if (authenticatorSelection->authenticatorAttachment == AuthenticatorAttachment::CrossPlatform) {
75 auto addResult = result.add(AuthenticatorTransport::Usb);
76 ASSERT_UNUSED(addResult, addResult.isNewEntry);
77 addResult = result.add(AuthenticatorTransport::Nfc);
78 ASSERT_UNUSED(addResult, addResult.isNewEntry);
79 return result;
80 }
81
82 ASSERT_NOT_REACHED();
83 return result;
84}

86// FIXME(188625): Support BLE authenticators.
87// The goal is to find a union of different transports from allowCredentials.
88// If it is not specified or any of its credentials doesn't specify its own. We should discover all.
89// This is a variant of Step. 18.*.4 from https://www.w3.org/TR/webauthn/#discover-from-external-source
90// as of 7 August 2018.
91static AuthenticatorManager::TransportSet collectTransports(const Vector<PublicKeyCredentialDescriptor>& allowCredentials, const std::optional<AuthenticatorAttachment>& authenticatorAttachment)
92{
93 AuthenticatorManager::TransportSet result;
94 if (allowCredentials.isEmpty()) {
95 auto addResult = result.add(AuthenticatorTransport::Internal);
96 ASSERT_UNUSED(addResult, addResult.isNewEntry);
97 addResult = result.add(AuthenticatorTransport::Usb);
98 ASSERT_UNUSED(addResult, addResult.isNewEntry);
99 addResult = result.add(AuthenticatorTransport::Nfc);
100 ASSERT_UNUSED(addResult, addResult.isNewEntry);
101 }
102
103 for (auto& allowCredential : allowCredentials) {
104 if (allowCredential.transports.isEmpty()) {
105 result.add(AuthenticatorTransport::Internal);
106 result.add(AuthenticatorTransport::Usb);
107 result.add(AuthenticatorTransport::Nfc);
108
109 break;
110 }
111
112 for (const auto& transport : allowCredential.transports) {
113 if (transport == AuthenticatorTransport::Ble)
114 continue;
115
116 result.add(transport);
117
118 if (result.size() >= AuthenticatorManager::maxTransportNumber)
119 break;
120 }
121 }
122
123 if (authenticatorAttachment) {
124 if (authenticatorAttachment == AuthenticatorAttachment::Platform) {
125 result.remove(AuthenticatorTransport::Usb);
126 result.remove(AuthenticatorTransport::Nfc);
127 }
128
129 if (authenticatorAttachment == AuthenticatorAttachment::CrossPlatform)
130 result.remove(AuthenticatorTransport::Internal);
131 }
132
133 ASSERT(result.size() <= AuthenticatorManager::maxTransportNumber);
134 return result;
135}

137// Only roaming authenticators are supported for Google legacy AppID support.
138static void processGoogleLegacyAppIdSupportExtension(const std::optional<AuthenticationExtensionsClientInputs>& extensions, AuthenticatorManager::TransportSet& transports)
139{
140 if (!extensions) {
141 // AuthenticatorCoordinator::create should always set it.
142 ASSERT_NOT_REACHED();
143 return;
144 }
145 if (!extensions->googleLegacyAppidSupport)
146 return;
147 transports.remove(AuthenticatorTransport::Internal);
148}

173void AuthenticatorManager::handleRequest(WebAuthenticationRequestData&& data, Callback&& callback)
174{
175 if (m_pendingCompletionHandler) {
176 invokePendingCompletionHandler(ExceptionData { NotAllowedError, "This request has been cancelled by a new request."_s });
177 m_requestTimeOutTimer.stop();
178 }
179 clearState();
180
181 // 1. Save request for async operations.
182 m_pendingRequestData = WTFMove(data);
183 m_pendingCompletionHandler = WTFMove(callback);
184
185 // 2. Ask clients to show appropriate UI if any and then start the request.
186 initTimeOutTimer();
187
188 // FIXME<rdar://problem/70822834>: The m_mode is used to determine whether or not we are in the UIProcess.
189 // If so, continue to the old route. Otherwise, use the modern WebAuthn process way.
190 if (m_mode == Mode::Compatible) {
191 runPanel();
192 return;
193 }
194 runPresenter();
195}

197void AuthenticatorManager::cancelRequest(const PageIdentifier& pageID, const std::optional<FrameIdentifier>& frameID)
198{
199 if (!m_pendingCompletionHandler)
200 return;
201 if (auto pendingFrameID = m_pendingRequestData.frameID) {
202 if (pendingFrameID->pageID != pageID)
203 return;
204 if (frameID && frameID != pendingFrameID->frameID)
205 return;
206 }
207 cancelRequest();
208}

210// The following implements part of Step 20. of https://www.w3.org/TR/webauthn/#createCredential
211// and part of Step 18. of https://www.w3.org/TR/webauthn/#getAssertion as of 4 March 2019:
212// "If the user exercises a user agent user-interface option to cancel the process,".
213void AuthenticatorManager::cancelRequest(const API::WebAuthenticationPanel& panel)
214{
215 RELEASE_ASSERT(RunLoop::isMain());
216 if (!m_pendingCompletionHandler || m_pendingRequestData.panel.get() != &panel)
217 return;
218 cancelRequest();
219}

221void AuthenticatorManager::cancel()

226 cancelRequest();

259{

281void AuthenticatorManager::respondReceived(Respond&& respond)
282{
283 ASSERT(RunLoop::isMain());
284 if (!m_requestTimeOutTimer.isActive())
285 return;
286 ASSERT(m_pendingCompletionHandler);
287
288 auto shouldComplete = WTF::holds_alternative<Ref<AuthenticatorResponse>>(respond);
289 if (!shouldComplete)
290 shouldComplete = WTF::get<ExceptionData>(respond).code == InvalidStateError;
291 if (shouldComplete) {
292 invokePendingCompletionHandler(WTFMove(respond));
293 clearStateAsync();
294 m_requestTimeOutTimer.stop();
295 return;
296 }
297 respondReceivedInternal(WTFMove(respond));
298 restartDiscovery();
299}

329void AuthenticatorManager::requestPin(uint64_t retries, CompletionHandler<void(const WTF::String&)>&& completionHandler)
330{
331 // Cache the PIN to improve NFC user experience so that a momentary movement of the NFC key away from the scanner doesn't
332 // force the PIN entry to be re-entered.
333 // We don't distinguish USB and NFC here becuase there is no harms to have this optimization for USB even though it is useless.
334 if (!m_pendingRequestData.cachedPin.isNull()) {
335 completionHandler(m_pendingRequestData.cachedPin);
336 m_pendingRequestData.cachedPin = String();
337 return;
338 }
339
340 auto callback = [weakThis = makeWeakPtr(*this), this, completionHandler = WTFMove(completionHandler)] (const WTF::String& pin) mutable {
341 if (!weakThis)
342 return;
343
344 m_pendingRequestData.cachedPin = pin;
345 completionHandler(pin);
346 };
347
348 // This is for the new UI.
349 if (m_presenter) {
350 m_presenter->requestPin(retries, WTFMove(callback));
351 return;
352 }
353
354 dispatchPanelClientCall([retries, callback = WTFMove(callback)] (const API::WebAuthenticationPanel& panel) mutable {
355 panel.client().requestPin(retries, WTFMove(callback));
356 });
357}

403void AuthenticatorManager::filterTransports(TransportSet& transports) const
404{
405 if (!NfcService::isAvailable())
406 transports.remove(AuthenticatorTransport::Nfc);
407 if (!LocalService::isAvailable())
408 transports.remove(AuthenticatorTransport::Internal);
409
410 // For the modern UI, we should only consider invoking it when the operation is triggered by users.
411 if (!m_pendingRequestData.processingUserGesture)
412 transports.clear();
413}

415void AuthenticatorManager::startDiscovery(const TransportSet& transports)
416{
417 ASSERT(RunLoop::isMain());
418 ASSERT(m_services.isEmpty() && transports.size() <= maxTransportNumber);
419 for (auto& transport : transports) {
420 auto service = createService(transport, *this);
421 service->startDiscovery();
422 m_services.append(WTFMove(service));
423 }
424}

426void AuthenticatorManager::initTimeOutTimer()
427{
428 std::optional<unsigned> timeOutInMs;
429 WTF::switchOn(m_pendingRequestData.options, [&](const PublicKeyCredentialCreationOptions& options) {
430 timeOutInMs = options.timeout;
431 }, [&](const PublicKeyCredentialRequestOptions& options) {
432 timeOutInMs = options.timeout;
433 });
434
435 unsigned timeOutInMsValue = std::min(maxTimeOutValue, timeOutInMs.value_or(maxTimeOutValue));
436 m_requestTimeOutTimer.startOneShot(Seconds::fromMilliseconds(timeOutInMsValue));
437}

445void AuthenticatorManager::runPanel()
446{
447 auto* page = m_pendingRequestData.page.get();
448 if (!page)
449 return;
450 ASSERT(m_pendingRequestData.frameID && page->webPageID() == m_pendingRequestData.frameID->pageID);
451 auto* frame = page->process().webFrame(m_pendingRequestData.frameID->frameID);
452 if (!frame)
453 return;
454
455 // Get available transports and start discovering authenticators on them.
456 auto& options = m_pendingRequestData.options;
457 auto transports = getTransports();
458 if (transports.isEmpty()) {
459 cancel();
460 return;
461 }
462
463 m_pendingRequestData.panel = API::WebAuthenticationPanel::create(*this, getRpId(options), transports, getClientDataType(options), getUserName(options));
464 auto& panel = *m_pendingRequestData.panel;
465 page->uiClient().runWebAuthenticationPanel(*page, panel, *frame, FrameInfoData { m_pendingRequestData.frameInfo }, [transports = WTFMove(transports), weakPanel = makeWeakPtr(panel), weakThis = makeWeakPtr(*this), this] (WebAuthenticationPanelResult result) {
466 // The panel address is used to determine if the current pending request is still the same.
467 if (!weakThis || !weakPanel
468 || (result == WebAuthenticationPanelResult::DidNotPresent)
469 || (weakPanel.get() != m_pendingRequestData.panel.get()))
470 return;
471 startDiscovery(transports);
472 });
473}

475void AuthenticatorManager::runPresenter()
476{
477 // Get available transports and start discovering authenticators on them.
478 auto transports = getTransports();
479 if (transports.isEmpty()) {
480 cancel();
481 return;
482 }
483
484 startDiscovery(transports);
485
486 // For native API support, we skip the UI part. The native API will handle that.
487 if (m_mode == Mode::Native)
488 return;
489
490 runPresenterInternal(transports);
491}

493void AuthenticatorManager::runPresenterInternal(const TransportSet& transports)
494{
495 auto& options = m_pendingRequestData.options;
496 m_presenter = makeUnique<AuthenticatorPresenterCoordinator>(*this, getRpId(options), transports, getClientDataType(options), getUserName(options));
497}

499void AuthenticatorManager::invokePendingCompletionHandler(Respond&& respond)
500{
501 auto result = WTF::holds_alternative<Ref<AuthenticatorResponse>>(respond) ? WebAuthenticationResult::Succeeded : WebAuthenticationResult::Failed;
502
503 // This is for the new UI.
504 if (m_presenter)
505 m_presenter->dimissPresenter(result);
506 else {
507 dispatchPanelClientCall([result] (const API::WebAuthenticationPanel& panel) {
508 panel.client().dismissPanel(result);
509 });
510 }
511
512 m_pendingCompletionHandler(WTFMove(respond));
513}

521auto AuthenticatorManager::getTransports() const -> TransportSet
522{
523 TransportSet transports;
524 WTF::switchOn(m_pendingRequestData.options, [&](const PublicKeyCredentialCreationOptions& options) {
525 transports = collectTransports(options.authenticatorSelection);
526 processGoogleLegacyAppIdSupportExtension(options.extensions, transports);
527 }, [&](const PublicKeyCredentialRequestOptions& options) {
528 transports = collectTransports(options.allowCredentials, options.authenticatorAttachment);
529 });
530 filterTransports(transports);
531 return transports;
532}

534void AuthenticatorManager::dispatchPanelClientCall(Function<void(const API::WebAuthenticationPanel&)>&& call) const
535{
536 auto weakPanel = m_pendingRequestData.weakPanel;
537 if (!weakPanel)
538 weakPanel = makeWeakPtr(m_pendingRequestData.panel.get());
539 if (!weakPanel)
540 return;
541
542 // Call delegates in the next run loop to prevent clients' reentrance that would potentially modify the state
543 // of the current run loop in unexpected ways.
544 RunLoop::main().dispatch([weakPanel = WTFMove(weakPanel), call = WTFMove(call)] () {
545 if (!weakPanel)
546 return;
547 call(*weakPanel);
548 });
549}
