49class AuthenticatorManager;
50
51class AuthenticatorPresenterCoordinator : public CanMakeWeakPtr<AuthenticatorPresenterCoordinator> {
52 WTF_MAKE_FAST_ALLOCATED;
53 WTF_MAKE_NONCOPYABLE(AuthenticatorPresenterCoordinator);
54public:
55 using TransportSet = HashSet<WebCore::AuthenticatorTransport, WTF::IntHash<WebCore::AuthenticatorTransport>, WTF::StrongEnumHashTraits<WebCore::AuthenticatorTransport>>;
56 using CredentialRequestHandler = Function<void(ASCAppleIDCredential *, NSError *)>;
57
58 AuthenticatorPresenterCoordinator(const AuthenticatorManager&, const String& rpId, const TransportSet&, WebCore::ClientDataType, const String& username);
59 ~AuthenticatorPresenterCoordinator();
60
61 void updatePresenter(WebAuthenticationStatus);
62 void requestPin(uint64_t retries, CompletionHandler<void(const String&)>&&);
63 void selectAssertionResponse(Vector<Ref<WebCore::AuthenticatorAssertionResponse>>&&, WebAuthenticationSource, CompletionHandler<void(WebCore::AuthenticatorAssertionResponse*)>&&);
64 void requestLAContextForUserVerification(CompletionHandler<void(LAContext *)>&&);
65 void dimissPresenter(WebAuthenticationResult);
66
67 void setCredentialRequestHandler(CredentialRequestHandler&& handler) { m_credentialRequestHandler = WTFMove(handler); }
68 void setLAContext(LAContext *);
69 void didSelectAssertionResponse(const String& credentialName, LAContext *);
70 void setPin(const String&);
71
72private:
73 WeakPtr<AuthenticatorManager> m_manager;
74 RetainPtr<ASCAuthorizationPresentationContext> m_context;
75 RetainPtr<ASCAuthorizationPresenter> m_presenter;
76 RetainPtr<WKASCAuthorizationPresenterDelegate> m_presenterDelegate;
77 Function<void()> m_delayedPresentation;
78#if HAVE(ASC_AUTH_UI)
79 bool m_delayedPresentationNeedsSecurityKey { false };
80#endif
81
82 CredentialRequestHandler m_credentialRequestHandler;
83
84 CompletionHandler<void(LAContext *)> m_laContextHandler;
85 RetainPtr<LAContext> m_laContext;
86
87 CompletionHandler<void(WebCore::AuthenticatorAssertionResponse*)> m_responseHandler;
88 HashMap<String, RefPtr<WebCore::AuthenticatorAssertionResponse>> m_credentials;
89
90 CompletionHandler<void(const String&)> m_pinHandler;
91#if HAVE(ASC_AUTH_UI)
92 bool m_presentedPIN { false };
93#endif
94};
