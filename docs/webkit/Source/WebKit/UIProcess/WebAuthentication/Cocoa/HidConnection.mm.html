39static void reportReceived(void* context, IOReturn status, void*, IOHIDReportType type, uint32_t reportID, uint8_t* report, CFIndex reportLength)
40{
41 ASSERT(RunLoop::isMain());
42 auto* connection = static_cast<HidConnection*>(context);
43
44 if (status) {
45 LOG_ERROR("Couldn't receive report from the authenticator: %d", status);
46 connection->receiveReport({ });
47 return;
48 }
49 ASSERT(type == kIOHIDReportTypeInput);
50 // FIXME(191519): Determine report id and max report size dynamically.
51 ASSERT(reportID == kHidReportId);
52 ASSERT(reportLength == kHidMaxPacketSize);
53
54 connection->receiveReport({ report, static_cast<size_t>(reportLength) });
55}

57HidConnection::HidConnection(IOHIDDeviceRef device)
58 : m_device(device)
59{
60}
61
62HidConnection::~HidConnection()
63{
64 ASSERT(!m_isInitialized);
65}
66
67void HidConnection::initialize()
68{
69 IOHIDDeviceOpen(m_device.get(), kIOHIDOptionsTypeNone);
70 IOHIDDeviceScheduleWithRunLoop(m_device.get(), CFRunLoopGetCurrent(), kCFRunLoopDefaultMode);
71 m_inputBuffer.resize(kHidMaxPacketSize);
72 IOHIDDeviceRegisterInputReportCallback(m_device.get(), m_inputBuffer.data(), m_inputBuffer.size(), &reportReceived, this);
73 m_isInitialized = true;
74}
75
76void HidConnection::terminate()
77{
78 IOHIDDeviceRegisterInputReportCallback(m_device.get(), m_inputBuffer.data(), m_inputBuffer.size(), nullptr, nullptr);
79 IOHIDDeviceUnscheduleFromRunLoop(m_device.get(), CFRunLoopGetCurrent(), kCFRunLoopDefaultMode);
80 IOHIDDeviceClose(m_device.get(), kIOHIDOptionsTypeNone);
81 m_isInitialized = false;
82}

84auto HidConnection::sendSync(const Vector<uint8_t>& data) -> DataSent
85{
86 ASSERT(m_isInitialized);
87 auto status = IOHIDDeviceSetReport(m_device.get(), kIOHIDReportTypeOutput, kHidReportId, data.data(), data.size());
88 if (status) {
89 LOG_ERROR("Couldn't send report to the authenticator: %d", status);
90 return DataSent::No;
91 }
92 return DataSent::Yes;
93}
94
95void HidConnection::send(Vector<uint8_t>&& data, DataSentCallback&& callback)
96{
97 ASSERT(m_isInitialized);
98 auto task = makeBlockPtr([device = m_device, data = WTFMove(data), callback = WTFMove(callback)]() mutable {
99 ASSERT(!RunLoop::isMain());
100
101 DataSent sent = DataSent::Yes;
102 auto status = IOHIDDeviceSetReport(device.get(), kIOHIDReportTypeOutput, kHidReportId, data.data(), data.size());
103 if (status) {
104 LOG_ERROR("Couldn't send report to the authenticator: %d", status);
105 sent = DataSent::No;
106 }
107 RunLoop::main().dispatch([callback = WTFMove(callback), sent]() mutable {
108 callback(sent);
109 });
110 });
111 dispatch_async(dispatch_get_global_queue(DISPATCH_QUEUE_PRIORITY_DEFAULT, 0), task.get());
112}
113
114void HidConnection::registerDataReceivedCallback(DataReceivedCallback&& callback)
115{
116 ASSERT(m_isInitialized);
117 ASSERT(!m_inputCallback);
118 m_inputCallback = WTFMove(callback);
119 consumeReports();
120 registerDataReceivedCallbackInternal();
121}
122
123void HidConnection::unregisterDataReceivedCallback()
124{
125 m_inputCallback = nullptr;
126}
127
128void HidConnection::receiveReport(Vector<uint8_t>&& report)
129{
130 m_inputReports.append(WTFMove(report));
131 consumeReports();
132}
133
134void HidConnection::consumeReports()
135{
136 while (!m_inputReports.isEmpty() && m_inputCallback)
137 m_inputCallback(m_inputReports.takeFirst());
138}

142}
