64void LocalConnection::verifyUser(const String& rpId, ClientDataType type, SecAccessControlRef accessControl, UserVerificationRequirement uv, UserVerificationCallback&& completionHandler)
65{
66 String title = genericTouchIDPromptTitle();
67#if PLATFORM(MAC)
68 switch (type) {
69 case ClientDataType::Create:
70 title = makeCredentialTouchIDPromptTitle(bundleName(), rpId);
71 break;
72 case ClientDataType::Get:
73 title = getAssertionTouchIDPromptTitle(bundleName(), rpId);
74 break;
75 default:
76 ASSERT_NOT_REACHED();
77 }
78#endif
79
80 m_context = [allocLAContextInstance() init];
81
82 auto options = adoptNS([[NSMutableDictionary alloc] init]);
83 if ([m_context biometryType] == LABiometryTypeTouchID) {
84 [options setObject:title forKey:@(LAOptionAuthenticationTitle)];
85 [options setObject:@NO forKey:@(LAOptionFallbackVisible)];
86 }
87
88 auto reply = makeBlockPtr([context = m_context, completionHandler = WTFMove(completionHandler)] (NSDictionary *information, NSError *error) mutable {
89 UserVerification verification = UserVerification::Yes;
90 if (error) {
91 LOG_ERROR("Couldn't authenticate with biometrics: %@", error);
92 verification = UserVerification::No;
93 if (error.code == LAErrorUserCancel)
94 verification = UserVerification::Cancel;
95 }
96 if (information[@"UserPresence"])
97 verification = UserVerification::Presence;
98
99 // This block can be executed in another thread.
100 RunLoop::main().dispatch([completionHandler = WTFMove(completionHandler), verification, context = WTFMove(context)] () mutable {
101 completionHandler(verification, context.get());
102 });
103 });
104
105#if USE(APPLE_INTERNAL_SDK)
106 // Depending on certain internal requirements, accessControl might not require user verifications.
107 // Hence, here introduces a quirk to force the compatible mode to require user verifications if necessary.
108 if (shouldUseAlternateAttributes()) {
109 NSError *error = nil;
110 auto canEvaluatePolicy = [m_context canEvaluatePolicy:LAPolicyDeviceOwnerAuthenticationWithBiometrics error:&error];
111 if (error.code == LAErrorBiometryLockout)
112 canEvaluatePolicy = true;
113
114 if (uv == UserVerificationRequirement::Required || canEvaluatePolicy) {
115 [m_context evaluatePolicy:LAPolicyDeviceOwnerAuthentication options:options.get() reply:reply.get()];
116 return;
117 }
118
119 reply(@{ @"UserPresence": @YES }, nullptr);
120 return;
121 }
122#endif
123
124 [m_context evaluateAccessControl:accessControl operation:LAAccessControlOperationUseKeySign options:options.get() reply:reply.get()];
125}

127void LocalConnection::verifyUser(SecAccessControlRef accessControl, LAContext *context, CompletionHandler<void(UserVerification)>&& completionHandler)
128{
129 auto options = adoptNS([[NSMutableDictionary alloc] init]);
130 [options setObject:@YES forKey:@(LAOptionNotInteractive)];
131
132 auto reply = makeBlockPtr([completionHandler = WTFMove(completionHandler)] (NSDictionary *information, NSError *error) mutable {
133 UserVerification verification = UserVerification::Yes;
134 if (error) {
135 LOG_ERROR("Couldn't authenticate with biometrics: %@", error);
136 verification = UserVerification::No;
137 if (error.code == LAErrorUserCancel)
138 verification = UserVerification::Cancel;
139 }
140 if (information[@"UserPresence"])
141 verification = UserVerification::Presence;
142
143 // This block can be executed in another thread.
144 RunLoop::main().dispatch([completionHandler = WTFMove(completionHandler), verification] () mutable {
145 completionHandler(verification);
146 });
147 });
148
149#if USE(APPLE_INTERNAL_SDK)
150 // Depending on certain internal requirements, context might be nil. In that case, just check user presence.
151 if (shouldUseAlternateAttributes() && !context) {
152 reply(@{ @"UserPresence": @YES }, nullptr);
153 return;
154 }
155
156#if PLATFORM(IOS_FAMILY_SIMULATOR)
157 // Simulator doesn't support LAAccessControlOperationUseKeySign, but does support alternate attributes.
158 if (shouldUseAlternateAttributes()) {
159 reply(@{ }, nullptr);
160 return;
161 }
162#endif
163#endif // USE(APPLE_INTERNAL_SDK)
164
165 [context evaluateAccessControl:accessControl operation:LAAccessControlOperationUseKeySign options:options.get() reply:reply.get()];
166}

168RetainPtr<SecKeyRef> LocalConnection::createCredentialPrivateKey(LAContext *context, SecAccessControlRef accessControlRef, const String& secAttrLabel, NSData *secAttrApplicationTag) const
169{
170 NSDictionary *privateKeyAttributes = @{
171 (id)kSecAttrAccessControl: (id)accessControlRef,
172 (id)kSecAttrIsPermanent: @YES,
173 (id)kSecAttrAccessGroup: (id)String(LocalAuthenticatiorAccessGroup),
174 (id)kSecAttrLabel: secAttrLabel,
175 (id)kSecAttrApplicationTag: secAttrApplicationTag,
176 };
177
178 if (context) {
179 privateKeyAttributes = [privateKeyAttributes mutableCopy];
180 ((NSMutableDictionary *)privateKeyAttributes)[(id)kSecUseAuthenticationContext] = context;
181 }
182
183 NSDictionary *attributes = @{
184 (id)kSecAttrTokenID: (id)kSecAttrTokenIDSecureEnclave,
185 (id)kSecAttrKeyType: (id)kSecAttrKeyTypeECSECPrimeRandom,
186 (id)kSecAttrKeySizeInBits: @256,
187 (id)kSecPrivateKeyAttrs: privateKeyAttributes,
188 };
189
190 LOCAL_CONNECTION_ADDITIONS
191 CFErrorRef errorRef = nullptr;
192 auto credentialPrivateKey = adoptCF(SecKeyCreateRandomKey((__bridge CFDictionaryRef)attributes, &errorRef));
193 auto retainError = adoptCF(errorRef);
194 if (errorRef) {
195 LOG_ERROR("Couldn't create private key: %@", (NSError *)errorRef);
196 return nullptr;
197 }
198 return credentialPrivateKey;
199}
