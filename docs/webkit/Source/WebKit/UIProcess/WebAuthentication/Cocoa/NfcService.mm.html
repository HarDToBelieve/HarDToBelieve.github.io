42NfcService::NfcService(Observer& observer)
43 : FidoService(observer)
44 , m_restartTimer(RunLoop::main(), this, &NfcService::platformStartDiscovery)

46}

50}

52bool NfcService::isAvailable()
53{
54#if HAVE(NEAR_FIELD)
55 return [[getNFHardwareManagerClass() sharedHardwareManager] areFeaturesSupported:NFFeatureReaderMode outError:nil];
56#else
57 return false;
58#endif
59}
60
61void NfcService::didConnectTag()
62{
63#if HAVE(NEAR_FIELD)
64 auto connection = m_connection;
65 ASSERT(connection);
66 getInfo(WTF::makeUnique<CtapNfcDriver>(connection.releaseNonNull()));
67#endif
68}
69
70void NfcService::didDetectMultipleTags() const
71{
72 if (auto* observer = this->observer())
73 observer->serviceStatusUpdated(WebAuthenticationStatus::MultipleNFCTagsPresent);
74}

83void NfcService::startDiscoveryInternal()
84{
85 platformStartDiscovery();
86}
87
88void NfcService::restartDiscoveryInternal()
89{
90#if HAVE(NEAR_FIELD)
91 if (m_connection)
92 m_connection->stop();
93#endif
94 m_restartTimer.startOneShot(1_s); // Magic number to give users enough time for reactions.
95}
96
97void NfcService::platformStartDiscovery()
98{
99#if HAVE(NEAR_FIELD)
100 if (!isAvailable())
101 return;
102
103 // Will be executed in a different thread.
104 auto callback = makeBlockPtr([weakThis = makeWeakPtr(*this), this] (NFReaderSession *session, NSError *error) mutable {
105 ASSERT(!RunLoop::isMain());
106 if (error) {
107 LOG_ERROR("Couldn't start a NFC reader session: %@", error);
108 return;
109 }
110
111 RunLoop::main().dispatch([weakThis = WTFMove(weakThis), this, session = retainPtr(session)] () mutable {
112 if (!weakThis) {
113 [session endSession];
114 return;
115 }
116
117 // NfcConnection will take care of polling tags and connecting to them.
118 m_connection = NfcConnection::create(WTFMove(session), *this);
119 });
120 });
121 [[getNFHardwareManagerClass() sharedHardwareManager] startReaderSession:callback.get()];
122#endif // HAVE(NEAR_FIELD)
123}
