180MockNfcService::MockNfcService(Observer& observer, const WebCore::MockWebAuthenticationConfiguration& configuration)
181 : NfcService(observer)
182 , m_configuration(configuration)
183{
184}

186NSData* MockNfcService::transceive()
187{
188 if (m_configuration.nfc->payloadBase64.isEmpty())
189 return nil;
190
191 auto result = adoptNS([[NSData alloc] initWithBase64EncodedString:m_configuration.nfc->payloadBase64[0] options:NSDataBase64DecodingIgnoreUnknownCharacters]);
192 m_configuration.nfc->payloadBase64.remove(0);
193 return result.autorelease();
194}
195
196void MockNfcService::receiveStopPolling()
197{
198 // For purpose of restart polling.
199 m_configuration.nfc->multiplePhysicalTags = false;
200}
201
202void MockNfcService::receiveStartPolling()
203{
204 RunLoop::main().dispatch([weakThis = makeWeakPtr(*this)] {
205 if (!weakThis)
206 return;
207 weakThis->detectTags();
208 });
209}

211void MockNfcService::platformStartDiscovery()
212{
213#if HAVE(NEAR_FIELD)
214 if (!!m_configuration.nfc) {
215 globalNfcService = this;
216
217 Method methodToSwizzle1 = class_getInstanceMethod(getNFReaderSessionClass(), @selector(setDelegate:));
218 method_setImplementation(methodToSwizzle1, (IMP)NFReaderSessionSetDelegate);
219
220 Method methodToSwizzle2 = class_getInstanceMethod(getNFReaderSessionClass(), @selector(connectTag:));
221 if (m_configuration.nfc->error == Mock::NfcError::NoConnections)
222 method_setImplementation(methodToSwizzle2, (IMP)NFReaderSessionConnectTagFail);
223 else
224 method_setImplementation(methodToSwizzle2, (IMP)NFReaderSessionConnectTag);
225
226 Method methodToSwizzle3 = class_getInstanceMethod(getNFReaderSessionClass(), @selector(transceive:));
227 method_setImplementation(methodToSwizzle3, (IMP)NFReaderSessionTransceive);
228
229 Method methodToSwizzle4 = class_getInstanceMethod(getNFReaderSessionClass(), @selector(stopPolling));
230 method_setImplementation(methodToSwizzle4, (IMP)NFReaderSessionStopPolling);
231
232 Method methodToSwizzle5 = class_getInstanceMethod(getNFReaderSessionClass(), @selector(startPollingWithError:));
233 method_setImplementation(methodToSwizzle5, (IMP)NFReaderSessionStartPollingWithError);
234
235 auto readerSession = adoptNS([allocNFReaderSessionInstance() init]);
236 setConnection(NfcConnection::create(readerSession.get(), *this));
237 }
238 LOG_ERROR("No nfc authenticators is available.");
239#endif // HAVE(NEAR_FIELD)
240}

242void MockNfcService::detectTags() const
243{
244#if HAVE(NEAR_FIELD)
245 if (m_configuration.nfc->error == Mock::NfcError::NoTags)
246 return;
247
248 auto callback = makeBlockPtr([configuration = m_configuration] {
249 auto tags = adoptNS([[NSMutableArray alloc] init]);
250 if (configuration.nfc->error == Mock::NfcError::WrongTagType || configuration.nfc->multipleTags)
251 [tags addObject:adoptNS([[WKMockNFTag alloc] initWithType:NFTagTypeUnknown]).get()];
252 else
253 [tags addObject:adoptNS([[WKMockNFTag alloc] initWithType:NFTagTypeGeneric4A]).get()];
254
255 if (configuration.nfc->multipleTags)
256 [tags addObject:adoptNS([[WKMockNFTag alloc] initWithType:NFTagTypeGeneric4A]).get()];
257
258 if (configuration.nfc->multiplePhysicalTags)
259 [tags addObject:adoptNS([[WKMockNFTag alloc] initWithType:NFTagTypeGeneric4A tagID:adoptNS([[NSData alloc] initWithBytes:tagID2 length:sizeof(tagID2)]).get()]).get()];
260
261 [globalNFReaderSessionDelegate readerSession:nil didDetectTags:tags.get()];
262 });
263 dispatch_async(dispatch_get_global_queue(DISPATCH_QUEUE_PRIORITY_DEFAULT, 0), callback.get());
264#endif // HAVE(NEAR_FIELD)
265}
