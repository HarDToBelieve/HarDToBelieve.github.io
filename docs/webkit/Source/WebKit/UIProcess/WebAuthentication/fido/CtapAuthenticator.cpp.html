51WebAuthenticationStatus toStatus(const CtapDeviceResponseCode& error)
52{
53 switch (error) {
54 case CtapDeviceResponseCode::kCtap2ErrPinAuthInvalid:
55 case CtapDeviceResponseCode::kCtap2ErrPinInvalid:
56 return WebAuthenticationStatus::PinInvalid;
57 case CtapDeviceResponseCode::kCtap2ErrPinAuthBlocked:
58 return WebAuthenticationStatus::PinAuthBlocked;
59 case CtapDeviceResponseCode::kCtap2ErrPinBlocked:
60 return WebAuthenticationStatus::PinBlocked;
61 default:
62 ASSERT_NOT_REACHED();
63 return WebAuthenticationStatus::PinInvalid;
64 }
65}
66
67bool isPinError(const CtapDeviceResponseCode& error)
68{
69 switch (error) {
70 case CtapDeviceResponseCode::kCtap2ErrPinAuthInvalid:
71 case CtapDeviceResponseCode::kCtap2ErrPinAuthBlocked:
72 case CtapDeviceResponseCode::kCtap2ErrPinInvalid:
73 case CtapDeviceResponseCode::kCtap2ErrPinBlocked:
74 return true;
75 default:
76 return false;
77 }
78}

88void CtapAuthenticator::makeCredential()
89{
90 ASSERT(!m_isDowngraded);
91 if (processGoogleLegacyAppIdSupportExtension())
92 return;
93 Vector<uint8_t> cborCmd;
94 auto& options = WTF::get<PublicKeyCredentialCreationOptions>(requestData().options);
95 if (m_info.options().clientPinAvailability() == AuthenticatorSupportedOptions::ClientPinAvailability::kSupportedAndPinSet)
96 cborCmd = encodeMakeCredenitalRequestAsCBOR(requestData().hash, options, m_info.options().userVerificationAvailability(), PinParameters { pin::kProtocolVersion, m_pinAuth });
97 else
98 cborCmd = encodeMakeCredenitalRequestAsCBOR(requestData().hash, options, m_info.options().userVerificationAvailability());
99 driver().transact(WTFMove(cborCmd), [weakThis = makeWeakPtr(*this)](Vector<uint8_t>&& data) {
100 ASSERT(RunLoop::isMain());
101 if (!weakThis)
102 return;
103 weakThis->continueMakeCredentialAfterResponseReceived(WTFMove(data));
104 });
105}

107void CtapAuthenticator::continueMakeCredentialAfterResponseReceived(Vector<uint8_t>&& data)
108{
109 auto response = readCTAPMakeCredentialResponse(data, AuthenticatorAttachment::CrossPlatform, WTF::get<PublicKeyCredentialCreationOptions>(requestData().options).attestation);
110 if (!response) {
111 auto error = getResponseCode(data);
112
113 if (error == CtapDeviceResponseCode::kCtap2ErrCredentialExcluded) {
114 receiveRespond(ExceptionData { InvalidStateError, "At least one credential matches an entry of the excludeCredentials list in the authenticator."_s });
115 return;
116 }
117
118 if (isPinError(error)) {
119 if (!m_pinAuth.isEmpty() && observer()) // Skip the very first command that acts like wink.
120 observer()->authenticatorStatusUpdated(toStatus(error));
121 if (tryRestartPin(error))
122 return;
123 }
124
125 receiveRespond(ExceptionData { UnknownError, makeString("Unknown internal error. Error code: ", static_cast<uint8_t>(error)) });
126 return;
127 }
128 receiveRespond(response.releaseNonNull());
129}

131void CtapAuthenticator::getAssertion()
132{
133 ASSERT(!m_isDowngraded);
134 Vector<uint8_t> cborCmd;
135 auto& options = WTF::get<PublicKeyCredentialRequestOptions>(requestData().options);
136 if (m_info.options().clientPinAvailability() == AuthenticatorSupportedOptions::ClientPinAvailability::kSupportedAndPinSet && options.userVerification != UserVerificationRequirement::Discouraged)
137 cborCmd = encodeGetAssertionRequestAsCBOR(requestData().hash, options, m_info.options().userVerificationAvailability(), PinParameters { pin::kProtocolVersion, m_pinAuth });
138 else
139 cborCmd = encodeGetAssertionRequestAsCBOR(requestData().hash, options, m_info.options().userVerificationAvailability());
140 driver().transact(WTFMove(cborCmd), [weakThis = makeWeakPtr(*this)](Vector<uint8_t>&& data) {
141 ASSERT(RunLoop::isMain());
142 if (!weakThis)
143 return;
144 weakThis->continueGetAssertionAfterResponseReceived(WTFMove(data));
145 });
146}

148void CtapAuthenticator::continueGetAssertionAfterResponseReceived(Vector<uint8_t>&& data)
149{
150 auto response = readCTAPGetAssertionResponse(data, AuthenticatorAttachment::CrossPlatform);
151 if (!response) {
152 auto error = getResponseCode(data);
153
154 if (!isPinError(error) && tryDowngrade())
155 return;
156
157 if (isPinError(error)) {
158 if (!m_pinAuth.isEmpty() && observer()) // Skip the very first command that acts like wink.
159 observer()->authenticatorStatusUpdated(toStatus(error));
160 if (tryRestartPin(error))
161 return;
162 }
163
164 if (error == CtapDeviceResponseCode::kCtap2ErrNoCredentials && observer())
165 observer()->authenticatorStatusUpdated(WebAuthenticationStatus::NoCredentialsFound);
166
167 receiveRespond(ExceptionData { UnknownError, makeString("Unknown internal error. Error code: ", static_cast<uint8_t>(error)) });
168 return;
169 }
170
171 if (response->numberOfCredentials() <= 1) {
172 receiveRespond(response.releaseNonNull());
173 return;
174 }
175
176 m_remainingAssertionResponses = response->numberOfCredentials() - 1;
177 m_assertionResponses.reserveInitialCapacity(response->numberOfCredentials());
178 m_assertionResponses.uncheckedAppend(response.releaseNonNull());
179 driver().transact(encodeEmptyAuthenticatorRequest(CtapRequestCommand::kAuthenticatorGetNextAssertion), [weakThis = makeWeakPtr(*this)](Vector<uint8_t>&& data) {
180 ASSERT(RunLoop::isMain());
181 if (!weakThis)
182 return;
183 weakThis->continueGetNextAssertionAfterResponseReceived(WTFMove(data));
184 });
185}
186
187void CtapAuthenticator::continueGetNextAssertionAfterResponseReceived(Vector<uint8_t>&& data)
188{
189 auto response = readCTAPGetAssertionResponse(data, AuthenticatorAttachment::CrossPlatform);
190 if (!response) {
191 auto error = getResponseCode(data);
192 receiveRespond(ExceptionData { UnknownError, makeString("Unknown internal error. Error code: ", static_cast<uint8_t>(error)) });
193 return;
194 }
195 m_remainingAssertionResponses--;
196 m_assertionResponses.uncheckedAppend(response.releaseNonNull());
197
198 if (!m_remainingAssertionResponses) {
199 if (auto* observer = this->observer()) {
200 Vector<Ref<AuthenticatorAssertionResponse>> responsesCopy;
201 responsesCopy.reserveInitialCapacity(m_assertionResponses.size());
202 for (auto& response : m_assertionResponses)
203 responsesCopy.uncheckedAppend(response.copyRef());
204
205 observer->selectAssertionResponse(WTFMove(responsesCopy), WebAuthenticationSource::External, [this, weakThis = makeWeakPtr(*this)] (AuthenticatorAssertionResponse* response) {
206 ASSERT(RunLoop::isMain());
207 if (!weakThis)
208 return;
209 auto result = m_assertionResponses.findMatching([expectedResponse = response] (auto& response) {
210 return response.ptr() == expectedResponse;
211 });
212 if (result == notFound)
213 return;
214 receiveRespond(m_assertionResponses[result].copyRef());
215 });
216 }
217 return;
218 }
219
220 driver().transact(encodeEmptyAuthenticatorRequest(CtapRequestCommand::kAuthenticatorGetNextAssertion), [weakThis = makeWeakPtr(*this)](Vector<uint8_t>&& data) {
221 ASSERT(RunLoop::isMain());
222 if (!weakThis)
223 return;
224 weakThis->continueGetNextAssertionAfterResponseReceived(WTFMove(data));
225 });
226}

239void CtapAuthenticator::continueGetKeyAgreementAfterGetRetries(Vector<uint8_t>&& data)
240{
241 auto retries = pin::RetriesResponse::parse(data);
242 if (!retries) {
243 auto error = getResponseCode(data);
244 receiveRespond(ExceptionData { UnknownError, makeString("Unknown internal error. Error code: ", static_cast<uint8_t>(error)) });
245 return;
246 }
247
248 auto cborCmd = encodeAsCBOR(pin::KeyAgreementRequest { });
249 driver().transact(WTFMove(cborCmd), [weakThis = makeWeakPtr(*this), retries = retries->retries] (Vector<uint8_t>&& data) {
250 ASSERT(RunLoop::isMain());
251 if (!weakThis)
252 return;
253 weakThis->continueRequestPinAfterGetKeyAgreement(WTFMove(data), retries);
254 });
255}

257void CtapAuthenticator::continueRequestPinAfterGetKeyAgreement(Vector<uint8_t>&& data, uint64_t retries)
258{
259 auto keyAgreement = pin::KeyAgreementResponse::parse(data);
260 if (!keyAgreement) {
261 auto error = getResponseCode(data);
262 receiveRespond(ExceptionData { UnknownError, makeString("Unknown internal error. Error code: ", static_cast<uint8_t>(error)) });
263 return;
264 }
265
266 if (auto* observer = this->observer()) {
267 observer->requestPin(retries, [weakThis = makeWeakPtr(*this), keyAgreement = WTFMove(*keyAgreement)] (const String& pin) {
268 ASSERT(RunLoop::isMain());
269 if (!weakThis)
270 return;
271 weakThis->continueGetPinTokenAfterRequestPin(pin, keyAgreement.peerKey);
272 });
273 }
274}

276void CtapAuthenticator::continueGetPinTokenAfterRequestPin(const String& pin, const CryptoKeyEC& peerKey)
277{
278 if (pin.isNull()) {
279 receiveRespond(ExceptionData { UnknownError, "Pin is null."_s });
280 return;
281 }
282
283 auto pinUtf8 = pin::validateAndConvertToUTF8(pin);
284 if (!pinUtf8) {
285 // Fake a pin invalid response from the authenticator such that clients could show some error to the user.
286 if (auto* observer = this->observer())
287 observer->authenticatorStatusUpdated(WebAuthenticationStatus::PinInvalid);
288 tryRestartPin(CtapDeviceResponseCode::kCtap2ErrPinInvalid);
289 return;
290 }
291 auto tokenRequest = pin::TokenRequest::tryCreate(*pinUtf8, peerKey);
292 if (!tokenRequest) {
293 receiveRespond(ExceptionData { UnknownError, "Cannot create a TokenRequest."_s });
294 return;
295 }
296
297 auto cborCmd = encodeAsCBOR(*tokenRequest);
298 driver().transact(WTFMove(cborCmd), [weakThis = makeWeakPtr(*this), tokenRequest = WTFMove(*tokenRequest)] (Vector<uint8_t>&& data) {
299 ASSERT(RunLoop::isMain());
300 if (!weakThis)
301 return;
302 weakThis->continueRequestAfterGetPinToken(WTFMove(data), tokenRequest);
303 });
304}

306void CtapAuthenticator::continueRequestAfterGetPinToken(Vector<uint8_t>&& data, const fido::pin::TokenRequest& tokenRequest)
307{
308 auto token = pin::TokenResponse::parse(tokenRequest.sharedKey(), data);
309 if (!token) {
310 auto error = getResponseCode(data);
311
312 if (isPinError(error)) {
313 if (auto* observer = this->observer())
314 observer->authenticatorStatusUpdated(toStatus(error));
315 if (tryRestartPin(error))
316 return;
317 }
318
319 receiveRespond(ExceptionData { UnknownError, makeString("Unknown internal error. Error code: ", static_cast<uint8_t>(error)) });
320 return;
321 }
322
323 m_pinAuth = token->pinAuth(requestData().hash);
324 WTF::switchOn(requestData().options, [&](const PublicKeyCredentialCreationOptions& options) {
325 makeCredential();
326 }, [&](const PublicKeyCredentialRequestOptions& options) {
327 getAssertion();
328 });
329}

331bool CtapAuthenticator::tryRestartPin(const CtapDeviceResponseCode& error)
332{
333 switch (error) {
334 case CtapDeviceResponseCode::kCtap2ErrPinAuthInvalid:
335 case CtapDeviceResponseCode::kCtap2ErrPinInvalid:
336 getRetries();
337 return true;
338 default:
339 return false;
340 }
341}

343bool CtapAuthenticator::tryDowngrade()
344{
345 if (m_info.versions().find(ProtocolVersion::kU2f) == m_info.versions().end())
346 return false;
347 if (!observer())
348 return false;
349
350 bool isConvertible = false;
351 WTF::switchOn(requestData().options, [&](const PublicKeyCredentialCreationOptions& options) {
352 isConvertible = isConvertibleToU2fRegisterCommand(options);
353 }, [&](const PublicKeyCredentialRequestOptions& options) {
354 isConvertible = isConvertibleToU2fSignCommand(options);
355 });
356 if (!isConvertible)
357 return false;
358
359 m_isDowngraded = true;
360 driver().setProtocol(ProtocolVersion::kU2f);
361 observer()->downgrade(this, U2fAuthenticator::create(releaseDriver()));
362 return true;
363}

365// Only U2F protocol is supported for Google legacy AppID support.
366bool CtapAuthenticator::processGoogleLegacyAppIdSupportExtension()
367{
368 auto& extensions = WTF::get<PublicKeyCredentialCreationOptions>(requestData().options).extensions;
369 if (!extensions) {
370 // AuthenticatorCoordinator::create should always set it.
371 ASSERT_NOT_REACHED();
372 return false;
373 }
374 if (extensions->googleLegacyAppidSupport)
375 tryDowngrade();
376 return extensions->googleLegacyAppidSupport;
377}
