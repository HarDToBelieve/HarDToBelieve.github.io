26#pragma once
27
28#if ENABLE(WEB_AUTHN)
29
30#include "CtapDriver.h"
31#include "HidConnection.h"
32#include <WebCore/FidoHidMessage.h>
33#include <wtf/UniqueRef.h>
34
35namespace WebKit {
36
37// The following implements the CTAP HID protocol:
38// https://fidoalliance.org/specs/fido-v2.0-ps-20170927/fido-client-to-authenticator-protocol-v2.0-ps-20170927.html#usb
39// FSM: Idle => AllocateChannel => Ready
40class CtapHidDriver final : public CtapDriver {
41public:
42 enum class State : uint8_t {
43 Idle,
44 AllocateChannel,
45 Ready,
46 // FIXME(191528)
47 Busy
48 };
49
50 explicit CtapHidDriver(UniqueRef<HidConnection>&&);
51
52 void transact(Vector<uint8_t>&& data, ResponseCallback&&) final;
53 void cancel() final;
54
55private:
56 // Worker is the helper that maintains the transaction.
57 // https://fidoalliance.org/specs/fido-v2.0-ps-20170927/fido-client-to-authenticator-protocol-v2.0-ps-20170927.html#arbitration
58 // FSM: Idle => Write => Read.
59 class Worker : public CanMakeWeakPtr<Worker> {
60 WTF_MAKE_FAST_ALLOCATED;
61 WTF_MAKE_NONCOPYABLE(Worker);
62 public:
63 using MessageCallback = Function<void(std::optional<fido::FidoHidMessage>&&)>;
64
65 enum class State : uint8_t {
66 Idle,
67 Write,
68 Read
69 };
70
71 explicit Worker(UniqueRef<HidConnection>&&);
72 ~Worker();
73
74 void transact(fido::FidoHidMessage&&, MessageCallback&&);
75 void cancel(fido::FidoHidMessage&&);
76
77 private:
78 void write(HidConnection::DataSent);
79 void read(const Vector<uint8_t>&);
80 void returnMessage();
81 void reset();
82
83 UniqueRef<HidConnection> m_connection;
84 State m_state { State::Idle };
85 std::optional<fido::FidoHidMessage> m_requestMessage;
86 std::optional<fido::FidoHidMessage> m_responseMessage;
87 MessageCallback m_callback;
88 };
89
90 void continueAfterChannelAllocated(std::optional<fido::FidoHidMessage>&&);
91 void continueAfterResponseReceived(std::optional<fido::FidoHidMessage>&&);
92 void returnResponse(Vector<uint8_t>&&);
93 void reset();
94
95 UniqueRef<Worker> m_worker;
96 State m_state { State::Idle };
97 uint32_t m_channelId { fido::kHidBroadcastChannel };
98 // One request at a time.
99 Vector<uint8_t> m_requestData;
100 ResponseCallback m_responseCallback;
101 Vector<uint8_t> m_nonce;
102};
103
104} // namespace WebKit
105
106#endif // ENABLE(WEB_AUTHN)
