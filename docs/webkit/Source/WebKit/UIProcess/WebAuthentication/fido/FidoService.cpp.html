49void FidoService::getInfo(std::unique_ptr<CtapDriver>&& driver)
50{
51 // Get authenticator info from the device.
52 driver->transact(encodeEmptyAuthenticatorRequest(CtapRequestCommand::kAuthenticatorGetInfo), [weakThis = makeWeakPtr(*this), weakDriver = makeWeakPtr(*driver)] (Vector<uint8_t>&& response) mutable {
53 ASSERT(RunLoop::isMain());
54 if (!weakThis)
55 return;
56 weakThis->continueAfterGetInfo(WTFMove(weakDriver), WTFMove(response));
57 });
58 auto addResult = m_drivers.add(WTFMove(driver));
59 ASSERT_UNUSED(addResult, addResult.isNewEntry);
60}
61
62void FidoService::continueAfterGetInfo(WeakPtr<CtapDriver>&& weakDriver, Vector<uint8_t>&& response)
63{
64 if (!weakDriver)
65 return;
66
67 auto driver = m_drivers.take(weakDriver.get());
68 if (!driver || !observer() || response.isEmpty())
69 return;
70
71 auto info = readCTAPGetInfoResponse(response);
72 if (info && info->versions().find(ProtocolVersion::kCtap) != info->versions().end()) {
73 observer()->authenticatorAdded(CtapAuthenticator::create(WTFMove(driver), WTFMove(*info)));
74 return;
75 }
76 LOG_ERROR("Couldn't parse a ctap get info response.");
77 driver->setProtocol(ProtocolVersion::kU2f);
78 observer()->authenticatorAdded(U2fAuthenticator::create(WTFMove(driver)));
79}
