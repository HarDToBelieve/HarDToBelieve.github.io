48U2fAuthenticator::U2fAuthenticator(std::unique_ptr<CtapDriver>&& driver)
49 : FidoAuthenticator(WTFMove(driver))
50 , m_retryTimer(RunLoop::main(), this, &U2fAuthenticator::retryLastCommand)
51{
52}

54void U2fAuthenticator::makeCredential()
55{
56 auto& creationOptions = WTF::get<PublicKeyCredentialCreationOptions>(requestData().options);
57 if (!isConvertibleToU2fRegisterCommand(creationOptions)) {
58 receiveRespond(ExceptionData { NotSupportedError, "Cannot convert the request to U2F command."_s });
59 return;
60 }
61 if (!creationOptions.excludeCredentials.isEmpty()) {
62 ASSERT(!m_nextListIndex);
63 checkExcludeList(m_nextListIndex++);
64 return;
65 }
66 issueRegisterCommand();
67}
68
69void U2fAuthenticator::checkExcludeList(size_t index)
70{
71 auto& creationOptions = WTF::get<PublicKeyCredentialCreationOptions>(requestData().options);
72 if (index >= creationOptions.excludeCredentials.size()) {
73 issueRegisterCommand();
74 return;
75 }
76 auto u2fCmd = convertToU2fCheckOnlySignCommand(requestData().hash, creationOptions, creationOptions.excludeCredentials[index]);
77 ASSERT(u2fCmd);
78 issueNewCommand(WTFMove(*u2fCmd), CommandType::CheckOnlyCommand);
79}
80
81void U2fAuthenticator::issueRegisterCommand()
82{
83 auto u2fCmd = convertToU2fRegisterCommand(requestData().hash, WTF::get<PublicKeyCredentialCreationOptions>(requestData().options));
84 ASSERT(u2fCmd);
85 issueNewCommand(WTFMove(*u2fCmd), CommandType::RegisterCommand);
86}
87
88void U2fAuthenticator::getAssertion()
89{
90 if (!isConvertibleToU2fSignCommand(WTF::get<PublicKeyCredentialRequestOptions>(requestData().options))) {
91 receiveRespond(ExceptionData { NotSupportedError, "Cannot convert the request to U2F command."_s });
92 return;
93 }
94 ASSERT(!m_nextListIndex);
95 issueSignCommand(m_nextListIndex++);
96}
97
98void U2fAuthenticator::issueSignCommand(size_t index)
99{
100 auto& requestOptions = WTF::get<PublicKeyCredentialRequestOptions>(requestData().options);
101 if (index >= requestOptions.allowCredentials.size()) {
102 if (auto* observer = this->observer())
103 observer->authenticatorStatusUpdated(WebAuthenticationStatus::NoCredentialsFound);
104 receiveRespond(ExceptionData { NotAllowedError, "No credentials from the allowCredentials list is found in the authenticator."_s });
105 return;
106 }
107 auto u2fCmd = convertToU2fSignCommand(requestData().hash, requestOptions, requestOptions.allowCredentials[index].idVector, m_isAppId);
108 ASSERT(u2fCmd);
109 issueNewCommand(WTFMove(*u2fCmd), CommandType::SignCommand);
110}

112void U2fAuthenticator::issueNewCommand(Vector<uint8_t>&& command, CommandType type)
113{
114 m_lastCommand = WTFMove(command);
115 m_lastCommandType = type;
116 issueCommand(m_lastCommand, m_lastCommandType);
117}
118
119void U2fAuthenticator::issueCommand(const Vector<uint8_t>& command, CommandType type)
120{
121 driver().transact(Vector<uint8_t>(command), [weakThis = makeWeakPtr(*this), type](Vector<uint8_t>&& data) {
122 ASSERT(RunLoop::isMain());
123 if (!weakThis)
124 return;
125 weakThis->responseReceived(WTFMove(data), type);
126 });
127}
128
129void U2fAuthenticator::responseReceived(Vector<uint8_t>&& response, CommandType type)
130{
131 auto apduResponse = ApduResponse::createFromMessage(response);
132 if (!apduResponse) {
133 receiveRespond(ExceptionData { UnknownError, "Couldn't parse the APDU response."_s });
134 return;
135 }
136
137 switch (type) {
138 case CommandType::RegisterCommand:
139 continueRegisterCommandAfterResponseReceived(WTFMove(*apduResponse));
140 return;
141 case CommandType::CheckOnlyCommand:
142 continueCheckOnlyCommandAfterResponseReceived(WTFMove(*apduResponse));
143 return;
144 case CommandType::BogusCommand:
145 continueBogusCommandAfterResponseReceived(WTFMove(*apduResponse));
146 return;
147 case CommandType::SignCommand:
148 continueSignCommandAfterResponseReceived(WTFMove(*apduResponse));
149 return;
150 }
151 ASSERT_NOT_REACHED();
152}
153
154void U2fAuthenticator::continueRegisterCommandAfterResponseReceived(ApduResponse&& apduResponse)
155{
156 switch (apduResponse.status()) {
157 case ApduResponse::Status::SW_NO_ERROR: {
158 auto& options = WTF::get<PublicKeyCredentialCreationOptions>(requestData().options);
159 auto appId = processGoogleLegacyAppIdSupportExtension(options.extensions);
160 auto response = readU2fRegisterResponse(!appId ? options.rp.id : appId, apduResponse.data(), AuthenticatorAttachment::CrossPlatform, options.attestation);
161 if (!response) {
162 receiveRespond(ExceptionData { UnknownError, "Couldn't parse the U2F register response."_s });
163 return;
164 }
165 receiveRespond(response.releaseNonNull());
166 return;
167 }
168 case ApduResponse::Status::SW_CONDITIONS_NOT_SATISFIED:
169 // Polling is required during test of user presence.
170 m_retryTimer.startOneShot(Seconds::fromMilliseconds(retryTimeOutValueMs));
171 return;
172 default:
173 receiveRespond(ExceptionData { UnknownError, makeString("Unknown internal error. Error code: ", static_cast<unsigned>(apduResponse.status())) });
174 }
175}
176
177void U2fAuthenticator::continueCheckOnlyCommandAfterResponseReceived(ApduResponse&& apduResponse)
178{
179 switch (apduResponse.status()) {
180 case ApduResponse::Status::SW_NO_ERROR:
181 case ApduResponse::Status::SW_CONDITIONS_NOT_SATISFIED:
182 issueNewCommand(constructBogusU2fRegistrationCommand(), CommandType::BogusCommand);
183 return;
184 default:
185 checkExcludeList(m_nextListIndex++);
186 }
187}
188
189void U2fAuthenticator::continueBogusCommandAfterResponseReceived(ApduResponse&& apduResponse)
190{
191 switch (apduResponse.status()) {
192 case ApduResponse::Status::SW_NO_ERROR:
193 receiveRespond(ExceptionData { InvalidStateError, "At least one credential matches an entry of the excludeCredentials list in the authenticator."_s });
194 return;
195 case ApduResponse::Status::SW_CONDITIONS_NOT_SATISFIED:
196 // Polling is required during test of user presence.
197 m_retryTimer.startOneShot(Seconds::fromMilliseconds(retryTimeOutValueMs));
198 return;
199 default:
200 receiveRespond(ExceptionData { UnknownError, makeString("Unknown internal error. Error code: ", static_cast<unsigned>(apduResponse.status())) });
201 }
202}
203
204void U2fAuthenticator::continueSignCommandAfterResponseReceived(ApduResponse&& apduResponse)
205{
206 auto& requestOptions = WTF::get<PublicKeyCredentialRequestOptions>(requestData().options);
207 switch (apduResponse.status()) {
208 case ApduResponse::Status::SW_NO_ERROR: {
209 RefPtr<AuthenticatorAssertionResponse> response;
210 if (m_isAppId) {
211 ASSERT(requestOptions.extensions && !requestOptions.extensions->appid.isNull());
212 response = readU2fSignResponse(requestOptions.extensions->appid, requestOptions.allowCredentials[m_nextListIndex - 1].idVector, apduResponse.data(), AuthenticatorAttachment::CrossPlatform);
213 } else
214 response = readU2fSignResponse(requestOptions.rpId, requestOptions.allowCredentials[m_nextListIndex - 1].idVector, apduResponse.data(), AuthenticatorAttachment::CrossPlatform);
215 if (!response) {
216 receiveRespond(ExceptionData { UnknownError, "Couldn't parse the U2F sign response."_s });
217 return;
218 }
219 if (m_isAppId)
220 response->setExtensions({ m_isAppId });
221
222 receiveRespond(response.releaseNonNull());
223 return;
224 }
225 case ApduResponse::Status::SW_CONDITIONS_NOT_SATISFIED:
226 // Polling is required during test of user presence.
227 m_retryTimer.startOneShot(Seconds::fromMilliseconds(retryTimeOutValueMs));
228 return;
229 case ApduResponse::Status::SW_WRONG_DATA:
230 if (requestOptions.extensions && !requestOptions.extensions->appid.isNull()) {
231 if (!m_isAppId) {
232 m_isAppId = true;
233 issueSignCommand(m_nextListIndex - 1);
234 return;
235 }
236 m_isAppId = false;
237 }
238 issueSignCommand(m_nextListIndex++);
239 return;
240 default:
241 issueSignCommand(m_nextListIndex++);
242 }
