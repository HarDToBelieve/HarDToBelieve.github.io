35namespace WebKit {
36
37static const Seconds expirationDelay { 30_min };
38
39WebBackForwardCacheEntry::WebBackForwardCacheEntry(WebBackForwardCache& backForwardCache, WebCore::BackForwardItemIdentifier backForwardItemID, WebCore::ProcessIdentifier processIdentifier, std::unique_ptr<SuspendedPageProxy>&& suspendedPage)
40 : m_backForwardCache(backForwardCache)
41 , m_processIdentifier(processIdentifier)
42 , m_backForwardItemID(backForwardItemID)
43 , m_suspendedPage(WTFMove(suspendedPage))
44 , m_expirationTimer(RunLoop::main(), this, &WebBackForwardCacheEntry::expirationTimerFired)
45{
46 m_expirationTimer.startOneShot(expirationDelay);
47}
48
49WebBackForwardCacheEntry::~WebBackForwardCacheEntry()
50{
51 if (m_backForwardItemID && !m_suspendedPage) {
52 auto& process = this->process();
53 process.sendWithAsyncReply(Messages::WebProcess::ClearCachedPage(m_backForwardItemID), [] { });
54 }
55}
56
57std::unique_ptr<SuspendedPageProxy> WebBackForwardCacheEntry::takeSuspendedPage()
58{
59 ASSERT(m_suspendedPage);
60 m_backForwardItemID = { };
61 m_expirationTimer.stop();
62 return std::exchange(m_suspendedPage, nullptr);
63}
64
65WebProcessProxy& WebBackForwardCacheEntry::process() const
66{
67 auto* process = WebProcessProxy::processForIdentifier(m_processIdentifier);
68 ASSERT(process);
69 ASSERT(!m_suspendedPage || process == &m_suspendedPage->process());
70 return *process;
71}
72
73void WebBackForwardCacheEntry::expirationTimerFired()
74{
75 RELEASE_LOG(BackForwardCache, "%p - WebBackForwardCacheEntry::expirationTimerFired backForwardItemID=%s, hasSuspendedPage=%d", this, m_backForwardItemID.string().utf8().data(), !!m_suspendedPage);
76 ASSERT(m_backForwardItemID);
77 auto* item = WebBackForwardListItem::itemForID(m_backForwardItemID);
78 ASSERT(item);
79 m_backForwardCache.removeEntry(*item); // Will destroy |this|.
80}
