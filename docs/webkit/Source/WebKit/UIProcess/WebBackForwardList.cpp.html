73void WebBackForwardList::pageClosed()
74{
75 LOG(BackForward, "(Back/Forward) WebBackForwardList %p had its page closed with current size %zu", this, m_entries.size());
76
77 // We should have always started out with an m_page and we should never close the page twice.
78 ASSERT(m_page);
79
80 if (m_page) {
81 size_t size = m_entries.size();
82 for (size_t i = 0; i < size; ++i)
83 didRemoveItem(m_entries[i]);
84 }
85
86 m_page = nullptr;
87 m_entries.clear();
88 m_currentIndex = std::nullopt;
89}

91void WebBackForwardList::addItem(Ref<WebBackForwardListItem>&& newItem)
92{
93 ASSERT(!m_currentIndex || *m_currentIndex < m_entries.size());
94
95 if (!m_page)
96 return;
97
98 Vector<Ref<WebBackForwardListItem>> removedItems;
99
100 if (m_currentIndex) {
101 m_page->recordAutomaticNavigationSnapshot();
102
103 // Toss everything in the forward list.
104 unsigned targetSize = *m_currentIndex + 1;
105 removedItems.reserveCapacity(m_entries.size() - targetSize);
106 while (m_entries.size() > targetSize) {
107 didRemoveItem(m_entries.last());
108 removedItems.append(WTFMove(m_entries.last()));
109 m_entries.removeLast();
110 }
111
112 // Toss the first item if the list is getting too big, as long as we're not using it
113 // (or even if we are, if we only want 1 entry).
114 if (m_entries.size() >= DefaultCapacity && (*m_currentIndex)) {
115 didRemoveItem(m_entries[0]);
116 removedItems.append(WTFMove(m_entries[0]));
117 m_entries.remove(0);
118
119 if (m_entries.isEmpty())
120 m_currentIndex = std::nullopt;
121 else
122 --*m_currentIndex;
123 }
124 } else {
125 // If we have no current item index we should also not have any entries.
126 ASSERT(m_entries.isEmpty());
127
128 // But just in case it does happen in practice we'll get back in to a consistent state now before adding the new item.
129 size_t size = m_entries.size();
130 for (size_t i = 0; i < size; ++i) {
131 didRemoveItem(m_entries[i]);
132 removedItems.append(WTFMove(m_entries[i]));
133 }
134 m_entries.clear();
135 }
136
137 bool shouldKeepCurrentItem = true;
138
139 if (!m_currentIndex) {
140 ASSERT(m_entries.isEmpty());
141 m_currentIndex = 0;
142 } else {
143 shouldKeepCurrentItem = m_page->shouldKeepCurrentBackForwardListItemInList(m_entries[*m_currentIndex]);
144 if (shouldKeepCurrentItem)
145 ++*m_currentIndex;
146 }
147
148 auto* newItemPtr = newItem.ptr();
149 if (!shouldKeepCurrentItem) {
150 // m_current should never be pointing past the end of the entries Vector.
151 // If it is, something has gone wrong and we should not try to swap in the new item.
152 ASSERT(*m_currentIndex < m_entries.size());
153
154 removedItems.append(m_entries[*m_currentIndex].copyRef());
155 m_entries[*m_currentIndex] = WTFMove(newItem);
156 } else {
157 // m_current should never be pointing more than 1 past the end of the entries Vector.
158 // If it is, something has gone wrong and we should not try to insert the new item.
159 ASSERT(*m_currentIndex <= m_entries.size());
160
161 if (*m_currentIndex <= m_entries.size())
162 m_entries.insert(*m_currentIndex, WTFMove(newItem));
163 }
164
165 LOG(BackForward, "(Back/Forward) WebBackForwardList %p added an item. Current size %zu, current index %zu, threw away %zu items", this, m_entries.size(), *m_currentIndex, removedItems.size());
166 m_page->didChangeBackForwardList(newItemPtr, WTFMove(removedItems));
167}

169void WebBackForwardList::goToItem(WebBackForwardListItem& item)
170{
171 ASSERT(!m_currentIndex || *m_currentIndex < m_entries.size());
172
173 if (!m_entries.size() || !m_page || !m_currentIndex)
174 return;
175
176 size_t targetIndex = notFound;
177 for (size_t i = 0; i < m_entries.size(); ++i) {
178 if (m_entries[i].ptr() == &item) {
179 targetIndex = i;
180 break;
181 }
182 }
183
184 // If the target item wasn't even in the list, there's nothing else to do.
185 if (targetIndex == notFound) {
186 LOG(BackForward, "(Back/Forward) WebBackForwardList %p could not go to item %s (%s) because it was not found", this, item.itemID().logString(), item.url().utf8().data());
187 return;
188 }
189
190 if (targetIndex < *m_currentIndex) {
191 unsigned delta = m_entries.size() - targetIndex - 1;
192 String deltaValue = delta > 10 ? "over10"_s : String::number(delta);
193 m_page->logDiagnosticMessage(WebCore::DiagnosticLoggingKeys::backNavigationDeltaKey(), deltaValue, ShouldSample::No);
194 }
195
196 // If we're going to an item different from the current item, ask the client if the current
197 // item should remain in the list.
198 auto& currentItem = m_entries[*m_currentIndex];
199 bool shouldKeepCurrentItem = true;
200 if (currentItem.ptr() != &item) {
201 m_page->recordAutomaticNavigationSnapshot();
202 shouldKeepCurrentItem = m_page->shouldKeepCurrentBackForwardListItemInList(m_entries[*m_currentIndex]);
203 }
204
205 // If the client said to remove the current item, remove it and then update the target index.
206 Vector<Ref<WebBackForwardListItem>> removedItems;
207 if (!shouldKeepCurrentItem) {
208 removedItems.append(currentItem.copyRef());
209 m_entries.remove(*m_currentIndex);
210 targetIndex = notFound;
211 for (size_t i = 0; i < m_entries.size(); ++i) {
212 if (m_entries[i].ptr() == &item) {
213 targetIndex = i;
214 break;
215 }
216 }
217 ASSERT(targetIndex != notFound);
218 }
219
220 m_currentIndex = targetIndex;
221
222 LOG(BackForward, "(Back/Forward) WebBackForwardList %p going to item %s, is now at index %zu", this, item.itemID().logString(), targetIndex);
223 m_page->didChangeBackForwardList(nullptr, WTFMove(removedItems));
224}

247WebBackForwardListItem* WebBackForwardList::itemAtIndex(int index) const
248{
249 ASSERT(!m_currentIndex || *m_currentIndex < m_entries.size());
250
251 if (!m_currentIndex || !m_page)
252 return nullptr;
253
254 // Do range checks without doing math on index to avoid overflow.
255 if (index < 0 && static_cast<unsigned>(-index) > backListCount())
256 return nullptr;
257
258 if (index > 0 && static_cast<unsigned>(index) > forwardListCount())
259 return nullptr;
260
261 return m_entries[index + *m_currentIndex].ptr();
262}

293Ref<API::Array> WebBackForwardList::backListAsAPIArrayWithLimit(unsigned limit) const
294{
295 ASSERT(!m_currentIndex || *m_currentIndex < m_entries.size());
296
297 if (!m_page || !m_currentIndex)
298 return API::Array::create();
299
300 unsigned backListSize = static_cast<unsigned>(backListCount());
301 unsigned size = std::min(backListSize, limit);
302 if (!size)
303 return API::Array::create();
304
305 Vector<RefPtr<API::Object>> vector;
306 vector.reserveInitialCapacity(size);
307
308 ASSERT(backListSize >= size);
309 for (unsigned i = backListSize - size; i < backListSize; ++i)
310 vector.uncheckedAppend(m_entries[i].ptr());
311
312 return API::Array::create(WTFMove(vector));
313}

315Ref<API::Array> WebBackForwardList::forwardListAsAPIArrayWithLimit(unsigned limit) const
316{
317 ASSERT(!m_currentIndex || *m_currentIndex < m_entries.size());
318
319 if (!m_page || !m_currentIndex)
320 return API::Array::create();
321
322 unsigned size = std::min(static_cast<unsigned>(forwardListCount()), limit);
323 if (!size)
324 return API::Array::create();
325
326 Vector<RefPtr<API::Object>> vector;
327 vector.reserveInitialCapacity(size);
328
329 size_t last = *m_currentIndex + size;
330 ASSERT(last < m_entries.size());
331 for (size_t i = *m_currentIndex + 1; i <= last; ++i)
332 vector.uncheckedAppend(m_entries[i].ptr());
333
334 return API::Array::create(WTFMove(vector));
335}

337void WebBackForwardList::removeAllItems()
338{
339 ASSERT(!m_currentIndex || *m_currentIndex < m_entries.size());
340
341 LOG(BackForward, "(Back/Forward) WebBackForwardList %p removeAllItems (has %zu of them)", this, m_entries.size());
342
343 Vector<Ref<WebBackForwardListItem>> removedItems;
344
345 for (auto& entry : m_entries) {
346 didRemoveItem(entry);
347 removedItems.append(WTFMove(entry));
348 }
349
350 m_entries.clear();
351 m_currentIndex = std::nullopt;
352 m_page->didChangeBackForwardList(nullptr, WTFMove(removedItems));
353}

355void WebBackForwardList::clear()
356{
357 ASSERT(!m_currentIndex || *m_currentIndex < m_entries.size());
358
359 LOG(BackForward, "(Back/Forward) WebBackForwardList %p clear (has %zu of them)", this, m_entries.size());
360
361 size_t size = m_entries.size();
362 if (!m_page || size <= 1)
363 return;
364
365 RefPtr<WebBackForwardListItem> currentItem = this->currentItem();
366 Vector<Ref<WebBackForwardListItem>> removedItems;
367
368 if (!currentItem) {
369 // We should only ever have no current item if we also have no current item index.
370 ASSERT(!m_currentIndex);
371
372 // But just in case it does happen in practice we should get back into a consistent state now.
373 for (size_t i = 0; i < size; ++i) {
374 didRemoveItem(m_entries[i]);
375 removedItems.append(WTFMove(m_entries[i]));
376 }
377
378 m_entries.clear();
379 m_currentIndex = std::nullopt;
380 m_page->didChangeBackForwardList(nullptr, WTFMove(removedItems));
381
382 return;
383 }
384
385 for (size_t i = 0; i < size; ++i) {
386 if (m_entries[i].ptr() != currentItem)
387 didRemoveItem(m_entries[i]);
388 }
389
390 removedItems.reserveCapacity(size - 1);
391 for (size_t i = 0; i < size; ++i) {
392 if (m_currentIndex && i != *m_currentIndex)
393 removedItems.append(WTFMove(m_entries[i]));
394 }
395
396 m_currentIndex = 0;
397
398 m_entries.clear();
399 if (currentItem)
400 m_entries.append(currentItem.releaseNonNull());
401 else
402 m_currentIndex = std::nullopt;
403 m_page->didChangeBackForwardList(nullptr, WTFMove(removedItems));
404}

406BackForwardListState WebBackForwardList::backForwardListState(WTF::Function<bool (WebBackForwardListItem&)>&& filter) const
407{
408 ASSERT(!m_currentIndex || *m_currentIndex < m_entries.size());
409
410 BackForwardListState backForwardListState;
411 if (m_currentIndex)
412 backForwardListState.currentIndex = *m_currentIndex;
413
414 for (size_t i = 0; i < m_entries.size(); ++i) {
415 auto& entry = m_entries[i];
416
417 if (filter && !filter(entry)) {
418 auto& currentIndex = backForwardListState.currentIndex;
419 if (currentIndex && i <= currentIndex.value() && currentIndex.value())
420 --currentIndex.value();
421
422 continue;
423 }
424
425 backForwardListState.items.append(entry->itemState());
426 }
427
428 if (backForwardListState.items.isEmpty())
429 backForwardListState.currentIndex = std::nullopt;
430 else if (backForwardListState.items.size() <= backForwardListState.currentIndex.value())
431 backForwardListState.currentIndex = backForwardListState.items.size() - 1;
432
433 return backForwardListState;
434}

436void WebBackForwardList::restoreFromState(BackForwardListState backForwardListState)
437{
438 if (!m_page)
439 return;
440
441 Vector<Ref<WebBackForwardListItem>> items;
442 items.reserveInitialCapacity(backForwardListState.items.size());
443
444 // FIXME: Enable restoring resourceDirectoryURL.
445 for (auto& backForwardListItemState : backForwardListState.items) {
446 backForwardListItemState.identifier = { Process::identifier(), ObjectIdentifier<BackForwardItemIdentifier::ItemIdentifierType>::generate() };
447 items.uncheckedAppend(WebBackForwardListItem::create(WTFMove(backForwardListItemState), m_page->identifier()));
448 }
449 m_currentIndex = backForwardListState.currentIndex ? std::optional<size_t>(*backForwardListState.currentIndex) : std::nullopt;
450 m_entries = WTFMove(items);
451
452 LOG(BackForward, "(Back/Forward) WebBackForwardList %p restored from state (has %zu entries)", this, m_entries.size());
453}

455Vector<BackForwardListItemState> WebBackForwardList::filteredItemStates(Function<bool(WebBackForwardListItem&)>&& functor) const
456{
457 Vector<BackForwardListItemState> itemStates;
458 itemStates.reserveInitialCapacity(m_entries.size());
459
460 for (const auto& entry : m_entries) {
461 if (functor(entry))
462 itemStates.uncheckedAppend(entry->itemState());
463 }
464
465 return itemStates;
466}

475void WebBackForwardList::didRemoveItem(WebBackForwardListItem& backForwardListItem)
476{
477 backForwardListItem.wasRemovedFromBackForwardList();
478
479 m_page->backForwardRemovedItem(backForwardListItem.itemID());
480
481#if PLATFORM(COCOA) || PLATFORM(GTK)
482 backForwardListItem.setSnapshot(nullptr);
483#endif
484}

488const char* WebBackForwardList::loggingString()
489{
490 StringBuilder builder;
491
492 builder.append("WebBackForwardList 0x", hex(reinterpret_cast<uintptr_t>(this)), " - ", m_entries.size(), " entries, has current index ", m_currentIndex ? "YES" : "NO", " (", m_currentIndex ? *m_currentIndex : 0, ')');
493
494 for (size_t i = 0; i < m_entries.size(); ++i) {
495 const char* prefix;
496 if (m_currentIndex && *m_currentIndex == i)
497 prefix = " * ";
498 else
499 prefix = " - ";
500 builder.append('\n', prefix, m_entries[i]->loggingString());
501 }
502
503 return debugString("\n", builder.toString());
504}
