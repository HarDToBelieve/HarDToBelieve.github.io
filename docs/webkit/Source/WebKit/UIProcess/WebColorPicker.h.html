28#if ENABLE(INPUT_TYPE_COLOR)

30#include <wtf/RefCounted.h>
31#include <wtf/RefPtr.h>
32
33namespace WebCore {
34class Color;
35}
36
37namespace WebKit {
38
39class WebPageProxy;
40
41class WebColorPicker : public RefCounted<WebColorPicker> {
42public:
43 class Client {
44 protected:
45 virtual ~Client() { }
46
47 public:
48 virtual void didChooseColor(const WebCore::Color&) = 0;
49 virtual void didEndColorPicker() = 0;
50 };
51
52 static Ref<WebColorPicker> create(Client* client)
53 {
54 return adoptRef(*new WebColorPicker(client));
55 }
56
57 virtual ~WebColorPicker();
58
59 virtual void endPicker();
60 virtual void setSelectedColor(const WebCore::Color&);
61 virtual void showColorPicker(const WebCore::Color&);
62
63protected:
64 explicit WebColorPicker(Client*);
65
66 Client* m_client;
67};

69} // namespace WebKit
