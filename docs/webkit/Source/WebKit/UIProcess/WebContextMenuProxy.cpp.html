38WebContextMenuProxy::WebContextMenuProxy(WebPageProxy& page, ContextMenuContextData&& context, const UserData& userData)
39 : m_context(WTFMove(context))
40 , m_userData(userData)
41 , m_page(makeWeakPtr(page))
42{
43}
44
45WebContextMenuProxy::~WebContextMenuProxy() = default;
46
47Vector<Ref<WebContextMenuItem>> WebContextMenuProxy::proposedItems() const
48{
49 return WTF::map(m_context.menuItems(), [](auto& item) {
50 return WebContextMenuItem::create(item);
51 });
52}
53
54void WebContextMenuProxy::show()
55{
56 m_contextMenuListener = WebContextMenuListenerProxy::create(*this);
57 page()->contextMenuClient().getContextMenuFromProposedMenu(*page(), proposedItems(), *m_contextMenuListener, m_context.webHitTestResultData(), page()->process().transformHandlesToObjects(m_userData.object()).get());
58}
59
60void WebContextMenuProxy::useContextMenuItems(Vector<Ref<WebContextMenuItem>>&& items)
61{
62 m_contextMenuListener = nullptr;
63
64 auto page = makeRefPtr(this->page());
65 if (!page)
66 return;
67
68 // Since showContextMenuWithItems can spin a nested run loop we need to turn off the responsiveness timer.
69 page->process().stopResponsivenessTimer();
70
71 // Protect |this| from being deallocated if WebPageProxy code is re-entered from the menu runloop or delegates.
72 auto protectedThis = makeRef(*this);
73 showContextMenuWithItems(WTFMove(items));
74
75 // No matter the result of showContextMenuWithItems, always notify the WebProcess that the menu is hidden so it starts handling mouse events again.
76 page->send(Messages::WebPage::ContextMenuHidden());
77}
