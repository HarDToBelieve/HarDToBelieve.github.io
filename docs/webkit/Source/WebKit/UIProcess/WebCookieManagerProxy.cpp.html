43WebCookieManagerProxy::WebCookieManagerProxy(NetworkProcessProxy& networkProcess)
44 : m_networkProcess(makeWeakPtr(networkProcess))
45{
46 networkProcess.addMessageReceiver(Messages::WebCookieManagerProxy::messageReceiverName(), *this);
47}
48
49WebCookieManagerProxy::~WebCookieManagerProxy()
50{
51 if (m_networkProcess)
52 m_networkProcess->removeMessageReceiver(Messages::WebCookieManagerProxy::messageReceiverName());
53 ASSERT(m_cookieObservers.isEmpty());
54}

136void WebCookieManagerProxy::registerObserver(PAL::SessionID sessionID, Observer& observer)
137{
138 auto result = m_cookieObservers.set(sessionID, WeakHashSet<Observer>());
139 result.iterator->value.add(&observer);
140
141 if (result.isNewEntry)
142 startObservingCookieChanges(sessionID);
143}
144
145void WebCookieManagerProxy::unregisterObserver(PAL::SessionID sessionID, Observer& observer)
146{
147 auto iterator = m_cookieObservers.find(sessionID);
148 if (iterator == m_cookieObservers.end())
149 return;
150
151 iterator->value.remove(observer);
152 if (!iterator->value.computesEmpty())
153 return;
154
155 m_cookieObservers.remove(iterator);
156 stopObservingCookieChanges(sessionID);
157}
158
159void WebCookieManagerProxy::cookiesDidChange(PAL::SessionID sessionID)
160{
161 auto iterator = m_cookieObservers.find(sessionID);
162 if (iterator == m_cookieObservers.end())
163 return;
164
165 for (auto& observer : iterator->value)
166 observer.cookiesDidChange();
167}
