42void WebFormClient::willSubmitForm(WebPageProxy& page, WebFrameProxy& frame, WebFrameProxy& sourceFrame, const Vector<std::pair<String, String>>& textFieldValues, API::Object* userData, WTF::Function<void(void)>&& completionHandler)
43{
44 if (!m_client.willSubmitForm) {
45 completionHandler();
46 return;
47 }
48
49 API::Dictionary::MapType map;
50 for (size_t i = 0; i < textFieldValues.size(); ++i)
51 map.set(textFieldValues[i].first, API::String::create(textFieldValues[i].second));
52 auto textFieldsMap = API::Dictionary::create(WTFMove(map));
53 auto listener = WebFormSubmissionListenerProxy::create(WTFMove(completionHandler));
54 m_client.willSubmitForm(toAPI(&page), toAPI(&frame), toAPI(&sourceFrame), toAPI(textFieldsMap.ptr()), toAPI(userData), toAPI(listener.ptr()), m_client.base.clientInfo);
55}
