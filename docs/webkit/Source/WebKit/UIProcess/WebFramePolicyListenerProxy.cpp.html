38WebFramePolicyListenerProxy::WebFramePolicyListenerProxy(Reply&& reply, ShouldExpectSafeBrowsingResult expectSafeBrowsingResult, ShouldExpectAppBoundDomainResult expectAppBoundDomainResult)
39 : m_reply(WTFMove(reply))
40{
41 if (expectSafeBrowsingResult == ShouldExpectSafeBrowsingResult::No)
42 didReceiveSafeBrowsingResults({ });
43 if (expectAppBoundDomainResult == ShouldExpectAppBoundDomainResult::No)
44 didReceiveAppBoundDomainResult({ });
45}
46
47WebFramePolicyListenerProxy::~WebFramePolicyListenerProxy() = default;
48
49void WebFramePolicyListenerProxy::didReceiveAppBoundDomainResult(std::optional<NavigatingToAppBoundDomain> isNavigatingToAppBoundDomain)
50{
51 ASSERT(RunLoop::isMain());
52
53 if (m_policyResult && m_safeBrowsingWarning) {
54 if (m_reply)
55 m_reply(WebCore::PolicyAction::Use, m_policyResult->first.get(), m_policyResult->second, WTFMove(*m_safeBrowsingWarning), isNavigatingToAppBoundDomain);
56 } else
57 m_isNavigatingToAppBoundDomain = isNavigatingToAppBoundDomain;
58}
59
60void WebFramePolicyListenerProxy::didReceiveSafeBrowsingResults(RefPtr<SafeBrowsingWarning>&& safeBrowsingWarning)
61{
62 ASSERT(isMainRunLoop());
63 ASSERT(!m_safeBrowsingWarning);
64 if (m_policyResult && m_isNavigatingToAppBoundDomain) {
65 if (m_reply)
66 m_reply(WebCore::PolicyAction::Use, m_policyResult->first.get(), m_policyResult->second, WTFMove(safeBrowsingWarning), *m_isNavigatingToAppBoundDomain);
67 } else
68 m_safeBrowsingWarning = WTFMove(safeBrowsingWarning);
69}
70
71void WebFramePolicyListenerProxy::use(API::WebsitePolicies* policies, ProcessSwapRequestedByClient processSwapRequestedByClient)
72{
73 if (m_safeBrowsingWarning && m_isNavigatingToAppBoundDomain) {
74 if (m_reply)
75 m_reply(WebCore::PolicyAction::Use, policies, processSwapRequestedByClient, WTFMove(*m_safeBrowsingWarning), *m_isNavigatingToAppBoundDomain);
76 } else if (!m_policyResult)
77 m_policyResult = {{ policies, processSwapRequestedByClient }};
78}
79
80void WebFramePolicyListenerProxy::download()
81{
82 if (m_reply)
83 m_reply(WebCore::PolicyAction::Download, nullptr, ProcessSwapRequestedByClient::No, { }, { });
84}
85
86void WebFramePolicyListenerProxy::ignore()
87{
88 if (m_reply)
89 m_reply(WebCore::PolicyAction::Ignore, nullptr, ProcessSwapRequestedByClient::No, { }, { });
90}
