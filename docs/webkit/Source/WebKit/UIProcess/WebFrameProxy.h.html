60class WebFrameProxy : public API::ObjectImpl<API::Object::Type::Frame> {
61public:
62 static Ref<WebFrameProxy> create(WebPageProxy& page, WebCore::FrameIdentifier frameID)
63 {
64 return adoptRef(*new WebFrameProxy(page, frameID));
65 }
66
67 virtual ~WebFrameProxy();
68
69 WebCore::FrameIdentifier frameID() const { return m_frameID; }
70 WebPageProxy* page() const { return m_page.get(); }
71
72 bool pageIsClosed() const { return !m_page; } // Needs to be thread-safe.
73
74 void webProcessWillShutDown();
75
76 bool isMainFrame() const;
77
78 FrameLoadState& frameLoadState() { return m_frameLoadState; }
79
80 void loadURL(const URL&, const String& referrer = String());
81 // Sub frames only. For main frames, use WebPageProxy::loadData.
82 void loadData(const IPC::DataReference&, const String& MIMEType, const String& encodingName, const URL& baseURL);
83 void stopLoading() const;
84
85 const URL& url() const { return m_frameLoadState.url(); }
86 const URL& provisionalURL() const { return m_frameLoadState.provisionalURL(); }
87
88 void setUnreachableURL(const URL&);
89 const URL& unreachableURL() const { return m_frameLoadState.unreachableURL(); }
90
91 const String& mimeType() const { return m_MIMEType; }
92 bool containsPluginDocument() const { return m_containsPluginDocument; }
93
94 const String& title() const { return m_title; }
95
96 WebCertificateInfo* certificateInfo() const { return m_certificateInfo.get(); }
97
98 bool canProvideSource() const;
99 bool canShowMIMEType(const String& mimeType) const;
100
101 bool isDisplayingStandaloneImageDocument() const;
102 bool isDisplayingStandaloneMediaDocument() const;
103 bool isDisplayingMarkupDocument() const;
104 bool isDisplayingPDFDocument() const;
105
106 void getWebArchive(CompletionHandler<void(API::Data*)>&&);
107 void getMainResourceData(CompletionHandler<void(API::Data*)>&&);
108 void getResourceData(API::URL*, CompletionHandler<void(API::Data*)>&&);
109
110 void didStartProvisionalLoad(const URL&);
111 void didExplicitOpen(URL&&, String&& mimeType);
112 void didReceiveServerRedirectForProvisionalLoad(const URL&);
113 void didFailProvisionalLoad();
114 void didCommitLoad(const String& contentType, WebCertificateInfo&, bool containsPluginDocument);
115 void didFinishLoad();
116 void didFailLoad();
117 void didSameDocumentNavigation(const URL&); // eg. anchor navigation, session state change.
118 void didChangeTitle(const String&);
119
120 WebFramePolicyListenerProxy& setUpPolicyListenerProxy(CompletionHandler<void(WebCore::PolicyAction, API::WebsitePolicies*, ProcessSwapRequestedByClient, RefPtr<SafeBrowsingWarning>&&, std::optional<NavigatingToAppBoundDomain>)>&&, ShouldExpectSafeBrowsingResult, ShouldExpectAppBoundDomainResult);
121
122#if ENABLE(CONTENT_FILTERING)
123 void contentFilterDidBlockLoad(WebCore::ContentFilterUnblockHandler contentFilterUnblockHandler) { m_contentFilterUnblockHandler = WTFMove(contentFilterUnblockHandler); }
124 bool didHandleContentFilterUnblockNavigation(const WebCore::ResourceRequest&);
125#endif
126
127#if PLATFORM(GTK)
128 void collapseSelection();
129#endif
130
131private:
132 WebFrameProxy(WebPageProxy&, WebCore::FrameIdentifier);
133
134 WeakPtr<WebPageProxy> m_page;
135
136 FrameLoadState m_frameLoadState;
137
138 String m_MIMEType;
139 String m_title;
140 bool m_containsPluginDocument { false };
141 RefPtr<WebCertificateInfo> m_certificateInfo;
142 RefPtr<WebFramePolicyListenerProxy> m_activeListener;
143 WebCore::FrameIdentifier m_frameID;
144#if ENABLE(CONTENT_FILTERING)
145 WebCore::ContentFilterUnblockHandler m_contentFilterUnblockHandler;
146#endif
147};
