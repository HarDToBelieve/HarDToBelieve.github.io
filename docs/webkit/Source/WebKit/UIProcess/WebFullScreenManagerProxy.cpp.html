43WebFullScreenManagerProxy::WebFullScreenManagerProxy(WebPageProxy& page, WebFullScreenManagerProxyClient& client)
44 : m_page(page)
45 , m_client(client)
46{
47 m_page.process().addMessageReceiver(Messages::WebFullScreenManagerProxy::messageReceiverName(), m_page.webPageID(), *this);
48}
49
50WebFullScreenManagerProxy::~WebFullScreenManagerProxy()
51{
52 m_page.process().removeMessageReceiver(Messages::WebFullScreenManagerProxy::messageReceiverName(), m_page.webPageID());
53 m_client.closeFullScreenManager();
54 callCloseCompletionHandlers();
55}
56
57void WebFullScreenManagerProxy::willEnterFullScreen()
58{
59 m_page.fullscreenClient().willEnterFullscreen(&m_page);
60 m_page.send(Messages::WebFullScreenManager::WillEnterFullScreen());
61}
62
63void WebFullScreenManagerProxy::didEnterFullScreen()
64{
65 m_page.fullscreenClient().didEnterFullscreen(&m_page);
66 m_page.send(Messages::WebFullScreenManager::DidEnterFullScreen());
67
68 if (m_page.isControlledByAutomation()) {
69 if (WebAutomationSession* automationSession = m_page.process().processPool().automationSession())
70 automationSession->didEnterFullScreenForPage(m_page);
71 }
72}
73
74void WebFullScreenManagerProxy::willExitFullScreen()
75{
76 m_page.fullscreenClient().willExitFullscreen(&m_page);
77 m_page.send(Messages::WebFullScreenManager::WillExitFullScreen());
78}
79
80void WebFullScreenManagerProxy::callCloseCompletionHandlers()
81{
82 auto closeMediaCallbacks = WTFMove(m_closeCompletionHandlers);
83 for (auto& callback : closeMediaCallbacks)
84 callback();
85}
86
87void WebFullScreenManagerProxy::closeWithCallback(CompletionHandler<void()>&& completionHandler)
88{
89 m_closeCompletionHandlers.append(WTFMove(completionHandler));
90 close();
91}
92
93void WebFullScreenManagerProxy::didExitFullScreen()
94{
95 m_page.fullscreenClient().didExitFullscreen(&m_page);
96 m_page.send(Messages::WebFullScreenManager::DidExitFullScreen());
97
98 if (m_page.isControlledByAutomation()) {
99 if (WebAutomationSession* automationSession = m_page.process().processPool().automationSession())
100 automationSession->didExitFullScreenForPage(m_page);
101 }
102 callCloseCompletionHandlers();
103}
