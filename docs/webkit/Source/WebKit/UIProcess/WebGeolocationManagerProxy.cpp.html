68void WebGeolocationManagerProxy::processPoolDestroyed()
69{
70 bool wasUpdating = isUpdating();
71 m_updateRequesters.clear();
72
73 ASSERT(!isUpdating());
74 if (wasUpdating)
75 m_provider->stopUpdating(*this);
76}

93void WebGeolocationManagerProxy::providerDidChangePosition(WebGeolocationPosition* position)
94{
95 m_lastPosition = position->corePosition();
96
97 if (!processPool())
98 return;
99
100 processPool()->sendToAllProcesses(Messages::WebGeolocationManager::DidChangePosition(m_lastPosition.value()));
101}

118void WebGeolocationManagerProxy::startUpdating(IPC::Connection& connection, WebPageProxyIdentifier pageProxyID, const String& authorizationToken)
119{
120 auto* page = WebProcessProxy::webPage(pageProxyID);
121 MESSAGE_CHECK(page);
122
123 auto isValidAuthorizationToken = page->geolocationPermissionRequestManager().isValidAuthorizationToken(authorizationToken);
124 MESSAGE_CHECK(isValidAuthorizationToken);
125
126 bool wasUpdating = isUpdating();
127 m_updateRequesters.add(&connection.client());
128 if (!wasUpdating) {
129 m_provider->setEnableHighAccuracy(*this, isHighAccuracyEnabled());
130 m_provider->startUpdating(*this);
131 } else if (m_lastPosition)
132 connection.send(Messages::WebGeolocationManager::DidChangePosition(m_lastPosition.value()), 0);
133}
134
135void WebGeolocationManagerProxy::stopUpdating(IPC::Connection& connection)
136{
137 removeRequester(&connection.client());
138}
139
140void WebGeolocationManagerProxy::removeRequester(const IPC::Connection::Client* client)
141{
142 bool wasUpdating = isUpdating();
143 bool highAccuracyWasEnabled = isHighAccuracyEnabled();
144
145 m_highAccuracyRequesters.remove(client);
146 m_updateRequesters.remove(client);
147
148 if (wasUpdating && !isUpdating())
149 m_provider->stopUpdating(*this);
150 else {
151 bool highAccuracyShouldBeEnabled = isHighAccuracyEnabled();
152 if (highAccuracyShouldBeEnabled != highAccuracyWasEnabled)
153 m_provider->setEnableHighAccuracy(*this, highAccuracyShouldBeEnabled);
154 }
155}
156
157void WebGeolocationManagerProxy::setEnableHighAccuracy(IPC::Connection& connection, bool enabled)
158{
159 bool highAccuracyWasEnabled = isHighAccuracyEnabled();
160
161 if (enabled)
162 m_highAccuracyRequesters.add(&connection.client());
163 else
164 m_highAccuracyRequesters.remove(&connection.client());
165
166 bool highAccuracyShouldBeEnabled = isHighAccuracyEnabled();
167 if (isUpdating() && highAccuracyWasEnabled != highAccuracyShouldBeEnabled)
168 m_provider->setEnableHighAccuracy(*this, highAccuracyShouldBeEnabled);
169}
