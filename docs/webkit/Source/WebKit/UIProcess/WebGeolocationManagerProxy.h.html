46class WebGeolocationManagerProxy : public API::ObjectImpl<API::Object::Type::GeolocationManager>, public WebContextSupplement, private IPC::MessageReceiver {

48 static const char* supplementName();
49
50 static Ref<WebGeolocationManagerProxy> create(WebProcessPool*);
51
52 void setProvider(std::unique_ptr<API::GeolocationProvider>&&);
53
54 void providerDidChangePosition(WebGeolocationPosition*);

61 using API::Object::ref;

63
64private:

66
67 // WebContextSupplement
68 void processPoolDestroyed() override;
69 void processDidClose(WebProcessProxy*) override;

74 void didReceiveMessage(IPC::Connection&, IPC::Decoder&) override;

76 bool isUpdating() const { return !m_updateRequesters.isEmpty(); }

78
79 void startUpdating(IPC::Connection&, WebPageProxyIdentifier, const String& authorizationToken);
80 void stopUpdating(IPC::Connection&);
81 void removeRequester(const IPC::Connection::Client*);
82 void setEnableHighAccuracy(IPC::Connection&, bool);
83
84 HashSet<const IPC::Connection::Client*> m_updateRequesters;
85 HashSet<const IPC::Connection::Client*> m_highAccuracyRequesters;

87 std::unique_ptr<API::GeolocationProvider> m_provider;
88 std::optional<WebCore::GeolocationPositionData> m_lastPosition;
89};
