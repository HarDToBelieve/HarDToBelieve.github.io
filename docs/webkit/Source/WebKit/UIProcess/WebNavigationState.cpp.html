44Ref<API::Navigation> WebNavigationState::createLoadRequestNavigation(ResourceRequest&& request, WebBackForwardListItem* currentItem)
45{
46 auto navigation = API::Navigation::create(*this, WTFMove(request), currentItem);
47
48 m_navigations.set(navigation->navigationID(), navigation.ptr());
49
50 return navigation;
51}

89API::Navigation* WebNavigationState::navigation(uint64_t navigationID)
90{
91 ASSERT(navigationID);
92 ASSERT(m_navigations.contains(navigationID));
93
94 return m_navigations.get(navigationID);
95}
96
97RefPtr<API::Navigation> WebNavigationState::takeNavigation(uint64_t navigationID)
98{
99 ASSERT(navigationID);
100 ASSERT(m_navigations.contains(navigationID));
101
102 return m_navigations.take(navigationID);
103}

105void WebNavigationState::didDestroyNavigation(uint64_t navigationID)
106{
107 ASSERT(navigationID);
108
109 m_navigations.remove(navigationID);
