43static uint64_t generatePageGroupID()
44{
45 static uint64_t uniquePageGroupID = 1;
46 return uniquePageGroupID++;
47}
48
49typedef HashMap<uint64_t, WebPageGroup*> WebPageGroupMap;
50
51static WebPageGroupMap& webPageGroupMap()
52{
53 static NeverDestroyed<WebPageGroupMap> map;
54 return map;
55}
56
57Ref<WebPageGroup> WebPageGroup::create(const String& identifier)
58{
59 return adoptRef(*new WebPageGroup(identifier));
60}
61
62WebPageGroup* WebPageGroup::get(uint64_t pageGroupID)
63{
64 return webPageGroupMap().get(pageGroupID);
65}
66
67void WebPageGroup::forEach(Function<void(WebPageGroup&)>&& function)
68{
69 auto allGroups = copyToVectorOf<RefPtr<WebPageGroup>>(webPageGroupMap().values());
70 for (auto& group : allGroups) {
71 if (group)
72 function(*group);
73 }
74}
75
76static WebPageGroupData pageGroupData(const String& identifier)
77{
78 WebPageGroupData data;
79
80 static NeverDestroyed<HashMap<String, uint64_t>> map;
81 if (HashMap<String, uint64_t>::isValidKey(identifier)) {
82 data.pageGroupID = map.get().ensure(identifier, [] {
83 return generatePageGroupID();
84 }).iterator->value;
85 } else
86 data.pageGroupID = generatePageGroupID();
87
88 if (!identifier.isEmpty())
89 data.identifier = identifier;
90 else
91 data.identifier = makeString("__uniquePageGroupID-", data.pageGroupID);
92
93 return data;
94}

96// FIXME: Why does the WebPreferences object here use ".WebKit2" instead of "WebKit2." which all the other constructors use.
97// If it turns out that it's wrong, we can change it to to "WebKit2." and get rid of the globalDebugKeyPrefix from WebPreferences.
98WebPageGroup::WebPageGroup(const String& identifier)
99 : m_data(pageGroupData(identifier))
100 , m_preferences(WebPreferences::createWithLegacyDefaults(m_data.identifier, ".WebKit2", "WebKit2."))
101 , m_userContentController(WebUserContentControllerProxy::create())
102{
103 webPageGroupMap().set(m_data.pageGroupID, this);
104}
105
106WebPageGroup::~WebPageGroup()
107{
108 webPageGroupMap().remove(pageGroupID());
109}

121void WebPageGroup::setPreferences(WebPreferences* preferences)
122{
123 if (preferences == m_preferences)
124 return;
125
126 m_preferences = preferences;
127
128 for (auto& webPageProxy : m_pages)
129 webPageProxy.setPreferences(*m_preferences);
130}
131
132WebPreferences& WebPageGroup::preferences() const
133{
134 return *m_preferences;
135}
