45 return adoptRef(*new WebPreferences(identifier, keyPrefix, globalDebugKeyPrefix));
46}

48Ref<WebPreferences> WebPreferences::createWithLegacyDefaults(const String& identifier, const String& keyPrefix, const String& globalDebugKeyPrefix)
49{
50 auto preferences = WebPreferences::create(identifier, keyPrefix, globalDebugKeyPrefix);
51 // FIXME: The registerDefault...ValueForKey machinery is unnecessarily heavyweight and complicated.
52 // We can just compute different defaults for modern and legacy APIs in WebPreferencesDefinitions.h macros.
53 preferences->registerDefaultBoolValueForKey(WebPreferencesKey::javaEnabledKey(), true);
54 preferences->registerDefaultBoolValueForKey(WebPreferencesKey::javaEnabledForLocalFilesKey(), true);
55 preferences->registerDefaultBoolValueForKey(WebPreferencesKey::pluginsEnabledKey(), true);
56 preferences->registerDefaultUInt32ValueForKey(WebPreferencesKey::storageBlockingPolicyKey(), static_cast<uint32_t>(WebCore::StorageBlockingPolicy::AllowAll));
57 return preferences;
58}

99void WebPreferences::update()
100{
101 if (m_updateBatchCount) {
102 m_needUpdateAfterBatch = true;
103 return;
104 }
105
106 for (auto& webPageProxy : m_pages)
107 webPageProxy.preferencesDidChange();
108}
109
110void WebPreferences::startBatchingUpdates()
111{
112 if (!m_updateBatchCount)
113 m_needUpdateAfterBatch = false;
114
115 ++m_updateBatchCount;
116}
117
118void WebPreferences::endBatchingUpdates()
119{
120 ASSERT(m_updateBatchCount > 0);
121 --m_updateBatchCount;
122 if (!m_updateBatchCount && m_needUpdateAfterBatch)
123 update();
124}

166void WebPreferences::updateBoolValueForInternalDebugFeatureKey(const String& key, bool value)
167{
168 if (key == WebPreferencesKey::processSwapOnCrossSiteNavigationEnabledKey()) {
169 for (auto& page : m_pages)
170 page.process().processPool().configuration().setProcessSwapsOnNavigation(value);
171
172 return;
173 }
174 update(); // FIXME: Only send over the changed key and value.
175}

207void WebPreferences::registerDefaultBoolValueForKey(const String& key, bool value)
208{
209 m_store.setOverrideDefaultsBoolValueForKey(key, value);
210 bool userValue;
211 if (platformGetBoolUserValueForKey(key, userValue))
212 m_store.setBoolValueForKey(key, userValue);
213}
