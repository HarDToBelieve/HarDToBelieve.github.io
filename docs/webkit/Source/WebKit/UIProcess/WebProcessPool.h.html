796template<typename T>
797void WebProcessPool::sendToAllProcesses(const T& message)
798{
799 for (auto& process : m_processes) {
800 if (process->canSendMessage())
801 process->send(T(message), 0);
802 }

805template<typename T>
806void WebProcessPool::sendToAllProcessesForSession(const T& message, PAL::SessionID sessionID)
807{
808 for (auto& process : m_processes) {
809 if (process->canSendMessage() && !process->isPrewarmed() && process->sessionID() == sessionID)
810 process->send(T(message), 0);
811 }
