46class WebURLSchemeHandler : public RefCounted<WebURLSchemeHandler> {

48public:
49 virtual ~WebURLSchemeHandler();

58protected:

61private:

64 virtual void platformTaskCompleted(WebURLSchemeTask&) { };
65
66 void removeTaskFromPageMap(WebPageProxyIdentifier, uint64_t taskID);

69 uint64_t m_identifier;

71 HashMap<uint64_t, Ref<WebURLSchemeTask>> m_tasks;
