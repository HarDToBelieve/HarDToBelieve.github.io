70auto WebURLSchemeTask::willPerformRedirection(ResourceResponse&& response, ResourceRequest&& request, Function<void(ResourceRequest&&)>&& completionHandler) -> ExceptionType
71{
72 ASSERT(RunLoop::isMain());
73
74 if (m_stopped)
75 return m_shouldSuppressTaskStoppedExceptions ? ExceptionType::None : ExceptionType::TaskAlreadyStopped;
76
77 if (m_completed)
78 return ExceptionType::CompleteAlreadyCalled;
79
80 if (m_dataSent)
81 return ExceptionType::DataAlreadySent;
82
83 if (m_responseSent)
84 return ExceptionType::RedirectAfterResponse;
85
86 if (m_waitingForRedirectCompletionHandlerCallback)
87 return ExceptionType::WaitingForRedirectCompletionHandler;
88
89 if (isSync())
90 m_syncResponse = response;
91
92 {
93 Locker locker { m_requestLock };
94 m_request = request;
95 }
96
97 auto* page = WebProcessProxy::webPage(m_pageProxyID);
98 if (!page)
99 return ExceptionType::None;
100
101 m_waitingForRedirectCompletionHandlerCallback = true;
102
103 Function<void(ResourceRequest&&)> innerCompletionHandler = [protectedThis = makeRef(*this), this, completionHandler = WTFMove(completionHandler)] (ResourceRequest&& request) mutable {
104 m_waitingForRedirectCompletionHandlerCallback = false;
105 if (!m_stopped && !m_completed)
106 completionHandler(WTFMove(request));
107 };
108
109 page->sendWithAsyncReply(Messages::WebPage::URLSchemeTaskWillPerformRedirection(m_urlSchemeHandler->identifier(), m_identifier, response, request), WTFMove(innerCompletionHandler));
110
111 return ExceptionType::None;
112}

171auto WebURLSchemeTask::didReceiveData(Ref<SharedBuffer>&& buffer) -> ExceptionType
172{
173 ASSERT(RunLoop::isMain());
174
175 if (m_stopped)
176 return m_shouldSuppressTaskStoppedExceptions ? ExceptionType::None : ExceptionType::TaskAlreadyStopped;
177
178 if (m_completed)
179 return ExceptionType::CompleteAlreadyCalled;
180
181 if (!m_responseSent)
182 return ExceptionType::NoResponseSent;
183
184 if (m_waitingForRedirectCompletionHandlerCallback)
185 return ExceptionType::WaitingForRedirectCompletionHandler;
186
187 m_dataSent = true;
188
189 if (isSync()) {
190 if (m_syncData)
191 m_syncData->append(WTFMove(buffer));
192 else
193 m_syncData = WTFMove(buffer);
194 return ExceptionType::None;
195 }
196
197 m_process->send(Messages::WebPage::URLSchemeTaskDidReceiveData(m_urlSchemeHandler->identifier(), m_identifier, buffer.get()), m_webPageID);
198 return ExceptionType::None;
199}

201auto WebURLSchemeTask::didComplete(const ResourceError& error) -> ExceptionType
202{
203 ASSERT(RunLoop::isMain());
204
205 if (m_stopped)
206 return m_shouldSuppressTaskStoppedExceptions ? ExceptionType::None : ExceptionType::TaskAlreadyStopped;
207
208 if (m_completed)
209 return ExceptionType::CompleteAlreadyCalled;
210
211 if (!m_responseSent && error.isNull())
212 return ExceptionType::NoResponseSent;
213
214 if (m_waitingForRedirectCompletionHandlerCallback && error.isNull())
215 return ExceptionType::WaitingForRedirectCompletionHandler;
216
217 m_completed = true;
218
219 if (isSync()) {
220 Vector<uint8_t> data;
221 if (m_syncData)
222 data = { m_syncData->data(), m_syncData->size() };
223
224 m_syncCompletionHandler(m_syncResponse, error, WTFMove(data));
225 m_syncData = nullptr;
226 }
227
228 m_process->send(Messages::WebPage::URLSchemeTaskDidComplete(m_urlSchemeHandler->identifier(), m_identifier, error), m_webPageID);
229 m_urlSchemeHandler->taskCompleted(*this);
230
231 return ExceptionType::None;
232}

234void WebURLSchemeTask::pageDestroyed()
235{
236 ASSERT(RunLoop::isMain());
237
238 m_pageProxyID = { };
239 m_webPageID = { };
240 m_process = nullptr;
241 m_stopped = true;
242
243 if (isSync()) {
244 Locker locker { m_requestLock };
245 m_syncCompletionHandler({ }, failedCustomProtocolSyncLoad(m_request), { });
246 }
247}

249void WebURLSchemeTask::stop()

254 m_stopped = true;
255
256 if (isSync()) {
257 Locker locker { m_requestLock };
258 m_syncCompletionHandler({ }, failedCustomProtocolSyncLoad(m_request), { });
259 }
260}
