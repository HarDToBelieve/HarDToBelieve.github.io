91void WebsiteDataStore::platformSetNetworkParameters(WebsiteDataStoreParameters& parameters)
92{
93 ASSERT(hasProcessPrivilege(ProcessPrivilege::CanAccessRawCookies));
94
95 NSUserDefaults *defaults = [NSUserDefaults standardUserDefaults];
96 bool shouldLogCookieInformation = false;
97 bool enableResourceLoadStatisticsDebugMode = false;
98 auto sameSiteStrictEnforcementEnabled = WebCore::SameSiteStrictEnforcementEnabled::No;
99 auto firstPartyWebsiteDataRemovalMode = WebCore::FirstPartyWebsiteDataRemovalMode::AllButCookies;
100 WebCore::RegistrableDomain resourceLoadStatisticsManualPrevalentResource { };
101#if ENABLE(RESOURCE_LOAD_STATISTICS)
102 enableResourceLoadStatisticsDebugMode = [defaults boolForKey:@"ITPDebugMode"];
103
104 if ([defaults boolForKey:[NSString stringWithFormat:@"Experimental%@", WebPreferencesKey::isSameSiteStrictEnforcementEnabledKey().createCFString().get()]])
105 sameSiteStrictEnforcementEnabled = WebCore::SameSiteStrictEnforcementEnabled::Yes;
106
107 if ([defaults boolForKey:[NSString stringWithFormat:@"Experimental%@", WebPreferencesKey::isFirstPartyWebsiteDataRemovalDisabledKey().createCFString().get()]])
108 firstPartyWebsiteDataRemovalMode = WebCore::FirstPartyWebsiteDataRemovalMode::None;
109 else {
110 if ([defaults boolForKey:[NSString stringWithFormat:@"InternalDebug%@", WebPreferencesKey::isFirstPartyWebsiteDataRemovalReproTestingEnabledKey().createCFString().get()]])
111 firstPartyWebsiteDataRemovalMode = WebCore::FirstPartyWebsiteDataRemovalMode::AllButCookiesReproTestingTimeout;
112 else if ([defaults boolForKey:[NSString stringWithFormat:@"InternalDebug%@", WebPreferencesKey::isFirstPartyWebsiteDataRemovalLiveOnTestingEnabledKey().createCFString().get()]])
113 firstPartyWebsiteDataRemovalMode = WebCore::FirstPartyWebsiteDataRemovalMode::AllButCookiesLiveOnTestingTimeout;
114 else
115 firstPartyWebsiteDataRemovalMode = WebCore::FirstPartyWebsiteDataRemovalMode::AllButCookies;
116 }
117
118 if (auto manualPrevalentResource = [defaults stringForKey:@"ITPManualPrevalentResource"]) {
119 URL url { { }, manualPrevalentResource };
120 if (!url.isValid())
121 url = { { }, makeString("http://", manualPrevalentResource) };
122 if (url.isValid())
123 resourceLoadStatisticsManualPrevalentResource = WebCore::RegistrableDomain { url };
124 }
125#if !RELEASE_LOG_DISABLED
126 static NSString * const WebKitLogCookieInformationDefaultsKey = @"WebKitLogCookieInformation";
127 shouldLogCookieInformation = [defaults boolForKey:WebKitLogCookieInformationDefaultsKey];
128#endif
129#endif // ENABLE(RESOURCE_LOAD_STATISTICS)
130
131 URL httpProxy = m_configuration->httpProxy();
132 URL httpsProxy = m_configuration->httpsProxy();
133
134 bool isSafari = false;
135 bool isMiniBrowser = false;
136#if PLATFORM(IOS_FAMILY)
137 isSafari = WebCore::IOSApplication::isMobileSafari();
138 isMiniBrowser = WebCore::IOSApplication::isMiniBrowser();
139#elif PLATFORM(MAC)
140 isSafari = WebCore::MacApplication::isSafari();
141 isMiniBrowser = WebCore::MacApplication::isMiniBrowser();
142#endif
143 // FIXME: Remove these once Safari adopts _WKWebsiteDataStoreConfiguration.httpProxy and .httpsProxy.
144 if (!httpProxy.isValid() && (isSafari || isMiniBrowser))
145 httpProxy = URL(URL(), [defaults stringForKey:(NSString *)WebKit2HTTPProxyDefaultsKey]);
146 if (!httpsProxy.isValid() && (isSafari || isMiniBrowser))
147 httpsProxy = URL(URL(), [defaults stringForKey:(NSString *)WebKit2HTTPSProxyDefaultsKey]);
148
149#if HAVE(CFNETWORK_ALTERNATIVE_SERVICE)
150 bool http3Enabled = WebsiteDataStore::http3Enabled();
151 SandboxExtension::Handle alternativeServiceStorageDirectoryExtensionHandle;
152 String alternativeServiceStorageDirectory = resolvedAlternativeServicesStorageDirectory();
153 if (!alternativeServiceStorageDirectory.isEmpty())
154 SandboxExtension::createHandleForReadWriteDirectory(alternativeServiceStorageDirectory, alternativeServiceStorageDirectoryExtensionHandle);
155#endif
156
157 bool shouldIncludeLocalhostInResourceLoadStatistics = isSafari;
158
159 parameters.networkSessionParameters.proxyConfiguration = configuration().proxyConfiguration();
160 parameters.networkSessionParameters.sourceApplicationBundleIdentifier = configuration().sourceApplicationBundleIdentifier();
161 parameters.networkSessionParameters.sourceApplicationSecondaryIdentifier = configuration().sourceApplicationSecondaryIdentifier();
162 parameters.networkSessionParameters.shouldLogCookieInformation = shouldLogCookieInformation;
163 parameters.networkSessionParameters.httpProxy = WTFMove(httpProxy);
164 parameters.networkSessionParameters.httpsProxy = WTFMove(httpsProxy);
165#if HAVE(CFNETWORK_ALTERNATIVE_SERVICE)
166 parameters.networkSessionParameters.alternativeServiceDirectory = WTFMove(alternativeServiceStorageDirectory);
167 parameters.networkSessionParameters.alternativeServiceDirectoryExtensionHandle = WTFMove(alternativeServiceStorageDirectoryExtensionHandle);
168 parameters.networkSessionParameters.http3Enabled = WTFMove(http3Enabled);
169#endif
170 parameters.networkSessionParameters.resourceLoadStatisticsParameters.shouldIncludeLocalhost = shouldIncludeLocalhostInResourceLoadStatistics;
171 parameters.networkSessionParameters.resourceLoadStatisticsParameters.enableDebugMode = enableResourceLoadStatisticsDebugMode;
172 parameters.networkSessionParameters.resourceLoadStatisticsParameters.sameSiteStrictEnforcementEnabled = sameSiteStrictEnforcementEnabled;
173 parameters.networkSessionParameters.resourceLoadStatisticsParameters.firstPartyWebsiteDataRemovalMode = firstPartyWebsiteDataRemovalMode;
174 parameters.networkSessionParameters.resourceLoadStatisticsParameters.standaloneApplicationDomain = WebCore::RegistrableDomain { m_configuration->standaloneApplicationURL() };
175 parameters.networkSessionParameters.resourceLoadStatisticsParameters.manualPrevalentResource = WTFMove(resourceLoadStatisticsManualPrevalentResource);
176
177 auto cookieFile = resolvedCookieStorageFile();
178
179 if (m_uiProcessCookieStorageIdentifier.isEmpty()) {
180 auto utf8File = cookieFile.utf8();
181 auto url = adoptCF(CFURLCreateFromFileSystemRepresentation(kCFAllocatorDefault, (const UInt8 *)utf8File.data(), (CFIndex)utf8File.length(), true));
182 m_cfCookieStorage = adoptCF(CFHTTPCookieStorageCreateFromFile(kCFAllocatorDefault, url.get(), nullptr));
183 m_uiProcessCookieStorageIdentifier = identifyingDataFromCookieStorage(m_cfCookieStorage.get());
184 }
185
186 parameters.uiProcessCookieStorageIdentifier = m_uiProcessCookieStorageIdentifier;
187
188 if (!cookieFile.isEmpty())
189 SandboxExtension::createHandleForReadWriteDirectory(FileSystem::parentPath(cookieFile), parameters.cookieStoragePathExtensionHandle);
190}

331WTF::String WebsiteDataStore::tempDirectoryFileSystemRepresentation(const WTF::String& directoryName, ShouldCreateDirectory shouldCreateDirectory)
332{
333 static dispatch_once_t onceToken;
334 static NeverDestroyed<RetainPtr<NSURL>> tempURL;
335
336 dispatch_once(&onceToken, ^{
337 NSURL *url = [NSURL fileURLWithPath:NSTemporaryDirectory() isDirectory:YES];
338 if (!url)
339 RELEASE_ASSERT_NOT_REACHED();
340
341 if (!WebKit::processHasContainer()) {
342 NSString *bundleIdentifier = [NSBundle mainBundle].bundleIdentifier;
343 if (!bundleIdentifier)
344 bundleIdentifier = [NSProcessInfo processInfo].processName;
345 url = [url URLByAppendingPathComponent:bundleIdentifier isDirectory:YES];
346 }
347
348 tempURL.get() = [url URLByAppendingPathComponent:@"WebKit" isDirectory:YES];
349 });
350
351 NSURL *url = [tempURL.get() URLByAppendingPathComponent:directoryName isDirectory:YES];
352
353 if (shouldCreateDirectory == ShouldCreateDirectory::Yes
354 && (![[NSFileManager defaultManager] createDirectoryAtURL:url withIntermediateDirectories:YES attributes:nil error:nullptr]))
355 LOG_ERROR("Failed to create directory %@", url);
356
357 return url.absoluteURL.path.fileSystemRepresentation;
358}

432void WebsiteDataStore::initializeAppBoundDomains(ForceReinitialization forceReinitialization)
433{
434 ASSERT(RunLoop::isMain());
435
436 if (hasInitializedAppBoundDomains && forceReinitialization != ForceReinitialization::Yes)
437 return;
438
439 static const auto maxAppBoundDomainCount = 10;
440
441 appBoundDomainQueue().dispatch([forceReinitialization] () mutable {
442 if (hasInitializedAppBoundDomains && forceReinitialization != ForceReinitialization::Yes)
443 return;
444
445 NSArray<NSString *> *appBoundData = [[NSBundle mainBundle] objectForInfoDictionaryKey:@"WKAppBoundDomains"];
446 keyExists = appBoundData ? true : false;
447
448 RunLoop::main().dispatch([forceReinitialization, appBoundData = retainPtr(appBoundData)] {
449 if (hasInitializedAppBoundDomains && forceReinitialization != ForceReinitialization::Yes)
450 return;
451
452 if (forceReinitialization == ForceReinitialization::Yes)
453 appBoundDomains().clear();
454
455 for (NSString *data in appBoundData.get()) {
456 if (appBoundDomains().size() + appBoundSchemes().size() >= maxAppBoundDomainCount)
457 break;
458 if ([data hasSuffix:@":"]) {
459 auto appBoundScheme = String([data substringToIndex:[data length] - 1]);
460 if (!appBoundScheme.isEmpty()) {
461 appBoundSchemes().add(appBoundScheme);
462 continue;
463 }
464 }
465
466 URL url { URL(), data };
467 if (url.protocol().isEmpty())
468 url.setProtocol("https");
469 if (!url.isValid())
470 continue;
471 WebCore::RegistrableDomain appBoundDomain { url };
472 if (appBoundDomain.isEmpty())
473 continue;
474 appBoundDomains().add(appBoundDomain);
475 }
476 hasInitializedAppBoundDomains = true;
477 if (isAppBoundITPRelaxationEnabled)
478 forwardAppBoundDomainsToITPIfInitialized([] { });
479 });
480 });
481}

495void WebsiteDataStore::ensureAppBoundDomains(CompletionHandler<void(const HashSet<WebCore::RegistrableDomain>&, const HashSet<String>&)>&& completionHandler) const
496{
497 if (hasInitializedAppBoundDomains) {
498 if (m_configuration->enableInAppBrowserPrivacyForTesting())
499 addTestDomains();
500 completionHandler(appBoundDomains(), appBoundSchemes());
501 return;
502 }
503
504 // Hopping to the background thread then back to the main thread
505 // ensures that initializeAppBoundDomains() has finished.
506 appBoundDomainQueue().dispatch([this, protectedThis = makeRef(*this), completionHandler = WTFMove(completionHandler)] () mutable {
507 RunLoop::main().dispatch([this, protectedThis = WTFMove(protectedThis), completionHandler = WTFMove(completionHandler)] () mutable {
508 ASSERT(hasInitializedAppBoundDomains);
509 if (m_configuration->enableInAppBrowserPrivacyForTesting())
510 addTestDomains();
511 completionHandler(appBoundDomains(), appBoundSchemes());
512 });
513 });
514}

516static NavigatingToAppBoundDomain schemeOrDomainIsAppBound(const String& host, const String& protocol, const HashSet<WebCore::RegistrableDomain>& domains, const HashSet<String>& schemes)
517{
518 auto schemeIsAppBound = !protocol.isNull() && schemes.contains(protocol);
519 auto domainIsAppBound = domains.contains(WebCore::RegistrableDomain::uncheckedCreateFromHost(host));
520 return schemeIsAppBound || domainIsAppBound ? NavigatingToAppBoundDomain::Yes : NavigatingToAppBoundDomain::No;
521}

523void WebsiteDataStore::beginAppBoundDomainCheck(const String& host, const String& protocol, WebFramePolicyListenerProxy& listener)
524{
525 ASSERT(RunLoop::isMain());
526
527 ensureAppBoundDomains([&host, &protocol, listener = makeRef(listener)] (auto& domains, auto& schemes) mutable {
528 // Must check for both an empty app bound domains list and an empty key before returning nullopt
529 // because test cases may have app bound domains but no key.
530 bool hasAppBoundDomains = keyExists || !domains.isEmpty();
531 if (!hasAppBoundDomains) {
532 listener->didReceiveAppBoundDomainResult(std::nullopt);
533 return;
534 }
535 listener->didReceiveAppBoundDomainResult(schemeOrDomainIsAppBound(host, protocol, domains, schemes));
536 });
537}

587void WebsiteDataStore::sendNetworkProcessXPCEndpointToWebProcess(WebProcessProxy& process)
588{
589 if (process.state() != AuxiliaryProcessProxy::State::Running)
590 return;
591 auto* connection = process.connection();
592 if (!connection)
593 return;
594 auto message = networkProcess().xpcEndpointMessage();
595 if (!message)
596 return;
597 xpc_connection_send_message(connection->xpcConnection(), message);
598}
