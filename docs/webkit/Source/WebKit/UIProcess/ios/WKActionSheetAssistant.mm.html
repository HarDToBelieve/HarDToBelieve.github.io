156- (UIView *)superviewForSheet
157{
158 UIView *view = _view.getAutoreleased();
159 UIView *superview = [view window];
160
161 // FIXME: WebKit has a delegate to retrieve the superview for the image sheet (superviewForImageSheetForWebView)
162 // Do we need it in WK2?
163
164 // Find the top most view with a view controller
165 UIViewController *controller = nil;
166 UIView *currentView = view;
167 while (currentView) {
168 UIViewController *aController = [UIViewController viewControllerForView:currentView];
169 if (aController)
170 controller = aController;
171
172 currentView = [currentView superview];
173 }
174 if (controller)
175 superview = controller.view;
176
177 return superview;
178}

210static const CGFloat presentationElementRectPadding = 15;

212- (CGRect)presentationRectForElementUsingClosestIndicatedRect
213{
214 UIView *view = [self superviewForSheet];
215 auto delegate = _delegate.get();
216 if (!view || !delegate || !_positionInformation)
217 return CGRectZero;
218
219 auto indicator = _positionInformation->linkIndicator;
220 if (indicator.textRectsInBoundingRectCoordinates.isEmpty())
221 return CGRectZero;
222
223 WebCore::FloatPoint touchLocation = _positionInformation->request.point;
224 WebCore::FloatPoint linkElementLocation = indicator.textBoundingRectInRootViewCoordinates.location();
225 Vector<WebCore::FloatRect> indicatedRects;
226 for (auto rect : indicator.textRectsInBoundingRectCoordinates) {
227 rect.inflate(2);
228 rect.moveBy(linkElementLocation);
229 indicatedRects.append(rect);
230 }
231
232 for (const auto& path : WebCore::PathUtilities::pathsWithShrinkWrappedRects(indicatedRects, 0)) {
233 auto boundingRect = path.fastBoundingRect();
234 if (boundingRect.contains(touchLocation))
235 return CGRectInset([view convertRect:(CGRect)boundingRect fromView:_view.getAutoreleased()], -presentationElementRectPadding, -presentationElementRectPadding);
236 }
237
238 return CGRectZero;
239}

262- (CGRect)presentationRectInHostViewForSheet
263{
264 UIView *view = [self superviewForSheet];
265 auto delegate = _delegate.get();
266 if (!view || !delegate || !_positionInformation)
267 return CGRectZero;
268
269 CGRect boundingRect = _positionInformation->bounds;
270 CGPoint fromPoint = _positionInformation->request.point;
271
272 // FIXME: We must adjust our presentation point to take into account a change in document scale.
273
274 // Test to see if we are still within the target node as it may have moved after rotation.
275 if (!CGRectContainsPoint(boundingRect, fromPoint))
276 fromPoint = CGPointMake(CGRectGetMidX(boundingRect), CGRectGetMidY(boundingRect));
277
278 return [self _presentationRectForSheetGivenPoint:fromPoint inHostView:view];
279}

343- (void)_createSheetWithElementActions:(NSArray *)actions defaultTitle:(NSString *)defaultTitle showLinkTitle:(BOOL)showLinkTitle
344{
345 auto delegate = _delegate.get();
346 if (!delegate)
347 return;
348
349 if (!_positionInformation)
350 return;
351
352 NSURL *targetURL = _positionInformation->url;
353 // FIXME: We should check if Javascript is enabled in the preferences.
354
355 _interactionSheet = adoptNS([[WKActionSheet alloc] init]);
356 _interactionSheet.get().sheetDelegate = self;
357 _interactionSheet.get().preferredStyle = UIAlertControllerStyleActionSheet;
358
359 NSString *titleString = nil;
360 BOOL titleIsURL = NO;
361 if (showLinkTitle && [[targetURL absoluteString] length]) {
362 if (isJavaScriptURL(targetURL))
363 titleString = WEB_UI_STRING_KEY("JavaScript", "JavaScript Action Sheet Title", "Title for action sheet for JavaScript link");
364 else {
365 titleString = WTF::userVisibleString(targetURL);
366 titleIsURL = YES;
367 }
368 } else if (defaultTitle)
369 titleString = defaultTitle;
370 else
371 titleString = _positionInformation->title;
372
373 if ([titleString length]) {
374 [_interactionSheet setTitle:titleString];
375 // We should configure the text field's line breaking mode correctly here, based on whether
376 // the title is an URL or not, but the appropriate UIAlertController SPIs are not available yet.
377 // The code that used to do this in the UIActionSheet world has been saved for reference in
378 // <rdar://problem/17049781> Configure the UIAlertController's title appropriately.
379 }
380
381 for (_WKElementAction *action in actions) {
382 [_interactionSheet _addActionWithTitle:[action title] style:UIAlertActionStyleDefault handler:^{
383 [action _runActionWithElementInfo:_elementInfo.get() forActionSheetAssistant:self];
384 [self cleanupSheet];
385 } shouldDismissHandler:^{
386 return (BOOL)(!action.dismissalHandler || action.dismissalHandler());
387 }];
388 }
389
390 [_interactionSheet addAction:[UIAlertAction actionWithTitle:WEB_UI_STRING_KEY("Cancel", "Cancel button label in button bar", "Title for Cancel button label in button bar")
391 style:UIAlertActionStyleCancel
392 handler:^(UIAlertAction *action) {
393 [self cleanupSheet];
394 }]];
395
396 if ([delegate respondsToSelector:@selector(actionSheetAssistant:willStartInteractionWithElement:)])
397 [delegate actionSheetAssistant:self willStartInteractionWithElement:_elementInfo.get()];
398}

400- (void)showImageSheet
401{
402 ASSERT(!_elementInfo);
403
404 auto delegate = _delegate.get();
405 if (!delegate)
406 return;
407
408 if (![self synchronouslyRetrievePositionInformation])
409 return;
410
411 void (^showImageSheetWithAlternateURLBlock)(NSURL*, NSDictionary *userInfo) = ^(NSURL *alternateURL, NSDictionary *userInfo) {
412 NSURL *targetURL = _positionInformation->url;
413 NSURL *imageURL = _positionInformation->imageURL;
414 if (!targetURL)
415 targetURL = alternateURL;
416 auto elementBounds = _positionInformation->bounds;
417 auto elementInfo = adoptNS([[_WKActivatedElementInfo alloc] _initWithType:_WKActivatedElementTypeImage URL:targetURL imageURL:imageURL location:_positionInformation->request.point title:_positionInformation->title ID:_positionInformation->idAttribute rect:elementBounds image:_positionInformation->image.get() userInfo:userInfo]);
418 if ([delegate respondsToSelector:@selector(actionSheetAssistant:showCustomSheetForElement:)] && [delegate actionSheetAssistant:self showCustomSheetForElement:elementInfo.get()])
419 return;
420 auto defaultActions = [self defaultActionsForImageSheet:elementInfo.get()];
421
422 RetainPtr<NSArray> actions = [delegate actionSheetAssistant:self decideActionsForElement:elementInfo.get() defaultActions:WTFMove(defaultActions)];
423
424 if (![actions count])
425 return;
426
427 if (!alternateURL && userInfo) {
428ALLOW_DEPRECATED_DECLARATIONS_BEGIN
429 [UIApp _cancelAllTouches];
430ALLOW_DEPRECATED_DECLARATIONS_END
431 return;
432 }
433
434 [self _createSheetWithElementActions:actions.get() defaultTitle:nil showLinkTitle:YES];
435 if (!_interactionSheet)
436 return;
437
438 _elementInfo = WTFMove(elementInfo);
439
440 if (![_interactionSheet presentSheet:[self _presentationStyleForPositionInfo:_positionInformation.value() elementInfo:_elementInfo.get()]])
441 [self cleanupSheet];
442 };
443
444 if (_positionInformation->url.isEmpty() && _positionInformation->image && [delegate respondsToSelector:@selector(actionSheetAssistant:getAlternateURLForImage:completion:)]) {
445 RetainPtr<UIImage> uiImage = adoptNS([[UIImage alloc] initWithCGImage:_positionInformation->image->makeCGImageCopy().get()]);
446
447 _hasPendingActionSheet = YES;
448 RetainPtr<WKActionSheetAssistant> retainedSelf(self);
449 [delegate actionSheetAssistant:self getAlternateURLForImage:uiImage.get() completion:^(NSURL *alternateURL, NSDictionary *userInfo) {
450 if (!retainedSelf->_hasPendingActionSheet)
451 return;
452
453 retainedSelf->_hasPendingActionSheet = NO;
454 showImageSheetWithAlternateURLBlock(alternateURL, userInfo);
455 }];
456 return;
457 }
458
459 showImageSheetWithAlternateURLBlock(nil, nil);
460}

462- (WKActionSheetPresentationStyle)_presentationStyleForPositionInfo:(const WebKit::InteractionInformationAtPosition&)positionInfo elementInfo:(_WKActivatedElementInfo *)elementInfo
463{
464 auto apparentElementRect = [_view convertRect:positionInfo.bounds toView:[_view window]];
465 if (CGRectIsEmpty(apparentElementRect))
466 return WKActionSheetPresentAtTouchLocation;
467
468 CGRect visibleRect;
469 auto delegate = _delegate.get();
470 if ([delegate respondsToSelector:@selector(unoccludedWindowBoundsForActionSheetAssistant:)])
471 visibleRect = [delegate unoccludedWindowBoundsForActionSheetAssistant:self];
472 else
473 visibleRect = [[_view window] bounds];
474
475 apparentElementRect = CGRectIntersection(apparentElementRect, visibleRect);
476 auto leftInset = CGRectGetMinX(apparentElementRect) - CGRectGetMinX(visibleRect);
477 auto topInset = CGRectGetMinY(apparentElementRect) - CGRectGetMinY(visibleRect);
478 auto rightInset = CGRectGetMaxX(visibleRect) - CGRectGetMaxX(apparentElementRect);
479 auto bottomInset = CGRectGetMaxY(visibleRect) - CGRectGetMaxY(apparentElementRect);
480
481 // If at least this much of the window is available for the popover to draw in, then target the element rect when presenting the action menu popover.
482 // Otherwise, there is not enough space to position the popover around the element, so revert to using the touch location instead.
483 static const CGFloat minimumAvailableWidthOrHeightRatio = 0.4;
484 if (std::max(leftInset, rightInset) <= minimumAvailableWidthOrHeightRatio * CGRectGetWidth(visibleRect) && std::max(topInset, bottomInset) <= minimumAvailableWidthOrHeightRatio * CGRectGetHeight(visibleRect))
485 return WKActionSheetPresentAtTouchLocation;
486
487 if (elementInfo.type == _WKActivatedElementTypeLink && positionInfo.linkIndicator.textRectsInBoundingRectCoordinates.size())
488 return WKActionSheetPresentAtClosestIndicatorRect;
489
490 return WKActionSheetPresentAtElementRect;
491}

618- (void)showLinkSheet
619{
620 ASSERT(!_elementInfo);
621 if (!_delegate)
622 return;
623
624 _needsLinkIndicator = YES;
625 if (![self synchronouslyRetrievePositionInformation])
626 return;
627
628 NSURL *targetURL = _positionInformation->url;
629 if (!targetURL) {
630 _needsLinkIndicator = NO;

1000- (void)handleElementActionWithType:(_WKElementActionType)type element:(_WKActivatedElementInfo *)element needsInteraction:(BOOL)needsInteraction
1001{
1002 auto delegate = _delegate.get();
1003
1004 if (needsInteraction && [delegate respondsToSelector:@selector(actionSheetAssistant:willStartInteractionWithElement:)])
1005 [delegate actionSheetAssistant:self willStartInteractionWithElement:element];
1006
1007 switch (type) {
1008 case _WKElementActionTypeCopy:
1009 [delegate actionSheetAssistant:self performAction:WebKit::SheetAction::Copy];
1010 break;
1011 case _WKElementActionTypeOpen:
1012 [delegate actionSheetAssistant:self openElementAtLocation:element._interactionLocation];
1013 break;
1014 case _WKElementActionTypeSaveImage:
1015 [delegate actionSheetAssistant:self performAction:WebKit::SheetAction::SaveImage];
1016 break;
1017 case _WKElementActionTypeShare:
1018 if (URL(element.imageURL).protocolIsData() && element.image && [delegate respondsToSelector:@selector(actionSheetAssistant:shareElementWithImage:rect:)])
1019 [delegate actionSheetAssistant:self shareElementWithImage:element.image rect:element.boundingRect];
1020 else {
1021 auto urlToShare = element.URL && !isJavaScriptURL(element.URL) ? element.URL : element.imageURL;
1022 [delegate actionSheetAssistant:self shareElementWithURL:urlToShare rect:element.boundingRect];
1023 }
1024 break;
1025 case _WKElementActionTypeImageExtraction:
1026#if ENABLE(IMAGE_ANALYSIS)
1027 [delegate actionSheetAssistant:self showTextForImage:element.image imageURL:element.imageURL title:element.title imageBounds:element.boundingRect];
1028#endif
1029 break;
1030 case _WKElementActionTypeRevealImage:
1031#if ENABLE(IMAGE_ANALYSIS)
1032 [delegate actionSheetAssistant:self lookUpImage:element.image imageURL:element.imageURL title:element.title imageBounds:element.boundingRect];
1033#endif
1034 break;
1035 default:
1036 ASSERT_NOT_REACHED();
1037 break;
1038 }
1039
1040 if (needsInteraction && [delegate respondsToSelector:@selector(actionSheetAssistantDidStopInteraction:)])
1041 [delegate actionSheetAssistantDidStopInteraction:self];
