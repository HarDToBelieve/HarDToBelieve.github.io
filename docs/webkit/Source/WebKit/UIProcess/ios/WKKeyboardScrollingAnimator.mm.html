118static WebCore::FloatSize perpendicularAbsoluteUnitVector(WebCore::ScrollDirection direction)
119{
120 switch (direction) {
121 case WebCore::ScrollDirection::ScrollUp:
122 case WebCore::ScrollDirection::ScrollDown:
123 return { 1, 0 };
124 case WebCore::ScrollDirection::ScrollLeft:
125 case WebCore::ScrollDirection::ScrollRight:
126 return { 0, 1 };
127 }
128}

130static WebCore::BoxSide boxSide(WebCore::ScrollDirection direction)
131{
132 switch (direction) {
133 case WebCore::ScrollDirection::ScrollUp:
134 return WebCore::BoxSide::Top;
135 case WebCore::ScrollDirection::ScrollDown:
136 return WebCore::BoxSide::Bottom;
137 case WebCore::ScrollDirection::ScrollLeft:
138 return WebCore::BoxSide::Left;
139 case WebCore::ScrollDirection::ScrollRight:
140 return WebCore::BoxSide::Right;

144- (std::optional<WebCore::KeyboardScroll>)keyboardScrollForEvent:(::WebEvent *)event
145{
146 static const unsigned kWebSpaceKey = 0x20;
147
148 if (![_scrollable isKeyboardScrollable])
149 return std::nullopt;
150
151 if (event.keyboardFlags & WebEventKeyboardInputModifierFlagsChanged)
152 return std::nullopt;
153
154 NSString *charactersIgnoringModifiers = event.charactersIgnoringModifiers;
155 if (!charactersIgnoringModifiers.length)
156 return std::nullopt;
157
158 enum class Key : uint8_t { Other, LeftArrow, RightArrow, UpArrow, DownArrow, PageUp, PageDown, Space };
159
160 auto key = ^{
161 auto firstCharacter = [charactersIgnoringModifiers characterAtIndex:0];
162 switch (firstCharacter) {
163 case NSLeftArrowFunctionKey:
164 return Key::LeftArrow;
165 case NSRightArrowFunctionKey:
166 return Key::RightArrow;
167 case NSUpArrowFunctionKey:
168 return Key::UpArrow;
169 case NSDownArrowFunctionKey:
170 return Key::DownArrow;
171 case NSPageDownFunctionKey:
172 return Key::PageDown;
173 case NSPageUpFunctionKey:
174 return Key::PageUp;
175 case kWebSpaceKey:
176 return Key::Space;
177 default:
178 return Key::Other;
179 };
180 }();
181
182 if (key == Key::Other)
183 return std::nullopt;
184
185 BOOL shiftPressed = event.modifierFlags & WebEventFlagMaskShiftKey;
186 BOOL altPressed = event.modifierFlags & WebEventFlagMaskOptionKey;
187 BOOL cmdPressed = event.modifierFlags & WebEventFlagMaskCommandKey;
188
189 // No shortcuts include more than one modifier; we should not eat key events
190 // that contain more than one modifier because they might be used for other shortcuts.
191 if (shiftPressed + altPressed + cmdPressed > 1)
192 return std::nullopt;
193
194 auto allowedModifiers = ^ WebEventFlags {
195 switch (key) {
196 case Key::LeftArrow:
197 case Key::RightArrow:
198 return WebEventFlagMaskOptionKey;
199 case Key::UpArrow:
200 case Key::DownArrow:
201 return WebEventFlagMaskOptionKey | WebEventFlagMaskCommandKey;
202 case Key::PageUp:
203 case Key::PageDown:
204 return 0;
205 case Key::Space:
206 return WebEventFlagMaskShiftKey;
207 case Key::Other:
208 ASSERT_NOT_REACHED();
209 return 0;
210 };
211 }();
212
213 auto relevantModifierFlags = WebEventFlagMaskOptionKey | WebEventFlagMaskCommandKey | WebEventFlagMaskShiftKey;
214 if (event.modifierFlags & relevantModifierFlags & ~allowedModifiers)
215 return std::nullopt;
216
217 auto increment = ^{
218 switch (key) {
219 case Key::LeftArrow:
220 case Key::RightArrow:
221 if (altPressed)
222 return WebCore::ScrollGranularity::ScrollByPage;
223 return WebCore::ScrollGranularity::ScrollByLine;
224 case Key::UpArrow:
225 case Key::DownArrow:
226 if (altPressed)
227 return WebCore::ScrollGranularity::ScrollByPage;
228 if (cmdPressed)
229 return WebCore::ScrollGranularity::ScrollByDocument;
230 return WebCore::ScrollGranularity::ScrollByLine;
231 case Key::PageUp:
232 case Key::PageDown:
233 case Key::Space:
234 return WebCore::ScrollGranularity::ScrollByPage;
235 case Key::Other:
236 ASSERT_NOT_REACHED();
237 return WebCore::ScrollGranularity::ScrollByLine;
238 };
239 }();
240
241 auto direction = ^() {
242 switch (key) {
243 case Key::LeftArrow:
244 return WebCore::ScrollDirection::ScrollLeft;
245 case Key::RightArrow:
246 return WebCore::ScrollDirection::ScrollRight;
247 case Key::UpArrow:
248 case Key::PageUp:
249 return WebCore::ScrollDirection::ScrollUp;
250 case Key::DownArrow:
251 case Key::PageDown:
252 return WebCore::ScrollDirection::ScrollDown;
253 case Key::Space:
254 return shiftPressed ? WebCore::ScrollDirection::ScrollUp : WebCore::ScrollDirection::ScrollDown;
255 case Key::Other:
256 ASSERT_NOT_REACHED();
257 return WebCore::ScrollDirection::ScrollDown;
258 };
259 }();
260
261 // FIXME (227461): Replace with call to WebCore::KeyboardScroll constructor.
262
263 CGFloat scrollDistance = [_scrollable distanceForIncrement:increment inDirection:direction];
264
265 WebCore::KeyboardScroll scroll;
266 scroll.offset = WebCore::unitVectorForScrollDirection(direction).scaled(scrollDistance);
267 scroll.granularity = increment;
268 scroll.direction = direction;
269 scroll.maximumVelocity = scroll.offset.scaled(self.parameters.maximumVelocityMultiplier);
270
271 // Apply a constant force to achieve Vmax in timeToMaximumVelocity seconds.
272 // F_constant = m * Vmax / t
273 scroll.force = scroll.maximumVelocity.scaled(self.parameters.springMass / self.parameters.timeToMaximumVelocity);
274
275 return scroll;
276}

278- (BOOL)beginWithEvent:(::WebEvent *)event
279{
280 if (event.type != WebEventKeyDown)
281 return NO;
282
283 auto scroll = [self keyboardScrollForEvent:event];
284 if (!scroll)
285 return NO;
286
287 if (_scrollTriggeringKeyIsPressed)
288 return NO;
289
290 if (![_scrollable rubberbandableDirections].at(boxSide(scroll->direction)))
291 return NO;
292
293 _scrollTriggeringKeyIsPressed = YES;
294 _currentScroll = scroll;
295
296 if (scroll->granularity == WebCore::ScrollGranularity::ScrollByDocument) {
297 _velocity = { };
298 [self stopAnimatedScroll];
299 [self stopDisplayLink];
300 [_scrollable scrollWithScrollToExtentAnimationTo:[_scrollable boundedContentOffset:_currentPosition + scroll->offset]];
301 return YES;
302 }
303
304 [self startDisplayLinkIfNeeded];
305
306 _currentPosition = WebCore::FloatPoint([_scrollable contentOffset]);
307 _velocity += WebCore::FloatSize([_scrollable interactiveScrollVelocity]);
308 _idealPositionForMinimumTravel = _currentPosition + _currentScroll->offset;
309
310 return YES;
311}
312
313- (void)handleKeyEvent:(::WebEvent *)event
314{
315 if (!_scrollTriggeringKeyIsPressed)
316 return;
317
318 auto scroll = [self keyboardScrollForEvent:event];
319
320 // UIKit does not emit a keyup event when the Command key is down. See <rdar://problem/49523065>.
321 // For recognized key commands that include the Command key (e.g. Command + Arrow Up) we reset our
322 // state on keydown.
323 if (!scroll || event.type == WebEventKeyUp || (event.modifierFlags & WebEventFlagMaskCommandKey)) {
324 [self stopAnimatedScroll];
325 _scrollTriggeringKeyIsPressed = NO;
326 }
327}

329static WebCore::FloatPoint farthestPointInDirection(WebCore::FloatPoint a, WebCore::FloatPoint b, WebCore::ScrollDirection direction)
330{
331 switch (direction) {
332 case WebCore::ScrollDirection::ScrollUp:
333 return WebCore::FloatPoint(a.x(), std::min(a.y(), b.y()));
334 case WebCore::ScrollDirection::ScrollDown:
335 return WebCore::FloatPoint(a.x(), std::max(a.y(), b.y()));
336 case WebCore::ScrollDirection::ScrollLeft:
337 return WebCore::FloatPoint(std::min(a.x(), b.x()), a.y());
338 case WebCore::ScrollDirection::ScrollRight:
339 return WebCore::FloatPoint(std::max(a.x(), b.x()), a.y());
340 }
341
342 ASSERT_NOT_REACHED();
343 return { };
344}

346- (void)stopAnimatedScroll
347{
348 if (!_currentScroll)
349 return;
350
351 // Determine the settling position of the spring, conserving the system's current energy.
352 // Kinetic = elastic potential
353 // 1/2 * m * v^2 = 1/2 * k * x^2
354 // x = sqrt(v^2 * m / k)
355 auto displacementMagnitudeSquared = (_velocity * _velocity).scaled(self.parameters.springMass / self.parameters.springStiffness);
356 WebCore::FloatSize displacement = {
357 std::copysign(sqrt(displacementMagnitudeSquared.width()), _velocity.width()),
358 std::copysign(sqrt(displacementMagnitudeSquared.height()), _velocity.height())
359 };
360
361 // If the spring would settle before the minimum travel distance
362 // for an instantaneous tap, move the settling position of the spring
363 // out to that point.
364 _idealPosition = [_scrollable boundedContentOffset:farthestPointInDirection(_currentPosition + displacement, _idealPositionForMinimumTravel, _currentScroll->direction)];
365
366 _currentScroll = std::nullopt;
367}

397- (void)displayLinkFired:(CADisplayLink *)sender
398{
399 WebCore::FloatSize force;
400 WebCore::FloatSize axesToApplySpring = { 1, 1 };
401
402 if (_currentScroll) {
403 auto scrollableDirections = [_scrollable scrollableDirectionsFromOffset:_currentPosition];
404 auto direction = _currentScroll->direction;
405
406 if (scrollableDirections.at(boxSide(direction))) {
407 // Apply the scrolling force. Only apply the spring in the perpendicular axis,
408 // otherwise it drags against the direction of motion.
409 axesToApplySpring = perpendicularAbsoluteUnitVector(direction);
410 force = _currentScroll->force;
411 } else {
412 // The scroll view cannot scroll in this direction, and is rubber-banding.
413 // Apply a constant and significant force; otherwise, the force for a
414 // single-line increment is not strong enough to rubber-band perceptibly.
415 force = WebCore::unitVectorForScrollDirection(direction).scaled(self.parameters.rubberBandForce);
416 }
417
418 // If we've reached or exceeded the maximum velocity, stop applying any force.
419 // However, we won't let the spring snap, we'll just keep going at the same
420 // velocity until the user raises their finger or we hit an edge.
421 if (fabs(_velocity.width()) >= fabs(_currentScroll->maximumVelocity.width()))
422 force.setWidth(0);
423 if (fabs(_velocity.height()) >= fabs(_currentScroll->maximumVelocity.height()))
424 force.setHeight(0);
425 }
426
427 WebCore::FloatPoint idealPosition = [_scrollable boundedContentOffset:_currentScroll ? _currentPosition : _idealPosition];
428 WebCore::FloatSize displacement = _currentPosition - idealPosition;
429
430 // Compute the spring's force, and apply it in allowed directions.
431 // F_spring = -k * x - c * v
432 auto springForce = - displacement.scaled(self.parameters.springStiffness) - _velocity.scaled(self.parameters.springDamping);
433 force += springForce * axesToApplySpring;
434
435 // Integrate acceleration -> velocity -> position for this time step.
436 CFTimeInterval frameDuration = sender.targetTimestamp - sender.timestamp;
437 WebCore::FloatSize acceleration = force.scaled(1. / self.parameters.springMass);
438 _velocity += acceleration.scaled(frameDuration);
439 _currentPosition += _velocity.scaled(frameDuration);
440
441 [_scrollable scrollToContentOffset:_currentPosition animated:NO];
442
443 // If we've effectively stopped scrolling, and no key is pressed,
444 // shut down the display link.
445 if (!_scrollTriggeringKeyIsPressed && _velocity.diagonalLengthSquared() < 1) {
446 [_scrollable didFinishScrolling];
447 [self stopDisplayLink];
448 _velocity = { };
449 }
450}

535- (CGFloat)distanceForIncrement:(WebCore::ScrollGranularity)increment inDirection:(WebCore::ScrollDirection)direction
536{
537 auto scrollView = _scrollView.getAutoreleased();
538 if (!scrollView)
539 return 0;
540
541 const CGFloat defaultPageScrollFraction = 0.8;
542 const CGFloat defaultLineScrollHeight = 40;
543
544 BOOL directionIsHorizontal = direction == WebCore::ScrollDirection::ScrollLeft || direction == WebCore::ScrollDirection::ScrollRight;
545
546 if (!_delegateRespondsToDistanceForIncrement) {
547 switch (increment) {
548 case WebCore::ScrollGranularity::ScrollByDocument:
549 return directionIsHorizontal ? scrollView.contentSize.width : scrollView.contentSize.height;
550 case WebCore::ScrollGranularity::ScrollByPage:
551 return (directionIsHorizontal ? scrollView.frame.size.width : scrollView.frame.size.height) * defaultPageScrollFraction;
552 case WebCore::ScrollGranularity::ScrollByLine:
553 return defaultLineScrollHeight * scrollView.zoomScale;
554 case WebCore::ScrollGranularity::ScrollByPixel:
555 return 0;
556 }
557 ASSERT_NOT_REACHED();
558 return 0;
559 }
560
561 return [_delegate keyboardScrollViewAnimator:self distanceForIncrement:increment inDirection:direction];
562}

564static UIAxis axesForDelta(WebCore::FloatSize delta)
565{
566 UIAxis axes = UIAxisNeither;

568 axes = static_cast<UIAxis>(axes | UIAxisHorizontal);
569 if (delta.height())
570 axes = static_cast<UIAxis>(axes | UIAxisVertical);
571 return axes;
572}

620- (WebCore::RectEdges<bool>)scrollableDirectionsFromOffset:(CGPoint)offset
621{
622 auto scrollView = _scrollView.getAutoreleased();
623 if (!scrollView)
624 return { };
625
626 UIEdgeInsets contentInsets = scrollView.adjustedContentInset;
627
628 CGSize contentSize = scrollView.contentSize;
629 CGSize scrollViewSize = scrollView.bounds.size;
630
631 CGPoint minimumContentOffset = CGPointMake(-contentInsets.left, -contentInsets.top);
632 CGPoint maximumContentOffset = CGPointMake(std::max(minimumContentOffset.x, contentSize.width + contentInsets.right - scrollViewSize.width), std::max(minimumContentOffset.y, contentSize.height + contentInsets.bottom - scrollViewSize.height));
633
634 WebCore::RectEdges<bool> edges;
635
636 edges.setTop(offset.y > minimumContentOffset.y);
637 edges.setBottom(offset.y < maximumContentOffset.y);
638 edges.setLeft(offset.x > minimumContentOffset.x);
639 edges.setRight(offset.x < maximumContentOffset.x);
640
641 return edges;
642}

644- (WebCore::RectEdges<bool>)rubberbandableDirections
645{
646 auto scrollView = _scrollView.getAutoreleased();
647 if (!scrollView)
648 return { };
649
650 WebCore::RectEdges<bool> edges;
651
652 edges.setTop(scrollView._canScrollWithoutBouncingY);
653 edges.setBottom(edges.top());
654 edges.setLeft(scrollView._canScrollWithoutBouncingX);
655 edges.setRight(edges.left());
656
657 return edges;
658}
