187static inline float adjustedUnexposedEdge(float documentEdge, float exposedRectEdge, float factor)
188{
189 if (exposedRectEdge < documentEdge)
190 return documentEdge - factor * (documentEdge - exposedRectEdge);
191
192 return exposedRectEdge;
193}
194
195static inline float adjustedUnexposedMaxEdge(float documentEdge, float exposedRectEdge, float factor)
196{
197 if (exposedRectEdge > documentEdge)
198 return documentEdge + factor * (exposedRectEdge - documentEdge);
199
200 return exposedRectEdge;
201}

203WebCore::FloatRect WebPageProxy::computeLayoutViewportRect(const FloatRect& unobscuredContentRect, const FloatRect& unobscuredContentRectRespectingInputViewBounds, const FloatRect& currentLayoutViewportRect, double displayedContentScale, FrameView::LayoutViewportConstraint constraint) const
204{
205 FloatRect constrainedUnobscuredRect = unobscuredContentRect;
206 FloatRect documentRect = pageClient().documentRect();
207
208 if (constraint == FrameView::LayoutViewportConstraint::ConstrainedToDocumentRect)
209 constrainedUnobscuredRect.intersect(documentRect);
210
211 double minimumScale = pageClient().minimumZoomScale();
212 bool isBelowMinimumScale = displayedContentScale < minimumScale;
213 if (isBelowMinimumScale) {
214 const CGFloat slope = 12;
215 CGFloat factor = std::max<CGFloat>(1 - slope * (minimumScale - displayedContentScale), 0);
216
217 constrainedUnobscuredRect.setX(adjustedUnexposedEdge(documentRect.x(), constrainedUnobscuredRect.x(), factor));
218 constrainedUnobscuredRect.setY(adjustedUnexposedEdge(documentRect.y(), constrainedUnobscuredRect.y(), factor));
219 constrainedUnobscuredRect.setWidth(adjustedUnexposedMaxEdge(documentRect.maxX(), constrainedUnobscuredRect.maxX(), factor) - constrainedUnobscuredRect.x());
220 constrainedUnobscuredRect.setHeight(adjustedUnexposedMaxEdge(documentRect.maxY(), constrainedUnobscuredRect.maxY(), factor) - constrainedUnobscuredRect.y());
221 }
222
223 FloatSize constrainedSize = isBelowMinimumScale ? constrainedUnobscuredRect.size() : unobscuredContentRect.size();
224 FloatRect unobscuredContentRectForViewport = isBelowMinimumScale ? constrainedUnobscuredRect : unobscuredContentRectRespectingInputViewBounds;
225
226 auto layoutViewportSize = FrameView::expandedLayoutViewportSize(m_baseLayoutViewportSize, LayoutSize(documentRect.size()), m_preferences->layoutViewportHeightExpansionFactor());
227 FloatRect layoutViewportRect = FrameView::computeUpdatedLayoutViewportRect(LayoutRect(currentLayoutViewportRect), LayoutRect(documentRect), LayoutSize(constrainedSize), LayoutRect(unobscuredContentRectForViewport), layoutViewportSize, m_minStableLayoutViewportOrigin, m_maxStableLayoutViewportOrigin, constraint);
228
229 if (layoutViewportRect != currentLayoutViewportRect)
230 LOG_WITH_STREAM(VisibleRects, stream << "WebPageProxy::computeLayoutViewportRect: new layout viewport " << layoutViewportRect);
231 return layoutViewportRect;
232}

333static bool exceedsRenderTreeSizeSizeThreshold(uint64_t thresholdSize, uint64_t committedSize)
334{

336 return committedSize > thresholdSize * thesholdSizeFraction;

339void WebPageProxy::didCommitLayerTree(const WebKit::RemoteLayerTreeTransaction& layerTreeTransaction)
340{
341 themeColorChanged(layerTreeTransaction.themeColor());
342 pageExtendedBackgroundColorDidChange(layerTreeTransaction.pageExtendedBackgroundColor());
343 sampledPageTopColorChanged(layerTreeTransaction.sampledPageTopColor());
344
345 if (!m_hasUpdatedRenderingAfterDidCommitLoad) {
346 if (layerTreeTransaction.transactionID() >= m_firstLayerTreeTransactionIdAfterDidCommitLoad) {
347 m_hasUpdatedRenderingAfterDidCommitLoad = true;
348 stopMakingViewBlankDueToLackOfRenderingUpdateIfNecessary();
349 m_lastVisibleContentRectUpdate = VisibleContentRectUpdateInfo();
350 }
351 }
352
353 pageClient().didCommitLayerTree(layerTreeTransaction);
354
355 // FIXME: Remove this special mechanism and fold it into the transaction's layout milestones.
356 if (m_observedLayoutMilestones.contains(WebCore::ReachedSessionRestorationRenderTreeSizeThreshold) && !m_hitRenderTreeSizeThreshold
357 && exceedsRenderTreeSizeSizeThreshold(m_sessionRestorationRenderTreeSize, layerTreeTransaction.renderTreeSize())) {
358 m_hitRenderTreeSizeThreshold = true;
359 didReachLayoutMilestone(WebCore::ReachedSessionRestorationRenderTreeSizeThreshold);
360 }
361}

594

600#endif
601 m_process->send(Messages::WebPage::ApplicationDidEnterBackground(isSuspendedUnderLock), m_webPageID);
602}

1364static bool desktopClassBrowsingRecommended(const WebCore::ResourceRequest& request, WebCore::IntSize viewSize, IgnoreAppCompatibilitySafeguards ignoreSafeguards)

1371 return false;

1374 if (webViewSizeIsNarrow(viewSize))

1385 auto screenClass = MGGetSInt32Answer(kMGQMainScreenClass, MGScreenClassPad2);
1386 shouldRecommendDesktopClassBrowsing = screenClass != MGScreenClassPad3 && screenClass != MGScreenClassPad4 && desktopClassBrowsingSupported();

1397WebContentMode WebPageProxy::effectiveContentModeAfterAdjustingPolicies(API::WebsitePolicies& policies, const WebCore::ResourceRequest& request)
1398{

1400 // FIXME: This is a compatibility hack to ensure that turning MSE on via the existing preference still enables MSE.
1401 policies.setMediaSourcePolicy(WebsiteMediaSourcePolicy::Enable);
1402 }
1403
1404 auto viewSize = this->viewSize();
1405 bool useDesktopBrowsingMode;
1406 switch (policies.preferredContentMode()) {
1407 case WebContentMode::Recommended: {
1408 auto ignoreSafeguards = m_navigationClient->shouldBypassContentModeSafeguards() ? IgnoreAppCompatibilitySafeguards::Yes : IgnoreAppCompatibilitySafeguards::No;
1409 useDesktopBrowsingMode = desktopClassBrowsingRecommended(request, viewSize, ignoreSafeguards);
1410 break;
1411 }
1412 case WebContentMode::Mobile:
1413 useDesktopBrowsingMode = false;
1414 break;
1415 case WebContentMode::Desktop:
1416 useDesktopBrowsingMode = !policies.allowSiteSpecificQuirksToOverrideContentMode() || desktopClassBrowsingRecommendedForRequest(request) != RecommendDesktopClassBrowsingForRequest::No;
1417 break;
1418 default:
1419 ASSERT_NOT_REACHED();
1420 useDesktopBrowsingMode = false;
1421 break;
1422 }
1423
1424 m_preferFasterClickOverDoubleTap = false;
1425
1426 if (!useDesktopBrowsingMode) {
1427 policies.setAllowContentChangeObserverQuirk(true);
1428 policies.setIdempotentModeAutosizingOnlyHonorsPercentages(true);
1429 return WebContentMode::Mobile;
1430 }
1431
1432 if (policies.customUserAgent().isEmpty() && customUserAgent().isEmpty()) {

1435 applicationName = applicationNameForDesktopUserAgent();

1437 }
