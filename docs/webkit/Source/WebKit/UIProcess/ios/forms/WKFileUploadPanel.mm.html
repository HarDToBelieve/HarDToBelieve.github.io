67static bool setContainsUTIThatConformsTo(NSSet<NSString *> *typeIdentifiers, UTType *conformToUTType)
68{
69 for (NSString *uti in typeIdentifiers) {
70 UTType *type = [UTType typeWithIdentifier:uti];

72 return true;
73 }
74 return false;

223- (void)_chooseFiles:(NSArray *)fileURLs displayString:(NSString *)displayString iconImage:(UIImage *)iconImage
224{
225 NSUInteger count = [fileURLs count];
226 if (!count) {
227 [self _cancel];
228 return;
229 }
230
231 Vector<String> filenames;
232 filenames.reserveInitialCapacity(count);
233 for (NSURL *fileURL in fileURLs)
234 filenames.uncheckedAppend(String::fromUTF8(fileURL.fileSystemRepresentation));
235
236 NSData *jpeg = UIImageJPEGRepresentation(iconImage, 1.0);
237 RefPtr<API::Data> iconImageDataRef = adoptRef(toImpl(WKDataCreate(reinterpret_cast<const unsigned char*>([jpeg bytes]), [jpeg length])));
238
239 _listener->chooseFiles(filenames, displayString, iconImageDataRef.get());
240 [self _dispatchDidDismiss];
241}

245- (void)presentWithParameters:(API::OpenPanelParameters*)parameters resultListener:(WebKit::WebOpenPanelResultListenerProxy*)listener
246{
247 ASSERT(!_listener);
248
249 _listener = listener;
250 _allowMultipleFiles = parameters->allowMultipleFiles();
251 _interactionPoint = [_view lastInteractionLocation];
252
253 Ref<API::Array> acceptMimeTypes = parameters->acceptMIMETypes();
254 NSMutableArray *mimeTypes = [NSMutableArray arrayWithCapacity:acceptMimeTypes->size()];
255 for (auto mimeType : acceptMimeTypes->elementsOfType<API::String>())
256 [mimeTypes addObject:mimeType->string()];
257
258 Ref<API::Array> acceptFileExtensions = parameters->acceptFileExtensions();
259 for (auto extension : acceptFileExtensions->elementsOfType<API::String>()) {
260 String mimeType = WebCore::MIMETypeRegistry::mimeTypeForExtension(extension->stringView().substring(1).toString());
261 if (!mimeType.isEmpty())
262 [mimeTypes addObject:mimeType];
263 }
264
265 _acceptedUTIs = UTIsForMIMETypes(mimeTypes);
266 if (![_acceptedUTIs count])
267 _allowedImagePickerTypes.add({ WKFileUploadPanelImagePickerType::Image, WKFileUploadPanelImagePickerType::Video });
268 else {
269 if (setContainsUTIThatConformsTo(_acceptedUTIs.get(), UTTypeImage))
270 _allowedImagePickerTypes.add({ WKFileUploadPanelImagePickerType::Image });
271 if (setContainsUTIThatConformsTo(_acceptedUTIs.get(), UTTypeMovie))
272 _allowedImagePickerTypes.add({ WKFileUploadPanelImagePickerType::Video });
273 }
274
275 _mediaCaptureType = WebCore::MediaCaptureTypeNone;
276#if ENABLE(MEDIA_CAPTURE)
277 _mediaCaptureType = parameters->mediaCaptureType();
278#endif
279
280 if (![self platformSupportsPickerViewController]) {
281 [self _cancel];
282 return;
283 }
284
285 if ([self _shouldMediaCaptureOpenMediaDevice]) {
286 [self _adjustMediaCaptureType];
287
288 _usingCamera = YES;
289 [self _showPhotoPickerWithSourceType:UIImagePickerControllerSourceTypeCamera];
290
291 return;
292 }
293
294 [self showDocumentPickerMenu];
295}

352#pragma mark - Media Types
353
354static NSSet<NSString *> *UTIsForMIMETypes(NSArray *mimeTypes)
355{
356 NSMutableSet *mediaTypes = [NSMutableSet set];
357 for (NSString *mimeType in mimeTypes) {
358 if ([mimeType caseInsensitiveCompare:@"image/*"] == NSOrderedSame)
359 [mediaTypes addObject:UTTypeImage.identifier];
360 else if ([mimeType caseInsensitiveCompare:@"video/*"] == NSOrderedSame)
361 [mediaTypes addObject:UTTypeMovie.identifier];
362 else if ([mimeType caseInsensitiveCompare:@"audio/*"] == NSOrderedSame)
363 // UIImagePickerController doesn't allow audio-only recording, so show the video
364 // recorder for "audio/*".
365 [mediaTypes addObject:UTTypeMovie.identifier];
366 else {
367 auto uti = [UTType typeWithMIMEType:mimeType];
368 if (uti)
369 [mediaTypes addObject:uti.identifier];
370 }
371
372 }
373 return mediaTypes;
374}
375
376- (NSArray<NSString *> *)_mediaTypesForPickerSourceType:(UIImagePickerControllerSourceType)sourceType
377{
378 NSArray<NSString *> *availableMediaTypeUTIs = [UIImagePickerController availableMediaTypesForSourceType:sourceType];
379 NSSet<NSString *> *acceptedMediaTypeUTIs = _acceptedUTIs.get();
380 if (acceptedMediaTypeUTIs.count) {
381 NSMutableArray<NSString *> *mediaTypes = [NSMutableArray array];
382 for (NSString *availableMediaTypeUTI in availableMediaTypeUTIs) {
383 if ([acceptedMediaTypeUTIs containsObject:availableMediaTypeUTI])
384 [mediaTypes addObject:availableMediaTypeUTI];
385 else {
386 UTType *availableMediaType = [UTType typeWithIdentifier:availableMediaTypeUTI];
387 for (NSString *acceptedMediaTypeUTI in acceptedMediaTypeUTIs) {
388 UTType *acceptedMediaType = [UTType typeWithIdentifier:acceptedMediaTypeUTI];
389 if ([acceptedMediaType conformsToType:availableMediaType]) {
390 [mediaTypes addObject:availableMediaTypeUTI];
391 break;
392 }
393 }
394 }
395 }
396
397 ASSERT(mediaTypes.count);
398 if (mediaTypes.count)
399 return mediaTypes;
400 }
401
402 // Fallback to every supported media type if there is no filter.
403 return availableMediaTypeUTIs;
404}

448- (UIContextMenuConfiguration *)contextMenuInteraction:(UIContextMenuInteraction *)interaction configurationForMenuAtLocation:(CGPoint)location {
449
450 UIContextMenuActionProvider actionMenuProvider = [self, weakSelf = WeakObjCPtr<WKFileUploadPanel>(self)] (NSArray<UIMenuElement *> *) -> UIMenu * {
451 NSArray *actions;
452
453 auto strongSelf = weakSelf.get();
454 if (!strongSelf)
455 return nil;
456
457 self->_isPresentingSubMenu = NO;
458 UIAction *chooseAction = [UIAction actionWithTitle:[strongSelf _chooseFilesButtonLabel] image:[UIImage systemImageNamed:@"folder"] identifier:@"choose" handler:^(__kindof UIAction *action) {
459 self->_isPresentingSubMenu = YES;

461 }];
462
463 UIAction *photoAction = [UIAction actionWithTitle:[strongSelf _photoLibraryButtonLabel] image:[UIImage systemImageNamed:@"photo.on.rectangle"] identifier:@"photo" handler:^(__kindof UIAction *action) {
464 self->_isPresentingSubMenu = YES;
465 [self _showPhotoPickerWithSourceType:UIImagePickerControllerSourceTypePhotoLibrary];
466 }];
467
468 if ([UIImagePickerController isSourceTypeAvailable:UIImagePickerControllerSourceTypeCamera]) {
469 NSString *cameraString = [strongSelf _cameraButtonLabel];
470 UIAction *cameraAction = [UIAction actionWithTitle:cameraString image:[UIImage systemImageNamed:@"camera"] identifier:@"camera" handler:^(__kindof UIAction *action) {
471 _usingCamera = YES;
472 self->_isPresentingSubMenu = YES;
473 [self _showPhotoPickerWithSourceType:UIImagePickerControllerSourceTypeCamera];
474 }];
475 actions = @[photoAction, cameraAction, chooseAction];
476 } else
477 actions = @[photoAction, chooseAction];
478
479 return [UIMenu menuWithTitle:@"" children:actions];
480 };
481
482 return [UIContextMenuConfiguration configurationWithIdentifier:nil previewProvider:nil actionProvider:actionMenuProvider];
483}

568- (void)_showPhotoPickerWithSourceType:(UIImagePickerControllerSourceType)sourceType
569{
570 ASSERT([UIImagePickerController isSourceTypeAvailable:sourceType]);
571
572 _imagePicker = adoptNS([[UIImagePickerController alloc] init]);
573 [_imagePicker setSourceType:sourceType];
574 [_imagePicker setMediaTypes:[self _mediaTypesForPickerSourceType:sourceType]];
575 [_imagePicker setDelegate:self];
576 [_imagePicker setAllowsEditing:NO];
577 [_imagePicker setModalPresentationStyle:UIModalPresentationFullScreen];
578 [_imagePicker _setAllowsMultipleSelection:_allowMultipleFiles];
579 [_imagePicker _setRequiresPickingConfirmation:YES];
580 [_imagePicker _setShowsFileSizePicker:YES];
581
582 if (_mediaCaptureType != WebCore::MediaCaptureTypeNone)
583 [_imagePicker setCameraDevice:cameraDeviceForMediaCaptureType(_mediaCaptureType)];
584
585 // Use a popover on the iPad if the source type is not the camera.
586 // The camera will use a fullscreen, modal view controller.
587 BOOL usePopover = currentUserInterfaceIdiomIsPadOrMac() && sourceType != UIImagePickerControllerSourceTypeCamera;
588 if (usePopover)
589 [self _presentPopoverWithContentViewController:_imagePicker.get() animated:YES];
590 else
591 [self _presentFullscreenViewController:_imagePicker.get() animated:YES];
592}

666- (BOOL)_willMultipleSelectionDelegateBeCalled
667{
668 // The multiple selection delegate will not be called when the UIImagePicker was not multiple selection.
669 if (!_allowMultipleFiles)
670 return NO;
671
672 // The multiple selection delegate will not be called when we used the camera in the UIImagePicker.
673 if (_usingCamera)
674 return NO;
675
676 return YES;
677}
678
679- (void)imagePickerController:(UIImagePickerController *)imagePicker didFinishPickingMediaWithInfo:(NSDictionary *)info
680{
681 // Sometimes both delegates get called, sometimes just one. Always let the
682 // multiple selection delegate handle everything if it will get called.
683 if ([self _willMultipleSelectionDelegateBeCalled])
684 return;
685
686 [self _dismissDisplayAnimated:YES];
687
688 [self _processMediaInfoDictionaries:@[info]
689 successBlock:^(NSArray *processedResults, NSString *displayString) {
690 ASSERT([processedResults count] == 1);
691 _WKFileUploadItem *result = [processedResults objectAtIndex:0];
692 RunLoop::main().dispatch([self, strongSelf = retainPtr(self), result = retainPtr(result), displayString = retainPtr(displayString)] {
693 [self _chooseFiles:@[result.get().fileURL] displayString:displayString.get() iconImage:result.get().displayImage.get()];
694 });
695 }
696 failureBlock:^{
697 RunLoop::main().dispatch([self, strongSelf = retainPtr(self)] {
698 [self _cancel];
699 });
700 }
701 ];
702}

746- (void)_processMediaInfoDictionaries:(NSArray *)infos atIndex:(NSUInteger)index processedResults:(NSMutableArray *)processedResults processedImageCount:(NSUInteger)processedImageCount processedVideoCount:(NSUInteger)processedVideoCount successBlock:(void (^)(NSArray *processedResults, NSString *displayString))successBlock failureBlock:(void (^)(void))failureBlock
747{
748 NSUInteger count = [infos count];
749 if (index == count) {
750 NSString *displayString = (processedImageCount || processedVideoCount) ? [NSString localizedStringWithFormat:WEB_UI_NSSTRING(@"%lu photo(s) and %lu video(s)", "label next to file upload control; parameters are the number of photos and the number of videos"), (unsigned long)processedImageCount, (unsigned long)processedVideoCount] : nil;
751 successBlock(processedResults, displayString);
752 return;
753 }
754
755 NSDictionary *info = [infos objectAtIndex:index];
756 ASSERT(index < count);
757 index++;

760 NSUInteger newProcessedVideoCount = processedVideoCount + (uploadItem.isVideo ? 1 : 0);
761 NSUInteger newProcessedImageCount = processedImageCount + (uploadItem.isVideo ? 0 : 1);
762 [processedResults addObject:uploadItem];
763 [self _processMediaInfoDictionaries:infos atIndex:index processedResults:processedResults processedImageCount:newProcessedImageCount processedVideoCount:newProcessedVideoCount successBlock:successBlock failureBlock:failureBlock];
764 };
765
766 [self _uploadItemFromMediaInfo:info successBlock:uploadItemSuccessBlock failureBlock:failureBlock];

769- (void)_uploadItemForImageData:(NSData *)imageData imageName:(NSString *)imageName successBlock:(void (^)(_WKFileUploadItem *))successBlock failureBlock:(void (^)(void))failureBlock
770{
771 ASSERT_ARG(imageData, imageData);
772 ASSERT(!RunLoop::isMain());
773
774 NSString * const kTemporaryDirectoryName = @"WKWebFileUpload";
775
776 // Build temporary file path.
777 NSString *temporaryDirectory = FileSystem::createTemporaryDirectory(kTemporaryDirectoryName);
778 NSString *filePath = [temporaryDirectory stringByAppendingPathComponent:imageName];
779 if (!filePath) {
780 LOG_ERROR("WKFileUploadPanel: Failed to create temporary directory to save image");
781 failureBlock();
782 return;
783 }
784
785 // Save the image to the temporary file.
786 NSError *error = nil;
787 [imageData writeToFile:filePath options:NSDataWritingAtomic error:&error];
788 if (error) {
789 LOG_ERROR("WKFileUploadPanel: Error writing image data to temporary file: %@", error);
790 failureBlock();
791 return;
792 }
793
794 successBlock(adoptNS([[_WKImageFileUploadItem alloc] initWithFileURL:[NSURL fileURLWithPath:filePath isDirectory:NO]]).get());
795}

821- (void)_uploadItemFromMediaInfo:(NSDictionary *)info successBlock:(void (^)(_WKFileUploadItem *))successBlock failureBlock:(void (^)(void))failureBlock
822{
823 NSString *mediaTypeUTI = [info objectForKey:UIImagePickerControllerMediaType];
824 UTType *mediaType = [UTType typeWithIdentifier:mediaTypeUTI];
825
826 // For videos from the existing library or camera, the media URL will give us a file path.
827 if ([mediaType conformsToType:UTTypeMovie]) {
828 NSURL *mediaURL = [info objectForKey:UIImagePickerControllerMediaURL];
829 if (![mediaURL isFileURL]) {
830 LOG_ERROR("WKFileUploadPanel: Expected media URL to be a file path, it was not");
831 ASSERT_NOT_REACHED();
832 failureBlock();
833 return;
834 }
835
836 successBlock(adoptNS([[_WKVideoFileUploadItem alloc] initWithFileURL:mediaURL]).get());
837 return;
838 }
839
840 if (![mediaType conformsToType:UTTypeImage]) {
841 LOG_ERROR("WKFileUploadPanel: Unexpected media type. Expected image or video, got: %@", mediaType);
842 ASSERT_NOT_REACHED();
843 failureBlock();
844 return;
845 }
846
847#if PLATFORM(IOS_FAMILY)
848 if (NSURL *imageURL = info[UIImagePickerControllerImageURL]) {
849 if (!imageURL.isFileURL) {
850 LOG_ERROR("WKFileUploadPanel: Expected image URL to be a file path, it was not");
851 ASSERT_NOT_REACHED();
852 failureBlock();
853 return;
854 }
855
856 successBlock(adoptNS([[_WKImageFileUploadItem alloc] initWithFileURL:imageURL]).get());
857 return;
858 }
859#endif
860
861 UIImage *originalImage = [info objectForKey:UIImagePickerControllerOriginalImage];
862 if (!originalImage) {
863 LOG_ERROR("WKFileUploadPanel: Expected image data but there was none");
864 ASSERT_NOT_REACHED();
865 failureBlock();
866 return;
867 }
868
869 // Photos taken with the camera will not have an image URL. Fall back to a JPEG representation.
870 [self _uploadItemForJPEGRepresentationOfImage:originalImage successBlock:successBlock failureBlock:failureBlock];
871}
