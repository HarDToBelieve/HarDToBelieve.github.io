41@implementation WKFormRotatingAccessoryPopover
42
43- (id)initWithView:(WKContentView *)view
44{
45 if (!(self = [super initWithView:view]))
46 return nil;
47 self.dismissionDelegate = self;
48 return self;
49}
50
51- (void)accessoryDone
52{
53 [self.view accessoryDone];
54}
55
56- (UIPopoverArrowDirection)popoverArrowDirections
57{
58 UIPopoverArrowDirection directions = UIPopoverArrowDirectionUp | UIPopoverArrowDirectionDown;
59 if (UIInterfaceOrientationIsLandscape([[UIApplication sharedApplication] statusBarOrientation]) && [[UIPeripheralHost sharedInstance] isOnScreen])
60 directions = UIPopoverArrowDirectionLeft | UIPopoverArrowDirectionRight;
61 return directions;
62}

64- (void)popoverWasDismissed:(WKRotatingPopover *)popover
65{
66 [self accessoryDone];
67}

71@interface WKRotatingPopover () <UIPopoverControllerDelegate>
72@end
73
74@implementation WKRotatingPopover {
75 WKContentView *_view;
76
77 BOOL _isRotating;
78 BOOL _isPreservingFocus;
79 CGPoint _presentationPoint;
80 RetainPtr<UIPopoverController> _popoverController;
81 id <WKRotatingPopoverDelegate> _dismissionDelegate;
82}
83
84- (id)initWithView:(WKContentView *)view
85{
86 if (!(self = [super init]))
87 return nil;
88
89 _view = view;
90 self.presentationPoint = CGPointZero;
91
92 NSNotificationCenter *center = [NSNotificationCenter defaultCenter];
93 [center addObserver:self selector:@selector(willRotate:) name:UIWindowWillRotateNotification object:nil];
94 [center addObserver:self selector:@selector(didRotate:) name:UIWindowDidRotateNotification object:nil];
95
96 return self;
97}
98
99- (void)dealloc
100{
101 _view = nil;
102
103 [_popoverController dismissPopoverAnimated:YES];
104 [_popoverController setDelegate:nil];
105 self.popoverController = nil;
106
107 NSNotificationCenter *center = [NSNotificationCenter defaultCenter];
108 [center removeObserver:self name:UIWindowWillRotateNotification object:nil];
109 [center removeObserver:self name:UIWindowDidRotateNotification object:nil];
110
111 [super dealloc];
112}
113
114- (UIPopoverController *)popoverController
115{
116 return _popoverController.get();
117}
118
119- (void)setPopoverController:(UIPopoverController *)popoverController
120{
121 if (_popoverController == popoverController)
122 return;
123
124 [_popoverController setDelegate:nil];
125 _popoverController = popoverController;
126 [_popoverController setDelegate:self];
127}
128
129- (UIPopoverArrowDirection)popoverArrowDirections
130{
131 return UIPopoverArrowDirectionAny;
132}
133
134- (void)presentPopoverAnimated:(BOOL)animated
135{
136 auto directions = [self popoverArrowDirections];
137 CGRect presentationRect;
138 if (CGPointEqualToPoint(self.presentationPoint, CGPointZero))
139 presentationRect = _view.focusedElementInformation.interactionRect;
140 else {
141 auto scale = _view.page->pageScaleFactor();
142 presentationRect = CGRectMake(self.presentationPoint.x * scale, self.presentationPoint.y * scale, 1, 1);
143 }
144
145 if (!CGRectIntersectsRect(presentationRect, _view.bounds))
146 return;
147
148#if PLATFORM(MACCATALYST)
149 [_view startRelinquishingFirstResponderToFocusedElement];
150#endif
151 [_popoverController presentPopoverFromRect:CGRectIntegral(presentationRect) inView:_view permittedArrowDirections:directions animated:animated];
152}
153
154- (void)dismissPopoverAnimated:(BOOL)animated
155{
156#if PLATFORM(MACCATALYST)
157 [_view stopRelinquishingFirstResponderToFocusedElement];
158#endif
159
160 [_popoverController dismissPopoverAnimated:animated];
161}
162
163- (void)willRotate:(NSNotification *)notification
164{
165 _isRotating = YES;
166 [_popoverController dismissPopoverAnimated:NO];
167}
168
169- (void)didRotate:(NSNotification *)notification
170{
171 _isRotating = NO;
172 [self presentPopoverAnimated:NO];
173}
174
175ALLOW_DEPRECATED_IMPLEMENTATIONS_BEGIN
176- (void)popoverControllerDidDismissPopover:(UIPopoverController *)popoverController
177ALLOW_DEPRECATED_IMPLEMENTATIONS_END
178{
179 if (_isRotating)
180 return;
181
182 [_dismissionDelegate popoverWasDismissed:self];
183}
