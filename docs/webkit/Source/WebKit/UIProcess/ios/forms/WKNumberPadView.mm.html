69typedef struct ColumnAndRow {
70 NSUInteger column;
71 NSUInteger row;
72} ColumnAndRow;
73
74static ColumnAndRow columnAndRowForPosition(WKNumberPadButtonPosition position)
75{
76 if (!position)
77 return { 1, 3 };
78 if (position == WKNumberPadButtonPositionBottomLeft)
79 return { 0, 3 };
80 if (position == WKNumberPadButtonPositionBottomRight)
81 return { 2, 3 };
82 return { (static_cast<NSUInteger>(position) - 1) % 3, (static_cast<NSUInteger>(position) - 1) / 3 };
83}
84
85static WKNumberPadButtonPosition positionForColumnAndRow(ColumnAndRow columnAndRow)
86{
87 NSUInteger column = columnAndRow.column;
88 NSUInteger row = columnAndRow.row;
89 if (row == 3) {
90 if (!column)
91 return WKNumberPadButtonPositionBottomLeft;
92 if (column == 1)
93 return WKNumberPadButtonPosition0;
94 return WKNumberPadButtonPositionBottomRight;
95 }
96 return static_cast<WKNumberPadButtonPosition>(1 + (3 * row + column));
97}

99static BOOL shouldDrawBackgroundForKey(WKNumberPadKey key, WKNumberPadButtonMode mode)
100{
101 switch (key) {
102 case WKNumberPadKeyAccept:
103 case WKNumberPadKeyToggleMode:
104 case WKNumberPadKeyNone:
105 return NO;
106 case WKNumberPadKeyPlus:
107 return mode == WKNumberPadButtonModeAlternate;
108 default:
109 return YES;
110 }
111}

113static UIColor *buttonTitleColorForKey(WKNumberPadKey key)
114{
115 return key == WKNumberPadKeyAccept ? [UIColor systemBlueColor] : [UIColor whiteColor];
116}

118static NSString *buttonTitleForKey(WKNumberPadKey key)
119{
120 switch (key) {
121 case WKNumberPadKeyDash:
122 return @"-";
123 case WKNumberPadKeyAsterisk:
124 return @"*";
125 case WKNumberPadKeyOctothorpe:
126 return @"#";
127 case WKNumberPadKeyClosingParenthesis:
128 return @")";
129 case WKNumberPadKeyOpeningParenthesis:
130 return @"(";
131 case WKNumberPadKeyPlus:
132 return @"+";
133 case WKNumberPadKeyToggleMode:
134 return @"â€¦";
135 case WKNumberPadKeyAccept:
136 return WebCore::numberPadOKButtonTitle();
137 case WKNumberPadKeyNone:
138 return nil;
139 default:
140 return String::number(key);
141 }
142}

170- (instancetype)initWithFrame:(CGRect)frame

178 [_backgroundView setUserInteractionEnabled:NO];

193- (void)setVisible:(BOOL)visible duration:(NSTimeInterval)duration delay:(NSTimeInterval)startDelay
194{
195 [self setEnabled:visible];
196 if (visible == _visible)
197 return;
198
199 auto applyBlock = [visible, protectedSelf = retainPtr(self)] {
200 [protectedSelf setTransform:visible ? CGAffineTransformIdentity : CGAffineTransformMakeScale(numberPadNonVisibleButtonScaleFactor, numberPadNonVisibleButtonScaleFactor)];
201 [protectedSelf titleLabel].alpha = visible ? 1 : 0;
202 };
203
204 if (duration > 0)
205 [UIView animateWithDuration:duration delay:startDelay options:UIViewAnimationOptionAllowUserInteraction animations:applyBlock completion:nil];
206 else
207 applyBlock();
208
209 _visible = visible;
210}

254{
255 [_backgroundView setBackgroundColor:shouldDrawBackgroundForKey(self.currentKey, self.buttonMode) ? [UIColor colorWithWhite:numberPadHighlightedWhiteColorValue alpha:1] : [UIColor clearColor]];

274- (void)setHighlighted:(BOOL)highlighted animated:(BOOL)animated
275{
276 self.layer.zPosition = highlighted ? numberPadHighlightedButtonZPosition : 0;
277 [super setHighlighted:highlighted];
278 [self setNeedsLayout];
279 if (!animated) {
280 [self _updateBackgroundView];
281 return;
282 }
283
284 [UIView animateWithDuration:numberPadHighlightAnimationDuration delay:0 options:UIViewAnimationOptionAllowUserInteraction | UIViewAnimationOptionBeginFromCurrentState animations:[protectedSelf = retainPtr(self)] {
285 [protectedSelf layoutIfNeeded];
286 [protectedSelf _updateBackgroundView];
287 } completion:nil];
288}

307- (void)layoutSubviews
308{
309 CGRect bounds = self.bounds;
310
311 UIView *titleView = self.titleLabel;
312 titleView.transform = CGAffineTransformIdentity;
313
314 UIView *imageView = self.imageView;
315 imageView.transform = CGAffineTransformIdentity;
316
317 if (_buttonMode == WKNumberPadButtonModeAlternate) {
318 BOOL enabled = self.alternateKey != WKNumberPadKeyNone;
319 [self setEnabled:enabled];
320 // This matches behavior on iOS, wherein switching to symbol keys disables any keys that don't support alternatives, but leaves their titles unchanged.
321 [self setTitle:buttonTitleForKey(enabled ? self.alternateKey : self.defaultKey) forState:UIControlStateNormal];
322 [self setTitleColor:buttonTitleColorForKey(enabled ? self.alternateKey : self.defaultKey) forState:UIControlStateNormal];
323 } else {
324 [self setEnabled:self.defaultKey != WKNumberPadKeyNone];
325 [self setTitle:self.defaultTitle forState:UIControlStateNormal];
326 [self setTitleColor:buttonTitleColorForKey(self.defaultKey) forState:UIControlStateNormal];
327 }
328
329 [super layoutSubviews];
330
331 [_backgroundView setTransform:CGAffineTransformIdentity];
332 [_backgroundView setFrame:CGRectInset(bounds, _backgroundInsetAmount, _backgroundInsetAmount)];
333 [_backgroundView layer].cornerRadius = numberPadButtonBackgroundViewCornerRadius;
334
335 CGAffineTransform transform = self.isHighlighted && shouldDrawBackgroundForKey(self.currentKey, self.buttonMode) ? CGAffineTransformMakeScale(numberPadButtonHighlightScaleFactor, numberPadButtonHighlightScaleFactor) : CGAffineTransformIdentity;
336 [_backgroundView setTransform:transform];
337 titleView.transform = transform;
338 imageView.transform = transform;
339
340 CGRect backgroundBounds = [_backgroundView bounds];
341 CGPoint contentCenter = [self convertPoint:CGPointMake(backgroundBounds.size.width / 2, backgroundBounds.size.height / 2) fromView:_backgroundView.get()];
342 titleView.center = contentCenter;
343 imageView.center = contentCenter;
344}

370static WKNumberPadKey defaultKeyAtPosition(WKNumberPadButtonPosition position, WKNumberPadInputMode inputMode)
371{
372 switch (position) {
373 case WKNumberPadButtonPositionBottomLeft:
374 if (inputMode == WKNumberPadInputModeTelephone)
375 return WKNumberPadKeyPlus;
376 if (inputMode == WKNumberPadInputModeNumbersOnly)
377 return WKNumberPadKeyNone;
378 return WKNumberPadKeyToggleMode;
379 case WKNumberPadButtonPositionBottomRight:
380 return WKNumberPadKeyAccept;
381 case WKNumberPadButtonPosition0:
382 return WKNumberPadKey0;
383 case WKNumberPadButtonPosition1:
384 return WKNumberPadKey1;
385 case WKNumberPadButtonPosition2:
386 return WKNumberPadKey2;
387 case WKNumberPadButtonPosition3:
388 return WKNumberPadKey3;
389 case WKNumberPadButtonPosition4:
390 return WKNumberPadKey4;
391 case WKNumberPadButtonPosition5:
392 return WKNumberPadKey5;
393 case WKNumberPadButtonPosition6:
394 return WKNumberPadKey6;
395 case WKNumberPadButtonPosition7:
396 return WKNumberPadKey7;
397 case WKNumberPadButtonPosition8:
398 return WKNumberPadKey8;
399 case WKNumberPadButtonPosition9:
400 return WKNumberPadKey9;
401 default:
402 ASSERT_NOT_REACHED();
403 return WKNumberPadKeyNone;
404 }
405}

431- (WKNumberPadButton *)_buttonForPosition:(WKNumberPadButtonPosition)position
432{
433 auto button = adoptNS([[WKNumberPadButton alloc] init]);
434 [button setDefaultKey:defaultKeyAtPosition(position, [_controller inputMode])];
435 [button setAlternateKey:alternateKeyAtPosition(position)];
436 [button setButtonPosition:position];
437 [button titleLabel].font = [UIFont systemFontOfSize:numberPadLabelFontSize() weight:UIFontWeightSemibold design:(NSString *)kCTFontUIFontDesignRounded];
438 [button setUserInteractionEnabled:NO];
439 return button.autorelease();
440}

442- (void)_initKeypad
443{
444 UIView *view = _contentView.get();
445 _keypad = adoptNS([[UIView alloc] init]);
446 [_keypad setUserInteractionEnabled:YES];
447
448 CGFloat xAnchors[] = { 0.0, 0.5, 1.0 };
449 CGFloat yAnchors[] = { 0.0, 0.5, 0.5, 1.0 };
450
451 for (unsigned x = 0; x < 3; ++x) {
452 for (unsigned y = 0; y < 4; ++y) {
453 WKNumberPadButton *button = [self _buttonForPosition:positionForColumnAndRow({ x, y })];
454 button.subAnchor = CGPointMake(xAnchors[x], yAnchors[y]);
455 [_keypad addSubview:button];
456 }
457 }
458 [view addSubview:_keypad.get()];
459}

461- (void)layoutSubviews
462{
463 [super layoutSubviews];
464
465 [_contentView setFrame:self.bounds];
466
467 CGFloat buttonSpacing = numberPadButtonSpacing();
468 CGSize buttonSize = CGSizeMake((CGRectGetWidth(self.bounds) - 2 * buttonSpacing) / 3, (CGRectGetHeight(self.bounds) - 3 * buttonSpacing) / 4);
469
470 [_keypad setFrame:self.bounds];
471 [[_keypad subviews] enumerateObjectsUsingBlock:^(WKNumberPadButton *button, NSUInteger, BOOL *) {
472 if (!button.visible)
473 [button setVisible:YES duration:0 delay:0];
474
475 auto columnAndRow = columnAndRowForPosition(button.buttonPosition);
476 CGFloat insetAmount = buttonSpacing / 2;
477 CGRect baseFrame = CGRectMake(columnAndRow.column * (buttonSize.width + buttonSpacing), columnAndRow.row * (buttonSize.height + buttonSpacing), buttonSize.width, buttonSize.height);
478 [button setFrame:CGRectInset(baseFrame, -insetAmount, -insetAmount)];
479 [button setBackgroundInsetAmount:insetAmount];
480 }];
481}

483- (WKNumberPadButton *)_numberPadButtonForTouch:(UITouch *)touch
484{
485 CGPoint pointInViewCoordinates = [touch locationInView:self];
486 for (WKNumberPadButton *button in [_keypad subviews]) {
487 CGPoint pointInButtonCoordinates = [button convertPoint:pointInViewCoordinates fromView:self];
488 if ([button pointInside:pointInButtonCoordinates withEvent:nil])
489 return button.enabled ? button : nil;
490 }
491
492 return nil;
493}

495- (void)touchesBegan:(NSSet<UITouch *> *)touches withEvent:(UIEvent *)event
496{
497 if (_trackedTouch)
498 return;
499
500 _trackedTouch = touches.anyObject;
501 _highlightedButton = [self _numberPadButtonForTouch:_trackedTouch.get()];
502 [_highlightedButton setHighlighted:YES];
503}
504
505- (void)touchesMoved:(NSSet<UITouch *> *)touches withEvent:(UIEvent *)event
506{
507 for (UITouch *touch in touches) {
508 if (touch != _trackedTouch)
509 continue;
510
511 WKNumberPadButton *newHighlightedButton = [self _numberPadButtonForTouch:_trackedTouch.get()];
512 if (newHighlightedButton == _highlightedButton)
513 return;
514
515 [_highlightedButton setHighlighted:NO];
516 _highlightedButton = newHighlightedButton;
517 [_highlightedButton setHighlighted:YES];
518 return;
519 }
520}
521
522- (void)touchesEnded:(NSSet<UITouch *> *)touches withEvent:(UIEvent *)event
523{
524 for (UITouch *touch in touches) {
525 if (touch != _trackedTouch)
526 continue;
527
528 _trackedTouch = nil;
529
530 if (!_highlightedButton)
531 return;
532
533 auto selectedKey = [_highlightedButton currentKey];
534 [_controller didSelectKey:selectedKey];
535 [_highlightedButton setHighlighted:NO animated:YES];
536
537 if (selectedKey == WKNumberPadKeyToggleMode || [_highlightedButton buttonMode] == WKNumberPadButtonModeAlternate)
538 [self toggleAlternateKeys];
539
540 _highlightedButton = nil;
541 return;
542 }
543}
544
545- (void)touchesCancelled:(NSSet<UITouch *> *)touches withEvent:(UIEvent *)event
546{
547 for (UITouch *touch in touches) {
548 if (touch != _trackedTouch)
549 continue;
550
551 _trackedTouch = nil;
552 [_highlightedButton setHighlighted:NO animated:YES];
553 _highlightedButton = nil;
554 return;
555 }
556}
557
558- (void)toggleAlternateKeys
559{
560 for (WKNumberPadButton *button in [_keypad subviews]) {
561 if (button.buttonMode == WKNumberPadButtonModeAlternate)
562 button.buttonMode = WKNumberPadButtonModeDefault;
563 else
564 button.buttonMode = WKNumberPadButtonModeAlternate;
565 }
