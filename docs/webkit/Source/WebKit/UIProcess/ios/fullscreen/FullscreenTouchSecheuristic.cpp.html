33double FullscreenTouchSecheuristic::scoreOfNextTouch(CGPoint location)
34{
35 Seconds now = MonotonicTime::now().secondsSinceEpoch();
36
37 if (!m_lastTouchTime) {
38 m_lastTouchTime = WTFMove(now);
39 return 0;
40 }
41
42 Seconds deltaTime = now - m_lastTouchTime;
43 m_lastTouchTime = now;
44
45 return scoreOfNextTouch(location, deltaTime);
46}
47
48double FullscreenTouchSecheuristic::scoreOfNextTouch(CGPoint location, const Seconds& deltaTime)
49{
50 if (m_lastTouchLocation.x == -1 && m_lastTouchLocation.y == -1) {
51 m_lastTouchLocation = WTFMove(location);
52 return 0;
53 }
54
55 double coefficient = attenuationFactor(deltaTime);
56 m_lastScore = coefficient * distanceScore(location, m_lastTouchLocation, deltaTime) + (1 - coefficient) * m_lastScore;
57 m_lastTouchLocation = location;
58 return m_lastScore;
59}
60
61void FullscreenTouchSecheuristic::reset()
62{
63 m_lastTouchTime = 0_s;
64 m_lastTouchLocation = { -1, -1 };
65 m_lastScore = 0;
66}

68double FullscreenTouchSecheuristic::distanceScore(const CGPoint& nextLocation, const CGPoint& lastLocation, const Seconds& deltaTime)
69{
70 double distance = sqrt(
71 m_parameters.xWeight * pow(nextLocation.x - lastLocation.x, 2) +
72 m_parameters.yWeight * pow(nextLocation.y - lastLocation.y, 2));
73 double sizeFactor = sqrt(
74 m_parameters.xWeight * pow(m_size.width, 2) +
75 m_parameters.yWeight * pow(m_size.height, 2));
76 double scaledDistance = distance / sizeFactor;
77 if (scaledDistance <= m_parameters.gammaCutoff)
78 return scaledDistance * (m_parameters.rampUpSpeed / deltaTime);
79
80 double exponentialDistance = m_parameters.gammaCutoff + pow((scaledDistance - m_parameters.gammaCutoff) / (1 - m_parameters.gammaCutoff), m_parameters.gamma);
81 return exponentialDistance * (m_parameters.rampUpSpeed / deltaTime);
82}
83
84double FullscreenTouchSecheuristic::attenuationFactor(Seconds delta)
85{
86 double normalizedTimeDelta = delta / m_parameters.rampDownSpeed;
87 return std::max(std::min(normalizedTimeDelta * m_weight, 1.0), 0.0);
88}
