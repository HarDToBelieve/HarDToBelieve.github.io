35static inline NSCorrectionIndicatorType correctionIndicatorType(WebCore::AlternativeTextType alternativeTextType)
36{
37 switch (alternativeTextType) {
38 case WebCore::AlternativeTextTypeCorrection:
39 return NSCorrectionIndicatorTypeDefault;
40 case WebCore::AlternativeTextTypeReversion:
41 return NSCorrectionIndicatorTypeReversion;
42 case WebCore::AlternativeTextTypeSpellingSuggestions:
43 return NSCorrectionIndicatorTypeGuesses;
44 case WebCore::AlternativeTextTypeDictationAlternatives:
45 ASSERT_NOT_REACHED();
46 break;
47 }
48 ASSERT_NOT_REACHED();
49 return NSCorrectionIndicatorTypeDefault;
50}

55CorrectionPanel::CorrectionPanel()

59}

61CorrectionPanel::~CorrectionPanel()
62{
63 dismissInternal(ReasonForDismissingAlternativeTextIgnored, false);
64}

66void CorrectionPanel::show(NSView *view, WebViewImpl& webViewImpl, AlternativeTextType type, const FloatRect& boundingBoxOfReplacedString, const String& replacedString, const String& replacementString, const Vector<String>& alternativeReplacementStrings)
67{
68 dismissInternal(ReasonForDismissingAlternativeTextIgnored, false);
69
70 if (!view)
71 return;
72
73 NSInteger spellCheckerDocumentTag = webViewImpl.spellCheckerDocumentTag();
74
75 NSString *replacedStringAsNSString = replacedString;
76 NSString *replacementStringAsNSString = replacementString;
77
78 m_view = view;
79 m_spellCheckerDocumentTag = spellCheckerDocumentTag;
80 NSCorrectionIndicatorType indicatorType = correctionIndicatorType(type);
81
82 RetainPtr<NSArray> alternativeStrings;
83 if (!alternativeReplacementStrings.isEmpty())
84 alternativeStrings = createNSArray(alternativeReplacementStrings);
85
86 NSSpellChecker* spellChecker = [NSSpellChecker sharedSpellChecker];
87 [spellChecker showCorrectionIndicatorOfType:indicatorType primaryString:replacementStringAsNSString alternativeStrings:alternativeStrings.get() forStringInRect:boundingBoxOfReplacedString view:m_view.get() completionHandler:^(NSString* acceptedString) {
88 handleAcceptedReplacement(webViewImpl, acceptedString, replacedStringAsNSString, replacementStringAsNSString, indicatorType);
89 }];
90}
91
92String CorrectionPanel::dismiss(ReasonForDismissingAlternativeText reason)
93{
94 return dismissInternal(reason, true);
95}
96
97String CorrectionPanel::dismissInternal(ReasonForDismissingAlternativeText reason, bool dismissingExternally)
98{
99 if (!isShowing())
100 return String();
101
102 m_wasDismissedExternally = dismissingExternally;
103 m_reasonForDismissing = reason;
104 m_resultForDismissal.clear();
105 [[NSSpellChecker sharedSpellChecker] dismissCorrectionIndicatorForView:m_view.get()];
106 return m_resultForDismissal.get();
107}

109void CorrectionPanel::recordAutocorrectionResponse(WebViewImpl& webViewImpl, NSInteger spellCheckerDocumentTag, NSCorrectionResponse response, const String& replacedString, const String& replacementString)
110{
111 if (webViewImpl.page().sessionID().isEphemeral())
112 return;
113
114 [[NSSpellChecker sharedSpellChecker] recordResponse:response toCorrection:replacementString forWord:replacedString language:nil inSpellDocumentWithTag:spellCheckerDocumentTag];
115}

117void CorrectionPanel::handleAcceptedReplacement(WebViewImpl& webViewImpl, NSString* acceptedReplacement, NSString* replaced, NSString* proposedReplacement, NSCorrectionIndicatorType correctionIndicatorType)
118{
119 if (!m_view)
120 return;
121
122 switch (correctionIndicatorType) {
123 case NSCorrectionIndicatorTypeDefault:
124 if (acceptedReplacement)
125 recordAutocorrectionResponse(webViewImpl, m_spellCheckerDocumentTag, NSCorrectionResponseAccepted, replaced, acceptedReplacement);
126 else {
127 if (!m_wasDismissedExternally || m_reasonForDismissing == ReasonForDismissingAlternativeTextCancelled)
128 recordAutocorrectionResponse(webViewImpl, m_spellCheckerDocumentTag, NSCorrectionResponseRejected, replaced, proposedReplacement);
129 else
130 recordAutocorrectionResponse(webViewImpl, m_spellCheckerDocumentTag, NSCorrectionResponseIgnored, replaced, proposedReplacement);
131 }
132 break;
133 case NSCorrectionIndicatorTypeReversion:
134 if (acceptedReplacement)
135 recordAutocorrectionResponse(webViewImpl, m_spellCheckerDocumentTag, NSCorrectionResponseReverted, replaced, acceptedReplacement);
136 break;
137 case NSCorrectionIndicatorTypeGuesses:
138 if (acceptedReplacement)
139 recordAutocorrectionResponse(webViewImpl, m_spellCheckerDocumentTag, NSCorrectionResponseAccepted, replaced, acceptedReplacement);
140 break;
141 }
142
143 webViewImpl.handleAcceptedAlternativeText(acceptedReplacement);
144 m_spellCheckerDocumentTag = 0;
145 m_view = nullptr;
146 if (acceptedReplacement)
147 m_resultForDismissal = adoptNS([acceptedReplacement copy]);
148}
