213 void encodeFixedLengthData(const uint8_t* data, size_t size, unsigned alignment)
214 {
215 RELEASE_ASSERT(data || !size);
216 ASSERT(!(reinterpret_cast<uintptr_t>(data) % alignment));
217
218 uint8_t* buffer = grow(alignment, size);
219 memcpy(buffer, data, size);
220 }
221
222 uint8_t* grow(unsigned alignment, size_t size)
223 {
224 size_t alignedSize = ((m_bufferSize + alignment - 1) / alignment) * alignment;
225
226 growCapacity(alignedSize + size);
227
228 std::memset(m_buffer.get() + m_bufferSize, 0, alignedSize - m_bufferSize);
229
230 m_bufferSize = alignedSize + size;
231 m_bufferPointer = m_buffer.get() + m_bufferSize;
232
233 return m_buffer.get() + alignedSize;
234 }
235
236 void growCapacity(size_t newSize)
237 {
238 if (newSize <= m_bufferCapacity)
239 return;
240
241 size_t newCapacity = m_bufferCapacity * 2;
242 while (newCapacity < newSize)
243 newCapacity *= 2;
244
245 m_buffer.realloc(newCapacity);
246 m_bufferCapacity = newCapacity;
247 }
248
249 size_t m_bufferSize;
250 size_t m_bufferCapacity;
251 MallocPtr<uint8_t, HistoryEntryDataEncoderMalloc> m_buffer;
252 uint8_t* m_bufferPointer;
253};

273static void encodeFormDataElement(HistoryEntryDataEncoder& encoder, const HTTPBody::Element& element)
274{
275 switch (element.type) {
276 case HTTPBody::Element::Type::Data:
277 encoder << FormDataElementType::Data;
278 encoder << element.data;
279 break;
280
281 case HTTPBody::Element::Type::File:
282 encoder << FormDataElementType::EncodedFile;
283 encoder << element.filePath;
284
285 // Used to be generatedFilename.
286 encoder << String();
287
288 // Used to be shouldGenerateFile.
289 encoder << false;
290
291 encoder << element.fileStart;
292 encoder << element.fileLength.value_or(-1);
293 encoder << element.expectedFileModificationTime.value_or(WallTime::nan()).secondsSinceEpoch().value();
294 break;
295
296 case HTTPBody::Element::Type::Blob:
297 encoder << FormDataElementType::EncodedBlob;
298 encoder << element.blobURLString;
299 break;
300 }
301}

303static void encodeFormData(HistoryEntryDataEncoder& encoder, const HTTPBody& formData)
304{
305 // Used to be alwaysStream.
306 encoder << false;
307
308 // Used to be boundary.
309 encoder << Vector<uint8_t>();
310
311 encoder << static_cast<uint64_t>(formData.elements.size());
312 for (const auto& element : formData.elements)
313 encodeFormDataElement(encoder, element);
314
315 // Used to be hasGeneratedFiles.
316 encoder << false;
317
318 // Used to be identifier.
319 encoder << static_cast<int64_t>(0);
320}

322static void encodeFrameStateNode(HistoryEntryDataEncoder& encoder, const FrameState& frameState)
323{
324 encoder << static_cast<uint64_t>(frameState.children.size());
325
326 for (const auto& childFrameState : frameState.children) {
327 encoder << childFrameState.originalURLString;
328 encoder << childFrameState.urlString;
329
330 encodeFrameStateNode(encoder, childFrameState);
331 }
332
333 encoder << frameState.documentSequenceNumber;
334
335 frameState.validateDocumentState();
336 encoder << static_cast<uint64_t>(frameState.documentState().size());
337 for (const auto& documentState : frameState.documentState())
338 encoder << documentState;
339
340 if (frameState.httpBody) {
341 encoder << frameState.httpBody.value().contentType;
342 encoder << true;
343
344 encodeFormData(encoder, frameState.httpBody.value());
345 } else {
346 encoder << String();
347 encoder << false;
348 }
349
350 encoder << frameState.itemSequenceNumber;
351
352 encoder << frameState.referrer;
353
354 encoder << frameState.scrollPosition.x();
355 encoder << frameState.scrollPosition.y();
356
357 encoder << frameState.pageScaleFactor;
358
359 encoder << !!frameState.stateObjectData;
360 if (frameState.stateObjectData)
361 encoder << frameState.stateObjectData.value();
362
363 encoder << frameState.target;
364
365#if PLATFORM(IOS_FAMILY)
366 // FIXME: iOS should not use the legacy session state encoder.
367 encoder << frameState.exposedContentRect;
368 encoder << frameState.unobscuredContentRect;
369 encoder << frameState.minimumLayoutSizeInScrollViewCoordinates;
370 encoder << frameState.contentSize;
371 encoder << frameState.scaleIsInitial;
372#endif
373}

413static RetainPtr<CFDictionaryRef> createDictionary(std::initializer_list<std::pair<CFStringRef, CFTypeRef>> keyValuePairs)
414{
415 Vector<CFTypeRef> keys;
416 Vector<CFTypeRef> values;
417
418 keys.reserveInitialCapacity(keyValuePairs.size());
419 values.reserveInitialCapacity(keyValuePairs.size());
420
421 for (const auto& keyValuePair : keyValuePairs) {
422 keys.uncheckedAppend(keyValuePair.first);
423 values.uncheckedAppend(keyValuePair.second);
424 }
425
426 return adoptCF(CFDictionaryCreate(kCFAllocatorDefault, keys.data(), values.data(), keyValuePairs.size(), &kCFTypeDictionaryKeyCallBacks, &kCFTypeDictionaryValueCallBacks));
427}

429static RetainPtr<CFDictionaryRef> encodeSessionHistory(const BackForwardListState& backForwardListState)
430{
431 ASSERT(!backForwardListState.currentIndex || backForwardListState.currentIndex.value() < backForwardListState.items.size());
432
433 auto sessionHistoryVersionNumber = adoptCF(CFNumberCreate(kCFAllocatorDefault, kCFNumberSInt32Type, &sessionHistoryVersion));
434
435 if (!backForwardListState.currentIndex)
436 return createDictionary({ { sessionHistoryVersionKey, sessionHistoryVersionNumber.get() } });
437
438 auto entries = adoptCF(CFArrayCreateMutable(kCFAllocatorDefault, backForwardListState.items.size(), &kCFTypeArrayCallBacks));
439 size_t totalDataSize = 0;
440
441 for (const auto& item : backForwardListState.items) {
442 auto url = item.pageState.mainFrameState.urlString.createCFString();
443 auto title = item.pageState.title.createCFString();
444 auto originalURL = item.pageState.mainFrameState.originalURLString.createCFString();
445 auto data = totalDataSize <= maximumSessionStateDataSize ? encodeSessionHistoryEntryData(item.pageState.mainFrameState) : nullptr;
446 auto shouldOpenExternalURLsPolicyValue = static_cast<uint64_t>(item.pageState.shouldOpenExternalURLsPolicy);
447 auto shouldOpenExternalURLsPolicy = adoptCF(CFNumberCreate(kCFAllocatorDefault, kCFNumberSInt64Type, &shouldOpenExternalURLsPolicyValue));
448
449 RetainPtr<CFDictionaryRef> entryDictionary;
450
451 if (data) {
452 totalDataSize += CFDataGetLength(data.get());
453
454 entryDictionary = createDictionary({
455 { sessionHistoryEntryURLKey, url.get() },
456 { sessionHistoryEntryTitleKey, title.get() },
457 { sessionHistoryEntryOriginalURLKey, originalURL.get() },
458 { sessionHistoryEntryDataKey, data.get() },
459 { sessionHistoryEntryShouldOpenExternalURLsPolicyKey, shouldOpenExternalURLsPolicy.get() },
460 });
461 } else {
462 entryDictionary = createDictionary({
463 { sessionHistoryEntryURLKey, url.get() },
464 { sessionHistoryEntryTitleKey, title.get() },
465 { sessionHistoryEntryOriginalURLKey, originalURL.get() },
466 { sessionHistoryEntryShouldOpenExternalURLsPolicyKey, shouldOpenExternalURLsPolicy.get() },
467 });
468 }
469
470 CFArrayAppendValue(entries.get(), entryDictionary.get());
471 }
472
473 uint32_t currentIndex = backForwardListState.currentIndex.value();
474 auto currentIndexNumber = adoptCF(CFNumberCreate(kCFAllocatorDefault, kCFNumberSInt32Type, &currentIndex));
475
476 return createDictionary({ { sessionHistoryVersionKey, sessionHistoryVersionNumber.get() }, { sessionHistoryCurrentIndexKey, currentIndexNumber.get() }, { sessionHistoryEntriesKey, entries.get() } });
477}

479RefPtr<API::Data> encodeLegacySessionState(const SessionState& sessionState)
480{
481 auto sessionHistoryDictionary = encodeSessionHistory(sessionState.backForwardListState);
482 auto provisionalURLString = sessionState.provisionalURL.isNull() ? nullptr : sessionState.provisionalURL.string().createCFString();
483 RetainPtr<CFNumberRef> renderTreeSizeNumber(adoptCF(CFNumberCreate(kCFAllocatorDefault, kCFNumberSInt64Type, &sessionState.renderTreeSize)));
484
485 RetainPtr<CFDictionaryRef> stateDictionary;
486 if (provisionalURLString) {
487 stateDictionary = createDictionary({
488 { sessionHistoryKey, sessionHistoryDictionary.get() },
489 { provisionalURLKey, provisionalURLString.get() },
490 { renderTreeSizeKey, renderTreeSizeNumber.get() }
491 });
492 } else {
493 stateDictionary = createDictionary({
494 { sessionHistoryKey, sessionHistoryDictionary.get() },
495 { renderTreeSizeKey, renderTreeSizeNumber.get() }
496 });
497 }
498
499 auto writeStream = adoptCF(CFWriteStreamCreateWithAllocatedBuffers(kCFAllocatorDefault, nullptr));
500 if (!writeStream)
501 return nullptr;
502
503 if (!CFWriteStreamOpen(writeStream.get()))
504 return nullptr;
505
506 if (!CFPropertyListWrite(stateDictionary.get(), writeStream.get(), kCFPropertyListBinaryFormat_v1_0, 0, nullptr))
507 return nullptr;
508
509 auto data = adoptCF(static_cast<CFDataRef>(CFWriteStreamCopyProperty(writeStream.get(), kCFStreamPropertyDataWritten)));
510
511 CFIndex length = CFDataGetLength(data.get());
512
513 size_t bufferSize = length + sizeof(uint32_t);
514 auto buffer = MallocPtr<uint8_t, HistoryEntryDataEncoderMalloc>::malloc(bufferSize);
515
516 // Put the session state version number at the start of the buffer
517 buffer.get()[0] = (sessionStateDataVersion & 0xff000000) >> 24;
518 buffer.get()[1] = (sessionStateDataVersion & 0x00ff0000) >> 16;
519 buffer.get()[2] = (sessionStateDataVersion & 0x0000ff00) >> 8;
520 buffer.get()[3] = (sessionStateDataVersion & 0x000000ff);
521
522 // Copy in the actual session state data
523 CFDataGetBytes(data.get(), CFRangeMake(0, length), buffer.get() + sizeof(uint32_t));
524
525 return API::Data::createWithoutCopying(buffer.leakPtr(), bufferSize, [] (unsigned char* buffer, const void* context) {
526 HistoryEntryDataEncoderMalloc::free(buffer);
527 }, nullptr);
528}

576 HistoryEntryDataDecoder& operator>>(String& value)
577 {
578 value = String();
579
580 uint32_t length;
581 *this >> length;
582
583 if (length == std::numeric_limits<uint32_t>::max()) {
584 // This is the null string.
585 value = String();
586 return *this;
587 }
588
589 uint64_t lengthInBytes;
590 *this >> lengthInBytes;
591
592 if (lengthInBytes % sizeof(UChar) || lengthInBytes / sizeof(UChar) != length) {
593 markInvalid();
594 return *this;
595 }
596
597 if (!bufferIsLargeEnoughToContain<UChar>(length)) {
598 markInvalid();
599 return *this;
600 }
601
602 UChar* buffer;
603 auto string = String::createUninitialized(length, buffer);
604 decodeFixedLengthData(reinterpret_cast<uint8_t*>(buffer), length * sizeof(UChar), alignof(UChar));
605
606 value = string;
607 return *this;
608 }

728 bool isValid() const { return m_buffer <= m_bufferEnd; }

733private:
734 template<typename Type>
735 HistoryEntryDataDecoder& decodeArithmeticType(Type& value)
736 {
737 static_assert(std::is_arithmetic<Type>::value, "");
738 value = Type();
739
740 decodeFixedLengthData(reinterpret_cast<uint8_t*>(&value), sizeof(value), sizeof(value));

743
744 void decodeFixedLengthData(uint8_t* data, size_t size, unsigned alignment)
745 {
746 if (!alignBufferPosition(alignment, size))
747 return;
748
749 memcpy(data, m_buffer, size);
750 m_buffer += size;
751 }
752
753 bool alignBufferPosition(unsigned alignment, size_t size)
754 {
755 const uint8_t* alignedPosition = alignedBuffer(alignment);
756 if (!alignedBufferIsLargeEnoughToContain(alignedPosition, size)) {
757 // We've walked off the end of this buffer.
758 markInvalid();
759 return false;
760 }
761
762 m_buffer = alignedPosition;
763 return true;
764 }
765
766 const uint8_t* alignedBuffer(unsigned alignment) const
767 {
768 ASSERT(alignment && !(alignment & (alignment - 1)));
769
770 uintptr_t alignmentMask = alignment - 1;
771 return reinterpret_cast<uint8_t*>((reinterpret_cast<uintptr_t>(m_buffer) + alignmentMask) & ~alignmentMask);
772 }

774 template<typename T>
775 bool bufferIsLargeEnoughToContain(size_t numElements) const
776 {
777 static_assert(std::is_arithmetic<T>::value, "Type T must have a fixed, known encoded size!");
778
779 if (numElements > std::numeric_limits<size_t>::max() / sizeof(T))
780 return false;
781
782 return bufferIsLargeEnoughToContain(alignof(T), numElements * sizeof(T));
783 }
784
785 bool bufferIsLargeEnoughToContain(unsigned alignment, size_t size) const
786 {
787 return alignedBufferIsLargeEnoughToContain(alignedBuffer(alignment), size);

799static void decodeFormDataElement(HistoryEntryDataDecoder& decoder, HTTPBody::Element& formDataElement)
800{
801 std::optional<FormDataElementType> elementType;
802 decoder >> elementType;
803 if (!elementType)
804 return;
805
806 switch (elementType.value()) {
807 case FormDataElementType::Data:
808 formDataElement.type = HTTPBody::Element::Type::Data;
809 decoder >> formDataElement.data;
810 break;
811
812 case FormDataElementType::EncodedFile: {
813 decoder >> formDataElement.filePath;
814
815 String generatedFilename;
816 decoder >> generatedFilename;
817
818 bool shouldGenerateFile;
819 decoder >> shouldGenerateFile;
820
821 decoder >> formDataElement.fileStart;
822 if (formDataElement.fileStart < 0) {
823 decoder.markInvalid();
824 return;
825 }
826
827 int64_t fileLength;
828 decoder >> fileLength;
829 if (fileLength != -1) {
830 if (fileLength < formDataElement.fileStart)
831 return;
832
833 formDataElement.fileLength = fileLength;
834 }
835
836 double expectedFileModificationTime;
837 decoder >> expectedFileModificationTime;
838 if (!std::isnan(expectedFileModificationTime))
839 formDataElement.expectedFileModificationTime = WallTime::fromRawSeconds(expectedFileModificationTime);
840
841 break;
842 }
843
844 case FormDataElementType::EncodedBlob:
845 decoder >> formDataElement.blobURLString;
846 break;
847 }
848}

879{
880 uint64_t childCount;
881 decoder >> childCount;
882
883 for (uint64_t i = 0; i < childCount; ++i) {
884 FrameState childFrameState;
885 decoder >> childFrameState.originalURLString;
886 decoder >> childFrameState.urlString;
887
888 decodeBackForwardTreeNode(decoder, childFrameState);
889
890 if (!decoder.isValid())
891 return;
892
893 frameState.children.append(WTFMove(childFrameState));
894 }
895
896 decoder >> frameState.documentSequenceNumber;
897
898 uint64_t documentStateVectorSize;
899 decoder >> documentStateVectorSize;
900
901 Vector<String> documentState;
902 for (uint64_t i = 0; i < documentStateVectorSize; ++i) {
903 String state;
904 decoder >> state;
905
906 if (!decoder.isValid())
907 return;
908
909 documentState.append(WTFMove(state));
910 }
911 frameState.setDocumentState(documentState, FrameState::ShouldValidate::Yes);
912

914 decoder >> formContentType;
915
916 bool hasFormData;
917 decoder >> hasFormData;
918

923 decodeFormData(decoder, httpBody);

926 }

929
930 decoder >> frameState.referrer;
931
932 int32_t scrollPositionX;
933 decoder >> scrollPositionX;
934
935 int32_t scrollPositionY;
936 decoder >> scrollPositionY;
937
938 frameState.scrollPosition = WebCore::IntPoint(scrollPositionX, scrollPositionY);
939

941
942 bool hasStateObject;

944

946 Vector<uint8_t> stateObjectData;
947 decoder >> stateObjectData;

950 }

952 decoder >> frameState.target;
953
954#if PLATFORM(IOS_FAMILY)

1124bool decodeLegacySessionState(const uint8_t* bytes, size_t size, SessionState& sessionState)
1125{
1126 if (size < sizeof(uint32_t))
1127 return false;
1128
1129 uint32_t versionNumber = (bytes[0] << 24) + (bytes[1] << 16) + (bytes[2] << 8) + bytes[3];
1130
1131 if (versionNumber != sessionStateDataVersion)
1132 return false;
1133
1134 auto cfPropertyList = adoptCF(CFPropertyListCreateWithData(kCFAllocatorDefault, adoptCF(CFDataCreate(kCFAllocatorDefault, bytes + sizeof(uint32_t), size - sizeof(uint32_t))).get(), kCFPropertyListImmutable, nullptr, nullptr));
1135 auto sessionStateDictionary = dynamic_cf_cast<CFDictionaryRef>(cfPropertyList.get());
1136 if (!sessionStateDictionary)
1137 return false;
1138
1139 if (auto backForwardListDictionary = dynamic_cf_cast<CFDictionaryRef>(CFDictionaryGetValue(sessionStateDictionary, sessionHistoryKey))) {
1140 if (!decodeSessionHistory(backForwardListDictionary, sessionState.backForwardListState))
