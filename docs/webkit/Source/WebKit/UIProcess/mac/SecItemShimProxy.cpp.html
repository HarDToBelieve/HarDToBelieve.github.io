39SecItemShimProxy& SecItemShimProxy::singleton()
40{
41 static SecItemShimProxy* proxy;
42 static dispatch_once_t once;
43 dispatch_once(&once, ^{
44 proxy = adoptRef(new SecItemShimProxy).leakRef();
45 });
46 return *proxy;
47}

64void SecItemShimProxy::secItemRequest(const SecItemRequestData& request, CompletionHandler<void(std::optional<SecItemResponseData>&&)>&& response)
65{
66 switch (request.type()) {
67 case SecItemRequestData::Invalid:
68 LOG_ERROR("SecItemShimProxy::secItemRequest received an invalid data request. Please file a bug if you know how you caused this.");
69 response(SecItemResponseData { errSecParam, nullptr });
70 break;
71
72 case SecItemRequestData::CopyMatching: {
73 CFTypeRef resultObject = nullptr;
74 OSStatus resultCode = SecItemCopyMatching(request.query(), &resultObject);
75 response(SecItemResponseData { resultCode, adoptCF(resultObject) });
76 break;
77 }
78
79 case SecItemRequestData::Add: {
80 // Return value of SecItemAdd is often ignored. Even if it isn't, we don't have the ability to
81 // serialize SecKeychainItemRef.
82 OSStatus resultCode = SecItemAdd(request.query(), nullptr);
83 response(SecItemResponseData { resultCode, nullptr });
84 break;
85 }
86
87 case SecItemRequestData::Update: {
88 OSStatus resultCode = SecItemUpdate(request.query(), request.attributesToMatch());
89 response(SecItemResponseData { resultCode, nullptr });
90 break;
91 }
92
93 case SecItemRequestData::Delete: {
94 OSStatus resultCode = SecItemDelete(request.query());
95 response(SecItemResponseData { resultCode, nullptr });
96 break;
97 }
98 }
99}
100
101void SecItemShimProxy::secItemRequestSync(const SecItemRequestData& data, CompletionHandler<void(std::optional<SecItemResponseData>&&)>&& completionHandler)
102{
103 secItemRequest(data, WTFMove(completionHandler));
104}
