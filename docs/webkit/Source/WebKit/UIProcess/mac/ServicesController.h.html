31#include <wtf/Forward.h>
32#include <wtf/Noncopyable.h>
33#include <wtf/RetainPtr.h>
34
35namespace WebKit {
36
37class ServicesController {
38 WTF_MAKE_NONCOPYABLE(ServicesController);
39 friend NeverDestroyed<ServicesController>;
40public:
41 static ServicesController& singleton();
42
43 bool hasImageServices() const { return m_hasImageServices; }
44 bool hasSelectionServices() const { return m_hasSelectionServices; }
45 bool hasRichContentServices() const { return m_hasRichContentServices; }
46
47 void refreshExistingServices(bool refreshImmediately = true);
48
49private:
50 ServicesController();
51
52 dispatch_queue_t m_refreshQueue;
53 std::atomic_bool m_hasPendingRefresh;
54
55 std::atomic<bool> m_hasImageServices;
56 std::atomic<bool> m_hasSelectionServices;
57 std::atomic<bool> m_hasRichContentServices;
58
59 bool m_lastSentHasImageServices;
60 bool m_lastSentHasSelectionServices;
61 bool m_lastSentHasRichContentServices;
62
63 RetainPtr<id> m_extensionWatcher;
64 RetainPtr<id> m_uiExtensionWatcher;
65};
66
67} // namespace WebKit

69#endif // ENABLE(SERVICE_CONTROLS)
