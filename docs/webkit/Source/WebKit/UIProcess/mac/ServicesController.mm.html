47ServicesController::ServicesController()
48 : m_refreshQueue(dispatch_queue_create("com.apple.WebKit.ServicesController", DISPATCH_QUEUE_SERIAL))
49 , m_hasPendingRefresh(false)
50 , m_hasImageServices(false)
51 , m_hasSelectionServices(false)
52 , m_hasRichContentServices(false)
53{
54 refreshExistingServices();
55
56 auto refreshCallback = [this](NSArray *, NSError *) {
57 // We coalese refreshes from the notification callbacks because they can come in small batches.
58 refreshExistingServices(false);
59 };
60
61 auto extensionAttributes = @{ @"NSExtensionPointName" : @"com.apple.services" };
62 m_extensionWatcher = [NSExtension beginMatchingExtensionsWithAttributes:extensionAttributes completion:refreshCallback];
63 auto uiExtensionAttributes = @{ @"NSExtensionPointName" : @"com.apple.ui-services" };
64 m_uiExtensionWatcher = [NSExtension beginMatchingExtensionsWithAttributes:uiExtensionAttributes completion:refreshCallback];
65}

67static void hasCompatibleServicesForItems(dispatch_group_t group, NSArray *items, WTF::Function<void(bool)>&& completionHandler)
68{
69 NSSharingServiceMask servicesMask = NSSharingServiceMaskViewer | NSSharingServiceMaskEditor;
70
71 if ([NSSharingService respondsToSelector:@selector(getSharingServicesForItems:mask:completion:)]) {
72 dispatch_group_enter(group);
73 [NSSharingService getSharingServicesForItems:items mask:servicesMask completion:makeBlockPtr([completionHandler = WTFMove(completionHandler), group](NSArray *services) {
74 completionHandler(services.count);
75 dispatch_group_leave(group);
76 }).get()];
77 return;
78 }
79
80 completionHandler([NSSharingService sharingServicesForItems:items mask:servicesMask].count);
81}

83void ServicesController::refreshExistingServices(bool refreshImmediately)
84{
85 if (m_hasPendingRefresh)
86 return;
87
88 m_hasPendingRefresh = true;
89
90 auto refreshTime = dispatch_time(DISPATCH_TIME_NOW, refreshImmediately ? 0 : (int64_t)(1 * NSEC_PER_SEC));
91 dispatch_after(refreshTime, m_refreshQueue, ^{
92 auto serviceLookupGroup = adoptOSObject(dispatch_group_create());
93
94 static NSImage *image { [[NSImage alloc] init] };
95 hasCompatibleServicesForItems(serviceLookupGroup.get(), @[ image ], [this] (bool hasServices) {
96 m_hasImageServices = hasServices;
97 });
98
99 static NSAttributedString *attributedString { [[NSAttributedString alloc] initWithString:@"a"] };
100 hasCompatibleServicesForItems(serviceLookupGroup.get(), @[ attributedString ], [this] (bool hasServices) {
101 m_hasSelectionServices = hasServices;
102 });
103
104 static NeverDestroyed<RetainPtr<NSAttributedString>> attributedStringWithRichContent;
105 static std::once_flag attributedStringWithRichContentOnceFlag;
106 std::call_once(attributedStringWithRichContentOnceFlag, [&] {
107 WorkQueue::main().dispatchSync([&] {
108 auto attachment = adoptNS([[NSTextAttachment alloc] init]);
109 auto cell = adoptNS([[NSTextAttachmentCell alloc] initImageCell:image]);
110 [attachment setAttachmentCell:cell.get()];
111 auto richString = adoptNS([[NSAttributedString attributedStringWithAttachment:attachment.get()] mutableCopy]);
112 [richString appendAttributedString:attributedString];
113 attributedStringWithRichContent.get() = WTFMove(richString);
114 });
115 });
116
117 hasCompatibleServicesForItems(serviceLookupGroup.get(), @[ attributedStringWithRichContent.get().get() ], [this] (bool hasServices) {
118 m_hasRichContentServices = hasServices;
119 });
120
121 dispatch_group_notify(serviceLookupGroup.get(), dispatch_get_main_queue(), makeBlockPtr([this] {
122 bool availableServicesChanged = (m_lastSentHasImageServices != m_hasImageServices) || (m_lastSentHasSelectionServices != m_hasSelectionServices) || (m_lastSentHasRichContentServices != m_hasRichContentServices);
123
124 m_lastSentHasSelectionServices = m_hasSelectionServices;
125 m_lastSentHasImageServices = m_hasImageServices;
126 m_lastSentHasRichContentServices = m_hasRichContentServices;
127
128 if (availableServicesChanged) {
129 for (auto& processPool : WebProcessPool::allProcessPools())
130 processPool->sendToAllProcesses(Messages::WebProcess::SetEnabledServices(m_hasImageServices, m_hasSelectionServices, m_hasRichContentServices));
131 }
132
133 m_hasPendingRefresh = false;
134 }).get());
135 });
136}
