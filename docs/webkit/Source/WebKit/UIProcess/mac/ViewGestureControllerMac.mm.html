91double ViewGestureController::resistanceForDelta(double deltaScale, double currentScale)
92{
93 // Zoom out with slight acceleration, until we reach minimum scale.
94 if (deltaScale < 0 && currentScale > minMagnification)
95 return zoomOutBoost;
96
97 // Zoom in with no acceleration, until we reach maximum scale.
98 if (deltaScale > 0 && currentScale < maxMagnification)
99 return 1;
100
101 // Outside of the extremes, resist further scaling.
102 double limit = currentScale < minMagnification ? minMagnification : maxMagnification;
103 double scaleDistance = std::abs(limit - currentScale);
104 double scalePercent = std::min(std::max(scaleDistance / limit, 0.), 1.);
105 double resistance = zoomOutResistance + scalePercent * (0.01 - zoomOutResistance);
106
107 return resistance;
108}

116void ViewGestureController::handleMagnificationGestureEvent(NSEvent *event, FloatPoint origin)
117{
118 origin.setY(origin.y() - m_webPageProxy.topContentInset());
119
120 ASSERT(m_activeGestureType == ViewGestureType::None || m_activeGestureType == ViewGestureType::Magnification);
121
122 if (m_activeGestureType == ViewGestureType::None) {
123 if (event.phase != NSEventPhaseBegan)
124 return;
125
126 // FIXME: We drop the first frame of the gesture on the floor, because we don't have the visible content bounds yet.
127 prepareMagnificationGesture(origin);
128
129 return;
130 }
131
132 // We're still waiting for the DidCollectGeometry callback.
133 if (!m_visibleContentRectIsValid)
134 return;
135
136 willBeginGesture(ViewGestureType::Magnification);
137
138 double scale = event.magnification;
139 double scaleWithResistance = resistanceForDelta(scale, m_magnification) * scale;
140
141 m_magnification += m_magnification * scaleWithResistance;
142 m_magnification = std::min(std::max(m_magnification, minElasticMagnification), maxElasticMagnification);
143
144 m_magnificationOrigin = origin;
145
146 applyMagnification();
147
148 if (event.phase == NSEventPhaseEnded || event.phase == NSEventPhaseCancelled)
149 endMagnificationGesture();
150}

160static float maximumRectangleComponentDelta(FloatRect a, FloatRect b)
161{
162 return std::max(std::abs(a.x() - b.x()), std::max(std::abs(a.y() - b.y()), std::max(std::abs(a.width() - b.width()), std::abs(a.height() - b.height()))));
163}

165void ViewGestureController::didCollectGeometryForSmartMagnificationGesture(FloatPoint origin, FloatRect renderRect, FloatRect visibleContentRect, bool fitEntireRect, double viewportMinimumScale, double viewportMaximumScale)
166{
167 double currentScaleFactor = m_webPageProxy.pageScaleFactor();
168
169 FloatRect unscaledTargetRect = renderRect;
170
171 // If there was no usable element under the cursor, we'll scale towards the cursor instead.
172 if (unscaledTargetRect.isEmpty())
173 unscaledTargetRect.setLocation(origin);
174
175 unscaledTargetRect.scale(1 / currentScaleFactor);
176 unscaledTargetRect.inflateX(unscaledTargetRect.width() * smartMagnificationElementPadding);
177 unscaledTargetRect.inflateY(unscaledTargetRect.height() * smartMagnificationElementPadding);
178
179 double targetMagnification = visibleContentRect.width() / unscaledTargetRect.width();
180
181 FloatRect unscaledVisibleContentRect = visibleContentRect;
182 unscaledVisibleContentRect.scale(1 / currentScaleFactor);
183 FloatRect viewportConstrainedUnscaledTargetRect = unscaledTargetRect;
184 viewportConstrainedUnscaledTargetRect.intersect(unscaledVisibleContentRect);
185
186 if (unscaledTargetRect.width() > viewportConstrainedUnscaledTargetRect.width())
187 viewportConstrainedUnscaledTargetRect.setX(unscaledVisibleContentRect.x() + (origin.x() / currentScaleFactor) - viewportConstrainedUnscaledTargetRect.width() / 2);
188 if (unscaledTargetRect.height() > viewportConstrainedUnscaledTargetRect.height())
189 viewportConstrainedUnscaledTargetRect.setY(unscaledVisibleContentRect.y() + (origin.y() / currentScaleFactor) - viewportConstrainedUnscaledTargetRect.height() / 2);
190
191 // For replaced elements like images, we want to fit the whole element
192 // in the view, so scale it down enough to make both dimensions fit if possible.
193 if (fitEntireRect)
194 targetMagnification = std::min(targetMagnification, static_cast<double>(visibleContentRect.height() / viewportConstrainedUnscaledTargetRect.height()));
195
196 targetMagnification = std::min(std::max(targetMagnification, minMagnification), maxMagnification);
197
198 // Allow panning between elements via double-tap while magnified, unless the target rect is
199 // similar to the last one or the mouse has not moved, in which case we'll zoom all the way out.
200 if (currentScaleFactor > 1 && m_lastMagnificationGestureWasSmartMagnification) {
201 if (maximumRectangleComponentDelta(m_lastSmartMagnificationUnscaledTargetRect, unscaledTargetRect) < smartMagnificationPanScrollThreshold)
202 targetMagnification = 1;
203
204 if (m_lastSmartMagnificationOrigin == origin)
205 targetMagnification = 1;
206 }
207
208 FloatRect targetRect(viewportConstrainedUnscaledTargetRect);
209 targetRect.scale(targetMagnification);
210 FloatPoint targetOrigin(visibleContentRect.center());
211 targetOrigin.moveBy(-targetRect.center());
212
213 m_initialMagnification = m_webPageProxy.pageScaleFactor();
214 m_initialMagnificationOrigin = FloatPoint();
215
216 m_webPageProxy.drawingArea()->adjustTransientZoom(m_webPageProxy.pageScaleFactor(), scaledMagnificationOrigin(FloatPoint(), m_webPageProxy.pageScaleFactor()));
217 m_webPageProxy.drawingArea()->commitTransientZoom(targetMagnification, targetOrigin);
218
219 m_lastSmartMagnificationUnscaledTargetRect = unscaledTargetRect;
220 m_lastSmartMagnificationOrigin = origin;
221
222 m_lastMagnificationGestureWasSmartMagnification = true;
223}

252void ViewGestureController::trackSwipeGesture(PlatformScrollEvent event, SwipeDirection direction, RefPtr<WebBackForwardListItem> targetItem)
253{
254 BOOL swipingLeft = isPhysicallySwipingLeft(direction);
255 CGFloat maxProgress = swipingLeft ? 1 : 0;
256 CGFloat minProgress = !swipingLeft ? -1 : 0;
257
258 __block bool swipeCancelled = false;
259
260 ASSERT(!m_swipeCancellationTracker);
261 RetainPtr<WKSwipeCancellationTracker> swipeCancellationTracker = adoptNS([[WKSwipeCancellationTracker alloc] init]);
262 m_swipeCancellationTracker = swipeCancellationTracker;
263
264 [event trackSwipeEventWithOptions:NSEventSwipeTrackingConsumeMouseEvents dampenAmountThresholdMin:minProgress max:maxProgress usingHandler:^(CGFloat progress, NSEventPhase phase, BOOL isComplete, BOOL *stop) {
265 if ([swipeCancellationTracker isCancelled]) {
266 *stop = YES;
267 return;
268 }
269 if (phase == NSEventPhaseBegan)
270 this->beginSwipeGesture(targetItem.get(), direction);
271 CGFloat clampedProgress = std::min(std::max(progress, minProgress), maxProgress);
272 this->handleSwipeGesture(targetItem.get(), clampedProgress, direction);
273 if (phase == NSEventPhaseCancelled)
274 swipeCancelled = true;
275 if (phase == NSEventPhaseEnded || phase == NSEventPhaseCancelled)
276 this->willEndSwipeGesture(*targetItem, swipeCancelled);
277 if (isComplete)
278 this->endSwipeGesture(targetItem.get(), swipeCancelled);
279 }];
280}

291static CALayer *leastCommonAncestorLayer(const Vector<RetainPtr<CALayer>>& layers)
292{
293 Vector<Vector<CALayer *>> liveLayerPathsFromRoot(layers.size());
294
295 size_t shortestPathLength = std::numeric_limits<size_t>::max();
296
297 for (size_t layerIndex = 0; layerIndex < layers.size(); layerIndex++) {
298 CALayer *parent = [layers[layerIndex] superlayer];
299 while (parent) {
300 liveLayerPathsFromRoot[layerIndex].insert(0, parent);
301 shortestPathLength = std::min(shortestPathLength, liveLayerPathsFromRoot[layerIndex].size());
302 parent = parent.superlayer;
303 }
304 }
305
306 for (size_t pathIndex = 0; pathIndex < shortestPathLength; pathIndex++) {
307 CALayer *firstPathLayer = liveLayerPathsFromRoot[0][pathIndex];
308 for (size_t layerIndex = 1; layerIndex < layers.size(); layerIndex++) {
309 if (liveLayerPathsFromRoot[layerIndex][pathIndex] != firstPathLayer)
310 return firstPathLayer;
311 }
312 }
313
314 return liveLayerPathsFromRoot[0][shortestPathLength];
315}

317CALayer *ViewGestureController::determineSnapshotLayerParent() const
318{
319 if (m_currentSwipeLiveLayers.size() == 1)
320 return [m_currentSwipeLiveLayers[0] superlayer];
321
322 // We insert our snapshot into the first shared superlayer of the custom views' layer, above the frontmost or below the bottommost layer.
323 return leastCommonAncestorLayer(m_currentSwipeLiveLayers);
324}

326CALayer *ViewGestureController::determineLayerAdjacentToSnapshotForParent(SwipeDirection direction, CALayer *snapshotLayerParent) const
327{
328 // If we have custom swiping views, we assume that the views were passed to us in back-to-front z-order.
329 CALayer *layerAdjacentToSnapshot = isPhysicallySwipingLeft(direction) ? m_currentSwipeLiveLayers.first().get() : m_currentSwipeLiveLayers.last().get();
330
331 if (m_currentSwipeLiveLayers.size() == 1)
332 return layerAdjacentToSnapshot;
333
334 // If the layers are not all siblings, find the child of the layer we're going to insert the snapshot into which has the frontmost/bottommost layer as a child.
335 while (snapshotLayerParent != layerAdjacentToSnapshot.superlayer)
336 layerAdjacentToSnapshot = layerAdjacentToSnapshot.superlayer;
337 return layerAdjacentToSnapshot;
338}

340static bool layerGeometryFlippedToRoot(CALayer *layer)
341{
342 bool flipped = false;
343 CALayer *parent = layer;
344 while (parent) {
345 if (parent.isGeometryFlipped)
346 flipped = !flipped;
347 parent = parent.superlayer;
348 }
349 return flipped;
350}

365void ViewGestureController::beginSwipeGesture(WebBackForwardListItem* targetItem, SwipeDirection direction)
366{
367 ASSERT(m_currentSwipeLiveLayers.isEmpty());
368
369 m_webPageProxy.navigationGestureDidBegin();
370
371 willBeginGesture(ViewGestureType::Swipe);
372
373 CALayer *rootContentLayer = m_webPageProxy.acceleratedCompositingRootLayer();
374
375 m_swipeLayer = adoptNS([[CALayer alloc] init]);
376 m_swipeSnapshotLayer = adoptNS([[CALayer alloc] init]);
377 m_currentSwipeCustomViewBounds = windowRelativeBoundsForCustomSwipeViews();
378
379 FloatRect swipeArea;
380 float topContentInset = 0;
381 if (!m_customSwipeViews.isEmpty()) {
382 topContentInset = m_customSwipeViewsTopContentInset;
383 swipeArea = m_currentSwipeCustomViewBounds;
384 swipeArea.expand(0, topContentInset);
385
386 for (const auto& view : m_customSwipeViews) {
387 CALayer *layer = [view layer];
388 ASSERT(layer);
389 m_currentSwipeLiveLayers.append(layer);
390 }
391 } else {
392 swipeArea = [rootContentLayer convertRect:CGRectMake(0, 0, m_webPageProxy.viewSize().width(), m_webPageProxy.viewSize().height()) toLayer:nil];
393 topContentInset = m_webPageProxy.topContentInset();
394 m_currentSwipeLiveLayers.append(rootContentLayer);
395 }
396
397 CALayer *snapshotLayerParent = determineSnapshotLayerParent();
398 bool geometryIsFlippedToRoot = layerGeometryFlippedToRoot(snapshotLayerParent);
399
400 RetainPtr<CGColorRef> backgroundColor = CGColorGetConstantColor(kCGColorWhite);
401 if (ViewSnapshot* snapshot = targetItem->snapshot()) {
402 if (shouldUseSnapshotForSize(*snapshot, swipeArea.size(), topContentInset))
403 [m_swipeSnapshotLayer setContents:snapshot->asLayerContents()];
404
405 Color coreColor = snapshot->backgroundColor();
406 if (coreColor.isValid())
407 backgroundColor = cachedCGColor(coreColor);
408 m_currentSwipeSnapshot = snapshot;
409 }
410
411 [m_swipeLayer setBackgroundColor:backgroundColor.get()];
412 [m_swipeLayer setAnchorPoint:CGPointZero];
413 [m_swipeLayer setFrame:[snapshotLayerParent convertRect:swipeArea fromLayer:nil]];
414 [m_swipeLayer setName:@"Gesture Swipe Root Layer"];
415 [m_swipeLayer setGeometryFlipped:geometryIsFlippedToRoot];
416 [m_swipeLayer setDelegate:[WebActionDisablingCALayerDelegate shared]];
417
418 float deviceScaleFactor = m_webPageProxy.deviceScaleFactor();
419 [m_swipeSnapshotLayer setContentsGravity:kCAGravityTopLeft];
420 [m_swipeSnapshotLayer setContentsScale:deviceScaleFactor];
421 [m_swipeSnapshotLayer setAnchorPoint:CGPointZero];
422 [m_swipeSnapshotLayer setFrame:CGRectMake(0, 0, swipeArea.width(), swipeArea.height() - topContentInset)];
423 [m_swipeSnapshotLayer setName:@"Gesture Swipe Snapshot Layer"];
424 [m_swipeSnapshotLayer setDelegate:[WebActionDisablingCALayerDelegate shared]];
425
426 [m_swipeLayer addSublayer:m_swipeSnapshotLayer.get()];
427
428 if (m_webPageProxy.preferences().viewGestureDebuggingEnabled())
429 applyDebuggingPropertiesToSwipeViews();
430
431 m_didCallEndSwipeGesture = false;
432 m_removeSnapshotImmediatelyWhenGestureEnds = false;
433
434 CALayer *layerAdjacentToSnapshot = determineLayerAdjacentToSnapshotForParent(direction, snapshotLayerParent);
435 BOOL swipingLeft = isPhysicallySwipingLeft(direction);
436 if (swipingLeft)
437 [snapshotLayerParent insertSublayer:m_swipeLayer.get() below:layerAdjacentToSnapshot];
438 else
439 [snapshotLayerParent insertSublayer:m_swipeLayer.get() above:layerAdjacentToSnapshot];
440
441 // We don't know enough about the custom views' hierarchy to apply a shadow.
442 if (m_customSwipeViews.isEmpty()) {
443 FloatRect dimmingRect(FloatPoint(), m_webPageProxy.viewSize());
444 m_swipeDimmingLayer = adoptNS([[CALayer alloc] init]);
445 [m_swipeDimmingLayer setName:@"Gesture Swipe Dimming Layer"];
446 [m_swipeDimmingLayer setBackgroundColor:[NSColor blackColor].CGColor];
447 [m_swipeDimmingLayer setOpacity:swipeOverlayDimmingOpacity];
448 [m_swipeDimmingLayer setAnchorPoint:CGPointZero];
449 [m_swipeDimmingLayer setFrame:dimmingRect];
450 [m_swipeDimmingLayer setGeometryFlipped:geometryIsFlippedToRoot];
451 [m_swipeDimmingLayer setDelegate:[WebActionDisablingCALayerDelegate shared]];
452
453 FloatRect shadowRect(-swipeOverlayShadowWidth, topContentInset, swipeOverlayShadowWidth, m_webPageProxy.viewSize().height() - topContentInset);
454 m_swipeShadowLayer = adoptNS([[CAGradientLayer alloc] init]);
455 [m_swipeShadowLayer setName:@"Gesture Swipe Shadow Layer"];
456 [m_swipeShadowLayer setColors:@[
457 (__bridge id)adoptCF(CGColorCreateGenericGray(0, 1.)).get(),
458 (__bridge id)adoptCF(CGColorCreateGenericGray(0, 0.99)).get(),
459 (__bridge id)adoptCF(CGColorCreateGenericGray(0, 0.98)).get(),
460 (__bridge id)adoptCF(CGColorCreateGenericGray(0, 0.95)).get(),
461 (__bridge id)adoptCF(CGColorCreateGenericGray(0, 0.92)).get(),
462 (__bridge id)adoptCF(CGColorCreateGenericGray(0, 0.82)).get(),
463 (__bridge id)adoptCF(CGColorCreateGenericGray(0, 0.71)).get(),
464 (__bridge id)adoptCF(CGColorCreateGenericGray(0, 0.46)).get(),
465 (__bridge id)adoptCF(CGColorCreateGenericGray(0, 0.35)).get(),
466 (__bridge id)adoptCF(CGColorCreateGenericGray(0, 0.25)).get(),
467 (__bridge id)adoptCF(CGColorCreateGenericGray(0, 0.17)).get(),
468 (__bridge id)adoptCF(CGColorCreateGenericGray(0, 0.11)).get(),
469 (__bridge id)adoptCF(CGColorCreateGenericGray(0, 0.07)).get(),
470 (__bridge id)adoptCF(CGColorCreateGenericGray(0, 0.04)).get(),
471 (__bridge id)adoptCF(CGColorCreateGenericGray(0, 0.01)).get(),
472 (__bridge id)adoptCF(CGColorCreateGenericGray(0, 0.)).get(),
473 ]];
474 [m_swipeShadowLayer setLocations:@[
475 @0,
476 @0.03125,
477 @0.0625,
478 @0.0938,
479 @0.125,
480 @0.1875,
481 @0.25,
482 @0.375,
483 @0.4375,
484 @0.5,
485 @0.5625,
486 @0.625,
487 @0.6875,
488 @0.75,
489 @0.875,
490 @1,
491 ]];
492 [m_swipeShadowLayer setStartPoint:CGPointMake(1, 0)];
493 [m_swipeShadowLayer setEndPoint:CGPointMake(0, 0)];
494 [m_swipeShadowLayer setOpacity:swipeOverlayShadowOpacity];
495 [m_swipeShadowLayer setAnchorPoint:CGPointZero];
496 [m_swipeShadowLayer setFrame:shadowRect];
497 [m_swipeShadowLayer setGeometryFlipped:geometryIsFlippedToRoot];
498 [m_swipeShadowLayer setDelegate:[WebActionDisablingCALayerDelegate shared]];
499
500 if (swipingLeft)
501 [snapshotLayerParent insertSublayer:m_swipeDimmingLayer.get() above:m_swipeLayer.get()];
502 else
503 [snapshotLayerParent insertSublayer:m_swipeDimmingLayer.get() below:m_swipeLayer.get()];
504
505 [snapshotLayerParent insertSublayer:m_swipeShadowLayer.get() above:m_swipeLayer.get()];
506 }

509void ViewGestureController::handleSwipeGesture(WebBackForwardListItem* targetItem, double progress, SwipeDirection direction)
510{
511 ASSERT(m_activeGestureType == ViewGestureType::Swipe);
512
513 if (!m_webPageProxy.drawingArea())
514 return;
515
516 bool swipingLeft = isPhysicallySwipingLeft(direction);
517
518 double width;
519 if (!m_customSwipeViews.isEmpty())
520 width = m_currentSwipeCustomViewBounds.width();
521 else
522 width = m_webPageProxy.drawingArea()->size().width();
523
524 double swipingLayerOffset = floor(width * progress);
525
526 double dimmingProgress = swipingLeft ? 1 - progress : -progress;
527 dimmingProgress = std::min(1., std::max(dimmingProgress, 0.));
528 [m_swipeDimmingLayer setOpacity:dimmingProgress * swipeOverlayDimmingOpacity];
529
530 double absoluteProgress = std::abs(progress);
531 double remainingSwipeDistance = width - std::abs(absoluteProgress * width);
532 double shadowFadeDistance = [m_swipeShadowLayer bounds].size.width;
533 if (remainingSwipeDistance < shadowFadeDistance)
534 [m_swipeShadowLayer setOpacity:(remainingSwipeDistance / shadowFadeDistance) * swipeOverlayShadowOpacity];
535 else
536 [m_swipeShadowLayer setOpacity:swipeOverlayShadowOpacity];
537
538 [m_swipeShadowLayer setTransform:CATransform3DMakeTranslation((swipingLeft ? 0 : width) + swipingLayerOffset, 0, 0)];
539
540 if (swipingLeft) {
541 for (const auto& layer : m_currentSwipeLiveLayers)
542 [layer setTransform:CATransform3DMakeTranslation(swipingLayerOffset, 0, 0)];
543 } else {
544 [m_swipeLayer setTransform:CATransform3DMakeTranslation(width + swipingLayerOffset, 0, 0)];
545 didMoveSwipeSnapshotLayer();
546 }
547}

557void ViewGestureController::removeSwipeSnapshot()
558{
559 m_snapshotRemovalTracker.reset();
560
561 m_hasOutstandingRepaintRequest = false;
562
563 if (m_activeGestureType != ViewGestureType::Swipe)
564 return;
565
566 if (!m_didCallEndSwipeGesture) {
567 m_removeSnapshotImmediatelyWhenGestureEnds = true;
568 return;
569 }
570
571 resetState();
572}

574void ViewGestureController::resetState()
575{
576 if (m_currentSwipeSnapshot)
577 m_currentSwipeSnapshot->setVolatile(true);
578 m_currentSwipeSnapshot = nullptr;
579
580 if (m_swipeCancellationTracker)
581 [m_swipeCancellationTracker setIsCancelled:YES];
582 m_swipeCancellationTracker = nil;
583
584 for (const auto& layer : m_currentSwipeLiveLayers)
585 [layer setTransform:CATransform3DIdentity];
586
587 [m_swipeSnapshotLayer removeFromSuperlayer];
588 m_swipeSnapshotLayer = nullptr;
589
590 [m_swipeLayer removeFromSuperlayer];
591 m_swipeLayer = nullptr;
