112- (id)initWithWindow:(NSWindow *)window webView:(NSView *)webView page:(NakedRef<WebKit::WebPageProxy>)page
113{
114 self = [super initWithWindow:window];
115 if (!self)
116 return nil;
117 [window setDelegate:self];
118 [window setCollectionBehavior:([window collectionBehavior] | NSWindowCollectionBehaviorFullScreenPrimary)];
119
120 // Hide the titlebar during the animation to full screen so that only the WKWebView content is visible.
121 window.titlebarAlphaValue = 0;
122 window.animationBehavior = NSWindowAnimationBehaviorNone;
123
124 NSView *contentView = [window contentView];
125 contentView.hidden = YES;
126 contentView.autoresizesSubviews = YES;
127
128 _backgroundView = adoptNS([[NSView alloc] initWithFrame:contentView.bounds]);
129 _backgroundView.get().layer = [CALayer layer];
130 _backgroundView.get().wantsLayer = YES;
131 _backgroundView.get().autoresizingMask = NSViewWidthSizable | NSViewHeightSizable;
132 [contentView addSubview:_backgroundView.get()];
133
134 _clipView = adoptNS([[NSView alloc] initWithFrame:contentView.bounds]);
135 [_clipView setWantsLayer:YES];
136 [_clipView setAutoresizingMask:(NSViewWidthSizable | NSViewHeightSizable)];
137 [_backgroundView addSubview:_clipView.get()];
138
139 [self windowDidLoad];
140 [window displayIfNeeded];
141 _webView = webView;
142 _page = page.ptr();
143
144 _videoFullscreenManagerProxyClient.setParent(self);
145
146 [self videoControlsManagerDidChange];
147
148 return self;
149}

195- (void)cancelOperation:(id)sender
196{
197 // If the page doesn't respond in DefaultWatchdogTimerInterval seconds, it could be because
198 // the WebProcess has hung, so exit anyway.
199 if (!_watchdogTimer) {
200 [self _manager]->requestExitFullScreen();
201 _watchdogTimer = adoptNS([[NSTimer alloc] initWithFireDate:[NSDate dateWithTimeIntervalSinceNow:DefaultWatchdogTimerInterval] interval:0 target:self selector:@selector(_watchdogTimerFired:) userInfo:nil repeats:NO]);
202 [[NSRunLoop mainRunLoop] addTimer:_watchdogTimer.get() forMode:NSDefaultRunLoopMode];
203 }
204}

227static RetainPtr<CGImageRef> createImageWithCopiedData(CGImageRef sourceImage)
228{
229 size_t width = CGImageGetWidth(sourceImage);
230 size_t height = CGImageGetHeight(sourceImage);
231 size_t bitsPerComponent = CGImageGetBitsPerComponent(sourceImage);
232 size_t bitsPerPixel = CGImageGetBitsPerPixel(sourceImage);
233 size_t bytesPerRow = CGImageGetBytesPerRow(sourceImage);
234 CGColorSpaceRef colorSpace = CGImageGetColorSpace(sourceImage);
235 CGBitmapInfo bitmapInfo = CGImageGetBitmapInfo(sourceImage);
236 RetainPtr<CGDataProviderRef> provider = createImageProviderWithCopiedData(CGImageGetDataProvider(sourceImage));
237 bool shouldInterpolate = CGImageGetShouldInterpolate(sourceImage);
238 CGColorRenderingIntent intent = CGImageGetRenderingIntent(sourceImage);
239
240 return adoptCF(CGImageCreate(width, height, bitsPerComponent, bitsPerPixel, bytesPerRow, colorSpace, bitmapInfo, provider.get(), 0, shouldInterpolate, intent));

243- (void)enterFullScreen:(NSScreen *)screen
244{
245 if ([self isFullScreen])
246 return;
247 _fullScreenState = WaitingToEnterFullScreen;
248
249 if (!screen)
250 screen = [NSScreen mainScreen];
251 NSRect screenFrame = [screen frame];
252
253 NSRect webViewFrame = convertRectToScreen([_webView window], [_webView convertRect:[_webView frame] toView:nil]);
254
255 // Flip coordinate system:
256 webViewFrame.origin.y = NSMaxY([[[NSScreen screens] objectAtIndex:0] frame]) - NSMaxY(webViewFrame);
257
258 CGWindowID windowID = [[_webView window] windowNumber];
259 RetainPtr<CGImageRef> webViewContents = adoptCF(CGWindowListCreateImage(NSRectToCGRect(webViewFrame), kCGWindowListOptionIncludingWindow, windowID, kCGWindowImageShouldBeOpaque));
260
261 // Using the returned CGImage directly would result in calls to the WindowServer every time
262 // the image was painted. Instead, copy the image data into our own process to eliminate that
263 // future overhead.
264 webViewContents = createImageWithCopiedData(webViewContents.get());
265
266 ALLOW_DEPRECATED_DECLARATIONS_BEGIN
267 // Screen updates to be re-enabled in _startEnterFullScreenAnimationWithDuration:
268 NSDisableScreenUpdates();
269 [[self window] setAutodisplay:NO];
270 ALLOW_DEPRECATED_DECLARATIONS_END
271
272 [self _manager]->saveScrollPosition();
273 _savedTopContentInset = _page->topContentInset();
274 _page->setTopContentInset(0);
275 [[self window] setFrame:screenFrame display:NO];
276
277 // Painting is normally suspended when the WKView is removed from the window, but this is
278 // unnecessary in the full-screen animation case, and can cause bugs; see
279 // https://bugs.webkit.org/show_bug.cgi?id=88940 and https://bugs.webkit.org/show_bug.cgi?id=88374
280 // We will resume the normal behavior in _startEnterFullScreenAnimationWithDuration:
281 _page->setSuppressVisibilityUpdates(true);
282
283 // Swap the webView placeholder into place.
284 if (!_webViewPlaceholder)
285 _webViewPlaceholder = adoptNS([[WebCoreFullScreenPlaceholderView alloc] initWithFrame:[_webView frame]]);
286 [_webViewPlaceholder setTarget:nil];
287 [_webViewPlaceholder setContents:(__bridge id)webViewContents.get()];
288 self.savedConstraints = _webView.superview.constraints;
289 [self _replaceView:_webView with:_webViewPlaceholder.get()];
290
291 // Then insert the WebView into the full screen window
292 NSView *contentView = [[self window] contentView];
293 [_clipView addSubview:_webView positioned:NSWindowBelow relativeTo:nil];
294 _webView.frame = NSInsetRect(contentView.bounds, 0, -_page->topContentInset());
295
296 _savedScale = _page->pageScaleFactor();
297 _page->scalePage(1, WebCore::IntPoint());
298 [self _manager]->setAnimatingFullScreen(true);
299 [self _manager]->willEnterFullScreen();
300}

302- (void)beganEnterFullScreenWithInitialFrame:(NSRect)initialFrame finalFrame:(NSRect)finalFrame
303{
304 if (_fullScreenState != WaitingToEnterFullScreen)
305 return;
306 _fullScreenState = EnteringFullScreen;
307
308 _initialFrame = initialFrame;
309 _finalFrame = finalFrame;
310
311 [CATransaction begin];
312 [CATransaction setDisableActions:YES];
313
314 auto clipLayer = _clipView.get().layer;
315 NSView *contentView = [[self window] contentView];
316
317 // Give the initial animations a speed of "0". We want the animations in place when we order in
318 // the window, but to not start animating until we get the callback from AppKit with the required
319 // animation duration. These animations will be replaced with the final animations in
320 // -_startEnterFullScreenAnimationWithDuration:
321 [clipLayer addAnimation:zoomAnimation(_initialFrame, _finalFrame, self.window.screen.frame, 1, 0, AnimateIn) forKey:@"fullscreen"];
322 clipLayer.mask = createMask(contentView.bounds);
323 [clipLayer.mask addAnimation:maskAnimation(_initialFrame, _finalFrame, self.window.screen.frame, 1, 0, AnimateIn) forKey:@"fullscreen"];
324 contentView.hidden = NO;
325
326 NSWindow* window = self.window;
327 NSWindowCollectionBehavior behavior = [window collectionBehavior];
328 [window setCollectionBehavior:(behavior | NSWindowCollectionBehaviorCanJoinAllSpaces)];
329 [window makeFirstResponder:_webView];
330 [window makeKeyAndOrderFront:self];
331 [window setCollectionBehavior:behavior];
332
333 _page->setSuppressVisibilityUpdates(false);
334 ALLOW_DEPRECATED_DECLARATIONS_BEGIN
335 [[self window] setAutodisplay:YES];
336 ALLOW_DEPRECATED_DECLARATIONS_END
337 [[self window] displayIfNeeded];
338 ALLOW_DEPRECATED_DECLARATIONS_BEGIN
339 NSEnableScreenUpdates();
340 ALLOW_DEPRECATED_DECLARATIONS_END
341
342 [CATransaction commit];
343
344 [self.window enterFullScreenMode:self];
345}

347static const float minVideoWidth = 468; // Keep in sync with `--controls-bar-width`.
348
349- (void)finishedEnterFullScreenAnimation:(bool)completed
350{
351 if (_fullScreenState != EnteringFullScreen)
352 return;
353
354 if (completed) {
355 _fullScreenState = InFullScreen;
356
357 ALLOW_DEPRECATED_DECLARATIONS_BEGIN
358 // Screen updates to be re-enabled ta the end of the current block.
359 NSDisableScreenUpdates();
360 ALLOW_DEPRECATED_DECLARATIONS_END
361 [self _manager]->didEnterFullScreen();
362 [self _manager]->setAnimatingFullScreen(false);
363
364 [_backgroundView.get().layer removeAllAnimations];
365 [[_clipView layer] removeAllAnimations];
366 [[_clipView layer] setMask:nil];
367
368 [_webViewPlaceholder setExitWarningVisible:YES];
369 [_webViewPlaceholder setTarget:self];
370
371 NSSize minContentSize = self.window.contentMinSize;
372 minContentSize.width = minVideoWidth;
373 self.window.contentMinSize = minContentSize;
374
375 // Always show the titlebar in full screen mode.
376 self.window.titlebarAlphaValue = 1;
377 } else {
378 // Transition to fullscreen failed. Clean up.
379 _fullScreenState = NotInFullScreen;
380
381 ALLOW_DEPRECATED_DECLARATIONS_BEGIN
382 [[self window] setAutodisplay:YES];
383 ALLOW_DEPRECATED_DECLARATIONS_END
384 _page->setSuppressVisibilityUpdates(false);
385
386 NSResponder *firstResponder = [[self window] firstResponder];
387 [self _replaceView:_webViewPlaceholder.get() with:_webView];
388 BEGIN_BLOCK_OBJC_EXCEPTIONS
389 [NSLayoutConstraint activateConstraints:self.savedConstraints];
390 END_BLOCK_OBJC_EXCEPTIONS
391 self.savedConstraints = nil;
392 makeResponderFirstResponderIfDescendantOfView(_webView.window, firstResponder, _webView);
393 [[_webView window] makeKeyAndOrderFront:self];
394
395 _page->scalePage(_savedScale, WebCore::IntPoint());
396 [self _manager]->restoreScrollPosition();
397 _page->setTopContentInset(_savedTopContentInset);
398 [self _manager]->didExitFullScreen();
399 [self _manager]->setAnimatingFullScreen(false);
400
401 // FIXME(53342): remove once pointer events fire when elements move out from under the pointer.
402 NSEvent *fakeEvent = [NSEvent mouseEventWithType:NSEventTypeMouseMoved
403 location:[NSEvent mouseLocation]
404 modifierFlags:[[NSApp currentEvent] modifierFlags]
405 timestamp:[NSDate timeIntervalSinceReferenceDate]
406 windowNumber:[[_webView window] windowNumber]
407 context:nullptr
408 eventNumber:0
409 clickCount:0
410 pressure:0];
411 WebKit::NativeWebMouseEvent webEvent(fakeEvent, nil, _webView);
412 _page->handleMouseEvent(webEvent);
413 }
414
415 ALLOW_DEPRECATED_DECLARATIONS_BEGIN
416 NSEnableScreenUpdates();
417 ALLOW_DEPRECATED_DECLARATIONS_END
418
419 if (_requestedExitFullScreen) {
420 _requestedExitFullScreen = NO;
421 [self exitFullScreen];
422 }
423}

425- (void)exitFullScreen
426{
427 if (_fullScreenState == EnteringFullScreen
428 || _fullScreenState == WaitingToEnterFullScreen) {
429 // Do not try to exit fullscreen during the enter animation; remember
430 // that exit was requested and perform the exit upon enter fullscreen
431 // animation complete.
432 _requestedExitFullScreen = YES;
433 return;
434 }
435
436 if (_watchdogTimer) {
437 [_watchdogTimer invalidate];
438 _watchdogTimer.clear();
439 }
440
441 if (![self isFullScreen])
442 return;
443 _fullScreenState = WaitingToExitFullScreen;
444
445 [_webViewPlaceholder setExitWarningVisible:NO];
446
447 ALLOW_DEPRECATED_DECLARATIONS_BEGIN
448 // Screen updates to be re-enabled in _startExitFullScreenAnimationWithDuration: or beganExitFullScreenWithInitialFrame:finalFrame:
449 NSDisableScreenUpdates();
450 [[self window] setAutodisplay:NO];
451 ALLOW_DEPRECATED_DECLARATIONS_END
452
453 // See the related comment in enterFullScreen:
454 // We will resume the normal behavior in _startExitFullScreenAnimationWithDuration:
455 _page->setSuppressVisibilityUpdates(true);
456 [_webViewPlaceholder setTarget:nil];
457
458 [self _manager]->setAnimatingFullScreen(true);
459 [self _manager]->willExitFullScreen();
460}

462- (void)exitFullScreenImmediately
463{
464 if (![self isFullScreen])
465 return;
466
467 [self _manager]->requestExitFullScreen();
468 [_webViewPlaceholder setExitWarningVisible:NO];
469 [self _manager]->willExitFullScreen();
470 _fullScreenState = ExitingFullScreen;
471 [self finishedExitFullScreenAnimation:YES];
472}

474- (void)requestExitFullScreen

476 [self _manager]->requestExitFullScreen();
477}

479- (void)beganExitFullScreenWithInitialFrame:(NSRect)initialFrame finalFrame:(NSRect)finalFrame
480{
481 if (_fullScreenState != WaitingToExitFullScreen)
482 return;
483 _fullScreenState = ExitingFullScreen;
484
485 if (![[self window] isOnActiveSpace]) {
486 // If the full screen window is not in the active space, the NSWindow full screen animation delegate methods
487 // will never be called. So call finishedExitFullScreenAnimation explicitly.
488 [self finishedExitFullScreenAnimation:YES];
489
490 // Because we are breaking the normal animation pattern, re-enable screen updates
491 // as exitFullScreen has disabled them, but _startExitFullScreenAnimationWithDuration:
492 // will never be called.
493 ALLOW_DEPRECATED_DECLARATIONS_BEGIN
494 NSEnableScreenUpdates();
495 ALLOW_DEPRECATED_DECLARATIONS_END
496 }
497
498 [[self window] exitFullScreenMode:self];
499}

503static RetainPtr<CGImageRef> takeWindowSnapshot(CGSWindowID windowID, bool captureAtNominalResolution)
504{
505 CGSWindowCaptureOptions options = kCGSCaptureIgnoreGlobalClipShape;
506 if (captureAtNominalResolution)
507 options |= kCGSWindowCaptureNominalResolution;
508 RetainPtr<CFArrayRef> windowSnapshotImages = adoptCF(CGSHWCaptureWindowList(CGSMainConnectionID(), &windowID, 1, options));
509
510 if (windowSnapshotImages && CFArrayGetCount(windowSnapshotImages.get()))
511 return checked_cf_cast<CGImageRef>(CFArrayGetValueAtIndex(windowSnapshotImages.get(), 0));
512
513 // Fall back to the non-hardware capture path if we didn't get a snapshot
514 // (which usually happens if the window is fully off-screen).
515 CGWindowImageOption imageOptions = kCGWindowImageBoundsIgnoreFraming | kCGWindowImageShouldBeOpaque;
516 if (captureAtNominalResolution)
517 imageOptions |= kCGWindowImageNominalResolution;
518 return adoptCF(CGWindowListCreateImage(CGRectNull, kCGWindowListOptionIncludingWindow, windowID, imageOptions));
519}

521- (void)finishedExitFullScreenAnimation:(bool)completed
522{
523 if (_fullScreenState == InFullScreen) {
524 // If we are currently in the InFullScreen state, this notification is unexpected, meaning
525 // fullscreen was exited without being initiated by WebKit. Do not return early, but continue to
526 // clean up our state by calling those methods which would have been called by -exitFullscreen,
527 // and proceed to close the fullscreen window.
528 [self _manager]->requestExitFullScreen();
529 [_webViewPlaceholder setTarget:nil];
530 [self _manager]->setAnimatingFullScreen(false);
531 [self _manager]->willExitFullScreen();
532 } else if (_fullScreenState != ExitingFullScreen)
533 return;
534 _fullScreenState = NotInFullScreen;
535
536 // Hide the titlebar at the end of the animation so that it can slide away without turning blank.
537 self.window.titlebarAlphaValue = 0;
538
539 NSResponder *firstResponder = [[self window] firstResponder];
540
541 [CATransaction begin];
542 [CATransaction setDisableActions:YES];
543 NSRect exitPlaceholderScreenRect = _initialFrame;
544 exitPlaceholderScreenRect.origin.y = NSMaxY([[[NSScreen screens] objectAtIndex:0] frame]) - NSMaxY(exitPlaceholderScreenRect);
545
546 RetainPtr<CGImageRef> webViewContents = takeWindowSnapshot([[_webView window] windowNumber], true);
547 webViewContents = adoptCF(CGImageCreateWithImageInRect(webViewContents.get(), NSRectToCGRect(exitPlaceholderScreenRect)));
548
549 _exitPlaceholder = adoptNS([[NSView alloc] initWithFrame:[_webView frame]]);
550 [_exitPlaceholder setWantsLayer: YES];
551 [_exitPlaceholder setAutoresizesSubviews: YES];
552 [_exitPlaceholder setLayerContentsPlacement: NSViewLayerContentsPlacementScaleProportionallyToFit];
553 [_exitPlaceholder setLayerContentsRedrawPolicy: NSViewLayerContentsRedrawNever];
554 [_exitPlaceholder setFrame:[_webView frame]];
555 [[_exitPlaceholder layer] setContents:(__bridge id)webViewContents.get()];
556 [[_webView superview] addSubview:_exitPlaceholder.get() positioned:NSWindowAbove relativeTo:_webView];
557
558 [CATransaction commit];
559 [CATransaction flush];
560
561 [CATransaction begin];
562 [CATransaction setDisableActions:YES];
563
564 [_backgroundView.get().layer removeAllAnimations];
565 _page->setSuppressVisibilityUpdates(true);
566 [_webView removeFromSuperview];
567 [_webView setFrame:[_webViewPlaceholder frame]];
568 [_webView setAutoresizingMask:[_webViewPlaceholder autoresizingMask]];
569 [[_webViewPlaceholder superview] addSubview:_webView positioned:NSWindowBelow relativeTo:_webViewPlaceholder.get()];
570
571 BEGIN_BLOCK_OBJC_EXCEPTIONS
572 [NSLayoutConstraint activateConstraints:self.savedConstraints];
573 END_BLOCK_OBJC_EXCEPTIONS
574 self.savedConstraints = nil;
575 makeResponderFirstResponderIfDescendantOfView(_webView.window, firstResponder, _webView);
576
577 // These messages must be sent after the swap or flashing will occur during forceRepaint:
578 [self _manager]->didExitFullScreen();
579 [self _manager]->setAnimatingFullScreen(false);
580 _page->scalePage(_savedScale, WebCore::IntPoint());
581 [self _manager]->restoreScrollPosition();
582 _page->setTopContentInset(_savedTopContentInset);
583
584 _page->forceRepaint([weakSelf = WeakObjCPtr<WKFullScreenWindowController>(self)] {
585 [weakSelf completeFinishExitFullScreenAnimationAfterRepaint];
586 });
587
588 [CATransaction commit];
589 [CATransaction flush];
590}
591
592- (void)completeFinishExitFullScreenAnimationAfterRepaint
593{
594 [CATransaction begin];
595 [CATransaction setDisableActions:YES];
596
597 [_webViewPlaceholder removeFromSuperview];
598 [[self window] orderOut:self];
599 NSView *contentView = [[self window] contentView];
600 contentView.hidden = YES;
601 [_exitPlaceholder removeFromSuperview];
602 [[_exitPlaceholder layer] setContents:nil];
603 _exitPlaceholder = nil;
604
605 [[_webView window] makeKeyAndOrderFront:self];
606 _webViewPlaceholder = nil;
607
608 _page->setSuppressVisibilityUpdates(false);
609 _page->setNeedsDOMWindowResizeEvent();
610
611 [CATransaction commit];
612 [CATransaction flush];
613}

621- (void)close
622{
623 // We are being asked to close rapidly, most likely because the page
624 // has closed or the web process has crashed. Just walk through our
625 // normal exit full screen sequence, but don't wait to be called back
626 // in response.
627 if ([self isFullScreen])
628 [self exitFullScreenImmediately];
629
630 if (_fullScreenState == ExitingFullScreen)
631 [self finishedExitFullScreenAnimation:YES];
632
633 [super close];
634
635 _webView = nil;
636}

750enum AnimationDirection { AnimateIn, AnimateOut };
751static CAAnimation *zoomAnimation(const WebCore::FloatRect& initialFrame, const WebCore::FloatRect& finalFrame, const WebCore::FloatRect& screenFrame, CFTimeInterval duration, float speed, AnimationDirection direction)
752{
753 CABasicAnimation *scaleAnimation = [CABasicAnimation animationWithKeyPath:@"transform"];
754 WebCore::FloatRect scaleRect = smallestRectWithAspectRatioAroundRect(finalFrame.size().aspectRatio(), initialFrame);
755 CGAffineTransform resetOriginTransform = CGAffineTransformMakeTranslation(screenFrame.x() - finalFrame.x(), screenFrame.y() - finalFrame.y());
756 CGAffineTransform scaleTransform = CGAffineTransformMakeScale(scaleRect.width() / finalFrame.width(), scaleRect.height() / finalFrame.height());
757 CGAffineTransform translateTransform = CGAffineTransformMakeTranslation(scaleRect.x() - screenFrame.x(), scaleRect.y() - screenFrame.y());
758
759 CGAffineTransform finalTransform = CGAffineTransformConcat(CGAffineTransformConcat(resetOriginTransform, scaleTransform), translateTransform);
760 NSValue *scaleValue = [NSValue valueWithCATransform3D:CATransform3DMakeAffineTransform(finalTransform)];
761 if (direction == AnimateIn)
762 scaleAnimation.fromValue = scaleValue;
763 else
764 scaleAnimation.toValue = scaleValue;
765
766 scaleAnimation.duration = duration;
767 scaleAnimation.removedOnCompletion = NO;
768 scaleAnimation.fillMode = kCAFillModeBoth;
769 scaleAnimation.timingFunction = timingFunctionForDuration(duration);
770 return scaleAnimation;
771}

783static CAAnimation *maskAnimation(const WebCore::FloatRect& initialFrame, const WebCore::FloatRect& finalFrame, const WebCore::FloatRect& screenFrame, CFTimeInterval duration, float speed, AnimationDirection direction)
784{
785 CABasicAnimation *boundsAnimation = [CABasicAnimation animationWithKeyPath:@"bounds"];
786 WebCore::FloatRect boundsRect = largestRectWithAspectRatioInsideRect(initialFrame.size().aspectRatio(), finalFrame);
787 NSValue *boundsValue = [NSValue valueWithRect:WebCore::FloatRect(WebCore::FloatPoint(), boundsRect.size())];
788 if (direction == AnimateIn)
789 boundsAnimation.fromValue = boundsValue;
790 else
791 boundsAnimation.toValue = boundsValue;
792
793 CABasicAnimation *positionAnimation = [CABasicAnimation animationWithKeyPath:@"position"];
794 NSValue *positionValue = [NSValue valueWithPoint:WebCore::FloatPoint(boundsRect.location() - screenFrame.location())];
795 if (direction == AnimateIn)
796 positionAnimation.fromValue = positionValue;
797 else
798 positionAnimation.toValue = positionValue;
799
800 CAAnimationGroup *animation = [CAAnimationGroup animation];
801 animation.animations = @[boundsAnimation, positionAnimation];
802 animation.duration = duration;
803 animation.removedOnCompletion = NO;
804 animation.fillMode = kCAFillModeBoth;
805 animation.timingFunction = timingFunctionForDuration(duration);
806 return animation;
807}

809static CAAnimation *fadeAnimation(CFTimeInterval duration, AnimationDirection direction)
810{
811 CABasicAnimation *fadeAnimation = [CABasicAnimation animationWithKeyPath:@"backgroundColor"];
812 if (direction == AnimateIn)
813 fadeAnimation.toValue = (id)CGColorGetConstantColor(kCGColorBlack);
814 else
815 fadeAnimation.fromValue = (id)CGColorGetConstantColor(kCGColorBlack);
816 fadeAnimation.duration = duration;
817 fadeAnimation.removedOnCompletion = NO;
818 fadeAnimation.fillMode = kCAFillModeBoth;
819 fadeAnimation.timingFunction = timingFunctionForDuration(duration);
820 return fadeAnimation;
821}

823- (void)_startEnterFullScreenAnimationWithDuration:(NSTimeInterval)duration
824{
825 [CATransaction begin];
826 [CATransaction setDisableActions:YES];
827
828 auto clipLayer = _clipView.get().layer;
829 [clipLayer addAnimation:zoomAnimation(_initialFrame, _finalFrame, self.window.screen.frame, duration, 1, AnimateIn) forKey:@"fullscreen"];
830 [clipLayer.mask addAnimation:maskAnimation(_initialFrame, _finalFrame, self.window.screen.frame, duration, 1, AnimateIn) forKey:@"fullscreen"];
831 [_backgroundView.get().layer addAnimation:fadeAnimation(duration, AnimateIn) forKey:@"fullscreen"];
832
833 [CATransaction commit];
834}

836- (void)_startExitFullScreenAnimationWithDuration:(NSTimeInterval)duration
837{
838 if ([self isFullScreen]) {
839 // We still believe we're in full screen mode, so we must have been asked to exit full
840 // screen by the system full screen button.
841 [self _manager]->requestExitFullScreen();
842 [self exitFullScreen];
843 _fullScreenState = ExitingFullScreen;
844 }
845
846 [[_clipView layer] addAnimation:zoomAnimation(_initialFrame, _finalFrame, self.window.screen.frame, duration, 1, AnimateOut) forKey:@"fullscreen"];
847 NSView* contentView = [[self window] contentView];
848 CALayer *maskLayer = createMask(contentView.bounds);
849 [maskLayer addAnimation:maskAnimation(_initialFrame, _finalFrame, self.window.screen.frame, duration, 1, AnimateOut) forKey:@"fullscreen"];
850 [_clipView layer].mask = maskLayer;
851
852 contentView.hidden = NO;
853 [_backgroundView.get().layer addAnimation:fadeAnimation(duration, AnimateOut) forKey:@"fullscreen"];
854
855 _page->setSuppressVisibilityUpdates(false);
856 ALLOW_DEPRECATED_DECLARATIONS_BEGIN
857 [[self window] setAutodisplay:YES];
858 ALLOW_DEPRECATED_DECLARATIONS_END
859 [[self window] displayIfNeeded];
860 ALLOW_DEPRECATED_DECLARATIONS_BEGIN
861 NSEnableScreenUpdates();
862 ALLOW_DEPRECATED_DECLARATIONS_END
863}
