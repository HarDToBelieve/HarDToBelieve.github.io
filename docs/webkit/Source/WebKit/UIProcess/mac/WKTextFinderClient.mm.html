59class TextFinderFindClient : public API::FindMatchesClient, public API::FindClient {
60 WTF_MAKE_FAST_ALLOCATED;
61public:
62 explicit TextFinderFindClient(WKTextFinderClient *textFinderClient)
63 : m_textFinderClient(textFinderClient)
64 {
65 }
66
67private:
68 void didFindStringMatches(WebPageProxy* page, const String&, const Vector<Vector<WebCore::IntRect>>& matchRects, int32_t) override
69 {
70 [m_textFinderClient didFindStringMatchesWithRects:matchRects didWrapAround:NO];
71 }
72
73 void didGetImageForMatchResult(WebPageProxy* page, WebImage* image, int32_t index) override
74 {
75 [m_textFinderClient didGetImageForMatchResult:image];
76 }
77
78 void didFindString(WebPageProxy*, const String&, const Vector<WebCore::IntRect>& matchRects, uint32_t matchCount, int32_t matchIndex, bool didWrapAround) override
79 {
80 Vector<Vector<WebCore::IntRect>> allMatches;
81 if (matchCount != static_cast<unsigned>(kWKMoreThanMaximumMatchCount)) {
82 // Synthesize a vector of match rects for all `matchCount` matches,
83 // filling in the actual rects for the one that we received.
84 // The rest will remain empty, but it's important to NSTextFinder
85 // that they at least exist.
86 allMatches.resize(matchCount);
87 // FIXME: Clean this up and figure out why we are getting a -1 index
88 if (matchIndex >= 0 && static_cast<uint32_t>(matchIndex) < matchCount)
89 allMatches[matchIndex].appendVector(matchRects);
90 }
91
92 [m_textFinderClient didFindStringMatchesWithRects:allMatches didWrapAround:didWrapAround];
93 }
94
95 void didFailToFindString(WebPageProxy*, const String& string) override
96 {
97 [m_textFinderClient didFindStringMatchesWithRects:{ } didWrapAround:NO];
98 }
99
100 RetainPtr<WKTextFinderClient> m_textFinderClient;
101};

163- (instancetype)initWithPage:(NakedRef<WebKit::WebPageProxy>)page view:(NSView *)view usePlatformFindUI:(BOOL)usePlatformFindUI
164{
165 self = [super init];
166
167 if (!self)
168 return nil;

172 _usePlatformFindUI = usePlatformFindUI;
173

191- (void)replaceMatches:(NSArray *)matches withString:(NSString *)replacementText inSelectionOnly:(BOOL)selectionOnly resultCollector:(void (^)(NSUInteger replacementCount))resultCollector
192{
193 Vector<uint32_t> matchIndices;
194 matchIndices.reserveCapacity(matches.count);
195 for (id match in matches) {
196 if ([match isKindOfClass:WKTextFinderMatch.class])
197 matchIndices.uncheckedAppend([(WKTextFinderMatch *)match index]);
198 }
199 _page->replaceMatches(WTFMove(matchIndices), replacementText, selectionOnly, [collector = makeBlockPtr(resultCollector)] (uint64_t numberOfReplacements) {
200 collector(numberOfReplacements);
201 });
202}

204- (void)findMatchesForString:(NSString *)targetString relativeToMatch:(id <NSTextFinderAsynchronousDocumentFindMatch>)relativeMatch findOptions:(NSTextFinderAsynchronousDocumentFindOptions)findOptions maxResults:(NSUInteger)maxResults resultCollector:(void (^)(NSArray *matches, BOOL didWrap))resultCollector
205{
206 // Limit the number of results, for performance reasons; NSTextFinder always
207 // passes either 1 or NSUIntegerMax, which results in search time being
208 // proportional to document length.
209 maxResults = std::min(maxResults, maximumFindMatches);
210
211 OptionSet<WebKit::FindOptions> kitFindOptions;
212
213 if (findOptions & NSTextFinderAsynchronousDocumentFindOptionsBackwards)
214 kitFindOptions.add(WebKit::FindOptions::Backwards);
215 if (findOptions & NSTextFinderAsynchronousDocumentFindOptionsWrap)
216 kitFindOptions.add(WebKit::FindOptions::WrapAround);
217 if (findOptions & NSTextFinderAsynchronousDocumentFindOptionsCaseInsensitive)
218 kitFindOptions.add(WebKit::FindOptions::CaseInsensitive);
219 if (findOptions & NSTextFinderAsynchronousDocumentFindOptionsStartsWith)
220 kitFindOptions.add(WebKit::FindOptions::AtWordStarts);
221
222 if (!_usePlatformFindUI) {
223 kitFindOptions.add(WebKit::FindOptions::ShowOverlay);
224 kitFindOptions.add(WebKit::FindOptions::ShowFindIndicator);
225 kitFindOptions.add(WebKit::FindOptions::DetermineMatchIndex);
226 }
227
228 RetainPtr<NSProgress> progress = [NSProgress progressWithTotalUnitCount:1];
229 auto copiedResultCollector = Block_copy(resultCollector);
230 _findReplyCallbacks.append([progress, copiedResultCollector] (NSArray *matches, bool didWrap) {
231 [progress setCompletedUnitCount:1];
232 copiedResultCollector(matches, didWrap);
233 Block_release(copiedResultCollector);
234 });
235
236 if (maxResults == 1)
237 _page->findString(targetString, kitFindOptions, maxResults);
238 else
239 _page->findStringMatches(targetString, kitFindOptions, maxResults);
240}

272#pragma mark - FindMatchesClient
273
274- (void)didFindStringMatchesWithRects:(const Vector<Vector<WebCore::IntRect>>&)rectsForMatches didWrapAround:(BOOL)didWrapAround
275{
276 if (_findReplyCallbacks.isEmpty())
277 return;
278
279 unsigned index = 0;
280 auto matchObjects = createNSArray(rectsForMatches, [&] (auto& rects) {
281 RetainPtr<NSArray> rectsArray;
282 if (_usePlatformFindUI)
283 rectsArray = createNSArray(rects);
284 else
285 rectsArray = @[];
286 return adoptNS([[WKTextFinderMatch alloc] initWithClient:self view:_view index:index++ rects:rectsArray.get()]);
287 });
288
289 _findReplyCallbacks.takeFirst()(matchObjects.get(), didWrapAround);
290}
291
292- (void)didGetImageForMatchResult:(WebKit::WebImage *)image
293{
294 if (_imageReplyCallbacks.isEmpty())
295 return;
296
297 WebCore::IntSize size = image->bitmap().size();
298 size.scale(1 / _page->deviceScaleFactor());
299
300 auto imageCallback = _imageReplyCallbacks.takeFirst();
301 imageCallback(adoptNS([[NSImage alloc] initWithCGImage:image->bitmap().makeCGImage().get() size:size]).get());
302}
303
304#pragma mark - WKTextFinderMatch callbacks
305
306- (void)getImageForMatchResult:(id <NSTextFinderAsynchronousDocumentFindMatch>)findMatch completionHandler:(void (^)(NSImage *generatedImage))completionHandler
307{
308 if (!_usePlatformFindUI) {
309 completionHandler(nil);
310 return;
311 }
312
313 ASSERT([findMatch isKindOfClass:[WKTextFinderMatch class]]);
314
315 WKTextFinderMatch *textFinderMatch = static_cast<WKTextFinderMatch *>(findMatch);
316
317 auto copiedImageCallback = Block_copy(completionHandler);
318 _imageReplyCallbacks.append([copiedImageCallback] (NSImage *image) {
319 copiedImageCallback(image);
320 Block_release(copiedImageCallback);
321 });
322
323 // FIXME: There is no guarantee that this will ever result in didGetImageForMatchResult
324 // being called (and thus us calling our completion handler); we should harden this
325 // against all of the early returns in FindController::getImageForFindMatch.
326 _page->getImageForFindMatch(textFinderMatch.index);
327}
