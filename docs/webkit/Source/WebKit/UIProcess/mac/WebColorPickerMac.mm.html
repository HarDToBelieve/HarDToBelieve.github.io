131@implementation WKPopoverColorWell

133+ (NSPopover *)_colorPopoverCreateIfNecessary:(BOOL)forceCreation
134{
135 static NeverDestroyed<RetainPtr<NSPopover>> colorPopover;
136 if (forceCreation) {
137 auto popover = adoptNS([[NSPopover alloc] init]);
138 [popover _setRequiresCorrectContentAppearance:YES];
139 [popover setBehavior:NSPopoverBehaviorTransient];
140
141 auto controller = adoptNS([[NSClassFromString(@"NSColorPopoverController") alloc] init]);
142 [popover setContentViewController:controller.get()];
143 [controller setPopover:popover.get()];
144
145 colorPopover.get() = WTFMove(popover);
146 }
147
148 return colorPopover.get().get();
149}
150
151- (id <WKPopoverColorWellDelegate>)webDelegate
152{
153 return _webDelegate.getAutoreleased();
154}
155
156- (void)setWebDelegate:(id <WKPopoverColorWellDelegate>)webDelegate
157{
158 _webDelegate = webDelegate;
159}

161- (void)_showPopover
162{
163 NSPopover *popover = [[self class] _colorPopoverCreateIfNecessary:YES];
164 popover.delegate = self;
165
166 [self deactivate];
167
168 // Deactivate previous NSPopoverColorWell
169 NSColorWell *owner = [NSColorWell _exclusiveColorPanelOwner];
170 if ([owner isKindOfClass:[NSPopoverColorWell class]])
171 [owner deactivate];
172
173 NSColorPopoverController *controller = (NSColorPopoverController *)[popover contentViewController];
174 controller.delegate = self;
175
176 if (_suggestedColors) {
177 NSUInteger numColors = [[_suggestedColors allKeys] count];
178 CGFloat swatchWidth = (colorPickerMatrixNumColumns * colorPickerMatrixSwatchWidth + (colorPickerMatrixNumColumns * colorPickerMatrixBorderWidth - numColors)) / numColors;
179 CGFloat swatchHeight = colorPickerMatrixSwatchWidth;
180
181 // topBarMatrixView cannot be accessed until view has been loaded
182 if (!controller.isViewLoaded)
183 [controller loadView];
184
185 NSColorPickerMatrixView *topMatrix = controller.topBarMatrixView;
186 [topMatrix setNumberOfColumns:numColors];
187 [topMatrix setSwatchSize:NSMakeSize(swatchWidth, swatchHeight)];
188 [topMatrix setColorList:_suggestedColors.get()];
189 }
190
191 [self activate:YES];
192 [popover showRelativeToRect:self.bounds ofView:self preferredEdge:NSMinYEdge];
193}

211@implementation WKColorPopoverMac
212- (id)initWithFrame:(const WebCore::IntRect &)rect inView:(NSView *)view
213{
214 if(!(self = [super init]))
215 return self;
216
217 _popoverWell = adoptNS([[WKPopoverColorWell alloc] initWithFrame:[view convertRect:NSRectFromCGRect(rect) toView:nil]]);
218 if (!_popoverWell)
219 return self;
220
221 [_popoverWell setAlphaValue:0.0];
222 [[view window].contentView addSubview:_popoverWell.get()];
223
224 return self;
225}
226
227- (void)setAndShowPicker:(WebKit::WebColorPickerMac*)picker withColor:(NSColor *)color suggestions:(Vector<WebCore::Color>&&)suggestions
228{
229 _picker = picker;
230
231 [_popoverWell setTarget:self];
232 [_popoverWell setWebDelegate:self];
233 [_popoverWell setAction:@selector(didChooseColor:)];
234 [_popoverWell setColor:color];
235
236 RetainPtr<NSColorList> suggestedColors;
237 if (suggestions.size()) {
238 suggestedColors = adoptNS([[NSColorList alloc] init]);
239 for (size_t i = 0; i < std::min(suggestions.size(), maxColorSuggestions); i++)
240 [suggestedColors insertColor:nsColor(suggestions.at(i)) key:@(i).stringValue atIndex:i];
241 }
242
243 [_popoverWell setSuggestedColors:suggestedColors.get()];
244 [_popoverWell _showPopover];
245
246 [[NSColorPanel sharedColorPanel] setDelegate:self];
247
248 _lastChangedByUser = YES;
249}
250
251- (void)invalidate
252{
253 [_popoverWell removeFromSuperviewWithoutNeedingDisplay];
254 [_popoverWell setTarget:nil];
255 [_popoverWell setAction:nil];
256 [_popoverWell deactivate];
257
258 _popoverWell = nil;
259 _picker = nil;
260
261 NSColorPanel *panel = [NSColorPanel sharedColorPanel];
262 if (panel.delegate == self) {
263 panel.delegate = nil;
264 [panel close];
265 }
266}
267
268- (void)windowWillClose:(NSNotification *)notification
269{
270 if (!_picker)
271 return;
272
273 if (notification.object == [NSColorPanel sharedColorPanel]) {
274 _lastChangedByUser = YES;
275 _picker->endPicker();
276 }
277}
278
279- (void)didChooseColor:(id)sender
280{
281 if (sender != _popoverWell)
282 return;
283
284 // Handle the case where the <input type='color'> value is programmatically set.
285 if (!_lastChangedByUser) {
286 _lastChangedByUser = YES;
287 return;
288 }
289
290 _picker->didChooseColor(WebCore::colorFromNSColor([_popoverWell color]));
291}
292
293- (void)setColor:(NSColor *)color
294{
295 _lastChangedByUser = NO;
296 [_popoverWell setColor:color];
297}
298
299- (void)didClosePopover
300{
301 if (!_picker)
302 return;
303
304 if (![NSColorPanel sharedColorPanel].isVisible)
305 _picker->endPicker();
306}
