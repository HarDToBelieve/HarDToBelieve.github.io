34constexpr CGFloat kCalendarWidth = 139;
35constexpr CGFloat kCalendarHeight = 148;
36constexpr CGFloat kCalendarCornerRadius = 10;
37constexpr CGFloat kWindowBorderSize = 0.5;
38constexpr NSString * kDateFormatString = @"yyyy-MM-dd";
39constexpr NSString * kDateTimeFormatString = @"yyyy-MM-dd'T'HH:mm";
40constexpr NSString * kDateTimeWithSecondsFormatString = @"yyyy-MM-dd'T'HH:mm:ss";
41constexpr NSString * kDateTimeWithMillisecondsFormatString = @"yyyy-MM-dd'T'HH:mm:ss.SSS";
42constexpr NSString * kDefaultLocaleIdentifier = @"en_US_POSIX";
43constexpr NSString * kDefaultTimeZoneIdentifier = @"UTC";

62Ref<WebDateTimePickerMac> WebDateTimePickerMac::create(WebPageProxy& page, NSView *view)
63{
64 return adoptRef(*new WebDateTimePickerMac(page, view));
65}
66
67WebDateTimePickerMac::~WebDateTimePickerMac()
68{
69 [m_picker invalidate];
70}
71
72WebDateTimePickerMac::WebDateTimePickerMac(WebPageProxy& page, NSView *view)
73 : WebDateTimePicker(page)
74 , m_view(view)
75{
76}
77
78void WebDateTimePickerMac::endPicker()
79{
80 [m_picker invalidate];
81 m_picker = nil;
82 WebDateTimePicker::endPicker();
83}
84
85void WebDateTimePickerMac::showDateTimePicker(WebCore::DateTimeChooserParameters&& params)
86{
87 if (m_picker) {
88 [m_picker updatePicker:WTFMove(params)];
89 return;
90 }
91
92 m_picker = adoptNS([[WKDateTimePicker alloc] initWithParams:WTFMove(params) inView:m_view.get().get()]);
93 [m_picker showPicker:*this];
94}
95
96void WebDateTimePickerMac::didChooseDate(StringView date)
97{
98 if (!m_page)
99 return;
100
101 m_page->didChooseDate(date);
102}

110- (id)initWithContentRect:(NSRect)contentRect styleMask:(NSUInteger)styleMask backing:(NSBackingStoreType)backingStoreType defer:(BOOL)defer
111{
112 self = [super initWithContentRect:contentRect styleMask:styleMask backing:backingStoreType defer:defer];
113 if (!self)
114 return nil;
115
116 self.hasShadow = YES;
117 self.releasedWhenClosed = NO;
118 self.titleVisibility = NSWindowTitleHidden;
119 self.titlebarAppearsTransparent = YES;
120 self.movable = NO;
121 self.backgroundColor = [NSColor clearColor];
122 self.opaque = NO;
123
124 _backdropView = adoptNS([[WKDateTimePickerBackdropView alloc] initWithFrame:contentRect]);
125 [_backdropView setAutoresizingMask:NSViewWidthSizable | NSViewHeightSizable];
126 [self setContentView:_backdropView.get()];
127
128 return self;
129}

148@implementation WKDateTimePickerBackdropView

150- (void)drawRect:(NSRect)dirtyRect
151{
152 [NSGraphicsContext saveGraphicsState];
153
154 [[NSColor controlBackgroundColor] setFill];
155
156 NSRect rect = NSInsetRect(self.frame, kWindowBorderSize, 0);
157 NSPoint topLeft = NSMakePoint(NSMinX(rect), NSMaxY(rect));
158 NSPoint topRight = NSMakePoint(NSMaxX(rect), NSMaxY(rect));
159 NSPoint bottomRight = NSMakePoint(NSMaxX(rect), NSMinY(rect));
160 NSPoint bottomLeft = NSMakePoint(NSMinX(rect), NSMinY(rect));
161
162 NSBezierPath *path = [NSBezierPath bezierPath];
163 [path moveToPoint:topLeft];
164 [path lineToPoint:NSMakePoint(topRight.x - kCalendarCornerRadius, topRight.y)];
165 [path curveToPoint:NSMakePoint(topRight.x, topRight.y - kCalendarCornerRadius) controlPoint1:topRight controlPoint2:topRight];
166 [path lineToPoint:NSMakePoint(bottomRight.x, bottomRight.y + kCalendarCornerRadius)];
167 [path curveToPoint:NSMakePoint(bottomRight.x - kCalendarCornerRadius, bottomRight.y) controlPoint1:bottomRight controlPoint2:bottomRight];
168 [path lineToPoint:NSMakePoint(bottomLeft.x + kCalendarCornerRadius, bottomLeft.y)];
169 [path curveToPoint:NSMakePoint(bottomLeft.x, bottomLeft.y + kCalendarCornerRadius) controlPoint1:bottomLeft controlPoint2:bottomLeft];
170 [path lineToPoint:topLeft];
171
172 [path fill];
173
174 [NSGraphicsContext restoreGraphicsState];
175}

179@implementation WKDateTimePicker {
180 WeakPtr<WebKit::WebDateTimePickerMac> _picker;
181 WebCore::DateTimeChooserParameters _params;
182 WeakObjCPtr<NSView> _presentingView;
183
184 RetainPtr<WKDateTimePickerWindow> _enclosingWindow;
185 RetainPtr<NSDatePicker> _datePicker;
186 RetainPtr<NSDateFormatter> _dateFormatter;
187}
188
189- (id)initWithParams:(WebCore::DateTimeChooserParameters&&)params inView:(NSView *)view
190{
191 if (!(self = [super init]))
192 return self;
193
194 _presentingView = view;
195
196 NSRect windowRect = [[_presentingView window] convertRectToScreen:[_presentingView convertRect:params.anchorRectInRootView toView:nil]];
197 windowRect.origin.y = NSMinY(windowRect) - kCalendarHeight;
198 windowRect.size.width = kCalendarWidth;
199 windowRect.size.height = kCalendarHeight;
200
201 // Use a UTC timezone as all incoming double values are UTC timestamps. This also ensures that
202 // the date value of the NSDatePicker matches the date value returned by JavaScript. The timezone
203 // has no effect on the value returned to the WebProcess, as a timezone-agnostic format string is
204 // used to return the date.
205 NSTimeZone *timeZone = [NSTimeZone timeZoneWithName:kDefaultTimeZoneIdentifier];
206
207 _enclosingWindow = adoptNS([[WKDateTimePickerWindow alloc] initWithContentRect:NSZeroRect styleMask:NSWindowStyleMaskBorderless backing:NSBackingStoreBuffered defer:NO]);
208 [_enclosingWindow setFrame:windowRect display:YES];
209
210 _datePicker = adoptNS([[NSDatePicker alloc] initWithFrame:[_enclosingWindow contentView].bounds]);
211 [_datePicker setBezeled:NO];
212 [_datePicker setDrawsBackground:NO];
213 [_datePicker setDatePickerStyle:NSDatePickerStyleClockAndCalendar];
214 [_datePicker setDatePickerElements:NSDatePickerElementFlagYearMonthDay];
215 [_datePicker setTimeZone:timeZone];
216 [_datePicker setTarget:self];
217 [_datePicker setAction:@selector(didChooseDate:)];
218
219 auto englishLocale = adoptNS([[NSLocale alloc] initWithLocaleIdentifier:kDefaultLocaleIdentifier]);
220 _dateFormatter = adoptNS([[NSDateFormatter alloc] init]);
221 [_dateFormatter setLocale:englishLocale.get()];
222 [_dateFormatter setTimeZone:timeZone];
223
224 [self updatePicker:WTFMove(params)];
225
226 return self;
227}
228
229- (void)showPicker:(WebKit::WebDateTimePickerMac&)picker
230{
231 _picker = makeWeakPtr(picker);
232
233 [[_enclosingWindow contentView] addSubview:_datePicker.get()];
234 [[_presentingView window] addChildWindow:_enclosingWindow.get() ordered:NSWindowAbove];
235}
236
237- (void)updatePicker:(WebCore::DateTimeChooserParameters&&)params
238{
239 _params = WTFMove(params);
240
241 NSString *currentDateValueString = _params.currentValue;
242
243 [_dateFormatter setDateFormat:[self dateFormatStringForType:_params.type]];
244
245 if (![currentDateValueString length])
246 [_datePicker setDateValue:[self initialDateForEmptyValue]];
247 else
248 [_datePicker setDateValue:[_dateFormatter dateFromString:currentDateValueString]];
249
250 [_datePicker setMinDate:[NSDate dateWithTimeIntervalSince1970:_params.minimum / 1000.0]];
251 [_datePicker setMaxDate:[NSDate dateWithTimeIntervalSince1970:_params.maximum / 1000.0]];
252
253 [_enclosingWindow setAppearance:[NSAppearance appearanceNamed:_params.useDarkAppearance ? NSAppearanceNameDarkAqua : NSAppearanceNameAqua]];
254}
255
256- (void)invalidate
257{
258 [_datePicker removeFromSuperviewWithoutNeedingDisplay];
259 [_datePicker setTarget:nil];
260 [_datePicker setAction:nil];
261 _datePicker = nil;
262
263 _dateFormatter = nil;
264
265 [[_presentingView window] removeChildWindow:_enclosingWindow.get()];
266 [_enclosingWindow close];
267 _enclosingWindow = nil;
268}
269
270- (void)didChooseDate:(id)sender
271{
272 if (sender != _datePicker)
273 return;
274
275 String dateString = [_dateFormatter stringFromDate:[_datePicker dateValue]];
276 _picker->didChooseDate(StringView(dateString));
277}
278
279- (NSString *)dateFormatStringForType:(NSString *)type
280{
281 if ([type isEqualToString:@"datetime-local"]) {
282 if (_params.hasMillisecondField)
283 return kDateTimeWithMillisecondsFormatString;
284 if (_params.hasSecondField)
285 return kDateTimeWithSecondsFormatString;
286 return kDateTimeFormatString;
287 }
288
289 return kDateFormatString;
290}
291
292- (NSDate *)initialDateForEmptyValue
293{
294 NSDate *now = [NSDate date];
295 NSTimeZone *defaultTimeZone = [NSTimeZone defaultTimeZone];
296 NSInteger offset = [defaultTimeZone secondsFromGMTForDate:now];
297 return [now dateByAddingTimeInterval:offset];
298}
