36void WindowServerConnection::applicationWindowModificationsStopped(bool stopped)
37{
38 if (m_applicationWindowModificationsHaveStopped == stopped)
39 return;
40 m_applicationWindowModificationsHaveStopped = stopped;
41 windowServerConnectionStateChanged();
42}
43
44void WindowServerConnection::windowServerConnectionStateChanged()
45{
46 for (auto& processPool : WebProcessPool::allProcessPools())
47 processPool->windowServerConnectionStateChanged();
48}

50WindowServerConnection& WindowServerConnection::singleton()
51{
52 static WindowServerConnection& windowServerConnection = *new WindowServerConnection;
53 return windowServerConnection;
54}

56static bool registerOcclusionNotificationHandler(CGSNotificationType type, CGSNotifyConnectionProcPtr handler)
57{
58 CGSConnectionID mainConnection = CGSMainConnectionID();
59 static bool notificationsEnabled;
60 if (!notificationsEnabled) {
61 if (CGSPackagesEnableConnectionOcclusionNotifications(mainConnection, true, nullptr) != kCGErrorSuccess)
62 return false;
63 if (CGSPackagesEnableConnectionWindowModificationNotifications(mainConnection, true, nullptr) != kCGErrorSuccess)
64 return false;
65 notificationsEnabled = true;
66 }
67
68 return CGSRegisterConnectionNotifyProc(mainConnection, handler, type, nullptr) == kCGErrorSuccess;
69}
70
71WindowServerConnection::WindowServerConnection()
72 : m_applicationWindowModificationsHaveStopped(false)
73{
74 struct OcclusionNotificationHandler {
75 CGSNotificationType notificationType;
76 CGSNotifyConnectionProcPtr handler;
77 const char* name;
78 };
79
80 static auto windowModificationsStarted = [](CGSNotificationType, void*, uint32_t, void*, CGSConnectionID) {
81 WindowServerConnection::singleton().applicationWindowModificationsStopped(false);
82 };
83
84 static auto windowModificationsStopped = [](CGSNotificationType, void*, uint32_t, void*, CGSConnectionID) {
85 WindowServerConnection::singleton().applicationWindowModificationsStopped(true);
86 };
87
88 static const OcclusionNotificationHandler occlusionNotificationHandlers[] = {
89 { kCGSConnectionWindowModificationsStarted, windowModificationsStarted, "Application Window Modifications Started" },
90 { kCGSConnectionWindowModificationsStopped, windowModificationsStopped, "Application Window Modifications Stopped" },
91 };
92
93 for (const auto& occlusionNotificationHandler : occlusionNotificationHandlers) {
94 bool result = registerOcclusionNotificationHandler(occlusionNotificationHandler.notificationType, occlusionNotificationHandler.handler);
95 UNUSED_PARAM(result);
96 ASSERT_WITH_MESSAGE(result, "Registration of \"%s\" notification handler failed.\n", occlusionNotificationHandler.name);
97 }
98}
