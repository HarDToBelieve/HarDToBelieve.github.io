26#pragma once
27
28#include <WebCore/DOMWindow.h>
29#include <wtf/Forward.h>
30
31namespace WebCore {
32class Frame;
33}
34
35namespace WebKit {
36
37class WebAutomationDOMWindowObserver final : public RefCounted<WebAutomationDOMWindowObserver>, public WebCore::DOMWindow::Observer {
38public:
39 static Ref<WebAutomationDOMWindowObserver> create(WebCore::DOMWindow& window, WTF::Function<void(WebAutomationDOMWindowObserver&)>&& callback)
40 {
41 return adoptRef(*new WebAutomationDOMWindowObserver(window, WTFMove(callback)));
42 }
43
44 ~WebAutomationDOMWindowObserver();
45
46 // All of these observer callbacks are interpreted as a signal that a frame has been detached and
47 // can no longer accept new commands nor finish pending commands (eg, evaluating JavaScript).
48 void willDestroyGlobalObjectInCachedFrame() final;
49 void willDestroyGlobalObjectInFrame() final;
50 void willDetachGlobalObjectFromFrame() final;
51
52private:
53 WebAutomationDOMWindowObserver(WebCore::DOMWindow&, WTF::Function<void(WebAutomationDOMWindowObserver&)>&&);
54
55 WeakPtr<WebCore::DOMWindow> m_window;
56 bool m_wasDetached { false };
57 WTF::Function<void(WebAutomationDOMWindowObserver&)> m_callback;
58};
59
60} // namespace WebKit
