72template <typename T>
73static JSObjectRef toJSArray(JSContextRef context, const Vector<T>& data, JSValueRef (*converter)(JSContextRef, const T&), JSValueRef* exception)
74{
75 ASSERT_ARG(converter, converter);
76
77 if (data.isEmpty())
78 return JSObjectMakeArray(context, 0, nullptr, exception);
79
80 Vector<JSValueRef, 8> convertedData;
81 convertedData.reserveCapacity(data.size());
82
83 for (auto& originalValue : data) {
84 JSValueRef convertedValue = converter(context, originalValue);
85 JSValueProtect(context, convertedValue);
86 convertedData.uncheckedAppend(convertedValue);
87 }
88
89 JSObjectRef array = JSObjectMakeArray(context, convertedData.size(), convertedData.data(), exception);
90
91 for (auto& convertedValue : convertedData)
92 JSValueUnprotect(context, convertedValue);
93
94 return array;
95}

102static inline JSValueRef callPropertyFunction(JSContextRef context, JSObjectRef object, const String& propertyName, size_t argumentCount, const JSValueRef arguments[], JSValueRef* exception)
103{
104 ASSERT_ARG(object, object);
105 ASSERT_ARG(object, JSValueIsObject(context, object));
106
107 JSObjectRef function = const_cast<JSObjectRef>(JSObjectGetProperty(context, object, OpaqueJSString::tryCreate(propertyName).get(), exception));
108 ASSERT(JSObjectIsFunction(context, function));
109
110 return JSObjectCallAsFunction(context, function, object, argumentCount, arguments, exception);
111}

127static bool isValidNodeHandle(const String& nodeHandle)
128{
129 // Node identifier has the following format:
130 // node-XXXXXXXX-XXXX-XXXX-XXXX-XXXXXXXXXXXX
131 // 01234567890123456789012345678901234567890
132 // where X is a valid hexadecimal value in upper case.
133 if (nodeHandle.length() != 41)
134 return false;
135
136 if (nodeHandle[0] != 'n' || nodeHandle[1] != 'o' || nodeHandle[2] != 'd' || nodeHandle[3] != 'e')
137 return false;
138
139 for (unsigned i = 4; i < 41; ++i) {
140 switch (i) {
141 case 4:
142 case 13:
143 case 18:
144 case 23:
145 case 28:
146 if (nodeHandle[i] != '-')
147 return false;
148 break;
149 default:
150 if (!(nodeHandle[i] >= '0' && nodeHandle[i] <= '9') && !(nodeHandle[i] >= 'A' && nodeHandle[i] <= 'F'))
151 return false;
152 break;
153 }
154 }
155
156 return true;
157}

159static JSValueRef isValidNodeIdentifier(JSContextRef context, JSObjectRef function, JSObjectRef thisObject, size_t argumentCount, const JSValueRef arguments[], JSValueRef* exception)
160{
161 ASSERT_ARG(argumentCount, argumentCount == 1);
162 ASSERT_ARG(arguments, JSValueIsString(context, arguments[0]));
163
164 if (argumentCount != 1)
165 return JSValueMakeUndefined(context);
166
167 auto nodeIdentifier = adoptRef(JSValueToStringCopy(context, arguments[0], exception));
168 return JSValueMakeBoolean(context, isValidNodeHandle(nodeIdentifier->string()));
169}

188static JSValueRef evaluateJavaScriptCallback(JSContextRef context, JSObjectRef function, JSObjectRef thisObject, size_t argumentCount, const JSValueRef arguments[], JSValueRef* exception)
189{
190 ASSERT_ARG(argumentCount, argumentCount == 3);
191 ASSERT_ARG(arguments, JSValueIsNumber(context, arguments[0]));
192 ASSERT_ARG(arguments, JSValueIsNumber(context, arguments[1]));
193 ASSERT_ARG(arguments, JSValueIsObject(context, arguments[2]) || JSValueIsString(context, arguments[2]));
194
195 auto automationSessionProxy = WebProcess::singleton().automationSessionProxy();
196 if (!automationSessionProxy)
197 return JSValueMakeUndefined(context);
198
199 WebCore::FrameIdentifier frameID = makeObjectIdentifier<WebCore::FrameIdentifierType>(JSValueToNumber(context, arguments[0], exception));
200 uint64_t callbackID = JSValueToNumber(context, arguments[1], exception);
201 if (JSValueIsString(context, arguments[2])) {
202 auto result = adoptRef(JSValueToStringCopy(context, arguments[2], exception));
203 automationSessionProxy->didEvaluateJavaScriptFunction(frameID, callbackID, result->string(), { });
204 } else if (JSValueIsObject(context, arguments[2])) {
205 JSObjectRef error = JSValueToObject(context, arguments[2], exception);
206 JSValueRef nameValue = JSObjectGetProperty(context, error, OpaqueJSString::tryCreate("name"_s).get(), exception);
207 String exceptionName = adoptRef(JSValueToStringCopy(context, nameValue, nullptr))->string();
208 String errorType = Inspector::Protocol::AutomationHelpers::getEnumConstantValue(Inspector::Protocol::Automation::ErrorMessage::JavaScriptError);
209 if (exceptionName == "JavaScriptTimeout")
210 errorType = Inspector::Protocol::AutomationHelpers::getEnumConstantValue(Inspector::Protocol::Automation::ErrorMessage::JavaScriptTimeout);
211 else if (exceptionName == "NodeNotFound")
212 errorType = Inspector::Protocol::AutomationHelpers::getEnumConstantValue(Inspector::Protocol::Automation::ErrorMessage::NodeNotFound);
213 else if (exceptionName == "InvalidNodeIdentifier")
214 errorType = Inspector::Protocol::AutomationHelpers::getEnumConstantValue(Inspector::Protocol::Automation::ErrorMessage::InvalidNodeIdentifier);
215 else if (exceptionName == "InvalidElementState")
216 errorType = Inspector::Protocol::AutomationHelpers::getEnumConstantValue(Inspector::Protocol::Automation::ErrorMessage::InvalidElementState);
217 else if (exceptionName == "InvalidParameter")
218 errorType = Inspector::Protocol::AutomationHelpers::getEnumConstantValue(Inspector::Protocol::Automation::ErrorMessage::InvalidParameter);
219 else if (exceptionName == "InvalidSelector")
220 errorType = Inspector::Protocol::AutomationHelpers::getEnumConstantValue(Inspector::Protocol::Automation::ErrorMessage::InvalidSelector);
221 else if (exceptionName == "ElementNotInteractable")
222 errorType = Inspector::Protocol::AutomationHelpers::getEnumConstantValue(Inspector::Protocol::Automation::ErrorMessage::ElementNotInteractable);
223
224 JSValueRef messageValue = JSObjectGetProperty(context, error, OpaqueJSString::tryCreate("message"_s).get(), exception);
225 auto exceptionMessage = adoptRef(JSValueToStringCopy(context, messageValue, exception))->string();
226 automationSessionProxy->didEvaluateJavaScriptFunction(frameID, callbackID, exceptionMessage, errorType);
227 } else {
228 String errorType = Inspector::Protocol::AutomationHelpers::getEnumConstantValue(Inspector::Protocol::Automation::ErrorMessage::InternalError);
229 automationSessionProxy->didEvaluateJavaScriptFunction(frameID, callbackID, { }, errorType);
230 }
231
232 return JSValueMakeUndefined(context);
233}

235JSObjectRef WebAutomationSessionProxy::scriptObject(JSGlobalContextRef context)
236{
237 JSC::JSGlobalObject* globalObject = toJS(context);
238 JSC::VM& vm = globalObject->vm();
239 JSC::JSLockHolder locker(vm);
240 auto scriptObjectID = JSC::Identifier::fromUid(m_scriptObjectIdentifier);
241 if (!globalObject->hasProperty(globalObject, scriptObjectID))
242 return nullptr;
243
244 return const_cast<JSObjectRef>(toRef(globalObject, globalObject->get(globalObject, scriptObjectID)));
245}

248{
249 JSC::JSGlobalObject* globalObject = toJS(context);
250 JSC::VM& vm = globalObject->vm();
251 JSC::JSLockHolder locker(vm);
252 auto scriptObjectID = JSC::Identifier::fromUid(m_scriptObjectIdentifier);
253 JSC::PutPropertySlot slot(globalObject);
254 globalObject->methodTable(vm)->put(globalObject, globalObject, scriptObjectID, toJS(globalObject, object), slot);
255}

257JSObjectRef WebAutomationSessionProxy::scriptObjectForFrame(WebFrame& frame)
258{
259 JSGlobalContextRef context = frame.jsContext();
260 if (auto* scriptObject = this->scriptObject(context))
261 return scriptObject;
262
263 JSValueRef exception = nullptr;
264 String script = StringImpl::createWithoutCopying(WebAutomationSessionProxyScriptSource, sizeof(WebAutomationSessionProxyScriptSource));
265 JSObjectRef scriptObjectFunction = const_cast<JSObjectRef>(JSEvaluateScript(context, OpaqueJSString::tryCreate(script).get(), nullptr, nullptr, 0, &exception));
266 ASSERT(JSValueIsObject(context, scriptObjectFunction));
267
268 JSValueRef sessionIdentifier = toJSValue(context, m_sessionIdentifier);
269 JSObjectRef evaluateFunction = JSObjectMakeFunctionWithCallback(context, nullptr, evaluate);
270 JSObjectRef createUUIDFunction = JSObjectMakeFunctionWithCallback(context, nullptr, createUUID);
271 JSObjectRef isValidNodeIdentifierFunction = JSObjectMakeFunctionWithCallback(context, nullptr, isValidNodeIdentifier);
272 JSValueRef arguments[] = { sessionIdentifier, evaluateFunction, createUUIDFunction, isValidNodeIdentifierFunction };
273 JSObjectRef scriptObject = const_cast<JSObjectRef>(JSObjectCallAsFunction(context, scriptObjectFunction, nullptr, WTF_ARRAY_LENGTH(arguments), arguments, &exception));
274 ASSERT(JSValueIsObject(context, scriptObject));
275
276 setScriptObject(context, scriptObject);
277 return scriptObject;
278}

280WebCore::Element* WebAutomationSessionProxy::elementForNodeHandle(WebFrame& frame, const String& nodeHandle)
281{
282 // Don't use scriptObjectForFrame() since we can assume if the script object
283 // does not exist, there are no nodes mapped to handles. Using scriptObjectForFrame()
284 // will make a new script object if it can't find one, preventing us from returning fast.
285 JSGlobalContextRef context = frame.jsContext();
286 auto* scriptObject = this->scriptObject(context);
287 if (!scriptObject)
288 return nullptr;
289
290 JSValueRef functionArguments[] = {
291 toJSValue(context, nodeHandle)
292 };
293
294 JSValueRef result = callPropertyFunction(context, scriptObject, "nodeForIdentifier"_s, WTF_ARRAY_LENGTH(functionArguments), functionArguments, nullptr);
295 JSObjectRef element = JSValueToObject(context, result, nullptr);
296 if (!element)
297 return nullptr;
298
299 auto elementWrapper = JSC::jsDynamicCast<WebCore::JSElement*>(toJS(context)->vm(), toJS(element));
300 if (!elementWrapper)
301 return nullptr;
302
303 return &elementWrapper->wrapped();
304}

306void WebAutomationSessionProxy::ensureObserverForFrame(WebFrame& frame)
307{
308 // If the frame and DOMWindow have become disconnected, then frame is already being destroyed
309 // and there is no way to get access to the frame from the observer's DOMWindow reference.
310 if (!frame.coreFrame()->window() || !frame.coreFrame()->window()->frame())
311 return;
312
313 if (m_frameObservers.contains(frame.frameID()))
314 return;
315
316 auto frameID = frame.frameID();
317 m_frameObservers.set(frameID, WebAutomationDOMWindowObserver::create(*frame.coreFrame()->window(), [this, frameID] (WebAutomationDOMWindowObserver&) {
318 willDestroyGlobalObjectForFrame(frameID);
319 }));
320}

322void WebAutomationSessionProxy::didClearWindowObjectForFrame(WebFrame& frame)
323{
324 willDestroyGlobalObjectForFrame(frame.frameID());
325}

327void WebAutomationSessionProxy::willDestroyGlobalObjectForFrame(WebCore::FrameIdentifier frameID)
328{
329 // The observer is no longer needed, let it become GC'd and unregister itself from DOMWindow.
330 if (m_frameObservers.contains(frameID))
331 m_frameObservers.remove(frameID);
332
333 String errorMessage = "Callback was not called before the unload event."_s;
334 String errorType = Inspector::Protocol::AutomationHelpers::getEnumConstantValue(Inspector::Protocol::Automation::ErrorMessage::FrameNotFound);
335
336 auto pendingFrameCallbacks = m_webFramePendingEvaluateJavaScriptCallbacksMap.take(frameID);
337 for (uint64_t callbackID : pendingFrameCallbacks)
338 WebProcess::singleton().parentProcessConnection()->send(Messages::WebAutomationSession::DidEvaluateJavaScriptFunction(callbackID, errorMessage, errorType), 0);
339}

341void WebAutomationSessionProxy::evaluateJavaScriptFunction(WebCore::PageIdentifier pageID, std::optional<WebCore::FrameIdentifier> optionalFrameID, const String& function, Vector<String> arguments, bool expectsImplicitCallbackArgument, std::optional<double> callbackTimeout, uint64_t callbackID)
342{
343 WebPage* page = WebProcess::singleton().webPage(pageID);
344 if (!page) {
345 WebProcess::singleton().parentProcessConnection()->send(Messages::WebAutomationSession::DidEvaluateJavaScriptFunction(callbackID, { },
346 Inspector::Protocol::AutomationHelpers::getEnumConstantValue(Inspector::Protocol::Automation::ErrorMessage::WindowNotFound)), 0);
347 return;
348 }
349 auto* frame = optionalFrameID ? WebProcess::singleton().webFrame(*optionalFrameID) : &page->mainWebFrame();
350 if (!frame || !frame->coreFrame()->window() || !frame->coreFrame()->window()->frame()) {
351 WebProcess::singleton().parentProcessConnection()->send(Messages::WebAutomationSession::DidEvaluateJavaScriptFunction(callbackID, { },
352 Inspector::Protocol::AutomationHelpers::getEnumConstantValue(Inspector::Protocol::Automation::ErrorMessage::FrameNotFound)), 0);
353 return;
354 }
355
356 // No need to track the main frame, this is handled by didClearWindowObjectForFrame.
357 if (!frame->coreFrame()->isMainFrame())
358 ensureObserverForFrame(*frame);
359
360 JSObjectRef scriptObject = scriptObjectForFrame(*frame);
361 ASSERT(scriptObject);
362
363 auto frameID = frame->frameID();
364 JSValueRef exception = nullptr;
365 JSGlobalContextRef context = frame->jsContext();
366
367 if (expectsImplicitCallbackArgument) {
368 auto result = m_webFramePendingEvaluateJavaScriptCallbacksMap.add(frameID, Vector<uint64_t>());
369 result.iterator->value.append(callbackID);
370 }
371
372 JSValueRef functionArguments[] = {
373 toJSValue(context, function),
374 toJSArray(context, arguments, toJSValue, &exception),
375 JSValueMakeBoolean(context, expectsImplicitCallbackArgument),
376 JSValueMakeNumber(context, frameID.toUInt64()),
377 JSValueMakeNumber(context, callbackID),
378 JSObjectMakeFunctionWithCallback(context, nullptr, evaluateJavaScriptCallback),
379 JSValueMakeNumber(context, callbackTimeout.value_or(-1))
380 };
381
382 {
383 WebCore::UserGestureIndicator gestureIndicator(WebCore::ProcessingUserGesture, frame->coreFrame()->document());
384 callPropertyFunction(context, scriptObject, "evaluateJavaScriptFunction"_s, WTF_ARRAY_LENGTH(functionArguments), functionArguments, &exception);
385 }
386
387 if (!exception)
388 return;
389
390 String errorType = Inspector::Protocol::AutomationHelpers::getEnumConstantValue(Inspector::Protocol::Automation::ErrorMessage::InternalError);
391
392 String exceptionMessage;
393 if (JSValueIsObject(context, exception)) {
394 JSValueRef messageValue = JSObjectGetProperty(context, const_cast<JSObjectRef>(exception), OpaqueJSString::tryCreate("message"_s).get(), nullptr);
395 exceptionMessage = adoptRef(JSValueToStringCopy(context, messageValue, nullptr))->string();
396 } else
397 exceptionMessage = adoptRef(JSValueToStringCopy(context, exception, nullptr))->string();
398
399 didEvaluateJavaScriptFunction(frameID, callbackID, exceptionMessage, errorType);
400}

402void WebAutomationSessionProxy::didEvaluateJavaScriptFunction(WebCore::FrameIdentifier frameID, uint64_t callbackID, const String& result, const String& errorType)
403{
404 auto findResult = m_webFramePendingEvaluateJavaScriptCallbacksMap.find(frameID);
405 if (findResult != m_webFramePendingEvaluateJavaScriptCallbacksMap.end()) {
406 findResult->value.removeFirst(callbackID);
407 ASSERT(!findResult->value.contains(callbackID));
408 if (findResult->value.isEmpty())
409 m_webFramePendingEvaluateJavaScriptCallbacksMap.remove(findResult);
410 }
411
412 WebProcess::singleton().parentProcessConnection()->send(Messages::WebAutomationSession::DidEvaluateJavaScriptFunction(callbackID, result, errorType), 0);
413}

567static WebCore::Element* containerElementForElement(WebCore::Element& element)
568{
569 // 13. Element State.
570 // https://w3c.github.io/webdriver/webdriver-spec.html#dfn-container.
571 if (is<WebCore::HTMLOptionElement>(element)) {
572#if ENABLE(DATALIST_ELEMENT)
573 if (auto* parentElement = WebCore::ancestorsOfType<WebCore::HTMLDataListElement>(element).first())
574 return parentElement;
575#endif
576 if (auto* parentElement = downcast<WebCore::HTMLOptionElement>(element).ownerSelectElement())
577 return parentElement;
578
579 return nullptr;
580 }
581
582 if (is<WebCore::HTMLOptGroupElement>(element)) {
583 if (auto* parentElement = downcast<WebCore::HTMLOptGroupElement>(element).ownerSelectElement())
584 return parentElement;
585
586 return nullptr;
587 }
588
589 return &element;
590}

592static WebCore::FloatRect convertRectFromFrameClientToRootView(WebCore::FrameView* frameView, WebCore::FloatRect clientRect)
593{
594 if (!frameView->delegatesScrolling())
595 return frameView->contentsToRootView(frameView->clientToDocumentRect(clientRect));
596
597 // If the frame delegates scrolling, contentsToRootView doesn't take into account scroll/zoom/scale.
598 auto& frame = frameView->frame();
599 clientRect.scale(frame.pageZoomFactor() * frame.frameScaleFactor());
600 clientRect.moveBy(frameView->contentsScrollPosition());
601 return clientRect;
602}

605{
606 if (!frameView->delegatesScrolling())

608
609 // If the frame delegates scrolling, contentsToRootView doesn't take into account scroll/zoom/scale.
610 auto& frame = frameView->frame();
611 clientPoint.scale(frame.pageZoomFactor() * frame.frameScaleFactor());

616void WebAutomationSessionProxy::computeElementLayout(WebCore::PageIdentifier pageID, std::optional<WebCore::FrameIdentifier> frameID, String nodeHandle, bool scrollIntoViewIfNeeded, CoordinateSystem coordinateSystem, CompletionHandler<void(std::optional<String>, WebCore::IntRect, std::optional<WebCore::IntPoint>, bool)>&& completionHandler)
617{
618 WebPage* page = WebProcess::singleton().webPage(pageID);
619 if (!page) {
620 String windowNotFoundErrorType = Inspector::Protocol::AutomationHelpers::getEnumConstantValue(Inspector::Protocol::Automation::ErrorMessage::WindowNotFound);
621 completionHandler(windowNotFoundErrorType, { }, std::nullopt, false);
622 return;
623 }
624
625 auto* frame = frameID ? WebProcess::singleton().webFrame(*frameID) : &page->mainWebFrame();
626 if (!frame || !frame->coreFrame() || !frame->coreFrame()->view()) {
627 String frameNotFoundErrorType = Inspector::Protocol::AutomationHelpers::getEnumConstantValue(Inspector::Protocol::Automation::ErrorMessage::FrameNotFound);
628 completionHandler(frameNotFoundErrorType, { }, std::nullopt, false);
629 return;
630 }
631
632 if (!isValidNodeHandle(nodeHandle)) {
633 String invalidNodeIdentifierrrorType = Inspector::Protocol::AutomationHelpers::getEnumConstantValue(Inspector::Protocol::Automation::ErrorMessage::InvalidNodeIdentifier);
634 completionHandler(invalidNodeIdentifierrrorType, { }, std::nullopt, false);
635 return;
636 }
637
638 WebCore::Element* coreElement = elementForNodeHandle(*frame, nodeHandle);
639 if (!coreElement) {
640 String nodeNotFoundErrorType = Inspector::Protocol::AutomationHelpers::getEnumConstantValue(Inspector::Protocol::Automation::ErrorMessage::NodeNotFound);
641 completionHandler(nodeNotFoundErrorType, { }, std::nullopt, false);
642 return;
643 }
644
645 auto* containerElement = containerElementForElement(*coreElement);
646 if (scrollIntoViewIfNeeded && containerElement) {
647 // 14.1 Element Click. Step 4. Scroll into view the elements container.
648 // https://w3c.github.io/webdriver/webdriver-spec.html#element-click
649 containerElement->scrollIntoViewIfNotVisible(false);
650 // FIXME: Wait in an implementation-specific way up to the session implicit wait timeout for the element to become in view.
651 }
652
653 WebCore::FrameView* frameView = frame->coreFrame()->view();
654 WebCore::FrameView* mainView = frame->coreFrame()->mainFrame().view();
655
656 WebCore::IntRect resultElementBounds;
657 std::optional<WebCore::IntPoint> resultInViewCenterPoint;
658 bool isObscured = false;
659
660 switch (coordinateSystem) {
661 case CoordinateSystem::Page:
662 resultElementBounds = enclosingIntRect(coreElement->boundingClientRect());
663 break;
664 case CoordinateSystem::LayoutViewport: {
665 auto elementBoundsInRootCoordinates = convertRectFromFrameClientToRootView(frameView, coreElement->boundingClientRect());
666 resultElementBounds = enclosingIntRect(mainView->absoluteToLayoutViewportRect(mainView->rootViewToContents(elementBoundsInRootCoordinates)));
667 break;
668 }
669 }
670
671 // If an <option> or <optgroup> does not have an associated <select> or <datalist> element, then give up.
672 if (!containerElement) {
673 String elementNotInteractableErrorType = Inspector::Protocol::AutomationHelpers::getEnumConstantValue(Inspector::Protocol::Automation::ErrorMessage::ElementNotInteractable);
674 completionHandler(elementNotInteractableErrorType, resultElementBounds, resultInViewCenterPoint, isObscured);
675 return;
676 }
677
678 // 12.1 Element Interactability.
679 // https://www.w3.org/TR/webdriver/#dfn-in-view-center-point
680 auto* firstElementRect = containerElement->getClientRects()->item(0);
681 if (!firstElementRect) {
682 String elementNotInteractableErrorType = Inspector::Protocol::AutomationHelpers::getEnumConstantValue(Inspector::Protocol::Automation::ErrorMessage::ElementNotInteractable);
683 completionHandler(elementNotInteractableErrorType, resultElementBounds, resultInViewCenterPoint, isObscured);
684 return;
685 }
686
687 // The W3C WebDriver specification does not explicitly intersect the element with the visual viewport.
688 // Do that here so that the IVCP for an element larger than the viewport is within the viewport.
689 // See spec bug here: https://github.com/w3c/webdriver/issues/1402
690 auto viewportRect = frameView->documentToClientRect(frameView->visualViewportRect());
691 auto elementRect = WebCore::FloatRect(firstElementRect->x(), firstElementRect->y(), firstElementRect->width(), firstElementRect->height());
692 auto visiblePortionOfElementRect = intersection(viewportRect, elementRect);
693
694 // If the element is entirely outside the viewport, still calculate it's bounds.
695 if (visiblePortionOfElementRect.isEmpty()) {
696 completionHandler(std::nullopt, resultElementBounds, resultInViewCenterPoint, isObscured);
697 return;
698 }
699
700 auto elementInViewCenterPoint = visiblePortionOfElementRect.center();
701 auto elementList = containerElement->treeScope().elementsFromPoint(elementInViewCenterPoint);
702 auto index = elementList.findMatching([containerElement] (auto& item) { return item.get() == containerElement; });
703 if (elementList.isEmpty() || index == notFound) {
704 // We hit this case if the element is visibility:hidden or opacity:0, in which case it will not hit test
705 // at the calculated IVCP. An element is technically not "in view" if it is not within its own paint/hit test tree,
706 // so it cannot have an in-view center point either. And without an IVCP, the definition of 'obscured' makes no sense.
707 // See <https://w3c.github.io/webdriver/webdriver-spec.html#dfn-in-view>.
708 String elementNotInteractableErrorType = Inspector::Protocol::AutomationHelpers::getEnumConstantValue(Inspector::Protocol::Automation::ErrorMessage::ElementNotInteractable);
709 completionHandler(elementNotInteractableErrorType, resultElementBounds, resultInViewCenterPoint, isObscured);
710 return;
711 }
712
713 // Check the case where a non-descendant element hit tests before the target element. For example, a child <option>
714 // of a <select> does not obscure the <select>, but two sibling <div> that overlap at the IVCP will obscure each other.
715 // Node::isDescendantOf() is not self-inclusive, so that is explicitly checked here.
716 isObscured = elementList[0] != containerElement && !elementList[0]->isDescendantOrShadowDescendantOf(containerElement);
717
718 switch (coordinateSystem) {
719 case CoordinateSystem::Page:
720 resultInViewCenterPoint = roundedIntPoint(elementInViewCenterPoint);
721 break;
722 case CoordinateSystem::LayoutViewport: {
723 auto inViewCenterPointInRootCoordinates = convertPointFromFrameClientToRootView(frameView, elementInViewCenterPoint);
724 resultInViewCenterPoint = roundedIntPoint(mainView->absoluteToLayoutViewportPoint(mainView->rootViewToContents(inViewCenterPointInRootCoordinates)));
725 break;
726 }
727 }
728
729 completionHandler(std::nullopt, resultElementBounds, resultInViewCenterPoint, isObscured);
730}
