43WebCacheStorageConnection::WebCacheStorageConnection(WebCacheStorageProvider& provider)
44 : m_provider(provider)
45{
46}
47
48WebCacheStorageConnection::~WebCacheStorageConnection()
49{
50 ASSERT(isMainRunLoop());
51}

53IPC::Connection& WebCacheStorageConnection::connection()
54{
55 return WebProcess::singleton().ensureNetworkProcessConnection().connection();
56}

58void WebCacheStorageConnection::open(const WebCore::ClientOrigin& origin, const String& cacheName, WebCore::DOMCacheEngine::CacheIdentifierCallback&& callback)
59{
60 connection().sendWithAsyncReply(Messages::CacheStorageEngineConnection::Open(origin, cacheName), WTFMove(callback));
61}

63void WebCacheStorageConnection::remove(uint64_t cacheIdentifier, WebCore::DOMCacheEngine::CacheIdentifierCallback&& callback)
64{
65 connection().sendWithAsyncReply(Messages::CacheStorageEngineConnection::Remove(cacheIdentifier), WTFMove(callback));
66}

78void WebCacheStorageConnection::batchDeleteOperation(uint64_t cacheIdentifier, const WebCore::ResourceRequest& request, WebCore::CacheQueryOptions&& options, WebCore::DOMCacheEngine::RecordIdentifiersCallback&& callback)
