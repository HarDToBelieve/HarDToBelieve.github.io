45MediaKeySystemPermissionRequestManager::MediaKeySystemPermissionRequestManager(WebPage& page)
46 : m_page(page)
47{
48}

50void MediaKeySystemPermissionRequestManager::startMediaKeySystemRequest(MediaKeySystemRequest& request)
51{
52 Document* document = request.document();
53 Frame* frame = document ? document->frame() : nullptr;
54
55 if (!frame || !document->page()) {
56 request.deny(emptyString());
57 return;
58 }
59
60 if (document->page()->canStartMedia()) {
61 sendMediaKeySystemRequest(request);
62 return;
63 }
64
65 auto& pendingRequests = m_pendingMediaKeySystemRequests.add(document, Vector<Ref<MediaKeySystemRequest>>()).iterator->value;
66 if (pendingRequests.isEmpty())
67 document->addMediaCanStartListener(*this);
68 pendingRequests.append(request);
69}
70
71void MediaKeySystemPermissionRequestManager::sendMediaKeySystemRequest(MediaKeySystemRequest& userRequest)
72{
73 auto* frame = userRequest.document() ? userRequest.document()->frame() : nullptr;
74 if (!frame) {
75 userRequest.deny(emptyString());
76 return;
77 }
78
79 m_ongoingMediaKeySystemRequests.add(userRequest.identifier(), makeRef(userRequest));
80
81 WebFrame* webFrame = WebFrame::fromCoreFrame(*frame);
82 ASSERT(webFrame);
83
84 auto* topLevelDocumentOrigin = userRequest.topLevelDocumentOrigin();
85 m_page.send(Messages::WebPageProxy::RequestMediaKeySystemPermissionForFrame(userRequest.identifier().toUInt64(), webFrame->frameID(), topLevelDocumentOrigin->data(), userRequest.keySystem()));
86}
87
88void MediaKeySystemPermissionRequestManager::cancelMediaKeySystemRequest(MediaKeySystemRequest& request)
89{
90 if (auto removedRequest = m_ongoingMediaKeySystemRequests.take(request.identifier()))
91 return;
92
93 auto* document = request.document();
94 if (!document)
95 return;
96
97 auto iterator = m_pendingMediaKeySystemRequests.find(document);
98 if (iterator == m_pendingMediaKeySystemRequests.end())
99 return;
100
101 auto& pendingRequests = iterator->value;
102 pendingRequests.removeFirstMatching([&request](auto& item) {
103 return &request == item.ptr();
104 });
105
106 if (!pendingRequests.isEmpty())
107 return;
108
109 document->removeMediaCanStartListener(*this);
110 m_pendingMediaKeySystemRequests.remove(iterator);
111}
112
113void MediaKeySystemPermissionRequestManager::mediaCanStart(Document& document)
114{
115 ASSERT(document.page()->canStartMedia());
116
117 auto pendingRequests = m_pendingMediaKeySystemRequests.take(&document);
118 for (auto& pendingRequest : pendingRequests)
119 sendMediaKeySystemRequest(pendingRequest);
120}
121
122void MediaKeySystemPermissionRequestManager::mediaKeySystemWasGranted(uint64_t requestID, CompletionHandler<void()>&& completionHandler)
123{
124 auto request = m_ongoingMediaKeySystemRequests.take(makeObjectIdentifier<MediaKeySystemRequestIdentifierType>(requestID));
125 if (!request) {
126 completionHandler();
127 return;
128 }
129
130 request->allow(WTFMove(completionHandler));
131}
132
133void MediaKeySystemPermissionRequestManager::mediaKeySystemWasDenied(uint64_t requestID, String&& message)
134{
135 auto request = m_ongoingMediaKeySystemRequests.take(makeObjectIdentifier<MediaKeySystemRequestIdentifierType>(requestID));
136 if (!request)
137 return;
138
139 request->deny(WTFMove(message));
140}
