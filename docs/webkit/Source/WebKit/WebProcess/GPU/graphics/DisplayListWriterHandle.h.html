26#pragma once
27
28#include "SharedDisplayListHandle.h"
29#include <wtf/FastMalloc.h>
30
31namespace WebKit {
32
33class DisplayListWriterHandle : public SharedDisplayListHandle {
34 WTF_MAKE_NONCOPYABLE(DisplayListWriterHandle); WTF_MAKE_FAST_ALLOCATED;
35public:
36 static RefPtr<DisplayListWriterHandle> create(WebCore::DisplayList::ItemBufferIdentifier identifier, Ref<SharedMemory>&& sharedMemory)
37 {
38 if (sharedMemory->size() <= headerSize())
39 return nullptr;
40 return adoptRef(new DisplayListWriterHandle(identifier, WTFMove(sharedMemory)));
41 }
42
43 size_t writableOffset() const { return m_writableOffset; }
44 size_t availableCapacity() const;
45
46 bool moveWritableOffsetToStartIfPossible();
47
48 size_t advance(size_t amount);
49 WebCore::DisplayList::ItemBufferHandle createHandle() const;
50
51 bool tryToResume(SharedDisplayListHandle::ResumeReadingInformation&& info)
52 {
53 auto& header = this->header();
54 header.resumeReadingInfo = WTFMove(info);
55 return header.waitingStatus.compareExchangeWeak(
56 static_cast<SharedDisplayListHandle::WaitingStatusStorageType>(SharedDisplayListHandle::WaitingStatus::Waiting),
57 static_cast<SharedDisplayListHandle::WaitingStatusStorageType>(SharedDisplayListHandle::WaitingStatus::Resuming)
58 );
59 }
60
61private:
62 DisplayListWriterHandle(WebCore::DisplayList::ItemBufferIdentifier identifier, Ref<SharedMemory>&& sharedMemory)
63 : SharedDisplayListHandle(identifier, WTFMove(sharedMemory))
64 , m_writableOffset(SharedDisplayListHandle::headerSize())
65 {
66 }
67
68 size_t m_writableOffset { 0 };
69};
70
71} // namespace WebKit
