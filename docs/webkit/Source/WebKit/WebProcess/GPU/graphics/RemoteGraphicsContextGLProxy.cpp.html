47RefPtr<RemoteGraphicsContextGLProxy> RemoteGraphicsContextGLProxy::create(const GraphicsContextGLAttributes& attributes, RenderingBackendIdentifier renderingBackend)
48{
49 return adoptRef(new RemoteGraphicsContextGLProxy(WebProcess::singleton().ensureGPUProcessConnection(), attributes, renderingBackend));
50}

52static constexpr size_t defaultStreamSize = 1 << 21;
53
54RemoteGraphicsContextGLProxy::RemoteGraphicsContextGLProxy(GPUProcessConnection& gpuProcessConnection, const GraphicsContextGLAttributes& attributes, RenderingBackendIdentifier renderingBackend)
55 : RemoteGraphicsContextGLProxyBase(attributes)
56 , m_gpuProcessConnection(&gpuProcessConnection)
57 , m_streamConnection(gpuProcessConnection.connection(), defaultStreamSize)
58{
59 m_gpuProcessConnection->addClient(*this);
60 m_gpuProcessConnection->messageReceiverMap().addMessageReceiver(Messages::RemoteGraphicsContextGLProxy::messageReceiverName(), m_graphicsContextGLIdentifier.toUInt64(), *this);
61 connection().send(Messages::GPUConnectionToWebProcess::CreateGraphicsContextGL(attributes, m_graphicsContextGLIdentifier, renderingBackend, m_streamConnection.streamBuffer()), 0, IPC::SendOption::DispatchMessageEvenWhenWaitingForSyncReply);
62 // TODO: We must wait until initialized, because at the moment we cannot receive IPC messages
63 // during wait while in synchronous stream send. Should be fixed as part of https://bugs.webkit.org/show_bug.cgi?id=217211.
64 waitUntilInitialized();
65}

75void RemoteGraphicsContextGLProxy::reshape(int width, int height)
76{
77 if (isContextLost())
78 return;
79 m_currentWidth = width;
80 m_currentHeight = height;
81 auto sendResult = send(Messages::RemoteGraphicsContextGL::Reshape(width, height));
82 if (!sendResult)
83 markContextLost();
84}

86void RemoteGraphicsContextGLProxy::prepareForDisplay()
87{
88 if (isContextLost())
89 return;
90#if PLATFORM(COCOA)
91 MachSendRight displayBufferSendRight;
92 auto sendResult = sendSync(Messages::RemoteGraphicsContextGL::PrepareForDisplay(), Messages::RemoteGraphicsContextGL::PrepareForDisplay::Reply(displayBufferSendRight));
93 if (!sendResult) {
94 markContextLost();
95 return;
96 }
97 auto displayBuffer = IOSurface::createFromSendRight(WTFMove(displayBufferSendRight), WebCore::DestinationColorSpace::SRGB());
98 if (displayBuffer) {
99 auto& sc = platformSwapChain();
100 sc.recycleBuffer();
101 sc.present({ WTFMove(displayBuffer), nullptr });
102 }
103#else
104 auto sendResult = sendSync(Messages::RemoteGraphicsContextGL::PrepareForDisplay(), Messages::RemoteGraphicsContextGL::PrepareForDisplay::Reply());
105 if (!sendResult) {
106 markContextLost();
107 return;
108 }
109#endif
110 markLayerComposited();
111}

131void RemoteGraphicsContextGLProxy::paintRenderingResultsToCanvas(ImageBuffer& buffer)
132{
133 // FIXME: the buffer is "relatively empty" always, but for consistency, we need to ensure
134 // no pending operations are targeted for the `buffer`.
135 buffer.flushDrawingContext();
136
137 // FIXME: We cannot synchronize so that we would know no pending operations are using the `buffer`.
138
139 // FIXME: Currently RemoteImageBufferProxy::getPixelBuffer et al do not wait for the flushes of the images
140 // inside the display lists. Rather, it assumes that processing its sequence (e.g. the display list) will equal to read flush being
141 // fulfilled. For below, we cannot create a new flush id since we go through different sequence (RemoteGraphicsContextGL sequence)
142
143 // FIXME: Maybe implement IPC::Fence or something similar.
144 auto sendResult = sendSync(Messages::RemoteGraphicsContextGL::PaintRenderingResultsToCanvas(buffer.renderingResourceIdentifier()), Messages::RemoteGraphicsContextGL::PaintRenderingResultsToCanvas::Reply());
145 if (!sendResult) {
146 markContextLost();
147 return;
148 }
149}

161bool RemoteGraphicsContextGLProxy::copyTextureFromMedia(MediaPlayer& mediaPlayer, PlatformGLObject texture, GCGLenum target, GCGLint level, GCGLenum internalFormat, GCGLenum format, GCGLenum type, bool premultiplyAlpha, bool flipY)
162{
163 bool result = false;
164 auto sendResult = sendSync(Messages::RemoteGraphicsContextGL::CopyTextureFromMedia(mediaPlayer.identifier(), texture, target, level, internalFormat, format, type, premultiplyAlpha, flipY), Messages::RemoteGraphicsContextGL::CopyTextureFromMedia::Reply(result));
165 if (!sendResult) {

173void RemoteGraphicsContextGLProxy::synthesizeGLError(GCGLenum error)
174{

177 if (!sendResult)

180 }

184GCGLenum RemoteGraphicsContextGLProxy::getError()
185{
186 if (!isContextLost()) {
187 uint32_t returnValue = 0;
188 auto sendResult = sendSync(Messages::RemoteGraphicsContextGL::GetError(), Messages::RemoteGraphicsContextGL::GetError::Reply(returnValue));
189 if (!sendResult)
190 markContextLost();
191 return static_cast<GCGLenum>(returnValue);
192 }
193 return std::exchange(m_errorWhenContextIsLost, NO_ERROR);
194}

205void RemoteGraphicsContextGLProxy::wasCreated(bool didSucceed, IPC::Semaphore&& semaphore, String&& availableExtensions, String&& requestedExtensions)
206{
207 if (isContextLost())
208 return;
209 if (!didSucceed) {
210 markContextLost();
211 return;
212 }
213 ASSERT(!m_didInitialize);
214 m_streamConnection.setWakeUpSemaphore(WTFMove(semaphore));
215 m_didInitialize = true;
216 initialize(availableExtensions, requestedExtensions);
217}

219void RemoteGraphicsContextGLProxy::wasLost()
220{
221 if (isContextLost())
222 return;
223 markContextLost();
224}

226void RemoteGraphicsContextGLProxy::wasChanged()
227{
228 if (isContextLost())
229 return;
230 for (auto* client : copyToVector(m_clients))
231 client->dispatchContextChangedNotification();
232}

234void RemoteGraphicsContextGLProxy::markContextLost()
235{
236 disconnectGpuProcessIfNeeded();
237 for (auto* client : copyToVector(m_clients))
238 client->forceContextLost();
239}
240
241void RemoteGraphicsContextGLProxy::waitUntilInitialized()
242{
243 if (isContextLost())
244 return;
245 if (m_didInitialize)
246 return;
247 if (connection().waitForAndDispatchImmediately<Messages::RemoteGraphicsContextGLProxy::WasCreated>(m_graphicsContextGLIdentifier, defaultSendTimeout))
248 return;
249 markContextLost();
250}

252void RemoteGraphicsContextGLProxy::gpuProcessConnectionDidClose(GPUProcessConnection&)
253{
254 ASSERT(!isContextLost());
255 abandonGpuProcess();
256 markContextLost();
257}
258
259void RemoteGraphicsContextGLProxy::abandonGpuProcess()
260{
261 auto gpuProcessConnection = std::exchange(m_gpuProcessConnection, nullptr);
262 gpuProcessConnection->removeClient(*this);
263 gpuProcessConnection->messageReceiverMap().removeMessageReceiver(Messages::RemoteGraphicsContextGLProxy::messageReceiverName(), m_graphicsContextGLIdentifier.toUInt64());
264 m_gpuProcessConnection = nullptr;
265}
266
267void RemoteGraphicsContextGLProxy::disconnectGpuProcessIfNeeded()
268{
269 if (auto gpuProcessConnection = std::exchange(m_gpuProcessConnection, nullptr)) {
270 gpuProcessConnection->removeClient(*this);
271 gpuProcessConnection->messageReceiverMap().removeMessageReceiver(Messages::RemoteGraphicsContextGLProxy::messageReceiverName(), m_graphicsContextGLIdentifier.toUInt64());
272 gpuProcessConnection->connection().send(Messages::GPUConnectionToWebProcess::ReleaseGraphicsContextGL(m_graphicsContextGLIdentifier), 0, IPC::SendOption::DispatchMessageEvenWhenWaitingForSyncReply);
273 }
274 ASSERT(isContextLost());
275}
