62RemoteRenderingBackendProxy::~RemoteRenderingBackendProxy()
63{
64 if (!m_gpuProcessConnection)
65 return;
66
67 // Un-register itself as a MessageReceiver.
68 m_gpuProcessConnection->messageReceiverMap().removeMessageReceiver(*this);
69
70 // Release the RemoteRenderingBackend.
71 send(Messages::GPUConnectionToWebProcess::ReleaseRenderingBackend(renderingBackendIdentifier()), 0, IPC::SendOption::DispatchMessageEvenWhenWaitingForSyncReply);
72}
73
74GPUProcessConnection& RemoteRenderingBackendProxy::ensureGPUProcessConnection()
75{
76 if (!m_gpuProcessConnection) {
77 auto& gpuProcessConnection = WebProcess::singleton().ensureGPUProcessConnection();
78 gpuProcessConnection.addClient(*this);
79 gpuProcessConnection.messageReceiverMap().addMessageReceiver(Messages::RemoteRenderingBackendProxy::messageReceiverName(), renderingBackendIdentifier().toUInt64(), *this);
80 gpuProcessConnection.connection().send(Messages::GPUConnectionToWebProcess::CreateRenderingBackend(m_parameters), 0, IPC::SendOption::DispatchMessageEvenWhenWaitingForSyncReply);
81 m_gpuProcessConnection = makeWeakPtr(gpuProcessConnection);
82 }
83 return *m_gpuProcessConnection;
84}

86void RemoteRenderingBackendProxy::gpuProcessConnectionDidClose(GPUProcessConnection& previousConnection)
87{
88 previousConnection.removeClient(*this);
89 m_gpuProcessConnection = nullptr;
90 m_remoteResourceCacheProxy.remoteResourceCacheWasDestroyed();
91
92 m_identifiersOfReusableHandles.clear();
93 m_sharedDisplayListHandles.clear();
94 m_currentDestinationImageBufferIdentifier = std::nullopt;
95 m_deferredWakeupMessageArguments = std::nullopt;
96 m_remainingItemsToAppendBeforeSendingWakeup = 0;
97
98 if (m_destroyGetPixelBufferSharedMemoryTimer.isActive())
99 m_destroyGetPixelBufferSharedMemoryTimer.stop();
100 m_getPixelBufferSemaphore = std::nullopt;
101 m_getPixelBufferSharedMemoryLength = 0;
102 m_getPixelBufferSharedMemory = nullptr;
103}

134RefPtr<ImageBuffer> RemoteRenderingBackendProxy::createImageBuffer(const FloatSize& size, RenderingMode renderingMode, float resolutionScale, const DestinationColorSpace& colorSpace, PixelFormat pixelFormat)
135{
136 RefPtr<ImageBuffer> imageBuffer;
137
138 if (renderingMode == RenderingMode::Accelerated) {
139 // Unless DOM rendering is always enabled when any GPU process rendering is enabled,
140 // we need to create ImageBuffers for e.g. Canvas that are actually mapped into the
141 // Web Content process, so they can be painted into the tiles.
142 if (!WebProcess::singleton().shouldUseRemoteRenderingFor(RenderingPurpose::DOM))
143 imageBuffer = AcceleratedRemoteImageBufferMappedProxy::create(size, resolutionScale, colorSpace, pixelFormat, *this);
144 else
145 imageBuffer = AcceleratedRemoteImageBufferProxy::create(size, resolutionScale, colorSpace, pixelFormat, *this);
146 }
147
148 if (!imageBuffer)
149 imageBuffer = UnacceleratedRemoteImageBufferProxy::create(size, resolutionScale, colorSpace, pixelFormat, *this);
150
151 if (imageBuffer) {
152 createRemoteImageBuffer(*imageBuffer);
153 return imageBuffer;
154 }
155
156 return nullptr;
157}

159SharedMemory* RemoteRenderingBackendProxy::sharedMemoryForGetPixelBuffer(size_t dataSize, IPC::Timeout timeout)
160{
161 sendDeferredWakeupMessageIfNeeded();
162
163 bool needsSharedMemory = !m_getPixelBufferSharedMemory || dataSize > m_getPixelBufferSharedMemoryLength;
164 bool needsSemaphore = !m_getPixelBufferSemaphore;
165
166 if (needsSharedMemory)
167 m_getPixelBufferSharedMemory = nullptr;
168
169 SharedMemory::IPCHandle handle;
170 IPC::Semaphore semaphore;
171
172 if (needsSharedMemory && needsSemaphore)
173 sendSync(Messages::RemoteRenderingBackend::UpdateSharedMemoryAndSemaphoreForGetPixelBuffer(dataSize), Messages::RemoteRenderingBackend::UpdateSharedMemoryAndSemaphoreForGetPixelBuffer::Reply(handle, semaphore), renderingBackendIdentifier(), timeout, IPC::SendSyncOption::MaintainOrderingWithAsyncMessages);
174 else if (needsSharedMemory)
175 sendSync(Messages::RemoteRenderingBackend::UpdateSharedMemoryForGetPixelBuffer(dataSize), Messages::RemoteRenderingBackend::UpdateSharedMemoryForGetPixelBuffer::Reply(handle), renderingBackendIdentifier(), timeout, IPC::SendSyncOption::MaintainOrderingWithAsyncMessages);
176 else if (needsSemaphore)
177 sendSync(Messages::RemoteRenderingBackend::SemaphoreForGetPixelBuffer(), Messages::RemoteRenderingBackend::SemaphoreForGetPixelBuffer::Reply(semaphore), renderingBackendIdentifier(), timeout);
178
179 if (!handle.handle.isNull()) {
180 m_getPixelBufferSharedMemory = SharedMemory::map(handle.handle, SharedMemory::Protection::ReadOnly);
181 m_getPixelBufferSharedMemoryLength = handle.dataSize;
182 }
183 if (needsSemaphore)
184 m_getPixelBufferSemaphore = WTFMove(semaphore);
185
186 if (m_destroyGetPixelBufferSharedMemoryTimer.isActive())
187 m_destroyGetPixelBufferSharedMemoryTimer.stop();
188 m_destroyGetPixelBufferSharedMemoryTimer.startOneShot(5_s);
189
190 return m_getPixelBufferSharedMemory.get();
191}

258void RemoteRenderingBackendProxy::didCreateImageBufferBackend(ImageBufferBackendHandle handle, RenderingResourceIdentifier renderingResourceIdentifier)
259{
260 auto imageBuffer = m_remoteResourceCacheProxy.cachedImageBuffer(renderingResourceIdentifier);
261 if (!imageBuffer)
262 return;
263
264 if (imageBuffer->renderingMode() == RenderingMode::Unaccelerated)
265 imageBuffer->setBackend(UnacceleratedImageBufferShareableBackend::create(imageBuffer->parameters(), WTFMove(handle)));
266 else if (imageBuffer->canMapBackingStore())
267 imageBuffer->setBackend(AcceleratedImageBufferShareableMappedBackend::create(imageBuffer->parameters(), WTFMove(handle)));
268 else
269 imageBuffer->setBackend(AcceleratedImageBufferShareableBackend::create(imageBuffer->parameters(), WTFMove(handle)));
270}

278void RemoteRenderingBackendProxy::willAppendItem(RenderingResourceIdentifier newDestinationIdentifier)
279{
280 if (m_currentDestinationImageBufferIdentifier == newDestinationIdentifier)
281 return;
282
283 if (auto previousDestinationBufferIdentifier = std::exchange(m_currentDestinationImageBufferIdentifier, newDestinationIdentifier)) {
284 if (auto imageBuffer = m_remoteResourceCacheProxy.cachedImageBuffer(*previousDestinationBufferIdentifier))
285 imageBuffer->changeDestinationImageBuffer(newDestinationIdentifier);
286 else
287 ASSERT_NOT_REACHED();
288 }
289
290 auto handle = mostRecentlyUsedDisplayListHandle();
291 if (UNLIKELY(!handle))
292 return;
293
294 auto newDestination = m_remoteResourceCacheProxy.cachedImageBuffer(newDestinationIdentifier);
295 if (UNLIKELY(!newDestination)) {
296 ASSERT_NOT_REACHED();
297 return;
298 }
299
300 handle->moveWritableOffsetToStartIfPossible();
301 newDestination->prepareToAppendDisplayListItems(handle->createHandle());
302}

310void RemoteRenderingBackendProxy::sendDeferredWakeupMessageIfNeeded()
311{
312 auto arguments = std::exchange(m_deferredWakeupMessageArguments, std::nullopt);
313 if (!arguments)
314 return;
315
316 sendWakeupMessage(*arguments);
317 m_remainingItemsToAppendBeforeSendingWakeup = 0;
318}

320void RemoteRenderingBackendProxy::didAppendData(const DisplayList::ItemBufferHandle& handle, size_t numberOfBytes, DisplayList::DidChangeItemBuffer didChangeItemBuffer, RenderingResourceIdentifier destinationImageBuffer)
321{
322 auto* sharedHandle = m_sharedDisplayListHandles.get(handle.identifier);
323 if (UNLIKELY(!sharedHandle))
324 RELEASE_ASSERT_NOT_REACHED();
325
326 bool wasEmpty = sharedHandle->advance(numberOfBytes) == numberOfBytes;
327 if (!wasEmpty || didChangeItemBuffer == DisplayList::DidChangeItemBuffer::Yes) {
328 if (m_deferredWakeupMessageArguments) {
329 if (sharedHandle->tryToResume({ m_deferredWakeupMessageArguments->offset, m_deferredWakeupMessageArguments->destinationImageBufferIdentifier.toUInt64() })) {
330 m_parameters.resumeDisplayListSemaphore.signal();
331 m_deferredWakeupMessageArguments = std::nullopt;
332 m_remainingItemsToAppendBeforeSendingWakeup = 0;
333 } else if (!--m_remainingItemsToAppendBeforeSendingWakeup) {
334 m_deferredWakeupMessageArguments->reason = GPUProcessWakeupReason::ItemCountHysteresisExceeded;
335 sendWakeupMessage(*std::exchange(m_deferredWakeupMessageArguments, std::nullopt));
336 }
337 }
338 return;
339 }
340
341 sendDeferredWakeupMessageIfNeeded();
342
343 auto offsetToRead = sharedHandle->writableOffset() - numberOfBytes;
344 if (sharedHandle->tryToResume({ offsetToRead, destinationImageBuffer.toUInt64() })) {
345 m_parameters.resumeDisplayListSemaphore.signal();
346 return;
347 }
348
349 // Instead of sending the wakeup message immediately, wait for some additional data. This gives the
350 // web process a "head start", decreasing the likelihood that the GPU process will encounter frequent
351 // wakeups when processing a large amount of display list items.
352 constexpr unsigned itemCountHysteresisBeforeSendingWakeup = 512;
353
354 m_remainingItemsToAppendBeforeSendingWakeup = itemCountHysteresisBeforeSendingWakeup;
355 m_deferredWakeupMessageArguments = {{ handle.identifier, offsetToRead, destinationImageBuffer }};
356}

366RefPtr<DisplayListWriterHandle> RemoteRenderingBackendProxy::findReusableDisplayListHandle(size_t capacity)
367{
368 auto mostRecentlyUsedHandle = mostRecentlyUsedDisplayListHandle();
369 if (UNLIKELY(!mostRecentlyUsedHandle))
370 return nullptr;
371
372 mostRecentlyUsedHandle->moveWritableOffsetToStartIfPossible();
373 if (mostRecentlyUsedHandle->availableCapacity() >= capacity)
374 return mostRecentlyUsedHandle;
375
376 m_identifiersOfReusableHandles.append(m_identifiersOfReusableHandles.takeFirst());
377
378 auto leastRecentlyUsedIdentifier = m_identifiersOfReusableHandles.first();
379 if (leastRecentlyUsedIdentifier != mostRecentlyUsedHandle->identifier()) {
380 auto handle = makeRefPtr(m_sharedDisplayListHandles.get(leastRecentlyUsedIdentifier));
381 if (handle->moveWritableOffsetToStartIfPossible() && handle->availableCapacity() >= capacity)
382 return handle;
383 }
384
385 return nullptr;
386}

388DisplayList::ItemBufferHandle RemoteRenderingBackendProxy::createItemBuffer(size_t capacity, RenderingResourceIdentifier destinationBufferIdentifier)
389{
390 if (auto handle = findReusableDisplayListHandle(capacity)) {
391 LOG_WITH_STREAM(SharedDisplayLists, stream << "Reusing Items[" << handle->identifier() << "] => Image(" << destinationBufferIdentifier << ") (remaining capacity: " << handle->availableCapacity() << ")");
392 return handle->createHandle();
393 }
394
395 static constexpr size_t defaultSharedItemBufferSize = 1 << 16;
396 static_assert(defaultSharedItemBufferSize > SharedDisplayListHandle::headerSize());
397
398 auto sharedMemory = SharedMemory::allocate(std::max(defaultSharedItemBufferSize, capacity + SharedDisplayListHandle::headerSize()));
399 if (!sharedMemory)
400 return { };
401
402 SharedMemory::Handle sharedMemoryHandle;
403 sharedMemory->createHandle(sharedMemoryHandle, SharedMemory::Protection::ReadWrite);
404
405 auto identifier = DisplayList::ItemBufferIdentifier::generate();
406 send(Messages::RemoteRenderingBackend::DidCreateSharedDisplayListHandle(identifier, { WTFMove(sharedMemoryHandle), sharedMemory->size() }, destinationBufferIdentifier), renderingBackendIdentifier(), IPC::SendOption::DispatchMessageEvenWhenWaitingForSyncReply);
407
408 auto newHandle = DisplayListWriterHandle::create(identifier, sharedMemory.releaseNonNull());
409 RELEASE_ASSERT(newHandle, "There must be enough space to create the handle.");
410 auto displayListHandle = newHandle->createHandle();
411
412 m_identifiersOfReusableHandles.prepend(identifier);
413 m_sharedDisplayListHandles.set(identifier, WTFMove(newHandle));
414
415 LOG_WITH_STREAM(SharedDisplayLists, stream << "Allocated Items[" << identifier << "] => Image(" << destinationBufferIdentifier << ")");
416 return displayListHandle;
417}
