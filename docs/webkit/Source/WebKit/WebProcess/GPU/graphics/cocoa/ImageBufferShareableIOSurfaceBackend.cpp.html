56std::unique_ptr<ImageBufferShareableIOSurfaceBackend> ImageBufferShareableIOSurfaceBackend::create(const Parameters& parameters, ImageBufferBackendHandle handle)
57{
58 if (!WTF::holds_alternative<MachSendRight>(handle)) {
59 RELEASE_ASSERT_NOT_REACHED();
60 return nullptr;
61 }
62
63 return makeUnique<ImageBufferShareableIOSurfaceBackend>(parameters, WTFMove(handle));
64}
65
66ImageBufferBackendHandle ImageBufferShareableIOSurfaceBackend::createImageBufferBackendHandle() const
67{
68 return WTF::get<MachSendRight>(m_handle).copySendRight();
69}
