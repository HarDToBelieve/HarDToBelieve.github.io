49Ref<MediaSourcePrivateRemote> MediaSourcePrivateRemote::create(GPUProcessConnection& gpuProcessConnection, RemoteMediaSourceIdentifier identifier, RemoteMediaPlayerMIMETypeCache& mimeTypeCache, const MediaPlayerPrivateRemote& mediaPlayerPrivate, MediaSourcePrivateClient* client)
50{
51 auto mediaSourcePrivate = adoptRef(*new MediaSourcePrivateRemote(gpuProcessConnection, identifier, mimeTypeCache, mediaPlayerPrivate, client));
52 client->setPrivateAndOpen(mediaSourcePrivate.copyRef());
53 return mediaSourcePrivate;
54}

56MediaSourcePrivateRemote::MediaSourcePrivateRemote(GPUProcessConnection& gpuProcessConnection, RemoteMediaSourceIdentifier identifier, RemoteMediaPlayerMIMETypeCache& mimeTypeCache, const MediaPlayerPrivateRemote& mediaPlayerPrivate, MediaSourcePrivateClient* client)
57 : m_gpuProcessConnection(makeWeakPtr(gpuProcessConnection))
58 , m_identifier(identifier)
59 , m_mimeTypeCache(mimeTypeCache)
60 , m_mediaPlayerPrivate(makeWeakPtr(mediaPlayerPrivate))
61 , m_client(client)
62#if !RELEASE_LOG_DISABLED
63 , m_logger(m_mediaPlayerPrivate->mediaPlayerLogger())
64 , m_logIdentifier(m_mediaPlayerPrivate->mediaPlayerLogIdentifier())
65#endif
66{
67 ALWAYS_LOG(LOGIDENTIFIER);
68
69 m_gpuProcessConnection->messageReceiverMap().addMessageReceiver(Messages::MediaSourcePrivateRemote::messageReceiverName(), m_identifier.toUInt64(), *this);
70
71#if !RELEASE_LOG_DISABLED
72 m_client->setLogIdentifier(m_logIdentifier);
73#endif
74}

76MediaSourcePrivateRemote::~MediaSourcePrivateRemote()
77{
78 ALWAYS_LOG(LOGIDENTIFIER);
79 if (m_gpuProcessConnection)
80 m_gpuProcessConnection->messageReceiverMap().removeMessageReceiver(Messages::MediaSourcePrivateRemote::messageReceiverName(), m_identifier.toUInt64());
81
82 for (auto it = m_sourceBuffers.begin(), end = m_sourceBuffers.end(); it != end; ++it)
83 (*it)->clearMediaSource();
84}

86MediaSourcePrivate::AddStatus MediaSourcePrivateRemote::addSourceBuffer(const ContentType& contentType, bool, RefPtr<SourceBufferPrivate>& outPrivate)
87{
88 DEBUG_LOG(LOGIDENTIFIER, contentType);
89
90 MediaEngineSupportParameters parameters;
91 parameters.isMediaSource = true;
92 parameters.type = contentType;
93 if (m_mimeTypeCache.supportsTypeAndCodecs(parameters) == MediaPlayer::SupportsType::IsNotSupported)
94 return AddStatus::NotSupported;
95
96 AddStatus status = AddStatus::NotSupported;
97 if (!m_gpuProcessConnection)
98 return status;
99
100 std::optional<RemoteSourceBufferIdentifier> remoteSourceBufferIdentifier;
101 m_gpuProcessConnection->connection().sendSync(Messages::RemoteMediaSourceProxy::AddSourceBuffer(contentType), Messages::RemoteMediaSourceProxy::AddSourceBuffer::Reply(status, remoteSourceBufferIdentifier), m_identifier);
102
103 if (status == AddStatus::Ok) {
104 ASSERT(remoteSourceBufferIdentifier.has_value());
105 auto newSourceBuffer = SourceBufferPrivateRemote::create(*m_gpuProcessConnection, *remoteSourceBufferIdentifier, *this, *m_mediaPlayerPrivate);
106 outPrivate = newSourceBuffer.copyRef();
107 m_sourceBuffers.append(WTFMove(newSourceBuffer));
108 }
109
110 return status;
111}
