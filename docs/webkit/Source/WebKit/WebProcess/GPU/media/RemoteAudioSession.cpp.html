47RemoteAudioSession::RemoteAudioSession(WebProcess& process)
48 : m_process(process)
49{
50}
51
52RemoteAudioSession::~RemoteAudioSession()
53{
54 if (m_gpuProcessConnection)
55 m_gpuProcessConnection->messageReceiverMap().removeMessageReceiver(Messages::RemoteAudioSession::messageReceiverName());
56}
57
58void RemoteAudioSession::gpuProcessConnectionDidClose(GPUProcessConnection& connection)
59{
60 ASSERT(m_gpuProcessConnection.get() == &connection);
61 m_gpuProcessConnection = nullptr;
62 connection.messageReceiverMap().removeMessageReceiver(Messages::RemoteAudioSession::messageReceiverName());
63 connection.removeClient(*this);
64}
65
66IPC::Connection& RemoteAudioSession::ensureConnection()
67{
68 if (!m_gpuProcessConnection) {
69 m_gpuProcessConnection = makeWeakPtr(m_process.ensureGPUProcessConnection());
70 m_gpuProcessConnection->addClient(*this);
71 m_gpuProcessConnection->messageReceiverMap().addMessageReceiver(Messages::RemoteAudioSession::messageReceiverName(), *this);
72
73 RemoteAudioSessionConfiguration configuration;
74 ensureConnection().sendSync(Messages::GPUConnectionToWebProcess::EnsureAudioSession(), Messages::GPUConnectionToWebProcess::EnsureAudioSession::Reply(configuration), { });
75 m_configuration = WTFMove(configuration);
76 }
77 return m_gpuProcessConnection->connection();
78}
79
80const RemoteAudioSessionConfiguration& RemoteAudioSession::configuration() const
81{
82 if (!m_configuration)
83 const_cast<RemoteAudioSession*>(this)->ensureConnection();
84 return *m_configuration;
85}
86
87RemoteAudioSessionConfiguration& RemoteAudioSession::configuration()
88{
89 if (!m_configuration)
90 ensureConnection();
91 return *m_configuration;
92}

94void RemoteAudioSession::setCategory(CategoryType type, RouteSharingPolicy policy)
95{
96#if PLATFORM(COCOA)
97 if (type == m_category && policy == m_routeSharingPolicy)
98 return;
99
100 m_category = type;
101 m_routeSharingPolicy = policy;
102
103 ensureConnection().send(Messages::RemoteAudioSessionProxy::SetCategory(type, policy), { });
104#else
105 UNUSED_PARAM(type);
106 UNUSED_PARAM(policy);
107#endif
108}
109
110void RemoteAudioSession::setPreferredBufferSize(size_t size)
111{
112 ensureConnection().send(Messages::RemoteAudioSessionProxy::SetPreferredBufferSize(size), { });
113}
114
115bool RemoteAudioSession::tryToSetActiveInternal(bool active)
116{
117 bool succeeded;
118 ensureConnection().sendSync(Messages::RemoteAudioSessionProxy::TryToSetActive(active), Messages::RemoteAudioSessionProxy::TryToSetActive::Reply(succeeded), { });
119 if (succeeded)
120 configuration().isActive = active;
121 return succeeded;
122}

124AudioSession::CategoryType RemoteAudioSession::category() const
125{
126#if PLATFORM(COCOA)
127 return m_category;
128#else
129 return AudioSession::CategoryType::None;
130#endif
131}
