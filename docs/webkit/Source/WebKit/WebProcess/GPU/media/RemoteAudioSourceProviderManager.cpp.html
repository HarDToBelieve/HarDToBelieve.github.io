41RemoteAudioSourceProviderManager::RemoteAudioSourceProviderManager()
42 : m_queue(WorkQueue::create("RemoteAudioSourceProviderManager", WorkQueue::Type::Serial, WorkQueue::QOS::UserInteractive))

46RemoteAudioSourceProviderManager::~RemoteAudioSourceProviderManager()

48 ASSERT(!m_connection);

51void RemoteAudioSourceProviderManager::stopListeningForIPC()
52{
53 setConnection(nullptr);
54}

56void RemoteAudioSourceProviderManager::setConnection(IPC::Connection* connection)
57{
58 if (m_connection == connection)
59 return;
60
61 if (m_connection)
62 m_connection->removeWorkQueueMessageReceiver(Messages::RemoteAudioSourceProviderManager::messageReceiverName());
63
64 m_connection = WTFMove(connection);
65
66 if (m_connection)
67 m_connection->addWorkQueueMessageReceiver(Messages::RemoteAudioSourceProviderManager::messageReceiverName(), m_queue, this);
68}
69
70void RemoteAudioSourceProviderManager::addProvider(Ref<RemoteAudioSourceProvider>&& provider)
71{
72 ASSERT(WTF::isMainRunLoop());
73 setConnection(&WebProcess::singleton().ensureGPUProcessConnection().connection());
74
75 m_queue->dispatch([this, protectedThis = makeRef(*this), provider = WTFMove(provider)]() mutable {
76 auto identifier = provider->identifier();
77
78 ASSERT(!m_providers.contains(identifier));
79 m_providers.add(identifier, makeUnique<RemoteAudio>(WTFMove(provider)));
80 });
81}
82
83void RemoteAudioSourceProviderManager::removeProvider(MediaPlayerIdentifier identifier)
84{
85 ASSERT(WTF::isMainRunLoop());
86
87 m_queue->dispatch([this, protectedThis = makeRef(*this), identifier] {
88 ASSERT(m_providers.contains(identifier));
89 m_providers.remove(identifier);
90 });
91}

93void RemoteAudioSourceProviderManager::audioStorageChanged(MediaPlayerIdentifier identifier, const SharedMemory::IPCHandle& ipcHandle, const WebCore::CAAudioStreamDescription& description, uint64_t numberOfFrames)
94{
95 ASSERT(!WTF::isMainRunLoop());
96
97 auto iterator = m_providers.find(identifier);
98 if (iterator == m_providers.end()) {
99 RELEASE_LOG_ERROR(Media, "Unable to find provider %llu for storageChanged", identifier.toUInt64());
100 return;
101 }
102 iterator->value->setStorage(ipcHandle.handle, description, numberOfFrames);
103}
104
105void RemoteAudioSourceProviderManager::audioSamplesAvailable(MediaPlayerIdentifier identifier, uint64_t startFrame, uint64_t numberOfFrames)
106{
107 ASSERT(!WTF::isMainRunLoop());
108
109 auto iterator = m_providers.find(identifier);
110 if (iterator == m_providers.end()) {
111 RELEASE_LOG_ERROR(Media, "Unable to find provider %llu for audioSamplesAvailable", identifier.toUInt64());
112 return;

123void RemoteAudioSourceProviderManager::RemoteAudio::setStorage(const SharedMemory::Handle& handle, const WebCore::CAAudioStreamDescription& description, uint64_t numberOfFrames)
124{
125 m_description = description;
126
127 // Take ownership of shared memory and mark it as media-related memory.
128 handle.takeOwnershipOfMemory(MemoryLedger::Media);
129
130 m_ringBuffer = CARingBuffer::adoptStorage(makeUniqueRef<ReadOnlySharedRingBufferStorage>(handle), description, numberOfFrames).moveToUniquePtr();
131 m_buffer = makeUnique<WebAudioBufferList>(description, numberOfFrames);
132}
133
134void RemoteAudioSourceProviderManager::RemoteAudio::audioSamplesAvailable(uint64_t startFrame, uint64_t numberOfFrames)
135{
136 if (!m_buffer) {
137 RELEASE_LOG_ERROR(Media, "buffer for audio provider %llu is null", m_provider->identifier().toUInt64());
138 return;
139 }
140
141 if (!WebAudioBufferList::isSupportedDescription(m_description, numberOfFrames)) {
142 RELEASE_LOG_ERROR(Media, "Unable to support description with given number of frames for audio provider %llu", m_provider->identifier().toUInt64());
143 return;
144 }
145
146 m_buffer->setSampleCount(numberOfFrames);
147
148 m_ringBuffer->fetch(m_buffer->list(), numberOfFrames, startFrame);
149
150 m_provider->audioSamplesAvailable(*m_buffer, m_description, numberOfFrames);
151}
