28#if ENABLE(GPU_PROCESS) && ENABLE(ENCRYPTED_MEDIA)

30#include <wtf/text/WTFString.h>
31
32namespace WebKit {
33
34struct RemoteCDMConfiguration {
35 Vector<AtomString> supportedInitDataTypes;
36 Vector<AtomString> supportedRobustnesses;
37 bool supportsServerCertificates;
38 bool supportsSessions;
39
40 template<class Encoder>
41 void encode(Encoder& encoder) const
42 {
43 encoder << supportedInitDataTypes;
44 encoder << supportedRobustnesses;
45 encoder << supportsServerCertificates;
46 encoder << supportsSessions;
47 }
48
49 template <class Decoder>
50 static std::optional<RemoteCDMConfiguration> decode(Decoder& decoder)
51 {
52 std::optional<Vector<AtomString>> supportedInitDataTypes;
53 decoder >> supportedInitDataTypes;
54 if (!supportedInitDataTypes)
55 return std::nullopt;
56
57 std::optional<Vector<AtomString>> supportedRobustnesses;
58 decoder >> supportedRobustnesses;
59 if (!supportedRobustnesses)
60 return std::nullopt;
61
62 std::optional<bool> supportsServerCertificates;
63 decoder >> supportsServerCertificates;
64 if (!supportsServerCertificates)
65 return std::nullopt;
66
67 std::optional<bool> supportsSessions;
68 decoder >> supportsSessions;
69 if (!supportsSessions)
70 return std::nullopt;
71
72 return {{
73 WTFMove(*supportedInitDataTypes),
74 WTFMove(*supportedRobustnesses),
75 *supportsServerCertificates,
76 *supportsSessions,
77 }};
78 }
79};
80
81} // namespace WebKit
82
83#endif
