42RemoteCDMFactory::RemoteCDMFactory(WebProcess& process)
43 : m_process(process)

45}

49void RemoteCDMFactory::registerFactory(Vector<CDMFactory*>& factories)
50{
51 factories.append(this);
52}

54const char* RemoteCDMFactory::supplementName()
55{
56 return "RemoteCDMFactory";
57}
58
59GPUProcessConnection& RemoteCDMFactory::gpuProcessConnection()
60{
61 return m_process.ensureGPUProcessConnection();
62}
63
64bool RemoteCDMFactory::supportsKeySystem(const String& keySystem)
65{
66 bool supported = false;
67 gpuProcessConnection().connection().sendSync(Messages::RemoteCDMFactoryProxy::SupportsKeySystem(keySystem), Messages::RemoteCDMFactoryProxy::SupportsKeySystem::Reply(supported), { });
68 return supported;
69}
70
71std::unique_ptr<CDMPrivate> RemoteCDMFactory::createCDM(const String& keySystem)
72{
73 RemoteCDMIdentifier identifier;
74 RemoteCDMConfiguration configuration;
75 gpuProcessConnection().connection().sendSync(Messages::RemoteCDMFactoryProxy::CreateCDM(keySystem), Messages::RemoteCDMFactoryProxy::CreateCDM::Reply(identifier, configuration), { });
76 if (!identifier)
77 return nullptr;
78 return RemoteCDM::create(makeWeakPtr(this), WTFMove(identifier), WTFMove(configuration));
79}
80
81void RemoteCDMFactory::addSession(Ref<RemoteCDMInstanceSession>&& session)
82{
83 ASSERT(!m_sessions.contains(session->identifier()));
84 m_sessions.set(session->identifier(), WTFMove(session));
85}
86
87void RemoteCDMFactory::removeSession(RemoteCDMInstanceSessionIdentifier identifier)
88{
89 ASSERT(m_sessions.contains(identifier));
90 m_sessions.remove(identifier);
91}
92
93void RemoteCDMFactory::didReceiveSessionMessage(IPC::Connection& connection, IPC::Decoder& decoder)
94{
95 if (auto* session = m_sessions.get(makeObjectIdentifier<RemoteCDMInstanceSessionIdentifierType>(decoder.destinationID())))
96 session->didReceiveMessage(connection, decoder);
97}
