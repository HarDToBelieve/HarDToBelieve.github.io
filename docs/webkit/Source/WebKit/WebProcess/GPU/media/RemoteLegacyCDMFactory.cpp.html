51void RemoteLegacyCDMFactory::registerFactory()
52{
53 LegacyCDM::clearFactories();
54 LegacyCDM::registerCDMFactory(
55 [weakThis = makeWeakPtr(this)] (LegacyCDM* privateCDM) -> std::unique_ptr<WebCore::CDMPrivateInterface> {
56 if (weakThis)
57 return weakThis->createCDM(privateCDM);
58 return nullptr;
59 },
60 [weakThis = makeWeakPtr(this)] (const String& keySystem) {
61 return weakThis ? weakThis->supportsKeySystem(keySystem) : false;
62 },
63 [weakThis = makeWeakPtr(this)] (const String& keySystem, const String& mimeType) {
64 return weakThis ? weakThis->supportsKeySystemAndMimeType(keySystem, mimeType) : false;
65 }
66 );
67}

79bool RemoteLegacyCDMFactory::supportsKeySystem(const String& keySystem)
80{
81 auto foundInCache = m_supportsKeySystemCache.find(keySystem);
82 if (foundInCache != m_supportsKeySystemCache.end())
83 return foundInCache->value;
84
85 bool supported = false;
86 gpuProcessConnection().connection().sendSync(Messages::RemoteLegacyCDMFactoryProxy::SupportsKeySystem(keySystem, std::nullopt), Messages::RemoteLegacyCDMFactoryProxy::SupportsKeySystem::Reply(supported), { });
87 m_supportsKeySystemCache.set(keySystem, supported);
88 return supported;
89}
90
91bool RemoteLegacyCDMFactory::supportsKeySystemAndMimeType(const String& keySystem, const String& mimeType)
92{
93 auto key = std::make_pair(keySystem, mimeType);
94 auto foundInCache = m_supportsKeySystemAndMimeTypeCache.find(key);
95 if (foundInCache != m_supportsKeySystemAndMimeTypeCache.end())
96 return foundInCache->value;
97
98 bool supported = false;
99 gpuProcessConnection().connection().sendSync(Messages::RemoteLegacyCDMFactoryProxy::SupportsKeySystem(keySystem, mimeType), Messages::RemoteLegacyCDMFactoryProxy::SupportsKeySystem::Reply(supported), { });
100 m_supportsKeySystemAndMimeTypeCache.set(key, supported);
101 return supported;
102}

104std::unique_ptr<CDMPrivateInterface> RemoteLegacyCDMFactory::createCDM(WebCore::LegacyCDM* cdm)
105{
106 if (!cdm) {
107 ASSERT_NOT_REACHED();
108 return nullptr;
109 }
110
111 std::optional<MediaPlayerIdentifier> playerId;
112 if (auto player = cdm->mediaPlayer())
113 playerId = gpuProcessConnection().mediaPlayerManager().findRemotePlayerId(player->playerPrivate());
114
115 RemoteLegacyCDMIdentifier identifier;
116 gpuProcessConnection().connection().sendSync(Messages::RemoteLegacyCDMFactoryProxy::CreateCDM(cdm->keySystem(), WTFMove(playerId)), Messages::RemoteLegacyCDMFactoryProxy::CreateCDM::Reply(identifier), { });
117 if (!identifier)
118 return nullptr;
119 auto remoteCDM = RemoteLegacyCDM::create(makeWeakPtr(this), identifier);
120 m_cdms.set(identifier, makeWeakPtr(remoteCDM.get()));
121 return remoteCDM;
122}

124void RemoteLegacyCDMFactory::addSession(RemoteLegacyCDMSessionIdentifier identifier, std::unique_ptr<RemoteLegacyCDMSession>&& session)
125{
126 ASSERT(!m_sessions.contains(identifier));
127 m_sessions.set(identifier, WTFMove(session));
128}
129
130void RemoteLegacyCDMFactory::removeSession(RemoteLegacyCDMSessionIdentifier identifier)
131{
132 ASSERT(m_sessions.contains(identifier));
133 m_sessions.remove(identifier);
134}
135
136RemoteLegacyCDM* RemoteLegacyCDMFactory::findCDM(CDMPrivateInterface* privateInterface) const
137{
138 for (auto& cdm : m_cdms.values()) {
139 if (privateInterface == cdm.get())
140 return cdm.get();
141 }
142 return nullptr;
143}

145void RemoteLegacyCDMFactory::didReceiveSessionMessage(IPC::Connection& connection, IPC::Decoder& decoder)
146{
147 if (auto* session = m_sessions.get(makeObjectIdentifier<RemoteLegacyCDMSessionIdentifierType>(decoder.destinationID())))
148 session->didReceiveMessage(connection, decoder);
149}
