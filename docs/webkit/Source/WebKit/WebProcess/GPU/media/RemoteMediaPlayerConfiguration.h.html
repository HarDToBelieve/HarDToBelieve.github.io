34namespace WebKit {
35
36struct RemoteMediaPlayerConfiguration {
37 String engineDescription;
38 double maximumDurationToCacheMediaTime;
39 bool supportsScanning { false };
40 bool supportsFullscreen { false };
41 bool supportsPictureInPicture { false };
42 bool supportsAcceleratedRendering { false };
43 bool canPlayToWirelessPlaybackTarget { false };
44 bool shouldIgnoreIntrinsicSize { false };
45
46 template<class Encoder>
47 void encode(Encoder& encoder) const
48 {
49 encoder << engineDescription;
50 encoder << maximumDurationToCacheMediaTime;
51 encoder << supportsScanning;
52 encoder << supportsFullscreen;
53 encoder << supportsPictureInPicture;
54 encoder << supportsAcceleratedRendering;
55 encoder << canPlayToWirelessPlaybackTarget;
56 encoder << shouldIgnoreIntrinsicSize;
57 }
58
59 template <class Decoder>
60 static std::optional<RemoteMediaPlayerConfiguration> decode(Decoder& decoder)
61 {
62 std::optional<String> engineDescription;
63 decoder >> engineDescription;
64 if (!engineDescription)
65 return std::nullopt;
66
67 std::optional<double> maximumDurationToCacheMediaTime;
68 decoder >> maximumDurationToCacheMediaTime;
69 if (!maximumDurationToCacheMediaTime)
70 return std::nullopt;
71
72 std::optional<bool> supportsScanning;
73 decoder >> supportsScanning;
74 if (!supportsScanning)
75 return std::nullopt;
76
77 std::optional<bool> supportsFullscreen;
78 decoder >> supportsFullscreen;
79 if (!supportsFullscreen)
80 return std::nullopt;
81
82 std::optional<bool> supportsPictureInPicture;
83 decoder >> supportsPictureInPicture;
84 if (!supportsPictureInPicture)
85 return std::nullopt;
86
87 std::optional<bool> supportsAcceleratedRendering;
88 decoder >> supportsAcceleratedRendering;
89 if (!supportsAcceleratedRendering)
90 return std::nullopt;
91
92 std::optional<bool> canPlayToWirelessPlaybackTarget;
93 decoder >> canPlayToWirelessPlaybackTarget;
94 if (!canPlayToWirelessPlaybackTarget)
95 return std::nullopt;
96
97 std::optional<bool> shouldIgnoreIntrinsicSize;
98 decoder >> shouldIgnoreIntrinsicSize;
99 if (!shouldIgnoreIntrinsicSize)
100 return std::nullopt;
101
102 return {{
103 WTFMove(*engineDescription),
104 *maximumDurationToCacheMediaTime,
105 *supportsScanning,
106 *supportsFullscreen,
107 *supportsPictureInPicture,
108 *supportsAcceleratedRendering,
109 *canPlayToWirelessPlaybackTarget,
110 *shouldIgnoreIntrinsicSize,
111 }};
112 }
113};
