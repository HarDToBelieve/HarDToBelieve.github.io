142std::unique_ptr<MediaPlayerPrivateInterface> RemoteMediaPlayerManager::createRemoteMediaPlayer(MediaPlayer* player, MediaPlayerEnums::MediaEngineIdentifier remoteEngineIdentifier)
143{
144 RemoteMediaPlayerProxyConfiguration proxyConfiguration;
145 proxyConfiguration.referrer = player->referrer();
146 proxyConfiguration.userAgent = player->userAgent();
147 proxyConfiguration.sourceApplicationIdentifier = player->sourceApplicationIdentifier();
148#if PLATFORM(IOS_FAMILY)
149 proxyConfiguration.networkInterfaceName = player->mediaPlayerNetworkInterfaceName();
150#endif
151 proxyConfiguration.mediaContentTypesRequiringHardwareSupport = player->mediaContentTypesRequiringHardwareSupport();
152 proxyConfiguration.preferredAudioCharacteristics = player->preferredAudioCharacteristics();
153#if !RELEASE_LOG_DISABLED
154 proxyConfiguration.logIdentifier = reinterpret_cast<uint64_t>(player->mediaPlayerLogIdentifier());
155#endif
156 proxyConfiguration.shouldUsePersistentCache = player->shouldUsePersistentCache();
157 proxyConfiguration.isVideo = player->isVideoPlayer();
158
159#if ENABLE(AVF_CAPTIONS)
160 for (const auto& track : player->outOfBandTrackSources())
161 proxyConfiguration.outOfBandTrackData.append(track->data());
162#endif
163
164 auto documentSecurityOrigin = player->documentSecurityOrigin();
165 proxyConfiguration.documentSecurityOrigin = documentSecurityOrigin;
166
167 auto identifier = MediaPlayerIdentifier::generate();
168 gpuProcessConnection().connection().send(Messages::RemoteMediaPlayerManagerProxy::CreateMediaPlayer(identifier, remoteEngineIdentifier, proxyConfiguration), 0);
169
170 auto remotePlayer = MediaPlayerPrivateRemote::create(player, remoteEngineIdentifier, identifier, *this);
171 m_players.add(identifier, makeWeakPtr(*remotePlayer));
172
173 return remotePlayer;
174}

211HashSet<SecurityOriginData> RemoteMediaPlayerManager::originsInMediaCache(MediaPlayerEnums::MediaEngineIdentifier remoteEngineIdentifier, const String& path)
212{
213 HashSet<SecurityOriginData> originData;
214 if (!gpuProcessConnection().connection().sendSync(Messages::RemoteMediaPlayerManagerProxy::OriginsInMediaCache(remoteEngineIdentifier, path), Messages::RemoteMediaPlayerManagerProxy::OriginsInMediaCache::Reply(originData), 0))

217}

235void RemoteMediaPlayerManager::setUseGPUProcess(bool useGPUProcess)
236{
237 auto registerEngine = [this](MediaEngineRegistrar registrar, MediaPlayerEnums::MediaEngineIdentifier remoteEngineIdentifier) {
238 registrar(makeUnique<MediaPlayerRemoteFactory>(remoteEngineIdentifier, *this));
239 };
240
241 RemoteMediaPlayerSupport::setRegisterRemotePlayerCallback(useGPUProcess ? WTFMove(registerEngine) : RemoteMediaPlayerSupport::RegisterRemotePlayerCallback());
242
243#if PLATFORM(COCOA) && ENABLE(MEDIA_STREAM)
244 if (useGPUProcess) {
245 WebCore::SampleBufferDisplayLayer::setCreator([](auto& client) {
246 return WebProcess::singleton().ensureGPUProcessConnection().sampleBufferDisplayLayerManager().createLayer(client);
247 });
248 }
249#endif
250}

252GPUProcessConnection& RemoteMediaPlayerManager::gpuProcessConnection() const
253{
254 if (!m_gpuProcessConnection) {
255 m_gpuProcessConnection = &WebProcess::singleton().ensureGPUProcessConnection();
256 m_gpuProcessConnection->addClient(*this);
257 }
258
259 return *m_gpuProcessConnection;
260}
261
262void RemoteMediaPlayerManager::gpuProcessConnectionDidClose(GPUProcessConnection& connection)
263{
264 ASSERT(m_gpuProcessConnection == &connection);
265 connection.removeClient(*this);
266
267 m_gpuProcessConnection = nullptr;
268
269 auto players = m_players;
270 for (auto& player : players.values()) {
271 if (player) {
272 player->player()->reloadAndResumePlaybackIfNeeded();
273 ASSERT_WITH_MESSAGE(!player, "reloadAndResumePlaybackIfNeeded should destroy this player and construct a new one");
274 }
275 }
