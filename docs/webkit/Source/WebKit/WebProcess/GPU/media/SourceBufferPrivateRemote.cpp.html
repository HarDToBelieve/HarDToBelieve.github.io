51Ref<SourceBufferPrivateRemote> SourceBufferPrivateRemote::create(GPUProcessConnection& gpuProcessConnection, RemoteSourceBufferIdentifier remoteSourceBufferIdentifier, const MediaSourcePrivateRemote& mediaSourcePrivate, const MediaPlayerPrivateRemote& mediaPlayerPrivate)
52{
53 return adoptRef(*new SourceBufferPrivateRemote(gpuProcessConnection, remoteSourceBufferIdentifier, mediaSourcePrivate, mediaPlayerPrivate));

56SourceBufferPrivateRemote::SourceBufferPrivateRemote(GPUProcessConnection& gpuProcessConnection, RemoteSourceBufferIdentifier remoteSourceBufferIdentifier, const MediaSourcePrivateRemote& mediaSourcePrivate, const MediaPlayerPrivateRemote& mediaPlayerPrivate)
57 : m_gpuProcessConnection(makeWeakPtr(gpuProcessConnection))
58 , m_remoteSourceBufferIdentifier(remoteSourceBufferIdentifier)
59 , m_mediaSourcePrivate(makeWeakPtr(mediaSourcePrivate))
60 , m_mediaPlayerPrivate(makeWeakPtr(mediaPlayerPrivate))
61#if !RELEASE_LOG_DISABLED
62 , m_logger(m_mediaSourcePrivate->logger())
63 , m_logIdentifier(m_mediaSourcePrivate->nextSourceBufferLogIdentifier())
64#endif
65{
66 ALWAYS_LOG(LOGIDENTIFIER);
67 m_gpuProcessConnection->messageReceiverMap().addMessageReceiver(Messages::SourceBufferPrivateRemote::messageReceiverName(), m_remoteSourceBufferIdentifier.toUInt64(), *this);
68}

70SourceBufferPrivateRemote::~SourceBufferPrivateRemote()
71{
72 ALWAYS_LOG(LOGIDENTIFIER);

77
78 m_gpuProcessConnection->messageReceiverMap().removeMessageReceiver(Messages::SourceBufferPrivateRemote::messageReceiverName(), m_remoteSourceBufferIdentifier.toUInt64());
79}

89void SourceBufferPrivateRemote::abort()
90{
91 if (!m_gpuProcessConnection)
92 return;
93
94 // When abort() is being issued; this will force the RemoteSourceBufferProxy to abort the current remote source buffer operation
95 // The totalTrackBufferSizeInBytes will be recalculated after the next operation which allows for potentially having
96 // m_totalTrackBufferSizeInBytes being temporarily stale as it won't be used until then.
97 m_gpuProcessConnection->connection().send(Messages::RemoteSourceBufferProxy::Abort(), m_remoteSourceBufferIdentifier);
98}

189void SourceBufferPrivateRemote::removeCodedFrames(const MediaTime& start, const MediaTime& end, const MediaTime& currentMediaTime, bool isEnded, CompletionHandler<void()>&& completionHandler)
190{
191 if (!m_gpuProcessConnection)
192 return;
193
194 m_gpuProcessConnection->connection().sendWithAsyncReply(
195 Messages::RemoteSourceBufferProxy::RemoveCodedFrames(start, end, currentMediaTime, isEnded), [this, protectedThis = makeRef(*this), completionHandler = WTFMove(completionHandler)](auto&& buffered, uint64_t totalTrackBufferSizeInBytes) mutable {
196 setBufferedRanges(buffered);
197 m_totalTrackBufferSizeInBytes = totalTrackBufferSizeInBytes;
198 completionHandler();
199 },
200 m_remoteSourceBufferIdentifier);
201}
202
203void SourceBufferPrivateRemote::evictCodedFrames(uint64_t newDataSize, uint64_t maximumBufferSize, const MediaTime& currentTime, const MediaTime& duration, bool isEnded)
204{
205 if (!m_gpuProcessConnection)
206 return;
207
208 uint64_t totalBufferSizeInBytes = 0;
209 if (m_gpuProcessConnection->connection().sendSync(Messages::RemoteSourceBufferProxy::EvictCodedFrames(newDataSize, maximumBufferSize, currentTime, duration, isEnded), Messages::RemoteSourceBufferProxy::EvictCodedFrames::Reply(totalBufferSizeInBytes), m_remoteSourceBufferIdentifier)) {
210 m_totalTrackBufferSizeInBytes = totalBufferSizeInBytes;
211 }
212}

329void SourceBufferPrivateRemote::updateTrackIds(Vector<std::pair<AtomString, AtomString>>&& trackIdPairs)
330{
331 if (!m_gpuProcessConnection)
332 return;
333
334 Vector<std::pair<TrackPrivateRemoteIdentifier, TrackPrivateRemoteIdentifier>> identifierPairs;
335
336 for (auto& trackIdPair : trackIdPairs) {
337 ASSERT(m_trackIdentifierMap.contains(trackIdPair.first));
338 ASSERT(m_trackIdentifierMap.contains(trackIdPair.second));
339
340 auto oldIdentifier = m_trackIdentifierMap.take(trackIdPair.first);
341 auto newIdentifier = m_trackIdentifierMap.get(trackIdPair.second);
342 identifierPairs.append(std::make_pair(oldIdentifier, newIdentifier));
343 }
344
345 m_gpuProcessConnection->connection().send(Messages::RemoteSourceBufferProxy::UpdateTrackIds(identifierPairs), m_remoteSourceBufferIdentifier);
346}

368void SourceBufferPrivateRemote::sourceBufferPrivateDidReceiveInitializationSegment(InitializationSegmentInfo&& segmentInfo, CompletionHandler<void()>&& completionHandler)
369{
370 if (!m_client || !m_mediaPlayerPrivate) {
371 completionHandler();
372 return;
373 }
374
375 SourceBufferPrivateClient::InitializationSegment segment;
376 segment.duration = segmentInfo.duration;
377
378 for (auto& audioTrack : segmentInfo.audioTracks) {
379 SourceBufferPrivateClient::InitializationSegment::AudioTrackInformation info;
380 info.track = m_mediaPlayerPrivate->audioTrackPrivateRemote(audioTrack.identifier);
381 info.description = RemoteMediaDescription::create(audioTrack.description);
382 segment.audioTracks.append(info);
383 m_trackIdentifierMap.add(info.track->id(), audioTrack.identifier);
384 }
385
386 for (auto& videoTrack : segmentInfo.videoTracks) {
387 SourceBufferPrivateClient::InitializationSegment::VideoTrackInformation info;
388 info.track = m_mediaPlayerPrivate->videoTrackPrivateRemote(videoTrack.identifier);
389 info.description = RemoteMediaDescription::create(videoTrack.description);
390 segment.videoTracks.append(info);
391 m_trackIdentifierMap.add(info.track->id(), videoTrack.identifier);
392 }
393
394 for (auto& textTrack : segmentInfo.textTracks) {
395 SourceBufferPrivateClient::InitializationSegment::TextTrackInformation info;
396 info.track = m_mediaPlayerPrivate->textTrackPrivateRemote(textTrack.identifier);
397 info.description = RemoteMediaDescription::create(textTrack.description);
398 segment.textTracks.append(info);
399 m_trackIdentifierMap.add(info.track->id(), textTrack.identifier);
400 }
401
402 m_client->sourceBufferPrivateDidReceiveInitializationSegment(WTFMove(segment), WTFMove(completionHandler));
403}
