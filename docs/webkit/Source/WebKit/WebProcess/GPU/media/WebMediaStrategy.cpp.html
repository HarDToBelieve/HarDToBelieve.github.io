48#if ENABLE(WEB_AUDIO)
49Ref<WebCore::AudioDestination> WebMediaStrategy::createAudioDestination(WebCore::AudioIOCallback& callback, const String& inputDeviceId,
50 unsigned numberOfInputChannels, unsigned numberOfOutputChannels, float sampleRate)
51{
52#if ENABLE(GPU_PROCESS)
53 if (m_useGPUProcess)
54 return RemoteAudioDestinationProxy::create(callback, inputDeviceId, numberOfInputChannels, numberOfOutputChannels, sampleRate);
55#endif
56 return WebCore::AudioDestination::create(callback, inputDeviceId, numberOfInputChannels, numberOfOutputChannels, sampleRate);
57}
58#endif

60std::unique_ptr<WebCore::NowPlayingManager> WebMediaStrategy::createNowPlayingManager() const
61{
62#if ENABLE(GPU_PROCESS)
63 if (m_useGPUProcess) {
64 class NowPlayingInfoForGPUManager : public WebCore::NowPlayingManager {
65 void clearNowPlayingInfoPrivate() final
66 {
67 if (auto* connection = WebProcess::singleton().existingGPUProcessConnection())
68 connection->connection().send(Messages::GPUConnectionToWebProcess::ClearNowPlayingInfo { }, 0);
69 }
70
71 void setNowPlayingInfoPrivate(const WebCore::NowPlayingInfo& nowPlayingInfo) final
72 {
73 auto& connection = WebProcess::singleton().ensureGPUProcessConnection().connection();
74 connection.send(Messages::GPUConnectionToWebProcess::SetNowPlayingInfo { nowPlayingInfo }, 0);
75 }
76 };
77 return makeUnique<NowPlayingInfoForGPUManager>();
78 }
79#endif
80 return WebCore::MediaStrategy::createNowPlayingManager();
81}
