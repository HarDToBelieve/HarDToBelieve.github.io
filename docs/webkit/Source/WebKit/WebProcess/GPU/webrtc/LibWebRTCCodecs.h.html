62class LibWebRTCCodecs : public IPC::Connection::ThreadMessageReceiverRefCounted, public GPUProcessConnection::Client {
63 WTF_MAKE_FAST_ALLOCATED;
64public:
65 static Ref<LibWebRTCCodecs> create();
66 ~LibWebRTCCodecs();
67
68 static void setCallbacks(bool useGPUProcess);
69
70 enum class Type { H264, H265, VP9 };
71 struct Decoder {
72 WTF_MAKE_FAST_ALLOCATED;
73 public:
74 RTCDecoderIdentifier identifier;
75 Type type;
76 void* decodedImageCallback WTF_GUARDED_BY_LOCK(decodedImageCallbackLock) { nullptr };
77 Lock decodedImageCallbackLock;
78 bool hasError { false };
79 RefPtr<IPC::Connection> connection;
80 };
81
82 Decoder* createDecoder(Type);
83 int32_t releaseDecoder(Decoder&);
84 int32_t decodeFrame(Decoder&, uint32_t timeStamp, const uint8_t*, size_t, uint16_t width, uint16_t height);
85 void registerDecodeFrameCallback(Decoder&, void* decodedImageCallback);
86
87 struct EncoderInitializationData {
88 uint16_t width;
89 uint16_t height;
90 unsigned startBitRate;
91 unsigned maxBitRate;
92 unsigned minBitRate;
93 uint32_t maxFrameRate;
94 };
95 struct Encoder {
96 WTF_MAKE_FAST_ALLOCATED;
97 public:
98 RTCEncoderIdentifier identifier;
99 webrtc::VideoCodecType codecType { webrtc::kVideoCodecGeneric };
100 Vector<std::pair<String, String>> parameters;
101 std::optional<EncoderInitializationData> initializationData;
102 void* encodedImageCallback WTF_GUARDED_BY_LOCK(encodedImageCallbackLock) { nullptr };
103 Lock encodedImageCallbackLock;
104 RefPtr<IPC::Connection> connection;
105 };
106
107 Encoder* createEncoder(Type, const std::map<std::string, std::string>&);
108 int32_t releaseEncoder(Encoder&);
109 int32_t initializeEncoder(Encoder&, uint16_t width, uint16_t height, unsigned startBitrate, unsigned maxBitrate, unsigned minBitrate, uint32_t maxFramerate);
110 int32_t encodeFrame(Encoder&, const webrtc::VideoFrame&, bool shouldEncodeAsKeyFrame);
111 void registerEncodeFrameCallback(Encoder&, void* encodedImageCallback);
112 void setEncodeRates(Encoder&, uint32_t bitRate, uint32_t frameRate);
113
114 CVPixelBufferPoolRef pixelBufferPool(size_t width, size_t height, OSType);
115
116 void didReceiveMessage(IPC::Connection&, IPC::Decoder&) final;
117
118 void setVP9VTBSupport(bool supportVP9VTB) { m_supportVP9VTB = supportVP9VTB; }
119 bool supportVP9VTB() const { return m_supportVP9VTB; }
120 void setLoggingLevel(WTFLogLevel);
121
122private:
123 LibWebRTCCodecs();
124 void ensureGPUProcessConnectionAndDispatchToThread(Function<void()>&&);
125 void ensureGPUProcessConnectionOnMainThreadWithLock() WTF_REQUIRES_LOCK(m_connectionLock);
126 void gpuProcessConnectionMayNoLongerBeNeeded();
127
128 void failedDecoding(RTCDecoderIdentifier);
129 void completedDecoding(RTCDecoderIdentifier, uint32_t timeStamp, WebCore::RemoteVideoSample&&);
130 void completedEncoding(RTCEncoderIdentifier, IPC::DataReference&&, const webrtc::WebKitEncodedFrameInfo&);
131 RetainPtr<CVPixelBufferRef> convertToBGRA(CVPixelBufferRef);
132
133 // IPC::Connection::ThreadMessageReceiver
134 void dispatchToThread(Function<void()>&&) final;
135
136 // GPUProcessConnection::Client
137 void gpuProcessConnectionDidClose(GPUProcessConnection&);
138
139private:
140 HashMap<RTCDecoderIdentifier, std::unique_ptr<Decoder>> m_decoders;
141 HashSet<RTCDecoderIdentifier> m_decodingErrors;
142
143 HashMap<RTCEncoderIdentifier, std::unique_ptr<Encoder>> m_encoders;
144
145 std::atomic<bool> m_needsGPUProcessConnection;
146
147 Lock m_connectionLock;
148 RefPtr<IPC::Connection> m_connection WTF_GUARDED_BY_LOCK(m_connectionLock);
149 Vector<Function<void()>> m_tasksToDispatchAfterEstablishingConnection;
150
151 Ref<WorkQueue> m_queue;
152 std::unique_ptr<WebCore::ImageTransferSessionVT> m_imageTransferSession;
153 std::unique_ptr<WebCore::PixelBufferConformerCV> m_pixelBufferConformer;
154 RetainPtr<CVPixelBufferPoolRef> m_pixelBufferPool;
155 size_t m_pixelBufferPoolWidth { 0 };
156 size_t m_pixelBufferPoolHeight { 0 };
157 bool m_supportVP9VTB { false };
158 std::optional<WTFLogLevel> m_loggingLevel;
159};
