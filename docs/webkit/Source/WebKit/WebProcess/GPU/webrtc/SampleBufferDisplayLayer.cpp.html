45std::unique_ptr<SampleBufferDisplayLayer> SampleBufferDisplayLayer::create(SampleBufferDisplayLayerManager& manager, Client& client)

47 return std::unique_ptr<SampleBufferDisplayLayer>(new SampleBufferDisplayLayer(manager, client));

50SampleBufferDisplayLayer::SampleBufferDisplayLayer(SampleBufferDisplayLayerManager& manager, Client& client)
51 : WebCore::SampleBufferDisplayLayer(client)
52 , m_manager(makeWeakPtr(manager))
53 , m_connection(WebProcess::singleton().ensureGPUProcessConnection().connection())
54 , m_identifier(SampleBufferDisplayLayerIdentifier::generate())
55{
56 manager.addLayer(*this);
57}
58
59void SampleBufferDisplayLayer::initialize(bool hideRootLayer, IntSize size, CompletionHandler<void(bool)>&& callback)
60{
61 m_connection->sendWithAsyncReply(Messages::RemoteSampleBufferDisplayLayerManager::CreateLayer { m_identifier, hideRootLayer, size }, [this, weakThis = makeWeakPtr(this), callback = WTFMove(callback)](auto contextId) mutable {
62 if (!weakThis)
63 return callback(false);
64 if (contextId)
65 m_videoLayer = LayerHostingContext::createPlatformLayerForHostingContext(*contextId);
66 callback(!!m_videoLayer);
67 });
68}
69
70SampleBufferDisplayLayer::~SampleBufferDisplayLayer()
71{
72 m_connection->send(Messages::RemoteSampleBufferDisplayLayerManager::ReleaseLayer { m_identifier }, 0);
73 if (m_manager)
74 m_manager->removeLayer(*this);
75}

119void SampleBufferDisplayLayer::enqueueSample(MediaSample& sample)
120{
121 if (m_paused)
122 return;
123 if (auto remoteSample = RemoteVideoSample::create(sample))
124 m_connection->send(Messages::RemoteSampleBufferDisplayLayer::EnqueueSample { *remoteSample }, m_identifier);
125}
