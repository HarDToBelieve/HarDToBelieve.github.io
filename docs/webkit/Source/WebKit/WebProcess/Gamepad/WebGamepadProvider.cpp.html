56void WebGamepadProvider::setInitialGamepads(const Vector<GamepadData>& gamepadDatas)
57{
58 ASSERT(m_gamepads.isEmpty());
59
60 m_gamepads.resize(gamepadDatas.size());
61 m_rawGamepads.resize(gamepadDatas.size());
62 for (size_t i = 0; i < gamepadDatas.size(); ++i) {
63 if (gamepadDatas[i].isNull())
64 continue;
65
66 m_gamepads[i] = makeUnique<WebGamepad>(gamepadDatas[i]);
67 m_rawGamepads[i] = m_gamepads[i].get();
68 }
69}
70
71void WebGamepadProvider::gamepadConnected(const GamepadData& gamepadData, EventMakesGamepadsVisible eventVisibility)
72{
73 LOG(Gamepad, "WebGamepadProvider::gamepadConnected - Gamepad index %u attached (visibility: %i)\n", gamepadData.index(), (int)eventVisibility);
74
75 if (m_gamepads.size() <= gamepadData.index()) {
76 m_gamepads.resize(gamepadData.index() + 1);
77 m_rawGamepads.resize(gamepadData.index() + 1);
78 }
79
80 ASSERT(!m_gamepads[gamepadData.index()]);
81
82 m_gamepads[gamepadData.index()] = makeUnique<WebGamepad>(gamepadData);
83 m_rawGamepads[gamepadData.index()] = m_gamepads[gamepadData.index()].get();
84
85 for (auto* client : m_clients)
86 client->platformGamepadConnected(*m_gamepads[gamepadData.index()], eventVisibility);
87}
88
89void WebGamepadProvider::gamepadDisconnected(unsigned index)
90{
91 ASSERT(m_gamepads.size() > index);
92
93 std::unique_ptr<WebGamepad> disconnectedGamepad = WTFMove(m_gamepads[index]);
94 m_rawGamepads[index] = nullptr;
95
96 LOG(Gamepad, "WebGamepadProvider::gamepadDisconnected - Gamepad index %u detached (m_gamepads size %zu, m_rawGamepads size %zu\n", index, m_gamepads.size(), m_rawGamepads.size());
97
98 for (auto* client : m_clients)
99 client->platformGamepadDisconnected(*disconnectedGamepad);
100}

102void WebGamepadProvider::gamepadActivity(const Vector<GamepadData>& gamepadDatas, EventMakesGamepadsVisible eventVisibility)
103{
104 LOG(Gamepad, "WebGamepadProvider::gamepadActivity - %zu gamepad datas with %zu local web gamepads\n", gamepadDatas.size(), m_gamepads.size());
105
106 ASSERT(m_gamepads.size() == gamepadDatas.size());
107
108 for (size_t i = 0; i < m_gamepads.size(); ++i) {
109 if (m_gamepads[i])
110 m_gamepads[i]->updateValues(gamepadDatas[i]);
111 }
112
113 for (auto* client : m_clients)
114 client->platformGamepadInputActivity(eventVisibility);
115}

117void WebGamepadProvider::startMonitoringGamepads(GamepadProviderClient& client)
118{
119 bool processHadGamepadClients = !m_clients.isEmpty();
120
121 ASSERT(!m_clients.contains(&client));
122 m_clients.add(&client);
123
124 if (!processHadGamepadClients)
125 WebProcess::singleton().send(Messages::WebProcessPool::StartedUsingGamepads(), 0);
126}
127
128void WebGamepadProvider::stopMonitoringGamepads(GamepadProviderClient& client)
129{
130 bool processHadGamepadClients = !m_clients.isEmpty();
131
132 ASSERT(m_clients.contains(&client));
133 m_clients.remove(&client);
134
135 if (processHadGamepadClients && m_clients.isEmpty())
136 WebProcess::singleton().send(Messages::WebProcessPool::StoppedUsingGamepads(), 0);
137}
