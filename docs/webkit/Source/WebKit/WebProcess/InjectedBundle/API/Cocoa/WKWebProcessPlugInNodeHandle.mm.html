66- (CocoaImage *)renderedImageWithOptions:(WKSnapshotOptions)options width:(NSNumber *)width
67{
68 std::optional<float> optionalWidth;
69 if (width)
70 optionalWidth = width.floatValue;
71
72 RefPtr<WebKit::WebImage> image = _nodeHandle->renderedImage(WebKit::toSnapshotOptions(options), options & kWKSnapshotOptionsExcludeOverflow, optionalWidth);
73 if (!image)
74 return nil;
75
76#if USE(APPKIT)
77 return adoptNS([[NSImage alloc] initWithCGImage:image->bitmap().makeCGImage().get() size:NSZeroSize]).autorelease();
78#else
79 return adoptNS([[UIImage alloc] initWithCGImage:image->bitmap().makeCGImage().get()]).autorelease();
80#endif
81}

113static WebCore::AutoFillButtonType toAutoFillButtonType(_WKAutoFillButtonType autoFillButtonType)
114{
115 switch (autoFillButtonType) {
116 case _WKAutoFillButtonTypeNone:
117 return WebCore::AutoFillButtonType::None;
118 case _WKAutoFillButtonTypeContacts:
119 return WebCore::AutoFillButtonType::Contacts;
120 case _WKAutoFillButtonTypeCredentials:
121 return WebCore::AutoFillButtonType::Credentials;
122 case _WKAutoFillButtonTypeStrongPassword:
123 return WebCore::AutoFillButtonType::StrongPassword;
124 case _WKAutoFillButtonTypeCreditCard:
125 return WebCore::AutoFillButtonType::CreditCard;
126 }
127 ASSERT_NOT_REACHED();
128 return WebCore::AutoFillButtonType::None;
129}
130
131static _WKAutoFillButtonType toWKAutoFillButtonType(WebCore::AutoFillButtonType autoFillButtonType)
132{
133 switch (autoFillButtonType) {
134 case WebCore::AutoFillButtonType::None:
135 return _WKAutoFillButtonTypeNone;
136 case WebCore::AutoFillButtonType::Contacts:
137 return _WKAutoFillButtonTypeContacts;
138 case WebCore::AutoFillButtonType::Credentials:
139 return _WKAutoFillButtonTypeCredentials;
140 case WebCore::AutoFillButtonType::StrongPassword:
141 return _WKAutoFillButtonTypeStrongPassword;
142 case WebCore::AutoFillButtonType::CreditCard:
143 return _WKAutoFillButtonTypeCreditCard;
144 }
145 ASSERT_NOT_REACHED();
146 return _WKAutoFillButtonTypeNone;
147
148}
