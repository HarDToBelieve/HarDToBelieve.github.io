119WKArrayRef WKBundleGetLiveDocumentURLsForTesting(WKBundleRef bundleRef, bool excludeDocumentsInPageGroupPages)
120{
121 auto liveDocuments = WebKit::toImpl(bundleRef)->liveDocumentURLs(excludeDocumentsInPageGroupPages);
122
123 auto liveURLs = adoptWK(WKMutableArrayCreate());
124
125 for (const auto& it : liveDocuments) {
126 auto urlInfo = adoptWK(WKMutableDictionaryCreate());
127
128 auto documentIDKey = adoptWK(WKStringCreateWithUTF8CString("id"));
129 auto documentURLKey = adoptWK(WKStringCreateWithUTF8CString("url"));
130
131 auto documentIDValue = adoptWK(WKUInt64Create(it.key));
132 auto documentURLValue = adoptWK(WebKit::toCopiedAPI(it.value));
133
134 WKDictionarySetItem(urlInfo.get(), documentIDKey.get(), documentIDValue.get());
135 WKDictionarySetItem(urlInfo.get(), documentURLKey.get(), documentURLValue.get());
136
137 WKArrayAppendItem(liveURLs.get(), urlInfo.get());
138 }
139
140 return liveURLs.leakRef();
141}
