37static WebCore::AutoFillButtonType toAutoFillButtonType(WKAutoFillButtonType wkAutoFillButtonType)
38{
39 switch (wkAutoFillButtonType) {
40 case kWKAutoFillButtonTypeNone:
41 return WebCore::AutoFillButtonType::None;
42 case kWKAutoFillButtonTypeContacts:
43 return WebCore::AutoFillButtonType::Contacts;
44 case kWKAutoFillButtonTypeCredentials:
45 return WebCore::AutoFillButtonType::Credentials;
46 case kWKAutoFillButtonTypeStrongPassword:
47 return WebCore::AutoFillButtonType::StrongPassword;
48 case kWKAutoFillButtonTypeCreditCard:
49 return WebCore::AutoFillButtonType::CreditCard;
50 }
51 ASSERT_NOT_REACHED();
52 return WebCore::AutoFillButtonType::None;
53}
54
55static WKAutoFillButtonType toWKAutoFillButtonType(WebCore::AutoFillButtonType autoFillButtonType)
56{
57 switch (autoFillButtonType) {
58 case WebCore::AutoFillButtonType::None:
59 return kWKAutoFillButtonTypeNone;
60 case WebCore::AutoFillButtonType::Contacts:
61 return kWKAutoFillButtonTypeContacts;
62 case WebCore::AutoFillButtonType::Credentials:
63 return kWKAutoFillButtonTypeCredentials;
64 case WebCore::AutoFillButtonType::StrongPassword:
65 return kWKAutoFillButtonTypeStrongPassword;
66 case WebCore::AutoFillButtonType::CreditCard:
67 return kWKAutoFillButtonTypeCreditCard;
68 }
69 ASSERT_NOT_REACHED();
70 return kWKAutoFillButtonTypeNone;
71}
