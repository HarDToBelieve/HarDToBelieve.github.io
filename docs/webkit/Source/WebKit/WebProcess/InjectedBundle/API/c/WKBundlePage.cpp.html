244void* WKAccessibilityRootObject(WKBundlePageRef pageRef)
245{
246#if ENABLE(ACCESSIBILITY)
247 if (!pageRef)
248 return 0;
249
250 WebCore::Page* page = WebKit::toImpl(pageRef)->corePage();
251 if (!page)
252 return 0;
253
254 WebCore::Frame& core = page->mainFrame();
255 if (!core.document())
256 return 0;
257
258 WebCore::AXObjectCache::enableAccessibility();
259
260 WebCore::AXCoreObject* root = core.document()->axObjectCache()->rootObject();
261 if (!root)
262 return 0;
263
264 return root->wrapper();
265#else
266 UNUSED_PARAM(pageRef);
267 return 0;
268#endif
269}

271void* WKAccessibilityFocusedObject(WKBundlePageRef pageRef)
272{
273#if ENABLE(ACCESSIBILITY)
274 if (!pageRef)
275 return 0;
276
277 WebCore::Page* page = WebKit::toImpl(pageRef)->corePage();
278 if (!page)
279 return 0;
280
281 auto* focusedDocument = page->focusController().focusedOrMainFrame().document();
282 if (!focusedDocument)
283 return 0;
284
285 WebCore::AXObjectCache::enableAccessibility();
286
287 auto* axObjectCache = focusedDocument->axObjectCache();
288 if (!axObjectCache)
289 return 0;
290
291 auto* focusedObject = axObjectCache->focusedUIElementForPage(page);
292 if (!focusedObject)
293 return 0;
294
295 return focusedObject->wrapper();
296#else
297 UNUSED_PARAM(pageRef);
298 return 0;
299#endif
300}

499void WKBundlePageReplaceStringMatches(WKBundlePageRef pageRef, WKArrayRef matchIndicesRef, WKStringRef replacementText, bool selectionOnly)
500{
501 auto* matchIndices = WebKit::toImpl(matchIndicesRef);
502
503 Vector<uint32_t> indices;
504 indices.reserveInitialCapacity(matchIndices->size());
505
506 auto numberOfMatchIndices = matchIndices->size();
507 for (size_t i = 0; i < numberOfMatchIndices; ++i) {
508 if (auto* indexAsObject = matchIndices->at<API::UInt64>(i))
509 indices.uncheckedAppend(indexAsObject->value());
510 }
511 WebKit::toImpl(pageRef)->replaceStringMatchesFromInjectedBundle(indices, WebKit::toWTFString(replacementText), selectionOnly);
512}

621void WKBundlePageSetComposition(WKBundlePageRef pageRef, WKStringRef text, int from, int length, bool suppressUnderline, WKArrayRef highlightData)
622{
623 Vector<WebCore::CompositionHighlight> highlights;
624 if (highlightData) {
625 auto* highlightDataArray = WebKit::toImpl(highlightData);
626 highlights.reserveInitialCapacity(highlightDataArray->size());
627 for (auto dictionary : highlightDataArray->elementsOfType<API::Dictionary>()) {
628 auto startOffset = static_cast<API::UInt64*>(dictionary->get("from"))->value();
629 highlights.uncheckedAppend({
630 static_cast<unsigned>(startOffset),
631 static_cast<unsigned>(startOffset + static_cast<API::UInt64*>(dictionary->get("length"))->value()),
632 WebCore::CSSParser::parseColor(static_cast<API::String*>(dictionary->get("color"))->string())
633 });
634 }
635 }
636 WebKit::toImpl(pageRef)->setCompositionForTesting(WebKit::toWTFString(text), from, length, suppressUnderline, highlights);
637}
