36WKTypeID WKBundleRangeHandleGetTypeID()
37{
38 return WebKit::toAPI(WebKit::InjectedBundleRangeHandle::APIType);
39}
40
41WKBundleRangeHandleRef WKBundleRangeHandleCreate(JSContextRef contextRef, JSObjectRef objectRef)
42{
43 RefPtr<WebKit::InjectedBundleRangeHandle> rangeHandle = WebKit::InjectedBundleRangeHandle::getOrCreate(contextRef, objectRef);
44 return toAPI(rangeHandle.leakRef());
45}
46
47WKRect WKBundleRangeHandleGetBoundingRectInWindowCoordinates(WKBundleRangeHandleRef rangeHandleRef)
48{
49 WebCore::IntRect boundingRect = WebKit::toImpl(rangeHandleRef)->boundingRectInWindowCoordinates();
50 return WKRectMake(boundingRect.x(), boundingRect.y(), boundingRect.width(), boundingRect.height());
51}
52
53WKImageRef WKBundleRangeHandleCopySnapshotWithOptions(WKBundleRangeHandleRef rangeHandleRef, WKSnapshotOptions options)
54{
55 RefPtr<WebKit::WebImage> image = WebKit::toImpl(rangeHandleRef)->renderedImage(WebKit::toSnapshotOptions(options));
56 return toAPI(image.leakRef());
57}
