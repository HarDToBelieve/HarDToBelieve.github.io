45class PageBannerClientImpl : API::Client<WKBundlePageBannerClientBase>, public PageBanner::Client {
46 WTF_MAKE_FAST_ALLOCATED;
47public:
48 explicit PageBannerClientImpl(WKBundlePageBannerClientBase* client)
49 {
50 initialize(client);
51 }
52
53 virtual ~PageBannerClientImpl()
54 {
55 }
56
57private:
58 // PageBanner::Client.
59 void pageBannerDestroyed(PageBanner*) override
60 {
61 delete this;
62 }
63
64 bool mouseEvent(PageBanner* pageBanner, WebEvent::Type type, WebMouseEvent::Button button, const IntPoint& position) override
65 {
66 switch (type) {
67 case WebEvent::MouseDown: {
68 if (!m_client.mouseDown)
69 return false;
70
71 return m_client.mouseDown(toAPI(pageBanner), toAPI(position), toAPI(button), m_client.base.clientInfo);
72 }
73 case WebEvent::MouseUp: {
74 if (!m_client.mouseUp)
75 return false;
76
77 return m_client.mouseUp(toAPI(pageBanner), toAPI(position), toAPI(button), m_client.base.clientInfo);
78 }
79 case WebEvent::MouseMove: {
80 if (button == WebMouseEvent::NoButton) {
81 if (!m_client.mouseMoved)
82 return false;
83
84 return m_client.mouseMoved(toAPI(pageBanner), toAPI(position), m_client.base.clientInfo);
85 }
86
87 // This is a MouseMove event with a mouse button pressed. Call mouseDragged.
88 if (!m_client.mouseDragged)
89 return false;
90
91 return m_client.mouseDragged(toAPI(pageBanner), toAPI(position), toAPI(button), m_client.base.clientInfo);
92 }
93
94 default:
95 return false;
96 }
97 }
98};
99
100}

102WKBundlePageBannerRef WKBundlePageBannerCreateBannerWithCALayer(CALayer *layer, int height, WKBundlePageBannerClientBase* wkClient)
103{
104 if (wkClient && wkClient->version)
105 return 0;
106
107 auto clientImpl = makeUnique<WebKit::PageBannerClientImpl>(wkClient);
108 return toAPI(&WebKit::PageBanner::create(layer, height, clientImpl.release()).leakRef());
109}
110
111CALayer *WKBundlePageBannerGetLayer(WKBundlePageBannerRef pageBanner)
112{
113 return WebKit::toImpl(pageBanner)->layer();
114}
