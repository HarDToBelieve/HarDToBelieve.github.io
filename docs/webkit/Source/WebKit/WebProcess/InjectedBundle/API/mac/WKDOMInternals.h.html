57namespace WebKit {
58
59template<typename WebCoreType, typename WKDOMType>
60class DOMCache {
61public:
62 DOMCache()
63 {
64 }
65
66 void add(WebCoreType core, WKDOMType kit)
67 {
68 m_map.add(core, kit);
69 }
70
71 WKDOMType get(WebCoreType core)
72 {
73 return m_map.get(core);
74 }
75
76 void remove(WebCoreType core)
77 {
78 m_map.remove(core);
79 }
80
81private:
82 // This class should only ever be used as a singleton.
83 ~DOMCache() = delete;
84
85 HashMap<WebCoreType, WKDOMType> m_map;
86};
