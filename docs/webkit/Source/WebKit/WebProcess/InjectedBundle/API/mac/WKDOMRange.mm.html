38@implementation WKDOMRange
39
40- (id)_initWithImpl:(WebCore::Range*)impl
41{
42 self = [super init];
43 if (!self)
44 return nil;
45
46 _impl = impl;
47 WebKit::WKDOMRangeCache().add(impl, self);
48
49 return self;
50}
51
52- (id)initWithDocument:(WKDOMDocument *)document
53{
54 return [self _initWithImpl:WebCore::Range::create(*WebKit::toWebCoreDocument(document)).ptr()];
55}
56
57- (void)dealloc
58{
59 ensureOnMainRunLoop([range = WTFMove(_impl)] {
60 WebKit::WKDOMRangeCache().remove(range.get());
61 });
62 [super dealloc];
63}

65- (void)setStart:(WKDOMNode *)node offset:(int)offset
66{
67 if (!node)
68 return;
69 _impl->setStart(*WebKit::toWebCoreNode(node), offset);
70}
71
72- (void)setEnd:(WKDOMNode *)node offset:(int)offset
73{
74 if (!node)
75 return;
76 _impl->setEnd(*WebKit::toWebCoreNode(node), offset);
77}

79- (void)collapse:(BOOL)toStart

81 _impl->collapse(toStart);
82}

118- (NSString *)text
119{
120 auto range = makeSimpleRange(*_impl);
121 range.start.document().updateLayout();
122 return plainText(range);
123}
124
125- (BOOL)isCollapsed

127 return _impl->collapsed();

129
130- (NSArray *)textRects
131{
132 auto range = makeSimpleRange(*_impl);
133 range.start.document().updateLayoutIgnorePendingStylesheets();
134 return createNSArray(WebCore::RenderObject::absoluteTextRects(range)).autorelease();
135}
136
137- (WKDOMRange *)rangeByExpandingToWordBoundaryByCharacters:(NSUInteger)characters inDirection:(WKDOMRangeDirection)direction
138{
139 auto range = makeSimpleRange(*_impl);
140 auto newRange = rangeExpandedByCharactersInDirectionAtWordBoundary(makeDeprecatedLegacyPosition(direction == WKDOMRangeDirectionForward ? range.end : range.start), characters, direction == WKDOMRangeDirectionForward ? WebCore::SelectionDirection::Forward : WebCore::SelectionDirection::Backward);
141 return adoptNS([[WKDOMRange alloc] _initWithImpl:createLiveRange(newRange).get()]).autorelease();
142}
