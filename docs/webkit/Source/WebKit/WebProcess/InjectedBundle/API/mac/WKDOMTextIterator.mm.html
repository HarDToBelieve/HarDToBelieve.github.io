26#import "config.h"
27#import "WKDOMTextIterator.h"
28
29#import "WKDOMInternals.h"
30#import <WebCore/TextIterator.h>
31#import <WebKit/WKDOMRange.h>
32
33@interface WKDOMTextIterator () {
34@private
35 std::unique_ptr<WebCore::TextIterator> _textIterator;
36 Vector<unichar> _upconvertedText;
37}
38@end
39
40@implementation WKDOMTextIterator
41
42- (id)initWithRange:(WKDOMRange *)range
43{
44 self = [super init];
45 if (!self)
46 return nil;
47
48 if (!range)
49 return self;
50
51 _textIterator = makeUnique<WebCore::TextIterator>(makeSimpleRange(*WebKit::toWebCoreRange(range)));
52 return self;
53}
54
55- (void)advance
56{
57 if (_textIterator)
58 _textIterator->advance();
59 _upconvertedText.shrink(0);
60}
61
62- (BOOL)atEnd
63{
64 return _textIterator && _textIterator->atEnd();
65}
66
67- (WKDOMRange *)currentRange
68{
69 return _textIterator ? WebKit::toWKDOMRange(createLiveRange(_textIterator->range()).ptr()) : nil;
70}
71
72// FIXME: Consider deprecating this method and creating one that does not require copying 8-bit characters.
73- (const unichar*)currentTextPointer
74{
75 if (!_textIterator)
76 return nullptr;
77 StringView text = _textIterator->text();
78 unsigned length = text.length();
79 if (!length)
80 return nullptr;
81 if (!text.is8Bit())
82 return reinterpret_cast<const unichar*>(text.characters16());
83 if (_upconvertedText.isEmpty())
84 _upconvertedText.appendRange(text.characters8(), text.characters8() + length);
85 ASSERT(_upconvertedText.size() == text.length());
86 return _upconvertedText.data();
87}
88
89- (NSUInteger)currentTextLength
90{
91 return _textIterator ? _textIterator->text().length() : 0;
92}
93
94@end
