270static WKURLRequestRef willSendRequestForFrame(WKBundlePageRef, WKBundleFrameRef frame, uint64_t resourceIdentifier, WKURLRequestRef request, WKURLResponseRef redirectResponse, const void* clientInfo)
271{
272 auto pluginContextController = (__bridge WKWebProcessPlugInBrowserContextController *)clientInfo;
273 auto loadDelegate = pluginContextController->_loadDelegate.get();
274
275 if ([loadDelegate respondsToSelector:@selector(webProcessPlugInBrowserContextController:frame:willSendRequestForResource:request:redirectResponse:)]) {
276 NSURLRequest *originalRequest = wrapper(*WebKit::toImpl(request));
277 RetainPtr<NSURLRequest> substituteRequest = [loadDelegate webProcessPlugInBrowserContextController:pluginContextController frame:wrapper(*WebKit::toImpl(frame)) willSendRequestForResource:resourceIdentifier
278 request:originalRequest redirectResponse:WebKit::toImpl(redirectResponse)->resourceResponse().nsURLResponse()];
279
280 if (substituteRequest != originalRequest)
281 return substituteRequest ? WKURLRequestCreateWithNSURLRequest(substituteRequest.get()) : nullptr;
282 } else if ([loadDelegate respondsToSelector:@selector(webProcessPlugInBrowserContextController:frame:willSendRequest:redirectResponse:)]) {
283 NSURLRequest *originalRequest = wrapper(*WebKit::toImpl(request));
284 RetainPtr<NSURLRequest> substituteRequest = [loadDelegate webProcessPlugInBrowserContextController:pluginContextController frame:wrapper(*WebKit::toImpl(frame)) willSendRequest:originalRequest
285 redirectResponse:WebKit::toImpl(redirectResponse)->resourceResponse().nsURLResponse()];
286
287 if (substituteRequest != originalRequest)
288 return substituteRequest ? WKURLRequestCreateWithNSURLRequest(substituteRequest.get()) : nullptr;
289 }
290
291 WKRetain(request);
292 return request;
293}

481 static void encodeUserObject(NSObject <NSSecureCoding> *userObject, RefPtr<API::Object>& userData)
482 {
483 if (!userObject)
484 return;
485 auto archiver = adoptNS([[NSKeyedArchiver alloc] initRequiringSecureCoding:YES]);
486 @try {
487 [archiver encodeObject:userObject forKey:@"userObject"];
488 } @catch (NSException *exception) {
489 LOG_ERROR("Failed to encode user object: %@", exception);
490 return;
491 }
492 [archiver finishEncoding];
493 userData = API::Data::createWithoutCopying(archiver.get().encodedData);
494 }
