27#include "InjectedBundleCSSStyleDeclarationHandle.h"

29#include <WebCore/CSSStyleDeclaration.h>
30#include <wtf/HashMap.h>
31#include <wtf/NeverDestroyed.h>
32
33namespace WebKit {
34using namespace WebCore;
35
36typedef HashMap<CSSStyleDeclaration*, InjectedBundleCSSStyleDeclarationHandle*> DOMStyleDeclarationHandleCache;
37
38static DOMStyleDeclarationHandleCache& domStyleDeclarationHandleCache()
39{
40 static NeverDestroyed<DOMStyleDeclarationHandleCache> cache;
41 return cache;
42}
43
44RefPtr<InjectedBundleCSSStyleDeclarationHandle> InjectedBundleCSSStyleDeclarationHandle::getOrCreate(CSSStyleDeclaration* styleDeclaration)
45{
46 if (!styleDeclaration)
47 return nullptr;
48
49 DOMStyleDeclarationHandleCache::AddResult result = domStyleDeclarationHandleCache().add(styleDeclaration, nullptr);
50 if (!result.isNewEntry)
51 return result.iterator->value;
52
53 auto styleDeclarationHandle = adoptRef(*new InjectedBundleCSSStyleDeclarationHandle(*styleDeclaration));
54 result.iterator->value = styleDeclarationHandle.ptr();
55 return styleDeclarationHandle;
56}
57
58InjectedBundleCSSStyleDeclarationHandle::InjectedBundleCSSStyleDeclarationHandle(CSSStyleDeclaration& styleDeclaration)
59 : m_styleDeclaration(styleDeclaration)
60{
61}
62
63InjectedBundleCSSStyleDeclarationHandle::~InjectedBundleCSSStyleDeclarationHandle()
64{
65 domStyleDeclarationHandleCache().remove(m_styleDeclaration.ptr());
66}
