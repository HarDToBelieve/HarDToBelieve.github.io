59using DOMRangeHandleCache = HashMap<Range*, InjectedBundleRangeHandle*>;
60
61static DOMRangeHandleCache& domRangeHandleCache()
62{
63 static NeverDestroyed<DOMRangeHandleCache> cache;
64 return cache;
65}

67RefPtr<InjectedBundleRangeHandle> InjectedBundleRangeHandle::getOrCreate(JSContextRef context, JSObjectRef object)
68{
69 return getOrCreate(JSRange::toWrapped(toJS(context)->vm(), toJS(object)));
70}
71
72RefPtr<InjectedBundleRangeHandle> InjectedBundleRangeHandle::getOrCreate(Range* range)
73{
74 if (!range)
75 return nullptr;
76 auto result = domRangeHandleCache().add(range, nullptr);
77 if (!result.isNewEntry)
78 return result.iterator->value;
79 auto rangeHandle = adoptRef(*new InjectedBundleRangeHandle(*range));
80 result.iterator->value = rangeHandle.ptr();
81 return rangeHandle;
82}

104WebCore::IntRect InjectedBundleRangeHandle::boundingRectInWindowCoordinates() const
105{
106 auto range = makeSimpleRange(m_range);
107 auto frame = range.start.document().frame();
108 if (!frame)
109 return { };
110 auto view = frame->view();
111 if (!view)
112 return { };
113 return view->contentsToWindow(enclosingIntRect(unionRectIgnoringZeroRects(RenderObject::absoluteBorderAndTextRects(range))));
114}

116RefPtr<WebImage> InjectedBundleRangeHandle::renderedImage(SnapshotOptions options)
117{
118 auto range = makeSimpleRange(m_range);
119
120 auto document = makeRef(range.start.document());
121
122 auto frame = makeRefPtr(document->frame());
123 if (!frame)
124 return nullptr;
125
126 auto frameView = frame->view();
127 if (!frameView)
128 return nullptr;
129
130#if PLATFORM(MAC)
131 LocalDefaultSystemAppearance localAppearance(frameView->useDarkAppearance());
132#endif
133
134 VisibleSelection oldSelection = frame->selection().selection();
135 frame->selection().setSelection(range);
136
137 float scaleFactor = (options & SnapshotOptionsExcludeDeviceScaleFactor) ? 1 : frame->page()->deviceScaleFactor();
138 IntRect paintRect = enclosingIntRect(unionRectIgnoringZeroRects(RenderObject::absoluteBorderAndTextRects(range)));
139 IntSize backingStoreSize = paintRect.size();
140 backingStoreSize.scale(scaleFactor);
141
142 auto backingStore = ShareableBitmap::createShareable(backingStoreSize, { });
143 if (!backingStore)
144 return nullptr;
145
146 auto graphicsContext = backingStore->createGraphicsContext();
147 if (!graphicsContext)
148 return nullptr;
149
150 graphicsContext->scale(scaleFactor);
151
152 paintRect.move(frameView->frameRect().x(), frameView->frameRect().y());
153 paintRect.moveBy(-frameView->scrollPosition());
154
155 graphicsContext->translate(-paintRect.location());
156
157 OptionSet<PaintBehavior> oldPaintBehavior = frameView->paintBehavior();
158 OptionSet<PaintBehavior> paintBehavior = oldPaintBehavior;
159 paintBehavior.add({ PaintBehavior::SelectionOnly, PaintBehavior::FlattenCompositingLayers, PaintBehavior::Snapshotting });
160 if (options & SnapshotOptionsForceBlackText)
161 paintBehavior.add(PaintBehavior::ForceBlackText);
162 if (options & SnapshotOptionsForceWhiteText)
163 paintBehavior.add(PaintBehavior::ForceWhiteText);
164
165 frameView->setPaintBehavior(paintBehavior);
166 document->updateLayout();
167
168 frameView->paint(*graphicsContext, paintRect);
169 frameView->setPaintBehavior(oldPaintBehavior);
170
171 frame->selection().setSelection(oldSelection);
172
173 return WebImage::create(backingStore.releaseNonNull());
174}

176String InjectedBundleRangeHandle::text() const
177{
178 auto range = makeSimpleRange(m_range);
179 range.start.document().updateLayout();
180 return plainText(range);
181}
