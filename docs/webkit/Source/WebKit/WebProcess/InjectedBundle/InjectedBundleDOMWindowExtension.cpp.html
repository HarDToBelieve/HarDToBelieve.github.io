39namespace WebKit {
40using namespace WebCore;
41
42typedef HashMap<WebCore::DOMWindowExtension*, InjectedBundleDOMWindowExtension*> ExtensionMap;
43static ExtensionMap& allExtensions()
44{
45 static NeverDestroyed<ExtensionMap> map;
46 return map;
47}
48
49Ref<InjectedBundleDOMWindowExtension> InjectedBundleDOMWindowExtension::create(WebFrame* frame, InjectedBundleScriptWorld* world)
50{
51 return adoptRef(*new InjectedBundleDOMWindowExtension(frame, world));
52}
53
54InjectedBundleDOMWindowExtension* InjectedBundleDOMWindowExtension::get(DOMWindowExtension* extension)
55{
56 ASSERT(allExtensions().contains(extension));
57 return allExtensions().get(extension);
58}
59
60InjectedBundleDOMWindowExtension::InjectedBundleDOMWindowExtension(WebFrame* frame, InjectedBundleScriptWorld* world)
61 : m_coreExtension(DOMWindowExtension::create(frame->coreFrame() ? frame->coreFrame()->window() : nullptr, world->coreWorld()))
62{
63 allExtensions().add(m_coreExtension.get(), this);
64}
65
66InjectedBundleDOMWindowExtension::~InjectedBundleDOMWindowExtension()
67{
68 ASSERT(allExtensions().contains(m_coreExtension.get()));
69 allExtensions().remove(m_coreExtension.get());
70}
71
72WebFrame* InjectedBundleDOMWindowExtension::frame() const
73{
74 Frame* frame = m_coreExtension->frame();
75 if (!frame)
76 return nullptr;
77
78 return WebFrame::fromCoreFrame(*frame);
79}
80
81InjectedBundleScriptWorld* InjectedBundleDOMWindowExtension::world() const
82{
83 if (!m_world)
84 m_world = InjectedBundleScriptWorld::getOrCreate(m_coreExtension->world());
85
86 return m_world.get();
87}
