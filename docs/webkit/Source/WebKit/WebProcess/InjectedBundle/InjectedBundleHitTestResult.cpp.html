60WebFrame* InjectedBundleHitTestResult::frame() const
61{
62 Node* node = m_hitTestResult.innerNonSharedNode();
63 if (!node)
64 return nullptr;
65
66 Frame* frame = node->document().frame();
67 if (!frame)
68 return nullptr;
69
70 return WebFrame::fromCoreFrame(*frame);
71}

79 return WebFrame::fromCoreFrame(*frame);
80}

143IntRect InjectedBundleHitTestResult::imageRect() const
144{
145 IntRect imageRect = m_hitTestResult.imageRect();
146 if (imageRect.isEmpty())
147 return imageRect;
148
149 // The image rect in HitTestResult is in frame coordinates, but we need it in WKView
150 // coordinates since WebKit2 clients don't have enough context to do the conversion themselves.
151 WebFrame* webFrame = frame();
152 if (!webFrame)
153 return imageRect;
154
155 Frame* coreFrame = webFrame->coreFrame();
156 if (!coreFrame)
157 return imageRect;
158
159 FrameView* view = coreFrame->view();
160 if (!view)
161 return imageRect;
162
163 return view->contentsToRootView(imageRect);
164}

166RefPtr<WebImage> InjectedBundleHitTestResult::image() const
167{
168 Image* image = m_hitTestResult.image();
169 // For now, we only handle bitmap images.
170 if (!is<BitmapImage>(image))
171 return nullptr;
172
173 BitmapImage& bitmapImage = downcast<BitmapImage>(*image);
174 IntSize size(bitmapImage.size());
175 auto webImage = WebImage::create(size, static_cast<ImageOptions>(0));
176
177 // FIXME: need to handle EXIF rotation.
178 auto graphicsContext = webImage->bitmap().createGraphicsContext();
179 if (!graphicsContext)
180 return nullptr;
181
182 graphicsContext->drawImage(bitmapImage, {{ }, size});
183
184 return webImage;
185}
