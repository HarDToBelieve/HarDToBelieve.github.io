43class WebFrame;

45class InjectedBundleNavigationAction : public API::ObjectImpl<API::Object::Type::BundleNavigationAction> {
46public:
47 static Ref<InjectedBundleNavigationAction> create(WebFrame*, const WebCore::NavigationAction&, RefPtr<WebCore::FormState>&&);
48
49 static OptionSet<WebEvent::Modifier> modifiersForNavigationAction(const WebCore::NavigationAction&);
50 static WebMouseEvent::Button mouseButtonForNavigationAction(const WebCore::NavigationAction&);
51 static WebMouseEvent::SyntheticClickType syntheticClickTypeForNavigationAction(const WebCore::NavigationAction&);
52 static WebCore::FloatPoint clickLocationInRootViewCoordinatesForNavigationAction(const WebCore::NavigationAction&);
53
54 WebCore::NavigationType navigationType() const { return m_navigationType; }
55 OptionSet<WebEvent::Modifier> modifiers() const { return m_modifiers; }
56 WebMouseEvent::Button mouseButton() const { return m_mouseButton; }
57 InjectedBundleHitTestResult* hitTestResult() const { return m_hitTestResult.get(); }
58 InjectedBundleNodeHandle* formElement() const { return m_formElement.get(); }
59 WebMouseEvent::SyntheticClickType syntheticClickType() const { return m_syntheticClickType; }
60 WebCore::FloatPoint clickLocationInRootViewCoordinates() const { return m_clickLocationInRootViewCoordinates; }
61
62 bool shouldOpenExternalURLs() const { return m_shouldOpenExternalURLs; }
63 bool shouldTryAppLinks() const { return m_shouldTryAppLinks; }
64 AtomString downloadAttribute() const { return m_downloadAttribute; }
65
66private:
67 InjectedBundleNavigationAction(WebFrame*, const WebCore::NavigationAction&, RefPtr<WebCore::FormState>&&);
68
69 WebCore::NavigationType m_navigationType;
70 OptionSet<WebEvent::Modifier> m_modifiers;
71 WebMouseEvent::Button m_mouseButton;
72 WebMouseEvent::SyntheticClickType m_syntheticClickType { WebMouseEvent::NoTap };
73 WebCore::FloatPoint m_clickLocationInRootViewCoordinates;
74 RefPtr<InjectedBundleHitTestResult> m_hitTestResult;
75 RefPtr<InjectedBundleNodeHandle> m_formElement;
76 AtomString m_downloadAttribute;
77 bool m_shouldOpenExternalURLs;
78 bool m_shouldTryAppLinks;
79};
