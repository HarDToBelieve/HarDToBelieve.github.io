49bool InjectedBundlePageContextMenuClient::getCustomMenuFromDefaultItems(WebPage& page, const HitTestResult& hitTestResult, const Vector<ContextMenuItem>& proposedMenu, Vector<WebContextMenuItemData>& newMenu, RefPtr<API::Object>& userData)
50{
51 if (!m_client.getContextMenuFromDefaultMenu)
52 return false;
53
54 Vector<WebContextMenuItemData> defaultMenu = kitItems(proposedMenu);
55 Vector<RefPtr<API::Object>> defaultMenuItems;
56 defaultMenuItems.reserveInitialCapacity(defaultMenu.size());
57 for (const auto& item : defaultMenu)
58 defaultMenuItems.uncheckedAppend(WebContextMenuItem::create(item));
59
60 WKArrayRef newMenuWK = nullptr;
61 WKTypeRef userDataToPass = nullptr;
62 m_client.getContextMenuFromDefaultMenu(toAPI(&page), toAPI(&InjectedBundleHitTestResult::create(hitTestResult).get()), toAPI(API::Array::create(WTFMove(defaultMenuItems)).ptr()), &newMenuWK, &userDataToPass, m_client.base.clientInfo);
63 RefPtr<API::Array> array = adoptRef(toImpl(newMenuWK));
64 userData = adoptRef(toImpl(userDataToPass));
65
66 newMenu.clear();
67
68 if (!array || !array->size())
69 return true;
70
71 size_t size = array->size();
72 for (size_t i = 0; i < size; ++i) {
73 WebContextMenuItem* item = array->at<WebContextMenuItem>(i);
74 if (!item) {
75 LOG(ContextMenu, "New menu entry at index %i is not a WebContextMenuItem", (int)i);
76 continue;
77 }
78
79 newMenu.append(item->data());
80 }
81
82 return true;
83}

85void InjectedBundlePageContextMenuClient::prepareForImmediateAction(WebPage& page, const HitTestResult& hitTestResult, RefPtr<API::Object>& userData)
86{
87 if (!m_client.prepareForActionMenu)
88 return;
89
90 WKTypeRef userDataToPass = nullptr;
91 m_client.prepareForActionMenu(toAPI(&page), toAPI(&InjectedBundleHitTestResult::create(hitTestResult).get()), &userDataToPass, m_client.base.clientInfo);
92 userData = adoptRef(toImpl(userDataToPass));
93}
