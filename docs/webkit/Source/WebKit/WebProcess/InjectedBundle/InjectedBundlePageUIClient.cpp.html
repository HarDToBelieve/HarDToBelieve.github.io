96static API::InjectedBundle::PageUIClient::UIElementVisibility toUIElementVisibility(WKBundlePageUIElementVisibility visibility)
97{
98 switch (visibility) {
99 case WKBundlePageUIElementVisibilityUnknown:
100 return API::InjectedBundle::PageUIClient::UIElementVisibility::Unknown;
101 case WKBundlePageUIElementVisible:
102 return API::InjectedBundle::PageUIClient::UIElementVisibility::Visible;
103 case WKBundlePageUIElementHidden:
104 return API::InjectedBundle::PageUIClient::UIElementVisibility::Hidden;
105 }
106
107 ASSERT_NOT_REACHED();
108 return API::InjectedBundle::PageUIClient::UIElementVisibility::Unknown;
109}
