58RemoteWebInspectorUI::RemoteWebInspectorUI(WebPage& page)
59 : m_page(page)
60 , m_frontendAPIDispatcher(InspectorFrontendAPIDispatcher::create(*page.corePage()))
61{
62 WebInspectorUI::enableFrontendFeatures(page);
63}

67void RemoteWebInspectorUI::initialize(DebuggableInfoData&& debuggableInfo, const String& backendCommandsURL)
68{
69#if ENABLE(INSPECTOR_EXTENSIONS)
70 m_extensionController = makeUnique<WebInspectorUIExtensionController>(*this);
71#endif
72
73 m_debuggableInfo = WTFMove(debuggableInfo);
74 m_backendCommandsURL = backendCommandsURL;
75
76 m_page.corePage()->inspectorController().setInspectorFrontendClient(this);
77
78 m_frontendAPIDispatcher->reset();
79 m_frontendAPIDispatcher->dispatchCommandWithResultAsync("setDockingUnavailable"_s, { JSON::Value::create(true) });
80}

107void RemoteWebInspectorUI::windowObjectCleared()
108{
109 if (m_frontendHost)
110 m_frontendHost->disconnectClient();
111
112 m_frontendHost = InspectorFrontendHost::create(this, m_page.corePage());
113 m_frontendHost->addSelfToGlobalObjectInWorld(mainThreadNormalWorld());
114}
115
116void RemoteWebInspectorUI::frontendLoaded()
117{
118 m_frontendAPIDispatcher->frontendLoaded();
119
120 m_frontendAPIDispatcher->dispatchCommandWithResultAsync("setIsVisible"_s, { JSON::Value::create(true) });
121
122 WebProcess::singleton().parentProcessConnection()->send(Messages::RemoteWebInspectorUIProxy::FrontendLoaded(), m_page.identifier());
123
124 bringToFront();
125}

164bool RemoteWebInspectorUI::supportsDockSide(DockSide dockSide)
165{
166 switch (dockSide) {
167 case DockSide::Undocked:
168 return true;
169
170 case DockSide::Right:
171 case DockSide::Left:
172 case DockSide::Bottom:
173 return false;
174 }
175
176 ASSERT_NOT_REACHED();
177 return false;
178}

185void RemoteWebInspectorUI::closeWindow()
186{
187 m_page.corePage()->inspectorController().setInspectorFrontendClient(nullptr);
188
189#if ENABLE(INSPECTOR_EXTENSIONS)
190 m_extensionController = nullptr;
191#endif
192
193 WebProcess::singleton().parentProcessConnection()->send(Messages::RemoteWebInspectorUIProxy::FrontendDidClose(), m_page.identifier());
194}

267void RemoteWebInspectorUI::setDiagnosticLoggingAvailable(bool available)
268{
269 // Inspector's diagnostic logging client should never be used unless the page setting is also enabled.
270 ASSERT(!available || supportsDiagnosticLogging());
271 m_diagnosticLoggingAvailable = available;
272
273 m_frontendAPIDispatcher->dispatchCommandWithResultAsync("setDiagnosticLoggingAvailable"_s, { JSON::Value::create(m_diagnosticLoggingAvailable) });
274}
