76void WebInspectorUI::establishConnection(WebPageProxyIdentifier inspectedPageIdentifier, const DebuggableInfoData& debuggableInfo, bool underTest, unsigned inspectionLevel)
77{
78 m_inspectedPageIdentifier = inspectedPageIdentifier;
79 m_debuggableInfo = debuggableInfo;
80 m_underTest = underTest;
81 m_inspectionLevel = inspectionLevel;
82
83#if ENABLE(INSPECTOR_EXTENSIONS)
84 if (!m_extensionController)
85 m_extensionController = makeUnique<WebInspectorUIExtensionController>(*this);
86#endif
87
88 m_frontendAPIDispatcher->reset();
89 m_frontendController = &m_page.corePage()->inspectorController();
90 m_frontendController->setInspectorFrontendClient(this);
91
92 updateConnection();
93}

95void WebInspectorUI::updateConnection()
96{
97 if (m_backendConnection) {
98 m_backendConnection->invalidate();
99 m_backendConnection = nullptr;
100 }
101
102#if USE(UNIX_DOMAIN_SOCKETS)
103 IPC::Connection::SocketPair socketPair = IPC::Connection::createPlatformConnection();
104 IPC::Connection::Identifier connectionIdentifier(socketPair.server);
105 UNUSED_PARAM(connectionIdentifier);
106 IPC::Attachment connectionClientPort(socketPair.client);
107#elif OS(DARWIN)
108 mach_port_t listeningPort = MACH_PORT_NULL;
109 if (mach_port_allocate(mach_task_self(), MACH_PORT_RIGHT_RECEIVE, &listeningPort) != KERN_SUCCESS)
110 CRASH();
111
112 if (mach_port_insert_right(mach_task_self(), listeningPort, listeningPort, MACH_MSG_TYPE_MAKE_SEND) != KERN_SUCCESS)
113 CRASH();
114
115 IPC::Connection::Identifier connectionIdentifier(listeningPort);
116 UNUSED_PARAM(connectionIdentifier);
117 IPC::Attachment connectionClientPort(listeningPort, MACH_MSG_TYPE_MOVE_SEND);
118#elif PLATFORM(WIN)
119 IPC::Connection::Identifier connectionIdentifier, connClient;
120 IPC::Connection::createServerAndClientIdentifiers(connectionIdentifier, connClient);
121 IPC::Attachment connectionClientPort(connClient);
122#else
123 notImplemented();
124 return;
125#endif
126
127#if USE(UNIX_DOMAIN_SOCKETS) || OS(DARWIN) || PLATFORM(WIN)
128 m_backendConnection = IPC::Connection::createServerConnection(connectionIdentifier, *this);
129 m_backendConnection->open();
130#endif
131
132 WebProcess::singleton().parentProcessConnection()->send(Messages::WebInspectorUIProxy::SetFrontendConnection(connectionClientPort), m_inspectedPageIdentifier);
133}

135void WebInspectorUI::windowObjectCleared()
136{
137 if (m_frontendHost)
138 m_frontendHost->disconnectClient();
139
140 m_frontendHost = InspectorFrontendHost::create(this, m_page.corePage());
141 m_frontendHost->addSelfToGlobalObjectInWorld(mainThreadNormalWorld());
142}

144void WebInspectorUI::frontendLoaded()
145{
146 m_frontendAPIDispatcher->frontendLoaded();
147
148 // Tell the new frontend about the current dock state. If the window object
149 // cleared due to a reload, the dock state won't be resent from UIProcess.
150 setDockingUnavailable(m_dockingUnavailable);
151 setDockSide(m_dockSide);
152 setIsVisible(m_isVisible);
153
154 WebProcess::singleton().parentProcessConnection()->send(Messages::WebInspectorUIProxy::FrontendLoaded(), m_inspectedPageIdentifier);
155
156 bringToFront();
157}

176void WebInspectorUI::closeWindow()
177{
178 WebProcess::singleton().parentProcessConnection()->send(Messages::WebInspectorUIProxy::DidClose(), m_inspectedPageIdentifier);
179
180 if (m_backendConnection) {
181 m_backendConnection->invalidate();
182 m_backendConnection = nullptr;
183 }
184
185 if (m_frontendController) {
186 m_frontendController->setInspectorFrontendClient(nullptr);
187 m_frontendController = nullptr;
188 }
189
190 if (m_frontendHost)
191 m_frontendHost->disconnectClient();
192
193 m_inspectedPageIdentifier = { };
194 m_underTest = false;
195
196#if ENABLE(INSPECTOR_EXTENSIONS)
197 m_extensionController = nullptr;
198#endif
199}

221bool WebInspectorUI::supportsDockSide(DockSide dockSide)
222{
223 switch (dockSide) {
224 case DockSide::Undocked:
225 case DockSide::Right:
226 case DockSide::Left:
227 case DockSide::Bottom:
228 return true;
229 }
230
231 ASSERT_NOT_REACHED();
232 return false;
233}
234
235void WebInspectorUI::requestSetDockSide(DockSide dockSide)
236{
237 auto& webProcess = WebProcess::singleton();
238 switch (dockSide) {
239 case DockSide::Undocked:
240 webProcess.parentProcessConnection()->send(Messages::WebInspectorUIProxy::Detach(), m_inspectedPageIdentifier);
241 break;
242 case DockSide::Right:
243 webProcess.parentProcessConnection()->send(Messages::WebInspectorUIProxy::AttachRight(), m_inspectedPageIdentifier);
244 break;
245 case DockSide::Left:
246 webProcess.parentProcessConnection()->send(Messages::WebInspectorUIProxy::AttachLeft(), m_inspectedPageIdentifier);
247 break;
248 case DockSide::Bottom:
249 webProcess.parentProcessConnection()->send(Messages::WebInspectorUIProxy::AttachBottom(), m_inspectedPageIdentifier);
250 break;
251 }
252}

254void WebInspectorUI::setDockSide(DockSide dockSide)
255{
256 ASCIILiteral dockSideString { ASCIILiteral::null() };
257
258 switch (dockSide) {
259 case DockSide::Undocked:
260 dockSideString = "undocked"_s;
261 break;
262
263 case DockSide::Right:
264 dockSideString = "right"_s;
265 break;
266
267 case DockSide::Left:
268 dockSideString = "left"_s;
269 break;
270
271 case DockSide::Bottom:
272 dockSideString = "bottom"_s;
273 break;
274 }
275
276 m_dockSide = dockSide;
277
278 m_frontendAPIDispatcher->dispatchCommandWithResultAsync("setDockSide"_s, { JSON::Value::create(String(dockSideString)) });
279}
