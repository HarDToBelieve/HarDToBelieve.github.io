62std::optional<Inspector::ExtensionError> WebInspectorUIExtensionController::parseExtensionErrorFromEvaluationResult(InspectorFrontendAPIDispatcher::EvaluationResult result)
63{
64 if (!result) {
65 switch (result.error()) {
66 case WebCore::InspectorFrontendAPIDispatcher::EvaluationError::ContextDestroyed:
67 return Inspector::ExtensionError::ContextDestroyed;
68 case WebCore::InspectorFrontendAPIDispatcher::EvaluationError::ExecutionSuspended:
69 return Inspector::ExtensionError::InternalError;
70 case WebCore::InspectorFrontendAPIDispatcher::EvaluationError::InternalError:
71 return Inspector::ExtensionError::InternalError;
72 }
73 }
74
75 ASSERT(m_frontendClient);
76 auto globalObject = m_frontendClient->frontendAPIDispatcher().frontendGlobalObject();
77 if (!globalObject)
78 return Inspector::ExtensionError::ContextDestroyed;
79
80 auto valueOrException = result.value();
81 if (!valueOrException.has_value()) {
82 LOG(Inspector, "Encountered exception while evaluating upon the frontend: %s", valueOrException.error().message.utf8().data());
83 return Inspector::ExtensionError::InternalError;
84 }
85
86 // If the evaluation result is a string, the frontend returned an error string.
87 // If there was a JS exception, log it and return error code InternalError.
88 // Anything else (falsy values, objects, arrays, DOM, etc.) is interpreted as success.
89 JSC::JSValue scriptValue = valueOrException.value();
90 if (scriptValue.isString()) {
91 auto resultString = scriptValue.toWTFString(globalObject);
92 if (resultString == "ContextDestroyed"_s)
93 return Inspector::ExtensionError::ContextDestroyed;
94 if (resultString == "InternalError"_s)
95 return Inspector::ExtensionError::InternalError;
96 if (resultString == "InvalidRequest"_s)
97 return Inspector::ExtensionError::InvalidRequest;
98 if (resultString == "RegistrationFailed"_s)
99 return Inspector::ExtensionError::RegistrationFailed;
100 if (resultString == "NotImplemented"_s)
101 return Inspector::ExtensionError::NotImplemented;
102
103 ASSERT_NOT_REACHED();
104 return Inspector::ExtensionError::InternalError;
105 }
106
107 return std::nullopt;
108}

112void WebInspectorUIExtensionController::registerExtension(const Inspector::ExtensionID& extensionID, const String& displayName, CompletionHandler<void(Expected<bool, Inspector::ExtensionError>)>&& completionHandler)
113{
114 if (!m_frontendClient) {
115 completionHandler(makeUnexpected(Inspector::ExtensionError::InvalidRequest));
116 return;
117 }
118
119 Vector<Ref<JSON::Value>> arguments {
120 JSON::Value::create(extensionID),
121 JSON::Value::create(displayName),
122 };
123 m_frontendClient->frontendAPIDispatcher().dispatchCommandWithResultAsync("registerExtension"_s, WTFMove(arguments), [weakThis = makeWeakPtr(this), completionHandler = WTFMove(completionHandler)](WebCore::InspectorFrontendAPIDispatcher::EvaluationResult&& result) mutable {
124 if (!weakThis || !result) {
125 completionHandler(makeUnexpected(Inspector::ExtensionError::ContextDestroyed));
126 return;
127 }
128
129 if (auto parsedError = weakThis->parseExtensionErrorFromEvaluationResult(result)) {
130 completionHandler(makeUnexpected(parsedError.value()));
131 return;
132 }
133
134 completionHandler(true);
135 });
136}

161JSC::JSObject* WebInspectorUIExtensionController::unwrapEvaluationResultAsObject(InspectorFrontendAPIDispatcher::EvaluationResult result)
162{
163 if (!result)
164 return nullptr;
165
166 auto valueOrException = result.value();
167 if (!valueOrException.has_value())
168 return nullptr;
169
170 return valueOrException.value().getObject();
171}

173void WebInspectorUIExtensionController::createTabForExtension(const Inspector::ExtensionID& extensionID, const String& tabName, const URL& tabIconURL, const URL& sourceURL, WTF::CompletionHandler<void(Expected<Inspector::ExtensionTabID, Inspector::ExtensionError>)>&& completionHandler)
174{
175 if (!m_frontendClient) {
176 completionHandler(makeUnexpected(Inspector::ExtensionError::InvalidRequest));
177 return;
178 }
179
180 Vector<Ref<JSON::Value>> arguments {
181 JSON::Value::create(extensionID),
182 JSON::Value::create(tabName),
183 JSON::Value::create(tabIconURL.string()),
184 JSON::Value::create(sourceURL.string()),
185 };
186 m_frontendClient->frontendAPIDispatcher().dispatchCommandWithResultAsync("createTabForExtension"_s, WTFMove(arguments), [weakThis = makeWeakPtr(this), completionHandler = WTFMove(completionHandler)](InspectorFrontendAPIDispatcher::EvaluationResult&& result) mutable {
187 if (!weakThis || !result) {
188 completionHandler(makeUnexpected(Inspector::ExtensionError::ContextDestroyed));
189 return;
190 }
191
192 if (auto parsedError = weakThis->parseExtensionErrorFromEvaluationResult(result)) {
193 completionHandler(makeUnexpected(parsedError.value()));
194 return;
195 }
196
197 // Expected result is either an ErrorString or {extensionTabID: <string>}.
198 auto objectResult = weakThis->unwrapEvaluationResultAsObject(result);
199 if (!objectResult) {
200 LOG(Inspector, "Unexpected non-object value returned from InspectorFrontendAPI.createTabForExtension().");
201 completionHandler(makeUnexpected(Inspector::ExtensionError::InternalError));
202 return;
203 }
204
205 auto* frontendGlobalObject = weakThis->m_frontendClient->frontendAPIDispatcher().frontendGlobalObject();
206 JSC::JSValue foundProperty = objectResult->get(frontendGlobalObject, JSC::Identifier::fromString(frontendGlobalObject->vm(), "extensionTabID"_s));
207 if (!foundProperty || !foundProperty.isString()) {
208 completionHandler(makeUnexpected(Inspector::ExtensionError::InternalError));
209 return;
210 }
211
212 completionHandler({ foundProperty.toWTFString(frontendGlobalObject) });
213 });
214}

216void WebInspectorUIExtensionController::evaluateScriptForExtension(const Inspector::ExtensionID& extensionID, const String& scriptSource, const std::optional<URL>& frameURL, const std::optional<URL>& contextSecurityOrigin, const std::optional<bool>& useContentScriptContext, CompletionHandler<void(const IPC::DataReference&, const std::optional<WebCore::ExceptionDetails>&, const std::optional<Inspector::ExtensionError>&)>&& completionHandler)
217{
218 if (!m_frontendClient) {
219 completionHandler({ }, std::nullopt, Inspector::ExtensionError::InvalidRequest);
220 return;
221 }
222
223 Ref<JSON::Object> optionalArguments = JSON::Object::create();
224 if (frameURL)
225 optionalArguments->setString("frameURL"_s, frameURL->string());
226 if (contextSecurityOrigin)
227 optionalArguments->setString("contextSecurityOrigin"_s, contextSecurityOrigin->string());
228 if (useContentScriptContext)
229 optionalArguments->setBoolean("useContentScriptContext"_s, *useContentScriptContext);
230
231 Vector<Ref<JSON::Value>> arguments {
232 JSON::Value::create(extensionID),
233 JSON::Value::create(scriptSource),
234 WTFMove(optionalArguments)
235 };
236
237 m_frontendClient->frontendAPIDispatcher().dispatchCommandWithResultAsync("evaluateScriptForExtension"_s, WTFMove(arguments), [weakThis = makeWeakPtr(this), completionHandler = WTFMove(completionHandler)](InspectorFrontendAPIDispatcher::EvaluationResult&& result) mutable {
238 if (!weakThis) {
239 completionHandler({ }, std::nullopt, Inspector::ExtensionError::ContextDestroyed);
240 return;
241 }
242
243 auto* frontendGlobalObject = weakThis->m_frontendClient->frontendAPIDispatcher().frontendGlobalObject();
244 if (!frontendGlobalObject) {
245 completionHandler({ }, std::nullopt, Inspector::ExtensionError::ContextDestroyed);
246 return;
247 }
248
249 if (auto parsedError = weakThis->parseExtensionErrorFromEvaluationResult(result)) {
250 auto exceptionDetails = result.value().error();
251 LOG(Inspector, "Internal error encountered while evaluating upon the frontend: at %s:%d:%d: %s", exceptionDetails.sourceURL.utf8().data(), exceptionDetails.lineNumber, exceptionDetails.columnNumber, exceptionDetails.message.utf8().data());
252 completionHandler({ }, std::nullopt, parsedError);
253 return;
254 }
255
256 // Expected result is either an ErrorString or {result: <any>}.
257 auto objectResult = weakThis->unwrapEvaluationResultAsObject(result);
258 if (!objectResult) {
259 LOG(Inspector, "Unexpected non-object value returned from InspectorFrontendAPI.createTabForExtension().");
260 completionHandler({ }, std::nullopt, Inspector::ExtensionError::InternalError);
261 return;
262 }
263 ASSERT(result.value());
264
265 JSC::JSValue errorPayload = objectResult->get(frontendGlobalObject, JSC::Identifier::fromString(frontendGlobalObject->vm(), "error"_s));
266 if (!errorPayload.isUndefined()) {
267 if (!errorPayload.isString()) {
268 completionHandler({ }, std::nullopt, Inspector::ExtensionError::InternalError);
269 return;
270 }
271
272 completionHandler({ }, ExceptionDetails { errorPayload.toWTFString(frontendGlobalObject) }, std::nullopt);
273 return;
274 }
275
276 JSC::JSValue resultPayload = objectResult->get(frontendGlobalObject, JSC::Identifier::fromString(frontendGlobalObject->vm(), "result"_s));
277 RefPtr<SerializedScriptValue> serializedResultValue = SerializedScriptValue::create(*frontendGlobalObject, resultPayload);
278 if (!serializedResultValue) {
279 completionHandler({ }, std::nullopt, Inspector::ExtensionError::InternalError);
280 return;
281 }
282
283 completionHandler(serializedResultValue->data(), std::nullopt, std::nullopt);
284 });
285}

287void WebInspectorUIExtensionController::reloadForExtension(const Inspector::ExtensionID& extensionID, const std::optional<bool>& ignoreCache, const std::optional<String>& userAgent, const std::optional<String>& injectedScript, CompletionHandler<void(const std::optional<Inspector::ExtensionError>&)>&& completionHandler)
288{
289 if (!m_frontendClient) {
290 completionHandler(Inspector::ExtensionError::InvalidRequest);
291 return;
292 }
293
294 Ref<JSON::Object> optionalArguments = JSON::Object::create();
295 if (ignoreCache)
296 optionalArguments->setBoolean("ignoreCache"_s, *ignoreCache);
297 if (userAgent)
298 optionalArguments->setString("userAgent"_s, *userAgent);
299 if (injectedScript)
300 optionalArguments->setString("injectedScript"_s, *injectedScript);
301
302 Vector<Ref<JSON::Value>> arguments {
303 JSON::Value::create(extensionID),
304 WTFMove(optionalArguments)
305 };
306
307 m_frontendClient->frontendAPIDispatcher().dispatchCommandWithResultAsync("reloadForExtension"_s, WTFMove(arguments), [weakThis = makeWeakPtr(this), completionHandler = WTFMove(completionHandler)](InspectorFrontendAPIDispatcher::EvaluationResult&& result) mutable {
308 if (!weakThis) {
309 completionHandler(Inspector::ExtensionError::ContextDestroyed);
310 return;
311 }
312
313 auto* frontendGlobalObject = weakThis->m_frontendClient->frontendAPIDispatcher().frontendGlobalObject();
314 if (!frontendGlobalObject) {
315 completionHandler(Inspector::ExtensionError::ContextDestroyed);
316 return;
317 }
318
319 if (auto parsedError = weakThis->parseExtensionErrorFromEvaluationResult(result)) {
320 LOG(Inspector, "Internal error encountered while evaluating upon the frontend: %s", Inspector::extensionErrorToString(*parsedError).utf8().data());
321 completionHandler(parsedError);
322 return;
323 }
324
325 completionHandler(std::nullopt);
326 });
327}
