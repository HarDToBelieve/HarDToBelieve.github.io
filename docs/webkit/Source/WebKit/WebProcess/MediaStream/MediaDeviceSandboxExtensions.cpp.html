33namespace WebKit {
34
35MediaDeviceSandboxExtensions::MediaDeviceSandboxExtensions(Vector<String> ids, SandboxExtension::HandleArray&& handles)
36 : m_ids(ids)
37 , m_handles(WTFMove(handles))
38{
39 RELEASE_ASSERT_WITH_SECURITY_IMPLICATION(m_ids.size() == m_handles.size());
40}
41
42void MediaDeviceSandboxExtensions::encode(IPC::Encoder& encoder) const
43{
44 encoder << m_ids;
45 m_handles.encode(encoder);
46}
47
48bool MediaDeviceSandboxExtensions::decode(IPC::Decoder& decoder, MediaDeviceSandboxExtensions& result)
49{
50 if (!decoder.decode(result.m_ids))
51 return false;
52
53 std::optional<SandboxExtension::HandleArray> handles;
54 decoder >> handles;
55 if (!handles)
56 return false;
57 result.m_handles = WTFMove(*handles);
58
59 return true;
60}
61
62std::pair<String, RefPtr<SandboxExtension>> MediaDeviceSandboxExtensions::operator[](size_t i)
63{
64 RELEASE_ASSERT_WITH_SECURITY_IMPLICATION(m_ids.size() == m_handles.size());
65 RELEASE_ASSERT_WITH_SECURITY_IMPLICATION(i < m_ids.size());
66 return { m_ids[i], SandboxExtension::create(WTFMove(m_handles[i])) };
67}
68
69size_t MediaDeviceSandboxExtensions::size() const
70{
71 return m_ids.size();
72}
