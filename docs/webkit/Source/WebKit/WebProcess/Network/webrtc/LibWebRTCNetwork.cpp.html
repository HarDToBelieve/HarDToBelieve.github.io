43void LibWebRTCNetwork::setAsActive()
44{
45 ASSERT(!m_isActive);
46 m_isActive = true;
47#if USE(LIBWEBRTC)
48 if (m_connection)
49 setSocketFactoryConnection();
50#endif
51}
52
53void LibWebRTCNetwork::networkProcessCrashed()
54{
55 setConnection(nullptr);
56
57#if USE(LIBWEBRTC)
58 m_webNetworkMonitor.networkProcessCrashed();
59#endif
60}
61
62void LibWebRTCNetwork::setConnection(RefPtr<IPC::Connection>&& connection)
63{
64#if USE(LIBWEBRTC)
65 if (m_connection)
66 m_connection->removeThreadMessageReceiver(Messages::LibWebRTCNetwork::messageReceiverName());
67#endif
68 m_connection = WTFMove(connection);
69#if USE(LIBWEBRTC)
70 if (m_isActive)
71 setSocketFactoryConnection();
72 if (m_connection)
73 m_connection->addThreadMessageReceiver(Messages::LibWebRTCNetwork::messageReceiverName(), this);
74#endif
75}
76
77#if USE(LIBWEBRTC)
78void LibWebRTCNetwork::setSocketFactoryConnection()
79{
80 if (!m_connection) {
81 WebCore::LibWebRTCProvider::callOnWebRTCNetworkThread([this]() mutable {
82 m_socketFactory.setConnection(nullptr);
83 });
84 return;
85 }
86 m_connection->sendWithAsyncReply(Messages::NetworkConnectionToWebProcess::CreateRTCProvider(), [this, connection = m_connection]() mutable {
87 if (!connection->isValid())
88 return;
89
90 WebCore::LibWebRTCProvider::callOnWebRTCNetworkThread([this, connection = WTFMove(connection)]() mutable {
91 m_socketFactory.setConnection(WTFMove(connection));
92 });
93 }, 0);
94}
95#endif

97void LibWebRTCNetwork::dispatchToThread(Function<void()>&& callback)
98{
99 if (!m_isActive) {
100 RELEASE_LOG_ERROR(WebRTC, "Received WebRTCSocket message while libWebRTCNetwork is not active");
101 return;
102 }
103
104#if USE(LIBWEBRTC)
105 WebCore::LibWebRTCProvider::callOnWebRTCNetworkThread(WTFMove(callback));
106#else
107 UNUSED_PARAM(callback);
108#endif
109}

147void LibWebRTCNetwork::signalNewConnection(WebCore::LibWebRTCSocketIdentifier identifier, WebCore::LibWebRTCSocketIdentifier newSocketIdentifier, const RTCNetwork::SocketAddress& remoteAddress)
