47 void networkProcessCrashed();
48
49 bool isActive() const { return m_isActive; }

51#if USE(LIBWEBRTC)
52 void didReceiveMessage(IPC::Connection&, IPC::Decoder&);
53
54 WebRTCMonitor& monitor() { return m_webNetworkMonitor; }
55 LibWebRTCSocketFactory& socketFactory() { return m_socketFactory; }
56
57 void disableNonLocalhostConnections() { socketFactory().disableNonLocalhostConnections(); }
58
59 WebRTCResolver resolver(LibWebRTCResolverIdentifier identifier) { return WebRTCResolver(socketFactory(), identifier); }
60#endif
61
62#if ENABLE(WEB_RTC)

65
66 void setAsActive();

68private:
69#if USE(LIBWEBRTC)
70 void setSocketFactoryConnection();
71
72 void signalReadPacket(WebCore::LibWebRTCSocketIdentifier, const IPC::DataReference&, const RTCNetwork::IPAddress&, uint16_t port, int64_t);
73 void signalSentPacket(WebCore::LibWebRTCSocketIdentifier, int, int64_t);

75 void signalConnect(WebCore::LibWebRTCSocketIdentifier);
76 void signalClose(WebCore::LibWebRTCSocketIdentifier, int);
77 void signalNewConnection(WebCore::LibWebRTCSocketIdentifier socketIdentifier, WebCore::LibWebRTCSocketIdentifier newSocketIdentifier, const WebKit::RTCNetwork::SocketAddress&);
78#endif
79
80 // IPC::Connection::ThreadMessageReceiver
81 void dispatchToThread(Function<void()>&&) final;

87#if ENABLE(WEB_RTC)
88 WebMDNSRegister m_mdnsRegister;

90 bool m_isActive { false };
91 RefPtr<IPC::Connection> m_connection;
92};
