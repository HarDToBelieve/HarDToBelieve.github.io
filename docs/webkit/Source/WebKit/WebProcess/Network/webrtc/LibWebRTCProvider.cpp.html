57rtc::scoped_refptr<webrtc::PeerConnectionInterface> LibWebRTCProvider::createPeerConnection(webrtc::PeerConnectionObserver& observer, rtc::PacketSocketFactory* socketFactory, webrtc::PeerConnectionInterface::RTCConfiguration&& configuration)
58{
59#if ENABLE(GPU_PROCESS) && PLATFORM(COCOA) && !PLATFORM(MACCATALYST)
60 if (!m_didInitializeCallback) {
61 // We initialize only once since callbacks are used in background threads.
62 auto* page = m_webPage.corePage();
63 LibWebRTCCodecs::setCallbacks(page && page->settings().webRTCPlatformCodecsInGPUProcessEnabled());
64 m_didInitializeCallback = true;
65 }
66#endif
67
68 auto& networkMonitor = WebProcess::singleton().libWebRTCNetwork().monitor();
69 networkMonitor.setEnumeratingAllNetworkInterfacesEnabled(isEnumeratingAllNetworkInterfacesEnabled());
70 return WebCore::LibWebRTCProvider::createPeerConnection(observer, networkMonitor, *socketFactory, WTFMove(configuration), makeUnique<AsyncResolverFactory>());
71}

88class RTCSocketFactory final : public LibWebRTCProvider::SuspendableSocketFactory {
89 WTF_MAKE_FAST_ALLOCATED;
90public:
91 RTCSocketFactory(String&& userAgent, bool isFirstParty, RegistrableDomain&&);
92
93private:
94 // SuspendableSocketFactory
95 rtc::AsyncPacketSocket* CreateUdpSocket(const rtc::SocketAddress&, uint16_t minPort, uint16_t maxPort) final;
96 rtc::AsyncPacketSocket* CreateServerTcpSocket(const rtc::SocketAddress&, uint16_t minPort, uint16_t maxPort, int options) final;
97 rtc::AsyncPacketSocket* CreateClientTcpSocket(const rtc::SocketAddress& localAddress, const rtc::SocketAddress& remoteAddress, const rtc::ProxyInfo&, const std::string&, const rtc::PacketSocketTcpOptions&) final;
98 rtc::AsyncResolverInterface* CreateAsyncResolver() final;
99 void suspend() final;
100 void resume() final;
101 void disableRelay() final { m_isRelayDisabled = true; }
102
103private:
104 String m_userAgent;
105 bool m_isFirstParty { false };
106 bool m_isRelayDisabled { false };
107 RegistrableDomain m_domain;
108};

137void RTCSocketFactory::suspend()
138{
139 WebCore::LibWebRTCProvider::callOnWebRTCNetworkThread([socketGroup = this] {
140 WebProcess::singleton().libWebRTCNetwork().socketFactory().forSocketInGroup(socketGroup, [](auto& socket) {
141 socket.suspend();
142 });
143 });
144}
145
146void RTCSocketFactory::resume()
147{
148 WebCore::LibWebRTCProvider::callOnWebRTCNetworkThread([socketGroup = this] {
149 WebProcess::singleton().libWebRTCNetwork().socketFactory().forSocketInGroup(socketGroup, [](auto& socket) {
150 socket.resume();
151 });
152 });
153}
