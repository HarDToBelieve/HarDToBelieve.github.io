43LibWebRTCSocket::LibWebRTCSocket(LibWebRTCSocketFactory& factory, const void* socketGroup, Type type, const rtc::SocketAddress& localAddress, const rtc::SocketAddress& remoteAddress)
44 : m_factory(factory)
45 , m_identifier(WebCore::LibWebRTCSocketIdentifier::generate())
46 , m_type(type)
47 , m_localAddress(localAddress)
48 , m_remoteAddress(remoteAddress)
49 , m_socketGroup(socketGroup)
50{
51 m_factory.addSocket(*this);
52}

54LibWebRTCSocket::~LibWebRTCSocket()
55{
56 Close();
57 m_factory.removeSocket(*this);
58}

70void LibWebRTCSocket::signalAddressReady(const rtc::SocketAddress& address)
71{
72 m_localAddress = address;
73 m_state = (m_type == Type::ClientTCP) ? STATE_CONNECTED : STATE_BOUND;
74 SignalAddressReady(this, m_localAddress);
75}

77void LibWebRTCSocket::signalReadPacket(const uint8_t* data, size_t size, rtc::SocketAddress&& address, int64_t timestamp)
78{
79 if (m_isSuspended)
80 return;
81
82 m_remoteAddress = WTFMove(address);
83 SignalReadPacket(this, reinterpret_cast<const char*>(data), size, m_remoteAddress, timestamp);
84}
85
86void LibWebRTCSocket::signalSentPacket(int rtcPacketID, int64_t sendTimeMs)
87{
88 if (m_beingSentPacketSizes.isEmpty())
89 return;
90
91 m_availableSendingBytes += m_beingSentPacketSizes.takeFirst();
92 SignalSentPacket(this, rtc::SentPacket(rtcPacketID, sendTimeMs));
93 if (m_shouldSignalReadyToSend) {
94 m_shouldSignalReadyToSend = false;
95 SignalReadyToSend(this);
96 }
97}

117bool LibWebRTCSocket::willSend(size_t size)
118{
119 if (size > m_availableSendingBytes) {
120 m_shouldSignalReadyToSend = true;
121 setError(EWOULDBLOCK);
122 return false;
123 }
124 m_availableSendingBytes -= size;
125 m_beingSentPacketSizes.append(size);
126 return true;
127}
128
129int LibWebRTCSocket::SendTo(const void *value, size_t size, const rtc::SocketAddress& address, const rtc::PacketOptions& options)
130{
131 auto* connection = m_factory.connection();
132 if (!connection || !willSend(size))
133 return -1;
134
135 if (m_isSuspended)
136 return size;
137
138 IPC::DataReference data(static_cast<const uint8_t*>(value), size);
139 connection->send(Messages::NetworkRTCProvider::SendToSocket { m_identifier, data, RTCNetwork::SocketAddress { address }, RTCPacketOptions { options } }, 0);
140
141 return size;
142}
143
144int LibWebRTCSocket::Close()
145{
146 auto* connection = m_factory.connection();
147 if (!connection || m_state == STATE_CLOSED)
148 return 0;
149
150 m_state = STATE_CLOSED;
151
152 connection->send(Messages::NetworkRTCProvider::CloseSocket { m_identifier }, 0);
153
154 return 0;
155}

157int LibWebRTCSocket::GetOption(rtc::Socket::Option option, int* value)
158{

161 *value = *storedValue;

164 return -1;

167int LibWebRTCSocket::SetOption(rtc::Socket::Option option, int value)
168{
169 ASSERT(option < MAX_SOCKET_OPTION);
170
171 m_options[option] = value;
172
173 if (auto* connection = m_factory.connection())
174 connection->send(Messages::NetworkRTCProvider::SetSocketOption { m_identifier, option, value }, 0);
175
176 return 0;
177}

179void LibWebRTCSocket::resume()
180{
181 m_isSuspended = false;
182
183 // On resume, we notify libwebrtc that TCP sockets are errored.
184 // We notify libwebrtc that all pending UDP packets have been sent even though we actually dropped them.
185 if (m_type != Type::UDP) {
186 signalClose(-1);
187 return;
188 }
189
190 auto currentTime = rtc::TimeMillis();
191 while (!m_beingSentPacketSizes.isEmpty())
192 signalSentPacket(-1, currentTime);
193}
194
195void LibWebRTCSocket::suspend()
196{
197 m_isSuspended = true;
198
199 // On suspend, we close TCP sockets as we cannot make sure packets are delivered reliably.
200 if (m_type == Type::UDP)
201 return;
202
203 if (auto* connection = m_factory.connection())
204 connection->send(Messages::NetworkRTCProvider::CloseSocket { m_identifier }, 0);
205}
