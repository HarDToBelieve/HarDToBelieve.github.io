52void RTCDataChannelRemoteManager::setConnection(IPC::Connection* connection)
53{
54 if (m_connection == connection)
55 return;
56
57 if (m_connection)
58 m_connection->removeWorkQueueMessageReceiver(Messages::RTCDataChannelRemoteManager::messageReceiverName());
59
60 m_connection = connection;
61
62 if (m_connection)
63 m_connection->addWorkQueueMessageReceiver(Messages::RTCDataChannelRemoteManager::messageReceiverName(), m_queue, this);
64}

66bool RTCDataChannelRemoteManager::connectToRemoteSource(WebCore::RTCDataChannelIdentifier localIdentifier, WebCore::RTCDataChannelIdentifier remoteIdentifier)
67{
68 ASSERT(WebCore::Process::identifier() == localIdentifier.processIdentifier);
69 if (WebCore::Process::identifier() != localIdentifier.processIdentifier)
70 return false;
71
72 auto handler = WebCore::RTCDataChannel::handlerFromIdentifier(localIdentifier.channelIdentifier);
73 if (!handler)
74 return false;
75
76 auto iterator = m_sources.add(remoteIdentifier.channelIdentifier, WebCore::RTCDataChannelRemoteSource::create(remoteIdentifier, makeUniqueRefFromNonNullUniquePtr(WTFMove(handler)), remoteSourceConnection()));
77 return iterator.isNewEntry;
78}

94void RTCDataChannelRemoteManager::postTaskToHandler(WebCore::RTCDataChannelIdentifier handlerIdentifier, Function<void(WebCore::RTCDataChannelRemoteHandler&)>&& function)
95{
96 ASSERT(WebCore::Process::identifier() == handlerIdentifier.processIdentifier);
97 if (WebCore::Process::identifier() != handlerIdentifier.processIdentifier)
98 return;
99
100 auto iterator = m_handlers.find(handlerIdentifier.channelIdentifier);
101 if (iterator == m_handlers.end())
102 return;
103 auto& remoteHandler = iterator->value;
104
105 WebCore::ScriptExecutionContext::postTaskTo(remoteHandler.contextIdentifier, [handler = remoteHandler.handler, function = WTFMove(function)](auto&) mutable {
106 if (handler)
107 function(*handler);
108 });
109}

111WebCore::RTCDataChannelRemoteSource* RTCDataChannelRemoteManager::sourceFromIdentifier(WebCore::RTCDataChannelIdentifier sourceIdentifier)
112{
113 ASSERT(WebCore::Process::identifier() == sourceIdentifier.processIdentifier);
114 if (WebCore::Process::identifier() != sourceIdentifier.processIdentifier)
115 return nullptr;
116
117 return m_sources.get(sourceIdentifier.channelIdentifier);
118}

120void RTCDataChannelRemoteManager::sendData(WebCore::RTCDataChannelIdentifier sourceIdentifier, bool isRaw, const IPC::DataReference& data)
121{
122 if (auto* source = sourceFromIdentifier(sourceIdentifier)) {
123 if (isRaw)
124 source->sendRawData(data.data(), data.size());
125 else
126 source->sendStringData(CString(data.data(), data.size()));
127 }
128}

130void RTCDataChannelRemoteManager::close(WebCore::RTCDataChannelIdentifier sourceIdentifier)
131{
132 if (auto* source = sourceFromIdentifier(sourceIdentifier))
133 source->close();
134}

143void RTCDataChannelRemoteManager::receiveData(WebCore::RTCDataChannelIdentifier handlerIdentifier, bool isRaw, const IPC::DataReference& data)
144{
145 Vector<uint8_t> buffer;
146 String text;
147 if (isRaw)
148 buffer = data.vector();
149 else
150 text = String::fromUTF8(data.data(), data.size());
151
152 postTaskToHandler(handlerIdentifier, [isRaw, text = WTFMove(text).isolatedCopy(), buffer = WTFMove(buffer)](auto& handler) mutable {
153 if (isRaw)
154 handler.didReceiveRawData(buffer.data(), buffer.size());
155 else
156 handler.didReceiveStringData(WTFMove(text));
157 });
158}

185void RTCDataChannelRemoteManager::RemoteHandlerConnection::connectToSource(WebCore::RTCDataChannelRemoteHandler& handler, WebCore::ScriptExecutionContextIdentifier contextIdentifier, WebCore::RTCDataChannelIdentifier localIdentifier, WebCore::RTCDataChannelIdentifier remoteIdentifier)
186{
187 m_queue->dispatch([handler = makeWeakPtr(handler), contextIdentifier, localIdentifier]() mutable {
188 RTCDataChannelRemoteManager::sharedManager().m_handlers.add(localIdentifier.channelIdentifier, RemoteHandler { WTFMove(handler), contextIdentifier });
189 });
190 m_connection->sendWithAsyncReply(Messages::NetworkConnectionToWebProcess::ConnectToRTCDataChannelRemoteSource { localIdentifier, remoteIdentifier }, [localIdentifier](auto&& result) {
191 RTCDataChannelRemoteManager::sharedManager().postTaskToHandler(localIdentifier, [result](auto& handler) {
192 if (!result || !*result) {
193 handler.didDetectError();
194 return;
195 }
196 handler.readyToSend();
197 });
198 }, 0);
199}
