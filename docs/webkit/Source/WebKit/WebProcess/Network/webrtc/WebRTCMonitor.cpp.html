44void WebRTCMonitor::sendOnMainThread(Function<void(IPC::Connection&)>&& callback)
45{
46 callOnMainRunLoop([callback = WTFMove(callback)]() {
47 callback(WebProcess::singleton().ensureNetworkProcessConnection().connection());
48 });
49}

56void WebRTCMonitor::StartUpdating()
57{
58 WEBRTC_RELEASE_LOG("StartUpdating");
59 if (m_receivedNetworkList) {
60 WebCore::LibWebRTCProvider::callOnWebRTCNetworkThread([this] {
61 SignalNetworksChanged();
62 });
63 }
64
65 sendOnMainThread([this](auto& connection) {
66 WEBRTC_RELEASE_LOG("StartUpdating - Asking network process to start updating");
67 connection.send(Messages::NetworkRTCMonitor::StartUpdatingIfNeeded(m_enableEnumeratingAllNetworkInterfaces), 0);
68 });
69 ++m_clientCount;
70}
71
72void WebRTCMonitor::StopUpdating()
73{
74 WEBRTC_RELEASE_LOG("StopUpdating");
75 ASSERT(m_clientCount);
76 if (--m_clientCount)
77 return;
78
79 sendOnMainThread([this](auto& connection) {
80 WEBRTC_RELEASE_LOG("StopUpdating - Asking network process to stop updating");
81 connection.send(Messages::NetworkRTCMonitor::StopUpdating(), 0);
82 });
83}

85void WebRTCMonitor::networksChanged(const Vector<RTCNetwork>& networks, const RTCNetwork::IPAddress& ipv4, const RTCNetwork::IPAddress& ipv6)
86{
87 WEBRTC_RELEASE_LOG("networksChanged");
88 // No need to protect 'this' as it has the lifetime of LibWebRTC which has the lifetime of the web process.
89 WebCore::LibWebRTCProvider::callOnWebRTCNetworkThread([this, networks, ipv4, ipv6] {
90 WEBRTC_RELEASE_LOG("networksChanged - Signaling");
91 std::vector<rtc::Network*> networkList(networks.size());
92 for (size_t index = 0; index < networks.size(); ++index)
93 networkList[index] = new rtc::Network(networks[index].value());
94
95 bool forceSignaling = !m_receivedNetworkList;
96 m_receivedNetworkList = true;
97
98 bool hasChanged;
99 set_default_local_addresses(ipv4.value, ipv6.value);
100 MergeNetworkList(networkList, &hasChanged);
101 if (hasChanged || forceSignaling)
102 SignalNetworksChanged();
103
104 });
105}
106
107void WebRTCMonitor::networkProcessCrashed()
108{
109 m_receivedNetworkList = false;
110 if (!WebCore::LibWebRTCProvider::hasWebRTCThreads())
111 return;
112
113 // In case we have clients waiting for networksChanged, we call SignalNetworksChanged to make sure they do not wait for nothing.
114 WebCore::LibWebRTCProvider::callOnWebRTCNetworkThread([this] {
115 SignalNetworksChanged();
116 });
117}
