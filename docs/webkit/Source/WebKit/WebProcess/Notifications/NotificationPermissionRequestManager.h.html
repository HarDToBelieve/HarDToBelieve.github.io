45/// FIXME: Need to keep a queue of pending notifications which permission is still being requested.
46class NotificationPermissionRequestManager : public RefCounted<NotificationPermissionRequestManager> {
47public:
48 static Ref<NotificationPermissionRequestManager> create(WebPage*);
49 ~NotificationPermissionRequestManager();
50
51 using Permission = WebCore::NotificationClient::Permission;
52 using PermissionHandler = WebCore::NotificationClient::PermissionHandler;
53
54#if ENABLE(NOTIFICATIONS)
55 void startRequest(const WebCore::SecurityOriginData&, PermissionHandler&&);
56#endif
57
58 Permission permissionLevel(const WebCore::SecurityOriginData&);
59
60 // For testing purposes only.
61 void setPermissionLevelForTesting(const String& originString, bool allowed);
62 void removeAllPermissionsForTesting();
63
64private:
65 NotificationPermissionRequestManager(WebPage*);
66
67#if ENABLE(NOTIFICATIONS)
68 using PermissionHandlers = Vector<PermissionHandler>;
69 static void callPermissionHandlersWith(PermissionHandlers&, Permission);
70
71 HashMap<WebCore::SecurityOriginData, PermissionHandlers> m_requestsPerOrigin;
72 WebPage* m_page;
73#endif
74};
75
76inline bool isRequestIDValid(uint64_t id)
77{
78 // This check makes sure that the ID is not equal to values needed by
79 // HashMap for bucketing.
80 return id && id != static_cast<uint64_t>(-1);
81}
