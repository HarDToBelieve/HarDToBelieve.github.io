104NotificationClient::Permission WebNotificationManager::policyForOrigin(const String& originString) const
105{
106#if ENABLE(NOTIFICATIONS)
107 if (!originString)
108 return NotificationClient::Permission::Default;
109
110 auto it = m_permissionsMap.find(originString);
111 if (it != m_permissionsMap.end())
112 return it->value ? NotificationClient::Permission::Granted : NotificationClient::Permission::Denied;
113#else
114 UNUSED_PARAM(originString);
115#endif
116
117 return NotificationClient::Permission::Default;
118}

139bool WebNotificationManager::show(Notification* notification, WebPage* page)
140{
141#if ENABLE(NOTIFICATIONS)
142 if (!notification || !page->corePage()->settings().notificationsEnabled())
143 return false;
144
145 uint64_t notificationID = generateNotificationID();
146 m_notificationMap.set(notification, notificationID);
147 m_notificationIDMap.set(notificationID, notification);
148
149 auto it = m_notificationContextMap.add(notification->scriptExecutionContext(), Vector<uint64_t>()).iterator;
150 it->value.append(notificationID);
151
152 m_process.parentProcessConnection()->send(Messages::WebPageProxy::ShowNotification(notification->title(), notification->body(), notification->icon().string(), notification->tag(), notification->lang(), notification->dir(), notification->scriptExecutionContext()->securityOrigin()->toString(), notificationID), page->identifier());
153 return true;
154#else
155 UNUSED_PARAM(notification);
156 UNUSED_PARAM(page);
157 return false;
158#endif
159}

178void WebNotificationManager::clearNotifications(WebCore::ScriptExecutionContext* context, WebPage* page)
179{
180#if ENABLE(NOTIFICATIONS)
181 NotificationContextMap::iterator it = m_notificationContextMap.find(context);
182 if (it == m_notificationContextMap.end())
183 return;
184
185 Vector<uint64_t>& notificationIDs = it->value;
186 m_process.parentProcessConnection()->send(Messages::WebPageProxy::ClearNotifications(notificationIDs), page->identifier());
187 size_t count = notificationIDs.size();
188 for (size_t i = 0; i < count; ++i) {
189 RefPtr<Notification> notification = m_notificationIDMap.take(notificationIDs[i]);
190 if (!notification)
191 continue;
192 notification->finalize();
193 m_notificationMap.remove(notification);
194 }
195
196 m_notificationContextMap.remove(it);
197#else
198 UNUSED_PARAM(context);
199 UNUSED_PARAM(page);
200#endif
201}

203void WebNotificationManager::didDestroyNotification(Notification* notification, WebPage* page)
204{
205#if ENABLE(NOTIFICATIONS)
206 auto protectedNotification = makeRefPtr(notification);
207
208 uint64_t notificationID = m_notificationMap.take(notification);
209 if (!notificationID)
210 return;
211
212 m_notificationIDMap.remove(notificationID);
213 removeNotificationFromContextMap(notificationID, notification);
214 m_process.parentProcessConnection()->send(Messages::WebPageProxy::DidDestroyNotification(notificationID), page->identifier());
215#else
216 UNUSED_PARAM(notification);
217 UNUSED_PARAM(page);
218#endif
219}

255void WebNotificationManager::didCloseNotifications(const Vector<uint64_t>& notificationIDs)
256{
257#if ENABLE(NOTIFICATIONS)
258 size_t count = notificationIDs.size();
259 for (size_t i = 0; i < count; ++i) {
260 uint64_t notificationID = notificationIDs[i];
261 if (!isNotificationIDValid(notificationID))
262 continue;
263
264 RefPtr<Notification> notification = m_notificationIDMap.take(notificationID);
265 if (!notification)
266 continue;
267
268 m_notificationMap.remove(notification);
269 removeNotificationFromContextMap(notificationID, notification.get());
270
271 notification->dispatchCloseEvent();
272 }
273#else
274 UNUSED_PARAM(notificationIDs);
275#endif
276}

278#if ENABLE(NOTIFICATIONS)
279void WebNotificationManager::removeNotificationFromContextMap(uint64_t notificationID, Notification* notification)
280{
281 // This is a helper function for managing the hash maps.
282 NotificationContextMap::iterator it = m_notificationContextMap.find(notification->scriptExecutionContext());
283 ASSERT(it != m_notificationContextMap.end());
284 size_t index = it->value.find(notificationID);
285 ASSERT(index != notFound);
286 it->value.remove(index);
287 if (it->value.isEmpty())
288 m_notificationContextMap.remove(it);
289}
290#endif
