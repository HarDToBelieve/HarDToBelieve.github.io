26#ifndef JSNPMethod_h
27#define JSNPMethod_h
28
29#if ENABLE(NETSCAPE_PLUGIN_API)
30
31#include <JavaScriptCore/FunctionPrototype.h>
32#include <JavaScriptCore/InternalFunction.h>
33#include <JavaScriptCore/JSGlobalObject.h>
34
35typedef void* NPIdentifier;
36
37namespace WebKit {
38
39// A JSObject that wraps an NPMethod.
40class JSNPMethod final : public JSC::InternalFunction {
41public:
42 using Base = JSC::InternalFunction;
43
44 template<typename CellType, JSC::SubspaceAccess>
45 static JSC::IsoSubspace* subspaceFor(JSC::VM& vm)
46 {
47 return subspaceForImpl(vm);
48 }
49
50 static JSNPMethod* create(JSC::JSGlobalObject* globalObject, const String& name, NPIdentifier npIdent)
51 {
52 JSC::VM& vm = globalObject->vm();
53 JSC::Structure* structure = createStructure(globalObject->vm(), globalObject, globalObject->functionPrototype());
54 JSNPMethod* method = new (JSC::allocateCell<JSNPMethod>(vm.heap)) JSNPMethod(globalObject, structure, npIdent);
55 method->finishCreation(vm, name);
56 return method;
57 }
58
59 DECLARE_INFO;
60
61 NPIdentifier npIdentifier() const { return m_npIdentifier; }
62
63private:
64 static JSC::IsoSubspace* subspaceForImpl(JSC::VM&);
65
66 JSNPMethod(JSC::JSGlobalObject*, JSC::Structure*, NPIdentifier);
67 void finishCreation(JSC::VM&, const String& name);
68
69 static JSC::Structure* createStructure(JSC::VM& vm, JSC::JSGlobalObject* globalObject, JSC::JSValue prototype)
70 {
71 return JSC::Structure::create(vm, globalObject, prototype, JSC::TypeInfo(JSC::InternalFunctionType, StructureFlags), info());
72 }
73
74 NPIdentifier m_npIdentifier;
75};
76
77
78} // namespace WebKit
79
80#endif // ENABLE(NETSCAPE_PLUGIN_API)
81
82#endif // JSNPMethod_h
