70NPObject* NPRuntimeObjectMap::getOrCreateNPObject(VM& vm, JSObject* jsObject)
71{
72 // If this is a JSNPObject, we can just get its underlying NPObject.
73 if (jsObject->classInfo(vm) == JSNPObject::info()) {
74 JSNPObject* jsNPObject = jsCast<JSNPObject*>(jsObject);
75 NPObject* npObject = jsNPObject->npObject();
76
77 retainNPObject(npObject);
78 return npObject;
79 }
80
81 // First, check if we already know about this object.
82 if (NPJSObject* npJSObject = m_npJSObjects.get(jsObject)) {
83 retainNPObject(npJSObject);
84 return npJSObject;
85 }
86
87 NPJSObject* npJSObject = NPJSObject::create(vm, this, jsObject);
88 m_npJSObjects.set(jsObject, npJSObject);
89
90 return npJSObject;
91}

93void NPRuntimeObjectMap::npJSObjectDestroyed(NPJSObject* npJSObject)
94{
95 // Remove the object from the map.
96 ASSERT(m_npJSObjects.contains(npJSObject->jsObject()));
97 m_npJSObjects.remove(npJSObject->jsObject());
98}

100JSObject* NPRuntimeObjectMap::getOrCreateJSObject(JSGlobalObject* globalObject, NPObject* npObject)
101{
102 // If this is an NPJSObject, we can just get the JSObject that it's wrapping.
103 if (NPJSObject::isNPJSObject(npObject))
104 return NPJSObject::toNPJSObject(npObject)->jsObject();
105
106 if (JSNPObject* jsNPObject = m_jsNPObjects.get(npObject))
107 return jsNPObject;
108
109 JSNPObject* jsNPObject = JSNPObject::create(globalObject, this, npObject);
110 weakAdd(m_jsNPObjects, npObject, JSC::Weak<JSNPObject>(jsNPObject, this, npObject));
111 return jsNPObject;
112}

114JSValue NPRuntimeObjectMap::convertNPVariantToJSValue(JSC::JSGlobalObject* globalObject, const NPVariant& variant)
115{
116 switch (variant.type) {
117 case NPVariantType_Void:
118 return jsUndefined();
119
120 case NPVariantType_Null:
121 return jsNull();
122
123 case NPVariantType_Bool:
124 return jsBoolean(variant.value.boolValue);
125
126 case NPVariantType_Int32:
127 return jsNumber(variant.value.intValue);
128
129 case NPVariantType_Double:
130 return jsNumber(variant.value.doubleValue);
131
132 case NPVariantType_String:
133 return jsString(globalObject->vm(), String::fromUTF8WithLatin1Fallback(variant.value.stringValue.UTF8Characters,
134 variant.value.stringValue.UTF8Length));
135 case NPVariantType_Object:
136 return getOrCreateJSObject(globalObject, variant.value.objectValue);
137 }
138
139 ASSERT_NOT_REACHED();
140 return jsUndefined();
141}

143void NPRuntimeObjectMap::convertJSValueToNPVariant(JSGlobalObject* lexicalGlobalObject, JSValue value, NPVariant& variant)
144{
145 VM& vm = lexicalGlobalObject->vm();
146 JSLockHolder lock(lexicalGlobalObject);
147
148 VOID_TO_NPVARIANT(variant);
149
150 if (value.isNull()) {
151 NULL_TO_NPVARIANT(variant);
152 return;
153 }
154
155 if (value.isUndefined()) {
156 VOID_TO_NPVARIANT(variant);
157 return;
158 }
159
160 if (value.isBoolean()) {
161 BOOLEAN_TO_NPVARIANT(value.toBoolean(lexicalGlobalObject), variant);
162 return;
163 }
164
165 if (value.isNumber()) {
166 DOUBLE_TO_NPVARIANT(value.toNumber(lexicalGlobalObject), variant);
167 return;
168 }
169
170 if (value.isString()) {
171 NPString npString = createNPString(value.toString(lexicalGlobalObject)->value(lexicalGlobalObject).utf8());
172 STRINGN_TO_NPVARIANT(npString.UTF8Characters, npString.UTF8Length, variant);
173 return;
174 }
175
176 if (value.isObject()) {
177 NPObject* npObject = getOrCreateNPObject(vm, asObject(value));
178 OBJECT_TO_NPVARIANT(npObject, variant);
179 return;
180 }
181
182 ASSERT_NOT_REACHED();
183}

185bool NPRuntimeObjectMap::evaluate(NPObject* npObject, const String& scriptString, NPVariant* result)
186{
187 Strong<JSGlobalObject> globalObject(this->globalObject()->vm(), this->globalObject());
188 if (!globalObject)
189 return false;
190
191 JSLockHolder lock(globalObject.get());
192 JSValue thisValue = getOrCreateJSObject(globalObject.get(), npObject);
193
194 JSValue resultValue = JSC::evaluate(globalObject.get(), makeSource(scriptString, { }), thisValue);
195
196 convertJSValueToNPVariant(globalObject.get(), resultValue, *result);
197 return true;
198}

200void NPRuntimeObjectMap::invalidate()
201{
202 // Deallocate all the object wrappers so we won't leak any JavaScript objects.
203 for (auto& npJSObject : copyToVector(m_npJSObjects.values()))
204 deallocateNPObject(npJSObject);
205
206 // We shouldn't have any NPJSObjects left now.
207 ASSERT(m_npJSObjects.isEmpty());
208
209 Vector<NPObject*> objects;
210
211 for (HashMap<NPObject*, JSC::Weak<JSNPObject>>::iterator ptr = m_jsNPObjects.begin(), end = m_jsNPObjects.end(); ptr != end; ++ptr) {
212 JSNPObject* jsNPObject = ptr->value.get();
213 if (!jsNPObject) // Skip zombies.
214 continue;
215 objects.append(jsNPObject->leakNPObject());
216 }
217
218 m_jsNPObjects.clear();
219
220 for (size_t i = 0; i < objects.size(); ++i)
221 releaseNPObject(objects[i]);
222
223 // Deal with any objects that were scheduled for delayed destruction
224 if (m_npObjectsToFinalize.isEmpty())
225 return;
226 ASSERT(m_finalizationTimer.isActive());
227 m_finalizationTimer.stop();
228 invalidateQueuedObjects();
229}

231JSGlobalObject* NPRuntimeObjectMap::globalObject() const
232{
233 Frame* frame = m_pluginView->frame();
234 if (!frame)
235 return 0;
236
237 return frame->script().globalObject(pluginWorld());

251void NPRuntimeObjectMap::moveGlobalExceptionToExecState(JSGlobalObject* lexicalGlobalObject)
252{
253 VM& vm = lexicalGlobalObject->vm();
254 auto scope = DECLARE_THROW_SCOPE(vm);
255
256 if (globalExceptionString().isNull())
257 return;
258
259 {
260 JSLockHolder lock(vm);
261 throwException(lexicalGlobalObject, scope, createError(lexicalGlobalObject, globalExceptionString()));
262 }
263
264 globalExceptionString() = String();
265}

267void NPRuntimeObjectMap::invalidateQueuedObjects()
268{
269 ASSERT(m_npObjectsToFinalize.size());
270 // We deliberately re-request m_npObjectsToFinalize.size() as custom dealloc
271 // functions may execute JS and so get more objects added to the dealloc queue
272 for (size_t i = 0; i < m_npObjectsToFinalize.size(); ++i)
273 deallocateNPObject(m_npObjectsToFinalize[i]);
274 m_npObjectsToFinalize.clear();
275}
276
277void NPRuntimeObjectMap::addToInvalidationQueue(NPObject* npObject)
278{
279 if (trySafeReleaseNPObject(npObject))
280 return;
281 if (m_npObjectsToFinalize.isEmpty())
282 m_finalizationTimer.startOneShot(0_s);
283 ASSERT(m_finalizationTimer.isActive());
284 m_npObjectsToFinalize.append(npObject);
285}

287void NPRuntimeObjectMap::finalize(JSC::Handle<JSC::Unknown> handle, void* context)
288{
289 JSNPObject* object = static_cast<JSNPObject*>(handle.get().asCell());
290 weakRemove(m_jsNPObjects, static_cast<NPObject*>(context), object);
291 addToInvalidationQueue(object->leakNPObject());
292}
