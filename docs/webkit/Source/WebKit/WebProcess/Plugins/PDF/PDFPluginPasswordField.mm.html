43Ref<PDFPluginPasswordField> PDFPluginPasswordField::create(PDFLayerController *pdfLayerController, PDFPlugin* plugin)
44{
45 return adoptRef(*new PDFPluginPasswordField(pdfLayerController, plugin));
46}
47
48PDFPluginPasswordField::~PDFPluginPasswordField()
49{
50 element()->removeEventListener(eventNames().keyupEvent, *eventListener(), false);
51}
52
53Ref<Element> PDFPluginPasswordField::createAnnotationElement()
54{
55 auto element = PDFPluginTextAnnotation::createAnnotationElement();
56 element->setAttribute(typeAttr, "password");
57 element->addEventListener(eventNames().keyupEvent, *eventListener(), false);
58 return element;
59}

61void PDFPluginPasswordField::updateGeometry()

63}
64
65bool PDFPluginPasswordField::handleEvent(WebCore::Event& event)
66{
67 if (event.isKeyboardEvent() && event.type() == eventNames().keyupEvent) {
68 auto& keyboardEvent = downcast<KeyboardEvent>(event);
69
70 if (keyboardEvent.keyIdentifier() == "Enter") {
71 plugin()->attemptToUnlockPDF(value());
72 event.preventDefault();
73 return true;
74 }
75 }
76
77 return false;
78}
