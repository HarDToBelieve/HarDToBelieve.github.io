74class PluginController;

76enum PluginType {
77 PluginProxyType,
78 NetscapePluginType,
79 PDFPluginType,
80};
81
82enum class LayerHostingMode : uint8_t;
83
84class Plugin : public ThreadSafeRefCounted<Plugin> {
85public:
86 struct Parameters {
87 URL url;
88 Vector<String> names;
89 Vector<String> values;
90 String mimeType;
91 bool isFullFramePlugin;
92 bool shouldUseManualLoader;
93#if PLATFORM(COCOA)
94 LayerHostingMode layerHostingMode;
95#endif
96
97 void encode(IPC::Encoder&) const;
98 static WARN_UNUSED_RETURN bool decode(IPC::Decoder&, Parameters&);
99 };
100
101 bool initialize(PluginController&, const Parameters&);
102
103 void destroyPlugin();
104
105 bool isBeingDestroyed() const { return m_isBeingDestroyed; }
106
107 PluginController* controller();
108 const PluginController* controller() const;
109
110 virtual ~Plugin();
111
112 PluginType type() const { return m_type; }
113
114 bool isPluginProxy() const { return m_type == PluginProxyType; }
115 bool isNetscapePlugin() const { return m_type == NetscapePluginType; }
116 bool isPDFPlugin() const { return m_type == PDFPluginType; }
117
118private:
119 // Initializes the plug-in. If the plug-in fails to initialize this should return false.
120 // This is only called by the other initialize overload so it can be made private.
121 virtual bool initialize(const Parameters&) = 0;
122
123 // Destroys the plug-in.
124 virtual void destroy() = 0;
125
126public:
127
128 // Tells the plug-in to paint itself into the given graphics context. The passed-in context and
129 // dirty rect are in window coordinates. The context is saved/restored by the caller.
130 virtual void paint(WebCore::GraphicsContext&, const WebCore::IntRect& dirtyRect) = 0;
131
132 // Invalidate native tintable controls. The passed-in context is in window coordinates.
133 virtual void updateControlTints(WebCore::GraphicsContext&);
134
135 // Returns whether the plug-in supports snapshotting or not.
136 virtual bool supportsSnapshotting() const = 0;
137
138 // Tells the plug-in to draw itself into a bitmap, and return that.
139 virtual RefPtr<ShareableBitmap> snapshot() = 0;
140
141#if PLATFORM(COCOA)
142 // If a plug-in is using the Core Animation drawing model, this returns its plug-in layer.
143 virtual PlatformLayer* pluginLayer() = 0;
144#endif
145
146 // Returns whether the plug-in is transparent or not.
147 virtual bool isTransparent() = 0;
148
149 // Returns whether we should send wheel events to this plug-in.
150 virtual bool wantsWheelEvents() = 0;
151
152 // Tells the plug-in that its geometry has changed. The clip rect is in plug-in coordinates, and the affine transform can be used
153 // to convert from root view coordinates to plug-in coordinates.
154 virtual void geometryDidChange(const WebCore::IntSize& pluginSize, const WebCore::IntRect& clipRect, const WebCore::AffineTransform& pluginToRootViewTransform) = 0;
155
156 // Tells the plug-in that it has been explicitly hidden or shown. (Note that this is not called when the plug-in becomes obscured from view on screen.)
157 virtual void visibilityDidChange(bool isVisible) = 0;
158
159 // Tells the plug-in that a frame load request that the plug-in made by calling PluginController::loadURL has finished.
160 virtual void frameDidFinishLoading(uint64_t requestID) = 0;
161
162 // Tells the plug-in that a frame load request that the plug-in made by calling PluginController::loadURL has failed.
163 virtual void frameDidFail(uint64_t requestID, bool wasCancelled) = 0;
164
165 // Tells the plug-in that a request to evaluate JavaScript (using PluginController::loadURL) has been fulfilled and passes
166 // back the result. If evaluating the script failed, result will be null.
167 virtual void didEvaluateJavaScript(uint64_t requestID, const String& result) = 0;
168
169 // Tells the plug-in that a stream may send an HTTP request.
170 virtual void streamWillSendRequest(uint64_t streamID, const URL& requestURL, const URL& responseURL, int responseStatusCode) = 0;
171
172 // Tells the plug-in that a stream has received its HTTP response.
173 virtual void streamDidReceiveResponse(uint64_t streamID, const URL& responseURL, uint32_t streamLength,
174 uint32_t lastModifiedTime, const String& mimeType, const String& headers, const String& suggestedFileName) = 0;
175
176 // Tells the plug-in that a stream did receive data.
177 virtual void streamDidReceiveData(uint64_t streamID, const uint8_t* bytes, int length) = 0;
178
179 // Tells the plug-in that a stream has finished loading.
180 virtual void streamDidFinishLoading(uint64_t streamID) = 0;
181
182 // Tells the plug-in that a stream has failed to load, either because of network errors or because the load was cancelled.
183 virtual void streamDidFail(uint64_t streamID, bool wasCancelled) = 0;
184
185 // Tells the plug-in that the manual stream has received its HTTP response.
186 virtual void manualStreamDidReceiveResponse(const URL& responseURL, uint32_t streamLength,
187 uint32_t lastModifiedTime, const String& mimeType, const String& headers, const String& suggestedFileName) = 0;
188
189 // Tells the plug-in that the manual stream did receive data.
190 virtual void manualStreamDidReceiveData(const uint8_t* bytes, int length) = 0;
191
192 // Tells the plug-in that a stream has finished loading.
193 virtual void manualStreamDidFinishLoading() = 0;
194
195 // Tells the plug-in that a stream has failed to load, either because of network errors or because the load was cancelled.
196 virtual void manualStreamDidFail(bool wasCancelled) = 0;
197
198 // Tells the plug-in to handle the passed in mouse event. The plug-in should return true if it processed the event.
199 virtual bool handleMouseEvent(const WebMouseEvent&) = 0;
200
201 // Tells the plug-in to handle the passed in wheel event. The plug-in should return true if it processed the event.
202 virtual bool handleWheelEvent(const WebWheelEvent&) = 0;
203
204 // Tells the plug-in to handle the passed in mouse over event. The plug-in should return true if it processed the event.
205 virtual bool handleMouseEnterEvent(const WebMouseEvent&) = 0;
206
207 // Tells the plug-in to handle the passed in mouse leave event. The plug-in should return true if it processed the event.
208 virtual bool handleMouseLeaveEvent(const WebMouseEvent&) = 0;
209
210 // Tells the plug-in to handle the passed in context menu event. The plug-in should return true if it processed the event.
211 virtual bool handleContextMenuEvent(const WebMouseEvent&) = 0;
212
213 // Tells the plug-in to handle the passed in keyboard event. The plug-in should return true if it processed the event.
214 virtual bool handleKeyboardEvent(const WebKeyboardEvent&) = 0;
215
216 // Tells the plug-in to handle the passed in editing command. The plug-in should return true if it executed the command.
217 virtual bool handleEditingCommand(const String& commandName, const String& argument) = 0;
218
219 // Ask the plug-in whether it will be able to handle the given editing command.
220 virtual bool isEditingCommandEnabled(const String&) = 0;
221
222 // Ask the plug-in whether it should be allowed to execute JavaScript or navigate to JavaScript URLs.
223 virtual bool shouldAllowScripting() = 0;
224
225 // Ask the plug-in whether it wants URLs and files dragged onto it to cause navigation.
226 virtual bool shouldAllowNavigationFromDrags() = 0;
227
228 // Ask the plug-in whether it wants to override full-page zoom.
229 virtual bool handlesPageScaleFactor() const = 0;
230
231 // Tells the plug-in about focus changes.
232 virtual void setFocus(bool) = 0;
233
234 // Get the NPObject that corresponds to the plug-in's scriptable object. Returns a retained object.
235 virtual NPObject* pluginScriptableNPObject() = 0;
236
237 // Tells the plug-in about window focus changes.
238 virtual void windowFocusChanged(bool) = 0;
239
240 // Tells the plug-in about window visibility changes.
241 virtual void windowVisibilityChanged(bool) = 0;
242
243#if PLATFORM(COCOA)
244 // Tells the plug-in about window and plug-in frame changes.
245 virtual void windowAndViewFramesChanged(const WebCore::IntRect& windowFrameInScreenCoordinates, const WebCore::IntRect& viewFrameInWindowCoordinates) = 0;
246
247 // Get the per complex text input identifier.
248 virtual uint64_t pluginComplexTextInputIdentifier() const = 0;
249
250 // Send the complex text input to the plug-in.
251 virtual void sendComplexTextInput(const String& textInput) = 0;
252
253 // Tells the plug-in about changes to the layer hosting mode.
254 virtual void setLayerHostingMode(LayerHostingMode) = 0;
255#endif
256
257 // Tells the plug-in about scale factor changes.
258 virtual void contentsScaleFactorChanged(float) = 0;
259
260 // Called when the storage blocking policy for this plug-in changes.
261 virtual void storageBlockingStateChanged(bool) = 0;
262
263 // Called when the private browsing state for this plug-in changes.
264 virtual void privateBrowsingStateChanged(bool) = 0;
265
266 // Gets the form value representation for the plug-in, letting plug-ins participate in form submission.
267 virtual bool getFormValue(String& formValue) = 0;
268
269 // Tells the plug-in that it should scroll. The plug-in should return true if it did scroll.
270 virtual bool handleScroll(WebCore::ScrollDirection, WebCore::ScrollGranularity) = 0;
271
272 // A plug-in can use WebCore scroll bars. Make them known, so that hit testing can find them.
273 // FIXME: This code should be in PluginView or its base class, not in individual plug-ins.
274 virtual WebCore::Scrollbar* horizontalScrollbar() = 0;
275 virtual WebCore::Scrollbar* verticalScrollbar() = 0;
276
277#if PLATFORM(COCOA)
278 virtual RetainPtr<PDFDocument> pdfDocumentForPrinting() const { return 0; }
279 virtual WebCore::FloatSize pdfDocumentSizeForPrinting() const;
280 virtual NSObject *accessibilityObject() const { return 0; }
281 virtual id accessibilityAssociatedPluginParentForElement(WebCore::Element*) const { return nullptr; }
282#endif
283
284 virtual unsigned countFindMatches(const String& target, WebCore::FindOptions, unsigned maxMatchCount) = 0;
285
286 virtual bool findString(const String& target, WebCore::FindOptions, unsigned maxMatchCount) = 0;
287
288 virtual WebCore::IntPoint convertToRootView(const WebCore::IntPoint& pointInLocalCoordinates) const;
289
290 virtual RefPtr<WebCore::SharedBuffer> liveResourceData() const = 0;
291
292 virtual bool performDictionaryLookupAtLocation(const WebCore::FloatPoint&) = 0;
293
294 virtual String getSelectionString() const = 0;
295 virtual String getSelectionForWordAtPoint(const WebCore::FloatPoint&) const = 0;
296 virtual bool existingSelectionContainsPoint(const WebCore::FloatPoint&) const = 0;
297
298 virtual void mutedStateChanged(bool) { }
299
300 virtual bool canCreateTransientPaintingSnapshot() const { return true; }
301
302 virtual bool requiresUnifiedScaleFactor() const { return false; }
303
304 virtual void willDetachRenderer() { }
305
306 virtual bool pluginHandlesContentOffsetForAccessibilityHitTest() const { return false; }
307
308protected:
309 Plugin(PluginType);
310
311 PluginType m_type;
312
313 bool m_isBeingDestroyed { false };
314
315private:
316 WeakPtr<PluginController> m_pluginController;
317};
318
