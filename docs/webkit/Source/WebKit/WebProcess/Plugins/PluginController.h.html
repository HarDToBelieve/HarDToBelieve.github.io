28#include <wtf/Forward.h>

31#if PLATFORM(COCOA)
32#include "PluginComplexTextInputState.h"

35struct NPObject;
36typedef struct _NPVariant NPVariant;
37typedef void* NPIdentifier;
38
39namespace WTF {
40class MachSendRight;
41}
42
43namespace WebCore {
44class HTTPHeaderMap;
45class IntRect;
46class ProtectionSpace;
47}
48
49namespace WebKit {
50
51class PluginController : public CanMakeWeakPtr<PluginController> {
52public:
53 // Tells the controller that the plug-in wants the given rect to be repainted. The rect is in the plug-in's coordinate system.
54 virtual void invalidate(const WebCore::IntRect&) = 0;
55
56 // Returns the user agent string.
57 virtual String userAgent() = 0;
58
59 // Loads the given URL and associates it with the request ID.
60 //
61 // If a target is specified, then the URL will be loaded in the window or frame that the target refers to.
62 // Once the URL finishes loading, Plugin::frameDidFinishLoading will be called with the given requestID. If the URL
63 // fails to load, Plugin::frameDidFailToLoad will be called.
64 //
65 // If the URL is a JavaScript URL, the JavaScript code will be evaluated and the result sent back using Plugin::didEvaluateJavaScript.
66 virtual void loadURL(uint64_t requestID, const String& method, const String& urlString, const String& target,
67 const WebCore::HTTPHeaderMap& headerFields, const Vector<uint8_t>& httpBody, bool allowPopups) = 0;
68
69 // Continues the load of a stream that was requested by loadURL.
70 virtual void continueStreamLoad(uint64_t streamID) = 0;
71
72 // Cancels the load of a stream that was requested by loadURL.
73 virtual void cancelStreamLoad(uint64_t streamID) = 0;
74
75 // Cancels the load of the manual stream.
76 virtual void cancelManualStreamLoad() = 0;
77
78#if ENABLE(NETSCAPE_PLUGIN_API)
79 // Get the NPObject that corresponds to the window JavaScript object. Returns a retained object.
80 virtual NPObject* windowScriptNPObject() = 0;
81
82 // Get the NPObject that corresponds to the plug-in's element. Returns a retained object.
83 virtual NPObject* pluginElementNPObject() = 0;
84
85 // Evaluates the given script string in the context of the given NPObject.
86 virtual bool evaluate(NPObject*, const String& scriptString, NPVariant* result, bool allowPopups) = 0;
87
88 // Called by the Netscape plug-in when it starts or stops playing audio.
89 virtual void setPluginIsPlayingAudio(bool) = 0;
90
91 // Returns whether the plugin should be muted.
92 virtual bool isMuted() const = 0;
93#endif
94
95 // Set the statusbar text.
96 virtual void setStatusbarText(const String&) = 0;
97
98 // Return whether accelerated compositing is enabled.
99 virtual bool isAcceleratedCompositingEnabled() = 0;
100
101 // Tells the controller that the plug-in process has crashed.
102 virtual void pluginProcessCrashed() = 0;
103
104#if PLATFORM(COCOA)
105 // Tells the controller that the plug-in focus or window focus did change.
106 virtual void pluginFocusOrWindowFocusChanged(bool) = 0;
107
108 // Tells the controller that complex text input be enabled or disabled for the plug-in.
109 virtual void setComplexTextInputState(PluginComplexTextInputState) = 0;
110
111 // Returns the mach port of the compositing render server.
112 virtual const WTF::MachSendRight& compositingRenderServerPort() = 0;
113#endif
114
115 // Returns the contents scale factor.
116 virtual float contentsScaleFactor() = 0;
117
118 // Returns the proxies for the given URL or null on failure.
119 virtual String proxiesForURL(const String&) = 0;
120
121 // Returns the cookies for the given URL or null on failure.
122 virtual String cookiesForURL(const String&) = 0;
123
124 // Sets the cookies for the given URL.
125 virtual void setCookiesForURL(const String& urlString, const String& cookieString) = 0;
126
127 // Get authentication credentials for the given protection space.
128 virtual bool getAuthenticationInfo(const WebCore::ProtectionSpace&, String& username, String& password) = 0;
129
130 // Returns whether private browsing is enabled.
131 virtual bool isPrivateBrowsingEnabled() = 0;
132
133 // Returns whether or not asynchronous plugin initialization is enabled.
134 virtual bool asynchronousPluginInitializationEnabled() const { return false; }
135
136 // Returns whether or not asynchronous plugin initialization should be attempted for all plugins.
137 virtual bool asynchronousPluginInitializationEnabledForAllPlugins() const { return false; }
138
139 // Returns the articifical plugin delay to use for testing of asynchronous plugin initialization.
140 virtual bool artificialPluginInitializationDelayEnabled() const { return false; }
141
142 // Increments a counter that prevents the plug-in from being destroyed.
143 virtual void protectPluginFromDestruction() = 0;
144
145 // Decrements a counter that, when it reaches 0, stops preventing the plug-in from being destroyed.
146 virtual void unprotectPluginFromDestruction() = 0;
147
148 // Called when the a plug-in instance is successfully initialized, either synchronously or asynchronously.
149 virtual void didInitializePlugin() = 0;
150
151 // Called when the a plug-in instance fails to initialized, either synchronously or asynchronously.
152 virtual void didFailToInitializePlugin() = 0;
153
154 // Helper class for delaying destruction of a plug-in.
155 class PluginDestructionProtector {
156 WTF_MAKE_FAST_ALLOCATED;
157 public:
158 explicit PluginDestructionProtector(PluginController* pluginController)
159 : m_pluginController(pluginController)
160 {
161 m_pluginController->protectPluginFromDestruction();
162 }
163
164 ~PluginDestructionProtector()
165 {
166 m_pluginController->unprotectPluginFromDestruction();
167 }
168
169 private:
170 PluginController* m_pluginController;
171 };
172
173protected:
174 virtual ~PluginController() { }
175};
176
177} // namespace WebKit
