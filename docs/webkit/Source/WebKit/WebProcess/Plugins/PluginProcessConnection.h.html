26#ifndef PluginProcessConnection_h
27#define PluginProcessConnection_h
28
29#if ENABLE(NETSCAPE_PLUGIN_API)
30
31#include "Connection.h"
32#include "Plugin.h"
33#include "PluginProcess.h"
34#include "PluginProcessAttributes.h"
35#include <wtf/RefCounted.h>
36
37namespace WebKit {
38
39class NPRemoteObjectMap;
40class PluginProcessConnectionManager;
41class PluginProxy;
42
43class PluginProcessConnection : public RefCounted<PluginProcessConnection>, IPC::Connection::Client {
44public:
45 static Ref<PluginProcessConnection> create(PluginProcessConnectionManager* pluginProcessConnectionManager, uint64_t pluginProcessToken, IPC::Connection::Identifier connectionIdentifier, bool supportsAsynchronousPluginInitialization)
46 {
47 return adoptRef(*new PluginProcessConnection(pluginProcessConnectionManager, pluginProcessToken, connectionIdentifier, supportsAsynchronousPluginInitialization));
48 }
49 ~PluginProcessConnection();
50
51 uint64_t pluginProcessToken() const { return m_pluginProcessToken; }
52
53 IPC::Connection* connection() const { return m_connection.get(); }
54
55 void addPluginProxy(PluginProxy*);
56 void removePluginProxy(PluginProxy*);
57
58 NPRemoteObjectMap* npRemoteObjectMap() const { return m_npRemoteObjectMap.get(); }
59
60 bool supportsAsynchronousPluginInitialization() const { return m_supportsAsynchronousPluginInitialization; }
61
62private:
63 PluginProcessConnection(PluginProcessConnectionManager*, uint64_t pluginProcessToken, IPC::Connection::Identifier connectionIdentifier, bool supportsAsynchronousInitialization);
64
65 // IPC::Connection::Client
66 void didReceiveMessage(IPC::Connection&, IPC::Decoder&) override;
67 bool didReceiveSyncMessage(IPC::Connection&, IPC::Decoder&, UniqueRef<IPC::Encoder>&) override;
68 void didClose(IPC::Connection&) override;
69 void didReceiveInvalidMessage(IPC::Connection&, IPC::MessageName) override;
70
71 // Message handlers.
72 bool didReceiveSyncPluginProcessConnectionMessage(IPC::Connection&, IPC::Decoder&, UniqueRef<IPC::Encoder>&);
73 void setException(const String&, CompletionHandler<void()>&&);
74
75 PluginProcessConnectionManager* m_pluginProcessConnectionManager;
76 uint64_t m_pluginProcessToken;
77
78 // The connection from the web process to the plug-in process.
79 RefPtr<IPC::Connection> m_connection;
80
81 // The plug-ins. We use a weak reference to the plug-in proxies because the plug-in view holds the strong reference.
82 HashMap<uint64_t, PluginProxy*> m_plugins;
83
84 RefPtr<NPRemoteObjectMap> m_npRemoteObjectMap;
85
86 bool m_supportsAsynchronousPluginInitialization;
87};
88
89} // namespace WebKit
90
91#endif // ENABLE(NETSCAPE_PLUGIN_API)
92
93#endif // PluginProcessConnection_h
