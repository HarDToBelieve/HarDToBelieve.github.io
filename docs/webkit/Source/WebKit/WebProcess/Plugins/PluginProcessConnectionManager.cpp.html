65PluginProcessConnection* PluginProcessConnectionManager::getPluginProcessConnection(uint64_t pluginProcessToken)
66{
67 for (size_t i = 0; i < m_pluginProcessConnections.size(); ++i) {
68 if (m_pluginProcessConnections[i]->pluginProcessToken() == pluginProcessToken)
69 return m_pluginProcessConnections[i].get();
70 }
71
72 IPC::Attachment encodedConnectionIdentifier;
73 bool supportsAsynchronousInitialization;
74 if (!WebProcess::singleton().parentProcessConnection()->sendSync(Messages::WebProcessProxy::GetPluginProcessConnection(pluginProcessToken),
75 Messages::WebProcessProxy::GetPluginProcessConnection::Reply(encodedConnectionIdentifier, supportsAsynchronousInitialization), 0))
76 return nullptr;
77
78#if USE(UNIX_DOMAIN_SOCKETS)
79 IPC::Connection::Identifier connectionIdentifier = encodedConnectionIdentifier.releaseFileDescriptor();
80#elif OS(DARWIN)
81 IPC::Connection::Identifier connectionIdentifier(encodedConnectionIdentifier.port());
82#elif OS(WINDOWS)
83 IPC::Connection::Identifier connectionIdentifier = encodedConnectionIdentifier.handle();
84#endif
85 if (!IPC::Connection::identifierIsValid(connectionIdentifier))
86 return nullptr;
87
88 auto pluginProcessConnection = PluginProcessConnection::create(this, pluginProcessToken, connectionIdentifier, supportsAsynchronousInitialization);
89 m_pluginProcessConnections.append(pluginProcessConnection.copyRef());
90
91 {
92 Locker locker { m_tokensAndConnectionsLock };
93 ASSERT(!m_tokensAndConnections.contains(pluginProcessToken));
94
95 m_tokensAndConnections.set(pluginProcessToken, pluginProcessConnection->connection());
96 }
97
98 return pluginProcessConnection.ptr();
99}
100
101void PluginProcessConnectionManager::removePluginProcessConnection(PluginProcessConnection* pluginProcessConnection)
102{
103 size_t vectorIndex = m_pluginProcessConnections.find(pluginProcessConnection);
104 ASSERT(vectorIndex != notFound);
105
106 {
107 Locker locker { m_tokensAndConnectionsLock };
108 ASSERT(m_tokensAndConnections.contains(pluginProcessConnection->pluginProcessToken()));
109
110 m_tokensAndConnections.remove(pluginProcessConnection->pluginProcessToken());
111 }
112
113 m_pluginProcessConnections.remove(vectorIndex);
114}
115
116void PluginProcessConnectionManager::pluginProcessCrashed(uint64_t pluginProcessToken)
117{
118 Locker locker { m_tokensAndConnectionsLock };
119 IPC::Connection* connection = m_tokensAndConnections.get(pluginProcessToken);
120
121 // It's OK for connection to be null here; it will happen if this web process doesn't know
122 // anything about the plug-in process.
123 if (!connection)
124 return;
125
126 connection->postConnectionDidCloseOnConnectionWorkQueue();
127}
