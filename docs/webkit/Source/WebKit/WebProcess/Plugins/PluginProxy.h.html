187 uint64_t m_pluginProcessToken { 0 };
188
189 RefPtr<PluginProcessConnection> m_connection;
190 uint64_t m_pluginInstanceID { 0 };
191
192 WebCore::IntSize m_pluginSize;
193
194 // The clip rect in plug-in coordinates.

216 bool m_wantsWheelEvents { false };
217

219 uint32_t m_remoteLayerClientID { 0 };

221 std::unique_ptr<PluginCreationParameters> m_pendingPluginCreationParameters;
222 bool m_waitingOnAsynchronousInitialization { false };
223
