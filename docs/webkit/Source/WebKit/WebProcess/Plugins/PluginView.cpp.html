115 void cancel();
116 void continueLoad();
117

120private:
121 Stream(PluginView* pluginView, uint64_t streamID, const ResourceRequest& request)

124 , m_request(request)

129 // NetscapePluginStreamLoaderClient

131 void didReceiveResponse(NetscapePlugInStreamLoader*, const ResourceResponse&) override;

157void PluginView::Stream::start()
158{
159 ASSERT(m_pluginView->m_plugin);
160 ASSERT(!m_loader);
161
162 Frame* frame = m_pluginView->m_pluginElement->document().frame();
163 ASSERT(frame);
164
165 WebProcess::singleton().webLoaderStrategy().schedulePluginStreamLoad(*frame, *this, ResourceRequest {m_request}, [this, protectedThis = makeRef(*this)](RefPtr<NetscapePlugInStreamLoader>&& loader) {
166 m_loader = WTFMove(loader);
167 });
168}

174 m_streamWasCancelled = true;
175 m_loader->cancel(m_loader->cancelledError());
176 m_loader = nullptr;
177}

187static String buildHTTPHeaders(const ResourceResponse& response, long long& expectedContentLength)
188{
189 if (!response.isInHTTPFamily())
190 return String();
191
192 StringBuilder header;
193 header.append("HTTP ", response.httpStatusCode(), ' ', response.httpStatusText(), '\n');
194 for (auto& field : response.httpHeaderFields())
195 header.append(field.key, ": ", field.value, '\n');
196
197 // If the content is encoded (most likely compressed), then don't send its length to the plugin,
198 // which is only interested in the decoded length, not yet known at the moment.
199 // <rdar://problem/4470599> tracks a request for -[NSURLResponse expectedContentLength] to incorporate this logic.
200 String contentEncoding = response.httpHeaderField(HTTPHeaderName::ContentEncoding);
201 if (!contentEncoding.isNull() && contentEncoding != "identity")
202 expectedContentLength = -1;
203
204 return header.toString();
205}

207static uint32_t lastModifiedDateMS(const ResourceResponse& response)
208{
209 auto lastModified = response.lastModified();
210 if (!lastModified)
211 return 0;
212
213 return lastModified.value().secondsSinceEpoch().millisecondsAs<uint32_t>();
214}

216void PluginView::Stream::willSendRequest(NetscapePlugInStreamLoader*, ResourceRequest&& request, const ResourceResponse& redirectResponse, CompletionHandler<void(ResourceRequest&&)>&& decisionHandler)
217{
218 const URL& requestURL = request.url();
219 const URL& redirectResponseURL = redirectResponse.url();
220
221 m_loadCallback = WTFMove(decisionHandler);
222 m_request = request;
223 m_pluginView->m_plugin->streamWillSendRequest(m_streamID, requestURL, redirectResponseURL, redirectResponse.httpStatusCode());
224}

226void PluginView::Stream::didReceiveResponse(NetscapePlugInStreamLoader*, const ResourceResponse& response)
227{
228 // Compute the stream related data from the resource response.
229 const URL& responseURL = response.url();
230 const String& mimeType = response.mimeType();
231 long long expectedContentLength = response.expectedContentLength();
232
233 String headers = buildHTTPHeaders(response, expectedContentLength);
234
235 uint32_t streamLength = 0;
236 if (expectedContentLength > 0)
237 streamLength = expectedContentLength;
238
239 m_pluginView->m_plugin->streamDidReceiveResponse(m_streamID, responseURL, streamLength, lastModifiedDateMS(response), mimeType, headers, response.suggestedFilename());
240}

244 m_pluginView->m_plugin->streamDidReceiveData(m_streamID, bytes, length);
245}

247void PluginView::Stream::didFail(NetscapePlugInStreamLoader*, const ResourceError& error)
248{
249 // Calling streamDidFail could cause us to be deleted, so we hold on to a reference here.
250 Ref<Stream> protect(*this);
251
252 // We only want to call streamDidFail if the stream was not explicitly cancelled by the plug-in.
253 if (!m_streamWasCancelled)
254 m_pluginView->m_plugin->streamDidFail(m_streamID, error.isCancellation());
255
256 m_pluginView->removeStream(this);
257 m_pluginView = 0;
258}
259
260void PluginView::Stream::didFinishLoading(NetscapePlugInStreamLoader*)
261{
262 // Calling streamDidFinishLoading could cause us to be deleted, so we hold on to a reference here.
263 Ref<Stream> protect(*this);
264
265#if ENABLE(NETSCAPE_PLUGIN_API)
266 // Protect the plug-in while we're calling into it.
267 NPRuntimeObjectMap::PluginProtector pluginProtector(&m_pluginView->m_npRuntimeObjectMap);
268#endif
269 m_pluginView->m_plugin->streamDidFinishLoading(m_streamID);
270
271 m_pluginView->removeStream(this);
272 m_pluginView = 0;
273}

303PluginView::~PluginView()
304{
305 if (m_webPage)
306 m_webPage->removePluginView(this);
307
308 ASSERT(!m_plugin || !m_plugin->isBeingDestroyed());
309
310 if (m_isWaitingUntilMediaCanStart)
311 m_pluginElement->document().removeMediaCanStartListener(*this);
312
313 m_pluginElement->document().removeAudioProducer(*this);
314
315 destroyPluginAndReset();
316
317 // Null out the plug-in element explicitly so we'll crash earlier if we try to use
318 // the plug-in view after it's been destroyed.
319 m_pluginElement = nullptr;
320}

322void PluginView::destroyPluginAndReset()
323{
324 // Cancel all pending frame loads.
325 for (FrameLoadMap::iterator it = m_pendingFrameLoads.begin(), end = m_pendingFrameLoads.end(); it != end; ++it)
326 it->key->setLoadListener(nullptr);
327
328 if (m_plugin) {
329 m_plugin->destroyPlugin();
330
331 m_pendingURLRequests.clear();
332 m_pendingURLRequestsTimer.stop();
333
334#if PLATFORM(COCOA)
335 if (m_webPage)
336 pluginFocusOrWindowFocusChanged(false);
337#endif
338 }
339
340#if ENABLE(NETSCAPE_PLUGIN_API)
341 // Invalidate the object map.
342 m_npRuntimeObjectMap.invalidate();
343#endif
344
345 cancelAllStreams();
346}

356 m_isWaitingUntilMediaCanStart = false;
357 m_manualStreamState = ManualStreamState::Initial;

359
360 initializePlugin();
361}

383void PluginView::manualLoadDidReceiveResponse(const ResourceResponse& response)
384{
385 // The plug-in can be null here if it failed to initialize.
386 if (!m_plugin)
387 return;
388
389 if (!m_isInitialized) {
390 ASSERT(m_manualStreamState == ManualStreamState::Initial);
391 m_manualStreamState = ManualStreamState::HasReceivedResponse;
392 m_manualStreamResponse = response;
393 return;
394 }
395
396 // Compute the stream related data from the resource response.
397 const URL& responseURL = response.url();
398 const String& mimeType = response.mimeType();
399 long long expectedContentLength = response.expectedContentLength();
400
401 String headers = buildHTTPHeaders(response, expectedContentLength);
402
403 uint32_t streamLength = 0;
404 if (expectedContentLength > 0)
405 streamLength = expectedContentLength;
406
407 m_plugin->manualStreamDidReceiveResponse(responseURL, streamLength, lastModifiedDateMS(response), mimeType, headers, response.suggestedFilename());
408}

559void PluginView::initializePlugin()
560{
561 if (m_isInitialized)
562 return;
563
564 if (!m_plugin) {
565 // We've already tried and failed to initialize the plug-in.
566 return;
567 }
568
569 if (Frame* frame = m_pluginElement->document().frame()) {
570 if (Page* page = frame->page()) {
571
572 // We shouldn't initialize the plug-in right now, add a listener.
573 if (!page->canStartMedia()) {
574 if (m_isWaitingUntilMediaCanStart)
575 return;
576
577 m_isWaitingUntilMediaCanStart = true;
578 m_pluginElement->document().addMediaCanStartListener(*this);
579 return;
580 }
581 }
582 }
583
584 m_pluginElement->document().addAudioProducer(*this);
585
586 m_plugin->initialize(*this, m_parameters);
587
588 // Plug-in initialization continued in didFailToInitializePlugin() or didInitializePlugin().
589}
590
591void PluginView::didFailToInitializePlugin()
592{
593 m_plugin = nullptr;
594
595#if ENABLE(NETSCAPE_PLUGIN_API)
596 String frameURLString = frame()->loader().documentLoader()->responseURL().string();
597 String pageURLString = m_webPage->corePage()->mainFrame().loader().documentLoader()->responseURL().string();
598 m_webPage->send(Messages::WebPageProxy::DidFailToInitializePlugin(m_parameters.mimeType, frameURLString, pageURLString));
599#endif
600}
601
602void PluginView::didInitializePlugin()
603{
604 m_isInitialized = true;
605
606#if PLATFORM(COCOA)
607 windowAndViewFramesChanged(m_webPage->windowFrameInScreenCoordinates(), m_webPage->viewFrameInWindowCoordinates());
608#endif
609
610 viewVisibilityDidChange();
611 viewGeometryDidChange();
612
613 if (m_pluginElement->document().focusedElement() == m_pluginElement)
614 m_plugin->setFocus(true);
615
616 redeliverManualStream();
617
618#if PLATFORM(COCOA)
619 if (m_plugin->pluginLayer() && frame()) {
620 frame()->view()->enterCompositingMode();
621 m_pluginElement->invalidateStyleAndLayerComposition();
622 }
623 m_plugin->visibilityDidChange(isVisible());
624 m_plugin->windowVisibilityChanged(m_webPage->isVisibleOrOccluded());
625 m_plugin->windowFocusChanged(m_webPage->windowIsFocused());
626#endif
627
628 if (wantsWheelEvents()) {
629 if (Frame* frame = m_pluginElement->document().frame()) {
630 if (FrameView* frameView = frame->view())
631 frameView->setNeedsLayoutAfterViewConfigurationChange();
632 }
633 }
634
635 if (Frame* frame = m_pluginElement->document().frame()) {
636 auto* webFrame = WebFrame::fromCoreFrame(*frame);
637 if (webFrame->isMainFrame())
638 webFrame->page()->send(Messages::WebPageProxy::MainFramePluginHandlesPageScaleGestureDidChange(handlesPageScaleFactor()));
639 }
640}

653JSObject* PluginView::scriptObject(JSGlobalObject* globalObject)
654{
655 // If we're already waiting for synchronous initialization of the plugin,
656 // calls to scriptObject() are from the plug-in itself and need to return 0;
657 if (m_isWaitingForSynchronousInitialization)
658 return 0;
659
660 // We might not have started initialization of the plug-in yet, the plug-in might be in the middle
661 // of being initializing asynchronously, or initialization might have previously failed.
662 if (!m_isInitialized || !m_plugin)
663 return 0;
664
665#if ENABLE(NETSCAPE_PLUGIN_API)
666 NPObject* scriptableNPObject = m_plugin->pluginScriptableNPObject();
667 if (!scriptableNPObject)
668 return 0;
669
670 JSObject* jsObject = m_npRuntimeObjectMap.getOrCreateJSObject(globalObject, scriptableNPObject);
671 releaseNPObject(scriptableNPObject);
672
673 return jsObject;
674#else
675 UNUSED_PARAM(globalObject);
676 return 0;
677#endif
678}

751void PluginView::paint(GraphicsContext& context, const IntRect& /*dirtyRect*/, Widget::SecurityOriginPaintPolicy, EventRegionContext*)
752{
753 if (!m_plugin || !m_isInitialized)
754 return;
755
756 if (context.paintingDisabled()) {
757 if (context.invalidatingControlTints())
758 m_plugin->updateControlTints(context);
759 return;
760 }
761
762 // FIXME: We should try to intersect the dirty rect with the plug-in's clip rect here.
763 IntRect paintRect = IntRect(IntPoint(), frameRect().size());
764
765 if (paintRect.isEmpty())
766 return;
767
768 if (m_transientPaintingSnapshot) {
769 m_transientPaintingSnapshot->paint(context, contentsScaleFactor(), frameRect().location(), m_transientPaintingSnapshot->bounds());
770 return;
771 }
772
773 GraphicsContextStateSaver stateSaver(context);
774
775 // Translate the coordinate system so that the origin is in the top-left corner of the plug-in.
776 context.translate(frameRect().location().x(), frameRect().location().y());
777
778 m_plugin->paint(context, paintRect);
779}

824std::unique_ptr<WebEvent> PluginView::createWebEvent(MouseEvent& event) const
825{
826 WebEvent::Type type = WebEvent::NoType;
827 unsigned clickCount = 1;
828 if (event.type() == eventNames().mousedownEvent)
829 type = WebEvent::MouseDown;
830 else if (event.type() == eventNames().mouseupEvent)
831 type = WebEvent::MouseUp;
832 else if (event.type() == eventNames().mouseoverEvent) {
833 type = WebEvent::MouseMove;
834 clickCount = 0;
835 } else if (event.type() == eventNames().clickEvent)
836 return nullptr;
837 else
838 ASSERT_NOT_REACHED();
839
840 WebMouseEvent::Button button = WebMouseEvent::NoButton;
841 switch (event.button()) {
842 case WebCore::LeftButton:
843 button = WebMouseEvent::LeftButton;
844 break;
845 case WebCore::MiddleButton:
846 button = WebMouseEvent::MiddleButton;
847 break;
848 case WebCore::RightButton:
849 button = WebMouseEvent::RightButton;
850 break;
851 default:
852 ASSERT_NOT_REACHED();
853 break;
854 }
855
856 OptionSet<WebEvent::Modifier> modifiers;
857 if (event.shiftKey())
858 modifiers.add(WebEvent::Modifier::ShiftKey);
859 if (event.ctrlKey())
860 modifiers.add(WebEvent::Modifier::ControlKey);
861 if (event.altKey())
862 modifiers.add(WebEvent::Modifier::AltKey);
863 if (event.metaKey())
864 modifiers.add(WebEvent::Modifier::MetaKey);
865
866 return makeUnique<WebMouseEvent>(type, button, event.buttons(), m_plugin->convertToRootView(IntPoint(event.offsetX(), event.offsetY())), event.screenLocation(), 0, 0, 0, clickCount, modifiers, WallTime { }, 0);
867}

869void PluginView::handleEvent(Event& event)
870{
871 if (!m_isInitialized || !m_plugin)
872 return;
873
874 const WebEvent* currentEvent = WebPage::currentEvent();
875 std::unique_ptr<WebEvent> simulatedWebEvent;
876 if (is<MouseEvent>(event) && downcast<MouseEvent>(event).isSimulated()) {
877 simulatedWebEvent = createWebEvent(downcast<MouseEvent>(event));
878 currentEvent = simulatedWebEvent.get();
879 }
880 if (!currentEvent)
881 return;
882
883 bool didHandleEvent = false;
884
885 if ((event.type() == eventNames().mousemoveEvent && currentEvent->type() == WebEvent::MouseMove)
886 || (event.type() == eventNames().mousedownEvent && currentEvent->type() == WebEvent::MouseDown)
887 || (event.type() == eventNames().mouseupEvent && currentEvent->type() == WebEvent::MouseUp)) {
888 // FIXME: Clicking in a scroll bar should not change focus.
889 if (currentEvent->type() == WebEvent::MouseDown) {
890 focusPluginElement();
891 frame()->eventHandler().setCapturingMouseEventsElement(m_pluginElement.get());
892 } else if (currentEvent->type() == WebEvent::MouseUp)
893 frame()->eventHandler().setCapturingMouseEventsElement(nullptr);
894
895 didHandleEvent = m_plugin->handleMouseEvent(static_cast<const WebMouseEvent&>(*currentEvent));
896 } else if (eventNames().isWheelEventType(event.type())
897 && currentEvent->type() == WebEvent::Wheel && m_plugin->wantsWheelEvents())
898 didHandleEvent = m_plugin->handleWheelEvent(static_cast<const WebWheelEvent&>(*currentEvent));
899 else if (event.type() == eventNames().mouseoverEvent && currentEvent->type() == WebEvent::MouseMove)
900 didHandleEvent = m_plugin->handleMouseEnterEvent(static_cast<const WebMouseEvent&>(*currentEvent));
901 else if (event.type() == eventNames().mouseoutEvent && currentEvent->type() == WebEvent::MouseMove)
902 didHandleEvent = m_plugin->handleMouseLeaveEvent(static_cast<const WebMouseEvent&>(*currentEvent));
903 else if (event.type() == eventNames().contextmenuEvent && currentEvent->type() == WebEvent::MouseDown)
904 didHandleEvent = m_plugin->handleContextMenuEvent(static_cast<const WebMouseEvent&>(*currentEvent));
905 else if ((event.type() == eventNames().keydownEvent && currentEvent->type() == WebEvent::KeyDown)
906 || (event.type() == eventNames().keyupEvent && currentEvent->type() == WebEvent::KeyUp))
907 didHandleEvent = m_plugin->handleKeyboardEvent(static_cast<const WebKeyboardEvent&>(*currentEvent));
908
909 if (didHandleEvent)
910 event.setDefaultHandled();
911}

1045void PluginView::viewGeometryDidChange()
1046{
1047 if (!m_isInitialized || !m_plugin || !parent())
1048 return;
1049
1050 ASSERT(frame());
1051 float pageScaleFactor = frame()->page() ? frame()->page()->pageScaleFactor() : 1;
1052
1053 IntPoint scaledFrameRectLocation(frameRect().location().x() * pageScaleFactor, frameRect().location().y() * pageScaleFactor);
1054 IntPoint scaledLocationInRootViewCoordinates(parent()->contentsToRootView(scaledFrameRectLocation));
1055
1056 // FIXME: We still don't get the right coordinates for transformed plugins.
1057 AffineTransform transform;
1058 transform.translate(scaledLocationInRootViewCoordinates.x(), scaledLocationInRootViewCoordinates.y());
1059 transform.scale(pageScaleFactor);
1060
1061 // FIXME: The way we calculate this clip rect isn't correct.
1062 // But it is still important to distinguish between empty and non-empty rects so we can notify the plug-in when it becomes invisible.
1063 // Making the rect actually correct is covered by https://bugs.webkit.org/show_bug.cgi?id=95362
1064 IntRect clipRect = boundsRect();
1065
1066 // FIXME: We can only get a semi-reliable answer from clipRectInWindowCoordinates() when the page is not scaled.
1067 // Fixing that is tracked in <rdar://problem/9026611> - Make the Widget hierarchy play nicely with transforms, for zoomed plug-ins and iframes
1068 if (pageScaleFactor == 1) {
1069 clipRect = clipRectInWindowCoordinates();
1070 if (!clipRect.isEmpty())
1071 clipRect = boundsRect();
1072 }
1073
1074 m_plugin->geometryDidChange(size(), clipRect, transform);
1075}

1085IntRect PluginView::clipRectInWindowCoordinates() const
1086{
1087 // Get the frame rect in window coordinates.
1088 IntRect frameRectInWindowCoordinates = parent()->contentsToWindow(frameRect());
1089
1090 Frame* frame = this->frame();
1091
1092 // Get the window clip rect for the plugin element (in window coordinates).
1093 IntRect windowClipRect = frame->view()->windowClipRectForFrameOwner(m_pluginElement.get(), true);
1094
1095 // Intersect the two rects to get the view clip rect in window coordinates.
1096 frameRectInWindowCoordinates.intersect(windowClipRect);
1097
1098 return frameRectInWindowCoordinates;

1124void PluginView::performURLRequest(URLRequest* request)
1125{
1126 // This protector is needed to make sure the PluginView is not destroyed while it is still needed.
1127 Ref<PluginView> protect(*this);
1128
1129 // First, check if this is a javascript: url.
1130 if (request->request().url().protocolIsJavaScript()) {
1131 performJavaScriptURLRequest(request);
1132 return;
1133 }
1134
1135 if (!request->target().isNull()) {
1136 performFrameLoadURLRequest(request);
1137 return;
1138 }
1139
1140 // This request is to load a URL and create a stream.
1141 auto stream = PluginView::Stream::create(this, request->requestID(), request->request());
1142 addStream(stream.ptr());
1143 stream->start();
1144}
1145
1146void PluginView::performFrameLoadURLRequest(URLRequest* request)
1147{
1148 ASSERT(!request->target().isNull());
1149
1150 Frame* frame = m_pluginElement->document().frame();
1151 if (!frame)
1152 return;
1153
1154 if (!m_pluginElement->document().securityOrigin().canDisplay(request->request().url())) {
1155 // We can't load the request, send back a reply to the plug-in.
1156 m_plugin->frameDidFail(request->requestID(), false);
1157 return;
1158 }
1159
1160 UserGestureIndicator gestureIndicator(request->allowPopups() ? std::optional<ProcessingUserGestureState>(ProcessingUserGesture) : std::nullopt);
1161
1162 // First, try to find a target frame.
1163 Frame* targetFrame = frame->loader().findFrameForNavigation(request->target());
1164 if (!targetFrame) {
1165 // We did not find a target frame. Ask our frame to load the page. This may or may not create a popup window.
1166 FrameLoadRequest frameLoadRequest { *frame, request->request() };
1167 frameLoadRequest.setFrameName(request->target());
1168 frameLoadRequest.setShouldCheckNewWindowPolicy(true);
1169 frame->loader().load(WTFMove(frameLoadRequest));
1170
1171 // FIXME: We don't know whether the window was successfully created here so we just assume that it worked.
1172 // It's better than not telling the plug-in anything.
1173 m_plugin->frameDidFinishLoading(request->requestID());
1174 return;
1175 }
1176
1177 // Now ask the frame to load the request.
1178 targetFrame->loader().load(FrameLoadRequest(*targetFrame, request->request() ));
1179
1180 auto* targetWebFrame = WebFrame::fromCoreFrame(*targetFrame);
1181 ASSERT(targetWebFrame);
1182
1183 if (WebFrame::LoadListener* loadListener = targetWebFrame->loadListener()) {
1184 // Check if another plug-in view or even this view is waiting for the frame to load.
1185 // If it is, tell it that the load was cancelled because it will be anyway.
1186 loadListener->didFailLoad(targetWebFrame, true);
1187 }
1188
1189 m_pendingFrameLoads.set(targetWebFrame, request);
1190 targetWebFrame->setLoadListener(this);
1191}
1192
1193void PluginView::performJavaScriptURLRequest(URLRequest* request)
1194{
1195 ASSERT(request->request().url().protocolIsJavaScript());
1196
1197 RefPtr<Frame> frame = m_pluginElement->document().frame();
1198 if (!frame)
1199 return;
1200
1201 String jsString = decodeURLEscapeSequences(request->request().url().string().substring(sizeof("javascript:") - 1));
1202
1203 if (!request->target().isNull()) {
1204 // For security reasons, only allow JS requests to be made on the frame that contains the plug-in.
1205 if (frame->tree().find(request->target(), *frame) != frame) {
1206 // Let the plug-in know that its frame load failed.
1207 m_plugin->frameDidFail(request->requestID(), false);
1208 return;
1209 }
1210 }
1211
1212 // Evaluate the JavaScript code. Note that running JavaScript here could cause the plug-in to be destroyed, so we
1213 // grab references to the plug-in here.
1214 RefPtr<Plugin> plugin = m_plugin;
1215 auto result = frame->script().executeScriptIgnoringException(jsString, request->allowPopups());
1216
1217 if (!result)
1218 return;
1219
1220 // Check if evaluating the JavaScript destroyed the plug-in.
1221 if (!plugin->controller())
1222 return;
1223
1224 // Don't notify the plug-in at all about targeted javascript: requests. This matches Mozilla and WebKit1.
1225 if (!request->target().isNull())
1226 return;
1227
1228 JSGlobalObject* globalObject = frame->script().globalObject(pluginWorld());
1229 String resultString;
1230 result.getString(globalObject, resultString);
1231
1232 // Send the result back to the plug-in.
1233 plugin->didEvaluateJavaScript(request->requestID(), resultString);
1234}

1596void PluginView::unprotectPluginFromDestruction()
1597{
1598 if (!m_plugin || m_plugin->isBeingDestroyed())
1599 return;
1600
1601 // A plug-in may ask us to evaluate JavaScript that removes the plug-in from the
1602 // page, but expect the object to still be alive when the call completes. Flash,
1603 // for example, may crash if the plug-in is destroyed and we return to code for
1604 // the destroyed object higher on the stack. To prevent this, if the plug-in has
1605 // only one remaining reference, call deref() asynchronously.
1606 if (hasOneRef()) {
1607 RunLoop::main().dispatch([lastRef = adoptRef(*this)] {
1608 });
1609 return;
1610 }
1611
1612 deref();
1613}

1633bool PluginView::shouldCreateTransientPaintingSnapshot() const
1634{
1635 if (!m_plugin)
1636 return false;
1637
1638 if (!m_isInitialized)
1639 return false;
1640
1641 if (FrameView* frameView = frame()->view()) {
1642 if (frameView->paintBehavior().containsAny({ PaintBehavior::SelectionOnly, PaintBehavior::SelectionAndBackgroundsOnly, PaintBehavior::ForceBlackText})) {
1643 // This paint behavior is used when drawing the find indicator and there's no need to
1644 // snapshot plug-ins, because they can never be painted as part of the find indicator.
1645 return false;
1646 }
1647 }
1648
1649 if (!m_plugin->canCreateTransientPaintingSnapshot())
1650 return false;
1651
1652 return true;
1653}
