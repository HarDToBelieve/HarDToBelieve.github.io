278inline WebCore::MediaProducer::MediaStateFlags PluginView::mediaState() const
279{
280 WebCore::MediaProducer::MediaStateFlags mediaState;
281 if (m_pluginIsPlayingAudio)
282 mediaState.add(MediaProducer::MediaState::IsPlayingAudio);
283 return mediaState;
284}
