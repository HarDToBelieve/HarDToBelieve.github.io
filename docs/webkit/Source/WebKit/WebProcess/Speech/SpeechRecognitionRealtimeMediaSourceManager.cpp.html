55class SpeechRecognitionRealtimeMediaSourceManager::Source
56 : private RealtimeMediaSource::Observer
57 , private RealtimeMediaSource::AudioSampleObserver
58{
59 WTF_MAKE_FAST_ALLOCATED;
60public:
61 Source(RealtimeMediaSourceIdentifier identifier, Ref<RealtimeMediaSource>&& source, Ref<IPC::Connection>&& connection)
62 : m_identifier(identifier)
63 , m_source(WTFMove(source))
64 , m_connection(WTFMove(connection))
65#if PLATFORM(COCOA)
66 , m_ringBuffer(makeUniqueRef<SharedRingBufferStorage>(std::bind(&Source::storageChanged, this, std::placeholders::_1)))
67#endif
68 {
69 m_source->addObserver(*this);
70 m_source->addAudioSampleObserver(*this);
71 }
72
73 ~Source()
74 {
75#if PLATFORM(COCOA)
76 storage().invalidate();
77#endif
78 m_source->removeAudioSampleObserver(*this);
79 m_source->removeObserver(*this);
80 }
81
82 void start()
83 {
84 m_source->start();
85 }
86
87 void stop()
88 {
89 m_source->stop();
90 }
91
92#if PLATFORM(COCOA)
93 SharedRingBufferStorage& storage()
94 {
95 return static_cast<SharedRingBufferStorage&>(m_ringBuffer.storage());
96 }
97#endif
98
99private:
100 void sourceStopped() final
101 {
102 if (m_source->captureDidFail()) {
103 m_connection->send(Messages::SpeechRecognitionRemoteRealtimeMediaSourceManager::RemoteCaptureFailed(m_identifier), 0);
104 return;
105 }
106 m_connection->send(Messages::SpeechRecognitionRemoteRealtimeMediaSourceManager::RemoteSourceStopped(m_identifier), 0);
107 }
108
109 void audioSamplesAvailable(const MediaTime& time, const PlatformAudioData& audioData, const AudioStreamDescription& description, size_t numberOfFrames) final
110 {
111#if PLATFORM(COCOA)
112 DisableMallocRestrictionsForCurrentThreadScope scope;
113 if (m_description != description) {
114 ASSERT(description.platformDescription().type == PlatformDescription::CAAudioStreamBasicType);
115 m_description = *WTF::get<const AudioStreamBasicDescription*>(description.platformDescription().description);
116
117 m_numberOfFrames = m_description.sampleRate() * 2;
118 m_ringBuffer.allocate(m_description.streamDescription(), m_numberOfFrames);
119 }
120
121 ASSERT(is<WebAudioBufferList>(audioData));
122 m_ringBuffer.store(downcast<WebAudioBufferList>(audioData).list(), numberOfFrames, time.timeValue());
123 m_connection->send(Messages::SpeechRecognitionRemoteRealtimeMediaSourceManager::RemoteAudioSamplesAvailable(m_identifier, time, numberOfFrames), 0);
124#else
125 UNUSED_PARAM(time);
126 UNUSED_PARAM(audioData);
127 UNUSED_PARAM(description);
128 UNUSED_PARAM(numberOfFrames);
129#endif
130 }
131
132#if PLATFORM(COCOA)
133
134 void storageChanged(SharedMemory* storage)
135 {
136 DisableMallocRestrictionsForCurrentThreadScope scope;
137 SharedMemory::Handle handle;
138 if (storage)
139 storage->createHandle(handle, SharedMemory::Protection::ReadOnly);
140#if OS(DARWIN) || OS(WINDOWS)
141 uint64_t dataSize = handle.size();
142#else
143 uint64_t dataSize = 0;
144#endif
145 m_connection->send(Messages::SpeechRecognitionRemoteRealtimeMediaSourceManager::SetStorage(m_identifier, SharedMemory::IPCHandle { WTFMove(handle), dataSize }, m_description, m_numberOfFrames), 0);
146 }
147
148#endif
149
150 void audioUnitWillStart() final
151 {
152#if USE(AUDIO_SESSION)
153 auto bufferSize = AudioSession::sharedSession().sampleRate() / 50;
154 if (AudioSession::sharedSession().preferredBufferSize() > bufferSize)
155 AudioSession::sharedSession().setPreferredBufferSize(bufferSize);
156 AudioSession::sharedSession().setCategory(AudioSession::CategoryType::PlayAndRecord, RouteSharingPolicy::Default);
157#endif
158 }
159
160 RealtimeMediaSourceIdentifier m_identifier;
161 Ref<RealtimeMediaSource> m_source;
162 Ref<IPC::Connection> m_connection;
163
164#if PLATFORM(COCOA)
165 uint64_t m_numberOfFrames { 0 };
166 CARingBuffer m_ringBuffer;
167 CAAudioStreamDescription m_description { };
168#endif
169};

172 : m_connection(WTFMove(connection))

174 WebProcess::singleton().addMessageReceiver(Messages::SpeechRecognitionRealtimeMediaSourceManager::messageReceiverName(), *this);

179 WebProcess::singleton().removeMessageReceiver(*this);

184void SpeechRecognitionRealtimeMediaSourceManager::grantSandboxExtensions(SandboxExtension::Handle&& sandboxHandleForTCCD, SandboxExtension::Handle&& sandboxHandleForMicrophone)
185{
186 m_sandboxExtensionForTCCD = SandboxExtension::create(WTFMove(sandboxHandleForTCCD));
187 if (!m_sandboxExtensionForTCCD)
188 RELEASE_LOG_ERROR(Media, "Failed to create sandbox extension for tccd");
189 else
190 m_sandboxExtensionForTCCD->consume();
191
192 m_sandboxExtensionForMicrophone = SandboxExtension::create(WTFMove(sandboxHandleForMicrophone));
193 if (!m_sandboxExtensionForMicrophone)
194 RELEASE_LOG_ERROR(Media, "Failed to create sandbox extension for microphone");
195 else
196 m_sandboxExtensionForMicrophone->consume();
197}
198
199void SpeechRecognitionRealtimeMediaSourceManager::revokeSandboxExtensions()
200{
201 if (m_sandboxExtensionForTCCD) {
202 m_sandboxExtensionForTCCD->revoke();
203 m_sandboxExtensionForTCCD = nullptr;
204 }
205
206 if (m_sandboxExtensionForMicrophone) {
207 m_sandboxExtensionForMicrophone->revoke();
208 m_sandboxExtensionForMicrophone = nullptr;
209 }
210}

214void SpeechRecognitionRealtimeMediaSourceManager::createSource(RealtimeMediaSourceIdentifier identifier, const CaptureDevice& device)
215{
216 auto result = SpeechRecognitionCaptureSource::createRealtimeMediaSource(device);
217 if (!result) {
218 RELEASE_LOG_ERROR(Media, "Failed to create realtime source");
219 send(Messages::SpeechRecognitionRemoteRealtimeMediaSourceManager::RemoteCaptureFailed(identifier), 0);
220 return;
221 }
222
223 ASSERT(!m_sources.contains(identifier));
224 m_sources.add(identifier, makeUnique<Source>(identifier, result.source(), makeRef(*messageSenderConnection())));
225}
226
227void SpeechRecognitionRealtimeMediaSourceManager::deleteSource(RealtimeMediaSourceIdentifier identifier)
228{
229 m_sources.remove(identifier);
230}
231
232void SpeechRecognitionRealtimeMediaSourceManager::start(RealtimeMediaSourceIdentifier identifier)
233{
234 if (auto source = m_sources.get(identifier))
235 source->start();
236}
237
238void SpeechRecognitionRealtimeMediaSourceManager::stop(RealtimeMediaSourceIdentifier identifier)
239{
240 if (auto source = m_sources.get(identifier))
241 source->stop();
242}
