99void WebSWClientConnection::scheduleUnregisterJobInServer(ServiceWorkerRegistrationIdentifier registrationIdentifier, WebCore::DocumentOrWorkerIdentifier documentIdentifier, CompletionHandler<void(ExceptionOr<bool>&&)>&& completionHandler)
100{
101 sendWithAsyncReply(Messages::WebSWServerConnection::ScheduleUnregisterJobInServer { ServiceWorkerJobIdentifier::generateThreadSafe(), registrationIdentifier, documentIdentifier }, [completionHandler = WTFMove(completionHandler)](auto&& result) mutable {
102 if (!result.has_value())
103 return completionHandler(result.error().toException());
104 completionHandler(result.value());
105 });
106}

128bool WebSWClientConnection::mayHaveServiceWorkerRegisteredForOrigin(const SecurityOriginData& origin) const
129{
130 if (!m_swOriginTable->isImported())
131 return true;
132
133 return m_swOriginTable->contains(origin);
134}

148void WebSWClientConnection::didMatchRegistration(uint64_t matchingRequest, std::optional<ServiceWorkerRegistrationData>&& result)
149{
150 ASSERT(isMainRunLoop());

164void WebSWClientConnection::matchRegistration(SecurityOriginData&& topOrigin, const URL& clientURL, RegistrationCallback&& callback)
165{
166 ASSERT(isMainRunLoop());
167
168 if (!mayHaveServiceWorkerRegisteredForOrigin(topOrigin)) {
169 callback(std::nullopt);
170 return;
171 }
172
173 runOrDelayTaskForImport([this, callback = WTFMove(callback), topOrigin = WTFMove(topOrigin), clientURL]() mutable {
174 uint64_t callbackID = ++m_previousCallbackIdentifier;
175 m_ongoingMatchRegistrationTasks.add(callbackID, WTFMove(callback));
176 send(Messages::WebSWServerConnection::MatchRegistration(callbackID, topOrigin, clientURL));
177 });
178}
179
180void WebSWClientConnection::runOrDelayTaskForImport(Function<void()>&& task)
181{
182 if (m_swOriginTable->isImported()) {
183 task();
184 return;
185 }
186 m_tasksPendingOriginImport.append(WTFMove(task));
187}

189void WebSWClientConnection::whenRegistrationReady(const SecurityOriginData& topOrigin, const URL& clientURL, WhenRegistrationReadyCallback&& callback)
190{
191 uint64_t callbackID = ++m_previousCallbackIdentifier;
192 m_ongoingRegistrationReadyTasks.add(callbackID, WTFMove(callback));
193 send(Messages::WebSWServerConnection::WhenRegistrationReady(callbackID, topOrigin, clientURL));
194}
195
196void WebSWClientConnection::registrationReady(uint64_t callbackID, ServiceWorkerRegistrationData&& registrationData)
197{
198 ASSERT(registrationData.activeWorker);
199 if (auto callback = m_ongoingRegistrationReadyTasks.take(callbackID))
200 callback(WTFMove(registrationData));
201}

203void WebSWClientConnection::setDocumentIsControlled(DocumentIdentifier documentIdentifier, ServiceWorkerRegistrationData&& data, CompletionHandler<void(bool)>&& completionHandler)
204{
205 auto* documentLoader = DocumentLoader::fromTemporaryDocumentIdentifier(documentIdentifier);

210void WebSWClientConnection::getRegistrations(SecurityOriginData&& topOrigin, const URL& clientURL, GetRegistrationsCallback&& callback)
211{
212 ASSERT(isMainRunLoop());
213
214 if (!mayHaveServiceWorkerRegisteredForOrigin(topOrigin)) {
215 callback({ });
216 return;
217 }
218
219 runOrDelayTaskForImport([this, callback = WTFMove(callback), topOrigin = WTFMove(topOrigin), clientURL]() mutable {
220 uint64_t callbackID = ++m_previousCallbackIdentifier;
221 m_ongoingGetRegistrationsTasks.add(callbackID, WTFMove(callback));
222 send(Messages::WebSWServerConnection::GetRegistrations { callbackID, topOrigin, clientURL });
223 });
224}

231void WebSWClientConnection::clear()
232{
233 auto registrationTasks = WTFMove(m_ongoingMatchRegistrationTasks);
234 for (auto& callback : registrationTasks.values())
235 callback(std::nullopt);
236
237 auto getRegistrationTasks = WTFMove(m_ongoingGetRegistrationsTasks);
238 for (auto& callback : getRegistrationTasks.values())
239 callback({ });
240
241 m_ongoingRegistrationReadyTasks.clear();
242
243 clearPendingJobs();
244}
