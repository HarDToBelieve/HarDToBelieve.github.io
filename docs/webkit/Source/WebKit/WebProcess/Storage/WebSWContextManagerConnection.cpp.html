144void WebSWContextManagerConnection::installServiceWorker(ServiceWorkerContextData&& data, String&& userAgent)
145{
146 auto pageConfiguration = pageConfigurationWithEmptyClients(WebProcess::singleton().sessionID());
147
148 pageConfiguration.databaseProvider = WebDatabaseProvider::getOrCreate(m_pageGroupID);
149 pageConfiguration.socketProvider = WebSocketProvider::create(m_webPageProxyID);
150 pageConfiguration.userContentProvider = m_userContentController;
151#if ENABLE(WEB_RTC)
152 pageConfiguration.libWebRTCProvider = makeUniqueRef<ServiceWorkerLibWebRTCProvider>();
153#endif
154
155 auto effectiveUserAgent = WTFMove(userAgent);
156 if (effectiveUserAgent.isNull())
157 effectiveUserAgent = m_userAgent;
158
159 pageConfiguration.loaderClientForMainFrame = makeUniqueRef<ServiceWorkerFrameLoaderClient>(m_webPageProxyID, m_pageID, FrameIdentifier::generate(), effectiveUserAgent);
160
161#if !RELEASE_LOG_DISABLED
162 auto serviceWorkerIdentifier = data.serviceWorkerIdentifier;
163#endif
164 auto serviceWorkerThreadProxy = ServiceWorkerThreadProxy::create(WTFMove(pageConfiguration), WTFMove(data), WTFMove(effectiveUserAgent), WebProcess::singleton().cacheStorageProvider(), m_storageBlockingPolicy);
165 SWContextManager::singleton().registerServiceWorkerThreadForInstall(WTFMove(serviceWorkerThreadProxy));
166
167 RELEASE_LOG(ServiceWorker, "Created service worker %" PRIu64 " in process PID %i", serviceWorkerIdentifier.toUInt64(), getCurrentProcessID());
168}

185static inline bool isValidFetch(const ResourceRequest& request, const FetchOptions& options, const URL& serviceWorkerURL, const String& referrer)
186{
187 // For exotic service workers, do not enforce checks.
188 if (!serviceWorkerURL.protocolIsInHTTPFamily())
189 return true;
190
191 if (options.mode == FetchOptions::Mode::Navigate) {
192 if (!protocolHostAndPortAreEqual(request.url(), serviceWorkerURL)) {
193 RELEASE_LOG_ERROR(ServiceWorker, "Should not intercept a navigation load that is not same-origin as the service worker URL");
194 RELEASE_ASSERT_WITH_MESSAGE(request.url().host() == serviceWorkerURL.host(), "Hosts do not match");
195 RELEASE_ASSERT_WITH_MESSAGE(request.url().protocol() == serviceWorkerURL.protocol(), "Protocols do not match");
196 RELEASE_ASSERT_WITH_MESSAGE(request.url().port() == serviceWorkerURL.port(), "Ports do not match");
197 return false;
198 }
199 return true;
200 }
201
202 String origin = request.httpOrigin();
203 URL url { URL(), origin.isEmpty() ? referrer : origin };
204 if (url.protocolIsInHTTPFamily() && !protocolHostAndPortAreEqual(url, serviceWorkerURL)) {
205 RELEASE_LOG_ERROR(ServiceWorker, "Should not intercept a non navigation load that is not originating from a same-origin context as the service worker URL");
206 ASSERT(url.host() == serviceWorkerURL.host());
207 ASSERT(url.protocol() == serviceWorkerURL.protocol());
208 ASSERT(url.port() == serviceWorkerURL.port());
209 return false;
210 }
211 return true;
212}

229void WebSWContextManagerConnection::startFetch(SWServerConnectionIdentifier serverConnectionIdentifier, ServiceWorkerIdentifier serviceWorkerIdentifier, FetchIdentifier fetchIdentifier, ResourceRequest&& request, FetchOptions&& options, IPC::FormDataReference&& formData, String&& referrer)
230{
231 auto* serviceWorkerThreadProxy = SWContextManager::singleton().serviceWorkerThreadProxy(serviceWorkerIdentifier);
232 if (!serviceWorkerThreadProxy) {
233 m_connectionToNetworkProcess->send(Messages::ServiceWorkerFetchTask::DidNotHandle { }, fetchIdentifier);
234 return;
235 }
236
237 serviceWorkerThreadProxy->setLastNavigationWasAppBound(request.isAppBound());
238
239 if (!isValidFetch(request, options, serviceWorkerThreadProxy->scriptURL(), referrer)) {
240 m_connectionToNetworkProcess->send(Messages::ServiceWorkerFetchTask::DidNotHandle { }, fetchIdentifier);
241 return;
242 }
243
244 auto client = WebServiceWorkerFetchTaskClient::create(m_connectionToNetworkProcess.copyRef(), serviceWorkerIdentifier, serverConnectionIdentifier, fetchIdentifier, request.requester() == ResourceRequest::Requester::Main);
245 std::optional<ServiceWorkerClientIdentifier> clientId;
246 if (options.clientIdentifier)
247 clientId = ServiceWorkerClientIdentifier { serverConnectionIdentifier, options.clientIdentifier.value() };
248
249 request.setHTTPBody(formData.takeData());
250 serviceWorkerThreadProxy->startFetch(serverConnectionIdentifier, fetchIdentifier, WTFMove(client), WTFMove(clientId), WTFMove(request), WTFMove(referrer), WTFMove(options));
251}
