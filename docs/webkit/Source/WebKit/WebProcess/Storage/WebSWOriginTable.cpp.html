33namespace WebKit {
34using namespace WebCore;
35
36bool WebSWOriginTable::contains(const SecurityOriginData& origin) const
37{
38 return m_serviceWorkerOriginTable.contains(computeSharedStringHash(origin.toString()));
39}
40
41void WebSWOriginTable::setSharedMemory(const SharedMemory::Handle& handle)
42{
43 auto sharedMemory = SharedMemory::map(handle, SharedMemory::Protection::ReadOnly);
44 if (!sharedMemory)
45 return;
46
47 m_serviceWorkerOriginTable.setSharedMemory(sharedMemory.releaseNonNull());
48}
