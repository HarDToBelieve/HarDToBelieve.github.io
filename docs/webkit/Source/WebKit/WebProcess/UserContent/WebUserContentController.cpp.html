71Ref<WebUserContentController> WebUserContentController::getOrCreate(UserContentControllerIdentifier identifier)
72{
73 auto& userContentControllerPtr = userContentControllers().add(identifier, nullptr).iterator->value;
74 if (userContentControllerPtr)
75 return *userContentControllerPtr;
76
77 RefPtr<WebUserContentController> userContentController = adoptRef(new WebUserContentController(identifier));
78 userContentControllerPtr = userContentController.get();
79
80 return userContentController.releaseNonNull();
81}

104InjectedBundleScriptWorld* WebUserContentController::addContentWorld(const std::pair<ContentWorldIdentifier, String>& world)
105{
106 ASSERT(world.first);
107 if (world.first == pageContentWorldIdentifier())
108 return nullptr;
109
110 auto addResult = worldMap().ensure(world.first, [&] {
111#if PLATFORM(GTK) || PLATFORM(WPE)
112 // The GLib API doesn't allow to create script worlds from the UI process. We need to
113 // use the existing world created by the web extension if any. The world name is used
114 // as the identifier.
115 if (auto* existingWorld = InjectedBundleScriptWorld::find(world.second))
116 return std::make_pair(Ref<InjectedBundleScriptWorld>(*existingWorld), 1);
117#endif
118 return std::make_pair(InjectedBundleScriptWorld::create(world.second, InjectedBundleScriptWorld::Type::User), 1);
119 });
120
121 if (addResult.isNewEntry)
122 return addResult.iterator->value.first.get();
123 return nullptr;
124}

132void WebUserContentController::removeContentWorlds(const Vector<ContentWorldIdentifier>& worldIdentifiers)
133{
134 for (auto& worldIdentifier : worldIdentifiers) {
135 ASSERT(worldIdentifier);
136 ASSERT(worldIdentifier != pageContentWorldIdentifier());
137
138 auto it = worldMap().find(worldIdentifier);
139 if (it == worldMap().end()) {
140 WTFLogAlways("Trying to remove a ContentWorld (id=%" PRIu64 ") that does not exist.", worldIdentifier.toUInt64());
141 return;
142 }
143
144 it->value.second--;
145
146 if (!it->value.second)
147 worldMap().remove(it);
148 }
149}

152{

162 }

248private:
249 WebUserMessageHandlerDescriptorProxy(WebUserContentController* controller, const String& name, InjectedBundleScriptWorld& world, uint64_t identifier)
250 : WebCore::UserMessageHandlerDescriptor(name, world.coreWorld())
251 , m_controller(controller)
252 , m_identifier(identifier)

255
256 // WebCore::UserMessageHandlerDescriptor
257 void didPostMessage(WebCore::UserMessageHandler& handler, WebCore::SerializedScriptValue* value, WTF::Function<void(SerializedScriptValue*, const String&)>&& completionHandler) override
258 {
259 WebCore::Frame* frame = handler.frame();
260 if (!frame)
261 return;
262
263 WebFrame* webFrame = WebFrame::fromCoreFrame(*frame);
264 if (!webFrame)
265 return;
266
267 WebPage* webPage = webFrame->page();
268 if (!webPage)
269 return;
270
271 auto messageReplyHandler = [completionHandler = WTFMove(completionHandler)](const IPC::DataReference& resultValue, const String& errorMessage) {
272 if (!errorMessage.isNull()) {
273 completionHandler(nullptr, errorMessage);
274 return;
275 }
276
277 auto value = SerializedScriptValue::createFromWireBytes(resultValue.vector());
278 completionHandler(value.ptr(), { });
279 };
280
281 WebProcess::singleton().parentProcessConnection()->sendWithAsyncReply(Messages::WebUserContentControllerProxy::DidPostMessage(webPage->webPageProxyIdentifier(), webFrame->info(), m_identifier, IPC::DataReference(value->data())), WTFMove(messageReplyHandler), m_controller->identifier());
282 }
283
284 RefPtr<WebUserContentController> m_controller;
285 uint64_t m_identifier;
286};

356void WebUserContentController::addUserScriptMessageHandlerInternal(InjectedBundleScriptWorld& world, uint64_t userScriptMessageHandlerIdentifier, const String& name)
357{
358 auto& messageHandlersInWorld = m_userMessageHandlers.ensure(&world, [] { return Vector<std::pair<uint64_t, RefPtr<WebUserMessageHandlerDescriptorProxy>>> { }; }).iterator->value;
359 if (messageHandlersInWorld.findMatching([&](auto& pair) { return pair.first == userScriptMessageHandlerIdentifier; }) != notFound)

361 messageHandlersInWorld.append(std::make_pair(userScriptMessageHandlerIdentifier, WebUserMessageHandlerDescriptorProxy::create(this, name, world, userScriptMessageHandlerIdentifier)));

364void WebUserContentController::removeUserScriptMessageHandlerInternal(InjectedBundleScriptWorld& world, uint64_t userScriptMessageHandlerIdentifier)
365{
366 auto it = m_userMessageHandlers.find(&world);
367 if (it == m_userMessageHandlers.end())
368 return;
369
370 auto protectedThis = makeRef(*this);
371
372 auto& userMessageHandlers = it->value;
373 bool userMessageHandlersChanged = userMessageHandlers.removeFirstMatching([userScriptMessageHandlerIdentifier](auto& pair) {
374 return pair.first == userScriptMessageHandlerIdentifier;
375 });
376
377 if (!userMessageHandlersChanged)
378 return;
379
380 if (userMessageHandlers.isEmpty())
381 m_userMessageHandlers.remove(it);
382
383 invalidateAllRegisteredUserMessageHandlerInvalidationClients();

408void WebUserContentController::addUserScriptInternal(InjectedBundleScriptWorld& world, const std::optional<uint64_t>& userScriptIdentifier, UserScript&& userScript, InjectUserScriptImmediately immediately)
409{
410 if (immediately == InjectUserScriptImmediately::Yes) {
411 Page::forEachPage([&] (auto& page) {
412 if (&page.userContentProvider() != this)
413 return;
414
415 auto& mainFrame = page.mainFrame();
416 if (userScript.injectedFrames() == UserContentInjectedFrames::InjectInTopFrameOnly) {
417 mainFrame.injectUserScriptImmediately(world.coreWorld(), userScript);
418 return;
419 }
420
421 for (auto* frame = &mainFrame; frame; frame = frame->tree().traverseNext(&mainFrame))
422 frame->injectUserScriptImmediately(world.coreWorld(), userScript);
423 });
424 }
425
426 auto& scriptsInWorld = m_userScripts.ensure(&world, [] { return Vector<std::pair<std::optional<uint64_t>, WebCore::UserScript>>(); }).iterator->value;
427 if (userScriptIdentifier && scriptsInWorld.findMatching([&](auto& pair) { return pair.first == userScriptIdentifier; }) != notFound)
428 return;
429
430 scriptsInWorld.append(std::make_pair(userScriptIdentifier, WTFMove(userScript)));
431}

473void WebUserContentController::addUserStyleSheetInternal(InjectedBundleScriptWorld& world, const std::optional<uint64_t>& userStyleSheetIdentifier, UserStyleSheet&& userStyleSheet)
474{
475 auto& styleSheetsInWorld = m_userStyleSheets.ensure(&world, [] { return Vector<std::pair<std::optional<uint64_t>, WebCore::UserStyleSheet>>(); }).iterator->value;
476 if (userStyleSheetIdentifier && styleSheetsInWorld.findMatching([&](auto& pair) { return pair.first == userStyleSheetIdentifier; }) != notFound)
477 return;
478
479 if (auto pageID = userStyleSheet.pageID()) {
480 if (auto* webPage = WebProcess::singleton().webPage(*pageID)) {
481 if (auto* page = webPage->corePage())
482 page->injectUserStyleSheet(userStyleSheet);
483 }
484 }
485
486 styleSheetsInWorld.append(std::make_pair(userStyleSheetIdentifier, WTFMove(userStyleSheet)));
487}

515void WebUserContentController::removeUserStyleSheetInternal(InjectedBundleScriptWorld& world, uint64_t userStyleSheetIdentifier)
516{
517 auto it = m_userStyleSheets.find(&world);
518 if (it == m_userStyleSheets.end())
519 return;
520
521 auto& stylesheets = it->value;
522
523 bool sheetsChanged = stylesheets.removeFirstMatching([&](auto& pair) {
524 if (pair.first != userStyleSheetIdentifier)
525 return false;
526
527 auto& userStyleSheet = pair.second;
528 if (auto pageID = userStyleSheet.pageID()) {
529 if (auto* webPage = WebProcess::singleton().webPage(*pageID)) {
530 if (auto* page = webPage->corePage())
531 page->removeInjectedUserStyleSheet(userStyleSheet);
532 }
533 }
534 return true;
535 });
536
537 if (!sheetsChanged)
538 return;
539
540 if (stylesheets.isEmpty())
541 m_userStyleSheets.remove(it);
542
543 invalidateInjectedStyleSheetCacheInAllFramesInAllPages();
544}
