59 void addUserScript(InjectedBundleScriptWorld&, WebCore::UserScript&&);

69 void addContentWorlds(const Vector<std::pair<ContentWorldIdentifier, String>>&);

72 void addUserStyleSheets(const Vector<WebUserStyleSheetData>&);
