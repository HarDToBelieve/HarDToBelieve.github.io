31#include "WebCoreArgumentCoders.h"
32#include "WebPage.h"
33#include "WebPageProxyMessages.h"
34#include "WebProcess.h"
35#include <WebCore/ColorChooserClient.h>

37namespace WebKit {
38using namespace WebCore;
39
40WebColorChooser::WebColorChooser(WebPage* page, ColorChooserClient* client, const Color& initialColor)
41 : m_colorChooserClient(client)
42 , m_page(page)
43{
44 m_page->setActiveColorChooser(this);
45 WebProcess::singleton().parentProcessConnection()->send(Messages::WebPageProxy::ShowColorPicker(initialColor, client->elementRectRelativeToRootView(), client->suggestedColors()), m_page->identifier());
46}
47
48WebColorChooser::~WebColorChooser()
49{
50 if (!m_page)
51 return;
52
53 m_page->setActiveColorChooser(0);
54}
55
56void WebColorChooser::didChooseColor(const Color& color)
57{
58 m_colorChooserClient->didChooseColor(color);
59}
60
61void WebColorChooser::didEndChooser()
62{
63 m_colorChooserClient->didEndChooser();
64}
65
66void WebColorChooser::disconnectFromPage()
67{
68 m_page = 0;
69}
70
71void WebColorChooser::reattachColorChooser(const Color& color)
72{
73 ASSERT(m_page);
74 m_page->setActiveColorChooser(this);
75
76 ASSERT(m_colorChooserClient);
77 WebProcess::singleton().parentProcessConnection()->send(Messages::WebPageProxy::ShowColorPicker(color, m_colorChooserClient->elementRectRelativeToRootView(), m_colorChooserClient->suggestedColors()), m_page->identifier());
78}
79
80void WebColorChooser::setSelectedColor(const Color& color)
81{
82 if (!m_page)
83 return;
84
85 if (m_page->activeColorChooser() != this)
86 return;
87
88 WebProcess::singleton().parentProcessConnection()->send(Messages::WebPageProxy::SetColorPickerColor(color), m_page->identifier());
89}
90
91void WebColorChooser::endChooser()
92{
93 if (!m_page)
94 return;
95
96 WebProcess::singleton().parentProcessConnection()->send(Messages::WebPageProxy::EndColorPicker(), m_page->identifier());
97}

99} // namespace WebKit
