46void WebDiagnosticLoggingClient::logDiagnosticMessage(const String& message, const String& description, WebCore::ShouldSample shouldSample)
47{
48 ASSERT(!m_page.corePage() || m_page.corePage()->settings().diagnosticLoggingEnabled());
49
50 if (!shouldLogAfterSampling(shouldSample))
51 return;
52
53 m_page.send(Messages::WebPageProxy::LogDiagnosticMessageFromWebProcess(message, description, ShouldSample::No));
54}
