28#include <wtf/HashMap.h>

33namespace WebKit {
34
35class WebPasteboardOverrides {
36 friend NeverDestroyed<WebPasteboardOverrides>;
37public:
38 static WebPasteboardOverrides& sharedPasteboardOverrides();
39
40 void addOverride(const String& pasteboardName, const String& type, const Vector<uint8_t>&);
41 void removeOverride(const String& pasteboardName, const String& type);
42
43 Vector<String> overriddenTypes(const String& pasteboardName);
44
45 bool getDataForOverride(const String& pasteboardName, const String& type, Vector<uint8_t>&) const;
46
47private:
48 WebPasteboardOverrides();
49
50 // The m_overridesMap maps string pasteboard names to pasteboard entries.
51 // Each pasteboard entry is a map of a string type to a data buffer.
52 HashMap<String, std::unique_ptr<HashMap<String, Vector<uint8_t>>>> m_overridesMap;
53};
