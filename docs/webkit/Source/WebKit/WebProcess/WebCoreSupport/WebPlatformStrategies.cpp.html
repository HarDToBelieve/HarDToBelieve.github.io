124void WebPlatformStrategies::getTypes(Vector<String>& types, const String& pasteboardName, const PasteboardContext* context)
125{
126 // First check the overrides.
127 // The purpose of the overrides is to avoid messaging back to the UI process.
128 // Therefore, if there are any overridden types, we return just those.
129 types = WebPasteboardOverrides::sharedPasteboardOverrides().overriddenTypes(pasteboardName);
130 if (!types.isEmpty())
131 return;
132
133 WebProcess::singleton().parentProcessConnection()->sendSync(Messages::WebPasteboardProxy::GetPasteboardTypes(pasteboardName, pageIdentifier(context)), Messages::WebPasteboardProxy::GetPasteboardTypes::Reply(types), 0);
134}

136RefPtr<WebCore::SharedBuffer> WebPlatformStrategies::bufferForType(const String& pasteboardType, const String& pasteboardName, const PasteboardContext* context)
137{
138 // First check the overrides.
139 Vector<uint8_t> overrideBuffer;
140 if (WebPasteboardOverrides::sharedPasteboardOverrides().getDataForOverride(pasteboardName, pasteboardType, overrideBuffer))
141 return SharedBuffer::create(WTFMove(overrideBuffer));
142
143 // Fallback to messaging the UI process for native pasteboard content.
144 SharedMemory::IPCHandle ipcHandle;
145 WebProcess::singleton().parentProcessConnection()->sendSync(Messages::WebPasteboardProxy::GetPasteboardBufferForType(pasteboardName, pasteboardType, pageIdentifier(context)), Messages::WebPasteboardProxy::GetPasteboardBufferForType::Reply(ipcHandle), 0);
146 if (ipcHandle.handle.isNull())
147 return nullptr;
148 RefPtr<SharedMemory> sharedMemoryBuffer = SharedMemory::map(ipcHandle.handle, SharedMemory::Protection::ReadOnly);
149 return SharedBuffer::create(static_cast<unsigned char *>(sharedMemoryBuffer->data()), ipcHandle.dataSize);
150}

211int64_t WebPlatformStrategies::setBufferForType(SharedBuffer* buffer, const String& pasteboardType, const String& pasteboardName, const PasteboardContext* context)
212{
213 SharedMemory::Handle handle;
214 if (buffer && buffer->size()) {
215 RefPtr<SharedMemory> sharedMemoryBuffer = SharedMemory::allocate(buffer->size());
216 // FIXME: Null check prevents crashing, but it is not great that we will have empty pasteboard content for this type,
217 // because we've already set the types.
218 if (sharedMemoryBuffer) {
219 memcpy(sharedMemoryBuffer->data(), buffer->data(), buffer->size());
220 sharedMemoryBuffer->createHandle(handle, SharedMemory::Protection::ReadOnly);
221 }
222 }
223 int64_t newChangeCount { 0 };
224 WebProcess::singleton().parentProcessConnection()->sendSync(Messages::WebPasteboardProxy::SetPasteboardBufferForType(pasteboardName, pasteboardType, SharedMemory::IPCHandle { WTFMove(handle), buffer ? buffer->size() : 0 }, pageIdentifier(context)), Messages::WebPasteboardProxy::SetPasteboardBufferForType::Reply(newChangeCount), 0);
225 return newChangeCount;
226}
