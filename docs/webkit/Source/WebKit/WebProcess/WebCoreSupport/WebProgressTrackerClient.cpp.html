29#include "WebPage.h"
30#include "WebPageProxyMessages.h"
31#include <WebCore/Frame.h>
32#include <WebCore/Page.h>
33#include <WebCore/ProgressTracker.h>
34
35namespace WebKit {
36using namespace WebCore;
37
38WebProgressTrackerClient::WebProgressTrackerClient(WebPage& webPage)
39 : m_webPage(webPage)
40{
41}
42
43void WebProgressTrackerClient::progressStarted(Frame& originatingProgressFrame)
44{
45 if (!originatingProgressFrame.isMainFrame())
46 return;
47
48 m_webPage.setMainFrameProgressCompleted(false);
49 m_webPage.send(Messages::WebPageProxy::DidStartProgress());
50}
51
52void WebProgressTrackerClient::progressEstimateChanged(Frame& originatingProgressFrame)
53{
54 if (!originatingProgressFrame.isMainFrame())
55 return;
56
57 double progress = m_webPage.corePage()->progress().estimatedProgress();
58 m_webPage.send(Messages::WebPageProxy::DidChangeProgress(progress));
59}
60
61void WebProgressTrackerClient::progressFinished(Frame& originatingProgressFrame)
62{
63 if (!originatingProgressFrame.isMainFrame())
64 return;
65
66 m_webPage.setMainFrameProgressCompleted(true);
67
68 // Notify the bundle client.
69 m_webPage.injectedBundleLoaderClient().didFinishProgress(m_webPage);
70
71 m_webPage.send(Messages::WebPageProxy::DidFinishProgress());
72}
