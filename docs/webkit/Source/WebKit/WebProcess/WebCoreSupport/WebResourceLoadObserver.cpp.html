59static bool is3xxRedirect(const ResourceResponse& response)
60{
61 return response.httpStatusCode() >= 300 && response.httpStatusCode() <= 399;
62}

76void WebResourceLoadObserver::requestStorageAccessUnderOpener(const RegistrableDomain& domainInNeedOfStorageAccess, WebPage& openerPage, Document& openerDocument)
77{
78 auto openerUrl = openerDocument.url();
79 RegistrableDomain openerDomain { openerUrl };
80 if (domainInNeedOfStorageAccess != openerDomain
81 && !openerDocument.hasRequestedPageSpecificStorageAccessWithUserInteraction(domainInNeedOfStorageAccess)
82 && !openerUrl.isAboutBlank()) {
83 WebProcess::singleton().ensureNetworkProcessConnection().connection().send(Messages::NetworkConnectionToWebProcess::RequestStorageAccessUnderOpener(domainInNeedOfStorageAccess, openerPage.identifier(), openerDomain), 0);
84
85 openerPage.addDomainWithPageLevelStorageAccess(openerDomain, domainInNeedOfStorageAccess);
86
87 // Remember user interaction-based requests since they don't need to be repeated.
88 openerDocument.setHasRequestedPageSpecificStorageAccessWithUserInteraction(domainInNeedOfStorageAccess);
89 }
90}

101void WebResourceLoadObserver::scheduleNotificationIfNeeded()
102{
103 if (m_resourceStatisticsMap.isEmpty()) {
104 m_notificationTimer.stop();
105 return;
106 }
107
108 if (!m_notificationTimer.isActive())
109 m_notificationTimer.startOneShot(minimumNotificationInterval);
110}

112void WebResourceLoadObserver::updateCentralStatisticsStore(CompletionHandler<void()>&& completionHandler)
113{
114 m_notificationTimer.stop();
115 WebProcess::singleton().ensureNetworkProcessConnection().connection().sendWithAsyncReply(Messages::NetworkConnectionToWebProcess::ResourceLoadStatisticsUpdated(takeStatistics()), WTFMove(completionHandler));
116}

119String WebResourceLoadObserver::statisticsForURL(const URL& url)
120{
121 auto* statistics = m_resourceStatisticsMap.get(RegistrableDomain { url });
122 if (!statistics)
123 return emptyString();
124
125 return makeString("Statistics for ", url.host().toString(), ":\n", statistics->toString());
126}
127
128Vector<ResourceLoadStatistics> WebResourceLoadObserver::takeStatistics()
129{
130 Vector<ResourceLoadStatistics> statistics;
131 statistics.reserveInitialCapacity(m_resourceStatisticsMap.size());
132 for (auto& statistic : m_resourceStatisticsMap.values())
133 statistics.uncheckedAppend(ResourceLoadStatistics(WTFMove(*statistic)));
134 m_resourceStatisticsMap.clear();
135 return statistics;
136}

150void WebResourceLoadObserver::logFontLoad(const Document& document, const String& familyName, bool loadStatus)
151{
152 if (isEphemeral())
153 return;
154
155#if ENABLE(WEB_API_STATISTICS)
156 RegistrableDomain registrableDomain { document.url() };
157 auto& statistics = ensureResourceStatisticsForRegistrableDomain(registrableDomain);
158 bool shouldCallNotificationCallback = false;
159 if (!loadStatus) {
160 if (statistics.fontsFailedToLoad.add(familyName).isNewEntry)
161 shouldCallNotificationCallback = true;
162 } else {
163 if (statistics.fontsSuccessfullyLoaded.add(familyName).isNewEntry)
164 shouldCallNotificationCallback = true;
165 }
166 RegistrableDomain mainFrameRegistrableDomain { document.topDocument().url() };
167 if (statistics.topFrameRegistrableDomainsWhichAccessedWebAPIs.add(mainFrameRegistrableDomain).isNewEntry)
168 shouldCallNotificationCallback = true;
169 if (shouldCallNotificationCallback)
170 scheduleNotificationIfNeeded();
171#else
172 UNUSED_PARAM(document);
173 UNUSED_PARAM(familyName);
174 UNUSED_PARAM(loadStatus);
175#endif
176}

265void WebResourceLoadObserver::logSubresourceLoading(const Frame* frame, const ResourceRequest& newRequest, const ResourceResponse& redirectResponse, FetchDestinationIsScriptLike isScriptLike)
266{
267 if (isEphemeral())
268 return;
269
270 ASSERT(frame->page());
271
272 if (!frame)
273 return;
274
275 auto* page = frame->page();
276 if (!page)
277 return;
278
279 bool isRedirect = is3xxRedirect(redirectResponse);
280 const URL& redirectedFromURL = redirectResponse.url();
281 const URL& targetURL = newRequest.url();
282 const URL& topFrameURL = frame ? frame->mainFrame().document()->url() : URL();
283
284 auto targetHost = targetURL.host();
285 auto topFrameHost = topFrameURL.host();
286
287 if (targetHost.isEmpty() || topFrameHost.isEmpty() || targetHost == topFrameHost || (isRedirect && targetHost == redirectedFromURL.host()))
288 return;
289
290 RegistrableDomain targetDomain { targetURL };
291 RegistrableDomain topFrameDomain { topFrameURL };
292 RegistrableDomain redirectedFromDomain { redirectedFromURL };
293
294 if (targetDomain == topFrameDomain || (isRedirect && targetDomain == redirectedFromDomain))
295 return;
296
297 {
298 auto& targetStatistics = ensureResourceStatisticsForRegistrableDomain(targetDomain);
299 auto lastSeen = ResourceLoadStatistics::reduceTimeResolution(WallTime::now());
300 targetStatistics.lastSeen = lastSeen;
301 targetStatistics.subresourceUnderTopFrameDomains.add(topFrameDomain);
302
303 scheduleNotificationIfNeeded();
304 }
305
306 if (frame->isMainFrame() && isScriptLike == FetchDestinationIsScriptLike::Yes) {
307 auto& topFrameStatistics = ensureResourceStatisticsForRegistrableDomain(topFrameDomain);
308 topFrameStatistics.topFrameLoadedThirdPartyScripts.add(targetDomain);
309
310 scheduleNotificationIfNeeded();
311 }
312
313 if (isRedirect) {
314 auto& redirectingOriginStatistics = ensureResourceStatisticsForRegistrableDomain(redirectedFromDomain);
315 redirectingOriginStatistics.subresourceUniqueRedirectsTo.add(targetDomain);
316 auto& targetStatistics = ensureResourceStatisticsForRegistrableDomain(targetDomain);
317 targetStatistics.subresourceUniqueRedirectsFrom.add(redirectedFromDomain);
318
319 scheduleNotificationIfNeeded();
320 }
321}

323void WebResourceLoadObserver::logWebSocketLoading(const URL& targetURL, const URL& mainFrameURL)
324{
325 if (isEphemeral())
326 return;
327
328 auto targetHost = targetURL.host();
329 auto mainFrameHost = mainFrameURL.host();
330
331 if (targetHost.isEmpty() || mainFrameHost.isEmpty() || targetHost == mainFrameHost)
332 return;
333
334 RegistrableDomain targetDomain { targetURL };
335 RegistrableDomain topFrameDomain { mainFrameURL };
336
337 if (targetDomain == topFrameDomain)
338 return;
339
340 auto lastSeen = ResourceLoadStatistics::reduceTimeResolution(WallTime::now());
341
342 auto& targetStatistics = ensureResourceStatisticsForRegistrableDomain(targetDomain);
343 targetStatistics.lastSeen = lastSeen;
344 targetStatistics.subresourceUnderTopFrameDomains.add(topFrameDomain);
345
346 scheduleNotificationIfNeeded();
347}

349void WebResourceLoadObserver::logUserInteractionWithReducedTimeResolution(const Document& document)
350{
351 auto& url = document.url();
352 if (url.protocolIsAbout() || url.isLocalFile() || url.isEmpty())
353 return;
354
355 RegistrableDomain topFrameDomain { url };
356 auto newTime = ResourceLoadStatistics::reduceTimeResolution(WallTime::now());
357 auto lastReportedUserInteraction = m_lastReportedUserInteractionMap.get(topFrameDomain);
358 if (newTime == lastReportedUserInteraction)
359 return;
360
361 m_lastReportedUserInteractionMap.set(topFrameDomain, newTime);
362
363 if (!isEphemeral()) {
364 auto& statistics = ensureResourceStatisticsForRegistrableDomain(topFrameDomain);
365 statistics.hadUserInteraction = true;
366 statistics.lastSeen = newTime;
367 statistics.mostRecentUserInteractionTime = newTime;
368 }
369
370 if (auto* frame = document.frame()) {
371 if (auto* opener = frame->loader().opener()) {
372 if (auto* openerDocument = opener->document()) {
373 if (auto* openerPage = openerDocument->page())
374 requestStorageAccessUnderOpener(topFrameDomain, WebPage::fromCorePage(*openerPage), *openerDocument);
375 }
376 }
377 }
378
379 // We notify right away in case of a user interaction instead of waiting the usual 5 seconds because we want
380 // to update cookie blocking state as quickly as possible.
381 WebProcess::singleton().ensureNetworkProcessConnection().connection().send(Messages::NetworkConnectionToWebProcess::LogUserInteraction(topFrameDomain), 0);
382
383#if !RELEASE_LOG_DISABLED
384 if (shouldLogUserInteraction) {
385 auto counter = ++m_loggingCounter;
386#define LOCAL_LOG(str, ...) \
387 RELEASE_LOG(ResourceLoadStatistics, "ResourceLoadObserver::logUserInteraction: counter=%" PRIu64 ": " str, counter, ##__VA_ARGS__)
388
389 auto escapeForJSON = [](String s) {
390 s.replace('\\', "\\\\").replace('"', "\\\"");
391 return s;
392 };
393 auto escapedURL = escapeForJSON(url.string());
394 auto escapedDomain = escapeForJSON(topFrameDomain.string());
395
396 LOCAL_LOG(R"({ "url": "%{public}s",)", escapedURL.utf8().data());
397 LOCAL_LOG(R"( "domain" : "%{public}s",)", escapedDomain.utf8().data());
398 LOCAL_LOG(R"( "until" : %f })", newTime.secondsSinceEpoch().seconds());
399
400#undef LOCAL_LOG
401 }
402#endif
403}

421bool WebResourceLoadObserver::hasCrossPageStorageAccess(const SubFrameDomain& subDomain, const TopFrameDomain& topDomain) const
422{
423 auto it = m_domainsWithCrossPageStorageAccess.find(topDomain);
424
425 if (it != m_domainsWithCrossPageStorageAccess.end())
426 return it->value.contains(subDomain);
427
428 return false;
429}

431void WebResourceLoadObserver::setDomainsWithCrossPageStorageAccess(HashMap<TopFrameDomain, SubFrameDomain>&& domains, CompletionHandler<void()>&& completionHandler)
432{
433 for (auto& topDomain : domains.keys()) {
434 m_domainsWithCrossPageStorageAccess.ensure(topDomain, [] { return HashSet<RegistrableDomain> { };
435 }).iterator->value.add(domains.get(topDomain));
436
437 // Some sites have quirks where multiple login domains require storage access.
438 if (auto additionalLoginDomain = WebCore::NetworkStorageSession::findAdditionalLoginDomain(topDomain, domains.get(topDomain))) {
439 m_domainsWithCrossPageStorageAccess.ensure(topDomain, [] { return HashSet<RegistrableDomain> { };
440 }).iterator->value.add(*additionalLoginDomain);
441 }
442 }
443 completionHandler();
444}
