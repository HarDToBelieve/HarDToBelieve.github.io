40Ref<WebSpeechRecognitionConnection> WebSpeechRecognitionConnection::create(SpeechRecognitionConnectionIdentifier identifier)
41{
42 return adoptRef(*new WebSpeechRecognitionConnection(identifier));
43}

45WebSpeechRecognitionConnection::WebSpeechRecognitionConnection(SpeechRecognitionConnectionIdentifier identifier)
46 : m_identifier(identifier)
47{
48 WebProcess::singleton().addMessageReceiver(Messages::WebSpeechRecognitionConnection::messageReceiverName(), m_identifier, *this);
49 send(Messages::WebProcessProxy::CreateSpeechRecognitionServer(m_identifier), 0);
50
51#if ENABLE(MEDIA_STREAM)
52 WebProcess::singleton().ensureSpeechRecognitionRealtimeMediaSourceManager();
53#endif
54}
55
56WebSpeechRecognitionConnection::~WebSpeechRecognitionConnection()
57{
58 send(Messages::WebProcessProxy::DestroySpeechRecognitionServer(m_identifier), 0);
59 WebProcess::singleton().removeMessageReceiver(*this);
60}

62void WebSpeechRecognitionConnection::registerClient(WebCore::SpeechRecognitionConnectionClient& client)
63{
64 m_clientMap.add(client.identifier(), makeWeakPtr(client));
65}

67void WebSpeechRecognitionConnection::unregisterClient(WebCore::SpeechRecognitionConnectionClient& client)
68{
69 m_clientMap.remove(client.identifier());
70}

92void WebSpeechRecognitionConnection::didReceiveUpdate(WebCore::SpeechRecognitionUpdate&& update)
93{
94 auto clientIdentifier = update.clientIdentifier();
95 if (!m_clientMap.contains(clientIdentifier))
96 return;
97
98 auto client = m_clientMap.get(clientIdentifier);
99 if (!client) {
100 m_clientMap.remove(clientIdentifier);
101 // Inform server that client does not exist any more.
102 invalidate(clientIdentifier);
103 return;
104 }
105
106 switch (update.type()) {
107 case WebCore::SpeechRecognitionUpdateType::Start:
108 client->didStart();
109 break;
110 case WebCore::SpeechRecognitionUpdateType::AudioStart:
111 client->didStartCapturingAudio();
112 break;
113 case WebCore::SpeechRecognitionUpdateType::SoundStart:
114 client->didStartCapturingSound();
115 break;
116 case WebCore::SpeechRecognitionUpdateType::SpeechStart:
117 client->didStartCapturingSpeech();
118 break;
119 case WebCore::SpeechRecognitionUpdateType::SpeechEnd:
120 client->didStopCapturingSpeech();
121 break;
122 case WebCore::SpeechRecognitionUpdateType::SoundEnd:
123 client->didStopCapturingSound();
124 break;
125 case WebCore::SpeechRecognitionUpdateType::AudioEnd:
126 client->didStopCapturingAudio();
127 break;
128 case WebCore::SpeechRecognitionUpdateType::NoMatch:
129 client->didFindNoMatch();
130 break;
131 case WebCore::SpeechRecognitionUpdateType::Result:
132 client->didReceiveResult(update.result());
133 break;
134 case WebCore::SpeechRecognitionUpdateType::Error:
135 client->didError(update.error());
136 break;
137 case WebCore::SpeechRecognitionUpdateType::End:
138 client->didEnd();
139 }
140}
