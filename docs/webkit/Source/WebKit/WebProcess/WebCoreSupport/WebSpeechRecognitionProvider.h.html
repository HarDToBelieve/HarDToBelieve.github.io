26#pragma once
27
28#include "WebPage.h"
29#include "WebSpeechRecognitionConnection.h"
30#include <WebCore/SpeechRecognitionProvider.h>
31
32namespace WebKit {
33
34class WebSpeechRecognitionProvider final : public WebCore::SpeechRecognitionProvider {
35public:
36 explicit WebSpeechRecognitionProvider(WebCore::PageIdentifier identifier)
37 : m_pageIdentifier(identifier)
38 {
39 }
40
41 WebCore::SpeechRecognitionConnection& speechRecognitionConnection() final
42 {
43 if (!m_connection)
44 m_connection = WebSpeechRecognitionConnection::create(m_pageIdentifier);
45
46 return *m_connection;
47 }
48
49private:
50 WebCore::PageIdentifier m_pageIdentifier;
51 RefPtr<WebSpeechRecognitionConnection> m_connection;
52};
53
54} // namespace WebKit
