37const Vector<RefPtr<WebCore::PlatformSpeechSynthesisVoice>>& WebSpeechSynthesisClient::voiceList()
38{
39 // FIXME: this message should not be sent synchronously. Instead, the UI process should
40 // get the list of voices and pass it on to the WebContent processes, see
41 // https://bugs.webkit.org/show_bug.cgi?id=195723
42 Vector<WebSpeechSynthesisVoice> voiceList;
43 m_page.sendSync(Messages::WebPageProxy::SpeechSynthesisVoiceList(), voiceList);
44
45 m_voices.clear();
46 for (auto& voice : voiceList)
47 m_voices.append(WebCore::PlatformSpeechSynthesisVoice::create(voice.voiceURI, voice.name, voice.lang, voice.localService, voice.defaultLang));
48 return m_voices;
49}

51WebCore::SpeechSynthesisClientObserver* WebSpeechSynthesisClient::corePageObserver() const
52{
53 if (m_page.corePage() && m_page.corePage()->speechSynthesisClient() && m_page.corePage()->speechSynthesisClient()->observer())
54 return m_page.corePage()->speechSynthesisClient()->observer().get();
55 return nullptr;
56}

58void WebSpeechSynthesisClient::speak(RefPtr<WebCore::PlatformSpeechSynthesisUtterance> utterance)
59{
60 WTF::CompletionHandler<void()> startedCompletionHandler = [this, weakThis = makeWeakPtr(*this)]() mutable {
61 if (!weakThis)
62 return;
63 if (auto observer = corePageObserver())
64 observer->didStartSpeaking();
65 };
66
67 WTF::CompletionHandler<void()> finishedCompletionHandler = [this, weakThis = makeWeakPtr(*this)]() mutable {
68 if (!weakThis)
69 return;
70 if (auto observer = corePageObserver())
71 observer->didFinishSpeaking();
72 };
73
74 auto voice = utterance->voice();
75 auto voiceURI = voice ? voice->voiceURI() : "";
76 auto name = voice ? voice->name() : "";
77 auto lang = voice ? voice->lang() : "";
78 auto localService = voice ? voice->localService() : false;
79 auto isDefault = voice ? voice->isDefault() : false;
80
81 m_page.sendWithAsyncReply(Messages::WebPageProxy::SpeechSynthesisSetFinishedCallback(), WTFMove(finishedCompletionHandler));
82 m_page.sendWithAsyncReply(Messages::WebPageProxy::SpeechSynthesisSpeak(utterance->text(), utterance->lang(), utterance->volume(), utterance->rate(), utterance->pitch(), utterance->startTime(), voiceURI, name, lang, localService, isDefault), WTFMove(startedCompletionHandler));
83}

90void WebSpeechSynthesisClient::pause()
91{
92 WTF::CompletionHandler<void()> completionHandler = [this, weakThis = makeWeakPtr(*this)]() mutable {
93 if (!weakThis)
94 return;
95 if (auto observer = corePageObserver())
96 observer->didPauseSpeaking();
97 };
98
99 m_page.sendWithAsyncReply(Messages::WebPageProxy::SpeechSynthesisPause(), WTFMove(completionHandler));
100}

102void WebSpeechSynthesisClient::resume()
103{
104 WTF::CompletionHandler<void()> completionHandler = [this, weakThis = makeWeakPtr(*this)]() mutable {
105 if (!weakThis)
106 return;
107 if (auto observer = corePageObserver())
108 observer->didResumeSpeaking();
109 };
110
111 m_page.sendWithAsyncReply(Messages::WebPageProxy::SpeechSynthesisResume(), WTFMove(completionHandler));
112}
