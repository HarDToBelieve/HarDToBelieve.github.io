66void WebContextMenuClient::searchWithGoogle(const Frame* frame)
67{

70
71 m_page->send(Messages::WebPageProxy::SearchTheWeb(searchString));

74void WebContextMenuClient::searchWithSpotlight()
75{
76 // FIXME: Why do we need to search all the frames like this?
77 // Isn't there any function in WebCore that can do this?
78 // If not, can we find a place in WebCore to put this?
79
80 Frame& mainFrame = m_page->corePage()->mainFrame();
81
82 Frame* selectionFrame = &mainFrame;
83 for (; selectionFrame; selectionFrame = selectionFrame->tree().traverseNext()) {
84 if (selectionFrame->selection().isRange())
85 break;
86 }
87 if (!selectionFrame)
88 selectionFrame = &mainFrame;
89
90 String selectedString = selectionFrame->displayStringModifiedByEncoding(selectionFrame->editor().selectedText());
91
92 if (selectedString.isEmpty())
93 return;
94
95 m_page->send(Messages::WebPageProxy::SearchWithSpotlight(selectedString));
96}
