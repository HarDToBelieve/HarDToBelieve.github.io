66void WebPage::platformDidReceiveLoadParameters(const LoadParameters& parameters)
67{
68#if HAVE(LSDATABASECONTEXT)
69 static bool hasWaitedForLaunchServicesDatabase = false;
70 if (!hasWaitedForLaunchServicesDatabase) {
71 auto startTime = WallTime::now();
72 bool databaseUpdated = LaunchServicesDatabaseManager::singleton().waitForDatabaseUpdate(5_s);
73 auto elapsedTime = WallTime::now() - startTime;
74 if (elapsedTime.value() > 0.5)
75 RELEASE_LOG(Loading, "Waiting for Launch Services database update took %f seconds", elapsedTime.value());
76 ASSERT_UNUSED(databaseUpdated, databaseUpdated);
77 if (!databaseUpdated)
78 RELEASE_LOG_ERROR(Loading, "Timed out waiting for Launch Services database update.");
79 hasWaitedForLaunchServicesDatabase = true;
80 }
81#endif
82
83 m_dataDetectionContext = parameters.dataDetectionContext;
84
85 consumeNetworkExtensionSandboxExtensions(parameters.networkExtensionSandboxExtensionHandles);
86
87#if PLATFORM(IOS)
88 if (parameters.contentFilterExtensionHandle)
89 SandboxExtension::consumePermanently(*parameters.contentFilterExtensionHandle);
90 ParentalControlsContentFilter::setHasConsumedSandboxExtension(parameters.contentFilterExtensionHandle.has_value());
91
92 if (parameters.frontboardServiceExtensionHandle)
93 SandboxExtension::consumePermanently(*parameters.frontboardServiceExtensionHandle);
94#endif
95}

97void WebPage::requestActiveNowPlayingSessionInfo(CompletionHandler<void(bool, bool, const String&, double, double, uint64_t)>&& completionHandler)
98{
99 bool hasActiveSession = false;
100 String title = emptyString();
101 double duration = NAN;
102 double elapsedTime = NAN;
103 uint64_t uniqueIdentifier = 0;
104 bool registeredAsNowPlayingApplication = false;
105 if (auto* sharedManager = WebCore::PlatformMediaSessionManager::sharedManagerIfExists()) {
106 hasActiveSession = sharedManager->hasActiveNowPlayingSession();
107 title = sharedManager->lastUpdatedNowPlayingTitle();
108 duration = sharedManager->lastUpdatedNowPlayingDuration();
109 elapsedTime = sharedManager->lastUpdatedNowPlayingElapsedTime();
110 uniqueIdentifier = sharedManager->lastUpdatedNowPlayingInfoUniqueIdentifier().toUInt64();
111 registeredAsNowPlayingApplication = sharedManager->registeredAsNowPlayingApplication();
112 }
113
114 completionHandler(hasActiveSession, registeredAsNowPlayingApplication, title, duration, elapsedTime, uniqueIdentifier);
115}

117void WebPage::performDictionaryLookupAtLocation(const FloatPoint& floatPoint)
118{
119 if (auto* pluginView = pluginViewForFrame(&m_page->mainFrame())) {
120 if (pluginView->performDictionaryLookupAtLocation(floatPoint))
121 return;
122 }
123
124 // Find the frame the point is over.
125 constexpr OptionSet<HitTestRequest::Type> hitType { HitTestRequest::Type::ReadOnly, HitTestRequest::Type::Active, HitTestRequest::Type::DisallowUserAgentShadowContent, HitTestRequest::Type::AllowChildFrameContent };
126 auto result = m_page->mainFrame().eventHandler().hitTestResultAtPoint(m_page->mainFrame().view()->windowToContents(roundedIntPoint(floatPoint)), hitType);
127
128 auto* frame = result.innerNonSharedNode() ? result.innerNonSharedNode()->document().frame() : &m_page->focusController().focusedOrMainFrame();
129 if (!frame)
130 return;
131
132 auto rangeResult = DictionaryLookup::rangeAtHitTestResult(result);
133 if (!rangeResult)
134 return;
135
136 auto [range, options] = WTFMove(*rangeResult);
137 performDictionaryLookupForRange(*frame, range, options, TextIndicatorPresentationTransition::Bounce);
138}

161DictionaryPopupInfo WebPage::dictionaryPopupInfoForRange(Frame& frame, const SimpleRange& range, NSDictionary *options, TextIndicatorPresentationTransition presentationTransition)
162{
163 Editor& editor = frame.editor();
164 editor.setIsGettingDictionaryPopupInfo(true);
165
166 // FIXME: Inefficient to call stripWhiteSpace to detect whether a string has a non-whitespace character in it.
167 if (plainText(range).stripWhiteSpace().isEmpty()) {
168 editor.setIsGettingDictionaryPopupInfo(false);
169 return { };
170 }
171
172 auto quads = RenderObject::absoluteTextQuads(range);
173 if (quads.isEmpty()) {
174 editor.setIsGettingDictionaryPopupInfo(false);
175 return { };
176 }
177
178 DictionaryPopupInfo dictionaryPopupInfo;
179
180 IntRect rangeRect = frame.view()->contentsToWindow(quads[0].enclosingBoundingBox());
181
182 const RenderStyle* style = range.startContainer().renderStyle();
183 float scaledAscent = style ? style->fontMetrics().ascent() * pageScaleFactor() : 0;
184 dictionaryPopupInfo.origin = FloatPoint(rangeRect.x(), rangeRect.y() + scaledAscent);
185 dictionaryPopupInfo.options = options;
186
187#if PLATFORM(MAC)
188 auto attributedString = editingAttributedString(range, IncludeImages::No).string;
189 auto scaledAttributedString = adoptNS([[NSMutableAttributedString alloc] initWithString:[attributedString string]]);
190 NSFontManager *fontManager = [NSFontManager sharedFontManager];
191 [attributedString enumerateAttributesInRange:NSMakeRange(0, [attributedString length]) options:0 usingBlock:^(NSDictionary *attributes, NSRange range, BOOL *stop) {
192 RetainPtr<NSMutableDictionary> scaledAttributes = adoptNS([attributes mutableCopy]);
193 NSFont *font = [scaledAttributes objectForKey:NSFontAttributeName];
194 if (font)
195 font = [fontManager convertFont:font toSize:font.pointSize * pageScaleFactor()];
196 if (font)
197 [scaledAttributes setObject:font forKey:NSFontAttributeName];
198 [scaledAttributedString addAttributes:scaledAttributes.get() range:range];
199 }];
200#endif // PLATFORM(MAC)
201
202 OptionSet<TextIndicatorOption> indicatorOptions { TextIndicatorOption::UseBoundingRectAndPaintAllContentForComplexRanges };
203 if (HTMLElement::isInsideImageOverlay(range))
204 indicatorOptions.add({ TextIndicatorOption::PaintAllContent, TextIndicatorOption::PaintBackgrounds });
205
206 if (presentationTransition == TextIndicatorPresentationTransition::BounceAndCrossfade)
207 indicatorOptions.add(TextIndicatorOption::IncludeSnapshotWithSelectionHighlight);
208
209 auto textIndicator = TextIndicator::createWithRange(range, indicatorOptions, presentationTransition);
210 if (!textIndicator) {
211 editor.setIsGettingDictionaryPopupInfo(false);
212 return dictionaryPopupInfo;
213 }
214
215 dictionaryPopupInfo.textIndicator = textIndicator->data();
216#if PLATFORM(MAC)
217 dictionaryPopupInfo.attributedString = scaledAttributedString;
218#elif PLATFORM(MACCATALYST)
219 dictionaryPopupInfo.attributedString = adoptNS([[NSMutableAttributedString alloc] initWithString:plainText(range)]);
220#endif
221
222 editor.setIsGettingDictionaryPopupInfo(false);
223 return dictionaryPopupInfo;
224}

226void WebPage::insertDictatedTextAsync(const String& text, const EditingRange& replacementEditingRange, const Vector<WebCore::DictationAlternative>& dictationAlternativeLocations, InsertTextOptions&& options)
227{
228 auto& frame = m_page->focusController().focusedOrMainFrame();
229 Ref<Frame> protector { frame };
230
231 if (replacementEditingRange.location != notFound) {
232 auto replacementRange = EditingRange::toRange(frame, replacementEditingRange);
233 if (replacementRange)
234 frame.selection().setSelection(VisibleSelection { *replacementRange });
235 }
236
237 if (options.registerUndoGroup)
238 send(Messages::WebPageProxy::RegisterInsertionUndoGrouping { });
239
240 RefPtr<Element> focusedElement = frame.document() ? frame.document()->focusedElement() : nullptr;
241 if (focusedElement && options.shouldSimulateKeyboardInput)
242 focusedElement->dispatchEvent(Event::create(eventNames().keydownEvent, Event::CanBubble::Yes, Event::IsCancelable::Yes));
243
244 ASSERT(!frame.editor().hasComposition());
245 frame.editor().insertDictatedText(text, dictationAlternativeLocations, nullptr /* triggeringEvent */);
246
247 if (focusedElement && options.shouldSimulateKeyboardInput) {
248 focusedElement->dispatchEvent(Event::create(eventNames().keyupEvent, Event::CanBubble::Yes, Event::IsCancelable::Yes));
249 focusedElement->dispatchEvent(Event::create(eventNames().changeEvent, Event::CanBubble::Yes, Event::IsCancelable::Yes));
250 }
251}

290RetainPtr<CFDataRef> WebPage::pdfSnapshotAtSize(IntRect rect, IntSize bitmapSize, SnapshotOptions options)
291{
292 Frame* coreFrame = m_mainFrame->coreFrame();
293 if (!coreFrame)
294 return nullptr;
295
296 FrameView* frameView = coreFrame->view();
297 if (!frameView)
298 return nullptr;
299
300 auto data = adoptCF(CFDataCreateMutable(kCFAllocatorDefault, 0));
301
302 auto dataConsumer = adoptCF(CGDataConsumerCreateWithCFData(data.get()));
303 auto mediaBox = CGRectMake(0, 0, bitmapSize.width(), bitmapSize.height());
304 auto pdfContext = adoptCF(CGPDFContextCreate(dataConsumer.get(), &mediaBox, nullptr));
305
306 int64_t remainingHeight = bitmapSize.height();
307 int64_t nextRectY = rect.y();
308 while (remainingHeight > 0) {
309 // PDFs have a per-page height limit of 200 inches at 72dpi.
310 // We'll export one PDF page at a time, up to that maximum height.
311 static const int64_t maxPageHeight = 72 * 200;
312 bitmapSize.setHeight(std::min(remainingHeight, maxPageHeight));
313 rect.setHeight(bitmapSize.height());
314 rect.setY(nextRectY);
315
316 CGRect mediaBox = CGRectMake(0, 0, bitmapSize.width(), bitmapSize.height());
317 auto mediaBoxData = adoptCF(CFDataCreate(NULL, (const UInt8 *)&mediaBox, sizeof(CGRect)));
318 auto dictionary = (CFDictionaryRef)@{
319 (NSString *)kCGPDFContextMediaBox : (NSData *)mediaBoxData.get()
320 };
321
322 CGPDFContextBeginPage(pdfContext.get(), dictionary);
323
324 GraphicsContextCG graphicsContext { pdfContext.get() };
325 graphicsContext.scale({ 1, -1 });
326 graphicsContext.translate(0, -bitmapSize.height());
327
328 paintSnapshotAtSize(rect, bitmapSize, options, *coreFrame, *frameView, graphicsContext);
329
330 CGPDFContextEndPage(pdfContext.get());
331
332 nextRectY += bitmapSize.height();
333 remainingHeight -= maxPageHeight;
334 }
335
336 CGPDFContextClose(pdfContext.get());
337
338 return data;
339}

350void WebPage::getPlatformEditorStateCommon(const Frame& frame, EditorState& result) const
351{
352 if (result.isMissingPostLayoutData)
353 return;
354
355 const auto& selection = frame.selection().selection();
356
357 if (!result.isContentEditable || selection.isNone())
358 return;
359
360 auto& postLayoutData = result.postLayoutData();
361 if (auto editingStyle = EditingStyle::styleAtSelectionStart(selection)) {
362 if (editingStyle->hasStyle(CSSPropertyFontWeight, "bold"_s))
363 postLayoutData.typingAttributes |= AttributeBold;
364
365 if (editingStyle->hasStyle(CSSPropertyFontStyle, "italic"_s) || editingStyle->hasStyle(CSSPropertyFontStyle, "oblique"_s))
366 postLayoutData.typingAttributes |= AttributeItalics;
367
368 if (editingStyle->hasStyle(CSSPropertyWebkitTextDecorationsInEffect, "underline"_s))
369 postLayoutData.typingAttributes |= AttributeUnderline;
370
371 if (auto* styleProperties = editingStyle->style()) {
372 bool isLeftToRight = styleProperties->propertyAsValueID(CSSPropertyDirection) == CSSValueLtr;
373 switch (styleProperties->propertyAsValueID(CSSPropertyTextAlign)) {
374 case CSSValueRight:
375 case CSSValueWebkitRight:
376 postLayoutData.textAlignment = RightAlignment;
377 break;
378 case CSSValueLeft:
379 case CSSValueWebkitLeft:
380 postLayoutData.textAlignment = LeftAlignment;
381 break;
382 case CSSValueCenter:
383 case CSSValueWebkitCenter:
384 postLayoutData.textAlignment = CenterAlignment;
385 break;
386 case CSSValueJustify:
387 postLayoutData.textAlignment = JustifiedAlignment;
388 break;
389 case CSSValueStart:
390 postLayoutData.textAlignment = isLeftToRight ? LeftAlignment : RightAlignment;
391 break;
392 case CSSValueEnd:
393 postLayoutData.textAlignment = isLeftToRight ? RightAlignment : LeftAlignment;
394 break;
395 default:
396 break;
397 }
398 if (auto textColor = styleProperties->propertyAsColor(CSSPropertyColor))
399 postLayoutData.textColor = *textColor;
400 }
401 }
402
403 if (auto* enclosingListElement = enclosingList(selection.start().containerNode())) {
404 if (is<HTMLUListElement>(*enclosingListElement))
405 postLayoutData.enclosingListType = UnorderedList;
406 else if (is<HTMLOListElement>(*enclosingListElement))
407 postLayoutData.enclosingListType = OrderedList;
408 else
409 ASSERT_NOT_REACHED();
410 }
411
412 postLayoutData.baseWritingDirection = frame.editor().baseWritingDirectionForSelectionStart();
413}

425void WebPage::getPDFFirstPageSize(WebCore::FrameIdentifier frameID, CompletionHandler<void(WebCore::FloatSize)>&& completionHandler)
426{
427 auto* webFrame = WebProcess::singleton().webFrame(frameID);
428 if (!webFrame)
429 return completionHandler({ });
430
431 auto* coreFrame = webFrame->coreFrame();
432 if (!coreFrame)
433 return completionHandler({ });
434
435 auto* pluginView = pluginViewForFrame(coreFrame);
436 if (!pluginView)
437 return completionHandler({ });
438
439 auto* plugin = pluginView->plugin();
440 if (!plugin)
441 return completionHandler({ });
442
443 completionHandler(FloatSize(plugin->pdfDocumentSizeForPrinting()));
444}
