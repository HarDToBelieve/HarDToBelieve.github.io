47std::unique_ptr<DrawingArea> DrawingArea::create(WebPage& webPage, const WebPageCreationParameters& parameters)
48{
49 switch (parameters.drawingAreaType) {
50#if PLATFORM(COCOA)
51#if !PLATFORM(IOS_FAMILY)
52 case DrawingAreaTypeTiledCoreAnimation:
53 return makeUnique<TiledCoreAnimationDrawingArea>(webPage, parameters);
54#endif
55 case DrawingAreaTypeRemoteLayerTree:
56 return makeUnique<RemoteLayerTreeDrawingArea>(webPage, parameters);
57#elif USE(COORDINATED_GRAPHICS) || USE(TEXTURE_MAPPER)
58 case DrawingAreaTypeCoordinatedGraphics:
59 return makeUnique<DrawingAreaCoordinatedGraphics>(webPage, parameters);
60#endif
61 }
62
63 return nullptr;
64}
65
66DrawingArea::DrawingArea(DrawingAreaType type, DrawingAreaIdentifier identifier, WebPage& webPage)
67 : m_type(type)
68 , m_identifier(identifier)
69 , m_webPage(webPage)
70{
71 WebProcess::singleton().addMessageReceiver(Messages::DrawingArea::messageReceiverName(), m_identifier, *this);
72}
73
74DrawingArea::~DrawingArea()
75{
76 removeMessageReceiverIfNeeded();
77}
78
79void DrawingArea::dispatchAfterEnsuringUpdatedScrollPosition(WTF::Function<void ()>&& function)
80{
81 // Scroll position updates are synchronous by default so we can just call the function right away here.
82 function();
83}
84
85void DrawingArea::removeMessageReceiverIfNeeded()
86{
87 if (m_hasRemovedMessageReceiver)
88 return;
89 m_hasRemovedMessageReceiver = true;
90 WebProcess::singleton().removeMessageReceiver(Messages::DrawingArea::messageReceiverName(), m_identifier);
91}
92
93RefPtr<WebCore::DisplayRefreshMonitor> DrawingArea::createDisplayRefreshMonitor(WebCore::PlatformDisplayID)
94{
95 return nullptr;
96}
97
98bool DrawingArea::supportsGPUProcessRendering(DrawingAreaType type)
99{
100 switch (type) {
101#if PLATFORM(COCOA)
102#if !PLATFORM(IOS_FAMILY)
103 case DrawingAreaTypeTiledCoreAnimation:
104 return false;
105#endif
106 case DrawingAreaTypeRemoteLayerTree:
107 return true;
108#elif USE(COORDINATED_GRAPHICS) || USE(TEXTURE_MAPPER)
109 case DrawingAreaTypeCoordinatedGraphics:
110 return false;
111#endif
112 default:
113 return false;
114 }
115}
