73#if ENABLE(SCROLLING_THREAD)
74void EventDispatcher::addScrollingTreeForPage(WebPage* webPage)
75{
76 Locker locker { m_scrollingTreesLock };
77
78 ASSERT(webPage->corePage()->scrollingCoordinator());
79 ASSERT(!m_scrollingTrees.contains(webPage->identifier()));
80
81 AsyncScrollingCoordinator& scrollingCoordinator = downcast<AsyncScrollingCoordinator>(*webPage->corePage()->scrollingCoordinator());
82 m_scrollingTrees.set(webPage->identifier(), downcast<ThreadedScrollingTree>(scrollingCoordinator.scrollingTree()));
83}
84
85void EventDispatcher::removeScrollingTreeForPage(WebPage* webPage)
86{
87 Locker locker { m_scrollingTreesLock };
88 ASSERT(m_scrollingTrees.contains(webPage->identifier()));
89
90 m_scrollingTrees.remove(webPage->identifier());
91}

99void EventDispatcher::wheelEvent(PageIdentifier pageID, const WebWheelEvent& wheelEvent, bool canRubberBandAtLeft, bool canRubberBandAtRight, bool canRubberBandAtTop, bool canRubberBandAtBottom)
100{
101#if PLATFORM(COCOA) || ENABLE(SCROLLING_THREAD)
102 PlatformWheelEvent platformWheelEvent = platform(wheelEvent);
103#endif
104
105#if PLATFORM(COCOA)
106 switch (wheelEvent.phase()) {
107 case WebWheelEvent::PhaseBegan:
108 m_recentWheelEventDeltaFilter->beginFilteringDeltas();
109 break;
110 case WebWheelEvent::PhaseEnded:
111 m_recentWheelEventDeltaFilter->endFilteringDeltas();
112 break;
113 default:
114 break;
115 }
116
117 if (m_recentWheelEventDeltaFilter->isFilteringDeltas()) {
118 m_recentWheelEventDeltaFilter->updateFromDelta(FloatSize(platformWheelEvent.deltaX(), platformWheelEvent.deltaY()));
119 FloatSize filteredDelta = m_recentWheelEventDeltaFilter->filteredDelta();
120 platformWheelEvent = platformWheelEvent.copyWithDeltasAndVelocity(filteredDelta.width(), filteredDelta.height(), m_recentWheelEventDeltaFilter->filteredVelocity());
121 }
122#endif
123
124 auto processingSteps = OptionSet<WebCore::WheelEventProcessingSteps> { WheelEventProcessingSteps::MainThreadForScrolling, WheelEventProcessingSteps::MainThreadForBlockingDOMEventDispatch };
125#if ENABLE(SCROLLING_THREAD)
126 do {
127 Locker locker { m_scrollingTreesLock };
128
129 auto scrollingTree = m_scrollingTrees.get(pageID);
130 if (!scrollingTree) {
131 dispatchWheelEventViaMainThread(pageID, wheelEvent, processingSteps);
132 break;
133 }
134
135 // FIXME: It's pretty horrible that we're updating the back/forward state here.
136 // WebCore should always know the current state and know when it changes so the
137 // scrolling tree can be notified.
138 // We only need to do this at the beginning of the gesture.
139 if (platformWheelEvent.phase() == PlatformWheelEventPhase::Began)
140 scrollingTree->setMainFrameCanRubberBand({ canRubberBandAtTop, canRubberBandAtRight, canRubberBandAtBottom, canRubberBandAtLeft });
141
142 auto processingSteps = scrollingTree->determineWheelEventProcessing(platformWheelEvent);
143
144 scrollingTree->willProcessWheelEvent();
145
146 ScrollingThread::dispatch([scrollingTree, wheelEvent, platformWheelEvent, processingSteps, pageID, protectedThis = makeRef(*this)] {
147 if (processingSteps.contains(WheelEventProcessingSteps::MainThreadForScrolling)) {
148 scrollingTree->willSendEventToMainThread(platformWheelEvent);
149 protectedThis->dispatchWheelEventViaMainThread(pageID, wheelEvent, processingSteps);
150 scrollingTree->waitForEventToBeProcessedByMainThread(platformWheelEvent);
151 return;
152 }
153
154 auto result = scrollingTree->handleWheelEvent(platformWheelEvent, processingSteps);
155
156 if (result.needsMainThreadProcessing()) {
157 protectedThis->dispatchWheelEventViaMainThread(pageID, wheelEvent, result.steps);
158 if (result.steps.contains(WheelEventProcessingSteps::MainThreadForScrolling))
159 return;
160 }
161
162 // If we scrolled on the scrolling thread (even if we send the event to the main thread for passive event handlers)
163 // respond to the UI process that the event was handled.
164 protectedThis->sendDidReceiveEvent(pageID, wheelEvent.type(), result.wasHandled);
165 });
166 } while (false);
167#else
168 UNUSED_PARAM(canRubberBandAtLeft);
169 UNUSED_PARAM(canRubberBandAtRight);
170 UNUSED_PARAM(canRubberBandAtTop);
171 UNUSED_PARAM(canRubberBandAtBottom);
172
173 dispatchWheelEventViaMainThread(pageID, wheelEvent, processingSteps);
174#endif
175}

198void EventDispatcher::touchEvent(PageIdentifier pageID, const WebTouchEvent& touchEvent, CompletionHandler<void(bool)>&& completionHandler)
199{
200 bool updateListWasEmpty;
201 {
202 Locker locker { m_touchEventsLock };
203 updateListWasEmpty = m_touchEvents.isEmpty();
204 auto addResult = m_touchEvents.add(pageID, TouchEventQueue());
205 if (addResult.isNewEntry)
206 addResult.iterator->value.append({ touchEvent, WTFMove(completionHandler) });
207 else {
208 auto& queuedEvents = addResult.iterator->value;
209 ASSERT(!queuedEvents.isEmpty());
210 auto& lastEventAndCallback = queuedEvents.last();
211 // Coalesce touch move events.
212 if (touchEvent.type() == WebEvent::TouchMove && lastEventAndCallback.first.type() == WebEvent::TouchMove && !completionHandler && !lastEventAndCallback.second)
213 queuedEvents.last() = { touchEvent, nullptr };
214 else
215 queuedEvents.append({ touchEvent, WTFMove(completionHandler) });
216 }
217 }
218
219 if (updateListWasEmpty) {
220 RunLoop::main().dispatch([protectedThis = makeRef(*this)]() mutable {
221 protectedThis->dispatchTouchEvents();
222 });
223 }
224}

226void EventDispatcher::dispatchTouchEvents()
227{
228 TraceScope traceScope(DispatchTouchEventsStart, DispatchTouchEventsEnd);
229
230 HashMap<PageIdentifier, TouchEventQueue> localCopy;
231 {
232 Locker locker { m_touchEventsLock };
233 localCopy.swap(m_touchEvents);
234 }
235
236 for (auto& slot : localCopy) {
237 if (auto* webPage = WebProcess::singleton().webPage(slot.key))
238 webPage->dispatchAsynchronousTouchEvents(WTFMove(slot.value));
239 }
240}

243void EventDispatcher::dispatchWheelEventViaMainThread(WebCore::PageIdentifier pageID, const WebWheelEvent& wheelEvent, OptionSet<WheelEventProcessingSteps> processingSteps)
244{
245 ASSERT(!RunLoop::isMain());
246 RunLoop::main().dispatch([protectedThis = makeRef(*this), pageID, wheelEvent, steps = processingSteps - WheelEventProcessingSteps::ScrollingThread]() mutable {
247 protectedThis->dispatchWheelEvent(pageID, wheelEvent, steps);
248 });
249}
250
251void EventDispatcher::dispatchWheelEvent(PageIdentifier pageID, const WebWheelEvent& wheelEvent, OptionSet<WheelEventProcessingSteps> processingSteps)
252{
253 ASSERT(RunLoop::isMain());
254
255 WebPage* webPage = WebProcess::singleton().webPage(pageID);
256 if (!webPage)
257 return;
258
259 webPage->wheelEvent(wheelEvent, processingSteps);
260}

282void EventDispatcher::notifyScrollingTreesDisplayWasRefreshed(PlatformDisplayID displayID)
283{
284#if ENABLE(SCROLLING_THREAD)
285 Locker locker { m_scrollingTreesLock };
286 for (auto keyValuePair : m_scrollingTrees)
287 keyValuePair.value->displayDidRefresh(displayID);
288#endif
289}

292void EventDispatcher::displayWasRefreshed(PlatformDisplayID displayID, const DisplayUpdate& displayUpdate, bool sendToMainThread)
293{
294 tracePoint(DisplayRefreshDispatchingToMainThread, displayID, sendToMainThread);
295
296 ASSERT(!RunLoop::isMain());
297 notifyScrollingTreesDisplayWasRefreshed(displayID);
298
299 if (!sendToMainThread)
300 return;
301
302 RunLoop::main().dispatch([displayID, displayUpdate]() {
