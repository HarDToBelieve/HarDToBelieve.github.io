59WebCore::FindOptions core(OptionSet<FindOptions> options)
60{
61 WebCore::FindOptions result;
62 if (options.contains(FindOptions::CaseInsensitive))
63 result.add(WebCore::CaseInsensitive);
64 if (options.contains(FindOptions::AtWordStarts))
65 result.add(WebCore::AtWordStarts);
66 if (options.contains(FindOptions::TreatMedialCapitalAsWordStart))
67 result.add(WebCore::TreatMedialCapitalAsWordStart);
68 if (options.contains(FindOptions::Backwards))
69 result.add(WebCore::Backwards);
70 if (options.contains(FindOptions::WrapAround))
71 result.add(WebCore::WrapAround);
72 return result;
73}

84void FindController::countStringMatches(const String& string, OptionSet<FindOptions> options, unsigned maxMatchCount)
85{
86 if (maxMatchCount == std::numeric_limits<unsigned>::max())
87 --maxMatchCount;
88
89 auto* pluginView = WebPage::pluginViewForFrame(m_webPage->mainFrame());
90
91 unsigned matchCount;
92 if (pluginView)
93 matchCount = pluginView->countFindMatches(string, core(options), maxMatchCount + 1);
94 else {
95 matchCount = m_webPage->corePage()->countFindMatches(string, core(options), maxMatchCount + 1);
96 m_webPage->corePage()->unmarkAllTextMatches();
97 }
98
99 if (matchCount > maxMatchCount)
100 matchCount = static_cast<unsigned>(kWKMoreThanMaximumMatchCount);
101
102 m_webPage->send(Messages::WebPageProxy::DidCountStringMatches(string, matchCount));
103}

105uint32_t FindController::replaceMatches(const Vector<uint32_t>& matchIndices, const String& replacementText, bool selectionOnly)
106{
107 if (matchIndices.isEmpty())
108 return m_webPage->corePage()->replaceSelectionWithText(replacementText);
109
110 // FIXME: This is an arbitrary cap on the maximum number of matches to try and replace, to prevent the web process from
111 // hanging while replacing an enormous amount of matches. In the future, we should handle replacement in batches, and
112 // periodically update an NSProgress in the UI process when a batch of find-in-page matches are replaced.
113 const uint32_t maximumNumberOfMatchesToReplace = 1000;
114
115 Vector<SimpleRange> rangesToReplace;
116 rangesToReplace.reserveCapacity(std::min<uint32_t>(maximumNumberOfMatchesToReplace, matchIndices.size()));
117 for (auto index : matchIndices) {
118 if (index < m_findMatches.size())
119 rangesToReplace.uncheckedAppend(m_findMatches[index]);
120 if (rangesToReplace.size() >= maximumNumberOfMatchesToReplace)
121 break;
122 }
123 return m_webPage->corePage()->replaceRangesWithText(rangesToReplace, replacementText, selectionOnly);
124}

126static Frame* frameWithSelection(Page* page)
127{
128 for (Frame* frame = &page->mainFrame(); frame; frame = frame->tree().traverseNext()) {
129 if (frame->selection().isRange())
130 return frame;
131 }
132
133 return 0;
134}

136void FindController::updateFindUIAfterPageScroll(bool found, const String& string, OptionSet<FindOptions> options, unsigned maxMatchCount, DidWrap didWrap, FindUIOriginator originator)
137{
138 Frame* selectedFrame = frameWithSelection(m_webPage->corePage());
139
140 auto* pluginView = WebPage::pluginViewForFrame(m_webPage->mainFrame());
141
142 bool shouldShowOverlay = false;
143
144 if (!found) {
145 if (!pluginView)
146 m_webPage->corePage()->unmarkAllTextMatches();
147
148 if (selectedFrame)
149 selectedFrame->selection().clear();
150
151 hideFindIndicator();
152 resetMatchIndex();
153 didFailToFindString();
154
155 m_webPage->send(Messages::WebPageProxy::DidFailToFindString(string));
156 } else {
157 shouldShowOverlay = options.contains(FindOptions::ShowOverlay);
158 bool shouldShowHighlight = options.contains(FindOptions::ShowHighlight);
159 bool shouldDetermineMatchIndex = options.contains(FindOptions::DetermineMatchIndex);
160 unsigned matchCount = 1;
161
162 if (shouldDetermineMatchIndex) {
163 if (pluginView)
164 matchCount = pluginView->countFindMatches(string, core(options), maxMatchCount + 1);
165 else
166 matchCount = m_webPage->corePage()->countFindMatches(string, core(options), maxMatchCount + 1);
167 }
168
169 if (shouldShowOverlay || shouldShowHighlight) {
170 if (maxMatchCount == std::numeric_limits<unsigned>::max())
171 --maxMatchCount;
172
173 if (pluginView) {
174 if (!shouldDetermineMatchIndex)
175 matchCount = pluginView->countFindMatches(string, core(options), maxMatchCount + 1);
176 shouldShowOverlay = false;
177 } else {
178 m_webPage->corePage()->unmarkAllTextMatches();
179 matchCount = m_webPage->corePage()->markAllMatchesForText(string, core(options), shouldShowHighlight, maxMatchCount + 1);
180 }
181
182 // If we have a large number of matches, we don't want to take the time to paint the overlay.
183 if (matchCount > maxMatchCount) {
184 shouldShowOverlay = false;
185 matchCount = static_cast<unsigned>(kWKMoreThanMaximumMatchCount);
186 }
187 }
188 if (matchCount == static_cast<unsigned>(kWKMoreThanMaximumMatchCount))
189 m_foundStringMatchIndex = -1;
190 else {
191 if (m_foundStringMatchIndex < 0)
192 m_foundStringMatchIndex += matchCount; // FIXME: Shouldn't this just be "="? Why is it correct to add to -1 here?
193 if (m_foundStringMatchIndex >= (int) matchCount)
194 m_foundStringMatchIndex -= matchCount;
195 }
196
197 // If updating UI after finding an individual match, update the current
198 // match rects and inform the UI process that we succeeded.
199 // If we're doing a multi-result search and just updating the indicator,
200 // this would blow away the results for the other matches.
201 // FIXME: This whole class needs a much clearer division between these two paths.
202 if (originator == FindUIOriginator::FindString) {
203 m_findMatches.clear();
204 Vector<IntRect> matchRects;
205 if (auto range = m_webPage->corePage()->selection().firstRange()) {
206 matchRects = RenderObject::absoluteTextRects(*range);
207 m_findMatches.append(*range);
208 }
209 m_webPage->send(Messages::WebPageProxy::DidFindString(string, matchRects, matchCount, m_foundStringMatchIndex, didWrap == DidWrap::Yes));
210 }
211 }
212
213 if (!shouldShowOverlay) {
214 if (m_findPageOverlay)
215 m_webPage->corePage()->pageOverlayController().uninstallPageOverlay(*m_findPageOverlay, PageOverlay::FadeMode::Fade);
216 } else {
217 if (!m_findPageOverlay) {
218 auto findPageOverlay = PageOverlay::create(*this, PageOverlay::OverlayType::Document);
219 m_findPageOverlay = findPageOverlay.ptr();
220 m_webPage->corePage()->pageOverlayController().installPageOverlay(WTFMove(findPageOverlay), PageOverlay::FadeMode::Fade);
221 }
222 m_findPageOverlay->setNeedsDisplay();
223 }
224
225 if (found && (!options.contains(FindOptions::ShowFindIndicator) || !selectedFrame || !updateFindIndicator(*selectedFrame, shouldShowOverlay)))
226 hideFindIndicator();
227}

229void FindController::findString(const String& string, OptionSet<FindOptions> options, unsigned maxMatchCount, CompletionHandler<void(bool)>&& completionHandler)
230{
231 auto* pluginView = WebPage::pluginViewForFrame(m_webPage->mainFrame());
232
233 WebCore::FindOptions coreOptions = core(options);
234
235 // iOS will reveal the selection through a different mechanism, and
236 // we need to avoid sending the non-painted selection change to the UI process
237 // so that it does not clear the selection out from under us.
238#if PLATFORM(IOS_FAMILY)
239 coreOptions.add(DoNotRevealSelection);
240#endif
241
242 willFindString();
243
244 bool foundStringStartsAfterSelection = false;
245 if (!pluginView) {
246 if (Frame* selectedFrame = frameWithSelection(m_webPage->corePage())) {
247 if (selectedFrame->selection().selectionBounds().isEmpty()) {
248 auto result = m_webPage->corePage()->findTextMatches(string, coreOptions, maxMatchCount);
249 m_findMatches = WTFMove(result.ranges);
250 m_foundStringMatchIndex = result.indexForSelection;
251 foundStringStartsAfterSelection = true;
252 }
253 }
254 }
255
256 m_findMatches.clear();
257
258 bool found;
259 DidWrap didWrap = DidWrap::No;
260 if (pluginView)
261 found = pluginView->findString(string, coreOptions, maxMatchCount);
262 else
263 found = m_webPage->corePage()->findString(string, coreOptions, &didWrap);
264
265 if (found) {
266 didFindString();
267
268 if (!foundStringStartsAfterSelection) {
269 if (options.contains(FindOptions::Backwards))
270 m_foundStringMatchIndex--;
271 else if (!options.contains(FindOptions::NoIndexChange))
272 m_foundStringMatchIndex++;
273 }
274 }
275
276 RefPtr<WebPage> protectedWebPage = m_webPage;
277 m_webPage->drawingArea()->dispatchAfterEnsuringUpdatedScrollPosition([protectedWebPage, found, string, options, maxMatchCount, didWrap] () {
278 protectedWebPage->findController().updateFindUIAfterPageScroll(found, string, options, maxMatchCount, didWrap, FindUIOriginator::FindString);
279 });
280
281 completionHandler(found);
282}

284void FindController::findStringMatches(const String& string, OptionSet<FindOptions> options, unsigned maxMatchCount)
285{
286 auto result = m_webPage->corePage()->findTextMatches(string, core(options), maxMatchCount);
287 m_findMatches = WTFMove(result.ranges);
288
289 Vector<Vector<IntRect>> matchRects;
290 for (auto& range : m_findMatches)
291 matchRects.append(RenderObject::absoluteTextRects(range));
292
293 m_webPage->send(Messages::WebPageProxy::DidFindStringMatches(string, matchRects, result.indexForSelection));
294
295 if (!options.contains(FindOptions::ShowOverlay) && !options.contains(FindOptions::ShowFindIndicator))
296 return;
297
298 bool found = !m_findMatches.isEmpty();
299 m_webPage->drawingArea()->dispatchAfterEnsuringUpdatedScrollPosition([protectedWebPage = makeRefPtr(m_webPage), found, string, options, maxMatchCount] () {
300 protectedWebPage->findController().updateFindUIAfterPageScroll(found, string, options, maxMatchCount, DidWrap::No, FindUIOriginator::FindStringMatches);
301 });
302}

304void FindController::getImageForFindMatch(uint32_t matchIndex)
305{
306 if (matchIndex >= m_findMatches.size())
307 return;
308 Frame* frame = m_findMatches[matchIndex].start.document().frame();
309 if (!frame)
310 return;
311
312 VisibleSelection oldSelection = frame->selection().selection();
313 frame->selection().setSelection(m_findMatches[matchIndex]);
314
315 RefPtr<ShareableBitmap> selectionSnapshot = WebFrame::fromCoreFrame(*frame)->createSelectionSnapshot();
316
317 frame->selection().setSelection(oldSelection);
318
319 if (!selectionSnapshot)
320 return;
321
322 ShareableBitmap::Handle handle;
323 selectionSnapshot->createHandle(handle);
324
325 if (handle.isNull())
326 return;
327
328 m_webPage->send(Messages::WebPageProxy::DidGetImageForFindMatch(handle, matchIndex));
329#if USE(DIRECT2D)
330 // Don't destroy the shared handle in the WebContent process. It will be destroyed in the UIProcess.
331 selectionSnapshot->leakSharedResource();
332#endif
333}

345void FindController::indicateFindMatch(uint32_t matchIndex)
346{
347 selectFindMatch(matchIndex);
348
349 Frame* selectedFrame = frameWithSelection(m_webPage->corePage());
350 if (!selectedFrame)
351 return;
352
353 selectedFrame->selection().revealSelection();
354
355 updateFindIndicator(*selectedFrame, !!m_findPageOverlay);
356}

358void FindController::hideFindUI()
359{
360 m_findMatches.clear();
361 if (m_findPageOverlay)
362 m_webPage->corePage()->pageOverlayController().uninstallPageOverlay(*m_findPageOverlay, PageOverlay::FadeMode::Fade);
363
364 if (auto* pluginView = WebPage::pluginViewForFrame(m_webPage->mainFrame()))
365 pluginView->findString(emptyString(), { }, 0);
366 else
367 m_webPage->corePage()->unmarkAllTextMatches();
368
369 hideFindIndicator();
370 resetMatchIndex();
371}

375bool FindController::updateFindIndicator(Frame& selectedFrame, bool isShowingOverlay, bool shouldAnimate)
376{
377 OptionSet<TextIndicatorOption> textIndicatorOptions { TextIndicatorOption::IncludeMarginIfRangeMatchesSelection };
378 if (auto selectedRange = selectedFrame.selection().selection().range(); selectedRange && HTMLElement::isInsideImageOverlay(*selectedRange))
379 textIndicatorOptions.add({ TextIndicatorOption::PaintAllContent, TextIndicatorOption::PaintBackgrounds });
380
381 auto indicator = TextIndicator::createWithSelectionInFrame(selectedFrame, textIndicatorOptions, shouldAnimate ? TextIndicatorPresentationTransition::Bounce : TextIndicatorPresentationTransition::None);
382 if (!indicator)
383 return false;
384
385 m_findIndicatorRect = enclosingIntRect(indicator->selectionRectInRootViewCoordinates());
386#if PLATFORM(COCOA)
387 m_webPage->send(Messages::WebPageProxy::SetTextIndicator(indicator->data(), static_cast<uint64_t>(isShowingOverlay ? WebCore::TextIndicatorLifetime::Permanent : WebCore::TextIndicatorLifetime::Temporary)));
388#endif
389 m_isShowingFindIndicator = true;
390
391 return true;

466Vector<FloatRect> FindController::rectsForTextMatchesInRect(IntRect clipRect)
467{
468 Vector<FloatRect> rects;
469
470 FrameView* mainFrameView = m_webPage->corePage()->mainFrame().view();
471
472 for (Frame* frame = &m_webPage->corePage()->mainFrame(); frame; frame = frame->tree().traverseNext()) {
473 Document* document = frame->document();
474 if (!document)
475 continue;
476
477 for (FloatRect rect : document->markers().renderedRectsForMarkers(DocumentMarker::TextMatch)) {
478 if (!frame->isMainFrame())
479 rect = mainFrameView->windowToContents(frame->view()->contentsToWindow(enclosingIntRect(rect)));
480
481 if (rect.isEmpty() || !rect.intersects(clipRect))
482 continue;
483
484 rects.append(rect);
485 }
486 }
487
488 return rects;
489}

504const float shadowOffsetX = 0;
505const float shadowOffsetY = 0;
506const float shadowBlurRadius = 1;
507
508void FindController::drawRect(PageOverlay&, GraphicsContext& graphicsContext, const IntRect& dirtyRect)
509{
510 const int borderWidth = 1;
511
512 constexpr auto overlayBackgroundColor = SRGBA<uint8_t> { 26, 26, 26, 64 };
513 constexpr auto shadowColor = Color::black.colorWithAlphaByte(128);
514
515 IntRect borderInflatedDirtyRect = dirtyRect;
516 borderInflatedDirtyRect.inflate(borderWidth);
517 Vector<FloatRect> rects = rectsForTextMatchesInRect(borderInflatedDirtyRect);
518
519 // Draw the background.
520 graphicsContext.fillRect(dirtyRect, overlayBackgroundColor);
521
522 Vector<Path> whiteFramePaths = PathUtilities::pathsWithShrinkWrappedRects(rects, findIndicatorRadius());
523
524 GraphicsContextStateSaver stateSaver(graphicsContext);
525
526 // Draw white frames around the holes.
527 // We double the thickness because half of the stroke will be erased when we clear the holes.
528 graphicsContext.setShadow(FloatSize(shadowOffsetX, shadowOffsetY), shadowBlurRadius, shadowColor);
529 graphicsContext.setStrokeColor(Color::white);
530 graphicsContext.setStrokeThickness(borderWidth * 2);
531 for (auto& path : whiteFramePaths)
532 graphicsContext.strokePath(path);
533
534 graphicsContext.clearShadow();
535
536 // Clear out the holes.
537 graphicsContext.setCompositeOperation(CompositeOperator::Clear);
538 for (auto& path : whiteFramePaths)
539 graphicsContext.fillPath(path);
540
541 if (!m_isShowingFindIndicator)
542 return;
543
544 if (Frame* selectedFrame = frameWithSelection(m_webPage->corePage())) {
545 IntRect findIndicatorRect = selectedFrame->view()->contentsToRootView(enclosingIntRect(selectedFrame->selection().selectionBounds(FrameSelection::ClipToVisibleContent::No)));
546
547 if (findIndicatorRect != m_findIndicatorRect) {
548 // We are underneath painting, so it's not safe to mutate the layer tree synchronously.
549 callOnMainRunLoop([weakWebPage = makeWeakPtr(m_webPage)] {
550 if (!weakWebPage)
551 return;
552 weakWebPage->findController().didScrollAffectingFindIndicatorPosition();
553 });
554 }
555 }
556}
