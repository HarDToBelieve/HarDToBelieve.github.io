38GraphicsLayerCARemote::GraphicsLayerCARemote(Type layerType, GraphicsLayerClient& client, RemoteLayerTreeContext& context)
39 : GraphicsLayerCA(layerType, client)
40 , m_context(&context)
41{
42 context.graphicsLayerDidEnterContext(*this);
43}
44
45GraphicsLayerCARemote::~GraphicsLayerCARemote()
46{
47 if (m_context)
48 m_context->graphicsLayerWillLeaveContext(*this);
49}
50
51bool GraphicsLayerCARemote::filtersCanBeComposited(const FilterOperations& filters)
52{
53 return PlatformCALayerRemote::filtersCanBeComposited(filters);
54}
55
56Ref<PlatformCALayer> GraphicsLayerCARemote::createPlatformCALayer(PlatformCALayer::LayerType layerType, PlatformCALayerClient* owner)
57{
58 auto result = PlatformCALayerRemote::create(layerType, owner, *m_context);
59
60 if (result->canHaveBackingStore())
61 result->setWantsDeepColorBackingStore(screenSupportsExtendedColor());
62
63 return WTFMove(result);
64}
65
66Ref<PlatformCALayer> GraphicsLayerCARemote::createPlatformCALayer(PlatformLayer* platformLayer, PlatformCALayerClient* owner)
67{
68 return PlatformCALayerRemote::create(platformLayer, owner, *m_context);
69}

78Ref<PlatformCAAnimation> GraphicsLayerCARemote::createPlatformCAAnimation(PlatformCAAnimation::AnimationType type, const String& keyPath)

81}
82
83void GraphicsLayerCARemote::moveToContext(RemoteLayerTreeContext& context)
84{
85 if (m_context)
86 m_context->graphicsLayerWillLeaveContext(*this);
87
88 m_context = &context;
89
90 context.graphicsLayerDidEnterContext(*this);
91}
