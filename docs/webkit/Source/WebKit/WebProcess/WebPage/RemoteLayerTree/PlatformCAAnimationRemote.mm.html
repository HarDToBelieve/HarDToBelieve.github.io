44static MonotonicTime mediaTimeToCurrentTime(CFTimeInterval t)
45{
46 return WTF::MonotonicTime::now() + Seconds(t - CACurrentMediaTime());
47}

49static NSString * const WKExplicitBeginTimeFlag = @"WKPlatformCAAnimationExplicitBeginTimeFlag";

56- (instancetype)initWithLayerID:(WebCore::GraphicsLayer::PlatformLayerID)layerID layerTreeHost:(WebKit::RemoteLayerTreeHost*)layerTreeHost
57{
58 if ((self = [super init])) {
59 _layerID = layerID;
60 _layerTreeHost = layerTreeHost;
61 }
62 return self;
63}
64
65- (void)invalidate
66{
67 _layerTreeHost = nullptr;
68}
69
70- (void)animationDidStart:(CAAnimation *)animation
71{
72 if (!_layerTreeHost)
73 return;
74
75 bool hasExplicitBeginTime = [[animation valueForKey:WKExplicitBeginTimeFlag] boolValue];
76 MonotonicTime startTime;
77
78 if (hasExplicitBeginTime) {
79 // We don't know what time CA used to commit the animation, so just use the current time
80 // (even though this will be slightly off).
81 startTime = mediaTimeToCurrentTime(CACurrentMediaTime());
82 } else
83 startTime = mediaTimeToCurrentTime([animation beginTime]);
84
85 _layerTreeHost->animationDidStart(_layerID, animation, startTime);
86}
87
88- (void)animationDidStop:(CAAnimation *)animation finished:(BOOL)finished
89{
90 if (!_layerTreeHost)
91 return;
92
93 _layerTreeHost->animationDidEnd(_layerID, animation);
94}

96@end

101static void encodeTimingFunction(IPC::Encoder& encoder, TimingFunction* timingFunction)
102{
103 switch (timingFunction->type()) {
104 case TimingFunction::LinearFunction:
105 encoder << *static_cast<LinearTimingFunction*>(timingFunction);
106 break;
107
108 case TimingFunction::CubicBezierFunction:
109 encoder << *static_cast<CubicBezierTimingFunction*>(timingFunction);
110 break;
111
112 case TimingFunction::StepsFunction:
113 encoder << *static_cast<StepsTimingFunction*>(timingFunction);
114 break;
115
116 case TimingFunction::SpringFunction:
117 encoder << *static_cast<SpringTimingFunction*>(timingFunction);
118 break;
119 }
120}
121
122static std::optional<RefPtr<TimingFunction>> decodeTimingFunction(IPC::Decoder& decoder)
123{
124 TimingFunction::TimingFunctionType type;
125 if (!decoder.decode(type))
126 return std::nullopt;
127
128 RefPtr<TimingFunction> timingFunction;
129 switch (type) {
130 case TimingFunction::LinearFunction:
131 timingFunction = LinearTimingFunction::create();
132 if (!decoder.decode(*static_cast<LinearTimingFunction*>(timingFunction.get())))
133 return std::nullopt;
134 break;
135
136 case TimingFunction::CubicBezierFunction:
137 timingFunction = CubicBezierTimingFunction::create();
138 if (!decoder.decode(*static_cast<CubicBezierTimingFunction*>(timingFunction.get())))
139 return std::nullopt;
140 break;
141
142 case TimingFunction::StepsFunction:
143 timingFunction = StepsTimingFunction::create();
144 if (!decoder.decode(*static_cast<StepsTimingFunction*>(timingFunction.get())))
145 return std::nullopt;
146 break;
147
148 case TimingFunction::SpringFunction:
149 timingFunction = SpringTimingFunction::create();
150 if (!decoder.decode(*static_cast<SpringTimingFunction*>(timingFunction.get())))
151 return std::nullopt;
152 break;
153 }
154
155 return timingFunction;
156}

158void PlatformCAAnimationRemote::Properties::encode(IPC::Encoder& encoder) const
159{
160 encoder << keyPath;
161 encoder << animationType;
162
163 encoder << beginTime;
164 encoder << duration;
165 encoder << timeOffset;
166 encoder << repeatCount;
167 encoder << speed;
168
169 encoder << fillMode;
170 encoder << valueFunction;
171
172 bool hasTimingFunction = !!timingFunction;
173 encoder << hasTimingFunction;
174 if (hasTimingFunction)
175 encodeTimingFunction(encoder, timingFunction.get());
176
177 encoder << autoReverses;
178 encoder << removedOnCompletion;
179 encoder << additive;
180 encoder << reverseTimingFunctions;
181 encoder << hasExplicitBeginTime;
182
183 encoder << keyValues;
184 encoder << keyTimes;
185
186 encoder << static_cast<uint64_t>(timingFunctions.size());
187 for (const auto& timingFunction : timingFunctions)
188 encodeTimingFunction(encoder, timingFunction.get());
189
190 encoder << animations;
191}

193std::optional<PlatformCAAnimationRemote::Properties> PlatformCAAnimationRemote::Properties::decode(IPC::Decoder& decoder)
194{
195 PlatformCAAnimationRemote::Properties properties;
196 if (!decoder.decode(properties.keyPath))
197 return std::nullopt;
198
199 if (!decoder.decode(properties.animationType))
200 return std::nullopt;
201
202 if (!decoder.decode(properties.beginTime))
203 return std::nullopt;
204
205 if (!decoder.decode(properties.duration))
206 return std::nullopt;
207
208 if (!decoder.decode(properties.timeOffset))
209 return std::nullopt;
210
211 if (!decoder.decode(properties.repeatCount))
212 return std::nullopt;
213
214 if (!decoder.decode(properties.speed))
215 return std::nullopt;
216
217 if (!decoder.decode(properties.fillMode))
218 return std::nullopt;
219
220 if (!decoder.decode(properties.valueFunction))
221 return std::nullopt;
222
223 bool hasTimingFunction;
224 if (!decoder.decode(hasTimingFunction))
225 return std::nullopt;
226
227 if (hasTimingFunction) {
228 if (auto timingFunction = decodeTimingFunction(decoder))
229 properties.timingFunction = WTFMove(*timingFunction);
230 else
231 return std::nullopt;
232 }
233
234 if (!decoder.decode(properties.autoReverses))
235 return std::nullopt;
236
237 if (!decoder.decode(properties.removedOnCompletion))
238 return std::nullopt;
239
240 if (!decoder.decode(properties.additive))
241 return std::nullopt;
242
243 if (!decoder.decode(properties.reverseTimingFunctions))
244 return std::nullopt;
245
246 if (!decoder.decode(properties.hasExplicitBeginTime))
247 return std::nullopt;
248
249 if (!decoder.decode(properties.keyValues))
250 return std::nullopt;
251
252 if (!decoder.decode(properties.keyTimes))
253 return std::nullopt;
254
255 uint64_t numTimingFunctions;
256 if (!decoder.decode(numTimingFunctions))
257 return std::nullopt;
258
259 if (numTimingFunctions) {
260 properties.timingFunctions.reserveInitialCapacity(numTimingFunctions);
261
262 for (size_t i = 0; i < numTimingFunctions; ++i) {
263 if (auto timingFunction = decodeTimingFunction(decoder))
264 properties.timingFunctions.uncheckedAppend(WTFMove(*timingFunction));
265 else
266 return std::nullopt;
267 }
268 }
269
270 if (!decoder.decode(properties.animations))
271 return std::nullopt;
272
273 return WTFMove(properties);
274}

281Ref<PlatformCAAnimation> PlatformCAAnimationRemote::copy() const
282{
283 auto animation = create(animationType(), keyPath());
284
285 animation->setBeginTime(beginTime());
286 animation->setDuration(duration());
287 animation->setSpeed(speed());
288 animation->setTimeOffset(timeOffset());
289 animation->setRepeatCount(repeatCount());
290 animation->setAutoreverses(autoreverses());
291 animation->setFillMode(fillMode());
292 animation->setRemovedOnCompletion(isRemovedOnCompletion());
293 animation->setAdditive(isAdditive());
294 animation->copyTimingFunctionFrom(*this);
295 animation->setValueFunction(valueFunction());
296
297 downcast<PlatformCAAnimationRemote>(animation.get()).setHasExplicitBeginTime(hasExplicitBeginTime());
298
299 // Copy the specific Basic or Keyframe values.
300 if (animationType() == Keyframe) {
301 animation->copyValuesFrom(*this);
302 animation->copyKeyTimesFrom(*this);
303 animation->copyTimingFunctionsFrom(*this);
304 } else {
305 animation->copyFromValueFrom(*this);
306 animation->copyToValueFrom(*this);
307 }
308
309 animation->copyAnimationsFrom(*this);
310
311 return animation;
312}

679static RetainPtr<NSObject> animationValueFromKeyframeValue(const PlatformCAAnimationRemote::KeyframeValue& keyframeValue)
680{
681 return WTF::switchOn(keyframeValue,
682 [&](const float number) -> RetainPtr<NSObject> { return @(number); },
683 [&](const WebCore::Color color) -> RetainPtr<NSObject> {
684 auto [r, g, b, a] = color.toSRGBALossy<uint8_t>();
685 return @[ @(r), @(g), @(b), @(a) ];
686 },
687 [&](const WebCore::FloatPoint3D point) -> RetainPtr<NSObject> {
688 return @[ @(point.x()), @(point.y()), @(point.z()) ];
689 },
690 [&](const WebCore::TransformationMatrix matrix) -> RetainPtr<NSObject> {
691 return [NSValue valueWithCATransform3D:matrix];
692 },
693 [&](const RefPtr<WebCore::FilterOperation> filter) -> RetainPtr<NSObject> {
694 return PlatformCAFilters::filterValueForOperation(filter.get(), 0 /* unused */);
695 }
696 );
697}

699static RetainPtr<CAAnimation> createAnimation(CALayer *layer, RemoteLayerTreeHost* layerTreeHost, const PlatformCAAnimationRemote::Properties& properties)
700{
701 RetainPtr<CAAnimation> caAnimation;
702 switch (properties.animationType) {
703 case PlatformCAAnimation::Basic: {
704 auto basicAnimation = [CABasicAnimation animationWithKeyPath:properties.keyPath];
705
706 if (properties.keyValues.size() > 1) {
707 [basicAnimation setFromValue:animationValueFromKeyframeValue(properties.keyValues[0]).get()];
708 [basicAnimation setToValue:animationValueFromKeyframeValue(properties.keyValues[1]).get()];
709 }
710
711 if (properties.timingFunctions.size())
712 [basicAnimation setTimingFunction:toCAMediaTimingFunction(properties.timingFunctions[0].get(), properties.reverseTimingFunctions)];
713
714 caAnimation = basicAnimation;
715 break;
716 }
717 case PlatformCAAnimation::Group: {
718 auto animationGroup = [CAAnimationGroup animation];
719
720 if (properties.animations.size()) {
721 [animationGroup setAnimations:createNSArray(properties.animations, [&] (auto& animationProperties) {
722 return createAnimation(layer, layerTreeHost, animationProperties).get();
723 }).get()];
724 }
725
726 caAnimation = animationGroup;
727 break;
728 }
729 case PlatformCAAnimation::Keyframe: {
730 auto keyframeAnimation = [CAKeyframeAnimation animationWithKeyPath:properties.keyPath];
731
732 if (properties.keyValues.size()) {
733 [keyframeAnimation setValues:createNSArray(properties.keyValues, [] (auto& value) {
734 return animationValueFromKeyframeValue(value);
735 }).get()];
736 }
737
738 if (properties.keyTimes.size()) {
739 [keyframeAnimation setKeyTimes:createNSArray(properties.keyTimes, [] (auto& time) {
740 return @(time);
741 }).get()];
742 }
743
744 if (properties.timingFunction)
745 [keyframeAnimation setTimingFunction:toCAMediaTimingFunction(properties.timingFunction.get(), false)]; // FIXME: handle reverse.
746
747 if (properties.timingFunctions.size()) {
748 [keyframeAnimation setTimingFunctions:createNSArray(properties.timingFunctions, [&] (auto& function) {
749 return toCAMediaTimingFunction(function.get(), properties.reverseTimingFunctions);
750 }).get()];
751 }
752
753 caAnimation = keyframeAnimation;
754 break;
755 }
756 case PlatformCAAnimation::Spring: {
757 auto springAnimation = [CASpringAnimation animationWithKeyPath:properties.keyPath];
758
759 if (properties.keyValues.size() > 1) {
760 [springAnimation setFromValue:animationValueFromKeyframeValue(properties.keyValues[0]).get()];
761 [springAnimation setToValue:animationValueFromKeyframeValue(properties.keyValues[1]).get()];
762 }
763
764 if (properties.timingFunctions.size()) {
765 auto& timingFunction = properties.timingFunctions[0];
766 if (timingFunction->isSpringTimingFunction()) {
767 auto& function = *static_cast<const SpringTimingFunction*>(timingFunction.get());
768 [springAnimation setMass:function.mass()];
769 [springAnimation setStiffness:function.stiffness()];
770 [springAnimation setDamping:function.damping()];
771 [springAnimation setInitialVelocity:function.initialVelocity()];
772 }
773 }
774 caAnimation = springAnimation;
775 break;
776 }
777 }
778
779 [caAnimation setBeginTime:properties.beginTime];
780 [caAnimation setDuration:properties.duration];
781 [caAnimation setTimeOffset:properties.timeOffset];
782 [caAnimation setRepeatCount:properties.repeatCount];
783 [caAnimation setSpeed:properties.speed];
784 [caAnimation setAutoreverses:properties.autoReverses];
785 [caAnimation setRemovedOnCompletion:properties.removedOnCompletion];
786
787 if ([caAnimation isKindOfClass:[CAPropertyAnimation class]]) {
788 [(CAPropertyAnimation *)caAnimation setAdditive:properties.additive];
789 if (properties.valueFunction != PlatformCAAnimation::NoValueFunction)
790 [(CAPropertyAnimation *)caAnimation setValueFunction:[CAValueFunction functionWithName:toCAValueFunctionType(properties.valueFunction)]];
791 }
792
793 if (properties.fillMode != PlatformCAAnimation::NoFillMode)
794 [caAnimation setFillMode:toCAFillModeType(properties.fillMode)];
795
796 if (properties.hasExplicitBeginTime)
797 [caAnimation setValue:@YES forKey:WKExplicitBeginTimeFlag];
798
799 if (layerTreeHost) {
800 GraphicsLayer::PlatformLayerID layerID = RemoteLayerTreeNode::layerID(layer);
801
802 RetainPtr<WKAnimationDelegate>& delegate = layerTreeHost->animationDelegates().add(layerID, nullptr).iterator->value;
803 if (!delegate)
804 delegate = adoptNS([[WKAnimationDelegate alloc] initWithLayerID:layerID layerTreeHost:layerTreeHost]);
805
806 [caAnimation setDelegate:delegate.get()];
807 }

809 return caAnimation;

906 ts.decreaseIndent();

909 ts << "\n";
910 ts.increaseIndent();
911 ts.writeIndent();
