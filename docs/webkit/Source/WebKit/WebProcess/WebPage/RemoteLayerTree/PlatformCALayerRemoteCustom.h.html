28#include "PlatformCALayerRemote.h"

30namespace WebKit {
31
32class LayerHostingContext;
33
34// PlatformCALayerRemoteCustom is used for CALayers that live in the web process and are hosted into the UI process via remote context.
35class PlatformCALayerRemoteCustom final : public PlatformCALayerRemote {
36 friend class PlatformCALayerRemote;
37public:
38 static Ref<PlatformCALayerRemote> create(PlatformLayer *, WebCore::PlatformCALayerClient*, RemoteLayerTreeContext&);
39
40 virtual ~PlatformCALayerRemoteCustom();
41
42 PlatformLayer* platformLayer() const override { return m_platformLayer.get(); }
43
44 uint32_t hostingContextID() override;
45
46 void setNeedsDisplayInRect(const WebCore::FloatRect& dirtyRect) override;
47 void setNeedsDisplay() override;
48
49private:
50 PlatformCALayerRemoteCustom(WebCore::PlatformCALayer::LayerType, PlatformLayer *, WebCore::PlatformCALayerClient* owner, RemoteLayerTreeContext&);
51
52 Ref<WebCore::PlatformCALayer> clone(WebCore::PlatformCALayerClient* owner) const override;
53
54 void populateCreationProperties(RemoteLayerTreeTransaction::LayerCreationProperties&, const RemoteLayerTreeContext&, WebCore::PlatformCALayer::LayerType) override;
55
56 bool isPlatformCALayerRemoteCustom() const override { return true; }
57
58 CFTypeRef contents() const override;
59 void setContents(CFTypeRef) override;
60
61 std::unique_ptr<LayerHostingContext> m_layerHostingContext;
62 RetainPtr<PlatformLayer> m_platformLayer;
63 bool m_providesContents;
64};
65
66} // namespace WebKit
