32namespace WebKit {
33using namespace WebCore;
34
35constexpr FramesPerSecond DefaultPreferredFramesPerSecond = 60;
36
37RemoteLayerTreeDisplayRefreshMonitor::RemoteLayerTreeDisplayRefreshMonitor(PlatformDisplayID displayID, RemoteLayerTreeDrawingArea& drawingArea)
38 : DisplayRefreshMonitor(displayID)
39 , m_drawingArea(makeWeakPtr(drawingArea))
40 , m_preferredFramesPerSecond(DefaultPreferredFramesPerSecond)
41 , m_currentUpdate({ 0, m_preferredFramesPerSecond })
42{
43}
44
45RemoteLayerTreeDisplayRefreshMonitor::~RemoteLayerTreeDisplayRefreshMonitor()
46{
47 if (m_drawingArea)
48 m_drawingArea->willDestroyDisplayRefreshMonitor(this);
49}
50
51void RemoteLayerTreeDisplayRefreshMonitor::adjustPreferredFramesPerSecond(FramesPerSecond preferredFramesPerSecond)
52{
53 if (preferredFramesPerSecond == m_preferredFramesPerSecond)
54 return;
55
56 LOG_WITH_STREAM(DisplayLink, stream << "RemoteLayerTreeDisplayRefreshMonitor::adjustMaxPreferredFramesPerSecond to " << preferredFramesPerSecond);
57
58 m_preferredFramesPerSecond = preferredFramesPerSecond;
59 m_currentUpdate = { 0, m_preferredFramesPerSecond };
60
61 if (m_drawingArea)
62 m_drawingArea->setPreferredFramesPerSecond(preferredFramesPerSecond);
63}
64
65bool RemoteLayerTreeDisplayRefreshMonitor::requestRefreshCallback()
66{
67 if (!m_drawingArea)
68 return false;
69
70 if (!isScheduled()) {
71 LOG_WITH_STREAM(DisplayLink, stream << "RemoteLayerTreeDisplayRefreshMonitor::requestRefreshCallback - triggering update");
72 static_cast<DrawingArea&>(*m_drawingArea.get()).triggerRenderingUpdate();
73 }
74
75 setIsScheduled(true);
76 return true;
77}
78
79void RemoteLayerTreeDisplayRefreshMonitor::didUpdateLayers()
80{
81 setIsScheduled(false);
82
83 if (!isPreviousFrameDone())
84 return;
85
86 setIsPreviousFrameDone(false);
87 displayDidRefresh(m_currentUpdate);
88 m_currentUpdate = m_currentUpdate.nextUpdate();
89}
90
91void RemoteLayerTreeDisplayRefreshMonitor::updateDrawingArea(RemoteLayerTreeDrawingArea& drawingArea)
92{
93 m_drawingArea = makeWeakPtr(drawingArea);
94}
95
96std::optional<FramesPerSecond> RemoteLayerTreeDisplayRefreshMonitor::displayNominalFramesPerSecond()
97{
98 return m_preferredFramesPerSecond;
99}
