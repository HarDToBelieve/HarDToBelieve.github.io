41namespace WebKit {
42
43class WebPage;
44class RemoteScrollingCoordinatorTransaction;
45class RemoteScrollingUIState;
46
47class RemoteScrollingCoordinator : public WebCore::AsyncScrollingCoordinator, public IPC::MessageReceiver {
48public:
49 static Ref<RemoteScrollingCoordinator> create(WebPage* page)
50 {
51 return adoptRef(*new RemoteScrollingCoordinator(page));
52 }
53
54 void buildTransaction(RemoteScrollingCoordinatorTransaction&);
55
56 void scrollingStateInUIProcessChanged(const RemoteScrollingUIState&);
57
58 void addNodeWithActiveRubberBanding(WebCore::ScrollingNodeID);
59 void removeNodeWithActiveRubberBanding(WebCore::ScrollingNodeID);
60
61private:
62 RemoteScrollingCoordinator(WebPage*);
63 virtual ~RemoteScrollingCoordinator();
64
65 bool isRemoteScrollingCoordinator() const override { return true; }
66
67 // ScrollingCoordinator
68 bool coordinatesScrollingForFrameView(const WebCore::FrameView&) const override;
69 void scheduleTreeStateCommit() override;
70
71 bool isRubberBandInProgress(WebCore::ScrollingNodeID) const final;
72 bool isUserScrollInProgress(WebCore::ScrollingNodeID) const final;
73 bool isScrollSnapInProgress(WebCore::ScrollingNodeID) const final;
74
75 void setScrollPinningBehavior(WebCore::ScrollPinningBehavior) override;
76
77 // IPC::MessageReceiver
78 void didReceiveMessage(IPC::Connection&, IPC::Decoder&) override;
79
80 // Respond to UI process changes.
81 void scrollPositionChangedForNode(WebCore::ScrollingNodeID, const WebCore::FloatPoint& scrollPosition, bool syncLayerPosition);
82 void currentSnapPointIndicesChangedForNode(WebCore::ScrollingNodeID, std::optional<unsigned> horizontal, std::optional<unsigned> vertical);
83
84 WebPage* m_webPage;
85
86 HashSet<WebCore::ScrollingNodeID> m_nodesWithActiveRubberBanding;
87 HashSet<WebCore::ScrollingNodeID> m_nodesWithActiveScrollSnap;
88 HashSet<WebCore::ScrollingNodeID> m_nodesWithActiveUserScrolls;
89};
90
91} // namespace WebKit
