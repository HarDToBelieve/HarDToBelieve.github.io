52RemoteScrollingCoordinator::RemoteScrollingCoordinator(WebPage* page)
53 : AsyncScrollingCoordinator(page->corePage())
54 , m_webPage(page)
55{
56 WebProcess::singleton().addMessageReceiver(Messages::RemoteScrollingCoordinator::messageReceiverName(), m_webPage->identifier(), *this);
57}
58
59RemoteScrollingCoordinator::~RemoteScrollingCoordinator()
60{
61 WebProcess::singleton().removeMessageReceiver(Messages::RemoteScrollingCoordinator::messageReceiverName(), m_webPage->identifier());
62}
63
64void RemoteScrollingCoordinator::scheduleTreeStateCommit()
65{
66 m_webPage->drawingArea()->triggerRenderingUpdate();
67}
68
69bool RemoteScrollingCoordinator::coordinatesScrollingForFrameView(const FrameView& frameView) const
70{
71 RenderView* renderView = frameView.renderView();
72 return renderView && renderView->usesCompositing();
73}

75bool RemoteScrollingCoordinator::isRubberBandInProgress(ScrollingNodeID nodeID) const
76{
77 if (!nodeID)
78 return false;
79 return m_nodesWithActiveRubberBanding.contains(nodeID);
80}
81
82bool RemoteScrollingCoordinator::isUserScrollInProgress(ScrollingNodeID nodeID) const
83{
84 if (!nodeID)
85 return false;
86 return m_nodesWithActiveUserScrolls.contains(nodeID);
87}
88
89bool RemoteScrollingCoordinator::isScrollSnapInProgress(ScrollingNodeID nodeID) const
90{
91 if (!nodeID)
92 return false;
93 return m_nodesWithActiveScrollSnap.contains(nodeID);

101void RemoteScrollingCoordinator::buildTransaction(RemoteScrollingCoordinatorTransaction& transaction)
102{
103 willCommitTree();
104 transaction.setStateTreeToEncode(scrollingStateTree()->commit(LayerRepresentation::PlatformLayerIDRepresentation));
105}

107// Notification from the UI process that we scrolled.
108void RemoteScrollingCoordinator::scrollPositionChangedForNode(ScrollingNodeID nodeID, const FloatPoint& scrollPosition, bool syncLayerPosition)
109{
110 applyScrollUpdate(nodeID, scrollPosition, std::nullopt, ScrollType::User, syncLayerPosition ? ScrollingLayerPositionAction::Sync : ScrollingLayerPositionAction::Set);
111}
112
113void RemoteScrollingCoordinator::currentSnapPointIndicesChangedForNode(ScrollingNodeID nodeID, std::optional<unsigned> horizontal, std::optional<unsigned> vertical)
114{
115 setActiveScrollSnapIndices(nodeID, horizontal, vertical);
116}

118void RemoteScrollingCoordinator::scrollingStateInUIProcessChanged(const RemoteScrollingUIState& uiState)
119{
120 // FIXME: Also track m_nodesWithActiveRubberBanding.
121 if (uiState.changes().contains(RemoteScrollingUIState::Changes::ScrollSnapNodes))
122 m_nodesWithActiveScrollSnap = uiState.nodesWithActiveScrollSnap();
123
124 if (uiState.changes().contains(RemoteScrollingUIState::Changes::UserScrollNodes))
125 m_nodesWithActiveUserScrolls = uiState.nodesWithActiveUserScrolls();
126}

128void RemoteScrollingCoordinator::addNodeWithActiveRubberBanding(ScrollingNodeID nodeID)

131}

133void RemoteScrollingCoordinator::removeNodeWithActiveRubberBanding(ScrollingNodeID nodeID)

135 m_nodesWithActiveRubberBanding.remove(nodeID);
136}
