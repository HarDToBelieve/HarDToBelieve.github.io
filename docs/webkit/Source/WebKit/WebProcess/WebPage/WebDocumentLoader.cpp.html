31namespace WebKit {
32using namespace WebCore;
33
34WebDocumentLoader::WebDocumentLoader(const ResourceRequest& request, const SubstituteData& substituteData)
35 : DocumentLoader(request, substituteData)
36 , m_navigationID(0)
37{
38}
39
40void WebDocumentLoader::detachFromFrame()
41{
42 if (m_navigationID)
43 WebFrame::fromCoreFrame(*frame())->documentLoaderDetached(m_navigationID);
44
45 m_navigationID = 0;
46
47 DocumentLoader::detachFromFrame();
48}
49
50void WebDocumentLoader::setNavigationID(uint64_t navigationID)
51{
52 ASSERT(navigationID);
53
54 m_navigationID = navigationID;
55}
