116Ref<WebFrame> WebFrame::createSubframe(WebPage* page, const String& frameName, HTMLFrameOwnerElement* ownerElement)
117{
118 auto frame = create();
119 page->send(Messages::WebPageProxy::DidCreateSubframe(frame->frameID()));
120
121 auto coreFrame = Frame::create(page->corePage(), ownerElement, makeUniqueRef<WebFrameLoaderClient>(frame.get()));
122 frame->m_coreFrame = makeWeakPtr(coreFrame.get());
123
124 coreFrame->tree().setName(frameName);
125 if (ownerElement) {
126 ASSERT(ownerElement->document().frame());
127 ownerElement->document().frame()->tree().appendChild(coreFrame.get());
128 }
129 coreFrame->init();
130
131 return frame;
132}

238void WebFrame::invalidatePolicyListener()
239{
240 if (!m_policyListenerID)
241 return;
242
243 m_policyDownloadID = { };
244 m_policyListenerID = 0;
245 auto identifier = m_policyIdentifier;
246 m_policyIdentifier = std::nullopt;
247 if (auto function = std::exchange(m_policyFunction, nullptr))
248 function(PolicyAction::Ignore, *identifier);
249 m_policyFunctionForNavigationAction = ForNavigationAction::No;
250
251 auto willSubmitFormCompletionHandlers = WTFMove(m_willSubmitFormCompletionHandlers);
252 for (auto& completionHandler : willSubmitFormCompletionHandlers.values())
253 completionHandler();
254}

256void WebFrame::didReceivePolicyDecision(uint64_t listenerID, PolicyDecision&& policyDecision)
257{
258 if (!m_coreFrame || !m_policyListenerID || listenerID != m_policyListenerID || !m_policyFunction)
259 return;
260
261 ASSERT(policyDecision.identifier == m_policyIdentifier);
262 m_policyIdentifier = std::nullopt;
263
264 FramePolicyFunction function = WTFMove(m_policyFunction);
265 bool forNavigationAction = m_policyFunctionForNavigationAction == ForNavigationAction::Yes;
266
267 invalidatePolicyListener();
268
269 if (forNavigationAction && frameLoaderClient() && policyDecision.websitePoliciesData) {
270 ASSERT(page());
271 if (page())
272 page()->setAllowsContentJavaScriptFromMostRecentNavigation(policyDecision.websitePoliciesData->allowsContentJavaScript);
273 frameLoaderClient()->applyToDocumentLoader(WTFMove(*policyDecision.websitePoliciesData));
274 }
275
276 m_policyDownloadID = policyDecision.downloadID;
277 if (policyDecision.navigationID) {
278 if (WebDocumentLoader* documentLoader = static_cast<WebDocumentLoader*>(m_coreFrame->loader().policyDocumentLoader()))
279 documentLoader->setNavigationID(policyDecision.navigationID);
280 }
281
282 if (policyDecision.policyAction == PolicyAction::Use && policyDecision.sandboxExtensionHandle) {
283 if (auto* page = this->page())
284 page->sandboxExtensionTracker().beginLoad(&page->mainWebFrame(), WTFMove(*(policyDecision.sandboxExtensionHandle)));
285 }
286
287 function(policyDecision.policyAction, policyDecision.identifier);
288}

290void WebFrame::startDownload(const WebCore::ResourceRequest& request, const String& suggestedName)
291{
292 if (!m_policyDownloadID) {
293 ASSERT_NOT_REACHED();
294 return;
295 }
296 auto policyDownloadID = *std::exchange(m_policyDownloadID, std::nullopt);
297
298 std::optional<NavigatingToAppBoundDomain> isAppBound = NavigatingToAppBoundDomain::No;
299 isAppBound = m_isNavigatingToAppBoundDomain;
300 WebProcess::singleton().ensureNetworkProcessConnection().connection().send(Messages::NetworkConnectionToWebProcess::StartDownload(policyDownloadID, request, isAppBound, suggestedName), 0);
301}

303void WebFrame::convertMainResourceLoadToDownload(DocumentLoader* documentLoader, const ResourceRequest& request, const ResourceResponse& response)
304{
305 if (!m_policyDownloadID) {
306 ASSERT_NOT_REACHED();
307 return;
308 }
309 auto policyDownloadID = *std::exchange(m_policyDownloadID, std::nullopt);
310
311 SubresourceLoader* mainResourceLoader = documentLoader->mainResourceLoader();
312
313 auto& webProcess = WebProcess::singleton();
314 // Use 0 to indicate that the resource load can't be converted and a new download must be started.
315 // This can happen if there is no loader because the main resource is in the WebCore memory cache,
316 // or because the conversion was attempted when not calling SubresourceLoader::didReceiveResponse().
317 uint64_t mainResourceLoadIdentifier;
318 if (mainResourceLoader)
319 mainResourceLoadIdentifier = mainResourceLoader->identifier();
320 else
321 mainResourceLoadIdentifier = 0;
322
323 std::optional<NavigatingToAppBoundDomain> isAppBound = NavigatingToAppBoundDomain::No;
324 isAppBound = m_isNavigatingToAppBoundDomain;
325 webProcess.ensureNetworkProcessConnection().connection().send(Messages::NetworkConnectionToWebProcess::ConvertMainResourceLoadToDownload(mainResourceLoadIdentifier, policyDownloadID, request, response, isAppBound), 0);
326}

336String WebFrame::source() const
337{
338 if (!m_coreFrame)
339 return String();
340 Document* document = m_coreFrame->document();
341 if (!document)
342 return String();
343 TextResourceDecoder* decoder = document->decoder();
344 if (!decoder)
345 return String();
346 DocumentLoader* documentLoader = m_coreFrame->loader().activeDocumentLoader();
347 if (!documentLoader)
348 return String();
349 RefPtr<SharedBuffer> mainResourceData = documentLoader->mainResourceData();
350 if (!mainResourceData)
351 return String();
352 return decoder->encoding().decode(mainResourceData->data(), mainResourceData->size());
353}

355String WebFrame::contentsAsString() const
356{
357 if (!m_coreFrame)
358 return String();
359
360 if (isFrameSet()) {
361 StringBuilder builder;
362 for (Frame* child = m_coreFrame->tree().firstChild(); child; child = child->tree().nextSibling()) {
363 if (!builder.isEmpty())
364 builder.append(' ');
365
366 WebFrame* webFrame = WebFrame::fromCoreFrame(*child);
367 ASSERT(webFrame);
368
369 builder.append(webFrame->contentsAsString());
370 }
371 // FIXME: It may make sense to use toStringPreserveCapacity() here.
372 return builder.toString();
373 }
374
375 auto document = m_coreFrame->document();
376 if (!document)
377 return String();
378
379 auto documentElement = document->documentElement();
380 if (!documentElement)
381 return String();
382
383 return plainText(makeRangeSelectingNodeContents(*documentElement));
384}

480Ref<API::Array> WebFrame::childFrames()
481{
482 if (!m_coreFrame)
483 return API::Array::create();
484
485 size_t size = m_coreFrame->tree().childCount();
486 if (!size)
487 return API::Array::create();
488
489 Vector<RefPtr<API::Object>> vector;
490 vector.reserveInitialCapacity(size);
491
492 for (Frame* child = m_coreFrame->tree().firstChild(); child; child = child->tree().nextSibling()) {
493 WebFrame* webFrame = WebFrame::fromCoreFrame(*child);
494 ASSERT(webFrame);
495 vector.uncheckedAppend(webFrame);
496 }
497
498 return API::Array::create(WTFMove(vector));
499}

762String WebFrame::suggestedFilenameForResourceWithURL(const URL& url) const
763{
764 if (!m_coreFrame)
765 return String();
766
767 DocumentLoader* loader = m_coreFrame->loader().documentLoader();
768 if (!loader)
769 return String();
770
771 // First, try the main resource.
772 if (loader->url() == url)
773 return loader->response().suggestedFilename();
774
775 // Next, try subresources.
776 RefPtr<ArchiveResource> resource = loader->subresource(url);
777 if (resource)
778 return resource->response().suggestedFilename();
779
780 return String();
781}

843RefPtr<ShareableBitmap> WebFrame::createSelectionSnapshot() const
844{
845 auto snapshot = snapshotSelection(*coreFrame(), { { WebCore::SnapshotFlags::ForceBlackText }, PixelFormat::BGRA8, DestinationColorSpace::SRGB() });
846 if (!snapshot)
847 return nullptr;
848
849 auto sharedSnapshot = ShareableBitmap::createShareable(snapshot->backendSize(), { });
850 if (!sharedSnapshot)
851 return nullptr;
852
853 // FIXME: We should consider providing a way to use subpixel antialiasing for the snapshot
854 // if we're compositing this image onto a solid color (e.g. the modern find indicator style).
855 auto graphicsContext = sharedSnapshot->createGraphicsContext();
856 if (!graphicsContext)
857 return nullptr;
858
859 float deviceScaleFactor = coreFrame()->page()->deviceScaleFactor();
860 graphicsContext->scale(deviceScaleFactor);
861 graphicsContext->drawConsumingImageBuffer(WTFMove(snapshot), FloatPoint());
862
863 return sharedSnapshot;
864}

867bool WebFrame::shouldEnableInAppBrowserPrivacyProtections()
868{
869 if (page() && page()->needsInAppBrowserPrivacyQuirks())
870 return false;
871
872 bool treeHasNonAppBoundFrame = m_isNavigatingToAppBoundDomain && m_isNavigatingToAppBoundDomain == NavigatingToAppBoundDomain::No;
873 if (!treeHasNonAppBoundFrame) {
874 for (WebFrame* frame = this; frame; frame = frame->parentFrame()) {
875 if (frame->isNavigatingToAppBoundDomain() && frame->isNavigatingToAppBoundDomain() == NavigatingToAppBoundDomain::No) {
876 treeHasNonAppBoundFrame = true;
877 break;
878 }
879 }
880 }
881 return treeHasNonAppBoundFrame;
