46WebURLSchemeHandlerProxy::WebURLSchemeHandlerProxy(WebPage& page, uint64_t identifier)
47 : m_webPage(page)
48 , m_identifier(identifier)
49{
50}
51
52WebURLSchemeHandlerProxy::~WebURLSchemeHandlerProxy()
53{
54 ASSERT(m_tasks.isEmpty());
55}
56
57void WebURLSchemeHandlerProxy::startNewTask(ResourceLoader& loader, WebFrame& webFrame)
58{
59 auto result = m_tasks.add(loader.identifier(), WebURLSchemeTaskProxy::create(*this, loader, webFrame));
60 ASSERT(result.isNewEntry);
61
62 WebProcess::singleton().webLoaderStrategy().addURLSchemeTaskProxy(*result.iterator->value);
63 result.iterator->value->startLoading();
64}
65
66void WebURLSchemeHandlerProxy::loadSynchronously(ResourceLoadIdentifier loadIdentifier, WebFrame& webFrame, const ResourceRequest& request, ResourceResponse& response, ResourceError& error, Vector<uint8_t>& data)
67{
68 data.shrink(0);
69 if (!m_webPage.sendSync(Messages::WebPageProxy::LoadSynchronousURLSchemeTask(URLSchemeTaskParameters { m_identifier, loadIdentifier, request, webFrame.info() }), Messages::WebPageProxy::LoadSynchronousURLSchemeTask::Reply(response, error, data))) {
70 error = failedCustomProtocolSyncLoad(request);
71 return;
72 }
73}
74
75void WebURLSchemeHandlerProxy::stopAllTasks()
76{
77 while (!m_tasks.isEmpty())
78 m_tasks.begin()->value->stopLoading();
79}
80
81void WebURLSchemeHandlerProxy::taskDidPerformRedirection(uint64_t taskIdentifier, WebCore::ResourceResponse&& redirectResponse, WebCore::ResourceRequest&& newRequest, CompletionHandler<void(WebCore::ResourceRequest&&)>&& completionHandler)
82{
83 auto* task = m_tasks.get(taskIdentifier);
84 if (!task)
85 return;
86
87 task->didPerformRedirection(WTFMove(redirectResponse), WTFMove(newRequest), WTFMove(completionHandler));
88}

90void WebURLSchemeHandlerProxy::taskDidReceiveResponse(uint64_t taskIdentifier, const ResourceResponse& response)
91{
92 auto* task = m_tasks.get(taskIdentifier);

96 task->didReceiveResponse(response);
97}

99void WebURLSchemeHandlerProxy::taskDidReceiveData(uint64_t taskIdentifier, size_t size, const uint8_t* data)
100{
101 auto* task = m_tasks.get(taskIdentifier);
102 if (!task)

105 task->didReceiveData(size, data);

107
108void WebURLSchemeHandlerProxy::taskDidComplete(uint64_t taskIdentifier, const ResourceError& error)
109{
110 if (auto task = removeTask(taskIdentifier))
111 task->didComplete(error);
112}
113
114void WebURLSchemeHandlerProxy::taskDidStopLoading(WebURLSchemeTaskProxy& task)
115{
116 ASSERT(m_tasks.get(task.identifier()) == &task);
117 removeTask(task.identifier());
118}
119
120RefPtr<WebURLSchemeTaskProxy> WebURLSchemeHandlerProxy::removeTask(uint64_t identifier)
121{
122 auto task = m_tasks.take(identifier);
123 if (!task)
124 return nullptr;
125
126 WebProcess::singleton().webLoaderStrategy().removeURLSchemeTaskProxy(*task);
127
128 return task;
129}
