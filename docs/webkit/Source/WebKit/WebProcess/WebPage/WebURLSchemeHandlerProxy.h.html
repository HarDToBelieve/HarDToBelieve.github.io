26#pragma once
27
28#include "WebURLSchemeTaskProxy.h"
29#include <wtf/HashMap.h>
30#include <wtf/RefCounted.h>
31
32namespace WebCore {
33class ResourceError;
34class ResourceLoader;
35class ResourceResponse;
36class ResourceRequest;
37}
38
39namespace WebKit {
40
41class WebPage;
42typedef uint64_t ResourceLoadIdentifier;
43
44class WebURLSchemeHandlerProxy : public RefCounted<WebURLSchemeHandlerProxy> {
45public:
46 static Ref<WebURLSchemeHandlerProxy> create(WebPage& page, uint64_t identifier)
47 {
48 return adoptRef(*new WebURLSchemeHandlerProxy(page, identifier));
49 }
50 ~WebURLSchemeHandlerProxy();
51
52 void startNewTask(WebCore::ResourceLoader&, WebFrame&);
53 void stopAllTasks();
54
55 void loadSynchronously(ResourceLoadIdentifier, WebFrame&, const WebCore::ResourceRequest&, WebCore::ResourceResponse&, WebCore::ResourceError&, Vector<uint8_t>&);
56
57 uint64_t identifier() const { return m_identifier; }
58 WebPage& page() { return m_webPage; }
59
60 void taskDidPerformRedirection(uint64_t taskIdentifier, WebCore::ResourceResponse&&, WebCore::ResourceRequest&&, CompletionHandler<void(WebCore::ResourceRequest&&)>&&);
61 void taskDidReceiveResponse(uint64_t taskIdentifier, const WebCore::ResourceResponse&);
62 void taskDidReceiveData(uint64_t taskIdentifier, size_t, const uint8_t* data);
63 void taskDidComplete(uint64_t taskIdentifier, const WebCore::ResourceError&);
64 void taskDidStopLoading(WebURLSchemeTaskProxy&);
65
66private:
67 WebURLSchemeHandlerProxy(WebPage&, uint64_t identifier);
68
69 RefPtr<WebURLSchemeTaskProxy> removeTask(uint64_t identifier);
70
71 WebPage& m_webPage;
72 uint64_t m_identifier { 0 };
73
74 HashMap<uint64_t, RefPtr<WebURLSchemeTaskProxy>> m_tasks;
75}; // class WebURLSchemeHandlerProxy
76
77} // namespace WebKit
