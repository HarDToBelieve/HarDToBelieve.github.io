50static uint64_t pageIDFromWebFrame(const RefPtr<WebFrame>& frame)
51{
52 if (frame) {
53 if (auto* page = frame->page())
54 return page->identifier().toUInt64();

56 return 0;

67WebURLSchemeTaskProxy::WebURLSchemeTaskProxy(WebURLSchemeHandlerProxy& handler, ResourceLoader& loader, WebFrame& frame)
68 : m_urlSchemeHandler(handler)
69 , m_coreLoader(&loader)
70 , m_frame(&frame)
71 , m_request(loader.request())
72 , m_identifier(loader.identifier())
73{
74}
75
76void WebURLSchemeTaskProxy::startLoading()
77{
78 ASSERT(m_coreLoader);
79 ASSERT(m_frame);
80 WEBURLSCHEMETASKPROXY_RELEASE_LOG("startLoading");
81 m_urlSchemeHandler.page().send(Messages::WebPageProxy::StartURLSchemeTask(URLSchemeTaskParameters { m_urlSchemeHandler.identifier(), m_coreLoader->identifier(), m_request, m_frame->info() }));
82}
83
84void WebURLSchemeTaskProxy::stopLoading()
85{
86 ASSERT(m_coreLoader);
87 WEBURLSCHEMETASKPROXY_RELEASE_LOG("stopLoading");
88 m_urlSchemeHandler.page().send(Messages::WebPageProxy::StopURLSchemeTask(m_urlSchemeHandler.identifier(), m_coreLoader->identifier()));
89 m_coreLoader = nullptr;
90 m_frame = nullptr;
91
92 // This line will result in this being deleted.
93 m_urlSchemeHandler.taskDidStopLoading(*this);
94}

96void WebURLSchemeTaskProxy::didPerformRedirection(WebCore::ResourceResponse&& redirectResponse, WebCore::ResourceRequest&& request, CompletionHandler<void(WebCore::ResourceRequest&&)>&& completionHandler)
97{
98 if (!hasLoader()) {
99 completionHandler({ });
100 return;
101 }
102
103 if (m_waitingForCompletionHandler) {
104 WEBURLSCHEMETASKPROXY_RELEASE_LOG("didPerformRedirection: Received redirect during previous redirect processing, queuing it.");
105 queueTask([this, protectedThis = makeRef(*this), redirectResponse = WTFMove(redirectResponse), request = WTFMove(request), completionHandler = WTFMove(completionHandler)]() mutable {
106 didPerformRedirection(WTFMove(redirectResponse), WTFMove(request), WTFMove(completionHandler));
107 });
108 return;
109 }
110 m_waitingForCompletionHandler = true;
111
112 auto innerCompletionHandler = [this, protectedThis = makeRef(*this), originalRequest = request, completionHandler = WTFMove(completionHandler)] (ResourceRequest&& request) mutable {
113 m_waitingForCompletionHandler = false;
114
115 completionHandler(WTFMove(request));
116
117 processNextPendingTask();
118 };
119
120 m_coreLoader->willSendRequest(WTFMove(request), redirectResponse, WTFMove(innerCompletionHandler));
121}

123void WebURLSchemeTaskProxy::didReceiveResponse(const ResourceResponse& response)
124{
125 if (m_waitingForCompletionHandler) {
126 WEBURLSCHEMETASKPROXY_RELEASE_LOG("didReceiveResponse: Received response during redirect processing, queuing it.");
127 queueTask([this, protectedThis = makeRef(*this), response] {
128 didReceiveResponse(response);
129 });
130 return;
131 }
132
133 if (!hasLoader())
134 return;
135
136 m_waitingForCompletionHandler = true;
137 m_coreLoader->didReceiveResponse(response, [this, protectedThis = makeRef(*this)] {
138 m_waitingForCompletionHandler = false;
139 processNextPendingTask();
140 });
141}

143void WebURLSchemeTaskProxy::didReceiveData(size_t size, const uint8_t* data)
144{
145 if (!hasLoader())
146 return;
147
148 if (m_waitingForCompletionHandler) {
149 WEBURLSCHEMETASKPROXY_RELEASE_LOG("didReceiveData: Received data during response processing, queuing it.");
150 queueTask([this, protectedThis = makeRef(*this), dataVector = Vector { data, size }] {
151 didReceiveData(dataVector.size(), dataVector.data());
152 });
153 return;
154 }
155
156 auto protectedThis = makeRef(*this);
157 m_coreLoader->didReceiveData(data, size, 0, DataPayloadType::DataPayloadBytes);
158 processNextPendingTask();
159}
160
161void WebURLSchemeTaskProxy::didComplete(const ResourceError& error)
162{
163 WEBURLSCHEMETASKPROXY_RELEASE_LOG("didComplete");
164 if (!hasLoader())
165 return;
166
167 if (m_waitingForCompletionHandler) {
168 queueTask([this, protectedThis = makeRef(*this), error] {
169 didComplete(error);
170 });
171 return;
172 }
173
174 if (error.isNull())
175 m_coreLoader->didFinishLoading(NetworkLoadMetrics());
176 else
177 m_coreLoader->didFail(error);
178
179 m_coreLoader = nullptr;
180 m_frame = nullptr;
181}
182
183bool WebURLSchemeTaskProxy::hasLoader()
184{
185 if (m_coreLoader && m_coreLoader->reachedTerminalState()) {
186 m_coreLoader = nullptr;
187 m_frame = nullptr;
188 }
189
190 return m_coreLoader;
191}
192
193void WebURLSchemeTaskProxy::processNextPendingTask()
194{
195 if (!m_queuedTasks.isEmpty())
196 m_queuedTasks.takeFirst()();
197}
