26#pragma once
27
28#include "WebUndoStepID.h"
29#include <WebCore/UndoStep.h>
30#include <wtf/Ref.h>
31
32namespace WebKit {
33
34class WebUndoStep : public RefCounted<WebUndoStep> {
35public:
36 static Ref<WebUndoStep> create(Ref<WebCore::UndoStep>&&);
37 ~WebUndoStep();
38
39 WebCore::UndoStep& step() const { return m_step.get(); }
40 WebUndoStepID stepID() const { return m_stepID; }
41
42 void didRemoveFromUndoManager() { m_step->didRemoveFromUndoManager(); }
43
44private:
45 WebUndoStep(Ref<WebCore::UndoStep>&& step, WebUndoStepID stepID)
46 : m_step(WTFMove(step))
47 , m_stepID(stepID)
48 {
49 }
50
51 Ref<WebCore::UndoStep> m_step;
52 WebUndoStepID m_stepID;
53};
54
55} // namespace WebKit
