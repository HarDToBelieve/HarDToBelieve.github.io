43// Avoid repeated start/stop IPC when rescheduled inside the callback.
44constexpr unsigned maxUnscheduledFireCount { 1 };
45
46DisplayRefreshMonitorMac::DisplayRefreshMonitorMac(PlatformDisplayID displayID)
47 : DisplayRefreshMonitor(displayID)
48 , m_observerID(DisplayLinkObserverID::generate())
49{
50 ASSERT(isMainRunLoop());
51 setMaxUnscheduledFireCount(maxUnscheduledFireCount);
52}
53
54DisplayRefreshMonitorMac::~DisplayRefreshMonitorMac()
55{
56 // stop() should have been called.
57 ASSERT(!m_displayLinkIsActive);
58}
59
60void DisplayRefreshMonitorMac::dispatchDisplayDidRefresh(const DisplayUpdate& displayUpdate)
61{
62 // FIXME: This will perturb displayUpdate.
63 if (!m_firstCallbackInCurrentRunloop)
64 return;
65
66 DisplayRefreshMonitor::dispatchDisplayDidRefresh(displayUpdate);
67}
68
69bool DisplayRefreshMonitorMac::startNotificationMechanism()
70{
71 if (m_displayLinkIsActive)
72 return true;
73
74 LOG_WITH_STREAM(DisplayLink, stream << "[Web] DisplayRefreshMonitorMac::requestRefreshCallback for display " << displayID() << " - starting");
75 WebProcess::singleton().parentProcessConnection()->send(Messages::WebProcessProxy::StartDisplayLink(m_observerID, displayID(), maxClientPreferredFramesPerSecond().value_or(FullSpeedFramesPerSecond)), 0);
76 if (!m_runLoopObserver) {
77 // The RunLoopObserver repeats.
78 m_runLoopObserver = makeUnique<RunLoopObserver>(kCFRunLoopEntry, [this]() {
79 m_firstCallbackInCurrentRunloop = true;
80 });
81 }
82
83 m_runLoopObserver->schedule(CFRunLoopGetCurrent());
84 m_displayLinkIsActive = true;
85
86 return true;
87}
88
89void DisplayRefreshMonitorMac::stopNotificationMechanism()
90{
91 if (!m_displayLinkIsActive)
92 return;
93
94 LOG_WITH_STREAM(DisplayLink, stream << "[Web] DisplayRefreshMonitorMac::requestRefreshCallback - stopping");
95 WebProcess::singleton().parentProcessConnection()->send(Messages::WebProcessProxy::StopDisplayLink(m_observerID, displayID()), 0);
96 m_runLoopObserver->invalidate();
97
98 m_displayLinkIsActive = false;
99}
100
101void DisplayRefreshMonitorMac::adjustPreferredFramesPerSecond(FramesPerSecond preferredFramesPerSecond)
102{
103 LOG_WITH_STREAM(DisplayLink, stream << "[Web] DisplayRefreshMonitorMac::adjustPreferredFramesPerSecond for display link on display " << displayID() << " to " << preferredFramesPerSecond);
104 WebProcess::singleton().parentProcessConnection()->send(Messages::WebProcessProxy::SetDisplayLinkPreferredFramesPerSecond(m_observerID, displayID(), preferredFramesPerSecond), 0);
105
106}
