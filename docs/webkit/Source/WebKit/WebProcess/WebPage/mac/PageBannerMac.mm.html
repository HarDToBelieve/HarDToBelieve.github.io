38Ref<PageBanner> PageBanner::create(CALayer *layer, int height, Client* client)

43PageBanner::PageBanner(CALayer *layer, int height, Client* client)
44 : m_client(client)
45 , m_layer(layer)
46 , m_height(height)
47{
48}

50void PageBanner::addToPage(Type type, WebPage* webPage)
51{
52 m_type = type;
53 m_webPage = webPage;
54
55 ASSERT(m_type != NotSet);
56 ASSERT(m_webPage);
57
58 switch (m_type) {
59 case Header:
60 m_webPage->corePage()->setHeaderHeight(m_height);
61 break;
62 case Footer:
63 m_webPage->corePage()->setFooterHeight(m_height);
64 break;
65 case NotSet:
66 ASSERT_NOT_REACHED();
67 }
68}

70void PageBanner::didAddParentLayer(GraphicsLayer* parentLayer)
71{
72 if (!parentLayer)
73 return;
74
75 m_layer.get().bounds = CGRectMake(0, 0, parentLayer->size().width(), parentLayer->size().height());
76 [parentLayer->platformLayer() addSublayer:m_layer.get()];
77}
78
79void PageBanner::detachFromPage()
80{
81 if (!m_webPage)
82 return;
83
84 // m_webPage->corePage() can be null when this is called from WebPage::~WebPage() after
85 // the web page has been closed.
86 if (m_webPage->corePage()) {
87 // We can hide the banner by removing the parent layer that hosts it.
88 if (m_type == Header)
89 m_webPage->corePage()->setHeaderHeight(0);
90 else if (m_type == Footer)
91 m_webPage->corePage()->setFooterHeight(0);
92 }
93
94 m_type = NotSet;
95 m_webPage = 0;
96}
97
98void PageBanner::hide()
99{
100 // We can hide the banner by removing the parent layer that hosts it.
101 if (m_type == Header)
102 m_webPage->corePage()->setHeaderHeight(0);
103 else if (m_type == Footer)
104 m_webPage->corePage()->setFooterHeight(0);
105
106 m_isHidden = true;
107}
108
109void PageBanner::showIfHidden()
110{
111 if (!m_isHidden)
112 return;
113 m_isHidden = false;
114
115 // This will re-create a parent layer in the WebCore layer tree, and we will re-add
116 // m_layer as a child of it.
117 addToPage(m_type, m_webPage);
118}

126bool PageBanner::mouseEvent(const WebMouseEvent& mouseEvent)
127{
128 if (m_isHidden)
129 return false;
130
131 FrameView* frameView = m_webPage->mainFrameView();
132 if (!frameView)
133 return false;
134
135 IntPoint positionInBannerSpace;
136
137 switch (m_type) {
138 case Header: {
139 positionInBannerSpace = frameView->rootViewToTotalContents(mouseEvent.position());
140 break;
141 }
142 case Footer: {
143 positionInBannerSpace = frameView->rootViewToTotalContents(mouseEvent.position()) - IntSize(0, frameView->totalContentsSize().height() - m_height);
144 break;
145 }
146 case NotSet:
147 ASSERT_NOT_REACHED();
148 }
149
150 if (!m_mouseDownInBanner && (positionInBannerSpace.y() < 0 || positionInBannerSpace.y() > m_height))
151 return false;
152
153 if (mouseEvent.type() == WebEvent::MouseDown)
154 m_mouseDownInBanner = true;
155 else if (mouseEvent.type() == WebEvent::MouseUp)
156 m_mouseDownInBanner = false;
157
158 return m_client->mouseEvent(this, mouseEvent.type(), mouseEvent.button(), positionInBannerSpace);
159}
