47@implementation WKAccessibilityWebPageObjectBase
48
49- (NakedPtr<WebCore::AXObjectCache>)axObjectCache
50{
51 ASSERT(isMainRunLoop());
52
53 if (!m_page)
54 return nullptr;
55
56 auto page = m_page->corePage();
57 if (!page)
58 return nullptr;
59
60 auto& core = page->mainFrame();
61 if (!core.document())
62 return nullptr;
63
64 return core.document()->axObjectCache();
65}
66
67- (id)accessibilityPluginObject
68{
69 ASSERT(isMainRunLoop());
70 auto retrieveBlock = [&self]() -> id {
71 id axPlugin = nil;
72 callOnMainRunLoopAndWait([&axPlugin, &self] {
73 if (self->m_page)
74 axPlugin = self->m_page->accessibilityObjectForMainFramePlugin();
75 });
76 return axPlugin;
77 };
78
79 return retrieveBlock();
80}
81
82- (id)accessibilityRootObjectWrapper
83{
84 return ax::retrieveAutoreleasedValueFromMainThread<id>([protectedSelf = retainPtr(self)] () -> RetainPtr<id> {
85 if (!WebCore::AXObjectCache::accessibilityEnabled())
86 WebCore::AXObjectCache::enableAccessibility();
87
88 if (protectedSelf.get()->m_hasMainFramePlugin)
89 return protectedSelf.get().accessibilityPluginObject;
90
91 if (auto cache = protectedSelf.get().axObjectCache) {
92 if (auto* root = cache->rootObject())
93 return root->wrapper();
94 }
95
96 return nil;
97 });
98}
99
100- (void)setWebPage:(NakedPtr<WebKit::WebPage>)page
101{
102 ASSERT(isMainRunLoop());
103
104 m_page = page;
105
106 if (page) {
107 m_pageID = page->identifier();
108
109 auto* frame = page->mainFrame();
110 m_hasMainFramePlugin = frame && frame->document() ? frame->document()->isPluginDocument() : false;
111 } else {
112 m_pageID = { };
113 m_hasMainFramePlugin = false;
114 }
115}
116
117- (void)setHasMainFramePlugin:(bool)hasPlugin
118{
119 ASSERT(isMainRunLoop());
120 m_hasMainFramePlugin = hasPlugin;
121}
122
123- (void)setRemoteParent:(id)parent
124{
125 ASSERT(isMainRunLoop());
126 m_parent = parent;
127}
128
129- (id)accessibilityFocusedUIElement
130{
131 return [[self accessibilityRootObjectWrapper] accessibilityFocusedUIElement];
132}
