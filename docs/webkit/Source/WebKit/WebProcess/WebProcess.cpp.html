377void WebProcess::initializeWebProcess(WebProcessCreationParameters&& parameters)
378{
379 TraceScope traceScope(InitializeWebProcessStart, InitializeWebProcessEnd);
380
381 ASSERT(m_pageMap.isEmpty());
382
383 if (parameters.websiteDataStoreParameters)
384 setWebsiteDataStoreParameters(WTFMove(*parameters.websiteDataStoreParameters));
385
386 WebCore::setPresentingApplicationPID(parameters.presentingApplicationPID);
387
388#if OS(LINUX)
389 MemoryPressureHandler::ReliefLogger::setLoggingEnabled(parameters.shouldEnableMemoryPressureReliefLogging);
390#endif
391
392 platformInitializeWebProcess(parameters);
393
394 // Match the QoS of the UIProcess and the scrolling thread but use a slightly lower priority.
395 WTF::Thread::setCurrentThreadIsUserInteractive(-1);
396
397 m_suppressMemoryPressureHandler = parameters.shouldSuppressMemoryPressureHandler;
398 if (!m_suppressMemoryPressureHandler) {
399 auto& memoryPressureHandler = MemoryPressureHandler::singleton();
400 memoryPressureHandler.setLowMemoryHandler([this] (Critical critical, Synchronous synchronous) {
401 // If this process contains only non-visible content (e.g. only contains background
402 // tabs), then treat the memory warning as if it was a critical warning to maximize the
403 // amount of memory released for foreground apps to use.
404 if (m_pagesInWindows.isEmpty() && critical == Critical::No)
405 critical = Critical::Yes;
406
407#if PLATFORM(MAC)
408 // If this is a process we keep around for performance, kill it on memory pressure instead of trying to free up its memory.
409 if (m_processType == ProcessType::CachedWebContent || m_processType == ProcessType::PrewarmedWebContent || areAllPagesSuspended()) {
410 if (m_processType == ProcessType::CachedWebContent)
411 WEBPROCESS_RELEASE_LOG(Process, "initializeWebProcess: Cached WebProcess is exiting due to memory pressure");
412 else if (m_processType == ProcessType::PrewarmedWebContent)
413 WEBPROCESS_RELEASE_LOG(Process, "initializeWebProcess: Prewarmed WebProcess is exiting due to memory pressure");
414 else
415 WEBPROCESS_RELEASE_LOG(Process, "initializeWebProcess: Suspended WebProcess is exiting due to memory pressure");
416 stopRunLoop();
417 return;
418 }
419#endif
420
421 auto maintainBackForwardCache = m_isSuspending ? WebCore::MaintainBackForwardCache::Yes : WebCore::MaintainBackForwardCache::No;
422 auto maintainMemoryCache = m_isSuspending && m_hasSuspendedPageProxy ? WebCore::MaintainMemoryCache::Yes : WebCore::MaintainMemoryCache::No;
423 WebCore::releaseMemory(critical, synchronous, maintainBackForwardCache, maintainMemoryCache);
424 for (auto& page : m_pageMap.values())
425 page->releaseMemory(critical);
426 });
427#if ENABLE(PERIODIC_MEMORY_MONITOR)
428 memoryPressureHandler.setShouldUsePeriodicMemoryMonitor(true);
429 memoryPressureHandler.setMemoryKillCallback([this] () {
430 WebCore::logMemoryStatisticsAtTimeOfDeath();
431 if (MemoryPressureHandler::singleton().processState() == WebsamProcessState::Active)
432 parentProcessConnection()->send(Messages::WebProcessProxy::DidExceedActiveMemoryLimit(), 0);
433 else
434 parentProcessConnection()->send(Messages::WebProcessProxy::DidExceedInactiveMemoryLimit(), 0);
435 });
436 memoryPressureHandler.setDidExceedInactiveLimitWhileActiveCallback([this] () {
437 parentProcessConnection()->send(Messages::WebProcessProxy::DidExceedInactiveMemoryLimitWhileActive(), 0);
438 });
439#endif
440 memoryPressureHandler.setMemoryPressureStatusChangedCallback([this](bool isUnderMemoryPressure) {
441 if (parentProcessConnection())
442 parentProcessConnection()->send(Messages::WebProcessProxy::MemoryPressureStatusChanged(isUnderMemoryPressure), 0);
443 });
444 memoryPressureHandler.install();
445 }
446
447 SandboxExtension::consumePermanently(parameters.additionalSandboxExtensionHandles);
448
449 if (!parameters.injectedBundlePath.isEmpty())
450 m_injectedBundle = InjectedBundle::create(parameters, transformHandlesToObjects(parameters.initializationUserData.object()).get());
451
452 for (auto& supplement : m_supplements.values())
453 supplement->initialize(parameters);
454
455 setCacheModel(parameters.cacheModel);
456
457 if (!parameters.overrideLanguages.isEmpty())
458 overrideUserPreferredLanguages(parameters.overrideLanguages);
459
460 m_textCheckerState = parameters.textCheckerState;
461
462 m_fullKeyboardAccessEnabled = parameters.fullKeyboardAccessEnabled;
463
464#if HAVE(MOUSE_DEVICE_OBSERVATION)
465 m_hasMouseDevice = parameters.hasMouseDevice;
466#endif
467
468#if HAVE(STYLUS_DEVICE_OBSERVATION)
469 m_hasStylusDevice = parameters.hasStylusDevice;
470#endif
471
472 for (auto& scheme : parameters.urlSchemesRegisteredAsEmptyDocument)
473 registerURLSchemeAsEmptyDocument(scheme);
474
475 for (auto& scheme : parameters.urlSchemesRegisteredAsSecure)
476 registerURLSchemeAsSecure(scheme);
477
478 for (auto& scheme : parameters.urlSchemesRegisteredAsBypassingContentSecurityPolicy)
479 registerURLSchemeAsBypassingContentSecurityPolicy(scheme);
480
481 for (auto& scheme : parameters.urlSchemesForWhichDomainRelaxationIsForbidden)
482 setDomainRelaxationForbiddenForURLScheme(scheme);
483
484 for (auto& scheme : parameters.urlSchemesRegisteredAsLocal)
485 registerURLSchemeAsLocal(scheme);
486
487 for (auto& scheme : parameters.urlSchemesRegisteredAsNoAccess)
488 registerURLSchemeAsNoAccess(scheme);
489
490 for (auto& scheme : parameters.urlSchemesRegisteredAsDisplayIsolated)
491 registerURLSchemeAsDisplayIsolated(scheme);
492
493 for (auto& scheme : parameters.urlSchemesRegisteredAsCORSEnabled)
494 LegacySchemeRegistry::registerURLSchemeAsCORSEnabled(scheme);
495
496 for (auto& scheme : parameters.urlSchemesRegisteredAsAlwaysRevalidated)
497 registerURLSchemeAsAlwaysRevalidated(scheme);
498
499 for (auto& scheme : parameters.urlSchemesRegisteredAsCachePartitioned)
500 registerURLSchemeAsCachePartitioned(scheme);
501
502 for (auto& scheme : parameters.urlSchemesRegisteredAsCanDisplayOnlyIfCanRequest)
503 registerURLSchemeAsCanDisplayOnlyIfCanRequest(scheme);
504
505 setDefaultRequestTimeoutInterval(parameters.defaultRequestTimeoutInterval);
506
507 setBackForwardCacheCapacity(parameters.backForwardCacheCapacity);
508
509 setAlwaysUsesComplexTextCodePath(parameters.shouldAlwaysUseComplexTextCodePath);
510
511 setShouldUseFontSmoothing(parameters.shouldUseFontSmoothing);
512
513 setTerminationTimeout(parameters.terminationTimeout);
514
515 setMemoryCacheDisabled(parameters.memoryCacheDisabled);
516
517 WebCore::RuntimeEnabledFeatures::sharedFeatures().setAttrStyleEnabled(parameters.attrStyleEnabled);
518
519 commonVM().setGlobalConstRedeclarationShouldThrow(parameters.shouldThrowExceptionForGlobalConstantRedeclaration);
520
521#if ENABLE(SERVICE_CONTROLS)
522 setEnabledServices(parameters.hasImageServices, parameters.hasSelectionServices, parameters.hasRichContentServices);
523#endif
524
525#if ENABLE(REMOTE_INSPECTOR) && PLATFORM(COCOA)
526#if PLATFORM(IOS)
527 Inspector::RemoteInspector::setNeedMachSandboxExtension(true);
528#endif
529 if (std::optional<audit_token_t> auditToken = parentProcessConnection()->getAuditToken()) {
530 RetainPtr<CFDataRef> auditData = adoptCF(CFDataCreate(nullptr, (const UInt8*)&*auditToken, sizeof(*auditToken)));
531 Inspector::RemoteInspector::singleton().setParentProcessInformation(WebCore::presentingApplicationPID(), auditData);
532 }
533#endif
534
535#if ENABLE(GAMEPAD)
536 GamepadProvider::singleton().setSharedProvider(WebGamepadProvider::singleton());
537#endif
538
539#if ENABLE(SERVICE_WORKER)
540 ServiceWorkerProvider::setSharedProvider(WebServiceWorkerProvider::singleton());
541#endif
542
543#if ENABLE(WEBASSEMBLY)
544 JSC::Wasm::enableFastMemory();
545#endif
546
547#if ENABLE(RESOURCE_LOAD_STATISTICS) && !RELEASE_LOG_DISABLED
548 WebResourceLoadObserver::setShouldLogUserInteraction(parameters.shouldLogUserInteraction);
549#endif
550
551#if PLATFORM(COCOA)
552 if (m_processType == ProcessType::PrewarmedWebContent)
553 prewarmGlobally();
554#endif
555
556 WEBPROCESS_RELEASE_LOG(Process, "initializeWebProcess: Presenting processPID=%d", WebCore::presentingApplicationPID());
557}

617void WebProcess::setIsInProcessCache(bool isInProcessCache, CompletionHandler<void()>&& completionHandler)
618{
619#if PLATFORM(COCOA)
620 if (isInProcessCache) {
621 ASSERT(m_processType == ProcessType::WebContent);
622 m_processType = ProcessType::CachedWebContent;
623 } else {
624 ASSERT(m_processType == ProcessType::CachedWebContent);
625 m_processType = ProcessType::WebContent;
626 }
627
628 updateProcessName(IsInProcessInitialization::No);
629#else
630 UNUSED_PARAM(isInProcessCache);
631#endif
632
633#if ENABLE(SERVICE_WORKER)
634 if (isInProcessCache) {
635 // When calling the completion handler, the parent process may suspend us. We do not want to get suspended until pending requests to
636 // stop service workers are processed because this would be wasteful from a memory standpoint. Also, service worker termination
637 // requests have a timeout and this timeout would be reached if we suspended while it is being processed.

1083static NetworkProcessConnectionInfo getNetworkProcessConnection(IPC::Connection& connection)
1084{
1085 NetworkProcessConnectionInfo connectionInfo;
1086 auto requestConnection = [&] {
1087 if (!connection.isValid()) {
1088 // Connection to UIProcess has been severed, exit cleanly.
1089 exit(0);
1090 }
1091 if (!connection.sendSync(Messages::WebProcessProxy::GetNetworkProcessConnection(), Messages::WebProcessProxy::GetNetworkProcessConnection::Reply(connectionInfo), 0))
1092 return false;
1093 return IPC::Connection::identifierIsValid(connectionInfo.identifier());
1094 };
1095
1096 static constexpr unsigned maxFailedAttempts = 10;
1097 unsigned failedAttempts = 0;
1098 while (!requestConnection()) {
1099 if (++failedAttempts >= maxFailedAttempts)
1100 CRASH();
1101
1102 RELEASE_LOG_ERROR(Process, "getNetworkProcessConnection: Failed to get connection to network process, will retry...");
1103
1104 // If we failed, retry after a delay. The attachment may have become invalid
1105 // before it was received by the web process if the network process crashed.
1106 sleep(100_ms);
1107 }
1108 return connectionInfo;
1109}

1111NetworkProcessConnection& WebProcess::ensureNetworkProcessConnection()
1112{
1113 RELEASE_ASSERT(RunLoop::isMain());
1114 ASSERT(m_sessionID);
1115
1116 // If we've lost our connection to the network process (e.g. it crashed) try to re-establish it.
1117 if (!m_networkProcessConnection) {
1118 auto connectionInfo = getNetworkProcessConnection(*parentProcessConnection());
1119
1120 m_networkProcessConnection = NetworkProcessConnection::create(connectionInfo.releaseIdentifier(), connectionInfo.cookieAcceptPolicy);
1121#if HAVE(AUDIT_TOKEN)
1122 m_networkProcessConnection->setNetworkProcessAuditToken(WTFMove(connectionInfo.auditToken));
1123#endif
1124 m_networkProcessConnection->connection().send(Messages::NetworkConnectionToWebProcess::RegisterURLSchemesAsCORSEnabled(WebCore::LegacySchemeRegistry::allURLSchemesRegisteredAsCORSEnabled()), 0);
1125
1126#if ENABLE(SERVICE_WORKER)
1127 if (!Document::allDocuments().isEmpty())
1128 m_networkProcessConnection->serviceWorkerConnection().registerServiceWorkerClients();
1129#endif
1130
1131 // This can be called during a WebPage's constructor, so wait until after the constructor returns to touch the WebPage.
1132 RunLoop::main().dispatch([this] {
1133 for (auto& webPage : m_pageMap.values())
1134 webPage->synchronizeCORSDisablingPatternsWithNetworkProcess();
1135 });
1136 }
1137
1138 return *m_networkProcessConnection;
1139}
