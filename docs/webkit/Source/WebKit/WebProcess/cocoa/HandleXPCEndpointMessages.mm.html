35namespace WebKit {
36
37void handleXPCEndpointMessages(xpc_object_t event, const char* messageName)
38{
39 ASSERT_UNUSED(event, xpc_get_type(event) == XPC_TYPE_DICTIONARY);
40 ASSERT_UNUSED(messageName, messageName);
41
42#if HAVE(LSDATABASECONTEXT)
43 if (!strcmp(messageName, LaunchServicesDatabaseXPCConstants::xpcLaunchServicesDatabaseXPCEndpointMessageName)) {
44 auto xpcEndPoint = xpc_dictionary_get_value(event, LaunchServicesDatabaseXPCConstants::xpcLaunchServicesDatabaseXPCEndpointNameKey);
45 if (!xpcEndPoint || xpc_get_type(xpcEndPoint) != XPC_TYPE_ENDPOINT)
46 return;
47 LaunchServicesDatabaseManager::singleton().setEndpoint(xpcEndPoint);
48 return;
49 }
50#endif
51}
