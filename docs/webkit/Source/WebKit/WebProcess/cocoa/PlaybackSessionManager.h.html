58class PlaybackSessionInterfaceContext final

63 {
64 return adoptRef(*new PlaybackSessionInterfaceContext(manager, contextId));

66 virtual ~PlaybackSessionInterfaceContext();

68 void invalidate() { m_manager = nullptr; }
