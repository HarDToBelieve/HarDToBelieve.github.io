164PlaybackSessionManager::PlaybackSessionManager(WebPage& page)
165 : m_page(&page)
166{
167 WebProcess::singleton().addMessageReceiver(Messages::PlaybackSessionManager::messageReceiverName(), page.identifier(), *this);
168}
169
170PlaybackSessionManager::~PlaybackSessionManager()
171{
172 for (auto& [model, interface] : m_contextMap.values()) {
173 model->removeClient(*interface);
174 model->setMediaElement(nullptr);
175
176 interface->invalidate();
177 }
178
179 m_contextMap.clear();
180 m_mediaElements.clear();
181 m_clientCounts.clear();
182
183 if (m_page)
184 WebProcess::singleton().removeMessageReceiver(Messages::PlaybackSessionManager::messageReceiverName(), m_page->identifier());
185}
186
187void PlaybackSessionManager::invalidate()
188{
189 ASSERT(m_page);
190 WebProcess::singleton().removeMessageReceiver(Messages::PlaybackSessionManager::messageReceiverName(), m_page->identifier());
191 m_page = nullptr;
192}

194PlaybackSessionManager::ModelInterfaceTuple PlaybackSessionManager::createModelAndInterface(PlaybackSessionContextIdentifier contextId)
195{
196 auto model = PlaybackSessionModelMediaElement::create();
197 auto interface = PlaybackSessionInterfaceContext::create(*this, contextId);
198 model->addClient(interface.get());
199
200 return std::make_tuple(WTFMove(model), WTFMove(interface));
201}
202
203PlaybackSessionManager::ModelInterfaceTuple& PlaybackSessionManager::ensureModelAndInterface(PlaybackSessionContextIdentifier contextId)
204{
205 auto addResult = m_contextMap.add(contextId, ModelInterfaceTuple());
206 if (addResult.isNewEntry)
207 addResult.iterator->value = createModelAndInterface(contextId);
208 return addResult.iterator->value;
209}

211WebCore::PlaybackSessionModelMediaElement& PlaybackSessionManager::ensureModel(PlaybackSessionContextIdentifier contextId)
212{
213 return *std::get<0>(ensureModelAndInterface(contextId));
214}
215
216PlaybackSessionInterfaceContext& PlaybackSessionManager::ensureInterface(PlaybackSessionContextIdentifier contextId)
217{
218 return *std::get<1>(ensureModelAndInterface(contextId));
219}

221void PlaybackSessionManager::removeContext(PlaybackSessionContextIdentifier contextId)
222{
223 auto& [model, interface] = ensureModelAndInterface(contextId);
224
225 RefPtr<HTMLMediaElement> mediaElement = model->mediaElement();
226 model->setMediaElement(nullptr);
227 model->removeClient(*interface);
228 interface->invalidate();
229 m_mediaElements.remove(mediaElement.get());
230 m_contextMap.remove(contextId);
231}

233void PlaybackSessionManager::addClientForContext(PlaybackSessionContextIdentifier contextId)
234{
235 m_clientCounts.add(contextId);
236}
237
238void PlaybackSessionManager::removeClientForContext(PlaybackSessionContextIdentifier contextId)
239{
240 ASSERT(m_clientCounts.contains(contextId));
241 if (m_clientCounts.remove(contextId))
242 removeContext(contextId);
243}

245void PlaybackSessionManager::setUpPlaybackControlsManager(WebCore::HTMLMediaElement& mediaElement)
246{
247 auto foundIterator = m_mediaElements.find(&mediaElement);
248 if (foundIterator != m_mediaElements.end()) {
249 auto contextId = foundIterator->value;
250 if (m_controlsManagerContextId == contextId)
251 return;
252
253 auto previousContextId = m_controlsManagerContextId;
254 m_controlsManagerContextId = contextId;
255 if (previousContextId)
256 removeClientForContext(previousContextId);
257 } else {
258 auto contextId = m_mediaElements.ensure(&mediaElement, [&] {
259 return PlaybackSessionContextIdentifier::generate();
260 }).iterator->value;
261
262 auto previousContextId = m_controlsManagerContextId;
263 m_controlsManagerContextId = contextId;
264 if (previousContextId)
265 removeClientForContext(previousContextId);
266
267 ensureModel(contextId).setMediaElement(&mediaElement);
268 }
269
270 addClientForContext(m_controlsManagerContextId);
271
272 m_page->videoControlsManagerDidChange();
273 m_page->send(Messages::PlaybackSessionManagerProxy::SetUpPlaybackControlsManagerWithID(m_controlsManagerContextId));
274}
275
276void PlaybackSessionManager::clearPlaybackControlsManager()
277{
278 if (!m_controlsManagerContextId)
279 return;
280
281 removeClientForContext(m_controlsManagerContextId);
282 m_controlsManagerContextId = { };
283
284 m_page->videoControlsManagerDidChange();
285 m_page->send(Messages::PlaybackSessionManagerProxy::ClearPlaybackControlsManager());
286}

288void PlaybackSessionManager::mediaEngineChanged()
289{
290 if (!m_controlsManagerContextId)
291 return;
292
293 auto it = m_contextMap.find(m_controlsManagerContextId);
294 if (it == m_contextMap.end())
295 return;
296
297 std::get<0>(it->value)->mediaEngineChanged();
298}
299
300PlaybackSessionContextIdentifier PlaybackSessionManager::contextIdForMediaElement(WebCore::HTMLMediaElement& mediaElement)
301{
302 auto addResult = m_mediaElements.ensure(&mediaElement, [&] {
303 return PlaybackSessionContextIdentifier::generate();
304 });
305 auto contextId = addResult.iterator->value;
306 ensureModel(contextId).setMediaElement(&mediaElement);
307 return contextId;
308}
309
310WebCore::HTMLMediaElement* PlaybackSessionManager::currentPlaybackControlsElement() const
311{
312 if (!m_controlsManagerContextId)
313 return nullptr;
314
315 auto iter = m_contextMap.find(m_controlsManagerContextId);
316 if (iter == m_contextMap.end())
317 return nullptr;
318
319 return std::get<0>(iter->value)->mediaElement();
320}

349void PlaybackSessionManager::seekableRangesChanged(PlaybackSessionContextIdentifier contextId, const WebCore::TimeRanges& timeRanges, double lastModifiedTime, double liveUpdateInterval)
350{
351 Vector<std::pair<double, double>> rangesVector;
352 for (unsigned i = 0; i < timeRanges.length(); i++) {
353 double start = timeRanges.ranges().start(i).toDouble();
354 double end = timeRanges.ranges().end(i).toDouble();
355 rangesVector.append({ start, end });
356 }
357 m_page->send(Messages::PlaybackSessionManagerProxy::SeekableRangesVectorChanged(contextId, WTFMove(rangesVector), lastModifiedTime, liveUpdateInterval));
358}
