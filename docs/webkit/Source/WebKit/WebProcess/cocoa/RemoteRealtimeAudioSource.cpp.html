46Ref<RealtimeMediaSource> RemoteRealtimeAudioSource::create(const CaptureDevice& device, const MediaConstraints* constraints, String&& name, String&& hashSalt, UserMediaCaptureManager& manager, bool shouldCaptureInGPUProcess)
47{
48 auto source = adoptRef(*new RemoteRealtimeAudioSource(RealtimeMediaSourceIdentifier::generate(), device, constraints, WTFMove(name), WTFMove(hashSalt), manager, shouldCaptureInGPUProcess));
49 manager.addAudioSource(source.copyRef());
50 manager.remoteCaptureSampleManager().addSource(source.copyRef());
51 source->createRemoteMediaSource();
52 return source;
53}

55RemoteRealtimeAudioSource::RemoteRealtimeAudioSource(RealtimeMediaSourceIdentifier identifier, const CaptureDevice& device, const MediaConstraints* constraints, String&& name, String&& hashSalt, UserMediaCaptureManager& manager, bool shouldCaptureInGPUProcess)
56 : RealtimeMediaSource(RealtimeMediaSource::Type::Audio, WTFMove(name), String::number(identifier.toUInt64()), WTFMove(hashSalt))
57 , m_proxy(identifier, device, shouldCaptureInGPUProcess, constraints)
58 , m_manager(manager)
59{
60 ASSERT(device.type() == CaptureDevice::DeviceType::Microphone);
61#if PLATFORM(IOS_FAMILY)
62 RealtimeMediaSourceCenter::singleton().audioCaptureFactory().setActiveSource(*this);
63#endif
64}
65
66void RemoteRealtimeAudioSource::createRemoteMediaSource()
67{
68 m_proxy.createRemoteMediaSource(deviceIDHashSalt(), [this, protectedThis = makeRef(*this)](bool succeeded, auto&& errorMessage, auto&& settings, auto&& capabilities, auto&&, auto, auto) {
69 if (!succeeded) {
70 m_proxy.didFail(WTFMove(errorMessage));
71 return;
72 }
73
74 setSettings(WTFMove(settings));
75 setCapabilities(WTFMove(capabilities));
76 setName(String { m_settings.label().string() });
77
78 m_proxy.setAsReady();
79 if (m_proxy.shouldCaptureInGPUProcess())
80 WebProcess::singleton().ensureGPUProcessConnection().addClient(*this);
81 });
82}
83
84RemoteRealtimeAudioSource::~RemoteRealtimeAudioSource()
85{
86 if (m_proxy.shouldCaptureInGPUProcess()) {
87 if (auto* connection = WebProcess::singleton().existingGPUProcessConnection())
88 connection->removeClient(*this);
89 }
90
91#if PLATFORM(IOS_FAMILY)
92 RealtimeMediaSourceCenter::singleton().audioCaptureFactory().unsetActiveSource(*this);
93#endif
94}

120void RemoteRealtimeAudioSource::hasEnded()
121{
122 m_proxy.hasEnded();
123 m_manager.removeAudioSource(identifier());
124 m_manager.remoteCaptureSampleManager().removeSource(identifier());
125}

127void RemoteRealtimeAudioSource::captureStopped()

129 stop();

131}

133void RemoteRealtimeAudioSource::captureFailed()

145#if ENABLE(GPU_PROCESS)
146void RemoteRealtimeAudioSource::gpuProcessConnectionDidClose(GPUProcessConnection&)
147{
148 ASSERT(m_proxy.shouldCaptureInGPUProcess());
149 if (isEnded())
150 return;
151
152#if PLATFORM(IOS_FAMILY)
153 if (this != RealtimeMediaSourceCenter::singleton().audioCaptureFactory().activeSource()) {
154 // Track is muted and has no chance of being unmuted, let's end it.
155 captureFailed();
156 return;
157 }
158#endif
159
160 m_manager.remoteCaptureSampleManager().didUpdateSourceConnection(connection());
161 m_proxy.resetReady();
162 createRemoteMediaSource();
163
164 m_proxy.failApplyConstraintCallbacks("GPU Process terminated"_s);
165 if (m_constraints)
166 m_proxy.applyConstraints(*m_constraints, [](auto) { });
167
168 if (isProducingData())
169 startProducingData();
170
171}
172#endif
