44namespace WebKit {
45
46class RemoteRealtimeMediaSourceProxy {
47public:
48 RemoteRealtimeMediaSourceProxy(WebCore::RealtimeMediaSourceIdentifier, const WebCore::CaptureDevice&, bool shouldCaptureInGPUProcess, const WebCore::MediaConstraints*);
49 ~RemoteRealtimeMediaSourceProxy();
50
51 IPC::Connection* connection();
52 WebCore::RealtimeMediaSourceIdentifier identifier() const { return m_identifier; }
53 WebCore::CaptureDevice::DeviceType deviceType() const { return m_device.type(); }
54 bool shouldCaptureInGPUProcess() const { return m_shouldCaptureInGPUProcess; }
55
56 using CreateCallback = CompletionHandler<void(bool, String&&, WebCore::RealtimeMediaSourceSettings&&, WebCore::RealtimeMediaSourceCapabilities&&, Vector<WebCore::VideoPresetData>&&, WebCore::IntSize, double)>;
57 void createRemoteMediaSource(const String&, CreateCallback&&);
58
59 void applyConstraintsSucceeded();
60 void applyConstraintsFailed(String&& failedConstraint, String&& errorMessage);
61 void failApplyConstraintCallbacks(const String& errorMessage);
62
63 void hasEnded();
64 void startProducingData();
65 void stopProducingData();
66 void applyConstraints(const WebCore::MediaConstraints&, WebCore::RealtimeMediaSource::ApplyConstraintsHandler&&);
67
68 void whenReady(CompletionHandler<void(String)>&&);
69 void setAsReady();
70 void resetReady() { m_isReady = false; }
71 bool isReady() const { return m_isReady; }
72
73 void didFail(String&& errorMessage);
74
75private:
76 WebCore::RealtimeMediaSourceIdentifier m_identifier;
77 WebCore::CaptureDevice m_device;
78 bool m_shouldCaptureInGPUProcess { false };
79
80 WebCore::MediaConstraints m_constraints;
81 Deque<WebCore::RealtimeMediaSource::ApplyConstraintsHandler> m_pendingApplyConstraintsCallbacks;
82 bool m_isReady { false };
83 CompletionHandler<void(String)> m_callback;
84 String m_errorMessage;
85};
86
87inline RemoteRealtimeMediaSourceProxy::RemoteRealtimeMediaSourceProxy(WebCore::RealtimeMediaSourceIdentifier identifier, const WebCore::CaptureDevice& device, bool shouldCaptureInGPUProcess, const WebCore::MediaConstraints* constraints)
88 : m_identifier(identifier)
89 , m_device(device)
90 , m_shouldCaptureInGPUProcess(shouldCaptureInGPUProcess)
91{
92 if (constraints)
93 m_constraints = *constraints;
94}
