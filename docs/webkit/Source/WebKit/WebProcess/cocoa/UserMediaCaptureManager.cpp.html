49UserMediaCaptureManager::UserMediaCaptureManager(WebProcess& process)
50 : m_process(process)
51 , m_audioFactory(*this)
52 , m_videoFactory(*this)
53 , m_displayFactory(*this)
54{
55 m_process.addMessageReceiver(Messages::UserMediaCaptureManager::messageReceiverName(), *this);
56}
57
58UserMediaCaptureManager::~UserMediaCaptureManager()
59{
60 RealtimeMediaSourceCenter::singleton().unsetAudioCaptureFactory(m_audioFactory);
61 RealtimeMediaSourceCenter::singleton().unsetDisplayCaptureFactory(m_displayFactory);
62 RealtimeMediaSourceCenter::singleton().unsetVideoCaptureFactory(m_videoFactory);
63 m_process.removeMessageReceiver(Messages::UserMediaCaptureManager::messageReceiverName());
64 m_remoteCaptureSampleManager.stopListeningForIPC();
65}

72void UserMediaCaptureManager::setupCaptureProcesses(bool shouldCaptureAudioInUIProcess, bool shouldCaptureAudioInGPUProcess, bool shouldCaptureVideoInUIProcess, bool shouldCaptureVideoInGPUProcess, bool shouldCaptureDisplayInUIProcess)
73{
74 MockRealtimeMediaSourceCenter::singleton().setMockAudioCaptureEnabled(!shouldCaptureAudioInUIProcess && !shouldCaptureAudioInGPUProcess);
75 MockRealtimeMediaSourceCenter::singleton().setMockVideoCaptureEnabled(!shouldCaptureVideoInUIProcess && !shouldCaptureVideoInGPUProcess);
76 MockRealtimeMediaSourceCenter::singleton().setMockDisplayCaptureEnabled(!shouldCaptureDisplayInUIProcess);
77
78 m_audioFactory.setShouldCaptureInGPUProcess(shouldCaptureAudioInGPUProcess);
79 m_videoFactory.setShouldCaptureInGPUProcess(shouldCaptureVideoInGPUProcess);
80
81 if (shouldCaptureAudioInGPUProcess) {
82 WebCore::AudioMediaStreamTrackRendererUnit::setCreateInternalUnitFunction([](auto&& renderCallback) {
83 return WebProcess::singleton().audioMediaStreamTrackRendererInternalUnitManager().createRemoteInternalUnit(WTFMove(renderCallback));
84 });
85 }
86
87 if (shouldCaptureAudioInUIProcess || shouldCaptureAudioInGPUProcess)
88 RealtimeMediaSourceCenter::singleton().setAudioCaptureFactory(m_audioFactory);
89 if (shouldCaptureVideoInUIProcess || shouldCaptureVideoInGPUProcess)
90 RealtimeMediaSourceCenter::singleton().setVideoCaptureFactory(m_videoFactory);
91 if (shouldCaptureDisplayInUIProcess)
92 RealtimeMediaSourceCenter::singleton().setDisplayCaptureFactory(m_displayFactory);
93}

100}

165CaptureSourceOrError UserMediaCaptureManager::AudioFactory::createAudioCaptureSource(const CaptureDevice& device, String&& hashSalt, const MediaConstraints* constraints)
166{
167#if !ENABLE(GPU_PROCESS)
168 if (m_shouldCaptureInGPUProcess)
169 return CaptureSourceOrError { "Audio capture in GPUProcess is not implemented"_s };
170#endif
171
172#if PLATFORM(IOS_FAMILY) || ENABLE(ROUTING_ARBITRATION)
173 // FIXME: Remove disabling of the audio session category management once we move all media playing to GPUProcess.
174 if (m_shouldCaptureInGPUProcess)
175 DeprecatedGlobalSettings::setShouldManageAudioSessionCategory(true);
176#endif
177
178 return RemoteRealtimeAudioSource::create(device, constraints, { }, WTFMove(hashSalt), m_manager, m_shouldCaptureInGPUProcess);
179}
180
181void UserMediaCaptureManager::AudioFactory::setShouldCaptureInGPUProcess(bool value)
182{
183 m_shouldCaptureInGPUProcess = value;
184}

186CaptureSourceOrError UserMediaCaptureManager::VideoFactory::createVideoCaptureSource(const CaptureDevice& device, String&& hashSalt, const MediaConstraints* constraints)
187{
188#if !ENABLE(GPU_PROCESS)
189 if (m_shouldCaptureInGPUProcess)
190 return CaptureSourceOrError { "Video capture in GPUProcess is not implemented"_s };
191#endif
192
193 return CaptureSourceOrError(RealtimeVideoSource::create(RemoteRealtimeVideoSource::create(device, constraints, { }, WTFMove(hashSalt), m_manager, m_shouldCaptureInGPUProcess)));
194}
195
196CaptureSourceOrError UserMediaCaptureManager::DisplayFactory::createDisplayCaptureSource(const CaptureDevice& device, const MediaConstraints* constraints)
197{
198 return CaptureSourceOrError(RealtimeVideoSource::create(RemoteRealtimeVideoSource::create(device, constraints, { }, { }, m_manager, false)));
199}
