62static FloatRect inlineVideoFrame(HTMLVideoElement& element)
63{
64 auto& document = element.document();
65 if (!document.hasLivingRenderTree() || document.activeDOMObjectsAreStopped())
66 return { };
67
68 document.updateLayoutIgnorePendingStylesheets();
69 auto* renderer = element.renderer();
70 if (!renderer)
71 return { };
72
73 if (renderer->hasLayer() && renderer->enclosingLayer()->isComposited()) {
74 FloatQuad contentsBox = static_cast<FloatRect>(renderer->enclosingLayer()->backing()->contentsBox());
75 contentsBox = renderer->localToAbsoluteQuad(contentsBox);
76 return element.document().view()->contentsToRootView(contentsBox.boundingBox());
77 }
78
79 auto rect = renderer->videoBox();
80 rect.moveBy(renderer->absoluteBoundingBoxRect().location());
81 return element.document().view()->contentsToRootView(rect);
82}

127VideoFullscreenManager::~VideoFullscreenManager()
128{
129 for (auto& [model, interface] : m_contextMap.values()) {
130 model->setVideoElement(nullptr);
131 model->removeClient(*interface);
132
133 interface->invalidate();
134 }
135
136 m_contextMap.clear();
137 m_videoElements.clear();
138 m_clientCounts.clear();
139
140 if (m_page)
141 WebProcess::singleton().removeMessageReceiver(Messages::VideoFullscreenManager::messageReceiverName(), m_page->identifier());
142}

186void VideoFullscreenManager::removeContext(PlaybackSessionContextIdentifier contextId)
187{
188 auto [model, interface] = ensureModelAndInterface(contextId);
189
190 m_playbackSessionManager->removeClientForContext(contextId);
191
192 RefPtr<HTMLVideoElement> videoElement = model->videoElement();
193 model->setVideoElement(nullptr);
194 model->removeClient(*interface);
195 interface->invalidate();
196 m_videoElements.remove(videoElement.get());
197 m_contextMap.remove(contextId);
198}
199
200void VideoFullscreenManager::addClientForContext(PlaybackSessionContextIdentifier contextId)
201{
202 auto addResult = m_clientCounts.add(contextId, 1);
203 if (!addResult.isNewEntry)
204 addResult.iterator->value++;
205}
206
207void VideoFullscreenManager::removeClientForContext(PlaybackSessionContextIdentifier contextId)
208{
209 ASSERT(m_clientCounts.contains(contextId));
210
211 int clientCount = m_clientCounts.get(contextId);
212 ASSERT(clientCount > 0);
213 clientCount--;
214
215 if (clientCount <= 0) {
216 m_clientCounts.remove(contextId);
217 removeContext(contextId);
218 return;
219 }
220
221 m_clientCounts.set(contextId, clientCount);
222}

249void VideoFullscreenManager::enterVideoFullscreenForVideoElement(HTMLVideoElement& videoElement, HTMLMediaElementEnums::VideoFullscreenMode mode, bool standby)
250{
251 ASSERT(m_page);
252 ASSERT(standby || mode != HTMLMediaElementEnums::VideoFullscreenModeNone);
253 LOG(Fullscreen, "VideoFullscreenManager::enterVideoFullscreenForVideoElement(%p)", this);
254
255 auto contextId = m_playbackSessionManager->contextIdForMediaElement(videoElement);
256 auto addResult = m_videoElements.add(&videoElement, contextId);
257 UNUSED_PARAM(addResult);
258 ASSERT(addResult.iterator->value == contextId);
259
260 auto [model, interface] = ensureModelAndInterface(contextId);
261 HTMLMediaElementEnums::VideoFullscreenMode oldMode = interface->fullscreenMode();
262
263 if (oldMode == HTMLMediaElementEnums::VideoFullscreenModeNone)
264 addClientForContext(contextId);
265
266 if (!interface->layerHostingContext())
267 interface->setLayerHostingContext(LayerHostingContext::createForExternalHostingProcess());
268
269 auto videoRect = inlineVideoFrame(videoElement);
270 FloatRect videoLayerFrame = FloatRect(0, 0, videoRect.width(), videoRect.height());
271
272 if (mode != HTMLMediaElementEnums::VideoFullscreenModeNone)
273 interface->setTargetIsFullscreen(true);
274 else
275 interface->setTargetIsFullscreen(false);
276
277 if (mode == HTMLMediaElementEnums::VideoFullscreenModePictureInPicture)
278 m_videoElementInPictureInPicture = makeWeakPtr(videoElement);
279
280 interface->setFullscreenMode(mode);
281 interface->setFullscreenStandby(standby);
282 model->setVideoElement(&videoElement);
283 if (oldMode == HTMLMediaElementEnums::VideoFullscreenModeNone && mode != HTMLMediaElementEnums::VideoFullscreenModeNone)
284 model->setVideoLayerFrame(videoLayerFrame);
285
286 if (interface->animationState() != VideoFullscreenInterfaceContext::AnimationType::None)
287 return;
288 interface->setAnimationState(VideoFullscreenInterfaceContext::AnimationType::IntoFullscreen);
289
290 bool allowsPictureInPicture = videoElement.webkitSupportsPresentationMode(HTMLVideoElement::VideoPresentationMode::PictureInPicture);
291
292 if (!interface->layerHostingContext()->rootLayer()) {
293 auto videoLayer = model->createVideoFullscreenLayer();
294 [videoLayer setDelegate:[WebActionDisablingCALayerDelegate shared]];
295 [videoLayer setName:@"Web Video Fullscreen Layer"];
296 [videoLayer setPosition:CGPointMake(0, 0)];
297 [videoLayer setBackgroundColor:cachedCGColor(WebCore::Color::transparentBlack)];
298
299 // Set a scale factor here to make convertRect:toLayer:nil take scale factor into account. <rdar://problem/18316542>.
300 // This scale factor is inverted in the hosting process.
301 float hostingScaleFactor = m_page->deviceScaleFactor();
302 [videoLayer setTransform:CATransform3DMakeScale(hostingScaleFactor, hostingScaleFactor, 1)];
303
304 interface->layerHostingContext()->setRootLayer(videoLayer.get());
305 }
306
307 m_page->send(Messages::VideoFullscreenManagerProxy::SetupFullscreenWithID(contextId, interface->layerHostingContext()->contextID(), videoRect, FloatSize(videoElement.videoWidth(), videoElement.videoHeight()), m_page->deviceScaleFactor(), interface->fullscreenMode(), allowsPictureInPicture, standby, videoElement.document().quirks().blocksReturnToFullscreenFromPictureInPictureQuirk()));
308}

310void VideoFullscreenManager::exitVideoFullscreenForVideoElement(HTMLVideoElement& videoElement, CompletionHandler<void(bool)>&& completionHandler)
311{
312 LOG(Fullscreen, "VideoFullscreenManager::exitVideoFullscreenForVideoElement(%p)", this);
313 ASSERT(m_page);
314 ASSERT(m_videoElements.contains(&videoElement));
315
316 auto contextId = m_videoElements.get(&videoElement);
317 auto& interface = ensureInterface(contextId);
318 if (interface.animationState() != VideoFullscreenInterfaceContext::AnimationType::None) {
319 completionHandler(false);
320 return;
321 }
322
323 m_page->sendWithAsyncReply(Messages::VideoFullscreenManagerProxy::ExitFullscreen(contextId, inlineVideoFrame(videoElement)), [protectedThis = makeRefPtr(this), this, contextId, videoElementPtr = &videoElement, completionHandler = WTFMove(completionHandler)](auto success) mutable {
324 if (!success) {
325 completionHandler(false);
326 return;
327 }
328
329 if (m_videoElementInPictureInPicture == videoElementPtr)
330 m_videoElementInPictureInPicture = nullptr;
331
332 auto& interface = ensureInterface(contextId);
333 interface.setTargetIsFullscreen(false);
334 interface.setAnimationState(VideoFullscreenInterfaceContext::AnimationType::FromFullscreen);
335 completionHandler(true);
336 });
337}

339void VideoFullscreenManager::exitVideoFullscreenToModeWithoutAnimation(HTMLVideoElement& videoElement, WebCore::HTMLMediaElementEnums::VideoFullscreenMode targetMode)
340{
341 LOG(Fullscreen, "VideoFullscreenManager::exitVideoFullscreenToModeWithoutAnimation(%p)", this);
342
343 ASSERT(m_page);
344 ASSERT(m_videoElements.contains(&videoElement));
345
346 if (m_videoElementInPictureInPicture == &videoElement)
347 m_videoElementInPictureInPicture = nullptr;
348
349 auto contextId = m_videoElements.get(&videoElement);
350 auto& interface = ensureInterface(contextId);
351
352 interface.setTargetIsFullscreen(false);
353
354 // didCleanupFullscreen() will call removeClientForContext() on Mac
355#if PLATFORM(IOS_FAMILY)
356 removeClientForContext(contextId);
357#endif
358
359 m_page->send(Messages::VideoFullscreenManagerProxy::ExitFullscreenWithoutAnimationToMode(contextId, targetMode));
360}

467void VideoFullscreenManager::didEnterFullscreen(PlaybackSessionContextIdentifier contextId, std::optional<WebCore::FloatSize> size)
468{
469 LOG(Fullscreen, "VideoFullscreenManager::didEnterFullscreen(%p, %x)", this, contextId);
470
471 auto [model, interface] = ensureModelAndInterface(contextId);
472
473 interface->setAnimationState(VideoFullscreenInterfaceContext::AnimationType::None);
474 interface->setIsFullscreen(false);
475
476 RefPtr<HTMLVideoElement> videoElement = model->videoElement();
477 if (!videoElement)
478 return;
479
480 videoElement->didEnterFullscreenOrPictureInPicture(size.value_or(WebCore::FloatSize()));
481
482 if (interface->targetIsFullscreen() || interface->fullscreenStandby())
483 return;
484
485 // exit fullscreen now if it was previously requested during an animation.
486 RunLoop::main().dispatch([protectedThis = makeRefPtr(this), videoElement] {
487 if (protectedThis->m_page)
488 protectedThis->exitVideoFullscreenForVideoElement(*videoElement, [](bool) { });
489 });
490}

492void VideoFullscreenManager::didExitFullscreen(PlaybackSessionContextIdentifier contextId)
493{
494 LOG(Fullscreen, "VideoFullscreenManager::didExitFullscreen(%p, %x)", this, contextId);
495
496 RefPtr<VideoFullscreenModelVideoElement> model;
497 RefPtr<VideoFullscreenInterfaceContext> interface;
498 std::tie(model, interface) = ensureModelAndInterface(contextId);
499
500#if PLATFORM(IOS_FAMILY)
501 RunLoop::main().dispatch([protectedThis = makeRefPtr(this), contextId, interface] {
502 if (protectedThis->m_page)
503 protectedThis->m_page->send(Messages::VideoFullscreenManagerProxy::CleanupFullscreen(contextId));
504 });
505#else
506 model->waitForPreparedForInlineThen([protectedThis = makeRefPtr(this), contextId, interface, model]() mutable {
507 RunLoop::main().dispatch([protectedThis = WTFMove(protectedThis), contextId, interface, model] () mutable {
508 model->setVideoFullscreenLayer(nil, [protectedThis = WTFMove(protectedThis), contextId, interface] () mutable {
509 RunLoop::main().dispatch([protectedThis = WTFMove(protectedThis), contextId, interface] {
510 if (interface->layerHostingContext()) {
511 interface->layerHostingContext()->setRootLayer(nullptr);
512 interface->setLayerHostingContext(nullptr);
513 }
514 if (protectedThis->m_page)
515 protectedThis->m_page->send(Messages::VideoFullscreenManagerProxy::CleanupFullscreen(contextId));
516 });
517 });
518 });
519 });
520#endif
521}

523void VideoFullscreenManager::didCleanupFullscreen(PlaybackSessionContextIdentifier contextId)
524{
525 LOG(Fullscreen, "VideoFullscreenManager::didCleanupFullscreen(%p, %x)", this, contextId);
526
527 auto [model, interface] = ensureModelAndInterface(contextId);
528
529 if (interface->layerHostingContext()) {
530 interface->layerHostingContext()->setRootLayer(nullptr);
531 interface->setLayerHostingContext(nullptr);
532 }
533
534 interface->setAnimationState(VideoFullscreenInterfaceContext::AnimationType::None);
535 interface->setIsFullscreen(false);
536 HTMLMediaElementEnums::VideoFullscreenMode mode = interface->fullscreenMode();
537 bool standby = interface->fullscreenStandby();
538 bool targetIsFullscreen = interface->targetIsFullscreen();
539
540 model->setVideoFullscreenLayer(nil);
541 RefPtr<HTMLVideoElement> videoElement = model->videoElement();
542 if (videoElement)
543 videoElement->didExitFullscreenOrPictureInPicture();
544
545 interface->setFullscreenMode(HTMLMediaElementEnums::VideoFullscreenModeNone);
546 interface->setFullscreenStandby(false);
547 removeClientForContext(contextId);
548
549 if (!videoElement || !targetIsFullscreen)
550 return;
551
552 RunLoop::main().dispatch([protectedThis = makeRefPtr(this), videoElement, mode, standby] {
553 if (protectedThis->m_page)
554 protectedThis->enterVideoFullscreenForVideoElement(*videoElement, mode, standby);
555 });
556}

580void VideoFullscreenManager::setVideoLayerFrameFenced(PlaybackSessionContextIdentifier contextId, WebCore::FloatRect bounds, const WTF::MachSendRight& machSendRight)
581{
582 LOG(Fullscreen, "VideoFullscreenManager::setVideoLayerFrameFenced(%p, %x)", this, contextId);
583
584 auto [model, interface] = ensureModelAndInterface(contextId);
585
586 if (std::isnan(bounds.x()) || std::isnan(bounds.y()) || std::isnan(bounds.width()) || std::isnan(bounds.height())) {
587 auto videoRect = inlineVideoFrame(*model->videoElement());
588 bounds = FloatRect(0, 0, videoRect.width(), videoRect.height());
589 }
590
591 if (auto* context = interface->layerHostingContext())
592 context->setFencePort(machSendRight.sendRight());
593 model->setVideoLayerFrame(bounds);
594}
