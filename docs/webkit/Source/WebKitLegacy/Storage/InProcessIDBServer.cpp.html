59InProcessIDBServer::~InProcessIDBServer()
60{
61 BinarySemaphore semaphore;
62 dispatchTask([this, &semaphore] {
63 {
64 Locker locker { m_serverLock };
65 m_server = nullptr;
66 }
67 m_connectionToClient = nullptr;
68 semaphore.signal();
69 });
70 semaphore.wait();
71}

73StorageQuotaManager* InProcessIDBServer::quotaManager(const ClientOrigin& origin)
74{
75 return m_quotaManagers.ensure(origin, [] {
76 return StorageQuotaManager::create(StorageQuotaManager::defaultQuota(), [] {
77 return 0;
78 }, [](uint64_t quota, uint64_t currentSpace, uint64_t spaceIncrease, auto callback) {
79 callback(quota + currentSpace + spaceIncrease);
80 });
81 }).iterator->value.get();
82}
83
84static inline IDBServer::IDBServer::StorageQuotaManagerSpaceRequester storageQuotaManagerSpaceRequester(InProcessIDBServer& server)
85{
86 return [server = &server, weakServer = makeWeakPtr(server)](const ClientOrigin& origin, uint64_t spaceRequested) mutable {
87 auto* storageQuotaManager = weakServer ? server->quotaManager(origin) : nullptr;
88 return storageQuotaManager ? storageQuotaManager->requestSpaceOnBackgroundThread(spaceRequested) : StorageQuotaManager::Decision::Deny;
89 };
90}

92InProcessIDBServer::InProcessIDBServer(PAL::SessionID sessionID, const String& databaseDirectoryPath)
93 : m_queue(WorkQueue::create("com.apple.WebKit.IndexedDBServer"))
94{
95 ASSERT(isMainThread());
96 m_connectionToServer = IDBClient::IDBConnectionToServer::create(*this);
97 dispatchTask([this, protectedThis = makeRef(*this), sessionID, directory = databaseDirectoryPath.isolatedCopy(), spaceRequester = storageQuotaManagerSpaceRequester(*this)] () mutable {
98 m_connectionToClient = IDBServer::IDBConnectionToClient::create(*this);
99
100 Locker locker { m_serverLock };
101 m_server = makeUnique<IDBServer::IDBServer>(sessionID, directory, WTFMove(spaceRequester), m_serverLock);
102 m_server->registerConnection(*m_connectionToClient);
103 });
104}

500void InProcessIDBServer::dispatchTask(Function<void()>&& function)
501{
502 ASSERT(isMainThread());
503 m_queue->dispatch(WTFMove(function));
504}

506void InProcessIDBServer::dispatchTaskReply(Function<void()>&& function)
507{
508 ASSERT(!isMainThread());
509 callOnMainThread(WTFMove(function));
510}
