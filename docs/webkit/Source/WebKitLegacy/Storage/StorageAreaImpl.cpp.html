47inline StorageAreaImpl::StorageAreaImpl(StorageType storageType, const SecurityOriginData& origin, RefPtr<StorageSyncManager>&& syncManager, unsigned quota)
48 : m_storageType(storageType)
49 , m_securityOrigin(origin)
50 , m_storageMap(quota)
51 , m_storageSyncManager(WTFMove(syncManager))
52 , m_accessCount(0)
53 , m_closeDatabaseTimer(*this, &StorageAreaImpl::closeDatabaseTimerFired)
54{
55 ASSERT(isMainThread());
56
57 // Accessing the shared global StorageTracker when a StorageArea is created
58 // ensures that the tracker is properly initialized before anyone actually needs to use it.
59 StorageTracker::tracker();
60}

62Ref<StorageAreaImpl> StorageAreaImpl::create(StorageType storageType, const SecurityOriginData& origin, RefPtr<StorageSyncManager>&& syncManager, unsigned quota)
63{
64 Ref<StorageAreaImpl> area = adoptRef(*new StorageAreaImpl(storageType, origin, WTFMove(syncManager), quota));
65 // FIXME: If there's no backing storage for LocalStorage, the default WebKit behavior should be that of private browsing,
66 // not silently ignoring it. https://bugs.webkit.org/show_bug.cgi?id=25894
67 if (area->m_storageSyncManager) {
68 area->m_storageAreaSync = StorageAreaSync::create(area->m_storageSyncManager.get(), area.copyRef(), area->m_securityOrigin.databaseIdentifier());
69 ASSERT(area->m_storageAreaSync);
70 }
71 return area;
72}

124void StorageAreaImpl::setItem(Frame* sourceFrame, const String& key, const String& value, bool& quotaException)
125{
126 ASSERT(!m_isShutdown);
127 ASSERT(!value.isNull());
128 blockUntilImportComplete();
129
130 String oldValue;
131 m_storageMap.setItem(key, value, oldValue, quotaException);
132 if (quotaException)
133 return;
134
135 if (oldValue == value)
136 return;
137
138 if (m_storageAreaSync)
139 m_storageAreaSync->scheduleItemForSync(key, value);
140
141 dispatchStorageEvent(key, oldValue, value, sourceFrame);
142}
143
144void StorageAreaImpl::removeItem(Frame* sourceFrame, const String& key)
145{
146 ASSERT(!m_isShutdown);
147 blockUntilImportComplete();
148
149 String oldValue;
150 m_storageMap.removeItem(key, oldValue);
151 if (oldValue.isNull())
152 return;
153
154 if (m_storageAreaSync)
155 m_storageAreaSync->scheduleItemForSync(key, String());
156
157 dispatchStorageEvent(key, oldValue, String(), sourceFrame);
158}

160void StorageAreaImpl::clear(Frame* sourceFrame)
161{
162 ASSERT(!m_isShutdown);
163 blockUntilImportComplete();
164
165 if (!m_storageMap.length())
166 return;
167
168 m_storageMap.clear();
169
170 if (m_storageAreaSync)
171 m_storageAreaSync->scheduleClear();
172
173 dispatchStorageEvent(String(), String(), String(), sourceFrame);
174}

235void StorageAreaImpl::incrementAccessCount()
236{
237 m_accessCount++;
238
239 if (m_closeDatabaseTimer.isActive())
240 m_closeDatabaseTimer.stop();
241}
242
243void StorageAreaImpl::decrementAccessCount()
244{
245 ASSERT(m_accessCount);
246 --m_accessCount;
247
248 if (!m_accessCount) {
249 if (m_closeDatabaseTimer.isActive())
250 m_closeDatabaseTimer.stop();
251 m_closeDatabaseTimer.startOneShot(StorageTracker::tracker().storageDatabaseIdleInterval());
252 }
253}
254
255void StorageAreaImpl::closeDatabaseTimerFired()
256{
257 blockUntilImportComplete();
258 if (m_storageAreaSync)
259 m_storageAreaSync->scheduleCloseDatabase();
260}
261
262void StorageAreaImpl::closeDatabaseIfIdle()
263{
264 if (m_closeDatabaseTimer.isActive()) {
265 ASSERT(!m_accessCount);
266 m_closeDatabaseTimer.stop();
267
268 closeDatabaseTimerFired();
269 }
270}

272void StorageAreaImpl::dispatchStorageEvent(const String& key, const String& oldValue, const String& newValue, Frame* sourceFrame)
273{
274 if (isLocalStorage(m_storageType))
275 StorageEventDispatcher::dispatchLocalStorageEvents(key, oldValue, newValue, m_securityOrigin, sourceFrame);
276 else
277 StorageEventDispatcher::dispatchSessionStorageEvents(key, oldValue, newValue, m_securityOrigin, sourceFrame);
278}

280void StorageAreaImpl::sessionChanged(bool isNewSessionPersistent)
281{
282 ASSERT(isMainThread());
283
284 // If import is not completed, background storage thread may be modifying m_storageMap.
285 blockUntilImportComplete();
286
287 m_storageMap.clear();
288
289 if (isNewSessionPersistent && !m_storageAreaSync && m_storageSyncManager) {
290 m_storageAreaSync = StorageAreaSync::create(m_storageSyncManager.get(), *this, m_securityOrigin.databaseIdentifier());
291 return;
292 }
293
294 if (!isNewSessionPersistent && m_storageAreaSync) {
295 m_storageAreaSync->scheduleFinalSync();
296 m_storageAreaSync = nullptr;
297 }
298}
