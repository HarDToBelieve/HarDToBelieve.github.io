52// If there is no document referencing a storage database, close the underlying database
53// after it has been idle for m_StorageDatabaseIdleInterval seconds.
54static const Seconds defaultStorageDatabaseIdleInterval { 300_s };

56void StorageTracker::initializeTracker(const String& storagePath, StorageTrackerClient* client)
57{
58 ASSERT(isMainThread());
59 ASSERT(!storageTracker || !storageTracker->m_client);
60
61 if (!storageTracker)
62 storageTracker = new StorageTracker(storagePath);
63
64 storageTracker->m_client = client;
65 storageTracker->m_needsInitialization = true;
66}
67
68void StorageTracker::internalInitialize()
69{
70 m_needsInitialization = false;
71
72 ASSERT(isMainThread());
73
74 // Make sure text encoding maps have been built on the main thread, as the StorageTracker thread might try to do it there instead.
75 // FIXME (<rdar://problem/9127819>): Is there a more explicit way of doing this besides accessing the UTF8Encoding?
76 UTF8Encoding();
77
78 storageTracker->setIsActive(true);
79 storageTracker->m_thread->start();
80 storageTracker->importOriginIdentifiers();
81
82 m_thread->dispatch([this] {
83 FileSystem::deleteFile(FileSystem::pathByAppendingComponent(m_storageDirectoryPath, "StorageTracker.db"));
84 });
85}
86
87StorageTracker& StorageTracker::tracker()
88{
89 if (!storageTracker)
90 storageTracker = new StorageTracker("");
91 if (storageTracker->m_needsInitialization)
92 storageTracker->internalInitialize();
93
94 return *storageTracker;
95}

121void StorageTracker::openTrackerDatabase(bool createIfDoesNotExist)
122{
123 ASSERT(m_isActive);
124 ASSERT(!isMainThread());
125
126 SQLiteTransactionInProgressAutoCounter transactionCounter;
127
128 ASSERT(!m_databaseMutex.tryLock());
129
130 if (m_database.isOpen())
131 return;
132
133 String databasePath = trackerDatabasePath();
134
135 if (!ensureDatabaseFileExists(databasePath, createIfDoesNotExist)) {
136 if (createIfDoesNotExist)
137 LOG_ERROR("Failed to create database file '%s'", databasePath.ascii().data());
138 return;
139 }
140
141 if (!m_database.open(databasePath)) {
142 LOG_ERROR("Failed to open databasePath %s.", databasePath.ascii().data());
143 return;
144 }
145
146 m_database.disableThreadingChecks();
147
148 if (!m_database.tableExists("Origins")) {
149 if (!m_database.executeCommand("CREATE TABLE Origins (origin TEXT UNIQUE ON CONFLICT REPLACE, path TEXT);"_s))
150 LOG_ERROR("Failed to create Origins table.");
151 }
152}

173
174void StorageTracker::syncImportOriginIdentifiers()
175{
176 ASSERT(m_isActive);
177
178 ASSERT(!isMainThread());
179
180 {
181 Locker locker { m_databaseMutex };
182
183 // Don't force creation of StorageTracker's db just because a tracker
184 // was initialized. It will be created if local storage dbs are found
185 // by syncFileSystemAndTrackerDatabse() or the next time a local storage
186 // db is created by StorageAreaSync.
187 openTrackerDatabase(false);
188
189 if (m_database.isOpen()) {
190 SQLiteTransactionInProgressAutoCounter transactionCounter;
191
192 auto statement = m_database.prepareStatement("SELECT origin FROM Origins"_s);
193 if (!statement) {
194 LOG_ERROR("Failed to prepare statement.");
195 return;
196 }
197
198 int result;
199
200 {
201 Locker lockOrigins { m_originSetMutex };
202 while ((result = statement->step()) == SQLITE_ROW)
203 m_originSet.add(statement->columnText(0).isolatedCopy());
204 }
205
206 if (result != SQLITE_DONE) {
207 LOG_ERROR("Failed to read in all origins from the database.");
208 return;
209 }
210 }
211 }
212
213 syncFileSystemAndTrackerDatabase();
214
215 {
216 Locker locker { m_clientMutex };
217
218 if (m_client) {
219 Locker locker { m_originSetMutex };
220 OriginSet::const_iterator end = m_originSet.end();
221 for (OriginSet::const_iterator it = m_originSet.begin(); it != end; ++it)
222 m_client->dispatchDidModifyOrigin(*it);
223 }
224 }
225
226 callOnMainThread([this] {
227 finishedImportingOriginIdentifiers();
228 });
229}

231void StorageTracker::syncFileSystemAndTrackerDatabase()
232{
233 ASSERT(!isMainThread());
234
235 SQLiteTransactionInProgressAutoCounter transactionCounter;
236
237 ASSERT(m_isActive);
238
239 Vector<String> fileNames;
240 {
241 Locker locker { m_databaseMutex };
242 fileNames = FileSystem::listDirectory(m_storageDirectoryPath);
243 }
244
245 // Use a copy of m_originSet to find expired entries and to schedule their
246 // deletions from disk and from m_originSet.
247 OriginSet originSetCopy;
248 {
249 Locker locker { m_originSetMutex };
250 for (OriginSet::const_iterator it = m_originSet.begin(), end = m_originSet.end(); it != end; ++it)
251 originSetCopy.add((*it).isolatedCopy());
252 }
253
254 // Add missing StorageTracker records.
255 OriginSet foundOrigins;
256 String fileExtension = ".localstorage"_s;
257
258 for (auto& fileName : fileNames) {
259 if (fileName.length() <= fileExtension.length() || !fileName.endsWith(fileExtension))
260 continue;
261
262 auto filePath = FileSystem::pathByAppendingComponent(m_storageDirectoryPath, fileName);
263 String originIdentifier = fileName.substring(0, fileName.length() - fileExtension.length());
264 if (!originSetCopy.contains(originIdentifier))
265 syncSetOriginDetails(originIdentifier, filePath);
266
267 foundOrigins.add(originIdentifier);
268 }
269
270 // Delete stale StorageTracker records.
271 for (OriginSet::const_iterator it = originSetCopy.begin(), end = originSetCopy.end(); it != end; ++it) {
272 const String& originIdentifier = *it;
273 if (foundOrigins.contains(originIdentifier))
274 continue;
275
276 callOnMainThread([this, originIdentifier = originIdentifier.isolatedCopy()] {
277 deleteOriginWithIdentifier(originIdentifier);
278 });
279 }
280}

282void StorageTracker::setOriginDetails(const String& originIdentifier, const String& databaseFile)
283{
284 if (!m_isActive)
285 return;
286
287 {
288 Locker locker { m_originSetMutex };
289
290 if (m_originSet.contains(originIdentifier))
291 return;
292
293 m_originSet.add(originIdentifier);
294 }
295
296 auto function = [this, originIdentifier = originIdentifier.isolatedCopy(), databaseFile = databaseFile.isolatedCopy()] {
297 syncSetOriginDetails(originIdentifier, databaseFile);
298 };
299
300 // FIXME: This weird ping-ponging was done to fix a deadlock. We should figure out a cleaner way to avoid it instead.
301 ensureOnMainThread([this, function = WTFMove(function)]() mutable {
302 m_thread->dispatch(WTFMove(function));
303 });
304}

306void StorageTracker::syncSetOriginDetails(const String& originIdentifier, const String& databaseFile)
307{
308 ASSERT(!isMainThread());
309
310 SQLiteTransactionInProgressAutoCounter transactionCounter;
311
312 Locker locker { m_databaseMutex };
313
314 openTrackerDatabase(true);
315
316 if (!m_database.isOpen())
317 return;
318
319 auto statement = m_database.prepareStatement("INSERT INTO Origins VALUES (?, ?)"_s);
320 if (!statement) {
321 LOG_ERROR("Unable to establish origin '%s' in the tracker", originIdentifier.ascii().data());
322 return;
323 }
324
325 statement->bindText(1, originIdentifier);
326 statement->bindText(2, databaseFile);
327
328 if (statement->step() != SQLITE_DONE)
329 LOG_ERROR("Unable to establish origin '%s' in the tracker", originIdentifier.ascii().data());
330
331 {
332 Locker locker { m_originSetMutex };
333 if (!m_originSet.contains(originIdentifier))
334 m_originSet.add(originIdentifier);
335 }
336
337 {
338 Locker locker { m_clientMutex };
339 if (m_client)
340 m_client->dispatchDidModifyOrigin(originIdentifier);
341 }
342}

344Vector<SecurityOriginData> StorageTracker::origins()
345{
346 ASSERT(m_isActive);
347
348 if (!m_isActive)
349 return { };
350
351 Locker locker { m_originSetMutex };
352
353 Vector<SecurityOriginData> result;
354 result.reserveInitialCapacity(m_originSet.size());
355 for (auto& identifier : m_originSet) {
356 auto origin = SecurityOriginData::fromDatabaseIdentifier(identifier);
357 if (!origin) {
358 ASSERT_NOT_REACHED();
359 continue;
360 }
361 result.uncheckedAppend(origin.value());
362 }
363 return result;
364}

366void StorageTracker::deleteAllOrigins()
367{
368 ASSERT(m_isActive);
369 ASSERT(isMainThread());
370 ASSERT(m_thread);
371
372 if (!m_isActive)
373 return;
374
375 {
376 Locker locker { m_originSetMutex };
377 willDeleteAllOrigins();
378 m_originSet.clear();
379 }
380
381 WebStorageNamespaceProvider::clearLocalStorageForAllOrigins();
382
383 m_thread->dispatch([this] {
384 syncDeleteAllOrigins();
385 });

395void StorageTracker::syncDeleteAllOrigins()
396{
397 ASSERT(!isMainThread());
398
399 SQLiteTransactionInProgressAutoCounter transactionCounter;
400
401 Locker locker { m_databaseMutex };
402
403 openTrackerDatabase(false);
404 if (!m_database.isOpen())
405 return;
406
407 auto statement = m_database.prepareStatement("SELECT origin, path FROM Origins"_s);
408 if (!statement) {
409 LOG_ERROR("Failed to prepare statement.");
410 return;
411 }
412
413 int result;
414 while ((result = statement->step()) == SQLITE_ROW) {
415 if (!canDeleteOrigin(statement->columnText(0)))
416 continue;
417
418 FileSystem::deleteFile(statement->columnText(1));
419
420 {
421 Locker locker { m_clientMutex };
422 if (m_client)
423 m_client->dispatchDidModifyOrigin(statement->columnText(0));
424 }
425 }
426
427 if (result != SQLITE_DONE)
428 LOG_ERROR("Failed to read in all origins from the database.");
429
430 if (m_database.isOpen()) {
431#if PLATFORM(IOS_FAMILY)
432 truncateDatabaseFile(m_database);
433#endif
434 m_database.close();
435 }
436
437#if !PLATFORM(IOS_FAMILY)
438 if (!FileSystem::deleteFile(trackerDatabasePath())) {
439 // In the case where it is not possible to delete the database file (e.g some other program
440 // like a virus scanner is accessing it), make sure to remove all entries.
441 openTrackerDatabase(false);
442 if (!m_database.isOpen())
443 return;
444 auto deleteStatement = m_database.prepareStatement("DELETE FROM Origins"_s);
445 if (!deleteStatement) {
446 LOG_ERROR("Unable to prepare deletion of all origins");
447 return;
448 }
449 if (!deleteStatement->executeCommand()) {
450 LOG_ERROR("Unable to execute deletion of all origins");
451 return;
452 }
453 }
454 FileSystem::deleteEmptyDirectory(m_storageDirectoryPath);
455#endif
456}

468void StorageTracker::deleteOrigin(const SecurityOriginData& origin)
469{
470 ASSERT(m_isActive);
471 ASSERT(isMainThread());
472 ASSERT(m_thread);
473
474 if (!m_isActive)
475 return;
476
477 // Before deleting database, we need to clear in-memory local storage data
478 // in StorageArea, and to close the StorageArea db. It's possible for an
479 // item to be added immediately after closing the db and cause StorageAreaSync
480 // to reopen the db before the db is deleted by a StorageTracker thread.
481 // In this case, reopening the db in StorageAreaSync will cancel a pending
482 // StorageTracker db deletion.
483 WebStorageNamespaceProvider::clearLocalStorageForOrigin(origin);
484
485 String originId = origin.databaseIdentifier();
486
487 {
488 Locker locker { m_originSetMutex };
489 willDeleteOrigin(originId);
490 m_originSet.remove(originId);
491 }
492
493 m_thread->dispatch([this, originId = originId.isolatedCopy()] {
494 syncDeleteOrigin(originId);
495 });
496}
497
498void StorageTracker::syncDeleteOrigin(const String& originIdentifier)
499{
500 ASSERT(!isMainThread());
501
502 SQLiteTransactionInProgressAutoCounter transactionCounter;
503
504 Locker locker { m_databaseMutex };
505
506 if (!canDeleteOrigin(originIdentifier)) {
507 LOG_ERROR("Attempted to delete origin '%s' while it was being created\n", originIdentifier.ascii().data());
508 return;
509 }
510
511 openTrackerDatabase(false);
512 if (!m_database.isOpen())
513 return;
514
515 String path = databasePathForOrigin(originIdentifier);
516 if (path.isEmpty()) {
517 // It is possible to get a request from the API to delete the storage for an origin that
518 // has no such storage.
519 return;
520 }
521
522 {
523 auto deleteStatement = m_database.prepareStatement("DELETE FROM Origins where origin=?"_s);
524 if (!deleteStatement) {
525 LOG_ERROR("Unable to prepare deletion of origin '%s'", originIdentifier.ascii().data());
526 return;
527 }
528 deleteStatement->bindText(1, originIdentifier);
529 if (!deleteStatement->executeCommand()) {
530 LOG_ERROR("Unable to execute deletion of origin '%s'", originIdentifier.ascii().data());
531 return;
532 }
533 }
534
535 FileSystem::deleteFile(path);
536
537 bool shouldDeleteTrackerFiles = false;
538 {
539 Locker locker { m_originSetMutex };
540 m_originSet.remove(originIdentifier);
541 shouldDeleteTrackerFiles = m_originSet.isEmpty();
542 }
543
544 if (shouldDeleteTrackerFiles) {
545#if PLATFORM(IOS_FAMILY)
546 truncateDatabaseFile(m_database);
547#endif
548 m_database.close();
549#if !PLATFORM(IOS_FAMILY)
550 FileSystem::deleteFile(trackerDatabasePath());
551 FileSystem::deleteEmptyDirectory(m_storageDirectoryPath);
552#endif
553 }
554
555 {
556 Locker locker { m_clientMutex };
557 if (m_client)
558 m_client->dispatchDidModifyOrigin(originIdentifier);
559 }
560}

609String StorageTracker::databasePathForOrigin(const String& originIdentifier)
610{
611 ASSERT(!m_databaseMutex.tryLock());
612 ASSERT(m_isActive);
613
614 if (!m_database.isOpen())
615 return String();
616
617 SQLiteTransactionInProgressAutoCounter transactionCounter;
618
619 auto pathStatement = m_database.prepareStatement("SELECT path FROM Origins WHERE origin=?"_s);
620 if (!pathStatement) {
621 LOG_ERROR("Unable to prepare selection of path for origin '%s'", originIdentifier.ascii().data());
622 return String();
623 }
624 pathStatement->bindText(1, originIdentifier);
625 int result = pathStatement->step();
626 if (result != SQLITE_ROW)
627 return String();
628
629 return pathStatement->columnText(0);
630}

632uint64_t StorageTracker::diskUsageForOrigin(SecurityOrigin* origin)
633{
634 if (!m_isActive)
635 return 0;
636
637 Locker locker { m_databaseMutex };
638
639 String path = databasePathForOrigin(origin->data().databaseIdentifier());
640 if (path.isEmpty())
641 return 0;
642
643 return FileSystem::fileSize(path).value_or(0);
644}
