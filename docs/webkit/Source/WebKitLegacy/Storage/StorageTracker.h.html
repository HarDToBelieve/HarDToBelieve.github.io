42namespace WebKit {
43
44class StorageTracker {
45 WTF_MAKE_NONCOPYABLE(StorageTracker);
46 WTF_MAKE_FAST_ALLOCATED;
47public:
48 static void initializeTracker(const String& storagePath, WebCore::StorageTrackerClient*);
49 static StorageTracker& tracker();
50
51 void setOriginDetails(const String& originIdentifier, const String& databaseFile);
52
53 void deleteAllOrigins();
54 void deleteOrigin(const WebCore::SecurityOriginData&);
55 void deleteOriginWithIdentifier(const String& originIdentifier);
56 Vector<WebCore::SecurityOriginData> origins();
57 uint64_t diskUsageForOrigin(WebCore::SecurityOrigin*);
58
59 void cancelDeletingOrigin(const String& originIdentifier);
60
61 bool isActive();
62
63 Seconds storageDatabaseIdleInterval() { return m_StorageDatabaseIdleInterval; }
64 void setStorageDatabaseIdleInterval(Seconds interval) { m_StorageDatabaseIdleInterval = interval; }
65
66 void syncFileSystemAndTrackerDatabase();
67
68private:
69 explicit StorageTracker(const String& storagePath);
70
71 void internalInitialize();
72
73 String trackerDatabasePath();
74 void openTrackerDatabase(bool createIfDoesNotExist);
75
76 void importOriginIdentifiers();
77 void finishedImportingOriginIdentifiers();
78
79 void deleteTrackerFiles();
80 String databasePathForOrigin(const String& originIdentifier);
81
82 bool canDeleteOrigin(const String& originIdentifier);
83 void willDeleteOrigin(const String& originIdentifier);
84 void willDeleteAllOrigins();
85
86 void originFilePaths(Vector<String>& paths);
87
88 void setIsActive(bool);
89
90 // Sync to disk on background thread.
91 void syncDeleteAllOrigins();
92 void syncDeleteOrigin(const String& originIdentifier);
93 void syncSetOriginDetails(const String& originIdentifier, const String& databaseFile);
94 void syncImportOriginIdentifiers();
95
96 // Mutex for m_database and m_storageDirectoryPath.
97 Lock m_databaseMutex;
98 WebCore::SQLiteDatabase m_database;
99 String m_storageDirectoryPath;
100
101 Lock m_clientMutex;
102 WebCore::StorageTrackerClient* m_client;
103
104 // Guard for m_originSet and m_originsBeingDeleted.
105 Lock m_originSetMutex;
106 typedef HashSet<String> OriginSet;
107 OriginSet m_originSet;
108 OriginSet m_originsBeingDeleted;
109
110 std::unique_ptr<WebCore::StorageThread> m_thread;
111
112 bool m_isActive;
113 bool m_needsInitialization;
114 Seconds m_StorageDatabaseIdleInterval;
115};
