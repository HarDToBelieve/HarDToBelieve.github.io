37// Any named groups will live for the lifetime of the process, thanks to the reference held by the RefPtr.
38static HashMap<String, RefPtr<WebViewGroup>>& webViewGroups()
39{
40 static NeverDestroyed<HashMap<String, RefPtr<WebViewGroup>>> webViewGroups;
41
42 return webViewGroups;
43}
44
45Ref<WebViewGroup> WebViewGroup::getOrCreate(const String& name, const String& localStorageDatabasePath)
46{
47 if (name.isEmpty())
48 return adoptRef(*new WebViewGroup(String(), localStorageDatabasePath));
49
50 auto& webViewGroup = webViewGroups().add(name, nullptr).iterator->value;
51 if (!webViewGroup) {
52 auto result = adoptRef(*new WebViewGroup(name, localStorageDatabasePath));
53 webViewGroup = result.copyRef();
54 return result;
55 }
56
57 if (!webViewGroup->m_storageNamespaceProvider && webViewGroup->m_localStorageDatabasePath.isEmpty() && !localStorageDatabasePath.isEmpty())
58 webViewGroup->m_localStorageDatabasePath = localStorageDatabasePath;
59
60 return *webViewGroup;
61}
62
63WebViewGroup* WebViewGroup::get(const String& name)
64{
65 ASSERT(!name.isEmpty());
66
67 return webViewGroups().get(name);
68}

70WebViewGroup::WebViewGroup(const String& name, const String& localStorageDatabasePath)
71 : m_name(name)
72 , m_localStorageDatabasePath(localStorageDatabasePath)
73 , m_userContentController(UserContentController::create())
74 , m_visitedLinkStore(WebVisitedLinkStore::create())
75{
76}
77
78WebViewGroup::~WebViewGroup()
79{
80 ASSERT(m_name.isEmpty());
81 ASSERT(m_webViews.isEmpty());
82}
83
84void WebViewGroup::addWebView(WebView *webView)
85{
86 ASSERT(!m_webViews.contains(webView));
87
88 m_webViews.add(webView);
89}
90
91void WebViewGroup::removeWebView(WebView *webView)
92{
93 ASSERT(m_webViews.contains(webView));
94
95 m_webViews.remove(webView);
96}
97
98StorageNamespaceProvider& WebViewGroup::storageNamespaceProvider()
99{
100 if (!m_storageNamespaceProvider)
101 m_storageNamespaceProvider = WebKit::WebStorageNamespaceProvider::create(m_localStorageDatabasePath);
102
103 return *m_storageNamespaceProvider;
104}
