163static Class lookupElementClass(const QualifiedName& tag)
164{
165 // Do a special lookup to ignore element prefixes
166 if (tag.hasPrefix())
167 return elementClassMap->get(QualifiedName(nullAtom(), tag.localName(), tag.namespaceURI()).impl());
168
169 return elementClassMap->get(tag.impl());
170}
171
172static Class elementClass(const QualifiedName& tag, Class defaultClass)
173{
174 if (!elementClassMap)
175 createElementClassMap();
176 Class objcClass = lookupElementClass(tag);
177 if (!objcClass)
178 objcClass = defaultClass;
179 return objcClass;
180}

260Class kitClass(Node* impl)
261{
262 switch (impl->nodeType()) {
263 case Node::ELEMENT_NODE:
264 if (is<HTMLElement>(*impl))
265 return elementClass(downcast<HTMLElement>(*impl).tagQName(), [DOMHTMLElement class]);
266 return [DOMElement class];
267 case Node::ATTRIBUTE_NODE:
268 return [DOMAttr class];
269 case Node::TEXT_NODE:
270 return [DOMText class];
271 case Node::CDATA_SECTION_NODE:
272 return [DOMCDATASection class];
273 case Node::PROCESSING_INSTRUCTION_NODE:
274 return [DOMProcessingInstruction class];
275 case Node::COMMENT_NODE:
276 return [DOMComment class];
277 case Node::DOCUMENT_NODE:
278 if (static_cast<Document*>(impl)->isHTMLDocument())
279 return [DOMHTMLDocument class];
280 return [DOMDocument class];
281 case Node::DOCUMENT_TYPE_NODE:
282 return [DOMDocumentType class];
283 case Node::DOCUMENT_FRAGMENT_NODE:
284 return [DOMDocumentFragment class];
285 }
286 ASSERT_NOT_REACHED();
287 return nil;
288}

290id <DOMEventTarget> kit(EventTarget* target)

294}

447 boundingBox.setHeight(boundingBox.height() - style.borderBottomWidth() - style.borderTopWidth());

451}

517- (void)getPreviewSnapshotImage:(CGImageRef*)cgImage andRects:(NSArray **)rects
518{
519 if (!cgImage || !rects)
520 return;
521
522 *cgImage = nullptr;
523 *rects = nullptr;
524
525 auto& node = *core(self);
526
527 constexpr OptionSet<TextIndicatorOption> options {
528 TextIndicatorOption::TightlyFitContent,
529 TextIndicatorOption::RespectTextColor,
530 TextIndicatorOption::PaintBackgrounds,
531 TextIndicatorOption::UseBoundingRectAndPaintAllContentForComplexRanges,
532 TextIndicatorOption::IncludeMarginIfRangeMatchesSelection
533 };
534 const float margin = 4 / node.document().page()->pageScaleFactor();
535 auto textIndicator = TextIndicator::createWithRange(makeRangeSelectingNodeContents(node), options, TextIndicatorPresentationTransition::None, FloatSize(margin, margin));
536
537 if (textIndicator) {
538 if (Image* image = textIndicator->contentImage()) {
539 auto contentImage = image->nativeImage()->platformImage();
540 *cgImage = contentImage.autorelease();
541 }
542 }
543
544 if (!*cgImage) {
545 if (auto* renderer = node.renderer()) {
546 FloatRect boundingBox;
547 if (renderer->isRenderImage())
548 boundingBox = downcast<RenderImage>(*renderer).absoluteContentQuad().enclosingBoundingBox();
549 else
550 boundingBox = renderer->absoluteBoundingBoxRect();
551 boundingBox.inflate(margin);
552 *rects = @[makeNSArrayElement(node.document().frame()->view()->contentsToWindow(enclosingIntRect(boundingBox)))];
553 }
554 return;
555 }
556
557 FloatPoint origin = textIndicator->textBoundingRectInRootViewCoordinates().location();
558 *rects = createNSArray(textIndicator->textRectsInBoundingRectCoordinates(), [&] (CGRect rect) {
559 rect.origin.x += origin.x();
560 rect.origin.y += origin.y();
561 return makeNSArrayElement(node.document().frame()->view()->contentsToWindow(enclosingIntRect(rect)));
562 }).autorelease();
563}

684- (BOOL)_mediaQueryMatchesForOrientation:(int)orientation
685{
686 Document& document = static_cast<HTMLLinkElement*>(core(self))->document();
687 FrameView* frameView = document.frame() ? document.frame()->view() : 0;
688 if (!frameView)
689 return false;
690 int layoutWidth = frameView->layoutWidth();
691 int layoutHeight = frameView->layoutHeight();
692 IntSize savedFixedLayoutSize = frameView->fixedLayoutSize();
693 bool savedUseFixedLayout = frameView->useFixedLayout();
694 if ((orientation == WebMediaQueryOrientationPortrait && layoutWidth > layoutHeight) ||
695 (orientation == WebMediaQueryOrientationLandscape && layoutWidth < layoutHeight)) {
696 // temporarily swap the orientation for the evaluation
697 frameView->setFixedLayoutSize(IntSize(layoutHeight, layoutWidth));
698 frameView->setUseFixedLayout(true);
699 }
700
701 bool result = [self _mediaQueryMatches];
702
703 frameView->setFixedLayoutSize(savedFixedLayoutSize);
704 frameView->setUseFixedLayout(savedUseFixedLayout);
705
706 return result;
707}
708
709- (BOOL)_mediaQueryMatches
710{
711 HTMLLinkElement& link = *static_cast<HTMLLinkElement*>(core(self));
712
713 auto& media = link.attributeWithoutSynchronization(HTMLNames::mediaAttr);
714 if (media.isEmpty())
