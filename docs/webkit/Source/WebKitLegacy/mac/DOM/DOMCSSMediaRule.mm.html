42#define IMPL static_cast<WebCore::CSSMediaRule*>(reinterpret_cast<WebCore::CSSRule*>(_internal))
43
44@implementation DOMCSSMediaRule
45
46- (DOMMediaList *)media
47{
48 WebCore::JSMainThreadNullState state;
49 return kit(WTF::getPtr(IMPL->media()));
50}
51
52- (DOMCSSRuleList *)cssRules
53{
54 WebCore::JSMainThreadNullState state;
55 return kit(WTF::getPtr(IMPL->cssRules()));
56}
57
58- (unsigned)insertRule:(NSString *)rule index:(unsigned)index
59{
60 WebCore::JSMainThreadNullState state;
61 return raiseOnDOMError(IMPL->insertRule(rule, index));
62}
63
64- (void)deleteRule:(unsigned)index
65{
66 WebCore::JSMainThreadNullState state;
67 return raiseOnDOMError(IMPL->deleteRule(index));
68}
69
70@end

72@implementation DOMCSSMediaRule (DOMCSSMediaRuleDeprecated)
73
74- (unsigned)insertRule:(NSString *)rule :(unsigned)index
75{
76 return [self insertRule:rule index:index];
77}
78
79@end
