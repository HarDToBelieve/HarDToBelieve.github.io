141WebCore::CSSStyleDeclaration* core(DOMCSSStyleDeclaration *wrapper)
142{
143 return wrapper ? reinterpret_cast<WebCore::CSSStyleDeclaration*>(wrapper->_internal) : 0;
144}
145
146DOMCSSStyleDeclaration *kit(WebCore::CSSStyleDeclaration* value)
147{
148 WebCoreThreadViolationCheckRoundOne();
149 if (!value)
150 return nil;
151 if (DOMCSSStyleDeclaration *wrapper = getDOMWrapper(value))
152 return retainPtr(wrapper).autorelease();
153 auto wrapper = adoptNS([[DOMCSSStyleDeclaration alloc] _init]);
154 wrapper->_internal = reinterpret_cast<DOMObjectInternal*>(value);
155 value->ref();
156 addDOMWrapper(wrapper.get(), value);
157 return wrapper.autorelease();
158}
