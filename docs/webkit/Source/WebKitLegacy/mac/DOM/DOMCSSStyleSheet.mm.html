43#define IMPL static_cast<WebCore::CSSStyleSheet*>(reinterpret_cast<WebCore::StyleSheet*>(_internal))
44
45@implementation DOMCSSStyleSheet
46
47- (DOMCSSRule *)ownerRule
48{
49 WebCore::JSMainThreadNullState state;
50 return kit(WTF::getPtr(IMPL->ownerRule()));
51}
52
53- (DOMCSSRuleList *)cssRules
54{
55 WebCore::JSMainThreadNullState state;
56 return kit(WTF::getPtr(IMPL->cssRules()));
57}
58
59- (DOMCSSRuleList *)rules
60{
61 WebCore::JSMainThreadNullState state;
62 // Calling IMPL->cssRules (not IMPL->rules) is intentional, as `rules` should just be an alias for `cssRules`.
63 // See https://bugs.webkit.org/show_bug.cgi?id=197725 for more information.
64 return kit(WTF::getPtr(IMPL->cssRules()));
65}
66
67- (unsigned)insertRule:(NSString *)rule index:(unsigned)index
68{
69 WebCore::JSMainThreadNullState state;
70 return raiseOnDOMError(IMPL->insertRule(rule, index));
71}
72
73- (void)deleteRule:(unsigned)index
74{
75 WebCore::JSMainThreadNullState state;
76 raiseOnDOMError(IMPL->deleteRule(index));
77}
78
79- (int)addRule:(NSString *)selector style:(NSString *)style index:(unsigned)index
80{
81 WebCore::JSMainThreadNullState state;
82 return raiseOnDOMError(IMPL->addRule(selector, style, index));
83}
84
85- (void)removeRule:(unsigned)index
86{
87 WebCore::JSMainThreadNullState state;
88 raiseOnDOMError(IMPL->removeRule(index));
89}
90
91@end
92
93@implementation DOMCSSStyleSheet (DOMCSSStyleSheetDeprecated)
94
95- (unsigned)insertRule:(NSString *)rule :(unsigned)index
96{
97 return [self insertRule:rule index:index];
98}
99
100@end
101
102DOMCSSStyleSheet *kit(WebCore::CSSStyleSheet* value)
103{
104 WebCoreThreadViolationCheckRoundOne();
105 return static_cast<DOMCSSStyleSheet*>(kit(static_cast<WebCore::StyleSheet*>(value)));
106}
107
108#undef IMPL
