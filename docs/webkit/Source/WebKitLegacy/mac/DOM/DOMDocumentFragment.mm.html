26#import "DOMDocumentFragmentInternal.h"
27
28#import "DOMElementInternal.h"
29#import "DOMHTMLCollectionInternal.h"
30#import "ExceptionHandlers.h"
31#import <WebCore/DocumentFragment.h>
32#import <WebCore/JSExecState.h>
33#import <WebCore/ThreadCheck.h>
34
35#define IMPL static_cast<WebCore::DocumentFragment*>(reinterpret_cast<WebCore::Node*>(_internal))
36
37@implementation DOMDocumentFragment
38
39- (DOMHTMLCollection *)children
40{
41 WebCore::JSMainThreadNullState state;
42 return kit(WTF::getPtr(IMPL->children()));
43}
44
45- (DOMElement *)firstElementChild
46{
47 WebCore::JSMainThreadNullState state;
48 return kit(WTF::getPtr(IMPL->firstElementChild()));
49}
50
51- (DOMElement *)lastElementChild
52{
53 WebCore::JSMainThreadNullState state;
54 return kit(WTF::getPtr(IMPL->lastElementChild()));
55}
56
57- (unsigned)childElementCount
58{
59 WebCore::JSMainThreadNullState state;
60 return IMPL->childElementCount();
61}
62
63- (DOMElement *)getElementById:(NSString *)elementId
64{
65 WebCore::JSMainThreadNullState state;
66 return kit(WTF::getPtr(IMPL->getElementById(AtomString(elementId))));
67}
68
69- (DOMElement *)querySelector:(NSString *)selectors
70{
71 WebCore::JSMainThreadNullState state;
72 return kit(raiseOnDOMError(IMPL->querySelector(selectors)));
73}
74
75- (DOMNodeList *)querySelectorAll:(NSString *)selectors
76{
77 WebCore::JSMainThreadNullState state;
78 return kit(raiseOnDOMError(IMPL->querySelectorAll(selectors)).ptr());
79}
80
81@end
82
83WebCore::DocumentFragment* core(DOMDocumentFragment *wrapper)
84{
85 return wrapper ? reinterpret_cast<WebCore::DocumentFragment*>(wrapper->_internal) : 0;
86}
87
88DOMDocumentFragment *kit(WebCore::DocumentFragment* value)
89{
90 WebCoreThreadViolationCheckRoundOne();
91 return static_cast<DOMDocumentFragment*>(kit(static_cast<WebCore::Node*>(value)));
92}
93
94#undef IMPL
