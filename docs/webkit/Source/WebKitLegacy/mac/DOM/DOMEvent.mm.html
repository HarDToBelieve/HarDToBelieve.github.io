180WebCore::Event* core(DOMEvent *wrapper)
181{
182 return wrapper ? reinterpret_cast<WebCore::Event*>(wrapper->_internal) : 0;
183}
184
185DOMEvent *kit(WebCore::Event* value)
186{
187 WebCoreThreadViolationCheckRoundOne();
188 if (!value)
189 return nil;
190 if (DOMEvent *wrapper = getDOMWrapper(value))
191 return retainPtr(wrapper).autorelease();
192 RetainPtr<DOMEvent> wrapper = adoptNS([[kitClass(value) alloc] _init]);
193 if (!wrapper)
194 return nil;
195 wrapper->_internal = reinterpret_cast<DOMObjectInternal*>(value);
196 value->ref();
197 addDOMWrapper(wrapper.get(), value);
198 return wrapper.autorelease();
199}
