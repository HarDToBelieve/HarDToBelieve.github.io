69- (int)scrollXOffset
70{
71 auto* renderer = core(self)->renderer();
72 if (!renderer)
73 return 0;
74
75 if (!is<WebCore::RenderBlockFlow>(*renderer))
76 renderer = renderer->containingBlock();
77
78 if (!is<WebCore::RenderBox>(*renderer) || !renderer->hasOverflowClip())
79 return 0;
80
81 auto* layer = downcast<WebCore::RenderBox>(*renderer).layer();
82 if (!layer)
83 return 0;
84 auto* scrollableArea = layer->scrollableArea();
85 if (!scrollableArea)
86 return 0;
87
88 return scrollableArea->scrollOffset().x();
89}
90
91- (int)scrollYOffset
92{
93 auto* renderer = core(self)->renderer();
94 if (!renderer)
95 return 0;
96
97 if (!is<WebCore::RenderBlockFlow>(*renderer))
98 renderer = renderer->containingBlock();
99 if (!is<WebCore::RenderBox>(*renderer) || !renderer->hasOverflowClip())
100 return 0;
101
102 auto* layer = downcast<WebCore::RenderBox>(*renderer).layer();
103 if (!layer)
104 return 0;
105 auto* scrollableArea = layer->scrollableArea();
106 if (!scrollableArea)
107 return 0;
108
109 return scrollableArea->scrollOffset().y();
110}
111
112- (void)setScrollXOffset:(int)x scrollYOffset:(int)y
113{
114 [self setScrollXOffset:x scrollYOffset:y adjustForIOSCaret:NO];
115}
116
117- (void)setScrollXOffset:(int)x scrollYOffset:(int)y adjustForIOSCaret:(BOOL)adjustForIOSCaret
118{
119 auto* renderer = core(self)->renderer();
120 if (!renderer)
121 return;
122
123 if (!is<WebCore::RenderBlockFlow>(*renderer))
124 renderer = renderer->containingBlock();
125 if (!renderer->hasOverflowClip() || !is<WebCore::RenderBox>(*renderer))
126 return;
127
128 auto* layer = downcast<WebCore::RenderBox>(*renderer).layer();
129 if (!layer)
130 return;
131 auto* scrollableArea = layer->ensureLayerScrollableArea();
132
133 if (adjustForIOSCaret)
134 scrollableArea->setAdjustForIOSCaretWhenScrolling(true);
135
136 auto scrollPositionChangeOptions = WebCore::ScrollPositionChangeOptions::createProgrammatic();
137 scrollPositionChangeOptions.clamping = WebCore::ScrollClamping::Unclamped;
138 scrollableArea->scrollToOffset(WebCore::ScrollOffset(x, y), scrollPositionChangeOptions);
139 if (adjustForIOSCaret)
140 scrollableArea->setAdjustForIOSCaretWhenScrolling(false);
141}
142
143- (void)absolutePosition:(int *)x :(int *)y :(int *)w :(int *)h
144{
145 auto* renderer = core(self)->renderBox();
146 if (renderer) {
147 if (w)
148 *w = renderer->width();
149 if (h)
150 *h = renderer->width();
151 if (x && y) {
152 WebCore::FloatPoint floatPoint(*x, *y);
153 renderer->localToAbsolute(floatPoint);
154 WebCore::IntPoint point = roundedIntPoint(floatPoint);
155 *x = point.x();
156 *y = point.y();
157 }
158 }
159}

216- (void)_activateItemAtIndex:(int)index allowMultipleSelection:(BOOL)allowMultipleSelection
217{
218 // Use the setSelectedIndexByUser function so a change event will be fired. <rdar://problem/6760590>

222 select->optionSelectedByUser(index, true, allowMultipleSelection);

247static WebAutocapitalizeType webAutocapitalizeType(WebCore::AutocapitalizeType type)
248{
249 switch (type) {
250 case WebCore::AutocapitalizeType::Default:
251 return WebAutocapitalizeTypeDefault;
252 case WebCore::AutocapitalizeType::None:
253 return WebAutocapitalizeTypeNone;
254 case WebCore::AutocapitalizeType::Words:
255 return WebAutocapitalizeTypeWords;
256 case WebCore::AutocapitalizeType::Sentences:
257 return WebAutocapitalizeTypeSentences;
258 case WebCore::AutocapitalizeType::AllCharacters:
259 return WebAutocapitalizeTypeAllCharacters;
260 }
261}
