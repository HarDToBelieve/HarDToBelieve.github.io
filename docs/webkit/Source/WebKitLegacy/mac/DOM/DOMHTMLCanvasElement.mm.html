40@implementation DOMHTMLCanvasElement
41
42- (int)width
43{
44 WebCore::JSMainThreadNullState state;
45 return IMPL->width();
46}
47
48- (void)setWidth:(int)newWidth
49{
50 WebCore::JSMainThreadNullState state;
51 IMPL->setWidth(newWidth);
52}
53
54- (int)height
55{
56 WebCore::JSMainThreadNullState state;
57 return IMPL->height();
58}
59
60- (void)setHeight:(int)newHeight
61{
62 WebCore::JSMainThreadNullState state;
63 IMPL->setHeight(newHeight);
64}
65
66- (NSString *)toDataURL:(NSString *)type
67{
68 WebCore::JSMainThreadNullState state;
69
70 auto exceptionOrReturnValue = IMPL->toDataURL(type);
71 if (exceptionOrReturnValue.hasException())
72 raiseDOMErrorException(exceptionOrReturnValue.releaseException());
73
74 return exceptionOrReturnValue.releaseReturnValue().string;
75}
76
77@end
