43#define IMPL reinterpret_cast<WebCore::HTMLCollection*>(_internal)
44
45@implementation DOMHTMLCollection
46
47- (void)dealloc
48{
49 if (WebCoreObjCScheduleDeallocateOnMainThread([DOMHTMLCollection class], self))
50 return;
51
52 if (_internal)
53 IMPL->deref();
54 [super dealloc];
55}
56
57- (unsigned)length
58{
59 WebCore::JSMainThreadNullState state;
60 return IMPL->length();
61}
62
63- (DOMNode *)item:(unsigned)index
64{
65 WebCore::JSMainThreadNullState state;
66 return kit(WTF::getPtr(IMPL->item(index)));
67}
68
69- (DOMNode *)namedItem:(NSString *)name
70{
71 WebCore::JSMainThreadNullState state;
72 return kit(WTF::getPtr(IMPL->namedItem(name)));
73}
74
75- (DOMNodeList *)tags:(NSString *)name
76{
77 if (!name)
78 return nullptr;
79
80 WebCore::JSMainThreadNullState state;
81 return kit(static_cast<WebCore::NodeList*>(WTF::getPtr(IMPL->ownerNode().getElementsByTagName(name))));
82}
83
84@end
85
86DOMHTMLCollection *kit(WebCore::HTMLCollection* value)
87{
88 WebCoreThreadViolationCheckRoundOne();
89 if (!value)
90 return nil;
91 if (DOMHTMLCollection *wrapper = getDOMWrapper(value))
92 return retainPtr(wrapper).autorelease();
93 RetainPtr<DOMHTMLCollection> wrapper = adoptNS([[kitClass(value) alloc] _init]);
94 if (!wrapper)
95 return nil;
96 wrapper->_internal = reinterpret_cast<DOMObjectInternal*>(value);
97 value->ref();
98 addDOMWrapper(wrapper.get(), value);
99 return wrapper.autorelease();
100}
