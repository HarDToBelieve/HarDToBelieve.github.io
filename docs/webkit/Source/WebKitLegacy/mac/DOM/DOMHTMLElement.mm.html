293WebCore::HTMLElement* core(DOMHTMLElement *wrapper)
294{
295 return wrapper ? reinterpret_cast<WebCore::HTMLElement*>(wrapper->_internal) : 0;
296}
297
298DOMHTMLElement *kit(WebCore::HTMLElement* value)
299{
300 WebCoreThreadViolationCheckRoundOne();
301 return static_cast<DOMHTMLElement*>(kit(static_cast<WebCore::Node*>(value)));
302}
