43@implementation DOMHTMLVideoElement
44
45- (unsigned)width
46{
47 WebCore::JSMainThreadNullState state;
48 return IMPL->getUnsignedIntegralAttribute(WebCore::HTMLNames::widthAttr);
49}
50
51- (void)setWidth:(unsigned)newWidth
52{
53 WebCore::JSMainThreadNullState state;
54 IMPL->setUnsignedIntegralAttribute(WebCore::HTMLNames::widthAttr, newWidth);
55}
56
57- (unsigned)height
58{
59 WebCore::JSMainThreadNullState state;
60 return IMPL->getUnsignedIntegralAttribute(WebCore::HTMLNames::heightAttr);
61}
62
63- (void)setHeight:(unsigned)newHeight
64{
65 WebCore::JSMainThreadNullState state;
66 IMPL->setUnsignedIntegralAttribute(WebCore::HTMLNames::heightAttr, newHeight);
67}
68
69- (unsigned)videoWidth
70{
71 WebCore::JSMainThreadNullState state;
72 return IMPL->videoWidth();
73}
74
75- (unsigned)videoHeight
76{
77 WebCore::JSMainThreadNullState state;
78 return IMPL->videoHeight();
79}
80
81- (NSString *)poster
82{
83 WebCore::JSMainThreadNullState state;
84 return IMPL->getURLAttribute(WebCore::HTMLNames::posterAttr).string();
85}
86
87- (void)setPoster:(NSString *)newPoster
88{
89 WebCore::JSMainThreadNullState state;
90 IMPL->setAttributeWithoutSynchronization(WebCore::HTMLNames::posterAttr, newPoster);
91}
92
93- (BOOL)playsInline
94{
95 WebCore::JSMainThreadNullState state;
96 return IMPL->hasAttributeWithoutSynchronization(WebCore::HTMLNames::playsinlineAttr);
97}
98
99- (void)setPlaysInline:(BOOL)newPlaysInline
100{
101 WebCore::JSMainThreadNullState state;
102 IMPL->setBooleanAttribute(WebCore::HTMLNames::playsinlineAttr, newPlaysInline);
103}
104
105- (BOOL)webkitSupportsFullscreen
106{
107 WebCore::JSMainThreadNullState state;
108 return IMPL->webkitSupportsFullscreen();
109}
110
111- (BOOL)webkitDisplayingFullscreen
112{
113 WebCore::JSMainThreadNullState state;
114 return IMPL->webkitDisplayingFullscreen();
115}
116
117- (void)webkitEnterFullscreen
118{
119 WebCore::JSMainThreadNullState state;
120 raiseOnDOMError(IMPL->webkitEnterFullscreen());
121}
122
123- (void)webkitExitFullscreen
124{
125 WebCore::JSMainThreadNullState state;
126 IMPL->webkitExitFullscreen();
127}
128
129- (void)webkitEnterFullScreen

131 [self webkitEnterFullscreen];

136 [self webkitExitFullscreen];
137}
