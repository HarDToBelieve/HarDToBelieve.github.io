44@implementation DOMImplementation

46static inline WebCore::DOMImplementation& unwrap(DOMImplementation& wrapper)
47{
48 return *reinterpret_cast<WebCore::DOMImplementation*>(wrapper._internal);
49}
50
51- (void)dealloc
52{
53 if (WebCoreObjCScheduleDeallocateOnMainThread([DOMImplementation class], self))
54 return;
55 if (_internal)
56 unwrap(*self).deref();
57 [super dealloc];
58}

115DOMImplementation *kit(WebCore::DOMImplementation* value)
116{
117 WebCoreThreadViolationCheckRoundOne();
118 if (!value)
119 return nil;
120 if (DOMImplementation *wrapper = getDOMWrapper(value))
121 return retainPtr(wrapper).autorelease();
122 auto wrapper = adoptNS([[DOMImplementation alloc] _init]);
123 wrapper->_internal = reinterpret_cast<DOMObjectInternal*>(value);
124 value->ref();
125 addDOMWrapper(wrapper.get(), value);
126 return wrapper.autorelease();
127}
