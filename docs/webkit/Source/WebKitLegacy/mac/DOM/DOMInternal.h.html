62template <class Source>
63inline id getDOMWrapper(Source impl)
64{
65 return getDOMWrapper(reinterpret_cast<DOMObjectInternal*>(impl));
66}
67
68template <class Source>

81{
82 return msSinceEpoch / 1000.0 - NSTimeIntervalSince1970;
83}

85inline double core(NSTimeInterval sec)
86{
