50- (void)dealloc
51{
52 if (WebCoreObjCScheduleDeallocateOnMainThread([DOMRGBColor class], self))
53 return;
54
55 if (_internal)
56 IMPL->deref();
57 [super dealloc];
58}

100DOMRGBColor *kit(WebCore::DeprecatedCSSOMRGBColor* value)
101{
102 WebCoreThreadViolationCheckRoundOne();
103 if (!value)
104 return nil;
105 if (DOMRGBColor *wrapper = getDOMWrapper(value))
106 return retainPtr(wrapper).autorelease();
107 auto wrapper = adoptNS([[DOMRGBColor alloc] _init]);
108 wrapper->_internal = reinterpret_cast<DOMObjectInternal*>(value);
109 value->ref();
110 addDOMWrapper(wrapper.get(), value);
111 return wrapper.autorelease();
112}
