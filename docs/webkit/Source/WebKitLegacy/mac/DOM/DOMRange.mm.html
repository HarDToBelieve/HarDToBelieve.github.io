296WebCore::Range* core(DOMRange *wrapper)
297{
298 return wrapper ? reinterpret_cast<WebCore::Range*>(wrapper->_internal) : 0;
299}
300
301DOMRange *kit(WebCore::Range* value)
302{
303 WebCoreThreadViolationCheckRoundOne();
304 if (!value)
305 return nil;
306 if (DOMRange *wrapper = getDOMWrapper(value))
307 return retainPtr(wrapper).autorelease();
308 auto wrapper = adoptNS([[DOMRange alloc] _init]);
309 wrapper->_internal = reinterpret_cast<DOMObjectInternal*>(value);
310 value->ref();
311 addDOMWrapper(wrapper.get(), value);
312 return wrapper.autorelease();
313}
