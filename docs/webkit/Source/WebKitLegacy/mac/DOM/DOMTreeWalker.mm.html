173DOMTreeWalker *kit(WebCore::TreeWalker* value)
174{
175 WebCoreThreadViolationCheckRoundOne();
176 if (!value)
177 return nil;
178 if (DOMTreeWalker *wrapper = getDOMWrapper(value))
179 return retainPtr(wrapper).autorelease();
180 auto wrapper = adoptNS([[DOMTreeWalker alloc] _init]);
181 wrapper->_internal = reinterpret_cast<DOMObjectInternal*>(value);
182 value->ref();
183 addDOMWrapper(wrapper.get(), value);
184 return wrapper.autorelease();
185}
