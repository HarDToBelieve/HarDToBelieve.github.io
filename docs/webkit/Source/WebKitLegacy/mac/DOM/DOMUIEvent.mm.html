41#define IMPL static_cast<WebCore::UIEvent*>(reinterpret_cast<WebCore::Event*>(_internal))
42
43@implementation DOMUIEvent
44
45- (DOMAbstractView *)view
46{
47 WebCore::JSMainThreadNullState state;
48 return kit(WTF::getPtr(IMPL->view()));
49}
50
51- (int)detail
52{
53 WebCore::JSMainThreadNullState state;
54 return IMPL->detail();
55}
56
57- (int)keyCode
58{
59 WebCore::JSMainThreadNullState state;
60 if (is<WebCore::KeyboardEvent>(*IMPL))
61 return downcast<WebCore::KeyboardEvent>(*IMPL).keyCode();
62 return 0;
63}
64
65- (int)charCode
66{
67 WebCore::JSMainThreadNullState state;
68 if (is<WebCore::KeyboardEvent>(*IMPL))
69 return downcast<WebCore::KeyboardEvent>(*IMPL).charCode();
70 return 0;
71}
72
73- (int)layerX
74{
75 WebCore::JSMainThreadNullState state;
76 return IMPL->layerX();
77}
78
79- (int)layerY
80{
81 WebCore::JSMainThreadNullState state;
82 return IMPL->layerY();
83}
84
85- (int)pageX
86{
87 WebCore::JSMainThreadNullState state;
88 return IMPL->pageX();
89}
90
91- (int)pageY
92{
93 WebCore::JSMainThreadNullState state;
94 return IMPL->pageY();
95}
96
97- (int)which
98{
99 WebCore::JSMainThreadNullState state;
100 return IMPL->which();
101}
102
103- (void)initUIEvent:(NSString *)type canBubble:(BOOL)canBubble cancelable:(BOOL)cancelable view:(DOMAbstractView *)inView detail:(int)inDetail
104{
105 WebCore::JSMainThreadNullState state;
106 IMPL->initUIEvent(type, canBubble, cancelable, toWindowProxy(inView), inDetail);
107}
108
109@end

116}

118@end
