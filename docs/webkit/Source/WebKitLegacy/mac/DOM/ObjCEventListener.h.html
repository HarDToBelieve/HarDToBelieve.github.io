29#include <WebCore/EventListener.h>
30#include <wtf/Forward.h>
31#include <wtf/RetainPtr.h>
32
33@protocol DOMEventListener;
34
35namespace WebCore {
36
37 class ObjCEventListener : public EventListener {
38 public:
39 typedef id<DOMEventListener> ObjCListener;
40 static RefPtr<ObjCEventListener> wrap(ObjCListener);
41
42 static const ObjCEventListener* cast(const EventListener* listener)
43 {
44 return listener->type() == ObjCEventListenerType
45 ? static_cast<const ObjCEventListener*>(listener)
46 : nullptr;
47 }
48
49 private:
50 static ObjCEventListener* find(ObjCListener);
51
52 ObjCEventListener(ObjCListener);
53 virtual ~ObjCEventListener();
54 virtual bool operator==(const EventListener&) const;
55 virtual void handleEvent(ScriptExecutionContext&, Event&);
56
57 RetainPtr<ObjCListener> m_listener;
58 };

60} // namespace WebCore
