37namespace WebCore {
38
39typedef HashMap<id, ObjCEventListener*> ListenerMap;
40static ListenerMap* listenerMap;
41
42ObjCEventListener* ObjCEventListener::find(ObjCListener listener)
43{
44 ListenerMap* map = listenerMap;
45 if (!map)
46 return 0;
47 return map->get(listener);
48}
49
50RefPtr<ObjCEventListener> ObjCEventListener::wrap(ObjCListener listener)
51{
52 if (!listener)
53 return nullptr;
54 if (RefPtr<ObjCEventListener> wrapper = find(listener))
55 return wrapper;
56 return adoptRef(new ObjCEventListener(listener));
57}
58
59ObjCEventListener::ObjCEventListener(ObjCListener listener)
60 : EventListener(ObjCEventListenerType)
61 , m_listener(listener)
62{
63 ListenerMap* map = listenerMap;
64 if (!map) {
65 map = new ListenerMap;
66 listenerMap = map;
67 }
68 map->set(listener, this);
69}
70
71ObjCEventListener::~ObjCEventListener()
72{
73 listenerMap->remove(m_listener.get());
74 // Avoid executing arbitrary code during GC; e.g. inside Node::~Node. Use CF* to be ARC safe.
75 CFRetain((__bridge CFTypeRef)m_listener.get());
76 CFAutorelease((__bridge CFTypeRef)m_listener.get());
77}
78
79void ObjCEventListener::handleEvent(ScriptExecutionContext&, Event& event)
80{
81 ObjCListener listener = m_listener.get();
82 [listener handleEvent:kit(&event)];
83}
84
85bool ObjCEventListener::operator==(const EventListener& listener) const
86{
87 if (const ObjCEventListener* objCEventListener = ObjCEventListener::cast(&listener))
88 return m_listener == objCEventListener->m_listener;
89 return false;
90}
