69@implementation DOMElement (WebDOMElementOperationsPrivate)
70
71+ (DOMElement *)_DOMElementFromJSContext:(JSContextRef)context value:(JSValueRef)value
72{
73 if (!context)
74 return 0;
75
76 if (!value)
77 return 0;
78
79 JSGlobalObject* lexicalGlobalObject = toJS(context);
80 JSLockHolder lock(lexicalGlobalObject);
81 return kit(JSElement::toWrapped(lexicalGlobalObject->vm(), toJS(lexicalGlobalObject, value)));
82}

133@implementation DOMNode (WebDOMNodeOperationsPendingPublic)
134
135- (NSString *)markupString
136{
137 auto& node = *core(self);
138
139 String markupString = serializeFragment(node, SerializedNodes::SubtreeIncludingNode);
140 Node::NodeType nodeType = node.nodeType();
141 if (nodeType != Node::DOCUMENT_NODE && nodeType != Node::DOCUMENT_TYPE_NODE)
142 markupString = documentTypeString(node.document()) + markupString;
143
144 return markupString;
145}

173- (DOMRange *)_documentRange
174{
175 DOMRange *range = [self createRange];
176
177 if (DOMNode* documentElement = [self documentElement])
178 [range selectNode:documentElement];
179
180 return range;

251#if !PLATFORM(IOS_FAMILY)
252static NSEventPhase toNSEventPhase(PlatformWheelEventPhase platformPhase)
253{
254 switch (platformPhase) {
255 case PlatformWheelEventPhase::None:
256 return NSEventPhaseNone;
257 case PlatformWheelEventPhase::Began:
258 return NSEventPhaseBegan;
259 case PlatformWheelEventPhase::Stationary:
260 return NSEventPhaseStationary;
261 case PlatformWheelEventPhase::Changed:
262 return NSEventPhaseChanged;
263 case PlatformWheelEventPhase::Ended:
264 return NSEventPhaseEnded;
265 case PlatformWheelEventPhase::Cancelled:
266 return NSEventPhaseCancelled;
267 case PlatformWheelEventPhase::MayBegin:
268 return NSEventPhaseMayBegin;
269 }
270
271 return NSEventPhaseNone;
272}
