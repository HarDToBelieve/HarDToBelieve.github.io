54// Return a object with vanilla implementations of the protocol's methods
55// Note this feature relies on our default delegate being stateless. This
56// is probably an invalid assumption for the WebUIDelegate.
57// If we add any real functionality to this default delegate we probably
58// won't be able to use a singleton.
59+ (WebDefaultUIDelegate *)sharedUIDelegate
60{
61 static NeverDestroyed<RetainPtr<WebDefaultUIDelegate>> sharedDelegate = adoptNS([[WebDefaultUIDelegate alloc] init]);
62 return sharedDelegate.get().get();
63}

65- (WebView *)webView: (WebView *)wv createWebViewWithRequest:(NSURLRequest *)request windowFeatures:(NSDictionary *)features
66{
67 // If the new API method doesn't exist, fallback to the old version of createWebViewWithRequest
68 // for backwards compatability
69 if (![[wv UIDelegate] respondsToSelector:@selector(webView:createWebViewWithRequest:windowFeatures:)] && [[wv UIDelegate] respondsToSelector:@selector(webView:createWebViewWithRequest:)])
70 return [[wv UIDelegate] webView:wv createWebViewWithRequest:request];
71 return nil;
72}

192#if PLATFORM(MAC)
193 auto panel = adoptNS([[WebJavaScriptTextInputPanel alloc] initWithPrompt:prompt text:defaultText]);

195 NSString *result;
196 if ([NSApp runModalForWindow:[panel window]])
197 result = [panel text];
