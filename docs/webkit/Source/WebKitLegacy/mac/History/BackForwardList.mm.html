33static const unsigned DefaultCapacity = 100;
34static const unsigned NoCurrentItemIndex = UINT_MAX;

36BackForwardList::BackForwardList(WebView *webView)
37 : m_webView(webView)
38 , m_current(NoCurrentItemIndex)
39 , m_capacity(DefaultCapacity)
40 , m_closed(true)
41 , m_enabled(true)
42{
43}

45BackForwardList::~BackForwardList()
46{
47 ASSERT(m_closed);
48}

50void BackForwardList::addItem(Ref<HistoryItem>&& newItem)
51{
52 if (!m_capacity || !m_enabled)
53 return;
54
55 // Toss anything in the forward list
56 if (m_current != NoCurrentItemIndex) {
57 unsigned targetSize = m_current + 1;
58 while (m_entries.size() > targetSize) {
59 Ref<HistoryItem> item = m_entries.takeLast();
60 m_entryHash.remove(item.ptr());
61 BackForwardCache::singleton().remove(item);
62 }
63 }
64
65 // Toss the first item if the list is getting too big, as long as we're not using it
66 // (or even if we are, if we only want 1 entry).
67 if (m_entries.size() == m_capacity && (m_current || m_capacity == 1)) {
68 Ref<HistoryItem> item = WTFMove(m_entries[0]);
69 m_entries.remove(0);
70 m_entryHash.remove(item.ptr());
71 BackForwardCache::singleton().remove(item);
72 --m_current;
73 }
74
75 m_entryHash.add(newItem.ptr());
76 m_entries.insert(m_current + 1, WTFMove(newItem));
77 ++m_current;
78}

80void BackForwardList::goBack()
81{
82 ASSERT(m_current > 0);
83 if (m_current > 0)
84 m_current--;
85}
86
87void BackForwardList::goForward()
88{
89 ASSERT(m_current < m_entries.size() - 1);
90 if (m_current < m_entries.size() - 1)
91 m_current++;
92}
93
94void BackForwardList::goToItem(HistoryItem& item)
95{
96 if (!m_entries.size())
97 return;
98
99 unsigned index = 0;
100 for (; index < m_entries.size(); ++index) {
101 if (m_entries[index].ptr() == &item)
102 break;
103 }
104
105 if (index < m_entries.size())
106 m_current = index;
107}

109RefPtr<HistoryItem> BackForwardList::backItem()
110{
111 if (m_current && m_current != NoCurrentItemIndex)
112 return m_entries[m_current - 1].copyRef();
113 return nullptr;
114}

116RefPtr<HistoryItem> BackForwardList::currentItem()
117{
118 if (m_current != NoCurrentItemIndex)
119 return m_entries[m_current].copyRef();
120 return nullptr;
121}
122
123RefPtr<HistoryItem> BackForwardList::forwardItem()
124{
125 if (m_entries.size() && m_current < m_entries.size() - 1)
126 return m_entries[m_current + 1].copyRef();
127 return nullptr;
128}

130void BackForwardList::backListWithLimit(int limit, Vector<Ref<HistoryItem>>& list)
131{
132 list.clear();
133 if (m_current != NoCurrentItemIndex) {
134 unsigned first = std::max(static_cast<int>(m_current) - limit, 0);
135 for (; first < m_current; ++first)
136 list.append(m_entries[first].get());
137 }
138}
139
140void BackForwardList::forwardListWithLimit(int limit, Vector<Ref<HistoryItem>>& list)
141{
142 ASSERT(limit > -1);
143 list.clear();
144 if (!m_entries.size())
145 return;
146
147 unsigned lastEntry = m_entries.size() - 1;
148 if (m_current < lastEntry) {
149 int last = std::min(m_current + limit, lastEntry);
150 limit = m_current + 1;
151 for (; limit <= last; ++limit)
152 list.append(m_entries[limit].get());
153 }
154}

161void BackForwardList::setCapacity(int size)
162{
163 while (size < static_cast<int>(m_entries.size())) {
164 Ref<HistoryItem> item = m_entries.takeLast();
165 m_entryHash.remove(item.ptr());
166 BackForwardCache::singleton().remove(item);
167 }
168
169 if (!size)
170 m_current = NoCurrentItemIndex;
171 else if (m_current > m_entries.size() - 1)
172 m_current = m_entries.size() - 1;
173
174 m_capacity = size;
175}

182void BackForwardList::setEnabled(bool enabled)
183{
184 m_enabled = enabled;

188 setCapacity(capacity);
189 }

192unsigned BackForwardList::backListCount() const
193{
194 return m_current == NoCurrentItemIndex ? 0 : m_current;
195}
196
197unsigned BackForwardList::forwardListCount() const
198{
199 return m_current == NoCurrentItemIndex ? 0 : m_entries.size() - m_current - 1;
200}
201
202RefPtr<HistoryItem> BackForwardList::itemAtIndex(int index)
203{
204 // Do range checks without doing math on index to avoid overflow.
205 if (index < -static_cast<int>(m_current))
206 return nullptr;
207
208 if (index > static_cast<int>(forwardListCount()))
209 return nullptr;
210
211 return m_entries[index + m_current].copyRef();
212}

239void BackForwardList::removeItem(HistoryItem& item)
240{
241 for (unsigned i = 0; i < m_entries.size(); ++i) {
242 if (m_entries[i].ptr() == &item) {
243 m_entries.remove(i);
244 m_entryHash.remove(&item);
245 if (m_current == NoCurrentItemIndex || m_current < i)
246 break;
247 if (m_current > i)
248 m_current--;
249 else {
250 size_t count = m_entries.size();
251 if (m_current >= count)
252 m_current = count ? count - 1 : NoCurrentItemIndex;
253 }
254 break;
255 }
256 }
257}

259bool BackForwardList::containsItem(HistoryItem& entry)
260{
261 return m_entryHash.contains(&entry);
262}
