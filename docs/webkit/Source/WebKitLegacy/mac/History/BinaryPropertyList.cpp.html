82 static unsigned hash(const IntegerArray&);

87unsigned IntegerArrayHash::hash(const IntegerArray& array)
88{
89 return StringHasher::hashMemory(array.integers(), array.size() * sizeof(int));
90}

92bool IntegerArrayHash::equal(const IntegerArray& a, const IntegerArray& b)
93{
94 if (a.isDeletedValue() || b.isDeletedValue())
95 return a.isDeletedValue() == b.isDeletedValue();
96 if (a.size() != b.size())
97 return false;
98 for (size_t i = 0; i < a.size(); ++i) {
99 if (a.integers()[i] != b.integers()[i])
100 return false;
101 }
102 return true;
103}

178{
179 if (integer <= 0xFF)
180 return 2;
181 if (integer <= 0xFFFF)
182 return 3;

186 return 9;

192void BinaryPropertyListPlan::writeInteger(int integer)
193{
194 ASSERT(integer >= 0);
195 ++m_currentAggregateSize;
196 if (!integer) {
197 if (m_integerZeroObjectReference != invalidObjectReference())
198 return;
199 m_integerZeroObjectReference = m_currentObjectReference;
200 } else {

202 return;

410static int bytesNeeded(size_t count)
411{
412 ASSERT(count);
413 int bytesNeeded = 1;
414 for (size_t mask = std::numeric_limits<size_t>::max() << 8; count & mask; mask <<= 8)
415 ++bytesNeeded;
416 return bytesNeeded;
417}

419static inline void storeLength(UInt8* destination, size_t length)
420{
421#ifdef __LP64__
422 destination[0] = length >> 56;
423 destination[1] = length >> 48;
424 destination[2] = length >> 40;
425 destination[3] = length >> 32;
426#else
427 destination[0] = 0;
428 destination[1] = 0;
429 destination[2] = 0;
430 destination[3] = 0;
431#endif
432 destination[4] = length >> 24;
433 destination[5] = length >> 16;
434 destination[6] = length >> 8;
435 destination[7] = length;
436}

438// Like memmove, but reverses the bytes.
439static void moveAndReverseBytes(UInt8* destination, const UInt8* source, size_t length)
440{
441 ASSERT(length);
442 memmove(destination, source, length);
443 UInt8* start = destination;
444 UInt8* end = destination + length;
445 while (end - start > 1)
446 std::swap(*start++, *--end);
447}

449// The serializer uses a single buffer for the property list.
450// The buffer contains:
451//
452// 8-byte header
453// object data
454// offset table
455// 32-byte trailer
456//
457// While serializing object, the offset table entry for each object is written just before
458// the object data for that object is written. Aggregates, arrays and dictionaries, are a
459// special case. The objects that go into an aggregate are written before the aggregate is.
460// As each object is written, the object reference is put in the aggregate buffer. Then,
461// when the aggregate is written, the aggregate buffer is copied into place in the object
462// data. Finally, the header and trailer are written.
463//
464// The aggregate buffer shares space with the object data, like this:
465//
466// 8-byte header
467// object data
468// >>> aggregate buffer <<<
469// offset table
470// 32-byte trailer
471//
472// To make it easy to build it incrementally, the buffer starts at the end of the object
473// data space, and grows backwards. We're guaranteed the aggregate buffer will never collide
474// with the object data pointer because we know that the object data is correctly sized
475// based on our plan, and all the data in the aggregate buffer will be used to create the
476// actual aggregate objects; in the worst case the aggregate buffer will already be in
477// exactly the right place, but backwards.
478
479BinaryPropertyListSerializer::BinaryPropertyListSerializer(BinaryPropertyListWriter& client)
480 : m_plan(client)
481 , m_objectReferenceSize(bytesNeeded(m_plan.objectCount()))
482 , m_offsetTableStart(headerSize + m_plan.byteCount() + m_plan.objectReferenceCount() * m_objectReferenceSize)
483 , m_offsetSize(bytesNeeded(m_offsetTableStart))
484 , m_bufferSize(m_offsetTableStart + m_plan.objectCount() * m_offsetSize + trailerSize)
485 , m_buffer(client.buffer(m_bufferSize))
486 , m_currentObjectReference(0)
487{
488 ASSERT(m_objectReferenceSize > 0);
489 ASSERT(m_offsetSize > 0);
490
491#ifdef __LP64__
492 ASSERT(m_objectReferenceSize <= 8);
493 ASSERT(m_offsetSize <= 8);
494#else
495 ASSERT(m_objectReferenceSize <= 4);
496 ASSERT(m_offsetSize <= 4);
497#endif
498
499 if (!m_buffer)
500 return;
501
502 // Write objects and offset table.
503 m_currentByte = m_buffer + headerSize;
504 m_currentAggregateBufferByte = m_buffer + m_offsetTableStart;
505 client.writeObjects(*this);
506 ASSERT(m_currentObjectReference == m_plan.objectCount());
507 ASSERT(m_currentAggregateBufferByte == m_buffer + m_offsetTableStart);
508 ASSERT(m_currentByte == m_buffer + m_offsetTableStart);
509
510 // Write header.
511 memcpy(m_buffer, "bplist00", headerSize);
512
513 // Write trailer.
514 UInt8* trailer = m_buffer + m_bufferSize - trailerSize;
515 memset(trailer, 0, 6);
516 trailer[6] = m_offsetSize;
517 trailer[7] = m_objectReferenceSize;
518 storeLength(trailer + 8, m_plan.objectCount());
519 storeLength(trailer + 16, m_plan.objectCount() - 1);
520 storeLength(trailer + 24, m_offsetTableStart);
521}

589void BinaryPropertyListSerializer::writeArrayEnd(size_t savedAggregateBufferOffset)
590{
591 ObjectReference reference = m_currentObjectReference;
592 startObject();
593 size_t aggregateBufferByteCount = savedAggregateBufferOffset - (m_currentAggregateBufferByte - m_buffer);
594 ASSERT(aggregateBufferByteCount);
595 ASSERT(!(aggregateBufferByteCount % m_objectReferenceSize));
596 size_t size = aggregateBufferByteCount / m_objectReferenceSize;
597 if (size <= maxLengthInMarkerByte)
598 appendByte(arrayMarkerByte | size);
599 else {
600 appendByte(arrayWithSeparateLengthMarkerByte);
601 appendInteger(size);
602 }
603 m_currentAggregateBufferByte = m_buffer + savedAggregateBufferOffset;
604 ASSERT(m_currentByte <= m_currentAggregateBufferByte);
605 moveAndReverseBytes(m_currentByte, m_currentAggregateBufferByte - aggregateBufferByteCount, aggregateBufferByteCount);
606 m_currentByte += aggregateBufferByteCount;
607 ASSERT(m_currentByte <= m_currentAggregateBufferByte);
608 if (m_currentObjectReference < m_plan.objectCount())
609 addAggregateObjectReference(reference);
610 else
611 ASSERT(m_currentObjectReference == m_plan.objectCount());
612}

651void BinaryPropertyListSerializer::appendInteger(size_t integer)
652{
653 if (integer <= 0xFF) {
654 appendByte(oneByteIntegerMarkerByte);
655 appendByte(integer);
656 return;
657 }
658 if (integer <= 0xFFFF) {
659 appendByte(twoByteIntegerMarkerByte);
660 appendByte(integer >> 8);
661 appendByte(integer);
662 return;

664#ifdef __LP64__
665 if (integer <= 0xFFFFFFFFULL) {
666#endif
667 appendByte(fourByteIntegerMarkerByte);
668 appendByte(integer >> 24);
669 appendByte(integer >> 16);
670 appendByte(integer >> 8);
671 appendByte(integer);
672#ifdef __LP64__
673 return;
674 }
675 appendByte(eightByteIntegerMarkerByte);
676 appendByte(integer >> 56);
677 appendByte(integer >> 48);
678 appendByte(integer >> 40);
679 appendByte(integer >> 32);
680 appendByte(integer >> 24);
681 appendByte(integer >> 16);
682 appendByte(integer >> 8);
683 appendByte(integer);
684#endif
685}

687void BinaryPropertyListSerializer::appendStringObject(const String& string)
688{
689 startObject();
690 unsigned length = string.length();
691 if (string.isAllASCII()) {
692 if (length <= maxLengthInMarkerByte)
693 appendByte(static_cast<unsigned char>(asciiStringMarkerByte | length));
694 else {
695 appendByte(asciiStringWithSeparateLengthMarkerByte);
696 appendInteger(length);
697 }
698 for (unsigned i = 0; i < length; ++i)
699 appendByte(string[i]);
700 } else {
701 if (length <= maxLengthInMarkerByte)
702 appendByte(static_cast<unsigned char>(unicodeStringMarkerByte | length));
703 else {
704 appendByte(unicodeStringWithSeparateLengthMarkerByte);
705 appendInteger(length);
706 }
707 for (unsigned i = 0; i < length; ++i) {
708 appendByte(string[i] >> 8);
709 appendByte(string[i]);
710 }
711 }
712}

741void BinaryPropertyListSerializer::appendObjectReference(ObjectReference reference)
742{
743 switch (m_objectReferenceSize) {
744#ifdef __LP64__
745 case 8:
746 appendByte(reference >> 56);
747 FALLTHROUGH;
748 case 7:
749 appendByte(reference >> 48);
750 FALLTHROUGH;
751 case 6:
752 appendByte(reference >> 40);
753 FALLTHROUGH;
754 case 5:
755 appendByte(reference >> 32);

772void BinaryPropertyListSerializer::startObject()
773{
774 ObjectReference reference = m_currentObjectReference++;
775
776 size_t offset = m_currentByte - m_buffer;
777
778 UInt8* offsetTableEntry = m_buffer + m_offsetTableStart + reference * m_offsetSize + m_offsetSize;
779 switch (m_offsetSize) {
780#ifdef __LP64__
781 case 8:
782 offsetTableEntry[-8] = offset >> 56;
783 FALLTHROUGH;
784 case 7:
785 offsetTableEntry[-7] = offset >> 48;
786 FALLTHROUGH;
787 case 6:
788 offsetTableEntry[-6] = offset >> 40;
789 FALLTHROUGH;
790 case 5:
791 offsetTableEntry[-5] = offset >> 32;
792 FALLTHROUGH;
793#endif
794 case 4:
795 offsetTableEntry[-4] = offset >> 24;
796 FALLTHROUGH;
797 case 3:
798 offsetTableEntry[-3] = offset >> 16;
799 FALLTHROUGH;
800 case 2:
801 offsetTableEntry[-2] = offset >> 8;
802 FALLTHROUGH;
