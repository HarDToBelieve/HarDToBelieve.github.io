67BackForwardList* core(WebBackForwardList *webBackForwardList)
68{
69 if (!webBackForwardList)
70 return 0;
71
72 return reinterpret_cast<BackForwardList*>(webBackForwardList->_private);
73}
74
75WebBackForwardList *kit(BackForwardList* backForwardList)
76{
77 if (!backForwardList)
78 return nil;
79
80 if (WebBackForwardList *webBackForwardList = backForwardLists().get(backForwardList))
81 return webBackForwardList;
82
83 return adoptNS([[WebBackForwardList alloc] initWithBackForwardList:*backForwardList]).autorelease();
84}

86- (id)initWithBackForwardList:(Ref<BackForwardList>&&)backForwardList
87{
88 WebCoreThreadViolationCheckRoundOne();
89 self = [super init];
90 if (!self)
91 return nil;
92
93 _private = reinterpret_cast<WebBackForwardListPrivate*>(&backForwardList.leakRef());
94 backForwardLists().set(core(self), self);
95 return self;
96}

112- (void)dealloc
113{
114 if (WebCoreObjCScheduleDeallocateOnMainThread([WebBackForwardList class], self))
115 return;
116
117 BackForwardList* backForwardList = core(self);
118 ASSERT(backForwardList);
119 if (backForwardList) {
120 ASSERT(backForwardList->closed());
121 backForwardLists().remove(backForwardList);
122 backForwardList->deref();
123 }
124
125 [super dealloc];
126}

156constexpr auto WebBackForwardListDictionaryEntriesKey = @"entries";
157constexpr auto WebBackForwardListDictionaryCapacityKey = @"capacity";
158constexpr auto WebBackForwardListDictionaryCurrentKey = @"current";
159
160- (NSDictionary *)dictionaryRepresentation
161{
162 auto& list = *core(self);
163 auto entries = createNSArray(list.entries(), [] (auto& item) {
164 return [kit(const_cast<WebCore::HistoryItem*>(item.ptr())) dictionaryRepresentationIncludingChildren:NO];
165 });
166 return @{
167 WebBackForwardListDictionaryEntriesKey: entries.get(),
168 WebBackForwardListDictionaryCurrentKey: @(list.current()),
169 WebBackForwardListDictionaryCapacityKey: @(list.capacity()),
170 };
171}
172
173- (void)setToMatchDictionaryRepresentation:(NSDictionary *)dictionary
174{
175 auto& list = *core(self);
176
177 list.setCapacity([[dictionary objectForKey:WebBackForwardListDictionaryCapacityKey] unsignedIntValue]);
178 for (NSDictionary *itemDictionary in [dictionary objectForKey:WebBackForwardListDictionaryEntriesKey])
179 list.addItem(*core(adoptNS([[WebHistoryItem alloc] initFromDictionaryRepresentation:itemDictionary]).get()));
180
181 unsigned currentIndex = [[dictionary objectForKey:WebBackForwardListDictionaryCurrentKey] unsignedIntValue];
182 size_t listSize = list.entries().size();
183 if (currentIndex >= listSize)
184 currentIndex = listSize - 1;
185 list.setCurrent(currentIndex);
186}

279-(NSString *)description
280{
281 NSMutableString *result;
282
283 result = [NSMutableString stringWithCapacity:512];
284
285 [result appendString:@"\n--------------------------------------------\n"];
286 [result appendString:@"WebBackForwardList:\n"];
287
288 BackForwardList* backForwardList = core(self);
289 auto& entries = backForwardList->entries();
290
291 for (unsigned i = 0; i < entries.size(); ++i) {
292 if (entries[i].ptr() == backForwardList->currentItem()) {
293 [result appendString:@" >>>"];
294 } else {
295 [result appendString:@" "];
296 }
297 [result appendFormat:@"%2d) ", i];
298 int currPos = [result length];
299 [result appendString:[kit(const_cast<WebCore::HistoryItem*>(entries[i].ptr())) description]];
300
301 // shift all the contents over. a bit slow, but this is for debugging
302 NSRange replRange = { static_cast<NSUInteger>(currPos), [result length] - currPos };
303 [result replaceOccurrencesOfString:@"\n" withString:@"\n " options:0 range:replRange];
304
305 [result appendString:@"\n"];
306 }
307
308 [result appendString:@"\n--------------------------------------------\n"];
309
310 return result;
311}
