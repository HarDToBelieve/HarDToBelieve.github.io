144static void getDayBoundaries(NSTimeInterval interval, NSTimeInterval& beginningOfDay, NSTimeInterval& beginningOfNextDay)
145{
146 NSDate *date = [NSDate dateWithTimeIntervalSinceReferenceDate:interval];
147
148 NSCalendar *calendar = [NSCalendar calendarWithIdentifier:NSCalendarIdentifierGregorian];
149
150 NSDate *beginningOfDayDate = nil;
151 NSTimeInterval dayLength;
152 [calendar rangeOfUnit:NSCalendarUnitDay startDate:&beginningOfDayDate interval:&dayLength forDate:date];
153
154 beginningOfDay = beginningOfDayDate.timeIntervalSinceReferenceDate;
155 beginningOfNextDay = beginningOfDay + dayLength;
156}
157
158static inline NSTimeInterval beginningOfDay(NSTimeInterval date)
159{
160 static NSTimeInterval cachedBeginningOfDay = NAN;
161 static NSTimeInterval cachedBeginningOfNextDay;
162 if (!(date >= cachedBeginningOfDay && date < cachedBeginningOfNextDay))
163 getDayBoundaries(date, cachedBeginningOfDay, cachedBeginningOfNextDay);
164 return cachedBeginningOfDay;
165}

167static inline WebHistoryDateKey dateKey(NSTimeInterval date)
168{
169 // Converting from double (NSTimeInterval) to int64_t (WebHistoryDateKey) is
170 // safe here because all sensible dates are in the range -2**48 .. 2**47 which
171 // safely fits in an int64_t.
172 return beginningOfDay(date);
173}

175// Returns whether the day is already in the list of days,
176// and fills in *key with the key used to access its location
177- (BOOL)findKey:(WebHistoryDateKey*)key forDay:(NSTimeInterval)date
178{
179 ASSERT_ARG(key, key);
180 *key = dateKey(date);
181 return _entriesByDate->contains(*key);

184- (void)insertItem:(WebHistoryItem *)entry forDateKey:(WebHistoryDateKey)dateKey
185{
186 ASSERT_ARG(entry, entry != nil);
187 ASSERT(_entriesByDate->contains(dateKey));
188
189 NSMutableArray *entriesForDate = _entriesByDate->get(dateKey).get();
190 NSTimeInterval entryDate = [entry lastVisitedTimeInterval];
191
192 unsigned count = [entriesForDate count];
193
194 // The entries for each day are stored in a sorted array with the most recent entry first
195 // Check for the common cases of the entry being newer than all existing entries or the first entry of the day
196 if (!count || [[entriesForDate objectAtIndex:0] lastVisitedTimeInterval] < entryDate) {
197 [entriesForDate insertObject:entry atIndex:0];
198 return;
199 }
200 // .. or older than all existing entries
201 if (count > 0 && [[entriesForDate objectAtIndex:count - 1] lastVisitedTimeInterval] >= entryDate) {
202 [entriesForDate insertObject:entry atIndex:count];
203 return;
204 }
205
206 unsigned low = 0;
207 unsigned high = count;
208 while (low < high) {
209 unsigned mid = low + (high - low) / 2;
210 if ([[entriesForDate objectAtIndex:mid] lastVisitedTimeInterval] >= entryDate)
211 low = mid + 1;
212 else
213 high = mid;
214 }
215
216 // low is now the index of the first entry that is older than entryDate
217 [entriesForDate insertObject:entry atIndex:low];
218}

220- (BOOL)removeItemFromDateCaches:(WebHistoryItem *)entry
221{
222 WebHistoryDateKey dateKey;
223 BOOL foundDate = [self findKey:&dateKey forDay:[entry lastVisitedTimeInterval]];
224
225 if (!foundDate)
226 return NO;
227
228 DateToEntriesMap::iterator it = _entriesByDate->find(dateKey);
229 NSMutableArray *entriesForDate = it->value.get();
230 [entriesForDate removeObjectIdenticalTo:entry];
231
232 // remove this date entirely if there are no other entries on it
233 if ([entriesForDate count] == 0) {
234 _entriesByDate->remove(it);
235 // Clear _orderedLastVisitedDays so it will be regenerated when next requested.
236 _orderedLastVisitedDays = nil;
237 }
238
239 return YES;
240}

242- (BOOL)removeItemForURLString:(NSString *)URLString
243{
244 WebHistoryItem *entry = [_entriesByURL objectForKey:URLString];
245 if (!entry)
246 return NO;
247
248 [_entriesByURL removeObjectForKey:URLString];
249
250#if !ASSERT_ENABLED
251 [self removeItemFromDateCaches:entry];
252#else
253 BOOL itemWasInDateCaches = [self removeItemFromDateCaches:entry];
254 ASSERT(itemWasInDateCaches);
255#endif
256
257 if (![_entriesByURL count])
258 WebVisitedLinkStore::removeAllVisitedLinks();
259
260 return YES;
261}

263- (void)addItemToDateCaches:(WebHistoryItem *)entry
264{
265 WebHistoryDateKey dateKey;
266 if ([self findKey:&dateKey forDay:[entry lastVisitedTimeInterval]])
267 // other entries already exist for this date
268 [self insertItem:entry forDateKey:dateKey];
269 else {
270 // no other entries exist for this date
271 auto entries = adoptNS([[NSMutableArray alloc] initWithObjects:&entry count:1]);
272 _entriesByDate->set(dateKey, entries.get());
273 // Clear _orderedLastVisitedDays so it will be regenerated when next requested.
274 _orderedLastVisitedDays = nil;
275 }
276}

278- (WebHistoryItem *)visitedURL:(NSURL *)url withTitle:(NSString *)title
279{
280 ASSERT(url);
281 ASSERT(title);
282
283 NSString *URLString = [url _web_originalDataAsString];
284 if (!URLString)
285 URLString = @"";
286 auto entry = retainPtr([_entriesByURL objectForKey:URLString]);
287
288 if (entry) {
289 LOG(History, "Updating global history entry %@", entry.get());
290 // Remove the item from date caches before changing its last visited date. Otherwise we might get duplicate entries
291 // as seen in <rdar://problem/6570573>.
292 BOOL itemWasInDateCaches = [self removeItemFromDateCaches:entry.get()];
293 ASSERT_UNUSED(itemWasInDateCaches, itemWasInDateCaches);
294
295 [entry _visitedWithTitle:title];
296 } else {
297 LOG(History, "Adding new global history entry for %@", url);
298 entry = adoptNS([[WebHistoryItem alloc] initWithURLString:URLString title:title lastVisitedTimeInterval:[NSDate timeIntervalSinceReferenceDate]]);
299 [_entriesByURL setObject:entry.get() forKey:URLString];
300 }
301
302 [self addItemToDateCaches:entry.get()];
303
304 return entry.autorelease();
305}

307- (BOOL)addItem:(WebHistoryItem *)entry discardDuplicate:(BOOL)discardDuplicate
308{
309 ASSERT_ARG(entry, entry);
310 ASSERT_ARG(entry, [entry lastVisitedTimeInterval] != 0);
311
312 NSString *URLString = [entry URLString];
313
314#if !PLATFORM(IOS_FAMILY)
315 if (auto oldEntry = retainPtr([_entriesByURL objectForKey:URLString])) {
316 if (discardDuplicate)
317 return NO;
318
319 [self removeItemForURLString:URLString];
320 }
321
322 [self addItemToDateCaches:entry];
323 [_entriesByURL setObject:entry forKey:URLString];
324#else
325 if (auto otherEntry = retainPtr([_entriesByURL objectForKey:URLString])) {
326 if (discardDuplicate)
327 return NO;
328
329 if ([otherEntry lastVisitedTimeInterval] < [entry lastVisitedTimeInterval]) {
330 [self removeItemForURLString:URLString];
331 [self addItemToDateCaches:entry];
332 [_entriesByURL setObject:entry forKey:URLString];
333 } else
334 return NO; // Special case for merges when new items may be older than pre-existing entries.
335 } else {
336 [self addItemToDateCaches:entry];
337 [_entriesByURL setObject:entry forKey:URLString];
338 }
339#endif
340
341 return YES;
342}

344- (void)rebuildHistoryByDayIfNeeded:(WebHistory *)webHistory
345{
346 // We clear all the values to present a consistent state when sending the notifications.
347 // We keep a reference to the entries for rebuilding the history after the notification.
348 auto entryArrays = copyToVector(_entriesByDate->values());
349 _entriesByDate->clear();
350
351 auto entriesByURL = std::exchange(_entriesByURL, nil);
352
353 _orderedLastVisitedDays = nil;
354
355 NSArray *allEntries = [entriesByURL allValues];
356 [webHistory _sendNotification:WebHistoryAllItemsRemovedNotification entries:allEntries];
357
358 // Next, we rebuild the history, restore the states, and notify the clients.
359 _entriesByURL = entriesByURL;
360 for (size_t dayIndex = 0; dayIndex < entryArrays.size(); ++dayIndex) {
361 for (WebHistoryItem *entry in (entryArrays[dayIndex]).get())
362 [self addItemToDateCaches:entry];
363 }
364 [webHistory _sendNotification:WebHistoryItemsAddedNotification entries:allEntries];
365}

367- (BOOL)removeItem:(WebHistoryItem *)entry
368{
369 NSString *URLString = [entry URLString];
370
371 // If this exact object isn't stored, then make no change.
372 // FIXME: Is this the right behavior if this entry isn't present, but another entry for the same URL is?
373 // Maybe need to change the API to make something like removeEntryForURLString public instead.
374 WebHistoryItem *matchingEntry = [_entriesByURL objectForKey:URLString];
375 if (matchingEntry != entry)
376 return NO;
377
378 [self removeItemForURLString:URLString];
379
380 return YES;
381}

383- (BOOL)removeItems:(NSArray *)entries
384{
385 NSUInteger count = [entries count];
386 if (!count)
387 return NO;
388
389 for (NSUInteger index = 0; index < count; ++index)

392 return YES;

395- (BOOL)removeAllItems
396{
397 if (_entriesByDate->isEmpty())
398 return NO;
399
400 _entriesByDate->clear();
401 [_entriesByURL removeAllObjects];
402
403 // Clear _orderedLastVisitedDays so it will be regenerated when next requested.
404 _orderedLastVisitedDays = nil;
405
406 WebVisitedLinkStore::removeAllVisitedLinks();
407
408 return YES;

411- (void)addItems:(NSArray *)newEntries
412{
413 // There is no guarantee that the incoming entries are in any particular
414 // order, but if this is called with a set of entries that were created by
415 // iterating through the results of orderedLastVisitedDays and orderedItemsLastVisitedOnDayy
416 // then they will be ordered chronologically from newest to oldest. We can make adding them
417 // faster (fewer compares) by inserting them from oldest to newest.
418 NSEnumerator *enumerator = [newEntries reverseObjectEnumerator];
419 while (WebHistoryItem *entry = [enumerator nextObject])
420 [self addItem:entry discardDuplicate:NO];
421}

427- (NSArray *)orderedLastVisitedDays
428{
429 if (!_orderedLastVisitedDays) {
430 Vector<int> daysAsTimeIntervals;
431 daysAsTimeIntervals.reserveCapacity(_entriesByDate->size());
432 DateToEntriesMap::const_iterator end = _entriesByDate->end();
433 for (DateToEntriesMap::const_iterator it = _entriesByDate->begin(); it != end; ++it)
434 daysAsTimeIntervals.append(it->key);
435
436 std::sort(daysAsTimeIntervals.begin(), daysAsTimeIntervals.end());
437 size_t count = daysAsTimeIntervals.size();
438 _orderedLastVisitedDays = adoptNS([[NSMutableArray alloc] initWithCapacity:count]);
439 for (int i = count - 1; i >= 0; i--) {
440 NSTimeInterval interval = daysAsTimeIntervals[i];
441 auto date = adoptNS([[NSCalendarDate alloc] initWithTimeIntervalSinceReferenceDate:interval]);
442 [_orderedLastVisitedDays addObject:date.get()];
443 }
444 }
445 return _orderedLastVisitedDays.get();
446}

448- (NSArray *)orderedItemsLastVisitedOnDay:(NSCalendarDate *)date
449{
450 WebHistoryDateKey dateKey;
451 if (![self findKey:&dateKey forDay:[date timeIntervalSinceReferenceDate]])
452 return nil;
453 return _entriesByDate->get(dateKey).get();
454}

520- (BOOL)loadHistoryGutsFromURL:(NSURL *)URL savedItemsCount:(int *)numberOfItemsLoaded collectDiscardedItemsInto:(NSMutableArray *)discardedItems error:(NSError **)error
521{
522 *numberOfItemsLoaded = 0;
523 NSDictionary *dictionary = nil;
524
525 // Optimize loading from local file, which is faster than using the general URL loading mechanism
526 if ([URL isFileURL]) {
527 dictionary = [NSDictionary dictionaryWithContentsOfFile:[URL path]];
528 if (!dictionary) {
529#if !LOG_DISABLED
530 if ([[NSFileManager defaultManager] fileExistsAtPath:[URL path]])
531 LOG_ERROR("unable to read history from file %@; perhaps contents are corrupted", [URL path]);
532#endif
533 // else file doesn't exist, which is normal the first time
534 return NO;
535 }
536 } else {
537 ALLOW_DEPRECATED_DECLARATIONS_BEGIN
538 NSData *data = [NSURLConnection sendSynchronousRequest:[NSURLRequest requestWithURL:URL] returningResponse:nil error:error];
539 ALLOW_DEPRECATED_DECLARATIONS_END
540 if (data.length)
541 dictionary = [NSPropertyListSerialization propertyListWithData:data options:NSPropertyListImmutable format:nullptr error:nullptr];
542 }
543
544 // We used to support NSArrays here, but that was before Safari 1.0 shipped. We will no longer support
545 // that ancient format, so anything that isn't an NSDictionary is bogus.
546 if (![dictionary isKindOfClass:[NSDictionary class]])
547 return NO;
548
549 NSNumber *fileVersionObject = [dictionary objectForKey:FileVersionKey];
550 int fileVersion;
551 // we don't trust data obtained from elsewhere, so double-check
552 if (!fileVersionObject || ![fileVersionObject isKindOfClass:[NSNumber class]]) {
553 LOG_ERROR("history file version can't be determined, therefore not loading");
554 return NO;
555 }
556 fileVersion = [fileVersionObject intValue];
557 if (fileVersion > currentFileVersion) {
558 LOG_ERROR("history file version is %d, newer than newest known version %d, therefore not loading", fileVersion, currentFileVersion);
559 return NO;
560 }
561
562 NSArray *array = [dictionary objectForKey:DatesArrayKey];
563
564 int itemCountLimit = [self historyItemLimit];
565 NSTimeInterval ageLimitDate = [[self ageLimitDate] timeIntervalSinceReferenceDate];
566 BOOL ageLimitPassed = NO;
567 BOOL itemLimitPassed = NO;
568 ASSERT(*numberOfItemsLoaded == 0);
569
570 for (NSDictionary *itemAsDictionary in array) {
571 @autoreleasepool {
572 auto item = adoptNS([[WebHistoryItem alloc] initFromDictionaryRepresentation:itemAsDictionary]);
573
574 // item without URL is useless; data on disk must have been bad; ignore
575 if ([item URLString]) {
576 // Test against date limit. Since the items are ordered newest to oldest, we can stop comparing
577 // once we've found the first item that's too old.
578 if (!ageLimitPassed && [item lastVisitedTimeInterval] <= ageLimitDate)
579 ageLimitPassed = YES;
580
581 if (ageLimitPassed || itemLimitPassed)
582 [discardedItems addObject:item.get()];
583 else {
584 if ([self addItem:item.get() discardDuplicate:YES])
585 ++(*numberOfItemsLoaded);
586 if (*numberOfItemsLoaded == itemCountLimit)
587 itemLimitPassed = YES;
588 }
589 }
590 }
591 }
592
593 return YES;
594}

596- (BOOL)loadFromURL:(NSURL *)URL collectDiscardedItemsInto:(NSMutableArray *)discardedItems error:(NSError **)error
597{
598#if !LOG_DISABLED
599 double start = CFAbsoluteTimeGetCurrent();
600#endif
601
602 int numberOfItems;
603 if (![self loadHistoryGutsFromURL:URL savedItemsCount:&numberOfItems collectDiscardedItemsInto:discardedItems error:error])
604 return NO;
605
606#if !LOG_DISABLED
607 double duration = CFAbsoluteTimeGetCurrent() - start;
608 LOG(Timing, "loading %d history entries from %@ took %f seconds", numberOfItems, URL, duration);
609#endif
610
611 return YES;
612}

614- (NSData *)data
615{
616 if (_entriesByDate->isEmpty()) {
617 static NSData *emptyHistoryData = [[NSData alloc] init];
618 return emptyHistoryData;
619 }
620
621 // Ignores the date and item count limits; these are respected when loading instead of when saving, so
622 // that clients can learn of discarded items by listening to WebHistoryItemsDiscardedWhileLoadingNotification.
623 WebHistoryWriter writer(_entriesByDate.get());
624 writer.writePropertyList();
625 return retainPtr((NSData *)writer.releaseData().get()).autorelease();
626}
627
628- (BOOL)saveToURL:(NSURL *)URL error:(NSError **)error
629{
630#if !LOG_DISABLED
631 double start = CFAbsoluteTimeGetCurrent();
632#endif
633
634 BOOL result = [[self data] writeToURL:URL options:0 error:error];
635
636#if !LOG_DISABLED
637 double duration = CFAbsoluteTimeGetCurrent() - start;
638 LOG(Timing, "saving history to %@ took %f seconds", URL, duration);
639#endif
640
641 return result;
642}

867WebHistoryWriter::WebHistoryWriter(DateToEntriesMap* entriesByDate)
868 : m_entriesByDate(entriesByDate)
869{
870 m_dateKeys.reserveCapacity(m_entriesByDate->size());
871 DateToEntriesMap::const_iterator end = m_entriesByDate->end();
872 for (DateToEntriesMap::const_iterator it = m_entriesByDate->begin(); it != end; ++it)
873 m_dateKeys.append(it->key);
874 std::sort(m_dateKeys.begin(), m_dateKeys.end());
875}

877void WebHistoryWriter::writeHistoryItems(BinaryPropertyListObjectStream& stream)
878{
879 for (int dateIndex = m_dateKeys.size() - 1; dateIndex >= 0; dateIndex--) {
880 NSArray *entries = m_entriesByDate->get(m_dateKeys[dateIndex]).get();
881 NSUInteger entryCount = [entries count];
882 for (NSUInteger entryIndex = 0; entryIndex < entryCount; ++entryIndex)
883 writeHistoryItem(stream, [entries objectAtIndex:entryIndex]);
884 }
