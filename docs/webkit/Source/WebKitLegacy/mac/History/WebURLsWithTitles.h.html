33#import <Cocoa/Cocoa.h>

35#define WebURLsWithTitlesPboardType @"WebURLsWithTitlesPboardType"
36
37// Convenience class for getting URLs and associated titles on and off an NSPasteboard
38
39@interface WebURLsWithTitles : NSObject
40
41// Writes parallel arrays of URLs and titles to the pasteboard. These items can be retrieved by
42// calling URLsFromPasteboard and titlesFromPasteboard. URLs must consist of NSURL objects.
43// titles must consist of NSStrings, or be nil. If titles is nil, or if titles is a different
44// length than URLs, empty strings will be used for all titles. If URLs is nil, this method
45// returns without doing anything. You must declare an WebURLsWithTitlesPboardType data type
46// for pasteboard before invoking this method, or it will return without doing anything.
47+ (void)writeURLs:(NSArray *)URLs andTitles:(NSArray *)titles toPasteboard:(NSPasteboard *)pasteboard;
48
49// Reads an array of NSURLs off the pasteboard. Returns nil if pasteboard does not contain
50// data of type WebURLsWithTitlesPboardType. This array consists of the URLs that correspond to
51// the titles returned from titlesFromPasteboard.
52+ (NSArray *)URLsFromPasteboard:(NSPasteboard *)pasteboard;
53
54// Reads an array of NSStrings off the pasteboard. Returns nil if pasteboard does not contain
55// data of type WebURLsWithTitlesPboardType. This array consists of the titles that correspond to
56// the URLs returned from URLsFromPasteboard.
57+ (NSArray *)titlesFromPasteboard:(NSPasteboard *)pasteboard;
58
59@end
60
61#endif
