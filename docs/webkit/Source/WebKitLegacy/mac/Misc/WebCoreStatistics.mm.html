83static RetainPtr<NSCountedSet> createNSCountedSet(const HashCountedSet<const char*>& set)
84{
85 auto result = adoptNS([[NSCountedSet alloc] initWithCapacity:set.size()]);
86 for (auto& entry : set) {
87 auto key = [NSString stringWithUTF8String:entry.key];
88 for (unsigned i = 0; i < entry.value; ++i)
89 [result addObject:key];
90 }
91 return result;
92}

193+ (NSDictionary *)memoryStatistics
194{
195 auto fastMallocStatistics = WTF::fastMallocStatistics();
196 JSLockHolder lock(commonVM());
197 size_t heapSize = commonVM().heap.size();
198 size_t heapFree = commonVM().heap.capacity() - heapSize;
199 auto globalMemoryStats = globalMemoryStatistics();
200 return @{
201 @"FastMallocReservedVMBytes": @(fastMallocStatistics.reservedVMBytes),
202 @"FastMallocCommittedVMBytes": @(fastMallocStatistics.committedVMBytes),
203 @"FastMallocFreeListBytes": @(fastMallocStatistics.freeListBytes),
204 @"JavaScriptHeapSize": @(heapSize),
205 @"JavaScriptFreeSize": @(heapFree),
206 @"JavaScriptStackSize": @(globalMemoryStats.stackBytes),
207 @"JavaScriptJITSize": @(globalMemoryStats.JITBytes),
208 };
209}

268static OptionSet<RenderAsTextFlag> toRenderAsTextFlags(WebRenderTreeAsTextOptions options)
269{
270 OptionSet<RenderAsTextFlag> flags;
271
272 if (options & WebRenderTreeAsTextShowAllLayers)
273 flags.add(RenderAsTextFlag::ShowAllLayers);
274 if (options & WebRenderTreeAsTextShowLayerNesting)
275 flags.add(RenderAsTextFlag::ShowLayerNesting);
276 if (options & WebRenderTreeAsTextShowCompositedLayers)
277 flags.add(RenderAsTextFlag::ShowCompositedLayers);
278 if (options & WebRenderTreeAsTextShowOverflow)
279 flags.add(RenderAsTextFlag::ShowOverflow);
280 if (options & WebRenderTreeAsTextShowSVGGeometry)
281 flags.add(RenderAsTextFlag::ShowSVGGeometry);
282 if (options & WebRenderTreeAsTextShowLayerFragments)
283 flags.add(RenderAsTextFlag::ShowLayerFragments);
284
285 return flags;
286}
287
288- (NSString *)renderTreeAsExternalRepresentationWithOptions:(WebRenderTreeAsTextOptions)options
289{
290 return externalRepresentation(_private->coreFrame, toRenderAsTextFlags(options));
291}

293- (int)numberOfPagesWithPageWidth:(float)pageWidthInPixels pageHeight:(float)pageHeightInPixels
294{
295 Frame* coreFrame = _private->coreFrame;
296 if (!coreFrame)
297 return -1;
298
299 return PrintContext::numberOfPages(*coreFrame, FloatSize(pageWidthInPixels, pageHeightInPixels));
300}
301
302- (void)printToCGContext:(CGContextRef)cgContext pageWidth:(float)pageWidthInPixels pageHeight:(float)pageHeightInPixels
303{
304 Frame* coreFrame = _private->coreFrame;
305 if (!coreFrame)
306 return;
307
308 GraphicsContextCG graphicsContext(cgContext);
309 PrintContext::spoolAllPagesWithBoundaries(*coreFrame, graphicsContext, FloatSize(pageWidthInPixels, pageHeightInPixels));
310}
