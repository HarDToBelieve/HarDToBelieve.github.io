46static bool shouldCallOnNetworkThread()
47{
48#if PLATFORM(MAC)
49 static bool isOldEpsonSoftwareUpdater = WebCore::MacApplication::isEpsonSoftwareUpdater() && dyld_get_program_sdk_version() < DYLD_MACOSX_VERSION_10_15;
50 return isOldEpsonSoftwareUpdater;
51#else
52 return false;
53#endif
54}
55
56static void callOnDelegateThread(Function<void()>&& function)
57{
58 if (shouldCallOnNetworkThread())
59 function();
60 callOnMainThread(WTFMove(function));
61}
62
63template<typename Callable>
64static void callOnDelegateThreadAndWait(Callable&& work)
65{
66 if (shouldCallOnNetworkThread() || isMainThread())
67 work();
68 else {
69 WorkQueue::main().dispatchSync([work = WTFMove(work)]() mutable {
70 work();
71 });
72 }
73}

95- (BOOL)respondsToSelector:(SEL)selector
96{
97 if (selector == @selector(downloadDidBegin:) ||
98 selector == @selector(download:willSendRequest:redirectResponse:) ||
99 selector == @selector(download:didReceiveResponse:) ||
100 selector == @selector(download:didReceiveDataOfLength:) ||
101 selector == @selector(download:shouldDecodeSourceDataOfMIMEType:) ||
102 selector == @selector(download:decideDestinationWithSuggestedFilename:) ||
103 selector == @selector(download:didCreateDestination:) ||
104 selector == @selector(downloadDidFinish:) ||
105 selector == @selector(download:didFailWithError:)) {
106 return [realDelegate respondsToSelector:selector];
107 }
108
109 return [super respondsToSelector:selector];
110}
111
112- (void)downloadDidBegin:(NSURLDownload *)download
113{
114 callOnDelegateThread([realDelegate = realDelegate, download = retainPtr(download)] {
115 [realDelegate downloadDidBegin:download.get()];
116 });
117}
118
119- (NSURLRequest *)download:(NSURLDownload *)download willSendRequest:(NSURLRequest *)request redirectResponse:(NSURLResponse *)redirectResponse
120{
121 RetainPtr<NSURLRequest> returnValue;
122 auto work = [&] {
123 ASSERT(isMainThread());
124 returnValue = [realDelegate download:download willSendRequest:request redirectResponse:redirectResponse];
125 };
126 callOnDelegateThreadAndWait(WTFMove(work));
127 return returnValue.autorelease();
128}
129
130- (void)download:(NSURLDownload *)download didReceiveAuthenticationChallenge:(NSURLAuthenticationChallenge *)challenge
131{
132#if !PLATFORM(IOS_FAMILY)
133 // Try previously stored credential first.
134 if (![challenge previousFailureCount]) {
135 NSURLCredential *credential = NetworkStorageSessionMap::defaultStorageSession().credentialStorage().get(emptyString(), ProtectionSpace([challenge protectionSpace])).nsCredential();
136 if (credential) {
137 [[challenge sender] useCredential:credential forAuthenticationChallenge:challenge];
138 return;
139 }
140 }
141
142 if ([realDelegate respondsToSelector:@selector(download:didReceiveAuthenticationChallenge:)]) {
143 callOnDelegateThread([realDelegate = realDelegate, download = retainPtr(download), challenge = retainPtr(challenge)] {
144 [realDelegate download:download.get() didReceiveAuthenticationChallenge:challenge.get()];
145 });
146 } else {
147 callOnDelegateThread([realDelegate = realDelegate, download = retainPtr(download), challenge = retainPtr(challenge)] {
148 NSWindow *window = nil;
149 if ([realDelegate respondsToSelector:@selector(downloadWindowForAuthenticationSheet:)])
150 window = [realDelegate downloadWindowForAuthenticationSheet:(WebDownload *)download.get()];
151
152 [[WebPanelAuthenticationHandler sharedHandler] startAuthentication:challenge.get() window:window];
153 });
154 }
155#endif
156}

212@implementation WebDownload

214- (void)_setRealDelegate:(id)delegate
215{
216 if (_webInternal == nil) {
217 _webInternal = [[WebDownloadInternal alloc] init];
218 [_webInternal setRealDelegate:delegate];
219 } else {
220 ASSERT(_webInternal == delegate);
221 }
222}
223
224- (id)init
225{
226 self = [super init];
227 if (self != nil) {
228 // _webInternal can be set up before init by _setRealDelegate
229 if (_webInternal == nil) {
230 _webInternal = [[WebDownloadInternal alloc] init];
231 }
232 }
233 return self;
234}
235
236- (void)dealloc
237{
238 [_webInternal release];
239 [super dealloc];
240}
