66@end

68@implementation NSError (WebKitInternal)

70- (instancetype)_webkit_initWithDomain:(NSString *)domain code:(int)code URL:(NSURL *)URL
71{
72 // Insert a localized string here for those folks not savvy to our category methods.
73 NSString *localizedDescription = [[descriptions objectForKey:domain] objectForKey:@(code)];
74 NSDictionary *userInfo = [NSDictionary dictionaryWithObjectsAndKeys:
75 URL, @"NSErrorFailingURLKey",
76 [URL absoluteString], NSURLErrorFailingURLStringErrorKey,
77 localizedDescription, NSLocalizedDescriptionKey,
78 nil];
79 return [self initWithDomain:domain code:code userInfo:userInfo];
80}

84@implementation NSError (WebKitExtras)

86+ (void)_registerWebKitErrors
87{
88 static dispatch_once_t flag;
89 dispatch_once(&flag, ^{
90 @autoreleasepool {
91 NSDictionary *dict = [NSDictionary dictionaryWithObjectsAndKeys:
92 // Policy errors
93 WebKitErrorDescriptionCannotShowMIMEType, @(WebKitErrorCannotShowMIMEType),
94 WebKitErrorDescriptionCannotShowURL, @(WebKitErrorCannotShowURL),
95 WebKitErrorDescriptionFrameLoadInterruptedByPolicyChange, @(WebKitErrorFrameLoadInterruptedByPolicyChange),
96 WebKitErrorDescriptionCannotUseRestrictedPort, @(WebKitErrorCannotUseRestrictedPort),
97 WebKitErrorDescriptionFrameLoadBlockedByContentFilter, @(WebKitErrorFrameLoadBlockedByContentFilter),
98
99 // Plug-in and java errors
100 WebKitErrorDescriptionCannotFindPlugin, @(WebKitErrorCannotFindPlugIn),
101 WebKitErrorDescriptionCannotLoadPlugin, @(WebKitErrorCannotLoadPlugIn),
102 WebKitErrorDescriptionJavaUnavailable, @(WebKitErrorJavaUnavailable),
103 WebKitErrorDescriptionPlugInCancelledConnection, @(WebKitErrorPlugInCancelledConnection),
104 WebKitErrorDescriptionPlugInWillHandleLoad, @(WebKitErrorPlugInWillHandleLoad),
105
106 // Geolocation errors
107 WebKitErrorDescriptionGeolocationLocationUnknown, @(WebKitErrorGeolocationLocationUnknown),
108 nil];
109
110 [NSError _webkit_addErrorsWithCodesAndDescriptions:dict inDomain:WebKitErrorDomain];
111 }
112 });
113}

115+(id)_webkit_errorWithDomain:(NSString *)domain code:(int)code URL:(NSURL *)URL
116{
117 return [[[self alloc] _webkit_initWithDomain:domain code:code URL:URL] autorelease];
118}

120+ (NSError *)_webKitErrorWithDomain:(NSString *)domain code:(int)code URL:(NSURL *)URL
121{
122 [self _registerWebKitErrors];
123 return [self _webkit_errorWithDomain:domain code:code URL:URL];
124}

126+ (NSError *)_webKitErrorWithCode:(int)code failingURL:(NSString *)URLString
127{
128 return [self _webKitErrorWithDomain:WebKitErrorDomain code:code URL:[NSURL _webkit_URLWithUserTypedString:URLString]];
129}

131- (id)_initWithPluginErrorCode:(int)code
132 contentURL:(NSURL *)contentURL
133 pluginPageURL:(NSURL *)pluginPageURL
134 pluginName:(NSString *)pluginName
135 MIMEType:(NSString *)MIMEType
136{
137 [[self class] _registerWebKitErrors];
138
139 NSMutableDictionary *userInfo = [[NSMutableDictionary alloc] init];
140 NSDictionary *descriptionsForWebKitErrorDomain = [descriptions objectForKey:WebKitErrorDomain];
141 NSString *localizedDescription = [descriptionsForWebKitErrorDomain objectForKey:@(code)];
142 if (localizedDescription)
143 [userInfo setObject:localizedDescription forKey:NSLocalizedDescriptionKey];
144 if (contentURL) {
145 [userInfo setObject:contentURL forKey:@"NSErrorFailingURLKey"];
146 [userInfo setObject:[contentURL _web_userVisibleString] forKey:NSURLErrorFailingURLStringErrorKey];
147 }
148 if (pluginPageURL) {
149 [userInfo setObject:[pluginPageURL _web_userVisibleString] forKey:WebKitErrorPlugInPageURLStringKey];
150 }
151 if (pluginName) {
152 [userInfo setObject:pluginName forKey:WebKitErrorPlugInNameKey];
153 }
154 if (MIMEType) {
155 [userInfo setObject:MIMEType forKey:WebKitErrorMIMETypeKey];
156 }
157
158 NSDictionary *userInfoCopy = [userInfo count] > 0 ? [[NSDictionary alloc] initWithDictionary:userInfo] : nil;
159 [userInfo release];
160 NSError *error = [self initWithDomain:WebKitErrorDomain code:code userInfo:userInfoCopy];
161 [userInfoCopy release];
162
163 return error;
164}

166+ (void)_webkit_addErrorsWithCodesAndDescriptions:(NSDictionary *)dictionary inDomain:(NSString *)domain
167{
168 if (!descriptions)
169 descriptions = [[NSMutableDictionary alloc] init];
170
171 [descriptions setObject:dictionary forKey:domain];
172}
